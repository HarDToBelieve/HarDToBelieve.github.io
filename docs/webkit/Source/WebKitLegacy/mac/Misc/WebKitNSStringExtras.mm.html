49static bool canUseFastRenderer(const UniChar* buffer, unsigned length)
50{
51 for (unsigned i = 0; i < length; i++) {
52 if (!isLatin1(buffer[i])) {
53 auto direction = u_charDirection(buffer[i]);
54 if (direction == U_RIGHT_TO_LEFT || (direction > U_OTHER_NEUTRAL && direction != U_DIR_NON_SPACING_MARK && direction != U_BOUNDARY_NEUTRAL))
55 return false;
56 }
57 }
58 return true;
59}

61- (void)_web_drawAtPoint:(NSPoint)point font:(NSFont *)font textColor:(NSColor *)textColor
62{
63 if (!font)
64 return;
65
66 unsigned length = [self length];
67 Vector<UniChar, 2048> buffer(length);
68 [self getCharacters:buffer.data()];
69
70 if (canUseFastRenderer(buffer.data(), length)) {
71 FontCascade webCoreFont(FontPlatformData((__bridge CTFontRef)font, [font pointSize]));
72 TextRun run(StringView(reinterpret_cast<const UChar*>(buffer.data()), length));
73
74 // The following is a half-assed attempt to match AppKit's rounding rules for drawAtPoint.
75 // If you change it, be sure to test all the text drawn this way in Safari, including
76 // the status bar, bookmarks bar, tab bar, and activity window.
77 point.y = CGCeiling(point.y);
78
79 NSGraphicsContext *nsContext = [NSGraphicsContext currentContext];
80 CGContextRef cgContext = [nsContext CGContext];
81 GraphicsContextCG graphicsContext { cgContext };
82
83 // WebCore requires a flipped graphics context.
84 bool flipped = [nsContext isFlipped];
85 if (!flipped)
86 CGContextScaleCTM(cgContext, 1, -1);
87
88 graphicsContext.setFillColor(colorFromNSColor(textColor));
89 webCoreFont.drawText(graphicsContext, run, FloatPoint(point.x, flipped ? point.y : -point.y));
90
91 if (!flipped)
92 CGContextScaleCTM(cgContext, 1, -1);
93 } else {
94 // The given point is on the baseline.
95 if ([[NSView focusView] isFlipped])
96 point.y -= [font ascender];
97 else
98 point.y += [font descender];
99
100 [self drawAtPoint:point withAttributes:@{ NSFontAttributeName: font, NSForegroundColorAttributeName: textColor }];
101 }
102}

104- (float)_web_widthWithFont:(NSFont *)font
105{
106 unsigned length = [self length];
107 Vector<UniChar, 2048> buffer(length);
108 [self getCharacters:buffer.data()];
109
110 if (canUseFastRenderer(buffer.data(), length)) {
111 FontCascade webCoreFont(FontPlatformData((__bridge CTFontRef)font, [font pointSize]));
112 TextRun run(StringView(reinterpret_cast<const UChar*>(buffer.data()), length));
113 return webCoreFont.width(run);
114 }
115
116 return [self sizeWithAttributes:@{ NSFontAttributeName: font }].width;
117}

121- (NSString *)_web_stringByAbbreviatingWithTildeInPath
122{
123 // Handles home directories that have symlinks in their paths as well as what stringByAbbreviatingWithTildeInPath handles.
124 // This works around Radar bug 2774250.
125
126 NSString *resolvedHomeDirectory = [NSHomeDirectory() stringByResolvingSymlinksInPath];
127 NSString *path;
128
129 if ([self hasPrefix:resolvedHomeDirectory]) {
130 NSString *relativePath = [self substringFromIndex:[resolvedHomeDirectory length]];
131 path = [NSHomeDirectory() stringByAppendingPathComponent:relativePath];
132 } else {
133 path = self;
134 }
135
136 return [path stringByAbbreviatingWithTildeInPath];
137}

139- (BOOL)_webkit_isCaseInsensitiveEqualToString:(NSString *)string
140{

177+ (NSString *)_webkit_localCacheDirectoryWithBundleIdentifier:(NSString*)bundleIdentifier
178{
179 NSString *cacheDirectory = [[NSUserDefaults standardUserDefaults] objectForKey:WebKitLocalCacheDefaultsKey];
180
181 if (!cacheDirectory || ![cacheDirectory isKindOfClass:[NSString class]]) {
182#if PLATFORM(IOS_FAMILY)
183 cacheDirectory = [NSHomeDirectory() stringByAppendingPathComponent:@"Library/Caches"];
184#endif
185#if PLATFORM(MAC)
186 char buffer[MAXPATHLEN];
187 if (size_t length = confstr(_CS_DARWIN_USER_CACHE_DIR, buffer, MAXPATHLEN))
188 cacheDirectory = [[NSFileManager defaultManager] stringWithFileSystemRepresentation:buffer length:length - 1];
189#endif
190 }
191
192 return [cacheDirectory stringByAppendingPathComponent:bundleIdentifier];
193}
