31#import <mach-o/dyld.h>
32#import <mutex>

34static int WebKitLinkTimeVersion(void);
35static int overriddenWebKitLinkTimeVersion;
36
37BOOL WebKitLinkedOnOrAfter(int version)
38{
39#if !PLATFORM(IOS_FAMILY)
40 return (WebKitLinkTimeVersion() >= version);
41#else
42 int32_t linkTimeVersion = WebKitLinkTimeVersion();
43 int32_t majorVersion = linkTimeVersion >> 16 & 0x0000FFFF;
44
45 // The application was not linked against UIKit so assume most recent WebKit
46 if (linkTimeVersion == -1)
47 return YES;
48
49 return (majorVersion >= version);
50#endif
51}
52
53void setWebKitLinkTimeVersion(int version)
54{
55 overriddenWebKitLinkTimeVersion = version;
56}
57
58static int WebKitLinkTimeVersion(void)
59{
60 if (overriddenWebKitLinkTimeVersion)
61 return overriddenWebKitLinkTimeVersion;
62
63#if !PLATFORM(IOS_FAMILY)
64 return NSVersionOfLinkTimeLibrary("WebKit");
65#else
66 // <rdar://problem/6627758> Need to implement WebKitLinkedOnOrAfter
67 // Third party applications do not link against WebKit, but rather against UIKit.
68 return NSVersionOfLinkTimeLibrary("UIKit");
69#endif
70}
