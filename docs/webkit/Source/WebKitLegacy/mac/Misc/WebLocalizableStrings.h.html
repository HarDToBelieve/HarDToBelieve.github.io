29#pragma once
30
31#ifdef __OBJC__
32@class NSBundle;
33typedef NSString *WebLocalizedStringType;
34#else
35#ifdef __cplusplus
36class NSBundle;
37#else
38typedef struct NSBundle NSBundle;
39#endif
40typedef CFStringRef WebLocalizedStringType;
41#endif
42
43typedef struct {
44 const char *identifier;
45 __unsafe_unretained NSBundle *bundle;
46} WebLocalizableStringsBundle;
47
48#ifdef __cplusplus
49extern "C" {
50#endif
51
52WebLocalizedStringType WebLocalizedString(WebLocalizableStringsBundle* bundle, const char* key);
53
54#ifdef __cplusplus
55}
56#endif
57
58static inline __attribute__((format_arg(3))) WebLocalizedStringType WebLocalizedStringWithValue(WebLocalizableStringsBundle* bundle, const char* key, const char* value)
59{
60 return WebLocalizedString(bundle, key);
61}
