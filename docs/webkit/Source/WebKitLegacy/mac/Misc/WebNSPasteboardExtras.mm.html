74static NSArray *_writableTypesForImageWithoutArchive (void)
75{
76 static auto types = makeNeverDestroyed([] {
77 auto types = adoptNS([[NSMutableArray alloc] initWithObjects:legacyTIFFPasteboardType(), nil]);
78 [types addObjectsFromArray:[NSPasteboard _web_writableTypesForURL]];
79 return types;
80 }());
81 return types.get().get();
82}

84static NSArray *_writableTypesForImageWithArchive (void)
85{
86 static auto types = makeNeverDestroyed([] {
87 auto types = adoptNS([_writableTypesForImageWithoutArchive() mutableCopy]);
88 [types addObject:legacyRTFDPasteboardType()];
89 [types addObject:WebArchivePboardType];
90 return types;
91 }());
92 return types.get().get();
93}

115- (NSURL *)_web_bestURL
116{
117 NSArray *types = [self types];
118
119 if ([types containsObject:legacyURLPasteboardType()]) {
120 NSURL *URLFromPasteboard = [NSURL URLFromPasteboard:self];
121 NSString *scheme = [URLFromPasteboard scheme];
122 if ([scheme isEqualToString:@"http"] || [scheme isEqualToString:@"https"]) {
123 return [URLFromPasteboard _webkit_canonicalize];
124 }
125 }
126
127 if ([types containsObject:legacyStringPasteboardType()]) {
128 NSString *URLString = [self stringForType:legacyStringPasteboardType()];
129 if ([URLString _webkit_looksLikeAbsoluteURL]) {
130 NSURL *URL = [[NSURL _webkit_URLWithUserTypedString:URLString] _webkit_canonicalize];
131 if (URL) {
132 return URL;
133 }
134 }
135 }
136
137 if ([types containsObject:legacyFilenamesPasteboardType()]) {
138 NSArray *files = [self propertyListForType:legacyFilenamesPasteboardType()];
139 // FIXME: Maybe it makes more sense to allow multiple files and only use the first one?
140 if ([files count] == 1) {
141 NSString *file = [files objectAtIndex:0];
142 // FIXME: We are filtering out directories because that's what the original code used to
143 // do. Without this check, if the URL points to a local directory, Safari will open the
144 // parent directory of the directory in Finder. This check should go away as soon as
145 // possible.
146 BOOL isDirectory;
147 if ([[NSFileManager defaultManager] fileExistsAtPath:file isDirectory:&isDirectory] && isDirectory)
148 return nil;
149 return [[NSURL fileURLWithPath:file] _webkit_canonicalize];
150 }
151 }
152
153 return nil;
154}

156- (void)_web_writeURL:(NSURL *)URL andTitle:(NSString *)title types:(NSArray *)types
157{
158 ASSERT(URL);
159
160 if ([title length] == 0) {
161 title = [[URL path] lastPathComponent];
162 if ([title length] == 0)
163 title = [URL _web_userVisibleString];
164 }
165
166 if ([types containsObject:legacyURLPasteboardType()])
167 [URL writeToPasteboard:self];
168 if ([types containsObject:WebURLPboardType])
169 [self setString:[URL _web_originalDataAsString] forType:WebURLPboardType];
170 if ([types containsObject:WebURLNamePboardType])
171 [self setString:title forType:WebURLNamePboardType];
172 if ([types containsObject:legacyStringPasteboardType()])
173 [self setString:[URL _web_userVisibleString] forType:legacyStringPasteboardType()];
174 if ([types containsObject:WebURLsWithTitlesPboardType])
175 [WebURLsWithTitles writeURLs:@[URL] andTitles:@[title] toPasteboard:self];
176}

179{
180 ALLOW_DEPRECATED_DECLARATIONS_BEGIN

182 ALLOW_DEPRECATED_DECLARATIONS_END
183 [findPasteboard declareTypes:@[legacyStringPasteboardType()] owner:owner];
184 [findPasteboard setString:string forType:legacyStringPasteboardType()];
185 return [findPasteboard changeCount];

198- (void)_web_writePromisedRTFDFromArchive:(WebArchive*)archive containsImage:(BOOL)containsImage
199{
200 ASSERT(archive);
201 // This image data is either the only subresource of an archive (HTML image case)
202 // or the main resource (standalone image case).
203 NSArray *subresources = [archive subresources];
204 WebResource *resource = [archive mainResource];
205 if (containsImage && [subresources count] > 0) {
206 WebResource *subresource = [subresources objectAtIndex:0];
207 NSString *subresourceMIMEType = [subresource MIMEType];
208 if (MIMETypeRegistry::isSupportedImageMIMEType(subresourceMIMEType) || MIMETypeRegistry::isPDFOrPostScriptMIMEType(subresourceMIMEType))
209 resource = subresource;
210 }
211 ASSERT(resource != nil);
212
213 ASSERT(!containsImage || MIMETypeRegistry::isSupportedImageMIMEType([resource MIMEType]) || MIMETypeRegistry::isPDFOrPostScriptMIMEType([resource MIMEType]));
214 if (!containsImage || MIMETypeRegistry::isSupportedImageMIMEType([resource MIMEType]) || MIMETypeRegistry::isPDFOrPostScriptMIMEType([resource MIMEType]))
215 [self _web_writeFileWrapperAsRTFDAttachment:[resource _fileWrapperRepresentation]];
216
217}

219static CachedImage* imageFromElement(DOMElement *domElement)
220{
221 auto* element = core(domElement);
222 if (!element)
223 return nullptr;
224 auto* renderer = element->renderer();
225 if (!is<RenderImage>(renderer))
226 return nullptr;
227 auto* image = downcast<RenderImage>(*renderer).cachedImage();
228 if (!image || image->errorOccurred())
229 return nullptr;
230 return image;
231}

233- (void)_web_writeImage:(NSImage *)image
234 element:(DOMElement *)element
235 URL:(NSURL *)URL
236 title:(NSString *)title
237 archive:(WebArchive *)archive
238 types:(NSArray *)types
239 source:(WebHTMLView *)source
240{
241 ASSERT(image || element);
242 ASSERT(URL);
243
244 [self _web_writeURL:URL andTitle:title types:types];
245
246 if ([types containsObject:legacyTIFFPasteboardType()]) {
247 if (image)
248 [self setData:[image TIFFRepresentation] forType:legacyTIFFPasteboardType()];
249 else if (source && element)
250 [source setPromisedDragTIFFDataSource:imageFromElement(element)];
251 else if (element)
252 [self setData:[element _imageTIFFRepresentation] forType:legacyTIFFPasteboardType()];
253 }
254
255 if (archive) {
256 if ([types containsObject:WebArchivePboardType])
257 [self setData:[archive data] forType:WebArchivePboardType];
258 return;
259 }
260
261 // We should not have declared types that we aren't going to write (4031826).
262 ASSERT(![types containsObject:legacyRTFDPasteboardType()]);
263 ASSERT(![types containsObject:WebArchivePboardType]);
264}

266- (id)_web_declareAndWriteDragImageForElement:(DOMElement *)element
267 URL:(NSURL *)URL
268 title:(NSString *)title
269 archive:(WebArchive *)archive
270 source:(WebHTMLView *)source
271{
272 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
273 ASSERT(self == [NSPasteboard pasteboardWithName:NSDragPboard]);
274 ALLOW_DEPRECATED_DECLARATIONS_END
275
276 NSString *extension = @"";
277 RetainPtr<NSMutableArray> types = adoptNS([[NSMutableArray alloc] initWithObjects:legacyFilesPromisePasteboardType(), nil]);
278 NSString *originIdentifier = core(element)->document().originIdentifierForPasteboard();
279 RetainPtr<NSData> customDataBuffer;
280 if (originIdentifier.length) {
281 [types addObject:@(PasteboardCustomData::cocoaType())];
282 PasteboardCustomData customData;
283 customData.setOrigin(originIdentifier);
284 customDataBuffer = customData.createSharedBuffer()->createNSData();
285 }
286
287 if (auto* renderer = core(element)->renderer()) {
288 if (is<RenderImage>(*renderer)) {
289 if (auto* image = downcast<RenderImage>(*renderer).cachedImage()) {
290 // FIXME: This doesn't check errorOccured the way imageFromElement does.
291 extension = image->image()->filenameExtension();
292 if (![extension length])
293 return nullptr;
294 [types addObjectsFromArray:[NSPasteboard _web_writableTypesForImageIncludingArchive:(archive != nil)]];
295 [self declareTypes:types.get() owner:source];
296 }
297 }
298#if ENABLE(ATTACHMENT_ELEMENT)
299 else if (is<RenderAttachment>(*renderer)) {
300 extension = URL.pathExtension;
301 [types addObjectsFromArray:[NSPasteboard _web_dragTypesForURL]];
302 [self declareTypes:types.get() owner:source];
303 [self setPropertyList:@[title] forType:legacyFilenamesPasteboardType()];
304 }
305#endif
306 }
307
308 [self _web_writeImage:nil element:element URL:URL title:title archive:archive types:types.get() source:source];
309 if (customDataBuffer)
310 [self setData:customDataBuffer.get() forType:@(PasteboardCustomData::cocoaType())];
311
312 [self setPropertyList:@[extension] forType:legacyFilesPromisePasteboardType()];
313
314 return source;
315}
