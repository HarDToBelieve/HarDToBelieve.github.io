166-(NSData *)_web_schemeSeparatorWithoutColon
167{
168 NSData *result = nil;
169 CFRange rangeWithSeparators;
170 CFRange range = CFURLGetByteRangeForComponent((CFURLRef)self, kCFURLComponentScheme, &rangeWithSeparators);
171 if (rangeWithSeparators.location != kCFNotFound) {
172 NSString *absoluteString = [self absoluteString];
173 NSRange separatorsRange = NSMakeRange(range.location + range.length + 1, rangeWithSeparators.length - range.length - 1);
174 if (separatorsRange.location + separatorsRange.length <= [absoluteString length]) {
175 NSString *slashes = [absoluteString substringWithRange:separatorsRange];
176 result = [slashes dataUsingEncoding:NSISOLatin1StringEncoding];
177 }
178 }
179 return result;
180}

192-(NSData *)_web_hostData
193{
194 NSData *result = WTF::dataForURLComponentType(self, kCFURLComponentHost);
195 NSData *scheme = [self _web_schemeData];
196 // Take off localhost for file
197 if ([scheme _web_isCaseInsensitiveEqualToCString:"file"]) {
198 return ([result _web_isCaseInsensitiveEqualToCString:"localhost"]) ? nil : result;
199 }
200 return result;
201}

274-(NSRange)_webkit_rangeOfURLScheme
275{
276 NSRange colon = [self rangeOfString:@":"];
277 if (colon.location != NSNotFound && colon.location > 0) {
278 NSRange scheme = {0, colon.location};
279 static auto inverseSchemeCharacterSet = makeNeverDestroyed([] {
280 /*
281 This stuff is very expensive. 10-15 msec on a 2x1.2GHz. If not cached it swamps
282 everything else when adding items to the autocomplete DB. Makes me wonder if we
283 even need to enforce the character set here.
284 */
285 NSString *acceptableCharacters = @"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+.-";
286 return retainPtr([[NSCharacterSet characterSetWithCharactersInString:acceptableCharacters] invertedSet]);
287 }());
288 NSRange illegals = [self rangeOfCharacterFromSet:inverseSchemeCharacterSet.get().get() options:0 range:scheme];
289 if (illegals.location == NSNotFound)
290 return scheme;
291 }
292 return NSMakeRange(NSNotFound, 0);
293}
