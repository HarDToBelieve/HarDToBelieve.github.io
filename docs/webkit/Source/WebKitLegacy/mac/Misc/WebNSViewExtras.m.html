56- (NSView *)_web_superviewOfClass:(Class)class
57{
58 NSView *view = [self superview];
59 while (view && ![view isKindOfClass:class])
60 view = [view superview];
61 return view;
62}

64- (WebFrameView *)_web_parentWebFrameView
65{
66 return (WebFrameView *)[self _web_superviewOfClass:[WebFrameView class]];

76/* Determine whether a mouse down should turn into a drag; started as copy of NSTableView code */
77- (BOOL)_web_dragShouldBeginFromMouseDown:(NSEvent *)mouseDownEvent
78 withExpiration:(NSDate *)expiration
79 xHysteresis:(float)xHysteresis
80 yHysteresis:(float)yHysteresis
81{
82 NSEvent *nextEvent, *firstEvent, *dragEvent, *mouseUp;
83 BOOL dragIt;
84
85 if ([mouseDownEvent type] != NSEventTypeLeftMouseDown) {
86 return NO;
87 }
88
89 nextEvent = nil;
90 firstEvent = nil;
91 dragEvent = nil;
92 mouseUp = nil;
93 dragIt = NO;
94
95 while ((nextEvent = [[self window] nextEventMatchingMask:(NSEventMaskLeftMouseUp | NSEventMaskLeftMouseDragged)
96 untilDate:expiration
97 inMode:NSEventTrackingRunLoopMode
98 dequeue:YES]) != nil) {
99 if (firstEvent == nil) {
100 firstEvent = nextEvent;
101 }
102
103 if ([nextEvent type] == NSEventTypeLeftMouseDragged) {
104 float deltax = ABS([nextEvent locationInWindow].x - [mouseDownEvent locationInWindow].x);
105 float deltay = ABS([nextEvent locationInWindow].y - [mouseDownEvent locationInWindow].y);
106 dragEvent = nextEvent;
107
108 if (deltax >= xHysteresis) {
109 dragIt = YES;
110 break;
111 }
112
113 if (deltay >= yHysteresis) {
114 dragIt = YES;
115 break;
116 }
117 } else if ([nextEvent type] == NSEventTypeLeftMouseUp) {
118 mouseUp = nextEvent;
119 break;
120 }
121 }
122
123 // Since we've been dequeuing the events (If we don't, we'll never see the mouse up...),
124 // we need to push some of the events back on. It makes sense to put the first and last
125 // drag events and the mouse up if there was one.
126 if (mouseUp != nil) {
127 [NSApp postEvent:mouseUp atStart:YES];
128 }
129 if (dragEvent != nil) {
130 [NSApp postEvent:dragEvent atStart:YES];
131 }
132 if (firstEvent != mouseUp && firstEvent != dragEvent) {
133 [NSApp postEvent:firstEvent atStart:YES];
134 }
135
136 return dragIt;
137}

139- (BOOL)_web_dragShouldBeginFromMouseDown:(NSEvent *)mouseDownEvent
140 withExpiration:(NSDate *)expiration
141{
142 return [self _web_dragShouldBeginFromMouseDown:mouseDownEvent
143 withExpiration:expiration
144 xHysteresis:WebDragStartHysteresisX
145 yHysteresis:WebDragStartHysteresisY];
146}

149- (NSDragOperation)_web_dragOperationForDraggingInfo:(id <NSDraggingInfo>)sender
150{
151 if (![NSApp modalWindow] &&
152 ![[self window] attachedSheet] &&
153 [sender draggingSource] != self &&
154 [[sender draggingPasteboard] _web_bestURL]) {
155
156 return NSDragOperationCopy;
157 }
158
159 return NSDragOperationNone;
160}

164- (BOOL)_web_firstResponderIsSelfOrDescendantView
165{
166 NSResponder *responder = [[self window] firstResponder];
167 return (responder &&
168 (responder == self ||
169 ([responder isKindOfClass:[NSView class]] && [(NSView *)responder isDescendantOf:self])));
170}
