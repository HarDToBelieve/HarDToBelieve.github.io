124- (void)didShareImageData:(NSData *)data confirmDataIsValidTIFFData:(BOOL)confirmData
125{
126 auto* page = _pickerClient->pageForSharingServicePicker(*self);
127 if (!page)
128 return;
129
130 if (confirmData) {
131 RetainPtr<NSImage> nsImage = adoptNS([[NSImage alloc] initWithData:data]);
132 if (!nsImage) {
133 LOG_ERROR("Shared image data cannot create a valid NSImage");
134 return;
135 }
136
137 data = [nsImage TIFFRepresentation];
138 }
139
140 NSPasteboard *pasteboard = [NSPasteboard pasteboardWithName:serviceControlsPasteboardName];
141 [pasteboard declareTypes:@[ NSPasteboardTypeTIFF ] owner:nil];
142 [pasteboard setData:data forType:NSPasteboardTypeTIFF];
143
144 if (auto* node = page->contextMenuController().context().hitTestResult().innerNode()) {
145 if (auto* frame = node->document().frame())
146 frame->editor().replaceNodeFromPasteboard(node, serviceControlsPasteboardName);
147 }
148
149 [self clear];
150}

154- (NSArray *)sharingServicePicker:(NSSharingServicePicker *)sharingServicePicker sharingServicesForItems:(NSArray *)items mask:(NSSharingServiceMask)mask proposedSharingServices:(NSArray *)proposedServices
155{
156 if (_includeEditorServices)
157 return proposedServices;
158
159 NSMutableArray *services = [NSMutableArray arrayWithCapacity:proposedServices.count];
160
161 for (NSSharingService *service in proposedServices) {
162 if (service.type != NSSharingServiceTypeEditor)
163 [services addObject:service];
164 }
165
166 return services;
167}

182- (void)sharingService:(NSSharingService *)sharingService didShareItems:(NSArray *)items
183{
184 if (!_handleEditingReplacement)
185 return;
186
187 // We only send one item, so we should only get one item back.
188 if ([items count] != 1)
189 return;
190
191 id item = [items objectAtIndex:0];
192
193 if ([item isKindOfClass:[NSImage class]])
194 [self didShareImageData:[item TIFFRepresentation] confirmDataIsValidTIFFData:NO];
195 else if ([item isKindOfClass:[NSItemProvider class]]) {
196 NSItemProvider *itemProvider = (NSItemProvider *)item;
197 NSString *itemUTI = itemProvider.registeredTypeIdentifiers.firstObject;
198
199 [itemProvider loadItemForTypeIdentifier:itemUTI options:nil completionHandler:^(id receivedData, NSError *dataError) {
200 if (!receivedData) {
201 LOG_ERROR("Did not receive data from NSItemProvider");
202 return;
203 }
204
205 if (![receivedData isKindOfClass:[NSData class]]) {
206 LOG_ERROR("Data received from NSItemProvider is not of type NSData");
207 return;
208 }
209
210 [[NSOperationQueue mainQueue] addOperationWithBlock:^{
211 [self didShareImageData:receivedData confirmDataIsValidTIFFData:YES];
212 }];
213
214 }];
215 }
216 else if ([item isKindOfClass:[NSAttributedString class]]) {
217 auto& frame = _pickerClient->pageForSharingServicePicker(*self)->focusController().focusedOrMainFrame();
218 frame.editor().replaceSelectionWithAttributedString(item);
219 } else
220 LOG_ERROR("sharingService:didShareItems: - Unknown item type returned\n");
221}
