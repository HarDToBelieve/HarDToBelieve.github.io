25#import "WebUserContentURLPattern.h"
26
27#import <WebCore/UserContentURLPattern.h>
28#import <wtf/URL.h>
29
30using namespace WebCore;
31
32@interface WebUserContentURLPatternPrivate : NSObject
33{
34@public
35 UserContentURLPattern pattern;
36}
37@end
38
39@implementation WebUserContentURLPatternPrivate
40@end
41
42@implementation WebUserContentURLPattern
43
44- (id)initWithPatternString:(NSString *)patternString
45{
46 self = [super init];
47 if (!self)
48 return nil;
49
50 _private = [[WebUserContentURLPatternPrivate alloc] init];
51 _private->pattern = UserContentURLPattern(patternString);
52
53 return self;
54}
55
56- (void)dealloc
57{
58 [_private release];
59 _private = nil;
60
61 [super dealloc];
62}
63
64- (BOOL)isValid
65{
66 return _private->pattern.isValid();
67}
68
69- (NSString *)scheme
70{
71 return _private->pattern.scheme();
72}
73
74- (NSString *)host
75{
76 return _private->pattern.host();
77}
78
79- (BOOL)matchesSubdomains
80{
81 return _private->pattern.matchSubdomains();
82}
83
84- (BOOL)matchesURL:(NSURL *)url
85{
86 return _private->pattern.matches(url);
87}
88
89@end
