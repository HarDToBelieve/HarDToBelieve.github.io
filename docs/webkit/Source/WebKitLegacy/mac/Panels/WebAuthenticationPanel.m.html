69- (IBAction)cancel:(id)sender
70{
71 // This is required because the body of this method is going to
72 // remove all of the panel's remaining refs, which can cause a
73 // crash later when finishing button hit tracking. So we make
74 // sure it lives on a bit longer.
75 [[panel retain] autorelease];
76
77 // This is required as a workaround for AppKit issue 4118422
78 [[self retain] autorelease];
79
80 if (usingSheet) {
81 [panel.sheetParent endSheet:panel returnCode:NSModalResponseCancel];
82 } else {
83 [panel orderOut:sender];
84 [[NSApplication sharedApplication] stopModalWithCode:1];
85 }
86}
87
88- (IBAction)logIn:(id)sender
89{
90 // This is required because the body of this method is going to
91 // remove all of the panel's remaining refs, which can cause a
92 // crash later when finishing button hit tracking. So we make
93 // sure it lives on a bit longer.
94 [[panel retain] autorelease];
95
96 if (usingSheet) {
97 [panel.sheetParent endSheet:panel returnCode:NSModalResponseOK];
98 } else {
99 [panel orderOut:sender];
100 [[NSApplication sharedApplication] stopModalWithCode:0];
101 }
102}

104- (BOOL)loadNib
105{
106 if (!nibLoaded) {
107 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
108 if ([NSBundle loadNibNamed:WebAuthenticationPanelNibName owner:self]) {
109 ALLOW_DEPRECATED_DECLARATIONS_END
110 nibLoaded = YES;
111 [imageView setImage:[NSImage imageNamed:@"NSApplicationIcon"]];
112 } else {
113 LOG_ERROR("couldn't load nib named '%@'", WebAuthenticationPanelNibName);
114 return FALSE;
115 }
116 }
117 return TRUE;
118}

122-(void)setUpForChallenge:(NSURLAuthenticationChallenge *)chall
123{
124 [self loadNib];
125
126 NSURLProtectionSpace *space = [chall protectionSpace];
127
128 NSString *host;
129 if ([space port] == 0) {
130 host = [[space host] _webkit_decodeHostName];
131 } else {
132 host = [NSString stringWithFormat:@"%@:%ld", [[space host] _webkit_decodeHostName], (long)[space port]];
133 }
134
135 NSString *realm = [space realm];
136 if (!realm)
137 realm = @"";
138 NSString *message;
139
140 // Consider the realm name to be "simple" if it does not contain any whitespace or newline characters.
141 // If the realm name is determined to be complex, we will use a slightly different sheet layout, designed
142 // to keep a malicious realm name from spoofing the wording in the sheet text.
143 BOOL realmNameIsSimple = [realm rangeOfCharacterFromSet:[NSCharacterSet whitespaceAndNewlineCharacterSet]].location == NSNotFound;
144
145 if ([chall previousFailureCount] == 0) {
146 if ([space isProxy]) {
147 message = [NSString stringWithFormat:UI_STRING_INTERNAL("To view this page, you must log in to the %@ proxy server %@.",
148 "prompt string in authentication panel"),
149 [space proxyType], host];
150 } else {
151 if (realmNameIsSimple)
152 message = [NSString stringWithFormat:UI_STRING_INTERNAL("To view this page, you must log in to area “%@” on %@.",
153 "prompt string in authentication panel"), realm, host];
154 else
155 message = [NSString stringWithFormat:UI_STRING_INTERNAL("To view this page, you must log in to this area on %@:",
156 "prompt string in authentication panel"), host];
157 }
158 } else {
159 if ([space isProxy]) {
160 message = [NSString stringWithFormat:UI_STRING_INTERNAL("The user name or password you entered for the %@ proxy server %@ was incorrect. Make sure you’re entering them correctly, and then try again.",
161 "prompt string in authentication panel"),
162 [space proxyType], host];
163 } else {
164 if (realmNameIsSimple)
165 message = [NSString stringWithFormat:UI_STRING_INTERNAL("The user name or password you entered for area “%@” on %@ was incorrect. Make sure you’re entering them correctly, and then try again.",
166 "prompt string in authentication panel"), realm, host];
167 else
168 message = [NSString stringWithFormat:UI_STRING_INTERNAL("The user name or password you entered for this area on %@ was incorrect. Make sure you’re entering them correctly, and then try again.",
169 "prompt string in authentication panel"), host];
170 }
171 }
172
173 if (![space isProxy] && !realmNameIsSimple) {
174 [separateRealmLabel setHidden:NO];
175 [separateRealmLabel setStringValue:realm];
176 [separateRealmLabel setAutoresizingMask:NSViewMinYMargin];
177 [separateRealmLabel sizeToFitAndAdjustWindowHeight];
178 [separateRealmLabel setAutoresizingMask:NSViewMaxYMargin];
179 } else {
180 // In the proxy or "simple" realm name case, we need to hide the 'separateRealmLabel'
181 // and move the rest of the contents up appropriately to fill the space.
182 NSRect mainLabelFrame = [mainLabel frame];
183 NSRect realmFrame = [separateRealmLabel frame];
184 NSRect smallLabelFrame = [smallLabel frame];
185
186 // Find the distance between the 'smallLabel' and the label above it, initially the 'separateRealmLabel'.
187 // Then, find the current distance between 'smallLabel' and 'mainLabel'. The difference between
188 // these two is how much shorter the panel needs to be after hiding the 'separateRealmLabel'.
189 CGFloat smallLabelMargin = NSMinY(realmFrame) - NSMaxY(smallLabelFrame);
190 CGFloat smallLabelToMainLabel = NSMinY(mainLabelFrame) - NSMaxY(smallLabelFrame);
191 CGFloat deltaMargin = smallLabelToMainLabel - smallLabelMargin;
192
193 [separateRealmLabel setHidden:YES];
194 NSRect windowFrame = [panel frame];
195 windowFrame.size.height -= deltaMargin;
196 [panel setFrame:windowFrame display:NO];
197 }
198
199 [mainLabel setStringValue:message];
200 [mainLabel sizeToFitAndAdjustWindowHeight];
201
202 if ([space receivesCredentialSecurely] || [[space protocol] _webkit_isCaseInsensitiveEqualToString:@"https"]) {
203 [smallLabel setStringValue:
204 UI_STRING_INTERNAL("Your login information will be sent securely.",
205 "message in authentication panel")];
206 } else {
207 // Use this scary-sounding phrase only when using basic auth with non-https servers. In this case the password
208 // could be sniffed by intercepting the network traffic.
209 [smallLabel setStringValue:
210 UI_STRING_INTERNAL("Your password will be sent unencrypted.",
211 "message in authentication panel")];
212 }
213
214 if ([[chall proposedCredential] user] != nil) {
215 [username setStringValue:[[chall proposedCredential] user]];
216 [panel setInitialFirstResponder:password];
217 } else {
218 [username setStringValue:@""];
219 [password setStringValue:@""];
220 [panel setInitialFirstResponder:username];
221 }
222}

224- (void)runAsModalDialogWithChallenge:(NSURLAuthenticationChallenge *)chall
225{
226 [self setUpForChallenge:chall];
227
228 usingSheet = FALSE;
229 [chall retain];
230 NSURLCredential *credential = nil;
231
232 if ([[NSApplication sharedApplication] runModalForWindow:panel] == 0) {
233 credential = [[NSURLCredential alloc] initWithUser:[username stringValue] password:[password stringValue] persistence:([remember state] == NSControlStateValueOn) ? NSURLCredentialPersistencePermanent : NSURLCredentialPersistenceForSession];
234 }
235
236 [callback performSelector:selector withObject:chall withObject:credential];
237 [credential release];
238 [chall release];
239}
240
241- (void)runAsSheetOnWindow:(NSWindow *)window withChallenge:(NSURLAuthenticationChallenge *)chall
242{
243 ASSERT(!usingSheet);
244
245 [self setUpForChallenge:chall];
246
247 usingSheet = TRUE;
248 challenge = [chall retain];
249
250 [window beginSheet:panel completionHandler:^(NSModalResponse modalResponse) {
251 int returnCode = (modalResponse == NSModalResponseCancel) ? 1 : 0;
252 [self sheetDidEnd:panel returnCode:returnCode contextInfo:NULL];
253 }];
254}
255
256- (void)sheetDidEnd:(NSWindow *)sheet returnCode:(int)returnCode contextInfo:(void *)contextInfo
257{
258 NSURLCredential *credential = nil;
259 NSURLAuthenticationChallenge *chall;
260
261 ASSERT(usingSheet);
262 ASSERT(challenge != nil);
263
264 if (returnCode == 0) {
265 credential = [[NSURLCredential alloc] initWithUser:[username stringValue] password:[password stringValue] persistence:([remember state] == NSControlStateValueOn) ? NSURLCredentialPersistencePermanent : NSURLCredentialPersistenceForSession];
266 }
267
268 // We take this tricky approach to nilling out and releasing the challenge
269 // because the callback below might remove our last ref.
270 chall = challenge;
271 challenge = nil;
272 [callback performSelector:selector withObject:chall withObject:credential];
273 [credential release];
274 [chall release];
275}

281- (BOOL)_blocksActionWhenModal:(SEL)theAction
282{
283 // This override of a private AppKit method allows the user to quit when a login dialog
284 // is onscreen, which is nice in general but in particular prevents pathological cases
285 // like 3744583 from requiring a Force Quit.
286 //
287 // It would be nice to allow closing the individual window as well as quitting the app when
288 // a login sheet is up, but this _blocksActionWhenModal: mechanism doesn't support that.
289 // This override matches those in NSOpenPanel and NSToolbarConfigPanel.
290 if (theAction == @selector(terminate:)) {
291 return NO;
292 }
293 return YES;
294}
