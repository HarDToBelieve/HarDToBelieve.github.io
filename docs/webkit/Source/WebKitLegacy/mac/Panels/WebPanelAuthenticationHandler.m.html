31#import <WebKitLegacy/WebPanelAuthenticationHandler.h>

33#import <Foundation/NSURLAuthenticationChallenge.h>
34#import <WebKitLegacy/WebAuthenticationPanel.h>
35#import <wtf/Assertions.h>
36
37static NSString *WebModalDialogPretendWindow = @"WebModalDialogPretendWindow";
38
39@implementation WebPanelAuthenticationHandler
40
41WebPanelAuthenticationHandler *sharedHandler;
42
43+ (id)sharedHandler
44{
45 if (sharedHandler == nil)
46 sharedHandler = [[self alloc] init];
47 return sharedHandler;
48}
49
50-(id)init
51{
52 self = [super init];
53 if (self != nil) {
54 windowToPanel = [[NSMapTable alloc] initWithKeyOptions:NSPointerFunctionsStrongMemory valueOptions:NSPointerFunctionsStrongMemory capacity:0];
55 challengeToWindow = [[NSMapTable alloc] initWithKeyOptions:NSPointerFunctionsStrongMemory valueOptions:NSPointerFunctionsStrongMemory capacity:0];
56 windowToChallengeQueue = [[NSMapTable alloc] initWithKeyOptions:NSPointerFunctionsStrongMemory valueOptions:NSPointerFunctionsStrongMemory capacity:0];
57 }
58
59 return self;
60}
61
62-(void)dealloc
63{
64 [windowToPanel release];
65 [challengeToWindow release];
66 [windowToChallengeQueue release];
67 [super dealloc];
68}
69
70-(void)enqueueChallenge:(NSURLAuthenticationChallenge *)challenge forWindow:(id)window
71{
72 NSMutableArray *queue = [windowToChallengeQueue objectForKey:window];
73 if (!queue) {
74 queue = [[NSMutableArray alloc] init];
75 [windowToChallengeQueue setObject:queue forKey:window];
76 [queue release];
77 }
78 [queue addObject:challenge];
79}
80
81-(void)tryNextChallengeForWindow:(id)window
82{
83 NSMutableArray *queue = [windowToChallengeQueue objectForKey:window];
84 if (!queue)
85 return;
86
87 NSURLAuthenticationChallenge *challenge = [[queue objectAtIndex:0] retain];
88 [queue removeObjectAtIndex:0];
89 if (![queue count])
90 [windowToChallengeQueue removeObjectForKey:window];
91
92 NSURLCredential *latestCredential = [[NSURLCredentialStorage sharedCredentialStorage] defaultCredentialForProtectionSpace:[challenge protectionSpace]];
93
94 if ([latestCredential hasPassword]) {
95 [[challenge sender] useCredential:latestCredential forAuthenticationChallenge:challenge];
96 [challenge release];
97 return;
98 }
99
100 [self startAuthentication:challenge window:(window == WebModalDialogPretendWindow ? nil : window)];
101 [challenge release];
102}
103
104
105-(void)startAuthentication:(NSURLAuthenticationChallenge *)challenge window:(NSWindow *)w
106{
107 id window = w ? (id)w : (id)WebModalDialogPretendWindow;
108
109 if ([windowToPanel objectForKey:window] != nil) {
110 [self enqueueChallenge:challenge forWindow:window];
111 return;
112 }
113
114 // In this case, we have an attached sheet that's not one of our
115 // authentication panels, so enqueing is not an option. Just
116 // cancel loading instead, since this case is fairly
117 // unlikely (how would you be loading a page if you had an error
118 // sheet up?)
119 if ([w attachedSheet] != nil) {
120 [[challenge sender] cancelAuthenticationChallenge:challenge];
121 return;
122 }
123
124 WebAuthenticationPanel *panel = [[WebAuthenticationPanel alloc] initWithCallback:self selector:@selector(_authenticationDoneWithChallenge:result:)];
125 [challengeToWindow setObject:window forKey:challenge];
126 [windowToPanel setObject:panel forKey:window];
127 [panel release];
128
129 if (window == WebModalDialogPretendWindow)
130 [panel runAsModalDialogWithChallenge:challenge];
131 else
132 [panel runAsSheetOnWindow:window withChallenge:challenge];
133}
134
135-(void)cancelAuthentication:(NSURLAuthenticationChallenge *)challenge
136{
137 id window = [challengeToWindow objectForKey:challenge];
138 if (!window)
139 return;
140
141 WebAuthenticationPanel *panel = [windowToPanel objectForKey:window];
142 [panel cancel:self];
143}
144
145-(void)_authenticationDoneWithChallenge:(NSURLAuthenticationChallenge *)challenge result:(NSURLCredential *)credential
146{
147 id window = [challengeToWindow objectForKey:challenge];
148 [window retain];
149 if (window != nil) {
150 [windowToPanel removeObjectForKey:window];
151 [challengeToWindow removeObjectForKey:challenge];
152 }
153
154 if (credential == nil) {
155 [[challenge sender] continueWithoutCredentialForAuthenticationChallenge:challenge];
156 } else {
157 [[challenge sender] useCredential:credential forAuthenticationChallenge:challenge];
158 }
159
160 [self tryNextChallengeForWindow:window];
161 [window release];
162}
163
164@end
