133uint32_t NetscapePluginInstanceProxy::LocalObjectMap::idForObject(VM& vm, JSObject* object)
134{
135 // This method creates objects with refcount of 1, but doesn't increase refcount when returning
136 // found objects. This extra count accounts for the main "reference" kept by plugin process.
137
138 // To avoid excessive IPC, plugin process doesn't send each NPObject release/retain call to
139 // Safari. It only sends one when the last reference is removed, and it can destroy the proxy
140 // NPObject.
141
142 // However, the browser may be sending the same object out to plug-in as a function call
143 // argument at the same time - neither side can know what the other one is doing. So,
144 // is to make PCForgetBrowserObject call return a boolean result, making it possible for
145 // the browser to make plugin host keep the proxy with zero refcount for a little longer.
146
147 uint32_t objectID = 0;
148
149 HashMap<JSC::JSObject*, std::pair<uint32_t, uint32_t>>::iterator iter = m_jsObjectToIDMap.find(object);
150 if (iter != m_jsObjectToIDMap.end())
151 return iter->value.first;
152
153 do {
154 objectID = ++m_objectIDCounter;
155 } while (!m_objectIDCounter || m_objectIDCounter == static_cast<uint32_t>(-1) || m_idToJSObjectMap.contains(objectID));
156
157 m_idToJSObjectMap.set(objectID, Strong<JSObject>(vm, object));
158 m_jsObjectToIDMap.set(object, std::make_pair(objectID, 1));
159
160 return objectID;
161}
162
163void NetscapePluginInstanceProxy::LocalObjectMap::retain(JSC::JSObject* object)
164{
165 HashMap<JSC::JSObject*, std::pair<uint32_t, uint32_t>>::iterator iter = m_jsObjectToIDMap.find(object);
166 ASSERT(iter != m_jsObjectToIDMap.end());
167
168 iter->value.second = iter->value.second + 1;
169}
170
171void NetscapePluginInstanceProxy::LocalObjectMap::release(JSC::JSObject* object)
172{
173 HashMap<JSC::JSObject*, std::pair<uint32_t, uint32_t>>::iterator iter = m_jsObjectToIDMap.find(object);
174 ASSERT(iter != m_jsObjectToIDMap.end());
175
176 ASSERT(iter->value.second > 0);
177 iter->value.second = iter->value.second - 1;
178 if (!iter->value.second) {
179 m_idToJSObjectMap.remove(iter->value.first);
180 m_jsObjectToIDMap.remove(iter);
181 }
182}

190bool NetscapePluginInstanceProxy::LocalObjectMap::forget(uint32_t objectID)
191{
192 if (objectID == HashTraits<uint32_t>::emptyValue() || HashTraits<uint32_t>::isDeletedValue(objectID)) {
193 LOG_ERROR("NetscapePluginInstanceProxy::LocalObjectMap::forget: local object id %u is not valid.", objectID);
194 return true;
195 }
196
197 HashMap<uint32_t, JSC::Strong<JSC::JSObject>>::iterator iter = m_idToJSObjectMap.find(objectID);
198 if (iter == m_idToJSObjectMap.end()) {
199 LOG_ERROR("NetscapePluginInstanceProxy::LocalObjectMap::forget: local object %u doesn't exist.", objectID);
200 return true;
201 }
202
203 HashMap<JSC::JSObject*, std::pair<uint32_t, uint32_t>>::iterator rIter = m_jsObjectToIDMap.find(iter->value.get());
204
205 // If the object is being sent to plug-in right now, then it's not the time to forget.
206 if (rIter->value.second != 1)
207 return false;
208
209 m_jsObjectToIDMap.remove(rIter);
210 m_idToJSObjectMap.remove(iter);
211 return true;
212}

220NetscapePluginInstanceProxy::NetscapePluginInstanceProxy(NetscapePluginHostProxy* pluginHostProxy, WebHostedNetscapePluginView *pluginView, bool fullFramePlugin)
221 : m_pluginHostProxy(pluginHostProxy)
222 , m_pluginView(pluginView)
223 , m_requestTimer(*this, &NetscapePluginInstanceProxy::requestTimerFired)
224 , m_currentURLRequestID(0)
225 , m_renderContextID(0)
226 , m_rendererType(UseSoftwareRenderer)
227 , m_waitingForReply(false)
228 , m_pluginFunctionCallDepth(0)
229 , m_shouldStopSoon(false)
230 , m_currentRequestID(0)
231 , m_pluginIsWaitingForDraw(false)
232{
233 ASSERT(m_pluginView);
234
235 if (fullFramePlugin) {
236 // For full frame plug-ins, the first requestID will always be the one for the already
237 // open stream.
238 ++m_currentURLRequestID;
239 }
240
241 // Assign a plug-in ID.
242 do {
243 m_pluginID = ++pluginIDCounter;
244 } while (pluginHostProxy->pluginInstance(m_pluginID) || !m_pluginID);
245
246#ifndef NDEBUG
247 netscapePluginInstanceProxyCounter.increment();
248#endif
249}

301void NetscapePluginInstanceProxy::cleanup()
302{
303 stopAllStreams();
304
305 m_requestTimer.stop();
306
307 // Clear the object map, this will cause any outstanding JS objects that the plug-in had a reference to
308 // to go away when the next garbage collection takes place.
309 m_localObjects.clear();
310
311 if (Frame* frame = core([m_pluginView webFrame]))
312 frame->script().cleanupScriptObjectsForPlugin((__bridge void*)m_pluginView);
313
314 ProxyInstanceSet instances;
315 instances.swap(m_instances);
316
317 // Invalidate all proxy instances.
318 ProxyInstanceSet::const_iterator end = instances.end();
319 for (ProxyInstanceSet::const_iterator it = instances.begin(); it != end; ++it)
320 (*it)->invalidate();
321
322 m_pluginView = nil;
323 m_manualStream = nullptr;
324}

336void NetscapePluginInstanceProxy::destroy()
337{
338 uint32_t requestID = nextRequestID();
339
340 ASSERT(!m_inDestroy);
341 m_inDestroy = true;
342
343 FrameLoadMap::iterator end = m_pendingFrameLoads.end();
344 for (FrameLoadMap::iterator it = m_pendingFrameLoads.begin(); it != end; ++it)
345 [(it->key) _setInternalLoadDelegate:nil];
346
347 _WKPHDestroyPluginInstance(m_pluginHostProxy->port(), m_pluginID, requestID);
348
349 // If the plug-in host crashes while we're waiting for a reply, the last reference to the instance proxy
350 // will go away. Prevent this by protecting it here.
351 Ref<NetscapePluginInstanceProxy> protect(*this);
352
353 // We don't care about the reply here - we just want to block until the plug-in instance has been torn down.
354 waitForReply<NetscapePluginInstanceProxy::BooleanReply>(requestID);
355
356 m_inDestroy = false;
357
358 cleanup();
359 invalidate();
360}

504void NetscapePluginInstanceProxy::print(CGContextRef context, unsigned width, unsigned height)
505{
506 uint32_t requestID = nextRequestID();
507 _WKPHPluginInstancePrint(m_pluginHostProxy->port(), m_pluginID, requestID, width, height);
508
509 auto reply = waitForReply<NetscapePluginInstanceProxy::BooleanAndDataReply>(requestID);
510 if (!reply || !reply->m_returnValue)

512
513 RetainPtr<CGDataProvider> dataProvider = adoptCF(CGDataProviderCreateWithCFData(reply->m_result.get()));
514 RetainPtr<CGColorSpaceRef> colorSpace = adoptCF(CGColorSpaceCreateDeviceRGB());
515 RetainPtr<CGImageRef> image = adoptCF(CGImageCreate(width, height, 8, 32, width * 4, colorSpace.get(), kCGImageAlphaFirst, dataProvider.get(), 0, false, kCGRenderingIntentDefault));
516
517 // Flip the context and draw the image.
518 CGContextSaveGState(context);
519 CGContextTranslateCTM(context, 0.0, height);
520 CGContextScaleCTM(context, 1.0, -1.0);
521
522 CGContextDrawImage(context, CGRectMake(0, 0, width, height), image.get());
523
524 CGContextRestoreGState(context);
525}

558NPError NetscapePluginInstanceProxy::loadURL(const char* url, const char* target, const char* postData, uint32_t postLen, LoadURLFlags flags, uint32_t& streamID)
559{
560 if (!url)
561 return NPERR_INVALID_PARAM;
562
563 NSMutableURLRequest *request = [m_pluginView requestWithURLCString:url];
564
565 if (flags & IsPost) {
566 NSData *httpBody = nil;
567
568 if (flags & PostDataIsFile) {
569 // If we're posting a file, buf is either a file URL or a path to the file.
570 if (!postData)
571 return NPERR_INVALID_PARAM;
572 RetainPtr<CFStringRef> bufString = adoptCF(CFStringCreateWithCString(kCFAllocatorDefault, postData, kCFStringEncodingWindowsLatin1));
573 if (!bufString)
574 return NPERR_INVALID_PARAM;
575
576 NSURL *fileURL = [NSURL _web_URLWithDataAsString:(__bridge NSString *)bufString.get()];
577 NSString *path;
578 if ([fileURL isFileURL])
579 path = [fileURL path];
580 else
581 path = (__bridge NSString *)bufString.get();
582 httpBody = [NSData dataWithContentsOfFile:path];
583 if (!httpBody)
584 return NPERR_FILE_NOT_FOUND;
585 } else
586 httpBody = [NSData dataWithBytes:postData length:postLen];
587
588 if (![httpBody length])
589 return NPERR_INVALID_PARAM;
590
591 [request setHTTPMethod:@"POST"];
592
593 if (flags & AllowHeadersInPostData) {
594 if ([httpBody _web_startsWithBlankLine])
595 httpBody = [httpBody subdataWithRange:NSMakeRange(1, [httpBody length] - 1)];
596 else {
597 NSInteger location = [httpBody _web_locationAfterFirstBlankLine];
598 if (location != NSNotFound) {
599 // If the blank line is somewhere in the middle of postData, everything before is the header.
600 NSData *headerData = [httpBody subdataWithRange:NSMakeRange(0, location)];
601 NSMutableDictionary *header = [headerData _webkit_parseRFC822HeaderFields];
602 unsigned dataLength = [httpBody length] - location;
603
604 // Sometimes plugins like to set Content-Length themselves when they post,
605 // but CFNetwork does not like that. So we will remove the header
606 // and instead truncate the data to the requested length.
607 NSString *contentLength = [header objectForKey:@"Content-Length"];
608
609 if (contentLength)
610 dataLength = std::min(static_cast<unsigned>([contentLength intValue]), dataLength);
611 [header removeObjectForKey:@"Content-Length"];
612
613 if ([header count] > 0)
614 [request setAllHTTPHeaderFields:header];
615
616 // Everything after the blank line is the actual content of the POST.
617 httpBody = [httpBody subdataWithRange:NSMakeRange(location, dataLength)];
618 }
619 }
620 }
621
622 if (![httpBody length])
623 return NPERR_INVALID_PARAM;
624
625 // Plug-ins expect to receive uncached data when doing a POST (3347134).
626 [request setCachePolicy:NSURLRequestReloadIgnoringCacheData];
627 [request setHTTPBody:httpBody];
628 }
629
630 return loadRequest(request, target, flags & AllowPopups, streamID);
631}

633void NetscapePluginInstanceProxy::performRequest(PluginRequest* pluginRequest)
634{
635 // Loading the request can cause the instance proxy to go away, so protect it.
636 Ref<NetscapePluginInstanceProxy> protect(*this);
637
638 ASSERT(m_pluginView);
639
640 NSURLRequest *request = pluginRequest->request();
641 NSString *frameName = pluginRequest->frameName();
642 WebFrame *frame = nil;
643
644 NSURL *URL = [request URL];
645 NSString *JSString = [URL _webkit_scriptIfJavaScriptURL];
646
647 ASSERT(frameName || JSString);
648 if (frameName) {
649 // FIXME - need to get rid of this window creation which
650 // bypasses normal targeted link handling
651 frame = kit(core([m_pluginView webFrame])->loader().findFrameForNavigation(frameName));
652 if (!frame) {

656 if (!newWebView) {
657 _WKPHLoadURLNotify(m_pluginHostProxy->port(), m_pluginID, pluginRequest->requestID(), NPERR_GENERIC_ERROR);

659 }
660
661 frame = [newWebView mainFrame];
662 core(frame)->tree().setName(frameName);
663 [[newWebView _UIDelegateForwarder] webViewShow:newWebView];
664 }
665 }
666
667 if (JSString) {
668 ASSERT(!frame || [m_pluginView webFrame] == frame);
669 evaluateJavaScript(pluginRequest);
670 } else {
671 [frame loadRequest:request];
672
673 // Check if another plug-in view or even this view is waiting for the frame to load.
674 // If it is, tell it that the load was cancelled because it will be anyway.
675 WebHostedNetscapePluginView *view = [frame _internalLoadDelegate];
676 if (view != nil) {
677 ASSERT([view isKindOfClass:[WebHostedNetscapePluginView class]]);
678 [view webFrame:frame didFinishLoadWithReason:NPRES_USER_BREAK];
679 }

699void NetscapePluginInstanceProxy::evaluateJavaScript(PluginRequest* pluginRequest)
700{
701 NSURL *URL = [pluginRequest->request() URL];
702 NSString *JSString = [URL _webkit_scriptIfJavaScriptURL];
703 ASSERT(JSString);
704
705 Ref<NetscapePluginInstanceProxy> protect(*this); // Executing arbitrary JavaScript can destroy the proxy.
706
707 NSString *result = [[m_pluginView webFrame] _stringByEvaluatingJavaScriptFromString:JSString forceUserGesture:pluginRequest->allowPopups()];
708
709 // Don't continue if stringByEvaluatingJavaScriptFromString caused the plug-in to stop.
710 if (!m_pluginHostProxy)
711 return;
712
713 if (pluginRequest->frameName() != nil)
714 return;
715
716 if ([result length] > 0) {
717 // Don't call NPP_NewStream and other stream methods if there is no JS result to deliver. This is what Mozilla does.
718 NSData *JSData = [result dataUsingEncoding:NSUTF8StringEncoding];
719
720 auto stream = HostedNetscapePluginStream::create(this, pluginRequest->requestID(), pluginRequest->request());
721 m_streams.add(stream->streamID(), stream.copyRef());
722
723 RetainPtr<NSURLResponse> response = adoptNS([[NSURLResponse alloc] initWithURL:URL
724 MIMEType:@"text/plain"
725 expectedContentLength:[JSData length]
726 textEncodingName:nil]);
727 stream->startStreamWithResponse(response.get());
728 stream->didReceiveData(0, static_cast<const uint8_t*>([JSData bytes]), [JSData length]);
729 stream->didFinishLoading(0);
730 }

747NPError NetscapePluginInstanceProxy::loadRequest(NSURLRequest *request, const char* cTarget, bool allowPopups, uint32_t& requestID)
748{
749 NSURL *URL = [request URL];
750
751 if (!URL)
752 return NPERR_INVALID_URL;
753
754 // Don't allow requests to be loaded when the document loader is stopping all loaders.
755 DocumentLoader* documentLoader = [[m_pluginView dataSource] _documentLoader];
756 if (!documentLoader || documentLoader->isStopping())
757 return NPERR_GENERIC_ERROR;
758
759 NSString *target = nil;
760 if (cTarget) {
761 // Find the frame given the target string.
762 target = [NSString stringWithCString:cTarget encoding:NSISOLatin1StringEncoding];
763 }
764 WebFrame *frame = [m_pluginView webFrame];
765
766 // don't let a plugin start any loads if it is no longer part of a document that is being
767 // displayed unless the loads are in the same frame as the plugin.
768 if (documentLoader != core([m_pluginView webFrame])->loader().activeDocumentLoader() &&
769 (!cTarget || [frame findFrameNamed:target] != frame)) {
770 return NPERR_GENERIC_ERROR;
771 }
772
773 NSString *JSString = [URL _webkit_scriptIfJavaScriptURL];
774 if (JSString != nil) {
775 if (![[[m_pluginView webView] preferences] isJavaScriptEnabled]) {
776 // Return NPERR_GENERIC_ERROR if JS is disabled. This is what Mozilla does.
777 return NPERR_GENERIC_ERROR;
778 }
779 } else {
780 if (!core([m_pluginView webFrame])->document()->securityOrigin().canDisplay(URL))
781 return NPERR_GENERIC_ERROR;
782 }
783
784 // FIXME: Handle wraparound
785 requestID = ++m_currentURLRequestID;
786
787 if (cTarget || JSString) {
788 // Make when targeting a frame or evaluating a JS string, perform the request after a delay because we don't
789 // want to potentially kill the plug-in inside of its URL request.
790
791 if (JSString && target && [frame findFrameNamed:target] != frame) {
792 // For security reasons, only allow JS requests to be made on the frame that contains the plug-in.
793 return NPERR_INVALID_PARAM;
794 }
795
796 auto pluginRequest = PluginRequest::create(requestID, request, target, allowPopups);
797 m_pluginRequests.append(WTFMove(pluginRequest));
798 m_requestTimer.startOneShot(0_s);
799 } else {
800 auto stream = HostedNetscapePluginStream::create(this, requestID, request);
801
802 ASSERT(!m_streams.contains(requestID));
803 m_streams.add(requestID, stream.copyRef());
804 stream->start();
805 }
806
807 return NPERR_NO_ERROR;
808}

810std::unique_ptr<NetscapePluginInstanceProxy::Reply> NetscapePluginInstanceProxy::processRequestsAndWaitForReply(uint32_t requestID)
811{
812 ASSERT(m_pluginHostProxy);
813
814 std::unique_ptr<Reply> reply;
815
816 while (!(reply = m_replies.take(requestID))) {
817 if (!m_pluginHostProxy->processRequests())

823 return nullptr;
824 }
825
826 return reply;

1313{

1315 JSLockHolder lock(vm);
1316

1319 [array addObject:asString(value)->value(lexicalGlobalObject)];

1370bool NetscapePluginInstanceProxy::demarshalValueFromArray(JSGlobalObject* lexicalGlobalObject, NSArray *array, NSUInteger& index, JSValue& result)
1371{
1372 if (index == [array count])
1373 return false;
1374
1375 int type = [[array objectAtIndex:index++] intValue];
1376 switch (type) {
1377 case VoidValueType:
1378 result = jsUndefined();
1379 return true;
1380 case NullValueType:
1381 result = jsNull();
1382 return true;
1383 case BoolValueType:
1384 result = jsBoolean([[array objectAtIndex:index++] boolValue]);
1385 return true;
1386 case DoubleValueType:
1387 result = jsNumber([[array objectAtIndex:index++] doubleValue]);
1388 return true;
1389 case StringValueType: {
1390 NSString *string = [array objectAtIndex:index++];
1391
1392 result = jsString(lexicalGlobalObject->vm(), String(string));
1393 return true;
1394 }
1395 case JSObjectValueType: {
1396 uint32_t objectID = [[array objectAtIndex:index++] intValue];
1397
1398 result = m_localObjects.get(objectID);
1399 ASSERT(result);
1400 return true;
1401 }
1402 case NPObjectValueType: {
1403 uint32_t objectID = [[array objectAtIndex:index++] intValue];
1404
1405 Frame* frame = core([m_pluginView webFrame]);
1406 if (!frame)
1407 return false;
1408
1409 if (!frame->script().canExecuteScripts(NotAboutToExecuteScript))
1410 return false;
1411
1412 auto rootObject = frame->script().createRootObject((__bridge void*)m_pluginView);
1413 result = ProxyInstance::create(WTFMove(rootObject), this, objectID)->createRuntimeObject(lexicalGlobalObject);
1414 return true;
1415 }
1416 default:
1417 ASSERT_NOT_REACHED();
1418 return false;
1419 }
1420}
