31#import "NetscapePluginInstanceProxy.h"

36namespace WebKit {
37
38class ProxyClass;
39
40class ProxyInstance : public JSC::Bindings::Instance {
41public:
42 static Ref<ProxyInstance> create(Ref<JSC::Bindings::RootObject>&& rootObject, NetscapePluginInstanceProxy* instanceProxy, uint32_t objectID)
43 {
44 return adoptRef(*new ProxyInstance(WTFMove(rootObject), instanceProxy, objectID));
45 }
46 ~ProxyInstance();
47
48 JSC::Bindings::Method* methodNamed(JSC::PropertyName);
49 JSC::Bindings::Field* fieldNamed(JSC::PropertyName);
50
51 JSC::JSValue fieldValue(JSC::JSGlobalObject*, const JSC::Bindings::Field*) const;
52 bool setFieldValue(JSC::JSGlobalObject*, const JSC::Bindings::Field*, JSC::JSValue) const;
53
54
55 NetscapePluginInstanceProxy* instanceProxy() { return m_instanceProxy; }
56
57 void invalidate();
58
59 uint32_t objectID() const { return m_objectID; }
60
61private:
62 ProxyInstance(Ref<JSC::Bindings::RootObject>&&, NetscapePluginInstanceProxy*, uint32_t objectID);
63
64 virtual JSC::Bindings::RuntimeObject* newRuntimeObject(JSC::JSGlobalObject*);
65
66 virtual JSC::Bindings::Class* getClass() const;
67
68 virtual JSC::JSValue getMethod(JSC::JSGlobalObject*, JSC::PropertyName);
69 virtual JSC::JSValue invokeMethod(JSC::JSGlobalObject*, JSC::CallFrame*, JSC::RuntimeMethod*);
70
71 virtual bool supportsInvokeDefaultMethod() const;
72 virtual JSC::JSValue invokeDefaultMethod(JSC::JSGlobalObject*, JSC::CallFrame*);
73
74 virtual bool supportsConstruct() const;
75 virtual JSC::JSValue invokeConstruct(JSC::JSGlobalObject*, JSC::CallFrame*, const JSC::ArgList&);
76
77 virtual JSC::JSValue defaultValue(JSC::JSGlobalObject*, JSC::PreferredPrimitiveType) const;
78 virtual JSC::JSValue valueOf(JSC::JSGlobalObject*) const;
79
80 virtual void getPropertyNames(JSC::JSGlobalObject*, JSC::PropertyNameArray&);
81
82 JSC::JSValue stringValue(JSC::JSGlobalObject*) const;
83 JSC::JSValue numberValue(JSC::JSGlobalObject*) const;
84 JSC::JSValue booleanValue() const;
85
86 JSC::JSValue invoke(JSC::JSGlobalObject*, JSC::CallFrame*, InvokeType, uint64_t identifier, const JSC::ArgList&);
87
88 template <typename T>
89 std::unique_ptr<T> waitForReply(uint32_t requestID) const
90 {
91 auto reply = m_instanceProxy->waitForReply<T>(requestID);
92
93 // If the instance proxy was invalidated, just return a null reply.
94 if (!m_instanceProxy)
95 return nullptr;
96
97 return reply;
98 }
99
100 NetscapePluginInstanceProxy* m_instanceProxy;
101 uint32_t m_objectID;
102 HashMap<RefPtr<StringImpl>, std::unique_ptr<JSC::Bindings::Field>> m_fields;
103 HashMap<RefPtr<StringImpl>, std::unique_ptr<JSC::Bindings::Method>> m_methods;
104};
105
106}
