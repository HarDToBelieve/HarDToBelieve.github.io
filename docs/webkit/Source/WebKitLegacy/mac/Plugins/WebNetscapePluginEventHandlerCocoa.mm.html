40static inline void initializeEvent(NPCocoaEvent* event, NPCocoaEventType type)
41{
42 event->type = type;
43 event->version = 0;
44}
45
46void WebNetscapePluginEventHandlerCocoa::drawRect(CGContextRef context, const NSRect& rect)
47{
48 NPCocoaEvent event;
49
50 initializeEvent(&event, NPCocoaEventDrawRect);
51 event.data.draw.context = context;
52 event.data.draw.x = rect.origin.x;
53 event.data.draw.y = rect.origin.y;
54 event.data.draw.width = rect.size.width;
55 event.data.draw.height = rect.size.height;
56
57 RetainPtr<CGContextRef> protect(context);
58
59 sendEvent(&event);
60}

97bool WebNetscapePluginEventHandlerCocoa::sendMouseEvent(NSEvent *nsEvent, NPCocoaEventType type)
98{
99 NPCocoaEvent event;
100
101 NSPoint point = [m_pluginView convertPoint:[nsEvent locationInWindow] fromView:nil];
102
103 int clickCount;
104 if (type == NPCocoaEventMouseEntered || type == NPCocoaEventMouseExited || type == NPCocoaEventScrollWheel)
105 clickCount = 0;
106 else
107 clickCount = [nsEvent clickCount];
108
109 initializeEvent(&event, type);
110 event.data.mouse.modifierFlags = [nsEvent modifierFlags];
111 event.data.mouse.buttonNumber = [nsEvent buttonNumber];
112 event.data.mouse.clickCount = clickCount;
113 event.data.mouse.pluginX = point.x;
114 event.data.mouse.pluginY = point.y;
115 event.data.mouse.deltaX = [nsEvent deltaX];
116 event.data.mouse.deltaY = [nsEvent deltaY];
117 event.data.mouse.deltaZ = [nsEvent deltaZ];
118
119 return sendEvent(&event);
120}

147void WebNetscapePluginEventHandlerCocoa::syntheticKeyDownWithCommandModifier(int keyCode, char character)
148{
149 char nullTerminatedString[] = { character, '\0' };
150
151 RetainPtr<NSString> characters = adoptNS([[NSString alloc] initWithUTF8String:nullTerminatedString]);
152
153 NPCocoaEvent event;
154 initializeEvent(&event, NPCocoaEventKeyDown);
155 event.data.key.modifierFlags = NSEventModifierFlagCommand;
156 event.data.key.keyCode = keyCode;
157 event.data.key.isARepeat = false;
158 event.data.key.characters = (NPNSString *)characters.get();
159 event.data.key.charactersIgnoringModifiers = (NPNSString *)characters.get();
160
161 sendEvent(&event);
162}

164bool WebNetscapePluginEventHandlerCocoa::sendKeyEvent(NSEvent* nsEvent, NPCocoaEventType type)
165{
166 NPCocoaEvent event;
167
168 initializeEvent(&event, type);
169 event.data.key.modifierFlags = [nsEvent modifierFlags];
170 event.data.key.keyCode = [nsEvent keyCode];
171 event.data.key.isARepeat = [nsEvent isARepeat];
172 event.data.key.characters = (NPNSString *)[nsEvent characters];
173 event.data.key.charactersIgnoringModifiers = (NPNSString *)[nsEvent charactersIgnoringModifiers];
174
175 return sendEvent(&event);

203bool WebNetscapePluginEventHandlerCocoa::sendEvent(NPCocoaEvent* event)
204{
205 switch (event->type) {
206 case NPCocoaEventMouseDown:
207 case NPCocoaEventMouseUp:
208 case NPCocoaEventMouseDragged:
209 case NPCocoaEventKeyDown:
210 case NPCocoaEventKeyUp:
211 case NPCocoaEventFlagsChanged:
212 case NPCocoaEventScrollWheel:
213 m_currentEventIsUserGesture = true;
214 break;
215 default:
216 m_currentEventIsUserGesture = false;
217 }
218
219 bool result = [m_pluginView sendEvent:event isDrawRect:event->type == NPCocoaEventDrawRect];
220
221 m_currentEventIsUserGesture = false;
222 return result;
223}
