67- (BOOL)_initWithPath:(NSString *)pluginPath
68{
69 resourceRef = -1;
70
71 OSType type = 0;
72
73 if (!cfBundle)
74 return NO;
75
76 CFBundleGetPackageInfo(cfBundle.get(), &type, NULL);
77
78 if (type != FOUR_CHAR_CODE('BRPL'))
79 return NO;
80
81#if USE(PLUGIN_HOST_PROCESS)
82 auto archs = adoptCF(CFBundleCopyExecutableArchitectures(cfBundle.get()));
83 if ([(__bridge NSArray *)archs.get() containsObject:@(NSBundleExecutableArchitectureX86_64)])
84 pluginHostArchitecture = CPU_TYPE_X86_64;
85 else if ([(__bridge NSArray *)archs.get() containsObject:@(NSBundleExecutableArchitectureI386)])
86 pluginHostArchitecture = CPU_TYPE_X86;
87 else
88 return NO;
89#else
90 RetainPtr<CFURLRef> executableURL = adoptCF(CFBundleCopyExecutableURL(cfBundle.get()));
91 if (!executableURL)
92 return NO;
93 NSFileHandle *executableFile = [NSFileHandle fileHandleForReadingAtPath:[(NSURL *)executableURL.get() path]];
94 NSData *data = [executableFile readDataOfLength:512];
95 [executableFile closeFile];
96
97 if (![self isNativeLibraryData:data])
98 return NO;
99#endif
100
101 if (![self getPluginInfoFromPLists])
102 return NO;
103
104 return YES;
105}

107- (id)initWithPath:(NSString *)pluginPath
108{
109 if (!(self = [super initWithPath:pluginPath]))
110 return nil;
111
112 // Initializing a plugin package can cause it to be loaded. If there was an error initializing the plugin package,
113 // ensure that it is unloaded before deallocating it (WebBasePluginPackage requires & asserts this).
114 if (![self _initWithPath:pluginPath]) {
115 [self _unloadWithShutdown:YES];
116 [self release];
117 return nil;
118 }
119
120 return self;
121}

142- (BOOL)_tryLoad
143{
144 NP_GetEntryPointsFuncPtr NP_GetEntryPoints = NULL;
145 NP_InitializeFuncPtr NP_Initialize = NULL;
146 NPError npErr;
147
148#if !LOG_DISABLED
149 CFAbsoluteTime start = CFAbsoluteTimeGetCurrent();
150 CFAbsoluteTime currentTime;
151 CFAbsoluteTime duration;
152#endif
153 LOG(Plugins, "%f Load timing started for: %@", start, (NSString *)[self pluginInfo].name);
154
155 if (isLoaded)
156 return YES;
157
158 if (!CFBundleLoadExecutable(cfBundle.get()))
159 return NO;
160#if !LOG_DISABLED
161 currentTime = CFAbsoluteTimeGetCurrent();
162 duration = currentTime - start;
163#endif
164 LOG(Plugins, "%f CFBundleLoadExecutable took %f seconds", currentTime, duration);
165 isLoaded = YES;
166
167 NP_Initialize = (NP_InitializeFuncPtr)CFBundleGetFunctionPointerForName(cfBundle.get(), CFSTR("NP_Initialize"));
168 NP_GetEntryPoints = (NP_GetEntryPointsFuncPtr)CFBundleGetFunctionPointerForName(cfBundle.get(), CFSTR("NP_GetEntryPoints"));
169 NP_Shutdown = (NPP_ShutdownProcPtr)CFBundleGetFunctionPointerForName(cfBundle.get(), CFSTR("NP_Shutdown"));
170 if (!NP_Initialize || !NP_GetEntryPoints || !NP_Shutdown)
171 return NO;
172
173 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
174 // Plugins (at least QT) require that you call UseResFile on the resource file before loading it.
175 resourceRef = [self openResourceFile];
176 if (resourceRef != -1) {
177 UseResFile(resourceRef);
178 }
179 ALLOW_DEPRECATED_DECLARATIONS_END

296- (void)open
297{
298 instanceCount++;
299
300 // Handle the case where all instances close a plug-in package, but another
301 // instance opens the package before it is unloaded (which only happens when
302 // the plug-in database is refreshed)
303 needsUnload = NO;
304
305 if (!isLoaded) {
306 // Should load when the first instance opens the plug-in package
307 ASSERT(instanceCount == 1);
308 [self load];
309 }
310}
311
312- (void)close
313{
314 ASSERT(instanceCount > 0);
315 instanceCount--;
316 if (instanceCount == 0 && needsUnload)
317 [self _unloadWithShutdown:YES];
318}

321- (BOOL)supportsSnapshotting
322{
323 if ([self bundleIdentifier] != "com.macromedia.Flash Player.plugin")
324 return YES;
325
326 // Flash has a bogus Info.plist entry for CFBundleVersionString, so use CFBundleShortVersionString.
327 NSString *versionString = (NSString *)CFDictionaryGetValue(CFBundleGetInfoDictionary(cfBundle.get()), CFSTR("CFBundleShortVersionString"));
328
329 if (![versionString hasPrefix:@"10.1"])
330 return YES;
331
332 // Some prerelease versions of Flash 10.1 crash when sent a drawRect event using the CA drawing model: <rdar://problem/7739922>
333 return CFStringCompare((CFStringRef)versionString, CFSTR("10.1.53.60"), kCFCompareNumerically) != kCFCompareLessThan;
334}

340- (void)_unloadWithShutdown:(BOOL)shutdown
341{
342 if (!isLoaded)
343 return;
344
345 LOG(Plugins, "Unloading %@...", (NSString *)pluginInfo.name);
346
347 // Cannot unload a plug-in package while an instance is still using it
348 if (instanceCount > 0) {
349 needsUnload = YES;
350 return;
351 }
352
353 if (shutdown && NP_Shutdown)
354 NP_Shutdown();
355
356 if (resourceRef != -1)
357 [self closeResourceFile:resourceRef];
358
359 LOG(Plugins, "Plugin Unloaded");
360 isLoaded = NO;
361}
