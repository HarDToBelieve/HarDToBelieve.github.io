29#import <Foundation/Foundation.h>
30
31#if !TARGET_OS_IPHONE
32#import <AppKit/AppKit.h>
33@class WebFrame;
34#endif
35
36/*!
37 This informal protocol enables a plug-in to request that its containing application
38 perform certain operations.
39*/
40
41@interface NSObject (WebPlugInContainer)
42
43/*!
44 @method webPlugInContainerLoadRequest:inFrame:
45 @abstract Tell the application to show a URL in a target frame
46 @param request The request to be loaded.
47 @param target The target frame. If the frame with the specified target is not
48 found, a new window is opened and the main frame of the new window is named
49 with the specified target. If nil is specified, the frame that contains
50 the applet is targeted.
51*/
52- (void)webPlugInContainerLoadRequest:(NSURLRequest *)request inFrame:(NSString *)target;
53
54/*!
55 @method webPlugInContainerShowStatus:
56 @abstract Tell the application to show the specified status message.
57 @param message The string to be shown.
58*/
59- (void)webPlugInContainerShowStatus:(NSString *)message;
60
61#if !TARGET_OS_IPHONE
62/*!
63 @property webPlugInContainerSelectionColor
64 @abstract The color that should be used for any special drawing when
65 plug-in is selected.
66*/
67@property (nonatomic, readonly, strong) NSColor *webPlugInContainerSelectionColor;
68#endif
69
70/*!
71 @property webFrame
72 @abstract Allows the plug-in to access the WebFrame that
73 contains the plug-in. This method will not be implemented by containers that
74 are not WebKit based.
75*/
76@property (nonatomic, readonly, strong) WebFrame *webFrame;
77
78@end
