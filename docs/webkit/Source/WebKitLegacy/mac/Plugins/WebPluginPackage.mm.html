41@implementation WebPluginPackage
42
43- (id)initWithPath:(NSString *)pluginPath
44{
45 if (!(self = [super initWithPath:pluginPath]))
46 return nil;
47
48 nsBundle = [[NSBundle alloc] initWithPath:path];
49
50 if (!nsBundle) {
51 [self release];
52 return nil;
53 }
54
55 if (![[pluginPath pathExtension] _webkit_isCaseInsensitiveEqualToString:@"webplugin"]) {
56 UInt32 type = 0;
57 CFBundleGetPackageInfo(cfBundle.get(), &type, NULL);
58 if (type != FOUR_CHAR_CODE('WBPL')) {
59 [self release];
60 return nil;
61 }
62 }
63
64#if !PLATFORM(IOS_FAMILY)
65 NSFileHandle *executableFile = [NSFileHandle fileHandleForReadingAtPath:[nsBundle executablePath]];
66 NSData *data = [executableFile readDataOfLength:512];
67 [executableFile closeFile];
68 if (![self isNativeLibraryData:data]) {
69 [self release];
70 return nil;
71 }
72#endif
73
74 if (![self getPluginInfoFromPLists]) {
75 [self release];
76 return nil;
77 }
78
79 return self;
80}
81
82- (void)dealloc
83{
84 [nsBundle release];
85
86 [super dealloc];
87}
88
89- (Class)viewFactory
90{
91 return [nsBundle principalClass];
92}
93
94- (BOOL)load
95{
96#if !LOG_DISABLED
97 CFAbsoluteTime start = CFAbsoluteTimeGetCurrent();
98#endif
99
100 // Load the bundle
101 if (![nsBundle isLoaded]) {
102 if (![nsBundle load])
103 return NO;
104 }
105
106#if !LOG_DISABLED
107 CFAbsoluteTime duration = CFAbsoluteTimeGetCurrent() - start;
108 LOG(Plugins, "principalClass took %f seconds for: %@", duration, (NSString *)[self pluginInfo].name);
109#endif
110 return [super load];
111}
112
113- (NSBundle *)bundle
114{
115 return nsBundle;
116}
117
118@end
