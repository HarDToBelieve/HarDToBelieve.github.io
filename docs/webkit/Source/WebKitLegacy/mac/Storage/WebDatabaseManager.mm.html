72- (id)init
73{
74 if (!(self = [super init]))
75 return nil;
76
77 WebPlatformStrategies::initializeIfNecessary();
78
79 DatabaseManager& dbManager = DatabaseManager::singleton();
80
81 // Set the database root path in WebCore
82 dbManager.initialize(databasesDirectoryPath());
83
84 // Set the DatabaseManagerClient
85 dbManager.setClient(WebDatabaseManagerClient::sharedWebDatabaseManagerClient());
86
87 return self;
88}

90- (NSArray *)origins
91{
92 return createNSArray(DatabaseTracker::singleton().origins(), [] (auto& origin) {
93 return adoptNS([[WebSecurityOrigin alloc] _initWithWebCoreSecurityOrigin:origin.securityOrigin().ptr()]);
94 }).autorelease();
95}

97- (NSArray *)databasesWithOrigin:(WebSecurityOrigin *)origin
98{
99 if (!origin)
100 return nil;
101 return createNSArray(DatabaseTracker::singleton().databaseNames([origin _core]->data())).autorelease();
102}
103
104- (NSDictionary *)detailsForDatabase:(NSString *)databaseIdentifier withOrigin:(WebSecurityOrigin *)origin
105{
106 if (!origin)
107 return nil;
108
109 auto details = DatabaseManager::singleton().detailsForNameAndOrigin(databaseIdentifier, *[origin _core]);
110 if (details.name().isNull())
111 return nil;
112
113 return @{
114 WebDatabaseDisplayNameKey: details.displayName().isEmpty() ? databaseIdentifier : (NSString *)details.displayName(),
115 WebDatabaseExpectedSizeKey: @(details.expectedUsage()),
116 WebDatabaseUsageKey: @(details.currentUsage()),
117 };
118}

148static bool isFileHidden(NSString *file)
149{
150 ASSERT([file length]);
151 return [file characterAtIndex:0] == '.';
152}
153
154+ (void)removeEmptyDatabaseFiles
155{
156 NSString *databasesDirectory = databasesDirectoryPath();
157 NSFileManager *fileManager = [NSFileManager defaultManager];
158 NSArray *array = [fileManager contentsOfDirectoryAtPath:databasesDirectory error:0];
159 if (!array)
160 return;
161
162 NSUInteger count = [array count];
163 for (NSUInteger i = 0; i < count; ++i) {
164 NSString *fileName = [array objectAtIndex:i];
165 // Skip hidden files.
166 if (![fileName length] || isFileHidden(fileName))
167 continue;
168
169 NSString *path = [databasesDirectory stringByAppendingPathComponent:fileName];
170 // Look for directories that contain database files belonging to the same origins.
171 BOOL isDirectory;
172 if (![fileManager fileExistsAtPath:path isDirectory:&isDirectory] || !isDirectory)
173 continue;
174
175 // Make sure the directory is not a symbolic link that points to something else.
176 NSDictionary *attributes = [fileManager attributesOfItemAtPath:path error:0];
177 if ([attributes fileType] == NSFileTypeSymbolicLink)
178 continue;
179
180 NSArray *databaseFilesInOrigin = [fileManager contentsOfDirectoryAtPath:path error:0];
181 NSUInteger databaseFileCount = [databaseFilesInOrigin count];
182 NSUInteger deletedDatabaseFileCount = 0;
183 for (NSUInteger j = 0; j < databaseFileCount; ++j) {
184 NSString *dbFileName = [databaseFilesInOrigin objectAtIndex:j];
185 // Skip hidden files.
186 if (![dbFileName length] || isFileHidden(dbFileName))
187 continue;
188
189 NSString *dbFilePath = [path stringByAppendingPathComponent:dbFileName];
190
191 // There shouldn't be any directories in this folder - but check for it anyway.
192 if (![fileManager fileExistsAtPath:dbFilePath isDirectory:&isDirectory] || isDirectory)
193 continue;
194
195 if (DatabaseTracker::deleteDatabaseFileIfEmpty(dbFilePath))
196 ++deletedDatabaseFileCount;
197 }
198
199 // If we have removed every database file for this origin, delete the folder for this origin.
200 if (databaseFileCount == deletedDatabaseFileCount || ![fileManager contentsOfDirectoryAtPath:path error:nullptr].count) {
201 // Use rmdir - we don't want the deletion to happen if the folder is not empty.
202 rmdir([path fileSystemRepresentation]);
203 }
204 }
205}

207+ (void)scheduleEmptyDatabaseRemoval
208{
209 DatabaseTracker::emptyDatabaseFilesRemovalTaskWillBeScheduled();
210
211 dispatch_async(dispatch_get_global_queue(0, 0), ^{
212 [WebDatabaseManager removeEmptyDatabaseFiles];
213 DatabaseTracker::emptyDatabaseFilesRemovalTaskDidFinish();
214 });
215}

221static NSString *databasesDirectoryPath()
222{
223 NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
224 NSString *databasesDirectory = [defaults objectForKey:WebDatabaseDirectoryDefaultsKey];
225 if (!databasesDirectory || ![databasesDirectory isKindOfClass:[NSString class]])
226 databasesDirectory = @"~/Library/WebKit/Databases";
227
228 return [databasesDirectory stringByStandardizingPath];
229}
