97+ (NSString *)_storageDirectoryPath
98{
99 static NeverDestroyed<RetainPtr<NSString>> sLocalStoragePath;
100 static dispatch_once_t flag;
101 dispatch_once(&flag, ^{
102 NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
103 RetainPtr<NSString> localStoragePath = [defaults objectForKey:WebStorageDirectoryDefaultsKey];
104 if (!localStoragePath || ![localStoragePath isKindOfClass:[NSString class]]) {
105 NSArray *paths = NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSUserDomainMask, YES);
106 NSString *libraryDirectory = [paths objectAtIndex:0];
107 localStoragePath = [libraryDirectory stringByAppendingPathComponent:@"WebKit/LocalStorage"];
108 }
109 sLocalStoragePath.get() = [localStoragePath stringByStandardizingPath];
110 });
111 return sLocalStoragePath.get().get();
112}

124void WebKitInitializeStorageIfNecessary()
125{
126 static BOOL initialized = NO;
127 if (initialized)
128 return;
129
130 auto *storagePath = [WebStorageManager _storageDirectoryPath];
131 WebKit::StorageTracker::initializeTracker(storagePath, WebStorageTrackerClient::sharedWebStorageTrackerClient());
132
133#if PLATFORM(IOS_FAMILY)
134 [[NSURL fileURLWithPath:storagePath] setResourceValue:@YES forKey:NSURLIsExcludedFromBackupKey error:nil];
135#endif
136
137 initialized = YES;
138}
