35@class WebView;

37class CorrectionPanel {
38 WTF_MAKE_NONCOPYABLE(CorrectionPanel);
39public:
40 CorrectionPanel();
41 ~CorrectionPanel();

43 String dismiss(WebCore::ReasonForDismissingAlternativeText);

46private:

48 String dismissInternal(WebCore::ReasonForDismissingAlternativeText, bool dismissingExternally);

51 bool m_wasDismissedExternally;
52 WebCore::ReasonForDismissingAlternativeText m_reasonForDismissing;
53 RetainPtr<WebView> m_view;
54 RetainPtr<NSString> m_resultForDismissal;
55};
