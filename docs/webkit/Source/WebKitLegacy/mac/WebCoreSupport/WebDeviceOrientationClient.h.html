27#import <WebCore/DeviceOrientationClient.h>

29namespace WebCore {
30class DeviceOrientationController;
31}

33@class WebView;
34
35// This class is the Mac implementation of DeviceOrientationClient. It is
36// passed to the Page constructor by the WebView. It is a simple proxy to
37// either the real or mock client which is passed to the WebView. It is
38// required because the WebView must pass a client to the Page constructor,
39// but the real or mock client can not be specified until after the Page has
40// been constructed.
41class WebDeviceOrientationClient : public WebCore::DeviceOrientationClient {
42public:
43 WebDeviceOrientationClient(WebView*);
44
45 // DeviceOrientationClient methods
46 void setController(WebCore::DeviceOrientationController*) override;
47 void startUpdating() override;
48 void stopUpdating() override;
49 WebCore::DeviceOrientationData* lastOrientation() const override;
50 void deviceOrientationControllerDestroyed() override;
51
52private:
53 id<WebDeviceOrientationProvider> getProvider() const;
54
55 WebView* m_webView;
56 WebCore::DeviceOrientationController* m_controller;
57 mutable id<WebDeviceOrientationProvider> m_provider;
58};
