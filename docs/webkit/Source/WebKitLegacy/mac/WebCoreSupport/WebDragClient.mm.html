72static OptionSet<WebCore::DragSourceAction> coreDragSourceActionMask(WebDragSourceAction action)
73{
74 OptionSet<WebCore::DragSourceAction> result;
75
76 if (action & WebDragSourceActionDHTML)
77 result.add(WebCore::DragSourceAction::DHTML);
78 if (action & WebDragSourceActionImage)
79 result.add(WebCore::DragSourceAction::Image);
80 if (action & WebDragSourceActionLink)
81 result.add(WebCore::DragSourceAction::Link);
82 if (action & WebDragSourceActionSelection)
83 result.add(WebCore::DragSourceAction::Selection);
84
85 return result;
86}
87
88static WebDragDestinationAction kit(WebCore::DragDestinationAction action)
89{
90 switch (action) {
91 case WebCore::DragDestinationAction::DHTML:
92 return WebDragDestinationActionDHTML;
93 case WebCore::DragDestinationAction::Edit:
94 return WebDragDestinationActionEdit;
95 case WebCore::DragDestinationAction::Load:
96 return WebDragDestinationActionLoad;
97 }
98 ASSERT_NOT_REACHED();
99 return WebDragDestinationActionNone;
100}

111static WebHTMLView *getTopHTMLView(Frame* frame)
112{
113 ASSERT(frame);
114 ASSERT(frame->page());

135void WebDragClient::startDrag(DragItem dragItem, DataTransfer& dataTransfer, Frame& frame)
136{
137 auto& dragImage = dragItem.image;
138 auto dragLocationInContentCoordinates = dragItem.dragLocationInContentCoordinates;
139
140 RetainPtr<WebHTMLView> htmlView = (WebHTMLView*)[[kit(&frame) frameView] documentView];
141 if (![htmlView.get() isKindOfClass:[WebHTMLView class]])
142 return;
143
144 NSEvent *event = (dragItem.sourceAction && *dragItem.sourceAction == DragSourceAction::Link) ? frame.eventHandler().currentNSEvent() : [htmlView.get() _mouseDownEvent];
145 WebHTMLView* topHTMLView = getTopHTMLView(&frame);
146 RetainPtr<WebHTMLView> topViewProtector = topHTMLView;
147
148 [topHTMLView _stopAutoscrollTimer];
149 NSPasteboard *pasteboard = [NSPasteboard pasteboardWithName:dataTransfer.pasteboard().name()];
150
151 NSImage *dragNSImage = dragImage.get().get();
152 WebHTMLView *sourceHTMLView = htmlView.get();
153
154 IntSize size([dragNSImage size]);
155 size.scale(1 / frame.page()->deviceScaleFactor());
156 [dragNSImage setSize:size];
157
158 id delegate = [m_webView UIDelegate];
159 SEL selector = @selector(webView:dragImage:at:offset:event:pasteboard:source:slideBack:forView:);
160 if ([delegate respondsToSelector:selector]) {
161 @try {
162 [delegate webView:m_webView dragImage:dragNSImage at:dragLocationInContentCoordinates offset:NSZeroSize event:event pasteboard:pasteboard source:sourceHTMLView slideBack:YES forView:topHTMLView];
163 } @catch (id exception) {
164 ReportDiscardedDelegateException(selector, exception);
165 }
166 } else
167 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
168 [topHTMLView dragImage:dragNSImage at:dragLocationInContentCoordinates offset:NSZeroSize event:event pasteboard:pasteboard source:sourceHTMLView slideBack:YES];
169 ALLOW_DEPRECATED_DECLARATIONS_END
170}

172void WebDragClient::beginDrag(DragItem dragItem, Frame& frame, const IntPoint& mouseDownPosition, const IntPoint& mouseDraggedPosition, DataTransfer& dataTransfer, DragSourceAction dragSourceAction)
173{
174 ASSERT(!dataTransfer.pasteboard().hasData());
175
176 RetainPtr<WebHTMLView> topWebHTMLView = dynamic_objc_cast<WebHTMLView>(m_webView.mainFrame.frameView.documentView);
177 ASSERT(topWebHTMLView);
178
179 [topWebHTMLView _stopAutoscrollTimer];
180
181 auto draggingItem = adoptNS([[NSDraggingItem alloc] initWithPasteboardWriter:createPasteboardWriter(dragItem.data).get()]);
182
183 auto dragImageSize = IntSize { [dragItem.image.get() size] };
184
185 dragImageSize.scale(1 / frame.page()->deviceScaleFactor());
186 [dragItem.image.get() setSize:dragImageSize];
187
188 NSRect draggingFrame = NSMakeRect(mouseDraggedPosition.x() - dragImageSize.width() * dragItem.imageAnchorPoint.x(), mouseDraggedPosition.y() - dragImageSize.height() * dragItem.imageAnchorPoint.y(), dragImageSize.width(), dragImageSize.height());
189 [draggingItem setDraggingFrame:draggingFrame contents:dragItem.image.get().get()];
190
191 // FIXME: We should be able to make a fake event with the mosue dragged coordinates.
192 NSEvent *event = frame.eventHandler().currentNSEvent();
193 [topWebHTMLView.get() beginDraggingSessionWithItems:@[ draggingItem.get() ] event:event source:topWebHTMLView.get()];
194}

197{
198 ASSERT(pasteboardName);
199 [[NSPasteboard pasteboardWithName:pasteboardName] _web_declareAndWriteDragImageForElement:kit(&element) URL:url title:title archive:[kit(&element) webArchive] source:getTopHTMLView(frame)];
200}
