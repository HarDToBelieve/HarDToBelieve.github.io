219inline NSSelectionAffinity kit(WebCore::Affinity affinity)
220{
221 switch (affinity) {
222 case WebCore::Affinity::Upstream:
223 return NSSelectionAffinityUpstream;
224 case WebCore::Affinity::Downstream:
225 return NSSelectionAffinityDownstream;
226 }
227 ASSERT_NOT_REACHED();
228 return NSSelectionAffinityDownstream;
229}
230
231inline WebCore::Affinity core(NSSelectionAffinity affinity)
232{
233 switch (affinity) {
234 case NSSelectionAffinityUpstream:
235 return WebCore::Affinity::Upstream;
236 case NSSelectionAffinityDownstream:
237 return WebCore::Affinity::Downstream;
238 }
239 ASSERT_NOT_REACHED();
240 return WebCore::Affinity::Downstream;
241}
