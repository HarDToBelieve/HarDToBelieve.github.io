108FrontendChannel* WebInspectorClient::openLocalFrontend(InspectorController* inspectedPageController)
109{
110 RetainPtr<WebInspectorWindowController> windowController = adoptNS([[WebInspectorWindowController alloc] initWithInspectedWebView:m_inspectedWebView isUnderTest:inspectedPageController->isUnderTest()]);
111 [windowController.get() setInspectorClient:this];
112
113 m_frontendPage = core([windowController.get() frontendWebView]);
114 m_frontendClient = makeUnique<WebInspectorFrontendClient>(m_inspectedWebView, windowController.get(), inspectedPageController, m_frontendPage, createFrontendSettings());
115
116 RetainPtr<WebInspectorFrontend> webInspectorFrontend = adoptNS([[WebInspectorFrontend alloc] initWithFrontendClient:m_frontendClient.get()]);
117 [[m_inspectedWebView inspector] setFrontend:webInspectorFrontend.get()];
118
119 m_frontendPage->inspectorController().setInspectorFrontendClient(m_frontendClient.get());
120
121 return this;
122}

200void WebInspectorFrontendClient::frontendLoaded()
201{
202 [m_frontendWindowController.get() showWindow:nil];
203 if ([m_frontendWindowController.get() attached])
204 restoreAttachedWindowHeight();
205
206 InspectorFrontendClientLocal::frontendLoaded();
207
208 WebFrame *frame = [m_inspectedWebView mainFrame];
209
210 WebFrameLoadDelegateImplementationCache* implementations = WebViewGetFrameLoadDelegateImplementations(m_inspectedWebView);
211 if (implementations->didClearInspectorWindowObjectForFrameFunc)
212 CallFrameLoadDelegate(implementations->didClearInspectorWindowObjectForFrameFunc, m_inspectedWebView,
213 @selector(webView:didClearInspectorWindowObject:forFrame:), [frame windowObject], frame);
214
215 bool attached = [m_frontendWindowController.get() attached];
216 setAttachedWindow(attached ? DockSide::Bottom : DockSide::Undocked);
217}

342void WebInspectorFrontendClient::showCertificate(const CertificateInfo& certificateInfo)
343{
344 ASSERT(!certificateInfo.isEmpty());
345
346 RetainPtr<SFCertificatePanel> certificatePanel = adoptNS([[SFCertificatePanel alloc] init]);
347
348 NSWindow *window = [[m_frontendWindowController frontendWebView] window];
349 if (!window)
350 window = [NSApp keyWindow];
351
352#if HAVE(SEC_TRUST_SERIALIZATION)
353 [certificatePanel beginSheetForWindow:window modalDelegate:nil didEndSelector:NULL contextInfo:nullptr trust:certificateInfo.trust() showGroup:YES];
354#else
355 [certificatePanel beginSheetForWindow:window modalDelegate:nil didEndSelector:NULL contextInfo:nullptr certificates:(NSArray *)certificateInfo.certificateChain() showGroup:YES];
356#endif
357
358 // This must be called after the trust panel has been displayed, because the certificateView doesn't exist beforehand.
359 SFCertificateView *certificateView = [certificatePanel certificateView];
360 [certificateView setDisplayTrust:YES];
361 [certificateView setEditableTrust:NO];
362 [certificateView setDisplayDetails:YES];
363 [certificateView setDetailsDisclosed:YES];
364}

386void WebInspectorFrontendClient::save(const String& suggestedURL, const String& content, bool base64Encoded, bool forceSaveDialog)
387{
388 ASSERT(!suggestedURL.isEmpty());
389
390 NSURL *platformURL = m_suggestedToActualURLMap.get(suggestedURL).get();
391 if (!platformURL) {
392 platformURL = [NSURL URLWithString:suggestedURL];
393 // The user must confirm new filenames before we can save to them.
394 forceSaveDialog = true;
395 }
396
397 ASSERT(platformURL);
398 if (!platformURL)
399 return;
400
401 // Necessary for the block below.
402 String suggestedURLCopy = suggestedURL;
403 String contentCopy = content;
404
405 auto saveToURL = ^(NSURL *actualURL) {
406 ASSERT(actualURL);
407
408 m_suggestedToActualURLMap.set(suggestedURLCopy, actualURL);
409
410 if (base64Encoded) {
411 auto decodedData = base64Decode(content, Base64DecodeOptions::ValidatePadding);
412 if (!decodedData)
413 return;
414 RetainPtr<NSData> dataContent = adoptNS([[NSData alloc] initWithBytes:decodedData->data() length:decodedData->size()]);
415 [dataContent writeToURL:actualURL atomically:YES];
416 } else
417 [contentCopy writeToURL:actualURL atomically:YES encoding:NSUTF8StringEncoding error:NULL];
418
419 core([m_frontendWindowController frontendWebView])->mainFrame().script().executeScriptIgnoringException([NSString stringWithFormat:@"InspectorFrontendAPI.savedURL(\"%@\")", actualURL.absoluteString]);
420 };
421
422 if (!forceSaveDialog) {
423 saveToURL(platformURL);
424 return;
425 }
426
427 NSSavePanel *panel = [NSSavePanel savePanel];
428 panel.nameFieldStringValue = platformURL.lastPathComponent;
429
430 // If we have a file URL we've already saved this file to a path and
431 // can provide a good directory to show. Otherwise, use the system's
432 // default behavior for the initial directory to show in the dialog.
433 if (platformURL.isFileURL)
434 panel.directoryURL = [platformURL URLByDeletingLastPathComponent];
435
436 auto completionHandler = ^(NSInteger result) {
437 if (result == NSModalResponseCancel)
438 return;
439 ASSERT(result == NSModalResponseOK);
440 saveToURL(panel.URL);
441 };
442
443 NSWindow *frontendWindow = [[m_frontendWindowController frontendWebView] window];
444 NSWindow *window = frontendWindow ? frontendWindow : [NSApp keyWindow];
445 if (window)
446 [panel beginSheetModalForWindow:window completionHandler:completionHandler];
447 else
448 completionHandler([panel runModal]);
449}

451void WebInspectorFrontendClient::append(const String& suggestedURL, const String& content)
452{
453 ASSERT(!suggestedURL.isEmpty());
454
455 RetainPtr<NSURL> actualURL = m_suggestedToActualURLMap.get(suggestedURL);
456 // do not append unless the user has already confirmed this filename in save().
457 if (!actualURL)
458 return;
459
460 NSFileHandle *handle = [NSFileHandle fileHandleForWritingToURL:actualURL.get() error:NULL];
461 [handle seekToEndOfFile];
462 [handle writeData:[content dataUsingEncoding:NSUTF8StringEncoding]];
463 [handle closeFile];
464
465 core([m_frontendWindowController frontendWebView])->mainFrame().script().executeScriptIgnoringException([NSString stringWithFormat:@"InspectorFrontendAPI.appendedToURL(\"%@\")", [actualURL absoluteString]]);
466}

600- (void)close
601{
602 if (!_visible)
603 return;
604
605 _visible = NO;
606
607 if (_attachedToInspectedWebView) {
608 if ([_inspectedWebView.get() _isClosed])
609 return;
610
611 [_frontendWebView removeFromSuperview];
612
613 WebFrameView *frameView = [[_inspectedWebView.get() mainFrame] frameView];
614 NSRect frameViewRect = [frameView frame];
615
616 // Setting the height based on the previous height is done to work with
617 // Safari's find banner. This assumes the previous height is the Y origin.
618 frameViewRect.size.height += NSMinY(frameViewRect);
619 frameViewRect.origin.y = 0.0;
620
621 [frameView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable)];
622 [frameView setFrame:frameViewRect];
623
624 [_inspectedWebView.get() displayIfNeeded];
625 } else
626 [super close];
627}

634- (IBAction)showWindow:(id)sender
635{
636 if (_visible) {
637 if (!_attachedToInspectedWebView)
638 [super showWindow:sender]; // call super so the window will be ordered front if needed
639 return;
640 }
641
642 _visible = YES;
643
644 _shouldAttach = _inspectorClient->inspectorStartsAttached() && _frontendClient->canAttach();
645
646 if (_shouldAttach) {
647 WebFrameView *frameView = [[_inspectedWebView.get() mainFrame] frameView];
648
649 [_frontendWebView removeFromSuperview];
650 [_inspectedWebView.get() addSubview:_frontendWebView.get() positioned:NSWindowBelow relativeTo:(NSView *)frameView];
651 [[_inspectedWebView.get() window] makeFirstResponder:_frontendWebView.get()];
652
653 [_frontendWebView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable | NSViewMaxYMargin)];
654 [frameView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable | NSViewMinYMargin)];
655
656 _attachedToInspectedWebView = YES;
657 } else {
658 _attachedToInspectedWebView = NO;
659
660 NSView *contentView = [[self window] contentView];
661 [_frontendWebView setFrame:[contentView frame]];
662 [_frontendWebView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable)];
663 [_frontendWebView removeFromSuperview];
664 [contentView addSubview:_frontendWebView.get()];
665
666 [super showWindow:nil];
667 }
668}

721 WebFrameView *frameView = [[_inspectedWebView.get() mainFrame] frameView];
722 NSRect frameViewRect = [frameView frame];
723
724 // Setting the height based on the difference is done to work with
725 // Safari's find banner. This assumes the previous height is the Y origin.
726 CGFloat heightDifference = (NSMinY(frameViewRect) - height);
727 frameViewRect.size.height += heightDifference;
728 frameViewRect.origin.y = height;
729
730 [_frontendWebView setFrame:NSMakeRect(0.0, 0.0, NSWidth(frameViewRect), height)];
731 [frameView setFrame:frameViewRect];
732}

740{
741 RetainPtr<WebInspectorWindowController> protect(self);
742
743 if (Page* frontendPage = _frontendClient->frontendPage())
744 frontendPage->inspectorController().setInspectorFrontendClient(nullptr);

759 [_frontendWebView close];

823
824- (void)webView:(WebView *)webView decidePolicyForNavigationAction:(NSDictionary *)actionInformation request:(NSURLRequest *)request frame:(WebFrame *)frame decisionListener:(id<WebPolicyDecisionListener>)listener
825{
826 // Allow non-main frames to navigate anywhere.
827 if (frame != [webView mainFrame]) {
828 [listener use];
829 return;
830 }
831
832 // Allow loading of the main inspector file.
833 if ([[request URL] isFileURL] && [[[request URL] path] isEqualToString:[self inspectorPagePath]]) {
834 [listener use];
835 return;
836 }
837
838 // Allow loading of the test inspector file.
839 NSString *testPagePath = [self inspectorTestPagePath];
840 if (testPagePath && [[request URL] isFileURL] && [[[request URL] path] isEqualToString:testPagePath]) {
841 [listener use];
842 return;
843 }
844
845 // Prevent everything else from loading in the inspector's page.
846 [listener ignore];
847
848 // And instead load it in the inspected page.
849 [[_inspectedWebView.get() mainFrame] loadRequest:request];
850}
