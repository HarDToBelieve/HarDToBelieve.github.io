57Vector<WebCore::PluginInfo> WebPluginInfoProvider::pluginInfo(WebCore::Page& page, std::optional<Vector<SupportedPluginIdentifier>>&)
58{
59 Vector<WebCore::PluginInfo> plugins;
60
61 BEGIN_BLOCK_OBJC_EXCEPTIONS
62
63
64 // WebKit1 has no application plug-ins, so we don't need to add them here.
65 if (!page.mainFrame().arePluginsEnabled())
66 return plugins;
67
68 for (WebPluginPackage *plugin in [WebPluginDatabase sharedDatabase].plugins)
69 plugins.append(plugin.pluginInfo);
70
71 END_BLOCK_OBJC_EXCEPTIONS
72
73 return plugins;
74}
75
76Vector<WebCore::PluginInfo> WebPluginInfoProvider::webVisiblePluginInfo(WebCore::Page& page, const URL&)
77{
78 std::optional<Vector<SupportedPluginIdentifier>> supportedPluginIdentifiers;
79 return pluginInfo(page, supportedPluginIdentifiers);
80}
