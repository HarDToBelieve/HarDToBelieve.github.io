26#import "WebValidationMessageClient.h"
27
28#import "WebView.h"
29#import "WebViewInternal.h"
30#import <WebCore/Element.h>
31
32using namespace WebCore;
33
34WebValidationMessageClient::WebValidationMessageClient(WebView* view)
35 : m_view(view)
36{
37}
38
39WebValidationMessageClient::~WebValidationMessageClient()
40{
41 if (m_currentAnchor)
42 hideValidationMessage(*m_currentAnchor);
43}
44
45void WebValidationMessageClient::documentDetached(Document& document)
46{
47 if (!m_currentAnchor)
48 return;
49 if (&m_currentAnchor->document() == &document)
50 hideValidationMessage(*m_currentAnchor);
51}
52
53void WebValidationMessageClient::showValidationMessage(const Element& anchor, const String& message)
54{
55 if (m_currentAnchor)
56 hideValidationMessage(*m_currentAnchor);
57
58 m_currentAnchor = &anchor;
59 m_currentAnchorRect = anchor.clientRect();
60 [m_view showFormValidationMessage:message withAnchorRect:m_currentAnchorRect];
61}
62
63void WebValidationMessageClient::hideValidationMessage(const Element& anchor)
64{
65 if (!isValidationMessageVisible(anchor))
66 return;
67
68 m_currentAnchor = nullptr;
69 m_currentAnchorRect = { };
70 [m_view hideFormValidationMessage];
71}
72
73void WebValidationMessageClient::hideAnyValidationMessage()
74{
75 if (!m_currentAnchor)
76 return;
77
78 m_currentAnchor = nullptr;
79 m_currentAnchorRect = { };
80 [m_view hideFormValidationMessage];
81}
82
83bool WebValidationMessageClient::isValidationMessageVisible(const Element& anchor)
84{
85 return m_currentAnchor == &anchor;
86}
87
88void WebValidationMessageClient::updateValidationBubbleStateIfNeeded()
89{
90 if (!m_currentAnchor)
91 return;
92
93 // We currently hide the validation bubble if its position is outdated instead of trying
94 // to update its position.
95 if (m_currentAnchorRect != m_currentAnchor->clientRect())
96 hideValidationMessage(*m_currentAnchor);
97}
