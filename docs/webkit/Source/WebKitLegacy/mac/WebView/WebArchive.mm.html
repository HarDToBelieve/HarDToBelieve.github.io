132static BOOL isArrayOfClass(id object, Class elementClass)
133{
134 if (![object isKindOfClass:[NSArray class]])
135 return NO;
136 NSArray *array = (NSArray *)object;
137 NSUInteger count = [array count];
138 for (NSUInteger i = 0; i < count; ++i)
139 if (![[array objectAtIndex:i] isKindOfClass:elementClass])
140 return NO;
141 return YES;
142}

144- (instancetype)initWithMainResource:(WebResource *)mainResource subresources:(NSArray *)subresources subframeArchives:(NSArray *)subframeArchives
145{
146 WebCoreThreadViolationCheckRoundTwo();
147
148 self = [super init];
149 if (!self)
150 return nil;
151
152 _private = [[WebArchivePrivate alloc] init];
153
154 _private->cachedMainResource = mainResource;
155 if (!_private->cachedMainResource) {
156 [self release];
157 return nil;
158 }
159
160 if (!subresources || isArrayOfClass(subresources, [WebResource class]))
161 _private->cachedSubresources = subresources;
162 else {
163 [self release];
164 return nil;
165 }
166
167 if (!subframeArchives || isArrayOfClass(subframeArchives, [WebArchive class]))
168 _private->cachedSubframeArchives = subframeArchives;
169 else {
170 [self release];
171 return nil;
172 }
173
174 Vector<Ref<ArchiveResource>> coreResources;
175 for (WebResource *subresource in subresources)
176 coreResources.append([subresource _coreResource].get());
177
178 Vector<Ref<LegacyWebArchive>> coreArchives;
179 for (WebArchive *subframeArchive in subframeArchives)
180 coreArchives.append(*[subframeArchive->_private coreArchive]);
181
182 [_private setCoreArchive:LegacyWebArchive::create([mainResource _coreResource].get(), WTFMove(coreResources), WTFMove(coreArchives))];
183 return self;
184}

186- (instancetype)initWithData:(NSData *)data
187{
188 WebCoreThreadViolationCheckRoundTwo();
189
190 self = [super init];
191 if (!self)
192 return nil;
193
194#if !LOG_DISABLED
195 CFAbsoluteTime start = CFAbsoluteTimeGetCurrent();
196#endif
197
198 _private = [[WebArchivePrivate alloc] init];
199 auto coreArchive = LegacyWebArchive::create(SharedBuffer::create(data));
200 if (!coreArchive) {
201 [self release];
202 return nil;
203 }
204
205 [_private setCoreArchive:coreArchive.releaseNonNull()];
206
207#if !LOG_DISABLED
208 CFAbsoluteTime end = CFAbsoluteTimeGetCurrent();
209 CFAbsoluteTime duration = end - start;
210#endif
211 LOG(Timing, "Parsing web archive with [NSPropertyListSerialization propertyListFromData::::] took %f seconds", duration);
212
213 return self;
214}

216- (instancetype)initWithCoder:(NSCoder *)decoder
217{
218 WebResource *mainResource = nil;
219 NSArray *subresources = nil;
220 NSArray *subframeArchives = nil;
221
222 @try {
223 id object = [decoder decodeObjectForKey:WebMainResourceKey];
224 if ([object isKindOfClass:[WebResource class]])
225 mainResource = object;
226 object = [decoder decodeObjectForKey:WebSubresourcesKey];
227 if (isArrayOfClass(object, [WebResource class]))
228 subresources = object;
229 object = [decoder decodeObjectForKey:WebSubframeArchivesKey];
230 if (isArrayOfClass(object, [WebArchive class]))
231 subframeArchives = object;
232 } @catch(id) {
233 [self release];
234 return nil;
235 }
236
237 return [self initWithMainResource:mainResource subresources:subresources subframeArchives:subframeArchives];
238}

258- (WebResource *)mainResource
259{
260 WebCoreThreadViolationCheckRoundTwo();
261
262 // Currently from WebKit API perspective, WebArchives are entirely immutable once created
263 // If they ever become mutable, we'll need to rethink this.
264 if (!_private->cachedMainResource) {
265 if (auto* coreArchive = [_private coreArchive]) {
266 if (auto* mainResource = coreArchive->mainResource())
267 _private->cachedMainResource = adoptNS([[WebResource alloc] _initWithCoreResource:*mainResource]);
268 }
269 }
270
271 auto cachedMainResourceCopy = _private->cachedMainResource;
272 return cachedMainResourceCopy.autorelease();
273}
274
275- (NSArray *)subresources
276{
277 WebCoreThreadViolationCheckRoundTwo();
278
279 // Currently from WebKit API perspective, WebArchives are entirely immutable once created
280 // If they ever become mutable, we'll need to rethink this.
281 if (!_private->cachedSubresources) {
282 auto coreArchive = [_private coreArchive];
283 if (!coreArchive)
284 _private->cachedSubresources = adoptNS([[NSArray alloc] init]);
285 else {
286 _private->cachedSubresources = createNSArray(coreArchive->subresources(), [] (auto& subresource) {
287 return adoptNS([[WebResource alloc] _initWithCoreResource:subresource.get()]);
288 });
289 }
290 }
291 // Maintain the WebKit 3 behavior of this API, which is documented and
292 // relied upon by some clients, of returning nil if there are no subresources.
293 if (![_private->cachedSubresources count])
294 return nil;
295
296 auto cachedSubresourcesCopy = _private->cachedSubresources;
297 return cachedSubresourcesCopy.autorelease();
298}
299
300- (NSArray *)subframeArchives
301{
302 WebCoreThreadViolationCheckRoundTwo();
303
304 // Currently from WebKit API perspective, WebArchives are entirely immutable once created
305 // If they ever become mutable, we'll need to rethink this.
306 if (!_private->cachedSubframeArchives) {
307 auto* coreArchive = [_private coreArchive];
308 if (!coreArchive)
309 _private->cachedSubframeArchives = adoptNS([[NSArray alloc] init]);
310 else {
311 _private->cachedSubframeArchives = createNSArray(coreArchive->subframeArchives(), [] (auto& archive) {
312 return adoptNS([[WebArchive alloc] _initWithCoreLegacyWebArchive:static_cast<LegacyWebArchive*>(archive.ptr())]);
313 });
314 }
315 }
316
317 auto cachedSubframeArchivesCopy = _private->cachedSubframeArchives;
318 return cachedSubframeArchivesCopy.autorelease();
319}

321- (NSData *)data
322{
323 WebCoreThreadViolationCheckRoundTwo();
324
325#if !LOG_DISABLED
326 CFAbsoluteTime start = CFAbsoluteTimeGetCurrent();
327#endif
328
329 RetainPtr<CFDataRef> data = [_private coreArchive]->rawDataRepresentation();
330
331#if !LOG_DISABLED
332 CFAbsoluteTime end = CFAbsoluteTimeGetCurrent();
333 CFAbsoluteTime duration = end - start;
334#endif
335 LOG(Timing, "Serializing web archive to raw CFPropertyList data took %f seconds", duration);
336
337 return retainPtr((NSData *)data.get()).autorelease();
338}

345{
346 WebCoreThreadViolationCheckRoundTwo();
347
348 self = [super init];
349 if (!self)
350 return nil;
351
352 _private = [[WebArchivePrivate alloc] initWithCoreArchive:WTFMove(coreLegacyWebArchive)];
353 if (!_private) {
354 [self release];
355 return nil;
356 }
357
358 return self;
