28#import "WebDashboardRegion.h"
29
30#if ENABLE(DASHBOARD_SUPPORT)
31
32@implementation WebDashboardRegion
33
34- (id)initWithRect:(NSRect)r clip:(NSRect)c type:(WebDashboardRegionType)t
35{
36 self = [super init];
37 rect = r;
38 clip = c;
39 type = t;
40 return self;
41}
42
43- (id)copyWithZone:(NSZone *)unusedZone
44{
45 UNUSED_PARAM(unusedZone);
46
47 return [self retain];
48}
49
50- (NSRect)dashboardRegionClip
51{
52 return clip;
53}
54
55- (NSRect)dashboardRegionRect
56{
57 return rect;
58}
59
60- (WebDashboardRegionType)dashboardRegionType
61{
62 return type;
63}
64
65static const char* typeName(WebDashboardRegionType type)
66{
67 switch (type) {
68 case WebDashboardRegionTypeNone:
69 return "None";
70 case WebDashboardRegionTypeCircle:
71 return "Circle";
72 case WebDashboardRegionTypeRectangle:
73 return "Rectangle";
74 case WebDashboardRegionTypeScrollerRectangle:
75 return "ScrollerRectangle";
76 }
77 return "Unknown";
78}
79
80- (NSString *)description
81{
82 return [NSString stringWithFormat:@"rect:%@ clip:%@ type:%s", NSStringFromRect(rect), NSStringFromRect(clip), typeName(type)];
83}
84
85// FIXME: Overriding isEqual: without overriding hash will cause trouble if this ever goes into an NSSet or is the key in an NSDictionary,
86// since two equal objects could have different hashes.
87- (BOOL)isEqual:(id)other
88{
89 if (![other isKindOfClass:[WebDashboardRegion class]])
90 return NO;
91
92 return NSEqualRects(rect, [other dashboardRegionRect]) && NSEqualRects(clip, [other dashboardRegionClip]) && type == [other dashboardRegionType];
93}
94
95@end
96
97#endif
