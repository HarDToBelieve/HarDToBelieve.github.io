81class WebDataSourcePrivate
82{
83public:
84 WebDataSourcePrivate(Ref<WebDocumentLoaderMac>&& loader)
85 : loader(WTFMove(loader))
86 , representationFinishedLoading(NO)
87 , includedInWebKitStatistics(NO)
88#if PLATFORM(IOS_FAMILY)
89 , _dataSourceDelegate(nil)
90#endif
91 {
92 }
93 ~WebDataSourcePrivate()
94 {
95 // We might run in to infinite recursion if we're stopping loading as the result of detaching from the frame.
96 // Therefore, DocumentLoader::detachFromFrame() did some smart things to stop the recursion.
97 // As a result of breaking the resursion, DocumentLoader::m_subresourceLoader
98 // and DocumentLoader::m_plugInStreamLoaders might not be empty at this time.
99 // See <rdar://problem/9673866> for more details.
100 ASSERT(!loader->isLoading() || loader->isStopping());
101 loader->detachDataSource();
102 }
103
104 Ref<WebDocumentLoaderMac> loader;
105 RetainPtr<id<WebDocumentRepresentation> > representation;
106 BOOL representationFinishedLoading;
107 BOOL includedInWebKitStatistics;
108#if PLATFORM(IOS_FAMILY)
109 NSObject<WebDataSourcePrivateDelegate> *_dataSourceDelegate;
110#endif
111#if USE(QUICK_LOOK)
112 RetainPtr<NSDictionary> _quickLookContent;
113 RefPtr<WebCore::LegacyPreviewLoaderClient> _quickLookPreviewLoaderClient;
114#endif
115};

133void addTypesFromClass(NSMutableDictionary *allTypes, Class objCClass, NSArray *supportTypes)
134{
135 NSEnumerator *enumerator = [supportTypes objectEnumerator];
136 ASSERT(enumerator != nil);
137 NSString *mime = nil;
138 while ((mime = [enumerator nextObject]) != nil) {
139 // Don't clobber previously-registered classes.
140 if ([allTypes objectForKey:mime] == nil)
141 [allTypes setObject:objCClass forKey:mime];
142 }
143}

271+ (NSMutableDictionary *)_repTypesAllowImageTypeOmission:(BOOL)allowImageTypeOmission
272{
273 static auto repTypes = makeNeverDestroyed([] {
274 auto types = adoptNS([[NSMutableDictionary alloc] init]);
275 addTypesFromClass(types.get(), [WebHTMLRepresentation class], [WebHTMLRepresentation supportedNonImageMIMETypes]);
276 addTypesFromClass(types.get(), [WebHTMLRepresentation class], [WebHTMLRepresentation supportedMediaMIMETypes]);
277
278 // Since this is a "secret default" we don't both registering it.
279 BOOL omitPDFSupport = [[NSUserDefaults standardUserDefaults] boolForKey:@"WebKitOmitPDFSupport"];
280 if (!omitPDFSupport) {
281#if PLATFORM(IOS_FAMILY)
282#define WebPDFRepresentation ([WebView _getPDFRepresentationClass])
283#endif
284 addTypesFromClass(types.get(), [WebPDFRepresentation class], [WebPDFRepresentation supportedMIMETypes]);
285#if PLATFORM(IOS_FAMILY)
286#undef WebPDFRepresentation
287#endif
288 }
289 return types;
290 }());
291 static BOOL addedImageTypes = NO;
292 if (!addedImageTypes && !allowImageTypeOmission) {
293 addTypesFromClass(repTypes.get().get(), [WebHTMLRepresentation class], [WebHTMLRepresentation supportedImageMIMETypes]);
294 addedImageTypes = YES;
295 }
296
297 return repTypes.get().get();
298}

307// FIXME: There are few reasons why this method and many of its related methods can't be pushed entirely into WebCore in the future.
308- (DOMDocumentFragment *)_documentFragmentWithArchive:(WebArchive *)archive
309{
310 ASSERT(archive);
311 WebResource *mainResource = [archive mainResource];
312 if (mainResource) {
313 NSString *MIMEType = [mainResource MIMEType];
314 if ([WebView canShowMIMETypeAsHTML:MIMEType]) {
315 auto markupString = adoptNS([[NSString alloc] initWithData:[mainResource data] encoding:NSUTF8StringEncoding]);
316
317 // FIXME: seems poor form to do this as a side effect of getting a document fragment
318 toPrivate(_private)->loader->addAllArchiveResources(*[archive _coreLegacyWebArchive]);
319
320 return [[self webFrame] _documentFragmentWithMarkupString:markupString.get() baseURLString:[[mainResource URL] _web_originalDataAsString]];
321 } else if (WebCore::MIMETypeRegistry::isSupportedImageMIMEType(MIMEType))
322 return [self _documentFragmentWithImageResource:mainResource];
323 }
324 return nil;
325}

327- (DOMDocumentFragment *)_documentFragmentWithImageResource:(WebResource *)resource
328{
329 DOMElement *imageElement = [self _imageElementWithImageResource:resource];
330 if (!imageElement)
331 return nil;
332 DOMDocumentFragment *fragment = [[[self webFrame] DOMDocument] createDocumentFragment];
333 [fragment appendChild:imageElement];
334 return fragment;
335}
336
337- (DOMElement *)_imageElementWithImageResource:(WebResource *)resource
338{
339 if (!resource)
340 return 0;
341
342 [self addSubresource:resource];
343
344 DOMElement *imageElement = [[[self webFrame] DOMDocument] createElement:@"img"];
345
346 // FIXME: calling _web_originalDataAsString on a file URL returns an absolute path. Workaround this.
347 NSURL *URL = [resource URL];
348 [imageElement setAttribute:@"src" value:[URL isFileURL] ? [URL absoluteString] : [URL _web_originalDataAsString]];
349
350 return imageElement;
351}

373- (void)_makeRepresentation
374{
375 Class repClass = [[self class] _representationClassForMIMEType:[self _responseMIMEType] allowingPlugins:[[[self _webView] preferences] arePlugInsEnabled]];
376
377#if PLATFORM(IOS_FAMILY)
378 if ([repClass respondsToSelector:@selector(_representationClassForWebFrame:)])
379 repClass = [repClass performSelector:@selector(_representationClassForWebFrame:) withObject:[self webFrame]];
380#endif
381
382 // Check if the data source was already bound?
383 if (![[self representation] isKindOfClass:repClass]) {
384 RetainPtr<id> newRep = repClass ? adoptNS([[repClass alloc] init]) : nil;
385 [self _setRepresentation:(id <WebDocumentRepresentation>)newRep.get()];
386 }
387
388 id<WebDocumentRepresentation> representation = toPrivate(_private)->representation.get();
389 [representation setDataSource:self];
390#if PLATFORM(IOS_FAMILY)
391 toPrivate(_private)->loader->setResponseMIMEType([self _responseMIMEType]);
392#endif
393}

400- (id)_initWithDocumentLoader:(Ref<WebDocumentLoaderMac>&&)loader
401{
402 self = [super init];
403 if (!self)
404 return nil;
405
406 _private = static_cast<void*>(new WebDataSourcePrivate(WTFMove(loader)));
407
408 LOG(Loading, "creating datasource for %@", static_cast<NSURL *>(toPrivate(_private)->loader->request().url()));
409
410 if ((toPrivate(_private)->includedInWebKitStatistics = [[self webFrame] _isIncludedInWebKitStatistics]))
411 ++WebDataSourceCount;
412
413 return self;
414}

442- (void)dealloc
443{
444 if (WebCoreObjCScheduleDeallocateOnMainThread([WebDataSource class], self))
445 return;
446
447 if (toPrivate(_private) && toPrivate(_private)->includedInWebKitStatistics)
448 --WebDataSourceCount;
449
450#if USE(QUICK_LOOK)
451 // Added in -[WebCoreResourceHandleAsDelegate connection:didReceiveResponse:].
452 if (NSURL *url = [[self response] URL])
453 WebCore::removeQLPreviewConverterForURL(url);
454#endif
455
456 delete toPrivate(_private);
457
458 [super dealloc];
459}
