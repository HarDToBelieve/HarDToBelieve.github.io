47 void setDataSource(WebDataSource *, WebView*);

54private:

60 void retainDataSource();
61 void releaseDataSource();

63 WebDataSource *m_dataSource;
64 bool m_isDataSourceRetained;
65 RetainPtr<id> m_resourceLoadDelegate;
66 RetainPtr<id> m_downloadDelegate;
67 HashSet<unsigned long> m_loadingResources;
68};
