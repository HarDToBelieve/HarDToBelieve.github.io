81@protocol WebDocumentIncrementalSearching
82/*!
83@method searchFor:direction:caseSensitive:wrap:startInSelection:
84 @abstract Searches a document view for a string and highlights the string if it is found.
85 @param string The string to search for.
86 @param forward YES to search forward, NO to seach backwards.
87 @param caseFlag YES to for case-sensitive search, NO for case-insensitive search.
88 @param wrapFlag YES to wrap around, NO to avoid wrapping.
89 @param startInSelection YES to begin search in the selected text (useful for incremental searching), NO to begin search after the selected text.
90 @result YES if found, NO if not found.
91 */
92- (BOOL)searchFor:(NSString *)string direction:(BOOL)forward caseSensitive:(BOOL)caseFlag wrap:(BOOL)wrapFlag startInSelection:(BOOL)startInSelection;
93@end
