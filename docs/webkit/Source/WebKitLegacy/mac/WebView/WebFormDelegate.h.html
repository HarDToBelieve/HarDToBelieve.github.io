52/*!
53 @protocol WebFormDelegate
54*/
55WEBKIT_DEPRECATED_MAC(10_3, 10_14)
56@protocol WebFormDelegate <NSObject>
57
58// Various methods send by controls that edit text to their delegates, which are all
59// analogous to similar methods in AppKit/NSControl.h.
60// These methods are forwarded from widgets used in forms to the WebFormDelegate.
61
62- (void)textFieldDidBeginEditing:(DOMHTMLInputElement *)element inFrame:(WebFrame *)frame;
63- (void)textFieldDidEndEditing:(DOMHTMLInputElement *)element inFrame:(WebFrame *)frame;
64- (void)textDidChangeInTextField:(DOMHTMLInputElement *)element inFrame:(WebFrame *)frame;
65- (void)textDidChangeInTextArea:(DOMHTMLTextAreaElement *)element inFrame:(WebFrame *)frame;
66- (void)didFocusTextField:(DOMHTMLInputElement *)element inFrame:(WebFrame *)frame;
67
68- (BOOL)textField:(DOMHTMLInputElement *)element doCommandBySelector:(SEL)commandSelector inFrame:(WebFrame *)frame;
69#if !TARGET_OS_IPHONE
70- (BOOL)textField:(DOMHTMLInputElement *)element shouldHandleEvent:(NSEvent *)event inFrame:(WebFrame *)frame;
71#endif
72// Sent when a form is just about to be submitted (before the load is started)
73// listener must be sent continue when the delegate is done.
74- (void)frame:(WebFrame *)frame sourceFrame:(WebFrame *)sourceFrame willSubmitForm:(DOMElement *)form
75 withValues:(NSDictionary *)values submissionListener:(id <WebFormSubmissionListener>)listener;
76
77- (void)willSendSubmitEventToForm:(DOMHTMLFormElement *)element inFrame:(WebFrame *)sourceFrame withValues:(NSDictionary *)values;
78
79@end
80
81/*!
82 @class WebFormDelegate
83 @discussion The WebFormDelegate class responds to all WebFormDelegate protocol
84 methods by doing nothing. It's provided for the convenience of clients who only want
85 to implement some of the above methods and ignore others.
86*/
87@interface WebFormDelegate : NSObject <WebFormDelegate>
88@end
