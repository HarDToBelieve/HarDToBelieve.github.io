146- (void)_setDocumentView:(NSView <WebDocumentView> *)view
147{
148 WebDynamicScrollBarsView *sv = [self _scrollView];
149
150#if ENABLE(DRAG_SUPPORT)
151 core([self _webView])->dragController().setDidInitiateDrag(false);
152#endif
153
154#if !PLATFORM(IOS_FAMILY)
155 [sv setSuppressLayout:YES];
156
157 // If the old view is the first responder, transfer first responder status to the new view as
158 // a convenience and so that we don't leave the window pointing to a view that's no longer in it.
159 NSWindow *window = [sv window];
160 NSResponder *firstResponder = [window firstResponder];
161 bool makeNewViewFirstResponder = [firstResponder isKindOfClass:[NSView class]] && [(NSView *)firstResponder isDescendantOf:[sv documentView]];
162
163 // Suppress the resetting of drag margins since we know we can't affect them.
164 BOOL resetDragMargins = [window _needsToResetDragMargins];
165 [window _setNeedsToResetDragMargins:NO];
166#endif
167 [sv setDocumentView:view];
168#if !PLATFORM(IOS_FAMILY)
169 [window _setNeedsToResetDragMargins:resetDragMargins];
170
171 if (makeNewViewFirstResponder)
172 [window makeFirstResponder:view];
173 [sv setSuppressLayout:NO];
174#else
175 ASSERT(_private->webFrame);
176
177 auto* frame = core(_private->webFrame);
178
179 ASSERT(frame);
180 ASSERT(frame->page());
181
182 if (frame == &frame->page()->mainFrame())
183 [[self window] makeFirstResponder:[self documentView]];
184#endif
185}

187-(NSView <WebDocumentView> *)_makeDocumentViewForDataSource:(WebDataSource *)dataSource
188{
189 NSString* MIMEType = [dataSource _responseMIMEType];
190 if (!MIMEType)
191 MIMEType = @"text/html";
192 Class viewClass = [self _viewClassForMIMEType:MIMEType];
193 RetainPtr<NSView <WebDocumentView>> documentView;
194 if (viewClass) {
195 // If the dataSource's representation has already been created, and it is also the
196 // same class as the desired documentView, then use it as the documentView instead
197 // of creating another one (Radar 4340787).
198 id <WebDocumentRepresentation> dataSourceRepresentation = [dataSource representation];
199 if (dataSourceRepresentation && [dataSourceRepresentation class] == viewClass)
200 documentView = (NSView <WebDocumentView> *)dataSourceRepresentation;
201 else
202 documentView = adoptNS([(NSView <WebDocumentView> *)[viewClass alloc] init]);
203 }
204
205 [self _setDocumentView:documentView.get()];
206 return documentView.autorelease();
207}

293- (void)_install
294{
295 ASSERT(_private->webFrame);
296 ASSERT(_private->frameScrollView);
297
298 auto* frame = core(_private->webFrame);
299
300 ASSERT(frame);
301 ASSERT(frame->page());
302
303 // If this isn't the main frame, it must have an owner element set, or it
304 // won't ever get installed in the view hierarchy.
305 ASSERT(frame->isMainFrame() || frame->ownerElement());
306
307 auto* view = frame->view();
308
309 view->setPlatformWidget(_private->frameScrollView.get());
310
311 // FIXME: Frame tries to do this too. Is this code needed?
312 if (WebCore::RenderWidget* owner = frame->ownerRenderer()) {
313 owner->setWidget(view);
314 // Now the RenderWidget owns the view, so we don't any more.
315 }
316
317 view->updateCanHaveScrollbars();
318}

335- (id)initWithFrame:(NSRect)frame
336{
337 self = [super initWithFrame:frame];
338 if (!self)
339 return nil;
340
341 static bool didFirstTimeInitialization;
342 if (!didFirstTimeInitialization) {
343 didFirstTimeInitialization = true;
344
345 // Need to tell WebCore what function to call for the "History Item has Changed" notification.
346 // Note: We also do this in WebHistoryItem's init method.
347 // FIXME: This means that if we mix legacy WebKit and modern WebKit in the same process, we won't get both notifications.
348 WebCore::notifyHistoryItemChanged = WKNotifyHistoryItemChanged;
349
350#if !PLATFORM(IOS_FAMILY)
351 if (!WebKitLinkedOnOrAfter(WEBKIT_FIRST_VERSION_WITH_MAIN_THREAD_EXCEPTIONS))
352 setDefaultThreadViolationBehavior(WebCore::LogOnFirstThreadViolation, WebCore::ThreadViolationRoundOne);
353
354 bool throwExceptionsForRoundTwo = WebKitLinkedOnOrAfter(WEBKIT_FIRST_VERSION_WITH_ROUND_TWO_MAIN_THREAD_EXCEPTIONS);
355 if (!throwExceptionsForRoundTwo)
356 setDefaultThreadViolationBehavior(WebCore::LogOnFirstThreadViolation, WebCore::ThreadViolationRoundTwo);
357
358 bool throwExceptionsForRoundThree = WebKitLinkedOnOrAfter(WEBKIT_FIRST_VERSION_WITH_ROUND_THREE_MAIN_THREAD_EXCEPTIONS);
359 if (!throwExceptionsForRoundThree)
360 setDefaultThreadViolationBehavior(WebCore::LogOnFirstThreadViolation, WebCore::ThreadViolationRoundThree);
361#endif
362 }
363
364 _private = [[WebFrameViewPrivate alloc] init];
365
366 auto scrollView = adoptNS([[WebDynamicScrollBarsView alloc] initWithFrame:NSMakeRect(0.0f, 0.0f, frame.size.width, frame.size.height)]);
367 _private->frameScrollView = scrollView;
368#if PLATFORM(IOS_FAMILY)
369 [scrollView setDelegate:self];
370#else
371 [scrollView setContentView:adoptNS([[WebClipView alloc] initWithFrame:[scrollView bounds]]).get()];
372#endif
373 [scrollView setDrawsBackground:NO];
374 [scrollView setHasVerticalScroller:NO];
375 [scrollView setHasHorizontalScroller:NO];
376 [scrollView setAutoresizingMask:NSViewWidthSizable | NSViewHeightSizable];
377 [scrollView setLineScroll:WebCore::Scrollbar::pixelsPerLineStep()];
378 [self addSubview:scrollView.get()];
379
380 // Don't call our overridden version of setNextKeyView here; we need to make the standard NSView
381 // link between us and our subview so that previousKeyView and previousValidKeyView work as expected.
382 // This works together with our becomeFirstResponder and setNextKeyView overrides.
383 [super setNextKeyView:scrollView.get()];
384
385 return self;
386}

440- (BOOL)becomeFirstResponder
441{
442 // This works together with setNextKeyView to splice the WebFrameView into
443 // the key loop similar to the way NSScrollView does this. Note that
444 // WebView has similar code.
445
446 NSWindow *window = [self window];
447 if ([window keyViewSelectionDirection] == NSSelectingPrevious) {
448 NSView *previousValidKeyView = [self previousValidKeyView];
449 // If we couldn't find a previous valid key view, ask the WebView. This handles frameset
450 // cases (one is mentioned in Radar bug 3748628). Note that previousValidKeyView should
451 // never be self but can be due to AppKit oddness (mentioned in Radar bug 3748628).
452 if (previousValidKeyView == nil || previousValidKeyView == self)
453 previousValidKeyView = [[[self webFrame] webView] previousValidKeyView];
454 [window makeFirstResponder:previousValidKeyView];
455 } else {
456 // If the scroll view won't accept first-responderness now, then just become
457 // the first responder ourself like a normal view. This lets us be the first
458 // responder in cases where no page has yet been loaded.
459 if ([[self _scrollView] acceptsFirstResponder])
460 [window makeFirstResponder:[self _scrollView]];
461 }
462
463 return YES;
464}

546- (NSRect)visibleRect
547{
548 // This method can be called beneath -[NSView dealloc] after we have cleared _private.
549 if (!_private)
550 return [super visibleRect];
551
552 // FIXME: <rdar://problem/6213380> This method does not work correctly with transforms, for two reasons:
553 // 1) [super visibleRect] does not account for the transform, since it is not represented
554 // in the NSView hierarchy.
555 // 2) -_getVisibleRect: does not correct for transforms.
556
557 NSRect rendererVisibleRect;
558 if (![[self webFrame] _getVisibleRect:&rendererVisibleRect])
559 return [super visibleRect];
560
561 if (NSIsEmptyRect(rendererVisibleRect))
562 return NSZeroRect;
563
564 NSRect viewVisibleRect = [super visibleRect];
565 if (NSIsEmptyRect(viewVisibleRect))
566 return NSZeroRect;
567
568 NSRect frame = [self frame];
569 // rendererVisibleRect is in the parent's coordinate space, and frame is in the superview's coordinate space.
570 // The return value from this method needs to be in this view's coordinate space. We get that right by subtracting
571 // the origins (and correcting for flipping), but when we support transforms, we will need to do better than this.
572 rendererVisibleRect.origin.x -= frame.origin.x;
573 rendererVisibleRect.origin.y = NSMaxY(frame) - NSMaxY(rendererVisibleRect);
574 return NSIntersectionRect(rendererVisibleRect, viewVisibleRect);
575}

634- (BOOL)_scrollToBeginningOfDocument
635{
636 if ([self _scrollOverflowInDirection:WebCore::ScrollUp granularity:WebCore::ScrollByDocument])
637 return YES;
638 if (![self _isScrollable])
639 return NO;
640 NSPoint point = [(NSView *)[[self _scrollView] documentView] frame].origin;
641 point.x += [[self _scrollView] scrollOrigin].x;
642 point.y += [[self _scrollView] scrollOrigin].y;
643 return [[self _contentView] _scrollTo:&point animate:YES];
644}
645
646- (BOOL)_scrollToEndOfDocument
647{
648 if ([self _scrollOverflowInDirection:WebCore::ScrollDown granularity:WebCore::ScrollByDocument])
649 return YES;
650 if (![self _isScrollable])
651 return NO;
652 NSRect frame = [(NSView *)[[self _scrollView] documentView] frame];
653
654 bool isVertical = [self _isVerticalDocument];
655 bool isFlipped = [self _isFlippedDocument];
656
657 NSPoint point;
658 if (isVertical) {
659 if (!isFlipped)
660 point = NSMakePoint(frame.origin.x, NSMaxY(frame));
661 else
662 point = NSMakePoint(frame.origin.x, NSMinY(frame));
663 } else {
664 if (!isFlipped)
665 point = NSMakePoint(NSMaxX(frame), frame.origin.y);
666 else
667 point = NSMakePoint(NSMinX(frame), frame.origin.y);
668 }
669
670 // Reset the position opposite to the block progression direction.
671 if (isVertical)
672 point.x += [[self _scrollView] scrollOrigin].x;
673 else
674 point.y += [[self _scrollView] scrollOrigin].y;
675 return [[self _contentView] _scrollTo:&point animate:YES];
676}

869- (void)keyDown:(NSEvent *)event
870#else
871- (void)keyDown:(WebEvent *)event
872#endif
873{
874 // Implement common browser behaviors for all kinds of content.
875
876 // FIXME: This is not a good time to execute commands for WebHTMLView. We should run these at the time commands sent by key bindings
877 // are executed for consistency.
878 // This doesn't work automatically because most of the keys handled here are translated into moveXXX commands, which are not handled
879 // by Editor when focus is not in editable content.
880
881#if !PLATFORM(IOS_FAMILY)
882 NSString *characters = [event characters];
883 int modifierFlags = [event modifierFlags];
884#else
885 NSString *characters = event.characters;
886 int modifierFlags = event.modifierFlags;
887#endif
888 int index, count;
889 BOOL callSuper = YES;
890 auto coreFrame = [self _web_frame];
891 BOOL maintainsBackForwardList = coreFrame && static_cast<BackForwardList&>(coreFrame->page()->backForward().client()).enabled() ? YES : NO;
892
893 count = [characters length];
894 for (index = 0; index < count; ++index) {
895 switch ([characters characterAtIndex:index]) {
896 case NSDeleteCharacter:
897 if (!maintainsBackForwardList || ![[[self _webView] preferences] backspaceKeyNavigationEnabled]) {
898 callSuper = YES;
899 break;
900 }
901 // This odd behavior matches some existing browsers,
902 // including Windows IE
903 if (modifierFlags & NSEventModifierFlagShift) {
904 [self _goForward];
905 } else {
906 [self _goBack];
907 }
908 callSuper = NO;
909 break;
910 case SpaceKey:
911 // Checking for a control will allow events to percolate
912 // correctly when the focus is on a form control and we
913 // are in full keyboard access mode.
914 if ((![self allowsScrolling] && ![self _largestScrollableChild]) || [self _firstResponderIsFormControl]) {
915 callSuper = YES;
916 break;
917 }
918 if (modifierFlags & NSEventModifierFlagShift) {
919 [self scrollPageUp:nil];
920 } else {
921 [self scrollPageDown:nil];
922 }
923 callSuper = NO;
924 break;
925 case NSPageUpFunctionKey:
926 if (![self allowsScrolling] && ![self _largestScrollableChild]) {
927 callSuper = YES;
928 break;
929 }
930 [self scrollPageUp:nil];
931 callSuper = NO;
932 break;
933 case NSPageDownFunctionKey:
934 if (![self allowsScrolling] && ![self _largestScrollableChild]) {
935 callSuper = YES;
936 break;
937 }
938 [self scrollPageDown:nil];
939 callSuper = NO;
940 break;
941 case NSHomeFunctionKey:
942 if (![self allowsScrolling] && ![self _largestScrollableChild]) {
943 callSuper = YES;
944 break;
945 }
946 [self scrollToBeginningOfDocument:nil];
947 callSuper = NO;
948 break;
949 case NSEndFunctionKey:
950 if (![self allowsScrolling] && ![self _largestScrollableChild]) {
951 callSuper = YES;
952 break;
953 }
954 [self scrollToEndOfDocument:nil];
955 callSuper = NO;
956 break;
957 case NSUpArrowFunctionKey:
958 // We don't handle shifted or control-arrow keys here, so let super have a chance.
959 if (modifierFlags & (NSEventModifierFlagShift | NSEventModifierFlagControl)) {
960 callSuper = YES;
961 break;
962 }
963#if !PLATFORM(IOS_FAMILY)
964 if ((![self allowsScrolling] && ![self _largestScrollableChild]) ||
965 [[[self window] firstResponder] isKindOfClass:[NSPopUpButton class]]) {
966 // Let arrow keys go through to pop up buttons
967 // <rdar://problem/3455910>: hitting up or down arrows when focus is on a
968 // pop-up menu should pop the menu
969 callSuper = YES;
970 break;
971 }
972#endif
973 if (modifierFlags & NSEventModifierFlagCommand) {
974 [self scrollToBeginningOfDocument:nil];
975 } else if (modifierFlags & NSEventModifierFlagOption) {
976 [self scrollPageUp:nil];
977 } else {
978 [self scrollLineUp:nil];
979 }
980 callSuper = NO;
981 break;
982 case NSDownArrowFunctionKey:
983 // We don't handle shifted or control-arrow keys here, so let super have a chance.
984 if (modifierFlags & (NSEventModifierFlagShift | NSEventModifierFlagControl)) {
985 callSuper = YES;
986 break;
987 }
988#if !PLATFORM(IOS_FAMILY)
989 if ((![self allowsScrolling] && ![self _largestScrollableChild]) ||
990 [[[self window] firstResponder] isKindOfClass:[NSPopUpButton class]]) {
991 // Let arrow keys go through to pop up buttons
992 // <rdar://problem/3455910>: hitting up or down arrows when focus is on a
993 // pop-up menu should pop the menu
994 callSuper = YES;
995 break;
996 }
997#endif
998 if (modifierFlags & NSEventModifierFlagCommand) {
999 [self scrollToEndOfDocument:nil];
1000 } else if (modifierFlags & NSEventModifierFlagOption) {
1001 [self scrollPageDown:nil];
1002 } else {
1003 [self scrollLineDown:nil];
1004 }
1005 callSuper = NO;
1006 break;
1007 case NSLeftArrowFunctionKey:
1008 // We don't handle shifted or control-arrow keys here, so let super have a chance.
1009 if (modifierFlags & (NSEventModifierFlagShift | NSEventModifierFlagControl)) {
1010 callSuper = YES;
1011 break;
1012 }
1013 // Check back/forward related keys.
1014 if (modifierFlags & NSEventModifierFlagCommand) {
1015 if (!maintainsBackForwardList) {
1016 callSuper = YES;
1017 break;
1018 }
1019 [self _goBack];
1020 } else {
1021 // Now check scrolling related keys.
1022 if ((![self allowsScrolling] && ![self _largestScrollableChild])) {
1023 callSuper = YES;
1024 break;
1025 }
1026
1027 if (modifierFlags & NSEventModifierFlagOption) {
1028 [self _pageHorizontally:YES];
1029 } else {
1030 [self _scrollLineHorizontally:YES];
1031 }
1032 }
1033 callSuper = NO;
1034 break;
1035 case NSRightArrowFunctionKey:
1036 // We don't handle shifted or control-arrow keys here, so let super have a chance.
1037 if (modifierFlags & (NSEventModifierFlagShift | NSEventModifierFlagControl)) {
1038 callSuper = YES;
1039 break;
1040 }
1041 // Check back/forward related keys.
1042 if (modifierFlags & NSEventModifierFlagCommand) {
1043 if (!maintainsBackForwardList) {
1044 callSuper = YES;
1045 break;
1046 }
1047 [self _goForward];
1048 } else {
1049 // Now check scrolling related keys.
1050 if ((![self allowsScrolling] && ![self _largestScrollableChild])) {
1051 callSuper = YES;
1052 break;
1053 }
1054
1055 if (modifierFlags & NSEventModifierFlagOption) {
1056 [self _pageHorizontally:NO];
1057 } else {
1058 [self _scrollLineHorizontally:NO];
1059 }
1060 }
1061 callSuper = NO;
1062 break;
1063 }
1064 }
1065 if (callSuper) {
1066 [super keyDown:event];
1067 } else {
1068#if PLATFORM(MAC)
1069 // If we did something useful, get the cursor out of the way.
1070 [NSCursor setHiddenUntilMouseMoves:YES];
1071#endif
1072 }
1073}
