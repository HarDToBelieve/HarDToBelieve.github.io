52static WebCore::IntRect screenRectOfContents(WebCore::Element* element)
53{
54 ASSERT(element);
55 if (element->renderer() && element->renderer()->hasLayer() && element->renderer()->enclosingLayer()->isComposited()) {
56 WebCore::FloatQuad contentsBox = static_cast<WebCore::FloatRect>(element->renderer()->enclosingLayer()->backing()->contentsBox());
57 contentsBox = element->renderer()->localToAbsoluteQuad(contentsBox);
58 return element->renderer()->view().frameView().contentsToScreen(contentsBox.enclosingBoundingBox());
59 }
60 return element->screenRect();
61}

160- (void)applicationDidResignActive:(NSNotification*)notification
161{
162 NSWindow* fullscreenWindow = [self window];
163
164 // Replicate the QuickTime Player (X) behavior when losing active application status:
165 // Is the fullscreen screen the main screen? (Note: this covers the case where only a
166 // single screen is available.) Is the fullscreen screen on the current space? IFF so,
167 // then exit fullscreen mode.
168 if (fullscreenWindow.screen == [NSScreen screens][0] && fullscreenWindow.onActiveSpace)
169 [self cancelOperation:self];
170}

172- (void)applicationDidChangeScreenParameters:(NSNotification*)notification
173{
174 // The user may have changed the main screen by moving the menu bar, or they may have changed
175 // the Dock's size or location, or they may have changed the fullscreen screen's dimensions.
176 // Update our presentation parameters, and ensure that the full screen window occupies the
177 // entire screen:
178 [self _updateMenuAndDockForFullScreen];
179 NSWindow* window = [self window];
180 NSRect screenFrame = [[window screen] frame];
181 [window setFrame:screenFrame display:YES];
182 [_backgroundWindow.get() setFrame:screenFrame display:YES];
183}

188- (void)enterFullScreen:(NSScreen *)screen
189{
190 if (_isFullScreen)
191 return;
192 _isFullScreen = YES;
193
194 [self _updateMenuAndDockForFullScreen];
195
196 if (!screen)
197 screen = [NSScreen mainScreen];
198 NSRect screenFrame = [screen frame];
199
200 NSRect webViewFrame = convertRectToScreen([_webView window], [_webView convertRect:[_webView frame] toView:nil]);
201
202 // Flip coordinate system:
203 webViewFrame.origin.y = NSMaxY([[[NSScreen screens] objectAtIndex:0] frame]) - NSMaxY(webViewFrame);
204
205 CGWindowID windowID = [[_webView window] windowNumber];
206 RetainPtr<CGImageRef> webViewContents = adoptCF(CGWindowListCreateImage(NSRectToCGRect(webViewFrame), kCGWindowListOptionIncludingWindow, windowID, kCGWindowImageShouldBeOpaque));
207
208 // Screen updates to be re-enabled in beganEnterFullScreenWithInitialFrame:finalFrame:
209 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
210 NSDisableScreenUpdates();
211 [[self window] setAutodisplay:NO];
212 ALLOW_DEPRECATED_DECLARATIONS_END
213
214 NSResponder *webWindowFirstResponder = [[_webView window] firstResponder];
215 [[self window] setFrame:screenFrame display:NO];
216
217 _initialFrame = screenRectOfContents(_element.get());
218
219 // Swap the webView placeholder into place.
220 if (!_webViewPlaceholder) {
221 _webViewPlaceholder = adoptNS([[NSView alloc] init]);
222 [_webViewPlaceholder.get() setLayer:[CALayer layer]];
223 [_webViewPlaceholder.get() setWantsLayer:YES];
224 }
225 [[_webViewPlaceholder.get() layer] setContents:(__bridge id)webViewContents.get()];
226 _scrollPosition = [_webView _mainCoreFrame]->view()->scrollPosition();
227 [self _swapView:_webView with:_webViewPlaceholder.get()];
228
229 // Then insert the WebView into the full screen window
230 NSView* contentView = [[self window] contentView];
231 [contentView addSubview:_webView positioned:NSWindowBelow relativeTo:nil];
232 [_webView setFrame:[contentView bounds]];
233 [[_webViewPlaceholder.get() window] recalculateKeyViewLoop];
234
235 [[self window] makeResponder:webWindowFirstResponder firstResponderIfDescendantOfView:_webView];
236
237 _savedScale = [_webView _viewScaleFactor];
238 [_webView _scaleWebView:1 atOrigin:NSMakePoint(0, 0)];
239 [self _manager]->willEnterFullscreen(*_element);
240 [self _manager]->setAnimatingFullscreen(true);
241 [self _document]->updateLayout();
242
243 _finalFrame = screenRectOfContents(_element.get());
244
245 [self _updateMenuAndDockForFullScreen];
246
247 [self _startEnterFullScreenAnimationWithDuration:defaultAnimationDuration];
248
249 _isEnteringFullScreen = true;
250}

252static void setClipRectForWindow(NSWindow *window, NSRect clipRect)
253{
254 CGSWindowID windowNumber = (CGSWindowID)window.windowNumber;
255 CGSRegionObj shape;
256 CGRect cgClipRect = NSRectToCGRect(clipRect);
257 CGSNewRegionWithRect(&cgClipRect, &shape);
258 CGSSetWindowClipShape(CGSMainConnectionID(), windowNumber, shape);
259 CGSReleaseRegion(shape);
260}

262- (void)finishedEnterFullScreenAnimation:(bool)completed
263{
264 if (!_isEnteringFullScreen)
265 return;
266 _isEnteringFullScreen = NO;
267
268 if (completed) {
269 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
270 // Screen updates to be re-enabled at the end of this block
271 NSDisableScreenUpdates();
272 ALLOW_DEPRECATED_DECLARATIONS_END
273 [self _manager]->setAnimatingFullscreen(false);
274 [self _manager]->didEnterFullscreen();
275
276 NSRect windowBounds = [[self window] frame];
277 windowBounds.origin = NSZeroPoint;
278 setClipRectForWindow(self.window, windowBounds);
279
280 NSWindow *webWindow = [_webViewPlaceholder.get() window];
281 // In Lion, NSWindow will animate into and out of orderOut operations. Suppress that
282 // behavior here, making sure to reset the animation behavior afterward.
283 NSWindowAnimationBehavior animationBehavior = [webWindow animationBehavior];
284 [webWindow setAnimationBehavior:NSWindowAnimationBehaviorNone];
285 [webWindow orderOut:self];
286 [webWindow setAnimationBehavior:animationBehavior];
287
288 [_fadeAnimation.get() stopAnimation];
289 [_fadeAnimation.get() setWindow:nil];
290 _fadeAnimation = nullptr;
291
292 [_backgroundWindow.get() orderOut:self];
293 [_backgroundWindow.get() setFrame:NSZeroRect display:YES];
294 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
295 NSEnableScreenUpdates();
296 ALLOW_DEPRECATED_DECLARATIONS_END
297 } else
298 [_scaleAnimation.get() stopAnimation];
299}

308- (void)exitFullScreen
309{
310 if (!_isFullScreen)
311 return;
312 _isFullScreen = NO;
313
314 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
315 // Screen updates to be re-enabled in beganExitFullScreenWithInitialFrame:finalFrame:
316 NSDisableScreenUpdates();
317 [[self window] setAutodisplay:NO];
318 ALLOW_DEPRECATED_DECLARATIONS_END
319
320 _finalFrame = screenRectOfContents(_element.get());
321
322 [self _manager]->willExitFullscreen();
323 [self _manager]->setAnimatingFullscreen(true);
324
325 if (_isEnteringFullScreen)
326 [self finishedEnterFullScreenAnimation:NO];
327
328 [self _updateMenuAndDockForFullScreen];
329
330 NSWindow* webWindow = [_webViewPlaceholder.get() window];
331 // In Lion, NSWindow will animate into and out of orderOut operations. Suppress that
332 // behavior here, making sure to reset the animation behavior afterward.
333 NSWindowAnimationBehavior animationBehavior = [webWindow animationBehavior];
334 [webWindow setAnimationBehavior:NSWindowAnimationBehaviorNone];
335 // If the user has moved the fullScreen window into a new space, temporarily change
336 // the collectionBehavior of the webView's window so that it is pulled into the active space:
337 if (!webWindow.onActiveSpace) {
338 NSWindowCollectionBehavior behavior = [webWindow collectionBehavior];
339 [webWindow setCollectionBehavior:NSWindowCollectionBehaviorCanJoinAllSpaces];
340 [webWindow orderWindow:NSWindowBelow relativeTo:[[self window] windowNumber]];
341 [webWindow setCollectionBehavior:behavior];
342 } else
343 [webWindow orderWindow:NSWindowBelow relativeTo:[[self window] windowNumber]];
344 [webWindow setAnimationBehavior:animationBehavior];
345
346 [self _startExitFullScreenAnimationWithDuration:defaultAnimationDuration];
347 _isExitingFullScreen = YES;
348}

350- (void)finishedExitFullScreenAnimation:(bool)completed
351{
352 if (!_isExitingFullScreen)
353 return;
354 _isExitingFullScreen = NO;
355
356 [self _updateMenuAndDockForFullScreen];
357
358 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
359 // Screen updates to be re-enabled at the end of this function
360 NSDisableScreenUpdates();
361 ALLOW_DEPRECATED_DECLARATIONS_END
362
363 [self _manager]->setAnimatingFullscreen(false);
364 [self _manager]->didExitFullscreen();
365 [_webView _scaleWebView:_savedScale atOrigin:NSMakePoint(0, 0)];
366
367 NSResponder *firstResponder = [[self window] firstResponder];
368 [self _swapView:_webViewPlaceholder.get() with:_webView];
369 [_webView _mainCoreFrame]->view()->setScrollPosition(_scrollPosition);
370 [[_webView window] makeResponder:firstResponder firstResponderIfDescendantOfView:_webView];
371
372 NSRect windowBounds = [[self window] frame];
373 windowBounds.origin = NSZeroPoint;
374 setClipRectForWindow(self.window, windowBounds);
375
376 [[self window] orderOut:self];
377 [[self window] setFrame:NSZeroRect display:YES];
378
379 [_fadeAnimation.get() stopAnimation];
380 [_fadeAnimation.get() setWindow:nil];
381 _fadeAnimation = nullptr;
382
383 [_backgroundWindow.get() orderOut:self];
384 [_backgroundWindow.get() setFrame:NSZeroRect display:YES];
385
386 [[_webView window] makeKeyAndOrderFront:self];
387
388 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
389 NSEnableScreenUpdates();
390 ALLOW_DEPRECATED_DECLARATIONS_END
391}

399- (void)close
400{
401 // We are being asked to close rapidly, most likely because the page
402 // has closed or the web process has crashed. Just walk through our
403 // normal exit full screen sequence, but don't wait to be called back
404 // in response.
405 if (_isFullScreen)
406 [self exitFullScreen];
407
408 if (_isExitingFullScreen)
409 [self finishedExitFullScreenAnimation:YES];
410
411 [super close];
412}

428- (void)_updateMenuAndDockForFullScreen
429{
430 NSApplicationPresentationOptions options = NSApplicationPresentationDefault;
431 NSScreen* fullscreenScreen = [[self window] screen];
432
433 if (_isFullScreen) {
434 // Auto-hide the menu bar if the fullscreenScreen contains the menu bar:
435 // NOTE: if the fullscreenScreen contains the menu bar but not the dock, we must still
436 // auto-hide the dock, or an exception will be thrown.
437 if ([[NSScreen screens] objectAtIndex:0] == fullscreenScreen)
438 options |= (NSApplicationPresentationAutoHideMenuBar | NSApplicationPresentationAutoHideDock);
439 // Check if the current screen contains the dock by comparing the screen's frame to its
440 // visibleFrame; if a dock is present, the visibleFrame will differ. If the current screen
441 // contains the dock, hide it.
442 else if (!NSEqualRects([fullscreenScreen frame], [fullscreenScreen visibleFrame]))
443 options |= NSApplicationPresentationAutoHideDock;
444 }
445
446 NSApp.presentationOptions = options;
447}

462- (void)_swapView:(NSView*)view with:(NSView*)otherView
463{
464 [CATransaction begin];
465 [CATransaction setDisableActions:YES];
466 [otherView setFrame:[view frame]];
467 [otherView setAutoresizingMask:[view autoresizingMask]];
468 [otherView removeFromSuperview];
469 [[view superview] addSubview:otherView positioned:NSWindowAbove relativeTo:view];
470 [view removeFromSuperview];
471 [CATransaction commit];
472}

474static RetainPtr<NSWindow> createBackgroundFullscreenWindow(NSRect frame)
475{
476 auto window = adoptNS([[NSWindow alloc] initWithContentRect:frame styleMask:NSWindowStyleMaskBorderless backing:NSBackingStoreBuffered defer:NO]);
477 [window setOpaque:YES];
478 [window setBackgroundColor:[NSColor blackColor]];
479 [window setReleasedWhenClosed:NO];
480 return window;
481}

483static NSRect windowFrameFromApparentFrames(NSRect screenFrame, NSRect initialFrame, NSRect finalFrame)
484{
485 NSRect initialWindowFrame;
486 if (!NSWidth(initialFrame) || !NSWidth(finalFrame) || !NSHeight(initialFrame) || !NSHeight(finalFrame))
487 return screenFrame;
488
489 CGFloat xScale = NSWidth(screenFrame) / NSWidth(finalFrame);
490 CGFloat yScale = NSHeight(screenFrame) / NSHeight(finalFrame);
491 CGFloat xTrans = NSMinX(screenFrame) - NSMinX(finalFrame);
492 CGFloat yTrans = NSMinY(screenFrame) - NSMinY(finalFrame);
493 initialWindowFrame.size = NSMakeSize(NSWidth(initialFrame) * xScale, NSHeight(initialFrame) * yScale);
494 initialWindowFrame.origin = NSMakePoint
495 ( NSMinX(initialFrame) + xTrans / (NSWidth(finalFrame) / NSWidth(initialFrame))
496 , NSMinY(initialFrame) + yTrans / (NSHeight(finalFrame) / NSHeight(initialFrame)));
497 return initialWindowFrame;
498}

500- (void)_startEnterFullScreenAnimationWithDuration:(NSTimeInterval)duration
501{
502 NSRect screenFrame = [[[self window] screen] frame];
503 NSRect initialWindowFrame = windowFrameFromApparentFrames(screenFrame, _initialFrame, _finalFrame);
504
505 _scaleAnimation = adoptNS([[WebWindowScaleAnimation alloc] initWithHintedDuration:duration window:[self window] initalFrame:initialWindowFrame finalFrame:screenFrame]);
506
507 [_scaleAnimation.get() setAnimationBlockingMode:NSAnimationNonblocking];
508 [_scaleAnimation.get() setDelegate:self];
509 [_scaleAnimation.get() setCurrentProgress:0];
510 [_scaleAnimation.get() startAnimation];
511
512 // setClipRectForWindow takes window coordinates, so convert from screen coordinates here:
513 NSRect finalBounds = _finalFrame;
514 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
515 finalBounds.origin = [[self window] convertScreenToBase:finalBounds.origin];
516 ALLOW_DEPRECATED_DECLARATIONS_END
517 setClipRectForWindow(self.window, finalBounds);
518
519 [[self window] makeKeyAndOrderFront:self];
520
521 if (!_backgroundWindow)
522 _backgroundWindow = createBackgroundFullscreenWindow(screenFrame);
523 else
524 [_backgroundWindow.get() setFrame:screenFrame display:NO];
525
526 CGFloat currentAlpha = 0;
527 if (_fadeAnimation) {
528 currentAlpha = [_fadeAnimation.get() currentAlpha];
529 [_fadeAnimation.get() stopAnimation];
530 [_fadeAnimation.get() setWindow:nil];
531 }
532
533 _fadeAnimation = adoptNS([[WebWindowFadeAnimation alloc] initWithDuration:duration
534 window:_backgroundWindow.get()
535 initialAlpha:currentAlpha
536 finalAlpha:1]);
537 [_fadeAnimation.get() setAnimationBlockingMode:NSAnimationNonblocking];
538 [_fadeAnimation.get() setCurrentProgress:0];
539 [_fadeAnimation.get() startAnimation];
540
541 [_backgroundWindow.get() orderWindow:NSWindowBelow relativeTo:[[self window] windowNumber]];
542
543 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
544 [[self window] setAutodisplay:YES];
545 ALLOW_DEPRECATED_DECLARATIONS_END
546 [[self window] displayIfNeeded];
547 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
548 // Screen updates disabled in enterFullScreen:
549 NSEnableScreenUpdates();
550 ALLOW_DEPRECATED_DECLARATIONS_END
551}

553- (void)_startExitFullScreenAnimationWithDuration:(NSTimeInterval)duration
554{
555 NSRect screenFrame = [[[self window] screen] frame];
556 NSRect initialWindowFrame = windowFrameFromApparentFrames(screenFrame, _initialFrame, _finalFrame);
557
558 NSRect currentFrame = _scaleAnimation ? [_scaleAnimation.get() currentFrame] : [[self window] frame];
559 _scaleAnimation = adoptNS([[WebWindowScaleAnimation alloc] initWithHintedDuration:duration window:[self window] initalFrame:currentFrame finalFrame:initialWindowFrame]);
560
561 [_scaleAnimation.get() setAnimationBlockingMode:NSAnimationNonblocking];
562 [_scaleAnimation.get() setDelegate:self];
563 [_scaleAnimation.get() setCurrentProgress:0];
564 [_scaleAnimation.get() startAnimation];
565
566 if (!_backgroundWindow)
567 _backgroundWindow = createBackgroundFullscreenWindow(screenFrame);
568 else
569 [_backgroundWindow.get() setFrame:screenFrame display:NO];
570
571 CGFloat currentAlpha = 1;
572 if (_fadeAnimation) {
573 currentAlpha = [_fadeAnimation.get() currentAlpha];
574 [_fadeAnimation.get() stopAnimation];
575 [_fadeAnimation.get() setWindow:nil];
576 }
577 _fadeAnimation = adoptNS([[WebWindowFadeAnimation alloc] initWithDuration:duration
578 window:_backgroundWindow.get()
579 initialAlpha:currentAlpha
580 finalAlpha:0]);
581 [_fadeAnimation.get() setAnimationBlockingMode:NSAnimationNonblocking];
582 [_fadeAnimation.get() setCurrentProgress:0];
583 [_fadeAnimation.get() startAnimation];
584
585 [_backgroundWindow.get() orderWindow:NSWindowBelow relativeTo:[[self window] windowNumber]];
586
587 // setClipRectForWindow takes window coordinates, so convert from screen coordinates here:
588 NSRect finalBounds = _finalFrame;
589 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
590 finalBounds.origin = [[self window] convertScreenToBase:finalBounds.origin];
591 ALLOW_DEPRECATED_DECLARATIONS_END
592 setClipRectForWindow(self.window, finalBounds);
593
594 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
595 [[self window] setAutodisplay:YES];
596 ALLOW_DEPRECATED_DECLARATIONS_END
597 [[self window] displayIfNeeded];
598
599 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
600 // Screen updates disabled in exitFullScreen:
601 NSEnableScreenUpdates();
602 ALLOW_DEPRECATED_DECLARATIONS_END
603}
