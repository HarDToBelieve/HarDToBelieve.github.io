171- (void)receivedData:(NSData *)data withDataSource:(WebDataSource *)dataSource
172{
173 auto protectedSelf = retainPtr(self);
174 WebFrame *webFrame = [dataSource webFrame];
175 if (!webFrame)
176 return;
177
178 if (!_private->pluginView)
179 [webFrame _commitData:data];
180
181 // If the document is a stand-alone media document, now is the right time to cancel the WebKit load
182 Frame* coreFrame = core(webFrame);
183 if (coreFrame->document()->isMediaDocument() && coreFrame->loader().documentLoader())
184 coreFrame->loader().documentLoader()->cancelMainResourceLoad(coreFrame->loader().client().pluginWillHandleLoadError(coreFrame->loader().documentLoader()->response()));
185
186 if (_private->pluginView) {
187 if (!_private->hasSentResponseToPlugin) {
188 [_private->manualLoader pluginView:_private->pluginView receivedResponse:[dataSource response]];
189 _private->hasSentResponseToPlugin = YES;
190 }
191
192 [_private->manualLoader pluginView:_private->pluginView receivedData:data];
193 }
194}

204{

206
207 if (_private->pluginView) {
208 [_private->manualLoader pluginViewFinishedLoading:_private->pluginView];
209 return;
210 }

212 if (!webFrame)

229- (NSString *)documentSource
230{
231 if ([self _isDisplayingWebArchive]) {
232 SharedBuffer *parsedArchiveData = [_private->dataSource _documentLoader]->parsedArchiveData();
233 return adoptNS([[NSString alloc] initWithData:parsedArchiveData ? parsedArchiveData->createNSData().get() : nil encoding:NSUTF8StringEncoding]).autorelease();
234 }
235
236 Frame* coreFrame = core([_private->dataSource webFrame]);
237 if (!coreFrame)
238 return nil;
239 Document* document = coreFrame->document();
240 if (!document)
241 return nil;
242 TextResourceDecoder* decoder = document->decoder();
243 if (!decoder)
244 return nil;
245 NSData *data = [_private->dataSource data];
246 if (!data)
247 return nil;
248 return decoder->encoding().decode(reinterpret_cast<const char*>([data bytes]), [data length]);
249}

332- (NSArray *)controlsInForm:(DOMElement *)form
333{
334 auto formElement = formElementFromDOMElement(form);
335 if (!formElement)
336 return nil;
337
338 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
339 auto result = createNSArray(formElement->unsafeAssociatedElements(), [] (auto& weakElement) -> DOMElement * {
340 auto coreElement = makeRefPtr(weakElement.get());
341 if (!coreElement->asFormAssociatedElement()->isEnumeratable()) // Skip option elements, other duds
342 return nil;
343 return kit(coreElement.get());
344 });
345 return [result count] ? result.autorelease() : nil;

348// Either get cached regexp or build one that matches any of the labels.
349// The regexp we build is of the form: (STR1|STR2|STRN)
350static RegularExpression* regExpForLabels(NSArray *labels)
351{
352 // All the ObjC calls in this method are simple array and string
353 // calls which we can assume do not raise exceptions
354
355 // Parallel arrays that we use to cache regExps. In practice the number of expressions
356 // that the app will use is equal to the number of locales is used in searching.
357 static const unsigned int regExpCacheSize = 4;
358 static NeverDestroyed<RetainPtr<NSMutableArray>> regExpLabels = adoptNS([[NSMutableArray alloc] initWithCapacity:regExpCacheSize]);
359 static NeverDestroyed<Vector<RegularExpression*>> regExps;
360 static NeverDestroyed<RegularExpression> wordRegExp("\\w");
361
362 RegularExpression* result;
363 CFIndex cacheHit = [regExpLabels.get() indexOfObject:labels];
364 if (cacheHit != NSNotFound)
365 result = regExps.get().at(cacheHit);
366 else {
367 StringBuilder pattern;
368 pattern.append('(');
369 unsigned numLabels = [labels count];
370 unsigned i;
371 for (i = 0; i < numLabels; i++) {
372 String label = [labels objectAtIndex:i];
373
374 bool startsWithWordCharacter = false;
375 bool endsWithWordCharacter = false;
376 if (label.length()) {
377 startsWithWordCharacter = wordRegExp.get().match(label.substring(0, 1)) >= 0;
378 endsWithWordCharacter = wordRegExp.get().match(label.substring(label.length() - 1, 1)) >= 0;
379 }
380
381 // Search for word boundaries only if label starts/ends with "word characters".
382 // If we always searched for word boundaries, this wouldn't work for languages such as Japanese.
383 pattern.append(i ? "|" : "", startsWithWordCharacter ? "\\b" : "", label, endsWithWordCharacter ? "\\b" : "");
384 }
385 pattern.append(')');
386 result = new RegularExpression(pattern.toString(), JSC::Yarr::TextCaseInsensitive);
387 }
388
389 // add regexp to the cache, making sure it is at the front for LRU ordering
390 if (cacheHit != 0) {
391 if (cacheHit != NSNotFound) {
392 // remove from old spot
393 [regExpLabels.get() removeObjectAtIndex:cacheHit];
394 regExps.get().remove(cacheHit);
395 }
396 // add to start
397 [regExpLabels.get() insertObject:labels atIndex:0];
398 regExps.get().insert(0, result);
399 // trim if too big
400 if ([regExpLabels.get() count] > regExpCacheSize) {
401 [regExpLabels.get() removeObjectAtIndex:regExpCacheSize];
402 RegularExpression* last = regExps.get().last();
403 regExps.get().removeLast();
404 delete last;
405 }
406 }
407 return result;
408}

411static NSString* searchForLabelsBeforeElement(Frame* frame, NSArray* labels, Element* element, size_t* resultDistance, bool* resultIsInCellAbove)
412{
413 ASSERT(element);
414 RegularExpression* regExp = regExpForLabels(labels);
415 // We stop searching after we've seen this many chars
416 constexpr unsigned charsSearchedThreshold = 500;
417 // This is the absolute max we search. We allow a little more slop than
418 // charsSearchedThreshold, to make it more likely that we'll search whole nodes.
419 constexpr unsigned maxCharsSearched = 600;
420 // If the starting element is within a table, the cell that contains it
421 HTMLTableCellElement* startingTableCell = nullptr;
422 bool searchedCellAbove = false;
423
424 if (resultDistance)
425 *resultDistance = notFound;
426 if (resultIsInCellAbove)
427 *resultIsInCellAbove = false;
428
429 // walk backwards in the node tree, until another element, or form, or end of tree
430 unsigned lengthSearched = 0;
431 Node* n;
432 for (n = NodeTraversal::previous(*element); n && lengthSearched < charsSearchedThreshold; n = NodeTraversal::previous(*n)) {
433 if (is<HTMLFormElement>(*n) || is<HTMLFormControlElement>(*n)) {
434 // We hit another form element or the start of the form - bail out
435 break;
436 }
437 if (n->hasTagName(tdTag) && !startingTableCell) {
438 startingTableCell = static_cast<HTMLTableCellElement*>(n);
439 } else if (n->hasTagName(trTag) && startingTableCell) {
440 NSString *result = frame->searchForLabelsAboveCell(*regExp, startingTableCell, resultDistance);
441 if ([result length]) {
442 if (resultIsInCellAbove)
443 *resultIsInCellAbove = true;
444 return result;
445 }
446 searchedCellAbove = true;
447 } else if (n->isTextNode() && n->renderer() && n->renderer()->style().visibility() == Visibility::Visible) {
448 // For each text chunk, run the regexp
449 String nodeString = n->nodeValue();
450 // add 100 for slop, to make it more likely that we'll search whole nodes
451 if (lengthSearched + nodeString.length() > maxCharsSearched)
452 nodeString = nodeString.right(charsSearchedThreshold - lengthSearched);
453 int pos = regExp->searchRev(nodeString);
454 if (pos >= 0) {
455 if (resultDistance)
456 *resultDistance = lengthSearched;
457 return nodeString.substring(pos, regExp->matchedLength());
458 }
459 lengthSearched += nodeString.length();
460 }
461 }
462
463 // If we started in a cell, but bailed because we found the start of the form or the
464 // previous element, we still might need to search the row above us for a label.
465 if (startingTableCell && !searchedCellAbove) {
466 NSString *result = frame->searchForLabelsAboveCell(*regExp, startingTableCell, resultDistance);
467 if ([result length]) {
468 if (resultIsInCellAbove)
469 *resultIsInCellAbove = true;
470 return result;
471 }
472 }
473
474 return nil;
475}

477static NSString *matchLabelsAgainstString(NSArray *labels, const String& stringToMatch)
478{
479 if (stringToMatch.isEmpty())
480 return nil;
481
482 String mutableStringToMatch = stringToMatch;
483
484 // Make numbers and _'s in field names behave like word boundaries, e.g., "address2"
485 replace(mutableStringToMatch, RegularExpression("\\d"), " ");
486 mutableStringToMatch.replace('_', ' ');
487
488 RegularExpression* regExp = regExpForLabels(labels);
489 // Use the largest match we can find in the whole string
490 int pos;
491 int length;
492 int bestPos = -1;
493 int bestLength = -1;
494 int start = 0;
495 do {
496 pos = regExp->match(mutableStringToMatch, start);
497 if (pos != -1) {
498 length = regExp->matchedLength();
499 if (length >= bestLength) {
500 bestPos = pos;
501 bestLength = length;
502 }
503 start = pos + 1;
504 }
505 } while (pos != -1);
506
507 if (bestPos != -1)
508 return mutableStringToMatch.substring(bestPos, bestLength);
509 return nil;
510}

512static NSString *matchLabelsAgainstElement(NSArray *labels, Element* element)
513{
514 if (!element)
515 return nil;
516
517 // Match against the name element, then against the id element if no match is found for the name element.
518 // See 7538330 for one popular site that benefits from the id element check.
519 auto resultFromNameAttribute = matchLabelsAgainstString(labels, element->attributeWithoutSynchronization(nameAttr));
520 if (resultFromNameAttribute.length)
521 return resultFromNameAttribute;
522
523 return matchLabelsAgainstString(labels, element->attributeWithoutSynchronization(idAttr));
524}

532- (NSString *)searchForLabels:(NSArray *)labels beforeElement:(DOMElement *)element resultDistance:(NSUInteger*)outDistance resultIsInCellAbove:(BOOL*)outIsInCellAbove
533{
534 size_t distance;
535 bool isInCellAbove;
536
537 NSString *result = searchForLabelsBeforeElement(core([_private->dataSource webFrame]), labels, core(element), &distance, &isInCellAbove);
538
539 if (outDistance) {
540 if (distance == notFound)
541 *outDistance = NSNotFound;
542 else
543 *outDistance = distance;
544 }
545
546 if (outIsInCellAbove)
547 *outIsInCellAbove = isInCellAbove;
548
549 return result;
550}
