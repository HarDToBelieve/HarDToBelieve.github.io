129- (void)_cancelImmediateAction
130{
131 // Reset the recognizer by turning it off and on again.
132 if ([_immediateActionRecognizer isEnabled]) {
133 [_immediateActionRecognizer setEnabled:NO];
134 [_immediateActionRecognizer setEnabled:YES];
135 }
136
137 [self _clearImmediateActionState];
138 [_webView _clearTextIndicatorWithAnimation:WebCore::TextIndicatorDismissalAnimation::FadeOut];
139}

141- (void)_clearImmediateActionState
142{
143 if (!PAL::isDataDetectorsFrameworkAvailable())
144 return;
145
146 DDActionsManager *actionsManager = [PAL::getDDActionsManagerClass() sharedManager];
147 [actionsManager requestBubbleClosureUnanchorOnFailure:YES];
148
149 if (_currentActionContext && _hasActivatedActionContext) {
150 _hasActivatedActionContext = NO;
151 [PAL::getDDActionsManagerClass() didUseActions];
152 }
153
154 _type = WebImmediateActionNone;
155 _currentActionContext = nil;
156 _currentQLPreviewMenuItem = nil;
157 _contentPreventsDefault = NO;
158}

160- (void)performHitTestAtPoint:(NSPoint)viewPoint
161{
162 using namespace WebCore;
163
164 auto* coreFrame = core([_webView _selectedOrMainFrame]);
165 if (!coreFrame)
166 return;
167
168 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent, HitTestRequest::Type::AllowChildFrameContent };
169 _hitTestResult = coreFrame->eventHandler().hitTestResultAtPoint(WebCore::IntPoint(viewPoint), hitType);
170 coreFrame->mainFrame().eventHandler().setImmediateActionStage(WebCore::ImmediateActionStage::PerformedHitTest);
171
172 if (auto* element = _hitTestResult.targetElement())
173 _contentPreventsDefault = element->dispatchMouseForceWillBegin();
174}

178- (void)immediateActionRecognizerWillPrepare:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
179{
180 if (!_webView)
181 return;
182
183 NSView *documentView = [[[_webView _selectedOrMainFrame] frameView] documentView];
184 if (![documentView isKindOfClass:[WebHTMLView class]]) {
185 [self _cancelImmediateAction];
186 return;
187 }
188
189 if (immediateActionRecognizer != _immediateActionRecognizer)
190 return;
191
192 [_webView _setMaintainsInactiveSelection:YES];
193
194 NSPoint locationInDocumentView = [immediateActionRecognizer locationInView:documentView];
195 [self performHitTestAtPoint:locationInDocumentView];
196 [self _updateImmediateActionItem];
197
198 if (![_immediateActionRecognizer animationController]) {
199 // FIXME: We should be able to remove the dispatch_async when rdar://problem/19502927 is resolved.
200 RunLoop::main().dispatch([self, strongSelf = retainPtr(self)] {
201 [self _cancelImmediateAction];
202 });
203 }
204}

206- (void)immediateActionRecognizerWillBeginAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
207{
208 if (!PAL::isDataDetectorsFrameworkAvailable())
209 return;
210
211 if (immediateActionRecognizer != _immediateActionRecognizer)
212 return;
213
214 if (_currentActionContext) {
215 _hasActivatedActionContext = YES;
216 if (![PAL::getDDActionsManagerClass() shouldUseActionsWithContext:_currentActionContext.get()])
217 [self _cancelImmediateAction];
218 }
219}

221- (void)immediateActionRecognizerDidUpdateAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
222{
223 if (immediateActionRecognizer != _immediateActionRecognizer)
224 return;
225
226 if (WebCore::Frame* coreFrame = [_webView _mainCoreFrame])
227 coreFrame->eventHandler().setImmediateActionStage(WebCore::ImmediateActionStage::ActionUpdated);
228
229 if (_contentPreventsDefault)
230 return;
231
232 [_webView _setTextIndicatorAnimationProgress:[immediateActionRecognizer animationProgress]];
233}
234
235- (void)immediateActionRecognizerDidCancelAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
236{
237 if (immediateActionRecognizer != _immediateActionRecognizer)
238 return;
239
240 if (WebCore::Frame* coreFrame = [_webView _mainCoreFrame]) {
241 WebCore::ImmediateActionStage lastStage = coreFrame->eventHandler().immediateActionStage();
242 if (lastStage == WebCore::ImmediateActionStage::ActionUpdated)
243 coreFrame->eventHandler().setImmediateActionStage(WebCore::ImmediateActionStage::ActionCancelledAfterUpdate);
244 else
245 coreFrame->eventHandler().setImmediateActionStage(WebCore::ImmediateActionStage::ActionCancelledWithoutUpdate);
246 }
247
248 [_webView _setTextIndicatorAnimationProgress:0];
249 [self _clearImmediateActionState];
250 [_webView _clearTextIndicatorWithAnimation:WebCore::TextIndicatorDismissalAnimation::None];
251 [_webView _setMaintainsInactiveSelection:NO];
252}
253
254- (void)immediateActionRecognizerDidCompleteAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
255{
256 if (immediateActionRecognizer != _immediateActionRecognizer)
257 return;
258
259 if (WebCore::Frame* coreFrame = [_webView _mainCoreFrame])
260 coreFrame->eventHandler().setImmediateActionStage(WebCore::ImmediateActionStage::ActionCompleted);
261
262 [_webView _setTextIndicatorAnimationProgress:1];
263 [_webView _setMaintainsInactiveSelection:NO];
264}

268- (id <NSImmediateActionAnimationController>)_defaultAnimationController
269{
270 if (_contentPreventsDefault)
271 return adoptNS([[WebAnimationController alloc] init]).autorelease();
272
273 NSURL *url = _hitTestResult.absoluteLinkURL();
274 String absoluteURLString = [url absoluteString];
275 if (url && _hitTestResult.URLElement()) {
276 if (WTF::protocolIs(absoluteURLString, "mailto")) {
277 _type = WebImmediateActionMailtoLink;
278 return [self _animationControllerForDataDetectedLink];
279 }
280
281 if (WTF::protocolIs(absoluteURLString, "tel")) {
282 _type = WebImmediateActionTelLink;
283 return [self _animationControllerForDataDetectedLink];
284 }
285
286 if (WTF::protocolIsInHTTPFamily(absoluteURLString)) {
287 _type = WebImmediateActionLinkPreview;
288
289 auto linkRange = makeRangeSelectingNodeContents(*_hitTestResult.URLElement());
290 auto indicator = WebCore::TextIndicator::createWithRange(linkRange, { WebCore::TextIndicatorOption::UseBoundingRectAndPaintAllContentForComplexRanges }, WebCore::TextIndicatorPresentationTransition::FadeIn);
291 if (indicator)
292 [_webView _setTextIndicator:*indicator withLifetime:WebCore::TextIndicatorLifetime::Permanent];
293
294 QLPreviewMenuItem *item = [NSMenuItem standardQuickLookMenuItem];
295 item.previewStyle = QLPreviewStylePopover;
296 item.delegate = self;
297 _currentQLPreviewMenuItem = item;
298 return (id <NSImmediateActionAnimationController>)item;
299 }
300 }
301
302 WebCore::Node* node = _hitTestResult.innerNode();
303 if ((node && node->isTextNode()) || _hitTestResult.isOverTextInsideFormControlElement()) {
304 if (auto animationController = [self _animationControllerForDataDetectedText]) {
305 _type = WebImmediateActionDataDetectedItem;
306 return animationController;
307 }
308
309 if (auto animationController = [self _animationControllerForText]) {
310 _type = WebImmediateActionText;
311 return animationController;
312 }
313 }
314
315 return nil;
316}

318- (void)_updateImmediateActionItem
319{
320 _type = WebImmediateActionNone;
321
322 id <NSImmediateActionAnimationController> defaultAnimationController = [self _defaultAnimationController];
323
324 if (_contentPreventsDefault) {
325 [_immediateActionRecognizer setAnimationController:defaultAnimationController];
326 return;
327 }
328
329 // Allow clients the opportunity to override the default immediate action.
330 id customClientAnimationController = nil;
331 if ([[_webView UIDelegate] respondsToSelector:@selector(_webView:immediateActionAnimationControllerForHitTestResult:withType:)]) {
332 RetainPtr<WebElementDictionary> webHitTestResult = adoptNS([[WebElementDictionary alloc] initWithHitTestResult:_hitTestResult]);
333 customClientAnimationController = [(id)[_webView UIDelegate] _webView:_webView immediateActionAnimationControllerForHitTestResult:webHitTestResult.get() withType:_type];
334 }
335
336 if (customClientAnimationController == [NSNull null]) {
337 [self _cancelImmediateAction];
338 return;
339 }
340
341#if PLATFORM(MAC)
342 // FIXME: We should not permanently disable this for iTunes. rdar://problem/19461358
343 if (WebCore::MacApplication::isITunes()) {
344 [self _cancelImmediateAction];
345 return;
346 }
347#endif
348
349 if (customClientAnimationController && [customClientAnimationController conformsToProtocol:@protocol(NSImmediateActionAnimationController)])
350 [_immediateActionRecognizer setAnimationController:(id <NSImmediateActionAnimationController>)customClientAnimationController];
351 else
352 [_immediateActionRecognizer setAnimationController:defaultAnimationController];
353}

381static WebCore::IntRect elementBoundingBoxInWindowCoordinatesFromNode(WebCore::Node* node)
382{
383 if (!node)
384 return { };
385
386 WebCore::Frame* frame = node->document().frame();
387 if (!frame)
388 return { };
389
390 WebCore::FrameView* view = frame->view();
391 if (!view)
392 return { };
393
394 WebCore::RenderObject* renderer = node->renderer();
395 if (!renderer)
396 return { };
397
398 return view->contentsToWindow(renderer->absoluteBoundingBoxRect());
399}

425- (id <NSImmediateActionAnimationController>)_animationControllerForDataDetectedText
426{
427 if (!PAL::isDataDetectorsFrameworkAvailable())
428 return nil;
429
430 std::optional<WebCore::DetectedItem> detectedItem;
431
432 if ([[_webView UIDelegate] respondsToSelector:@selector(_webView:actionContextForHitTestResult:range:)]) {
433 DOMRange *customDataDetectorsRange;
434 auto actionContext = [(id)[_webView UIDelegate] _webView:_webView
435 actionContextForHitTestResult:adoptNS([[WebElementDictionary alloc] initWithHitTestResult:_hitTestResult]).get()
436 range:&customDataDetectorsRange];
437 if (actionContext && customDataDetectorsRange) {
438 detectedItem = { {
439 actionContext,
440 { }, // FIXME: Seems like an empty rect isn't really OK.
441 makeSimpleRange(*core(customDataDetectorsRange))
442 } };
443 }
444 }
445
446 // If the client didn't give us an action context, try to scan around the hit point.
447 if (!detectedItem)
448 detectedItem = WebCore::DataDetection::detectItemAroundHitTestResult(_hitTestResult);
449
450 if (!detectedItem)
451 return nil;
452
453 [detectedItem->actionContext setAltMode:YES];
454 [detectedItem->actionContext setImmediate:YES];
455 if (![[PAL::getDDActionsManagerClass() sharedManager] hasActionsForResult:[detectedItem->actionContext mainResult] actionContext:detectedItem->actionContext.get()])
456 return nil;
457
458 auto indicator = WebCore::TextIndicator::createWithRange(detectedItem->range, { }, WebCore::TextIndicatorPresentationTransition::FadeIn);
459
460 _currentActionContext = [detectedItem->actionContext contextForView:_webView altMode:YES interactionStartedHandler:^() {
461 } interactionChangedHandler:^() {
462 if (indicator)
463 [_webView _setTextIndicator:*indicator withLifetime:WebCore::TextIndicatorLifetime::Permanent];
464 } interactionStoppedHandler:^() {
465 [_webView _clearTextIndicatorWithAnimation:WebCore::TextIndicatorDismissalAnimation::FadeOut];
466 }];
467
468 [_currentActionContext setHighlightFrame:[_webView.window convertRectToScreen:detectedItem->boundingBox]];
469
470 NSArray *menuItems = [[PAL::getDDActionsManagerClass() sharedManager] menuItemsForResult:[_currentActionContext mainResult] actionContext:_currentActionContext.get()];
471 if (menuItems.count != 1)
472 return nil;
473
474 return menuItems.lastObject;
475}

477- (id <NSImmediateActionAnimationController>)_animationControllerForDataDetectedLink
478{
479 if (!PAL::isDataDetectorsFrameworkAvailable())
480 return nil;
481
482 auto actionContext = adoptNS([PAL::allocDDActionContextInstance() init]);
483
484 if (!actionContext)
485 return nil;
486
487 [actionContext setAltMode:YES];
488 [actionContext setImmediate:YES];
489
490 auto linkRange = makeRangeSelectingNodeContents(*_hitTestResult.URLElement());
491 auto indicator = WebCore::TextIndicator::createWithRange(linkRange, { }, WebCore::TextIndicatorPresentationTransition::FadeIn);
492
493 _currentActionContext = [actionContext contextForView:_webView altMode:YES interactionStartedHandler:^() {
494 } interactionChangedHandler:^() {
495 if (indicator)
496 [_webView _setTextIndicator:*indicator withLifetime:WebCore::TextIndicatorLifetime::Permanent];
497 } interactionStoppedHandler:^() {
498 [_webView _clearTextIndicatorWithAnimation:WebCore::TextIndicatorDismissalAnimation::FadeOut];
499 }];
500
501 [_currentActionContext setHighlightFrame:[_webView.window convertRectToScreen:elementBoundingBoxInWindowCoordinatesFromNode(_hitTestResult.URLElement())]];
502
503 NSArray *menuItems = [[PAL::getDDActionsManagerClass() sharedManager] menuItemsForTargetURL:_hitTestResult.absoluteLinkURL().string() actionContext:_currentActionContext.get()];
504 if (menuItems.count != 1)
505 return nil;
506
507 return menuItems.lastObject;
508}

510#pragma mark Text action

512+ (WebCore::DictionaryPopupInfo)_dictionaryPopupInfoForRange:(const WebCore::SimpleRange&)range inFrame:(WebCore::Frame*)frame withLookupOptions:(NSDictionary *)lookupOptions indicatorOptions:(OptionSet<WebCore::TextIndicatorOption>)indicatorOptions transition:(WebCore::TextIndicatorPresentationTransition)presentationTransition
513{
514 auto& editor = frame->editor();
515 editor.setIsGettingDictionaryPopupInfo(true);
516
517 // Dictionary API will accept a whitespace-only string and display UI as if it were real text,
518 // so bail out early to avoid that.
519 WebCore::DictionaryPopupInfo popupInfo;
520 if (plainText(range).stripWhiteSpace().isEmpty()) {
521 editor.setIsGettingDictionaryPopupInfo(false);
522 return popupInfo;
523 }
524
525 auto style = range.start.container->renderStyle();
526 float scaledDescent = style ? style->fontMetrics().descent() * frame->page()->pageScaleFactor() : 0;
527
528 auto quads = WebCore::RenderObject::absoluteTextQuads(range);
529 if (quads.isEmpty()) {
530 editor.setIsGettingDictionaryPopupInfo(false);
531 return popupInfo;
532 }
533
534 auto rangeRect = frame->view()->contentsToWindow(quads[0].enclosingBoundingBox());
535
536 popupInfo.origin = NSMakePoint(rangeRect.x(), rangeRect.y() + scaledDescent);
537 popupInfo.options = lookupOptions;
538
539 auto attributedString = editingAttributedString(range, WebCore::IncludeImages::No).string;
540 auto scaledAttributedString = adoptNS([[NSMutableAttributedString alloc] initWithString:[attributedString string]]);
541 NSFontManager *fontManager = [NSFontManager sharedFontManager];
542
543 [attributedString enumerateAttributesInRange:NSMakeRange(0, [attributedString length]) options:0 usingBlock:^(NSDictionary *attributes, NSRange attributeRange, BOOL *stop) {
544 RetainPtr<NSMutableDictionary> scaledAttributes = adoptNS([attributes mutableCopy]);
545 NSFont *font = [scaledAttributes objectForKey:NSFontAttributeName];
546 if (font)
547 font = [fontManager convertFont:font toSize:font.pointSize * frame->page()->pageScaleFactor()];
548 if (font)
549 [scaledAttributes setObject:font forKey:NSFontAttributeName];
550 [scaledAttributedString addAttributes:scaledAttributes.get() range:attributeRange];
551 }];
552
553 popupInfo.attributedString = scaledAttributedString.get();
554
555 if (auto textIndicator = WebCore::TextIndicator::createWithRange(range, indicatorOptions, presentationTransition))
556 popupInfo.textIndicator = textIndicator->data();
557
558 editor.setIsGettingDictionaryPopupInfo(false);
559 return popupInfo;
560}

562- (id<NSImmediateActionAnimationController>)_animationControllerForText
563{
564 if (!PAL::getLULookupDefinitionModuleClass())
565 return nil;
566
567 auto node = _hitTestResult.innerNode();
568 if (!node)
569 return nil;
570
571 auto frame = node->document().frame();
572 if (!frame)
573 return nil;
574
575 auto range = WebCore::DictionaryLookup::rangeAtHitTestResult(_hitTestResult);
576 if (!range)
577 return nil;
578
579 auto [dictionaryRange, options] = WTFMove(*range);
580 auto dictionaryPopupInfo = [WebImmediateActionController _dictionaryPopupInfoForRange:dictionaryRange inFrame:frame withLookupOptions:options indicatorOptions: { } transition: WebCore::TextIndicatorPresentationTransition::FadeIn];
581 if (!dictionaryPopupInfo.attributedString)
582 return nil;
583
584 return [_webView _animationControllerForDictionaryLookupPopupInfo:dictionaryPopupInfo];
585}
