26#import "WebJSPDFDoc.h"
27
28#import "WebDataSource.h"
29#import "WebDelegateImplementationCaching.h"
30#import "WebFrame.h"
31#import "WebUIDelegate.h"
32#import "WebView.h"
33#import <JavaScriptCore/JSObjectRef.h>
34
35static void jsPDFDocInitialize(JSContextRef ctx, JSObjectRef object)
36{
37 CFRetain(JSObjectGetPrivate(object));
38}
39
40static void jsPDFDocFinalize(JSObjectRef object)
41{
42 CFRelease(JSObjectGetPrivate(object));
43}
44
45static JSValueRef jsPDFDocPrint(JSContextRef ctx, JSObjectRef function, JSObjectRef thisObject, size_t argumentCount, const JSValueRef arguments[], JSValueRef* exception)
46{
47 WebDataSource *dataSource = (__bridge WebDataSource *)JSObjectGetPrivate(thisObject);
48
49 WebView *webView = [[dataSource webFrame] webView];
50 CallUIDelegate(webView, @selector(webView:printFrameView:), [[dataSource webFrame] frameView]);
51
52 return JSValueMakeUndefined(ctx);
53}
54
55static JSStaticFunction jsPDFDocStaticFunctions[] = {
56 { "print", jsPDFDocPrint, kJSPropertyAttributeReadOnly | kJSPropertyAttributeDontDelete },
57 { 0, 0, 0 },
58};
59
60static JSClassDefinition jsPDFDocClassDefinition = {
61 0,
62 kJSClassAttributeNone,
63 "Doc",
64 0,
65 0,
66 jsPDFDocStaticFunctions,
67 jsPDFDocInitialize, jsPDFDocFinalize, 0, 0, 0, 0, 0, 0, 0, 0, 0
68};
69
70JSObjectRef makeJSPDFDoc(JSContextRef ctx, WebDataSource *dataSource)
71{
72 static JSClassRef jsPDFDocClass = JSClassCreate(&jsPDFDocClassDefinition);
73
74 return JSObjectMake(ctx, jsPDFDocClass, (__bridge void*)dataSource);
75}
