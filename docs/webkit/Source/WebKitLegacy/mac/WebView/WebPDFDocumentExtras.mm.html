35static void appendValuesInPDFNameSubtreeToVector(CGPDFDictionaryRef subtree, Vector<CGPDFObjectRef>& values)
36{
37 CGPDFArrayRef names;
38 if (CGPDFDictionaryGetArray(subtree, "Names", &names)) {
39 size_t nameCount = CGPDFArrayGetCount(names) / 2;
40 for (size_t i = 0; i < nameCount; ++i) {
41 CGPDFObjectRef object;
42 CGPDFArrayGetObject(names, 2 * i + 1, &object);
43 values.append(object);
44 }
45 return;
46 }
47
48 CGPDFArrayRef kids;
49 if (!CGPDFDictionaryGetArray(subtree, "Kids", &kids))
50 return;
51
52 size_t kidCount = CGPDFArrayGetCount(kids);
53 for (size_t i = 0; i < kidCount; ++i) {
54 CGPDFDictionaryRef kid;
55 if (!CGPDFArrayGetDictionary(kids, i, &kid))
56 continue;
57 appendValuesInPDFNameSubtreeToVector(kid, values);
58 }
59}

61static void getAllValuesInPDFNameTree(CGPDFDictionaryRef tree, Vector<CGPDFObjectRef>& allValues)
62{
63 appendValuesInPDFNameSubtreeToVector(tree, allValues);
64}

66NSArray *allScriptsInPDFDocument(CGPDFDocumentRef pdfDocument)
67{
68 NSMutableArray *scripts = [NSMutableArray array];
69 if (!pdfDocument)
70 return scripts;
71
72 CGPDFDictionaryRef pdfCatalog = CGPDFDocumentGetCatalog(pdfDocument);
73 if (!pdfCatalog)
74 return scripts;
75
76 // Get the dictionary of all document-level name trees.
77 CGPDFDictionaryRef namesDictionary;
78 if (!CGPDFDictionaryGetDictionary(pdfCatalog, "Names", &namesDictionary))
79 return scripts;
80
81 // Get the document-level "JavaScript" name tree.
82 CGPDFDictionaryRef javaScriptNameTree;
83 if (!CGPDFDictionaryGetDictionary(namesDictionary, "JavaScript", &javaScriptNameTree))
84 return scripts;
85
86 // The names are arbitrary. We are only interested in the values.
87 Vector<CGPDFObjectRef> objects;
88 getAllValuesInPDFNameTree(javaScriptNameTree, objects);
89 size_t objectCount = objects.size();
90
91 for (size_t i = 0; i < objectCount; ++i) {
92 CGPDFDictionaryRef javaScriptAction;
93 if (!CGPDFObjectGetValue(reinterpret_cast<CGPDFObjectRef>(objects[i]), kCGPDFObjectTypeDictionary, &javaScriptAction))
94 continue;
95
96 // A JavaScript action must have an action type of "JavaScript".
97 const char* actionType;
98 if (!CGPDFDictionaryGetName(javaScriptAction, "S", &actionType) || strcmp(actionType, "JavaScript"))
99 continue;
100
101 const UInt8* bytes = nullptr;
102 CFIndex length = 0;
103 CGPDFStreamRef stream;
104 CGPDFStringRef string;
105 RetainPtr<CFDataRef> data;
106 if (CGPDFDictionaryGetStream(javaScriptAction, "JS", &stream)) {
107 CGPDFDataFormat format;
108 data = adoptCF(CGPDFStreamCopyData(stream, &format));
109 if (!data)
110 continue;
111 bytes = CFDataGetBytePtr(data.get());
112 length = CFDataGetLength(data.get());
113 } else if (CGPDFDictionaryGetString(javaScriptAction, "JS", &string)) {
114 bytes = CGPDFStringGetBytePtr(string);
115 length = CGPDFStringGetLength(string);
116 }
117 if (!bytes || !length)
118 continue;
119
120 NSStringEncoding encoding = (length > 1 && bytes[0] == 0xFE && bytes[1] == 0xFF) ? NSUnicodeStringEncoding : NSUTF8StringEncoding;
121 auto script = adoptNS([[NSString alloc] initWithBytes:bytes length:length encoding:encoding]);
122 if (!script)
123 continue;
124
125 [scripts addObject:script.get()];
126 }
127
128 return scripts;
129}
