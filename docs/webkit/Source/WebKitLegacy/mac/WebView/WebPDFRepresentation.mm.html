86- (NSData *)convertPostScriptDataSourceToPDF:(NSData *)data
87{
88 // Convert PostScript to PDF using Quartz 2D API
89 // http://developer.apple.com/documentation/GraphicsImaging/Conceptual/drawingwithquartz2d/dq_ps_convert/chapter_16_section_1.html
90
91 CGPSConverterCallbacks callbacks = { 0, 0, 0, 0, 0, 0, 0, 0 };
92 RetainPtr<CGPSConverterRef> converter = adoptCF(CGPSConverterCreate(0, &callbacks, 0));
93 ASSERT(converter.get());
94
95 RetainPtr<CGDataProviderRef> provider = adoptCF(CGDataProviderCreateWithCFData((CFDataRef)data));
96 ASSERT(provider.get());
97
98 RetainPtr<CFMutableDataRef> result = adoptCF(CFDataCreateMutable(kCFAllocatorDefault, 0));
99 ASSERT(result.get());
100
101 RetainPtr<CGDataConsumerRef> consumer = adoptCF(CGDataConsumerCreateWithCFData(result.get()));
102 ASSERT(consumer.get());
103
104 // Error handled by detecting zero-length 'result' in caller
105 CGPSConverterConvert(converter.get(), provider.get(), consumer.get(), 0);
106
107 return result.bridgingAutorelease();
108}

110- (void)finishedLoadingWithDataSource:(WebDataSource *)dataSource
111{
112 NSData *data = [dataSource data];
113
114 NSArray *postScriptMIMETypes = [[self class] postScriptMIMETypes];
115 NSString *mimeType = [dataSource _responseMIMEType];
116 if ([postScriptMIMETypes containsObject:mimeType]) {
117 data = [self convertPostScriptDataSourceToPDF:data];
118 if ([data length] == 0)
119 return;
120 }
121
122 WebPDFView *view = (WebPDFView *)[[[dataSource webFrame] frameView] documentView];
123 auto document = adoptNS([[[[self class] PDFDocumentClass] alloc] initWithData:data]);
124 [view setPDFDocument:document.get()];
125
126 NSArray *scripts = allScriptsInPDFDocument(document.get());
127 if (![scripts count])
128 return;
129
130 JSGlobalContextRef ctx = JSGlobalContextCreate(0);
131 JSObjectRef jsPDFDoc = makeJSPDFDoc(ctx, dataSource);
132 for (NSString *script in scripts)
133 JSEvaluateScript(ctx, OpaqueJSString::tryCreate(script).get(), jsPDFDoc, nullptr, 0, nullptr);
134 JSGlobalContextRelease(ctx);
135}
