105static void _applicationInfoForMIMEType(NSString *type, NSString **name, NSImage **image)
106{
107 CFURLRef appURL = nullptr;
108
109 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
110 OSStatus error = LSCopyApplicationForMIMEType((__bridge CFStringRef)type, kLSRolesAll, &appURL);
111 ALLOW_DEPRECATED_DECLARATIONS_END
112 if (error != noErr)
113 return;
114
115 NSString *appPath = [(__bridge NSURL *)appURL path];
116 CFRelease(appURL);
117
118 *image = [[NSWorkspace sharedWorkspace] iconForFile:appPath];
119 [*image setSize:NSMakeSize(16.f,16.f)];
120
121 NSString *appName = [[NSFileManager defaultManager] displayNameAtPath:appPath];
122 *name = appName;
123}

125// FIXME 4182876: We can eliminate this function in favor if -isEqual: if [PDFSelection isEqual:] is overridden
126// to compare contents.
127static BOOL _PDFSelectionsAreEqual(PDFSelection *selectionA, PDFSelection *selectionB)
128{
129 NSArray *aPages = [selectionA pages];
130 NSArray *bPages = [selectionB pages];
131
132 if (![aPages isEqual:bPages])
133 return NO;
134
135 int count = [aPages count];
136 int i;
137 for (i = 0; i < count; ++i) {
138 NSRect aBounds = [selectionA boundsForPage:[aPages objectAtIndex:i]];
139 NSRect bBounds = [selectionB boundsForPage:[bPages objectAtIndex:i]];
140 if (!NSEqualRects(aBounds, bBounds)) {
141 return NO;
142 }
143 }
144
145 return YES;
146}

264- (BOOL)becomeFirstResponder
265{
266 // This works together with setNextKeyView to splice our PDFSubview into
267 // the key loop similar to the way NSScrollView does this.
268 NSWindow *window = [self window];
269 id newFirstResponder = nil;
270
271 if ([window keyViewSelectionDirection] == NSSelectingPrevious) {
272 NSView *previousValidKeyView = [self previousValidKeyView];
273 if ((previousValidKeyView != self) && (previousValidKeyView != PDFSubview))
274 newFirstResponder = previousValidKeyView;
275 } else {
276 NSView *PDFDocumentView = [PDFSubview documentView];
277 if ([PDFDocumentView acceptsFirstResponder])
278 newFirstResponder = PDFDocumentView;
279 }
280
281 if (!newFirstResponder)
282 return NO;
283
284 if (![window makeFirstResponder:newFirstResponder])
285 return NO;
286
287 [[dataSource webFrame] _clearSelectionInOtherFrames];
288
289 return YES;
290}

304{
305 self = [super initWithFrame:frame];

314 [self addSubview:PDFSubview];

321 }
322

379- (NSMenu *)menuForEvent:(NSEvent *)theEvent
380{
381 // Start with the menu items supplied by PDFKit, with WebKit tags applied
382 NSMutableArray *items = [self _menuItemsFromPDFKitForEvent:theEvent];
383
384 // Add in an "Open with <default PDF viewer>" item
385 NSString *appName = nil;
386 NSImage *appIcon = nil;
387
388 _applicationInfoForMIMEType([dataSource _responseMIMEType], &appName, &appIcon);
389 if (!appName)
390 appName = UI_STRING_INTERNAL("Finder", "Default application name for Open With context menu");
391
392 // To match the PDFKit style, we'll add Open with Preview even when there's no document yet to view, and
393 // disable it using validateUserInterfaceItem.
394 NSString *title = [NSString stringWithFormat:UI_STRING_INTERNAL("Open with %@", "context menu item for PDF"), appName];
395 auto item = adoptNS([[NSMenuItem alloc] initWithTitle:title action:@selector(_openWithFinder:) keyEquivalent:@""]);
396 [item setTag:WebMenuItemTagOpenWithDefaultApplication];
397 if (appIcon)
398 [item setImage:appIcon];
399 [items insertObject:item.get() atIndex:0];
400
401 [items insertObject:[NSMenuItem separatorItem] atIndex:1];
402
403 // pass the items off to the WebKit context menu mechanism
404 WebView *webView = [[dataSource webFrame] webView];
405 ASSERT(webView);
406 return [webView _menuForElement:[self elementAtPoint:[self convertPoint:[theEvent locationInWindow] fromView:nil]] defaultItems:items];
407}

409- (void)setNextKeyView:(NSView *)aView
410{
411 // This works together with becomeFirstResponder to splice PDFSubview into
412 // the key loop similar to the way NSScrollView and NSClipView do this.
413 NSView *documentView = [PDFSubview documentView];
414 if (documentView) {
415 [documentView setNextKeyView:aView];
416
417 // We need to make the documentView be the next view in the keyview loop.
418 // It would seem more sensible to do this in our init method, but it turns out
419 // that [NSClipView setDocumentView] won't call this method if our next key view
420 // is already set, so we wait until we're called before adding this connection.
421 // We'll also clear it when we're called with nil, so this could go through the
422 // same code path more than once successfully.
423 [super setNextKeyView: aView ? documentView : nil];
424 } else
425 [super setNextKeyView:aView];
426}

640- (NSUInteger)countMatchesForText:(NSString *)string inDOMRange:(DOMRange *)range options:(WebFindOptions)options limit:(NSUInteger)limit markMatches:(BOOL)markMatches
641{
642 if (range && !containsCrossingDocumentBoundaries(makeSimpleRange(*core(range)), *core([dataSource webFrame])->document()))
643 return 0;
644
645 PDFSelection *previousMatch = nil;
646 auto matches = adoptNS([[NSMutableArray alloc] initWithCapacity:limit]);
647
648 for (;;) {
649 PDFSelection *nextMatch = [self _nextMatchFor:string direction:YES caseSensitive:!(options & WebFindOptionsCaseInsensitive) wrap:NO fromSelection:previousMatch startInSelection:NO];
650 if (!nextMatch)
651 break;
652
653 [matches addObject:nextMatch];
654 previousMatch = nextMatch;
655
656 if ([matches count] >= limit)
657 break;
658 }
659
660 [self _setTextMatches:matches.get()];
661
662 return [matches count];
663}

673 NSSet *visiblePages = [self _visiblePDFPages];
674 NSEnumerator *matchEnumerator = [textMatches objectEnumerator];
675 PDFSelection *match;
676
677 while ((match = [matchEnumerator nextObject]) != nil) {
678 NSEnumerator *pages = [[match pages] objectEnumerator];
679 PDFPage *page;
680 while ((page = [pages nextObject]) != nil) {
681
682 // Skip pages that aren't visible (needed for non-continuous modes, see 5362989)
683 if (![visiblePages containsObject:page])
684 continue;
685
686 NSRect selectionOnPageInPDFViewCoordinates = [PDFSubview convertRect:[match boundsForPage:page] fromPage:page];
687 [result addObject:[NSValue valueWithRect:selectionOnPageInPDFViewCoordinates]];
688 }
689 }
690
691 return result;
692}

1076static void removeUselessMenuItemSeparators(NSMutableArray *menuItems)
1077{
1078 // Starting with a mutable array of NSMenuItems, removes any separators at the start,
1079 // removes any separators at the end, and collapses any other adjacent separators to
1080 // a single separator.
1081
1082 // Start this with YES so very last item will be removed if it's a separator.
1083 BOOL removePreviousItemIfSeparator = YES;
1084
1085 for (NSInteger index = menuItems.count - 1; index >= 0; --index) {
1086 NSMenuItem *item = [menuItems objectAtIndex:index];
1087 ASSERT([item isKindOfClass:[NSMenuItem class]]);
1088
1089 BOOL itemIsSeparator = [item isSeparatorItem];
1090 if (itemIsSeparator && (removePreviousItemIfSeparator || !index))
1091 [menuItems removeObjectAtIndex:index];
1092
1093 removePreviousItemIfSeparator = itemIsSeparator;
1094 }
1095
1096 // This could leave us with one initial separator; kill it off too
1097 if (menuItems.count && [[menuItems objectAtIndex:0] isSeparatorItem])
1098 [menuItems removeObjectAtIndex:0];
1099}

1101- (NSMutableArray *)_menuItemsFromPDFKitForEvent:(NSEvent *)theEvent
1102{
1103 NSMutableArray *copiedItems = [NSMutableArray array];
1104
1105IGNORE_WARNINGS_BEGIN("undeclared-selector")
1106 auto actionsToTags = @{
1107 NSStringFromSelector(@selector(_setActualSize:)): @(WebMenuItemPDFActualSize),
1108 NSStringFromSelector(@selector(zoomIn:)): @(WebMenuItemPDFZoomIn),
1109 NSStringFromSelector(@selector(zoomOut:)): @(WebMenuItemPDFZoomOut),
1110 NSStringFromSelector(@selector(_setAutoSize:)): @(WebMenuItemPDFAutoSize),
1111 NSStringFromSelector(@selector(_setSinglePage:)): @(WebMenuItemPDFSinglePage),
1112 NSStringFromSelector(@selector(_setSinglePageScrolling:)): @(WebMenuItemPDFSinglePageScrolling),
1113 NSStringFromSelector(@selector(_setDoublePage:)): @(WebMenuItemPDFFacingPages),
1114 NSStringFromSelector(@selector(_setDoublePageScrolling:)): @(WebMenuItemPDFFacingPagesScrolling),
1115 NSStringFromSelector(@selector(_toggleContinuous:)): @(WebMenuItemPDFContinuous),
1116 NSStringFromSelector(@selector(goToNextPage:)): @(WebMenuItemPDFNextPage),
1117 NSStringFromSelector(@selector(goToPreviousPage:)): @(WebMenuItemPDFPreviousPage),
1118 };
1119IGNORE_WARNINGS_END
1120
1121 // Leave these menu items out, since WebKit inserts equivalent ones. Note that we leave out PDFKit's "Look Up in Dictionary"
1122 // item here because WebKit already includes an item with the same title and purpose. We map WebKit's to PDFKit's
1123 // "Look Up in Dictionary" via the implementation of -[WebPDFView _lookUpInDictionaryFromMenu:].
1124 auto unwantedActions = adoptNS([[NSSet alloc] initWithObjects:
1125IGNORE_WARNINGS_BEGIN("undeclared-selector")
1126 NSStringFromSelector(@selector(_searchInSpotlight:)),
1127 NSStringFromSelector(@selector(_searchInGoogle:)),
1128 NSStringFromSelector(@selector(_searchInDictionary:)),
1129IGNORE_WARNINGS_END
1130 NSStringFromSelector(@selector(copy:)),
1131 nil]);
1132
1133 NSEnumerator *e = [[[PDFSubview menuForEvent:theEvent] itemArray] objectEnumerator];
1134 NSMenuItem *item;
1135 while ((item = [e nextObject]) != nil) {
1136
1137 NSString *actionString = NSStringFromSelector([item action]);
1138
1139 if ([unwantedActions containsObject:actionString])
1140 continue;
1141
1142 // Copy items since a menu item can be in only one menu at a time, and we don't
1143 // want to modify the original menu supplied by PDFKit.
1144 RetainPtr<NSMenuItem> itemCopy = adoptNS([item copy]);
1145 [copiedItems addObject:itemCopy.get()];
1146
1147 // Include all of PDFKit's separators for now. At the end we'll remove any ones that were made
1148 // useless by removing PDFKit's menu items.
1149 if ([itemCopy isSeparatorItem])
1150 continue;
1151
1152 NSNumber *tagNumber = [actionsToTags objectForKey:actionString];
1153
1154 int tag;
1155 if (tagNumber != nil)
1156 tag = [tagNumber intValue];
1157 else {
1158 // This should happen only if PDFKit updates behind WebKit's back. It's non-ideal because clients that only include tags
1159 // that they recognize (like Safari) won't get these PDFKit additions until WebKit is updated to match.
1160 tag = WebMenuItemTagOther;
1161 LOG_ERROR("no WebKit menu item tag found for PDF context menu item action \"%@\", using WebMenuItemTagOther", actionString);
1162 }
1163
1164 if ([itemCopy tag] == 0) {
1165 [itemCopy setTag:tag];
1166 if ([itemCopy target] == PDFSubview) {
1167 // Note that updating the defaults is cheap because it catches redundant settings, so installing
1168 // the proxy for actions that don't impact the defaults is OK
1169 [itemCopy setTarget:PDFSubviewProxy];
1170 }
1171 } else
1172 LOG_ERROR("PDF context menu item %@ came with tag %d, so no WebKit tag was applied. This could mean that the item doesn't appear in clients such as Safari.", [itemCopy title], [itemCopy tag]);
1173 }
1174
1175 // Since we might have removed elements supplied by PDFKit, and we want to minimize our hardwired
1176 // knowledge of the order and arrangement of PDFKit's menu items, we need to remove any bogus
1177 // separators that were left behind.
1178 removeUselessMenuItemSeparators(copiedItems);
1179
1180 return copiedItems;
1181}

1183- (PDFSelection *)_nextMatchFor:(NSString *)string direction:(BOOL)forward caseSensitive:(BOOL)caseFlag wrap:(BOOL)wrapFlag fromSelection:(PDFSelection *)initialSelection startInSelection:(BOOL)startInSelection
1184{
1185 if (![string length])
1186 return nil;
1187
1188 int options = 0;
1189 if (!forward)
1190 options |= NSBackwardsSearch;
1191
1192 if (!caseFlag)
1193 options |= NSCaseInsensitiveSearch;
1194
1195 PDFDocument *document = [PDFSubview document];
1196
1197 auto selectionForInitialSearch = adoptNS([initialSelection copy]);
1198 if (startInSelection) {
1199 // Initially we want to include the selected text in the search. PDFDocument's API always searches from just
1200 // past the passed-in selection, so we need to pass a selection that's modified appropriately.
1201 // FIXME 4182863: Ideally we'd use a zero-length selection at the edge of the current selection, but zero-length
1202 // selections don't work in PDFDocument. So instead we make a one-length selection just before or after the
1203 // current selection, which works for our purposes even when the current selection is at an edge of the
1204 // document.
1205 int initialSelectionLength = [[initialSelection string] length];
1206 if (forward) {
1207 [selectionForInitialSearch extendSelectionAtStart:1];
1208 [selectionForInitialSearch extendSelectionAtEnd:-initialSelectionLength];
1209 } else {
1210 [selectionForInitialSearch extendSelectionAtEnd:1];
1211 [selectionForInitialSearch extendSelectionAtStart:-initialSelectionLength];
1212 }
1213 }
1214 PDFSelection *foundSelection = [document findString:string fromSelection:selectionForInitialSearch.get() withOptions:options];
1215
1216 // If we first searched in the selection, and we found the selection, search again from just past the selection
1217 if (startInSelection && _PDFSelectionsAreEqual(foundSelection, initialSelection))
1218 foundSelection = [document findString:string fromSelection:initialSelection withOptions:options];
1219
1220 if (!foundSelection && wrapFlag) {
1221 auto emptySelection = adoptNS([[[[self class] _PDFViewClass] alloc] initWithDocument:document]);
1222 foundSelection = [document findString:string fromSelection:emptySelection.get() withOptions:options];
1223 }
1224
1225 return foundSelection;
1226}

1253- (NSString *)_path
1254{
1255 // Generate path once.
1256 if (path)
1257 return path;
1258
1259 NSString *filename = [[dataSource response] suggestedFilename];
1260 NSFileManager *manager = [NSFileManager defaultManager];
1261 NSString *temporaryPDFDirectoryPath = [self _temporaryPDFDirectoryPath];
1262
1263 if (!temporaryPDFDirectoryPath) {
1264 // This should never happen; if it does we'll fail silently on non-debug builds.
1265 ASSERT_NOT_REACHED();
1266 return nil;
1267 }
1268
1269 path = [temporaryPDFDirectoryPath stringByAppendingPathComponent:filename];
1270 if ([manager fileExistsAtPath:path]) {
1271 NSString *pathTemplatePrefix = [temporaryPDFDirectoryPath stringByAppendingPathComponent:@"XXXXXX-"];
1272 NSString *pathTemplate = [pathTemplatePrefix stringByAppendingString:filename];
1273 // fileSystemRepresentation returns a const char *; copy it into a char * so we can modify it safely
1274 char *cPath = strdup([pathTemplate fileSystemRepresentation]);
1275 int fd = mkstemps(cPath, strlen(cPath) - strlen([pathTemplatePrefix fileSystemRepresentation]) + 1);
1276 if (fd < 0) {
1277 // Couldn't create a temporary file! Should never happen; if it does we'll fail silently on non-debug builds.
1278 ASSERT_NOT_REACHED();
1279 path = nil;
1280 } else {
1281 close(fd);
1282 path = [manager stringWithFileSystemRepresentation:cPath length:strlen(cPath)];
1283 }
1284 free(cPath);
1285 }
1286
1287 [path retain];
1288
1289 return path;
1290}

1329- (NSAttributedString *)_scaledAttributedString:(NSAttributedString *)unscaledAttributedString
1330{
1331 if (!unscaledAttributedString)
1332 return nil;
1333
1334 float scaleFactor = [PDFSubview scaleFactor];
1335 if (scaleFactor == 1.0)
1336 return unscaledAttributedString;
1337
1338 auto result = adoptNS([unscaledAttributedString mutableCopy]);
1339 unsigned int length = [result length];
1340 NSRange effectiveRange = NSMakeRange(0,0);
1341
1342 [result beginEditing];
1343 while (NSMaxRange(effectiveRange) < length) {
1344 NSFont *unscaledFont = [result attribute:NSFontAttributeName atIndex:NSMaxRange(effectiveRange) effectiveRange:&effectiveRange];
1345
1346 if (!unscaledFont) {
1347 // FIXME: We can't scale the font if we don't know what it is. We should always know what it is,
1348 // but sometimes don't due to PDFKit issue 5089411. When that's addressed, we can remove this
1349 // early continue.
1350 LOG_ERROR("no font attribute found in range %@ for attributed string \"%@\" on page %@ (see radar 5089411)", NSStringFromRange(effectiveRange), result.get(), [[dataSource request] URL]);
1351 continue;
1352 }
1353
1354 NSFont *scaledFont = [NSFont fontWithName:[unscaledFont fontName] size:[unscaledFont pointSize]*scaleFactor];
1355 [result addAttribute:NSFontAttributeName value:scaledFont range:effectiveRange];
1356 }
1357 [result endEditing];
1358
1359 return result.autorelease();
1360}

1369- (NSString *)_temporaryPDFDirectoryPath
1370{
1371 // Returns nil if the temporary PDF directory didn't exist and couldn't be created
1372
1373 static NSString *_temporaryPDFDirectoryPath = nil;
1374
1375 if (!_temporaryPDFDirectoryPath) {

1384 // so only the current user can add to it or view its contents.

1435- (NSSet *)_visiblePDFPages
1436{
1437 // Returns the set of pages that are at least partly visible, used to avoid processing non-visible pages
1438 PDFDocument *pdfDocument = [PDFSubview document];
1439 if (!pdfDocument)
1440 return nil;
1441
1442 NSRect pdfViewBounds = [PDFSubview bounds];
1443 PDFPage *topLeftPage = [PDFSubview pageForPoint:NSMakePoint(NSMinX(pdfViewBounds), NSMaxY(pdfViewBounds)) nearest:YES];
1444 PDFPage *bottomRightPage = [PDFSubview pageForPoint:NSMakePoint(NSMaxX(pdfViewBounds), NSMinY(pdfViewBounds)) nearest:YES];
1445
1446 // only page-free documents should return nil for either of these two since we passed YES for nearest:
1447 if (!topLeftPage) {
1448 ASSERT(!bottomRightPage);
1449 return nil;
1450 }
1451
1452 NSUInteger firstVisiblePageIndex = [pdfDocument indexForPage:topLeftPage];
1453 NSUInteger lastVisiblePageIndex = [pdfDocument indexForPage:bottomRightPage];
1454
1455 if (firstVisiblePageIndex > lastVisiblePageIndex) {
1456 NSUInteger swap = firstVisiblePageIndex;
1457 firstVisiblePageIndex = lastVisiblePageIndex;
1458 lastVisiblePageIndex = swap;
1459 }
1460
1461 NSMutableSet *result = [NSMutableSet set];
1462 NSUInteger pageIndex;
1463 for (pageIndex = firstVisiblePageIndex; pageIndex <= lastVisiblePageIndex; ++pageIndex)
1464 [result addObject:[pdfDocument pageAtIndex:pageIndex]];
1465
1466 return result;
1467}
