93template<unsigned size> static bool contains(const char* const (&array)[size], const char* item)
94{
95 if (!item)
96 return false;
97 for (auto* string : array) {
98 if (equalIgnoringASCIICase(string, item))
99 return true;
100 }
101 return false;
102}

232{

254- (instancetype)initWithIdentifier:(NSString *)anIdentifier sendChangeNotification:(BOOL)sendChangeNotification
255#else
256- (instancetype)initWithIdentifier:(NSString *)anIdentifier
257#endif
258{
259 WebPreferences *instance = [[self class] _getInstanceForIdentifier:anIdentifier];
260 if (instance) {
261 [self release];
262 return [instance retain];
263 }
264
265 self = [super init];
266 if (!self)
267 return nil;
268
269 _private = new WebPreferencesPrivate;
270 _private->values = adoptNS([[NSMutableDictionary alloc] init]);
271 _private->identifier = adoptNS([anIdentifier copy]);
272 _private->automaticallyDetectsCacheModel = YES;
273
274 [[self class] _setInstance:self forIdentifier:_private->identifier.get()];
275
276 [self _updatePrivateBrowsingStateTo:[self privateBrowsingEnabled]];
277
278#if PLATFORM(IOS_FAMILY)
279 if (sendChangeNotification) {
280 [self _postPreferencesChangedNotification];
281 [self _postCacheModelChangedNotification];
282 }
283#else
284 [self _postPreferencesChangedNotification];
285 [self _postCacheModelChangedNotification];
286#endif
287
288 return self;
289}

291- (instancetype)initWithCoder:(NSCoder *)decoder
292{
293 self = [super init];
294 if (!self)
295 return nil;
296
297 _private = new WebPreferencesPrivate;
298 _private->automaticallyDetectsCacheModel = YES;
299
300 @try {
301 id identifier = nil;
302 id values = nil;
303 if ([decoder allowsKeyedCoding]) {
304 identifier = [decoder decodeObjectForKey:@"Identifier"];
305 values = [decoder decodeObjectForKey:@"Values"];
306 } else {
307 int version;
308 [decoder decodeValueOfObjCType:@encode(int) at:&version size:sizeof(int)];
309 if (version == 1) {
310 identifier = [decoder decodeObject];
311 values = [decoder decodeObject];
312 }
313 }
314
315 if ([identifier isKindOfClass:[NSString class]])
316 _private->identifier = adoptNS([identifier copy]);
317 if ([values isKindOfClass:[NSDictionary class]])
318 _private->values = adoptNS([values mutableCopy]); // ensure dictionary is mutable
319
320 LOG(Encoding, "Identifier = %@, Values = %@\n", _private->identifier.get(), _private->values.get());
321 } @catch(id) {
322 [self release];
323 return nil;
324 }
325
326 // If we load a nib multiple times, or have instances in multiple
327 // nibs with the same name, the first guy up wins.
328 WebPreferences *instance = [[self class] _getInstanceForIdentifier:_private->identifier.get()];
329 if (instance) {
330 [self release];
331 self = [instance retain];
332 } else {
333 [[self class] _setInstance:self forIdentifier:_private->identifier.get()];
334 [self _updatePrivateBrowsingStateTo:[self privateBrowsingEnabled]];
335 }
336
337 return self;
338}

340- (void)encodeWithCoder:(NSCoder *)encoder
341{
342 if ([encoder allowsKeyedCoding]){
343 [encoder encodeObject:_private->identifier.get() forKey:@"Identifier"];
344#if PLATFORM(IOS_FAMILY)
345 dispatch_sync(_private->readWriteQueue.get(), ^{
346#endif
347 [encoder encodeObject:_private->values.get() forKey:@"Values"];
348 LOG (Encoding, "Identifier = %@, Values = %@\n", _private->identifier.get(), _private->values.get());
349#if PLATFORM(IOS_FAMILY)
350 });
351#endif
352 }
353 else {
354 int version = WebPreferencesVersion;
355 [encoder encodeValueOfObjCType:@encode(int) at:&version];
356 [encoder encodeObject:_private->identifier.get()];
357#if PLATFORM(IOS_FAMILY)
358 dispatch_sync(_private->readWriteQueue.get(), ^{
359#endif
360 [encoder encodeObject:_private->values.get()];
361#if PLATFORM(IOS_FAMILY)
362 });
363#endif
364 }

459- (id)_valueForKey:(NSString *)key
460{
461 NSString *_key = KEY(key);
462#if PLATFORM(IOS_FAMILY)
463 __block id o = nil;
464 dispatch_sync(_private->readWriteQueue.get(), ^{
465 o = [_private->values.get() objectForKey:_key];
466 });
467#else
468 id o = [_private->values.get() objectForKey:_key];
469#endif
470 if (o)
471 return o;
472 o = [[NSUserDefaults standardUserDefaults] objectForKey:_key];
473 if (!o && key != _key)
474 o = [[NSUserDefaults standardUserDefaults] objectForKey:key];
475 return o;
476}

484- (void)_setStringValue:(NSString *)value forKey:(NSString *)key
485{
486 if ([[self _stringValueForKey:key] isEqualToString:value])
487 return;
488 NSString *_key = KEY(key);
489#if PLATFORM(IOS_FAMILY)
490 dispatch_barrier_sync(_private->readWriteQueue.get(), ^{
491#endif
492 [_private->values.get() setObject:value forKey:_key];
493#if PLATFORM(IOS_FAMILY)
494 });
495#endif
496 if (_private->autosaves)
497 [[NSUserDefaults standardUserDefaults] setObject:value forKey:_key];
498 [self _postPreferencesChangedNotification];
499}

502{
503 id value = [self _valueForKey:key];
504 if (![value isKindOfClass:[NSArray class]])
505 return nil;
506
507 NSArray *array = (NSArray *)value;
508 for (id object in array) {
509 if (![object isKindOfClass:[NSString class]])
510 return nil;
511 }
512
513 return (NSArray<NSString *> *)array;

978- (void)_updatePrivateBrowsingStateTo:(BOOL)enabled
979{
980 if (!_private) {
981 ASSERT(!enabled);
982 return;
983 }
984
985 if (enabled == _private->inPrivateBrowsing)
986 return;
987 if (enabled > _private->inPrivateBrowsing) {
988 WebFrameNetworkingContext::ensurePrivateBrowsingSession();
989 ++webPreferencesInstanceCountWithPrivateBrowsingEnabled;
990 } else {
991 ASSERT(webPreferencesInstanceCountWithPrivateBrowsingEnabled);
992 --webPreferencesInstanceCountWithPrivateBrowsingEnabled;
993 if (!webPreferencesInstanceCountWithPrivateBrowsingEnabled)
994 WebFrameNetworkingContext::destroyPrivateBrowsingSession();
995 }
996 _private->inPrivateBrowsing = enabled;
