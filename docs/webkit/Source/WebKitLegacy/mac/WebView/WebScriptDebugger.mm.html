50static NSString *toNSString(JSC::SourceProvider* sourceProvider)
51{
52 const String& sourceString = sourceProvider->source().toString();
53 if (sourceString.isEmpty())
54 return nil;
55 return sourceString;
56}

58static WebFrame *toWebFrame(JSC::JSGlobalObject* globalObject)
59{
60 WebCore::JSDOMWindow* window = static_cast<WebCore::JSDOMWindow*>(globalObject);
61 return kit(window->wrapped().frame());
62}

64WebScriptDebugger::WebScriptDebugger(JSC::JSGlobalObject* globalObject)
65 : Debugger(globalObject->vm())
66 , m_callingDelegate(false)
67 , m_globalObject(globalObject->vm(), globalObject)
68{
69 setPauseOnAllExceptionsBreakpoint(JSC::Breakpoint::create(JSC::noBreakpointID));
70 deactivateBreakpoints();
71 attach(globalObject);
72}

74// callbacks - relay to delegate
75void WebScriptDebugger::sourceParsed(JSC::JSGlobalObject* lexicalGlobalObject, JSC::SourceProvider* sourceProvider, int errorLine, const String& errorMsg)
76{
77 if (m_callingDelegate)
78 return;
79
80 m_callingDelegate = true;
81
82 NSString *nsSource = toNSString(sourceProvider);
83 NSURL *nsURL = sourceProvider->sourceOrigin().url();
84 int firstLine = sourceProvider->startPosition().m_line.oneBasedInt();
85
86 JSC::VM& vm = lexicalGlobalObject->vm();
87 WebFrame *webFrame = toWebFrame(vm.deprecatedVMEntryGlobalObject(lexicalGlobalObject));
88 WebView *webView = [webFrame webView];
89 WebScriptDebugDelegateImplementationCache* implementations = WebViewGetScriptDebugDelegateImplementations(webView);
90
91 if (errorLine == -1) {
92 if (implementations->didParseSourceFunc) {
93 if (implementations->didParseSourceExpectsBaseLineNumber)
94 CallScriptDebugDelegate(implementations->didParseSourceFunc, webView, @selector(webView:didParseSource:baseLineNumber:fromURL:sourceId:forWebFrame:), nsSource, firstLine, nsURL, sourceProvider->asID(), webFrame);
95 else
96 CallScriptDebugDelegate(implementations->didParseSourceFunc, webView, @selector(webView:didParseSource:fromURL:sourceId:forWebFrame:), nsSource, [nsURL absoluteString], sourceProvider->asID(), webFrame);
97 }
98 } else {
99 NSDictionary *info;
100 if (errorMsg.isEmpty()) {
101 info = @{
102 WebScriptErrorLineNumberKey: @(errorLine),
103 };
104 } else {
105 info = @{
106 WebScriptErrorDescriptionKey: (NSString *)errorMsg,
107 WebScriptErrorLineNumberKey: @(errorLine),
108 };
109 }
110 auto error = adoptNS([[NSError alloc] initWithDomain:WebScriptErrorDomain code:WebScriptGeneralErrorCode userInfo:info]);
111
112 if (implementations->failedToParseSourceFunc)
113 CallScriptDebugDelegate(implementations->failedToParseSourceFunc, webView, @selector(webView:failedToParseSource:baseLineNumber:fromURL:withError:forWebFrame:), nsSource, firstLine, nsURL, error.get(), webFrame);
114 }
115
116 m_callingDelegate = false;
117}

119void WebScriptDebugger::handlePause(JSC::JSGlobalObject* globalObject, Debugger::ReasonForPause reason)
120{
121 if (m_callingDelegate)
122 return;
123
124 if (reason != Debugger::PausedForException)
125 return;
126
127 m_callingDelegate = true;
128
129 WebFrame *webFrame = toWebFrame(globalObject);
130 WebView *webView = [webFrame webView];
131 JSC::DebuggerCallFrame& debuggerCallFrame = currentDebuggerCallFrame();
132 JSC::JSValue exceptionValue = currentException();
133 String functionName = debuggerCallFrame.functionName();
134 RetainPtr<WebScriptCallFrame> webCallFrame = adoptNS([[WebScriptCallFrame alloc] _initWithGlobalObject:core(webFrame)->script().windowScriptObject() functionName:functionName exceptionValue:exceptionValue]);
135
136 WebScriptDebugDelegateImplementationCache* cache = WebViewGetScriptDebugDelegateImplementations(webView);
137 if (cache->exceptionWasRaisedFunc) {
138 if (cache->exceptionWasRaisedExpectsHasHandlerFlag) {
139 bool hasHandler = hasHandlerForExceptionCallback();
140 CallScriptDebugDelegate(cache->exceptionWasRaisedFunc, webView, @selector(webView:exceptionWasRaised:hasHandler:sourceId:line:forWebFrame:), webCallFrame.get(), hasHandler, debuggerCallFrame.sourceID(), debuggerCallFrame.line(), webFrame);
141 } else
142 CallScriptDebugDelegate(cache->exceptionWasRaisedFunc, webView, @selector(webView:exceptionWasRaised:sourceId:line:forWebFrame:), webCallFrame.get(), debuggerCallFrame.sourceID(), debuggerCallFrame.line(), webFrame);
143 }
144
145 m_callingDelegate = false;
146}
