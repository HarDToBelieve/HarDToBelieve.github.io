26#import <Foundation/Foundation.h>
27
28@class DOMRange;
29@class DOMNode;
30@class WebTextIteratorPrivate;
31
32@interface WebTextIterator : NSObject {
33@private
34 WebTextIteratorPrivate *_private;
35}
36
37- (id)initWithRange:(DOMRange *)range;
38
39/*!
40 @method advance
41 @abstract Moves the WebTextIterator to the next bit of text or boundary between runs of text.
42 The iterator can break up runs of text however it finds convenient, so clients need to handle
43 text runs that are broken up into arbitrary pieces.
44 */
45- (void)advance;
46
47/*!
48 @method atEnd
49 @result YES if the WebTextIterator has reached the end of the DOMRange.
50 */
51- (BOOL)atEnd;
52
53/*!
54 @method currentTextLength
55 @result Length of the current text. Length of zero means that the iterator is at a boundary,
56 such as an image, that separates runs of text.
57 */
58- (NSUInteger)currentTextLength;
59
60/*!
61 @method currentTextPointer
62 @result A pointer to the current text. Like the WebTextIterator itself, the pointer becomes
63 invalid after any modification is made to the document; it must be used before the document
64 is changed or the iterator is advanced.
65 */
66- (const unichar *)currentTextPointer;
67
68/*!
69 @method currentRange
70 @abstract A function that identifies the specific document range that text corresponds to.
71 This can be quite costly to compute for non-text items, so when possible this should only
72 be called once the caller has determined that the text is text it wants to process. If you
73 call currentRange every time you advance the iterator, performance will be extremely slow
74 due to the cost of computing a DOM range.
75 @result A DOM range indicating the position within the document of the current text.
76 */
77- (DOMRange *)currentRange;
78
79@end
80
81@interface WebTextIterator (WebTextIteratorDeprecated)
82
83/*!
84 @method currentNode
85 @abstract A convenience method that finds the first node in currentRange; it's almost always better to use currentRange instead.
86 @result The current DOMNode in the WebTextIterator
87 */
88- (DOMNode *)currentNode;
89
90/*!
91 @method currentText
92 @abstract A convenience method that makes an NSString out of the current text; it's almost always better to use currentTextPointer and currentTextLength instead.
93 @result The current text in the WebTextIterator.
94 */
95- (NSString *)currentText;
96
97@end
