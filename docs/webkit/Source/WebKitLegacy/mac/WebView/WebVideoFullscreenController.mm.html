77static id<WebAVPlayerViewDelegate> WebAVPlayerView_webDelegate(id aSelf, SEL)
78{
79 void* webDelegate = nil;
80 object_getInstanceVariable(aSelf, "_webDelegate", &webDelegate);
81 return static_cast<id<WebAVPlayerViewDelegate>>(webDelegate);
82}
83
84static void WebAVPlayerView_setWebDelegate(id aSelf, SEL, id<WebAVPlayerViewDelegate> webDelegate)
85{
86 object_setInstanceVariable(aSelf, "_webDelegate", webDelegate);
87}
88
89static BOOL WebAVPlayerView_isFullScreen(id aSelf, SEL)
90{
91 WebAVPlayerView *playerView = aSelf;
92 return [playerView.webDelegate playerViewIsFullScreen:playerView];
93}
94
95static void WebAVPlayerView_enterFullScreen(id aSelf, SEL, id sender)
96{
97 WebAVPlayerView *playerView = aSelf;
98 [playerView.webDelegate playerViewRequestEnterFullscreen:playerView];
99}

101static void WebAVPlayerView_exitFullScreen(id aSelf, SEL, id sender)
102{
103 WebAVPlayerView *playerView = aSelf;
104 [playerView.webDelegate playerViewRequestExitFullscreen:playerView];
105}

107static WebAVPlayerView *allocWebAVPlayerViewInstance()
108{
109 static Class theClass = nil;
110 static dispatch_once_t onceToken;
111 dispatch_once(&onceToken, ^{
112 ASSERT(getAVPlayerViewClass());
113 Class aClass = objc_allocateClassPair(getAVPlayerViewClass(), "WebAVPlayerView", 0);
114 theClass = aClass;
115 class_addMethod(theClass, @selector(setWebDelegate:), (IMP)WebAVPlayerView_setWebDelegate, "v@:@");
116 class_addMethod(theClass, @selector(webDelegate), (IMP)WebAVPlayerView_webDelegate, "@@:");
117 class_addMethod(theClass, @selector(isFullScreen), (IMP)WebAVPlayerView_isFullScreen, "B@:");
118 class_addMethod(theClass, @selector(enterFullScreen:), (IMP)WebAVPlayerView_enterFullScreen, "v@:@");
119 class_addMethod(theClass, @selector(exitFullScreen:), (IMP)WebAVPlayerView_exitFullScreen, "v@:@");
120
121 class_addIvar(theClass, "_webDelegate", sizeof(id), log2(sizeof(id)), "@");
122 class_addIvar(theClass, "_webIsFullScreen", sizeof(BOOL), log2(sizeof(BOOL)), "B");
123
124 objc_registerClassPair(theClass);
125 });
126 return (WebAVPlayerView *)[theClass alloc];
127}

129@interface WebVideoFullscreenController () <WebAVPlayerViewDelegate, NSWindowDelegate> {
130 RefPtr<WebCore::PlaybackSessionModelMediaElement> _playbackModel;
131 RefPtr<WebCore::PlaybackSessionInterfaceAVKit> _playbackInterface;
132 RetainPtr<NSView> _contentOverlay;
133 BOOL _isFullScreen;
134}
135@property (readonly) WebCoreFullScreenWindow* fullscreenWindow;
136@property (readonly) WebAVPlayerView* playerView;
137@end
138
139@implementation WebVideoFullscreenController
140
141- (id)init
142{
143 // Do not defer window creation, to make sure -windowNumber is created (needed by WebWindowScaleAnimation).
144 auto window = adoptNS([[WebCoreFullScreenWindow alloc] initWithContentRect:NSZeroRect styleMask:(NSWindowStyleMaskFullSizeContentView | NSWindowStyleMaskResizable) backing:NSBackingStoreBuffered defer:NO]);
145 [window setCollectionBehavior:([window collectionBehavior] | NSWindowCollectionBehaviorFullScreenPrimary)];
146 [window setDelegate: self];
147 self = [super initWithWindow:window.get()];
148 if (!self)
149 return nil;
150 _playbackModel = WebCore::PlaybackSessionModelMediaElement::create();
151 _playbackInterface = WebCore::PlaybackSessionInterfaceAVKit::create(*_playbackModel);
152 _contentOverlay = adoptNS([[NSView alloc] initWithFrame:NSZeroRect]);
153 _contentOverlay.get().layerContentsRedrawPolicy = NSViewLayerContentsRedrawNever;
154 _contentOverlay.get().layer = adoptNS([[WebVideoFullscreenOverlayLayer alloc] init]).get();
155 [_contentOverlay setWantsLayer:YES];
156 [_contentOverlay setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable)];
157 [self windowDidLoad];
158
159 return self;
160}
161- (void)dealloc
162{
163 ASSERT(!_backgroundFullscreenWindow);
164 ASSERT(!_fadeAnimation);
165 _playerView.webDelegate = nil;
166 _playbackModel = nil;
167 [super dealloc];
168}
169
170- (WebCoreFullScreenWindow *)fullscreenWindow
171{
172 return (WebCoreFullScreenWindow *)[super window];
173}
174
175- (void)windowDidLoad
176{
177 auto window = [self fullscreenWindow];
178
179 [window setHasShadow:YES]; // This is nicer with a shadow.
180 [window setLevel:NSPopUpMenuWindowLevel-1];
181
182 _playerView = [allocWebAVPlayerViewInstance() initWithFrame:window.contentLayoutRect];
183 _playerView.controlsStyle = AVPlayerViewControlsStyleNone;
184 _playerView.showsFullScreenToggleButton = YES;
185 _playerView.showsAudioOnlyIndicatorView = NO;
186 _playerView.webDelegate = self;
187 window.contentView = _playerView;
188 [_contentOverlay setFrame:_playerView.contentOverlayView.bounds];
189 [_playerView.contentOverlayView addSubview:_contentOverlay.get()];
190
191 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(applicationDidResignActive:) name:NSApplicationDidResignActiveNotification object:NSApp];
192}

194- (NakedPtr<WebCore::HTMLVideoElement>)videoElement

196 return _videoElement.get();

199// FIXME: This method is not really a setter. The caller relies on its side effects, and it's
200// called once each time we enter full screen. So it should have a different name.
201- (void)setVideoElement:(NakedPtr<WebCore::HTMLVideoElement>)videoElement
202{
203 ASSERT(videoElement);
204 _videoElement = videoElement;
205
206 if (![self isWindowLoaded])
207 return;
208
209 _playbackModel->setMediaElement(videoElement);
210 self.playerView.playerController = (AVPlayerController*)_playbackInterface->playerController();
211}
212
213- (void)enterFullscreen:(NSScreen *)screen
214{
215 if (!_videoElement)
216 return;
217 [NSAnimationContext beginGrouping];
218 _videoElement->setVideoFullscreenLayer(_contentOverlay.get().layer, [self, protectedSelf = retainPtr(self)] {
219 [self.fullscreenWindow setFrame:self.videoElementRect display:YES];
220 [self.fullscreenWindow makeKeyAndOrderFront:self];
221 [self.fullscreenWindow enterFullScreenMode:self];
222 [NSAnimationContext endGrouping];
223 });
224}
225
226- (void)exitFullscreen
227{
228 [self.fullscreenWindow exitFullScreenMode:self];
229}

231- (NSRect)videoElementRect

233 return _videoElement->screenRect();

236- (void)applicationDidResignActive:(NSNotification*)notification
237{
238 UNUSED_PARAM(notification);
239 NSWindow* fullscreenWindow = [self fullscreenWindow];
240
241 // Replicate the QuickTime Player (X) behavior when losing active application status:
242 // Is the fullscreen screen the main screen? (Note: this covers the case where only a
243 // single screen is available.) Is the fullscreen screen on the current space? IFF so,
244 // then exit fullscreen mode.
245 if (fullscreenWindow.screen == [NSScreen screens][0] && fullscreenWindow.onActiveSpace)
246 [self _requestExit];
247}
248
249- (void)_requestExit
250{
251 [self.fullscreenWindow exitFullScreenMode:self];
252}

254- (void)_requestEnter
255{
256 if (_videoElement)

258}

308- (void)windowDidEnterFullScreen:(NSNotification *)notification
309{
310 _playerView.controlsStyle = AVPlayerViewControlsStyleFloating;
311 [_playerView willChangeValueForKey:@"isFullScreen"];
312 _isFullScreen = YES;
313 [_playerView didChangeValueForKey:@"isFullScreen"];
314 if (_videoElement)
315 _videoElement->didBecomeFullscreenElement();
316}
317
318- (void)windowWillExitFullScreen:(NSNotification *)notification
319{
320 _playerView.controlsStyle = AVPlayerViewControlsStyleNone;
321}
322
323- (void)windowDidExitFullScreen:(NSNotification *)notification
324{
325 [_playerView willChangeValueForKey:@"isFullScreen"];
326 _isFullScreen = NO;
327 [_playerView didChangeValueForKey:@"isFullScreen"];
328
329 if (!_videoElement) {
330 [self.fullscreenWindow close];
331 return;
332 }
333
334 [NSAnimationContext beginGrouping];
335 _videoElement->setVideoFullscreenLayer(nil, [self, protectedSelf = retainPtr(self)] {
336 [self.fullscreenWindow close];
337 [NSAnimationContext endGrouping];
338 });
339
340 if (_videoElement->isFullscreen())
341 _videoElement->exitFullscreen();
342}
