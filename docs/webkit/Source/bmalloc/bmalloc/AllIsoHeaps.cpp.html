28namespace bmalloc {
29
30DEFINE_STATIC_PER_PROCESS_STORAGE(AllIsoHeaps);
31
32AllIsoHeaps::AllIsoHeaps(const LockHolder&)
33{
34}
35
36void AllIsoHeaps::add(IsoHeapImplBase* heap)
37{
38 LockHolder locker(mutex());
39 heap->m_next = m_head;
40 m_head = heap;
41}
42
43IsoHeapImplBase* AllIsoHeaps::head()
44{
45 LockHolder locker(mutex());
46 return m_head;
47}
