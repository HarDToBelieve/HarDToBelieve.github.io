35namespace bmalloc {
36
37class AllIsoHeaps;
38
39class BEXPORT IsoHeapImplBase {
40 MAKE_BMALLOCED;
41 IsoHeapImplBase(const IsoHeapImplBase&) = delete;
42 IsoHeapImplBase& operator=(const IsoHeapImplBase&) = delete;
43public:
44 static constexpr unsigned maxAllocationFromShared = 8;
45 static constexpr unsigned maxAllocationFromSharedMask = (1U << maxAllocationFromShared) - 1U;
46 static_assert(maxAllocationFromShared <= bmalloc::alignment, "");
47 static_assert(isPowerOfTwo(maxAllocationFromShared), "");
48
49 virtual ~IsoHeapImplBase();
50
51 virtual void scavenge(Vector<DeferredDecommit>&) = 0;
52#if BUSE(PARTIAL_SCAVENGE)
53 virtual void scavengeToHighWatermark(Vector<DeferredDecommit>&) = 0;
54#endif
55
56 void scavengeNow();
57 static void finishScavenging(Vector<DeferredDecommit>&);
58
59 void didCommit(void* ptr, size_t bytes);
60 void didDecommit(void* ptr, size_t bytes);
61
62 void isNowFreeable(void* ptr, size_t bytes);
63 void isNoLongerFreeable(void* ptr, size_t bytes);
64
65 size_t freeableMemory();
66 size_t footprint();
67
68 void addToAllIsoHeaps();
69
70protected:
71 IsoHeapImplBase(Mutex&);
72
73 friend class IsoSharedPage;
74 friend class AllIsoHeaps;
75
76public:
77 // It's almost always the caller's responsibility to grab the lock. This lock comes from the
78 // (*PerProcess<IsoTLSEntryHolder<IsoTLSDeallocatorEntry<Config>>>::get())->lock. That's pretty weird, and we don't
79 // try to disguise the fact that it's weird. We only do that because heaps in the same size class
80 // share the same deallocator log, so it makes sense for them to also share the same lock to
81 // amortize lock acquisition costs.
82 Mutex& lock;
83protected:
84 IsoHeapImplBase* m_next { nullptr };
85 std::chrono::steady_clock::time_point m_lastSlowPathTime;
86 size_t m_footprint { 0 };
87 size_t m_freeableMemory { 0 };
88#if ENABLE_PHYSICAL_PAGE_MAP
89 PhysicalPageMap m_physicalPageMap;
90#endif
91 std::array<PackedAlignedPtr<uint8_t, bmalloc::alignment>, maxAllocationFromShared> m_sharedCells { };
92protected:
93 unsigned m_numberOfAllocationsFromSharedInOneCycle { 0 };
94 unsigned m_availableShared { maxAllocationFromSharedMask };
95 AllocationMode m_allocationMode { AllocationMode::Init };
96 bool m_isInlineDirectoryEligibleOrDecommitted { true };
97 static_assert(sizeof(m_availableShared) * 8 >= maxAllocationFromShared, "");
98};
99
100template<typename Config>
101class IsoHeapImpl final : public IsoHeapImplBase {
102 // Pick a size that makes us most efficiently use the bitvectors.
103 static constexpr unsigned numPagesInInlineDirectory = 32;
104
105public:
106 IsoHeapImpl();
107
108 EligibilityResult<Config> takeFirstEligible(const LockHolder&);
109
110 // Callbacks from directory.
111 void didBecomeEligibleOrDecommited(const LockHolder&, IsoDirectory<Config, numPagesInInlineDirectory>*);
112 void didBecomeEligibleOrDecommited(const LockHolder&, IsoDirectory<Config, IsoDirectoryPage<Config>::numPages>*);
113
114 void scavenge(Vector<DeferredDecommit>&) override;
115#if BUSE(PARTIAL_SCAVENGE)
116 void scavengeToHighWatermark(Vector<DeferredDecommit>&) override;
117#endif
118
119 unsigned allocatorOffset();
120 unsigned deallocatorOffset();
121
122 // White-box testing functions.
123 unsigned numLiveObjects();
124 unsigned numCommittedPages();
125
126 template<typename Func>
127 void forEachDirectory(const LockHolder&, const Func&);
128
129 template<typename Func>
130 void forEachCommittedPage(const LockHolder&, const Func&);
131
132 // This is only accurate when all threads are scavenged. Otherwise it will overestimate.
133 template<typename Func>
134 void forEachLiveObject(const LockHolder&, const Func&);
135
136 AllocationMode updateAllocationMode();
137 void* allocateFromShared(const LockHolder&, bool abortOnFailure);
138
139private:
140 PackedPtr<IsoDirectoryPage<Config>> m_headDirectory { nullptr };
141 PackedPtr<IsoDirectoryPage<Config>> m_tailDirectory { nullptr };
142 PackedPtr<IsoDirectoryPage<Config>> m_firstEligibleOrDecommitedDirectory { nullptr };
143 IsoDirectory<Config, numPagesInInlineDirectory> m_inlineDirectory;
144 unsigned m_nextDirectoryPageIndex { 1 }; // We start at 1 so that the high water mark being zero means we've only allocated in the inline directory since the last scavenge.
145 unsigned m_directoryHighWatermark { 0 };
146 IsoTLSEntryHolder<IsoTLSAllocatorEntry<Config>> m_allocator;
147};
