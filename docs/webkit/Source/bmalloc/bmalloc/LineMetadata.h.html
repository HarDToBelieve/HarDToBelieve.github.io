28#include "Sizes.h"

30namespace bmalloc {
31
32struct LineMetadata {
33 unsigned char startOffset { };
34 unsigned char objectCount { };
35};
36
37static_assert(
38 smallLineSize - alignment <= std::numeric_limits<unsigned char>::max(),
39 "maximum object offset must fit in LineMetadata::startOffset");
40
41static_assert(
42 smallLineSize / alignment <= std::numeric_limits<unsigned char>::max(),
43 "maximum object count must fit in LineMetadata::objectCount");
44
45} // namespace bmalloc
