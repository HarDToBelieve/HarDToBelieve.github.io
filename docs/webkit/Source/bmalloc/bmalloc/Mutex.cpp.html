38static inline void yield()
39{
40#if BOS(DARWIN)
41 constexpr mach_msg_timeout_t timeoutInMS = 1;
42 thread_switch(MACH_PORT_NULL, SWITCH_OPTION_DEPRESS, timeoutInMS);
43#else
44 sched_yield();
45#endif
46}
47
48void Mutex::lockSlowCase()
49{
50 // The longest critical section in bmalloc is much shorter than the
51 // time it takes to make a system call to yield to the OS scheduler.
52 // So, we try again a lot before we yield.
53 static constexpr size_t aLot = 256;
54
55 if (!m_isSpinning.exchange(true)) {
56 auto clear = makeScopeExit([&] { m_isSpinning.store(false); });
57
58 for (size_t i = 0; i < aLot; ++i) {
59 if (try_lock())
60 return;
61 }
62 }
63
64 // Avoid spinning pathologically.
65 while (!try_lock())
66 yield();
67}
