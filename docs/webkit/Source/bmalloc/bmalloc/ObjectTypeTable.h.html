37namespace bmalloc {
38
39class Chunk;
40
41// Querying ObjectType for Chunk without locking.
42class ObjectTypeTable {
43public:
44 ObjectTypeTable();
45
46 static constexpr unsigned shiftAmount = 20;
47 static_assert((1ULL << shiftAmount) == chunkSize);
48 static_assert((BOS_EFFECTIVE_ADDRESS_WIDTH - shiftAmount) <= 32);
49
50 class Bits;
51
52 ObjectType get(Chunk*);
53 void set(UniqueLockHolder&, Chunk*, ObjectType);
54
55private:
56 static unsigned convertToIndex(Chunk* chunk)
57 {
58 uintptr_t address = reinterpret_cast<uintptr_t>(chunk);
59 BASSERT(!(address & (~chunkMask)));
60 return static_cast<unsigned>(address >> shiftAmount);
61 }
62
63 Bits* m_bits;
64};
65
66class ObjectTypeTable::Bits {
67public:
68 using WordType = unsigned;
69 static constexpr unsigned bitCountPerWord = sizeof(WordType) * 8;
70 static constexpr WordType one = 1;
71 constexpr Bits(Bits* previous, unsigned begin, unsigned end)
72 : m_previous(previous)
73 , m_begin(begin)
74 , m_end(end)
75 {
76 }
77
78 bool get(unsigned index);
79 void set(unsigned index, bool);
80
81 Bits* previous() const { return m_previous; }
82 unsigned begin() const { return m_begin; }
83 unsigned end() const { return m_end; }
84 unsigned count() const { return m_end - m_begin; }
85 unsigned sizeInBytes() const { return count() / 8; }
86
87 const WordType* words() const { return const_cast<Bits*>(this)->words(); }
88 WordType* words() { return reinterpret_cast<WordType*>(reinterpret_cast<uintptr_t>(this) + sizeof(Bits)); }
89
90 WordType* wordForIndex(unsigned);
91
92private:
93 Bits* m_previous { nullptr }; // Keeping the previous Bits* just to suppress Leaks warnings.
94 unsigned m_begin { 0 };
95 unsigned m_end { 0 };
96};
97static_assert(!(sizeof(ObjectTypeTable::Bits) % sizeof(ObjectTypeTable::Bits::WordType)));
98
99extern BEXPORT ObjectTypeTable::Bits sentinelBits;
100
101inline ObjectTypeTable::ObjectTypeTable()
102 : m_bits(&sentinelBits)
103{
104}
105
106inline ObjectType ObjectTypeTable::get(Chunk* chunk)
107{
108 Bits* bits = m_bits;
109 unsigned index = convertToIndex(chunk);
110 BASSERT(bits);
111 if (bits->begin() <= index && index < bits->end())
112 return static_cast<ObjectType>(bits->get(index));
113 return { };
114}
115
116inline bool ObjectTypeTable::Bits::get(unsigned index)
117{
118 unsigned n = index - begin();
119 return words()[n / bitCountPerWord] & (one << (n % bitCountPerWord));
120}
121
122inline void ObjectTypeTable::Bits::set(unsigned index, bool value)
123{
124 unsigned n = index - begin();
125 if (value)
126 words()[n / bitCountPerWord] |= (one << (n % bitCountPerWord));
127 else
128 words()[n / bitCountPerWord] &= ~(one << (n % bitCountPerWord));
129}
130
131inline ObjectTypeTable::Bits::WordType* ObjectTypeTable::Bits::wordForIndex(unsigned index)
132{
133 unsigned n = index - begin();
134 return &words()[n / bitCountPerWord];
135}
