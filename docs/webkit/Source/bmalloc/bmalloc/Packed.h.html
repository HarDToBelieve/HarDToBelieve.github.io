38template<typename T>
39class Packed {
40public:
41 static_assert(std::is_trivial<T>::value);
42 static constexpr bool isPackedType = true;
43
44 Packed()
45 : Packed(T { })
46 {
47 }
48
49 Packed(const T& value)
50 {
51 memcpy(m_storage.data(), &value, sizeof(T));
52 }
53
54 T get() const
55 {
56 T value { };
57 memcpy(&value, m_storage.data(), sizeof(T));
58 return value;
59 }
60
61 void set(const T& value)
62 {
63 memcpy(m_storage.data(), &value, sizeof(T));
64 }
65
66 Packed<T>& operator=(const T& value)
67 {
68 set(value);
69 return *this;
70 }
71
72 template<class U>
73 T exchange(U&& newValue)
74 {
75 T oldValue = get();
76 set(std::forward<U>(newValue));
77 return oldValue;
78 }
79
80 void swap(Packed& other)
81 {
82 m_storage.swap(other.m_storage);
83 }
84
85 template<typename Other, typename = std::enable_if_t<Other::isPackedType>>
86 void swap(Other& other)
87 {
88 T t1 = get();
89 T t2 = other.get();
90 set(t2);
91 other.set(t1);
92 }
93
94 void swap(T& t2)
95 {
96 T t1 = get();
97 std::swap(t1, t2);
98 set(t1);
99 }
100
101private:
102 std::array<uint8_t, sizeof(T)> m_storage;
103};

105// PackedAlignedPtr can take alignment parameter too. PackedAlignedPtr only uses this alignment information if it is profitable: we use
106// alignment information only when we can reduce the size of the storage.
107template<typename T, size_t alignment = alignof(T)>
108class PackedAlignedPtr {
109public:
110 static_assert(isPowerOfTwo(alignment), "Alignment needs to be power-of-two");
111 static constexpr bool isPackedType = true;
112 static constexpr unsigned alignmentShiftSizeIfProfitable = getLSBSetNonZeroConstexpr(alignment);
113 static constexpr unsigned storageSizeWithoutAlignmentShift = roundUpToMultipleOf<8, uintptr_t>(BOS_EFFECTIVE_ADDRESS_WIDTH) / 8;
114 static constexpr unsigned storageSizeWithAlignmentShift = roundUpToMultipleOf<8, uintptr_t>(BOS_EFFECTIVE_ADDRESS_WIDTH - alignmentShiftSizeIfProfitable) / 8;
115 static constexpr bool isAlignmentShiftProfitable = storageSizeWithoutAlignmentShift > storageSizeWithAlignmentShift;
116 static constexpr unsigned alignmentShiftSize = isAlignmentShiftProfitable ? alignmentShiftSizeIfProfitable : 0;
117 static constexpr unsigned storageSize = storageSizeWithAlignmentShift;
118
119 constexpr PackedAlignedPtr()
120 : m_storage()
121 {
122 }
123
124 constexpr PackedAlignedPtr(std::nullptr_t)
125 : m_storage()
126 {
127 }
128
129 PackedAlignedPtr(T* value)
130 {
131 set(value);
132 }
133
134 T* get() const
135 {
136 // FIXME: PackedPtr<> can load memory with one mov by checking page boundary.
137 // https://bugs.webkit.org/show_bug.cgi?id=197754
138 uintptr_t value = 0;
139#if BCPU(LITTLE_ENDIAN)
140 memcpy(&value, m_storage.data(), storageSize);
141#else
142 memcpy(bitwise_cast<uint8_t*>(&value) + (sizeof(void*) - storageSize), m_storage.data(), storageSize);
143#endif
144 if (isAlignmentShiftProfitable)
145 value <<= alignmentShiftSize;
146 return bitwise_cast<T*>(value);
147 }
148
149 void set(T* passedValue)
150 {
151 uintptr_t value = bitwise_cast<uintptr_t>(passedValue);
152 if (isAlignmentShiftProfitable)
153 value >>= alignmentShiftSize;
154#if BCPU(LITTLE_ENDIAN)
155 memcpy(m_storage.data(), &value, storageSize);
156#else
157 memcpy(m_storage.data(), bitwise_cast<uint8_t*>(&value) + (sizeof(void*) - storageSize), storageSize);
158#endif
159 }
160
161 void clear()
162 {
163 set(nullptr);
164 }
165
166 T* operator->() const { return get(); }
167 T& operator*() const { return *get(); }
168 bool operator!() const { return !get(); }
169
170 // This conversion operator allows implicit conversion to bool but not to other integer types.
171 typedef T* (PackedAlignedPtr::*UnspecifiedBoolType);
172 operator UnspecifiedBoolType() const { return get() ? &PackedAlignedPtr::m_storage : nullptr; }
173 explicit operator bool() const { return get(); }
174
175 PackedAlignedPtr& operator=(T* value)
176 {
177 set(value);
178 return *this;
179 }
180
181 template<class U>
182 T* exchange(U&& newValue)
183 {
184 T* oldValue = get();
185 set(std::forward<U>(newValue));
186 return oldValue;
187 }
188
189 void swap(std::nullptr_t) { clear(); }
190
191 void swap(PackedAlignedPtr& other)
192 {
193 m_storage.swap(other.m_storage);
194 }
195
196 template<typename Other, typename = std::enable_if_t<Other::isPackedType>>
197 void swap(Other& other)
198 {
199 T* t1 = get();
200 T* t2 = other.get();
201 set(t2);
202 other.set(t1);
203 }
204
205 void swap(T* t2)
206 {
207 T* t1 = get();
208 std::swap(t1, t2);
209 set(t1);
210 }
211
212private:
213 std::array<uint8_t, storageSize> m_storage;
214};
215
216template<typename T>
217class Packed<T*> : public PackedAlignedPtr<T, 1> {
218public:
219 using Base = PackedAlignedPtr<T, 1>;
220 using Base::Base;
221};
222
223template<typename T>
224using PackedPtr = Packed<T*>;
225
226template<typename T>
227struct PackedPtrTraits {
228 template<typename U> using RebindTraits = PackedPtrTraits<U>;
229
230 using StorageType = PackedPtr<T>;
231
232 template<class U> static T* exchange(StorageType& ptr, U&& newValue) { return ptr.exchange(newValue); }
233
234 template<typename Other> static void swap(PackedPtr<T>& a, Other& b) { a.swap(b); }
235
236 static T* unwrap(const StorageType& ptr) { return ptr.get(); }
237};
