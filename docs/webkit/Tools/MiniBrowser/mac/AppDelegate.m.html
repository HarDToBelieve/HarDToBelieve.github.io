94static WKWebsiteDataStore *persistentDataStore()
95{
96 static WKWebsiteDataStore *dataStore;
97
98 if (!dataStore) {
99 _WKWebsiteDataStoreConfiguration *configuration = [[[_WKWebsiteDataStoreConfiguration alloc] init] autorelease];
100 configuration.networkCacheSpeculativeValidationEnabled = YES;
101 dataStore = [[WKWebsiteDataStore alloc] _initWithConfiguration:configuration];
102 }
103
104 return dataStore;
105}

107- (WKWebViewConfiguration *)defaultConfiguration
108{
109 static WKWebViewConfiguration *configuration;
110
111 if (!configuration) {
112 configuration = [[WKWebViewConfiguration alloc] init];
113 configuration.websiteDataStore = persistentDataStore();
114
115 _WKProcessPoolConfiguration *processConfiguration = [[[_WKProcessPoolConfiguration alloc] init] autorelease];
116 if (_settingsController.perWindowWebProcessesDisabled)
117 processConfiguration.usesSingleWebProcess = YES;
118 if (_settingsController.processSwapOnWindowOpenWithOpenerEnabled)
119 processConfiguration.processSwapsOnWindowOpenWithOpener = true;
120
121 configuration.processPool = [[[WKProcessPool alloc] _initWithConfiguration:processConfiguration] autorelease];
122
123 NSArray<_WKExperimentalFeature *> *experimentalFeatures = [WKPreferences _experimentalFeatures];
124 for (_WKExperimentalFeature *feature in experimentalFeatures) {
125 BOOL enabled;
126 if ([[NSUserDefaults standardUserDefaults] objectForKey:feature.key])
127 enabled = [[NSUserDefaults standardUserDefaults] boolForKey:feature.key];
128 else
129 enabled = [feature defaultValue];
130 [configuration.preferences _setEnabled:enabled forExperimentalFeature:feature];
131 }
132
133 NSArray<_WKInternalDebugFeature *> *internalDebugFeatures = [WKPreferences _internalDebugFeatures];
134 for (_WKInternalDebugFeature *feature in internalDebugFeatures) {
135 BOOL enabled;
136 if ([[NSUserDefaults standardUserDefaults] objectForKey:feature.key])
137 enabled = [[NSUserDefaults standardUserDefaults] boolForKey:feature.key];
138 else
139 enabled = [feature defaultValue];
140 [configuration.preferences _setEnabled:enabled forInternalDebugFeature:feature];
141 }
142
143 configuration.preferences._fullScreenEnabled = YES;
144 configuration.preferences._allowsPictureInPictureMediaPlayback = YES;
145 configuration.preferences._developerExtrasEnabled = YES;
146 configuration.preferences._mockCaptureDevicesEnabled = YES;
147 configuration.preferences._accessibilityIsolatedTreeEnabled = YES;
148 configuration.preferences._logsPageMessagesToSystemConsoleEnabled = YES;
149 }
150
151 configuration.suppressesIncrementalRendering = _settingsController.incrementalRenderingSuppressed;
152 configuration.websiteDataStore._resourceLoadStatisticsEnabled = _settingsController.resourceLoadStatisticsEnabled;
153 return configuration;
154}

161- (BrowserWindowController *)createBrowserWindowController:(id)sender
162{
163 BrowserWindowController *controller = nil;
164 BOOL useWebKit2 = NO;
165 BOOL makeEditable = NO;
166
167 if (![sender respondsToSelector:@selector(tag)]) {
168 useWebKit2 = _settingsController.useWebKit2ByDefault;
169 makeEditable = _settingsController.createEditorByDefault;
170 } else {
171 useWebKit2 = [sender tag] == WebKit2NewWindowTag || [sender tag] == WebKit2NewEditorTag;
172 makeEditable = [sender tag] == WebKit1NewEditorTag || [sender tag] == WebKit2NewEditorTag;
173 }
174
175 if (!useWebKit2)
176 controller = [[WK1BrowserWindowController alloc] initWithWindowNibName:@"BrowserWindow"];
177 else
178 controller = [[WK2BrowserWindowController alloc] initWithConfiguration:[self defaultConfiguration]];
179
180 if (makeEditable)
181 controller.editable = YES;
182
183 if (!controller)
184 return nil;
185
186 [_browserWindowControllers addObject:controller];
187
188 return controller;
189}

201- (IBAction)newPrivateWindow:(id)sender
202{
203 WKWebViewConfiguration *privateConfiguraton = [self.defaultConfiguration copy];
204 privateConfiguraton.websiteDataStore = [WKWebsiteDataStore nonPersistentDataStore];
205
206 BrowserWindowController *controller = [[WK2BrowserWindowController alloc] initWithConfiguration:privateConfiguraton];
207 [privateConfiguraton release];
208
209 [[controller window] makeKeyAndOrderFront:sender];
210 [_browserWindowControllers addObject:controller];
211
212 [controller loadURLString:_settingsController.defaultURL];
213}

230- (void)applicationDidFinishLaunching:(NSNotification *)aNotification
231{
232 WebHistory *webHistory = [[WebHistory alloc] init];
233 [WebHistory setOptionalSharedHistory:webHistory];
234 [webHistory release];
235
236 [self _updateNewWindowKeyEquivalents];
237
238 if (!_openNewWindowAtStartup)
239 return;
240
241 if (_settingsController.createEditorByDefault)
242 [self newEditorWindow:self];
243 else
244 [self newWindow:self];
245}

247- (BrowserWindowController *)frontmostBrowserWindowController
248{
249 for (NSWindow* window in [NSApp windows]) {
250 id delegate = [window delegate];
251
252 if (![delegate isKindOfClass:[BrowserWindowController class]])
253 continue;
254
255 BrowserWindowController *controller = (BrowserWindowController *)delegate;
256 assert([_browserWindowControllers containsObject:controller]);
257 return controller;
258 }
259
260 return nil;
261}

263- (BOOL)application:(NSApplication *)theApplication openFile:(NSString *)filename
264{
265 BrowserWindowController *controller = [self createBrowserWindowController:nil];
266 if (!controller)
267 return NO;
268
269 [controller.window makeKeyAndOrderFront:self];
270 [controller loadURLString:[NSURL fileURLWithPath:filename].absoluteString];
271 _openNewWindowAtStartup = false;
272 return YES;
273}

275- (IBAction)openDocument:(id)sender
276{
277 BrowserWindowController *browserWindowController = [self frontmostBrowserWindowController];
278
279 if (browserWindowController) {
280 NSOpenPanel *openPanel = [[NSOpenPanel openPanel] retain];
281 [openPanel beginSheetModalForWindow:browserWindowController.window completionHandler:^(NSInteger result) {
282 if (result != NSModalResponseOK)
283 return;
284
285 NSURL *url = [openPanel.URLs objectAtIndex:0];
286 [browserWindowController loadURLString:[url absoluteString]];
287 }];
288 return;
289 }
290
291 NSOpenPanel *openPanel = [NSOpenPanel openPanel];
292 [openPanel beginWithCompletionHandler:^(NSInteger result) {
293 if (result != NSModalResponseOK)
294 return;
295
296 BrowserWindowController *controller = [self createBrowserWindowController:nil];
297 [controller.window makeKeyAndOrderFront:self];
298
299 NSURL *url = [openPanel.URLs objectAtIndex:0];
300 [controller loadURLString:[url absoluteString]];
301 }];
302}

304- (void)didChangeSettings
305{
306 [self _updateNewWindowKeyEquivalents];

308 // Let all of the BrowserWindowControllers know that a setting changed, so they can attempt to dynamically update.
309 for (BrowserWindowController *browserWindowController in _browserWindowControllers)
310 [browserWindowController didChangeSettings];
311}

313- (void)_updateNewWindowKeyEquivalents
314{
315 NSEventModifierFlags webKit1Flags = _settingsController.useWebKit2ByDefault ? NSEventModifierFlagOption : 0;
316 NSEventModifierFlags webKit2Flags = _settingsController.useWebKit2ByDefault ? 0 : NSEventModifierFlagOption;
317
318 NSString *normalWindowEquivalent = _settingsController.createEditorByDefault ? @"N" : @"n";
319 NSString *editorEquivalent = _settingsController.createEditorByDefault ? @"n" : @"N";
320
321 _newWebKit1WindowItem.keyEquivalentModifierMask = NSEventModifierFlagCommand | webKit1Flags;
322 _newWebKit2WindowItem.keyEquivalentModifierMask = NSEventModifierFlagCommand | webKit2Flags;
323 _newWebKit1EditorItem.keyEquivalentModifierMask = NSEventModifierFlagCommand | webKit1Flags;
324 _newWebKit2EditorItem.keyEquivalentModifierMask = NSEventModifierFlagCommand | webKit2Flags;
325
326 _newWebKit1WindowItem.keyEquivalent = normalWindowEquivalent;
327 _newWebKit2WindowItem.keyEquivalent = normalWindowEquivalent;
328 _newWebKit1EditorItem.keyEquivalent = editorEquivalent;
329 _newWebKit2EditorItem.keyEquivalent = editorEquivalent;
330}

349- (IBAction)fetchAndClearDefaultStoreWebsiteData:(id)sender
350{
351 [persistentDataStore() fetchDataRecordsOfTypes:[WKWebsiteDataStore allWebsiteDataTypes] completionHandler:^(NSArray *websiteDataRecords) {
352 [persistentDataStore() removeDataOfTypes:[WKWebsiteDataStore allWebsiteDataTypes] forDataRecords:websiteDataRecords completionHandler:^{

354 NSLog(@"did clear default store website data, after clearing data is %@.", websiteDataRecords);
355 }];
356 }];
357 }];
358}
