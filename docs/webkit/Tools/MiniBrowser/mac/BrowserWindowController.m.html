75- (NSString *)addProtocolIfNecessary:(NSString *)address
76{
77 if ([address rangeOfString:@"://"].length > 0)
78 return address;
79
80 if ([address hasPrefix:@"data:"])
81 return address;
82
83 if ([address hasPrefix:@"about:"])
84 return address;
85
86 return [@"http://" stringByAppendingString:address];
87}

149- (IBAction)toggleFullWindowWebView:(id)sender
150{
151 BOOL newFillWindow = ![self webViewFillsWindow];
152 [self setWebViewFillsWindow:newFillWindow];
153
154 SettingsController *settings = [[NSApplication sharedApplication] browserAppDelegate].settingsController;
155 settings.webViewFillsWindow = newFillWindow;
156}
157
158- (BOOL)webViewFillsWindow
159{
160 return NSEqualRects(containerView.bounds, self.mainContentView.frame);
161}
162
163- (void)setWebViewFillsWindow:(BOOL)fillWindow
164{
165 if (fillWindow)
166 [self.mainContentView setFrame:containerView.bounds];
167 else {
168 const CGFloat viewInset = 100.0f;
169 NSRect viewRect = NSInsetRect(containerView.bounds, viewInset, viewInset);
170 // Make it not vertically centered, to reveal y-flipping bugs.
171 viewRect = NSOffsetRect(viewRect, 0, -25);
172 [self.mainContentView setFrame:viewRect];
173 }
174}

295- (NSRect)sharingService:(NSSharingService *)sharingService sourceFrameOnScreenForShareItem:(id)item
296{
297 NSRect rect = [self.window convertRectToScreen:self.mainContentView.bounds];
298
299 return rect;
300}

302static CGRect coreGraphicsScreenRectForAppKitScreenRect(NSRect rect)
303{
304 NSScreen *firstScreen = [NSScreen screens][0];
305 return CGRectMake(NSMinX(rect), NSHeight(firstScreen.frame) - NSMinY(rect) - NSHeight(rect), NSWidth(rect), NSHeight(rect));
306}
307
308- (NSImage *)sharingService:(NSSharingService *)sharingService transitionImageForShareItem:(id)item contentRect:(NSRect *)contentRect
309{
310 NSRect contentFrame = [self.window convertRectToScreen:self.mainContentView.bounds];
311
312 CGRect frame = coreGraphicsScreenRectForAppKitScreenRect(contentFrame);
313 CGImageRef imageRef = CGWindowListCreateImage(frame, kCGWindowListOptionIncludingWindow, (CGWindowID)[self.window windowNumber], kCGWindowImageBoundsIgnoreFraming);
314
315 if (!imageRef)
316 return nil;
317
318 NSImage *image = [[NSImage alloc] initWithCGImage:imageRef size:NSZeroSize];
319 CGImageRelease(imageRef);
320
321 return [image autorelease];
322}
