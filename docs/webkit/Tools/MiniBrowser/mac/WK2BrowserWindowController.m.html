93{

100
101 [_webView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable)];
102 [containerView addSubview:_webView];
103
104 [progressIndicator bind:NSHiddenBinding toObject:_webView withKeyPath:@"loading" options:@{ NSValueTransformerNameBindingOption : NSNegateBooleanTransformerName }];
105 [progressIndicator bind:NSValueBinding toObject:_webView withKeyPath:@"estimatedProgress" options:nil];
106
107 [_webView addObserver:self forKeyPath:@"title" options:0 context:keyValueObservingContext];
108 [_webView addObserver:self forKeyPath:@"URL" options:0 context:keyValueObservingContext];
109 [_webView addObserver:self forKeyPath:@"hasOnlySecureContent" options:0 context:keyValueObservingContext];

111 _webView.navigationDelegate = self;
112 _webView.UIDelegate = self;
113
114 SettingsController *settingsController = [[NSApplication sharedApplication] browserAppDelegate].settingsController;
115 // This setting installs the new WK2 Icon Loading Delegate and tests that mechanism by
116 // telling WebKit to load every icon referenced by the page.
117 if (settingsController.loadsAllSiteIcons)
118 _webView._iconLoadingDelegate = self;
119
120 _webView._observedRenderingProgressEvents = _WKRenderingProgressEventFirstLayout
121 | _WKRenderingProgressEventFirstVisuallyNonEmptyLayout
122 | _WKRenderingProgressEventFirstPaintWithSignificantArea
123 | _WKRenderingProgressEventFirstLayoutAfterSuppressedIncrementalRendering
124 | _WKRenderingProgressEventFirstPaintAfterSuppressedIncrementalRendering;
125
126
127 if (settingsController.customUserAgent)
128 _webView.customUserAgent = settingsController.customUserAgent;
129
130 _webView._usePlatformFindUI = NO;
131
132 _textFinder = [[MiniBrowserNSTextFinder alloc] init];
133 _textFinder.incrementalSearchingEnabled = YES;
134 _textFinder.incrementalSearchingShouldDimContentView = NO;
135 _textFinder.client = _webView;
136 _textFinder.findBarContainer = self;
137
138#if __has_feature(objc_arc)

140#else
141 WKWebView *weakWebView = _webView;
142#endif

145 [webView _hideFindUI];
146 };
147
148 _zoomTextOnly = NO;

201- (CGFloat)viewScaleForMenuItemTag:(NSInteger)tag
202{
203 if (tag == 1)
204 return 1;
205 if (tag == 2)
206 return 0.75;
207 if (tag == 3)
208 return 0.5;
209 if (tag == 4)
210 return 0.25;
211
212 return 1;
213}

215- (IBAction)setViewScale:(id)sender
216{
217 CGFloat scale = [self viewScaleForMenuItemTag:[sender tag]];
218 CGFloat oldScale = [_webView _viewScale];
219
220 if (scale == oldScale)
221 return;
222
223 [_webView _setLayoutMode:_WKLayoutModeDynamicSizeComputedFromViewScale];
224
225 NSRect oldFrame = self.window.frame;
226 NSSize newFrameSize = NSMakeSize(oldFrame.size.width * (scale / oldScale), oldFrame.size.height * (scale / oldScale));
227 [self.window setFrame:NSMakeRect(oldFrame.origin.x, oldFrame.origin.y - (newFrameSize.height - oldFrame.size.height), newFrameSize.width, newFrameSize.height) display:NO animate:NO];
228
229 [_webView _setViewScale:scale];
230}

232static BOOL areEssentiallyEqual(double a, double b)
233{
234 double tolerance = 0.001;
235 return (fabs(a - b) <= tolerance);
236}

239#pragma GCC diagnostic ignored "-Wdeprecated-implementations"
240- (BOOL)validateMenuItem:(NSMenuItem *)menuItem
241#pragma GCC diagnostic pop
242{
243 SEL action = menuItem.action;
244
245 if (action == @selector(saveAsPDF:))
246 return YES;
247 if (action == @selector(saveAsWebArchive:))
248 return YES;
249
250 if (action == @selector(zoomIn:))
251 return [self canZoomIn];
252 if (action == @selector(zoomOut:))
253 return [self canZoomOut];
254 if (action == @selector(resetZoom:))
255 return [self canResetZoom];
256
257 // Disabled until missing WK2 functionality is exposed via API/SPI.
258 if (action == @selector(dumpSourceToConsole:)
259 || action == @selector(forceRepaint:))
260 return NO;
261
262 if (action == @selector(showHideWebView:))
263 [menuItem setTitle:[_webView isHidden] ? @"Show Web View" : @"Hide Web View"];
264 else if (action == @selector(removeReinsertWebView:))
265 [menuItem setTitle:[_webView window] ? @"Remove Web View" : @"Insert Web View"];
266 else if (action == @selector(toggleFullWindowWebView:))
267 [menuItem setTitle:[self webViewFillsWindow] ? @"Inset Web View" : @"Fit Web View to Window"];
268 else if (action == @selector(toggleZoomMode:))
269 [menuItem setState:_zoomTextOnly ? NSControlStateValueOn : NSControlStateValueOff];
270 else if (action == @selector(toggleEditable:))
271 [menuItem setState:self.isEditable ? NSControlStateValueOn : NSControlStateValueOff];
272 else if (action == @selector(showHideWebInspector:))
273 [menuItem setTitle:_webView._inspector.isVisible ? @"Close Web Inspector" : @"Show Web Inspector"];
274 else if (action == @selector(toggleAlwaysShowsHorizontalScroller:))
275 menuItem.state = _webView._alwaysShowsHorizontalScroller ? NSControlStateValueOn : NSControlStateValueOff;
276 else if (action == @selector(toggleAlwaysShowsVerticalScroller:))
277 menuItem.state = _webView._alwaysShowsVerticalScroller ? NSControlStateValueOn : NSControlStateValueOff;
278
279 if (action == @selector(setPageScale:))
280 [menuItem setState:areEssentiallyEqual([_webView _pageScale], [self pageScaleForMenuItemTag:[menuItem tag]])];
281
282 if (action == @selector(setViewScale:))
283 [menuItem setState:areEssentiallyEqual([_webView _viewScale], [self viewScaleForMenuItemTag:[menuItem tag]])];
284
285 return YES;
286}

315- (IBAction)toggleZoomMode:(id)sender
316{
317 if (_zoomTextOnly) {
318 _zoomTextOnly = NO;
319 double currentTextZoom = _webView._textZoomFactor;
320 _webView._textZoomFactor = 1;
321 _webView.pageZoom = currentTextZoom;
322 } else {
323 _zoomTextOnly = YES;
324 double currentPageZoom = _webView._pageZoomFactor;
325 _webView._textZoomFactor = currentPageZoom;
326 _webView.pageZoom = 1;
327 }
328}

421#define DefaultMinimumZoomFactor (.5)
422#define DefaultMaximumZoomFactor (3.0)
423#define DefaultZoomFactorRatio (1.2)
424
425- (CGFloat)currentZoomFactor
426{
427 return _zoomTextOnly ? _webView._textZoomFactor : _webView.pageZoom;
428}
429
430- (void)setCurrentZoomFactor:(CGFloat)factor
431{
432 if (_zoomTextOnly)
433 _webView._textZoomFactor = factor;
434 else
435 _webView.pageZoom = factor;
436}
437
438- (BOOL)canZoomIn
439{
440 return self.currentZoomFactor * DefaultZoomFactorRatio < DefaultMaximumZoomFactor;
441}
442
443- (void)zoomIn:(id)sender
444{
445 if (!self.canZoomIn)
446 return;
447
448 self.currentZoomFactor *= DefaultZoomFactorRatio;
449}
450
451- (BOOL)canZoomOut
452{
453 return self.currentZoomFactor / DefaultZoomFactorRatio > DefaultMinimumZoomFactor;
454}
455
456- (void)zoomOut:(id)sender
457{
458 if (!self.canZoomIn)
459 return;
460
461 self.currentZoomFactor /= DefaultZoomFactorRatio;
462}

464- (void)didChangeSettings
465{
466 SettingsController *settings = [[NSApplication sharedApplication] browserAppDelegate].settingsController;
467 WKPreferences *preferences = _webView.configuration.preferences;
468
469 _webView._useSystemAppearance = settings.useSystemAppearance;
470
471 preferences._tiledScrollingIndicatorVisible = settings.tiledScrollingIndicatorVisible;
472 preferences._compositingBordersVisible = settings.layerBordersVisible;
473 preferences._compositingRepaintCountersVisible = settings.layerBordersVisible;
474 preferences._legacyLineLayoutVisualCoverageEnabled = settings.legacyLineLayoutVisualCoverageEnabled;
475 preferences._acceleratedDrawingEnabled = settings.acceleratedDrawingEnabled;
476 preferences._resourceUsageOverlayVisible = settings.resourceUsageOverlayVisible;
477 preferences._displayListDrawingEnabled = settings.displayListDrawingEnabled;
478 preferences._subpixelAntialiasedLayerTextEnabled = settings.subpixelAntialiasedLayerTextEnabled;
479 preferences._largeImageAsyncDecodingEnabled = settings.largeImageAsyncDecodingEnabled;
480 preferences._animatedImageAsyncDecodingEnabled = settings.animatedImageAsyncDecodingEnabled;
481 preferences._colorFilterEnabled = settings.appleColorFilterEnabled;
482 preferences._punchOutWhiteBackgroundsInDarkMode = settings.punchOutWhiteBackgroundsInDarkMode;
483
484 _webView.configuration.websiteDataStore._resourceLoadStatisticsEnabled = settings.resourceLoadStatisticsEnabled;
485
486 [self setWebViewFillsWindow:settings.webViewFillsWindow];
487
488 BOOL useTransparentWindows = settings.useTransparentWindows;
489 if (useTransparentWindows != !_webView._drawsBackground) {
490 [self.window setOpaque:!useTransparentWindows];
491 [self.window setBackgroundColor:[NSColor clearColor]];
492 [self.window setHasShadow:!useTransparentWindows];
493
494 _webView._drawsBackground = !useTransparentWindows;
495
496 [self.window display];
497 }
498
499 BOOL usePaginatedMode = settings.usePaginatedMode;
500 if (usePaginatedMode != (_webView._paginationMode != _WKPaginationModeUnpaginated)) {
501 if (usePaginatedMode) {
502 _webView._paginationMode = _WKPaginationModeLeftToRight;
503 _webView._pageLength = _webView.bounds.size.width / 2;
504 _webView._gapBetweenPages = 10;
505 } else
506 _webView._paginationMode = _WKPaginationModeUnpaginated;
507 }
508
509 NSUInteger visibleOverlayRegions = 0;
510 if (settings.nonFastScrollableRegionOverlayVisible)
511 visibleOverlayRegions |= _WKNonFastScrollableRegion;
512 if (settings.wheelEventHandlerRegionOverlayVisible)
513 visibleOverlayRegions |= _WKWheelEventHandlerRegion;
514
515 preferences._visibleDebugOverlayRegions = visibleOverlayRegions;
516
517 [_webView _setHeaderBannerHeight:[settings isSpaceReservedForBanners] ? testHeaderBannerHeight : 0];
518 [_webView _setFooterBannerHeight:[settings isSpaceReservedForBanners] ? testFooterBannerHeight : 0];
519}

769- (void)webView:(WKWebView *)webView didReceiveAuthenticationChallenge:(NSURLAuthenticationChallenge *)challenge completionHandler:(void (^)(NSURLSessionAuthChallengeDisposition disposition, NSURLCredential *__nullable credential))completionHandler
770{
771 LOG(@"didReceiveAuthenticationChallenge: %@", challenge);
772 if ([challenge.protectionSpace.authenticationMethod isEqualToString:NSURLAuthenticationMethodHTTPBasic]) {
773 NSAlert *alert = [[NSAlert alloc] init];
774 NSView *container = [[[NSView alloc] initWithFrame:NSMakeRect(0, 0, 200, 48)] autorelease];
775 NSTextField *userInput = [[[NSTextField alloc] initWithFrame:NSMakeRect(0, 24, 200, 24)] autorelease];
776 NSTextField *passwordInput = [[[NSSecureTextField alloc] initWithFrame:NSMakeRect(0, 0, 200, 24)] autorelease];
777
778 [alert setMessageText:[NSString stringWithFormat:@"Log in to %@:%lu.", challenge.protectionSpace.host, challenge.protectionSpace.port]];
779 [alert addButtonWithTitle:@"Log in"];
780 [alert addButtonWithTitle:@"Cancel"];
781 [container addSubview:userInput];
782 [container addSubview:passwordInput];
783 [alert setAccessoryView:container];
784 [userInput setNextKeyView:passwordInput];
785 [alert.window setInitialFirstResponder:userInput];
786
787 [alert beginSheetModalForWindow:self.window completionHandler:^(NSModalResponse response) {
788 [userInput validateEditing];
789 if (response == NSAlertFirstButtonReturn)
790 completionHandler(NSURLSessionAuthChallengeUseCredential, [[[NSURLCredential alloc] initWithUser:[userInput stringValue] password:[passwordInput stringValue] persistence:NSURLCredentialPersistenceForSession] autorelease]);
791 else
792 completionHandler(NSURLSessionAuthChallengeRejectProtectionSpace, nil);
793 [alert release];
794 }];
795 return;
796 }
797 completionHandler(NSURLSessionAuthChallengeRejectProtectionSpace, nil);
