667JSC::IsoSubspace* JSEvent::subspaceForImpl(JSC::VM& vm)
668{
669 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
670 auto& spaces = clientData.subspaces();
671 if (auto* space = spaces.m_subspaceForEvent.get())
672 return space;
673 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSEvent> || !JSEvent::needsDestruction);
674 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSEvent>)
675 spaces.m_subspaceForEvent = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSEvent);
676 else
677 spaces.m_subspaceForEvent = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSEvent);
678 auto* space = spaces.m_subspaceForEvent.get();
679IGNORE_WARNINGS_BEGIN("unreachable-code")
680IGNORE_WARNINGS_BEGIN("tautological-compare")
681 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSEvent::visitOutputConstraints;
682 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
683 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
684 clientData.outputConstraintSpaces().append(space);
685IGNORE_WARNINGS_END
686IGNORE_WARNINGS_END
687 return space;
688}
