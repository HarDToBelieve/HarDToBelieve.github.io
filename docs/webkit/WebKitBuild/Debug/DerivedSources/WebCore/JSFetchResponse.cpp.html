660JSC::IsoSubspace* JSFetchResponse::subspaceForImpl(JSC::VM& vm)
661{
662 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
663 auto& spaces = clientData.subspaces();
664 if (auto* space = spaces.m_subspaceForFetchResponse.get())
665 return space;
666 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSFetchResponse> || !JSFetchResponse::needsDestruction);
667 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSFetchResponse>)
668 spaces.m_subspaceForFetchResponse = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSFetchResponse);
669 else
670 spaces.m_subspaceForFetchResponse = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSFetchResponse);
671 auto* space = spaces.m_subspaceForFetchResponse.get();
672IGNORE_WARNINGS_BEGIN("unreachable-code")
673IGNORE_WARNINGS_BEGIN("tautological-compare")
674 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSFetchResponse::visitOutputConstraints;
675 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
676 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
677 clientData.outputConstraintSpaces().append(space);
678IGNORE_WARNINGS_END
679IGNORE_WARNINGS_END
680 return space;
681}
