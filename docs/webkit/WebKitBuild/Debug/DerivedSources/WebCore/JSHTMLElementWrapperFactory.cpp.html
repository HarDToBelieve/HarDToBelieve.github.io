793{
794 static const auto functions = makeNeverDestroyed(createHTMLWrapperMap());
795 if (auto function = functions.get().get(element->localName()))
796 return function(globalObject, WTFMove(element));
797 if (!element->isUnknownElement())

799 return createWrapper<HTMLUnknownElement>(globalObject, WTFMove(element));
