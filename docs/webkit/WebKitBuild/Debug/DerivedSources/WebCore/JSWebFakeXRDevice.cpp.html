429 if (auto* space = spaces.m_subspaceForWebFakeXRDevice.get())
430 return space;
431 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSWebFakeXRDevice> || !JSWebFakeXRDevice::needsDestruction);
432 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSWebFakeXRDevice>)
433 spaces.m_subspaceForWebFakeXRDevice = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSWebFakeXRDevice);
434 else
435 spaces.m_subspaceForWebFakeXRDevice = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSWebFakeXRDevice);
436 auto* space = spaces.m_subspaceForWebFakeXRDevice.get();
437IGNORE_WARNINGS_BEGIN("unreachable-code")
438IGNORE_WARNINGS_BEGIN("tautological-compare")
439 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSWebFakeXRDevice::visitOutputConstraints;
440 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
441 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
442 clientData.outputConstraintSpaces().append(space);
443IGNORE_WARNINGS_END
444IGNORE_WARNINGS_END
445 return space;
446}
