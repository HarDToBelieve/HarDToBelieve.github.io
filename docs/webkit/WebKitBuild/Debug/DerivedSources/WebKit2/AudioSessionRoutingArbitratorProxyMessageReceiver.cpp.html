40void BeginRoutingArbitrationWithCategory::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError&&, WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged&&)>&& completionHandler)
41{
42 std::optional<WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError> error;
43 decoder >> error;
44 if (!error) {
45 ASSERT_NOT_REACHED();
46 cancelReply(WTFMove(completionHandler));
47 return;
48 }
49 std::optional<WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged> defaultRouteChanged;
50 decoder >> defaultRouteChanged;
51 if (!defaultRouteChanged) {
52 ASSERT_NOT_REACHED();
53 cancelReply(WTFMove(completionHandler));
54 return;
55 }
56 completionHandler(WTFMove(*error), WTFMove(*defaultRouteChanged));
57}
58
59void BeginRoutingArbitrationWithCategory::cancelReply(CompletionHandler<void(WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError&&, WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged&&)>&& completionHandler)
60{
61 completionHandler(IPC::AsyncReplyError<WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError>::create(), IPC::AsyncReplyError<WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged>::create());
62}
63
64void BeginRoutingArbitrationWithCategory::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError error, WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged defaultRouteChanged)
65{
66 encoder.get() << error;
67 encoder.get() << defaultRouteChanged;
68 connection.sendSyncReply(WTFMove(encoder));
69}

77void AudioSessionRoutingArbitratorProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
78{
79 if (decoder.messageName() == Messages::AudioSessionRoutingArbitratorProxy::BeginRoutingArbitrationWithCategory::name())
80 return IPC::handleMessageAsync<Messages::AudioSessionRoutingArbitratorProxy::BeginRoutingArbitrationWithCategory>(connection, decoder, this, &AudioSessionRoutingArbitratorProxy::beginRoutingArbitrationWithCategory);
81 if (decoder.messageName() == Messages::AudioSessionRoutingArbitratorProxy::EndRoutingArbitration::name())
82 return IPC::handleMessage<Messages::AudioSessionRoutingArbitratorProxy::EndRoutingArbitration>(decoder, this, &AudioSessionRoutingArbitratorProxy::endRoutingArbitration);
83 UNUSED_PARAM(connection);
84 UNUSED_PARAM(decoder);
85#if ENABLE(IPC_TESTING_API)
86 if (connection.ignoreInvalidMessageForTesting())
87 return;
88#endif // ENABLE(IPC_TESTING_API)
89 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
90}
