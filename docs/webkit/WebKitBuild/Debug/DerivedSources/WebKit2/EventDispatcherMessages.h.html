53class WheelEvent {
54public:
55 using Arguments = std::tuple<const WebCore::PageIdentifier&, const WebKit::WebWheelEvent&, bool, bool, bool, bool>;
56
57 static IPC::MessageName name() { return IPC::MessageName::EventDispatcher_WheelEvent; }
58 static constexpr bool isSync = false;
59
60 WheelEvent(const WebCore::PageIdentifier& pageID, const WebKit::WebWheelEvent& event, bool canRubberBandAtLeft, bool canRubberBandAtRight, bool canRubberBandAtTop, bool canRubberBandAtBottom)
61 : m_arguments(pageID, event, canRubberBandAtLeft, canRubberBandAtRight, canRubberBandAtTop, canRubberBandAtBottom)
62 {
63 }
64
65 const Arguments& arguments() const
66 {
67 return m_arguments;
68 }
69
70private:
71 Arguments m_arguments;
72};
