81void GPUConnectionToWebProcess::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
82{
83 auto protectedThis = makeRef(*this);
84 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::CreateRenderingBackend::name())
85 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::CreateRenderingBackend>(decoder, this, &GPUConnectionToWebProcess::createRenderingBackend);
86 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::ReleaseRenderingBackend::name())
87 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::ReleaseRenderingBackend>(decoder, this, &GPUConnectionToWebProcess::releaseRenderingBackend);
88#if ENABLE(WEBGL)
89 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::CreateGraphicsContextGL::name())
90 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::CreateGraphicsContextGL>(decoder, this, &GPUConnectionToWebProcess::createGraphicsContextGL);
91#endif
92#if ENABLE(WEBGL)
93 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::ReleaseGraphicsContextGL::name())
94 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::ReleaseGraphicsContextGL>(decoder, this, &GPUConnectionToWebProcess::releaseGraphicsContextGL);
95#endif
96 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::ClearNowPlayingInfo::name())
97 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::ClearNowPlayingInfo>(decoder, this, &GPUConnectionToWebProcess::clearNowPlayingInfo);
98 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::SetNowPlayingInfo::name())
99 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::SetNowPlayingInfo>(decoder, this, &GPUConnectionToWebProcess::setNowPlayingInfo);
100#if PLATFORM(IOS_FAMILY)
101 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::EnsureMediaSessionHelper::name())
102 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::EnsureMediaSessionHelper>(decoder, this, &GPUConnectionToWebProcess::ensureMediaSessionHelper);
103#endif
104#if ENABLE(VP9)
105 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::EnableVP9Decoders::name())
106 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::EnableVP9Decoders>(decoder, this, &GPUConnectionToWebProcess::enableVP9Decoders);
107#endif
108#if HAVE(VISIBILITY_PROPAGATION_VIEW)
109 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::CreateVisibilityPropagationContextForPage::name())
110 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::CreateVisibilityPropagationContextForPage>(decoder, this, &GPUConnectionToWebProcess::createVisibilityPropagationContextForPage);
111#endif
112#if HAVE(VISIBILITY_PROPAGATION_VIEW)
113 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::DestroyVisibilityPropagationContextForPage::name())
114 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::DestroyVisibilityPropagationContextForPage>(decoder, this, &GPUConnectionToWebProcess::destroyVisibilityPropagationContextForPage);
115#endif
116 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::SetMediaOverridesForTesting::name())
117 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::SetMediaOverridesForTesting>(decoder, this, &GPUConnectionToWebProcess::setMediaOverridesForTesting);
118 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::CreateAudioHardwareListener::name())
119 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::CreateAudioHardwareListener>(decoder, this, &GPUConnectionToWebProcess::createAudioHardwareListener);
120 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::ReleaseAudioHardwareListener::name())
121 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::ReleaseAudioHardwareListener>(decoder, this, &GPUConnectionToWebProcess::releaseAudioHardwareListener);
122 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::CreateRemoteCommandListener::name())
123 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::CreateRemoteCommandListener>(decoder, this, &GPUConnectionToWebProcess::createRemoteCommandListener);
124 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::ReleaseRemoteCommandListener::name())
125 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::ReleaseRemoteCommandListener>(decoder, this, &GPUConnectionToWebProcess::releaseRemoteCommandListener);
126 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::SetUserPreferredLanguages::name())
127 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::SetUserPreferredLanguages>(decoder, this, &GPUConnectionToWebProcess::setUserPreferredLanguages);
128 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::ConfigureLoggingChannel::name())
129 return IPC::handleMessage<Messages::GPUConnectionToWebProcess::ConfigureLoggingChannel>(decoder, this, &GPUConnectionToWebProcess::configureLoggingChannel);
130 if (dispatchMessage(connection, decoder))
131 return;
132 UNUSED_PARAM(connection);
133 UNUSED_PARAM(decoder);
134#if ENABLE(IPC_TESTING_API)
135 if (connection.ignoreInvalidMessageForTesting())
136 return;
137#endif // ENABLE(IPC_TESTING_API)
138 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
139}

141bool GPUConnectionToWebProcess::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
142{
143 auto protectedThis = makeRef(*this);
144#if USE(AUDIO_SESSION)
145 if (decoder.messageName() == Messages::GPUConnectionToWebProcess::EnsureAudioSession::name())
146 return IPC::handleMessageSynchronous<Messages::GPUConnectionToWebProcess::EnsureAudioSession>(connection, decoder, replyEncoder, this, &GPUConnectionToWebProcess::ensureAudioSession);
147#endif
148 if (dispatchSyncMessage(connection, decoder, replyEncoder))
149 return true;
150 UNUSED_PARAM(connection);
151 UNUSED_PARAM(decoder);
152 UNUSED_PARAM(replyEncoder);
153#if ENABLE(IPC_TESTING_API)
154 if (connection.ignoreInvalidMessageForTesting())
155 return false;
156#endif // ENABLE(IPC_TESTING_API)
157 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
158 return false;
159}
