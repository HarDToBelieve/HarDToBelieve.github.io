58void CreateGPUConnectionToWebProcess::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<IPC::Attachment>&&)>&& completionHandler)
59{
60 std::optional<std::optional<IPC::Attachment>> connectionIdentifier;
61 decoder >> connectionIdentifier;
62 if (!connectionIdentifier) {
63 ASSERT_NOT_REACHED();
64 cancelReply(WTFMove(completionHandler));
65 return;
66 }
67 completionHandler(WTFMove(*connectionIdentifier));
68}
69
70void CreateGPUConnectionToWebProcess::cancelReply(CompletionHandler<void(std::optional<IPC::Attachment>&&)>&& completionHandler)
71{
72 completionHandler(IPC::AsyncReplyError<std::optional<IPC::Attachment>>::create());
73}
74
75void CreateGPUConnectionToWebProcess::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<IPC::Attachment>& connectionIdentifier)
76{
77 encoder.get() << connectionIdentifier;
78 connection.sendSyncReply(WTFMove(encoder));
79}

121void GPUProcess::didReceiveGPUProcessMessage(IPC::Connection& connection, IPC::Decoder& decoder)
122{
123 auto protectedThis = makeRef(*this);
124 if (decoder.messageName() == Messages::GPUProcess::InitializeGPUProcess::name())
125 return IPC::handleMessage<Messages::GPUProcess::InitializeGPUProcess>(decoder, this, &GPUProcess::initializeGPUProcess);
126 if (decoder.messageName() == Messages::GPUProcess::CreateGPUConnectionToWebProcess::name())
127 return IPC::handleMessageAsync<Messages::GPUProcess::CreateGPUConnectionToWebProcess>(connection, decoder, this, &GPUProcess::createGPUConnectionToWebProcess);
128 if (decoder.messageName() == Messages::GPUProcess::ProcessDidTransitionToForeground::name())
129 return IPC::handleMessage<Messages::GPUProcess::ProcessDidTransitionToForeground>(decoder, this, &GPUProcess::processDidTransitionToForeground);
130 if (decoder.messageName() == Messages::GPUProcess::ProcessDidTransitionToBackground::name())
131 return IPC::handleMessage<Messages::GPUProcess::ProcessDidTransitionToBackground>(decoder, this, &GPUProcess::processDidTransitionToBackground);
132 if (decoder.messageName() == Messages::GPUProcess::PrepareToSuspend::name())
133 return IPC::handleMessageAsync<Messages::GPUProcess::PrepareToSuspend>(connection, decoder, this, &GPUProcess::prepareToSuspend);
134 if (decoder.messageName() == Messages::GPUProcess::ProcessDidResume::name())
135 return IPC::handleMessage<Messages::GPUProcess::ProcessDidResume>(decoder, this, &GPUProcess::processDidResume);
136 if (decoder.messageName() == Messages::GPUProcess::AddSession::name())
137 return IPC::handleMessage<Messages::GPUProcess::AddSession>(decoder, this, &GPUProcess::addSession);
138 if (decoder.messageName() == Messages::GPUProcess::RemoveSession::name())
139 return IPC::handleMessage<Messages::GPUProcess::RemoveSession>(decoder, this, &GPUProcess::removeSession);
140#if ENABLE(MEDIA_STREAM)
141 if (decoder.messageName() == Messages::GPUProcess::SetMockCaptureDevicesEnabled::name())
142 return IPC::handleMessage<Messages::GPUProcess::SetMockCaptureDevicesEnabled>(decoder, this, &GPUProcess::setMockCaptureDevicesEnabled);
143#endif
144#if ENABLE(MEDIA_STREAM)
145 if (decoder.messageName() == Messages::GPUProcess::SetOrientationForMediaCapture::name())
146 return IPC::handleMessage<Messages::GPUProcess::SetOrientationForMediaCapture>(decoder, this, &GPUProcess::setOrientationForMediaCapture);
147#endif
148#if ENABLE(MEDIA_STREAM)
149 if (decoder.messageName() == Messages::GPUProcess::UpdateSandboxAccess::name())
150 return IPC::handleMessage<Messages::GPUProcess::UpdateSandboxAccess>(decoder, this, &GPUProcess::updateSandboxAccess);
151#endif
152#if ENABLE(MEDIA_STREAM)
153 if (decoder.messageName() == Messages::GPUProcess::UpdateCaptureAccess::name())
154 return IPC::handleMessageAsync<Messages::GPUProcess::UpdateCaptureAccess>(connection, decoder, this, &GPUProcess::updateCaptureAccess);
155#endif
156#if ENABLE(MEDIA_STREAM)
157 if (decoder.messageName() == Messages::GPUProcess::AddMockMediaDevice::name())
158 return IPC::handleMessage<Messages::GPUProcess::AddMockMediaDevice>(decoder, this, &GPUProcess::addMockMediaDevice);
159#endif
160#if ENABLE(MEDIA_STREAM)
161 if (decoder.messageName() == Messages::GPUProcess::ClearMockMediaDevices::name())
162 return IPC::handleMessage<Messages::GPUProcess::ClearMockMediaDevices>(decoder, this, &GPUProcess::clearMockMediaDevices);
163#endif
164#if ENABLE(MEDIA_STREAM)
165 if (decoder.messageName() == Messages::GPUProcess::RemoveMockMediaDevice::name())
166 return IPC::handleMessage<Messages::GPUProcess::RemoveMockMediaDevice>(decoder, this, &GPUProcess::removeMockMediaDevice);
167#endif
168#if ENABLE(MEDIA_STREAM)
169 if (decoder.messageName() == Messages::GPUProcess::ResetMockMediaDevices::name())
170 return IPC::handleMessage<Messages::GPUProcess::ResetMockMediaDevices>(decoder, this, &GPUProcess::resetMockMediaDevices);
171#endif
172#if PLATFORM(MAC)
173 if (decoder.messageName() == Messages::GPUProcess::DisplayConfigurationChanged::name())
174 return IPC::handleMessage<Messages::GPUProcess::DisplayConfigurationChanged>(decoder, this, &GPUProcess::displayConfigurationChanged);
175#endif
176#if ENABLE(MEDIA_SOURCE) && ENABLE(VP9)
177 if (decoder.messageName() == Messages::GPUProcess::SetWebMParserEnabled::name())
178 return IPC::handleMessage<Messages::GPUProcess::SetWebMParserEnabled>(decoder, this, &GPUProcess::setWebMParserEnabled);
179#endif
180#if ENABLE(WEBM_FORMAT_READER)
181 if (decoder.messageName() == Messages::GPUProcess::SetWebMFormatReaderEnabled::name())
182 return IPC::handleMessage<Messages::GPUProcess::SetWebMFormatReaderEnabled>(decoder, this, &GPUProcess::setWebMFormatReaderEnabled);
183#endif
184#if ENABLE(OPUS)
185 if (decoder.messageName() == Messages::GPUProcess::SetOpusDecoderEnabled::name())
186 return IPC::handleMessage<Messages::GPUProcess::SetOpusDecoderEnabled>(decoder, this, &GPUProcess::setOpusDecoderEnabled);
187#endif
188#if ENABLE(VORBIS)
189 if (decoder.messageName() == Messages::GPUProcess::SetVorbisDecoderEnabled::name())
190 return IPC::handleMessage<Messages::GPUProcess::SetVorbisDecoderEnabled>(decoder, this, &GPUProcess::setVorbisDecoderEnabled);
191#endif
192 UNUSED_PARAM(connection);
193 UNUSED_PARAM(decoder);
194#if ENABLE(IPC_TESTING_API)
195 if (connection.ignoreInvalidMessageForTesting())
196 return;
197#endif // ENABLE(IPC_TESTING_API)
198 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
199}
