44void LibWebRTCCodecsProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
45{
46 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::CreateH264Decoder::name())
47 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::CreateH264Decoder>(decoder, this, &LibWebRTCCodecsProxy::createH264Decoder);
48 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::CreateH265Decoder::name())
49 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::CreateH265Decoder>(decoder, this, &LibWebRTCCodecsProxy::createH265Decoder);
50 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::CreateVP9Decoder::name())
51 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::CreateVP9Decoder>(decoder, this, &LibWebRTCCodecsProxy::createVP9Decoder);
52 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::ReleaseDecoder::name())
53 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::ReleaseDecoder>(decoder, this, &LibWebRTCCodecsProxy::releaseDecoder);
54 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::DecodeFrame::name())
55 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::DecodeFrame>(decoder, this, &LibWebRTCCodecsProxy::decodeFrame);
56 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::SetFrameSize::name())
57 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::SetFrameSize>(decoder, this, &LibWebRTCCodecsProxy::setFrameSize);
58 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::CreateEncoder::name())
59 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::CreateEncoder>(decoder, this, &LibWebRTCCodecsProxy::createEncoder);
60 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::ReleaseEncoder::name())
61 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::ReleaseEncoder>(decoder, this, &LibWebRTCCodecsProxy::releaseEncoder);
62 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::InitializeEncoder::name())
63 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::InitializeEncoder>(decoder, this, &LibWebRTCCodecsProxy::initializeEncoder);
64 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::EncodeFrame::name())
65 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::EncodeFrame>(decoder, this, &LibWebRTCCodecsProxy::encodeFrame);
66 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::SetEncodeRates::name())
67 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::SetEncodeRates>(decoder, this, &LibWebRTCCodecsProxy::setEncodeRates);
68 if (decoder.messageName() == Messages::LibWebRTCCodecsProxy::SetRTCLoggingLevel::name())
69 return IPC::handleMessage<Messages::LibWebRTCCodecsProxy::SetRTCLoggingLevel>(decoder, this, &LibWebRTCCodecsProxy::setRTCLoggingLevel);
70 UNUSED_PARAM(connection);
71 UNUSED_PARAM(decoder);
72#if ENABLE(IPC_TESTING_API)
73 if (connection.ignoreInvalidMessageForTesting())
74 return;
75#endif // ENABLE(IPC_TESTING_API)
76 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
77}
