39void LibWebRTCNetwork::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
40{
41 if (decoder.messageName() == Messages::LibWebRTCNetwork::SignalReadPacket::name())
42 return IPC::handleMessage<Messages::LibWebRTCNetwork::SignalReadPacket>(decoder, this, &LibWebRTCNetwork::signalReadPacket);
43 if (decoder.messageName() == Messages::LibWebRTCNetwork::SignalSentPacket::name())
44 return IPC::handleMessage<Messages::LibWebRTCNetwork::SignalSentPacket>(decoder, this, &LibWebRTCNetwork::signalSentPacket);
45 if (decoder.messageName() == Messages::LibWebRTCNetwork::SignalAddressReady::name())
46 return IPC::handleMessage<Messages::LibWebRTCNetwork::SignalAddressReady>(decoder, this, &LibWebRTCNetwork::signalAddressReady);
47 if (decoder.messageName() == Messages::LibWebRTCNetwork::SignalConnect::name())
48 return IPC::handleMessage<Messages::LibWebRTCNetwork::SignalConnect>(decoder, this, &LibWebRTCNetwork::signalConnect);
49 if (decoder.messageName() == Messages::LibWebRTCNetwork::SignalClose::name())
50 return IPC::handleMessage<Messages::LibWebRTCNetwork::SignalClose>(decoder, this, &LibWebRTCNetwork::signalClose);
51 if (decoder.messageName() == Messages::LibWebRTCNetwork::SignalNewConnection::name())
52 return IPC::handleMessage<Messages::LibWebRTCNetwork::SignalNewConnection>(decoder, this, &LibWebRTCNetwork::signalNewConnection);
53 UNUSED_PARAM(connection);
54 UNUSED_PARAM(decoder);
55#if ENABLE(IPC_TESTING_API)
56 if (connection.ignoreInvalidMessageForTesting())
57 return;
58#endif // ENABLE(IPC_TESTING_API)
59 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
60}
