81void SendH2Ping::callReply(IPC::Decoder& decoder, CompletionHandler<void(Expected<Seconds, WebCore::ResourceError>&&)>&& completionHandler)
82{
83 std::optional<Expected<Seconds, WebCore::ResourceError>> result;
84 decoder >> result;
85 if (!result) {
86 ASSERT_NOT_REACHED();
87 cancelReply(WTFMove(completionHandler));
88 return;
89 }
90 completionHandler(WTFMove(*result));
91}
92
93void SendH2Ping::cancelReply(CompletionHandler<void(Expected<Seconds, WebCore::ResourceError>&&)>&& completionHandler)
94{
95 completionHandler(IPC::AsyncReplyError<Expected<Seconds, WebCore::ResourceError>>::create());
96}
97
98void SendH2Ping::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const Expected<Seconds, WebCore::ResourceError>& result)
99{
100 encoder.get() << result;
101 connection.sendSyncReply(WTFMove(encoder));
102}
