65void ConnectToRTCDataChannelRemoteSource::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<bool>&&)>&& completionHandler)
66{
67 std::optional<std::optional<bool>> result;
68 decoder >> result;
69 if (!result) {
70 ASSERT_NOT_REACHED();
71 cancelReply(WTFMove(completionHandler));
72 return;
73 }
74 completionHandler(WTFMove(*result));
75}
76
77void ConnectToRTCDataChannelRemoteSource::cancelReply(CompletionHandler<void(std::optional<bool>&&)>&& completionHandler)
78{
79 completionHandler(IPC::AsyncReplyError<std::optional<bool>>::create());
80}
81
82void ConnectToRTCDataChannelRemoteSource::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<bool>& result)
83{
84 encoder.get() << result;
85 connection.sendSyncReply(WTFMove(encoder));
86}

119void NetworkProcessConnection::didReceiveNetworkProcessConnectionMessage(IPC::Connection& connection, IPC::Decoder& decoder)
120{
121 auto protectedThis = makeRef(*this);
122#if ENABLE(SHAREABLE_RESOURCE)
123 if (decoder.messageName() == Messages::NetworkProcessConnection::DidCacheResource::name())
124 return IPC::handleMessage<Messages::NetworkProcessConnection::DidCacheResource>(decoder, this, &NetworkProcessConnection::didCacheResource);
125#endif
126 if (decoder.messageName() == Messages::NetworkProcessConnection::DidFinishPingLoad::name())
127 return IPC::handleMessage<Messages::NetworkProcessConnection::DidFinishPingLoad>(decoder, this, &NetworkProcessConnection::didFinishPingLoad);
128 if (decoder.messageName() == Messages::NetworkProcessConnection::DidFinishPreconnection::name())
129 return IPC::handleMessage<Messages::NetworkProcessConnection::DidFinishPreconnection>(decoder, this, &NetworkProcessConnection::didFinishPreconnection);
130 if (decoder.messageName() == Messages::NetworkProcessConnection::SetOnLineState::name())
131 return IPC::handleMessage<Messages::NetworkProcessConnection::SetOnLineState>(decoder, this, &NetworkProcessConnection::setOnLineState);
132 if (decoder.messageName() == Messages::NetworkProcessConnection::CookieAcceptPolicyChanged::name())
133 return IPC::handleMessage<Messages::NetworkProcessConnection::CookieAcceptPolicyChanged>(decoder, this, &NetworkProcessConnection::cookieAcceptPolicyChanged);
134#if HAVE(COOKIE_CHANGE_LISTENER_API)
135 if (decoder.messageName() == Messages::NetworkProcessConnection::CookiesAdded::name())
136 return IPC::handleMessage<Messages::NetworkProcessConnection::CookiesAdded>(decoder, this, &NetworkProcessConnection::cookiesAdded);
137#endif
138#if HAVE(COOKIE_CHANGE_LISTENER_API)
139 if (decoder.messageName() == Messages::NetworkProcessConnection::CookiesDeleted::name())
140 return IPC::handleMessage<Messages::NetworkProcessConnection::CookiesDeleted>(decoder, this, &NetworkProcessConnection::cookiesDeleted);
141#endif
142#if HAVE(COOKIE_CHANGE_LISTENER_API)
143 if (decoder.messageName() == Messages::NetworkProcessConnection::AllCookiesDeleted::name())
144 return IPC::handleMessage<Messages::NetworkProcessConnection::AllCookiesDeleted>(decoder, this, &NetworkProcessConnection::allCookiesDeleted);
145#endif
146#if ENABLE(WEB_RTC)
147 if (decoder.messageName() == Messages::NetworkProcessConnection::ConnectToRTCDataChannelRemoteSource::name())
148 return IPC::handleMessageAsync<Messages::NetworkProcessConnection::ConnectToRTCDataChannelRemoteSource>(connection, decoder, this, &NetworkProcessConnection::connectToRTCDataChannelRemoteSource);
149#endif
150 if (decoder.messageName() == Messages::NetworkProcessConnection::CheckProcessLocalPortForActivity::name())
151 return IPC::handleMessageAsync<Messages::NetworkProcessConnection::CheckProcessLocalPortForActivity>(connection, decoder, this, &NetworkProcessConnection::checkProcessLocalPortForActivity);
152 if (decoder.messageName() == Messages::NetworkProcessConnection::MessagesAvailableForPort::name())
153 return IPC::handleMessage<Messages::NetworkProcessConnection::MessagesAvailableForPort>(decoder, this, &NetworkProcessConnection::messagesAvailableForPort);
154 if (decoder.messageName() == Messages::NetworkProcessConnection::BroadcastConsoleMessage::name())
155 return IPC::handleMessage<Messages::NetworkProcessConnection::BroadcastConsoleMessage>(decoder, this, &NetworkProcessConnection::broadcastConsoleMessage);
156 UNUSED_PARAM(connection);
157 UNUSED_PARAM(decoder);
158#if ENABLE(IPC_TESTING_API)
159 if (connection.ignoreInvalidMessageForTesting())
160 return;
161#endif // ENABLE(IPC_TESTING_API)
162 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
163}
