44void NetworkRTCProvider::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
45{
46 auto protectedThis = makeRef(*this);
47 if (decoder.messageName() == Messages::NetworkRTCProvider::CreateUDPSocket::name())
48 return IPC::handleMessage<Messages::NetworkRTCProvider::CreateUDPSocket>(decoder, this, &NetworkRTCProvider::createUDPSocket);
49 if (decoder.messageName() == Messages::NetworkRTCProvider::CreateServerTCPSocket::name())
50 return IPC::handleMessage<Messages::NetworkRTCProvider::CreateServerTCPSocket>(decoder, this, &NetworkRTCProvider::createServerTCPSocket);
51 if (decoder.messageName() == Messages::NetworkRTCProvider::CreateClientTCPSocket::name())
52 return IPC::handleMessage<Messages::NetworkRTCProvider::CreateClientTCPSocket>(decoder, this, &NetworkRTCProvider::createClientTCPSocket);
53 if (decoder.messageName() == Messages::NetworkRTCProvider::WrapNewTCPConnection::name())
54 return IPC::handleMessage<Messages::NetworkRTCProvider::WrapNewTCPConnection>(decoder, this, &NetworkRTCProvider::wrapNewTCPConnection);
55 if (decoder.messageName() == Messages::NetworkRTCProvider::SetPlatformTCPSocketsEnabled::name())
56 return IPC::handleMessage<Messages::NetworkRTCProvider::SetPlatformTCPSocketsEnabled>(decoder, this, &NetworkRTCProvider::setPlatformTCPSocketsEnabled);
57 if (decoder.messageName() == Messages::NetworkRTCProvider::SetPlatformUDPSocketsEnabled::name())
58 return IPC::handleMessage<Messages::NetworkRTCProvider::SetPlatformUDPSocketsEnabled>(decoder, this, &NetworkRTCProvider::setPlatformUDPSocketsEnabled);
59 if (decoder.messageName() == Messages::NetworkRTCProvider::CreateResolver::name())
60 return IPC::handleMessage<Messages::NetworkRTCProvider::CreateResolver>(decoder, this, &NetworkRTCProvider::createResolver);
61 if (decoder.messageName() == Messages::NetworkRTCProvider::StopResolver::name())
62 return IPC::handleMessage<Messages::NetworkRTCProvider::StopResolver>(decoder, this, &NetworkRTCProvider::stopResolver);
63 if (decoder.messageName() == Messages::NetworkRTCProvider::SendToSocket::name())
64 return IPC::handleMessage<Messages::NetworkRTCProvider::SendToSocket>(decoder, this, &NetworkRTCProvider::sendToSocket);
65 if (decoder.messageName() == Messages::NetworkRTCProvider::CloseSocket::name())
66 return IPC::handleMessage<Messages::NetworkRTCProvider::CloseSocket>(decoder, this, &NetworkRTCProvider::closeSocket);
67 if (decoder.messageName() == Messages::NetworkRTCProvider::SetSocketOption::name())
68 return IPC::handleMessage<Messages::NetworkRTCProvider::SetSocketOption>(decoder, this, &NetworkRTCProvider::setSocketOption);
69 UNUSED_PARAM(connection);
70 UNUSED_PARAM(decoder);
71#if ENABLE(IPC_TESTING_API)
72 if (connection.ignoreInvalidMessageForTesting())
73 return;
74#endif // ENABLE(IPC_TESTING_API)
75 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
76}
