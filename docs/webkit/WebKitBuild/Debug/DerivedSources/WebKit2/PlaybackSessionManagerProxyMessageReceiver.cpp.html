44void PlaybackSessionManagerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
45{
46 auto protectedThis = makeRef(*this);
47 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::CurrentTimeChanged::name())
48 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::CurrentTimeChanged>(decoder, this, &PlaybackSessionManagerProxy::currentTimeChanged);
49 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::BufferedTimeChanged::name())
50 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::BufferedTimeChanged>(decoder, this, &PlaybackSessionManagerProxy::bufferedTimeChanged);
51 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::SeekableRangesVectorChanged::name())
52 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::SeekableRangesVectorChanged>(decoder, this, &PlaybackSessionManagerProxy::seekableRangesVectorChanged);
53 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::CanPlayFastReverseChanged::name())
54 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::CanPlayFastReverseChanged>(decoder, this, &PlaybackSessionManagerProxy::canPlayFastReverseChanged);
55 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::AudioMediaSelectionOptionsChanged::name())
56 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::AudioMediaSelectionOptionsChanged>(decoder, this, &PlaybackSessionManagerProxy::audioMediaSelectionOptionsChanged);
57 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::LegibleMediaSelectionOptionsChanged::name())
58 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::LegibleMediaSelectionOptionsChanged>(decoder, this, &PlaybackSessionManagerProxy::legibleMediaSelectionOptionsChanged);
59 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::AudioMediaSelectionIndexChanged::name())
60 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::AudioMediaSelectionIndexChanged>(decoder, this, &PlaybackSessionManagerProxy::audioMediaSelectionIndexChanged);
61 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::LegibleMediaSelectionIndexChanged::name())
62 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::LegibleMediaSelectionIndexChanged>(decoder, this, &PlaybackSessionManagerProxy::legibleMediaSelectionIndexChanged);
63 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::ExternalPlaybackPropertiesChanged::name())
64 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::ExternalPlaybackPropertiesChanged>(decoder, this, &PlaybackSessionManagerProxy::externalPlaybackPropertiesChanged);
65 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::WirelessVideoPlaybackDisabledChanged::name())
66 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::WirelessVideoPlaybackDisabledChanged>(decoder, this, &PlaybackSessionManagerProxy::wirelessVideoPlaybackDisabledChanged);
67 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::DurationChanged::name())
68 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::DurationChanged>(decoder, this, &PlaybackSessionManagerProxy::durationChanged);
69 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::PlaybackStartedTimeChanged::name())
70 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::PlaybackStartedTimeChanged>(decoder, this, &PlaybackSessionManagerProxy::playbackStartedTimeChanged);
71 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::RateChanged::name())
72 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::RateChanged>(decoder, this, &PlaybackSessionManagerProxy::rateChanged);
73 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::MutedChanged::name())
74 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::MutedChanged>(decoder, this, &PlaybackSessionManagerProxy::mutedChanged);
75 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::VolumeChanged::name())
76 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::VolumeChanged>(decoder, this, &PlaybackSessionManagerProxy::volumeChanged);
77 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::PictureInPictureSupportedChanged::name())
78 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::PictureInPictureSupportedChanged>(decoder, this, &PlaybackSessionManagerProxy::pictureInPictureSupportedChanged);
79 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::SetUpPlaybackControlsManagerWithID::name())
80 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::SetUpPlaybackControlsManagerWithID>(decoder, this, &PlaybackSessionManagerProxy::setUpPlaybackControlsManagerWithID);
81 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::ClearPlaybackControlsManager::name())
82 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::ClearPlaybackControlsManager>(decoder, this, &PlaybackSessionManagerProxy::clearPlaybackControlsManager);
83 if (decoder.messageName() == Messages::PlaybackSessionManagerProxy::HandleControlledElementIDResponse::name())
84 return IPC::handleMessage<Messages::PlaybackSessionManagerProxy::HandleControlledElementIDResponse>(decoder, this, &PlaybackSessionManagerProxy::handleControlledElementIDResponse);
85 UNUSED_PARAM(connection);
86 UNUSED_PARAM(decoder);
87#if ENABLE(IPC_TESTING_API)
88 if (connection.ignoreInvalidMessageForTesting())
89 return;
90#endif // ENABLE(IPC_TESTING_API)
91 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
92}
