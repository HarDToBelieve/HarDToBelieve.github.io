51bool PluginProcessConnection::didReceiveSyncPluginProcessConnectionMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
52{
53 auto protectedThis = makeRef(*this);
54 if (decoder.messageName() == Messages::PluginProcessConnection::SetException::name())
55 return IPC::handleMessageSynchronous<Messages::PluginProcessConnection::SetException>(connection, decoder, replyEncoder, this, &PluginProcessConnection::setException);
56 UNUSED_PARAM(connection);
57 UNUSED_PARAM(decoder);
58 UNUSED_PARAM(replyEncoder);
59#if ENABLE(IPC_TESTING_API)
60 if (connection.ignoreInvalidMessageForTesting())
61 return false;
62#endif // ENABLE(IPC_TESTING_API)
63 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
64 return false;
65}
