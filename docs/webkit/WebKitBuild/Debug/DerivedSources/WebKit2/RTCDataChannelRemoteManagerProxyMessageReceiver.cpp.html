39void RTCDataChannelRemoteManagerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
40{
41 if (decoder.messageName() == Messages::RTCDataChannelRemoteManagerProxy::SendData::name())
42 return IPC::handleMessage<Messages::RTCDataChannelRemoteManagerProxy::SendData>(decoder, this, &RTCDataChannelRemoteManagerProxy::sendData);
43 if (decoder.messageName() == Messages::RTCDataChannelRemoteManagerProxy::Close::name())
44 return IPC::handleMessage<Messages::RTCDataChannelRemoteManagerProxy::Close>(decoder, this, &RTCDataChannelRemoteManagerProxy::close);
45 if (decoder.messageName() == Messages::RTCDataChannelRemoteManagerProxy::ChangeReadyState::name())
46 return IPC::handleMessage<Messages::RTCDataChannelRemoteManagerProxy::ChangeReadyState>(decoder, this, &RTCDataChannelRemoteManagerProxy::changeReadyState);
47 if (decoder.messageName() == Messages::RTCDataChannelRemoteManagerProxy::ReceiveData::name())
48 return IPC::handleMessage<Messages::RTCDataChannelRemoteManagerProxy::ReceiveData>(decoder, this, &RTCDataChannelRemoteManagerProxy::receiveData);
49 if (decoder.messageName() == Messages::RTCDataChannelRemoteManagerProxy::DetectError::name())
50 return IPC::handleMessage<Messages::RTCDataChannelRemoteManagerProxy::DetectError>(decoder, this, &RTCDataChannelRemoteManagerProxy::detectError);
51 if (decoder.messageName() == Messages::RTCDataChannelRemoteManagerProxy::BufferedAmountIsDecreasing::name())
52 return IPC::handleMessage<Messages::RTCDataChannelRemoteManagerProxy::BufferedAmountIsDecreasing>(decoder, this, &RTCDataChannelRemoteManagerProxy::bufferedAmountIsDecreasing);
53 UNUSED_PARAM(connection);
54 UNUSED_PARAM(decoder);
55#if ENABLE(IPC_TESTING_API)
56 if (connection.ignoreInvalidMessageForTesting())
57 return;
58#endif // ENABLE(IPC_TESTING_API)
59 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
60}
