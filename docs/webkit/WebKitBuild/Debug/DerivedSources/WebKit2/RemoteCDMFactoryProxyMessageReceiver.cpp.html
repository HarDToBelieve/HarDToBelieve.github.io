41void CreateCDM::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebKit::RemoteCDMIdentifier& identifier, const WebKit::RemoteCDMConfiguration& configuration)
42{
43 encoder.get() << identifier;
44 encoder.get() << configuration;
45 connection.sendSyncReply(WTFMove(encoder));
46}

48void SupportsKeySystem::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, bool result)
49{
50 encoder.get() << result;
51 connection.sendSyncReply(WTFMove(encoder));
52}

61bool RemoteCDMFactoryProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
62{
63 if (decoder.messageName() == Messages::RemoteCDMFactoryProxy::CreateCDM::name())
64 return IPC::handleMessageSynchronous<Messages::RemoteCDMFactoryProxy::CreateCDM>(connection, decoder, replyEncoder, this, &RemoteCDMFactoryProxy::createCDM);
65 if (decoder.messageName() == Messages::RemoteCDMFactoryProxy::SupportsKeySystem::name())
66 return IPC::handleMessageSynchronous<Messages::RemoteCDMFactoryProxy::SupportsKeySystem>(connection, decoder, replyEncoder, this, &RemoteCDMFactoryProxy::supportsKeySystem);
67 UNUSED_PARAM(connection);
68 UNUSED_PARAM(decoder);
69 UNUSED_PARAM(replyEncoder);
70#if ENABLE(IPC_TESTING_API)
71 if (connection.ignoreInvalidMessageForTesting())
72 return false;
73#endif // ENABLE(IPC_TESTING_API)
74 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
75 return false;
76}
