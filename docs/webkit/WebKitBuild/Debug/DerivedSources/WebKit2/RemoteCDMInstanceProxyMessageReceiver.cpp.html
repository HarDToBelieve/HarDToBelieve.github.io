50void InitializeWithConfiguration::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::CDMInstance::SuccessValue&&)>&& completionHandler)
51{
52 std::optional<WebCore::CDMInstance::SuccessValue> success;
53 decoder >> success;
54 if (!success) {
55 ASSERT_NOT_REACHED();
56 cancelReply(WTFMove(completionHandler));
57 return;
58 }
59 completionHandler(WTFMove(*success));
60}

62void InitializeWithConfiguration::cancelReply(CompletionHandler<void(WebCore::CDMInstance::SuccessValue&&)>&& completionHandler)
63{
64 completionHandler(IPC::AsyncReplyError<WebCore::CDMInstance::SuccessValue>::create());
65}

102void RemoteCDMInstanceProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
103{
104 if (decoder.messageName() == Messages::RemoteCDMInstanceProxy::InitializeWithConfiguration::name())
105 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceProxy::InitializeWithConfiguration>(connection, decoder, this, &RemoteCDMInstanceProxy::initializeWithConfiguration);
106 if (decoder.messageName() == Messages::RemoteCDMInstanceProxy::SetServerCertificate::name())
107 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceProxy::SetServerCertificate>(connection, decoder, this, &RemoteCDMInstanceProxy::setServerCertificate);
108 if (decoder.messageName() == Messages::RemoteCDMInstanceProxy::SetStorageDirectory::name())
109 return IPC::handleMessage<Messages::RemoteCDMInstanceProxy::SetStorageDirectory>(decoder, this, &RemoteCDMInstanceProxy::setStorageDirectory);
110 UNUSED_PARAM(connection);
111 UNUSED_PARAM(decoder);
112#if ENABLE(IPC_TESTING_API)
113 if (connection.ignoreInvalidMessageForTesting())
114 return;
115#endif // ENABLE(IPC_TESTING_API)
116 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
117}
118
119bool RemoteCDMInstanceProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
120{
121 if (decoder.messageName() == Messages::RemoteCDMInstanceProxy::CreateSession::name())
122 return IPC::handleMessageSynchronous<Messages::RemoteCDMInstanceProxy::CreateSession>(connection, decoder, replyEncoder, this, &RemoteCDMInstanceProxy::createSession);
123 UNUSED_PARAM(connection);
124 UNUSED_PARAM(decoder);
125 UNUSED_PARAM(replyEncoder);
126#if ENABLE(IPC_TESTING_API)
127 if (connection.ignoreInvalidMessageForTesting())
128 return false;
129#endif // ENABLE(IPC_TESTING_API)
130 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
131 return false;
132}
