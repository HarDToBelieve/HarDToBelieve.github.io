76void RemoteImageDecoderAVFProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
77{
78 if (decoder.messageName() == Messages::RemoteImageDecoderAVFProxy::DeleteDecoder::name())
79 return IPC::handleMessage<Messages::RemoteImageDecoderAVFProxy::DeleteDecoder>(decoder, this, &RemoteImageDecoderAVFProxy::deleteDecoder);
80 if (decoder.messageName() == Messages::RemoteImageDecoderAVFProxy::SetExpectedContentSize::name())
81 return IPC::handleMessage<Messages::RemoteImageDecoderAVFProxy::SetExpectedContentSize>(decoder, this, &RemoteImageDecoderAVFProxy::setExpectedContentSize);
82 if (decoder.messageName() == Messages::RemoteImageDecoderAVFProxy::ClearFrameBufferCache::name())
83 return IPC::handleMessage<Messages::RemoteImageDecoderAVFProxy::ClearFrameBufferCache>(decoder, this, &RemoteImageDecoderAVFProxy::clearFrameBufferCache);
84 UNUSED_PARAM(connection);
85 UNUSED_PARAM(decoder);
86#if ENABLE(IPC_TESTING_API)
87 if (connection.ignoreInvalidMessageForTesting())
88 return;
89#endif // ENABLE(IPC_TESTING_API)
90 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
91}
92
93bool RemoteImageDecoderAVFProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
94{
95 if (decoder.messageName() == Messages::RemoteImageDecoderAVFProxy::CreateDecoder::name())
96 return IPC::handleMessageSynchronous<Messages::RemoteImageDecoderAVFProxy::CreateDecoder>(connection, decoder, replyEncoder, this, &RemoteImageDecoderAVFProxy::createDecoder);
97 if (decoder.messageName() == Messages::RemoteImageDecoderAVFProxy::SetData::name())
98 return IPC::handleMessageSynchronous<Messages::RemoteImageDecoderAVFProxy::SetData>(connection, decoder, replyEncoder, this, &RemoteImageDecoderAVFProxy::setData);
99 if (decoder.messageName() == Messages::RemoteImageDecoderAVFProxy::CreateFrameImageAtIndex::name())
100 return IPC::handleMessageSynchronous<Messages::RemoteImageDecoderAVFProxy::CreateFrameImageAtIndex>(connection, decoder, replyEncoder, this, &RemoteImageDecoderAVFProxy::createFrameImageAtIndex);
101 UNUSED_PARAM(connection);
102 UNUSED_PARAM(decoder);
103 UNUSED_PARAM(replyEncoder);
104#if ENABLE(IPC_TESTING_API)
105 if (connection.ignoreInvalidMessageForTesting())
106 return false;
107#endif // ENABLE(IPC_TESTING_API)
108 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
109 return false;
110}
