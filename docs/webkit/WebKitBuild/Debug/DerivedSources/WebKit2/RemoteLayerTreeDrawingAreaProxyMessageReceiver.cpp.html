37void RemoteLayerTreeDrawingAreaProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
38{
39 if (decoder.messageName() == Messages::RemoteLayerTreeDrawingAreaProxy::SetPreferredFramesPerSecond::name())
40 return IPC::handleMessage<Messages::RemoteLayerTreeDrawingAreaProxy::SetPreferredFramesPerSecond>(decoder, this, &RemoteLayerTreeDrawingAreaProxy::setPreferredFramesPerSecond);
41 if (decoder.messageName() == Messages::RemoteLayerTreeDrawingAreaProxy::WillCommitLayerTree::name())
42 return IPC::handleMessage<Messages::RemoteLayerTreeDrawingAreaProxy::WillCommitLayerTree>(decoder, this, &RemoteLayerTreeDrawingAreaProxy::willCommitLayerTree);
43 if (decoder.messageName() == Messages::RemoteLayerTreeDrawingAreaProxy::CommitLayerTree::name())
44 return IPC::handleMessage<Messages::RemoteLayerTreeDrawingAreaProxy::CommitLayerTree>(decoder, this, &RemoteLayerTreeDrawingAreaProxy::commitLayerTree);
45 DrawingAreaProxy::didReceiveMessage(connection, decoder);
46}
