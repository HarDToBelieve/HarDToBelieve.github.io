61void RemoteLegacyCDMProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
62{
63 if (decoder.messageName() == Messages::RemoteLegacyCDMProxy::SetPlayerId::name())
64 return IPC::handleMessage<Messages::RemoteLegacyCDMProxy::SetPlayerId>(decoder, this, &RemoteLegacyCDMProxy::setPlayerId);
65 UNUSED_PARAM(connection);
66 UNUSED_PARAM(decoder);
67#if ENABLE(IPC_TESTING_API)
68 if (connection.ignoreInvalidMessageForTesting())
69 return;
70#endif // ENABLE(IPC_TESTING_API)
71 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
72}
73
74bool RemoteLegacyCDMProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
75{
76 if (decoder.messageName() == Messages::RemoteLegacyCDMProxy::SupportsMIMEType::name())
77 return IPC::handleMessageSynchronous<Messages::RemoteLegacyCDMProxy::SupportsMIMEType>(connection, decoder, replyEncoder, this, &RemoteLegacyCDMProxy::supportsMIMEType);
78 if (decoder.messageName() == Messages::RemoteLegacyCDMProxy::CreateSession::name())
79 return IPC::handleMessageSynchronous<Messages::RemoteLegacyCDMProxy::CreateSession>(connection, decoder, replyEncoder, this, &RemoteLegacyCDMProxy::createSession);
80 UNUSED_PARAM(connection);
81 UNUSED_PARAM(decoder);
82 UNUSED_PARAM(replyEncoder);
83#if ENABLE(IPC_TESTING_API)
84 if (connection.ignoreInvalidMessageForTesting())
85 return false;
86#endif // ENABLE(IPC_TESTING_API)
87 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
88 return false;
89}
