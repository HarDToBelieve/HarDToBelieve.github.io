78void RemoteMediaPlayerManagerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
79{
80 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::CreateMediaPlayer::name())
81 return IPC::handleMessage<Messages::RemoteMediaPlayerManagerProxy::CreateMediaPlayer>(decoder, this, &RemoteMediaPlayerManagerProxy::createMediaPlayer);
82 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::DeleteMediaPlayer::name())
83 return IPC::handleMessage<Messages::RemoteMediaPlayerManagerProxy::DeleteMediaPlayer>(decoder, this, &RemoteMediaPlayerManagerProxy::deleteMediaPlayer);
84 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::ClearMediaCache::name())
85 return IPC::handleMessage<Messages::RemoteMediaPlayerManagerProxy::ClearMediaCache>(decoder, this, &RemoteMediaPlayerManagerProxy::clearMediaCache);
86 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::ClearMediaCacheForOrigins::name())
87 return IPC::handleMessage<Messages::RemoteMediaPlayerManagerProxy::ClearMediaCacheForOrigins>(decoder, this, &RemoteMediaPlayerManagerProxy::clearMediaCacheForOrigins);
88 UNUSED_PARAM(connection);
89 UNUSED_PARAM(decoder);
90#if ENABLE(IPC_TESTING_API)
91 if (connection.ignoreInvalidMessageForTesting())
92 return;
93#endif // ENABLE(IPC_TESTING_API)
94 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
95}
96
97bool RemoteMediaPlayerManagerProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
98{
99 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::GetSupportedTypes::name())
100 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerManagerProxy::GetSupportedTypes>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerManagerProxy::getSupportedTypes);
101 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::SupportsTypeAndCodecs::name())
102 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerManagerProxy::SupportsTypeAndCodecs>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerManagerProxy::supportsTypeAndCodecs);
103 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::OriginsInMediaCache::name())
104 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerManagerProxy::OriginsInMediaCache>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerManagerProxy::originsInMediaCache);
105 if (decoder.messageName() == Messages::RemoteMediaPlayerManagerProxy::SupportsKeySystem::name())
106 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerManagerProxy::SupportsKeySystem>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerManagerProxy::supportsKeySystem);
107 UNUSED_PARAM(connection);
108 UNUSED_PARAM(decoder);
109 UNUSED_PARAM(replyEncoder);
110#if ENABLE(IPC_TESTING_API)
111 if (connection.ignoreInvalidMessageForTesting())
112 return false;
113#endif // ENABLE(IPC_TESTING_API)
114 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
115 return false;
116}
