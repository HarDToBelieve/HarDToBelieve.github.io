78void PrepareForPlayback::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<WebKit::LayerHostingContextID>&&)>&& completionHandler)
79{
80 std::optional<std::optional<WebKit::LayerHostingContextID>> inlineLayerHostingContextId;
81 decoder >> inlineLayerHostingContextId;
82 if (!inlineLayerHostingContextId) {
83 ASSERT_NOT_REACHED();
84 cancelReply(WTFMove(completionHandler));
85 return;
86 }
87 completionHandler(WTFMove(*inlineLayerHostingContextId));
88}

93}

174void PerformTaskAtMediaTime::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<MediaTime>&&, std::optional<MonotonicTime>&&)>&& completionHandler)
175{
176 std::optional<std::optional<MediaTime>> mediaTime;
177 decoder >> mediaTime;
178 if (!mediaTime) {
179 ASSERT_NOT_REACHED();
180 cancelReply(WTFMove(completionHandler));
181 return;
182 }
183 std::optional<std::optional<MonotonicTime>> monotonicTime;
184 decoder >> monotonicTime;
185 if (!monotonicTime) {
186 ASSERT_NOT_REACHED();
187 cancelReply(WTFMove(completionHandler));
188 return;
189 }
190 completionHandler(WTFMove(*mediaTime), WTFMove(*monotonicTime));
191}

196}

406bool RemoteMediaPlayerProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
407{
408 if (decoder.messageName() == Messages::RemoteMediaPlayerProxy::WouldTaintOrigin::name())
409 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerProxy::WouldTaintOrigin>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerProxy::wouldTaintOrigin);
410#if PLATFORM(IOS_FAMILY)
411 if (decoder.messageName() == Messages::RemoteMediaPlayerProxy::ErrorLog::name())
412 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerProxy::ErrorLog>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerProxy::errorLog);
413#endif
414#if PLATFORM(IOS_FAMILY)
415 if (decoder.messageName() == Messages::RemoteMediaPlayerProxy::AccessLog::name())
416 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerProxy::AccessLog>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerProxy::accessLog);
417#endif
418#if PLATFORM(COCOA)
419 if (decoder.messageName() == Messages::RemoteMediaPlayerProxy::NativeImageForCurrentTime::name())
420 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerProxy::NativeImageForCurrentTime>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerProxy::nativeImageForCurrentTime);
421#endif
422#if USE(AVFOUNDATION)
423 if (decoder.messageName() == Messages::RemoteMediaPlayerProxy::PixelBufferForCurrentTime::name())
424 return IPC::handleMessageSynchronous<Messages::RemoteMediaPlayerProxy::PixelBufferForCurrentTime>(connection, decoder, replyEncoder, this, &RemoteMediaPlayerProxy::pixelBufferForCurrentTime);
425#endif
426 UNUSED_PARAM(connection);
427 UNUSED_PARAM(decoder);
428 UNUSED_PARAM(replyEncoder);
429#if ENABLE(IPC_TESTING_API)
430 if (connection.ignoreInvalidMessageForTesting())
431 return false;
432#endif // ENABLE(IPC_TESTING_API)
433 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
434 return false;
435}
