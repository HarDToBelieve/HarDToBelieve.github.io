44void CreateRecorder::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<WebCore::ExceptionData>&&, String&&, unsigned&&, unsigned&&)>&& completionHandler)
45{
46 std::optional<std::optional<WebCore::ExceptionData>> creationError;
47 decoder >> creationError;
48 if (!creationError) {
49 ASSERT_NOT_REACHED();
50 cancelReply(WTFMove(completionHandler));
51 return;
52 }
53 std::optional<String> mimeType;
54 decoder >> mimeType;
55 if (!mimeType) {
56 ASSERT_NOT_REACHED();
57 cancelReply(WTFMove(completionHandler));
58 return;
59 }
60 std::optional<unsigned> audioBitRate;
61 decoder >> audioBitRate;
62 if (!audioBitRate) {
63 ASSERT_NOT_REACHED();
64 cancelReply(WTFMove(completionHandler));
65 return;
66 }
67 std::optional<unsigned> videoBitRate;
68 decoder >> videoBitRate;
69 if (!videoBitRate) {
70 ASSERT_NOT_REACHED();
71 cancelReply(WTFMove(completionHandler));
72 return;
73 }
74 completionHandler(WTFMove(*creationError), WTFMove(*mimeType), WTFMove(*audioBitRate), WTFMove(*videoBitRate));
75}
76
77void CreateRecorder::cancelReply(CompletionHandler<void(std::optional<WebCore::ExceptionData>&&, String&&, unsigned&&, unsigned&&)>&& completionHandler)
78{
79 completionHandler(IPC::AsyncReplyError<std::optional<WebCore::ExceptionData>>::create(), IPC::AsyncReplyError<String>::create(), IPC::AsyncReplyError<unsigned>::create(), IPC::AsyncReplyError<unsigned>::create());
80}
81
82void CreateRecorder::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<WebCore::ExceptionData>& creationError, const String& mimeType, const unsigned& audioBitRate, const unsigned& videoBitRate)
83{
84 encoder.get() << creationError;
85 encoder.get() << mimeType;
86 encoder.get() << audioBitRate;
87 encoder.get() << videoBitRate;
88 connection.sendSyncReply(WTFMove(encoder));
89}

97void RemoteMediaRecorderManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
98{
99 if (decoder.messageName() == Messages::RemoteMediaRecorderManager::CreateRecorder::name())
100 return IPC::handleMessageAsync<Messages::RemoteMediaRecorderManager::CreateRecorder>(connection, decoder, this, &RemoteMediaRecorderManager::createRecorder);
101 if (decoder.messageName() == Messages::RemoteMediaRecorderManager::ReleaseRecorder::name())
102 return IPC::handleMessage<Messages::RemoteMediaRecorderManager::ReleaseRecorder>(decoder, this, &RemoteMediaRecorderManager::releaseRecorder);
103 UNUSED_PARAM(connection);
104 UNUSED_PARAM(decoder);
105#if ENABLE(IPC_TESTING_API)
106 if (connection.ignoreInvalidMessageForTesting())
107 return;
108#endif // ENABLE(IPC_TESTING_API)
109 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
110}
