43void FetchData::callReply(IPC::Decoder& decoder, CompletionHandler<void(IPC::DataReference&&, double&&)>&& completionHandler)
44{
45 std::optional<IPC::DataReference> buffer;
46 decoder >> buffer;
47 if (!buffer) {
48 ASSERT_NOT_REACHED();
49 cancelReply(WTFMove(completionHandler));
50 return;
51 }
52 std::optional<double> timeCode;
53 decoder >> timeCode;
54 if (!timeCode) {
55 ASSERT_NOT_REACHED();
56 cancelReply(WTFMove(completionHandler));
57 return;
58 }
59 completionHandler(WTFMove(*buffer), WTFMove(*timeCode));
60}
61
62void FetchData::cancelReply(CompletionHandler<void(IPC::DataReference&&, double&&)>&& completionHandler)
63{
64 completionHandler(IPC::AsyncReplyError<IPC::DataReference>::create(), IPC::AsyncReplyError<double>::create());
65}
66
67void FetchData::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const IPC::DataReference& buffer, double timeCode)
68{
69 encoder.get() << buffer;
70 encoder.get() << timeCode;
71 connection.sendSyncReply(WTFMove(encoder));
72}

125void RemoteMediaRecorder::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
126{
127 if (decoder.messageName() == Messages::RemoteMediaRecorder::AudioSamplesStorageChanged::name())
128 return IPC::handleMessage<Messages::RemoteMediaRecorder::AudioSamplesStorageChanged>(decoder, this, &RemoteMediaRecorder::audioSamplesStorageChanged);
129 if (decoder.messageName() == Messages::RemoteMediaRecorder::AudioSamplesAvailable::name())
130 return IPC::handleMessage<Messages::RemoteMediaRecorder::AudioSamplesAvailable>(decoder, this, &RemoteMediaRecorder::audioSamplesAvailable);
131 if (decoder.messageName() == Messages::RemoteMediaRecorder::VideoSampleAvailable::name())
132 return IPC::handleMessage<Messages::RemoteMediaRecorder::VideoSampleAvailable>(decoder, this, &RemoteMediaRecorder::videoSampleAvailable);
133 if (decoder.messageName() == Messages::RemoteMediaRecorder::FetchData::name())
134 return IPC::handleMessageAsync<Messages::RemoteMediaRecorder::FetchData>(connection, decoder, this, &RemoteMediaRecorder::fetchData);
135 if (decoder.messageName() == Messages::RemoteMediaRecorder::StopRecording::name())
136 return IPC::handleMessageAsync<Messages::RemoteMediaRecorder::StopRecording>(connection, decoder, this, &RemoteMediaRecorder::stopRecording);
137 if (decoder.messageName() == Messages::RemoteMediaRecorder::Pause::name())
138 return IPC::handleMessageAsync<Messages::RemoteMediaRecorder::Pause>(connection, decoder, this, &RemoteMediaRecorder::pause);
139 if (decoder.messageName() == Messages::RemoteMediaRecorder::Resume::name())
140 return IPC::handleMessageAsync<Messages::RemoteMediaRecorder::Resume>(connection, decoder, this, &RemoteMediaRecorder::resume);
141 UNUSED_PARAM(connection);
142 UNUSED_PARAM(decoder);
143#if ENABLE(IPC_TESTING_API)
144 if (connection.ignoreInvalidMessageForTesting())
145 return;
146#endif // ENABLE(IPC_TESTING_API)
147 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
148}
