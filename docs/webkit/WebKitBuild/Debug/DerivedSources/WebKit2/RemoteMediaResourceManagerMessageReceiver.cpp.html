45void ResponseReceived::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::ShouldContinuePolicyCheck&&)>&& completionHandler)
46{
47 std::optional<WebCore::ShouldContinuePolicyCheck> shouldContinue;
48 decoder >> shouldContinue;
49 if (!shouldContinue) {
50 ASSERT_NOT_REACHED();
51 cancelReply(WTFMove(completionHandler));
52 return;
53 }
54 completionHandler(WTFMove(*shouldContinue));
55}
56
57void ResponseReceived::cancelReply(CompletionHandler<void(WebCore::ShouldContinuePolicyCheck&&)>&& completionHandler)
58{
59 completionHandler(IPC::AsyncReplyError<WebCore::ShouldContinuePolicyCheck>::create());
60}
61
62void ResponseReceived::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, WebCore::ShouldContinuePolicyCheck shouldContinue)
63{
64 encoder.get() << shouldContinue;
65 connection.sendSyncReply(WTFMove(encoder));
66}
67
68void RedirectReceived::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler)
69{
70 std::optional<WebCore::ResourceRequest> returnRequest;
71 decoder >> returnRequest;
72 if (!returnRequest) {
73 ASSERT_NOT_REACHED();
74 cancelReply(WTFMove(completionHandler));
75 return;
76 }
77 completionHandler(WTFMove(*returnRequest));
78}
79
80void RedirectReceived::cancelReply(CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler)
81{
82 completionHandler(IPC::AsyncReplyError<WebCore::ResourceRequest>::create());
83}
84
85void RedirectReceived::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebCore::ResourceRequest& returnRequest)
86{
87 encoder.get() << returnRequest;
88 connection.sendSyncReply(WTFMove(encoder));
89}

97void RemoteMediaResourceManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
98{
99 if (decoder.messageName() == Messages::RemoteMediaResourceManager::ResponseReceived::name())
100 return IPC::handleMessageAsync<Messages::RemoteMediaResourceManager::ResponseReceived>(connection, decoder, this, &RemoteMediaResourceManager::responseReceived);
101 if (decoder.messageName() == Messages::RemoteMediaResourceManager::RedirectReceived::name())
102 return IPC::handleMessageAsync<Messages::RemoteMediaResourceManager::RedirectReceived>(connection, decoder, this, &RemoteMediaResourceManager::redirectReceived);
103 if (decoder.messageName() == Messages::RemoteMediaResourceManager::DataSent::name())
104 return IPC::handleMessage<Messages::RemoteMediaResourceManager::DataSent>(decoder, this, &RemoteMediaResourceManager::dataSent);
105 if (decoder.messageName() == Messages::RemoteMediaResourceManager::DataReceived::name())
106 return IPC::handleMessage<Messages::RemoteMediaResourceManager::DataReceived>(decoder, this, &RemoteMediaResourceManager::dataReceived);
107 if (decoder.messageName() == Messages::RemoteMediaResourceManager::AccessControlCheckFailed::name())
108 return IPC::handleMessage<Messages::RemoteMediaResourceManager::AccessControlCheckFailed>(decoder, this, &RemoteMediaResourceManager::accessControlCheckFailed);
109 if (decoder.messageName() == Messages::RemoteMediaResourceManager::LoadFailed::name())
110 return IPC::handleMessage<Messages::RemoteMediaResourceManager::LoadFailed>(decoder, this, &RemoteMediaResourceManager::loadFailed);
111 if (decoder.messageName() == Messages::RemoteMediaResourceManager::LoadFinished::name())
112 return IPC::handleMessage<Messages::RemoteMediaResourceManager::LoadFinished>(decoder, this, &RemoteMediaResourceManager::loadFinished);
113 UNUSED_PARAM(connection);
114 UNUSED_PARAM(decoder);
115#if ENABLE(IPC_TESTING_API)
116 if (connection.ignoreInvalidMessageForTesting())
117 return;
118#endif // ENABLE(IPC_TESTING_API)
119 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
120}
