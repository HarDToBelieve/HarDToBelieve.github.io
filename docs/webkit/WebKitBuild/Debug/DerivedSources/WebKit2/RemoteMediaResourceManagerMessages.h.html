54class ResponseReceived {
55public:
56 using Arguments = std::tuple<const WebKit::RemoteMediaResourceIdentifier&, const WebCore::ResourceResponse&, bool>;
57
58 static IPC::MessageName name() { return IPC::MessageName::RemoteMediaResourceManager_ResponseReceived; }
59 static constexpr bool isSync = false;
60
61 static void callReply(IPC::Decoder&, CompletionHandler<void(WebCore::ShouldContinuePolicyCheck&&)>&&);
62 static void cancelReply(CompletionHandler<void(WebCore::ShouldContinuePolicyCheck&&)>&&);
63 static IPC::MessageName asyncMessageReplyName() { return IPC::MessageName::RemoteMediaResourceManager_ResponseReceivedReply; }
64 using AsyncReply = ResponseReceivedAsyncReply;
65 static constexpr auto callbackThread = WTF::CompletionHandlerCallThread::ConstructionThread;
66 static void send(UniqueRef<IPC::Encoder>&&, IPC::Connection&, WebCore::ShouldContinuePolicyCheck shouldContinue);
67 using Reply = std::tuple<WebCore::ShouldContinuePolicyCheck&>;
68 using ReplyArguments = std::tuple<WebCore::ShouldContinuePolicyCheck>;
69 ResponseReceived(const WebKit::RemoteMediaResourceIdentifier& identifier, const WebCore::ResourceResponse& response, bool didPassAccessControlCheck)
70 : m_arguments(identifier, response, didPassAccessControlCheck)
71 {
72 }
73
74 const Arguments& arguments() const
75 {
76 return m_arguments;
77 }
78
79private:
80 Arguments m_arguments;
81};
