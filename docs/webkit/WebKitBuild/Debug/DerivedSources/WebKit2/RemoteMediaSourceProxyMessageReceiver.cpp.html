46void AddSourceBuffer::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, WebCore::MediaSourcePrivate::AddStatus status, const std::optional<WebKit::RemoteSourceBufferIdentifier>& remoteSourceBufferIdentifier)
47{
48 encoder.get() << status;
49 encoder.get() << remoteSourceBufferIdentifier;
50 connection.sendSyncReply(WTFMove(encoder));
51}

59void RemoteMediaSourceProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
60{
61 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::DurationChanged::name())
62 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::DurationChanged>(decoder, this, &RemoteMediaSourceProxy::durationChanged);
63 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::BufferedChanged::name())
64 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::BufferedChanged>(decoder, this, &RemoteMediaSourceProxy::bufferedChanged);
65 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::SetReadyState::name())
66 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::SetReadyState>(decoder, this, &RemoteMediaSourceProxy::setReadyState);
67 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::SetIsSeeking::name())
68 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::SetIsSeeking>(decoder, this, &RemoteMediaSourceProxy::setIsSeeking);
69 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::WaitForSeekCompleted::name())
70 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::WaitForSeekCompleted>(decoder, this, &RemoteMediaSourceProxy::waitForSeekCompleted);
71 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::SeekCompleted::name())
72 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::SeekCompleted>(decoder, this, &RemoteMediaSourceProxy::seekCompleted);
73 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::SetTimeFudgeFactor::name())
74 return IPC::handleMessage<Messages::RemoteMediaSourceProxy::SetTimeFudgeFactor>(decoder, this, &RemoteMediaSourceProxy::setTimeFudgeFactor);
75 UNUSED_PARAM(connection);
76 UNUSED_PARAM(decoder);
77#if ENABLE(IPC_TESTING_API)
78 if (connection.ignoreInvalidMessageForTesting())
79 return;
80#endif // ENABLE(IPC_TESTING_API)
81 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
82}
83
84bool RemoteMediaSourceProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
85{
86 if (decoder.messageName() == Messages::RemoteMediaSourceProxy::AddSourceBuffer::name())
87 return IPC::handleMessageSynchronous<Messages::RemoteMediaSourceProxy::AddSourceBuffer>(connection, decoder, replyEncoder, this, &RemoteMediaSourceProxy::addSourceBuffer);
88 UNUSED_PARAM(connection);
89 UNUSED_PARAM(decoder);
90 UNUSED_PARAM(replyEncoder);
91#if ENABLE(IPC_TESTING_API)
92 if (connection.ignoreInvalidMessageForTesting())
93 return false;
94#endif // ENABLE(IPC_TESTING_API)
95 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
96 return false;
97}
