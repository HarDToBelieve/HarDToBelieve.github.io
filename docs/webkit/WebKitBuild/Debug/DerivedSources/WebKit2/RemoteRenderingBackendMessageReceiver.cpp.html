96void RemoteRenderingBackend::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
97{
98 auto protectedThis = makeRef(*this);
99 if (decoder.messageName() == Messages::RemoteRenderingBackend::CreateImageBuffer::name())
100 return IPC::handleMessage<Messages::RemoteRenderingBackend::CreateImageBuffer>(decoder, this, &RemoteRenderingBackend::createImageBuffer);
101 if (decoder.messageName() == Messages::RemoteRenderingBackend::WakeUpAndApplyDisplayList::name())
102 return IPC::handleMessage<Messages::RemoteRenderingBackend::WakeUpAndApplyDisplayList>(decoder, this, &RemoteRenderingBackend::wakeUpAndApplyDisplayList);
103 if (decoder.messageName() == Messages::RemoteRenderingBackend::DestroyGetPixelBufferSharedMemory::name())
104 return IPC::handleMessage<Messages::RemoteRenderingBackend::DestroyGetPixelBufferSharedMemory>(decoder, this, &RemoteRenderingBackend::destroyGetPixelBufferSharedMemory);
105 if (decoder.messageName() == Messages::RemoteRenderingBackend::CacheNativeImage::name())
106 return IPC::handleMessage<Messages::RemoteRenderingBackend::CacheNativeImage>(decoder, this, &RemoteRenderingBackend::cacheNativeImage);
107 if (decoder.messageName() == Messages::RemoteRenderingBackend::CacheFont::name())
108 return IPC::handleMessage<Messages::RemoteRenderingBackend::CacheFont>(decoder, this, &RemoteRenderingBackend::cacheFont);
109 if (decoder.messageName() == Messages::RemoteRenderingBackend::DeleteAllFonts::name())
110 return IPC::handleMessage<Messages::RemoteRenderingBackend::DeleteAllFonts>(decoder, this, &RemoteRenderingBackend::deleteAllFonts);
111 if (decoder.messageName() == Messages::RemoteRenderingBackend::DidCreateSharedDisplayListHandle::name())
112 return IPC::handleMessage<Messages::RemoteRenderingBackend::DidCreateSharedDisplayListHandle>(decoder, this, &RemoteRenderingBackend::didCreateSharedDisplayListHandle);
113 if (decoder.messageName() == Messages::RemoteRenderingBackend::ReleaseRemoteResource::name())
114 return IPC::handleMessage<Messages::RemoteRenderingBackend::ReleaseRemoteResource>(decoder, this, &RemoteRenderingBackend::releaseRemoteResource);
115 UNUSED_PARAM(connection);
116 UNUSED_PARAM(decoder);
117#if ENABLE(IPC_TESTING_API)
118 if (connection.ignoreInvalidMessageForTesting())
119 return;
120#endif // ENABLE(IPC_TESTING_API)
121 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
122}

124bool RemoteRenderingBackend::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
125{
126 auto protectedThis = makeRef(*this);
127 if (decoder.messageName() == Messages::RemoteRenderingBackend::UpdateSharedMemoryForGetPixelBuffer::name())
128 return IPC::handleMessageSynchronous<Messages::RemoteRenderingBackend::UpdateSharedMemoryForGetPixelBuffer>(connection, decoder, replyEncoder, this, &RemoteRenderingBackend::updateSharedMemoryForGetPixelBuffer);
129 if (decoder.messageName() == Messages::RemoteRenderingBackend::SemaphoreForGetPixelBuffer::name())
130 return IPC::handleMessageSynchronous<Messages::RemoteRenderingBackend::SemaphoreForGetPixelBuffer>(connection, decoder, replyEncoder, this, &RemoteRenderingBackend::semaphoreForGetPixelBuffer);
131 if (decoder.messageName() == Messages::RemoteRenderingBackend::UpdateSharedMemoryAndSemaphoreForGetPixelBuffer::name())
132 return IPC::handleMessageSynchronous<Messages::RemoteRenderingBackend::UpdateSharedMemoryAndSemaphoreForGetPixelBuffer>(connection, decoder, replyEncoder, this, &RemoteRenderingBackend::updateSharedMemoryAndSemaphoreForGetPixelBuffer);
133 if (decoder.messageName() == Messages::RemoteRenderingBackend::GetDataURLForImageBuffer::name())
134 return IPC::handleMessageSynchronous<Messages::RemoteRenderingBackend::GetDataURLForImageBuffer>(connection, decoder, replyEncoder, this, &RemoteRenderingBackend::getDataURLForImageBuffer);
135 if (decoder.messageName() == Messages::RemoteRenderingBackend::GetDataForImageBuffer::name())
136 return IPC::handleMessageSynchronous<Messages::RemoteRenderingBackend::GetDataForImageBuffer>(connection, decoder, replyEncoder, this, &RemoteRenderingBackend::getDataForImageBuffer);
137 if (decoder.messageName() == Messages::RemoteRenderingBackend::GetShareableBitmapForImageBuffer::name())
138 return IPC::handleMessageSynchronous<Messages::RemoteRenderingBackend::GetShareableBitmapForImageBuffer>(connection, decoder, replyEncoder, this, &RemoteRenderingBackend::getShareableBitmapForImageBuffer);
139 UNUSED_PARAM(connection);
140 UNUSED_PARAM(decoder);
141 UNUSED_PARAM(replyEncoder);
142#if ENABLE(IPC_TESTING_API)
143 if (connection.ignoreInvalidMessageForTesting())
144 return false;
145#endif // ENABLE(IPC_TESTING_API)
146 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
147 return false;
148}
