43void CreateLayer::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<WebKit::LayerHostingContextID>&&)>&& completionHandler)
44{
45 std::optional<std::optional<WebKit::LayerHostingContextID>> contextID;
46 decoder >> contextID;
47 if (!contextID) {
48 ASSERT_NOT_REACHED();
49 cancelReply(WTFMove(completionHandler));
50 return;
51 }
52 completionHandler(WTFMove(*contextID));
53}
54
55void CreateLayer::cancelReply(CompletionHandler<void(std::optional<WebKit::LayerHostingContextID>&&)>&& completionHandler)
56{
57 completionHandler(IPC::AsyncReplyError<std::optional<WebKit::LayerHostingContextID>>::create());
58}
59
60void CreateLayer::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<WebKit::LayerHostingContextID>& contextID)
61{
62 encoder.get() << contextID;
63 connection.sendSyncReply(WTFMove(encoder));
64}

72void RemoteSampleBufferDisplayLayerManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
73{
74 auto protectedThis = makeRef(*this);
75 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayerManager::CreateLayer::name())
76 return IPC::handleMessageAsync<Messages::RemoteSampleBufferDisplayLayerManager::CreateLayer>(connection, decoder, this, &RemoteSampleBufferDisplayLayerManager::createLayer);
77 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayerManager::ReleaseLayer::name())
78 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayerManager::ReleaseLayer>(decoder, this, &RemoteSampleBufferDisplayLayerManager::releaseLayer);
79 if (dispatchMessage(connection, decoder))
80 return;
81 UNUSED_PARAM(connection);
82 UNUSED_PARAM(decoder);
83#if ENABLE(IPC_TESTING_API)
84 if (connection.ignoreInvalidMessageForTesting())
85 return;
86#endif // ENABLE(IPC_TESTING_API)
87 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
88}
