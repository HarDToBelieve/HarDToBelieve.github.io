61void RemoveCodedFrames::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::PlatformTimeRanges&&, uint64_t&&)>&& completionHandler)
62{
63 std::optional<WebCore::PlatformTimeRanges> buffered;
64 decoder >> buffered;
65 if (!buffered) {
66 ASSERT_NOT_REACHED();
67 cancelReply(WTFMove(completionHandler));
68 return;
69 }
70 std::optional<uint64_t> totalTrackBufferSizeInBytes;
71 decoder >> totalTrackBufferSizeInBytes;
72 if (!totalTrackBufferSizeInBytes) {
73 ASSERT_NOT_REACHED();
74 cancelReply(WTFMove(completionHandler));
75 return;
76 }
77 completionHandler(WTFMove(*buffered), WTFMove(*totalTrackBufferSizeInBytes));
78}
79
80void RemoveCodedFrames::cancelReply(CompletionHandler<void(WebCore::PlatformTimeRanges&&, uint64_t&&)>&& completionHandler)
81{
82 completionHandler(IPC::AsyncReplyError<WebCore::PlatformTimeRanges>::create(), IPC::AsyncReplyError<uint64_t>::create());
83}

150void RemoteSourceBufferProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
151{
152 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::SetActive::name())
153 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::SetActive>(decoder, this, &RemoteSourceBufferProxy::setActive);
154 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::SetMode::name())
155 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::SetMode>(decoder, this, &RemoteSourceBufferProxy::setMode);
156 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::Append::name())
157 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::Append>(decoder, this, &RemoteSourceBufferProxy::append);
158 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::Abort::name())
159 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::Abort>(decoder, this, &RemoteSourceBufferProxy::abort);
160 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::ResetParserState::name())
161 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::ResetParserState>(decoder, this, &RemoteSourceBufferProxy::resetParserState);
162 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::RemovedFromMediaSource::name())
163 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::RemovedFromMediaSource>(decoder, this, &RemoteSourceBufferProxy::removedFromMediaSource);
164 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::SetMediaSourceEnded::name())
165 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::SetMediaSourceEnded>(decoder, this, &RemoteSourceBufferProxy::setMediaSourceEnded);
166 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::SetReadyState::name())
167 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::SetReadyState>(decoder, this, &RemoteSourceBufferProxy::setReadyState);
168 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::StartChangingType::name())
169 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::StartChangingType>(decoder, this, &RemoteSourceBufferProxy::startChangingType);
170 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::AddTrackBuffer::name())
171 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::AddTrackBuffer>(decoder, this, &RemoteSourceBufferProxy::addTrackBuffer);
172 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::ResetTrackBuffers::name())
173 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::ResetTrackBuffers>(decoder, this, &RemoteSourceBufferProxy::resetTrackBuffers);
174 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::ClearTrackBuffers::name())
175 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::ClearTrackBuffers>(decoder, this, &RemoteSourceBufferProxy::clearTrackBuffers);
176 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::SetAllTrackBuffersNeedRandomAccess::name())
177 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::SetAllTrackBuffersNeedRandomAccess>(decoder, this, &RemoteSourceBufferProxy::setAllTrackBuffersNeedRandomAccess);
178 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::RemoveCodedFrames::name())
179 return IPC::handleMessageAsync<Messages::RemoteSourceBufferProxy::RemoveCodedFrames>(connection, decoder, this, &RemoteSourceBufferProxy::removeCodedFrames);
180 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::ReenqueueMediaIfNeeded::name())
181 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::ReenqueueMediaIfNeeded>(decoder, this, &RemoteSourceBufferProxy::reenqueueMediaIfNeeded);

186 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::SetShouldGenerateTimestamps::name())

189 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::ResetTimestampOffsetInTrackBuffers>(decoder, this, &RemoteSourceBufferProxy::resetTimestampOffsetInTrackBuffers);

191 return IPC::handleMessage<Messages::RemoteSourceBufferProxy::SetTimestampOffset>(decoder, this, &RemoteSourceBufferProxy::setTimestampOffset);

213bool RemoteSourceBufferProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
214{
215 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::CanSwitchToType::name())
216 return IPC::handleMessageSynchronous<Messages::RemoteSourceBufferProxy::CanSwitchToType>(connection, decoder, replyEncoder, this, &RemoteSourceBufferProxy::canSwitchToType);
217 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::UpdateBufferedFromTrackBuffers::name())
218 return IPC::handleMessageSynchronous<Messages::RemoteSourceBufferProxy::UpdateBufferedFromTrackBuffers>(connection, decoder, replyEncoder, this, &RemoteSourceBufferProxy::updateBufferedFromTrackBuffers);
219 if (decoder.messageName() == Messages::RemoteSourceBufferProxy::EvictCodedFrames::name())
220 return IPC::handleMessageSynchronous<Messages::RemoteSourceBufferProxy::EvictCodedFrames>(connection, decoder, replyEncoder, this, &RemoteSourceBufferProxy::evictCodedFrames);
221 UNUSED_PARAM(connection);
222 UNUSED_PARAM(decoder);
223 UNUSED_PARAM(replyEncoder);
224#if ENABLE(IPC_TESTING_API)
225 if (connection.ignoreInvalidMessageForTesting())
226 return false;
227#endif // ENABLE(IPC_TESTING_API)
228 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
229 return false;
230}
