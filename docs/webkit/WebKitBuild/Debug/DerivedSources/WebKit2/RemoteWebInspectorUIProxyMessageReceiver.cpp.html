40void RemoteWebInspectorUIProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
41{
42 auto protectedThis = makeRef(*this);
43 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::FrontendLoaded::name())
44 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::FrontendLoaded>(decoder, this, &RemoteWebInspectorUIProxy::frontendLoaded);
45 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::FrontendDidClose::name())
46 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::FrontendDidClose>(decoder, this, &RemoteWebInspectorUIProxy::frontendDidClose);
47 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::Reopen::name())
48 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::Reopen>(decoder, this, &RemoteWebInspectorUIProxy::reopen);
49 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::ResetState::name())
50 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::ResetState>(decoder, this, &RemoteWebInspectorUIProxy::resetState);
51 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::BringToFront::name())
52 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::BringToFront>(decoder, this, &RemoteWebInspectorUIProxy::bringToFront);
53 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::Save::name())
54 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::Save>(decoder, this, &RemoteWebInspectorUIProxy::save);
55 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::Append::name())
56 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::Append>(decoder, this, &RemoteWebInspectorUIProxy::append);
57 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::SetForcedAppearance::name())
58 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::SetForcedAppearance>(decoder, this, &RemoteWebInspectorUIProxy::setForcedAppearance);
59 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::SetSheetRect::name())
60 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::SetSheetRect>(decoder, this, &RemoteWebInspectorUIProxy::setSheetRect);
61 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::StartWindowDrag::name())
62 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::StartWindowDrag>(decoder, this, &RemoteWebInspectorUIProxy::startWindowDrag);
63 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::OpenURLExternally::name())
64 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::OpenURLExternally>(decoder, this, &RemoteWebInspectorUIProxy::openURLExternally);
65 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::ShowCertificate::name())
66 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::ShowCertificate>(decoder, this, &RemoteWebInspectorUIProxy::showCertificate);
67 if (decoder.messageName() == Messages::RemoteWebInspectorUIProxy::SendMessageToBackend::name())
68 return IPC::handleMessage<Messages::RemoteWebInspectorUIProxy::SendMessageToBackend>(decoder, this, &RemoteWebInspectorUIProxy::sendMessageToBackend);
69 UNUSED_PARAM(connection);
70 UNUSED_PARAM(decoder);
71#if ENABLE(IPC_TESTING_API)
72 if (connection.ignoreInvalidMessageForTesting())
73 return;
74#endif // ENABLE(IPC_TESTING_API)
75 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
76}
