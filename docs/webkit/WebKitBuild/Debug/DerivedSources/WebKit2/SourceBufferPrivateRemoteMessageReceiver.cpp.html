63void SourceBufferPrivateRemote::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
64{
65 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidReceiveInitializationSegment::name())
66 return IPC::handleMessageAsync<Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidReceiveInitializationSegment>(connection, decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateDidReceiveInitializationSegment);
67 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateStreamEndedWithDecodeError::name())
68 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateStreamEndedWithDecodeError>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateStreamEndedWithDecodeError);
69 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateAppendError::name())
70 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateAppendError>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateAppendError);
71 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateAppendComplete::name())
72 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateAppendComplete>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateAppendComplete);
73 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateHighestPresentationTimestampChanged::name())
74 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateHighestPresentationTimestampChanged>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateHighestPresentationTimestampChanged);
75 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateDurationChanged::name())
76 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateDurationChanged>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateDurationChanged);
77 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidParseSample::name())
78 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidParseSample>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateDidParseSample);
79 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidDropSample::name())
80 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidDropSample>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateDidDropSample);
81 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateBufferedDirtyChanged::name())
82 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateBufferedDirtyChanged>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateBufferedDirtyChanged);
83 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidReceiveRenderingError::name())
84 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidReceiveRenderingError>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateDidReceiveRenderingError);
85 if (decoder.messageName() == Messages::SourceBufferPrivateRemote::SourceBufferPrivateReportExtraMemoryCost::name())
86 return IPC::handleMessage<Messages::SourceBufferPrivateRemote::SourceBufferPrivateReportExtraMemoryCost>(decoder, this, &SourceBufferPrivateRemote::sourceBufferPrivateReportExtraMemoryCost);
87 UNUSED_PARAM(connection);
88 UNUSED_PARAM(decoder);
89#if ENABLE(IPC_TESTING_API)
90 if (connection.ignoreInvalidMessageForTesting())
91 return;
92#endif // ENABLE(IPC_TESTING_API)
93 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
94}
