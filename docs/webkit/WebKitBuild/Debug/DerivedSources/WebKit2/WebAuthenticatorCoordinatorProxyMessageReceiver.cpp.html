47void MakeCredential::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::AuthenticatorResponseData&&, WebCore::AuthenticatorAttachment&&, WebCore::ExceptionData&&)>&& completionHandler)
48{
49 std::optional<WebCore::AuthenticatorResponseData> data;
50 decoder >> data;
51 if (!data) {
52 ASSERT_NOT_REACHED();
53 cancelReply(WTFMove(completionHandler));
54 return;
55 }
56 std::optional<WebCore::AuthenticatorAttachment> attachment;
57 decoder >> attachment;
58 if (!attachment) {
59 ASSERT_NOT_REACHED();
60 cancelReply(WTFMove(completionHandler));
61 return;
62 }
63 std::optional<WebCore::ExceptionData> exception;
64 decoder >> exception;
65 if (!exception) {
66 ASSERT_NOT_REACHED();
67 cancelReply(WTFMove(completionHandler));
68 return;
69 }
70 completionHandler(WTFMove(*data), WTFMove(*attachment), WTFMove(*exception));
71}
72
73void MakeCredential::cancelReply(CompletionHandler<void(WebCore::AuthenticatorResponseData&&, WebCore::AuthenticatorAttachment&&, WebCore::ExceptionData&&)>&& completionHandler)
74{
75 completionHandler(IPC::AsyncReplyError<WebCore::AuthenticatorResponseData>::create(), IPC::AsyncReplyError<WebCore::AuthenticatorAttachment>::create(), IPC::AsyncReplyError<WebCore::ExceptionData>::create());
76}
77
78void MakeCredential::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebCore::AuthenticatorResponseData& data, WebCore::AuthenticatorAttachment attachment, const WebCore::ExceptionData& exception)
79{
80 encoder.get() << data;
81 encoder.get() << attachment;
82 encoder.get() << exception;
83 connection.sendSyncReply(WTFMove(encoder));
84}

125void IsUserVerifyingPlatformAuthenticatorAvailable::callReply(IPC::Decoder& decoder, CompletionHandler<void(bool&&)>&& completionHandler)
126{
127 std::optional<bool> result;
128 decoder >> result;
129 if (!result) {
130 ASSERT_NOT_REACHED();

132 return;
133 }
134 completionHandler(WTFMove(*result));
135}

154void WebAuthenticatorCoordinatorProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
155{
156 if (decoder.messageName() == Messages::WebAuthenticatorCoordinatorProxy::MakeCredential::name())
157 return IPC::handleMessageAsync<Messages::WebAuthenticatorCoordinatorProxy::MakeCredential>(connection, decoder, this, &WebAuthenticatorCoordinatorProxy::makeCredential);
158 if (decoder.messageName() == Messages::WebAuthenticatorCoordinatorProxy::GetAssertion::name())
159 return IPC::handleMessageAsync<Messages::WebAuthenticatorCoordinatorProxy::GetAssertion>(connection, decoder, this, &WebAuthenticatorCoordinatorProxy::getAssertion);
160 if (decoder.messageName() == Messages::WebAuthenticatorCoordinatorProxy::IsUserVerifyingPlatformAuthenticatorAvailable::name())
161 return IPC::handleMessageAsync<Messages::WebAuthenticatorCoordinatorProxy::IsUserVerifyingPlatformAuthenticatorAvailable>(connection, decoder, this, &WebAuthenticatorCoordinatorProxy::isUserVerifyingPlatformAuthenticatorAvailable);
162 UNUSED_PARAM(connection);
163 UNUSED_PARAM(decoder);
164#if ENABLE(IPC_TESTING_API)
165 if (connection.ignoreInvalidMessageForTesting())
166 return;
167#endif // ENABLE(IPC_TESTING_API)
168 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
169}
