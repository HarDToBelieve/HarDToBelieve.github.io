43void CreateWebAuthnConnectionToWebProcess::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<IPC::Attachment>&&)>&& completionHandler)
44{
45 std::optional<std::optional<IPC::Attachment>> connectionIdentifier;
46 decoder >> connectionIdentifier;
47 if (!connectionIdentifier) {
48 ASSERT_NOT_REACHED();
49 cancelReply(WTFMove(completionHandler));
50 return;
51 }
52 completionHandler(WTFMove(*connectionIdentifier));
53}
54
55void CreateWebAuthnConnectionToWebProcess::cancelReply(CompletionHandler<void(std::optional<IPC::Attachment>&&)>&& completionHandler)
56{
57 completionHandler(IPC::AsyncReplyError<std::optional<IPC::Attachment>>::create());
58}
59
60void CreateWebAuthnConnectionToWebProcess::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<IPC::Attachment>& connectionIdentifier)
61{
62 encoder.get() << connectionIdentifier;
63 connection.sendSyncReply(WTFMove(encoder));
64}

72void WebAuthnProcess::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
73{
74 auto protectedThis = makeRef(*this);
75 if (decoder.messageName() == Messages::WebAuthnProcess::InitializeWebAuthnProcess::name())
76 return IPC::handleMessage<Messages::WebAuthnProcess::InitializeWebAuthnProcess>(decoder, this, &WebAuthnProcess::initializeWebAuthnProcess);
77 if (decoder.messageName() == Messages::WebAuthnProcess::CreateWebAuthnConnectionToWebProcess::name())
78 return IPC::handleMessageAsync<Messages::WebAuthnProcess::CreateWebAuthnConnectionToWebProcess>(connection, decoder, this, &WebAuthnProcess::createWebAuthnConnectionToWebProcess);
79 if (decoder.messageName() == Messages::WebAuthnProcess::ProcessDidTransitionToForeground::name())
80 return IPC::handleMessage<Messages::WebAuthnProcess::ProcessDidTransitionToForeground>(decoder, this, &WebAuthnProcess::processDidTransitionToForeground);
81 if (decoder.messageName() == Messages::WebAuthnProcess::ProcessDidTransitionToBackground::name())
82 return IPC::handleMessage<Messages::WebAuthnProcess::ProcessDidTransitionToBackground>(decoder, this, &WebAuthnProcess::processDidTransitionToBackground);
83 UNUSED_PARAM(connection);
84 UNUSED_PARAM(decoder);
85#if ENABLE(IPC_TESTING_API)
86 if (connection.ignoreInvalidMessageForTesting())
87 return;
88#endif // ENABLE(IPC_TESTING_API)
89 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
90}
