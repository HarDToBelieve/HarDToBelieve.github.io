47class HandleMessage {

49 using Arguments = std::tuple<const String&, const WebKit::UserData&>;

51 static IPC::MessageName name() { return IPC::MessageName::WebConnection_HandleMessage; }

53
54 HandleMessage(const String& messageName, const WebKit::UserData& messageBody)
55 : m_arguments(messageName, messageBody)

57 }
58

62 }

64private:
65 Arguments m_arguments;
66};
