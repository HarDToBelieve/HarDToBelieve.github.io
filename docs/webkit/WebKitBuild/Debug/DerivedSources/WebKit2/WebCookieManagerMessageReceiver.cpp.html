48void GetHostnamesWithCookies::callReply(IPC::Decoder& decoder, CompletionHandler<void(Vector<String>&&)>&& completionHandler)
49{
50 std::optional<Vector<String>> hostnames;
51 decoder >> hostnames;
52 if (!hostnames) {
53 ASSERT_NOT_REACHED();
54 cancelReply(WTFMove(completionHandler));
55 return;
56 }
57 completionHandler(WTFMove(*hostnames));
58}
59
60void GetHostnamesWithCookies::cancelReply(CompletionHandler<void(Vector<String>&&)>&& completionHandler)
61{
62 completionHandler(IPC::AsyncReplyError<Vector<String>>::create());
63}
64
65void GetHostnamesWithCookies::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const Vector<String>& hostnames)
66{
67 encoder.get() << hostnames;
68 connection.sendSyncReply(WTFMove(encoder));
69}

221void WebCookieManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
222{
223 if (decoder.messageName() == Messages::WebCookieManager::GetHostnamesWithCookies::name())
224 return IPC::handleMessageAsync<Messages::WebCookieManager::GetHostnamesWithCookies>(connection, decoder, this, &WebCookieManager::getHostnamesWithCookies);
225 if (decoder.messageName() == Messages::WebCookieManager::DeleteCookiesForHostnames::name())
226 return IPC::handleMessage<Messages::WebCookieManager::DeleteCookiesForHostnames>(decoder, this, &WebCookieManager::deleteCookiesForHostnames);
227 if (decoder.messageName() == Messages::WebCookieManager::DeleteAllCookies::name())
228 return IPC::handleMessage<Messages::WebCookieManager::DeleteAllCookies>(decoder, this, &WebCookieManager::deleteAllCookies);
229 if (decoder.messageName() == Messages::WebCookieManager::SetCookie::name())
230 return IPC::handleMessageAsync<Messages::WebCookieManager::SetCookie>(connection, decoder, this, &WebCookieManager::setCookie);
231 if (decoder.messageName() == Messages::WebCookieManager::SetCookies::name())
232 return IPC::handleMessageAsync<Messages::WebCookieManager::SetCookies>(connection, decoder, this, &WebCookieManager::setCookies);
233 if (decoder.messageName() == Messages::WebCookieManager::GetAllCookies::name())
234 return IPC::handleMessageAsync<Messages::WebCookieManager::GetAllCookies>(connection, decoder, this, &WebCookieManager::getAllCookies);
235 if (decoder.messageName() == Messages::WebCookieManager::GetCookies::name())
236 return IPC::handleMessageAsync<Messages::WebCookieManager::GetCookies>(connection, decoder, this, &WebCookieManager::getCookies);
237 if (decoder.messageName() == Messages::WebCookieManager::DeleteCookie::name())
238 return IPC::handleMessageAsync<Messages::WebCookieManager::DeleteCookie>(connection, decoder, this, &WebCookieManager::deleteCookie);
239 if (decoder.messageName() == Messages::WebCookieManager::DeleteAllCookiesModifiedSince::name())
240 return IPC::handleMessageAsync<Messages::WebCookieManager::DeleteAllCookiesModifiedSince>(connection, decoder, this, &WebCookieManager::deleteAllCookiesModifiedSince);
241 if (decoder.messageName() == Messages::WebCookieManager::SetHTTPCookieAcceptPolicy::name())
242 return IPC::handleMessageAsync<Messages::WebCookieManager::SetHTTPCookieAcceptPolicy>(connection, decoder, this, &WebCookieManager::setHTTPCookieAcceptPolicy);
243 if (decoder.messageName() == Messages::WebCookieManager::GetHTTPCookieAcceptPolicy::name())
244 return IPC::handleMessageAsync<Messages::WebCookieManager::GetHTTPCookieAcceptPolicy>(connection, decoder, this, &WebCookieManager::getHTTPCookieAcceptPolicy);
245 if (decoder.messageName() == Messages::WebCookieManager::StartObservingCookieChanges::name())
246 return IPC::handleMessage<Messages::WebCookieManager::StartObservingCookieChanges>(decoder, this, &WebCookieManager::startObservingCookieChanges);
247 if (decoder.messageName() == Messages::WebCookieManager::StopObservingCookieChanges::name())
248 return IPC::handleMessage<Messages::WebCookieManager::StopObservingCookieChanges>(decoder, this, &WebCookieManager::stopObservingCookieChanges);
249#if USE(SOUP)
250 if (decoder.messageName() == Messages::WebCookieManager::SetCookiePersistentStorage::name())
251 return IPC::handleMessage<Messages::WebCookieManager::SetCookiePersistentStorage>(decoder, this, &WebCookieManager::setCookiePersistentStorage);
252#endif
253 UNUSED_PARAM(connection);
254 UNUSED_PARAM(decoder);
255#if ENABLE(IPC_TESTING_API)
256 if (connection.ignoreInvalidMessageForTesting())
257 return;
258#endif // ENABLE(IPC_TESTING_API)
259 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
260}
