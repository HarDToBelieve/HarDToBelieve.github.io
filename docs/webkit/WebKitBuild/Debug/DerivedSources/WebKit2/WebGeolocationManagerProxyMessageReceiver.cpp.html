37void WebGeolocationManagerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
38{
39 auto protectedThis = makeRef(*this);
40 if (decoder.messageName() == Messages::WebGeolocationManagerProxy::StartUpdating::name())
41 return IPC::handleMessage<Messages::WebGeolocationManagerProxy::StartUpdating>(connection, decoder, this, &WebGeolocationManagerProxy::startUpdating);
42 if (decoder.messageName() == Messages::WebGeolocationManagerProxy::StopUpdating::name())
43 return IPC::handleMessage<Messages::WebGeolocationManagerProxy::StopUpdating>(connection, decoder, this, &WebGeolocationManagerProxy::stopUpdating);
44 if (decoder.messageName() == Messages::WebGeolocationManagerProxy::SetEnableHighAccuracy::name())
45 return IPC::handleMessage<Messages::WebGeolocationManagerProxy::SetEnableHighAccuracy>(connection, decoder, this, &WebGeolocationManagerProxy::setEnableHighAccuracy);
46 UNUSED_PARAM(connection);
47 UNUSED_PARAM(decoder);
48#if ENABLE(IPC_TESTING_API)
49 if (connection.ignoreInvalidMessageForTesting())
50 return;
51#endif // ENABLE(IPC_TESTING_API)
52 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
53}
