37void WebInspectorUIExtensionControllerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
38{
39 auto protectedThis = makeRef(*this);
40 if (decoder.messageName() == Messages::WebInspectorUIExtensionControllerProxy::DidShowExtensionTab::name())
41 return IPC::handleMessage<Messages::WebInspectorUIExtensionControllerProxy::DidShowExtensionTab>(decoder, this, &WebInspectorUIExtensionControllerProxy::didShowExtensionTab);
42 if (decoder.messageName() == Messages::WebInspectorUIExtensionControllerProxy::DidHideExtensionTab::name())
43 return IPC::handleMessage<Messages::WebInspectorUIExtensionControllerProxy::DidHideExtensionTab>(decoder, this, &WebInspectorUIExtensionControllerProxy::didHideExtensionTab);
44 UNUSED_PARAM(connection);
45 UNUSED_PARAM(decoder);
46#if ENABLE(IPC_TESTING_API)
47 if (connection.ignoreInvalidMessageForTesting())
48 return;
49#endif // ENABLE(IPC_TESTING_API)
50 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
51}
