38void WebInspectorUI::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
39{
40 auto protectedThis = makeRef(*this);
41 if (decoder.messageName() == Messages::WebInspectorUI::EstablishConnection::name())
42 return IPC::handleMessage<Messages::WebInspectorUI::EstablishConnection>(decoder, this, &WebInspectorUI::establishConnection);
43 if (decoder.messageName() == Messages::WebInspectorUI::UpdateConnection::name())
44 return IPC::handleMessage<Messages::WebInspectorUI::UpdateConnection>(decoder, this, &WebInspectorUI::updateConnection);
45 if (decoder.messageName() == Messages::WebInspectorUI::AttachedBottom::name())
46 return IPC::handleMessage<Messages::WebInspectorUI::AttachedBottom>(decoder, this, &WebInspectorUI::attachedBottom);
47 if (decoder.messageName() == Messages::WebInspectorUI::AttachedRight::name())
48 return IPC::handleMessage<Messages::WebInspectorUI::AttachedRight>(decoder, this, &WebInspectorUI::attachedRight);
49 if (decoder.messageName() == Messages::WebInspectorUI::AttachedLeft::name())
50 return IPC::handleMessage<Messages::WebInspectorUI::AttachedLeft>(decoder, this, &WebInspectorUI::attachedLeft);
51 if (decoder.messageName() == Messages::WebInspectorUI::Detached::name())
52 return IPC::handleMessage<Messages::WebInspectorUI::Detached>(decoder, this, &WebInspectorUI::detached);
53 if (decoder.messageName() == Messages::WebInspectorUI::SetDockingUnavailable::name())
54 return IPC::handleMessage<Messages::WebInspectorUI::SetDockingUnavailable>(decoder, this, &WebInspectorUI::setDockingUnavailable);
55 if (decoder.messageName() == Messages::WebInspectorUI::SetIsVisible::name())
56 return IPC::handleMessage<Messages::WebInspectorUI::SetIsVisible>(decoder, this, &WebInspectorUI::setIsVisible);
57 if (decoder.messageName() == Messages::WebInspectorUI::UpdateFindString::name())
58 return IPC::handleMessage<Messages::WebInspectorUI::UpdateFindString>(decoder, this, &WebInspectorUI::updateFindString);
59#if ENABLE(INSPECTOR_TELEMETRY)
60 if (decoder.messageName() == Messages::WebInspectorUI::SetDiagnosticLoggingAvailable::name())
61 return IPC::handleMessage<Messages::WebInspectorUI::SetDiagnosticLoggingAvailable>(decoder, this, &WebInspectorUI::setDiagnosticLoggingAvailable);
62#endif
63 if (decoder.messageName() == Messages::WebInspectorUI::ShowConsole::name())
64 return IPC::handleMessage<Messages::WebInspectorUI::ShowConsole>(decoder, this, &WebInspectorUI::showConsole);
65 if (decoder.messageName() == Messages::WebInspectorUI::ShowResources::name())
66 return IPC::handleMessage<Messages::WebInspectorUI::ShowResources>(decoder, this, &WebInspectorUI::showResources);
67 if (decoder.messageName() == Messages::WebInspectorUI::ShowMainResourceForFrame::name())
68 return IPC::handleMessage<Messages::WebInspectorUI::ShowMainResourceForFrame>(decoder, this, &WebInspectorUI::showMainResourceForFrame);
69 if (decoder.messageName() == Messages::WebInspectorUI::StartPageProfiling::name())
70 return IPC::handleMessage<Messages::WebInspectorUI::StartPageProfiling>(decoder, this, &WebInspectorUI::startPageProfiling);
71 if (decoder.messageName() == Messages::WebInspectorUI::StopPageProfiling::name())
72 return IPC::handleMessage<Messages::WebInspectorUI::StopPageProfiling>(decoder, this, &WebInspectorUI::stopPageProfiling);
73 if (decoder.messageName() == Messages::WebInspectorUI::StartElementSelection::name())
74 return IPC::handleMessage<Messages::WebInspectorUI::StartElementSelection>(decoder, this, &WebInspectorUI::startElementSelection);
75 if (decoder.messageName() == Messages::WebInspectorUI::StopElementSelection::name())
76 return IPC::handleMessage<Messages::WebInspectorUI::StopElementSelection>(decoder, this, &WebInspectorUI::stopElementSelection);
77 if (decoder.messageName() == Messages::WebInspectorUI::DidSave::name())
78 return IPC::handleMessage<Messages::WebInspectorUI::DidSave>(decoder, this, &WebInspectorUI::didSave);
79 if (decoder.messageName() == Messages::WebInspectorUI::DidAppend::name())
80 return IPC::handleMessage<Messages::WebInspectorUI::DidAppend>(decoder, this, &WebInspectorUI::didAppend);
81 if (decoder.messageName() == Messages::WebInspectorUI::SendMessageToFrontend::name())
82 return IPC::handleMessage<Messages::WebInspectorUI::SendMessageToFrontend>(decoder, this, &WebInspectorUI::sendMessageToFrontend);
83 if (decoder.messageName() == Messages::WebInspectorUI::EvaluateInFrontendForTesting::name())
84 return IPC::handleMessage<Messages::WebInspectorUI::EvaluateInFrontendForTesting>(decoder, this, &WebInspectorUI::evaluateInFrontendForTesting);
85 UNUSED_PARAM(connection);
86 UNUSED_PARAM(decoder);
87#if ENABLE(IPC_TESTING_API)
88 if (connection.ignoreInvalidMessageForTesting())
89 return;
90#endif // ENABLE(IPC_TESTING_API)
91 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
92}
