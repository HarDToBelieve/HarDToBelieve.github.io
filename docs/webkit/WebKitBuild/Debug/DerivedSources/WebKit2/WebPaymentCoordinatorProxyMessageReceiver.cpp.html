59void CanMakePaymentsWithActiveCard::callReply(IPC::Decoder& decoder, CompletionHandler<void(bool&&)>&& completionHandler)
60{
61 std::optional<bool> canMakePayments;
62 decoder >> canMakePayments;
63 if (!canMakePayments) {
64 ASSERT_NOT_REACHED();
65 cancelReply(WTFMove(completionHandler));
66 return;
67 }

69}

163void WebPaymentCoordinatorProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
164{
165 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CanMakePaymentsWithActiveCard::name())
166 return IPC::handleMessageAsync<Messages::WebPaymentCoordinatorProxy::CanMakePaymentsWithActiveCard>(connection, decoder, this, &WebPaymentCoordinatorProxy::canMakePaymentsWithActiveCard);
167 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::OpenPaymentSetup::name())
168 return IPC::handleMessageAsync<Messages::WebPaymentCoordinatorProxy::OpenPaymentSetup>(connection, decoder, this, &WebPaymentCoordinatorProxy::openPaymentSetup);
169 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CompleteMerchantValidation::name())
170 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CompleteMerchantValidation>(decoder, this, &WebPaymentCoordinatorProxy::completeMerchantValidation);
171 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CompleteShippingMethodSelection::name())
172 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CompleteShippingMethodSelection>(decoder, this, &WebPaymentCoordinatorProxy::completeShippingMethodSelection);
173 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CompleteShippingContactSelection::name())
174 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CompleteShippingContactSelection>(decoder, this, &WebPaymentCoordinatorProxy::completeShippingContactSelection);
175 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CompletePaymentMethodSelection::name())
176 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CompletePaymentMethodSelection>(decoder, this, &WebPaymentCoordinatorProxy::completePaymentMethodSelection);
177#if ENABLE(APPLE_PAY_COUPON_CODE)
178 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CompleteCouponCodeChange::name())
179 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CompleteCouponCodeChange>(decoder, this, &WebPaymentCoordinatorProxy::completeCouponCodeChange);
180#endif
181 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CompletePaymentSession::name())
182 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CompletePaymentSession>(decoder, this, &WebPaymentCoordinatorProxy::completePaymentSession);
183 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::AbortPaymentSession::name())
184 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::AbortPaymentSession>(decoder, this, &WebPaymentCoordinatorProxy::abortPaymentSession);
185 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CancelPaymentSession::name())
186 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::CancelPaymentSession>(decoder, this, &WebPaymentCoordinatorProxy::cancelPaymentSession);
187 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::GetSetupFeatures::name())
188 return IPC::handleMessageAsync<Messages::WebPaymentCoordinatorProxy::GetSetupFeatures>(connection, decoder, this, &WebPaymentCoordinatorProxy::getSetupFeatures);
189 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::BeginApplePaySetup::name())
190 return IPC::handleMessageAsync<Messages::WebPaymentCoordinatorProxy::BeginApplePaySetup>(connection, decoder, this, &WebPaymentCoordinatorProxy::beginApplePaySetup);
191 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::EndApplePaySetup::name())
192 return IPC::handleMessage<Messages::WebPaymentCoordinatorProxy::EndApplePaySetup>(decoder, this, &WebPaymentCoordinatorProxy::endApplePaySetup);
193 UNUSED_PARAM(connection);
194 UNUSED_PARAM(decoder);
195#if ENABLE(IPC_TESTING_API)
196 if (connection.ignoreInvalidMessageForTesting())
197 return;
198#endif // ENABLE(IPC_TESTING_API)
199 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
200}

202bool WebPaymentCoordinatorProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
203{
204 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::CanMakePayments::name())
205 return IPC::handleMessageSynchronous<Messages::WebPaymentCoordinatorProxy::CanMakePayments>(connection, decoder, replyEncoder, this, &WebPaymentCoordinatorProxy::canMakePayments);
206 if (decoder.messageName() == Messages::WebPaymentCoordinatorProxy::ShowPaymentUI::name())
207 return IPC::handleMessageSynchronous<Messages::WebPaymentCoordinatorProxy::ShowPaymentUI>(connection, decoder, replyEncoder, this, &WebPaymentCoordinatorProxy::showPaymentUI);
208 UNUSED_PARAM(connection);
209 UNUSED_PARAM(decoder);
210 UNUSED_PARAM(replyEncoder);
211#if ENABLE(IPC_TESTING_API)
212 if (connection.ignoreInvalidMessageForTesting())
213 return false;
214#endif // ENABLE(IPC_TESTING_API)
215 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
216 return false;
217}
