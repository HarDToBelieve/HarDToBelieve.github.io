44void WebResourceLoader::didReceiveWebResourceLoaderMessage(IPC::Connection& connection, IPC::Decoder& decoder)
45{
46 auto protectedThis = makeRef(*this);
47 if (decoder.messageName() == Messages::WebResourceLoader::WillSendRequest::name())
48 return IPC::handleMessage<Messages::WebResourceLoader::WillSendRequest>(decoder, this, &WebResourceLoader::willSendRequest);
49 if (decoder.messageName() == Messages::WebResourceLoader::DidSendData::name())
50 return IPC::handleMessage<Messages::WebResourceLoader::DidSendData>(decoder, this, &WebResourceLoader::didSendData);
51 if (decoder.messageName() == Messages::WebResourceLoader::DidReceiveResponse::name())
52 return IPC::handleMessage<Messages::WebResourceLoader::DidReceiveResponse>(decoder, this, &WebResourceLoader::didReceiveResponse);
53 if (decoder.messageName() == Messages::WebResourceLoader::DidReceiveData::name())
54 return IPC::handleMessage<Messages::WebResourceLoader::DidReceiveData>(decoder, this, &WebResourceLoader::didReceiveData);
55 if (decoder.messageName() == Messages::WebResourceLoader::DidFinishResourceLoad::name())
56 return IPC::handleMessage<Messages::WebResourceLoader::DidFinishResourceLoad>(decoder, this, &WebResourceLoader::didFinishResourceLoad);
57 if (decoder.messageName() == Messages::WebResourceLoader::DidFailResourceLoad::name())
58 return IPC::handleMessage<Messages::WebResourceLoader::DidFailResourceLoad>(decoder, this, &WebResourceLoader::didFailResourceLoad);
59 if (decoder.messageName() == Messages::WebResourceLoader::DidFailServiceWorkerLoad::name())
60 return IPC::handleMessage<Messages::WebResourceLoader::DidFailServiceWorkerLoad>(decoder, this, &WebResourceLoader::didFailServiceWorkerLoad);
61 if (decoder.messageName() == Messages::WebResourceLoader::ServiceWorkerDidNotHandle::name())
62 return IPC::handleMessage<Messages::WebResourceLoader::ServiceWorkerDidNotHandle>(decoder, this, &WebResourceLoader::serviceWorkerDidNotHandle);
63 if (decoder.messageName() == Messages::WebResourceLoader::DidBlockAuthenticationChallenge::name())
64 return IPC::handleMessage<Messages::WebResourceLoader::DidBlockAuthenticationChallenge>(decoder, this, &WebResourceLoader::didBlockAuthenticationChallenge);
65 if (decoder.messageName() == Messages::WebResourceLoader::StopLoadingAfterXFrameOptionsOrContentSecurityPolicyDenied::name())
66 return IPC::handleMessage<Messages::WebResourceLoader::StopLoadingAfterXFrameOptionsOrContentSecurityPolicyDenied>(decoder, this, &WebResourceLoader::stopLoadingAfterXFrameOptionsOrContentSecurityPolicyDenied);
67#if ENABLE(SHAREABLE_RESOURCE)
68 if (decoder.messageName() == Messages::WebResourceLoader::DidReceiveResource::name())
69 return IPC::handleMessage<Messages::WebResourceLoader::DidReceiveResource>(decoder, this, &WebResourceLoader::didReceiveResource);
70#endif
71 UNUSED_PARAM(connection);
72 UNUSED_PARAM(decoder);
73#if ENABLE(IPC_TESTING_API)
74 if (connection.ignoreInvalidMessageForTesting())
75 return;
76#endif // ENABLE(IPC_TESTING_API)
77 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
78}
