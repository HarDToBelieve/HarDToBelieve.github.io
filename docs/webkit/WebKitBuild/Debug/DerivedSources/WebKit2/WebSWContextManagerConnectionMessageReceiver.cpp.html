60void WebSWContextManagerConnection::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
61{
62 if (decoder.messageName() == Messages::WebSWContextManagerConnection::InstallServiceWorker::name())
63 return IPC::handleMessage<Messages::WebSWContextManagerConnection::InstallServiceWorker>(decoder, this, &WebSWContextManagerConnection::installServiceWorker);
64 if (decoder.messageName() == Messages::WebSWContextManagerConnection::UpdateAppBoundValue::name())
65 return IPC::handleMessage<Messages::WebSWContextManagerConnection::UpdateAppBoundValue>(decoder, this, &WebSWContextManagerConnection::updateAppBoundValue);
66 if (decoder.messageName() == Messages::WebSWContextManagerConnection::StartFetch::name())
67 return IPC::handleMessage<Messages::WebSWContextManagerConnection::StartFetch>(decoder, this, &WebSWContextManagerConnection::startFetch);
68 if (decoder.messageName() == Messages::WebSWContextManagerConnection::CancelFetch::name())
69 return IPC::handleMessage<Messages::WebSWContextManagerConnection::CancelFetch>(decoder, this, &WebSWContextManagerConnection::cancelFetch);
70 if (decoder.messageName() == Messages::WebSWContextManagerConnection::ContinueDidReceiveFetchResponse::name())
71 return IPC::handleMessage<Messages::WebSWContextManagerConnection::ContinueDidReceiveFetchResponse>(decoder, this, &WebSWContextManagerConnection::continueDidReceiveFetchResponse);
72 if (decoder.messageName() == Messages::WebSWContextManagerConnection::PostMessageToServiceWorker::name())
73 return IPC::handleMessage<Messages::WebSWContextManagerConnection::PostMessageToServiceWorker>(decoder, this, &WebSWContextManagerConnection::postMessageToServiceWorker);
74 if (decoder.messageName() == Messages::WebSWContextManagerConnection::FireInstallEvent::name())
75 return IPC::handleMessage<Messages::WebSWContextManagerConnection::FireInstallEvent>(decoder, this, &WebSWContextManagerConnection::fireInstallEvent);
76 if (decoder.messageName() == Messages::WebSWContextManagerConnection::FireActivateEvent::name())
77 return IPC::handleMessage<Messages::WebSWContextManagerConnection::FireActivateEvent>(decoder, this, &WebSWContextManagerConnection::fireActivateEvent);
78 if (decoder.messageName() == Messages::WebSWContextManagerConnection::TerminateWorker::name())
79 return IPC::handleMessage<Messages::WebSWContextManagerConnection::TerminateWorker>(decoder, this, &WebSWContextManagerConnection::terminateWorker);
80#if ENABLE(SHAREABLE_RESOURCE) && PLATFORM(COCOA)
81 if (decoder.messageName() == Messages::WebSWContextManagerConnection::DidSaveScriptsToDisk::name())
82 return IPC::handleMessage<Messages::WebSWContextManagerConnection::DidSaveScriptsToDisk>(decoder, this, &WebSWContextManagerConnection::didSaveScriptsToDisk);
83#endif
84 if (decoder.messageName() == Messages::WebSWContextManagerConnection::FindClientByIdentifierCompleted::name())
85 return IPC::handleMessage<Messages::WebSWContextManagerConnection::FindClientByIdentifierCompleted>(decoder, this, &WebSWContextManagerConnection::findClientByIdentifierCompleted);
86 if (decoder.messageName() == Messages::WebSWContextManagerConnection::MatchAllCompleted::name())
87 return IPC::handleMessage<Messages::WebSWContextManagerConnection::MatchAllCompleted>(decoder, this, &WebSWContextManagerConnection::matchAllCompleted);
88 if (decoder.messageName() == Messages::WebSWContextManagerConnection::SetUserAgent::name())
89 return IPC::handleMessage<Messages::WebSWContextManagerConnection::SetUserAgent>(decoder, this, &WebSWContextManagerConnection::setUserAgent);
90 if (decoder.messageName() == Messages::WebSWContextManagerConnection::UpdatePreferencesStore::name())
91 return IPC::handleMessage<Messages::WebSWContextManagerConnection::UpdatePreferencesStore>(decoder, this, &WebSWContextManagerConnection::updatePreferencesStore);
92 if (decoder.messageName() == Messages::WebSWContextManagerConnection::Close::name())
93 return IPC::handleMessage<Messages::WebSWContextManagerConnection::Close>(decoder, this, &WebSWContextManagerConnection::close);
94 if (decoder.messageName() == Messages::WebSWContextManagerConnection::SetThrottleState::name())
95 return IPC::handleMessage<Messages::WebSWContextManagerConnection::SetThrottleState>(decoder, this, &WebSWContextManagerConnection::setThrottleState);
96 UNUSED_PARAM(connection);
97 UNUSED_PARAM(decoder);
98#if ENABLE(IPC_TESTING_API)
99 if (connection.ignoreInvalidMessageForTesting())
100 return;
101#endif // ENABLE(IPC_TESTING_API)
102 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
103}
