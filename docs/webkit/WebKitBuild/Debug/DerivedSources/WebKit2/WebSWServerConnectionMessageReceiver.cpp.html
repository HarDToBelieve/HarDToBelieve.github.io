53void ScheduleUnregisterJobInServer::callReply(IPC::Decoder& decoder, CompletionHandler<void(Expected<bool, WebCore::ExceptionData>&&)>&& completionHandler)
54{
55 std::optional<Expected<bool, WebCore::ExceptionData>> result;
56 decoder >> result;
57 if (!result) {
58 ASSERT_NOT_REACHED();
59 cancelReply(WTFMove(completionHandler));
60 return;
61 }
62 completionHandler(WTFMove(*result));
63}
64
65void ScheduleUnregisterJobInServer::cancelReply(CompletionHandler<void(Expected<bool, WebCore::ExceptionData>&&)>&& completionHandler)
66{
67 completionHandler(IPC::AsyncReplyError<Expected<bool, WebCore::ExceptionData>>::create());
68}

70void ScheduleUnregisterJobInServer::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const Expected<bool, WebCore::ExceptionData>& result)
71{
72 encoder.get() << result;
73 connection.sendSyncReply(WTFMove(encoder));
74}

127void WebSWServerConnection::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
128{
129 if (decoder.messageName() == Messages::WebSWServerConnection::ScheduleJobInServer::name())
130 return IPC::handleMessage<Messages::WebSWServerConnection::ScheduleJobInServer>(decoder, this, &WebSWServerConnection::scheduleJobInServer);
131 if (decoder.messageName() == Messages::WebSWServerConnection::ScheduleUnregisterJobInServer::name())
132 return IPC::handleMessageAsync<Messages::WebSWServerConnection::ScheduleUnregisterJobInServer>(connection, decoder, this, &WebSWServerConnection::scheduleUnregisterJobInServer);
133 if (decoder.messageName() == Messages::WebSWServerConnection::FinishFetchingScriptInServer::name())
134 return IPC::handleMessage<Messages::WebSWServerConnection::FinishFetchingScriptInServer>(decoder, this, &WebSWServerConnection::finishFetchingScriptInServer);
135 if (decoder.messageName() == Messages::WebSWServerConnection::AddServiceWorkerRegistrationInServer::name())
136 return IPC::handleMessage<Messages::WebSWServerConnection::AddServiceWorkerRegistrationInServer>(decoder, this, &WebSWServerConnection::addServiceWorkerRegistrationInServer);
137 if (decoder.messageName() == Messages::WebSWServerConnection::RemoveServiceWorkerRegistrationInServer::name())
138 return IPC::handleMessage<Messages::WebSWServerConnection::RemoveServiceWorkerRegistrationInServer>(decoder, this, &WebSWServerConnection::removeServiceWorkerRegistrationInServer);
139 if (decoder.messageName() == Messages::WebSWServerConnection::PostMessageToServiceWorker::name())
140 return IPC::handleMessage<Messages::WebSWServerConnection::PostMessageToServiceWorker>(decoder, this, &WebSWServerConnection::postMessageToServiceWorker);
141 if (decoder.messageName() == Messages::WebSWServerConnection::DidResolveRegistrationPromise::name())
142 return IPC::handleMessage<Messages::WebSWServerConnection::DidResolveRegistrationPromise>(decoder, this, &WebSWServerConnection::didResolveRegistrationPromise);
143 if (decoder.messageName() == Messages::WebSWServerConnection::MatchRegistration::name())
144 return IPC::handleMessage<Messages::WebSWServerConnection::MatchRegistration>(decoder, this, &WebSWServerConnection::matchRegistration);
145 if (decoder.messageName() == Messages::WebSWServerConnection::WhenRegistrationReady::name())
146 return IPC::handleMessage<Messages::WebSWServerConnection::WhenRegistrationReady>(decoder, this, &WebSWServerConnection::whenRegistrationReady);
147 if (decoder.messageName() == Messages::WebSWServerConnection::GetRegistrations::name())
148 return IPC::handleMessage<Messages::WebSWServerConnection::GetRegistrations>(decoder, this, &WebSWServerConnection::getRegistrations);
149 if (decoder.messageName() == Messages::WebSWServerConnection::RegisterServiceWorkerClient::name())
150 return IPC::handleMessage<Messages::WebSWServerConnection::RegisterServiceWorkerClient>(decoder, this, &WebSWServerConnection::registerServiceWorkerClient);
151 if (decoder.messageName() == Messages::WebSWServerConnection::UnregisterServiceWorkerClient::name())
152 return IPC::handleMessage<Messages::WebSWServerConnection::UnregisterServiceWorkerClient>(decoder, this, &WebSWServerConnection::unregisterServiceWorkerClient);
153 if (decoder.messageName() == Messages::WebSWServerConnection::TerminateWorkerFromClient::name())
154 return IPC::handleMessageAsync<Messages::WebSWServerConnection::TerminateWorkerFromClient>(connection, decoder, this, &WebSWServerConnection::terminateWorkerFromClient);
155 if (decoder.messageName() == Messages::WebSWServerConnection::WhenServiceWorkerIsTerminatedForTesting::name())
156 return IPC::handleMessageAsync<Messages::WebSWServerConnection::WhenServiceWorkerIsTerminatedForTesting>(connection, decoder, this, &WebSWServerConnection::whenServiceWorkerIsTerminatedForTesting);
157 if (decoder.messageName() == Messages::WebSWServerConnection::SetThrottleState::name())
158 return IPC::handleMessage<Messages::WebSWServerConnection::SetThrottleState>(decoder, this, &WebSWServerConnection::setThrottleState);
159 if (decoder.messageName() == Messages::WebSWServerConnection::StoreRegistrationsOnDisk::name())
160 return IPC::handleMessageAsync<Messages::WebSWServerConnection::StoreRegistrationsOnDisk>(connection, decoder, this, &WebSWServerConnection::storeRegistrationsOnDisk);
161 UNUSED_PARAM(connection);
162 UNUSED_PARAM(decoder);
163#if ENABLE(IPC_TESTING_API)
164 if (connection.ignoreInvalidMessageForTesting())
165 return;
166#endif // ENABLE(IPC_TESTING_API)
167 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
168}
