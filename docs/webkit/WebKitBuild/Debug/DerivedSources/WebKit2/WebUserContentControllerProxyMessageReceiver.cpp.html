41void DidPostMessage::callReply(IPC::Decoder& decoder, CompletionHandler<void(IPC::DataReference&&, String&&)>&& completionHandler)
42{
43 std::optional<IPC::DataReference> resultValue;
44 decoder >> resultValue;
45 if (!resultValue) {
46 ASSERT_NOT_REACHED();
47 cancelReply(WTFMove(completionHandler));
48 return;
49 }
50 std::optional<String> errorMessage;
51 decoder >> errorMessage;
52 if (!errorMessage) {
53 ASSERT_NOT_REACHED();
54 cancelReply(WTFMove(completionHandler));
55 return;
56 }
57 completionHandler(WTFMove(*resultValue), WTFMove(*errorMessage));
58}
59
60void DidPostMessage::cancelReply(CompletionHandler<void(IPC::DataReference&&, String&&)>&& completionHandler)
61{
62 completionHandler(IPC::AsyncReplyError<IPC::DataReference>::create(), IPC::AsyncReplyError<String>::create());
63}
64
65void DidPostMessage::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const IPC::DataReference& resultValue, const String& errorMessage)
66{
67 encoder.get() << resultValue;
68 encoder.get() << errorMessage;
69 connection.sendSyncReply(WTFMove(encoder));
70}

78void WebUserContentControllerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
79{
80 auto protectedThis = makeRef(*this);
81 if (decoder.messageName() == Messages::WebUserContentControllerProxy::DidPostMessage::name())
82 return IPC::handleMessageAsync<Messages::WebUserContentControllerProxy::DidPostMessage>(connection, decoder, this, &WebUserContentControllerProxy::didPostMessage);
83 UNUSED_PARAM(connection);
84 UNUSED_PARAM(decoder);
85#if ENABLE(IPC_TESTING_API)
86 if (connection.ignoreInvalidMessageForTesting())
87 return;
88#endif // ENABLE(IPC_TESTING_API)
89 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
90}
