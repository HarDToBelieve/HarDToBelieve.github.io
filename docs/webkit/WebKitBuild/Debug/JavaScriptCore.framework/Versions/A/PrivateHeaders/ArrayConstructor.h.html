63// ES6 7.2.2
64// https://tc39.github.io/ecma262/#sec-isarray
65inline bool isArray(JSGlobalObject* globalObject, JSValue argumentValue)
66{
67 if (!argumentValue.isObject())
68 return false;
69
70 JSObject* argument = jsCast<JSObject*>(argumentValue);
71 if (argument->type() == ArrayType || argument->type() == DerivedArrayType)
72 return true;
73
74 if (argument->type() != ProxyObjectType)
75 return false;
76 return isArraySlow(globalObject, jsCast<ProxyObject*>(argument));
77}
