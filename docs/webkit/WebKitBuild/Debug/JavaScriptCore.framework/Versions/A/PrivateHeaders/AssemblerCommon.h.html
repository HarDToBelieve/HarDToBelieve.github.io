48ALWAYS_INLINE bool isInt9(int32_t value)
49{
50 return value == ((value << 23) >> 23);
51}

53template<typename Type>
54ALWAYS_INLINE bool isUInt12(Type value)
55{
56 return !(value & ~static_cast<Type>(0xfff));
57}

59template<int datasize>
60ALWAYS_INLINE bool isValidScaledUImm12(int32_t offset)
61{
62 int32_t maxPImm = 4095 * (datasize / 8);
63 if (offset < 0)
64 return false;
65 if (offset > maxPImm)
66 return false;
67 if (offset & ((datasize / 8) - 1))
68 return false;
69 return true;
70}

77ALWAYS_INLINE bool isValidSignedImm7(int32_t value, int alignmentShiftAmount)
78{
79 constexpr int32_t disallowedHighBits = 32 - 7;
80 int32_t shiftedValue = value >> alignmentShiftAmount;
81 bool fitsIn7Bits = shiftedValue == ((shiftedValue << disallowedHighBits) >> disallowedHighBits);
82 bool hasCorrectAlignment = value == (shiftedValue << alignmentShiftAmount);
83 return fitsIn7Bits && hasCorrectAlignment;
84}

86class ARM64LogicalImmediate {
87public:
88 static ARM64LogicalImmediate create32(uint32_t value)
89 {
90 // Check for 0, -1 - these cannot be encoded.
91 if (!value || !~value)
92 return InvalidLogicalImmediate;
93
94 // First look for a 32-bit pattern, then for repeating 16-bit
95 // patterns, 8-bit, 4-bit, and finally 2-bit.
96
97 unsigned hsb, lsb;
98 bool inverted;
99 if (findBitRange<32>(value, hsb, lsb, inverted))
100 return encodeLogicalImmediate<32>(hsb, lsb, inverted);
101
102 if ((value & 0xffff) != (value >> 16))
103 return InvalidLogicalImmediate;
104 value &= 0xffff;
105
106 if (findBitRange<16>(value, hsb, lsb, inverted))
107 return encodeLogicalImmediate<16>(hsb, lsb, inverted);
108
109 if ((value & 0xff) != (value >> 8))
110 return InvalidLogicalImmediate;
111 value &= 0xff;
112
113 if (findBitRange<8>(value, hsb, lsb, inverted))
114 return encodeLogicalImmediate<8>(hsb, lsb, inverted);
115
116 if ((value & 0xf) != (value >> 4))
117 return InvalidLogicalImmediate;
118 value &= 0xf;
119
120 if (findBitRange<4>(value, hsb, lsb, inverted))
121 return encodeLogicalImmediate<4>(hsb, lsb, inverted);
122
123 if ((value & 0x3) != (value >> 2))
124 return InvalidLogicalImmediate;
125 value &= 0x3;
126
127 if (findBitRange<2>(value, hsb, lsb, inverted))
128 return encodeLogicalImmediate<2>(hsb, lsb, inverted);
129
130 return InvalidLogicalImmediate;
131 }
132
133 static ARM64LogicalImmediate create64(uint64_t value)
134 {
135 // Check for 0, -1 - these cannot be encoded.
136 if (!value || !~value)
137 return InvalidLogicalImmediate;
138
139 // Look for a contiguous bit range.
140 unsigned hsb, lsb;
141 bool inverted;
142 if (findBitRange<64>(value, hsb, lsb, inverted))
143 return encodeLogicalImmediate<64>(hsb, lsb, inverted);
144
145 // If the high & low 32 bits are equal, we can try for a 32-bit (or narrower) pattern.
146 if (static_cast<uint32_t>(value) == static_cast<uint32_t>(value >> 32))
147 return create32(static_cast<uint32_t>(value));
148 return InvalidLogicalImmediate;
149 }
150
151 int value() const
152 {
153 ASSERT(isValid());
154 return m_value;
155 }
156
157 bool isValid() const
158 {
159 return m_value != InvalidLogicalImmediate;
160 }
161
162 bool is64bit() const
163 {
164 return m_value & (1 << 12);
165 }

167private:
168 ARM64LogicalImmediate(int value)
169 : m_value(value)

173 // Generate a mask with bits in the range hsb..0 set, for example:
174 // hsb:63 = 0xffffffffffffffff
175 // hsb:42 = 0x000007ffffffffff
176 // hsb: 0 = 0x0000000000000001
177 static uint64_t mask(unsigned hsb)
178 {
179 ASSERT(hsb < 64);
180 return 0xffffffffffffffffull >> (63 - hsb);
181 }
182
183 template<unsigned N>
184 static void partialHSB(uint64_t& value, unsigned&result)
185 {
186 if (value & (0xffffffffffffffffull << N)) {
187 result += N;
188 value >>= N;
189 }
190 }
191
192 // Find the bit number of the highest bit set in a non-zero value, for example:
193 // 0x8080808080808080 = hsb:63
194 // 0x0000000000000001 = hsb: 0
195 // 0x000007ffffe00000 = hsb:42
196 static unsigned highestSetBit(uint64_t value)
197 {
198 ASSERT(value);
199 unsigned hsb = 0;
200 partialHSB<32>(value, hsb);
201 partialHSB<16>(value, hsb);
202 partialHSB<8>(value, hsb);
203 partialHSB<4>(value, hsb);
204 partialHSB<2>(value, hsb);
205 partialHSB<1>(value, hsb);
206 return hsb;
207 }
208
209 // This function takes a value and a bit width, where value obeys the following constraints:
210 // * bits outside of the width of the value must be zero.
211 // * bits within the width of value must neither be all clear or all set.
212 // The input is inspected to detect values that consist of either two or three contiguous
213 // ranges of bits. The output range hsb..lsb will describe the second range of the value.
214 // if the range is set, inverted will be false, and if the range is clear, inverted will
215 // be true. For example (with width 8):
216 // 00001111 = hsb:3, lsb:0, inverted:false
217 // 11110000 = hsb:3, lsb:0, inverted:true
218 // 00111100 = hsb:5, lsb:2, inverted:false
219 // 11000011 = hsb:5, lsb:2, inverted:true
220 template<unsigned width>
221 static bool findBitRange(uint64_t value, unsigned& hsb, unsigned& lsb, bool& inverted)
222 {
223 ASSERT(value & mask(width - 1));
224 ASSERT(value != mask(width - 1));
225 ASSERT(!(value & ~mask(width - 1)));
226
227 // Detect cases where the top bit is set; if so, flip all the bits & set invert.
228 // This halves the number of patterns we need to look for.
229 const uint64_t msb = 1ull << (width - 1);
230 if ((inverted = (value & msb)))
231 value ^= mask(width - 1);
232
233 // Find the highest set bit in value, generate a corresponding mask & flip all
234 // bits under it.
235 hsb = highestSetBit(value);
236 value ^= mask(hsb);
237 if (!value) {
238 // If this cleared the value, then the range hsb..0 was all set.
239 lsb = 0;
240 return true;
241 }
242
243 // Try making one more mask, and flipping the bits!
244 lsb = highestSetBit(value);
245 value ^= mask(lsb);
246 if (!value) {
247 // Success - but lsb actually points to the hsb of a third range - add one
248 // to get to the lsb of the mid range.
249 ++lsb;
250 return true;
251 }
252
253 return false;
254 }

256 // Encodes the set of immN:immr:imms fields found in a logical immediate.
257 template<unsigned width>
258 static int encodeLogicalImmediate(unsigned hsb, unsigned lsb, bool inverted)
259 {
260 // Check width is a power of 2!
261 ASSERT(!(width & (width -1)));
262 ASSERT(width <= 64 && width >= 2);
263 ASSERT(hsb >= lsb);
264 ASSERT(hsb < width);
265
266 int immN = 0;
267 int imms = 0;
268 int immr = 0;
269
270 // For 64-bit values this is easy - just set immN to true, and imms just
271 // contains the bit number of the highest set bit of the set range. For
272 // values with narrower widths, these are encoded by a leading set of
273 // one bits, followed by a zero bit, followed by the remaining set of bits
274 // being the high bit of the range. For a 32-bit immediate there are no
275 // leading one bits, just a zero followed by a five bit number. For a
276 // 16-bit immediate there is one one bit, a zero bit, and then a four bit
277 // bit-position, etc.
278 if (width == 64)
279 immN = 1;
280 else
281 imms = 63 & ~(width + width - 1);
282
283 if (inverted) {
284 // if width is 64 & hsb is 62, then we have a value something like:
285 // 0x80000000ffffffff (in this case with lsb 32).
286 // The ror should be by 1, imms (effectively set width minus 1) is
287 // 32. Set width is full width minus cleared width.
288 immr = (width - 1) - hsb;
289 imms |= (width - ((hsb - lsb) + 1)) - 1;
290 } else {
291 // if width is 64 & hsb is 62, then we have a value something like:
292 // 0x7fffffff00000000 (in this case with lsb 32).
293 // The value is effectively rol'ed by lsb, which is equivalent to
294 // a ror by width - lsb (or 0, in the case where lsb is 0). imms
295 // is hsb - lsb.
296 immr = (width - lsb) & (width - 1);
297 imms |= hsb - lsb;
298 }
299
300 return immN << 12 | immr << 6 | imms;
301 }
302
303 static constexpr int InvalidLogicalImmediate = -1;
304
305 int m_value;
306};
