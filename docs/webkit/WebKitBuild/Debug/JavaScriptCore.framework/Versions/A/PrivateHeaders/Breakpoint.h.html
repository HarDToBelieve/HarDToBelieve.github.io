26#pragma once
27
28#include "DebuggerPrimitives.h"
29#include <wtf/FastMalloc.h>
30#include <wtf/Noncopyable.h>
31#include <wtf/Ref.h>
32#include <wtf/RefCounted.h>
33#include <wtf/Vector.h>
34#include <wtf/text/WTFString.h>
35
36namespace JSC {
37
38class Debugger;
39class JSGlobalObject;
40
41class Breakpoint : public RefCounted<Breakpoint> {
42 WTF_MAKE_NONCOPYABLE(Breakpoint);
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 struct Action {
46 enum class Type : uint8_t {
47 Log,
48 Evaluate,
49 Sound,
50 Probe,
51 };
52
53 Action(Type);
54
55 Type type;
56 String data;
57 BreakpointActionID id { noBreakpointActionID };
58 bool emulateUserGesture { false };
59 };
60
61 using ActionsVector = Vector<Action>;
62
63 JS_EXPORT_PRIVATE static Ref<Breakpoint> create(BreakpointID, const String& condition = nullString(), ActionsVector&& = { }, bool autoContinue = false, size_t ignoreCount = 0);
64
65 BreakpointID id() const { return m_id; }
66
67 SourceID sourceID() const { return m_sourceID; }
68 unsigned lineNumber() const { return m_lineNumber; }
69 unsigned columnNumber() const { return m_columnNumber; }
70
71 const String& condition() const { return m_condition; }
72 const ActionsVector& actions() const { return m_actions; }
73 bool isAutoContinue() const { return m_autoContinue; }
74
75 void resetHitCount() { m_hitCount = 0; }
76
77 // Associates this breakpoint with a position in a specific source code.
78 bool link(SourceID, unsigned lineNumber, unsigned columnNumber);
79 bool isLinked() const { return m_sourceID != noSourceID; }
80
81 // Adjust the previously associated position to the next pause opportunity.
82 bool resolve(unsigned lineNumber, unsigned columnNumber);
83 bool isResolved() const { return m_resolved; }
84
85 bool shouldPause(Debugger&, JSGlobalObject*);
86
87private:
88 Breakpoint(BreakpointID, String condition = nullString(), ActionsVector&& = { }, bool autoContinue = false, size_t ignoreCount = 0);
89
90 BreakpointID m_id { noBreakpointID };
91
92 SourceID m_sourceID { noSourceID };
93
94 // FIXME: <https://webkit.org/b/162771> Web Inspector: Adopt TextPosition in Inspector to avoid oneBasedInt/zeroBasedInt ambiguity
95 unsigned m_lineNumber { 0 };
96 unsigned m_columnNumber { 0 };
97
98 bool m_resolved { false };
99
100 String m_condition;
101 ActionsVector m_actions;
102 bool m_autoContinue { false };
103 size_t m_ignoreCount { 0 };
104 size_t m_hitCount { 0 };
105};
106
107using BreakpointsVector = Vector<Ref<JSC::Breakpoint>>;
108
109} // namespace JSC
