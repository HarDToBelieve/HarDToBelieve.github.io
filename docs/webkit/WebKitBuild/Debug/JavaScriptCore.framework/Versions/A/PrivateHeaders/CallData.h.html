36class ArgList;
37class Exception;
38class FunctionExecutable;

41struct CallData {
42 enum class Type : uint8_t { None, Native, JS };
43 Type type { Type::None };
44
45 union {
46 struct {
47 TaggedNativeFunction function;
48 } native;
49 struct {
50 FunctionExecutable* functionExecutable;
51 JSScope* scope;
52 } js;
53 };
54};
55
56enum class ProfilingReason : uint8_t {
57 API,
58 Microtask,
59 Other
60};
