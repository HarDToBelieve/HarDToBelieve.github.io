92 enum class CallFrameSlot {
93 codeBlock = CallerFrameAndPC::sizeInRegisters,
94 callee = codeBlock + 1,
95 argumentCountIncludingThis = callee + 1,
96 thisArgument = argumentCountIncludingThis + 1,
97 firstArgument = thisArgument + 1,
98 };

110 // when you've guaranteed the callee can't be a Wasm callee, and can
111 // be an arbitrary JSValue. This function should basically never be used.
112 // Its only use right now is when we are making a call, and we're not
113 // yet sure if the callee is a cell. In general, a JS callee is guaranteed
114 // to be a cell, however, there is a brief window where we need to check
115 // to see if it's a cell, and if it's not, we throw an exception.

201 static int argumentOffset(int argument) { return (CallFrameSlot::firstArgument + argument); }
202 static int argumentOffsetIncludingThis(int argument) { return (CallFrameSlot::thisArgument + argument); }
203
204 // In the following (argument() and setArgument()), the 'argument'
205 // parameter is the index of the arguments of the target function of
206 // this frame. The index starts at 0 for the first arg, 1 for the
207 // second, etc.
208 //
209 // The arguments (in this case) do not include the 'this' value.
210 // arguments(0) will not fetch the 'this' value. To get/set 'this',
211 // use thisValue() and setThisValue() below.
212
213 JSValue* addressOfArgumentsStart() const { return bitwise_cast<JSValue*>(this + argumentOffset(0)); }
214 JSValue argument(size_t argument) const
215 {
216 if (argument >= argumentCount())
217 return jsUndefined();
218 return getArgumentUnsafe(argument);
219 }
220 JSValue uncheckedArgument(size_t argument) const
221 {
222 ASSERT(argument < argumentCount());
223 return getArgumentUnsafe(argument);
224 }
225 void setArgument(size_t argument, JSValue value)
226 {
227 this[argumentOffset(argument)] = value;
228 }

230 JSValue getArgumentUnsafe(size_t argIndex) const
231 {
232 // User beware! This method does not verify that there is a valid
233 // argument at the specified argIndex. This is used for debugging
234 // and verification code only. The caller is expected to know what
235 // he/she is doing when calling this method.
236 return this[argumentOffset(argIndex)].jsValue();
237 }

268 // CallFrame::iterate() expects a Functor that implements the following method:
269 // StackVisitor::Status operator()(StackVisitor&) const;
270 // FIXME: This method is improper. We rely on the fact that we can call it with a null
271 // receiver. We should always be using StackVisitor directly.
272 // It's only valid to call this from a non-wasm top frame.
273 template <StackVisitor::EmptyEntryFrameAction action = StackVisitor::ContinueIfTopEntryFrameIsEmpty, typename Functor> void iterate(VM& vm, const Functor& functor)
274 {
275 void* rawThis = this;
276 if (!!rawThis)
277 RELEASE_ASSERT(callee().isCell());
278 StackVisitor::visit<action, Functor>(this, vm, functor);
279 }

293 // The following are for internal use in debugging and verification
294 // code only and not meant as an API for general usage:
295
296 size_t argIndexForRegister(Register* reg)
297 {
298 // The register at 'offset' number of slots from the frame pointer
299 // i.e.
300 // reg = frame[offset];
301 // ==> reg = frame + offset;
302 // ==> offset = reg - frame;
303 int offset = reg - this->registers();
304
305 // The offset is defined (based on argumentOffset()) to be:
306 // offset = CallFrameSlot::firstArgument - argIndex;
307 // Hence:
308 // argIndex = CallFrameSlot::firstArgument - offset;
309 size_t argIndex = offset - CallFrameSlot::firstArgument;
310 return argIndex;
311 }
