89 static CallMode callModeFor(CallType callType)
90 {
91 switch (callType) {
92 case Call:
93 case CallVarargs:
94 case DirectCall:
95 return CallMode::Regular;
96 case TailCall:
97 case TailCallVarargs:
98 case DirectTailCall:
99 return CallMode::Tail;
100 case Construct:
101 case ConstructVarargs:
102 case DirectConstruct:
103 return CallMode::Construct;
104 case None:
105 RELEASE_ASSERT_NOT_REACHED();
106 }
107
108 RELEASE_ASSERT_NOT_REACHED();
109 }

111 static bool isDirect(CallType callType)
112 {
113 switch (callType) {
114 case DirectCall:
115 case DirectTailCall:
116 case DirectConstruct:
117 return true;
118 case Call:
119 case CallVarargs:
120 case TailCall:
121 case TailCallVarargs:
122 case Construct:
123 case ConstructVarargs:
124 return false;
125 case None:
126 RELEASE_ASSERT_NOT_REACHED();
127 return false;
128 }
129
130 RELEASE_ASSERT_NOT_REACHED();
131 return false;
132 }

358 template<typename Functor>
359 void forEachDependentCell(const Functor& functor) const
360 {
361 if (isLinked()) {
362 if (stub())
363 stub()->forEachDependentCell(functor);
364 else {
365 functor(m_calleeOrCodeBlock.get());
366 if (isDirect())
367 functor(m_lastSeenCalleeOrExecutable.get());
368 }
369 }
370 if (!isDirect() && haveLastSeenCallee())
371 functor(lastSeenCallee());
372 }
