28#include "DeferGC.h"
29#include <wtf/Lock.h>
30#include <wtf/NoLock.h>

32namespace JSC {
33
34using ConcurrentJSLock = Lock;
35using ConcurrentJSLockerImpl = LockHolder;
36
37static_assert(sizeof(ConcurrentJSLock) == 1, "Regardless of status of concurrent JS flag, size of ConurrentJSLock is always one byte.");
38
39class ConcurrentJSLockerBase : public AbstractLocker {
40 WTF_MAKE_NONCOPYABLE(ConcurrentJSLockerBase);
41public:
42 explicit ConcurrentJSLockerBase(ConcurrentJSLock& lockable)
43 {
44 m_locker.emplace(lockable);
45 }
46 explicit ConcurrentJSLockerBase(ConcurrentJSLock* lockable)
47 {
48 if (lockable)
49 m_locker.emplace(*lockable);
50 }
51
52 explicit ConcurrentJSLockerBase(NoLockingNecessaryTag)
53 {
54 }
55
56 ~ConcurrentJSLockerBase()
57 {
58 }
59
60 void unlockEarly() WTF_IGNORES_THREAD_SAFETY_ANALYSIS
61 {
62 if (m_locker)
63 m_locker->unlockEarly();
64 }
65
66private:
67 std::optional<ConcurrentJSLockerImpl> m_locker;
68};
69
70class GCSafeConcurrentJSLocker : public ConcurrentJSLockerBase {
71public:
72 GCSafeConcurrentJSLocker(ConcurrentJSLock& lockable, Heap& heap)
73 : ConcurrentJSLockerBase(lockable)
74 , m_deferGC(heap)
75 {
76 }
77
78 GCSafeConcurrentJSLocker(ConcurrentJSLock* lockable, Heap& heap)
79 : ConcurrentJSLockerBase(lockable)
80 , m_deferGC(heap)
81 {
82 }
83
84 ~GCSafeConcurrentJSLocker()
85 {
86 // We have to unlock early due to the destruction order of base
87 // vs. derived classes. If we didn't, then we would destroy the
88 // DeferGC object before unlocking the lock which could cause a GC
89 // and resulting deadlock.
90 unlockEarly();
91 }
92
93private:
94 DeferGC m_deferGC;
95};
96
97class ConcurrentJSLocker : public ConcurrentJSLockerBase {
98public:
99 ConcurrentJSLocker(ConcurrentJSLock& lockable)
100 : ConcurrentJSLockerBase(lockable)
101#if !defined(NDEBUG)
102 , m_disallowGC(std::in_place)
103#endif
104 {
105 }
106
107 ConcurrentJSLocker(ConcurrentJSLock* lockable)
108 : ConcurrentJSLockerBase(lockable)
109#if !defined(NDEBUG)
110 , m_disallowGC(std::in_place)
111#endif
112 {
113 }
114
115 ConcurrentJSLocker(NoLockingNecessaryTag)
116 : ConcurrentJSLockerBase(NoLockingNecessary)
117#if !defined(NDEBUG)
118 , m_disallowGC(std::nullopt)
119#endif
120 {
121 }
122
123 ConcurrentJSLocker(int) = delete;
124
125#if !defined(NDEBUG)
126private:
127 std::optional<DisallowGC> m_disallowGC;
128#endif
129};
