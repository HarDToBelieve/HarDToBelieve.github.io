33namespace JSC {
34
35class JSObject;
36
37// Convenience wrapper so you don't need to deal with CallData unless you are going to use it.
38JS_EXPORT_PRIVATE JSObject* construct(JSGlobalObject*, JSValue functionObject, const ArgList&, const char* errorMessage);
39JS_EXPORT_PRIVATE JSObject* construct(JSGlobalObject*, JSValue functionObject, JSValue newTarget, const ArgList&, const char* errorMessage);
40
41JS_EXPORT_PRIVATE JSObject* construct(JSGlobalObject*, JSValue constructor, const CallData&, const ArgList&, JSValue newTarget);
42
43ALWAYS_INLINE JSObject* construct(JSGlobalObject* globalObject, JSValue constructorObject, const CallData& callData, const ArgList& args)
44{
45 return construct(globalObject, constructorObject, callData, args, constructorObject);
46}
47
48JS_EXPORT_PRIVATE JSObject* profiledConstruct(JSGlobalObject*, ProfilingReason, JSValue constructor, const CallData&, const ArgList&, JSValue newTarget);
49
50ALWAYS_INLINE JSObject* profiledConstruct(JSGlobalObject* globalObject, ProfilingReason reason, JSValue constructorObject, const CallData& callData, const ArgList& args)
51{
52 return profiledConstruct(globalObject, reason, constructorObject, callData, args, constructorObject);
53}
