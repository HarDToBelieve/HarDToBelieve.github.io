29#pragma once
30
31#include "CallFrame.h"
32#include "DebuggerPrimitives.h"
33#include "ShadowChicken.h"
34#include "Strong.h"
35#include <wtf/NakedPtr.h>
36#include <wtf/RefCounted.h>
37#include <wtf/text/TextPosition.h>
38
39namespace JSC {
40
41class DebuggerScope;
42class Exception;
43
44class DebuggerCallFrame : public RefCounted<DebuggerCallFrame> {
45public:
46 enum Type { ProgramType, FunctionType };
47
48 static Ref<DebuggerCallFrame> create(VM&, CallFrame*);
49
50 JS_EXPORT_PRIVATE RefPtr<DebuggerCallFrame> callerFrame();
51 JSGlobalObject* globalObject();
52 JS_EXPORT_PRIVATE SourceID sourceID() const;
53
54 // line and column are in base 0 e.g. the first line is line 0.
55 int line() const { return m_position.m_line.zeroBasedInt(); }
56 int column() const { return m_position.m_column.zeroBasedInt(); }
57 JS_EXPORT_PRIVATE const TextPosition& position() const { return m_position; }
58
59 JS_EXPORT_PRIVATE JSGlobalObject* deprecatedVMEntryGlobalObject() const;
60 JS_EXPORT_PRIVATE DebuggerScope* scope();
61 JS_EXPORT_PRIVATE String functionName() const;
62 JS_EXPORT_PRIVATE Type type() const;
63 JS_EXPORT_PRIVATE JSValue thisValue(VM&) const;
64
65 JSValue evaluateWithScopeExtension(const String&, JSObject* scopeExtensionObject, NakedPtr<Exception>&);
66
67 bool isValid() const { return !!m_validMachineFrame || isTailDeleted(); }
68 JS_EXPORT_PRIVATE void invalidate();
69
70 // The following are only public for the Debugger's use only. They will be
71 // made private soon. Other clients should not use these.
72
73 JS_EXPORT_PRIVATE TextPosition currentPosition(VM&);
74 JS_EXPORT_PRIVATE static TextPosition positionForCallFrame(VM&, CallFrame*);
75 JS_EXPORT_PRIVATE static SourceID sourceIDForCallFrame(CallFrame*);
76
77 bool isTailDeleted() const { return m_shadowChickenFrame.isTailDeleted; }
78
79private:
80 DebuggerCallFrame(VM&, CallFrame*, const ShadowChicken::Frame&);
81
82 CallFrame* m_validMachineFrame;
83 RefPtr<DebuggerCallFrame> m_caller;
84 TextPosition m_position;
85 // The DebuggerPausedScope is responsible for calling invalidate() which,
86 // in turn, will clear this strong ref.
87 Strong<DebuggerScope> m_scope;
88 ShadowChicken::Frame m_shadowChickenFrame;
89};
90
91} // namespace JSC
