33class SourceProvider;
34class VM;

36// The order of the constants here matters as it is used for
37// sorting pause positions that have same offset.
38enum class DebuggerPausePositionType { Invalid, Enter, Pause, Leave };
39struct DebuggerPausePosition {
40 DebuggerPausePositionType type { DebuggerPausePositionType::Invalid };
41 JSTextPosition position;
42};
43
44class DebuggerPausePositions {
45public:
46 DebuggerPausePositions() { }
47 ~DebuggerPausePositions() { }
48
49 void appendPause(const JSTextPosition& position)
50 {
51 m_positions.append({ DebuggerPausePositionType::Pause, position });
52 }
53
54 void appendEntry(const JSTextPosition& position)
55 {
56 m_positions.append({ DebuggerPausePositionType::Enter, position });
57 }
58
59 void appendLeave(const JSTextPosition& position)
60 {
61 m_positions.append({ DebuggerPausePositionType::Leave, position });
62 }
63
64 std::optional<JSTextPosition> breakpointLocationForLineColumn(int line, int column);
65
66 void sort();
67
68private:
69 Vector<DebuggerPausePosition> m_positions;
70};
71
72
73struct DebuggerParseData {
74 DebuggerParseData() { }
75 ~DebuggerParseData() { }
76
77 DebuggerPausePositions pausePositions;
78};

80bool gatherDebuggerParseDataForSource(VM&, SourceProvider*, DebuggerParseData&);
