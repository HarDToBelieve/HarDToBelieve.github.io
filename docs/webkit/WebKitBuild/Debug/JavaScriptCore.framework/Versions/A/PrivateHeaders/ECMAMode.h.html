28namespace WTF {
29class PrintStream;
30};
31
32namespace JSC {
33
34struct ECMAMode {
35 static constexpr uint8_t StrictMode = 0;
36 static constexpr uint8_t SloppyMode = 1;
37
38public:
39 static constexpr ECMAMode fromByte(uint8_t byte) { return ECMAMode(byte); }
40 static constexpr ECMAMode fromBool(bool isStrict) { return isStrict ? strict() : sloppy(); }
41 static constexpr ECMAMode strict() { return ECMAMode(StrictMode); }
42 static constexpr ECMAMode sloppy() { return ECMAMode(SloppyMode); }
43
44 ALWAYS_INLINE bool isStrict() const { return m_value == StrictMode; }
45 ALWAYS_INLINE uint8_t value() const { return m_value; }
46
47 void dump(WTF::PrintStream&) const;
48
49private:
50 constexpr ECMAMode(uint8_t value)
51 : m_value(value)
52 {
53 ASSERT(m_value == StrictMode || m_value == SloppyMode, m_value);
54 }
55
56 uint8_t m_value;
57};
58
59} // namespace JSC
