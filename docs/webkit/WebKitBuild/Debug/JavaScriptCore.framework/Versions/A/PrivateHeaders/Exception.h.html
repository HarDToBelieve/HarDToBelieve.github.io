26#pragma once
27
28#include "JSDestructibleObject.h"
29#include "StackFrame.h"
30#include <wtf/Vector.h>
31
32namespace JSC {
33
34class Exception final : public JSCell {
35public:
36 using Base = JSCell;
37 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38 static constexpr bool needsDestruction = true;
39
40 template<typename CellType, SubspaceAccess mode>
41 static IsoSubspace* subspaceFor(VM& vm)
42 {
43 return &vm.exceptionSpace;
44 }
45
46 enum StackCaptureAction {
47 CaptureStack,
48 DoNotCaptureStack
49 };
50 JS_EXPORT_PRIVATE static Exception* create(VM&, JSValue thrownValue, StackCaptureAction = CaptureStack);
51
52 static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
53
54 DECLARE_VISIT_CHILDREN;
55
56 DECLARE_EXPORT_INFO;
57
58 static ptrdiff_t valueOffset()
59 {
60 return OBJECT_OFFSETOF(Exception, m_value);
61 }
62
63 JSValue value() const { return m_value.get(); }
64 const Vector<StackFrame>& stack() const { return m_stack; }
65
66 bool didNotifyInspectorOfThrow() const { return m_didNotifyInspectorOfThrow; }
67 void setDidNotifyInspectorOfThrow() { m_didNotifyInspectorOfThrow = true; }
68
69 ~Exception();
70
71private:
72 Exception(VM&);
73 void finishCreation(VM&, JSValue thrownValue, StackCaptureAction);
74 static void destroy(JSCell*);
75
76 WriteBarrier<Unknown> m_value;
77 Vector<StackFrame> m_stack;
78 bool m_didNotifyInspectorOfThrow { false };
79
80 friend class LLIntOffsetsExtractor;
81};
82
83} // namespace JSC
