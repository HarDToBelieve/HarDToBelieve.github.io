33namespace JSC {
34
35class CodeBlock;
36class LLIntOffsetsExtractor;
37
38class ExecutableToCodeBlockEdge final : public JSCell {
39public:
40 typedef JSCell Base;
41 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
42
43 template<typename CellType, SubspaceAccess>
44 static IsoSubspace* subspaceFor(VM& vm)
45 {
46 return &vm.executableToCodeBlockEdgeSpace;
47 }
48
49 static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
50
51 static ExecutableToCodeBlockEdge* create(VM&, CodeBlock*);
52
53 DECLARE_INFO;
54
55 CodeBlock* codeBlock() const { return m_codeBlock.get(); }
56
57 DECLARE_VISIT_CHILDREN;
58 DECLARE_VISIT_OUTPUT_CONSTRAINTS;
59 void finalizeUnconditionally(VM&);
60
61 static CodeBlock* unwrap(ExecutableToCodeBlockEdge* edge)
62 {
63 if (!edge)
64 return nullptr;
65 return edge->codeBlock();
66 }
67
68 static CodeBlock* deactivateAndUnwrap(ExecutableToCodeBlockEdge* edge);
69
70 static ExecutableToCodeBlockEdge* wrap(CodeBlock* codeBlock);
71
72 static ExecutableToCodeBlockEdge* wrapAndActivate(CodeBlock* codeBlock);
73
74private:
75 friend class LLIntOffsetsExtractor;
76
77 ExecutableToCodeBlockEdge(VM&, CodeBlock*);
78
79 void finishCreation(VM&);
80
81 void activate();
82 void deactivate();
83 bool isActive() const;
84
85 template<typename Visitor> void runConstraint(const ConcurrentJSLocker&, VM&, Visitor&);
86
87 WriteBarrier<CodeBlock> m_codeBlock;
88};

90} // namespace JSC
