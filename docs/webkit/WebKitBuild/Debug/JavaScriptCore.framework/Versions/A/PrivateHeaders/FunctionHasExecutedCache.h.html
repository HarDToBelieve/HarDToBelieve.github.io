26#pragma once
27
28#include <wtf/HashMethod.h>
29#include <wtf/StdUnorderedMap.h>
30#include <wtf/Vector.h>
31
32namespace JSC {
33
34class FunctionHasExecutedCache {
35public:
36 struct FunctionRange {
37 FunctionRange() {}
38 bool operator==(const FunctionRange& other) const
39 {
40 return m_start == other.m_start && m_end == other.m_end;
41 }
42 unsigned hash() const
43 {
44 return m_start * m_end;
45 }
46
47 unsigned m_start;
48 unsigned m_end;
49 };
50
51 bool hasExecutedAtOffset(intptr_t id, unsigned offset);
52 void insertUnexecutedRange(intptr_t id, unsigned start, unsigned end);
53 void removeUnexecutedRange(intptr_t id, unsigned start, unsigned end);
54 Vector<std::tuple<bool, unsigned, unsigned>> getFunctionRanges(intptr_t id);
55
56private:
57 using RangeMap = StdUnorderedMap<FunctionRange, bool, HashMethod<FunctionRange>>;
58 using SourceIDToRangeMap = StdUnorderedMap<intptr_t, RangeMap>;
59 SourceIDToRangeMap m_rangeMap;
60};
61
62} // namespace JSC
