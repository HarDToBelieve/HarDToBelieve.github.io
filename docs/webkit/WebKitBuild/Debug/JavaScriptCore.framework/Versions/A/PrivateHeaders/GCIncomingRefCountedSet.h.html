32// T = some subtype of GCIncomingRefCounted, must support a gcSizeEstimateInBytes()
33// method.
34template<typename T>
35class GCIncomingRefCountedSet {
36public:
37 GCIncomingRefCountedSet();
38
39 void lastChanceToFinalize();
40
41 // Returns true if the native object is new to this set.
42 bool addReference(JSCell*, T*);
43
44 void sweep(VM&);
45
46 size_t size() const { return m_bytes; };
47
48private:
49 Vector<T*> m_vector;
50 size_t m_bytes;
51};
