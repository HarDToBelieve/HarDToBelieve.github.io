26#pragma once
27
28#include "JSCJSValue.h"
29
30namespace JSC {
31
32typedef JSValue* HandleSlot;
33
34template<typename T> struct HandleTypes {
35 typedef T* ExternalType;
36 static ExternalType getFromSlot(HandleSlot slot) { return (slot && *slot) ? reinterpret_cast<ExternalType>(static_cast<void*>(slot->asCell())) : nullptr; }
37 static JSValue toJSValue(T* cell) { return reinterpret_cast<JSCell*>(cell); }
38 template<typename U> static void validateUpcast() { T* temp; temp = (U*)0; }
39};
40
41template<> struct HandleTypes<Unknown> {
42 typedef JSValue ExternalType;
43 static ExternalType getFromSlot(HandleSlot slot) { return slot ? *slot : JSValue(); }
44 static JSValue toJSValue(const JSValue& v) { return v; }
45 template<typename U> static void validateUpcast() { }
46};
47
48} // namespace JSC
