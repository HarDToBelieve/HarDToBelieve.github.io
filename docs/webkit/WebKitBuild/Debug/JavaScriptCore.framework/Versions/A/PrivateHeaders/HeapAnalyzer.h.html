27
28#include "RootMarkReason.h"
29#include <wtf/text/UniquedStringImpl.h>
30#include <wtf/text/WTFString.h>

32namespace JSC {
33
34class JSCell;
35
36class JS_EXPORT_PRIVATE HeapAnalyzer {
37public:
38 virtual ~HeapAnalyzer() = default;
39
40 // A root or marked cell.
41 virtual void analyzeNode(JSCell*) = 0;
42
43 // A reference from one cell to another.
44 virtual void analyzeEdge(JSCell* from, JSCell* to, RootMarkReason) = 0;
45 virtual void analyzePropertyNameEdge(JSCell* from, JSCell* to, UniquedStringImpl* propertyName) = 0;
46 virtual void analyzeVariableNameEdge(JSCell* from, JSCell* to, UniquedStringImpl* variableName) = 0;
47 virtual void analyzeIndexEdge(JSCell* from, JSCell* to, uint32_t index) = 0;
48
49 virtual void setOpaqueRootReachabilityReasonForCell(JSCell*, const char*) = 0;
50 virtual void setWrappedObjectForCell(JSCell*, void*) = 0;
51 virtual void setLabelForCell(JSCell*, const String&) = 0;
52};
53
54} // namespace JSC
