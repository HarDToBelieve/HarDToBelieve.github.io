26#pragma once
27
28#include "PropertyStorage.h"
29
30namespace JSC {
31
32class ArrayBuffer;
33class Butterfly;
34class LLIntOffsetsExtractor;
35class Structure;
36struct ArrayStorage;
37
38class IndexingHeader {
39public:
40 // Define the maximum storage vector length to be 2^32 / sizeof(JSValue) / 2 to ensure that
41 // there is no risk of overflow.
42 static constexpr unsigned maximumLength = 0x10000000;
43
44 static ptrdiff_t offsetOfIndexingHeader() { return -static_cast<ptrdiff_t>(sizeof(IndexingHeader)); }
45
46 static ptrdiff_t offsetOfArrayBuffer() { return OBJECT_OFFSETOF(IndexingHeader, u.typedArray.buffer); }
47 static ptrdiff_t offsetOfPublicLength() { return OBJECT_OFFSETOF(IndexingHeader, u.lengths.publicLength); }
48 static ptrdiff_t offsetOfVectorLength() { return OBJECT_OFFSETOF(IndexingHeader, u.lengths.vectorLength); }
49
50 IndexingHeader()
51 {
52 u.lengths.publicLength = 0;
53 u.lengths.vectorLength = 0;
54 }
55
56 uint32_t vectorLength() const { return u.lengths.vectorLength; }
57
58 void setVectorLength(uint32_t length)
59 {
60 RELEASE_ASSERT(length <= maximumLength);
61 u.lengths.vectorLength = length;
62 }
63
64 uint32_t publicLength() const { return u.lengths.publicLength; }
65 void setPublicLength(uint32_t auxWord) { u.lengths.publicLength = auxWord; }
66
67 ArrayBuffer* arrayBuffer() { return u.typedArray.buffer; }
68 void setArrayBuffer(ArrayBuffer* buffer) { u.typedArray.buffer = buffer; }
69
70 static IndexingHeader* from(Butterfly* butterfly)
71 {
72 return reinterpret_cast<IndexingHeader*>(butterfly) - 1;
73 }
74
75 static const IndexingHeader* from(const Butterfly* butterfly)
76 {
77 return reinterpret_cast<const IndexingHeader*>(butterfly) - 1;
78 }
79
80 static IndexingHeader* from(ArrayStorage* arrayStorage)
81 {
82 return const_cast<IndexingHeader*>(from(const_cast<const ArrayStorage*>(arrayStorage)));
83 }
84
85 static const IndexingHeader* from(const ArrayStorage* arrayStorage)
86 {
87 return reinterpret_cast<const IndexingHeader*>(arrayStorage) - 1;
88 }
89
90 static IndexingHeader* fromEndOf(PropertyStorage propertyStorage)
91 {
92 return reinterpret_cast<IndexingHeader*>(propertyStorage);
93 }
94
95 PropertyStorage propertyStorage()
96 {
97 return reinterpret_cast_ptr<PropertyStorage>(this);
98 }
99
100 ConstPropertyStorage propertyStorage() const
101 {
102 return reinterpret_cast_ptr<ConstPropertyStorage>(this);
103 }
104
105 ArrayStorage* arrayStorage()
106 {
107 return reinterpret_cast<ArrayStorage*>(this + 1);
108 }
109
110 Butterfly* butterfly()
111 {
112 return reinterpret_cast<Butterfly*>(this + 1);
113 }
114
115 // These methods are not standalone in the sense that they cannot be
116 // used on a copy of the IndexingHeader.
117 size_t preCapacity(Structure*);
118 size_t indexingPayloadSizeInBytes(Structure*);
119
120private:
121 friend class LLIntOffsetsExtractor;
122
123 union {
124 struct {
125 // FIXME: vectorLength should be least significant, so that it's really hard to craft a pointer by
126 // mucking with the butterfly.
127 // https://bugs.webkit.org/show_bug.cgi?id=174927
128 uint32_t publicLength; // The meaning of this field depends on the array type, but for all JSArrays we rely on this being the publicly visible length (array.length).
129 uint32_t vectorLength; // The length of the indexed property storage. The actual size of the storage depends on this, and the type.
130 } lengths;
131
132 struct {
133 ArrayBuffer* buffer;
134 } typedArray;
135 } u;
136};
137
138} // namespace JSC
