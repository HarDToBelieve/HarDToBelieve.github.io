34inline size_t IndexingHeader::preCapacity(Structure* structure)
35{
36 if (LIKELY(!hasAnyArrayStorage(structure->indexingType())))
37 return 0;
38
39 return arrayStorage()->m_indexBias;
40}
41
42inline size_t IndexingHeader::indexingPayloadSizeInBytes(Structure* structure)
43{
44 switch (structure->indexingType()) {
45 case ALL_UNDECIDED_INDEXING_TYPES:
46 case ALL_INT32_INDEXING_TYPES:
47 case ALL_DOUBLE_INDEXING_TYPES:
48 case ALL_CONTIGUOUS_INDEXING_TYPES:
49 return vectorLength() * sizeof(EncodedJSValue);
50
51 case ALL_ARRAY_STORAGE_INDEXING_TYPES:
52 return ArrayStorage::sizeFor(arrayStorage()->vectorLength());
53
54 default:
55 ASSERT(!hasIndexedProperties(structure->indexingType()));
56 return 0;
57 }
58}
