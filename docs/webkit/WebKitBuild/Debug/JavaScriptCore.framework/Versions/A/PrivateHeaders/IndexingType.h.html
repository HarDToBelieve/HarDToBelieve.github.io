34/*
35 Structure of the IndexingType
36 =============================
37 Conceptually, the IndexingTypeAndMisc looks like this:
38
39 struct IndexingTypeAndMisc {
40 struct IndexingModeIncludingHistory {
41 struct IndexingMode {
42 struct IndexingType {
43 uint8_t isArray:1; // bit 0
44 uint8_t shape:3; // bit 1 - 3
45 };
46 uint8_t copyOnWrite:1; // bit 4
47 };
48 uint8_t mayHaveIndexedAccessors:1; // bit 5
49 };
50 uint8_t cellLockBits:2; // bit 6 - 7
51 };
52
53 The shape values (e.g. Int32Shape, ContiguousShape, etc) are an enumeration of
54 various shapes (though not necessarily sequential in terms of their values).
55 Hence, shape values are not bitwise exclusive with respect to each other.
56
57 It's also common to refer to shape + copyOnWrite as IndexingShapeWithWritability.
58*/
59
60typedef uint8_t IndexingType;
61
62// Flags for testing the presence of capabilities.
63static const IndexingType IsArray = 0x01;
64
65// The shape of the indexed property storage.
66static const IndexingType NoIndexingShape = 0x00;
67static const IndexingType UndecidedShape = 0x02; // Only useful for arrays.
68static const IndexingType Int32Shape = 0x04;
69static const IndexingType DoubleShape = 0x06;
70static const IndexingType ContiguousShape = 0x08;
71static const IndexingType ArrayStorageShape = 0x0A;
72static const IndexingType SlowPutArrayStorageShape = 0x0C;
73
74static const IndexingType IndexingShapeMask = 0x0E;
75static const IndexingType IndexingShapeShift = 1;
76static const IndexingType NumberOfIndexingShapes = 7;
77static const IndexingType IndexingTypeMask = IndexingShapeMask | IsArray;
78
79// Whether or not the butterfly is copy on write. If it is copy on write then the butterfly is actually a JSImmutableButterfly. This should only ever be set if there are no named properties.
80static const IndexingType CopyOnWrite = 0x10;
81static const IndexingType IndexingShapeAndWritabilityMask = CopyOnWrite | IndexingShapeMask;
82static const IndexingType IndexingModeMask = CopyOnWrite | IndexingTypeMask;
83static const IndexingType NumberOfCopyOnWriteIndexingModes = 3; // We only have copy on write for int32, double, and contiguous shapes.
84static const IndexingType NumberOfArrayIndexingModes = NumberOfIndexingShapes + NumberOfCopyOnWriteIndexingModes;
85
86// Additional flags for tracking the history of the type. These are usually
87// masked off unless you ask for them directly.
88// FIXME: We only seem to use MayHaveIndexedAccessors for checking if the prototype chain may intercept indexed properties. We should probably be able to compute that same information via (indexingShape != SlowPutArrayStorageShape || classInfo().getOwnPropertySlotByIndex != JSObject::getOwnPropertySlotByIndex || globalObject().isHavingABadTime()) since we will already get the structure in order to do the prototype chain walk.
89// See: https://bugs.webkit.org/show_bug.cgi?id=209757
90static const IndexingType MayHaveIndexedAccessors = 0x20;
91
92// The IndexingType field of JSCells is stolen for locks and remembering if the object has been a
93// prototype.
94static const IndexingType IndexingTypeLockIsHeld = 0x40;
95static const IndexingType IndexingTypeLockHasParked = 0x80;
96
97// List of acceptable array types.
98static const IndexingType NonArray = 0x0;
99static const IndexingType NonArrayWithInt32 = Int32Shape;
100static const IndexingType NonArrayWithDouble = DoubleShape;
101static const IndexingType NonArrayWithContiguous = ContiguousShape;
102static const IndexingType NonArrayWithArrayStorage = ArrayStorageShape;
103static const IndexingType NonArrayWithSlowPutArrayStorage = SlowPutArrayStorageShape;
104static const IndexingType ArrayClass = IsArray; // I'd want to call this "Array" but this would lead to disastrous namespace pollution.
105static const IndexingType ArrayWithUndecided = IsArray | UndecidedShape;
106static const IndexingType ArrayWithInt32 = IsArray | Int32Shape;
107static const IndexingType ArrayWithDouble = IsArray | DoubleShape;
108static const IndexingType ArrayWithContiguous = IsArray | ContiguousShape;
109static const IndexingType ArrayWithArrayStorage = IsArray | ArrayStorageShape;
110static const IndexingType ArrayWithSlowPutArrayStorage = IsArray | SlowPutArrayStorageShape;
111static const IndexingType CopyOnWriteArrayWithInt32 = IsArray | Int32Shape | CopyOnWrite;
112static const IndexingType CopyOnWriteArrayWithDouble = IsArray | DoubleShape | CopyOnWrite;
113static const IndexingType CopyOnWriteArrayWithContiguous = IsArray | ContiguousShape | CopyOnWrite;
114
115#define ALL_BLANK_INDEXING_TYPES \
116 NonArray: \
117 case ArrayClass
118
119#define ALL_UNDECIDED_INDEXING_TYPES \
120 ArrayWithUndecided
121
122#define ALL_WRITABLE_INT32_INDEXING_TYPES \
123 NonArrayWithInt32: \
124 case ArrayWithInt32
125
126#define ALL_INT32_INDEXING_TYPES \
127 ALL_WRITABLE_INT32_INDEXING_TYPES: \
128 case CopyOnWriteArrayWithInt32
129
130#define ALL_WRITABLE_DOUBLE_INDEXING_TYPES \
131 NonArrayWithDouble: \
132 case ArrayWithDouble \
133
134#define ALL_DOUBLE_INDEXING_TYPES \
135 ALL_WRITABLE_DOUBLE_INDEXING_TYPES: \
136 case CopyOnWriteArrayWithDouble
137
138#define ALL_WRITABLE_CONTIGUOUS_INDEXING_TYPES \
139 NonArrayWithContiguous: \
140 case ArrayWithContiguous \
141
142#define ALL_CONTIGUOUS_INDEXING_TYPES \
143 ALL_WRITABLE_CONTIGUOUS_INDEXING_TYPES: \
144 case CopyOnWriteArrayWithContiguous
145
146#define ARRAY_WITH_ARRAY_STORAGE_INDEXING_TYPES \
147 ArrayWithArrayStorage: \
148 case ArrayWithSlowPutArrayStorage
149
150#define ALL_ARRAY_STORAGE_INDEXING_TYPES \
151 NonArrayWithArrayStorage: \
152 case NonArrayWithSlowPutArrayStorage: \
153 case ARRAY_WITH_ARRAY_STORAGE_INDEXING_TYPES

155inline bool hasIndexedProperties(IndexingType indexingType)
156{
157 return (indexingType & IndexingShapeMask) != NoIndexingShape;
158}
159
160inline bool hasUndecided(IndexingType indexingType)
161{
162 return (indexingType & IndexingShapeMask) == UndecidedShape;
163}
164
165inline bool hasInt32(IndexingType indexingType)
166{
167 return (indexingType & IndexingShapeMask) == Int32Shape;
168}
169
170inline bool hasDouble(IndexingType indexingType)
171{
172 return (indexingType & IndexingShapeMask) == DoubleShape;
173}
174
175inline bool hasContiguous(IndexingType indexingType)
176{
177 return (indexingType & IndexingShapeMask) == ContiguousShape;
178}
179
180inline bool hasArrayStorage(IndexingType indexingType)
181{
182 return (indexingType & IndexingShapeMask) == ArrayStorageShape;
183}
184
185inline bool hasAnyArrayStorage(IndexingType indexingType)
186{
187 return static_cast<uint8_t>(indexingType & IndexingShapeMask) >= ArrayStorageShape;
188}
189
190inline bool hasSlowPutArrayStorage(IndexingType indexingType)
191{
192 return (indexingType & IndexingShapeMask) == SlowPutArrayStorageShape;
193}
194
195inline bool shouldUseSlowPut(IndexingType indexingType)
196{
197 return hasSlowPutArrayStorage(indexingType);
198}
199
200constexpr bool isCopyOnWrite(IndexingType indexingMode)
201{
202 return indexingMode & CopyOnWrite;
203}
204
205inline unsigned arrayIndexFromIndexingType(IndexingType indexingType)
206{
207 if (isCopyOnWrite(indexingType))
208 return ((indexingType & IndexingShapeMask) - UndecidedShape + SlowPutArrayStorageShape) >> IndexingShapeShift;
209 return (indexingType & IndexingShapeMask) >> IndexingShapeShift;
210}
211
212inline IndexingType indexingTypeForValue(JSValue value)
213{
214 if (value.isInt32())
215 return Int32Shape;
216
217 if (value.isNumber() && value.asNumber() == value.asNumber())
218 return DoubleShape;
219
220 return ContiguousShape;
221}
