31namespace Inspector {
32
33// FIXME: Add DedicatedWorker Inspector Targets
34// FIXME: Add ServiceWorker Inspector Targets
35enum class InspectorTargetType : uint8_t {
36 Page,
37 DedicatedWorker,
38 ServiceWorker,
39};
40
41class JS_EXPORT_PRIVATE InspectorTarget {
42public:
43 virtual ~InspectorTarget() = default;
44
45 // State.
46 virtual String identifier() const = 0;
47 virtual InspectorTargetType type() const = 0;
48
49 virtual bool isProvisional() const { return false; }
50 bool isPaused() const { return m_isPaused; }
51 void pause();
52 void resume();
53 void setResumeCallback(WTF::Function<void()>&&);
54
55 // Connection management.
56 virtual void connect(FrontendChannel::ConnectionType) = 0;
57 virtual void disconnect() = 0;
58 virtual void sendMessageToTargetBackend(const String&) = 0;
59
60private:
61 WTF::Function<void()> m_resumeCallback;
62 bool m_isPaused { false };
63};
64
65} // namespace Inspector

74 Inspector::InspectorTargetType::ServiceWorker
75 >;
76};
