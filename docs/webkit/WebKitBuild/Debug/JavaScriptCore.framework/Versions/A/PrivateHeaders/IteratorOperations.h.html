61template<typename CallBackType>
62void forEachInIterable(JSGlobalObject* globalObject, JSValue iterable, const CallBackType& callback)
63{
64 auto& vm = getVM(globalObject);
65 auto scope = DECLARE_THROW_SCOPE(vm);
66
67 if (getIterationMode(vm, globalObject, iterable) == IterationMode::FastArray) {
68 auto* array = jsCast<JSArray*>(iterable);
69 for (unsigned index = 0; index < array->length(); ++index) {
70 JSValue nextValue = array->getIndex(globalObject, index);
71 RETURN_IF_EXCEPTION(scope, void());
72 callback(vm, globalObject, nextValue);
73 if (UNLIKELY(scope.exception())) {
74 scope.release();
75 JSArrayIterator* iterator = JSArrayIterator::create(vm, globalObject->arrayIteratorStructure(), array, IterationKind::Values);
76 iterator->internalField(JSArrayIterator::Field::Index).setWithoutWriteBarrier(jsNumber(index + 1));
77 iteratorClose(globalObject, iterator);
78 return;
79 }
80 }
81 return;
82 }
83
84 IterationRecord iterationRecord = iteratorForIterable(globalObject, iterable);
85 RETURN_IF_EXCEPTION(scope, void());
86 while (true) {
87 JSValue next = iteratorStep(globalObject, iterationRecord);
88 if (UNLIKELY(scope.exception()) || next.isFalse())
89 return;
90
91 JSValue nextValue = iteratorValue(globalObject, next);
92 RETURN_IF_EXCEPTION(scope, void());
93
94 callback(vm, globalObject, nextValue);
95 if (UNLIKELY(scope.exception())) {
96 scope.release();
97 iteratorClose(globalObject, iterationRecord.iterator);
98 return;
99 }
100 }
101}

103template<typename CallBackType>
104void forEachInIterable(JSGlobalObject& globalObject, JSObject* iterable, JSValue iteratorMethod, const CallBackType& callback)
105{
106 auto& vm = getVM(&globalObject);
107 auto scope = DECLARE_THROW_SCOPE(vm);
108
109 if (getIterationMode(vm, &globalObject, iterable, iteratorMethod) == IterationMode::FastArray) {
110 auto* array = jsCast<JSArray*>(iterable);
111 for (unsigned index = 0; index < array->length(); ++index) {
112 JSValue nextValue = array->getIndex(&globalObject, index);
113 RETURN_IF_EXCEPTION(scope, void());
114 callback(vm, globalObject, nextValue);
115 if (UNLIKELY(scope.exception())) {
116 scope.release();
117 JSArrayIterator* iterator = JSArrayIterator::create(vm, globalObject.arrayIteratorStructure(), array, IterationKind::Values);
118 iterator->internalField(JSArrayIterator::Field::Index).setWithoutWriteBarrier(jsNumber(index + 1));
119 iteratorClose(&globalObject, iterator);
120 return;
121 }
122 }
123 return;
124 }
125
126 auto iterationRecord = iteratorForIterable(&globalObject, iterable, iteratorMethod);
127 RETURN_IF_EXCEPTION(scope, void());
128 while (true) {
129 JSValue next = iteratorStep(&globalObject, iterationRecord);
130 if (UNLIKELY(scope.exception()) || next.isFalse())
131 return;
132
133 JSValue nextValue = iteratorValue(&globalObject, next);
134 RETURN_IF_EXCEPTION(scope, void());
135
136 callback(vm, globalObject, nextValue);
137 if (UNLIKELY(scope.exception())) {
138 scope.release();
139 iteratorClose(&globalObject, iterationRecord.iterator);
140 return;
141 }
142 }
143}
