53enum class JITType : uint8_t {
54 None = 0b000,
55 HostCallThunk = 0b001,
56 InterpreterThunk = 0b010,
57 BaselineJIT = 0b011,
58 DFGJIT = 0b100,
59 FTLJIT = 0b101,
60};
61static constexpr unsigned widthOfJITType = 3;
62static_assert(WTF::getMSBSetConstexpr(static_cast<std::underlying_type_t<JITType>>(JITType::FTLJIT)) + 1 == widthOfJITType);
63
64class JITCode : public ThreadSafeRefCounted<JITCode> {
65public:
66 template<PtrTag tag> using CodePtr = MacroAssemblerCodePtr<tag>;
67 template<PtrTag tag> using CodeRef = MacroAssemblerCodeRef<tag>;
68
69 static const char* typeName(JITType);
70
71 static JITType bottomTierJIT()
72 {
73 return JITType::BaselineJIT;
74 }
75
76 static JITType topTierJIT()
77 {
78 return JITType::FTLJIT;
79 }
80
81 static JITType nextTierJIT(JITType jitType)
82 {
83 switch (jitType) {
84 case JITType::BaselineJIT:
85 return JITType::DFGJIT;
86 case JITType::DFGJIT:
87 return JITType::FTLJIT;
88 default:
89 RELEASE_ASSERT_NOT_REACHED();
90 return JITType::None;
91 }
92 }
93
94 static bool isExecutableScript(JITType jitType)
95 {
96 switch (jitType) {
97 case JITType::None:
98 case JITType::HostCallThunk:
99 return false;
100 default:
101 return true;
102 }
103 }
104
105 static bool couldBeInterpreted(JITType jitType)
106 {
107 switch (jitType) {
108 case JITType::InterpreterThunk:
109 case JITType::BaselineJIT:
110 return true;
111 default:
112 return false;
113 }
114 }
115
116 static bool isJIT(JITType jitType)
117 {
118 switch (jitType) {
119 case JITType::BaselineJIT:
120 case JITType::DFGJIT:
121 case JITType::FTLJIT:
122 return true;
123 default:
124 return false;
125 }
126 }
127
128 static bool isLowerTier(JITType expectedLower, JITType expectedHigher)
129 {
130 RELEASE_ASSERT(isExecutableScript(expectedLower));
131 RELEASE_ASSERT(isExecutableScript(expectedHigher));
132 return expectedLower < expectedHigher;
133 }
134
135 static bool isHigherTier(JITType expectedHigher, JITType expectedLower)
136 {
137 return isLowerTier(expectedLower, expectedHigher);
138 }
139
140 static bool isLowerOrSameTier(JITType expectedLower, JITType expectedHigher)
141 {
142 return !isHigherTier(expectedLower, expectedHigher);
143 }
144
145 static bool isHigherOrSameTier(JITType expectedHigher, JITType expectedLower)
146 {
147 return isLowerOrSameTier(expectedLower, expectedHigher);
148 }
149
150 static bool isOptimizingJIT(JITType jitType)
151 {
152 return jitType == JITType::DFGJIT || jitType == JITType::FTLJIT;
153 }
154
155 static bool isBaselineCode(JITType jitType)
156 {
157 return jitType == JITType::InterpreterThunk || jitType == JITType::BaselineJIT;
158 }
159
160 static bool useDataIC(JITType jitType)
161 {
162 if (!Options::useDataIC())
163 return false;
164 if (JITCode::isBaselineCode(jitType))
165 return true;
166 return Options::useDataICInOptimizingJIT();
167 }
168
169 virtual const DOMJIT::Signature* signature() const { return nullptr; }
170
171 enum class ShareAttribute : uint8_t {
172 NotShared,
173 Shared
174 };
175
176protected:
177 JITCode(JITType, JITCode::ShareAttribute = JITCode::ShareAttribute::NotShared);
178
179public:
180 virtual ~JITCode();
181
182 JITType jitType() const
183 {
184 return m_jitType;
185 }
186
187 template<typename PointerType>
188 static JITType jitTypeFor(PointerType jitCode)
189 {
190 if (!jitCode)
191 return JITType::None;
192 return jitCode->jitType();
193 }
194
195 virtual CodePtr<JSEntryPtrTag> addressForCall(ArityCheckMode) = 0;
196 virtual void* executableAddressAtOffset(size_t offset) = 0;
197 void* executableAddress() { return executableAddressAtOffset(0); }
198 virtual void* dataAddressAtOffset(size_t offset) = 0;
199 virtual unsigned offsetOf(void* pointerIntoCode) = 0;
200
201 virtual DFG::CommonData* dfgCommon();
202 virtual DFG::JITCode* dfg();
203 virtual FTL::JITCode* ftl();
204 virtual FTL::ForOSREntryJITCode* ftlForOSREntry();
205 virtual void shrinkToFit(const ConcurrentJSLocker&);
206
207 virtual void validateReferences(const TrackedReferences&);
208
209 JSValue execute(VM*, ProtoCallFrame*);
210
211 void* start() { return dataAddressAtOffset(0); }
212 virtual size_t size() = 0;
213 void* end() { return reinterpret_cast<void*>(reinterpret_cast<uintptr_t>(start()) + size()); }
214
215 virtual bool contains(void*) = 0;
216
217#if ENABLE(JIT)
218 virtual RegisterSet liveRegistersToPreserveAtExceptionHandlingCallSite(CodeBlock*, CallSiteIndex);
219 virtual std::optional<CodeOrigin> findPC(CodeBlock*, void* pc) { UNUSED_PARAM(pc); return std::nullopt; }
220#endif
221
222 Intrinsic intrinsic() { return m_intrinsic; }
223
224 bool isShared() const { return m_shareAttribute == ShareAttribute::Shared; }
225
226private:
227 JITType m_jitType;
228 ShareAttribute m_shareAttribute;
229protected:
230 Intrinsic m_intrinsic { NoIntrinsic }; // Effective only in NativeExecutable.
231};
