26#pragma once
27
28#if ENABLE(JIT)
29
30#include "MacroAssemblerCodeRef.h"
31#include <wtf/FastMalloc.h>
32#include <wtf/Noncopyable.h>
33
34namespace JSC {
35
36class VM;
37
38class OpaqueByproducts;
39
40// This class is a way to keep the result of a compilation alive and runnable.
41
42class Compilation {
43 WTF_MAKE_NONCOPYABLE(Compilation);
44 WTF_MAKE_FAST_ALLOCATED;
45
46public:
47 JS_EXPORT_PRIVATE Compilation(MacroAssemblerCodeRef<JITCompilationPtrTag>, std::unique_ptr<OpaqueByproducts>);
48 JS_EXPORT_PRIVATE Compilation(Compilation&&);
49 JS_EXPORT_PRIVATE ~Compilation();
50
51 MacroAssemblerCodePtr<JITCompilationPtrTag> code() const { return m_codeRef.code(); }
52 MacroAssemblerCodeRef<JITCompilationPtrTag> codeRef() const { return m_codeRef; }
53
54 CString disassembly() const { return m_codeRef.disassembly(); }
55
56private:
57 MacroAssemblerCodeRef<JITCompilationPtrTag> m_codeRef;
58 std::unique_ptr<OpaqueByproducts> m_byproducts;
59};
60
61} // namespace JSC
62
63#endif // ENABLE(JIT)
