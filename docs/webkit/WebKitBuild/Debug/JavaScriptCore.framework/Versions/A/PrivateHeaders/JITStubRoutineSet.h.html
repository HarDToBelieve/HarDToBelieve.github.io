28#include "JITStubRoutine.h"

32#include <wtf/Vector.h>

34namespace JSC {
35
36class GCAwareJITStubRoutine;
37
38#if ENABLE(JIT)
39
40class JITStubRoutineSet {
41 WTF_MAKE_NONCOPYABLE(JITStubRoutineSet);
42 WTF_MAKE_FAST_ALLOCATED;
43
44public:
45 JITStubRoutineSet();
46 ~JITStubRoutineSet();
47
48 void add(GCAwareJITStubRoutine*);
49
50 void clearMarks();
51
52 void mark(void* candidateAddress)
53 {
54 uintptr_t address = removeCodePtrTag<uintptr_t>(candidateAddress);
55 if (!m_range.contains(address))
56 return;
57 markSlow(address);
58 }
59
60 void prepareForConservativeScan();
61
62 void deleteUnmarkedJettisonedStubRoutines(Seconds timeSlice);
63
64 template<typename Visitor> void traceMarkedStubRoutines(Visitor&);
65
66 static bool mayHaveRoutinesToDelete() { return s_mayHaveRoutinesToDelete; }
67 static void notifyHaveRoutinesToDelete() { s_mayHaveRoutinesToDelete = true; }
68
69private:
70 void markSlow(uintptr_t address);
71
72 struct Routine {
73 uintptr_t startAddress;
74 GCAwareJITStubRoutine* routine;
75 };
76 Vector<Routine> m_routines;
77 WTF::Range<uintptr_t> m_range { 0, 0 };
78
79 static bool s_mayHaveRoutinesToDelete;
80};
81
82#else // !ENABLE(JIT)
83
84class JITStubRoutineSet {
85 WTF_MAKE_NONCOPYABLE(JITStubRoutineSet);
86 WTF_MAKE_FAST_ALLOCATED;
87
88public:
89 JITStubRoutineSet() { }
90 ~JITStubRoutineSet() { }
91
92 void add(GCAwareJITStubRoutine*) { }
93 void clearMarks() { }
94 void mark(void*) { }
95 void prepareForConservativeScan() { }
96 void deleteUnmarkedJettisonedStubRoutines(Seconds) { }
97 template<typename Visitor> void traceMarkedStubRoutines(Visitor&) { }
98
99 static bool mayHaveRoutinesToDelete() { return false; }
100 static void notifyHaveRoutinesToDelete() { }
101};
102
103#endif // !ENABLE(JIT)
104
105} // namespace JSC
