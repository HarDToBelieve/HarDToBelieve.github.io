23#pragma once
24
25#include "CallFrame.h"
26#include "JSCJSValue.h"
27#include "JSCast.h"
28#include "Structure.h"
29
30namespace JSC {
31
32class JSAPIValueWrapper final : public JSCell {
33 friend JSValue jsAPIValueWrapper(JSGlobalObject*, JSValue);
34public:
35 using Base = JSCell;
36 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
37
38 template<typename CellType, SubspaceAccess mode>
39 static IsoSubspace* subspaceFor(VM& vm)
40 {
41 return vm.apiValueWrapperSpace<mode>();
42 }
43
44 JSValue value() const { return m_value.get(); }
45
46 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
47 {
48 return Structure::create(vm, globalObject, prototype, TypeInfo(APIValueWrapperType, StructureFlags), info());
49 }
50
51 DECLARE_EXPORT_INFO;
52
53 static JSAPIValueWrapper* create(VM& vm, JSValue value)
54 {
55 JSAPIValueWrapper* wrapper = new (NotNull, allocateCell<JSAPIValueWrapper>(vm.heap)) JSAPIValueWrapper(vm);
56 wrapper->finishCreation(vm, value);
57 return wrapper;
58 }
59
60private:
61 void finishCreation(VM& vm, JSValue value)
62 {
63 Base::finishCreation(vm);
64 m_value.set(vm, this, value);
65 ASSERT(!value.isCell());
66 }
67
68 JSAPIValueWrapper(VM& vm)
69 : JSCell(vm, vm.apiWrapperStructure.get())
70 {
71 }
72
73 WriteBarrier<Unknown> m_value;
74};
75
76} // namespace JSC
