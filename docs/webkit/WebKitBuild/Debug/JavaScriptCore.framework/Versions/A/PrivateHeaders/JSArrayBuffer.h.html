26#pragma once
27
28#include "ArrayBuffer.h"
29#include "JSObject.h"

31namespace JSC {
32
33class JSArrayBuffer final : public JSNonFinalObject {
34public:
35 using Base = JSNonFinalObject;
36 static constexpr unsigned StructureFlags = Base::StructureFlags;
37
38 template<typename CellType, SubspaceAccess mode>
39 static IsoSubspace* subspaceFor(VM& vm)
40 {
41 return vm.arrayBufferSpace<mode>();
42 }
43
44 // This function will register the new wrapper with the vm's TypedArrayController.
45 JS_EXPORT_PRIVATE static JSArrayBuffer* create(VM&, Structure*, RefPtr<ArrayBuffer>&&);
46
47 ArrayBuffer* impl() const { return m_impl; }
48
49 static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
50
51 JS_EXPORT_PRIVATE bool isShared() const;
52 ArrayBufferSharingMode sharingMode() const;
53
54 DECLARE_EXPORT_INFO;
55
56 // This is the default DOM unwrapping. It calls toUnsharedArrayBuffer().
57 static ArrayBuffer* toWrapped(VM&, JSValue);
58 static ArrayBuffer* toWrappedAllowShared(VM&, JSValue);
59
60private:
61 JSArrayBuffer(VM&, Structure*, RefPtr<ArrayBuffer>&&);
62 void finishCreation(VM&, JSGlobalObject*);
63
64 static size_t estimatedSize(JSCell*, VM&);
65
66 ArrayBuffer* m_impl;
67};
68
69inline ArrayBuffer* toPossiblySharedArrayBuffer(VM& vm, JSValue value)
70{
71 JSArrayBuffer* wrapper = jsDynamicCast<JSArrayBuffer*>(vm, value);
72 if (!wrapper)
73 return nullptr;
74 return wrapper->impl();
75}
76
77inline ArrayBuffer* toUnsharedArrayBuffer(VM& vm, JSValue value)
78{
79 ArrayBuffer* result = toPossiblySharedArrayBuffer(vm, value);
80 if (!result || result->isShared())
81 return nullptr;
82 return result;
83}
84
85inline ArrayBuffer* JSArrayBuffer::toWrapped(VM& vm, JSValue value)
86{
87 return toUnsharedArrayBuffer(vm, value);
88}
89
90inline ArrayBuffer* JSArrayBuffer::toWrappedAllowShared(VM& vm, JSValue value)
91{
92 return toPossiblySharedArrayBuffer(vm, value);
93}
94
95} // namespace JSC
