26#pragma once
27
28#include "JSInternalFieldObjectImpl.h"
29
30namespace JSC {
31
32class JSArrayIterator final : public JSInternalFieldObjectImpl<3> {
33public:
34 using Base = JSInternalFieldObjectImpl<3>;
35
36 enum class Field : uint8_t {
37 Index = 0,
38 IteratedObject,
39 Kind,
40 };
41 static_assert(numberOfInternalFields == 3);
42
43 static size_t allocationSize(Checked<size_t> inlineCapacity)
44 {
45 ASSERT_UNUSED(inlineCapacity, inlineCapacity == 0U);
46 return sizeof(JSArrayIterator);
47 }
48
49 template<typename CellType, SubspaceAccess mode>
50 static IsoSubspace* subspaceFor(VM& vm)
51 {
52 return vm.arrayIteratorSpace<mode>();
53 }
54
55 static std::array<JSValue, numberOfInternalFields> initialValues()
56 {
57 return { {
58 jsNumber(0),
59 jsNull(),
60 jsNumber(0),
61 } };
62 }
63
64 static constexpr int64_t doneIndex = -1;
65 const WriteBarrier<Unknown>& internalField(Field field) const { return Base::internalField(static_cast<uint32_t>(field)); }
66 WriteBarrier<Unknown>& internalField(Field field) { return Base::internalField(static_cast<uint32_t>(field)); }
67
68 IterationKind kind() const { return static_cast<IterationKind>(internalField(Field::Kind).get().asUInt32AsAnyInt()); }
69 JSObject* iteratedObject() const { return jsCast<JSObject*>(internalField(Field::IteratedObject).get()); }
70
71 JS_EXPORT_PRIVATE static JSArrayIterator* create(VM&, Structure*, JSObject* iteratedObject, JSValue kind);
72 static JSArrayIterator* create(VM& vm, Structure* structure, JSObject* iteratedObject, IterationKind kind)
73 {
74 return create(vm, structure, iteratedObject, jsNumber(static_cast<unsigned>(kind)));
75 }
76 static JSArrayIterator* createWithInitialValues(VM&, Structure*);
77 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
78
79 DECLARE_INFO;
80
81 DECLARE_VISIT_CHILDREN;
82
83private:
84 JSArrayIterator(VM&, Structure*);
85 void finishCreation(VM&);
86};
87
88} // namespace JSC
