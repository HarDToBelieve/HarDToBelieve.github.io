375 static JSValue leftShift(JSGlobalObject*, JSBigInt* x, JSBigInt* y);
376#if USE(BIGINT32)
377 static JSValue leftShift(JSGlobalObject*, JSBigInt* x, int32_t y);
378 static JSValue leftShift(JSGlobalObject*, int32_t x, JSBigInt* y);
379private:
380 static JSValue leftShiftSlow(JSGlobalObject*, int32_t x, int32_t y);
381public:
382 static JSValue leftShift(JSGlobalObject* globalObject, int32_t x, int32_t y)
383 {
384 if (y < 0) {
385 // Shifts one less than requested, but doesn't matter since lhs is int32
386 return signedRightShift(globalObject, x, y == INT32_MIN ? INT32_MAX : -y);
387 }
388
389 // Do some checks to detect overflow of left-shift. But this is much cheaper compared to allocating two JSBigInt and perform shift operations in JSBigInt.
390 if (!x)
391 return jsBigInt32(0);
392 if (y < 32)
393 return asInt32OrHeapCell(globalObject, static_cast<int64_t>(x) << y);
394 return leftShiftSlow(globalObject, x, y);
395 }
396#endif
397
398 static JSValue signedRightShift(JSGlobalObject*, JSBigInt* x, JSBigInt* y);
399#if USE(BIGINT32)
400 static JSValue signedRightShift(JSGlobalObject*, JSBigInt* x, int32_t y);
401 static JSValue signedRightShift(JSGlobalObject*, int32_t x, JSBigInt* y);
402 static JSValue signedRightShift(JSGlobalObject* globalObject, int32_t x, int32_t y)
403 {
404 if (y < 0) {
405 // Shifts one less than requested, but doesn't matter since lhs is int32
406 return leftShift(globalObject, x, y == INT32_MIN ? INT32_MAX : -y);
407 }
408
409 return jsBigInt32(x >> std::min(y, 31));
410 }
411#endif

654ALWAYS_INLINE JSValue tryConvertToBigInt32(JSBigInt* bigInt)
655{
656#if USE(BIGINT32)
657 if (UNLIKELY(!bigInt))
658 return JSValue();
659
660 if (bigInt->length() <= 1) {
661 if (!bigInt->length())
662 return jsBigInt32(0);
663 JSBigInt::Digit digit = bigInt->digit(0);
664 if (bigInt->sign()) {
665 static constexpr uint64_t maxValue = -static_cast<int64_t>(std::numeric_limits<int32_t>::min());
666 if (digit <= maxValue)
667 return jsBigInt32(static_cast<int32_t>(-static_cast<int64_t>(digit)));
668 } else {
669 static constexpr uint64_t maxValue = static_cast<uint64_t>(std::numeric_limits<int32_t>::max());
670 if (digit <= maxValue)
671 return jsBigInt32(static_cast<int32_t>(digit));
672 }
673 }
674#endif

676 return bigInt;
