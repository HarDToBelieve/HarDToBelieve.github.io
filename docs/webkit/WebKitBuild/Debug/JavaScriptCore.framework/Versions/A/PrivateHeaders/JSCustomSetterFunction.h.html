26#pragma once
27
28#include "JSFunction.h"
29
30namespace JSC {
31
32class JSCustomSetterFunction final : public JSFunction {
33public:
34 typedef JSFunction Base;
35 static constexpr unsigned StructureFlags = Base::StructureFlags;
36
37 using CustomFunctionPointer = PutValueFunc;
38
39 static constexpr bool needsDestruction = true;
40 static void destroy(JSCell*);
41
42 template<typename CellType, SubspaceAccess mode>
43 static IsoSubspace* subspaceFor(VM& vm)
44 {
45 return vm.customSetterFunctionSpace<mode>();
46 }
47
48 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
49 {
50 ASSERT(globalObject);
51 return Structure::create(vm, globalObject, prototype, TypeInfo(JSFunctionType, StructureFlags), info());
52 }
53
54 JS_EXPORT_PRIVATE static JSCustomSetterFunction* create(VM&, JSGlobalObject*, const PropertyName&, CustomFunctionPointer);
55
56 DECLARE_EXPORT_INFO;
57
58 const Identifier& propertyName() const { return m_propertyName; }
59 CustomFunctionPointer setter() const { return m_setter; };
60 CustomFunctionPointer customFunctionPointer() const { return m_setter; };
61
62private:
63 JSCustomSetterFunction(VM&, NativeExecutable*, JSGlobalObject*, Structure*, const PropertyName&, CustomFunctionPointer);
64
65 Identifier m_propertyName;
66 CustomFunctionPointer m_setter;
67};
68
69} // namespace JSC
