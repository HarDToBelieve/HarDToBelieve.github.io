51class JSGlobalObject;

55static constexpr double minECMAScriptTime = -8.64E15;

57// We do not expose icu::TimeZone in this header file. And we cannot use icu::TimeZone forward declaration
58// because icu namespace can be an alias to icu$verNum namespace.
59struct OpaqueICUTimeZoneDeleter {
60 JS_EXPORT_PRIVATE void operator()(OpaqueICUTimeZone*);
61};
62
63struct LocalTimeOffsetCache {
64 LocalTimeOffsetCache() = default;
65
66 LocalTimeOffset offset;
67 double start { 0.0 };
68 double end { -1.0 };
69 double incrementStart { 0.0 };
70 double incrementEnd { 0.0 };
71};
72
73class DateCache {
74 WTF_MAKE_FAST_ALLOCATED;
75 WTF_MAKE_NONCOPYABLE(DateCache);
76public:
77 DateCache();
78 ~DateCache();
79
80 JS_EXPORT_PRIVATE void reset();
81
82 String defaultTimeZone();
83
84 Ref<DateInstanceData> cachedDateInstanceData(double millisecondsFromEpoch);
85
86 void msToGregorianDateTime(double millisecondsFromEpoch, WTF::TimeType outputTimeType, GregorianDateTime&);
87 double gregorianDateTimeToMS(const GregorianDateTime&, double milliseconds, WTF::TimeType inputTimeType);
88 JS_EXPORT_PRIVATE double parseDate(JSGlobalObject*, VM&, const WTF::String&);
89
90private:
91 void timeZoneCacheSlow();
92 LocalTimeOffset localTimeOffset(double millisecondsFromEpoch, WTF::TimeType inputTimeType = WTF::UTCTime);
93 LocalTimeOffset calculateLocalTimeOffset(double millisecondsFromEpoch, WTF::TimeType inputTimeType);
94
95 OpaqueICUTimeZone* timeZoneCache()
96 {
97 if (!m_timeZoneCache)
98 timeZoneCacheSlow();
99 return m_timeZoneCache.get();
100 }
101
102 std::unique_ptr<OpaqueICUTimeZone, OpaqueICUTimeZoneDeleter> m_timeZoneCache;
103 LocalTimeOffsetCache m_utcTimeOffsetCache;
104 LocalTimeOffsetCache m_localTimeOffsetCache;
105 String m_cachedDateString;
106 double m_cachedDateStringValue;
107 DateInstanceCache m_dateInstanceCache;
108};
109
110ALWAYS_INLINE bool isUTCEquivalent(StringView timeZone)
111{
112 return timeZone == "Etc/UTC"_s || timeZone == "Etc/GMT"_s;
113}
