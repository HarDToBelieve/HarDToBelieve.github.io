27
28#include "JSObject.h"
29
30namespace JSC {
31
32struct ClassInfo;
33
34class JSDestructibleObject : public JSNonFinalObject {
35public:
36 using Base = JSNonFinalObject;
37
38 static constexpr bool needsDestruction = true;
39
40 const ClassInfo* classInfo() const { return m_classInfo; }
41
42protected:
43 JSDestructibleObject(VM& vm, Structure* structure, Butterfly* butterfly = nullptr)
44 : JSNonFinalObject(vm, structure, butterfly)
45 , m_classInfo(structure->classInfo())
46 {
47 ASSERT(m_classInfo);
48 }
49
50private:
51 const ClassInfo* m_classInfo;
52};

54} // namespace JSC
