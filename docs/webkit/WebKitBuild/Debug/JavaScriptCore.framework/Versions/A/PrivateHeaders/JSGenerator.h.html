26#pragma once
27
28#include "JSInternalFieldObjectImpl.h"
29
30namespace JSC {
31
32class JSGenerator final : public JSInternalFieldObjectImpl<5> {
33public:
34 using Base = JSInternalFieldObjectImpl<5>;
35
36 // JSGenerator has one inline storage slot, which is pointing internalField(0).
37 static size_t allocationSize(Checked<size_t> inlineCapacity)
38 {
39 ASSERT_UNUSED(inlineCapacity, inlineCapacity == 0U);
40 return sizeof(JSGenerator);
41 }
42
43 template<typename CellType, SubspaceAccess mode>
44 static IsoSubspace* subspaceFor(VM& vm)
45 {
46 return vm.generatorSpace<mode>();
47 }
48
49 enum class ResumeMode : int32_t {
50 NormalMode = 0,
51 ReturnMode = 1,
52 ThrowMode = 2
53 };
54
55 enum class State : int32_t {
56 Completed = -1,
57 Executing = -2,
58 Init = 0,
59 };
60
61 // [this], @generator, @generatorState, @generatorValue, @generatorResumeMode, @generatorFrame.
62 enum class Argument : int32_t {
63 ThisValue = 0,
64 Generator = 1,
65 State = 2,
66 Value = 3,
67 ResumeMode = 4,
68 Frame = 5,
69 NumberOfArguments = Frame,
70 };
71
72 enum class Field : uint32_t {
73 // FIXME: JSGenerator should support PolyProto, since generator tends to be created with poly proto mode.
74 // We reserve the first internal field for PolyProto property. This offset is identical to JSFinalObject's first inline storage slot which will be used for PolyProto.
75 PolyProto = 0,
76 State,
77 Next,
78 This,
79 Frame,
80 };
81 static_assert(numberOfInternalFields == 5);
82 static std::array<JSValue, numberOfInternalFields> initialValues()
83 {
84 return { {
85 jsNull(),
86 jsNumber(static_cast<int32_t>(State::Init)),
87 jsUndefined(),
88 jsUndefined(),
89 jsUndefined(),
90 } };
91 }
92
93 static JSGenerator* create(VM&, Structure*);
94 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
95
96 DECLARE_EXPORT_INFO;
97
98 DECLARE_VISIT_CHILDREN;
99
100private:
101 JSGenerator(VM&, Structure*);
102 void finishCreation(VM&);
103};
104
105OVERLOAD_RELATIONAL_OPERATORS_FOR_ENUM_CLASS_WITH_INTEGRALS(JSGenerator::Argument);
106
107} // namespace JSC
