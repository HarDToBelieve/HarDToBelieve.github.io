99template<typename Adaptor>
100JSGenericTypedArrayView<Adaptor>* JSGenericTypedArrayView<Adaptor>::create(
101 JSGlobalObject* globalObject, Structure* structure, RefPtr<ArrayBuffer>&& buffer,
102 unsigned byteOffset, unsigned length)
103{
104 VM& vm = globalObject->vm();
105 auto scope = DECLARE_THROW_SCOPE(vm);
106 size_t size = sizeof(typename Adaptor::Type);
107 ASSERT(buffer);
108 if (!ArrayBufferView::verifySubRangeLength(*buffer, byteOffset, length, size)) {
109 throwException(globalObject, scope, createRangeError(globalObject, "Length out of range of buffer"));
110 return nullptr;
111 }
112 if (!ArrayBufferView::verifyByteOffsetAlignment(byteOffset, size)) {
113 throwException(globalObject, scope, createRangeError(globalObject, "Byte offset is not aligned"));
114 return nullptr;
115 }
116 ConstructionContext context(vm, structure, WTFMove(buffer), byteOffset, length);
117 ASSERT(context);
118 JSGenericTypedArrayView* result =
119 new (NotNull, allocateCell<JSGenericTypedArrayView>(vm.heap))
120 JSGenericTypedArrayView(vm, context);
121 result->finishCreation(vm);
122 return result;
123}

146template<typename Adaptor>
147bool JSGenericTypedArrayView<Adaptor>::validateRange(
148 JSGlobalObject* globalObject, unsigned offset, unsigned length)
149{
150 VM& vm = globalObject->vm();
151 auto scope = DECLARE_THROW_SCOPE(vm);
152 if (canAccessRangeQuickly(offset, length))
153 return true;
154
155 throwException(globalObject, scope, createRangeError(globalObject, "Range consisting of offset and length are out of bounds"));
156 return false;

159template<typename Adaptor>
160template<typename OtherAdaptor>
161bool JSGenericTypedArrayView<Adaptor>::setWithSpecificType(
162 JSGlobalObject* globalObject, unsigned offset, JSGenericTypedArrayView<OtherAdaptor>* other,
163 unsigned otherOffset, unsigned length, CopyType type)
164{
165 VM& vm = globalObject->vm();
166 auto scope = DECLARE_THROW_SCOPE(vm);
167
168 // Handle the hilarious case: the act of getting the length could have resulted
169 // in detaching. Well, no. That'll never happen because there cannot be
170 // side-effects on getting the length of a typed array. But predicting where there
171 // are, or aren't, side-effects is a fool's game so we resort to this cheap
172 // check. Worst case, if we're wrong, people start seeing less things get copied
173 // but we won't have a security vulnerability.
174 length = std::min(length, other->length());
175
176 RELEASE_ASSERT(other->canAccessRangeQuickly(otherOffset, length));
177 bool success = validateRange(globalObject, offset, length);
178 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception() == success);
179 if (!success)
180 return false;
181
182 if constexpr (Adaptor::contentType != OtherAdaptor::contentType) {
183 throwTypeError(globalObject, scope, "Content types of source and destination typed arrays are different"_s);
184 return false;
185 }
186
187 // This method doesn't support copying between the same array. Note that
188 // set() will only call this if the types differ, which implicitly guarantees
189 // that we can't be the same array. This is relevant because the way we detect
190 // non-overlapping is by checking if either (a) either array doesn't have a
191 // backing buffer or (b) the backing buffers are different, but that doesn't
192 // catch the case where it's the *same* array - fortunately though, this code
193 // path never needs to worry about that case.
194 ASSERT(static_cast<JSCell*>(this) != static_cast<JSCell*>(other));
195
196 // 1) If the two arrays are non-overlapping, we can copy in any order we like
197 // and we don't need an intermediate buffer. Arrays are definitely
198 // non-overlapping if either one of them has no backing buffer (that means
199 // that it *owns* its philosophical backing buffer) or if they have
200 // different backing buffers.
201 // 2) If the two arrays overlap but have the same element size, we can do a
202 // memmove-like copy where we flip-flop direction based on which vector
203 // starts before the other:
204 // A) If the destination vector is before the source vector, then a forward
205 // copy is in order.
206 // B) If the destination vector is after the source vector, then a backward
207 // copy is in order.
208 // 3) If we have different element sizes and there is a chance of overlap then
209 // we need an intermediate vector.
210
211 // NB. Comparisons involving elementSize will be constant-folded by template
212 // specialization.
213
214 unsigned otherElementSize = sizeof(typename OtherAdaptor::Type);
215
216 // Handle cases (1) and (2A).
217 if (!hasArrayBuffer() || !other->hasArrayBuffer()
218 || existingBuffer() != other->existingBuffer()
219 || (elementSize == otherElementSize && vector() <= other->vector())
220 || type == CopyType::LeftToRight) {
221 for (unsigned i = 0; i < length; ++i) {
222 setIndexQuicklyToNativeValue(
223 offset + i, OtherAdaptor::template convertTo<Adaptor>(
224 other->getIndexQuicklyAsNativeValue(i + otherOffset)));
225 }
226 return true;
227 }
228
229 // Now we either have (2B) or (3) - so first we try to cover (2B).
230 if (elementSize == otherElementSize) {
231 for (unsigned i = length; i--;) {
232 setIndexQuicklyToNativeValue(
233 offset + i, OtherAdaptor::template convertTo<Adaptor>(
234 other->getIndexQuicklyAsNativeValue(i + otherOffset)));
235 }
236 return true;
237 }
238
239 // Fail: we need an intermediate transfer buffer (i.e. case (3)).
240 Vector<typename Adaptor::Type, 32> transferBuffer(length);
241 for (unsigned i = length; i--;) {
242 transferBuffer[i] = OtherAdaptor::template convertTo<Adaptor>(
243 other->getIndexQuicklyAsNativeValue(i + otherOffset));
244 }
245 for (unsigned i = length; i--;)
246 setIndexQuicklyToNativeValue(offset + i, transferBuffer[i]);
247
248 return true;
249}

251template<typename Adaptor>
252bool JSGenericTypedArrayView<Adaptor>::set(
253 JSGlobalObject* globalObject, unsigned offset, JSObject* object, unsigned objectOffset, unsigned length, CopyType type)
254{
255 VM& vm = globalObject->vm();
256 auto scope = DECLARE_THROW_SCOPE(vm);
257
258 const ClassInfo* ci = object->classInfo(vm);
259 if (ci->typedArrayStorageType == Adaptor::typeValue) {
260 // The super fast case: we can just memcpy since we're the same type.
261 JSGenericTypedArrayView* other = jsCast<JSGenericTypedArrayView*>(object);
262 length = std::min(length, other->length());
263
264 RELEASE_ASSERT(other->canAccessRangeQuickly(objectOffset, length));
265 bool success = validateRange(globalObject, offset, length);
266 EXCEPTION_ASSERT(!scope.exception() == success);
267 if (!success)
268 return false;
269
270 memmove(typedVector() + offset, other->typedVector() + objectOffset, length * elementSize);
271 return true;
272 }
273
274 switch (ci->typedArrayStorageType) {
275 case TypeInt8:
276 RELEASE_AND_RETURN(scope, setWithSpecificType<Int8Adaptor>(
277 globalObject, offset, jsCast<JSInt8Array*>(object), objectOffset, length, type));
278 case TypeInt16:
279 RELEASE_AND_RETURN(scope, setWithSpecificType<Int16Adaptor>(
280 globalObject, offset, jsCast<JSInt16Array*>(object), objectOffset, length, type));
281 case TypeInt32:
282 RELEASE_AND_RETURN(scope, setWithSpecificType<Int32Adaptor>(
283 globalObject, offset, jsCast<JSInt32Array*>(object), objectOffset, length, type));
284 case TypeUint8:
285 RELEASE_AND_RETURN(scope, setWithSpecificType<Uint8Adaptor>(
286 globalObject, offset, jsCast<JSUint8Array*>(object), objectOffset, length, type));
287 case TypeUint8Clamped:
288 RELEASE_AND_RETURN(scope, setWithSpecificType<Uint8ClampedAdaptor>(
289 globalObject, offset, jsCast<JSUint8ClampedArray*>(object), objectOffset, length, type));
290 case TypeUint16:
291 RELEASE_AND_RETURN(scope, setWithSpecificType<Uint16Adaptor>(
292 globalObject, offset, jsCast<JSUint16Array*>(object), objectOffset, length, type));
293 case TypeUint32:
294 RELEASE_AND_RETURN(scope, setWithSpecificType<Uint32Adaptor>(
295 globalObject, offset, jsCast<JSUint32Array*>(object), objectOffset, length, type));
296 case TypeFloat32:
297 RELEASE_AND_RETURN(scope, setWithSpecificType<Float32Adaptor>(
298 globalObject, offset, jsCast<JSFloat32Array*>(object), objectOffset, length, type));
299 case TypeFloat64:
300 RELEASE_AND_RETURN(scope, setWithSpecificType<Float64Adaptor>(
301 globalObject, offset, jsCast<JSFloat64Array*>(object), objectOffset, length, type));
302 case TypeBigInt64:
303 RELEASE_AND_RETURN(scope, setWithSpecificType<BigInt64Adaptor>(
304 globalObject, offset, jsCast<JSBigInt64Array*>(object), objectOffset, length, type));
305 case TypeBigUint64:
306 RELEASE_AND_RETURN(scope, setWithSpecificType<BigUint64Adaptor>(
307 globalObject, offset, jsCast<JSBigUint64Array*>(object), objectOffset, length, type));
308 case NotTypedArray:
309 case TypeDataView: {
310 bool success = validateRange(globalObject, offset, length);
311 EXCEPTION_ASSERT(!scope.exception() == success);
312 if (!success)
313 return false;
314
315 // We could optimize this case. But right now, we don't.
316 for (unsigned i = 0; i < length; ++i) {
317 JSValue value = object->get(globalObject, i + objectOffset);
318 RETURN_IF_EXCEPTION(scope, false);
319 bool success = setIndex(globalObject, offset + i, value);
320 EXCEPTION_ASSERT(!scope.exception() || !success);
321 if (!success) {
322 if (isDetached())
323 throwTypeError(globalObject, scope, typedArrayBufferHasBeenDetachedErrorMessage);
324 return false;
325 }
326 }
327 return true;
328 } }
329
330 RELEASE_ASSERT_NOT_REACHED();
331 return false;
332}

352template<typename Adaptor>
353bool JSGenericTypedArrayView<Adaptor>::getOwnPropertySlot(
354 JSObject* object, JSGlobalObject* globalObject, PropertyName propertyName, PropertySlot& slot)
355{
356 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(object);
357
358 if (std::optional<uint32_t> index = parseIndex(propertyName)) {
359 static_assert(std::is_final_v<JSGenericTypedArrayView<Adaptor>>, "getOwnPropertySlotByIndex must not be overridden");
360 return getOwnPropertySlotByIndex(thisObject, globalObject, index.value(), slot);
361 }
362
363 if (isCanonicalNumericIndexString(propertyName))
364 return false;
365
366 return Base::getOwnPropertySlot(thisObject, globalObject, propertyName, slot);
367}

369template<typename Adaptor>
370bool JSGenericTypedArrayView<Adaptor>::put(
371 JSCell* cell, JSGlobalObject* globalObject, PropertyName propertyName, JSValue value,
372 PutPropertySlot& slot)
373{
374 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(cell);
375
376 if (std::optional<uint32_t> index = parseIndex(propertyName)) {
377 static_assert(std::is_final_v<JSGenericTypedArrayView<Adaptor>>, "putByIndex must not be overridden");
378 return putByIndex(thisObject, globalObject, index.value(), value, slot.isStrictMode());
379 }
380
381 if (isCanonicalNumericIndexString(propertyName)) {
382 // Cases like '-0', '1.1', etc. are still obliged to give the RHS a chance to throw.
383 toNativeFromValue<Adaptor>(globalObject, value);
384 return true;
385 }
386
387 return Base::put(thisObject, globalObject, propertyName, value, slot);
388}

390template<typename Adaptor>
391bool JSGenericTypedArrayView<Adaptor>::defineOwnProperty(
392 JSObject* object, JSGlobalObject* globalObject, PropertyName propertyName,
393 const PropertyDescriptor& descriptor, bool shouldThrow)
394{
395 VM& vm = globalObject->vm();
396 auto scope = DECLARE_THROW_SCOPE(vm);
397 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(object);
398
399 if (std::optional<uint32_t> index = parseIndex(propertyName)) {
400 auto throwTypeErrorIfNeeded = [&] (const char* errorMessage) -> bool {
401 if (shouldThrow)
402 throwTypeError(globalObject, scope, makeString(errorMessage, *index));
403 return false;
404 };
405
406 if (thisObject->isDetached())
407 return typeError(globalObject, scope, shouldThrow, typedArrayBufferHasBeenDetachedErrorMessage);
408
409 if (!thisObject->inBounds(index.value()))
410 return throwTypeErrorIfNeeded("Attempting to store out-of-bounds property on a typed array at index: ");
411
412 if (descriptor.isAccessorDescriptor())
413 return throwTypeErrorIfNeeded("Attempting to store accessor property on a typed array at index: ");
414
415 if (descriptor.configurablePresent() && !descriptor.configurable())
416 return throwTypeErrorIfNeeded("Attempting to store non-configurable property on a typed array at index: ");
417
418 if (descriptor.enumerablePresent() && !descriptor.enumerable())
419 return throwTypeErrorIfNeeded("Attempting to store non-enumerable property on a typed array at index: ");
420
421 if (descriptor.writablePresent() && !descriptor.writable())
422 return throwTypeErrorIfNeeded("Attempting to store non-writable property on a typed array at index: ");
423
424 scope.release();
425 if (descriptor.value())
426 thisObject->setIndex(globalObject, index.value(), descriptor.value());
427
428 return true;
429 }
430
431 if (isCanonicalNumericIndexString(propertyName))
432 return typeError(globalObject, scope, shouldThrow, "Attempting to store canonical numeric string property on a typed array"_s);
433
434 RELEASE_AND_RETURN(scope, Base::defineOwnProperty(thisObject, globalObject, propertyName, descriptor, shouldThrow));
435}

454template<typename Adaptor>
455bool JSGenericTypedArrayView<Adaptor>::getOwnPropertySlotByIndex(
456 JSObject* object, JSGlobalObject* globalObject, unsigned propertyName, PropertySlot& slot)
457{
458 VM& vm = globalObject->vm();
459 auto scope = DECLARE_THROW_SCOPE(vm);
460 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(object);
461
462 if (thisObject->isDetached() || !thisObject->inBounds(propertyName))
463 return false;
464
465 JSValue value;
466 if constexpr (Adaptor::canConvertToJSQuickly) {
467 UNUSED_VARIABLE(scope);
468 value = thisObject->getIndexQuickly(propertyName);
469 } else {
470 auto nativeValue = thisObject->getIndexQuicklyAsNativeValue(propertyName);
471 value = Adaptor::toJSValue(globalObject, nativeValue);
472 RETURN_IF_EXCEPTION(scope, false);
473 }
474 slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::None), value);
475 return true;
476}

487template<typename Adaptor>
488bool JSGenericTypedArrayView<Adaptor>::deletePropertyByIndex(
489 JSCell* cell, JSGlobalObject*, unsigned propertyName)
490{
491 // Integer-indexed elements can't be deleted, so we must return false when the index is valid.
492 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(cell);
493 return thisObject->isDetached() || propertyName >= thisObject->m_length;

511template<typename Adaptor>
512size_t JSGenericTypedArrayView<Adaptor>::estimatedSize(JSCell* cell, VM& vm)
513{
514 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(cell);
515
516 if (thisObject->m_mode == OversizeTypedArray)
517 return Base::estimatedSize(thisObject, vm) + thisObject->byteSize();
518 if (thisObject->m_mode == FastTypedArray && thisObject->hasVector())
519 return Base::estimatedSize(thisObject, vm) + thisObject->byteSize();
520
521 return Base::estimatedSize(thisObject, vm);
522}

524template<typename Adaptor>
525template<typename Visitor>
526void JSGenericTypedArrayView<Adaptor>::visitChildrenImpl(JSCell* cell, Visitor& visitor)
527{
528 JSGenericTypedArrayView* thisObject = jsCast<JSGenericTypedArrayView*>(cell);
529 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
530 Base::visitChildren(thisObject, visitor);
531
532 TypedArrayMode mode;
533 void* vector;
534 size_t byteSize;
535
536 {
537 Locker locker { thisObject->cellLock() };
538 mode = thisObject->m_mode;
539 vector = thisObject->vector();
540 byteSize = thisObject->byteSize();
541 }
542
543 switch (mode) {
544 case FastTypedArray: {
545 if (vector)
546 visitor.markAuxiliary(vector);
547 break;
548 }
549
550 case OversizeTypedArray: {
551 visitor.reportExtraMemoryVisited(byteSize);
552 break;
553 }
554
555 case WastefulTypedArray:
556 break;
557
558 case DataViewMode:
559 RELEASE_ASSERT_NOT_REACHED();
560 break;
561 }
