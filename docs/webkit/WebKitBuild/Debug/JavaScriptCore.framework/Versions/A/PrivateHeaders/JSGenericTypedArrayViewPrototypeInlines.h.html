30namespace JSC {
31
32template<typename ViewClass>
33JSGenericTypedArrayViewPrototype<ViewClass>::JSGenericTypedArrayViewPrototype(VM& vm, Structure* structure)
34 : Base(vm, structure)
35{
36}
37
38template<typename ViewClass>
39void JSGenericTypedArrayViewPrototype<ViewClass>::finishCreation(
40 VM& vm, JSGlobalObject*)
41{
42 Base::finishCreation(vm);
43
44 ASSERT(inherits(vm, info()));
45
46 putDirect(vm, vm.propertyNames->BYTES_PER_ELEMENT, jsNumber(ViewClass::elementSize), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly | PropertyAttribute::DontDelete);
47
48}
49
50template<typename ViewClass>
51JSGenericTypedArrayViewPrototype<ViewClass>*
52JSGenericTypedArrayViewPrototype<ViewClass>::create(
53 VM& vm, JSGlobalObject* globalObject, Structure* structure)
54{
55 JSGenericTypedArrayViewPrototype* prototype =
56 new (NotNull, allocateCell<JSGenericTypedArrayViewPrototype>(vm.heap))
57 JSGenericTypedArrayViewPrototype(vm, structure);
58 prototype->finishCreation(vm, globalObject);
59 return prototype;
60}
61
62template<typename ViewClass>
63Structure* JSGenericTypedArrayViewPrototype<ViewClass>::createStructure(
64 VM& vm, JSGlobalObject* globalObject, JSValue prototype)
65{
66 return Structure::create(
67 vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
68}
