37class JSImmutableButterfly : public JSCell {
38 using Base = JSCell;
39
40public:
41 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
42
43 DECLARE_INFO;
44
45 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype, IndexingType indexingType)
46 {
47 return Structure::create(vm, globalObject, prototype, TypeInfo(JSImmutableButterflyType, StructureFlags), info(), indexingType);
48 }
49
50 ALWAYS_INLINE static JSImmutableButterfly* tryCreate(VM& vm, Structure* structure, unsigned length)
51 {
52 if (UNLIKELY(length > IndexingHeader::maximumLength))
53 return nullptr;
54
55 // Because of the above maximumLength requirement, allocationSize can never overflow.
56 void* buffer = tryAllocateCell<JSImmutableButterfly>(vm.heap, allocationSize(length));
57 if (UNLIKELY(!buffer))
58 return nullptr;
59 JSImmutableButterfly* result = new (NotNull, buffer) JSImmutableButterfly(vm, structure, length);
60 result->finishCreation(vm);
61 return result;
62 }
63
64 static JSImmutableButterfly* create(VM& vm, IndexingType indexingType, unsigned length)
65 {
66 auto* array = tryCreate(vm, vm.immutableButterflyStructures[arrayIndexFromIndexingType(indexingType) - NumberOfIndexingShapes].get(), length);
67 RELEASE_ASSERT(array);
68 return array;
69 }
70
71 ALWAYS_INLINE static JSImmutableButterfly* createFromArray(JSGlobalObject* globalObject, VM& vm, JSArray* array)
72 {
73 auto throwScope = DECLARE_THROW_SCOPE(vm);
74
75 IndexingType indexingType = array->indexingType() & IndexingShapeMask;
76 unsigned length = array->length();
77
78 // FIXME: JSImmutableButterfly::createFromArray should support re-using non contiguous indexing types as well.
79 if (isCopyOnWrite(indexingType)) {
80 if (hasContiguous(indexingType))
81 return JSImmutableButterfly::fromButterfly(array->butterfly());
82 }
83
84 JSImmutableButterfly* result = JSImmutableButterfly::tryCreate(vm, vm.immutableButterflyStructures[arrayIndexFromIndexingType(CopyOnWriteArrayWithContiguous) - NumberOfIndexingShapes].get(), length);
85 if (UNLIKELY(!result)) {
86 throwOutOfMemoryError(globalObject, throwScope);
87 return nullptr;
88 }
89
90 if (!length)
91 return result;
92
93 if (indexingType == ContiguousShape || indexingType == Int32Shape) {
94 for (unsigned i = 0; i < length; i++) {
95 JSValue value = array->butterfly()->contiguous().at(array, i).get();
96 value = !!value ? value : jsUndefined();
97 result->setIndex(vm, i, value);
98 }
99 return result;
100 }
101
102 if (indexingType == DoubleShape) {
103 for (unsigned i = 0; i < length; i++) {
104 double d = array->butterfly()->contiguousDouble().at(array, i);
105 JSValue value = std::isnan(d) ? jsUndefined() : JSValue(JSValue::EncodeAsDouble, d);
106 result->setIndex(vm, i, value);
107 }
108 return result;
109 }
110
111 for (unsigned i = 0; i < length; i++) {
112 JSValue value = array->getDirectIndex(globalObject, i);
113 if (!value) {
114 // When we see a hole, we assume that it's safe to assume the get would have returned undefined.
115 // We may still call into this function when !globalObject->isArrayIteratorProtocolFastAndNonObservable(),
116 // however, if we do that, we ensure we're calling in with an array with all self properties between
117 // [0, length).
118 //
119 // We may also call into this during OSR exit to materialize a phantom fixed array.
120 // We may be creating a fixed array during OSR exit even after the iterator protocol changed.
121 // But, when the phantom would have logically been created, the protocol hadn't been
122 // changed. Therefore, it is sound to assume empty indices are jsUndefined().
123 value = jsUndefined();
124 }
125 RETURN_IF_EXCEPTION(throwScope, nullptr);
126 result->setIndex(vm, i, value);
127 }
128 return result;
129 }
130
131 unsigned publicLength() const { return m_header.publicLength(); }
132 unsigned vectorLength() const { return m_header.vectorLength(); }
133 unsigned length() const { return m_header.publicLength(); }
134
135 Butterfly* toButterfly() const { return bitwise_cast<Butterfly*>(bitwise_cast<char*>(this) + offsetOfData()); }
136 static JSImmutableButterfly* fromButterfly(Butterfly* butterfly) { return bitwise_cast<JSImmutableButterfly*>(bitwise_cast<char*>(butterfly) - offsetOfData()); }
137
138 JSValue get(unsigned index) const
139 {
140 if (!hasDouble(indexingMode()))
141 return toButterfly()->contiguous().at(this, index).get();
142 double value = toButterfly()->contiguousDouble().at(this, index);
143 // Holes are not supported yet.
144 ASSERT(!std::isnan(value));
145 return jsDoubleNumber(value);
146 }
147
148 DECLARE_VISIT_CHILDREN;
149
150 void copyToArguments(JSGlobalObject*, JSValue* firstElementDest, unsigned offset, unsigned length);
151
152 template<typename, SubspaceAccess>
153 static CompleteSubspace* subspaceFor(VM& vm)
154 {
155 // We allocate out of the JSValue gigacage as other code expects all butterflies to live there.
156 return &vm.immutableButterflyJSValueGigacageAuxiliarySpace;
157 }
158
159 // Only call this if you just allocated this butterfly.
160 void setIndex(VM& vm, unsigned index, JSValue value)
161 {
162 if (hasDouble(indexingType()))
163 toButterfly()->contiguousDouble().atUnsafe(index) = value.asNumber();
164 else
165 toButterfly()->contiguous().atUnsafe(index).set(vm, this, value);
166 }
167
168 static constexpr size_t offsetOfData()
169 {
170 return WTF::roundUpToMultipleOf<sizeof(WriteBarrier<Unknown>)>(sizeof(JSImmutableButterfly));
171 }
172
173 static ptrdiff_t offsetOfPublicLength()
174 {
175 return OBJECT_OFFSETOF(JSImmutableButterfly, m_header) + IndexingHeader::offsetOfPublicLength();
176 }
177
178 static ptrdiff_t offsetOfVectorLength()
179 {
180 return OBJECT_OFFSETOF(JSImmutableButterfly, m_header) + IndexingHeader::offsetOfVectorLength();
181 }
182
183 static Checked<size_t> allocationSize(Checked<size_t> numItems)
184 {
185 return offsetOfData() + numItems * sizeof(WriteBarrier<Unknown>);
186 }
187
188private:
189 JSImmutableButterfly(VM& vm, Structure* structure, unsigned length)
190 : Base(vm, structure)
191 {
192 m_header.setVectorLength(length);
193 m_header.setPublicLength(length);
194 if (hasContiguous(indexingType())) {
195 for (unsigned index = 0; index < length; ++index)
196 toButterfly()->contiguous().at(this, index).setStartingValue(JSValue());
197 }
198 }
199
200 IndexingHeader m_header;
201};
