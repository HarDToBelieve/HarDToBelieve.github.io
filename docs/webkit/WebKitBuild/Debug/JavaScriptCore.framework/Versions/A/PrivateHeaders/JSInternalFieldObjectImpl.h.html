27
28#include "JSObject.h"

30namespace JSC {
31
32// This is used for sharing interface and implementation. It should not have its own classInfo.
33template<unsigned passedNumberOfInternalFields = 1>
34class JSInternalFieldObjectImpl : public JSNonFinalObject {
35public:
36 friend class LLIntOffsetsExtractor;
37
38 using Base = JSNonFinalObject;
39 static constexpr unsigned numberOfInternalFields = passedNumberOfInternalFields;
40
41 template<typename CellType, SubspaceAccess>
42 static void subspaceFor(VM&)
43 {
44 RELEASE_ASSERT_NOT_REACHED();
45 }
46
47 static size_t allocationSize(Checked<size_t> inlineCapacity)
48 {
49 ASSERT_UNUSED(inlineCapacity, !inlineCapacity);
50 return sizeof(JSInternalFieldObjectImpl);
51 }
52
53 const WriteBarrier<Unknown>& internalField(unsigned index) const
54 {
55 ASSERT(index < numberOfInternalFields);
56 return m_internalFields[index];
57 }
58
59 WriteBarrier<Unknown>& internalField(unsigned index)
60 {
61 ASSERT(index < numberOfInternalFields);
62 return m_internalFields[index];
63 }
64
65 static ptrdiff_t offsetOfInternalFields() { return OBJECT_OFFSETOF(JSInternalFieldObjectImpl, m_internalFields); }
66 static ptrdiff_t offsetOfInternalField(unsigned index) { return OBJECT_OFFSETOF(JSInternalFieldObjectImpl, m_internalFields) + index * sizeof(WriteBarrier<Unknown>); }
67
68protected:
69 DECLARE_VISIT_CHILDREN;
70
71 JSInternalFieldObjectImpl(VM& vm, Structure* structure)
72 : Base(vm, structure)
73 {
74 }
75
76 WriteBarrier<Unknown> m_internalFields[numberOfInternalFields] { };
77};
78
79} // namespace JSC
