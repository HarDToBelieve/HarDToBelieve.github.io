26#pragma once
27
28#include "JSObject.h"
29
30namespace JSC {
31
32class JSProxy : public JSNonFinalObject {
33public:
34 using Base = JSNonFinalObject;
35 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetOwnPropertyNames | OverridesPut | OverridesGetPrototype | InterceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero;
36
37 template<typename CellType, SubspaceAccess>
38 static IsoSubspace* subspaceFor(VM& vm)
39 {
40 static_assert(sizeof(CellType) == sizeof(JSProxy));
41 return &vm.jsProxySpace;
42 }
43
44 static JSProxy* create(VM& vm, Structure* structure, JSObject* target)
45 {
46 JSProxy* proxy = new (NotNull, allocateCell<JSProxy>(vm.heap)) JSProxy(vm, structure);
47 proxy->finishCreation(vm, target);
48 return proxy;
49 }
50
51 static JSProxy* create(VM& vm, Structure* structure)
52 {
53 JSProxy* proxy = new (NotNull, allocateCell<JSProxy>(vm.heap)) JSProxy(vm, structure);
54 proxy->finishCreation(vm);
55 return proxy;
56 }
57
58 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
59 {
60 return Structure::create(vm, globalObject, prototype, TypeInfo(PureForwardingProxyType, StructureFlags), info());
61 }
62
63 DECLARE_EXPORT_INFO;
64
65 JSObject* target() const { return m_target.get(); }
66 static ptrdiff_t targetOffset() { return OBJECT_OFFSETOF(JSProxy, m_target); }
67
68 JS_EXPORT_PRIVATE void setTarget(VM&, JSGlobalObject*);
69
70protected:
71 JSProxy(VM& vm, Structure* structure)
72 : Base(vm, structure)
73 {
74 }
75
76 void finishCreation(VM& vm)
77 {
78 Base::finishCreation(vm);
79 }
80
81 void finishCreation(VM& vm, JSObject* target)
82 {
83 Base::finishCreation(vm);
84 m_target.set(vm, this, target);
85 }
86
87 DECLARE_VISIT_CHILDREN_WITH_MODIFIER(JS_EXPORT_PRIVATE);
88
89 JS_EXPORT_PRIVATE static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
90 JS_EXPORT_PRIVATE static bool getOwnPropertySlotByIndex(JSObject*, JSGlobalObject*, unsigned, PropertySlot&);
91 JS_EXPORT_PRIVATE static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
92 JS_EXPORT_PRIVATE static bool putByIndex(JSCell*, JSGlobalObject*, unsigned, JSValue, bool shouldThrow);
93 JS_EXPORT_PRIVATE static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
94 JS_EXPORT_PRIVATE static bool deletePropertyByIndex(JSCell*, JSGlobalObject*, unsigned);
95 JS_EXPORT_PRIVATE static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, DontEnumPropertiesMode);
96 JS_EXPORT_PRIVATE static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
97 JS_EXPORT_PRIVATE static bool setPrototype(JSObject*, JSGlobalObject*, JSValue, bool shouldThrowIfCantSet);
98 JS_EXPORT_PRIVATE static JSValue getPrototype(JSObject*, JSGlobalObject*);
99 JS_EXPORT_PRIVATE static bool isExtensible(JSObject*, JSGlobalObject*);
100 JS_EXPORT_PRIVATE static bool preventExtensions(JSObject*, JSGlobalObject*);
101
102private:
103 WriteBarrier<JSObject> m_target;
104};
105
106} // namespace JSC
