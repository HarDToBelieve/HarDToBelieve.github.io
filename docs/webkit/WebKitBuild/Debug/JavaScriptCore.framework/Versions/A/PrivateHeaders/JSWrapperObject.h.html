22#pragma once
23
24#include "JSInternalFieldObjectImpl.h"
25
26namespace JSC {
27
28// This class is used as a base for classes such as String,
29// Number, Boolean and Symbol which are wrappers for primitive types.
30class JSWrapperObject : public JSInternalFieldObjectImpl<1> {
31public:
32 using Base = JSInternalFieldObjectImpl<1>;
33
34 template<typename, SubspaceAccess>
35 static void subspaceFor(VM&)
36 {
37 RELEASE_ASSERT_NOT_REACHED();
38 }
39
40 static size_t allocationSize(Checked<size_t> inlineCapacity)
41 {
42 ASSERT_UNUSED(inlineCapacity, !inlineCapacity);
43 return sizeof(JSWrapperObject);
44 }
45
46 enum class Field : uint32_t {
47 WrappedValue = 0,
48 };
49 static_assert(numberOfInternalFields == 1);
50
51 JSValue internalValue() const;
52 void setInternalValue(VM&, JSValue);
53
54 static ptrdiff_t internalValueOffset() { return offsetOfInternalField(static_cast<unsigned>(Field::WrappedValue)); }
55 static ptrdiff_t internalValueCellOffset()
56 {
57#if USE(JSVALUE64)
58 return internalValueOffset();
59#else
60 return internalValueOffset() + OBJECT_OFFSETOF(EncodedValueDescriptor, asBits.payload);
61#endif
62 }
63
64protected:
65 explicit JSWrapperObject(VM&, Structure*);
66
67 DECLARE_VISIT_CHILDREN_WITH_MODIFIER(JS_EXPORT_PRIVATE);
68};
69
70inline JSWrapperObject::JSWrapperObject(VM& vm, Structure* structure)
71 : Base(vm, structure)
72{
73}
74
75inline JSValue JSWrapperObject::internalValue() const
76{
77 return internalField(static_cast<unsigned>(Field::WrappedValue)).get();
78}
79
80inline void JSWrapperObject::setInternalValue(VM& vm, JSValue value)
81{
82 ASSERT(value);
83 ASSERT(!value.isObject());
84 internalField(static_cast<unsigned>(Field::WrappedValue)).set(vm, this, value);
85}
86
87} // namespace JSC
