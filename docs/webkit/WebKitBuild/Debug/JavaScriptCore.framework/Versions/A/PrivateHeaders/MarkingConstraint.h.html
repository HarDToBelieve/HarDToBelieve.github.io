26#pragma once
27
28#include "ConstraintConcurrency.h"
29#include "ConstraintParallelism.h"
30#include "ConstraintVolatility.h"
31#include <limits.h>
32#include <wtf/FastMalloc.h>
33#include <wtf/Lock.h>
34#include <wtf/Noncopyable.h>
35#include <wtf/SharedTask.h>
36#include <wtf/text/CString.h>
37
38namespace JSC {
39
40class MarkingConstraintSet;
41class AbstractSlotVisitor;
42class SlotVisitor;
43
44class MarkingConstraint {
45 WTF_MAKE_NONCOPYABLE(MarkingConstraint);
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 JS_EXPORT_PRIVATE MarkingConstraint(
49 CString abbreviatedName, CString name, ConstraintVolatility,
50 ConstraintConcurrency = ConstraintConcurrency::Concurrent,
51 ConstraintParallelism = ConstraintParallelism::Sequential);
52
53 JS_EXPORT_PRIVATE virtual ~MarkingConstraint();
54
55 unsigned index() const { return m_index; }
56
57 const char* abbreviatedName() const { return m_abbreviatedName.data(); }
58 const char* name() const { return m_name.data(); }
59
60 void resetStats();
61
62 size_t lastVisitCount() const { return m_lastVisitCount; }
63
64 // The following functions are only used by the real GC via the MarkingConstraintSolver.
65 // Hence, we only need the SlotVisitor version.
66 void execute(SlotVisitor&);
67
68 JS_EXPORT_PRIVATE virtual double quickWorkEstimate(SlotVisitor&);
69
70 double workEstimate(SlotVisitor&);
71
72 void prepareToExecute(const AbstractLocker& constraintSolvingLocker, SlotVisitor&);
73
74 void doParallelWork(SlotVisitor&, SharedTask<void(SlotVisitor&)>&);
75
76 ConstraintVolatility volatility() const { return m_volatility; }
77
78 ConstraintConcurrency concurrency() const { return m_concurrency; }
79 ConstraintParallelism parallelism() const { return m_parallelism; }
80
81protected:
82 virtual void executeImpl(AbstractSlotVisitor&) = 0;
83 virtual void executeImpl(SlotVisitor&) = 0;
84 JS_EXPORT_PRIVATE virtual void prepareToExecuteImpl(const AbstractLocker& constraintSolvingLocker, AbstractSlotVisitor&);
85
86 // This function is only used by the verifier GC via Heap::verifyGC().
87 // Hence, we only need the AbstractSlotVisitor version.
88 void executeSynchronously(AbstractSlotVisitor&);
89
90private:
91 friend class MarkingConstraintSet; // So it can set m_index.
92
93 CString m_abbreviatedName;
94 CString m_name;
95 size_t m_lastVisitCount { 0 };
96 unsigned m_index { UINT_MAX };
97 ConstraintVolatility m_volatility;
98 ConstraintConcurrency m_concurrency;
99 ConstraintParallelism m_parallelism;
100 Lock m_lock;
101};
102
103} // namespace JSC
