26#pragma once
27
28#include <wtf/NotFound.h>
29#include <wtf/PrintStream.h>
30
31namespace JSC {
32
33struct MatchResult;
34#if CPU(ADDRESS32)
35using EncodedMatchResult = uint64_t;
36#else
37struct EncodedMatchResult {
38 size_t start;
39 size_t end;
40};
41#endif
42
43struct MatchResult {
44 MatchResult()
45 : start(WTF::notFound)
46 , end(0)
47 {
48 }
49
50 ALWAYS_INLINE MatchResult(size_t start, size_t end)
51 : start(start)
52 , end(end)
53 {
54 }
55
56#if CPU(ADDRESS32)
57 ALWAYS_INLINE MatchResult(EncodedMatchResult match)
58 : start(bitwise_cast<MatchResult>(match).start)
59 , end(bitwise_cast<MatchResult>(match).end)
60 {
61 }
62#else
63 ALWAYS_INLINE MatchResult(EncodedMatchResult match)
64 : start(match.start)
65 , end(match.end)
66 {
67 }
68#endif
69
70 ALWAYS_INLINE static MatchResult failed()
71 {
72 return MatchResult();
73 }
74
75 ALWAYS_INLINE explicit operator bool() const
76 {
77 return start != WTF::notFound;
78 }
79
80 ALWAYS_INLINE bool empty()
81 {
82 return start == end;
83 }
84
85 void dump(PrintStream&) const;
86
87 size_t start;
88 size_t end;
89};
90
91static_assert(sizeof(EncodedMatchResult) == 2 * sizeof(size_t), "https://bugs.webkit.org/show_bug.cgi?id=198518#c11");
92static_assert(sizeof(MatchResult) == sizeof(EncodedMatchResult), "Match result and EncodedMatchResult should be the same size");
93
94} // namespace JSC
