26#ifndef OpaqueJSString_h
27#define OpaqueJSString_h
28
29#include <atomic>
30#include <wtf/ThreadSafeRefCounted.h>
31#include <wtf/text/WTFString.h>
32
33namespace JSC {
34 class Identifier;
35 class VM;
36}
37
38struct OpaqueJSString : public ThreadSafeRefCounted<OpaqueJSString> {
39 static Ref<OpaqueJSString> create()
40 {
41 return adoptRef(*new OpaqueJSString);
42 }
43
44 static Ref<OpaqueJSString> create(const LChar* characters, unsigned length)
45 {
46 return adoptRef(*new OpaqueJSString(characters, length));
47 }
48
49 static Ref<OpaqueJSString> create(const UChar* characters, unsigned length)
50 {
51 return adoptRef(*new OpaqueJSString(characters, length));
52 }
53
54 JS_EXPORT_PRIVATE static RefPtr<OpaqueJSString> tryCreate(const String&);
55 JS_EXPORT_PRIVATE static RefPtr<OpaqueJSString> tryCreate(String&&);
56
57 JS_EXPORT_PRIVATE ~OpaqueJSString();
58
59 bool is8Bit() { return m_string.is8Bit(); }
60 const LChar* characters8() { return m_string.characters8(); }
61 const UChar* characters16() { return m_string.characters16(); }
62 unsigned length() { return m_string.length(); }
63
64 const UChar* characters();
65
66 JS_EXPORT_PRIVATE String string() const;
67 JSC::Identifier identifier(JSC::VM*) const;
68
69 static bool equal(const OpaqueJSString*, const OpaqueJSString*);
70
71private:
72 friend class WTF::ThreadSafeRefCounted<OpaqueJSString>;
73
74 OpaqueJSString()
75 : m_characters(nullptr)
76 {
77 }
78
79 OpaqueJSString(const String& string)
80 : m_string(string.isolatedCopy())
81 , m_characters(m_string.impl() && m_string.is8Bit() ? nullptr : const_cast<UChar*>(m_string.characters16()))
82 {
83 }
84
85 explicit OpaqueJSString(String&& string)
86 : m_string(WTFMove(string))
87 , m_characters(m_string.impl() && m_string.is8Bit() ? nullptr : const_cast<UChar*>(m_string.characters16()))
88 {
89 }
90
91 OpaqueJSString(const LChar* characters, unsigned length)
92 : m_string(characters, length)
93 , m_characters(nullptr)
94 {
95 }
96
97 OpaqueJSString(const UChar* characters, unsigned length)
98 : m_string(characters, length)
99 , m_characters(m_string.impl() && m_string.is8Bit() ? nullptr : const_cast<UChar*>(m_string.characters16()))
100 {
101 }
102
103 String m_string;
104
105 // This will be initialized on demand when characters() is called if the string needs up-conversion.
106 std::atomic<UChar*> m_characters;
107};
108
109#endif
