39enum {
40 // Token Bitfield: 0b000000000RTE00IIIIPPPPKUXXXXXXXX
41 // R = right-associative bit
42 // T = unterminated error flag
43 // E = error flag
44 // I = binary operator allows 'in'
45 // P = binary operator precedence
46 // K = keyword flag
47 // U = unary operator flag
48 //
49 // We must keep the upper 8bit (1byte) region empty. JSTokenType must be 24bits.
50 UnaryOpTokenFlag = 1 << 8,
51 KeywordTokenFlag = 1 << 9,
52 BinaryOpTokenPrecedenceShift = 10,
53 BinaryOpTokenAllowsInPrecedenceAdditionalShift = 4,
54 BinaryOpTokenPrecedenceMask = 15 << BinaryOpTokenPrecedenceShift,
55 CanBeErrorTokenFlag = 1 << (BinaryOpTokenAllowsInPrecedenceAdditionalShift + BinaryOpTokenPrecedenceShift + 6),
56 UnterminatedCanBeErrorTokenFlag = CanBeErrorTokenFlag << 1,
57 RightAssociativeBinaryOpTokenFlag = UnterminatedCanBeErrorTokenFlag << 1
58};
59
60#define BINARY_OP_PRECEDENCE(prec) (((prec) << BinaryOpTokenPrecedenceShift) | ((prec) << (BinaryOpTokenPrecedenceShift + BinaryOpTokenAllowsInPrecedenceAdditionalShift)))
61#define IN_OP_PRECEDENCE(prec) ((prec) << (BinaryOpTokenPrecedenceShift + BinaryOpTokenAllowsInPrecedenceAdditionalShift))

206struct JSTextPosition {
207 JSTextPosition() = default;
208 JSTextPosition(int _line, int _offset, int _lineStartOffset)
209 : line(_line)
210 , offset(_offset)
211 , lineStartOffset(_lineStartOffset)
212 {
213 checkConsistency();
214 }
215
216 JSTextPosition operator+(int adjustment) const { return JSTextPosition(line, offset + adjustment, lineStartOffset); }
217 JSTextPosition operator+(unsigned adjustment) const { return *this + static_cast<int>(adjustment); }
218 JSTextPosition operator-(int adjustment) const { return *this + (- adjustment); }
219 JSTextPosition operator-(unsigned adjustment) const { return *this + (- static_cast<int>(adjustment)); }
220
221 operator int() const { return offset; }
222
223 bool operator==(const JSTextPosition& other) const
224 {
225 return line == other.line
226 && offset == other.offset
227 && lineStartOffset == other.lineStartOffset;
228 }
229 bool operator!=(const JSTextPosition& other) const
230 {
231 return !(*this == other);
232 }
233
234 int column() const { return offset - lineStartOffset; }
235 void checkConsistency()
236 {
237 // FIXME: We should test ASSERT(offset >= lineStartOffset); but that breaks a lot of tests.
238 ASSERT(line >= 0);
239 ASSERT(offset >= 0);
240 ASSERT(lineStartOffset >= 0);
241 }
242
243 // FIXME: these should be unsigned.
244 int line { -1 };
245 int offset { -1 };
246 int lineStartOffset { -1 };
247};

294ALWAYS_INLINE bool isUpdateOp(JSTokenType token)
295{
296 return token >= PLUSPLUS && token <= AUTOMINUSMINUS;
297}
298
299ALWAYS_INLINE bool isUnaryOp(JSTokenType token)
300{
301 return token & UnaryOpTokenFlag;
302}
