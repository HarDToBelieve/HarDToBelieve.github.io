33namespace JSC {

35class PropertyName {
36public:
37 PropertyName(UniquedStringImpl* propertyName)
38 : m_impl(propertyName)
39 {
40 }
41
42 PropertyName(const Identifier& propertyName)
43 : PropertyName(propertyName.impl())
44 {
45 }
46
47 PropertyName(const PrivateName& propertyName)
48 : m_impl(&propertyName.uid())
49 {
50 ASSERT(m_impl);
51 ASSERT(m_impl->isSymbol());
52 }
53
54 bool isNull() const { return !m_impl; }
55
56 bool isSymbol() const
57 {
58 return m_impl && m_impl->isSymbol();
59 }
60
61 bool isPrivateName() const
62 {
63 return isSymbol() && static_cast<const SymbolImpl*>(m_impl)->isPrivate();
64 }
65
66 UniquedStringImpl* uid() const
67 {
68 return m_impl;
69 }
70
71 AtomStringImpl* publicName() const
72 {
73 return (!m_impl || m_impl->isSymbol()) ? nullptr : static_cast<AtomStringImpl*>(m_impl);
74 }
75
76 void dump(PrintStream& out) const
77 {
78 if (m_impl)
79 out.print(m_impl);
80 else
81 out.print("<null property name>");
82 }
83
84private:
85 UniquedStringImpl* m_impl;
86};
87static_assert(sizeof(PropertyName) == sizeof(UniquedStringImpl*), "UniquedStringImpl* and PropertyName should be compatible to invoke easily from JIT code.");

89inline bool operator==(PropertyName a, const Identifier& b)
90{
91 return a.uid() == b.impl();
92}
93
94inline bool operator==(const Identifier& a, PropertyName b)
95{
96 return a.impl() == b.uid();
97}
98
99inline bool operator==(PropertyName a, PropertyName b)
100{
101 return a.uid() == b.uid();
102}
103
104inline bool operator==(PropertyName a, const char* b)
105{
106 return equal(a.uid(), b);
107}
108
109inline bool operator!=(PropertyName a, const Identifier& b)
110{
111 return a.uid() != b.impl();
112}
113
114inline bool operator!=(const Identifier& a, PropertyName b)
115{
116 return a.impl() != b.uid();
117}
118
119inline bool operator!=(PropertyName a, PropertyName b)
120{
121 return a.uid() != b.uid();
122}

124ALWAYS_INLINE std::optional<uint32_t> parseIndex(PropertyName propertyName)
125{
126 auto uid = propertyName.uid();
127 if (!uid)
128 return std::nullopt;
129 if (uid->isSymbol())
130 return std::nullopt;
131 return parseIndex(*uid);
132}

134// https://www.ecma-international.org/ecma-262/9.0/index.html#sec-canonicalnumericindexstring
135ALWAYS_INLINE bool isCanonicalNumericIndexString(const PropertyName& propertyName)
136{
137 StringImpl* property = propertyName.uid();
138 if (!property)
139 return false;
140 if (property->isSymbol())
141 return false;
142 if (equal(property, "-0"))
143 return true;
144 double index = jsToNumber(property);
145 NumberToStringBuffer buffer;
146 const char* indexString = WTF::numberToString(index, buffer);
147 if (!equal(property, indexString))
148 return false;
149 return true;
150}
