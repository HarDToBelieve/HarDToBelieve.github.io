38namespace Inspector {
39
40class FrontendChannel;
41
42using TargetID = unsigned;
43
44class JS_EXPORT_PRIVATE RemoteControllableTarget {
45public:
46 virtual ~RemoteControllableTarget();
47
48 void init();
49 void update();
50
51 virtual void connect(FrontendChannel&, bool isAutomaticConnection = false, bool immediatelyPause = false) = 0;
52 virtual void disconnect(FrontendChannel&) = 0;
53
54 TargetID targetIdentifier() const { return m_identifier; }
55 void setTargetIdentifier(TargetID identifier) { m_identifier = identifier; }
56
57 enum class Type {
58 Automation,
59 ITML,
60 JavaScript,
61 Page,
62 ServiceWorker,
63 WebPage,
64 };
65 virtual Type type() const = 0;
66 virtual bool remoteControlAllowed() const = 0;
67 virtual void dispatchMessageFromRemote(const String& message) = 0;
68
69#if USE(CF)
70 // The dispatch block will be scheduled on a global run loop if null is returned.
71 virtual CFRunLoopRef targetRunLoop() const { return nullptr; }
72#endif
73
74private:
75 TargetID m_identifier { 0 };
76};
77
78} // namespace Inspector

82 static bool isType(const Inspector::RemoteControllableTarget& target) { return target.type() == Inspector::RemoteControllableTarget::Type::ToClassType; } \
83SPECIALIZE_TYPE_TRAITS_END()
