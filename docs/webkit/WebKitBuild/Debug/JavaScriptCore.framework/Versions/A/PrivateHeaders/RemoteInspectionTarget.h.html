26#pragma once
27
28#if ENABLE(REMOTE_INSPECTOR)
29
30#include "RemoteControllableTarget.h"
31#include <wtf/RetainPtr.h>
32#include <wtf/TypeCasts.h>
33#include <wtf/text/WTFString.h>
34
35namespace Inspector {
36
37class FrontendChannel;
38
39class JS_EXPORT_PRIVATE RemoteInspectionTarget : public RemoteControllableTarget {
40public:
41 bool remoteDebuggingAllowed() const { return m_allowed; }
42 void setRemoteDebuggingAllowed(bool);
43
44#if USE(CF)
45 CFRunLoopRef targetRunLoop() const final { return m_runLoop.get(); }
46 void setTargetRunLoop(CFRunLoopRef runLoop) { m_runLoop = runLoop; }
47#endif
48
49 virtual String name() const { return String(); } // ITML JavaScript Page ServiceWorker WebPage
50 virtual String url() const { return String(); } // Page ServiceWorker WebPage
51 virtual bool hasLocalDebugger() const = 0;
52
53 virtual void setIndicating(bool) { } // Default is to do nothing.
54
55 virtual bool automaticInspectionAllowed() const { return false; }
56 virtual void pauseWaitingForAutomaticInspection();
57 virtual void unpauseForInitializedInspector();
58
59 // RemoteControllableTarget overrides.
60 bool remoteControlAllowed() const final;
61
62private:
63 bool m_allowed { false };
64#if USE(CF)
65 RetainPtr<CFRunLoopRef> m_runLoop;
66#endif
67};
68
69} // namespace Inspector
70
71SPECIALIZE_TYPE_TRAITS_BEGIN(Inspector::RemoteInspectionTarget)
72 static bool isType(const Inspector::RemoteControllableTarget& target)
73 {
74 return target.type() != Inspector::RemoteControllableTarget::Type::Automation;
75 }
76SPECIALIZE_TYPE_TRAITS_END()
77
78#endif // ENABLE(REMOTE_INSPECTOR)
