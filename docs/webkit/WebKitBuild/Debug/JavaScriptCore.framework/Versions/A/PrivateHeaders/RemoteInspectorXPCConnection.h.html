26#pragma once
27
28#if ENABLE(REMOTE_INSPECTOR)
29
30#import <dispatch/dispatch.h>
31#import <wtf/Lock.h>
32#import <wtf/OSObjectPtr.h>
33#import <wtf/ThreadSafeRefCounted.h>
34#import <wtf/spi/darwin/XPCSPI.h>
35
36OBJC_CLASS NSDictionary;
37OBJC_CLASS NSString;
38
39namespace Inspector {
40
41class RemoteInspectorXPCConnection : public ThreadSafeRefCounted<RemoteInspectorXPCConnection> {
42public:
43 class Client {
44 public:
45 virtual ~Client() { }
46 virtual void xpcConnectionReceivedMessage(RemoteInspectorXPCConnection*, NSString *messageName, NSDictionary *userInfo) = 0;
47 virtual void xpcConnectionFailed(RemoteInspectorXPCConnection*) = 0;
48 virtual void xpcConnectionUnhandledMessage(RemoteInspectorXPCConnection*, xpc_object_t) = 0;
49 };
50
51 RemoteInspectorXPCConnection(xpc_connection_t, dispatch_queue_t, Client*);
52 virtual ~RemoteInspectorXPCConnection();
53
54 void close();
55 void closeFromMessage();
56 void sendMessage(NSString *messageName, NSDictionary *userInfo);
57
58private:
59 NSDictionary *deserializeMessage(xpc_object_t);
60 void handleEvent(xpc_object_t);
61 void closeOnQueue();
62
63 // We handle XPC events on the queue, but a client may call close() on any queue.
64 // We make sure that m_client is thread safe and immediately cleared in close().
65 Lock m_mutex;
66
67 OSObjectPtr<xpc_connection_t> m_connection;
68 OSObjectPtr<dispatch_queue_t> m_queue;
69 Client* m_client;
70 bool m_closed { false };
71#if PLATFORM(MAC)
72 bool m_validated { false };
73#endif
74};
75
76} // namespace Inspector
77
78#endif // ENABLE(REMOTE_INSPECTOR)
