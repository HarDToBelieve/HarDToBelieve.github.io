143 // Attempts to perform an increment of draining that involves only walking `bytes` worth of data. This
144 // is likely to accidentally walk more or less than that. It will usually mark more than bytes. It may

146 // rare cases happen temporarily even if we're not reaching termination).

244class ParallelModeEnabler {
245public:
246 ParallelModeEnabler(SlotVisitor& stack)
247 : m_stack(stack)
248 {
249 ASSERT(!m_stack.m_isInParallelMode);
250 m_stack.m_isInParallelMode = true;
251 }
252
253 ~ParallelModeEnabler()
254 {
255 ASSERT(m_stack.m_isInParallelMode);
256 m_stack.m_isInParallelMode = false;
257 }
258
259private:
260 SlotVisitor& m_stack;
