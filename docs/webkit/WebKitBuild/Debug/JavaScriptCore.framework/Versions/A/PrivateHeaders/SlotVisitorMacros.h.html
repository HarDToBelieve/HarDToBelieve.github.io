47#define DECLARE_VISIT_AGGREGATE_WITH_MODIFIER(postModifier) \
48private: \
49 template<typename Visitor> ALWAYS_INLINE void visitAggregateImpl(Visitor&) postModifier; \
50public: \
51 void visitAggregate(AbstractSlotVisitor&) postModifier; \
52 void visitAggregate(SlotVisitor&) postModifier

54#define DECLARE_VISIT_AGGREGATE \
55 DECLARE_VISIT_AGGREGATE_WITH_MODIFIER(UNUSED_MODIFIER_ARGUMENT)

57#define DEFINE_VISIT_AGGREGATE_WITH_MODIFIER(className, postModifier) \
58 void className::visitAggregate(AbstractSlotVisitor& visitor) postModifier { visitAggregateImpl(visitor); } \
59 void className::visitAggregate(SlotVisitor& visitor) postModifier { visitAggregateImpl(visitor); } \
