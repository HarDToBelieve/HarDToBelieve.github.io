34namespace JSC {

36// It is quite unfortunate that 32 bit environment exists on DFG! This means that JSValueRegs contains 2 registers
37// in such an environment. If we use GPRReg and FPRReg in SnippetParams, SnippetParams may contain
38// different number of registers in 32bit and 64bit environments when we pass JSValueRegs, it is confusing.
39// Therefore, we introduce an abstraction that SnippetReg, which is a polymorphic register class. It can refer FPRReg,
40// GPRReg, and "JSValueRegs". Note that isGPR() will return false if the target Reg is "JSValueRegs" even if the
41// environment is 64bit.
42//
43// FIXME: Eventually we should move this class into JSC and make is available for other JIT code.
44// https://bugs.webkit.org/show_bug.cgi?id=162990
45class SnippetReg {
46public:
47 enum class Type : uint8_t {
48 GPR = 0,
49 FPR = 1,
50 JSValue = 2,
51 };
52
53 SnippetReg(GPRReg reg)
54 : m_variant(reg)
55 {
56 }
57
58 SnippetReg(FPRReg reg)
59 : m_variant(reg)
60 {
61 }
62
63 SnippetReg(JSValueRegs regs)
64 : m_variant(regs)
65 {
66 }
67
68 bool isGPR() const { return m_variant.index() == static_cast<unsigned>(Type::GPR); }
69 bool isFPR() const { return m_variant.index() == static_cast<unsigned>(Type::FPR); }
70 bool isJSValueRegs() const { return m_variant.index() == static_cast<unsigned>(Type::JSValue); }
71
72 GPRReg gpr() const
73 {
74 ASSERT(isGPR());
75 return WTF::get<GPRReg>(m_variant);
76 }
77 FPRReg fpr() const
78 {
79 ASSERT(isFPR());
80 return WTF::get<FPRReg>(m_variant);
81 }
82 JSValueRegs jsValueRegs() const
83 {
84 ASSERT(isJSValueRegs());
85 return WTF::get<JSValueRegs>(m_variant);
86 }
87
88private:
89 Variant<GPRReg, FPRReg, JSValueRegs> m_variant;
90};
