31namespace JSC {

33template <typename T, ShouldStrongDestructorGrabLock shouldStrongDestructorGrabLock>
34inline Strong<T, shouldStrongDestructorGrabLock>::Strong(VM& vm, ExternalType value)
35 : Handle<T>(vm.heap.handleSet()->allocate())
36{
37 set(value);
38}
39
40template <typename T, ShouldStrongDestructorGrabLock shouldStrongDestructorGrabLock>
41inline Strong<T, shouldStrongDestructorGrabLock>::Strong(VM& vm, Handle<T> handle)
42 : Handle<T>(vm.heap.handleSet()->allocate())
43{
44 set(handle.get());
45}
46
47template <typename T, ShouldStrongDestructorGrabLock shouldStrongDestructorGrabLock>
48inline void Strong<T, shouldStrongDestructorGrabLock>::set(VM& vm, ExternalType value)
49{
50 if (!slot())
51 setSlot(vm.heap.handleSet()->allocate());
52 set(value);
53}
