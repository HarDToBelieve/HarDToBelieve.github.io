26#pragma once
27
28#include "CellState.h"
29#include "IndexingType.h"
30#include "JSTypeInfo.h"
31#include "StructureIDTable.h"
32
33namespace JSC {
34
35class StructureIDBlob {
36 friend class LLIntOffsetsExtractor;
37public:
38 StructureIDBlob()
39 {
40 u.doubleWord = 0xbbadbeef;
41 }
42
43 StructureIDBlob(StructureID structureID, IndexingType indexingModeIncludingHistory, const TypeInfo& typeInfo)
44 {
45 u.fields.structureID = structureID;
46 u.fields.indexingModeIncludingHistory = indexingModeIncludingHistory;
47 u.fields.type = typeInfo.type();
48 u.fields.inlineTypeFlags = typeInfo.inlineTypeFlags();
49 u.fields.defaultCellState = CellState::DefinitelyWhite;
50 }
51
52 void operator=(const StructureIDBlob& other) { u.doubleWord = other.u.doubleWord; }
53
54 StructureID structureID() const { return u.fields.structureID; }
55 IndexingType indexingModeIncludingHistory() const { return u.fields.indexingModeIncludingHistory; }
56 void setIndexingModeIncludingHistory(IndexingType indexingModeIncludingHistory) { u.fields.indexingModeIncludingHistory = indexingModeIncludingHistory; }
57 JSType type() const { return u.fields.type; }
58 TypeInfo::InlineTypeFlags inlineTypeFlags() const { return u.fields.inlineTypeFlags; }
59
60 TypeInfo typeInfo(TypeInfo::OutOfLineTypeFlags outOfLineTypeFlags) const { return TypeInfo(type(), inlineTypeFlags(), outOfLineTypeFlags); }
61
62 int32_t blobExcludingStructureID() const { return u.words.word2; }
63 int64_t blob() const { return u.doubleWord; }
64
65 static ptrdiff_t structureIDOffset()
66 {
67 return OBJECT_OFFSETOF(StructureIDBlob, u.fields.structureID);
68 }
69
70 static ptrdiff_t indexingModeIncludingHistoryOffset()
71 {
72 return OBJECT_OFFSETOF(StructureIDBlob, u.fields.indexingModeIncludingHistory);
73 }
74
75private:
76 union {
77 struct {
78 StructureID structureID;
79 IndexingType indexingModeIncludingHistory;
80 JSType type;
81 TypeInfo::InlineTypeFlags inlineTypeFlags;
82 CellState defaultCellState;
83 } fields;
84 struct {
85 int32_t word1;
86 int32_t word2;
87 } words;
88 int64_t doubleWord;
89 } u;
90};
91
92} // namespace JSC
