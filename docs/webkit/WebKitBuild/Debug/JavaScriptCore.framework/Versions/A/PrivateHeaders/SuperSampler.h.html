32extern JS_EXPORT_PRIVATE volatile uint32_t g_superSamplerCount;
33
34void initializeSuperSampler();
35
36class SuperSamplerScope {
37public:
38 SuperSamplerScope(bool doSample = true)
39 : m_doSample(doSample)
40 {
41 if (m_doSample)
42 g_superSamplerCount++;
43 }
44
45 ~SuperSamplerScope()
46 {
47 if (m_doSample)
48 g_superSamplerCount--;
49 }
50
51 void release()
52 {
53 ASSERT(m_doSample);
54 g_superSamplerCount--;
55 m_doSample = false;
56 }
57
58private:
59 bool m_doSample;
60};
