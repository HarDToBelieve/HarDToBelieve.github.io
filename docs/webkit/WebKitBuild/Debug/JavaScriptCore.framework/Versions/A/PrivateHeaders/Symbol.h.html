27#pragma once
28
29#include "JSString.h"
30#include "PrivateName.h"
31
32namespace JSC {
33
34class Symbol final : public JSCell {
35public:
36 typedef JSCell Base;
37 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal | OverridesToThis;
38
39 DECLARE_EXPORT_INFO;
40
41 static constexpr bool needsDestruction = true;
42
43 template<typename CellType, SubspaceAccess mode>
44 static IsoSubspace* subspaceFor(VM& vm)
45 {
46 return vm.symbolSpace<mode>();
47 }
48
49 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
50 {
51 return Structure::create(vm, globalObject, prototype, TypeInfo(SymbolType, StructureFlags), info());
52 }
53
54 static Symbol* create(VM&);
55 static Symbol* createWithDescription(VM&, const String&);
56 JS_EXPORT_PRIVATE static Symbol* create(VM&, SymbolImpl& uid);
57
58 SymbolImpl& uid() const { return m_privateName.uid(); }
59 PrivateName privateName() const { return m_privateName; }
60 String descriptiveString() const;
61 String description() const;
62
63 JSValue toPrimitive(JSGlobalObject*, PreferredPrimitiveType) const;
64 JSObject* toObject(JSGlobalObject*) const;
65 double toNumber(JSGlobalObject*) const;
66
67 static ptrdiff_t offsetOfSymbolImpl()
68 {
69 // PrivateName is just a Ref<SymbolImpl> which can just be used as a SymbolImpl*.
70 return OBJECT_OFFSETOF(Symbol, m_privateName);
71 }
72
73private:
74 static void destroy(JSCell*);
75
76 Symbol(VM&);
77 Symbol(VM&, const String&);
78 Symbol(VM&, SymbolImpl& uid);
79
80 void finishCreation(VM&);
81
82 PrivateName m_privateName;
83};
84
85Symbol* asSymbol(JSValue);
86
87inline Symbol* asSymbol(JSValue value)
88{
89 ASSERT(value.asCell()->isSymbol());
90 return jsCast<Symbol*>(value.asCell());
91}
92
93} // namespace JSC
