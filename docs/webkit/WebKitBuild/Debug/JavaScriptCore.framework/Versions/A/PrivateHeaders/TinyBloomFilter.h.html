26#pragma once
27
28namespace JSC {
29
30typedef uintptr_t Bits;
31
32class TinyBloomFilter {
33public:
34 TinyBloomFilter() = default;
35 TinyBloomFilter(Bits);
36
37 void add(Bits);
38 void add(TinyBloomFilter&);
39 bool ruleOut(Bits) const; // True for 0.
40 void reset();
41 Bits bits() const { return m_bits; }
42
43private:
44 Bits m_bits { 0 };
45};
46
47inline TinyBloomFilter::TinyBloomFilter(Bits bits)
48 : m_bits(bits)
49{
50}
51
52inline void TinyBloomFilter::add(Bits bits)
53{
54 m_bits |= bits;
55}
56
57inline void TinyBloomFilter::add(TinyBloomFilter& other)
58{
59 m_bits |= other.m_bits;
60}
61
62inline bool TinyBloomFilter::ruleOut(Bits bits) const
63{
64 if (!bits)
65 return true;
66
67 if ((bits & m_bits) != bits)
68 return true;
69
70 return false;
71}
72
73inline void TinyBloomFilter::reset()
74{
75 m_bits = 0;
76}
77
78} // namespace JSC
