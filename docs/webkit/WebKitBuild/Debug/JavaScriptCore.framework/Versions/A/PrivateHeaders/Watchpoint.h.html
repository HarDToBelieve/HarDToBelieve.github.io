97// Really unfortunately, we do not have the way to dispatch appropriate destructor in base class' destructor
98// based on enum type. If we call destructor explicitly in the base class, it ends up calling the base destructor
99// twice. C++20 allows this by using std::std::destroying_delete_t. But we are not using C++20 right now.
100//
101// Because we cannot dispatch destructors of derived classes in the destructor of the base class, what it means is,
102// 1. Calling Watchpoint::~Watchpoint directly is illegal.
103// 2. `delete watchpoint` where watchpoint is non-final derived class is illegal. If watchpoint is final derived class, it works.
104// 3. If we really want to do (2), we need to call `watchpoint->destroy()` instead, and dispatch an appropriate destructor in Watchpoint::destroy.
105//
106// Luckily, none of our derived watchpoint classes have members which require destructors. So we do not dispatch
107// the destructor call to the drived class in the base class. If it becomes really required, we can introduce
108// a custom deleter for some classes which directly call "delete" to the allocated non-final Watchpoint class
109// (e.g. std::unique_ptr<Watchpoint>, RefPtr<Watchpoint>), and call Watchpoint::destroy instead of "delete"
110// operator. But since we do not require it for now, we are doing the simplest thing.
111#define JSC_WATCHPOINT_TYPES_WITHOUT_JIT(macro) \
112 macro(AdaptiveInferredPropertyValueStructure, AdaptiveInferredPropertyValueWatchpointBase::StructureWatchpoint) \
113 macro(AdaptiveInferredPropertyValueProperty, AdaptiveInferredPropertyValueWatchpointBase::PropertyWatchpoint) \
114 macro(CodeBlockJettisoning, CodeBlockJettisoningWatchpoint) \
115 macro(LLIntPrototypeLoadAdaptiveStructure, LLIntPrototypeLoadAdaptiveStructureWatchpoint) \
116 macro(FunctionRareDataAllocationProfileClearing, FunctionRareData::AllocationProfileClearingWatchpoint) \
117 macro(CachedSpecialPropertyAdaptiveStructure, CachedSpecialPropertyAdaptiveStructureWatchpoint)
118
119#if ENABLE(JIT)
120#define JSC_WATCHPOINT_TYPES_WITHOUT_DFG(macro) \
121 JSC_WATCHPOINT_TYPES_WITHOUT_JIT(macro) \
122 macro(StructureTransitionStructureStubClearing, StructureTransitionStructureStubClearingWatchpoint)
123
124#if ENABLE(DFG_JIT)
125#define JSC_WATCHPOINT_TYPES(macro) \
126 JSC_WATCHPOINT_TYPES_WITHOUT_DFG(macro) \
127 macro(AdaptiveStructure, DFG::AdaptiveStructureWatchpoint)
128#else
129#define JSC_WATCHPOINT_TYPES(macro) \
130 JSC_WATCHPOINT_TYPES_WITHOUT_DFG(macro)
131#endif
132
133#else
134#define JSC_WATCHPOINT_TYPES(macro) \
135 JSC_WATCHPOINT_TYPES_WITHOUT_JIT(macro)
136#endif

171// Make sure that the state can be represented in 2 bits.
172enum WatchpointState : uint8_t {
173 ClearWatchpoint = 0,
174 IsWatched = 1,
175 IsInvalidated = 2

184class WatchpointSet : public ThreadSafeRefCounted<WatchpointSet> {
185 WTF_MAKE_STRUCT_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(WatchpointSet);
186 friend class LLIntOffsetsExtractor;
187 friend class DeferredWatchpointFire;
188public:
189 // FIXME: In many cases, it would be amazing if this *did* fire the watchpoints. I suspect that
190 // this might be hard to get right, but still, it might be awesome.
191 JS_EXPORT_PRIVATE ~WatchpointSet(); // Note that this will not fire any of the watchpoints; if you need to know when a WatchpointSet dies then you need a separate mechanism for this.
192
193 static Ref<WatchpointSet> create(WatchpointState state)
194 {
195 return adoptRef(*new WatchpointSet(state));
196 }
197
198 // Fast way of getting the state, which only works from the main thread.
199 WatchpointState stateOnJSThread() const
200 {
201 return static_cast<WatchpointState>(m_state);
202 }
203
204 // It is safe to call this from another thread. It may return an old
205 // state. Guarantees that if *first* read the state() of the thing being
206 // watched and it returned IsWatched and *second* you actually read its
207 // value then it's safe to assume that if the state being watched changes
208 // then also the watchpoint state() will change to IsInvalidated.
209 WatchpointState state() const
210 {
211 WatchpointState result = static_cast<WatchpointState>(m_state);
212 return result;
213 }
214
215 // It is safe to call this from another thread. It may return true
216 // even if the set actually had been invalidated, but that ought to happen
217 // only in the case of races, and should be rare. Guarantees that if you
218 // call this after observing something that must imply that the set is
219 // invalidated, then you will see this return false. This is ensured by
220 // issuing a load-load fence prior to querying the state.
221 bool isStillValid() const
222 {
223 return state() != IsInvalidated;
224 }
225 // Fast way of testing isStillValid(), which only works from the main thread.
226 bool isStillValidOnJSThread() const
227 {
228 return stateOnJSThread() != IsInvalidated;
229 }
230 // Like isStillValid(), may be called from another thread.
231 bool hasBeenInvalidated() const { return !isStillValid(); }
232
233 // As a convenience, this will ignore 0. That's because code paths in the DFG
234 // that create speculation watchpoints may choose to bail out if speculation
235 // had already been terminated.
236 void add(Watchpoint*);
237
238 // Force the watchpoint set to behave as if it was being watched even if no
239 // watchpoints have been installed. This will result in invalidation if the
240 // watchpoint would have fired. That's a pretty good indication that you
241 // probably don't want to set watchpoints, since we typically don't want to
242 // set watchpoints that we believe will actually be fired.
243 void startWatching()
244 {
245 ASSERT(m_state != IsInvalidated);
246 if (m_state == IsWatched)
247 return;
248 WTF::storeStoreFence();
249 m_state = IsWatched;
250 WTF::storeStoreFence();
251 }
252
253 template <typename T>
254 void fireAll(VM& vm, T& fireDetails)
255 {
256 if (LIKELY(m_state != IsWatched))
257 return;
258 fireAllSlow(vm, fireDetails);
259 }
260
261 void touch(VM& vm, const FireDetail& detail)
262 {
263 if (state() == ClearWatchpoint)
264 startWatching();
265 else
266 fireAll(vm, detail);
267 }
268
269 void touch(VM& vm, const char* reason)
270 {
271 touch(vm, StringFireDetail(reason));
272 }
273
274 void invalidate(VM& vm, const FireDetail& detail)
275 {
276 if (state() == IsWatched)
277 fireAll(vm, detail);
278 m_state = IsInvalidated;
279 }
280
281 void invalidate(VM& vm, const char* reason)
282 {
283 invalidate(vm, StringFireDetail(reason));
284 }
285
286 bool isBeingWatched() const
287 {
288 return m_setIsNotEmpty;
289 }
290
291 int8_t* addressOfState() { return &m_state; }
292 static ptrdiff_t offsetOfState() { return OBJECT_OFFSETOF(WatchpointSet, m_state); }
293 int8_t* addressOfSetIsNotEmpty() { return &m_setIsNotEmpty; }
294
295 JS_EXPORT_PRIVATE void fireAllSlow(VM&, const FireDetail&); // Call only if you've checked isWatched.
296 JS_EXPORT_PRIVATE void fireAllSlow(VM&, DeferredWatchpointFire* deferredWatchpoints); // Ditto.
297 JS_EXPORT_PRIVATE void fireAllSlow(VM&, const char* reason); // Ditto.
298
299protected:
300 JS_EXPORT_PRIVATE WatchpointSet(WatchpointState);
301
302private:
303 void fireAllWatchpoints(VM&, const FireDetail&);
304 void take(WatchpointSet* other);
305
306 friend class InlineWatchpointSet;
307
308 int8_t m_state;
309 int8_t m_setIsNotEmpty;
310
311 SentinelLinkedList<Watchpoint, PackedRawSentinelNode<Watchpoint>> m_set;
312};

314// InlineWatchpointSet is a low-overhead, non-copyable watchpoint set in which
315// it is not possible to quickly query whether it is being watched in a single
316// branch. There is a fairly simple tradeoff between WatchpointSet and
317// InlineWatchpointSet:
318//
319// Do you have to emit JIT code that rapidly tests whether the watchpoint set
320// is being watched? If so, use WatchpointSet.
321//
322// Do you need multiple parties to have pointers to the same WatchpointSet?
323// If so, use WatchpointSet.
324//
325// Do you have to allocate a lot of watchpoint sets? If so, use
326// InlineWatchpointSet unless you answered "yes" to the previous questions.
327//
328// InlineWatchpointSet will use just one pointer-width word of memory unless
329// you actually add watchpoints to it, in which case it internally inflates
330// to a pointer to a WatchpointSet, and transfers its state to the
331// WatchpointSet.
332
333class InlineWatchpointSet {
334 WTF_MAKE_NONCOPYABLE(InlineWatchpointSet);
335public:
336 InlineWatchpointSet(WatchpointState state)
337 : m_data(encodeState(state))
338 {
339 }
340
341 ~InlineWatchpointSet()
342 {
343 if (isThin())
344 return;
345 freeFat();
346 }
347
348 // Fast way of getting the state, which only works from the main thread.
349 WatchpointState stateOnJSThread() const
350 {
351 uintptr_t data = m_data;
352 if (isFat(data))
353 return fat(data)->stateOnJSThread();
354 return decodeState(data);
355 }
356
357 // It is safe to call this from another thread. It may return a prior state,
358 // but that should be fine since you should only perform actions based on the
359 // state if you also add a watchpoint.
360 WatchpointState state() const
361 {
362 uintptr_t data = m_data;
363 if (isFat(data))
364 return fat(data)->state();
365 return decodeState(data);
366 }
367
368 // It is safe to call this from another thread. It may return false
369 // even if the set actually had been invalidated, but that ought to happen
370 // only in the case of races, and should be rare.
371 bool hasBeenInvalidated() const
372 {
373 return state() == IsInvalidated;
374 }
375
376 // Like hasBeenInvalidated(), may be called from another thread.
377 bool isStillValid() const
378 {
379 return !hasBeenInvalidated();
380 }
381
382 void add(Watchpoint*);
383
384 void startWatching()
385 {
386 if (isFat()) {
387 fat()->startWatching();
388 return;
389 }
390 ASSERT(decodeState(m_data) != IsInvalidated);
391 m_data = encodeState(IsWatched);
392 }
393
394 template <typename T>
395 void fireAll(VM& vm, T fireDetails)
396 {
397 if (isFat()) {
398 fat()->fireAll(vm, fireDetails);
399 return;
400 }
401 if (decodeState(m_data) == ClearWatchpoint)
402 return;
403 m_data = encodeState(IsInvalidated);
404 WTF::storeStoreFence();
405 }
406
407 void invalidate(VM& vm, const FireDetail& detail)
408 {
409 if (isFat())
410 fat()->invalidate(vm, detail);
411 else
412 m_data = encodeState(IsInvalidated);
413 }
414
415 JS_EXPORT_PRIVATE void fireAll(VM&, const char* reason);
416
417 void touch(VM& vm, const FireDetail& detail)
418 {
419 if (isFat()) {
420 fat()->touch(vm, detail);
421 return;
422 }
423 uintptr_t data = m_data;
424 if (decodeState(data) == IsInvalidated)
425 return;
426 WTF::storeStoreFence();
427 if (decodeState(data) == ClearWatchpoint)
428 m_data = encodeState(IsWatched);
429 else
430 m_data = encodeState(IsInvalidated);
431 WTF::storeStoreFence();
432 }
433
434 void touch(VM& vm, const char* reason)
435 {
436 touch(vm, StringFireDetail(reason));
437 }

439 // Note that for any watchpoint that is visible from the DFG, it would be incorrect to write code like:
440 //
441 // if (w.isBeingWatched())
442 // w.fireAll()
443 //
444 // Concurrently to this, the DFG could do:
445 //
446 // if (w.isStillValid())
447 // perform optimizations;
448 // if (!w.isStillValid())
449 // retry compilation;
450 //
451 // Note that the DFG algorithm is widespread, and sound, because fireAll() and invalidate() will leave
452 // the watchpoint in a !isStillValid() state. Hence, if fireAll() or invalidate() interleaved between
453 // the first isStillValid() check and the second one, then it would simply cause the DFG to retry
454 // compilation later.
455 //
456 // But, if you change some piece of state that the DFG might optimize for, but invalidate the
457 // watchpoint by doing:
458 //
459 // if (w.isBeingWatched())
460 // w.fireAll()
461 //
462 // then the DFG would never know that you invalidated state between the two checks.
463 //
464 // There are two ways to work around this:
465 //
466 // - Call fireAll() without a isBeingWatched() check. Then, the DFG will know that the watchpoint has
467 // been invalidated when it does its second check.
468 //
469 // - Do not expose the watchpoint set to the DFG directly, and have your own way of validating whether
470 // the assumptions that the DFG thread used are still valid when the DFG code is installed.
471 bool isBeingWatched() const
472 {
473 if (isFat())
474 return fat()->isBeingWatched();
475 return false;
476 }
477
478 // We expose this because sometimes a client knows its about to start
479 // watching this InlineWatchpointSet, hence it'll become inflated regardless.
480 // Such clients may find it useful to have a WatchpointSet* pointer, for example,
481 // if they collect a Vector of WatchpointSet*.
482 WatchpointSet* inflate()
483 {
484 if (LIKELY(isFat()))
485 return fat();
486 return inflateSlow();
487 }
488
489private:
490 static constexpr uintptr_t IsThinFlag = 1;
491 static constexpr uintptr_t StateMask = 6;
492 static constexpr uintptr_t StateShift = 1;
493
494 static bool isThin(uintptr_t data) { return data & IsThinFlag; }
495 static bool isFat(uintptr_t data) { return !isThin(data); }
496
497 static WatchpointState decodeState(uintptr_t data)
498 {
499 ASSERT(isThin(data));
500 return static_cast<WatchpointState>((data & StateMask) >> StateShift);
501 }
502
503 static uintptr_t encodeState(WatchpointState state)
504 {
505 return (static_cast<uintptr_t>(state) << StateShift) | IsThinFlag;
506 }
507
508 bool isThin() const { return isThin(m_data); }
509 bool isFat() const { return isFat(m_data); };
510
511 static WatchpointSet* fat(uintptr_t data)
512 {
513 return bitwise_cast<WatchpointSet*>(data);
514 }
515
516 WatchpointSet* fat()
517 {
518 ASSERT(isFat());
519 return fat(m_data);
520 }
521
522 const WatchpointSet* fat() const
523 {
524 ASSERT(isFat());
525 return fat(m_data);
526 }
527
528 JS_EXPORT_PRIVATE WatchpointSet* inflateSlow();
529 JS_EXPORT_PRIVATE void freeFat();
530
531 uintptr_t m_data;
532};
