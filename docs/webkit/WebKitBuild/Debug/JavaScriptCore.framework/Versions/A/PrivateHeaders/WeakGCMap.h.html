28#include "DeferGC.h"
29#include "Weak.h"
30#include "WeakGCHashTable.h"
31#include <wtf/HashMap.h>

33namespace JSC {
34
35// A HashMap with Weak<JSCell> values, which automatically removes values once they're garbage collected.
36
37template<typename KeyArg, typename ValueArg, typename HashArg = DefaultHash<KeyArg>, typename KeyTraitsArg = HashTraits<KeyArg>>
38class WeakGCMap final : public WeakGCHashTable {
39 WTF_MAKE_FAST_ALLOCATED;
40 typedef Weak<ValueArg> ValueType;
41 typedef HashMap<KeyArg, ValueType, HashArg, KeyTraitsArg> HashMapType;
42
43public:
44 typedef typename HashMapType::KeyType KeyType;
45 typedef typename HashMapType::AddResult AddResult;
46 typedef typename HashMapType::iterator iterator;
47 typedef typename HashMapType::const_iterator const_iterator;
48
49 explicit WeakGCMap(VM&);
50 ~WeakGCMap() final;
51
52 ValueArg* get(const KeyType& key) const
53 {
54 return m_map.get(key);
55 }
56
57 AddResult set(const KeyType& key, ValueType value)
58 {
59 return m_map.set(key, WTFMove(value));
60 }
61
62 template<typename Functor>
63 ValueArg* ensureValue(const KeyType& key, Functor&& functor)
64 {
65 // If functor invokes GC, GC can prune WeakGCMap, and manipulate HashMap while we are touching it in ensure function.
66 // The functor must not invoke GC.
67 DisallowGC disallowGC;
68 AddResult result = m_map.ensure(key, std::forward<Functor>(functor));
69 ValueArg* value = result.iterator->value.get();
70 if (!result.isNewEntry && !value) {
71 value = functor();
72 result.iterator->value = WTFMove(value);
73 }
74 return value;
75 }
76
77 bool remove(const KeyType& key)
78 {
79 return m_map.remove(key);
80 }
81
82 void clear()
83 {
84 m_map.clear();
85 }
86
87 bool isEmpty() const
88 {
89 const_iterator it = m_map.begin();
90 const_iterator end = m_map.end();
91 while (it != end) {
92 if (it->value)
93 return true;
94 }
95 return false;
96 }
97
98 inline iterator find(const KeyType& key);
99
100 inline const_iterator find(const KeyType& key) const;
101
102 inline bool contains(const KeyType& key) const;
103
104 void pruneStaleEntries() final;
105
106private:
107 HashMapType m_map;
108 VM& m_vm;
109};

111} // namespace JSC
