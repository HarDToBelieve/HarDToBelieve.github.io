32namespace JSC {

34template<typename ValueArg, typename HashArg, typename TraitsArg>
35inline WeakGCSet<ValueArg, HashArg, TraitsArg>::WeakGCSet(VM& vm)
36 : m_vm(vm)
37{
38 vm.heap.registerWeakGCHashTable(this);
39}
40
41template<typename ValueArg, typename HashArg, typename TraitsArg>
42inline WeakGCSet<ValueArg, HashArg, TraitsArg>::~WeakGCSet()
43{
44 m_vm.heap.unregisterWeakGCHashTable(this);
45}
46
47template<typename ValueArg, typename HashArg, typename TraitsArg>
48inline typename WeakGCSet<ValueArg, HashArg, TraitsArg>::iterator WeakGCSet<ValueArg, HashArg, TraitsArg>::find(const ValueType& value)
49{
50 iterator it = m_set.find(value);
51 iterator end = m_set.end();
52 if (it != end && !*it) // Found a zombie value.
53 return end;
54 return it;
55}
56
57template<typename ValueArg, typename HashArg, typename TraitsArg>
58inline typename WeakGCSet<ValueArg, HashArg, TraitsArg>::const_iterator WeakGCSet<ValueArg, HashArg, TraitsArg>::find(const ValueType& value) const
59{
60 return const_cast<WeakGCSet*>(this)->find(value);
61}
62
63template<typename ValueArg, typename HashArg, typename TraitsArg>
64inline bool WeakGCSet<ValueArg, HashArg, TraitsArg>::contains(const ValueType& value) const
65{
66 return find(value) != m_set.end();
67}
68
69template<typename ValueArg, typename HashArg, typename TraitsArg>
70NEVER_INLINE void WeakGCSet<ValueArg, HashArg, TraitsArg>::pruneStaleEntries()
71{
72 m_set.removeIf([](auto& entry) {
73 return !entry;
74 });
75}
