33inline WeakImpl* WeakSet::allocate(JSValue jsValue, WeakHandleOwner* weakHandleOwner, void* context)
34{
35 CellContainer container = jsValue.asCell()->cellContainer();
36 WeakSet& weakSet = container.weakSet();
37 WeakBlock::FreeCell* allocator = weakSet.m_allocator;
38 if (UNLIKELY(!allocator))
39 allocator = weakSet.findAllocator(container);
40 weakSet.m_allocator = allocator->next;
41
42 WeakImpl* weakImpl = WeakBlock::asWeakImpl(allocator);
43 return new (NotNull, weakImpl) WeakImpl(jsValue, weakHandleOwner, context);
44}
45
46inline void WeakBlock::finalize(WeakImpl* weakImpl)
47{
48 ASSERT(weakImpl->state() == WeakImpl::Dead);
49 weakImpl->setState(WeakImpl::Finalized);
50 WeakHandleOwner* weakHandleOwner = weakImpl->weakHandleOwner();
51 if (!weakHandleOwner)
52 return;
53 weakHandleOwner->finalize(Handle<Unknown>::wrapSlot(&const_cast<JSValue&>(weakImpl->jsValue())), weakImpl->context());
54}
