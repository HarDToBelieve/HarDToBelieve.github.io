47enum class JITFailureReason : uint8_t {
48 DecodeSurrogatePair,
49 BackReference,
50 ForwardReference,
51 VariableCountedParenthesisWithNonZeroMinimum,
52 ParenthesizedSubpattern,
53 FixedCountParenthesizedSubpattern,
54 ParenthesisNestedTooDeep,
55 ExecutableMemoryAllocationFailure,
56};

58class MatchingContextHolder {
59 WTF_FORBID_HEAP_ALLOCATION;
60public:
61 MatchingContextHolder(VM&, YarrCodeBlock*, MatchFrom matchFrom = MatchFrom::VMThread);
62 ~MatchingContextHolder();
63
64 static ptrdiff_t offsetOfStackLimit() { return OBJECT_OFFSETOF(MatchingContextHolder, m_stackLimit); }
65#if ENABLE(YARR_JIT_ALL_PARENS_EXPRESSIONS)
66 static ptrdiff_t offsetOfPatternContextBuffer() { return OBJECT_OFFSETOF(MatchingContextHolder, m_patternContextBuffer); }
67 static ptrdiff_t offsetOfPatternContextBufferSize() { return OBJECT_OFFSETOF(MatchingContextHolder, m_patternContextBufferSize); }
68#endif
69
70private:
71 VM& m_vm;
72 void* m_stackLimit;
73#if ENABLE(YARR_JIT_ALL_PARENS_EXPRESSIONS)
74 void* m_patternContextBuffer { nullptr };
75 unsigned m_patternContextBufferSize { 0 };
76#endif
77};

84class YarrCodeBlock {
85 WTF_MAKE_FAST_ALLOCATED;
86 WTF_MAKE_NONCOPYABLE(YarrCodeBlock);
87
88 using YarrJITCode8 = EncodedMatchResult (*)(const LChar* input, unsigned start, unsigned length, int* output, MatchingContextHolder& matchingContext) YARR_CALL;
89 using YarrJITCode16 = EncodedMatchResult (*)(const UChar* input, unsigned start, unsigned length, int* output, MatchingContextHolder& matchingContext) YARR_CALL;
90 using YarrJITCodeMatchOnly8 = EncodedMatchResult (*)(const LChar* input, unsigned start, unsigned length, void*, MatchingContextHolder& matchingContext) YARR_CALL;
91 using YarrJITCodeMatchOnly16 = EncodedMatchResult (*)(const UChar* input, unsigned start, unsigned length, void*, MatchingContextHolder& matchingContext) YARR_CALL;
92
93public:
94 YarrCodeBlock() = default;
95
96 void setFallBackWithFailureReason(JITFailureReason failureReason) { m_failureReason = failureReason; }
97 std::optional<JITFailureReason> failureReason() { return m_failureReason; }
98
99 bool has8BitCode() { return m_ref8.size(); }
100 bool has16BitCode() { return m_ref16.size(); }
101 void set8BitCode(MacroAssemblerCodeRef<Yarr8BitPtrTag> ref) { m_ref8 = ref; }
102 void set16BitCode(MacroAssemblerCodeRef<Yarr16BitPtrTag> ref) { m_ref16 = ref; }
103
104 bool has8BitCodeMatchOnly() { return m_matchOnly8.size(); }
105 bool has16BitCodeMatchOnly() { return m_matchOnly16.size(); }
106 void set8BitCodeMatchOnly(MacroAssemblerCodeRef<YarrMatchOnly8BitPtrTag> matchOnly) { m_matchOnly8 = matchOnly; }
107 void set16BitCodeMatchOnly(MacroAssemblerCodeRef<YarrMatchOnly16BitPtrTag> matchOnly) { m_matchOnly16 = matchOnly; }
108
109 bool usesPatternContextBuffer() { return m_usesPatternContextBuffer; }
110#if ENABLE(YARR_JIT_ALL_PARENS_EXPRESSIONS)
111 void setUsesPatternContextBuffer() { m_usesPatternContextBuffer = true; }
112#endif
113
114 MatchResult execute(const LChar* input, unsigned start, unsigned length, int* output, MatchingContextHolder& matchingContext)
115 {
116 ASSERT(has8BitCode());
117#if CPU(ARM64E)
118 if (Options::useJITCage())
119 return MatchResult(vmEntryToYarrJIT(input, start, length, output, &matchingContext, retagCodePtr<Yarr8BitPtrTag, YarrEntryPtrTag>(m_ref8.code().executableAddress())));
120#endif
121 return MatchResult(untagCFunctionPtr<YarrJITCode8, Yarr8BitPtrTag>(m_ref8.code().executableAddress())(input, start, length, output, matchingContext));
122 }
123
124 MatchResult execute(const UChar* input, unsigned start, unsigned length, int* output, MatchingContextHolder& matchingContext)
125 {
126 ASSERT(has16BitCode());
127#if CPU(ARM64E)
128 if (Options::useJITCage())
129 return MatchResult(vmEntryToYarrJIT(input, start, length, output, &matchingContext, retagCodePtr<Yarr16BitPtrTag, YarrEntryPtrTag>(m_ref16.code().executableAddress())));
130#endif
131 return MatchResult(untagCFunctionPtr<YarrJITCode16, Yarr16BitPtrTag>(m_ref16.code().executableAddress())(input, start, length, output, matchingContext));
132 }
133
134 MatchResult execute(const LChar* input, unsigned start, unsigned length, MatchingContextHolder& matchingContext)
135 {
136 ASSERT(has8BitCodeMatchOnly());
137#if CPU(ARM64E)
138 if (Options::useJITCage())
139 return MatchResult(vmEntryToYarrJIT(input, start, length, nullptr, &matchingContext, retagCodePtr<YarrMatchOnly8BitPtrTag, YarrEntryPtrTag>(m_matchOnly8.code().executableAddress())));
140#endif
141 return MatchResult(untagCFunctionPtr<YarrJITCodeMatchOnly8, YarrMatchOnly8BitPtrTag>(m_matchOnly8.code().executableAddress())(input, start, length, nullptr, matchingContext));
142 }
143
144 MatchResult execute(const UChar* input, unsigned start, unsigned length, MatchingContextHolder& matchingContext)
145 {
146 ASSERT(has16BitCodeMatchOnly());
147#if CPU(ARM64E)
148 if (Options::useJITCage())
149 return MatchResult(vmEntryToYarrJIT(input, start, length, nullptr, &matchingContext, retagCodePtr<YarrMatchOnly16BitPtrTag, YarrEntryPtrTag>(m_matchOnly16.code().executableAddress())));
150#endif
151 return MatchResult(untagCFunctionPtr<YarrJITCodeMatchOnly16, YarrMatchOnly16BitPtrTag>(m_matchOnly16.code().executableAddress())(input, start, length, nullptr, matchingContext));
152 }
153
154#if ENABLE(REGEXP_TRACING)
155 void *get8BitMatchOnlyAddr()
156 {
157 if (!has8BitCodeMatchOnly())
158 return 0;
159
160 return m_matchOnly8.code().executableAddress();
161 }
162
163 void *get16BitMatchOnlyAddr()
164 {
165 if (!has16BitCodeMatchOnly())
166 return 0;
167
168 return m_matchOnly16.code().executableAddress();
169 }
170
171 void *get8BitMatchAddr()
172 {
173 if (!has8BitCode())
174 return 0;
175
176 return m_ref8.code().executableAddress();
177 }
178
179 void *get16BitMatchAddr()
180 {
181 if (!has16BitCode())
182 return 0;
183
184 return m_ref16.code().executableAddress();
185 }
186#endif
187
188 size_t size() const
189 {
190 return m_ref8.size() + m_ref16.size() + m_matchOnly8.size() + m_matchOnly16.size();
191 }
192
193 void clear()
194 {
195 m_ref8 = MacroAssemblerCodeRef<Yarr8BitPtrTag>();
196 m_ref16 = MacroAssemblerCodeRef<Yarr16BitPtrTag>();
197 m_matchOnly8 = MacroAssemblerCodeRef<YarrMatchOnly8BitPtrTag>();
198 m_matchOnly16 = MacroAssemblerCodeRef<YarrMatchOnly16BitPtrTag>();
199 m_failureReason = std::nullopt;
200 }
201
202private:
203 MacroAssemblerCodeRef<Yarr8BitPtrTag> m_ref8;
204 MacroAssemblerCodeRef<Yarr16BitPtrTag> m_ref16;
205 MacroAssemblerCodeRef<YarrMatchOnly8BitPtrTag> m_matchOnly8;
206 MacroAssemblerCodeRef<YarrMatchOnly16BitPtrTag> m_matchOnly16;
207 bool m_usesPatternContextBuffer { false };
208 std::optional<JITFailureReason> m_failureReason;
209};
