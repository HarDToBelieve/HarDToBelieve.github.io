325class AXIsolatedTree : public ThreadSafeRefCounted<AXIsolatedTree> {
326 WTF_MAKE_NONCOPYABLE(AXIsolatedTree); WTF_MAKE_FAST_ALLOCATED;
327 friend WTF::TextStream& operator<<(WTF::TextStream&, AXIsolatedTree&);
328public:
329 static Ref<AXIsolatedTree> create(AXObjectCache*);
330 virtual ~AXIsolatedTree();
331
332 static void removeTreeForPageID(PageIdentifier);
333
334 static RefPtr<AXIsolatedTree> treeForPageID(PageIdentifier);
335 static RefPtr<AXIsolatedTree> treeForID(AXIsolatedTreeID);
336 AXObjectCache* axObjectCache() const;
337
338 RefPtr<AXIsolatedObject> rootNode();
339 RefPtr<AXIsolatedObject> focusedNode();
340 RefPtr<AXIsolatedObject> nodeForID(AXID) const;
341 Vector<RefPtr<AXCoreObject>> objectsForIDs(const Vector<AXID>&) const;
342 Vector<AXID> idsForObjects(const Vector<RefPtr<AXCoreObject>>&) const;
343
344 struct NodeChange {
345 Ref<AXIsolatedObject> isolatedObject;
346 RetainPtr<AccessibilityObjectWrapper> wrapper;
347 };
348
349 void generateSubtree(AXCoreObject&, AXCoreObject*, bool attachWrapper);
350 void updateNode(AXCoreObject&);
351 void updateNodeProperty(const AXCoreObject&, AXPropertyName);
352 void updateSubtree(AXCoreObject&);
353 void updateChildren(AXCoreObject&);
354
355 // Removes the given node leaving all descendants alone.
356 void removeNode(AXID);
357 // Removes the given node and all its descendants.
358 void removeSubtree(AXID);
359
360 // Both setRootNodeID and setFocusedNodeID are called during the generation
361 // of the IsolatedTree.
362 // Focused node updates in AXObjectCache use setFocusNodeID.
363 void setRootNode(AXIsolatedObject*) WTF_REQUIRES_LOCK(m_changeLogLock);
364 void setFocusedNodeID(AXID);
365
366 // Called on AX thread from WebAccessibilityObjectWrapper methods.
367 // During layout tests, it is called on the main thread.
368 void applyPendingChanges();
369
370 AXIsolatedTreeID treeID() const { return m_treeID; }
371
372private:
373 AXIsolatedTree(AXObjectCache*);
374 void clear();
375
376 static Lock s_cacheLock;
377 static HashMap<AXIsolatedTreeID, Ref<AXIsolatedTree>>& treeIDCache() WTF_REQUIRES_LOCK(s_cacheLock);
378 static HashMap<PageIdentifier, Ref<AXIsolatedTree>>& treePageCache() WTF_REQUIRES_LOCK(s_cacheLock);
379
380 // Call on main thread
381 Ref<AXIsolatedObject> createSubtree(AXCoreObject&, AXID parentID, bool attachWrapper);
382 // Called on main thread to update both m_nodeMap and m_pendingChildrenUpdates.
383 void updateChildrenIDs(AXID parentID, Vector<AXID>&& childrenIDs) WTF_REQUIRES_LOCK(m_changeLogLock);
384
385 AXIsolatedTreeID m_treeID;
386 AXObjectCache* m_axObjectCache { nullptr };
387 bool m_usedOnAXThread { true };
388
389 // Only accessed on main thread.
390 HashMap<AXID, Vector<AXID>> m_nodeMap;
391 // Only accessed on AX thread requesting data.
392 HashMap<AXID, Ref<AXIsolatedObject>> m_readerThreadNodeMap;
393
394 // Written to by main thread under lock, accessed and applied by AX thread.
395 RefPtr<AXIsolatedObject> m_rootNode WTF_GUARDED_BY_LOCK(m_changeLogLock);
396 Vector<NodeChange> m_pendingAppends WTF_GUARDED_BY_LOCK(m_changeLogLock); // Nodes to be added to the tree and platform-wrapped.
397 Vector<AXPropertyChange> m_pendingPropertyChanges WTF_GUARDED_BY_LOCK(m_changeLogLock);
398 Vector<AXID> m_pendingNodeRemovals WTF_GUARDED_BY_LOCK(m_changeLogLock); // Nodes to be removed from the tree.
399 Vector<AXID> m_pendingSubtreeRemovals WTF_GUARDED_BY_LOCK(m_changeLogLock); // Nodes whose subtrees are to be removed from the tree.
400 Vector<std::pair<AXID, Vector<AXID>>> m_pendingChildrenUpdates WTF_GUARDED_BY_LOCK(m_changeLogLock);
401 AXID m_pendingFocusedNodeID WTF_GUARDED_BY_LOCK(m_changeLogLock) { InvalidAXID };
402 AXID m_focusedNodeID { InvalidAXID };
403 Lock m_changeLogLock;
404};
405
406inline AXObjectCache* AXIsolatedTree::axObjectCache() const
407{
408 ASSERT(isMainThread());
409 return m_axObjectCache;
410}
