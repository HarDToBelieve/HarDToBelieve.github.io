26#pragma once
27
28namespace WebCore {
29
30enum AXTextStateChangeType {
31 AXTextStateChangeTypeUnknown,
32 AXTextStateChangeTypeEdit,
33 AXTextStateChangeTypeSelectionMove,
34 AXTextStateChangeTypeSelectionExtend,
35 AXTextStateChangeTypeSelectionBoundary
36};
37
38enum AXTextEditType {
39 AXTextEditTypeUnknown,
40 AXTextEditTypeDelete, // Generic text delete
41 AXTextEditTypeInsert, // Generic text insert
42 AXTextEditTypeTyping, // Insert via typing
43 AXTextEditTypeDictation, // Insert via dictation
44 AXTextEditTypeCut, // Delete via Cut
45 AXTextEditTypePaste, // Insert via Paste
46 AXTextEditTypeAttributesChange // Change font, style, alignment, color, etc.
47};
48
49enum AXTextSelectionDirection {
50 AXTextSelectionDirectionUnknown,
51 AXTextSelectionDirectionBeginning,
52 AXTextSelectionDirectionEnd,
53 AXTextSelectionDirectionPrevious,
54 AXTextSelectionDirectionNext,
55 AXTextSelectionDirectionDiscontiguous
56};
57
58enum AXTextSelectionGranularity {
59 AXTextSelectionGranularityUnknown,
60 AXTextSelectionGranularityCharacter,
61 AXTextSelectionGranularityWord,
62 AXTextSelectionGranularityLine,
63 AXTextSelectionGranularitySentence,
64 AXTextSelectionGranularityParagraph,
65 AXTextSelectionGranularityPage,
66 AXTextSelectionGranularityDocument,
67 AXTextSelectionGranularityAll // All granularity represents the action of selecting the whole document as a single action. Extending selection by some other granularity until it encompasses the whole document will not result in a all granularity notification.
68};
69
70struct AXTextSelection {
71 AXTextSelectionDirection direction;
72 AXTextSelectionGranularity granularity;
73 bool focusChange;
74};
75
76struct AXTextStateChangeIntent {
77 AXTextStateChangeType type;
78 union {
79 AXTextSelection selection;
80 AXTextEditType change;
81 };
82
83 AXTextStateChangeIntent(AXTextStateChangeType type = AXTextStateChangeTypeUnknown, AXTextSelection selection = AXTextSelection())
84 : type(type)
85 , selection(selection)
86 { }
87
88 AXTextStateChangeIntent(AXTextEditType change)
89 : type(AXTextStateChangeTypeEdit)
90 , change(change)
91 { }
92};
93
94} // namespace WebCore
