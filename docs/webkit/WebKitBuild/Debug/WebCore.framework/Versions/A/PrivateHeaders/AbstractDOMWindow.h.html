34namespace WebCore {
35
36class AbstractFrame;
37
38// FIXME: Rename DOMWindow to LocalWindow and AbstractDOMWindow to DOMWindow.
39class AbstractDOMWindow : public RefCounted<AbstractDOMWindow>, public EventTargetWithInlineData {
40 WTF_MAKE_ISO_ALLOCATED(AbstractDOMWindow);
41public:
42 virtual ~AbstractDOMWindow();
43
44 static HashMap<GlobalWindowIdentifier, AbstractDOMWindow*>& allWindows();
45
46 const GlobalWindowIdentifier& identifier() const { return m_identifier; }
47
48 virtual AbstractFrame* frame() const = 0;
49
50 virtual bool isLocalDOMWindow() const = 0;
51 virtual bool isRemoteDOMWindow() const = 0;
52
53 using RefCounted::ref;
54 using RefCounted::deref;
55
56protected:
57 explicit AbstractDOMWindow(GlobalWindowIdentifier&&);
58
59 EventTargetInterface eventTargetInterface() const final { return DOMWindowEventTargetInterfaceType; }
60 void refEventTarget() final { ref(); }
61 void derefEventTarget() final { deref(); }
62
63private:
64 GlobalWindowIdentifier m_identifier;
65};
