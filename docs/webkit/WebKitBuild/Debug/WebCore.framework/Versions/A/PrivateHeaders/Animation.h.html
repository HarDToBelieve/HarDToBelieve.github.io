25#pragma once
26
27#include "CSSPropertyNames.h"
28#include "RenderStyleConstants.h"
29#include "StyleScopeOrdinal.h"
30#include "TimingFunction.h"
31
32namespace WebCore {
33
34class Animation : public RefCounted<Animation> {
35public:
36 WEBCORE_EXPORT ~Animation();
37
38 static Ref<Animation> create() { return adoptRef(*new Animation); }
39 static Ref<Animation> create(const Animation& other) { return adoptRef(*new Animation(other)); }
40
41 bool isDelaySet() const { return m_delaySet; }
42 bool isDirectionSet() const { return m_directionSet; }
43 bool isDurationSet() const { return m_durationSet; }
44 bool isFillModeSet() const { return m_fillModeSet; }
45 bool isIterationCountSet() const { return m_iterationCountSet; }
46 bool isNameSet() const { return m_nameSet; }
47 bool isPlayStateSet() const { return m_playStateSet; }
48 bool isPropertySet() const { return m_propertySet; }
49 bool isTimingFunctionSet() const { return m_timingFunctionSet; }
50
51 // Flags this to be the special "none" animation (animation-name: none)
52 bool isNoneAnimation() const { return m_isNone; }
53
54 // We can make placeholder Animation objects to keep the comma-separated lists
55 // of properties in sync. isValidAnimation means this is not a placeholder.
56 bool isValidAnimation() const { return !m_isNone && !m_name.string.isEmpty(); }
57
58 bool isEmpty() const
59 {
60 return !m_directionSet && !m_durationSet && !m_fillModeSet
61 && !m_nameSet && !m_playStateSet && !m_iterationCountSet
62 && !m_delaySet && !m_timingFunctionSet && !m_propertySet;
63 }
64
65 bool isEmptyOrZeroDuration() const
66 {
67 return isEmpty() || (m_duration == 0 && m_delay <= 0);
68 }
69
70 void clearDelay() { m_delaySet = false; }
71 void clearDirection() { m_directionSet = false; }
72 void clearDuration() { m_durationSet = false; }
73 void clearFillMode() { m_fillModeSet = false; }
74 void clearIterationCount() { m_iterationCountSet = false; }
75 void clearName() { m_nameSet = false; }
76 void clearPlayState() { m_playStateSet = false; }
77 void clearProperty() { m_propertySet = false; }
78 void clearTimingFunction() { m_timingFunctionSet = false; }
79
80 void clearAll()
81 {
82 clearDelay();
83 clearDirection();
84 clearDuration();
85 clearFillMode();
86 clearIterationCount();
87 clearName();
88 clearPlayState();
89 clearProperty();
90 clearTimingFunction();
91 }
92
93 double delay() const { return m_delay; }
94
95 enum class TransitionMode : uint8_t {
96 All,
97 None,
98 SingleProperty,
99 UnknownProperty
100 };
101
102 struct TransitionProperty {
103 TransitionMode mode;
104 CSSPropertyID id;
105 };
106
107 enum AnimationDirection {
108 AnimationDirectionNormal,
109 AnimationDirectionAlternate,
110 AnimationDirectionReverse,
111 AnimationDirectionAlternateReverse
112 };
113
114 AnimationDirection direction() const { return static_cast<AnimationDirection>(m_direction); }
115 bool directionIsForwards() const { return m_direction == AnimationDirectionNormal || m_direction == AnimationDirectionAlternate; }
116
117 AnimationFillMode fillMode() const { return static_cast<AnimationFillMode>(m_fillMode); }
118
119 double duration() const { return m_duration; }
120 double playbackRate() const { return m_playbackRate; }
121
122 struct Name {
123 String string;
124 bool isIdentifier { false };
125 };
126
127 enum { IterationCountInfinite = -1 };
128 double iterationCount() const { return m_iterationCount; }
129 const Name& name() const { return m_name; }
130 Style::ScopeOrdinal nameStyleScopeOrdinal() const { return m_nameStyleScopeOrdinal; }
131 AnimationPlayState playState() const { return static_cast<AnimationPlayState>(m_playState); }
132 TransitionProperty property() const { return m_property; }
133 const String& unknownProperty() const { return m_unknownProperty; }
134 TimingFunction* timingFunction() const { return m_timingFunction.get(); }
135 TimingFunction* defaultTimingFunctionForKeyframes() const { return m_defaultTimingFunctionForKeyframes.get(); }
136
137 void setDelay(double c) { m_delay = c; m_delaySet = true; }
138 void setDirection(AnimationDirection d) { m_direction = d; m_directionSet = true; }
139 void setDuration(double d) { ASSERT(d >= 0); m_duration = d; m_durationSet = true; }
140 void setPlaybackRate(double d) { m_playbackRate = d; }
141 void setFillMode(AnimationFillMode f) { m_fillMode = static_cast<unsigned>(f); m_fillModeSet = true; }
142 void setIterationCount(double c) { m_iterationCount = c; m_iterationCountSet = true; }
143 void setName(const Name& name, Style::ScopeOrdinal scope = Style::ScopeOrdinal::Element)
144 {
145 m_name = name;
146 m_nameStyleScopeOrdinal = scope;
147 m_nameSet = true;
148 }
149 void setPlayState(AnimationPlayState d) { m_playState = static_cast<unsigned>(d); m_playStateSet = true; }
150 void setProperty(TransitionProperty t) { m_property = t; m_propertySet = true; }
151 void setUnknownProperty(const String& property) { m_unknownProperty = property; }
152 void setTimingFunction(RefPtr<TimingFunction>&& function) { m_timingFunction = WTFMove(function); m_timingFunctionSet = true; }
153 void setDefaultTimingFunctionForKeyframes(RefPtr<TimingFunction>&& function) { m_defaultTimingFunctionForKeyframes = WTFMove(function); }
154
155 void setIsNoneAnimation(bool n) { m_isNone = n; }
156
157 Animation& operator=(const Animation&);
158
159 // return true if all members of this class match (excluding m_next)
160 bool animationsMatch(const Animation&, bool matchProperties = true) const;
161
162 // return true every Animation in the chain (defined by m_next) match
163 bool operator==(const Animation& o) const { return animationsMatch(o); }
164 bool operator!=(const Animation& o) const { return !(*this == o); }
165
166 bool fillsBackwards() const { return m_fillModeSet && (fillMode() == AnimationFillMode::Backwards || fillMode() == AnimationFillMode::Both); }
167 bool fillsForwards() const { return m_fillModeSet && (fillMode() == AnimationFillMode::Forwards || fillMode() == AnimationFillMode::Both); }
168
169private:
170 WEBCORE_EXPORT Animation();
171 Animation(const Animation&);
172
173 // Packs with m_refCount from the base class.
174 TransitionProperty m_property { TransitionMode::All, CSSPropertyInvalid };
175
176 Name m_name;
177 String m_unknownProperty;
178 double m_iterationCount;
179 double m_delay;
180 double m_duration;
181 double m_playbackRate { 1 };
182 RefPtr<TimingFunction> m_timingFunction;
183 RefPtr<TimingFunction> m_defaultTimingFunctionForKeyframes;
184
185 Style::ScopeOrdinal m_nameStyleScopeOrdinal { Style::ScopeOrdinal::Element };
186
187 unsigned m_direction : 2; // AnimationDirection
188 unsigned m_fillMode : 2; // AnimationFillMode
189 unsigned m_playState : 2; // AnimationPlayState
190
191 bool m_delaySet : 1;
192 bool m_directionSet : 1;
193 bool m_durationSet : 1;
194 bool m_fillModeSet : 1;
195 bool m_iterationCountSet : 1;
196 bool m_nameSet : 1;
197 bool m_playStateSet : 1;
198 bool m_propertySet : 1;
199 bool m_timingFunctionSet : 1;
200
201 bool m_isNone : 1;
202
203public:
204 static double initialDelay() { return 0; }
205 static AnimationDirection initialDirection() { return AnimationDirectionNormal; }
206 static double initialDuration() { return 0; }
207 static AnimationFillMode initialFillMode() { return AnimationFillMode::None; }
208 static double initialIterationCount() { return 1.0; }
209 static const Name& initialName();
210 static AnimationPlayState initialPlayState() { return AnimationPlayState::Playing; }
211 static TransitionProperty initialProperty() { return { TransitionMode::All, CSSPropertyInvalid }; }
212 static Ref<TimingFunction> initialTimingFunction() { return CubicBezierTimingFunction::create(); }
213};
