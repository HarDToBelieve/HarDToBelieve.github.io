37using FramesPerSecond = unsigned;

39enum class ThrottlingReason : uint8_t {
40 VisuallyIdle = 1 << 0,
41 OutsideViewport = 1 << 1,
42 LowPowerMode = 1 << 2,
43 NonInteractedCrossOriginFrame = 1 << 3,
44};
45
46// Allow a little more than 60fps to make sure we can at least hit that frame rate.
47constexpr const Seconds FullSpeedAnimationInterval { 15_ms };
48// Allow a little more than 30fps to make sure we can at least hit that frame rate.
49constexpr const Seconds HalfSpeedThrottlingAnimationInterval { 30_ms };
50constexpr const Seconds AggressiveThrottlingAnimationInterval { 10_s };
51constexpr const int IntervalThrottlingFactor { 2 };
52
53constexpr const FramesPerSecond FullSpeedFramesPerSecond = 60;
54constexpr const FramesPerSecond HalfSpeedThrottlingFramesPerSecond = 30;

56WEBCORE_EXPORT FramesPerSecond framesPerSecondNearestFullSpeed(FramesPerSecond);

58// This will return std::nullopt if throttling results in a frequency < 1fps.
59WEBCORE_EXPORT std::optional<FramesPerSecond> preferredFramesPerSecond(OptionSet<ThrottlingReason>, std::optional<FramesPerSecond> nominalFramesPerSecond, bool preferFrameRatesNear60FPS);

61WEBCORE_EXPORT Seconds preferredFrameInterval(OptionSet<ThrottlingReason>, std::optional<FramesPerSecond> nominalFramesPerSecond, bool preferFrameRatesNear60FPS);

63WEBCORE_EXPORT FramesPerSecond preferredFramesPerSecondFromInterval(Seconds);
