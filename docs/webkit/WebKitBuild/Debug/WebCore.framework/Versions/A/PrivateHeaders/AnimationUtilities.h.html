26#pragma once
27
28#include "IntPoint.h"
29#include "LayoutPoint.h"
30
31namespace WebCore {
32
33struct BlendingContext {
34 double progress { 0 };
35 bool isDiscrete { false };
36
37 BlendingContext(double progress = 0, bool isDiscrete = false)
38 : progress(progress)
39 , isDiscrete(isDiscrete)
40 {
41 }
42};
43
44inline int blend(int from, int to, const BlendingContext& context)
45{
46 return static_cast<int>(lround(static_cast<double>(from) + static_cast<double>(to - from) * context.progress));
47}
48
49inline unsigned blend(unsigned from, unsigned to, const BlendingContext& context)
50{
51 return static_cast<unsigned>(lround(to > from ? static_cast<double>(from) + static_cast<double>(to - from) * context.progress : static_cast<double>(from) - static_cast<double>(from - to) * context.progress));
52}
53
54inline double blend(double from, double to, const BlendingContext& context)
55{
56 return from + (to - from) * context.progress;
57}
58
59inline float blend(float from, float to, const BlendingContext& context)
60{
61 return static_cast<float>(from + (to - from) * context.progress);
62}
63
64inline LayoutUnit blend(LayoutUnit from, LayoutUnit to, const BlendingContext& context)
65{
66 return LayoutUnit(blend(from.toFloat(), to.toFloat(), context));
67}
68
69inline IntPoint blend(const IntPoint& from, const IntPoint& to, const BlendingContext& context)
70{
71 return IntPoint(blend(from.x(), to.x(), context),
72 blend(from.y(), to.y(), context));
73}
74
75inline LayoutPoint blend(const LayoutPoint& from, const LayoutPoint& to, const BlendingContext& context)
76{
77 return LayoutPoint(blend(from.x(), to.x(), context),
78 blend(from.y(), to.y(), context));
79}
80
81} // namespace WebCore
