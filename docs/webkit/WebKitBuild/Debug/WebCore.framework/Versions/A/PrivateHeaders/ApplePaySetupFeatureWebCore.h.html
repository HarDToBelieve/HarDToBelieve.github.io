30#include <wtf/RefCounted.h>
31#include <wtf/RetainPtr.h>
32
33OBJC_CLASS PKPaymentSetupFeature;
34
35namespace WebCore {
36
37enum class ApplePaySetupFeatureState : uint8_t;
38enum class ApplePaySetupFeatureType : uint8_t;
39
40class ApplePaySetupFeature : public RefCounted<ApplePaySetupFeature> {
41public:
42 static Ref<ApplePaySetupFeature> create(PKPaymentSetupFeature *feature)
43 {
44 return adoptRef(*new ApplePaySetupFeature(feature));
45 }
46
47 WEBCORE_EXPORT virtual ~ApplePaySetupFeature();
48
49 virtual ApplePaySetupFeatureState state() const;
50 virtual ApplePaySetupFeatureType type() const;
51
52 PKPaymentSetupFeature *platformFeature() const { return m_feature.get(); }
53
54#if ENABLE(APPLE_PAY_INSTALLMENTS)
55 virtual bool supportsInstallments() const;
56#endif
57
58protected:
59 WEBCORE_EXPORT ApplePaySetupFeature();
60
61private:
62 WEBCORE_EXPORT explicit ApplePaySetupFeature(PKPaymentSetupFeature *);
63
64 RetainPtr<PKPaymentSetupFeature> m_feature;
65};
66
67} // namespace WebCore

69#endif // ENABLE(APPLE_PAY)
