29#pragma once
30
31#include "ArchiveResource.h"
32#include <wtf/HashSet.h>
33
34namespace WebCore {
35
36class Archive : public RefCounted<Archive> {
37public:
38 virtual ~Archive();
39
40 virtual bool shouldLoadFromArchiveOnly() const = 0;
41 virtual bool shouldOverrideBaseURL() const = 0;
42 virtual bool shouldUseMainResourceEncoding() const = 0;
43 virtual bool shouldUseMainResourceURL() const = 0;
44
45 ArchiveResource* mainResource() { return m_mainResource.get(); }
46 const Vector<Ref<ArchiveResource>>& subresources() const { return m_subresources; }
47 const Vector<Ref<Archive>>& subframeArchives() const { return m_subframeArchives; }
48
49protected:
50 // These methods are meant for subclasses for different archive types to add resources in to the archive,
51 // and should not be exposed as archives should be immutable to clients
52 void setMainResource(Ref<ArchiveResource>&& mainResource) { m_mainResource = WTFMove(mainResource); }
53 void addSubresource(Ref<ArchiveResource>&& resource) { m_subresources.append(WTFMove(resource)); }
54 void addSubframeArchive(Ref<Archive>&& subframeArchive) { m_subframeArchives.append(WTFMove(subframeArchive)); }
55
56 void clearAllSubframeArchives();
57
58private:
59 void clearAllSubframeArchives(HashSet<Archive*>&);
60
61 RefPtr<ArchiveResource> m_mainResource;
62 Vector<Ref<ArchiveResource>> m_subresources;
63 Vector<Ref<Archive>> m_subframeArchives;
64};
65
66} // namespace WebCore
