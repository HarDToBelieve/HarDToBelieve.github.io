35class ArchiveResource : public SubstituteResource {
36public:
37 static RefPtr<ArchiveResource> create(RefPtr<SharedBuffer>&&, const URL&, const ResourceResponse&);
38 WEBCORE_EXPORT static RefPtr<ArchiveResource> create(RefPtr<SharedBuffer>&&, const URL&,
39 const String& mimeType, const String& textEncoding, const String& frameName,
40 const ResourceResponse& = ResourceResponse());
41
42 const String& mimeType() const { return m_mimeType; }
43 const String& textEncoding() const { return m_textEncoding; }
44 const String& frameName() const { return m_frameName; }
45
46 void ignoreWhenUnarchiving() { m_shouldIgnoreWhenUnarchiving = true; }
47 bool shouldIgnoreWhenUnarchiving() const { return m_shouldIgnoreWhenUnarchiving; }
48
49private:
50 ArchiveResource(Ref<SharedBuffer>&&, const URL&, const String& mimeType, const String& textEncoding, const String& frameName, const ResourceResponse&);
51
52 String m_mimeType;
53 String m_textEncoding;
54 String m_frameName;
55
56 bool m_shouldIgnoreWhenUnarchiving;
57};
