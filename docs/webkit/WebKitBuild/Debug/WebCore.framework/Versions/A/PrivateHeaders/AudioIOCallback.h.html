29#pragma once
30
31#include <wtf/MonotonicTime.h>
32#include <wtf/Seconds.h>
33
34namespace WebCore {
35
36class AudioBus;
37
38struct AudioIOPosition {
39 // Audio stream position in seconds.
40 Seconds position;
41 // System's monotonic time in seconds corresponding to the contained |position| value.
42 MonotonicTime timestamp;
43
44 template<typename Encoder>
45 void encode(Encoder& encoder) const
46 {
47 encoder << position;
48 encoder << timestamp;
49 }
50
51 template<typename Decoder> static std::optional<AudioIOPosition> decode(Decoder& decoder)
52 {
53 std::optional<Seconds> position;
54 decoder >> position;
55 if (!position)
56 return std::nullopt;
57
58 std::optional<MonotonicTime> timestamp;
59 decoder >> timestamp;
60 if (!timestamp)
61 return std::nullopt;
62
63 return AudioIOPosition { *position, *timestamp };
64 }
65};
66
67// Abstract base-class for isochronous audio I/O client.
68class AudioIOCallback {
69public:
70 // render() is called periodically to get the next render quantum of audio into destinationBus.
71 // Optional audio input is given in sourceBus (if it's not 0).
72 virtual void render(AudioBus* sourceBus, AudioBus* destinationBus, size_t framesToProcess, const AudioIOPosition& outputPosition) = 0;
73
74 virtual void isPlayingDidChange() = 0;
75
76 virtual ~AudioIOCallback() = default;
77};
