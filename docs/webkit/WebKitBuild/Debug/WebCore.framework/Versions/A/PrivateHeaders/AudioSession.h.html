51class WEBCORE_EXPORT AudioSession {
52 WTF_MAKE_FAST_ALLOCATED;
53 WTF_MAKE_NONCOPYABLE(AudioSession);
54 friend class UniqueRef<AudioSession>;
55 friend UniqueRef<AudioSession> WTF::makeUniqueRefWithoutFastMallocCheck<AudioSession>();
56public:
57 static UniqueRef<AudioSession> create();
58 static void setSharedSession(UniqueRef<AudioSession>&&);
59 static AudioSession& sharedSession();
60
61 using ChangedObserver = WTF::Observer<void(AudioSession&)>;
62 static void addAudioSessionChangedObserver(const ChangedObserver&);
63
64 virtual ~AudioSession();
65
66 enum class CategoryType : uint8_t {
67 None,
68 AmbientSound,
69 SoloAmbientSound,
70 MediaPlayback,
71 RecordAudio,
72 PlayAndRecord,
73 AudioProcessing,
74 };
75 virtual void setCategory(CategoryType, RouteSharingPolicy);
76 virtual CategoryType category() const;
77
78 virtual void setCategoryOverride(CategoryType);
79 virtual CategoryType categoryOverride() const;
80
81 virtual RouteSharingPolicy routeSharingPolicy() const;
82 virtual String routingContextUID() const;
83
84 virtual float sampleRate() const;
85 virtual size_t bufferSize() const;
86 virtual size_t numberOfOutputChannels() const;
87 virtual size_t maximumNumberOfOutputChannels() const;
88
89 bool tryToSetActive(bool);
90
91 virtual size_t preferredBufferSize() const;
92 virtual void setPreferredBufferSize(size_t);
93
94 class MutedStateObserver {
95 public:
96 virtual ~MutedStateObserver() = default;
97
98 virtual void hardwareMutedStateDidChange(AudioSession*) = 0;
99 };
100
101 virtual void addMutedStateObserver(MutedStateObserver*) { }
102 virtual void removeMutedStateObserver(MutedStateObserver*) { }
103
104 virtual void audioOutputDeviceChanged();
105 virtual void setIsPlayingToBluetoothOverride(std::optional<bool>);
106
107 virtual bool isMuted() const { return false; }
108 virtual void handleMutedStateChange() { }
109
110 virtual void beginInterruption();
111 enum class MayResume { No, Yes };
112 virtual void endInterruption(MayResume);
113
114 class InterruptionObserver : public CanMakeWeakPtr<InterruptionObserver> {
115 public:
116 virtual ~InterruptionObserver() = default;
117
118 virtual void beginAudioSessionInterruption() = 0;
119 virtual void endAudioSessionInterruption(MayResume) = 0;
120 };
121 virtual void addInterruptionObserver(InterruptionObserver&);
122 virtual void removeInterruptionObserver(InterruptionObserver&);
123
124 virtual bool isActive() const { return m_active; }
125
126 virtual void setRoutingArbitrationClient(WeakPtr<AudioSessionRoutingArbitrationClient>&& client) { m_routingArbitrationClient = client; }
127
128protected:
129 friend class NeverDestroyed<AudioSession>;
130 AudioSession();
131
132 virtual bool tryToSetActiveInternal(bool);
133
134 HashSet<MutedStateObserver*> m_observers;
135#if PLATFORM(IOS_FAMILY)
136 WeakHashSet<InterruptionObserver> m_interruptionObservers;
137#endif
138
139 WeakPtr<AudioSessionRoutingArbitrationClient> m_routingArbitrationClient;
140 bool m_active { false }; // Used only for testing.
141};

143class WEBCORE_EXPORT AudioSessionRoutingArbitrationClient {
144public:
145 virtual ~AudioSessionRoutingArbitrationClient() = default;
146
147 enum class RoutingArbitrationError : uint8_t { None, Failed, Cancelled };
148 enum class DefaultRouteChanged : bool { No, Yes };
149
150 using ArbitrationCallback = CompletionHandler<void(RoutingArbitrationError, DefaultRouteChanged)>;
151
152 virtual void beginRoutingArbitrationWithCategory(AudioSession::CategoryType, ArbitrationCallback&&) = 0;
153 virtual void leaveRoutingAbritration() = 0;
154
155 using WeakValueType = AudioSessionRoutingArbitrationClient;
156};
