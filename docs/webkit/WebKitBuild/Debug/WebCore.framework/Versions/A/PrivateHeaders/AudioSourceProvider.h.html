29#ifndef AudioSourceProvider_h
30#define AudioSourceProvider_h
31
32namespace WebCore {
33
34class AudioBus;
35class AudioSourceProviderClient;
36
37// Abstract base-class for a pull-model client.
38class AudioSourceProvider {
39public:
40 // provideInput() gets called repeatedly to render time-slices of a continuous audio stream.
41 virtual void provideInput(AudioBus* bus, size_t framesToProcess) = 0;
42
43 // If a client is set, we call it back when the audio format is available or changes.
44 virtual void setClient(AudioSourceProviderClient*) { };
45
46 virtual bool isHandlingAVPlayer() const { return false; }
47
48 virtual ~AudioSourceProvider() = default;
49};
50
51} // WebCore
52
53#endif // AudioSourceProvider_h
