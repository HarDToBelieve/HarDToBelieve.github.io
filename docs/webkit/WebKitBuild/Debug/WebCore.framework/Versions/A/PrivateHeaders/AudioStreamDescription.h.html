26#pragma once
27
28#include <wtf/Variant.h>
29
30typedef struct AudioStreamBasicDescription AudioStreamBasicDescription;
31
32namespace WebCore {
33
34struct PlatformDescription {
35 enum {
36 None,
37 CAAudioStreamBasicType,
38 GStreamerAudioStreamDescription,
39 } type;
40 Variant<std::nullptr_t, const AudioStreamBasicDescription*> description;
41};
42
43class AudioStreamDescription {
44public:
45 virtual ~AudioStreamDescription() = default;
46
47 virtual const PlatformDescription& platformDescription() const = 0;
48
49 enum PCMFormat {
50 None,
51 Int16,
52 Int32,
53 Float32,
54 Float64
55 };
56 virtual PCMFormat format() const = 0;
57
58 virtual double sampleRate() const = 0;
59
60 virtual bool isPCM() const { return format() != None; }
61 virtual bool isInterleaved() const = 0;
62 virtual bool isSignedInteger() const = 0;
63 virtual bool isFloat() const = 0;
64 virtual bool isNativeEndian() const = 0;
65
66 virtual uint32_t numberOfInterleavedChannels() const = 0;
67 virtual uint32_t numberOfChannelStreams() const = 0;
68 virtual uint32_t numberOfChannels() const = 0;
69 virtual uint32_t sampleWordSize() const = 0;
70};
71
72}
