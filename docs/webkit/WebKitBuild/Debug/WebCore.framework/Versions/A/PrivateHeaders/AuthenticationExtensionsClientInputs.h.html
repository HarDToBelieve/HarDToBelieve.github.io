26#pragma once
27
28#if ENABLE(WEB_AUTHN)
29
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34struct AuthenticationExtensionsClientInputs {
35 String appid;
36 bool googleLegacyAppidSupport;
37
38 template<class Encoder> void encode(Encoder&) const;
39 template<class Decoder> static std::optional<AuthenticationExtensionsClientInputs> decode(Decoder&);
40};
41
42template<class Encoder>
43void AuthenticationExtensionsClientInputs::encode(Encoder& encoder) const
44{
45 encoder << appid << googleLegacyAppidSupport;
46}
47
48template<class Decoder>
49std::optional<AuthenticationExtensionsClientInputs> AuthenticationExtensionsClientInputs::decode(Decoder& decoder)
50{
51 AuthenticationExtensionsClientInputs result;
52
53 std::optional<String> appid;
54 decoder >> appid;
55 if (!appid)
56 return std::nullopt;
57 result.appid = WTFMove(*appid);
58
59 std::optional<bool> googleLegacyAppidSupport;
60 decoder >> googleLegacyAppidSupport;
61 if (!googleLegacyAppidSupport)
62 return std::nullopt;
63 result.googleLegacyAppidSupport = WTFMove(*googleLegacyAppidSupport);
64
65 return result;
66}
67
68} // namespace WebCore
69
70#endif // ENABLE(WEB_AUTHN)
