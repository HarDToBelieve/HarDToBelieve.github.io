26#pragma once
27
28#include <wtf/URL.h>
29
30namespace WebCore {
31
32class BlobPart {
33public:
34 enum class Type : bool {
35 Data,
36 Blob
37 };
38
39 BlobPart()
40 : m_type(Type::Data)
41 {
42 }
43
44 BlobPart(Vector<uint8_t>&& data)
45 : m_type(Type::Data)
46 , m_data(WTFMove(data))
47 {
48 }
49
50 BlobPart(const URL& url)
51 : m_type(Type::Blob)
52 , m_url(url)
53 {
54 }
55
56 Type type() const { return m_type; }
57
58 const Vector<uint8_t>& data() const
59 {
60 ASSERT(m_type == Type::Data);
61 return m_data;
62 }
63
64 Vector<uint8_t> moveData()
65 {
66 ASSERT(m_type == Type::Data);
67 return WTFMove(m_data);
68 }
69
70 const URL& url() const
71 {
72 ASSERT(m_type == Type::Blob);
73 return m_url;
74 }
75
76 void detachFromCurrentThread()
77 {
78 m_url = m_url.isolatedCopy();
79 }
80
81private:
82 Type m_type;
83 Vector<uint8_t> m_data;
84 URL m_url;
85};
86
87} // namespace WebCore
