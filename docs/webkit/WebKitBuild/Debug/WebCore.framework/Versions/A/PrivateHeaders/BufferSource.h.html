26#pragma once
27
28#include <JavaScriptCore/ArrayBuffer.h>
29#include <JavaScriptCore/ArrayBufferView.h>
30#include <wtf/RefPtr.h>
31#include <wtf/Variant.h>
32
33namespace WebCore {
34
35class BufferSource {
36public:
37 using VariantType = WTF::Variant<RefPtr<JSC::ArrayBufferView>, RefPtr<JSC::ArrayBuffer>>;
38
39 BufferSource() { }
40 BufferSource(VariantType&& variant)
41 : m_variant(WTFMove(variant))
42 { }
43
44 const VariantType& variant() const { return m_variant; }
45
46 const uint8_t* data() const
47 {
48 return WTF::visit([](auto& buffer) -> const uint8_t* {
49 return buffer ? static_cast<const uint8_t*>(buffer->data()) : nullptr;
50 }, m_variant);
51 }
52
53 size_t length() const
54 {
55 return WTF::visit([](auto& buffer) -> size_t {
56 return buffer ? buffer->byteLength() : 0;
57 }, m_variant);
58 }
59
60private:
61 VariantType m_variant;
62};
63
64} // namespace WebCore
