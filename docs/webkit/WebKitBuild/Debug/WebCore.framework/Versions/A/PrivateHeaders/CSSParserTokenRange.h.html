32#include "CSSParserToken.h"
33#include <wtf/Forward.h>

35namespace WebCore {
36
37// A CSSParserTokenRange is an iterator over a subrange of a vector of CSSParserTokens.
38// Accessing outside of the range will return an endless stream of EOF tokens.
39// This class refers to half-open intervals [first, last).
40class CSSParserTokenRange {
41public:
42 template<size_t inlineBuffer>
43 CSSParserTokenRange(const Vector<CSSParserToken, inlineBuffer>& vector)
44 : m_first(vector.begin())
45 , m_last(vector.end())
46 {
47 }
48
49 // This should be called on a range with tokens returned by that range.
50 CSSParserTokenRange makeSubRange(const CSSParserToken* first, const CSSParserToken* last) const;
51
52 bool atEnd() const { return m_first == m_last; }
53 const CSSParserToken* end() const { return m_last; }
54
55 const CSSParserToken& peek(unsigned offset = 0) const
56 {
57 if (m_first + offset >= m_last)
58 return eofToken();
59 return *(m_first + offset);
60 }
61
62 const CSSParserToken& consume()
63 {
64 if (m_first == m_last)
65 return eofToken();
66 return *m_first++;
67 }
68
69 const CSSParserToken& consumeIncludingWhitespace()
70 {
71 const CSSParserToken& result = consume();
72 consumeWhitespace();
73 return result;
74 }
75
76 // The returned range doesn't include the brackets
77 CSSParserTokenRange consumeBlock();
78 CSSParserTokenRange consumeBlockCheckingForEditability(StyleSheetContents*);
79
80 void consumeComponentValue();
81
82 void consumeWhitespace()
83 {
84 while (peek().type() == WhitespaceToken)
85 ++m_first;
86 }
87
88 String serialize() const;
89
90 const CSSParserToken* begin() const { return m_first; }
91
92 static CSSParserToken& eofToken();
93
94private:
95 CSSParserTokenRange(const CSSParserToken* first, const CSSParserToken* last)
96 : m_first(first)
97 , m_last(last)
98 { }
99
100 const CSSParserToken* m_first;
101 const CSSParserToken* m_last;
102};
103
104} // namespace WebCore
