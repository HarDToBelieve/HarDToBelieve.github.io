94 // This value is used to handle quirky margins in reflow roots (body, td, and th) like WinIE.
95 // The basic idea is that a stylesheet can use the value __qem (for quirky em) instead of em.
96 // When the quirky value is used, if you're in quirks mode, the margin will collapse away
97 // inside a table cell. This quirk is specified in the HTML spec but our impl is different.
98 CSS_QUIRKY_EMS = 120
