34namespace WebCore {
35
36class CachedFrame;
37class CachedFramePlatformData;
38class Document;
39class DocumentLoader;
40class FrameView;
41class Node;
42enum class HasInsecureContent : bool;
43enum class UsedLegacyTLS : bool;
44
45class CachedFrameBase {
46public:
47 void restore();
48
49 Document* document() const { return m_document.get(); }
50 FrameView* view() const { return m_view.get(); }
51 const URL& url() const { return m_url; }
52 bool isMainFrame() { return m_isMainFrame; }
53
54protected:
55 CachedFrameBase(Frame&);
56 ~CachedFrameBase();
57
58 void pruneDetachedChildFrames();
59
60 RefPtr<Document> m_document;
61 RefPtr<DocumentLoader> m_documentLoader;
62 RefPtr<FrameView> m_view;
63 URL m_url;
64 std::unique_ptr<ScriptCachedFrameData> m_cachedFrameScriptData;
65 std::unique_ptr<CachedFramePlatformData> m_cachedFramePlatformData;
66 bool m_isMainFrame;
67
68 Vector<UniqueRef<CachedFrame>> m_childFrames;
69};
70
71class CachedFrame : private CachedFrameBase {
72 WTF_MAKE_FAST_ALLOCATED;
73public:
74 explicit CachedFrame(Frame&);
75
76 void open();
77 void clear();
78 void destroy();
79
80 WEBCORE_EXPORT void setCachedFramePlatformData(std::unique_ptr<CachedFramePlatformData>);
81 WEBCORE_EXPORT CachedFramePlatformData* cachedFramePlatformData();
82
83 HasInsecureContent hasInsecureContent() const;
84 UsedLegacyTLS usedLegacyTLS() const;
85
86 using CachedFrameBase::document;
87 using CachedFrameBase::view;
88 using CachedFrameBase::url;
89 DocumentLoader* documentLoader() const { return m_documentLoader.get(); }
90
91 size_t descendantFrameCount() const;
92};
