23#pragma once
24
25#include "CachedResourceClient.h"
26#include "ImageTypes.h"
27
28namespace WebCore {
29
30class CachedImage;
31class Document;
32class IntRect;
33
34enum class VisibleInViewportState { Unknown, Yes, No };
35
36class CachedImageClient : public CachedResourceClient {
37public:
38 virtual ~CachedImageClient() = default;
39 static CachedResourceClientType expectedType() { return ImageType; }
40 CachedResourceClientType resourceClientType() const override { return expectedType(); }
41
42 // Called whenever a frame of an image changes because we got more data from the network.
43 // If not null, the IntRect is the changed rect of the image.
44 virtual void imageChanged(CachedImage*, const IntRect* = nullptr) { }
45
46 virtual bool canDestroyDecodedData() { return true; }
47
48 // Called when a new decoded frame for a large image is available or when an animated image is ready to advance to the next frame.
49 virtual VisibleInViewportState imageFrameAvailable(CachedImage& image, ImageAnimatingState, const IntRect* changeRect) { imageChanged(&image, changeRect); return VisibleInViewportState::No; }
50 virtual VisibleInViewportState imageVisibleInViewport(const Document&) const { return VisibleInViewportState::No; }
51
52 virtual void didRemoveCachedImageClient(CachedImage&) { }
53
54 virtual void scheduleRenderingUpdateForImage(CachedImage&) { }
55};
56
57} // namespace WebCore
