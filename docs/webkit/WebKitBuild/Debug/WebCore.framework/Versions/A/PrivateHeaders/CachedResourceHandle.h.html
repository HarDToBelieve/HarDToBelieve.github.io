26#pragma once
27
28#include <wtf/Forward.h>

30namespace WebCore {
31
32class CachedResource;
33
34class WEBCORE_EXPORT CachedResourceHandleBase {
35public:
36 ~CachedResourceHandleBase();
37
38 CachedResource* get() const { return m_resource; }
39
40 bool operator!() const { return !m_resource; }
41
42 // This conversion operator allows implicit conversion to bool but not to other integer types.
43 typedef CachedResource* CachedResourceHandleBase::*UnspecifiedBoolType;
44 operator UnspecifiedBoolType() const { return m_resource ? &CachedResourceHandleBase::m_resource : 0; }
45
46protected:
47 CachedResourceHandleBase();
48 CachedResourceHandleBase(CachedResource*);
49 CachedResourceHandleBase(const CachedResourceHandleBase&);
50
51 void setResource(CachedResource*);
52
53private:
54 CachedResourceHandleBase& operator=(const CachedResourceHandleBase&) { return *this; }
55
56 friend class CachedResource;
57
58 CachedResource* m_resource;
59};
60
61template <class R> class CachedResourceHandle : public CachedResourceHandleBase {
62public:
63 CachedResourceHandle() { }
64 CachedResourceHandle(R* res) : CachedResourceHandleBase(res) { }
65 CachedResourceHandle(const CachedResourceHandle<R>& o) : CachedResourceHandleBase(o) { }
66 template<typename U> CachedResourceHandle(const CachedResourceHandle<U>& o) : CachedResourceHandleBase(o.get()) { }
67
68 R* get() const { return reinterpret_cast<R*>(CachedResourceHandleBase::get()); }
69 R* operator->() const { return get(); }
70 R& operator*() const { ASSERT(get()); return *get(); }
71
72 CachedResourceHandle& operator=(R* res) { setResource(res); return *this; }
73 CachedResourceHandle& operator=(const CachedResourceHandle& o) { setResource(o.get()); return *this; }
74 template<typename U> CachedResourceHandle& operator=(const CachedResourceHandle<U>& o) { setResource(o.get()); return *this; }
75
76 bool operator==(const CachedResourceHandleBase& o) const { return get() == o.get(); }
77 bool operator!=(const CachedResourceHandleBase& o) const { return get() != o.get(); }
78};
79
80template <class R, class RR> bool operator==(const CachedResourceHandle<R>& h, const RR* res)
81{
82 return h.get() == res;
83}
84template <class R, class RR> bool operator==(const RR* res, const CachedResourceHandle<R>& h)
85{
86 return h.get() == res;
87}
88template <class R, class RR> bool operator!=(const CachedResourceHandle<R>& h, const RR* res)
89{
90 return h.get() != res;
91}
92template <class R, class RR> bool operator!=(const RR* res, const CachedResourceHandle<R>& h)
93{
94 return h.get() != res;
95}
