35
36class CachedScript;
37class Document;
38
39class CachedScriptFetcher : public JSC::ScriptFetcher {
40public:
41 virtual CachedResourceHandle<CachedScript> requestModuleScript(Document&, const URL& sourceURL, String&& integrity) const;
42
43 static Ref<CachedScriptFetcher> create(const String& charset);
44
45protected:
46 CachedScriptFetcher(const String& nonce, ReferrerPolicy referrerPolicy, const String& charset, const AtomString& initiatorName, bool isInUserAgentShadowTree)
47 : m_nonce(nonce)
48 , m_charset(charset)
49 , m_initiatorName(initiatorName)
50 , m_isInUserAgentShadowTree(isInUserAgentShadowTree)
51 , m_referrerPolicy(referrerPolicy)
52 {
53 }
54
55 CachedScriptFetcher(const String& charset)
56 : m_charset(charset)
57 {
58 }
59
60 CachedResourceHandle<CachedScript> requestScriptWithCache(Document&, const URL& sourceURL, const String& crossOriginMode, String&& integrity, std::optional<ResourceLoadPriority>) const;
61
62private:
63 String m_nonce;
64 String m_charset;
65 AtomString m_initiatorName;
66 bool m_isInUserAgentShadowTree { false };
67 ReferrerPolicy m_referrerPolicy { ReferrerPolicy::EmptyString };
68};
