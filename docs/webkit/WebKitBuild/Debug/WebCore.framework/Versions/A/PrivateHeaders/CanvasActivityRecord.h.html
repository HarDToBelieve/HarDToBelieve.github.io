26#pragma once
27#include <wtf/HashSet.h>
28#include <wtf/text/StringHash.h>
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32struct CanvasActivityRecord {
33 HashSet<String> textWritten;
34 bool wasDataRead { false };
35
36 WEBCORE_EXPORT bool recordWrittenOrMeasuredText(const String&);
37 void mergeWith(const CanvasActivityRecord&);
38
39 template <class Encoder> void encode(Encoder&) const;
40 template <class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, CanvasActivityRecord&);
41};
42
43template <class Encoder>
44void CanvasActivityRecord::encode(Encoder& encoder) const
45{
46 encoder << textWritten;
47 encoder << wasDataRead;
48}
49
50template <class Decoder>
51bool CanvasActivityRecord::decode(Decoder& decoder, CanvasActivityRecord& canvasActivityRecord)
52{
53 if (!decoder.decode(canvasActivityRecord.textWritten))
54 return false;
55 if (!decoder.decode(canvasActivityRecord.wasDataRead))
56 return false;
57 return true;
58}
59} // namespace WebCore
