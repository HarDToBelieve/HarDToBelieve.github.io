26#pragma once
27
28#include "Color.h"
29#include <wtf/HashTraits.h>
30
31namespace WTF {
32
33struct ColorHash {
34 static unsigned hash(const WebCore::Color& key) { return key.hash(); }
35 static bool equal(const WebCore::Color& a, const WebCore::Color& b) { return a == b; }
36 static const bool safeToCompareToEmptyOrDeleted = true;
37};
38
39template<> struct DefaultHash<WebCore::Color> : ColorHash { };
40
41template<> struct HashTraits<WebCore::Color> : GenericHashTraits<WebCore::Color> {
42 static const bool emptyValueIsZero = false;
43 static WebCore::Color emptyValue() { return WebCore::Color(HashTableEmptyValue); }
44
45 static void constructDeletedValue(WebCore::Color& slot) { new (NotNull, &slot) WebCore::Color(HashTableDeletedValue); }
46 static bool isDeletedValue(const WebCore::Color& value) { return value.isHashTableDeletedValue(); }
47};
48
49} // namespace WTF
