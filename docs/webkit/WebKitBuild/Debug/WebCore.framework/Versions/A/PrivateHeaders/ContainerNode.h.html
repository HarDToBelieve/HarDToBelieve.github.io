76 struct ChildChange {
77 enum class Type : uint8_t { ElementInserted, ElementRemoved, TextInserted, TextRemoved, TextChanged, AllChildrenRemoved, NonContentsChildRemoved, NonContentsChildInserted, AllChildrenReplaced };
78 enum class Source : bool { Parser, API };
79
80 ChildChange::Type type;
81 Element* previousSiblingElement;
82 Element* nextSiblingElement;
83 ChildChange::Source source;
84
85 bool isInsertion() const
86 {
87 switch (type) {
88 case ChildChange::Type::ElementInserted:
89 case ChildChange::Type::TextInserted:
90 case ChildChange::Type::NonContentsChildInserted:
91 case ChildChange::Type::AllChildrenReplaced:
92 return true;
93 case ChildChange::Type::ElementRemoved:
94 case ChildChange::Type::TextRemoved:
95 case ChildChange::Type::TextChanged:
96 case ChildChange::Type::AllChildrenRemoved:
97 case ChildChange::Type::NonContentsChildRemoved:
98 return false;
99 }
100 ASSERT_NOT_REACHED();
101 return false;
102 }
103 };

216class ChildNodesLazySnapshot {
217 WTF_MAKE_NONCOPYABLE(ChildNodesLazySnapshot);
218 WTF_MAKE_FAST_ALLOCATED;
219public:
220 explicit ChildNodesLazySnapshot(Node& parentNode)
221 : m_currentNode(parentNode.firstChild())
222 , m_currentIndex(0)
223 , m_hasSnapshot(false)
224 {
225 m_nextSnapshot = latestSnapshot;
226 latestSnapshot = this;
227 }
228
229 ALWAYS_INLINE ~ChildNodesLazySnapshot()
230 {
231 latestSnapshot = m_nextSnapshot;
232 }
233
234 // Returns 0 if there is no next Node.
235 RefPtr<Node> nextNode()
236 {
237 if (LIKELY(!hasSnapshot())) {
238 RefPtr<Node> node = WTFMove(m_currentNode);
239 if (node)
240 m_currentNode = node->nextSibling();
241 return node;
242 }
243 if (m_currentIndex >= m_snapshot.size())
244 return nullptr;
245 return m_snapshot[m_currentIndex++];
246 }
247
248 void takeSnapshot()
249 {
250 if (hasSnapshot())
251 return;
252 m_hasSnapshot = true;
253 Node* node = m_currentNode.get();
254 while (node) {
255 m_snapshot.append(node);
256 node = node->nextSibling();
257 }
258 }
259
260 ChildNodesLazySnapshot* nextSnapshot() { return m_nextSnapshot; }
261 bool hasSnapshot() { return m_hasSnapshot; }
262
263 static void takeChildNodesLazySnapshot()
264 {
265 ChildNodesLazySnapshot* snapshot = latestSnapshot;
266 while (snapshot && !snapshot->hasSnapshot()) {
267 snapshot->takeSnapshot();
268 snapshot = snapshot->nextSnapshot();
269 }
270 }
271
272private:
273 static ChildNodesLazySnapshot* latestSnapshot;
274
275 RefPtr<Node> m_currentNode;
276 unsigned m_currentIndex;
277 bool m_hasSnapshot;
278 Vector<RefPtr<Node>> m_snapshot; // Lazily instantiated.
279 ChildNodesLazySnapshot* m_nextSnapshot;
280};
