35namespace WebCore {
36namespace ContentExtensions {
37
38class ContentExtensionCompilationClient {
39public:
40 virtual ~ContentExtensionCompilationClient() = default;
41
42 // Functions should be called in this order. All except writeActions and finalize can be called multiple times, though.
43 virtual void writeSource(String&&) = 0;
44 virtual void writeActions(Vector<SerializedActionByte>&&, bool conditionsApplyOnlyToDomain) = 0;
45 virtual void writeFiltersWithoutConditionsBytecode(Vector<DFABytecode>&&) = 0;
46 virtual void writeFiltersWithConditionsBytecode(Vector<DFABytecode>&&) = 0;
47 virtual void writeTopURLFiltersBytecode(Vector<DFABytecode>&&) = 0;
48 virtual void finalize() = 0;
49};
50
51WEBCORE_EXPORT std::error_code compileRuleList(ContentExtensionCompilationClient&, String&& ruleJSON, Vector<ContentExtensionRule>&&);
52
53} // namespace ContentExtensions
54} // namespace WebCore
