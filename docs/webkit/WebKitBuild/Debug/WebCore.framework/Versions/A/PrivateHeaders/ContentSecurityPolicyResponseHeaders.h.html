26#pragma once
27
28#include <wtf/Vector.h>
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class ContentSecurityPolicy;
34class ResourceResponse;
35
36enum class ContentSecurityPolicyHeaderType : bool {
37 Report,
38 Enforce,
39};
40
41class ContentSecurityPolicyResponseHeaders {
42public:
43 ContentSecurityPolicyResponseHeaders() = default;
44 WEBCORE_EXPORT explicit ContentSecurityPolicyResponseHeaders(const ResourceResponse&);
45
46 ContentSecurityPolicyResponseHeaders isolatedCopy() const;
47
48 template <class Encoder> void encode(Encoder&) const;
49 template <class Decoder> static std::optional<ContentSecurityPolicyResponseHeaders> decode(Decoder&);
50
51 enum EmptyTag { Empty };
52 struct MarkableTraits {
53 static bool isEmptyValue(const ContentSecurityPolicyResponseHeaders& identifier)
54 {
55 return identifier.m_emptyForMarkable;
56 }
57
58 static ContentSecurityPolicyResponseHeaders emptyValue()
59 {
60 return ContentSecurityPolicyResponseHeaders(Empty);
61 }
62 };
63
64private:
65 friend class ContentSecurityPolicy;
66 ContentSecurityPolicyResponseHeaders(EmptyTag)
67 : m_emptyForMarkable(true)
68 { }
69
70 Vector<std::pair<String, ContentSecurityPolicyHeaderType>> m_headers;
71 int m_httpStatusCode { 0 };
72 bool m_emptyForMarkable { false };
73};
74
75template <class Encoder>
76void ContentSecurityPolicyResponseHeaders::encode(Encoder& encoder) const
77{
78 encoder << static_cast<uint64_t>(m_headers.size());
79 for (auto& pair : m_headers) {
80 encoder << pair.first;
81 encoder << pair.second;
82 }
83 encoder << m_httpStatusCode;
84}
85
86template <class Decoder>
87std::optional<ContentSecurityPolicyResponseHeaders> ContentSecurityPolicyResponseHeaders::decode(Decoder& decoder)
88{
89 ContentSecurityPolicyResponseHeaders headers;
90
91 std::optional<uint64_t> headersSize;
92 decoder >> headersSize;
93 if (!headersSize)
94 return std::nullopt;
95 for (size_t i = 0; i < *headersSize; ++i) {
96 std::optional<String> header;
97 decoder >> header;
98 if (!header)
99 return std::nullopt;
100 std::optional<ContentSecurityPolicyHeaderType> headerType;
101 decoder >> headerType;
102 if (!headerType)
103 return std::nullopt;
104 headers.m_headers.append(std::make_pair(WTFMove(*header), WTFMove(*headerType)));
105 }
106 headers.m_headers.shrinkToFit();
107
108 std::optional<int> httpStatusCode;
109 decoder >> httpStatusCode;
110 if (!httpStatusCode)
111 return std::nullopt;
112 headers.m_httpStatusCode = *httpStatusCode;
113
114 return headers;
115}
