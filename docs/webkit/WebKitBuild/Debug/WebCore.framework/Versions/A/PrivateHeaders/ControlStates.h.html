26#ifndef ControlStates_h
27#define ControlStates_h
28
29#include <wtf/OptionSet.h>
30#include <wtf/RetainPtr.h>
31#include <wtf/Seconds.h>
32
33#if PLATFORM(COCOA)
34#ifndef __OBJC__
35typedef struct objc_object *id;
36#endif
37#endif
38
39namespace WebCore {
40
41#if PLATFORM(COCOA)
42typedef id PlatformControlInstance;
43#endif
44
45class ControlStates {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 enum class States : uint16_t {
49 Hovered = 1 << 0,
50 Pressed = 1 << 1,
51 Focused = 1 << 2,
52 Enabled = 1 << 3,
53 Checked = 1 << 4,
54 Default = 1 << 5,
55 WindowInactive = 1 << 6,
56 Indeterminate = 1 << 7,
57 SpinUp = 1 << 8, // Sub-state for HoverState and PressedState.
58 Presenting = 1 << 9,
59 };
60
61 ControlStates(OptionSet<States> states)
62 : m_states(states)
63 {
64 }
65
66 ControlStates() = default;
67
68 OptionSet<States> states() const { return m_states; }
69 void setStates(OptionSet<States> newStates)
70 {
71 if (newStates == m_states)
72 return;
73 m_states = newStates;
74 m_isDirty = m_initialized;
75 m_initialized = true;
76 }
77
78 bool needsRepaint() const { return m_needsRepaint; }
79 void setNeedsRepaint(bool r) { m_needsRepaint = r; }
80
81 bool isDirty() const { return m_isDirty; }
82 void setDirty(bool d) { m_isDirty = d; }
83
84 Seconds timeSinceControlWasFocused() const { return m_timeSinceControlWasFocused; }
85 void setTimeSinceControlWasFocused(Seconds time) { m_timeSinceControlWasFocused = time; }
86
87#if PLATFORM(COCOA)
88 PlatformControlInstance platformControl() const { return m_controlInstance.get(); }
89 void setPlatformControl(PlatformControlInstance instance) { m_controlInstance = instance; }
90#endif
91
92private:
93 OptionSet<States> m_states;
94 bool m_initialized { false };
95 bool m_needsRepaint { false };
96 bool m_isDirty { false };
97 Seconds m_timeSinceControlWasFocused { 0_s };
98#if PLATFORM(COCOA)
99 RetainPtr<PlatformControlInstance> m_controlInstance;
100#endif
101};
102
103}
104
105#endif
