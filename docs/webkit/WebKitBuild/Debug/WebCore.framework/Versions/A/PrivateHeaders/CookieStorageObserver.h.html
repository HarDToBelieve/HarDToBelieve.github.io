36namespace WebCore {
37
38// Use eager initialization for the WeakPtrFactory since we call makeWeakPtr() from a non-main thread.
39class WEBCORE_EXPORT CookieStorageObserver : public CanMakeWeakPtr<CookieStorageObserver, WeakPtrFactoryInitialization::Eager> {
40 WTF_MAKE_FAST_ALLOCATED;
41 WTF_MAKE_NONCOPYABLE(CookieStorageObserver);
42public:
43 explicit CookieStorageObserver(NSHTTPCookieStorage *);
44 ~CookieStorageObserver();
45
46 void startObserving(Function<void()>&& callback);
47 void stopObserving();
48
49 void cookiesDidChange();
50
51private:
52 RetainPtr<NSHTTPCookieStorage> m_cookieStorage;
53 bool m_hasRegisteredInternalsForNotifications { false };
54 RetainPtr<WebCookieObserverAdapter> m_observerAdapter;
55 WTF::Function<void()> m_cookieChangeCallback;
56};
