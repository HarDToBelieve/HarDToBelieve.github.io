21#pragma once
22
23#include <wtf/HashMap.h>
24#include <wtf/text/AtomStringHash.h>
25
26namespace WebCore {
27
28struct CounterDirectives {
29 std::optional<int> resetValue;
30 std::optional<int> incrementValue;
31};
32
33constexpr bool operator==(const CounterDirectives& a, const CounterDirectives& b)
34{
35 return a.incrementValue == b.incrementValue && a.resetValue == b.resetValue;
36}
37
38constexpr bool operator!=(const CounterDirectives& a, const CounterDirectives& b)
39{
40 return !(a == b);
41}
42
43using CounterDirectiveMap = HashMap<AtomString, CounterDirectives>;
44
45} // namespace WebCore
