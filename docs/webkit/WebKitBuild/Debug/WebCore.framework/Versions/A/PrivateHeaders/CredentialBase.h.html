26#pragma once
27
28#include <wtf/EnumTraits.h>
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class Credential;
34
35enum CredentialPersistence {
36 CredentialPersistenceNone,
37 CredentialPersistenceForSession,
38 CredentialPersistencePermanent
39};
40
41class CredentialBase {
42public:
43 WEBCORE_EXPORT bool isEmpty() const;
44
45 WEBCORE_EXPORT const String& user() const;
46 WEBCORE_EXPORT const String& password() const;
47 WEBCORE_EXPORT bool hasPassword() const;
48 WEBCORE_EXPORT CredentialPersistence persistence() const;
49
50 bool encodingRequiresPlatformData() const { return false; }
51
52 WEBCORE_EXPORT static bool compare(const Credential&, const Credential&);
53
54 WEBCORE_EXPORT String serializationForBasicAuthorizationHeader() const;
55
56protected:
57 WEBCORE_EXPORT CredentialBase();
58 WEBCORE_EXPORT CredentialBase(const String& user, const String& password, CredentialPersistence);
59 CredentialBase(const Credential& original, CredentialPersistence);
60
61 static bool platformCompare(const Credential&, const Credential&) { return true; }
62
63private:
64 String m_user;
65 String m_password;
66 CredentialPersistence m_persistence;
67};
68
69inline bool operator==(const Credential& a, const Credential& b) { return CredentialBase::compare(a, b); }
70inline bool operator!=(const Credential& a, const Credential& b) { return !(a == b); }
71
72} // namespace WebCore
73
74namespace WTF {
75
76template<> struct EnumTraits<WebCore::CredentialPersistence> {
77 using values = EnumValues<
78 WebCore::CredentialPersistence,
79 WebCore::CredentialPersistence::CredentialPersistenceNone,
80 WebCore::CredentialPersistence::CredentialPersistenceForSession,
81 WebCore::CredentialPersistence::CredentialPersistencePermanent
82 >;
83};
84
85} // namespace WTF
