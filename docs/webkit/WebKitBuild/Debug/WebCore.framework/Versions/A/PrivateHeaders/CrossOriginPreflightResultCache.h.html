38namespace WebCore {
39
40class HTTPHeaderMap;
41class ResourceResponse;
42
43class CrossOriginPreflightResultCacheItem {
44 WTF_MAKE_NONCOPYABLE(CrossOriginPreflightResultCacheItem); WTF_MAKE_FAST_ALLOCATED;
45public:
46 static Expected<UniqueRef<CrossOriginPreflightResultCacheItem>, String> create(StoredCredentialsPolicy, const ResourceResponse&);
47
48 CrossOriginPreflightResultCacheItem(MonotonicTime, StoredCredentialsPolicy, HashSet<String>&&, HashSet<String, ASCIICaseInsensitiveHash>&&);
49
50 std::optional<String> validateMethodAndHeaders(const String& method, const HTTPHeaderMap&) const;
51 bool allowsRequest(StoredCredentialsPolicy, const String& method, const HTTPHeaderMap&) const;
52
53private:
54 bool allowsCrossOriginMethod(const String&, StoredCredentialsPolicy) const;
55 std::optional<String> validateCrossOriginHeaders(const HTTPHeaderMap&, StoredCredentialsPolicy) const;
56
57 // FIXME: A better solution to holding onto the absolute expiration time might be
58 // to start a timer for the expiration delta that removes this from the cache when
59 // it fires.
60 MonotonicTime m_absoluteExpiryTime;
61 StoredCredentialsPolicy m_storedCredentialsPolicy;
62 HashSet<String> m_methods;
63 HashSet<String, ASCIICaseInsensitiveHash> m_headers;
64};
65
66class CrossOriginPreflightResultCache {
67 WTF_MAKE_NONCOPYABLE(CrossOriginPreflightResultCache); WTF_MAKE_FAST_ALLOCATED;
68public:
69 WEBCORE_EXPORT static CrossOriginPreflightResultCache& singleton();
70 WEBCORE_EXPORT void appendEntry(PAL::SessionID, const String& origin, const URL&, std::unique_ptr<CrossOriginPreflightResultCacheItem>);
71 WEBCORE_EXPORT bool canSkipPreflight(PAL::SessionID, const String& origin, const URL&, StoredCredentialsPolicy, const String& method, const HTTPHeaderMap& requestHeaders);
72 WEBCORE_EXPORT void clear();
73
74private:
75 friend NeverDestroyed<CrossOriginPreflightResultCache>;
76 CrossOriginPreflightResultCache();
77
78 HashMap<std::tuple<PAL::SessionID, String, URL>, std::unique_ptr<CrossOriginPreflightResultCacheItem>> m_preflightHashMap;
79};
80
81inline CrossOriginPreflightResultCacheItem::CrossOriginPreflightResultCacheItem(MonotonicTime absoluteExpiryTime, StoredCredentialsPolicy storedCredentialsPolicy, HashSet<String>&& methods, HashSet<String, ASCIICaseInsensitiveHash>&& headers)
82 : m_absoluteExpiryTime(absoluteExpiryTime)
83 , m_storedCredentialsPolicy(storedCredentialsPolicy)
84 , m_methods(WTFMove(methods))
85 , m_headers(WTFMove(headers))
86{
87}
