96struct CacheIdentifierOperationResult {
97 template<class Encoder> void encode(Encoder&) const;
98 template<class Decoder> static std::optional<CacheIdentifierOperationResult> decode(Decoder&);
99
100 uint64_t identifier { 0 };
101 // True in case storing cache list on the filesystem failed.
102 bool hadStorageError { false };
103};

120template<class Encoder> inline void CacheInfos::encode(Encoder& encoder) const
121{
122 encoder << infos;
123 encoder << updateCounter;
124}
125
126template<class Decoder> inline std::optional<CacheInfos> CacheInfos::decode(Decoder& decoder)
127{
128 std::optional<Vector<CacheInfo>> infos;
129 decoder >> infos;
130 if (!infos)
131 return std::nullopt;
132
133 std::optional<uint64_t> updateCounter;
134 decoder >> updateCounter;
135 if (!updateCounter)
136 return std::nullopt;
137
138 return {{ WTFMove(*infos), WTFMove(*updateCounter) }};
139}
140
141template<class Encoder> inline void CacheIdentifierOperationResult::encode(Encoder& encoder) const
142{
143 encoder << identifier;
144 encoder << hadStorageError;
145}
146
147template<class Decoder> inline std::optional<CacheIdentifierOperationResult> CacheIdentifierOperationResult::decode(Decoder& decoder)
148{
149 std::optional<uint64_t> identifier;
150 decoder >> identifier;
151 if (!identifier)
152 return std::nullopt;
153
154 std::optional<bool> hadStorageError;
155 decoder >> hadStorageError;
156 if (!hadStorageError)
157 return std::nullopt;
158 return {{ WTFMove(*identifier), WTFMove(*hadStorageError) }};
159}
