26#pragma once
27
28#if ENABLE(DATA_DETECTION)
29
30#include "ImageOverlayDataDetectionResultIdentifier.h"
31#include <wtf/HashMap.h>
32#include <wtf/Noncopyable.h>
33#include <wtf/RetainPtr.h>
34
35OBJC_CLASS DDScannerResult;
36OBJC_CLASS NSArray;
37
38namespace WebCore {
39
40class DataDetectionResultsStorage {
41 WTF_MAKE_NONCOPYABLE(DataDetectionResultsStorage);
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 DataDetectionResultsStorage() = default;
45
46 void setDocumentLevelResults(NSArray *results) { m_documentLevelResults = results; }
47 NSArray *documentLevelResults() const { return m_documentLevelResults.get(); }
48
49 DDScannerResult *imageOverlayDataDetectionResult(ImageOverlayDataDetectionResultIdentifier identifier) { return m_imageOverlayResults.get(identifier).get(); }
50 ImageOverlayDataDetectionResultIdentifier addImageOverlayDataDetectionResult(DDScannerResult *result)
51 {
52 auto newIdentifier = ImageOverlayDataDetectionResultIdentifier::generate();
53 m_imageOverlayResults.set(newIdentifier, result);
54 return newIdentifier;
55 }
56
57private:
58 RetainPtr<NSArray> m_documentLevelResults;
59 HashMap<ImageOverlayDataDetectionResultIdentifier, RetainPtr<DDScannerResult>> m_imageOverlayResults;
60};
61
62} // namespace WebCore
63
64#endif // ENABLE(DATA_DETECTION)
