34namespace WebCore {
35
36enum class DataListSuggestionActivationType : uint8_t {
37 ControlClicked,
38 IndicatorClicked,
39 TextChanged,
40};
41
42struct DataListSuggestion {
43 String value;
44 String label;
45
46 template<class Encoder> void encode(Encoder&) const;
47 template<class Decoder> static std::optional<DataListSuggestion> decode(Decoder&);
48};
49
50template<class Encoder>
51void DataListSuggestion::encode(Encoder& encoder) const
52{
53 encoder << value;
54 encoder << label;
55}
56
57template<class Decoder>
58std::optional<DataListSuggestion> DataListSuggestion::decode(Decoder& decoder)
59{
60 std::optional<String> value;
61 decoder >> value;
62 if (!value)
63 return std::nullopt;
64
65 std::optional<String> label;
66 decoder >> label;
67 if (!label)
68 return std::nullopt;
69
70 return {{ *value, *label }};
71}
72
73struct DataListSuggestionInformation {
74 DataListSuggestionActivationType activationType;
75 Vector<DataListSuggestion> suggestions;
76 IntRect elementRect;
77
78 template<class Encoder> void encode(Encoder&) const;
79 template<class Decoder> static std::optional<DataListSuggestionInformation> decode(Decoder&);
80};
81
82template<class Encoder>
83void DataListSuggestionInformation::encode(Encoder& encoder) const
84{
85 encoder << activationType;
86 encoder << suggestions;
87 encoder << elementRect;
88}
89
90template<class Decoder>
91std::optional<DataListSuggestionInformation> DataListSuggestionInformation::decode(Decoder& decoder)
92{
93 DataListSuggestionActivationType activationType;
94 if (!decoder.decode(activationType))
95 return std::nullopt;
96
97 std::optional<Vector<DataListSuggestion>> suggestions;
98 decoder >> suggestions;
99 if (!suggestions)
100 return std::nullopt;
101
102 std::optional<IntRect> elementRect;
103 decoder >> elementRect;
104 if (!elementRect)
105 return std::nullopt;
106
107 return {{ activationType, *suggestions, *elementRect }};
108}

110} // namespace WebCore
