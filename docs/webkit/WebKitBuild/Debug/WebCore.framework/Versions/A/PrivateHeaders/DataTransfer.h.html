31namespace WebCore {
32
33class CachedImage;
34class DataTransferItemList;
35class Document;
36class DragData;
37class DragImageLoader;
38class Element;
39class FileList;
40class File;
41class Pasteboard;
42class ScriptExecutionContext;
43enum class WebContentReadingPolicy : bool;

45class DataTransfer : public RefCounted<DataTransfer> {
46public:
47 // https://html.spec.whatwg.org/multipage/dnd.html#drag-data-store-mode
48 enum class StoreMode { Invalid, ReadWrite, Readonly, Protected };
49
50 static Ref<DataTransfer> create();
51 static Ref<DataTransfer> createForCopyAndPaste(const Document&, StoreMode, std::unique_ptr<Pasteboard>&&);
52 static Ref<DataTransfer> createForInputEvent(const String& plainText, const String& htmlText);
53
54 WEBCORE_EXPORT ~DataTransfer();
55
56 String dropEffect() const;
57 void setDropEffect(const String&);
58
59 String effectAllowed() const;
60 void setEffectAllowed(const String&);
61
62 DataTransferItemList& items(Document&);
63 Vector<String> types() const;
64 Vector<String> typesForItemList() const;
65
66 FileList& files(Document*) const;
67 FileList& files(Document&) const;
68
69 void clearData(const String& type = String());
70
71 String getData(Document&, const String& type) const;
72 String getDataForItem(Document&, const String& type) const;
73
74 void setData(const String& type, const String& data);
75 void setDataFromItemList(const String& type, const String& data);
76
77 void setDragImage(Element&, int x, int y);
78
79 void makeInvalidForSecurity() { m_storeMode = StoreMode::Invalid; }
80
81 bool canReadTypes() const;
82 bool canReadData() const;
83 bool canWriteData() const;
84
85 bool hasFileOfType(const String&);
86 bool hasStringOfType(const String&);
87
88 Pasteboard& pasteboard() { return *m_pasteboard; }
89 void commitToPasteboard(Pasteboard&);
90
91#if ENABLE(DRAG_SUPPORT)
92 static Ref<DataTransfer> createForDrag(const Document&);
93 static Ref<DataTransfer> createForDragStartEvent(const Document&);
94 static Ref<DataTransfer> createForDrop(const Document&, std::unique_ptr<Pasteboard>&&, OptionSet<DragOperation>, bool draggingFiles);
95 static Ref<DataTransfer> createForUpdatingDropTarget(const Document&, std::unique_ptr<Pasteboard>&&, OptionSet<DragOperation>, bool draggingFiles);
96
97 bool dropEffectIsUninitialized() const { return m_dropEffect == "uninitialized"; }
98
99 OptionSet<DragOperation> sourceOperationMask() const;
100 OptionSet<DragOperation> destinationOperationMask() const;
101 void setSourceOperationMask(OptionSet<DragOperation>);
102 void setDestinationOperationMask(OptionSet<DragOperation>);
103
104 void setDragHasStarted() { m_shouldUpdateDragImage = true; }
105 DragImageRef createDragImage(IntPoint& dragLocation) const;
106 void updateDragImage();
107 RefPtr<Element> dragImageElement() const;
108
109 void moveDragState(Ref<DataTransfer>&&);
110 bool hasDragImage() const;
111#endif
112
113 void didAddFileToItemList();
114 void updateFileList(ScriptExecutionContext*);
115
116private:
117 enum class Type { CopyAndPaste, DragAndDropData, DragAndDropFiles, InputEvent };
118 DataTransfer(StoreMode, std::unique_ptr<Pasteboard>, Type = Type::CopyAndPaste, String&& effectAllowed = "uninitialized"_s);
119
120#if ENABLE(DRAG_SUPPORT)
121 bool forDrag() const { return m_type == Type::DragAndDropData || m_type == Type::DragAndDropFiles; }
122 bool forFileDrag() const { return m_type == Type::DragAndDropFiles; }
123#else
124 bool forDrag() const { return false; }
125 bool forFileDrag() const { return false; }
126#endif
127
128 String readStringFromPasteboard(Document&, const String& lowercaseType, WebContentReadingPolicy) const;
129 bool shouldSuppressGetAndSetDataToAvoidExposingFilePaths() const;
130
131 enum class AddFilesType { No, Yes };
132 Vector<String> types(AddFilesType) const;
133 Vector<Ref<File>> filesFromPasteboardAndItemList(ScriptExecutionContext*) const;
134
135 String m_originIdentifier;
136 StoreMode m_storeMode;
137 std::unique_ptr<Pasteboard> m_pasteboard;
138 std::unique_ptr<DataTransferItemList> m_itemList;
139
140 mutable RefPtr<FileList> m_fileList;
141
142#if ENABLE(DRAG_SUPPORT)
143 Type m_type;
144 String m_dropEffect;
145 String m_effectAllowed;
146 bool m_shouldUpdateDragImage;
147 IntPoint m_dragLocation;
148 CachedResourceHandle<CachedImage> m_dragImage;
149 RefPtr<Element> m_dragImageElement;
150 std::unique_ptr<DragImageLoader> m_dragImageLoader;
151#endif
152};
