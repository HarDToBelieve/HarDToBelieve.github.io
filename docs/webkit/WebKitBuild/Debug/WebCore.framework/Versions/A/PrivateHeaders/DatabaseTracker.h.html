45class Database;

52enum class CurrentQueryBehavior { Interrupt, RunToCompletion };
53
54class DatabaseTracker {
55 WTF_MAKE_NONCOPYABLE(DatabaseTracker); WTF_MAKE_FAST_ALLOCATED;
56public:
57 // FIXME: This is a hack so we can easily delete databases from the UI process in WebKit2.
58 WEBCORE_EXPORT static std::unique_ptr<DatabaseTracker> trackerWithDatabasePath(const String& databasePath);
59
60 static void initializeTracker(const String& databasePath);
61
62 WEBCORE_EXPORT static DatabaseTracker& singleton();
63 WEBCORE_EXPORT static bool isInitialized();
64 // This singleton will potentially be used from multiple worker threads and the page's context thread simultaneously. To keep this safe, it's
65 // currently using 4 locks. In order to avoid deadlock when taking multiple locks, you must take them in the correct order:
66 // m_databaseGuard before quotaManager if both locks are needed.
67 // m_openDatabaseMapGuard before quotaManager if both locks are needed.
68 // m_databaseGuard and m_openDatabaseMapGuard currently don't overlap.
69 // notificationMutex() is currently independent of the other locks.
70
71 ExceptionOr<void> canEstablishDatabase(DatabaseContext&, const String& name, uint64_t estimatedSize);
72 ExceptionOr<void> retryCanEstablishDatabase(DatabaseContext&, const String& name, uint64_t estimatedSize);
73
74 void setDatabaseDetails(const SecurityOriginData&, const String& name, const String& displayName, uint64_t estimatedSize);
75 WEBCORE_EXPORT String fullPathForDatabase(const SecurityOriginData&, const String& name, bool createIfDoesNotExist);
76
77 Vector<Ref<Database>> openDatabases();
78 void addOpenDatabase(Database&);
79 void removeOpenDatabase(Database&);
80
81 uint64_t maximumSize(Database&);
82
83 WEBCORE_EXPORT void closeAllDatabases(CurrentQueryBehavior = CurrentQueryBehavior::RunToCompletion);
84
85 WEBCORE_EXPORT Vector<SecurityOriginData> origins();
86 WEBCORE_EXPORT Vector<String> databaseNames(const SecurityOriginData&);
87
88 DatabaseDetails detailsForNameAndOrigin(const String&, const SecurityOriginData&);
89
90 WEBCORE_EXPORT uint64_t usage(const SecurityOriginData&);
91 WEBCORE_EXPORT uint64_t quota(const SecurityOriginData&);
92 WEBCORE_EXPORT void setQuota(const SecurityOriginData&, uint64_t);
93 RefPtr<OriginLock> originLockFor(const SecurityOriginData&);
94
95 WEBCORE_EXPORT void deleteAllDatabasesImmediately();
96 WEBCORE_EXPORT void deleteDatabasesModifiedSince(WallTime);
97 WEBCORE_EXPORT bool deleteOrigin(const SecurityOriginData&);
98 WEBCORE_EXPORT bool deleteDatabase(const SecurityOriginData&, const String& name);
99
100#if PLATFORM(IOS_FAMILY)
101 WEBCORE_EXPORT void removeDeletedOpenedDatabases();
102 WEBCORE_EXPORT static bool deleteDatabaseFileIfEmpty(const String&);
103
104 // MobileSafari will grab this mutex on the main thread before dispatching the task to
105 // clean up zero byte database files. Any operations to open new database will have to
106 // wait for that task to finish by waiting on this mutex.
107 static Lock& openDatabaseMutex();
108
109 WEBCORE_EXPORT static void emptyDatabaseFilesRemovalTaskWillBeScheduled();
110 WEBCORE_EXPORT static void emptyDatabaseFilesRemovalTaskDidFinish();
111#endif
112
113 void setClient(DatabaseManagerClient*);
114
115 // From a secondary thread, must be thread safe with its data
116 void scheduleNotifyDatabaseChanged(const SecurityOriginData&, const String& name);
117
118 void doneCreatingDatabase(Database&);
119
120private:
121 explicit DatabaseTracker(const String& databasePath);
122
123 ExceptionOr<void> hasAdequateQuotaForOrigin(const SecurityOriginData&, uint64_t estimatedSize) WTF_REQUIRES_LOCK(m_databaseGuard);
124
125 bool hasEntryForOriginNoLock(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
126 String fullPathForDatabaseNoLock(const SecurityOriginData&, const String& name, bool createIfDoesNotExist) WTF_REQUIRES_LOCK(m_databaseGuard);
127 Vector<String> databaseNamesNoLock(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard) WTF_REQUIRES_LOCK(m_databaseGuard);
128 uint64_t quotaNoLock(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
129
130 String trackerDatabasePath() const WTF_REQUIRES_LOCK(m_databaseGuard);
131
132 enum TrackerCreationAction {
133 DontCreateIfDoesNotExist,
134 CreateIfDoesNotExist
135 };
136 void openTrackerDatabase(TrackerCreationAction) WTF_REQUIRES_LOCK(m_databaseGuard);
137
138 String originPath(const SecurityOriginData&) const;
139
140 bool hasEntryForDatabase(const SecurityOriginData&, const String& databaseIdentifier) WTF_REQUIRES_LOCK(m_databaseGuard);
141
142 bool addDatabase(const SecurityOriginData&, const String& name, const String& path) WTF_REQUIRES_LOCK(m_databaseGuard);
143
144 enum class DeletionMode {
145 Immediate,
146#if PLATFORM(IOS_FAMILY)
147 // Deferred deletion is currently only supported on iOS
148 // (see removeDeletedOpenedDatabases etc, above).
149 Deferred,
150 Default = Deferred
151#else
152 Default = Immediate
153#endif
154 };
155
156 bool deleteOrigin(const SecurityOriginData&, DeletionMode);
157 bool deleteDatabaseFile(const SecurityOriginData&, const String& name, DeletionMode);
158
159 void deleteOriginLockFor(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
160
161 using DatabaseSet = HashSet<Database*>;
162 using DatabaseNameMap = HashMap<String, DatabaseSet*>;
163 using DatabaseOriginMap = HashMap<SecurityOriginData, DatabaseNameMap*>;
164
165 Lock m_openDatabaseMapGuard;
166 mutable std::unique_ptr<DatabaseOriginMap> m_openDatabaseMap WTF_GUARDED_BY_LOCK(m_openDatabaseMapGuard);
167
168 // This lock protects m_database, m_originLockMap, m_databaseDirectoryPath, m_originsBeingDeleted, m_beingCreated, and m_beingDeleted.
169 Lock m_databaseGuard;
170 SQLiteDatabase m_database WTF_GUARDED_BY_LOCK(m_databaseGuard);
171
172 using OriginLockMap = HashMap<String, RefPtr<OriginLock>>;
173 OriginLockMap m_originLockMap WTF_GUARDED_BY_LOCK(m_databaseGuard);
174
175 String m_databaseDirectoryPath;
176
177 DatabaseManagerClient* m_client { nullptr };
178
179 HashMap<SecurityOriginData, std::unique_ptr<HashCountedSet<String>>> m_beingCreated WTF_GUARDED_BY_LOCK(m_databaseGuard);
180 HashMap<SecurityOriginData, std::unique_ptr<HashSet<String>>> m_beingDeleted WTF_GUARDED_BY_LOCK(m_databaseGuard);
181 HashSet<SecurityOriginData> m_originsBeingDeleted WTF_GUARDED_BY_LOCK(m_databaseGuard);
182 bool isDeletingDatabaseOrOriginFor(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
183 void recordCreatingDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
184 void doneCreatingDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
185 bool creatingDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
186 bool canDeleteDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
187 void recordDeletingDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
188 void doneDeletingDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
189 bool isDeletingDatabase(const SecurityOriginData&, const String& name) WTF_REQUIRES_LOCK(m_databaseGuard);
190 bool canDeleteOrigin(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
191 bool isDeletingOrigin(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
192 void recordDeletingOrigin(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
193 void doneDeletingOrigin(const SecurityOriginData&) WTF_REQUIRES_LOCK(m_databaseGuard);
194
195 static void scheduleForNotification();
196 static void notifyDatabasesChanged();
197};
