26#pragma once
27
28#include <wtf/Function.h>
29#include <wtf/MainThread.h>
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class MainThreadDeferrableTask : public CanMakeWeakPtr<MainThreadDeferrableTask> {
35public:
36 MainThreadDeferrableTask() = default;
37
38 void close()
39 {
40 cancelTask();
41 m_isClosed = true;
42 }
43
44 void cancelTask()
45 {
46 weakPtrFactory().revokeAll();
47 m_isPending = false;
48 }
49
50 bool isPending() const { return m_isPending; }
51
52 void scheduleTask(Function<void()>&& task)
53 {
54 if (m_isClosed)
55 return;
56
57 cancelTask();
58
59 m_isPending = true;
60 callOnMainThread([weakThis = makeWeakPtr(*this), task = WTFMove(task)] {
61 if (!weakThis)
62 return;
63 ASSERT(weakThis->isPending());
64 weakThis->m_isPending = false;
65 task();
66 });
67 }
68
69private:
70 bool m_isPending { false };
71 bool m_isClosed { false };
72};
73
74} // namespace WebCore
