37class DeprecatedCSSOMPrimitiveValue : public DeprecatedCSSOMValue {
38public:

40 enum UnitType {
41 CSS_UNKNOWN = 0,
42 CSS_NUMBER = 1,
43 CSS_PERCENTAGE = 2,
44 CSS_EMS = 3,
45 CSS_EXS = 4,
46 CSS_PX = 5,
47 CSS_CM = 6,
48 CSS_MM = 7,
49 CSS_IN = 8,
50 CSS_PT = 9,
51 CSS_PC = 10,
52 CSS_DEG = 11,
53 CSS_RAD = 12,
54 CSS_GRAD = 13,
55 CSS_MS = 14,
56 CSS_S = 15,
57 CSS_HZ = 16,
58 CSS_KHZ = 17,
59 CSS_DIMENSION = 18,
60 CSS_STRING = 19,
61 CSS_URI = 20,
62 CSS_IDENT = 21,
63 CSS_ATTR = 22,
64 CSS_COUNTER = 23,
65 CSS_RECT = 24,
66 CSS_RGBCOLOR = 25,
67 CSS_VW = 26,
68 CSS_VH = 27,
69 CSS_VMIN = 28,
70 CSS_VMAX = 29
71 };
72
73 static Ref<DeprecatedCSSOMPrimitiveValue> create(const CSSPrimitiveValue& value, CSSStyleDeclaration& owner)
74 {
75 return adoptRef(*new DeprecatedCSSOMPrimitiveValue(value, owner));
76 }
77
78 bool equals(const DeprecatedCSSOMPrimitiveValue& other) const { return m_value->equals(other.m_value); }
79 String cssText() const { return m_value->cssText(); }
80
81 WEBCORE_EXPORT unsigned short primitiveType() const;
82 WEBCORE_EXPORT ExceptionOr<float> getFloatValue(unsigned short unitType) const;
83 WEBCORE_EXPORT ExceptionOr<String> getStringValue() const;
84 WEBCORE_EXPORT ExceptionOr<Ref<DeprecatedCSSOMCounter>> getCounterValue() const;
85 WEBCORE_EXPORT ExceptionOr<Ref<DeprecatedCSSOMRect>> getRectValue() const;
86 WEBCORE_EXPORT ExceptionOr<Ref<DeprecatedCSSOMRGBColor>> getRGBColorValue() const;
87
88 static ExceptionOr<void> setFloatValue(unsigned short, double) { return Exception { NoModificationAllowedError }; }
89 static ExceptionOr<void> setStringValue(unsigned short, const String&) { return Exception { NoModificationAllowedError }; }
90
91 String stringValue() const { return m_value->stringValue(); }
92
93private:
94 DeprecatedCSSOMPrimitiveValue(const CSSPrimitiveValue& value, CSSStyleDeclaration& owner)
95 : DeprecatedCSSOMValue(DeprecatedPrimitiveValueClass, owner)
96 , m_value(const_cast<CSSPrimitiveValue&>(value))
97 {
98 }
99
100 Ref<CSSPrimitiveValue> m_value;
101};

103} // namespace WebCore
