26#pragma once
27
28#include "ConcreteImageBuffer.h"
29#include "DisplayListDrawingContext.h"
30#include "InMemoryDisplayList.h"
31
32namespace WebCore {
33namespace DisplayList {
34
35template<typename BackendType>
36class ImageBuffer : public ConcreteImageBuffer<BackendType> {
37 using BaseConcreteImageBuffer = ConcreteImageBuffer<BackendType>;
38 using BaseConcreteImageBuffer::logicalSize;
39 using BaseConcreteImageBuffer::baseTransform;
40
41public:
42 static auto create(const FloatSize& size, float resolutionScale, const DestinationColorSpace& colorSpace, PixelFormat pixelFormat, const HostWindow* hostWindow)
43 {
44 return BaseConcreteImageBuffer::template create<ImageBuffer>(size, resolutionScale, colorSpace, pixelFormat, hostWindow);
45 }
46
47 static auto create(const FloatSize& size, const GraphicsContext& context)
48 {
49 return BaseConcreteImageBuffer::template create<ImageBuffer>(size, context);
50 }
51
52 ImageBuffer(const ImageBufferBackend::Parameters& parameters, std::unique_ptr<BackendType>&& backend)
53 : BaseConcreteImageBuffer(parameters, WTFMove(backend))
54 , m_drawingContext(logicalSize(), baseTransform())
55 , m_writingClient(WTF::makeUnique<InMemoryDisplayList::WritingClient>())
56 , m_readingClient(WTF::makeUnique<InMemoryDisplayList::ReadingClient>())
57 {
58 m_drawingContext.displayList().setItemBufferWritingClient(m_writingClient.get());
59 m_drawingContext.displayList().setItemBufferReadingClient(m_readingClient.get());
60 }
61
62 ImageBuffer(const ImageBufferBackend::Parameters& parameters, Recorder::Delegate* delegate = nullptr)
63 : BaseConcreteImageBuffer(parameters)
64 , m_drawingContext(logicalSize(), baseTransform(), delegate)
65 , m_writingClient(WTF::makeUnique<InMemoryDisplayList::WritingClient>())
66 , m_readingClient(WTF::makeUnique<InMemoryDisplayList::ReadingClient>())
67 {
68 m_drawingContext.displayList().setItemBufferWritingClient(m_writingClient.get());
69 m_drawingContext.displayList().setItemBufferReadingClient(m_readingClient.get());
70 }
71
72 ~ImageBuffer()
73 {
74 flushDrawingContext();
75 }
76
77 GraphicsContext& context() const override
78 {
79 return m_drawingContext.context();
80 }
81
82 DrawingContext* drawingContext() override { return &m_drawingContext; }
83
84 void flushDrawingContext() override
85 {
86 if (!m_drawingContext.displayList().isEmpty())
87 m_drawingContext.replayDisplayList(BaseConcreteImageBuffer::context());
88 }
89
90 void clearBackend() override
91 {
92 m_drawingContext.displayList().clear();
93 BaseConcreteImageBuffer::clearBackend();
94 }
95
96protected:
97 DrawingContext m_drawingContext;
98 std::unique_ptr<ItemBufferWritingClient> m_writingClient;
99 std::unique_ptr<ItemBufferReadingClient> m_readingClient;
100};
101
102} // DisplayList
103} // namespace WebCore
