36// Used to represent a given update. An value of { 3, 60 } indicates that this is the third update in a 1-second interval
37// on a 60fps cadence. updateIndex will reset to zero every second, so { 59, 60 } is followed by { 0, 60 }.
38struct DisplayUpdate {
39 unsigned updateIndex { 0 };
40 FramesPerSecond updatesPerSecond { 0 };
41
42 DisplayUpdate nextUpdate() const
43 {
44 return { (updateIndex + 1) % updatesPerSecond, updatesPerSecond };
45 }
46
47 WEBCORE_EXPORT bool relevantForUpdateFrequency(FramesPerSecond) const;
48
49 template<class Encoder> void encode(Encoder&) const;
50 template<class Decoder> static std::optional<DisplayUpdate> decode(Decoder&);
51};
52
53template<class Encoder>
54void DisplayUpdate::encode(Encoder& encoder) const
55{
56 encoder << updateIndex;
57 encoder << updatesPerSecond;
58}
59
60template<class Decoder>
61std::optional<DisplayUpdate> DisplayUpdate::decode(Decoder& decoder)
62{
63 std::optional<unsigned> updateIndex;
64 decoder >> updateIndex;
65 if (!updateIndex)
66 return std::nullopt;
67
68 std::optional<FramesPerSecond> updatesPerSecond;
69 decoder >> updatesPerSecond;
70 if (!updatesPerSecond)
71 return std::nullopt;
72
73 return {{ *updateIndex, *updatesPerSecond }};
74}

76WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, const DisplayUpdate&);
