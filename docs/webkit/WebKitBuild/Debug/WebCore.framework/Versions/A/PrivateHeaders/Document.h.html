354 // Nodes belonging to this document increase referencingNodeCount -
355 // these are enough to keep the document from being destroyed, but
356 // not enough to keep it from removing its children. This allows a
357 // node that outlives its document to still have a valid document
358 // pointer without introducing reference cycles.
359 void incrementReferencingNodeCount()
360 {
361 ASSERT(!m_deletionHasBegun);
362 ++m_referencingNodeCount;
363 }
364
365 void decrementReferencingNodeCount()
366 {
367 ASSERT(!m_deletionHasBegun || !m_referencingNodeCount);
368 --m_referencingNodeCount;
369 if (!m_referencingNodeCount && !refCount()) {
370#if ASSERT_ENABLED
371 m_deletionHasBegun = true;
372#endif
373 m_refCountAndParentBit = s_refCountIncrement; // Avoid double destruction through use of Ref<T>/RefPtr<T>. (This is a security mitigation in case of programmer error. It will ASSERT in debug builds.)
374 delete this;
375 }
376 }
