27#pragma once
28
29#include "ResourceLoadTiming.h"
30
31namespace WebCore {
32
33class DocumentLoadTiming : public ResourceLoadTiming {
34public:
35 // https://www.w3.org/TR/hr-time-2/#dfn-time-origin
36 MonotonicTime timeOrigin() const { return startTime(); }
37
38 void markUnloadEventStart() { m_unloadEventStart = MonotonicTime::now(); }
39 void markUnloadEventEnd() { m_unloadEventEnd = MonotonicTime::now(); }
40 void markLoadEventStart() { m_loadEventStart = MonotonicTime::now(); }
41 void markLoadEventEnd() { m_loadEventEnd = MonotonicTime::now(); }
42
43 void setHasSameOriginAsPreviousDocument(bool value) { m_hasSameOriginAsPreviousDocument = value; }
44
45 MonotonicTime unloadEventStart() const { return m_unloadEventStart; }
46 MonotonicTime unloadEventEnd() const { return m_unloadEventEnd; }
47 MonotonicTime loadEventStart() const { return m_loadEventStart; }
48 MonotonicTime loadEventEnd() const { return m_loadEventEnd; }
49 bool hasSameOriginAsPreviousDocument() const { return m_hasSameOriginAsPreviousDocument; }
50
51private:
52 MonotonicTime m_unloadEventStart;
53 MonotonicTime m_unloadEventEnd;
54 MonotonicTime m_loadEventStart;
55 MonotonicTime m_loadEventEnd;
56 bool m_hasSameOriginAsPreviousDocument { false };
57};
58
59} // namespace WebCore
