31#include <wtf/WeakPtr.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class Document;
37class DocumentParser;
38class Frame;
39class TextResourceDecoder;
40
41class DocumentWriter {
42 WTF_MAKE_NONCOPYABLE(DocumentWriter);
43public:
44 DocumentWriter() = default;
45
46 void replaceDocumentWithResultOfExecutingJavascriptURL(const String&, Document* ownerDocument);
47
48 bool begin();
49 bool begin(const URL&, bool dispatchWindowObjectAvailable = true, Document* ownerDocument = nullptr);
50 void addData(const uint8_t* bytes, size_t length);
51 void insertDataSynchronously(const String&); // For an internal use only to prevent the parser from yielding.
52 WEBCORE_EXPORT void end();
53
54 void setFrame(Frame&);
55
56 WEBCORE_EXPORT void setEncoding(const String& encoding, bool userChosen);
57
58 const String& mimeType() const { return m_mimeType; }
59 void setMIMEType(const String& type) { m_mimeType = type; }
60
61 // Exposed for DocumentParser::appendBytes.
62 TextResourceDecoder& decoder();
63 void reportDataReceived();
64
65 void setDocumentWasLoadedAsPartOfNavigation();
66
67private:
68 Ref<Document> createDocument(const URL&);
69 void clear();
70
71 WeakPtr<Frame> m_frame;
72
73 bool m_hasReceivedSomeData { false };
74 String m_mimeType;
75
76 bool m_encodingWasChosenByUser { false };
77 String m_encoding;
78 RefPtr<TextResourceDecoder> m_decoder;
79 RefPtr<DocumentParser> m_parser;
80
81 enum class State { NotStarted, Started, Finished };
82 State m_state { State::NotStarted };
83};

85} // namespace WebCore
