60namespace WebCore {
61
62enum class DragApplicationFlags : uint8_t {
63 IsModal = 1,
64 IsSource = 2,
65 HasAttachedSheet = 4,
66 IsCopyKeyDown = 8
67};
68
69class PasteboardContext;
70
71#if PLATFORM(WIN)
72typedef HashMap<unsigned, Vector<String>> DragDataMap;
73#endif
74
75class DragData {
76public:
77 enum FilenameConversionPolicy { DoNotConvertFilenames, ConvertFilenames };
78 enum class DraggingPurpose { ForEditing, ForFileUpload, ForColorControl };
79
80 // clientPosition is taken to be the position of the drag event within the target window, with (0,0) at the top left
81 WEBCORE_EXPORT DragData(DragDataRef, const IntPoint& clientPosition, const IntPoint& globalPosition, OptionSet<DragOperation>, OptionSet<DragApplicationFlags> = { }, OptionSet<DragDestinationAction> = anyDragDestinationAction(), std::optional<PageIdentifier> pageID = std::nullopt);
82 WEBCORE_EXPORT DragData(const String& dragStorageName, const IntPoint& clientPosition, const IntPoint& globalPosition, OptionSet<DragOperation>, OptionSet<DragApplicationFlags> = { }, OptionSet<DragDestinationAction> = anyDragDestinationAction(), std::optional<PageIdentifier> pageID = std::nullopt);
83 // This constructor should used only by WebKit2 IPC because DragData
84 // is initialized by the decoder and not in the constructor.
85 DragData() = default;
86#if PLATFORM(WIN)
87 WEBCORE_EXPORT DragData(const DragDataMap&, const IntPoint& clientPosition, const IntPoint& globalPosition, OptionSet<DragOperation> sourceOperationMask, OptionSet<DragApplicationFlags> = { }, std::optional<PageIdentifier> pageID = std::nullopt);
88 const DragDataMap& dragDataMap();
89 void getDragFileDescriptorData(int& size, String& pathname);
90 void getDragFileContentData(int size, void* dataBlob);
91#endif
92 const IntPoint& clientPosition() const { return m_clientPosition; }
93 const IntPoint& globalPosition() const { return m_globalPosition; }
94 OptionSet<DragApplicationFlags> flags() const { return m_applicationFlags; }
95 DragDataRef platformData() const { return m_platformDragData; }
96 OptionSet<DragOperation> draggingSourceOperationMask() const { return m_draggingSourceOperationMask; }
97 bool containsURL(FilenameConversionPolicy = ConvertFilenames) const;
98 bool containsPlainText() const;
99 bool containsCompatibleContent(DraggingPurpose = DraggingPurpose::ForEditing) const;
100 String asURL(FilenameConversionPolicy = ConvertFilenames, String* title = nullptr) const;
101 String asPlainText() const;
102 Vector<String> asFilenames() const;
103 Color asColor() const;
104 bool canSmartReplace() const;
105 bool containsColor() const;
106 bool containsFiles() const;
107 unsigned numberOfFiles() const;
108 OptionSet<DragDestinationAction> dragDestinationActionMask() const { return m_dragDestinationActionMask; }
109 void setFileNames(Vector<String>& fileNames) { m_fileNames = WTFMove(fileNames); }
110 const Vector<String>& fileNames() const { return m_fileNames; }
111#if PLATFORM(COCOA)
112 const String& pasteboardName() const { return m_pasteboardName; }
113 bool containsURLTypeIdentifier() const;
114 bool containsPromise() const;
115#endif
116
117 std::optional<PageIdentifier> pageID() const { return m_pageID; }
118
119 std::unique_ptr<PasteboardContext> createPasteboardContext() const;
120
121private:
122 IntPoint m_clientPosition;
123 IntPoint m_globalPosition;
124 DragDataRef m_platformDragData;
125 OptionSet<DragOperation> m_draggingSourceOperationMask;
126 OptionSet<DragApplicationFlags> m_applicationFlags;
127 Vector<String> m_fileNames;
128 OptionSet<DragDestinationAction> m_dragDestinationActionMask;
129 std::optional<PageIdentifier> m_pageID;
130#if PLATFORM(COCOA)
131 String m_pasteboardName;
132#endif
133#if PLATFORM(WIN)
134 DragDataMap m_dragDataMap;
135#endif
136};
