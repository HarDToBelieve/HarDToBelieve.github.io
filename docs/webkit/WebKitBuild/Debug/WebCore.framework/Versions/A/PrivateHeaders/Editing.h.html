208inline bool deprecatedIsEditingWhitespace(UChar c)
209{
210 return c == noBreakSpace || c == ' ' || c == '\n' || c == '\t';
211}
212
213// FIXME: Can't really answer this question correctly without knowing the white-space mode.
214inline bool deprecatedIsCollapsibleWhitespace(UChar c)
215{
216 return c == ' ' || c == '\n';
217}
218
219bool isAmbiguousBoundaryCharacter(UChar);
220
221inline bool editingIgnoresContent(const Node& node)
222{
223 return !node.canContainRangeEndPoint();
224}
225
226inline bool positionBeforeOrAfterNodeIsCandidate(Node& node)
227{
228 return isRenderedTable(&node) || editingIgnoresContent(node);
229}
230
231inline Position firstPositionInOrBeforeNode(Node* node)
232{
233 if (!node)
234 return { };
235 return editingIgnoresContent(*node) ? positionBeforeNode(node) : firstPositionInNode(node);
236}
237
238inline Position lastPositionInOrAfterNode(Node* node)
239{
240 if (!node)
241 return { };
242 return editingIgnoresContent(*node) ? positionAfterNode(node) : lastPositionInNode(node);
243}
