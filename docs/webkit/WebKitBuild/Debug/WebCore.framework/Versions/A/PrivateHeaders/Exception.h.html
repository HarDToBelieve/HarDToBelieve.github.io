27#pragma once
28
29#include "ExceptionCode.h"
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34class Exception {
35public:
36 explicit Exception(ExceptionCode, String = { });
37
38 ExceptionCode code() const { return m_code; }
39 const String& message() const { return m_message; }
40 String&& releaseMessage() { return WTFMove(m_message); }
41
42 Exception isolatedCopy() const
43 {
44 return Exception { m_code, m_message.isolatedCopy() };
45 }
46
47private:
48 ExceptionCode m_code;
49 String m_message;
50};
51
52Exception isolatedCopy(Exception&&);
53
54inline Exception::Exception(ExceptionCode code, String message)
55 : m_code { code }
56 , m_message { WTFMove(message) }
57{
58}
59
60inline Exception isolatedCopy(Exception&& value)
61{
62 return Exception { value.code(), value.releaseMessage().isolatedCopy() };
63}
64
65}
