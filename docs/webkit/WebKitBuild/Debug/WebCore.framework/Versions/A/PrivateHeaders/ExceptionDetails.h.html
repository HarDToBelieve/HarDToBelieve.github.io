26#pragma once
27
28#include <wtf/text/WTFString.h>
29
30namespace WebCore {
31
32struct ExceptionDetails {
33 enum class Type : uint8_t {
34 Script,
35 InvalidTargetFrame,
36 AppBoundDomain,
37 };
38
39 String message;
40 int lineNumber { 0 };
41 int columnNumber { 0 };
42 Type type { Type::Script };
43
44 // This bizarre explicit initialization of String is because older compilers (like on High Sierra)
45 // don't properly handle partial initialization lists unless every struct member has an explicit default value.
46 // Once we stop building on those platforms we can remove this.
47 String sourceURL { };
48};
49
50} // namespace WebCore
51
52namespace WTF {
53template<> struct EnumTraits<WebCore::ExceptionDetails::Type> {
54 using values = EnumValues<
55 WebCore::ExceptionDetails::Type,
56 WebCore::ExceptionDetails::Type::Script,
57 WebCore::ExceptionDetails::Type::InvalidTargetFrame,
58 WebCore::ExceptionDetails::Type::AppBoundDomain
59 >;
60};
61}
