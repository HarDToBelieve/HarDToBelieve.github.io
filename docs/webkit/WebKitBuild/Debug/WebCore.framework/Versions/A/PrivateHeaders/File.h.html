26#pragma once
27
28#include "Blob.h"
29#include <wtf/IsoMalloc.h>
30#include <wtf/Ref.h>
31#include <wtf/TypeCasts.h>
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class File final : public Blob {
37 WTF_MAKE_ISO_ALLOCATED_EXPORT(File, WEBCORE_EXPORT);
38public:
39 struct PropertyBag : BlobPropertyBag {
40 std::optional<int64_t> lastModified;
41 };
42
43 // Create a file with an optional name exposed to the author (via File.name and associated DOM properties) that differs from the one provided in the path.
44 WEBCORE_EXPORT static Ref<File> create(ScriptExecutionContext*, const String& path, const String& replacementPath = { }, const String& nameOverride = { });
45
46 // Create a File using the 'new File' constructor.
47 static Ref<File> create(ScriptExecutionContext& context, Vector<BlobPartVariant>&& blobPartVariants, const String& filename, const PropertyBag& propertyBag)
48 {
49 auto file = adoptRef(*new File(context, WTFMove(blobPartVariants), filename, propertyBag));
50 file->suspendIfNeeded();
51 return file;
52 }
53
54 static Ref<File> deserialize(ScriptExecutionContext* context, const String& path, const URL& srcURL, const String& type, const String& name, const std::optional<int64_t>& lastModified = std::nullopt)
55 {
56 auto file = adoptRef(*new File(deserializationContructor, context, path, srcURL, type, name, lastModified));
57 file->suspendIfNeeded();
58 return file;
59 }
60
61 static Ref<File> create(ScriptExecutionContext* context, const Blob& blob, const String& name)
62 {
63 auto file = adoptRef(*new File(context, blob, name));
64 file->suspendIfNeeded();
65 return file;
66 }
67
68 static Ref<File> create(ScriptExecutionContext* context, const File& existingFile, const String& name)
69 {
70 auto file = adoptRef(*new File(context, existingFile, name));
71 file->suspendIfNeeded();
72 return file;
73 }
74
75 static Ref<File> createWithRelativePath(ScriptExecutionContext*, const String& path, const String& relativePath);
76
77 bool isFile() const override { return true; }
78
79 const String& path() const { return m_path; }
80 const String& relativePath() const { return m_relativePath; }
81 void setRelativePath(const String& relativePath) { m_relativePath = relativePath; }
82 const String& name() const { return m_name; }
83 WEBCORE_EXPORT int64_t lastModified() const; // Number of milliseconds since Epoch.
84 const std::optional<int64_t>& lastModifiedOverride() const { return m_lastModifiedDateOverride; } // Number of milliseconds since Epoch.
85
86 static String contentTypeForFile(const String& path);
87
88#if ENABLE(FILE_REPLACEMENT)
89 static bool shouldReplaceFile(const String& path);
90#endif
91
92 bool isDirectory() const;
93
94private:
95 WEBCORE_EXPORT explicit File(ScriptExecutionContext*, const String& path);
96 File(ScriptExecutionContext*, URL&&, String&& type, String&& path, String&& name);
97 File(ScriptExecutionContext&, Vector<BlobPartVariant>&& blobPartVariants, const String& filename, const PropertyBag&);
98 File(ScriptExecutionContext*, const Blob&, const String& name);
99 File(ScriptExecutionContext*, const File&, const String& name);
100
101 File(DeserializationContructor, ScriptExecutionContext*, const String& path, const URL& srcURL, const String& type, const String& name, const std::optional<int64_t>& lastModified);
102
103 static void computeNameAndContentType(const String& path, const String& nameOverride, String& effectiveName, String& effectiveContentType);
104#if ENABLE(FILE_REPLACEMENT)
105 static void computeNameAndContentTypeForReplacedFile(const String& path, const String& nameOverride, String& effectiveName, String& effectiveContentType);
106#endif
107
108 // ActiveDOMObject.
109 const char* activeDOMObjectName() const final;
110
111 String m_path;
112 String m_relativePath;
113 String m_name;
114
115 std::optional<int64_t> m_lastModifiedDateOverride;
116 mutable std::optional<bool> m_isDirectory;
117};
118
119} // namespace WebCore
120
121SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::File)
122 static bool isType(const WebCore::Blob& blob) { return blob.isFile(); }
123SPECIALIZE_TYPE_TRAITS_END()
