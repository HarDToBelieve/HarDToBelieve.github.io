30#pragma once
31
32#include <wtf/EnumTraits.h>
33#include <wtf/RefCounted.h>
34#include <wtf/Vector.h>
35#include <wtf/text/WTFString.h>
36
37namespace WebCore {
38
39enum MediaCaptureType {
40 MediaCaptureTypeNone,
41 MediaCaptureTypeUser,
42 MediaCaptureTypeEnvironment
43};
44
45class FileChooser;
46class Icon;
47
48struct FileChooserFileInfo {
49 FileChooserFileInfo isolatedCopy() const
50 {
51 return { path.isolatedCopy(), replacementPath.isolatedCopy(), displayName.isolatedCopy() };
52 }
53
54 const String path;
55 const String replacementPath;
56 const String displayName;
57};
58
59struct FileChooserSettings {
60 bool allowsDirectories { false };
61 bool allowsMultipleFiles { false };
62 Vector<String> acceptMIMETypes;
63 Vector<String> acceptFileExtensions;
64 Vector<String> selectedFiles;
65#if ENABLE(MEDIA_CAPTURE)
66 MediaCaptureType mediaCaptureType { MediaCaptureTypeNone };
67#endif
68};
69
70class FileChooserClient {
71public:
72 virtual ~FileChooserClient() = default;
73
74 virtual void filesChosen(const Vector<FileChooserFileInfo>&, const String& displayString = { }, Icon* = nullptr) = 0;
75};
76
77class FileChooser : public RefCounted<FileChooser> {
78public:
79 static Ref<FileChooser> create(FileChooserClient*, const FileChooserSettings&);
80 WEBCORE_EXPORT ~FileChooser();
81
82 void invalidate();
83
84 WEBCORE_EXPORT void chooseFile(const String& path);
85 WEBCORE_EXPORT void chooseFiles(const Vector<String>& paths, const Vector<String>& replacementPaths = { });
86#if PLATFORM(IOS_FAMILY)
87 // FIXME: This function is almost identical to FileChooser::chooseFiles(). We should merge this
88 // function with FileChooser::chooseFiles() and hence remove the PLATFORM(IOS_FAMILY)-guard.
89 WEBCORE_EXPORT void chooseMediaFiles(const Vector<String>& paths, const String& displayString, Icon*);
90#endif
91
92 // FIXME: We should probably just pass file paths that could be virtual paths with proper display names rather than passing structs.
93 void chooseFiles(const Vector<FileChooserFileInfo>& files);
94
95 const FileChooserSettings& settings() const { return m_settings; }
96
97private:
98 FileChooser(FileChooserClient*, const FileChooserSettings&);
99
100 FileChooserClient* m_client { nullptr };
101 FileChooserSettings m_settings;
102};
103
104} // namespace WebCore

115};
