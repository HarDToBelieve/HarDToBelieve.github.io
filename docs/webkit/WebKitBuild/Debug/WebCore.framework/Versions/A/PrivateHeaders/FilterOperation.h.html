51class FilterOperation : public ThreadSafeRefCounted<FilterOperation> {
52public:
53 enum OperationType {
54 REFERENCE, // url(#somefilter)
55 GRAYSCALE,
56 SEPIA,
57 SATURATE,
58 HUE_ROTATE,
59 INVERT,
60 APPLE_INVERT_LIGHTNESS,
61 OPACITY,
62 BRIGHTNESS,
63 CONTRAST,
64 BLUR,
65 DROP_SHADOW,
66 PASSTHROUGH,
67 DEFAULT,
68 NONE
69 };
70
71 virtual ~FilterOperation() = default;
72
73 virtual Ref<FilterOperation> clone() const = 0;
74
75 virtual bool operator==(const FilterOperation&) const = 0;
76 bool operator!=(const FilterOperation& o) const { return !(*this == o); }
77
78 virtual RefPtr<FilterOperation> blend(const FilterOperation* /*from*/, const BlendingContext&, bool /*blendToPassthrough*/ = false)
79 {
80 return nullptr;
81 }
82
83 virtual bool transformColor(SRGBA<float>&) const { return false; }
84 virtual bool inverseTransformColor(SRGBA<float>&) const { return false; }
85
86 OperationType type() const { return m_type; }
87
88 bool isBasicColorMatrixFilterOperation() const
89 {
90 return m_type == GRAYSCALE || m_type == SEPIA || m_type == SATURATE || m_type == HUE_ROTATE;
91 }
92
93 bool isBasicComponentTransferFilterOperation() const
94 {
95 return m_type == INVERT || m_type == BRIGHTNESS || m_type == CONTRAST || m_type == OPACITY;
96 }
97
98 bool isSameType(const FilterOperation& o) const { return o.type() == m_type; }
99
100 // True if the alpha channel of any pixel can change under this operation.
101 virtual bool affectsOpacity() const { return false; }
102 // True if the value of one pixel can affect the value of another pixel under this operation, such as blur.
103 virtual bool movesPixels() const { return false; }
104 // True if the filter should not be allowed to work on content that is not available from this security origin.
105 virtual bool shouldBeRestrictedBySecurityOrigin() const { return false; }
106
107protected:
108 FilterOperation(OperationType type)
109 : m_type(type)
110 {
111 }
112
113 OperationType m_type;
114};

166class ReferenceFilterOperation : public FilterOperation {
167public:
168 static Ref<ReferenceFilterOperation> create(const String& url, const String& fragment)
169 {
170 return adoptRef(*new ReferenceFilterOperation(url, fragment));
171 }
172 virtual ~ReferenceFilterOperation();
173
174 Ref<FilterOperation> clone() const final
175 {
176 // Reference filters cannot be cloned.
177 RELEASE_ASSERT_NOT_REACHED();
178 }
179
180 bool affectsOpacity() const override { return true; }
181 bool movesPixels() const override { return true; }
182 // FIXME: This only needs to return true for graphs that include ConvolveMatrix, DisplacementMap, Morphology and possibly Lighting.
183 // https://bugs.webkit.org/show_bug.cgi?id=171753
184 bool shouldBeRestrictedBySecurityOrigin() const override { return true; }
185
186 const String& url() const { return m_url; }
187 const String& fragment() const { return m_fragment; }
188
189 void loadExternalDocumentIfNeeded(CachedResourceLoader&, const ResourceLoaderOptions&);
190
191 CachedSVGDocumentReference* cachedSVGDocumentReference() const { return m_cachedSVGDocumentReference.get(); }
192
193private:
194 ReferenceFilterOperation(const String& url, const String& fragment);
195
196 bool operator==(const FilterOperation&) const override;
197
198 String m_url;
199 String m_fragment;
200 std::unique_ptr<CachedSVGDocumentReference> m_cachedSVGDocumentReference;
201};

215 }
216
217 double amount() const { return m_amount; }
218
219 RefPtr<FilterOperation> blend(const FilterOperation* from, const BlendingContext&, bool blendToPassthrough = false) override;
220
221private:
222 bool operator==(const FilterOperation&) const override;
223
224 double passthroughAmount() const;
225
226 BasicColorMatrixFilterOperation(double amount, OperationType type)
227 : FilterOperation(type)

229 {
230 }

329class WEBCORE_EXPORT DropShadowFilterOperation : public FilterOperation {
330public:
331 static Ref<DropShadowFilterOperation> create(const IntPoint& location, int stdDeviation, const Color& color)

333 return adoptRef(*new DropShadowFilterOperation(location, stdDeviation, color));

335

337 {
338 return adoptRef(*new DropShadowFilterOperation(location(), stdDeviation(), color()));

342 int y() const { return m_location.y(); }
343 IntPoint location() const { return m_location; }
344 int stdDeviation() const { return m_stdDeviation; }
345 const Color& color() const { return m_color; }
346
347 bool affectsOpacity() const override { return true; }
348 bool movesPixels() const override { return true; }
349
350 RefPtr<FilterOperation> blend(const FilterOperation* from, const BlendingContext&, bool blendToPassthrough = false) override;
351
352private:
353 bool operator==(const FilterOperation&) const override;
354
355 DropShadowFilterOperation(const IntPoint& location, int stdDeviation, const Color& color)
356 : FilterOperation(DROP_SHADOW)
357 , m_location(location)
358 , m_stdDeviation(stdDeviation)
359 , m_color(color)
360 {
361 }
362
363 IntPoint m_location; // FIXME: should location be in Lengths?
364 int m_stdDeviation;
365 Color m_color;
