29#pragma once
30
31#include "FloatPoint.h"
32#include "FloatQuad.h"
33#include "FloatRect.h"
34#include <optional>
35
36namespace WebCore {
37
38class FloatLine {
39public:
40 FloatLine() { }
41 FloatLine(const FloatPoint& start, const FloatPoint& end)
42 : m_start(start)
43 , m_end(end)
44 , m_length(sqrtf(powf(start.x() - end.x(), 2) + powf(start.y() - end.y(), 2)))
45 {
46 }
47
48 const FloatPoint& start() const { return m_start; }
49 const FloatPoint& end() const { return m_end; }
50
51 float length() const { return m_length; }
52
53 WEBCORE_EXPORT const FloatPoint pointAtAbsoluteDistance(float) const;
54 const FloatPoint pointAtRelativeDistance(float) const;
55 const FloatLine extendedToBounds(const FloatRect&) const;
56 const std::optional<FloatPoint> intersectionWith(const FloatLine&) const;
57
58 template<class Encoder> void encode(Encoder&) const;
59 template<class Decoder> static std::optional<FloatLine> decode(Decoder&);
60
61private:
62 FloatPoint m_start { 0, 0 };
63 FloatPoint m_end { 0, 0 };
64 float m_length { 0 };
65};
66
67template<class Encoder> void FloatLine::encode(Encoder& encoder) const
68{
69 encoder << m_start;
70 encoder << m_end;
71}
72
73template<class Decoder> std::optional<FloatLine> FloatLine::decode(Decoder& decoder)
74{
75 std::optional<FloatPoint> start;
76 decoder >> start;
77 if (!start)
78 return std::nullopt;
79
80 std::optional<FloatPoint> end;
81 decoder >> end;
82 if (!end)
83 return std::nullopt;
84
85 return {{ *start, *end }};
86}
