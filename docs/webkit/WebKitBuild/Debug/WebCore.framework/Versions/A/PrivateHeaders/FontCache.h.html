81// This key contains the FontDescription fields other than family that matter when fetching FontDatas (platform fonts).
82struct FontDescriptionKey {
83 FontDescriptionKey() = default;
84
85 FontDescriptionKey(const FontDescription& description)
86 : m_size(description.computedPixelSize())
87 , m_fontSelectionRequest(description.fontSelectionRequest())
88 , m_flags(makeFlagsKey(description))
89 , m_locale(description.specifiedLocale())
90 , m_featureSettings(description.featureSettings())
91 , m_variationSettings(description.variationSettings())
92 { }
93
94 explicit FontDescriptionKey(WTF::HashTableDeletedValueType)
95 : m_isDeletedValue(true)
96 { }
97
98 bool operator==(const FontDescriptionKey& other) const
99 {
100 return m_isDeletedValue == other.m_isDeletedValue
101 && m_size == other.m_size
102 && m_fontSelectionRequest == other.m_fontSelectionRequest
103 && m_flags == other.m_flags
104 && m_locale == other.m_locale
105 && m_variationSettings == other.m_variationSettings
106 && m_featureSettings == other.m_featureSettings;
107 }
108
109 bool operator!=(const FontDescriptionKey& other) const
110 {
111 return !(*this == other);
112 }
113
114 bool isHashTableDeletedValue() const { return m_isDeletedValue; }
115
116 friend void add(Hasher&, const FontDescriptionKey&);
117
118private:
119 static std::array<unsigned, 2> makeFlagsKey(const FontDescription& description)
120 {
121 unsigned first = static_cast<unsigned>(description.script()) << 15
122 | static_cast<unsigned>(description.shouldDisableLigaturesForSpacing()) << 14
123 | static_cast<unsigned>(description.shouldAllowUserInstalledFonts()) << 13
124 | static_cast<unsigned>(description.fontStyleAxis() == FontStyleAxis::slnt) << 12
125 | static_cast<unsigned>(description.opticalSizing()) << 11
126 | static_cast<unsigned>(description.textRenderingMode()) << 9
127 | static_cast<unsigned>(description.fontSynthesis()) << 6
128 | static_cast<unsigned>(description.widthVariant()) << 4
129 | static_cast<unsigned>(description.nonCJKGlyphOrientation()) << 3
130 | static_cast<unsigned>(description.orientation()) << 2
131 | static_cast<unsigned>(description.renderingMode());
132 unsigned second = static_cast<unsigned>(description.variantEastAsianRuby()) << 27
133 | static_cast<unsigned>(description.variantEastAsianWidth()) << 25
134 | static_cast<unsigned>(description.variantEastAsianVariant()) << 22
135 | static_cast<unsigned>(description.variantAlternates()) << 21
136 | static_cast<unsigned>(description.variantNumericSlashedZero()) << 20
137 | static_cast<unsigned>(description.variantNumericOrdinal()) << 19
138 | static_cast<unsigned>(description.variantNumericFraction()) << 17
139 | static_cast<unsigned>(description.variantNumericSpacing()) << 15
140 | static_cast<unsigned>(description.variantNumericFigure()) << 13
141 | static_cast<unsigned>(description.variantCaps()) << 10
142 | static_cast<unsigned>(description.variantPosition()) << 8
143 | static_cast<unsigned>(description.variantContextualAlternates()) << 6
144 | static_cast<unsigned>(description.variantHistoricalLigatures()) << 4
145 | static_cast<unsigned>(description.variantDiscretionaryLigatures()) << 2
146 | static_cast<unsigned>(description.variantCommonLigatures());
147 return {{ first, second }};
148 }
149
150 bool m_isDeletedValue { false };
151 unsigned m_size { 0 };
152 FontSelectionRequest m_fontSelectionRequest;
153 std::array<unsigned, 2> m_flags {{ 0, 0 }};
154 AtomString m_locale;
155 FontFeatureSettings m_featureSettings;
156 FontVariationSettings m_variationSettings;
157};
158
159inline void add(Hasher& hasher, const FontDescriptionKey& key)
160{
161 add(hasher, key.m_size, key.m_fontSelectionRequest, key.m_flags, key.m_locale, key.m_featureSettings, key.m_variationSettings);
162}

164struct FontDescriptionKeyHash {
165 static unsigned hash(const FontDescriptionKey& key) { return computeHash(key); }
166 static bool equal(const FontDescriptionKey& a, const FontDescriptionKey& b) { return a == b; }
167 static constexpr bool safeToCompareToEmptyOrDeleted = true;
168};

208struct FontCascadeCacheKeyHashTraits : HashTraits<FontCascadeCacheKey> {
209 static FontCascadeCacheKey emptyValue() { return { }; }
210 static void constructDeletedValue(FontCascadeCacheKey& slot) { new (NotNull, &slot.fontDescriptionKey) FontDescriptionKey(WTF::HashTableDeletedValue); }
211 static bool isDeletedValue(const FontCascadeCacheKey& key) { return key.fontDescriptionKey.isHashTableDeletedValue(); }

352inline bool FontCache::PrewarmInformation::isEmpty() const
353{
354 return seenFamilies.isEmpty() && fontNamesRequiringSystemFallback.isEmpty();
355}
356
357inline FontCache::PrewarmInformation FontCache::PrewarmInformation::isolatedCopy() const
358{
359 return { seenFamilies.isolatedCopy(), fontNamesRequiringSystemFallback.isolatedCopy() };
360}
361
362template<class Encoder>
363void FontCache::PrewarmInformation::encode(Encoder& encoder) const
364{
365 encoder << seenFamilies;
366 encoder << fontNamesRequiringSystemFallback;
367}
368
369template<class Decoder>
370std::optional<FontCache::PrewarmInformation> FontCache::PrewarmInformation::decode(Decoder& decoder)
371{
372 PrewarmInformation prewarmInformation;
373 if (!decoder.decode(prewarmInformation.seenFamilies))
374 return { };
375 if (!decoder.decode(prewarmInformation.fontNamesRequiringSystemFallback))
376 return { };
377
378 return prewarmInformation;
379}
