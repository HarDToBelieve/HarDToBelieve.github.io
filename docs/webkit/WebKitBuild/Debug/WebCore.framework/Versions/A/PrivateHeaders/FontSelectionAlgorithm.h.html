36// Unclamped, unchecked, signed fixed-point number representing a value used for font variations.
37// Sixteen bits in total, one sign bit, two fractional bits, smallest positive value is 0.25,
38// maximum value is 8191.75, and minimum value is -8192.
39class FontSelectionValue {
40public:
41 using BackingType = int16_t;
42
43 FontSelectionValue() = default;
44
45 // Explicit because it won't work correctly for values outside the representable range.
46 explicit constexpr FontSelectionValue(int);
47
48 // Explicit because it won't work correctly for values outside the representable range and because precision can be lost.
49 explicit constexpr FontSelectionValue(float);
50
51 // Precision can be lost, but value will be clamped to the representable range.
52 static constexpr FontSelectionValue clampFloat(float);
53
54 // Since floats have 23 mantissa bits, every value can be represented losslessly.
55 constexpr operator float() const;
56
57 static constexpr FontSelectionValue maximumValue();
58 static constexpr FontSelectionValue minimumValue();
59
60 friend constexpr FontSelectionValue operator+(FontSelectionValue, FontSelectionValue);
61 friend constexpr FontSelectionValue operator-(FontSelectionValue, FontSelectionValue);
62 friend constexpr FontSelectionValue operator*(FontSelectionValue, FontSelectionValue);
63 friend constexpr FontSelectionValue operator/(FontSelectionValue, FontSelectionValue);
64 friend constexpr FontSelectionValue operator-(FontSelectionValue);
65
66 constexpr BackingType rawValue() const { return m_backing; }
67
68 template<class Encoder>
69 void encode(Encoder&) const;
70
71 template<class Decoder>
72 static std::optional<FontSelectionValue> decode(Decoder&);
73
74private:
75 enum class RawTag { RawTag };
76 constexpr FontSelectionValue(int, RawTag);
77
78 static constexpr int fractionalEntropy = 4;
79 BackingType m_backing { 0 };
80};
81
82template<class Encoder>
83void FontSelectionValue::encode(Encoder& encoder) const
84{
85 encoder << m_backing;
86}
87
88template<class Decoder>
89std::optional<FontSelectionValue> FontSelectionValue::decode(Decoder& decoder)
90{
91 std::optional<FontSelectionValue::BackingType> backing;
92 decoder >> backing;
93 if (!backing)
94 return std::nullopt;
95
96 FontSelectionValue result;
97 result.m_backing = *backing;
98 return result;
99}

299struct FontSelectionRange {
300 using Value = FontSelectionValue;
301
302 constexpr FontSelectionRange(Value minimum, Value maximum)
303 : minimum(minimum)
304 , maximum(maximum)
305 {
306 }
307
308 explicit constexpr FontSelectionRange(Value value)
309 : minimum(value)
310 , maximum(value)
311 {
312 }
313
314 constexpr bool operator==(const FontSelectionRange& other) const
315 {
316 return WTF::tie(minimum, maximum) == WTF::tie(other.minimum, other.maximum);
317 }
318
319 constexpr bool isValid() const
320 {
321 return minimum <= maximum;
322 }
323
324 void expand(const FontSelectionRange& other)
325 {
326 ASSERT(other.isValid());
327 if (!isValid())
328 *this = other;
329 else {
330 minimum = std::min(minimum, other.minimum);
331 maximum = std::max(maximum, other.maximum);
332 }
333 ASSERT(isValid());
334 }
335
336 constexpr bool includes(Value target) const
337 {
338 return target >= minimum && target <= maximum;
339 }
340
341 template<class Encoder>
342 void encode(Encoder&) const;
343
344 template<class Decoder>
345 static std::optional<FontSelectionRange> decode(Decoder&);
346
347 Value minimum { 1 };
348 Value maximum { 0 };
349};
350
351template<class Encoder>
352void FontSelectionRange::encode(Encoder& encoder) const
353{
354 encoder << minimum;
355 encoder << maximum;
356}
357
358template<class Decoder>
359std::optional<FontSelectionRange> FontSelectionRange::decode(Decoder& decoder)
360{
361 std::optional<FontSelectionRange::Value> minimum;
362 decoder >> minimum;
363 if (!minimum)
364 return std::nullopt;
365
366 std::optional<FontSelectionRange::Value> maximum;
367 decoder >> maximum;
368 if (!maximum)
369 return std::nullopt;
370
371 return {{ *minimum, *maximum }};
372}
373
374inline void add(Hasher& hasher, const FontSelectionRange& range)
375{
376 add(hasher, range.minimum, range.maximum);
377}

379struct FontSelectionRequest {
380 using Value = FontSelectionValue;
381
382 Value weight;
383 Value width;
384
385 // FIXME: We are using an optional here to be able to distinguish between an explicit
386 // or implicit slope (for "italic" and "oblique") and the "normal" value which has no
387 // slope. The "italic" and "oblique" values can be distinguished by looking at the
388 // "fontStyleAxis" on the FontDescription. We should come up with a tri-state member
389 // so that it's a lot clearer whether we're dealing with a "normal", "italic" or explicit
390 // "oblique" font style. See webkit.org/b/187774.
391 std::optional<Value> slope;
392
393 std::tuple<Value, Value, std::optional<Value>> tied() const
394 {
395 return WTF::tie(weight, width, slope);
396 }
397};

426struct FontSelectionCapabilities {
427 using Range = FontSelectionRange;
428
429 FontSelectionCapabilities& operator=(const FontSelectionCapabilities&) = default;

431 constexpr std::tuple<Range, Range, Range> tied() const
432 {
433 return WTF::tie(weight, width, slope);
434 }
435
436 void expand(const FontSelectionCapabilities& capabilities)
437 {
438 weight.expand(capabilities.weight);
439 width.expand(capabilities.width);

444 Range width { normalStretchValue() };

458struct FontSelectionSpecifiedCapabilities {
459 using Capabilities = FontSelectionCapabilities;
460 using Range = FontSelectionRange;
461 using OptionalRange = std::optional<Range>;
462
463 constexpr Capabilities computeFontSelectionCapabilities() const
464 {
465 return { computeWeight(), computeWidth(), computeSlope() };
466 }
467
468 constexpr std::tuple<OptionalRange&, OptionalRange&, OptionalRange&> tied()
469 {
470 return WTF::tie(weight, width, slope);
471 }
472
473 constexpr std::tuple<const OptionalRange&, const OptionalRange&, const OptionalRange&> tied() const
474 {
475 return WTF::tie(weight, width, slope);
476 }
477
478 FontSelectionSpecifiedCapabilities& operator=(const Capabilities& other)
479 {
480 tied() = other.tied();
481 return *this;
482 }
483
484 constexpr Range computeWeight() const
485 {
486 return weight.value_or(Range { normalWeightValue() });
487 }
488
489 constexpr Range computeWidth() const
490 {
491 return width.value_or(Range { normalStretchValue() });
492 }
493
494 constexpr Range computeSlope() const
495 {
496 return slope.value_or(Range { normalItalicValue() });
497 }
498
499 template<class Encoder>
500 void encode(Encoder&) const;
501
502 template<class Decoder>
503 static std::optional<FontSelectionSpecifiedCapabilities> decode(Decoder&);
504
505 OptionalRange weight;
506 OptionalRange width;
507 OptionalRange slope;
508};
509
510template<class Encoder>
511void FontSelectionSpecifiedCapabilities::encode(Encoder& encoder) const
512{
513 encoder << weight;
514 encoder << width;
515 encoder << slope;
516}
517
518template<class Decoder>
519std::optional<FontSelectionSpecifiedCapabilities> FontSelectionSpecifiedCapabilities::decode(Decoder& decoder)
520{
521 std::optional<OptionalRange> weight;
522 decoder >> weight;
523 if (!weight)
524 return std::nullopt;
525
526 std::optional<OptionalRange> width;
527 decoder >> width;
528 if (!width)
529 return std::nullopt;
530
531 std::optional<OptionalRange> slope;
532 decoder >> slope;
533 if (!slope)
534 return std::nullopt;
535
536 return {{ *weight, *width, *slope }};
537}

542}
