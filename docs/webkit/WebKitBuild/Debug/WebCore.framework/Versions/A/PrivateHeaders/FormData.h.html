38struct FormDataElement {
39 struct EncodedFileData;
40 struct EncodedBlobData;
41 using Data = Variant<Vector<uint8_t>, EncodedFileData, EncodedBlobData>;
42
43 FormDataElement() = default;
44 explicit FormDataElement(Data&& data)
45 : data(WTFMove(data)) { }
46 explicit FormDataElement(Vector<uint8_t>&& array)
47 : data(WTFMove(array)) { }
48 FormDataElement(const String& filename, int64_t fileStart, int64_t fileLength, std::optional<WallTime> expectedFileModificationTime)
49 : data(EncodedFileData { filename, fileStart, fileLength, expectedFileModificationTime }) { }
50 explicit FormDataElement(const URL& blobURL)
51 : data(EncodedBlobData { blobURL }) { }
52
53 uint64_t lengthInBytes(const Function<uint64_t(const URL&)>&) const;
54 uint64_t lengthInBytes() const;
55
56 FormDataElement isolatedCopy() const;
57
58 template<typename Encoder> void encode(Encoder& encoder) const
59 {
60 encoder << data;
61 }
62 template<typename Decoder> static std::optional<FormDataElement> decode(Decoder& decoder)
63 {
64 std::optional<Data> data;
65 decoder >> data;
66 if (!data)
67 return std::nullopt;
68 return FormDataElement(WTFMove(*data));
69 }

71 struct EncodedFileData {
72 String filename;
73 int64_t fileStart { 0 };
74 int64_t fileLength { 0 };
75 std::optional<WallTime> expectedFileModificationTime;
76
77 bool fileModificationTimeMatchesExpectation() const;
78
79 EncodedFileData isolatedCopy() const
80 {
81 return { filename.isolatedCopy(), fileStart, fileLength, expectedFileModificationTime };
82 }
83
84 bool operator==(const EncodedFileData& other) const
85 {
86 return filename == other.filename
87 && fileStart == other.fileStart
88 && fileLength == other.fileLength
89 && expectedFileModificationTime == other.expectedFileModificationTime;
90 }
91 template<typename Encoder> void encode(Encoder& encoder) const
92 {
93 encoder << filename << fileStart << fileLength << expectedFileModificationTime;
94 }
95 template<typename Decoder> static std::optional<EncodedFileData> decode(Decoder& decoder)
96 {
97 std::optional<String> filename;
98 decoder >> filename;
99 if (!filename)
100 return std::nullopt;
101
102 std::optional<int64_t> fileStart;
103 decoder >> fileStart;
104 if (!fileStart)
105 return std::nullopt;
106
107 std::optional<int64_t> fileLength;
108 decoder >> fileLength;
109 if (!fileLength)
110 return std::nullopt;
111
112 std::optional<std::optional<WallTime>> expectedFileModificationTime;
113 decoder >> expectedFileModificationTime;
114 if (!expectedFileModificationTime)
115 return std::nullopt;
116
117 return {{
118 WTFMove(*filename),
119 WTFMove(*fileStart),
120 WTFMove(*fileLength),
121 WTFMove(*expectedFileModificationTime)
122 }};
123 }
124
125 };
126
127 struct EncodedBlobData {
128 URL url;
129
130 bool operator==(const EncodedBlobData& other) const
131 {
132 return url == other.url;
133 }
134 template<typename Encoder> void encode(Encoder& encoder) const
135 {
136 encoder << url;
137 }
138 template<typename Decoder> static std::optional<EncodedBlobData> decode(Decoder& decoder)
139 {
140 std::optional<URL> url;
141 decoder >> url;
142 if (!url)
143 return std::nullopt;
144
145 return {{ WTFMove(*url) }};
146 }
147 };
148
149 bool operator==(const FormDataElement& other) const
150 {
151 if (&other == this)
152 return true;
153 if (data.index() != other.data.index())
154 return false;
155 if (!data.index())
156 return WTF::get<0>(data) == WTF::get<0>(other.data);
157 if (data.index() == 1)
158 return WTF::get<1>(data) == WTF::get<1>(other.data);
159 return WTF::get<2>(data) == WTF::get<2>(other.data);
160 }
161 bool operator!=(const FormDataElement& other) const
162 {
163 return !(*this == other);
164 }
165
166 Data data;
167};

248 if (equalLettersIgnoringASCIICase(type, "text/plain"))

287template<typename Encoder>
288void FormData::encode(Encoder& encoder) const
289{
290 encoder << m_alwaysStream;
291 encoder << m_boundary;
292 encoder << m_elements;
293 encoder << m_identifier;
294 // FIXME: Does not encode m_containsPasswordData. Why is that OK?
295}
296
297template<typename Decoder>
298RefPtr<FormData> FormData::decode(Decoder& decoder)
299{
300 auto data = FormData::create();
301
302 if (!decoder.decode(data->m_alwaysStream))
303 return nullptr;
304
305 if (!decoder.decode(data->m_boundary))
306 return nullptr;
307
308 if (!decoder.decode(data->m_elements))
309 return nullptr;
310
311 if (!decoder.decode(data->m_identifier))
312 return nullptr;
313
314 return data;
315}
