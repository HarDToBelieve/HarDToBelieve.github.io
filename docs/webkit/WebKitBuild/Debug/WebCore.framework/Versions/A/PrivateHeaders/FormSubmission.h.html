31#pragma once
32
33#include "FormState.h"
34#include "FrameLoaderTypes.h"
35#include "RuntimeEnabledFeatures.h"
36#include <wtf/URL.h>
37#include <wtf/WeakPtr.h>
38
39namespace WebCore {
40
41class Event;
42class FormData;
43class FrameLoadRequest;
44class HTMLFormControlElement;
45
46class FormSubmission : public RefCounted<FormSubmission>, public CanMakeWeakPtr<FormSubmission> {
47public:
48 enum class Method : uint8_t { Get, Post, Dialog };
49
50 class Attributes {
51 public:
52 Method method() const { return m_method; }
53 static Method parseMethodType(const String&);
54 void updateMethodType(const String&);
55 static ASCIILiteral methodString(Method);
56
57 const String& action() const { return m_action; }
58 void parseAction(const String&);
59
60 const String& target() const { return m_target; }
61 void setTarget(const String& target) { m_target = target; }
62
63 const String& encodingType() const { return m_encodingType; }
64 static String parseEncodingType(const String&);
65 void updateEncodingType(const String&);
66 bool isMultiPartForm() const { return m_isMultiPartForm; }
67
68 const String& acceptCharset() const { return m_acceptCharset; }
69 void setAcceptCharset(const String& value) { m_acceptCharset = value; }
70
71 private:
72 Method m_method { Method::Get };
73 bool m_isMultiPartForm { false };
74 String m_action;
75 String m_target;
76 String m_encodingType { "application/x-www-form-urlencoded"_s };
77 String m_acceptCharset;
78 };
79
80 static Ref<FormSubmission> create(HTMLFormElement&, HTMLFormControlElement* overrideSubmitter, const Attributes&, Event*, LockHistory, FormSubmissionTrigger);
81
82 void populateFrameLoadRequest(FrameLoadRequest&);
83 URL requestURL() const;
84
85 Method method() const { return m_method; }
86 const URL& action() const { return m_action; }
87 const String& target() const { return m_target; }
88 const String& contentType() const { return m_contentType; }
89 FormState& state() const { return *m_formState; }
90 Ref<FormState> takeState() { return m_formState.releaseNonNull(); }
91 FormData& data() const { return *m_formData; }
92 const String boundary() const { return m_boundary; }
93 LockHistory lockHistory() const { return m_lockHistory; }
94 Event* event() const { return m_event.get(); }
95 const String& referrer() const { return m_referrer; }
96 const String& origin() const { return m_origin; }
97
98 const String& returnValue() const { return m_returnValue; }
99
100 void clearTarget() { m_target = { }; }
101 void setReferrer(const String& referrer) { m_referrer = referrer; }
102 void setOrigin(const String& origin) { m_origin = origin; }
103
104 void cancel() { m_wasCancelled = true; }
105 bool wasCancelled() const { return m_wasCancelled; }
106
107private:
108 // dialog form submissions
109 FormSubmission(Method, const String& returnValue, const URL& action, const String& target, const String& contentType, LockHistory, Event*);
110
111 // get/post form submissions
112 FormSubmission(Method, const URL& action, const String& target, const String& contentType, Ref<FormState>&&, Ref<FormData>&&, const String& boundary, LockHistory, Event*);
113
114 // FIXME: Hold an instance of Attributes instead of individual members.
115 Method m_method;
116 bool m_wasCancelled { false };
117 URL m_action;
118 String m_target;
119 String m_contentType;
120 RefPtr<FormState> m_formState;
121 RefPtr<FormData> m_formData;
122 String m_boundary;
123 LockHistory m_lockHistory;
124 RefPtr<Event> m_event;
125 String m_referrer;
126 String m_origin;
127
128 String m_returnValue; // for form[method=dialog]
129};
130
131} // namespace WebCore
