26#pragma once
27
28#include <wtf/text/WTFString.h>
29
30namespace WebCore {
31
32struct FourCC {
33 FourCC() = default;
34 FourCC(uint32_t value) : value(value) { }
35
36 template<std::size_t N>
37 constexpr FourCC(const char (&data)[N])
38 {
39 static_assert((N - 1) == 4, "FourCC literals must be exactly 4 characters long");
40 value = data[0] << 24 | data[1] << 16 | data[2] << 8 | data[3];
41 }
42
43 String toString() const;
44 WEBCORE_EXPORT static std::optional<FourCC> fromString(const String&);
45
46 bool operator==(const FourCC& other) const { return value == other.value; }
47 bool operator!=(const FourCC& other) const { return value != other.value; }
48
49 uint32_t value { 0 };
50};
51
52} // namespace WebCore
53
54namespace WTF {
55
56template<typename> struct LogArgument;
57
58template<> struct LogArgument<WebCore::FourCC> {
59 static String toString(const WebCore::FourCC& code) { return code.toString(); }
60};
