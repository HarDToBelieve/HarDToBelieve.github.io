74class PolicyCheckIdentifier {
75public:
76 PolicyCheckIdentifier() = default;
77
78 static PolicyCheckIdentifier create();
79
80 bool isValidFor(PolicyCheckIdentifier);
81 bool operator==(const PolicyCheckIdentifier& other) const { return m_process == other.m_process && m_policyCheck == other.m_policyCheck; }
82
83 template<class Encoder> void encode(Encoder&) const;
84 template<class Decoder> static std::optional<PolicyCheckIdentifier> decode(Decoder&);
85
86private:
87 PolicyCheckIdentifier(ProcessIdentifier process, uint64_t policyCheck)
88 : m_process(process)
89 , m_policyCheck(policyCheck)
90 { }
91
92 ProcessIdentifier m_process;
93 uint64_t m_policyCheck { 0 };
94};
95
96template<class Encoder>
97void PolicyCheckIdentifier::encode(Encoder& encoder) const
98{
99 encoder << m_process << m_policyCheck;
100}
101
102template<class Decoder>
103std::optional<PolicyCheckIdentifier> PolicyCheckIdentifier::decode(Decoder& decoder)
104{
105 auto process = ProcessIdentifier::decode(decoder);
106 if (!process)
107 return std::nullopt;
108
109 uint64_t policyCheck;
110 if (!decoder.decode(policyCheck))
111 return std::nullopt;
112
113 return PolicyCheckIdentifier { *process, policyCheck };
114}

189struct SystemPreviewInfo {
190 ElementContext element;
191
192 IntRect previewRect;
193 bool isPreview { false };
194
195 template<class Encoder> void encode(Encoder&) const;
196 template<class Decoder> static std::optional<SystemPreviewInfo> decode(Decoder&);
197};
198
199template<class Encoder>
200void SystemPreviewInfo::encode(Encoder& encoder) const
201{
202 encoder << element << previewRect << isPreview;
203}
204
205template<class Decoder>
206std::optional<SystemPreviewInfo> SystemPreviewInfo::decode(Decoder& decoder)
207{
208 std::optional<ElementContext> element;
209 decoder >> element;
210 if (!element)
211 return std::nullopt;
212
213 std::optional<IntRect> previewRect;
214 decoder >> previewRect;
215 if (!previewRect)
216 return std::nullopt;
217
218 std::optional<bool> isPreview;
219 decoder >> isPreview;
220 if (!isPreview)
221 return std::nullopt;
222
223 return { { WTFMove(*element), WTFMove(*previewRect), WTFMove(*isPreview) } };
224}
