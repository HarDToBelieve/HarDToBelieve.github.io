20#pragma once
21
22#include <wtf/Forward.h>
23#include <wtf/WeakPtr.h>
24#include <wtf/text/AtomString.h>
25
26namespace WebCore {
27
28enum class CanWrap : bool { No, Yes };
29enum class DidWrap : bool { No, Yes };
30
31class Frame;
32class TreeScope;
33
34class FrameTree {
35 WTF_MAKE_NONCOPYABLE(FrameTree);
36public:
37 static constexpr unsigned invalidCount = static_cast<unsigned>(-1);
38
39 FrameTree(Frame& thisFrame, Frame* parentFrame);
40
41 ~FrameTree();
42
43 const AtomString& name() const { return m_name; }
44 const AtomString& uniqueName() const { return m_uniqueName; }
45 WEBCORE_EXPORT void setName(const AtomString&);
46 WEBCORE_EXPORT void clearName();
47 WEBCORE_EXPORT Frame* parent() const;
48
49 Frame* nextSibling() const { return m_nextSibling.get(); }
50 Frame* previousSibling() const { return m_previousSibling.get(); }
51 Frame* firstChild() const { return m_firstChild.get(); }
52 Frame* lastChild() const { return m_lastChild.get(); }
53
54 Frame* firstRenderedChild() const;
55 Frame* nextRenderedSibling() const;
56
57 WEBCORE_EXPORT bool isDescendantOf(const Frame* ancestor) const;
58
59 WEBCORE_EXPORT Frame* traverseNext(const Frame* stayWithin = nullptr) const;
60 // Rendered means being the main frame or having an ownerRenderer. It may not have been parented in the Widget tree yet (see WidgetHierarchyUpdatesSuspensionScope).
61 WEBCORE_EXPORT Frame* traverseNextRendered(const Frame* stayWithin = nullptr) const;
62 WEBCORE_EXPORT Frame* traverseNext(CanWrap, DidWrap* = nullptr) const;
63 WEBCORE_EXPORT Frame* traversePrevious(CanWrap, DidWrap* = nullptr) const;
64
65 Frame* traverseNextInPostOrder(CanWrap) const;
66
67 WEBCORE_EXPORT void appendChild(Frame&);
68 void detachFromParent() { m_parent = nullptr; }
69 void removeChild(Frame&);
70
71 Frame* child(unsigned index) const;
72 Frame* child(const AtomString& name) const;
73 WEBCORE_EXPORT Frame* find(const AtomString& name, Frame& activeFrame) const;
74 WEBCORE_EXPORT unsigned childCount() const;
75 unsigned descendantCount() const;
76 WEBCORE_EXPORT Frame& top() const;
77
78 WEBCORE_EXPORT Frame* scopedChild(unsigned index) const;
79 WEBCORE_EXPORT Frame* scopedChild(const AtomString& name) const;
80 unsigned scopedChildCount() const;
81
82 void resetFrameIdentifiers() { m_frameIDGenerator = 0; }
83
84private:
85 Frame* deepFirstChild() const;
86 Frame* deepLastChild() const;
87
88 bool scopedBy(TreeScope*) const;
89 Frame* scopedChild(unsigned index, TreeScope*) const;
90 Frame* scopedChild(const AtomString& name, TreeScope*) const;
91 unsigned scopedChildCount(TreeScope*) const;
92
93 AtomString uniqueChildName(const AtomString& requestedName) const;
94 AtomString generateUniqueName() const;
95
96 Frame& m_thisFrame;
97
98 WeakPtr<Frame> m_parent;
99 AtomString m_name; // The actual frame name (may be empty).
100 AtomString m_uniqueName;
101
102 RefPtr<Frame> m_nextSibling;
103 WeakPtr<Frame> m_previousSibling;
104 RefPtr<Frame> m_firstChild;
105 WeakPtr<Frame> m_lastChild;
106 mutable unsigned m_scopedChildCount { invalidCount };
107 mutable uint64_t m_frameIDGenerator { 0 };
108};
109
110} // namespace WebCore
111

115#endif
