39namespace WebCore {
40
41// This class wraps the lenght of a gap used by properties column-gap and row-gap.
42// The valid values are "normal" or a non-negative <length-percentage>.
43class GapLength {
44public:
45 GapLength()
46 : m_isNormal(true)
47 {
48 }
49
50 GapLength(const Length& length)
51 : m_isNormal(false)
52 , m_length(length)
53 {
54 }
55
56 bool isNormal() const { return m_isNormal; }
57 const Length& length() const { ASSERT(!m_isNormal); return m_length; }
58
59 bool operator==(const GapLength& o) const
60 {
61 return m_isNormal == o.m_isNormal && m_length == o.m_length;
62 }
63
64private:
65 bool m_isNormal;
66 Length m_length;
67};
68
69WTF::TextStream& operator<<(WTF::TextStream&, const GapLength&);
