26#pragma once
27
28#include <wtf/RefCounted.h>
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class GeolocationPositionError : public RefCounted<GeolocationPositionError> {
34public:
35 enum ErrorCode {
36 PERMISSION_DENIED = 1,
37 POSITION_UNAVAILABLE = 2,
38 TIMEOUT = 3
39 };
40
41 static Ref<GeolocationPositionError> create(ErrorCode code, const String& message) { return adoptRef(*new GeolocationPositionError(code, message)); }
42
43 ErrorCode code() const { return m_code; }
44 const String& message() const { return m_message; }
45 void setIsFatal(bool isFatal) { m_isFatal = isFatal; }
46 bool isFatal() const { return m_isFatal; }
47
48private:
49 GeolocationPositionError(ErrorCode code, const String& message)
50 : m_code(code)
51 , m_message(message)
52 , m_isFatal(false)
53 {
54 }
55
56 ErrorCode m_code;
57 String m_message;
58 // Whether the error is fatal, such that no request can ever obtain a good
59 // position fix in the future.
60 bool m_isFatal;
61};
62
63} // namespace WebCore
