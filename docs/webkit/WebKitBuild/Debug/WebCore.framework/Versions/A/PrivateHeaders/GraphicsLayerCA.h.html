344 // Used to track the path down the tree for replica layers.
345 struct ReplicaState {
346 WTF_MAKE_STRUCT_FAST_ALLOCATED;
347 static const size_t maxReplicaDepth = 16;
348 enum ReplicaBranchType { ChildBranch = 0, ReplicaBranch = 1 };
349 ReplicaState(ReplicaBranchType firstBranch)
350 : m_replicaDepth(0)
351 {
352 push(firstBranch);
353 }
354
355 // Called as we walk down the tree to build replicas.
356 void push(ReplicaBranchType branchType)
357 {
358 m_replicaBranches.append(branchType);
359 if (branchType == ReplicaBranch)
360 ++m_replicaDepth;
361 }
362
363 void setBranchType(ReplicaBranchType branchType)
364 {
365 ASSERT(!m_replicaBranches.isEmpty());
366
367 if (m_replicaBranches.last() != branchType) {
368 if (branchType == ReplicaBranch)
369 ++m_replicaDepth;
370 else
371 --m_replicaDepth;
372 }
373
374 m_replicaBranches.last() = branchType;
375 }
376
377 void pop()
378 {
379 if (m_replicaBranches.last() == ReplicaBranch)
380 --m_replicaDepth;
381 m_replicaBranches.removeLast();
382 }
383
384 size_t depth() const { return m_replicaBranches.size(); }
385 size_t replicaDepth() const { return m_replicaDepth; }
386
387 CloneID cloneID() const;
388
389 private:
390 Vector<ReplicaBranchType> m_replicaBranches;
391 size_t m_replicaDepth;
392 };
