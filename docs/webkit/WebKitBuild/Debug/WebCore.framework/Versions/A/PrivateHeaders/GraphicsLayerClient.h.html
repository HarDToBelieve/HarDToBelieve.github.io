42enum class GraphicsLayerPaintingPhase {
43 Background = 1 << 0,

52enum AnimatedPropertyID {
53 AnimatedPropertyInvalid,
54 AnimatedPropertyTranslate,
55 AnimatedPropertyScale,
56 AnimatedPropertyRotate,
57 AnimatedPropertyTransform,
58 AnimatedPropertyOpacity,
59 AnimatedPropertyBackgroundColor,
60 AnimatedPropertyFilter,
61#if ENABLE(FILTERS_LEVEL_2)
62 AnimatedPropertyWebkitBackdropFilter,
63#endif
64};
65
66inline bool animatedPropertyIsTransformOrRelated(AnimatedPropertyID property)
67{
68 return property == AnimatedPropertyTransform || property == AnimatedPropertyTranslate || property == AnimatedPropertyScale || property == AnimatedPropertyRotate;
69}

103class GraphicsLayerClient {
104public:
105 virtual ~GraphicsLayerClient() = default;
106
107 virtual void tiledBackingUsageChanged(const GraphicsLayer*, bool /*usingTiledBacking*/) { }
108
109 // Callback for when hardware-accelerated animation started.
110 virtual void notifyAnimationStarted(const GraphicsLayer*, const String& /*animationKey*/, MonotonicTime /*time*/) { }
111 virtual void notifyAnimationEnded(const GraphicsLayer*, const String& /*animationKey*/) { }
112
113 // Notification that a layer property changed that requires a subsequent call to flushCompositingState()
114 // to appear on the screen.
115 virtual void notifyFlushRequired(const GraphicsLayer*) { }
116
117 // Notification that this layer requires a flush before the next display refresh.
118 virtual void notifyFlushBeforeDisplayRefresh(const GraphicsLayer*) { }
119
120 virtual void paintContents(const GraphicsLayer*, GraphicsContext&, const FloatRect& /* inClip */, GraphicsLayerPaintBehavior) { }
121 virtual void didChangePlatformLayerForLayer(const GraphicsLayer*) { }
122
123 // Provides current transform (taking transform-origin and animations into account). Input matrix has been
124 // initialized to identity already. Returns false if the layer has no transform.
125 virtual bool getCurrentTransform(const GraphicsLayer*, TransformationMatrix&) const { return false; }
126
127 // Allows the client to modify a layer position used during the visibleRect calculation, for example to ignore
128 // scroll overhang.
129 virtual void customPositionForVisibleRectComputation(const GraphicsLayer*, FloatPoint&) const { }
130
131 // Multiplier for backing store size, related to high DPI.
132 virtual float deviceScaleFactor() const { return 1; }
133 // Page scale factor.
134 virtual float pageScaleFactor() const { return 1; }
135 virtual float zoomedOutPageScaleFactor() const { return 0; }
136
137 virtual float contentsScaleMultiplierForNewTiles(const GraphicsLayer*) const { return 1; }
138 virtual bool paintsOpaquelyAtNonIntegralScales(const GraphicsLayer*) const { return false; }
139
140 virtual bool isTrackingRepaints() const { return false; }
141
142 virtual bool shouldSkipLayerInDump(const GraphicsLayer*, LayerTreeAsTextBehavior) const { return false; }
143 virtual bool shouldDumpPropertyForLayer(const GraphicsLayer*, const char*, LayerTreeAsTextBehavior) const { return true; }
144
145 virtual bool shouldAggressivelyRetainTiles(const GraphicsLayer*) const { return false; }
146 virtual bool shouldTemporarilyRetainTileCohorts(const GraphicsLayer*) const { return true; }
147
148 virtual bool useGiantTiles() const { return false; }
149
150 virtual bool needsPixelAligment() const { return false; }
151
152 virtual bool needsIOSDumpRenderTreeMainFrameRenderViewLayerIsAlwaysOpaqueHack(const GraphicsLayer&) const { return false; }
153
154 virtual void logFilledVisibleFreshTile(unsigned) { };
155
156 virtual TransformationMatrix transformMatrixForProperty(AnimatedPropertyID) const { return { }; }
157
158#ifndef NDEBUG
159 // RenderLayerBacking overrides this to verify that it is not
160 // currently painting contents. An ASSERT fails, if it is.
161 // This is executed in GraphicsLayer construction and destruction
162 // to verify that we don't create or destroy GraphicsLayers
163 // while painting.
164 virtual void verifyNotPainting() { }
165#endif
166};
