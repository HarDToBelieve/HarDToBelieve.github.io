34// GCGL types match the corresponding GL types as defined in OpenGL ES 2.0
35// header file gl2.h from khronos.org.
36typedef unsigned GCGLenum;
37typedef unsigned char GCGLboolean;
38typedef unsigned GCGLbitfield;
39typedef signed char GCGLbyte;
40typedef unsigned char GCGLubyte;
41typedef short GCGLshort;
42typedef unsigned short GCGLushort;
43typedef int GCGLint;
44typedef int GCGLsizei;
45typedef unsigned GCGLuint;
46typedef float GCGLfloat;
47typedef unsigned short GCGLhalffloat;
48typedef float GCGLclampf;
49typedef char GCGLchar;
50typedef struct __GLsync* GCGLsync;
51typedef void GCGLvoid;
52
53// These GCGL types do not strictly match the GL types as defined in OpenGL ES 2.0
54// header file for all platforms.
55typedef intptr_t GCGLintptr;
56typedef intptr_t GCGLsizeiptr;
57typedef intptr_t GCGLvoidptr;
58typedef int64_t GCGLint64;
59typedef uint64_t GCGLuint64;
60
61typedef GCGLuint PlatformGLObject;
62
63#if !PLATFORM(COCOA)
64typedef unsigned GLuint;
65#endif

67inline constexpr size_t gcGLSpanDynamicExtent = std::numeric_limits<size_t>::max();
68
69template<typename T, size_t Extent = gcGLSpanDynamicExtent>
70struct GCGLSpan;
71template<typename T, size_t Extent>
72struct GCGLSpan {
73 explicit GCGLSpan(T* array, size_t size = Extent)
74 : data(array)
75 {
76 ASSERT_UNUSED(size, size == Extent);
77 }
78 GCGLSpan(T (&array)[Extent])
79 : data(array)
80 { }
81 template<typename U>
82 GCGLSpan(const GCGLSpan<U, Extent>& other, std::enable_if_t<std::is_convertible_v<U(*)[], T(*)[]>, std::nullptr_t> = nullptr)
83 : data(other.data)
84 { }
85 GCGLSpan(std::array<T, Extent>& array)
86 : data(array.data())
87 { }
88 GCGLSpan(const std::array<T, Extent>& array)
89 : data(array.data())
90 { }
91 T& operator[](size_t i) { RELEASE_ASSERT(data && i < bufSize); return data[i]; }
92 T& operator*() { RELEASE_ASSERT(data && bufSize); return *data; }
93 T* data;
94 constexpr static size_t bufSize = Extent;
95};
96
97template<typename T>
98struct GCGLSpan<T, gcGLSpanDynamicExtent> {
99 GCGLSpan(T* array, size_t bufSize_)
100 : data(array)
101 , bufSize(bufSize_)
102 { }
103 template<size_t Sz>
104 GCGLSpan(T (&array)[Sz])
105 : data(array)
106 , bufSize(Sz)
107 { }
108 template<typename U, size_t UExtent>
109 GCGLSpan(const GCGLSpan<U, UExtent>& other, std::enable_if_t<std::is_convertible_v<U(*)[], T(*)[]>, std::nullptr_t> = nullptr)
110 : data(other.data)
111 , bufSize(other.bufSize)
112 { }
113 template<typename U, size_t inlineCapacity>
114 GCGLSpan(WTF::Vector<U, inlineCapacity>& array, std::enable_if_t<std::is_convertible_v<U(*)[], T(*)[]>, std::nullptr_t> = nullptr)
115 : data(array.data())
116 , bufSize(array.size())
117 { }
118 T& operator[](size_t i) { RELEASE_ASSERT(data && i < bufSize); return data[i]; }
119 T& operator*() { RELEASE_ASSERT(data && bufSize); return *data; }
120 T* data;
121 const size_t bufSize;
122};

124template<>
125struct GCGLSpan<GCGLvoid> {
126 GCGLSpan(GCGLvoid* array, size_t bufSize_)
127 : data(array)
128 , bufSize(bufSize_)
129 { }
130 template<typename U, size_t Sz>
131 GCGLSpan(U (&array)[Sz])
132 : data(array)
133 , bufSize(Sz * sizeof(U))
134 { }
135 template<typename U, size_t UExtent>
136 GCGLSpan(const GCGLSpan<U, UExtent>& other)
137 : data(other.data)
138 , bufSize(other.bufSize * sizeof(U))
139 { }
140 template<typename U, size_t inlineCapacity>
141 GCGLSpan(WTF::Vector<U, inlineCapacity>& array)
142 : data(array.data())
143 , bufSize(array.size() * sizeof(U))
144 { }
145 GCGLvoid* data;
146 const size_t bufSize;
147};
148
149template<>
150struct GCGLSpan<const GCGLvoid> {
151 GCGLSpan(const GCGLvoid* array, size_t bufSize_)
152 : data(array)
153 , bufSize(bufSize_)
154 { }
155 template<typename U, size_t Sz>
156 GCGLSpan(U (&array)[Sz])
157 : data(array)
158 , bufSize(Sz * sizeof(U))
159 { }
160 GCGLSpan(const GCGLSpan<GCGLvoid>& other)
161 : data(other.data)
162 , bufSize(other.bufSize)
163 { }
164 template<typename U, size_t UExtent>
165 GCGLSpan(const GCGLSpan<U, UExtent>& other)
166 : data(other.data)
167 , bufSize(other.bufSize * sizeof(U))
168 { }
169 const GCGLvoid* data;
170 const size_t bufSize;
171};

173template<typename T, size_t N>
174GCGLSpan(T (&)[N]) -> GCGLSpan<T, N>;
175
176template<typename T, size_t N>
177GCGLSpan(std::array<T, N>&) -> GCGLSpan<T, N>;
178
179template<typename T, size_t inlineCapacity>
180GCGLSpan(WTF::Vector<T, inlineCapacity>&) -> GCGLSpan<T>;
181
182template<typename T>
183GCGLSpan<T> makeGCGLSpan(T* data, size_t count)
184{
185 return GCGLSpan<T> { data, count };
186}
187
188template<size_t Extent, typename T>
189GCGLSpan<T, Extent> makeGCGLSpan(T* data)
190{
191 return GCGLSpan<T, Extent> { data };
192}
