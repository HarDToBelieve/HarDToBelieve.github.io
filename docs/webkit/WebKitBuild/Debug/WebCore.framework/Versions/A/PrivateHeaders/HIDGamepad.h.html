27
28#if ENABLE(GAMEPAD) && PLATFORM(MAC)
29
30#include "HIDDevice.h"
31#include "HIDGamepadElement.h"
32#include "PlatformGamepad.h"
33#include <IOKit/hid/IOHIDDevice.h>
34#include <wtf/HashMap.h>
35
36namespace WebCore {
37
38class HIDGamepad : public PlatformGamepad {
39public:
40 static std::unique_ptr<HIDGamepad> create(IOHIDDeviceRef, unsigned index);
41
42 const HIDDevice& hidDevice() const { return m_device; }
43
44 void initialize();
45 HIDInputType valueChanged(IOHIDValueRef);
46
47 const Vector<SharedGamepadValue>& axisValues() const final { return m_axisValues; }
48 const Vector<SharedGamepadValue>& buttonValues() const final { return m_buttonValues; }
49
50 const char* source() const final { return "HID"_s; }
51
52protected:
53 HIDGamepad(HIDDevice&&, unsigned index);
54
55 HashMap<IOHIDElementCookie, std::unique_ptr<HIDGamepadElement>> m_elementMap;
56 Vector<SharedGamepadValue> m_buttonValues;
57 Vector<SharedGamepadValue> m_axisValues;
58
59private:
60 HIDDevice m_device;
61};
62
63} // namespace WebCore
64
65#endif // ENABLE(GAMEPAD) && PLATFORM(MAC)
