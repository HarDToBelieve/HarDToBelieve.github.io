26#pragma once
27
28#if ENABLE(GAMEPAD) && PLATFORM(MAC)
29
30#include "HIDElement.h"
31#include "SharedGamepadValue.h"
32
33namespace WebCore {
34
35enum class HIDInputType {
36 ButtonPress,
37 NotAButtonPress,
38};
39
40class HIDGamepadElement : public HIDElement {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 virtual ~HIDGamepadElement() { }
44
45 virtual HIDInputType gamepadValueChanged(IOHIDValueRef) = 0;
46
47 void refreshCurrentValue();
48
49protected:
50 HIDGamepadElement(const HIDElement&);
51
52 virtual double normalizedValue();
53 virtual bool isButton() const { return false; }
54 virtual bool isAxis() const { return false; }
55};
56
57class HIDGamepadButton final : public HIDGamepadElement {
58public:
59 HIDGamepadButton(const HIDElement& element, SharedGamepadValue& value)
60 : HIDGamepadElement(element)
61 , m_value(value)
62 {
63 }
64
65 bool isButton() const final { return true; }
66
67private:
68 HIDInputType gamepadValueChanged(IOHIDValueRef) override;
69
70 SharedGamepadValue m_value;
71};
72
73class HIDGamepadAxis final : public HIDGamepadElement {
74public:
75 HIDGamepadAxis(const HIDElement& element, SharedGamepadValue& value)
76 : HIDGamepadElement(element)
77 , m_value(value)
78 {
79 }
80
81 bool isAxis() const final { return true; }
82
83private:
84 HIDInputType gamepadValueChanged(IOHIDValueRef) override;
85 double normalizedValue() final;
86
87 SharedGamepadValue m_value;
88};
89
90class HIDGamepadHatswitch : public HIDGamepadElement {
91public:
92 HIDGamepadHatswitch(const HIDElement& element, Vector<SharedGamepadValue>&& buttonValues)
93 : HIDGamepadElement(element)
94 , m_buttonValues(WTFMove(buttonValues))
95 {
96 }
97
98 // Treat hatswitch value changes as a button press
99 bool isButton() const final { return true; }
100
101private:
102 HIDInputType gamepadValueChanged(IOHIDValueRef) final;
103 double normalizedValue() final;
104
105 Vector<SharedGamepadValue> m_buttonValues;
106};
107
108} // namespace WebCore
