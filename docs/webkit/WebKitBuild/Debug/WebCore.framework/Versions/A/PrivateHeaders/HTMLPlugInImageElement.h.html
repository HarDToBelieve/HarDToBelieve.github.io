22
23#include "HTMLPlugInElement.h"
24
25namespace WebCore {
26
27class HTMLImageLoader;
28
29enum class CreatePlugins { No, Yes };
30
31// Base class for HTMLEmbedElement and HTMLObjectElement.
32// FIXME: This is the only class that derives from HTMLPlugInElement, so we could merge the two classes.
33class HTMLPlugInImageElement : public HTMLPlugInElement {
34 WTF_MAKE_ISO_ALLOCATED(HTMLPlugInImageElement);
35public:
36 virtual ~HTMLPlugInImageElement();
37
38 RenderEmbeddedObject* renderEmbeddedObject() const;
39
40 virtual void updateWidget(CreatePlugins) = 0;
41
42 const String& serviceType() const { return m_serviceType; }
43 const String& url() const { return m_url; }
44
45 bool needsWidgetUpdate() const { return m_needsWidgetUpdate; }
46 void setNeedsWidgetUpdate(bool needsWidgetUpdate) { m_needsWidgetUpdate = needsWidgetUpdate; }
47
48protected:
49 HTMLPlugInImageElement(const QualifiedName& tagName, Document&);
50
51 void didMoveToNewDocument(Document& oldDocument, Document& newDocument) override;
52
53 bool requestObject(const String& url, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues) final;
54
55 bool isImageType();
56 HTMLImageLoader* imageLoader() { return m_imageLoader.get(); }
57 void updateImageLoaderWithNewURLSoon();
58
59 bool canLoadURL(const String& relativeURL) const;
60 bool wouldLoadAsPlugIn(const String& relativeURL, const String& serviceType);
61
62 void scheduleUpdateForAfterStyleResolution();
63
64 String m_serviceType;
65 String m_url;
66
67private:
68 bool isPlugInImageElement() const final { return true; }
69
70 bool shouldBypassCSPForPDFPlugin(const String&) const;
71 bool canLoadPlugInContent(const String& relativeURL, const String& mimeType) const;
72 bool canLoadURL(const URL&) const;
73
74 RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) override;
75 bool childShouldCreateRenderer(const Node&) const override;
76 void willRecalcStyle(Style::Change) final;
77 void didRecalcStyle(Style::Change) final;
78 void didAttachRenderers() final;
79 void willDetachRenderers() final;
80
81 void prepareForDocumentSuspension() final;
82 void resumeFromDocumentSuspension() final;
83
84 void updateAfterStyleResolution();
85
86 bool m_needsWidgetUpdate { false };
87 bool m_needsDocumentActivationCallbacks { false };
88 std::unique_ptr<HTMLImageLoader> m_imageLoader;
89 bool m_needsImageReload { false };
90 bool m_hasUpdateScheduledForAfterStyleResolution { false };
91};
92
93} // namespace WebCore
94
95SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::HTMLPlugInImageElement)
96 static bool isType(const WebCore::HTMLPlugInElement& element) { return element.isPlugInImageElement(); }
97 static bool isType(const WebCore::Node& node) { return is<WebCore::HTMLPlugInElement>(node) && isType(downcast<WebCore::HTMLPlugInElement>(node)); }
98SPECIALIZE_TYPE_TRAITS_END()
