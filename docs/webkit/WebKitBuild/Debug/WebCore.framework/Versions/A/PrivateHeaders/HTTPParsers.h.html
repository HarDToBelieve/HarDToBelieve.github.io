113inline bool isHTTPSpace(UChar character)
114{
115 return character <= ' ' && (character == ' ' || character == '\n' || character == '\t' || character == '\r');
116}
117
118// Strip leading and trailing whitespace as defined in https://fetch.spec.whatwg.org/#concept-header-value-normalize.
119inline String stripLeadingAndTrailingHTTPSpaces(const String& string)
120{
121 return string.stripLeadingAndTrailingCharacters(isHTTPSpace);
122}
123
124inline StringView stripLeadingAndTrailingHTTPSpaces(StringView string)
125{
126 return string.stripLeadingAndTrailingMatchedCharacters(isHTTPSpace);
127}

129template<class HashType>
130bool addToAccessControlAllowList(const String& string, unsigned start, unsigned end, HashSet<String, HashType>& set)
131{
132 StringImpl* stringImpl = string.impl();
133 if (!stringImpl)
134 return true;
135
136 // Skip white space from start.
137 while (start <= end && isHTTPSpace((*stringImpl)[start]))
138 ++start;
139
140 // only white space
141 if (start > end)
142 return true;
143
144 // Skip white space from end.
145 while (end && isHTTPSpace((*stringImpl)[end]))
146 --end;
147
148 auto token = string.substring(start, end - start + 1);
149 if (!isValidHTTPToken(token))
150 return false;
151
152 set.add(WTFMove(token));
153 return true;
154}
155
156template<class HashType = DefaultHash<String>>
157std::optional<HashSet<String, HashType>> parseAccessControlAllowList(const String& string)
158{
159 HashSet<String, HashType> set;
160 unsigned start = 0;
161 size_t end;
162 while ((end = string.find(',', start)) != notFound) {
163 if (start != end) {
164 if (!addToAccessControlAllowList(string, start, end - 1, set))
165 return { };
166 }
167 start = end + 1;
168 }
169 if (start != string.length()) {
170 if (!addToAccessControlAllowList(string, start, string.length() - 1, set))
171 return { };
172 }
173 return set;
174}
