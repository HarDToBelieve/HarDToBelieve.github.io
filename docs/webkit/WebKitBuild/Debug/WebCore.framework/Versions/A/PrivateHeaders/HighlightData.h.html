42class RenderRange {
43public:
44 RenderRange() = default;
45 RenderRange(RenderObject* start, RenderObject* end, unsigned startOffset, unsigned endOffset)
46 : m_start(makeWeakPtr(start))
47 , m_end(makeWeakPtr(end))
48 , m_startOffset(startOffset)
49 , m_endOffset(endOffset)
50 {
51 }
52
53 RenderObject* start() const { return m_start.get(); }
54 RenderObject* end() const { return m_end.get(); }
55 unsigned startOffset() const { return m_startOffset; }
56 unsigned endOffset() const { return m_endOffset; }
57
58 bool operator==(const RenderRange& other) const
59 {
60 return m_start == other.m_start && m_end == other.m_end && m_startOffset == other.m_startOffset && m_endOffset == other.m_endOffset;
61 }
62
63private:
64 WeakPtr<RenderObject> m_start;
65 WeakPtr<RenderObject> m_end;
66 unsigned m_startOffset { 0 };
67 unsigned m_endOffset { 0 };
68};

83class HighlightData {
84public:
85 enum IsSelectionTag { IsSelection };
86 HighlightData() = default;
87 HighlightData(IsSelectionTag)
88 : m_isSelection(true)
89 { }
90
91 void setRenderRange(const RenderRange&);
92 bool setRenderRange(const HighlightRangeData&); // Returns true if successful.
93 const RenderRange& get() const { return m_renderRange; }
94
95 RenderObject* start() const { return m_renderRange.start(); }
96 RenderObject* end() const { return m_renderRange.end(); }
97
98 unsigned startOffset() const { return m_renderRange.startOffset(); }
99 unsigned endOffset() const { return m_renderRange.endOffset(); }
100
101 RenderObject::HighlightState highlightStateForRenderer(const RenderObject&);
102 RenderObject::HighlightState highlightStateForTextBox(const RenderText&, const TextBoxSelectableRange&);
103 std::pair<unsigned, unsigned> rangeForTextBox(const RenderText&, const TextBoxSelectableRange&);
104
105protected:
106 RenderRange m_renderRange;
107 const bool m_isSelection { false };
108};
