23#pragma once
24
25#include <wtf/Assertions.h>
26#include <wtf/OptionSet.h>
27
28namespace WebCore {
29
30class HitTestRequest {
31public:
32 enum class Type {
33 ReadOnly = 1 << 0,
34 Active = 1 << 1,
35 Move = 1 << 2,
36 Release = 1 << 3,
37 IgnoreCSSPointerEventsProperty = 1 << 4,
38 IgnoreClipping = 1 << 5,
39 SVGClipContent = 1 << 6,
40 TouchEvent = 1 << 7,
41 DisallowUserAgentShadowContent = 1 << 8,
42 DisallowUserAgentShadowContentExceptForImageOverlays = 1 << 9,
43 AllowFrameScrollbars = 1 << 10,
44 AllowChildFrameContent = 1 << 11,
45 AllowVisibleChildFrameContentOnly = 1 << 12,
46 ChildFrameHitTest = 1 << 13,
47 AccessibilityHitTest = 1 << 14,
48 // Collect a list of nodes instead of just one. Used for elementsFromPoint and rect-based tests.
49 CollectMultipleElements = 1 << 15,
50 // When using list-based testing, continue hit testing even after a hit has been found.
51 IncludeAllElementsUnderPoint = 1 << 16,
52 };
53
54 HitTestRequest(OptionSet<Type> type = { Type::ReadOnly, Type::Active, Type::DisallowUserAgentShadowContent })
55 : m_type { type }
56 {
57 ASSERT(!type.containsAll({ Type::DisallowUserAgentShadowContentExceptForImageOverlays, Type::DisallowUserAgentShadowContent }));
58 ASSERT_IMPLIES(type.contains(Type::IncludeAllElementsUnderPoint), type.contains(Type::CollectMultipleElements));
59 }
60
61 bool readOnly() const { return m_type.contains(Type::ReadOnly); }
62 bool active() const { return m_type.contains(Type::Active); }
63 bool move() const { return m_type.contains(Type::Move); }
64 bool release() const { return m_type.contains(Type::Release); }
65 bool ignoreCSSPointerEventsProperty() const { return m_type.contains(Type::IgnoreCSSPointerEventsProperty); }
66 bool ignoreClipping() const { return m_type.contains(Type::IgnoreClipping); }
67 bool svgClipContent() const { return m_type.contains(Type::SVGClipContent); }
68 bool touchEvent() const { return m_type.contains(Type::TouchEvent); }
69 bool mouseEvent() const { return !touchEvent(); }
70 bool disallowsUserAgentShadowContent() const { return m_type.contains(Type::DisallowUserAgentShadowContent); }
71 bool disallowsUserAgentShadowContentExceptForImageOverlays() const { return m_type.contains(Type::DisallowUserAgentShadowContentExceptForImageOverlays); }
72 bool allowsFrameScrollbars() const { return m_type.contains(Type::AllowFrameScrollbars); }
73 bool allowsChildFrameContent() const { return m_type.contains(Type::AllowChildFrameContent); }
74 bool allowsVisibleChildFrameContent() const { return m_type.contains(Type::AllowVisibleChildFrameContentOnly); }
75 bool isChildFrameHitTest() const { return m_type.contains(Type::ChildFrameHitTest); }
76 bool resultIsElementList() const { return m_type.contains(Type::CollectMultipleElements); }
77 bool includesAllElementsUnderPoint() const { return m_type.contains(Type::IncludeAllElementsUnderPoint); }
78
79 // Convenience functions
80 bool touchMove() const { return move() && touchEvent(); }
81 bool touchRelease() const { return release() && touchEvent(); }
82
83 OptionSet<Type> type() const { return m_type; }
84
85private:
86 OptionSet<Type> m_type;
87};
88
89} // namespace WebCore
