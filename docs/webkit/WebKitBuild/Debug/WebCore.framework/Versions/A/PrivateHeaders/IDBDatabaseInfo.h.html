26#pragma once
27
28#include "IDBObjectStoreInfo.h"
29#include <wtf/HashMap.h>
30#include <wtf/IsoMalloc.h>
31
32namespace WebCore {
33
34class IDBDatabaseInfo {
35 WTF_MAKE_ISO_ALLOCATED_EXPORT(IDBDatabaseInfo, WEBCORE_EXPORT);
36public:
37 explicit IDBDatabaseInfo(const String& name, uint64_t version, uint64_t maxIndexID);
38
39 enum IsolatedCopyTag { IsolatedCopy };
40 IDBDatabaseInfo(const IDBDatabaseInfo&, IsolatedCopyTag);
41
42 IDBDatabaseInfo isolatedCopy() const;
43
44 const String& name() const { return m_name; }
45
46 void setVersion(uint64_t version) { m_version = version; }
47 uint64_t version() const { return m_version; }
48
49 bool hasObjectStore(const String& name) const;
50 IDBObjectStoreInfo createNewObjectStore(const String& name, std::optional<IDBKeyPath>&&, bool autoIncrement);
51 void addExistingObjectStore(const IDBObjectStoreInfo&);
52 IDBObjectStoreInfo* infoForExistingObjectStore(uint64_t objectStoreIdentifier);
53 IDBObjectStoreInfo* infoForExistingObjectStore(const String& objectStoreName);
54 const IDBObjectStoreInfo* infoForExistingObjectStore(uint64_t objectStoreIdentifier) const;
55 const IDBObjectStoreInfo* infoForExistingObjectStore(const String& objectStoreName) const;
56
57 void renameObjectStore(uint64_t objectStoreIdentifier, const String& newName);
58
59 Vector<String> objectStoreNames() const;
60 const HashMap<uint64_t, IDBObjectStoreInfo>& objectStoreMap() const { return m_objectStoreMap; }
61
62 void deleteObjectStore(const String& objectStoreName);
63 void deleteObjectStore(uint64_t objectStoreIdentifier);
64
65 WEBCORE_EXPORT IDBDatabaseInfo();
66
67 template<class Encoder> void encode(Encoder&) const;
68 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBDatabaseInfo&);
69
70 void setMaxIndexID(uint64_t maxIndexID);
71 uint64_t generateNextIndexID() { return ++m_maxIndexID; }
72
73#if !LOG_DISABLED
74 String loggingString() const;
75#endif
76
77private:
78 IDBObjectStoreInfo* getInfoForExistingObjectStore(const String& objectStoreName);
79 IDBObjectStoreInfo* getInfoForExistingObjectStore(uint64_t objectStoreIdentifier);
80
81 String m_name;
82 uint64_t m_version { 0 };
83 uint64_t m_maxObjectStoreID { 0 };
84 uint64_t m_maxIndexID { 0 };
85
86 HashMap<uint64_t, IDBObjectStoreInfo> m_objectStoreMap;
87
88};
89
90template<class Encoder>
91void IDBDatabaseInfo::encode(Encoder& encoder) const
92{
93 encoder << m_name << m_version << m_maxObjectStoreID << m_maxIndexID << m_objectStoreMap;
94}
95
96template<class Decoder>
97bool IDBDatabaseInfo::decode(Decoder& decoder, IDBDatabaseInfo& info)
98{
99 if (!decoder.decode(info.m_name))
100 return false;
101
102 if (!decoder.decode(info.m_version))
103 return false;
104
105 if (!decoder.decode(info.m_maxObjectStoreID))
106 return false;
107
108 if (!decoder.decode(info.m_maxIndexID))
109 return false;
110
111 if (!decoder.decode(info.m_objectStoreMap))
112 return false;
113
114 return true;
115}
