26#pragma once
27
28#include <wtf/text/WTFString.h>
29
30namespace WebCore {
31
32struct IDBDatabaseNameAndVersion {
33 String name;
34 uint64_t version { 0 };
35
36 template<class Encoder> void encode(Encoder&) const;
37 template<class Decoder> static std::optional<IDBDatabaseNameAndVersion> decode(Decoder&);
38
39 IDBDatabaseNameAndVersion isolatedCopy() const;
40};
41
42template<class Encoder>
43void IDBDatabaseNameAndVersion::encode(Encoder& encoder) const
44{
45 encoder << name << version;
46}
47
48template<class Decoder>
49std::optional<IDBDatabaseNameAndVersion> IDBDatabaseNameAndVersion::decode(Decoder& decoder)
50{
51 IDBDatabaseNameAndVersion info;
52
53 if (!decoder.decode(info.name))
54 return std::nullopt;
55
56 if (!decoder.decode(info.version))
57 return std::nullopt;
58
59 return info;
60}
61
62inline IDBDatabaseNameAndVersion IDBDatabaseNameAndVersion::isolatedCopy() const
63{
64 return { name.isolatedCopy(), version };
65}
66
67} // namespace WebCore
