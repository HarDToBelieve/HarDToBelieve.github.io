44 }

46 IDBGetResult(const IDBKeyData& keyData)

49 }
50

112 Vector<IDBCursorRecord> m_prefetchedRecords;
113 bool m_isDefined { true };
114};
115
116template<class Encoder>
117void IDBGetResult::encode(Encoder& encoder) const
118{
119 encoder << m_keyData << m_primaryKeyData << m_keyPath << m_isDefined << m_value << m_prefetchedRecords;
120}
121
122template<class Decoder>
123bool IDBGetResult::decode(Decoder& decoder, IDBGetResult& result)
124{
125 std::optional<IDBKeyData> keyData;
126 decoder >> keyData;
127 if (!keyData)
128 return false;
129 result.m_keyData = WTFMove(*keyData);
130
131 std::optional<IDBKeyData> primaryKeyData;
132 decoder >> primaryKeyData;
133 if (!primaryKeyData)
134 return false;
135 result.m_primaryKeyData = WTFMove(*primaryKeyData);
136
137 if (!decoder.decode(result.m_keyPath))
138 return false;
139
140 if (!decoder.decode(result.m_isDefined))
141 return false;
142
143 std::optional<IDBValue> value;
144 decoder >> value;
145 if (!value)
146 return false;
147 result.m_value = WTFMove(*value);
148
149 std::optional<Vector<IDBCursorRecord>> prefetchedRecords;
150 decoder >> prefetchedRecords;
151 if (!prefetchedRecords)
152 return false;
153 result.m_prefetchedRecords = WTFMove(*prefetchedRecords);
154
155 return true;
156}
