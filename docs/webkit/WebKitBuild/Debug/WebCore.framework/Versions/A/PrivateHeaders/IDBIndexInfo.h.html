26#pragma once
27
28#include "IDBKeyPath.h"
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class IDBIndexInfo {
34public:
35 WEBCORE_EXPORT IDBIndexInfo();
36 IDBIndexInfo(uint64_t identifier, uint64_t objectStoreIdentifier, const String& name, IDBKeyPath&&, bool unique, bool multiEntry);
37
38 WEBCORE_EXPORT IDBIndexInfo isolatedCopy() const;
39
40 uint64_t identifier() const { return m_identifier; }
41 uint64_t objectStoreIdentifier() const { return m_objectStoreIdentifier; }
42 const String& name() const { return m_name; }
43 const IDBKeyPath& keyPath() const { return m_keyPath; }
44 bool unique() const { return m_unique; }
45 bool multiEntry() const { return m_multiEntry; }
46
47 void rename(const String& newName) { m_name = newName; }
48
49 template<class Encoder> void encode(Encoder&) const;
50 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBIndexInfo&);
51
52#if !LOG_DISABLED
53 String loggingString(int indent = 0) const;
54 String condensedLoggingString() const;
55#endif
56
57 // FIXME: Remove the need for this.
58 static const int64_t InvalidId = -1;
59
60 void setIdentifier(uint64_t identifier) { m_identifier = identifier; }
61private:
62 uint64_t m_identifier { 0 };
63 uint64_t m_objectStoreIdentifier { 0 };
64 String m_name;
65 IDBKeyPath m_keyPath;
66 bool m_unique { true };
67 bool m_multiEntry { false };
68};
69
70template<class Encoder>
71void IDBIndexInfo::encode(Encoder& encoder) const
72{
73 encoder << m_identifier << m_objectStoreIdentifier << m_name << m_keyPath << m_unique << m_multiEntry;
74}
75
76template<class Decoder>
77bool IDBIndexInfo::decode(Decoder& decoder, IDBIndexInfo& info)
78{
79 if (!decoder.decode(info.m_identifier))
80 return false;
81
82 if (!decoder.decode(info.m_objectStoreIdentifier))
83 return false;
84
85 if (!decoder.decode(info.m_name))
86 return false;
87
88 if (!decoder.decode(info.m_keyPath))
89 return false;
90
91 if (!decoder.decode(info.m_unique))
92 return false;
93
94 if (!decoder.decode(info.m_multiEntry))
95 return false;
96
97 return true;
98}
99
100} // namespace WebCore
