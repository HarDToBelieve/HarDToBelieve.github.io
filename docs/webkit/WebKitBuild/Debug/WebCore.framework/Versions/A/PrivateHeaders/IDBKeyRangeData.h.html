26#pragma once
27
28#include "IDBKeyData.h"
29#include "IDBKeyRange.h"
30
31namespace WebCore {
32
33class IDBKey;
34
35struct IDBKeyRangeData {
36 IDBKeyRangeData()
37 : isNull(true)
38 {
39 }
40
41 static IDBKeyRangeData allKeys()
42 {
43 IDBKeyRangeData result;
44 result.isNull = false;
45 result.lowerKey = IDBKeyData::minimum();
46 result.upperKey = IDBKeyData::maximum();
47 return result;
48 }
49
50 IDBKeyRangeData(IDBKey*);
51 IDBKeyRangeData(const IDBKeyData&);
52
53 IDBKeyRangeData(IDBKeyRange* keyRange)
54 : isNull(!keyRange)
55 {
56 if (isNull)
57 return;
58
59 lowerKey = keyRange->lower();
60 upperKey = keyRange->upper();
61 lowerOpen = keyRange->lowerOpen();
62 upperOpen = keyRange->upperOpen();
63 }
64
65 WEBCORE_EXPORT IDBKeyRangeData isolatedCopy() const;
66
67 WEBCORE_EXPORT RefPtr<IDBKeyRange> maybeCreateIDBKeyRange() const;
68
69 WEBCORE_EXPORT bool isExactlyOneKey() const;
70 bool containsKey(const IDBKeyData&) const;
71 bool isValid() const;
72
73 template<class Encoder> void encode(Encoder&) const;
74 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBKeyRangeData&);
75
76 IDBKeyData lowerKey;
77 IDBKeyData upperKey;
78
79 bool lowerOpen { false };
80 bool upperOpen { false };
81
82 bool isNull;
83
84#if !LOG_DISABLED
85 String loggingString() const;
86#endif
87};
88
89template<class Encoder>
90void IDBKeyRangeData::encode(Encoder& encoder) const
91{
92 encoder << isNull;
93 if (isNull)
94 return;
95
96 encoder << upperKey << lowerKey << upperOpen << lowerOpen;
97}
98
99template<class Decoder>
100bool IDBKeyRangeData::decode(Decoder& decoder, IDBKeyRangeData& keyRange)
101{
102 if (!decoder.decode(keyRange.isNull))
103 return false;
104
105 if (keyRange.isNull)
106 return true;
107
108 std::optional<IDBKeyData> upperKey;
109 decoder >> upperKey;
110 if (!upperKey)
111 return false;
112 keyRange.upperKey = WTFMove(*upperKey);
113
114 std::optional<IDBKeyData> lowerKey;
115 decoder >> lowerKey;
116 if (!lowerKey)
117 return false;
118 keyRange.lowerKey = WTFMove(*lowerKey);
119
120 if (!decoder.decode(keyRange.upperOpen))
121 return false;
122
123 if (!decoder.decode(keyRange.lowerOpen))
124 return false;
125
126 return true;
127}

129} // namespace WebCore
