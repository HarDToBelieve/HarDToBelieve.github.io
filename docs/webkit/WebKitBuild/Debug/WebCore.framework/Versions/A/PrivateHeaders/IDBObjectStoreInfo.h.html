26#pragma once
27
28#include "IDBIndexInfo.h"
29#include "IDBKeyPath.h"
30#include <wtf/HashMap.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35class IDBObjectStoreInfo {
36public:
37 WEBCORE_EXPORT IDBObjectStoreInfo();
38 IDBObjectStoreInfo(uint64_t identifier, const String& name, std::optional<IDBKeyPath>&&, bool autoIncrement);
39
40 uint64_t identifier() const { return m_identifier; }
41 const String& name() const { return m_name; }
42 const std::optional<IDBKeyPath>& keyPath() const { return m_keyPath; }
43 bool autoIncrement() const { return m_autoIncrement; }
44
45 void rename(const String& newName) { m_name = newName; }
46
47 WEBCORE_EXPORT IDBObjectStoreInfo isolatedCopy() const;
48
49 IDBIndexInfo createNewIndex(uint64_t indexID, const String& name, IDBKeyPath&&, bool unique, bool multiEntry);
50 void addExistingIndex(const IDBIndexInfo&);
51 bool hasIndex(const String& name) const;
52 bool hasIndex(uint64_t indexIdentifier) const;
53 IDBIndexInfo* infoForExistingIndex(const String& name);
54 IDBIndexInfo* infoForExistingIndex(uint64_t identifier);
55
56 Vector<String> indexNames() const;
57 const HashMap<uint64_t, IDBIndexInfo>& indexMap() const { return m_indexMap; }
58
59 void deleteIndex(const String& indexName);
60 void deleteIndex(uint64_t indexIdentifier);
61
62 template<class Encoder> void encode(Encoder&) const;
63 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBObjectStoreInfo&);
64
65#if !LOG_DISABLED
66 String loggingString(int indent = 0) const;
67 String condensedLoggingString() const;
68#endif
69
70private:
71 uint64_t m_identifier { 0 };
72 String m_name;
73 std::optional<IDBKeyPath> m_keyPath;
74 bool m_autoIncrement { false };
75
76 HashMap<uint64_t, IDBIndexInfo> m_indexMap;
77};
78
79template<class Encoder>
80void IDBObjectStoreInfo::encode(Encoder& encoder) const
81{
82 encoder << m_identifier << m_name << m_keyPath << m_autoIncrement << m_indexMap;
83}
84
85template<class Decoder>
86bool IDBObjectStoreInfo::decode(Decoder& decoder, IDBObjectStoreInfo& info)
87{
88 if (!decoder.decode(info.m_identifier))
89 return false;
90
91 if (!decoder.decode(info.m_name))
92 return false;
93
94 if (!decoder.decode(info.m_keyPath))
95 return false;
96
97 if (!decoder.decode(info.m_autoIncrement))
98 return false;
99
100 if (!decoder.decode(info.m_indexMap))
101 return false;
102
103 return true;
104}

106} // namespace WebCore
