26#pragma once
27
28#include "IDBDatabaseIdentifier.h"
29#include "IDBRequest.h"
30
31namespace WebCore {
32
33class IDBResultData;
34
35class IDBOpenDBRequest final : public IDBRequest {
36 WTF_MAKE_ISO_ALLOCATED(IDBOpenDBRequest);
37public:
38 static Ref<IDBOpenDBRequest> createDeleteRequest(ScriptExecutionContext&, IDBClient::IDBConnectionProxy&, const IDBDatabaseIdentifier&);
39 static Ref<IDBOpenDBRequest> createOpenRequest(ScriptExecutionContext&, IDBClient::IDBConnectionProxy&, const IDBDatabaseIdentifier&, uint64_t version);
40
41 virtual ~IDBOpenDBRequest();
42
43 const IDBDatabaseIdentifier& databaseIdentifier() const { return m_databaseIdentifier; }
44 uint64_t version() const { return m_version; }
45
46 void requestCompleted(const IDBResultData&);
47 void requestBlocked(uint64_t oldVersion, uint64_t newVersion);
48
49 void versionChangeTransactionDidFinish();
50 void fireSuccessAfterVersionChangeCommit();
51 void fireErrorAfterVersionChangeCompletion();
52
53 void setIsContextSuspended(bool);
54 bool isContextSuspended() const { return m_isContextSuspended; }
55
56private:
57 IDBOpenDBRequest(ScriptExecutionContext&, IDBClient::IDBConnectionProxy&, const IDBDatabaseIdentifier&, uint64_t version, IndexedDB::RequestType);
58
59 void dispatchEvent(Event&) final;
60
61 void cancelForStop() final;
62
63 void onError(const IDBResultData&);
64 void onSuccess(const IDBResultData&);
65 void onUpgradeNeeded(const IDBResultData&);
66 void onDeleteDatabaseSuccess(const IDBResultData&);
67
68 bool isOpenDBRequest() const final { return true; }
69
70 IDBDatabaseIdentifier m_databaseIdentifier;
71 uint64_t m_version { 0 };
72
73 bool m_isContextSuspended { false };
74 bool m_isBlocked { false };
75};
76
77} // namespace WebCore
