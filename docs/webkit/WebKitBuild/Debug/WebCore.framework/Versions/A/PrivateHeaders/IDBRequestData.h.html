46class IDBRequestData {
47public:
48 IDBRequestData(const IDBClient::IDBConnectionProxy&, const IDBOpenDBRequest&);
49 explicit IDBRequestData(IDBClient::TransactionOperation&);
50 WEBCORE_EXPORT IDBRequestData(const IDBRequestData&);
51 IDBRequestData(IDBRequestData&&) = default;
52 IDBRequestData& operator=(IDBRequestData&&) = default;
53
54 enum IsolatedCopyTag { IsolatedCopy };
55 IDBRequestData(const IDBRequestData&, IsolatedCopyTag);
56 WEBCORE_EXPORT IDBRequestData isolatedCopy() const;
57
58 IDBConnectionIdentifier serverConnectionIdentifier() const;
59 IDBResourceIdentifier requestIdentifier() const;
60 IDBResourceIdentifier transactionIdentifier() const;
61 uint64_t objectStoreIdentifier() const;
62 uint64_t indexIdentifier() const;
63 IndexedDB::IndexRecordType indexRecordType() const;
64 IDBResourceIdentifier cursorIdentifier() const;
65
66 const IDBDatabaseIdentifier& databaseIdentifier() const;
67 uint64_t requestedVersion() const;
68
69 bool isOpenRequest() const { return m_requestType == IndexedDB::RequestType::Open; }
70 bool isDeleteRequest() const { return m_requestType == IndexedDB::RequestType::Delete; }
71
72 IDBRequestData isolatedCopy();
73
74 WEBCORE_EXPORT IDBRequestData();
75
76 template<class Encoder> void encode(Encoder&) const;
77 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBRequestData&);
78
79private:
80 static void isolatedCopy(const IDBRequestData& source, IDBRequestData& destination);
81
82 IDBConnectionIdentifier m_serverConnectionIdentifier;
83 std::unique_ptr<IDBResourceIdentifier> m_requestIdentifier;
84 std::unique_ptr<IDBResourceIdentifier> m_transactionIdentifier;
85 std::unique_ptr<IDBResourceIdentifier> m_cursorIdentifier;
86 uint64_t m_objectStoreIdentifier { 0 };
87 uint64_t m_indexIdentifier { 0 };
88 IndexedDB::IndexRecordType m_indexRecordType { IndexedDB::IndexRecordType::Key };
89
90 mutable std::optional<IDBDatabaseIdentifier> m_databaseIdentifier;
91 uint64_t m_requestedVersion { 0 };
92
93 IndexedDB::RequestType m_requestType { IndexedDB::RequestType::Other };
94};
95
96inline const IDBDatabaseIdentifier& IDBRequestData::databaseIdentifier() const
97{
98 ASSERT(m_databaseIdentifier);
99 if (!m_databaseIdentifier)
100 m_databaseIdentifier = IDBDatabaseIdentifier { };
101 return *m_databaseIdentifier;
102}
103
104template<class Encoder>
105void IDBRequestData::encode(Encoder& encoder) const
106{
107 encoder << m_serverConnectionIdentifier << m_objectStoreIdentifier << m_indexIdentifier << m_databaseIdentifier << m_requestedVersion;
108
109 encoder << m_indexRecordType;
110 encoder << m_requestType;
111
112 encoder << !!m_requestIdentifier;
113 if (m_requestIdentifier)
114 encoder << *m_requestIdentifier;
115
116 encoder << !!m_transactionIdentifier;
117 if (m_transactionIdentifier)
118 encoder << *m_transactionIdentifier;
119
120 encoder << !!m_cursorIdentifier;
121 if (m_cursorIdentifier)
122 encoder << *m_cursorIdentifier;
123}
124
125template<class Decoder>
126bool IDBRequestData::decode(Decoder& decoder, IDBRequestData& request)
127{
128 if (!decoder.decode(request.m_serverConnectionIdentifier))
129 return false;
130
131 if (!decoder.decode(request.m_objectStoreIdentifier))
132 return false;
133
134 if (!decoder.decode(request.m_indexIdentifier))
135 return false;
136
137 std::optional<std::optional<IDBDatabaseIdentifier>> databaseIdentifier;
138 decoder >> databaseIdentifier;
139 if (!databaseIdentifier)
140 return false;
141 request.m_databaseIdentifier = WTFMove(*databaseIdentifier);
142
143 if (!decoder.decode(request.m_requestedVersion))
144 return false;
145
146 if (!decoder.decode(request.m_indexRecordType))
147 return false;
148
149 if (!decoder.decode(request.m_requestType))
150 return false;
151
152 bool hasObject;
153
154 if (!decoder.decode(hasObject))
155 return false;
156 if (hasObject) {
157 std::unique_ptr<IDBResourceIdentifier> object = makeUnique<IDBResourceIdentifier>();
158 if (!decoder.decode(*object))
159 return false;
160 request.m_requestIdentifier = WTFMove(object);
161 }
162
163 if (!decoder.decode(hasObject))
164 return false;
165 if (hasObject) {
166 std::unique_ptr<IDBResourceIdentifier> object = makeUnique<IDBResourceIdentifier>();
167 if (!decoder.decode(*object))
168 return false;
169 request.m_transactionIdentifier = WTFMove(object);
170 }
171
172 if (!decoder.decode(hasObject))
173 return false;
174 if (hasObject) {
175 std::unique_ptr<IDBResourceIdentifier> object = makeUnique<IDBResourceIdentifier>();
176 if (!decoder.decode(*object))
177 return false;
178 request.m_cursorIdentifier = WTFMove(object);
179 }
180
181 return true;
182}
