40namespace WebCore {
41
42class DestinatationColorSpace;
43
44class IOSurfacePool {
45 WTF_MAKE_NONCOPYABLE(IOSurfacePool);
46 WTF_MAKE_FAST_ALLOCATED;
47 friend class LazyNeverDestroyed<IOSurfacePool>;
48
49public:
50 WEBCORE_EXPORT static IOSurfacePool& sharedPool();
51
52 std::unique_ptr<IOSurface> takeSurface(IntSize, const DestinationColorSpace&, IOSurface::Format);
53 WEBCORE_EXPORT void addSurface(std::unique_ptr<IOSurface>);
54
55 void discardAllSurfaces();
56
57 WEBCORE_EXPORT void setPoolSize(size_t);
58
59private:
60 IOSurfacePool();
61
62 struct CachedSurfaceDetails {
63 CachedSurfaceDetails()
64 : hasMarkedPurgeable(false)
65 { }
66
67 void resetLastUseTime() { lastUseTime = MonotonicTime::now(); }
68
69 MonotonicTime lastUseTime;
70 bool hasMarkedPurgeable;
71 };
72
73 typedef Deque<std::unique_ptr<IOSurface>> CachedSurfaceQueue;
74 typedef HashMap<IntSize, CachedSurfaceQueue> CachedSurfaceMap;
75 typedef HashMap<IOSurface*, CachedSurfaceDetails> CachedSurfaceDetailsMap;
76
77 static constexpr size_t defaultMaximumBytesCached { 1024 * 1024 * 64 };
78
79 // We'll never allow more than 1/2 of the cache to be filled with in-use surfaces, because
80 // they can't be immediately returned when requested (but will be freed up in the future).
81 static constexpr size_t maximumInUseBytes = defaultMaximumBytesCached / 2;
82
83 bool shouldCacheSurface(const IOSurface&) const WTF_REQUIRES_LOCK(m_lock);
84
85 void willAddSurface(IOSurface&, bool inUse) WTF_REQUIRES_LOCK(m_lock);
86 void didRemoveSurface(IOSurface&, bool inUse) WTF_REQUIRES_LOCK(m_lock);
87 void didUseSurfaceOfSize(IntSize) WTF_REQUIRES_LOCK(m_lock);
88
89 void insertSurfaceIntoPool(std::unique_ptr<IOSurface>) WTF_REQUIRES_LOCK(m_lock);
90
91 void evict(size_t additionalSize) WTF_REQUIRES_LOCK(m_lock);
92 void tryEvictInUseSurface() WTF_REQUIRES_LOCK(m_lock);
93 void tryEvictOldestCachedSurface() WTF_REQUIRES_LOCK(m_lock);
94
95 void scheduleCollectionTimer() WTF_REQUIRES_LOCK(m_lock);
96 void collectionTimerFired();
97 void collectInUseSurfaces() WTF_REQUIRES_LOCK(m_lock);
98 bool markOlderSurfacesPurgeable() WTF_REQUIRES_LOCK(m_lock);
99
100 void platformGarbageCollectNow();
101
102 void discardAllSurfacesInternal() WTF_REQUIRES_LOCK(m_lock);
103
104 void showPoolStatistics(const char*);
105
106 Lock m_lock;
107 RunLoop::Timer<IOSurfacePool> m_collectionTimer WTF_GUARDED_BY_LOCK(m_lock);
108 CachedSurfaceMap m_cachedSurfaces WTF_GUARDED_BY_LOCK(m_lock);
109 CachedSurfaceQueue m_inUseSurfaces WTF_GUARDED_BY_LOCK(m_lock);
110 CachedSurfaceDetailsMap m_surfaceDetails WTF_GUARDED_BY_LOCK(m_lock);
111 Vector<IntSize> m_sizesInPruneOrder WTF_GUARDED_BY_LOCK(m_lock);
112
113 size_t m_bytesCached WTF_GUARDED_BY_LOCK(m_lock) { 0 };
114 size_t m_inUseBytesCached WTF_GUARDED_BY_LOCK(m_lock) { 0 };
115 size_t m_maximumBytesCached WTF_GUARDED_BY_LOCK(m_lock) { defaultMaximumBytesCached };
116};
117
118}
119#endif // HAVE(IOSURFACE)
