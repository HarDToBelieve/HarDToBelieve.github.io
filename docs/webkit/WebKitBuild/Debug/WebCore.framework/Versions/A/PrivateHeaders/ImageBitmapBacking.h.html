26#pragma once
27
28#include "ImageBuffer.h"
29#include <wtf/OptionSet.h>
30
31namespace WebCore {
32
33enum class SerializationState : uint8_t {
34 OriginClean = 1 << 0,
35 PremultiplyAlpha = 1 << 1,
36 ForciblyPremultiplyAlpha = 1 << 2
37};
38
39class ImageBitmapBacking {
40public:
41 ImageBitmapBacking(RefPtr<ImageBuffer>&&, OptionSet<SerializationState> = SerializationState::OriginClean);
42
43 ImageBuffer* buffer() const;
44 RefPtr<ImageBuffer> takeImageBuffer();
45
46 unsigned width() const;
47 unsigned height() const;
48
49 bool originClean() const { return m_serializationState.contains(SerializationState::OriginClean); }
50
51 bool premultiplyAlpha() const { return m_serializationState.contains(SerializationState::PremultiplyAlpha); }
52 // When WebGL consumes an Image coming from an ImageBitmap's ImageBuffer, it typically honors
53 // the alpha mode of that native image - CGImageAlphaInfo in the Core Graphics backend. For
54 // ImageBitmaps created from ImageBitmaps, this information is not accurate, and callers must be
55 // told to ignore the alpha mode, and forcibly premultiply the alpha channel.
56 bool forciblyPremultiplyAlpha() const { return m_serializationState.contains(SerializationState::ForciblyPremultiplyAlpha); }
57
58 OptionSet<SerializationState> serializationState() const { return m_serializationState; }
59
60private:
61 RefPtr<ImageBuffer> m_bitmapData;
62 OptionSet<SerializationState> m_serializationState;
63};

65} // namespace WebCore
