51 // Create an image buffer compatible with the context, with suitable resolution for drawing into the buffer and then into this context.

53 static RefPtr<ImageBuffer> createCompatibleBuffer(const FloatSize&, const DestinationColorSpace&, const GraphicsContext&);

87 virtual IntSize logicalSize() const = 0;

93 virtual IntSize backendSize() const = 0;

96 virtual size_t memoryCost() const = 0;

99 virtual bool isInUse() const = 0;
