72class ImageBufferBackend {
73public:
74 struct Parameters {
75 FloatSize logicalSize;
76 float resolutionScale;
77 DestinationColorSpace colorSpace;
78 PixelFormat pixelFormat;
79 };
80
81 WEBCORE_EXPORT virtual ~ImageBufferBackend();
82
83 WEBCORE_EXPORT static IntSize calculateBackendSize(const Parameters&);
84 WEBCORE_EXPORT static size_t calculateMemoryCost(const IntSize& backendSize, unsigned bytesPerRow);
85 static size_t calculateExternalMemoryCost(const Parameters&) { return 0; }
86
87 virtual GraphicsContext& context() const = 0;
88 virtual void flushContext() { }
89
90 virtual IntSize backendSize() const { return { }; }
91
92 virtual RefPtr<NativeImage> copyNativeImage(BackingStoreCopy) const = 0;
93 virtual RefPtr<Image> copyImage(BackingStoreCopy, PreserveResolution) const = 0;
94
95 WEBCORE_EXPORT virtual void draw(GraphicsContext&, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions&) = 0;
96 WEBCORE_EXPORT virtual void drawPattern(GraphicsContext&, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions&) = 0;
97
98 WEBCORE_EXPORT virtual RefPtr<NativeImage> sinkIntoNativeImage();
99 WEBCORE_EXPORT virtual RefPtr<Image> sinkIntoImage(PreserveResolution);
100 WEBCORE_EXPORT virtual void drawConsuming(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions&);
101
102 virtual void clipToMask(GraphicsContext&, const FloatRect&) { }
103
104 WEBCORE_EXPORT void convertToLuminanceMask();
105 virtual void transformToColorSpace(const DestinationColorSpace&) { }
106
107 virtual String toDataURL(const String& mimeType, std::optional<double> quality, PreserveResolution) const = 0;
108 virtual Vector<uint8_t> toData(const String& mimeType, std::optional<double> quality) const = 0;
109
110 virtual std::optional<PixelBuffer> getPixelBuffer(const PixelBufferFormat& outputFormat, const IntRect&) const = 0;
111 virtual void putPixelBuffer(const PixelBuffer&, const IntRect& srcRect, const IntPoint& destPoint, AlphaPremultiplication destFormat) = 0;
112
113 virtual PlatformLayer* platformLayer() const { return nullptr; }
114 virtual bool copyToPlatformTexture(GraphicsContextGL&, GCGLenum, PlatformGLObject, GCGLenum, bool, bool) const { return false; }
115
116 virtual bool isInUse() const { return false; }
117 virtual void releaseGraphicsContext() { ASSERT_NOT_REACHED(); }
118 virtual VolatilityState setVolatile(bool) { return VolatilityState::Valid; }
119 virtual void releaseBufferToPool() { }
120
121 virtual std::unique_ptr<ThreadSafeImageBufferFlusher> createFlusher() { return nullptr; }
122
123 static constexpr bool isOriginAtBottomLeftCorner = false;
124 static constexpr bool canMapBackingStore = true;
125 static constexpr RenderingMode renderingMode = RenderingMode::Unaccelerated;
126
127protected:
128 WEBCORE_EXPORT ImageBufferBackend(const Parameters&);
129
130 virtual unsigned bytesPerRow() const = 0;
131
132 template<typename T>
133 T toBackendCoordinates(T t) const
134 {
135 static_assert(std::is_same<T, IntPoint>::value || std::is_same<T, IntSize>::value || std::is_same<T, IntRect>::value);
136 if (resolutionScale() != 1)
137 t.scale(resolutionScale());
138 return t;
139 }
140
141 IntSize logicalSize() const { return IntSize(m_parameters.logicalSize); }
142 float resolutionScale() const { return m_parameters.resolutionScale; }
143 const DestinationColorSpace& colorSpace() const { return m_parameters.colorSpace; }
144 PixelFormat pixelFormat() const { return m_parameters.pixelFormat; }
145
146 IntRect logicalRect() const { return IntRect(IntPoint::zero(), logicalSize()); };
147 IntRect backendRect() const { return IntRect(IntPoint::zero(), backendSize()); };
148
149 WEBCORE_EXPORT std::optional<PixelBuffer> getPixelBuffer(const PixelBufferFormat& outputFormat, const IntRect& srcRect, void* data) const;
150 WEBCORE_EXPORT void putPixelBuffer(const PixelBuffer&, const IntRect& srcRect, const IntPoint& destPoint, AlphaPremultiplication destFormat, void* data);
151
152 Parameters m_parameters;
153};
