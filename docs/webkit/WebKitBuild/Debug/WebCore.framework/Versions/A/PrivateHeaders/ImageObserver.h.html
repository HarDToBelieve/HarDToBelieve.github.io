26#pragma once
27
28#include "ImageTypes.h"
29
30namespace WebCore {
31
32class Image;
33class IntRect;
34
35// Interface for notification about changes to an image, including decoding,
36// drawing, and animating.
37class ImageObserver {
38protected:
39 virtual ~ImageObserver() = default;
40public:
41 virtual URL sourceUrl() const = 0;
42 virtual String mimeType() const = 0;
43 virtual long long expectedContentLength() const = 0;
44
45 virtual void encodedDataStatusChanged(const Image&, EncodedDataStatus) { };
46 virtual void decodedSizeChanged(const Image&, long long delta) = 0;
47
48 virtual void didDraw(const Image&) = 0;
49
50 virtual bool canDestroyDecodedData(const Image&) = 0;
51 virtual void imageFrameAvailable(const Image&, ImageAnimatingState, const IntRect* changeRect = nullptr, DecodingStatus = DecodingStatus::Invalid) = 0;
52 virtual void changedInRect(const Image&, const IntRect* changeRect = nullptr) = 0;
53 virtual void scheduleRenderingUpdate(const Image&) = 0;
54};
55
56}
