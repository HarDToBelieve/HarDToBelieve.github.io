26#pragma once
27
28#include "DecodingOptions.h"
29#include "GraphicsTypes.h"
30#include "ImageOrientation.h"

32namespace WebCore {
33
34struct ImagePaintingOptions {
35 template<typename... Options>
36 ImagePaintingOptions(Options... options)
37 {
38 setOption(options...);
39 }
40
41 template<typename... Overrides>
42 ImagePaintingOptions(const ImagePaintingOptions& other, Overrides... overrides)
43 : ImagePaintingOptions(other)
44 {
45 setOption(overrides...);
46 }
47
48 ImagePaintingOptions() = default;
49 ImagePaintingOptions(const ImagePaintingOptions&) = default;
50
51 CompositeOperator compositeOperator() const { return m_compositeOperator; }
52 BlendMode blendMode() const { return m_blendMode; }
53 DecodingMode decodingMode() const { return m_decodingMode; }
54 ImageOrientation orientation() const { return m_orientation; }
55 InterpolationQuality interpolationQuality() const { return m_interpolationQuality; }
56
57 template<class Encoder> void encode(Encoder&) const;
58 template<class Decoder> static std::optional<ImagePaintingOptions> decode(Decoder&);
59
60private:
61 template <typename First, typename... Rest>
62 void setOption(First first, Rest... rest)
63 {
64 setOption(first);
65 setOption(rest...);
66 }
67
68 void setOption(CompositeOperator compositeOperator) { m_compositeOperator = compositeOperator; }
69 void setOption(BlendMode blendMode) { m_blendMode = blendMode; }
70 void setOption(DecodingMode decodingMode) { m_decodingMode = decodingMode; }
71 void setOption(ImageOrientation orientation) { m_orientation = orientation; }
72 void setOption(ImageOrientation::Orientation orientation) { m_orientation = orientation; }
73 void setOption(InterpolationQuality interpolationQuality) { m_interpolationQuality = interpolationQuality; }
74
75 CompositeOperator m_compositeOperator { CompositeOperator::SourceOver };
76 BlendMode m_blendMode { BlendMode::Normal };
77 DecodingMode m_decodingMode { DecodingMode::Synchronous };
78 ImageOrientation m_orientation { ImageOrientation::None };
79 InterpolationQuality m_interpolationQuality { InterpolationQuality::Default };
80};
81
82template<class Encoder>
83void ImagePaintingOptions::encode(Encoder& encoder) const
84{
85 encoder << m_compositeOperator;
86 encoder << m_blendMode;
87 encoder << m_decodingMode;
88 encoder << ImageOrientation::Orientation(m_orientation);
89 encoder << m_interpolationQuality;
90}
91
92template<class Decoder>
93std::optional<ImagePaintingOptions> ImagePaintingOptions::decode(Decoder& decoder)
94{
95 std::optional<CompositeOperator> compositeOperator;
96 decoder >> compositeOperator;
97 if (!compositeOperator)
98 return std::nullopt;
99
100 std::optional<BlendMode> blendMode;
101 decoder >> blendMode;
102 if (!blendMode)
103 return std::nullopt;
104
105 std::optional<DecodingMode> decodingMode;
106 decoder >> decodingMode;
107 if (!decodingMode)
108 return std::nullopt;
109
110 std::optional<ImageOrientation::Orientation> orientation;
111 decoder >> orientation;
112 if (!orientation)
113 return std::nullopt;
114
115 std::optional<InterpolationQuality> interpolationQuality;
116 decoder >> interpolationQuality;
117 if (!interpolationQuality)
118 return std::nullopt;
119
120 return ImagePaintingOptions { *compositeOperator, *blendMode, *decodingMode, *orientation, *interpolationQuality };
121}
