35namespace WebCore {
36
37class MediaSample;
38
39class ImageRotationSessionVT final {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 struct RotationProperties {
43 bool flipX { false };
44 bool flipY { false };
45 unsigned angle { 0 };
46
47 bool isIdentity() const { return !flipX && !flipY && !angle; }
48 };
49
50 enum class IsCGImageCompatible {
51 No,
52 Yes,
53 };
54
55 ImageRotationSessionVT(AffineTransform&&, FloatSize, IsCGImageCompatible);
56 ImageRotationSessionVT(const RotationProperties&, FloatSize, IsCGImageCompatible);
57 ImageRotationSessionVT() = default;
58
59 const std::optional<AffineTransform>& transform() const { return m_transform; }
60 const RotationProperties& rotationProperties() const { return m_rotationProperties; }
61 const FloatSize& size() { return m_size; }
62 const FloatSize& rotatedSize() { return m_rotatedSize; }
63
64 RetainPtr<CVPixelBufferRef> rotate(CVPixelBufferRef);
65 WEBCORE_EXPORT RetainPtr<CVPixelBufferRef> rotate(MediaSample&, const RotationProperties&, IsCGImageCompatible);
66
67private:
68 void initialize(const RotationProperties&, FloatSize, IsCGImageCompatible);
69
70 RotationProperties m_rotationProperties;
71 FloatSize m_size;
72 std::optional<AffineTransform> m_transform;
73 OSType m_pixelFormat;
74 IsCGImageCompatible m_isCGImageCompatible;
75 FloatSize m_rotatedSize;
76 RetainPtr<VTImageRotationSessionRef> m_rotationSession;
77 RetainPtr<CVPixelBufferPoolRef> m_rotationPool;
78};
79
80inline bool operator==(const ImageRotationSessionVT::RotationProperties& rotation1, const ImageRotationSessionVT::RotationProperties& rotation2)
81{
82 return rotation1.flipX == rotation2.flipX && rotation1.flipY == rotation2.flipY && rotation1.angle == rotation2.angle;
83}
84
85inline bool operator!=(const ImageRotationSessionVT::RotationProperties& rotation1, const ImageRotationSessionVT::RotationProperties& rotation2)
86{
87 return !(rotation1 == rotation2);
88}
