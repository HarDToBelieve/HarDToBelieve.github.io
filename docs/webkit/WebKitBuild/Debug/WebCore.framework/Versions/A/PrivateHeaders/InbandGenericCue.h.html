36struct GenericCueData {
37
38 enum class Alignment : uint8_t { None, Start, Middle, End };
39 enum class Status : uint8_t { Uninitialized, Partial, Complete };
40
41 GenericCueData() = default;
42 GenericCueData(InbandGenericCueIdentifier uniqueId, const MediaTime& startTime, const MediaTime& endTime, const String& id, const String& content, const String& fontName, double line, double position, double size, double baseFontSize, double relativeFontSize, const Color& foregroundColor, const Color& backgroundColor, const Color& highlightColor, GenericCueData::Alignment align, GenericCueData::Status status)
43 : m_uniqueId(uniqueId)
44 , m_startTime(startTime)
45 , m_endTime(endTime)
46 , m_id(id)
47 , m_content(content)
48 , m_fontName(fontName)
49 , m_line(line)
50 , m_position(position)
51 , m_size(size)
52 , m_baseFontSize(baseFontSize)
53 , m_relativeFontSize(relativeFontSize)
54 , m_foregroundColor(foregroundColor)
55 , m_backgroundColor(backgroundColor)
56 , m_highlightColor(highlightColor)
57 , m_align(align)
58 , m_status(status)
59 {
60 ASSERT(isValid());
61 }
62
63 bool isValid() const { return !!m_uniqueId; }
64 bool equalNotConsideringTimesOrId(const GenericCueData&) const;
65
66 InbandGenericCueIdentifier m_uniqueId;
67 MediaTime m_startTime;
68 MediaTime m_endTime;
69 String m_id;
70 String m_content;
71 String m_fontName;
72 double m_line { -1 };
73 double m_position { -1 };
74 double m_size { -1 };
75 double m_baseFontSize { 0 };
76 double m_relativeFontSize { 0 };
77 Color m_foregroundColor;
78 Color m_backgroundColor;
79 Color m_highlightColor;
80 Alignment m_align { Alignment::None };
81 Status m_status { Status::Uninitialized };
82
83 template<class Encoder> void encode(Encoder&) const;
84 template<class Decoder> static std::optional<GenericCueData> decode(Decoder&);
85};

87template <class Decoder>
88std::optional<GenericCueData> GenericCueData::decode(Decoder& decoder)
89{
90 std::optional<InbandGenericCueIdentifier> uniqueId;
91 decoder >> uniqueId;
92 if (!uniqueId)
93 return std::nullopt;
94
95 std::optional<MediaTime> startTime;
96 decoder >> startTime;
97 if (!startTime)
98 return std::nullopt;
99
100 std::optional<MediaTime> endTime;
101 decoder >> endTime;
102 if (!endTime)
103 return std::nullopt;
104
105 std::optional<String> identifier;
106 decoder >> identifier;
107 if (!identifier)
108 return std::nullopt;
109
110 std::optional<String> content;
111 decoder >> content;
112 if (!content)
113 return std::nullopt;
114
115 std::optional<String> fontName;
116 decoder >> fontName;
117 if (!fontName)
118 return std::nullopt;
119
120 std::optional<double> line;
121 decoder >> line;
122 if (!line)
123 return std::nullopt;
124
125 std::optional<double> position;
126 decoder >> position;
127 if (!position)
128 return std::nullopt;
129
130 std::optional<double> size;
131 decoder >> size;
132 if (!size)
133 return std::nullopt;
134
135 std::optional<double> baseFontSize;
136 decoder >> baseFontSize;
137 if (!baseFontSize)
138 return std::nullopt;
139
140 std::optional<double> relativeFontSize;
141 decoder >> relativeFontSize;
142 if (!relativeFontSize)
143 return std::nullopt;
144
145 std::optional<Color> foregroundColor;
146 decoder >> foregroundColor;
147 if (!foregroundColor)
148 return std::nullopt;
149
150 std::optional<Color> backgroundColor;
151 decoder >> backgroundColor;
152 if (!backgroundColor)
153 return std::nullopt;
154
155 std::optional<Color> highlightColor;
156 decoder >> highlightColor;
157 if (!highlightColor)
158 return std::nullopt;
159
160 std::optional<Alignment> alignment;
161 decoder >> alignment;
162 if (!alignment)
163 return std::nullopt;
164
165 std::optional<Status> status;
166 decoder >> status;
167 if (!status)
168 return std::nullopt;
169
170 GenericCueData data = {
171
172 WTFMove(*uniqueId),
173
174 WTFMove(*startTime),
175 WTFMove(*endTime),
176
177 WTFMove(*identifier),
178 WTFMove(*content),
179 WTFMove(*fontName),
180
181 WTFMove(*line),
182 WTFMove(*position),
183 WTFMove(*size),
184 WTFMove(*baseFontSize),
185 WTFMove(*relativeFontSize),
186
187 WTFMove(*foregroundColor),
188 WTFMove(*backgroundColor),
189 WTFMove(*highlightColor),
190
191 WTFMove(*alignment),
192 WTFMove(*status),
193
194 };
195
196 if (!data.isValid())
197 return std::nullopt;
198
199 return data;
200}

240 void setContent(const String& content) { m_cueData.m_content = content; }
241
242 double line() const { return m_cueData.m_line; }
243 void setLine(double line) { m_cueData.m_line = line; }
244
245 double position() const { return m_cueData.m_position; }
