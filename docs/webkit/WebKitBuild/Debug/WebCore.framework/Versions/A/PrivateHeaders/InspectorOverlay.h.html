276 encoder << areas;
277 encoder << labels;

280template<class Decoder> std::optional<InspectorOverlay::Highlight::GridHighlightOverlay> InspectorOverlay::Highlight::GridHighlightOverlay::decode(Decoder& decoder)
281{
282 InspectorOverlay::Highlight::GridHighlightOverlay gridHighlightOverlay;
283 if (!decoder.decode(gridHighlightOverlay.color))
284 return { };
285 if (!decoder.decode(gridHighlightOverlay.gridLines))
286 return { };
287 if (!decoder.decode(gridHighlightOverlay.gaps))
288 return { };
289 if (!decoder.decode(gridHighlightOverlay.areas))
290 return { };
291 if (!decoder.decode(gridHighlightOverlay.labels))
292 return { };
293 return { gridHighlightOverlay };
294}

296template<class Encoder> void InspectorOverlay::Highlight::GridHighlightOverlay::Label::encode(Encoder& encoder) const
297{
298 encoder << text;
299 encoder << location;
300 encoder << backgroundColor;
301 encoder << static_cast<uint32_t>(arrowDirection);
302 encoder << static_cast<uint32_t>(arrowEdgePosition);
303}
304
305template<class Decoder> std::optional<InspectorOverlay::Highlight::GridHighlightOverlay::Label> InspectorOverlay::Highlight::GridHighlightOverlay::Label::decode(Decoder& decoder)
306{
307 InspectorOverlay::Highlight::GridHighlightOverlay::Label label;
308 if (!decoder.decode(label.text))
309 return { };
310 if (!decoder.decode(label.location))
311 return { };
312 if (!decoder.decode(label.backgroundColor))
313 return { };
314
315 uint32_t arrowDirection;
316 if (!decoder.decode(arrowDirection))
317 return { };
318 label.arrowDirection = (InspectorOverlay::LabelArrowDirection)arrowDirection;
319
320 uint32_t arrowEdgePosition;
321 if (!decoder.decode(arrowEdgePosition))
322 return { };
323 label.arrowEdgePosition = (InspectorOverlay::LabelArrowEdgePosition)arrowEdgePosition;
324
325 return { label };
326}
