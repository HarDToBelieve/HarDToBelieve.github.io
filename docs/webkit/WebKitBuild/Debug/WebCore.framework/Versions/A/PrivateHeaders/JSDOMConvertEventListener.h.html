27
28#include "IDLTypes.h"
29#include "JSDOMConvertBase.h"
30
31namespace WebCore {
32
33template<typename T> struct Converter<IDLEventListener<T>> : DefaultConverter<IDLEventListener<T>> {
34 using ReturnType = RefPtr<T>;
35
36 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, JSC::JSObject& thisObject)
37 {
38 auto scope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
39
40 auto listener = T::create(value, thisObject, false, currentWorld(lexicalGlobalObject));
41 if (!listener)
42 throwTypeError(&lexicalGlobalObject, scope);
43
44 return listener;
45 }
46};
47
48} // namespace WebCore
