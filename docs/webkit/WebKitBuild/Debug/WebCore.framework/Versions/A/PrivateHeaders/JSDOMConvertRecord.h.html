63template<typename K, typename V> struct Converter<IDLRecord<K, V>> : DefaultConverter<IDLRecord<K, V>> {
64 using ReturnType = typename IDLRecord<K, V>::ImplementationType;
65 using KeyType = typename K::ImplementationType;
66 using ValueType = typename V::ImplementationType;
67
68 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, JSDOMGlobalObject& globalObject)
69 {
70 return convertRecord<JSDOMGlobalObject&>(lexicalGlobalObject, value, globalObject);
71 }
72
73 static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
74 {
75 return convertRecord(lexicalGlobalObject, value);
76 }
77
78private:
79 template<class...Args>
80 static ReturnType convertRecord(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, Args ... args)
81 {
82 auto& vm = JSC::getVM(&lexicalGlobalObject);
83 auto scope = DECLARE_THROW_SCOPE(vm);
84
85 // 1. Let result be a new empty instance of record<K, V>.
86 // 2. If Type(O) is Undefined or Null, return result.
87 if (value.isUndefinedOrNull())
88 return { };
89
90 // 3. If Type(O) is not Object, throw a TypeError.
91 if (!value.isObject()) {
92 throwTypeError(&lexicalGlobalObject, scope);
93 return { };
94 }
95
96 JSC::JSObject* object = JSC::asObject(value);
97
98 ReturnType result;
99 HashMap<KeyType, size_t> resultMap;
100
101 // 4. Let keys be ? O.[[OwnPropertyKeys]]().
102 JSC::PropertyNameArray keys(vm, JSC::PropertyNameMode::StringsAndSymbols, JSC::PrivateSymbolMode::Exclude);
103 object->methodTable(vm)->getOwnPropertyNames(object, &lexicalGlobalObject, keys, JSC::DontEnumPropertiesMode::Include);
104 RETURN_IF_EXCEPTION(scope, { });
105
106 // 5. Repeat, for each element key of keys in List order:
107 for (auto& key : keys) {
108 // 1. Let desc be ? O.[[GetOwnProperty]](key).
109 JSC::PropertySlot slot(object, JSC::PropertySlot::InternalMethodType::GetOwnProperty);
110 bool hasProperty = object->methodTable(vm)->getOwnPropertySlot(object, &lexicalGlobalObject, key, slot);
111 RETURN_IF_EXCEPTION(scope, { });
112
113 // 2. If desc is not undefined and desc.[[Enumerable]] is true:
114
115 // It's necessary to filter enumerable here rather than using DontEnumPropertiesMode::Exclude,
116 // to prevent an observable extra [[GetOwnProperty]] operation in the case of ProxyObject records.
117 if (hasProperty && !(slot.attributes() & JSC::PropertyAttribute::DontEnum)) {
118 // 1. Let typedKey be key converted to an IDL value of type K.
119 auto typedKey = Detail::IdentifierConverter<K>::convert(lexicalGlobalObject, key);
120 RETURN_IF_EXCEPTION(scope, { });
121
122 // 2. Let value be ? Get(O, key).
123 JSC::JSValue subValue;
124 if (LIKELY(!slot.isTaintedByOpaqueObject()))
125 subValue = slot.getValue(&lexicalGlobalObject, key);
126 else
127 subValue = object->get(&lexicalGlobalObject, key);
128 RETURN_IF_EXCEPTION(scope, { });
129
130 // 3. Let typedValue be value converted to an IDL value of type V.
131 auto typedValue = Converter<V>::convert(lexicalGlobalObject, subValue, args...);
132 RETURN_IF_EXCEPTION(scope, { });
133
134 // 4. Set result[typedKey] to typedValue.
135 // Note: It's possible that typedKey is already in result if K is USVString and key contains unpaired surrogates.
136 if constexpr (std::is_same_v<K, IDLUSVString>) {
137 if (!typedKey.is8Bit()) {
138 auto addResult = resultMap.add(typedKey, result.size());
139 if (!addResult.isNewEntry) {
140 ASSERT(result[addResult.iterator->value].key == typedKey);
141 result[addResult.iterator->value].value = WTFMove(typedValue);
142 continue;
143 }
144 }
145 } else
146 UNUSED_VARIABLE(resultMap);
147
148 // 5. Otherwise, append to result a mapping (typedKey, typedValue).
149 result.append({ WTFMove(typedKey), WTFMove(typedValue) });
150 }
151 }
152
153 // 6. Return result.
154 return result;
155 }
156};

158template<typename K, typename V> struct JSConverter<IDLRecord<K, V>> {
159 static constexpr bool needsState = true;
160 static constexpr bool needsGlobalObject = true;
161
162 template<typename MapType>
163 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const MapType& map)
164 {
165 auto& vm = JSC::getVM(&lexicalGlobalObject);
166
167 // 1. Let result be ! ObjectCreate(%ObjectPrototype%).
168 auto result = constructEmptyObject(&lexicalGlobalObject, globalObject.objectPrototype());
169
170 // 2. Repeat, for each mapping (key, value) in D:
171 for (const auto& keyValuePair : map) {
172 // 1. Let esKey be key converted to an ECMAScript value.
173 // Note, this step is not required, as we need the key to be
174 // an Identifier, not a JSValue.
175
176 // 2. Let esValue be value converted to an ECMAScript value.
177 auto esValue = toJS<V>(lexicalGlobalObject, globalObject, keyValuePair.value);
178
179 // 3. Let created be ! CreateDataProperty(result, esKey, esValue).
180 bool created = result->putDirect(vm, JSC::Identifier::fromString(vm, keyValuePair.key), esValue);
181
182 // 4. Assert: created is true.
183 ASSERT_UNUSED(created, created);
184 }
185
186 // 3. Return result.
187 return result;
188 }
189};
