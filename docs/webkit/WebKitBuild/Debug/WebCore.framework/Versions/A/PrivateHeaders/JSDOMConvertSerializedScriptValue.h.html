33template<typename T> struct Converter<IDLSerializedScriptValue<T>> : DefaultConverter<IDLSerializedScriptValue<T>> {
34 static RefPtr<T> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
35 {
36 return T::create(lexicalGlobalObject, value);
37 }
38};
39
40template<typename T> struct JSConverter<IDLSerializedScriptValue<T>> {
41 static constexpr bool needsState = true;
42 static constexpr bool needsGlobalObject = true;
43
44 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, RefPtr<T> value)
45 {
46 return value ? value->deserialize(lexicalGlobalObject, &globalObject) : JSC::jsNull();
47 }
48};
