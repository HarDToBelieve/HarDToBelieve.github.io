26#include "JSDOMCastThisValue.h"
27#include "JSDOMExceptionHandling.h"

29namespace WebCore {
30
31template<typename JSClass>
32class IDLOperation {
33public:
34 using ClassParameter = JSClass*;
35 using Operation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*, ClassParameter);
36 using StaticOperation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*);
37
38 // FIXME: Remove this after FunctionCallResolveNode is fixed not to pass resolved scope as |this| value.
39 // https://bugs.webkit.org/show_bug.cgi?id=225397
40 static JSClass* cast(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame)
41 {
42 if constexpr (std::is_base_of_v<JSDOMGlobalObject, JSClass>)
43 return castThisValue<JSClass>(lexicalGlobalObject, callFrame.thisValue().toThis(&lexicalGlobalObject, JSC::ECMAMode::strict()));
44 else
45 return castThisValue<JSClass>(lexicalGlobalObject, callFrame.thisValue());
46 }
47
48 template<Operation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
49 static JSC::EncodedJSValue call(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char* operationName)
50 {
51 auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
52
53 auto* thisObject = cast(lexicalGlobalObject, callFrame);
54 if constexpr (shouldThrow != CastedThisErrorBehavior::Assert) {
55 if (UNLIKELY(!thisObject))
56 return throwThisTypeError(lexicalGlobalObject, throwScope, JSClass::info()->className, operationName);
57 } else
58 ASSERT(thisObject);
59
60 ASSERT_GC_OBJECT_INHERITS(thisObject, JSClass::info());
61
62 // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject and thisObject.
63 RELEASE_AND_RETURN(throwScope, (operation(&lexicalGlobalObject, &callFrame, thisObject)));
64 }
65
66 template<StaticOperation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
67 static JSC::EncodedJSValue callStatic(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char*)
68 {
69 // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject.
70 return operation(&lexicalGlobalObject, &callFrame);
71 }
72};
