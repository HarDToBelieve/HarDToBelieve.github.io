40class DeferredPromise : public DOMGuarded<JSC::JSPromise> {
41public:
42 enum class Mode {
43 ClearPromiseOnResolve,
44 RetainPromiseOnResolve

47 static RefPtr<DeferredPromise> create(JSDOMGlobalObject& globalObject, Mode mode = Mode::ClearPromiseOnResolve)
48 {
49 JSC::VM& vm = JSC::getVM(&globalObject);
50 auto* promise = JSC::JSPromise::create(vm, globalObject.promiseStructure());
51 ASSERT(promise);
52 return adoptRef(new DeferredPromise(globalObject, *promise, mode));
53 }
54
55 static Ref<DeferredPromise> create(JSDOMGlobalObject& globalObject, JSC::JSPromise& deferred, Mode mode = Mode::ClearPromiseOnResolve)
56 {
57 return adoptRef(*new DeferredPromise(globalObject, deferred, mode));
58 }

60 template<class IDLType>
61 void resolve(typename IDLType::ParameterType value)
62 {
63 if (shouldIgnoreRequestToFulfill())
64 return;
65
66 ASSERT(deferred());
67 ASSERT(globalObject());
68 JSC::JSGlobalObject* lexicalGlobalObject = globalObject();
69 JSC::JSLockHolder locker(lexicalGlobalObject);
70 resolve(*lexicalGlobalObject, toJS<IDLType>(*lexicalGlobalObject, *globalObject(), std::forward<typename IDLType::ParameterType>(value)));
71 }

142 template<typename Callback>
143 void resolveWithCallback(Callback callback)
144 {
145 if (shouldIgnoreRequestToFulfill())
146 return;
147
148 ASSERT(deferred());
149 ASSERT(globalObject());
150 auto* lexicalGlobalObject = globalObject();
151 JSC::VM& vm = lexicalGlobalObject->vm();
152 JSC::JSLockHolder locker(vm);
153 auto scope = DECLARE_CATCH_SCOPE(vm);
154 resolve(*lexicalGlobalObject, callback(*globalObject()));
155 if (UNLIKELY(scope.exception()))
156 handleUncaughtException(scope, *lexicalGlobalObject);
157 }
158
159 template<typename Callback>
160 void rejectWithCallback(Callback callback, RejectAsHandled rejectAsHandled = RejectAsHandled::No)
161 {
162 if (shouldIgnoreRequestToFulfill())
163 return;
164
165 ASSERT(deferred());
166 ASSERT(globalObject());
167 auto* lexicalGlobalObject = globalObject();
168 JSC::VM& vm = lexicalGlobalObject->vm();
169 JSC::JSLockHolder locker(vm);
170 auto scope = DECLARE_CATCH_SCOPE(vm);
171 reject(*lexicalGlobalObject, callback(*globalObject()), rejectAsHandled);
172 if (UNLIKELY(scope.exception()))
173 handleUncaughtException(scope, *lexicalGlobalObject);
174 }

176 JSC::JSValue promise() const;
177
178 void whenSettled(Function<void()>&&);
179
180private:
181 DeferredPromise(JSDOMGlobalObject& globalObject, JSC::JSPromise& deferred, Mode mode)
182 : DOMGuarded<JSC::JSPromise>(globalObject, deferred)
183 , m_mode(mode)
184 {
185 }
186
187 bool shouldIgnoreRequestToFulfill() const { return isEmpty(); }
188
189 JSC::JSPromise* deferred() const { return guarded(); }
190
191 enum class ResolveMode { Resolve, Reject, RejectAsHandled };
192 WEBCORE_EXPORT void callFunction(JSC::JSGlobalObject&, ResolveMode, JSC::JSValue resolution);
193
194 void resolve(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue resolution) { callFunction(lexicalGlobalObject, ResolveMode::Resolve, resolution); }
195 void reject(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue resolution, RejectAsHandled rejectAsHandled)
196 {
197 callFunction(lexicalGlobalObject, rejectAsHandled == RejectAsHandled::Yes ? ResolveMode::RejectAsHandled : ResolveMode::Reject, resolution);
198 }
199
200 bool handleTerminationExceptionIfNeeded(JSC::CatchScope&, JSDOMGlobalObject& lexicalGlobalObject);
201 void handleUncaughtException(JSC::CatchScope&, JSDOMGlobalObject& lexicalGlobalObject);
202
203 Mode m_mode;
204};

219 DOMPromiseDeferredBase(const DOMPromiseDeferredBase& other)
220 : m_promise(other.m_promise.copyRef())

264template<typename IDLType>
265class DOMPromiseDeferred : public DOMPromiseDeferredBase {
266public:
267 using DOMPromiseDeferredBase::DOMPromiseDeferredBase;
268 using DOMPromiseDeferredBase::operator=;
269 using DOMPromiseDeferredBase::promise;
270 using DOMPromiseDeferredBase::reject;
271
272 void resolve(typename IDLType::ParameterType value)
273 {
274 m_promise->resolve<IDLType>(std::forward<typename IDLType::ParameterType>(value));
275 }
276
277 void settle(ExceptionOr<typename IDLType::ParameterType>&& result)
278 {
279 if (result.hasException()) {
280 reject(result.releaseException());
281 return;
282 }
283 resolve(result.releaseReturnValue());
284 }

317using PromiseFunction = void(JSC::JSGlobalObject&, JSC::CallFrame&, Ref<DeferredPromise>&&);

319template<PromiseFunction promiseFunction>
320inline JSC::JSValue callPromiseFunction(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame)
321{
322 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
323 auto catchScope = DECLARE_CATCH_SCOPE(vm);
324
325 auto& globalObject = *JSC::jsSecureCast<JSDOMGlobalObject*>(vm, &lexicalGlobalObject);
326 auto* promise = JSC::JSPromise::create(vm, globalObject.promiseStructure());
327 ASSERT(promise);
328
329 promiseFunction(lexicalGlobalObject, callFrame, DeferredPromise::create(globalObject, *promise));
330
331 rejectPromiseWithExceptionIfAny(lexicalGlobalObject, globalObject, *promise, catchScope);
332 // FIXME: We could have error since any JS call can throw stack-overflow errors.
333 // https://bugs.webkit.org/show_bug.cgi?id=203402
334 RETURN_IF_EXCEPTION(catchScope, JSC::jsUndefined());
335 return promise;
336}
337
338template<typename PromiseFunctor>
339inline JSC::JSValue callPromiseFunction(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, PromiseFunctor functor)
340{
341 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
342 auto catchScope = DECLARE_CATCH_SCOPE(vm);
343
344 auto& globalObject = *JSC::jsSecureCast<JSDOMGlobalObject*>(vm, &lexicalGlobalObject);
345 auto* promise = JSC::JSPromise::create(vm, globalObject.promiseStructure());
346 ASSERT(promise);
347
348 functor(lexicalGlobalObject, callFrame, DeferredPromise::create(globalObject, *promise));
349
350 rejectPromiseWithExceptionIfAny(lexicalGlobalObject, globalObject, *promise, catchScope);
351 // FIXME: We could have error since any JS call can throw stack-overflow errors.
352 // https://bugs.webkit.org/show_bug.cgi?id=203402
353 RETURN_IF_EXCEPTION(catchScope, JSC::jsUndefined());
354 return promise;
355}

358template<BindingPromiseFunction bindingFunction>
359inline void bindingPromiseFunctionAdapter(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, Ref<DeferredPromise>&& promise)
360{
361 bindingFunction(&lexicalGlobalObject, &callFrame, WTFMove(promise));
362}
