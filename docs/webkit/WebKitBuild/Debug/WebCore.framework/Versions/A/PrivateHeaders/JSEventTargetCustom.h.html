33namespace WebCore {

35// Wrapper type for JSEventTarget's castedThis because JSDOMWindow and JSWorkerGlobalScope do not inherit JSEventTarget.
36class JSEventTargetWrapper {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 JSEventTargetWrapper(EventTarget& wrapped, JSC::JSObject& wrapper)
40 : m_wrapped(wrapped)
41 , m_wrapper(wrapper)
42 { }
43
44 EventTarget& wrapped() { return m_wrapped; }
45
46 operator JSC::JSObject&() { return m_wrapper; }
47
48private:
49 EventTarget& m_wrapped;
50 JSC::JSObject& m_wrapper;
51};
52
53std::unique_ptr<JSEventTargetWrapper> jsEventTargetCast(JSC::VM&, JSC::JSValue thisValue);
54
55template<> class IDLOperation<JSEventTarget> {
56public:
57 using ClassParameter = JSEventTargetWrapper*;
58 using Operation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*, ClassParameter);
59
60 template<Operation operation, CastedThisErrorBehavior = CastedThisErrorBehavior::Throw>
61 static JSC::EncodedJSValue call(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char* operationName)
62 {
63 auto& vm = JSC::getVM(&lexicalGlobalObject);
64 auto throwScope = DECLARE_THROW_SCOPE(vm);
65
66 auto thisValue = callFrame.thisValue().toThis(&lexicalGlobalObject, JSC::ECMAMode::strict());
67 auto thisObject = jsEventTargetCast(vm, thisValue.isUndefinedOrNull() ? JSC::JSValue(&lexicalGlobalObject) : thisValue);
68 if (UNLIKELY(!thisObject))
69 return throwThisTypeError(lexicalGlobalObject, throwScope, "EventTarget", operationName);
70
71 auto& wrapped = thisObject->wrapped();
72 if (is<DOMWindow>(wrapped)) {
73 auto& window = downcast<DOMWindow>(wrapped);
74 if (!window.frame() || !BindingSecurity::shouldAllowAccessToDOMWindow(&lexicalGlobalObject, window, ThrowSecurityError))
75 return JSC::JSValue::encode(JSC::jsUndefined());
76 }
77
78 RELEASE_AND_RETURN(throwScope, (operation(&lexicalGlobalObject, &callFrame, thisObject.get())));
79 }
80
81};
