35template<>
36struct InheritsTraits<WebCore::JSNode> {
37 static constexpr std::optional<JSTypeRange> typeRange { { static_cast<JSType>(WebCore::JSNodeType), static_cast<JSType>(WebCore::JSNodeType + WebCore::JSNodeTypeMask) } };
38 static_assert(std::numeric_limits<uint8_t>::max() == typeRange->last);
39 template<typename From>
40 static inline bool inherits(VM& vm, From* from)
41 {
42 return inheritsJSTypeImpl<WebCore::JSNode>(vm, from, *typeRange);
43 }
44};

51WEBCORE_EXPORT JSC::JSValue createWrapper(JSC::JSGlobalObject*, JSDOMGlobalObject*, Ref<Node>&&);
52WEBCORE_EXPORT JSC::JSObject* getOutOfLineCachedWrapper(JSDOMGlobalObject*, Node&);
53
54inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Node& node)
55{
56 if (LIKELY(globalObject->worldIsNormal())) {
57 if (auto* wrapper = node.wrapper())
58 return wrapper;
59 } else {
60 if (auto* wrapper = getOutOfLineCachedWrapper(globalObject, node))
61 return wrapper;
62 }
63
64 return createWrapper(lexicalGlobalObject, globalObject, node);
65}
66
67// In the C++ DOM, a node tree survives as long as there is a reference to its
68// root. In the JavaScript DOM, a node tree survives as long as there is a
69// reference to any node in the tree. To model the JavaScript DOM on top of
70// the C++ DOM, we ensure that the root of every tree has a JavaScript wrapper.
71void willCreatePossiblyOrphanedTreeByRemovalSlowCase(Node* root);
72inline void willCreatePossiblyOrphanedTreeByRemoval(Node* root)
73{
74 if (root->wrapper())
75 return;
76
77 if (!root->hasChildNodes())
78 return;
79
80 willCreatePossiblyOrphanedTreeByRemovalSlowCase(root);
81}
82
83inline void* root(Node* node)
84{
85 return node ? node->opaqueRoot() : nullptr;
86}
87
88inline void* root(Node& node)
89{
90 return root(&node);
91}
92
93ALWAYS_INLINE JSC::JSValue JSNode::nodeType(JSC::JSGlobalObject&) const
94{
95 return JSC::jsNumber(static_cast<uint8_t>(type()) & JSNodeTypeMask);
96}
