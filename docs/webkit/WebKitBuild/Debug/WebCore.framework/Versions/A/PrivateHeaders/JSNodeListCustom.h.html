32WEBCORE_EXPORT JSC::JSValue createWrapper(JSDOMGlobalObject&, Ref<NodeList>&&);
33
34ALWAYS_INLINE JSC::JSValue toJS(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, NodeList& nodeList)
35{
36 if (auto wrapper = getCachedWrapper(globalObject->world(), nodeList))
37 return wrapper;
38 return createWrapper(*globalObject, nodeList);
39}
