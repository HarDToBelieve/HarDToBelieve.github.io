26#ifndef KeypressCommand_h
27#define KeypressCommand_h
28
29#include <wtf/Assertions.h>
30#include <wtf/text/WTFString.h>
31
32#if PLATFORM(COCOA)
33
34namespace WebCore {
35
36struct KeypressCommand {
37 KeypressCommand() { }
38 explicit KeypressCommand(const String& commandName) : commandName(commandName) { ASSERT(isASCIILower(commandName[0U])); }
39 KeypressCommand(const String& commandName, const String& text) : commandName(commandName), text(text) { ASSERT(commandName == "insertText:"); }
40
41 String commandName; // Actually, a selector name - it may have a trailing colon, and a name that can be different from an editor command name.
42 String text;
43};
44
45} // namespace WebCore
46
47#endif
48
49#endif // KeypressCommand_h
