26#pragma once
27
28#include "ClipRect.h"
29
30namespace WebCore {
31
32class LayerFragment {
33public:
34 LayerFragment() = default;
35
36 void setRects(const LayoutRect& bounds, const ClipRect& background, const ClipRect& foreground, const std::optional<LayoutRect>& bbox)
37 {
38 layerBounds = bounds;
39 backgroundRect = background;
40 foregroundRect = foreground;
41 boundingBox = bbox;
42 }
43
44 void moveBy(const LayoutPoint& offset)
45 {
46 layerBounds.moveBy(offset);
47 backgroundRect.moveBy(offset);
48 foregroundRect.moveBy(offset);
49 paginationClip.moveBy(offset);
50 if (boundingBox)
51 boundingBox->moveBy(offset);
52 }
53
54 void intersect(const LayoutRect& rect)
55 {
56 backgroundRect.intersect(rect);
57 foregroundRect.intersect(rect);
58 if (boundingBox)
59 boundingBox->intersect(rect);
60 }
61
62 void intersect(const ClipRect& clipRect)
63 {
64 backgroundRect.intersect(clipRect);
65 foregroundRect.intersect(clipRect);
66 }
67
68 bool shouldPaintContent = false;
69 std::optional<LayoutRect> boundingBox;
70
71 LayoutRect layerBounds;
72 ClipRect backgroundRect;
73 ClipRect foregroundRect;
74
75 // Unique to paginated fragments. The physical translation to apply to shift the layer when painting/hit-testing.
76 LayoutSize paginationOffset;
77
78 // Also unique to paginated fragments. An additional clip that applies to the layer. It is in layer-local
79 // (physical) coordinates.
80 LayoutRect paginationClip;
81};
82
83typedef Vector<LayerFragment, 1> LayerFragments;
84
85} // namespace WebCore
