46class PathRun {
47public:
48 using PathVariant = Variant<
49#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
50 RunIteratorModernPath,
51#endif
52 RunIteratorLegacyPath
53 >;
54
55 PathRun(PathVariant&&);
56
57 bool isText() const;
58
59 FloatRect rect() const;
60
61 float logicalTop() const { return isHorizontal() ? rect().y() : rect().x(); }
62 float logicalBottom() const { return isHorizontal() ? rect().maxY() : rect().maxX(); }
63 float logicalLeft() const { return isHorizontal() ? rect().x() : rect().y(); }
64 float logicalRight() const { return isHorizontal() ? rect().maxX() : rect().maxY(); }
65 float logicalWidth() const { return isHorizontal() ? rect().width() : rect().height(); }
66 float logicalHeight() const { return isHorizontal() ? rect().height() : rect().width(); }
67
68 bool isHorizontal() const;
69 bool dirOverride() const;
70 bool isLineBreak() const;
71
72 unsigned minimumCaretOffset() const;
73 unsigned maximumCaretOffset() const;
74 unsigned leftmostCaretOffset() const { return isLeftToRightDirection() ? minimumCaretOffset() : maximumCaretOffset(); }
75 unsigned rightmostCaretOffset() const { return isLeftToRightDirection() ? maximumCaretOffset() : minimumCaretOffset(); }
76
77 unsigned char bidiLevel() const;
78 TextDirection direction() const { return bidiLevel() % 2 ? TextDirection::RTL : TextDirection::LTR; }
79 bool isLeftToRightDirection() const { return direction() == TextDirection::LTR; }
80
81 const RenderObject& renderer() const;
82
83 // For intermediate porting steps only.
84 LegacyInlineBox* legacyInlineBox() const;
85
86protected:
87 friend class RunIterator;
88 friend class TextRunIterator;
89
90 // To help with debugging.
91#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
92 const RunIteratorModernPath& modernPath() const;
93#endif
94 const RunIteratorLegacyPath& legacyPath() const;
95
96 PathVariant m_pathVariant;
97};

119class RunIterator {
120public:
121 RunIterator() : m_run(RunIteratorLegacyPath { nullptr, { } }) { };
122 RunIterator(PathRun::PathVariant&&);
123
124 explicit operator bool() const { return !atEnd(); }
125
126 bool operator==(const RunIterator&) const;
127 bool operator!=(const RunIterator& other) const { return !(*this == other); }
128
129 bool operator==(EndIterator) const { return atEnd(); }
130 bool operator!=(EndIterator) const { return !atEnd(); }
131
132 const PathRun& operator*() const { return m_run; }
133 const PathRun* operator->() const { return &m_run; }
134
135 bool atEnd() const;
136
137 RunIterator nextOnLine() const;
138 RunIterator previousOnLine() const;
139 RunIterator nextOnLineIgnoringLineBreak() const;
140 RunIterator previousOnLineIgnoringLineBreak() const;
141
142 RunIterator& traverseNextOnLine();
143 RunIterator& traversePreviousOnLine();
144 RunIterator& traverseNextOnLineIgnoringLineBreak();
145 RunIterator& traversePreviousOnLineIgnoringLineBreak();
146 RunIterator& traverseNextOnLineInLogicalOrder();
147 RunIterator& traversePreviousOnLineInLogicalOrder();
148
149 LineIterator line() const;
150
151protected:
152 PathRun m_run;
153};
