34namespace WebCore {
35namespace LayoutIntegration {
36
37class RunIteratorLegacyPath {
38public:
39 RunIteratorLegacyPath(const LegacyInlineBox* inlineBox, Vector<const LegacyInlineBox*>&& sortedInlineBoxes = { }, size_t sortedInlineBoxIndex = 0)
40 : m_inlineBox(inlineBox)
41 , m_logicalOrderCache(WTFMove(sortedInlineBoxes))
42 , m_logicalOrderCacheIndex(sortedInlineBoxIndex)
43 { }
44
45 enum class LogicalOrder { Start, End };
46 RunIteratorLegacyPath(const LegacyRootInlineBox& root, LogicalOrder order)
47 : m_logicalOrderCache(inlineBoxesInLogicalOrder(root))
48 {
49 if (!m_logicalOrderCache.isEmpty()) {
50 m_logicalOrderCacheIndex = order == LogicalOrder::Start ? 0 : m_logicalOrderCache.size() - 1;
51 m_inlineBox = m_logicalOrderCache[m_logicalOrderCacheIndex];
52 }
53 }
54
55 bool isText() const { return m_inlineBox->isInlineTextBox(); }
56
57 FloatRect rect() const { return m_inlineBox->frameRect(); }
58
59 bool isHorizontal() const { return m_inlineBox->isHorizontal(); }
60 bool dirOverride() const { return m_inlineBox->dirOverride(); }
61 bool isLineBreak() const { return m_inlineBox->isLineBreak(); }
62
63 unsigned minimumCaretOffset() const { return m_inlineBox->caretMinOffset(); }
64 unsigned maximumCaretOffset() const { return m_inlineBox->caretMaxOffset(); }
65
66 unsigned char bidiLevel() const { return m_inlineBox->bidiLevel(); }
67
68 bool hasHyphen() const { return inlineTextBox()->hasHyphen(); }
69 StringView text() const { return StringView(inlineTextBox()->renderer().text()).substring(inlineTextBox()->start(), inlineTextBox()->len()); }
70 unsigned start() const { return inlineTextBox()->start(); }
71 unsigned end() const { return inlineTextBox()->end(); }
72 unsigned length() const { return inlineTextBox()->len(); }
73
74 unsigned offsetForPosition(float x) const { return inlineTextBox()->offsetForPosition(x); }
75 float positionForOffset(unsigned offset) const { return inlineTextBox()->positionForOffset(offset); }
76
77 bool isSelectable(unsigned start, unsigned end) const { return inlineTextBox()->isSelectable(start, end); }
78 LayoutRect selectionRect(unsigned start, unsigned end) const { return inlineTextBox()->localSelectionRect(start, end); }
79
80 const RenderObject& renderer() const
81 {
82 return m_inlineBox->renderer();
83 }
84
85 void traverseNextTextRun() { m_inlineBox = inlineTextBox()->nextTextBox(); }
86 void traverseNextTextRunInTextOrder()
87 {
88 if (!m_logicalOrderCache.isEmpty()) {
89 traverseNextInlineBoxInCacheOrder();
90 ASSERT(!m_inlineBox || is<LegacyInlineTextBox>(m_inlineBox));
91 return;
92 }
93 traverseNextTextRun();
94 }
95
96 void traverseNextOnLine()
97 {
98 m_inlineBox = m_inlineBox->nextLeafOnLine();
99 }
100
101 void traversePreviousOnLine()
102 {
103 m_inlineBox = m_inlineBox->previousLeafOnLine();
104 }
105
106 void traverseNextOnLineInLogicalOrder()
107 {
108 initializeLogicalOrderCacheForLine();
109 traverseNextInlineBoxInCacheOrder();
110 }
111
112 void traversePreviousOnLineInLogicalOrder()
113 {
114 initializeLogicalOrderCacheForLine();
115 traversePreviousInlineBoxInCacheOrder();
116 }
117
118 bool operator==(const RunIteratorLegacyPath& other) const { return m_inlineBox == other.m_inlineBox; }
119
120 bool atEnd() const { return !m_inlineBox; }
121
122 LegacyInlineBox* legacyInlineBox() const { return const_cast<LegacyInlineBox*>(m_inlineBox); }
123 const LegacyRootInlineBox& rootInlineBox() const { return m_inlineBox->root(); }
124
125private:
126 const LegacyInlineTextBox* inlineTextBox() const { return downcast<LegacyInlineTextBox>(m_inlineBox); }
127
128 static Vector<const LegacyInlineBox*> inlineBoxesInLogicalOrder(const LegacyRootInlineBox& root)
129 {
130 Vector<LegacyInlineBox*> inlineBoxes;
131 root.collectLeafBoxesInLogicalOrder(inlineBoxes);
132 return reinterpret_cast<Vector<const LegacyInlineBox*>&>(inlineBoxes);
133 }
134 void initializeLogicalOrderCacheForLine()
135 {
136 if (!m_inlineBox || !m_logicalOrderCache.isEmpty())
137 return;
138 m_logicalOrderCache = inlineBoxesInLogicalOrder(m_inlineBox->root());
139 for (m_logicalOrderCacheIndex = 0; m_logicalOrderCacheIndex < m_logicalOrderCache.size(); ++m_logicalOrderCacheIndex) {
140 if (m_logicalOrderCache[m_logicalOrderCacheIndex] == m_inlineBox)
141 return;
142 }
143 ASSERT_NOT_REACHED();
144 }
145
146 void traverseNextInlineBoxInCacheOrder();
147 void traversePreviousInlineBoxInCacheOrder();
148
149 const LegacyInlineBox* m_inlineBox { nullptr };
150 RefCountedArray<const LegacyInlineBox*> m_logicalOrderCache;
151 size_t m_logicalOrderCacheIndex { 0 };
152};
153
154
155inline void RunIteratorLegacyPath::traverseNextInlineBoxInCacheOrder()
156{
157 ASSERT(!m_logicalOrderCache.isEmpty());
158 ++m_logicalOrderCacheIndex;
159 m_inlineBox = m_logicalOrderCacheIndex < m_logicalOrderCache.size() ? m_logicalOrderCache[m_logicalOrderCacheIndex] : nullptr;
160}
161
162inline void RunIteratorLegacyPath::traversePreviousInlineBoxInCacheOrder()
163{
164 ASSERT(!m_logicalOrderCache.isEmpty());
165 if (!m_logicalOrderCacheIndex) {
166 m_inlineBox = nullptr;
167 return;
168 }
169 m_inlineBox = m_logicalOrderCache[--m_logicalOrderCacheIndex];
170}
