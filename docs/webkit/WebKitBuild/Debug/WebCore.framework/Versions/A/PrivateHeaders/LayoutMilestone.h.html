30namespace WebCore {
31
32// FIXME: Some of these milestones are about layout, and others are about painting.
33// We should either re-name them to something more generic, or split them into
34// two enums -- one for painting and one for layout.
35enum LayoutMilestone {
36 DidFirstLayout = 1 << 0,
37 DidFirstVisuallyNonEmptyLayout = 1 << 1,
38 DidHitRelevantRepaintedObjectsAreaThreshold = 1 << 2,
39 DidFirstFlushForHeaderLayer = 1 << 3,
40 DidFirstLayoutAfterSuppressedIncrementalRendering = 1 << 4,
41 DidFirstPaintAfterSuppressedIncrementalRendering = 1 << 5,
42 ReachedSessionRestorationRenderTreeSizeThreshold = 1 << 6, // FIXME: only implemented by WK2 currently.
43 DidRenderSignificantAmountOfText = 1 << 7,
44 DidFirstMeaningfulPaint = 1 << 8,
45};
46
47} // namespace WebCore
48
49namespace WTF {
50
51template<> struct EnumTraits<WebCore::LayoutMilestone> {
52 using values = EnumValues<
53 WebCore::LayoutMilestone,
54 WebCore::LayoutMilestone::DidFirstLayout,
55 WebCore::LayoutMilestone::DidFirstVisuallyNonEmptyLayout,
56 WebCore::LayoutMilestone::DidHitRelevantRepaintedObjectsAreaThreshold,
57 WebCore::LayoutMilestone::DidFirstFlushForHeaderLayer,
58 WebCore::LayoutMilestone::DidFirstLayoutAfterSuppressedIncrementalRendering,
59 WebCore::LayoutMilestone::DidFirstPaintAfterSuppressedIncrementalRendering,
60 WebCore::LayoutMilestone::ReachedSessionRestorationRenderTreeSizeThreshold,
61 WebCore::LayoutMilestone::DidRenderSignificantAmountOfText,
62 WebCore::LayoutMilestone::DidFirstMeaningfulPaint
63 >;
64};

66} // namespace WTF
