33namespace WebCore {
34
35class LegacyCDMSessionClient {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 virtual ~LegacyCDMSessionClient() = default;
39 virtual void sendMessage(Uint8Array*, String destinationURL) = 0;
40
41 enum : uint8_t {
42 MediaKeyErrorUnknown = 1,
43 MediaKeyErrorClient,
44 MediaKeyErrorService,
45 MediaKeyErrorOutput,
46 MediaKeyErrorHardwareChange,
47 MediaKeyErrorDomain,
48 };
49 typedef unsigned short MediaKeyErrorCode;
50 virtual void sendError(MediaKeyErrorCode, uint32_t systemCode) = 0;
51
52 virtual String mediaKeysStorageDirectory() const = 0;
53};
54
55enum LegacyCDMSessionType {
56 CDMSessionTypeUnknown,
57 CDMSessionTypeClearKey,
58 CDMSessionTypeAVFoundationObjC,
59 CDMSessionTypeAVStreamSession,
60 CDMSessionTypeAVContentKeySession,
61 CDMSessionTypeRemote,
62};
63
64class WEBCORE_EXPORT LegacyCDMSession {
65public:
66 virtual ~LegacyCDMSession() = default;
67
68 virtual LegacyCDMSessionType type() { return CDMSessionTypeUnknown; }
69 virtual void setClient(LegacyCDMSessionClient*) = 0;
70 virtual const String& sessionId() const = 0;
71 virtual RefPtr<Uint8Array> generateKeyRequest(const String& mimeType, Uint8Array* initData, String& destinationURL, unsigned short& errorCode, uint32_t& systemCode) = 0;
72 virtual void releaseKeys() = 0;
73 virtual bool update(Uint8Array*, RefPtr<Uint8Array>& nextMessage, unsigned short& errorCode, uint32_t& systemCode) = 0;
74 virtual RefPtr<ArrayBuffer> cachedKeyForKeyID(const String&) const { return nullptr; }
75};

77}
