30namespace WebCore {

32class LegacyInlineElementBox : public LegacyInlineBox {
33 WTF_MAKE_ISO_ALLOCATED(LegacyInlineElementBox);
34public:
35 explicit LegacyInlineElementBox(RenderBoxModelObject& renderer)
36 : LegacyInlineBox(renderer)
37 {
38 }
39
40 LegacyInlineElementBox(RenderObject& renderer, FloatPoint topLeft, float logicalWidth, bool firstLine, bool constructed, bool dirty, bool extracted, bool isHorizontal, LegacyInlineBox* next, LegacyInlineBox* prev, LegacyInlineFlowBox* parent)
41 : LegacyInlineBox(renderer, topLeft, logicalWidth, firstLine, constructed, dirty, extracted, isHorizontal, next, prev, parent)
42 {
43 }
44
45 RenderBoxModelObject& renderer() const { return downcast<RenderBoxModelObject>(LegacyInlineBox::renderer()); }
46
47 void deleteLine() final;
48 void extractLine() final;
49 void attachLine() final;
50
51 void paint(PaintInfo&, const LayoutPoint&, LayoutUnit lineTop, LayoutUnit lineBottom) override;
52 bool nodeAtPoint(const HitTestRequest&, HitTestResult&, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, LayoutUnit lineTop, LayoutUnit lineBottom, HitTestAction) override;
53
54private:
55 bool isInlineElementBox() const final { return true; }
56};

58} // namespace WebCore
