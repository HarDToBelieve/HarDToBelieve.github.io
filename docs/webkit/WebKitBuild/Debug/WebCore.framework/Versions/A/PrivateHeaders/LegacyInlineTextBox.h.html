41const unsigned short cNoTruncation = USHRT_MAX;
42const unsigned short cFullTruncation = USHRT_MAX - 1;

44class LegacyInlineTextBox : public LegacyInlineBox {
45 WTF_MAKE_ISO_ALLOCATED(LegacyInlineTextBox);
46public:
47 explicit LegacyInlineTextBox(RenderText& renderer)
48 : LegacyInlineBox(renderer)
49 {
50 setBehavesLikeText(true);
51 }
52
53 virtual ~LegacyInlineTextBox();
54
55 RenderText& renderer() const { return downcast<RenderText>(LegacyInlineBox::renderer()); }
56 const RenderStyle& lineStyle() const { return isFirstLine() ? renderer().firstLineStyle() : renderer().style(); }
57
58 LegacyInlineTextBox* prevTextBox() const { return m_prevTextBox; }
59 LegacyInlineTextBox* nextTextBox() const { return m_nextTextBox; }
60 void setNextTextBox(LegacyInlineTextBox* n) { m_nextTextBox = n; }
61 void setPreviousTextBox(LegacyInlineTextBox* p) { m_prevTextBox = p; }
62
63 bool hasTextContent() const;
64
65 // These functions do not account for combined text. For combined text this box will always have len() == 1
66 // regardless of whether the resulting composition is the empty string. Use hasTextContent() if you want to
67 // know whether this box has text content.
68 //
69 // FIXME: These accessors should ASSERT(!isDirty()). See https://bugs.webkit.org/show_bug.cgi?id=97264
70 // Note len() == 1 for combined text regardless of whether the composition is empty. Use hasTextContent() to
71 unsigned start() const { return m_start; }
72 unsigned end() const { return m_start + m_len; }
73 unsigned len() const { return m_len; }
74
75 void setStart(unsigned start) { m_start = start; }
76 void setLen(unsigned len) { m_len = len; }
77
78 void offsetRun(int d) { ASSERT(!isDirty()); ASSERT(d > 0 || m_start >= static_cast<unsigned>(-d)); m_start += d; }
79
80 unsigned short truncation() const { return m_truncation; }
81
82 TextBoxSelectableRange selectableRange() const;
83
84 void markDirty(bool dirty = true) final;
85
86 using LegacyInlineBox::hasHyphen;
87 using LegacyInlineBox::setHasHyphen;
88 using LegacyInlineBox::canHaveLeftExpansion;
89 using LegacyInlineBox::setCanHaveLeftExpansion;
90 using LegacyInlineBox::canHaveRightExpansion;
91 using LegacyInlineBox::setCanHaveRightExpansion;
92 using LegacyInlineBox::forceRightExpansion;
93 using LegacyInlineBox::setForceRightExpansion;
94 using LegacyInlineBox::forceLeftExpansion;
95 using LegacyInlineBox::setForceLeftExpansion;
96
97 static inline bool compareByStart(const LegacyInlineTextBox* first, const LegacyInlineTextBox* second) { return first->start() < second->start(); }
98
99 LayoutUnit baselinePosition(FontBaseline) const final;
100 LayoutUnit lineHeight() const final;
101
102 std::optional<bool> emphasisMarkExistsAndIsAbove(const RenderStyle&) const;
103
104 LayoutRect logicalOverflowRect() const;
105 void setLogicalOverflowRect(const LayoutRect&);
106 LayoutUnit logicalTopVisualOverflow() const { return logicalOverflowRect().y(); }
107 LayoutUnit logicalBottomVisualOverflow() const { return logicalOverflowRect().maxY(); }
108 LayoutUnit logicalLeftVisualOverflow() const { return logicalOverflowRect().x(); }
109 LayoutUnit logicalRightVisualOverflow() const { return logicalOverflowRect().maxX(); }
110
111 virtual void dirtyOwnLineBoxes() { dirtyLineBoxes(); }
112
113#if ENABLE(TREE_DEBUGGING)
114 void outputLineBox(WTF::TextStream&, bool mark, int depth) const final;
115 const char* boxName() const final;

179 void paintMarkedTextForeground(PaintInfo&, const FloatRect& boxRect, const StyledMarkedText&);

197 unsigned m_start { 0 };

200 // Where to truncate when text overflow is applied. We use special constants to

202 unsigned short m_truncation { cNoTruncation };
