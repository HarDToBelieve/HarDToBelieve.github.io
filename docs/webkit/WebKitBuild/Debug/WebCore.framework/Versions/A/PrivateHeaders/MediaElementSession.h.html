114 // Restrictions to modify default behaviors.
115 enum BehaviorRestrictionFlags : unsigned {
116 NoRestrictions = 0,
117 RequireUserGestureForLoad = 1 << 0,
118 RequireUserGestureForVideoRateChange = 1 << 1,
119 RequireUserGestureForFullscreen = 1 << 2,
120 RequirePageConsentToLoadMedia = 1 << 3,
121 RequirePageConsentToResumeMedia = 1 << 4,
122 RequireUserGestureForAudioRateChange = 1 << 5,
123 RequireUserGestureToShowPlaybackTargetPicker = 1 << 6,
124 WirelessVideoPlaybackDisabled = 1 << 7,
125 RequireUserGestureToAutoplayToExternalDevice = 1 << 8,
126 AutoPreloadingNotPermitted = 1 << 10,
127 InvisibleAutoplayNotPermitted = 1 << 11,
128 OverrideUserGestureRequirementForMainContent = 1 << 12,
129 RequireUserGestureToControlControlsManager = 1 << 13,
130 RequirePlaybackToControlControlsManager = 1 << 14,
131 RequireUserGestureForVideoDueToLowPowerMode = 1 << 15,
132 AllRestrictions = ~NoRestrictions,
133 };
134 typedef unsigned BehaviorRestrictions;
