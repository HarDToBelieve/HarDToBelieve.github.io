93struct MediaEngineSupportParameters {
94 ContentType type;
95 URL url;
96 bool isMediaSource { false };
97 bool isMediaStream { false };
98 Vector<ContentType> contentTypesRequiringHardwareSupport;
99
100 template<class Encoder>
101 void encode(Encoder& encoder) const
102 {
103 encoder << type;
104 encoder << url;
105 encoder << isMediaSource;
106 encoder << isMediaStream;
107 encoder << contentTypesRequiringHardwareSupport;
108 }
109
110 template <class Decoder>
111 static std::optional<MediaEngineSupportParameters> decode(Decoder& decoder)
112 {
113 std::optional<ContentType> type;
114 decoder >> type;
115 if (!type)
116 return std::nullopt;
117
118 std::optional<URL> url;
119 decoder >> url;
120 if (!url)
121 return std::nullopt;
122
123 std::optional<bool> isMediaSource;
124 decoder >> isMediaSource;
125 if (!isMediaSource)
126 return std::nullopt;
127
128 std::optional<bool> isMediaStream;
129 decoder >> isMediaStream;
130 if (!isMediaStream)
131 return std::nullopt;
132
133 std::optional<Vector<ContentType>> typesRequiringHardware;
134 decoder >> typesRequiringHardware;
135 if (!typesRequiringHardware)
136 return std::nullopt;
137
138 return {{ WTFMove(*type), WTFMove(*url), *isMediaSource, *isMediaStream, *typesRequiringHardware }};
139 }
140};
