60class WEBCORE_EXPORT MediaSample : public ThreadSafeRefCounted<MediaSample> {
61public:
62 virtual ~MediaSample() = default;
63
64 virtual MediaTime presentationTime() const = 0;
65 virtual MediaTime decodeTime() const = 0;
66 virtual MediaTime duration() const = 0;
67 virtual AtomString trackID() const = 0;
68 virtual void setTrackID(const String&) = 0;
69 virtual size_t sizeInBytes() const = 0;
70 virtual FloatSize presentationSize() const = 0;
71 virtual void offsetTimestampsBy(const MediaTime&) = 0;
72 virtual void setTimestamps(const MediaTime&, const MediaTime&) = 0;
73 virtual bool isDivisable() const = 0;
74 enum DivideFlags { BeforePresentationTime, AfterPresentationTime };
75 enum class UseEndTime : bool {
76 DoNotUse,
77 Use,
78 };
79 virtual std::pair<RefPtr<MediaSample>, RefPtr<MediaSample>> divide(const MediaTime& presentationTime, UseEndTime = UseEndTime::DoNotUse) = 0;
80 virtual Ref<MediaSample> createNonDisplayingCopy() const = 0;
81
82 virtual RefPtr<JSC::Uint8ClampedArray> getRGBAImageData() const { return nullptr; }
83
84 enum SampleFlags {
85 None = 0,
86 IsSync = 1 << 0,
87 IsNonDisplaying = 1 << 1,
88 HasAlpha = 1 << 2,
89 HasSyncInfo = 1 << 3,
90 };
91 virtual SampleFlags flags() const = 0;
92 virtual PlatformSample platformSample() = 0;
93
94 struct ByteRange {
95 size_t byteOffset { 0 };
96 size_t byteLength { 0 };
97 };
98 virtual std::optional<ByteRange> byteRange() const = 0;
99
100 enum class VideoRotation {
101 None = 0,
102 UpsideDown = 180,
103 Right = 90,
104 Left = 270,
105 };
106 virtual VideoRotation videoRotation() const { return VideoRotation::None; }
107 virtual bool videoMirrored() const { return false; }
108 virtual uint32_t videoPixelFormat() const { return 0; }
109
110 bool isSync() const { return flags() & IsSync; }
111 bool isNonDisplaying() const { return flags() & IsNonDisplaying; }
112 bool hasAlpha() const { return flags() & HasAlpha; }
113 bool hasSyncInfo() const { return flags() & HasSyncInfo; }
114
115 virtual void dump(PrintStream&) const = 0;
116 String toJSONString() const
117 {
118 auto object = JSON::Object::create();
119
120 object->setObject("pts"_s, presentationTime().toJSONObject());
121 object->setObject("dts"_s, decodeTime().toJSONObject());
122 object->setObject("duration"_s, duration().toJSONObject());
123 object->setInteger("flags"_s, static_cast<unsigned>(flags()));
124 object->setObject("presentationSize"_s, presentationSize().toJSONObject());
125
126 return object->toJSONString();
127 }
128};
