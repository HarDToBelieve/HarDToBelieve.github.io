38struct MockMicrophoneProperties {
39 template<class Encoder>
40 void encode(Encoder& encoder) const
41 {
42 encoder << static_cast<int32_t>(defaultSampleRate);
43 }
44
45 template <class Decoder>
46 static std::optional<MockMicrophoneProperties> decode(Decoder& decoder)
47 {
48 std::optional<int32_t> defaultSampleRate;
49 decoder >> defaultSampleRate;
50 if (!defaultSampleRate)
51 return std::nullopt;
52 return MockMicrophoneProperties { *defaultSampleRate };
53 }
54
55 int defaultSampleRate { 44100 };
56};
57
58struct MockSpeakerProperties {
59 template<class Encoder>
60 void encode(Encoder& encoder) const
61 {
62 encoder << relatedMicrophoneId << static_cast<int32_t>(defaultSampleRate);
63 }
64
65 template <class Decoder>
66 static std::optional<MockSpeakerProperties> decode(Decoder& decoder)
67 {
68 std::optional<int32_t> defaultSampleRate;
69 decoder >> defaultSampleRate;
70 if (!defaultSampleRate)
71 return std::nullopt;
72
73 std::optional<String> relatedMicrophoneId;
74 decoder >> relatedMicrophoneId;
75 if (!relatedMicrophoneId)
76 return std::nullopt;
77
78 return MockSpeakerProperties { WTFMove(*relatedMicrophoneId), *defaultSampleRate };
79 }
80
81 String relatedMicrophoneId;
82 int defaultSampleRate { 44100 };
83};

106 if (!facingMode)
107 return std::nullopt;
108
109 std::optional<Vector<VideoPresetData>> presets;
110 decoder >> presets;
111 if (!presets)
112 return std::nullopt;
113
114 std::optional<Color> fillColor;
115 decoder >> fillColor;
116 if (!fillColor)
117 return std::nullopt;
118
119 return MockCameraProperties { *defaultFrameRate, *facingMode, WTFMove(*presets), *fillColor };
120 }
121
122 double defaultFrameRate { 30 };
123 RealtimeMediaSourceSettings::VideoFacingMode facingMode { RealtimeMediaSourceSettings::VideoFacingMode::User };
124 Vector<VideoPresetData> presets { { { 640, 480 }, { { 30, 30}, { 15, 15 } } } };
125 Color fillColor { Color::black };

162struct MockMediaDevice {
163 bool isMicrophone() const { return WTF::holds_alternative<MockMicrophoneProperties>(properties); }
164 bool isSpeaker() const { return WTF::holds_alternative<MockSpeakerProperties>(properties); }
165 bool isCamera() const { return WTF::holds_alternative<MockCameraProperties>(properties); }
166 bool isDisplay() const { return WTF::holds_alternative<MockDisplayProperties>(properties); }
167
168 CaptureDevice captureDevice() const
169 {
170 if (isMicrophone())
171 return CaptureDevice { persistentId, CaptureDevice::DeviceType::Microphone, label, persistentId };
172 if (isSpeaker())
173 return CaptureDevice { persistentId, CaptureDevice::DeviceType::Speaker, label, speakerProperties()->relatedMicrophoneId };
174 if (isCamera())
175 return CaptureDevice { persistentId, CaptureDevice::DeviceType::Camera, label, persistentId };
176
177 ASSERT(isDisplay());
178 return CaptureDevice { persistentId, CaptureDevice::DeviceType::Screen, label, persistentId };
179 }
180
181 CaptureDevice::DeviceType type() const
182 {
183 if (isMicrophone())
184 return CaptureDevice::DeviceType::Microphone;
185 if (isSpeaker())
186 return CaptureDevice::DeviceType::Speaker;
187 if (isCamera())
188 return CaptureDevice::DeviceType::Camera;
189
190 ASSERT(isDisplay());
191 return WTF::get<MockDisplayProperties>(properties).type;
192 }
193
194 const MockSpeakerProperties* speakerProperties() const
195 {
196 return isSpeaker() ? &WTF::get<MockSpeakerProperties>(properties) : nullptr;
197 }
198
199 template<class Encoder>
200 void encode(Encoder& encoder) const
201 {
202 encoder << persistentId;
203 encoder << label;
204 switchOn(properties, [&](const MockMicrophoneProperties& properties) {
205 encoder << (uint8_t)1;
206 encoder << properties;
207 }, [&](const MockSpeakerProperties& properties) {
208 encoder << (uint8_t)2;
209 encoder << properties;
210 }, [&](const MockCameraProperties& properties) {
211 encoder << (uint8_t)3;
212 encoder << properties;
213 }, [&](const MockDisplayProperties& properties) {
214 encoder << (uint8_t)4;
215 encoder << properties;
216 });
217 }
218
219 template <typename Properties, typename Decoder>
220 static std::optional<MockMediaDevice> decodeMockMediaDevice(Decoder& decoder, String&& persistentId, String&& label)
221 {
222 std::optional<Properties> properties;
223 decoder >> properties;
224 if (!properties)
225 return std::nullopt;
226 return MockMediaDevice { WTFMove(persistentId), WTFMove(label), WTFMove(*properties) };
227 }
228
229 template <class Decoder>
230 static std::optional<MockMediaDevice> decode(Decoder& decoder)
231 {
232 std::optional<String> persistentId;
233 decoder >> persistentId;
234 if (!persistentId)
235 return std::nullopt;
236
237 std::optional<String> label;
238 decoder >> label;
239 if (!label)
240 return std::nullopt;
241
242 std::optional<uint8_t> index;
243 decoder >> index;
244 if (!index)
245 return std::nullopt;
246
247 switch (*index) {
248 case 1:
249 return decodeMockMediaDevice<MockMicrophoneProperties>(decoder, WTFMove(*persistentId), WTFMove(*label));
250 case 2:
251 return decodeMockMediaDevice<MockSpeakerProperties>(decoder, WTFMove(*persistentId), WTFMove(*label));
252 case 3:
253 return decodeMockMediaDevice<MockCameraProperties>(decoder, WTFMove(*persistentId), WTFMove(*label));
254 case 4:
255 return decodeMockMediaDevice<MockDisplayProperties>(decoder, WTFMove(*persistentId), WTFMove(*label));
256 }
257 return std::nullopt;
258 }
259
260 String persistentId;
261 String label;
262 Variant<MockMicrophoneProperties, MockSpeakerProperties, MockCameraProperties, MockDisplayProperties> properties;
263};
