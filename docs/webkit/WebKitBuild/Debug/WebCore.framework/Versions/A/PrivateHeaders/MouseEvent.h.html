48 const IntPoint& screenLocation, const IntPoint& windowLocation, const IntPoint& movementDelta, OptionSet<Modifier>, short button, unsigned short buttons,
49 EventTarget* relatedTarget, double force, unsigned short syntheticClickType, IsSimulated = IsSimulated::No, IsTrusted = IsTrusted::Yes);

51 WEBCORE_EXPORT static Ref<MouseEvent> create(const AtomString& eventType, RefPtr<WindowProxy>&&, const PlatformMouseEvent&, int detail, Node* relatedTarget);
52
53 static Ref<MouseEvent> create(const AtomString& eventType, CanBubble, IsCancelable, IsComposed, RefPtr<WindowProxy>&&, int detail,
54 int screenX, int screenY, int clientX, int clientY, OptionSet<Modifier>, short button, unsigned short buttons,
55 unsigned short syntheticClickType, EventTarget* relatedTarget);

57 static Ref<MouseEvent> createForBindings() { return adoptRef(*new MouseEvent); }

59 static Ref<MouseEvent> create(const AtomString& eventType, const MouseEventInit&);

64
65 virtual ~MouseEvent();
66
67 WEBCORE_EXPORT void initMouseEvent(const AtomString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&&,
68 int detail, int screenX, int screenY, int clientX, int clientY, bool ctrlKey, bool altKey, bool shiftKey, bool metaKey,
69 short button, EventTarget* relatedTarget);

73 short button, JSC::JSValue relatedTarget);
74
75 short button() const { return m_button; }
76 unsigned short buttons() const { return m_buttons; }
77 unsigned short syntheticClickType() const { return m_syntheticClickType; }
78 bool buttonDown() const { return m_buttonDown; }
79 EventTarget* relatedTarget() const final { return m_relatedTarget.get(); }

88 int which() const final;

90protected:
91 MouseEvent(const AtomString& type, CanBubble, IsCancelable, IsComposed, MonotonicTime timestamp, RefPtr<WindowProxy>&&, int detail,

93 EventTarget* relatedTarget, double force, unsigned short syntheticClickType, IsSimulated, IsTrusted);
94
95 MouseEvent(const AtomString& type, CanBubble, IsCancelable, IsComposed, RefPtr<WindowProxy>&&, int detail,

97 unsigned short syntheticClickType, EventTarget* relatedTarget);

99 MouseEvent(const AtomString& type, const MouseEventInit&);

101 MouseEvent();
102
103private:

113 RefPtr<EventTarget> m_relatedTarget;
114 double m_force { 0 };
115};
