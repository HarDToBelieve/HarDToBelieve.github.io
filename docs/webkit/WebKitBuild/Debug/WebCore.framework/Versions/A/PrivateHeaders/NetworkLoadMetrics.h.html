62class NetworkLoadMetricsWithoutNonTimingData {
63 WTF_MAKE_FAST_ALLOCATED(NetworkLoadMetricsWithoutNonTimingData);
64public:
65 bool isComplete() const { return complete; }
66 void markComplete() { complete = true; }
67
68 // https://www.w3.org/TR/resource-timing-2/#attribute-descriptions
69 MonotonicTime redirectStart;
70 MonotonicTime fetchStart;
71 MonotonicTime domainLookupStart;
72 MonotonicTime domainLookupEnd;
73 MonotonicTime connectStart;
74 MonotonicTime secureConnectionStart;
75 MonotonicTime connectEnd;
76 MonotonicTime requestStart;
77 MonotonicTime responseStart;
78 MonotonicTime responseEnd;
79
80 // ALPN Protocol ID: https://w3c.github.io/resource-timing/#bib-RFC7301
81 String protocol;
82
83 uint16_t redirectCount { 0 };
84
85 // FIXME: These could all be made bit fields.
86 bool complete { false };
87 bool cellular { false };
88 bool expensive { false };
89 bool constrained { false };
90 bool multipath { false };
91 bool isReusedConnection { false };
92 bool failsTAOCheck { false };
93 bool hasCrossOriginRedirect { false };
94};

96class NetworkLoadMetrics : public NetworkLoadMetricsWithoutNonTimingData {
97public:
98 NetworkLoadMetrics()
99 : NetworkLoadMetricsWithoutNonTimingData()
100 {
101 }
102
103 NetworkLoadMetrics isolatedCopy() const
104 {
105 NetworkLoadMetrics copy;
106
107 copy.redirectStart = redirectStart;
108 copy.fetchStart = fetchStart;
109
110 copy.domainLookupStart = domainLookupStart;
111 copy.domainLookupEnd = domainLookupEnd;
112 copy.connectStart = connectStart;
113 copy.secureConnectionStart = secureConnectionStart;
114 copy.connectEnd = connectEnd;
115 copy.requestStart = requestStart;
116 copy.responseStart = responseStart;
117 copy.responseEnd = responseEnd;
118 copy.complete = complete;
119 copy.protocol = protocol.isolatedCopy();
120 copy.redirectCount = redirectCount;
121 copy.cellular = cellular;
122 copy.expensive = expensive;
123 copy.constrained = constrained;
124 copy.multipath = multipath;
125 copy.isReusedConnection = isReusedConnection;
126 copy.failsTAOCheck = failsTAOCheck;
127 copy.hasCrossOriginRedirect = hasCrossOriginRedirect;
128
129 copy.remoteAddress = remoteAddress.isolatedCopy();
130 copy.connectionIdentifier = connectionIdentifier.isolatedCopy();
131 copy.tlsProtocol = tlsProtocol.isolatedCopy();
132 copy.tlsCipher = tlsCipher.isolatedCopy();
133 copy.priority = priority;
134 copy.privacyStance = privacyStance;
135 copy.requestHeaders = requestHeaders.isolatedCopy();
136
137 copy.requestHeaderBytesSent = requestHeaderBytesSent;
138 copy.requestBodyBytesSent = requestBodyBytesSent;
139 copy.responseHeaderBytesReceived = responseHeaderBytesReceived;
140 copy.responseBodyBytesReceived = responseBodyBytesReceived;
141 copy.responseBodyDecodedSize = responseBodyDecodedSize;
142
143 return copy;
144 }

178 && responseBodyBytesReceived == other.responseBodyBytesReceived
179 && responseBodyDecodedSize == other.responseBodyDecodedSize;
