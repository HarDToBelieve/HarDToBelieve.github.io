49class PDFDocumentImage final : public Image {
50public:
51 static Ref<PDFDocumentImage> create(ImageObserver* observer)
52 {
53#if ENABLE(GPU_PROCESS)
54 RELEASE_ASSERT(!isInGPUProcess());

57 }
58
59 void setPdfImageCachingPolicy(PDFImageCachingPolicy);

64
65 unsigned cachingCountForTesting() const { return m_cachingCountForTesting; }
66
67private:
68 PDFDocumentImage(ImageObserver*);

71 bool isPDFDocumentImage() const override { return true; }

73 String filenameExtension() const override;

75 bool hasSingleSecurityOrigin() const override { return true; }
76
77 EncodedDataStatus dataChanged(bool allDataReceived) override;

79 void destroyDecodedData(bool /*destroyAll*/ = true) override;

81 void computeIntrinsicDimensions(Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio) override;

102#if USE(PDFKIT_FOR_PDFDOCUMENTIMAGE)

108 RefPtr<ImageBuffer> m_cachedImageBuffer;
