185enum class RenderingUpdateStep : uint16_t {
186 Resize = 1 << 0,
187 Scroll = 1 << 1,
188 MediaQueryEvaluation = 1 << 2,
189 Animations = 1 << 3,
190 Fullscreen = 1 << 4,
191 AnimationFrameCallbacks = 1 << 5,
192#if ENABLE(INTERSECTION_OBSERVER)
193 IntersectionObservations = 1 << 6,
194#endif
195#if ENABLE(RESIZE_OBSERVER)
196 ResizeObservations = 1 << 7,
197#endif
198 Images = 1 << 8,
199 WheelEventMonitorCallbacks = 1 << 9,
200 CursorUpdate = 1 << 10,
201 EventRegionUpdate = 1 << 11,
202 LayerFlush = 1 << 12,
203#if ENABLE(ASYNC_SCROLLING)
204 ScrollingTreeUpdate = 1 << 13,
205#endif
206};
207
208constexpr OptionSet<RenderingUpdateStep> updateRenderingSteps = {
209 RenderingUpdateStep::Resize,
210 RenderingUpdateStep::Scroll,
211 RenderingUpdateStep::MediaQueryEvaluation,
212 RenderingUpdateStep::Animations,
213 RenderingUpdateStep::Fullscreen,
214 RenderingUpdateStep::AnimationFrameCallbacks,
215#if ENABLE(INTERSECTION_OBSERVER)
216 RenderingUpdateStep::IntersectionObservations,
217#endif
218#if ENABLE(RESIZE_OBSERVER)
219 RenderingUpdateStep::ResizeObservations,
220#endif
221 RenderingUpdateStep::Images,
222 RenderingUpdateStep::WheelEventMonitorCallbacks,
223 RenderingUpdateStep::CursorUpdate,
224 RenderingUpdateStep::EventRegionUpdate,
225};
226
227constexpr auto allRenderingUpdateSteps = updateRenderingSteps | OptionSet<RenderingUpdateStep> {
228 RenderingUpdateStep::LayerFlush,
229#if ENABLE(ASYNC_SCROLLING)
230 RenderingUpdateStep::ScrollingTreeUpdate,
231#endif
232};
