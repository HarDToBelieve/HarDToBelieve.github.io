26#pragma once
27
28namespace WebCore {
29
30class BlobRegistry;
31class LoaderStrategy;
32class MediaStrategy;
33class PasteboardStrategy;
34
35class PlatformStrategies {
36public:
37 LoaderStrategy* loaderStrategy()
38 {
39 if (!m_loaderStrategy)
40 m_loaderStrategy = createLoaderStrategy();
41 return m_loaderStrategy;
42 }
43
44 PasteboardStrategy* pasteboardStrategy()
45 {
46 if (!m_pasteboardStrategy)
47 m_pasteboardStrategy = createPasteboardStrategy();
48 return m_pasteboardStrategy;
49 }
50
51 MediaStrategy& mediaStrategy()
52 {
53 if (!m_mediaStrategy)
54 m_mediaStrategy = createMediaStrategy();
55 return *m_mediaStrategy;
56 }
57
58 BlobRegistry* blobRegistry()
59 {
60 if (!m_blobRegistry)
61 m_blobRegistry = createBlobRegistry();
62 return m_blobRegistry;
63 }
64
65protected:
66 PlatformStrategies() = default;
67
68 virtual ~PlatformStrategies()
69 {
70 }
71
72private:
73 virtual LoaderStrategy* createLoaderStrategy() = 0;
74 virtual PasteboardStrategy* createPasteboardStrategy() = 0;
75 virtual MediaStrategy* createMediaStrategy() = 0;
76 virtual BlobRegistry* createBlobRegistry() = 0;
77
78 LoaderStrategy* m_loaderStrategy { };
79 PasteboardStrategy* m_pasteboardStrategy { };
80 MediaStrategy* m_mediaStrategy { };
81 BlobRegistry* m_blobRegistry { };
82};
83
84bool hasPlatformStrategies();
85WEBCORE_EXPORT PlatformStrategies* platformStrategies();
86WEBCORE_EXPORT void setPlatformStrategies(PlatformStrategies*);
87
88} // namespace WebCore
