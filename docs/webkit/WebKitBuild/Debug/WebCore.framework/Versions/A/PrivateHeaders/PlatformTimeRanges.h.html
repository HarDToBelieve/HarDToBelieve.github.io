26#ifndef PlatformTimeRanges_h
27#define PlatformTimeRanges_h
28
29#include <algorithm>
30#include <wtf/MediaTime.h>
31#include <wtf/Vector.h>

33namespace WTF {
34class PrintStream;
35}
36
37namespace WebCore {
38
39class WEBCORE_EXPORT PlatformTimeRanges {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 explicit PlatformTimeRanges() { }
43 PlatformTimeRanges(const MediaTime& start, const MediaTime& end);
44
45 MediaTime start(unsigned index) const;
46 MediaTime start(unsigned index, bool& valid) const;
47 MediaTime end(unsigned index) const;
48 MediaTime end(unsigned index, bool& valid) const;
49 MediaTime duration(unsigned index) const;
50 MediaTime maximumBufferedTime() const;
51
52 void invert();
53 void intersectWith(const PlatformTimeRanges&);
54 void unionWith(const PlatformTimeRanges&);
55
56 unsigned length() const { return m_ranges.size(); }
57
58 void add(const MediaTime& start, const MediaTime& end);
59 void clear();
60
61 bool contain(const MediaTime&) const;
62
63 size_t find(const MediaTime&) const;
64
65 MediaTime nearest(const MediaTime&) const;
66
67 MediaTime totalDuration() const;
68
69 void dump(WTF::PrintStream&) const;
70
71 template<class Encoder> void encode(Encoder&) const;
72 template<class Decoder> static std::optional<PlatformTimeRanges> decode(Decoder&);
73
74private:
75 // We consider all the Ranges to be semi-bounded as follow: [start, end[
76 struct Range {
77 Range() { }
78 Range(const MediaTime& start, const MediaTime& end)
79 : m_start(start)
80 , m_end(end)
81 {
82 }
83
84 MediaTime m_start;
85 MediaTime m_end;
86
87 template<class Encoder>
88 void encode(Encoder& encoder) const
89 {
90 encoder << m_start;
91 encoder << m_end;
92 }
93
94 template <class Decoder>
95 static std::optional<Range> decode(Decoder& decoder)
96 {
97 std::optional<MediaTime> start;
98 decoder >> start;
99 if (!start)
100 return std::nullopt;
101
102 std::optional<MediaTime> end;
103 decoder >> end;
104 if (!end)
105 return std::nullopt;
106
107 return {{ WTFMove(*start), WTFMove(*end) }};
108 }
109
110 inline bool isPointInRange(const MediaTime& point) const
111 {
112 return m_start <= point && point < m_end;
113 }
114
115 inline bool isOverlappingRange(const Range& range) const
116 {
117 return isPointInRange(range.m_start) || isPointInRange(range.m_end) || range.isPointInRange(m_start);
118 }
119
120 inline bool isContiguousWithRange(const Range& range) const
121 {
122 return range.m_start == m_end || range.m_end == m_start;
123 }
124
125 inline Range unionWithOverlappingOrContiguousRange(const Range& range) const
126 {
127 Range ret;
128
129 ret.m_start = std::min(m_start, range.m_start);
130 ret.m_end = std::max(m_end, range.m_end);
131
132 return ret;
133 }
134
135 inline bool isBeforeRange(const Range& range) const
136 {
137 return range.m_start >= m_end;
138 }
139 };
140
141 PlatformTimeRanges(Vector<Range>&&);
142
143 Vector<Range> m_ranges;
144};
145
146template<class Encoder>
147void PlatformTimeRanges::encode(Encoder& encoder) const
148{
149 encoder << m_ranges;
150}
151
152template <class Decoder>
153std::optional<PlatformTimeRanges> PlatformTimeRanges::decode(Decoder& decoder)
154{
155 std::optional<Vector<Range>> buffered;
156 decoder >> buffered;
157 if (!buffered)
158 return std::nullopt;
159
160 return {{ WTFMove(*buffered) }};
161}
