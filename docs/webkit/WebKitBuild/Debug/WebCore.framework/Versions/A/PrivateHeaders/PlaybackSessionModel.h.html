38class TimeRanges;
39class PlaybackSessionModelClient;
40struct MediaSelectionOption;

42class PlaybackSessionModel : public CanMakeWeakPtr<PlaybackSessionModel> {
43public:
44 virtual ~PlaybackSessionModel() { };
45 virtual void addClient(PlaybackSessionModelClient&) = 0;
46 virtual void removeClient(PlaybackSessionModelClient&) = 0;
47
48 virtual void play() = 0;
49 virtual void pause() = 0;
50 virtual void togglePlayState() = 0;
51 virtual void beginScrubbing() = 0;
52 virtual void endScrubbing() = 0;
53 virtual void seekToTime(double time, double toleranceBefore = 0, double toleranceAfter = 0) = 0;
54 virtual void fastSeek(double time) = 0;
55 virtual void beginScanningForward() = 0;
56 virtual void beginScanningBackward() = 0;
57 virtual void endScanning() = 0;
58 virtual void setDefaultPlaybackRate(double) = 0;
59 virtual void setPlaybackRate(double) = 0;
60 virtual void selectAudioMediaOption(uint64_t index) = 0;
61 virtual void selectLegibleMediaOption(uint64_t index) = 0;
62 virtual void togglePictureInPicture() = 0;
63 virtual void toggleMuted() = 0;
64 virtual void setMuted(bool) = 0;
65 virtual void setVolume(double) = 0;
66 virtual void setPlayingOnSecondScreen(bool) = 0;
67 virtual void sendRemoteCommand(PlatformMediaSession::RemoteControlCommandType, const PlatformMediaSession::RemoteCommandArgument&) { };
68
69 enum ExternalPlaybackTargetType { TargetTypeNone, TargetTypeAirPlay, TargetTypeTVOut };
70
71 virtual double playbackStartedTime() const = 0;
72 virtual double duration() const = 0;
73 virtual double currentTime() const = 0;
74 virtual double bufferedTime() const = 0;
75
76 enum class PlaybackState {
77 Playing = 1 << 0,
78 Stalled = 1 << 1,
79 };
80 virtual bool isPlaying() const = 0;
81 virtual bool isStalled() const = 0;
82 virtual bool isScrubbing() const = 0;
83 virtual double defaultPlaybackRate() const = 0;
84 virtual double playbackRate() const = 0;
85 virtual Ref<TimeRanges> seekableRanges() const = 0;
86 virtual double seekableTimeRangesLastModifiedTime() const = 0;
87 virtual double liveUpdateInterval() const = 0;
88 virtual bool canPlayFastReverse() const = 0;
89 virtual Vector<MediaSelectionOption> audioMediaSelectionOptions() const = 0;
90 virtual uint64_t audioMediaSelectedIndex() const = 0;
91 virtual Vector<MediaSelectionOption> legibleMediaSelectionOptions() const = 0;
92 virtual uint64_t legibleMediaSelectedIndex() const = 0;
93 virtual bool externalPlaybackEnabled() const = 0;
94 virtual ExternalPlaybackTargetType externalPlaybackTargetType() const = 0;
95 virtual String externalPlaybackLocalizedDeviceName() const = 0;
96 virtual bool wirelessVideoPlaybackDisabled() const = 0;
97 virtual bool isMuted() const = 0;
98 virtual double volume() const = 0;
99 virtual bool isPictureInPictureSupported() const = 0;
100 virtual bool isPictureInPictureActive() const = 0;
101};

103class PlaybackSessionModelClient {
104public:
105 virtual ~PlaybackSessionModelClient() { };
106 virtual void durationChanged(double) { }
107 virtual void currentTimeChanged(double /* currentTime */, double /* anchorTime */) { }
108 virtual void bufferedTimeChanged(double) { }
109 virtual void playbackStartedTimeChanged(double /* playbackStartedTime */) { }
110 virtual void rateChanged(OptionSet<PlaybackSessionModel::PlaybackState>, double /* playbackRate */, double /* defaultPlaybackRate */) { }
111 virtual void seekableRangesChanged(const TimeRanges&, double /* lastModified */, double /* liveInterval */) { }
112 virtual void canPlayFastReverseChanged(bool) { }
113 virtual void audioMediaSelectionOptionsChanged(const Vector<MediaSelectionOption>& /* options */, uint64_t /* selectedIndex */) { }
114 virtual void legibleMediaSelectionOptionsChanged(const Vector<MediaSelectionOption>& /* options */, uint64_t /* selectedIndex */) { }
115 virtual void audioMediaSelectionIndexChanged(uint64_t) { }
116 virtual void legibleMediaSelectionIndexChanged(uint64_t) { }
117 virtual void externalPlaybackChanged(bool /* enabled */, PlaybackSessionModel::ExternalPlaybackTargetType, const String& /* localizedDeviceName */) { }
118 virtual void wirelessVideoPlaybackDisabledChanged(bool) { }
119 virtual void mutedChanged(bool) { }
120 virtual void volumeChanged(double) { }
121 virtual void isPictureInPictureSupportedChanged(bool) { }
122 virtual void pictureInPictureActiveChanged(bool) { }
123 virtual void ensureControlsManager() { }
124 virtual void modelDestroyed() { }
125};
