34enum class PluginLoadClientPolicy : uint8_t {
35 // No client-specific plug-in load policy has been defined. The plug-in should be visible in navigator.plugins and WebKit should synchronously
36 // ask the client whether the plug-in should be loaded.
37 Undefined = 0,
38
39 // The plug-in module should be blocked from being instantiated. The plug-in should be hidden in navigator.plugins.
40 Block,
41
42 // WebKit should synchronously ask the client whether the plug-in should be loaded. The plug-in should be visible in navigator.plugins.
43 Ask,
44
45 // The plug-in module may be loaded if WebKit is not blocking it.
46 Allow,
47
48 // The plug-in module should be loaded irrespective of whether WebKit has asked it to be blocked.
49 AllowAlways,
50};

52struct MimeClassInfo {
53 String type;
54 String desc;
55 Vector<String> extensions;
56};
57
58inline bool operator==(const MimeClassInfo& a, const MimeClassInfo& b)
59{
60 return a.type == b.type && a.desc == b.desc && a.extensions == b.extensions;
61}
62
63struct PluginInfo {
64 String name;
65 String file;
66 String desc;
67 Vector<MimeClassInfo> mimes;
68 bool isApplicationPlugin { false };
69
70 PluginLoadClientPolicy clientLoadPolicy { PluginLoadClientPolicy::Undefined };
71
72 String bundleIdentifier;
73#if PLATFORM(MAC)
74 String versionString;
75#endif
76};
77
78inline bool operator==(PluginInfo& a, PluginInfo& b)
79{
80 bool result = a.name == b.name && a.file == b.file && a.desc == b.desc && a.mimes == b.mimes && a.isApplicationPlugin == b.isApplicationPlugin && a.clientLoadPolicy == b.clientLoadPolicy && a.bundleIdentifier == b.bundleIdentifier;
81#if PLATFORM(MAC)
82 result = result && a.versionString == b.versionString;
83#endif
84 return result;
85}

132inline bool isSupportedPlugin(const Vector<SupportedPluginIdentifier>& pluginIdentifiers, const URL& pageURL, const String& pluginIdentifier)
133{
134 return pluginIdentifiers.findMatching([&] (auto&& plugin) {
135 return pageURL.isMatchingDomain(plugin.matchingDomain) && plugin.pluginIdentifier == pluginIdentifier;
136 }) != notFound;
137}
138
139template<class Decoder> inline std::optional<SupportedPluginIdentifier> SupportedPluginIdentifier::decode(Decoder& decoder)
140{
141 std::optional<String> matchingDomain;
142 decoder >> matchingDomain;
143 if (!matchingDomain)
144 return std::nullopt;
145
146 std::optional<String> pluginIdentifier;
147 decoder >> pluginIdentifier;
148 if (!pluginIdentifier)
149 return std::nullopt;
150
151 return SupportedPluginIdentifier { WTFMove(matchingDomain.value()), WTFMove(pluginIdentifier.value()) };
152}
153
154template<class Encoder> inline void SupportedPluginIdentifier::encode(Encoder& encoder) const
155{
156 encoder << matchingDomain;
157 encoder << pluginIdentifier;
158}
