25#pragma once
26
27#include "ExceptionOr.h"
28#include "PointerID.h"
29#include <wtf/HashMap.h>
30
31namespace WebCore {
32
33class Document;
34class Element;
35class EventTarget;
36class IntPoint;
37class MouseEvent;
38class Page;
39class PlatformTouchEvent;
40class PointerEvent;
41class WindowProxy;
42
43class PointerCaptureController {
44 WTF_MAKE_NONCOPYABLE(PointerCaptureController);
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 explicit PointerCaptureController(Page&);
48
49 Element* pointerCaptureElement(Document*, PointerID) const;
50 ExceptionOr<void> setPointerCapture(Element*, PointerID);
51 ExceptionOr<void> releasePointerCapture(Element*, PointerID);
52 bool hasPointerCapture(Element*, PointerID);
53 void reset();
54
55 void pointerLockWasApplied();
56 void elementWasRemoved(Element&);
57
58 RefPtr<PointerEvent> pointerEventForMouseEvent(const MouseEvent&, PointerID, const String& pointerType);
59
60#if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
61 void dispatchEventForTouchAtIndex(EventTarget&, const PlatformTouchEvent&, unsigned, bool isPrimary, WindowProxy&);
62#endif
63
64 WEBCORE_EXPORT void touchWithIdentifierWasRemoved(PointerID);
65 bool hasCancelledPointerEventForIdentifier(PointerID) const;
66 bool preventsCompatibilityMouseEventsForIdentifier(PointerID) const;
67 void dispatchEvent(PointerEvent&, EventTarget*);
68 WEBCORE_EXPORT void cancelPointer(PointerID, const IntPoint&);
69 void processPendingPointerCapture(PointerID);
70
71private:
72 struct CapturingData {
73 RefPtr<Element> pendingTargetOverride;
74 RefPtr<Element> targetOverride;
75#if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
76 RefPtr<Element> previousTarget;
77#endif
78 bool hasAnyElement() const {
79 return pendingTargetOverride || targetOverride
80#if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
81 || previousTarget
82#endif
83 ;
84 }
85 String pointerType;
86 enum class State : uint8_t {
87 Ready,
88 Finished,
89 Cancelled,
90 };
91 State state { State::Ready };
92 bool isPrimary { false };
93 bool preventsCompatibilityMouseEvents { false };
94 bool pointerIsPressed { false };
95 short previousMouseButton { -1 };
96 };
97
98 CapturingData& ensureCapturingDataForPointerEvent(const PointerEvent&);
99 void pointerEventWillBeDispatched(const PointerEvent&, EventTarget*);
100 void pointerEventWasDispatched(const PointerEvent&);
101
102 void updateHaveAnyCapturingElement();
103
104 Page& m_page;
105 // While PointerID is defined as int32_t, we use int64_t here so that we may use a value outside of the int32_t range to have safe
106 // empty and removed values, allowing any int32_t to be provided through the API for lookup in this hashmap.
107 using PointerIdToCapturingDataMap = HashMap<int64_t, CapturingData, WTF::IntHash<int64_t>, WTF::SignedWithZeroKeyHashTraits<int64_t>>;
108 PointerIdToCapturingDataMap m_activePointerIdsToCapturingData;
109 bool m_processingPendingPointerCapture { false };
110 bool m_haveAnyCapturingElement { false };
111};
112
113} // namespace WebCore
