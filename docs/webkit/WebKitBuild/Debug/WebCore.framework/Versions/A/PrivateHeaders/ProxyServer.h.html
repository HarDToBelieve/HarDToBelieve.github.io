28#include <wtf/Vector.h>

31namespace WebCore {
32
33class NetworkingContext;
34
35// Represents a single proxy server.
36class ProxyServer {
37public:
38 enum Type {
39 Direct,
40 HTTP,
41 HTTPS,
42 SOCKS,
43 };
44
45 ProxyServer()
46 : m_type(Direct)
47 , m_port(-1)
48 {
49 }
50
51 ProxyServer(Type type, const String& hostName, int port)
52 : m_type(type)
53 , m_hostName(hostName)
54 , m_port(port)
55 {
56 }
57
58 Type type() const { return m_type; }
59 const String& hostName() const { return m_hostName; }
60 int port() const { return m_port; }
61
62private:
63 Type m_type;
64 String m_hostName;
65 int m_port;
66};
67
68// Return a vector of proxy servers for the given URL.
69WEBCORE_EXPORT Vector<ProxyServer> proxyServersForURL(const URL&);
70
71// Converts the given vector of proxy servers to a PAC string, as described in
72// http://web.archive.org/web/20060424005037/wp.netscape.com/eng/mozilla/2.0/relnotes/demo/proxy-live.html
73WEBCORE_EXPORT String toString(const Vector<ProxyServer>&);
