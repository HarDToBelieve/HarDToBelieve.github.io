37class RTCDataChannelRemoteSource : public RTCDataChannelHandlerClient {
38public:
39 static Ref<RTCDataChannelRemoteSource> create(RTCDataChannelIdentifier identifier, UniqueRef<RTCDataChannelHandler>&& handler, Ref<RTCDataChannelRemoteSourceConnection>&& connection) { return adoptRef(*new RTCDataChannelRemoteSource(identifier, WTFMove(handler), WTFMove(connection))); }
40 ~RTCDataChannelRemoteSource();
41
42 void sendStringData(const CString& text) { m_handler->sendStringData(text); }
43 void sendRawData(const uint8_t* data, size_t size) { m_handler->sendRawData(data, size); }
44 void close() { m_handler->close(); }

46private:
47 WEBCORE_EXPORT RTCDataChannelRemoteSource(RTCDataChannelIdentifier, UniqueRef<RTCDataChannelHandler>&&, Ref<RTCDataChannelRemoteSourceConnection>&&);
48
49 // RTCDataChannelHandlerClient
50 void didChangeReadyState(RTCDataChannelState state) final { m_connection->didChangeReadyState(m_identifier, state); }
51 void didReceiveStringData(const String& text) final { m_connection->didReceiveStringData(m_identifier, text); }
52 void didReceiveRawData(const uint8_t* data, size_t size) final { m_connection->didReceiveRawData(m_identifier, data, size); }
53 void didDetectError() final { m_connection->didDetectError(m_identifier); }
54 void bufferedAmountIsDecreasing(size_t amount) final { m_connection->bufferedAmountIsDecreasing(m_identifier, amount); }
55
56 RTCDataChannelIdentifier m_identifier;
57 UniqueRef<RTCDataChannelHandler> m_handler;
58 Ref<RTCDataChannelRemoteSourceConnection> m_connection;
59};
