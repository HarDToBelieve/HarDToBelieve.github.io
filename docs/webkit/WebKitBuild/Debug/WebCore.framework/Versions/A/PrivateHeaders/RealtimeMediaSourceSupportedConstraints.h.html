39enum class MediaConstraintType : uint8_t {
40 Unknown,
41 Width,
42 Height,
43 AspectRatio,
44 FrameRate,
45 FacingMode,
46 Volume,
47 SampleRate,
48 SampleSize,
49 EchoCancellation,
50 DeviceId,
51 GroupId,
52 DisplaySurface,
53 LogicalSurface,
54};

56class RealtimeMediaSourceSupportedConstraints {
57public:
58 RealtimeMediaSourceSupportedConstraints()
59 {
60 }
61
62 bool supportsWidth() const { return m_supportsWidth; }
63 void setSupportsWidth(bool value) { m_supportsWidth = value; }
64
65 bool supportsHeight() const { return m_supportsHeight; }
66 void setSupportsHeight(bool value) { m_supportsHeight = value; }
67
68 bool supportsAspectRatio() const { return m_supportsAspectRatio; }
69 void setSupportsAspectRatio(bool value) { m_supportsAspectRatio = value; }
70
71 bool supportsFrameRate() const { return m_supportsFrameRate; }
72 void setSupportsFrameRate(bool value) { m_supportsFrameRate = value; }
73
74 bool supportsFacingMode() const { return m_supportsFacingMode; }
75 void setSupportsFacingMode(bool value) { m_supportsFacingMode = value; }
76
77 bool supportsVolume() const { return m_supportsVolume; }
78 void setSupportsVolume(bool value) { m_supportsVolume = value; }
79
80 bool supportsSampleRate() const { return m_supportsSampleRate; }
81 void setSupportsSampleRate(bool value) { m_supportsSampleRate = value; }
82
83 bool supportsSampleSize() const { return m_supportsSampleSize; }

85
86 bool supportsEchoCancellation() const { return m_supportsEchoCancellation; }
87 void setSupportsEchoCancellation(bool value) { m_supportsEchoCancellation = value; }
88
89 bool supportsDeviceId() const { return m_supportsDeviceId; }
90 void setSupportsDeviceId(bool value) { m_supportsDeviceId = value; }

92 bool supportsGroupId() const { return m_supportsGroupId; }
93 void setSupportsGroupId(bool value) { m_supportsGroupId = value; }

95 bool supportsDisplaySurface() const { return m_supportsDisplaySurface; }
96 void setSupportsDisplaySurface(bool value) { m_supportsDisplaySurface = value; }

98 bool supportsLogicalSurface() const { return m_supportsLogicalSurface; }
99 void setSupportsLogicalSurface(bool value) { m_supportsLogicalSurface = value; }
100
101 bool supportsConstraint(MediaConstraintType) const;
102
103 template<class Encoder> void encode(Encoder&) const;
104 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, RealtimeMediaSourceSupportedConstraints&);
105
106private:
107 bool m_supportsWidth { false };
108 bool m_supportsHeight { false };
109 bool m_supportsAspectRatio { false };
110 bool m_supportsFrameRate { false };
111 bool m_supportsFacingMode { false };
112 bool m_supportsVolume { false };
113 bool m_supportsSampleRate { false };
114 bool m_supportsSampleSize { false };
115 bool m_supportsEchoCancellation { false };
116 bool m_supportsDeviceId { false };
117 bool m_supportsGroupId { false };
118 bool m_supportsDisplaySurface { false };
119 bool m_supportsLogicalSurface { false };
120};
121
122template<class Encoder>
123void RealtimeMediaSourceSupportedConstraints::encode(Encoder& encoder) const
124{
125 encoder << m_supportsWidth
126 << m_supportsHeight
127 << m_supportsAspectRatio
128 << m_supportsFrameRate
129 << m_supportsFacingMode
130 << m_supportsVolume
131 << m_supportsSampleRate
132 << m_supportsSampleSize
133 << m_supportsEchoCancellation
134 << m_supportsDeviceId
135 << m_supportsGroupId
136 << m_supportsDisplaySurface
137 << m_supportsLogicalSurface;
138}
139
140template<class Decoder>
141bool RealtimeMediaSourceSupportedConstraints::decode(Decoder& decoder, RealtimeMediaSourceSupportedConstraints& constraints)
142{
143 return decoder.decode(constraints.m_supportsWidth)
144 && decoder.decode(constraints.m_supportsHeight)
145 && decoder.decode(constraints.m_supportsAspectRatio)
146 && decoder.decode(constraints.m_supportsFrameRate)
147 && decoder.decode(constraints.m_supportsFacingMode)
148 && decoder.decode(constraints.m_supportsVolume)
149 && decoder.decode(constraints.m_supportsSampleRate)
150 && decoder.decode(constraints.m_supportsSampleSize)
151 && decoder.decode(constraints.m_supportsEchoCancellation)
152 && decoder.decode(constraints.m_supportsDeviceId)
153 && decoder.decode(constraints.m_supportsGroupId)
154 && decoder.decode(constraints.m_supportsDisplaySurface)
155 && decoder.decode(constraints.m_supportsLogicalSurface);
156}
