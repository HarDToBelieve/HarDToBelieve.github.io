33namespace WebCore {
34
35enum class BoxSideFlag : uint8_t {
36 Top = 1 << static_cast<unsigned>(BoxSide::Top),
37 Right = 1 << static_cast<unsigned>(BoxSide::Right),
38 Bottom = 1 << static_cast<unsigned>(BoxSide::Bottom),
39 Left = 1 << static_cast<unsigned>(BoxSide::Left)
40};
41
42using BoxSideSet = OptionSet<BoxSideFlag>;
43
44template<typename T> class RectEdges {
45public:
46 RectEdges() = default;
47
48 RectEdges(const RectEdges&) = default;
49
50 template<typename U>
51 RectEdges(U&& top, U&& right, U&& bottom, U&& left)
52 : m_sides({ { std::forward<T>(top), std::forward<T>(right), std::forward<T>(bottom), std::forward<T>(left) } })
53 { }
54
55 T& at(BoxSide side) { return m_sides[static_cast<size_t>(side)]; }
56 T& operator[](BoxSide side) { return m_sides[static_cast<size_t>(side)]; }
57 T& top() { return at(BoxSide::Top); }
58 T& right() { return at(BoxSide::Right); }
59 T& bottom() { return at(BoxSide::Bottom); }
60 T& left() { return at(BoxSide::Left); }
61
62 const T& at(BoxSide side) const { return m_sides[static_cast<size_t>(side)]; }
63 const T& operator[](BoxSide side) const { return m_sides[static_cast<size_t>(side)]; }
64 const T& top() const { return at(BoxSide::Top); }
65 const T& right() const { return at(BoxSide::Right); }
66 const T& bottom() const { return at(BoxSide::Bottom); }
67 const T& left() const { return at(BoxSide::Left); }
68
69 void setAt(BoxSide side, const T& v) { at(side) = v; }
70 void setTop(const T& top) { setAt(BoxSide::Top, top); }
71 void setRight(const T& right) { setAt(BoxSide::Right, right); }
72 void setBottom(const T& bottom) { setAt(BoxSide::Bottom, bottom); }
73 void setLeft(const T& left) { setAt(BoxSide::Left, left); }
74
75 T& before(WritingMode writingMode) { return at(mapLogicalSideToPhysicalSide(writingMode, LogicalBoxSide::BlockStart)); }
76 T& after(WritingMode writingMode) { return at(mapLogicalSideToPhysicalSide(writingMode, LogicalBoxSide::BlockEnd)); }
77 T& start(WritingMode writingMode, TextDirection direction = TextDirection::LTR) { return at(mapLogicalSideToPhysicalSide(makeTextFlow(writingMode, direction), LogicalBoxSide::InlineStart)); }
78 T& end(WritingMode writingMode, TextDirection direction = TextDirection::LTR) { return at(mapLogicalSideToPhysicalSide(makeTextFlow(writingMode, direction), LogicalBoxSide::InlineEnd)); }
79
80 const T& before(WritingMode writingMode) const { return at(mapLogicalSideToPhysicalSide(writingMode, LogicalBoxSide::BlockStart)); }
81 const T& after(WritingMode writingMode) const { return at(mapLogicalSideToPhysicalSide(writingMode, LogicalBoxSide::BlockEnd)); }
82 const T& start(WritingMode writingMode, TextDirection direction = TextDirection::LTR) const { return at(mapLogicalSideToPhysicalSide(makeTextFlow(writingMode, direction), LogicalBoxSide::InlineStart)); }
83 const T& end(WritingMode writingMode, TextDirection direction = TextDirection::LTR) const { return at(mapLogicalSideToPhysicalSide(makeTextFlow(writingMode, direction), LogicalBoxSide::InlineEnd)); }
84
85 void setBefore(const T& before, WritingMode writingMode) { this->before(writingMode) = before; }
86 void setAfter(const T& after, WritingMode writingMode) { this->after(writingMode) = after; }
87 void setStart(const T& start, WritingMode writingMode, TextDirection direction = TextDirection::LTR) { this->start(writingMode, direction) = start; }
88 void setEnd(const T& end, WritingMode writingMode, TextDirection direction = TextDirection::LTR) { this->end(writingMode, direction) = end; }
89
90 bool operator==(const RectEdges& other) const { return m_sides == other.m_sides; }
91 bool operator!=(const RectEdges& other) const { return m_sides != other.m_sides; }
92
93private:
94 std::array<T, 4> m_sides { };
95};
96
97
98template<typename T>
99TextStream& operator<<(TextStream& ts, const RectEdges<T>& edges)
100{
101 ts << "[top " << edges.top() << " right " << edges.right() << " bottom " << edges.bottom() << " left " << edges.left() << "]";
102 return ts;
103}
