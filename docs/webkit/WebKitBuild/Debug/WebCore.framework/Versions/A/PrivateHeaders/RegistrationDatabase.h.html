26#pragma once
27
28#if ENABLE(SERVICE_WORKER)
29
30#include "SecurityOrigin.h"
31#include "ServiceWorkerRegistrationKey.h"
32#include <wtf/ThreadSafeRefCounted.h>
33#include <wtf/WeakPtr.h>
34#include <wtf/WorkQueue.h>
35#include <wtf/text/WTFString.h>
36
37namespace WebCore {
38
39class RegistrationStore;
40class SQLiteDatabase;
41class SWScriptStorage;
42struct ServiceWorkerContextData;
43
44WEBCORE_EXPORT String serviceWorkerRegistrationDatabaseFilename(const String& databaseDirectory);
45
46class RegistrationDatabase : public ThreadSafeRefCounted<RegistrationDatabase, WTF::DestructionThread::Main> {
47WTF_MAKE_FAST_ALLOCATED;
48public:
49 static Ref<RegistrationDatabase> create(RegistrationStore& store, String&& databaseDirectory)
50 {
51 return adoptRef(*new RegistrationDatabase(store, WTFMove(databaseDirectory)));
52 }
53
54 ~RegistrationDatabase();
55
56 void pushChanges(const HashMap<ServiceWorkerRegistrationKey, std::optional<ServiceWorkerContextData>>&, CompletionHandler<void()>&&);
57 void clearAll(CompletionHandler<void()>&&);
58 void close(CompletionHandler<void()>&&);
59
60private:
61 RegistrationDatabase(RegistrationStore&, String&& databaseDirectory);
62
63 String databaseDirectoryIsolatedCopy() const { return m_databaseDirectory.isolatedCopy(); }
64
65 enum class ShouldRetry { No, Yes };
66 void schedulePushChanges(Vector<ServiceWorkerContextData>&&, Vector<ServiceWorkerRegistrationKey>&&, ShouldRetry, CompletionHandler<void()>&&);
67 void postTaskToWorkQueue(Function<void()>&&);
68
69 // Methods to be run on the work queue.
70 bool openSQLiteDatabase(const String& fullFilename);
71 String ensureValidRecordsTable();
72 String importRecords();
73 void importRecordsIfNecessary();
74 bool doPushChanges(const Vector<ServiceWorkerContextData>&, const Vector<ServiceWorkerRegistrationKey>&);
75 void doClearOrigin(const SecurityOrigin&);
76 SWScriptStorage& scriptStorage();
77 String scriptStorageDirectory() const;
78
79 // Replies to the main thread.
80 void addRegistrationToStore(ServiceWorkerContextData&&);
81 void databaseFailedToOpen();
82 void databaseOpenedAndRecordsImported();
83
84 Ref<WorkQueue> m_workQueue;
85 WeakPtr<RegistrationStore> m_store;
86 std::unique_ptr<SWScriptStorage> m_scriptStorage;
87 String m_databaseDirectory;
88 String m_databaseFilePath;
89 std::unique_ptr<SQLiteDatabase> m_database;
90 uint64_t m_pushCounter { 0 };
91};
92
93} // namespace WebCore
94
95#endif // ENABLE(SERVICE_WORKER)
