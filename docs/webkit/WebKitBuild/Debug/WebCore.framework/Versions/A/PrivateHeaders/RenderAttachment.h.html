28#if ENABLE(ATTACHMENT_ELEMENT)

30#include "HTMLAttachmentElement.h"
31#include "RenderReplaced.h"

33namespace WebCore {
34
35class RenderAttachment final : public RenderReplaced {
36 WTF_MAKE_ISO_ALLOCATED(RenderAttachment);
37public:
38 RenderAttachment(HTMLAttachmentElement&, RenderStyle&&);
39
40 HTMLAttachmentElement& attachmentElement() const;
41
42 void setShouldDrawBorder(bool drawBorder) { m_shouldDrawBorder = drawBorder; }
43 bool shouldDrawBorder() const;
44
45 void invalidate();
46
47private:
48 void element() const = delete;
49 bool isAttachment() const override { return true; }
50 const char* renderName() const override { return "RenderAttachment"; }
51
52 bool shouldDrawSelectionTint() const override { return false; }
53 void paintReplaced(PaintInfo&, const LayoutPoint& offset) final;
54
55 void layout() override;
56
57 LayoutUnit baselinePosition(FontBaseline, bool, LineDirectionMode, LinePositionMode) const override;
58
59 LayoutUnit m_minimumIntrinsicWidth;
60 bool m_shouldDrawBorder { true };
61};
62
63inline RenderAttachment* HTMLAttachmentElement::renderer() const
64{
65 return downcast<RenderAttachment>(HTMLElement::renderer());
66}
67
68} // namespace WebCore
