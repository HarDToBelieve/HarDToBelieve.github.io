23#pragma once
24
25#include "RenderElement.h"
26
27namespace WebCore {
28
29class KeyframeList;
30class RenderLayer;
31
32struct LayerRepaintRects {
33 LayoutRect clippedOverflowRect;
34 LayoutRect outlineBoundsRect;
35};
36
37class RenderLayerModelObject : public RenderElement {
38 WTF_MAKE_ISO_ALLOCATED(RenderLayerModelObject);
39public:
40 virtual ~RenderLayerModelObject();
41
42 void destroyLayer();
43
44 bool hasSelfPaintingLayer() const;
45 RenderLayer* layer() const { return m_layer.get(); }
46
47 void styleWillChange(StyleDifference, const RenderStyle& newStyle) override;
48 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
49 virtual void updateFromStyle() { }
50
51 virtual bool requiresLayer() const = 0;
52
53 // Returns true if the background is painted opaque in the given rect.
54 // The query rect is given in local coordinate system.
55 virtual bool backgroundIsKnownToBeOpaqueInRect(const LayoutRect&) const { return false; }
56
57 virtual bool isScrollableOrRubberbandableBox() const { return false; }
58
59 bool shouldPlaceVerticalScrollbarOnLeft() const;
60
61 std::optional<LayerRepaintRects> layerRepaintRects() const;
62
63 bool startAnimation(double timeOffset, const Animation&, const KeyframeList&) override;
64 void animationPaused(double timeOffset, const String& name) override;
65 void animationFinished(const String& name) override;
66 void transformRelatedPropertyDidChange() override;
67
68 void suspendAnimations(MonotonicTime = MonotonicTime()) override;
69
70protected:
71 RenderLayerModelObject(Element&, RenderStyle&&, BaseTypeFlags);
72 RenderLayerModelObject(Document&, RenderStyle&&, BaseTypeFlags);
73
74 void createLayer();
75 void willBeDestroyed() override;
76
77private:
78 std::unique_ptr<RenderLayer> m_layer;
79
80 // Used to store state between styleWillChange and styleDidChange
81 static bool s_wasFloating;
82 static bool s_hadLayer;
83 static bool s_hadTransform;
84 static bool s_layerWasSelfPainting;
85};
86
87} // namespace WebCore
