31namespace WebCore {
32
33class RenderSelectionInfoBase {
34 WTF_MAKE_NONCOPYABLE(RenderSelectionInfoBase); WTF_MAKE_FAST_ALLOCATED;
35public:
36 explicit RenderSelectionInfoBase(RenderObject& renderer);
37 RenderLayerModelObject* repaintContainer() const { return m_repaintContainer; }
38 RenderObject::HighlightState state() const { return m_state; }
39
40protected:
41 void repaintRectangle(const LayoutRect& repaintRect);
42
43 RenderObject& m_renderer;
44 RenderLayerModelObject* m_repaintContainer;
45
46private:
47 RenderObject::HighlightState m_state;
48};
49
50// This struct is used when the selection changes to cache the old and new state of the selection for each RenderObject.
51class RenderSelectionInfo : public RenderSelectionInfoBase {
52public:
53 RenderSelectionInfo(RenderObject& renderer, bool clipToVisibleContent);
54
55 void repaint();
56 const Vector<FloatQuad>& collectedSelectionQuads() const { return m_collectedSelectionQuads; }
57 LayoutRect rect() const { return m_rect; }
58
59private:
60 Vector<FloatQuad> m_collectedSelectionQuads; // relative to repaint container
61 LayoutRect m_rect; // relative to repaint container
62};
63
64
65// This struct is used when the selection changes to cache the old and new state of the selection for each RenderBlock.
66class RenderBlockSelectionInfo : public RenderSelectionInfoBase {
67public:
68 explicit RenderBlockSelectionInfo(RenderBlock& renderer);
69
70 void repaint();
71 GapRects rects() const { return m_rects; }
72
73private:
74 GapRects m_rects; // relative to repaint container
75};
