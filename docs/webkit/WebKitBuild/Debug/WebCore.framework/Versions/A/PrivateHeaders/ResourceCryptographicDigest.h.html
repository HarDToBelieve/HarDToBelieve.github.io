39struct ResourceCryptographicDigest {
40 enum class Algorithm {
41 SHA256 = 1 << 0,
42 SHA384 = 1 << 1,
43 SHA512 = 1 << 2,
44 };
45
46 // Number of bytes to hold SHA-512 digest
47 static constexpr size_t maximumDigestLength = 64;
48
49 Algorithm algorithm;
50 Vector<uint8_t> value;
51
52 bool operator==(const ResourceCryptographicDigest& other) const
53 {
54 return algorithm == other.algorithm && value == other.value;
55 }
56
57 bool operator!=(const ResourceCryptographicDigest& other) const
58 {
59 return !(*this == other);
60 }
61};
62
63struct EncodedResourceCryptographicDigest {
64 using Algorithm = ResourceCryptographicDigest::Algorithm;
65
66 Algorithm algorithm;
67 String digest;
68};

85template<> struct DefaultHash<WebCore::ResourceCryptographicDigest> {
86 static unsigned hash(const WebCore::ResourceCryptographicDigest& digest)
87 {
88 return pairIntHash(intHash(static_cast<unsigned>(digest.algorithm)), StringHasher::computeHash(digest.value.data(), digest.value.size()));
89 }
90 static bool equal(const WebCore::ResourceCryptographicDigest& a, const WebCore::ResourceCryptographicDigest& b)
91 {
92 return a == b;
93 }
94 static const bool safeToCompareToEmptyOrDeleted = true;
95};

97template<> struct HashTraits<WebCore::ResourceCryptographicDigest> : GenericHashTraits<WebCore::ResourceCryptographicDigest> {
98 using Algorithm = WebCore::ResourceCryptographicDigest::Algorithm;
99 using AlgorithmUnderlyingType = typename std::underlying_type<Algorithm>::type;
100 static constexpr auto emptyAlgorithmValue = static_cast<Algorithm>(std::numeric_limits<AlgorithmUnderlyingType>::max());
101 static constexpr auto deletedAlgorithmValue = static_cast<Algorithm>(std::numeric_limits<AlgorithmUnderlyingType>::max() - 1);
102
103 static const bool emptyValueIsZero = false;
104
105 static WebCore::ResourceCryptographicDigest emptyValue()
106 {
107 return { emptyAlgorithmValue, { } };
108 }
109
110 static void constructDeletedValue(WebCore::ResourceCryptographicDigest& slot)
111 {
112 slot.algorithm = deletedAlgorithmValue;
113 }
114
115 static bool isDeletedValue(const WebCore::ResourceCryptographicDigest& slot)
116 {
117 return slot.algorithm == deletedAlgorithmValue;
118 }
119};
