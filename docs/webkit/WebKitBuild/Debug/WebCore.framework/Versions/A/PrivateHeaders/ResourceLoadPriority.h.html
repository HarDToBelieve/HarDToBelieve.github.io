28#include <wtf/EnumTraits.h>

30namespace WebCore {
31
32enum class ResourceLoadPriority : uint8_t {
33 VeryLow,
34 Low,
35 Medium,
36 High,
37 VeryHigh,
38 Lowest = VeryLow,
39 Highest = VeryHigh,
40};
41static constexpr unsigned bitWidthOfResourceLoadPriority = 3;
42static_assert(static_cast<unsigned>(ResourceLoadPriority::Highest) <= ((1U << bitWidthOfResourceLoadPriority) - 1));
43
44static const unsigned resourceLoadPriorityCount { static_cast<unsigned>(ResourceLoadPriority::Highest) + 1 };
45
46inline ResourceLoadPriority& operator++(ResourceLoadPriority& priority)
47{
48 ASSERT(priority != ResourceLoadPriority::Highest);
49 return priority = static_cast<ResourceLoadPriority>(static_cast<int>(priority) + 1);
50}
51
52inline ResourceLoadPriority& operator--(ResourceLoadPriority& priority)
53{
54 ASSERT(priority != ResourceLoadPriority::Lowest);
55 return priority = static_cast<ResourceLoadPriority>(static_cast<int>(priority) - 1);
56}
57
58} // namespace WebCore
59
60namespace WTF {

64 WebCore::ResourceLoadPriority,
65 WebCore::ResourceLoadPriority::VeryLow,
66 WebCore::ResourceLoadPriority::Low,
67 WebCore::ResourceLoadPriority::Medium,
68 WebCore::ResourceLoadPriority::High,
69 WebCore::ResourceLoadPriority::VeryHigh
70 >;
71};
