148struct ResourceLoaderOptions : public FetchOptions {
149 ResourceLoaderOptions()
150 : ResourceLoaderOptions(FetchOptions())
151 {
152 }
153
154 ResourceLoaderOptions(FetchOptions options)
155 : FetchOptions { WTFMove(options) }
156 , sendLoadCallbacks(SendCallbackPolicy::DoNotSendCallbacks)
157 , sniffContent(ContentSniffingPolicy::DoNotSniffContent)
158 , sniffContentEncoding(ContentEncodingSniffingPolicy::Sniff)
159 , dataBufferingPolicy(DataBufferingPolicy::BufferData)
160 , storedCredentialsPolicy(StoredCredentialsPolicy::DoNotUse)
161 , securityCheck(SecurityCheckPolicy::DoSecurityCheck)
162 , certificateInfoPolicy(CertificateInfoPolicy::DoNotIncludeCertificateInfo)
163 , contentSecurityPolicyImposition(ContentSecurityPolicyImposition::DoPolicyCheck)
164 , defersLoadingPolicy(DefersLoadingPolicy::AllowDefersLoading)
165 , cachingPolicy(CachingPolicy::AllowCaching)
166 , sameOriginDataURLFlag(SameOriginDataURLFlag::Unset)
167 , initiatorContext(InitiatorContext::Document)
168 , serviceWorkersMode(ServiceWorkersMode::All)
169 , applicationCacheMode(ApplicationCacheMode::Use)
170 , clientCredentialPolicy(ClientCredentialPolicy::CannotAskClientForCredentials)
171 , preflightPolicy(PreflightPolicy::Consider)
172 , loadedFromOpaqueSource(LoadedFromOpaqueSource::No)
173 { }
174
175 ResourceLoaderOptions(SendCallbackPolicy sendLoadCallbacks, ContentSniffingPolicy sniffContent, DataBufferingPolicy dataBufferingPolicy, StoredCredentialsPolicy storedCredentialsPolicy, ClientCredentialPolicy credentialPolicy, FetchOptions::Credentials credentials, SecurityCheckPolicy securityCheck, FetchOptions::Mode mode, CertificateInfoPolicy certificateInfoPolicy, ContentSecurityPolicyImposition contentSecurityPolicyImposition, DefersLoadingPolicy defersLoadingPolicy, CachingPolicy cachingPolicy)
176 : sendLoadCallbacks(sendLoadCallbacks)
177 , sniffContent(sniffContent)
178 , sniffContentEncoding(ContentEncodingSniffingPolicy::Sniff)
179 , dataBufferingPolicy(dataBufferingPolicy)
180 , storedCredentialsPolicy(storedCredentialsPolicy)
181 , securityCheck(securityCheck)
182 , certificateInfoPolicy(certificateInfoPolicy)
183 , contentSecurityPolicyImposition(contentSecurityPolicyImposition)
184 , defersLoadingPolicy(defersLoadingPolicy)
185 , cachingPolicy(cachingPolicy)
186 , sameOriginDataURLFlag(SameOriginDataURLFlag::Unset)
187 , initiatorContext(InitiatorContext::Document)
188 , serviceWorkersMode(ServiceWorkersMode::All)
189 , applicationCacheMode(ApplicationCacheMode::Use)
190 , clientCredentialPolicy(credentialPolicy)
191 , preflightPolicy(PreflightPolicy::Consider)
192 , loadedFromOpaqueSource(LoadedFromOpaqueSource::No)
193
194 {
195 this->credentials = credentials;
196 this->mode = mode;
197 }
198
199#if ENABLE(SERVICE_WORKER)
200 Markable<ServiceWorkerRegistrationIdentifier, ServiceWorkerRegistrationIdentifier::MarkableTraits> serviceWorkerRegistrationIdentifier;
201#endif
202 Markable<ContentSecurityPolicyResponseHeaders, ContentSecurityPolicyResponseHeaders::MarkableTraits> cspResponseHeaders;
203 OptionSet<HTTPHeadersToKeepFromCleaning> httpHeadersToKeep;
204 uint8_t maxRedirectCount { 20 };
205
206 SendCallbackPolicy sendLoadCallbacks : bitWidthOfSendCallbackPolicy;
207 ContentSniffingPolicy sniffContent : bitWidthOfContentSniffingPolicy;
208 ContentEncodingSniffingPolicy sniffContentEncoding : bitWidthOfContentEncodingSniffingPolicy;
209 DataBufferingPolicy dataBufferingPolicy : bitWidthOfDataBufferingPolicy;
210 StoredCredentialsPolicy storedCredentialsPolicy : bitWidthOfStoredCredentialsPolicy;
211 SecurityCheckPolicy securityCheck : bitWidthOfSecurityCheckPolicy;
212 CertificateInfoPolicy certificateInfoPolicy : bitWidthOfCertificateInfoPolicy;
213 ContentSecurityPolicyImposition contentSecurityPolicyImposition : bitWidthOfContentSecurityPolicyImposition;
214 DefersLoadingPolicy defersLoadingPolicy : bitWidthOfDefersLoadingPolicy;
215 CachingPolicy cachingPolicy : bitWidthOfCachingPolicy;
216 SameOriginDataURLFlag sameOriginDataURLFlag : bitWidthOfSameOriginDataURLFlag;
217 InitiatorContext initiatorContext : bitWidthOfInitiatorContext;
218 ServiceWorkersMode serviceWorkersMode : bitWidthOfServiceWorkersMode;
219 ApplicationCacheMode applicationCacheMode : bitWidthOfApplicationCacheMode;

221 PreflightPolicy preflightPolicy : bitWidthOfPreflightPolicy;
222 LoadedFromOpaqueSource loadedFromOpaqueSource : bitWidthOfLoadedFromOpaqueSource;
223};
