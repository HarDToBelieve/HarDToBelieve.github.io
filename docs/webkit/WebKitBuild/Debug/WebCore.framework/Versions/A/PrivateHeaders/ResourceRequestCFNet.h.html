47#if HAVE(CFNETWORK_NSURLSESSION_CONNECTION_CACHE_LIMITS)
48inline ResourceLoadPriority toResourceLoadPriority(CFURLRequestPriority priority)
49{
50 switch (priority) {
51 case -1:
52 case 0:
53 return ResourceLoadPriority::VeryLow;
54 case 1:
55 return ResourceLoadPriority::Low;
56 case 2:
57 return ResourceLoadPriority::Medium;
58 case 3:
59 return ResourceLoadPriority::High;
60 case 4:
61 return ResourceLoadPriority::VeryHigh;
62 default:
63 ASSERT_NOT_REACHED();
64 return ResourceLoadPriority::Lowest;
65 }
66}

129inline RetainPtr<CFStringRef> httpHeaderValueUsingSuitableEncoding(HTTPHeaderMap::const_iterator::KeyValue header)
130{
131 if (header.keyAsHTTPHeaderName && *header.keyAsHTTPHeaderName == HTTPHeaderName::LastEventID && !header.value.isAllASCII()) {
132 auto utf8Value = header.value.utf8();
133 // Constructing a string with the UTF-8 bytes but claiming that it’s Latin-1 is the way to get CFNetwork to put those UTF-8 bytes on the wire.
134 return adoptCF(CFStringCreateWithBytes(nullptr, utf8Value.dataAsUInt8Ptr(), utf8Value.length(), kCFStringEncodingISOLatin1, false));
135 }
136 return header.value.createCFString();
137}
