26#pragma once
27
28#include "ResourceResponseBase.h"
29#include <wtf/RetainPtr.h>
30
31#if USE(CFURLCONNECTION)
32#include <pal/spi/win/CFNetworkSPIWin.h>
33#endif
34
35OBJC_CLASS NSURLResponse;
36
37namespace WebCore {
38
39class ResourceResponse : public ResourceResponseBase {
40public:
41 ResourceResponse()
42 {
43 m_initLevel = AllFields;
44 }
45
46#if USE(CFURLCONNECTION)
47 ResourceResponse(CFURLResponseRef cfResponse)
48 : m_cfResponse(cfResponse)
49 {
50 m_initLevel = Uninitialized;
51 m_isNull = !cfResponse;
52 }
53#else
54 ResourceResponse(NSURLResponse *nsResponse)
55 : m_nsResponse(nsResponse)
56 {
57 m_initLevel = Uninitialized;
58 m_isNull = !nsResponse;
59 }
60#endif
61
62 ResourceResponse(const URL& url, const String& mimeType, long long expectedLength, const String& textEncodingName)
63 : ResourceResponseBase(url, mimeType, expectedLength, textEncodingName)
64 {
65 m_initLevel = AllFields;
66 }
67
68#if PLATFORM(COCOA)
69 WEBCORE_EXPORT void disableLazyInitialization();
70#endif
71
72 unsigned memoryUsage() const
73 {
74 // FIXME: Find some programmatic lighweight way to calculate ResourceResponse and associated classes.
75 // This is a rough estimate of resource overhead based on stats collected from memory usage tests.
76 return 3800;
77 /* 1280 * 2 + // average size of ResourceResponse. Doubled to account for the WebCore copy and the CF copy.
78 // Mostly due to the size of the hash maps, the Header Map strings and the URL.
79 256 * 2 // Overhead from ResourceRequest, doubled to account for WebCore copy and CF copy.
80 // Mostly due to the URL and Header Map.
81 */
82 }
83
84#if USE(CFURLCONNECTION)
85 WEBCORE_EXPORT CFURLResponseRef cfURLResponse() const;
86#endif
87#if PLATFORM(COCOA)
88 WEBCORE_EXPORT NSURLResponse *nsURLResponse() const;
89#endif
90
91#if USE(QUICK_LOOK)
92 bool isQuickLook() const { return m_isQuickLook; }
93 void setIsQuickLook(bool isQuickLook) { m_isQuickLook = isQuickLook; }
94#endif
95
96private:
97 friend class ResourceResponseBase;
98
99 void platformLazyInit(InitLevel);
100 String platformSuggestedFilename() const;
101 CertificateInfo platformCertificateInfo() const;
102
103#if PLATFORM(COCOA)
104 void initNSURLResponse() const;
105#endif
106
107 static bool platformCompare(const ResourceResponse& a, const ResourceResponse& b);
108
109#if USE(QUICK_LOOK)
110 bool m_isQuickLook { false };
111#endif
112
113#if USE(CFURLCONNECTION)
114 mutable RetainPtr<CFURLResponseRef> m_cfResponse;
115#endif
116#if PLATFORM(COCOA)
117 mutable RetainPtr<NSURLResponse> m_nsResponse;
118#endif
119};
120
121} // namespace WebCore
