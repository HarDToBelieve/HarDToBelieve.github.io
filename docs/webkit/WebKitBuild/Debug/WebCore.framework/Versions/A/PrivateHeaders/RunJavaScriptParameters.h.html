26#pragma once
27
28#include <wtf/HashMap.h>
29#include <wtf/URL.h>
30#include <wtf/Vector.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35enum class RunAsAsyncFunction : bool { No, Yes };
36enum class ForceUserGesture : bool { No, Yes };
37
38using ArgumentWireBytesMap = HashMap<String, Vector<uint8_t>>;
39
40struct RunJavaScriptParameters {
41 RunJavaScriptParameters(String&& source, URL&& sourceURL, RunAsAsyncFunction runAsAsyncFunction, std::optional<ArgumentWireBytesMap>&& arguments, ForceUserGesture forceUserGesture)
42 : source(WTFMove(source))
43 , sourceURL(WTFMove(sourceURL))
44 , runAsAsyncFunction(runAsAsyncFunction)
45 , arguments(WTFMove(arguments))
46 , forceUserGesture(forceUserGesture)
47 {
48 }
49
50 RunJavaScriptParameters(const String& source, URL&& sourceURL, bool runAsAsyncFunction, std::optional<ArgumentWireBytesMap>&& arguments, bool forceUserGesture)
51 : source(source)
52 , sourceURL(WTFMove(sourceURL))
53 , runAsAsyncFunction(runAsAsyncFunction ? RunAsAsyncFunction::Yes : RunAsAsyncFunction::No)
54 , arguments(WTFMove(arguments))
55 , forceUserGesture(forceUserGesture ? ForceUserGesture::Yes : ForceUserGesture::No)
56 {
57 }
58
59 RunJavaScriptParameters(String&& source, URL&& sourceURL, bool runAsAsyncFunction, std::optional<ArgumentWireBytesMap>&& arguments, bool forceUserGesture)
60 : source(WTFMove(source))
61 , sourceURL(WTFMove(sourceURL))
62 , runAsAsyncFunction(runAsAsyncFunction ? RunAsAsyncFunction::Yes : RunAsAsyncFunction::No)
63 , arguments(WTFMove(arguments))
64 , forceUserGesture(forceUserGesture ? ForceUserGesture::Yes : ForceUserGesture::No)
65 {
66 }
67
68 String source;
69 URL sourceURL;
70 RunAsAsyncFunction runAsAsyncFunction;
71 std::optional<ArgumentWireBytesMap> arguments;
72 ForceUserGesture forceUserGesture;
73
74 template<typename Encoder> void encode(Encoder& encoder) const
75 {
76 encoder << source << sourceURL << runAsAsyncFunction << arguments << forceUserGesture;
77 }
78
79 template<typename Decoder> static std::optional<RunJavaScriptParameters> decode(Decoder& decoder)
80 {
81 String source;
82 if (!decoder.decode(source))
83 return std::nullopt;
84
85 URL sourceURL;
86 if (!decoder.decode(sourceURL))
87 return std::nullopt;
88
89 RunAsAsyncFunction runAsAsyncFunction;
90 if (!decoder.decode(runAsAsyncFunction))
91 return std::nullopt;
92
93 std::optional<ArgumentWireBytesMap> arguments;
94 if (!decoder.decode(arguments))
95 return std::nullopt;
96
97 ForceUserGesture forceUserGesture;
98 if (!decoder.decode(forceUserGesture))
99 return std::nullopt;
100
101 return { RunJavaScriptParameters { WTFMove(source), WTFMove(sourceURL), runAsAsyncFunction, WTFMove(arguments), forceUserGesture } };
102 }
103};
104
105} // namespace WebCore
