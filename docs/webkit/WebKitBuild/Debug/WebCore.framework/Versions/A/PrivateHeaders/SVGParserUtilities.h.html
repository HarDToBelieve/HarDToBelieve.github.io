59// SVG allows several different whitespace characters:
60// http://www.w3.org/TR/SVG/paths.html#PathDataBNF
61template<typename CharacterType> constexpr bool isSVGSpace(CharacterType c)
62{
63 return c == ' ' || c == '\t' || c == '\n' || c == '\r';
64}
65
66template<typename CharacterType> constexpr bool isSVGSpaceOrComma(CharacterType c)
67{
68 return isSVGSpace(c) || c == ',';
69}
70
71template<typename CharacterType> constexpr bool skipOptionalSVGSpaces(const CharacterType*& ptr, const CharacterType* end)
72{
73 skipWhile<isSVGSpace>(ptr, end);
74 return ptr < end;
75}
76
77template<typename CharacterType> constexpr bool skipOptionalSVGSpaces(StringParsingBuffer<CharacterType>& characters)
78{
79 skipWhile<isSVGSpace>(characters);
80 return characters.hasCharactersRemaining();
81}
82
83template<typename CharacterType> constexpr bool skipOptionalSVGSpacesOrDelimiter(const CharacterType*& ptr, const CharacterType* end, char delimiter = ',')
84{
85 if (ptr < end && !isSVGSpace(*ptr) && *ptr != delimiter)
86 return false;
87 if (skipOptionalSVGSpaces(ptr, end)) {
88 if (ptr < end && *ptr == delimiter) {
89 ptr++;
90 skipOptionalSVGSpaces(ptr, end);
91 }
92 }
93 return ptr < end;
94}
95
96template<typename CharacterType> constexpr bool skipOptionalSVGSpacesOrDelimiter(StringParsingBuffer<CharacterType>& characters, char delimiter = ',')
97{
98 if (characters.hasCharactersRemaining() && !isSVGSpace(*characters) && *characters != delimiter)
99 return false;
100 if (skipOptionalSVGSpaces(characters)) {
101 if (characters.hasCharactersRemaining() && *characters == delimiter) {
102 characters++;
103 skipOptionalSVGSpaces(characters);
104 }
105 }
106 return characters.hasCharactersRemaining();
107}
