21#pragma once
22
23#include "SVGPropertyOwner.h"
24#include <wtf/RefCounted.h>
25#include <wtf/text/WTFString.h>
26
27namespace WebCore {
28
29enum class SVGPropertyAccess : uint8_t { ReadWrite, ReadOnly };
30enum class SVGPropertyState : uint8_t { Clean, Dirty };
31
32class SVGProperty : public RefCounted<SVGProperty> {
33public:
34 virtual ~SVGProperty() = default;
35
36 // Managing the relationship with the owner.
37 bool isAttached() const { return m_owner; }
38 virtual void attach(SVGPropertyOwner* owner, SVGPropertyAccess access)
39 {
40 ASSERT(!m_owner);
41 ASSERT(m_state == SVGPropertyState::Clean);
42 m_owner = owner;
43 m_access = access;
44 }
45
46 virtual void detach()
47 {
48 m_owner = nullptr;
49 m_access = SVGPropertyAccess::ReadWrite;
50 m_state = SVGPropertyState::Clean;
51 }
52
53 void reattach(SVGPropertyOwner* owner, SVGPropertyAccess access)
54 {
55 ASSERT_UNUSED(owner, owner == m_owner);
56 m_access = access;
57 m_state = SVGPropertyState::Clean;
58 }
59
60 const SVGElement* contextElement() const
61 {
62 if (!m_owner)
63 return nullptr;
64 return m_owner->attributeContextElement();
65 }
66
67 void commitChange()
68 {
69 if (!m_owner)
70 return;
71 m_owner->commitPropertyChange(this);
72 }
73
74 // DOM access.
75 SVGPropertyAccess access() const { return m_access; }
76 bool isReadOnly() const { return m_access == SVGPropertyAccess::ReadOnly; }
77
78 // Synchronizing the SVG attribute and its reflection here.
79 bool isDirty() const { return m_state == SVGPropertyState::Dirty; }
80 void setDirty() { m_state = SVGPropertyState::Dirty; }
81 std::optional<String> synchronize()
82 {
83 if (m_state == SVGPropertyState::Clean)
84 return std::nullopt;
85 m_state = SVGPropertyState::Clean;
86 return valueAsString();
87 }
88
89 // This is used when calling setAttribute().
90 virtual String valueAsString() const { return emptyString(); }
91
92 // Visual Studio doesn't seem to see these private constructors from subclasses.
93 // FIXME: See what it takes to remove this hack.
94#if !COMPILER(MSVC)
95protected:
96#endif
97 SVGProperty(SVGPropertyOwner* owner = nullptr, SVGPropertyAccess access = SVGPropertyAccess::ReadWrite)
98 : m_owner(owner)
99 , m_access(access)
100 {
101 }
102
103 SVGPropertyOwner* m_owner { nullptr };
104 SVGPropertyAccess m_access { SVGPropertyAccess::ReadWrite };
105 SVGPropertyState m_state { SVGPropertyState::Clean };
106};
107
108} // namespace WebCore
