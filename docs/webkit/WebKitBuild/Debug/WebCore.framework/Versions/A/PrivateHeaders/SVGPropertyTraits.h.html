32namespace WebCore {
33
34template<typename PropertyType>
35struct SVGPropertyTraits { };
36
37template<>
38struct SVGPropertyTraits<bool> {
39 static bool initialValue() { return false; }
40 static bool fromString(const String& string) { return string == "true"; }
41 static std::optional<bool> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return initialValue(); }
42 static String toString(bool type) { return type ? "true" : "false"; }
43};
44
45template<>
46struct SVGPropertyTraits<Color> {
47 static Color initialValue() { return Color(); }
48 static Color fromString(const String& string) { return CSSParser::parseColor(string.stripWhiteSpace()); }
49 static std::optional<Color> parse(const QualifiedName&, const String& string)
50 {
51 Color color = CSSParser::parseColor(string.stripWhiteSpace());
52 if (!color.isValid())
53 return std::nullopt;
54 return color;
55 }
56 static String toString(const Color& type) { return serializationForHTML(type); }
57};
58
59template<>
60struct SVGPropertyTraits<unsigned> {
61 static unsigned initialValue() { return 0; }
62 static std::optional<unsigned> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return initialValue(); }
63 static String toString(unsigned type) { return String::number(type); }
64};
65
66template<>
67struct SVGPropertyTraits<int> {
68 static int initialValue() { return 0; }
69 static int fromString(const String&);
70 static std::optional<int> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return initialValue(); }
71 static String toString(int type) { return String::number(type); }
72};
73
74template<>
75struct SVGPropertyTraits<std::pair<int, int>> {
76 static std::pair<int, int> initialValue() { return { }; }
77 static std::pair<int, int> fromString(const String& string)
78 {
79 auto result = parseNumberOptionalNumber(string);
80 if (!result)
81 return { };
82 return std::make_pair(static_cast<int>(std::round(result->first)), static_cast<int>(std::round(result->second)));
83 }
84 static std::optional<std::pair<int, int>> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return initialValue(); }
85 static String toString(std::pair<int, int>) { ASSERT_NOT_REACHED(); return emptyString(); }
86};
87
88template<>
89struct SVGPropertyTraits<float> {
90 static float initialValue() { return 0; }
91 static float fromString(const String& string)
92 {
93 return parseNumber(string).value_or(0);
94 }
95 static std::optional<float> parse(const QualifiedName&, const String& string)
96 {
97 return parseNumber(string);
98 }
99 static String toString(float type) { return String::number(type); }
100};
101
102template<>
103struct SVGPropertyTraits<std::pair<float, float>> {
104 static std::pair<float, float> initialValue() { return { }; }
105 static std::pair<float, float> fromString(const String& string)
106 {
107 return parseNumberOptionalNumber(string).value_or(std::pair<float, float> { });
108 }
109 static std::optional<std::pair<float, float>> parse(const QualifiedName&, const String&) { ASSERT_NOT_REACHED(); return initialValue(); }
110 static String toString(std::pair<float, float>) { ASSERT_NOT_REACHED(); return emptyString(); }
111};
112
113template<>
114struct SVGPropertyTraits<FloatPoint> {
115 static FloatPoint initialValue() { return FloatPoint(); }
116 static FloatPoint fromString(const String& string)
117 {
118 return parsePoint(string).value_or(FloatPoint { });
119 }
120 static std::optional<FloatPoint> parse(const QualifiedName&, const String& string)
121 {
122 return parsePoint(string);
123 }
124 static String toString(const FloatPoint& type)
125 {
126 return makeString(type.x(), ' ', type.y());
127 }
128};
129
130template<>
131struct SVGPropertyTraits<FloatRect> {
132 static FloatRect initialValue() { return FloatRect(); }
133 static FloatRect fromString(const String& string)
134 {
135 return parseRect(string).value_or(FloatRect { });
136 }
137 static std::optional<FloatRect> parse(const QualifiedName&, const String& string)
138 {
139 return parseRect(string);
140 }
141 static String toString(const FloatRect& type)
142 {
143 return makeString(type.x(), ' ', type.y(), ' ', type.width(), ' ', type.height());
144 }
145};
146
147template<>
148struct SVGPropertyTraits<String> {
149 static String initialValue() { return String(); }
150 static String fromString(const String& string) { return string; }
151 static std::optional<String> parse(const QualifiedName&, const String& string) { return string; }
152 static String toString(const String& string) { return string; }
153};

155template<typename EnumType>
156struct SVGIDLEnumLimits {
157 // Specialize this function for a particular enumeration to limit the values that are exposed through the DOM.
158 static unsigned highestExposedEnumValue() { return SVGPropertyTraits<EnumType>::highestEnumValue(); }
159};
