36namespace WebCore {
37
38struct ScreenData {
39 FloatRect screenAvailableRect;
40 FloatRect screenRect;
41 DestinationColorSpace colorSpace { DestinationColorSpace::SRGB() };
42 int screenDepth { 0 };
43 int screenDepthPerComponent { 0 };
44 bool screenSupportsExtendedColor { false };
45 bool screenHasInvertedColors { false };
46 bool screenSupportsHighDynamicRange { false };
47#if PLATFORM(MAC)
48 bool screenIsMonochrome { false };
49 uint32_t displayMask { 0 };
50 IORegistryGPUID gpuID { 0 };
51 DynamicRangeMode preferredDynamicRangeMode { DynamicRangeMode::Standard };
52#endif
53
54#if PLATFORM(MAC) || PLATFORM(IOS_FAMILY)
55 float scaleFactor { 1 };
56#endif
57
58 template<class Encoder> void encode(Encoder&) const;
59 template<class Decoder> static std::optional<ScreenData> decode(Decoder&);
60};
61
62using ScreenDataMap = HashMap<PlatformDisplayID, ScreenData>;
63
64struct ScreenProperties {
65 PlatformDisplayID primaryDisplayID { 0 };
66 ScreenDataMap screenDataMap;
67
68 template<class Encoder> void encode(Encoder&) const;
69 template<class Decoder> static std::optional<ScreenProperties> decode(Decoder&);
70};
71
72template<class Encoder>
73void ScreenProperties::encode(Encoder& encoder) const
74{
75 encoder << primaryDisplayID << screenDataMap;
76}
77
78template<class Decoder>
79std::optional<ScreenProperties> ScreenProperties::decode(Decoder& decoder)
80{
81 std::optional<PlatformDisplayID> primaryDisplayID;
82 decoder >> primaryDisplayID;
83 if (!primaryDisplayID)
84 return std::nullopt;
85
86 std::optional<ScreenDataMap> screenDataMap;
87 decoder >> screenDataMap;
88 if (!screenDataMap)
89 return std::nullopt;
90
91 return { { *primaryDisplayID, WTFMove(*screenDataMap) } };
92}
93
94template<class Encoder>
95void ScreenData::encode(Encoder& encoder) const
96{
97 encoder << screenAvailableRect << screenRect << colorSpace << screenDepth << screenDepthPerComponent << screenSupportsExtendedColor << screenHasInvertedColors << screenSupportsHighDynamicRange;
98
99#if PLATFORM(MAC)
100 encoder << screenIsMonochrome << displayMask << gpuID << preferredDynamicRangeMode;
101#endif
102
103#if PLATFORM(MAC) || PLATFORM(IOS_FAMILY)
104 encoder << scaleFactor;
105#endif
106}
107
108template<class Decoder>
109std::optional<ScreenData> ScreenData::decode(Decoder& decoder)
110{
111 std::optional<FloatRect> screenAvailableRect;
112 decoder >> screenAvailableRect;
113 if (!screenAvailableRect)
114 return std::nullopt;
115
116 std::optional<FloatRect> screenRect;
117 decoder >> screenRect;
118 if (!screenRect)
119 return std::nullopt;
120
121 std::optional<DestinationColorSpace> screenColorSpace;
122 decoder >> screenColorSpace;
123 if (!screenColorSpace)
124 return std::nullopt;
125
126 std::optional<int> screenDepth;
127 decoder >> screenDepth;
128 if (!screenDepth)
129 return std::nullopt;
130
131 std::optional<int> screenDepthPerComponent;
132 decoder >> screenDepthPerComponent;
133 if (!screenDepthPerComponent)
134 return std::nullopt;
135
136 std::optional<bool> screenSupportsExtendedColor;
137 decoder >> screenSupportsExtendedColor;
138 if (!screenSupportsExtendedColor)
139 return std::nullopt;
140
141 std::optional<bool> screenHasInvertedColors;
142 decoder >> screenHasInvertedColors;
143 if (!screenHasInvertedColors)
144 return std::nullopt;
145
146 std::optional<bool> screenSupportsHighDynamicRange;
147 decoder >> screenSupportsHighDynamicRange;
148 if (!screenSupportsHighDynamicRange)
149 return std::nullopt;
150
151#if PLATFORM(MAC)
152 std::optional<bool> screenIsMonochrome;
153 decoder >> screenIsMonochrome;
154 if (!screenIsMonochrome)
155 return std::nullopt;
156
157 std::optional<uint32_t> displayMask;
158 decoder >> displayMask;
159 if (!displayMask)
160 return std::nullopt;
161
162 std::optional<IORegistryGPUID> gpuID;
163 decoder >> gpuID;
164 if (!gpuID)
165 return std::nullopt;
166
167 std::optional<DynamicRangeMode> preferredDynamicRangeMode;
168 decoder >> preferredDynamicRangeMode;
169 if (!preferredDynamicRangeMode)
170 return std::nullopt;
171#endif
172
173#if PLATFORM(MAC) || PLATFORM(IOS_FAMILY)
174 std::optional<float> scaleFactor;
175 decoder >> scaleFactor;
176 if (!scaleFactor)
177 return std::nullopt;
178#endif
179
180 return { {
181 WTFMove(*screenAvailableRect),
182 WTFMove(*screenRect),
183 WTFMove(*screenColorSpace),
184 WTFMove(*screenDepth),
185 WTFMove(*screenDepthPerComponent),
186 WTFMove(*screenSupportsExtendedColor),
187 WTFMove(*screenHasInvertedColors),
188 WTFMove(*screenSupportsHighDynamicRange),
189#if PLATFORM(MAC)
190 WTFMove(*screenIsMonochrome),
191 WTFMove(*displayMask),
192 WTFMove(*gpuID),
193 WTFMove(*preferredDynamicRangeMode),
194#endif
195#if PLATFORM(MAC) || PLATFORM(IOS_FAMILY)
196 WTFMove(*scaleFactor),
197#endif
198 } };
199}
