34class ContainerNode;

40class ScriptElement {
41public:
42 virtual ~ScriptElement() = default;
43
44 Element& element() { return m_element; }
45 const Element& element() const { return m_element; }
46
47 enum LegacyTypeSupport { DisallowLegacyTypeInTypeAttribute, AllowLegacyTypeInTypeAttribute };
48 bool prepareScript(const TextPosition& scriptStartPosition = TextPosition(), LegacyTypeSupport = DisallowLegacyTypeInTypeAttribute);
49
50 String scriptCharset() const { return m_characterEncoding; }
51 WEBCORE_EXPORT String scriptContent() const;
52 void executeClassicScript(const ScriptSourceCode&);
53 void executeModuleScript(LoadableModuleScript&);
54
55 void executePendingScript(PendingScript&);
56
57 virtual bool hasAsyncAttribute() const = 0;
58 virtual bool hasDeferAttribute() const = 0;
59 virtual bool hasSourceAttribute() const = 0;
60 virtual bool hasNoModuleAttribute() const = 0;
61
62 // XML parser calls these
63 virtual void dispatchLoadEvent() = 0;
64 virtual void dispatchErrorEvent();
65
66 bool haveFiredLoadEvent() const { return m_haveFiredLoad; }
67 bool errorOccurred() const { return m_errorOccurred; }
68 bool willBeParserExecuted() const { return m_willBeParserExecuted; }
69 bool readyToBeParserExecuted() const { return m_readyToBeParserExecuted; }
70 bool willExecuteWhenDocumentFinishedParsing() const { return m_willExecuteWhenDocumentFinishedParsing; }
71 bool willExecuteInOrder() const { return m_willExecuteInOrder; }
72 LoadableScript* loadableScript() { return m_loadableScript.get(); }
73
74 // https://html.spec.whatwg.org/multipage/scripting.html#concept-script-type
75 enum class ScriptType { Classic, Module };
76 ScriptType scriptType() const { return m_isModuleScript ? ScriptType::Module : ScriptType::Classic; }
77
78 void ref();
79 void deref();
80
81protected:
82 ScriptElement(Element&, bool createdByParser, bool isEvaluated);
83
84 void setHaveFiredLoadEvent(bool haveFiredLoad) { m_haveFiredLoad = haveFiredLoad; }
85 void setErrorOccurred(bool errorOccurred) { m_errorOccurred = errorOccurred; }
86 bool isParserInserted() const { return m_parserInserted; }
87 bool alreadyStarted() const { return m_alreadyStarted; }
88 bool forceAsync() const { return m_forceAsync; }
89
90 // Helper functions used by our parent classes.
91 Node::InsertedIntoAncestorResult insertedIntoAncestor(Node::InsertionType insertionType, ContainerNode&) const
92 {
93 if (insertionType.connectedToDocument && !m_parserInserted)
94 return Node::InsertedIntoAncestorResult::NeedsPostInsertionCallback;
95 return Node::InsertedIntoAncestorResult::Done;
96 }
97
98 void didFinishInsertingNode();
99 void childrenChanged(const ContainerNode::ChildChange&);
100 void handleSourceAttribute(const String& sourceURL);
101 void handleAsyncAttribute();
102
103private:
104 void executeScriptAndDispatchEvent(LoadableScript&);
105
106 std::optional<ScriptType> determineScriptType(LegacyTypeSupport) const;
107 bool ignoresLoadRequest() const;
108 bool isScriptForEventSupported() const;
109 void dispatchLoadEventRespectingUserGestureIndicator();
110
111 bool requestClassicScript(const String& sourceURL);
112 bool requestModuleScript(const TextPosition& scriptStartPosition);
113
114 virtual String sourceAttributeValue() const = 0;
115 virtual String charsetAttributeValue() const = 0;
116 virtual String typeAttributeValue() const = 0;
117 virtual String languageAttributeValue() const = 0;
118 virtual String forAttributeValue() const = 0;
119 virtual String eventAttributeValue() const = 0;
120 virtual ReferrerPolicy referrerPolicy() const = 0;
121
122 Element& m_element;
123 WTF::OrdinalNumber m_startLineNumber;
124 bool m_parserInserted : 1;
125 bool m_isExternalScript : 1;
126 bool m_alreadyStarted : 1;
127 bool m_haveFiredLoad : 1;
128 bool m_errorOccurred : 1;
129 bool m_willBeParserExecuted : 1; // Same as "The parser will handle executing the script."
130 bool m_readyToBeParserExecuted : 1;
131 bool m_willExecuteWhenDocumentFinishedParsing : 1;
132 bool m_forceAsync : 1;
133 bool m_willExecuteInOrder : 1;
134 bool m_isModuleScript : 1;
135 String m_characterEncoding;
136 String m_fallbackCharacterEncoding;
137 RefPtr<LoadableScript> m_loadableScript;
138
139 MonotonicTime m_creationTime;
140 RefPtr<UserGestureToken> m_userGestureToken;
141};
