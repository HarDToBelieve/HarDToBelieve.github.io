178 class Task {
179 WTF_MAKE_FAST_ALLOCATED;
180 public:
181 enum CleanupTaskTag { CleanupTask };
182
183 template<typename T, typename = typename std::enable_if<!std::is_base_of<Task, T>::value && std::is_convertible<T, WTF::Function<void (ScriptExecutionContext&)>>::value>::type>
184 Task(T task)
185 : m_task(WTFMove(task))
186 , m_isCleanupTask(false)
187 {
188 }
189
190 Task(WTF::Function<void ()>&& task)
191 : m_task([task = WTFMove(task)](ScriptExecutionContext&) { task(); })
192 , m_isCleanupTask(false)
193 {
194 }
195
196 template<typename T, typename = typename std::enable_if<std::is_convertible<T, WTF::Function<void (ScriptExecutionContext&)>>::value>::type>
197 Task(CleanupTaskTag, T task)
198 : m_task(WTFMove(task))
199 , m_isCleanupTask(true)
200 {
201 }
202
203 void performTask(ScriptExecutionContext& context) { m_task(context); }
204 bool isCleanupTask() const { return m_isCleanupTask; }
205
206 protected:
207 WTF::Function<void (ScriptExecutionContext&)> m_task;
208 bool m_isCleanupTask;
209 };
