257 // scrollOffset() anchors its (0,0) point at the ScrollableArea's origin. When the Page has a
258 // header, the header is positioned at (0,0), ABOVE the start of the Document. So when a page with
259 // a header is pinned to the top, the scrollOffset() is (0,0), but the Document is actually at
260 // (0, -headerHeight()). documentScrollPositionRelativeToScrollableAreaOrigin() will return this
261 // version of the offset, which tracks the top of Document relative to where scrolling was achored.
262 ScrollPosition documentScrollPositionRelativeToScrollableAreaOrigin() const;
263
264 // scrollPostion() anchors its (0,0) point at the ScrollableArea's origin. The top of the scrolling
265 // layer does not represent the top of the view when there is a topContentInset. Additionally, as
266 // detailed above, the origin of the scrolling layer also does not necessarily correspond with the
267 // top of the document anyway, since there could also be header. documentScrollPositionRelativeToViewOrigin()
268 // will return a version of the current scroll offset which tracks the top of the Document
269 // relative to the very top of the view.
270 WEBCORE_EXPORT ScrollPosition documentScrollPositionRelativeToViewOrigin() const;

351 IntPoint convertChildToSelf(const Widget* child, const IntPoint& point) const
352 {
353 IntPoint newPoint = point;
354 if (!isScrollViewScrollbar(child))
355 newPoint = point - toIntSize(scrollPosition());
356 newPoint.moveBy(child->location());
357 return newPoint;
358 }
359
360 FloatPoint convertChildToSelf(const Widget* child, const FloatPoint& point) const
361 {
362 FloatPoint newPoint = point;
363 if (!isScrollViewScrollbar(child))
364 newPoint -= toFloatSize(scrollPosition());
365 newPoint.moveBy(child->location());
366 return newPoint;
367 }
368
369 IntPoint convertSelfToChild(const Widget* child, const IntPoint& point) const
370 {
371 IntPoint newPoint = point;
372 if (!isScrollViewScrollbar(child))
373 newPoint = point + toIntSize(scrollPosition());
374 newPoint.moveBy(-child->location());
375 return newPoint;
376 }
