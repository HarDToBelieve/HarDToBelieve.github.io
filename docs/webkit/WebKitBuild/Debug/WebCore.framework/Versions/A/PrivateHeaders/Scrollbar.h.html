26#pragma once
27
28#include "ScrollTypes.h"
29#include "Timer.h"
30#include "Widget.h"

32namespace WebCore {
33
34class GraphicsContext;
35class IntRect;
36class PlatformMouseEvent;
37class ScrollableArea;
38class ScrollbarTheme;
39
40class Scrollbar : public Widget {
41public:
42 // Must be implemented by platforms that can't simply use the Scrollbar base class. Right now the only platform that is not using the base class is GTK.
43 WEBCORE_EXPORT static Ref<Scrollbar> createNativeScrollbar(ScrollableArea&, ScrollbarOrientation, ScrollbarControlSize);
44
45 virtual ~Scrollbar();
46
47 // Called by the ScrollableArea when the scroll offset changes.
48 void offsetDidChange();
49
50 static int pixelsPerLineStep() { return 40; }
51 static float minFractionToStepWhenPaging() { return 0.8; }
52 WEBCORE_EXPORT static int maxOverlapBetweenPages();
53 static int pageStep(int viewWidthOrHeight, int contentWidthOrHeight) { return std::max(std::max<int>(lroundf(viewWidthOrHeight * Scrollbar::minFractionToStepWhenPaging()), lroundf(contentWidthOrHeight - Scrollbar::maxOverlapBetweenPages())), 1); }
54 static int pageStep(int viewWidthOrHeight) { return pageStep(viewWidthOrHeight, viewWidthOrHeight); }
55 static float pageStepDelta(int widthOrHeight) { return std::max(std::max(static_cast<float>(widthOrHeight) * Scrollbar::minFractionToStepWhenPaging(), static_cast<float>(widthOrHeight) - Scrollbar::maxOverlapBetweenPages()), 1.0f); }
56
57 ScrollableArea& scrollableArea() const { return m_scrollableArea; }
58
59 bool isCustomScrollbar() const { return m_isCustomScrollbar; }
60 ScrollbarOrientation orientation() const { return m_orientation; }
61
62 int value() const { return lroundf(m_currentPos); }
63 float currentPos() const { return m_currentPos; }
64 int pressedPos() const { return m_pressedPos; }
65 int visibleSize() const { return m_visibleSize; }
66 int totalSize() const { return m_totalSize; }
67 int maximum() const { return m_totalSize - m_visibleSize; }
68 ScrollbarControlSize controlSize() const { return m_controlSize; }
69
70 int occupiedWidth() const;
71 int occupiedHeight() const;
72
73 int lineStep() const { return m_lineStep; }
74 int pageStep() const { return m_pageStep; }
75 float pixelStep() const { return m_pixelStep; }
76
77 ScrollbarPart pressedPart() const { return m_pressedPart; }
78 ScrollbarPart hoveredPart() const { return m_hoveredPart; }
79 virtual void setHoveredPart(ScrollbarPart);
80 virtual void setPressedPart(ScrollbarPart);
81
82 WEBCORE_EXPORT void setSteps(int lineStep, int pageStep, int pixelsPerStep = 1);
83 WEBCORE_EXPORT void setProportion(int visibleSize, int totalSize);
84 void setPressedPos(int p) { m_pressedPos = p; }
85
86 void paint(GraphicsContext&, const IntRect& damageRect, Widget::SecurityOriginPaintPolicy = SecurityOriginPaintPolicy::AnyOrigin, EventRegionContext* = nullptr) override;
87
88 bool enabled() const { return m_enabled; }
89 virtual void setEnabled(bool);
90
91 virtual bool isOverlayScrollbar() const;
92 bool shouldParticipateInHitTesting();
93 virtual bool isHiddenByStyle() const { return false; }
94
95 bool isWindowActive() const;
96
97 // These methods are used for platform scrollbars to give :hover feedback. They will not get called
98 // when the mouse went down in a scrollbar, since it is assumed the scrollbar will start
99 // grabbing all events in that case anyway.
100#if !PLATFORM(IOS_FAMILY)
101 WEBCORE_EXPORT bool mouseMoved(const PlatformMouseEvent&);
102#endif
103 WEBCORE_EXPORT void mouseEntered();
104 WEBCORE_EXPORT bool mouseExited();
105
106 // Used by some platform scrollbars to know when they've been released from capture.
107 WEBCORE_EXPORT bool mouseUp(const PlatformMouseEvent&);
108
109 WEBCORE_EXPORT bool mouseDown(const PlatformMouseEvent&);
110
111 ScrollbarTheme& theme() const { return m_theme; }
112
113 void invalidateRect(const IntRect&) override;
114
115 bool suppressInvalidation() const { return m_suppressInvalidation; }
116 void setSuppressInvalidation(bool s) { m_suppressInvalidation = s; }
117
118 virtual void styleChanged() { }
119
120 IntRect convertToContainingView(const IntRect&) const override;
121 IntRect convertFromContainingView(const IntRect&) const override;
122
123 IntPoint convertToContainingView(const IntPoint&) const override;
124 IntPoint convertFromContainingView(const IntPoint&) const override;
125
126 void moveThumb(int pos, bool draggingDocument = false);
127
128#if !PLATFORM(COCOA)
129 float opacity() const { return m_opacity; }
130 void setOpacity(float opacity) { m_opacity = opacity; }
131#endif
132
133 bool supportsUpdateOnSecondaryThread() const;
134
135protected:
136 Scrollbar(ScrollableArea&, ScrollbarOrientation, ScrollbarControlSize, ScrollbarTheme* = nullptr, bool isCustomScrollbar = false);
137
138 void updateThumb();
139 virtual void updateThumbPosition();
140 virtual void updateThumbProportion();
141
142 void autoscrollTimerFired();
143 void startTimerIfNeeded(Seconds delay);
144 void stopTimerIfNeeded();
145 void autoscrollPressedPart(Seconds delay);
146 ScrollDirection pressedPartScrollDirection();
147 ScrollGranularity pressedPartScrollGranularity();
148
149 ScrollableArea& m_scrollableArea;
150 ScrollbarOrientation m_orientation;
151 ScrollbarControlSize m_controlSize;
152 ScrollbarTheme& m_theme;
153
154 int m_visibleSize { 0 };
155 int m_totalSize { 0 };
156 float m_currentPos { 0 };
157 float m_dragOrigin { 0 };
158 int m_lineStep { 0 };
159 int m_pageStep { 0 };
160 float m_pixelStep { 1 };
161
162 ScrollbarPart m_hoveredPart { NoPart };
163 ScrollbarPart m_pressedPart { NoPart };
164 int m_pressedPos { 0 };
165 bool m_draggingDocument { false };
166 int m_documentDragPos { 0 };
167
168 bool m_enabled { true };
169 bool m_isCustomScrollbar { false };
170
171 Timer m_scrollTimer;
172
173 bool m_suppressInvalidation { false };
174
175#if !PLATFORM(COCOA)
176 float m_opacity { 1 };
177#endif
178
179private:
180 bool isScrollbar() const override { return true; }
181};
182
183} // namespace WebCore
