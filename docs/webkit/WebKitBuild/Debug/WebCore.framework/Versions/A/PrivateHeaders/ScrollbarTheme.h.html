26#ifndef ScrollbarTheme_h
27#define ScrollbarTheme_h
28
29#include "GraphicsContext.h"
30#include "IntRect.h"
31#include "ScrollTypes.h"
32
33namespace WebCore {
34
35class PlatformMouseEvent;
36class Scrollbar;
37class ScrollView;
38
39#if ENABLE(RUBBER_BANDING)
40class GraphicsLayer;
41#endif
42
43class ScrollbarTheme {
44 WTF_MAKE_NONCOPYABLE(ScrollbarTheme); WTF_MAKE_FAST_ALLOCATED;
45public:
46 ScrollbarTheme() = default;
47 virtual ~ScrollbarTheme() {};
48
49 virtual void updateEnabledState(Scrollbar&) { }
50
51 virtual bool paint(Scrollbar&, GraphicsContext&, const IntRect& /*damageRect*/) { return false; }
52 virtual ScrollbarPart hitTest(Scrollbar&, const IntPoint&) { return NoPart; }
53
54 virtual int scrollbarThickness(ScrollbarControlSize = ScrollbarControlSize::Regular, ScrollbarExpansionState = ScrollbarExpansionState::Expanded) { return 0; }
55
56 virtual ScrollbarButtonsPlacement buttonsPlacement() const { return ScrollbarButtonsSingle; }
57
58 virtual bool supportsControlTints() const { return false; }
59 virtual bool usesOverlayScrollbars() const { return false; }
60 virtual void usesOverlayScrollbarsChanged() { }
61 virtual void updateScrollbarOverlayStyle(Scrollbar&) { }
62
63 virtual void themeChanged() {}
64
65 virtual bool invalidateOnMouseEnterExit() { return false; }
66
67 void invalidateParts(Scrollbar& scrollbar, ScrollbarControlPartMask mask)
68 {
69 if (mask & BackButtonStartPart)
70 invalidatePart(scrollbar, BackButtonStartPart);
71 if (mask & ForwardButtonStartPart)
72 invalidatePart(scrollbar, ForwardButtonStartPart);
73 if (mask & BackTrackPart)
74 invalidatePart(scrollbar, BackTrackPart);
75 if (mask & ThumbPart)
76 invalidatePart(scrollbar, ThumbPart);
77 if (mask & ForwardTrackPart)
78 invalidatePart(scrollbar, ForwardTrackPart);
79 if (mask & BackButtonEndPart)
80 invalidatePart(scrollbar, BackButtonEndPart);
81 if (mask & ForwardButtonEndPart)
82 invalidatePart(scrollbar, ForwardButtonEndPart);
83 }
84
85 virtual void invalidatePart(Scrollbar&, ScrollbarPart) { }
86
87 virtual void paintScrollCorner(GraphicsContext& context, const IntRect& cornerRect) { defaultPaintScrollCorner(context, cornerRect); }
88 static void defaultPaintScrollCorner(GraphicsContext& context, const IntRect& cornerRect) { context.fillRect(cornerRect, Color::white); }
89
90 virtual void paintTickmarks(GraphicsContext&, Scrollbar&, const IntRect&) { }
91 virtual void paintOverhangAreas(ScrollView&, GraphicsContext&, const IntRect&, const IntRect&, const IntRect&) { }
92
93#if ENABLE(RUBBER_BANDING)
94 virtual void setUpOverhangAreasLayerContents(GraphicsLayer*, const Color&) { }
95 virtual void setUpContentShadowLayer(GraphicsLayer*) { }
96#endif
97
98 virtual ScrollbarButtonPressAction handleMousePressEvent(Scrollbar&, const PlatformMouseEvent&, ScrollbarPart);
99 virtual bool shouldSnapBackToDragOrigin(Scrollbar&, const PlatformMouseEvent&) { return false; }
100 virtual bool shouldDragDocumentInsteadOfThumb(Scrollbar&, const PlatformMouseEvent&) { return false; }
101 virtual int thumbPosition(Scrollbar&) { return 0; } // The position of the thumb relative to the track.
102 virtual int thumbLength(Scrollbar&) { return 0; } // The length of the thumb along the axis of the scrollbar.
103 virtual int trackPosition(Scrollbar&) { return 0; } // The position of the track relative to the scrollbar.
104 virtual int trackLength(Scrollbar&) { return 0; } // The length of the track along the axis of the scrollbar.
105
106 virtual int maxOverlapBetweenPages() { return std::numeric_limits<int>::max(); }
107
108 virtual Seconds initialAutoscrollTimerDelay() { return 250_ms; }
109 virtual Seconds autoscrollTimerDelay() { return 50_ms; }
110
111 virtual void registerScrollbar(Scrollbar&) { }
112 virtual void unregisterScrollbar(Scrollbar&) { }
113
114 virtual bool isMockTheme() const { return false; }
115
116 WEBCORE_EXPORT static ScrollbarTheme& theme();
117
118private:
119 static ScrollbarTheme& nativeTheme(); // Must be implemented to return the correct theme subclass.
120};
121
122}
123#endif
