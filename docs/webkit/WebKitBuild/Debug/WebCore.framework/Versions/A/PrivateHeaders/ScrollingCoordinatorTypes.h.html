70struct ScrollableAreaParameters {
71 ScrollElasticity horizontalScrollElasticity { ScrollElasticityNone };
72 ScrollElasticity verticalScrollElasticity { ScrollElasticityNone };
73
74 ScrollbarMode horizontalScrollbarMode { ScrollbarAuto };
75 ScrollbarMode verticalScrollbarMode { ScrollbarAuto };
76
77 bool allowsHorizontalScrolling { false };
78 bool allowsVerticalScrolling { false };
79
80 bool horizontalScrollbarHiddenByStyle { false };
81 bool verticalScrollbarHiddenByStyle { false };
82
83 bool useDarkAppearanceForScrollbars { false };
84
85 bool operator==(const ScrollableAreaParameters& other) const
86 {
87 return horizontalScrollElasticity == other.horizontalScrollElasticity
88 && verticalScrollElasticity == other.verticalScrollElasticity
89 && horizontalScrollbarMode == other.horizontalScrollbarMode
90 && verticalScrollbarMode == other.verticalScrollbarMode
91 && allowsHorizontalScrolling == other.allowsHorizontalScrolling
92 && allowsVerticalScrolling == other.allowsVerticalScrolling
93 && horizontalScrollbarHiddenByStyle == other.horizontalScrollbarHiddenByStyle
94 && verticalScrollbarHiddenByStyle == other.verticalScrollbarHiddenByStyle
95 && useDarkAppearanceForScrollbars == other.useDarkAppearanceForScrollbars;
96 }
97};
