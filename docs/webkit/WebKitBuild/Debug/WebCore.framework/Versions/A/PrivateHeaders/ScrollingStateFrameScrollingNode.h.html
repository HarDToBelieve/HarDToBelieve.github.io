52 WEBCORE_EXPORT void setEventTrackingRegions(const EventTrackingRegions&);

54 ScrollBehaviorForFixedElements scrollBehaviorForFixedElements() const { return m_behaviorForFixed; }
55 WEBCORE_EXPORT void setScrollBehaviorForFixedElements(ScrollBehaviorForFixedElements);

58 WEBCORE_EXPORT void setLayoutViewport(const FloatRect&);

60 FloatPoint minLayoutViewportOrigin() const { return m_minLayoutViewportOrigin; }
61 WEBCORE_EXPORT void setMinLayoutViewportOrigin(const FloatPoint&);
62
63 FloatPoint maxLayoutViewportOrigin() const { return m_maxLayoutViewportOrigin; }
64 WEBCORE_EXPORT void setMaxLayoutViewportOrigin(const FloatPoint&);
65
66 std::optional<FloatSize> overrideVisualViewportSize() const { return m_overrideVisualViewportSize; };
67 WEBCORE_EXPORT void setOverrideVisualViewportSize(std::optional<FloatSize>);
68
69 int headerHeight() const { return m_headerHeight; }
70 WEBCORE_EXPORT void setHeaderHeight(int);
71
72 int footerHeight() const { return m_footerHeight; }
73 WEBCORE_EXPORT void setFooterHeight(int);
74
75 float topContentInset() const { return m_topContentInset; }
76 WEBCORE_EXPORT void setTopContentInset(float);
77
78 const LayerRepresentation& rootContentsLayer() const { return m_rootContentsLayer; }
79 WEBCORE_EXPORT void setRootContentsLayer(const LayerRepresentation&);
80
81 // This is a layer moved in the opposite direction to scrolling, for example for background-attachment:fixed
82 const LayerRepresentation& counterScrollingLayer() const { return m_counterScrollingLayer; }
83 WEBCORE_EXPORT void setCounterScrollingLayer(const LayerRepresentation&);
84
85 // This is a clipping layer that will scroll with the page for all y-delta scroll values between 0
86 // and topContentInset(). Once the y-deltas get beyond the content inset point, this layer no longer
87 // needs to move. If the topContentInset() is 0, this layer does not need to move at all. This is
88 // only used on the Mac.
89 const LayerRepresentation& insetClipLayer() const { return m_insetClipLayer; }
90 WEBCORE_EXPORT void setInsetClipLayer(const LayerRepresentation&);
91
92 const LayerRepresentation& contentShadowLayer() const { return m_contentShadowLayer; }
93 WEBCORE_EXPORT void setContentShadowLayer(const LayerRepresentation&);
94
95 // The header and footer layers scroll vertically with the page, they should remain fixed when scrolling horizontally.
96 const LayerRepresentation& headerLayer() const { return m_headerLayer; }
97 WEBCORE_EXPORT void setHeaderLayer(const LayerRepresentation&);
98
99 // The header and footer layers scroll vertically with the page, they should remain fixed when scrolling horizontally.
100 const LayerRepresentation& footerLayer() const { return m_footerLayer; }
101 WEBCORE_EXPORT void setFooterLayer(const LayerRepresentation&);
102
103 // True when the visual viewport is smaller than the layout viewport, indicating that panning should be possible.
104 bool visualViewportIsSmallerThanLayoutViewport() const { return m_visualViewportIsSmallerThanLayoutViewport; }
105 WEBCORE_EXPORT void setVisualViewportIsSmallerThanLayoutViewport(bool);
106
107 // These are more like Settings, and should probably move to the Scrolling{State}Tree itself.
108 bool fixedElementsLayoutRelativeToFrame() const { return m_fixedElementsLayoutRelativeToFrame; }
109 WEBCORE_EXPORT void setFixedElementsLayoutRelativeToFrame(bool);

111 bool asyncFrameOrOverflowScrollingEnabled() const { return m_asyncFrameOrOverflowScrollingEnabled; }
112 void setAsyncFrameOrOverflowScrollingEnabled(bool);
113
114 bool scrollingPerformanceTestingEnabled() const { return m_scrollingPerformanceTestingEnabled; }
115 WEBCORE_EXPORT void setScrollingPerformanceTestingEnabled(bool);
116
117 bool wheelEventGesturesBecomeNonBlocking() const { return m_wheelEventGesturesBecomeNonBlocking; }
118 WEBCORE_EXPORT void setWheelEventGesturesBecomeNonBlocking(bool);
119
120 void dumpProperties(WTF::TextStream&, ScrollingStateTreeAsTextBehavior) const override;
121
122private:
123 ScrollingStateFrameScrollingNode(ScrollingStateTree&, ScrollingNodeType, ScrollingNodeID);
124 ScrollingStateFrameScrollingNode(const ScrollingStateFrameScrollingNode&, ScrollingStateTree&);
125
126 OptionSet<ScrollingStateNode::Property> applicableProperties() const final;

128 LayerRepresentation m_rootContentsLayer;
129 LayerRepresentation m_counterScrollingLayer;
130 LayerRepresentation m_insetClipLayer;
131 LayerRepresentation m_contentShadowLayer;
132 LayerRepresentation m_headerLayer;
133 LayerRepresentation m_footerLayer;
134
135 EventTrackingRegions m_eventTrackingRegions;
136
137 FloatRect m_layoutViewport;
138 FloatPoint m_minLayoutViewportOrigin;
139 FloatPoint m_maxLayoutViewportOrigin;
140 std::optional<FloatSize> m_overrideVisualViewportSize;
141
142 float m_frameScaleFactor { 1 };
143 float m_topContentInset { 0 };
144 int m_headerHeight { 0 };
145 int m_footerHeight { 0 };
146 ScrollBehaviorForFixedElements m_behaviorForFixed { StickToDocumentBounds };
147 bool m_fixedElementsLayoutRelativeToFrame { false };
148 bool m_visualViewportIsSmallerThanLayoutViewport { false };
149 bool m_asyncFrameOrOverflowScrollingEnabled { false };
150 bool m_wheelEventGesturesBecomeNonBlocking { false };
151 bool m_scrollingPerformanceTestingEnabled { false };
152};
