61struct WheelEventHandlingResult {
62 OptionSet<WheelEventProcessingSteps> steps;
63 bool wasHandled { false };
64 bool needsMainThreadProcessing() const { return steps.containsAny({ WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch }); }
65
66 static WheelEventHandlingResult handled(OptionSet<WheelEventProcessingSteps> steps = { })
67 {
68 return { steps, true };
69 }
70 static WheelEventHandlingResult unhandled(OptionSet<WheelEventProcessingSteps> steps = { })
71 {
72 return { steps, false };
73 }
74 static WheelEventHandlingResult result(bool handled)
75 {
76 return { { }, handled };
77 }
78};

227 struct ScrollUpdate {
228 ScrollingNodeID nodeID { 0 };
229 FloatPoint scrollPosition;
230 std::optional<FloatPoint> layoutViewportOrigin;
231 ScrollingLayerPositionAction updateLayerPositionAction { ScrollingLayerPositionAction::Sync };
232
233 bool canMerge(const ScrollUpdate& other) const
234 {
235 return nodeID == other.nodeID && updateLayerPositionAction == other.updateLayerPositionAction;
236 }
237
238 void merge(ScrollUpdate&& other)
239 {
240 scrollPosition = other.scrollPosition;
241 layoutViewportOrigin = other.layoutViewportOrigin;
242 }
243 };
