31#include "FetchOptions.h"
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class SecurityOrigin;
37class UserContentURLPattern;
38
39class SecurityPolicy {
40public:
41 // True if the referrer should be omitted according to ReferrerPolicy::Default.
42 // If you intend to send a referrer header, you should use generateReferrerHeader instead.
43 WEBCORE_EXPORT static bool shouldHideReferrer(const URL&, const String& referrer);
44
45 // Returns the referrer's security origin plus a / to make it a canonical URL
46 // and thus useable as referrer.
47 static String referrerToOriginString(const String& referrer);
48
49 // Returns the referrer modified according to the referrer policy for a
50 // navigation to a given URL. If the referrer returned is empty, the
51 // referrer header should be omitted.
52 WEBCORE_EXPORT static String generateReferrerHeader(ReferrerPolicy, const URL&, const String& referrer);
53
54 static String generateOriginHeader(ReferrerPolicy, const URL&, const SecurityOrigin&);
55
56 static bool shouldInheritSecurityOriginFromOwner(const URL&);
57
58 static bool isBaseURLSchemeAllowed(const URL&);
59
60 enum LocalLoadPolicy {
61 AllowLocalLoadsForAll, // No restriction on local loads.
62 AllowLocalLoadsForLocalAndSubstituteData,
63 AllowLocalLoadsForLocalOnly,
64 };
65
66 WEBCORE_EXPORT static void setLocalLoadPolicy(LocalLoadPolicy);
67 static bool restrictAccessToLocal();
68 static bool allowSubstituteDataAccessToLocal();
69
70 WEBCORE_EXPORT static void allowAccessTo(const UserContentURLPattern&);
71 WEBCORE_EXPORT static void addOriginAccessAllowlistEntry(const SecurityOrigin& sourceOrigin, const String& destinationProtocol, const String& destinationDomain, bool allowDestinationSubdomains);
72 WEBCORE_EXPORT static void removeOriginAccessAllowlistEntry(const SecurityOrigin& sourceOrigin, const String& destinationProtocol, const String& destinationDomain, bool allowDestinationSubdomains);
73 WEBCORE_EXPORT static void resetOriginAccessAllowlists();
74
75 static bool isAccessAllowed(const SecurityOrigin& activeOrigin, const SecurityOrigin& targetOrigin, const URL& targetURL);
76 static bool isAccessAllowed(const SecurityOrigin& activeOrigin, const URL& targetURL);
77};
78
79} // namespace WebCore
