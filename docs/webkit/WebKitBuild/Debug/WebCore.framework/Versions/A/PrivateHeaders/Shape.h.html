58class BasicShape;
59class Image;
60class RoundedRect;
61
62// A representation of a BasicShape that enables layout code to determine how to break a line up into segments
63// that will fit within or around a shape. The line is defined by a pair of logical Y coordinates and the
64// computed segments are returned as pairs of logical X coordinates. The BasicShape itself is defined in
65// physical coordinates.
66
67class Shape {
68 WTF_MAKE_FAST_ALLOCATED;
69public:
70 struct DisplayPaths {
71 Path shape;
72 Path marginShape;
73 };
74
75 static std::unique_ptr<Shape> createShape(const BasicShape&, const LayoutSize& logicalBoxSize, WritingMode, float margin);
76 static std::unique_ptr<Shape> createRasterShape(Image*, float threshold, const LayoutRect& imageRect, const LayoutRect& marginRect, WritingMode, float margin);
77 static std::unique_ptr<Shape> createBoxShape(const RoundedRect&, WritingMode, float margin);
78
79 virtual ~Shape() = default;
80
81 virtual LayoutRect shapeMarginLogicalBoundingBox() const = 0;
82 virtual bool isEmpty() const = 0;
83 virtual LineSegment getExcludedInterval(LayoutUnit logicalTop, LayoutUnit logicalHeight) const = 0;
84
85 bool lineOverlapsShapeMarginBounds(LayoutUnit lineTop, LayoutUnit lineHeight) const { return lineOverlapsBoundingBox(lineTop, lineHeight, shapeMarginLogicalBoundingBox()); }
86
87 virtual void buildDisplayPaths(DisplayPaths&) const = 0;
88
89protected:
90 float shapeMargin() const { return m_margin; }
91
92private:
93 bool lineOverlapsBoundingBox(LayoutUnit lineTop, LayoutUnit lineHeight, const LayoutRect& rect) const
94 {
95 if (rect.isEmpty())
96 return false;
97 return (lineTop < rect.maxY() && lineTop + lineHeight > rect.y()) || (!lineHeight && lineTop == rect.y());
98 }
99
100 WritingMode m_writingMode;
101 float m_margin;
102};
