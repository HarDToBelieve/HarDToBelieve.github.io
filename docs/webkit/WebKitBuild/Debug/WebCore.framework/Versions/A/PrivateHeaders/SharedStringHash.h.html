32namespace WebCore {
33
34using SharedStringHash = uint32_t;
35
36// This is a hash value, but it can be used as a key in HashMap. So, we need to avoid producing deleted-value in HashMap, which is -1.
37struct SharedStringHashHash {
38 static unsigned hash(SharedStringHash key) { return static_cast<unsigned>(key); }
39 static bool equal(SharedStringHash a, SharedStringHash b) { return a == b; }
40 static const bool safeToCompareToEmptyOrDeleted = true;
41 static constexpr SharedStringHash deletedValue = std::numeric_limits<SharedStringHash>::max();
42};
43
44using SharedStringHashMarkableTraits = IntegralMarkableTraits<SharedStringHash, SharedStringHashHash::deletedValue>;
45
46// Returns the hash of the string that will be used for visited link coloring.
47WEBCORE_EXPORT SharedStringHash computeSharedStringHash(const String& url);
48WEBCORE_EXPORT SharedStringHash computeSharedStringHash(const UChar* url, unsigned length);
49
50// Resolves the potentially relative URL "attributeURL" relative to the given
51// base URL, and returns the hash of the string that will be used for visited
52// link coloring. It will return the special value of 0 if attributeURL does not
53// look like a relative URL.
54SharedStringHash computeVisitedLinkHash(const URL& base, const AtomString& attributeURL);

56} // namespace WebCore
