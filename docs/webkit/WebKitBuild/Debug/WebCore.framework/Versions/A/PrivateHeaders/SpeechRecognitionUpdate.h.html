34namespace WebCore {
35
36enum class SpeechRecognitionUpdateType {
37 Start,
38 AudioStart,
39 SoundStart,
40 SpeechStart,
41 SpeechEnd,
42 SoundEnd,
43 AudioEnd,
44 Result,
45 NoMatch,
46 Error,
47 End
48};
49
50String convertEnumerationToString(SpeechRecognitionUpdateType);
51
52class SpeechRecognitionUpdate {
53public:
54 WEBCORE_EXPORT static SpeechRecognitionUpdate create(SpeechRecognitionConnectionClientIdentifier, SpeechRecognitionUpdateType);
55 WEBCORE_EXPORT static SpeechRecognitionUpdate createError(SpeechRecognitionConnectionClientIdentifier, const SpeechRecognitionError&);
56 WEBCORE_EXPORT static SpeechRecognitionUpdate createResult(SpeechRecognitionConnectionClientIdentifier, const Vector<SpeechRecognitionResultData>&);
57
58 SpeechRecognitionConnectionClientIdentifier clientIdentifier() const { return m_clientIdentifier; }
59 SpeechRecognitionUpdateType type() const { return m_type; }
60 WEBCORE_EXPORT SpeechRecognitionError error() const;
61 WEBCORE_EXPORT Vector<SpeechRecognitionResultData> result() const;
62
63 template<class Encoder> void encode(Encoder&) const;
64 template<class Decoder> static std::optional<SpeechRecognitionUpdate> decode(Decoder&);
65
66private:
67 using Content = Variant<WTF::Monostate, SpeechRecognitionError, Vector<SpeechRecognitionResultData>>;
68 WEBCORE_EXPORT SpeechRecognitionUpdate(SpeechRecognitionConnectionClientIdentifier, SpeechRecognitionUpdateType, Content);
69
70 SpeechRecognitionConnectionClientIdentifier m_clientIdentifier;
71 SpeechRecognitionUpdateType m_type;
72 Content m_content;
73};
74
75template<class Encoder>
76void SpeechRecognitionUpdate::encode(Encoder& encoder) const
77{
78 encoder << m_clientIdentifier << m_type << m_content;
79}
80
81template<class Decoder>
82std::optional<SpeechRecognitionUpdate> SpeechRecognitionUpdate::decode(Decoder& decoder)
83{
84 std::optional<SpeechRecognitionConnectionClientIdentifier> clientIdentifier;
85 decoder >> clientIdentifier;
86 if (!clientIdentifier)
87 return std::nullopt;
88
89 std::optional<SpeechRecognitionUpdateType> type;
90 decoder >> type;
91 if (!type)
92 return std::nullopt;
93
94 std::optional<Content> content;
95 decoder >> content;
96 if (!content)
97 return std::nullopt;
98
99 return SpeechRecognitionUpdate { WTFMove(*clientIdentifier), WTFMove(*type), WTFMove(*content) };
100}
