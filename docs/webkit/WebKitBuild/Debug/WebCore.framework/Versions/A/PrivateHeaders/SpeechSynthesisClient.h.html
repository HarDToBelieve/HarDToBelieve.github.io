26#pragma once
27
28#if ENABLE(SPEECH_SYNTHESIS)
29
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class PlatformSpeechSynthesisUtterance;
35class SpeechSynthesisClientObserver;
36class PlatformSpeechSynthesisVoice;
37
38class SpeechSynthesisClient : public CanMakeWeakPtr<SpeechSynthesisClient> {
39public:
40 virtual ~SpeechSynthesisClient() = default;
41
42 virtual void setObserver(WeakPtr<SpeechSynthesisClientObserver>) = 0;
43 virtual WeakPtr<SpeechSynthesisClientObserver> observer() const = 0;
44
45 virtual const Vector<RefPtr<PlatformSpeechSynthesisVoice>>& voiceList() = 0;
46 virtual void speak(RefPtr<PlatformSpeechSynthesisUtterance>) = 0;
47 virtual void cancel() = 0;
48 virtual void pause() = 0;
49 virtual void resume() = 0;
50
51};
52
53class SpeechSynthesisClientObserver : public CanMakeWeakPtr<SpeechSynthesisClientObserver> {
54public:
55 virtual ~SpeechSynthesisClientObserver() = default;
56
57 virtual void didStartSpeaking() = 0;
58 virtual void didFinishSpeaking() = 0;
59 virtual void didPauseSpeaking() = 0;
60 virtual void didResumeSpeaking() = 0;
61 virtual void speakingErrorOccurred() = 0;
62 virtual void boundaryEventOccurred(bool wordBoundary, unsigned charIndex) = 0;
63 virtual void voicesChanged() = 0;
64};
65
66} // namespace WebCore
67
68#endif // ENABLE(SPEECH_SYNTHESIS)
