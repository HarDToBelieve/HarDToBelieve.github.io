31namespace WebCore {
32
33class StyleCustomPropertyData : public RefCounted<StyleCustomPropertyData> {
34public:
35 static Ref<StyleCustomPropertyData> create() { return adoptRef(*new StyleCustomPropertyData); }
36 Ref<StyleCustomPropertyData> copy() const { return adoptRef(*new StyleCustomPropertyData(*this)); }
37
38 bool operator==(const StyleCustomPropertyData& other) const
39 {
40 if (values.size() != other.values.size())
41 return false;
42
43 for (auto& entry : values) {
44 auto otherEntry = other.values.find(entry.key);
45 if (otherEntry == other.values.end() || !entry.value->equals(*otherEntry->value))
46 return false;
47 }
48
49 return true;
50 }
51
52 bool operator!=(const StyleCustomPropertyData& other) const { return !(*this == other); }
53
54 void setCustomPropertyValue(const AtomString& name, Ref<CSSCustomPropertyValue>&& value)
55 {
56 values.set(name, WTFMove(value));
57 }
58
59 CustomPropertyValueMap values;
60
61private:
62 StyleCustomPropertyData() = default;
63 StyleCustomPropertyData(const StyleCustomPropertyData& other)
64 : RefCounted<StyleCustomPropertyData>()
65 , values(other.values)
66 { }
67};
