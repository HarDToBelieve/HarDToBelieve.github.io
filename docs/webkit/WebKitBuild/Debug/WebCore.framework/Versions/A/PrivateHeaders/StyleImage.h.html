33namespace WebCore {
34
35class CachedImage;
36class CachedResourceLoader;
37class CSSValue;
38class RenderElement;
39class RenderObject;
40struct ResourceLoaderOptions;
41
42typedef const void* WrappedImagePtr;
43
44class StyleImage : public RefCounted<StyleImage> {
45public:
46 virtual ~StyleImage() = default;
47
48 virtual bool operator==(const StyleImage& other) const = 0;
49
50 virtual Ref<CSSValue> cssValue() const = 0;
51
52 virtual bool canRender(const RenderElement*, float /*multiplier*/) const { return true; }
53 virtual bool isPending() const = 0;
54 virtual void load(CachedResourceLoader&, const ResourceLoaderOptions&) = 0;
55 virtual bool isLoaded() const { return true; }
56 virtual bool errorOccurred() const { return false; }
57 virtual FloatSize imageSize(const RenderElement*, float multiplier) const = 0;
58 virtual void computeIntrinsicDimensions(const RenderElement*, Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio) = 0;
59 virtual bool imageHasRelativeWidth() const = 0;
60 virtual bool imageHasRelativeHeight() const = 0;
61 virtual bool usesImageContainerSize() const = 0;
62 virtual void setContainerContextForRenderer(const RenderElement&, const FloatSize&, float) = 0;
63 virtual void addClient(RenderElement&) = 0;
64 virtual void removeClient(RenderElement&) = 0;
65 virtual bool hasClient(RenderElement&) const = 0;
66 virtual RefPtr<Image> image(RenderElement*, const FloatSize&) const = 0;
67 virtual WrappedImagePtr data() const = 0;
68 virtual float imageScaleFactor() const { return 1; }
69 virtual bool knownToBeOpaque(const RenderElement&) const = 0;
70 virtual CachedImage* cachedImage() const { return 0; }
71 virtual StyleImage* selectedImage() { return this; }
72 virtual const StyleImage* selectedImage() const { return this; }
73
74 ALWAYS_INLINE bool isCachedImage() const { return m_isCachedImage; }
75 ALWAYS_INLINE bool isGeneratedImage() const { return m_isGeneratedImage; }
76 ALWAYS_INLINE bool isCursorImage() const { return m_isCursorImage; }
77 ALWAYS_INLINE bool isImageSet() const { return m_isImageSet; }
78
79 bool hasCachedImage() const { return m_isCachedImage || selectedImage()->isCachedImage(); }
80
81 virtual bool usesDataProtocol() const { return false; }
82
83protected:
84 StyleImage()
85 : m_isCachedImage(false)
86 , m_isGeneratedImage(false)
87 , m_isImageSet(false)
88 , m_isCursorImage(false)
89 {
90 }
91 bool m_isCachedImage : 1;
92 bool m_isGeneratedImage : 1;
93 bool m_isImageSet : 1;
94 bool m_isCursorImage : 1;
95};
96
97} // namespace WebCore
