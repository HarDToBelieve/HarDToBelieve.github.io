198inline bool Scope::hasPendingSheets() const
199{
200 return hasPendingSheetsBeforeBody() || !m_elementsInBodyWithPendingSheets.isEmpty();
201}
202
203inline bool Scope::hasPendingSheetsBeforeBody() const
204{
205 return !m_elementsInHeadWithPendingSheets.isEmpty() || !m_processingInstructionsWithPendingSheets.isEmpty();
206}
207
208inline bool Scope::hasPendingSheetsInBody() const
209{
210 return !m_elementsInBodyWithPendingSheets.isEmpty();
211}
212
213inline void Scope::flushPendingUpdate()
214{
215 if (m_hasDescendantWithPendingUpdate)
216 flushPendingDescendantUpdates();
217 if (m_pendingUpdate)
218 flushPendingSelfUpdate();
219}
