33namespace WebCore {
34
35 class DocumentFragment;
36
37 enum class MailBlockquoteHandling;
38
39 class TextEvent final : public UIEvent {
40 WTF_MAKE_ISO_ALLOCATED(TextEvent);
41 public:
42 static Ref<TextEvent> create(RefPtr<WindowProxy>&&, const String& data, TextEventInputType = TextEventInputKeyboard);
43 static Ref<TextEvent> createForBindings();
44 static Ref<TextEvent> createForPlainTextPaste(RefPtr<WindowProxy>&&, const String& data, bool shouldSmartReplace);
45 static Ref<TextEvent> createForFragmentPaste(RefPtr<WindowProxy>&&, RefPtr<DocumentFragment>&& data, bool shouldSmartReplace, bool shouldMatchStyle, MailBlockquoteHandling);
46 static Ref<TextEvent> createForDrop(RefPtr<WindowProxy>&&, const String& data);
47 static Ref<TextEvent> createForDictation(RefPtr<WindowProxy>&&, const String& data, const Vector<DictationAlternative>& dictationAlternatives);
48
49 virtual ~TextEvent();
50
51 WEBCORE_EXPORT void initTextEvent(const AtomString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&&, const String& data);
52
53 String data() const { return m_data; }
54
55 EventInterface eventInterface() const override;
56
57 bool isLineBreak() const { return m_inputType == TextEventInputLineBreak; }
58 bool isComposition() const { return m_inputType == TextEventInputComposition; }
59 bool isBackTab() const { return m_inputType == TextEventInputBackTab; }
60 bool isPaste() const { return m_inputType == TextEventInputPaste; }
61 bool isDrop() const { return m_inputType == TextEventInputDrop; }
62 bool isDictation() const { return m_inputType == TextEventInputDictation; }
63 bool isAutocompletion() const { return m_inputType == TextEventInputAutocompletion; }
64 bool isKeyboard() const { return m_inputType == TextEventInputKeyboard; }
65
66 bool shouldSmartReplace() const { return m_shouldSmartReplace; }
67 bool shouldMatchStyle() const { return m_shouldMatchStyle; }
68 MailBlockquoteHandling mailBlockquoteHandling() const { return m_mailBlockquoteHandling; }
69 DocumentFragment* pastingFragment() const { return m_pastingFragment.get(); }
70 const Vector<DictationAlternative>& dictationAlternatives() const { return m_dictationAlternatives; }
71
72 private:
73 TextEvent();
74
75 TextEvent(RefPtr<WindowProxy>&&, const String& data, TextEventInputType = TextEventInputKeyboard);
76 TextEvent(RefPtr<WindowProxy>&&, const String& data, RefPtr<DocumentFragment>&&, bool shouldSmartReplace, bool shouldMatchStyle, MailBlockquoteHandling);
77 TextEvent(RefPtr<WindowProxy>&&, const String& data, const Vector<DictationAlternative>& dictationAlternatives);
78
79 bool isTextEvent() const override;
80
81 TextEventInputType m_inputType;
82 String m_data;
83
84 RefPtr<DocumentFragment> m_pastingFragment;
85 bool m_shouldSmartReplace;
86 bool m_shouldMatchStyle;
87 MailBlockquoteHandling m_mailBlockquoteHandling;
88 Vector<DictationAlternative> m_dictationAlternatives;
89 };
