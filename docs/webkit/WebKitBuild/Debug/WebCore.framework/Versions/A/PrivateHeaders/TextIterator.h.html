90// Iterates through the DOM range, returning all the text, and 0-length boundaries
91// at points where replaced elements break up the text flow. The text is delivered in
92// the chunks it's already stored in, to avoid copying any text.
93
94class TextIterator {
95 WTF_MAKE_FAST_ALLOCATED;
96public:
97 WEBCORE_EXPORT explicit TextIterator(const SimpleRange&, TextIteratorBehaviors = { });
98 WEBCORE_EXPORT ~TextIterator();
99
100 bool atEnd() const { return !m_positionNode; }
101 WEBCORE_EXPORT void advance();
102
103 StringView text() const { ASSERT(!atEnd()); return m_text; }
104 WEBCORE_EXPORT SimpleRange range() const;
105 WEBCORE_EXPORT Node* node() const;
106
107 const TextIteratorCopyableText& copyableText() const { ASSERT(!atEnd()); return m_copyableText; }
108 void appendTextToStringBuilder(StringBuilder& builder) const { copyableText().appendToStringBuilder(builder); }
109
110private:
111 void init();
112 void exitNode(Node*);
113 bool shouldRepresentNodeOffsetZero();
114 bool shouldEmitSpaceBeforeAndAfterNode(Node&);
115 void representNodeOffsetZero();
116 bool handleTextNode();
117 bool handleReplacedElement();
118 bool handleNonTextNode();
119 void handleTextRun();
120 void handleTextNodeFirstLetter(RenderTextFragment&);
121 void emitCharacter(UChar, Node& characterNode, Node* offsetBaseNode, int textStartOffset, int textEndOffset);
122 void emitText(Text& textNode, RenderText&, int textStartOffset, int textEndOffset);
123
124 Node* baseNodeForEmittingNewLine() const;
125
126 const TextIteratorBehaviors m_behaviors;
127
128 // Current position, not necessarily of the text being returned, but position as we walk through the DOM tree.
129 Node* m_node { nullptr };
130 int m_offset { 0 };
131 bool m_handledNode { false };
132 bool m_handledChildren { false };
133 BitStack m_fullyClippedStack;
134
135 // The range.
136 Node* m_startContainer { nullptr };
137 int m_startOffset { 0 };
138 Node* m_endContainer { nullptr };
139 int m_endOffset { 0 };
140 Node* m_pastEndNode { nullptr };
141
142 // The current text and its position, in the form to be returned from the iterator.
143 Node* m_positionNode { nullptr };
144 mutable Node* m_positionOffsetBaseNode { nullptr };
145 mutable int m_positionStartOffset { 0 };
146 mutable int m_positionEndOffset { 0 };
147 TextIteratorCopyableText m_copyableText;
148 StringView m_text;
149
150 // Used when there is still some pending text from the current node; when these are false and null, we go back to normal iterating.
151 Node* m_nodeForAdditionalNewline { nullptr };
152 LayoutIntegration::TextRunIterator m_textRun;
153
154 // Used when iterating over :first-letter text to save pointer to remaining text box.
155 LayoutIntegration::TextRunIterator m_remainingTextRun;
156
157 // Used to point to RenderText object for :first-letter.
158 RenderText* m_firstLetterText { nullptr };
159
160 // Used to do the whitespace collapsing logic.
161 Text* m_lastTextNode { nullptr };
162 bool m_lastTextNodeEndedWithCollapsedSpace { false };
163 UChar m_lastCharacter { 0 };
164
165 // Used when deciding whether to emit a "positioning" (e.g. newline) before any other content
166 bool m_hasEmitted { false };
167
168 // Used when deciding text fragment created by :first-letter should be looked into.
169 bool m_handledFirstLetter { false };
170};

172// Iterates through the DOM range, returning all the text, and 0-length boundaries
173// at points where replaced elements break up the text flow. The text comes back in
174// chunks so as to optimize for performance of the iteration.
175class SimplifiedBackwardsTextIterator {
176public:
177 WEBCORE_EXPORT explicit SimplifiedBackwardsTextIterator(const SimpleRange&);
178
179 bool atEnd() const { return !m_positionNode; }
180 WEBCORE_EXPORT void advance();
181
182 StringView text() const { ASSERT(!atEnd()); return m_text; }
183 WEBCORE_EXPORT SimpleRange range() const;
184 Node* node() const { ASSERT(!atEnd()); return m_node; }
185
186private:
187 void exitNode();
188 bool handleTextNode();
189 RenderText* handleFirstLetter(int& startOffset, int& offsetInNode);
190 bool handleReplacedElement();
191 bool handleNonTextNode();
192 void emitCharacter(UChar, Node&, int startOffset, int endOffset);
193 bool advanceRespectingRange(Node*);
194
195 const TextIteratorBehaviors m_behaviors;
196
197 // Current position, not necessarily of the text being returned, but position as we walk through the DOM tree.
198 Node* m_node { nullptr };
199 int m_offset { 0 };
200 bool m_handledNode { false };
201 bool m_handledChildren { false };
202 BitStack m_fullyClippedStack;
203
204 // The range.
205 Node* m_startContainer { nullptr };
206 int m_startOffset { 0 };
207 Node* m_endContainer { nullptr };
208 int m_endOffset { 0 };
209
210 // The current text and its position, in the form to be returned from the iterator.
211 Node* m_positionNode { nullptr };
212 int m_positionStartOffset { 0 };
213 int m_positionEndOffset { 0 };
214 TextIteratorCopyableText m_copyableText;
215 StringView m_text;
216
217 // Used to do the whitespace logic.
218 Text* m_lastTextNode { nullptr };
219 UChar m_lastCharacter { 0 };
220
221 // Whether m_node has advanced beyond the iteration range (i.e. m_startContainer).
222 bool m_havePassedStartContainer { false };
223
224 // Should handle first-letter renderer in the next call to handleTextNode.
225 bool m_shouldHandleFirstLetter { false };
226};

228// Builds on the text iterator, adding a character position so we can walk one
229// character at a time, or faster, as needed. Useful for searching.
230class CharacterIterator {
231public:
232 WEBCORE_EXPORT explicit CharacterIterator(const SimpleRange&, TextIteratorBehaviors = { });
233
234 bool atEnd() const { return m_underlyingIterator.atEnd(); }
235 WEBCORE_EXPORT void advance(int numCharacters);
236
237 StringView text() const { return m_underlyingIterator.text().substring(m_runOffset); }
238 WEBCORE_EXPORT SimpleRange range() const;
239
240 bool atBreak() const { return m_atBreak; }
241 unsigned characterOffset() const { return m_offset; }
242
243private:
244 TextIterator m_underlyingIterator;
245
246 unsigned m_offset { 0 };
247 unsigned m_runOffset { 0 };
248 bool m_atBreak { true };
249};

256 void advance(int numCharacters);
257

259

262

265 bool m_atBreak { true };
266};

268// Similar to the TextIterator, except that the chunks of text returned are "well behaved", meaning
269// they never split up a word. This is useful for spell checking and perhaps one day for searching as well.
270class WordAwareIterator {
271public:
272 explicit WordAwareIterator(const SimpleRange&);
273
274 bool atEnd() const { return !m_didLookAhead && m_underlyingIterator.atEnd(); }
275 void advance();
276
277 StringView text() const;
278
279private:
280 TextIterator m_underlyingIterator;
281
282 // Text from the previous chunk from the text iterator.
283 TextIteratorCopyableText m_previousText;
284
285 // Many chunks from text iterator concatenated.
286 Vector<UChar> m_buffer;
287
288 // Did we have to look ahead in the text iterator to confirm the current chunk?
289 bool m_didLookAhead { true };
290};

292inline CharacterRange characterRange(const BoundaryPoint& start, const SimpleRange& range, TextIteratorBehaviors behaviors)
293{
294 return { characterCount({ start, range.start }, behaviors), characterCount(range, behaviors) };
295}
296
297inline CharacterRange characterRange(const SimpleRange& scope, const SimpleRange& range, TextIteratorBehaviors behaviors)
298{
299 return characterRange(scope.start, range, behaviors);
300}
301
302inline BoundaryPoint resolveCharacterLocation(const SimpleRange& scope, uint64_t location, TextIteratorBehaviors behaviors)
303{
304 return resolveCharacterRange(scope, { location, 0 }, behaviors).start;
305}
