27#ifndef ThreadGlobalData_h
28#define ThreadGlobalData_h
29
30#include <wtf/ThreadSafeRefCounted.h>
31#include <wtf/text/StringHash.h>
32
33namespace JSC {
34class CallFrame;
35class JSGlobalObject;
36}
37
38namespace WebCore {
39
40class QualifiedNameCache;
41class ThreadTimers;
42
43struct CachedResourceRequestInitiators;
44struct EventNames;
45struct ICUConverterWrapper;
46struct MIMETypeRegistryThreadGlobalData;
47
48#if USE(WEB_THREAD)
49class ThreadGlobalData : public ThreadSafeRefCounted<ThreadGlobalData> {
50#else
51class ThreadGlobalData {
52#endif
53 WTF_MAKE_NONCOPYABLE(ThreadGlobalData);
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 WEBCORE_EXPORT ThreadGlobalData();
57 WEBCORE_EXPORT ~ThreadGlobalData();
58 void destroy(); // called on workers to clean up the ThreadGlobalData before the thread exits.
59
60 const CachedResourceRequestInitiators& cachedResourceRequestInitiators()
61 {
62 if (UNLIKELY(!m_cachedResourceRequestInitiators))
63 initializeCachedResourceRequestInitiators();
64 return *m_cachedResourceRequestInitiators;
65 }
66 EventNames& eventNames()
67 {
68 if (UNLIKELY(!m_eventNames))
69 initializeEventNames();
70 return *m_eventNames;
71 }
72 QualifiedNameCache& qualifiedNameCache()
73 {
74 if (UNLIKELY(!m_qualifiedNameCache))
75 initializeQualifiedNameCache();
76 return *m_qualifiedNameCache;
77 }
78 const MIMETypeRegistryThreadGlobalData& mimeTypeRegistryThreadGlobalData()
79 {
80 if (UNLIKELY(!m_MIMETypeRegistryThreadGlobalData))
81 initializeMimeTypeRegistryThreadGlobalData();
82 return *m_MIMETypeRegistryThreadGlobalData;
83 }
84
85 ThreadTimers& threadTimers() { return *m_threadTimers; }
86 ICUConverterWrapper& cachedConverterICU() { return *m_cachedConverterICU; }
87
88 JSC::JSGlobalObject* currentState() const { return m_currentState; }
89 void setCurrentState(JSC::JSGlobalObject* state) { m_currentState = state; }
90
91#if USE(WEB_THREAD)
92 void setWebCoreThreadData();
93#endif
94
95 bool isInRemoveAllEventListeners() const { return m_isInRemoveAllEventListeners; }
96 void setIsInRemoveAllEventListeners(bool value) { m_isInRemoveAllEventListeners = value; }
97
98private:
99 WEBCORE_EXPORT void initializeCachedResourceRequestInitiators();
100 WEBCORE_EXPORT void initializeEventNames();
101 WEBCORE_EXPORT void initializeQualifiedNameCache();
102 WEBCORE_EXPORT void initializeMimeTypeRegistryThreadGlobalData();
103
104 std::unique_ptr<CachedResourceRequestInitiators> m_cachedResourceRequestInitiators;
105 std::unique_ptr<EventNames> m_eventNames;
106 std::unique_ptr<ThreadTimers> m_threadTimers;
107 std::unique_ptr<QualifiedNameCache> m_qualifiedNameCache;
108 JSC::JSGlobalObject* m_currentState { nullptr };
109 std::unique_ptr<ICUConverterWrapper> m_cachedConverterICU;
110 std::unique_ptr<MIMETypeRegistryThreadGlobalData> m_MIMETypeRegistryThreadGlobalData;
111
112#ifndef NDEBUG
113 bool m_isMainThread;
114#endif
115
116 bool m_isInRemoveAllEventListeners { false };
117
118 WEBCORE_EXPORT friend ThreadGlobalData& threadGlobalData();
119};
120
121#if USE(WEB_THREAD)
122WEBCORE_EXPORT ThreadGlobalData& threadGlobalData();
123#else
124WEBCORE_EXPORT ThreadGlobalData& threadGlobalData() PURE_FUNCTION;
125#endif
126
127} // namespace WebCore
128
129#endif // ThreadGlobalData_h
