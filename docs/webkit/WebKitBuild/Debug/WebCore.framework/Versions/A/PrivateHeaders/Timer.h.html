108class Timer : public TimerBase {
109 WTF_MAKE_FAST_ALLOCATED;
110public:
111 static void schedule(Seconds delay, WTF::Function<void()>&& function)
112 {
113 auto* timer = new Timer([] { });
114 timer->m_function = [timer, function = WTFMove(function)] {
115 function();
116 delete timer;
117 };
118 timer->startOneShot(delay);
119 }
120
121 template <typename TimerFiredClass, typename TimerFiredBaseClass>
122 Timer(TimerFiredClass& object, void (TimerFiredBaseClass::*function)())
123 : m_function(std::bind(function, &object))
124 {
125 }
126
127 Timer(WTF::Function<void ()>&& function)
128 : m_function(WTFMove(function))
129 {
130 }
131
132private:
133 void fired() override
134 {
135 m_function();
136 }
137
138 WTF::Function<void ()> m_function;
139};

141inline bool TimerBase::isActive() const
142{
143 // FIXME: Write this in terms of USE(WEB_THREAD) instead of PLATFORM(IOS_FAMILY).
144#if !PLATFORM(IOS_FAMILY)
145 ASSERT(m_thread.ptr() == &Thread::current());
146#else
147 ASSERT(WebThreadIsCurrent() || pthread_main_np() || m_thread.ptr() == &Thread::current());
148#endif // PLATFORM(IOS_FAMILY)
149 return static_cast<bool>(nextFireTime());
150}

152class DeferrableOneShotTimer : protected TimerBase {
153 WTF_MAKE_FAST_ALLOCATED;
154public:
155 template<typename TimerFiredClass>
156 DeferrableOneShotTimer(TimerFiredClass& object, void (TimerFiredClass::*function)(), Seconds delay)
157 : DeferrableOneShotTimer(std::bind(function, &object), delay)
158 {
159 }
160
161 DeferrableOneShotTimer(WTF::Function<void()>&& function, Seconds delay)
162 : m_function(WTFMove(function))
163 , m_delay(delay)
164 , m_shouldRestartWhenTimerFires(false)
165 {
166 }
167
168 void restart()
169 {
170 // Setting this boolean is much more efficient than calling startOneShot
171 // again, which might result in rescheduling the system timer which
172 // can be quite expensive.
173
174 if (isActive()) {
175 m_shouldRestartWhenTimerFires = true;
176 return;
177 }
178 startOneShot(m_delay);
179 }
180
181 void stop()
182 {
183 m_shouldRestartWhenTimerFires = false;
184 TimerBase::stop();
185 }
186
187 using TimerBase::isActive;
188
189private:
190 void fired() override
191 {
192 if (m_shouldRestartWhenTimerFires) {
193 m_shouldRestartWhenTimerFires = false;
194 startOneShot(m_delay);
195 return;
196 }
197
198 m_function();
199 }
200
201 WTF::Function<void ()> m_function;
202
203 Seconds m_delay;
204 bool m_shouldRestartWhenTimerFires;
205};
