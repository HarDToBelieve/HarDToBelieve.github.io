25#pragma once
26
27#include "LayoutSize.h"
28#include "TransformOperation.h"
29#include <wtf/RefPtr.h>
30#include <wtf/Vector.h>
31
32namespace WebCore {
33
34struct BlendingContext;
35
36class TransformOperations {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 explicit TransformOperations(bool makeIdentity = false);
40
41 bool operator==(const TransformOperations& o) const;
42 bool operator!=(const TransformOperations& o) const
43 {
44 return !(*this == o);
45 }
46
47 void apply(const FloatSize& sz, TransformationMatrix& t) const
48 {
49 for (unsigned i = 0; i < m_operations.size(); ++i)
50 m_operations[i]->apply(t, sz);
51 }
52
53 // Return true if any of the operation types are 3D operation types (even if the
54 // values describe affine transforms)
55 bool has3DOperation() const
56 {
57 for (const auto& operation : m_operations) {
58 if (operation->is3DOperation())
59 return true;
60 }
61 return false;
62 }
63
64 bool isRepresentableIn2D() const
65 {
66 for (const auto& operation : m_operations) {
67 if (!operation->isRepresentableIn2D())
68 return false;
69 }
70 return true;
71 }
72
73 bool operationsMatch(const TransformOperations&) const;
74
75 void clear()
76 {
77 m_operations.clear();
78 }
79
80 bool affectedByTransformOrigin() const;
81
82 Vector<RefPtr<TransformOperation>>& operations() { return m_operations; }
83 const Vector<RefPtr<TransformOperation>>& operations() const { return m_operations; }
84
85 size_t size() const { return m_operations.size(); }
86 const TransformOperation* at(size_t index) const { return index < m_operations.size() ? m_operations.at(index).get() : 0; }
87
88 TransformOperations blendByMatchingOperations(const TransformOperations& from, const BlendingContext&) const;
89 TransformOperations blendByUsingMatrixInterpolation(const TransformOperations& from, const BlendingContext&, const LayoutSize&) const;
90 TransformOperations blend(const TransformOperations& from, const BlendingContext&, const LayoutSize&) const;
91
92private:
93 Vector<RefPtr<TransformOperation>> m_operations;
94};
95
96WTF::TextStream& operator<<(WTF::TextStream&, const TransformOperations&);
97
98} // namespace WebCore
