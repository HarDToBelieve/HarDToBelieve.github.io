25#pragma once
26
27#include "Length.h"
28#include "LengthFunctions.h"
29#include "TransformOperation.h"
30#include <wtf/Ref.h>
31
32namespace WebCore {
33
34struct BlendingContext;
35
36class TranslateTransformOperation final : public TransformOperation {
37public:
38 static Ref<TranslateTransformOperation> create(const Length& tx, const Length& ty, OperationType type)
39 {
40 return adoptRef(*new TranslateTransformOperation(tx, ty, Length(0, LengthType::Fixed), type));
41 }
42
43 static Ref<TranslateTransformOperation> create(const Length& tx, const Length& ty, const Length& tz, OperationType type)
44 {
45 return adoptRef(*new TranslateTransformOperation(tx, ty, tz, type));
46 }
47
48 Ref<TransformOperation> clone() const override
49 {
50 return adoptRef(*new TranslateTransformOperation(m_x, m_y, m_z, type()));
51 }
52
53 float xAsFloat(const FloatSize& borderBoxSize) const { return floatValueForLength(m_x, borderBoxSize.width()); }
54 float yAsFloat(const FloatSize& borderBoxSize) const { return floatValueForLength(m_y, borderBoxSize.height()); }
55 float zAsFloat() const { return floatValueForLength(m_z, 1); }
56
57 Length x() const { return m_x; }
58 Length y() const { return m_y; }
59 Length z() const { return m_z; }
60
61 bool apply(TransformationMatrix& transform, const FloatSize& borderBoxSize) const final
62 {
63 transform.translate3d(xAsFloat(borderBoxSize), yAsFloat(borderBoxSize), zAsFloat());
64 return m_x.isPercent() || m_y.isPercent();
65 }
66
67 bool isIdentity() const final { return !floatValueForLength(m_x, 1) && !floatValueForLength(m_y, 1) && !floatValueForLength(m_z, 1); }
68
69 bool operator==(const TransformOperation&) const final;
70
71 Ref<TransformOperation> blend(const TransformOperation* from, const BlendingContext&, bool blendToIdentity = false) final;
72
73 bool isRepresentableIn2D() const final { return m_z.isZero(); }
74
75private:
76
77 void dump(WTF::TextStream&) const final;
78
79 TranslateTransformOperation(const Length& tx, const Length& ty, const Length& tz, OperationType type)
80 : TransformOperation(type)
81 , m_x(tx)
82 , m_y(ty)
83 , m_z(tz)
84 {
85 ASSERT(isTranslateTransformOperationType());
86 }
87
88 Length m_x;
89 Length m_y;
90 Length m_z;
91};
92
93} // namespace WebCore
94
95SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::TranslateTransformOperation, isTranslateTransformOperationType())
