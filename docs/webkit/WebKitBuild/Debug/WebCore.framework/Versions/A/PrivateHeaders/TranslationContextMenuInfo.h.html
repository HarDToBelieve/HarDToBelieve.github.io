35namespace WebCore {
36
37enum class TranslationContextMenuMode : bool { NonEditable, Editable };
38
39struct TranslationContextMenuInfo {
40 String text;
41 IntRect selectionBoundsInRootView;
42 IntPoint locationInRootView;
43 TranslationContextMenuMode mode { TranslationContextMenuMode::NonEditable };
44
45 template<class Encoder> void encode(Encoder&) const;
46 template<class Decoder> static std::optional<TranslationContextMenuInfo> decode(Decoder&);
47};
48
49template<class Encoder> void TranslationContextMenuInfo::encode(Encoder& encoder) const
50{
51 encoder << text;
52 encoder << selectionBoundsInRootView;
53 encoder << locationInRootView;
54 encoder << mode;
55}
56
57template<class Decoder> std::optional<TranslationContextMenuInfo> TranslationContextMenuInfo::decode(Decoder& decoder)
58{
59 std::optional<String> text;
60 decoder >> text;
61 if (!text)
62 return std::nullopt;
63
64 std::optional<IntRect> selectionBoundsInRootView;
65 decoder >> selectionBoundsInRootView;
66 if (!selectionBoundsInRootView)
67 return std::nullopt;
68
69 std::optional<IntPoint> locationInRootView;
70 decoder >> locationInRootView;
71 if (!locationInRootView)
72 return std::nullopt;
73
74 std::optional<TranslationContextMenuMode> mode;
75 decoder >> mode;
76 if (!mode)
77 return std::nullopt;
78
79 return {{ WTFMove(*text), WTFMove(*selectionBoundsInRootView), WTFMove(*locationInRootView), *mode }};
80}
81
82} // namespace WebCore

91 >;
