43namespace fido {

45// Checks whether the request can be translated to valid U2F request
46// parameter. Namely, U2F request does not support resident key and
47// user verification, and ES256 algorithm must be used for public key
48// credential.
49// https://fidoalliance.org/specs/fido-v2.0-id-20180227/fido-client-to-authenticator-protocol-v2.0-id-20180227.html#u2f-authenticatorMakeCredential-interoperability
50WEBCORE_EXPORT bool isConvertibleToU2fRegisterCommand(const WebCore::PublicKeyCredentialCreationOptions&);
51
52// Checks whether user verification is not required and that allow list is
53// not empty.
54// https://fidoalliance.org/specs/fido-v2.0-id-20180227/fido-client-to-authenticator-protocol-v2.0-id-20180227.html#u2f-authenticatorGetAssertion-interoperability
55WEBCORE_EXPORT bool isConvertibleToU2fSignCommand(const WebCore::PublicKeyCredentialRequestOptions&);
56
57// Extracts APDU encoded U2F register command from PublicKeyCredentialCreationOptions.
58WEBCORE_EXPORT std::optional<Vector<uint8_t>> convertToU2fRegisterCommand(const Vector<uint8_t>& clientDataHash, const WebCore::PublicKeyCredentialCreationOptions&);
59
60// Extracts APDU encoded U2F check only sign command from
61// PublicKeyCredentialCreationOptions. Invoked when U2F register operation includes key
62// handles in exclude list.
63WEBCORE_EXPORT std::optional<Vector<uint8_t>> convertToU2fCheckOnlySignCommand(const Vector<uint8_t>& clientDataHash, const WebCore::PublicKeyCredentialCreationOptions&, const WebCore::PublicKeyCredentialDescriptor&);
64
65// Extracts APDU encoded U2F sign command from PublicKeyCredentialRequestOptions.
66WEBCORE_EXPORT std::optional<Vector<uint8_t>> convertToU2fSignCommand(const Vector<uint8_t>& clientDataHash, const WebCore::PublicKeyCredentialRequestOptions&, const Vector<uint8_t>& keyHandle, bool isAppId = false);
67
68WEBCORE_EXPORT Vector<uint8_t> constructBogusU2fRegistrationCommand();

71WEBCORE_EXPORT String processGoogleLegacyAppIdSupportExtension(const std::optional<WebCore::AuthenticationExtensionsClientInputs>&);
