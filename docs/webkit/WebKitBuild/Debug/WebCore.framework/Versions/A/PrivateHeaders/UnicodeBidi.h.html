26#ifndef UnicodeBidi_h
27#define UnicodeBidi_h
28
29namespace WebCore {
30
31enum EUnicodeBidi {
32 UBNormal,
33 Embed,
34 Override,
35 Isolate,
36 Plaintext,
37 IsolateOverride,
38};
39
40inline bool isIsolated(const EUnicodeBidi& unicodeBidi)
41{
42 return unicodeBidi == Isolate || unicodeBidi == IsolateOverride || unicodeBidi == Plaintext;
43}
44
45inline bool isOverride(EUnicodeBidi unicodeBidi)
46{
47 return unicodeBidi == Override || unicodeBidi == IsolateOverride;
48}
49
50}
