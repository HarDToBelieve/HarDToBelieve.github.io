53class UniqueIDBDatabaseTransaction : public RefCounted<UniqueIDBDatabaseTransaction> {
54public:
55 static Ref<UniqueIDBDatabaseTransaction> create(UniqueIDBDatabaseConnection&, const IDBTransactionInfo&);
56
57 ~UniqueIDBDatabaseTransaction();
58
59 UniqueIDBDatabaseConnection& databaseConnection() { return *m_databaseConnection; }
60 const IDBTransactionInfo& info() const { return m_transactionInfo; }
61 bool isVersionChange() const;
62 bool isReadOnly() const;
63
64 IDBDatabaseInfo* originalDatabaseInfo() const;
65
66 void abort();
67 void abortWithoutCallback();
68 void commit();
69
70 void createObjectStore(const IDBRequestData&, const IDBObjectStoreInfo&);
71 void deleteObjectStore(const IDBRequestData&, const String& objectStoreName);
72 void renameObjectStore(const IDBRequestData&, uint64_t objectStoreIdentifier, const String& newName);
73 void clearObjectStore(const IDBRequestData&, uint64_t objectStoreIdentifier);
74 void createIndex(const IDBRequestData&, const IDBIndexInfo&);
75 void deleteIndex(const IDBRequestData&, uint64_t objectStoreIdentifier, const String& indexName);
76 void renameIndex(const IDBRequestData&, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const String& newName);
77 void putOrAdd(const IDBRequestData&, const IDBKeyData&, const IDBValue&, IndexedDB::ObjectStoreOverwriteMode);
78 void getRecord(const IDBRequestData&, const IDBGetRecordData&);
79 void getAllRecords(const IDBRequestData&, const IDBGetAllRecordsData&);
80 void getCount(const IDBRequestData&, const IDBKeyRangeData&);
81 void deleteRecord(const IDBRequestData&, const IDBKeyRangeData&);
82 void openCursor(const IDBRequestData&, const IDBCursorInfo&);
83 void iterateCursor(const IDBRequestData&, const IDBIterateCursorData&);
84
85 void didActivateInBackingStore(const IDBError&);
86
87 const Vector<uint64_t>& objectStoreIdentifiers();
88
89 void setMainThreadAbortResult(const IDBError& error) { m_mainThreadAbortResult = { error }; }
90 const std::optional<IDBError>& mainThreadAbortResult() const { return m_mainThreadAbortResult; }
91
92private:
93 UniqueIDBDatabaseTransaction(UniqueIDBDatabaseConnection&, const IDBTransactionInfo&);
94
95 UniqueIDBDatabaseConnection* m_databaseConnection;
96 IDBTransactionInfo m_transactionInfo;
97
98 std::unique_ptr<IDBDatabaseInfo> m_originalDatabaseInfo;
99
100 Vector<uint64_t> m_objectStoreIdentifiers;
101
102 std::optional<IDBError> m_mainThreadAbortResult;
103};
