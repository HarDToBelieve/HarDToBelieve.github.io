39namespace WebCore {

41class DOMWrapperWorld;
42class DocumentLoader;
43class Page;
44class UserContentProvider;
45class UserMessageHandlerDescriptor;

49class UserContentProviderInvalidationClient : public CanMakeWeakPtr<UserContentProviderInvalidationClient> {
50public:
51 virtual ~UserContentProviderInvalidationClient()
52 {
53 }
54
55 virtual void didInvalidate(UserContentProvider&) = 0;
56};

58class UserContentProvider : public RefCounted<UserContentProvider> {
59public:
60 WEBCORE_EXPORT UserContentProvider();
61 WEBCORE_EXPORT virtual ~UserContentProvider();
62
63 virtual void forEachUserScript(Function<void(DOMWrapperWorld&, const UserScript&)>&&) const = 0;
64 virtual void forEachUserStyleSheet(Function<void(const UserStyleSheet&)>&&) const = 0;
65#if ENABLE(USER_MESSAGE_HANDLERS)
66 virtual void forEachUserMessageHandler(Function<void(const UserMessageHandlerDescriptor&)>&&) const = 0;
67#endif
68#if ENABLE(CONTENT_EXTENSIONS)
69 virtual ContentExtensions::ContentExtensionsBackend& userContentExtensionBackend() = 0;
70#endif
71
72 void registerForUserMessageHandlerInvalidation(UserContentProviderInvalidationClient&);
73 void unregisterForUserMessageHandlerInvalidation(UserContentProviderInvalidationClient&);
74
75 void addPage(Page&);
76 void removePage(Page&);
77
78#if ENABLE(CONTENT_EXTENSIONS)
79 ContentRuleListResults processContentRuleListsForLoad(Page&, const URL&, OptionSet<ContentExtensions::ResourceType>, DocumentLoader& initiatingDocumentLoader, const URL& redirectFrom = { });
80#endif
81
82protected:
83 WEBCORE_EXPORT void invalidateAllRegisteredUserMessageHandlerInvalidationClients();
84 WEBCORE_EXPORT void invalidateInjectedStyleSheetCacheInAllFramesInAllPages();
85
86private:
87 WeakHashSet<Page> m_pages;
88 WeakHashSet<UserContentProviderInvalidationClient> m_userMessageHandlerInvalidationClients;
89};
