37namespace WebCore {

39class Document;
40
41enum ProcessingUserGestureState {
42 ProcessingUserGesture,
43 ProcessingPotentialUserGesture,
44 NotProcessingUserGesture
45};
46
47enum class UserGestureType { EscapeKey, Other };
48
49class UserGestureToken : public RefCounted<UserGestureToken>, public CanMakeWeakPtr<UserGestureToken> {
50public:
51 static constexpr Seconds maximumIntervalForUserGestureForwarding { 1_s }; // One second matches Gecko.
52 static const Seconds& maximumIntervalForUserGestureForwardingForFetch();
53 WEBCORE_EXPORT static void setMaximumIntervalForUserGestureForwardingForFetchForTesting(Seconds);
54
55 static Ref<UserGestureToken> create(ProcessingUserGestureState state, UserGestureType gestureType, Document* document = nullptr)
56 {
57 return adoptRef(*new UserGestureToken(state, gestureType, document));
58 }
59
60 WEBCORE_EXPORT ~UserGestureToken();
61
62 ProcessingUserGestureState state() const { return m_state; }
63 bool processingUserGesture() const { return m_scope == GestureScope::All && m_state == ProcessingUserGesture; }
64 bool processingUserGestureForMedia() const { return m_state == ProcessingUserGesture || m_state == ProcessingPotentialUserGesture; }
65 UserGestureType gestureType() const { return m_gestureType; }
66
67 void addDestructionObserver(WTF::Function<void (UserGestureToken&)>&& observer)
68 {
69 m_destructionObservers.append(WTFMove(observer));
70 }
71
72 DOMPasteAccessPolicy domPasteAccessPolicy() const { return m_domPasteAccessPolicy; }
73 void didRequestDOMPasteAccess(DOMPasteAccessResponse response)
74 {
75 switch (response) {
76 case DOMPasteAccessResponse::DeniedForGesture:
77 m_domPasteAccessPolicy = DOMPasteAccessPolicy::Denied;
78 break;
79 case DOMPasteAccessResponse::GrantedForCommand:
80 break;
81 case DOMPasteAccessResponse::GrantedForGesture:
82 m_domPasteAccessPolicy = DOMPasteAccessPolicy::Granted;
83 break;
84 }
85 }
86 void resetDOMPasteAccess() { m_domPasteAccessPolicy = DOMPasteAccessPolicy::NotRequestedYet; }
87
88 enum class GestureScope { All, MediaOnly };
89 void setScope(GestureScope scope) { m_scope = scope; }
90 void resetScope() { m_scope = GestureScope::All; }
91
92 // Expand the following methods if more propagation sources are added later.
93 enum class IsPropagatedFromFetch { Yes, No };
94 void setIsPropagatedFromFetch(IsPropagatedFromFetch is) { m_isPropagatedFromFetch = is; }
95 void resetIsPropagatedFromFetch() { m_isPropagatedFromFetch = IsPropagatedFromFetch::No; }
96 bool isPropagatedFromFetch() const { return m_isPropagatedFromFetch == IsPropagatedFromFetch::Yes; }
97
98 bool hasExpired(Seconds expirationInterval) const
99 {
100 return m_startTime + expirationInterval < MonotonicTime::now();
101 }
102
103 MonotonicTime startTime() const { return m_startTime; }
104
105 bool isValidForDocument(const Document&) const;
106
107private:
108 UserGestureToken(ProcessingUserGestureState, UserGestureType, Document*);
109
110 ProcessingUserGestureState m_state = NotProcessingUserGesture;
111 Vector<WTF::Function<void (UserGestureToken&)>> m_destructionObservers;
112 UserGestureType m_gestureType;
113 WeakHashSet<Document> m_documentsImpactedByUserGesture;
114 DOMPasteAccessPolicy m_domPasteAccessPolicy { DOMPasteAccessPolicy::NotRequestedYet };
115 GestureScope m_scope { GestureScope::All };
116 MonotonicTime m_startTime { MonotonicTime::now() };
117 IsPropagatedFromFetch m_isPropagatedFromFetch { IsPropagatedFromFetch::No };
118};
119
120class UserGestureIndicator {
121 WTF_MAKE_FAST_ALLOCATED;
122 WTF_MAKE_NONCOPYABLE(UserGestureIndicator);
123public:
124 WEBCORE_EXPORT static RefPtr<UserGestureToken> currentUserGesture();
125
126 WEBCORE_EXPORT static bool processingUserGesture(const Document* = nullptr);
127 WEBCORE_EXPORT static bool processingUserGestureForMedia();
128
129 // If a document is provided, its last known user gesture timestamp is updated.
130 enum class ProcessInteractionStyle { Immediate, Delayed };
131 WEBCORE_EXPORT explicit UserGestureIndicator(std::optional<ProcessingUserGestureState>, Document* = nullptr, UserGestureType = UserGestureType::Other, ProcessInteractionStyle = ProcessInteractionStyle::Immediate);
132 WEBCORE_EXPORT explicit UserGestureIndicator(RefPtr<UserGestureToken>, UserGestureToken::GestureScope = UserGestureToken::GestureScope::All, UserGestureToken::IsPropagatedFromFetch = UserGestureToken::IsPropagatedFromFetch::No);
133 WEBCORE_EXPORT ~UserGestureIndicator();
134
135private:
136 RefPtr<UserGestureToken> m_previousToken;
137};
