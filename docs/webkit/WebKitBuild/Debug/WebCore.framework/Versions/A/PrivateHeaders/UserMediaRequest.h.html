37#include "ActiveDOMObject.h"

43#include <wtf/CompletionHandler.h>
44#include <wtf/ObjectIdentifier.h>
45#include <wtf/UniqueRef.h>
46
47namespace WebCore {
48
49class MediaStream;
50class SecurityOrigin;
51
52template<typename IDLType> class DOMPromiseDeferred;
53
54enum UserMediaRequestIdentifierType { };
55using UserMediaRequestIdentifier = ObjectIdentifier<UserMediaRequestIdentifierType>;
56
57class UserMediaRequest : public RefCounted<UserMediaRequest>, public ActiveDOMObject {
58public:
59 static Ref<UserMediaRequest> create(Document&, MediaStreamRequest&&, DOMPromiseDeferred<IDLInterface<MediaStream>>&&);
60 virtual ~UserMediaRequest();
61
62 UserMediaRequestIdentifier identifier() const { return m_identifier; }
63 void start();
64
65 WEBCORE_EXPORT void setAllowedMediaDeviceUIDs(const String& audioDeviceUID, const String& videoDeviceUID);
66 WEBCORE_EXPORT void allow(CaptureDevice&& audioDevice, CaptureDevice&& videoDevice, String&& deviceIdentifierHashSalt, CompletionHandler<void()>&&);
67
68 enum MediaAccessDenialReason { NoConstraints, UserMediaDisabled, NoCaptureDevices, InvalidConstraint, HardwareError, PermissionDenied, InvalidAccess, IllegalConstraint, OtherFailure };
69 WEBCORE_EXPORT void deny(MediaAccessDenialReason, const String& errorMessage = emptyString());
70
71 const Vector<String>& audioDeviceUIDs() const { return m_audioDeviceUIDs; }
72 const Vector<String>& videoDeviceUIDs() const { return m_videoDeviceUIDs; }
73
74 const MediaConstraints& audioConstraints() const { return m_request.audioConstraints; }
75 const MediaConstraints& videoConstraints() const { return m_request.videoConstraints; }
76
77 WEBCORE_EXPORT SecurityOrigin* userMediaDocumentOrigin() const;
78 WEBCORE_EXPORT SecurityOrigin* topLevelDocumentOrigin() const;
79 WEBCORE_EXPORT Document* document() const;
80
81 const MediaStreamRequest& request() const { return m_request; }
82
83private:
84 UserMediaRequest(Document&, MediaStreamRequest&&, DOMPromiseDeferred<IDLInterface<MediaStream>>&&);
85
86 void stop() final;
87 const char* activeDOMObjectName() const final;
88
89 void mediaStreamDidFail(RealtimeMediaSource::Type);
90
91 UserMediaRequestIdentifier m_identifier;
92
93 Vector<String> m_videoDeviceUIDs;
94 Vector<String> m_audioDeviceUIDs;
95
96 UniqueRef<DOMPromiseDeferred<IDLInterface<MediaStream>>> m_promise;
97 CompletionHandler<void()> m_allowCompletionHandler;
98 MediaStreamRequest m_request;
99};
100
101} // namespace WebCore
102
103#endif // ENABLE(MEDIA_STREAM)
