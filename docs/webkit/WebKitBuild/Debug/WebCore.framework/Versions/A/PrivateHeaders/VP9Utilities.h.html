130struct ScreenDataOverrides {
131 double width { 0 };
132 double height { 0 };
133 double scale { 1 };
134
135 template<class Encoder>
136 void encode(Encoder& encoder) const
137 {
138 encoder << width;
139 encoder << height;
140 encoder << scale;
141 }
142
143 template <class Decoder>
144 static std::optional<ScreenDataOverrides> decode(Decoder& decoder)
145 {
146#define DECODE(name, type) \
147 std::optional<type> name; \
148 decoder >> name; \
149 if (!name) \
150 return std::nullopt; \
151
152 DECODE(width, double);
153 DECODE(height, double);
154 DECODE(scale, double);
155#undef DECODE
156
157 return {{ WTFMove(*width), WTFMove(*height), WTFMove(*scale) }};
158 }
159};
160
161inline bool operator==(const ScreenDataOverrides& a, const ScreenDataOverrides& b)
162{
163 return a.width == b.width && a.height == b.height && a.scale == b.scale;
164}
165
166inline bool operator!=(const ScreenDataOverrides& a, const ScreenDataOverrides& b)
167{
168 return !(a == b);
169}
