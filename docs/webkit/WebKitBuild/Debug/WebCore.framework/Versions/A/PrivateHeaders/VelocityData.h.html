37struct VelocityData {
38 float horizontalVelocity { 0 };
39 float verticalVelocity { 0 };
40 float scaleChangeRate { 0 };
41 MonotonicTime lastUpdateTime;
42
43 VelocityData(float horizontal = 0, float vertical = 0, float scaleChange = 0, MonotonicTime updateTime = MonotonicTime())
44 : horizontalVelocity(horizontal)
45 , verticalVelocity(vertical)
46 , scaleChangeRate(scaleChange)
47 , lastUpdateTime(updateTime)
48 {
49 }
50
51 bool velocityOrScaleIsChanging() const
52 {
53 return horizontalVelocity || verticalVelocity || scaleChangeRate;
54 }
55
56 bool equalIgnoringTimestamp(const VelocityData& other) const
57 {
58 return horizontalVelocity == other.horizontalVelocity
59 && verticalVelocity == other.verticalVelocity
60 && scaleChangeRate == other.scaleChangeRate;
61 }
62};

64WEBCORE_EXPORT TextStream& operator<<(TextStream&, const VelocityData&);
