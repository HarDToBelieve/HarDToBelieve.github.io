32namespace WebCore {
33
34enum class SelectionDirection : uint8_t { Forward, Backward, Right, Left };
35
36class VisibleSelection {
37public:
38 WEBCORE_EXPORT VisibleSelection();
39
40 static constexpr auto defaultAffinity = VisiblePosition::defaultAffinity;
41
42 VisibleSelection(const Position& anchor, const Position& focus, Affinity = defaultAffinity, bool isDirectional = false);
43
44 VisibleSelection(const Position&, Affinity, bool isDirectional = false);
45 WEBCORE_EXPORT VisibleSelection(const SimpleRange&, Affinity = defaultAffinity, bool isDirectional = false);
46 WEBCORE_EXPORT VisibleSelection(const VisiblePosition&, bool isDirectional = false);
47 WEBCORE_EXPORT VisibleSelection(const VisiblePosition& anchor, const VisiblePosition& focus, bool isDirectional = false);
48
49 WEBCORE_EXPORT static VisibleSelection selectionFromContentsOfNode(Node*);
50
51 void setAffinity(Affinity affinity) { m_affinity = affinity; }
52 Affinity affinity() const { return m_affinity; }
53
54 // FIXME: Move to the terms "focus" and "anchor" instead of "base" and "extent".
55
56 void setBase(const Position&);
57 void setBase(const VisiblePosition&);
58 void setExtent(const Position&);
59 void setExtent(const VisiblePosition&);
60
61 // These functions return the values that were passed in, without the canonicalization done by VisiblePosition.
62 // FIXME: When we expand granularity, we canonicalize as a side effect, so expanded values have been made canonical.
63 // FIXME: Replace start/range/base/end/firstRange with these, renaming these to the shorter names.
64 Position uncanonicalizedStart() const;
65 Position uncanonicalizedEnd() const;
66 Position anchor() const;
67 Position focus() const;
68 WEBCORE_EXPORT std::optional<SimpleRange> range() const;
69
70 // FIXME: Rename these to include the word "canonical" or remove.
71 Position base() const { return m_base; }
72 Position extent() const { return m_extent; }
73 Position start() const { return m_start; }
74 Position end() const { return m_end; }
75
76 VisiblePosition visibleStart() const { return VisiblePosition(m_start, isRange() ? Affinity::Downstream : affinity()); }
77 VisiblePosition visibleEnd() const { return VisiblePosition(m_end, isRange() ? Affinity::Upstream : affinity()); }
78 VisiblePosition visibleBase() const { return VisiblePosition(m_base, isRange() ? (isBaseFirst() ? Affinity::Upstream : Affinity::Downstream) : affinity()); }
79 VisiblePosition visibleExtent() const { return VisiblePosition(m_extent, isRange() ? (isBaseFirst() ? Affinity::Downstream : Affinity::Upstream) : affinity()); }
80
81 operator VisiblePositionRange() const { return { visibleStart(), visibleEnd() }; }
82
83 bool isNone() const { return m_type == Type::None; }
84 bool isCaret() const { return m_type == Type::Caret; }
85 bool isRange() const { return m_type == Type::Range; }
86 bool isCaretOrRange() const { return !isNone(); }
87 bool isNonOrphanedRange() const { return isRange() && !start().isOrphan() && !end().isOrphan(); }
88 bool isNoneOrOrphaned() const { return isNone() || start().isOrphan() || end().isOrphan(); }
89 bool isOrphan() const;
90 RefPtr<Document> document() const;
91
92 bool isBaseFirst() const { return m_anchorIsFirst; }
93 bool isDirectional() const { return m_isDirectional; }
94 void setIsDirectional(bool isDirectional) { m_isDirectional = isDirectional; }
95
96 WEBCORE_EXPORT bool isAll(EditingBoundaryCrossingRule) const;
97
98 void appendTrailingWhitespace();
99
100 WEBCORE_EXPORT bool expandUsingGranularity(TextGranularity granularity);
101
102 // FIXME: Rename to include the word "canonical" and remove the word "first" or remove.
103 // We don't yet support multi-range selections, so we only ever have one range to return.
104 WEBCORE_EXPORT std::optional<SimpleRange> firstRange() const;
105
106 // FIXME: Most callers probably don't want this function, and should use range instead.
107 // This function contracts the range around text and moves the caret upstream before returning the range.
108 WEBCORE_EXPORT std::optional<SimpleRange> toNormalizedRange() const;
109
110 WEBCORE_EXPORT Element* rootEditableElement() const;
111 WEBCORE_EXPORT bool isContentEditable() const;
112 WEBCORE_EXPORT bool hasEditableStyle() const;
113 WEBCORE_EXPORT bool isContentRichlyEditable() const;
114
115 // Returns a shadow tree node for legacy shadow trees, a child of the
116 // ShadowRoot node for new shadow trees, or 0 for non-shadow trees.
117 Node* nonBoundaryShadowTreeRootNode() const;
118
119 WEBCORE_EXPORT bool isInPasswordField() const;
120
121 WEBCORE_EXPORT static Position adjustPositionForEnd(const Position& currentPosition, Node* startContainerNode);
122 WEBCORE_EXPORT static Position adjustPositionForStart(const Position& currentPosition, Node* startContainerNode);
123
124#if ENABLE(TREE_DEBUGGING)
125 void debugPosition() const;
126 String debugDescription() const;
127 void showTreeForThis() const;
128#endif
129
130 void setWithoutValidation(const Position&, const Position&);
131
132private:
133 void validate(TextGranularity = TextGranularity::CharacterGranularity);
134
135 // Support methods for validate()
136 void setBaseAndExtentToDeepEquivalents();
137 void adjustSelectionRespectingGranularity(TextGranularity);
138 void adjustSelectionToAvoidCrossingShadowBoundaries();
139 void adjustSelectionToAvoidCrossingEditingBoundaries();
140 void updateSelectionType();
141
142 // We store only Positions because VisibleSelection is used to store values in
143 // editing commands for use when undoing the command. We need to be able to store
144 // a selection that, while currently invalid, will be valid once the changes are undone.
145
146 // FIXME: Consider doing canonicalization only as part of editing operations, and keeping all selection endpoints non-canonical outside of that code.
147 // FIXME: Rename m_base to m_canonicalAnchor.
148 // FIXME: Rename m_extent to m_canonicalFocus.
149 Position m_anchor; // Where the first click happened.
150 Position m_focus; // Where the end click/release happened.
151 Position m_base; // Anchor, canonical, not yet expended to respect granularity.
152 Position m_extent; // Focus, canonical, but not expended to respect granularity.
153 Position m_start; // First of anchor and focus, caonicalized and expanded to respect granularity.
154 Position m_end; // Last of anchor and focus, caonicalized and expanded to respect granularity.
155
156 Affinity m_affinity { defaultAffinity };
157
158 // These are cached, can be recalculated by validate()
159 enum class Type : uint8_t { None, Caret, Range };
160 Type m_type { Type::None };
161 bool m_anchorIsFirst : 1; // True if the anchor is before the focus.
162 bool m_isDirectional : 1; // On Mac, Shift-arrow keys move the anchor in a directional selection and moves either end to always extend in a non-directional selection.
163};
164
165inline bool operator==(const VisibleSelection& a, const VisibleSelection& b)
166{
167 return a.start() == b.start() && a.end() == b.end() && a.affinity() == b.affinity() && a.isBaseFirst() == b.isBaseFirst() && a.isDirectional() == b.isDirectional();
168}
169
170inline bool operator!=(const VisibleSelection& a, const VisibleSelection& b)
171{
172 return !(a == b);
173}
