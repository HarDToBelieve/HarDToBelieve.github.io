33struct AudioBuffer;
34struct AudioBufferList;

38namespace WebCore {
39
40class CAAudioStreamDescription;
41
42class WebAudioBufferList final : public PlatformAudioData {
43public:
44 WEBCORE_EXPORT WebAudioBufferList(const CAAudioStreamDescription&);
45 WEBCORE_EXPORT WebAudioBufferList(const CAAudioStreamDescription&, uint32_t sampleCount);
46 WebAudioBufferList(const CAAudioStreamDescription&, CMSampleBufferRef);
47
48 void reset();
49 WEBCORE_EXPORT void setSampleCount(uint32_t);
50
51 AudioBufferList* list() const { return m_list.get(); }
52 operator AudioBufferList&() const { return *m_list; }
53
54 uint32_t bufferCount() const;
55 uint32_t channelCount() const { return m_channelCount; }
56 AudioBuffer* buffer(uint32_t index) const;
57 WTF::IteratorRange<AudioBuffer*> buffers() const;
58
59 WEBCORE_EXPORT static bool isSupportedDescription(const CAAudioStreamDescription&, uint32_t sampleCount);
60
61 WEBCORE_EXPORT void zeroFlatBuffer();
62
63private:
64 Kind kind() const { return Kind::WebAudioBufferList; }
65
66 size_t m_listBufferSize { 0 };
67 uint32_t m_bytesPerFrame { 0 };
68 uint32_t m_channelCount { 0 };
69 uint32_t m_sampleCount { 0 };
70 std::unique_ptr<AudioBufferList> m_canonicalList;
71 std::unique_ptr<AudioBufferList> m_list;
72 RetainPtr<CMBlockBufferRef> m_blockBuffer;
73 Vector<uint8_t> m_flatBuffer;
74};
75
76}
