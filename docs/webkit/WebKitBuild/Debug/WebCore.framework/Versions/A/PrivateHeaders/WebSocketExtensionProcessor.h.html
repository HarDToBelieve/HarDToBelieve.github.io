31#pragma once
32
33#include <wtf/HashMap.h>
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38class WebSocketExtensionProcessor {
39public:
40 virtual ~WebSocketExtensionProcessor() = default;
41
42 String extensionToken() const { return m_extensionToken; }
43
44 // The return value of this method will be a part of the value of
45 // Sec-WebSocket-Extensions.
46 virtual String handshakeString() = 0;
47
48 // This should validate the server's response parameters which are passed
49 // as HashMap<key, value>. This may also do something for the extension.
50 // Note that this method may be called more than once when the server
51 // response contains duplicate extension token that matches extensionToken().
52 virtual bool processResponse(const HashMap<String, String>&) = 0;
53
54 // If procecssResponse() returns false, this should provide the reason.
55 virtual String failureReason() { return "Extension " + m_extensionToken + " failed"; }
56
57protected:
58 explicit WebSocketExtensionProcessor(const String& extensionToken)
59 : m_extensionToken(extensionToken)
60 {
61 }
62
63private:
64 String m_extensionToken;
65};
66
67} // namespace WebCore
