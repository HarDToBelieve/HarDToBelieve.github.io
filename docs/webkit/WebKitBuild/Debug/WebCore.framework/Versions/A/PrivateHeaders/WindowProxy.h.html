21#pragma once

23#include <JavaScriptCore/Strong.h>
24#include <wtf/HashMap.h>
25#include <wtf/RefCounted.h>
26#include <wtf/UniqueRef.h>
27#include <wtf/WeakPtr.h>
28
29namespace JSC {
30class Debugger;
31}
32
33namespace WebCore {
34
35class AbstractDOMWindow;
36class AbstractFrame;
37class DOMWrapperWorld;
38class JSDOMGlobalObject;
39class JSWindowProxy;
40
41class WindowProxy : public RefCounted<WindowProxy> {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 using ProxyMap = HashMap<RefPtr<DOMWrapperWorld>, JSC::Strong<JSWindowProxy>>;
45
46 static Ref<WindowProxy> create(AbstractFrame& frame)
47 {
48 return adoptRef(*new WindowProxy(frame));
49 }
50
51 WEBCORE_EXPORT ~WindowProxy();
52
53 WEBCORE_EXPORT AbstractFrame* frame() const;
54 void detachFromFrame();
55
56 void destroyJSWindowProxy(DOMWrapperWorld&);
57
58 ProxyMap::ValuesConstIteratorRange jsWindowProxies() const;
59 Vector<JSC::Strong<JSWindowProxy>> jsWindowProxiesAsVector() const;
60
61 WEBCORE_EXPORT ProxyMap releaseJSWindowProxies();
62 WEBCORE_EXPORT void setJSWindowProxies(ProxyMap&&);
63
64 JSWindowProxy* jsWindowProxy(DOMWrapperWorld& world)
65 {
66 if (!m_frame)
67 return nullptr;
68
69 if (auto* existingProxy = existingJSWindowProxy(world))
70 return existingProxy;
71
72 return &createJSWindowProxyWithInitializedScript(world);
73 }
74
75 JSWindowProxy* existingJSWindowProxy(DOMWrapperWorld& world) const
76 {
77 auto it = m_jsWindowProxies->find(&world);
78 return (it != m_jsWindowProxies->end()) ? it->value.get() : nullptr;
79 }
80
81 WEBCORE_EXPORT JSDOMGlobalObject* globalObject(DOMWrapperWorld&);
82
83 void clearJSWindowProxiesNotMatchingDOMWindow(AbstractDOMWindow*, bool goingIntoBackForwardCache);
84
85 WEBCORE_EXPORT void setDOMWindow(AbstractDOMWindow*);
86
87 // Debugger can be nullptr to detach any existing Debugger.
88 void attachDebugger(JSC::Debugger*); // Attaches/detaches in all worlds/window proxies.
89
90 WEBCORE_EXPORT AbstractDOMWindow* window() const;
91
92private:
93 explicit WindowProxy(AbstractFrame&);
94
95 JSWindowProxy& createJSWindowProxy(DOMWrapperWorld&);
96 WEBCORE_EXPORT JSWindowProxy& createJSWindowProxyWithInitializedScript(DOMWrapperWorld&);
97
98 WeakPtr<AbstractFrame> m_frame;
99 UniqueRef<ProxyMap> m_jsWindowProxies;
100};
101
102} // namespace WebCore
