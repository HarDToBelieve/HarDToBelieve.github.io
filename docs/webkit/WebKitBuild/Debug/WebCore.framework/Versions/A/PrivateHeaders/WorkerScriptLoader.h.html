45namespace WebCore {
46
47class Exception;
48class ScriptExecutionContext;
49class TextResourceDecoder;
50class WorkerScriptLoaderClient;
51enum class CertificateInfoPolicy : uint8_t;
52
53class WorkerScriptLoader : public RefCounted<WorkerScriptLoader>, public ThreadableLoaderClient {
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 static Ref<WorkerScriptLoader> create()
57 {
58 return adoptRef(*new WorkerScriptLoader);
59 }
60
61 std::optional<Exception> loadSynchronously(ScriptExecutionContext*, const URL&, FetchOptions::Mode, FetchOptions::Cache, ContentSecurityPolicyEnforcement, const String& initiatorIdentifier);
62 void loadAsynchronously(ScriptExecutionContext&, ResourceRequest&&, FetchOptions&&, ContentSecurityPolicyEnforcement, ServiceWorkersMode, WorkerScriptLoaderClient&, String&& taskMode);
63
64 void notifyError();
65
66 const ScriptBuffer& script() { return m_script; }
67 const ContentSecurityPolicyResponseHeaders& contentSecurityPolicy() const { return m_contentSecurityPolicy; }
68 const String& referrerPolicy() const { return m_referrerPolicy; }
69 const URL& url() const { return m_url; }
70 const URL& responseURL() const;
71 ResourceResponse::Source responseSource() const { return m_responseSource; }
72 bool isRedirected() const { return m_isRedirected; }
73 const CertificateInfo& certificateInfo() const { return m_certificateInfo; }
74 const String& responseMIMEType() const { return m_responseMIMEType; }
75 bool failed() const { return m_failed; }
76 unsigned long identifier() const { return m_identifier; }
77 const ResourceError& error() const { return m_error; }
78
79 void didReceiveResponse(unsigned long identifier, const ResourceResponse&) override;
80 void didReceiveData(const uint8_t* data, int dataLength) override;
81 void didFinishLoading(unsigned long identifier) override;
82 void didFail(const ResourceError&) override;
83
84 void cancel();
85
86 WEBCORE_EXPORT static ResourceError validateWorkerResponse(const ResourceResponse&, FetchOptions::Destination);
87
88private:
89 friend class WTF::RefCounted<WorkerScriptLoader>;
90 friend struct std::default_delete<WorkerScriptLoader>;
91
92 WorkerScriptLoader();
93 ~WorkerScriptLoader();
94
95 std::unique_ptr<ResourceRequest> createResourceRequest(const String& initiatorIdentifier);
96 void notifyFinished();
97
98 WorkerScriptLoaderClient* m_client { nullptr };
99 RefPtr<ThreadableLoader> m_threadableLoader;
100 String m_responseEncoding;
101 RefPtr<TextResourceDecoder> m_decoder;
102 ScriptBuffer m_script;
103 URL m_url;
104 URL m_responseURL;
105 CertificateInfo m_certificateInfo;
106 String m_responseMIMEType;
107 FetchOptions::Destination m_destination;
108 ContentSecurityPolicyResponseHeaders m_contentSecurityPolicy;
109 String m_referrerPolicy;
110 unsigned long m_identifier { 0 };
111 bool m_failed { false };
112 bool m_finishing { false };
113 bool m_isRedirected { false };
114 ResourceResponse::Source m_responseSource { ResourceResponse::Source::Unknown };
115 ResourceError m_error;
116};
