26#ifndef KJS_RUNTIME_OBJECT_H
27#define KJS_RUNTIME_OBJECT_H
28
29#include "BridgeJSC.h"
30#include <JavaScriptCore/JSGlobalObject.h>
31
32namespace JSC {
33namespace Bindings {
34
35Exception* throwRuntimeObjectInvalidAccessError(JSGlobalObject*, ThrowScope&);
36
37class WEBCORE_EXPORT RuntimeObject : public JSNonFinalObject {
38public:
39 using Base = JSNonFinalObject;
40 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetOwnPropertyNames | OverridesGetCallData | OverridesPut | GetOwnPropertySlotMayBeWrongAboutDontEnum;
41 static constexpr bool needsDestruction = true;
42
43 template<typename CellType, JSC::SubspaceAccess>
44 static IsoSubspace* subspaceFor(JSC::VM& vm)
45 {
46 static_assert(sizeof(CellType) == sizeof(RuntimeObject), "RuntimeObject subclasses that add fields need to override subspaceFor<>()");
47 static_assert(CellType::destroy == RuntimeObject::destroy);
48 return subspaceForImpl(vm);
49 }
50
51 static RuntimeObject* create(VM& vm, Structure* structure, RefPtr<Instance>&& instance)
52 {
53 RuntimeObject* object = new (NotNull, allocateCell<RuntimeObject>(vm.heap)) RuntimeObject(vm, structure, WTFMove(instance));
54 object->finishCreation(vm);
55 return object;
56 }
57
58 static void destroy(JSCell*);
59
60 static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
61 static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
62 static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
63 static CallData getCallData(JSCell*);
64 static CallData getConstructData(JSCell*);
65
66 static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, DontEnumPropertiesMode);
67
68 void invalidate();
69
70 Instance* getInternalInstance() const { return m_instance.get(); }
71
72 DECLARE_INFO;
73
74 static ObjectPrototype* createPrototype(VM&, JSGlobalObject& globalObject)
75 {
76 return globalObject.objectPrototype();
77 }
78
79 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
80 {
81 return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
82 }
83
84protected:
85 RuntimeObject(VM&, Structure*, RefPtr<Instance>&&);
86 void finishCreation(VM&);
87
88private:
89 static IsoSubspace* subspaceForImpl(VM&);
90
91 RefPtr<Instance> m_instance;
92};
