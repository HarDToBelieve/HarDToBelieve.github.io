39/*! @enum WKNavigationActionPolicy
40 @abstract The policy to pass back to the decision handler from the
41 webView:decidePolicyForNavigationAction:decisionHandler: method.
42 @constant WKNavigationActionPolicyCancel Cancel the navigation.
43 @constant WKNavigationActionPolicyAllow Allow the navigation to continue.
44 @constant WKNavigationActionPolicyDownload Turn the navigation into a download.
45 */
46typedef NS_ENUM(NSInteger, WKNavigationActionPolicy) {
47 WKNavigationActionPolicyCancel,
48 WKNavigationActionPolicyAllow,
49 WKNavigationActionPolicyDownload WK_API_AVAILABLE(macos(12.0), ios(15.0)),
50} WK_API_AVAILABLE(macos(10.10), ios(8.0));
51
52/*! @enum WKNavigationResponsePolicy
53 @abstract The policy to pass back to the decision handler from the webView:decidePolicyForNavigationResponse:decisionHandler: method.
54 @constant WKNavigationResponsePolicyCancel Cancel the navigation.
55 @constant WKNavigationResponsePolicyAllow Allow the navigation to continue.
56 @constant WKNavigationResponsePolicyDownload Turn the navigation into a download.

59 WKNavigationResponsePolicyCancel,
60 WKNavigationResponsePolicyAllow,
61 WKNavigationResponsePolicyDownload WK_API_AVAILABLE(macos(12.0), ios(15.0)),

64/*! A class conforming to the WKNavigationDelegate protocol can provide
65 methods for tracking progress for main frame navigations and for deciding
66 policy for main frame and subframe navigations.
67 */
68@protocol WKNavigationDelegate <NSObject>
69
70@optional
71
72/*! @abstract Decides whether to allow or cancel a navigation.
73 @param webView The web view invoking the delegate method.
74 @param navigationAction Descriptive information about the action
75 triggering the navigation request.
76 @param decisionHandler The decision handler to call to allow or cancel the
77 navigation. The argument is one of the constants of the enumerated type WKNavigationActionPolicy.
78 @discussion If you do not implement this method, the web view will load the request or, if appropriate, forward it to another application.
79 */
80- (void)webView:(WKWebView *)webView decidePolicyForNavigationAction:(WKNavigationAction *)navigationAction decisionHandler:(void (^)(WKNavigationActionPolicy))decisionHandler WK_SWIFT_ASYNC(3);
81
82/*! @abstract Decides whether to allow or cancel a navigation.
83 @param webView The web view invoking the delegate method.
84 @param navigationAction Descriptive information about the action
85 triggering the navigation request.
86 @param preferences The default set of webpage preferences. This may be
87 changed by setting defaultWebpagePreferences on WKWebViewConfiguration.
88 @param decisionHandler The policy decision handler to call to allow or cancel
89 the navigation. The arguments are one of the constants of the enumerated type
90 WKNavigationActionPolicy, as well as an instance of WKWebpagePreferences.
91 @discussion If you implement this method,
92 -webView:decidePolicyForNavigationAction:decisionHandler: will not be called.
93 */
94- (void)webView:(WKWebView *)webView decidePolicyForNavigationAction:(WKNavigationAction *)navigationAction preferences:(WKWebpagePreferences *)preferences decisionHandler:(void (^)(WKNavigationActionPolicy, WKWebpagePreferences *))decisionHandler WK_SWIFT_ASYNC(4) WK_API_AVAILABLE(macos(10.15), ios(13.0));
95
96/*! @abstract Decides whether to allow or cancel a navigation after its
97 response is known.
98 @param webView The web view invoking the delegate method.
99 @param navigationResponse Descriptive information about the navigation
100 response.
101 @param decisionHandler The decision handler to call to allow or cancel the
102 navigation. The argument is one of the constants of the enumerated type WKNavigationResponsePolicy.
103 @discussion If you do not implement this method, the web view will allow the response, if the web view can show it.
104 */
105- (void)webView:(WKWebView *)webView decidePolicyForNavigationResponse:(WKNavigationResponse *)navigationResponse decisionHandler:(void (^)(WKNavigationResponsePolicy))decisionHandler WK_SWIFT_ASYNC(3);
106
107/*! @abstract Invoked when a main frame navigation starts.
108 @param webView The web view invoking the delegate method.
109 @param navigation The navigation.
110 */
111- (void)webView:(WKWebView *)webView didStartProvisionalNavigation:(null_unspecified WKNavigation *)navigation;

124 @param error The error that occurred.
125 */
126- (void)webView:(WKWebView *)webView didFailProvisionalNavigation:(null_unspecified WKNavigation *)navigation withError:(NSError *)error;

138- (void)webView:(WKWebView *)webView didFinishNavigation:(null_unspecified WKNavigation *)navigation;

140/*! @abstract Invoked when an error occurs during a committed main frame

146- (void)webView:(WKWebView *)webView didFailNavigation:(null_unspecified WKNavigation *)navigation withError:(NSError *)error;
147
148/*! @abstract Invoked when the web view needs to respond to an authentication challenge.
149 @param webView The web view that received the authentication challenge.
150 @param challenge The authentication challenge.
151 @param completionHandler The completion handler you must invoke to respond to the challenge. The
152 disposition argument is one of the constants of the enumerated type
153 NSURLSessionAuthChallengeDisposition. When disposition is NSURLSessionAuthChallengeUseCredential,
154 the credential argument is the credential to use, or nil to indicate continuing without a
155 credential.
156 @discussion If you do not implement this method, the web view will respond to the authentication challenge with the NSURLSessionAuthChallengeRejectProtectionSpace disposition.
157 */
158- (void)webView:(WKWebView *)webView didReceiveAuthenticationChallenge:(NSURLAuthenticationChallenge *)challenge completionHandler:(void (^)(NSURLSessionAuthChallengeDisposition disposition, NSURLCredential * _Nullable credential))completionHandler WK_SWIFT_ASYNC_NAME(webView(_:respondTo:));
159
160/*! @abstract Invoked when the web view's web content process is terminated.
161 @param webView The web view whose underlying web content process was terminated.
162 */
163- (void)webViewWebContentProcessDidTerminate:(WKWebView *)webView WK_API_AVAILABLE(macos(10.11), ios(9.0));

165/*! @abstract Invoked when the web view is establishing a network connection using a deprecated version of TLS.
166 @param webView The web view initiating the connection.
167 @param challenge The authentication challenge.
168 @param decisionHandler The decision handler you must invoke to respond to indicate whether or not to continue with the connection establishment.
169 */
170- (void)webView:(WKWebView *)webView authenticationChallenge:(NSURLAuthenticationChallenge *)challenge shouldAllowDeprecatedTLS:(void (^)(BOOL))decisionHandler WK_SWIFT_ASYNC_NAME(webView(_:shouldAllowDeprecatedTLSFor:)) WK_SWIFT_ASYNC(3) WK_API_AVAILABLE(macos(11.0), ios(14.0));
