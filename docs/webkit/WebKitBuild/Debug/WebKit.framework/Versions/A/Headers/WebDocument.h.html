37/*!
38 @protocol WebDocumentView
39 @discussion Protocol implemented by the document view of WebFrameView
40*/
41WEBKIT_DEPRECATED_MAC(10_3, 10_14)
42@protocol WebDocumentView <NSObject>
43
44/*!
45 @method setDataSource:
46 @abstract Called when the corresponding data source has been created.
47 @param dataSource The corresponding data source.
48*/
49- (void)setDataSource:(WebDataSource *)dataSource;
50
51/*!
52 @method dataSourceUpdated:
53 @abstract Called when the corresponding data source has received data.
54 @param dataSource The corresponding data source.
55*/
56- (void)dataSourceUpdated:(WebDataSource *)dataSource;
57
58/*!
59 @method setNeedsLayout:
60 @discussion Called when WebKit has determined that the document view needs to layout.
61 This method should simply set a flag and call layout from drawRect if the flag is YES.
62 @param flag YES to cause a layout, no to not cause a layout.
63*/
64- (void)setNeedsLayout:(BOOL)flag;
65
66/*!
67 @method layout
68 @discussion Called when the document view must immediately layout. For simple views,
69 setting the frame is a sufficient implementation of this method.
70*/
71- (void)layout;
72
73/*!
74 @method viewWillMoveToHostWindow:
75 @param hostWindow The host window for the document view.
76 @abstract Called before the host window is set on the parent web view.
77*/
78- (void)viewWillMoveToHostWindow:(NSWindow *)hostWindow;
79
80/*!
81 @method viewDidMoveToHostWindow
82 @abstract Called after the host window is set on the parent web view.
83*/
84- (void)viewDidMoveToHostWindow;
85
86@end

89/*!
90 @protocol WebDocumentSearching
91 @discussion Optional protocol for searching document view of WebFrameView.
92*/
93WEBKIT_DEPRECATED_MAC(10_3, 10_14)
94@protocol WebDocumentSearching <NSObject>
95/*!
96 @method searchFor:direction:caseSensitive:wrap:
97 @abstract Searches a document view for a string and highlights the string if it is found.
98 @param string The string to search for.
99 @param forward YES to search forward, NO to seach backwards.
100 @param caseFlag YES to for case-sensitive search, NO for case-insensitive search.
101 @param wrapFlag YES to wrap around, NO to avoid wrapping.
102 @result YES if found, NO if not found.
103*/
104- (BOOL)searchFor:(NSString *)string direction:(BOOL)forward caseSensitive:(BOOL)caseFlag wrap:(BOOL)wrapFlag;
105@end

161/*!
162 @protocol WebDocumentRepresentation
163 @discussion Protocol implemented by the document representation of a data source.
164*/
165WEBKIT_DEPRECATED_MAC(10_3, 10_14)
166@protocol WebDocumentRepresentation <NSObject>
167/*!
168 @method setDataSource:
169 @abstract Called soon after the document representation is created.
170 @param dataSource The data source that is set.
171*/
172- (void)setDataSource:(WebDataSource *)dataSource;
173
174/*!
175 @method receivedData:withDataSource:
176 @abstract Called when the data source has received data.
177 @param data The data that the data source has received.
178 @param dataSource The data source that has received data.
179*/
180- (void)receivedData:(NSData *)data withDataSource:(WebDataSource *)dataSource;
181
182/*!
183 @method receivedError:withDataSource:
184 @abstract Called when the data source has received an error.
185 @param error The error that the data source has received.
186 @param dataSource The data source that has received the error.
187*/
188- (void)receivedError:(NSError *)error withDataSource:(WebDataSource *)dataSource;
189
190/*!
191 @method finishedLoadingWithDataSource:
192 @abstract Called when the data source has finished loading.
193 @param dataSource The datasource that has finished loading.
194*/
195- (void)finishedLoadingWithDataSource:(WebDataSource *)dataSource;
196
197/*!
198 @method canProvideDocumentSource
199 @result Returns true if the representation can provide document source.
200*/
201- (BOOL)canProvideDocumentSource;
202
203/*!
204 @method documentSource
205 @result Returns the textual source representation of the document. For HTML documents
206 this is the original HTML source.
207*/
208- (NSString *)documentSource;
209
210/*!
211 @method title
212 @result Return the title for the document.
213*/
214- (NSString *)title;
215
216@end
