29#import <Foundation/Foundation.h>
30#import <WebKit/WebKitAvailability.h>
31
32#ifndef WebDownload_h
33#define WebDownload_h
34
35#if (defined TARGET_OS_MACCATALYST && TARGET_OS_MACCATALYST)
36#import <CFNetwork/CFNSURLConnection.h>
37#else
38#import <Foundation/NSURLDownload.h>
39#endif
40
41@class NSWindow;
42@class WebDownloadInternal;
43
44/*!
45 @class WebDownload
46 @discussion A WebDownload works just like an NSURLDownload, with
47 one extra feature: if you do not implement the
48 authentication-related delegate methods, it will automatically
49 prompt for authentication using the standard WebKit authentication
50 panel, as either a sheet or window. It provides no extra methods,
51 but does have one additional delegate method.
52*/
53WEBKIT_CLASS_DEPRECATED_MAC(10_4, 10_14)
54@interface WebDownload : NSURLDownload
55{
56@package
57 WebDownloadInternal *_webInternal;
58}
59
60@end
61
62/*!
63 @protocol WebDownloadDelegate
64 @discussion The WebDownloadDelegate delegate has one extra method used to choose
65 the right window when automatically prompting with a sheet.
66*/
67WEBKIT_DEPRECATED_MAC(10_4, 10_14)
68@protocol WebDownloadDelegate <NSURLDownloadDelegate>
69
70@optional
71
72/*!
73 @method downloadWindowForAuthenticationSheet:
74*/
75- (NSWindow *)downloadWindowForAuthenticationSheet:(WebDownload *)download;
76
77@end
78
79#endif /* WebDownload_h */
