44/*!
45 @category WebFrameLoadDelegate
46 @discussion A WebView's WebFrameLoadDelegate tracks the loading progress of its frames.
47 When a data source of a frame starts to load, the data source is considered "provisional".
48 Once at least one byte is received, the data source is considered "committed". This is done
49 so the contents of the frame will not be lost if the new data source fails to successfully load.
50*/
51WEBKIT_DEPRECATED_MAC(10_3, 10_14)
52@protocol WebFrameLoadDelegate <NSObject>
53
54@optional
55
56/*!
57 @method webView:didStartProvisionalLoadForFrame:
58 @abstract Notifies the delegate that the provisional load of a frame has started
59 @param sender The WebView sending the message
60 @param frame The frame for which the provisional load has started
61 @discussion This method is called after the provisional data source of a frame
62 has started to load.
63*/
64- (void)webView:(WebView *)sender didStartProvisionalLoadForFrame:(WebFrame *)frame;
65
66/*!
67 @method webView:didReceiveServerRedirectForProvisionalLoadForFrame:
68 @abstract Notifies the delegate that a server redirect occurred during the provisional load
69 @param sender The WebView sending the message
70 @param frame The frame for which the redirect occurred
71*/
72- (void)webView:(WebView *)sender didReceiveServerRedirectForProvisionalLoadForFrame:(WebFrame *)frame;
73
74/*!
75 @method webView:didFailProvisionalLoadWithError:forFrame:
76 @abstract Notifies the delegate that the provisional load has failed
77 @param sender The WebView sending the message
78 @param error The error that occurred
79 @param frame The frame for which the error occurred
80 @discussion This method is called after the provisional data source has failed to load.
81 The frame will continue to display the contents of the committed data source if there is one.
82*/
83- (void)webView:(WebView *)sender didFailProvisionalLoadWithError:(NSError *)error forFrame:(WebFrame *)frame;
84
85/*!
86 @method webView:didCommitLoadForFrame:
87 @abstract Notifies the delegate that the load has changed from provisional to committed
88 @param sender The WebView sending the message
89 @param frame The frame for which the load has committed
90 @discussion This method is called after the provisional data source has become the
91 committed data source.
92
93 In some cases, a single load may be committed more than once. This happens
94 in the case of multipart/x-mixed-replace, also known as "server push". In this case,
95 a single location change leads to multiple documents that are loaded in sequence. When
96 this happens, a new commit will be sent for each document.
97*/
98- (void)webView:(WebView *)sender didCommitLoadForFrame:(WebFrame *)frame;
99
100/*!
101 @method webView:didReceiveTitle:forFrame:
102 @abstract Notifies the delegate that the page title for a frame has been received
103 @param sender The WebView sending the message
104 @param title The new page title
105 @param frame The frame for which the title has been received
106 @discussion The title may update during loading; clients should be prepared for this.
107*/
108- (void)webView:(WebView *)sender didReceiveTitle:(NSString *)title forFrame:(WebFrame *)frame;
109
110/*!
111 @method webView:didReceiveIcon:forFrame:
112 @abstract Notifies the delegate that a page icon image for a frame has been received
113 @param webView The WebView sending the message
114 @param image The icon image. Also known as a "favicon".
115 @param frame The frame for which a page icon has been received
116*/
117- (void)webView:(WebView *)sender didReceiveIcon:(NSImage *)image forFrame:(WebFrame *)frame;
118
119/*!
120 @method webView:didFinishLoadForFrame:
121 @abstract Notifies the delegate that the committed load of a frame has completed
122 @param sender The WebView sending the message
123 @param frame The frame that finished loading
124 @discussion This method is called after the committed data source of a frame has successfully loaded
125 and will only be called when all subresources such as images and stylesheets are done loading.
126 Plug-In content and JavaScript-requested loads may occur after this method is called.
127*/
128- (void)webView:(WebView *)sender didFinishLoadForFrame:(WebFrame *)frame;
129
130/*!
131 @method webView:didFailLoadWithError:forFrame:
132 @abstract Notifies the delegate that the committed load of a frame has failed
133 @param sender The WebView sending the message
134 @param error The error that occurred
135 @param frame The frame that failed to load
136 @discussion This method is called after a data source has committed but failed to completely load.
137*/
138- (void)webView:(WebView *)sender didFailLoadWithError:(NSError *)error forFrame:(WebFrame *)frame;
139
140/*!
141 @method webView:didChangeLocationWithinPageForFrame:
142 @abstract Notifies the delegate that the scroll position in a frame has changed
143 @param sender The WebView sending the message
144 @param frame The frame that scrolled
145 @discussion This method is called when anchors within a page have been clicked.
146*/
147- (void)webView:(WebView *)sender didChangeLocationWithinPageForFrame:(WebFrame *)frame;
148
149/*!
150 @method webView:willPerformClientRedirectToURL:delay:fireDate:forFrame:
151 @abstract Notifies the delegate that a frame will perform a client-side redirect
152 @param sender The WebView sending the message
153 @param URL The URL to be redirected to
154 @param seconds Seconds in which the redirect will happen
155 @param date The fire date
156 @param frame The frame on which the redirect will occur
157 @discussion This method can be used to continue progress feedback while a client-side
158 redirect is pending.
159*/
160- (void)webView:(WebView *)sender willPerformClientRedirectToURL:(NSURL *)URL delay:(NSTimeInterval)seconds fireDate:(NSDate *)date forFrame:(WebFrame *)frame;
161
162/*!
163 @method webView:didCancelClientRedirectForFrame:
164 @abstract Notifies the delegate that a pending client-side redirect has been cancelled
165 @param sender The WebView sending the message
166 @param frame The frame for which the pending redirect was cancelled
167 @discussion A client-side redirect can be cancelled if a frame changes location before the timeout.
168*/
169- (void)webView:(WebView *)sender didCancelClientRedirectForFrame:(WebFrame *)frame;
170
171/*!
172 @method webView:willCloseFrame:
173 @abstract Notifies the delegate that a frame will be closed
174 @param sender The WebView sending the message
175 @param frame The frame that will be closed
176 @discussion This method is called right before WebKit is done with the frame
177 and the objects that it contains.
178*/
179- (void)webView:(WebView *)sender willCloseFrame:(WebFrame *)frame;
180
181/*!
182 @method webView:didClearWindowObject:forFrame:
183 @abstract Notifies the delegate that the JavaScript window object in a frame has
184 been cleared in preparation for a new load. This is the preferred place to set custom
185 properties on the window object using the WebScriptObject and JavaScriptCore APIs.
186 @param webView The webView sending the message.
187 @param windowObject The WebScriptObject representing the frame's JavaScript window object.
188 @param frame The WebFrame to which windowObject belongs.
189 @discussion If a delegate implements both webView:didClearWindowObject:forFrame:
190 and webView:windowScriptObjectAvailable:, only webView:didClearWindowObject:forFrame:
191 will be invoked. This enables a delegate to implement both methods for backwards
192 compatibility with older versions of WebKit.
193*/
194- (void)webView:(WebView *)webView didClearWindowObject:(WebScriptObject *)windowObject forFrame:(WebFrame *)frame;
195
196/*!
197 @method webView:windowScriptObjectAvailable:
198 @abstract Notifies the delegate that the scripting object for a page is available. This is called
199 before the page is loaded. It may be useful to allow delegates to bind native objects to the window.
200 @param webView The webView sending the message.
201 @param windowScriptObject The WebScriptObject for the window in the scripting environment.
202 @discussion This method is deprecated. Consider using webView:didClearWindowObject:forFrame:
203 instead.
204*/
205- (void)webView:(WebView *)webView windowScriptObjectAvailable:(WebScriptObject *)windowScriptObject WEBKIT_DEPRECATED_MAC(10_4, 10_5);
206
207#if JSC_OBJC_API_ENABLED
208/*!
209 @method webView:didCreateJavaScriptContext:contextForFrame:
210 @abstract Notifies the delegate that a new JavaScript context has been created created.
211 @param webView The WebView sending the message.
212 @param context The JSContext representing the frame's JavaScript window object.
213 @param frame The WebFrame to which the context belongs.
214 @discussion If a delegate implements webView:didCreateJavaScriptContext:forFrame: along with either
215 webView:didClearWindowObject:forFrame: or webView:windowScriptObjectAvailable:, only
216 webView:didCreateJavaScriptContext:forFrame will be invoked. This enables a delegate to implement
217 multiple versions to maintain backwards compatibility with older versions of WebKit.
218*/
219- (void)webView:(WebView *)webView didCreateJavaScriptContext:(JSContext *)context forFrame:(WebFrame *)frame;
220#endif
