29#import <Foundation/Foundation.h>
30#import <WebKit/WebKitAvailability.h>
31
32#import <AppKit/AppKit.h>
33
34@class WebDataSource;
35@class WebFrame;
36@class WebFrameViewPrivate;
37
38@protocol WebDocumentView;
39
40/*!
41 @class WebFrameView
42*/
43WEBKIT_CLASS_DEPRECATED_MAC(10_3, 10_14)
44@interface WebFrameView : NSView
45{
46@package
47 WebFrameViewPrivate *_private;
48}
49
50/*!
51 @property webFrame
52 @abstract The WebFrame associated with this WebFrameView
53*/
54@property (nonatomic, readonly, strong) WebFrame *webFrame;
55
56/*!
57 @property documentView
58 @abstract The WebFrameView's document subview
59 @discussion The subview that renders the WebFrameView's contents
60*/
61@property (nonatomic, readonly, strong) NSView<WebDocumentView> *documentView;
62
63/*!
64 @property allowsScrolling
65 @abstract Whether the WebFrameView allows its document to be scrolled
66*/
67@property (nonatomic) BOOL allowsScrolling;
68
69/*!
70 @property canPrintHeadersAndFooters
71 @abstract Whether this frame can print headers and footers
72*/
73@property (nonatomic, readonly) BOOL canPrintHeadersAndFooters;
74
75/*!
76 @method printOperationWithPrintInfo
77 @abstract Creates a print operation set up to print this frame
78 @result A newly created print operation object
79*/
80- (NSPrintOperation *)printOperationWithPrintInfo:(NSPrintInfo *)printInfo;
81
82/*!
83 @property documentViewShouldHandlePrint
84 @abstract Called by the host application before it initializes and runs a print operation.
85 @discussion If NO is returned, the host application will abort its print operation and call -printDocumentView on the
86 WebFrameView. The document view is then expected to run its own print operation. If YES is returned, the host
87 application's print operation will continue as normal.
88*/
89@property (nonatomic, readonly) BOOL documentViewShouldHandlePrint;
90
91/*!
92 @method printDocumentView
93 @abstract Called by the host application when the WebFrameView returns YES from -documentViewShouldHandlePrint.
94*/
95- (void)printDocumentView;
96
97@end
