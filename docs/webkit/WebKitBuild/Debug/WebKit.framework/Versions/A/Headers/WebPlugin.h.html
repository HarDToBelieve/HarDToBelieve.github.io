29#import <Foundation/Foundation.h>
30#import <WebKit/WebKitAvailability.h>
31
32#import <AppKit/AppKit.h>
33
34/*!
35 WebPlugIn is an informal protocol that enables interaction between an application
36 and web related plug-ins it may contain.
37*/
38
39@interface NSObject (WebPlugIn)
40
41/*!
42 @method webPlugInInitialize
43 @abstract Tell the plug-in to perform one-time initialization.
44 @discussion This method must be only called once per instance of the plug-in
45 object and must be called before any other methods in this protocol.
46*/
47- (void)webPlugInInitialize;
48
49/*!
50 @method webPlugInStart
51 @abstract Tell the plug-in to start normal operation.
52 @discussion The plug-in usually begins drawing, playing sounds and/or
53 animation in this method. This method must be called before calling webPlugInStop.
54 This method may called more than once, provided that the application has
55 already called webPlugInInitialize and that each call to webPlugInStart is followed
56 by a call to webPlugInStop.
57*/
58- (void)webPlugInStart;
59
60/*!
61 @method webPlugInStop
62 @abstract Tell the plug-in to stop normal operation.
63 @discussion webPlugInStop must be called before this method. This method may be
64 called more than once, provided that the application has already called
65 webPlugInInitialize and that each call to webPlugInStop is preceded by a call to
66 webPlugInStart.
67*/
68- (void)webPlugInStop;
69
70/*!
71 @method webPlugInDestroy
72 @abstract Tell the plug-in perform cleanup and prepare to be deallocated.
73 @discussion The plug-in typically releases memory and other resources in this
74 method. If the plug-in has retained the WebPlugInContainer, it must release
75 it in this mehthod. This method must be only called once per instance of the
76 plug-in object. No other methods in this interface may be called after the
77 application has called webPlugInDestroy.
78*/
79- (void)webPlugInDestroy;
80
81/*!
82 @method webPlugInSetIsSelected:
83 @discusssion Informs the plug-in whether or not it is selected. This is typically
84 used to allow the plug-in to alter it's appearance when selected.
85*/
86- (void)webPlugInSetIsSelected:(BOOL)isSelected;
87
88/*!
89 @property objectForWebScript
90 @discussion objectForWebScript is used to expose a plug-in's scripting interface. The
91 methods of the object are exposed to the script environment. See the WebScripting
92 informal protocol for more details.
93 @result Returns the object that exposes the plug-in's interface. The class of this
94 object can implement methods from the WebScripting informal protocol.
95*/
96@property (nonatomic, readonly, strong) id objectForWebScript;
97
98/*!
99 @method webPlugInMainResourceDidReceiveResponse:
100 @abstract Called on the plug-in when WebKit receives -connection:didReceiveResponse:
101 for the plug-in's main resource.
102 @discussion This method is only sent to the plug-in if the
103 WebPlugInShouldLoadMainResourceKey argument passed to the plug-in was NO.
104*/
105- (void)webPlugInMainResourceDidReceiveResponse:(NSURLResponse *)response WEBKIT_AVAILABLE_MAC(10_6);
106
107/*!
108 @method webPlugInMainResourceDidReceiveData:
109 @abstract Called on the plug-in when WebKit recieves -connection:didReceiveData:
110 for the plug-in's main resource.
111 @discussion This method is only sent to the plug-in if the
112 WebPlugInShouldLoadMainResourceKey argument passed to the plug-in was NO.
113*/
114- (void)webPlugInMainResourceDidReceiveData:(NSData *)data WEBKIT_AVAILABLE_MAC(10_6);
115
116/*!
117 @method webPlugInMainResourceDidFailWithError:
118 @abstract Called on the plug-in when WebKit receives -connection:didFailWithError:
119 for the plug-in's main resource.
120 @discussion This method is only sent to the plug-in if the
121 WebPlugInShouldLoadMainResourceKey argument passed to the plug-in was NO.
122*/
123- (void)webPlugInMainResourceDidFailWithError:(NSError *)error WEBKIT_AVAILABLE_MAC(10_6);
124
125/*!
126 @method webPlugInMainResourceDidFinishLoading
127 @abstract Called on the plug-in when WebKit receives -connectionDidFinishLoading:
128 for the plug-in's main resource.
129 @discussion This method is only sent to the plug-in if the
130 WebPlugInShouldLoadMainResourceKey argument passed to the plug-in was NO.
131*/
132- (void)webPlugInMainResourceDidFinishLoading WEBKIT_AVAILABLE_MAC(10_6);
133
134@end
