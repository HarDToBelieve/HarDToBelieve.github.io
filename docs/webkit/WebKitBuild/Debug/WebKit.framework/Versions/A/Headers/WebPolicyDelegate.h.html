41/*!
42 @enum WebNavigationType
43 @abstract The type of action that triggered a possible navigation.
44 @constant WebNavigationTypeLinkClicked A link with an href was clicked.
45 @constant WebNavigationTypeFormSubmitted A form was submitted.
46 @constant WebNavigationTypeBackForward The user chose back or forward.
47 @constant WebNavigationTypeReload The User hit the reload button.
48 @constant WebNavigationTypeFormResubmitted A form was resubmitted (by virtue of doing back, forward or reload).
49 @constant WebNavigationTypeOther Navigation is taking place for some other reason.

67/*!
68 @protocol WebPolicyDecisionListener
69 @discussion This protocol is used to call back with the results of a
70 policy decision. This provides the ability to make these decisions
71 asyncrhonously, which means the decision can be made by prompting
72 with a sheet, for example.
73*/
74
75WEBKIT_DEPRECATED_MAC(10_3, 10_14)
76@protocol WebPolicyDecisionListener <NSObject>
77
78/*!
79 @method use
80 @abstract Use the resource
81 @discussion If there remain more policy decisions to be made, then
82 the next policy delegate method gets to decide. This will be
83 either the next navigation policy delegate if there is a redirect,
84 or the content policy delegate. If there are no more policy
85 decisions to be made, the resource will be displayed inline if
86 possible. If there is no view available to display the resource
87 inline, then unableToImplementPolicyWithError:frame: will be
88 called with an appropriate error.
89
90 <p>If a new window is going to be created for this navigation as a
91 result of frame targeting, then it will be created once you call
92 this method.
93*/
94- (void)use;
95/*!
96 @method download
97 @abstract Download the resource instead of displaying it.
98 @discussion This method is more than just a convenience because it
99 allows an in-progress navigation to be converted to a download
100 based on content type, without having to stop and restart the
101 load.
102*/
103- (void)download;
104
105/*!
106 @method ignore
107 @abstract Do nothing (but the client may choose to handle the request itself)
108 @discussion A policy of ignore prevents WebKit from doing anything
109 further with the load, however, the client is still free to handle
110 the request in some other way, such as opening a new window,
111 opening a new window behind the current one, opening the URL in an
112 external app, revealing the location in Finder if a file URL, etc.
113*/
114- (void)ignore;
115
116@end

119/*!
120 @category WebPolicyDelegate
121 @discussion While loading a URL, WebKit asks the WebPolicyDelegate for
122 policies that determine the action of what to do with the URL or the data that
123 the URL represents. Typically, the policy handler methods are called in this order:
124
125 decidePolicyForNewWindowAction:request:newFrameName:decisionListener: (at most once)<BR>
126 decidePolicyForNavigationAction:request:frame:decisionListener: (zero or more times)<BR>
127 decidePolicyForMIMEType:request:frame: (zero or more times)<BR>
128
129 New window policy is always checked. Navigation policy is checked
130 for the initial load and every redirect unless blocked by an
131 earlier policy. Content policy is checked once the content type is
132 known, unless an earlier policy prevented it.
133
134 In rare cases, content policy might be checked more than
135 once. This occurs when loading a "multipart/x-mixed-replace"
136 document, also known as "server push". In this case, multiple
137 documents come in one navigation, with each replacing the last. In
138 this case, conent policy will be checked for each one.
139*/
140WEBKIT_DEPRECATED_MAC(10_3, 10_14)
141@protocol WebPolicyDelegate <NSObject>
142
143@optional
144
145/*!
146 @method webView:decidePolicyForNavigationAction:request:frame:decisionListener:
147 @abstract This method is called to decide what to do with a proposed navigation.
148 @param actionInformation Dictionary that describes the action that triggered this navigation.
149 @param request The request for the proposed navigation
150 @param frame The WebFrame in which the navigation is happening
151 @param listener The object to call when the decision is made
152 @discussion This method will be called before loading starts, and
153 on every redirect.
154*/
155- (void)webView:(WebView *)webView decidePolicyForNavigationAction:(NSDictionary *)actionInformation
156 request:(NSURLRequest *)request
157 frame:(WebFrame *)frame
158 decisionListener:(id<WebPolicyDecisionListener>)listener;
159
160/*!
161 @method webView:decidePolicyForNewWindowAction:request:newFrameName:decisionListener:
162 @discussion This method is called to decide what to do with an targetted nagivation that would open a new window.
163 @param actionInformation Dictionary that describes the action that triggered this navigation.
164 @param request The request for the proposed navigation
165 @param frameName The frame in which the navigation is taking place
166 @param listener The object to call when the decision is made
167 @discussion This method is provided so that modified clicks on a targetted link which
168 opens a new frame can prevent the new window from being opened if they decide to
169 do something else, like download or present the new frame in a specialized way.
170
171 <p>If this method picks a policy of Use, the new window will be
172 opened, and decidePolicyForNavigationAction:request:frame:decisionListner:
173 will be called with a WebNavigationType of WebNavigationTypeOther
174 in its action. This is to avoid possible confusion about the modifiers.
175*/
176- (void)webView:(WebView *)webView decidePolicyForNewWindowAction:(NSDictionary *)actionInformation
177 request:(NSURLRequest *)request
178 newFrameName:(NSString *)frameName
179 decisionListener:(id<WebPolicyDecisionListener>)listener;
180
181/*!
182 @method webView:decidePolicyForMIMEType:request:frame:
183 @discussion Returns the policy for content which has been partially loaded.
184 Sent after webView:didStartProvisionalLoadForFrame: is sent on the WebFrameLoadDelegate.
185 @param type MIME type for the resource.
186 @param request A NSURLRequest for the partially loaded content.
187 @param frame The frame which is loading the URL.
188 @param listener The object to call when the decision is made
189*/
190- (void)webView:(WebView *)webView decidePolicyForMIMEType:(NSString *)type
191 request:(NSURLRequest *)request
192 frame:(WebFrame *)frame
193 decisionListener:(id<WebPolicyDecisionListener>)listener;
194
195/*!
196 @method webView:unableToImplementPolicy:error:forURL:inFrame:
197 @discussion Called when a WebPolicy could not be implemented. It is up to the client to display appropriate feedback.
198 @param error The error that caused the policy to not be implemented.
199 @param frame The frame in which the policy could not be implemented.
200*/
201- (void)webView:(WebView *)webView unableToImplementPolicyWithError:(NSError *)error frame:(WebFrame *)frame;
