84/*!
85 @class WebView
86 WebView manages the interaction between WebFrameViews and WebDataSources. Modification
87 of the policies and behavior of the WebKit is largely managed by WebViews and their
88 delegates.
89
90 <p>
91 Typical usage:
92 </p>
93 <pre>
94 WebView *webView;
95 WebFrame *mainFrame;
96
97 webView = [[WebView alloc] initWithFrame: NSMakeRect (0,0,640,480)];
98 mainFrame = [webView mainFrame];
99 [mainFrame loadRequest:request];
100 </pre>
101
102 WebViews have the following delegates: WebUIDelegate, WebResourceLoadDelegate,
103 WebFrameLoadDelegate, and WebPolicyDelegate.
104
105 WebKit depends on the WebView's WebUIDelegate for all window
106 related management, including opening new windows and controlling the user interface
107 elements in those windows.
108
109 WebResourceLoadDelegate is used to monitor the progress of resources as they are
110 loaded. This delegate may be used to present users with a progress monitor.
111
112 The WebFrameLoadDelegate receives messages when the URL in a WebFrame is
113 changed.
114
115 WebView's WebPolicyDelegate can make determinations about how
116 content should be handled, based on the resource's URL and MIME type.
117*/
118WEBKIT_CLASS_DEPRECATED_MAC(10_3, 10_14, "No longer supported; please adopt WKWebView.")
119@interface WebView : NSView
120{
121@private
122 WebViewPrivate *_private;
123}
