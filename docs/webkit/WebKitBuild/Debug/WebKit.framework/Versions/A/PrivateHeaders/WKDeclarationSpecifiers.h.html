27#ifndef WKDeclarationSpecifiers_h
28#define WKDeclarationSpecifiers_h
29
30#ifndef __has_declspec_attribute
31#define __has_declspec_attribute(x) 0
32#endif
33
34#undef WK_EXPORT
35#if defined(WK_NO_EXPORT)
36#define WK_EXPORT
37#elif defined(WIN32) || defined(_WIN32) || defined(__CC_ARM) || defined(__ARMCC__) || (__has_declspec_attribute(dllimport) && __has_declspec_attribute(dllexport))
38#if defined(BUILDING_WebKit)
39#define WK_EXPORT __declspec(dllexport)
40#else
41#define WK_EXPORT __declspec(dllimport)
42#endif /* defined(BUILDING_WebKit) */
43#elif defined(__GNUC__)
44#define WK_EXPORT __attribute__((visibility("default")))
45#else /* !defined(WK_NO_EXPORT) */
46#define WK_EXPORT
47#endif /* defined(WK_NO_EXPORT) */
48
49#if !defined(WK_INLINE)
50#if defined(__cplusplus)
51#define WK_INLINE static inline
52#elif defined(__GNUC__)
53#define WK_INLINE static __inline__
54#else
55#define WK_INLINE static
56#endif
57#endif /* !defined(WK_INLINE) */
58
59#ifndef __has_extension
60#define __has_extension(x) 0
61#endif
62
63#if __has_extension(enumerator_attributes) && __has_extension(attribute_unavailable_with_message)
64#define WK_C_DEPRECATED(message) __attribute__((deprecated(message)))
65#else
66#define WK_C_DEPRECATED(message)
67#endif
68
69#ifndef __has_attribute
70#define __has_attribute(x) 0
71#endif
72
73#if __has_attribute(unavailable)
74#define WK_UNAVAILABLE __attribute__((unavailable))
75#else
76#define WK_UNAVAILABLE
77#endif
78
79#endif /* WKDeclarationSpecifiers_h */
