29#import <Foundation/Foundation.h>
30#import <WebKitLegacy/WebKitAvailability.h>
31
32@class WebArchivePrivate;
33@class WebResource;
34
35/*!
36 @const WebArchivePboardType
37 @abstract The pasteboard type constant used when adding or accessing a WebArchive on the pasteboard.
38*/
39extern NSString *WebArchivePboardType WEBKIT_DEPRECATED_MAC(10_3, 10_14);
40
41/*!
42 @class WebArchive
43 @discussion WebArchive represents a main resource as well as all the subresources and subframes associated with the main resource.
44 The main resource can be an entire web page, a portion of a web page, or some other kind of data such as an image.
45 This class can be used for saving standalone web pages, representing portions of a web page on the pasteboard, or any other
46 application where one class is needed to represent rich web content.
47*/
48WEBKIT_CLASS_DEPRECATED_MAC(10_3, 10_14)
49@interface WebArchive : NSObject <NSCoding, NSCopying>
50{
51@package
52 WebArchivePrivate *_private;
53}
54
55/*!
56 @method initWithMainResource:subresources:subframeArchives:
57 @abstract The initializer for WebArchive.
58 @param mainResource The main resource of the archive.
59 @param subresources The subresources of the archive (can be nil).
60 @param subframeArchives The archives representing the subframes of the archive (can be nil).
61 @result An initialized WebArchive.
62*/
63- (instancetype)initWithMainResource:(WebResource *)mainResource subresources:(NSArray *)subresources subframeArchives:(NSArray *)subframeArchives;
64
65/*!
66 @method initWithData:
67 @abstract The initializer for creating a WebArchive from data.
68 @param data The data representing the archive. This can be obtained using WebArchive's data method.
69 @result An initialized WebArchive.
70*/
71- (instancetype)initWithData:(NSData *)data;
72
73/*!
74 @property mainResource
75 @abstract The main resource of the archive.
76*/
77@property (nonatomic, readonly, strong) WebResource *mainResource;
78
79/*!
80 @property subresources
81 @abstract The subresource of the archive (can be nil).
82*/
83@property (nonatomic, readonly, copy) NSArray *subresources;
84
85/*!
86 @property subframeArchives
87 @abstract The archives representing the subframes of the archive (can be nil).
88*/
89@property (nonatomic, readonly, copy) NSArray *subframeArchives;
90
91/*!
92 @property data
93 @abstract The data representation of the archive.
94 @discussion The data returned by this method can be used to save a web archive to a file or to place a web archive on the pasteboard
95 using WebArchivePboardType. To create a WebArchive using the returned data, call initWithData:.
96*/
97@property (nonatomic, readonly, copy) NSData *data;
98
99@end
