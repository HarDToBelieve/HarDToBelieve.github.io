29#import <Foundation/Foundation.h>
30
31@class WebResourcePrivate;
32
33
34/*!
35 @class WebResource
36 @discussion A WebResource represents a fully downloaded URL.
37 It includes the data of the resource as well as the metadata associated with the resource.
38*/
39@interface WebResource : NSObject <NSCoding, NSCopying>
40{
41@package
42 WebResourcePrivate *_private;
43}
44
45/*!
46 @method initWithData:URL:MIMEType:textEncodingName:frameName
47 @abstract The initializer for WebResource.
48 @param data The data of the resource.
49 @param URL The URL of the resource.
50 @param MIMEType The MIME type of the resource.
51 @param textEncodingName The text encoding name of the resource (can be nil).
52 @param frameName The frame name of the resource if the resource represents the contents of an entire HTML frame (can be nil).
53 @result An initialized WebResource.
54*/
55- (instancetype)initWithData:(NSData *)data URL:(NSURL *)URL MIMEType:(NSString *)MIMEType textEncodingName:(NSString *)textEncodingName frameName:(NSString *)frameName;
56
57/*!
58 @property data
59 @abstract The data of the resource.
60*/
61@property (nonatomic, readonly, copy) NSData *data;
62
63/*!
64 @property URL
65 @abstract The URL of the resource.
66*/
67@property (nonatomic, readonly, strong) NSURL *URL;
68
69/*!
70 @property MIMEType
71 @abstract The MIME type of the resource.
72*/
73@property (nonatomic, readonly, copy) NSString *MIMEType;
74
75/*!
76 @property textEncodingName
77 @abstract The text encoding name of the resource (can be nil).
78*/
79@property (nonatomic, readonly, copy) NSString *textEncodingName;
80
81/*!
82 @property frameName
83 @abstract The frame name of the resource if the resource represents the contents of an entire HTML frame (can be nil).
84*/
85@property (nonatomic, readonly, copy) NSString *frameName;
86
87@end
