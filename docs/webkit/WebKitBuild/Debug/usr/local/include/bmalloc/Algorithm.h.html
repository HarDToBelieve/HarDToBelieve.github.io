41// Versions of min and max that are compatible with compile-time constants.
42template<typename T> constexpr T max(T a, T b)
43{
44 return a > b ? a : b;

52template<typename T> constexpr T mask(T value, uintptr_t mask)
53{
54 static_assert(sizeof(T) == sizeof(uintptr_t), "sizeof(T) must be equal to sizeof(uintptr_t).");
55 return static_cast<T>(static_cast<uintptr_t>(value) & mask);
56}
57
58template<typename T> inline T* mask(T* value, uintptr_t mask)
59{
60 return reinterpret_cast<T*>(reinterpret_cast<uintptr_t>(value) & mask);
61}
62
63template<typename T> constexpr bool test(T value, uintptr_t mask)
64{
65 return !!(reinterpret_cast<uintptr_t>(value) & mask);
66}

68template <typename T>
69constexpr bool isPowerOfTwo(T size)
70{
71 static_assert(std::is_integral<T>::value, "");
72 return size && !(size & (size - 1));
73}

75template<typename T> constexpr T roundUpToMultipleOfImpl(size_t divisor, T x)
76{
77 static_assert(sizeof(T) == sizeof(uintptr_t), "sizeof(T) must be equal to sizeof(uintptr_t).");
78 return static_cast<T>((static_cast<uintptr_t>(x) + (divisor - 1)) & ~(divisor - 1));
79}
80
81template<typename T> inline T roundUpToMultipleOf(size_t divisor, T x)
82{
83 BASSERT(isPowerOfTwo(divisor));
84 return roundUpToMultipleOfImpl(divisor, x);
85}
86
87template<size_t divisor, typename T> constexpr T roundUpToMultipleOf(T x)
88{
89 static_assert(isPowerOfTwo(divisor), "'divisor' must be a power of two.");
90 return roundUpToMultipleOfImpl(divisor, x);
91}

93template<typename T> inline T* roundUpToMultipleOf(size_t divisor, T* x)
94{
95 BASSERT(isPowerOfTwo(divisor));
96 return reinterpret_cast<T*>((reinterpret_cast<uintptr_t>(x) + (divisor - 1)) & ~(divisor - 1));
97}
98
99template<size_t divisor, typename T> inline T* roundUpToMultipleOf(T* x)
100{
101 static_assert(isPowerOfTwo(divisor), "'divisor' must be a power of two.");
102 return roundUpToMultipleOf(divisor, x);
103}

105template<typename T> inline T roundDownToMultipleOf(size_t divisor, T x)
106{
107 BASSERT(isPowerOfTwo(divisor));
108 static_assert(sizeof(T) == sizeof(uintptr_t), "sizeof(T) must be equal to sizeof(uintptr_t).");
109 return static_cast<T>(mask(static_cast<uintptr_t>(x), ~(divisor - 1ul)));
110}
111
112template<typename T> inline T* roundDownToMultipleOf(size_t divisor, T* x)
113{
114 BASSERT(isPowerOfTwo(divisor));
115 return reinterpret_cast<T*>(mask(reinterpret_cast<uintptr_t>(x), ~(divisor - 1ul)));
116}
117
118template<size_t divisor, typename T> constexpr T roundDownToMultipleOf(T x)
119{
120 static_assert(isPowerOfTwo(divisor), "'divisor' must be a power of two.");
121 return roundDownToMultipleOf(divisor, x);
122}

124template<typename T> inline void divideRoundingUp(T numerator, T denominator, T& quotient, T& remainder)
125{
126 // We expect the compiler to emit a single divide instruction to extract both the quotient and the remainder.
127 quotient = numerator / denominator;
128 remainder = numerator % denominator;
129 if (remainder)
130 quotient += 1;
131}
132
133template<typename T> constexpr T divideRoundingUp(T numerator, T denominator)
134{
135 return (numerator + denominator - 1) / denominator;
136}
137
138template<typename T> inline T roundUpToMultipleOfNonPowerOfTwo(size_t divisor, T x)
139{
140 return divideRoundingUp(x, divisor) * divisor;
141}

155#if BOS(WINDOWS)
156template<int depth> __forceinline constexpr unsigned long clzl(unsigned long value)
157{
158 return value & (1UL << (bitCount<unsigned long>() - 1)) ? 0 : 1 + clzl<depth - 1>(value << 1);
159}
160
161template<> __forceinline constexpr unsigned long clzl<1>(unsigned long value)
162{
163 return 0;
164}
165
166__forceinline constexpr unsigned long __builtin_clzl(unsigned long value)
167{
168 return value == 0 ? 32 : clzl<bitCount<unsigned long>()>(value);
169}
170#endif

172template <typename T>
173constexpr unsigned clzConstexpr(T value)
174{
175 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
176
177 using UT = typename std::make_unsigned<T>::type;
178 UT uValue = value;
179
180 unsigned zeroCount = 0;
181 for (int i = bitSize - 1; i >= 0; i--) {
182 if (uValue >> i)
183 break;
184 zeroCount++;
185 }
186 return zeroCount;
187}
188
189constexpr unsigned long log2(unsigned long value)
190{
191 return bitCount<unsigned long>() - 1 - __builtin_clzl(value);
192}

194#define BOFFSETOF(class, field) (reinterpret_cast<ptrdiff_t>(&(reinterpret_cast<class*>(0x4000)->field)) - 0x4000)

196template <typename T>
197constexpr unsigned ctzConstexpr(T value)
198{
199 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
200
201 using UT = typename std::make_unsigned<T>::type;
202 UT uValue = value;
203
204 unsigned zeroCount = 0;
205 for (unsigned i = 0; i < bitSize; i++) {
206 if (uValue & 1)
207 break;
208
209 zeroCount++;
210 uValue >>= 1;
211 }
212 return zeroCount;
213}
214
215template<typename T>
216inline unsigned ctz(T value)
217{
218 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
219
220 using UT = typename std::make_unsigned<T>::type;
221 UT uValue = value;
222
223#if BCOMPILER(GCC_COMPATIBLE)
224 if (uValue)
225 return __builtin_ctzll(uValue);
226 return bitSize;
227#elif BCOMPILER(MSVC) && !BCPU(X86)
228 unsigned long ret = 0;
229 if (_BitScanForward64(&ret, uValue))
230 return ret;
231 return bitSize;
232#else
233 UNUSED_PARAM(bitSize);
234 UNUSED_PARAM(uValue);
235 return ctzConstexpr(value);
236#endif
237}

239template<typename T>
240bool findBitInWord(T word, size_t& startOrResultIndex, size_t endIndex, bool value)
241{
242 static_assert(std::is_unsigned<T>::value, "Type used in findBitInWord must be unsigned");
243 constexpr size_t bitsInWord = sizeof(word) * 8;
244 BASSERT(startOrResultIndex <= bitsInWord && endIndex <= bitsInWord);
245 BUNUSED(bitsInWord);
246
247 size_t index = startOrResultIndex;
248 word >>= index;
249
250#if BCOMPILER(GCC_COMPATIBLE) && (BCPU(X86_64) || BCPU(ARM64))
251 // We should only use ctz() when we know that ctz() is implementated using
252 // a fast hardware instruction. Otherwise, this will actually result in
253 // worse performance.
254
255 word ^= (static_cast<T>(value) - 1);
256 index += ctz(word);
257 if (index < endIndex) {
258 startOrResultIndex = index;
259 return true;
260 }
261#else
262 while (index < endIndex) {
263 if ((word & 1) == static_cast<T>(value)) {
264 startOrResultIndex = index;
265 return true;
266 }
267 index++;
268 word >>= 1;
269 }
270#endif
271
272 startOrResultIndex = endIndex;
273 return false;
274}

282template<typename T>
283constexpr unsigned getMSBSetConstexpr(T t)
284{
285 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
286 return bitSize - 1 - clzConstexpr(t);

289// From http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
290constexpr uint32_t roundUpToPowerOfTwo(uint32_t v)
291{
292 v--;
293 v |= v >> 1;
294 v |= v >> 2;
295 v |= v >> 4;
296 v |= v >> 8;
297 v |= v >> 16;
298 v++;
299 return v;
300}
