67constexpr bool hasCapacityToUseLargeGigacage = BOS_EFFECTIVE_ADDRESS_WIDTH > 36;
68
69#if GIGACAGE_ENABLED
70
71constexpr size_t primitiveGigacageSize = (hasCapacityToUseLargeGigacage ? 32 : 2) * bmalloc::Sizes::GB;
72constexpr size_t jsValueGigacageSize = (hasCapacityToUseLargeGigacage ? 16 : 2) * bmalloc::Sizes::GB;
73constexpr size_t maximumCageSizeReductionForSlide = hasCapacityToUseLargeGigacage ? 4 * bmalloc::Sizes::GB : bmalloc::Sizes::GB / 4;
74
75
76// In Linux, if `vm.overcommit_memory = 2` is specified, mmap with large size can fail if it exceeds the size of RAM.
77// So we specify GIGACAGE_ALLOCATION_CAN_FAIL = 1.
78#if BOS(LINUX)
79#define GIGACAGE_ALLOCATION_CAN_FAIL 1
80#else
81#define GIGACAGE_ALLOCATION_CAN_FAIL 0
82#endif
83
84
85static_assert(bmalloc::isPowerOfTwo(primitiveGigacageSize), "");
86static_assert(bmalloc::isPowerOfTwo(jsValueGigacageSize), "");
87static_assert(primitiveGigacageSize > maximumCageSizeReductionForSlide, "");
88static_assert(jsValueGigacageSize > maximumCageSizeReductionForSlide, "");
89
90constexpr size_t gigacageSizeToMask(size_t size) { return size - 1; }
91
92constexpr size_t primitiveGigacageMask = gigacageSizeToMask(primitiveGigacageSize);
93constexpr size_t jsValueGigacageMask = gigacageSizeToMask(jsValueGigacageSize);

123BINLINE bool isEnabled(Kind kind)
124{
125 if (kind == Primitive)
126 return g_gigacageConfig.basePtr(Primitive) && (disablingPrimitiveGigacageIsForbidden() || disableNotRequestedForPrimitiveGigacage());
127 return g_gigacageConfig.basePtr(kind);
128}
129
130BINLINE void* basePtr(Kind kind)
131{
132 BASSERT(isEnabled(kind));
133 return g_gigacageConfig.basePtr(kind);
134}

136BINLINE void* addressOfBasePtr(Kind kind)

142BINLINE size_t maxSize(Kind kind)
143{
144 switch (kind) {
145 case Primitive:
146 return static_cast<size_t>(primitiveGigacageSize);
147 case JSValue:
148 return static_cast<size_t>(jsValueGigacageSize);
149 case NumberOfKinds:
150 break;
151 }
152 BCRASH();
153 return 0;
154}
155
156BINLINE size_t alignment(Kind kind)
157{
158 return maxSize(kind);
159}
160
161BINLINE size_t mask(Kind kind)
162{
163 return gigacageSizeToMask(maxSize(kind));
164}

176template<typename T>
177BINLINE T* caged(Kind kind, T* ptr)
178{
179 BASSERT(ptr);
180 if (!isEnabled(kind))
181 return ptr;
182 void* gigacageBasePtr = basePtr(kind);
183 return reinterpret_cast<T*>(
184 reinterpret_cast<uintptr_t>(gigacageBasePtr) + (
185 reinterpret_cast<uintptr_t>(ptr) & mask(kind)));
186}
187
188template<typename T>
189BINLINE T* cagedMayBeNull(Kind kind, T* ptr)
190{
191 if (!ptr)
192 return ptr;
193 return caged(kind, ptr);
194}
195
196BINLINE bool isCaged(Kind kind, const void* ptr)
197{
198 return caged(kind, ptr) == ptr;
199}
200
201BINLINE bool contains(const void* ptr)
202{
203 auto* start = reinterpret_cast<const uint8_t*>(g_gigacageConfig.start);
204 auto* p = reinterpret_cast<const uint8_t*>(ptr);
205 return static_cast<size_t>(p - start) < g_gigacageConfig.totalSize;
206}
