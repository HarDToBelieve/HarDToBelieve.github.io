26#pragma once
27
28#include "BMalloced.h"
29#include "IsoDirectory.h"
30
31namespace bmalloc {
32
33template<typename Config>
34class IsoDirectoryPage {
35 MAKE_BMALLOCED;
36public:
37 // By my math, this results in an IsoDirectoryPage using 4036 bytes on 64-bit platforms. We allocate it
38 // with malloc, so it doesn't really matter how much memory it uses. But it's nice that this about a
39 // page, since that's quite intuitive.
40 //
41 // On 32-bit platforms, I think that this will be 2112 bytes. That's still quite intuitive.
42 //
43 // Note that this is a multiple of 32 so that it uses bitvectors efficiently.
44 static constexpr unsigned numPages = 480;
45
46 IsoDirectoryPage(IsoHeapImpl<Config>&, unsigned index);
47
48 static IsoDirectoryPage* pageFor(IsoDirectory<Config, numPages>* payload);
49
50 unsigned index() const { return m_index; }
51
52 IsoDirectory<Config, numPages> payload;
53 IsoDirectoryPage* next { nullptr };
54
55private:
56 unsigned m_index;
57};
58
59} // namespace bmalloc
