82template<typename Type>
83bool IsoHeap<Type>::isInitialized()
84{
85 auto* atomic = reinterpret_cast<std::atomic<IsoHeapImpl<Config>*>*>(&m_impl);
86 return atomic->load(std::memory_order_acquire);
87}
88
89template<typename Type>
90void IsoHeap<Type>::initialize()
91{
92 // We are using m_impl field as a guard variable of the initialization of IsoHeap.
93 // IsoHeap::isInitialized gets m_impl with "acquire", and IsoHeap::initialize stores
94 // the value to m_impl with "release". To make IsoHeap changes visible to any threads
95 // when IsoHeap::isInitialized returns true, we need to store the value to m_impl *after*
96 // all the initialization finishes.
97 auto* heap = new IsoHeapImpl<Config>();
98 heap->addToAllIsoHeaps();
99 setAllocatorOffset(heap->allocatorOffset());
100 setDeallocatorOffset(heap->deallocatorOffset());
101 auto* atomic = reinterpret_cast<std::atomic<IsoHeapImpl<Config>*>*>(&m_impl);
102 atomic->store(heap, std::memory_order_release);
103}
104
105template<typename Type>
106auto IsoHeap<Type>::impl() -> IsoHeapImpl<Config>&
107{
108 IsoTLS::ensureHeap(*this);
109 return *m_impl;
110}
