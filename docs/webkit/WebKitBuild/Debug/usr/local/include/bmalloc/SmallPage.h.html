26#ifndef SmallPage_h
27#define SmallPage_h
28
29#include "BAssert.h"
30#include "List.h"
31#include "Mutex.h"
32#include "VMAllocate.h"
33#include <mutex>
34
35namespace bmalloc {
36
37class SmallLine;
38
39class SmallPage : public ListNode<SmallPage> {
40public:
41 void ref(UniqueLockHolder&);
42 bool deref(UniqueLockHolder&);
43 unsigned refCount(UniqueLockHolder&) { return m_refCount; }
44
45 size_t sizeClass() { return m_sizeClass; }
46 void setSizeClass(size_t sizeClass) { m_sizeClass = sizeClass; }
47
48 bool hasFreeLines(UniqueLockHolder&) const { return m_hasFreeLines; }
49 void setHasFreeLines(UniqueLockHolder&, bool hasFreeLines) { m_hasFreeLines = hasFreeLines; }
50
51 bool hasPhysicalPages() { return m_hasPhysicalPages; }
52 void setHasPhysicalPages(bool hasPhysicalPages) { m_hasPhysicalPages = hasPhysicalPages; }
53
54#if !BUSE(PARTIAL_SCAVENGE)
55 bool usedSinceLastScavenge() { return m_usedSinceLastScavenge; }
56 void clearUsedSinceLastScavenge() { m_usedSinceLastScavenge = false; }
57 void setUsedSinceLastScavenge() { m_usedSinceLastScavenge = true; }
58#endif
59
60 SmallLine* begin();
61
62 unsigned char slide() const { return m_slide; }
63 void setSlide(unsigned char slide) { m_slide = slide; }
64
65private:
66 unsigned char m_hasFreeLines: 1;
67 unsigned char m_hasPhysicalPages: 1;
68#if !BUSE(PARTIAL_SCAVENGE)
69 unsigned char m_usedSinceLastScavenge: 1;
70#endif
71 unsigned char m_refCount: 7;
72 unsigned char m_sizeClass;
73 unsigned char m_slide;
74
75static_assert(
76 sizeClassCount <= std::numeric_limits<decltype(m_sizeClass)>::max(),
77 "Largest size class must fit in SmallPage metadata");
78};
79
80using LineCache = std::array<List<SmallPage>, sizeClassCount>;
81
82inline void SmallPage::ref(UniqueLockHolder&)
83{
84 BASSERT(!m_slide);
85 ++m_refCount;
86 BASSERT(m_refCount);
87}
88
89inline bool SmallPage::deref(UniqueLockHolder&)
90{
91 BASSERT(!m_slide);
92 BASSERT(m_refCount);
93 --m_refCount;
94 return !m_refCount;
95}
96
97} // namespace bmalloc
98
99#endif // SmallPage_h
