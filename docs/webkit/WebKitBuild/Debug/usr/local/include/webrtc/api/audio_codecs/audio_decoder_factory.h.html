12#define API_AUDIO_CODECS_AUDIO_DECODER_FACTORY_H_

17#include "absl/types/optional.h"
18#include "api/audio_codecs/audio_codec_pair_id.h"
19#include "api/audio_codecs/audio_decoder.h"
20#include "api/audio_codecs/audio_format.h"

23namespace webrtc {
24
25// A factory that creates AudioDecoders.
26class AudioDecoderFactory : public rtc::RefCountInterface {
27 public:
28 virtual std::vector<AudioCodecSpec> GetSupportedDecoders() = 0;
29
30 virtual bool IsSupportedDecoder(const SdpAudioFormat& format) = 0;
31
32 // Create a new decoder instance. The `codec_pair_id` argument is used to link
33 // encoders and decoders that talk to the same remote entity: if a
34 // AudioEncoderFactory::MakeAudioEncoder() and a
35 // AudioDecoderFactory::MakeAudioDecoder() call receive non-null IDs that
36 // compare equal, the factory implementations may assume that the encoder and
37 // decoder form a pair. (The intended use case for this is to set up
38 // communication between the AudioEncoder and AudioDecoder instances, which is
39 // needed for some codecs with built-in bandwidth adaptation.)
40 //
41 // Returns null if the format isn't supported.
42 //
43 // Note: Implementations need to be robust against combinations other than
44 // one encoder, one decoder getting the same ID; such decoders must still
45 // work.
46 virtual std::unique_ptr<AudioDecoder> MakeAudioDecoder(
47 const SdpAudioFormat& format,
48 absl::optional<AudioCodecPairId> codec_pair_id) = 0;
49};
50
51} // namespace webrtc

53#endif // API_AUDIO_CODECS_AUDIO_DECODER_FACTORY_H_
