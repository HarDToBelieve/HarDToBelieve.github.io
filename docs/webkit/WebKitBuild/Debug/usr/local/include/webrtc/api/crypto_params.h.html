11#ifndef API_CRYPTO_PARAMS_H_
12#define API_CRYPTO_PARAMS_H_
13
14#include <string>
15
16namespace cricket {
17
18// Parameters for SRTP negotiation, as described in RFC 4568.
19// TODO(benwright) - Rename to SrtpCryptoParams as these only apply to SRTP and
20// not generic crypto parameters for WebRTC.
21struct CryptoParams {
22 CryptoParams() : tag(0) {}
23 CryptoParams(int t,
24 const std::string& cs,
25 const std::string& kp,
26 const std::string& sp)
27 : tag(t), cipher_suite(cs), key_params(kp), session_params(sp) {}
28
29 bool Matches(const CryptoParams& params) const {
30 return (tag == params.tag && cipher_suite == params.cipher_suite);
31 }
32
33 int tag;
34 std::string cipher_suite;
35 std::string key_params;
36 std::string session_params;
37};
38
39} // namespace cricket
40
41#endif // API_CRYPTO_PARAMS_H_
