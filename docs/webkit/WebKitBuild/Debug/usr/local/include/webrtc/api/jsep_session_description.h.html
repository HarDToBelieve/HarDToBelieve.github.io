31namespace webrtc {
32
33// Implementation of SessionDescriptionInterface.
34class JsepSessionDescription : public SessionDescriptionInterface {
35 public:
36 explicit JsepSessionDescription(SdpType type);
37 // TODO(steveanton): Remove this once callers have switched to SdpType.
38 explicit JsepSessionDescription(const std::string& type);
39 JsepSessionDescription(
40 SdpType type,
41 std::unique_ptr<cricket::SessionDescription> description,
42 absl::string_view session_id,
43 absl::string_view session_version);
44 virtual ~JsepSessionDescription();
45
46 // Takes ownership of |description|.
47 bool Initialize(std::unique_ptr<cricket::SessionDescription> description,
48 const std::string& session_id,
49 const std::string& session_version);
50
51 virtual std::unique_ptr<SessionDescriptionInterface> Clone() const;
52
53 virtual cricket::SessionDescription* description() {
54 return description_.get();
55 }
56 virtual const cricket::SessionDescription* description() const {
57 return description_.get();
58 }
59 virtual std::string session_id() const { return session_id_; }
60 virtual std::string session_version() const { return session_version_; }
61 virtual SdpType GetType() const { return type_; }
62 virtual std::string type() const { return SdpTypeToString(type_); }
63 // Allows changing the type. Used for testing.
64 virtual bool AddCandidate(const IceCandidateInterface* candidate);
65 virtual size_t RemoveCandidates(
66 const std::vector<cricket::Candidate>& candidates);
67 virtual size_t number_of_mediasections() const;
68 virtual const IceCandidateCollection* candidates(
69 size_t mediasection_index) const;
70 virtual bool ToString(std::string* out) const;
71
72 static const int kDefaultVideoCodecId;
73 static const char kDefaultVideoCodecName[];
74
75 private:
76 std::unique_ptr<cricket::SessionDescription> description_;
77 std::string session_id_;
78 std::string session_version_;
79 SdpType type_;
80 std::vector<JsepCandidateCollection> candidate_collection_;
81
82 bool GetMediasectionIndex(const IceCandidateInterface* candidate,
83 size_t* index);
84 int GetMediasectionIndex(const cricket::Candidate& candidate);
85
86 RTC_DISALLOW_COPY_AND_ASSIGN(JsepSessionDescription);
87};
88
89} // namespace webrtc
