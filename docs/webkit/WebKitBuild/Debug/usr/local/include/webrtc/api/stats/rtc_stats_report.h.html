30namespace webrtc {
31
32// A collection of stats.
33// This is accessible as a map from |RTCStats::id| to |RTCStats|.
34class RTC_EXPORT RTCStatsReport final
35 : public rtc::RefCountedNonVirtual<RTCStatsReport> {
36 public:
37 typedef std::map<std::string, std::unique_ptr<const RTCStats>> StatsMap;
38
39 class RTC_EXPORT ConstIterator {
40 public:
41 ConstIterator(ConstIterator&& other);
42 ~ConstIterator();
43
44 ConstIterator& operator++();
45 ConstIterator& operator++(int);
46 const RTCStats& operator*() const;
47 const RTCStats* operator->() const;
48 bool operator==(const ConstIterator& other) const;
49 bool operator!=(const ConstIterator& other) const;
50
51 private:
52 friend class RTCStatsReport;
53 ConstIterator(const rtc::scoped_refptr<const RTCStatsReport>& report,
54 StatsMap::const_iterator it);
55
56 // Reference report to make sure it is kept alive.
57 rtc::scoped_refptr<const RTCStatsReport> report_;
58 StatsMap::const_iterator it_;
59 };
60
61 // TODO(hbos): Remove "= 0" once Chromium unittest has been updated to call
62 // with a parameter. crbug.com/627816
63 static rtc::scoped_refptr<RTCStatsReport> Create(int64_t timestamp_us = 0);
64
65 explicit RTCStatsReport(int64_t timestamp_us);
66 RTCStatsReport(const RTCStatsReport& other) = delete;
67 rtc::scoped_refptr<RTCStatsReport> Copy() const;
68
69 int64_t timestamp_us() const { return timestamp_us_; }
70 void AddStats(std::unique_ptr<const RTCStats> stats);
71 const RTCStats* Get(const std::string& id) const;
72 size_t size() const { return stats_.size(); }
73
74 // Gets the stat object of type |T| by ID, where |T| is any class descending
75 // from |RTCStats|.
76 // Returns null if there is no stats object for the given ID or it is the
77 // wrong type.
78 template <typename T>
79 const T* GetAs(const std::string& id) const {
80 const RTCStats* stats = Get(id);
81 if (!stats || stats->type() != T::kType) {
82 return nullptr;
83 }
84 return &stats->cast_to<const T>();
85 }
86
87 // Removes the stats object from the report, returning ownership of it or null
88 // if there is no object with |id|.
89 std::unique_ptr<const RTCStats> Take(const std::string& id);
90 // Takes ownership of all the stats in |other|, leaving it empty.
91 void TakeMembersFrom(rtc::scoped_refptr<RTCStatsReport> other);
92
93 // Stats iterators. Stats are ordered lexicographically on |RTCStats::id|.
94 ConstIterator begin() const;
95 ConstIterator end() const;
96
97 // Gets the subset of stats that are of type |T|, where |T| is any class
98 // descending from |RTCStats|.
99 template <typename T>
100 std::vector<const T*> GetStatsOfType() const {
101 std::vector<const T*> stats_of_type;
102 for (const RTCStats& stats : *this) {
103 if (stats.type() == T::kType)
104 stats_of_type.push_back(&stats.cast_to<const T>());
105 }
106 return stats_of_type;
107 }
108
109 // Creates a JSON readable string representation of the report,
110 // listing all of its stats objects.
111 std::string ToJson() const;
112
113 protected:
114 friend class rtc::RefCountedNonVirtual<RTCStatsReport>;
115 ~RTCStatsReport() = default;
116
117 private:
118 int64_t timestamp_us_;
119 StatsMap stats_;
120};
