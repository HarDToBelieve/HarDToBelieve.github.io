34// Abstract interface for buffer storage. Intended to support buffers owned by
35// external encoders with special release requirements, e.g, java encoders with
36// releaseOutputBuffer.
37class EncodedImageBufferInterface : public rtc::RefCountInterface {
38 public:
39 virtual const uint8_t* data() const = 0;
40 // TODO(bugs.webrtc.org/9378): Make interface essentially read-only, delete
41 // this non-const data method.
42 virtual uint8_t* data() = 0;
43 virtual size_t size() const = 0;
44};
45
46// Basic implementation of EncodedImageBufferInterface.
47class RTC_EXPORT EncodedImageBuffer : public EncodedImageBufferInterface {
48 public:
49 static rtc::scoped_refptr<EncodedImageBuffer> Create() { return Create(0); }
50 static rtc::scoped_refptr<EncodedImageBuffer> Create(size_t size);
51 static rtc::scoped_refptr<EncodedImageBuffer> Create(const uint8_t* data,
52 size_t size);
53
54 const uint8_t* data() const override;
55 uint8_t* data() override;
56 size_t size() const override;
57 void Realloc(size_t t);
58
59 protected:
60 explicit EncodedImageBuffer(size_t size);
61 EncodedImageBuffer(const uint8_t* data, size_t size);
62 ~EncodedImageBuffer();
63
64 size_t size_;
65 uint8_t* buffer_;
66};

68// TODO(bug.webrtc.org/9378): This is a legacy api class, which is slowly being
69// cleaned up. Direct use of its members is strongly discouraged.
70class RTC_EXPORT EncodedImage {
71 public:
72 EncodedImage();
73 EncodedImage(EncodedImage&&);
74 EncodedImage(const EncodedImage&);
75
76 ~EncodedImage();
77
78 EncodedImage& operator=(EncodedImage&&);
79 EncodedImage& operator=(const EncodedImage&);

81 // TODO(nisse): Change style to timestamp(), set_timestamp(), for consistency
82 // with the VideoFrame class.
83 // Set frame timestamp (90kHz).
84 void SetTimestamp(uint32_t timestamp) { timestamp_rtp_ = timestamp; }
85
86 // Get frame timestamp (90kHz).
87 uint32_t Timestamp() const { return timestamp_rtp_; }
88
89 void SetEncodeTime(int64_t encode_start_ms, int64_t encode_finish_ms);
90
91 int64_t NtpTimeMs() const { return ntp_time_ms_; }
92
93 absl::optional<int> SpatialIndex() const { return spatial_index_; }
94 void SetSpatialIndex(absl::optional<int> spatial_index) {
95 RTC_DCHECK_GE(spatial_index.value_or(0), 0);
96 RTC_DCHECK_LT(spatial_index.value_or(0), kMaxSpatialLayers);
97 spatial_index_ = spatial_index;
98 }
99
100 // These methods can be used to set/get size of subframe with spatial index
101 // |spatial_index| on encoded frames that consist of multiple spatial layers.
102 absl::optional<size_t> SpatialLayerFrameSize(int spatial_index) const;
103 void SetSpatialLayerFrameSize(int spatial_index, size_t size_bytes);
104
105 const webrtc::ColorSpace* ColorSpace() const {
106 return color_space_ ? &*color_space_ : nullptr;
107 }
108 void SetColorSpace(const absl::optional<webrtc::ColorSpace>& color_space) {
109 color_space_ = color_space;
110 }
111
112 // These methods along with the private member video_frame_tracking_id_ are
113 // meant for media quality testing purpose only.
114 absl::optional<uint16_t> VideoFrameTrackingId() const {
115 return video_frame_tracking_id_;
116 }
117 void SetVideoFrameTrackingId(absl::optional<uint16_t> tracking_id) {
118 video_frame_tracking_id_ = tracking_id;
119 }
120
121 const RtpPacketInfos& PacketInfos() const { return packet_infos_; }
122 void SetPacketInfos(RtpPacketInfos packet_infos) {
123 packet_infos_ = std::move(packet_infos);
124 }
125
126 bool RetransmissionAllowed() const { return retransmission_allowed_; }
127 void SetRetransmissionAllowed(bool retransmission_allowed) {
128 retransmission_allowed_ = retransmission_allowed;
129 }
130
131 size_t size() const { return size_; }
132 void set_size(size_t new_size) {
133 // Allow set_size(0) even if we have no buffer.
134 RTC_DCHECK_LE(new_size, new_size == 0 ? 0 : capacity());
135 size_ = new_size;
136 }
137
138 void SetEncodedData(
139 rtc::scoped_refptr<EncodedImageBufferInterface> encoded_data) {
140 encoded_data_ = encoded_data;
141 size_ = encoded_data->size();
142 }
143
144 void ClearEncodedData() {
145 encoded_data_ = nullptr;
146 size_ = 0;
147 }
148
149 rtc::scoped_refptr<EncodedImageBufferInterface> GetEncodedData() const {
150 return encoded_data_;
151 }
152
153 const uint8_t* data() const {
154 return encoded_data_ ? encoded_data_->data() : nullptr;
155 }
156
157 uint32_t _encodedWidth = 0;
158 uint32_t _encodedHeight = 0;
159 // NTP time of the capture time in local timebase in milliseconds.
160 // TODO(minyue): make this member private.
161 int64_t ntp_time_ms_ = 0;
162 int64_t capture_time_ms_ = 0;
163 VideoFrameType _frameType = VideoFrameType::kVideoFrameDelta;
164 VideoRotation rotation_ = kVideoRotation_0;
165 VideoContentType content_type_ = VideoContentType::UNSPECIFIED;
166 int qp_ = -1; // Quantizer value.
167
168 // When an application indicates non-zero values here, it is taken as an
169 // indication that all future frames will be constrained with those limits
170 // until the application indicates a change again.
171 VideoPlayoutDelay playout_delay_;
172
173 struct Timing {
174 uint8_t flags = VideoSendTiming::kInvalid;
175 int64_t encode_start_ms = 0;
176 int64_t encode_finish_ms = 0;
177 int64_t packetization_finish_ms = 0;
178 int64_t pacer_exit_ms = 0;
179 int64_t network_timestamp_ms = 0;
180 int64_t network2_timestamp_ms = 0;
181 int64_t receive_start_ms = 0;
182 int64_t receive_finish_ms = 0;
183 } timing_;
184
185 private:
186 size_t capacity() const { return encoded_data_ ? encoded_data_->size() : 0; }
187
188 rtc::scoped_refptr<EncodedImageBufferInterface> encoded_data_;
189 size_t size_ = 0; // Size of encoded frame data.
190 uint32_t timestamp_rtp_ = 0;
191 absl::optional<int> spatial_index_;
192 std::map<int, size_t> spatial_layer_frame_size_bytes_;
193 absl::optional<webrtc::ColorSpace> color_space_;
194 // This field is meant for media quality testing purpose only. When enabled it
195 // carries the webrtc::VideoFrame id field from the sender to the receiver.
196 absl::optional<uint16_t> video_frame_tracking_id_;
197 // Information about packets used to assemble this video frame. This is needed
198 // by |SourceTracker| when the frame is delivered to the RTCRtpReceiver's
199 // MediaStreamTrack, in order to implement getContributingSources(). See:
200 // https://w3c.github.io/webrtc-pc/#dom-rtcrtpreceiver-getcontributingsources
201 RtpPacketInfos packet_infos_;
202 bool retransmission_allowed_ = true;
203};
