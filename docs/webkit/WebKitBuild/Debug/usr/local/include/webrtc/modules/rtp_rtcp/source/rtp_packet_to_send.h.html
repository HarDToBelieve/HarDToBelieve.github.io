27namespace webrtc {
28// Class to hold rtp packet with metadata for sender side.
29// The metadata is not send over the wire, but packet sender may use it to
30// create rtp header extensions or other data that is sent over the wire.
31class RtpPacketToSend : public RtpPacket {
32 public:
33 // RtpPacketToSend::Type is deprecated. Use RtpPacketMediaType directly.
34 using Type = RtpPacketMediaType;
35
36 explicit RtpPacketToSend(const ExtensionManager* extensions);
37 RtpPacketToSend(const ExtensionManager* extensions, size_t capacity);
38 RtpPacketToSend(const RtpPacketToSend& packet);
39 RtpPacketToSend(RtpPacketToSend&& packet);
40
41 RtpPacketToSend& operator=(const RtpPacketToSend& packet);
42 RtpPacketToSend& operator=(RtpPacketToSend&& packet);
43
44 ~RtpPacketToSend();
45
46 // Time in local time base as close as it can to frame capture time.
47 int64_t capture_time_ms() const { return capture_time_ms_; }
48
49 void set_capture_time_ms(int64_t time) { capture_time_ms_ = time; }
50
51 void set_packet_type(RtpPacketMediaType type) { packet_type_ = type; }
52 absl::optional<RtpPacketMediaType> packet_type() const {
53 return packet_type_;
54 }
55
56 // If this is a retransmission, indicates the sequence number of the original
57 // media packet that this packet represents. If RTX is used this will likely
58 // be different from SequenceNumber().
59 void set_retransmitted_sequence_number(uint16_t sequence_number) {
60 retransmitted_sequence_number_ = sequence_number;
61 }
62 absl::optional<uint16_t> retransmitted_sequence_number() {
63 return retransmitted_sequence_number_;
64 }
65
66 void set_allow_retransmission(bool allow_retransmission) {
67 allow_retransmission_ = allow_retransmission;
68 }
69 bool allow_retransmission() { return allow_retransmission_; }
70
71 // An application can attach arbitrary data to an RTP packet using
72 // `additional_data`. The additional data does not affect WebRTC processing.
73 rtc::scoped_refptr<rtc::RefCountedBase> additional_data() const {
74 return additional_data_;
75 }
76 void set_additional_data(rtc::scoped_refptr<rtc::RefCountedBase> data) {
77 additional_data_ = std::move(data);
78 }
79
80 void set_packetization_finish_time_ms(int64_t time) {
81 SetExtension<VideoTimingExtension>(
82 VideoSendTiming::GetDeltaCappedMs(capture_time_ms_, time),
83 VideoTimingExtension::kPacketizationFinishDeltaOffset);
84 }
85
86 void set_pacer_exit_time_ms(int64_t time) {
87 SetExtension<VideoTimingExtension>(
88 VideoSendTiming::GetDeltaCappedMs(capture_time_ms_, time),
89 VideoTimingExtension::kPacerExitDeltaOffset);
90 }
91
92 void set_network_time_ms(int64_t time) {
93 SetExtension<VideoTimingExtension>(
94 VideoSendTiming::GetDeltaCappedMs(capture_time_ms_, time),
95 VideoTimingExtension::kNetworkTimestampDeltaOffset);
96 }
97
98 void set_network2_time_ms(int64_t time) {
99 SetExtension<VideoTimingExtension>(
100 VideoSendTiming::GetDeltaCappedMs(capture_time_ms_, time),
101 VideoTimingExtension::kNetwork2TimestampDeltaOffset);
102 }
103
104 // Indicates if packet is the first packet of a video frame.
105 void set_first_packet_of_frame(bool is_first_packet) {
106 is_first_packet_of_frame_ = is_first_packet;
107 }
108 bool is_first_packet_of_frame() const { return is_first_packet_of_frame_; }
109
110 // Indicates if packet contains payload for a video key-frame.
111 void set_is_key_frame(bool is_key_frame) { is_key_frame_ = is_key_frame; }
112 bool is_key_frame() const { return is_key_frame_; }
113
114 // Indicates if packets should be protected by FEC (Forward Error Correction).
115 void set_fec_protect_packet(bool protect) { fec_protect_packet_ = protect; }
116 bool fec_protect_packet() const { return fec_protect_packet_; }
117
118 // Indicates if packet is using RED encapsulation, in accordance with
119 // https://tools.ietf.org/html/rfc2198
120 void set_is_red(bool is_red) { is_red_ = is_red; }
121 bool is_red() const { return is_red_; }
122
123 private:
124 int64_t capture_time_ms_ = 0;
125 absl::optional<RtpPacketMediaType> packet_type_;
126 bool allow_retransmission_ = false;
127 absl::optional<uint16_t> retransmitted_sequence_number_;
128 rtc::scoped_refptr<rtc::RefCountedBase> additional_data_;
129 bool is_first_packet_of_frame_ = false;
130 bool is_key_frame_ = false;
131 bool fec_protect_packet_ = false;
132 bool is_red_ = false;
133};
134
135} // namespace webrtc
136#endif // MODULES_RTP_RTCP_SOURCE_RTP_PACKET_TO_SEND_H_
