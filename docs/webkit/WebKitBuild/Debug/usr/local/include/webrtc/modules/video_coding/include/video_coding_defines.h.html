24// Error codes

33enum {
34 // Timing frames settings. Timing frames are sent every
35 // |kDefaultTimingFramesDelayMs|, or if the frame is at least
36 // |kDefaultOutliserFrameSizePercent| in size of average frame.
37 kDefaultTimingFramesDelayMs = 200,
38 kDefaultOutlierFrameSizePercent = 500,
39 // Maximum number of frames for what we store encode start timing information.
40 kMaxEncodeStartTimeListSize = 150,
41};

43enum VCMVideoProtection {
44 kProtectionNack,
45 kProtectionNackFEC,
46};

48// Callback class used for passing decoded frames which are ready to be
49// rendered.
50class VCMReceiveCallback {
51 public:
52 virtual int32_t FrameToRender(VideoFrame& videoFrame, // NOLINT
53 absl::optional<uint8_t> qp,
54 int32_t decode_time_ms,
55 VideoContentType content_type) = 0;
56
57 virtual void OnDroppedFrames(uint32_t frames_dropped);
58
59 // Called when the current receive codec changes.
60 virtual void OnIncomingPayloadType(int payload_type);
61 virtual void OnDecoderImplementationName(const char* implementation_name);
62
63 protected:
64 virtual ~VCMReceiveCallback() {}
65};
66
67// Callback class used for informing the user of the incoming bit rate and frame
68// rate.
69class VCMReceiveStatisticsCallback {
70 public:
71 virtual void OnCompleteFrame(bool is_keyframe,
72 size_t size_bytes,
73 VideoContentType content_type) = 0;
74
75 virtual void OnDroppedFrames(uint32_t frames_dropped) = 0;
76
77 virtual void OnFrameBufferTimingsUpdated(int max_decode_ms,
78 int current_delay_ms,
79 int target_delay_ms,
80 int jitter_buffer_ms,
81 int min_playout_delay_ms,
82 int render_delay_ms) = 0;
83
84 virtual void OnTimingFrameInfoUpdated(const TimingFrameInfo& info) = 0;
85
86 protected:
87 virtual ~VCMReceiveStatisticsCallback() {}
88};
89
90// Callback class used for telling the user about what frame type needed to
91// continue decoding.
92// Typically a key frame when the stream has been corrupted in some way.
93class VCMFrameTypeCallback {
94 public:
95 virtual int32_t RequestKeyFrame() = 0;
96
97 protected:
98 virtual ~VCMFrameTypeCallback() {}
99};

101// Callback class used for telling the user about which packet sequence numbers

104// TODO(philipel): Deprecate VCMPacketRequestCallback

110
111 protected:
112 virtual ~VCMPacketRequestCallback() {}
113};
