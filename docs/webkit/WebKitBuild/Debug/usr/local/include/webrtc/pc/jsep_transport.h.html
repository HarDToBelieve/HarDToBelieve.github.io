58struct JsepTransportDescription {
59 public:
60 JsepTransportDescription();
61 JsepTransportDescription(
62 bool rtcp_mux_enabled,
63 const std::vector<CryptoParams>& cryptos,
64 const std::vector<int>& encrypted_header_extension_ids,
65 int rtp_abs_sendtime_extn_id,
66 const TransportDescription& transport_description);
67 JsepTransportDescription(const JsepTransportDescription& from);
68 ~JsepTransportDescription();
69
70 JsepTransportDescription& operator=(const JsepTransportDescription& from);
71
72 bool rtcp_mux_enabled = true;
73 std::vector<CryptoParams> cryptos;
74 std::vector<int> encrypted_header_extension_ids;
75 int rtp_abs_sendtime_extn_id = -1;
76 // TODO(zhihuang): Add the ICE and DTLS related variables and methods from
77 // TransportDescription and remove this extra layer of abstraction.
78 TransportDescription transport_desc;
79};

81// Helper class used by JsepTransportController that processes
82// TransportDescriptions. A TransportDescription represents the
83// transport-specific properties of an SDP m= section, processed according to
84// JSEP. Each transport consists of DTLS and ICE transport channels for RTP
85// (and possibly RTCP, if rtcp-mux isn't used).
86//
87// On Threading: JsepTransport performs work solely on the network thread, and
88// so its methods should only be called on the network thread.
89class JsepTransport : public sigslot::has_slots<> {
90 public:
91 // |mid| is just used for log statements in order to identify the Transport.
92 // Note that |local_certificate| is allowed to be null since a remote
93 // description may be set before a local certificate is generated.
94 JsepTransport(
95 const std::string& mid,
96 const rtc::scoped_refptr<rtc::RTCCertificate>& local_certificate,
97 rtc::scoped_refptr<webrtc::IceTransportInterface> ice_transport,
98 rtc::scoped_refptr<webrtc::IceTransportInterface> rtcp_ice_transport,
99 std::unique_ptr<webrtc::RtpTransport> unencrypted_rtp_transport,
100 std::unique_ptr<webrtc::SrtpTransport> sdes_transport,
101 std::unique_ptr<webrtc::DtlsSrtpTransport> dtls_srtp_transport,
102 std::unique_ptr<DtlsTransportInternal> rtp_dtls_transport,
103 std::unique_ptr<DtlsTransportInternal> rtcp_dtls_transport,
104 std::unique_ptr<SctpTransportInternal> sctp_transport);
105
106 ~JsepTransport() override;
107
108 // Returns the MID of this transport. This is only used for logging.
109 const std::string& mid() const { return mid_; }
110
111 // Must be called before applying local session description.
112 // Needed in order to verify the local fingerprint.
113 void SetLocalCertificate(
114 const rtc::scoped_refptr<rtc::RTCCertificate>& local_certificate) {
115 RTC_DCHECK_RUN_ON(network_thread_);
116 local_certificate_ = local_certificate;
117 }
118
119 // Return the local certificate provided by SetLocalCertificate.
120 rtc::scoped_refptr<rtc::RTCCertificate> GetLocalCertificate() const {
121 RTC_DCHECK_RUN_ON(network_thread_);
122 return local_certificate_;
123 }
124
125 webrtc::RTCError SetLocalJsepTransportDescription(
126 const JsepTransportDescription& jsep_description,
127 webrtc::SdpType type);
128
129 // Set the remote TransportDescription to be used by DTLS and ICE channels
130 // that are part of this Transport.
131 webrtc::RTCError SetRemoteJsepTransportDescription(
132 const JsepTransportDescription& jsep_description,
133 webrtc::SdpType type);
134 webrtc::RTCError AddRemoteCandidates(const Candidates& candidates);
135
136 // Set the "needs-ice-restart" flag as described in JSEP. After the flag is
137 // set, offers should generate new ufrags/passwords until an ICE restart
138 // occurs.
139 //
140 // This and |needs_ice_restart()| must be called on the network thread.
141 void SetNeedsIceRestartFlag();
142
143 // Returns true if the ICE restart flag above was set, and no ICE restart has
144 // occurred yet for this transport (by applying a local description with
145 // changed ufrag/password).
146 bool needs_ice_restart() const {
147 RTC_DCHECK_RUN_ON(network_thread_);
148 return needs_ice_restart_;
149 }
150
151 // Returns role if negotiated, or empty absl::optional if it hasn't been
152 // negotiated yet.
153 absl::optional<rtc::SSLRole> GetDtlsRole() const;
154
155 // TODO(deadbeef): Make this const. See comment in transportcontroller.h.
156 bool GetStats(TransportStats* stats);
157
158 const JsepTransportDescription* local_description() const {
159 RTC_DCHECK_RUN_ON(network_thread_);
160 return local_description_.get();
161 }
162
163 const JsepTransportDescription* remote_description() const {
164 RTC_DCHECK_RUN_ON(network_thread_);
165 return remote_description_.get();
166 }
167
168 // Returns the rtp transport, if any.
169 webrtc::RtpTransportInternal* rtp_transport() const {
170 if (dtls_srtp_transport_) {
171 return dtls_srtp_transport_.get();
172 }
173 if (sdes_transport_) {
174 return sdes_transport_.get();
175 }
176 if (unencrypted_rtp_transport_) {
177 return unencrypted_rtp_transport_.get();
178 }
179 return nullptr;
180 }
181
182 const DtlsTransportInternal* rtp_dtls_transport() const {
183 if (rtp_dtls_transport_) {
184 return rtp_dtls_transport_->internal();
185 }
186 return nullptr;
187 }
188
189 DtlsTransportInternal* rtp_dtls_transport() {
190 if (rtp_dtls_transport_) {
191 return rtp_dtls_transport_->internal();
192 }
193 return nullptr;
194 }
195
196 const DtlsTransportInternal* rtcp_dtls_transport() const {
197 RTC_DCHECK_RUN_ON(network_thread_);
198 if (rtcp_dtls_transport_) {
199 return rtcp_dtls_transport_->internal();
200 }
201 return nullptr;
202 }
203
204 DtlsTransportInternal* rtcp_dtls_transport() {
205 RTC_DCHECK_RUN_ON(network_thread_);
206 if (rtcp_dtls_transport_) {
207 return rtcp_dtls_transport_->internal();
208 }
209 return nullptr;
210 }
211
212 rtc::scoped_refptr<webrtc::DtlsTransport> RtpDtlsTransport() {
213 return rtp_dtls_transport_;
214 }
215
216 rtc::scoped_refptr<webrtc::SctpTransport> SctpTransport() const {
217 return sctp_transport_;
218 }
219
220 // TODO(bugs.webrtc.org/9719): Delete method, update callers to use
221 // SctpTransport() instead.
222 webrtc::DataChannelTransportInterface* data_channel_transport() const {
223 if (sctp_data_channel_transport_) {
224 return sctp_data_channel_transport_.get();
225 }
226 return nullptr;
227 }
228
229 // This is signaled when RTCP-mux becomes active and
230 // |rtcp_dtls_transport_| is destroyed. The JsepTransportController will
231 // handle the signal and update the aggregate transport states.
232 sigslot::signal<> SignalRtcpMuxActive;
233
234 // TODO(deadbeef): The methods below are only public for testing. Should make
235 // them utility functions or objects so they can be tested independently from
236 // this class.
237
238 // Returns an error if the certificate's identity does not match the
239 // fingerprint, or either is NULL.
240 webrtc::RTCError VerifyCertificateFingerprint(
241 const rtc::RTCCertificate* certificate,
242 const rtc::SSLFingerprint* fingerprint) const;
243
244 void SetActiveResetSrtpParams(bool active_reset_srtp_params);
245
246 private:
247 bool SetRtcpMux(bool enable, webrtc::SdpType type, ContentSource source);
248
249 void ActivateRtcpMux() RTC_RUN_ON(network_thread_);
250
251 bool SetSdes(const std::vector<CryptoParams>& cryptos,
252 const std::vector<int>& encrypted_extension_ids,
253 webrtc::SdpType type,
254 ContentSource source);
255
256 // Negotiates and sets the DTLS parameters based on the current local and
257 // remote transport description, such as the DTLS role to use, and whether
258 // DTLS should be activated.
259 //
260 // Called when an answer TransportDescription is applied.
261 webrtc::RTCError NegotiateAndSetDtlsParameters(
262 webrtc::SdpType local_description_type);
263
264 // Negotiates the DTLS role based off the offer and answer as specified by
265 // RFC 4145, section-4.1. Returns an RTCError if role cannot be determined
266 // from the local description and remote description.
267 webrtc::RTCError NegotiateDtlsRole(
268 webrtc::SdpType local_description_type,
269 ConnectionRole local_connection_role,
270 ConnectionRole remote_connection_role,
271 absl::optional<rtc::SSLRole>* negotiated_dtls_role);
272
273 // Pushes down the ICE parameters from the remote description.
274 void SetRemoteIceParameters(const IceParameters& ice_parameters,
275 IceTransportInternal* ice);
276
277 // Pushes down the DTLS parameters obtained via negotiation.
278 static webrtc::RTCError SetNegotiatedDtlsParameters(
279 DtlsTransportInternal* dtls_transport,
280 absl::optional<rtc::SSLRole> dtls_role,
281 rtc::SSLFingerprint* remote_fingerprint);
282
283 bool GetTransportStats(DtlsTransportInternal* dtls_transport,
284 int component,
285 TransportStats* stats);
286
287 // Owning thread, for safety checks
288 const rtc::Thread* const network_thread_;
289 const std::string mid_;
290 // needs-ice-restart bit as described in JSEP.
291 bool needs_ice_restart_ RTC_GUARDED_BY(network_thread_) = false;
292 rtc::scoped_refptr<rtc::RTCCertificate> local_certificate_
293 RTC_GUARDED_BY(network_thread_);
294 std::unique_ptr<JsepTransportDescription> local_description_
295 RTC_GUARDED_BY(network_thread_);
296 std::unique_ptr<JsepTransportDescription> remote_description_
297 RTC_GUARDED_BY(network_thread_);
298
299 // Ice transport which may be used by any of upper-layer transports (below).
300 // Owned by JsepTransport and guaranteed to outlive the transports below.
301 const rtc::scoped_refptr<webrtc::IceTransportInterface> ice_transport_;
302 const rtc::scoped_refptr<webrtc::IceTransportInterface> rtcp_ice_transport_;
303
304 // To avoid downcasting and make it type safe, keep three unique pointers for
305 // different SRTP mode and only one of these is non-nullptr.
306 const std::unique_ptr<webrtc::RtpTransport> unencrypted_rtp_transport_;
307 const std::unique_ptr<webrtc::SrtpTransport> sdes_transport_;
308 const std::unique_ptr<webrtc::DtlsSrtpTransport> dtls_srtp_transport_;
309
310 const rtc::scoped_refptr<webrtc::DtlsTransport> rtp_dtls_transport_;
311 // The RTCP transport is const for all usages, except that it is cleared
312 // when RTCP multiplexing is turned on; this happens on the network thread.
313 rtc::scoped_refptr<webrtc::DtlsTransport> rtcp_dtls_transport_
314 RTC_GUARDED_BY(network_thread_);
315
316 const std::unique_ptr<webrtc::DataChannelTransportInterface>
317 sctp_data_channel_transport_;
318 const rtc::scoped_refptr<webrtc::SctpTransport> sctp_transport_;
319
320 SrtpFilter sdes_negotiator_ RTC_GUARDED_BY(network_thread_);
321 RtcpMuxFilter rtcp_mux_negotiator_ RTC_GUARDED_BY(network_thread_);
322
323 // Cache the encrypted header extension IDs for SDES negoitation.
324 absl::optional<std::vector<int>> send_extension_ids_
325 RTC_GUARDED_BY(network_thread_);
326 absl::optional<std::vector<int>> recv_extension_ids_
327 RTC_GUARDED_BY(network_thread_);
328
329 RTC_DISALLOW_COPY_AND_ASSIGN(JsepTransport);
330};
