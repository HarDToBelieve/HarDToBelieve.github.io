54class RtcEventLog;
55
56class PeerConnectionFactory : public PeerConnectionFactoryInterface {
57 public:
58 // Creates a PeerConnectionFactory. It returns nullptr on initialization
59 // error.
60 //
61 // The Dependencies structure allows simple management of all new
62 // dependencies being added to the PeerConnectionFactory.
63 static rtc::scoped_refptr<PeerConnectionFactory> Create(
64 PeerConnectionFactoryDependencies dependencies);
65
66 void SetOptions(const Options& options) override;
67
68 RTCErrorOr<rtc::scoped_refptr<PeerConnectionInterface>>
69 CreatePeerConnectionOrError(
70 const PeerConnectionInterface::RTCConfiguration& configuration,
71 PeerConnectionDependencies dependencies) override;
72
73 RtpCapabilities GetRtpSenderCapabilities(
74 cricket::MediaType kind) const override;
75
76 RtpCapabilities GetRtpReceiverCapabilities(
77 cricket::MediaType kind) const override;
78
79 rtc::scoped_refptr<MediaStreamInterface> CreateLocalMediaStream(
80 const std::string& stream_id) override;
81
82 rtc::scoped_refptr<AudioSourceInterface> CreateAudioSource(
83 const cricket::AudioOptions& options) override;
84
85 rtc::scoped_refptr<VideoTrackInterface> CreateVideoTrack(
86 const std::string& id,
87 VideoTrackSourceInterface* video_source) override;
88
89 rtc::scoped_refptr<AudioTrackInterface> CreateAudioTrack(
90 const std::string& id,
91 AudioSourceInterface* audio_source) override;
92
93 bool StartAecDump(FILE* file, int64_t max_size_bytes) override;
94 void StopAecDump() override;
95
96 SctpTransportFactoryInterface* sctp_transport_factory() {
97 return context_->sctp_transport_factory();
98 }
99
100 virtual cricket::ChannelManager* channel_manager();
101
102 rtc::Thread* signaling_thread() const {
103 // This method can be called on a different thread when the factory is
104 // created in CreatePeerConnectionFactory().
105 return context_->signaling_thread();
106 }
107
108 rtc::Thread* worker_thread() const { return context_->worker_thread(); }
109
110 const Options& options() const {
111 RTC_DCHECK_RUN_ON(signaling_thread());
112 return options_;
113 }
114
115 const WebRtcKeyValueConfig& trials() const { return context_->trials(); }
116
117 protected:
118 // Constructor used by the static Create() method. Modifies the dependencies.
119 PeerConnectionFactory(rtc::scoped_refptr<ConnectionContext> context,
120 PeerConnectionFactoryDependencies* dependencies);
121
122 // Constructor for use in testing. Ignores the possibility of initialization
123 // failure. The dependencies are passed in by std::move().
124 explicit PeerConnectionFactory(
125 PeerConnectionFactoryDependencies dependencies);
126
127 virtual ~PeerConnectionFactory();
128
129 private:
130 rtc::Thread* network_thread() const { return context_->network_thread(); }
131
132 bool IsTrialEnabled(absl::string_view key) const;
133 const cricket::ChannelManager* channel_manager() const {
134 return context_->channel_manager();
135 }
136
137 std::unique_ptr<RtcEventLog> CreateRtcEventLog_w();
138 std::unique_ptr<Call> CreateCall_w(RtcEventLog* event_log);
139
140 rtc::scoped_refptr<ConnectionContext> context_;
141 PeerConnectionFactoryInterface::Options options_
142 RTC_GUARDED_BY(signaling_thread());
143 std::unique_ptr<TaskQueueFactory> task_queue_factory_;
144 std::unique_ptr<RtcEventLogFactoryInterface> event_log_factory_;
145 std::unique_ptr<FecControllerFactoryInterface> fec_controller_factory_;
146 std::unique_ptr<NetworkStatePredictorFactoryInterface>
147 network_state_predictor_factory_;
148 std::unique_ptr<NetworkControllerFactoryInterface>
149 injected_network_controller_factory_;
150 std::unique_ptr<NetEqFactory> neteq_factory_;
151};
