52// All public methods of the collector are to be called on the signaling thread.
53// Stats are gathered on the signaling, worker and network threads
54// asynchronously. The callback is invoked on the signaling thread. Resulting
55// reports are cached for |cache_lifetime_| ms.
56class RTCStatsCollector : public rtc::RefCountInterface,
57 public sigslot::has_slots<> {
58 public:
59 static rtc::scoped_refptr<RTCStatsCollector> Create(
60 PeerConnectionInternal* pc,
61 int64_t cache_lifetime_us = 50 * rtc::kNumMicrosecsPerMillisec);
62
63 // Gets a recent stats report. If there is a report cached that is still fresh
64 // it is returned, otherwise new stats are gathered and returned. A report is
65 // considered fresh for |cache_lifetime_| ms. const RTCStatsReports are safe
66 // to use across multiple threads and may be destructed on any thread.
67 // If the optional selector argument is used, stats are filtered according to
68 // stats selection algorithm before delivery.
69 // https://w3c.github.io/webrtc-pc/#dfn-stats-selection-algorithm
70 void GetStatsReport(rtc::scoped_refptr<RTCStatsCollectorCallback> callback);
71 // If |selector| is null the selection algorithm is still applied (interpreted
72 // as: no RTP streams are sent by selector). The result is empty.
73 void GetStatsReport(rtc::scoped_refptr<RtpSenderInternal> selector,
74 rtc::scoped_refptr<RTCStatsCollectorCallback> callback);
75 // If |selector| is null the selection algorithm is still applied (interpreted
76 // as: no RTP streams are received by selector). The result is empty.
77 void GetStatsReport(rtc::scoped_refptr<RtpReceiverInternal> selector,
78 rtc::scoped_refptr<RTCStatsCollectorCallback> callback);
79 // Clears the cache's reference to the most recent stats report. Subsequently
80 // calling |GetStatsReport| guarantees fresh stats.
81 void ClearCachedStatsReport();
82
83 // If there is a |GetStatsReport| requests in-flight, waits until it has been
84 // completed. Must be called on the signaling thread.
85 void WaitForPendingRequest();
86
87 protected:
88 RTCStatsCollector(PeerConnectionInternal* pc, int64_t cache_lifetime_us);
89 ~RTCStatsCollector();

151 // Structure for tracking stats about each RtpTransceiver managed by the
152 // PeerConnection. This can either by a Plan B style or Unified Plan style
153 // transceiver (i.e., can have 0 or many senders and receivers).
154 // Some fields are copied from the RtpTransceiver/BaseChannel object so that
155 // they can be accessed safely on threads other than the signaling thread.
156 // If a BaseChannel is not available (e.g., if signaling has not started),
157 // then |mid| and |transport_name| will be null.
158 struct RtpTransceiverStatsInfo {
159 rtc::scoped_refptr<RtpTransceiver> transceiver;
160 cricket::MediaType media_type;
161 absl::optional<std::string> mid;
162 absl::optional<std::string> transport_name;
163 std::unique_ptr<TrackMediaInfoMap> track_media_info_map;
164 };

270 // Cleared and set in `PrepareTransceiverStatsInfosAndCallStats_s_w_n`,
271 // starting out on the signaling thread, then network. Later read on the
272 // network and signaling threads as part of collecting stats and finally
273 // reset when the work is done. Initially this variable was added and not
274 // passed around as an arguments to avoid copies. This is thread safe due to
275 // how operations are sequenced and we don't start the stats collection
276 // sequence if one is in progress. As a future improvement though, we could
277 // now get rid of the variable and keep the data scoped within a stats
278 // collection sequence.

291 // Data recorded and maintained by the stats collector during its lifetime.
292 // Some stats are produced from this record instead of other components.
293 struct InternalRecord {
294 InternalRecord() : data_channels_opened(0), data_channels_closed(0) {}
295
296 // The opened count goes up when a channel is fully opened and the closed
297 // count goes up if a previously opened channel has fully closed. The opened
298 // count does not go down when a channel closes, meaning (opened - closed)
299 // is the number of channels currently opened. A channel that is closed
300 // before reaching the open state does not affect these counters.
301 uint32_t data_channels_opened;
302 uint32_t data_channels_closed;
303 // Identifies by address channels that have been opened, which remain in the
304 // set until they have been fully closed.
305 std::set<uintptr_t> opened_data_channels;
306 };
307 InternalRecord internal_record_;
