11// This file contain functions for parsing and serializing SDP messages.
12// Related RFC/draft including:
13// * RFC 4566 - SDP
14// * RFC 5245 - ICE
15// * RFC 3388 - Grouping of Media Lines in SDP
16// * RFC 4568 - SDP Security Descriptions for Media Streams
17// * draft-lennox-mmusic-sdp-source-selection-02 -
18// Mechanisms for Media Source Selection in SDP
19

33namespace cricket {
34class Candidate;
35} // namespace cricket

37namespace rtc {
38class StringBuilder;
39} // namespace rtc

41namespace webrtc {
42class IceCandidateInterface;
43class JsepIceCandidate;
44class JsepSessionDescription;
45struct SdpParseError;
46
47// Serializes the passed in JsepSessionDescription.
48// Serialize SessionDescription including candidates if
49// JsepSessionDescription has candidates.
50// jdesc - The JsepSessionDescription object to be serialized.
51// unified_plan_sdp - If set to true, include "a=msid" lines where appropriate.
52// return - SDP string serialized from the arguments.
53std::string SdpSerialize(const JsepSessionDescription& jdesc);
54
55// Serializes the passed in IceCandidateInterface to a SDP string.
56// candidate - The candidate to be serialized.
57std::string SdpSerializeCandidate(const IceCandidateInterface& candidate);
58
59// Serializes a cricket Candidate.
60// candidate - The candidate to be serialized.
61RTC_EXPORT std::string SdpSerializeCandidate(
62 const cricket::Candidate& candidate);
63
64// Deserializes the passed in SDP string to a JsepSessionDescription.
65// message - SDP string to be Deserialized.
66// jdesc - The JsepSessionDescription deserialized from the SDP string.
67// error - The detail error information when parsing fails.
68// return - true on success, false on failure.
69bool SdpDeserialize(const std::string& message,
70 JsepSessionDescription* jdesc,
71 SdpParseError* error);
72
73// Deserializes the passed in SDP string to one JsepIceCandidate.
74// The first line must be a=candidate line and only the first line will be
75// parsed.
76// message - The SDP string to be Deserialized.
77// candidates - The JsepIceCandidate from the SDP string.
78// error - The detail error information when parsing fails.
79// return - true on success, false on failure.
80RTC_EXPORT bool SdpDeserializeCandidate(const std::string& message,
81 JsepIceCandidate* candidate,
82 SdpParseError* error);
83
84// Deserializes the passed in SDP string to a cricket Candidate.
85// The first line must be a=candidate line and only the first line will be
86// parsed.
87// transport_name - The transport name (MID) of the candidate.
88// message - The SDP string to be deserialized.
89// candidate - The cricket Candidate from the SDP string.
90// error - The detail error information when parsing fails.
91// return - true on success, false on failure.
92RTC_EXPORT bool SdpDeserializeCandidate(const std::string& transport_name,
93 const std::string& message,
94 cricket::Candidate* candidate,
95 SdpParseError* error);
96
97// Parses |message| according to the grammar defined in RFC 5245, Section 15.1
98// and, if successful, stores the result in |candidate| and returns true.
99// If unsuccessful, returns false and stores error information in |error| if
100// |error| is not null.
101// If |is_raw| is false, |message| is expected to be prefixed with "a=".
102// If |is_raw| is true, no prefix is expected in |messaage|.
103RTC_EXPORT bool ParseCandidate(const std::string& message,
104 cricket::Candidate* candidate,
105 SdpParseError* error,
106 bool is_raw);
107
108// Generates an FMTP line based on |parameters|. Please note that some
109// parameters are not considered to be part of the FMTP line, see the function
110// IsFmtpParam(). Returns true if the set of FMTP parameters is nonempty, false
111// otherwise.
112bool WriteFmtpParameters(const cricket::CodecParameterMap& parameters,
113 rtc::StringBuilder* os);
