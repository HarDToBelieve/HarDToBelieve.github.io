32template<size_t size>
33using BitmapWordType = std::conditional_t<(size <= 32 && sizeof(UCPURegister) > sizeof(uint32_t)), uint32_t, UCPURegister>;
34
35template<size_t bitmapSize, typename WordType = BitmapWordType<bitmapSize>>
36class Bitmap final {
37 WTF_MAKE_FAST_ALLOCATED;
38
39 static_assert(sizeof(WordType) <= sizeof(UCPURegister), "WordType must not be bigger than the CPU atomic word size");
40public:
41 constexpr Bitmap();
42
43 static constexpr size_t size()
44 {
45 return bitmapSize;
46 }
47
48 bool get(size_t, Dependency = Dependency()) const;
49 void set(size_t);
50 void set(size_t, bool);
51 bool testAndSet(size_t);
52 bool testAndClear(size_t);
53 bool concurrentTestAndSet(size_t, Dependency = Dependency());
54 bool concurrentTestAndClear(size_t, Dependency = Dependency());
55 size_t nextPossiblyUnset(size_t) const;
56 void clear(size_t);
57 void clearAll();
58 void invert();
59 int64_t findRunOfZeros(size_t runLength) const;
60 size_t count(size_t start = 0) const;
61 bool isEmpty() const;
62 bool isFull() const;
63
64 void merge(const Bitmap&);
65 void filter(const Bitmap&);
66 void exclude(const Bitmap&);
67
68 void concurrentFilter(const Bitmap&);
69
70 bool subsumes(const Bitmap&) const;
71
72 template<typename Func>
73 void forEachSetBit(const Func&) const;
74
75 size_t findBit(size_t startIndex, bool value) const;
76
77 class iterator {
78 WTF_MAKE_FAST_ALLOCATED;
79 public:
80 iterator()
81 : m_bitmap(nullptr)
82 , m_index(0)
83 {
84 }
85
86 iterator(const Bitmap& bitmap, size_t index)
87 : m_bitmap(&bitmap)
88 , m_index(index)
89 {
90 }
91
92 size_t operator*() const { return m_index; }
93
94 iterator& operator++()
95 {
96 m_index = m_bitmap->findBit(m_index + 1, true);
97 return *this;
98 }
99
100 bool operator==(const iterator& other) const
101 {
102 return m_index == other.m_index;
103 }
104
105 bool operator!=(const iterator& other) const
106 {
107 return !(*this == other);
108 }
109
110 private:
111 const Bitmap* m_bitmap;
112 size_t m_index;
113 };
114
115 // Use this to iterate over set bits.
116 iterator begin() const { return iterator(*this, findBit(0, true)); }
117 iterator end() const { return iterator(*this, bitmapSize); }
118
119 void mergeAndClear(Bitmap&);
120 void setAndClear(Bitmap&);
121
122 void setEachNthBit(size_t n, size_t start = 0, size_t end = bitmapSize);
123
124 bool operator==(const Bitmap&) const;
125 bool operator!=(const Bitmap&) const;
126
127 void operator|=(const Bitmap&);
128 void operator&=(const Bitmap&);
129 void operator^=(const Bitmap&);
130
131 unsigned hash() const;
132
133private:
134 static constexpr unsigned wordSize = sizeof(WordType) * 8;
135 static constexpr unsigned words = (bitmapSize + wordSize - 1) / wordSize;
136
137 // the literal '1' is of type signed int. We want to use an unsigned
138 // version of the correct size when doing the calculations because if
139 // WordType is larger than int, '1 << 31' will first be sign extended
140 // and then casted to unsigned, meaning that set(31) when WordType is
141 // a 64 bit unsigned int would give 0xffff8000
142 static constexpr WordType one = 1;
143
144 std::array<WordType, words> bits;
145};

147template<size_t bitmapSize, typename WordType>
148constexpr Bitmap<bitmapSize, WordType>::Bitmap()
149{
150 clearAll();
151}
152
153template<size_t bitmapSize, typename WordType>
154inline bool Bitmap<bitmapSize, WordType>::get(size_t n, Dependency dependency) const
155{
156 return !!(dependency.consume(this)->bits[n / wordSize] & (one << (n % wordSize)));
157}
158
159template<size_t bitmapSize, typename WordType>
160inline void Bitmap<bitmapSize, WordType>::set(size_t n)
161{
162 bits[n / wordSize] |= (one << (n % wordSize));
163}
164
165template<size_t bitmapSize, typename WordType>
166inline void Bitmap<bitmapSize, WordType>::set(size_t n, bool value)
167{
168 if (value)
169 set(n);
170 else
171 clear(n);
172}
173
174template<size_t bitmapSize, typename WordType>
175inline bool Bitmap<bitmapSize, WordType>::testAndSet(size_t n)
176{
177 WordType mask = one << (n % wordSize);
178 size_t index = n / wordSize;
179 bool result = bits[index] & mask;
180 bits[index] |= mask;
181 return result;
182}
183
184template<size_t bitmapSize, typename WordType>
185inline bool Bitmap<bitmapSize, WordType>::testAndClear(size_t n)
186{
187 WordType mask = one << (n % wordSize);
188 size_t index = n / wordSize;
189 bool result = bits[index] & mask;
190 bits[index] &= ~mask;
191 return result;
192}

194template<size_t bitmapSize, typename WordType>
195ALWAYS_INLINE bool Bitmap<bitmapSize, WordType>::concurrentTestAndSet(size_t n, Dependency dependency)
196{
197 WordType mask = one << (n % wordSize);
198 size_t index = n / wordSize;
199 WordType* data = dependency.consume(bits.data()) + index;
200 return !bitwise_cast<Atomic<WordType>*>(data)->transactionRelaxed(
201 [&] (WordType& value) -> bool {
202 if (value & mask)
203 return false;
204
205 value |= mask;
206 return true;
207 });
208}
209
210template<size_t bitmapSize, typename WordType>
211ALWAYS_INLINE bool Bitmap<bitmapSize, WordType>::concurrentTestAndClear(size_t n, Dependency dependency)
212{
213 WordType mask = one << (n % wordSize);
214 size_t index = n / wordSize;
215 WordType* data = dependency.consume(bits.data()) + index;
216 return !bitwise_cast<Atomic<WordType>*>(data)->transactionRelaxed(
217 [&] (WordType& value) -> bool {
218 if (!(value & mask))
219 return false;
220
221 value &= ~mask;
222 return true;
223 });
224}

226template<size_t bitmapSize, typename WordType>
227inline void Bitmap<bitmapSize, WordType>::clear(size_t n)
228{
229IGNORE_ARRAY_BOUNDS_WARNINGS_BEGIN
230 bits[n / wordSize] &= ~(one << (n % wordSize));
231IGNORE_ARRAY_BOUNDS_WARNINGS_END
232}

234template<size_t bitmapSize, typename WordType>
235inline void Bitmap<bitmapSize, WordType>::clearAll()
236{
237 memset(bits.data(), 0, sizeof(bits));
238}
239
240template<size_t bitmapSize, typename WordType>
241inline void Bitmap<bitmapSize, WordType>::invert()
242{
243 for (size_t i = 0; i < words; ++i)
244 bits[i] = ~bits[i];
245 if constexpr (!!(bitmapSize % wordSize)) {
246 constexpr size_t remainingBits = bitmapSize % wordSize;
247 constexpr WordType mask = (static_cast<WordType>(1) << remainingBits) - 1;
248 bits[words - 1] &= mask;
249 }
250}

252template<size_t bitmapSize, typename WordType>
253inline size_t Bitmap<bitmapSize, WordType>::nextPossiblyUnset(size_t start) const
254{
255 if (!~bits[start / wordSize])
256 return ((start / wordSize) + 1) * wordSize;
257 return start + 1;
258}

260template<size_t bitmapSize, typename WordType>
261inline int64_t Bitmap<bitmapSize, WordType>::findRunOfZeros(size_t runLength) const
262{
263 if (!runLength)
264 runLength = 1;
265
266 if (runLength > bitmapSize)
267 return -1;
268
269 for (size_t i = 0; i <= (bitmapSize - runLength) ; i++) {
270 bool found = true;
271 for (size_t j = i; j <= (i + runLength - 1) ; j++) {
272 if (get(j)) {
273 found = false;
274 break;
275 }
276 }
277 if (found)
278 return i;
279 }
280 return -1;
281}

283template<size_t bitmapSize, typename WordType>
284inline size_t Bitmap<bitmapSize, WordType>::count(size_t start) const
285{
286 size_t result = 0;
287 for ( ; (start % wordSize); ++start) {
288 if (get(start))
289 ++result;
290 }
291 for (size_t i = start / wordSize; i < words; ++i)
292 result += WTF::bitCount(bits[i]);
293 return result;
294}

296template<size_t bitmapSize, typename WordType>
297inline bool Bitmap<bitmapSize, WordType>::isEmpty() const
298{
299 for (size_t i = 0; i < words; ++i)
300 if (bits[i])
301 return false;
302 return true;
303}

305template<size_t bitmapSize, typename WordType>
306inline bool Bitmap<bitmapSize, WordType>::isFull() const
307{
308 for (size_t i = 0; i < words; ++i)
309 if (~bits[i]) {
310 if constexpr (!!(bitmapSize % wordSize)) {
311 if (i == words - 1) {
312 constexpr size_t remainingBits = bitmapSize % wordSize;
313 constexpr WordType mask = (static_cast<WordType>(1) << remainingBits) - 1;
314 if ((bits[i] & mask) == mask)
315 return true;
316 }
317 }
318 return false;
319 }
320 return true;
321}

344template<size_t bitmapSize, typename WordType>
345inline void Bitmap<bitmapSize, WordType>::concurrentFilter(const Bitmap& other)
346{
347 for (size_t i = 0; i < words; ++i) {
348 for (;;) {
349 WordType otherBits = other.bits[i];
350 if (!otherBits) {
351 bits[i] = 0;
352 break;
353 }
354 WordType oldBits = bits[i];
355 WordType filteredBits = oldBits & otherBits;
356 if (oldBits == filteredBits)
357 break;
358 if (atomicCompareExchangeWeakRelaxed(&bits[i], oldBits, filteredBits))
359 break;
360 }
361 }
362}

364template<size_t bitmapSize, typename WordType>
365inline bool Bitmap<bitmapSize, WordType>::subsumes(const Bitmap& other) const
366{
367 for (size_t i = 0; i < words; ++i) {
368 WordType myBits = bits[i];
369 WordType otherBits = other.bits[i];
370 if ((myBits | otherBits) != myBits)
371 return false;
372 }
373 return true;
374}

376template<size_t bitmapSize, typename WordType>
377template<typename Func>
378inline void Bitmap<bitmapSize, WordType>::forEachSetBit(const Func& func) const
379{
380 for (size_t i = 0; i < words; ++i) {
381 WordType word = bits[i];
382 if (!word)
383 continue;
384 size_t base = i * wordSize;
385 for (size_t j = 0; j < wordSize; ++j) {
386 if (word & 1)
387 func(base + j);
388 word >>= 1;
389 }
390 }
391}

393template<size_t bitmapSize, typename WordType>
394inline size_t Bitmap<bitmapSize, WordType>::findBit(size_t startIndex, bool value) const
395{
396 WordType skipValue = -(static_cast<WordType>(value) ^ 1);
397 size_t wordIndex = startIndex / wordSize;
398 size_t startIndexInWord = startIndex - wordIndex * wordSize;
399
400 while (wordIndex < words) {
401 WordType word = bits[wordIndex];
402 if (word != skipValue) {
403 size_t index = startIndexInWord;
404 if (findBitInWord(word, index, wordSize, value))
405 return wordIndex * wordSize + index;
406 }
407
408 wordIndex++;
409 startIndexInWord = 0;
410 }
411
412 return bitmapSize;
413}

433template<size_t bitmapSize, typename WordType>
434inline void Bitmap<bitmapSize, WordType>::setEachNthBit(size_t n, size_t start, size_t end)
435{
436 ASSERT(start <= end);
437 ASSERT(end <= bitmapSize);
438
439 size_t wordIndex = start / wordSize;
440 size_t endWordIndex = end / wordSize;
441 size_t index = start - wordIndex * wordSize;
442 while (wordIndex < endWordIndex) {
443 while (index < wordSize) {
444 bits[wordIndex] |= (one << index);
445 index += n;
446 }
447 index -= wordSize;
448 wordIndex++;
449 }
450
451 size_t endIndex = end - endWordIndex * wordSize;
452 while (index < endIndex) {
453 bits[wordIndex] |= (one << index);
454 index += n;
455 }
456
457 if constexpr (!!(bitmapSize % wordSize)) {
458 constexpr size_t remainingBits = bitmapSize % wordSize;
459 constexpr WordType mask = (static_cast<WordType>(1) << remainingBits) - 1;
460 bits[words - 1] &= mask;
461 }
462}

464template<size_t bitmapSize, typename WordType>
465inline bool Bitmap<bitmapSize, WordType>::operator==(const Bitmap& other) const
466{
467 for (size_t i = 0; i < words; ++i) {
468 if (bits[i] != other.bits[i])
469 return false;
470 }
471 return true;

501template<size_t bitmapSize, typename WordType>
502inline unsigned Bitmap<bitmapSize, WordType>::hash() const
503{
504 unsigned result = 0;
505 for (size_t i = 0; i < words; ++i)
506 result ^= IntHash<WordType>::hash(bits[i]);
