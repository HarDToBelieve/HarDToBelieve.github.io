50// Because ARC enablement is a compile-time choice, and we compile this header
51// both ways, we need a separate copy of our code when ARC is enabled.
52#if __has_feature(objc_arc)
53#define BlockPtr BlockPtrArc
54#endif
55
56namespace WTF {
57
58extern "C" void* _NSConcreteMallocBlock[32];
59
60template<typename> class BlockPtr;
61
62template<typename R, typename... Args>
63class BlockPtr<R (Args...)> {
64public:
65 using BlockType = R (^)(Args...);
66
67 template<typename F>
68 static BlockPtr fromCallable(F function)
69 {
70 struct Descriptor {
71 uintptr_t reserved;
72 uintptr_t size;
73 void (*WTF_COPY_FUNCTION_POINTER_QUALIFIER copy)(void *dst, const void *src);
74 void (*WTF_DISPOSE_FUNCTION_POINTER_QUALIFIER dispose)(const void *);
75 };
76
77 struct Block {
78 void* WTF_ISA_POINTER_QUALIFIER isa;
79 int32_t flags;
80 int32_t reserved;
81 R (*WTF_INVOKE_FUNCTION_POINTER_QUALIFIER invoke)(void *, Args...);
82 const struct Descriptor* descriptor;
83 F f;
84 };
85
86 static const Descriptor descriptor {
87 0,
88 sizeof(Block),
89
90 // We keep the copy function null - the block is already on the heap
91 // so it should never be copied.
92 nullptr,
93
94 [](const void* ptr) {
95 static_cast<Block*>(const_cast<void*>(ptr))->f.~F();
96 }
97 };
98
99 Block* block = static_cast<Block*>(malloc(sizeof(Block)));
100 block->isa = _NSConcreteMallocBlock;
101
102 enum {
103 BLOCK_NEEDS_FREE = (1 << 24),
104 BLOCK_HAS_COPY_DISPOSE = (1 << 25),
105 };
106 const unsigned retainCount = 1;
107
108 block->flags = BLOCK_HAS_COPY_DISPOSE | BLOCK_NEEDS_FREE | (retainCount << 1);
109 block->reserved = 0;
110 block->invoke = [](void *ptr, Args... args) -> R {
111 return static_cast<Block*>(ptr)->f(std::forward<Args>(args)...);
112 };
113 block->descriptor = &descriptor;
114
115 new (&block->f) F { std::move(function) };
116
117#if __has_feature(objc_arc)
118 return BlockPtr { (__bridge_transfer BlockType)block };
119#else
120 BlockPtr blockPtr;
121 blockPtr.m_block = reinterpret_cast<BlockType>(block);
122 return blockPtr;
123#endif
124 }
125
126 BlockPtr()
127 : m_block(nullptr)
128 {
129 }
130
131 BlockPtr(BlockType block)
132#if __has_feature(objc_arc)
133 : m_block(WTFMove(block))
134#else
135 : m_block(Block_copy(block))
136#endif
137 {
138 }
139
140 BlockPtr(const BlockPtr& other)
141#if __has_feature(objc_arc)
142 : m_block(other.m_block)
143#else
144 : m_block(Block_copy(other.m_block))
145#endif
146 {
147 }
148
149 BlockPtr(BlockPtr&& other)
150 : m_block(std::exchange(other.m_block, nullptr))
151 {
152 }
153
154 ~BlockPtr()
155 {
156#if !__has_feature(objc_arc)
157 Block_release(m_block);
158#endif
159 }
160
161 BlockPtr& operator=(const BlockPtr& other)
162 {
163#if __has_feature(objc_arc)
164 m_block = other.m_block;
165#else
166 if (this != &other) {
167 Block_release(m_block);
168 m_block = Block_copy(other.m_block);
169 }
170#endif
171
172 return *this;
173 }
174
175 BlockPtr& operator=(BlockPtr&& other)
176 {
177 ASSERT(this != &other);
178
179#if !__has_feature(objc_arc)
180 Block_release(m_block);
181#endif
182 m_block = std::exchange(other.m_block, nullptr);
183
184 return *this;
185 }
186
187 BlockType get() const { return m_block; }
188
189 explicit operator bool() const { return m_block; }
190 bool operator!() const { return !m_block; }
191
192 R operator()(Args... arguments) const
193 {
194 ASSERT(m_block);
195
196 return m_block(std::forward<Args>(arguments)...);
197 }
198
199private:
200 BlockType m_block;
201};
202
203template<typename R, typename... Args>
204inline BlockPtr<R (Args...)> makeBlockPtr(R (^block)(Args...))
205{
206 return BlockPtr<R (Args...)>(block);
207}
208
209template<typename F, typename Class, typename R, typename... Args>
210inline auto makeBlockPtr(F&& function, R (Class::*)(Args...) const)
211{
212 return BlockPtr<R (Args...)>::fromCallable(std::forward<F>(function));
213}
214
215template<typename F, typename Class, typename R, typename... Args>
216inline auto makeBlockPtr(F&& function, R (Class::*)(Args...))
217{
218 return BlockPtr<R (Args...)>::fromCallable(std::forward<F>(function));
219}
220
221template<typename F>
222inline auto makeBlockPtr(F&& function)
223{
224 return makeBlockPtr(std::forward<F>(function), &F::operator());
225}
226
227}
