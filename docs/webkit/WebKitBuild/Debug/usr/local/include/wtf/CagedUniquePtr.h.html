26#pragma once
27
28#include <wtf/CagedPtr.h>
29
30namespace WTF {
31
32template<Gigacage::Kind kind, typename T, bool shouldTag = false>
33class CagedUniquePtr : public CagedPtr<kind, T, shouldTag> {
34 static_assert(std::is_trivially_destructible<T>::value, "We expect the contents of a caged pointer to be trivially destructable.");
35public:
36 using Base = CagedPtr<kind, T, shouldTag>;
37 CagedUniquePtr() = default;
38
39 CagedUniquePtr(T* ptr, size_t size)
40 : Base(ptr, size)
41 { }
42
43 CagedUniquePtr(CagedUniquePtr&& ptr)
44 : Base(std::forward<CagedUniquePtr&&>(ptr))
45 { }
46
47 CagedUniquePtr(const CagedUniquePtr&) = delete;
48
49 template<typename... Arguments>
50 static CagedUniquePtr create(size_t length, Arguments&&... arguments)
51 {
52 T* result = static_cast<T*>(Gigacage::malloc(kind, sizeof(T) * length));
53 while (length--)
54 new (result + length) T(arguments...);
55 return CagedUniquePtr(result, length);
56 }
57
58 template<typename... Arguments>
59 static CagedUniquePtr tryCreate(size_t length, Arguments&&... arguments)
60 {
61 T* result = static_cast<T*>(Gigacage::tryMalloc(kind, sizeof(T) * length));
62 if (!result)
63 return { };
64 while (length--)
65 new (result + length) T(arguments...);
66 return CagedUniquePtr(result, length);
67 }
68
69 CagedUniquePtr& operator=(CagedUniquePtr&& ptr)
70 {
71 destroy();
72 this->m_ptr = ptr.m_ptr;
73 ptr.m_ptr = nullptr;
74 return *this;
75 }
76
77 CagedUniquePtr& operator=(const CagedUniquePtr&) = delete;
78
79 ~CagedUniquePtr()
80 {
81 destroy();
82 }
83
84private:
85 void destroy()
86 {
87 T* ptr = Base::getUnsafe();
88 if (!ptr)
89 return;
90 ptr->~T();
91 Gigacage::free(kind, ptr);
92 }
93};
94
95} // namespace WTF
96
97using WTF::CagedUniquePtr;
