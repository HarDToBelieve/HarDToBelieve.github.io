26#pragma once
27
28#include <wtf/Function.h>
29#include <wtf/UniqueRef.h>
30#include <wtf/WeakPtr.h>
31
32namespace WTF {
33
34class CancellableTask;
35
36class TaskCancellationGroup : public CanMakeWeakPtr<TaskCancellationGroup> {
37public:
38 TaskCancellationGroup() : m_impl(makeUniqueRef<Impl>()) { }
39 void cancel() { m_impl->cancel(); }
40 bool hasPendingTask() const { return m_impl->hasPendingTask(); }
41
42private:
43 friend class CancellableTask;
44 class Impl : public CanMakeWeakPtr<Impl> {
45 WTF_MAKE_FAST_ALLOCATED;
46 public:
47 void cancel() { weakPtrFactory().revokeAll(); }
48 bool hasPendingTask() const { return weakPtrFactory().weakPtrCount(); }
49 };
50
51 class Handle {
52 public:
53 bool isCancelled() const { return !m_impl; }
54 void clear() { m_impl = nullptr; }
55 private:
56 friend class TaskCancellationGroup;
57 explicit Handle(Impl& impl) : m_impl(makeWeakPtr(impl)) { }
58 WeakPtr<Impl> m_impl;
59 };
60 Handle createHandle() { return Handle { m_impl }; }
61
62 UniqueRef<Impl> m_impl;
63};
64
65class CancellableTask {
66public:
67 CancellableTask(TaskCancellationGroup&, Function<void()>&&);
68 void operator()();
69
70private:
71 TaskCancellationGroup::Handle m_cancellationGroup;
72 Function<void()> m_task;
73};
74
75inline CancellableTask::CancellableTask(TaskCancellationGroup& cancellationGroup, Function<void()>&& task)
76 : m_cancellationGroup(cancellationGroup.createHandle())
77 , m_task(WTFMove(task))
78{ }
79
80inline void CancellableTask::operator()()
81{
82 if (m_cancellationGroup.isCancelled())
83 return;
84
85 m_cancellationGroup.clear();
86 m_task();
87}
88
89} // namespace WTF
