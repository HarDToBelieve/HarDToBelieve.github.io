31namespace WTF {

33template<typename T, typename PtrTraits = RawPtrTraits<T>>
34class CheckedPtr {
35 WTF_MAKE_FAST_ALLOCATED;
36public:
37
38 CheckedPtr()
39 : m_ptr(nullptr)
40 { }
41
42 CheckedPtr(std::nullptr_t)
43 : m_ptr(nullptr)
44 { }
45
46 ALWAYS_INLINE CheckedPtr(T* ptr)
47 : m_ptr { ptr }
48 {
49 refIfNotNull();
50 }
51
52 ALWAYS_INLINE CheckedPtr(const CheckedPtr& other)
53 : m_ptr { other.m_ptr }
54 {
55 refIfNotNull();
56 }
57
58 ALWAYS_INLINE CheckedPtr(CheckedPtr&& other)
59 : m_ptr { PtrTraits::exchange(other.m_ptr, nullptr) }
60 { }
61
62 ALWAYS_INLINE ~CheckedPtr()
63 {
64 derefIfNotNull();
65 }
66
67 template<typename OtherType, typename OtherPtrTraits> CheckedPtr(const CheckedPtr<OtherType, OtherPtrTraits>& other)
68 : m_ptr { other.m_ptr }
69 {
70 refIfNotNull();
71 }
72
73 template<typename OtherType, typename OtherPtrTraits> CheckedPtr(CheckedPtr<OtherType, OtherPtrTraits>&& other)
74 : m_ptr { PtrTraits::exchange(other.m_ptr, nullptr) }
75 { }
76
77 CheckedPtr(HashTableDeletedValueType)
78 : m_ptr(PtrTraits::hashTableDeletedValue())
79 { }
80
81 bool isHashTableDeletedValue() const { return PtrTraits::isHashTableDeletedValue(m_ptr); }
82
83 // This conversion operator allows implicit conversion to bool but not to other integer types.
84 using UnspecifiedBoolType = void (CheckedPtr::*)() const;
85 operator UnspecifiedBoolType() const { return m_ptr ? &CheckedPtr::unspecifiedBoolTypeInstance : nullptr; }
86
87 ALWAYS_INLINE bool operator!() const { return !PtrTraits::unwrap(m_ptr); }
88
89 ALWAYS_INLINE const T* get() const { return PtrTraits::unwrap(m_ptr); }
90 ALWAYS_INLINE T* get() { return PtrTraits::unwrap(m_ptr); }
91 ALWAYS_INLINE const T& operator*() const { ASSERT(this); return *get(); }
92 ALWAYS_INLINE T& operator*() { ASSERT(this); return *get(); }
93 ALWAYS_INLINE const T* operator->() const { return get(); }
94 ALWAYS_INLINE T* operator->() { return get(); }
95
96 bool operator==(const T* other) const { return m_ptr == other; }
97 template<typename U> bool operator==(U* other) const { return m_ptr == other; }
98
99 bool operator==(const CheckedPtr& other) const { return m_ptr == other.m_ptr; }
100
101 template<typename OtherType, typename OtherPtrTraits>
102 bool operator==(const CheckedPtr<OtherType, OtherPtrTraits>& other) const { return m_ptr == other.m_ptr; }
103
104 CheckedPtr& operator=(std::nullptr_t)
105 {
106 derefIfNotNull();
107 m_ptr = nullptr;
108 return *this;
109 }
110
111 CheckedPtr& operator=(T* ptr)
112 {
113 CheckedPtr copy { ptr };
114 PtrTraits::swap(m_ptr, copy.m_ptr);
115 return *this;
116 }
117
118 CheckedPtr& operator=(const CheckedPtr& other)
119 {
120 CheckedPtr copy { other };
121 PtrTraits::swap(m_ptr, copy.m_ptr);
122 return *this;
123 }
124
125 template<typename OtherType, typename OtherPtrTraits> CheckedPtr& operator=(const CheckedPtr<OtherType, OtherPtrTraits>& other)
126 {
127 CheckedPtr copy { other };
128 PtrTraits::swap(m_ptr, copy.m_ptr);
129 return *this;
130 }
131
132 CheckedPtr& operator=(CheckedPtr&& other)
133 {
134 CheckedPtr moved { WTFMove(other) };
135 PtrTraits::swap(m_ptr, moved.m_ptr);
136 return *this;
137 }
138
139 template<typename OtherType, typename OtherPtrTraits> CheckedPtr& operator=(CheckedPtr<OtherType, OtherPtrTraits>&& other)
140 {
141 CheckedPtr moved { WTFMove(other) };
142 PtrTraits::swap(m_ptr, moved.m_ptr);
143 return *this;
144 }
145
146private:
147 template<typename OtherType, typename OtherPtrTraits> friend class CheckedPtr;
148
149 void unspecifiedBoolTypeInstance() const { }
150
151 ALWAYS_INLINE void refIfNotNull()
152 {
153 if (T* ptr = PtrTraits::unwrap(m_ptr); LIKELY(ptr))
154 ptr->incrementPtrCount();
155 }
156
157 ALWAYS_INLINE void derefIfNotNull()
158 {
159 if (T* ptr = PtrTraits::unwrap(m_ptr); LIKELY(ptr))
160 ptr->decrementPtrCount();
161 }
162
163 typename PtrTraits::StorageType m_ptr;
164};

166template<typename T, typename PtrTraits = RawPtrTraits<T>> inline CheckedPtr<T, PtrTraits> makeCheckedPtr(T* pointer)
167{
168 return pointer;
169}
170
171template<typename T, typename PtrTraits = RawPtrTraits<T>> inline CheckedPtr<T, PtrTraits> makeCheckedPtr(T& reference)
172{
173 return &reference;
174}

176template <typename T, typename PtrTraits>
177struct GetPtrHelper<CheckedPtr<T, PtrTraits>> {
178 typedef T* PtrType;
179 static T* getPtr(const CheckedPtr<T, PtrTraits>& p) { return const_cast<T*>(p.get()); }
180};

182template <typename T, typename U>
183struct IsSmartPtr<CheckedPtr<T, U>> {
184 static constexpr bool value = true;
185};

187template<typename ExpectedType, typename ArgType, typename ArgPtrTraits>
188inline bool is(CheckedPtr<ArgType, ArgPtrTraits>& source)
189{
190 return is<ExpectedType>(source.get());
191}
192
193template<typename ExpectedType, typename ArgType, typename ArgPtrTraits>
194inline bool is(const CheckedPtr<ArgType, ArgPtrTraits>& source)
195{
196 return is<ExpectedType>(source.get());
197}

199class CanMakeCheckedPtr {
200public:
201 ~CanMakeCheckedPtr() { RELEASE_ASSERT(!m_count); }
202
203 uint16_t ptrCount() const { return m_count; }
204 void incrementPtrCount() { ++m_count; }
205 void decrementPtrCount() { ASSERT(m_count); --m_count; }
206private:
207 uint16_t m_count { 0 };
208};

210template<typename P> struct HashTraits<CheckedPtr<P>> : SimpleClassHashTraits<CheckedPtr<P>> {
211 static P* emptyValue() { return nullptr; }
212
213 typedef P* PeekType;
214 static PeekType peek(const CheckedPtr<P>& value) { return value.get(); }
215 static PeekType peek(P* value) { return value; }
216
217 static void customDeleteBucket(CheckedPtr<P>& value)
218 {
219 // See unique_ptr's customDeleteBucket() for an explanation.
220 ASSERT(!SimpleClassHashTraits<CheckedPtr<P>>::isDeletedValue(value));
221 auto valueToBeDestroyed = WTFMove(value);
222 SimpleClassHashTraits<CheckedPtr<P>>::constructDeletedValue(value);
223 }
224};

226template<typename P> struct DefaultHash<CheckedPtr<P>> : PtrHash<CheckedPtr<P>> { };
