38namespace WTF {
39
40// The goal of this class is folding a pointer and 2 bytes value into 8 bytes in both 32bit and 64bit architectures.
41// 32bit architecture just has a pair of byte and pointer, which should be 8 bytes.
42// We are assuming 48bit pointers here, which is also assumed in JSValue anyway.
43template<typename PointerType, typename Type>
44class CompactPointerTuple final {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 static_assert(sizeof(Type) <= 2, "");
48 static_assert(std::is_pointer<PointerType>::value, "");
49 static_assert(std::is_integral<Type>::value || std::is_enum<Type>::value, "");
50 using UnsignedType = std::make_unsigned_t<std::conditional_t<std::is_same_v<Type, bool>, uint8_t, Type>>;
51 static_assert(sizeof(UnsignedType) == sizeof(Type));
52
53 CompactPointerTuple() = default;
54
55#if CPU(ADDRESS64)
56public:
57 static constexpr unsigned maxNumberOfBitsInPointer = 48;
58 static_assert(OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH) <= maxNumberOfBitsInPointer);
59
60#if CPU(LITTLE_ENDIAN)
61 static ptrdiff_t offsetOfType()
62 {
63 return maxNumberOfBitsInPointer / 8;
64 }
65#endif
66
67 static constexpr uint64_t pointerMask = (1ULL << maxNumberOfBitsInPointer) - 1;
68
69 CompactPointerTuple(PointerType pointer, Type type)
70 : m_data(encode(pointer, type))
71 {
72 ASSERT(this->type() == type);
73 ASSERT(this->pointer() == pointer);
74 }
75
76 template<typename OtherPointerType, typename = std::enable_if<std::is_pointer<PointerType>::value && std::is_convertible<OtherPointerType, PointerType>::value>>
77 CompactPointerTuple(CompactPointerTuple<OtherPointerType, Type>&& other)
78 : m_data { std::exchange(other.m_data, { }) }
79 {
80 }
81
82 PointerType pointer() const { return bitwise_cast<PointerType>(m_data & pointerMask); }
83 void setPointer(PointerType pointer)
84 {
85 m_data = encode(pointer, type());
86 ASSERT(this->pointer() == pointer);
87 }
88
89 Type type() const { return decodeType(m_data); }
90 void setType(Type type)
91 {
92 m_data = encode(pointer(), type);
93 ASSERT(this->type() == type);
94 }
95
96 uint64_t data() const { return m_data; }
97
98private:
99 static constexpr uint64_t encodeType(Type type)
100 {
101 return static_cast<uint64_t>(static_cast<UnsignedType>(type)) << maxNumberOfBitsInPointer;
102 }
103 static constexpr Type decodeType(uint64_t value)
104 {
105 return static_cast<Type>(static_cast<UnsignedType>(value >> maxNumberOfBitsInPointer));
106 }
107
108 static uint64_t encode(PointerType pointer, Type type)
109 {
110 return bitwise_cast<uint64_t>(pointer) | encodeType(type);
111 }
112
113 uint64_t m_data { 0 };
114#else
115public:
116 CompactPointerTuple(PointerType pointer, Type type)
117 : m_pointer(pointer)
118 , m_type(type)
119 {
120 }
121
122 template<typename OtherPointerType, typename = std::enable_if<std::is_pointer<PointerType>::value && std::is_convertible<OtherPointerType, PointerType>::value>>
123 CompactPointerTuple(CompactPointerTuple<OtherPointerType, Type>&& other)
124 : m_pointer { std::exchange(other.m_pointer, { }) }
125 , m_type { std::exchange(other.m_type, { }) }
126 {
127 }
128
129 PointerType pointer() const { return m_pointer; }
130 void setPointer(PointerType pointer) { m_pointer = pointer; }
131 Type type() const { return m_type; }
132 void setType(Type type) { m_type = type; }
133
134private:
135 PointerType m_pointer { nullptr };
136 Type m_type { 0 };
137#endif
138
139 template<typename, typename> friend class CompactPointerTuple;
140};
141
142} // namespace WTF
