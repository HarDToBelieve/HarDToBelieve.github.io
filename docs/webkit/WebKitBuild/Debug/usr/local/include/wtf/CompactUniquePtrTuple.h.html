26#pragma once

28#include <memory>
29#include <wtf/CompactPointerTuple.h>
30#include <wtf/Noncopyable.h>
31
32namespace WTF {
33
34template<typename T, typename Type, typename Deleter = std::default_delete<T>> class CompactUniquePtrTuple;
35
36template<typename T, typename Type, typename... Args>
37ALWAYS_INLINE CompactUniquePtrTuple<T, Type> makeCompactUniquePtr(Args&&... args)
38{
39 return CompactUniquePtrTuple<T, Type>(makeUnique<T>(std::forward<Args>(args)...));
40}
41
42template<typename T, typename Type, typename Deleter, typename... Args>
43ALWAYS_INLINE CompactUniquePtrTuple<T, Type, Deleter> makeCompactUniquePtr(Args&&... args)
44{
45 return CompactUniquePtrTuple<T, Type, Deleter>(makeUnique<T>(std::forward<Args>(args)...));
46}
47
48template<typename T, typename Type, typename Deleter>
49class CompactUniquePtrTuple final {
50 WTF_MAKE_FAST_ALLOCATED;
51 WTF_MAKE_NONCOPYABLE(CompactUniquePtrTuple);
52public:
53 CompactUniquePtrTuple() = default;
54
55 template <typename U, typename UDeleter, typename = std::enable_if_t<std::is_same<UDeleter, Deleter>::value || std::is_same<UDeleter, std::default_delete<U>>::value>>
56 CompactUniquePtrTuple(CompactUniquePtrTuple<U, Type, UDeleter>&& other)
57 : m_data { std::exchange(other.m_data, { }) }
58 {
59 }
60
61 ~CompactUniquePtrTuple()
62 {
63 setPointer(nullptr);
64 }
65
66 template <typename U, typename UDeleter, typename = std::enable_if_t<std::is_same<UDeleter, Deleter>::value || std::is_same<UDeleter, std::default_delete<U>>::value>>
67 CompactUniquePtrTuple<T, Type, Deleter>& operator=(CompactUniquePtrTuple<U, Type, UDeleter>&& other)
68 {
69 CompactUniquePtrTuple moved { WTFMove(other) };
70 std::swap(m_data, moved.m_data);
71 return *this;
72 }
73
74 T* pointer() const { return m_data.pointer(); }
75
76 std::unique_ptr<T, Deleter> moveToUniquePtr()
77 {
78 T* pointer = m_data.pointer();
79 m_data.setPointer(nullptr);
80 return std::unique_ptr<T, Deleter>(pointer);
81 }
82
83 void setPointer(std::nullptr_t)
84 {
85 deletePointer();
86 m_data.setPointer(nullptr);
87 }
88
89 template <typename U, typename UDeleter, typename = std::enable_if_t<std::is_same<UDeleter, Deleter>::value || std::is_same<UDeleter, std::default_delete<U>>::value>>
90 void setPointer(std::unique_ptr<U, UDeleter>&& pointer)
91 {
92 deletePointer();
93 m_data.setPointer(pointer.release());
94 }
95
96 Type type() const { return m_data.type(); }
97
98 void setType(Type type)
99 {
100 m_data.setType(type);
101 }
102
103private:
104 CompactUniquePtrTuple(std::unique_ptr<T>&& pointer)
105 {
106 m_data.setPointer(pointer.release());
107 }
108
109 void deletePointer()
110 {
111 if (T* pointer = m_data.pointer())
112 Deleter()(pointer);
113 }
114
115 template<typename U, typename E, typename... Args> friend CompactUniquePtrTuple<U, E> makeCompactUniquePtr(Args&&... args);
116 template<typename U, typename E, typename D, typename... Args> friend CompactUniquePtrTuple<U, E, D> makeCompactUniquePtr(Args&&... args);
117
118 template <typename, typename, typename> friend class CompactUniquePtrTuple;
119
120 CompactPointerTuple<T*, Type> m_data;
121};
122
123} // namespace WTF
