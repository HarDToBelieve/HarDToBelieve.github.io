31namespace WTF {

33template<typename> class CompletionHandler;
34enum class CompletionHandlerCallThread { MainThread, ConstructionThread };
35
36// Wraps a Function to make sure it is always called once and only once.
37template <typename Out, typename... In>
38class CompletionHandler<Out(In...)> {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 CompletionHandler() = default;
42
43 template<typename CallableType, class = typename std::enable_if<std::is_rvalue_reference<CallableType&&>::value>::type>
44 CompletionHandler(CallableType&& callable, CompletionHandlerCallThread callThread = CompletionHandlerCallThread::ConstructionThread)
45 : m_function(WTFMove(callable))
46#if ASSERT_ENABLED
47 , m_shouldBeCalledOnMainThread(callThread == CompletionHandlerCallThread::MainThread || isMainThread())
48#endif
49 {
50 UNUSED_PARAM(callThread);
51 }
52
53 CompletionHandler(CompletionHandler&&) = default;
54 CompletionHandler& operator=(CompletionHandler&&) = default;
55
56 ~CompletionHandler()
57 {
58 ASSERT_WITH_MESSAGE(!m_function, "Completion handler should always be called");
59 }
60
61 explicit operator bool() const { return !!m_function; }
62
63 Out operator()(In... in)
64 {
65 ASSERT(m_shouldBeCalledOnMainThread == isMainThread());
66 ASSERT_WITH_MESSAGE(m_function, "Completion handler should not be called more than once");
67 return std::exchange(m_function, nullptr)(std::forward<In>(in)...);
68 }
69
70private:
71 Function<Out(In...)> m_function;
72#if ASSERT_ENABLED
73 bool m_shouldBeCalledOnMainThread;
74#endif
75};
76
77// Wraps a Function to make sure it is called at most once.
78// If the CompletionHandlerWithFinalizer is destroyed and the function hasn't yet been called,
79// the finalizer is invoked with the function as its argument.
80template<typename> class CompletionHandlerWithFinalizer;
81template <typename Out, typename... In>
82class CompletionHandlerWithFinalizer<Out(In...)> {
83 WTF_MAKE_FAST_ALLOCATED;
84public:
85 template<typename CallableType, class = typename std::enable_if<std::is_rvalue_reference<CallableType&&>::value>::type>
86 CompletionHandlerWithFinalizer(CallableType&& callable, Function<void(Function<Out(In...)>&)>&& finalizer)
87 : m_function(WTFMove(callable))
88 , m_finalizer(WTFMove(finalizer))
89 {
90 }
91
92 CompletionHandlerWithFinalizer(CompletionHandlerWithFinalizer&&) = default;
93 CompletionHandlerWithFinalizer& operator=(CompletionHandlerWithFinalizer&&) = default;
94
95 ~CompletionHandlerWithFinalizer()
96 {
97 if (!m_function)
98 return;
99
100 m_finalizer(m_function);
101 }
102
103 explicit operator bool() const { return !!m_function; }
104
105 Out operator()(In... in)
106 {
107 ASSERT(m_wasConstructedOnMainThread == isMainThread());
108 ASSERT_WITH_MESSAGE(m_function, "Completion handler should not be called more than once");
109 return std::exchange(m_function, nullptr)(std::forward<In>(in)...);
110 }
111
112private:
113 Function<Out(In...)> m_function;
114 Function<void(Function<Out(In...)>&)> m_finalizer;
115#if ASSERT_ENABLED
116 bool m_wasConstructedOnMainThread { isMainThread() };
117#endif
118};

120namespace Detail {
121
122template<typename Out, typename... In>
123class CallableWrapper<CompletionHandler<Out(In...)>, Out, In...> : public CallableWrapperBase<Out, In...> {
124 WTF_MAKE_FAST_ALLOCATED;
125public:
126 explicit CallableWrapper(CompletionHandler<Out(In...)>&& completionHandler)
127 : m_completionHandler(WTFMove(completionHandler))
128 {
129 RELEASE_ASSERT(m_completionHandler);
130 }
131 Out call(In... in) final { return m_completionHandler(std::forward<In>(in)...); }
132private:
133 CompletionHandler<Out(In...)> m_completionHandler;
134};

138class CompletionHandlerCallingScope final {
139 WTF_MAKE_FAST_ALLOCATED;
140public:
141 CompletionHandlerCallingScope() = default;
142
143 CompletionHandlerCallingScope(CompletionHandler<void()>&& completionHandler)
144 : m_completionHandler(WTFMove(completionHandler))
145 { }
146
147 ~CompletionHandlerCallingScope()
148 {
149 if (m_completionHandler)
150 m_completionHandler();
151 }
152
153 CompletionHandlerCallingScope(CompletionHandlerCallingScope&&) = default;
154 CompletionHandlerCallingScope& operator=(CompletionHandlerCallingScope&&) = default;
155
156 CompletionHandler<void()> release() { return WTFMove(m_completionHandler); }
157
158private:
159 CompletionHandler<void()> m_completionHandler;
160};
