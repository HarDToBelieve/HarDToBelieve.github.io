35namespace WTF {
36
37DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(ConcurrentBuffer);
38
39// ConcurrentBuffer is suitable for when you plan to store immutable data and sometimes append to it.
40// It supports storing data that is not copy-constructable but bit-copyable.
41template<typename T>
42class ConcurrentBuffer final {
43 WTF_MAKE_NONCOPYABLE(ConcurrentBuffer);
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46
47 ConcurrentBuffer()
48 {
49 }
50
51 ~ConcurrentBuffer()
52 {
53 if (Array* array = m_array) {
54 for (size_t i = 0; i < array->size; ++i)
55 array->data[i].~T();
56 }
57 for (Array* array : m_allArrays)
58 ConcurrentBufferMalloc::free(array);
59 }
60
61 // Growing is not concurrent. This assumes you are holding some other lock before you do this.
62 void growExact(size_t newSize)
63 {
64 Array* array = m_array;
65 if (array && newSize <= array->size)
66 return;
67 Array* newArray = createArray(newSize);
68 // This allows us to do ConcurrentBuffer<std::unique_ptr<>>.
69 // static_cast<void*> avoids triggering -Wclass-memaccess.
70 if (array)
71 memcpy(static_cast<void*>(newArray->data), array->data, sizeof(T) * array->size);
72 for (size_t i = array ? array->size : 0; i < newSize; ++i)
73 new (newArray->data + i) T();
74 WTF::storeStoreFence();
75 m_array = newArray;
76 WTF::storeStoreFence();
77 m_allArrays.append(newArray);
78 }
79
80 void grow(size_t newSize)
81 {
82 size_t size = m_array ? m_array->size : 0;
83 if (newSize <= size)
84 return;
85 growExact(std::max(newSize, size * 2));
86 }
87
88 struct Array {
89 size_t size; // This is an immutable size.
90 T data[1];
91 };
92
93 Array* array() const { return m_array; }
94
95 T& operator[](size_t index) { return m_array->data[index]; }
96 const T& operator[](size_t index) const { return m_array->data[index]; }
97
98private:
99 Array* createArray(size_t size)
100 {
101 Checked<size_t> objectSize = sizeof(T);
102 objectSize *= size;
103 objectSize += static_cast<size_t>(OBJECT_OFFSETOF(Array, data));
104 Array* result = static_cast<Array*>(ConcurrentBufferMalloc::malloc(objectSize));
105 result->size = size;
106 return result;
107 }
108
109 Array* m_array { nullptr };
110 Vector<Array*> m_allArrays;
111};
