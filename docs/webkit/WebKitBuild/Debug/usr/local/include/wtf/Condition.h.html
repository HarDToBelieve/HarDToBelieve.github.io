33namespace WTF {

35// This is a condition variable that is suitable for use with any lock-like object, including
36// our own WTF::Lock. It features standard wait()/notifyOne()/notifyAll() methods in addition to
37// a variety of wait-with-timeout methods. This includes methods that use WTF's own notion of
38// time, like wall-clock time (i.e. WallTime) and monotonic time (i.e. MonotonicTime). This is
39// a very efficient condition variable. It only requires one byte of memory. notifyOne() and
40// notifyAll() require just a load and branch for the fast case where no thread is waiting.
41// This condition variable, when used with WTF::Lock, can outperform a system condition variable
42// and lock by up to 58x.
43class Condition final {
44 WTF_MAKE_NONCOPYABLE(Condition);
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 // Condition will accept any kind of time and convert it internally, but this typedef tells
48 // you what kind of time Condition would be able to use without conversions. However, if you
49 // are unlikely to be affected by the cost of conversions, it is better to use MonotonicTime.
50 using Time = ParkingLot::Time;
51
52 constexpr Condition() = default;
53
54 // Wait on a parking queue while releasing the given lock. It will unlock the lock just before
55 // parking, and relock it upon wakeup. Returns true if we woke up due to some call to
56 // notifyOne() or notifyAll(). Returns false if we woke up due to a timeout. Note that this form
57 // of waitUntil() has some quirks:
58 //
59 // No spurious wake-up: in order for this to return before the timeout, some notifyOne() or
60 // notifyAll() call must have happened. No scenario other than timeout or notify can lead to this
61 // method returning. This means, for example, that you can't use pthread cancelation or signals to
62 // cause early return.
63 //
64 // Past timeout: it's possible for waitUntil() to be called with a timeout in the past. In that
65 // case, waitUntil() will still release the lock and reacquire it. waitUntil() will always return
66 // false in that case. This is subtly different from some pthread_cond_timedwait() implementations,
67 // which may not release the lock for past timeout. But, this behavior is consistent with OpenGroup
68 // documentation for timedwait().
69 template<typename LockType>
70 bool waitUntil(LockType& lock, const TimeWithDynamicClockType& timeout)
71 {
72 return waitUntilUnchecked(lock, timeout);
73 }
74
75 bool waitUntil(Lock& lock, const TimeWithDynamicClockType& timeout) WTF_REQUIRES_LOCK(lock)
76 {
77 return waitUntilUnchecked(lock, timeout);
78 }
79
80 // Wait until the given predicate is satisfied. Returns true if it is satisfied in the end.
81 // May return early due to timeout.
82 template<typename LockType, typename Functor>
83 bool waitUntil(LockType& lock, const TimeWithDynamicClockType& timeout, const Functor& predicate)
84 {
85 return waitUntilUnchecked(lock, timeout, predicate);
86 }
87
88 template<typename Functor>
89 bool waitUntil(Lock& lock, const TimeWithDynamicClockType& timeout, const Functor& predicate) WTF_REQUIRES_LOCK(lock)
90 {
91 return waitUntilUnchecked(lock, timeout, predicate);
92 }
93
94 // Wait until the given predicate is satisfied. Returns true if it is satisfied in the end.
95 // May return early due to timeout.
96 template<typename LockType, typename Functor>
97 bool waitFor(LockType& lock, Seconds relativeTimeout, const Functor& predicate)
98 {
99 return waitUntil(lock, MonotonicTime::now() + relativeTimeout, predicate);
100 }

102 template<typename Functor>
103 bool waitFor(Lock& lock, Seconds relativeTimeout, const Functor& predicate) WTF_REQUIRES_LOCK(lock)
104 {
105 return waitUntil(lock, MonotonicTime::now() + relativeTimeout, predicate);
106 }

119 template<typename LockType>
120 void wait(LockType& lock)
121 {
122 waitUntil(lock, Time::infinity());
123 }
124
125 void wait(Lock& lock) WTF_REQUIRES_LOCK(lock)

128 }

130 template<typename LockType, typename Functor>
131 void wait(LockType& lock, const Functor& predicate)
132 {
133 while (!predicate())
134 wait(lock);
135 }

138 void wait(Lock& lock, const Functor& predicate) WTF_REQUIRES_LOCK(lock)
139 {
140 while (!predicate())
141 wait(lock);
142 }
143
144 // Note that this method is extremely fast when nobody is waiting. It is not necessary to try to
145 // avoid calling this method. This returns true if someone was actually woken up.
146 bool notifyOne()
147 {
148 if (!m_hasWaiters.load()) {
149 // At this exact instant, there is nobody waiting on this condition. The way to visualize
150 // this is that if unparkOne() ran to completion without obstructions at this moment, it
151 // wouldn't wake anyone up. Hence, we have nothing to do!
152 return false;
153 }
154
155 bool didNotifyThread = false;
156 ParkingLot::unparkOne(
157 &m_hasWaiters,
158 [&] (ParkingLot::UnparkResult result) -> intptr_t {
159 if (!result.mayHaveMoreThreads)
160 m_hasWaiters.store(false);
161 didNotifyThread = result.didUnparkThread;
162 return 0;
163 });
164 return didNotifyThread;
165 }
166
167 void notifyAll()
168 {
169 if (!m_hasWaiters.load()) {
170 // See above.
171 return;
172 }
173
174 // It's totally safe for us to set this to false without any locking, because this thread is
175 // guaranteed to then unparkAll() anyway. So, if there is a race with some thread calling
176 // wait() just before this store happens, that thread is guaranteed to be awoken by the call to
177 // unparkAll(), below.
178 m_hasWaiters.store(false);
179
180 ParkingLot::unparkAll(&m_hasWaiters);
181 }
182
183private:
184 template<typename LockType>
185 bool waitUntilUnchecked(LockType& lock, const TimeWithDynamicClockType& timeout) WTF_IGNORES_THREAD_SAFETY_ANALYSIS
186 {
187 bool result;
188 if (timeout < timeout.nowWithSameClock()) {
189 lock.unlock();
190 result = false;
191 } else {
192 result = ParkingLot::parkConditionally(
193 &m_hasWaiters,
194 [this] () -> bool {
195 // Let everyone know that we will be waiting. Do this while we hold the queue lock,
196 // to prevent races with notifyOne().
197 m_hasWaiters.store(true);
198 return true;
199 },
200 [&lock] () WTF_IGNORES_THREAD_SAFETY_ANALYSIS {
201 lock.unlock();
202 },
203 timeout).wasUnparked;
204 }
205 lock.lock();
206 return result;
207 }
208
209 template<typename LockType, typename Functor>
210 bool waitUntilUnchecked(LockType& lock, const TimeWithDynamicClockType& timeout, const Functor& predicate) WTF_IGNORES_THREAD_SAFETY_ANALYSIS
211 {
212 while (!predicate()) {
213 if (!waitUntil(lock, timeout))
214 return predicate();
215 }
216 return true;
217 }
218
219 Atomic<bool> m_hasWaiters { false };
220};
