34namespace WTF {
35
36class CrossThreadTask {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 CrossThreadTask() = default;
40
41 CrossThreadTask(Function<void ()>&& taskFunction)
42 : m_taskFunction(WTFMove(taskFunction))
43 {
44 ASSERT(m_taskFunction);
45 }
46
47 void performTask()
48 {
49 m_taskFunction();
50 }
51
52 explicit operator bool() const { return !!m_taskFunction; }
53
54protected:
55 Function<void ()> m_taskFunction;
56};
57
58template <typename F, typename ArgsTuple, size_t... ArgsIndex>
59void callFunctionForCrossThreadTaskImpl(F function, ArgsTuple&& args, std::index_sequence<ArgsIndex...>)
60{
61 function(std::get<ArgsIndex>(std::forward<ArgsTuple>(args))...);
62}
63
64template <typename F, typename ArgsTuple, typename ArgsIndices = std::make_index_sequence<std::tuple_size<ArgsTuple>::value>>
65void callFunctionForCrossThreadTask(F function, ArgsTuple&& args)
66{
67 callFunctionForCrossThreadTaskImpl(function, std::forward<ArgsTuple>(args), ArgsIndices());
68}
69
70template<typename... Parameters, typename... Arguments>
71CrossThreadTask createCrossThreadTask(void (*method)(Parameters...), const Arguments&... arguments)
72{
73 return CrossThreadTask([method, arguments = std::make_tuple(crossThreadCopy(arguments)...)]() mutable {
74 callFunctionForCrossThreadTask(method, WTFMove(arguments));
75 });
76}
77
78template <typename C, typename MF, typename ArgsTuple, size_t... ArgsIndex>
79void callMemberFunctionForCrossThreadTaskImpl(C* object, MF function, ArgsTuple&& args, std::index_sequence<ArgsIndex...>)
80{
81 (object->*function)(std::get<ArgsIndex>(std::forward<ArgsTuple>(args))...);
82}
83
84template <typename C, typename MF, typename ArgsTuple, typename ArgsIndicies = std::make_index_sequence<std::tuple_size<ArgsTuple>::value>>
85void callMemberFunctionForCrossThreadTask(C* object, MF function, ArgsTuple&& args)
86{
87 callMemberFunctionForCrossThreadTaskImpl(object, function, std::forward<ArgsTuple>(args), ArgsIndicies());
88}
89
90template<typename T, typename std::enable_if<std::is_base_of<ThreadSafeRefCounted<T>, T>::value, int>::type = 0, typename... Parameters, typename... Arguments>
91CrossThreadTask createCrossThreadTask(T& callee, void (T::*method)(Parameters...), const Arguments&... arguments)
92{
93 return CrossThreadTask([callee = makeRefPtr(&callee), method, arguments = std::make_tuple(crossThreadCopy(arguments)...)]() mutable {
94 callMemberFunctionForCrossThreadTask(callee.get(), method, WTFMove(arguments));
95 });
96}
97
98template<typename T, typename std::enable_if<!std::is_base_of<ThreadSafeRefCounted<T>, T>::value, int>::type = 0, typename... Parameters, typename... Arguments>
99CrossThreadTask createCrossThreadTask(T& callee, void (T::*method)(Parameters...), const Arguments&... arguments)
100{
101 return CrossThreadTask([callee = &callee, method, arguments = std::make_tuple(crossThreadCopy(arguments)...)]() mutable {
102 callMemberFunctionForCrossThreadTask(callee, method, WTFMove(arguments));
103 });
104}
