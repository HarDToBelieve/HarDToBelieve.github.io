28#include <stdarg.h>
29#include <stdio.h>
30#include <wtf/PrintStream.h>
31#include <wtf/StdLibExtras.h>

33namespace WTF {
34
35WTF_EXPORT_PRIVATE PrintStream& dataFile();
36WTF_EXPORT_PRIVATE void setDataFile(const char* path);
37WTF_EXPORT_PRIVATE void setDataFile(std::unique_ptr<PrintStream>&&);
38
39WTF_EXPORT_PRIVATE void dataLogFV(const char* format, va_list) WTF_ATTRIBUTE_PRINTF(1, 0);
40WTF_EXPORT_PRIVATE void dataLogF(const char* format, ...) WTF_ATTRIBUTE_PRINTF(1, 2);
41WTF_EXPORT_PRIVATE void dataLogFString(const char*);
42
43template<typename... Types>
44NEVER_INLINE void dataLog(const Types&... values)
45{
46 dataFile().print(values...);
47}
48
49template<typename... Types>
50void dataLogLn(const Types&... values)
51{
52 dataLog(values..., "\n");
53}
54
55template<typename... Types>
56ALWAYS_INLINE void dataLogIf(bool shouldLog, const Types&... values)
57{
58 if (UNLIKELY(shouldLog))
59 dataLog(values...);
60}
61
62template<typename... Types>
63ALWAYS_INLINE void dataLogLnIf(bool shouldLog, const Types&... values)
64{
65 if (UNLIKELY(shouldLog))
66 dataLogLn(values...);
67}
68
69} // namespace WTF
