59struct LocalTimeOffset {
60 WTF_MAKE_STRUCT_FAST_ALLOCATED;
61
62 LocalTimeOffset() = default;
63 constexpr LocalTimeOffset(bool isDST, int offset)
64 : isDST(isDST)
65 , offset(offset)
66 {
67 }
68
69 bool operator==(const LocalTimeOffset& other)
70 {
71 return isDST == other.isDST && offset == other.offset;
72 }
73
74 bool operator!=(const LocalTimeOffset& other)
75 {
76 return isDST != other.isDST || offset != other.offset;
77 }
78
79 bool isDST { false };
80 int offset { 0 };
81};

121class TimeClippedPositiveMilliseconds {
122public:
123 static constexpr int64_t hoursPerDay = 24;
124 static constexpr int64_t minutesPerHour = 60;
125 static constexpr int64_t secondsPerMinute = 60;
126 static constexpr int64_t msPerSecond = 1000;
127 static constexpr int64_t msPerMonth = 2592000000;
128 static constexpr int64_t secondsPerHour = secondsPerMinute * minutesPerHour;
129 static constexpr int64_t secondsPerDay = secondsPerHour * hoursPerDay;
130 static constexpr int64_t msPerMinute = msPerSecond * secondsPerMinute;
131 static constexpr int64_t msPerHour = msPerSecond * secondsPerHour;
132 static constexpr int64_t msPerDay = msPerSecond * secondsPerDay;
133 static constexpr int64_t maxECMAScriptTime = 8.64E15;
134
135 explicit TimeClippedPositiveMilliseconds(int64_t value)
136 : m_value(value)
137 {
138 ASSERT(value >= 0);
139 }
140
141 int64_t value() const { return m_value; }
142 double asDouble() const { return static_cast<double>(m_value); }
143private:
144 int64_t m_value;
145};

147inline double timeClip(double t)
148{
149 if (std::abs(t) > maxECMAScriptTime)
150 return std::numeric_limits<double>::quiet_NaN();
151 return std::trunc(t) + 0.0;
152}

154inline double daysFrom1970ToYear(int year)
155{
156 // The Gregorian Calendar rules for leap years:
157 // Every fourth year is a leap year. 2004, 2008, and 2012 are leap years.
158 // However, every hundredth year is not a leap year. 1900 and 2100 are not leap years.
159 // Every four hundred years, there's a leap year after all. 2000 and 2400 are leap years.
160
161 static constexpr int leapDaysBefore1971By4Rule = 1970 / 4;
162 static constexpr int excludedLeapDaysBefore1971By100Rule = 1970 / 100;
163 static constexpr int leapDaysBefore1971By400Rule = 1970 / 400;
164
165 const double yearMinusOne = static_cast<double>(year) - 1;
166 const double yearsToAddBy4Rule = floor(yearMinusOne / 4.0) - leapDaysBefore1971By4Rule;
167 const double yearsToExcludeBy100Rule = floor(yearMinusOne / 100.0) - excludedLeapDaysBefore1971By100Rule;
168 const double yearsToAddBy400Rule = floor(yearMinusOne / 400.0) - leapDaysBefore1971By400Rule;
169
170 return 365.0 * (year - 1970.0) + yearsToAddBy4Rule - yearsToExcludeBy100Rule + yearsToAddBy400Rule;
171}
172
173inline int64_t daysFrom1970ToYearTimeClippedPositive(int year)
174{
175 static constexpr int leapDaysBefore1971By4Rule = 1970 / 4;
176 static constexpr int excludedLeapDaysBefore1971By100Rule = 1970 / 100;
177 static constexpr int leapDaysBefore1971By400Rule = 1970 / 400;
178
179 ASSERT(year >= 1970);
180 const int64_t yearMinusOne = year - 1;
181 const int64_t yearsToAddBy4Rule = yearMinusOne / 4.0 - leapDaysBefore1971By4Rule;
182 const int64_t yearsToExcludeBy100Rule = yearMinusOne / 100.0 - excludedLeapDaysBefore1971By100Rule;
183 const int64_t yearsToAddBy400Rule = yearMinusOne / 400.0 - leapDaysBefore1971By400Rule;
184
185 return 365 * (year - 1970) + yearsToAddBy4Rule - yearsToExcludeBy100Rule + yearsToAddBy400Rule;
186}

188inline bool isLeapYear(int year)
189{
190 if (year % 4 != 0)
191 return false;
192 if (year % 400 == 0)
193 return true;
194 if (year % 100 == 0)
195 return false;
196 return true;
197}

219inline int dayInYear(double ms, int year)
220{
221 double result = msToDays(ms) - daysFrom1970ToYear(year);
222 return std::isnan(result) ? 0 : static_cast<int>(result);
223}

225inline int dayInYear(TimeClippedPositiveMilliseconds ms, int year)
226{
227 return static_cast<int>(msToDays(ms) - daysFrom1970ToYearTimeClippedPositive(year));
228}

230// Returns the number of days from 1970-01-01 to the specified date.
231inline double dateToDaysFrom1970(int year, int month, int day)
232{
233 year += month / 12;
234
235 month %= 12;
236 if (month < 0) {
237 month += 12;
238 --year;
239 }
240
241 double yearday = floor(daysFrom1970ToYear(year));
242 ASSERT((year >= 1970 && yearday >= 0) || (year < 1970 && yearday < 0));
243 return yearday + dayInYear(year, month, day);
244}

246inline int msToYear(double ms)
247{
248 double msAsYears = std::floor(ms / (msPerDay * 365.2425));
249 if (std::isnan(msAsYears))
250 msAsYears = 0;
251 int approxYear = static_cast<int>(msAsYears + 1970);
252 double msFromApproxYearTo1970 = msPerDay * daysFrom1970ToYear(approxYear);
253 if (msFromApproxYearTo1970 > ms)
254 return approxYear - 1;
255 if (msFromApproxYearTo1970 + msPerDay * daysInYear(approxYear) <= ms)
256 return approxYear + 1;
257 return approxYear;
258}

260inline int msToMinutes(double ms)
261{
262 double result = fmod(floor(ms / msPerMinute), minutesPerHour);
263 if (result < 0)
264 result += minutesPerHour;
265 return static_cast<int>(result);
266}

306inline int msToWeekDay(double ms)
307{
308 int wd = (static_cast<int>(msToDays(ms)) + 4) % 7;
309 if (wd < 0)

312}

315{

321inline int monthFromDayInYear(int dayInYear, bool leapYear)
322{
323 const int d = dayInYear;
324 int step;
325
326 if (d < (step = 31))
327 return 0;
328 step += (leapYear ? 29 : 28);
329 if (d < step)
330 return 1;
331 if (d < (step += 31))
332 return 2;
333 if (d < (step += 30))
334 return 3;
335 if (d < (step += 31))
336 return 4;
337 if (d < (step += 30))
338 return 5;
339 if (d < (step += 31))
340 return 6;
341 if (d < (step += 31))
342 return 7;
343 if (d < (step += 30))
344 return 8;
345 if (d < (step += 31))
346 return 9;
347 if (d < step + 30)
348 return 10;
349 return 11;
350}

352inline int dayInMonthFromDayInYear(int dayInYear, bool leapYear)
353{
354 auto checkMonth = [] (int dayInYear, int& startDayOfThisMonth, int& startDayOfNextMonth, int daysInThisMonth) -> bool {
355 startDayOfThisMonth = startDayOfNextMonth;
356 startDayOfNextMonth += daysInThisMonth;
357 return (dayInYear <= startDayOfNextMonth);
358 };
359
360 const int d = dayInYear;
361 int step;
362 int next = 30;
363
364 if (d <= next)
365 return d + 1;
366 const int daysInFeb = (leapYear ? 29 : 28);
367 if (checkMonth(d, step, next, daysInFeb))
368 return d - step;
369 if (checkMonth(d, step, next, 31))
370 return d - step;
371 if (checkMonth(d, step, next, 30))
372 return d - step;
373 if (checkMonth(d, step, next, 31))
374 return d - step;
375 if (checkMonth(d, step, next, 30))
376 return d - step;
377 if (checkMonth(d, step, next, 31))
378 return d - step;
379 if (checkMonth(d, step, next, 31))
380 return d - step;
381 if (checkMonth(d, step, next, 30))
382 return d - step;
383 if (checkMonth(d, step, next, 31))
384 return d - step;
385 if (checkMonth(d, step, next, 30))
386 return d - step;
387 step = next;
388 return d - step;
389}
