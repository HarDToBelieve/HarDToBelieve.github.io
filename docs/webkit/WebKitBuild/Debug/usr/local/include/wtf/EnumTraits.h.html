26#pragma once
27
28#include <type_traits>
29
30namespace WTF {
31
32template<typename> struct EnumTraits;
33
34template<typename E, E...> struct EnumValues;
35
36template<typename T, typename E> struct EnumValueChecker;
37
38template<typename T, typename E, E e, E... es>
39struct EnumValueChecker<T, EnumValues<E, e, es...>> {
40 static constexpr bool isValidEnum(T t)
41 {
42 return (static_cast<T>(e) == t) ? true : EnumValueChecker<T, EnumValues<E, es...>>::isValidEnum(t);
43 }
44};
45
46template<typename T, typename E>
47struct EnumValueChecker<T, EnumValues<E>> {
48 static constexpr bool isValidEnum(T)
49 {
50 return false;
51 }
52};
53
54template<typename E, typename T, typename = std::enable_if_t<!std::is_same_v<std::underlying_type_t<E>, bool>>>
55constexpr auto isValidEnum(T t) -> decltype(EnumValueChecker<T, typename EnumTraits<E>::values>::isValidEnum(t), bool())
56{
57 static_assert(sizeof(T) >= sizeof(std::underlying_type_t<E>), "Integral type must be at least the size of the underlying enum type");
58
59 return EnumValueChecker<T, typename EnumTraits<E>::values>::isValidEnum(t);
60}
61
62template<typename E, typename T, typename = std::enable_if_t<!std::is_same_v<std::underlying_type_t<E>, bool>>>
63auto isValidEnum(T t) -> decltype(EnumTraits<E>::isValidEnum(t), bool())
64{
65 return EnumTraits<E>::isValidEnum(t);
66}
67
68template<typename E, typename T, typename = std::enable_if_t<std::is_same_v<std::underlying_type_t<E>, bool>>>
69constexpr bool isValidEnum(T t)
70{
71 return !t || t == 1;
72}
73
74template<typename E>
75constexpr auto enumToUnderlyingType(E e)
76{
77 return static_cast<std::underlying_type_t<E>>(e);
78}
79
80}
