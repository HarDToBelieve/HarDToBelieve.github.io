26#pragma once
27
28#include <wtf/RefCountedArray.h>
29
30namespace WTF {
31
32// Wrapper around RefCountedArray, fixed-sized, memory compact Vector.
33// Copy constructor / assignment operator work differently from RefCountedArray: works as like a Vector.
34template<typename T>
35class FixedVector {
36public:
37 using iterator = T*;
38 using const_iterator = const T*;
39 using reverse_iterator = std::reverse_iterator<iterator>;
40 using const_reverse_iterator = std::reverse_iterator<const_iterator>;
41
42 FixedVector() = default;
43 FixedVector(const FixedVector& other)
44 : m_storage(other.m_storage.clone())
45 { }
46 FixedVector(FixedVector&& other) = default;
47
48 FixedVector& operator=(const FixedVector& other)
49 {
50 FixedVector tmp(other);
51 swap(tmp);
52 return *this;
53 }
54
55 FixedVector& operator=(FixedVector&& other)
56 {
57 FixedVector tmp(WTFMove(other));
58 swap(tmp);
59 return *this;
60 }
61
62 explicit FixedVector(size_t size)
63 : m_storage(size)
64 { }
65
66 template<size_t inlineCapacity, typename OverflowHandler>
67 explicit FixedVector(const Vector<T, inlineCapacity, OverflowHandler>& other)
68 : m_storage(other)
69 { }
70
71 template<size_t inlineCapacity, typename OverflowHandler>
72 FixedVector& operator=(const Vector<T, inlineCapacity, OverflowHandler>& other)
73 {
74 m_storage = other;
75 return *this;
76 }
77
78 template<size_t inlineCapacity, typename OverflowHandler>
79 explicit FixedVector(Vector<T, inlineCapacity, OverflowHandler>&& other)
80 : m_storage(WTFMove(other))
81 { }
82
83 template<size_t inlineCapacity, typename OverflowHandler>
84 FixedVector& operator=(Vector<T, inlineCapacity, OverflowHandler>&& other)
85 {
86 m_storage = WTFMove(other);
87 return *this;
88 }
89
90 size_t size() const { return m_storage.size(); }
91 bool isEmpty() const { return m_storage.isEmpty(); }
92 size_t byteSize() const { return m_storage.byteSize(); }
93
94 T* data() { return m_storage.data(); }
95 iterator begin() { return m_storage.begin(); }
96 iterator end() { return m_storage.end(); }
97
98 const T* data() const { return const_cast<FixedVector*>(this)->data(); }
99 const_iterator begin() const { return const_cast<FixedVector*>(this)->begin(); }
100 const_iterator end() const { return const_cast<FixedVector*>(this)->end(); }
101
102 reverse_iterator rbegin() { return m_storage.rbegin(); }
103 reverse_iterator rend() { return m_storage.rend(); }
104 const_reverse_iterator rbegin() const { return m_storage.rbegin(); }
105 const_reverse_iterator rend() const { return m_storage.rend(); }
106
107 T& at(size_t i) { return m_storage.at(i); }
108 const T& at(size_t i) const { return m_storage.at(i); }
109
110 T& operator[](size_t i) { return at(i); }
111 const T& operator[](size_t i) const { return at(i); }
112
113 T& first() { return (*this)[0]; }
114 const T& first() const { return (*this)[0]; }
115 T& last() { return (*this)[size() - 1]; }
116 const T& last() const { return (*this)[size() - 1]; }
117
118 void fill(const T& val) { m_storage.fill(val); }
119
120 bool operator==(const FixedVector<T>& other) const { return m_storage == other.m_storage; }
121
122 void swap(FixedVector<T>& other)
123 {
124 m_storage.swap(other.m_storage);
125 }
126
127 static ptrdiff_t offsetOfStorage() { return OBJECT_OFFSETOF(FixedVector, m_storage); }
128
129private:
130 friend class JSC::LLIntOffsetsExtractor;
131
132 RefCountedArray<T> m_storage;
133};
134static_assert(sizeof(FixedVector<int>) == sizeof(int*));
135
136template<typename T>
137inline void swap(FixedVector<T>& a, FixedVector<T>& b)
138{
139 a.swap(b);
140}
141
142} // namespace WTF
143
144using WTF::FixedVector;
