41enum Kind {
42 Primitive,
43 JSValue,
44 NumberOfKinds
45};

47inline void ensureGigacage() { }

56ALWAYS_INLINE const char* name(Kind kind)
57{
58 switch (kind) {
59 case Primitive:
60 return "Primitive";
61 case JSValue:
62 return "JSValue";
63 case NumberOfKinds:
64 break;
65 }
66 RELEASE_ASSERT_NOT_REACHED();
67 return nullptr;
68}
