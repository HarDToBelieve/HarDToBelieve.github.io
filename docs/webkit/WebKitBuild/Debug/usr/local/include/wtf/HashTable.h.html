308 // HashTableCapacityForSize computes the upper power of two capacity to hold the size parameter.
309 // This is done at compile time to initialize the HashTraits.
310 template<unsigned size>
311 struct HashTableCapacityForSize {
312 // Load-factor for small table is 75%.
313 static constexpr unsigned smallMaxLoadNumerator = 3;
314 static constexpr unsigned smallMaxLoadDenominator = 4;
315 // Load-factor for large table is 50%.
316 static constexpr unsigned largeMaxLoadNumerator = 1;
317 static constexpr unsigned largeMaxLoadDenominator = 2;
318 static constexpr unsigned maxSmallTableCapacity = 1024;
319 static constexpr unsigned minLoad = 6;
320
321 static constexpr bool shouldExpand(uint64_t keyAndDeleteCount, uint64_t tableSize)
322 {
323 if (tableSize <= maxSmallTableCapacity)
324 return keyAndDeleteCount * smallMaxLoadDenominator >= tableSize * smallMaxLoadNumerator;
325 return keyAndDeleteCount * largeMaxLoadDenominator >= tableSize * largeMaxLoadNumerator;
326 }
327
328 static constexpr unsigned capacityForSize(uint32_t sizeArg)
329 {
330 if (!sizeArg)
331 return 0;
332 constexpr unsigned maxCapacity = 1U << 31;
333 UNUSED_PARAM(maxCapacity);
334 ASSERT_UNDER_CONSTEXPR_CONTEXT(sizeArg <= maxCapacity);
335 uint32_t capacity = roundUpToPowerOfTwo(sizeArg);
336 ASSERT_UNDER_CONSTEXPR_CONTEXT(capacity <= maxCapacity);
337 if (shouldExpand(sizeArg, capacity)) {
338 ASSERT_UNDER_CONSTEXPR_CONTEXT((static_cast<uint64_t>(capacity) * 2) <= maxCapacity);
339 return capacity * 2;
340 }
341 return capacity;
342 }
343
344 static constexpr unsigned value = capacityForSize(size);
345 static_assert(size > 0, "HashTableNonZeroMinimumCapacity");
346 static_assert(!static_cast<unsigned>(value >> 31), "HashTableNoCapacityOverflow");
347 };

667 {
668 static_assert(sizeof(Value) <= 128, "Your HashTable types are too big to efficiently move when rehashing. Consider using UniqueRef instead");
669 checkKey<HashTranslator>(key);
670
671 unsigned k = 0;
672 ValueType* table = m_table;
673 if (!table)
674 return nullptr;
675
676 unsigned sizeMask = tableSizeMask();
677 unsigned h = HashTranslator::hash(key);
678 unsigned i = h & sizeMask;
679
680#if DUMP_HASHTABLE_STATS
681 ++HashTableStats::numAccesses;
682 unsigned probeCount = 0;
683#endif
684
685#if DUMP_HASHTABLE_STATS_PER_TABLE
686 ++m_stats->numAccesses;
687#endif
688
689 while (true) {

694 if (HashTranslator::equal(Extractor::extract(*entry), key))

698 return nullptr;

701 return nullptr;

706#if DUMP_HASHTABLE_STATS

708 HashTableStats::recordCollisionAtCount(probeCount);
709#endif
710
711#if DUMP_HASHTABLE_STATS_PER_TABLE
712 m_stats->recordCollisionAtCount(probeCount);
713#endif
714
715 if (k == 0)
716 k = 1 | doubleHash(h);
717 i = (i + k) & sizeMask;
718 }
719 }

1173 {

1175 // make a function call, which prevents the compiler from keeping
1176 // the values in register.
1177 unsigned removedBucketCount = 0;
1178 ValueType* table = m_table;
1179
1180 for (unsigned i = tableSize(); i--;) {
1181 ValueType& bucket = table[i];
1182 if (isEmptyOrDeletedBucket(bucket))
1183 continue;
1184
1185 if (!functor(bucket))
1186 continue;
1187
1188 deleteBucket(bucket);
1189 ++removedBucketCount;
1190 }
1191 if (removedBucketCount) {
1192 setDeletedCount(deletedCount() + removedBucketCount);
1193 setKeyCount(keyCount() - removedBucketCount);
1194 }
1195
1196 if (shouldShrink())
1197 shrinkToBestSize();
1198
1199 internalCheckTableConsistency();
1200 return removedBucketCount;
1201 }

1248 template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits>
1249 constexpr unsigned HashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits>::computeBestTableSize(unsigned keyCount)
1250 {
1251 unsigned bestTableSize = WTF::roundUpToPowerOfTwo(keyCount);
1252
1253 if (HashTableSizePolicy::shouldExpand(keyCount, bestTableSize))
1254 bestTableSize *= 2;
1255
1256 auto aboveThresholdForEagerExpansion = [](double loadFactor, unsigned keyCount, unsigned tableSize)
1257 {
1258 // Here is the rationale behind this calculation, using 3/4 load-factor.
1259 // With maxLoad at 3/4 and minLoad at 1/6, our average load is 11/24.
1260 // If we are getting half-way between 11/24 and 3/4, we double the size
1261 // to avoid being too close to loadMax and bring the ratio close to 11/24. This
1262 // give us a load in the bounds [9/24, 15/24).
1263 double maxLoadRatio = loadFactor;
1264 double minLoadRatio = 1.0 / minLoad;
1265 double averageLoadRatio = (maxLoadRatio + minLoadRatio) / 2;
1266 double halfWayBetweenAverageAndMaxLoadRatio = (averageLoadRatio + maxLoadRatio) / 2;
1267 return keyCount >= tableSize * halfWayBetweenAverageAndMaxLoadRatio;
1268 };
1269
1270 if (bestTableSize <= maxSmallTableCapacity) {
1271 constexpr double smallLoadFactor = static_cast<double>(smallMaxLoadNumerator) / smallMaxLoadDenominator;
1272 if (aboveThresholdForEagerExpansion(smallLoadFactor, keyCount, bestTableSize))
1273 bestTableSize *= 2;
1274 } else {
1275 constexpr double largeLoadFactor = static_cast<double>(largeMaxLoadNumerator) / largeMaxLoadDenominator;
1276 if (aboveThresholdForEagerExpansion(largeLoadFactor, keyCount, bestTableSize))
1277 bestTableSize *= 2;
1278 }
1279 unsigned minimumTableSize = KeyTraits::minimumTableSize;
1280 return std::max(bestTableSize, minimumTableSize);
1281 }

1345 oldEntry.~ValueType();

1355 }
1356 }

1361 internalCheckTableConsistency();
