36template<bool isInteger, typename T> struct GenericHashTraitsBase;

38template<typename T> struct GenericHashTraitsBase<false, T> {
39 // The emptyValueIsZero flag is used to optimize allocation of empty hash tables with zeroed memory.
40 static constexpr bool emptyValueIsZero = false;
41
42 // The hasIsEmptyValueFunction flag allows the hash table to automatically generate code to check
43 // for the empty value when it can be done with the equality operator, but allows custom functions
44 // for cases like String that need them.
45 static constexpr bool hasIsEmptyValueFunction = false;
46
47 // Used by WeakPtr to indicate that the value may become deleted without being explicitly removed.
48 static constexpr bool hasIsReleasedWeakValueFunction = false;
49
50 // The starting table size. Can be overridden when we know beforehand that
51 // a hash table will have at least N entries.
52 static constexpr unsigned minimumTableSize = 8;
53};
54
55// Default integer traits disallow both 0 and -1 as keys (max value instead of -1 for unsigned).
56template<typename T> struct GenericHashTraitsBase<true, T> : GenericHashTraitsBase<false, T> {
57 static constexpr bool emptyValueIsZero = true;
58 static void constructDeletedValue(T& slot) { slot = static_cast<T>(-1); }
59 static bool isDeletedValue(T value) { return value == static_cast<T>(-1); }
60};

62template<typename T> struct GenericHashTraits : GenericHashTraitsBase<std::is_integral<T>::value, T> {
63 typedef T TraitType;
64 typedef T EmptyValueType;
65
66 static T emptyValue() { return T(); }
67
68 template<typename U, typename V>
69 static void assignToEmpty(U& emptyValue, V&& value)
70 {
71 emptyValue = std::forward<V>(value);
72 }
73
74 template <typename Traits>
75 static void constructEmptyValue(T& slot)
76 {
77 new (NotNull, std::addressof(slot)) T(Traits::emptyValue());
78 }
79
80 // Type for return value of functions that do not transfer ownership, such as get.
81 typedef T PeekType;
82 template<typename U> static U&& peek(U&& value) { return std::forward<U>(value); }
83
84 typedef T TakeType;
85 template<typename U> static TakeType take(U&& value) { return std::forward<U>(value); }
86};

145template<typename T, typename Deleter> struct HashTraits<std::unique_ptr<T, Deleter>> : SimpleClassHashTraits<std::unique_ptr<T, Deleter>> {
146 typedef std::nullptr_t EmptyValueType;
147 static EmptyValueType emptyValue() { return nullptr; }
148
149 static void constructDeletedValue(std::unique_ptr<T, Deleter>& slot) { new (NotNull, std::addressof(slot)) std::unique_ptr<T, Deleter> { reinterpret_cast<T*>(-1) }; }
150 static bool isDeletedValue(const std::unique_ptr<T, Deleter>& value) { return value.get() == reinterpret_cast<T*>(-1); }
151
152 typedef T* PeekType;
153 static T* peek(const std::unique_ptr<T, Deleter>& value) { return value.get(); }
154 static T* peek(std::nullptr_t) { return nullptr; }
155
156 static void customDeleteBucket(std::unique_ptr<T, Deleter>& value)
157 {
158 // The custom delete function exists to avoid a dead store before the value is destructed.
159 // The normal destruction sequence of a bucket would be:
160 // 1) Call the destructor of unique_ptr.
161 // 2) unique_ptr store a zero for its internal pointer.
162 // 3) unique_ptr destroys its value.
163 // 4) Call constructDeletedValue() to set the bucket as destructed.
164 //
165 // The problem is the call in (3) prevents the compile from eliminating the dead store in (2)
166 // becase a side effect of free() could be observing the value.
167 //
168 // This version of deleteBucket() ensures the dead 2 stores changing "value"
169 // are on the same side of the function call.
170 ASSERT(!isDeletedValue(value));
171 T* pointer = value.release();
172 constructDeletedValue(value);
173
174 // The null case happens if a caller uses std::move() to remove the pointer before calling remove()
175 // with an iterator. This is very uncommon.
176 if (LIKELY(pointer))
177 Deleter()(pointer);
178 }
179};

181template<typename T> struct HashTraits<UniqueRef<T>> : SimpleClassHashTraits<UniqueRef<T>> {
182 typedef std::nullptr_t EmptyValueType;
183 static EmptyValueType emptyValue() { return nullptr; }
184
185 static void constructDeletedValue(UniqueRef<T>& slot) { new (NotNull, std::addressof(slot)) UniqueRef<T> { reinterpret_cast<T*>(-1) }; }
186 static bool isDeletedValue(const UniqueRef<T>& value) { return value.get() == reinterpret_cast<T*>(-1); }
187
188 typedef T* PeekType;
189 static const T* peek(const UniqueRef<T>& value) { return &value.get(); }
190 static T* peek(UniqueRef<T>& value) { return &value.get(); }
191 static T* peek(std::nullptr_t) { return nullptr; }
192
193 using TakeType = std::unique_ptr<T>;
194 static TakeType take(UniqueRef<T>&& value) { return value.moveToUniquePtr(); }
195 static TakeType take(std::nullptr_t) { return nullptr; }
196};

198template<typename P> struct HashTraits<RefPtr<P>> : SimpleClassHashTraits<RefPtr<P>> {
199 static P* emptyValue() { return nullptr; }
200
201 typedef P* PeekType;
202 static PeekType peek(const RefPtr<P>& value) { return value.get(); }
203 static PeekType peek(P* value) { return value; }
204
205 static void customDeleteBucket(RefPtr<P>& value)
206 {
207 // See unique_ptr's customDeleteBucket() for an explanation.
208 ASSERT(!SimpleClassHashTraits<RefPtr<P>>::isDeletedValue(value));
209 auto valueToBeDestroyed = WTFMove(value);
210 SimpleClassHashTraits<RefPtr<P>>::constructDeletedValue(value);
211 }
212};

214template<typename P> struct RefHashTraits : SimpleClassHashTraits<Ref<P>> {
215 static constexpr bool emptyValueIsZero = true;
216 static Ref<P> emptyValue() { return HashTableEmptyValue; }
217
218 template <typename>
219 static void constructEmptyValue(Ref<P>& slot)
220 {
221 new (NotNull, std::addressof(slot)) Ref<P>(HashTableEmptyValue);
222 }
223
224 static constexpr bool hasIsEmptyValueFunction = true;
225 static bool isEmptyValue(const Ref<P>& value) { return value.isHashTableEmptyValue(); }
226
227 using PeekType = P*;
228 static PeekType peek(const Ref<P>& value) { return const_cast<PeekType>(value.ptrAllowingHashTableEmptyValue()); }
229 static PeekType peek(P* value) { return value; }
230
231 using TakeType = RefPtr<P>;
232 static TakeType take(Ref<P>&& value) { return isEmptyValue(value) ? nullptr : RefPtr<P>(WTFMove(value)); }
233};

257// This struct template is an implementation detail of the isHashTraitsEmptyValue function,
258// which selects either the emptyValue function or the isEmptyValue function to check for empty values.
259template<typename Traits, bool hasEmptyValueFunction> struct HashTraitsEmptyValueChecker;
260template<typename Traits> struct HashTraitsEmptyValueChecker<Traits, true> {
261 template<typename T> static bool isEmptyValue(const T& value) { return Traits::isEmptyValue(value); }
262};
263template<typename Traits> struct HashTraitsEmptyValueChecker<Traits, false> {
264 template<typename T> static bool isEmptyValue(const T& value) { return value == Traits::emptyValue(); }
265};
266template<typename Traits, typename T> inline bool isHashTraitsEmptyValue(const T& value)
267{
268 return HashTraitsEmptyValueChecker<Traits, Traits::hasIsEmptyValueFunction>::isEmptyValue(value);
269}

283template<typename Traits, typename T>
284struct HashTraitHasCustomDelete {
285 static T& bucketArg;
286 template<typename X> static std::true_type TestHasCustomDelete(X*, decltype(X::customDeleteBucket(bucketArg))* = nullptr);
287 static std::false_type TestHasCustomDelete(...);
288 typedef decltype(TestHasCustomDelete(static_cast<Traits*>(nullptr))) ResultType;
289 static constexpr bool value = ResultType::value;
290};
291
292template<typename Traits, typename T>
293typename std::enable_if<HashTraitHasCustomDelete<Traits, T>::value>::type
294hashTraitsDeleteBucket(T& value)
295{
296 Traits::customDeleteBucket(value);
297}
298
299template<typename Traits, typename T>
300typename std::enable_if<!HashTraitHasCustomDelete<Traits, T>::value>::type
301hashTraitsDeleteBucket(T& value)
302{
303 value.~T();
304 Traits::constructDeletedValue(value);
305}

307template<typename FirstTraitsArg, typename SecondTraitsArg>
308struct PairHashTraits : GenericHashTraits<std::pair<typename FirstTraitsArg::TraitType, typename SecondTraitsArg::TraitType>> {
309 typedef FirstTraitsArg FirstTraits;
310 typedef SecondTraitsArg SecondTraits;
311 typedef std::pair<typename FirstTraits::TraitType, typename SecondTraits::TraitType> TraitType;
312 typedef std::pair<typename FirstTraits::EmptyValueType, typename SecondTraits::EmptyValueType> EmptyValueType;
313
314 static constexpr bool emptyValueIsZero = FirstTraits::emptyValueIsZero && SecondTraits::emptyValueIsZero;
315 static EmptyValueType emptyValue() { return std::make_pair(FirstTraits::emptyValue(), SecondTraits::emptyValue()); }
316
317 static constexpr unsigned minimumTableSize = FirstTraits::minimumTableSize;
318
319 static void constructDeletedValue(TraitType& slot) { FirstTraits::constructDeletedValue(slot.first); }
320 static bool isDeletedValue(const TraitType& value) { return FirstTraits::isDeletedValue(value.first); }
321};

326template<typename FirstTrait, typename... Traits>
327struct TupleHashTraits : GenericHashTraits<std::tuple<typename FirstTrait::TraitType, typename Traits::TraitType...>> {
328 typedef std::tuple<typename FirstTrait::TraitType, typename Traits::TraitType...> TraitType;
329 typedef std::tuple<typename FirstTrait::EmptyValueType, typename Traits::EmptyValueType...> EmptyValueType;
330
331 // We should use emptyValueIsZero = Traits::emptyValueIsZero &&... whenever we switch to C++17. We can't do anything
332 // better here right now because GCC can't do C++.
333 template<typename BoolType>
334 static constexpr bool allTrue(BoolType value) { return value; }
335 template<typename BoolType, typename... BoolTypes>
336 static constexpr bool allTrue(BoolType value, BoolTypes... values) { return value && allTrue(values...); }
337 static constexpr bool emptyValueIsZero = allTrue(FirstTrait::emptyValueIsZero, Traits::emptyValueIsZero...);
338 static EmptyValueType emptyValue() { return std::make_tuple(FirstTrait::emptyValue(), Traits::emptyValue()...); }
339
340 static constexpr unsigned minimumTableSize = FirstTrait::minimumTableSize;
341
342 static void constructDeletedValue(TraitType& slot) { FirstTrait::constructDeletedValue(std::get<0>(slot)); }
343 static bool isDeletedValue(const TraitType& value) { return FirstTrait::isDeletedValue(std::get<0>(value)); }
344};

349template<typename KeyTraitsArg, typename ValueTraitsArg>
350struct KeyValuePairHashTraits : GenericHashTraits<KeyValuePair<typename KeyTraitsArg::TraitType, typename ValueTraitsArg::TraitType>> {
351 typedef KeyTraitsArg KeyTraits;
352 typedef ValueTraitsArg ValueTraits;
353 typedef KeyValuePair<typename KeyTraits::TraitType, typename ValueTraits::TraitType> TraitType;
354 typedef KeyValuePair<typename KeyTraits::EmptyValueType, typename ValueTraits::EmptyValueType> EmptyValueType;
355 typedef typename ValueTraitsArg::TraitType ValueType;
356
357 static constexpr bool emptyValueIsZero = KeyTraits::emptyValueIsZero && ValueTraits::emptyValueIsZero;
358 static EmptyValueType emptyValue() { return KeyValuePair<typename KeyTraits::EmptyValueType, typename ValueTraits::EmptyValueType>(KeyTraits::emptyValue(), ValueTraits::emptyValue()); }
359
360 template <typename>
361 static void constructEmptyValue(TraitType& slot)
362 {
363 KeyTraits::template constructEmptyValue<KeyTraits>(slot.key);
364 ValueTraits::template constructEmptyValue<ValueTraits>(slot.value);
365 }
366
367 static constexpr unsigned minimumTableSize = KeyTraits::minimumTableSize;
368
369 static void constructDeletedValue(TraitType& slot) { KeyTraits::constructDeletedValue(slot.key); }
370 static bool isDeletedValue(const TraitType& value) { return KeyTraits::isDeletedValue(value.key); }
371
372 static void customDeleteBucket(TraitType& value)
373 {
374 static_assert(std::is_trivially_destructible<KeyValuePair<int, int>>::value,
375 "The wrapper itself has to be trivially destructible for customDeleteBucket() to make sense, since we do not destruct the wrapper itself.");
376
377 hashTraitsDeleteBucket<KeyTraits>(value.key);
378 value.value.~ValueType();
379 }
380};

399// Useful for classes that want complete control over what is empty and what is deleted,
400// and how to construct both.
401template<typename T>
402struct CustomHashTraits : public GenericHashTraits<T> {
403 static constexpr bool emptyValueIsZero = false;
404 static constexpr bool hasIsEmptyValueFunction = true;
405
406 static void constructDeletedValue(T& slot)
407 {
408 new (NotNull, std::addressof(slot)) T(T::DeletedValue);
409 }
410
411 static bool isDeletedValue(const T& value)
412 {
413 return value.isDeletedValue();
414 }
415
416 static T emptyValue()
417 {
418 return T(T::EmptyValue);
419 }
420
421 static bool isEmptyValue(const T& value)
422 {
423 return value.isEmptyValue();
424 }
425};
