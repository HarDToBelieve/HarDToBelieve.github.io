26#pragma once
27
28#include <wtf/PrintStream.h>
29#include <wtf/text/WTFString.h>
30
31namespace WTF {
32
33class Indenter {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 Indenter(unsigned count = 0, String string = " "_s)
37 : m_count(count)
38 , m_string(string)
39 { }
40
41 Indenter(const Indenter& other)
42 : m_count(other.m_count)
43 , m_string(other.m_string)
44 { }
45
46 void dump(PrintStream& out) const
47 {
48 unsigned levels = m_count;
49 while (levels--)
50 out.print(m_string);
51 }
52
53 unsigned operator++() { return ++m_count; }
54 unsigned operator++(int) { return m_count++; }
55 unsigned operator--() { return --m_count; }
56 unsigned operator--(int) { return m_count--; }
57
58private:
59 unsigned m_count;
60 String m_string;
61};
62
63} // namespace WTF
64
65using WTF::Indenter;
