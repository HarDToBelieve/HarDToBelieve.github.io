26#pragma once
27
28#include <type_traits>
29
30namespace WTF {
31
32template<class Container>
33class IndexedContainerIterator {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 IndexedContainerIterator()
37 : m_container(nullptr)
38 , m_index(0)
39 {
40 }
41
42 IndexedContainerIterator(const Container& container, unsigned index)
43 : m_container(&container)
44 , m_index(findNext(index))
45 {
46 }
47
48 auto operator*() -> typename std::result_of<decltype(&Container::at)(const Container, unsigned)>::type
49 {
50 return m_container->at(m_index);
51 }
52
53 IndexedContainerIterator& operator++()
54 {
55 m_index = findNext(m_index + 1);
56 return *this;
57 }
58
59 bool operator==(const IndexedContainerIterator& other) const
60 {
61 ASSERT(m_container == other.m_container);
62 return m_index == other.m_index;
63 }
64
65 bool operator!=(const IndexedContainerIterator& other) const
66 {
67 return !(*this == other);
68 }
69
70private:
71 unsigned findNext(unsigned index)
72 {
73 while (index < m_container->size() && !m_container->at(index))
74 index++;
75 return index;
76 }
77
78 const Container* m_container;
79 unsigned m_index;
80};
81
82} // namespace WTF
