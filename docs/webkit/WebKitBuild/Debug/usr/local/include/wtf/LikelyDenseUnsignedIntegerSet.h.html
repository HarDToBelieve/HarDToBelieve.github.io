38DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(LikelyDenseUnsignedIntegerSet);

40// This is effectively a Variant<HashSet, Pair<BitVector, IndexType>>
41// If it is in BitVector mode, it keeps track of the minimum value in the set, and has the bitVector shifted by the same amount.
42// So for example {4000, 4002, 4003} would be represented as the bitVector 1101 with a m_min of 4000.
43// It shifts between the two modes whenever that would at least halve its memory usage. So it will never use more than twice the optimal amount of memory, and yet should not ping-pong between the two modes too often.
44template<typename IndexType>
45class LikelyDenseUnsignedIntegerSet {
46 WTF_MAKE_FAST_ALLOCATED;
47 WTF_MAKE_NONCOPYABLE(LikelyDenseUnsignedIntegerSet);
48 static_assert(std::is_unsigned<IndexType>::value);
49 using Set = HashSet<IndexType, WTF::IntHash<IndexType>, WTF::UnsignedWithZeroKeyHashTraits<IndexType> >;
50public:
51 LikelyDenseUnsignedIntegerSet()
52 : m_size(0)
53 , m_min(0)
54 , m_max(0)
55 {
56 new (NotNull, &m_inline.bitVector) BitVector;
57 }
58
59 ~LikelyDenseUnsignedIntegerSet()
60 {
61 if (isBitVector())
62 m_inline.bitVector.~BitVector();
63 else
64 m_inline.hashSet.~Set();
65 }
66
67 LikelyDenseUnsignedIntegerSet(LikelyDenseUnsignedIntegerSet&& other)
68 : m_size(other.m_size)
69 , m_min(other.m_min)
70 , m_max(other.m_max)
71 {
72 if (isBitVector())
73 new (NotNull, &m_inline.bitVector) BitVector(WTFMove(other.m_inline.bitVector));
74 else
75 new (NotNull, &m_inline.hashSet) Set(WTFMove(other.m_inline.hashSet));
76 }
77
78 void clear()
79 {
80 if (isBitVector())
81 m_inline.bitVector.~BitVector();
82 else
83 m_inline.hashSet.~HashSet();
84 new (NotNull, &m_inline.bitVector) BitVector;
85 m_size = 0;
86 m_min = 0;
87 m_max = 0;
88 }
89
90 bool contains(IndexType value) const
91 {
92 ASSERT(isValidValue(value));
93
94 if (isBitVector()) {
95 if (m_min > value)
96 return false;
97 return m_inline.bitVector.get(value - m_min);
98 }
99
100 return m_inline.hashSet.contains(value);
101 }
102
103 // Providing an iterator in this would be possible, but none of our clients need it.
104 struct AddResult {
105 bool isNewEntry;
106 };
107
108 AddResult add(IndexType value)
109 {
110 ASSERT(isValidValue(value));
111
112 if (!m_size) {
113 ASSERT(isBitVector());
114 m_min = value;
115 m_max = value;
116 m_size = 1;
117 m_inline.bitVector.quickSet(0);
118 return { true };
119 }
120
121 if (!isBitVector()) {
122 bool isNewEntry = m_inline.hashSet.add(value).isNewEntry;
123 if (!isNewEntry)
124 return { false };
125 m_min = std::min(m_min, value);
126 m_max = std::max(m_max, value);
127 unsigned hashSetSize = m_inline.hashSet.capacity() * sizeof(IndexType);
128 unsigned wouldBeBitVectorSize = (m_max - m_min) / 8;
129 if (wouldBeBitVectorSize * 2 < hashSetSize)
130 transitionToBitVector();
131 return { true };
132 }
133
134 if (value >= m_min && value <= m_max) {
135 bool isNewEntry = !m_inline.bitVector.quickSet(value - m_min);
136 m_size += isNewEntry;
137 return { isNewEntry };
138 }
139
140 // We are in BitVector mode, and value is not in the bounds: we will definitely insert it as a new entry.
141 ++m_size;
142
143 IndexType newMin = std::min(m_min, value);
144 IndexType newMax = std::max(m_max, value);
145 unsigned bitVectorSize = (newMax - newMin) / 8;
146 unsigned wouldBeHashSetSize = estimateHashSetSize(m_size);
147 if (wouldBeHashSetSize * 2 < bitVectorSize) {
148 transitionToHashSet();
149 auto result = m_inline.hashSet.add(value);
150 ASSERT_UNUSED(result, result.isNewEntry);
151 m_min = newMin;
152 m_max = newMax;
153 return { true };
154 }
155
156 if (m_min <= value) {
157 bool isNewEntry = !m_inline.bitVector.set(value - m_min);
158 ASSERT_UNUSED(isNewEntry, isNewEntry);
159 ASSERT(newMax == value);
160 m_max = value;
161 return { true };
162 }
163
164 BitVector newBitVector;
165 newBitVector.resize(m_max - value + 1);
166 IndexType shiftReduction = m_min - value;
167 for (IndexType oldIndex : m_inline.bitVector)
168 newBitVector.quickSet(oldIndex + shiftReduction);
169 newBitVector.quickSet(0);
170 m_inline.bitVector = WTFMove(newBitVector);
171 ASSERT(newMin == value);
172 m_min = value;
173 return { true };
174 }
175
176 unsigned size() const
177 {
178 if (isBitVector())
179 return m_size;
180 return m_inline.hashSet.size();
181 }
182
183 class iterator {
184 WTF_MAKE_FAST_ALLOCATED;
185 public:
186 iterator(BitVector::iterator it, IndexType shift)
187 : m_underlying({ it })
188 , m_shift(shift)
189 {
190 }
191 iterator(typename Set::iterator it, IndexType shift)
192 : m_underlying({ it })
193 , m_shift(shift)
194 {
195 }
196
197 iterator& operator++()
198 {
199 WTF::switchOn(m_underlying,
200 [](BitVector::iterator& it) { ++it; },
201 [](typename Set::iterator& it) { ++it; });
202 return *this;
203 }
204
205 IndexType operator*() const
206 {
207 return WTF::switchOn(m_underlying,
208 [&](const BitVector::iterator& it) { return *it + m_shift; },
209 [](const typename Set::iterator& it) { return *it; });
210 }
211
212 bool operator==(const iterator& other) const
213 {
214 return m_underlying == other.m_underlying && m_shift == other.m_shift;
215 }
216 bool operator!=(const iterator& other) const
217 {
218 return m_underlying != other.m_underlying || m_shift != other.m_shift;
219 }
220
221 private:
222 WTF::Variant<BitVector::iterator, typename Set::iterator> m_underlying;
223 IndexType m_shift;
224 };
225
226 iterator begin() const
227 {
228 if (isBitVector())
229 return { m_inline.bitVector.begin(), m_min };
230 return { m_inline.hashSet.begin(), m_min };
231 }
232 iterator end() const
233 {
234 if (isBitVector())
235 return { m_inline.bitVector.end(), m_min };
236 return { m_inline.hashSet.end(), m_min };
237 }
238
239 unsigned memoryUse() const
240 {
241 unsigned result = sizeof(LikelyDenseUnsignedIntegerSet);
242 if (isBitVector())
243 result += m_inline.bitVector.outOfLineMemoryUse();
244 else
245 result += m_inline.hashSet.capacity() * sizeof(IndexType);
246 return result;
247 }
248
249 void validate() const
250 {
251 IndexType min = std::numeric_limits<IndexType>::max(), max = 0;
252 if (isBitVector()) {
253 unsigned numElements = 0;
254 for (IndexType shiftedIndex : m_inline.bitVector) {
255 ++numElements;
256 IndexType correctedIndex = m_min + shiftedIndex;
257 min = std::min(min, correctedIndex);
258 max = std::max(max, correctedIndex);
259 }
260 RELEASE_ASSERT(m_size == numElements);
261 } else {
262 for (IndexType index : m_inline.hashSet) {
263 min = std::min(min, index);
264 max = std::max(max, index);
265 }
266 }
267 if (m_size) {
268 RELEASE_ASSERT(m_min == min);
269 RELEASE_ASSERT(m_max == max);
270 }
271 }
272
273private:
274 bool isBitVector() const { return m_size != std::numeric_limits<unsigned>::max(); }
275
276 bool isValidValue(IndexType value) const
277 {
278 return value != UnsignedWithZeroKeyHashTraits<IndexType>::emptyValue()
279 && !UnsignedWithZeroKeyHashTraits<IndexType>::isDeletedValue(value);
280 }
281
282 unsigned estimateHashSetSize(unsigned n)
283 {
284 unsigned hashSetEstimatedOccupancyOverhead = 3; // max occupancy for a large HashSet is 50%
285 unsigned wouldBeHashSetCapacity = std::max(8u, n) * hashSetEstimatedOccupancyOverhead;
286 return wouldBeHashSetCapacity * sizeof(IndexType);
287 }
288
289 void transitionToHashSet()
290 {
291 ASSERT(isBitVector());
292
293 Set newSet;
294 newSet.reserveInitialCapacity(m_size + 1);
295 for (IndexType oldIndex : m_inline.bitVector)
296 newSet.add(oldIndex + m_min);
297 m_inline.bitVector.~BitVector();
298 new (NotNull, &m_inline.hashSet) Set(WTFMove(newSet));
299 m_size = std::numeric_limits<unsigned>::max();
300
301 ASSERT(!isBitVector());
302 }
303
304 void transitionToBitVector()
305 {
306 ASSERT(!isBitVector());
307
308 BitVector newBitVector;
309 newBitVector.ensureSize(m_max - m_min + 1);
310 m_size = 0;
311 for (IndexType oldValue : m_inline.hashSet) {
312 newBitVector.quickSet(oldValue - m_min);
313 ++m_size;
314 }
315 ASSERT(newBitVector.quickGet(0));
316 m_inline.hashSet.~Set();
317 new (NotNull, &m_inline.bitVector) BitVector(WTFMove(newBitVector));
318
319 ASSERT(isBitVector());
320 }
321
322 union U {
323 BitVector bitVector;
324 Set hashSet;
325
326 U() { }
327 ~U() { }
328 } m_inline;
329 unsigned m_size; // Only updated in BitVector mode, std::numeric_limits<unsigned>::max() to indicate HashSet mode
330 IndexType m_min;
331 IndexType m_max;
332
333};
