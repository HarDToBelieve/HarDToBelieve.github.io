41typedef LockAlgorithm<uint8_t, 1, 2> DefaultLockAlgorithm;
42
43// This is a fully adaptive mutex that only requires 1 byte of storage. It has fast paths that are
44// competetive to a spinlock (uncontended locking is inlined and is just a CAS, microcontention is
45// handled by spinning and yielding), and a slow path that is competetive to std::mutex (if a lock
46// cannot be acquired in a short period of time, the thread is put to sleep until the lock is
47// available again). It uses less memory than a std::mutex. This lock guarantees eventual stochastic
48// fairness, even in programs that relock the lock immediately after unlocking it. Except when there
49// are collisions between this lock and other locks in the ParkingLot, this lock will guarantee that
50// at worst one call to unlock() per millisecond will do a direct hand-off to the thread that is at
51// the head of the queue. When there are collisions, each collision increases the fair unlock delay
52// by one millisecond in the worst case.
53//
54// This lock type supports thread safety analysis.
55// To annotate a member variable or a global variable with thread ownership information,
56// use lock capability annotations defined in ThreadSafetyAnalysis.h.
57class WTF_CAPABILITY_LOCK Lock {
58 WTF_MAKE_NONCOPYABLE(Lock);
59 WTF_MAKE_FAST_ALLOCATED;
60public:
61 constexpr Lock() = default;
62
63 void lock() WTF_ACQUIRES_LOCK()
64 {
65 if (UNLIKELY(!DefaultLockAlgorithm::lockFastAssumingZero(m_byte)))
66 lockSlow();
67 }
68
69 bool tryLock() WTF_ACQUIRES_LOCK_IF(true) // NOLINT: Intentional deviation to support std::scoped_lock.
70 {
71 return DefaultLockAlgorithm::tryLock(m_byte);
72 }
73
74 // Need this version for std::unique_lock.
75 bool try_lock() WTF_ACQUIRES_LOCK_IF(true)
76 {
77 return tryLock();
78 }
79
80 WTF_EXPORT_PRIVATE bool tryLockWithTimeout(Seconds timeout) WTF_ACQUIRES_LOCK_IF(true);
81
82 // Relinquish the lock. Either one of the threads that were waiting for the lock, or some other
83 // thread that happens to be running, will be able to grab the lock. This bit of unfairness is
84 // called barging, and we allow it because it maximizes throughput. However, we bound how unfair
85 // barging can get by ensuring that every once in a while, when there is a thread waiting on the
86 // lock, we hand the lock to that thread directly. Every time unlock() finds a thread waiting,
87 // we check if the last time that we did a fair unlock was more than roughly 1ms ago; if so, we
88 // unlock fairly. Fairness matters most for long critical sections, and this virtually
89 // guarantees that long critical sections always get a fair lock.
90 void unlock() WTF_RELEASES_LOCK()
91 {
92 if (UNLIKELY(!DefaultLockAlgorithm::unlockFastAssumingZero(m_byte)))
93 unlockSlow();
94 }
95
96 // This is like unlock() but it guarantees that we unlock the lock fairly. For short critical
97 // sections, this is much slower than unlock(). For long critical sections, unlock() will learn
98 // to be fair anyway. However, if you plan to relock the lock right after unlocking and you want
99 // to ensure that some other thread runs in the meantime, this is probably the function you
100 // want.
101 void unlockFairly() WTF_RELEASES_LOCK()
102 {
103 if (UNLIKELY(!DefaultLockAlgorithm::unlockFastAssumingZero(m_byte)))
104 unlockFairlySlow();
105 }
106
107 void safepoint()
108 {
109 if (UNLIKELY(!DefaultLockAlgorithm::safepointFast(m_byte)))
110 safepointSlow();
111 }
112
113 bool isHeld() const
114 {
115 return DefaultLockAlgorithm::isLocked(m_byte);
116 }
117
118 bool isLocked() const
119 {
120 return isHeld();
121 }
122
123private:
124 friend struct TestWebKitAPI::LockInspector;
125
126 static constexpr uint8_t isHeldBit = 1;
127 static constexpr uint8_t hasParkedBit = 2;
128
129 WTF_EXPORT_PRIVATE void lockSlow();
130 WTF_EXPORT_PRIVATE void unlockSlow();
131 WTF_EXPORT_PRIVATE void unlockFairlySlow();
132 WTF_EXPORT_PRIVATE void safepointSlow();
133
134 // Method used for testing only.
135 bool isFullyReset() const
136 {
137 return !m_byte.load();
138 }
139
140 Atomic<uint8_t> m_byte { 0 };
141};
142
143// Asserts that the lock is held.
144// This can be used in cases where the annotations cannot be added to the function
145// declaration.
146inline void assertIsHeld(const Lock& lock) WTF_ASSERTS_ACQUIRED_LOCK(lock) { ASSERT_UNUSED(lock, lock.isHeld()); }

148// Locker specialization to use with Lock.
149// Non-movable simple scoped lock holder.
150// Example: Locker locker { m_lock };
151template <>
152class WTF_CAPABILITY_SCOPED_LOCK Locker<Lock> : public AbstractLocker {
153public:
154 explicit Locker(Lock& lock) WTF_ACQUIRES_LOCK(lock)
155 : m_lock(lock)
156 , m_isLocked(true)
157 {
158 m_lock.lock();
159 }
160 Locker(AdoptLockTag, Lock& lock) WTF_REQUIRES_LOCK(lock)
161 : m_lock(lock)
162 , m_isLocked(true)
163 {
164 }
165 ~Locker() WTF_RELEASES_LOCK()
166 {
167 if (m_isLocked)
168 m_lock.unlock();
169 }
170 void unlockEarly() WTF_RELEASES_LOCK()
171 {
172 ASSERT(m_isLocked);
173 m_isLocked = false;
174 m_lock.unlock();
175 }
176 Locker(const Locker<Lock>&) = delete;
177 Locker& operator=(const Locker<Lock>&) = delete;
178
179private:
180 // Support DropLockForScope even though it doesn't support thread safety analysis.
181 template<typename>
182 friend class DropLockForScope;
183
184 void lock() WTF_ACQUIRES_LOCK(m_lock)
185 {
186 m_lock.lock();
187 compilerFence();
188 }
189
190 void unlock() WTF_RELEASES_LOCK(m_lock)
191 {
192 compilerFence();
193 m_lock.unlock();
194 }
195
196 Lock& m_lock;
197 bool m_isLocked { false };
198};
199
200Locker(Lock&) -> Locker<Lock>;
201Locker(AdoptLockTag, Lock&) -> Locker<Lock>;
