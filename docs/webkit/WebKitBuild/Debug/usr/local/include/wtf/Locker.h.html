41enum NoLockingNecessaryTag { NoLockingNecessary };
42
43class AbstractLocker {
44 WTF_MAKE_NONCOPYABLE(AbstractLocker);
45public:
46 AbstractLocker(NoLockingNecessaryTag)
47 {
48 }
49
50protected:
51 AbstractLocker()
52 {
53 }
54};
55
56template<typename T> class DropLockForScope;
57
58using AdoptLockTag = std::adopt_lock_t;
59constexpr AdoptLockTag AdoptLock;
60
61template<typename T>
62class Locker : public AbstractLocker {
63public:
64 explicit Locker(T& lockable) : m_lockable(&lockable) { lock(); }
65 explicit Locker(T* lockable) : m_lockable(lockable) { lock(); }
66 explicit Locker(AdoptLockTag, T& lockable) : m_lockable(&lockable) { }
67
68 // You should be wary of using this constructor. It's only applicable
69 // in places where there is a locking protocol for a particular object
70 // but it's not necessary to engage in that protocol yet. For example,
71 // this often happens when an object is newly allocated and it can not
72 // be accessed concurrently.
73 Locker(NoLockingNecessaryTag) : m_lockable(nullptr) { }
74
75 Locker(std::underlying_type_t<NoLockingNecessaryTag>) = delete;
76
77 ~Locker()
78 {
79 unlock();
80 }
81
82 static Locker tryLock(T& lockable)
83 {
84 Locker result(NoLockingNecessary);
85 if (lockable.tryLock()) {
86 result.m_lockable = &lockable;
87 return result;
88 }
89 return result;
90 }
91
92 T* lockable() { return m_lockable; }
93
94 explicit operator bool() const { return !!m_lockable; }
95
96 void unlockEarly()
97 {
98 unlock();
99 m_lockable = nullptr;
100 }
101
102 // It's great to be able to pass lockers around. It enables custom locking adaptors like
103 // JSC::LockDuringMarking.
104 Locker(Locker&& other)
105 : m_lockable(other.m_lockable)
106 {
107 ASSERT(&other != this);
108 other.m_lockable = nullptr;
109 }
110
111 Locker& operator=(Locker&& other)
112 {
113 ASSERT(&other != this);
114 m_lockable = other.m_lockable;
115 other.m_lockable = nullptr;
116 return *this;
117 }
118
119private:
120 template<typename>
121 friend class DropLockForScope;
122
123 void unlock()
124 {
125 compilerFence();
126 if (m_lockable) {
127 TSAN_ANNOTATE_HAPPENS_BEFORE(m_lockable);
128 m_lockable->unlock();
129 }
130 }
131
132 void lock()
133 {
134 if (m_lockable) {
135 m_lockable->lock();
136 TSAN_ANNOTATE_HAPPENS_AFTER(m_lockable);
137 }
138 compilerFence();
139 }
140
141 T* m_lockable;
142};
143
144template<typename LockType>
145class DropLockForScope : private AbstractLocker {
146 WTF_FORBID_HEAP_ALLOCATION(DropLockForScope);
147public:
148 DropLockForScope(Locker<LockType>& lock)
149 : m_lock(lock)
150 {
151 m_lock.unlock();
152 }
153
154 ~DropLockForScope()
155 {
156 m_lock.lock();
157 }
158
159private:
160 Locker<LockType>& m_lock;
161};

163// This is a close replica of Locker, but for generic lock/unlock functions.
164template<typename T, void (lockFunction)(T*), void (*unlockFunction)(T*)>
165class ExternalLocker: public WTF::AbstractLocker {
166public:
167 explicit ExternalLocker(T* lockable)
168 : m_lockable(lockable)
169 {
170 ASSERT(lockable);
171 lock();
172 }
173
174 ~ExternalLocker()
175 {
176 unlock();
177 }
178
179 T* lockable() { return m_lockable; }
180
181 explicit operator bool() const { return !!m_lockable; }
182
183 void unlockEarly()
184 {
185 unlock();
186 m_lockable = nullptr;
187 }
188
189 ExternalLocker(ExternalLocker&& other)
190 : m_lockable(other.m_lockable)
191 {
192 ASSERT(&other != this);
193 other.m_lockable = nullptr;
194 }
195
196 ExternalLocker& operator=(ExternalLocker&& other)
197 {
198 ASSERT(&other != this);
199 m_lockable = other.m_lockable;
200 other.m_lockable = nullptr;
201 return *this;
202 }
203
204private:
205 template<typename>
206 friend class DropLockForScope;
207
208 void unlock()
209 {
210 if (m_lockable)
211 unlockFunction(m_lockable);
212 }
213
214 void lock()
215 {
216 if (m_lockable)
217 lockFunction(m_lockable);
218 }
219
220 T* m_lockable;
221};
