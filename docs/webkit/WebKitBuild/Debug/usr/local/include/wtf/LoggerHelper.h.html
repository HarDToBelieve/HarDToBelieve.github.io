26#pragma once

28#include <wtf/CryptographicallyRandomNumber.h>
29#include <wtf/Logger.h>
30
31namespace WTF {
32
33class LoggerHelper {
34public:
35 virtual ~LoggerHelper() = default;
36
37 virtual const Logger& logger() const = 0;
38 virtual const char* logClassName() const = 0;
39 virtual WTFLogChannel& logChannel() const = 0;
40 virtual const void* logIdentifier() const = 0;
41
42#if !RELEASE_LOG_DISABLED
43
44#define LOGIDENTIFIER WTF::Logger::LogSiteIdentifier(logClassName(), __func__, logIdentifier())
45
46#if VERBOSE_RELEASE_LOG
47#define ALWAYS_LOG(...) logger().logAlwaysVerbose(logChannel(), __FILE__, __func__, __LINE__, __VA_ARGS__)
48#define ERROR_LOG(...) logger().errorVerbose(logChannel(), __FILE__, __func__, __LINE__, __VA_ARGS__)
49#define WARNING_LOG(...) logger().warningVerbose(logChannel(), __FILE__, __func__, __LINE__, __VA_ARGS__)
50#define INFO_LOG(...) logger().infoVerbose(logChannel(), __FILE__, __func__, __LINE__, __VA_ARGS__)
51#define DEBUG_LOG(...) logger().debugVerbose(logChannel(), __FILE__, __func__, __LINE__, __VA_ARGS__)
52#else
53#define ALWAYS_LOG(...) logger().logAlways(logChannel(), __VA_ARGS__)
54#define ERROR_LOG(...) logger().error(logChannel(), __VA_ARGS__)
55#define WARNING_LOG(...) logger().warning(logChannel(), __VA_ARGS__)
56#define INFO_LOG(...) logger().info(logChannel(), __VA_ARGS__)
57#define DEBUG_LOG(...) logger().debug(logChannel(), __VA_ARGS__)
58#endif
59
60#define WILL_LOG(_level_) logger().willLog(logChannel(), _level_)
61
62#define ALWAYS_LOG_IF(condition, ...) if (condition) ALWAYS_LOG(__VA_ARGS__)
63#define ERROR_LOG_IF(condition, ...) if (condition) ERROR_LOG(__VA_ARGS__)
64#define WARNING_LOG_IF(condition, ...) if (condition) WARNING_LOG(__VA_ARGS__)
65#define INFO_LOG_IF(condition, ...) if (condition) INFO_LOG(__VA_ARGS__)
66#define DEBUG_LOG_IF(condition, ...) if (condition) DEBUG_LOG(__VA_ARGS__)
67
68#define ALWAYS_LOG_IF_POSSIBLE(...) if (loggerPtr()) loggerPtr()->logAlways(logChannel(), __VA_ARGS__)
69#define ERROR_LOG_IF_POSSIBLE(...) if (loggerPtr()) loggerPtr()->error(logChannel(), __VA_ARGS__)
70#define WARNING_LOG_IF_POSSIBLE(...) if (loggerPtr()) loggerPtr()->warning(logChannel(), __VA_ARGS__)
71#define INFO_LOG_IF_POSSIBLE(...) if (loggerPtr()) loggerPtr()->info(logChannel(), __VA_ARGS__)
72#define DEBUG_LOG_IF_POSSIBLE(...) if (loggerPtr()) loggerPtr()->debug(logChannel(), __VA_ARGS__)
73#define WILL_LOG_IF_POSSIBLE(_level_) if (loggerPtr()) loggerPtr()->willLog(logChannel(), _level_)
74
75 static const void* childLogIdentifier(const void* parentIdentifier, uint64_t childIdentifier)
76 {
77 static constexpr uint64_t parentMask = 0xffffffffffff0000ull;
78 static constexpr uint64_t maskLowerWord = 0xffffull;
79 return reinterpret_cast<const void*>((bitwise_cast<uintptr_t>(parentIdentifier) & parentMask) | (childIdentifier & maskLowerWord));
80 }
81
82 static const void* uniqueLogIdentifier()
83 {
84 uint64_t highWord = cryptographicallyRandomNumber();
85 uint64_t lowWord = cryptographicallyRandomNumber();
86 return reinterpret_cast<const void*>((highWord << 32) + lowWord);
87 }
88#else // RELEASE_LOG_DISABLED
89
90#define LOGIDENTIFIER (std::nullopt)
91
92#define ALWAYS_LOG(channelName, ...) (UNUSED_PARAM(channelName))
93#define ERROR_LOG(channelName, ...) (UNUSED_PARAM(channelName))
94#define WARNING_LOG(channelName, ...) (UNUSED_PARAM(channelName))
95#define NOTICE_LOG(channelName, ...) (UNUSED_PARAM(channelName))
96#define INFO_LOG(channelName, ...) (UNUSED_PARAM(channelName))
97#define DEBUG_LOG(channelName, ...) (UNUSED_PARAM(channelName))
98#define WILL_LOG(_level_) false
99
100#define ALWAYS_LOG_IF(condition, ...) ((void)0)
101#define ERROR_LOG_IF(condition, ...) ((void)0)
102#define WARNING_LOG_IF(condition, ...) ((void)0)
103#define INFO_LOG_IF(condition, ...) ((void)0)
104#define DEBUG_LOG_IF(condition, ...) ((void)0)
105
106#define ALWAYS_LOG_IF_POSSIBLE(...) ((void)0)
107#define ERROR_LOG_IF_POSSIBLE(...) ((void)0)
108#define WARNING_LOG_IF_POSSIBLE(...) ((void)0)
109#define INFO_LOG_IF_POSSIBLE(...) ((void)0)
110#define DEBUG_LOG_IF_POSSIBLE(...) ((void)0)
111#define WILL_LOG_IF_POSSIBLE(_level_) ((void)0)
112
113#endif // RELEASE_LOG_DISABLED
114
115};
116
117} // namespace WTF
