76// Work around a bug in Win, where atan2(+-infinity, +-infinity) yields NaN instead of specific values.
77extern "C" inline double wtf_atan2(double x, double y)
78{
79 double posInf = std::numeric_limits<double>::infinity();
80 double negInf = -std::numeric_limits<double>::infinity();
81 double nan = std::numeric_limits<double>::quiet_NaN();
82
83 double result = nan;
84
85 if (x == posInf && y == posInf)
86 result = piOverFourDouble;
87 else if (x == posInf && y == negInf)
88 result = 3 * piOverFourDouble;
89 else if (x == negInf && y == posInf)
90 result = -piOverFourDouble;
91 else if (x == negInf && y == negInf)
92 result = -3 * piOverFourDouble;
93 else
94 result = ::atan2(x, y);
95
96 return result;
97}

148clampTo(SourceType value, TargetType min = defaultMinimumForClamp<TargetType>(), TargetType max = defaultMaximumForClamp<TargetType>())

157// Floating point source.
158template<typename TargetType, typename SourceType>
159typename std::enable_if<!std::is_same<TargetType, SourceType>::value
160 && std::is_floating_point<SourceType>::value
161 && !(std::is_floating_point<TargetType>::value && sizeof(TargetType) > sizeof(SourceType)), TargetType>::type
162clampTo(SourceType value, TargetType min = defaultMinimumForClamp<TargetType>(), TargetType max = defaultMaximumForClamp<TargetType>())
163{
164 if (value >= static_cast<SourceType>(max))
165 return max;
166 // This will return min if value is NaN.
167 if (!(value > static_cast<SourceType>(min)))
168 return min;
169 return static_cast<TargetType>(value);
170}

393// decompose 'number' to its sign, exponent, and mantissa components.
394// The result is interpreted as:
395// (sign ? -1 : 1) * pow(2, exponent) * (mantissa / (1 << 52))
396inline void decomposeDouble(double number, bool& sign, int32_t& exponent, uint64_t& mantissa)
397{
398 ASSERT(std::isfinite(number));
399
400 sign = std::signbit(number);
401
402 uint64_t bits = WTF::bitwise_cast<uint64_t>(number);
403 exponent = (static_cast<int32_t>(bits >> 52) & 0x7ff) - 0x3ff;
404 mantissa = bits & 0xFFFFFFFFFFFFFull;
405
406 // Check for zero/denormal values; if so, adjust the exponent,
407 // if not insert the implicit, omitted leading 1 bit.
408 if (exponent == -0x3ff)
409 exponent = mantissa ? -0x3fe : 0;
410 else
411 mantissa |= 0x10000000000000ull;
412}

427// Calculate d % 2^{64}.
428inline void doubleToInteger(double d, unsigned long long& value)
429{
430 if (std::isnan(d) || std::isinf(d))
431 value = 0;
432 else {
433 // -2^{64} < fmodValue < 2^{64}.
434 double fmodValue = fmod(trunc(d), maxPlusOne<unsigned long long>);
435 if (fmodValue >= 0) {
436 // 0 <= fmodValue < 2^{64}.
437 // 0 <= value < 2^{64}. This cast causes no loss.
438 value = static_cast<unsigned long long>(fmodValue);
439 } else {
440 // -2^{64} < fmodValue < 0.
441 // 0 < fmodValueInUnsignedLongLong < 2^{64}. This cast causes no loss.
442 unsigned long long fmodValueInUnsignedLongLong = static_cast<unsigned long long>(-fmodValue);
443 // -1 < (std::numeric_limits<unsigned long long>::max() - fmodValueInUnsignedLongLong) < 2^{64} - 1.
444 // 0 < value < 2^{64}.
445 value = std::numeric_limits<unsigned long long>::max() - fmodValueInUnsignedLongLong + 1;
446 }
447 }

452// From http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
453constexpr uint32_t roundUpToPowerOfTwo(uint32_t v)
454{
455 v--;
456 v |= v >> 1;
457 v |= v >> 2;
458 v |= v >> 4;
459 v |= v >> 8;
460 v |= v >> 16;
461 v++;
462 return v;
463}

472inline unsigned fastLog2(unsigned i)
473{
474 unsigned log2 = 0;
475 if (i & (i - 1))
476 log2 += 1;
477 if (i >> 16) {
478 log2 += 16;
479 i >>= 16;
480 }
481 if (i >> 8) {
482 log2 += 8;
483 i >>= 8;
484 }
485 if (i >> 4) {
486 log2 += 4;
487 i >>= 4;
488 }
489 if (i >> 2) {
490 log2 += 2;
491 i >>= 2;
492 }
493 if (i >> 1)
494 log2 += 1;
495 return log2;
496}

506template <typename T>
507inline typename std::enable_if<std::is_floating_point<T>::value, T>::type safeFPDivision(T u, T v)
508{
509 // Protect against overflow / underflow.
510 if (v < 1 && u > v * std::numeric_limits<T>::max())
511 return std::numeric_limits<T>::max();
512 if (v > 1 && u < v * std::numeric_limits<T>::min())
513 return 0;
514 return u / v;
515}

517// Floating point numbers comparison:
518// u is "essentially equal" [1][2] to v if: | u - v | / |u| <= e and | u - v | / |v| <= e
519//
520// [1] Knuth, D. E. "Accuracy of Floating Point Arithmetic." The Art of Computer Programming. 3rd ed. Vol. 2.
521// Boston: Addison-Wesley, 1998. 229-45.
522// [2] http://www.boost.org/doc/libs/1_34_0/libs/test/doc/components/test_tools/floating_point_comparison.html
523template <typename T>
524inline typename std::enable_if<std::is_floating_point<T>::value, bool>::type areEssentiallyEqual(T u, T v, T epsilon = std::numeric_limits<T>::epsilon())
525{
526 if (u == v)
527 return true;
528
529 const T delta = std::abs(u - v);
530 return safeFPDivision(delta, std::abs(u)) <= epsilon && safeFPDivision(delta, std::abs(v)) <= epsilon;
531}

559template<typename T>
560inline T leftShiftWithSaturation(T value, unsigned shiftAmount, T max = std::numeric_limits<T>::max())
561{
562 T result = value << shiftAmount;
563 // We will have saturated if shifting right doesn't recover the original value.
564 if (result >> shiftAmount != value)
565 return max;
566 if (result > max)
567 return max;
568 return result;
569}

571// Check if two ranges overlap assuming that neither range is empty.
572template<typename T>
573inline bool nonEmptyRangesOverlap(T leftMin, T leftMax, T rightMin, T rightMax)
574{
575 ASSERT(leftMin < leftMax);
576 ASSERT(rightMin < rightMax);
577
578 return leftMax > rightMin && rightMax > leftMin;
579}

581// Pass ranges with the min being inclusive and the max being exclusive. For example, this should
582// return false:
583//
584// rangesOverlap(0, 8, 8, 16)
585template<typename T>
586inline bool rangesOverlap(T leftMin, T leftMax, T rightMin, T rightMax)
587{
588 ASSERT(leftMin <= leftMax);
589 ASSERT(rightMin <= rightMax);
590
591 // Empty ranges interfere with nothing.
592 if (leftMin == leftMax)
593 return false;
594 if (rightMin == rightMax)
595 return false;
596
597 return nonEmptyRangesOverlap(leftMin, leftMax, rightMin, rightMax);
598}

600template<typename VectorType, typename RandomFunc>
601void shuffleVector(VectorType& vector, size_t size, const RandomFunc& randomFunc)
602{
603 for (size_t i = 0; i + 1 < size; ++i)
604 std::swap(vector[i], vector[i + randomFunc(size - i)]);
605}

613template <typename T>
614constexpr unsigned clzConstexpr(T value)
615{
616 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
617
618 using UT = typename std::make_unsigned<T>::type;
619 UT uValue = value;
620
621 unsigned zeroCount = 0;
622 for (int i = bitSize - 1; i >= 0; i--) {
623 if (uValue >> i)
624 break;
625 zeroCount++;
626 }
627 return zeroCount;
628}

630template<typename T>
631inline unsigned clz(T value)
632{
633 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
634
635 using UT = typename std::make_unsigned<T>::type;
636 UT uValue = value;
637
638#if COMPILER(GCC_COMPATIBLE)
639 constexpr unsigned bitSize64 = sizeof(uint64_t) * CHAR_BIT;
640 if (uValue)
641 return __builtin_clzll(uValue) - (bitSize64 - bitSize);
642 return bitSize;
643#elif COMPILER(MSVC) && !CPU(X86)
644 // Visual Studio 2008 or upper have __lzcnt, but we can't detect Intel AVX at compile time.
645 // So we use bit-scan-reverse operation to calculate clz.
646 // _BitScanReverse64 is defined in X86_64 and ARM in MSVC supported environments.
647 unsigned long ret = 0;
648 if (_BitScanReverse64(&ret, uValue))
649 return bitSize - 1 - ret;
650 return bitSize;
651#else
652 UNUSED_PARAM(bitSize);
653 UNUSED_PARAM(uValue);
654 return clzConstexpr(value);
655#endif
656}

659constexpr unsigned ctzConstexpr(T value)
660{
661 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
662
663 using UT = typename std::make_unsigned<T>::type;
664 UT uValue = value;
665
666 unsigned zeroCount = 0;
667 for (unsigned i = 0; i < bitSize; i++) {
668 if (uValue & 1)
669 break;
670
671 zeroCount++;
672 uValue >>= 1;
673 }
674 return zeroCount;
675}

677template<typename T>
678inline unsigned ctz(T value)
679{
680 constexpr unsigned bitSize = sizeof(T) * CHAR_BIT;
681
682 using UT = typename std::make_unsigned<T>::type;
683 UT uValue = value;
684
685#if COMPILER(GCC_COMPATIBLE)
686 if (uValue)
687 return __builtin_ctzll(uValue);
688 return bitSize;
689#elif COMPILER(MSVC) && !CPU(X86)
690 unsigned long ret = 0;
691 if (_BitScanForward64(&ret, uValue))
692 return ret;
693 return bitSize;
694#else
695 UNUSED_PARAM(bitSize);
696 UNUSED_PARAM(uValue);
697 return ctzConstexpr(value);
698#endif
699}
