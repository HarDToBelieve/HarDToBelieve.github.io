44class WTF_EXPORT_PRIVATE MediaTime final {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 enum {
48 Valid = 1 << 0,
49 HasBeenRounded = 1 << 1,
50 PositiveInfinite = 1 << 2,
51 NegativeInfinite = 1 << 3,
52 Indefinite = 1 << 4,
53 DoubleValue = 1 << 5,
54 };
55
56 constexpr MediaTime();
57 constexpr MediaTime(int64_t value, uint32_t scale, uint8_t flags = Valid);
58 MediaTime(const MediaTime& rhs);
59
60 static MediaTime createWithFloat(float floatTime);
61 static MediaTime createWithFloat(float floatTime, uint32_t timeScale);
62 static MediaTime createWithDouble(double doubleTime);
63 static MediaTime createWithDouble(double doubleTime, uint32_t timeScale);
64
65 float toFloat() const;
66 double toDouble() const;
67
68 MediaTime& operator=(const MediaTime& rhs);
69 MediaTime& operator+=(const MediaTime& rhs) { return *this = *this + rhs; }
70 MediaTime& operator-=(const MediaTime& rhs) { return *this = *this - rhs; }
71 MediaTime operator+(const MediaTime& rhs) const;
72 MediaTime operator-(const MediaTime& rhs) const;
73 MediaTime operator-() const;
74 MediaTime operator*(int32_t) const;
75 bool operator<(const MediaTime& rhs) const { return compare(rhs) == LessThan; }
76 bool operator>(const MediaTime& rhs) const { return compare(rhs) == GreaterThan; }
77 bool operator!=(const MediaTime& rhs) const { return compare(rhs) != EqualTo; }
78 bool operator==(const MediaTime& rhs) const { return compare(rhs) == EqualTo; }
79 bool operator>=(const MediaTime& rhs) const { return compare(rhs) >= EqualTo; }
80 bool operator<=(const MediaTime& rhs) const { return compare(rhs) <= EqualTo; }
81 bool operator!() const;
82 explicit operator bool() const;
83
84 typedef enum {
85 LessThan = -1,
86 EqualTo = 0,
87 GreaterThan = 1,
88 } ComparisonFlags;
89
90 ComparisonFlags compare(const MediaTime& rhs) const;
91 bool isBetween(const MediaTime&, const MediaTime&) const;
92
93 bool isValid() const { return m_timeFlags & Valid; }
94 bool isInvalid() const { return !isValid(); }
95 bool hasBeenRounded() const { return m_timeFlags & HasBeenRounded; }
96 bool isPositiveInfinite() const { return m_timeFlags & PositiveInfinite; }
97 bool isNegativeInfinite() const { return m_timeFlags & NegativeInfinite; }
98 bool isIndefinite() const { return m_timeFlags & Indefinite; }
99 bool isFinite() const { return !isInvalid() && !isIndefinite() && !isPositiveInfinite() && !isNegativeInfinite(); }
100 bool hasDoubleValue() const { return m_timeFlags & DoubleValue; }
101 uint8_t timeFlags() const { return m_timeFlags; }
102
103 static const MediaTime& zeroTime();
104 static const MediaTime& invalidTime();
105 static const MediaTime& positiveInfiniteTime();
106 static const MediaTime& negativeInfiniteTime();
107 static const MediaTime& indefiniteTime();
108
109 const int64_t& timeValue() const { return m_timeValue; }
110 const uint32_t& timeScale() const { return m_timeScale; }
111
112 void dump(PrintStream& out) const;
113 String toString() const;
114 String toJSONString() const;
115 Ref<JSON::Object> toJSONObject() const;
116
117 // Make the following casts errors:
118 operator double() const = delete;
119 MediaTime(double) = delete;
120 operator int() const = delete;
121 MediaTime(int) = delete;
122
123 friend WTF_EXPORT_PRIVATE MediaTime abs(const MediaTime& rhs);

125 static const uint32_t DefaultTimeScale = 10000000;
126 static const uint32_t MaximumTimeScale;
127
128 enum class RoundingFlags {
129 HalfAwayFromZero = 0,
130 TowardZero,
131 AwayFromZero,
132 TowardPositiveInfinity,
133 TowardNegativeInfinity,
134 };
135
136 MediaTime toTimeScale(uint32_t, RoundingFlags = RoundingFlags::HalfAwayFromZero) const;
137
138 template<class Encoder> void encode(Encoder&) const;
139 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, MediaTime&);
140
141private:
142 void setTimeScale(uint32_t, RoundingFlags = RoundingFlags::HalfAwayFromZero);
143
144 union {
145 int64_t m_timeValue;
146 double m_timeValueAsDouble;
147 };
148 uint32_t m_timeScale;
149 uint8_t m_timeFlags;
150};
151
152constexpr MediaTime::MediaTime()
153 : m_timeValue(0)
154 , m_timeScale(DefaultTimeScale)
155 , m_timeFlags(Valid)
156{
157}
158
159constexpr MediaTime::MediaTime(int64_t value, uint32_t scale, uint8_t flags)
160 : m_timeValue(value)
161 , m_timeScale(scale)
162 , m_timeFlags(flags)
163{
164 if (scale || !(flags & Valid))
165 return;
166
167 *this = value < 0 ? negativeInfiniteTime() : positiveInfiniteTime();
168}
169
170inline MediaTime operator*(int32_t lhs, const MediaTime& rhs) { return rhs.operator*(lhs); }
171
172WTF_EXPORT_PRIVATE extern MediaTime abs(const MediaTime& rhs);

174struct WTF_EXPORT_PRIVATE MediaTimeRange {
175 WTF_MAKE_STRUCT_FAST_ALLOCATED;
176
177 String toJSONString() const;
178
179 const MediaTime start;
180 const MediaTime end;
181};
182
183template<class Encoder>
184void MediaTime::encode(Encoder& encoder) const
185{
186 encoder << m_timeValue << m_timeScale << m_timeFlags;
187}
188
189template<class Decoder>
190bool MediaTime::decode(Decoder& decoder, MediaTime& time)
191{
192 return decoder.decode(time.m_timeValue)
193 && decoder.decode(time.m_timeScale)
194 && decoder.decode(time.m_timeFlags);
195}
