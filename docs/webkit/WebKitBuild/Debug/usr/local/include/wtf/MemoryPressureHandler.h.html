45
46enum class MemoryUsagePolicy : uint8_t {
47 Unrestricted, // Allocate as much as you want
48 Conservative, // Maybe you don't cache every single thing
49 Strict, // Time to start pinching pennies for real
50};
51
52enum class WebsamProcessState : uint8_t {
53 Active,
54 Inactive,
55};
56
57enum class Critical : uint8_t { No, Yes };
58enum class Synchronous : uint8_t { No, Yes };
59
60typedef WTF::Function<void(Critical, Synchronous)> LowMemoryHandler;
61
62class MemoryPressureHandler {
63 WTF_MAKE_FAST_ALLOCATED;
64 friend class WTF::LazyNeverDestroyed<MemoryPressureHandler>;
65public:
66 WTF_EXPORT_PRIVATE static MemoryPressureHandler& singleton();
67
68 WTF_EXPORT_PRIVATE void install();
69
70 WTF_EXPORT_PRIVATE void setShouldUsePeriodicMemoryMonitor(bool);
71
72#if OS(LINUX) || OS(FREEBSD)
73 WTF_EXPORT_PRIVATE void triggerMemoryPressureEvent(bool isCritical);
74#endif
75
76 void setMemoryKillCallback(WTF::Function<void()>&& function) { m_memoryKillCallback = WTFMove(function); }
77 void setMemoryPressureStatusChangedCallback(WTF::Function<void(bool)>&& function) { m_memoryPressureStatusChangedCallback = WTFMove(function); }
78 void setDidExceedInactiveLimitWhileActiveCallback(WTF::Function<void()>&& function) { m_didExceedInactiveLimitWhileActiveCallback = WTFMove(function); }
79
80 void setLowMemoryHandler(LowMemoryHandler&& handler)
81 {
82 m_lowMemoryHandler = WTFMove(handler);
83 }
84
85 bool isUnderMemoryPressure() const
86 {
87 return m_underMemoryPressure
88#if PLATFORM(MAC)
89 || m_memoryUsagePolicy >= MemoryUsagePolicy::Strict
90#endif
91 || m_isSimulatingMemoryPressure;
92 }
93 bool isSimulatingMemoryPressure() const { return m_isSimulatingMemoryPressure; }
94 void setUnderMemoryPressure(bool);
95
96 WTF_EXPORT_PRIVATE static MemoryUsagePolicy currentMemoryUsagePolicy();
97
98#if PLATFORM(COCOA)
99 WTF_EXPORT_PRIVATE void setDispatchQueue(OSObjectPtr<dispatch_queue_t>&&);
100#endif
101
102 class ReliefLogger {
103 WTF_MAKE_FAST_ALLOCATED;
104 public:
105 explicit ReliefLogger(const char *log)
106 : m_logString(log)
107 , m_initialMemory(loggingEnabled() ? platformMemoryUsage() : MemoryUsage { })
108 {
109 }
110
111 ~ReliefLogger()
112 {
113 if (loggingEnabled())
114 logMemoryUsageChange();
115 }
116
117
118 const char* logString() const { return m_logString; }
119 static void setLoggingEnabled(bool enabled) { s_loggingEnabled = enabled; }
120 static bool loggingEnabled()
121 {
122#if RELEASE_LOG_DISABLED
123 return s_loggingEnabled;
124#else
125 return true;
126#endif
127 }
128
129 private:
130 struct MemoryUsage {
131 WTF_MAKE_STRUCT_FAST_ALLOCATED;
132 MemoryUsage() = default;
133 MemoryUsage(size_t resident, size_t physical)
134 : resident(resident)
135 , physical(physical)
136 {
137 }
138 size_t resident { 0 };
139 size_t physical { 0 };
140 };
141 std::optional<MemoryUsage> platformMemoryUsage();
142 void logMemoryUsageChange();
143
144 const char* m_logString;
145 std::optional<MemoryUsage> m_initialMemory;
146
147 WTF_EXPORT_PRIVATE static bool s_loggingEnabled;
148 };
149
150 WTF_EXPORT_PRIVATE void releaseMemory(Critical, Synchronous = Synchronous::No);
151
152 WTF_EXPORT_PRIVATE void beginSimulatedMemoryPressure();
153 WTF_EXPORT_PRIVATE void endSimulatedMemoryPressure();
154
155 WTF_EXPORT_PRIVATE void setProcessState(WebsamProcessState);
156 WebsamProcessState processState() const { return m_processState; }
157
158 WTF_EXPORT_PRIVATE static void setPageCount(unsigned);
159
160 void setShouldLogMemoryMemoryPressureEvents(bool shouldLog) { m_shouldLogMemoryMemoryPressureEvents = shouldLog; }
161
162private:
163 std::optional<size_t> thresholdForMemoryKill();
164 void memoryPressureStatusChanged();
165
166 void uninstall();
167
168 void holdOff(Seconds);
169
170 MemoryPressureHandler();
171 ~MemoryPressureHandler() = delete;
172
173 void respondToMemoryPressure(Critical, Synchronous = Synchronous::No);
174 void platformReleaseMemory(Critical);
175 void platformInitialize();
176
177 void measurementTimerFired();
178 void shrinkOrDie(size_t killThreshold);
179 void setMemoryUsagePolicyBasedOnFootprint(size_t);
180 void doesExceedInactiveLimitWhileActive();
181 void doesNotExceedInactiveLimitWhileActive();
182
183 unsigned m_pageCount { 0 };
184
185 std::atomic<bool> m_underMemoryPressure { false };
186 bool m_installed { false };
187 bool m_isSimulatingMemoryPressure { false };
188 bool m_shouldLogMemoryMemoryPressureEvents { true };
189 bool m_hasInvokedDidExceedInactiveLimitWhileActiveCallback { false };
190
191 WebsamProcessState m_processState { WebsamProcessState::Inactive };
192
193 MemoryUsagePolicy m_memoryUsagePolicy { MemoryUsagePolicy::Unrestricted };
194
195 std::unique_ptr<RunLoop::Timer<MemoryPressureHandler>> m_measurementTimer;
196 WTF::Function<void()> m_memoryKillCallback;
197 WTF::Function<void(bool)> m_memoryPressureStatusChangedCallback;
198 WTF::Function<void()> m_didExceedInactiveLimitWhileActiveCallback;
199 LowMemoryHandler m_lowMemoryHandler;
200
201#if OS(WINDOWS)
202 void windowsMeasurementTimerFired();
203 RunLoop::Timer<MemoryPressureHandler> m_windowsMeasurementTimer;
204 Win32Handle m_lowMemoryHandle;
205#endif
206
207#if OS(LINUX) || OS(FREEBSD)
208 RunLoop::Timer<MemoryPressureHandler> m_holdOffTimer;
209 void holdOffTimerFired();
210#endif
211
212#if PLATFORM(COCOA)
213 OSObjectPtr<dispatch_queue_t> m_dispatchQueue;
214#endif
215};
