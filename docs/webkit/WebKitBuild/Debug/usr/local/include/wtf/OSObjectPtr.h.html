28#include <os/object.h>
29#include <wtf/StdLibExtras.h>

31// Because ARC enablement is a compile-time choice, and we compile this header
32// both ways, we need a separate copy of our code when ARC is enabled.
33#if __has_feature(objc_arc)
34#define retainOSObject retainOSObjectArc
35#define releaseOSObject releaseOSObjectArc
36#endif
37
38namespace WTF {
39
40template<typename> class OSObjectPtr;
41template<typename T> OSObjectPtr<T> adoptOSObject(T);
42
43template<typename T> static inline void retainOSObject(T ptr)
44{
45#if __has_feature(objc_arc)
46 UNUSED_PARAM(ptr);
47#else
48 os_retain(ptr);
49#endif
50}
51
52template<typename T> static inline void releaseOSObject(T ptr)
53{
54#if __has_feature(objc_arc)
55 UNUSED_PARAM(ptr);
56#else
57 os_release(ptr);
58#endif
59}
60
61template<typename T> class OSObjectPtr {
62public:
63 OSObjectPtr()
64 : m_ptr(nullptr)
65 {
66 }
67
68 ~OSObjectPtr()
69 {
70 if (m_ptr)
71 releaseOSObject(m_ptr);
72 }
73
74 T get() const { return m_ptr; }
75
76 explicit operator bool() const { return m_ptr; }
77 bool operator!() const { return !m_ptr; }
78
79 OSObjectPtr(const OSObjectPtr& other)
80 : m_ptr(other.m_ptr)
81 {
82 if (m_ptr)
83 retainOSObject(m_ptr);
84 }
85
86 OSObjectPtr(OSObjectPtr&& other)
87 : m_ptr(WTFMove(other.m_ptr))
88 {
89 other.m_ptr = nullptr;
90 }
91
92 OSObjectPtr(T ptr)
93 : m_ptr(WTFMove(ptr))
94 {
95 if (m_ptr)
96 retainOSObject(m_ptr);
97 }
98
99 OSObjectPtr& operator=(const OSObjectPtr& other)
100 {
101 OSObjectPtr ptr = other;
102 swap(ptr);
103 return *this;
104 }
105
106 OSObjectPtr& operator=(OSObjectPtr&& other)
107 {
108 OSObjectPtr ptr = WTFMove(other);
109 swap(ptr);
110 return *this;
111 }
112
113 OSObjectPtr& operator=(std::nullptr_t)
114 {
115 if (m_ptr)
116 releaseOSObject(m_ptr);
117 m_ptr = nullptr;
118 return *this;
119 }
120
121 OSObjectPtr& operator=(T other)
122 {
123 OSObjectPtr ptr = WTFMove(other);
124 swap(ptr);
125 return *this;
126 }
127
128 void swap(OSObjectPtr& other)
129 {
130 std::swap(m_ptr, other.m_ptr);
131 }
132
133 T leakRef() WARN_UNUSED_RETURN
134 {
135 return std::exchange(m_ptr, nullptr);
136 }
137
138 friend OSObjectPtr adoptOSObject<T>(T);
139
140private:
141 struct AdoptOSObject { };
142 OSObjectPtr(AdoptOSObject, T ptr)
143 : m_ptr(WTFMove(ptr))
144 {
145 }
146
147 T m_ptr;
148};
149
150template<typename T> inline OSObjectPtr<T> adoptOSObject(T ptr)
151{
152 return OSObjectPtr<T> { typename OSObjectPtr<T>::AdoptOSObject { }, WTFMove(ptr) };
153}
154
155} // namespace WTF
156
157using WTF::OSObjectPtr;
158using WTF::adoptOSObject;
