42template<typename T>
43class Packed {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 static constexpr bool isPackedType = true;
47
48 Packed()
49 : Packed(T { })
50 {
51 }
52
53 Packed(const T& value)
54 {
55 unalignedStore<T>(m_storage.data(), value);
56 }
57
58 T get() const
59 {
60 return unalignedLoad<T>(m_storage.data());
61 }
62
63 void set(const T& value)
64 {
65 unalignedStore<T>(m_storage.data(), value);
66 }
67
68 Packed<T>& operator=(const T& value)
69 {
70 set(value);
71 return *this;
72 }
73
74 template<class U>
75 T exchange(U&& newValue)
76 {
77 T oldValue = get();
78 set(std::forward<U>(newValue));
79 return oldValue;
80 }
81
82 void swap(Packed& other)
83 {
84 m_storage.swap(other.m_storage);
85 }
86
87 template<typename Other, typename = std::enable_if_t<Other::isPackedType>>
88 void swap(Other& other)
89 {
90 T t1 = get();
91 T t2 = other.get();
92 set(t2);
93 other.set(t1);
94 }
95
96 void swap(T& t2)
97 {
98 T t1 = get();
99 std::swap(t1, t2);
100 set(t1);
101 }
102
103private:
104 std::array<uint8_t, sizeof(T)> m_storage;
105};

107// PackedAlignedPtr can take alignment parameter too. PackedAlignedPtr only uses this alignment information if it is profitable: we use
108// alignment information only when we can reduce the size of the storage.
109template<typename T, size_t passedAlignment>
110class PackedAlignedPtr {
111 WTF_MAKE_FAST_ALLOCATED;
112public:
113 static_assert(hasOneBitSet(passedAlignment), "Alignment needs to be power-of-two");
114 static constexpr size_t alignment = passedAlignment;
115 static constexpr bool isPackedType = true;
116 static constexpr unsigned alignmentShiftSizeIfProfitable = getLSBSetConstexpr(alignment);
117 static constexpr unsigned storageSizeWithoutAlignmentShift = roundUpToMultipleOf<8>(OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH)) / 8;
118 static constexpr unsigned storageSizeWithAlignmentShift = roundUpToMultipleOf<8>(OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH) - alignmentShiftSizeIfProfitable) / 8;
119 static constexpr bool isAlignmentShiftProfitable = storageSizeWithoutAlignmentShift > storageSizeWithAlignmentShift;
120 static constexpr unsigned alignmentShiftSize = isAlignmentShiftProfitable ? alignmentShiftSizeIfProfitable : 0;
121 static constexpr unsigned storageSize = storageSizeWithAlignmentShift;
122 static_assert(storageSize <= sizeof(uintptr_t));
123
124 constexpr PackedAlignedPtr()
125 : m_storage()
126 {
127 }
128
129 constexpr PackedAlignedPtr(std::nullptr_t)
130 : m_storage()
131 {
132 }
133
134 PackedAlignedPtr(T* value)
135 {
136 set(value);
137 }
138
139 T* get() const
140 {
141 // FIXME: PackedPtr<> can load memory with one mov by checking page boundary.
142 // https://bugs.webkit.org/show_bug.cgi?id=197754
143 uintptr_t value = 0;
144
145IGNORE_ARRAY_BOUNDS_WARNINGS_BEGIN
146#if CPU(LITTLE_ENDIAN)
147 memcpy(&value, m_storage.data(), storageSize);
148#else
149 memcpy(bitwise_cast<uint8_t*>(&value) + (sizeof(void*) - storageSize), m_storage.data(), storageSize);
150#endif
151IGNORE_ARRAY_BOUNDS_WARNINGS_END
152
153 if (isAlignmentShiftProfitable)
154 value <<= alignmentShiftSize;
155
156#if CPU(X86_64) && !(OS(DARWIN) || OS(LINUX) || OS(WINDOWS))
157 // The AMD specification requires that the most significant 16
158 // bits of any virtual address, bits 48 through 63, must be
159 // copies of bit 47 (in a manner akin to sign extension).
160 //
161 // The above-named OSes will never allocate user space addresses
162 // with bit 47 set, thus are already in canonical form.
163 //
164 // Reference: https://en.wikipedia.org/wiki/X86-64#Virtual_address_space_details
165 constexpr unsigned shiftBits = countOfBits<uintptr_t> - OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH);
166 value = (bitwise_cast<intptr_t>(value) << shiftBits) >> shiftBits;
167#endif
168
169 return bitwise_cast<T*>(value);
170 }
171
172 void set(T* passedValue)
173 {
174 uintptr_t value = bitwise_cast<uintptr_t>(passedValue);
175 if (isAlignmentShiftProfitable)
176 value >>= alignmentShiftSize;
177#if CPU(LITTLE_ENDIAN)
178 memcpy(m_storage.data(), &value, storageSize);
179#else
180 memcpy(m_storage.data(), bitwise_cast<uint8_t*>(&value) + (sizeof(void*) - storageSize), storageSize);
181#endif
182 ASSERT(bitwise_cast<uintptr_t>(get()) == value);
183 }
184
185 void clear()
186 {
187 set(nullptr);
188 }
189
190 T* operator->() const { return get(); }
191 T& operator*() const { return *get(); }
192 bool operator!() const { return !get(); }
193
194 // This conversion operator allows implicit conversion to bool but not to other integer types.
195 typedef T* (PackedAlignedPtr::*UnspecifiedBoolType);
196 operator UnspecifiedBoolType() const { return get() ? &PackedAlignedPtr::m_storage : nullptr; }
197 explicit operator bool() const { return get(); }
198
199 PackedAlignedPtr& operator=(T* value)
200 {
201 set(value);
202 return *this;
203 }
204
205 template<class U>
206 T* exchange(U&& newValue)
207 {
208 T* oldValue = get();
209 set(std::forward<U>(newValue));
210 return oldValue;
211 }
212
213 void swap(std::nullptr_t) { clear(); }
214
215 void swap(PackedAlignedPtr& other)
216 {
217 m_storage.swap(other.m_storage);
218 }
219
220 template<typename Other, typename = std::enable_if_t<Other::isPackedType>>
221 void swap(Other& other)
222 {
223 T* t1 = get();
224 T* t2 = other.get();
225 set(t2);
226 other.set(t1);
227 }
228
229 void swap(T* t2)
230 {
231 T* t1 = get();
232 std::swap(t1, t2);
233 set(t1);
234 }
235
236private:
237 std::array<uint8_t, storageSize> m_storage;
238};

240template<typename T>
241class Packed<T*> : public PackedAlignedPtr<T, 1> {
242public:
243 using Base = PackedAlignedPtr<T, 1>;
244 using Base::Base;
245
246 // Hash table deleted values, which are only constructed and never copied or destroyed.
247 Packed(HashTableDeletedValueType) : Base(bitwise_cast<T*>(static_cast<uintptr_t>(Base::alignment))) { }
248 bool isHashTableDeletedValue() const { return Base::get() == bitwise_cast<T*>(static_cast<uintptr_t>(Base::alignment)); }
249};

251template<typename T>
252using PackedPtr = Packed<T*>;

254template <typename T>
255struct GetPtrHelper<PackedPtr<T>> {
256 using PtrType = T*;
257 static T* getPtr(const PackedPtr<T>& p) { return const_cast<T*>(p.get()); }
258};

264
265template<typename T>
266struct PackedPtrTraits {
267 template<typename U> using RebindTraits = PackedPtrTraits<U>;
268
269 using StorageType = PackedPtr<T>;
270
271 template<class U> static ALWAYS_INLINE T* exchange(StorageType& ptr, U&& newValue) { return ptr.exchange(newValue); }
272
273 template<typename Other> static ALWAYS_INLINE void swap(PackedPtr<T>& a, Other& b) { a.swap(b); }
274
275 static ALWAYS_INLINE T* unwrap(const StorageType& ptr) { return ptr.get(); }
276
277 // We assume that,
278 // 1. The alignment is < 4KB. (It is tested by HashTraits).
279 // 2. The first page (including nullptr) is never mapped.
280 static StorageType hashTableDeletedValue() { return StorageType { bitwise_cast<T*>(static_cast<uintptr_t>(StorageType::alignment)) }; }
281 static ALWAYS_INLINE bool isHashTableDeletedValue(const StorageType& ptr) { return ptr.get() == bitwise_cast<T*>(static_cast<uintptr_t>(StorageType::alignment)); }
282};
