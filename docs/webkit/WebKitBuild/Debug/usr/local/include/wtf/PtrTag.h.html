47enum PtrTag : uintptr_t {
48 NoPtrTag,
49 CFunctionPtrTag,
50};

52enum class PtrTagCallerType : uint8_t { Native, JIT, None };

55template<PtrTag tag, typename PtrType>
56ALWAYS_INLINE static PtrType tagNativeCodePtrImpl(PtrType ptr)
57{
58#if CPU(ARM64E)
59 if constexpr (tag == NoPtrTag)
60 return ptr;
61 if constexpr (tag == CFunctionPtrTag)
62 return ptrauth_sign_unauthenticated(ptr, ptrauth_key_function_pointer, 0);
63 return ptrauth_sign_unauthenticated(ptr, ptrauth_key_process_dependent_code, tag);
64#else
65 return ptr;
66#endif
67}

69template<PtrTag tag, typename PtrType>
70ALWAYS_INLINE static PtrType untagNativeCodePtrImpl(PtrType ptr)
71{
72#if CPU(ARM64E)
73 if constexpr (tag == NoPtrTag)
74 return ptr;
75 if constexpr (tag == CFunctionPtrTag)
76 return __builtin_ptrauth_auth(ptr, ptrauth_key_function_pointer, 0);
77 return __builtin_ptrauth_auth(ptr, ptrauth_key_process_dependent_code, tag);
78#else
79 return ptr;
80#endif
81}

83template<PtrTag passedTag>
84struct PtrTagTraits {
85 static constexpr PtrTag tag = passedTag;
86 static constexpr PtrTagCallerType callerType = PtrTagCallerType::Native;
87 static constexpr PtrTagCalleeType calleeType = PtrTagCalleeType::Native;
88 static constexpr bool isSpecialized = false;
89
90 template<typename PtrType>
91 ALWAYS_INLINE static PtrType tagCodePtr(PtrType ptr)
92 {
93 return tagNativeCodePtrImpl<tag>(ptr);
94 }
95
96 template<typename PtrType>
97 ALWAYS_INLINE static PtrType untagCodePtr(PtrType ptr)
98 {
99 return untagNativeCodePtrImpl<tag>(ptr);
100 }
101};

108template<size_t N>
109constexpr uintptr_t makePtrTagHash(const char (&str)[N])
110{
111 uintptr_t result = 134775813;
112 for (size_t i = 0; i < N; ++i)
113 result += ((result * str[i]) ^ (result >> 16));
114 return result & 0xffff;
115}
116
117#define WTF_PTRTAG_HASH(tag) WTF::makePtrTagHash(#tag)
118#endif // not CPU(ARM64E)

120#define WTF_DECLARE_PTRTAG(tag) \
121 constexpr PtrTag tag = static_cast<PtrTag>(WTF_PTRTAG_HASH(#tag)); \
122 static_assert(tag != NoPtrTag && tag != CFunctionPtrTag, "");

124static_assert(static_cast<uintptr_t>(NoPtrTag) == static_cast<uintptr_t>(0), "");
125static_assert(static_cast<uintptr_t>(CFunctionPtrTag) == static_cast<uintptr_t>(1), "");

138struct PtrTagLookup {
139 using TagForPtrFunc = const char* (*)(const void*);
140 using PtrTagNameFunc = const char* (*)(PtrTag);
141
142 void initialize(TagForPtrFunc tagForPtr, PtrTagNameFunc ptrTagName)
143 {
144 this->tagForPtr = tagForPtr;
145 this->ptrTagName = ptrTagName;
146 }
147
148 TagForPtrFunc tagForPtr;
149 PtrTagNameFunc ptrTagName;
150 PtrTagLookup* next;
151};

156WTF_EXPORT_PRIVATE void registerPtrTagLookup(PtrTagLookup*);

159#if ENABLE(PTRTAG_DEBUGGING)
160constexpr bool enablePtrTagDebugAssert = true;
161#define REPORT_BAD_TAG(success, ptr, expectedTag) do { \
162 if (UNLIKELY(!success)) \
163 reportBadTag(reinterpret_cast<const void*>(ptr), expectedTag); \
164 } while (false)
165#else
166constexpr bool enablePtrTagDebugAssert = false;
167#define REPORT_BAD_TAG(success, ptr, expectedTag)
168#endif
169
170#define WTF_PTRTAG_ASSERT(action, ptr, expectedTag, assertion) \
171 do { \
172 if constexpr (action == PtrTagAction::ReleaseAssert \
173 || (WTF::enablePtrTagDebugAssert && action == PtrTagAction::DebugAssert)) { \
174 bool passed = (assertion); \
175 REPORT_BAD_TAG(passed, ptr, expectedTag); \
176 RELEASE_ASSERT(passed && #assertion); \
177 } \
178 } while (false)
179#else
180
181inline void registerPtrTagLookup(PtrTagLookup*) { }
182inline void reportBadTag(const void*, PtrTag) { }
183
184#define WTF_PTRTAG_ASSERT(action, ptr, expectedTag, assertion) \
185 do { \
186 if constexpr (action == PtrTagAction::ReleaseAssert) { \
187 UNUSED_PARAM(ptr); \
188 RELEASE_ASSERT(assertion); \
189 } \
190 } while (false)
191#endif

193enum class PtrTagAction {
194 ReleaseAssert,
195 DebugAssert,
196 NoAssert,

201template<typename T, typename PtrType, typename = std::enable_if_t<std::is_pointer<PtrType>::value && !std::is_same<T, PtrType>::value>>
202inline constexpr T removeCodePtrTag(PtrType ptr)
203{
204#if CPU(ARM64E)
205 return bitwise_cast<T>(ptrauth_strip(ptr, ptrauth_key_process_dependent_code));
206#else
207 return bitwise_cast<T>(ptr);
208#endif

221template<PtrTagAction tagAction, PtrTag tag, typename PtrType>
222inline PtrType tagCodePtrImpl(PtrType ptr)
223{
224 if (!ptr)
225 return nullptr;
226 WTF_PTRTAG_ASSERT(tagAction, ptr, NoPtrTag, removeCodePtrTag(ptr) == ptr);
227 return PtrTagTraits<tag>::tagCodePtr(ptr);
228}

239template<PtrTagAction tagAction, PtrTag tag, typename PtrType>
240inline PtrType untagCodePtrImpl(PtrType ptr)
241{
242 if (!ptr)
243 return nullptr;
244 PtrType result = PtrTagTraits<tag>::untagCodePtr(ptr);
245 WTF_PTRTAG_ASSERT(tagAction, ptr, tag, removeCodePtrTag(ptr) == result);
246 return result;

258template<PtrTagAction tagAction, PtrTag oldTag, PtrTag newTag, typename PtrType>
259inline PtrType retagCodePtrImplHelper(PtrType ptr)
260{
261 if constexpr (oldTag == newTag || (oldTag == NoPtrTag && newTag == NoPtrTag))
262 return ptr;
263 if constexpr (newTag == NoPtrTag)
264 return untagCodePtrImpl<tagAction, oldTag>(ptr);
265 if constexpr (oldTag == NoPtrTag)
266 return tagCodePtrImpl<tagAction, newTag>(ptr);
267#if CPU(ARM64E)
268 if constexpr (PtrTagTraits<oldTag>::isSpecialized || PtrTagTraits<newTag>::isSpecialized)
269 return tagCodePtrImpl<tagAction, newTag>(untagCodePtrImpl<tagAction, oldTag>(ptr));
270 if constexpr (oldTag == CFunctionPtrTag)
271 return ptrauth_auth_and_resign(ptr, ptrauth_key_function_pointer, 0, ptrauth_key_process_dependent_code, newTag);
272 if constexpr (newTag == CFunctionPtrTag)
273 return ptrauth_auth_and_resign(ptr, ptrauth_key_process_dependent_code, oldTag, ptrauth_key_function_pointer, 0);
274 return ptrauth_auth_and_resign(ptr, ptrauth_key_process_dependent_code, oldTag, ptrauth_key_process_dependent_code, newTag);
275#else
276 return tagCodePtrImpl<tagAction, newTag>(untagCodePtrImpl<tagAction, oldTag>(ptr));
277#endif
278}

280template<PtrTagAction tagAction, PtrTag oldTag, PtrTag newTag, typename PtrType>
281inline PtrType retagCodePtrImpl(PtrType ptr)
282{
283 if (!ptr)
284 return nullptr;
285 WTF_PTRTAG_ASSERT(tagAction, ptr, oldTag, ptr == (tagCodePtrImpl<PtrTagAction::NoAssert, oldTag>(removeCodePtrTag(ptr))));
286 PtrType result = retagCodePtrImplHelper<tagAction, oldTag, newTag>(ptr);
287 WTF_PTRTAG_ASSERT(tagAction, ptr, newTag, result == (tagCodePtrImpl<PtrTagAction::NoAssert, newTag>(removeCodePtrTag(ptr))));
288 return result;
289}

335template<PtrTag tag, typename PtrType>
336bool isTaggedWith(PtrType value)
337{
338 void* ptr = bitwise_cast<void*>(value);
339 if (tag == NoPtrTag)
340 return ptr == removeCodePtrTag(ptr);
341 return ptr == tagCodePtrImpl<PtrTagAction::NoAssert, tag>(removeCodePtrTag(ptr));
342}
