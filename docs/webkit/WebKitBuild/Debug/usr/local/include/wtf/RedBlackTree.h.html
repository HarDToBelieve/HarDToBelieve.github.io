37// This implements a red-black tree with the following properties:
38// - The allocation of nodes in the tree is entirely up to the user.
39// - If you are in possession of a pointer to a node, you can delete
40// it from the tree. The tree will subsequently no longer have a
41// reference to this node.
42// - The key type must implement operator< and ==.
43
44template<class NodeType, typename KeyType>
45class RedBlackTree final {
46 WTF_MAKE_NONCOPYABLE(RedBlackTree);
47 WTF_MAKE_FAST_ALLOCATED;
48private:
49 static constexpr unsigned s_maximumTreeDepth = 128;
50
51 enum Color {
52 Red = 1,
53 Black
54 };
55
56public:
57 class Node {
58 friend class RedBlackTree;
59
60 public:
61 const NodeType* successor() const
62 {
63 const Node* x = this;
64 if (x->right())
65 return treeMinimum(x->right());
66 const NodeType* y = x->parent();
67 unsigned depth = 0;
68 while (y && x == y->right()) {
69 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
70 x = y;
71 y = y->parent();
72 }
73 return y;
74 }
75
76 const NodeType* predecessor() const
77 {
78 const Node* x = this;
79 if (x->left())
80 return treeMaximum(x->left());
81 const NodeType* y = x->parent();
82 unsigned depth = 0;
83 while (y && x == y->left()) {
84 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
85 x = y;
86 y = y->parent();
87 }
88 return y;
89 }
90
91 NodeType* successor()
92 {
93 return const_cast<NodeType*>(const_cast<const Node*>(this)->successor());
94 }
95
96 NodeType* predecessor()
97 {
98 return const_cast<NodeType*>(const_cast<const Node*>(this)->predecessor());
99 }
100
101 private:
102 void reset()
103 {
104 m_left = nullptr;
105 m_right = nullptr;
106 m_parentAndRed = 1; // initialize to red
107 }
108
109 // NOTE: these methods should pack the parent and red into a single
110 // word. But doing so appears to reveal a bug in the compiler.
111 NodeType* parent() const
112 {
113 return reinterpret_cast<NodeType*>(m_parentAndRed & ~static_cast<uintptr_t>(1));
114 }
115
116 void setParent(NodeType* newParent)
117 {
118 m_parentAndRed = reinterpret_cast<uintptr_t>(newParent) | (m_parentAndRed & 1);
119 }
120
121 NodeType* left() const
122 {
123 return m_left;
124 }
125
126 void setLeft(NodeType* node)
127 {
128 m_left = node;
129 }
130
131 NodeType* right() const
132 {
133 return m_right;
134 }
135
136 void setRight(NodeType* node)
137 {
138 m_right = node;
139 }
140
141 Color color() const
142 {
143 if (m_parentAndRed & 1)
144 return Red;
145 return Black;
146 }
147
148 void setColor(Color value)
149 {
150 if (value == Red)
151 m_parentAndRed |= 1;
152 else
153 m_parentAndRed &= ~static_cast<uintptr_t>(1);
154 }
155
156 NodeType* m_left;
157 NodeType* m_right;
158 uintptr_t m_parentAndRed;
159 };
160
161 RedBlackTree()
162 : m_root(nullptr)
163 {
164 }
165
166 void insert(NodeType* x)
167 {
168 x->reset();
169 treeInsert(x);
170 x->setColor(Red);
171
172 unsigned depth = 0;
173 while (x != m_root && x->parent()->color() == Red) {
174 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
175 if (x->parent() == x->parent()->parent()->left()) {
176 NodeType* y = x->parent()->parent()->right();
177 if (y && y->color() == Red) {
178 // Case 1
179 x->parent()->setColor(Black);
180 y->setColor(Black);
181 x->parent()->parent()->setColor(Red);
182 x = x->parent()->parent();
183 } else {
184 if (x == x->parent()->right()) {
185 // Case 2
186 x = x->parent();
187 leftRotate(x);
188 }
189 // Case 3
190 x->parent()->setColor(Black);
191 x->parent()->parent()->setColor(Red);
192 rightRotate(x->parent()->parent());
193 }
194 } else {
195 // Same as "then" clause with "right" and "left" exchanged.
196 NodeType* y = x->parent()->parent()->left();
197 if (y && y->color() == Red) {
198 // Case 1
199 x->parent()->setColor(Black);
200 y->setColor(Black);
201 x->parent()->parent()->setColor(Red);
202 x = x->parent()->parent();
203 } else {
204 if (x == x->parent()->left()) {
205 // Case 2
206 x = x->parent();
207 rightRotate(x);
208 }
209 // Case 3
210 x->parent()->setColor(Black);
211 x->parent()->parent()->setColor(Red);
212 leftRotate(x->parent()->parent());
213 }
214 }
215 }
216
217 m_root->setColor(Black);
218 }
219
220 NodeType* remove(NodeType* z)
221 {
222 ASSERT(z);
223 ASSERT(z->parent() || z == m_root);
224
225 // Y is the node to be unlinked from the tree.
226 NodeType* y;
227 if (!z->left() || !z->right())
228 y = z;
229 else
230 y = z->successor();
231
232 // Y is guaranteed to be non-null at this point.
233 NodeType* x;
234 if (y->left())
235 x = y->left();
236 else
237 x = y->right();
238
239 // X is the child of y which might potentially replace y in
240 // the tree. X might be null at this point.
241 NodeType* xParent;
242 if (x) {
243 x->setParent(y->parent());
244 xParent = x->parent();
245 } else
246 xParent = y->parent();
247 if (!y->parent())
248 m_root = x;
249 else {
250 if (y == y->parent()->left())
251 y->parent()->setLeft(x);
252 else
253 y->parent()->setRight(x);
254 }
255
256 if (y != z) {
257 if (y->color() == Black)
258 removeFixup(x, xParent);
259
260 y->setParent(z->parent());
261 y->setColor(z->color());
262 y->setLeft(z->left());
263 y->setRight(z->right());
264
265 if (z->left())
266 z->left()->setParent(y);
267 if (z->right())
268 z->right()->setParent(y);
269 if (z->parent()) {
270 if (z->parent()->left() == z)
271 z->parent()->setLeft(y);
272 else
273 z->parent()->setRight(y);
274 } else {
275 ASSERT(m_root == z);
276 m_root = y;
277 }
278 } else if (y->color() == Black)
279 removeFixup(x, xParent);
280
281 return z;
282 }
283
284 NodeType* remove(const KeyType& key)
285 {
286 NodeType* result = findExact(key);
287 if (!result)
288 return 0;
289 return remove(result);
290 }
291
292 NodeType* findExact(const KeyType& key) const
293 {
294 unsigned depth = 0;
295 for (NodeType* current = m_root; current;) {
296 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
297 if (current->key() == key)
298 return current;
299 if (key < current->key())
300 current = current->left();
301 else
302 current = current->right();
303 }
304 return 0;
305 }
306
307 NodeType* findLeastGreaterThanOrEqual(const KeyType& key) const
308 {
309 NodeType* best = nullptr;
310 unsigned depth = 0;
311 for (NodeType* current = m_root; current;) {
312 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
313 if (current->key() == key)
314 return current;
315 if (current->key() < key)
316 current = current->right();
317 else {
318 best = current;
319 current = current->left();
320 }
321 }
322 return best;
323 }
324
325 NodeType* findGreatestLessThanOrEqual(const KeyType& key) const
326 {
327 NodeType* best = nullptr;
328 unsigned depth = 0;
329 for (NodeType* current = m_root; current;) {
330 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
331 if (current->key() == key)
332 return current;
333 if (current->key() > key)
334 current = current->left();
335 else {
336 best = current;
337 current = current->right();
338 }
339 }
340 return best;
341 }

343 template <typename Function>
344 void iterate(Function function)
345 {
346 if (!m_root)
347 return;
348
349 Vector<NodeType*, 16> toIterate;
350 unsigned size = 0;
351 toIterate.append(m_root);
352 while (toIterate.size()) {
353 RELEASE_ASSERT(++size < std::numeric_limits<unsigned>::max());
354 NodeType& current = *toIterate.takeLast();
355 bool iterateLeft = false;
356 bool iterateRight = false;
357 function(current, iterateLeft, iterateRight);
358 if (iterateLeft && current.left())
359 toIterate.append(current.left());
360 if (iterateRight && current.right())
361 toIterate.append(current.right());
362 }
363 }
364
365 NodeType* first() const
366 {
367 if (!m_root)
368 return nullptr;
369 return treeMinimum(m_root);
370 }
371
372 NodeType* last() const
373 {
374 if (!m_root)
375 return 0;
376 return treeMaximum(m_root);
377 }
378
379 // This is an O(n) operation.
380 size_t size()
381 {
382 size_t result = 0;
383 for (NodeType* current = first(); current; current = current->successor())
384 result++;
385 return result;
386 }
387
388 // This is an O(1) operation.
389 bool isEmpty()
390 {
391 return !m_root;
392 }

394private:
395 // Finds the minimum element in the sub-tree rooted at the given
396 // node.
397 static NodeType* treeMinimum(NodeType* x)
398 {
399 unsigned depth = 0;
400 while (x->left()) {
401 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
402 x = x->left();
403 }
404 return x;
405 }
406
407 static NodeType* treeMaximum(NodeType* x)
408 {
409 unsigned depth = 0;
410 while (x->right()) {
411 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
412 x = x->right();
413 }
414 return x;
415 }
416
417 static const NodeType* treeMinimum(const NodeType* x)
418 {
419 unsigned depth = 0;
420 while (x->left()) {
421 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
422 x = x->left();
423 }
424 return x;
425 }
426
427 static const NodeType* treeMaximum(const NodeType* x)
428 {
429 unsigned depth = 0;
430 while (x->right()) {
431 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
432 x = x->right();
433 }
434 return x;
435 }

437 void treeInsert(NodeType* z)
438 {
439 ASSERT(!z->left());
440 ASSERT(!z->right());
441 ASSERT(!z->parent());
442 ASSERT(z->color() == Red);
443
444 NodeType* y = nullptr;
445 NodeType* x = m_root;
446 unsigned depth = 0;
447 while (x) {
448 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
449 y = x;
450 if (z->key() < x->key())
451 x = x->left();
452 else
453 x = x->right();
454 }
455 z->setParent(y);
456 if (!y)
457 m_root = z;
458 else {
459 if (z->key() < y->key())
460 y->setLeft(z);
461 else
462 y->setRight(z);
463 }
464 }

466 //----------------------------------------------------------------------
467 // Red-Black tree operations
468 //
469
470 // Left-rotates the subtree rooted at x.
471 // Returns the new root of the subtree (x's right child).
472 NodeType* leftRotate(NodeType* x)
473 {
474 // Set y.
475 NodeType* y = x->right();
476
477 // Turn y's left subtree into x's right subtree.
478 x->setRight(y->left());
479 if (y->left())
480 y->left()->setParent(x);
481
482 // Link x's parent to y.
483 y->setParent(x->parent());
484 if (!x->parent())
485 m_root = y;
486 else {
487 if (x == x->parent()->left())
488 x->parent()->setLeft(y);
489 else
490 x->parent()->setRight(y);
491 }
492
493 // Put x on y's left.
494 y->setLeft(x);
495 x->setParent(y);
496
497 return y;
498 }
499
500 // Right-rotates the subtree rooted at y.
501 // Returns the new root of the subtree (y's left child).
502 NodeType* rightRotate(NodeType* y)
503 {
504 // Set x.
505 NodeType* x = y->left();
506
507 // Turn x's right subtree into y's left subtree.
508 y->setLeft(x->right());
509 if (x->right())
510 x->right()->setParent(y);
511
512 // Link y's parent to x.
513 x->setParent(y->parent());
514 if (!y->parent())
515 m_root = x;
516 else {
517 if (y == y->parent()->left())
518 y->parent()->setLeft(x);
519 else
520 y->parent()->setRight(x);
521 }
522
523 // Put y on x's right.
524 x->setRight(y);
525 y->setParent(x);
526
527 return x;
528 }
529
530 // Restores the red-black property to the tree after splicing out
531 // a node. Note that x may be null, which is why xParent must be
532 // supplied.
533 void removeFixup(NodeType* x, NodeType* xParent)
534 {
535 unsigned depth = 0;
536 while (x != m_root && (!x || x->color() == Black)) {
537 RELEASE_ASSERT(++depth <= s_maximumTreeDepth);
538 if (x == xParent->left()) {
539 // Note: the text points out that w can not be null.
540 // The reason is not obvious from simply looking at
541 // the code; it comes about from the properties of the
542 // red-black tree.
543 NodeType* w = xParent->right();
544 ASSERT(w); // x's sibling should not be null.
545 if (w->color() == Red) {
546 // Case 1
547 w->setColor(Black);
548 xParent->setColor(Red);
549 leftRotate(xParent);
550 w = xParent->right();
551 }
552 if ((!w->left() || w->left()->color() == Black)
553 && (!w->right() || w->right()->color() == Black)) {
554 // Case 2
555 w->setColor(Red);
556 x = xParent;
557 xParent = x->parent();
558 } else {
559 if (!w->right() || w->right()->color() == Black) {
560 // Case 3
561 w->left()->setColor(Black);
562 w->setColor(Red);
563 rightRotate(w);
564 w = xParent->right();
565 }
566 // Case 4
567 w->setColor(xParent->color());
568 xParent->setColor(Black);
569 if (w->right())
570 w->right()->setColor(Black);
571 leftRotate(xParent);
572 x = m_root;
573 xParent = x->parent();
574 }
575 } else {
576 // Same as "then" clause with "right" and "left"
577 // exchanged.
578
579 // Note: the text points out that w can not be null.
580 // The reason is not obvious from simply looking at
581 // the code; it comes about from the properties of the
582 // red-black tree.
583 NodeType* w = xParent->left();
584 ASSERT(w); // x's sibling should not be null.
585 if (w->color() == Red) {
586 // Case 1
587 w->setColor(Black);
588 xParent->setColor(Red);
589 rightRotate(xParent);
590 w = xParent->left();
591 }
592 if ((!w->right() || w->right()->color() == Black)
593 && (!w->left() || w->left()->color() == Black)) {
594 // Case 2
595 w->setColor(Red);
596 x = xParent;
597 xParent = x->parent();
598 } else {
599 if (!w->left() || w->left()->color() == Black) {
600 // Case 3
601 w->right()->setColor(Black);
602 w->setColor(Red);
603 leftRotate(w);
604 w = xParent->left();
605 }
606 // Case 4
607 w->setColor(xParent->color());
608 xParent->setColor(Black);
609 if (w->left())
610 w->left()->setColor(Black);
611 rightRotate(xParent);
612 x = m_root;
613 xParent = x->parent();
614 }
615 }
616 }
617 if (x)
618 x->setColor(Black);
619 }
