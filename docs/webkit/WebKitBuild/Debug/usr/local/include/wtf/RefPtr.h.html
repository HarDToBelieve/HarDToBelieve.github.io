32template<typename T> struct DefaultRefDerefTraits {
33 static ALWAYS_INLINE void refIfNotNull(T* ptr)
34 {
35 if (LIKELY(ptr != nullptr))
36 ptr->ref();
37 }
38
39 static ALWAYS_INLINE void derefIfNotNull(T* ptr)
40 {
41IGNORE_ERRONEOUS_GCC_NULL_CHECK_WARNINGS_BEGIN
42 if (LIKELY(ptr != nullptr))
43 ptr->deref();
44IGNORE_ERRONEOUS_GCC_NULL_CHECK_WARNINGS_END
45 }
46};
47
48template<typename T, typename PtrTraits, typename RefDerefTraits> class RefPtr;
49template<typename T, typename PtrTraits = RawPtrTraits<T>, typename RefDerefTraits = DefaultRefDerefTraits<T>> RefPtr<T, PtrTraits, RefDerefTraits> adoptRef(T*);
50
51template<typename T, typename _PtrTraits, typename _RefDerefTraits>
52class RefPtr {
53 WTF_MAKE_FAST_ALLOCATED;
54public:
55 using PtrTraits = _PtrTraits;
56 using RefDerefTraits = _RefDerefTraits;
57 typedef T ValueType;
58 typedef ValueType* PtrType;
59
60 static constexpr bool isRefPtr = true;
61
62 ALWAYS_INLINE constexpr RefPtr() : m_ptr(nullptr) { }
63 ALWAYS_INLINE RefPtr(T* ptr) : m_ptr(ptr) { RefDerefTraits::refIfNotNull(ptr); }
64 ALWAYS_INLINE RefPtr(const RefPtr& o) : m_ptr(o.m_ptr) { RefDerefTraits::refIfNotNull(PtrTraits::unwrap(m_ptr)); }
65 template<typename X, typename Y, typename Z> RefPtr(const RefPtr<X, Y, Z>& o) : m_ptr(o.get()) { RefDerefTraits::refIfNotNull(PtrTraits::unwrap(m_ptr)); }
66
67 ALWAYS_INLINE RefPtr(RefPtr&& o) : m_ptr(o.leakRef()) { }
68 template<typename X, typename Y, typename Z> RefPtr(RefPtr<X, Y, Z>&& o) : m_ptr(o.leakRef()) { }
69 template<typename X, typename Y> RefPtr(Ref<X, Y>&&);
70
71 // Hash table deleted values, which are only constructed and never copied or destroyed.
72 RefPtr(HashTableDeletedValueType) : m_ptr(PtrTraits::hashTableDeletedValue()) { }
73 bool isHashTableDeletedValue() const { return PtrTraits::isHashTableDeletedValue(m_ptr); }
74
75 ALWAYS_INLINE ~RefPtr() { RefDerefTraits::derefIfNotNull(PtrTraits::exchange(m_ptr, nullptr)); }
76
77 T* get() const { return PtrTraits::unwrap(m_ptr); }
78
79 Ref<T> releaseNonNull() { ASSERT(m_ptr); Ref<T> tmp(adoptRef(*m_ptr)); m_ptr = nullptr; return tmp; }
80 Ref<const T> releaseConstNonNull() { ASSERT(m_ptr); Ref<const T> tmp(adoptRef(*m_ptr)); m_ptr = nullptr; return tmp; }
81
82 T* leakRef() WARN_UNUSED_RETURN;
83
84 T& operator*() const { ASSERT(m_ptr); return *PtrTraits::unwrap(m_ptr); }
85 ALWAYS_INLINE T* operator->() const { return PtrTraits::unwrap(m_ptr); }
86
87 bool operator!() const { return !m_ptr; }
88
89 // This conversion operator allows implicit conversion to bool but not to other integer types.
90 using UnspecifiedBoolType = void (RefPtr::*)() const;
91 operator UnspecifiedBoolType() const { return m_ptr ? &RefPtr::unspecifiedBoolTypeInstance : nullptr; }
92
93 explicit operator bool() const { return !!m_ptr; }
94
95 RefPtr& operator=(const RefPtr&);
96 RefPtr& operator=(T*);
97 RefPtr& operator=(std::nullptr_t);
98 template<typename X, typename Y, typename Z> RefPtr& operator=(const RefPtr<X, Y, Z>&);
99 RefPtr& operator=(RefPtr&&);
100 template<typename X, typename Y, typename Z> RefPtr& operator=(RefPtr<X, Y, Z>&&);
101 template<typename X> RefPtr& operator=(Ref<X>&&);
102
103 template<typename X, typename Y, typename Z> void swap(RefPtr<X, Y, Z>&);
104
105 RefPtr copyRef() && = delete;
106 RefPtr copyRef() const & WARN_UNUSED_RETURN { return RefPtr(m_ptr); }
107
108private:
109 void unspecifiedBoolTypeInstance() const { }
110
111 friend RefPtr adoptRef<T, PtrTraits, RefDerefTraits>(T*);
112 template<typename X, typename Y, typename Z> friend class RefPtr;
113
114 enum AdoptTag { Adopt };
115 RefPtr(T* ptr, AdoptTag) : m_ptr(ptr) { }
116
117 typename PtrTraits::StorageType m_ptr;
118};

120template<typename T, typename U, typename V>
121template<typename X, typename Y>
122inline RefPtr<T, U, V>::RefPtr(Ref<X, Y>&& reference)
123 : m_ptr(&reference.leakRef())
124{
125}
126
127template<typename T, typename U, typename V>
128inline T* RefPtr<T, U, V>::leakRef()
129{
130 return U::exchange(m_ptr, nullptr);
131}

133template<typename T, typename U, typename V>
134inline RefPtr<T, U, V>& RefPtr<T, U, V>::operator=(const RefPtr& o)
135{
136 RefPtr ptr = o;
137 swap(ptr);
138 return *this;
139}

191template<class T, typename U, typename V>
192template<typename X, typename Y, typename Z>
193inline void RefPtr<T, U, V>::swap(RefPtr<X, Y, Z>& o)
194{
195 U::swap(m_ptr, o.m_ptr);
196}

198template<typename T, typename U, typename V>
199inline void swap(RefPtr<T, U, V>& a, RefPtr<T, U, V>& b)

201 a.swap(b);

240template<typename T, typename U = RawPtrTraits<T>, typename V = DefaultRefDerefTraits<T>, typename X, typename Y, typename Z>
241inline RefPtr<T, U, V> static_pointer_cast(const RefPtr<X, Y, Z>& p)
242{
243 return RefPtr<T, U, V>(static_cast<T*>(p.get()));
244}

251template<typename T, typename U, typename V>
252inline RefPtr<T, U, V> adoptRef(T* p)
253{
254 adopted(p);
255 return RefPtr<T, U, V>(p, RefPtr<T, U, V>::Adopt);
256}
