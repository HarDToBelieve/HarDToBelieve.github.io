61// 95% load factor. This a bit regress "insertion" performance, while it keeps lookup performance sane.
62// RobinHoodHashTable can work with 95% load factor because of maintained probe distance.
63struct MemoryCompactLookupOnlyRobinHoodHashTableSizePolicy {
64 static constexpr unsigned maxLoadNumerator = 19;
65 static constexpr unsigned maxLoadDenominator = 20;
66 static constexpr unsigned minLoad = 6;
67};

69// 90% load factor. RobinHoodHashTable can work with such a high load-factor.
70// Observed performance is slightly worse than HashTable (75% for small table, 50% for large table).
71struct MemoryCompactRobinHoodHashTableSizePolicy {
72 static constexpr unsigned maxLoadNumerator = 9;
73 static constexpr unsigned maxLoadDenominator = 10;
74 static constexpr unsigned minLoad = 6;
75};

77// 75% load factor, this maintains the performance same to HashTable, still the load factor
78// is higher (HashTable uses 75% for small table, 50 for large table).
79struct FastRobinHoodHashTableSizePolicy {
80 static constexpr unsigned maxLoadNumerator = 3;
81 static constexpr unsigned maxLoadDenominator = 4;
82 static constexpr unsigned minLoad = 6;
83};

85// RobinHood HashTable has several limitations compared to usual HashTable, that's why this is not a default one.
86// RobinHood HashTable computes hash much more frequently. This means that the Key should cache computed hash.
87// But our default HashTable does not cache hash value because of memory usage. This design means that Key type
88// should have hash value intrusively (e.g. WTF::String holds hash value internally).
89//
90// If the above requirements meet your use case, then you can try RobinHood HashTable.
91// The largest benefit is that we can use significantly high load-factor (90% - 95%)!
92//
93// The algorithm is RobinHood-Hashing + backward shift deletion, described in [1,2].
94//
95// Naive RobinHoodHashTable has some cases which cause O(N^2) when iterating it and inserting it to some new RobinHoodHashTable
96// without reserving capacity and this is because of high load-factor and exposed hash-ordering at iteration[3]. To mitigate it,
97// we calculate hash for each table, and do XOR with the hash value to make hash-ordering different for each table.
98//
99// [1]: https://codecapsule.com/2013/11/11/robin-hood-hashing/
100// [2]: https://codecapsule.com/2013/11/17/robin-hood-hashing-backward-shift-deletion/
101// [3]: https://accidentallyquadratic.tumblr.com/post/153545455987/rust-hash-iteration-reinsertion
102template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
103class RobinHoodHashTable {
104public:
105 using HashTableType = RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>;
106 using iterator = HashTableIterator<HashTableType, Key, Value, Extractor, HashFunctions, Traits, KeyTraits>;
107 using const_iterator = HashTableConstIterator<HashTableType, Key, Value, Extractor, HashFunctions, Traits, KeyTraits>;
108 using ValueTraits = Traits;
109 using KeyType = Key;
110 using ValueType = Value;
111 using IdentityTranslatorType = IdentityHashTranslator<ValueTraits, HashFunctions>;
112 using AddResult = HashTableAddResult<iterator>;

114 static constexpr unsigned probeDistanceThreshold = 128;

116 static_assert(!KeyTraits::hasIsReleasedWeakValueFunction);
117 static_assert(HashFunctions::hasHashInValue);

122 {
123 invalidateIterators(this);
124 if (m_table)
125 deallocateTable(m_table, tableSize());
126 }

132 RobinHoodHashTable(RobinHoodHashTable&&);

135 // When the hash table is empty, just return the same iterator for end as for begin.
136 // This is more efficient because we don't have to skip all the empty and deleted
137 // buckets, and iterating an empty table is a common case that's worth optimizing.
138 iterator begin() { return isEmpty() ? end() : makeIterator(m_table); }
139 iterator end() { return makeKnownGoodIterator(m_table + tableSize()); }
140 const_iterator begin() const { return isEmpty() ? end() : makeConstIterator(m_table); }
141 const_iterator end() const { return makeKnownGoodConstIterator(m_table + tableSize()); }
142
143 iterator random()
144 {
145 if (isEmpty())
146 return end();
147
148 while (true) {
149 auto& bucket = m_table[weakRandomUint32() & tableSizeMask()];
150 if (!isEmptyBucket(bucket))
151 return makeKnownGoodIterator(&bucket);
152 }
153 }
154
155 const_iterator random() const { return static_cast<const_iterator>(const_cast<RobinHoodHashTable*>(this)->random()); }
156
157 unsigned size() const { return keyCount(); }
158 unsigned capacity() const { return tableSize(); }
159 bool isEmpty() const { return !keyCount(); }
160
161 void reserveInitialCapacity(unsigned keyCount)
162 {
163 ASSERT(!m_table);
164 ASSERT(!tableSize());
165
166 unsigned minimumTableSize = KeyTraits::minimumTableSize;
167 unsigned newTableSize = std::max(minimumTableSize, computeBestTableSize(keyCount));
168
169 m_table = allocateTable(newTableSize);
170 m_tableSize = newTableSize;
171 m_keyCount = 0;
172 m_tableHash = computeTableHash(m_table);
173 m_willExpand = false;
174 internalCheckTableConsistency();
175 }

177 AddResult add(const ValueType& value) { return add<IdentityTranslatorType>(Extractor::extract(value), value); }
178 AddResult add(ValueType&& value) { return add<IdentityTranslatorType>(Extractor::extract(value), WTFMove(value)); }

181 // with some other type, to avoid the cost of type conversion if the object is already

183 template<typename HashTranslator, typename T, typename Extra> AddResult add(T&& key, Extra&&);
184 template<typename HashTranslator, typename T, typename Extra> AddResult addPassingHashCode(T&& key, Extra&&);

186 iterator find(const KeyType& key) { return find<IdentityTranslatorType>(key); }
187 const_iterator find(const KeyType& key) const { return find<IdentityTranslatorType>(key); }
188 bool contains(const KeyType& key) const { return contains<IdentityTranslatorType>(key); }

221 static constexpr bool shouldExpand(uint64_t keyCount, uint64_t tableSize)
222 {
223 return keyCount * maxLoadDenominator >= tableSize * maxLoadNumerator;
224 }

226private:
227 static ValueType* allocateTable(unsigned size);
228 static void deallocateTable(ValueType* table, unsigned size);

237 void removeAndInvalidate(ValueType*);

240 static unsigned computeTableHash(ValueType* table) { return DefaultHash<ValueType*>::hash(table); }
241
242 static constexpr unsigned computeBestTableSize(unsigned keyCount);
243 bool shouldExpand() const
244 {
245 unsigned keyCount = this->keyCount();
246 unsigned tableSize = this->tableSize();
247 if (shouldExpand(keyCount, tableSize))
248 return true;
249 // If probe-length exceeds probeDistanceThreshold, and 50%~ is filled, expand the table.
250 return m_willExpand && keyCount * 2 >= tableSize;
251 }
252 bool shouldShrink() const { return keyCount() * minLoad < tableSize() && tableSize() > KeyTraits::minimumTableSize; }
253 void expand();
254 void shrink() { rehash(tableSize() / 2); }
255 void shrinkToBestSize();
256
257 void rehash(unsigned newTableSize);
258 void reinsert(ValueType&&);
259
260 static void initializeBucket(ValueType& bucket);
261 static void deleteBucket(ValueType& bucket) { hashTraitsDeleteBucket<Traits>(bucket); }
262
263 static constexpr unsigned desiredIndex(unsigned hash, unsigned sizeMask)
264 {
265 return hash & sizeMask;
266 }
267
268 static constexpr unsigned probeDistance(unsigned hash, unsigned index, unsigned size, unsigned sizeMask)
269 {
270 return (index + size - desiredIndex(hash, sizeMask)) & sizeMask;
271 }
272
273 iterator makeIterator(ValueType* pos) { return iterator(this, pos, m_table + tableSize()); }
274 const_iterator makeConstIterator(ValueType* pos) const { return const_iterator(this, pos, m_table + tableSize()); }
275 iterator makeKnownGoodIterator(ValueType* pos) { return iterator(this, pos, m_table + tableSize(), HashItemKnownGood); }
276 const_iterator makeKnownGoodConstIterator(ValueType* pos) const { return const_iterator(this, pos, m_table + tableSize(), HashItemKnownGood); }
277
278#if ASSERT_ENABLED
279 void checkTableConsistencyExceptSize() const;
280#else
281 static void checkTableConsistencyExceptSize() { }
282#endif
283
284 static constexpr unsigned maxLoadNumerator = SizePolicy::maxLoadNumerator;
285 static constexpr unsigned maxLoadDenominator = SizePolicy::maxLoadDenominator;
286 static constexpr unsigned minLoad = SizePolicy::minLoad;
287
288 unsigned tableSize() const { return m_tableSize; }
289 unsigned tableSizeMask() const { return m_tableSize - 1; }
290 unsigned keyCount() const { return m_keyCount; }
291 unsigned tableHash() const { return m_tableHash; }
292
293 ValueType* m_table { nullptr };
294 unsigned m_tableSize { 0 };
295 unsigned m_keyCount { 0 };
296 unsigned m_tableHash { 0 };
297 bool m_willExpand { false };
298
299#if CHECK_HASHTABLE_ITERATORS
300public:
301 // All access to m_iterators should be guarded with m_mutex.
302 mutable const_iterator* m_iterators { nullptr };
303 // Use std::unique_ptr so HashTable can still be memmove'd or memcpy'ed.
304 mutable std::unique_ptr<Lock> m_mutex { makeUnique<Lock>() };
305#endif
306};

318template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
319template<typename HashTranslator, typename T>
320void RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::checkKey(const T& key)
321{
322 if (!HashFunctions::safeToCompareToEmptyOrDeleted)
323 return;
324 ASSERT(!HashTranslator::equal(KeyTraits::emptyValue(), key));
325 typename std::aligned_storage<sizeof(ValueType), std::alignment_of<ValueType>::value>::type deletedValueBuffer;
326 ValueType* deletedValuePtr = reinterpret_cast_ptr<ValueType*>(&deletedValueBuffer);
327 ValueType& deletedValue = *deletedValuePtr;
328 Traits::constructDeletedValue(deletedValue);
329 ASSERT(!HashTranslator::equal(Extractor::extract(deletedValue), key));
330}
331
332#endif // ASSERT_ENABLED

341template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
342template<typename HashTranslator, typename T>
343ALWAYS_INLINE auto RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::inlineLookup(const T& key) -> ValueType*
344{
345 checkKey<HashTranslator>(key);
346
347 ValueType* table = m_table;
348 if (!table)
349 return nullptr;
350
351 unsigned size = tableSize();
352 unsigned sizeMask = tableSizeMask();
353 unsigned tableHash = this->tableHash();
354 unsigned hash = HashTranslator::hash(key) ^ tableHash;
355 unsigned index = desiredIndex(hash, sizeMask);
356 unsigned distance = 0;
357
358 while (true) {
359 ValueType* entry = m_table + index;
360 if (isEmptyBucket(*entry))
361 return nullptr;
362
363 // RobinHoodHashTable maintains this invariant so that we can stop
364 // probing when distance exceeds probing distance of the existing entry.
365 auto& entryKey = Extractor::extract(*entry);
366 unsigned entryHash = IdentityTranslatorType::hash(entryKey) ^ tableHash;
367 if (distance > probeDistance(entryHash, index, size, sizeMask))
368 return nullptr;
369
370 if (entryHash == hash && HashTranslator::equal(entryKey, key))
371 return entry;
372
373 index = (index + 1) & sizeMask;
374 ++distance;
375 }
376}

384template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
385template<typename HashTranslator, typename T, typename Extra>
386ALWAYS_INLINE auto RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::add(T&& key, Extra&& extra) -> AddResult
387{
388 checkKey<HashTranslator>(key);
389
390 invalidateIterators(this);
391
392 // We should expand before potentially inserting an entry. If we expand a table after inserting an entry,
393 // we need to re-look up entry from the table since the inserted entry position is not stable during rehasing.
394 if (shouldExpand())
395 expand();
396
397 internalCheckTableConsistency();
398
399 ASSERT(m_table);
400
401 unsigned size = tableSize();
402 unsigned sizeMask = tableSizeMask();
403 unsigned tableHash = this->tableHash();
404 unsigned hash = HashTranslator::hash(key) ^ tableHash;
405 unsigned index = desiredIndex(hash, sizeMask);
406 unsigned distance = 0;
407
408 ValueType* entry = nullptr;
409 while (true) {
410 entry = m_table + index;
411 if (isEmptyBucket(*entry)) {
412 if (distance >= probeDistanceThreshold)
413 m_willExpand = true;
414 HashTranslator::translate(*entry, std::forward<T>(key), std::forward<Extra>(extra));
415 break;
416 }
417
418 auto& entryKey = Extractor::extract(*entry);
419 unsigned entryHash = IdentityTranslatorType::hash(entryKey) ^ tableHash;
420 unsigned entryDistance = probeDistance(entryHash, index, size, sizeMask);
421 if (distance > entryDistance) {
422 if (distance >= probeDistanceThreshold)
423 m_willExpand = true;
424 // Start swapping existing entry to maintain probe-distance invariant.
425 ValueType existingEntry = WTFMove(*entry);
426 entry->~ValueType();
427 initializeBucket(*entry);
428 HashTranslator::translate(*entry, std::forward<T>(key), std::forward<Extra>(extra));
429 maintainProbeDistanceForAdd(WTFMove(existingEntry), index, entryDistance, size, sizeMask, tableHash);
430 break;
431 }
432
433 if (entryHash == hash && HashTranslator::equal(entryKey, key))
434 return AddResult(makeKnownGoodIterator(entry), false);
435
436 index = (index + 1) & sizeMask;
437 ++distance;
438 }
439
440 m_keyCount += 1;
441
442 internalCheckTableConsistency();
443
444 return AddResult(makeKnownGoodIterator(entry), true);
445}

447template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
448ALWAYS_INLINE void RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::maintainProbeDistanceForAdd(ValueType&& value, unsigned index, unsigned distance, unsigned size, unsigned sizeMask, unsigned tableHash)
449{
450 using std::swap; // For C++ ADL.
451 index = (index + 1) & sizeMask;
452 ++distance;
453
454 while (true) {
455 ValueType* entry = m_table + index;
456 if (isEmptyBucket(*entry)) {
457 ValueTraits::assignToEmpty(*entry, WTFMove(value));
458 return;
459 }
460
461 unsigned entryHash = IdentityTranslatorType::hash(Extractor::extract(*entry)) ^ tableHash;
462 unsigned entryDistance = probeDistance(entryHash, index, size, sizeMask);
463 if (distance > entryDistance) {
464 swap(value, *entry);
465 distance = entryDistance;
466 }
467
468 index = (index + 1) & sizeMask;
469 ++distance;
470 }
471}

474template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
475template<typename HashTranslator, typename T, typename Extra>
476inline auto RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::addPassingHashCode(T&& key, Extra&& extra) -> AddResult
477{
478 checkKey<HashTranslator>(key);
479
480 invalidateIterators(this);
481
482 // We should expand before potentially inserting an entry. If we expand a table after inserting an entry,
483 // we need to re-look up entry from the table since the inserted entry position is not stable during rehasing.
484 if (shouldExpand())
485 expand();
486
487 internalCheckTableConsistency();
488
489 ASSERT(m_table);
490
491 unsigned size = tableSize();
492 unsigned sizeMask = tableSizeMask();
493 unsigned tableHash = this->tableHash();
494 unsigned originalHash = HashTranslator::hash(key);
495 unsigned hash = originalHash ^ tableHash;
496 unsigned index = desiredIndex(hash, sizeMask);
497 unsigned distance = 0;
498
499 ValueType* entry = nullptr;
500 while (true) {
501 entry = m_table + index;
502 if (isEmptyBucket(*entry)) {
503 if (distance >= probeDistanceThreshold)
504 m_willExpand = true;
505 HashTranslator::translate(*entry, std::forward<T>(key), std::forward<Extra>(extra), originalHash);
506 break;
507 }
508
509 auto& entryKey = Extractor::extract(*entry);
510 unsigned entryHash = IdentityTranslatorType::hash(entryKey) ^ tableHash;
511 unsigned entryDistance = probeDistance(entryHash, index, size, sizeMask);
512 if (distance > entryDistance) {
513 if (distance >= probeDistanceThreshold)
514 m_willExpand = true;
515 // Start swapping existing entry to maintain probe-distance invariant.
516 ValueType existingEntry = WTFMove(*entry);
517 entry->~ValueType();
518 initializeBucket(*entry);
519 HashTranslator::translate(*entry, std::forward<T>(key), std::forward<Extra>(extra), originalHash);
520 maintainProbeDistanceForAdd(WTFMove(existingEntry), index, entryDistance, size, sizeMask, tableHash);
521 break;
522 }
523
524 if (entryHash == hash && HashTranslator::equal(entryKey, key))
525 return AddResult(makeKnownGoodIterator(entry), false);
526
527 index = (index + 1) & sizeMask;
528 ++distance;
529 }
530
531 m_keyCount += 1;
532
533 internalCheckTableConsistency();
534
535 return AddResult(makeKnownGoodIterator(entry), true);
536}

560 distance = entryDistance;
561 }
562
563 index = (index + 1) & sizeMask;
564 ++distance;
565 }

621template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
622void RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::remove(ValueType* pos)
623{
624 // This is removal via "backward-shift-deletion". This basically shift existing entries to removed empty entry place so that we make
625 // the table as if no removal happened so far. This decreases distance-to-initial-bucket (DIB) of the subsequent entries by 1. This maintains
626 // DIB of the table low and relatively constant even if we have many removals, compared to using tombstones.
627 // https://codecapsule.com/2013/11/17/robin-hood-hashing-backward-shift-deletion/
628 deleteBucket(*pos);
629 initializeBucket(*pos);
630 m_keyCount -= 1;
631
632 unsigned size = tableSize();
633 unsigned sizeMask = tableSizeMask();
634 unsigned tableHash = this->tableHash();
635 unsigned indexPrevious = pos - m_table;
636 unsigned index = (indexPrevious + 1) & sizeMask;
637
638 while (true) {
639 Value* previousEntry = m_table + indexPrevious;
640 Value* entry = m_table + index;
641 if (isEmptyBucket(*entry))
642 break;
643
644 ASSERT(isEmptyBucket(*previousEntry));
645 auto& entryKey = Extractor::extract(*entry);
646 unsigned entryHash = IdentityTranslatorType::hash(entryKey) ^ tableHash;
647 if (!probeDistance(entryHash, index, size, sizeMask))
648 break;
649
650 ValueTraits::assignToEmpty(*previousEntry, WTFMove(*entry));
651 entry->~ValueType();
652 initializeBucket(*entry);
653
654 indexPrevious = index;
655 index = (index + 1) & sizeMask;
656 }
657
658 if (shouldShrink())
659 shrink();
660
661 internalCheckTableConsistency();
662}

732template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
733constexpr unsigned RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::computeBestTableSize(unsigned keyCount)
734{
735 unsigned bestTableSize = WTF::roundUpToPowerOfTwo(keyCount);
736
737 if (shouldExpand(keyCount, bestTableSize))
738 bestTableSize *= 2;
739
740 auto aboveThresholdForEagerExpansion = [](double loadFactor, unsigned keyCount, unsigned tableSize)
741 {
742 // Here is the rationale behind this calculation, using 3/4 load-factor.
743 // With maxLoad at 3/4 and minLoad at 1/6, our average load is 11/24.
744 // If we are getting half-way between 11/24 and 3/4, we double the size
745 // to avoid being too close to loadMax and bring the ratio close to 11/24. This
746 // give us a load in the bounds [9/24, 15/24).
747 double maxLoadRatio = loadFactor;
748 double minLoadRatio = 1.0 / minLoad;
749 double averageLoadRatio = (maxLoadRatio + minLoadRatio) / 2;
750 double halfWayBetweenAverageAndMaxLoadRatio = (averageLoadRatio + maxLoadRatio) / 2;
751 return keyCount >= tableSize * halfWayBetweenAverageAndMaxLoadRatio;
752 };
753
754 constexpr double loadFactor = static_cast<double>(maxLoadNumerator) / maxLoadDenominator;
755 if (aboveThresholdForEagerExpansion(loadFactor, keyCount, bestTableSize))
756 bestTableSize *= 2;
757 unsigned minimumTableSize = KeyTraits::minimumTableSize;
758 return std::max(bestTableSize, minimumTableSize);
759}

768template<typename Key, typename Value, typename Extractor, typename HashFunctions, typename Traits, typename KeyTraits, typename SizePolicy>
769void RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::rehash(unsigned newTableSize)
770{
771 internalCheckTableConsistencyExceptSize();
772
773 unsigned oldTableSize = tableSize();
774 ValueType* oldTable = m_table;
775
776 m_table = allocateTable(newTableSize);
777 m_tableSize = newTableSize;
778 m_tableHash = computeTableHash(m_table);
779 m_willExpand = false;
780
781 for (unsigned i = 0; i < oldTableSize; ++i) {
782 auto* oldEntry = oldTable + i;
783 if (!isEmptyBucket(*oldEntry))
784 reinsert(WTFMove(*oldEntry));
785 oldEntry->~ValueType();
786 }
787
788 if (oldTable)
789 HashTableMalloc::free(reinterpret_cast<char*>(oldTable));
790
791 internalCheckTableConsistency();
792}

811RobinHoodHashTable<Key, Value, Extractor, HashFunctions, Traits, KeyTraits, SizePolicy>::RobinHoodHashTable(const RobinHoodHashTable& other)
812{
813 if (!other.m_tableSize || !other.m_keyCount)
814 return;
815
816 m_table = allocateTable(other.m_tableSize);
817 m_tableSize = other.m_tableSize;
818 m_keyCount = other.m_keyCount;
819 m_tableHash = computeTableHash(m_table);
820 m_willExpand = other.m_willExpand;
821
822 for (unsigned index = 0; index < other.m_tableSize; ++index) {
823 ValueType& otherEntry = other.m_table[index];
824 if (!isEmptyBucket(otherEntry)) {
825 ValueType entry(otherEntry);
826 reinsert(WTFMove(entry));
827 }
828 }
829 internalCheckTableConsistency();
830}

891{

898 ValueType* entry = m_table + i;
899 if (isEmptyBucket(*entry))

911 ASSERT(this->tableSize() >= KeyTraits::minimumTableSize);
912 ASSERT(tableSizeMask());

914}
