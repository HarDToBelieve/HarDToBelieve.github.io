40class Seconds final {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 constexpr Seconds() { }
44
45 explicit constexpr Seconds(double value)
46 : m_value(value)
47 {
48 }
49
50 constexpr double value() const { return m_value; }
51
52 constexpr double minutes() const { return m_value / 60; }
53 constexpr double seconds() const { return m_value; }
54 constexpr double milliseconds() const { return seconds() * 1000; }
55 constexpr double microseconds() const { return milliseconds() * 1000; }
56 constexpr double nanoseconds() const { return microseconds() * 1000; }
57
58 // Keep in mind that Seconds is held in double. If the value is not in range of 53bit integer, the result may not be precise.
59 template<typename T> T minutesAs() const { static_assert(std::is_integral<T>::value, ""); return clampToAccepting64<T>(minutes()); }
60 template<typename T> T secondsAs() const { static_assert(std::is_integral<T>::value, ""); return clampToAccepting64<T>(seconds()); }
61 template<typename T> T millisecondsAs() const { static_assert(std::is_integral<T>::value, ""); return clampToAccepting64<T>(milliseconds()); }
62 template<typename T> T microsecondsAs() const { static_assert(std::is_integral<T>::value, ""); return clampToAccepting64<T>(microseconds()); }
63 template<typename T> T nanosecondsAs() const { static_assert(std::is_integral<T>::value, ""); return clampToAccepting64<T>(nanoseconds()); }
64
65 static constexpr Seconds fromMinutes(double minutes)
66 {
67 return Seconds(minutes * 60);
68 }
69
70 static constexpr Seconds fromHours(double hours)
71 {
72 return Seconds(hours * 3600);
73 }
74
75 static constexpr Seconds fromMilliseconds(double milliseconds)
76 {
77 return Seconds(milliseconds / 1000);
78 }
79
80 static constexpr Seconds fromMicroseconds(double microseconds)
81 {
82 return fromMilliseconds(microseconds / 1000);
83 }
84
85 static constexpr Seconds fromNanoseconds(double nanoseconds)
86 {
87 return fromMicroseconds(nanoseconds / 1000);
88 }
89
90 static constexpr Seconds infinity()
91 {
92 return Seconds(std::numeric_limits<double>::infinity());
93 }
94
95 static constexpr Seconds nan()
96 {
97 return Seconds(std::numeric_limits<double>::quiet_NaN());
98 }
99
100 explicit constexpr operator bool() const { return !!m_value; }
101
102 constexpr Seconds operator+(Seconds other) const
103 {
104 return Seconds(value() + other.value());
105 }
106
107 constexpr Seconds operator-(Seconds other) const
108 {
109 return Seconds(value() - other.value());
110 }
111
112 constexpr Seconds operator-() const
113 {
114 return Seconds(-value());
115 }
116
117 // It makes sense to consider scaling a duration, like, "I want to wait 5 times as long as
118 // last time!".
119 constexpr Seconds operator*(double scalar) const
120 {
121 return Seconds(value() * scalar);
122 }
123
124 constexpr Seconds operator/(double scalar) const
125 {
126 return Seconds(value() / scalar);
127 }
128
129 // It's reasonable to think about ratios between Seconds.
130 constexpr double operator/(Seconds other) const
131 {
132 return value() / other.value();
133 }
134
135 Seconds operator%(double scalar) const
136 {
137 return Seconds(fmod(value(), scalar));
138 }
139
140 // This solves for r, where:
141 //
142 // floor(this / other) + r / other = this / other
143 //
144 // Therefore, if this is Seconds then r is Seconds.
145 Seconds operator%(Seconds other) const
146 {
147 return Seconds(fmod(value(), other.value()));
148 }
149
150 Seconds& operator+=(Seconds other)
151 {
152 return *this = *this + other;
153 }
154
155 Seconds& operator-=(Seconds other)
156 {
157 return *this = *this - other;
158 }
159
160 Seconds& operator*=(double scalar)
161 {
162 return *this = *this * scalar;
163 }
164
165 Seconds& operator/=(double scalar)
166 {
167 return *this = *this / scalar;
168 }
169
170 Seconds& operator%=(double scalar)
171 {
172 return *this = *this % scalar;
173 }
174
175 Seconds& operator%=(Seconds other)
176 {
177 return *this = *this % other;
178 }

188 constexpr bool operator==(Seconds other) const
189 {
190 return m_value == other.m_value;
191 }
192
193 constexpr bool operator!=(Seconds other) const
194 {
195 return m_value != other.m_value;
196 }
197
198 constexpr bool operator<(Seconds other) const
199 {
200 return m_value < other.m_value;
201 }
202
203 constexpr bool operator>(Seconds other) const
204 {
205 return m_value > other.m_value;
206 }
207
208 constexpr bool operator<=(Seconds other) const
209 {
210 return m_value <= other.m_value;
211 }
212
213 constexpr bool operator>=(Seconds other) const
214 {
215 return m_value >= other.m_value;
216 }

218 WTF_EXPORT_PRIVATE void dump(PrintStream&) const;
219
220 Seconds isolatedCopy() const
221 {
222 return *this;
223 }
224
225 template<class Encoder>
226 void encode(Encoder& encoder) const
227 {
228 encoder << m_value;
229 }
230
231 template<class Decoder>
232 static std::optional<Seconds> decode(Decoder& decoder)
233 {
234 std::optional<double> seconds;
235 decoder >> seconds;
236 if (!seconds)
237 return std::nullopt;
238 return Seconds(*seconds);
239 }
240
241 template<class Decoder>
242 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, Seconds& seconds)
243 {
244 double value;
245 if (!decoder.decode(value))
246 return false;
247
248 seconds = Seconds(value);
249 return true;
250 }
251
252 struct MarkableTraits;
253
254private:
255 double m_value { 0 };
256};

260struct Seconds::MarkableTraits {
261 static bool isEmptyValue(Seconds seconds)
262 {
263 return std::isnan(seconds.value());
264 }
265
266 static constexpr Seconds emptyValue()
267 {
268 return Seconds::nan();
269 }
270};
