31#if !HAVE(STRNSTR)
32
33inline char* strnstr(const char* buffer, const char* target, size_t bufferLength)
34{
35 size_t targetLength = strlen(target);
36 if (targetLength == 0)
37 return const_cast<char*>(buffer);
38 for (const char* start = buffer; *start && start + targetLength <= buffer + bufferLength; start++) {
39 if (*start == *target && strncmp(start + 1, target + 1, targetLength - 1) == 0)
40 return const_cast<char*>(start);
41 }
42 return nullptr;
43}

47#if !HAVE(MEMMEM)
48
49inline const void* memmem(const void* haystack, size_t haystackLength, const void* needle, size_t needleLength)
50{
51 const char* pointer = static_cast<const char*>(haystack);
52 while (haystackLength >= needleLength) {
53 if (!memcmp(pointer, needle, needleLength))
54 return pointer;
55 pointer++;
56 haystackLength--;
57 }
58 return nullptr;
59}
