26#pragma once
27
28#include <wtf/PtrTag.h>
29
30namespace WTF {
31
32template<typename PtrType>
33class TaggedArrayStoragePtr {
34public:
35 TaggedArrayStoragePtr()
36 : m_ptr(tagArrayPtr<PtrType>(nullptr, 0))
37 { }
38
39 TaggedArrayStoragePtr(PtrType* ptr, unsigned length)
40 : m_ptr(tagArrayPtr(ptr, length))
41 { }
42
43 PtrType* get(unsigned length) const { return untagArrayPtr(m_ptr, length); }
44 PtrType* getUnsafe() const { return removeArrayPtrTag(m_ptr); }
45
46 void resize(unsigned oldLength, unsigned newLength)
47 {
48 m_ptr = retagArrayPtr(m_ptr, oldLength, newLength);
49 }
50
51 explicit operator bool() const { return !!getUnsafe(); }
52
53private:
54 PtrType* m_ptr;
55};
56
57}
