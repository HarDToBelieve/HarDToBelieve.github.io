26#pragma once
27
28#include <wtf/ScopedLambda.h>
29#include <wtf/Threading.h>
30
31namespace WTF {
32
33using ThreadMessage = ScopedLambda<void(PlatformRegisters&)>;
34
35enum class MessageStatus {
36 MessageRan,
37 ThreadExited,
38};
39
40// This method allows us to send a message which will be run in a signal handler on the desired thread.
41// There are several caveates to this method however, This function uses signals so your message should
42// be sync signal safe.
43WTF_EXPORT_PRIVATE MessageStatus sendMessageScoped(Thread&, const ThreadMessage&);
44
45template<typename Functor>
46MessageStatus sendMessage(Thread& targetThread, const Functor& func)
47{
48 auto lambda = scopedLambdaRef<void(PlatformRegisters&)>(func);
49 return sendMessageScoped(targetThread, lambda);
50}
51
52} // namespace WTF
53
54using WTF::sendMessage;
