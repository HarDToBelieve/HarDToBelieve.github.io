26#pragma once
27
28#include <memory>
29#include <wtf/Assertions.h>
30
31namespace WTF {
32
33template<typename T> class UniqueRef;
34
35template<typename T, class... Args>
36UniqueRef<T> makeUniqueRefWithoutFastMallocCheck(Args&&... args)
37{
38 return UniqueRef<T>(*new T(std::forward<Args>(args)...));
39}
40
41template<typename T, class... Args>
42UniqueRef<T> makeUniqueRef(Args&&... args)
43{
44 static_assert(std::is_same<typename T::webkitFastMalloced, int>::value, "T is FastMalloced");
45 return makeUniqueRefWithoutFastMallocCheck<T>(std::forward<Args>(args)...);
46}
47
48template<typename T>
49UniqueRef<T> makeUniqueRefFromNonNullUniquePtr(std::unique_ptr<T> ptr)
50{
51 return UniqueRef<T>(*ptr.release());
52}
53
54template<typename T>
55class UniqueRef {
56public:
57 template <typename U>
58 UniqueRef(UniqueRef<U>&& other)
59 : m_ref(other.m_ref.release())
60 {
61 ASSERT(m_ref);
62 }
63
64 explicit UniqueRef(T& other)
65 : m_ref(&other)
66 {
67 ASSERT(m_ref);
68 }
69
70 T& get() { ASSERT(m_ref); return *m_ref; }
71 const T& get() const { ASSERT(m_ref); return *m_ref; }
72
73 T* operator&() { ASSERT(m_ref); return m_ref.get(); }
74 const T* operator&() const { ASSERT(m_ref); return m_ref.get(); }
75
76 T* operator->() { ASSERT(m_ref); return m_ref.get(); }
77 const T* operator->() const { ASSERT(m_ref); return m_ref.get(); }
78
79 operator T&() { ASSERT(m_ref); return *m_ref; }
80 operator const T&() const { ASSERT(m_ref); return *m_ref; }
81
82 std::unique_ptr<T> moveToUniquePtr() { return WTFMove(m_ref); }
83
84private:
85 template<class U, class... Args> friend UniqueRef<U> makeUniqueRefWithoutFastMallocCheck(Args&&...);
86 template<class U> friend UniqueRef<U> makeUniqueRefFromNonNullUniquePtr(std::unique_ptr<U>);
87 template<class U> friend class UniqueRef;
88
89 std::unique_ptr<T> m_ref;
90};
91
92} // namespace WTF
