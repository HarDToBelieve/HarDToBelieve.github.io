28#include <wtf/Algorithms.h>
29#include <wtf/HashSet.h>
30#include <wtf/WeakPtr.h>

32namespace WTF {
33
34template<typename T, typename Counter = EmptyCounter>
35class WeakHashSet final {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 typedef HashSet<Ref<WeakPtrImpl<Counter>>> WeakPtrImplSet;
39 typedef typename WeakPtrImplSet::AddResult AddResult;
40
41 class WeakHashSetConstIterator : public std::iterator<std::forward_iterator_tag, T, std::ptrdiff_t, const T*, const T&> {
42 private:
43 WeakHashSetConstIterator(const WeakPtrImplSet& set, typename WeakPtrImplSet::const_iterator position)
44 : m_position(position), m_endPosition(set.end())
45 {
46 skipEmptyBuckets();
47 }
48
49 public:
50 T* get() const { return static_cast<T*>((*m_position)->template get<T>()); }
51 T& operator*() const { return *get(); }
52 T* operator->() const { return get(); }
53
54 WeakHashSetConstIterator& operator++()
55 {
56 ASSERT(m_position != m_endPosition);
57 ++m_position;
58 skipEmptyBuckets();
59 return *this;
60 }
61
62 void skipEmptyBuckets()
63 {
64 while (m_position != m_endPosition && !get())
65 ++m_position;
66 }
67
68 bool operator==(const WeakHashSetConstIterator& other) const
69 {
70 return m_position == other.m_position;
71 }
72
73 bool operator!=(const WeakHashSetConstIterator& other) const
74 {
75 return m_position != other.m_position;
76 }
77
78 private:
79 template <typename, typename> friend class WeakHashSet;
80
81 typename WeakPtrImplSet::const_iterator m_position;
82 typename WeakPtrImplSet::const_iterator m_endPosition;
83 };
84 typedef WeakHashSetConstIterator const_iterator;
85
86 WeakHashSet() { }
87
88 const_iterator begin() const { return WeakHashSetConstIterator(m_set, m_set.begin()); }
89 const_iterator end() const { return WeakHashSetConstIterator(m_set, m_set.end()); }
90
91 template <typename U>
92 AddResult add(const U& value)
93 {
94 return m_set.add(*makeWeakPtr<T>(const_cast<U&>(value)).m_impl);
95 }
96
97 template <typename U>
98 bool remove(const U& value)
99 {
100 auto& weakPtrImpl = value.weakPtrFactory().m_impl;
101 if (!weakPtrImpl || !*weakPtrImpl)
102 return false;
103 return m_set.remove(*weakPtrImpl);
104 }
105
106 void clear() { m_set.clear(); }
107
108 template <typename U>
109 bool contains(const U& value) const
110 {
111 auto& weakPtrImpl = value.weakPtrFactory().m_impl;
112 if (!weakPtrImpl || !*weakPtrImpl)
113 return false;
114 return m_set.contains(*weakPtrImpl);
115 }
116
117 unsigned capacity() const { return m_set.capacity(); }
118
119 bool computesEmpty() const { return begin() == end(); }
120
121 bool hasNullReferences() const
122 {
123 return WTF::anyOf(m_set, [] (auto& value) { return !value.get(); });
124 }
125
126 unsigned computeSize() const
127 {
128 const_cast<WeakPtrImplSet&>(m_set).removeIf([] (auto& value) { return !value.get(); });
129 return m_set.size();
130 }
131
132 void forEach(const Function<void(T&)>& callback)
133 {
134 auto items = map(m_set, [](const Ref<WeakPtrImpl<Counter>>& item) {
135 auto* pointer = static_cast<T*>(item->template get<T>());
136 return makeWeakPtr(pointer);
137 });
138 for (auto& item : items) {
139 if (item && m_set.contains(*item.m_impl))
140 callback(*item);
141 }
142 }
143
144#if ASSERT_ENABLED
145 void checkConsistency() const { m_set.checkConsistency(); }
146#else
147 void checkConsistency() const { }
148#endif
149
150private:
151 WeakPtrImplSet m_set;
152};
153
154template<typename MapFunction, typename T>
155struct Mapper<MapFunction, const WeakHashSet<T> &, void> {
156 using SourceItemType = T&;
157 using DestinationItemType = typename std::result_of<MapFunction(SourceItemType&)>::type;
158
159 static Vector<DestinationItemType> map(const WeakHashSet<T>& source, const MapFunction& mapFunction)
160 {
161 Vector<DestinationItemType> result;
162 result.reserveInitialCapacity(source.computeSize());
163 for (auto& item : source)
164 result.uncheckedAppend(mapFunction(item));
165 return result;
166 }
167};
168
169template<typename T>
170inline auto copyToVector(const WeakHashSet<T>& collection) -> Vector<WeakPtr<T>>
171{
172 return WTF::map(collection, [] (auto& v) -> WeakPtr<T> { return makeWeakPtr<T>(v); });
173}
