35namespace WTF {
36
37class WorkerPool : public ThreadSafeRefCounted<WorkerPool> {
38public:
39 WTF_EXPORT_PRIVATE void postTask(Function<void()>&&);
40
41 WTF_EXPORT_PRIVATE ~WorkerPool();
42
43 // If timeout is infinity, it means AutomaticThread will be never automatically destroyed.
44 static Ref<WorkerPool> create(ASCIILiteral name, unsigned numberOfWorkers = WTF::numberOfProcessorCores(), Seconds timeout = Seconds::infinity())
45 {
46 ASSERT(numberOfWorkers >= 1);
47 return adoptRef(*new WorkerPool(name, numberOfWorkers, timeout));
48 }
49
50 ASCIILiteral name() const { return m_name; }
51
52private:
53 class Worker;
54 friend class Worker;
55
56 WTF_EXPORT_PRIVATE WorkerPool(ASCIILiteral name, unsigned numberOfWorkers, Seconds timeout);
57
58 bool shouldSleep(const AbstractLocker&);
59
60 Box<Lock> m_lock;
61 Ref<AutomaticThreadCondition> m_condition;
62 Seconds m_timeout;
63 MonotonicTime m_lastTimeoutTime { MonotonicTime::nan() };
64 unsigned m_numberOfActiveWorkers { 0 };
65 Vector<Ref<Worker>> m_workers;
66 Deque<Function<void()>> m_tasks;
67 ASCIILiteral m_name;
68};
69
70}
