43template<typename T, typename U> struct Coder<std::pair<T, U>> {
44 static void encode(Encoder& encoder, const std::pair<T, U>& pair)
45 {
46 encoder << pair.first << pair.second;
47 }
48
49 static std::optional<std::pair<T, U>> decode(Decoder& decoder)
50 {
51 std::optional<T> first;
52 decoder >> first;
53 if (!first)
54 return std::nullopt;
55
56 std::optional<U> second;
57 decoder >> second;
58 if (!second)
59 return std::nullopt;
60
61 return {{ WTFMove(*first), WTFMove(*second) }};
62 }
63};
64
65template<typename T> struct Coder<std::optional<T>> {
66 static void encode(Encoder& encoder, const std::optional<T>& optional)
67 {
68 if (!optional) {
69 encoder << false;
70 return;
71 }
72
73 encoder << true;
74 encoder << optional.value();
75 }
76
77 static std::optional<std::optional<T>> decode(Decoder& decoder)
78 {
79 std::optional<bool> isEngaged;
80 decoder >> isEngaged;
81 if (!isEngaged)
82 return std::nullopt;
83 if (!*isEngaged)
84 return std::optional<std::optional<T>> { std::optional<T> { std::nullopt } };
85
86 std::optional<T> value;
87 decoder >> value;
88 if (!value)
89 return std::nullopt;
90
91 return std::optional<std::optional<T>> { std::optional<T> { WTFMove(*value) } };
92 }
93};

119template<typename T, size_t inlineCapacity> struct VectorCoder<false, T, inlineCapacity> {
120 static void encode(Encoder& encoder, const Vector<T, inlineCapacity>& vector)
121 {
122 encoder << static_cast<uint64_t>(vector.size());
123 for (size_t i = 0; i < vector.size(); ++i)
124 encoder << vector[i];
125 }
126
127 static std::optional<Vector<T, inlineCapacity>> decode(Decoder& decoder)
128 {
129 std::optional<uint64_t> size;
130 decoder >> size;
131 if (!size)
132 return std::nullopt;
133
134 Vector<T, inlineCapacity> tmp;
135 for (size_t i = 0; i < *size; ++i) {
136 std::optional<T> element;
137 decoder >> element;
138 if (!element)
139 return std::nullopt;
140 tmp.append(WTFMove(*element));
141 }
142
143 tmp.shrinkToFit();
144 return tmp;
145 }
146};

148template<typename T, size_t inlineCapacity> struct VectorCoder<true, T, inlineCapacity> {
149 static void encode(Encoder& encoder, const Vector<T, inlineCapacity>& vector)
150 {
151 encoder << static_cast<uint64_t>(vector.size());
152 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(vector.data()), vector.size() * sizeof(T));
153 }
154
155 static std::optional<Vector<T, inlineCapacity>> decode(Decoder& decoder)
156 {
157 std::optional<uint64_t> decodedSize;
158 decoder >> decodedSize;
159 if (!decodedSize)
160 return std::nullopt;
161
162 if (!isInBounds<size_t>(*decodedSize))
163 return std::nullopt;
164
165 auto size = static_cast<size_t>(*decodedSize);
166
167 // Since we know the total size of the elements, we can allocate the vector in
168 // one fell swoop. Before allocating we must however make sure that the decoder buffer
169 // is big enough.
170 if (!decoder.bufferIsLargeEnoughToContain<T>(size))
171 return std::nullopt;
172
173 Vector<T, inlineCapacity> temp;
174 temp.grow(size);
175
176 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(temp.data()), size * sizeof(T)))
177 return std::nullopt;
178
179 return temp;
180 }
181};

185template<typename KeyArg, typename MappedArg, typename HashArg, typename KeyTraitsArg, typename MappedTraitsArg> struct Coder<HashMap<KeyArg, MappedArg, HashArg, KeyTraitsArg, MappedTraitsArg>> {
186 typedef HashMap<KeyArg, MappedArg, HashArg, KeyTraitsArg, MappedTraitsArg> HashMapType;
187
188 static void encode(Encoder& encoder, const HashMapType& hashMap)
189 {
190 encoder << static_cast<uint64_t>(hashMap.size());
191 for (typename HashMapType::const_iterator it = hashMap.begin(), end = hashMap.end(); it != end; ++it)
192 encoder << *it;
193 }
194
195 static std::optional<HashMapType> decode(Decoder& decoder)
196 {
197 std::optional<uint64_t> hashMapSize;
198 decoder >> hashMapSize;
199 if (!hashMapSize)
200 return std::nullopt;
201
202 HashMapType tempHashMap;
203 tempHashMap.reserveInitialCapacity(static_cast<unsigned>(*hashMapSize));
204 for (uint64_t i = 0; i < *hashMapSize; ++i) {
205 std::optional<KeyArg> key;
206 decoder >> key;
207 if (!key)
208 return std::nullopt;
209 std::optional<MappedArg> value;
210 decoder >> value;
211 if (!value)
212 return std::nullopt;
213
214 if (!tempHashMap.add(WTFMove(*key), WTFMove(*value)).isNewEntry) {
215 // The hash map already has the specified key, bail.
216 return std::nullopt;
217 }
218 }
219
220 return tempHashMap;
221 }
222};

258template<> struct Coder<Seconds> {
259 static void encode(Encoder& encoder, const Seconds& seconds)
260 {
261 encoder << seconds.value();
262 }
263
264 static std::optional<Seconds> decode(Decoder& decoder)
265 {
266 std::optional<double> value;
267 decoder >> value;
268 if (!value)
269 return std::nullopt;
270 return Seconds(*value);
271 }
