26#pragma once
27
28#include <wtf/ASCIICType.h>
29#include <wtf/StdLibExtras.h>
30
31namespace WTF {
32
33class PrintStream;
34
35class ASCIILiteral final {
36public:
37 operator const char*() const { return m_characters; }
38
39 static constexpr ASCIILiteral fromLiteralUnsafe(const char* string)
40 {
41 return ASCIILiteral { string };
42 }
43
44 WTF_EXPORT_PRIVATE void dump(PrintStream& out) const;
45
46 static constexpr ASCIILiteral null()
47 {
48 return ASCIILiteral { nullptr };
49 }
50
51 constexpr const char* characters() const { return m_characters; }
52 const LChar* characters8() const { return bitwise_cast<const LChar*>(m_characters); }
53 size_t length() const { return strlen(m_characters); }
54
55 constexpr char characterAt(unsigned index) const { return m_characters[index]; }
56
57private:
58 constexpr explicit ASCIILiteral(const char* characters) : m_characters(characters) { }
59
60 const char* m_characters;
61};
62
63inline namespace StringLiterals {
64
65constexpr ASCIILiteral operator"" _s(const char* characters, size_t n)
66{
67#if ASSERT_ENABLED
68 for (size_t i = 0; i < n; ++i)
69 ASSERT_UNDER_CONSTEXPR_CONTEXT(isASCII(characters[i]));
70#else
71 UNUSED_PARAM(n);
72#endif
73 return ASCIILiteral::fromLiteralUnsafe(characters);
74}
75
76} // inline StringLiterals
77
78} // namespace WTF
