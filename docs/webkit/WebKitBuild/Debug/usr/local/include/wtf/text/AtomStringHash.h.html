29#pragma once
30
31#include <wtf/text/AtomString.h>
32#include <wtf/HashTraits.h>
33
34namespace WTF {
35
36 struct AtomStringHash {
37 static unsigned hash(const AtomString& key)
38 {
39 return key.impl()->existingHash();
40 }
41
42 static bool equal(const AtomString& a, const AtomString& b)
43 {
44 return a == b;
45 }
46
47 static constexpr bool safeToCompareToEmptyOrDeleted = false;
48 static constexpr bool hasHashInValue = true;
49 };
50
51 template<> struct HashTraits<WTF::AtomString> : SimpleClassHashTraits<WTF::AtomString> {
52 static constexpr bool hasIsEmptyValueFunction = true;
53 static bool isEmptyValue(const AtomString& value)
54 {
55 return value.isNull();
56 }
57
58 static void customDeleteBucket(AtomString& value)
59 {
60 // See unique_ptr's customDeleteBucket() for an explanation.
61 ASSERT(!isDeletedValue(value));
62 AtomString valueToBeDestroyed = WTFMove(value);
63 constructDeletedValue(value);
64 }
65 };
66
67 template<> struct DefaultHash<AtomString> : AtomStringHash { };
68}
69
70using WTF::AtomStringHash;
