38enum class Base64EncodePolicy {
39 DoNotInsertLFs,
40 InsertLFs,
41 URL // No padding, no LFs.
42};
43
44enum class Base64EncodeMap { Default, URL };
45
46enum class Base64DecodeOptions : uint8_t {
47 ValidatePadding = 1 << 0,
48 IgnoreSpacesAndNewLines = 1 << 1,
49 DiscardVerticalTab = 1 << 2,
50};

52enum class Base64DecodeMap { Default, URL };

54struct Base64Specification {

267template<typename CharacterType> bool isBase64OrBase64URLCharacter(CharacterType c)
268{
269 return isASCIIAlphanumeric(c) || c == '+' || c == '/' || c == '-' || c == '_';
270}

272inline unsigned calculateBase64EncodedSize(unsigned inputLength, Base64EncodePolicy policy)
273{
274 if (!inputLength)
275 return 0;
276
277 if (inputLength > maximumBase64EncoderInputBufferSize)
278 return 0;
279
280 auto basePaddedEncodedSize = [&] {
281 return ((inputLength + 2) / 3) * 4;
282 };
283
284 auto baseUnpaddedEncodedSize = [&] {
285 return ((inputLength * 4) + 2) / 3;
286 };
287
288 switch (policy) {
289 case Base64EncodePolicy::DoNotInsertLFs:
290 return basePaddedEncodedSize();
291
292 case Base64EncodePolicy::InsertLFs: {
293 auto result = basePaddedEncodedSize();
294 return result + ((result - 1) / maximumBase64LineLengthWhenInsertingLFs);
295 }
296
297 case Base64EncodePolicy::URL:
298 return baseUnpaddedEncodedSize();
299 }
300
301 ASSERT_NOT_REACHED();
302 return 0;
303}

345template<> class StringTypeAdapter<Base64Specification> {
346public:
347 StringTypeAdapter(const Base64Specification& base64)
348 : m_base64 { base64 }
349 , m_encodedLength { calculateBase64EncodedSize(base64.input.size(), base64.policy) }
350 {
351 }
352
353 unsigned length() const { return m_encodedLength; }
354 bool is8Bit() const { return true; }
355
356 template<typename CharacterType> void writeTo(CharacterType* destination) const
357 {
358 base64Encode(m_base64.input, Span { destination, m_encodedLength }, m_base64.policy, m_base64.map);
359 }
360
361private:
362 Base64Specification m_base64;
363 unsigned m_encodedLength;
364};
