26#pragma once
27
28#include <unicode/utypes.h>
29#include <wtf/Assertions.h>
30#include <wtf/text/LChar.h>
31
32namespace WTF {
33
34template<typename CharacterType>
35class CodePointIterator {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 ALWAYS_INLINE CodePointIterator() { }
39 ALWAYS_INLINE CodePointIterator(const CharacterType* begin, const CharacterType* end)
40 : m_begin(begin)
41 , m_end(end)
42 {
43 }
44
45 ALWAYS_INLINE CodePointIterator(const CodePointIterator& begin, const CodePointIterator& end)
46 : CodePointIterator(begin.m_begin, end.m_begin)
47 {
48 ASSERT(end.m_begin >= begin.m_begin);
49 }
50
51 ALWAYS_INLINE UChar32 operator*() const;
52 ALWAYS_INLINE CodePointIterator& operator++();
53
54 ALWAYS_INLINE bool operator==(const CodePointIterator& other) const
55 {
56 return m_begin == other.m_begin
57 && m_end == other.m_end;
58 }
59 ALWAYS_INLINE bool operator!=(const CodePointIterator& other) const { return !(*this == other); }
60
61 ALWAYS_INLINE bool atEnd() const
62 {
63 ASSERT(m_begin <= m_end);
64 return m_begin >= m_end;
65 }
66
67 ALWAYS_INLINE size_t codeUnitsSince(const CharacterType* reference) const
68 {
69 ASSERT(m_begin >= reference);
70 return m_begin - reference;
71 }
72
73 ALWAYS_INLINE size_t codeUnitsSince(const CodePointIterator& other) const
74 {
75 return codeUnitsSince(other.m_begin);
76 }
77
78private:
79 const CharacterType* m_begin { nullptr };
80 const CharacterType* m_end { nullptr };
81};
82
83template<>
84ALWAYS_INLINE UChar32 CodePointIterator<LChar>::operator*() const
85{
86 ASSERT(!atEnd());
87 return *m_begin;
88}
89
90template<>
91ALWAYS_INLINE auto CodePointIterator<LChar>::operator++() -> CodePointIterator&
92{
93 m_begin++;
94 return *this;
95}
96
97template<>
98ALWAYS_INLINE UChar32 CodePointIterator<UChar>::operator*() const
99{
100 ASSERT(!atEnd());
101 UChar32 c;
102 U16_GET(m_begin, 0, 0, m_end - m_begin, c);
103 return c;
104}
105
106template<>
107ALWAYS_INLINE auto CodePointIterator<UChar>::operator++() -> CodePointIterator&
108{
109 unsigned i = 0;
110 size_t length = m_end - m_begin;
111 U16_FWD_1(m_begin, i, length);
112 m_begin += i;
113 return *this;
114}
115
116} // namespace WTF
