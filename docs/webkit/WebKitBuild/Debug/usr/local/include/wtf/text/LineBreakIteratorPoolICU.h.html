36namespace WTF {
37
38class LineBreakIteratorPool {
39 WTF_MAKE_NONCOPYABLE(LineBreakIteratorPool);
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 LineBreakIteratorPool() = default;
43
44 WTF_EXPORT_PRIVATE static LineBreakIteratorPool& sharedPool();
45
46 static AtomString makeLocaleWithBreakKeyword(const AtomString& locale, LineBreakIteratorMode mode)
47 {
48 // The uloc functions model locales as char*, so we have to downconvert our AtomString.
49 auto utf8Locale = locale.string().utf8();
50 if (!utf8Locale.length())
51 return locale;
52 Vector<char> scratchBuffer(utf8Locale.length() + 11, 0);
53 memcpy(scratchBuffer.data(), utf8Locale.data(), utf8Locale.length());
54
55 const char* keywordValue = nullptr;
56 switch (mode) {
57 case LineBreakIteratorMode::Default:
58 // nullptr will cause any existing values to be removed.
59 break;
60 case LineBreakIteratorMode::Loose:
61 keywordValue = "loose";
62 break;
63 case LineBreakIteratorMode::Normal:
64 keywordValue = "normal";
65 break;
66 case LineBreakIteratorMode::Strict:
67 keywordValue = "strict";
68 break;
69 }
70
71 UErrorCode status = U_ZERO_ERROR;
72 int32_t lengthNeeded = uloc_setKeywordValue("lb", keywordValue, scratchBuffer.data(), scratchBuffer.size(), &status);
73 if (U_SUCCESS(status))
74 return AtomString::fromUTF8(scratchBuffer.data(), lengthNeeded);
75 if (needsToGrowToProduceBuffer(status)) {
76 scratchBuffer.grow(lengthNeeded + 1);
77 memset(scratchBuffer.data() + utf8Locale.length(), 0, scratchBuffer.size() - utf8Locale.length());
78 status = U_ZERO_ERROR;
79 int32_t lengthNeeded2 = uloc_setKeywordValue("lb", keywordValue, scratchBuffer.data(), scratchBuffer.size(), &status);
80 if (!U_SUCCESS(status) || lengthNeeded != lengthNeeded2)
81 return locale;
82 return AtomString::fromUTF8(scratchBuffer.data(), lengthNeeded);
83 }
84 return locale;
85 }
86
87 UBreakIterator* take(const AtomString& locale, LineBreakIteratorMode mode)
88 {
89 auto localeWithOptionalBreakKeyword = makeLocaleWithBreakKeyword(locale, mode);
90
91 UBreakIterator* iterator = nullptr;
92 for (size_t i = 0; i < m_pool.size(); ++i) {
93 if (m_pool[i].first == localeWithOptionalBreakKeyword) {
94 iterator = m_pool[i].second;
95 m_pool.remove(i);
96 break;
97 }
98 }
99
100 if (!iterator) {
101 iterator = openLineBreakIterator(localeWithOptionalBreakKeyword);
102 if (!iterator)
103 return nullptr;
104 }
105
106 ASSERT(!m_vendedIterators.contains(iterator));
107 m_vendedIterators.add(iterator, localeWithOptionalBreakKeyword);
108 return iterator;
109 }
110
111 void put(UBreakIterator* iterator)
112 {
113 ASSERT(m_vendedIterators.contains(iterator));
114 if (m_pool.size() == capacity) {
115 closeLineBreakIterator(m_pool[0].second);
116 m_pool.remove(0);
117 }
118 m_pool.uncheckedAppend({ m_vendedIterators.take(iterator), iterator });
119 }
120
121private:
122 static constexpr size_t capacity = 4;
123
124 Vector<std::pair<AtomString, UBreakIterator*>, capacity> m_pool;
125 HashMap<UBreakIterator*, AtomString> m_vendedIterators;
126
127 friend WTF::ThreadSpecific<LineBreakIteratorPool>::operator LineBreakIteratorPool*();
128};
