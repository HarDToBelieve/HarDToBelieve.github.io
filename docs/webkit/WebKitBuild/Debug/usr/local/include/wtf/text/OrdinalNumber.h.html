27namespace WTF {
28
29// An abstract number of element in a sequence. The sequence has a first element.
30// This type should be used instead of integer because 2 contradicting traditions can
31// call a first element '0' or '1' which makes integer type ambiguous.
32class OrdinalNumber {
33 WTF_MAKE_FAST_ALLOCATED;
34public:
35 static OrdinalNumber beforeFirst() { return OrdinalNumber(-1); }
36 static OrdinalNumber fromZeroBasedInt(int zeroBasedInt) { return OrdinalNumber(zeroBasedInt); }
37 static OrdinalNumber fromOneBasedInt(int oneBasedInt) { return OrdinalNumber(oneBasedInt - 1); }
38
39 OrdinalNumber() : m_zeroBasedValue(0) { }
40
41 int zeroBasedInt() const { return m_zeroBasedValue; }
42 int oneBasedInt() const { return m_zeroBasedValue + 1; }
43
44 bool operator==(OrdinalNumber other) const { return m_zeroBasedValue == other.m_zeroBasedValue; }
45 bool operator!=(OrdinalNumber other) const { return !((*this) == other); }
46 bool operator>(OrdinalNumber other) const { return m_zeroBasedValue > other.m_zeroBasedValue; }
47
48private:
49 OrdinalNumber(int zeroBasedInt) : m_zeroBasedValue(zeroBasedInt) { }
50 int m_zeroBasedValue;
51};
52
53}
