29#pragma once
30
31#include <limits>
32#include <unicode/utypes.h>
33#include <wtf/Assertions.h>
34#include <wtf/DebugHeap.h>
35#include <wtf/MallocPtr.h>
36#include <wtf/Noncopyable.h>
37
38namespace WTF {
39
40DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StringBuffer);
41
42template <typename CharType>
43class StringBuffer {
44 WTF_MAKE_NONCOPYABLE(StringBuffer);
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 explicit StringBuffer(unsigned length)
48 : m_length(length)
49 , m_data(m_length ? static_cast<CharType*>(StringBufferMalloc::malloc(Checked<size_t>(m_length) * sizeof(CharType))) : nullptr)
50 {
51 }
52
53 ~StringBuffer()
54 {
55 StringBufferMalloc::free(m_data);
56 }
57
58 void shrink(unsigned newLength)
59 {
60 ASSERT(newLength <= m_length);
61 m_length = newLength;
62 }
63
64 void resize(unsigned newLength)
65 {
66 if (newLength > m_length) {
67 if (newLength > std::numeric_limits<unsigned>::max() / sizeof(UChar))
68 CRASH();
69 m_data = static_cast<UChar*>(StringBufferMalloc::realloc(m_data, newLength * sizeof(UChar)));
70 }
71 m_length = newLength;
72 }
73
74 unsigned length() const { return m_length; }
75 CharType* characters() { return m_data; }
76
77 CharType& operator[](unsigned i) { ASSERT_WITH_SECURITY_IMPLICATION(i < m_length); return m_data[i]; }
78
79 MallocPtr<CharType, StringBufferMalloc> release()
80 {
81 CharType* data = m_data;
82 m_data = nullptr;
83 return adoptMallocPtr<CharType, StringBufferMalloc>(data);
84 }
85
86private:
87 unsigned m_length;
88 CharType* m_data;
89};
90
91} // namespace WTF
92
93using WTF::StringBuffer;
