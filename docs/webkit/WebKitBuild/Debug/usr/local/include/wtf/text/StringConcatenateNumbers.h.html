34template<typename Integer>
35class StringTypeAdapter<Integer, typename std::enable_if_t<std::is_integral_v<Integer>>> {
36public:
37 StringTypeAdapter(Integer number)
38 : m_number { number }
39 {
40 }
41
42 unsigned length() const { return lengthOfIntegerAsString(m_number); }
43 bool is8Bit() const { return true; }
44 template<typename CharacterType>
45 void writeTo(CharacterType* destination) const { writeIntegerToBuffer(m_number, destination); }
46
47private:
48 Integer m_number;
49};

51template<typename Enum>
52class StringTypeAdapter<Enum, typename std::enable_if_t<std::is_enum_v<Enum>>> {
53using UnderlyingType = typename std::underlying_type_t<Enum>;
54public:
55 StringTypeAdapter(Enum enumValue)
56 : m_enum { enumValue }
57 {
58 }
59
60 unsigned length() const { return lengthOfIntegerAsString(static_cast<UnderlyingType>(m_enum)); }
61 bool is8Bit() const { return true; }
62 template<typename CharacterType>
63 void writeTo(CharacterType* destination) const { writeIntegerToBuffer(static_cast<UnderlyingType>(m_enum), destination); }

65private:
66 Enum m_enum;

69template<typename FloatingPoint>
70class StringTypeAdapter<FloatingPoint, typename std::enable_if_t<std::is_floating_point<FloatingPoint>::value>> {
71public:
72 StringTypeAdapter(FloatingPoint number)
73 {
74 numberToString(number, m_buffer);
75 m_length = std::strlen(&m_buffer[0]);
76 }
77
78 unsigned length() const { return m_length; }
79 bool is8Bit() const { return true; }
80 template<typename CharacterType> void writeTo(CharacterType* destination) const { StringImpl::copyCharacters(destination, buffer(), m_length); }
81
82private:
83 const LChar* buffer() const { return reinterpret_cast<const LChar*>(&m_buffer[0]); }
84
85 NumberToStringBuffer m_buffer;
86 unsigned m_length;
87};

89class FormattedNumber {
90 WTF_MAKE_FAST_ALLOCATED;
91public:
92 static FormattedNumber fixedPrecision(double number, unsigned significantFigures = 6, TrailingZerosTruncatingPolicy trailingZerosTruncatingPolicy = TruncateTrailingZeros)
93 {
94 FormattedNumber numberFormatter;
95 numberToFixedPrecisionString(number, significantFigures, numberFormatter.m_buffer, trailingZerosTruncatingPolicy == TruncateTrailingZeros);
96 numberFormatter.m_length = std::strlen(&numberFormatter.m_buffer[0]);
97 return numberFormatter;
98 }
99
100 static FormattedNumber fixedWidth(double number, unsigned decimalPlaces)
101 {
102 FormattedNumber numberFormatter;
103 numberToFixedWidthString(number, decimalPlaces, numberFormatter.m_buffer);
104 numberFormatter.m_length = std::strlen(&numberFormatter.m_buffer[0]);
105 return numberFormatter;
106 }
107
108 unsigned length() const { return m_length; }
109 const LChar* buffer() const { return reinterpret_cast<const LChar*>(&m_buffer[0]); }
110
111private:
112 NumberToStringBuffer m_buffer;
113 unsigned m_length;
114};
115
116template<> class StringTypeAdapter<FormattedNumber> {
117public:
118 StringTypeAdapter(const FormattedNumber& number)
119 : m_number { number }
120 {
121 }
122
123 unsigned length() const { return m_number.length(); }
124 bool is8Bit() const { return true; }
125 template<typename CharacterType> void writeTo(CharacterType* destination) const { StringImpl::copyCharacters(destination, m_number.buffer(), m_number.length()); }
126
127private:
128 const FormattedNumber& m_number;
129};
