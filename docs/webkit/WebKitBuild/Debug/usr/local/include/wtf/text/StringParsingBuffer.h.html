26#pragma once
27
28#include <wtf/text/StringView.h>
29
30namespace WTF {
31
32template<typename T>
33class StringParsingBuffer final {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 using CharacterType = T;
37
38 constexpr StringParsingBuffer() = default;
39
40 constexpr StringParsingBuffer(const CharacterType* characters, unsigned length)
41 : m_position { characters }
42 , m_end { characters + length }
43 {
44 ASSERT(characters || !length);
45 }
46
47 constexpr StringParsingBuffer(const CharacterType* characters, const CharacterType* end)
48 : m_position { characters }
49 , m_end { end }
50 {
51 ASSERT(characters <= end);
52 ASSERT(!characters == !end);
53 ASSERT(static_cast<size_t>(end - characters) <= std::numeric_limits<unsigned>::max());
54 }
55
56 constexpr auto position() const { return m_position; }
57 constexpr auto end() const { return m_end; }
58
59 constexpr bool hasCharactersRemaining() const { return m_position < m_end; }
60 constexpr bool atEnd() const { return m_position == m_end; }
61
62 constexpr unsigned lengthRemaining() const { return m_end - m_position; }
63
64 StringView stringViewOfCharactersRemaining() { return { m_position, lengthRemaining() }; }
65
66 CharacterType operator[](unsigned i)
67 {
68 ASSERT(i < lengthRemaining());
69 return m_position[i];
70 }
71
72 constexpr CharacterType operator*() const
73 {
74 ASSERT(hasCharactersRemaining());
75 return *m_position;
76 }
77
78 constexpr void advance()
79 {
80 ASSERT(hasCharactersRemaining());
81 ++m_position;
82 }
83
84 constexpr void advanceBy(unsigned places)
85 {
86 ASSERT(places <= lengthRemaining());
87 m_position += places;
88 }
89
90 constexpr StringParsingBuffer& operator++()
91 {
92 advance();
93 return *this;
94 }
95
96 constexpr StringParsingBuffer operator++(int)
97 {
98 auto result = *this;
99 ++*this;
100 return result;
101 }
102
103 constexpr StringParsingBuffer& operator+=(int places)
104 {
105 advanceBy(places);
106 return *this;
107 }
108
109private:
110 const CharacterType* m_position { nullptr };
111 const CharacterType* m_end { nullptr };
112};
113
114template<typename StringType, typename Function> decltype(auto) readCharactersForParsing(StringType&& string, Function&& functor)
115{
116 if (string.is8Bit())
117 return functor(StringParsingBuffer { string.characters8(), string.length() });
118 return functor(StringParsingBuffer { string.characters16(), string.length() });
119}
120
121} // namespace WTF
122
123using WTF::StringParsingBuffer;
124using WTF::readCharactersForParsing;
