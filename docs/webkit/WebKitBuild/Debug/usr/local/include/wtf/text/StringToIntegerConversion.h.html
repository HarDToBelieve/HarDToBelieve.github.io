33// The parseInteger function template allows leading and trailing spaces as defined by isASCIISpace, and, after the leading spaces, allows a single leading "+".
34// The parseIntegerAllowingTrailingJunk function template is like parseInteger, but allows any characters after the integer.
35
36// FIXME: Should we add a version that does not allow "+"?
37// FIXME: Should we add a version that allows other definitions of spaces, like isHTMLSpace or isHTTPSpace?
38// FIXME: Should we add a version that does not allow leading and trailing spaces?
39
40template<typename IntegralType> std::optional<IntegralType> parseInteger(StringView, uint8_t base = 10);
41template<typename IntegralType> std::optional<IntegralType> parseIntegerAllowingTrailingJunk(StringView, uint8_t base = 10);
42
43enum class TrailingJunkPolicy { Disallow, Allow };
44
45template<typename IntegralType, typename CharacterType> std::optional<IntegralType> parseInteger(const CharacterType* data, size_t length, uint8_t base, TrailingJunkPolicy policy)
46{
47 if (!data)
48 return std::nullopt;
49
50 while (length && isASCIISpace(*data)) {
51 --length;
52 ++data;
53 }
54
55 bool isNegative = false;
56 if (std::is_signed_v<IntegralType> && length && *data == '-') {
57 --length;
58 ++data;
59 isNegative = true;
60 } else if (length && *data == '+') {
61 --length;
62 ++data;
63 }
64
65 auto isCharacterAllowedInBase = [] (auto character, auto base) {
66 if (isASCIIDigit(character))
67 return character - '0' < base;
68 return toASCIILowerUnchecked(character) >= 'a' && toASCIILowerUnchecked(character) < 'a' + std::min(base - 10, 26);
69 };
70
71 if (!(length && isCharacterAllowedInBase(*data, base)))
72 return std::nullopt;
73
74 Checked<IntegralType, RecordOverflow> value;
75 do {
76 IntegralType digitValue = isASCIIDigit(*data) ? *data - '0' : toASCIILowerUnchecked(*data) - 'a' + 10;
77 value *= static_cast<IntegralType>(base);
78 if (isNegative)
79 value -= digitValue;
80 else
81 value += digitValue;
82 } while (--length && isCharacterAllowedInBase(*++data, base));
83
84 if (UNLIKELY(value.hasOverflowed()))
85 return std::nullopt;
86
87 if (policy == TrailingJunkPolicy::Disallow) {
88 while (length && isASCIISpace(*data)) {
89 --length;
90 ++data;
91 }
92 if (length)
93 return std::nullopt;
94 }
95
96 return value.value();
97}
98
99template<typename IntegralType> std::optional<IntegralType> parseInteger(StringView string, uint8_t base)
100{
101 if (string.is8Bit())
102 return parseInteger<IntegralType>(string.characters8(), string.length(), base, TrailingJunkPolicy::Disallow);
103 return parseInteger<IntegralType>(string.characters16(), string.length(), base, TrailingJunkPolicy::Disallow);
104}
105
106template<typename IntegralType> std::optional<IntegralType> parseIntegerAllowingTrailingJunk(StringView string, uint8_t base)
107{
108 if (string.is8Bit())
109 return parseInteger<IntegralType>(string.characters8(), string.length(), base, TrailingJunkPolicy::Allow);
110 return parseInteger<IntegralType>(string.characters16(), string.length(), base, TrailingJunkPolicy::Allow);
111}
