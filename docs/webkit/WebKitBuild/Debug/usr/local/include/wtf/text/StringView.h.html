475inline StringView StringView::substring(unsigned start, unsigned length) const
476{
477 if (start >= this->length())
478 return empty();
479 unsigned maxLength = this->length() - start;
480
481 if (length >= maxLength) {
482 if (!start)
483 return *this;
484 length = maxLength;
485 }
486
487 if (is8Bit()) {
488 StringView result(characters8() + start, length);
489 result.setUnderlyingString(*this);
490 return result;
491 }
492 StringView result(characters16() + start, length);
493 result.setUnderlyingString(*this);
494 return result;
495}

542inline StringView::UpconvertedCharacters::UpconvertedCharacters(const StringView& string)
543{
544 if (!string.is8Bit()) {
545 m_characters = string.characters16();
546 return;
547 }
548 const LChar* characters8 = string.characters8();
549 unsigned length = string.m_length;
550 m_upconvertedCharacters.reserveInitialCapacity(length);
551 for (unsigned i = 0; i < length; ++i)
552 m_upconvertedCharacters.uncheckedAppend(characters8[i]);
553 m_characters = m_upconvertedCharacters.data();
554}

642inline bool equal(StringView a, StringView b)
643{
644 if (a.m_characters == b.m_characters) {
645 ASSERT(a.is8Bit() == b.is8Bit());
646 return a.length() == b.length();
647 }
648
649 return equalCommon(a, b);
650}
651
652inline bool equal(StringView a, const LChar* b)
653{
654 if (!b)
655 return !a.isEmpty();
656 if (a.isEmpty())
657 return !b;
658
659 unsigned aLength = a.length();
660 if (aLength != strlen(reinterpret_cast<const char*>(b)))
661 return false;
662
663 if (a.is8Bit())
664 return equal(a.characters8(), b, aLength);
665 return equal(a.characters16(), b, aLength);
666}

1032template<typename CharacterType, typename MatchedCharacterPredicate>
1033inline StringView StringView::stripLeadingAndTrailingMatchedCharacters(const CharacterType* characters, const MatchedCharacterPredicate& predicate)
1034{
1035 if (!m_length)
1036 return *this;
1037
1038 unsigned start = 0;
1039 unsigned end = m_length - 1;
1040
1041 while (start <= end && predicate(characters[start]))
1042 ++start;
1043
1044 if (start > end)
1045 return StringView::empty();
1046
1047 while (end && predicate(characters[end]))
1048 --end;
1049
1050 if (!start && end == m_length - 1)
1051 return *this;
1052
1053 StringView result(characters + start, end + 1 - start);
1054 result.setUnderlyingString(*this);
1055 return result;
1056}

1063 return stripLeadingAndTrailingMatchedCharacters<UChar>(characters16(), predicate);
