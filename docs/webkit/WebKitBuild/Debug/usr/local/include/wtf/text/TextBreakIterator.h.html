46class TextBreakIterator {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 enum class Mode {
50 Line,
51 Caret,
52 Delete
53 };
54
55 TextBreakIterator() = delete;
56 TextBreakIterator(const TextBreakIterator&) = delete;
57 TextBreakIterator(TextBreakIterator&&) = default;
58 TextBreakIterator& operator=(const TextBreakIterator&) = delete;
59 TextBreakIterator& operator=(TextBreakIterator&&) = default;
60
61 std::optional<unsigned> preceding(unsigned location) const
62 {
63 return switchOn(m_backing, [&](const auto& iterator) {
64 return iterator.preceding(location);
65 });
66 }
67
68 std::optional<unsigned> following(unsigned location) const
69 {

77 return switchOn(m_backing, [&](const auto& iterator) {

82private:

86 WTF_EXPORT_PRIVATE TextBreakIterator(StringView, Mode, const AtomString& locale);

89 {
90 return switchOn(m_backing, [&](auto& iterator) {

112class TextBreakIteratorCache {
113 WTF_MAKE_FAST_ALLOCATED;
114// Use CachedTextBreakIterator instead of dealing with the cache directly.
115private:
116 friend class LazyNeverDestroyed<TextBreakIteratorCache>;
117 friend class CachedTextBreakIterator;
118
119 WTF_EXPORT_PRIVATE static TextBreakIteratorCache& singleton();
120
121 TextBreakIteratorCache(const TextBreakIteratorCache&) = delete;
122 TextBreakIteratorCache(TextBreakIteratorCache&&) = delete;
123 TextBreakIteratorCache& operator=(const TextBreakIteratorCache&) = delete;
124 TextBreakIteratorCache& operator=(TextBreakIteratorCache&&) = delete;
125
126 TextBreakIterator take(StringView string, TextBreakIterator::Mode mode, const AtomString& locale)
127 {
128 auto iter = std::find_if(m_unused.begin(), m_unused.end(), [&](TextBreakIterator& candidate) {
129 return candidate.mode() == mode && candidate.locale() == locale;
130 });
131 if (iter == m_unused.end())
132 return TextBreakIterator(string, mode, locale);
133 auto result = WTFMove(*iter);
134 m_unused.remove(iter - m_unused.begin());
135 result.setText(string);
136 return result;
137
138 }
139
140 void put(TextBreakIterator&& iterator)
141 {
142 m_unused.append(WTFMove(iterator));
143 if (m_unused.size() > capacity)
144 m_unused.remove(0);
145 }
146
147 TextBreakIteratorCache()
148 {
149 }
150
151 static constexpr int capacity = 2;
152 // FIXME: Break this up into different Vectors per mode.
153 Vector<TextBreakIterator, capacity> m_unused;
154};

209class LazyLineBreakIterator {
210 WTF_MAKE_FAST_ALLOCATED;
211public:
212 LazyLineBreakIterator()
213 {
214 resetPriorContext();
215 }
216
217 explicit LazyLineBreakIterator(StringView stringView, const AtomString& locale = AtomString(), LineBreakIteratorMode mode = LineBreakIteratorMode::Default)
218 : m_stringView(stringView)
219 , m_locale(locale)
220 , m_mode(mode)
221 {
222 resetPriorContext();
223 }
224
225 ~LazyLineBreakIterator()
226 {
227 if (m_iterator)
228 releaseLineBreakIterator(m_iterator);
229 }
230
231 StringView stringView() const { return m_stringView; }
232 LineBreakIteratorMode mode() const { return m_mode; }
233
234 UChar lastCharacter() const
235 {
236 static_assert(WTF_ARRAY_LENGTH(m_priorContext) == 2, "UBreakIterator unexpected prior context length");
237 return m_priorContext[1];
238 }
239
240 UChar secondToLastCharacter() const
241 {
242 static_assert(WTF_ARRAY_LENGTH(m_priorContext) == 2, "UBreakIterator unexpected prior context length");
243 return m_priorContext[0];
244 }
245
246 void setPriorContext(UChar last, UChar secondToLast)
247 {
248 static_assert(WTF_ARRAY_LENGTH(m_priorContext) == 2, "UBreakIterator unexpected prior context length");
249 m_priorContext[0] = secondToLast;
250 m_priorContext[1] = last;
251 }
252
253 void updatePriorContext(UChar last)
254 {
255 static_assert(WTF_ARRAY_LENGTH(m_priorContext) == 2, "UBreakIterator unexpected prior context length");
256 m_priorContext[0] = m_priorContext[1];
257 m_priorContext[1] = last;
258 }
259
260 void resetPriorContext()
261 {
262 static_assert(WTF_ARRAY_LENGTH(m_priorContext) == 2, "UBreakIterator unexpected prior context length");
263 m_priorContext[0] = 0;
264 m_priorContext[1] = 0;
265 }
266
267 unsigned priorContextLength() const
268 {
269 unsigned priorContextLength = 0;
270 static_assert(WTF_ARRAY_LENGTH(m_priorContext) == 2, "UBreakIterator unexpected prior context length");
271 if (m_priorContext[1]) {
272 ++priorContextLength;
273 if (m_priorContext[0])
274 ++priorContextLength;
275 }
276 return priorContextLength;
277 }
278
279 // Obtain text break iterator, possibly previously cached, where this iterator is (or has been)
280 // initialized to use the previously stored string as the primary breaking context and using
281 // previously stored prior context if non-empty.
282 UBreakIterator* get(unsigned priorContextLength)
283 {
284 ASSERT(priorContextLength <= priorContextCapacity);
285 const UChar* priorContext = priorContextLength ? &m_priorContext[priorContextCapacity - priorContextLength] : nullptr;
286 if (!m_iterator) {
287 m_iterator = acquireLineBreakIterator(m_stringView, m_locale, priorContext, priorContextLength, m_mode);
288 m_cachedPriorContext = priorContext;
289 m_cachedPriorContextLength = priorContextLength;
290 } else if (priorContext != m_cachedPriorContext || priorContextLength != m_cachedPriorContextLength) {
291 resetStringAndReleaseIterator(m_stringView, m_locale, m_mode);
292 return this->get(priorContextLength);
293 }
294 return m_iterator;
295 }
296
297 void resetStringAndReleaseIterator(StringView stringView, const AtomString& locale, LineBreakIteratorMode mode)
298 {
299 if (m_iterator)
300 releaseLineBreakIterator(m_iterator);
301 m_stringView = stringView;
302 m_locale = locale;
303 m_iterator = nullptr;
304 m_cachedPriorContext = nullptr;
305 m_mode = mode;
306 m_cachedPriorContextLength = 0;
307 }
308
309private:
310 static constexpr unsigned priorContextCapacity = 2;
311 StringView m_stringView;
312 AtomString m_locale;
313 UBreakIterator* m_iterator { nullptr };
314 const UChar* m_cachedPriorContext { nullptr };
315 LineBreakIteratorMode m_mode { LineBreakIteratorMode::Default };
316 unsigned m_cachedPriorContextLength { 0 };
317 UChar m_priorContext[priorContextCapacity];
318};

322// version 4.0 only supports "legacy grapheme clusters".
323// Use this for general text processing, e.g. string truncation.
