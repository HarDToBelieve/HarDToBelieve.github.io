65 // Construct a string by copying the contents of a vector. To avoid
66 // copying, consider using String::adopt instead.
67 // This method will never create a null string. Vectors with size() == 0
68 // will return the empty string.
69 // NOTE: This is different from String(vector.data(), vector.size())
70 // which will sometimes return a null string when vector.data() is null
71 // which can only occur for vectors without inline capacity.
72 // See: https://bugs.webkit.org/show_bug.cgi?id=109792

571template<typename CharacterType>
572inline void appendNumber(Vector<CharacterType>& vector, unsigned char number)
573{
574 int numberLength = number > 99 ? 3 : (number > 9 ? 2 : 1);
575 size_t vectorSize = vector.size();
576 vector.grow(vectorSize + numberLength);
577
578 switch (numberLength) {
579 case 3:
580 vector[vectorSize + 2] = number % 10 + '0';
581 number /= 10;
582 FALLTHROUGH;
583
584 case 2:
585 vector[vectorSize + 1] = number % 10 + '0';
586 number /= 10;
587 FALLTHROUGH;
588
589 case 1:
590 vector[vectorSize] = number % 10 + '0';
591 }
592}
