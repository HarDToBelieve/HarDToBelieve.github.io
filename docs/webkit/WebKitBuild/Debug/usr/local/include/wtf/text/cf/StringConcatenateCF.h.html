32namespace WTF {
33
34template<> class StringTypeAdapter<CFStringRef> {
35public:
36 StringTypeAdapter(CFStringRef);
37 unsigned length() const { return m_string ? CFStringGetLength(m_string) : 0; }
38 bool is8Bit() const { return !m_string || CFStringGetCStringPtr(m_string, kCFStringEncodingISOLatin1); }
39 template<typename CharacterType> void writeTo(CharacterType*) const;
40
41private:
42 CFStringRef m_string;
43};
44
45inline StringTypeAdapter<CFStringRef>::StringTypeAdapter(CFStringRef string)
46 : m_string { string }
47{
48}
49
50template<> inline void StringTypeAdapter<CFStringRef>::writeTo<LChar>(LChar* destination) const
51{
52 if (m_string)
53 std::memcpy(destination, CFStringGetCStringPtr(m_string, kCFStringEncodingISOLatin1), CFStringGetLength(m_string));
54}
55
56template<> inline void StringTypeAdapter<CFStringRef>::writeTo<UChar>(UChar* destination) const
57{
58 if (m_string)
59 CFStringGetCharacters(m_string, CFRangeMake(0, CFStringGetLength(m_string)), reinterpret_cast<UniChar*>(destination));
60}
