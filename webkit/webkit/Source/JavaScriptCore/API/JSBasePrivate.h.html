36/*!
37@function
38@abstract Reports an object's non-GC memory payload to the garbage collector.
39@param ctx The execution context to use.
40@param size The payload's size, in bytes.
41@discussion Use this function to notify the garbage collector that a GC object
42owns a large non-GC memory region. Calling this function will encourage the
43garbage collector to collect soon, hoping to reclaim that large non-GC memory
44region.
45*/
46JS_EXPORT void JSReportExtraMemoryCost(JSContextRef ctx, size_t size) JSC_API_AVAILABLE(macos(10.6), ios(7.0));
47
48JS_EXPORT void JSDisableGCTimer(void);

51/*!
52@function JSConfigureSignalForGC
53@abstract Configure signals for GC in non-Apple and non-Windows platforms.
54@param signal The signal number to use.
55@result true if the signal is successfully configured, otherwise false.
56@discussion Call this function before any of JSC initialization starts. Otherwise, it fails.

58JS_EXPORT bool JSConfigureSignalForGC(int signal);

61/*!
62@function
63@abstract Produces an object with various statistics about current memory usage.
64@param ctx The execution context to use.
65@result An object containing GC heap status data.
66@discussion Specifically, the result object has the following integer-valued fields:
67 heapSize: current size of heap
68 heapCapacity: current capacity of heap
69 extraMemorySize: amount of non-GC memory referenced by GC objects (included in heap size / capacity)
70 objectCount: current count of GC objects
71 protectedObjectCount: current count of protected GC objects
72 globalObjectCount: current count of global GC objects
73 protectedGlobalObjectCount: current count of protected global GC objects
74*/
75JS_EXPORT JSObjectRef JSGetMemoryUsageStatistics(JSContextRef ctx);
