41static unsigned unitFlags = NSCalendarUnitSecond | NSCalendarUnitMinute | NSCalendarUnitHour | NSCalendarUnitDay | NSCalendarUnitMonth | NSCalendarUnitYear;
42
43@implementation DateTests
44+ (void) NSDateToJSDateTest
45{
46 JSContext *context = [[JSContext alloc] init];
47 NSDate *now = [NSDate dateWithTimeIntervalSinceNow:0];
48 NSDateComponents *components = [[NSCalendar currentCalendar] components:unitFlags fromDate:now];
49 JSValue *jsNow = [JSValue valueWithObject:now inContext:context];
50 int year = [[jsNow invokeMethod:@"getFullYear" withArguments:@[]] toInt32];
51 // Months are 0-indexed for JavaScript Dates.
52 int month = [[jsNow invokeMethod:@"getMonth" withArguments:@[]] toInt32] + 1;
53 int day = [[jsNow invokeMethod:@"getDate" withArguments:@[]] toInt32];
54 int hour = [[jsNow invokeMethod:@"getHours" withArguments:@[]] toInt32];
55 int minute = [[jsNow invokeMethod:@"getMinutes" withArguments:@[]] toInt32];
56 int second = [[jsNow invokeMethod:@"getSeconds" withArguments:@[]] toInt32];
57
58 checkResult(@"NSDate to JS Date", year == [components year]
59 && month == [components month]
60 && day == [components day]
61 && hour == [components hour]
62 && minute == [components minute]
63 && second == [components second]);
64}

66+ (void) JSDateToNSDateTest
67{
68 JSContext *context = [[JSContext alloc] init];
69 NSDateFormatter *formatter = [[NSDateFormatter alloc] init];
70 [formatter setDateFormat:@"MMMM dd',' yyyy hh:mm:ss"];
71 [formatter setLocale:[[NSLocale alloc] initWithLocaleIdentifier:@"en_US"]];
72 NSDate *februaryFourth2014 = [formatter dateFromString:@"February 4, 2014 11:40:03"];
73 NSDateComponents *components = [[NSCalendar currentCalendar] components:unitFlags fromDate:februaryFourth2014];
74 // Months are 0-indexed for JavaScript Dates.
75 JSValue *jsDate = [context[@"Date"] constructWithArguments:@[@2014, @1, @4, @11, @40, @3]];
76
77 int year = [[jsDate invokeMethod:@"getFullYear" withArguments:@[]] toInt32];
78 int month = [[jsDate invokeMethod:@"getMonth" withArguments:@[]] toInt32] + 1;
79 int day = [[jsDate invokeMethod:@"getDate" withArguments:@[]] toInt32];
80 int hour = [[jsDate invokeMethod:@"getHours" withArguments:@[]] toInt32];
81 int minute = [[jsDate invokeMethod:@"getMinutes" withArguments:@[]] toInt32];
82 int second = [[jsDate invokeMethod:@"getSeconds" withArguments:@[]] toInt32];
83
84 checkResult(@"JS Date to NSDate", year == [components year]
85 && month == [components month]
86 && day == [components day]
87 && hour == [components hour]
88 && minute == [components minute]
89 && second == [components second]);
90}

92+ (void) roundTripThroughJSDateTest
93{
94 JSContext *context = [[JSContext alloc] init];
95 [context evaluateScript:@"function jsReturnDate(date) { return date; }"];
96 NSDateFormatter *formatter = [[NSDateFormatter alloc] init];
97 [formatter setDateFormat:@"MMMM dd',' yyyy hh:mm:ss"];
98 [formatter setLocale:[[NSLocale alloc] initWithLocaleIdentifier:@"en_US"]];
99 NSDate *februaryFourth2014 = [formatter dateFromString:@"February 4, 2014 11:40:03"];
100 NSDateComponents *components = [[NSCalendar currentCalendar] components:unitFlags fromDate:februaryFourth2014];
101
102 JSValue *roundTripThroughJS = [context[@"jsReturnDate"] callWithArguments:@[februaryFourth2014]];
103 int year = [[roundTripThroughJS invokeMethod:@"getFullYear" withArguments:@[]] toInt32];
104 // Months are 0-indexed for JavaScript Dates.
105 int month = [[roundTripThroughJS invokeMethod:@"getMonth" withArguments:@[]] toInt32] + 1;
106 int day = [[roundTripThroughJS invokeMethod:@"getDate" withArguments:@[]] toInt32];
107 int hour = [[roundTripThroughJS invokeMethod:@"getHours" withArguments:@[]] toInt32];
108 int minute = [[roundTripThroughJS invokeMethod:@"getMinutes" withArguments:@[]] toInt32];
109 int second = [[roundTripThroughJS invokeMethod:@"getSeconds" withArguments:@[]] toInt32];
110
111 checkResult(@"JS date round trip", year == [components year]
112 && month == [components month]
113 && day == [components day]
114 && hour == [components hour]
115 && minute == [components minute]
116 && second == [components second]);
117}

119+ (void) roundTripThroughObjCDateTest
120{
121 JSContext *context = [[JSContext alloc] init];
122 context[@"objcReturnDate"] = ^(NSDate *date) {
123 return date;
124 };
125 [context evaluateScript:@"function test() {\
126 var date = new Date(2014, 1, 4, 11, 40, 3); \
127 var result = objcReturnDate(date); \
128 return date.getYear() === result.getYear() \
129 && date.getMonth() === result.getMonth() \
130 && date.getDate() === result.getDate() \
131 && date.getHours() === result.getHours() \
132 && date.getMinutes() === result.getMinutes() \
133 && date.getSeconds() === result.getSeconds() \
134 && date.getMilliseconds() === result.getMilliseconds();\
135 }"];
136
137 checkResult(@"ObjC date round trip", [[context[@"test"] callWithArguments:@[]] toBool]);
138}
