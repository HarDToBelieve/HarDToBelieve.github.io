107+ (void) exportInstanceMethodWithIdProtocolTest
108{
109 JSContext *context = [[JSContext alloc] init];
110 context[@"ExportMethodWithIdProtocol"] = [ExportMethodWithIdProtocol class];
111 context[@"makeTestObject"] = ^{
112 return [[ExportMethodWithIdProtocol alloc] init];
113 };
114 context[@"opaqueObject"] = [[TruthTeller alloc] init];
115 [context evaluateScript:@"makeTestObject().methodWithIdProtocol(opaqueObject);"];
116 checkResult(@"Successfully exported instance method", !context.exception);
117}

131+ (void) exportDynamicallyGeneratedProtocolTest
132{
133 JSContext *context = [[JSContext alloc] init];
134 Protocol *dynProtocol = objc_allocateProtocol("NSStringJSExport");
135 Protocol *jsExportProtocol = @protocol(JSExport);
136 protocol_addProtocol(dynProtocol, jsExportProtocol);
137 Method method = class_getInstanceMethod([NSString class], @selector(boolValue));
138 protocol_addMethodDescription(dynProtocol, @selector(boolValue), method_getTypeEncoding(method), YES, YES);
139 NSLog(@"type encoding = %s", method_getTypeEncoding(method));
140 protocol_addMethodDescription(dynProtocol, @selector(boolValue), "B@:", YES, YES);
141 objc_registerProtocol(dynProtocol);
142 class_addProtocol([NSString class], dynProtocol);
143
144 context[@"NSString"] = [NSString class];
145 context[@"myString"] = @"YES";
146 JSValue *value = [context evaluateScript:@"myString.boolValue()"];
147 checkResult(@"Dynamically generated JSExport-ed protocols are ignored", [value isUndefined] && !!context.exception);
148}

150+ (void)classNamePrefixedWithUnderscoreTest
151{
152 JSContext *context = [[JSContext alloc] init];
153
154 context[@"_UnderscorePrefixNoExport"] = [_UnderscorePrefixNoExport class];
155 context[@"_UnderscorePrefixWithExport"] = [_UnderscorePrefixWithExport class];
156
157 checkResult(@"Non-underscore-prefixed ancestor class used when there are no exports", [context[@"_UnderscorePrefixNoExport"] toObject] == [NoUnderscorePrefix class]);
158 checkResult(@"Underscore-prefixed class used when there are exports", [context[@"_UnderscorePrefixWithExport"] toObject] == [_UnderscorePrefixWithExport class]);
159
160 JSValue *withExportInstance = [context evaluateScript:@"new _UnderscorePrefixWithExport()"];
161 checkResult(@"Exports present on underscore-prefixed class", !context.exception && !withExportInstance.isUndefined);
162}

176static void wrapperLifetimeIsTiedToGlobalObject()
177{
178 JSGlobalContextRef contextRef;
179 @autoreleasepool {
180 JSContext *context = [[JSContext alloc] init];
181 contextRef = JSGlobalContextRetain(context.JSGlobalContextRef);
182 context[@"A"] = A.class;
183 checkResult(@"Initial wrapper's constructor is itself", [[context evaluateScript:@"new A().constructor === A"] toBool]);
184 }
185
186 @autoreleasepool {
187 JSContext *context = [JSContext contextWithJSGlobalContextRef:contextRef];
188 checkResult(@"New context's wrapper's constructor is itself", [[context evaluateScript:@"new A().constructor === A"] toBool]);
189 }
190
191 JSGlobalContextRelease(contextRef);
192}

194static void wrapperForNSObjectisObject()
195{
196 @autoreleasepool {
197 JSContext *context = [[JSContext alloc] init];
198 context[@"Object"] = [[NSNull alloc] init];
199 context.exception = nil;
200
201 context[@"A"] = NSObject.class;
202 checkResult(@"Should not throw an exception when wrapping NSObject and Object has been changed", ![context exception]);
203 }
204}
