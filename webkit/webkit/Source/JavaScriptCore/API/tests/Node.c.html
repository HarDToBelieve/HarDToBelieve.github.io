32Node* Node_new(void)
33{
34 Node* node = (Node*)malloc(sizeof(Node));
35 node->refCount = 0;
36 node->nodeType = "Node";
37 node->childNodesTail = NULL;
38
39 return node;
40}
41
42void Node_appendChild(Node* node, Node* child)
43{
44 Node_ref(child);
45 NodeLink* nodeLink = (NodeLink*)malloc(sizeof(NodeLink));
46 nodeLink->node = child;
47 nodeLink->prev = node->childNodesTail;
48 node->childNodesTail = nodeLink;
49}
50
51void Node_removeChild(Node* node, Node* child)
52{
53 /* Linear search from tail -- good enough for our purposes here */
54 NodeLink* current;
55 NodeLink** currentHandle;
56 for (currentHandle = &node->childNodesTail, current = *currentHandle; current; currentHandle = &current->prev, current = *currentHandle) {
57 if (current->node == child) {
58 Node_deref(current->node);
59 *currentHandle = current->prev;
60 free(current);
61 break;
62 }
63 }
64}
65
66void Node_replaceChild(Node* node, Node* newChild, Node* oldChild)
67{
68 /* Linear search from tail -- good enough for our purposes here */
69 NodeLink* current;
70 for (current = node->childNodesTail; current; current = current->prev) {
71 if (current->node == oldChild) {
72 Node_deref(current->node);
73 current->node = newChild;
74 }
75 }
76}
77
78void Node_ref(Node* node)
79{
80 ++node->refCount;
81}
82
83void Node_deref(Node* node)
84{
85 if (--node->refCount == 0)
86 free(node);
87}
