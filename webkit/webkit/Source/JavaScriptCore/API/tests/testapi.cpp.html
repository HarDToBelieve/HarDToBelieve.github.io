74class APIContext {
75 WTF_MAKE_NONCOPYABLE(APIContext);
76public:
77
78 APIContext()
79 : m_context(JSGlobalContextCreate(nullptr))
80 {
81 APIString print("print");
82 JSObjectRef printFunction = JSObjectMakeFunctionWithCallback(m_context, print, [] (JSContextRef ctx, JSObjectRef, JSObjectRef, size_t argumentCount, const JSValueRef arguments[], JSValueRef*) {
83
84 JSC::JSGlobalObject* globalObject = toJS(ctx);
85 for (unsigned i = 0; i < argumentCount; i++)
86 dataLog(toJS(globalObject, arguments[i]));
87 dataLogLn();
88 return JSValueMakeUndefined(ctx);
89 });
90
91 JSObjectSetProperty(m_context, JSContextGetGlobalObject(m_context), print, printFunction, kJSPropertyAttributeNone, nullptr);
92 }
93
94 ~APIContext()
95 {
96 JSGlobalContextRelease(m_context);
97 }
98
99 operator JSGlobalContextRef() { return m_context; }
100 operator JSC::JSGlobalObject*() { return toJS(m_context); }
101
102private:
103 JSGlobalContextRef m_context;
104};

106template<typename T>
107class APIVector : protected Vector<T> {
108 using Base = Vector<T>;
109public:
110 APIVector(APIContext& context)
111 : Base()
112 , m_context(context)
113 {
114 }
115
116 ~APIVector()
117 {
118 for (auto& value : *this)
119 JSValueUnprotect(m_context, value);
120 }
121
122 using Vector<T>::operator[];
123 using Vector<T>::size;
124 using Vector<T>::begin;
125 using Vector<T>::end;
126 using typename Vector<T>::iterator;
127
128 void append(T value)
129 {
130 JSValueProtect(m_context, value);
131 Base::append(WTFMove(value));
132 }
133
134private:
135 APIContext& m_context;
136};

189TestAPI::ScriptResult TestAPI::evaluateScript(const char* script, JSObjectRef thisObject)
190{
191 APIString scriptAPIString(script);
192 JSValueRef exception = nullptr;
193
194 JSValueRef result = JSEvaluateScript(context, scriptAPIString, thisObject, nullptr, 0, &exception);
195 if (exception)
196 return Unexpected<JSValueRef>(exception);
197 return ScriptResult(result);
198}

200template<typename... ArgumentTypes>
201TestAPI::ScriptResult TestAPI::callFunction(const char* functionSource, ArgumentTypes... arguments)
202{
203 JSValueRef function;
204 {
205 ScriptResult functionResult = evaluateScript(functionSource);
206 if (!functionResult)
207 return functionResult;
208 function = functionResult.value();
209 }
210
211 JSValueRef exception = nullptr;
212 if (JSObjectRef functionObject = JSValueToObject(context, function, &exception)) {
213 JSValueRef args[sizeof...(arguments)] { arguments... };
214 JSValueRef result = JSObjectCallAsFunction(context, functionObject, functionObject, sizeof...(arguments), args, &exception);
215 if (!exception)
216 return ScriptResult(result);
217 }
218
219 RELEASE_ASSERT(exception);
220 return Unexpected<JSValueRef>(exception);
221}

248bool TestAPI::functionReturnsTrue(const char* functionSource, ArgumentTypes... arguments)
249{
250 JSValueRef trueValue = JSValueMakeBoolean(context, true);
251 ScriptResult result = callFunction(functionSource, arguments...);
252 if (!result)
253 return false;
254 return JSValueIsStrictEqual(context, trueValue, result.value());

276template<typename JSFunctor, typename APIFunctor>
277void TestAPI::checkJSAndAPIMatch(const JSFunctor& jsFunctor, const APIFunctor& apiFunctor, const char* description)
278{
279 JSValueRef exception = nullptr;
280 JSValueRef result = apiFunctor(&exception);
281 ScriptResult jsResult = jsFunctor();
282 if (!jsResult) {
283 check(exception, "JS and API calls should both throw an exception while ", description);
284 check(functionReturnsTrue("(function(a, b) { return a.constructor === b.constructor; })", exception, jsResult.error()), "JS and API calls should both throw the same exception while ", description);
285 } else {
286 check(!exception, "JS and API calls should both not throw an exception while ", description);
287 check(JSValueIsStrictEqual(context, result, jsResult.value()), "JS result and API calls should return the same value while ", description);
288 }
289}

292{
293 APIVector<JSObjectRef> result(context);
294 JSObjectRef array = JSValueToObject(context, evaluateScript(

301 ASSERT(object);
302 result.append(object);
303 }
304
305 return result;

384void TestAPI::symbolsSetPropertyForKey()
385{
386 auto jsObjects = interestingObjects();
387 auto apiObjects = interestingObjects();
388 auto keys = interestingKeys();
389
390 JSValueRef theAnswer = JSValueMakeNumber(context, 42);
391 for (size_t i = 0; i < jsObjects.size(); i++) {
392 for (auto& key : keys) {
393 JSObjectRef jsObject = jsObjects[i];
394 JSObjectRef apiObject = apiObjects[i];
395 checkJSAndAPIMatch(
396 [&] {
397 return callFunction(setFunction, jsObject, key, theAnswer);
398 } , [&] (JSValueRef* exception) {
399 JSObjectSetPropertyForKey(context, apiObject, key, theAnswer, kJSPropertyAttributeNone, exception);
400 return JSValueMakeUndefined(context);
401 }, "setting property keys to the answer");
402 // Check get is the same on API object.
403 checkJSAndAPIMatch(
404 [&] {
405 return callFunction(getFunction, apiObject, key);
406 }, [&] (JSValueRef* exception) {
407 return JSObjectGetPropertyForKey(context, apiObject, key, exception);
408 }, "getting property keys from API objects");
409 // Check get is the same on respective objects.
410 checkJSAndAPIMatch(
411 [&] {
412 return callFunction(getFunction, jsObject, key);
413 }, [&] (JSValueRef* exception) {
414 return JSObjectGetPropertyForKey(context, apiObject, key, exception);
415 }, "getting property keys from respective objects");

481void TestAPI::promiseResolveTrue()
482{
483 JSObjectRef resolve;
484 JSObjectRef reject;
485 JSValueRef exception = nullptr;
486 JSObjectRef promise = JSObjectMakeDeferredPromise(context, &resolve, &reject, &exception);
487 check(!exception, "No exception should be thrown creating a deferred promise");
488
489 // Ugh, we don't have any C API that takes blocks... so we do this hack to capture the runner.
490 static TestAPI* tester = this;
491 static bool passedTrueCalled = false;
492
493 APIString trueString("passedTrue");
494 auto passedTrue = [](JSContextRef ctx, JSObjectRef, JSObjectRef, size_t argumentCount, const JSValueRef arguments[], JSValueRef*) -> JSValueRef {
495 tester->check(argumentCount && JSValueIsStrictEqual(ctx, arguments[0], JSValueMakeBoolean(ctx, true)), "function should have been called back with true");
496 passedTrueCalled = true;
497 return JSValueMakeUndefined(ctx);
498 };
499
500 APIString thenString("then");
501 JSValueRef thenFunction = JSObjectGetProperty(context, promise, thenString, &exception);
502 check(!exception && thenFunction && JSValueIsObject(context, thenFunction), "Promise should have a then object property");
503
504 JSValueRef passedTrueFunction = JSObjectMakeFunctionWithCallback(context, trueString, passedTrue);
505 JSObjectCallAsFunction(context, const_cast<JSObjectRef>(thenFunction), promise, 1, &passedTrueFunction, &exception);
506 check(!exception, "No exception should be thrown setting up callback");
507
508 auto trueValue = JSValueMakeBoolean(context, true);
509 JSObjectCallAsFunction(context, resolve, resolve, 1, &trueValue, &exception);
510 check(!exception, "No exception should be thrown resolving promise");
511 check(passedTrueCalled, "then response function should have been called.");
512}

547void TestAPI::promiseUnhandledRejection()
548{
549 JSObjectRef reject = nullptr;
550 JSValueRef exception = nullptr;
551 static auto promise = JSObjectMakeDeferredPromise(context, nullptr, &reject, &exception);
552 check(!exception, "creating a (reject-only) deferred promise should not throw");
553 static auto reason = JSValueMakeString(context, APIString("reason"));
554
555 static TestAPI* tester = this;
556 static bool callbackCalled = false;
557 auto callback = [](JSContextRef ctx, JSObjectRef, JSObjectRef, size_t argumentCount, const JSValueRef arguments[], JSValueRef*) -> JSValueRef {
558 tester->check(argumentCount && JSValueIsStrictEqual(ctx, arguments[0], promise), "callback should receive rejected promise as first argument");
559 tester->check(argumentCount > 1 && JSValueIsStrictEqual(ctx, arguments[1], reason), "callback should receive rejection reason as second argument");
560 tester->check(argumentCount == 2, "callback should not receive a third argument");
561 callbackCalled = true;
562 return JSValueMakeUndefined(ctx);
563 };
564 auto callbackFunction = JSObjectMakeFunctionWithCallback(context, APIString("callback"), callback);
565
566 JSGlobalContextSetUnhandledRejectionCallback(context, callbackFunction, &exception);
567 check(!exception, "setting unhandled rejection callback should not throw");
568
569 JSObjectCallAsFunction(context, reject, reject, 1, &reason, &exception);
570 check(!exception && callbackCalled, "unhandled rejection callback should be called upon unhandled rejection");
571}

573void TestAPI::promiseUnhandledRejectionFromUnhandledRejectionCallback()
574{
575 static JSObjectRef reject;
576 static JSValueRef exception = nullptr;
577 JSObjectMakeDeferredPromise(context, nullptr, &reject, &exception);
578 check(!exception, "creating a (reject-only) deferred promise should not throw");
579
580 static auto callbackCallCount = 0;
581 auto callback = [](JSContextRef ctx, JSObjectRef, JSObjectRef, size_t, const JSValueRef[], JSValueRef*) -> JSValueRef {
582 if (!callbackCallCount)
583 JSObjectCallAsFunction(ctx, reject, reject, 0, nullptr, &exception);
584 callbackCallCount++;
585 return JSValueMakeUndefined(ctx);
586 };
587 auto callbackFunction = JSObjectMakeFunctionWithCallback(context, APIString("callback"), callback);
588
589 JSGlobalContextSetUnhandledRejectionCallback(context, callbackFunction, &exception);
590 check(!exception, "setting unhandled rejection callback should not throw");
591
592 callFunction("(function () { Promise.reject(); })");
593 check(!exception && callbackCallCount == 2, "unhandled rejection from unhandled rejection callback should also trigger the callback");
594}

596void TestAPI::promiseEarlyHandledRejections()
597{
598 JSValueRef exception = nullptr;
599
600 static bool callbackCalled = false;
601 auto callback = [](JSContextRef ctx, JSObjectRef, JSObjectRef, size_t, const JSValueRef[], JSValueRef*) -> JSValueRef {
602 callbackCalled = true;
603 return JSValueMakeUndefined(ctx);
604 };
605 auto callbackFunction = JSObjectMakeFunctionWithCallback(context, APIString("callback"), callback);
606
607 JSGlobalContextSetUnhandledRejectionCallback(context, callbackFunction, &exception);
608 check(!exception, "setting unhandled rejection callback should not throw");
609
610 callFunction("(function () { const p = Promise.reject(); p.catch(() => {}); })");
611 check(!callbackCalled, "unhandled rejection callback should not be called for synchronous early-handled rejection");
612
613 callFunction("(function () { const p = Promise.reject(); Promise.resolve().then(() => { p.catch(() => {}); }); })");
614 check(!callbackCalled, "unhandled rejection callback should not be called for asynchronous early-handled rejection");
615}

639void TestAPI::topCallFrameAccess()
640{
641 {
642 JSObjectRef function = JSValueToObject(context, evaluateScript("(function () { })").value(), nullptr);
643 APIString argumentsString("arguments");
644 auto arguments = JSObjectGetProperty(context, function, argumentsString, nullptr);
645 check(JSValueIsNull(context, arguments), "vm.topCallFrame access from C++ world should use nullptr internally for arguments");
646 }
647 {
648 JSObjectRef arguments = JSValueToObject(context, evaluateScript("(function ok(v) { return ok.arguments; })(42)").value(), nullptr);
649 check(!JSValueIsNull(context, arguments), "vm.topCallFrame is materialized and we found the caller function's arguments");
650 }
651 {
652 JSObjectRef function = JSValueToObject(context, evaluateScript("(function () { })").value(), nullptr);
653 APIString callerString("caller");
654 auto caller = JSObjectGetProperty(context, function, callerString, nullptr);
655 check(JSValueIsNull(context, caller), "vm.topCallFrame access from C++ world should use nullptr internally for caller");
656 }
657 {
658 JSObjectRef caller = JSValueToObject(context, evaluateScript("(function () { return (function ok(v) { return ok.caller; })(42); })()").value(), nullptr);
659 check(!JSValueIsNull(context, caller), "vm.topCallFrame is materialized and we found the caller function's caller");
660 }
661 {
662 JSObjectRef caller = JSValueToObject(context, evaluateScript("(function ok(v) { return ok.caller; })(42)").value(), nullptr);
663 check(JSValueIsNull(context, caller), "vm.topCallFrame is materialized and we found the caller function's caller, but the caller is global code");
664 }

768int testCAPIViaCpp(const char* filter)
769{
770 dataLogLn("Starting C-API tests in C++");
771
772 Deque<RefPtr<SharedTask<void(TestAPI&)>>> tasks;
773
774 auto shouldRun = [&] (const char* testName) -> bool {
775 return !filter || WTF::findIgnoringASCIICaseWithoutLength(testName, filter) != WTF::notFound;
776 };
777
778 RUN(topCallFrameAccess());
779 RUN(basicSymbol());
780 RUN(symbolsTypeof());
781 RUN(symbolsDescription());
782 RUN(symbolsGetPropertyForKey());
783 RUN(symbolsSetPropertyForKey());
784 RUN(symbolsHasPropertyForKey());
785 RUN(symbolsDeletePropertyForKey());
786 RUN(promiseResolveTrue());
787 RUN(promiseRejectTrue());
788 RUN(promiseUnhandledRejection());
789 RUN(promiseUnhandledRejectionFromUnhandledRejectionCallback());
790 RUN(promiseDrainDoesNotEatExceptions());
791 RUN(promiseEarlyHandledRejections());
792 RUN(markedJSValueArrayAndGC());
793 RUN(classDefinitionWithJSSubclass());
794 RUN(proxyReturnedWithJSSubclassing());
795 RUN(testJSObjectSetOnGlobalObjectSubclassDefinition());
796
797 if (tasks.isEmpty())
798 return 0;
799
800 Lock lock;
801
802 static Atomic<int> failed { 0 };
803 Vector<Ref<Thread>> threads;
804 for (unsigned i = filter ? 1 : WTF::numberOfProcessorCores(); i--;) {
805 threads.append(Thread::create(
806 "Testapi via C++ thread",
807 [&] () {
808 TestAPI tester;
809 for (;;) {
810 RefPtr<SharedTask<void(TestAPI&)>> task;
811 {
812 Locker locker { lock };
813 if (tasks.isEmpty())
814 break;
815 task = tasks.takeFirst();
816 }
817
818 task->run(tester);
819 }
820 failed.exchangeAdd(tester.failed());
821 }));
822 }
823
824 for (auto& thread : threads)
825 thread->waitForCompletion();
826
827 dataLogLn("C-API tests in C++ had ", failed.load(), " failures");
828 return failed.load();
829}
