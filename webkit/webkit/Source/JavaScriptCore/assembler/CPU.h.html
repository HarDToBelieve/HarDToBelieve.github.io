26#pragma once
27
28#include "Options.h"
29#include <wtf/NumberOfCores.h>
30#include <wtf/StdIntExtras.h>
31
32namespace JSC {
33
34using UCPUStrictInt32 = UCPURegister;
35
36constexpr bool isARMv7IDIVSupported()
37{
38#if HAVE(ARM_IDIV_INSTRUCTIONS)
39 return true;
40#else
41 return false;
42#endif
43}
44
45constexpr bool isARM64()
46{
47#if CPU(ARM64)
48 return true;
49#else
50 return false;
51#endif
52}
53
54constexpr bool isARM64E()
55{
56#if CPU(ARM64E)
57 return true;
58#else
59 return false;
60#endif
61}
62
63constexpr bool isX86()
64{
65#if CPU(X86_64) || CPU(X86)
66 return true;
67#else
68 return false;
69#endif
70}
71
72constexpr bool isX86_64()
73{
74#if CPU(X86_64)
75 return true;
76#else
77 return false;
78#endif
79}
80
81constexpr bool is64Bit()
82{
83#if USE(JSVALUE64)
84 return true;
85#else
86 return false;
87#endif
88}
89
90constexpr bool is32Bit()
91{
92 return !is64Bit();
93}
94
95constexpr bool isAddress64Bit()
96{
97 return sizeof(void*) == 8;
98}
99
100constexpr bool isAddress32Bit()
101{
102 return !isAddress64Bit();
103}
104
105constexpr bool isMIPS()
106{
107#if CPU(MIPS)
108 return true;
109#else
110 return false;
111#endif
112}
113
114inline bool optimizeForARMv7IDIVSupported()
115{
116 return isARMv7IDIVSupported() && Options::useArchitectureSpecificOptimizations();
117}
118
119inline bool optimizeForARM64()
120{
121 return isARM64() && Options::useArchitectureSpecificOptimizations();
122}
123
124inline bool optimizeForX86()
125{
126 return isX86() && Options::useArchitectureSpecificOptimizations();
127}
128
129inline bool optimizeForX86_64()
130{
131 return isX86_64() && Options::useArchitectureSpecificOptimizations();
132}
133
134inline bool hasSensibleDoubleToInt()
135{
136 return optimizeForX86();
137}
138
139#if (CPU(X86) || CPU(X86_64)) && OS(DARWIN)
140bool isKernTCSMAvailable();
141bool enableKernTCSM();
142int kernTCSMAwareNumberOfProcessorCores();
143int64_t hwL3CacheSize();
144int32_t hwPhysicalCPUMax();
145#else
146ALWAYS_INLINE bool isKernTCSMAvailable() { return false; }
147ALWAYS_INLINE bool enableKernTCSM() { return false; }
148ALWAYS_INLINE int kernTCSMAwareNumberOfProcessorCores() { return WTF::numberOfProcessorCores(); }
149ALWAYS_INLINE int64_t hwL3CacheSize() { return 0; }
150ALWAYS_INLINE int32_t hwPhysicalCPUMax() { return kernTCSMAwareNumberOfProcessorCores(); }
151#endif
