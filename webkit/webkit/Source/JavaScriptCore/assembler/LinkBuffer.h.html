52// LinkBuffer:
53//
54// This class assists in linking code generated by the macro assembler, once code generation
55// has been completed, and the code has been copied to is final location in memory. At this
56// time pointers to labels within the code may be resolved, and relative offsets to external
57// addresses may be fixed.
58//
59// Specifically:
60// * Jump objects may be linked to external targets,
61// * The address of Jump objects may taken, such that it can later be relinked.
62// * The return address of a Call may be acquired.
63// * The address of a Label pointing into the code may be resolved.
64// * The value referenced by a DataLabel may be set.
65//
66class LinkBuffer {
67 WTF_MAKE_NONCOPYABLE(LinkBuffer); WTF_MAKE_FAST_ALLOCATED;

184 template<PtrTag tag>
185 void link(Call call, FunctionPtr<tag> function)
186 {
187 ASSERT(call.isFlagSet(Call::Linkable));
188 call.m_label = applyOffset(call.m_label);
189 MacroAssembler::linkCall(code(), call, function);
190 }

354#if ENABLE(BRANCH_COMPACTION)
355 int executableOffsetFor(int location)
356 {
357 // Returning 0 in this case works because at location <
358 // sizeof(int32_t), no compaction could have happened before this
359 // point as the assembler could not have placed a branch instruction
360 // within this space that required compaction.
361 if (location < static_cast<int>(sizeof(int32_t)))
362 return 0;
363 return bitwise_cast<int32_t*>(m_assemblerStorage.buffer())[location / sizeof(int32_t) - 1];
364 }
365#endif

367 template <typename T> T applyOffset(T src)
368 {
369#if ENABLE(BRANCH_COMPACTION)
370 src = src.labelAtOffset(-executableOffsetFor(src.offset()));
371#endif
372 return src;
373 }

383 template<PtrTag tag, typename T>
384 void* getLinkerAddress(T src)
385 {
386 void *code = this->code();
387 void* address = MacroAssembler::getLinkerAddress<tag>(code, src);
388 RELEASE_ASSERT(code <= untagCodePtr<tag>(address) && untagCodePtr<tag>(address) <= static_cast<char*>(code) + size());
389 return address;
390 }

454// Use this to finalize code, like so:
455//
456// CodeRef code = FINALIZE_CODE(linkBuffer, tag, "my super thingy number %d", number);
457//
458// Which, in disassembly mode, will print:
459//
460// Generated JIT code for my super thingy number 42:
461// Code at [0x123456, 0x234567]:
462// 0x123456: mov $0, 0
463// 0x12345a: ret
464//
465// ... and so on.
466//
467// Note that the format string and print arguments are only evaluated when dumpDisassembly
468// is true, so you can hide expensive disassembly-only computations inside there.
469
470#define FINALIZE_CODE(linkBufferReference, resultPtrTag, ...) \
471 FINALIZE_CODE_IF((JSC::Options::asyncDisassembly() || JSC::Options::dumpDisassembly()), linkBufferReference, resultPtrTag, __VA_ARGS__)
472
473#define FINALIZE_DFG_CODE(linkBufferReference, resultPtrTag, ...) \
474 FINALIZE_CODE_IF((JSC::Options::asyncDisassembly() || JSC::Options::dumpDisassembly() || Options::dumpDFGDisassembly()), linkBufferReference, resultPtrTag, __VA_ARGS__)
475
476#define FINALIZE_REGEXP_CODE(linkBufferReference, resultPtrTag, dataLogFArgumentsForHeading) \
477 FINALIZE_CODE_IF(JSC::Options::asyncDisassembly() || JSC::Options::dumpDisassembly() || Options::dumpRegExpDisassembly(), linkBufferReference, resultPtrTag, dataLogFArgumentsForHeading)
