26#pragma once
27
28#if ENABLE(B3_JIT)
29
30#include "B3Value.h"
31#include "Reg.h"
32
33namespace JSC { namespace B3 {
34
35class JS_EXPORT_PRIVATE ArgumentRegValue final : public Value {
36public:
37 static bool accepts(Kind kind) { return kind == ArgumentReg; }
38
39 ~ArgumentRegValue() final;
40
41 Reg argumentReg() const { return m_reg; }
42
43 B3_SPECIALIZE_VALUE_FOR_NO_CHILDREN
44
45private:
46 void dumpMeta(CommaPrinter&, PrintStream&) const final;
47
48 friend class Procedure;
49 friend class Value;
50
51 static Opcode opcodeFromConstructor(Origin, Reg) { return ArgumentReg; }
52
53 ArgumentRegValue(Origin origin, Reg reg)
54 : Value(CheckedOpcode, ArgumentReg, reg.isGPR() ? pointerType() : Double, Zero, origin)
55 , m_reg(reg)
56 {
57 ASSERT(reg.isSet());
58 }
59
60 Reg m_reg;
61};
62
63} } // namespace JSC::B3
64
65#endif // ENABLE(B3_JIT)
