36template<typename ValueType, typename... Arguments>
37ValueType* BasicBlock::appendNew(Procedure& procedure, Arguments... arguments)
38{
39 ValueType* result = procedure.add<ValueType>(arguments...);
40 append(result);
41 return result;
42}
43
44template<typename ValueType, typename... Arguments>
45ValueType* BasicBlock::replaceLastWithNew(Procedure& procedure, Arguments... arguments)
46{
47 ValueType* result = procedure.add<ValueType>(arguments...);
48 replaceLast(procedure, result);
49 return result;
50}

52inline const FrequentedBlock& BasicBlock::taken() const
53{
54 ASSERT(last()->opcode() == Jump || last()->opcode() == Branch);
55 return m_successors[0];
