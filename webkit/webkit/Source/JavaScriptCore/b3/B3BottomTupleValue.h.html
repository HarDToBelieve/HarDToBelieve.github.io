32namespace JSC { namespace B3 {

34class JS_EXPORT_PRIVATE BottomTupleValue final : public Value {
35public:
36 static bool accepts(Kind kind) { return kind == BottomTuple; }
37
38 ~BottomTupleValue() final;
39
40 B3_SPECIALIZE_VALUE_FOR_NO_CHILDREN
41
42private:
43 void dumpMeta(CommaPrinter&, PrintStream&) const final;
44
45 static Opcode opcodeFromConstructor(Origin, Type) { return BottomTuple; }
46
47 BottomTupleValue(Origin origin, Type type)
48 : Value(CheckedOpcode, BottomTuple, type, Zero, origin)
49 {
50 ASSERT(type.isTuple());
51 }
52
53 friend class Procedure;
54 friend class Value;
55};
