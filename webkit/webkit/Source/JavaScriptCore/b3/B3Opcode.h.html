36// Warning: In B3, an Opcode is just one part of a Kind. Kind is used the way that an opcode
37// would be used in simple IRs. See B3Kind.h.
38
39enum Opcode : uint8_t {
40 // A no-op that returns Void, useful for when you want to remove a value.
41 Nop,
42
43 // Polymorphic identity, usable with any value type.
44 Identity,
45
46 // This is an identity, but we prohibit the compiler from realizing this until the bitter end. This can
47 // be used to block reassociation and other compiler reasoning, if we find that it's wrong or
48 // unprofitable and we need an escape hatch.
49 Opaque,
50
51 // Constants. Use the ConstValue* classes. Constants exist in the control flow, so that we can
52 // reason about where we would construct them. Large constants are expensive to create.
53 Const32,
54 Const64,
55 ConstDouble,
56 ConstFloat,
57
58 // Tuple filled with zeros. This appears when Tuple Patchpoints are replaced with Bottom values.
59 BottomTuple,
60
61 // B3 supports non-SSA variables. These are accessed using Get and Set opcodes. Use the
62 // VariableValue class. It's a good idea to run fixSSA() to turn these into SSA. The
63 // optimizer will do that eventually, but if your input tends to use these opcodes, you
64 // should run fixSSA() directly before launching the optimizer.
65 Set,
66 Get,
67
68 // Gets the base address of a StackSlot.
69 SlotBase,
70
71 // The magical argument register. This is viewed as executing at the top of the program
72 // regardless of where in control flow you put it, and the compiler takes care to ensure that we
73 // don't clobber the value by register allocation or calls (either by saving the argument to the
74 // stack or preserving it in a callee-save register). Use the ArgumentRegValue class. The return
75 // type is either pointer() (for GPRs) or Double (for FPRs).
76 ArgumentReg,
77
78 // The frame pointer. You can put this anywhere in control flow but it will always yield the
79 // frame pointer, with a caveat: if our compiler changes the frame pointer temporarily for some
80 // silly reason, the FramePointer intrinsic will return where the frame pointer *should* be not
81 // where it happens to be right now.
82 FramePointer,
83
84 // Polymorphic math, usable with any value type.
85 Add,
86 Sub,
87 Mul,
88 Div, // All bets are off as to what will happen when you execute this for -2^31/-1 and x/0.
89 UDiv,
90 Mod, // All bets are off as to what will happen when you execute this for -2^31%-1 and x%0.
91 UMod,
92
93 // Polymorphic negation. Note that we only need this for floating point, since integer negation
94 // is exactly like Sub(0, x). But that's not true for floating point. Sub(0, 0) is 0, while
95 // Neg(0) is -0. Also, we canonicalize Sub(0, x) into Neg(x) in case of integers.
96 Neg,
97
98 // Integer math.
99 BitAnd,
100 BitOr,
101 BitXor,
102 Shl,
103 SShr, // Arithmetic Shift.
104 ZShr, // Logical Shift.
105 RotR, // Rotate Right.
106 RotL, // Rotate Left.
107 Clz, // Count leading zeros.
108
109 // Floating point math.
110 Abs,
111 Ceil,
112 Floor,
113 Sqrt,
114
115 // Casts and such.
116 // Bitwise Cast of Double->Int64 or Int64->Double
117 BitwiseCast,
118 // Takes and returns Int32:
119 SExt8,
120 SExt16,
121 // Takes Int32 and returns Int64:
122 SExt32,
123 ZExt32,
124 // Does a bitwise truncation of Int64->Int32 and Double->Float:
125 Trunc,
126 // Takes ints and returns floating point value. Note that we don't currently provide the opposite operation,
127 // because double-to-int conversions have weirdly different semantics on different platforms. Use
128 // a patchpoint if you need to do that.
129 IToD,
130 IToF,
131 // Convert between double and float.
132 FloatToDouble,
133 DoubleToFloat,
134
135 // Polymorphic comparisons, usable with any value type. Returns int32 0 or 1. Note that "Not"
136 // is just Equal(x, 0), and "ToBoolean" is just NotEqual(x, 0).
137 Equal,
138 NotEqual,
139 LessThan,
140 GreaterThan,
141 LessEqual,
142 GreaterEqual,
143
144 // Integer comparisons. Returns int32 0 or 1.
145 Above,
146 Below,
147 AboveEqual,
148 BelowEqual,
149
150 // Unordered floating point compare: values are equal or either one is NaN.
151 EqualOrUnordered,
152
153 // SSA form of conditional move. The first child is evaluated for truthiness. If true, the second child
154 // is returned. Otherwise, the third child is returned.
155 Select,
156
157 // Memory loads. Opcode indicates how we load and the loaded type. These use MemoryValue.
158 // These return Int32:
159 Load8Z,
160 Load8S,
161 Load16Z,
162 Load16S,
163 // This returns whatever the return type is:
164 Load,
165
166 // Memory stores. Opcode indicates how the value is stored. These use MemoryValue.
167 // These take an Int32 value:
168 Store8,
169 Store16,
170 // This is a polymorphic store for Int32, Int64, Float, and Double.
171 Store,
172
173 // Atomic compare and swap that returns a boolean. May choose to do nothing and return false. You can
174 // usually assume that this is faster and results in less code than AtomicStrongCAS, though that's
175 // not necessarily true on Intel, if instruction selection does its job. Imagine that this opcode is
176 // as if you did this atomically:
177 //
178 // template<typename T>
179 // bool AtomicWeakCAS(T expectedValue, T newValue, T* ptr)
180 // {
181 // if (!rand())
182 // return false; // Real world example of this: context switch on ARM while doing CAS.
183 // if (*ptr != expectedValue)
184 // return false;
185 // *ptr = newValue;
186 // return true;
187 // }
188 //
189 // Note that all atomics put the pointer last to be consistent with how loads and stores work. This
190 // is a goofy tradition, but it's harmless, and better than being inconsistent.
191 //
192 // Note that weak CAS has no fencing guarantees when it fails. This means that the following
193 // transformation is always valid:
194 //
195 // Before:
196 //
197 // Branch(AtomicWeakCAS(expected, new, ptr))
198 // Successors: Then:#success, Else:#fail
199 //
200 // After:
201 //
202 // Branch(Equal(Load(ptr), expected))
203 // Successors: Then:#attempt, Else:#fail
204 // BB#attempt:
205 // Branch(AtomicWeakCAS(expected, new, ptr))
206 // Successors: Then:#success, Else:#fail
207 //
208 // Both kinds of CAS for non-canonical widths (Width8 and Width16) ignore the irrelevant bits of the
209 // input.
210 AtomicWeakCAS,
211
212 // Atomic compare and swap that returns the old value. Does not have the nondeterminism of WeakCAS.
213 // This is a bit more code and a bit slower in some cases, though not by a lot. Imagine that this
214 // opcode is as if you did this atomically:
215 //
216 // template<typename T>
217 // T AtomicStrongCAS(T expectedValue, T newValue, T* ptr)
218 // {
219 // T oldValue = *ptr;
220 // if (oldValue == expectedValue)
221 // *ptr = newValue;
222 // return oldValue
223 // }
224 //
225 // AtomicStrongCAS sign-extends its result for subwidth operations.
226 //
227 // Note that AtomicWeakCAS and AtomicStrongCAS sort of have this kind of equivalence:
228 //
229 // AtomicWeakCAS(@exp, @new, @ptr) == Equal(AtomicStrongCAS(@exp, @new, @ptr), @exp)
230 //
231 // Assuming that the WeakCAS does not spuriously fail, of course.
232 AtomicStrongCAS,
233
234 // Atomically ___ a memory location and return the old value. Syntax:
235 //
236 // @oldValue = AtomicXchg___(@operand, @ptr)
237 //
238 // For non-canonical widths (Width8 and Width16), these return sign-extended results and ignore the
239 // irrelevant bits of their inputs.
240 AtomicXchgAdd,
241 AtomicXchgAnd,
242 AtomicXchgOr,
243 AtomicXchgSub,
244 AtomicXchgXor,
245
246 // FIXME: Maybe we should have AtomicXchgNeg.
247 // https://bugs.webkit.org/show_bug.cgi?id=169252
248
249 // Atomically exchange a value with a memory location. Syntax:
250 //
251 // @oldValue = AtomicXchg(@newValue, @ptr)
252 AtomicXchg,
253
254 // Introduce an invisible dependency for blocking motion of loads with respect to each other. Syntax:
255 //
256 // @result = Depend(@phantom)
257 //
258 // This is eventually codegenerated to have local semantics as if we did:
259 //
260 // @result = $0
261 //
262 // But it ensures that the users of @result cannot execute until @phantom is computed.
263 //
264 // The compiler is not allowed to reason about the fact that Depend codegenerates this way. Any kind
265 // of transformation or analysis that relies on the insight that Depend is really zero is unsound,
266 // because it unlocks reordering of users of @result and @phantom.
267 //
268 // On X86, this is lowered to a load-load fence and @result folds to zero.
269 //
270 // On ARM, this is lowered as if like:
271 //
272 // @result = BitXor(@phantom, @phantom)
273 //
274 // Except that the compiler never gets an opportunity to simplify out the BitXor.
275 Depend,
276
277 // This is used to compute the actual address of a Wasm memory operation. It takes an IntPtr
278 // and a pinned register then computes the appropriate IntPtr address. For the use-case of
279 // Wasm it is important that the first child initially be a ZExt32 so the top bits are cleared.
280 // We do WasmAddress(ZExt32(ptr), ...) so that we can avoid generating extraneous moves in Air.
281 WasmAddress,
282
283 // This is used to represent standalone fences - i.e. fences that are not part of other
284 // instructions. It's expressive enough to expose mfence on x86 and dmb ish/ishst on ARM. On
285 // x86, it also acts as a compiler store-store fence in those cases where it would have been a
286 // dmb ishst on ARM.
287 Fence,
288
289 // This is a regular ordinary C function call, using the system C calling convention. Make sure
290 // that the arguments are passed using the right types. The first argument is the callee.
291 CCall,
292
293 // This is a patchpoint. Use the PatchpointValue class. This is viewed as behaving like a call,
294 // but only emits code via a code generation callback. That callback gets to emit code inline.
295 // You can pass a stackmap along with constraints on how each stackmap argument must be passed.
296 // It's legal to request that a stackmap argument is in some register and it's legal to request
297 // that a stackmap argument is at some offset from the top of the argument passing area on the
298 // stack.
299 Patchpoint,
300
301 // This is a projection out of a tuple. Currently only Patchpoints, Get, Phi, and BottomTuple can produce tuples.
302 // It's assumumed that each entry in a tuple has a fixed Numeric B3 Type (i.e. not Void or Tuple).
303 Extract,
304
305 // Checked math. Use the CheckValue class. Like a Patchpoint, this takes a code generation
306 // callback. That callback gets to emit some code after the epilogue, and gets to link the jump
307 // from the check, and the choice of registers. You also get to supply a stackmap. Note that you
308 // are not allowed to jump back into the mainline code from your slow path, since the compiler
309 // will assume that the execution of these instructions proves that overflow didn't happen. For
310 // example, if you have two CheckAdd's:
311 //
312 // a = CheckAdd(x, y)
313 // b = CheckAdd(x, y)
314 //
315 // Then it's valid to change this to:
316 //
317 // a = CheckAdd(x, y)
318 // b = Identity(a)
319 //
320 // This is valid regardless of the callbacks used by the two CheckAdds. They may have different
321 // callbacks. Yet, this transformation is valid even if they are different because we know that
322 // after the first CheckAdd executes, the second CheckAdd could not have possibly taken slow
323 // path. Therefore, the second CheckAdd's callback is irrelevant.
324 //
325 // Note that the first two children of these operations have ValueRep's as input constraints but do
326 // not have output constraints.
327 CheckAdd,
328 CheckSub,
329 CheckMul,
330
331 // Check that side-exits. Use the CheckValue class. Like CheckAdd and friends, this has a
332 // stackmap with a generation callback. This takes an int argument that this branches on, with
333 // full branch fusion in the instruction selector. A true value jumps to the generator's slow
334 // path. Note that the predicate child is has both an input ValueRep. The input constraint must be
335 // WarmAny. It will not have an output constraint.
336 Check,
337
338 // Special Wasm opcode that takes a Int32, a special pinned gpr and an offset. This node exists
339 // to allow us to CSE WasmBoundsChecks if both use the same pointer and one dominates the other.
340 // Without some such node B3 would not have enough information about the inner workings of wasm
341 // to be able to perform such optimizations.
342 WasmBoundsCheck,
343
344 // SSA support, in the style of DFG SSA.
345 Upsilon, // This uses the UpsilonValue class.
346 Phi,
347
348 // Jump.
349 Jump,
350
351 // Polymorphic branch, usable with any integer type. Branches if not equal to zero. The 0-index
352 // successor is the true successor.
353 Branch,
354
355 // Switch. Switches over either Int32 or Int64. Uses the SwitchValue class.
356 Switch,
357
358 // Multiple entrypoints are supported via the EntrySwitch operation. Place this in the root
359 // block and list the entrypoints as the successors. All blocks backwards-reachable from
360 // EntrySwitch are duplicated for each entrypoint.
361 EntrySwitch,
362
363 // Return. Note that B3 procedures don't know their return type, so this can just return any
364 // type.
365 Return,
366
367 // This is a terminal that indicates that we will never get here.
368 Oops
369};

371inline bool isCheckMath(Opcode opcode)
372{
373 switch (opcode) {
374 case CheckAdd:
375 case CheckSub:
376 case CheckMul:
377 return true;
378 default:
379 return false;
380 }
381}

392inline bool isConstant(Opcode opcode)
393{
394 switch (opcode) {
395 case Const32:
396 case Const64:
397 case ConstDouble:
398 case ConstFloat:
399 return true;
400 default:
401 return false;
402 }
403}

405inline Opcode opcodeForConstant(Type type)
406{
407 switch (type.kind()) {
408 case Int32: return Const32;
409 case Int64: return Const64;
410 case Float: return ConstFloat;
411 case Double: return ConstDouble;
412 default:
413 RELEASE_ASSERT_NOT_REACHED();
414 }
415}

417inline bool isDefinitelyTerminal(Opcode opcode)
418{
419 switch (opcode) {
420 case Jump:
421 case Branch:
422 case Switch:
423 case Oops:
424 case Return:
425 return true;
426 default:
427 return false;

517inline bool isMemoryAccess(Opcode opcode)
518{

522inline Opcode signExtendOpcode(Width width)
523{
524 switch (width) {
525 case Width8:
526 return SExt8;
527 case Width16:
528 return SExt16;
529 default:
530 RELEASE_ASSERT_NOT_REACHED();
531 return Oops;
