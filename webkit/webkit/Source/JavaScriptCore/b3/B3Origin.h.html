32namespace JSC { namespace B3 {
33
34// Whoever generates B3IR can choose to put origins on values. When you do this, B3 will be able to
35// account, down to the machine code, which instruction corresponds to which origin. B3
36// transformations must preserve Origins carefully. It's an error to write a transformation that
37// either drops Origins or lies about them.
38class Origin {
39public:
40 explicit Origin(const void* data = nullptr)
41 : m_data(data)
42 {
43 }
44
45 explicit operator bool() const { return !!m_data; }
46
47 const void* data() const { return m_data; }
48
49 bool operator==(const Origin& other) const { return m_data == other.m_data; }
50
51 // You should avoid using this. Use OriginDump instead.
52 void dump(PrintStream&) const;
53
54private:
55 const void* m_data;
56};
57
58} } // namespace JSC::B3
