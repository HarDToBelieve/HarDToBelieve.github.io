26#pragma once
27
28#if ENABLE(B3_JIT)
29
30#include "B3Value.h"
31
32namespace JSC { namespace B3 {
33
34class JS_EXPORT_PRIVATE UpsilonValue final : public Value {
35public:
36 static bool accepts(Kind kind) { return kind == Upsilon; }
37
38 ~UpsilonValue() final;
39
40 Value* phi() const { return m_phi; }
41 void setPhi(Value* phi)
42 {
43 ASSERT(child(0)->type() == phi->type());
44 ASSERT(phi->opcode() == Phi);
45 m_phi = phi;
46 }
47
48 B3_SPECIALIZE_VALUE_FOR_FIXED_CHILDREN(1)
49 B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_FIXED_CHILDREN
50
51private:
52 void dumpMeta(CommaPrinter&, PrintStream&) const final;
53
54 friend class Procedure;
55 friend class Value;
56
57 static Opcode opcodeFromConstructor(Origin, Value*, Value* = nullptr) { return Upsilon; }
58 // Note that passing the Phi during construction is optional. A valid pattern is to first create
59 // the Upsilons without the Phi, then create the Phi, then go back and tell the Upsilons about
60 // the Phi. This allows you to emit code in its natural order.
61 UpsilonValue(Origin origin, Value* value, Value* phi = nullptr)
62 : Value(CheckedOpcode, Upsilon, Void, One, origin, value)
63 , m_phi(phi)
64 {
65 if (phi)
66 ASSERT(value->type() == phi->type());
67 }
68
69 Value* m_phi;
70};
71
72} } // namespace JSC::B3
73
74#endif // ENABLE(B3_JIT)
