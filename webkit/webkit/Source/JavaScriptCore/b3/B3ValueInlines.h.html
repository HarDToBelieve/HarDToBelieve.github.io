187template<typename BottomProvider>
188void Value::replaceWithBottom(const BottomProvider& bottomProvider)
189{
190 if (m_type == Void) {
191 replaceWithNop();
192 return;
193 }
194
195 if (isConstant())
196 return;
197
198 replaceWithIdentity(bottomProvider(m_origin, m_type));
199}

319inline bool Value::isNegativeZero() const
320{
321 if (hasDouble()) {
322 double value = asDouble();
323 return !value && std::signbit(value);
324 }
325 if (hasFloat()) {
326 float value = asFloat();
327 return !value && std::signbit(value);
328 }
329 return false;
330}

332template<typename T>
333inline bool Value::isRepresentableAs() const
334{
335 switch (opcode()) {
336 case Const32:

338 case Const64:

340 case ConstDouble:
341 return B3::isRepresentableAs<T>(asDouble());
342 case ConstFloat:
343 return B3::isRepresentableAs<T>(asFloat());
344 default:
345 return false;

366template<typename Functor>
367void Value::walk(const Functor& functor, PhiChildren* phiChildren)
368{
369 GraphNodeWorklist<Value*> worklist;
370 worklist.push(this);
371 while (Value* value = worklist.pop()) {
372 WalkStatus status = functor(value);
373 switch (status) {
374 case Continue:
375 if (value->opcode() == Phi) {
376 if (phiChildren)
377 worklist.pushAll(phiChildren->at(value).values());
378 } else
379 worklist.pushAll(value->children());
380 break;
381 case IgnoreChildren:
382 break;
383 case Stop:
384 return;
385 }
386 }
387}
