47// We use this class to describe value representations at stackmaps. It's used both to force a
48// representation and to get the representation. When the B3 client forces a representation, we say
49// that it's an input. When B3 tells the client what representation it picked, we say that it's an
50// output.
51
52class ValueRep {
53 WTF_MAKE_FAST_ALLOCATED;
54public:
55 enum Kind : uint8_t {
56 // As an input representation, this means that B3 can pick any representation. As an output
57 // representation, this means that we don't know. This will only arise as an output
58 // representation for the active arguments of Check/CheckAdd/CheckSub/CheckMul.
59 WarmAny,
60
61 // Same as WarmAny, but implies that the use is cold. A cold use is not counted as a use for
62 // computing the priority of the used temporary.
63 ColdAny,
64
65 // Same as ColdAny, but also implies that the use occurs after all other effects of the stackmap
66 // value.
67 LateColdAny,
68
69 // As an input representation, this means that B3 should pick some register. It could be a
70 // register that this claims to clobber!
71 SomeRegister,
72
73 // As an input representation, this means that B3 should pick some register but that this
74 // register is then cobbered with garbage. This only works for patchpoints.
75 SomeRegisterWithClobber,
76
77 // As an input representation, this tells us that B3 should pick some register, but implies
78 // that the def happens before any of the effects of the stackmap. This is only valid for
79 // the result constraint of a Patchpoint.
80 SomeEarlyRegister,
81
82 // As an input representation, this tells us that B3 should pick some register, but implies
83 // the use happens after any defs. This is only works for patchpoints.
84 SomeLateRegister,
85
86 // As an input representation, this forces a particular register. As an output
87 // representation, this tells us what register B3 picked.
88 Register,
89
90 // As an input representation, this forces a particular register and states that
91 // the register is used late. This means that the register is used after the result
92 // is defined (i.e, the result will interfere with this as an input).
93 // It's not a valid output representation.
94 LateRegister,
95
96 // As an output representation, this tells us what stack slot B3 picked. It's not a valid
97 // input representation.
98 Stack,
99
100 // As an input representation, this forces the value to end up in the argument area at some
101 // offset. As an output representation this tells us what offset from SP B3 picked.
102 StackArgument,
103
104 // As an output representation, this tells us that B3 constant-folded the value.
105 Constant
106 };
107
108 ValueRep()
109 : m_kind(WarmAny)
110 {
111 }
112
113 explicit ValueRep(Reg reg)
114 : m_kind(Register)
115 {
116 u.reg = reg;
117 }
118
119 ValueRep(const ValueRep&) = default;
120
121 ValueRep(Kind kind)
122 : m_kind(kind)
123 {
124 ASSERT(kind == WarmAny || kind == ColdAny || kind == LateColdAny || kind == SomeRegister || kind == SomeRegisterWithClobber || kind == SomeEarlyRegister || kind == SomeLateRegister);
125 }
126
127#if ENABLE(WEBASSEMBLY)
128 ValueRep(Wasm::ValueLocation location)
129 {
130 switch (location.kind()) {
131 case Wasm::ValueLocation::Kind::Register:
132 m_kind = Register;
133 u.reg = location.reg();
134 break;
135 case Wasm::ValueLocation::Kind::Stack:
136 m_kind = Stack;
137 u.offsetFromFP = location.offsetFromFP();
138 break;
139 case Wasm::ValueLocation::Kind::StackArgument:
140 m_kind = StackArgument;
141 u.offsetFromSP = location.offsetFromSP();
142 break;
143 default:
144 ASSERT_NOT_REACHED();
145 }
146 }
147#endif
148
149 static ValueRep reg(Reg reg)
150 {
151 return ValueRep(reg);
152 }
153
154 static ValueRep lateReg(Reg reg)
155 {
156 ValueRep result(reg);
157 result.m_kind = LateRegister;
158 return result;
159 }
160
161 static ValueRep stack(intptr_t offsetFromFP)
162 {
163 ValueRep result;
164 result.m_kind = Stack;
165 result.u.offsetFromFP = offsetFromFP;
166 return result;
167 }
168
169 static ValueRep stackArgument(intptr_t offsetFromSP)
170 {
171 ValueRep result;
172 result.m_kind = StackArgument;
173 result.u.offsetFromSP = offsetFromSP;
174 return result;
175 }
176
177 static ValueRep constant(int64_t value)
178 {
179 ValueRep result;
180 result.m_kind = Constant;
181 result.u.value = value;
182 return result;
183 }
184
185 static ValueRep constantDouble(double value)
186 {
187 return ValueRep::constant(bitwise_cast<int64_t>(value));
188 }
189
190 static ValueRep constantFloat(float value)
191 {
192 return ValueRep::constant(static_cast<uint64_t>(bitwise_cast<uint32_t>(value)));
193 }
194
195 Kind kind() const { return m_kind; }
196
197 bool operator==(const ValueRep& other) const
198 {
199 if (kind() != other.kind())
200 return false;
201 switch (kind()) {
202 case LateRegister:
203 case Register:
204 return u.reg == other.u.reg;
205 case Stack:
206 return u.offsetFromFP == other.u.offsetFromFP;
207 case StackArgument:
208 return u.offsetFromSP == other.u.offsetFromSP;
209 case Constant:
210 return u.value == other.u.value;
211 default:
212 return true;
213 }
214 }
215
216 bool operator!=(const ValueRep& other) const
217 {
218 return !(*this == other);
219 }
220
221 explicit operator bool() const { return kind() != WarmAny; }
222
223 bool isAny() const { return kind() == WarmAny || kind() == ColdAny || kind() == LateColdAny; }
224
225 bool isReg() const { return kind() == Register || kind() == LateRegister || kind() == SomeLateRegister; }
226
227 Reg reg() const
228 {
229 ASSERT(isReg());
230 return u.reg;
231 }
232
233 bool isGPR() const { return isReg() && reg().isGPR(); }
234 bool isFPR() const { return isReg() && reg().isFPR(); }
235
236 GPRReg gpr() const { return reg().gpr(); }
237 FPRReg fpr() const { return reg().fpr(); }
238
239 bool isStack() const { return kind() == Stack; }
240
241 intptr_t offsetFromFP() const
242 {
243 ASSERT(isStack());
244 return u.offsetFromFP;
245 }
246
247 bool isStackArgument() const { return kind() == StackArgument; }
248
249 intptr_t offsetFromSP() const
250 {
251 ASSERT(isStackArgument());
252 return u.offsetFromSP;
253 }
254
255 bool isConstant() const { return kind() == Constant; }
256
257 int64_t value() const
258 {
259 ASSERT(isConstant());
260 return u.value;
261 }
262
263 double doubleValue() const
264 {
265 return bitwise_cast<double>(value());
266 }
267
268 float floatValue() const
269 {
270 return bitwise_cast<float>(static_cast<uint32_t>(static_cast<uint64_t>(value())));
271 }
272
273 ValueRep withOffset(intptr_t offset) const
274 {
275 switch (kind()) {
276 case Stack:
277 return stack(offsetFromFP() + offset);
278 case StackArgument:
279 return stackArgument(offsetFromSP() + offset);
280 default:
281 return *this;
282 }
283 }

289 // Get the used registers for a vector of ValueReps.
290 template<typename VectorType>
291 static RegisterSet usedRegisters(const VectorType& vector)
292 {
293 RegisterSet result;
294 for (const ValueRep& value : vector)
295 value.addUsedRegistersTo(result);
296 return result;
297 }

305 // Computes the ValueRecovery assuming that the Value* was for a JSValue (i.e. Int64).
306 // NOTE: We should avoid putting JSValue-related methods in B3, but this was hard to avoid
307 // because some parts of JSC use ValueRecovery like a general "where my bits at" object, almost
308 // exactly like ValueRep.
309 ValueRecovery recoveryForJSValue() const;

311private:
312 union U {
313 Reg reg;
314 intptr_t offsetFromFP;
315 intptr_t offsetFromSP;
316 int64_t value;
317
318 U()
319 {
320 memset(static_cast<void*>(this), 0, sizeof(*this));
321 }
322 } u;
323 Kind m_kind;
324};
