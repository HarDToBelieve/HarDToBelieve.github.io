28#if ENABLE(B3_JIT)

30#include "B3Value.h"
31#include "CCallHelpers.h"

33namespace JSC { namespace B3 {
34
35class WasmBoundsCheckValue final : public Value {
36public:
37 static bool accepts(Kind kind) { return kind == WasmBoundsCheck; }
38
39 ~WasmBoundsCheckValue() final;
40
41 enum class Type {
42 Pinned,
43 Maximum,
44 };
45
46 union Bounds {
47 GPRReg pinnedSize;
48 size_t maximum;
49 };
50
51 unsigned offset() const { return m_offset; }
52 Type boundsType() const { return m_boundsType; }
53 Bounds bounds() const { return m_bounds; }
54
55 B3_SPECIALIZE_VALUE_FOR_FIXED_CHILDREN(1)
56 B3_SPECIALIZE_VALUE_FOR_FINAL_SIZE_FIXED_CHILDREN
57
58private:
59 void dumpMeta(CommaPrinter&, PrintStream&) const final;
60
61 friend class Procedure;
62 friend class Value;
63
64 static Opcode opcodeFromConstructor(Origin, GPRReg, Value*, unsigned) { return WasmBoundsCheck; }
65 JS_EXPORT_PRIVATE WasmBoundsCheckValue(Origin, GPRReg pinnedGPR, Value* ptr, unsigned offset);
66
67 static Opcode opcodeFromConstructor(Origin, Value*, unsigned, size_t) { return WasmBoundsCheck; }
68 JS_EXPORT_PRIVATE WasmBoundsCheckValue(Origin, Value* ptr, unsigned offset, size_t maximum);
69
70 unsigned m_offset;
71 Type m_boundsType;
72 Bounds m_bounds;
73
74};
75
76} } // namespace JSC::B3
77
78#endif // ENABLE(B3_JIT)
