37namespace JSC { namespace B3 {
38
39enum Width : int8_t {
40 Width8 = 0,
41 Width16,
42 Width32,
43 Width64
44};
45
46constexpr Width pointerWidth()
47{
48 if (sizeof(void*) == 8)
49 return Width64;
50 return Width32;
51}
52
53inline Width widthForType(Type type)
54{
55 switch (type.kind()) {
56 case Void:
57 case Tuple:
58 ASSERT_NOT_REACHED();
59 return Width8;
60 case Int32:
61 case Float:
62 return Width32;
63 case Int64:
64 case Double:
65 return Width64;
66 }
67 ASSERT_NOT_REACHED();
68 return Width8;
69}
70
71inline Width canonicalWidth(Width width)
72{
73 return std::max(Width32, width);
74}
75
76inline bool isCanonicalWidth(Width width)
77{
78 return width >= Width32;
79}
80
81Type bestType(Bank bank, Width width);
82
83inline Width conservativeWidth(Bank bank)
84{
85 return bank == GP ? pointerWidth() : Width64;
86}
87
88inline Width minimumWidth(Bank bank)
89{
90 return bank == GP ? Width8 : Width32;
91}
92
93inline unsigned bytes(Width width)
94{
95 return 1 << width;
96}
97
98inline Width widthForBytes(unsigned bytes)
99{
100 switch (bytes) {
101 case 0:
102 case 1:
103 return Width8;
104 case 2:
105 return Width16;
106 case 3:
107 case 4:
108 return Width32;
109 default:
110 return Width64;
111 }
112}
113
114inline unsigned bytesForWidth(Width width)
115{
116 switch (width) {
117 case Width8:
118 return 1;
119 case Width16:
120 return 2;
121 case Width32:
122 return 4;
123 case Width64:
124 return 8;
125 }
126 return 1;
127}
128
129inline uint64_t mask(Width width)
130{
131 switch (width) {
132 case Width8:
133 return 0x00000000000000ffllu;
134 case Width16:
135 return 0x000000000000ffffllu;
136 case Width32:
137 return 0x00000000ffffffffllu;
138 case Width64:
139 return 0xffffffffffffffffllu;
140 }
141 ASSERT_NOT_REACHED();
142}
