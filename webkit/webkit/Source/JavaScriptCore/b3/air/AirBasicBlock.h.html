47class BasicBlock {
48 WTF_MAKE_NONCOPYABLE(BasicBlock);
49 WTF_MAKE_FAST_ALLOCATED;
50public:
51 static const char* const dumpPrefix;
52 static constexpr unsigned uninsertedIndex = UINT_MAX;
53
54 typedef Vector<Inst> InstList;
55 typedef Vector<BasicBlock*, 2> PredecessorList;
56 typedef Vector<FrequentedBlock, 2> SuccessorList;
57
58 unsigned index() const { return m_index; }
59
60 // This method is exposed for phases that mess with the layout of basic blocks. Currently that means just
61 // optimizeBlockOrder().
62 void setIndex(unsigned index) { m_index = index; }
63
64 unsigned size() const { return m_insts.size(); }
65 InstList::iterator begin() { return m_insts.begin(); }
66 InstList::iterator end() { return m_insts.end(); }
67 InstList::const_iterator begin() const { return m_insts.begin(); }
68 InstList::const_iterator end() const { return m_insts.end(); }
69
70 const Inst& at(unsigned index) const { return m_insts[index]; }
71 Inst& at(unsigned index) { return m_insts[index]; }
72
73 Inst* get(unsigned index)
74 {
75 return index < size() ? &at(index) : nullptr;
76 }
77
78 const Inst& last() const { return m_insts.last(); }
79 Inst& last() { return m_insts.last(); }
80
81 void resize(unsigned size) { m_insts.resize(size); }
82
83 const InstList& insts() const { return m_insts; }
84 InstList& insts() { return m_insts; }
85
86 template<typename Inst>
87 Inst& appendInst(Inst&& inst)
88 {
89 m_insts.append(std::forward<Inst>(inst));
90 return m_insts.last();
91 }
92
93 template<typename... Arguments>
94 Inst& append(Arguments&&... arguments)
95 {
96 m_insts.append(Inst(std::forward<Arguments>(arguments)...));
97 return m_insts.last();
98 }
99
100 // The "0" case is the case to which the branch jumps, so the "then" case. The "1" case is the
101 // "else" case, and is used to represent the fall-through of a conditional branch.
102 unsigned numSuccessors() const { return m_successors.size(); }
103 FrequentedBlock successor(unsigned index) const { return m_successors[index]; }
104 FrequentedBlock& successor(unsigned index) { return m_successors[index]; }
105 const SuccessorList& successors() const { return m_successors; }
106 SuccessorList& successors() { return m_successors; }
107
108 JS_EXPORT_PRIVATE void setSuccessors(FrequentedBlock);
109 JS_EXPORT_PRIVATE void setSuccessors(FrequentedBlock, FrequentedBlock);
110
111 BasicBlock* successorBlock(unsigned index) const { return successor(index).block(); }
112 BasicBlock*& successorBlock(unsigned index) { return successor(index).block(); }
113 SuccessorCollection<BasicBlock, SuccessorList> successorBlocks()
114 {
115 return SuccessorCollection<BasicBlock, SuccessorList>(m_successors);
116 }
117 SuccessorCollection<const BasicBlock, const SuccessorList> successorBlocks() const
118 {
119 return SuccessorCollection<const BasicBlock, const SuccessorList>(m_successors);
120 }
121
122 unsigned numPredecessors() const { return m_predecessors.size(); }
123 BasicBlock* predecessor(unsigned index) const { return m_predecessors[index]; }
124 BasicBlock*& predecessor(unsigned index) { return m_predecessors[index]; }
125 const PredecessorList& predecessors() const { return m_predecessors; }
126 PredecessorList& predecessors() { return m_predecessors; }
127
128 bool addPredecessor(BasicBlock*);
129 bool removePredecessor(BasicBlock*);
130 bool replacePredecessor(BasicBlock* from, BasicBlock* to);
131 bool containsPredecessor(BasicBlock* predecessor) const { return m_predecessors.contains(predecessor); }
132
133 double frequency() const { return m_frequency; }
134
135 void dump(PrintStream&) const;
136 void deepDump(PrintStream&) const;
137
138 void dumpHeader(PrintStream&) const;
139 void dumpFooter(PrintStream&) const;
140
141private:
142 friend class BlockInsertionSet;
143 friend class Code;
144 friend class InsertionSet;
145 friend class PhaseInsertionSet;
146 template<typename> friend class B3::GenericBlockInsertionSet;
147
148 BasicBlock(unsigned index, double frequency);
149
150 unsigned m_index;
151 InstList m_insts;
152 SuccessorList m_successors;
153 PredecessorList m_predecessors;
154 double m_frequency;
155};
156
157class DeepBasicBlockDump {
158public:
159 DeepBasicBlockDump(const BasicBlock* block)
160 : m_block(block)
161 {
162 }
163
164 void dump(PrintStream& out) const
165 {
166 if (m_block)
167 m_block->deepDump(out);
168 else
169 out.print("<null>");
170 }
171
172private:
173 const BasicBlock* m_block;
174};
175
176inline DeepBasicBlockDump deepDump(const BasicBlock* block)
177{
178 return DeepBasicBlockDump(block);
179}
