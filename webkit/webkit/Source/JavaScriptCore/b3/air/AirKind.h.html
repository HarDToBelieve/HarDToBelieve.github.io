26#ifndef AirKind_h
27#define AirKind_h
28
29#if ENABLE(B3_JIT)

31#include "AirOpcode.h"
32#include <wtf/PrintStream.h>

34namespace JSC { namespace B3 { namespace Air {
35
36// Air opcodes are always carried around with some flags. These flags are understood as having no
37// meaning if they are set for an opcode to which they do not apply. This makes sense, since Air
38// is a complex instruction set and most of these flags can apply to basically any opcode. In
39// fact, it's recommended to only represent something as a flag if you believe that it is largely
40// opcode-agnostic.
41
42struct Kind {
43 Kind(Opcode opcode)
44 : opcode(opcode)
45 , effects(false)
46 {
47 }
48
49 Kind()
50 : Kind(Nop)
51 {
52 }
53
54 bool operator==(const Kind& other) const
55 {
56 return opcode == other.opcode
57 && effects == other.effects;
58 }
59
60 bool operator!=(const Kind& other) const
61 {
62 return !(*this == other);
63 }
64
65 unsigned hash() const
66 {
67 return static_cast<unsigned>(opcode) + (static_cast<unsigned>(effects) << 16);
68 }
69
70 explicit operator bool() const
71 {
72 return *this != Kind();
73 }
74
75 void dump(PrintStream&) const;
76
77 Opcode opcode;
78
79 // This is an opcode-agnostic flag that indicates that we expect that this instruction will do
80 // any of the following:
81 // - Trap.
82 // - Perform some non-arg non-control effect.
83 bool effects : 1;
84};
85
86} } } // namespace JSC::B3::Air
87
88#endif // ENABLE(B3_JIT)
89
90#endif // AirKind_h
