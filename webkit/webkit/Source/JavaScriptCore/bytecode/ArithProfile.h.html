37struct ObservedType {
38 constexpr ObservedType(uint8_t bits = TypeEmpty)
39 : m_bits(bits)
40 { }
41
42 constexpr bool sawInt32() const { return m_bits & TypeInt32; }
43 constexpr bool isOnlyInt32() const { return m_bits == TypeInt32; }
44 constexpr bool sawNumber() const { return m_bits & TypeNumber; }
45 constexpr bool isOnlyNumber() const { return m_bits == TypeNumber; }
46 constexpr bool sawNonNumber() const { return m_bits & TypeNonNumber; }
47 constexpr bool isOnlyNonNumber() const { return m_bits == TypeNonNumber; }
48 constexpr bool isEmpty() const { return !m_bits; }
49 constexpr uint8_t bits() const { return m_bits; }
50
51 constexpr ObservedType withInt32() const { return ObservedType(m_bits | TypeInt32); }
52 constexpr ObservedType withNumber() const { return ObservedType(m_bits | TypeNumber); }
53 constexpr ObservedType withNonNumber() const { return ObservedType(m_bits | TypeNonNumber); }
54 constexpr ObservedType withoutNonNumber() const { return ObservedType(m_bits & ~TypeNonNumber); }
55
56 constexpr bool operator==(const ObservedType& other) const { return m_bits == other.m_bits; }
57
58 static constexpr uint8_t TypeEmpty = 0x0;
59 static constexpr uint8_t TypeInt32 = 0x1;
60 static constexpr uint8_t TypeNumber = 0x02;
61 static constexpr uint8_t TypeNonNumber = 0x04;
62
63 static constexpr uint32_t numBitsNeeded = 3;
64
65private:
66 uint8_t m_bits { 0 };
67};
68
69class ObservedResults {
70public:
71 enum Tags : uint8_t {
72 NonNegZeroDouble = 1 << 0,
73 NegZeroDouble = 1 << 1,
74 NonNumeric = 1 << 2,
75 Int32Overflow = 1 << 3,
76 Int52Overflow = 1 << 4,
77 HeapBigInt = 1 << 5,
78 BigInt32 = 1 << 6,
79 };
80 static constexpr uint32_t numBitsNeeded = 7;
81
82 ObservedResults() = default;
83 explicit ObservedResults(uint8_t bits)
84 : m_bits(bits)
85 { }
86
87 bool didObserveNonInt32() { return m_bits & (NonNegZeroDouble | NegZeroDouble | NonNumeric | HeapBigInt | BigInt32); }
88 bool didObserveDouble() { return m_bits & (NonNegZeroDouble | NegZeroDouble); }
89 bool didObserveNonNegZeroDouble() { return m_bits & NonNegZeroDouble; }
90 bool didObserveNegZeroDouble() { return m_bits & NegZeroDouble; }
91 bool didObserveNonNumeric() { return m_bits & NonNumeric; }
92 bool didObserveBigInt() { return m_bits & (HeapBigInt | BigInt32); }
93 bool didObserveHeapBigInt() { return m_bits & HeapBigInt; }
94 bool didObserveBigInt32() { return m_bits & BigInt32; }
95 bool didObserveInt32Overflow() { return m_bits & Int32Overflow; }
96 bool didObserveInt52Overflow() { return m_bits & Int52Overflow; }
97
98private:
99 uint8_t m_bits { 0 };
100};

102template <typename BitfieldType>
103class ArithProfile {
104public:
105 ObservedResults observedResults() const
106 {
107 return ObservedResults(m_bits & ((1 << ObservedResults::numBitsNeeded) - 1));
108 }
109 bool didObserveNonInt32() const { return observedResults().didObserveNonInt32();}
110 bool didObserveDouble() const { return observedResults().didObserveDouble(); }
111 bool didObserveNonNegZeroDouble() const { return observedResults().didObserveNonNegZeroDouble(); }
112 bool didObserveNegZeroDouble() const { return observedResults().didObserveNegZeroDouble(); }
113 bool didObserveNonNumeric() const { return observedResults().didObserveNonNumeric(); }
114 bool didObserveBigInt() const { return observedResults().didObserveBigInt(); }
115 bool didObserveHeapBigInt() const { return observedResults().didObserveHeapBigInt(); }
116 bool didObserveBigInt32() const { return observedResults().didObserveBigInt32(); }
117 bool didObserveInt32Overflow() const { return observedResults().didObserveInt32Overflow(); }
118 bool didObserveInt52Overflow() const { return observedResults().didObserveInt52Overflow(); }
119
120 void setObservedNonNegZeroDouble() { setBit(ObservedResults::NonNegZeroDouble); }
121 void setObservedNegZeroDouble() { setBit(ObservedResults::NegZeroDouble); }
122 void setObservedNonNumeric() { setBit(ObservedResults::NonNumeric); }
123 void setObservedHeapBigInt() { setBit(ObservedResults::HeapBigInt); }
124 void setObservedBigInt32() { setBit(ObservedResults::BigInt32); }
125 void setObservedInt32Overflow() { setBit(ObservedResults::Int32Overflow); }
126 void setObservedInt52Overflow() { setBit(ObservedResults::Int52Overflow); }
127
128 void observeResult(JSValue value)
129 {
130 if (value.isInt32())
131 return;
132 if (value.isNumber()) {
133 m_bits |= ObservedResults::Int32Overflow | ObservedResults::Int52Overflow | ObservedResults::NonNegZeroDouble | ObservedResults::NegZeroDouble;
134 return;
135 }
136 if (value.isBigInt32()) {
137 m_bits |= ObservedResults::BigInt32;
138 return;
139 }
140 if (value && value.isHeapBigInt()) {
141 m_bits |= ObservedResults::HeapBigInt;
142 return;
143 }
144 m_bits |= ObservedResults::NonNumeric;
145 }
146
147 const void* addressOfBits() const { return &m_bits; }
148
149#if ENABLE(JIT)
150 // Sets (Int32Overflow | Int52Overflow | NonNegZeroDouble | NegZeroDouble) if it sees a
151 // double. Sets NonNumeric if it sees a non-numeric.
152 void emitObserveResult(CCallHelpers&, JSValueRegs, GPRReg tempGPR, TagRegistersMode = HaveTagRegisters);
153
154 // Sets (Int32Overflow | Int52Overflow | NonNegZeroDouble | NegZeroDouble).
155 bool shouldEmitSetDouble() const;
156 void emitSetDouble(CCallHelpers&) const;
157
158 void emitSetNonNumeric(CCallHelpers&) const;
159 bool shouldEmitSetNonNumeric() const;
160
161 bool shouldEmitSetHeapBigInt() const;
162 void emitSetHeapBigInt(CCallHelpers&) const;
163
164 bool shouldEmitSetBigInt32() const;
165#if USE(BIGINT32)
166 void emitSetBigInt32(CCallHelpers&) const;
167#endif
168
169 void emitUnconditionalSet(CCallHelpers&, BitfieldType) const;
170#endif // ENABLE(JIT)
171
172 constexpr uint32_t bits() const { return m_bits; }
173
174protected:
175 ArithProfile() = default;
176
177 bool hasBits(int mask) const { return m_bits & mask; }
178 void setBit(int mask) { m_bits |= mask; }
179
180 BitfieldType m_bits { 0 }; // We take care to update m_bits only in a single operation. We don't ever store an inconsistent bit representation to it.
181};

183/* This class stores the following components in 16 bits:
184 * - ObservedResults
185 * - ObservedType for the argument
186 */
187using UnaryArithProfileBase = uint16_t;
188class UnaryArithProfile : public ArithProfile<UnaryArithProfileBase> {
189 static constexpr unsigned argObservedTypeShift = ObservedResults::numBitsNeeded;
190
191 static_assert(argObservedTypeShift + ObservedType::numBitsNeeded <= sizeof(UnaryArithProfileBase) * 8, "Should fit in the type of the underlying bitfield.");
192
193 static constexpr UnaryArithProfileBase clearArgObservedTypeBitMask = static_cast<UnaryArithProfileBase>(~(0b111 << argObservedTypeShift));
194
195 static constexpr UnaryArithProfileBase observedTypeMask = (1 << ObservedType::numBitsNeeded) - 1;
196
197public:
198 UnaryArithProfile()
199 : ArithProfile<UnaryArithProfileBase>()
200 {
201 ASSERT(argObservedType().isEmpty());
202 ASSERT(argObservedType().isEmpty());
203 }
204
205 static constexpr UnaryArithProfileBase observedIntBits()
206 {
207 constexpr ObservedType observedInt32 { ObservedType().withInt32() };
208 constexpr UnaryArithProfileBase bits = observedInt32.bits() << argObservedTypeShift;
209 return bits;
210 }
211 static constexpr UnaryArithProfileBase observedNumberBits()
212 {
213 constexpr ObservedType observedNumber { ObservedType().withNumber() };
214 constexpr UnaryArithProfileBase bits = observedNumber.bits() << argObservedTypeShift;
215 return bits;
216 }
217
218 constexpr ObservedType argObservedType() const { return ObservedType((m_bits >> argObservedTypeShift) & observedTypeMask); }
219 void setArgObservedType(ObservedType type)
220 {
221 UnaryArithProfileBase bits = m_bits;
222 bits &= clearArgObservedTypeBitMask;
223 bits |= type.bits() << argObservedTypeShift;
224 m_bits = bits;
225 ASSERT(argObservedType() == type);
226 }
227
228 void argSawInt32() { setArgObservedType(argObservedType().withInt32()); }
229 void argSawNumber() { setArgObservedType(argObservedType().withNumber()); }
230 void argSawNonNumber() { setArgObservedType(argObservedType().withNonNumber()); }
231
232 void observeArg(JSValue arg)
233 {
234 UnaryArithProfile newProfile = *this;
235 if (arg.isNumber()) {
236 if (arg.isInt32())
237 newProfile.argSawInt32();
238 else
239 newProfile.argSawNumber();
240 } else
241 newProfile.argSawNonNumber();
242
243 m_bits = newProfile.bits();
244 }
245
246 bool isObservedTypeEmpty()
247 {
248 return argObservedType().isEmpty();
249 }
250
251 friend class JSC::LLIntOffsetsExtractor;
252};

254/* This class stores the following components in 16 bits:
255 * - ObservedResults
256 * - ObservedType for right-hand-side
257 * - ObservedType for left-hand-side
258 * - a bit used by division to indicate whether a special fast path was taken
259 */
260using BinaryArithProfileBase = uint16_t;
261class BinaryArithProfile : public ArithProfile<BinaryArithProfileBase> {
262 static constexpr uint32_t rhsObservedTypeShift = ObservedResults::numBitsNeeded;
263 static constexpr uint32_t lhsObservedTypeShift = rhsObservedTypeShift + ObservedType::numBitsNeeded;
264
265 static_assert(ObservedType::numBitsNeeded == 3, "We make a hard assumption about that here.");
266 static constexpr BinaryArithProfileBase clearRhsObservedTypeBitMask = static_cast<BinaryArithProfileBase>(~(0b111 << rhsObservedTypeShift));
267 static constexpr BinaryArithProfileBase clearLhsObservedTypeBitMask = static_cast<BinaryArithProfileBase>(~(0b111 << lhsObservedTypeShift));
268
269 static constexpr BinaryArithProfileBase observedTypeMask = (1 << ObservedType::numBitsNeeded) - 1;
270
271public:
272 static constexpr BinaryArithProfileBase specialFastPathBit = 1 << (lhsObservedTypeShift + ObservedType::numBitsNeeded);
273 static_assert((lhsObservedTypeShift + ObservedType::numBitsNeeded + 1) <= sizeof(BinaryArithProfileBase) * 8, "Should fit in the underlying type.");
274 static_assert(!(specialFastPathBit & ~clearLhsObservedTypeBitMask), "These bits should not intersect.");
275 static_assert(specialFastPathBit & clearLhsObservedTypeBitMask, "These bits should intersect.");
276 static_assert(static_cast<unsigned>(specialFastPathBit) > static_cast<unsigned>(static_cast<BinaryArithProfileBase>(~clearLhsObservedTypeBitMask)), "These bits should not intersect and specialFastPathBit should be a higher bit.");
277
278 BinaryArithProfile()
279 : ArithProfile<BinaryArithProfileBase> ()
280 {
281 ASSERT(lhsObservedType().isEmpty());
282 ASSERT(rhsObservedType().isEmpty());
283 }
284
285 static constexpr BinaryArithProfileBase observedIntIntBits()
286 {
287 constexpr ObservedType observedInt32 { ObservedType().withInt32() };
288 constexpr BinaryArithProfileBase bits = (observedInt32.bits() << lhsObservedTypeShift) | (observedInt32.bits() << rhsObservedTypeShift);
289 return bits;
290 }
291 static constexpr BinaryArithProfileBase observedNumberIntBits()
292 {
293 constexpr ObservedType observedNumber { ObservedType().withNumber() };
294 constexpr ObservedType observedInt32 { ObservedType().withInt32() };
295 constexpr BinaryArithProfileBase bits = (observedNumber.bits() << lhsObservedTypeShift) | (observedInt32.bits() << rhsObservedTypeShift);
296 return bits;
297 }
298 static constexpr BinaryArithProfileBase observedIntNumberBits()
299 {
300 constexpr ObservedType observedNumber { ObservedType().withNumber() };
301 constexpr ObservedType observedInt32 { ObservedType().withInt32() };
302 constexpr BinaryArithProfileBase bits = (observedInt32.bits() << lhsObservedTypeShift) | (observedNumber.bits() << rhsObservedTypeShift);
303 return bits;
304 }
305 static constexpr BinaryArithProfileBase observedNumberNumberBits()
306 {
307 constexpr ObservedType observedNumber { ObservedType().withNumber() };
308 constexpr BinaryArithProfileBase bits = (observedNumber.bits() << lhsObservedTypeShift) | (observedNumber.bits() << rhsObservedTypeShift);
309 return bits;
310 }
311
312 constexpr ObservedType lhsObservedType() const { return ObservedType((m_bits >> lhsObservedTypeShift) & observedTypeMask); }
313 constexpr ObservedType rhsObservedType() const { return ObservedType((m_bits >> rhsObservedTypeShift) & observedTypeMask); }
314 void setLhsObservedType(ObservedType type)
315 {
316 BinaryArithProfileBase bits = m_bits;
317 bits &= clearLhsObservedTypeBitMask;
318 bits |= type.bits() << lhsObservedTypeShift;
319 m_bits = bits;
320 ASSERT(lhsObservedType() == type);
321 }
322
323 void setRhsObservedType(ObservedType type)
324 {
325 BinaryArithProfileBase bits = m_bits;
326 bits &= clearRhsObservedTypeBitMask;
327 bits |= type.bits() << rhsObservedTypeShift;
328 m_bits = bits;
329 ASSERT(rhsObservedType() == type);
330 }
331
332 bool tookSpecialFastPath() const { return m_bits & specialFastPathBit; }
333
334 void lhsSawInt32() { setLhsObservedType(lhsObservedType().withInt32()); }
335 void lhsSawNumber() { setLhsObservedType(lhsObservedType().withNumber()); }
336 void lhsSawNonNumber() { setLhsObservedType(lhsObservedType().withNonNumber()); }
337 void rhsSawInt32() { setRhsObservedType(rhsObservedType().withInt32()); }
338 void rhsSawNumber() { setRhsObservedType(rhsObservedType().withNumber()); }
339 void rhsSawNonNumber() { setRhsObservedType(rhsObservedType().withNonNumber()); }
340
341 void observeLHS(JSValue lhs)
342 {
343 BinaryArithProfile newProfile = *this;
344 if (lhs.isNumber()) {
345 if (lhs.isInt32())
346 newProfile.lhsSawInt32();
347 else
348 newProfile.lhsSawNumber();
349 } else
350 newProfile.lhsSawNonNumber();
351
352 m_bits = newProfile.bits();
353 }
354
355 void observeLHSAndRHS(JSValue lhs, JSValue rhs)
356 {
357 observeLHS(lhs);
358
359 BinaryArithProfile newProfile = *this;
360 if (rhs.isNumber()) {
361 if (rhs.isInt32())
362 newProfile.rhsSawInt32();
363 else
364 newProfile.rhsSawNumber();
365 } else
366 newProfile.rhsSawNonNumber();
367
368 m_bits = newProfile.bits();
369 }
370
371 bool isObservedTypeEmpty()
372 {
373 return lhsObservedType().isEmpty() && rhsObservedType().isEmpty();
374 }
375
376 friend class JSC::LLIntOffsetsExtractor;
377};
