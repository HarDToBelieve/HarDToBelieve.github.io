26#pragma once
27
28#include "InstructionStream.h"
29#include "Opcode.h"
30#include <limits.h>
31#include <wtf/FastBitVector.h>
32#include <wtf/Vector.h>
33
34namespace JSC {
35
36class BytecodeGraph;
37class CodeBlock;
38class UnlinkedCodeBlock;
39class UnlinkedCodeBlockGenerator;
40struct Instruction;
41
42DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(BytecodeBasicBlock);
43
44class BytecodeBasicBlock {
45 WTF_MAKE_FAST_ALLOCATED(BytecodeBasicBlock);
46 WTF_MAKE_NONCOPYABLE(BytecodeBasicBlock);
47 friend class BytecodeGraph;
48public:
49 using BasicBlockVector = Vector<BytecodeBasicBlock, 0, UnsafeVectorOverflow, 16, BytecodeBasicBlockMalloc>;
50 static_assert(maxOpcodeLength <= UINT8_MAX);
51 enum SpecialBlockType { EntryBlock, ExitBlock };
52 inline BytecodeBasicBlock(const InstructionStream::Ref&, unsigned blockIndex);
53 inline BytecodeBasicBlock(SpecialBlockType, unsigned blockIndex);
54 BytecodeBasicBlock(BytecodeBasicBlock&&) = default;
55
56
57 bool isEntryBlock() { return !m_leaderOffset && !m_totalLength; }
58 bool isExitBlock() { return m_leaderOffset == UINT_MAX && m_totalLength == UINT_MAX; }
59
60 unsigned leaderOffset() const { return m_leaderOffset; }
61 unsigned totalLength() const { return m_totalLength; }
62
63 const Vector<uint8_t>& delta() const { return m_delta; }
64 const Vector<unsigned>& successors() const { return m_successors; }
65
66 FastBitVector& in() { return m_in; }
67 FastBitVector& out() { return m_out; }
68
69 unsigned index() const { return m_index; }
70
71 explicit operator bool() const { return true; }
72
73private:
74 // Only called from BytecodeGraph.
75 static BasicBlockVector compute(CodeBlock*, const InstructionStream& instructions);
76 static BasicBlockVector compute(UnlinkedCodeBlockGenerator*, const InstructionStream& instructions);
77 template<typename Block> static BasicBlockVector computeImpl(Block* codeBlock, const InstructionStream& instructions);
78 void shrinkToFit();
79
80 void addSuccessor(BytecodeBasicBlock& block)
81 {
82 if (!m_successors.contains(block.index()))
83 m_successors.append(block.index());
84 }
85
86 inline void addLength(unsigned);
87
88 InstructionStream::Offset m_leaderOffset;
89 unsigned m_totalLength;
90 unsigned m_index;
91
92 Vector<uint8_t> m_delta;
93 Vector<unsigned> m_successors;
94
95 FastBitVector m_in;
96 FastBitVector m_out;
97};
98
99} // namespace JSC
