38class BytecodeGraph;

40struct Instruction;

42class BytecodeDumperBase {
43public:
44 virtual ~BytecodeDumperBase()
45 {
46 }
47
48 void printLocationAndOp(InstructionStream::Offset location, const char* op);
49
50 template<typename T>
51 void dumpOperand(const char* operandName, T operand, bool isFirst = false)
52 {
53 if (!isFirst)
54 m_out.print(", ");
55 m_out.print(operandName);
56 m_out.print(":");
57 dumpValue(operand);
58 }
59
60 void dumpValue(VirtualRegister);
61
62 template<typename Traits>
63 void dumpValue(GenericBoundLabel<Traits>);
64
65 template<typename T>
66 void dumpValue(T v) { m_out.print(v); }
67
68protected:
69 virtual CString registerName(VirtualRegister) const = 0;
70 virtual int outOfLineJumpOffset(InstructionStream::Offset) const = 0;
71
72 BytecodeDumperBase(PrintStream& out)
73 : m_out(out)
74 {
75 }
76
77 PrintStream& m_out;
78 InstructionStream::Offset m_currentLocation { 0 };
79};
80
81template<class Block>
82class BytecodeDumper : public BytecodeDumperBase {
83public:
84 static void dumpBytecode(Block*, PrintStream& out, const InstructionStream::Ref& it, const ICStatusMap& = ICStatusMap());
85
86 BytecodeDumper(Block* block, PrintStream& out)
87 : BytecodeDumperBase(out)
88 , m_block(block)
89 {
90 }
91
92 ~BytecodeDumper() override { }
93
94protected:
95 Block* block() const { return m_block; }
96
97 void dumpBytecode(const InstructionStream::Ref& it, const ICStatusMap&);
98
99 CString registerName(VirtualRegister) const override;
100 int outOfLineJumpOffset(InstructionStream::Offset offset) const override;
101
102private:
103 virtual CString constantName(VirtualRegister) const;
104
105 Block* m_block;
106};
107
108template<class Block>
109class CodeBlockBytecodeDumper final : public BytecodeDumper<Block> {
110public:
111 static void dumpBlock(Block*, const InstructionStream&, PrintStream& out, const ICStatusMap& = ICStatusMap());
112 static void dumpGraph(Block*, const InstructionStream&, BytecodeGraph&, PrintStream& out = WTF::dataFile(), const ICStatusMap& = ICStatusMap());
113
114 void dumpIdentifiers();
115 void dumpConstants();
116 void dumpExceptionHandlers();
117 void dumpSwitchJumpTables();
118 void dumpStringSwitchJumpTables();
119
120private:
121 using BytecodeDumper<Block>::BytecodeDumper;
122
123 ALWAYS_INLINE VM& vm() const;
124
125 const Identifier& identifier(int index) const;
126};
