35class BytecodeLivenessAnalysis;
36class CodeBlock;

38// Note: Full bytecode liveness does not track any information about the liveness of temps.
39// If you want tmp liveness for a checkpoint ask tmpLivenessForCheckpoint.
40class FullBytecodeLiveness {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 explicit FullBytecodeLiveness(size_t size)
44 : m_usesBefore(size)
45 , m_usesAfter(size)
46 { }
47
48 const FastBitVector& getLiveness(BytecodeIndex bytecodeIndex, LivenessCalculationPoint point) const
49 {
50 // We don't have to worry about overflowing into the next bytecodeoffset in our vectors because we
51 // static assert that bytecode length is greater than the number of checkpoints in BytecodeStructs.h
52 switch (point) {
53 case LivenessCalculationPoint::BeforeUse:
54 return m_usesBefore[toIndex(bytecodeIndex)];
55 case LivenessCalculationPoint::AfterUse:
56 return m_usesAfter[toIndex(bytecodeIndex)];
57 }
58 RELEASE_ASSERT_NOT_REACHED();
59 }
60
61 bool virtualRegisterIsLive(VirtualRegister reg, BytecodeIndex bytecodeIndex, LivenessCalculationPoint point) const
62 {
63 return virtualRegisterIsAlwaysLive(reg) || virtualRegisterThatIsNotAlwaysLiveIsLive(getLiveness(bytecodeIndex, point), reg);
64 }
65
66private:
67 friend class BytecodeLivenessAnalysis;
68
69 static size_t toIndex(BytecodeIndex bytecodeIndex) { return bytecodeIndex.offset() + bytecodeIndex.checkpoint(); }
70
71 // FIXME: Use FastBitVector's view mechanism to make them compact.
72 // https://bugs.webkit.org/show_bug.cgi?id=204427
73 FixedVector<FastBitVector> m_usesBefore;
74 FixedVector<FastBitVector> m_usesAfter;
75};
