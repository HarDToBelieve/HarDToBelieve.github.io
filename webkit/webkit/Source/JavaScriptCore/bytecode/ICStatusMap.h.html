26#pragma once
27
28#include "CodeOrigin.h"
29#include "ExitingInlineKind.h"
30#include <wtf/HashMap.h>
31
32namespace JSC {
33
34class CallLinkInfo;
35class CallLinkStatus;
36class CodeBlock;
37class GetByStatus;
38class InByStatus;
39class PutByIdStatus;
40class DeleteByStatus;
41class StructureStubInfo;
42struct ByValInfo;
43
44struct ICStatus {
45 StructureStubInfo* stubInfo { nullptr };
46 CallLinkInfo* callLinkInfo { nullptr };
47 ByValInfo* byValInfo { nullptr };
48 CallLinkStatus* callStatus { nullptr };
49 GetByStatus* getStatus { nullptr };
50 InByStatus* inStatus { nullptr };
51 PutByIdStatus* putStatus { nullptr };
52 DeleteByStatus* deleteStatus { nullptr };
53};
54
55typedef HashMap<CodeOrigin, ICStatus, CodeOriginApproximateHash> ICStatusMap;
56
57struct ICStatusContext {
58 ICStatus get(CodeOrigin) const;
59 bool isInlined(CodeOrigin) const;
60 ExitingInlineKind inlineKind(CodeOrigin) const;
61
62 InlineCallFrame* inlineCallFrame { nullptr };
63 CodeBlock* optimizedCodeBlock { nullptr };
64 ICStatusMap map;
65};
66
67typedef Vector<ICStatusContext*, 8> ICStatusContextStack;
68
69} // namespace JSC
