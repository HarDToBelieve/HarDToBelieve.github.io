36class FunctionRareData;

38template<typename Derived>
39class ObjectAllocationProfileBase {
40 friend class LLIntOffsetsExtractor;
41public:
42 static ptrdiff_t offsetOfAllocator() { return OBJECT_OFFSETOF(ObjectAllocationProfileBase, m_allocator); }
43 static ptrdiff_t offsetOfStructure() { return OBJECT_OFFSETOF(ObjectAllocationProfileBase, m_structure); }
44
45 ObjectAllocationProfileBase() = default;
46
47 bool isNull() { return !m_structure; }
48
49 void initializeProfile(VM&, JSGlobalObject*, JSCell* owner, JSObject* prototype, unsigned inferredInlineCapacity, JSFunction* constructor = nullptr, FunctionRareData* = nullptr);
50
51 Structure* structure()
52 {
53 Structure* structure = m_structure.get();
54 // Ensure that if we see the structure, it has been properly created
55 WTF::dependentLoadLoadFence();
56 return structure;
57 }
58
59protected:
60 void clear()
61 {
62 m_allocator = Allocator();
63 m_structure.clear();
64 ASSERT(isNull());
65 }
66
67 template<typename Visitor>
68 void visitAggregate(Visitor& visitor)
69 {
70 visitor.append(m_structure);
71 }
72
73private:
74 unsigned possibleDefaultPropertyCount(VM&, JSObject* prototype);
75
76 Allocator m_allocator; // Precomputed to make things easier for generated code.
77 WriteBarrier<Structure> m_structure;
78};
79
80class ObjectAllocationProfile : public ObjectAllocationProfileBase<ObjectAllocationProfile> {
81public:
82 using Base = ObjectAllocationProfileBase<ObjectAllocationProfile>;
83
84 ObjectAllocationProfile() = default;
85
86 using Base::clear;
87 using Base::visitAggregate;
88
89 void setPrototype(VM&, JSCell*, JSObject*) { }
90};
91
92class ObjectAllocationProfileWithPrototype : public ObjectAllocationProfileBase<ObjectAllocationProfileWithPrototype> {
93public:
94 using Base = ObjectAllocationProfileBase<ObjectAllocationProfileWithPrototype>;
95
96 static ptrdiff_t offsetOfPrototype() { return OBJECT_OFFSETOF(ObjectAllocationProfileWithPrototype, m_prototype); }
97
98 ObjectAllocationProfileWithPrototype() = default;
99
100 JSObject* prototype()
101 {
102 JSObject* prototype = m_prototype.get();
103 WTF::dependentLoadLoadFence();
104 return prototype;
105 }
106
107 void clear()
108 {
109 Base::clear();
110 m_prototype.clear();
111 ASSERT(isNull());
112 }
113
114 template<typename Visitor>
115 void visitAggregate(Visitor& visitor)
116 {
117 Base::visitAggregate(visitor);
118 visitor.append(m_prototype);
119 }
120
121 void setPrototype(VM& vm, JSCell* owner, JSObject* object)
122 {
123 m_prototype.set(vm, owner, object);
124 }
125
126private:
127 WriteBarrier<JSObject> m_prototype;
128};
