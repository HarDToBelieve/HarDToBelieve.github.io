39constexpr unsigned maxNumCheckpointTmps = 4;

41// A OperandKind::Tmp is one that exists for exiting to a checkpoint but does not exist between bytecodes.
42enum class OperandKind : uint32_t { Argument, Local, Tmp }; // Keep bit-width in sync with Operand::operandKindBits' definition.
43static constexpr OperandKind lastOperandKind = OperandKind::Tmp;
44
45class Operand {
46public:
47 static constexpr unsigned kindBits = WTF::getMSBSetConstexpr(static_cast<std::underlying_type_t<OperandKind>>(lastOperandKind)) + 1;
48 static constexpr unsigned maxBits = 32 + kindBits;
49 static_assert(maxBits == 34);
50
51 Operand() = default;
52 Operand(const Operand&) = default;
53
54 Operand(VirtualRegister operand)
55 : Operand(operand.isLocal() ? OperandKind::Local : OperandKind::Argument, operand.offset())
56 { }
57
58 Operand(OperandKind kind, int operand)
59#if CPU(LITTLE_ENDIAN)
60 : m_operand(operand)
61 , m_kind(kind)
62#else
63 : m_kind(kind)
64 , m_operand(operand)
65#endif
66 {
67 ASSERT(kind == OperandKind::Tmp || VirtualRegister(operand).isLocal() == (kind == OperandKind::Local));
68 }
69 static Operand tmp(uint32_t index) { return Operand(OperandKind::Tmp, index); }
70
71 OperandKind kind() const { return m_kind; }
72 int value() const { return m_operand; }
73 VirtualRegister virtualRegister() const
74 {
75 ASSERT(m_kind != OperandKind::Tmp);
76 return VirtualRegister(m_operand);
77 }
78 uint64_t asBits() const
79 {
80 uint64_t bits = bitwise_cast<uint64_t>(*this);
81 ASSERT(bits < (1ULL << maxBits));
82 return bits;
83 }
84 static Operand fromBits(uint64_t value);
85
86 bool isTmp() const { return kind() == OperandKind::Tmp; }
87 bool isArgument() const { return kind() == OperandKind::Argument; }
88 bool isLocal() const { return kind() == OperandKind::Local && virtualRegister().isLocal(); }
89 bool isHeader() const { return kind() != OperandKind::Tmp && virtualRegister().isHeader(); }
90 bool isConstant() const { return kind() != OperandKind::Tmp && virtualRegister().isConstant(); }
91
92 int toArgument() const { ASSERT(isArgument()); return virtualRegister().toArgument(); }
93 int toLocal() const { ASSERT(isLocal()); return virtualRegister().toLocal(); }
94
95 inline bool isValid() const;
96
97 inline bool operator==(const Operand&) const;
98
99 void dump(PrintStream&) const;
100
101private:
102#if CPU(LITTLE_ENDIAN)
103 int m_operand { VirtualRegister::invalidVirtualRegister };
104 OperandKind m_kind { OperandKind::Argument };
105#else
106 OperandKind m_kind { OperandKind::Argument };
107 int m_operand { VirtualRegister::invalidVirtualRegister };
108#endif
109};
110
111ALWAYS_INLINE bool Operand::operator==(const Operand& other) const
112{
113 if (kind() != other.kind())
114 return false;
115 if (isTmp())
116 return value() == other.value();
117 return virtualRegister() == other.virtualRegister();
118}
119
120inline bool Operand::isValid() const
121{
122 if (isTmp())
123 return value() >= 0;
124 return virtualRegister().isValid();
125}
126
127inline Operand Operand::fromBits(uint64_t value)
128{
129 Operand result = bitwise_cast<Operand>(value);
130 ASSERT(result.isValid());
131 return result;
132}

134static_assert(sizeof(Operand) == sizeof(uint64_t), "Operand::asBits() relies on this.");

138template<typename T, typename StorageArg = std::conditional_t<std::is_same_v<T, bool>, FastBitVector, Vector<T, 0, UnsafeVectorOverflow>>>
139class Operands {
140public:
141 template<typename, typename> friend class Operands;
142
143 using Storage = StorageArg;
144 using RefType = std::conditional_t<std::is_same_v<T, bool>, FastBitReference, T&>;
145 using ConstRefType = std::conditional_t<std::is_same_v<T, bool>, bool, const T&>;
146
147 Operands() = default;
148
149 explicit Operands(size_t numArguments, size_t numLocals, size_t numTmps)
150 : m_values(numArguments + numLocals + numTmps)
151 , m_numArguments(numArguments)
152 , m_numLocals(numLocals)
153 {
154 if (!WTF::VectorTraits<T>::needsInitialization)
155 m_values.fill(T());
156 }
157
158 explicit Operands(size_t numArguments, size_t numLocals, size_t numTmps, const T& initialValue)
159 : m_values(numArguments + numLocals + numTmps)
160 , m_numArguments(numArguments)
161 , m_numLocals(numLocals)
162 {
163 m_values.fill(initialValue);
164 }
165
166 template<typename U, typename V>
167 explicit Operands(OperandsLikeTag, const Operands<U, V>& other, const T& initialValue = T())
168 : m_values(other.size())
169 , m_numArguments(other.numberOfArguments())
170 , m_numLocals(other.numberOfLocals())
171 {
172 m_values.fill(initialValue);
173 }
174
175 template<typename U>
176 explicit Operands(const Operands<T, U>& other)
177 : m_values(other.m_values)
178 , m_numArguments(other.m_numArguments)
179 , m_numLocals(other.m_numLocals)
180 {
181 }
182
183 size_t numberOfArguments() const { return m_numArguments; }
184 size_t numberOfLocals() const { return m_numLocals; }
185 size_t numberOfTmps() const { return m_values.size() - numberOfArguments() - numberOfLocals(); }
186
187 size_t tmpIndex(size_t idx) const
188 {
189 ASSERT(idx < numberOfTmps());
190 return idx + numberOfArguments() + numberOfLocals();
191 }
192 size_t argumentIndex(size_t idx) const
193 {
194 ASSERT(idx < numberOfArguments());
195 return idx;
196 }
197
198 size_t localIndex(size_t idx) const
199 {
200 ASSERT(idx < numberOfLocals());
201 return numberOfArguments() + idx;
202 }
203
204 RefType tmp(size_t idx) { return m_values[tmpIndex(idx)]; }
205 ConstRefType tmp(size_t idx) const { return m_values[tmpIndex(idx)]; }
206
207 RefType argument(size_t idx) { return m_values[argumentIndex(idx)]; }
208 ConstRefType argument(size_t idx) const { return m_values[argumentIndex(idx)]; }
209
210 RefType local(size_t idx) { return m_values[localIndex(idx)]; }
211 ConstRefType local(size_t idx) const { return m_values[localIndex(idx)]; }
212
213 template<OperandKind operandKind>
214 size_t sizeFor() const
215 {
216 switch (operandKind) {
217 case OperandKind::Tmp:
218 return numberOfTmps();
219 case OperandKind::Argument:
220 return numberOfArguments();
221 case OperandKind::Local:
222 return numberOfLocals();
223 }
224 RELEASE_ASSERT_NOT_REACHED();
225 return 0;
226 }
227 template<OperandKind operandKind>
228 RefType atFor(size_t idx)
229 {
230 switch (operandKind) {
231 case OperandKind::Tmp:
232 return tmp(idx);
233 case OperandKind::Argument:
234 return argument(idx);
235 case OperandKind::Local:
236 return local(idx);
237 }
238 RELEASE_ASSERT_NOT_REACHED();
239 return tmp(0);
240 }
241 template<OperandKind operandKind>
242 ConstRefType atFor(size_t idx) const
243 {
244 switch (operandKind) {
245 case OperandKind::Tmp:
246 return tmp(idx);
247 case OperandKind::Argument:
248 return argument(idx);
249 case OperandKind::Local:
250 return local(idx);
251 }
252 RELEASE_ASSERT_NOT_REACHED();
253 return tmp(0);
254 }
255
256 void ensureLocals(size_t size, const T& ensuredValue = T())
257 {
258 if (size <= numberOfLocals())
259 return;
260
261 size_t newSize = numberOfArguments() + numberOfTmps() + size;
262 size_t oldNumLocals = numberOfLocals();
263 size_t oldNumTmps = numberOfTmps();
264 m_values.grow(newSize);
265 for (size_t i = 0; i < oldNumTmps; ++i)
266 m_values[newSize - 1 - i] = m_values[tmpIndex(oldNumTmps - 1 - i)];
267
268 m_numLocals = size;
269 if (ensuredValue != T() || !WTF::VectorTraits<T>::needsInitialization) {
270 for (size_t i = 0; i < size - oldNumLocals; ++i)
271 m_values[localIndex(oldNumLocals + i)] = ensuredValue;
272 }
273 }
274
275 void ensureTmps(size_t size, const T& ensuredValue = T())
276 {
277 if (size <= numberOfTmps())
278 return;
279
280 size_t oldSize = m_values.size();
281 size_t newSize = numberOfArguments() + numberOfLocals() + size;
282 m_values.grow(newSize);
283
284 if (ensuredValue != T() || !WTF::VectorTraits<T>::needsInitialization) {
285 for (size_t i = oldSize; i < newSize; ++i)
286 m_values[i] = ensuredValue;
287 }
288 }
289
290 void setLocal(size_t idx, const T& value)
291 {
292 ensureLocals(idx + 1);
293 local(idx) = value;
294 }
295
296 T getLocal(size_t idx)
297 {
298 return idx >= numberOfLocals() ? T() : local(idx);
299 }
300
301 void setArgumentFirstTime(size_t idx, const T& value)
302 {
303 ASSERT(m_values[idx] == T());
304 argument(idx) = value;
305 }
306
307 void setLocalFirstTime(size_t idx, const T& value)
308 {
309 ASSERT(idx >= numberOfLocals() || local(idx) == T());
310 setLocal(idx, value);
311 }

313 RefType getForOperandIndex(size_t index) { return m_values[index]; }

316 size_t operandIndex(VirtualRegister operand) const
317 {
318 if (operand.isArgument())
319 return argumentIndex(operand.toArgument());
320 return localIndex(operand.toLocal());
321 }
322
323 size_t operandIndex(Operand op) const
324 {
325 if (!op.isTmp())
326 return operandIndex(op.virtualRegister());
327 return tmpIndex(op.value());
328 }
329
330 RefType operand(VirtualRegister operand)
331 {
332 if (operand.isArgument())
333 return argument(operand.toArgument());
334 return local(operand.toLocal());
335 }
336
337 RefType operand(Operand op)
338 {
339 if (!op.isTmp())
340 return operand(op.virtualRegister());
341 return tmp(op.value());
342 }
343
344 ConstRefType operand(VirtualRegister operand) const { return const_cast<Operands*>(this)->operand(operand); }
345 ConstRefType operand(Operand operand) const { return const_cast<Operands*>(this)->operand(operand); }
346
347 bool hasOperand(VirtualRegister operand) const
348 {
349 if (operand.isArgument())
350 return true;
351 return static_cast<size_t>(operand.toLocal()) < numberOfLocals();
352 }
353 bool hasOperand(Operand op) const
354 {
355 if (op.isTmp()) {
356 ASSERT(op.value() >= 0);
357 return static_cast<size_t>(op.value()) < numberOfTmps();
358 }
359 return hasOperand(op.virtualRegister());
360 }
361
362 void setOperand(Operand operand, const T& value)
363 {
364 this->operand(operand) = value;
365 }
366
367 size_t size() const { return m_values.size(); }
368 ConstRefType at(size_t index) const { return m_values[index]; }
369 RefType at(size_t index) { return m_values[index]; }
370 ConstRefType operator[](size_t index) const { return at(index); }
371 RefType operator[](size_t index) { return at(index); }
372
373 Operand operandForIndex(size_t index) const
374 {
375 if (index < numberOfArguments())
376 return virtualRegisterForArgumentIncludingThis(index);
377 else if (index < numberOfLocals() + numberOfArguments())
378 return virtualRegisterForLocal(index - numberOfArguments());
379 return Operand::tmp(index - (numberOfLocals() + numberOfArguments()));
380 }
381
382 void fill(T value)
383 {
384 for (size_t i = 0; i < m_values.size(); ++i)
385 m_values[i] = value;
386 }
387
388 void clear()
389 {
390 fill(T());
391 }
392
393 bool operator==(const Operands& other) const
394 {
395 ASSERT(numberOfArguments() == other.numberOfArguments());
396 ASSERT(numberOfLocals() == other.numberOfLocals());
397 ASSERT(numberOfTmps() == other.numberOfTmps());
398
399 return m_values == other.m_values;
400 }
401
402 void dumpInContext(PrintStream& out, DumpContext* context) const;
403 void dump(PrintStream& out) const;
404
405private:
406 // The first m_numArguments of m_values are arguments, the next m_numLocals are locals, and the rest are tmps.
407 Storage m_values;
408 unsigned m_numArguments { 0 };
409 unsigned m_numLocals { 0 };
410};
