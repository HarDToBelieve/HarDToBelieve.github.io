187private:
188 ALWAYS_INLINE bool considerCaching(VM& vm, CodeBlock* codeBlock, Structure* structure, CacheableIdentifier impl)
189 {
190 DisallowGC disallowGC;
191
192 // We never cache non-cells.
193 if (!structure) {
194 sawNonCell = true;
195 return false;
196 }
197
198 // This method is called from the Optimize variants of IC slow paths. The first part of this
199 // method tries to determine if the Optimize variant should really behave like the
200 // non-Optimize variant and leave the IC untouched.
201 //
202 // If we determine that we should do something to the IC then the next order of business is
203 // to determine if this Structure would impact the IC at all. We know that it won't, if we
204 // have already buffered something on its behalf. That's what the m_bufferedStructures set is
205 // for.
206
207 everConsidered = true;
208 if (!countdown) {
209 // Check if we have been doing repatching too frequently. If so, then we should cool off
210 // for a while.
211 WTF::incrementWithSaturation(repatchCount);
212 if (repatchCount > Options::repatchCountForCoolDown()) {
213 // We've been repatching too much, so don't do it now.
214 repatchCount = 0;
215 // The amount of time we require for cool-down depends on the number of times we've
216 // had to cool down in the past. The relationship is exponential. The max value we
217 // allow here is 2^256 - 2, since the slow paths may increment the count to indicate
218 // that they'd like to temporarily skip patching just this once.
219 countdown = WTF::leftShiftWithSaturation(
220 static_cast<uint8_t>(Options::initialCoolDownCount()),
221 numberOfCoolDowns,
222 static_cast<uint8_t>(std::numeric_limits<uint8_t>::max() - 1));
223 WTF::incrementWithSaturation(numberOfCoolDowns);
224
225 // We may still have had something buffered. Trigger generation now.
226 bufferingCountdown = 0;
227 return true;
228 }
229
230 // We don't want to return false due to buffering indefinitely.
231 if (!bufferingCountdown) {
232 // Note that when this returns true, it's possible that we will not even get an
233 // AccessCase because this may cause Repatch.cpp to simply do an in-place
234 // repatching.
235 return true;
236 }
237
238 bufferingCountdown--;
239
240 // Now protect the IC buffering. We want to proceed only if this is a structure that
241 // we don't already have a case buffered for. Note that if this returns true but the
242 // bufferingCountdown is not zero then we will buffer the access case for later without
243 // immediately generating code for it.
244 //
245 // NOTE: This will behave oddly for InstanceOf if the user varies the prototype but not
246 // the base's structure. That seems unlikely for the canonical use of instanceof, where
247 // the prototype is fixed.
248 bool isNewlyAdded = false;
249 {
250 Locker locker { m_bufferedStructuresLock };
251 isNewlyAdded = m_bufferedStructures.add({ structure, impl }).isNewEntry;
252 }
253 if (isNewlyAdded)
254 vm.heap.writeBarrier(codeBlock);
255 return isNewlyAdded;
256 }
257 countdown--;
258 return false;
259 }

269 class BufferedStructure {

271 static constexpr uintptr_t hashTableDeletedValue = 0x2;
272 BufferedStructure() = default;
273 BufferedStructure(Structure* structure, CacheableIdentifier byValId)
274 : m_structure(structure)
275 , m_byValId(byValId)
276 { }
277 BufferedStructure(WTF::HashTableDeletedValueType)
278 : m_structure(bitwise_cast<Structure*>(hashTableDeletedValue))
279 { }
280
281 bool isHashTableDeletedValue() const { return bitwise_cast<uintptr_t>(m_structure) == hashTableDeletedValue; }
282
283 unsigned hash() const
284 {
285 unsigned hash = PtrHash<Structure*>::hash(m_structure);
286 if (m_byValId)
287 hash += m_byValId.hash();
288 return hash;
289 }
290
291 friend bool operator==(const BufferedStructure& a, const BufferedStructure& b)
292 {
293 return a.m_structure == b.m_structure && a.m_byValId == b.m_byValId;
294 }
295
296 friend bool operator!=(const BufferedStructure& a, const BufferedStructure& b)
297 {
298 return !(a == b);
299 }

411inline auto appropriateOptimizingGetByIdFunction(AccessType type) -> decltype(&operationGetByIdOptimize)
412{
413 switch (type) {
414 case AccessType::GetById:
415 return operationGetByIdOptimize;
416 case AccessType::TryGetById:
417 return operationTryGetByIdOptimize;
418 case AccessType::GetByIdDirect:
