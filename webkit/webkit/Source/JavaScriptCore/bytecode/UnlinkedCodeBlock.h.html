77struct UnlinkedStringJumpTable {
78 struct OffsetLocation {
79 int32_t m_branchOffset;
80 unsigned m_indexInTable;
81 };
82
83 using StringOffsetTable = MemoryCompactLookupOnlyRobinHoodHashMap<RefPtr<StringImpl>, OffsetLocation>;
84 StringOffsetTable m_offsetTable;
85
86 inline int32_t offsetForValue(StringImpl* value, int32_t defaultOffset) const
87 {
88 auto loc = m_offsetTable.find(value);
89 if (loc == m_offsetTable.end())
90 return defaultOffset;
91 return loc->value.m_branchOffset;
92 }
93
94 inline unsigned indexForValue(StringImpl* value, unsigned defaultIndex) const
95 {
96 auto loc = m_offsetTable.find(value);
97 if (loc == m_offsetTable.end())
98 return defaultIndex;
99 return loc->value.m_indexInTable;
100 }
101};

103struct UnlinkedSimpleJumpTable {
104 FixedVector<int32_t> m_branchOffsets;
105 int32_t m_min;
106
107 inline int32_t offsetForValue(int32_t value, int32_t defaultOffset) const
108 {
109 if (value >= m_min && static_cast<uint32_t>(value - m_min) < m_branchOffsets.size()) {
110 int32_t offset = m_branchOffsets[value - m_min];
111 if (offset)
112 return offset;
113 }
114 return defaultOffset;
115 }
116
117 void add(int32_t key, int32_t offset)
118 {
119 if (!m_branchOffsets[key])
120 m_branchOffsets[key] = offset;
121 }
122};
