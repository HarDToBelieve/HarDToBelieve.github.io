137 // Abstractly executes the given node. The new abstract state is stored into an
138 // abstract stack stored in *this. Loads of local variables (that span
139 // basic blocks) interrogate the basic block's notion of the state at the head.
140 // Stores to local variables are handled in endBasicBlock(). This returns true
141 // if execution should continue past this node. Notably, it will return true
142 // for block terminals, so long as those terminals are not Return or Unreachable.
143 //
144 // This is guaranteed to be equivalent to doing:
145 //
146 // state.startExecuting()
147 // state.executeEdges(node);
148 // result = state.executeEffects(index);
149 bool execute(unsigned indexInBlock);
150 bool execute(Node*);
151
152 // Indicate the start of execution of a node. It resets any state in the node
153 // that is progressively built up by executeEdges() and executeEffects().
154 void startExecuting();
155
156 // Abstractly execute the edges of the given node. This runs filterEdgeByUse()
157 // on all edges of the node. You can skip this step, if you have already used
158 // filterEdgeByUse() (or some equivalent) on each edge.
159 void executeEdges(Node*);
160
161 void executeKnownEdgeTypes(Node*);
162
163 ALWAYS_INLINE void filterEdgeByUse(Edge& edge)
164 {
165 UseKind useKind = edge.useKind();
166 if (useKind == UntypedUse)
167 return;
168 filterByType(edge, typeFilterFor(useKind));
169 }
170
171 // Abstractly execute the effects of the given node. This changes the abstract
