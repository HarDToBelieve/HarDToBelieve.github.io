26#pragma once
27
28#if ENABLE(DFG_JIT)
29
30#include "DFGEpoch.h"
31#include "DFGStructureClobberState.h"
32#include <wtf/PrintStream.h>
33
34namespace JSC { namespace DFG {
35
36class AbstractValueClobberEpoch {
37public:
38 AbstractValueClobberEpoch()
39 {
40 }
41
42 static AbstractValueClobberEpoch first(StructureClobberState clobberState)
43 {
44 AbstractValueClobberEpoch result;
45 result.m_value = Epoch::first().toUnsigned() << epochShift;
46 switch (clobberState) {
47 case StructuresAreWatched:
48 result.m_value |= watchedFlag;
49 break;
50 case StructuresAreClobbered:
51 break;
52 }
53 return result;
54 }
55
56 void clobber()
57 {
58 m_value += step;
59 m_value &= ~watchedFlag;
60 }
61
62 void observeInvalidationPoint()
63 {
64 m_value |= watchedFlag;
65 }
66
67 bool operator==(const AbstractValueClobberEpoch& other) const
68 {
69 return m_value == other.m_value;
70 }
71
72 bool operator!=(const AbstractValueClobberEpoch& other) const
73 {
74 return !(*this == other);
75 }
76
77 StructureClobberState structureClobberState() const
78 {
79 return m_value & watchedFlag ? StructuresAreWatched : StructuresAreClobbered;
80 }
81
82 Epoch clobberEpoch() const
83 {
84 return Epoch::fromUnsigned(m_value >> epochShift);
85 }
86
87 void dump(PrintStream&) const;
88
89private:
90 static constexpr unsigned step = 2;
91 static constexpr unsigned watchedFlag = 1;
92 static constexpr unsigned epochShift = 1;
93
94 unsigned m_value { 0 };
95};
96
97} } // namespace JSC::DFG

99#endif // ENABLE(DFG_JIT)
