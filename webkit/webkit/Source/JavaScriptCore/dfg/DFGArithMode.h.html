40// Arith::Mode describes the mode of an arithmetic operation that speculates integer.
41// Note that not all modes are valid for all operations.
42namespace Arith {
43enum Mode {
44 NotSet, // Arithmetic mode is either not relevant because we're using doubles anyway or we are at a phase in compilation where we don't know what we're doing, yet. Should never see this after FixupPhase except for nodes that take doubles as inputs already.
45 Unchecked, // Don't check anything and just do the direct hardware operation.
46 CheckOverflow, // Check for overflow but don't bother with negative zero.
47 CheckOverflowAndNegativeZero, // Check for both overflow and negative zero.
48 DoOverflow // Up-convert to the smallest type that soundly represents all possible results after input type speculation.
49};
50
51// Define the type of operation the rounding operation will perform.
52enum class RoundingMode {
53 Int32, // The round operation produces a integer and -0 is considered as 0.
54 Int32WithNegativeZeroCheck, // The round operation produces a integer and checks for -0.
55 Double // The round operation produce a double. The result can be -0, NaN or (+/-)Infinity.
56};
57
58enum class UnaryType : uint32_t {
59#define DFG_ARITH_UNARY_ENUM(capitalizedName, lowerName) capitalizedName,
60 FOR_EACH_ARITH_UNARY_OP(DFG_ARITH_UNARY_ENUM)
61#undef DFG_ARITH_UNARY_ENUM
62};
63
64using UnaryFunction = double(JIT_OPERATION_ATTRIBUTES*)(double);
65using UnaryOperation = double(JIT_OPERATION_ATTRIBUTES*)(JSGlobalObject*, EncodedJSValue);
66
67} // namespace Arith

69inline bool doesOverflow(Arith::Mode mode)
70{
71 switch (mode) {
72 case Arith::NotSet:
73 ASSERT_NOT_REACHED();
74#if !ASSERT_ENABLED
75 FALLTHROUGH;
76#endif
77 case Arith::Unchecked:
78 case Arith::CheckOverflow:
79 case Arith::CheckOverflowAndNegativeZero:
80 return false;
81 case Arith::DoOverflow:
82 return true;
83 }
84 ASSERT_NOT_REACHED();
85 return true;
86}
87
88// It's only valid to call this once you've determined that you don't need to *do*
89// overflow. For most nodes, that's implicit.
90inline bool shouldCheckOverflow(Arith::Mode mode)
91{
92 switch (mode) {
93 case Arith::NotSet:
94 case Arith::DoOverflow:
95 ASSERT_NOT_REACHED();
96 return true;
97 case Arith::Unchecked:
98 return false;
99 case Arith::CheckOverflow:
100 case Arith::CheckOverflowAndNegativeZero:
101 return true;
102 }
103 ASSERT_NOT_REACHED();
104 return true;
105}
106
107inline bool shouldCheckNegativeZero(Arith::Mode mode)
108{
109 switch (mode) {
110 case Arith::NotSet:
111 case Arith::DoOverflow:
112 ASSERT_NOT_REACHED();
113 return true;
114 case Arith::Unchecked:
115 case Arith::CheckOverflow:
116 return false;
117 case Arith::CheckOverflowAndNegativeZero:
118 return true;
119 }
120 ASSERT_NOT_REACHED();
121 return true;
122}
123
124inline bool subsumes(Arith::Mode earlier, Arith::Mode later)
125{
126 switch (earlier) {
127 case Arith::CheckOverflow:
128 switch (later) {
129 case Arith::Unchecked:
130 case Arith::CheckOverflow:
131 return true;
132 default:
133 return false;
134 }
135 case Arith::CheckOverflowAndNegativeZero:
136 switch (later) {
137 case Arith::Unchecked:
138 case Arith::CheckOverflow:
139 case Arith::CheckOverflowAndNegativeZero:
140 return true;
141 default:
142 return false;
143 }
144 default:
145 return earlier == later;
146 }
147}
148
149inline bool producesInteger(Arith::RoundingMode mode)
150{
151 return mode == Arith::RoundingMode::Int32WithNegativeZeroCheck || mode == Arith::RoundingMode::Int32;
152}
153
154inline bool shouldCheckNegativeZero(Arith::RoundingMode mode)
155{
156 return mode == Arith::RoundingMode::Int32WithNegativeZeroCheck;
157}
