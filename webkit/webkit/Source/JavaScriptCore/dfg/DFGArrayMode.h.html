43// Use a namespace + enum instead of enum alone to avoid the namespace collision
44// that would otherwise occur, since we say things like "Int8Array" and "JSArray"
45// in lots of other places, to mean subtly different things.
46namespace Array {
47enum Action : uint8_t {
48 Read,
49 Write
50};
51
52enum Type : uint8_t {
53 SelectUsingPredictions, // Implies that we need predictions to decide. We will never get to the backend in this mode.
54 SelectUsingArguments, // Implies that we use the Node's arguments to decide. We will never get to the backend in this mode.
55 Unprofiled, // Implies that array profiling didn't see anything. But that could be because the operands didn't comply with basic type assumptions (base is cell, property is int). This either becomes Generic or ForceExit depending on value profiling.
56 ForceExit, // Implies that we have no idea how to execute this operation, so we should just give up.
57 Generic,
58 String,
59
60 Undecided,
61 Int32,
62 Double,
63 Contiguous,
64 ArrayStorage,
65 SlowPutArrayStorage,
66
67 DirectArguments,
68 ScopedArguments,
69
70 Int8Array,
71 Int16Array,
72 Int32Array,
73 Uint8Array,
74 Uint8ClampedArray,
75 Uint16Array,
76 Uint32Array,
77 Float32Array,
78 Float64Array,
79 BigInt64Array,
80 BigUint64Array,
81 AnyTypedArray
82};
83
84enum Class : uint8_t {
85 NonArray, // Definitely some object that is not a JSArray.
86 OriginalNonArray, // Definitely some object that is not a JSArray, but that object has the original structure.
87 Array, // Definitely a JSArray, and may or may not have custom properties or have undergone some other bizarre transitions.
88 OriginalArray, // Definitely a JSArray, and still has one of the primordial JSArray structures for the global object that this code block (possibly inlined code block) belongs to.
89 OriginalCopyOnWriteArray, // Definitely a copy on write JSArray, and still has one of the primordial JSArray copy on write structures for the global object that this code block (possibly inlined code block) belongs to.
90 PossiblyArray // Some object that may or may not be a JSArray.
91};
92
93enum Speculation : uint8_t {
94 InBoundsSaneChain, // In bounds and the array prototype chain is still intact, i.e. loading a hole doesn't require special treatment.
95
96 InBounds, // In bounds and not loading a hole.
97 ToHole, // Potentially storing to a hole.
98 OutOfBoundsSaneChain, // Out-of-bounds access, but sane chain, so there are no arbitrary effects. E.g, loading out of bounds doesn't require traversing the prototype chain if we're an original array structure.
99 OutOfBounds // Out-of-bounds access and anything can happen.
100};
101enum Conversion : uint8_t {
102 AsIs,
103 Convert
104};
105} // namespace Array

193 ArrayMode withSpeculationFromProfile(const ConcurrentJSLocker& locker, ArrayProfile* profile, bool makeSafe) const
194 {
195 Array::Speculation mySpeculation;
196
197 if (makeSafe)
198 mySpeculation = Array::OutOfBounds;
199 else if (profile->mayStoreToHole(locker))
200 mySpeculation = Array::ToHole;
201 else
202 mySpeculation = Array::InBounds;
203
204 return withSpeculation(mySpeculation);
205 }
206
207 ArrayMode withProfile(const ConcurrentJSLocker& locker, ArrayProfile* profile, bool makeSafe) const
208 {
209 Array::Class myArrayClass;
210
211 if (isJSArray()) {
212 if (profile->usesOriginalArrayStructures(locker) && benefitsFromOriginalArray()) {
213 ArrayModes arrayModes = profile->observedArrayModes(locker);
214 if (hasSeenCopyOnWriteArray(arrayModes) && !hasSeenWritableArray(arrayModes))
215 myArrayClass = Array::OriginalCopyOnWriteArray;
216 else if (!hasSeenCopyOnWriteArray(arrayModes) && hasSeenWritableArray(arrayModes))
217 myArrayClass = Array::OriginalArray;
218 else
219 myArrayClass = Array::Array;
220 } else
221 myArrayClass = Array::Array;
222 } else
223 myArrayClass = arrayClass();
224
225 return withArrayClass(myArrayClass).withSpeculationFromProfile(locker, profile, makeSafe);
226 }

442 ArrayModes arrayModesThatPassFiltering() const
443 {
444 ArrayModes result;
445 switch (type()) {
446 case Array::Generic:
447 return ALL_ARRAY_MODES;
448 case Array::Undecided:
449 return arrayModesWithIndexingShapes(UndecidedShape);
450 case Array::Int32:
451 result = arrayModesWithIndexingShapes(Int32Shape);
452 break;
453 case Array::Double:
454 result = arrayModesWithIndexingShapes(DoubleShape);
455 break;
456 case Array::Contiguous:
457 result = arrayModesWithIndexingShapes(ContiguousShape);
458 break;
459 case Array::ArrayStorage:
460 return arrayModesWithIndexingShapes(ArrayStorageShape);
461 case Array::SlowPutArrayStorage:
462 return arrayModesWithIndexingShapes(SlowPutArrayStorageShape, ArrayStorageShape);
463 case Array::DirectArguments:
464 case Array::ScopedArguments:
465 return arrayModesWithIndexingShapes(ArrayStorageShape, SlowPutArrayStorageShape, NonArray);
466 case Array::Int8Array:
467 return Int8ArrayMode;
468 case Array::Int16Array:
469 return Int16ArrayMode;
470 case Array::Int32Array:
471 return Int32ArrayMode;
472 case Array::Uint8Array:
473 return Uint8ArrayMode;
474 case Array::Uint8ClampedArray:
475 return Uint8ClampedArrayMode;
476 case Array::Uint16Array:
477 return Uint16ArrayMode;
478 case Array::Uint32Array:
479 return Uint32ArrayMode;
480 case Array::Float32Array:
481 return Float32ArrayMode;
482 case Array::Float64Array:
483 return Float64ArrayMode;
484 case Array::BigInt64Array:
485 return BigInt64ArrayMode;
486 case Array::BigUint64Array:
487 return BigUint64ArrayMode;
488 case Array::AnyTypedArray:
489 return ALL_TYPED_ARRAY_MODES;
490 default:
491 return asArrayModesIgnoringTypedArrays(NonArray);
492 }
493
494 if (action() == Array::Write)
495 result &= ~ALL_COPY_ON_WRITE_ARRAY_MODES;
496 return result;
497 }

537 ArrayModes arrayModesWithIndexingShapes(IndexingType shape) const
538 {
539 switch (arrayClass()) {
540 case Array::NonArray:
541 case Array::OriginalNonArray:
542 return asArrayModesIgnoringTypedArrays(shape);
543 case Array::OriginalCopyOnWriteArray:
544 ASSERT(hasInt32(shape) || hasDouble(shape) || hasContiguous(shape));
545 return asArrayModesIgnoringTypedArrays(shape | IsArray) | asArrayModesIgnoringTypedArrays(shape | IsArray | CopyOnWrite);
546 case Array::Array:
547 if (hasInt32(shape) || hasDouble(shape) || hasContiguous(shape))
548 return asArrayModesIgnoringTypedArrays(shape | IsArray) | asArrayModesIgnoringTypedArrays(shape | IsArray | CopyOnWrite);
549 FALLTHROUGH;
550 case Array::OriginalArray:
551 return asArrayModesIgnoringTypedArrays(shape | IsArray);
552 case Array::PossiblyArray:
553 if (hasInt32(shape) || hasDouble(shape) || hasContiguous(shape))
554 return asArrayModesIgnoringTypedArrays(shape) | asArrayModesIgnoringTypedArrays(shape | IsArray) | asArrayModesIgnoringTypedArrays(shape | IsArray | CopyOnWrite);
555 return asArrayModesIgnoringTypedArrays(shape) | asArrayModesIgnoringTypedArrays(shape | IsArray);
556 }
557 // This is only necessary for C++ compilers that don't understand enums.
558 return 0;
559 }
560
561 template <typename... Args>
562 ArrayModes arrayModesWithIndexingShapes(IndexingType shape1, Args... args) const
563 {
564 ArrayModes arrayMode1 = arrayModesWithIndexingShapes(shape1);
565 ArrayModes arrayMode2 = arrayModesWithIndexingShapes(args...);
566 return arrayMode1 | arrayMode2;
567 }
