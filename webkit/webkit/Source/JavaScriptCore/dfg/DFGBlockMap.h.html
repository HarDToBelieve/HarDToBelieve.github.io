28#if ENABLE(DFG_JIT)

30#include "DFGBasicBlock.h"

32namespace JSC { namespace DFG {
33
34class Graph;
35
36template<typename T>
37class BlockMap {
38 WTF_MAKE_FAST_ALLOCATED;
39public:
40 BlockMap()
41 {
42 }
43
44 BlockMap(Graph&);
45
46 BlockIndex size() const
47 {
48 return m_vector.size();
49 }
50
51 T& atIndex(BlockIndex blockIndex)
52 {
53 return m_vector[blockIndex];
54 }
55
56 const T& atIndex(BlockIndex blockIndex) const
57 {
58 return m_vector[blockIndex];
59 }
60
61 T& at(BlockIndex blockIndex)
62 {
63 return m_vector[blockIndex];
64 }
65
66 const T& at(BlockIndex blockIndex) const
67 {
68 return m_vector[blockIndex];
69 }
70
71 T& at(BasicBlock* block)
72 {
73 return m_vector[block->index];
74 }
75
76 const T& at(BasicBlock* block) const
77 {
78 return m_vector[block->index];
79 }
80
81 T& operator[](BlockIndex blockIndex)
82 {
83 return m_vector[blockIndex];
84 }
85
86 const T& operator[](BlockIndex blockIndex) const
87 {
88 return m_vector[blockIndex];
89 }
90
91 T& operator[](BasicBlock* block)
92 {
93 return m_vector[block->index];
94 }
95
96 const T& operator[](BasicBlock* block) const
97 {
98 return m_vector[block->index];
99 }
100
101private:
102 Vector<T> m_vector;
103};
104
105} } // namespace JSC::DFG
106
107#endif // ENABLE(DFG_JIT)
