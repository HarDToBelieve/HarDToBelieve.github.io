26#pragma once
27
28#if ENABLE(DFG_JIT)
29
30#include "CallFrame.h"
31#include <wtf/ThreadSafeRefCounted.h>
32
33namespace JSC { namespace DFG {
34
35class CodeOriginPool : public ThreadSafeRefCounted<CodeOriginPool> {
36public:
37 static Ref<CodeOriginPool> create()
38 {
39 return adoptRef(*new CodeOriginPool);
40 }
41
42 CallSiteIndex addCodeOrigin(CodeOrigin);
43 CallSiteIndex addUniqueCallSiteIndex(CodeOrigin);
44 CallSiteIndex lastCallSite() const;
45 DisposableCallSiteIndex addDisposableCallSiteIndex(CodeOrigin);
46 void removeDisposableCallSiteIndex(DisposableCallSiteIndex);
47 void shrinkToFit();
48
49 CodeOrigin get(unsigned index) { return m_codeOrigins[index]; }
50 unsigned size() const { return m_codeOrigins.size(); }
51
52private:
53 Vector<CodeOrigin, 0, UnsafeVectorOverflow> m_codeOrigins;
54 Vector<unsigned> m_callSiteIndexFreeList;
55};
56
57} } // namespace JSC::DFG
58
59#endif // ENABLE(DFG_JIT)
