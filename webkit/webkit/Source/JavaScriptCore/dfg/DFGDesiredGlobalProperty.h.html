34namespace JSC {
35
36class JSGlobalObject;
37
38namespace DFG {
39
40class DesiredGlobalProperty {
41public:
42 DesiredGlobalProperty() = default;
43
44 DesiredGlobalProperty(JSGlobalObject* globalObject, unsigned identifierNumber)
45 : m_globalObject(globalObject)
46 , m_identifierNumber(identifierNumber)
47 {
48 }
49
50 DesiredGlobalProperty(WTF::HashTableDeletedValueType)
51 : m_globalObject(nullptr)
52 , m_identifierNumber(UINT_MAX)
53 {
54 }
55
56 JSGlobalObject* globalObject() const { return m_globalObject; }
57 unsigned identifierNumber() const { return m_identifierNumber; }
58
59 bool operator==(const DesiredGlobalProperty& other) const
60 {
61 return m_globalObject == other.m_globalObject && m_identifierNumber == other.m_identifierNumber;
62 }
63
64 bool operator!=(const DesiredGlobalProperty& other) const
65 {
66 return !(*this == other);
67 }
68
69 bool isHashTableDeletedValue() const
70 {
71 return !m_globalObject && m_identifierNumber == UINT_MAX;
72 }
73
74 unsigned hash() const
75 {
76 return WTF::PtrHash<JSGlobalObject*>::hash(m_globalObject) + m_identifierNumber * 7;
77 }
78
79 void dumpInContext(PrintStream& out, DumpContext*) const
80 {
81 out.print(m_identifierNumber, " for ", RawPointer(m_globalObject));
82 }
83
84 void dump(PrintStream& out) const
85 {
86 dumpInContext(out, nullptr);
87 }
88
89private:
90 JSGlobalObject* m_globalObject { nullptr };
91 unsigned m_identifierNumber { 0 };
92};
93
94struct DesiredGlobalPropertyHash {
95 static unsigned hash(const DesiredGlobalProperty& key) { return key.hash(); }
96 static bool equal(const DesiredGlobalProperty& a, const DesiredGlobalProperty& b) { return a == b; }
97 static constexpr bool safeToCompareToEmptyOrDeleted = true;
98};
99
100} } // namespace JSC::DFG
