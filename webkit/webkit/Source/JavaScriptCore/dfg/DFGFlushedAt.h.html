30#include "DFGFlushFormat.h"
31#include "VirtualRegister.h"
32
33namespace JSC { namespace DFG {
34
35class FlushedAt {
36public:
37 FlushedAt()
38 : m_format(DeadFlush)
39 {
40 }
41
42 explicit FlushedAt(FlushFormat format)
43 : m_format(format)
44 {
45 ASSERT(format == DeadFlush || format == ConflictingFlush);
46 }
47
48 FlushedAt(FlushFormat format, VirtualRegister virtualRegister)
49 : m_format(format)
50 , m_virtualRegister(virtualRegister)
51 {
52 if (format == DeadFlush)
53 ASSERT(!virtualRegister.isValid());
54 }
55
56 bool operator!() const { return m_format == DeadFlush; }
57
58 FlushFormat format() const { return m_format; }
59 VirtualRegister virtualRegister() const { return m_virtualRegister; }
60
61 bool operator==(const FlushedAt& other) const
62 {
63 return m_format == other.m_format
64 && m_virtualRegister == other.m_virtualRegister;
65 }
66
67 bool operator!=(const FlushedAt& other) const { return !(*this == other); }
68
69 FlushedAt merge(const FlushedAt& other) const
70 {
71 if (!*this)
72 return other;
73 if (!other)
74 return *this;
75 if (*this == other)
76 return *this;
77 return FlushedAt(ConflictingFlush);
78 }
79
80 void dump(PrintStream&) const;
81 void dumpInContext(PrintStream&, DumpContext*) const;
82
83private:
84 FlushFormat m_format;
85 VirtualRegister m_virtualRegister;
86};
87
88} } // namespace JSC::DFG
