78#define DFG_NODE_DO_TO_CHILDREN(graph, node, thingToDo) do { \
79 Node* _node = (node); \
80 if (_node->flags() & NodeHasVarArgs) { \
81 for (unsigned _childIdx = _node->firstChild(); \
82 _childIdx < _node->firstChild() + _node->numChildren(); \
83 _childIdx++) { \
84 if (!!(graph).m_varArgChildren[_childIdx]) \
85 thingToDo(_node, (graph).m_varArgChildren[_childIdx]); \
86 } \
87 } else { \
88 for (unsigned _edgeIndex = 0; _edgeIndex < AdjacencyList::Size; _edgeIndex++) { \
89 Edge& _edge = _node->children.child(_edgeIndex); \
90 if (!_edge) \
91 break; \
92 thingToDo(_node, _edge); \
93 } \
94 } \
95 } while (false)

98 if (!!(assertion)) \
99 break; \
100 (graph).logAssertionFailure( \
101 (node), __FILE__, __LINE__, WTF_PRETTY_FUNCTION, #assertion); \
102 CRASH_WITH_SECURITY_IMPLICATION_AND_INFO(__VA_ARGS__); \
103 } while (false)

285 AddSpeculationMode addSpeculationMode(Node* add, bool leftShouldSpeculateInt32, bool rightShouldSpeculateInt32, PredictionPass pass)
286 {
287 ASSERT(add->op() == ValueAdd || add->op() == ValueSub || add->op() == ArithAdd || add->op() == ArithSub);
288
289 RareCaseProfilingSource source = add->sourceFor(pass);
290
291 Node* left = add->child1().node();
292 Node* right = add->child2().node();
293
294 if (left->hasConstant())
295 return addImmediateShouldSpeculateInt32(add, rightShouldSpeculateInt32, right, left, source);
296 if (right->hasConstant())
297 return addImmediateShouldSpeculateInt32(add, leftShouldSpeculateInt32, left, right, source);
298
299 return (leftShouldSpeculateInt32 && rightShouldSpeculateInt32 && add->canSpeculateInt32(source)) ? SpeculateInt32 : DontSpeculateInt32;
300 }

333 bool addShouldSpeculateInt52(Node* add)
334 {
335 if (!enableInt52())
336 return false;
337
338 Node* left = add->child1().node();
339 Node* right = add->child2().node();
340
341 if (hasExitSite(add, Int52Overflow))
342 return false;
343
344 if (Node::shouldSpeculateInt52(left, right))
345 return true;
346
347 auto shouldSpeculateInt52ForAdd = [] (Node* node) {
348 // When DoubleConstant node appears, it means that users explicitly write a constant in their code with double form instead of integer form (1.0 instead of 1).
349 // In that case, we should honor this decision: using it as integer is not appropriate.
350 if (node->op() == DoubleConstant)
351 return false;
352 return isIntAnyFormat(node->prediction());
353 };
354
355 // Allow Int52 ArithAdd only when the one side of the binary operation should be speculated Int52. It is a bit conservative
356 // decision. This is because Double to Int52 conversion is not so cheap. Frequent back-and-forth conversions between Double and Int52
357 // rather hurt the performance. If the one side of the operation is already Int52, the cost for constructing ArithAdd becomes
358 // cheap since only one Double to Int52 conversion could be required.
359 // This recovers some regression in assorted tests while keeping kraken crypto improvements.
360 if (!left->shouldSpeculateInt52() && !right->shouldSpeculateInt52())
361 return false;
362
363 auto usesAsNumbers = [](Node* node) {
364 NodeFlags flags = node->flags() & NodeBytecodeBackPropMask;
365 if (!flags)
366 return false;
367 return (flags & (NodeBytecodeUsesAsNumber | NodeBytecodeNeedsNegZero | NodeBytecodeUsesAsInt | NodeBytecodeUsesAsArrayIndex)) == flags;
368 };
369
370 // Wrapping Int52 to Value is also not so cheap. Thus, we allow Int52 addition only when the node is used as number.
371 if (!usesAsNumbers(add))
372 return false;
373
374 return shouldSpeculateInt52ForAdd(left) && shouldSpeculateInt52ForAdd(right);
375 }

617 void voteChildren(Node* node, unsigned ballot, float weight = 1)
618 {
619 if (node->flags() & NodeHasVarArgs) {
620 for (unsigned childIdx = node->firstChild();
621 childIdx < node->firstChild() + node->numChildren();
622 childIdx++) {
623 if (!!m_varArgChildren[childIdx])
624 voteNode(m_varArgChildren[childIdx], ballot, weight);
625 }
626 return;
627 }
628
629 if (!node->child1())
630 return;
631 voteNode(node->child1(), ballot, weight);
632 if (!node->child2())
633 return;
634 voteNode(node->child2(), ballot, weight);
635 if (!node->child3())
636 return;
637 voteNode(node->child3(), ballot, weight);
638 }

640 template<typename T> // T = Node* or Edge
641 void substitute(BasicBlock& block, unsigned startIndexInBlock, T oldThing, T newThing)
642 {
643 for (unsigned indexInBlock = startIndexInBlock; indexInBlock < block.size(); ++indexInBlock) {
644 Node* node = block[indexInBlock];
645 if (node->flags() & NodeHasVarArgs) {
646 for (unsigned childIdx = node->firstChild(); childIdx < node->firstChild() + node->numChildren(); ++childIdx) {
647 if (!!m_varArgChildren[childIdx])
648 compareAndSwap(m_varArgChildren[childIdx], oldThing, newThing);
649 }
650 continue;
651 }
652 if (!node->child1())
653 continue;
654 compareAndSwap(node->children.child1(), oldThing, newThing);
655 if (!node->child2())
656 continue;
657 compareAndSwap(node->children.child2(), oldThing, newThing);
658 if (!node->child3())
659 continue;
660 compareAndSwap(node->children.child3(), oldThing, newThing);
661 }
662 }

864 // Quickly get all of the non-argument locals and tmps live at the given point. This doesn't give you
865 // any arguments because those are all presumed live. You can call forAllLiveInBytecode() to
866 // also get the arguments. This is much faster than calling isLiveInBytecode() for each local.
867 template<typename Functor>
868 void forAllLocalsAndTmpsLiveInBytecode(CodeOrigin codeOrigin, const Functor& functor)
869 {
870 // Support for not redundantly reporting arguments. Necessary because in case of a varargs
871 // call, only the callee knows that arguments are live while in the case of a non-varargs
872 // call, both callee and caller will see the variables live.
873 VirtualRegister exclusionStart;
874 VirtualRegister exclusionEnd;
875
876 CodeOrigin* codeOriginPtr = &codeOrigin;
877
878 bool isCallerOrigin = false;
879 for (;;) {
880 InlineCallFrame* inlineCallFrame = codeOriginPtr->inlineCallFrame();
881 VirtualRegister stackOffset(inlineCallFrame ? inlineCallFrame->stackOffset : 0);
882
883 if (inlineCallFrame) {
884 if (inlineCallFrame->isClosureCall)
885 functor(stackOffset + CallFrameSlot::callee);
886 if (inlineCallFrame->isVarargs())
887 functor(stackOffset + CallFrameSlot::argumentCountIncludingThis);
888 }
889
890 CodeBlock* codeBlock = baselineCodeBlockFor(inlineCallFrame);
891 FullBytecodeLiveness& fullLiveness = livenessFor(codeBlock);
892 const auto& livenessAtBytecode = fullLiveness.getLiveness(codeOriginPtr->bytecodeIndex(), appropriateLivenessCalculationPoint(*codeOriginPtr, isCallerOrigin));
893 for (unsigned relativeLocal = codeBlock->numCalleeLocals(); relativeLocal--;) {
894 VirtualRegister reg = stackOffset + virtualRegisterForLocal(relativeLocal);
895
896 // Don't report if our callee already reported.
897 if (reg >= exclusionStart && reg < exclusionEnd)
898 continue;
899
900 if (livenessAtBytecode[relativeLocal])
901 functor(reg);
902 }
903
904 if (codeOriginPtr->bytecodeIndex().checkpoint()) {
905 ASSERT(codeBlock->numTmps());
906 auto liveTmps = tmpLivenessForCheckpoint(*codeBlock, codeOriginPtr->bytecodeIndex());
907 liveTmps.forEachSetBit([&] (size_t tmp) {
908 functor(remapOperand(inlineCallFrame, Operand::tmp(tmp)));
909 });
910 }
911
912 if (!inlineCallFrame)
913 break;
914
915 // Arguments are always live. This would be redundant if it wasn't for our
916 // op_call_varargs inlining. See the comment above.
917 exclusionStart = stackOffset + CallFrame::argumentOffsetIncludingThis(0);
918 exclusionEnd = stackOffset + CallFrame::argumentOffsetIncludingThis(inlineCallFrame->m_argumentsWithFixup.size());
919
920 // We will always have a "this" argument and exclusionStart should be a smaller stack
921 // offset than exclusionEnd.
922 ASSERT(exclusionStart < exclusionEnd);
923
924 for (VirtualRegister reg = exclusionStart; reg < exclusionEnd; reg += 1)
925 functor(reg);
926
927 // We need to handle tail callers because we may decide to exit to the
928 // the return bytecode following the tail call.
929 codeOriginPtr = &inlineCallFrame->directCaller;
930 isCallerOrigin = true;
931 }
932 }

938 LivenessCalculationPoint appropriateLivenessCalculationPoint(CodeOrigin origin, bool isCallerOrigin)
939 {
940 if (isCallerOrigin) {
941 // We do not need to keep used registers of call bytecodes live when terminating in inlined function,
942 // except for inlining invoked by non call bytecodes including getter/setter calls.
943 BytecodeIndex bytecodeIndex = origin.bytecodeIndex();
944 InlineCallFrame* inlineCallFrame = origin.inlineCallFrame();
945 CodeBlock* codeBlock = baselineCodeBlockFor(inlineCallFrame);
946 auto instruction = codeBlock->instructions().at(bytecodeIndex.offset());
947 switch (instruction->opcodeID()) {
948 case op_call_varargs:
949 case op_tail_call_varargs:
950 case op_construct_varargs:
951 // When inlining varargs call, uses include array used for varargs. But when we are in inlined function,
952 // the content of this is already read and flushed to the stack. So, at this point, we no longer need to
953 // keep these use registers. We can use the liveness at LivenessCalculationPoint::AfterUse point.
954 // This is important to kill arguments allocations in DFG (not in FTL) when calling a function in a
955 // `func.apply(undefined, arguments)` manner.
956 return LivenessCalculationPoint::AfterUse;
957 default:
958 // We could list up the other bytecodes here, like, `op_call`, `op_get_by_id` (getter inlining). But we don't do that.
959 // To list up bytecodes here, we must ensure that these bytecodes never use `uses` registers after inlining. So we cannot
960 // return LivenessCalculationPoint::AfterUse blindly if isCallerOrigin = true. And since excluding liveness in the other
961 // bytecodes does not offer practical benefit, we do not try it.
962 break;
963 }
964 }
965 return LivenessCalculationPoint::BeforeUse;
966 }

969 // extension to forAllLocalsOrTmpsLiveInBytecode(), since all arguments are always presumed live.
970 template<typename Functor>
971 void forAllLiveInBytecode(CodeOrigin codeOrigin, const Functor& functor)
972 {
973 forAllLocalsAndTmpsLiveInBytecode(codeOrigin, functor);
974
975 // Report all arguments as being live.
976 for (unsigned argument = block(0)->variablesAtHead.numberOfArguments(); argument--;)
977 functor(virtualRegisterForArgumentIncludingThis(argument));
978 }

1213 AddSpeculationMode addImmediateShouldSpeculateInt32(Node* add, bool variableShouldSpeculateInt32, Node* operand, Node*immediate, RareCaseProfilingSource source)
1214 {
1215 ASSERT(immediate->hasConstant());
1216
1217 JSValue immediateValue = immediate->asJSValue();
1218 if (!immediateValue.isNumber() && !immediateValue.isBoolean())
1219 return DontSpeculateInt32;
1220
1221 if (!variableShouldSpeculateInt32)
1222 return DontSpeculateInt32;
1223
1224 // Integer constants can be typed Double if they are written like a double in the source code (e.g. 42.0).
1225 // In that case, we stay conservative unless the other operand was explicitly typed as integer.
1226 NodeFlags operandResultType = operand->result();
1227 if (operandResultType != NodeResultInt32 && immediateValue.isDouble())
1228 return DontSpeculateInt32;
1229
1230 if (immediateValue.isBoolean() || jsNumber(immediateValue.asNumber()).isInt32())
1231 return add->canSpeculateInt32(source) ? SpeculateInt32 : DontSpeculateInt32;
1232
1233 double doubleImmediate = immediateValue.asDouble();
1234 const double twoToThe48 = 281474976710656.0;
1235 if (doubleImmediate < -twoToThe48 || doubleImmediate > twoToThe48)
1236 return DontSpeculateInt32;
1237
1238 return bytecodeCanTruncateInteger(add->arithNodeFlags()) ? SpeculateInt32AndTruncateConstants : DontSpeculateInt32;
