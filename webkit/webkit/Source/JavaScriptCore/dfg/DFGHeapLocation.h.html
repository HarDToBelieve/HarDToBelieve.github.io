84class HeapLocation {
85public:
86 HeapLocation(
87 LocationKind kind = InvalidLocationKind,
88 AbstractHeap heap = AbstractHeap(),
89 Node* base = nullptr, LazyNode index = LazyNode(), Node* descriptor = nullptr)
90 : m_kind(kind)
91 , m_heap(heap)
92 , m_base(base)
93 , m_index(index)
94 , m_descriptor(descriptor)
95 {
96 ASSERT((kind == InvalidLocationKind) == !heap);
97 ASSERT(!!m_heap || !m_base);
98 ASSERT(m_base || (!m_index && !m_descriptor));
99 }
100
101 HeapLocation(LocationKind kind, AbstractHeap heap, Node* base, Node* index, Node* descriptor = nullptr)
102 : HeapLocation(kind, heap, base, LazyNode(index), descriptor)
103 {
104 }
105
106 HeapLocation(LocationKind kind, AbstractHeap heap, Edge base, Edge index = Edge(), Edge descriptor = Edge())
107 : HeapLocation(kind, heap, base.node(), index.node(), descriptor.node())
108 {
109 }
110
111 HeapLocation(WTF::HashTableDeletedValueType)
112 : m_kind(InvalidLocationKind)
113 , m_heap(WTF::HashTableDeletedValue)
114 , m_base(nullptr)
115 , m_index(nullptr)
116 , m_descriptor(nullptr)
117 {
118 }
119
120 bool operator!() const { return !m_heap; }
121
122 LocationKind kind() const { return m_kind; }
123 AbstractHeap heap() const { return m_heap; }
124 Node* base() const { return m_base; }
125 LazyNode index() const { return m_index; }
126
127 unsigned hash() const
128 {
129 return m_kind + m_heap.hash() + m_index.hash() + static_cast<unsigned>(bitwise_cast<uintptr_t>(m_base)) + static_cast<unsigned>(bitwise_cast<uintptr_t>(m_descriptor));
130 }
131
132 bool operator==(const HeapLocation& other) const
133 {
134 return m_kind == other.m_kind
135 && m_heap == other.m_heap
136 && m_base == other.m_base
137 && m_index == other.m_index
138 && m_descriptor == other.m_descriptor;
139 }
140
141 bool isHashTableDeletedValue() const
142 {
143 return m_heap.isHashTableDeletedValue();
144 }
145
146 void dump(PrintStream& out) const;
147
148private:
149 LocationKind m_kind;
150 AbstractHeap m_heap;
151 Node* m_base;
152 LazyNode m_index;
153 Node* m_descriptor;
154};
155
156struct HeapLocationHash {
157 static unsigned hash(const HeapLocation& key) { return key.hash(); }
158 static bool equal(const HeapLocation& a, const HeapLocation& b) { return a == b; }
159 static constexpr bool safeToCompareToEmptyOrDeleted = true;
160};

162LocationKind indexedPropertyLocForResultType(NodeFlags);
163
164inline LocationKind indexedPropertyLocForResultType(NodeFlags canonicalResultRepresentation)
165{
166 if (!canonicalResultRepresentation)
167 return IndexedPropertyJSLoc;
168
169 ASSERT((canonicalResultRepresentation & NodeResultMask) == canonicalResultRepresentation);
170 switch (canonicalResultRepresentation) {
171 case NodeResultDouble:
172 return IndexedPropertyDoubleLoc;
173 case NodeResultInt52:
174 return IndexedPropertyInt52Loc;
175 case NodeResultInt32:
176 return IndexedPropertyInt32Loc;
177 case NodeResultJS:
178 return IndexedPropertyJSLoc;
179 case NodeResultStorage:
180 RELEASE_ASSERT_NOT_REACHED();
181 default:
182 break;
183 }
184 RELEASE_ASSERT_NOT_REACHED();
185}
