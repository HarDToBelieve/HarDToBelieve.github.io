41namespace JSC {
42
43class TrackedReferences;
44
45namespace DFG {
46
47class JITCompiler;
48
49class JITCode final : public DirectJITCode {
50public:
51 JITCode();
52 ~JITCode() final;
53
54 CommonData* dfgCommon() final;
55 JITCode* dfg() final;
56
57 OSREntryData* osrEntryDataForBytecodeIndex(BytecodeIndex bytecodeIndex)
58 {
59 return tryBinarySearch<OSREntryData, BytecodeIndex>(
60 m_osrEntry, m_osrEntry.size(), bytecodeIndex,
61 getOSREntryDataBytecodeIndex);
62 }
63
64 void finalizeOSREntrypoints(Vector<DFG::OSREntryData>&&);
65
66 void reconstruct(
67 CodeBlock*, CodeOrigin, unsigned streamIndex, Operands<ValueRecovery>& result);
68
69 // This is only applicable if we're at a point where all values are spilled to the
70 // stack. Currently, it also has the restriction that the values must be in their
71 // bytecode-designated stack slots.
72 void reconstruct(
73 CallFrame*, CodeBlock*, CodeOrigin, unsigned streamIndex, Operands<std::optional<JSValue>>& result);
74
75#if ENABLE(FTL_JIT)
76 // NB. All of these methods take CodeBlock* because they may want to use
77 // CodeBlock's logic about scaling thresholds. It should be a DFG CodeBlock.
78
79 bool checkIfOptimizationThresholdReached(CodeBlock*);
80 void optimizeNextInvocation(CodeBlock*);
81 void dontOptimizeAnytimeSoon(CodeBlock*);
82 void optimizeAfterWarmUp(CodeBlock*);
83 void optimizeSoon(CodeBlock*);
84 void forceOptimizationSlowPathConcurrently(CodeBlock*);
85 void setOptimizationThresholdBasedOnCompilationResult(CodeBlock*, CompilationResult);
86#endif // ENABLE(FTL_JIT)
87
88 void validateReferences(const TrackedReferences&) final;
89
90 void shrinkToFit(const ConcurrentJSLocker&) final;
91
92 RegisterSet liveRegistersToPreserveAtExceptionHandlingCallSite(CodeBlock*, CallSiteIndex) final;
93#if ENABLE(FTL_JIT)
94 CodeBlock* osrEntryBlock() { return m_osrEntryBlock.get(); }
95 void setOSREntryBlock(VM&, const JSCell* owner, CodeBlock* osrEntryBlock);
96 void clearOSREntryBlockAndResetThresholds(CodeBlock* dfgCodeBlock);
97#endif
98
99 static ptrdiff_t commonDataOffset() { return OBJECT_OFFSETOF(JITCode, common); }
100
101 std::optional<CodeOrigin> findPC(CodeBlock*, void* pc) final;
102
103 using DirectJITCode::initializeCodeRefForDFG;
104
105private:
106 friend class JITCompiler; // Allow JITCompiler to call setCodeRef().
107
108public:
109 CommonData common;
110 FixedVector<DFG::OSREntryData> m_osrEntry;
111 FixedVector<DFG::OSRExit> m_osrExit;
112 FixedVector<DFG::SpeculationRecovery> m_speculationRecovery;
113 DFG::VariableEventStream variableEventStream;
114 DFG::MinifiedGraph minifiedDFG;
115
116#if ENABLE(FTL_JIT)
117 uint8_t neverExecutedEntry { 1 };
118
119 UpperTierExecutionCounter tierUpCounter;
120
121 // For osrEntryPoint that are in inner loop, this maps their bytecode to the bytecode
122 // of the outerloop entry points in order (from innermost to outermost).
123 //
124 // The key may not always be a target for OSR Entry but the list in the value is guaranteed
125 // to be usable for OSR Entry.
126 HashMap<BytecodeIndex, FixedVector<BytecodeIndex>> tierUpInLoopHierarchy;
127
128 // Map each bytecode of CheckTierUpAndOSREnter to its stream index.
129 HashMap<BytecodeIndex, unsigned> bytecodeIndexToStreamIndex;
130
131 enum class TriggerReason : uint8_t {
132 DontTrigger,
133 CompilationDone,
134 StartCompilation,
135 };
136
137 // Map each bytecode of CheckTierUpAndOSREnter to its trigger forcing OSR Entry.
138 // This can never be modified after it has been initialized since the addresses of the triggers
139 // are used by the JIT.
140 HashMap<BytecodeIndex, TriggerReason> tierUpEntryTriggers;
141
142 WriteBarrier<CodeBlock> m_osrEntryBlock;
143 unsigned osrEntryRetry;
144 bool abandonOSREntry;
145#endif // ENABLE(FTL_JIT)
146};
147
148} } // namespace JSC::DFG
149
150#endif // ENABLE(DFG_JIT)
