35namespace JSC {
36
37class TrackedReferences;
38
39namespace DFG {
40
41class MinifiedGraph {
42public:
43 MinifiedGraph() { }
44
45 MinifiedNode* at(MinifiedID id)
46 {
47 return tryBinarySearch<MinifiedNode, MinifiedID>(
48 m_list, m_list.size(), id, MinifiedNode::getID);
49 }
50
51 void append(const MinifiedNode& node)
52 {
53 m_list.append(node);
54 }
55
56 void prepareAndShrink();
57
58 void validateReferences(const TrackedReferences&);
59
60private:
61 Vector<MinifiedNode> m_list;
62};

64} } // namespace JSC::DFG
