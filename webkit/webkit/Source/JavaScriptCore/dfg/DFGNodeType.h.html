564// This enum generates a monotonically increasing id for all Node types,
565// and is used by the subsequent enum to fill out the id (as accessed via the NodeIdMask).
566enum NodeType {
567#define DFG_OP_ENUM(opcode, flags) opcode,
568 FOR_EACH_DFG_OP(DFG_OP_ENUM)
569#undef DFG_OP_ENUM
570 LastNodeType
571};
572
573#define DFG_OP_COUNT(opcode, flags) + 1
574constexpr unsigned numberOfNodeTypes = FOR_EACH_DFG_OP(DFG_OP_COUNT);
575#undef DFG_OP_COUNT
576
577// Specifies the default flags for each node.
578inline NodeFlags defaultFlags(NodeType op)
579{
580 switch (op) {
581#define DFG_OP_ENUM(opcode, flags) case opcode: return flags;
582 FOR_EACH_DFG_OP(DFG_OP_ENUM)
583#undef DFG_OP_ENUM
584 default:
585 RELEASE_ASSERT_NOT_REACHED();
586 return 0;
587 }
588}
