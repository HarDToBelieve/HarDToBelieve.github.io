35struct BasicBlock;
36struct Node;
37
38// Provides for the OSR exit profiling functionality that is common between the DFG
39// and the FTL.
40
41struct OSRExitBase {
42 OSRExitBase(ExitKind kind, CodeOrigin origin, CodeOrigin originForProfile, bool wasHoisted, uint32_t dfgNodeIndex)
43 : m_kind(kind)
44 , m_wasHoisted(wasHoisted)
45 , m_codeOrigin(origin)
46 , m_codeOriginForExitProfile(originForProfile)
47 , m_dfgNodeIndex(dfgNodeIndex)
48 {
49 ASSERT(m_codeOrigin.isSet());
50 ASSERT(m_codeOriginForExitProfile.isSet());
51 }
52
53 uint32_t m_count { 0 };
54 ExitKind m_kind;
55 bool m_wasHoisted;
56
57 CodeOrigin m_codeOrigin;
58 CodeOrigin m_codeOriginForExitProfile;
59 CallSiteIndex m_exceptionHandlerCallSiteIndex;
60 uint32_t m_dfgNodeIndex;
61
62 ALWAYS_INLINE bool isExceptionHandler() const
63 {
64 return m_kind == ExceptionCheck || m_kind == GenericUnwind;
65 }
66
67 // True if this exit is used as an exception handler for unwinding. This happens to only be set when
68 // isExceptionHandler is true, but all this actually means is that the OSR exit will assume that the
69 // machine state is as it would be coming out of genericUnwind.
70 ALWAYS_INLINE bool isGenericUnwindHandler() const
71 {
72 return m_kind == GenericUnwind;
73 }
74
75 ALWAYS_INLINE bool isExitingToCheckpointHandler() const
76 {
77 return m_codeOrigin.bytecodeIndex().checkpoint();
78 }
79
80protected:
81 void considerAddingAsFrequentExitSite(CodeBlock* profiledCodeBlock, ExitingJITType jitType)
82 {
83 if (m_count)
84 considerAddingAsFrequentExitSiteSlow(profiledCodeBlock, jitType);
85 }
86
87private:
88 void considerAddingAsFrequentExitSiteSlow(CodeBlock* profiledCodeBlock, ExitingJITType);
89};
90
91} } // namespace JSC::DFG
