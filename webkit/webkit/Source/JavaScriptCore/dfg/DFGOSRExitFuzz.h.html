27
28#include "Options.h"
29
30namespace JSC { namespace DFG {
31
32extern unsigned g_numberOfStaticOSRExitFuzzChecks;
33
34inline bool doOSRExitFuzzing()
35{
36 ASSERT(Options::useOSRExitFuzz());
37
38 g_numberOfStaticOSRExitFuzzChecks++;
39 if (unsigned atStatic = Options::fireOSRExitFuzzAtStatic())
40 return atStatic == g_numberOfStaticOSRExitFuzzChecks;
41
42 return true;
43}
44
45// DFG- and FTL-generated code will query this on every speculation.
46extern unsigned g_numberOfOSRExitFuzzChecks;
47
48} } // namespace JSC::DFG
