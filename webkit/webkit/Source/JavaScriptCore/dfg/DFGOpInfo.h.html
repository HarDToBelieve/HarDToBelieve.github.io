36namespace JSC { namespace DFG {

38// This type used in passing an immediate argument to Node constructor;
39// distinguishes an immediate value (typically an index into a CodeBlock data structure -
40// a constant index, argument, or identifier) from a Node*.
41struct OpInfo {
42 OpInfo() : m_value(0) { }
43 template<
44 typename IntegralType,
45 typename Constraint = typename std::enable_if<(std::is_integral<IntegralType>::value || std::is_enum<IntegralType>::value) && sizeof(IntegralType) <= sizeof(uint64_t)>::type>
46 explicit OpInfo(IntegralType value)
47 : m_value(static_cast<uint64_t>(value)) { }
48 explicit OpInfo(RegisteredStructure structure) : m_value(static_cast<uint64_t>(bitwise_cast<uintptr_t>(structure))) { }
49 explicit OpInfo(Operand op) : m_value(op.asBits()) { }
50 explicit OpInfo(CacheableIdentifier identifier) : m_value(static_cast<uint64_t>(identifier.rawBits())) { }
51 explicit OpInfo(ECMAMode ecmaMode) : m_value(ecmaMode.value()) { }
52 explicit OpInfo(PrivateFieldPutKind putKind) : m_value(putKind.value()) { }
53
54 template <typename T>
55 explicit OpInfo(T* ptr)
56 {
57 static_assert(!std::is_base_of<HeapCell, T>::value, "To make an OpInfo with a cell in it, the cell must be registered or frozen.");
58 static_assert(!std::is_base_of<StructureSet, T>::value, "To make an OpInfo with a structure set in, make sure to use RegisteredStructureSet.");
59 m_value = static_cast<uint64_t>(reinterpret_cast<uintptr_t>(ptr));
60 }
61
62 uint64_t m_value;
63};
64
65} } // namespace JSC::DFG
