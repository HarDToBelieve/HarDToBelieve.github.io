54class Plan final : public JITPlan {
55 using Base = JITPlan;
56
57public:
58 Plan(
59 CodeBlock* codeBlockToCompile, CodeBlock* profiledDFGCodeBlock,
60 JITCompilationMode, BytecodeIndex osrEntryBytecodeIndex,
61 const Operands<std::optional<JSValue>>& mustHandleValues);
62 ~Plan();
63
64 size_t codeSize() const final;
65 void finalizeInGC() final;
66 CompilationResult finalize() override;
67
68 void notifyReady() final;
69 void cancel() final;
70
71 bool isKnownToBeLiveAfterGC() final;
72 bool isKnownToBeLiveDuringGC(AbstractSlotVisitor&) final;
73 bool iterateCodeBlocksForGC(AbstractSlotVisitor&, const Function<void(CodeBlock*)>&) final;
74 bool checkLivenessAndVisitChildren(AbstractSlotVisitor&) final;
75
76
77 bool canTierUpAndOSREnter() const { return !m_tierUpAndOSREnterBytecodes.isEmpty(); }
78
79 void cleanMustHandleValuesIfNecessary();
80
81 BytecodeIndex osrEntryBytecodeIndex() const { return m_osrEntryBytecodeIndex; }
82 const Operands<std::optional<JSValue>>& mustHandleValues() const { return m_mustHandleValues; }
83 Profiler::Compilation* compilation() const { return m_compilation.get(); }
84
85 Finalizer* finalizer() const { return m_finalizer.get(); }
86 void setFinalizer(std::unique_ptr<Finalizer>&& finalizer) { m_finalizer = WTFMove(finalizer); }
87
88 RefPtr<InlineCallFrameSet> inlineCallFrames() const { return m_inlineCallFrames; }
89 DesiredWatchpoints& watchpoints() { return m_watchpoints; }
90 DesiredIdentifiers& identifiers() { return m_identifiers; }
91 DesiredWeakReferences& weakReferences() { return m_weakReferences; }
92 DesiredTransitions& transitions() { return m_transitions; }
93 RecordedStatuses& recordedStatuses() { return m_recordedStatuses; }
94
95 bool willTryToTierUp() const { return m_willTryToTierUp; }
96 void setWillTryToTierUp(bool willTryToTierUp) { m_willTryToTierUp = willTryToTierUp; }
97
98 HashMap<BytecodeIndex, FixedVector<BytecodeIndex>>& tierUpInLoopHierarchy() { return m_tierUpInLoopHierarchy; }
99 Vector<BytecodeIndex>& tierUpAndOSREnterBytecodes() { return m_tierUpAndOSREnterBytecodes; }
100
101 DeferredCompilationCallback* callback() const { return m_callback.get(); }
102 void setCallback(Ref<DeferredCompilationCallback>&& callback) { m_callback = WTFMove(callback); }
103
104private:
105 CompilationPath compileInThreadImpl() override;
106
107 bool isStillValidOnMainThread();
108 bool isStillValid();
109 void reallyAdd(CommonData*);
110
111 // These can be raw pointers because we visit them during every GC in checkLivenessAndVisitChildren.
112 CodeBlock* m_profiledDFGCodeBlock;
113
114 Operands<std::optional<JSValue>> m_mustHandleValues;
115 bool m_mustHandleValuesMayIncludeGarbage WTF_GUARDED_BY_LOCK(m_mustHandleValueCleaningLock) { true };
116 Lock m_mustHandleValueCleaningLock;
117
118 bool m_willTryToTierUp { false };
119
120 const BytecodeIndex m_osrEntryBytecodeIndex;
121
122 RefPtr<Profiler::Compilation> m_compilation;
123
124 std::unique_ptr<Finalizer> m_finalizer;
125
126 RefPtr<InlineCallFrameSet> m_inlineCallFrames;
127 DesiredWatchpoints m_watchpoints;
128 DesiredIdentifiers m_identifiers;
129 DesiredWeakReferences m_weakReferences;
130 DesiredTransitions m_transitions;
131 RecordedStatuses m_recordedStatuses;
132
133 HashMap<BytecodeIndex, FixedVector<BytecodeIndex>> m_tierUpInLoopHierarchy;
134 Vector<BytecodeIndex> m_tierUpAndOSREnterBytecodes;
135
136 RefPtr<DeferredCompilationCallback> m_callback;
137};
