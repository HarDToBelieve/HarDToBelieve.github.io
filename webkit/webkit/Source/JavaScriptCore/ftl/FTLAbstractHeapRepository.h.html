189// This class is meant to be cacheable between compilations, but it doesn't have to be.
190// Doing so saves on creation of nodes. But clearing it will save memory.
191
192class AbstractHeapRepository {
193 WTF_MAKE_NONCOPYABLE(AbstractHeapRepository);
194public:
195 AbstractHeapRepository();
196 ~AbstractHeapRepository();
197
198 AbstractHeap root;
199
200#define ABSTRACT_HEAP_DECLARATION(name) AbstractHeap name;
201 FOR_EACH_ABSTRACT_HEAP(ABSTRACT_HEAP_DECLARATION)
202#undef ABSTRACT_HEAP_DECLARATION
203
204#define ABSTRACT_FIELD_DECLARATION(name, offset) AbstractHeap name;
205 FOR_EACH_ABSTRACT_FIELD(ABSTRACT_FIELD_DECLARATION)
206#undef ABSTRACT_FIELD_DECLARATION
207
208 AbstractHeap& JSCell_freeListNext;
209 AbstractHeap& ArrayStorage_publicLength;
210 AbstractHeap& ArrayStorage_vectorLength;
211
212#define INDEXED_ABSTRACT_HEAP_DECLARATION(name, offset, size) IndexedAbstractHeap name;
213 FOR_EACH_INDEXED_ABSTRACT_HEAP(INDEXED_ABSTRACT_HEAP_DECLARATION)
214#undef INDEXED_ABSTRACT_HEAP_DECLARATION
215
216#define NUMBERED_ABSTRACT_HEAP_DECLARATION(name) NumberedAbstractHeap name;
217 FOR_EACH_NUMBERED_ABSTRACT_HEAP(NUMBERED_ABSTRACT_HEAP_DECLARATION)
218#undef NUMBERED_ABSTRACT_HEAP_DECLARATION
219
220 AbstractHeap& JSString_value;
221 AbstractHeap& JSWrapperObject_internalValue;
222
223 AbsoluteAbstractHeap absolute;
224
225 IndexedAbstractHeap* forIndexingType(IndexingType indexingType)
226 {
227 switch (indexingType) {
228 case ALL_BLANK_INDEXING_TYPES:
229 case ALL_UNDECIDED_INDEXING_TYPES:
230 return nullptr;
231
232 case ALL_INT32_INDEXING_TYPES:
233 return &indexedInt32Properties;
234
235 case ALL_DOUBLE_INDEXING_TYPES:
236 return &indexedDoubleProperties;
237
238 case ALL_CONTIGUOUS_INDEXING_TYPES:
239 return &indexedContiguousProperties;
240
241 case ALL_ARRAY_STORAGE_INDEXING_TYPES:
242 return &ArrayStorage_vector;
243
244 default:
245 RELEASE_ASSERT_NOT_REACHED();
246 return nullptr;
247 }
248 }
249
250 IndexedAbstractHeap& forArrayType(DFG::Array::Type type)
251 {
252 switch (type) {
253 case DFG::Array::Int32:
254 return indexedInt32Properties;
255 case DFG::Array::Double:
256 return indexedDoubleProperties;
257 case DFG::Array::Contiguous:
258 return indexedContiguousProperties;
259 case DFG::Array::ArrayStorage:
260 case DFG::Array::SlowPutArrayStorage:
261 return ArrayStorage_vector;
262 default:
263 RELEASE_ASSERT_NOT_REACHED();
264 return indexedInt32Properties;
265 }
266 }
