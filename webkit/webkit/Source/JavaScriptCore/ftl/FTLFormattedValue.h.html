33namespace JSC { namespace FTL {
34
35// This class is mostly used for OSR; it's a way of specifying how a value is formatted
36// in cases where it wouldn't have been obvious from looking at other indicators (like
37// the type of the B3::Value* or the type of the DFG::Node).
38
39class FormattedValue {
40public:
41 FormattedValue()
42 : m_format(DataFormatNone)
43 , m_value(nullptr)
44 {
45 }
46
47 FormattedValue(DataFormat format, LValue value)
48 : m_format(format)
49 , m_value(value)
50 {
51 }
52
53 bool operator!() const
54 {
55 ASSERT((m_format == DataFormatNone) == !m_value);
56 return m_format == DataFormatNone;
57 }
58
59 DataFormat format() const { return m_format; }
60 LValue value() const { return m_value; }
61
62private:
63 DataFormat m_format;
64 LValue m_value;
65};
66
67static inline FormattedValue noValue() { return FormattedValue(); }
68static inline FormattedValue int32Value(LValue value) { return FormattedValue(DataFormatInt32, value); }
69static inline FormattedValue booleanValue(LValue value) { return FormattedValue(DataFormatBoolean, value); }
70static inline FormattedValue jsValueValue(LValue value) { return FormattedValue(DataFormatJS, value); }
71static inline FormattedValue doubleValue(LValue value) { return FormattedValue(DataFormatDouble, value); }
72
73} } // namespace JSC::FTL

75#endif // ENABLE(FTL_JIT)
