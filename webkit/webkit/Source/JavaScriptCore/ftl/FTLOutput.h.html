421 template<typename VectorType>
422 void switchInstruction(LValue value, const VectorType& cases, LBasicBlock fallThrough, Weight fallThroughWeight)
423 {
424 B3::SwitchValue* switchValue = m_block->appendNew<B3::SwitchValue>(m_proc, origin(), value);
425 switchValue->setFallThrough(B3::FrequentedBlock(fallThrough));
426 for (const SwitchCase& switchCase : cases) {
427 int64_t value = switchCase.value()->asInt();
428 B3::FrequentedBlock target(switchCase.target(), switchCase.weight().frequencyClass());
429 switchValue->appendCase(B3::SwitchCase(value, target));
430 }
431 }

472template<typename... Params>
473inline LValue Output::phi(LType type, ValueFromBlock value, Params... theRest)
474{
475 LValue phiNode = phi(type);
476 addIncomingToPhi(phiNode, value, theRest...);
477 return phiNode;
478}
479
480template<typename VectorType>
481inline LValue Output::phi(LType type, const VectorType& vector)
482{
483 LValue phiNode = phi(type);
484 for (const ValueFromBlock& valueFromBlock : vector)
485 addIncomingToPhi(phiNode, valueFromBlock);
486 return phiNode;
487}

489template<typename... Params>
490inline void Output::addIncomingToPhi(LValue phi, ValueFromBlock value, Params... theRest)
491{
492 addIncomingToPhi(phi, value);
493 addIncomingToPhi(phi, theRest...);
494}
