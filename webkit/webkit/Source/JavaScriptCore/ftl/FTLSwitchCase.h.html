33namespace JSC { namespace FTL {
34
35class SwitchCase {
36public:
37 SwitchCase()
38 : m_value(nullptr)
39 , m_target(nullptr)
40 {
41 }
42
43 SwitchCase(LValue value, LBasicBlock target, Weight weight = Weight())
44 : m_value(value)
45 , m_target(target)
46 , m_weight(weight)
47 {
48 }
49
50 LValue value() const { return m_value; }
51 LBasicBlock target() const { return m_target; }
52 Weight weight() const { return m_weight; }
53
54private:
55 LValue m_value;
56 LBasicBlock m_target;
57 Weight m_weight;
58};
