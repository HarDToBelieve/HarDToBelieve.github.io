26#pragma once
27
28#if ENABLE(FTL_JIT)
29
30#include "FTLAbbreviatedTypes.h"
31#include "FTLAbstractHeap.h"
32
33namespace JSC { namespace FTL {
34
35class TypedPointer {
36public:
37 TypedPointer()
38 : m_heap(nullptr)
39 , m_value(nullptr)
40 {
41 }
42
43 TypedPointer(const AbstractHeap& heap, LValue value)
44 : m_heap(&heap)
45 , m_value(value)
46 {
47 }
48
49 explicit operator bool() const
50 {
51 ASSERT(!m_heap == !m_value);
52 return !!m_heap;
53 }
54
55 const AbstractHeap* heap() const { return m_heap; }
56 LValue value() const { return m_value; }
57
58private:
59 const AbstractHeap* m_heap;
60 LValue m_value;
61};
62
63} } // namespace JSC::FTL

65#endif // ENABLE(FTL_JIT)
