32namespace JSC { namespace FTL {
33
34class ValueFromBlock {
35public:
36 ValueFromBlock()
37 : m_value(nullptr)
38 , m_block(nullptr)
39 {
40 }
41
42 ValueFromBlock(LValue value, LBasicBlock block)
43 : m_value(value)
44 , m_block(block)
45 {
46 }
47
48 explicit operator bool() const { return m_value || m_block; }
49
50 LValue value() const { return m_value; }
51 LBasicBlock block() const { return m_block; }
52
53private:
54 LValue m_value;
55 LBasicBlock m_block;
56};
