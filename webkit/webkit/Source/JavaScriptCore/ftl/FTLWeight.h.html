26#pragma once
27
28#if ENABLE(FTL_JIT)
29
30#include "B3FrequencyClass.h"
31#include <wtf/MathExtras.h>
32#include <wtf/StdLibExtras.h>
33
34namespace JSC { namespace FTL {
35
36class Weight {
37public:
38 Weight()
39 : m_value(std::numeric_limits<float>::quiet_NaN())
40 {
41 }
42
43 explicit Weight(float value)
44 : m_value(value)
45 {
46 }
47
48 bool isSet() const { return m_value == m_value; }
49 bool operator!() const { return !isSet(); }
50
51 float value() const { return m_value; }
52
53 B3::FrequencyClass frequencyClass() const { return value() ? B3::FrequencyClass::Normal : B3::FrequencyClass::Rare; }
54
55 // Inverse weight for a two-target branch.
56 Weight inverse() const
57 {
58 if (!isSet())
59 return Weight();
60 if (value())
61 return Weight(0);
62 return Weight(1);
63 }
64
65private:
66 float m_value;
67};
68
69} } // namespace JSC::FTL

71#endif // ENABLE(FTL_JIT)
