33namespace JSC { namespace FTL {
34
35class WeightedTarget {
36public:
37 WeightedTarget()
38 : m_target(nullptr)
39 {
40 }
41
42 WeightedTarget(LBasicBlock target, Weight weight)
43 : m_target(target)
44 , m_weight(weight)
45 {
46 }
47
48 WeightedTarget(LBasicBlock target, float weight)
49 : m_target(target)
50 , m_weight(weight)
51 {
52 }
53
54 LBasicBlock target() const { return m_target; }
55 Weight weight() const { return m_weight; }
56
57 B3::FrequentedBlock frequentedBlock() const
58 {
59 return B3::FrequentedBlock(target(), weight().frequencyClass());
60 }
61
62private:
63 LBasicBlock m_target;
64 Weight m_weight;
65};
66
67// Helpers for creating weighted targets for statically known (or unknown) branch
68// profiles.
69
70inline WeightedTarget usually(LBasicBlock block)
71{
72 return WeightedTarget(block, 1);
73}
74
75inline WeightedTarget rarely(LBasicBlock block)
76{
77 return WeightedTarget(block, 0);
78}
79
80// Currently in B3 this is the equivalent of "usually", but we like to make the distinction in
81// case we ever make B3 support proper branch weights. We used to do that in LLVM.
82inline WeightedTarget unsure(LBasicBlock block)
83{
84 return WeightedTarget(block, Weight());
85}
