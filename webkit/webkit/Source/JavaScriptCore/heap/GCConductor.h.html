30// Either the mutator has the conn (https://en.wikipedia.org/wiki/Conn_(nautical)), meaning that the
31// mutator will incrementally drive the collector when it calls into slow paths; or the collector has the
32// conn, meaning that the collector thread will drive the collector.
33enum class GCConductor : uint8_t {
34 Mutator,
35 Collector
36};
37
38const char* gcConductorShortName(GCConductor officer);

40} // namespace JSC
