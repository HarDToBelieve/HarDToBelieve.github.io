71ALWAYS_INLINE bool Heap::isMarked(const void* rawCell)
72{
73 ASSERT(!m_isMarkingForGCVerifier);
74 HeapCell* cell = bitwise_cast<HeapCell*>(rawCell);
75 if (cell->isPreciseAllocation())
76 return cell->preciseAllocation().isMarked();
77 MarkedBlock& block = cell->markedBlock();
78 return block.isMarked(m_objectSpace.markingVersion(), cell);
79}
80
81ALWAYS_INLINE bool Heap::testAndSetMarked(HeapVersion markingVersion, const void* rawCell)
82{
83 HeapCell* cell = bitwise_cast<HeapCell*>(rawCell);
84 if (cell->isPreciseAllocation())
85 return cell->preciseAllocation().testAndSetMarked();
86 MarkedBlock& block = cell->markedBlock();
87 Dependency dependency = block.aboutToMark(markingVersion);
88 return block.testAndSetMarked(cell, dependency);
89}

96inline void Heap::writeBarrier(const JSCell* from, JSValue to)
97{
98#if ENABLE(WRITE_BARRIER_PROFILING)
99 WriteBarrierCounters::countWriteBarrier();
100#endif
101 if (!to.isCell())
102 return;
103 writeBarrier(from, to.asCell());
104}
105
106inline void Heap::writeBarrier(const JSCell* from, JSCell* to)
107{
108#if ENABLE(WRITE_BARRIER_PROFILING)
109 WriteBarrierCounters::countWriteBarrier();
110#endif
111 if (!from)
112 return;
113 if (!isWithinThreshold(from->cellState(), barrierThreshold()))
114 return;
115 if (LIKELY(!to))
116 return;
117 writeBarrierSlowPath(from);
118}
119
120inline void Heap::writeBarrier(const JSCell* from)
121{
122 ASSERT_GC_OBJECT_LOOKS_VALID(const_cast<JSCell*>(from));
123 if (!from)
124 return;
125 if (UNLIKELY(isWithinThreshold(from->cellState(), barrierThreshold())))
126 writeBarrierSlowPath(from);
127}

129inline void Heap::writeBarrierWithoutFence(const JSCell* from)
130{
131 ASSERT_GC_OBJECT_LOOKS_VALID(const_cast<JSCell*>(from));
132 if (!from)
133 return;
134 if (UNLIKELY(isWithinThreshold(from->cellState(), blackThreshold)))
135 addToRememberedSet(from);
136}

176inline void Heap::incrementDeferralDepth()
177{
178 ASSERT(!Thread::mayBeGCThread() || m_worldIsStopped);
179 m_deferralDepth++;
180}
181
182inline void Heap::decrementDeferralDepth()
183{
184 ASSERT(!Thread::mayBeGCThread() || m_worldIsStopped);
185 m_deferralDepth--;
186}
187
188inline void Heap::decrementDeferralDepthAndGCIfNeeded()
189{
190 ASSERT(!Thread::mayBeGCThread() || m_worldIsStopped);
191 m_deferralDepth--;
192
193 if (UNLIKELY(m_didDeferGCWork) || Options::forceDidDeferGCWork()) {
194 decrementDeferralDepthAndGCIfNeededSlow();
195
196 // Here are the possible relationships between m_deferralDepth and m_didDeferGCWork.
197 // Note that prior to the call to decrementDeferralDepthAndGCIfNeededSlow,
198 // m_didDeferGCWork had to have been true. Now it can be either false or true. There is
199 // nothing we can reliably assert.
200 //
201 // Possible arrangements of m_didDeferGCWork and !!m_deferralDepth:
202 //
203 // Both false: We popped out of all DeferGCs and we did whatever work was deferred.
204 //
205 // Only m_didDeferGCWork is true: We stopped for GC and the GC did DeferGC. This is
206 // possible because of how we handle the baseline JIT's worklist. It's also perfectly
207 // safe because it only protects reportExtraMemory. We can just ignore this.
208 //
209 // Only !!m_deferralDepth is true: m_didDeferGCWork had been set spuriously. It is only
210 // cleared by decrementDeferralDepthAndGCIfNeededSlow(). So, if we had deferred work but
211 // then decrementDeferralDepth()'d, then we might have the bit set even if we GC'd since
212 // then.
213 //
214 // Both true: We're in a recursive ~DeferGC. We wanted to do something about the
215 // deferred work, but were unable to.
216 }
217}

238inline void Heap::acquireAccess()
239{
240 if constexpr (validateDFGDoesGC)
241 verifyCanGC();
242
243 if (m_worldState.compareExchangeWeak(0, hasAccessBit))
244 return;
245 acquireAccessSlow();
246}
247
248inline bool Heap::hasAccess() const
249{
250 return m_worldState.loadRelaxed() & hasAccessBit;
251}
252
253inline void Heap::releaseAccess()
254{
255 if (m_worldState.compareExchangeWeak(hasAccessBit, 0))
256 return;
257 releaseAccessSlow();
258}
259
260inline bool Heap::mayNeedToStop()
261{
262 return m_worldState.loadRelaxed() != hasAccessBit;
263}
264
265inline void Heap::stopIfNecessary()
266{
267 if constexpr (validateDFGDoesGC)
268 verifyCanGC();
269
270 if (mayNeedToStop())
271 stopIfNecessarySlow();
272}
