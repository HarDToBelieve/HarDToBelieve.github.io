26#pragma once
27
28#include "GPRInfo.h"
29
30namespace JSC {
31
32struct EntryFrame;
33class CallFrame;
34class JSGlobalObject;
35class JSObject;
36class VM;
37
38struct VMEntryRecord {
39 /*
40 * This record stored in a vmEntryTo{JavaScript,Host} allocated frame. It is allocated on the stack
41 * after callee save registers where local variables would go.
42 */
43 VM* m_vm;
44 CallFrame* m_prevTopCallFrame;
45 EntryFrame* m_prevTopEntryFrame;
46 JSObject* m_callee;
47
48 JSObject* callee() const { return m_callee; }
49
50#if !ENABLE(C_LOOP) && NUMBER_OF_CALLEE_SAVES_REGISTERS > 0
51 CPURegister calleeSaveRegistersBuffer[NUMBER_OF_CALLEE_SAVES_REGISTERS];
52#elif ENABLE(C_LOOP)
53 CPURegister calleeSaveRegistersBuffer[1];
54#endif
55
56 CallFrame* prevTopCallFrame() { return m_prevTopCallFrame; }
57 SUPPRESS_ASAN CallFrame* unsafePrevTopCallFrame() { return m_prevTopCallFrame; }
58
59 EntryFrame* prevTopEntryFrame() { return m_prevTopEntryFrame; }
60 SUPPRESS_ASAN EntryFrame* unsafePrevTopEntryFrame() { return m_prevTopEntryFrame; }
61};
62
63extern "C" VMEntryRecord* vmEntryRecord(EntryFrame*);
64
65} // namespace JSC
