114static ALWAYS_INLINE void* performJITMemcpy(void *dst, const void *src, size_t n)
115{
116#if CPU(ARM64)
117 static constexpr size_t instructionSize = sizeof(unsigned);
118 RELEASE_ASSERT(roundUpToMultipleOf<instructionSize>(dst) == dst);
119 RELEASE_ASSERT(roundUpToMultipleOf<instructionSize>(src) == src);
120#endif
121 if (isJITPC(dst)) {
122 RELEASE_ASSERT(!Gigacage::contains(src));
123 RELEASE_ASSERT(reinterpret_cast<uint8_t*>(dst) + n <= endOfFixedExecutableMemoryPool());
124
125 if (UNLIKELY(Options::dumpJITMemoryPath()))
126 dumpJITMemory(dst, src, n);
127
128 if (g_jscConfig.useFastJITPermissions) {
129 threadSelfRestrictRWXToRW();
130 memcpy(dst, src, n);
131 threadSelfRestrictRWXToRX();
132 return dst;
133 }
134
135#if ENABLE(SEPARATED_WX_HEAP)
136 if (g_jscConfig.jitWriteSeparateHeaps) {
137 // Use execute-only write thunk for writes inside the JIT region. This is a variant of
138 // memcpy that takes an offset into the JIT region as its destination (first) parameter.
139 off_t offset = (off_t)((uintptr_t)dst - startOfFixedExecutableMemoryPool<uintptr_t>());
140 retagCodePtr<JITThunkPtrTag, CFunctionPtrTag>(g_jscConfig.jitWriteSeparateHeaps)(offset, src, n);
141 RELEASE_ASSERT(!Gigacage::contains(src));
142 return dst;
143 }
144#endif
145 }
146
147 // Use regular memcpy for writes outside the JIT region.
148 return memcpy(dst, src, n);
149}
