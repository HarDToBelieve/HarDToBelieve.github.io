46// Use this stub routine if you know that your code might be on stack when
47// either GC or other kinds of stub deletion happen. Basicaly, if your stub
48// routine makes calls (either to JS code or to C++ code) then you should
49// assume that it's possible for that JS or C++ code to do something that
50// causes the system to try to delete your routine. Using this routine type
51// ensures that the actual deletion is delayed until the GC proves that the
52// routine is no longer running. You can also subclass this routine if you
53// want to mark additional objects during GC in those cases where the
54// routine is known to be executing, or if you want to force this routine to
55// keep other routines alive (for example due to the use of a slow-path
56// list which does not get reclaimed all at once).
57class GCAwareJITStubRoutine : public JITStubRoutine {
58public:
59 GCAwareJITStubRoutine(const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&);
60 ~GCAwareJITStubRoutine() override;
61
62 static Ref<JITStubRoutine> create(VM& vm, const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code)
63 {
64 auto stub = adoptRef(*new GCAwareJITStubRoutine(code));
65 stub->makeGCAware(vm);
66 return stub;
67 }
68
69 template<typename Visitor>
70 void markRequiredObjects(Visitor& visitor)
71 {
72 markRequiredObjectsInternal(visitor);
73 }
74
75 void deleteFromGC();
76
77 void makeGCAware(VM&);
78
79protected:
80 void observeZeroRefCount() override;
81
82 virtual void markRequiredObjectsInternal(AbstractSlotVisitor&) { }
83 virtual void markRequiredObjectsInternal(SlotVisitor&) { }
84
85private:
86 friend class JITStubRoutineSet;
87
88 bool m_mayBeExecuting { false };
89 bool m_isJettisoned { false };
90 bool m_isGCAware { false };
91};

93class PolymorphicAccessJITStubRoutine : public GCAwareJITStubRoutine {
94public:
95 using Base = GCAwareJITStubRoutine;
96
97 PolymorphicAccessJITStubRoutine(const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, VM&, FixedVector<RefPtr<AccessCase>>&&, FixedVector<StructureID>&&);
98
99 const FixedVector<RefPtr<AccessCase>>& cases() const { return m_cases; }
100 const FixedVector<StructureID>& weakStructures() const { return m_weakStructures; }
101
102 unsigned hash() const
103 {
104 if (!m_hash)
105 m_hash = computeHash(m_cases, m_weakStructures);
106 return m_hash;
107 }
108
109 static unsigned computeHash(const FixedVector<RefPtr<AccessCase>>&, const FixedVector<StructureID>&);
110
111protected:
112 void observeZeroRefCount() override;
113
114private:
115 VM& m_vm;
116 FixedVector<RefPtr<AccessCase>> m_cases;
117 FixedVector<StructureID> m_weakStructures;
118};

162// Helper for easily creating a GC-aware JIT stub routine. For the varargs,
163// pass zero or more JSCell*'s. This will either create a JITStubRoutine, a
164// GCAwareJITStubRoutine, or an ObjectMarkingGCAwareJITStubRoutine as
165// appropriate. Generally you only need to pass pointers that will be used
166// after the first call to C++ or JS.
167//
168// Ref<PolymorphicAccessJITStubRoutine> createICJITStubRoutine(
169// const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code,
170// VM& vm,
171// FixedVector<RefPtr<AccessCase>>&& cases,
172// const JSCell* owner,
173// bool makesCalls,
174// ...);
175//
176// Note that we don't actually use C-style varargs because that leads to
177// strange type-related problems. For example it would preclude us from using
178// our custom of passing '0' as NULL pointer. Besides, when I did try to write
179// this function using varargs, I ended up with more code than this simple
180// way.
181
182Ref<PolymorphicAccessJITStubRoutine> createICJITStubRoutine(
183 const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, FixedVector<RefPtr<AccessCase>>&& cases, FixedVector<StructureID>&& weakStructures, VM&, const JSCell* owner, bool makesCalls,
184 const Vector<JSCell*>&, Bag<CallLinkInfo>&& callLinkInfos,
185 CodeBlock* codeBlockForExceptionHandlers, DisposableCallSiteIndex exceptionHandlingCallSiteIndex);
