26#pragma once
27
28#if ENABLE(JIT)
29
30#include "CCallHelpers.h"
31#include "JITMathICInlineResult.h"
32#include "SnippetOperand.h"
33
34namespace JSC {
35
36struct MathICGenerationState;
37
38class JITAddGenerator {
39public:
40 JITAddGenerator() { }
41
42 JITAddGenerator(SnippetOperand leftOperand, SnippetOperand rightOperand,
43 JSValueRegs result, JSValueRegs left, JSValueRegs right,
44 FPRReg leftFPR, FPRReg rightFPR, GPRReg scratchGPR, FPRReg scratchFPR)
45 : m_leftOperand(leftOperand)
46 , m_rightOperand(rightOperand)
47 , m_result(result)
48 , m_left(left)
49 , m_right(right)
50 , m_leftFPR(leftFPR)
51 , m_rightFPR(rightFPR)
52 , m_scratchGPR(scratchGPR)
53 , m_scratchFPR(scratchFPR)
54 {
55 ASSERT(!m_leftOperand.isConstInt32() || !m_rightOperand.isConstInt32());
56 }
57
58 JITMathICInlineResult generateInline(CCallHelpers&, MathICGenerationState&, const BinaryArithProfile*);
59 bool generateFastPath(CCallHelpers&, CCallHelpers::JumpList& endJumpList, CCallHelpers::JumpList& slowPathJumpList, const BinaryArithProfile*, bool shouldEmitProfiling);
60
61 static bool isLeftOperandValidConstant(SnippetOperand leftOperand) { return leftOperand.isPositiveConstInt32(); }
62 static bool isRightOperandValidConstant(SnippetOperand rightOperand) { return rightOperand.isPositiveConstInt32(); }
63
64private:
65 SnippetOperand m_leftOperand;
66 SnippetOperand m_rightOperand;
67 JSValueRegs m_result;
68 JSValueRegs m_left;
69 JSValueRegs m_right;
70 FPRReg m_leftFPR;
71 FPRReg m_rightFPR;
72 GPRReg m_scratchGPR;
73 FPRReg m_scratchFPR;
74};
75
76} // namespace JSC
77
78#endif // ENABLE(JIT)
