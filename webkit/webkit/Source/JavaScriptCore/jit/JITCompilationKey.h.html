28#include "JITCompilationMode.h"
29#include <wtf/HashMap.h>

31namespace JSC {
32
33class CodeBlock;
34class CodeBlockSet;
35
36class JITCompilationKey {
37public:
38 JITCompilationKey()
39 : m_profiledBlock(nullptr)
40 , m_mode(JITCompilationMode::InvalidCompilation)
41 {
42 }
43
44 JITCompilationKey(WTF::HashTableDeletedValueType)
45 : m_profiledBlock(nullptr)
46 , m_mode(JITCompilationMode::DFG)
47 {
48 }
49
50 JITCompilationKey(CodeBlock* profiledBlock, JITCompilationMode mode)
51 : m_profiledBlock(profiledBlock)
52 , m_mode(mode)
53 {
54 }
55
56 bool operator!() const
57 {
58 return !m_profiledBlock && m_mode == JITCompilationMode::InvalidCompilation;
59 }
60
61 bool isHashTableDeletedValue() const
62 {
63 return !m_profiledBlock && m_mode != JITCompilationMode::InvalidCompilation;
64 }
65
66 CodeBlock* profiledBlock() const { return m_profiledBlock; }
67 JITCompilationMode mode() const { return m_mode; }
68
69 bool operator==(const JITCompilationKey& other) const
70 {
71 return m_profiledBlock == other.m_profiledBlock
72 && m_mode == other.m_mode;
73 }
74
75 unsigned hash() const
76 {
77 return WTF::pairIntHash(WTF::PtrHash<CodeBlock*>::hash(m_profiledBlock), static_cast<std::underlying_type<JITCompilationMode>::type>(m_mode));
78 }
79
80 void dump(PrintStream&) const;
81
82private:
83 CodeBlock* m_profiledBlock;
84 JITCompilationMode m_mode;
85};
86
87struct JITCompilationKeyHash {
88 static unsigned hash(const JITCompilationKey& key) { return key.hash(); }
89 static bool equal(const JITCompilationKey& a, const JITCompilationKey& b) { return a == b; }
90 static constexpr bool safeToCompareToEmptyOrDeleted = true;
91};
92
93} // namespace JSC
