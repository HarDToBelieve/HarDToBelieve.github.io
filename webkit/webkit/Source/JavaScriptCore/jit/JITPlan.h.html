26#pragma once
27
28#if ENABLE(JIT)
29
30#include "CompilationResult.h"
31#include "JITCompilationKey.h"
32#include "JITCompilationMode.h"
33#include "JITPlanStage.h"
34#include "ReleaseHeapAccessScope.h"
35#include <wtf/MonotonicTime.h>
36#include <wtf/ThreadSafeRefCounted.h>
37
38namespace JSC {
39
40class AbstractSlotVisitor;
41class CodeBlock;
42class JITWorklistThread;
43class VM;
44
45class JITPlan : public ThreadSafeRefCounted<JITPlan> {
46protected:
47 JITPlan(JITCompilationMode, CodeBlock*);
48
49public:
50 virtual ~JITPlan() { }
51
52 VM* vm() const { return m_vm; }
53 CodeBlock* codeBlock() const { return m_codeBlock; }
54 JITWorklistThread* thread() const { return m_thread; }
55
56 JITCompilationMode mode() const { return m_mode; }
57
58 JITPlanStage stage() const { return m_stage; }
59 bool isFTL() const { return ::JSC::isFTL(m_mode); }
60
61 enum class Tier { Baseline = 0, DFG = 1, FTL = 2, Count = 3 };
62 Tier tier() const;
63
64 JITCompilationKey key();
65
66 void compileInThread(JITWorklistThread*);
67
68 virtual size_t codeSize() const = 0;
69
70 virtual CompilationResult finalize() = 0;
71
72 virtual void finalizeInGC() { }
73
74 void notifyCompiling();
75 virtual void notifyReady();
76 virtual void cancel();
77
78 virtual bool isKnownToBeLiveAfterGC();
79 virtual bool isKnownToBeLiveDuringGC(AbstractSlotVisitor&);
80 virtual bool iterateCodeBlocksForGC(AbstractSlotVisitor&, const Function<void(CodeBlock*)>&);
81 virtual bool checkLivenessAndVisitChildren(AbstractSlotVisitor&);
82
83protected:
84 bool computeCompileTimes() const;
85 bool reportCompileTimes() const;
86
87 enum CompilationPath { FailPath, BaselinePath, DFGPath, FTLPath, CancelPath };
88 virtual CompilationPath compileInThreadImpl() = 0;
89
90 JITPlanStage m_stage { JITPlanStage::Preparing };
91 JITCompilationMode m_mode;
92 MonotonicTime m_timeBeforeFTL;
93 VM* m_vm;
94 CodeBlock* m_codeBlock;
95 JITWorklistThread* m_thread { nullptr };
96};
97
98} // namespace JSC
99
100#endif // ENABLE(JIT)
