26#pragma once
27
28#if ENABLE(ASSEMBLER)
29
30#include "Reg.h"
31#include <wtf/PrintStream.h>
32
33namespace JSC {
34
35class RegisterAtOffset {
36public:
37 RegisterAtOffset()
38 : m_offset(0)
39 {
40 }
41
42 RegisterAtOffset(Reg reg, ptrdiff_t offset)
43 : m_reg(reg)
44 , m_offset(offset)
45 {
46 }
47
48 bool operator!() const { return !m_reg; }
49
50 Reg reg() const { return m_reg; }
51 ptrdiff_t offset() const { return m_offset; }
52 int offsetAsIndex() const { ASSERT(!(offset() % sizeof(CPURegister))); return offset() / static_cast<int>(sizeof(CPURegister)); }
53
54 bool operator==(const RegisterAtOffset& other) const
55 {
56 return reg() == other.reg() && offset() == other.offset();
57 }
58
59 bool operator<(const RegisterAtOffset& other) const
60 {
61 if (reg() != other.reg())
62 return reg() < other.reg();
63 return offset() < other.offset();
64 }
65
66 static Reg getReg(RegisterAtOffset* value) { return value->reg(); }
67
68 void dump(PrintStream& out) const;
69
70private:
71 Reg m_reg;
72 ptrdiff_t m_offset : (sizeof(ptrdiff_t) - sizeof(Reg)) * CHAR_BIT;
73};
74
75} // namespace JSC
76
77#endif // ENABLE(ASSEMBLER)
