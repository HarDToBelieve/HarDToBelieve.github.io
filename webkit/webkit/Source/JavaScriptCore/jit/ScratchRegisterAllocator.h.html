33namespace JSC {
34
35class AssemblyHelpers;
36struct ScratchBuffer;
37
38// This class provides a low-level register allocator for use in stubs.
39
40class ScratchRegisterAllocator {
41public:
42 ScratchRegisterAllocator() { }
43 ScratchRegisterAllocator(const RegisterSet& usedRegisters);
44 ~ScratchRegisterAllocator();
45
46 void lock(GPRReg);
47 void lock(FPRReg);
48 void lock(JSValueRegs);
49
50 template<typename BankInfo>
51 typename BankInfo::RegisterType allocateScratch();
52
53 GPRReg allocateScratchGPR();
54 FPRReg allocateScratchFPR();
55
56 bool didReuseRegisters() const
57 {
58 return !!m_numberOfReusedRegisters;
59 }
60
61 unsigned numberOfReusedRegisters() const
62 {
63 return m_numberOfReusedRegisters;
64 }
65
66 RegisterSet usedRegisters() const { return m_usedRegisters; }
67
68 enum class ExtraStackSpace { SpaceForCCall, NoExtraSpace };
69
70 struct PreservedState {
71 PreservedState()
72 : numberOfBytesPreserved(std::numeric_limits<unsigned>::max())
73 , extraStackSpaceRequirement(ExtraStackSpace::SpaceForCCall)
74 { }
75
76 PreservedState(unsigned numberOfBytes, ExtraStackSpace extraStackSpace)
77 : numberOfBytesPreserved(numberOfBytes)
78 , extraStackSpaceRequirement(extraStackSpace)
79 { }
80
81 explicit operator bool() const { return numberOfBytesPreserved != std::numeric_limits<unsigned>::max(); }
82
83 unsigned numberOfBytesPreserved;
84 ExtraStackSpace extraStackSpaceRequirement;
85 };
86
87 PreservedState preserveReusedRegistersByPushing(AssemblyHelpers& jit, ExtraStackSpace);
88 void restoreReusedRegistersByPopping(AssemblyHelpers& jit, const PreservedState&);
89
90 RegisterSet usedRegistersForCall() const;
91
92 unsigned desiredScratchBufferSizeForCall() const;
93
94 static unsigned preserveRegistersToStackForCall(AssemblyHelpers& jit, const RegisterSet& usedRegisters, unsigned extraPaddingInBytes);
95 static void restoreRegistersFromStackForCall(AssemblyHelpers& jit, const RegisterSet& usedRegisters, const RegisterSet& ignore, unsigned numberOfStackBytesUsedForRegisterPreservation, unsigned extraPaddingInBytes);
96
97private:
98 RegisterSet m_usedRegisters;
99 TempRegisterSet m_lockedRegisters;
100 TempRegisterSet m_scratchRegisters;
101 unsigned m_numberOfReusedRegisters;
102};
103
104} // namespace JSC
