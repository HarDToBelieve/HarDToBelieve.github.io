149template<PtrTag tag>
150ALWAYS_INLINE MacroAssemblerCodePtr<tag> getCodePtrImpl(const Opcode opcode, const void* opcodeAddress)
151{
152 void* opcodeValue = reinterpret_cast<void*>(opcode);
153 void* untaggedOpcode = untagAddressDiversifiedCodePtr<BytecodePtrTag>(opcodeValue, opcodeAddress);
154 void* retaggedOpcode = tagCodePtr<tag>(untaggedOpcode);
155 return MacroAssemblerCodePtr<tag>::createFromExecutableAddress(retaggedOpcode);
156}
157
158#if ENABLE(ARM64E) && !ENABLE(COMPUTED_GOTO_OPCODES)
159#error ENABLE(ARM64E) requires ENABLE(COMPUTED_GOTO_OPCODES) for getCodePtr (and its variants).
160#endif
161
162template<PtrTag tag>
163ALWAYS_INLINE MacroAssemblerCodePtr<tag> getCodePtr(OpcodeID opcodeID)
164{
165#if ENABLE(COMPUTED_GOTO_OPCODES)
166 const Opcode* opcode = getOpcodeAddress(opcodeID);
167 return getCodePtrImpl<tag>(*opcode, opcode);
168#else
169 return getCodePtrImpl<tag>(getOpcode(opcodeID), nullptr);
170#endif
171}
