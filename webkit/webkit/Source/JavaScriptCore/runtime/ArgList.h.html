31class MarkedArgumentBuffer : public RecordOverflow {
32 WTF_MAKE_NONCOPYABLE(MarkedArgumentBuffer);
33 WTF_MAKE_NONMOVABLE(MarkedArgumentBuffer);
34 WTF_FORBID_HEAP_ALLOCATION;
35 friend class VM;
36 friend class ArgList;
37
38public:
39 using Base = RecordOverflow;
40 static constexpr size_t inlineCapacity = 8;
41 typedef HashSet<MarkedArgumentBuffer*> ListSet;
42
43 // Constructor for a read-write list, to which you may append values.
44 // FIXME: Remove all clients of this API, then remove this API.
45 MarkedArgumentBuffer()
46 : m_size(0)
47 , m_capacity(inlineCapacity)
48 , m_buffer(m_inlineBuffer)
49 , m_markSet(nullptr)
50 {
51 }
52
53 ~MarkedArgumentBuffer()
54 {
55 ASSERT(!m_needsOverflowCheck);
56 if (m_markSet)
57 m_markSet->remove(this);
58
59 if (EncodedJSValue* base = mallocBase())
60 Gigacage::free(Gigacage::JSValue, base);
61 }
62
63 size_t size() const { return m_size; }
64 bool isEmpty() const { return !m_size; }
65
66 JSValue at(int i) const
67 {
68 if (i >= m_size)
69 return jsUndefined();
70
71 return JSValue::decode(slotFor(i));
72 }
73
74 void clear()
75 {
76 ASSERT(!m_needsOverflowCheck);
77 clearOverflow();
78 m_size = 0;
79 }
80
81 enum OverflowCheckAction {
82 CrashOnOverflow,
83 WillCheckLater
84 };
85 template<OverflowCheckAction action>
86 void appendWithAction(JSValue v)
87 {
88 ASSERT(m_size <= m_capacity);
89 if (m_size == m_capacity || mallocBase()) {
90 slowAppend(v);
91 if (action == CrashOnOverflow)
92 RELEASE_ASSERT(!hasOverflowed());
93 return;
94 }
95
96 slotFor(m_size) = JSValue::encode(v);
97 ++m_size;
98 }
99 void append(JSValue v) { appendWithAction<WillCheckLater>(v); }
100 void appendWithCrashOnOverflow(JSValue v) { appendWithAction<CrashOnOverflow>(v); }
101
102 void removeLast()
103 {
104 ASSERT(m_size);
105 m_size--;
106 }
107
108 JSValue last()
109 {
110 ASSERT(m_size);
111 return JSValue::decode(slotFor(m_size - 1));
112 }
113
114 JSValue takeLast()
115 {
116 JSValue result = last();
117 removeLast();
118 return result;
119 }
120
121 template<typename Visitor> static void markLists(Visitor&, ListSet&);
122
123 void ensureCapacity(size_t requestedCapacity)
124 {
125 if (requestedCapacity > static_cast<size_t>(m_capacity))
126 slowEnsureCapacity(requestedCapacity);
127 }
128
129 bool hasOverflowed()
130 {
131 clearNeedsOverflowCheck();
132 return Base::hasOverflowed();
133 }
134
135 void overflowCheckNotNeeded() { clearNeedsOverflowCheck(); }
136
137 template<typename Functor>
138 void fill(size_t count, const Functor& func)
139 {
140 ASSERT(!m_size);
141 ensureCapacity(count);
142 if (Base::hasOverflowed())
143 return;
144 m_size = count;
145 func(reinterpret_cast<JSValue*>(&slotFor(0)));
146 }
147
148private:
149 void expandCapacity();
150 void expandCapacity(int newCapacity);
151 void slowEnsureCapacity(size_t requestedCapacity);
152
153 void addMarkSet(JSValue);
154
155 JS_EXPORT_PRIVATE void slowAppend(JSValue);
156
157 EncodedJSValue& slotFor(int item) const
158 {
159 return m_buffer[item];
160 }
161
162 EncodedJSValue* mallocBase()
163 {
164 if (m_buffer == m_inlineBuffer)
165 return nullptr;
166 return &slotFor(0);
167 }
168
169#if ASSERT_ENABLED
170 void setNeedsOverflowCheck() { m_needsOverflowCheck = true; }
171 void clearNeedsOverflowCheck() { m_needsOverflowCheck = false; }
172
173 bool m_needsOverflowCheck { false };
174#else
175 void setNeedsOverflowCheck() { }
176 void clearNeedsOverflowCheck() { }
177#endif // ASSERT_ENABLED
178 int m_size;
179 int m_capacity;
180 EncodedJSValue m_inlineBuffer[inlineCapacity];
181 EncodedJSValue* m_buffer;
182 ListSet* m_markSet;
183};

185class ArgList {
186 WTF_MAKE_FAST_ALLOCATED;
187 friend class Interpreter;
188 friend class JIT;
189public:
190 ArgList()
191 : m_args(nullptr)
192 , m_argCount(0)
193 {
194 }
195
196 ArgList(CallFrame* callFrame)
197 : m_args(reinterpret_cast<JSValue*>(&callFrame[CallFrame::argumentOffset(0)]))
198 , m_argCount(callFrame->argumentCount())
199 {
200 }
201
202 ArgList(const MarkedArgumentBuffer& args)
203 : m_args(reinterpret_cast<JSValue*>(args.m_buffer))
204 , m_argCount(args.size())
205 {
206 }
207
208 JSValue at(int i) const
209 {
210 if (i >= m_argCount)
211 return jsUndefined();
212 return m_args[i];
213 }
214
215 bool isEmpty() const { return !m_argCount; }
216 size_t size() const { return m_argCount; }
217
218 JS_EXPORT_PRIVATE void getSlice(int startIndex, ArgList& result) const;
219
220private:
221 JSValue* data() const { return m_args; }
222
223 JSValue* m_args;
224 int m_argCount;
225};
