28namespace JSC {
29
30enum CodeSpecializationKind : uint8_t { CodeForCall, CodeForConstruct };
31
32inline CodeSpecializationKind specializationFromIsCall(bool isCall)
33{
34 return isCall ? CodeForCall : CodeForConstruct;
35}
36
37inline CodeSpecializationKind specializationFromIsConstruct(bool isConstruct)
38{
39 return isConstruct ? CodeForConstruct : CodeForCall;
40}
41
42} // namespace JSC
