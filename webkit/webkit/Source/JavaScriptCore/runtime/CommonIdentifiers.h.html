315 class CommonIdentifiers {
316 WTF_MAKE_NONCOPYABLE(CommonIdentifiers); WTF_MAKE_FAST_ALLOCATED;
317 private:
318 CommonIdentifiers(VM&);
319 ~CommonIdentifiers();
320 friend class VM;
321
322 public:
323 const BuiltinNames& builtinNames() const { return *m_builtinNames; }
324 const Identifier nullIdentifier;
325 const Identifier emptyIdentifier;
326 const Identifier underscoreProto;
327 const Identifier useStrictIdentifier;
328 const Identifier timesIdentifier;
329 private:
330 std::unique_ptr<BuiltinNames> m_builtinNames;
331
332 public:
333
334#define JSC_IDENTIFIER_DECLARE_PARSER_PRIVATE_NAME(name) const Identifier name##PrivateName;
335 JSC_PARSER_PRIVATE_NAMES(JSC_IDENTIFIER_DECLARE_PARSER_PRIVATE_NAME)
336#undef JSC_IDENTIFIER_DECLARE_PARSER_PRIVATE_NAME
337
338#define JSC_IDENTIFIER_DECLARE_KEYWORD_NAME_GLOBAL(name) const Identifier name##Keyword;
339 JSC_COMMON_IDENTIFIERS_EACH_KEYWORD(JSC_IDENTIFIER_DECLARE_KEYWORD_NAME_GLOBAL)
340#undef JSC_IDENTIFIER_DECLARE_KEYWORD_NAME_GLOBAL
341
342#define JSC_IDENTIFIER_DECLARE_PROPERTY_NAME_GLOBAL(name) const Identifier name;
343 JSC_COMMON_IDENTIFIERS_EACH_PROPERTY_NAME(JSC_IDENTIFIER_DECLARE_PROPERTY_NAME_GLOBAL)
344#undef JSC_IDENTIFIER_DECLARE_PROPERTY_NAME_GLOBAL
345
346#define JSC_IDENTIFIER_DECLARE_PRIVATE_WELL_KNOWN_SYMBOL_GLOBAL(name) const Identifier name##Symbol;
347 JSC_COMMON_PRIVATE_IDENTIFIERS_EACH_WELL_KNOWN_SYMBOL(JSC_IDENTIFIER_DECLARE_PRIVATE_WELL_KNOWN_SYMBOL_GLOBAL)
348#undef JSC_IDENTIFIER_DECLARE_PRIVATE_WELL_KNOWN_SYMBOL_GLOBAL
349 const Identifier intlLegacyConstructedSymbol;
350
351#define JSC_IDENTIFIER_DECLARE_PRIVATE_FIELD_GLOBAL(name) const Identifier name##PrivateField;
352 JSC_COMMON_IDENTIFIERS_EACH_PRIVATE_FIELD(JSC_IDENTIFIER_DECLARE_PRIVATE_FIELD_GLOBAL)
353#undef JSC_IDENTIFIER_DECLARE_PRIVATE_FIELD_GLOBAL
354
355 // Callers of this method should make sure that identifiers given to this method
356 // survive the lifetime of CommonIdentifiers and related VM.
357 JS_EXPORT_PRIVATE void appendExternalName(const Identifier& publicName, const Identifier& privateName);
358 };
