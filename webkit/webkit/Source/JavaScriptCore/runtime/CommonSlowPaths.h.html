42// The purpose of this namespace is to include slow paths that are shared
43// between the interpreter and baseline JIT. They are written to be agnostic
44// with respect to the slow-path calling convention, but they do rely on the
45// JS code being executed more-or-less directly from bytecode (so the call
46// frame layout is unmodified, making it potentially awkward to use these
47// from any optimizing JIT, like the DFG).

51ALWAYS_INLINE int numberOfExtraSlots(int argumentCountIncludingThis)
52{
53 int frameSize = argumentCountIncludingThis + CallFrame::headerSizeInRegisters;
54 int alignedFrameSize = WTF::roundUpToMultipleOf(stackAlignmentRegisters(), frameSize);
55 return alignedFrameSize - frameSize;
56}
57
58ALWAYS_INLINE int numberOfStackPaddingSlots(CodeBlock* codeBlock, int argumentCountIncludingThis)
59{
60 if (static_cast<unsigned>(argumentCountIncludingThis) >= codeBlock->numParameters())
61 return 0;
62 int alignedFrameSize = WTF::roundUpToMultipleOf(stackAlignmentRegisters(), argumentCountIncludingThis + CallFrame::headerSizeInRegisters);
63 int alignedFrameSizeForParameters = WTF::roundUpToMultipleOf(stackAlignmentRegisters(), codeBlock->numParameters() + CallFrame::headerSizeInRegisters);
64 return alignedFrameSizeForParameters - alignedFrameSize;
65}
66
67ALWAYS_INLINE int numberOfStackPaddingSlotsWithExtraSlots(CodeBlock* codeBlock, int argumentCountIncludingThis)
68{
69 if (static_cast<unsigned>(argumentCountIncludingThis) >= codeBlock->numParameters())
70 return 0;
71 return numberOfStackPaddingSlots(codeBlock, argumentCountIncludingThis) + numberOfExtraSlots(argumentCountIncludingThis);
72}
73
74ALWAYS_INLINE CodeBlock* codeBlockFromCallFrameCallee(CallFrame* callFrame, CodeSpecializationKind kind)
75{
76 JSFunction* callee = jsCast<JSFunction*>(callFrame->jsCallee());
77 ASSERT(!callee->isHostFunction());
78 return callee->jsExecutable()->codeBlockFor(kind);
79}
80
81ALWAYS_INLINE int arityCheckFor(VM& vm, CallFrame* callFrame, CodeSpecializationKind kind)
82{
83 CodeBlock* newCodeBlock = codeBlockFromCallFrameCallee(callFrame, kind);
84 ASSERT(callFrame->argumentCountIncludingThis() < static_cast<unsigned>(newCodeBlock->numParameters()));
85 int padding = numberOfStackPaddingSlotsWithExtraSlots(newCodeBlock, callFrame->argumentCountIncludingThis());
86
87 Register* newStack = callFrame->registers() - WTF::roundUpToMultipleOf(stackAlignmentRegisters(), padding);
88
89 if (UNLIKELY(!vm.ensureStackCapacityFor(newStack)))
90 return -1;
91 return padding;
92}

94inline bool opInByVal(JSGlobalObject* globalObject, JSValue baseVal, JSValue propName, ArrayProfile* arrayProfile = nullptr)
95{
96 VM& vm = getVM(globalObject);
97 auto scope = DECLARE_THROW_SCOPE(vm);
98 if (!baseVal.isObject()) {
99 throwException(globalObject, scope, createInvalidInParameterError(globalObject, baseVal));
100 return false;
101 }
102
103 JSObject* baseObj = asObject(baseVal);
104 if (arrayProfile)
105 arrayProfile->observeStructure(baseObj->structure(vm));
106
107 uint32_t i;
108 if (propName.getUInt32(i)) {
109 if (arrayProfile)
110 arrayProfile->observeIndexedRead(vm, baseObj, i);
111 RELEASE_AND_RETURN(scope, baseObj->hasProperty(globalObject, i));
112 }
113
114 auto property = propName.toPropertyKey(globalObject);
115 RETURN_IF_EXCEPTION(scope, false);
116 RELEASE_AND_RETURN(scope, baseObj->hasProperty(globalObject, property));
117}

119inline bool canAccessArgumentIndexQuickly(JSObject& object, uint32_t index)
120{
121 switch (object.type()) {
122 case DirectArgumentsType: {
123 DirectArguments* directArguments = jsCast<DirectArguments*>(&object);
124 if (directArguments->isMappedArgumentInDFG(index))
125 return true;
126 break;
127 }
128 case ScopedArgumentsType: {
129 ScopedArguments* scopedArguments = jsCast<ScopedArguments*>(&object);
130 if (scopedArguments->isMappedArgumentInDFG(index))
131 return true;
132 break;
133 }
134 default:
135 break;
136 }
137 return false;
138}

140ALWAYS_INLINE Structure* originalStructureBeforePut(VM& vm, JSValue value)
141{
142 if (!value.isCell())
143 return nullptr;
144 if (value.asCell()->type() == PureForwardingProxyType)
145 return jsCast<JSProxy*>(value)->target()->structure(vm);
146 return value.asCell()->structure(vm);
147}

150static ALWAYS_INLINE void putDirectWithReify(VM& vm, JSGlobalObject* globalObject, JSObject* baseObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot, Structure** result = nullptr)
151{
152 auto scope = DECLARE_THROW_SCOPE(vm);
153 if (baseObject->inherits<JSFunction>(vm)) {
154 jsCast<JSFunction*>(baseObject)->reifyLazyPropertyIfNeeded(vm, globalObject, propertyName);
155 RETURN_IF_EXCEPTION(scope, void());
156 }
157 if (result)
158 *result = originalStructureBeforePut(vm, baseObject);
159 scope.release();
160 baseObject->putDirect(vm, propertyName, value, slot);
161}
162
163static ALWAYS_INLINE void putDirectAccessorWithReify(VM& vm, JSGlobalObject* globalObject, JSObject* baseObject, PropertyName propertyName, GetterSetter* accessor, unsigned attribute)
164{
165 auto scope = DECLARE_THROW_SCOPE(vm);
166 if (baseObject->inherits<JSFunction>(vm)) {
167 jsCast<JSFunction*>(baseObject)->reifyLazyPropertyIfNeeded(vm, globalObject, propertyName);
168 RETURN_IF_EXCEPTION(scope, void());
169 }
170 scope.release();
171 baseObject->putDirectAccessor(globalObject, propertyName, accessor, attribute);
172}

174inline JSArray* allocateNewArrayBuffer(VM& vm, Structure* structure, JSImmutableButterfly* immutableButterfly)
175{
176 JSGlobalObject* globalObject = structure->globalObject();
177 Structure* originalStructure = globalObject->originalArrayStructureForIndexingType(immutableButterfly->indexingMode());
178 ASSERT(originalStructure->indexingMode() == immutableButterfly->indexingMode());
179 ASSERT(isCopyOnWrite(immutableButterfly->indexingMode()));
180 ASSERT(!structure->outOfLineCapacity());
181
182 JSArray* result = JSArray::createWithButterfly(vm, nullptr, originalStructure, immutableButterfly->toButterfly());
183 // FIXME: This works but it's slow. If we cared enough about the perf when having a bad time then we could fix it.
184 if (UNLIKELY(originalStructure != structure)) {
185 ASSERT(hasSlowPutArrayStorage(structure->indexingMode()));
186 ASSERT(globalObject->isHavingABadTime());
187
188 result->switchToSlowPutArrayStorage(vm);
189 ASSERT(result->butterfly() != immutableButterfly->toButterfly());
190 ASSERT(!result->butterfly()->arrayStorage()->m_sparseMap.get());
191 ASSERT(result->structureID() == structure->id());
192 }
193
194 return result;
195}
