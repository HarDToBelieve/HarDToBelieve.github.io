31namespace JSC {
32namespace DOMJIT {
33
34class GetterSetter;
35

38class DOMAttributeGetterSetter final : public CustomGetterSetter {
39public:
40 using Base = CustomGetterSetter;
41
42 template<typename CellType, SubspaceAccess>
43 static IsoSubspace* subspaceFor(VM& vm)
44 {
45 return &vm.domAttributeGetterSetterSpace;
46 }
47
48 static DOMAttributeGetterSetter* create(VM& vm, CustomGetter customGetter, CustomSetter customSetter, DOMAttributeAnnotation domAttribute)
49 {
50 DOMAttributeGetterSetter* customGetterSetter = new (NotNull, allocateCell<DOMAttributeGetterSetter>(vm.heap)) DOMAttributeGetterSetter(vm, customGetter, customSetter, domAttribute);
51 customGetterSetter->finishCreation(vm);
52 return customGetterSetter;
53 }
54
55 DOMAttributeAnnotation domAttribute() const { return m_domAttribute; }
56
57 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
58 {
59 return Structure::create(vm, globalObject, prototype, TypeInfo(CustomGetterSetterType, StructureFlags), info());
60 }
61
62 DECLARE_EXPORT_INFO;
63
64private:
65 DOMAttributeGetterSetter(VM& vm, CustomGetter getter, CustomSetter setter, DOMAttributeAnnotation domAttribute)
66 : Base(vm, vm.domAttributeGetterSetterStructure.get(), getter, setter)
67 , m_domAttribute(domAttribute)
68 {
69 }
70
71 DOMAttributeAnnotation m_domAttribute;
72};
