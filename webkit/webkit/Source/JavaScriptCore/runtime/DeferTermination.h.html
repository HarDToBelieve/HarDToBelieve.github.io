26#pragma once
27
28#include "VM.h"
29#include <wtf/ForbidHeapAllocation.h>
30#include <wtf/Noncopyable.h>
31
32namespace JSC {
33
34template<VMTraps::DeferAction deferAction = VMTraps::DeferAction::DeferUntilEndOfScope>
35class DeferTermination {
36 WTF_MAKE_NONCOPYABLE(DeferTermination);
37 WTF_FORBID_HEAP_ALLOCATION;
38public:
39 DeferTermination(VM& vm)
40 : m_vm(vm)
41 {
42 m_vm.traps().deferTermination(deferAction);
43 }
44
45 ~DeferTermination()
46 {
47 m_vm.traps().undoDeferTermination(deferAction);
48 }
49
50private:
51 VM& m_vm;
52};
53
54using DeferTerminationForAWhile = DeferTermination<VMTraps::DeferAction::DeferForAWhile>;

56} // namespace JSC
