26#pragma once
27
28#include <optional>
29#include <wtf/TriState.h>
30
31namespace JSC {
32
33class DefinePropertyAttributes {
34public:
35 static_assert(!static_cast<uint8_t>(TriState::False), "TriState::False is 0.");
36 static_assert(static_cast<uint8_t>(TriState::True) == 1, "TriState::True is 1.");
37 static_assert(static_cast<uint8_t>(TriState::Indeterminate) == 2, "TriState::Indeterminate is 2.");
38
39 static constexpr unsigned ConfigurableShift = 0;
40 static constexpr unsigned EnumerableShift = 2;
41 static constexpr unsigned WritableShift = 4;
42 static constexpr unsigned ValueShift = 6;
43 static constexpr unsigned GetShift = 7;
44 static constexpr unsigned SetShift = 8;
45
46 DefinePropertyAttributes()
47 : m_attributes(
48 (static_cast<uint8_t>(TriState::Indeterminate) << ConfigurableShift)
49 | (static_cast<uint8_t>(TriState::Indeterminate) << EnumerableShift)
50 | (static_cast<uint8_t>(TriState::Indeterminate) << WritableShift)
51 | (static_cast<uint8_t>(TriState::False) << ValueShift)
52 | (static_cast<uint8_t>(TriState::False) << GetShift)
53 | (static_cast<uint8_t>(TriState::False) << SetShift))
54 {
55 }
56
57 explicit DefinePropertyAttributes(unsigned attributes)
58 : m_attributes(attributes)
59 {
60 }
61
62 unsigned rawRepresentation() const
63 {
64 return m_attributes;
65 }
66
67 bool hasValue() const
68 {
69 return m_attributes & (0b1 << ValueShift);
70 }
71
72 void setValue()
73 {
74 m_attributes = m_attributes | (0b1 << ValueShift);
75 }
76
77 bool hasGet() const
78 {
79 return m_attributes & (0b1 << GetShift);
80 }
81
82 void setGet()
83 {
84 m_attributes = m_attributes | (0b1 << GetShift);
85 }
86
87 bool hasSet() const
88 {
89 return m_attributes & (0b1 << SetShift);
90 }
91
92 void setSet()
93 {
94 m_attributes = m_attributes | (0b1 << SetShift);
95 }
96
97 bool hasWritable() const
98 {
99 return extractTriState(WritableShift) != TriState::Indeterminate;
100 }
101
102 std::optional<bool> writable() const
103 {
104 if (!hasWritable())
105 return std::nullopt;
106 return extractTriState(WritableShift) == TriState::True;
107 }
108
109 bool hasConfigurable() const
110 {
111 return extractTriState(ConfigurableShift) != TriState::Indeterminate;
112 }
113
114 std::optional<bool> configurable() const
115 {
116 if (!hasConfigurable())
117 return std::nullopt;
118 return extractTriState(ConfigurableShift) == TriState::True;
119 }
120
121 bool hasEnumerable() const
122 {
123 return extractTriState(EnumerableShift) != TriState::Indeterminate;
124 }
125
126 std::optional<bool> enumerable() const
127 {
128 if (!hasEnumerable())
129 return std::nullopt;
130 return extractTriState(EnumerableShift) == TriState::True;
131 }
132
133 void setWritable(bool value)
134 {
135 fillWithTriState(value ? TriState::True : TriState::False, WritableShift);
136 }
137
138 void setConfigurable(bool value)
139 {
140 fillWithTriState(value ? TriState::True : TriState::False, ConfigurableShift);
141 }
142
143 void setEnumerable(bool value)
144 {
145 fillWithTriState(value ? TriState::True : TriState::False, EnumerableShift);
146 }
147
148private:
149 void fillWithTriState(TriState state, unsigned shift)
150 {
151 unsigned mask = 0b11 << shift;
152 m_attributes = (m_attributes & ~mask) | (static_cast<uint8_t>(state) << shift);
153 }
154
155 TriState extractTriState(unsigned shift) const
156 {
157 return static_cast<TriState>((m_attributes >> shift) & 0b11);
158 }
159
160 unsigned m_attributes;
161};
162
163
164} // namespace JSC
