28namespace JSC {
29
30class ErrorInstance : public JSNonFinalObject {
31public:
32 using Base = JSNonFinalObject;
33 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetOwnSpecialPropertyNames | OverridesPut;
34 static constexpr bool needsDestruction = true;
35
36 static void destroy(JSCell* cell)
37 {
38 static_cast<ErrorInstance*>(cell)->ErrorInstance::~ErrorInstance();
39 }
40
41 template<typename CellType, SubspaceAccess mode>
42 static IsoSubspace* subspaceFor(VM& vm)
43 {
44 return vm.errorInstanceSpace<mode>();
45 }
46
47 enum SourceTextWhereErrorOccurred { FoundExactSource, FoundApproximateSource };
48 typedef String (*SourceAppender) (const String& originalMessage, const String& sourceText, RuntimeType, SourceTextWhereErrorOccurred);
49
50 DECLARE_EXPORT_INFO;
51
52 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
53 {
54 return Structure::create(vm, globalObject, prototype, TypeInfo(ErrorInstanceType, StructureFlags), info());
55 }
56
57 static ErrorInstance* create(JSGlobalObject* globalObject, VM& vm, Structure* structure, const String& message, JSValue cause, SourceAppender appender = nullptr, RuntimeType type = TypeNothing, ErrorType errorType = ErrorType::Error, bool useCurrentFrame = true)
58 {
59 ErrorInstance* instance = new (NotNull, allocateCell<ErrorInstance>(vm.heap)) ErrorInstance(vm, structure, errorType);
60 instance->finishCreation(vm, globalObject, message, cause, appender, type, useCurrentFrame);
61 return instance;
62 }
63
64 static ErrorInstance* create(JSGlobalObject*, Structure*, JSValue message, JSValue options, SourceAppender = nullptr, RuntimeType = TypeNothing, ErrorType = ErrorType::Error, bool useCurrentFrame = true);
65
66 bool hasSourceAppender() const { return !!m_sourceAppender; }
67 SourceAppender sourceAppender() const { return m_sourceAppender; }
68 void setSourceAppender(SourceAppender appender) { m_sourceAppender = appender; }
69 void clearSourceAppender() { m_sourceAppender = nullptr; }
70 void setRuntimeTypeForCause(RuntimeType type) { m_runtimeTypeForCause = type; }
71 RuntimeType runtimeTypeForCause() const { return m_runtimeTypeForCause; }
72 void clearRuntimeTypeForCause() { m_runtimeTypeForCause = TypeNothing; }
73
74 ErrorType errorType() const { return m_errorType; }
75 void setStackOverflowError() { m_stackOverflowError = true; }
76 bool isStackOverflowError() const { return m_stackOverflowError; }
77 void setOutOfMemoryError() { m_outOfMemoryError = true; }
78 bool isOutOfMemoryError() const { return m_outOfMemoryError; }
79
80 void setNativeGetterTypeError() { m_nativeGetterTypeError = true; }
81 bool isNativeGetterTypeError() const { return m_nativeGetterTypeError; }
82
83 JS_EXPORT_PRIVATE String sanitizedToString(JSGlobalObject*);
84 JS_EXPORT_PRIVATE String sanitizedMessageString(JSGlobalObject*);
85 JS_EXPORT_PRIVATE String sanitizedNameString(JSGlobalObject*);
86
87 Vector<StackFrame>* stackTrace() { return m_stackTrace.get(); }
88
89 bool materializeErrorInfoIfNeeded(VM&);
90 bool materializeErrorInfoIfNeeded(VM&, PropertyName);
91
92 void finalizeUnconditionally(VM&);
93
94protected:
95 explicit ErrorInstance(VM&, Structure*, ErrorType);
96
97 void finishCreation(VM&, JSGlobalObject*, const String& message, JSValue cause, SourceAppender = nullptr, RuntimeType = TypeNothing, bool useCurrentFrame = true);
98
99 static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
100 static void getOwnSpecialPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, DontEnumPropertiesMode);
101 static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
102 static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
103 static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
104
105 void computeErrorInfo(VM&);
106
107 SourceAppender m_sourceAppender { nullptr };
108 std::unique_ptr<Vector<StackFrame>> m_stackTrace;
109 unsigned m_line;
110 unsigned m_column;
111 String m_sourceURL;
112 String m_stackString;
113 RuntimeType m_runtimeTypeForCause { TypeNothing };
114 ErrorType m_errorType { ErrorType::Error };
115 bool m_stackOverflowError : 1;
116 bool m_outOfMemoryError : 1;
117 bool m_errorInfoMaterialized : 1;
118 bool m_nativeGetterTypeError : 1;
119};

121} // namespace JSC
