110static String canonicalizeTimeZoneName(const String& timeZoneName)
111{
112 // 6.4.1 IsValidTimeZoneName (timeZone)
113 // The abstract operation returns true if timeZone, converted to upper case as described in 6.1, is equal to one of the Zone or Link names of the IANA Time Zone Database, converted to upper case as described in 6.1. It returns false otherwise.
114 UErrorCode status = U_ZERO_ERROR;
115 UEnumeration* timeZones = ucal_openTimeZones(&status);
116 ASSERT(U_SUCCESS(status));
117
118 String canonical;
119 do {
120 status = U_ZERO_ERROR;
121 int32_t ianaTimeZoneLength;
122 // Time zone names are represented as UChar[] in all related ICU APIs.
123 const UChar* ianaTimeZone = uenum_unext(timeZones, &ianaTimeZoneLength, &status);
124 ASSERT(U_SUCCESS(status));
125
126 // End of enumeration.
127 if (!ianaTimeZone)
128 break;
129
130 StringView ianaTimeZoneView(ianaTimeZone, ianaTimeZoneLength);
131 if (!equalIgnoringASCIICase(timeZoneName, ianaTimeZoneView))
132 continue;
133
134 // Found a match, now canonicalize.
135 // 6.4.2 CanonicalizeTimeZoneName (timeZone) (ECMA-402 2.0)
136 // 1. Let ianaTimeZone be the Zone or Link name of the IANA Time Zone Database such that timeZone, converted to upper case as described in 6.1, is equal to ianaTimeZone, converted to upper case as described in 6.1.
137 // 2. If ianaTimeZone is a Link name, then let ianaTimeZone be the corresponding Zone name as specified in the “backward” file of the IANA Time Zone Database.
138
139 Vector<UChar, 32> buffer;
140 auto status = callBufferProducingFunction(ucal_getCanonicalTimeZoneID, ianaTimeZone, ianaTimeZoneLength, buffer, nullptr);
141 ASSERT_UNUSED(status, U_SUCCESS(status));
142 canonical = String(buffer);
143 } while (canonical.isNull());
144 uenum_close(timeZones);
145
146 // 3. If ianaTimeZone is "Etc/UTC" or "Etc/GMT", then return "UTC".
147 if (isUTCEquivalent(canonical))
148 return "UTC"_s;
149
150 // 4. Return ianaTimeZone.
151 return canonical;
152}

154Vector<String> IntlDateTimeFormat::localeData(const String& locale, RelevantExtensionKey key)
155{
156 Vector<String> keyLocaleData;
157 switch (key) {
158 case RelevantExtensionKey::Ca: {
159 UErrorCode status = U_ZERO_ERROR;
160 UEnumeration* calendars = ucal_getKeywordValuesForLocale("calendar", locale.utf8().data(), false, &status);
161 ASSERT(U_SUCCESS(status));
162
163 int32_t nameLength;
164 while (const char* availableName = uenum_next(calendars, &nameLength, &status)) {
165 ASSERT(U_SUCCESS(status));
166 String calendar = String(availableName, nameLength);
167 keyLocaleData.append(calendar);
168 // Ensure aliases used in language tag are allowed.
169 if (calendar == "gregorian")
170 keyLocaleData.append("gregory"_s);
171 else if (calendar == "islamic-civil")
172 keyLocaleData.append("islamicc"_s);
173 else if (calendar == "ethiopic-amete-alem")
174 keyLocaleData.append("ethioaa"_s);
175 }
176 uenum_close(calendars);
177 break;
178 }
179 case RelevantExtensionKey::Hc:
180 // Null default so we know to use 'j' in pattern.
181 keyLocaleData.append(String());
182 keyLocaleData.append("h11"_s);
183 keyLocaleData.append("h12"_s);
184 keyLocaleData.append("h23"_s);
185 keyLocaleData.append("h24"_s);
186 break;
187 case RelevantExtensionKey::Nu:
188 keyLocaleData = numberingSystemsForLocale(locale);
189 break;
190 default:
191 ASSERT_NOT_REACHED();
192 }
193 return keyLocaleData;
194}

196static JSObject* toDateTimeOptionsAnyDate(JSGlobalObject* globalObject, JSValue originalOptions)
197{
198 // 12.1.1 ToDateTimeOptions abstract operation (ECMA-402 2.0)
199 VM& vm = globalObject->vm();
200 auto scope = DECLARE_THROW_SCOPE(vm);
201
202 // 1. If options is undefined, then let options be null, else let options be ToObject(options).
203 // 2. ReturnIfAbrupt(options).
204 // 3. Let options be ObjectCreate(options).
205 JSObject* options;
206 if (originalOptions.isUndefined())
207 options = constructEmptyObject(vm, globalObject->nullPrototypeObjectStructure());
208 else {
209 JSObject* originalToObject = originalOptions.toObject(globalObject);
210 RETURN_IF_EXCEPTION(scope, { });
211 options = constructEmptyObject(globalObject, originalToObject);
212 }
213
214 // 4. Let needDefaults be true.
215 bool needDefaults = true;
216
217 // 5. If required is "date" or "any",
218 // Always "any".
219
220 // a. For each of the property names "weekday", "year", "month", "day":
221 // i. Let prop be the property name.
222 // ii. Let value be Get(options, prop).
223 // iii. ReturnIfAbrupt(value).
224 // iv. If value is not undefined, then let needDefaults be false.
225 JSValue weekday = options->get(globalObject, vm.propertyNames->weekday);
226 RETURN_IF_EXCEPTION(scope, { });
227 if (!weekday.isUndefined())
228 needDefaults = false;
229
230 JSValue year = options->get(globalObject, vm.propertyNames->year);
231 RETURN_IF_EXCEPTION(scope, { });
232 if (!year.isUndefined())
233 needDefaults = false;
234
235 JSValue month = options->get(globalObject, vm.propertyNames->month);
236 RETURN_IF_EXCEPTION(scope, { });
237 if (!month.isUndefined())
238 needDefaults = false;
239
240 JSValue day = options->get(globalObject, vm.propertyNames->day);
241 RETURN_IF_EXCEPTION(scope, { });
242 if (!day.isUndefined())
243 needDefaults = false;
244
245 // 6. If required is "time" or "any",
246 // Always "any".
247
248 // a. For each of the property names ""dayPeriod", hour", "minute", "second", "fractionalSecondDigits":
249 // i. Let prop be the property name.
250 // ii. Let value be Get(options, prop).
251 // iii. ReturnIfAbrupt(value).
252 // iv. If value is not undefined, then let needDefaults be false.
253 if (Options::useIntlDateTimeFormatDayPeriod()) {
254 JSValue dayPeriod = options->get(globalObject, vm.propertyNames->dayPeriod);
255 RETURN_IF_EXCEPTION(scope, { });
256 if (!dayPeriod.isUndefined())
257 needDefaults = false;
258 }
259
260 JSValue hour = options->get(globalObject, vm.propertyNames->hour);
261 RETURN_IF_EXCEPTION(scope, { });
262 if (!hour.isUndefined())
263 needDefaults = false;
264
265 JSValue minute = options->get(globalObject, vm.propertyNames->minute);
266 RETURN_IF_EXCEPTION(scope, { });
267 if (!minute.isUndefined())
268 needDefaults = false;
269
270 JSValue second = options->get(globalObject, vm.propertyNames->second);
271 RETURN_IF_EXCEPTION(scope, { });
272 if (!second.isUndefined())
273 needDefaults = false;
274
275 JSValue fractionalSecondDigits = options->get(globalObject, vm.propertyNames->fractionalSecondDigits);
276 RETURN_IF_EXCEPTION(scope, { });
277 if (!fractionalSecondDigits.isUndefined())
278 needDefaults = false;
279
280 JSValue dateStyle = options->get(globalObject, vm.propertyNames->dateStyle);
281 RETURN_IF_EXCEPTION(scope, { });
282 JSValue timeStyle = options->get(globalObject, vm.propertyNames->timeStyle);
283 RETURN_IF_EXCEPTION(scope, { });
284
285 if (!dateStyle.isUndefined() || !timeStyle.isUndefined())
286 needDefaults = false;
287
288 // 7. If needDefaults is true and defaults is either "date" or "all", then
289 // Defaults is always "date".
290 if (needDefaults) {
291 // a. For each of the property names "year", "month", "day":
292 // i. Let status be CreateDatePropertyOrThrow(options, prop, "numeric").
293 // ii. ReturnIfAbrupt(status).
294 JSString* numeric = jsNontrivialString(vm, "numeric"_s);
295
296 options->putDirect(vm, vm.propertyNames->year, numeric);
297 RETURN_IF_EXCEPTION(scope, { });
298
299 options->putDirect(vm, vm.propertyNames->month, numeric);
300 RETURN_IF_EXCEPTION(scope, { });
301
302 options->putDirect(vm, vm.propertyNames->day, numeric);
303 RETURN_IF_EXCEPTION(scope, { });
304 }
305
306 // 8. If needDefaults is true and defaults is either "time" or "all", then
307 // Defaults is always "date". Ignore this branch.
308
309 // 9. Return options.
310 return options;
311}

313void IntlDateTimeFormat::setFormatsFromPattern(const StringView& pattern)
314{
315 // Get all symbols from the pattern, and set format fields accordingly.
316 // http://unicode.org/reports/tr35/tr35-dates.html#Date_Field_Symbol_Table
317 unsigned length = pattern.length();
318 for (unsigned i = 0; i < length; ++i) {
319 UChar currentCharacter = pattern[i];
320 if (!isASCIIAlpha(currentCharacter))
321 continue;
322
323 unsigned count = 1;
324 while (i + 1 < length && pattern[i + 1] == currentCharacter) {
325 ++count;
326 ++i;
327 }
328
329 switch (currentCharacter) {
330 case 'G':
331 if (count <= 3)
332 m_era = Era::Short;
333 else if (count == 4)
334 m_era = Era::Long;
335 else if (count == 5)
336 m_era = Era::Narrow;
337 break;
338 case 'y':
339 if (count == 1)
340 m_year = Year::Numeric;
341 else if (count == 2)
342 m_year = Year::TwoDigit;
343 break;
344 case 'M':
345 case 'L':
346 if (count == 1)
347 m_month = Month::Numeric;
348 else if (count == 2)
349 m_month = Month::TwoDigit;
350 else if (count == 3)
351 m_month = Month::Short;
352 else if (count == 4)
353 m_month = Month::Long;
354 else if (count == 5)
355 m_month = Month::Narrow;
356 break;
357 case 'E':
358 case 'e':
359 case 'c':
360 if (count <= 3)
361 m_weekday = Weekday::Short;
362 else if (count == 4)
363 m_weekday = Weekday::Long;
364 else if (count == 5)
365 m_weekday = Weekday::Narrow;
366 break;
367 case 'd':
368 if (count == 1)
369 m_day = Day::Numeric;
370 else if (count == 2)
371 m_day = Day::TwoDigit;
372 break;
373 case 'a':
374 case 'b':
375 case 'B':
376 if (count <= 3)
377 m_dayPeriod = DayPeriod::Short;
378 else if (count == 4)
379 m_dayPeriod = DayPeriod::Long;
380 else if (count == 5)
381 m_dayPeriod = DayPeriod::Narrow;
382 break;
383 case 'h':
384 case 'H':
385 case 'k':
386 case 'K': {
387 // Populate hourCycle from actually generated patterns. It is possible that locale or option is specifying hourCycle explicitly,
388 // but the generated pattern does not include related part since the pattern does not include hours.
389 // This is tested in test262/test/intl402/DateTimeFormat/prototype/resolvedOptions/hourCycle-dateStyle.js and our stress tests.
390 // Example:
391 // new Intl.DateTimeFormat(`de-u-hc-h11`, {
392 // dateStyle: "full"
393 // }).resolvedOptions().hourCycle === undefined
394 m_hourCycle = hourCycleFromSymbol(currentCharacter);
395 if (count == 1)
396 m_hour = Hour::Numeric;
397 else if (count == 2)
398 m_hour = Hour::TwoDigit;
399 break;
400 }
401 case 'm':
402 if (count == 1)
403 m_minute = Minute::Numeric;
404 else if (count == 2)
405 m_minute = Minute::TwoDigit;
406 break;
407 case 's':
408 if (count == 1)
409 m_second = Second::Numeric;
410 else if (count == 2)
411 m_second = Second::TwoDigit;
412 break;
413 case 'z':
414 case 'v':
415 case 'V':
416 if (count == 1)
417 m_timeZoneName = TimeZoneName::Short;
418 else if (count == 4)
419 m_timeZoneName = TimeZoneName::Long;
420 break;
421 case 'S':
422 m_fractionalSecondDigits = count;
423 break;
424 }
425 }
426}

518// https://tc39.github.io/ecma402/#sec-initializedatetimeformat
519void IntlDateTimeFormat::initializeDateTimeFormat(JSGlobalObject* globalObject, JSValue locales, JSValue originalOptions)
520{
521 VM& vm = globalObject->vm();
522 auto scope = DECLARE_THROW_SCOPE(vm);
523
524 Vector<String> requestedLocales = canonicalizeLocaleList(globalObject, locales);
525 RETURN_IF_EXCEPTION(scope, void());
526
527 JSObject* options = toDateTimeOptionsAnyDate(globalObject, originalOptions);
528 RETURN_IF_EXCEPTION(scope, void());
529
530 ResolveLocaleOptions localeOptions;
531
532 LocaleMatcher localeMatcher = intlOption<LocaleMatcher>(globalObject, options, vm.propertyNames->localeMatcher, { { "lookup"_s, LocaleMatcher::Lookup }, { "best fit"_s, LocaleMatcher::BestFit } }, "localeMatcher must be either \"lookup\" or \"best fit\""_s, LocaleMatcher::BestFit);
533 RETURN_IF_EXCEPTION(scope, void());
534
535 String calendar = intlStringOption(globalObject, options, vm.propertyNames->calendar, { }, nullptr, nullptr);
536 RETURN_IF_EXCEPTION(scope, void());
537 if (!calendar.isNull()) {
538 if (!isUnicodeLocaleIdentifierType(calendar)) {
539 throwRangeError(globalObject, scope, "calendar is not a well-formed calendar value"_s);
540 return;
541 }
542 localeOptions[static_cast<unsigned>(RelevantExtensionKey::Ca)] = calendar;
543 }
544
545 String numberingSystem = intlStringOption(globalObject, options, vm.propertyNames->numberingSystem, { }, nullptr, nullptr);
546 RETURN_IF_EXCEPTION(scope, void());
547 if (!numberingSystem.isNull()) {
548 if (!isUnicodeLocaleIdentifierType(numberingSystem)) {
549 throwRangeError(globalObject, scope, "numberingSystem is not a well-formed numbering system value"_s);
550 return;
551 }
552 localeOptions[static_cast<unsigned>(RelevantExtensionKey::Nu)] = numberingSystem;
553 }
554
555 TriState hour12 = intlBooleanOption(globalObject, options, vm.propertyNames->hour12);
556 RETURN_IF_EXCEPTION(scope, void());
557
558 HourCycle hourCycle = intlOption<HourCycle>(globalObject, options, vm.propertyNames->hourCycle, { { "h11"_s, HourCycle::H11 }, { "h12"_s, HourCycle::H12 }, { "h23"_s, HourCycle::H23 }, { "h24"_s, HourCycle::H24 } }, "hourCycle must be \"h11\", \"h12\", \"h23\", or \"h24\""_s, HourCycle::None);
559 RETURN_IF_EXCEPTION(scope, void());
560 if (hour12 == TriState::Indeterminate) {
561 if (hourCycle != HourCycle::None)
562 localeOptions[static_cast<unsigned>(RelevantExtensionKey::Hc)] = String(hourCycleString(hourCycle));
563 } else {
564 // If there is hour12, hourCycle is ignored.
565 // We are setting null String explicitly here (localeOptions' entries are std::optional<String>). This leads us to use HourCycle::None later.
566 localeOptions[static_cast<unsigned>(RelevantExtensionKey::Hc)] = String();
567 }
568
569 const auto& availableLocales = intlDateTimeFormatAvailableLocales();
570 auto resolved = resolveLocale(globalObject, availableLocales, requestedLocales, localeMatcher, localeOptions, { RelevantExtensionKey::Ca, RelevantExtensionKey::Hc, RelevantExtensionKey::Nu }, localeData);
571
572 m_locale = resolved.locale;
573 if (m_locale.isEmpty()) {
574 throwTypeError(globalObject, scope, "failed to initialize DateTimeFormat due to invalid locale"_s);
575 return;
576 }
577
578 m_calendar = resolved.extensions[static_cast<unsigned>(RelevantExtensionKey::Ca)];
579 if (m_calendar == "gregorian")
580 m_calendar = "gregory"_s;
581 else if (m_calendar == "islamicc")
582 m_calendar = "islamic-civil"_s;
583 else if (m_calendar == "ethioaa")
584 m_calendar = "ethiopic-amete-alem"_s;
585
586 hourCycle = parseHourCycle(resolved.extensions[static_cast<unsigned>(RelevantExtensionKey::Hc)]);
587 m_numberingSystem = resolved.extensions[static_cast<unsigned>(RelevantExtensionKey::Nu)];
588 m_dataLocale = resolved.dataLocale;
589 CString dataLocaleWithExtensions = makeString(m_dataLocale, "-u-ca-", m_calendar, "-nu-", m_numberingSystem).utf8();
590
591 JSValue tzValue = options->get(globalObject, vm.propertyNames->timeZone);
592 RETURN_IF_EXCEPTION(scope, void());
593 String tz;
594 if (!tzValue.isUndefined()) {
595 String originalTz = tzValue.toWTFString(globalObject);
596 RETURN_IF_EXCEPTION(scope, void());
597 tz = canonicalizeTimeZoneName(originalTz);
598 if (tz.isNull()) {
599 throwRangeError(globalObject, scope, "invalid time zone: " + originalTz);
600 return;
601 }
602 } else
603 tz = vm.dateCache.defaultTimeZone();
604 m_timeZone = tz;
605
606 StringBuilder skeletonBuilder;
607
608 Weekday weekday = intlOption<Weekday>(globalObject, options, vm.propertyNames->weekday, { { "narrow"_s, Weekday::Narrow }, { "short"_s, Weekday::Short }, { "long"_s, Weekday::Long } }, "weekday must be \"narrow\", \"short\", or \"long\""_s, Weekday::None);
609 RETURN_IF_EXCEPTION(scope, void());
610 switch (weekday) {
611 case Weekday::Narrow:
612 skeletonBuilder.append("EEEEE");
613 break;
614 case Weekday::Short:
615 skeletonBuilder.append("EEE");
616 break;
617 case Weekday::Long:
618 skeletonBuilder.append("EEEE");
619 break;
620 case Weekday::None:
621 break;
622 }
623
624 Era era = intlOption<Era>(globalObject, options, vm.propertyNames->era, { { "narrow"_s, Era::Narrow }, { "short"_s, Era::Short }, { "long"_s, Era::Long } }, "era must be \"narrow\", \"short\", or \"long\""_s, Era::None);
625 RETURN_IF_EXCEPTION(scope, void());
626 switch (era) {
627 case Era::Narrow:
628 skeletonBuilder.append("GGGGG");
629 break;
630 case Era::Short:
631 skeletonBuilder.append("GGG");
632 break;
633 case Era::Long:
634 skeletonBuilder.append("GGGG");
635 break;
636 case Era::None:
637 break;
638 }
639
640 Year year = intlOption<Year>(globalObject, options, vm.propertyNames->year, { { "2-digit"_s, Year::TwoDigit }, { "numeric"_s, Year::Numeric } }, "year must be \"2-digit\" or \"numeric\""_s, Year::None);
641 RETURN_IF_EXCEPTION(scope, void());
642 switch (year) {
643 case Year::TwoDigit:
644 skeletonBuilder.append("yy");
645 break;
646 case Year::Numeric:
647 skeletonBuilder.append('y');
648 break;
649 case Year::None:
650 break;
651 }

842 // We cannot use this UDateFormat directly yet because we need to enforce specified hourCycle.
843 // First, we create UDateFormat via dateStyle and timeStyle. And then convert it to pattern string.
844 // After updating this pattern string with hourCycle, we create a final UDateFormat with the updated pattern string.
845 UErrorCode status = U_ZERO_ERROR;

859 // It is possible that timeStyle includes dayPeriod, which is sensitive to hour-cycle.
860 // If dayPeriod is included, just replacing hour based on hourCycle / hour12 produces strange results.
861 // Let's consider about the example. The formatted result looks like "02:12:47 PM Coordinated Universal Time"
862 // If we simply replace 02 to 14, this becomes "14:12:47 PM Coordinated Universal Time", this looks strange since "PM" is unnecessary!
863 //
864 // If the generated pattern's hour12 does not match against the option's one, we retrieve skeleton from the pattern, enforcing hour-cycle,
865 // and re-generating the best pattern from the modified skeleton. ICU will look into the generated skeleton, and pick the best format for the request.
866 // We do not care about h11 vs. h12 and h23 vs. h24 difference here since this will be later adjusted by replaceHourCycleInPattern.
867 //
868 // test262/test/intl402/DateTimeFormat/prototype/format/timedatestyle-en.js includes the test for this behavior.
869 if (m_timeStyle != DateTimeStyle::None && (hourCycle != HourCycle::None || hour12 != TriState::Indeterminate)) {
870 auto isHour12 = [](HourCycle hourCycle) {
871 return hourCycle == HourCycle::H11 || hourCycle == HourCycle::H12;
872 };
873 bool specifiedHour12 = false;
874 // If hour12 is specified, we prefer it and ignore hourCycle.
875 if (hour12 != TriState::Indeterminate)
876 specifiedHour12 = hour12 == TriState::True;
877 else
878 specifiedHour12 = isHour12(hourCycle);
879 HourCycle extractedHourCycle = hourCycleFromPattern(patternBuffer);
880 if (extractedHourCycle != HourCycle::None && isHour12(extractedHourCycle) != specifiedHour12) {
881 Vector<UChar, 32> skeleton;
882 auto status = callBufferProducingFunction(udatpg_getSkeleton, nullptr, patternBuffer.data(), patternBuffer.size(), skeleton);
883 if (U_FAILURE(status)) {
884 throwTypeError(globalObject, scope, "failed to initialize DateTimeFormat"_s);
885 return;
886 }
887 replaceHourCycleInSkeleton(skeleton, specifiedHour12);
888 dataLogLnIf(IntlDateTimeFormatInternal::verbose, "replaced:(", StringView(skeleton.data(), skeleton.size()), ")");
889
890 patternBuffer = vm.intlCache().getBestDateTimePattern(dataLocaleWithExtensions, skeleton.data(), skeleton.size(), status);
891 if (U_FAILURE(status)) {
892 throwTypeError(globalObject, scope, "failed to initialize DateTimeFormat"_s);
893 return;
894 }
895 }
896 }
897 } else {
898 UErrorCode status = U_ZERO_ERROR;
899 String skeleton = skeletonBuilder.toString();
900 patternBuffer = vm.intlCache().getBestDateTimePattern(dataLocaleWithExtensions, StringView(skeleton).upconvertedCharacters().get(), skeleton.length(), status);
901 if (U_FAILURE(status)) {
902 throwTypeError(globalObject, scope, "failed to initialize DateTimeFormat"_s);
903 return;
904 }
905 }
906
907 // After generating pattern from skeleton, we need to change h11 vs. h12 and h23 vs. h24 if hourCycle is specified.
908 if (hourCycle != HourCycle::None)
909 replaceHourCycleInPattern(patternBuffer, hourCycle);
910
911 StringView pattern(patternBuffer.data(), patternBuffer.size());
912 setFormatsFromPattern(pattern);
913
914 dataLogLnIf(IntlDateTimeFormatInternal::verbose, "locale:(", m_locale, "),dataLocale:(", dataLocaleWithExtensions, "),pattern:(", pattern, ")");
915
916 UErrorCode status = U_ZERO_ERROR;
917 StringView timeZoneView(m_timeZone);
918 m_dateFormat = std::unique_ptr<UDateFormat, UDateFormatDeleter>(udat_open(UDAT_PATTERN, UDAT_PATTERN, dataLocaleWithExtensions.data(), timeZoneView.upconvertedCharacters(), timeZoneView.length(), pattern.upconvertedCharacters(), pattern.length(), &status));
919 if (U_FAILURE(status)) {
920 throwTypeError(globalObject, scope, "failed to initialize DateTimeFormat"_s);
921 return;
922 }
923
924 // Gregorian calendar should be used from the beginning of ECMAScript time.
925 // Failure here means unsupported calendar, and can safely be ignored.
926 UCalendar* cal = const_cast<UCalendar*>(udat_getCalendar(m_dateFormat.get()));
927 ucal_setGregorianChange(cal, minECMAScriptTime, &status);
928}

1318 }
1319 previousEndIndex = endIndex;
1320

1322 auto type = jsString(vm, partTypeString(UDateFormatField(fieldType)));
1323 auto value = jsString(vm, resultString.substring(beginIndex, endIndex - beginIndex));
1324 JSObject* part = constructEmptyObject(globalObject);

1326 part->putDirect(vm, vm.propertyNames->value, value);
1327 if (sourceType)
1328 part->putDirect(vm, vm.propertyNames->source, sourceType);

1387static std::unique_ptr<UFormattedDateInterval, ICUDeleter<udtitvfmt_closeResult>> formattedValueFromDateRange(UDateIntervalFormat& dateIntervalFormat, UDateFormat& dateFormat, double startDate, double endDate, UErrorCode& status)
1388{
1389 auto result = std::unique_ptr<UFormattedDateInterval, ICUDeleter<udtitvfmt_closeResult>>(udtitvfmt_openResult(&status));
1390 if (U_FAILURE(status))
1391 return nullptr;
1392
1393 // After ICU 67, udtitvfmt_formatToResult's signature is changed.
1394#if U_ICU_VERSION_MAJOR_NUM >= 67
1395 // If a date is after Oct 15, 1582, the configuration of gregorian calendar change date in UCalendar does not affect
1396 // on the formatted string. To ensure that it is after Oct 15 in all timezones, we add one day to gregorian calendar
1397 // change date in UTC, so that this check can conservatively answer whether the date is definitely after gregorian
1398 // calendar change date.
1399 auto definitelyAfterGregorianCalendarChangeDate = [](double millisecondsFromEpoch) {
1400 constexpr double gregorianCalendarReformDateInUTC = -12219292800000.0;
1401 return millisecondsFromEpoch >= (gregorianCalendarReformDateInUTC + msPerDay);
1402 };
1403
1404 // UFormattedDateInterval does not have a way to configure gregorian calendar change date while ECMAScript requires that
1405 // gregorian calendar change should not have effect (we are setting ucal_setGregorianChange(cal, minECMAScriptTime, &status) explicitly).
1406 // As a result, if the input date is older than gregorian calendar change date (Oct 15, 1582), the formatted string becomes
1407 // julian calendar date.
1408 // udtitvfmt_formatCalendarToResult API offers the way to set calendar to each date of the input, so that we can use UDateFormat's
1409 // calendar which is already configured to meet ECMAScript's requirement (effectively clearing gregorian calendar change date).
1410 //
1411 // If we can ensure that startDate is after gregorian calendar change date, we can just use udtitvfmt_formatToResult since gregorian
1412 // calendar change date does not affect on the formatted string.
1413 //
1414 // https://unicode-org.atlassian.net/browse/ICU-20705
1415 if (definitelyAfterGregorianCalendarChangeDate(startDate))
1416 udtitvfmt_formatToResult(&dateIntervalFormat, startDate, endDate, result.get(), &status);
1417 else {
1418 auto createCalendarForDate = [](const UCalendar* calendar, double date, UErrorCode& status) -> std::unique_ptr<UCalendar, ICUDeleter<ucal_close>> {
1419 auto result = std::unique_ptr<UCalendar, ICUDeleter<ucal_close>>(ucal_clone(calendar, &status));
1420 if (U_FAILURE(status))
1421 return nullptr;
1422 ucal_setMillis(result.get(), date, &status);
1423 if (U_FAILURE(status))
1424 return nullptr;
1425 return result;
1426 };
1427
1428 auto calendar = udat_getCalendar(&dateFormat);
1429
1430 auto startCalendar = createCalendarForDate(calendar, startDate, status);
1431 if (U_FAILURE(status))
1432 return nullptr;
1433
1434 auto endCalendar = createCalendarForDate(calendar, endDate, status);
1435 if (U_FAILURE(status))
1436 return nullptr;
1437
1438 udtitvfmt_formatCalendarToResult(&dateIntervalFormat, startCalendar.get(), endCalendar.get(), result.get(), &status);
1439 }
1440#else
1441 UNUSED_PARAM(dateFormat);
1442 udtitvfmt_formatToResult(&dateIntervalFormat, result.get(), startDate, endDate, &status);
1443#endif
1444 return result;
1445}

1468{

1470
1471 VM& vm = globalObject->vm();
1472 auto scope = DECLARE_THROW_SCOPE(vm);
1473
1474 // http://tc39.es/proposal-intl-DateTimeFormat-formatRange/#sec-partitiondatetimerangepattern
1475 startDate = timeClip(startDate);
1476 endDate = timeClip(endDate);
1477 if (std::isnan(startDate) || std::isnan(endDate)) {
1478 throwRangeError(globalObject, scope, "Passed date is out of range"_s);
1479 return { };
1480 }
1481
1482 auto* dateIntervalFormat = createDateIntervalFormatIfNecessary(globalObject);
1483 RETURN_IF_EXCEPTION(scope, { });
1484
1485#if HAVE(ICU_U_DATE_INTERVAL_FORMAT_FORMAT_RANGE_TO_PARTS)
1486 UErrorCode status = U_ZERO_ERROR;
1487 auto result = formattedValueFromDateRange(*dateIntervalFormat, *m_dateFormat, startDate, endDate, status);
1488 if (U_FAILURE(status)) {
1489 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1490 return { };
1491 }

1493 // UFormattedValue is owned by UFormattedDateInterval. We do not need to close it.
1494 auto formattedValue = udtitvfmt_resultAsValue(result.get(), &status);
1495 if (U_FAILURE(status)) {
1496 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1497 return { };
1498 }

1500 // If the formatted parts of startDate and endDate are the same, it is possible that the resulted string does not look like range.
1501 // For example, if the requested format only includes "year" and startDate and endDate are the same year, the result just contains one year.
1502 // In that case, startDate and endDate are *practically-equal* (spec term), and we generate parts as we call `formatToParts(startDate)` with
1503 // `source: "shared"` additional fields.

1511 RELEASE_AND_RETURN(scope, format(globalObject, startDate));
1512
1513 int32_t formattedStringLength = 0;
1514 const UChar* formattedStringPointer = ufmtval_getString(formattedValue, &formattedStringLength, &status);
1515 if (U_FAILURE(status)) {
1516 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1517 return { };
1518 }
1519
1520 return jsString(vm, String(formattedStringPointer, formattedStringLength));
1521#else
1522 Vector<UChar, 32> buffer;
1523 auto status = callBufferProducingFunction(udtitvfmt_format, dateIntervalFormat, startDate, endDate, buffer, nullptr);
1524 if (U_FAILURE(status)) {
1525 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1526 return { };
1527 }

1529 return jsString(vm, String(buffer));

1533JSValue IntlDateTimeFormat::formatRangeToParts(JSGlobalObject* globalObject, double startDate, double endDate)
1534{
1535 ASSERT(m_dateFormat);
1536
1537 VM& vm = globalObject->vm();
1538 auto scope = DECLARE_THROW_SCOPE(vm);
1539
1540#if HAVE(ICU_U_DATE_INTERVAL_FORMAT_FORMAT_RANGE_TO_PARTS)
1541 // http://tc39.es/proposal-intl-DateTimeFormat-formatRange/#sec-partitiondatetimerangepattern
1542 startDate = timeClip(startDate);
1543 endDate = timeClip(endDate);
1544 if (std::isnan(startDate) || std::isnan(endDate)) {
1545 throwRangeError(globalObject, scope, "Passed date is out of range"_s);
1546 return { };
1547 }
1548
1549 auto* dateIntervalFormat = createDateIntervalFormatIfNecessary(globalObject);
1550 RETURN_IF_EXCEPTION(scope, { });
1551
1552 UErrorCode status = U_ZERO_ERROR;
1553 auto result = formattedValueFromDateRange(*dateIntervalFormat, *m_dateFormat, startDate, endDate, status);
1554 if (U_FAILURE(status)) {
1555 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1556 return { };
1557 }
1558
1559 // UFormattedValue is owned by UFormattedDateInterval. We do not need to close it.
1560 auto formattedValue = udtitvfmt_resultAsValue(result.get(), &status);
1561 if (U_FAILURE(status)) {
1562 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1563 return { };
1564 }
1565
1566 auto sharedString = jsNontrivialString(vm, "shared"_s);
1567
1568 // If the formatted parts of startDate and endDate are the same, it is possible that the resulted string does not look like range.
1569 // For example, if the requested format only includes "year" and startDate and endDate are the same year, the result just contains one year.
1570 // In that case, startDate and endDate are *practically-equal* (spec term), and we generate parts as we call `formatToParts(startDate)` with
1571 // `source: "shared"` additional fields.
1572 bool equal = dateFieldsPracticallyEqual(formattedValue, status);
1573 if (U_FAILURE(status)) {
1574 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1575 return { };
1576 }
1577
1578 if (equal)
1579 RELEASE_AND_RETURN(scope, formatToParts(globalObject, startDate, sharedString));
1580
1581 // ICU produces ranges for the formatted string, and we construct parts array from that.
1582 // For example, startDate = Jan 3, 2019, endDate = Jan 5, 2019 with en-US locale is,
1583 //
1584 // Formatted string: "1/3/2019 – 1/5/2019"
1585 // | | | | | | | |
1586 // B C | | F G | |
1587 // | +-D+ | +-H+
1588 // | | | |
1589 // +--A---+ +--E---+
1590 //
1591 // Ranges ICU generates:
1592 // A: (0, 8) UFIELD_CATEGORY_DATE_INTERVAL_SPAN startRange
1593 // B: (0, 1) UFIELD_CATEGORY_DATE month
1594 // C: (2, 3) UFIELD_CATEGORY_DATE day
1595 // D: (4, 8) UFIELD_CATEGORY_DATE year
1596 // E: (11, 19) UFIELD_CATEGORY_DATE_INTERVAL_SPAN endRange
1597 // F: (11, 12) UFIELD_CATEGORY_DATE month
1598 // G: (13, 14) UFIELD_CATEGORY_DATE day
1599 // H: (15, 19) UFIELD_CATEGORY_DATE year
1600 //
1601 // We use UFIELD_CATEGORY_DATE_INTERVAL_SPAN range to determine each part is either "startRange", "endRange", or "shared".
1602 // It is gurarnteed that UFIELD_CATEGORY_DATE_INTERVAL_SPAN comes first before any other parts including that range.
1603 // For example, in the above formatted string, " – " is "shared" part. For UFIELD_CATEGORY_DATE ranges, we generate corresponding
1604 // part object with types such as "month". And non populated parts (e.g. "/") become "literal" parts.
1605 // In the above case, expected parts are,
1606 //
1607 // { type: "month", value: "1", source: "startRange" },
1608 // { type: "literal", value: "/", source: "startRange" },
1609 // { type: "day", value: "3", source: "startRange" },
1610 // { type: "literal", value: "/", source: "startRange" },
1611 // { type: "year", value: "2019", source: "startRange" },
1612 // { type: "literal", value: " - ", source: "shared" },
1613 // { type: "month", value: "1", source: "endRange" },
1614 // { type: "literal", value: "/", source: "endRange" },
1615 // { type: "day", value: "5", source: "endRange" },
1616 // { type: "literal", value: "/", source: "endRange" },
1617 // { type: "year", value: "2019", source: "endRange" },
1618 //
1619
1620 JSArray* parts = JSArray::tryCreate(vm, globalObject->arrayStructureForIndexingTypeDuringAllocation(ArrayWithContiguous), 0);
1621 if (!parts) {
1622 throwOutOfMemoryError(globalObject, scope);
1623 return { };
1624 }
1625
1626 int32_t formattedStringLength = 0;
1627 const UChar* formattedStringPointer = ufmtval_getString(formattedValue, &formattedStringLength, &status);
1628 if (U_FAILURE(status)) {
1629 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1630 return { };
1631 }
1632 String resultString(formattedStringPointer, formattedStringLength);
1633
1634 // We care multiple categories (UFIELD_CATEGORY_DATE and UFIELD_CATEGORY_DATE_INTERVAL_SPAN).
1635 // So we do not constraint iterator.
1636 auto iterator = std::unique_ptr<UConstrainedFieldPosition, ICUDeleter<ucfpos_close>>(ucfpos_open(&status));
1637 if (U_FAILURE(status)) {
1638 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1639 return { };
1640 }
1641
1642 auto startRangeString = jsNontrivialString(vm, "startRange"_s);
1643 auto endRangeString = jsNontrivialString(vm, "endRange"_s);
1644 auto literalString = jsNontrivialString(vm, "literal"_s);
1645
1646 WTF::Range<int32_t> startRange { -1, -1 };
1647 WTF::Range<int32_t> endRange { -1, -1 };
1648
1649 auto createPart = [&] (JSString* type, int32_t beginIndex, int32_t length) {
1650 auto sourceType = [&](int32_t index) -> JSString* {
1651 if (startRange.contains(index))
1652 return startRangeString;
1653 if (endRange.contains(index))
1654 return endRangeString;
1655 return sharedString;
1656 };
1657
1658 auto value = jsString(vm, resultString.substring(beginIndex, length));
1659 JSObject* part = constructEmptyObject(globalObject);
1660 part->putDirect(vm, vm.propertyNames->type, type);
1661 part->putDirect(vm, vm.propertyNames->value, value);
1662 part->putDirect(vm, vm.propertyNames->source, sourceType(beginIndex));
1663 return part;
1664 };
1665
1666 int32_t resultLength = resultString.length();
1667 int32_t previousEndIndex = 0;
1668 while (true) {
1669 bool next = ufmtval_nextPosition(formattedValue, iterator.get(), &status);
1670 if (U_FAILURE(status)) {
1671 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1672 return { };
1673 }
1674 if (!next)
1675 break;
1676
1677 int32_t category = ucfpos_getCategory(iterator.get(), &status);
1678 if (U_FAILURE(status)) {
1679 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1680 return { };
1681 }
1682
1683 int32_t fieldType = ucfpos_getField(iterator.get(), &status);
1684 if (U_FAILURE(status)) {
1685 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1686 return { };
1687 }
1688
1689 int32_t beginIndex = 0;
1690 int32_t endIndex = 0;
1691 ucfpos_getIndexes(iterator.get(), &beginIndex, &endIndex, &status);
1692 if (U_FAILURE(status)) {
1693 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1694 return { };
1695 }
1696
1697 dataLogLnIf(IntlDateTimeFormatInternal::verbose, category, " ", fieldType, " (", beginIndex, ", ", endIndex, ")");
1698
1699 if (category != UFIELD_CATEGORY_DATE && category != UFIELD_CATEGORY_DATE_INTERVAL_SPAN)
1700 continue;
1701 if (category == UFIELD_CATEGORY_DATE && fieldType < 0)
1702 continue;
1703
1704 if (previousEndIndex < beginIndex) {
1705 JSObject* part = createPart(literalString, previousEndIndex, beginIndex - previousEndIndex);
1706 parts->push(globalObject, part);
1707 RETURN_IF_EXCEPTION(scope, { });
1708 previousEndIndex = beginIndex;
1709 }
1710
1711 if (category == UFIELD_CATEGORY_DATE_INTERVAL_SPAN) {
1712 // > The special field category UFIELD_CATEGORY_DATE_INTERVAL_SPAN is used to indicate which datetime
1713 // > primitives came from which arguments: 0 means fromCalendar, and 1 means toCalendar. The span category
1714 // > will always occur before the corresponding fields in UFIELD_CATEGORY_DATE in the nextPosition() iterator.
1715 // from ICU comment. So, field 0 is startRange, field 1 is endRange.
1716 if (!fieldType)
1717 startRange = WTF::Range<int32_t>(beginIndex, endIndex);
1718 else {
1719 ASSERT(fieldType == 1);
1720 endRange = WTF::Range<int32_t>(beginIndex, endIndex);
1721 }
1722 continue;
1723 }
1724
1725 ASSERT(category == UFIELD_CATEGORY_DATE);
1726
1727 auto type = jsString(vm, partTypeString(UDateFormatField(fieldType)));
1728 JSObject* part = createPart(type, beginIndex, endIndex - beginIndex);
1729 parts->push(globalObject, part);
1730 RETURN_IF_EXCEPTION(scope, { });
1731 previousEndIndex = endIndex;
1732 }
1733
1734 if (previousEndIndex < resultLength) {
1735 JSObject* part = createPart(literalString, previousEndIndex, resultLength - previousEndIndex);
1736 parts->push(globalObject, part);
1737 RETURN_IF_EXCEPTION(scope, { });
1738 }
1739
1740 return parts;
1741#else
1742 UNUSED_PARAM(startDate);
1743 UNUSED_PARAM(endDate);
1744 throwTypeError(globalObject, scope, "Failed to format date interval"_s);
1745 return { };
1746#endif
