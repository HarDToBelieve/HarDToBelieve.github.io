85// https://tc39.es/proposal-intl-list-format/#sec-Intl.ListFormat
86void IntlListFormat::initializeListFormat(JSGlobalObject* globalObject, JSValue locales, JSValue optionsValue)
87{
88 VM& vm = globalObject->vm();
89 auto scope = DECLARE_THROW_SCOPE(vm);
90
91 auto requestedLocales = canonicalizeLocaleList(globalObject, locales);
92 RETURN_IF_EXCEPTION(scope, void());
93
94 JSObject* options = intlGetOptionsObject(globalObject, optionsValue);
95 RETURN_IF_EXCEPTION(scope, void());
96
97 ResolveLocaleOptions localeOptions;
98
99 LocaleMatcher localeMatcher = intlOption<LocaleMatcher>(globalObject, options, vm.propertyNames->localeMatcher, { { "lookup"_s, LocaleMatcher::Lookup }, { "best fit"_s, LocaleMatcher::BestFit } }, "localeMatcher must be either \"lookup\" or \"best fit\""_s, LocaleMatcher::BestFit);
100 RETURN_IF_EXCEPTION(scope, void());
101
102 auto localeData = [](const String&, RelevantExtensionKey) -> Vector<String> {
103 return { };
104 };
105
106 const auto& availableLocales = intlListFormatAvailableLocales();
107 auto resolved = resolveLocale(globalObject, availableLocales, requestedLocales, localeMatcher, localeOptions, { }, localeData);
108
109 m_locale = resolved.locale;
110 if (m_locale.isEmpty()) {
111 throwTypeError(globalObject, scope, "failed to initialize ListFormat due to invalid locale"_s);
112 return;
113 }
114
115 m_type = intlOption<Type>(globalObject, options, vm.propertyNames->type, { { "conjunction"_s, Type::Conjunction }, { "disjunction"_s, Type::Disjunction }, { "unit"_s, Type::Unit } }, "type must be either \"conjunction\", \"disjunction\", or \"unit\""_s, Type::Conjunction);
116 RETURN_IF_EXCEPTION(scope, void());
117
118 m_style = intlOption<Style>(globalObject, options, vm.propertyNames->style, { { "long"_s, Style::Long }, { "short"_s, Style::Short }, { "narrow"_s, Style::Narrow } }, "style must be either \"long\", \"short\", or \"narrow\""_s, Style::Long);
119 RETURN_IF_EXCEPTION(scope, void());
120
121#if HAVE(ICU_U_LIST_FORMATTER)
122 auto toUListFormatterType = [](Type type) {
123 switch (type) {
124 case Type::Conjunction:
125 return ULISTFMT_TYPE_AND;
126 case Type::Disjunction:
127 return ULISTFMT_TYPE_OR;
128 case Type::Unit:
129 return ULISTFMT_TYPE_UNITS;
130 }
131 return ULISTFMT_TYPE_AND;
132 };
133
134 auto toUListFormatterWidth = [](Style style) {
135 switch (style) {
136 case Style::Long:
137 return ULISTFMT_WIDTH_WIDE;
138 case Style::Short:
139 return ULISTFMT_WIDTH_SHORT;
140 case Style::Narrow:
141 return ULISTFMT_WIDTH_NARROW;
142 }
143 return ULISTFMT_WIDTH_WIDE;
144 };
145
146 UErrorCode status = U_ZERO_ERROR;
147 m_listFormat = std::unique_ptr<UListFormatter, UListFormatterDeleter>(ulistfmt_openForType(m_locale.utf8().data(), toUListFormatterType(m_type), toUListFormatterWidth(m_style), &status));
148 if (U_FAILURE(status)) {
149 throwTypeError(globalObject, scope, "failed to initialize ListFormat"_s);
150 return;
151 }
152#else
153 throwTypeError(globalObject, scope, "Failed to initialize Intl.ListFormat since this feature is not supported in the linked ICU version"_s);
154 return;
155#endif
156}

158#if HAVE(ICU_U_LIST_FORMATTER)
159static Vector<String, 4> stringListFromIterable(JSGlobalObject* globalObject, JSValue iterable)
160{
161 Vector<String, 4> result;
162
163 if (iterable.isUndefined())
164 return result;
165
166 forEachInIterable(globalObject, iterable, [&](VM& vm, JSGlobalObject* globalObject, JSValue value) {
167 auto scope = DECLARE_THROW_SCOPE(vm);
168 if (!value.isString()) {
169 throwTypeError(globalObject, scope, "Iterable passed to ListFormat includes non String"_s);
170 return;
171 }
172 String item = value.toWTFString(globalObject);
173 RETURN_IF_EXCEPTION(scope, void());
174 result.append(item);
175 });
176 return result;
177}

179class ListFormatInput {
180 WTF_MAKE_NONCOPYABLE(ListFormatInput);
181public:
182 ListFormatInput(Vector<String, 4>&& strings)
183 : m_strings(WTFMove(strings))
184 {
185 m_stringPointers.reserveInitialCapacity(m_strings.size());
186 m_stringLengths.reserveInitialCapacity(m_strings.size());
187 for (auto& string : m_strings) {
188 if (string.is8Bit()) {
189 auto vector = makeUnique<Vector<UChar>>();
190 vector->resize(string.length());
191 StringImpl::copyCharacters(vector->data(), string.characters8(), string.length());
192 m_retainedUpconvertedStrings.append(WTFMove(vector));
193 m_stringPointers.append(m_retainedUpconvertedStrings.last()->data());
194 } else
195 m_stringPointers.append(string.characters16());
196 m_stringLengths.append(string.length());
197 }
198 }
199
200 int32_t size() const { return m_stringPointers.size(); }
201 const UChar* const* stringPointers() const { return m_stringPointers.data(); }
202 const int32_t* stringLengths() const { return m_stringLengths.data(); }
203
204private:
205 Vector<String, 4> m_strings;
206 Vector<std::unique_ptr<Vector<UChar>>, 4> m_retainedUpconvertedStrings;
207 Vector<const UChar*, 4> m_stringPointers;
208 Vector<int32_t, 4> m_stringLengths;
209};
210#endif

212// https://tc39.es/proposal-intl-list-format/#sec-Intl.ListFormat.prototype.format
213JSValue IntlListFormat::format(JSGlobalObject* globalObject, JSValue list) const
214{
215 VM& vm = globalObject->vm();
216 auto scope = DECLARE_THROW_SCOPE(vm);
217
218#if HAVE(ICU_U_LIST_FORMATTER)
219 auto stringList = stringListFromIterable(globalObject, list);
220 RETURN_IF_EXCEPTION(scope, { });
221
222 ListFormatInput input(WTFMove(stringList));
223
224 Vector<UChar, 32> result;
225 auto status = callBufferProducingFunction(ulistfmt_format, m_listFormat.get(), input.stringPointers(), input.stringLengths(), input.size(), result);
226 if (U_FAILURE(status))
227 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
228
229 return jsString(vm, String(result));
230#else
231 UNUSED_PARAM(list);
232 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
233#endif
234}

236// https://tc39.es/proposal-intl-list-format/#sec-Intl.ListFormat.prototype.formatToParts
237JSValue IntlListFormat::formatToParts(JSGlobalObject* globalObject, JSValue list) const
238{
239 VM& vm = globalObject->vm();
240 auto scope = DECLARE_THROW_SCOPE(vm);
241
242#if HAVE(ICU_U_LIST_FORMATTER)
243 auto stringList = stringListFromIterable(globalObject, list);
244 RETURN_IF_EXCEPTION(scope, { });
245
246 ListFormatInput input(WTFMove(stringList));
247
248 UErrorCode status = U_ZERO_ERROR;
249
250 auto result = std::unique_ptr<UFormattedList, ICUDeleter<ulistfmt_closeResult>>(ulistfmt_openResult(&status));
251 if (U_FAILURE(status))
252 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
253
254 ulistfmt_formatStringsToResult(m_listFormat.get(), input.stringPointers(), input.stringLengths(), input.size(), result.get(), &status);
255 if (U_FAILURE(status))
256 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
257
258 // UFormattedValue is owned by UFormattedList. We do not need to close it.
259 auto formattedValue = ulistfmt_resultAsValue(result.get(), &status);
260 if (U_FAILURE(status))
261 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
262
263 JSArray* parts = JSArray::tryCreate(vm, globalObject->arrayStructureForIndexingTypeDuringAllocation(ArrayWithContiguous), 0);
264 if (!parts)
265 return throwOutOfMemoryError(globalObject, scope);
266
267 int32_t formattedStringLength = 0;
268 const UChar* formattedStringPointer = ufmtval_getString(formattedValue, &formattedStringLength, &status);
269 if (U_FAILURE(status))
270 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
271 String resultString(formattedStringPointer, formattedStringLength);
272
273 auto iterator = std::unique_ptr<UConstrainedFieldPosition, ICUDeleter<ucfpos_close>>(ucfpos_open(&status));
274 if (U_FAILURE(status))
275 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
276
277 ucfpos_constrainField(iterator.get(), UFIELD_CATEGORY_LIST, ULISTFMT_ELEMENT_FIELD, &status);
278 if (U_FAILURE(status))
279 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
280
281 auto literalString = jsNontrivialString(vm, "literal"_s);
282 auto elementString = jsNontrivialString(vm, "element"_s);
283
284 auto createPart = [&] (JSString* type, JSString* value) {
285 JSObject* part = constructEmptyObject(globalObject);
286 part->putDirect(vm, vm.propertyNames->type, type);
287 part->putDirect(vm, vm.propertyNames->value, value);
288 return part;
289 };
290
291 int32_t resultLength = resultString.length();
292 int32_t previousEndIndex = 0;
293 while (true) {
294 bool next = ufmtval_nextPosition(formattedValue, iterator.get(), &status);
295 if (U_FAILURE(status))
296 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
297 if (!next)
298 break;
299
300 int32_t beginIndex = 0;
301 int32_t endIndex = 0;
302 ucfpos_getIndexes(iterator.get(), &beginIndex, &endIndex, &status);
303 if (U_FAILURE(status))
304 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
305
306 if (previousEndIndex < beginIndex) {
307 auto value = jsString(vm, resultString.substring(previousEndIndex, beginIndex - previousEndIndex));
308 JSObject* part = createPart(literalString, value);
309 parts->push(globalObject, part);
310 RETURN_IF_EXCEPTION(scope, { });
311 }
312 previousEndIndex = endIndex;
313
314 auto value = jsString(vm, resultString.substring(beginIndex, endIndex - beginIndex));
315 JSObject* part = createPart(elementString, value);
316 parts->push(globalObject, part);
317 RETURN_IF_EXCEPTION(scope, { });
318 }
319
320 if (previousEndIndex < resultLength) {
321 auto value = jsString(vm, resultString.substring(previousEndIndex, resultLength - previousEndIndex));
322 JSObject* part = createPart(literalString, value);
323 parts->push(globalObject, part);
324 RETURN_IF_EXCEPTION(scope, { });
325 }
326
327 return parts;
328#else
329 UNUSED_PARAM(list);
330 return throwTypeError(globalObject, scope, "failed to format list of strings"_s);
331#endif
332}

334// https://tc39.es/proposal-intl-list-format/#sec-Intl.ListFormat.prototype.resolvedOptions
335JSObject* IntlListFormat::resolvedOptions(JSGlobalObject* globalObject) const
336{
337 VM& vm = globalObject->vm();
338 JSObject* options = constructEmptyObject(globalObject);
339 options->putDirect(vm, vm.propertyNames->locale, jsString(vm, m_locale));
340 options->putDirect(vm, vm.propertyNames->type, jsNontrivialString(vm, typeString(m_type)));
341 options->putDirect(vm, vm.propertyNames->style, jsNontrivialString(vm, styleString(m_style)));
342 return options;
343}

345ASCIILiteral IntlListFormat::styleString(Style style)
346{
347 switch (style) {
