41enum class RelevantExtensionKey : uint8_t;

43struct UListFormatterDeleter {
44 JS_EXPORT_PRIVATE void operator()(UListFormatter*);

47class IntlListFormat final : public JSNonFinalObject {
48public:
49 using Base = JSNonFinalObject;
50
51 static constexpr bool needsDestruction = true;
52
53 static void destroy(JSCell* cell)
54 {
55 static_cast<IntlListFormat*>(cell)->IntlListFormat::~IntlListFormat();
56 }
57
58 template<typename CellType, SubspaceAccess mode>
59 static IsoSubspace* subspaceFor(VM& vm)
60 {
61 return vm.intlListFormatSpace<mode>();
62 }
63
64 static IntlListFormat* create(VM&, Structure*);
65 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
66
67 DECLARE_INFO;
68
69 void initializeListFormat(JSGlobalObject*, JSValue localesValue, JSValue optionsValue);
70
71 JSValue format(JSGlobalObject*, JSValue) const;
72 JSValue formatToParts(JSGlobalObject*, JSValue) const;
73 JSObject* resolvedOptions(JSGlobalObject*) const;
74
75private:
76 IntlListFormat(VM&, Structure*);
77 void finishCreation(VM&);
78
79 enum class Type : uint8_t { Conjunction, Disjunction, Unit };
80 enum class Style : uint8_t { Short, Long, Narrow };
81
82 static ASCIILiteral typeString(Type);
83 static ASCIILiteral styleString(Style);
84
85 std::unique_ptr<UListFormatter, UListFormatterDeleter> m_listFormat;
86 String m_locale;
87 Type m_type { Type::Conjunction };
88 Style m_style { Style::Long };
89};
