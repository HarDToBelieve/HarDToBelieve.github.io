58inline JSCell::JSCell(VM&, Structure* structure)
59 : m_structureID(structure->id())
60 , m_indexingTypeAndMisc(structure->indexingModeIncludingHistory())
61 , m_type(structure->typeInfo().type())
62 , m_flags(structure->typeInfo().inlineTypeFlags())
63 , m_cellState(CellState::DefinitelyWhite)
64{
65 ASSERT(!isCompilationThread());
66
67 // Note that in the constructor initializer list above, we are only using values
68 // inside structure but not necessarily the structure pointer itself. All these
69 // values are contained inside Structure::m_blob. Note also that this constructor
70 // is an inline function. Hence, the compiler may choose to pre-compute the address
71 // of structure->m_blob and discard the structure pointer itself. There's a chance
72 // that the GC may run while allocating this cell. In the event that the structure
73 // is newly instantiated just before calling this constructor, there may not be any
74 // other references to it. As a result, the structure may get collected before this
75 // cell is even constructed. To avoid this possibility, we need to ensure that the
76 // structure pointer is still alive at this point.
77 ensureStillAliveHere(structure);
78}

80inline void JSCell::finishCreation(VM& vm)
81{
82 // This object is ready to be escaped so the concurrent GC may see it at any time. We have
83 // to make sure that none of our stores sink below here.
84 vm.heap.mutatorFence();
85#if ENABLE(GC_VALIDATION)
86 ASSERT(vm.isInitializingObject());
87 vm.setInitializingObjectClass(0);
88#else
89 UNUSED_PARAM(vm);
90#endif
91 ASSERT(m_structureID);
92}
93
94inline void JSCell::finishCreation(VM& vm, Structure* structure, CreatingEarlyCellTag)
95{
96#if ENABLE(GC_VALIDATION)
97 ASSERT(vm.isInitializingObject());
98 vm.setInitializingObjectClass(0);
99 if (structure) {
100#endif
101 m_structureID = structure->id();
102 m_indexingTypeAndMisc = structure->indexingModeIncludingHistory();
103 m_type = structure->typeInfo().type();
104 m_flags = structure->typeInfo().inlineTypeFlags();
105#if ENABLE(GC_VALIDATION)
106 }
107#else
108 UNUSED_PARAM(vm);
109#endif
110 // Very first set of allocations won't have a real structure.
111 ASSERT(m_structureID || !vm.structureStructure);
112}

174template<typename T>
175ALWAYS_INLINE void* tryAllocateCellHelper(Heap& heap, size_t size, GCDeferralContext* deferralContext, AllocationFailureMode failureMode)
176{
177 VM& vm = heap.vm();
178 ASSERT(deferralContext || heap.isDeferred() || !DisallowGC::isInEffectOnCurrentThread());
179 ASSERT(size >= sizeof(T));
180 JSCell* result = static_cast<JSCell*>(subspaceFor<T>(vm)->allocateNonVirtual(vm, size, deferralContext, failureMode));
181 if (failureMode == AllocationFailureMode::ReturnNull && !result)
182 return nullptr;
183#if ENABLE(GC_VALIDATION)
184 ASSERT(!vm.isInitializingObject());
185 vm.setInitializingObjectClass(T::info());
186#endif
187 result->clearStructure();
188 return result;
189}

250// FIXME: Consider making getCallData concurrency-safe once NPAPI support is removed.
251// https://bugs.webkit.org/show_bug.cgi?id=215801
252template<Concurrency concurrency>
253ALWAYS_INLINE TriState JSCell::isCallableWithConcurrency(VM& vm)
254{
255 if (!isObject())
256 return TriState::False;
257 if (type() == JSFunctionType)
258 return TriState::True;
259 if (inlineTypeFlags() & OverridesGetCallData) {
260 if constexpr (concurrency == Concurrency::MainThread)
261 return (methodTable(vm)->getCallData(this).type != CallData::Type::None) ? TriState::True : TriState::False;
262 // We know that InternalFunction::getCallData is concurrency aware. Plus, derived classes of InternalFunction never
263 // override getCallData (this is ensured by ASSERT in InternalFunction).
264 if (type() == InternalFunctionType)
265 return (methodTable(vm)->getCallData(this).type != CallData::Type::None) ? TriState::True : TriState::False;
266 return TriState::Indeterminate;
267 }
268 return TriState::False;
269}
270
271template<Concurrency concurrency>
272inline TriState JSCell::isConstructorWithConcurrency(VM& vm)
273{
274 if (!isObject())
275 return TriState::False;
276 if constexpr (concurrency == Concurrency::MainThread)
277 return (methodTable(vm)->getConstructData(this).type != CallData::Type::None) ? TriState::True : TriState::False;
278 // We know that both getConstructData of both types are concurrency aware. Plus, derived classes of JSFunction and InternalFunction
279 // never override getConstructData (this is ensured by ASSERT in JSFunction and InternalFunction).
280 if (type() == JSFunctionType || type() == InternalFunctionType)
281 return (methodTable(vm)->getConstructData(this).type != CallData::Type::None) ? TriState::True : TriState::False;
282 return TriState::Indeterminate;
283}

285ALWAYS_INLINE bool JSCell::isCallable(VM& vm)
286{
287 auto result = isCallableWithConcurrency<Concurrency::MainThread>(vm);
288 ASSERT(result != TriState::Indeterminate);
289 return result == TriState::True;
290}

304ALWAYS_INLINE void JSCell::setStructure(VM& vm, Structure* structure)
305{
306 ASSERT(structure->classInfo() == this->structure(vm)->classInfo());
307 ASSERT(!this->structure(vm)
308 || this->structure(vm)->transitionWatchpointSetHasBeenInvalidated()
309 || Heap::heap(this)->structureIDTable().get(structure->id()) == structure);
310 m_structureID = structure->id();
311 m_flags = TypeInfo::mergeInlineTypeFlags(structure->typeInfo().inlineTypeFlags(), m_flags);
312 m_type = structure->typeInfo().type();
313 IndexingType newIndexingType = structure->indexingModeIncludingHistory();
314 if (m_indexingTypeAndMisc != newIndexingType) {
315 ASSERT(!(newIndexingType & ~AllArrayTypesAndHistory));
316 for (;;) {
317 IndexingType oldValue = m_indexingTypeAndMisc;
318 IndexingType newValue = (oldValue & ~AllArrayTypesAndHistory) | structure->indexingModeIncludingHistory();
319 if (WTF::atomicCompareExchangeWeakRelaxed(&m_indexingTypeAndMisc, oldValue, newValue))
320 break;
321 }
322 }
323 vm.heap.writeBarrier(this, structure);
324}

347ALWAYS_INLINE JSValue JSCell::fastGetOwnProperty(VM& vm, Structure& structure, PropertyName name)
348{
349 ASSERT(canUseFastGetOwnProperty(structure));
350 PropertyOffset offset = structure.get(vm, name);
351 if (offset != invalidOffset)
352 return asObject(this)->locationForOffset(offset)->get();
353 return JSValue();
354}

356inline bool JSCell::canUseFastGetOwnProperty(const Structure& structure)
357{
358 return !structure.hasGetterSetterProperties()
359 && !structure.hasCustomGetterSetterProperties()
360 && !structure.typeInfo().overridesGetOwnPropertySlot();
361}

363ALWAYS_INLINE const ClassInfo* JSCell::classInfo(VM& vm) const
364{
365 // What we really want to assert here is that we're not currently destructing this object (which makes its classInfo
366 // invalid). If mutatorState() == MutatorState::Running, then we're not currently sweeping, and therefore cannot be
367 // destructing the object. The GC thread or JIT threads, unlike the mutator thread, are able to access classInfo
368 // independent of whether the mutator thread is sweeping or not. Hence, we also check for !currentThreadIsHoldingAPILock()
369 // to allow the GC thread or JIT threads to pass this assertion.
370 ASSERT(vm.heap.mutatorState() != MutatorState::Sweeping || !vm.currentThreadIsHoldingAPILock());
371 return structure(vm)->classInfo();
372}

374inline bool JSCell::toBoolean(JSGlobalObject* globalObject) const
375{
376 if (isString())
377 return static_cast<const JSString*>(this)->toBoolean();
378 if (isHeapBigInt())
379 return static_cast<const JSBigInt*>(this)->toBoolean();
380 return !structure(getVM(globalObject))->masqueradesAsUndefined(globalObject);
381}
382
383inline TriState JSCell::pureToBoolean() const
384{
385 if (isString())
386 return static_cast<const JSString*>(this)->toBoolean() ? TriState::True : TriState::False;
387 if (isHeapBigInt())
388 return static_cast<const JSBigInt*>(this)->toBoolean() ? TriState::True : TriState::False;
389 if (isSymbol())
390 return TriState::True;
391 return TriState::Indeterminate;
392}

394inline void JSCellLock::lock()
395{
396 Atomic<IndexingType>* lock = bitwise_cast<Atomic<IndexingType>*>(&m_indexingTypeAndMisc);
397 if (UNLIKELY(!IndexingTypeLockAlgorithm::lockFast(*lock)))
398 lockSlow();
399}

425inline void JSCell::setPerCellBit(bool value)
426{
427 if (value == perCellBit())
428 return;
429
430 if (value)
431 m_flags |= static_cast<TypeInfo::InlineTypeFlags>(TypeInfoPerCellBit);
432 else
433 m_flags &= ~static_cast<TypeInfo::InlineTypeFlags>(TypeInfoPerCellBit);
434}

443ALWAYS_INLINE bool JSCell::putInline(JSGlobalObject* globalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
444{
445 auto putMethod = methodTable(getVM(globalObject))->put;
446 if (LIKELY(putMethod == JSObject::put))
447 return JSObject::putInlineForJSObject(asObject(this), globalObject, propertyName, value, slot);
448 return putMethod(this, globalObject, propertyName, value, slot);
449}
