56class JSFunction : public JSCallee {
57 friend class JIT;
58 friend class DFG::SpeculativeJIT;
59 friend class DFG::JITCompiler;
60 friend class VM;
61 friend class InternalFunction;
62
63public:
64 static constexpr uintptr_t rareDataTag = 0x1;
65
66 template<typename CellType, SubspaceAccess>
67 static IsoSubspace* subspaceFor(VM& vm)
68 {
69 return &vm.functionSpace;
70 }
71
72 typedef JSCallee Base;
73 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetOwnSpecialPropertyNames | OverridesGetCallData | OverridesPut;
74
75 static size_t allocationSize(Checked<size_t> inlineCapacity)
76 {
77 ASSERT_UNUSED(inlineCapacity, !inlineCapacity);
78 return sizeof(JSFunction);
79 }
80
81 static Structure* selectStructureForNewFuncExp(JSGlobalObject*, FunctionExecutable*);
82
83 JS_EXPORT_PRIVATE static JSFunction* create(VM&, JSGlobalObject*, unsigned length, const String& name, NativeFunction, Intrinsic = NoIntrinsic, NativeFunction nativeConstructor = callHostFunctionAsConstructor, const DOMJIT::Signature* = nullptr);
84
85 static JSFunction* createWithInvalidatedReallocationWatchpoint(VM&, FunctionExecutable*, JSScope*);
86
87 JS_EXPORT_PRIVATE static JSFunction* create(VM&, FunctionExecutable*, JSScope*);
88 static JSFunction* create(VM&, FunctionExecutable*, JSScope*, Structure*);
89
90 JS_EXPORT_PRIVATE String name(VM&);
91 JS_EXPORT_PRIVATE String displayName(VM&);
92 JS_EXPORT_PRIVATE const String calculatedDisplayName(VM&);
93 JS_EXPORT_PRIVATE JSString* toString(JSGlobalObject*);
94
95 JSString* asStringConcurrently(VM&) const;
96
97 ExecutableBase* executable() const
98 {
99 uintptr_t executableOrRareData = m_executableOrRareData;
100 if (executableOrRareData & rareDataTag)
101 return bitwise_cast<FunctionRareData*>(executableOrRareData & ~rareDataTag)->executable();
102 return bitwise_cast<ExecutableBase*>(executableOrRareData);
103 }
104
105 // To call any of these methods include JSFunctionInlines.h
106 bool isHostFunction() const;
107 FunctionExecutable* jsExecutable() const;
108 Intrinsic intrinsic() const;
109
110 JS_EXPORT_PRIVATE const SourceCode* sourceCode() const;
111
112 DECLARE_EXPORT_INFO;
113
114 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
115 {
116 ASSERT(globalObject);
117 return Structure::create(vm, globalObject, prototype, TypeInfo(JSFunctionType, StructureFlags), info());
118 }
119
120 TaggedNativeFunction nativeFunction();
121 TaggedNativeFunction nativeConstructor();
122
123 JS_EXPORT_PRIVATE static CallData getConstructData(JSCell*);
124 JS_EXPORT_PRIVATE static CallData getCallData(JSCell*);
125
126 static inline ptrdiff_t offsetOfExecutableOrRareData()
127 {
128 return OBJECT_OFFSETOF(JSFunction, m_executableOrRareData);
129 }
130
131 FunctionRareData* ensureRareData(VM& vm)
132 {
133 uintptr_t executableOrRareData = m_executableOrRareData;
134 if (UNLIKELY(!(executableOrRareData & rareDataTag)))
135 return allocateRareData(vm);
136 return bitwise_cast<FunctionRareData*>(executableOrRareData & ~rareDataTag);
137 }
138
139 FunctionRareData* ensureRareDataAndAllocationProfile(JSGlobalObject*, unsigned inlineCapacity);
140
141 FunctionRareData* rareData() const
142 {
143 uintptr_t executableOrRareData = m_executableOrRareData;
144 if (executableOrRareData & rareDataTag)
145 return bitwise_cast<FunctionRareData*>(executableOrRareData & ~rareDataTag);
146 return nullptr;
147 }
148
149 bool isHostOrBuiltinFunction() const;
150 bool isBuiltinFunction() const;
151 JS_EXPORT_PRIVATE bool isHostFunctionNonInline() const;
152 bool isClassConstructorFunction() const;
153
154 void setFunctionName(JSGlobalObject*, JSValue name);
155
156 // Returns the __proto__ for the |this| value if this JSFunction were to be constructed.
157 JSObject* prototypeForConstruction(VM&, JSGlobalObject*);
158
159 bool canUseAllocationProfile();
160 bool canUseAllocationProfileNonInline();
161
162 enum class PropertyStatus {
163 Eager,
164 Lazy,
165 Reified,
166 };
167 PropertyStatus reifyLazyPropertyIfNeeded(VM&, JSGlobalObject*, PropertyName);
168
169 bool canAssumeNameAndLengthAreOriginal(VM&);
170
171protected:
172 JS_EXPORT_PRIVATE JSFunction(VM&, NativeExecutable*, JSGlobalObject*, Structure*);
173 JSFunction(VM&, FunctionExecutable*, JSScope*, Structure*);
174
175 void finishCreation(VM&, NativeExecutable*, unsigned length, const String& name);
176 void finishCreation(VM&);
177
178 static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
179 static void getOwnSpecialPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, DontEnumPropertiesMode);
180 static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
181
182 static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
183
184 static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
185
186 DECLARE_VISIT_CHILDREN;
187
188private:
189 static JSFunction* createImpl(VM& vm, FunctionExecutable* executable, JSScope* scope, Structure* structure)
190 {
191 JSFunction* function = new (NotNull, allocateCell<JSFunction>(vm.heap)) JSFunction(vm, executable, scope, structure);
192 ASSERT(function->structure(vm)->globalObject());
193 function->finishCreation(vm);
194 return function;
195 }
196
197 FunctionRareData* allocateRareData(VM&);
198 FunctionRareData* allocateAndInitializeRareData(JSGlobalObject*, size_t inlineCapacity);
199 FunctionRareData* initializeRareData(JSGlobalObject*, size_t inlineCapacity);
200
201 bool hasReifiedLength() const;
202 bool hasReifiedName() const;
203 void reifyLength(VM&);
204 void reifyName(VM&, JSGlobalObject*);
205 void reifyName(VM&, JSGlobalObject*, String name);
206
207 static bool isLazy(PropertyStatus property) { return property == PropertyStatus::Lazy || property == PropertyStatus::Reified; }
208 static bool isReified(PropertyStatus property) { return property == PropertyStatus::Reified; }
209
210 PropertyStatus reifyLazyPropertyForHostOrBuiltinIfNeeded(VM&, JSGlobalObject*, PropertyName);
211 PropertyStatus reifyLazyLengthIfNeeded(VM&, JSGlobalObject*, PropertyName);
212 PropertyStatus reifyLazyNameIfNeeded(VM&, JSGlobalObject*, PropertyName);
213 PropertyStatus reifyLazyBoundNameIfNeeded(VM&, JSGlobalObject*, PropertyName);
214
215#if ASSERT_ENABLED
216 void assertTypeInfoFlagInvariants();
217#else
218 void assertTypeInfoFlagInvariants() { }
219#endif
220
221 friend class LLIntOffsetsExtractor;
222
223 uintptr_t m_executableOrRareData;
224};
