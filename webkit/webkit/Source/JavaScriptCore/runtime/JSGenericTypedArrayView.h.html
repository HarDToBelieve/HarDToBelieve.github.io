46// A typed array view is our representation of a typed array object as seen
47// from JavaScript. For example:
48//
49// var o = new Int8Array(100);
50//
51// Here, 'o' points to a JSGenericTypedArrayView<int8_t>.
52//
53// Views contain five fields:
54//
55// Structure* S // from JSCell
56// Butterfly* B // from JSObject
57// ElementType* V
58// uint32_t L
59// TypedArrayMode M
60//
61// These fields take up a total of four pointer-width words. FIXME: Make
62// it take less words!
63//
64// B is usually unused but may stored some additional "overflow" data for
65// one of the modes. V always points to the base of the typed array's data,
66// and may point to either GC-managed copied space, or data in the C heap;
67// which of those things it points to is governed by the mode although for
68// simple accesses to the view you can just read from the pointer either
69// way. M specifies the mode of the view. L is the length, in units that
70// depend on the view's type.
71
72// The JSGenericTypedArrayView is templatized by an Adaptor that controls
73// the element type and how it's converted; it should obey the following
74// interface; I use int8_t as an example:
75//
76// struct Adaptor {
77// typedef int8_t Type;
78// typedef Int8Array ViewType;
79// typedef JSInt8Array JSViewType;
80// static int8_t toNativeFromInt32(int32_t);
81// static int8_t toNativeFromUint32(uint32_t);
82// static int8_t toNativeFromDouble(double);
83// static JSValue toJSValue(int8_t);
84// template<T> static T::Type convertTo(uint8_t);
85// };
86
87enum class CopyType {
88 LeftToRight,
89 Unobservable,
90};

165 {

183 void sort()
184 {
185 RELEASE_ASSERT(!isDetached());
186 switch (Adaptor::typeValue) {
187 case TypeFloat32:
188 sortFloat<int32_t>();
189 break;
190 case TypeFloat64:
191 sortFloat<int64_t>();
192 break;
193 default: {
194 ElementType* array = typedVector();
195 std::sort(array, array + m_length);
196 break;
197 }
198 }

324 // The ECMA 6 spec states that floating point Typed Arrays should have the following ordering:
325 //
326 // -Inifinity < negative finite numbers < -0.0 < 0.0 < positive finite numbers < Infinity < NaN
327 // Note: regardless of the sign or exact representation of a NaN it is greater than all other values.
328 //
329 // An interesting fact about IEEE 754 floating point numbers is that have an adjacent representation
330 // i.e. for any finite floating point x there does not exist a finite floating point y such that
331 // ((float) ((int) x + 1)) > y > x (where int represents a signed bit integer with the same number
332 // of bits as float). Thus, if we have an array of floating points if we view it as an
333 // array of signed bit integers it will sort in the format we desire. Note, denormal
334 // numbers fit this property as they are floating point numbers with a exponent field of all
335 // zeros so they will be closer to the signed zeros than any normalized number.
336 //
337 // All the processors we support, however, use twos complement. Fortunately, if you compare a signed
338 // bit number as if it were twos complement the result will be correct assuming both numbers are not
339 // negative. e.g.
340 //
341 // - <=> - = reversed (-30 > -20 = true)
342 // + <=> + = ordered (30 > 20 = true)
343 // - <=> + = ordered (-30 > 20 = false)
344 // + <=> - = ordered (30 > -20 = true)
345 //
346 // For NaN, we normalize the NaN to a peticular representation; the sign bit is 0, all exponential bits
347 // are 1 and only the MSB of the mantissa is 1. So, NaN is recognized as the largest integral numbers.
348
349 void purifyArray()
350 {
351 ElementType* array = typedVector();
352 for (unsigned i = 0; i < m_length; i++)
353 array[i] = purifyNaN(array[i]);
354 }
355
356 template<typename IntegralType>
357 void sortFloat()
358 {
359 ASSERT(sizeof(IntegralType) == sizeof(ElementType));
360
361 // Since there might be another view that sets the bits of
362 // our floats to NaNs with negative sign bits we need to
363 // purify the array.
364 // We use a separate function here to avoid the strict aliasing rule.
365 // We could use a union but ASAN seems to frown upon that.
366 purifyArray();
367
368 IntegralType* array = reinterpret_cast_ptr<IntegralType*>(typedVector());
369 std::sort(array, array + m_length, [] (IntegralType a, IntegralType b) {
370 if (a >= 0 || b >= 0)
371 return a < b;
372 return a > b;
373 });
374
375 }

379template<typename Adaptor>
380inline RefPtr<typename Adaptor::ViewType> toPossiblySharedNativeTypedView(VM& vm, JSValue value)
381{
382 typename Adaptor::JSViewType* wrapper = jsDynamicCast<typename Adaptor::JSViewType*>(vm, value);
383 if (!wrapper)
384 return nullptr;
385 return wrapper->possiblySharedTypedImpl();
386}
387
388template<typename Adaptor>
389inline RefPtr<typename Adaptor::ViewType> toUnsharedNativeTypedView(VM& vm, JSValue value)
390{
391 RefPtr<typename Adaptor::ViewType> result = toPossiblySharedNativeTypedView<Adaptor>(vm, value);
392 if (!result || result->isShared())
393 return nullptr;
394 return result;
395}
