40ALWAYS_INLINE bool JSGlobalObject::objectPrototypeIsSane()
41{
42 return !hasIndexedProperties(m_objectPrototype->indexingType())
43 && m_objectPrototype->getPrototypeDirect(vm()).isNull();
44}
45
46ALWAYS_INLINE bool JSGlobalObject::arrayPrototypeChainIsSane()
47{
48 return !hasIndexedProperties(m_arrayPrototype->indexingType())
49 && m_arrayPrototype->getPrototypeDirect(vm()) == m_objectPrototype.get()
50 && objectPrototypeIsSane();
51}
52
53ALWAYS_INLINE bool JSGlobalObject::stringPrototypeChainIsSane()
54{
55 return !hasIndexedProperties(m_stringPrototype->indexingType())
56 && m_stringPrototype->getPrototypeDirect(vm()) == m_objectPrototype.get()
57 && objectPrototypeIsSane();
58}

60ALWAYS_INLINE bool JSGlobalObject::isArrayPrototypeIteratorProtocolFastAndNonObservable()
61{
62 // We're fast if we don't have any indexed properties on the prototype.
63 // We're non-observable if the iteration protocol hasn't changed.
64 //
65 // Note: it only makes sense to call this from the main thread. If you're
66 // trying to prove this behavior on the compiler thread, you'll want to
67 // carefully set up watchpoints to have correct ordering while JS code is
68 // executing concurrently.
69
70 return arrayIteratorProtocolWatchpointSet().isStillValid() && !isHavingABadTime() && arrayPrototypeChainIsSane();
71}
72
73// We're non-observable if the iteration protocol hasn't changed.
74//
75// Note: it only makes sense to call this from the main thread. If you're
76// trying to prove this behavior on the compiler thread, you'll want to
77// carefully set up watchpoints to have correct ordering while JS code is
78// executing concurrently.
79ALWAYS_INLINE bool JSGlobalObject::isMapPrototypeIteratorProtocolFastAndNonObservable()
80{
81 return mapIteratorProtocolWatchpointSet().isStillValid();
82}

104ALWAYS_INLINE Structure* JSGlobalObject::arrayStructureForIndexingTypeDuringAllocation(JSGlobalObject* globalObject, IndexingType indexingType, JSValue newTarget) const
105{
106 VM& vm = globalObject->vm();
107 auto scope = DECLARE_THROW_SCOPE(vm);
108 if (!newTarget || newTarget == globalObject->arrayConstructor())
109 return globalObject->arrayStructureForIndexingTypeDuringAllocation(indexingType);
110 auto* functionGlobalObject = getFunctionRealm(globalObject, asObject(newTarget));
111 RETURN_IF_EXCEPTION(scope, nullptr);
112 RELEASE_AND_RETURN(scope, InternalFunction::createSubclassStructure(globalObject, asObject(newTarget), functionGlobalObject->arrayStructureForIndexingTypeDuringAllocation(indexingType)));
113}

129template<typename T>
130inline unsigned JSGlobalObject::WeakCustomGetterOrSetterHash<T>::hash(const Weak<T>& value)
131{
132 if (!value)
133 return 0;
134 return hash(value->propertyName(), value->customFunctionPointer());
135}
136
137template<typename T>
138inline bool JSGlobalObject::WeakCustomGetterOrSetterHash<T>::equal(const Weak<T>& a, const Weak<T>& b)
139{
140 if (!a || !b)
141 return false;
142 return a == b;
143}
144
145template<typename T>
146inline unsigned JSGlobalObject::WeakCustomGetterOrSetterHash<T>::hash(const PropertyName& propertyName, typename T::CustomFunctionPointer functionPointer)
147{
148 unsigned hash = DefaultHash<typename T::CustomFunctionPointer>::hash(functionPointer);
149 if (!propertyName.isNull())
150 hash = WTF::pairIntHash(hash, propertyName.uid()->existingSymbolAwareHash());
151 return hash;
152}
