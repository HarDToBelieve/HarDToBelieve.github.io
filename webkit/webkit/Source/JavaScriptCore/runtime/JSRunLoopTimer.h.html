26#pragma once
27
28#include <wtf/HashSet.h>
29#include <wtf/Lock.h>
30#include <wtf/RefPtr.h>
31#include <wtf/RetainPtr.h>
32#include <wtf/RunLoop.h>
33#include <wtf/SharedTask.h>
34#include <wtf/ThreadSafeRefCounted.h>
35
36namespace JSC {
37
38class JSLock;
39class VM;
40
41class JSRunLoopTimer : public ThreadSafeRefCounted<JSRunLoopTimer> {
42public:
43 typedef void TimerNotificationType();
44 using TimerNotificationCallback = RefPtr<WTF::SharedTask<TimerNotificationType>>;
45
46 class Manager {
47 WTF_MAKE_FAST_ALLOCATED;
48 WTF_MAKE_NONCOPYABLE(Manager);
49 void timerDidFireCallback();
50
51 Manager() = default;
52
53 void timerDidFire();
54
55 public:
56 using EpochTime = Seconds;
57
58 static Manager& shared();
59 void registerVM(VM&);
60 void unregisterVM(VM&);
61 void scheduleTimer(JSRunLoopTimer&, Seconds nextFireTime);
62 void cancelTimer(JSRunLoopTimer&);
63
64 std::optional<Seconds> timeUntilFire(JSRunLoopTimer&);
65
66 private:
67 Lock m_lock;
68
69 class PerVMData {
70 WTF_MAKE_FAST_ALLOCATED;
71 WTF_MAKE_NONCOPYABLE(PerVMData);
72 public:
73 PerVMData(Manager&, WTF::RunLoop&);
74 ~PerVMData();
75
76 Ref<WTF::RunLoop> runLoop;
77 std::unique_ptr<RunLoop::Timer<Manager>> timer;
78 Vector<std::pair<Ref<JSRunLoopTimer>, EpochTime>> timers;
79 };
80
81 HashMap<Ref<JSLock>, std::unique_ptr<PerVMData>> m_mapping;
82 };
83
84 JSRunLoopTimer(VM&);
85 JS_EXPORT_PRIVATE virtual ~JSRunLoopTimer();
86 virtual void doWork(VM&) = 0;
87
88 void setTimeUntilFire(Seconds intervalInSeconds);
89 void cancelTimer();
90 bool isScheduled() const { return m_isScheduled; }
91
92 // Note: The only thing the timer notification callback cannot do is
93 // call setTimeUntilFire(). This will cause a deadlock. It would not
94 // be hard to make this work, however, there are no clients that need
95 // this behavior. We should implement it only if we find that we need it.
96 JS_EXPORT_PRIVATE void addTimerSetNotification(TimerNotificationCallback);
97 JS_EXPORT_PRIVATE void removeTimerSetNotification(TimerNotificationCallback);
98
99 JS_EXPORT_PRIVATE std::optional<Seconds> timeUntilFire();
100
101protected:
102 static constexpr Seconds s_decade { 60 * 60 * 24 * 365 * 10 };
103 Ref<JSLock> m_apiLock;
104
105private:
106 friend class Manager;
107
108 void timerDidFire();
109
110 HashSet<TimerNotificationCallback> m_timerSetCallbacks;
111 Lock m_timerCallbacksLock;
112
113 Lock m_lock;
114 bool m_isScheduled { false };
115};
116
117} // namespace JSC
