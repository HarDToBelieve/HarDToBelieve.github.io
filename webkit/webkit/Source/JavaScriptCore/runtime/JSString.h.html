70// In 64bit architecture, JSString and JSRopeString have the following memory layout to make sizeof(JSString) == 16 and sizeof(JSRopeString) == 32.
71// JSString has only one pointer. We use it for String. length() and is8Bit() queries go to StringImpl. In JSRopeString, we reuse the above pointer
72// place for the 1st fiber. JSRopeString has three fibers so its size is 48. To keep length and is8Bit flag information in JSRopeString, JSRopeString
73// encodes these information into the fiber pointers. is8Bit flag is encoded in the 1st fiber pointer. length is embedded directly, and two fibers
74// are compressed into 12bytes. isRope information is encoded in the first fiber's LSB.
75//
76// Since length of JSRopeString should be frequently accessed compared to each fiber, we put length in contiguous 32byte field, and compress 2nd
77// and 3rd fibers into the following 80byte fields. One problem is that now 2nd and 3rd fibers are split. Storing and loading 2nd and 3rd fibers
78// are not one pointer load operation. To make concurrent collector work correctly, we must initialize 2nd and 3rd fibers at JSRopeString creation
79// and we must not modify these part later.
80//
81// 0 8 10 16 20 24 26 28 32
82// JSString [ ID ][ header ][ String pointer 0]
83// JSRopeString [ ID ][ header ][ 1st fiber xyz][ length ][2nd lower32][2nd upper16][3rd lower16][3rd upper32]
84// ^
85// x:(is8Bit),y:(isSubstring),z:(isRope) bit flags

379 template <class OverflowHandler = CrashOnOverflow>
380 class RopeBuilder : public OverflowHandler {
381 WTF_FORBID_HEAP_ALLOCATION;
382 public:
383 RopeBuilder(VM& vm)
384 : m_vm(vm)
385 {
386 }
387
388 bool append(JSString* jsString)
389 {
390 if (UNLIKELY(this->hasOverflowed()))
391 return false;
392 if (!jsString->length())
393 return true;
394 if (m_strings.size() == JSRopeString::s_maxInternalRopeLength)
395 expand();
396
397 static_assert(JSString::MaxLength == std::numeric_limits<int32_t>::max(), "");
398 auto sum = checkedSum<int32_t>(m_length, jsString->length());
399 if (sum.hasOverflowed()) {
400 this->overflowed();
401 return false;
402 }
403 ASSERT(static_cast<unsigned>(sum) <= MaxLength);
404 m_strings.append(jsString);
405 m_length = static_cast<unsigned>(sum);
406 return true;
407 }
408
409 JSString* release()
410 {
411 RELEASE_ASSERT(!this->hasOverflowed());
412 JSString* result = nullptr;
413 switch (m_strings.size()) {
414 case 0: {
415 ASSERT(!m_length);
416 result = jsEmptyString(m_vm);
417 break;
418 }
419 case 1: {
420 result = asString(m_strings.at(0));
421 break;
422 }
423 case 2: {
424 result = JSRopeString::create(m_vm, asString(m_strings.at(0)), asString(m_strings.at(1)));
425 break;
426 }
427 case 3: {
428 result = JSRopeString::create(m_vm, asString(m_strings.at(0)), asString(m_strings.at(1)), asString(m_strings.at(2)));
429 break;
430 }
431 default:
432 ASSERT_NOT_REACHED();
433 break;
434 }
435 ASSERT(result->length() == m_length);
436 m_strings.clear();
437 m_length = 0;
438 return result;
439 }
440
441 unsigned length() const
442 {
443 ASSERT(!this->hasOverflowed());
444 return m_length;
445 }
446
447 private:
448 void expand();
449
450 VM& m_vm;
451 MarkedArgumentBuffer m_strings;
452 unsigned m_length { 0 };
453 };

552 // This JSRopeString is only used to simulate half-baked JSRopeString in DFG and FTL MakeRope. If OSR exit happens in
553 // the middle of MakeRope due to string length overflow, we have half-baked JSRopeString which is the same to the result
554 // of this function. This half-baked JSRopeString will not be exposed to users, but still collectors can see it due to
555 // the conservative stack scan. This JSRopeString is used to test the collector with such a half-baked JSRopeString.
556 // Because this JSRopeString breaks the JSString's invariant (only one singleton JSString can be zero length), almost all the
557 // operations in JS fail to handle this string correctly.

560 JSRopeString* newString = new (NotNull, allocateCell<JSRopeString>(vm.heap)) JSRopeString(vm);

563 ASSERT(newString->isRope());

691// JSString::is8Bit is safe to be called concurrently. Concurrent threads can access is8Bit even if the main thread
692// is in the middle of converting JSRopeString to JSString.
693ALWAYS_INLINE bool JSString::is8Bit() const
694{
695 uintptr_t pointer = m_fiber;
696 if (pointer & isRopeInPointer) {
697 // Do not load m_fiber twice. We should use the information in pointer.
698 // Otherwise, JSRopeString may be converted to JSString between the first and second accesses.
699 return pointer & JSRopeString::is8BitInPointer;
700 }
701 return bitwise_cast<StringImpl*>(pointer)->is8Bit();
702}
703
704// JSString::length is safe to be called concurrently. Concurrent threads can access length even if the main thread
705// is in the middle of converting JSRopeString to JSString. This is OK because we never override the length bits
706// when we resolve a JSRopeString.
707ALWAYS_INLINE unsigned JSString::length() const
708{
709 uintptr_t pointer = m_fiber;
710 if (pointer & isRopeInPointer)
711 return jsCast<const JSRopeString*>(this)->length();
712 return bitwise_cast<StringImpl*>(pointer)->length();
713}

837inline JSString* jsSubstring(VM& vm, JSGlobalObject* globalObject, JSString* base, unsigned offset, unsigned length)
838{
839 auto scope = DECLARE_THROW_SCOPE(vm);
840
841 ASSERT(offset <= base->length());
842 ASSERT(length <= base->length());
843 ASSERT(offset + length <= base->length());
844 if (!length)
845 return vm.smallStrings.emptyString();
846 if (!offset && length == base->length())
847 return base;
848
849 // For now, let's not allow substrings with a rope base.
850 // Resolve non-substring rope bases so we don't have to deal with it.
851 // FIXME: Evaluate if this would be worth adding more branches.
852 if (base->isSubstring()) {
853 JSRopeString* baseRope = jsCast<JSRopeString*>(base);
854 base = baseRope->substringBase();
855 offset = baseRope->substringOffset() + offset;
856 ASSERT(!base->isRope());
857 } else if (base->isRope()) {
858 jsCast<JSRopeString*>(base)->resolveRope(globalObject);
859 RETURN_IF_EXCEPTION(scope, nullptr);
860 }
861 return jsSubstringOfResolved(vm, nullptr, base, offset, length);
862}

864inline JSString* jsSubstringOfResolved(VM& vm, GCDeferralContext* deferralContext, JSString* s, unsigned offset, unsigned length)
865{
866 ASSERT(offset <= s->length());
867 ASSERT(length <= s->length());
868 ASSERT(offset + length <= s->length());
869 ASSERT(!s->isRope());
870 if (!length)
871 return vm.smallStrings.emptyString();
872 if (!offset && length == s->length())
873 return s;
874 if (length == 1) {
875 auto& base = s->valueInternal();
876 UChar character = base.characterAt(offset);
877 if (character <= maxSingleCharacterString)
878 return vm.smallStrings.singleCharacterString(character);
879 }
880 return JSRopeString::createSubstringOfResolved(vm, deferralContext, s, offset, length);
881}

924ALWAYS_INLINE JSString* jsStringWithCache(VM& vm, const String& s)
925{
926 StringImpl* stringImpl = s.impl();
927 if (!stringImpl || !stringImpl->length())
928 return jsEmptyString(vm);
929
930 if (stringImpl->length() == 1) {
931 UChar singleCharacter = (*stringImpl)[0u];
932 if (singleCharacter <= maxSingleCharacterString)
933 return vm.smallStrings.singleCharacterString(static_cast<unsigned char>(singleCharacter));
934 }
935
936 if (JSString* lastCachedString = vm.lastCachedString.get()) {
937 if (lastCachedString->tryGetValueImpl() == stringImpl)
938 return lastCachedString;
939 }
940
941 return jsStringWithCacheSlowCase(vm, *stringImpl);
942}

944ALWAYS_INLINE bool JSString::getStringPropertySlot(JSGlobalObject* globalObject, PropertyName propertyName, PropertySlot& slot)
945{
946 VM& vm = getVM(globalObject);
947 auto scope = DECLARE_THROW_SCOPE(vm);
948
949 if (propertyName == vm.propertyNames->length) {
950 slot.setValue(this, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly, jsNumber(length()));
951 return true;
952 }
953
954 std::optional<uint32_t> index = parseIndex(propertyName);
955 if (index && index.value() < length()) {
956 JSValue value = getIndex(globalObject, index.value());
957 RETURN_IF_EXCEPTION(scope, false);
958 slot.setValue(this, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly, value);
959 return true;
960 }
961
962 return false;
963}
