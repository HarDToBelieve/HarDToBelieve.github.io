21#pragma once
22
23#include "InternalFunction.h"
24#include "MathCommon.h"
25
26namespace JSC {
27
28class NumberPrototype;
29class GetterSetter;
30
31class NumberConstructor final : public InternalFunction {
32public:
33 typedef InternalFunction Base;
34 static constexpr unsigned StructureFlags = Base::StructureFlags | ImplementsHasInstance | HasStaticPropertyTable;
35
36 static NumberConstructor* create(VM& vm, Structure* structure, NumberPrototype* numberPrototype, GetterSetter*)
37 {
38 NumberConstructor* constructor = new (NotNull, allocateCell<NumberConstructor>(vm.heap)) NumberConstructor(vm, structure);
39 constructor->finishCreation(vm, numberPrototype);
40 return constructor;
41 }
42
43 DECLARE_INFO;
44
45 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
46 {
47 return Structure::create(vm, globalObject, proto, TypeInfo(InternalFunctionType, StructureFlags), info());
48 }
49
50 static bool isIntegerImpl(JSValue value)
51 {
52 return value.isInt32() || (value.isDouble() && isInteger(value.asDouble()));
53 }
54
55private:
56 NumberConstructor(VM&, Structure*);
57 void finishCreation(VM&, NumberPrototype*);
58};
59STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(NumberConstructor, InternalFunction);
60
61} // namespace JSC
