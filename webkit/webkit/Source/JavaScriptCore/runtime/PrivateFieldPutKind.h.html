27#pragma once

29namespace WTF {
30class PrintStream;
31};
32
33namespace JSC {
34
35struct PrivateFieldPutKind {
36 static constexpr uint8_t None = 0;
37 static constexpr uint8_t Set = 1;
38 static constexpr uint8_t Define = 2;
39
40public:
41 static constexpr PrivateFieldPutKind fromByte(uint8_t byte) { return PrivateFieldPutKind(byte); }
42 static constexpr PrivateFieldPutKind none() { return PrivateFieldPutKind(None); }
43 static constexpr PrivateFieldPutKind set() { return PrivateFieldPutKind(Set); }
44 static constexpr PrivateFieldPutKind define() { return PrivateFieldPutKind(Define); }
45
46 ALWAYS_INLINE bool isNone() const { return m_value == None; }
47 ALWAYS_INLINE bool isSet() const { return m_value == Set; }
48 ALWAYS_INLINE bool isDefine() const { return m_value == Define; }
49 ALWAYS_INLINE uint8_t value() const { return m_value; }
50
51 void dump(WTF::PrintStream&) const;
52
53private:
54 constexpr PrivateFieldPutKind(uint8_t value)
55 : m_value(value)
56 {
57 ASSERT(m_value == None || m_value == Set || m_value == Define, m_value);
58 }
59
60 uint8_t m_value;
61};
62
63} // namespace JSC
