32namespace JSC {
33
34class GetterSetter;
35
36class PropertyDescriptor {
37public:
38 PropertyDescriptor()
39 : m_attributes(defaultAttributes)
40 , m_seenAttributes(0)
41 {
42 }
43 PropertyDescriptor(JSValue value, unsigned attributes)
44 : m_value(value)
45 , m_attributes(attributes)
46 , m_seenAttributes(EnumerablePresent | ConfigurablePresent | WritablePresent)
47 {
48 ASSERT(m_value);
49 ASSERT(!m_value.isGetterSetter());
50 ASSERT(!m_value.isCustomGetterSetter());
51 }
52 JS_EXPORT_PRIVATE bool writable() const;
53 JS_EXPORT_PRIVATE bool enumerable() const;
54 JS_EXPORT_PRIVATE bool configurable() const;
55 JS_EXPORT_PRIVATE bool isDataDescriptor() const;
56 bool isGenericDescriptor() const;
57 JS_EXPORT_PRIVATE bool isAccessorDescriptor() const;
58 unsigned attributes() const { return m_attributes; }
59 JSValue value() const { return m_value; }
60 GetterSetter* slowGetterSetter(JSGlobalObject*) const; // Be aware that this will lazily allocate a GetterSetter object. It's much better to use getter() and setter() individually if possible.
61 JS_EXPORT_PRIVATE JSValue getter() const;
62 JS_EXPORT_PRIVATE JSValue setter() const;
63 JSObject* getterObject() const;
64 JSObject* setterObject() const;
65 JS_EXPORT_PRIVATE void setUndefined();
66 JS_EXPORT_PRIVATE void setDescriptor(JSValue, unsigned attributes);
67 JS_EXPORT_PRIVATE void setAccessorDescriptor(unsigned attributes);
68 JS_EXPORT_PRIVATE void setAccessorDescriptor(GetterSetter* accessor, unsigned attributes);
69 JS_EXPORT_PRIVATE void setWritable(bool);
70 JS_EXPORT_PRIVATE void setEnumerable(bool);
71 JS_EXPORT_PRIVATE void setConfigurable(bool);
72 void setValue(JSValue value) { m_value = value; }
73 JS_EXPORT_PRIVATE void setSetter(JSValue);
74 JS_EXPORT_PRIVATE void setGetter(JSValue);
75 bool isEmpty() const { return !(m_value || m_getter || m_setter || m_seenAttributes); }
76 bool writablePresent() const { return m_seenAttributes & WritablePresent; }
77 bool enumerablePresent() const { return m_seenAttributes & EnumerablePresent; }
78 bool configurablePresent() const { return m_seenAttributes & ConfigurablePresent; }
79 bool setterPresent() const { return !!m_setter; }
80 bool getterPresent() const { return !!m_getter; }
81 bool equalTo(JSGlobalObject*, const PropertyDescriptor& other) const;
82 bool attributesEqual(const PropertyDescriptor& other) const;
83 unsigned attributesOverridingCurrent(const PropertyDescriptor& current) const;
84
85private:
86 static constexpr unsigned defaultAttributes = PropertyAttribute::DontDelete | PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly;
87
88 bool operator==(const PropertyDescriptor&) { return false; }
89 enum { WritablePresent = 1, EnumerablePresent = 2, ConfigurablePresent = 4};
90 // May be a getter/setter
91 JSValue m_value;
92 JSValue m_getter;
93 JSValue m_setter;
94 unsigned m_attributes;
95 unsigned m_seenAttributes;
96};
97
98inline PropertyDescriptor toPropertyDescriptor(JSValue value, JSValue getter, JSValue setter, DefinePropertyAttributes attributes)
99{
100 // We assume that validation is already done.
101 PropertyDescriptor desc;
102
103 if (std::optional<bool> enumerable = attributes.enumerable())
104 desc.setEnumerable(enumerable.value());
105
106 if (std::optional<bool> configurable = attributes.configurable())
107 desc.setConfigurable(configurable.value());
108
109 if (attributes.hasValue())
110 desc.setValue(value);
111
112 if (std::optional<bool> writable = attributes.writable())
113 desc.setWritable(writable.value());
114
115 if (attributes.hasGet())
116 desc.setGetter(getter);
117
118 if (attributes.hasSet())
119 desc.setSetter(setter);
120
121 return desc;
122}
