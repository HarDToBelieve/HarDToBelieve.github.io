41class RegExp final : public JSCell {
42 friend class CachedRegExp;
43
44public:
45 using Base = JSCell;
46 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
47 static constexpr bool needsDestruction = true;
48
49 template<typename CellType, SubspaceAccess mode>
50 static IsoSubspace* subspaceFor(VM& vm)
51 {
52 return &vm.regExpSpace;
53 }
54
55 JS_EXPORT_PRIVATE static RegExp* create(VM&, const String& pattern, OptionSet<Yarr::Flags>);
56 static void destroy(JSCell*);
57 static size_t estimatedSize(JSCell*, VM&);
58 JS_EXPORT_PRIVATE static void dumpToStream(const JSCell*, PrintStream&);
59
60 bool global() const { return m_flags.contains(Yarr::Flags::Global); }
61 bool ignoreCase() const { return m_flags.contains(Yarr::Flags::IgnoreCase); }
62 bool multiline() const { return m_flags.contains(Yarr::Flags::Multiline); }
63 bool sticky() const { return m_flags.contains(Yarr::Flags::Sticky); }
64 bool globalOrSticky() const { return global() || sticky(); }
65 bool unicode() const { return m_flags.contains(Yarr::Flags::Unicode); }
66 bool dotAll() const { return m_flags.contains(Yarr::Flags::DotAll); }
67 bool hasIndices() const { return m_flags.contains(Yarr::Flags::HasIndices); }
68
69 const String& pattern() const { return m_patternString; }
70
71 bool isValid() const { return !Yarr::hasError(m_constructionErrorCode); }
72 const char* errorMessage() const { return Yarr::errorMessage(m_constructionErrorCode); }
73 JSObject* errorToThrow(JSGlobalObject* globalObject) { return Yarr::errorToThrow(globalObject, m_constructionErrorCode); }
74 void reset()
75 {
76 m_state = NotCompiled;
77 m_constructionErrorCode = Yarr::ErrorCode::NoError;
78 }
79
80 JS_EXPORT_PRIVATE int match(JSGlobalObject*, const String&, unsigned startOffset, Vector<int>& ovector);
81
82 // Returns false if we couldn't run the regular expression for any reason.
83 bool matchConcurrently(VM&, const String&, unsigned startOffset, int& position, Vector<int>& ovector);
84
85 JS_EXPORT_PRIVATE MatchResult match(JSGlobalObject*, const String&, unsigned startOffset);
86
87 bool matchConcurrently(VM&, const String&, unsigned startOffset, MatchResult&);
88
89 // Call these versions of the match functions if you're desperate for performance.
90 template<typename VectorType, Yarr::MatchFrom thread = Yarr::MatchFrom::VMThread>
91 int matchInline(JSGlobalObject* nullOrGlobalObject, VM&, const String&, unsigned startOffset, VectorType& ovector);
92 template<Yarr::MatchFrom thread = Yarr::MatchFrom::VMThread>
93 MatchResult matchInline(JSGlobalObject* nullOrGlobalObject, VM&, const String&, unsigned startOffset);
94
95 unsigned numSubpatterns() const { return m_numSubpatterns; }
96
97 bool hasNamedCaptures()
98 {
99 return m_rareData && !m_rareData->m_captureGroupNames.isEmpty();
100 }
101
102 String getCaptureGroupName(unsigned i)
103 {
104 if (!i || !m_rareData || m_rareData->m_captureGroupNames.size() <= i)
105 return String();
106 ASSERT(m_rareData);
107 return m_rareData->m_captureGroupNames[i];
108 }
109
110 unsigned subpatternForName(String groupName)
111 {
112 if (!m_rareData)
113 return 0;
114 auto it = m_rareData->m_namedGroupToParenIndex.find(groupName);
115 if (it == m_rareData->m_namedGroupToParenIndex.end())
116 return 0;
117 return it->value;
118 }
119
120 bool hasCode()
121 {
122 return m_state == JITCode || m_state == ByteCode;
123 }
124
125 bool hasCodeFor(Yarr::YarrCharSize);
126 bool hasMatchOnlyCodeFor(Yarr::YarrCharSize);
127
128 void deleteCode();
129
130#if ENABLE(REGEXP_TRACING)
131 void printTraceData();
132#endif
133
134 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
135 {
136 return Structure::create(vm, globalObject, prototype, TypeInfo(CellType, StructureFlags), info());
137 }
138
139 DECLARE_INFO;
140
141 RegExpKey key() { return RegExpKey(m_flags, m_patternString); }
142
143private:
144 friend class RegExpCache;
145 RegExp(VM&, const String&, OptionSet<Yarr::Flags>);
146 void finishCreation(VM&);
147
148 static RegExp* createWithoutCaching(VM&, const String&, OptionSet<Yarr::Flags>);
149
150 enum RegExpState : uint8_t {
151 ParseError,
152 JITCode,
153 ByteCode,
154 NotCompiled
155 };
156
157 void byteCodeCompileIfNecessary(VM*);
158
159 void compile(VM*, Yarr::YarrCharSize);
160 void compileIfNecessary(VM&, Yarr::YarrCharSize);
161
162 void compileMatchOnly(VM*, Yarr::YarrCharSize);
163 void compileIfNecessaryMatchOnly(VM&, Yarr::YarrCharSize);
164
165#if ENABLE(YARR_JIT_DEBUG)
166 void matchCompareWithInterpreter(const String&, int startOffset, int* offsetVector, int jitResult);
167#endif
168
169#if ENABLE(YARR_JIT)
170 Yarr::YarrCodeBlock& ensureRegExpJITCode()
171 {
172 if (!m_regExpJITCode)
173 m_regExpJITCode = makeUnique<Yarr::YarrCodeBlock>();
174 return *m_regExpJITCode.get();
175 }
176#endif
177
178 struct RareData {
179 WTF_MAKE_STRUCT_FAST_ALLOCATED;
180 Vector<String> m_captureGroupNames;
181 HashMap<String, unsigned> m_namedGroupToParenIndex;
182 };
183
184 String m_patternString;
185 RegExpState m_state { NotCompiled };
186 OptionSet<Yarr::Flags> m_flags;
187 Yarr::ErrorCode m_constructionErrorCode { Yarr::ErrorCode::NoError };
188 unsigned m_numSubpatterns { 0 };
189 std::unique_ptr<Yarr::BytecodePattern> m_regExpBytecode;
190#if ENABLE(YARR_JIT)
191 std::unique_ptr<Yarr::YarrCodeBlock> m_regExpJITCode;
192#endif
193 std::unique_ptr<RareData> m_rareData;
194#if ENABLE(REGEXP_TRACING)
195 double m_rtMatchOnlyTotalSubjectStringLen { 0.0 };
196 double m_rtMatchTotalSubjectStringLen { 0.0 };
197 unsigned m_rtMatchOnlyCallCount { 0 };
198 unsigned m_rtMatchOnlyFoundCount { 0 };
199 unsigned m_rtMatchCallCount { 0 };
200 unsigned m_rtMatchFoundCount { 0 };
201#endif
202};
