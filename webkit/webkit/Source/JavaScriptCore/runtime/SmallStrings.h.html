50namespace JSC {
51
52class VM;
53class JSString;
54
55static constexpr unsigned maxSingleCharacterString = 0xFF;
56
57class SmallStrings {
58 WTF_MAKE_NONCOPYABLE(SmallStrings);
59public:
60 SmallStrings();
61 ~SmallStrings();
62
63 JSString* emptyString()
64 {
65 return m_emptyString;
66 }
67
68 JSString* singleCharacterString(unsigned char character)
69 {
70 return m_singleCharacterStrings[character];
71 }
72
73 JS_EXPORT_PRIVATE Ref<AtomStringImpl> singleCharacterStringRep(unsigned char character);
74
75 void setIsInitialized(bool isInitialized) { m_isInitialized = isInitialized; }
76
77 JSString** singleCharacterStrings() { return &m_singleCharacterStrings[0]; }
78
79 void initializeCommonStrings(VM&);
80 template<typename Visitor> void visitStrongReferences(Visitor&);
81
82#define JSC_COMMON_STRINGS_ACCESSOR_DEFINITION(name) \
83 JSString* name##String() const \
84 { \
85 return m_##name; \
86 }
87 JSC_COMMON_STRINGS_EACH_NAME(JSC_COMMON_STRINGS_ACCESSOR_DEFINITION)
88#undef JSC_COMMON_STRINGS_ACCESSOR_DEFINITION
89
90 JSString* typeString(TypeofType type) const
91 {
92 switch (type) {
93 case TypeofType::Undefined:
94 return undefinedString();
95 case TypeofType::Boolean:
96 return booleanString();
97 case TypeofType::Number:
98 return numberString();
99 case TypeofType::String:
100 return stringString();
101 case TypeofType::Symbol:
102 return symbolString();
103 case TypeofType::Object:
104 return objectString();
105 case TypeofType::Function:
106 return functionString();
107 case TypeofType::BigInt:
108 return bigintString();
109 }
110
111 RELEASE_ASSERT_NOT_REACHED();
112 return nullptr;
113 }
114
115 JSString* objectStringStart() const { return m_objectStringStart; }
116 JSString* nullObjectString() const { return m_nullObjectString; }
117 JSString* undefinedObjectString() const { return m_undefinedObjectString; }
118 JSString* boundPrefixString() const { return m_boundPrefixString; }
119 JSString* notEqualString() const { return m_notEqualString; }
120 JSString* timedOutString() const { return m_timedOutString; }
121 JSString* okString() const { return m_okString; }
122
123 bool needsToBeVisited(CollectionScope scope) const
124 {
125 if (scope == CollectionScope::Full)
126 return true;
127 return m_needsToBeVisited;
128 }
129
130private:
131 static constexpr unsigned singleCharacterStringCount = maxSingleCharacterString + 1;
132
133 void initialize(VM*, JSString*&, const char* value);
134
135 JSString* m_emptyString { nullptr };
136#define JSC_COMMON_STRINGS_ATTRIBUTE_DECLARATION(name) JSString* m_##name { nullptr };
137 JSC_COMMON_STRINGS_EACH_NAME(JSC_COMMON_STRINGS_ATTRIBUTE_DECLARATION)
138#undef JSC_COMMON_STRINGS_ATTRIBUTE_DECLARATION
139 JSString* m_objectStringStart { nullptr };
140 JSString* m_nullObjectString { nullptr };
141 JSString* m_undefinedObjectString { nullptr };
142 JSString* m_boundPrefixString { nullptr };
143 JSString* m_notEqualString { nullptr };
144 JSString* m_timedOutString { nullptr };
145 JSString* m_okString { nullptr };
146 JSString* m_singleCharacterStrings[singleCharacterStringCount] { nullptr };
147 bool m_needsToBeVisited { true };
148 bool m_isInitialized { false };
149};
150
151} // namespace JSC
