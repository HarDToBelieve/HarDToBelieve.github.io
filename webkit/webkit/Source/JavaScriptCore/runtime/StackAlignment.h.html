32namespace JSC {
33
34// NB. Different platforms may have different requirements here. But 16 bytes is very common.
35constexpr unsigned stackAlignmentBytes() { return 16; }
36
37constexpr unsigned stackAlignmentRegisters()
38{
39 return stackAlignmentBytes() / sizeof(EncodedJSValue);
40}
41
42// Align argument count taking into account the CallFrameHeaderSize may be
43// an "unaligned" count of registers.
44inline unsigned roundArgumentCountToAlignFrame(unsigned argumentCount)
45{
46 return WTF::roundUpToMultipleOf(stackAlignmentRegisters(), argumentCount + CallFrame::headerSizeInRegisters) - CallFrame::headerSizeInRegisters;
47}
48
49// Align local register count to make the last local end on a stack aligned address given the
50// CallFrame is at an address that is stack aligned minus CallerFrameAndPC::sizeInRegisters
51inline unsigned roundLocalRegisterCountForFramePointerOffset(unsigned localRegisterCount)
52{
53 return WTF::roundUpToMultipleOf(stackAlignmentRegisters(), localRegisterCount + CallerFrameAndPC::sizeInRegisters) - CallerFrameAndPC::sizeInRegisters;
54}
55
56inline unsigned logStackAlignmentRegisters()
57{
58 return WTF::fastLog2(stackAlignmentRegisters());
59}
