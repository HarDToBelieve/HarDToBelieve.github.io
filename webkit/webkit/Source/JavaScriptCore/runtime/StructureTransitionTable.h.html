38using TransitionPropertyAttributes = uint8_t;
39
40enum class TransitionKind : uint8_t {
41 Unknown,
42 PropertyAddition,
43 PropertyDeletion,
44 PropertyAttributeChange,
45
46 // Support for transitions not related to properties.
47 // If any of these are used, the string portion of the key should be 0.
48 AllocateUndecided,
49 AllocateInt32,
50 AllocateDouble,
51 AllocateContiguous,
52 AllocateArrayStorage,
53 AllocateSlowPutArrayStorage,
54 SwitchToSlowPutArrayStorage,
55 AddIndexedAccessors,
56 PreventExtensions,
57 Seal,
58 Freeze,
59
60 // Support for transitions related with private brand
61 SetBrand
62};
63
64static constexpr auto FirstNonPropertyTransitionKind = TransitionKind::AllocateUndecided;
65
66inline bool changesIndexingType(TransitionKind transition)
67{
68 switch (transition) {
69 case TransitionKind::AllocateUndecided:
70 case TransitionKind::AllocateInt32:
71 case TransitionKind::AllocateDouble:
72 case TransitionKind::AllocateContiguous:
73 case TransitionKind::AllocateArrayStorage:
74 case TransitionKind::AllocateSlowPutArrayStorage:
75 case TransitionKind::SwitchToSlowPutArrayStorage:
76 case TransitionKind::AddIndexedAccessors:
77 return true;
78 default:
79 return false;
80 }
81}
82
83inline IndexingType newIndexingType(IndexingType oldType, TransitionKind transition)
84{
85 switch (transition) {
86 case TransitionKind::AllocateUndecided:
87 ASSERT(!hasIndexedProperties(oldType));
88 return oldType | UndecidedShape;
89 case TransitionKind::AllocateInt32:
90 ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || oldType == CopyOnWriteArrayWithInt32);
91 return (oldType & ~IndexingShapeAndWritabilityMask) | Int32Shape;
92 case TransitionKind::AllocateDouble:
93 ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || oldType == CopyOnWriteArrayWithDouble);
94 return (oldType & ~IndexingShapeAndWritabilityMask) | DoubleShape;
95 case TransitionKind::AllocateContiguous:
96 ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || hasDouble(oldType) || oldType == CopyOnWriteArrayWithContiguous);
97 return (oldType & ~IndexingShapeAndWritabilityMask) | ContiguousShape;
98 case TransitionKind::AllocateArrayStorage:
99 ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || hasDouble(oldType) || hasContiguous(oldType));
100 return (oldType & ~IndexingShapeAndWritabilityMask) | ArrayStorageShape;
101 case TransitionKind::AllocateSlowPutArrayStorage:
102 ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || hasDouble(oldType) || hasContiguous(oldType));
103 return (oldType & ~IndexingShapeAndWritabilityMask) | SlowPutArrayStorageShape;
104 case TransitionKind::SwitchToSlowPutArrayStorage:
105 ASSERT(hasArrayStorage(oldType));
106 return (oldType & ~IndexingShapeAndWritabilityMask) | SlowPutArrayStorageShape;
107 case TransitionKind::AddIndexedAccessors:
108 return oldType | MayHaveIndexedAccessors;
109 default:
110 return oldType;
111 }
112}

114inline bool preventsExtensions(TransitionKind transition)
115{
116 switch (transition) {
117 case TransitionKind::PreventExtensions:
118 case TransitionKind::Seal:
119 case TransitionKind::Freeze:
120 return true;
121 default:
122 return false;
123 }
124}

126inline bool setsDontDeleteOnAllProperties(TransitionKind transition)
127{
128 switch (transition) {
129 case TransitionKind::Seal:
130 case TransitionKind::Freeze:
131 return true;
132 default:
133 return false;
134 }

137inline bool setsReadOnlyOnNonAccessorProperties(TransitionKind transition)
138{
139 switch (transition) {
140 case TransitionKind::Freeze:
141 return true;
142 default:
143 return false;

147class StructureTransitionTable {
148 static constexpr intptr_t UsingSingleSlotFlag = 1;
149
150
151#if CPU(ADDRESS64)
152 struct Hash {
153 // Logically, Key is a tuple of (1) UniquedStringImpl*, (2) unsigned attributes, and (3) transitionKind.
154 struct Key {
155 friend struct Hash;
156 static_assert(WTF_OS_CONSTANT_EFFECTIVE_ADDRESS_WIDTH <= 48);
157 static constexpr unsigned attributesShift = 48;
158 static constexpr unsigned transitionKindShift = 56;
159 static constexpr uintptr_t stringMask = (1ULL << attributesShift) - 1;
160 static constexpr uintptr_t hashTableDeletedValue = 0x2;
161 static_assert(sizeof(TransitionPropertyAttributes) * 8 <= 8);
162 static_assert(sizeof(TransitionKind) * 8 <= 8);
163 static_assert(hashTableDeletedValue < alignof(UniquedStringImpl));
164
165 // Highest 8 bits are for TransitionKind; next 8 belong to TransitionPropertyAttributes.
166 // Remaining bits are for UniquedStringImpl*.
167 Key(UniquedStringImpl* impl, unsigned attributes, TransitionKind transitionKind)
168 : m_encodedData(bitwise_cast<uintptr_t>(impl) | (static_cast<uintptr_t>(attributes) << attributesShift) | (static_cast<uintptr_t>(transitionKind) << transitionKindShift))
169 {
170 ASSERT(impl == this->impl());
171 ASSERT(attributes <= UINT8_MAX);
172 ASSERT(attributes == this->attributes());
173 ASSERT(transitionKind != TransitionKind::Unknown);
174 ASSERT(transitionKind == this->transitionKind());
175 }
176
177 Key() = default;
178
179 Key(WTF::HashTableDeletedValueType)
180 : m_encodedData(hashTableDeletedValue)
181 { }
182
183 bool isHashTableDeletedValue() const { return m_encodedData == hashTableDeletedValue; }
184
185 UniquedStringImpl* impl() const { return bitwise_cast<UniquedStringImpl*>(m_encodedData & stringMask); }
186 TransitionPropertyAttributes attributes() const { return (m_encodedData >> attributesShift) & UINT8_MAX; }
187 TransitionKind transitionKind() const { return static_cast<TransitionKind>(m_encodedData >> transitionKindShift); }
188
189 friend bool operator==(const Key& a, const Key& b)
190 {
191 return a.m_encodedData == b.m_encodedData;
192 }
193
194 friend bool operator!=(const Key& a, const Key& b)
195 {
196 return a.m_encodedData != b.m_encodedData;
197 }
198
199 private:
200 uintptr_t m_encodedData { 0 };
201 };
202 using KeyTraits = SimpleClassHashTraits<Key>;
203
204 static unsigned hash(const Key& p)
205 {
206 return IntHash<uintptr_t>::hash(p.m_encodedData);
207 }
208
209 static bool equal(const Key& a, const Key& b)
210 {
211 return a == b;
212 }
213
214 static constexpr bool safeToCompareToEmptyOrDeleted = true;
215 };
216#else
217 struct Hash {
218 using Key = std::tuple<UniquedStringImpl*, unsigned, TransitionKind>;
219 using KeyTraits = HashTraits<Key>;
220
221 static unsigned hash(const Key& p)
222 {
223 return PtrHash<UniquedStringImpl*>::hash(std::get<0>(p)) + std::get<1>(p) + static_cast<unsigned>(std::get<2>(p));
224 }
225
226 static bool equal(const Key& a, const Key& b)
227 {
228 return a == b;
229 }
230
231 static constexpr bool safeToCompareToEmptyOrDeleted = true;
232 };
233#endif
234
235 typedef WeakGCMap<Hash::Key, Structure, Hash, Hash::KeyTraits> TransitionMap;
236
237public:
238 StructureTransitionTable()
239 : m_data(UsingSingleSlotFlag)
240 {
241 }
242
243 ~StructureTransitionTable()
244 {
245 if (!isUsingSingleSlot()) {
246 delete map();
247 return;
248 }
249
250 WeakImpl* impl = this->weakImpl();
251 if (!impl)
252 return;
253 WeakSet::deallocate(impl);
254 }
255
256 void add(VM&, Structure*);
257 bool contains(UniquedStringImpl*, unsigned attributes, TransitionKind) const;
258 Structure* get(UniquedStringImpl*, unsigned attributes, TransitionKind) const;
259
260private:
261 friend class SingleSlotTransitionWeakOwner;
262
263 bool isUsingSingleSlot() const
264 {
265 return m_data & UsingSingleSlotFlag;
266 }
267
268 TransitionMap* map() const
269 {
270 ASSERT(!isUsingSingleSlot());
271 return bitwise_cast<TransitionMap*>(m_data);
272 }
273
274 WeakImpl* weakImpl() const
275 {
276 ASSERT(isUsingSingleSlot());
277 return bitwise_cast<WeakImpl*>(m_data & ~UsingSingleSlotFlag);
278 }
279
280 void setMap(TransitionMap* map)
281 {
282 ASSERT(isUsingSingleSlot());
283
284 if (WeakImpl* impl = this->weakImpl())
285 WeakSet::deallocate(impl);
286
287 // This implicitly clears the flag that indicates we're using a single transition
288 m_data = bitwise_cast<intptr_t>(map);
289
290 ASSERT(!isUsingSingleSlot());
291 }
292
293 Structure* singleTransition() const;
294 void setSingleTransition(Structure*);
295
296 intptr_t m_data;
297};
