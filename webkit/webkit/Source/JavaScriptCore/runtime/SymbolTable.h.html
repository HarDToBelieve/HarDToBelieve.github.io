52// The bit twiddling in this class assumes that every register index is a
53// reasonably small positive or negative number, and therefore has its high
54// four bits all set or all unset.
55
56// In addition to implementing semantics-mandated variable attributes and
57// implementation-mandated variable indexing, this class also implements
58// watchpoints to be used for JIT optimizations. Because watchpoints are
59// meant to be relatively rare, this class optimizes heavily for the case
60// that they are not being used. To that end, this class uses the thin-fat
61// idiom: either it is thin, in which case it contains an in-place encoded
62// word that consists of attributes, the index, and a bit saying that it is
63// thin; or it is fat, in which case it contains a pointer to a malloc'd
64// data structure and a bit saying that it is fat. The malloc'd data
65// structure will be malloced a second time upon copy, to preserve the
66// property that in-place edits to SymbolTableEntry do not manifest in any
67// copies. However, the malloc'd FatEntry data structure contains a ref-
68// counted pointer to a shared WatchpointSet. Thus, in-place edits of the
69// WatchpointSet will manifest in all copies. Here's a picture:
70//
71// SymbolTableEntry --> FatEntry --> WatchpointSet
72//
73// If you make a copy of a SymbolTableEntry, you will have:
74//
75// original: SymbolTableEntry --> FatEntry --> WatchpointSet
76// copy: SymbolTableEntry --> FatEntry -----^
77

82 static VarOffset varOffsetFromBits(intptr_t bits)
83 {
84 VarKind kind;
85 intptr_t kindBits = bits & KindBitsMask;
86 if (kindBits <= UnwatchableScopeKindBits)
87 kind = VarKind::Scope;
88 else if (kindBits == StackKindBits)
89 kind = VarKind::Stack;
90 else
91 kind = VarKind::DirectArgument;
92 return VarOffset::assemble(kind, static_cast<int>(bits >> FlagBits));
93 }
94
95 static ScopeOffset scopeOffsetFromBits(intptr_t bits)
96 {
97 ASSERT((bits & KindBitsMask) <= UnwatchableScopeKindBits);
98 return ScopeOffset(static_cast<int>(bits >> FlagBits));
99 }

297 // This watchpoint set is initialized clear, and goes through the following state transitions:
298 //
299 // First write to this var, in any scope that has this symbol table: Clear->IsWatched.
300 //
301 // Second write to this var, in any scope that has this symbol table: IsWatched->IsInvalidated.
302 //
303 // We ensure that we touch the set (i.e. trigger its state transition) after we do the write. This
304 // means that if you're in the compiler thread, and you:
305 //
306 // 1) Observe that the set IsWatched and commit to adding your watchpoint.
307 // 2) Load a value from any scope that has this watchpoint set.
308 //
309 // Then you can be sure that that value is either going to be the correct value for that var forever,
310 // or the watchpoint set will invalidate and you'll get fired.
311 //
312 // It's possible to write a program that first creates multiple scopes with the same var, and then
313 // initializes that var in just one of them. This means that a compilation could constant-fold to one
314 // of the scopes that still has an undefined value for this variable. That's fine, because at that
315 // point any write to any of the instances of that variable would fire the watchpoint.
316 //
317 // Note that watchpointSet() returns nullptr if JIT is disabled.
318 WatchpointSet* watchpointSet()
319 {
320 if (!isFat())
321 return nullptr;
322 return fatEntry()->m_watchpoints.get();
323 }

401 void pack(VarOffset offset, bool isWatchable, bool readOnly, bool dontEnum)
402 {
403 ASSERT(!isFat());
404 intptr_t& bitsRef = bits();
405 bitsRef =
406 (static_cast<intptr_t>(offset.rawOffset()) << FlagBits) | NotNullFlag | SlimFlag;
407 if (readOnly)
408 bitsRef |= ReadOnlyFlag;
409 if (dontEnum)
410 bitsRef |= DontEnumFlag;
411 switch (offset.kind()) {
412 case VarKind::Scope:
413 if (isWatchable)
414 bitsRef |= ScopeKindBits;
415 else
416 bitsRef |= UnwatchableScopeKindBits;
417 break;
418 case VarKind::Stack:
419 bitsRef |= StackKindBits;
420 break;
421 case VarKind::DirectArgument:
422 bitsRef |= DirectArgumentKindBits;
423 break;
424 default:
425 RELEASE_ASSERT_NOT_REACHED();
426 break;
427 }
428 }
429
430 static bool isValidVarOffset(VarOffset offset)
431 {
432 return ((static_cast<intptr_t>(offset.rawOffset()) << FlagBits) >> FlagBits) == static_cast<intptr_t>(offset.rawOffset());
433 }

644 // The principle behind ScopedArgumentsTable modifications is that we will create one and
645 // leave it unlocked - thereby allowing in-place changes - until someone asks for a pointer to
646 // the table. Then, we will lock it. Then both our future changes and their future changes
647 // will first have to make a copy. This discipline means that usually when we create a
648 // ScopedArguments object, we don't have to make a copy of the ScopedArgumentsTable - instead
649 // we just take a reference to one that we already have.

651 uint32_t argumentsLength() const
652 {
653 if (!m_arguments)
654 return 0;
655 return m_arguments->length();
656 }
657
658 bool trySetArgumentsLength(VM& vm, uint32_t length)
659 {
660 if (UNLIKELY(!m_arguments)) {
661 ScopedArgumentsTable* table = ScopedArgumentsTable::tryCreate(vm, length);
662 if (UNLIKELY(!table))
663 return false;
664 m_arguments.set(vm, this, table);
665 } else {
666 ScopedArgumentsTable* table = m_arguments->trySetLength(vm, length);
667 if (UNLIKELY(!table))
668 return false;
669 m_arguments.set(vm, this, table);
670 }
671 return true;
672 }
