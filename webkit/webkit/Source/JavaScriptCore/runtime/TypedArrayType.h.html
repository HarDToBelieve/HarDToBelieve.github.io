35// Keep in sync with the order of JSType.
36#define FOR_EACH_TYPED_ARRAY_TYPE_EXCLUDING_DATA_VIEW(macro) \
37 macro(Int8) \
38 macro(Uint8) \
39 macro(Uint8Clamped) \
40 macro(Int16) \
41 macro(Uint16) \
42 macro(Int32) \
43 macro(Uint32) \
44 macro(Float32) \
45 macro(Float64) \
46 macro(BigInt64) \
47 macro(BigUint64)
48
49#define FOR_EACH_TYPED_ARRAY_TYPE(macro) \
50 FOR_EACH_TYPED_ARRAY_TYPE_EXCLUDING_DATA_VIEW(macro) \
51 macro(DataView)
52
53enum TypedArrayType {
54 NotTypedArray,
55#define DECLARE_TYPED_ARRAY_TYPE(name) Type ## name,
56 FOR_EACH_TYPED_ARRAY_TYPE(DECLARE_TYPED_ARRAY_TYPE)
57#undef DECLARE_TYPED_ARRAY_TYPE
58};
59
60enum class TypedArrayContentType : uint8_t {
61 None,
62 Number,
63 BigInt,
64};

66#define ASSERT_TYPED_ARRAY_TYPE(name) \
67 static_assert(static_cast<uint32_t>(Type ## name) == (static_cast<uint32_t>(name ## ArrayType) - FirstTypedArrayType + static_cast<uint32_t>(TypeInt8)), "");
68 FOR_EACH_TYPED_ARRAY_TYPE_EXCLUDING_DATA_VIEW(ASSERT_TYPED_ARRAY_TYPE)
69#undef ASSERT_TYPED_ARRAY_TYPE
70
71static_assert(TypeDataView == (DataViewType - FirstTypedArrayType + TypeInt8), "");
72
73inline unsigned toIndex(TypedArrayType type)
74{
75 return static_cast<unsigned>(type) - 1;
76}
77
78inline TypedArrayType indexToTypedArrayType(unsigned index)
79{
80 TypedArrayType result = static_cast<TypedArrayType>(index + 1);
81 ASSERT(result >= TypeInt8 && result <= TypeDataView);
82 return result;
83}
84
85inline bool isTypedView(TypedArrayType type)
86{
87 switch (type) {
88 case NotTypedArray:
89 case TypeDataView:
90 return false;
91 default:
92 return true;
93 }
94}
95
96inline bool isBigIntTypedView(TypedArrayType type)
97{
98 switch (type) {
99 case TypeBigInt64:
100 case TypeBigUint64:
101 return true;
102 default:
103 return false;
104 }
105}
106
107inline unsigned logElementSize(TypedArrayType type)
108{
109 switch (type) {
110 case NotTypedArray:
111 break;
112 case TypeInt8:
113 case TypeUint8:
114 case TypeUint8Clamped:
115 case TypeDataView:
116 return 0;
117 case TypeInt16:
118 case TypeUint16:
119 return 1;
120 case TypeInt32:
121 case TypeUint32:
122 case TypeFloat32:
123 return 2;
124 case TypeFloat64:
125 case TypeBigInt64:
126 case TypeBigUint64:
127 return 3;
128 }
129 RELEASE_ASSERT_NOT_REACHED();
130 return 0;
131}
132
133inline size_t elementSize(TypedArrayType type)
134{
135 return static_cast<size_t>(1) << logElementSize(type);
136}

140inline TypedArrayType typedArrayTypeForType(JSType type)
141{
142 if (type >= FirstTypedArrayType && type <= LastTypedArrayType)
143 return static_cast<TypedArrayType>(type - FirstTypedArrayType + TypeInt8);
144 return NotTypedArray;
145}
146
147inline JSType typeForTypedArrayType(TypedArrayType type)
148{
149 if (type >= TypeInt8 && type <= TypeDataView)
150 return static_cast<JSType>(type - TypeInt8 + FirstTypedArrayType);
151
152 RELEASE_ASSERT_NOT_REACHED();
153 return Int8ArrayType;
154}

156inline bool isInt(TypedArrayType type)
157{
158 switch (type) {
159 case TypeInt8:
160 case TypeUint8:
161 case TypeUint8Clamped:
162 case TypeInt16:
163 case TypeUint16:
164 case TypeInt32:
165 case TypeUint32:
166 return true;
167 default:
168 return false;
169 }
170}

194inline bool isSigned(TypedArrayType type)
195{
196 switch (type) {
197 case TypeInt8:
198 case TypeInt16:
199 case TypeInt32:
200 case TypeFloat32:
201 case TypeFloat64:
202 case TypeBigInt64:
203 return true;

208
209inline bool isClamped(TypedArrayType type)
210{
211 return type == TypeUint8Clamped;
212}
213
214inline constexpr TypedArrayContentType contentType(TypedArrayType type)
215{
216 switch (type) {
217 case TypeBigInt64:
218 case TypeBigUint64:
219 return TypedArrayContentType::BigInt;
220 case TypeInt8:
221 case TypeInt16:
222 case TypeInt32:
223 case TypeUint8:
224 case TypeUint16:
225 case TypeUint32:
226 case TypeFloat32:
227 case TypeFloat64:
228 case TypeUint8Clamped:
229 return TypedArrayContentType::Number;
230 case NotTypedArray:
231 case TypeDataView:
232 return TypedArrayContentType::None;
233 }
234 return TypedArrayContentType::None;
235}
