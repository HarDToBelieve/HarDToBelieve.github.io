28#include "Weak.h"
29#include "WeakGCHashTable.h"
30#include <wtf/HashSet.h>
31
32namespace JSC {
33
34// A HashSet with Weak<JSCell> values, which automatically removes values once they're garbage collected.
35
36template<typename T>
37struct WeakGCSetHashTraits : HashTraits<Weak<T>> {
38 static constexpr bool hasIsReleasedWeakValueFunction = true;
39 static bool isReleasedWeakValue(const Weak<T>& value)
40 {
41 return !value.isHashTableDeletedValue() && !value.isHashTableEmptyValue() && !value;
42 }
43};
44
45template<typename ValueArg, typename HashArg = DefaultHash<Weak<ValueArg>>, typename TraitsArg = WeakGCSetHashTraits<ValueArg>>
46class WeakGCSet final : public WeakGCHashTable {
47 WTF_MAKE_FAST_ALLOCATED;
48 using ValueType = Weak<ValueArg>;
49 using HashSetType = HashSet<ValueType, HashArg, TraitsArg>;
50
51public:
52 using AddResult = typename HashSetType::AddResult;
53 using iterator = typename HashSetType::iterator;
54 using const_iterator = typename HashSetType::const_iterator;
55
56 explicit WeakGCSet(VM&);
57 ~WeakGCSet() final;
58
59 void clear()
60 {
61 m_set.clear();
62 }
63
64 template<typename HashTranslator, typename T, typename Functor>
65 ValueArg* ensureValue(T&& key, Functor&& functor)
66 {
67 // If functor invokes GC, GC can prune WeakGCSet, and manipulate HashSet while we are touching it in the ensure function.
68 // The functor must not invoke GC.
69 DisallowGC disallowGC;
70
71 auto result = m_set.template ensure<HashTranslator>(std::forward<T>(key), std::forward<Functor>(functor));
72 return result.iterator->get();
73 }
74
75 inline iterator find(const ValueType& key);
76 inline const_iterator find(const ValueType& key) const;
77 inline bool contains(const ValueType&) const;
78
79private:
80 void pruneStaleEntries() final;
81
82 HashSetType m_set;
83 VM& m_vm;
84};
85
86} // namespace JSC
