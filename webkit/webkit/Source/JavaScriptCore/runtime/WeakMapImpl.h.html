37struct WeakMapBucketDataKey {
38 static const HashTableType Type = HashTableType::Key;
39 WriteBarrier<JSObject> key;
40};
41static_assert(sizeof(WeakMapBucketDataKey) == sizeof(void*), "");

43struct WeakMapBucketDataKeyValue {
44 static const HashTableType Type = HashTableType::KeyValue;
45 WriteBarrier<JSObject> key;
46#if USE(JSVALUE32_64)
47 uint32_t padding;
48#endif
49 WriteBarrier<Unknown> value;
50};
51static_assert(sizeof(WeakMapBucketDataKeyValue) == 16, "");

56template <typename Data>
57class WeakMapBucket {
58public:
59 ALWAYS_INLINE void setKey(VM& vm, JSCell* owner, JSObject* key)
60 {
61 m_data.key.set(vm, owner, key);
62 }
63
64 template <typename T = Data>
65 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value>::type setValue(VM& vm, JSCell* owner, JSValue value)
66 {
67 m_data.value.set(vm, owner, value);
68 }
69 template <typename T = Data>
70 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKey>::value>::type setValue(VM&, JSCell*, JSValue) { }
71
72 ALWAYS_INLINE JSObject* key() const { return m_data.key.get(); }
73
74 template <typename T = Data>
75 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value, JSValue>::type value() const
76 {
77 return m_data.value.get();
78 }
79 template <typename T = Data>
80 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKey>::value, JSValue>::type value() const { return JSValue(); }
81
82 template <typename T = Data>
83 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value>::type copyFrom(const WeakMapBucket& from)
84 {
85 m_data.key.copyFrom(from.m_data.key);
86 m_data.value.setWithoutWriteBarrier(from.m_data.value.get());
87 }
88 template <typename T = Data>
89 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKey>::value>::type copyFrom(const WeakMapBucket& from)
90 {
91 m_data.key.copyFrom(from.m_data.key);
92 }
93
94 static ptrdiff_t offsetOfKey()
95 {
96 return OBJECT_OFFSETOF(WeakMapBucket, m_data) + OBJECT_OFFSETOF(Data, key);
97 }
98
99 template <typename T = Data>
100 static typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value, ptrdiff_t>::type offsetOfValue()
101 {
102 return OBJECT_OFFSETOF(WeakMapBucket, m_data) + OBJECT_OFFSETOF(Data, value);
103 }
104
105 template <typename T = Data>
106 ALWAYS_INLINE static typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value, JSValue>::type extractValue(const WeakMapBucket& bucket)
107 {
108 return bucket.value();
109 }
110
111 template <typename T = Data>
112 ALWAYS_INLINE static typename std::enable_if<std::is_same<T, WeakMapBucketDataKey>::value, JSValue>::type extractValue(const WeakMapBucket&)
113 {
114 return JSValue();
115 }
116
117 bool isEmpty()
118 {
119 return !m_data.key.unvalidatedGet();
120 }
121
122 static JSObject* deletedKey()
123 {
124 return bitwise_cast<JSObject*>(static_cast<uintptr_t>(-3));
125 }
126
127 bool isDeleted()
128 {
129 return m_data.key.unvalidatedGet() == deletedKey();
130 }
131
132 void makeDeleted()
133 {
134 m_data.key.setWithoutWriteBarrier(deletedKey());
135 clearValue();
136 }
137
138 template <typename T = Data, typename Visitor>
139 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value>::type visitAggregate(Visitor& visitor)
140 {
141 visitor.append(m_data.value);
142 }
143
144private:
145 template <typename T = Data>
146 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKeyValue>::value>::type clearValue()
147 {
148 m_data.value.clear();
149 }
150 template <typename T = Data>
151 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucketDataKey>::value>::type clearValue() { }
152
153 Data m_data;
154};

156template <typename BucketType>
157class WeakMapBuffer {
158public:
159 WeakMapBuffer() = delete;
160
161 static size_t allocationSize(Checked<size_t> capacity)
162 {
163 return capacity * sizeof(BucketType);
164 }
165
166 ALWAYS_INLINE BucketType* buffer() const
167 {
168 return bitwise_cast<BucketType*>(this);
169 }
170
171 static MallocPtr<WeakMapBuffer, JSValueMalloc> create(uint32_t capacity)
172 {
173 size_t allocationSize = WeakMapBuffer::allocationSize(capacity);
174 auto buffer = MallocPtr<WeakMapBuffer, JSValueMalloc>::malloc(allocationSize);
175 buffer->reset(capacity);
176 return buffer;
177 }
178
179 ALWAYS_INLINE void reset(uint32_t capacity)
180 {
181 memset(this, 0, allocationSize(capacity));
182 }
183};

220 template <typename T = WeakMapBucketType>
221 ALWAYS_INLINE typename std::enable_if<std::is_same<T, WeakMapBucket<WeakMapBucketDataKeyValue>>::value, JSValue>::type get(JSObject* key)
222 {
223 DisallowGC disallowGC;
224 if (WeakMapBucketType* bucket = findBucket(key))
225 return bucket->value();
226 return jsUndefined();
227 }
228
229 ALWAYS_INLINE bool has(JSObject* key)
230 {
231 DisallowGC disallowGC;
232 return !!findBucket(key);
233 }
234
235 ALWAYS_INLINE void add(VM&, JSObject* key, JSValue = JSValue());
236 ALWAYS_INLINE void add(VM&, JSObject* key, JSValue, uint32_t hash);
237
238 ALWAYS_INLINE bool remove(JSObject* key)
239 {
240 DisallowGC disallowGC;
241 WeakMapBucketType* bucket = findBucket(key);
242 if (!bucket)
243 return false;
244
245 bucket->makeDeleted();
246
247 ++m_deleteCount;
248 RELEASE_ASSERT(m_keyCount > 0);
249 --m_keyCount;
250
251 if (shouldShrink())
252 rehash();
253
254 return true;
255 }
256
257 ALWAYS_INLINE uint32_t size() const
258 {
259 return m_keyCount;
260 }

314 enum class IterationState { Continue, Stop };
315 template<typename Functor>
316 void forEach(Functor functor)
317 {
318 auto* buffer = this->buffer();
319 for (uint32_t index = 0; index < m_capacity; ++index) {
320 auto* bucket = buffer + index;
321 if (bucket->isEmpty() || bucket->isDeleted())
322 continue;
323 if (functor(bucket->key(), bucket->value()) == IterationState::Stop)
324 return;
325 }
326 }

335 ALWAYS_INLINE static bool canUseBucket(WeakMapBucketType* bucket, JSObject* key)
336 {
337 return !bucket->isDeleted() && key == bucket->key();
338 }
339
340 ALWAYS_INLINE void addInternal(VM& vm, JSObject* key, JSValue value, uint32_t hash)
341 {
342 const uint32_t mask = m_capacity - 1;
343 uint32_t index = hash & mask;
344 WeakMapBucketType* buffer = this->buffer();
345 WeakMapBucketType* bucket = buffer + index;
346 while (!bucket->isEmpty()) {
347 if (canUseBucket(bucket, key)) {
348 ASSERT(!bucket->isDeleted());
349 bucket->setValue(vm, this, value);
350 return;
351 }
352 index = (index + 1) & mask;
353 bucket = buffer + index;
354 }
355
356 auto* newEntry = buffer + index;
357 newEntry->setKey(vm, this, key);
358 newEntry->setValue(vm, this, value);
359 ++m_keyCount;
360 }
361
362 ALWAYS_INLINE WeakMapBucketType* findBucketAlreadyHashed(JSObject* key, uint32_t hash)
363 {
364 const uint32_t mask = m_capacity - 1;
365 uint32_t index = hash & mask;
366 WeakMapBucketType* buffer = this->buffer();
367 WeakMapBucketType* bucket = buffer + index;
368
369 while (!bucket->isEmpty()) {
370 if (canUseBucket(bucket, key)) {
371 ASSERT(!bucket->isDeleted());
372 return buffer + index;
373 }
374 index = (index + 1) & mask;
375 bucket = buffer + index;
376 }
377 return nullptr;
378 }

383 ALWAYS_INLINE void checkConsistency() const
384 {
385 if (ASSERT_ENABLED) {
386 uint32_t size = 0;
387 auto* buffer = this->buffer();
388 for (uint32_t index = 0; index < m_capacity; ++index) {
389 auto* bucket = buffer + index;
390 if (bucket->isEmpty() || bucket->isDeleted())
391 continue;
392 ++size;
393 }
394 ASSERT(size == m_keyCount);
395 }
396 }

398 void makeAndSetNewBuffer(const AbstractLocker&, uint32_t capacity)
399 {
400 ASSERT(!(capacity & (capacity - 1)));
401
402 m_buffer = WeakMapBufferType::create(capacity);
403 m_capacity = capacity;
404 ASSERT(m_buffer);
405 assertBufferIsEmpty();
406 }
407
408 ALWAYS_INLINE void assertBufferIsEmpty() const
409 {
410 if (ASSERT_ENABLED) {
411 for (unsigned i = 0; i < m_capacity; i++)
412 ASSERT((buffer() + i)->isEmpty());
413 }
414 }
