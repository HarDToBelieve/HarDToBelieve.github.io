80class JITCallee : public Callee {
81public:
82 MacroAssemblerCodePtr<WasmEntryPtrTag> entrypoint() const override { return m_entrypoint.compilation->code().retagged<WasmEntryPtrTag>(); }
83 RegisterAtOffsetList* calleeSaveRegisters() override { return &m_entrypoint.calleeSaveRegisters; }
84 Vector<UnlinkedWasmToWasmCall>& wasmToWasmCallsites() { return m_wasmToWasmCallsites; }
85
86protected:
87 JS_EXPORT_PRIVATE JITCallee(Wasm::CompilationMode, Wasm::Entrypoint&&);
88 JS_EXPORT_PRIVATE JITCallee(Wasm::CompilationMode, Wasm::Entrypoint&&, size_t, std::pair<const Name*, RefPtr<NameSection>>&&, Vector<UnlinkedWasmToWasmCall>&&);
89
90 std::tuple<void*, void*> range() const override
91 {
92 void* start = m_entrypoint.compilation->codeRef().executableMemory()->start().untaggedPtr();
93 void* end = m_entrypoint.compilation->codeRef().executableMemory()->end().untaggedPtr();
94 return { start, end };
95 }
96
97private:
98 Vector<UnlinkedWasmToWasmCall> m_wasmToWasmCallsites;
99 Wasm::Entrypoint m_entrypoint;
100};
