43using ArgumentLocation = ValueLocation;
44enum class CallRole : uint8_t {
45 Caller,
46 Callee,
47};
48
49struct CallInformation {
50 CallInformation(Vector<ArgumentLocation>&& parameters, Vector<ArgumentLocation, 1>&& returnValues, size_t stackOffset)
51 : params(WTFMove(parameters))
52 , results(WTFMove(returnValues))
53 , headerAndArgumentStackSizeInBytes(stackOffset)
54 { }
55
56 RegisterAtOffsetList computeResultsOffsetList()
57 {
58 RegisterSet usedResultRegisters;
59 for (ValueLocation loc : results) {
60 if (loc.isReg())
61 usedResultRegisters.set(loc.reg());
62 }
63
64 RegisterAtOffsetList savedRegs(usedResultRegisters, RegisterAtOffsetList::ZeroBased);
65 return savedRegs;
66 }
67
68 bool argumentsIncludeI64 { false };
69 bool resultsIncludeI64 { false };
70 Vector<ArgumentLocation> params;
71 Vector<ArgumentLocation, 1> results;
72 // As a callee this includes CallerFrameAndPC as a caller it does not.
73 size_t headerAndArgumentStackSizeInBytes;
74};

76class WasmCallingConvention {
77public:
78 static constexpr unsigned headerSizeInBytes = CallFrame::headerSizeInRegisters * sizeof(Register);
79
80 WasmCallingConvention(Vector<Reg>&& gprs, Vector<Reg>&& fprs, Vector<GPRReg>&& scratches, RegisterSet&& calleeSaves, RegisterSet&& callerSaves)
81 : gprArgs(WTFMove(gprs))
82 , fprArgs(WTFMove(fprs))
83 , prologueScratchGPRs(WTFMove(scratches))
84 , calleeSaveRegisters(WTFMove(calleeSaves))
85 , callerSaveRegisters(WTFMove(callerSaves))
86 { }
87
88 WTF_MAKE_NONCOPYABLE(WasmCallingConvention);
89
90private:
91 ArgumentLocation marshallLocationImpl(CallRole role, const Vector<Reg>& regArgs, size_t& count, size_t& stackOffset) const
92 {
93 if (count < regArgs.size())
94 return ArgumentLocation::reg(regArgs[count++]);
95
96 count++;
97 ArgumentLocation result = role == CallRole::Caller ? ArgumentLocation::stackArgument(stackOffset) : ArgumentLocation::stack(stackOffset);
98 stackOffset += sizeof(Register);
99 return result;
100 }
101
102 ArgumentLocation marshallLocation(CallRole role, Type valueType, size_t& gpArgumentCount, size_t& fpArgumentCount, size_t& stackOffset) const
103 {
104 ASSERT(isValueType(valueType));
105 switch (valueType.kind) {
106 case TypeKind::I32:
107 case TypeKind::I64:
108 case TypeKind::Funcref:
109 case TypeKind::Externref:
110 case TypeKind::TypeIdx:
111 return marshallLocationImpl(role, gprArgs, gpArgumentCount, stackOffset);
112 case TypeKind::F32:
113 case TypeKind::F64:
114 return marshallLocationImpl(role, fprArgs, fpArgumentCount, stackOffset);
115 default:
116 break;
117 }
118 RELEASE_ASSERT_NOT_REACHED();
119 }
120
121public:
122 CallInformation callInformationFor(const Signature& signature, CallRole role = CallRole::Caller) const
123 {
124 bool argumentsIncludeI64 = false;
125 bool resultsIncludeI64 = false;
126 size_t gpArgumentCount = 0;
127 size_t fpArgumentCount = 0;
128 size_t argStackOffset = headerSizeInBytes;
129 if (role == CallRole::Caller)
130 argStackOffset -= sizeof(CallerFrameAndPC);
131
132 Vector<ArgumentLocation> params(signature.argumentCount());
133 for (size_t i = 0; i < signature.argumentCount(); ++i) {
134 argumentsIncludeI64 |= signature.argument(i).isI64();
135 params[i] = marshallLocation(role, signature.argument(i), gpArgumentCount, fpArgumentCount, argStackOffset);
136 }
137 gpArgumentCount = 0;
138 fpArgumentCount = 0;
139 size_t resultStackOffset = headerSizeInBytes;
140 if (role == CallRole::Caller)
141 resultStackOffset -= sizeof(CallerFrameAndPC);
142
143 Vector<ArgumentLocation, 1> results(signature.returnCount());
144 for (size_t i = 0; i < signature.returnCount(); ++i) {
145 resultsIncludeI64 |= signature.returnType(i).isI64();
146 results[i] = marshallLocation(role, signature.returnType(i), gpArgumentCount, fpArgumentCount, resultStackOffset);
147 }
148
149 CallInformation result(WTFMove(params), WTFMove(results), std::max(argStackOffset, resultStackOffset));
150 result.argumentsIncludeI64 = argumentsIncludeI64;
151 result.resultsIncludeI64 = resultsIncludeI64;
152 return result;
153 }
154
155 const Vector<Reg> gprArgs;
156 const Vector<Reg> fprArgs;
157 const Vector<GPRReg> prologueScratchGPRs;
158 const RegisterSet calleeSaveRegisters;
159 const RegisterSet callerSaveRegisters;
160};

162class JSCallingConvention {
163public:
164 static constexpr unsigned headerSizeInBytes = CallFrame::headerSizeInRegisters * sizeof(Register);
165
166 // vmEntryToWasm passes the JSWebAssemblyInstance corresponding to Wasm::Context*'s
167 // instance as the first JS argument when we're not using fast TLS to hold the
168 // Wasm::Context*'s instance.
169 static constexpr ptrdiff_t instanceStackOffset = CallFrameSlot::thisArgument * sizeof(EncodedJSValue);
170
171 JSCallingConvention(Vector<Reg>&& gprs, Vector<Reg>&& fprs, RegisterSet&& calleeSaves, RegisterSet&& callerSaves)
172 : gprArgs(WTFMove(gprs))
173 , fprArgs(WTFMove(fprs))
174 , calleeSaveRegisters(WTFMove(calleeSaves))
175 , callerSaveRegisters(WTFMove(callerSaves))
176 { }
177
178 WTF_MAKE_NONCOPYABLE(JSCallingConvention);
179private:
180 ArgumentLocation marshallLocationImpl(CallRole role, const Vector<Reg>& regArgs, size_t& count, size_t& stackOffset) const
181 {
182 if (count < regArgs.size())
183 return ArgumentLocation::reg(regArgs[count++]);
184
185 count++;
186 ArgumentLocation result = role == CallRole::Caller ? ArgumentLocation::stackArgument(stackOffset) : ArgumentLocation::stack(stackOffset);
187 stackOffset += sizeof(Register);
188 return result;
189 }
190
191 ArgumentLocation marshallLocation(CallRole role, Type valueType, size_t& gpArgumentCount, size_t& fpArgumentCount, size_t& stackOffset) const
192 {
193 ASSERT(isValueType(valueType));
194 switch (valueType.kind) {
195 case TypeKind::I32:
196 case TypeKind::I64:
197 case TypeKind::Funcref:
198 case TypeKind::Externref:
199 case TypeKind::TypeIdx:
200 return marshallLocationImpl(role, gprArgs, gpArgumentCount, stackOffset);
201 case TypeKind::F32:
202 case TypeKind::F64:
203 return marshallLocationImpl(role, fprArgs, fpArgumentCount, stackOffset);
204 default:
205 break;
206 }
207 RELEASE_ASSERT_NOT_REACHED();
208 }
209
210public:
211 CallInformation callInformationFor(const Signature& signature, CallRole role = CallRole::Callee) const
212 {
213 size_t gpArgumentCount = 0;
214 size_t fpArgumentCount = 0;
215 size_t stackOffset = headerSizeInBytes + sizeof(Register); // Skip the this value since wasm doesn't use it and we sometimes put the context there.
216 if (role == CallRole::Caller)
217 stackOffset -= sizeof(CallerFrameAndPC);
218
219 Vector<ArgumentLocation> params;
220 for (size_t i = 0; i < signature.argumentCount(); ++i)
221 params.append(marshallLocation(role, signature.argument(i), gpArgumentCount, fpArgumentCount, stackOffset));
222
223 Vector<ArgumentLocation, 1> results { ArgumentLocation::reg(GPRInfo::returnValueGPR) };
224 return CallInformation(WTFMove(params), WTFMove(results), stackOffset);
225 }
226
227 const Vector<Reg> gprArgs;
228 const Vector<Reg> fprArgs;
229 const RegisterSet calleeSaveRegisters;
230 const RegisterSet callerSaveRegisters;
231};
