61inline bool isValueType(Type type)
62{
63 switch (type.kind) {
64 case TypeKind::I32:
65 case TypeKind::I64:
66 case TypeKind::F32:
67 case TypeKind::F64:
68 return true;
69 case TypeKind::Externref:
70 case TypeKind::Funcref:
71 return Options::useWebAssemblyReferences();
72 case TypeKind::TypeIdx:
73 return Options::useWebAssemblyTypedFunctionReferences();
74 default:
75 break;
76 }
77 return false;
78}
79
80inline bool isSubtype(Type sub, Type parent)
81{
82 if (sub.isNullable() && !parent.isNullable())
83 return false;
84
85 if (sub.isTypeIdx() && parent.isFuncref())
86 return true;
87
88 return sub == parent;
89}
90
91inline bool isRefType(Type type)
92{
93 return type.isFuncref() || type.isExternref() || type.isTypeIdx();
94}
95
96inline bool isValidHeapTypeKind(TypeKind kind)
97{
98 switch (kind) {
99 case TypeKind::Funcref:
100 case TypeKind::Externref:
101 return true;
102 default:
103 break;
104 }
105 return false;
106}
107
108inline bool isDefaultableType(Type type)
109{
110 return !isRefType(type) || type.isNullable();
111}

113enum class ExternalKind : uint8_t {
114 // FIXME auto-generate this. https://bugs.webkit.org/show_bug.cgi?id=165231
115 Function = 0,
116 Table = 1,
117 Memory = 2,
118 Global = 3,
119};
120
121template<typename Int>
122inline bool isValidExternalKind(Int val)
123{
124 switch (val) {
125 case static_cast<Int>(ExternalKind::Function):
126 case static_cast<Int>(ExternalKind::Table):
127 case static_cast<Int>(ExternalKind::Memory):
128 case static_cast<Int>(ExternalKind::Global):
129 return true;
130 }
131 return false;
132}
133
134static_assert(static_cast<int>(ExternalKind::Function) == 0, "Wasm needs Function to have the value 0");
135static_assert(static_cast<int>(ExternalKind::Table) == 1, "Wasm needs Table to have the value 1");
136static_assert(static_cast<int>(ExternalKind::Memory) == 2, "Wasm needs Memory to have the value 2");
137static_assert(static_cast<int>(ExternalKind::Global) == 3, "Wasm needs Global to have the value 3");
138
139inline const char* makeString(ExternalKind kind)
140{
141 switch (kind) {
142 case ExternalKind::Function: return "function";
143 case ExternalKind::Table: return "table";
144 case ExternalKind::Memory: return "memory";
145 case ExternalKind::Global: return "global";
146 }
147 RELEASE_ASSERT_NOT_REACHED();
148 return "?";
149}

202class I32InitExpr {
203 WTF_MAKE_FAST_ALLOCATED;
204 enum Type : uint8_t {
205 Global,
206 Const
207 };
208
209 I32InitExpr(Type type, uint32_t bits)
210 : m_bits(bits)
211 , m_type(type)
212 { }
213
214public:
215 I32InitExpr() = delete;
216
217 static I32InitExpr globalImport(uint32_t globalImportNumber) { return I32InitExpr(Global, globalImportNumber); }
218 static I32InitExpr constValue(uint32_t constValue) { return I32InitExpr(Const, constValue); }
219
220 bool isConst() const { return m_type == Const; }
221 bool isGlobalImport() const { return m_type == Global; }
222 uint32_t constValue() const
223 {
224 RELEASE_ASSERT(isConst());
225 return m_bits;
226 }
227 uint32_t globalImportIndex() const
228 {
229 RELEASE_ASSERT(isGlobalImport());
230 return m_bits;
231 }
232
233private:
234 uint32_t m_bits;
235 Type m_type;
236};

238struct Segment {
239 WTF_MAKE_STRUCT_FAST_ALLOCATED;
240
241 enum class Kind : uint8_t {
242 Active,
243 Passive,
244 };
245
246 Kind kind;
247 uint32_t sizeInBytes;
248 std::optional<I32InitExpr> offsetIfActive;
249 // Bytes are allocated at the end.
250 uint8_t& byte(uint32_t pos)
251 {
252 ASSERT(pos < sizeInBytes);
253 return *reinterpret_cast<uint8_t*>(reinterpret_cast<char*>(this) + sizeof(Segment) + pos);
254 }
255
256 static void destroy(Segment*);
257 typedef std::unique_ptr<Segment, decltype(&Segment::destroy)> Ptr;
258 static Segment::Ptr create(std::optional<I32InitExpr>, uint32_t, Kind);
259
260 bool isActive() const { return kind == Kind::Active; }
261 bool isPassive() const { return kind == Kind::Passive; }
262};

264struct Element {
265 WTF_MAKE_STRUCT_FAST_ALLOCATED;
266
267 // nullFuncIndex represents the case when an element segment (of type funcref)
268 // contains a null element.
269 constexpr static uint32_t nullFuncIndex = UINT32_MAX;
270
271 enum class Kind : uint8_t {
272 Active,
273 Passive,
274 Declared,
275 };
276
277 Element(Element::Kind kind, TableElementType elementType, std::optional<uint32_t> tableIndex, std::optional<I32InitExpr> initExpr)
278 : kind(kind)
279 , elementType(elementType)
280 , tableIndexIfActive(WTFMove(tableIndex))
281 , offsetIfActive(WTFMove(initExpr))
282 { }
283
284 Element(Element::Kind kind, TableElementType elemType)
285 : Element(kind, elemType, std::nullopt, std::nullopt)
286 { }
287
288 uint32_t length() const { return functionIndices.size(); }
289
290 bool isActive() const { return kind == Kind::Active; }
291 bool isPassive() const { return kind == Kind::Passive; }
292
293 static bool isNullFuncIndex(uint32_t idx) { return idx == nullFuncIndex; }
294
295 Kind kind;
296 TableElementType elementType;
297 std::optional<uint32_t> tableIndexIfActive;
298 std::optional<I32InitExpr> offsetIfActive;
299
300 // Index may be nullFuncIndex.
301 Vector<uint32_t> functionIndices;
302};

379// WebAssembly direct calls and call_indirect use indices into "function index space". This space starts
380// with all imports, and then all internal functions. WasmToWasmImportableFunction and FunctionIndexSpace are only
381// meant as fast lookup tables for these opcodes and do not own code.
382struct WasmToWasmImportableFunction {
383 WTF_MAKE_STRUCT_FAST_ALLOCATED;
384 using LoadLocation = MacroAssemblerCodePtr<WasmEntryPtrTag>*;
385 static ptrdiff_t offsetOfSignatureIndex() { return OBJECT_OFFSETOF(WasmToWasmImportableFunction, signatureIndex); }
386 static ptrdiff_t offsetOfEntrypointLoadLocation() { return OBJECT_OFFSETOF(WasmToWasmImportableFunction, entrypointLoadLocation); }
387
388 // FIXME: Pack signature index and code pointer into one 64-bit value. See <https://bugs.webkit.org/show_bug.cgi?id=165511>.
389 SignatureIndex signatureIndex { Signature::invalidIndex };
390 LoadLocation entrypointLoadLocation;
391};
392using FunctionIndexSpace = Vector<WasmToWasmImportableFunction>;
