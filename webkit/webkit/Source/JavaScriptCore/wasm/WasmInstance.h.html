101 void setMemory(Ref<Memory>&& memory)
102 {
103 m_memory = WTFMove(memory);
104 m_memory.get()->registerInstance(this);
105 updateCachedMemory();
106 }
107 void updateCachedMemory()
108 {
109 if (m_memory != nullptr) {
110 m_cachedMemory = CagedPtr<Gigacage::Primitive, void, tagCagedPtr>(memory()->memory(), memory()->boundsCheckingSize());
111 m_cachedBoundsCheckingSize = memory()->boundsCheckingSize();
112 ASSERT(memory()->memory() == cachedMemory());
113 }
114 }

116 int32_t loadI32Global(unsigned i) const
117 {
118 Global::Value* slot = m_globals.get() + i;
119 if (m_globalsToBinding.get(i)) {
120 slot = slot->m_pointer;
121 if (!slot)
122 return 0;
123 }
124 return slot->m_primitive;
125 }
126 int64_t loadI64Global(unsigned i) const
127 {
128 Global::Value* slot = m_globals.get() + i;
129 if (m_globalsToBinding.get(i)) {
130 slot = slot->m_pointer;
131 if (!slot)
132 return 0;
133 }
134 return slot->m_primitive;
135 }
136 float loadF32Global(unsigned i) const { return bitwise_cast<float>(loadI32Global(i)); }
137 double loadF64Global(unsigned i) const { return bitwise_cast<double>(loadI64Global(i)); }
138 void setGlobal(unsigned i, int64_t bits)
139 {
140 Global::Value* slot = m_globals.get() + i;
141 if (m_globalsToBinding.get(i)) {
142 slot = slot->m_pointer;
143 if (!slot)
144 return;
145 }
146 slot->m_primitive = bits;
147 }

156 Wasm::Global* getGlobalBinding(unsigned i)
157 {
158 ASSERT(m_globalsToBinding.get(i));
159 Wasm::Global::Value* pointer = m_globals.get()[i].m_pointer;
160 if (!pointer)
161 return nullptr;
162 return &Wasm::Global::fromBinding(*pointer);
163 }

184 // Tail accessors.
185 static constexpr size_t offsetOfTail() { return WTF::roundUpToMultipleOf<sizeof(uint64_t)>(sizeof(Instance)); }
186 struct ImportFunctionInfo {
187 // Target instance and entrypoint are only set for wasm->wasm calls, and are otherwise nullptr. The embedder-specific logic occurs through import function.
188 Instance* targetInstance { nullptr };
189 WasmToWasmImportableFunction::LoadLocation wasmEntrypointLoadLocation { nullptr };
190 MacroAssemblerCodePtr<WasmEntryPtrTag> wasmToEmbedderStub;
191 void* importFunction { nullptr }; // In a JS embedding, this is a WriteBarrier<JSObject>.
192 };
193 unsigned numImportFunctions() const { return m_numImportFunctions; }
194 ImportFunctionInfo* importFunctionInfo(size_t importFunctionNum)
195 {
196 RELEASE_ASSERT(importFunctionNum < m_numImportFunctions);
197 return &bitwise_cast<ImportFunctionInfo*>(bitwise_cast<char*>(this) + offsetOfTail())[importFunctionNum];
198 }
199 static size_t offsetOfTargetInstance(size_t importFunctionNum) { return offsetOfTail() + importFunctionNum * sizeof(ImportFunctionInfo) + OBJECT_OFFSETOF(ImportFunctionInfo, targetInstance); }
200 static size_t offsetOfWasmEntrypointLoadLocation(size_t importFunctionNum) { return offsetOfTail() + importFunctionNum * sizeof(ImportFunctionInfo) + OBJECT_OFFSETOF(ImportFunctionInfo, wasmEntrypointLoadLocation); }
201 static size_t offsetOfWasmToEmbedderStub(size_t importFunctionNum) { return offsetOfTail() + importFunctionNum * sizeof(ImportFunctionInfo) + OBJECT_OFFSETOF(ImportFunctionInfo, wasmToEmbedderStub); }
202 static size_t offsetOfImportFunction(size_t importFunctionNum) { return offsetOfTail() + importFunctionNum * sizeof(ImportFunctionInfo) + OBJECT_OFFSETOF(ImportFunctionInfo, importFunction); }
203 template<typename T> T* importFunction(unsigned importFunctionNum) { return reinterpret_cast<T*>(&importFunctionInfo(importFunctionNum)->importFunction); }
204
205 static_assert(sizeof(ImportFunctionInfo) == WTF::roundUpToMultipleOf<sizeof(uint64_t)>(sizeof(ImportFunctionInfo)), "We rely on this for the alignment to be correct");
206 static constexpr size_t offsetOfTablePtr(unsigned numImportFunctions, unsigned i) { return offsetOfTail() + sizeof(ImportFunctionInfo) * numImportFunctions + sizeof(Table*) * i; }
