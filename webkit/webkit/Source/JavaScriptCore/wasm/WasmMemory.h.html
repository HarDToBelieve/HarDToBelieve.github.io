53class MemoryHandle final : public ThreadSafeRefCounted<MemoryHandle> {
54 WTF_MAKE_NONCOPYABLE(MemoryHandle);
55 WTF_MAKE_FAST_ALLOCATED;
56 friend LLIntOffsetsExtractor;
57public:
58 MemoryHandle(void*, size_t size, size_t mappedCapacity, PageCount initial, PageCount maximum, MemorySharingMode, MemoryMode);
59 JS_EXPORT_PRIVATE ~MemoryHandle();
60
61 void* memory() const;
62 size_t size() const { return m_size; }
63 size_t mappedCapacity() const { return m_mappedCapacity; }
64 size_t boundsCheckingSize() const { return m_mappedCapacity; }
65 PageCount initial() const { return m_initial; }
66 PageCount maximum() const { return m_maximum; }
67 MemorySharingMode sharingMode() const { return m_sharingMode; }
68 MemoryMode mode() const { return m_mode; }
69 static ptrdiff_t offsetOfSize() { return OBJECT_OFFSETOF(MemoryHandle, m_size); }
70 Lock& lock() { return m_lock; }
71
72 void growToSize(size_t size)
73 {
74 m_size = size;
75 }
76
77private:
78 using CagedMemory = CagedPtr<Gigacage::Primitive, void, tagCagedPtr>;
79
80 Lock m_lock;
81 MemorySharingMode m_sharingMode { MemorySharingMode::Default };
82 MemoryMode m_mode { MemoryMode::BoundsChecking };
83 CagedMemory m_memory;
84 size_t m_size { 0 };
85 size_t m_mappedCapacity { 0 };
86 PageCount m_initial;
87 PageCount m_maximum;
88};

90class Memory final : public RefCounted<Memory> {
91 WTF_MAKE_NONCOPYABLE(Memory);
92 WTF_MAKE_FAST_ALLOCATED;
93 friend LLIntOffsetsExtractor;
94public:
95 void dump(WTF::PrintStream&) const;
96
97 explicit operator bool() const { return !!m_handle->memory(); }
98
99 enum NotifyPressure { NotifyPressureTag };
100 enum SyncTryToReclaim { SyncTryToReclaimTag };
101 enum GrowSuccess { GrowSuccessTag };
102
103 static Ref<Memory> create();
104 JS_EXPORT_PRIVATE static Ref<Memory> create(Ref<MemoryHandle>&&, WTF::Function<void(NotifyPressure)>&& notifyMemoryPressure, WTF::Function<void(SyncTryToReclaim)>&& syncTryToReclaimMemory, WTF::Function<void(GrowSuccess, PageCount, PageCount)>&& growSuccessCallback);
105 static RefPtr<Memory> tryCreate(PageCount initial, PageCount maximum, MemorySharingMode, WTF::Function<void(NotifyPressure)>&& notifyMemoryPressure, WTF::Function<void(SyncTryToReclaim)>&& syncTryToReclaimMemory, WTF::Function<void(GrowSuccess, PageCount, PageCount)>&& growSuccessCallback);
106
107 JS_EXPORT_PRIVATE ~Memory();
108
109 static size_t fastMappedRedzoneBytes();
110 static size_t fastMappedBytes(); // Includes redzone.
111 static bool addressIsInGrowableOrFastMemory(void*);
112
113 void* memory() const { return m_handle->memory(); }
114 size_t size() const { return m_handle->size(); }
115 PageCount sizeInPages() const { return PageCount::fromBytes(size()); }
116 size_t boundsCheckingSize() const { return m_handle->boundsCheckingSize(); }
117 PageCount initial() const { return m_handle->initial(); }
118 PageCount maximum() const { return m_handle->maximum(); }
119 MemoryHandle& handle() { return m_handle.get(); }
120
121 MemorySharingMode sharingMode() const { return m_handle->sharingMode(); }
122 MemoryMode mode() const { return m_handle->mode(); }
123
124 enum class GrowFailReason {
125 InvalidDelta,
126 InvalidGrowSize,
127 WouldExceedMaximum,
128 OutOfMemory,
129 };
130 Expected<PageCount, GrowFailReason> grow(PageCount);
131 bool fill(uint32_t, uint8_t, uint32_t);
132 bool copy(uint32_t, uint32_t, uint32_t);
133 bool init(uint32_t, const uint8_t*, uint32_t);
134
135 void registerInstance(Instance*);
136
137 void check() { ASSERT(!deletionHasBegun()); }
138
139 static ptrdiff_t offsetOfHandle() { return OBJECT_OFFSETOF(Memory, m_handle); }
140
141private:
142 Memory();
143 Memory(Ref<MemoryHandle>&&, WTF::Function<void(NotifyPressure)>&& notifyMemoryPressure, WTF::Function<void(SyncTryToReclaim)>&& syncTryToReclaimMemory, WTF::Function<void(GrowSuccess, PageCount, PageCount)>&& growSuccessCallback);
144 Memory(PageCount initial, PageCount maximum, MemorySharingMode, WTF::Function<void(NotifyPressure)>&& notifyMemoryPressure, WTF::Function<void(SyncTryToReclaim)>&& syncTryToReclaimMemory, WTF::Function<void(GrowSuccess, PageCount, PageCount)>&& growSuccessCallback);
145
146 Expected<PageCount, GrowFailReason> growShared(PageCount);
147
148 Ref<MemoryHandle> m_handle;
149 WTF::Function<void(NotifyPressure)> m_notifyMemoryPressure;
150 WTF::Function<void(SyncTryToReclaim)> m_syncTryToReclaimMemory;
151 WTF::Function<void(GrowSuccess, PageCount, PageCount)> m_growSuccessCallback;
152 Vector<WeakPtr<Instance>> m_instances;
153};
