26#pragma once
27
28#if ENABLE(WEBASSEMBLY)
29
30#include "WasmFormat.h"
31
32#include <wtf/BitVector.h>
33
34namespace JSC { namespace Wasm {
35
36struct ModuleInformation : public ThreadSafeRefCounted<ModuleInformation> {
37 ModuleInformation();
38 ModuleInformation(const ModuleInformation&) = delete;
39 ModuleInformation(ModuleInformation&&) = delete;
40
41 static Ref<ModuleInformation> create()
42 {
43 return adoptRef(*new ModuleInformation);
44 }
45
46 JS_EXPORT_PRIVATE ~ModuleInformation();
47
48 size_t functionIndexSpaceSize() const { return importFunctionSignatureIndices.size() + internalFunctionSignatureIndices.size(); }
49 bool isImportedFunctionFromFunctionIndexSpace(size_t functionIndex) const
50 {
51 ASSERT(functionIndex < functionIndexSpaceSize());
52 return functionIndex < importFunctionSignatureIndices.size();
53 }
54 SignatureIndex signatureIndexFromFunctionIndexSpace(size_t functionIndex) const
55 {
56 return isImportedFunctionFromFunctionIndexSpace(functionIndex)
57 ? importFunctionSignatureIndices[functionIndex]
58 : internalFunctionSignatureIndices[functionIndex - importFunctionSignatureIndices.size()];
59 }
60
61 uint32_t importFunctionCount() const { return importFunctionSignatureIndices.size(); }
62 uint32_t internalFunctionCount() const { return internalFunctionSignatureIndices.size(); }
63
64 // Currently, our wasm implementation allows only one memory and table.
65 // If we need to remove this limitation, we would have MemoryInformation and TableInformation in the Vectors.
66 uint32_t memoryCount() const { return memory ? 1 : 0; }
67 uint32_t tableCount() const { return tables.size(); }
68 uint32_t elementCount() const { return elements.size(); }
69 uint32_t dataSegmentsCount() const { return numberOfDataSegments; }
70
71 const TableInformation& table(unsigned index) const { return tables[index]; }
72
73 const BitVector& referencedFunctions() const { return m_referencedFunctions; }
74 void addReferencedFunction(unsigned index) const { m_referencedFunctions.set(index); }
75
76 bool isDeclaredFunction(uint32_t index) const { return m_declaredFunctions.contains(index); }
77 void addDeclaredFunction(uint32_t index) { m_declaredFunctions.set(index); }
78
79 Vector<Import> imports;
80 Vector<SignatureIndex> importFunctionSignatureIndices;
81 Vector<SignatureIndex> internalFunctionSignatureIndices;
82 Vector<Ref<Signature>> usedSignatures;
83
84 MemoryInformation memory;
85
86 Vector<FunctionData> functions;
87
88 Vector<Export> exports;
89 std::optional<uint32_t> startFunctionIndexSpace;
90 Vector<Segment::Ptr> data;
91 Vector<Element> elements;
92 Vector<TableInformation> tables;
93 Vector<GlobalInformation> globals;
94 unsigned firstInternalGlobal { 0 };
95 uint32_t codeSectionSize { 0 };
96 Vector<CustomSection> customSections;
97 Ref<NameSection> nameSection;
98 uint32_t numberOfDataSegments { 0 };
99
100 BitVector m_declaredFunctions;
101 mutable BitVector m_referencedFunctions;
102};
103
104
105} } // namespace JSC::Wasm
106
107#endif // ENABLE(WEBASSEMBLY)
