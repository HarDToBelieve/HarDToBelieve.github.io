28#if ENABLE(WEBASSEMBLY)
29
30#include "B3Origin.h"
31#include "WasmFormat.h"
32
33#include <wtf/ForbidHeapAllocation.h>
34
35namespace JSC { namespace Wasm {
36
37class OpcodeOrigin {
38 WTF_FORBID_HEAP_ALLOCATION;
39public:
40 OpcodeOrigin() = default;
41 OpcodeOrigin(OpType opcode, size_t offset)
42 {
43 ASSERT(static_cast<uint32_t>(offset) == offset);
44 packedData = (static_cast<uint64_t>(opcode) << 32) | offset;
45 }
46
47 void dump(PrintStream&) const;
48
49 OpType opcode() const { return static_cast<OpType>(packedData >> 32); }
50 size_t location() const { return static_cast<uint32_t>(packedData); }
51
52private:
53 static_assert(sizeof(void*) == sizeof(uint64_t), "this packing doesn't work if this isn't the case");
54 uint64_t packedData { 0 };
55};
56
57} } // namespace JSC::Wasm

59#endif // ENABLE(WEBASSEMBLY)
