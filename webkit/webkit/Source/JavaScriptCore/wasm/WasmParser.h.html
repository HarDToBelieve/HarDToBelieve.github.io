54template<typename SuccessType>
55class Parser {
56public:
57 typedef String ErrorType;
58 typedef Unexpected<ErrorType> UnexpectedResult;
59 typedef Expected<void, ErrorType> PartialResult;
60 typedef Expected<SuccessType, ErrorType> Result;
61
62 const uint8_t* source() const { return m_source; }
63 size_t length() const { return m_sourceLength; }
64 size_t offset() const { return m_offset; }
65
66protected:
67 Parser(const uint8_t*, size_t);
68
69 bool WARN_UNUSED_RETURN consumeCharacter(char);
70 bool WARN_UNUSED_RETURN consumeString(const char*);
71 bool WARN_UNUSED_RETURN consumeUTF8String(Name&, size_t);
72
73 bool WARN_UNUSED_RETURN parseVarUInt1(uint8_t&);
74 bool WARN_UNUSED_RETURN parseInt7(int8_t&);
75 bool WARN_UNUSED_RETURN peekInt7(int8_t&);
76 bool WARN_UNUSED_RETURN parseUInt7(uint8_t&);
77 bool WARN_UNUSED_RETURN parseUInt8(uint8_t&);
78 bool WARN_UNUSED_RETURN parseUInt32(uint32_t&);
79 bool WARN_UNUSED_RETURN parseUInt64(uint64_t&);
80 bool WARN_UNUSED_RETURN parseVarUInt32(uint32_t&);
81 bool WARN_UNUSED_RETURN parseVarUInt64(uint64_t&);
82
83 bool WARN_UNUSED_RETURN parseVarInt32(int32_t&);
84 bool WARN_UNUSED_RETURN parseVarInt64(int64_t&);
85
86 PartialResult WARN_UNUSED_RETURN parseBlockSignature(const ModuleInformation&, BlockSignature&);
87 bool WARN_UNUSED_RETURN parseValueType(const ModuleInformation&, Type&);
88 bool WARN_UNUSED_RETURN parseRefType(const ModuleInformation&, Type&);
89 bool WARN_UNUSED_RETURN parseExternalKind(ExternalKind&);
90
91 size_t m_offset = 0;
92
93 template <typename ...Args>
94 NEVER_INLINE UnexpectedResult WARN_UNUSED_RETURN fail(Args... args) const
95 {
96 using namespace FailureHelper; // See ADL comment in namespace above.
97 return UnexpectedResult(makeString("WebAssembly.Module doesn't parse at byte "_s, String::number(m_offset), ": "_s, makeString(args)...));
98 }
99#define WASM_PARSER_FAIL_IF(condition, ...) do { \
100 if (UNLIKELY(condition)) \
101 return fail(__VA_ARGS__); \
102 } while (0)
103
104#define WASM_FAIL_IF_HELPER_FAILS(helper) do { \
105 auto helperResult = helper; \
106 if (UNLIKELY(!helperResult)) \
107 return makeUnexpected(WTFMove(helperResult.error())); \
108 } while (0)
109
110private:
111 const uint8_t* m_source;
112 size_t m_sourceLength;
113 // We keep a local reference to the global table so we don't have to fetch it to find thunk signatures.
114 const SignatureInformation& m_signatureInformation;
115};

118ALWAYS_INLINE Parser<SuccessType>::Parser(const uint8_t* sourceBuffer, size_t sourceLength)

121 , m_signatureInformation(SignatureInformation::singleton())

125template<typename SuccessType>
126ALWAYS_INLINE bool Parser<SuccessType>::consumeCharacter(char c)
127{
128 if (m_offset >= length())
129 return false;
130 if (c == source()[m_offset]) {
131 m_offset++;
132 return true;
133 }
134 return false;
135}
136
137template<typename SuccessType>
138ALWAYS_INLINE bool Parser<SuccessType>::consumeString(const char* str)
139{
140 unsigned start = m_offset;
141 if (m_offset >= length())
142 return false;
143 for (size_t i = 0; str[i]; i++) {
144 if (!consumeCharacter(str[i])) {
145 m_offset = start;
146 return false;
147 }
148 }
149 return true;
150}

152template<typename SuccessType>
153ALWAYS_INLINE bool Parser<SuccessType>::consumeUTF8String(Name& result, size_t stringLength)
154{
155 if (length() < stringLength || m_offset > length() - stringLength)
156 return false;
157 if (stringLength > maxStringSize)
158 return false;
159 if (!result.tryReserveCapacity(stringLength))
160 return false;
161
162 const uint8_t* stringStart = source() + m_offset;
163
164 // We don't cache the UTF-16 characters since it seems likely the string is ASCII.
165 if (UNLIKELY(!charactersAreAllASCII(stringStart, stringLength))) {
166 Vector<UChar, 1024> buffer(stringLength);
167 UChar* bufferStart = buffer.data();
168
169 UChar* bufferCurrent = bufferStart;
170 const char* stringCurrent = reinterpret_cast<const char*>(stringStart);
171 if (!WTF::Unicode::convertUTF8ToUTF16(stringCurrent, reinterpret_cast<const char *>(stringStart + stringLength), &bufferCurrent, bufferCurrent + buffer.size()))
172 return false;
173 }
174
175 result.grow(stringLength);
176 memcpy(result.data(), stringStart, stringLength);
177 m_offset += stringLength;
178 return true;
179}

205template<typename SuccessType>
206ALWAYS_INLINE bool Parser<SuccessType>::parseUInt32(uint32_t& result)
207{
208 if (length() < 4 || m_offset > length() - 4)
209 return false;
210 result = *reinterpret_cast<const uint32_t*>(source() + m_offset);
211 m_offset += 4;
212 return true;
213}

225template<typename SuccessType>
226ALWAYS_INLINE bool Parser<SuccessType>::parseUInt8(uint8_t& result)
227{
228 if (m_offset >= length())
229 return false;
230 result = source()[m_offset++];
231 return true;
232}

234template<typename SuccessType>
235ALWAYS_INLINE bool Parser<SuccessType>::parseInt7(int8_t& result)
236{
237 if (m_offset >= length())
238 return false;
239 uint8_t v = source()[m_offset++];
240 result = (v & 0x40) ? WTF::bitwise_cast<int8_t>(uint8_t(v | 0x80)) : v;
241 return (v & 0x80) == 0;
242}
243
244template<typename SuccessType>
245ALWAYS_INLINE bool Parser<SuccessType>::peekInt7(int8_t& result)
246{
247 if (m_offset >= length())
248 return false;
249 uint8_t v = source()[m_offset];
250 result = (v & 0x40) ? WTF::bitwise_cast<int8_t>(uint8_t(v | 0x80)) : v;
251 return (v & 0x80) == 0;
252}

254template<typename SuccessType>
255ALWAYS_INLINE bool Parser<SuccessType>::parseUInt7(uint8_t& result)
256{
257 if (m_offset >= length())
258 return false;
259 result = source()[m_offset++];
260 return result < 0x80;

263template<typename SuccessType>

268 return false;

275template<typename SuccessType>
276ALWAYS_INLINE typename Parser<SuccessType>::PartialResult Parser<SuccessType>::parseBlockSignature(const ModuleInformation& info, BlockSignature& result)
277{
278 int8_t typeKind;
279 if (peekInt7(typeKind) && isValidTypeKind(typeKind)) {
280 Type type = {static_cast<TypeKind>(typeKind), Nullable::Yes, 0};
281 WASM_PARSER_FAIL_IF(!(isValueType(type) || type.isVoid()), "result type of block: ", makeString(type.kind), " is not a value type or Void");
282 result = m_signatureInformation.thunkFor(type);
283 m_offset++;
284 return { };
285 }
286
287 WASM_PARSER_FAIL_IF(!Options::useWebAssemblyMultiValues(), "Type table indices for block signatures are not supported yet");
288
289 int64_t index;
290 WASM_PARSER_FAIL_IF(!parseVarInt64(index), "Block-like instruction doesn't return value type but can't decode type section index");
291 WASM_PARSER_FAIL_IF(index < 0, "Block-like instruction signature index is negative");
292 WASM_PARSER_FAIL_IF(static_cast<size_t>(index) >= info.usedSignatures.size(), "Block-like instruction signature index is out of bounds. Index: ", index, " type index space: ", info.usedSignatures.size());
293
294 result = &info.usedSignatures[index].get();
295 return { };
296}

298template<typename SuccessType>
299ALWAYS_INLINE bool Parser<SuccessType>::parseValueType(const ModuleInformation& info, Type& result)
300{
301 int8_t kind;
302 if (!parseInt7(kind))
303 return false;
304 if (!isValidTypeKind(kind))
305 return false;
306
307 TypeKind typeKind = static_cast<TypeKind>(kind);
308 bool isNullable = true;
309 SignatureIndex sigIndex = 0;
310 if (typeKind == TypeKind::Ref || typeKind == TypeKind::RefNull) {
311 if (!Options::useWebAssemblyTypedFunctionReferences())
312 return false;
313
314 int32_t heapType;
315 isNullable = typeKind == TypeKind::RefNull;
316
317 if (!parseVarInt32(heapType))
318 return false;
319 if (heapType < 0) {
320 TypeKind heapKind = static_cast<TypeKind>(heapType);
321 if (!isValidHeapTypeKind(heapKind))
322 return false;
323 typeKind = heapKind;
324 } else {
325 typeKind = TypeKind::TypeIdx;
326 if (static_cast<size_t>(heapType) >= info.usedSignatures.size())
327 return false;
328 sigIndex = SignatureInformation::get(info.usedSignatures[heapType].get());
329 }
330 }
331
332 Type type = { typeKind, static_cast<Nullable>(isNullable), sigIndex };
333 if (!isValueType(type))
334 return false;
335 result = type;
336 return true;
337}
338
339template<typename SuccessType>
340ALWAYS_INLINE bool Parser<SuccessType>::parseRefType(const ModuleInformation& info, Type& result)
341{
342 const bool parsed = parseValueType(info, result);
343 return parsed && isRefType(result);
344}

346template<typename SuccessType>
347ALWAYS_INLINE bool Parser<SuccessType>::parseExternalKind(ExternalKind& result)
348{
349 uint8_t value;
350 if (!parseUInt7(value))
351 return false;
352 if (!isValidExternalKind(value))
353 return false;
354 result = static_cast<ExternalKind>(value);
355 return true;
356}

358ALWAYS_INLINE I32InitExpr makeI32InitExpr(uint8_t opcode, uint32_t bits)
359{
360 RELEASE_ASSERT(opcode == I32Const || opcode == GetGlobal);
361 if (opcode == I32Const)
362 return I32InitExpr::constValue(bits);
363 return I32InitExpr::globalImport(bits);
