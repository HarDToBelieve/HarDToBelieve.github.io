38namespace JSC { namespace Wasm {
39
40class Instance;
41class FuncRefTable;
42
43class Table : public ThreadSafeRefCounted<Table> {
44 WTF_MAKE_NONCOPYABLE(Table);
45 WTF_MAKE_FAST_ALLOCATED(Table);
46public:
47 static RefPtr<Table> tryCreate(uint32_t initial, std::optional<uint32_t> maximum, TableElementType);
48
49 JS_EXPORT_PRIVATE ~Table() = default;
50
51 std::optional<uint32_t> maximum() const { return m_maximum; }
52 uint32_t length() const { return m_length; }
53
54 static ptrdiff_t offsetOfLength() { return OBJECT_OFFSETOF(Table, m_length); }
55 static ptrdiff_t offsetOfMask() { return OBJECT_OFFSETOF(Table, m_mask); }
56
57 static uint32_t allocatedLength(uint32_t length);
58 uint32_t mask() const { return m_mask; }
59
60 template<typename T> T* owner() const { return reinterpret_cast<T*>(m_owner); }
61 void setOwner(JSObject* owner)
62 {
63 ASSERT(!m_owner);
64 ASSERT(owner);
65 m_owner = owner;
66 }
67
68 TableElementType type() const { return m_type; }
69 bool isExternrefTable() const { return m_type == TableElementType::Externref; }
70 bool isFuncrefTable() const { return m_type == TableElementType::Funcref; }
71 Type wasmType() const;
72 FuncRefTable* asFuncrefTable();
73
74 static bool isValidLength(uint32_t length) { return length < maxTableEntries; }
75
76 void clear(uint32_t);
77 void set(uint32_t, JSValue);
78 JSValue get(uint32_t) const;
79
80 std::optional<uint32_t> grow(uint32_t delta, JSValue defaultValue);
81 void copy(const Table* srcTable, uint32_t dstIndex, uint32_t srcIndex);
82
83 DECLARE_VISIT_AGGREGATE;
84
85protected:
86 Table(uint32_t initial, std::optional<uint32_t> maximum, TableElementType = TableElementType::Externref);
87
88 void setLength(uint32_t);
89
90 uint32_t m_length;
91 uint32_t m_mask;
92 const TableElementType m_type;
93 const std::optional<uint32_t> m_maximum;
94
95 MallocPtr<WriteBarrier<Unknown>, VMMalloc> m_jsValues;
96 JSObject* m_owner;
97};
98
99class FuncRefTable : public Table {
100public:
101 JS_EXPORT_PRIVATE ~FuncRefTable() = default;
102
103 void setFunction(uint32_t, JSObject*, WasmToWasmImportableFunction, Instance*);
104 const WasmToWasmImportableFunction& function(uint32_t) const;
105 Instance* instance(uint32_t) const;
106
107 void copyFunction(const FuncRefTable* srcTable, uint32_t dstIndex, uint32_t srcIndex);
108
109 static ptrdiff_t offsetOfFunctions() { return OBJECT_OFFSETOF(FuncRefTable, m_importableFunctions); }
110 static ptrdiff_t offsetOfInstances() { return OBJECT_OFFSETOF(FuncRefTable, m_instances); }
111
112private:
113 FuncRefTable(uint32_t initial, std::optional<uint32_t> maximum);
114
115 MallocPtr<WasmToWasmImportableFunction, VMMalloc> m_importableFunctions;
116 // call_indirect needs to do an Instance check to potentially context switch when calling a function to another instance. We can hold raw pointers to Instance here because the embedder ensures that Table keeps all the instances alive. We couldn't hold a Ref here because it would cause cycles.
117 MallocPtr<Instance*, VMMalloc> m_instances;
118
119 friend class Table;
120};
121
122} } // namespace JSC::Wasm
