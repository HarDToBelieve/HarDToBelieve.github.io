31#include "FPRInfo.h"

33#include "Reg.h"
34#include <wtf/PrintStream.h>

36namespace JSC {
37
38namespace Wasm {
39
40class ValueLocation {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 enum Kind : uint8_t {
44 Register,
45 Stack,
46 StackArgument,
47 };
48
49 ValueLocation()
50 : m_kind(Register)
51 {
52 }
53
54 explicit ValueLocation(Reg reg)
55 : m_kind(Register)
56 {
57 u.reg = reg;
58 }
59
60 ValueLocation(const ValueLocation&) = default;
61
62 static ValueLocation reg(Reg reg)
63 {
64 return ValueLocation(reg);
65 }
66
67 static ValueLocation stack(intptr_t offsetFromFP)
68 {
69 ValueLocation result;
70 result.m_kind = Stack;
71 result.u.offsetFromFP = offsetFromFP;
72 return result;
73 }
74
75 static ValueLocation stackArgument(intptr_t offsetFromSP)
76 {
77 ValueLocation result;
78 result.m_kind = StackArgument;
79 result.u.offsetFromSP = offsetFromSP;
80 return result;
81 }
82
83 Kind kind() const { return m_kind; }
84
85 bool isReg() const { return kind() == Register; }
86
87 Reg reg() const
88 {
89 ASSERT(isReg());
90 return u.reg;
91 }
92
93 bool isGPR() const { return isReg() && reg().isGPR(); }
94 bool isFPR() const { return isReg() && reg().isFPR(); }
95
96 GPRReg gpr() const { return reg().gpr(); }
97 FPRReg fpr() const { return reg().fpr(); }
98
99 bool isStack() const { return kind() == Stack; }
100
101 intptr_t offsetFromFP() const
102 {
103 ASSERT(isStack());
104 return u.offsetFromFP;
105 }
106
107 bool isStackArgument() const { return kind() == StackArgument; }
108
109 intptr_t offsetFromSP() const
110 {
111 ASSERT(isStackArgument());
112 return u.offsetFromSP;
113 }
114
115 JS_EXPORT_PRIVATE void dump(PrintStream&) const;
116
117private:
118 union U {
119 Reg reg;
120 intptr_t offsetFromFP;
121 intptr_t offsetFromSP;
122
123 U()
124 {
125 memset(static_cast<void*>(this), 0, sizeof(*this));
126 }
127 } u;
128 Kind m_kind;
129};
130
131} } // namespace JSC::Wasm
