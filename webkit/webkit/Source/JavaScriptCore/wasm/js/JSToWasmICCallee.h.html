30namespace JSC {
31
32#if ENABLE(WEBASSEMBLY)
33
34class WebAssemblyFunction;
35
36// FIXME: Remove this type. Unwinding should just work by using WebAssemblyFunction instead of JSToWasmICCallee.
37// https://bugs.webkit.org/show_bug.cgi?id=204960
38class JSToWasmICCallee final : public JSCallee {
39public:
40 using Base = JSCallee;
41
42 template<typename CellType, SubspaceAccess mode>
43 static IsoSubspace* subspaceFor(VM& vm)
44 {
45 return vm.jsToWasmICCalleeSpace<mode>();
46 }
47
48 DECLARE_INFO;
49
50 static JSToWasmICCallee* create(VM&, JSGlobalObject*, WebAssemblyFunction*);
51 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
52
53 WebAssemblyFunction* function() { return m_function.get(); }
54
55private:
56 JSToWasmICCallee(VM& vm, JSGlobalObject* globalObject, Structure* structure)
57 : Base(vm, globalObject, structure)
58 { }
59 DECLARE_VISIT_CHILDREN;
60
61 WriteBarrier<WebAssemblyFunction> m_function;
62};
63
64#endif // ENABLE(WEBASSEMBLY)
