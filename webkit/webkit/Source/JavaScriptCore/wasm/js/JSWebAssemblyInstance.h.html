40namespace JSC {
41
42class JSModuleNamespaceObject;
43class JSWebAssemblyModule;
44class WebAssemblyModuleRecord;
45
46namespace Wasm {
47class CodeBlock;
48}
49
50class JSWebAssemblyInstance final : public JSNonFinalObject {
51public:
52 using Base = JSNonFinalObject;
53 static constexpr bool needsDestruction = true;
54 static void destroy(JSCell*);
55
56 template<typename CellType, SubspaceAccess mode>
57 static IsoSubspace* subspaceFor(VM& vm)
58 {
59 return vm.webAssemblyInstanceSpace<mode>();
60 }
61
62 static Identifier createPrivateModuleKey();
63
64 static JSWebAssemblyInstance* tryCreate(VM&, JSGlobalObject*, const Identifier& moduleKey, JSWebAssemblyModule*, JSObject* importObject, Structure*, Ref<Wasm::Module>&&, Wasm::CreationMode);
65 static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
66
67 DECLARE_EXPORT_INFO;
68
69 void finalizeCreation(VM&, JSGlobalObject*, Ref<Wasm::CodeBlock>&&, JSObject* importObject, Wasm::CreationMode);
70
71 Wasm::Instance& instance() { return m_instance.get(); }
72 WebAssemblyModuleRecord* moduleRecord() { return m_moduleRecord.get(); }
73
74 JSWebAssemblyMemory* memory() { return m_memory.get(); }
75 void setMemory(VM& vm, JSWebAssemblyMemory* value) {
76 ASSERT(!memory());
77 m_memory.set(vm, this, value);
78 instance().setMemory(makeRef(memory()->memory()));
79 }
80 Wasm::MemoryMode memoryMode() { return memory()->memory().mode(); }
81
82 JSWebAssemblyTable* table(unsigned i) { return m_tables[i].get(); }
83 void setTable(VM& vm, uint32_t index, JSWebAssemblyTable* value)
84 {
85 ASSERT(index < m_tables.size());
86 ASSERT(!table(index));
87 m_tables[index].set(vm, this, value);
88 instance().setTable(index, makeRef(*table(index)->table()));
89 }
90
91 void linkGlobal(VM& vm, uint32_t index, JSWebAssemblyGlobal* value)
92 {
93 ASSERT(value == value->global()->owner<JSWebAssemblyGlobal>());
94 instance().linkGlobal(index, makeRef(*value->global()));
95 vm.heap.writeBarrier(this, value);
96 }
97
98 JSGlobalObject* globalObject() const { return m_globalObject.get(); }
99 JSWebAssemblyModule* module() const { return m_module.get(); }
100
101 static size_t offsetOfInstance() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_instance); }
102 static size_t offsetOfModule() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_module); }
103 static size_t offsetOfGlobalObject() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_globalObject); }
104 static size_t offsetOfVM() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_vm); }
105
106private:
107 JSWebAssemblyInstance(VM&, Structure*, Ref<Wasm::Instance>&&);
108 void finishCreation(VM&, JSWebAssemblyModule*, WebAssemblyModuleRecord*);
109 DECLARE_VISIT_CHILDREN;
110
111 Ref<Wasm::Instance> m_instance;
112 VM* m_vm;
113
114 WriteBarrier<JSGlobalObject> m_globalObject;
115 WriteBarrier<JSWebAssemblyModule> m_module;
116 WriteBarrier<JSWebAssemblyCodeBlock> m_codeBlock;
117 WriteBarrier<WebAssemblyModuleRecord> m_moduleRecord;
118 WriteBarrier<JSWebAssemblyMemory> m_memory;
119 FixedVector<WriteBarrier<JSWebAssemblyTable>> m_tables;
120};
121
122} // namespace JSC

124#endif // ENABLE(WEBASSEMBLY)
