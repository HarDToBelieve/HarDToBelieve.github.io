26#pragma once
27
28#if ENABLE(WEBASSEMBLY)
29
30#include "JSFunction.h"
31#include "WasmFormat.h"
32
33namespace JSC {
34
35class JSGlobalObject;
36class JSWebAssemblyInstance;
37using Wasm::WasmToWasmImportableFunction;
38
39class WebAssemblyFunctionBase : public JSFunction {
40public:
41 using Base = JSFunction;
42
43 static constexpr unsigned StructureFlags = Base::StructureFlags;
44
45 DECLARE_INFO;
46
47 JSWebAssemblyInstance* instance() const { return m_instance.get(); }
48
49 Wasm::SignatureIndex signatureIndex() const { return m_importableFunction.signatureIndex; }
50 WasmToWasmImportableFunction::LoadLocation entrypointLoadLocation() const { return m_importableFunction.entrypointLoadLocation; }
51 WasmToWasmImportableFunction importableFunction() const { return m_importableFunction; }
52
53 static ptrdiff_t offsetOfInstance() { return OBJECT_OFFSETOF(WebAssemblyFunctionBase, m_instance); }
54
55 static ptrdiff_t offsetOfSignatureIndex() { return OBJECT_OFFSETOF(WebAssemblyFunctionBase, m_importableFunction) + WasmToWasmImportableFunction::offsetOfSignatureIndex(); }
56
57 static ptrdiff_t offsetOfEntrypointLoadLocation() { return OBJECT_OFFSETOF(WebAssemblyFunctionBase, m_importableFunction) + WasmToWasmImportableFunction::offsetOfEntrypointLoadLocation(); }
58
59protected:
60 DECLARE_VISIT_CHILDREN;
61 void finishCreation(VM&, NativeExecutable*, unsigned length, const String& name, JSWebAssemblyInstance*);
62 WebAssemblyFunctionBase(VM&, NativeExecutable*, JSGlobalObject*, Structure*, WasmToWasmImportableFunction);
63
64 WriteBarrier<JSWebAssemblyInstance> m_instance;
65
66 // It's safe to just hold the raw WasmToWasmImportableFunction because we have a reference
67 // to our Instance, which points to the CodeBlock, which points to the Module
68 // that exported us, which ensures that the actual Signature/code doesn't get deallocated.
69 WasmToWasmImportableFunction m_importableFunction;
70};
71
72} // namespace JSC
73
74#endif // ENABLE(WEBASSEMBLY)
