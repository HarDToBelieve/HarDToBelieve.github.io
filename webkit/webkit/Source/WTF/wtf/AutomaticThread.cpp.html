49void AutomaticThreadCondition::notifyOne(const AbstractLocker& locker)
50{
51 for (AutomaticThread* thread : m_threads) {
52 if (thread->isWaiting(locker)) {
53 thread->notify(locker);
54 return;
55 }
56 }
57
58 for (AutomaticThread* thread : m_threads) {
59 if (!thread->hasUnderlyingThread(locker)) {
60 thread->start(locker);
61 return;
62 }
63 }
64
65 m_condition.notifyOne();
66}
67
68void AutomaticThreadCondition::notifyAll(const AbstractLocker& locker)
69{
70 m_condition.notifyAll();
71
72 for (AutomaticThread* thread : m_threads) {
73 if (thread->isWaiting(locker))
74 thread->notify(locker);
75 else if (!thread->hasUnderlyingThread(locker))
76 thread->start(locker);
77 }
78}

112AutomaticThread::AutomaticThread(const AbstractLocker& locker, Box<Lock> lock, Ref<AutomaticThreadCondition>&& condition, ThreadType type, Seconds timeout)
113 : m_lock(lock)
114 , m_condition(WTFMove(condition))
115 , m_timeout(timeout)
116 , m_threadType(type)
117{
118 if (verbose)
119 dataLog(RawPointer(this), ": Allocated AutomaticThread.\n");
120 m_condition->add(locker, this);
121}

123AutomaticThread::~AutomaticThread()
124{
125 if (verbose)
126 dataLog(RawPointer(this), ": Deleting AutomaticThread.\n");
127 Locker locker { *m_lock };
128
129 // It's possible that we're in a waiting state with the thread shut down. This is a goofy way to
130 // die, but it could happen.
131 m_condition->remove(locker, this);
132}

134bool AutomaticThread::tryStop(const AbstractLocker&)
135{
136 if (!m_isRunning)
137 return true;
138 if (m_hasUnderlyingThread)
139 return false;
140 m_isRunning = false;
141 return true;
142}

144bool AutomaticThread::isWaiting(const AbstractLocker& locker)
145{
146 return hasUnderlyingThread(locker) && m_isWaiting;
147}
148
149bool AutomaticThread::notify(const AbstractLocker& locker)
150{
151 ASSERT_UNUSED(locker, hasUnderlyingThread(locker));
152 m_isWaiting = false;
153 return m_waitCondition.notifyOne();
154}

156void AutomaticThread::join()
157{
158 Locker locker { *m_lock };
159 while (m_isRunning)
160 m_isRunningCondition.wait(*m_lock);
161}

163void AutomaticThread::start(const AbstractLocker&)
164{
165 RELEASE_ASSERT(m_isRunning);
166
167 RefPtr<AutomaticThread> preserveThisForThread = this;
168
169 m_hasUnderlyingThread = true;
170
171 Thread::create(
172 name(),
173 [=] () {
174 if (verbose)
175 dataLog(RawPointer(this), ": Running automatic thread!\n");
176
177 RefPtr<AutomaticThread> thread = preserveThisForThread;
178 thread->threadDidStart();
179
180 if (ASSERT_ENABLED) {
181 Locker locker { *m_lock };
182 ASSERT(m_condition->contains(locker, this));
183 }
184
185 auto stopImpl = [&] (const AbstractLocker& locker) {
186 thread->threadIsStopping(locker);
187 thread->m_hasUnderlyingThread = false;
188 };
189
190 auto stopPermanently = [&] (const AbstractLocker& locker) {
191 m_isRunning = false;
192 m_isRunningCondition.notifyAll();
193 stopImpl(locker);
194 };
195
196 auto stopForTimeout = [&] (const AbstractLocker& locker) {
197 stopImpl(locker);
198 };
199
200 for (;;) {
201 {
202 Locker locker { *m_lock };
203 for (;;) {
204 PollResult result = poll(locker);
205 if (result == PollResult::Work)
206 break;
207 if (result == PollResult::Stop)
208 return stopPermanently(locker);
209 RELEASE_ASSERT(result == PollResult::Wait);
210
211 // Shut the thread down after a timeout.
212 m_isWaiting = true;
213 bool awokenByNotify =
214 m_waitCondition.waitFor(*m_lock, m_timeout);
215 if (verbose && !awokenByNotify && !m_isWaiting)
216 dataLog(RawPointer(this), ": waitFor timed out, but notified via m_isWaiting flag!\n");
217 if (m_isWaiting && shouldSleep(locker)) {
218 m_isWaiting = false;
219 if (verbose)
220 dataLog(RawPointer(this), ": Going to sleep!\n");
221 // It's important that we don't release the lock until we have completely
222 // indicated that the thread is kaput. Otherwise we'll have a a notify
223 // race that manifests as a deadlock on VM shutdown.
224 return stopForTimeout(locker);
225 }
226 }
227 }
228
229 WorkResult result = work();
230 if (result == WorkResult::Stop) {
231 Locker locker { *m_lock };
232 return stopPermanently(locker);
233 }
234 RELEASE_ASSERT(result == WorkResult::Continue);
235 }
236 }, m_threadType)->detach();
237}
