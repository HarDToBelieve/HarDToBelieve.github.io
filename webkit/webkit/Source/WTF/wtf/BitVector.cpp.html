41void BitVector::setSlow(const BitVector& other)
42{
43 uintptr_t newBitsOrPointer;
44 if (other.isInline() || other.isEmptyOrDeletedValue())
45 newBitsOrPointer = other.m_bitsOrPointer;
46 else {
47 OutOfLineBits* newOutOfLineBits = OutOfLineBits::create(other.size());
48 memcpy(newOutOfLineBits->bits(), other.bits(), byteCount(other.size()));
49 newBitsOrPointer = bitwise_cast<uintptr_t>(newOutOfLineBits) >> 1;
50 }
51 if (!isInline() && !isEmptyOrDeletedValue())
52 OutOfLineBits::destroy(outOfLineBits());
53 m_bitsOrPointer = newBitsOrPointer;
54}
55
56void BitVector::resize(size_t numBits)
57{
58 if (numBits <= maxInlineBits()) {
59 if (isInline())
60 return;
61
62 OutOfLineBits* myOutOfLineBits = outOfLineBits();
63 m_bitsOrPointer = makeInlineBits(*myOutOfLineBits->bits());
64 OutOfLineBits::destroy(myOutOfLineBits);
65 return;
66 }
67
68 resizeOutOfLine(numBits);
69}
70
71void BitVector::clearAll()
72{
73 if (isInline())
74 m_bitsOrPointer = makeInlineBits(0);
75 else
76 memset(outOfLineBits()->bits(), 0, byteCount(size()));
77}

79BitVector::OutOfLineBits* BitVector::OutOfLineBits::create(size_t numBits)
80{
81 numBits = (numBits + bitsInPointer() - 1) & ~(static_cast<size_t>(bitsInPointer()) - 1);
82 size_t size = sizeof(OutOfLineBits) + sizeof(uintptr_t) * (numBits / bitsInPointer());
83 OutOfLineBits* result = new (NotNull, BitVectorMalloc::malloc(size)) OutOfLineBits(numBits);
84 return result;
85}

87void BitVector::OutOfLineBits::destroy(OutOfLineBits* outOfLineBits)
88{
89 BitVectorMalloc::free(outOfLineBits);
90}

92void BitVector::resizeOutOfLine(size_t numBits)
93{
94 ASSERT(numBits > maxInlineBits());
95 OutOfLineBits* newOutOfLineBits = OutOfLineBits::create(numBits);
96 size_t newNumWords = newOutOfLineBits->numWords();
97 if (isInline()) {
98 // Make sure that all of the bits are zero in case we do a no-op resize.
99 *newOutOfLineBits->bits() = m_bitsOrPointer & ~(static_cast<uintptr_t>(1) << maxInlineBits());
100 memset(newOutOfLineBits->bits() + 1, 0, (newNumWords - 1) * sizeof(void*));
101 } else {
102 if (numBits > size()) {
103 size_t oldNumWords = outOfLineBits()->numWords();
104 memcpy(newOutOfLineBits->bits(), outOfLineBits()->bits(), oldNumWords * sizeof(void*));
105 memset(newOutOfLineBits->bits() + oldNumWords, 0, (newNumWords - oldNumWords) * sizeof(void*));
106 } else
107 memcpy(newOutOfLineBits->bits(), outOfLineBits()->bits(), newOutOfLineBits->numWords() * sizeof(void*));
108 OutOfLineBits::destroy(outOfLineBits());
109 }
110 m_bitsOrPointer = bitwise_cast<uintptr_t>(newOutOfLineBits) >> 1;
111}

113void BitVector::mergeSlow(const BitVector& other)
114{
115 if (other.isInline()) {
116 ASSERT(!isInline());
117 *bits() |= cleanseInlineBits(other.m_bitsOrPointer);
118 return;
119 }
120
121 ensureSize(other.size());
122 ASSERT(!isInline());
123 ASSERT(!other.isInline());
124
125 OutOfLineBits* a = outOfLineBits();
126 const OutOfLineBits* b = other.outOfLineBits();
127 for (unsigned i = a->numWords(); i--;)
128 a->bits()[i] |= b->bits()[i];
129}

131void BitVector::filterSlow(const BitVector& other)
132{
133 if (other.isInline()) {
134 ASSERT(!isInline());
135 *bits() &= cleanseInlineBits(other.m_bitsOrPointer);
136 return;
137 }
138
139 if (isInline()) {
140 ASSERT(!other.isInline());
141 m_bitsOrPointer &= *other.outOfLineBits()->bits();
142 m_bitsOrPointer |= (static_cast<uintptr_t>(1) << maxInlineBits());
143 ASSERT(isInline());
144 return;
145 }
146
147 OutOfLineBits* a = outOfLineBits();
148 const OutOfLineBits* b = other.outOfLineBits();
149 for (unsigned i = std::min(a->numWords(), b->numWords()); i--;)
150 a->bits()[i] &= b->bits()[i];
151
152 for (unsigned i = b->numWords(); i < a->numWords(); ++i)
153 a->bits()[i] = 0;
154}

156void BitVector::excludeSlow(const BitVector& other)
157{
158 if (other.isInline()) {
159 ASSERT(!isInline());
160 *bits() &= ~cleanseInlineBits(other.m_bitsOrPointer);
161 return;
162 }
163
164 if (isInline()) {
165 ASSERT(!other.isInline());
166 m_bitsOrPointer &= ~*other.outOfLineBits()->bits();
167 m_bitsOrPointer |= (static_cast<uintptr_t>(1) << maxInlineBits());
168 ASSERT(isInline());
169 return;
170 }
171
172 OutOfLineBits* a = outOfLineBits();
173 const OutOfLineBits* b = other.outOfLineBits();
174 for (unsigned i = std::min(a->numWords(), b->numWords()); i--;)
175 a->bits()[i] &= ~b->bits()[i];
176}

178size_t BitVector::bitCountSlow() const
179{
180 ASSERT(!isInline());
181 const OutOfLineBits* bits = outOfLineBits();
182 size_t result = 0;
183 for (unsigned i = bits->numWords(); i--;)
184 result += bitCount(bits->bits()[i]);
185 return result;
186}

188bool BitVector::isEmptySlow() const
189{
190 ASSERT(!isInline());
191 const OutOfLineBits* bits = outOfLineBits();
192 for (unsigned i = bits->numWords(); i--;) {
193 if (bits->bits()[i])
194 return false;
195 }
196 return true;
197}

199bool BitVector::equalsSlowCase(const BitVector& other) const
200{
201 bool result = equalsSlowCaseFast(other);
202 ASSERT(result == equalsSlowCaseSimple(other));
203 return result;
204}
205
206bool BitVector::equalsSlowCaseFast(const BitVector& other) const
207{
208 if (isInline() != other.isInline())
209 return equalsSlowCaseSimple(other);
210
211 const OutOfLineBits* myBits = outOfLineBits();
212 const OutOfLineBits* otherBits = other.outOfLineBits();
213
214 size_t myNumWords = myBits->numWords();
215 size_t otherNumWords = otherBits->numWords();
216 size_t minNumWords;
217 size_t maxNumWords;
218
219 const OutOfLineBits* longerBits;
220 if (myNumWords < otherNumWords) {
221 minNumWords = myNumWords;
222 maxNumWords = otherNumWords;
223 longerBits = otherBits;
224 } else {
225 minNumWords = otherNumWords;
226 maxNumWords = myNumWords;
227 longerBits = myBits;
228 }
229
230 for (size_t i = minNumWords; i < maxNumWords; ++i) {
231 if (longerBits->bits()[i])
232 return false;
233 }
234
235 for (size_t i = minNumWords; i--;) {
236 if (myBits->bits()[i] != otherBits->bits()[i])
237 return false;
238 }
239
240 return true;
241}

243bool BitVector::equalsSlowCaseSimple(const BitVector& other) const
244{
245 // This is really cheesy, but probably good enough for now.
246 for (unsigned i = std::max(size(), other.size()); i--;) {
247 if (get(i) != other.get(i))
248 return false;
249 }
250 return true;
251}

253uintptr_t BitVector::hashSlowCase() const
254{
255 ASSERT(!isInline());
256 const OutOfLineBits* bits = outOfLineBits();
257 uintptr_t result = 0;
258 for (unsigned i = bits->numWords(); i--;)
259 result ^= bits->bits()[i];
260 return result;
261}

263void BitVector::dump(PrintStream& out) const
264{
265 for (size_t i = 0; i < size(); ++i) {
266 if (get(i))
267 out.printf("1");
268 else
269 out.printf("-");
270 }
