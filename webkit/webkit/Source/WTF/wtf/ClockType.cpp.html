26#include "config.h"
27#include <wtf/ClockType.h>
28
29#include <wtf/PrintStream.h>
30
31namespace WTF {
32
33void printInternal(PrintStream& out, ClockType type)
34{
35 switch (type) {
36 case ClockType::Wall:
37 out.print("Wall");
38 return;
39 case ClockType::Monotonic:
40 out.print("Monotonic");
41 return;
42 }
43 RELEASE_ASSERT_NOT_REACHED();
44}
45
46} // namespace WTF
