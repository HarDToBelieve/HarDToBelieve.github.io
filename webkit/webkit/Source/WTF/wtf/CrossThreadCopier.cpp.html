32#include "config.h"
33#include <wtf/CrossThreadCopier.h>
34
35namespace WTF {
36
37// Test CrossThreadCopier using COMPILE_ASSERT.
38
39// Verify that ThreadSafeRefCounted objects get handled correctly.
40class CopierThreadSafeRefCountedTest : public ThreadSafeRefCounted<CopierThreadSafeRefCountedTest> {
41};
42
43COMPILE_ASSERT((std::is_same<
44 RefPtr<CopierThreadSafeRefCountedTest>,
45 CrossThreadCopier<RefPtr<CopierThreadSafeRefCountedTest>>::Type
46 >::value),
47 RefPtrTest);
48COMPILE_ASSERT((std::is_same<
49 RefPtr<CopierThreadSafeRefCountedTest>,
50 CrossThreadCopier<CopierThreadSafeRefCountedTest*>::Type
51 >::value),
52 RawPointerTest);
53
54// Add specializations for RefCounted types which will let us verify that no other template matches.
55template<typename T> struct CrossThreadCopierBase<false, false, RefPtr<T>> {
56 typedef int Type;
57};
58
59template<typename T> struct CrossThreadCopierBase<false, false, T*> {
60 typedef int Type;
61};
62
63// Verify that RefCounted objects only match the above templates which expose Type as int.
64class CopierRefCountedTest : public RefCounted<CopierRefCountedTest> {
65};
66
67static_assert((std::is_same<int, CrossThreadCopier<RefPtr<CopierRefCountedTest>>::Type>::value), "CrossThreadCopier specialization improperly applied to RefPtr<> of a RefCounted (but not ThreadSafeRefCounted) type");
68static_assert((std::is_same<int, CrossThreadCopier<CopierRefCountedTest*>::Type>::value), "CrossThreadCopier specialization improperly applied to raw pointer of a RefCounted (but not ThreadSafeRefCounted) type");
69
70} // namespace WTF
