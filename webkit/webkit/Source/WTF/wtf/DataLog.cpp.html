36#define DATA_LOG_TO_FILE 0
37
38// Set to 1 to use the temp directory from confstr instead of hardcoded directory.
39// The last component of DATA_LOG_FILENAME will still be used.
40#define DATA_LOG_TO_DARWIN_TEMP_DIR 0
41
42// Uncomment to force logging to the given file regardless of what the environment variable says.
43// Note that we will append ".<pid>.txt" where <pid> is the PID.
44// This path won't work on Windows, make sure to change to something like C:\\Users\\<more path>\\log.txt.
45#define DATA_LOG_FILENAME "/tmp/WTFLog"
46
47namespace WTF {
48
49static constexpr size_t maxPathLength = 1024;
50
51static PrintStream* s_file;
52static uint64_t s_fileData[(sizeof(FilePrintStream) + 7) / 8];
53static uint64_t s_lockedFileData[(sizeof(LockedPrintStream) + 7) / 8];
54
55static void initializeLogFileOnce()
56{
57 const char* filename = nullptr;
58
59 if (s_file)
60 return;
61
62#if DATA_LOG_TO_FILE
63#if DATA_LOG_TO_DARWIN_TEMP_DIR
64 char filenameBuffer[maxPathLength + 1];
65#if defined(DATA_LOG_FILENAME)
66 const char* logBasename = strrchr(DATA_LOG_FILENAME, '/');
67 if (!logBasename)
68 logBasename = (char*)DATA_LOG_FILENAME;
69#else
70 const char* logBasename = "WTFLog";
71#endif
72
73 bool success = confstr(_CS_DARWIN_USER_TEMP_DIR, filenameBuffer, sizeof(filenameBuffer));
74 if (success) {
75 // FIXME: Assert that the path ends with a slash instead of adding a slash if it does not exist
76 // once <rdar://problem/23579077> is fixed in all iOS Simulator versions that we use.
77 size_t lastComponentLength = strlen(logBasename) + 20; // More than enough for ".<pid>.txt"
78 size_t dirnameLength = strlen(filenameBuffer);
79 bool shouldAddPathSeparator = filenameBuffer[dirnameLength - 1] != '/' && logBasename[0] != '/';
80 if (lastComponentLength + shouldAddPathSeparator <= sizeof(filenameBuffer) - dirnameLength - 1) {
81 if (shouldAddPathSeparator)
82 strncat(filenameBuffer, "/", 1);
83 strncat(filenameBuffer, logBasename, sizeof(filenameBuffer) - strlen(filenameBuffer) - 1);
84 filename = filenameBuffer;
85 }
86 }
87#elif defined(DATA_LOG_FILENAME)
88 filename = DATA_LOG_FILENAME;
89#else
90 filename = getenv("WTF_DATA_LOG_FILENAME");
91#endif
92 char actualFilename[maxPathLength + 1];
93
94 if (filename && !strstr(filename, "%pid")) {
95 snprintf(actualFilename, sizeof(actualFilename), "%s.%%pid.txt", filename);
96 filename = actualFilename;
97 }
98#endif // DATA_LOG_TO_FILE
99
100 setDataFile(filename);
101}
102
103static void initializeLogFile()
104{
105 static std::once_flag once;
106 std::call_once(
107 once,
108 [] {
109 initializeLogFileOnce();
110 });
111}
112
113void setDataFile(const char* path)
114{
115 FilePrintStream* file = nullptr;
116 char formattedPath[maxPathLength + 1];
117 const char* pathToOpen = path;
118
119 if (path) {
120 const char* pidFormat = strstr(path, "%pid");
121 if (pidFormat) {
122 size_t leadingPathLength = pidFormat - path;
123 size_t pathCharactersAvailable = std::min(maxPathLength, leadingPathLength);
124 strncpy(formattedPath, path, pathCharactersAvailable);
125 char* nextDest = formattedPath + pathCharactersAvailable;
126 pathCharactersAvailable = maxPathLength - pathCharactersAvailable;
127 if (pathCharactersAvailable) {
128 int pidTextLength = snprintf(nextDest, pathCharactersAvailable, "%d", getCurrentProcessID());
129
130 if (pidTextLength >= 0 && static_cast<size_t>(pidTextLength) < pathCharactersAvailable) {
131 pathCharactersAvailable -= static_cast<size_t>(pidTextLength);
132 nextDest += pidTextLength;
133 strncpy(nextDest, pidFormat + 4, pathCharactersAvailable);
134 }
135 }
136 formattedPath[maxPathLength] = '\0';
137 pathToOpen = formattedPath;
138 }
139
140 file = FilePrintStream::open(pathToOpen, "w").release();
141 if (file)
142 WTFLogAlways("*** DataLog output to \"%s\" ***\n", pathToOpen);
143 else
144 WTFLogAlways("Warning: Could not open DataLog file %s for writing.\n", pathToOpen);
145 }
146
147 if (!file) {
148 // Use placement new; this makes it easier to use dataLog() to debug
149 // fastMalloc.
150 file = new (s_fileData) FilePrintStream(stderr, FilePrintStream::Borrow);
151 }
152
153 setvbuf(file->file(), nullptr, _IONBF, 0); // Prefer unbuffered output, so that we get a full log upon crash or deadlock.
154
155 if (s_file)
156 s_file->flush();
157
158 s_file = new (s_lockedFileData) LockedPrintStream(std::unique_ptr<FilePrintStream>(file));
159}
160
161void setDataFile(std::unique_ptr<PrintStream>&& file)
162{
163 s_file = file.release();
164}
165
166PrintStream& dataFile()
167{
168 initializeLogFile();
169 return *s_file;
170}
171
172void dataLogFV(const char* format, va_list argList)
173{
174 dataFile().vprintf(format, argList);
175}
176
177void dataLogF(const char* format, ...)
178{
179 va_list argList;
180 va_start(argList, format);
181 dataLogFV(format, argList);
182 va_end(argList);
183}
184
185void dataLogFString(const char* str)
186{
187 dataFile().printf("%s", str);
188}
