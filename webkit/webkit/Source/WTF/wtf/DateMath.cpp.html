127static inline double msToMilliseconds(double ms)
128{
129 double result = fmod(ms, msPerDay);
130 if (result < 0)
131 result += msPerDay;
132 return result;

136// for (rdar://problem/5052975).
137static inline int maximumYearForDST()
138{
139 return 2037;
140}

142static inline int minimumYearForDST()
143{
144 // Because of the 2038 issue (see maximumYearForDST) if the current year is
145 // greater than the max year minus 27 (2010), we want to use the max year
146 // minus 27 instead, to ensure there is a range of 28 years that all years
147 // can map to.
148 return std::min(msToYear(jsCurrentTime()), maximumYearForDST() - 27) ;
149}

151/*
152 * Find an equivalent year for the one given, where equivalence is deterined by
153 * the two years having the same leapness and the first day of the year, falling
154 * on the same day of the week.
155 *
156 * This function returns a year between this current year and 2037, however this
157 * function will potentially return incorrect results if the current year is after
158 * 2010, (rdar://problem/5052975), if the year passed in is before 1900 or after
159 * 2100, (rdar://problem/5055038).
160 */
161int equivalentYearForDST(int year)
162{
163 // It is ok if the cached year is not the current year as long as the rules
164 // for DST did not change between the two years; if they did the app would need
165 // to be restarted.
166 static int minYear = minimumYearForDST();
167 int maxYear = maximumYearForDST();
168
169 int difference;
170 if (year > maxYear)
171 difference = minYear - year;
172 else if (year < minYear)
173 difference = maxYear - year;
174 else
175 return year;
176
177 int quotient = difference / 28;
178 int product = (quotient) * 28;
179
180 year += product;
181 return year;
182}

204static int32_t calculateUTCOffset()
205{
206#if OS(WINDOWS)
207 TIME_ZONE_INFORMATION timeZoneInformation;
208 DWORD rc = 0;
209
210 if (callGetTimeZoneInformationForYear_t timeZoneFunction = timeZoneInformationForYearFunction()) {
211 // If available, use the Windows API call that takes into account the varying DST from
212 // year to year.
213 SYSTEMTIME systemTime;
214 ::GetSystemTime(&systemTime);
215 rc = timeZoneFunction(systemTime.wYear, nullptr, &timeZoneInformation);
216 if (rc == TIME_ZONE_ID_INVALID)
217 return 0;
218 } else {
219 rc = ::GetTimeZoneInformation(&timeZoneInformation);
220 if (rc == TIME_ZONE_ID_INVALID)
221 return 0;
222 }
223
224 int32_t bias = timeZoneInformation.Bias;
225
226 if (rc == TIME_ZONE_ID_DAYLIGHT)
227 bias += timeZoneInformation.DaylightBias;
228 else if (rc == TIME_ZONE_ID_STANDARD || rc == TIME_ZONE_ID_UNKNOWN)
229 bias += timeZoneInformation.StandardBias;
230
231 return -bias * 60 * 1000;
232#else
233 time_t localTime = time(0);
234 tm localt;
235 getLocalTime(&localTime, &localt);
236
237 // Get the difference between this time zone and UTC on the 1st of January of this year.
238 localt.tm_sec = 0;
239 localt.tm_min = 0;
240 localt.tm_hour = 0;
241 localt.tm_mday = 1;
242 localt.tm_mon = 0;
243 // Not setting localt.tm_year!
244 localt.tm_wday = 0;
245 localt.tm_yday = 0;
246 localt.tm_isdst = 0;
247#if HAVE(TM_GMTOFF)
248 localt.tm_gmtoff = 0;
249#endif
250#if HAVE(TM_ZONE)
251 localt.tm_zone = 0;
252#endif
253
254#if HAVE(TIMEGM)
255 time_t utcOffset = timegm(&localt) - mktime(&localt);
256#else
257 // Using a canned date of 01/01/2019 on platforms with weaker date-handling foo.
258 localt.tm_year = 119;
259 time_t utcOffset = 1546300800 - mktime(&localt);
260#endif
261
262 return static_cast<int32_t>(utcOffset * 1000);
263#endif
264}

281/*
282 * Get the DST offset for the time passed in.
283 */
284static double calculateDSTOffset(time_t localTime, double utcOffset)
285{
286 // input is UTC so we have to shift back to local time to determine DST thus the + getUTCOffset()
287 double offsetTime = (localTime * msPerSecond) + utcOffset;
288
289 // Offset from UTC but doesn't include DST obviously
290 int offsetHour = msToHours(offsetTime);
291 int offsetMinute = msToMinutes(offsetTime);
292
293#if OS(WINDOWS)
294 FILETIME utcFileTime;
295 UnixTimeToFileTime(localTime, &utcFileTime);
296 SYSTEMTIME utcSystemTime, localSystemTime;
297 if (!::FileTimeToSystemTime(&utcFileTime, &utcSystemTime))
298 return 0;
299 if (!::SystemTimeToTzSpecificLocalTime(nullptr, &utcSystemTime, &localSystemTime))
300 return 0;
301
302 double diff = ((localSystemTime.wHour - offsetHour) * secondsPerHour) + ((localSystemTime.wMinute - offsetMinute) * 60);
303#else
304 tm localTM;
305 getLocalTime(&localTime, &localTM);
306
307 double diff = ((localTM.tm_hour - offsetHour) * secondsPerHour) + ((localTM.tm_min - offsetMinute) * 60);
308#endif
309
310 if (diff < 0)
311 diff += secondsPerDay;
312
313 return (diff * msPerSecond);
314}

318// Returns combined offset in millisecond (UTC + DST).
319LocalTimeOffset calculateLocalTimeOffset(double ms, TimeType inputTimeType)
320{
321#if HAVE(TM_GMTOFF)
322 double localToUTCTimeOffset = inputTimeType == LocalTime ? calculateUTCOffset() : 0;
323#else
324 double localToUTCTimeOffset = calculateUTCOffset();
325#endif
326 if (inputTimeType == LocalTime)
327 ms -= localToUTCTimeOffset;
328
329 // On Mac OS X, the call to localtime (see calculateDSTOffset) will return historically accurate
330 // DST information (e.g. New Zealand did not have DST from 1946 to 1974) however the JavaScript
331 // standard explicitly dictates that historical information should not be considered when
332 // determining DST. For this reason we shift away from years that localtime can handle but would
333 // return historically accurate information.
334 int year = msToYear(ms);
335 int equivalentYear = equivalentYearForDST(year);
336 if (year != equivalentYear) {
337 bool leapYear = isLeapYear(year);
338 int dayInYearLocal = dayInYear(ms, year);
339 int dayInMonth = dayInMonthFromDayInYear(dayInYearLocal, leapYear);
340 int month = monthFromDayInYear(dayInYearLocal, leapYear);
341 double day = dateToDaysFrom1970(equivalentYear, month, dayInMonth);
342 ms = (day * msPerDay) + msToMilliseconds(ms);
343 }
344
345 double localTimeSeconds = ms / msPerSecond;
346 if (localTimeSeconds > maxUnixTime)
347 localTimeSeconds = maxUnixTime;
348 else if (localTimeSeconds < 0) // Go ahead a day to make localtime work (does not work with 0).
349 localTimeSeconds += secondsPerDay;
350 // FIXME: time_t has a potential problem in 2038.
351 time_t localTime = static_cast<time_t>(localTimeSeconds);
352
353#if HAVE(TM_GMTOFF)
354 tm localTM;
355 getLocalTime(&localTime, &localTM);
356 return LocalTimeOffset(localTM.tm_isdst, localTM.tm_gmtoff * msPerSecond);
357#else
358 double dstOffset = calculateDSTOffset(localTime, localToUTCTimeOffset);
359 return LocalTimeOffset(dstOffset, localToUTCTimeOffset + dstOffset);
360#endif
361}

374static inline double ymdhmsToMilliseconds(int year, long mon, long day, long hour, long minute, long second, double milliseconds)
375{
376 int mday = firstDayOfMonth[isLeapYear(year)][mon - 1];
377 double ydays = daysFrom1970ToYear(year);
378
379 double dateMilliseconds = milliseconds + second * msPerSecond + minute * (secondsPerMinute * msPerSecond) + hour * (secondsPerHour * msPerSecond) + (mday + day - 1 + ydays) * (secondsPerDay * msPerSecond);
380
381 // Clamp to EcmaScript standard (ecma262/#sec-time-values-and-time-range) of
382 // +/- 100,000,000 days from 01 January, 1970.
383 if (dateMilliseconds < -8640000000000000.0 || dateMilliseconds > 8640000000000000.0)
384 return std::numeric_limits<double>::quiet_NaN();
385
386 return dateMilliseconds;
387}

410inline static void skipSpacesAndComments(const char*& s)
411{
412 int nesting = 0;
413 char ch;
414 while ((ch = *s)) {
415 if (!isASCIISpace(ch)) {
416 if (ch == '(')
417 nesting++;
418 else if (ch == ')' && nesting > 0)
419 nesting--;
420 else if (nesting == 0)
421 break;
422 }
423 s++;
424 }
425}

427// returns 0-11 (Jan-Dec); -1 on failure
428static int findMonth(const char* monthStr)
429{
430 ASSERT(monthStr);
431 char needle[4];
432 for (int i = 0; i < 3; ++i) {
433 if (!*monthStr)
434 return -1;
435 needle[i] = static_cast<char>(toASCIILower(*monthStr++));
436 }
437 needle[3] = '\0';
438 const char *haystack = "janfebmaraprmayjunjulaugsepoctnovdec";
439 const char *str = strstr(haystack, needle);
440 if (str) {
441 int position = static_cast<int>(str - haystack);
442 if (position % 3 == 0)
443 return position / 3;
444 }
445 return -1;
446}

448static bool parseInt(const char* string, char** stopPosition, int base, int* result)
449{
450 long longResult = strtol(string, stopPosition, base);
451 // Avoid the use of errno as it is not available on Windows CE
452 if (string == *stopPosition || longResult <= std::numeric_limits<int>::min() || longResult >= std::numeric_limits<int>::max())
453 return false;
454 *result = static_cast<int>(longResult);
455 return true;
456}

458static bool parseLong(const char* string, char** stopPosition, int base, long* result)
459{
460 *result = strtol(string, stopPosition, base);
461 // Avoid the use of errno as it is not available on Windows CE
462 if (string == *stopPosition || *result == std::numeric_limits<long>::min() || *result == std::numeric_limits<long>::max())
463 return false;
464 return true;
465}

467// Parses a date with the format YYYY[-MM[-DD]].
468// Year parsing is lenient, allows any number of digits, and +/-.
469// Returns 0 if a parse error occurs, else returns the end of the parsed portion of the string.
470static char* parseES5DatePortion(const char* currentPosition, int& year, long& month, long& day)
471{
472 char* postParsePosition;
473
474 // This is a bit more lenient on the year string than ES5 specifies:
475 // instead of restricting to 4 digits (or 6 digits with mandatory +/-),
476 // it accepts any integer value. Consider this an implementation fallback.
477 if (!parseInt(currentPosition, &postParsePosition, 10, &year))
478 return nullptr;
479
480 // Check for presence of -MM portion.
481 if (*postParsePosition != '-')
482 return postParsePosition;
483 currentPosition = postParsePosition + 1;
484
485 if (!isASCIIDigit(*currentPosition))
486 return nullptr;
487 if (!parseLong(currentPosition, &postParsePosition, 10, &month))
488 return nullptr;
489 if ((postParsePosition - currentPosition) != 2)
490 return nullptr;
491
492 // Check for presence of -DD portion.
493 if (*postParsePosition != '-')
494 return postParsePosition;
495 currentPosition = postParsePosition + 1;
496
497 if (!isASCIIDigit(*currentPosition))
498 return nullptr;
499 if (!parseLong(currentPosition, &postParsePosition, 10, &day))
500 return nullptr;
501 if ((postParsePosition - currentPosition) != 2)
502 return nullptr;
503 return postParsePosition;
504}

506// Parses a time with the format HH:mm[:ss[.sss]][Z|(+|-)(00:00|0000|00)].
507// Fractional seconds parsing is lenient, allows any number of digits.
508// Returns 0 if a parse error occurs, else returns the end of the parsed portion of the string.
509static char* parseES5TimePortion(char* currentPosition, long& hours, long& minutes, long& seconds, double& milliseconds, bool& isLocalTime, long& timeZoneSeconds)
510{
511 isLocalTime = false;
512
513 char* postParsePosition;
514 if (!isASCIIDigit(*currentPosition))
515 return nullptr;
516 if (!parseLong(currentPosition, &postParsePosition, 10, &hours))
517 return nullptr;
518 if (*postParsePosition != ':' || (postParsePosition - currentPosition) != 2)
519 return nullptr;
520 currentPosition = postParsePosition + 1;
521
522 if (!isASCIIDigit(*currentPosition))
523 return nullptr;
524 if (!parseLong(currentPosition, &postParsePosition, 10, &minutes))
525 return nullptr;
526 if ((postParsePosition - currentPosition) != 2)
527 return nullptr;
528 currentPosition = postParsePosition;
529
530 // Seconds are optional.
531 if (*currentPosition == ':') {
532 ++currentPosition;
533
534 if (!isASCIIDigit(*currentPosition))
535 return nullptr;
536 if (!parseLong(currentPosition, &postParsePosition, 10, &seconds))
537 return nullptr;
538 if ((postParsePosition - currentPosition) != 2)
539 return nullptr;
540 if (*postParsePosition == '.') {
541 currentPosition = postParsePosition + 1;
542
543 // In ECMA-262-5 it's a bit unclear if '.' can be present without milliseconds, but
544 // a reasonable interpretation guided by the given examples and RFC 3339 says "no".
545 // We check the next character to avoid reading +/- timezone hours after an invalid decimal.
546 if (!isASCIIDigit(*currentPosition))
547 return nullptr;
548
549 // We are more lenient than ES5 by accepting more or less than 3 fraction digits.
550 long fracSeconds;
551 if (!parseLong(currentPosition, &postParsePosition, 10, &fracSeconds))
552 return nullptr;
553
554 long numFracDigits = postParsePosition - currentPosition;
555 milliseconds = fracSeconds * pow(10.0, static_cast<double>(-numFracDigits + 3));
556 }
557 currentPosition = postParsePosition;
558 }
559
560 if (*currentPosition == 'Z')
561 return currentPosition + 1;
562
563 // Parse (+|-)(00:00|0000|00).
564 bool tzNegative;
565 if (*currentPosition == '-')
566 tzNegative = true;
567 else if (*currentPosition == '+')
568 tzNegative = false;
569 else {
570 isLocalTime = true;
571 return currentPosition;
572 }
573 ++currentPosition;
574
575 long tzHours = 0;
576 long tzHoursAbs = 0;
577 long tzMinutes = 0;
578
579 if (!isASCIIDigit(*currentPosition))
580 return nullptr;
581 if (!parseLong(currentPosition, &postParsePosition, 10, &tzHours))
582 return nullptr;
583 if (*postParsePosition != ':') {
584 if ((postParsePosition - currentPosition) == 2) {
585 // "00" case.
586 tzHoursAbs = labs(tzHours);
587 } else if ((postParsePosition - currentPosition) == 4) {
588 // "0000" case.
589 tzHoursAbs = labs(tzHours);
590 tzMinutes = tzHoursAbs % 100;
591 tzHoursAbs = tzHoursAbs / 100;
592 } else
593 return nullptr;
594 } else {
595 // "00:00" case.
596 if ((postParsePosition - currentPosition) != 2)
597 return nullptr;
598 tzHoursAbs = labs(tzHours);
599 currentPosition = postParsePosition + 1; // Skip ":".
600
601 if (!isASCIIDigit(*currentPosition))
602 return nullptr;
603 if (!parseLong(currentPosition, &postParsePosition, 10, &tzMinutes))
604 return nullptr;
605 if ((postParsePosition - currentPosition) != 2)
606 return nullptr;
607 }
608 currentPosition = postParsePosition;
609
610 if (tzHoursAbs > 24)
611 return nullptr;
612 if (tzMinutes < 0 || tzMinutes > 59)
613 return nullptr;
614
615 timeZoneSeconds = 60 * (tzMinutes + (60 * tzHoursAbs));
616 if (tzNegative)
617 timeZoneSeconds = -timeZoneSeconds;
618
619 return currentPosition;
620}

622double parseES5DateFromNullTerminatedCharacters(const char* dateString, bool& isLocalTime)
623{
624 isLocalTime = false;
625
626 // This parses a date of the form defined in ecma262/#sec-date-time-string-format
627 // (similar to RFC 3339 / ISO 8601: YYYY-MM-DDTHH:mm:ss[.sss]Z).
628 // In most cases it is intentionally strict (e.g. correct field widths, no stray whitespace).
629
630 static const long daysPerMonth[12] = { 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
631
632 // The year must be present, but the other fields may be omitted - see ES5.1 15.9.1.15.
633 int year = 0;
634 long month = 1;
635 long day = 1;
636 long hours = 0;
637 long minutes = 0;
638 long seconds = 0;
639 double milliseconds = 0;
640 long timeZoneSeconds = 0;
641
642 // Parse the date YYYY[-MM[-DD]]
643 char* currentPosition = parseES5DatePortion(dateString, year, month, day);
644 if (!currentPosition)
645 return std::numeric_limits<double>::quiet_NaN();
646 // Look for a time portion.
647 // Note: As of ES2016, when a UTC offset is missing, date-time forms are local time while date-only forms are UTC.
648 if (*currentPosition == 'T') {
649 // Parse the time HH:mm[:ss[.sss]][Z|(+|-)(00:00|0000|00)]
650 currentPosition = parseES5TimePortion(currentPosition + 1, hours, minutes, seconds, milliseconds, isLocalTime, timeZoneSeconds);
651 if (!currentPosition)
652 return std::numeric_limits<double>::quiet_NaN();
653 }
654 // Check that we have parsed all characters in the string.
655 if (*currentPosition)
656 return std::numeric_limits<double>::quiet_NaN();
657
658 // A few of these checks could be done inline above, but since many of them are interrelated
659 // we would be sacrificing readability to "optimize" the (presumably less common) failure path.
660 if (month < 1 || month > 12)
661 return std::numeric_limits<double>::quiet_NaN();
662 if (day < 1 || day > daysPerMonth[month - 1])
663 return std::numeric_limits<double>::quiet_NaN();
664 if (month == 2 && day > 28 && !isLeapYear(year))
665 return std::numeric_limits<double>::quiet_NaN();
666 if (hours < 0 || hours > 24)
667 return std::numeric_limits<double>::quiet_NaN();
668 if (hours == 24 && (minutes || seconds))
669 return std::numeric_limits<double>::quiet_NaN();
670 if (minutes < 0 || minutes > 59)
671 return std::numeric_limits<double>::quiet_NaN();
672 if (seconds < 0 || seconds >= 61)
673 return std::numeric_limits<double>::quiet_NaN();
674 if (seconds == 60) {
675 // Discard leap seconds by clamping to the end of a minute.
676 milliseconds = 0;
677 }
678
679 return ymdhmsToMilliseconds(year, month, day, hours, minutes, seconds, milliseconds) - (timeZoneSeconds * msPerSecond);
680}

682// Odd case where 'exec' is allowed to be 0, to accomodate a caller in WebCore.
683double parseDateFromNullTerminatedCharacters(const char* dateString, bool& isLocalTime)
684{
685 isLocalTime = true;
686 int offset = 0;
687
688 // This parses a date in the form:
689 // Tuesday, 09-Nov-99 23:12:40 GMT
690 // or
691 // Sat, 01-Jan-2000 08:00:00 GMT
692 // or
693 // Sat, 01 Jan 2000 08:00:00 GMT
694 // or
695 // 01 Jan 99 22:00 +0100 (exceptions in rfc822/rfc2822)
696 // ### non RFC formats, added for Javascript:
697 // [Wednesday] January 09 1999 23:12:40 GMT
698 // [Wednesday] January 09 23:12:40 GMT 1999
699 //
700 // We ignore the weekday.
701
702 // Skip leading space
703 skipSpacesAndComments(dateString);
704
705 long month = -1;
706 const char *wordStart = dateString;
707 // Check contents of first words if not number
708 while (*dateString && !isASCIIDigit(*dateString)) {
709 if (isASCIISpace(*dateString) || *dateString == '(') {
710 if (dateString - wordStart >= 3)
711 month = findMonth(wordStart);
712 skipSpacesAndComments(dateString);
713 wordStart = dateString;
714 } else
715 dateString++;
716 }
717
718 // Missing delimiter between month and day (like "January29")?
719 if (month == -1 && wordStart != dateString)
720 month = findMonth(wordStart);
721
722 skipSpacesAndComments(dateString);
723
724 if (!*dateString)
725 return std::numeric_limits<double>::quiet_NaN();
726
727 // ' 09-Nov-99 23:12:40 GMT'
728 char* newPosStr;
729 long day;
730 if (!parseLong(dateString, &newPosStr, 10, &day))
731 return std::numeric_limits<double>::quiet_NaN();
732 dateString = newPosStr;
733
734 if (day < 0)
735 return std::numeric_limits<double>::quiet_NaN();
736
737 std::optional<int> year;
738 if (day > 31) {
739 // ### where is the boundary and what happens below?
740 if (*dateString != '/')
741 return std::numeric_limits<double>::quiet_NaN();
742 // looks like a YYYY/MM/DD date
743 if (!*++dateString)
744 return std::numeric_limits<double>::quiet_NaN();
745 if (day <= std::numeric_limits<int>::min() || day >= std::numeric_limits<int>::max())
746 return std::numeric_limits<double>::quiet_NaN();
747 year = static_cast<int>(day);
748 if (!parseLong(dateString, &newPosStr, 10, &month))
749 return std::numeric_limits<double>::quiet_NaN();
750 month -= 1;
751 dateString = newPosStr;
752 if (*dateString++ != '/' || !*dateString)
753 return std::numeric_limits<double>::quiet_NaN();
754 if (!parseLong(dateString, &newPosStr, 10, &day))
755 return std::numeric_limits<double>::quiet_NaN();
756 dateString = newPosStr;
757 } else if (*dateString == '/' && month == -1) {
758 dateString++;
759 // This looks like a MM/DD/YYYY date, not an RFC date.
760 month = day - 1; // 0-based
761 if (!parseLong(dateString, &newPosStr, 10, &day))
762 return std::numeric_limits<double>::quiet_NaN();
763 if (day < 1 || day > 31)
764 return std::numeric_limits<double>::quiet_NaN();
765 dateString = newPosStr;
766 if (*dateString == '/')
767 dateString++;
768 if (!*dateString)
769 return std::numeric_limits<double>::quiet_NaN();
770 } else {
771 if (*dateString == '-')
772 dateString++;
773
774 skipSpacesAndComments(dateString);
775
776 if (*dateString == ',')
777 dateString++;
778
779 if (month == -1) { // not found yet
780 month = findMonth(dateString);
781 if (month == -1)
782 return std::numeric_limits<double>::quiet_NaN();
783
784 while (*dateString && *dateString != '-' && *dateString != ',' && !isASCIISpace(*dateString))
785 dateString++;
786
787 if (!*dateString)
788 return std::numeric_limits<double>::quiet_NaN();
789
790 // '-99 23:12:40 GMT'
791 if (*dateString != '-' && *dateString != '/' && *dateString != ',' && !isASCIISpace(*dateString))
792 return std::numeric_limits<double>::quiet_NaN();
793 dateString++;
794 }
795 }
796
797 if (month < 0 || month > 11)
798 return std::numeric_limits<double>::quiet_NaN();
799
800 // '99 23:12:40 GMT'
801 if (*dateString && !year) {
802 int result = 0;
803 if (!parseInt(dateString, &newPosStr, 10, &result))
804 return std::numeric_limits<double>::quiet_NaN();
805 year = result;
806 }
807
808 // Don't fail if the time is missing.
809 long hour = 0;
810 long minute = 0;
811 long second = 0;
812 if (!*newPosStr)
813 dateString = newPosStr;
814 else {
815 // ' 23:12:40 GMT'
816 if (!(isASCIISpace(*newPosStr) || *newPosStr == ',')) {
817 if (*newPosStr != ':')
818 return std::numeric_limits<double>::quiet_NaN();
819 // There was no year; the number was the hour.
820 year = std::nullopt;
821 } else {
822 // in the normal case (we parsed the year), advance to the next number
823 dateString = ++newPosStr;
824 skipSpacesAndComments(dateString);
825 }
826
827 parseLong(dateString, &newPosStr, 10, &hour);
828 // Do not check for errno here since we want to continue
829 // even if errno was set becasue we are still looking
830 // for the timezone!
831
832 // Read a number? If not, this might be a timezone name.
833 if (newPosStr != dateString) {
834 dateString = newPosStr;
835
836 if (hour < 0 || hour > 23)
837 return std::numeric_limits<double>::quiet_NaN();
838
839 if (!*dateString)
840 return std::numeric_limits<double>::quiet_NaN();
841
842 // ':12:40 GMT'
843 if (*dateString++ != ':')
844 return std::numeric_limits<double>::quiet_NaN();
845
846 if (!parseLong(dateString, &newPosStr, 10, &minute))
847 return std::numeric_limits<double>::quiet_NaN();
848 dateString = newPosStr;
849
850 if (minute < 0 || minute > 59)
851 return std::numeric_limits<double>::quiet_NaN();
852
853 // ':40 GMT'
854 if (*dateString && *dateString != ':' && !isASCIISpace(*dateString))
855 return std::numeric_limits<double>::quiet_NaN();
856
857 // seconds are optional in rfc822 + rfc2822
858 if (*dateString ==':') {
859 dateString++;
860
861 if (!parseLong(dateString, &newPosStr, 10, &second))
862 return std::numeric_limits<double>::quiet_NaN();
863 dateString = newPosStr;
864
865 if (second < 0 || second > 59)
866 return std::numeric_limits<double>::quiet_NaN();
867 }
868
869 skipSpacesAndComments(dateString);
870
871 if (startsWithLettersIgnoringASCIICase(dateString, "am")) {
872 if (hour > 12)
873 return std::numeric_limits<double>::quiet_NaN();
874 if (hour == 12)
875 hour = 0;
876 dateString += 2;
877 skipSpacesAndComments(dateString);
878 } else if (startsWithLettersIgnoringASCIICase(dateString, "pm")) {
879 if (hour > 12)
880 return std::numeric_limits<double>::quiet_NaN();
881 if (hour != 12)
882 hour += 12;
883 dateString += 2;
884 skipSpacesAndComments(dateString);
885 }
886 }
887 }
888
889 // The year may be after the time but before the time zone.
890 if (isASCIIDigit(*dateString) && !year) {
891 int result = 0;
892 if (!parseInt(dateString, &newPosStr, 10, &result))
893 return std::numeric_limits<double>::quiet_NaN();
894 year = result;
895 dateString = newPosStr;
896 skipSpacesAndComments(dateString);
897 }
898
899 // Don't fail if the time zone is missing.
900 // Some websites omit the time zone (4275206).
901 if (*dateString) {
902 if (startsWithLettersIgnoringASCIICase(dateString, "gmt") || startsWithLettersIgnoringASCIICase(dateString, "utc")) {
903 dateString += 3;
904 isLocalTime = false;
905 }
906
907 if (*dateString == '+' || *dateString == '-') {
908 int o;
909 if (!parseInt(dateString, &newPosStr, 10, &o))
910 return std::numeric_limits<double>::quiet_NaN();
911 dateString = newPosStr;
912
913 if (o < -9959 || o > 9959)
914 return std::numeric_limits<double>::quiet_NaN();
915
916 int sgn = (o < 0) ? -1 : 1;
917 o = abs(o);
918 if (*dateString != ':') {
919 if (o >= 24)
920 offset = ((o / 100) * 60 + (o % 100)) * sgn;
921 else
922 offset = o * 60 * sgn;
923 } else { // GMT+05:00
924 ++dateString; // skip the ':'
925 int o2;
926 if (!parseInt(dateString, &newPosStr, 10, &o2))
927 return std::numeric_limits<double>::quiet_NaN();
928 dateString = newPosStr;
929 offset = (o * 60 + o2) * sgn;
930 }
931 isLocalTime = false;
932 } else {
933 for (auto& knownZone : knownZones) {
934 // Since the passed-in length is used for both strings, the following checks that
935 // dateString has the time zone name as a prefix, not that it is equal.
936 auto length = strlen(knownZone.tzName);
937 if (equalLettersIgnoringASCIICase(dateString, knownZone.tzName, length)) {
938 offset = knownZone.tzOffset;
939 dateString += length;
940 isLocalTime = false;
941 break;
942 }
943 }
944 }
945 }
946
947 skipSpacesAndComments(dateString);
948
949 if (*dateString && !year) {
950 int result = 0;
951 if (!parseInt(dateString, &newPosStr, 10, &result))
952 return std::numeric_limits<double>::quiet_NaN();
953 year = result;
954 dateString = newPosStr;
955 skipSpacesAndComments(dateString);
956 }
957
958 // Trailing garbage
959 if (*dateString)
960 return std::numeric_limits<double>::quiet_NaN();
961
962 // Y2K: Handle 2 digit years.
963 if (year) {
964 int yearValue = year.value();
965 if (yearValue >= 0 && yearValue < 100) {
966 if (yearValue < 50)
967 yearValue += 2000;
968 else
969 yearValue += 1900;
970 }
971 year = yearValue;
972 } else {
973 // We select 2000 as default value. This is because of the following reasons.
974 // 1. Year 2000 was used for the initial value of the variable `year`. While it won't be posed to users in WebKit,
975 // V8 used this 2000 as its default value. (As of April 2017, V8 is using the year 2001 and Spider Monkey is
976 // not doing this kind of fallback.)
977 // 2. It is a leap year. When using `new Date("Feb 29")`, we assume that people want to save month and day.
978 // Leap year can save user inputs if they is valid. If we use the current year instead, the current year
979 // may not be a leap year. In that case, `new Date("Feb 29").getMonth()` becomes 2 (March).
980 year = 2000;
981 }
982 ASSERT(year);
983
984 return ymdhmsToMilliseconds(year.value(), month + 1, day, hour, minute, second, 0) - offset * (secondsPerMinute * msPerSecond);
985}

987double parseDateFromNullTerminatedCharacters(const char* dateString)
988{
989 bool isLocalTime;
990 double value = parseDateFromNullTerminatedCharacters(dateString, isLocalTime);
991
992 if (isLocalTime)
993 value -= calculateLocalTimeOffset(value, LocalTime).offset;
994
995 return value;
996}

998// See http://tools.ietf.org/html/rfc2822#section-3.3 for more information.
999String makeRFC2822DateString(unsigned dayOfWeek, unsigned day, unsigned month, unsigned year, unsigned hours, unsigned minutes, unsigned seconds, int utcOffset)
1000{
1001 StringBuilder stringBuilder;
1002 stringBuilder.append(weekdayName[dayOfWeek], ", ", day, ' ', monthName[month], ' ', year, ' ');
1003
1004 appendTwoDigitNumber(stringBuilder, hours);
1005 stringBuilder.append(':');
1006 appendTwoDigitNumber(stringBuilder, minutes);
1007 stringBuilder.append(':');
1008 appendTwoDigitNumber(stringBuilder, seconds);
1009 stringBuilder.append(' ');
1010
1011 stringBuilder.append(utcOffset > 0 ? '+' : '-');
1012 int absoluteUTCOffset = abs(utcOffset);
1013 appendTwoDigitNumber(stringBuilder, absoluteUTCOffset / 60);
1014 appendTwoDigitNumber(stringBuilder, absoluteUTCOffset % 60);
1015
1016 return stringBuilder.toString();
1017}
