95#define ASSERT_IS_WITHIN_LIMIT(size) do { \
96 size_t size__ = (size); \
97 ASSERT_WITH_MESSAGE((size__) <= maxSingleAllocationSize, "Requested size (%zu) exceeds max single allocation size set for testing (%zu)", (size__), maxSingleAllocationSize); \
98 } while (false)
99
100#define FAIL_IF_EXCEEDS_LIMIT(size) do { \
101 if (UNLIKELY((size) > maxSingleAllocationSize)) \
102 return nullptr; \
103 } while (false)

112void* fastZeroedMalloc(size_t n)
113{
114 void* result = fastMalloc(n);
115 memset(result, 0, n);
116 return result;
117}
118
119char* fastStrDup(const char* src)
120{
121 size_t len = strlen(src) + 1;
122 char* dup = static_cast<char*>(fastMalloc(len));
123 memcpy(dup, src, len);
124 return dup;
125}
126
127void* fastMemDup(const void* mem, size_t bytes)
128{
129 if (!mem || !bytes)
130 return nullptr;
131
132 void* result = fastMalloc(bytes);
133 memcpy(result, mem, bytes);
134 return result;
135}
136
137TryMallocReturnValue tryFastZeroedMalloc(size_t n)
138{
139 void* result;
140 if (!tryFastMalloc(n).getValue(result))
141 return nullptr;
142 memset(result, 0, n);
143 return result;
144}

328static ThreadSpecificKey avoidRecordingCountKey { InvalidThreadSpecificKey };
329class AvoidRecordingScope {
330public:
331 AvoidRecordingScope();
332 ~AvoidRecordingScope();
333
334 static uintptr_t avoidRecordingCount()
335 {
336 return bitwise_cast<uintptr_t>(threadSpecificGet(avoidRecordingCountKey));
337 }
338};
339
340AvoidRecordingScope::AvoidRecordingScope()
341{
342 static std::once_flag onceKey;
343 std::call_once(onceKey, [] {
344 // The value stored in TLS is initially 0.
345 threadSpecificKeyCreate(&avoidRecordingCountKey, [](void*) { });
346 });
347 threadSpecificSet(avoidRecordingCountKey, bitwise_cast<void*>(avoidRecordingCount() + 1));
348}
349
350AvoidRecordingScope::~AvoidRecordingScope()
351{
352 threadSpecificSet(avoidRecordingCountKey, bitwise_cast<void*>(avoidRecordingCount() - 1));
353}

355class MallocCallTracker {
356public:
357 MallocCallTracker();
358
359 void recordMalloc(void*, size_t);
360 void recordRealloc(void* oldAddress, void* newAddress, size_t);
361 void recordFree(void*);
362
363 void dumpStats();
364
365 static MallocCallTracker& singleton();
366
367private:
368 struct MallocSiteData {
369 StackShot stack;
370 size_t size;
371
372 MallocSiteData(size_t stackSize, size_t allocationSize)
373 : stack(stackSize)
374 , size(allocationSize)
375 {
376 }
377 };
378
379 Lock m_lock;
380 HashMap<void*, std::unique_ptr<MallocSiteData>> m_addressMallocSiteData WTF_GUARDED_BY_LOCK(m_lock);
381};
382
383MallocCallTracker& MallocCallTracker::singleton()
384{
385 AvoidRecordingScope avoidRecording;
386 static LazyNeverDestroyed<MallocCallTracker> tracker;
387 static std::once_flag onceKey;
388 std::call_once(onceKey, [&] {
389 tracker.construct();
390 });
391 return tracker;
392}

403void MallocCallTracker::recordMalloc(void* address, size_t allocationSize)
404{
405 AvoidRecordingScope avoidRecording;
406
407 // Intentionally using std::make_unique not to use FastMalloc for data structure tracking FastMalloc.
408 const size_t stackSize = 10;
409 auto siteData = std::make_unique<MallocSiteData>(stackSize, allocationSize);
410
411 Locker locker { m_lock };
412 auto addResult = m_addressMallocSiteData.add(address, WTFMove(siteData));
413 UNUSED_PARAM(addResult);
414}
415
416void MallocCallTracker::recordRealloc(void* oldAddress, void* newAddress, size_t newSize)
417{
418 AvoidRecordingScope avoidRecording;
419
420 Locker locker { m_lock };
421
422 auto it = m_addressMallocSiteData.find(oldAddress);
423 if (it == m_addressMallocSiteData.end()) {
424 ASSERT_NOT_REACHED();
425 return;
426 }
427
428 it->value->size = newSize;
429 if (oldAddress != newAddress) {
430 auto value = WTFMove(it->value);
431 m_addressMallocSiteData.remove(it);
432 auto addResult = m_addressMallocSiteData.add(newAddress, WTFMove(value));
433 ASSERT_UNUSED(addResult, addResult.isNewEntry);
434 }
435}
436
437void MallocCallTracker::recordFree(void* address)
438{
439 AvoidRecordingScope avoidRecording;
440
441 Locker locker { m_lock };
442 bool removed = m_addressMallocSiteData.remove(address);
443 UNUSED_PARAM(removed);
444}

446void MallocCallTracker::dumpStats()
447{
448 AvoidRecordingScope avoidRecording;
449
450 {
451 Locker locker { m_lock };
452
453 // Build a hash of stack to address vector
454 struct MallocSiteTotals {
455 Vector<MallocSiteData*> siteData;
456 size_t count { 0 };
457 size_t totalSize { 0 };
458 };
459
460 size_t totalUntrackedSize = 0;
461 size_t totalUntrackedCount = 0;
462
463 HashMap<unsigned, std::unique_ptr<MallocSiteTotals>> callSiteToMallocData;
464 for (const auto& it : m_addressMallocSiteData) {
465 auto result = callSiteToMallocData.ensure(it.value->stack.hash(), [] () {
466 // Intentionally using std::make_unique not to use FastMalloc for data structure tracking FastMalloc.
467 return std::make_unique<MallocSiteTotals>();
468 });
469 auto& siteTotal = result.iterator->value;
470 siteTotal->siteData.append(it.value.get());
471 ++siteTotal->count;
472 siteTotal->totalSize += it.value->size;
473 totalUntrackedSize += it.value->size;
474 ++totalUntrackedCount;
475 }
476
477 Vector<unsigned> stackHashes;
478 auto stackKeys = callSiteToMallocData.keys();
479 for (auto key : stackKeys)
480 stackHashes.append(key);
481
482 // Sort by reverse total size.
483 std::sort(stackHashes.begin(), stackHashes.end(), [&] (unsigned a, unsigned b) {
484 const auto& aSiteTotals = callSiteToMallocData.get(a);
485 const auto& bSiteTotals = callSiteToMallocData.get(b);
486
487 return aSiteTotals->totalSize > bSiteTotals->totalSize;
488 });
489
490 WTFLogAlways("Total untracked bytes: %lu (%lu allocations)\n", totalUntrackedSize, totalUntrackedCount);
491
492 const size_t numStacksToDump = 100;
493 for (size_t i = 0; i < std::min(numStacksToDump, stackHashes.size()); ++i) {
494 const auto& mallocDataForStack = callSiteToMallocData.get(stackHashes[i]);
495
496 WTFLogAlways("Total allocation size: %lu (%lu allocations)\n", mallocDataForStack->totalSize, mallocDataForStack->count);
497 // FIXME: Add a way to remove some entries in StackShot in a programable way.
498 // https://bugs.webkit.org/show_bug.cgi?id=205701
499 const size_t framesToSkip = 6;
500 WTFPrintBacktrace(mallocDataForStack->siteData[0]->stack.array() + framesToSkip, mallocDataForStack->siteData[0]->stack.size() - framesToSkip);
501 WTFLogAlways("\n");
502 }
503 }

521void* fastMalloc(size_t size)
522{
523 ASSERT_IS_WITHIN_LIMIT(size);
524 ASSERT(!forbidMallocUseScopeCount || disableMallocRestrictionScopeCount);
525 void* result = bmalloc::api::malloc(size);
526#if ENABLE(MALLOC_HEAP_BREAKDOWN) && TRACK_MALLOC_CALLSTACK
527 if (!AvoidRecordingScope::avoidRecordingCount())
528 MallocCallTracker::singleton().recordMalloc(result, size);
529#endif
530 return result;
531}

533void* fastCalloc(size_t numElements, size_t elementSize)
534{
535 ASSERT_IS_WITHIN_LIMIT(numElements * elementSize);
536 Checked<size_t> checkedSize = elementSize;
537 checkedSize *= numElements;
538 void* result = fastZeroedMalloc(checkedSize);
539 if (!result)
540 CRASH();
541 return result;
542}

641FastMallocStatistics fastMallocStatistics()
642{
643
644 // FIXME: Can bmalloc itself report the stats instead of relying on the OS?
645 FastMallocStatistics statistics;
646 statistics.freeListBytes = 0;
647 statistics.reservedVMBytes = 0;
648
649#if OS(WINDOWS)
650 PROCESS_MEMORY_COUNTERS resourceUsage;
651 GetProcessMemoryInfo(GetCurrentProcess(), &resourceUsage, sizeof(resourceUsage));
652 statistics.committedVMBytes = resourceUsage.PeakWorkingSetSize;
653#elif HAVE(RESOURCE_H)
654 struct rusage resourceUsage;
655 getrusage(RUSAGE_SELF, &resourceUsage);
656
657#if OS(DARWIN)
658 statistics.committedVMBytes = resourceUsage.ru_maxrss;
659#else
660 statistics.committedVMBytes = resourceUsage.ru_maxrss * 1024;
661#endif // OS(DARWIN)
662
663#endif // OS(WINDOWS)
664 return statistics;
665}
