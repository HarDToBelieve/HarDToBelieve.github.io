26#include "config.h"
27#include <wtf/FilePrintStream.h>
28
29namespace WTF {
30
31FilePrintStream::FilePrintStream(FILE* file, AdoptionMode adoptionMode)
32 : m_file(file)
33 , m_adoptionMode(adoptionMode)
34{
35}
36
37FilePrintStream::~FilePrintStream()
38{
39 if (m_adoptionMode == Borrow)
40 return;
41 fclose(m_file);
42}
43
44std::unique_ptr<FilePrintStream> FilePrintStream::open(const char* filename, const char* mode)
45{
46 FILE* file = fopen(filename, mode);
47 if (!file)
48 return nullptr;
49
50 return makeUnique<FilePrintStream>(file);
51}
52
53void FilePrintStream::vprintf(const char* format, va_list argList)
54{
55 vfprintf(m_file, format, argList);
56}
57
58void FilePrintStream::flush()
59{
60 fflush(m_file);
61}
