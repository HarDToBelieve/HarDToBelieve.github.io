69// The following lower-ASCII characters need escaping to be used in a filename
70// across all systems, including Windows:
71// - Unprintable ASCII (00-1F)
72// - Space (20)
73// - Double quote (22)
74// - Percent (25) (escaped because it is our escape character)
75// - Asterisk (2A)
76// - Slash (2F)
77// - Colon (3A)
78// - Less-than (3C)
79// - Greater-than (3E)
80// - Question Mark (3F)
81// - Backslash (5C)
82// - Pipe (7C)
83// - Delete (7F)
84
85static const bool needsEscaping[128] = {
86 true, true, true, true, true, true, true, true, /* 00-07 */
87 true, true, true, true, true, true, true, true, /* 08-0F */
88
89 true, true, true, true, true, true, true, true, /* 10-17 */
90 true, true, true, true, true, true, true, true, /* 18-1F */
91
92 true, false, true, false, false, true, false, false, /* 20-27 */
93 false, false, true, false, false, false, false, true, /* 28-2F */
94
95 false, false, false, false, false, false, false, false, /* 30-37 */
96 false, false, true, false, true, false, true, true, /* 38-3F */
97
98 false, false, false, false, false, false, false, false, /* 40-47 */
99 false, false, false, false, false, false, false, false, /* 48-4F */
100
101 false, false, false, false, false, false, false, false, /* 50-57 */
102 false, false, false, false, true, false, false, false, /* 58-5F */
103
104 false, false, false, false, false, false, false, false, /* 60-67 */
105 false, false, false, false, false, false, false, false, /* 68-6F */
106
107 false, false, false, false, false, false, false, false, /* 70-77 */
108 false, false, false, false, true, false, false, true, /* 78-7F */
109};

111static inline bool shouldEscapeUChar(UChar character, UChar previousCharacter, UChar nextCharacter)
112{
113 if (character <= 127)
114 return needsEscaping[character];
115
116 if (U16_IS_LEAD(character) && !U16_IS_TRAIL(nextCharacter))
117 return true;
118
119 if (U16_IS_TRAIL(character) && !U16_IS_LEAD(previousCharacter))
120 return true;
121
122 return false;
123}

152String encodeForFileName(const String& inputString)
153{
154 unsigned length = inputString.length();
155 if (!length)
156 return inputString;
157
158 StringBuilder result;
159 result.reserveCapacity(length);
160
161 UChar previousCharacter = 0;
162 UChar nextCharacter = inputString[0];
163 for (unsigned i = 0; i < length; ++i) {
164 auto character = std::exchange(nextCharacter, i + 1 < length ? inputString[i + 1] : 0);
165 if (shouldEscapeUChar(character, previousCharacter, nextCharacter)) {
166 if (character <= 0xFF)
167 result.append('%', hex(character, 2));
168 else
169 result.append("%+", hex(static_cast<uint8_t>(character >> 8), 2), hex(static_cast<uint8_t>(character), 2));
170 } else
171 result.append(character);
172 previousCharacter = character;
173 }
174
175 return result.toString();
176}

178String decodeFromFilename(const String& inputString)
179{
180 unsigned length = inputString.length();
181 if (!length)
182 return inputString;
183
184 StringBuilder result;
185 result.reserveCapacity(length);
186
187 for (unsigned i = 0; i < length; ++i) {
188 if (inputString[i] != '%') {
189 result.append(inputString[i]);
190 continue;
191 }
192
193 // If the input string is a valid encoded filename, it must be at least 2 characters longer
194 // than the current index to account for this percent encoded value.
195 if (i + 2 >= length)
196 return { };
197
198 if (inputString[i+1] != '+') {
199 if (!isASCIIHexDigit(inputString[i + 1]))
200 return { };
201 if (!isASCIIHexDigit(inputString[i + 2]))
202 return { };
203 result.append(toASCIIHexValue(inputString[i + 1], inputString[i + 2]));
204 i += 2;
205 continue;
206 }
207
208 // If the input string is a valid encoded filename, it must be at least 5 characters longer
209 // than the current index to account for this percent encoded value.
210 if (i + 5 >= length)
211 return { };
212
213 if (!isASCIIHexDigit(inputString[i + 2]))
214 return { };
215 if (!isASCIIHexDigit(inputString[i + 3]))
216 return { };
217 if (!isASCIIHexDigit(inputString[i + 4]))
218 return { };
219 if (!isASCIIHexDigit(inputString[i + 5]))
220 return { };
221
222 UChar encodedCharacter = toASCIIHexValue(inputString[i + 2], inputString[i + 3]) << 8 | toASCIIHexValue(inputString[i + 4], inputString[i + 5]);
223 result.append(encodedCharacter);
224 i += 5;
225 }
226
227 return result.toString();
228}

230String lastComponentOfPathIgnoringTrailingSlash(const String& path)
231{
232#if OS(WINDOWS)
233 char pathSeparator = '\\';
234#else
235 char pathSeparator = '/';
236#endif
237
238 auto position = path.reverseFind(pathSeparator);
239 if (position == notFound)
240 return path;
241
242 size_t endOfSubstring = path.length() - 1;
243 if (position == endOfSubstring) {
244 --endOfSubstring;
245 position = path.reverseFind(pathSeparator, endOfSubstring);
246 }
247
248 return path.substring(position + 1, endOfSubstring - position);
249}

251bool appendFileContentsToFileHandle(const String& path, PlatformFileHandle& target)
252{
253 auto source = openFile(path, FileOpenMode::Read);
254
255 if (!isHandleValid(source))
256 return false;
257
258 static int bufferSize = 1 << 19;
259 Vector<uint8_t> buffer(bufferSize);
260
261 auto fileCloser = WTF::makeScopeExit([source]() {
262 PlatformFileHandle handle = source;
263 closeFile(handle);
264 });
265
266 do {
267 int readBytes = readFromFile(source, buffer.data(), bufferSize);
268
269 if (readBytes < 0)
270 return false;
271
272 if (writeToFile(target, buffer.data(), readBytes) != readBytes)
273 return false;
274
275 if (readBytes < bufferSize)
276 return true;
277 } while (true);
278
279 ASSERT_NOT_REACHED();
280}

283bool filesHaveSameVolume(const String& fileA, const String& fileB)
284{
285 auto fsRepFileA = fileSystemRepresentation(fileA);
286 auto fsRepFileB = fileSystemRepresentation(fileB);
287
288 if (fsRepFileA.isNull() || fsRepFileB.isNull())
289 return false;
290
291 bool result = false;
292
293 auto fileADev = getFileDeviceId(fsRepFileA);
294 auto fileBDev = getFileDeviceId(fsRepFileB);
295
296 if (fileADev && fileBDev)
297 result = (fileADev == fileBDev);
298
299 return result;
300}

337bool MappedFileData::mapFileHandle(PlatformFileHandle handle, FileOpenMode openMode, MappedFileMode mapMode)
338{
339 if (!isHandleValid(handle))
340 return false;
341
342 int fd;
343#if USE(GLIB)
344 auto* inputStream = g_io_stream_get_input_stream(G_IO_STREAM(handle));
345 fd = g_file_descriptor_based_get_fd(G_FILE_DESCRIPTOR_BASED(inputStream));
346#else
347 fd = handle;
348#endif
349
350 struct stat fileStat;
351 if (fstat(fd, &fileStat)) {
352 return false;
353 }
354
355 unsigned size;
356 if (!WTF::convertSafely(fileStat.st_size, size)) {
357 return false;
358 }
359
360 if (!size)
361 return true;
362
363 int pageProtection = PROT_READ;
364 switch (openMode) {
365 case FileOpenMode::Read:
366 pageProtection = PROT_READ;
367 break;
368 case FileOpenMode::Write:
369 pageProtection = PROT_WRITE;
370 break;
371 case FileOpenMode::ReadWrite:
372 pageProtection = PROT_READ | PROT_WRITE;
373 break;
374#if OS(DARWIN)
375 case FileOpenMode::EventsOnly:
376 ASSERT_NOT_REACHED();
377#endif
378 }
379
380 void* data = mmap(0, size, pageProtection, MAP_FILE | (mapMode == MappedFileMode::Shared ? MAP_SHARED : MAP_PRIVATE), fd, 0);
381
382 if (data == MAP_FAILED) {
383 return false;
384 }
385
386 m_fileData = data;
387 m_fileSize = size;
388 return true;
389}
390#endif

435MappedFileData mapToFile(const String& path, size_t bytesSize, Function<void(const Function<bool(const uint8_t*, size_t)>&)>&& apply, PlatformFileHandle* outputHandle)
436{
437 constexpr bool failIfFileExists = true;
438 auto handle = FileSystem::openFile(path, FileSystem::FileOpenMode::ReadWrite, FileSystem::FileAccessPermission::User, failIfFileExists);
439 if (!FileSystem::isHandleValid(handle) || !FileSystem::truncateFile(handle, bytesSize)) {
440 FileSystem::closeFile(handle);
441 return { };
442 }
443
444 FileSystem::makeSafeToUseMemoryMapForPath(path);
445 bool success;
446 FileSystem::MappedFileData mappedFile(handle, FileSystem::FileOpenMode::ReadWrite, FileSystem::MappedFileMode::Shared, success);
447 if (!success) {
448 FileSystem::closeFile(handle);
449 return { };
450 }
451
452 void* map = const_cast<void*>(mappedFile.data());
453 uint8_t* mapData = static_cast<uint8_t*>(map);
454
455 apply([&mapData](const uint8_t* chunk, size_t chunkSize) {
456 memcpy(mapData, chunk, chunkSize);
457 mapData += chunkSize;
458 return true;
459 });
460
461#if OS(WINDOWS)
462 DWORD oldProtection;
463 VirtualProtect(map, bytesSize, FILE_MAP_READ, &oldProtection);
464 FlushViewOfFile(map, bytesSize);
465#else
466 // Drop the write permission.
467 mprotect(map, bytesSize, PROT_READ);
468
469 // Flush (asynchronously) to file, turning this into clean memory.
470 msync(map, bytesSize, MS_ASYNC);
471#endif
472
473 if (outputHandle)
474 *outputHandle = handle;
475 else
476 FileSystem::closeFile(handle);
477
478 return mappedFile;
479}

481static Salt makeSalt()
482{
483 Salt salt;
484 static_assert(salt.size() == 8, "Salt size");
485 *reinterpret_cast<uint32_t*>(&salt[0]) = cryptographicallyRandomNumber();
486 *reinterpret_cast<uint32_t*>(&salt[4]) = cryptographicallyRandomNumber();
487 return salt;
488}

490std::optional<Salt> readOrMakeSalt(const String& path)
491{
492 if (FileSystem::fileExists(path)) {
493 auto file = FileSystem::openFile(path, FileSystem::FileOpenMode::Read);
494 Salt salt;
495 auto bytesRead = static_cast<std::size_t>(FileSystem::readFromFile(file, salt.data(), salt.size()));
496 FileSystem::closeFile(file);
497 if (bytesRead == salt.size())
498 return salt;
499
500 FileSystem::deleteFile(path);
501 }
502
503 Salt salt = makeSalt();
504 FileSystem::makeAllDirectories(FileSystem::parentPath(path));
505 auto file = FileSystem::openFile(path, FileSystem::FileOpenMode::Write, FileSystem::FileAccessPermission::User);
506 if (!FileSystem::isHandleValid(file))
507 return { };
508
509 bool success = static_cast<std::size_t>(FileSystem::writeToFile(file, salt.data(), salt.size())) == salt.size();
510 FileSystem::closeFile(file);
511 if (!success)
512 return { };
513
514 return salt;
515}

519bool deleteEmptyDirectory(const String& path)
520{
521 std::error_code ec;
522 auto fsPath = toStdFileSystemPath(path);
523
524 auto fileStatus = std::filesystem::symlink_status(fsPath, ec);
525 if (ec || fileStatus.type() != std::filesystem::file_type::directory)
526 return false;
527
528#if PLATFORM(MAC)
529 bool containsSingleDSStoreFile = false;
530 auto entries = std::filesystem::directory_iterator(fsPath, ec);
531 for (auto it = std::filesystem::begin(entries), end = std::filesystem::end(entries); !ec && it != end; it.increment(ec)) {
532 if (it->path().filename() == ".DS_Store")
533 containsSingleDSStoreFile = true;
534 else {
535 containsSingleDSStoreFile = false;
536 break;
537 }
538 }
539 if (containsSingleDSStoreFile)
540 std::filesystem::remove(fsPath / ".DS_Store", ec);
541#endif
542
543 // remove() returns false on error so no need to check ec.
544 return std::filesystem::remove(fsPath, ec);
545}

547bool moveFile(const String& oldPath, const String& newPath)
548{
549 auto fsOldPath = toStdFileSystemPath(oldPath);
550 auto fsNewPath = toStdFileSystemPath(newPath);
551
552 std::error_code ec;
553 std::filesystem::rename(fsOldPath, fsNewPath, ec);
554 if (!ec)
555 return true;
556
557 // Fall back to copying and then deleting source as rename() does not work across volumes.
558 ec = { };
559 std::filesystem::copy(fsOldPath, fsNewPath, std::filesystem::copy_options::overwrite_existing | std::filesystem::copy_options::recursive, ec);
560 if (ec)
561 return false;
562 return std::filesystem::remove_all(fsOldPath, ec);
563}

653enum class ShouldFollowSymbolicLinks { No, Yes };
654static std::optional<FileType> fileTypePotentiallyFollowingSymLinks(const String& path, ShouldFollowSymbolicLinks shouldFollowSymbolicLinks)
655{
656 std::error_code ec;
657 auto status = shouldFollowSymbolicLinks == ShouldFollowSymbolicLinks::Yes ? std::filesystem::status(toStdFileSystemPath(path), ec) : std::filesystem::symlink_status(toStdFileSystemPath(path), ec);

659 return std::nullopt;

661 case std::filesystem::file_type::directory:
662 return FileType::Directory;
663 case std::filesystem::file_type::symlink:
664 return FileType::SymbolicLink;
665 default:
666 break;
667 }
668 return FileType::Regular;
669}
