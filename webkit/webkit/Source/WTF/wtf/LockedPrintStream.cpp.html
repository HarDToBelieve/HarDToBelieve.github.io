26#include "config.h"
27#include <wtf/LockedPrintStream.h>
28
29namespace WTF {
30
31LockedPrintStream::LockedPrintStream(std::unique_ptr<PrintStream> target)
32 : m_target(WTFMove(target))
33{
34}
35
36LockedPrintStream::~LockedPrintStream()
37{
38}
39
40void LockedPrintStream::vprintf(const char* format, va_list args)
41{
42 Locker locker { m_lock };
43 m_target->vprintf(format, args);
44}
45
46void LockedPrintStream::flush()
47{
48 Locker locker { m_lock };
49 m_target->flush();
50}
51
52PrintStream& LockedPrintStream::begin()
53{
54 m_lock.lock();
55 return *m_target;
56}
57
58void LockedPrintStream::end()
59{
60 m_lock.unlock();
61}

63} // namespace WTF
