45static uint32_t greatestCommonDivisor(uint32_t a, uint32_t b)
46{
47 ASSERT(a);
48 ASSERT(b);
49
50 // Euclid's Algorithm
51 uint32_t temp = 0;
52 while (b) {
53 temp = b;
54 b = a % b;
55 a = temp;
56 }
57
58 ASSERT(a);
59 return a;
60}
61
62static uint32_t leastCommonMultiple(uint32_t a, uint32_t b, uint32_t &result)
63{
64 return safeMultiply(a, b / greatestCommonDivisor(a, b), result);
65}

67static int64_t signum(int64_t val)
68{
69 return (0 < val) - (val < 0);
70}

79MediaTime MediaTime::createWithFloat(float floatTime)
80{
81 if (floatTime != floatTime)
82 return invalidTime();
83 if (std::isinf(floatTime))
84 return std::signbit(floatTime) ? negativeInfiniteTime() : positiveInfiniteTime();
85
86 MediaTime value(0, DefaultTimeScale, Valid | DoubleValue);
87 value.m_timeValueAsDouble = floatTime;
88 return value;
89}
90
91MediaTime MediaTime::createWithFloat(float floatTime, uint32_t timeScale)
92{
93 if (floatTime != floatTime)
94 return invalidTime();
95 if (std::isinf(floatTime))
96 return std::signbit(floatTime) ? negativeInfiniteTime() : positiveInfiniteTime();
97 if (floatTime >= maxPlusOne<int64_t>)
98 return positiveInfiniteTime();
99 if (floatTime < std::numeric_limits<int64_t>::min())
100 return negativeInfiniteTime();
101 if (!timeScale)
102 return std::signbit(floatTime) ? negativeInfiniteTime() : positiveInfiniteTime();
103
104 while (floatTime * timeScale >= maxPlusOne<int64_t>)
105 timeScale /= 2;
106 return MediaTime(static_cast<int64_t>(floatTime * timeScale), timeScale, Valid);
107}

121MediaTime MediaTime::createWithDouble(double doubleTime, uint32_t timeScale)
122{
123 if (doubleTime != doubleTime)
124 return invalidTime();
125 if (std::isinf(doubleTime))
126 return std::signbit(doubleTime) ? negativeInfiniteTime() : positiveInfiniteTime();
127 if (doubleTime >= maxPlusOne<int64_t>)
128 return positiveInfiniteTime();
129 if (doubleTime < std::numeric_limits<int64_t>::min())
130 return negativeInfiniteTime();
131 if (!timeScale)
132 return std::signbit(doubleTime) ? negativeInfiniteTime() : positiveInfiniteTime();
133
134 while (doubleTime * timeScale >= maxPlusOne<int64_t>)
135 timeScale /= 2;
136 return MediaTime(static_cast<int64_t>(std::round(doubleTime * timeScale)), timeScale, Valid);
137}

139float MediaTime::toFloat() const
140{
141 if (isInvalid() || isIndefinite())
142 return std::numeric_limits<float>::quiet_NaN();

144 return std::numeric_limits<float>::infinity();

146 return -std::numeric_limits<float>::infinity();

173MediaTime MediaTime::operator+(const MediaTime& rhs) const
174{
175 if (rhs.isInvalid() || isInvalid())
176 return invalidTime();
177
178 if (rhs.isIndefinite() || isIndefinite())
179 return indefiniteTime();
180
181 if (isPositiveInfinite() && rhs.isNegativeInfinite())
182 return invalidTime();
183
184 if (isNegativeInfinite() && rhs.isPositiveInfinite())
185 return invalidTime();
186
187 if (isPositiveInfinite() || rhs.isPositiveInfinite())
188 return positiveInfiniteTime();
189
190 if (isNegativeInfinite() || rhs.isNegativeInfinite())
191 return negativeInfiniteTime();
192
193 if (hasDoubleValue() && rhs.hasDoubleValue())
194 return MediaTime::createWithDouble(m_timeValueAsDouble + rhs.m_timeValueAsDouble);
195
196 if (hasDoubleValue() || rhs.hasDoubleValue())
197 return MediaTime::createWithDouble(toDouble() + rhs.toDouble());
198
199 MediaTime a = *this;
200 MediaTime b = rhs;
201
202 uint32_t commonTimeScale;
203 if (!leastCommonMultiple(a.m_timeScale, b.m_timeScale, commonTimeScale) || commonTimeScale > MaximumTimeScale)
204 commonTimeScale = MaximumTimeScale;
205 a.setTimeScale(commonTimeScale);
206 b.setTimeScale(commonTimeScale);
207 while (!safeAdd(a.m_timeValue, b.m_timeValue, a.m_timeValue)) {
208 if (commonTimeScale == 1)
209 return a.m_timeValue > 0 ? positiveInfiniteTime() : negativeInfiniteTime();
210 commonTimeScale /= 2;
211 a.setTimeScale(commonTimeScale);
212 b.setTimeScale(commonTimeScale);
213 }
214 return a;
215}

217MediaTime MediaTime::operator-(const MediaTime& rhs) const
218{
219 if (rhs.isInvalid() || isInvalid())
220 return invalidTime();
221
222 if (rhs.isIndefinite() || isIndefinite())
223 return indefiniteTime();
224
225 if (isPositiveInfinite() && rhs.isPositiveInfinite())
226 return invalidTime();
227
228 if (isNegativeInfinite() && rhs.isNegativeInfinite())
229 return invalidTime();
230
231 if (isPositiveInfinite() || rhs.isNegativeInfinite())
232 return positiveInfiniteTime();
233
234 if (isNegativeInfinite() || rhs.isPositiveInfinite())
235 return negativeInfiniteTime();
236
237 if (hasDoubleValue() && rhs.hasDoubleValue())
238 return MediaTime::createWithDouble(m_timeValueAsDouble - rhs.m_timeValueAsDouble);
239
240 if (hasDoubleValue() || rhs.hasDoubleValue())
241 return MediaTime::createWithDouble(toDouble() - rhs.toDouble());
242
243 MediaTime a = *this;
244 MediaTime b = rhs;
245
246 uint32_t commonTimeScale;
247 if (!leastCommonMultiple(this->m_timeScale, rhs.m_timeScale, commonTimeScale) || commonTimeScale > MaximumTimeScale)
248 commonTimeScale = MaximumTimeScale;
249 a.setTimeScale(commonTimeScale);
250 b.setTimeScale(commonTimeScale);
251 while (!safeSub(a.m_timeValue, b.m_timeValue, a.m_timeValue)) {
252 if (commonTimeScale == 1)
253 return a.m_timeValue > 0 ? positiveInfiniteTime() : negativeInfiniteTime();
254 commonTimeScale /= 2;
255 a.setTimeScale(commonTimeScale);
256 b.setTimeScale(commonTimeScale);
257 }
258 return a;
259}

261MediaTime MediaTime::operator-() const
262{
263 if (isInvalid())
264 return invalidTime();
265
266 if (isIndefinite())
267 return indefiniteTime();
268
269 if (isPositiveInfinite())
270 return negativeInfiniteTime();
271
272 if (isNegativeInfinite())
273 return positiveInfiniteTime();
274
275 MediaTime negativeTime = *this;
276 if (negativeTime.hasDoubleValue())
277 negativeTime.m_timeValueAsDouble = -negativeTime.m_timeValueAsDouble;
278 else
279 negativeTime.m_timeValue = -negativeTime.m_timeValue;
280 return negativeTime;
281}

283MediaTime MediaTime::operator*(int32_t rhs) const
284{
285 if (isInvalid())
286 return invalidTime();
287
288 if (isIndefinite())
289 return indefiniteTime();
290
291 if (!rhs)
292 return zeroTime();
293
294 if (isPositiveInfinite()) {
295 if (rhs > 0)
296 return positiveInfiniteTime();
297 return negativeInfiniteTime();
298 }
299
300 if (isNegativeInfinite()) {
301 if (rhs > 0)
302 return negativeInfiniteTime();
303 return positiveInfiniteTime();
304 }
305
306 MediaTime a = *this;
307
308 if (a.hasDoubleValue()) {
309 a.m_timeValueAsDouble *= rhs;
310 return a;
311 }
312
313 while (!safeMultiply(a.m_timeValue, rhs, a.m_timeValue)) {
314 if (a.m_timeScale == 1)
315 return signum(a.m_timeValue) == signum(rhs) ? positiveInfiniteTime() : negativeInfiniteTime();
316 a.setTimeScale(a.m_timeScale / 2);
317 }
318
319 return a;
320}

322bool MediaTime::operator!() const
323{
324 return (m_timeFlags == Valid && !m_timeValue)
325 || (m_timeFlags == (Valid | DoubleValue) && !m_timeValueAsDouble)
326 || isInvalid();
327}

329MediaTime::operator bool() const

336MediaTime::ComparisonFlags MediaTime::compare(const MediaTime& rhs) const
337{
338 auto andFlags = m_timeFlags & rhs.m_timeFlags;
339 if (andFlags & (PositiveInfinite | NegativeInfinite | Indefinite))
340 return EqualTo;
341
342 auto orFlags = m_timeFlags | rhs.m_timeFlags;
343 if (!(orFlags & Valid))
344 return EqualTo;
345
346 if (!(andFlags & Valid))
347 return isInvalid() ? GreaterThan : LessThan;
348
349 if (orFlags & NegativeInfinite)
350 return isNegativeInfinite() ? LessThan : GreaterThan;
351
352 if (orFlags & PositiveInfinite)
353 return isPositiveInfinite() ? GreaterThan : LessThan;
354
355 if (orFlags & Indefinite)
356 return isIndefinite() ? GreaterThan : LessThan;
357
358 if (andFlags & DoubleValue) {
359 if (m_timeValueAsDouble == rhs.m_timeValueAsDouble)
360 return EqualTo;
361
362 return m_timeValueAsDouble < rhs.m_timeValueAsDouble ? LessThan : GreaterThan;
363 }
364
365 if (orFlags & DoubleValue) {
366 double a = toDouble();
367 double b = rhs.toDouble();
368 if (a > b)
369 return GreaterThan;
370 if (a < b)
371 return LessThan;
372 return EqualTo;
373 }
374
375 if ((m_timeValue < 0) != (rhs.m_timeValue < 0))
376 return m_timeValue < 0 ? LessThan : GreaterThan;
377
378 if (!m_timeValue && !rhs.m_timeValue)
379 return EqualTo;
380
381 if (m_timeScale == rhs.m_timeScale) {
382 if (m_timeValue == rhs.m_timeValue)
383 return EqualTo;
384 return m_timeValue < rhs.m_timeValue ? LessThan : GreaterThan;
385 }
386
387 if (m_timeValue == rhs.m_timeValue)
388 return m_timeScale < rhs.m_timeScale ? GreaterThan : LessThan;
389
390 if (m_timeValue >= 0) {
391 if (m_timeValue < rhs.m_timeValue && m_timeScale > rhs.m_timeScale)
392 return LessThan;
393
394 if (m_timeValue > rhs.m_timeValue && m_timeScale < rhs.m_timeScale)
395 return GreaterThan;
396 } else {
397 if (m_timeValue < rhs.m_timeValue && m_timeScale < rhs.m_timeScale)
398 return LessThan;
399
400 if (m_timeValue > rhs.m_timeValue && m_timeScale > rhs.m_timeScale)
401 return GreaterThan;
402 }
403
404 int64_t lhsFactor;
405 int64_t rhsFactor;
406 if (safeMultiply(m_timeValue, static_cast<int64_t>(rhs.m_timeScale), lhsFactor)
407 && safeMultiply(rhs.m_timeValue, static_cast<int64_t>(m_timeScale), rhsFactor)) {
408 if (lhsFactor == rhsFactor)
409 return EqualTo;
410 return lhsFactor < rhsFactor ? LessThan : GreaterThan;
411 }
412
413 int64_t rhsWhole = rhs.m_timeValue / rhs.m_timeScale;
414 int64_t lhsWhole = m_timeValue / m_timeScale;
415 if (lhsWhole > rhsWhole)
416 return GreaterThan;
417 if (lhsWhole < rhsWhole)
418 return LessThan;
419
420 int64_t rhsRemain = rhs.m_timeValue % rhs.m_timeScale;
421 int64_t lhsRemain = m_timeValue % m_timeScale;
422 lhsFactor = lhsRemain * rhs.m_timeScale;
423 rhsFactor = rhsRemain * m_timeScale;
424
425 if (lhsFactor == rhsFactor)
426 return EqualTo;
427 return lhsFactor > rhsFactor ? GreaterThan : LessThan;
428}

474void MediaTime::setTimeScale(uint32_t timeScale, RoundingFlags flags)
475{
476 if (hasDoubleValue()) {
477 *this = MediaTime::createWithDouble(m_timeValueAsDouble, timeScale);
478 return;
479 }
480
481 if (!timeScale) {
482 *this = m_timeValue < 0 ? negativeInfiniteTime() : positiveInfiniteTime();
483 return;
484 }
485
486 if (timeScale == m_timeScale)
487 return;
488
489 timeScale = std::min(MaximumTimeScale, timeScale);
490
491#if HAVE(INT128_T)
492 __int128_t newValue = static_cast<__int128_t>(m_timeValue) * timeScale;
493 int64_t remainder = newValue % m_timeScale;
494 newValue = newValue / m_timeScale;
495
496 if (newValue < std::numeric_limits<int64_t>::min()) {
497 *this = negativeInfiniteTime();
498 return;
499 }
500
501 if (newValue > std::numeric_limits<int64_t>::max()) {
502 *this = positiveInfiniteTime();
503 return;
504 }
505#else
506 int64_t newValue = m_timeValue / m_timeScale;
507 int64_t partialRemainder = (m_timeValue % m_timeScale) * timeScale;
508 int64_t remainder = partialRemainder % m_timeScale;
509
510 if (!safeMultiply<int64_t>(newValue, static_cast<int64_t>(timeScale), newValue)
511 || !safeAdd(newValue, partialRemainder / m_timeScale, newValue)) {
512 *this = newValue < 0 ? negativeInfiniteTime() : positiveInfiniteTime();
513 return;
514 }
515#endif
516
517 m_timeValue = newValue;
518 std::swap(m_timeScale, timeScale);
519
520 if (!remainder)
521 return;
522
523 m_timeFlags |= HasBeenRounded;
524 switch (flags) {
525 case RoundingFlags::HalfAwayFromZero:
526 if (static_cast<int64_t>(llabs(remainder)) * 2 >= static_cast<int64_t>(timeScale)) {
527 // round up (away from zero)
528 if (remainder < 0)
529 m_timeValue--;
530 else
531 m_timeValue++;
532 }
533 break;
534
535 case RoundingFlags::TowardZero:
536 break;
537
538 case RoundingFlags::AwayFromZero:
539 if (remainder < 0)
540 m_timeValue--;
541 else
542 m_timeValue++;
543 break;
544
545 case RoundingFlags::TowardPositiveInfinity:
546 if (remainder > 0)
547 m_timeValue++;
548 break;
549
550 case RoundingFlags::TowardNegativeInfinity:
551 if (remainder < 0)
552 m_timeValue--;
553 break;
554 }
555}

573Ref<JSON::Object> MediaTime::toJSONObject() const
574{
575 auto object = JSON::Object::create();
576
577 if (hasDoubleValue()) {
578 object->setDouble("value"_s, toDouble());
579 return object;
580 }
581
582 if (isInvalid())
583 object->setBoolean("invalid"_s, true);
584 else if (isIndefinite())
585 object->setString("value"_s, "NaN"_s);
586 else if (isPositiveInfinite())
587 object->setString("value"_s, "POSITIVE_INFINITY"_s);
588 else if (isNegativeInfinite())
589 object->setString("value"_s, "NEGATIVE_INFINITY"_s);
590 else
591 object->setDouble("value"_s, toDouble());
592
593 object->setDouble("numerator"_s, static_cast<double>(m_timeValue));
594 object->setInteger("denominator"_s, m_timeScale);
595 object->setInteger("flags"_s, m_timeFlags);
596
597 return object;
598}

605MediaTime abs(const MediaTime& rhs)
606{
607 if (rhs.isInvalid())
608 return MediaTime::invalidTime();
609 if (rhs.isNegativeInfinite() || rhs.isPositiveInfinite())
610 return MediaTime::positiveInfiniteTime();
611 if (rhs.hasDoubleValue())
612 return MediaTime::createWithDouble(fabs(rhs.m_timeValueAsDouble));
613
614 MediaTime val = rhs;
615 val.m_timeValue = std::abs(rhs.m_timeValue);
616 return val;
617}
