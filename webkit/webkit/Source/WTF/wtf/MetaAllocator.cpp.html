42MetaAllocator::~MetaAllocator()
43{
44 for (FreeSpaceNode* node = m_freeSpaceSizeMap.first(); node;) {
45 FreeSpaceNode* next = node->successor();
46 m_freeSpaceSizeMap.remove(node);
47 freeFreeSpaceNode(node);
48 node = next;
49 }
50#ifndef NDEBUG
51 ASSERT(!m_mallocBalance);
52#endif
53}

65void MetaAllocator::release(const LockHolder&, MetaAllocatorHandle& handle)
66{
67 if (handle.sizeInBytes()) {
68 MemoryPtr start = handle.start();
69 size_t sizeInBytes = handle.sizeInBytes();
70 decrementPageOccupancy(start.untaggedPtr(), sizeInBytes);
71 addFreeSpaceFromReleasedHandle(FreeSpacePtr(start), sizeInBytes);
72 }
73
74 if (UNLIKELY(!!m_tracker))
75 m_tracker->release(handle);
76}

93void MetaAllocatorHandle::shrink(size_t newSizeInBytes)
94{
95 size_t sizeInBytes = this->sizeInBytes();
96 ASSERT(newSizeInBytes <= sizeInBytes);
97
98 MetaAllocator& allocator = this->allocator();
99 Locker locker { allocator.m_lock };
100
101 newSizeInBytes = allocator.roundUp(newSizeInBytes);
102
103 ASSERT(newSizeInBytes <= sizeInBytes);
104
105 if (newSizeInBytes == sizeInBytes)
106 return;
107
108 MemoryPtr freeStart = m_start + newSizeInBytes;
109 size_t freeSize = sizeInBytes - newSizeInBytes;
110 uintptr_t freeStartValue = freeStart.untaggedPtr<uintptr_t>();
111 uintptr_t freeEnd = freeStartValue + freeSize;
112
113 uintptr_t firstCompletelyFreePage = roundUpToMultipleOf(allocator.m_pageSize, freeStartValue);
114 if (firstCompletelyFreePage < freeEnd)
115 allocator.decrementPageOccupancy(reinterpret_cast<void*>(firstCompletelyFreePage), freeSize - (firstCompletelyFreePage - freeStartValue));
116
117 allocator.addFreeSpaceFromReleasedHandle(MetaAllocator::FreeSpacePtr(freeStart), freeSize);
118
119 m_end = freeStart;
120}

127MetaAllocator::MetaAllocator(Lock& lock, size_t allocationGranule, size_t pageSize)
128 : m_allocationGranule(allocationGranule)
129 , m_pageSize(pageSize)
130 , m_bytesAllocated(0)
131 , m_bytesReserved(0)
132 , m_bytesCommitted(0)
133 , m_lock(lock)
134#ifndef NDEBUG
135 , m_mallocBalance(0)
136#endif
137#if ENABLE(META_ALLOCATOR_PROFILE)
138 , m_numAllocations(0)
139 , m_numFrees(0)
140#endif
141{
142 for (m_logPageSize = 0; m_logPageSize < 32; ++m_logPageSize) {
143 if (static_cast<size_t>(1) << m_logPageSize == m_pageSize)
144 break;
145 }
146
147 ASSERT(static_cast<size_t>(1) << m_logPageSize == m_pageSize);
148
149 for (m_logAllocationGranule = 0; m_logAllocationGranule < 32; ++m_logAllocationGranule) {
150 if (static_cast<size_t>(1) << m_logAllocationGranule == m_allocationGranule)
151 break;
152 }
153
154 ASSERT(static_cast<size_t>(1) << m_logAllocationGranule == m_allocationGranule);
155}

157RefPtr<MetaAllocatorHandle> MetaAllocator::allocate(const LockHolder&, size_t sizeInBytes)
158{
159 if (!sizeInBytes)
160 return nullptr;
161
162 sizeInBytes = roundUp(sizeInBytes);
163
164 FreeSpacePtr start = findAndRemoveFreeSpace(sizeInBytes);
165 if (!start) {
166 size_t requestedNumberOfPages = (sizeInBytes + m_pageSize - 1) >> m_logPageSize;
167 size_t numberOfPages = requestedNumberOfPages;
168
169 start = allocateNewSpace(numberOfPages);
170 if (!start)
171 return nullptr;
172
173 ASSERT(numberOfPages >= requestedNumberOfPages);
174
175 size_t roundedUpSize = numberOfPages << m_logPageSize;
176
177 ASSERT(roundedUpSize >= sizeInBytes);
178
179 m_bytesReserved += roundedUpSize;
180
181 if (roundedUpSize > sizeInBytes) {
182 FreeSpacePtr freeSpaceStart = start + sizeInBytes;
183 size_t freeSpaceSize = roundedUpSize - sizeInBytes;
184 addFreeSpace(freeSpaceStart, freeSpaceSize);
185 }
186 }
187 incrementPageOccupancy(start.untaggedPtr(), sizeInBytes);
188 m_bytesAllocated += sizeInBytes;
189#if ENABLE(META_ALLOCATOR_PROFILE)
190 m_numAllocations++;
191#endif
192
193 auto handle = adoptRef(*new MetaAllocatorHandle(*this, MemoryPtr(start), sizeInBytes));
194
195 if (UNLIKELY(!!m_tracker))
196 m_tracker->notify(*handle.ptr());
197
198 return handle;
199}

210MetaAllocator::FreeSpacePtr MetaAllocator::findAndRemoveFreeSpace(size_t sizeInBytes)
211{
212 FreeSpaceNode* node = m_freeSpaceSizeMap.findLeastGreaterThanOrEqual(sizeInBytes);
213
214 if (!node)
215 return nullptr;
216
217 size_t nodeSizeInBytes = node->sizeInBytes();
218 RELEASE_ASSERT(nodeSizeInBytes >= sizeInBytes);
219
220 m_freeSpaceSizeMap.remove(node);
221
222 FreeSpacePtr result;
223
224 if (nodeSizeInBytes == sizeInBytes) {
225 // Easy case: perfect fit, so just remove the node entirely.
226 result = node->m_start;
227
228 m_freeSpaceStartAddressMap.remove(node->m_start);
229 m_freeSpaceEndAddressMap.remove(node->m_end);
230 freeFreeSpaceNode(node);
231 } else {
232 // Try to be a good citizen and ensure that the returned chunk of memory
233 // straddles as few pages as possible, but only insofar as doing so will
234 // not increase fragmentation. The intuition is that minimizing
235 // fragmentation is a strictly higher priority than minimizing the number
236 // of committed pages, since in the long run, smaller fragmentation means
237 // fewer committed pages and fewer failures in general.
238
239 uintptr_t nodeStartAsInt = node->m_start.untaggedPtr<uintptr_t>();
240 uintptr_t firstPage = nodeStartAsInt >> m_logPageSize;
241 uintptr_t lastPage = (nodeStartAsInt + nodeSizeInBytes - 1) >> m_logPageSize;
242
243 uintptr_t lastPageForLeftAllocation = (nodeStartAsInt + sizeInBytes - 1) >> m_logPageSize;
244 uintptr_t firstPageForRightAllocation = (nodeStartAsInt + nodeSizeInBytes - sizeInBytes) >> m_logPageSize;
245
246 if (lastPageForLeftAllocation - firstPage + 1 <= lastPage - firstPageForRightAllocation + 1) {
247 // Allocate in the left side of the returned chunk, and slide the node to the right.
248 result = node->m_start;
249
250 m_freeSpaceStartAddressMap.remove(node->m_start);
251
252 node->m_start += sizeInBytes;
253 RELEASE_ASSERT(nodeStartAsInt < node->m_start.untaggedPtr<uintptr_t>() && node->m_start.untaggedPtr<uintptr_t>() < node->m_end.untaggedPtr<uintptr_t>());
254
255 m_freeSpaceSizeMap.insert(node);
256 m_freeSpaceStartAddressMap.add(node->m_start, node);
257 } else {
258 // Allocate in the right size of the returned chunk, and slide the node to the left;
259
260 result = node->m_end - sizeInBytes;
261
262 m_freeSpaceEndAddressMap.remove(node->m_end);
263
264 node->m_end = result;
265
266 m_freeSpaceSizeMap.insert(node);
267 m_freeSpaceEndAddressMap.add(result, node);
268 }
269 }
270
271#if ENABLE(META_ALLOCATOR_PROFILE)
272 dumpProfile();
273#endif
274
275 return result;
276}

295size_t MetaAllocator::debugFreeSpaceSize()
296{
297#ifndef NDEBUG
298 Locker locker { m_lock };
299 size_t result = 0;
300 for (FreeSpaceNode* node = m_freeSpaceSizeMap.first(); node; node = node->successor())
301 result += node->sizeInBytes();
302 return result;

309void MetaAllocator::addFreeSpace(FreeSpacePtr start, size_t sizeInBytes)
310{
311 FreeSpacePtr end = start + sizeInBytes;
312
313 HashMap<FreeSpacePtr, FreeSpaceNode*>::iterator leftNeighbor = m_freeSpaceEndAddressMap.find(start);
314 HashMap<FreeSpacePtr, FreeSpaceNode*>::iterator rightNeighbor = m_freeSpaceStartAddressMap.find(end);
315
316 if (leftNeighbor != m_freeSpaceEndAddressMap.end()) {
317 // We have something we can coalesce with on the left. Remove it from the tree, and
318 // remove its end from the end address map.
319
320 ASSERT(leftNeighbor->value->m_end == leftNeighbor->key);
321
322 FreeSpaceNode* leftNode = leftNeighbor->value;
323
324 FreeSpacePtr leftEnd = leftNode->m_end;
325
326 ASSERT(leftEnd == start);
327
328 m_freeSpaceSizeMap.remove(leftNode);
329 m_freeSpaceEndAddressMap.remove(leftEnd);
330
331 // Now check if there is also something to coalesce with on the right.
332 if (rightNeighbor != m_freeSpaceStartAddressMap.end()) {
333 // Freeing something in the middle of free blocks. Coalesce both left and
334 // right, whilst removing the right neighbor from the maps.
335
336 ASSERT(rightNeighbor->value->m_start == rightNeighbor->key);
337
338 FreeSpaceNode* rightNode = rightNeighbor->value;
339 FreeSpacePtr rightStart = rightNeighbor->key;
340 size_t rightSize = rightNode->sizeInBytes();
341 FreeSpacePtr rightEnd = rightNode->m_end;
342
343 ASSERT(rightStart == end);
344 ASSERT(leftNode->m_start + (leftNode->sizeInBytes() + sizeInBytes + rightSize) == rightEnd);
345
346 m_freeSpaceSizeMap.remove(rightNode);
347 m_freeSpaceStartAddressMap.remove(rightStart);
348 m_freeSpaceEndAddressMap.remove(rightEnd);
349
350 freeFreeSpaceNode(rightNode);
351
352 leftNode->m_end += (sizeInBytes + rightSize);
353
354 m_freeSpaceSizeMap.insert(leftNode);
355 m_freeSpaceEndAddressMap.add(rightEnd, leftNode);
356 } else {
357 leftNode->m_end += sizeInBytes;
358
359 m_freeSpaceSizeMap.insert(leftNode);
360 m_freeSpaceEndAddressMap.add(end, leftNode);
361 }
362 } else {
363 // Cannot coalesce with left; try to see if we can coalesce with right.
364
365 if (rightNeighbor != m_freeSpaceStartAddressMap.end()) {
366 FreeSpaceNode* rightNode = rightNeighbor->value;
367 FreeSpacePtr rightStart = rightNeighbor->key;
368
369 ASSERT(rightStart == end);
370 ASSERT(start + (sizeInBytes + rightNode->sizeInBytes()) == rightNode->m_end);
371
372 m_freeSpaceSizeMap.remove(rightNode);
373 m_freeSpaceStartAddressMap.remove(rightStart);
374
375 rightNode->m_start = start;
376
377 m_freeSpaceSizeMap.insert(rightNode);
378 m_freeSpaceStartAddressMap.add(start, rightNode);
379 } else {
380 // Nothing to coalesce with, so create a new free space node and add it.
381
382 FreeSpaceNode* node = allocFreeSpaceNode();
383
384 node->m_start = start;
385 node->m_end = start + sizeInBytes;
386
387 m_freeSpaceSizeMap.insert(node);
388 m_freeSpaceStartAddressMap.add(start, node);
389 m_freeSpaceEndAddressMap.add(end, node);
390 }
391 }

393#if ENABLE(META_ALLOCATOR_PROFILE)
394 dumpProfile();
395#endif

398void MetaAllocator::incrementPageOccupancy(void* address, size_t sizeInBytes)
399{
400 uintptr_t firstPage = reinterpret_cast<uintptr_t>(address) >> m_logPageSize;
401 uintptr_t lastPage = (reinterpret_cast<uintptr_t>(address) + sizeInBytes - 1) >> m_logPageSize;
402
403 uintptr_t currentPageStart = 0;
404 size_t count = 0;
405 auto flushNeedPages = [&] {
406 if (!currentPageStart)
407 return;
408 notifyNeedPage(reinterpret_cast<void*>(currentPageStart << m_logPageSize), count);
409 currentPageStart = 0;
410 count = 0;
411 };
412
413 for (uintptr_t page = firstPage; page <= lastPage; ++page) {
414 auto result = m_pageOccupancyMap.add(page, 1);
415 if (result.isNewEntry) {
416 m_bytesCommitted += m_pageSize;
417 if (!currentPageStart)
418 currentPageStart = page;
419 ++count;
420 } else {
421 result.iterator->value++;
422 flushNeedPages();
423 }
424 }
425 flushNeedPages();
426}

428void MetaAllocator::decrementPageOccupancy(void* address, size_t sizeInBytes)
429{
430 uintptr_t firstPage = reinterpret_cast<uintptr_t>(address) >> m_logPageSize;
431 uintptr_t lastPage = (reinterpret_cast<uintptr_t>(address) + sizeInBytes - 1) >> m_logPageSize;
432
433 uintptr_t currentPageStart = 0;
434 size_t count = 0;
435 auto flushFreePages = [&] {
436 if (!currentPageStart)
437 return;
438 notifyPageIsFree(reinterpret_cast<void*>(currentPageStart << m_logPageSize), count);
439 currentPageStart = 0;
440 count = 0;
441 };
442
443 for (uintptr_t page = firstPage; page <= lastPage; ++page) {
444 HashMap<uintptr_t, size_t>::iterator iter = m_pageOccupancyMap.find(page);
445 ASSERT(iter != m_pageOccupancyMap.end());
446 if (!--(iter->value)) {
447 m_pageOccupancyMap.remove(iter);
448 m_bytesCommitted -= m_pageSize;
449 if (!currentPageStart)
450 currentPageStart = page;
451 ++count;
452 } else
453 flushFreePages();
454 }
455 flushFreePages();
456}

458bool MetaAllocator::isInAllocatedMemory(const AbstractLocker&, void* address)
459{
460 ASSERT(m_lock.isLocked());
461 uintptr_t page = reinterpret_cast<uintptr_t>(address) >> m_logPageSize;
462 return m_pageOccupancyMap.contains(page);
463}

465size_t MetaAllocator::roundUp(size_t sizeInBytes)
466{
467 if (std::numeric_limits<size_t>::max() - m_allocationGranule <= sizeInBytes)
468 CRASH();
469 return (sizeInBytes + m_allocationGranule - 1) & ~(m_allocationGranule - 1);
470}

472MetaAllocator::FreeSpaceNode* MetaAllocator::allocFreeSpaceNode()
473{
474#ifndef NDEBUG
475 m_mallocBalance++;
476#endif
477 return new (NotNull, MetaAllocatorFreeSpaceMalloc::malloc(sizeof(FreeSpaceNode))) FreeSpaceNode();
478}
