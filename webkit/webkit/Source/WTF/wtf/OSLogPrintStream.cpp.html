49void OSLogPrintStream::vprintf(const char* format, va_list argList)
50{
51 Locker lock { m_stringLock };
52 size_t offset = m_offset;
53 size_t freeBytes = m_string.length() - offset;
54 va_list backup;
55 va_copy(backup, argList);
56 size_t bytesWritten = vsnprintf(m_string.mutableData() + offset, freeBytes, format, argList);
57 if (UNLIKELY(bytesWritten >= freeBytes)) {
58 size_t newLength = std::max(bytesWritten + m_string.length(), m_string.length() * 2);
59 m_string.grow(newLength);
60 freeBytes = newLength - offset;
61 bytesWritten = vsnprintf(m_string.mutableData() + offset, freeBytes, format, backup);
62 ASSERT(bytesWritten < freeBytes);
63 }
64
65 size_t newOffset = offset + bytesWritten;
66 char* buffer = m_string.mutableData();
67 bool loggedText = false;
68 do {
69 if (buffer[offset] == '\n') {
70 // Set the new line to a null character so os_log stops copying there.
71 buffer[offset] = '\0';
72 os_log_with_type(m_log, m_logType, "%{public}s", buffer);
73 buffer += offset + 1;
74 newOffset -= offset + 1;
75 offset = 0;
76 loggedText = true;
77 } else
78 offset++;
79 } while (offset < newOffset);
80
81 if (loggedText)
82 memmove(m_string.mutableData(), buffer, newOffset);
83 m_offset = newOffset;
84}
