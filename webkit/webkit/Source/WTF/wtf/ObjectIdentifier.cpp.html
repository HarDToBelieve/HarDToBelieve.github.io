32uint64_t ObjectIdentifierBase::generateIdentifierInternal()
33{
34 static uint64_t current;
35 return ++current;
36}
37
38uint64_t ObjectIdentifierBase::generateThreadSafeIdentifierInternal()
39{
40 static LazyNeverDestroyed<std::atomic<uint64_t>> current;
41 static std::once_flag initializeCurrentIdentifier;
42 std::call_once(initializeCurrentIdentifier, [] {
43 current.construct(0);
44 });
45 return ++current.get();
46}
