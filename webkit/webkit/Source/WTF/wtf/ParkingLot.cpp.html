78 void enqueue(ThreadData* data)
79 {
80 if (verbose)
81 dataLog(toString(Thread::current(), ": enqueueing ", RawPointer(data), " with address = ", RawPointer(data->address), " onto ", RawPointer(this), "\n"));
82 ASSERT(data->address);
83 ASSERT(!data->nextInQueue);

85 if (queueTail) {

87 queueTail = data;
88 return;
89 }
90
91 queueHead = data;
92 queueTail = data;

95 template<typename Functor>
96 void genericDequeue(const Functor& functor)
97 {
98 if (verbose)
99 dataLog(toString(Thread::current(), ": dequeueing from bucket at ", RawPointer(this), "\n"));
100
101 if (!queueHead) {
102 if (verbose)
103 dataLog(toString(Thread::current(), ": empty.\n"));
104 return;
105 }
106
107 // This loop is a very clever abomination. The induction variables are the pointer to the
108 // pointer to the current node, and the pointer to the previous node. This gives us everything
109 // we need to both proceed forward to the next node, and to remove nodes while maintaining the
110 // queueHead/queueTail and all of the nextInQueue links. For example, when we are at the head
111 // element, then removal means rewiring queueHead, and if it was also equal to queueTail, then
112 // we'd want queueTail to be set to nullptr. This works because:
113 //
114 // currentPtr == &queueHead
115 // previous == nullptr
116 //
117 // We remove by setting *currentPtr = (*currentPtr)->nextInQueue, i.e. changing the pointer
118 // that used to point to this node to instead point to this node's successor. Another example:
119 // if we were at the second node in the queue, then we'd have:
120 //
121 // currentPtr == &queueHead->nextInQueue
122 // previous == queueHead
123 //
124 // If this node is not equal to queueTail, then removing it simply means making
125 // queueHead->nextInQueue point to queueHead->nextInQueue->nextInQueue (which the algorithm
126 // achieves by mutating *currentPtr). If this node is equal to queueTail, then we want to set
127 // queueTail to previous, which in this case is queueHead - thus making the queue look like a
128 // proper one-element queue with queueHead == queueTail.
129 bool shouldContinue = true;
130 ThreadData** currentPtr = &queueHead;
131 ThreadData* previous = nullptr;
132
133 MonotonicTime time = MonotonicTime::now();
134 bool timeToBeFair = false;
135 if (time > nextFairTime)
136 timeToBeFair = true;
137
138 bool didDequeue = false;
139
140 while (shouldContinue) {
141 ThreadData* current = *currentPtr;
142 if (verbose)
143 dataLog(toString(Thread::current(), ": got thread ", RawPointer(current), "\n"));
144 if (!current)
145 break;
146 DequeueResult result = functor(current, timeToBeFair);
147 switch (result) {
148 case DequeueResult::Ignore:
149 if (verbose)
150 dataLog(toString(Thread::current(), ": currentPtr = ", RawPointer(currentPtr), ", *currentPtr = ", RawPointer(*currentPtr), "\n"));
151 previous = current;
152 currentPtr = &(*currentPtr)->nextInQueue;
153 break;
154 case DequeueResult::RemoveAndStop:
155 shouldContinue = false;
156 FALLTHROUGH;
157 case DequeueResult::RemoveAndContinue:
158 if (verbose)
159 dataLog(toString(Thread::current(), ": dequeueing ", RawPointer(current), " from ", RawPointer(this), "\n"));
160 if (current == queueTail)
161 queueTail = previous;
162 didDequeue = true;
163 *currentPtr = current->nextInQueue;
164 current->nextInQueue = nullptr;
165 break;
166 }
167 }
168
169 if (timeToBeFair && didDequeue)
170 nextFairTime = time + Seconds::fromMilliseconds(random.get());
171
172 ASSERT(!!queueHead == !!queueTail);
173 }
174
175 ThreadData* dequeue()
176 {
177 ThreadData* result = nullptr;
178 genericDequeue(
179 [&] (ThreadData* element, bool) -> DequeueResult {
180 result = element;
181 return DequeueResult::RemoveAndStop;
182 });
183 return result;
184 }

186 ThreadData* queueHead { nullptr };
187 ThreadData* queueTail { nullptr };

193 MonotonicTime nextFairTime;
194
195 WeakRandom random;
196
197 // Put some distane between buckets in memory. This is one of several mitigations against false
198 // sharing.
199 char padding[64];
200};

204// We track all allocated hashtables so that hashtable resizing doesn't anger leak detectors.
205Vector<Hashtable*>* hashtables;
206WordLock hashtablesLock;

208struct Hashtable {
209 unsigned size;
210 Atomic<Bucket*> data[1];
211
212 static Hashtable* create(unsigned size)
213 {
214 ASSERT(size >= 1);
215
216 Hashtable* result = static_cast<Hashtable*>(
217 fastZeroedMalloc(sizeof(Hashtable) + sizeof(Atomic<Bucket*>) * (size - 1)));
218 result->size = size;
219
220 {
221 // This is not fast and it's not data-access parallel, but that's fine, because
222 // hashtable resizing is guaranteed to be rare and it will never happen in steady
223 // state.
224 WordLockHolder locker(hashtablesLock);
225 if (!hashtables)
226 hashtables = new Vector<Hashtable*>();
227 hashtables->append(result);
228 }
229
230 return result;
231 }
232
233 static void destroy(Hashtable* hashtable)
234 {
235 {
236 // This is not fast, but that's OK. See comment in create().
237 WordLockHolder locker(hashtablesLock);
238 hashtables->removeFirst(hashtable);
239 }
240
241 fastFree(hashtable);
242 }
243};

248// With 64 bytes of padding per bucket, assuming a hashtable is fully populated with buckets, the
249// memory usage per thread will still be less than 1KB.
250const unsigned maxLoadFactor = 3;
251
252const unsigned growthFactor = 2;
253
254unsigned hashAddress(const void* address)
255{
256 return WTF::PtrHash<const void*>::hash(address);
257}

259Hashtable* ensureHashtable()
260{
261 for (;;) {
262 Hashtable* currentHashtable = hashtable.load();
263
264 if (currentHashtable)
265 return currentHashtable;
266
267 if (!currentHashtable) {
268 currentHashtable = Hashtable::create(maxLoadFactor);
269 if (hashtable.compareExchangeWeak(nullptr, currentHashtable)) {
270 if (verbose)
271 dataLog(toString(Thread::current(), ": created initial hashtable ", RawPointer(currentHashtable), "\n"));
272 return currentHashtable;
273 }
274
275 Hashtable::destroy(currentHashtable);
276 }
277 }
278}

280// Locks the hashtable. This reloops in case of rehashing, so the current hashtable may be different
281// after this returns than when you called it. Guarantees that there is a hashtable. This is pretty
282// slow and not scalable, so it's only used during thread creation and for debugging/testing.
283Vector<Bucket*> lockHashtable()
284{
285 for (;;) {
286 Hashtable* currentHashtable = ensureHashtable();
287
288 ASSERT(currentHashtable);
289
290 // Now find all of the buckets. This makes sure that the hashtable is full of buckets so that
291 // we can lock all of the buckets, not just the ones that are materialized.
292 Vector<Bucket*> buckets;
293 for (unsigned i = currentHashtable->size; i--;) {
294 Atomic<Bucket*>& bucketPointer = currentHashtable->data[i];
295
296 for (;;) {
297 Bucket* bucket = bucketPointer.load();
298
299 if (!bucket) {
300 bucket = new Bucket();
301 if (!bucketPointer.compareExchangeWeak(nullptr, bucket)) {
302 delete bucket;
303 continue;
304 }
305 }
306
307 buckets.append(bucket);
308 break;
309 }
310 }
311
312 // Now lock the buckets in the right order.
313 std::sort(buckets.begin(), buckets.end());
314 for (Bucket* bucket : buckets)
315 bucket->lock.lock();
316
317 // If the hashtable didn't change (wasn't rehashed) while we were locking it, then we own it
318 // now.
319 if (hashtable.load() == currentHashtable)
320 return buckets;
321
322 // The hashtable rehashed. Unlock everything and try again.
323 for (Bucket* bucket : buckets)
324 bucket->lock.unlock();
325 }
326}
327
328void unlockHashtable(const Vector<Bucket*>& buckets)
329{
330 for (Bucket* bucket : buckets)
331 bucket->lock.unlock();
332}

334// Rehash the hashtable to handle numThreads threads.
335void ensureHashtableSize(unsigned numThreads)
336{
337 // We try to ensure that the size of the hashtable used for thread queues is always large enough
338 // to avoid collisions. So, since we started a new thread, we may need to increase the size of the
339 // hashtable. This does just that. Note that we never free the old spine, since we never lock
340 // around spine accesses (i.e. the "hashtable" global variable).
341
342 // First do a fast check to see if rehashing is needed.
343 Hashtable* oldHashtable = hashtable.load();
344 if (oldHashtable && static_cast<double>(oldHashtable->size) / static_cast<double>(numThreads) >= maxLoadFactor) {
345 if (verbose)
346 dataLog(toString(Thread::current(), ": no need to rehash because ", oldHashtable->size, " / ", numThreads, " >= ", maxLoadFactor, "\n"));
347 return;
348 }
349
350 // Seems like we *might* have to rehash, so lock the hashtable and try again.
351 Vector<Bucket*> bucketsToUnlock = lockHashtable();
352
353 // Check again, since the hashtable could have rehashed while we were locking it. Also,
354 // lockHashtable() creates an initial hashtable for us.
355 oldHashtable = hashtable.load();
356 RELEASE_ASSERT(oldHashtable);
357 if (static_cast<double>(oldHashtable->size) / static_cast<double>(numThreads) >= maxLoadFactor) {
358 if (verbose)
359 dataLog(toString(Thread::current(), ": after locking, no need to rehash because ", oldHashtable->size, " / ", numThreads, " >= ", maxLoadFactor, "\n"));
360 unlockHashtable(bucketsToUnlock);
361 return;
362 }
363
364 Vector<Bucket*> reusableBuckets = bucketsToUnlock;
365
366 // OK, now we resize. First we gather all thread datas from the old hashtable. These thread datas
367 // are placed into the vector in queue order.
368 Vector<ThreadData*> threadDatas;
369 for (Bucket* bucket : reusableBuckets) {
370 while (ThreadData* threadData = bucket->dequeue())
371 threadDatas.append(threadData);
372 }
373
374 unsigned newSize = numThreads * growthFactor * maxLoadFactor;
375 RELEASE_ASSERT(newSize > oldHashtable->size);
376
377 Hashtable* newHashtable = Hashtable::create(newSize);
378 if (verbose)
379 dataLog(toString(Thread::current(), ": created new hashtable: ", RawPointer(newHashtable), "\n"));
380 for (ThreadData* threadData : threadDatas) {
381 if (verbose)
382 dataLog(toString(Thread::current(), ": rehashing thread data ", RawPointer(threadData), " with address = ", RawPointer(threadData->address), "\n"));
383 unsigned hash = hashAddress(threadData->address);
384 unsigned index = hash % newHashtable->size;
385 if (verbose)
386 dataLog(toString(Thread::current(), ": index = ", index, "\n"));
387 Bucket* bucket = newHashtable->data[index].load();
388 if (!bucket) {
389 if (reusableBuckets.isEmpty())
390 bucket = new Bucket();
391 else
392 bucket = reusableBuckets.takeLast();
393 newHashtable->data[index].store(bucket);
394 }
395
396 bucket->enqueue(threadData);
397 }
398
399 // At this point there may be some buckets left unreused. This could easily happen if the
400 // number of enqueued threads right now is low but the high watermark of the number of threads
401 // enqueued was high. We place these buckets into the hashtable basically at random, just to
402 // make sure we don't leak them.
403 for (unsigned i = 0; i < newHashtable->size && !reusableBuckets.isEmpty(); ++i) {
404 Atomic<Bucket*>& bucketPtr = newHashtable->data[i];
405 if (bucketPtr.load())
406 continue;
407 bucketPtr.store(reusableBuckets.takeLast());
408 }
409
410 // Since we increased the size of the hashtable, we should have exhausted our preallocated
411 // buckets by now.
412 ASSERT(reusableBuckets.isEmpty());
413
414 // OK, right now the old hashtable is locked up and the new hashtable is ready to rock and
415 // roll. After we install the new hashtable, we can release all bucket locks.
416
417 bool result = hashtable.compareExchangeStrong(oldHashtable, newHashtable) == oldHashtable;
418 RELEASE_ASSERT(result);
419
420 unlockHashtable(bucketsToUnlock);
421}

423ThreadData::ThreadData()
424 : thread(Thread::current())
425{
426 unsigned currentNumThreads;
427 for (;;) {
428 unsigned oldNumThreads = numThreads.load();
429 currentNumThreads = oldNumThreads + 1;
430 if (numThreads.compareExchangeWeak(oldNumThreads, currentNumThreads))
431 break;
432 }
433
434 ensureHashtableSize(currentNumThreads);
435}
436
437ThreadData::~ThreadData()
438{
439 for (;;) {
440 unsigned oldNumThreads = numThreads.load();
441 if (numThreads.compareExchangeWeak(oldNumThreads, oldNumThreads - 1))
442 break;
443 }
444}

464template<typename Functor>
465bool enqueue(const void* address, const Functor& functor)
466{
467 unsigned hash = hashAddress(address);
468
469 for (;;) {
470 Hashtable* myHashtable = ensureHashtable();
471 unsigned index = hash % myHashtable->size;
472 Atomic<Bucket*>& bucketPointer = myHashtable->data[index];
473 Bucket* bucket;
474 for (;;) {
475 bucket = bucketPointer.load();
476 if (!bucket) {
477 bucket = new Bucket();
478 if (!bucketPointer.compareExchangeWeak(nullptr, bucket)) {
479 delete bucket;
480 continue;
481 }
482 }
483 break;
484 }
485 if (verbose)
486 dataLog(toString(Thread::current(), ": enqueueing onto bucket ", RawPointer(bucket), " with index ", index, " for address ", RawPointer(address), " with hash ", hash, "\n"));
487 bucket->lock.lock();
488
489 // At this point the hashtable could have rehashed under us.
490 if (hashtable.load() != myHashtable) {
491 bucket->lock.unlock();
492 continue;
493 }
494
495 ThreadData* threadData = functor();
496 bool result;
497 if (threadData) {
498 if (verbose)
499 dataLog(toString(Thread::current(), ": proceeding to enqueue ", RawPointer(threadData), "\n"));
500 bucket->enqueue(threadData);
501 result = true;
502 } else
503 result = false;
504 bucket->lock.unlock();
505 return result;
506 }
507}

509enum class BucketMode {
510 EnsureNonEmpty,
511 IgnoreEmpty
512};
513
514template<typename DequeueFunctor, typename FinishFunctor>
515bool dequeue(
516 const void* address, BucketMode bucketMode, const DequeueFunctor& dequeueFunctor,
517 const FinishFunctor& finishFunctor)
518{
519 unsigned hash = hashAddress(address);
520
521 for (;;) {
522 Hashtable* myHashtable = ensureHashtable();
523 unsigned index = hash % myHashtable->size;
524 Atomic<Bucket*>& bucketPointer = myHashtable->data[index];
525 Bucket* bucket = bucketPointer.load();
526 if (!bucket) {
527 if (bucketMode == BucketMode::IgnoreEmpty)
528 return false;
529
530 for (;;) {
531 bucket = bucketPointer.load();
532 if (!bucket) {
533 bucket = new Bucket();
534 if (!bucketPointer.compareExchangeWeak(nullptr, bucket)) {
535 delete bucket;
536 continue;
537 }
538 }
539 break;
540 }
541 }
542
543 bucket->lock.lock();
544
545 // At this point the hashtable could have rehashed under us.
546 if (hashtable.load() != myHashtable) {
547 bucket->lock.unlock();
548 continue;
549 }
550
551 bucket->genericDequeue(dequeueFunctor);
552 bool result = !!bucket->queueHead;
553 finishFunctor(result);
554 bucket->lock.unlock();
555 return result;
556 }
557}

561NEVER_INLINE ParkingLot::ParkResult ParkingLot::parkConditionallyImpl(
562 const void* address,
563 const ScopedLambda<bool()>& validation,
564 const ScopedLambda<void()>& beforeSleep,
565 const TimeWithDynamicClockType& timeout)
566{
567 if (verbose)
568 dataLog(toString(Thread::current(), ": parking.\n"));
569
570 ThreadData* me = myThreadData();
571 me->token = 0;
572
573 // Guard against someone calling parkConditionally() recursively from beforeSleep().
574 RELEASE_ASSERT(!me->address);
575
576 bool enqueueResult = enqueue(
577 address,
578 [&] () -> ThreadData* {
579 if (!validation())
580 return nullptr;
581
582 me->address = address;
583 return me;
584 });
585
586 if (!enqueueResult)
587 return ParkResult();
588
589 beforeSleep();
590
591 bool didGetDequeued;
592 {
593 MutexLocker locker(me->parkingLock);
594 while (me->address && timeout.nowWithSameClock() < timeout) {
595 me->parkingCondition.timedWait(
596 me->parkingLock, timeout.approximateWallTime());
597
598 // It's possible for the OS to decide not to wait. If it does that then it will also
599 // decide not to release the lock. If there's a bug in the time math, then this could
600 // result in a deadlock. Flashing the lock means that at worst it's just a CPU-eating
601 // spin.
602 me->parkingLock.unlock();
603 me->parkingLock.lock();
604 }
605 ASSERT(!me->address || me->address == address);
606 didGetDequeued = !me->address;
607 }
608
609 if (didGetDequeued) {
610 // Great! We actually got dequeued rather than the timeout expiring.
611 ParkResult result;
612 result.wasUnparked = true;
613 result.token = me->token;
614 return result;
615 }
616
617 // Have to remove ourselves from the queue since we timed out and nobody has dequeued us yet.
618
619 bool didDequeue = false;
620 dequeue(
621 address, BucketMode::IgnoreEmpty,
622 [&] (ThreadData* element, bool) {
623 if (element == me) {
624 didDequeue = true;
625 return DequeueResult::RemoveAndStop;
626 }
627 return DequeueResult::Ignore;
628 },
629 [] (bool) { });
630
631 // If didDequeue is true, then we dequeued ourselves. This means that we were not unparked.
632 // If didDequeue is false, then someone unparked us.
633
634 RELEASE_ASSERT(!me->nextInQueue);
635
636 // Make sure that no matter what, me->address is null after this point.
637 {
638 MutexLocker locker(me->parkingLock);
639 if (!didDequeue) {
640 // If we did not dequeue ourselves, then someone else did. They will set our address to
641 // null. We don't want to proceed until they do this, because otherwise, they may set
642 // our address to null in some distant future when we're already trying to wait for
643 // other things.
644 while (me->address)
645 me->parkingCondition.wait(me->parkingLock);
646 }
647 me->address = nullptr;
648 }
649
650 ParkResult result;
651 result.wasUnparked = !didDequeue;
652 if (!didDequeue) {
653 // If we were unparked then there should be a token.
654 result.token = me->token;
655 }
656 return result;
657}

659NEVER_INLINE ParkingLot::UnparkResult ParkingLot::unparkOne(const void* address)
660{
661 if (verbose)
662 dataLog(toString(Thread::current(), ": unparking one.\n"));
663
664 UnparkResult result;
665
666 RefPtr<ThreadData> threadData;
667 result.mayHaveMoreThreads = dequeue(
668 address,
669 // Why is this here?
670 // FIXME: It seems like this could be IgnoreEmpty, but I switched this to EnsureNonEmpty
671 // without explanation in r199760. We need it to use EnsureNonEmpty if we need to perform
672 // some operation while holding the bucket lock, which usually goes into the finish func.
673 // But if that operation is a no-op, then it's not clear why we need this.
674 BucketMode::EnsureNonEmpty,
675 [&] (ThreadData* element, bool) {
676 if (element->address != address)
677 return DequeueResult::Ignore;
678 threadData = element;
679 result.didUnparkThread = true;
680 return DequeueResult::RemoveAndStop;
681 },
682 [] (bool) { });
683
684 if (!threadData) {
685 ASSERT(!result.didUnparkThread);
686 result.mayHaveMoreThreads = false;
687 return result;
688 }
689
690 ASSERT(threadData->address);
691
692 {
693 MutexLocker locker(threadData->parkingLock);
694 threadData->address = nullptr;
695 threadData->token = 0;
696 }
697 threadData->parkingCondition.signal();
698
699 return result;
700}

702NEVER_INLINE void ParkingLot::unparkOneImpl(
703 const void* address,
704 const ScopedLambda<intptr_t(ParkingLot::UnparkResult)>& callback)
705{
706 if (verbose)
707 dataLog(toString(Thread::current(), ": unparking one the hard way.\n"));
708
709 RefPtr<ThreadData> threadData;
710 bool timeToBeFair = false;
711 dequeue(
712 address,
713 BucketMode::EnsureNonEmpty,
714 [&] (ThreadData* element, bool passedTimeToBeFair) {
715 if (element->address != address)
716 return DequeueResult::Ignore;
717 threadData = element;
718 timeToBeFair = passedTimeToBeFair;
719 return DequeueResult::RemoveAndStop;
720 },
721 [&] (bool mayHaveMoreThreads) {
722 UnparkResult result;
723 result.didUnparkThread = !!threadData;
724 result.mayHaveMoreThreads = result.didUnparkThread && mayHaveMoreThreads;
725 if (timeToBeFair)
726 RELEASE_ASSERT(threadData);
727 result.timeToBeFair = timeToBeFair;
728 intptr_t token = callback(result);
729 if (threadData)
730 threadData->token = token;
731 });
732
733 if (!threadData)
734 return;
735
736 ASSERT(threadData->address);
737
738 {
739 MutexLocker locker(threadData->parkingLock);
740 threadData->address = nullptr;
741 }
742 // At this point, the threadData may die. Good thing we have a RefPtr<> on it.
743 threadData->parkingCondition.signal();
744}

746NEVER_INLINE unsigned ParkingLot::unparkCount(const void* address, unsigned count)
747{
748 if (!count)
749 return 0;
750
751 if (verbose)
752 dataLog(toString(Thread::current(), ": unparking count = ", count, " from ", RawPointer(address), ".\n"));
753
754 Vector<RefPtr<ThreadData>, 8> threadDatas;
755 dequeue(
756 address,
757 // FIXME: It seems like this ought to be EnsureNonEmpty if we follow what unparkOne() does,
758 // but that seems wrong.
759 BucketMode::IgnoreEmpty,
760 [&] (ThreadData* element, bool) {
761 if (verbose)
762 dataLog(toString(Thread::current(), ": Observing element with address = ", RawPointer(element->address), "\n"));
763 if (element->address != address)
764 return DequeueResult::Ignore;
765 threadDatas.append(element);
766 if (threadDatas.size() == count)
767 return DequeueResult::RemoveAndStop;
768 return DequeueResult::RemoveAndContinue;
769 },
770 [] (bool) { });
771
772 for (RefPtr<ThreadData>& threadData : threadDatas) {
773 if (verbose)
774 dataLog(toString(Thread::current(), ": unparking ", RawPointer(threadData.get()), " with address ", RawPointer(threadData->address), "\n"));
775 ASSERT(threadData->address);
776 {
777 MutexLocker locker(threadData->parkingLock);
778 threadData->address = nullptr;
779 }
780 threadData->parkingCondition.signal();
781 }
782
783 if (verbose)
784 dataLog(toString(Thread::current(), ": done unparking.\n"));
785
786 return threadDatas.size();
787}

790{
791 unparkCount(address, UINT_MAX);
792}

794NEVER_INLINE void ParkingLot::forEachImpl(const ScopedLambda<void(Thread&, const void*)>& callback)
795{
796 Vector<Bucket*> bucketsToUnlock = lockHashtable();
797
798 Hashtable* currentHashtable = hashtable.load();
799 for (unsigned i = currentHashtable->size; i--;) {
800 Bucket* bucket = currentHashtable->data[i].load();
801 if (!bucket)
802 continue;
803 for (ThreadData* currentThreadData = bucket->queueHead; currentThreadData; currentThreadData = currentThreadData->nextInQueue)
804 callback(currentThreadData->thread.get(), currentThreadData->address);
805 }
806
807 unlockHashtable(bucketsToUnlock);
808}
