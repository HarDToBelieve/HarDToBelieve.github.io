26#include "config.h"
27#include <wtf/ProcessPrivilege.h>
28
29#include <wtf/OptionSet.h>
30
31namespace WTF {
32
33OptionSet<ProcessPrivilege> allPrivileges()
34{
35 return {
36 ProcessPrivilege::CanAccessRawCookies,
37 ProcessPrivilege::CanAccessCredentials,
38 ProcessPrivilege::CanCommunicateWithWindowServer,
39 };
40}
41
42static OptionSet<ProcessPrivilege>& processPrivileges()
43{
44 static OptionSet<ProcessPrivilege> privileges = { };
45 return privileges;
46}
47
48void setProcessPrivileges(OptionSet<ProcessPrivilege> privileges)
49{
50 processPrivileges() = privileges;
51}
52
53bool hasProcessPrivilege(ProcessPrivilege privilege)
54{
55 return processPrivileges().contains(privilege);
56}
57
58void addProcessPrivilege(ProcessPrivilege privilege)
59{
60 processPrivileges().add(privilege);
61}
62
63void removeProcessPrivilege(ProcessPrivilege privilege)
64{
65 processPrivileges().remove(privilege);
66}
67
68} // namespace WTF
