52
53typedef HashCountedSet<const char*, PtrHash<const char*>> ReasonSet;
54static ReasonSet* leakMessageSuppressionReasons;
55
56void RefCountedLeakCounter::suppressMessages(const char* reason)
57{
58 if (!leakMessageSuppressionReasons)
59 leakMessageSuppressionReasons = new ReasonSet;
60 leakMessageSuppressionReasons->add(reason);
61}
62
63void RefCountedLeakCounter::cancelMessageSuppression(const char* reason)
64{
65 ASSERT(leakMessageSuppressionReasons);
66 ASSERT(leakMessageSuppressionReasons->contains(reason));
67 leakMessageSuppressionReasons->remove(reason);
68}
69
70RefCountedLeakCounter::RefCountedLeakCounter(const char* description)
71 : m_description(description)
72{
73}
74
75RefCountedLeakCounter::~RefCountedLeakCounter()
76{
77 static bool loggedSuppressionReason;
78 if (m_count) {
79 if (!leakMessageSuppressionReasons || leakMessageSuppressionReasons->isEmpty())
80 LOG(RefCountedLeaks, "LEAK: %u %s", m_count.load(), m_description);
81 else if (!loggedSuppressionReason) {
82 // This logs only one reason. Later we could change it so we log all the reasons.
83 LOG(RefCountedLeaks, "No leak checking done: %s", leakMessageSuppressionReasons->begin()->key);
84 loggedSuppressionReason = true;
85 }
86 }
87}
88
89void RefCountedLeakCounter::increment()
90{
91 ++m_count;
92}
93
94void RefCountedLeakCounter::decrement()
95{
96 --m_count;
97}
