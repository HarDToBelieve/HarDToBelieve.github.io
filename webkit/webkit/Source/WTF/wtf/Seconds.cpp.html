39WallTime Seconds::operator+(WallTime other) const
40{
41 return other + *this;
42}
43
44MonotonicTime Seconds::operator+(MonotonicTime other) const
45{
46 return other + *this;
47}
48
49TimeWithDynamicClockType Seconds::operator+(const TimeWithDynamicClockType& other) const
50{
51 return other + *this;
52}
53
54WallTime Seconds::operator-(WallTime other) const
55{
56 return WallTime::fromRawSeconds(value() - other.secondsSinceEpoch().value());
57}
58
59MonotonicTime Seconds::operator-(MonotonicTime other) const
60{
61 return MonotonicTime::fromRawSeconds(value() - other.secondsSinceEpoch().value());
62}
63
64TimeWithDynamicClockType Seconds::operator-(const TimeWithDynamicClockType& other) const
65{
66 return other.withSameClockAndRawSeconds(value() - other.secondsSinceEpoch().value());
67}

69void Seconds::dump(PrintStream& out) const
70{
71 out.print(m_value, " sec");

74TextStream& operator<<(TextStream& ts, Seconds seconds)
75{
76 ts << seconds.value() << "s";
77 return ts;
78}

80void sleep(Seconds value)
81{
82 // It's very challenging to find portable ways of sleeping for less than a second. On UNIX, you want to
83 // use usleep() but it's hard to #include it in a portable way (you'd think it's in unistd.h, but then
84 // you'd be wrong on some OSX SDKs). Also, usleep() won't save you on Windows. Hence, bottoming out in
85 // lock code, which already solves the sleeping problem, is probably for the best.
86
87 Lock fakeLock;
88 Condition fakeCondition;
89 Locker fakeLocker { fakeLock };
90 fakeCondition.waitFor(fakeLock, value);
91}
