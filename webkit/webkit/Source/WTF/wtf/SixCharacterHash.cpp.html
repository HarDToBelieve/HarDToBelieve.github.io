33unsigned sixCharacterHashStringToInteger(const char* string)
34{
35 unsigned hash = 0;
36
37 for (unsigned i = 0; i < 6; ++i) {
38 hash *= 62;
39 unsigned c = string[i];
40 RELEASE_ASSERT(c); // FIXME: Why does this need to be a RELEASE_ASSERT?
41 if (isASCIIUpper(c)) {
42 hash += c - 'A';
43 continue;
44 }
45 if (isASCIILower(c)) {
46 hash += c - 'a' + 26;
47 continue;
48 }
49 ASSERT(isASCIIDigit(c));
50 hash += c - '0' + 26 * 2;
51 }
52
53 RELEASE_ASSERT(!string[6]); // FIXME: Why does this need to be a RELEASE_ASSERT?
54
55 return hash;
56}
57
58std::array<char, 7> integerToSixCharacterHashString(unsigned hash)
59{
60 static const char table[63] = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
61 std::array<char, 7> buffer;
62 unsigned accumulator = hash;
63 for (unsigned i = 6; i--;) {
64 buffer[i] = table[accumulator % 62];
65 accumulator /= 62;
66 }
67 buffer[6] = 0;
68 return buffer;
69}
