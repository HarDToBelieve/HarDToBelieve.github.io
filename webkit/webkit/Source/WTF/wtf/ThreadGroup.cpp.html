26#include "config.h"
27#include <wtf/ThreadGroup.h>
28
29namespace WTF {
30
31ThreadGroup::~ThreadGroup()
32{
33 Locker locker { m_lock };
34 for (auto& thread : m_threads)
35 thread->removeFromThreadGroup(locker, *this);
36}
37
38ThreadGroupAddResult ThreadGroup::add(const AbstractLocker& locker, Thread& thread)
39{
40 return thread.addToThreadGroup(locker, *this);
41}
42
43ThreadGroupAddResult ThreadGroup::add(Thread& thread)
44{
45 Locker locker { m_lock };
46 return add(locker, thread);
47}
48
49ThreadGroupAddResult ThreadGroup::addCurrentThread()
50{
51 auto result = add(Thread::current());
52 ASSERT(result != ThreadGroupAddResult::NotAdded);
53 return result;
54}
