81StringView URL::lastPathComponent() const
82{
83 if (!hasPath())
84 return { };
85
86 unsigned end = m_pathEnd - 1;
87 if (m_string[end] == '/')
88 --end;
89
90 size_t start = m_string.reverseFind('/', end);
91 if (start < pathStart())
92 return { };
93 ++start;
94
95 return StringView(m_string).substring(start, end - start + 1);
96}

159static std::optional<LChar> decodeEscapeSequence(StringView input, unsigned index, unsigned length)
160{
161 if (index + 3 > length || input[index] != '%')
162 return std::nullopt;
163 auto digit1 = input[index + 1];
164 auto digit2 = input[index + 2];
165 if (!isASCIIHexDigit(digit1) || !isASCIIHexDigit(digit2))
166 return std::nullopt;
167 return toASCIIHexValue(digit1, digit2);
168}

170static String decodeEscapeSequencesFromParsedURL(StringView input)
171{
172 ASSERT(input.isAllASCII());
173
174 auto length = input.length();
175 if (length < 3 || !input.contains('%'))
176 return input.toString();
177
178 // FIXME: This 100 is arbitrary. Should make a histogram of how this function is actually used to choose a better value.
179 Vector<LChar, 100> percentDecoded;
180 percentDecoded.reserveInitialCapacity(length);
181 for (unsigned i = 0; i < length; ) {
182 if (auto decodedCharacter = decodeEscapeSequence(input, i, length)) {
183 percentDecoded.uncheckedAppend(*decodedCharacter);
184 i += 3;
185 } else {
186 percentDecoded.uncheckedAppend(input[i]);
187 ++i;
188 }
189 }
190
191 // FIXME: Is UTF-8 always the correct encoding?
192 // FIXME: This returns a null string when we encounter an invalid UTF-8 sequence. Is that OK?
193 return String::fromUTF8(percentDecoded.data(), percentDecoded.size());
194}

324bool URL::protocolIs(const char* protocol) const
325{
326 assertProtocolIsGood(protocol);
327
328 // JavaScript URLs are "valid" and should be executed even if URL decides they are invalid.
329 // The free function protocolIsJavaScript() should be used instead.
330 ASSERT(!equalLettersIgnoringASCIICase(StringView(protocol), "javascript"));
331
332 if (!m_isValid)
333 return false;
334
335 // Do the comparison without making a new string object.
336 for (unsigned i = 0; i < m_schemeEnd; ++i) {
337 if (!protocol[i] || !isASCIIAlphaCaselessEqual(m_string[i], protocol[i]))
338 return false;
339 }
340 return !protocol[m_schemeEnd]; // We should have consumed all characters in the argument.
341}

343bool URL::protocolIs(StringView protocol) const
344{
345 assertProtocolIsGood(protocol);
346
347 if (!m_isValid)
348 return false;
349
350 if (m_schemeEnd != protocol.length())
351 return false;

353 // Do the comparison without making a new string object.
354 for (unsigned i = 0; i < m_schemeEnd; ++i) {
355 if (!isASCIIAlphaCaselessEqual(m_string[i], protocol[i]))
356 return false;
357 }
358 return true;

377bool URL::setProtocol(StringView newProtocol)
378{
379 // Firefox and IE remove everything after the first ':'.
380 auto newProtocolPrefix = newProtocol.substring(0, newProtocol.find(':'));
381 auto newProtocolCanonicalized = URLParser::maybeCanonicalizeScheme(newProtocolPrefix);
382 if (!newProtocolCanonicalized)
383 return false;
384
385 if (!m_isValid) {
386 parse(makeString(*newProtocolCanonicalized, ':', m_string));
387 return true;
388 }
389
390 if ((m_passwordEnd != m_userStart || port()) && *newProtocolCanonicalized == "file")
391 return true;
392
393 if (isLocalFile() && host().isEmpty())
394 return true;
395
396 parse(makeString(*newProtocolCanonicalized, StringView(m_string).substring(m_schemeEnd)));
397 return true;
398}

400// Appends the punycoded hostname identified by the given string and length to
401// the output buffer. The result will not be null terminated.
402// Return value of false means error in encoding.
403static bool appendEncodedHostname(Vector<UChar, 512>& buffer, StringView string)
404{
405 // hostnameBuffer needs to be big enough to hold an IDN-encoded name.
406 // For host names bigger than this, we won't do IDN encoding, which is almost certainly OK.
407 if (string.length() > URLParser::hostnameBufferLength || string.isAllASCII()) {
408 append(buffer, string);
409 return true;
410 }
411
412 UChar hostnameBuffer[URLParser::hostnameBufferLength];
413 UErrorCode error = U_ZERO_ERROR;
414 UIDNAInfo processingDetails = UIDNA_INFO_INITIALIZER;
415 int32_t numCharactersConverted = uidna_nameToASCII(&URLParser::internationalDomainNameTranscoder(),
416 string.upconvertedCharacters(), string.length(), hostnameBuffer, URLParser::hostnameBufferLength, &processingDetails, &error);
417
418 if (U_SUCCESS(error) && !(processingDetails.errors & ~URLParser::allowedNameToASCIIErrors) && numCharactersConverted) {
419 buffer.append(hostnameBuffer, numCharactersConverted);
420 return true;
421 }
422 return false;
423}

451void URL::setHost(StringView newHost)
452{
453 if (!m_isValid)
454 return;
455
456 if (newHost.contains(':') && !newHost.startsWith('['))
457 return;
458
459 if (auto index = newHost.find(hasSpecialScheme() ? slashHashOrQuestionMark : forwardSlashHashOrQuestionMark); index != notFound)
460 newHost = newHost.substring(0, index);
461
462 Vector<UChar, 512> encodedHostName;
463 if (!appendEncodedHostname(encodedHostName, newHost))
464 return;
465
466 bool slashSlashNeeded = m_userStart == m_schemeEnd + 1U;
467 parse(makeString(
468 StringView(m_string).left(hostStart()),
469 slashSlashNeeded ? "//" : "",
470 StringView(encodedHostName.data(), encodedHostName.size()),
471 StringView(m_string).substring(m_hostEnd)
472 ));
473}

493void URL::setHostAndPort(StringView hostAndPort)
494{
495 if (!m_isValid)
496 return;
497
498 auto hostName = hostAndPort;
499 StringView portString;
500 auto colonIndex = hostName.reverseFind(':');
501 if (colonIndex != notFound) {
502 portString = hostName.substring(colonIndex + 1);
503 hostName = hostName.substring(0, colonIndex);
504 // Multiple colons are acceptable only in case of IPv6.
505 if (hostName.contains(':') && !hostName.startsWith('['))
506 return;
507 if (!parseInteger<uint16_t>(portString))
508 portString = { };
509 }
510 if (hostName.isEmpty()) {
511 remove(hostStart(), pathStart() - hostStart());
512 return;
513 }
514
515 Vector<UChar, 512> encodedHostName;
516 if (!appendEncodedHostname(encodedHostName, hostName))
517 return;
518
519 bool slashSlashNeeded = m_userStart == m_schemeEnd + 1U;
520 parse(makeString(
521 StringView(m_string).left(hostStart()),
522 slashSlashNeeded ? "//" : "",
523 StringView(encodedHostName.data(), encodedHostName.size()),
524 portString.isEmpty() ? "" : ":",
525 portString,
526 StringView(m_string).substring(pathStart())
527 ));
528}

530template<typename StringType>
531static String percentEncodeCharacters(const StringType& input, bool(*shouldEncode)(UChar))
532{
533 auto encode = [shouldEncode] (const StringType& input) {
534 CString utf8 = input.utf8();
535 auto* data = utf8.data();
536 StringBuilder builder;
537 auto length = utf8.length();
538 for (unsigned j = 0; j < length; j++) {
539 auto c = data[j];
540 if (shouldEncode(c)) {
541 builder.append('%');
542 builder.append(upperNibbleToASCIIHexDigit(c));
543 builder.append(lowerNibbleToASCIIHexDigit(c));
544 } else
545 builder.append(c);
546 }
547 return builder.toString();
548 };
549
550 for (size_t i = 0; i < input.length(); ++i) {
551 if (UNLIKELY(shouldEncode(input[i])))
552 return encode(input);
553 }
554 if constexpr (std::is_same_v<StringType, StringView>)
555 return input.toString();
556 else
557 return input;
558}

565void URL::remove(unsigned start, unsigned length)
566{
567 if (!length)
568 return;
569 ASSERT(start < m_string.length());
570 ASSERT(length <= m_string.length() - start);
571
572 auto stringAfterRemoval = WTFMove(m_string);
573 stringAfterRemoval.remove(start, length);
574 parse(stringAfterRemoval);
575}

577void URL::setUser(StringView newUser)
578{
579 if (!m_isValid)
580 return;
581
582 unsigned end = m_userEnd;
583 if (!newUser.isEmpty()) {
584 bool slashSlashNeeded = m_userStart == m_schemeEnd + 1U;
585 bool needSeparator = end == m_hostEnd || (end == m_passwordEnd && m_string[end] != '@');
586 parse(makeString(
587 StringView(m_string).left(m_userStart),
588 slashSlashNeeded ? "//" : "",
589 percentEncodeCharacters(newUser, URLParser::isInUserInfoEncodeSet),
590 needSeparator ? "@" : "",
591 StringView(m_string).substring(end)
592 ));
593 } else {
594 // Remove '@' if we now have neither user nor password.
595 if (m_userEnd == m_passwordEnd && end != m_hostEnd && m_string[end] == '@')
596 end += 1;
597 remove(m_userStart, end - m_userStart);
598 }
599}

601void URL::setPassword(StringView newPassword)
602{
603 if (!m_isValid)
604 return;
605
606 if (!newPassword.isEmpty()) {
607 bool needLeadingSlashes = m_userEnd == m_schemeEnd + 1U;
608 parse(makeString(
609 StringView(m_string).left(m_userEnd),
610 needLeadingSlashes ? "//:" : ":",
611 percentEncodeCharacters(newPassword, URLParser::isInUserInfoEncodeSet),
612 '@',
613 StringView(m_string).substring(credentialsEnd())
614 ));
615 } else {
616 unsigned end = m_userStart == m_userEnd ? credentialsEnd() : m_passwordEnd;
617 remove(m_userEnd, end - m_userEnd);
618 }
619}

712bool protocolHostAndPortAreEqual(const URL& a, const URL& b)
713{
714 if (a.m_schemeEnd != b.m_schemeEnd)
715 return false;
716
717 unsigned hostStartA = a.hostStart();
718 unsigned hostLengthA = a.m_hostEnd - hostStartA;
719 unsigned hostStartB = b.hostStart();
720 unsigned hostLengthB = b.m_hostEnd - b.hostStart();
721 if (hostLengthA != hostLengthB)
722 return false;
723
724 // Check the scheme
725 for (unsigned i = 0; i < a.m_schemeEnd; ++i) {
726 if (a.string()[i] != b.string()[i])
727 return false;
728 }
729
730 // And the host
731 for (unsigned i = 0; i < hostLengthA; ++i) {
732 if (a.string()[hostStartA + i] != b.string()[hostStartB + i])
733 return false;
734 }
735
736 if (a.port() != b.port())
737 return false;
738
739 return true;
740}

742bool URL::isMatchingDomain(StringView domain) const
743{
744 // FIXME: Consider moving this to an appropriate place in WebCore's plug-in code; don't want people tempted to use this instead of SecurityOrigin.
745
746 if (isNull())
747 return false;
748
749 if (domain.isEmpty())
750 return true;
751
752 if (!protocolIsInHTTPFamily())
753 return false;
754
755 auto host = this->host();
756 if (!host.endsWith(domain))
757 return false;
758
759 return host.length() == domain.length() || host[host.length() - domain.length() - 1] == '.';
760}

776static bool protocolIsInternal(StringView string, const char* protocol)
777{
778 assertProtocolIsGood(protocol);
779 bool isLeading = true;
780 for (auto codeUnit : string.codeUnits()) {
781 if (isLeading) {
782 // Skip leading whitespace and control characters.
783 if (shouldTrimFromURL(codeUnit))
784 continue;
785 isLeading = false;
786 } else {
787 // Skip tabs and newlines even later in the protocol.
788 if (codeUnit == '\t' || codeUnit == '\r' || codeUnit == '\n')
789 continue;
790 }
791
792 char expectedCharacter = *protocol++;
793 if (!expectedCharacter)
794 return codeUnit == ':';
795 if (!isASCIIAlphaCaselessEqual(codeUnit, expectedCharacter))
796 return false;
797 }
798 return false;
799}

1001String mimeTypeFromDataURL(StringView dataURL)
1002{
1003 ASSERT(protocolIsInternal(dataURL, "data"));
1004
1005 // FIXME: What's the right behavior when the URL has a comma first, but a semicolon later?
1006 // Currently this code will break at the semicolon in that case; should add a test.
1007 auto index = dataURL.find(';', 5);
1008 if (index == notFound)
1009 index = dataURL.find(',', 5);
1010 if (index == notFound) {
1011 // FIXME: There was an old comment here that made it sound like this should be returning text/plain.
1012 // But we have been returning empty string here for some time, so not changing its behavior at this time.
1013 return emptyString();
1014 }
1015 if (index == 5)
1016 return "text/plain"_s;
1017 ASSERT(index >= 5);
1018 return dataURL.substring(5, index - 5).convertToASCIILowercase();
1019}

1077static bool isIPv4Address(StringView string)
1078{
1079 auto count = 0;
1080
1081 for (const auto octet : string.splitAllowingEmptyEntries('.')) {
1082 if (count >= 4)
1083 return false;
1084
1085 const auto length = octet.length();
1086 if (!length || length > 3)
1087 return false;
1088
1089 auto value = 0;
1090 for (auto i = 0u; i < length; ++i) {
1091 const auto digit = octet[i];
1092
1093 // Prohibit leading zeroes.
1094 if (digit > '9' || digit < (!i && length > 1 ? '1' : '0'))
1095 return false;
1096
1097 value = 10 * value + (digit - '0');
1098 }
1099
1100 if (value > 255)
1101 return false;
1102
1103 count++;
1104 }
1105
1106 return (count == 4);
1107}

1109static bool isIPv6Address(StringView string)
1110{
1111 enum SkipState { None, WillSkip, Skipping, Skipped, Final };
1112 auto skipState = None;
1113 auto count = 0;
1114
1115 for (const auto hextet : string.splitAllowingEmptyEntries(':')) {
1116 if (count >= 8 || skipState == Final)
1117 return false;
1118
1119 const auto length = hextet.length();
1120 if (!length) {
1121 // :: may be used anywhere to skip 1 to 8 hextets, but only once.
1122 if (skipState == Skipped)
1123 return false;
1124
1125 if (skipState == None)
1126 skipState = !count ? WillSkip : Skipping;
1127 else if (skipState == WillSkip)
1128 skipState = Skipping;
1129 else
1130 skipState = Final;
1131 continue;
1132 }
1133
1134 if (skipState == WillSkip)
1135 return false;
1136
1137 if (skipState == Skipping)
1138 skipState = Skipped;
1139
1140 if (length > 4) {
1141 // An IPv4 address may be used in place of the final two hextets.
1142 if ((skipState == None && count != 6) || (skipState == Skipped && count >= 6) || !isIPv4Address(hextet))
1143 return false;
1144
1145 skipState = Final;
1146 continue;
1147 }
1148
1149 for (const auto codeUnit : hextet.codeUnits()) {
1150 // IPv6 allows leading zeroes.
1151 if (!isASCIIHexDigit(codeUnit))
1152 return false;
1153 }
1154
1155 count++;
1156 }
1157
1158 return (count == 8 && skipState == None) || skipState == Skipped || skipState == Final;
1159}

1161bool URL::hostIsIPAddress(StringView host)
1162{
1163 return host.contains(':') ? isIPv6Address(host) : isIPv4Address(host);
1164}
