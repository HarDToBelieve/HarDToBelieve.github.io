34void getURLBytes(CFURLRef url, URLCharBuffer& result)
35{
36 CFIndex bytesLength = CFURLGetBytes(url, nullptr, 0);
37 result.resize(bytesLength);
38 CFIndex finalLength = CFURLGetBytes(url, reinterpret_cast<UInt8*>(result.data()), bytesLength);
39 ASSERT_UNUSED(finalLength, finalLength == bytesLength);
40}
41
42void getURLBytes(CFURLRef url, CString& result)
43{
44 CFIndex bytesLength = CFURLGetBytes(url, nullptr, 0);
45 char* bytes;
46 result = CString::newUninitialized(bytesLength, bytes);
47 CFIndex finalLength = CFURLGetBytes(url, reinterpret_cast<UInt8*>(bytes), bytesLength);
48 ASSERT_UNUSED(finalLength, finalLength == bytesLength);
49}
50
51bool isCFURLSameOrigin(CFURLRef cfURL, const URL& url)
52{
53 ASSERT(url.protocolIsInHTTPFamily());
54
55 if (url.hasCredentials())
56 return protocolHostAndPortAreEqual(url, URL { cfURL });
57
58 URLCharBuffer bytes;
59 getURLBytes(cfURL, bytes);
60 StringView cfURLString { reinterpret_cast<const LChar*>(bytes.data()), static_cast<unsigned>(bytes.size()) };
61
62 if (!url.hasPath())
63 return StringView { url.string() } == cfURLString;
64
65 auto urlWithoutPath = StringView { url.string() }.substring(0, url.pathStart() + 1);
66 return cfURLString.startsWith(urlWithoutPath);
67}
