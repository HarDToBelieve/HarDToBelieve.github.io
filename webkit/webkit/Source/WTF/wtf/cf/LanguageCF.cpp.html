46static String httpStyleLanguageCode(CFStringRef language, ShouldMinimizeLanguages shouldMinimizeLanguages)
47{
48 RetainPtr<CFStringRef> preferredLanguageCode;
49 // If we can minimize the language list to reduce fingerprinting, we can afford to be more lossless when canonicalizing the locale list.
50 if (shouldMinimizeLanguages == ShouldMinimizeLanguages::No || canMinimizeLanguages())
51 preferredLanguageCode = adoptCF(CFLocaleCreateCanonicalLanguageIdentifierFromString(kCFAllocatorDefault, language));
52 else {
53 SInt32 languageCode;
54 SInt32 regionCode;
55 SInt32 scriptCode;
56 CFStringEncoding stringEncoding;
57
58 // FIXME: This transformation is very wrong:
59 // 1. There is no reason why CFBundle localization names would be at all related to language names as used on the Web.
60 // 2. Script Manager codes cannot represent all languages that are now supported by the platform, so the conversion is lossy.
61 // 3. This should probably match what is sent by the network layer as Accept-Language, but currently, that's implemented separately.
62 CFBundleGetLocalizationInfoForLocalization(language, &languageCode, &regionCode, &scriptCode, &stringEncoding);
63 preferredLanguageCode = adoptCF(CFBundleCopyLocalizationForLocalizationInfo(languageCode, regionCode, scriptCode, stringEncoding));
64 }
65
66 if (!preferredLanguageCode)
67 preferredLanguageCode = language;
68 auto mutableLanguageCode = adoptCF(CFStringCreateMutableCopy(kCFAllocatorDefault, 0, preferredLanguageCode.get()));
69
70 // Turn a '_' into a '-' if it appears after a 2-letter language code
71 if (CFStringGetLength(mutableLanguageCode.get()) >= 3 && CFStringGetCharacterAtIndex(mutableLanguageCode.get(), 2) == '_')
72 CFStringReplace(mutableLanguageCode.get(), CFRangeMake(2, 1), CFSTR("-"));
73
74 return mutableLanguageCode.get();
75}

77void listenForLanguageChangeNotifications()
78{
79#if PLATFORM(MAC)
80 static dispatch_once_t onceToken;
81 dispatch_once(&onceToken, ^{
82 CFNotificationCenterAddObserver(CFNotificationCenterGetDistributedCenter(), nullptr, &languagePreferencesDidChange, CFSTR("AppleLanguagePreferencesChangedNotification"), nullptr, CFNotificationSuspensionBehaviorCoalesce);
83 });
84#endif
85}

87Vector<String> platformUserPreferredLanguages(ShouldMinimizeLanguages shouldMinimizeLanguages)
88{
89 auto platformLanguages = adoptCF(CFLocaleCopyPreferredLanguages());
90
91 if (shouldMinimizeLanguages == ShouldMinimizeLanguages::Yes)
92 platformLanguages = minimizedLanguagesFromLanguages(platformLanguages.get());
93
94 CFIndex platformLanguagesCount = CFArrayGetCount(platformLanguages.get());
95 if (!platformLanguagesCount)
96 return { "en"_s };
97
98 Vector<String> languages;
99 for (CFIndex i = 0; i < platformLanguagesCount; i++) {
100 auto platformLanguage = static_cast<CFStringRef>(CFArrayGetValueAtIndex(platformLanguages.get(), i));
101 languages.append(httpStyleLanguageCode(platformLanguage, shouldMinimizeLanguages));
102 }
103 return languages;
104}
