36static RetainPtr<CFRunLoopTimerRef> createTimer(Seconds interval, bool repeat, void(*timerFired)(CFRunLoopTimerRef, void*), void* info)
37{
38 CFRunLoopTimerContext context = { 0, info, 0, 0, 0 };
39 Seconds repeatInterval = repeat ? interval : 0_s;
40 return adoptCF(CFRunLoopTimerCreate(kCFAllocatorDefault, CFAbsoluteTimeGetCurrent() + interval.seconds(), repeatInterval.seconds(), 0, 0, timerFired, &context));
41}
42
43void RunLoop::performWork(void* context)
44{
45 AutodrainedPool pool;
46 static_cast<RunLoop*>(context)->performWork();
47}
48
49RunLoop::RunLoop()
50 : m_runLoop(CFRunLoopGetCurrent())
51{
52 CFRunLoopSourceContext context = { 0, this, 0, 0, 0, 0, 0, 0, 0, performWork };
53 m_runLoopSource = adoptCF(CFRunLoopSourceCreate(kCFAllocatorDefault, 0, &context));
54 CFRunLoopAddSource(m_runLoop.get(), m_runLoopSource.get(), kCFRunLoopCommonModes);
55}
56
57RunLoop::~RunLoop()
58{
59 CFRunLoopSourceInvalidate(m_runLoopSource.get());
60}
61
62void RunLoop::wakeUp()
63{
64 CFRunLoopSourceSignal(m_runLoopSource.get());
65 CFRunLoopWakeUp(m_runLoop.get());
66}

68RunLoop::CycleResult RunLoop::cycle(RunLoopMode mode)
69{
70 CFTimeInterval timeInterval = 0.05;
71 CFRunLoopRunInMode(mode, timeInterval, true);
72 return CycleResult::Continue;
73}
74
75void RunLoop::run()
76{
77 AutodrainedPool pool;
78 CFRunLoopRun();
79}
80
81void RunLoop::stop()
82{
83 ASSERT(m_runLoop == CFRunLoopGetCurrent());
84 CFRunLoopStop(m_runLoop.get());
85}
86
87void RunLoop::dispatch(const SchedulePairHashSet& schedulePairs, Function<void()>&& function)
88{
89 auto timer = createTimer(0_s, false, [] (CFRunLoopTimerRef timer, void* context) {
90 AutodrainedPool pool;
91
92 CFRunLoopTimerInvalidate(timer);
93
94 auto function = adopt(static_cast<Function<void()>::Impl*>(context));
95 function();
96 }, function.leak());
97
98 for (auto& schedulePair : schedulePairs)
99 CFRunLoopAddTimer(schedulePair->runLoop(), timer.get(), schedulePair->mode());
100}

104RunLoop::TimerBase::TimerBase(RunLoop& runLoop)
105 : m_runLoop(runLoop)
106{
107}
108
109RunLoop::TimerBase::~TimerBase()
110{
111 stop();
112}
113
114void RunLoop::TimerBase::start(Seconds interval, bool repeat)
115{
116 if (m_timer)
117 stop();
118
119 m_timer = createTimer(interval, repeat, [] (CFRunLoopTimerRef cfTimer, void* context) {
120 AutodrainedPool pool;
121
122 auto timer = static_cast<TimerBase*>(context);
123 if (!CFRunLoopTimerDoesRepeat(cfTimer))
124 CFRunLoopTimerInvalidate(cfTimer);
125
126 timer->fired();
127 }, this);
128 CFRunLoopAddTimer(m_runLoop->m_runLoop.get(), m_timer.get(), kCFRunLoopCommonModes);
129}
130
131void RunLoop::TimerBase::stop()
132{
133 if (!m_timer)
134 return;
135
136 CFRunLoopTimerInvalidate(m_timer.get());
137 m_timer = nullptr;
138}
139
140bool RunLoop::TimerBase::isActive() const
141{
142 return m_timer && CFRunLoopTimerIsValid(m_timer.get());
143}
144
145Seconds RunLoop::TimerBase::secondsUntilFire() const
146{
147 if (isActive())
148 return std::max<Seconds>(Seconds { CFRunLoopTimerGetNextFireDate(m_timer.get()) - CFAbsoluteTimeGetCurrent() }, 0_s);
149 return 0_s;
150}
