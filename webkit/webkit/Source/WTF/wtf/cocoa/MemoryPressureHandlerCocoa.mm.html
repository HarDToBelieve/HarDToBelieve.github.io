44void MemoryPressureHandler::platformReleaseMemory(Critical critical)
45{
46 if (critical == Critical::Yes && (!isUnderMemoryPressure() || m_isSimulatingMemoryPressure)) {
47 // libcache listens to OS memory notifications, but for process suspension
48 // or memory pressure simulation, we need to prod it manually:
49 cache_simulate_memory_warning_event(DISPATCH_MEMORYPRESSURE_CRITICAL);
50 }
51}
52
53static dispatch_source_t memoryPressureEventSource = nullptr;
54static dispatch_source_t timerEventSource = nullptr;
55static int notifyTokens[3];
56
57// Disable memory event reception for a minimum of s_minimumHoldOffTime
58// seconds after receiving an event. Don't let events fire any sooner than
59// s_holdOffMultiplier times the last cleanup processing time. Effectively
60// this is 1 / s_holdOffMultiplier percent of the time.
61// These value seems reasonable and testing verifies that it throttles frequent
62// low memory events, greatly reducing CPU usage.
63static const Seconds s_minimumHoldOffTime { 5_s };
64#if !PLATFORM(IOS_FAMILY)
65static constexpr unsigned s_holdOffMultiplier = 20;
66#endif
67
68void MemoryPressureHandler::install()
69{
70 if (m_installed || timerEventSource)
71 return;
72
73 dispatch_async(m_dispatchQueue.get(), ^{
74 auto memoryStatusFlags = DISPATCH_MEMORYPRESSURE_NORMAL | DISPATCH_MEMORYPRESSURE_WARN | DISPATCH_MEMORYPRESSURE_CRITICAL | DISPATCH_MEMORYPRESSURE_PROC_LIMIT_WARN | DISPATCH_MEMORYPRESSURE_PROC_LIMIT_CRITICAL;
75 memoryPressureEventSource = dispatch_source_create(DISPATCH_SOURCE_TYPE_MEMORYPRESSURE, 0, memoryStatusFlags, m_dispatchQueue.get());
76
77 dispatch_source_set_event_handler(memoryPressureEventSource, ^{
78 auto status = dispatch_source_get_data(memoryPressureEventSource);
79 switch (status) {
80 // VM pressure events.
81 case DISPATCH_MEMORYPRESSURE_NORMAL:
82 setUnderMemoryPressure(false);
83 break;
84 case DISPATCH_MEMORYPRESSURE_WARN:
85 setUnderMemoryPressure(false);
86 respondToMemoryPressure(Critical::No);
87 break;
88 case DISPATCH_MEMORYPRESSURE_CRITICAL:
89 setUnderMemoryPressure(true);
90 respondToMemoryPressure(Critical::Yes);
91 break;
92 // Process memory limit events.
93 case DISPATCH_MEMORYPRESSURE_PROC_LIMIT_WARN:
94 respondToMemoryPressure(Critical::No);
95 break;
96 case DISPATCH_MEMORYPRESSURE_PROC_LIMIT_CRITICAL:
97 respondToMemoryPressure(Critical::Yes);
98 break;
99 }
100 if (m_shouldLogMemoryMemoryPressureEvents)
101 RELEASE_LOG(MemoryPressure, "Received memory pressure event %lu vm pressure %d", status, isUnderMemoryPressure());
102 });
103 dispatch_resume(memoryPressureEventSource);
104 });
105
106 // Allow simulation of memory pressure with "notifyutil -p org.WebKit.lowMemory"
107 notify_register_dispatch("org.WebKit.lowMemory", &notifyTokens[0], m_dispatchQueue.get(), ^(int) {
108#if ENABLE(FMW_FOOTPRINT_COMPARISON)
109 auto footprintBefore = pagesPerVMTag();
110#endif
111 beginSimulatedMemoryPressure();
112
113 WTF::releaseFastMallocFreeMemory();
114 malloc_zone_pressure_relief(nullptr, 0);
115
116#if ENABLE(FMW_FOOTPRINT_COMPARISON)
117 auto footprintAfter = pagesPerVMTag();
118 logFootprintComparison(footprintBefore, footprintAfter);
119#endif
120
121 dispatch_after(dispatch_time(DISPATCH_TIME_NOW, 2 * NSEC_PER_SEC), m_dispatchQueue.get(), ^{
122 endSimulatedMemoryPressure();
123 });
124 });
125
126 notify_register_dispatch("org.WebKit.lowMemory.begin", &notifyTokens[1], m_dispatchQueue.get(), ^(int) {
127 beginSimulatedMemoryPressure();
128 });
129 notify_register_dispatch("org.WebKit.lowMemory.end", &notifyTokens[2], m_dispatchQueue.get(), ^(int) {
130 endSimulatedMemoryPressure();
131 });
132
133 m_installed = true;
134}

136void MemoryPressureHandler::uninstall()
137{
138 if (!m_installed)
139 return;
140
141 dispatch_async(m_dispatchQueue.get(), ^{
142 if (memoryPressureEventSource) {
143 dispatch_source_cancel(memoryPressureEventSource);
144 memoryPressureEventSource = nullptr;
145 }
146
147 if (timerEventSource) {
148 dispatch_source_cancel(timerEventSource);
149 timerEventSource = nullptr;
150 }
151 });
152
153 m_installed = false;
154
155 for (auto& token : notifyTokens)
156 notify_cancel(token);
157}

159void MemoryPressureHandler::holdOff(Seconds seconds)
160{
161 dispatch_async(m_dispatchQueue.get(), ^{
162 timerEventSource = dispatch_source_create(DISPATCH_SOURCE_TYPE_TIMER, 0, 0, m_dispatchQueue.get());
163 if (timerEventSource) {
164 dispatch_set_context(timerEventSource, this);
165 // FIXME: The final argument `s_minimumHoldOffTime.seconds()` seems wrong.
166 // https://bugs.webkit.org/show_bug.cgi?id=183277
167 dispatch_source_set_timer(timerEventSource, dispatch_time(DISPATCH_TIME_NOW, seconds.seconds() * NSEC_PER_SEC), DISPATCH_TIME_FOREVER, s_minimumHoldOffTime.seconds());
168 dispatch_source_set_event_handler(timerEventSource, ^{
169 if (timerEventSource) {
170 dispatch_source_cancel(timerEventSource);
171 timerEventSource = nullptr;
172 }
173 MemoryPressureHandler::singleton().install();
174 });
175 dispatch_resume(timerEventSource);
176 }
177 });
178}
179
180void MemoryPressureHandler::respondToMemoryPressure(Critical critical, Synchronous synchronous)
181{
182#if !PLATFORM(IOS_FAMILY)
183 uninstall();
184 MonotonicTime startTime = MonotonicTime::now();
185#endif
186
187 releaseMemory(critical, synchronous);
188
189#if !PLATFORM(IOS_FAMILY)
190 Seconds holdOffTime = (MonotonicTime::now() - startTime) * s_holdOffMultiplier;
191 holdOff(std::max(holdOffTime, s_minimumHoldOffTime));
192#endif
193}

195std::optional<MemoryPressureHandler::ReliefLogger::MemoryUsage> MemoryPressureHandler::ReliefLogger::platformMemoryUsage()
196{
197 task_vm_info_data_t vmInfo;
198 mach_msg_type_number_t count = TASK_VM_INFO_COUNT;
199 kern_return_t err = task_info(mach_task_self(), TASK_VM_INFO, (task_info_t) &vmInfo, &count);
200 if (err != KERN_SUCCESS)
201 return std::nullopt;
202
203 return MemoryUsage {static_cast<size_t>(vmInfo.internal), static_cast<size_t>(vmInfo.phys_footprint)};
204}
