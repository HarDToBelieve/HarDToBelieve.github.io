37void FileSystem::setMetadataURL(const String& path, const String& metadataURLString, const String& referrer)
38{
39 String urlString;
40 if (NSURL *url = URLWithUserTypedString(metadataURLString, nil))
41 urlString = WTF::userVisibleString(WTF::URLByRemovingUserInfo(url));
42 else
43 urlString = metadataURLString;
44
45 // Call Metadata API on a background queue because it can take some time.
46 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), [path = path.isolatedCopy(), urlString = urlString.isolatedCopy(), referrer = referrer.isolatedCopy()] {
47 auto item = adoptCF(MDItemCreate(kCFAllocatorDefault, path.createCFString().get()));
48 if (!item)
49 return;
50
51 auto whereFromAttribute = adoptNS([[NSMutableArray alloc] initWithObjects:urlString, nil]);
52 if (!referrer.isNull())
53 [whereFromAttribute addObject:referrer];
54
55 MDItemSetAttribute(item.get(), kMDItemWhereFroms, (__bridge CFArrayRef)whereFromAttribute.get());
56 MDItemSetAttribute(item.get(), kMDItemDownloadedDate, (__bridge CFArrayRef)@[ [NSDate date] ]);
57 });
58}
59
60bool FileSystem::canExcludeFromBackup()
61{
62 return true;
63}
64
65bool FileSystem::excludeFromBackup(const String& path)
66{
67 // It is critical to pass FALSE for excludeByPath because excluding by path requires root privileges.
68 CSBackupSetItemExcluded(FileSystem::pathAsURL(path).get(), TRUE, FALSE);
69 return true;
70}
