35void Coder<AtomString>::encode(Encoder& encoder, const AtomString& atomString)
36{
37 encoder << atomString.string();
38}
39
40std::optional<AtomString> Coder<AtomString>::decode(Decoder& decoder)
41{
42 std::optional<String> string;
43 decoder >> string;
44 if (!string)
45 return std::nullopt;
46
47 return {{ WTFMove(*string) }};
48}

50void Coder<CString>::encode(Encoder& encoder, const CString& string)
51{
52 // Special case the null string.
53 if (string.isNull()) {
54 encoder << std::numeric_limits<uint32_t>::max();
55 return;
56 }
57
58 uint32_t length = string.length();
59 encoder << length;
60 encoder.encodeFixedLengthData(string.dataAsUInt8Ptr(), length);
61}
62
63std::optional<CString> Coder<CString>::decode(Decoder& decoder)
64{
65 std::optional<uint32_t> length;
66 decoder >> length;
67 if (!length)
68 return std::nullopt;
69
70 if (length == std::numeric_limits<uint32_t>::max()) {
71 // This is the null string.
72 return CString();
73 }
74
75 // Before allocating the string, make sure that the decoder buffer is big enough.
76 if (!decoder.bufferIsLargeEnoughToContain<char>(*length))
77 return std::nullopt;
78
79 char* buffer;
80 CString string = CString::newUninitialized(*length, buffer);
81 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(buffer), *length))
82 return std::nullopt;
83
84 return string;
85}
86
87void Coder<String>::encode(Encoder& encoder, const String& string)
88{
89 // Special case the null string.
90 if (string.isNull()) {
91 encoder << std::numeric_limits<uint32_t>::max();
92 return;
93 }
94
95 uint32_t length = string.length();
96 bool is8Bit = string.is8Bit();
97
98 encoder << length << is8Bit;
99
100 if (is8Bit)
101 encoder.encodeFixedLengthData(string.characters8(), length);
102 else
103 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(string.characters16()), length * sizeof(UChar));
104}
105
106template <typename CharacterType>
107static inline std::optional<String> decodeStringText(Decoder& decoder, uint32_t length)
108{
109 // Before allocating the string, make sure that the decoder buffer is big enough.
110 if (!decoder.bufferIsLargeEnoughToContain<CharacterType>(length))
111 return std::nullopt;
112
113 CharacterType* buffer;
114 String string = String::createUninitialized(length, buffer);
115 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(buffer), length * sizeof(CharacterType)))
116 return std::nullopt;
117
118 return string;
119}
120
121std::optional<String> Coder<String>::decode(Decoder& decoder)
122{
123 std::optional<uint32_t> length;
124 decoder >> length;
125 if (!length)
126 return std::nullopt;
127
128 if (*length == std::numeric_limits<uint32_t>::max()) {
129 // This is the null string.
130 return String();
131 }
132
133 std::optional<bool> is8Bit;
134 decoder >> is8Bit;
135 if (!is8Bit)
136 return std::nullopt;
137
138 if (*is8Bit)
139 return decodeStringText<LChar>(decoder, *length);
140 return decodeStringText<UChar>(decoder, *length);
141}

143void Coder<SHA1::Digest>::encode(Encoder& encoder, const SHA1::Digest& digest)
144{
145 encoder.encodeFixedLengthData(digest.data(), sizeof(digest));
146}
147
148std::optional<SHA1::Digest> Coder<SHA1::Digest>::decode(Decoder& decoder)
149{
150 SHA1::Digest tmp;
151 if (!decoder.decodeFixedLengthData(tmp.data(), sizeof(tmp)))
152 return std::nullopt;
153 return tmp;
154}
