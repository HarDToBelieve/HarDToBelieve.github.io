34Decoder::Decoder(const uint8_t* buffer, size_t bufferSize)
35 : m_buffer(buffer)
36 , m_bufferPosition(buffer)
37 , m_bufferEnd(buffer + bufferSize)
38{
39}
40
41Decoder::~Decoder()
42{
43}
44
45bool Decoder::bufferIsLargeEnoughToContain(size_t size) const
46{
47 return size <= static_cast<size_t>(m_bufferEnd - m_bufferPosition);
48}
49
50const uint8_t* Decoder::bufferPointerForDirectRead(size_t size)
51{
52 if (!bufferIsLargeEnoughToContain(size))
53 return nullptr;
54
55 auto data = m_bufferPosition;
56 m_bufferPosition += size;
57
58 Encoder::updateChecksumForData(m_sha1, data, size);
59 return data;
60}
61
62bool Decoder::decodeFixedLengthData(uint8_t* data, size_t size)
63{
64 auto buffer = bufferPointerForDirectRead(size);
65 if (!buffer)
66 return false;
67 memcpy(data, buffer, size);
68 return true;
69}
70
71bool Decoder::rewind(size_t size)
72{
73 if (m_bufferPosition - size >= m_buffer) {
74 m_bufferPosition -= size;
75 return true;
76 }
77 return false;
78}

80template<typename T>
81Decoder& Decoder::decodeNumber(std::optional<T>& optional)
82{
83 if (!bufferIsLargeEnoughToContain(sizeof(T)))
84 return *this;
85
86 T value;
87 memcpy(&value, m_bufferPosition, sizeof(T));
88 m_bufferPosition += sizeof(T);
89
90 Encoder::updateChecksumForNumber(m_sha1, value);
91 optional = value;
92 return *this;
93}

95Decoder& Decoder::operator>>(std::optional<bool>& result)
96{
97 return decodeNumber(result);

145bool Decoder::verifyChecksum()
146{
147 SHA1::Digest computedHash;
148 m_sha1.computeHash(computedHash);
149
150 SHA1::Digest savedHash;
151 if (!decodeFixedLengthData(savedHash.data(), sizeof(savedHash)))
152 return false;
153
154 return computedHash == savedHash;
155}
