51PlatformFileHandle openFile(const String& path, FileOpenMode mode, FileAccessPermission permission, bool failIfFileExists)
52{
53 CString fsRep = fileSystemRepresentation(path);
54
55 if (fsRep.isNull())
56 return invalidPlatformFileHandle;
57
58 int platformFlag = 0;
59 switch (mode) {
60 case FileOpenMode::Read:
61 platformFlag |= O_RDONLY;
62 break;
63 case FileOpenMode::Write:
64 platformFlag |= (O_WRONLY | O_CREAT | O_TRUNC);
65 break;
66 case FileOpenMode::ReadWrite:
67 platformFlag |= (O_RDWR | O_CREAT);
68 break;
69#if OS(DARWIN)
70 case FileOpenMode::EventsOnly:
71 platformFlag |= O_EVTONLY;
72 break;
73#endif
74 }
75
76 if (failIfFileExists)
77 platformFlag |= (O_CREAT | O_EXCL);
78
79 int permissionFlag = 0;
80 if (permission == FileAccessPermission::User)
81 permissionFlag |= (S_IRUSR | S_IWUSR);
82 else if (permission == FileAccessPermission::All)
83 permissionFlag |= (S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH);
84
85 return open(fsRep.data(), platformFlag, permissionFlag);
86}

88void closeFile(PlatformFileHandle& handle)
89{
90 if (isHandleValid(handle)) {
91 close(handle);
92 handle = invalidPlatformFileHandle;
93 }
94}

96long long seekFile(PlatformFileHandle handle, long long offset, FileSeekOrigin origin)
97{
98 int whence = SEEK_SET;
99 switch (origin) {
100 case FileSeekOrigin::Beginning:
101 whence = SEEK_SET;
102 break;
103 case FileSeekOrigin::Current:
104 whence = SEEK_CUR;
105 break;
106 case FileSeekOrigin::End:
107 whence = SEEK_END;
108 break;
109 default:
110 ASSERT_NOT_REACHED();
111 }
112 return static_cast<long long>(lseek(handle, offset, whence));
113}

115bool truncateFile(PlatformFileHandle handle, long long offset)
116{
117 // ftruncate returns 0 to indicate the success.
118 return !ftruncate(handle, offset);
119}

121int writeToFile(PlatformFileHandle handle, const void* data, int length)
122{
123 do {
124 int bytesWritten = write(handle, data, static_cast<size_t>(length));
125 if (bytesWritten >= 0)
126 return bytesWritten;
127 } while (errno == EINTR);
128 return -1;
129}
130
131int readFromFile(PlatformFileHandle handle, void* data, int length)
132{
133 do {
134 int bytesRead = read(handle, data, static_cast<size_t>(length));
135 if (bytesRead >= 0)
136 return bytesRead;
137 } while (errno == EINTR);
138 return -1;
139}

141#if USE(FILE_LOCK)
142bool lockFile(PlatformFileHandle handle, OptionSet<FileLockMode> lockMode)
143{
144 COMPILE_ASSERT(LOCK_SH == WTF::enumToUnderlyingType(FileLockMode::Shared), LockSharedEncodingIsAsExpected);
145 COMPILE_ASSERT(LOCK_EX == WTF::enumToUnderlyingType(FileLockMode::Exclusive), LockExclusiveEncodingIsAsExpected);
146 COMPILE_ASSERT(LOCK_NB == WTF::enumToUnderlyingType(FileLockMode::Nonblocking), LockNonblockingEncodingIsAsExpected);
147 int result = flock(handle, lockMode.toRaw());
148 return (result != -1);
149}
150
151bool unlockFile(PlatformFileHandle handle)
152{
153 int result = flock(handle, LOCK_UN);
154 return (result != -1);
155}
156#endif

158std::optional<uint64_t> fileSize(PlatformFileHandle handle)
159{
160 struct stat fileInfo;
161 if (fstat(handle, &fileInfo))
162 return std::nullopt;
163
164 return fileInfo.st_size;
165}

167std::optional<WallTime> fileCreationTime(const String& path)
168{
169#if OS(DARWIN) || OS(OPENBSD) || OS(NETBSD) || OS(FREEBSD)
170 CString fsRep = fileSystemRepresentation(path);
171
172 if (!fsRep.data() || fsRep.data()[0] == '\0')
173 return std::nullopt;
174
175 struct stat fileInfo;
176
177 if (stat(fsRep.data(), &fileInfo))
178 return std::nullopt;
179
180 return WallTime::fromRawSeconds(fileInfo.st_birthtime);
181#else
182 UNUSED_PARAM(path);
183 return std::nullopt;
184#endif
185}

202#if !PLATFORM(COCOA)
203String openTemporaryFile(const String& prefix, PlatformFileHandle& handle, const String& suffix)
204{
205 // FIXME: Suffix is not supported, but OK for now since the code using it is macOS-port-only.
206 ASSERT_UNUSED(suffix, suffix.isEmpty());
207
208 char buffer[PATH_MAX];
209 const char* tmpDir = getenv("TMPDIR");
210
211 if (!tmpDir)
212 tmpDir = "/tmp";
213
214 if (snprintf(buffer, PATH_MAX, "%s/%sXXXXXX", tmpDir, prefix.utf8().data()) >= PATH_MAX)
215 goto end;
216
217 handle = mkstemp(buffer);
218 if (handle < 0)
219 goto end;
220
221 return String::fromUTF8(buffer);
222
223end:
224 handle = invalidPlatformFileHandle;
225 return String();
226}
227#endif // !PLATFORM(COCOA)

229std::optional<int32_t> getFileDeviceId(const CString& fsFile)
230{
231 struct stat fileStat;
232 if (stat(fsFile.data(), &fileStat) == -1)
233 return std::nullopt;
234
235 return fileStat.st_dev;
236}

245bool deleteFile(const String& path)
246{
247 // unlink(...) returns 0 on successful deletion of the path and non-zero in any other case (including invalid permissions or non-existent file)
248 bool unlinked = !unlink(fileSystemRepresentation(path).data());
249 if (!unlinked && errno != ENOENT)
250 LOG_ERROR("File failed to delete. Error message: %s", strerror(errno));
251
252 return unlinked;
253}

255bool makeAllDirectories(const String& path)
256{
257 auto fullPath = fileSystemRepresentation(path);
258 if (!access(fullPath.data(), F_OK))
259 return true;
260
261 char* p = fullPath.mutableData() + 1;
262 int length = fullPath.length();
263 if (p[length - 1] == '/')
264 p[length - 1] = '\0';
265 for (; *p; ++p) {
266 if (*p == '/') {
267 *p = '\0';
268 if (access(fullPath.data(), F_OK)) {
269 if (mkdir(fullPath.data(), S_IRWXU))
270 return false;
271 }
272 *p = '/';
273 }
274 }
275 if (access(fullPath.data(), F_OK)) {
276 if (mkdir(fullPath.data(), S_IRWXU))
277 return false;
278 }
279
280 return true;
281}

283String pathByAppendingComponent(const String& path, const String& component)
284{
285 if (path.endsWith('/'))
286 return path + component;
287 return path + "/" + component;
288}
289
290String pathByAppendingComponents(StringView path, const Vector<StringView>& components)
291{
292 StringBuilder builder;
293 builder.append(path);
294 bool isFirstComponent = true;
295 for (auto& component : components) {
296 if (isFirstComponent) {
297 isFirstComponent = false;
298 if (path.endsWith('/')) {
299 builder.append(component);
300 continue;
301 }
302 }
303 builder.append('/', component);
304 }
305 return builder.toString();
306}
