117void Thread::signalHandlerSuspendResume(int, siginfo_t*, void* ucontext)
118{
119 // Touching a global variable atomic types from signal handlers is allowed.
120 Thread* thread = targetThread.load();
121
122 if (thread->m_suspendCount) {
123 // This is signal handler invocation that is intended to be used to resume sigsuspend.
124 // So this handler invocation itself should not process.
125 //
126 // When signal comes, first, the system calls signal handler. And later, sigsuspend will be resumed. Signal handler invocation always precedes.
127 // So, the problem never happens that suspended.store(true, ...) will be executed before the handler is called.
128 // http://pubs.opengroup.org/onlinepubs/009695399/functions/sigsuspend.html
129 return;
130 }
131
132 void* approximateStackPointer = currentStackPointer();
133 if (!thread->m_stack.contains(approximateStackPointer)) {
134 // This happens if we use an alternative signal stack.
135 // 1. A user-defined signal handler is invoked with an alternative signal stack.
136 // 2. In the middle of the execution of the handler, we attempt to suspend the target thread.
137 // 3. A nested signal handler is executed.
138 // 4. The stack pointer saved in the machine context will be pointing to the alternative signal stack.
139 // In this case, we back off the suspension and retry a bit later.
140 thread->m_platformRegisters = nullptr;
141 globalSemaphoreForSuspendResume->post();
142 return;
143 }
144
145#if HAVE(MACHINE_CONTEXT)
146 ucontext_t* userContext = static_cast<ucontext_t*>(ucontext);
147 thread->m_platformRegisters = &registersFromUContext(userContext);
148#else
149 UNUSED_PARAM(ucontext);
150 PlatformRegisters platformRegisters { approximateStackPointer };
151 thread->m_platformRegisters = &platformRegisters;
152#endif
153
154 // Allow suspend caller to see that this thread is suspended.
155 // sem_post is async-signal-safe function. It means that we can call this from a signal handler.
156 // http://pubs.opengroup.org/onlinepubs/009695399/functions/xsh_chap02_04.html#tag_02_04_03
157 //
158 // And sem_post emits memory barrier that ensures that PlatformRegisters are correctly saved.
159 // http://pubs.opengroup.org/onlinepubs/9699919799/basedefs/V1_chap04.html#tag_04_11
160 globalSemaphoreForSuspendResume->post();
161
162 // Reaching here, sigThreadSuspendResume is blocked in this handler (this is configured by sigaction's sa_mask).
163 // So before calling sigsuspend, sigThreadSuspendResume to this thread is deferred. This ensures that the handler is not executed recursively.
164 sigset_t blockedSignalSet;
165 sigfillset(&blockedSignalSet);
166 sigdelset(&blockedSignalSet, g_wtfConfig.sigThreadSuspendResume);
167 sigsuspend(&blockedSignalSet);
168
169 thread->m_platformRegisters = nullptr;
170
171 // Allow resume caller to see that this thread is resumed.
172 globalSemaphoreForSuspendResume->post();
173}

177void Thread::initializePlatformThreading()
178{
179 if (!g_wtfConfig.isUserSpecifiedThreadSuspendResumeSignalConfigured) {
180 g_wtfConfig.sigThreadSuspendResume = SIGUSR1;
181 if (const char* string = getenv("JSC_SIGNAL_FOR_GC")) {
182 int32_t value = 0;
183 if (sscanf(string, "%d", &value) == 1)
184 g_wtfConfig.sigThreadSuspendResume = value;
185 }
186 }
187 g_wtfConfig.isThreadSuspendResumeSignalConfigured = true;
188
189#if !OS(DARWIN)
190 globalSemaphoreForSuspendResume.construct(0);
191
192 // Signal handlers are process global configuration.
193 // Intentionally block sigThreadSuspendResume in the handler.
194 // sigThreadSuspendResume will be allowed in the handler by sigsuspend.
195 auto attemptToSetSignal = [](int signal) -> bool {
196 struct sigaction action;
197 sigemptyset(&action.sa_mask);
198 sigaddset(&action.sa_mask, signal);
199
200 action.sa_sigaction = &signalHandlerSuspendResume;
201 action.sa_flags = SA_RESTART | SA_SIGINFO;
202
203 // Theoretically, this can have race conditions but currently, there is no way to deal with it,
204 // plus, we do not expect that this initialization is executed concurrently with the other
205 // initialization which also installs specific signals. If this is the problem, applications should
206 // change how to initialize things.
207 struct sigaction oldAction;
208 if (sigaction(signal, nullptr, &oldAction))
209 return false;
210 // It has signal already.
211 if (oldAction.sa_handler != SIG_DFL || bitwise_cast<void*>(oldAction.sa_sigaction) != bitwise_cast<void*>(SIG_DFL))
212 WTFLogAlways("Overriding existing handler for signal %d. Set JSC_SIGNAL_FOR_GC if you want WebKit to use a different signal", signal);
213 return !sigaction(signal, &action, 0);
214 };
215
216 bool signalIsInstalled = attemptToSetSignal(g_wtfConfig.sigThreadSuspendResume);
217 RELEASE_ASSERT(signalIsInstalled);
218#endif
219}

263bool Thread::establishHandle(NewThreadContext* context, std::optional<size_t> stackSize, QOS qos)
264{
265 pthread_t threadHandle;
266 pthread_attr_t attr;
267 pthread_attr_init(&attr);
268#if HAVE(QOS_CLASSES)
269 pthread_attr_set_qos_class_np(&attr, dispatchQOSClass(qos), 0);
270#else
271 UNUSED_PARAM(qos);
272#endif
273 if (stackSize)
274 pthread_attr_setstacksize(&attr, stackSize.value());
275 int error = pthread_create(&threadHandle, &attr, wtfThreadEntryPoint, context);
276 pthread_attr_destroy(&attr);
277 if (error) {
278 LOG_ERROR("Failed to create pthread at entry point %p with context %p", wtfThreadEntryPoint, context);
279 return false;
280 }
281 establishPlatformSpecificHandle(threadHandle);
282 return true;
283}

314int Thread::waitForCompletion()
315{
316 pthread_t handle;
317 {
318 Locker locker { m_mutex };
319 handle = m_handle;
320 }
321
322 int joinResult = pthread_join(handle, 0);
323
324 if (joinResult == EDEADLK)
325 LOG_ERROR("Thread %p was found to be deadlocked trying to quit", this);
326 else if (joinResult)
327 LOG_ERROR("Thread %p was unable to be joined.\n", this);
328
329 Locker locker { m_mutex };
330 ASSERT(joinableState() == Joinable);
331
332 // If the thread has already exited, then do nothing. If the thread hasn't exited yet, then just signal that we've already joined on it.
333 // In both cases, Thread::destructTLS() will take care of destroying Thread.
334 if (!hasExited())
335 didJoin();
336
337 return joinResult;
338}

372auto Thread::suspend() -> Expected<void, PlatformSuspendError>
373{
374 RELEASE_ASSERT_WITH_MESSAGE(this != &Thread::current(), "We do not support suspending the current thread itself.");
375 // During suspend, suspend or resume should not be executed from the other threads.
376 // We use global lock instead of per thread lock.
377 // Consider the following case, there are threads A and B.
378 // And A attempt to suspend B and B attempt to suspend A.
379 // A and B send signals. And later, signals are delivered to A and B.
380 // In that case, both will be suspended.
381 //
382 // And it is important to use a global lock to suspend and resume. Let's consider using per-thread lock.
383 // Your issuing thread (A) attempts to suspend the target thread (B). Then, you will suspend the thread (C) additionally.
384 // This case frequently happens if you stop threads to perform stack scanning. But thread (B) may hold the lock of thread (C).
385 // In that case, dead lock happens. Using global lock here avoids this dead lock.
386 Locker locker { globalSuspendLock };
387#if OS(DARWIN)
388 kern_return_t result = thread_suspend(m_platformThread);
389 if (result != KERN_SUCCESS)
390 return makeUnexpected(result);
391 return { };
392#else
393 if (!m_suspendCount) {
394 targetThread.store(this);
395
396 while (true) {
397 // We must use pthread_kill to avoid queue-overflow problem with real-time signals.
398 int result = pthread_kill(m_handle, g_wtfConfig.sigThreadSuspendResume);
399 if (result)
400 return makeUnexpected(result);
401 globalSemaphoreForSuspendResume->wait();
402 if (m_platformRegisters)
403 break;
404 // Because of an alternative signal stack, we failed to suspend this thread.
405 // Retry suspension again after yielding.
406 Thread::yield();
407 }
408 }
409 ++m_suspendCount;
410 return { };
411#endif
412}
413
414void Thread::resume()
415{
416 // During resume, suspend or resume should not be executed from the other threads.
417 Locker locker { globalSuspendLock };
418#if OS(DARWIN)
419 thread_resume(m_platformThread);
420#else
421 if (m_suspendCount == 1) {
422 // When allowing sigThreadSuspendResume interrupt in the signal handler by sigsuspend and SigThreadSuspendResume is actually issued,
423 // the signal handler itself will be called once again.
424 // There are several ways to distinguish the handler invocation for suspend and resume.
425 // 1. Use different signal numbers. And check the signal number in the handler.
426 // 2. Use some arguments to distinguish suspend and resume in the handler.
427 // 3. Use thread's flag.
428 // In this implementaiton, we take (3). m_suspendCount is used to distinguish it.
429 // Note that we must use pthread_kill to avoid queue-overflow problem with real-time signals.
430 targetThread.store(this);
431 if (pthread_kill(m_handle, g_wtfConfig.sigThreadSuspendResume) == ESRCH)
432 return;
433 globalSemaphoreForSuspendResume->wait();
434 }
435 --m_suspendCount;
436#endif
437}

473size_t Thread::getRegisters(PlatformRegisters& registers)
474{
475 Locker locker { globalSuspendLock };
476#if OS(DARWIN)
477 auto metadata = threadStateMetadata();
478 kern_return_t result = thread_get_state(m_platformThread, metadata.flavor, (thread_state_t)&registers, &metadata.userCount);
479 if (result != KERN_SUCCESS) {
480 WTFReportFatalError(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, "JavaScript garbage collection failed because thread_get_state returned an error (%d). This is probably the result of running inside Rosetta, which is not supported.", result);
481 CRASH();
482 }
483 return metadata.userCount * sizeof(uintptr_t);
484#else
485 ASSERT_WITH_MESSAGE(m_suspendCount, "We can get registers only if the thread is suspended.");
486 ASSERT(m_platformRegisters);
487 registers = *m_platformRegisters;
488 return sizeof(PlatformRegisters);
489#endif
490}

508Thread& Thread::initializeTLS(Ref<Thread>&& thread)
509{
510 // We leak the ref to keep the Thread alive while it is held in TLS. destructTLS will deref it later at thread destruction time.
511 auto& threadInTLS = thread.leakRef();
512#if !HAVE(FAST_TLS)
513 ASSERT(s_key != InvalidThreadSpecificKey);
514 threadSpecificSet(s_key, &threadInTLS);
515#else
516 _pthread_setspecific_direct(WTF_THREAD_DATA_KEY, &threadInTLS);
517 pthread_key_init_np(WTF_THREAD_DATA_KEY, &destructTLS);
518#endif
519 return threadInTLS;
520}

522void Thread::destructTLS(void* data)
523{
524 Thread* thread = static_cast<Thread*>(data);
525 ASSERT(thread);
526
527 if (thread->m_isDestroyedOnce) {
528 thread->didExit();
529 thread->deref();
530 return;
531 }
532
533 thread->m_isDestroyedOnce = true;
534 // Re-setting the value for key causes another destructTLS() call after all other thread-specific destructors were called.
535#if !HAVE(FAST_TLS)
536 ASSERT(s_key != InvalidThreadSpecificKey);
537 threadSpecificSet(s_key, thread);
538#else
539 _pthread_setspecific_direct(WTF_THREAD_DATA_KEY, thread);
540 pthread_key_init_np(WTF_THREAD_DATA_KEY, &destructTLS);
541#endif
542}

586bool ThreadCondition::timedWait(Mutex& mutex, WallTime absoluteTime)
587{
588 if (absoluteTime < WallTime::now())
589 return false;
590
591 if (absoluteTime > WallTime::fromRawSeconds(INT_MAX)) {
592 wait(mutex);
593 return true;
594 }
595
596 double rawSeconds = absoluteTime.secondsSinceEpoch().value();
597
598 int timeSeconds = static_cast<int>(rawSeconds);
599 int timeNanoseconds = static_cast<int>((rawSeconds - timeSeconds) * 1E9);
600
601 timespec targetTime;
602 targetTime.tv_sec = timeSeconds;
603 targetTime.tv_nsec = timeNanoseconds;
604
605 return pthread_cond_timedwait(&m_condition, &mutex.impl(), &targetTime) == 0;
606}
