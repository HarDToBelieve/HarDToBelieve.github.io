37DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(CStringBuffer);
38
39Ref<CStringBuffer> CStringBuffer::createUninitialized(size_t length)
40{
41 // The +1 is for the terminating null character.
42 size_t size = Checked<size_t>(sizeof(CStringBuffer)) + length + 1U;
43 auto* stringBuffer = static_cast<CStringBuffer*>(CStringBufferMalloc::malloc(size));
44 return adoptRef(*new (NotNull, stringBuffer) CStringBuffer(length));
45}
46
47CString::CString(const char* str)
48{
49 if (!str)
50 return;
51
52 init(str, strlen(str));
53}
54
55CString::CString(const char* str, size_t length)
56{
57 if (!str) {
58 ASSERT(!length);
59 return;
60 }
61
62 init(str, length);
63}
64
65void CString::init(const char* str, size_t length)
66{
67 ASSERT(str);
68
69 m_buffer = CStringBuffer::createUninitialized(length);
70 memcpy(m_buffer->mutableData(), str, length);
71 m_buffer->mutableData()[length] = '\0';
72}
73
74char* CString::mutableData()
75{
76 copyBufferIfNeeded();
77 if (!m_buffer)
78 return nullptr;
79 return m_buffer->mutableData();
80}
81
82CString CString::newUninitialized(size_t length, char*& characterBuffer)
83{
84 CString result;
85 result.m_buffer = CStringBuffer::createUninitialized(length);
86 char* bytes = result.m_buffer->mutableData();
87 bytes[length] = '\0';
88 characterBuffer = bytes;
89 return result;
90}
91
92void CString::copyBufferIfNeeded()
93{
94 if (!m_buffer || m_buffer->hasOneRef())
95 return;
96
97 RefPtr<CStringBuffer> buffer = WTFMove(m_buffer);
98 size_t length = buffer->length();
99 m_buffer = CStringBuffer::createUninitialized(length);
100 memcpy(m_buffer->mutableData(), buffer->data(), length + 1);
101}
102
103bool CString::isSafeToSendToAnotherThread() const
104{
105 return !m_buffer || m_buffer->hasOneRef();
106}
107
108void CString::grow(size_t newLength)
109{
110 ASSERT(newLength > length());
111
112 auto newBuffer = CStringBuffer::createUninitialized(newLength);
113 memcpy(newBuffer->mutableData(), m_buffer->data(), length() + 1);
114 m_buffer = WTFMove(newBuffer);
115}

117bool operator==(const CString& a, const CString& b)
118{
119 if (a.isNull() != b.isNull())
120 return false;
121 if (a.length() != b.length())
122 return false;
123 return !memcmp(a.data(), b.data(), a.length());
124}
125
126bool operator==(const CString& a, const char* b)
127{
128 if (a.isNull() != !b)
129 return false;
130 if (!b)
131 return true;
132 return !strcmp(a.data(), b);
133}
134
135unsigned CString::hash() const
136{
137 if (isNull())
138 return 0;
139 StringHasher hasher;
140 for (const char* ptr = data(); *ptr; ++ptr)
141 hasher.addCharacter(*ptr);
142 return hasher.hash();
143}
144
145bool operator<(const CString& a, const CString& b)
146{
147 if (a.isNull())
148 return !b.isNull();
149 if (b.isNull())
150 return false;
151 return strcmp(a.data(), b.data()) < 0;
152}
153
154bool CStringHash::equal(const CString& a, const CString& b)
155{
156 if (a.isHashTableDeletedValue())
157 return b.isHashTableDeletedValue();
158 if (b.isHashTableDeletedValue())
159 return false;
160 return a == b;
161}
