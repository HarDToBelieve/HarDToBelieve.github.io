35static constexpr unsigned maxCapacity = String::MaxLength;
36
37unsigned StringBuilder::expandedCapacity(unsigned capacity, unsigned requiredCapacity)
38{
39 static constexpr unsigned minimumCapacity = 16;
40 return std::max(requiredCapacity, std::max(minimumCapacity, std::min(capacity * 2, maxCapacity)));
41}
42
43void StringBuilder::didOverflow()
44{
45 if (m_shouldCrashOnOverflow)
46 CRASH();
47 m_length = std::numeric_limits<unsigned>::max();
48}
49
50void StringBuilder::reifyString() const
51{
52 RELEASE_ASSERT(!hasOverflowed());
53
54 // Check if the string already exists.
55 if (!m_string.isNull()) {
56 ASSERT(m_string.length() == m_length);
57 return;
58 }
59
60 // Check for empty.
61 if (!m_length) {
62 m_string = StringImpl::empty();
63 return;
64 }
65
66 // Must be valid in the buffer, take a substring (unless string fills the buffer).
67 ASSERT(m_length <= m_buffer->length());
68 if (m_length == m_buffer->length())
69 m_string = m_buffer.get();
70 else
71 m_string = StringImpl::createSubstringSharingImpl(*m_buffer, 0, m_length);
72}
73
74void StringBuilder::shrink(unsigned newLength)
75{
76 if (hasOverflowed())
77 return;
78
79 ASSERT(newLength <= m_length);
80 if (newLength >= m_length) {
81 if (newLength > m_length)
82 didOverflow();
83 return;
84 }
85
86 m_length = newLength;
87
88 if (m_buffer) {
89 m_string = { }; // Clear the string to remove the reference to m_buffer if any before checking the reference count of m_buffer.
90 if (m_buffer->hasOneRef()) {
91 // If the old buffer had only the one ref, we can just reduce the length and keep using the buffer.
92 return;
93 }
94 // Allocate a fresh buffer, with a copy of the characters we are keeping.
95 if (m_buffer->is8Bit())
96 allocateBuffer<LChar>(m_buffer->characters<LChar>(), newLength);
97 else
98 allocateBuffer<UChar>(m_buffer->characters<UChar>(), newLength);
99 return;
100 }
101
102 // Since the old length was not 0 and m_buffer is null, m_string is guaranteed to be non-null.
103 m_string = StringImpl::createSubstringSharingImpl(*m_string.impl(), 0, newLength);
104}
105
106void StringBuilder::reallocateBuffer(unsigned requiredCapacity)
107{
108 if (is8Bit())
109 reallocateBuffer<LChar>(requiredCapacity);
110 else
111 reallocateBuffer<UChar>(requiredCapacity);
112}
113
114void StringBuilder::reserveCapacity(unsigned newCapacity)
115{
116 if (hasOverflowed())
117 return;
118
119 if (m_buffer) {
120 if (newCapacity > m_buffer->length())
121 reallocateBuffer(newCapacity);
122 } else {
123 if (newCapacity > m_length) {
124 if (!m_length)
125 allocateBuffer<LChar>(static_cast<LChar*>(nullptr), newCapacity);
126 else if (m_string.is8Bit())
127 allocateBuffer<LChar>(m_string.characters8(), newCapacity);
128 else
129 allocateBuffer<UChar>(m_string.characters16(), newCapacity);
130 }
131 }
132 ASSERT(hasOverflowed() || !newCapacity || m_buffer->length() >= newCapacity);
133}

135// Alterative extendBufferForAppending that can be called from the header without inlining.
136LChar* StringBuilder::extendBufferForAppendingLChar(unsigned requiredLength)
137{
138 return extendBufferForAppending<LChar>(requiredLength);
139}
140
141UChar* StringBuilder::extendBufferForAppendingWithUpconvert(unsigned requiredLength)
142{
143 if (is8Bit()) {
144 allocateBuffer<UChar>(characters<LChar>(), expandedCapacity(capacity(), requiredLength));
145 if (UNLIKELY(hasOverflowed()))
146 return nullptr;
147 return const_cast<UChar*>(m_buffer->characters<UChar>()) + std::exchange(m_length, requiredLength);
148 }
149 return extendBufferForAppending<UChar>(requiredLength);
150}
151
152void StringBuilder::appendCharacters(const UChar* characters, unsigned length)
153{
154 if (!length || hasOverflowed())
155 return;
156 if (length == 1 && isLatin1(characters[0]) && is8Bit()) {
157 append(static_cast<LChar>(characters[0]));
158 return;
159 }
160 if (auto destination = extendBufferForAppendingWithUpconvert(saturatedSum<uint32_t>(m_length, length)))
161 StringImpl::copyCharacters(destination, characters, length);
162}
163
164void StringBuilder::appendCharacters(const LChar* characters, unsigned length)
165{
166 if (!length || hasOverflowed())
167 return;
168 if (is8Bit()) {
169 if (auto destination = extendBufferForAppending<LChar>(saturatedSum<uint32_t>(m_length, length)))
170 StringImpl::copyCharacters(destination, characters, length);
171 } else {
172 if (auto destination = extendBufferForAppending<UChar>(saturatedSum<uint32_t>(m_length, length)))
173 StringImpl::copyCharacters(destination, characters, length);
174 }
175}

177bool StringBuilder::shouldShrinkToFit() const
178{
179 // Shrink the buffer if it's 80% full or less.
180 static_assert(static_cast<size_t>(String::MaxLength) + (String::MaxLength >> 2) <= static_cast<size_t>(std::numeric_limits<unsigned>::max()));
181 return !hasOverflowed() && m_buffer && m_buffer->length() > m_length + (m_length >> 2);
182}
183
184void StringBuilder::shrinkToFit()
185{
186 if (shouldShrinkToFit()) {
187 reallocateBuffer(m_length);
188 m_string = std::exchange(m_buffer, nullptr);
189 }
190}
