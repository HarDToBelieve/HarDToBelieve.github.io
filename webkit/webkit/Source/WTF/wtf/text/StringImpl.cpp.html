55void StringStats::removeString(StringImpl& string)
56{
57 unsigned length = string.length();
58 bool isSubString = string.isSubString();
59
60 --m_totalNumberStrings;
61
62 if (string.is8Bit()) {
63 --m_number8BitStrings;
64 if (!isSubString)
65 m_total8BitData -= length;
66 } else {
67 --m_number16BitStrings;
68 if (!isSubString)
69 m_total16BitData -= length;
70 }
71
72 if (!--s_stringRemovesTillPrintStats) {
73 s_stringRemovesTillPrintStats = s_printStringStatsFrequency;
74 printStats();
75 }
76}

107StringImpl::~StringImpl()
108{
109 ASSERT(!isStatic());
110
111 StringView::invalidate(*this);
112
113 STRING_STATS_REMOVE_STRING(*this);
114
115 if (isAtom()) {
116 ASSERT(!isSymbol());
117 if (length())
118 AtomStringImpl::remove(static_cast<AtomStringImpl*>(this));
119 } else if (isSymbol()) {
120 auto& symbol = static_cast<SymbolImpl&>(*this);
121 auto* symbolRegistry = symbol.symbolRegistry();
122 if (symbolRegistry)
123 symbolRegistry->remove(*symbol.asRegisteredSymbolImpl());
124 }
125
126 BufferOwnership ownership = bufferOwnership();
127
128 if (ownership == BufferInternal)
129 return;
130 if (ownership == BufferOwned) {
131 // We use m_data8, but since it is a union with m_data16 this works either way.
132 ASSERT(m_data8);
133 StringImplMalloc::free(const_cast<LChar*>(m_data8));
134 return;
135 }
136 if (ownership == BufferExternal) {
137 auto* external = static_cast<ExternalStringImpl*>(this);
138 external->freeExternalBuffer(const_cast<LChar*>(m_data8), sizeInBytes());
139 external->m_free.~ExternalStringImplFreeFunction();
140 return;
141 }
142
143 ASSERT(ownership == BufferSubstring);
144 ASSERT(substringBuffer());
145 substringBuffer()->deref();
146}

189template<typename CharacterType> inline Ref<StringImpl> StringImpl::createUninitializedInternalNonEmpty(unsigned length, CharacterType*& data)
190{
191 ASSERT(length);
192
193 // Allocate a single buffer large enough to contain the StringImpl
194 // struct as well as the data which it contains. This removes one
195 // heap allocation from this call.
196 if (length > maxInternalLength<CharacterType>())
197 CRASH();

199 data = string->tailPointer<CharacterType>();
200 return constructInternal<CharacterType>(*string, length);
201}

213template<typename CharacterType> inline Expected<Ref<StringImpl>, UTF8ConversionError> StringImpl::reallocateInternal(Ref<StringImpl>&& originalString, unsigned length, CharacterType*& data)
214{
215 ASSERT(originalString->hasOneRef());
216 ASSERT(originalString->bufferOwnership() == BufferInternal);
217
218 if (!length) {
219 data = nullptr;
220 return Ref<StringImpl>(*empty());
221 }
222
223 // Same as createUninitialized() except here we use fastRealloc.
224 if (length > maxInternalLength<CharacterType>())
225 return makeUnexpected(UTF8ConversionError::OutOfMemory);
226
227 originalString->~StringImpl();
228 auto* string = static_cast<StringImpl*>(StringImplMalloc::tryRealloc(&originalString.leakRef(), allocationSize<CharacterType>(length)));
229 if (!string)
230 return makeUnexpected(UTF8ConversionError::OutOfMemory);
231
232 data = string->tailPointer<CharacterType>();
233 return constructInternal<CharacterType>(*string, length);
234}

325{
326 if (start >= m_length)
327 return *empty();

351Ref<StringImpl> StringImpl::convertToLowercaseWithoutLocale()
352{
353 // Note: At one time this was a hot function in the Dromaeo benchmark, specifically the
354 // no-op code path that may return ourself if we find no upper case letters and no invalid
355 // ASCII letters.
356
357 // First scan the string for uppercase and non-ASCII characters:
358 if (is8Bit()) {
359 for (unsigned i = 0; i < m_length; ++i) {
360 LChar character = m_data8[i];
361 if (UNLIKELY(!isASCII(character) || isASCIIUpper(character)))
362 return convertToLowercaseWithoutLocaleStartingAtFailingIndex8Bit(i);
363 }
364
365 return *this;
366 }
367
368 bool noUpper = true;
369 unsigned ored = 0;
370
371 for (unsigned i = 0; i < m_length; ++i) {
372 UChar character = m_data16[i];
373 if (UNLIKELY(isASCIIUpper(character)))
374 noUpper = false;
375 ored |= character;
376 }
377 // Nothing to do if the string is all ASCII with no uppercase.
378 if (noUpper && !(ored & ~0x7F))
379 return *this;
380
381 if (!(ored & ~0x7F)) {
382 UChar* data16;
383 auto newImpl = createUninitializedInternalNonEmpty(m_length, data16);
384 for (unsigned i = 0; i < m_length; ++i)
385 data16[i] = toASCIILower(m_data16[i]);
386 return newImpl;
387 }
388
389 if (m_length > MaxLength)
390 CRASH();
391 int32_t length = m_length;
392
393 // Do a slower implementation for cases that include non-ASCII characters.
394 UChar* data16;
395 auto newImpl = createUninitializedInternalNonEmpty(m_length, data16);
396
397 UErrorCode status = U_ZERO_ERROR;
398 int32_t realLength = u_strToLower(data16, length, m_data16, m_length, "", &status);
399 if (U_SUCCESS(status) && realLength == length)
400 return newImpl;
401
402 newImpl = createUninitialized(realLength, data16);
403 status = U_ZERO_ERROR;
404 u_strToLower(data16, realLength, m_data16, m_length, "", &status);
405 if (U_FAILURE(status))
406 return *this;
407 return newImpl;
408}

410Ref<StringImpl> StringImpl::convertToLowercaseWithoutLocaleStartingAtFailingIndex8Bit(unsigned failingIndex)
411{
412 ASSERT(is8Bit());
413 LChar* data8;
414 auto newImpl = createUninitializedInternalNonEmpty(m_length, data8);
415
416 for (unsigned i = 0; i < failingIndex; ++i) {
417 ASSERT(isASCII(m_data8[i]));
418 ASSERT(!isASCIIUpper(m_data8[i]));
419 data8[i] = m_data8[i];
420 }
421
422 for (unsigned i = failingIndex; i < m_length; ++i) {
423 LChar character = m_data8[i];
424 if (isASCII(character))
425 data8[i] = toASCIILower(character);
426 else {
427 ASSERT(isLatin1(u_tolower(character)));
428 data8[i] = static_cast<LChar>(u_tolower(character));
429 }
430 }
431
432 return newImpl;
433}

435Ref<StringImpl> StringImpl::convertToUppercaseWithoutLocale()
436{
437 // This function could be optimized for no-op cases the way
438 // convertToLowercaseWithoutLocale() is, but in empirical testing,
439 // few actual calls to upper() are no-ops, so it wouldn't be worth
440 // the extra time for pre-scanning.
441
442 if (m_length > MaxLength)
443 CRASH();
444 int32_t length = m_length;
445
446 if (is8Bit()) {
447 LChar* data8;
448 auto newImpl = createUninitialized(m_length, data8);
449
450 // Do a faster loop for the case where all the characters are ASCII.
451 unsigned ored = 0;
452 for (int i = 0; i < length; ++i) {
453 LChar character = m_data8[i];
454 ored |= character;
455 data8[i] = toASCIIUpper(character);
456 }
457 if (!(ored & ~0x7F))
458 return newImpl;
459
460 // Do a slower implementation for cases that include non-ASCII Latin-1 characters.
461 int numberSharpSCharacters = 0;
462
463 // There are two special cases.
464 // 1. Some Latin-1 characters when converted to upper case are 16 bit characters.
465 // 2. Lower case sharp-S converts to "SS" (two characters)
466 for (int32_t i = 0; i < length; ++i) {
467 LChar character = m_data8[i];
468 if (UNLIKELY(character == smallLetterSharpS))
469 ++numberSharpSCharacters;
470 ASSERT(u_toupper(character) <= 0xFFFF);
471 UChar upper = u_toupper(character);
472 if (UNLIKELY(!isLatin1(upper))) {
473 // Since this upper-cased character does not fit in an 8-bit string, we need to take the 16-bit path.
474 goto upconvert;
475 }
476 data8[i] = static_cast<LChar>(upper);
477 }
478
479 if (!numberSharpSCharacters)
480 return newImpl;
481
482 // We have numberSSCharacters sharp-s characters, but none of the other special characters.
483 newImpl = createUninitialized(m_length + numberSharpSCharacters, data8);
484
485 LChar* dest = data8;
486
487 for (int32_t i = 0; i < length; ++i) {
488 LChar character = m_data8[i];
489 if (character == smallLetterSharpS) {
490 *dest++ = 'S';
491 *dest++ = 'S';
492 } else {
493 ASSERT(isLatin1(u_toupper(character)));
494 *dest++ = static_cast<LChar>(u_toupper(character));
495 }
496 }
497
498 return newImpl;
499 }
500
501upconvert:
502 auto upconvertedCharacters = StringView(*this).upconvertedCharacters();
503 const UChar* source16 = upconvertedCharacters;
504
505 UChar* data16;
506 auto newImpl = createUninitialized(m_length, data16);
507
508 // Do a faster loop for the case where all the characters are ASCII.
509 unsigned ored = 0;
510 for (int i = 0; i < length; ++i) {
511 UChar character = source16[i];
512 ored |= character;
513 data16[i] = toASCIIUpper(character);
514 }
515 if (!(ored & ~0x7F))
516 return newImpl;
517
518 // Do a slower implementation for cases that include non-ASCII characters.
519 UErrorCode status = U_ZERO_ERROR;
520 int32_t realLength = u_strToUpper(data16, length, source16, m_length, "", &status);
521 if (U_SUCCESS(status) && realLength == length)
522 return newImpl;
523 newImpl = createUninitialized(realLength, data16);
524 status = U_ZERO_ERROR;
525 u_strToUpper(data16, realLength, source16, m_length, "", &status);
526 if (U_FAILURE(status))
527 return *this;
528 return newImpl;
529}

531static inline bool needsTurkishCasingRules(const AtomString& localeIdentifier)
532{
533 // Either "tr" or "az" locale, with case sensitive comparison and allowing for an ignored subtag.
534 UChar first = localeIdentifier[0];
535 UChar second = localeIdentifier[1];
536 return ((isASCIIAlphaCaselessEqual(first, 't') && isASCIIAlphaCaselessEqual(second, 'r'))
537 || (isASCIIAlphaCaselessEqual(first, 'a') && isASCIIAlphaCaselessEqual(second, 'z')))
538 && (localeIdentifier.length() == 2 || localeIdentifier[2] == '-');
539}

541Ref<StringImpl> StringImpl::convertToLowercaseWithLocale(const AtomString& localeIdentifier)
542{
543 // Use the more-optimized code path most of the time.
544 // Assuming here that the only locale-specific lowercasing is the Turkish casing rules.
545 // FIXME: Could possibly optimize further by looking for the specific sequences
546 // that have locale-specific lowercasing. There are only three of them.
547 if (!needsTurkishCasingRules(localeIdentifier))
548 return convertToLowercaseWithoutLocale();
549
550 // FIXME: Could share more code with the main StringImpl::lower by factoring out
551 // this last part into a shared function that takes a locale string, since this is
552 // just like the end of that function.
553
554 if (m_length > MaxLength)
555 CRASH();
556 int length = m_length;
557
558 // Below, we pass in the hardcoded locale "tr". Passing that is more efficient than
559 // allocating memory just to turn localeIdentifier into a C string, and we assume
560 // there is no difference between the uppercasing for "tr" and "az" locales.
561 auto upconvertedCharacters = StringView(*this).upconvertedCharacters();
562 const UChar* source16 = upconvertedCharacters;
563 UChar* data16;
564 auto newString = createUninitialized(length, data16);
565 UErrorCode status = U_ZERO_ERROR;
566 int realLength = u_strToLower(data16, length, source16, length, "tr", &status);
567 if (U_SUCCESS(status) && realLength == length)
568 return newString;
569 newString = createUninitialized(realLength, data16);
570 status = U_ZERO_ERROR;
571 u_strToLower(data16, realLength, source16, length, "tr", &status);
572 if (U_FAILURE(status))
573 return *this;
574 return newString;
575}

608Ref<StringImpl> StringImpl::foldCase()
609{
610 if (is8Bit()) {
611 unsigned failingIndex;
612 for (unsigned i = 0; i < m_length; ++i) {
613 auto character = m_data8[i];
614 if (UNLIKELY(!isASCII(character) || isASCIIUpper(character))) {
615 failingIndex = i;
616 goto SlowPath;
617 }
618 }
619 // String was all ASCII and no uppercase, so just return as-is.
620 return *this;
621
622SlowPath:
623 bool need16BitCharacters = false;
624 for (unsigned i = failingIndex; i < m_length; ++i) {
625 auto character = m_data8[i];
626 if (character == 0xB5 || character == 0xDF) {
627 need16BitCharacters = true;
628 break;
629 }
630 }
631
632 if (!need16BitCharacters) {
633 LChar* data8;
634 auto folded = createUninitializedInternalNonEmpty(m_length, data8);
635 copyCharacters(data8, m_data8, failingIndex);
636 for (unsigned i = failingIndex; i < m_length; ++i) {
637 auto character = m_data8[i];
638 if (isASCII(character))
639 data8[i] = toASCIILower(character);
640 else {
641 ASSERT(isLatin1(u_foldCase(character, U_FOLD_CASE_DEFAULT)));
642 data8[i] = static_cast<LChar>(u_foldCase(character, U_FOLD_CASE_DEFAULT));
643 }
644 }
645 return folded;
646 }
647 } else {
648 // FIXME: Unclear why we use goto in the 8-bit case, and a different approach in the 16-bit case.
649 bool noUpper = true;
650 unsigned ored = 0;
651 for (unsigned i = 0; i < m_length; ++i) {
652 UChar character = m_data16[i];
653 if (UNLIKELY(isASCIIUpper(character)))
654 noUpper = false;
655 ored |= character;
656 }
657 if (!(ored & ~0x7F)) {
658 if (noUpper) {
659 // String was all ASCII and no uppercase, so just return as-is.
660 return *this;
661 }
662 UChar* data16;
663 auto folded = createUninitializedInternalNonEmpty(m_length, data16);
664 for (unsigned i = 0; i < m_length; ++i)
665 data16[i] = toASCIILower(m_data16[i]);
666 return folded;
667 }
668 }
669
670 if (m_length > MaxLength)
671 CRASH();
672
673 auto upconvertedCharacters = StringView(*this).upconvertedCharacters();
674
675 UChar* data;
676 auto folded = createUninitializedInternalNonEmpty(m_length, data);
677 int32_t length = m_length;
678 UErrorCode status = U_ZERO_ERROR;
679 int32_t realLength = u_strFoldCase(data, length, upconvertedCharacters, length, U_FOLD_CASE_DEFAULT, &status);
680 if (U_SUCCESS(status) && realLength == length)
681 return folded;
682 ASSERT(realLength > length);
683 folded = createUninitializedInternalNonEmpty(realLength, data);
684 status = U_ZERO_ERROR;
685 u_strFoldCase(data, realLength, upconvertedCharacters, length, U_FOLD_CASE_DEFAULT, &status);
686 if (U_FAILURE(status))
687 return *this;
688 return folded;
689}

764template<typename CharacterType, class UCharPredicate> inline Ref<StringImpl> StringImpl::simplifyMatchedCharactersToSpace(UCharPredicate predicate)
765{
766 StringBuffer<CharacterType> data(m_length);
767
768 auto* from = characters<CharacterType>();
769 auto* fromEnd = from + m_length;
770 unsigned outc = 0;
771 bool changedToSpace = false;
772
773 auto* to = data.characters();
774
775 while (true) {
776 while (from != fromEnd && predicate(*from)) {
777 if (*from != ' ')
778 changedToSpace = true;
779 ++from;
780 }
781 while (from != fromEnd && !predicate(*from))
782 to[outc++] = *from++;
783 if (from != fromEnd)
784 to[outc++] = ' ';
785 else
786 break;
787 }
788
789 if (outc && to[outc - 1] == ' ')
790 --outc;
791
792 if (outc == m_length && !changedToSpace)
793 return *this;
794
795 data.shrink(outc);
796
797 return adopt(WTFMove(data));
798}

838 if (!matchString)
839 return notFound;

841 if (matchStringLength > MaxLength)
842 CRASH();

844 if (!matchLength)
845 return std::min(index, length());
846
847 // Optimization 1: fast case for strings of length 1.
848 if (matchLength == 1) {
849 if (is8Bit())

851 return WTF::find(characters16(), length(), *matchString, index);

853
854 // Check index & matchLength are in range.
855 if (index > length())
856 return notFound;

858 if (matchLength > searchLength)
859 return notFound;

872 searchHash += searchCharacters[i];
873 matchHash += matchString[i];

875
876 unsigned i = 0;
877 while (searchHash != matchHash || !equal(searchCharacters + i, matchString, matchLength)) {
878 if (i == delta)
879 return notFound;
880 searchHash += searchCharacters[i + matchLength];
881 searchHash -= searchCharacters[i];
882 ++i;
883 }
884 return index + i;
885 }
886
887 const UChar* searchCharacters = characters16() + index;

891 for (unsigned i = 0; i < matchLength; ++i) {
892 searchHash += searchCharacters[i];

899 return notFound;
900 searchHash += searchCharacters[i + matchLength];
901 searchHash -= searchCharacters[i];
902 ++i;
903 }

986template <typename SearchCharacterType, typename MatchCharacterType>
987ALWAYS_INLINE static size_t reverseFindInner(const SearchCharacterType* searchCharacters, const MatchCharacterType* matchCharacters, unsigned index, unsigned length, unsigned matchLength)
988{
989 // Optimization: keep a running hash of the strings,
990 // only call equal if the hashes match.
991
992 // delta is the number of additional times to test; delta == 0 means test only once.
993 unsigned delta = std::min(index, length - matchLength);
994
995 unsigned searchHash = 0;
996 unsigned matchHash = 0;
997 for (unsigned i = 0; i < matchLength; ++i) {
998 searchHash += searchCharacters[delta + i];
999 matchHash += matchCharacters[i];
1000 }
1001
1002 // keep looping until we match
1003 while (searchHash != matchHash || !equal(searchCharacters + delta, matchCharacters, matchLength)) {
1004 if (!delta)
1005 return notFound;
1006 --delta;
1007 searchHash -= searchCharacters[delta + matchLength];
1008 searchHash += searchCharacters[delta];
1009 }
1010 return delta;
1011}

1408Ref<StringImpl> StringImpl::replace(StringImpl* pattern, StringImpl* replacement)
1409{
1410 if (!pattern || !replacement)
1411 return *this;
1412
1413 unsigned patternLength = pattern->length();
1414 if (!patternLength)
1415 return *this;
1416
1417 unsigned repStrLength = replacement->length();
1418 size_t srcSegmentStart = 0;
1419 unsigned matchCount = 0;
1420
1421 // Count the matches.
1422 while ((srcSegmentStart = find(pattern, srcSegmentStart)) != notFound) {
1423 ++matchCount;
1424 srcSegmentStart += patternLength;
1425 }
1426
1427 // If we have 0 matches, we don't have to do any more work
1428 if (!matchCount)
1429 return *this;
1430
1431 unsigned newSize = m_length - matchCount * patternLength;
1432 if (repStrLength && matchCount > MaxLength / repStrLength)
1433 CRASH();
1434
1435 if (newSize > (MaxLength - matchCount * repStrLength))
1436 CRASH();
1437
1438 newSize += matchCount * repStrLength;
1439
1440
1441 // Construct the new data
1442 size_t srcSegmentEnd;
1443 unsigned srcSegmentLength;
1444 srcSegmentStart = 0;
1445 unsigned dstOffset = 0;
1446 bool srcIs8Bit = is8Bit();
1447 bool replacementIs8Bit = replacement->is8Bit();
1448
1449 // There are 4 cases:
1450 // 1. This and replacement are both 8 bit.
1451 // 2. This and replacement are both 16 bit.
1452 // 3. This is 8 bit and replacement is 16 bit.
1453 // 4. This is 16 bit and replacement is 8 bit.
1454 if (srcIs8Bit && replacementIs8Bit) {
1455 // Case 1
1456 LChar* data;
1457 auto newImpl = createUninitialized(newSize, data);
1458 while ((srcSegmentEnd = find(pattern, srcSegmentStart)) != notFound) {
1459 srcSegmentLength = srcSegmentEnd - srcSegmentStart;
1460 copyCharacters(data + dstOffset, m_data8 + srcSegmentStart, srcSegmentLength);
1461 dstOffset += srcSegmentLength;
1462 copyCharacters(data + dstOffset, replacement->m_data8, repStrLength);
1463 dstOffset += repStrLength;
1464 srcSegmentStart = srcSegmentEnd + patternLength;
1465 }
1466
1467 srcSegmentLength = m_length - srcSegmentStart;
1468 copyCharacters(data + dstOffset, m_data8 + srcSegmentStart, srcSegmentLength);
1469
1470 ASSERT(dstOffset + srcSegmentLength == newImpl.get().length());
1471
1472 return newImpl;
1473 }
1474
1475 UChar* data;
1476 auto newImpl = createUninitialized(newSize, data);
1477 while ((srcSegmentEnd = find(pattern, srcSegmentStart)) != notFound) {
1478 srcSegmentLength = srcSegmentEnd - srcSegmentStart;
1479 if (srcIs8Bit) {
1480 // Case 3.
1481 copyCharacters(data + dstOffset, m_data8 + srcSegmentStart, srcSegmentLength);
1482 } else {
1483 // Case 2 & 4.
1484 copyCharacters(data + dstOffset, m_data16 + srcSegmentStart, srcSegmentLength);
1485 }
1486 dstOffset += srcSegmentLength;
1487 if (replacementIs8Bit) {
1488 // Cases 2 & 3.
1489 copyCharacters(data + dstOffset, replacement->m_data8, repStrLength);
1490 } else {
1491 // Case 4
1492 copyCharacters(data + dstOffset, replacement->m_data16, repStrLength);
1493 }
1494 dstOffset += repStrLength;
1495 srcSegmentStart = srcSegmentEnd + patternLength;
1496 }
1497
1498 srcSegmentLength = m_length - srcSegmentStart;
1499 if (srcIs8Bit) {
1500 // Case 3.
1501 copyCharacters(data + dstOffset, m_data8 + srcSegmentStart, srcSegmentLength);
1502 } else {
1503 // Cases 2 & 4.
1504 copyCharacters(data + dstOffset, m_data16 + srcSegmentStart, srcSegmentLength);
1505 }
1506
1507 ASSERT(dstOffset + srcSegmentLength == newImpl.get().length());
1508
1509 return newImpl;

1648// Helper to write a three-byte UTF-8 code point into the buffer; caller must ensure room is available.
1649static inline void putUTF8Triple(char*& buffer, UChar character)
1650{

1657UTF8ConversionError StringImpl::utf8Impl(const UChar* characters, unsigned length, char*& buffer, size_t bufferSize, ConversionMode mode)
1658{
1659 if (mode == StrictConversionReplacingUnpairedSurrogatesWithFFFD) {
1660 const UChar* charactersEnd = characters + length;
1661 char* bufferEnd = buffer + bufferSize;
1662 while (characters < charactersEnd) {
1663 // Use strict conversion to detect unpaired surrogates.
1664 auto result = convertUTF16ToUTF8(&characters, charactersEnd, &buffer, bufferEnd);
1665 ASSERT(result != TargetExhausted);
1666 // Conversion fails when there is an unpaired surrogate.
1667 // Put replacement character (U+FFFD) instead of the unpaired surrogate.
1668 if (result != ConversionOK) {
1669 ASSERT((0xD800 <= *characters && *characters <= 0xDFFF));
1670 // There should be room left, since one UChar hasn't been converted.
1671 ASSERT((buffer + 3) <= bufferEnd);
1672 putUTF8Triple(buffer, replacementCharacter);
1673 ++characters;
1674 }
1675 }
1676 } else {
1677 bool strict = mode == StrictConversion;
1678 const UChar* originalCharacters = characters;
1679 auto result = convertUTF16ToUTF8(&characters, characters + length, &buffer, buffer + bufferSize, strict);
1680 ASSERT(result != TargetExhausted); // (length * 3) should be sufficient for any conversion
1681
1682 // Only produced from strict conversion.
1683 if (result == SourceIllegal) {
1684 ASSERT(strict);
1685 return UTF8ConversionError::IllegalSource;
1686 }
1687
1688 // Check for an unconverted high surrogate.
1689 if (result == SourceExhausted) {
1690 if (strict)
1691 return UTF8ConversionError::SourceExhausted;
1692 // This should be one unpaired high surrogate. Treat it the same
1693 // was as an unpaired high surrogate would have been handled in
1694 // the middle of a string with non-strict conversion - which is
1695 // to say, simply encode it to UTF-8.
1696 ASSERT_UNUSED(
1697 originalCharacters, (characters + 1) == (originalCharacters + length));
1698 ASSERT((*characters >= 0xD800) && (*characters <= 0xDBFF));
1699 // There should be room left, since one UChar hasn't been converted.
1700 ASSERT((buffer + 3) <= (buffer + bufferSize));
1701 putUTF8Triple(buffer, *characters);
1702 }
1703 }
1704
1705 return UTF8ConversionError::None;
1706}

1737{
1738 ASSERT(offset <= this->length());
1739 ASSERT(offset + length <= this->length());
1740
1741 if (!length)
1742 return CString("", 0);
1743
1744 // Allocate a buffer big enough to hold all the characters
1745 // (an individual UTF-16 UChar can only expand to 3 UTF-8 bytes).
1746 // Optimization ideas, if we find this function is hot:
1747 // * We could speculatively create a CStringBuffer to contain 'length'
1748 // characters, and resize if necessary (i.e. if the buffer contains
1749 // non-ascii characters). (Alternatively, scan the buffer first for
1750 // ascii characters, so we know this will be sufficient).
1751 // * We could allocate a CStringBuffer with an appropriate size to
1752 // have a good chance of being able to write the string into the
1753 // buffer without reallocing (say, 1.5 x length).
