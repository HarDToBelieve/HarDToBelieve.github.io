112void StringView::SplitResult::Iterator::findNextSubstring()
113{
114 for (size_t separatorPosition; (separatorPosition = m_result.m_string.find(m_result.m_separator, m_position)) != notFound; ++m_position) {
115 if (m_result.m_allowEmptyEntries || separatorPosition > m_position) {
116 m_length = separatorPosition - m_position;
117 return;
118 }
119 }
120 m_length = m_result.m_string.length() - m_position;
121 if (!m_length && !m_result.m_allowEmptyEntries)
122 m_isDone = true;
123}
124
125auto StringView::SplitResult::Iterator::operator++() -> Iterator&
126{
127 ASSERT(m_position <= m_result.m_string.length() && !m_isDone);
128 m_position += m_length;
129 if (m_position < m_result.m_string.length()) {
130 ++m_position;
131 findNextSubstring();
132 } else if (!m_isDone)
133 m_isDone = true;
134 return *this;
135}

137class StringView::GraphemeClusters::Iterator::Impl {
138 WTF_MAKE_FAST_ALLOCATED;
139public:
140 Impl(const StringView& stringView, std::optional<NonSharedCharacterBreakIterator>&& iterator, unsigned index)
141 : m_stringView(stringView)
142 , m_iterator(WTFMove(iterator))
143 , m_index(index)
144 , m_indexEnd(computeIndexEnd())
145 {
146 }
147
148 void operator++()
149 {
150 ASSERT(m_indexEnd > m_index);
151 m_index = m_indexEnd;
152 m_indexEnd = computeIndexEnd();
153 }
154
155 StringView operator*() const
156 {
157 if (m_stringView.is8Bit())
158 return StringView(m_stringView.characters8() + m_index, m_indexEnd - m_index);
159 return StringView(m_stringView.characters16() + m_index, m_indexEnd - m_index);
160 }
161
162 bool operator==(const Impl& other) const
163 {
164 ASSERT(&m_stringView == &other.m_stringView);
165 auto result = m_index == other.m_index;
166 ASSERT(!result || m_indexEnd == other.m_indexEnd);
167 return result;
168 }
169
170 unsigned computeIndexEnd()
171 {
172 if (!m_iterator)
173 return 0;
174 if (m_index == m_stringView.length())
175 return m_index;
176 return ubrk_following(m_iterator.value(), m_index);
177 }
178
179private:
180 const StringView& m_stringView;
181 std::optional<NonSharedCharacterBreakIterator> m_iterator;
182 unsigned m_index;
183 unsigned m_indexEnd;
184};

221enum class ASCIICase { Lower, Upper };
222
223template<ASCIICase type, typename CharacterType>
224String convertASCIICase(const CharacterType* input, unsigned length)
225{
226 if (!input)
227 return { };
228
229 CharacterType* characters;
230 auto result = String::createUninitialized(length, characters);
231 for (unsigned i = 0; i < length; ++i)
232 characters[i] = type == ASCIICase::Lower ? toASCIILower(input[i]) : toASCIIUpper(input[i]);
233 return result;
234}
235
236String StringView::convertToASCIILowercase() const
237{
238 if (m_is8Bit)
239 return convertASCIICase<ASCIICase::Lower>(static_cast<const LChar*>(m_characters), m_length);
240 return convertASCIICase<ASCIICase::Lower>(static_cast<const UChar*>(m_characters), m_length);
241}
242
243String StringView::convertToASCIIUppercase() const
244{
245 if (m_is8Bit)
246 return convertASCIICase<ASCIICase::Upper>(static_cast<const LChar*>(m_characters), m_length);
247 return convertASCIICase<ASCIICase::Upper>(static_cast<const UChar*>(m_characters), m_length);
248}

250template<typename CharacterType>
251static AtomString convertASCIILowercaseAtom(const CharacterType* input, unsigned length)
252{
253 for (unsigned i = 0; i < length; ++i) {
254 if (UNLIKELY(isASCIIUpper(input[i]))) {
255 CharacterType* characters;
256 auto result = String::createUninitialized(length, characters);
257 StringImpl::copyCharacters(characters, input, i);
258 for (; i < length; ++i)
259 characters[i] = toASCIILower(input[i]);
260 return result;
261 }
262 }
263 // Fast path when the StringView is already all lowercase.
264 return AtomString(input, length);
265}
266
267AtomString StringView::convertToASCIILowercaseAtom() const
268{
269 if (m_is8Bit)
270 return convertASCIILowercaseAtom(characters8(), m_length);
271 return convertASCIILowercaseAtom(characters16(), m_length);
272}

274template<typename DestinationCharacterType, typename SourceCharacterType>
275void getCharactersWithASCIICaseInternal(StringView::CaseConvertType type, DestinationCharacterType* destination, const SourceCharacterType* source, unsigned length)
276{
277 static_assert(std::is_same<SourceCharacterType, LChar>::value || std::is_same<SourceCharacterType, UChar>::value);
278 static_assert(std::is_same<DestinationCharacterType, LChar>::value || std::is_same<DestinationCharacterType, UChar>::value);
279 static_assert(sizeof(DestinationCharacterType) >= sizeof(SourceCharacterType));
280 auto caseConvert = (type == StringView::CaseConvertType::Lower) ? toASCIILower<SourceCharacterType> : toASCIIUpper<SourceCharacterType>;
281 for (unsigned i = 0; i < length; ++i)
282 destination[i] = caseConvert(source[i]);
283}

285void StringView::getCharactersWithASCIICase(CaseConvertType type, LChar* destination) const
286{
287 ASSERT(is8Bit());
288 getCharactersWithASCIICaseInternal(type, destination, characters8(), m_length);
289}

300StringViewWithUnderlyingString normalizedNFC(StringView string)
301{
302 // Latin-1 characters are unaffected by normalization.
303 if (string.is8Bit())
304 return { string, { } };
305
306 UErrorCode status = U_ZERO_ERROR;
307 const UNormalizer2* normalizer = unorm2_getNFCInstance(&status);
308 ASSERT(U_SUCCESS(status));
309
310 // No need to normalize if already normalized.
311 UBool checkResult = unorm2_isNormalized(normalizer, string.characters16(), string.length(), &status);
312 if (checkResult)
313 return { string, { } };
314
315 unsigned normalizedLength = unorm2_normalize(normalizer, string.characters16(), string.length(), nullptr, 0, &status);
316 ASSERT(needsToGrowToProduceBuffer(status));
317
318 UChar* characters;
319 String result = String::createUninitialized(normalizedLength, characters);
320
321 status = U_ZERO_ERROR;
322 unorm2_normalize(normalizer, string.characters16(), string.length(), characters, normalizedLength, &status);
323 ASSERT(U_SUCCESS(status));
324
325 StringView view { result };
326 return { view, WTFMove(result) };
327}
328
329String normalizedNFC(const String& string)
330{
331 auto result = normalizedNFC(StringView { string });
332 if (result.underlyingString.isNull())
333 return string;
334 return result.underlyingString;
335}

337bool equalRespectingNullity(StringView a, StringView b)
338{
339 if (a.m_characters == b.m_characters) {
340 ASSERT(a.is8Bit() == b.is8Bit());
341 return a.length() == b.length();
342 }
343
344 if (a.isEmpty() && b.isEmpty())
345 return a.isNull() == b.isNull();
346
347 return equalCommon(a, b);
348}

355int codePointCompare(StringView lhs, StringView rhs)
356{
357 bool lhsIs8Bit = lhs.is8Bit();
358 bool rhsIs8Bit = rhs.is8Bit();
359 if (lhsIs8Bit) {
360 if (rhsIs8Bit)
361 return codePointCompare(lhs.characters8(), lhs.length(), rhs.characters8(), rhs.length());
362 return codePointCompare(lhs.characters8(), lhs.length(), rhs.characters16(), rhs.length());
363 }
364 if (rhsIs8Bit)
365 return codePointCompare(lhs.characters16(), lhs.length(), rhs.characters8(), rhs.length());
366 return codePointCompare(lhs.characters16(), lhs.length(), rhs.characters16(), rhs.length());
367}

371// Manage reference count manually so UnderlyingString does not need to be defined in the header.
372
373struct StringView::UnderlyingString {
374 WTF_MAKE_STRUCT_FAST_ALLOCATED;
375 std::atomic_uint refCount { 1u };
376 bool isValid { true };
377 const StringImpl& string;
378 explicit UnderlyingString(const StringImpl&);
379};
380
381StringView::UnderlyingString::UnderlyingString(const StringImpl& string)
382 : string(string)
383{
384}
385
386static Lock underlyingStringsLock;
387
388static HashMap<const StringImpl*, StringView::UnderlyingString*>& underlyingStrings() WTF_REQUIRES_LOCK(underlyingStringsLock)
389{
390 static NeverDestroyed<HashMap<const StringImpl*, StringView::UnderlyingString*>> map;
391 return map;
392}
393
394void StringView::invalidate(const StringImpl& stringToBeDestroyed)
395{
396 UnderlyingString* underlyingString;
397 {
398 Locker locker { underlyingStringsLock };
399 underlyingString = underlyingStrings().take(&stringToBeDestroyed);
400 if (!underlyingString)
401 return;
402 }
403 ASSERT(underlyingString->isValid);
404 underlyingString->isValid = false;
405}
406
407bool StringView::underlyingStringIsValid() const
408{
409 return !m_underlyingString || m_underlyingString->isValid;
410}
411
412void StringView::adoptUnderlyingString(UnderlyingString* underlyingString)
413{
414 if (m_underlyingString) {
415 Locker locker { underlyingStringsLock };
416 if (!--m_underlyingString->refCount) {
417 if (m_underlyingString->isValid) {
418 underlyingStrings().remove(&m_underlyingString->string);
419 }
420 delete m_underlyingString;
421 }
422 }
423 m_underlyingString = underlyingString;
424}
425
426void StringView::setUnderlyingString(const StringImpl* string)
427{
428 UnderlyingString* underlyingString;
429 if (!string)
430 underlyingString = nullptr;
431 else {
432 Locker locker { underlyingStringsLock };
433 auto result = underlyingStrings().add(string, nullptr);
434 if (result.isNewEntry)
435 result.iterator->value = new UnderlyingString(*string);
436 else
437 ++result.iterator->value->refCount;
438 underlyingString = result.iterator->value;
439 }
440 adoptUnderlyingString(underlyingString);
441}
442
443void StringView::setUnderlyingString(const StringView& otherString)
444{
445 UnderlyingString* underlyingString = otherString.m_underlyingString;
446 if (underlyingString) {
447 // It's safe to inc the refCount here without locking underlyingStringsLock
448 // because UnderlyingString::refCount is a std::atomic_uint, and we're
449 // guaranteed that the StringView we're copying it from will at least
450 // have 1 ref on it, thereby keeping it alive regardless of what other
451 // threads may be doing.
452 ++underlyingString->refCount;
453 }
454 adoptUnderlyingString(underlyingString);
455}
