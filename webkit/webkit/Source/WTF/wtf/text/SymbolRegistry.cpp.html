31SymbolRegistry::SymbolRegistry(Type type)
32 : m_symbolType(type)
33{
34}
35
36SymbolRegistry::~SymbolRegistry()
37{
38 for (auto& key : m_table) {
39 ASSERT(key.impl()->isSymbol());
40 static_cast<SymbolImpl*>(key.impl())->asRegisteredSymbolImpl()->clearSymbolRegistry();
41 }
42}
43
44Ref<RegisteredSymbolImpl> SymbolRegistry::symbolForKey(const String& rep)
45{
46 auto addResult = m_table.add(SymbolRegistryKey(rep.impl()));
47 if (!addResult.isNewEntry) {
48 ASSERT(addResult.iterator->impl()->isSymbol());
49 return *static_cast<SymbolImpl*>(addResult.iterator->impl())->asRegisteredSymbolImpl();
50 }
51
52 RefPtr<RegisteredSymbolImpl> symbol;
53 if (m_symbolType == Type::PrivateSymbol)
54 symbol = RegisteredSymbolImpl::createPrivate(*rep.impl(), *this);
55 else
56 symbol = RegisteredSymbolImpl::create(*rep.impl(), *this);
57
58 *addResult.iterator = SymbolRegistryKey(symbol.get());
59 return symbol.releaseNonNull();
60}
61
62void SymbolRegistry::remove(RegisteredSymbolImpl& uid)
63{
64 ASSERT(uid.symbolRegistry() == this);
65 auto iterator = m_table.find(SymbolRegistryKey(&uid));
66 ASSERT_WITH_MESSAGE(iterator != m_table.end(), "The string being removed is registered in the string table of an other thread!");
67 m_table.remove(iterator);
68}
