87void String::append(const String& otherString)
88{
89 // FIXME: This is extremely inefficient. So much so that we might want to take this out of String's API.
90
91 if (!m_impl) {
92 m_impl = otherString.m_impl;
93 return;
94 }
95
96 if (otherString.isEmpty())
97 return;
98
99 auto length = m_impl->length();
100 auto otherLength = otherString.m_impl->length();
101 if (otherLength > MaxLength - length)
102 CRASH();
103
104 if (m_impl->is8Bit() && otherString.m_impl->is8Bit()) {
105 LChar* data;
106 auto newImpl = StringImpl::createUninitialized(length + otherLength, data);
107 StringImpl::copyCharacters(data, m_impl->characters8(), length);
108 StringImpl::copyCharacters(data + length, otherString.m_impl->characters8(), otherLength);
109 m_impl = WTFMove(newImpl);
110 return;
111 }
112 UChar* data;
113 auto newImpl = StringImpl::createUninitialized(length + otherLength, data);
114 StringView(*m_impl).getCharactersWithUpconvert(data);
115 StringView(*otherString.m_impl).getCharactersWithUpconvert(data + length);
116 m_impl = WTFMove(newImpl);
117}

119void String::append(LChar character)
120{
121 // FIXME: This is extremely inefficient. So much so that we might want to take this out of String's API.
122
123 if (!m_impl) {
124 m_impl = StringImpl::create(&character, 1);
125 return;
126 }
127 if (!is8Bit()) {
128 append(static_cast<UChar>(character));
129 return;
130 }
131 if (m_impl->length() >= MaxLength)
132 CRASH();
133 LChar* data;
134 auto newImpl = StringImpl::createUninitialized(m_impl->length() + 1, data);
135 StringImpl::copyCharacters(data, m_impl->characters8(), m_impl->length());
136 data[m_impl->length()] = character;
137 m_impl = WTFMove(newImpl);
138}

166void String::insert(const String& string, unsigned position)
167{
168 // FIXME: This is extremely inefficient. So much so that we might want to take this out of String's API.
169
170 unsigned lengthToInsert = string.length();
171
172 if (!lengthToInsert) {
173 if (string.isNull())
174 return;
175 if (isNull())
176 m_impl = string.impl();
177 return;
178 }
179
180 if (position >= length()) {
181 append(string);
182 return;
183 }
184
185 if (lengthToInsert > MaxLength - length())
186 CRASH();
187
188 if (is8Bit() && string.is8Bit()) {
189 LChar* data;
190 auto newString = StringImpl::createUninitialized(length() + lengthToInsert, data);
191 StringView(*m_impl).substring(0, position).getCharactersWithUpconvert(data);
192 StringView(string).getCharactersWithUpconvert(data + position);
193 StringView(*m_impl).substring(position).getCharactersWithUpconvert(data + position + lengthToInsert);
194 m_impl = WTFMove(newString);
195 } else {
196 UChar* data;
197 auto newString = StringImpl::createUninitialized(length() + lengthToInsert, data);
198 StringView(*m_impl).substring(0, position).getCharactersWithUpconvert(data);
199 StringView(string).getCharactersWithUpconvert(data + position);
200 StringView(*m_impl).substring(position).getCharactersWithUpconvert(data + position + lengthToInsert);
201 m_impl = WTFMove(newString);
202 }
203}

205void String::append(const LChar* charactersToAppend, unsigned lengthToAppend)
206{
207 // FIXME: This is extremely inefficient. So much so that we might want to take this out of String's API.
208
209 if (!m_impl) {
210 if (!charactersToAppend)
211 return;
212 m_impl = StringImpl::create(charactersToAppend, lengthToAppend);
213 return;
214 }
215
216 if (!lengthToAppend)
217 return;
218
219 ASSERT(charactersToAppend);
220
221 unsigned strLength = m_impl->length();
222
223 if (m_impl->is8Bit()) {
224 if (lengthToAppend > MaxLength - strLength)
225 CRASH();
226 LChar* data;
227 auto newImpl = StringImpl::createUninitialized(strLength + lengthToAppend, data);
228 StringImpl::copyCharacters(data, m_impl->characters8(), strLength);
229 StringImpl::copyCharacters(data + strLength, charactersToAppend, lengthToAppend);
230 m_impl = WTFMove(newImpl);
231 return;
232 }
233
234 if (lengthToAppend > MaxLength - strLength)
235 CRASH();
236 UChar* data;
237 auto newImpl = StringImpl::createUninitialized(length() + lengthToAppend, data);
238 StringImpl::copyCharacters(data, m_impl->characters16(), strLength);
239 StringImpl::copyCharacters(data + strLength, charactersToAppend, lengthToAppend);
240 m_impl = WTFMove(newImpl);
241}

243void String::append(const UChar* charactersToAppend, unsigned lengthToAppend)
244{
245 // FIXME: This is extremely inefficient. So much so that we might want to take this out of String's API.
246
247 if (!m_impl) {
248 if (!charactersToAppend)
249 return;
250 m_impl = StringImpl::create(charactersToAppend, lengthToAppend);
251 return;
252 }
253
254 if (!lengthToAppend)

266 else

268 StringImpl::copyCharacters(data + strLength, charactersToAppend, lengthToAppend);
269 m_impl = WTFMove(newImpl);
270}

286template<typename CharacterType> inline void String::removeInternal(const CharacterType* characters, unsigned position, unsigned lengthToRemove)
287{
288 CharacterType* data;
289 auto newImpl = StringImpl::createUninitialized(length() - lengthToRemove, data);
290 StringImpl::copyCharacters(data, characters, position);
291 StringImpl::copyCharacters(data + position, characters + position + lengthToRemove, length() - lengthToRemove - position);
292 m_impl = WTFMove(newImpl);
293}

295void String::remove(unsigned position, unsigned lengthToRemove)
296{
297 if (!lengthToRemove)
298 return;
299 auto length = this->length();
300 if (position >= length)
301 return;
302 lengthToRemove = std::min(lengthToRemove, length - position);
303 if (is8Bit())
304 removeInternal(characters8(), position, lengthToRemove);
305 else
306 removeInternal(characters16(), position, lengthToRemove);
307}

315String String::substringSharingImpl(unsigned offset, unsigned length) const
316{
317 // FIXME: We used to check against a limit of Heap::minExtraCost / sizeof(UChar).
318
319 unsigned stringLength = this->length();
320 offset = std::min(offset, stringLength);
321 length = std::min(length, stringLength - offset);
322
323 if (!offset && length == stringLength)
324 return *this;
325 return StringImpl::createSubstringSharingImpl(*m_impl, offset, length);
326}

408Vector<UChar> String::charactersWithoutNullTermination() const
409{
410 Vector<UChar> result;
411
412 if (m_impl) {
413 result.reserveInitialCapacity(length() + 1);
414
415 if (is8Bit()) {
416 const LChar* characters8 = m_impl->characters8();
417 for (size_t i = 0; i < length(); ++i)
418 result.uncheckedAppend(characters8[i]);
419 } else {
420 const UChar* characters16 = m_impl->characters16();
421 result.append(characters16, m_impl->length());
422 }
423 }
424
425 return result;
426}

532bool String::isSafeToSendToAnotherThread() const
533{
534 // AtomStrings are not safe to send between threads, as ~StringImpl()

536 return isEmpty() || (m_impl->hasOneRef() && !m_impl->isAtom());

539template<bool allowEmptyEntries>
540inline Vector<String> String::splitInternal(const String& separator) const
541{
542 Vector<String> result;
543
544 unsigned startPos = 0;
545 size_t endPos;
546 while ((endPos = find(separator, startPos)) != notFound) {
547 if (allowEmptyEntries || startPos != endPos)
548 result.append(substring(startPos, endPos - startPos));
549 startPos = endPos + separator.length();
550 }
551 if (allowEmptyEntries || startPos != length())
552 result.append(substring(startPos));
553
554 return result;
555}
556
557template<bool allowEmptyEntries>
558inline void String::splitInternal(UChar separator, const SplitFunctor& functor) const
559{
560 StringView view(*this);
561
562 unsigned startPos = 0;
563 size_t endPos;
564 while ((endPos = find(separator, startPos)) != notFound) {
565 if (allowEmptyEntries || startPos != endPos)
566 functor(view.substring(startPos, endPos - startPos));
567 startPos = endPos + 1;
568 }
569 if (allowEmptyEntries || startPos != length())
570 functor(view.substring(startPos));
571}
572
573template<bool allowEmptyEntries>
574inline Vector<String> String::splitInternal(UChar separator) const
575{
576 Vector<String> result;
577 splitInternal<allowEmptyEntries>(separator, [&result](StringView item) {
578 result.append(item.toString());
579 });
580
581 return result;
582}

614CString String::ascii() const
615{
616 // Printable ASCII characters 32..127 and the null character are
617 // preserved, characters outside of this range are converted to '?'.
618
619 unsigned length = this->length();
620 if (!length) {
621 char* characterBuffer;
622 return CString::newUninitialized(length, characterBuffer);
623 }
624
625 if (this->is8Bit()) {
626 const LChar* characters = this->characters8();
627
628 char* characterBuffer;
629 CString result = CString::newUninitialized(length, characterBuffer);
630
631 for (unsigned i = 0; i < length; ++i) {
632 LChar ch = characters[i];
633 characterBuffer[i] = ch && (ch < 0x20 || ch > 0x7f) ? '?' : ch;
634 }
635
636 return result;
637 }
638
639 const UChar* characters = this->characters16();
640
641 char* characterBuffer;
642 CString result = CString::newUninitialized(length, characterBuffer);
643
644 for (unsigned i = 0; i < length; ++i) {
645 UChar ch = characters[i];
646 characterBuffer[i] = ch && (ch < 0x20 || ch > 0x7f) ? '?' : ch;
647 }
648
649 return result;
650}

652CString String::latin1() const
653{
654 // Basic Latin1 (ISO) encoding - Unicode characters 0..255 are
655 // preserved, characters outside of this range are converted to '?'.
656
657 unsigned length = this->length();
658
659 if (!length)
660 return CString("", 0);
661
662 if (is8Bit())
663 return CString(this->characters8(), length);
664
665 const UChar* characters = this->characters16();
666
667 char* characterBuffer;
668 CString result = CString::newUninitialized(length, characterBuffer);
669
670 for (unsigned i = 0; i < length; ++i) {
671 UChar ch = characters[i];
672 characterBuffer[i] = !isLatin1(ch) ? '?' : ch;
673 }
674
675 return result;
676}

726template<bool replaceInvalidSequences>
727String fromUTF8Impl(const LChar* stringStart, size_t length)
728{
729 if (length > StringImplShape::MaxLength)
730 CRASH();
731
732 if (!stringStart)
733 return String();
734
735 if (!length)
736 return emptyString();
737
738 if (charactersAreAllASCII(stringStart, length))
739 return StringImpl::create(stringStart, length);
740
741 Vector<UChar, 1024> buffer(length);
742 UChar* bufferStart = buffer.data();
743
744 UChar* bufferCurrent = bufferStart;
745 const char* stringCurrent = reinterpret_cast<const char*>(stringStart);
746 constexpr auto function = replaceInvalidSequences ? convertUTF8ToUTF16ReplacingInvalidSequences : convertUTF8ToUTF16;
747 if (!function(stringCurrent, reinterpret_cast<const char*>(stringStart + length), &bufferCurrent, bufferCurrent + buffer.size(), nullptr))
748 return String();
749
750 unsigned utf16Length = bufferCurrent - bufferStart;
751 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(utf16Length <= length);
752 return StringImpl::create(bufferStart, utf16Length);
753}

796template<typename CharacterType, TrailingJunkPolicy policy>
797static inline double toDoubleType(const CharacterType* data, size_t length, bool* ok, size_t& parsedLength)
798{
799 size_t leadingSpacesLength = 0;
800 while (leadingSpacesLength < length && isASCIISpace(data[leadingSpacesLength]))
801 ++leadingSpacesLength;
802
803 double number = parseDouble(data + leadingSpacesLength, length - leadingSpacesLength, parsedLength);
804 if (!parsedLength) {
805 if (ok)
806 *ok = false;
807 return 0.0;
808 }
809
810 parsedLength += leadingSpacesLength;
811 if (ok)
812 *ok = policy == TrailingJunkPolicy::Allow || parsedLength == length;
813 return number;
814}
