38 DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StringWrapperCFAllocator);

40 static StringImpl* currentString;
41
42 static const void* retain(const void* info)
43 {
44 return info;
45 }
46
47 NO_RETURN_DUE_TO_ASSERT
48 static void release(const void*)
49 {
50 ASSERT_NOT_REACHED();
51 }
52
53 static CFStringRef copyDescription(const void*)
54 {
55 return CFSTR("WTF::String-based allocator");
56 }
57
58 static void* allocate(CFIndex size, CFOptionFlags, void*)
59 {
60 StringImpl* underlyingString = nullptr;
61 if (isMainThread()) {
62 underlyingString = currentString;
63 if (underlyingString) {
64 currentString = nullptr;
65 underlyingString->ref(); // Balanced by call to deref in deallocate below.
66 }
67 }
68 StringImpl** header = static_cast<StringImpl**>(StringWrapperCFAllocatorMalloc::malloc(sizeof(StringImpl*) + size));
69 *header = underlyingString;
70 return header + 1;
71 }
72
73 static void* reallocate(void* pointer, CFIndex newSize, CFOptionFlags, void*)
74 {
75 size_t newAllocationSize = sizeof(StringImpl*) + newSize;
76 StringImpl** header = static_cast<StringImpl**>(pointer) - 1;
77 ASSERT(!*header);
78 header = static_cast<StringImpl**>(StringWrapperCFAllocatorMalloc::realloc(header, newAllocationSize));
79 return header + 1;
80 }
81
82 static void deallocate(void* pointer, void*)
83 {
84 StringImpl** header = static_cast<StringImpl**>(pointer) - 1;
85 if (!*header)
86 StringWrapperCFAllocatorMalloc::free(header);
87 else {
88 ensureOnMainThread([header] {
89 StringImpl* underlyingString = *header;
90 ASSERT(underlyingString);
91 underlyingString->deref(); // Balanced by call to ref in allocate above.
92 StringWrapperCFAllocatorMalloc::free(header);
93 });
94 }
95 }
96
97 static CFIndex preferredSize(CFIndex size, CFOptionFlags, void*)
98 {
99 // FIXME: If FastMalloc provided a "good size" callback, we'd want to use it here.
100 // Note that this optimization would help performance for strings created with the
101 // allocator that are mutable, and those typically are only created by callers who
102 // make a new string using the old string's allocator, such as some of the call
103 // sites in CFURL.
104 return size;
105 }
106
107 static CFAllocatorRef allocator()
108 {
109 static auto allocator = makeNeverDestroyed([] {
110 CFAllocatorContext context = { 0, nullptr, retain, release, copyDescription, allocate, reallocate, deallocate, preferredSize };
111 return adoptCF(CFAllocatorCreate(nullptr, &context));
112 }());
113 return allocator.get().get();
114 }
115
116}

118RetainPtr<CFStringRef> StringImpl::createCFString()
119{
120 if (!m_length || !isMainThread()) {
121 if (is8Bit())
122 return adoptCF(CFStringCreateWithBytes(nullptr, characters8(), m_length, kCFStringEncodingISOLatin1, false));
123 return adoptCF(CFStringCreateWithCharacters(nullptr, reinterpret_cast<const UniChar*>(characters16()), m_length));
124 }
125 CFAllocatorRef allocator = StringWrapperCFAllocator::allocator();
126
127 // Put pointer to the StringImpl in a global so the allocator can store it with the CFString.
128 ASSERT(!StringWrapperCFAllocator::currentString);
129 StringWrapperCFAllocator::currentString = this;
130
131 RetainPtr<CFStringRef> string;
132 if (is8Bit())
133 string = adoptCF(CFStringCreateWithBytesNoCopy(allocator, characters8(), m_length, kCFStringEncodingISOLatin1, false, kCFAllocatorNull));
134 else
135 string = adoptCF(CFStringCreateWithCharactersNoCopy(allocator, reinterpret_cast<const UniChar*>(characters16()), m_length, kCFAllocatorNull));
136 // CoreFoundation might not have to allocate anything, we clear currentString in case we did not execute allocate().
137 StringWrapperCFAllocator::currentString = nullptr;
138
139 return string;
140}

142// On StringImpl creation we could check if the allocator is the StringWrapperCFAllocator.
143// If it is, then we could find the original StringImpl and just return that. But to
144// do that we'd have to compute the offset from CFStringRef to the allocated block;
145// the CFStringRef is *not* at the start of an allocated block. Testing shows 1000x
146// more calls to createCFString than calls to the create functions with the appropriate
147// allocator, so it's probably not urgent optimize that case.
148
149}
