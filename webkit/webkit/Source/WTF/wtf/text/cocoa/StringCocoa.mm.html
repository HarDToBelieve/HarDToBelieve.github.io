33RetainPtr<id> makeNSArrayElement(const String& vectorElement)
34{
35 return adoptNS((__bridge_transfer id)vectorElement.createCFString().leakRef());
36}
37
38std::optional<String> makeVectorElement(const String*, id arrayElement)
39{
40 if (![arrayElement isKindOfClass:NSString.class])
41 return std::nullopt;
42 return { { arrayElement } };
43}
