30// Buffer sized to hold ASCII locale ID strings up to 32 characters long.
31using LocaleIDBuffer = std::array<char, 33>;
32
33static Variant<TextBreakIteratorICU, TextBreakIteratorPlatform> mapModeToBackingIterator(StringView string, TextBreakIterator::Mode mode, const AtomString& locale)
34{
35 switch (mode) {
36 case TextBreakIterator::Mode::Line:
37 return TextBreakIteratorICU(string, TextBreakIteratorICU::Mode::Line, locale.string().utf8().data());
38 case TextBreakIterator::Mode::Caret:
39 return TextBreakIteratorCF(string, TextBreakIteratorCF::Mode::Caret);
40 case TextBreakIterator::Mode::Delete:
41 return TextBreakIteratorCF(string, TextBreakIteratorCF::Mode::Delete);
42 }
43}
44
45TextBreakIterator::TextBreakIterator(StringView string, Mode mode, const AtomString& locale)
46 : m_backing(mapModeToBackingIterator(string, mode, locale))
47 , m_mode(mode)
48 , m_locale(locale)
49{
50}

52static RetainPtr<CFStringRef> textBreakLocalePreference()
53{
54 auto locale = adoptCF(CFPreferencesCopyValue(CFSTR("AppleTextBreakLocale"),
55 kCFPreferencesAnyApplication, kCFPreferencesCurrentUser, kCFPreferencesAnyHost));
56 if (!locale || CFGetTypeID(locale.get()) != CFStringGetTypeID())
57 return nullptr;
58 return static_cast<CFStringRef>(locale.get());
59}
60
61static RetainPtr<CFStringRef> topLanguagePreference()
62{
63 auto languagesArray = adoptCF(CFLocaleCopyPreferredLanguages());
64 if (!languagesArray || !CFArrayGetCount(languagesArray.get()))
65 return nullptr;
66 return static_cast<CFStringRef>(CFArrayGetValueAtIndex(languagesArray.get(), 0));
67}
68
69static LocaleIDBuffer localeIDInBuffer(CFStringRef string)
70{
71 // Empty string means "root locale", and is what we use if we can't get a preference.
72 LocaleIDBuffer buffer;
73 if (!string || !CFStringGetCString(string, buffer.data(), buffer.size(), kCFStringEncodingASCII))
74 buffer.front() = '\0';
75 return buffer;
76}
77
78const char* currentSearchLocaleID()
79{
80 static const auto buffer = localeIDInBuffer(topLanguagePreference().get());
81 return buffer.data();
82}
83
84static RetainPtr<CFStringRef> textBreakLocale()
85{
86 // If there is no text break locale, use the top language preference.
87 auto locale = textBreakLocalePreference();
88 if (!locale)
89 return topLanguagePreference();
90 if (auto canonicalLocale = adoptCF(CFLocaleCreateCanonicalLanguageIdentifierFromString(kCFAllocatorDefault, locale.get())))
91 return canonicalLocale;
92 return locale;
93}
94
95const char* currentTextBreakLocaleID()
96{
97 static const auto buffer = localeIDInBuffer(textBreakLocale().get());
98 return buffer.data();
99}
