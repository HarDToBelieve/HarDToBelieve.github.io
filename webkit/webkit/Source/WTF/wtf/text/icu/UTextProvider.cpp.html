34// Relocate pointer from source into destination as required.
35static inline void fixPointer(const UText* source, UText* destination, const void*& pointer)
36{
37 if (pointer >= source->pExtra && pointer < static_cast<char*>(source->pExtra) + source->extraSize) {
38 // Pointer references source extra buffer.
39 pointer = static_cast<char*>(destination->pExtra) + (static_cast<const char*>(pointer) - static_cast<const char*>(source->pExtra));
40 } else if (pointer >= source && pointer < reinterpret_cast<const char*>(source) + source->sizeOfStruct) {
41 // Pointer references source text structure, but not source extra buffer.
42 pointer = reinterpret_cast<char*>(destination) + (static_cast<const char*>(pointer) - reinterpret_cast<const char*>(source));
43 }
44}
45
46UText* uTextCloneImpl(UText* destination, const UText* source, UBool deep, UErrorCode* status)
47{
48 ASSERT_UNUSED(deep, !deep);
49 if (U_FAILURE(*status))
50 return nullptr;
51 int32_t extraSize = source->extraSize;
52 destination = utext_setup(destination, extraSize, status);
53 if (U_FAILURE(*status))
54 return destination;
55 void* extraNew = destination->pExtra;
56 int32_t flags = destination->flags;
57 int sizeToCopy = std::min(source->sizeOfStruct, destination->sizeOfStruct);
58 memcpy(destination, source, sizeToCopy);
59 destination->pExtra = extraNew;
60 destination->flags = flags;
61 memcpy(destination->pExtra, source->pExtra, extraSize);
62 fixPointer(source, destination, destination->context);
63 fixPointer(source, destination, destination->p);
64 fixPointer(source, destination, destination->q);
65 ASSERT(!destination->r);
66 const void* chunkContents = static_cast<const void*>(destination->chunkContents);
67 fixPointer(source, destination, chunkContents);
68 destination->chunkContents = static_cast<const UChar*>(chunkContents);
69 return destination;
70}
