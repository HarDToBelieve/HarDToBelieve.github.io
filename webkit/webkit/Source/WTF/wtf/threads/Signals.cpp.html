66void SignalHandlers::add(Signal signal, SignalHandler&& handler)
67{
68 Config::AssertNotFrozenScope assertScope;
69 static Lock lock;
70 Locker locker { lock };
71
72 size_t signalIndex = static_cast<size_t>(signal);
73 size_t nextFree = numberOfHandlers[signalIndex];
74#if HAVE(MACH_EXCEPTIONS)
75 if (signal != Signal::Usr)
76 addedExceptions |= toMachMask(signal);
77#endif
78 RELEASE_ASSERT(nextFree < maxNumberOfHandlers);
79 SignalHandlerMemory* memory = &handlers[signalIndex][nextFree];
80 new (memory) SignalHandler(WTFMove(handler));
81
82 // We deliberately do not want to increment the count until after we've
83 // fully initialized the memory. This way, forEachHandler() won't see a
84 // partially initialized handler.
85 storeStoreFence();
86 numberOfHandlers[signalIndex]++;
87 loadLoadFence();
88}

90template<typename Func>
91inline void SignalHandlers::forEachHandler(Signal signal, const Func& func) const
92{
93 size_t signalIndex = static_cast<size_t>(signal);
94 size_t handlerIndex = numberOfHandlers[signalIndex];
95 while (handlerIndex--) {
96 auto* memory = const_cast<SignalHandlerMemory*>(&handlers[signalIndex][handlerIndex]);
97 const SignalHandler& handler = *bitwise_cast<SignalHandler*>(memory);
98 func(handler);
99 }
100}

108static constexpr size_t maxMessageSize = 1 * KB;
109
110void startMachExceptionHandlerThread()
111{
112 static std::once_flag once;
113 std::call_once(once, [] {
114 Config::AssertNotFrozenScope assertScope;
115 SignalHandlers& handlers = g_wtfConfig.signalHandlers;
116 kern_return_t kr = mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &handlers.exceptionPort);
117 RELEASE_ASSERT(kr == KERN_SUCCESS);
118 kr = mach_port_insert_right(mach_task_self(), handlers.exceptionPort, handlers.exceptionPort, MACH_MSG_TYPE_MAKE_SEND);
119 RELEASE_ASSERT(kr == KERN_SUCCESS);
120
121 dispatch_source_t source = dispatch_source_create(
122 DISPATCH_SOURCE_TYPE_MACH_RECV, handlers.exceptionPort, 0, DISPATCH_TARGET_QUEUE_DEFAULT);
123 RELEASE_ASSERT(source);
124
125 dispatch_source_set_event_handler(source, ^{
126 UNUSED_PARAM(source); // Capture a pointer to source in user space to silence the leaks tool.
127
128 kern_return_t kr = mach_msg_server_once(
129 mach_exc_server, maxMessageSize, handlers.exceptionPort, MACH_MSG_TIMEOUT_NONE);
130 RELEASE_ASSERT(kr == KERN_SUCCESS);
131 });
132
133 // No need for a cancel handler because we never destroy exceptionPort.
134
135 dispatch_resume(source);
136 });
137}

139static Signal fromMachException(exception_type_t type)
140{
141 switch (type) {
142 case EXC_BAD_ACCESS: return Signal::AccessFault;
143 case EXC_BAD_INSTRUCTION: return Signal::IllegalInstruction;
144 case EXC_ARITHMETIC: return Signal::FloatingPoint;
145 case EXC_BREAKPOINT: return Signal::Breakpoint;
146 default: break;
147 }
148 return Signal::Unknown;
149}
150
151static exception_mask_t toMachMask(Signal signal)
152{
153 switch (signal) {
154 case Signal::AccessFault: return EXC_MASK_BAD_ACCESS;
155 case Signal::IllegalInstruction: return EXC_MASK_BAD_INSTRUCTION;
156 case Signal::FloatingPoint: return EXC_MASK_ARITHMETIC;
157 case Signal::Breakpoint: return EXC_MASK_BREAKPOINT;
158 default: break;
159 }
160 RELEASE_ASSERT_NOT_REACHED();
161}

163#if CPU(ARM64E) && OS(DARWIN)
164inline ptrauth_generic_signature_t hashThreadState(const thread_state_t source)
165{
166 constexpr size_t threadStatePCPointerIndex = (offsetof(arm_unified_thread_state, ts_64) + offsetof(arm_thread_state64_t, __opaque_pc)) / sizeof(uintptr_t);
167 constexpr size_t threadStateSizeInPointers = sizeof(arm_unified_thread_state) / sizeof(uintptr_t);
168
169 ptrauth_generic_signature_t hash = 0;
170
171 hash = ptrauth_sign_generic_data(hash, mach_thread_self());
172
173 const uintptr_t* srcPtr = reinterpret_cast<const uintptr_t*>(source);
174
175 for (size_t i = 0; i < threadStateSizeInPointers; ++i) {
176 if (i != threadStatePCPointerIndex)
177 hash = ptrauth_sign_generic_data(srcPtr[i], hash);
178 }
179
180 return hash;
181}
182#endif

201
202kern_return_t catch_mach_exception_raise_state(
203 mach_port_t port,
204 exception_type_t exceptionType,
205 const mach_exception_data_t exceptionData,
206 mach_msg_type_number_t dataCount,
207 int* stateFlavor,
208 const thread_state_t inState,
209 mach_msg_type_number_t inStateCount,
210 thread_state_t outState,
211 mach_msg_type_number_t* outStateCount)
212{
213 SignalHandlers& handlers = g_wtfConfig.signalHandlers;
214 RELEASE_ASSERT(port == handlers.exceptionPort);
215 // If we wanted to distinguish between SIGBUS and SIGSEGV for EXC_BAD_ACCESS on Darwin we could do:
216 // if (exceptionData[0] == KERN_INVALID_ADDRESS)
217 // signal = SIGSEGV;
218 // else
219 // signal = SIGBUS;
220 Signal signal = fromMachException(exceptionType);
221 RELEASE_ASSERT(signal != Signal::Unknown);
222
223#if CPU(ARM64E) && OS(DARWIN)
224 ptrauth_generic_signature_t inStateHash = hashThreadState(inState);
225#endif
226
227 memcpy(outState, inState, inStateCount * sizeof(inState[0]));
228
229#if CPU(X86_64)
230 RELEASE_ASSERT(*stateFlavor == x86_THREAD_STATE);
231 PlatformRegisters& registers = reinterpret_cast<x86_thread_state_t*>(outState)->uts.ts64;
232#elif CPU(X86)
233 RELEASE_ASSERT(*stateFlavor == x86_THREAD_STATE);
234 PlatformRegisters& registers = reinterpret_cast<x86_thread_state_t*>(outState)->uts.ts32;
235#elif CPU(ARM64)
236 RELEASE_ASSERT(*stateFlavor == ARM_THREAD_STATE);
237 PlatformRegisters& registers = reinterpret_cast<arm_unified_thread_state*>(outState)->ts_64;
238#elif CPU(ARM)
239 RELEASE_ASSERT(*stateFlavor == ARM_THREAD_STATE);
240 PlatformRegisters& registers = reinterpret_cast<arm_unified_thread_state*>(outState)->ts_32;
241#endif
242
243 SigInfo info;
244 if (signal == Signal::AccessFault) {
245 ASSERT_UNUSED(dataCount, dataCount == 2);
246 info.faultingAddress = reinterpret_cast<void*>(exceptionData[1]);
247#if CPU(ADDRESS64)
248 // If the faulting address is out of the range of any valid memory, we would
249 // not have any reason to handle it. Just let the default handler take care of it.
250 static constexpr unsigned validAddressBits = OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH);
251 static constexpr uintptr_t invalidAddressMask = ~((1ull << validAddressBits) - 1);
252 if (bitwise_cast<uintptr_t>(info.faultingAddress) & invalidAddressMask)
253 return KERN_FAILURE;
254#endif
255 }
256
257 bool didHandle = false;
258 handlers.forEachHandler(signal, [&] (const SignalHandler& handler) {
259 SignalAction handlerResult = handler(signal, info, registers);
260 didHandle |= handlerResult == SignalAction::Handled;
261 });
262
263 if (didHandle) {
264#if CPU(ARM64E) && OS(DARWIN)
265 RELEASE_ASSERT(inStateHash == hashThreadState(outState));
266#endif
267 *outStateCount = inStateCount;
268 return KERN_SUCCESS;
269 }
270
271 return KERN_FAILURE;
272}

282static exception_mask_t activeExceptions;
283inline void setExceptionPorts(const AbstractLocker& threadGroupLocker, Thread& thread)
284{
285 UNUSED_PARAM(threadGroupLocker);
286 SignalHandlers& handlers = g_wtfConfig.signalHandlers;
287 kern_return_t result = thread_set_exception_ports(thread.machThread(), handlers.addedExceptions &activeExceptions, handlers.exceptionPort, EXCEPTION_STATE | MACH_EXCEPTION_CODES, MACHINE_THREAD_STATE);
288 if (result != KERN_SUCCESS) {
289 dataLogLn("thread set port failed due to ", mach_error_string(result));
290 CRASH();
291 }
292}

314inline size_t offsetForSystemSignal(int sig)
315{
316 Signal signal = fromSystemSignal(sig);
317 return static_cast<size_t>(signal) + (sig == SIGBUS);
318}

320static void jscSignalHandler(int, siginfo_t*, void*);
321
322void addSignalHandler(Signal signal, SignalHandler&& handler)
323{
324 Config::AssertNotFrozenScope assertScope;
325 SignalHandlers& handlers = g_wtfConfig.signalHandlers;
326 ASSERT(signal < Signal::Unknown);
327 ASSERT(!handlers.useMach || signal != Signal::Usr);
328
329#if HAVE(MACH_EXCEPTIONS)
330 if (handlers.useMach)
331 startMachExceptionHandlerThread();
332#endif
333
334 static std::once_flag initializeOnceFlags[static_cast<size_t>(Signal::NumberOfSignals)];
335 std::call_once(initializeOnceFlags[static_cast<size_t>(signal)], [&] {
336 Config::AssertNotFrozenScope assertScope;
337 if (!handlers.useMach) {
338 struct sigaction action;
339 action.sa_sigaction = jscSignalHandler;
340 auto result = sigfillset(&action.sa_mask);
341 RELEASE_ASSERT(!result);
342 // Do not block this signal since it is used on non-Darwin systems to suspend and resume threads.
343 RELEASE_ASSERT(g_wtfConfig.isThreadSuspendResumeSignalConfigured);
344 result = sigdelset(&action.sa_mask, g_wtfConfig.sigThreadSuspendResume);
345 RELEASE_ASSERT(!result);
346 action.sa_flags = SA_SIGINFO;
347 auto systemSignals = toSystemSignal(signal);
348 result = sigaction(std::get<0>(systemSignals), &action, &handlers.oldActions[offsetForSystemSignal(std::get<0>(systemSignals))]);
349 if (std::get<1>(systemSignals))
350 result |= sigaction(*std::get<1>(systemSignals), &action, &handlers.oldActions[offsetForSystemSignal(*std::get<1>(systemSignals))]);
351 RELEASE_ASSERT(!result);
352 }
353 });
354
355 handlers.add(signal, WTFMove(handler));
356}

358void activateSignalHandlersFor(Signal signal)
359{
360 UNUSED_PARAM(signal);
361#if HAVE(MACH_EXCEPTIONS)
362 const SignalHandlers& handlers = g_wtfConfig.signalHandlers;
363 ASSERT(signal < Signal::Unknown);
364 ASSERT(!handlers.useMach || signal != Signal::Usr);
365
366 Locker locker { activeThreads().getLock() };
367 if (handlers.useMach) {
368 activeExceptions |= toMachMask(signal);
369
370 for (auto& thread : activeThreads().threads(locker))
371 setExceptionPorts(locker, thread.get());
372 }
373#endif
374}

376void jscSignalHandler(int sig, siginfo_t* info, void* ucontext)
377{
378 Signal signal = fromSystemSignal(sig);
379 SignalHandlers& handlers = g_wtfConfig.signalHandlers;
380
381 auto restoreDefault = [&] {
382 struct sigaction defaultAction;
383 defaultAction.sa_handler = SIG_DFL;
384 sigfillset(&defaultAction.sa_mask);
385 defaultAction.sa_flags = 0;
386 auto result = sigaction(sig, &defaultAction, nullptr);
387 dataLogLnIf(result == -1, "Unable to restore the default handler while proccessing signal ", sig, " the process is probably deadlocked. (errno: ", strerror(errno), ")");
388 };
389
390 // This shouldn't happen but we might as well be careful.
391 if (signal == Signal::Unknown) {
392 dataLogLn("We somehow got called for an unknown signal ", sig, ", help.");
393 restoreDefault();
394 return;
395 }
396
397 SigInfo sigInfo;
398 if (signal == Signal::AccessFault)
399 sigInfo.faultingAddress = info->si_addr;
400
401#if HAVE(MACHINE_CONTEXT)
402 PlatformRegisters& registers = registersFromUContext(reinterpret_cast<ucontext_t*>(ucontext));
403#else
404 PlatformRegisters registers { };
405#endif
406
407 bool didHandle = false;
408 bool restoreDefaultHandler = false;
409 handlers.forEachHandler(signal, [&] (const SignalHandler& handler) {
410 switch (handler(signal, sigInfo, registers)) {
411 case SignalAction::Handled:
412 didHandle = true;
413 break;
414 case SignalAction::ForceDefault:
415 restoreDefaultHandler = true;
416 break;
417 default:
418 break;
419 }
420 });
421
422 if (restoreDefaultHandler) {
423 restoreDefault();
424 return;
425 }
426
427 unsigned oldActionIndex = static_cast<size_t>(signal) + (sig == SIGBUS);
428 struct sigaction& oldAction = handlers.oldActions[static_cast<size_t>(oldActionIndex)];
429 if (signal == Signal::Usr) {
430 if (oldAction.sa_sigaction)
431 oldAction.sa_sigaction(sig, info, ucontext);
432 return;
433 }
434
435 if (!didHandle) {
436 if (oldAction.sa_sigaction) {
437 oldAction.sa_sigaction(sig, info, ucontext);
438 return;
439 }
440
441 restoreDefault();
442 return;
443 }
444}

446void SignalHandlers::initialize()
447{
448#if HAVE(MACH_EXCEPTIONS)
449 // In production configurations, this does not matter because signal handler
450 // installations will always trigger this initialization. However, in debugging
451 // configurations, we may end up disabling the use of all signal handlers but
452 // we still need this to be initialized. Hence, we need to initialize it
453 // eagerly to ensure that it is done before we freeze the WTF::Config.
454 activeThreads();
455#endif
456}
