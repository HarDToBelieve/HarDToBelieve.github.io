37bool convertLatin1ToUTF8(const LChar** sourceStart, const LChar* sourceEnd, char** targetStart, char* targetEnd)
38{
39 const LChar* source;
40 char* target = *targetStart;
41 int i = 0;
42 for (source = *sourceStart; source < sourceEnd; ++source) {
43 UBool sawError = false;
44 // Work around bug in either Windows compiler or old version of ICU, where passing a uint8_t to
45 // U8_APPEND warns, by converting from uint8_t to a wider type.
46 UChar32 character = *source;
47 U8_APPEND(reinterpret_cast<uint8_t*>(target), i, targetEnd - *targetStart, character, sawError);
48 if (sawError)
49 return false;
50 }
51 *sourceStart = source;
52 *targetStart = target + i;
53 return true;
54}

56ConversionResult convertUTF16ToUTF8(const UChar** sourceStart, const UChar* sourceEnd, char** targetStart, char* targetEnd, bool strict)
57{
58 ConversionResult result = ConversionOK;
59 const UChar* source = *sourceStart;
60 char* target = *targetStart;
61 UBool sawError = false;
62 int i = 0;
63 while (source < sourceEnd) {
64 UChar32 ch;
65 int j = 0;
66 U16_NEXT(source, j, sourceEnd - source, ch);
67 if (U_IS_SURROGATE(ch)) {
68 if (source + j == sourceEnd && U_IS_SURROGATE_LEAD(ch)) {
69 result = SourceExhausted;
70 break;
71 }
72 if (strict) {
73 result = SourceIllegal;
74 break;
75 }
76 ch = replacementCharacter;
77 }
78 U8_APPEND(reinterpret_cast<uint8_t*>(target), i, targetEnd - target, ch, sawError);
79 if (sawError) {
80 result = TargetExhausted;
81 break;
82 }
83 source += j;
84 }
85 *sourceStart = source;
86 *targetStart = target + i;
87 return result;
88}

90template<bool replaceInvalidSequences>
91bool convertUTF8ToUTF16Impl(const char* source, const char* sourceEnd, UChar** targetStart, UChar* targetEnd, bool* sourceAllASCII)
92{
93 RELEASE_ASSERT(sourceEnd - source <= std::numeric_limits<int>::max());
94 UBool error = false;
95 UChar* target = *targetStart;
96 RELEASE_ASSERT(targetEnd - target <= std::numeric_limits<int>::max());
97 UChar32 orAllData = 0;
98 int targetOffset = 0;
99 for (int sourceOffset = 0; sourceOffset < sourceEnd - source; ) {
100 UChar32 character;
101 if constexpr (replaceInvalidSequences) {
102 U8_NEXT_OR_FFFD(reinterpret_cast<const uint8_t*>(source), sourceOffset, sourceEnd - source, character);
103 } else {
104 U8_NEXT(reinterpret_cast<const uint8_t*>(source), sourceOffset, sourceEnd - source, character);
105 }
106 if (character < 0)
107 return false;
108 U16_APPEND(target, targetOffset, targetEnd - target, character, error);
109 if (error)
110 return false;
111 orAllData |= character;
112 }
113 RELEASE_ASSERT(target + targetOffset <= targetEnd);
114 *targetStart = target + targetOffset;
115 if (sourceAllASCII)
116 *sourceAllASCII = isASCII(orAllData);
117 return true;
118}

120bool convertUTF8ToUTF16(const char* source, const char* sourceEnd, UChar** targetStart, UChar* targetEnd, bool* sourceAllASCII)
121{
122 return convertUTF8ToUTF16Impl<false>(source, sourceEnd, targetStart, targetEnd, sourceAllASCII);
123}

125bool convertUTF8ToUTF16ReplacingInvalidSequences(const char* source, const char* sourceEnd, UChar** targetStart, UChar* targetEnd, bool* sourceAllASCII)

127 return convertUTF8ToUTF16Impl<true>(source, sourceEnd, targetStart, targetEnd, sourceAllASCII);

130unsigned calculateStringHashAndLengthFromUTF8MaskingTop8Bits(const char* data, const char* dataEnd, unsigned& dataLength, unsigned& utf16Length)
131{
132 StringHasher stringHasher;
133 utf16Length = 0;
134
135 int inputOffset = 0;
136 int inputLength = dataEnd - data;
137 while (inputOffset < inputLength) {
138 UChar32 character;
139 U8_NEXT(reinterpret_cast<const uint8_t*>(data), inputOffset, inputLength, character);
140 if (character < 0)
141 return 0;
142
143 if (U_IS_BMP(character)) {
144 ASSERT(!U_IS_SURROGATE(character));
145 stringHasher.addCharacter(character);
146 utf16Length++;
147 } else {
148 ASSERT(U_IS_SUPPLEMENTARY(character));
149 stringHasher.addCharacters(U16_LEAD(character), U16_TRAIL(character));
150 utf16Length += 2;
151 }
152 }
153
154 dataLength = inputOffset;
155 return stringHasher.hashWithTop8BitsMasked();
156}

158bool equalUTF16WithUTF8(const UChar* a, const char* b, const char* bEnd)
159{
160 while (b < bEnd) {
161 int offset = 0;
162 UChar32 character;
163 U8_NEXT(reinterpret_cast<const uint8_t*>(b), offset, bEnd - b, character);
164 if (character < 0)
165 return false;
166 b += offset;
167
168 if (U_IS_BMP(character)) {
169 ASSERT(!U_IS_SURROGATE(character));
170 if (*a++ != character)
171 return false;
172 } else {
173 ASSERT(U_IS_SUPPLEMENTARY(character));
174 if (*a++ != U16_LEAD(character))
175 return false;
176 if (*a++ != U16_TRAIL(character))
177 return false;
178 }
179 }
180
181 return true;
182}

184bool equalLatin1WithUTF8(const LChar* a, const char* b, const char* bEnd)
185{
186 while (b < bEnd) {
187 if (isASCII(*a) || isASCII(*b)) {
188 if (*a++ != *b++)
189 return false;
190 continue;
191 }
192
193 if (b + 1 == bEnd)
194 return false;
195
196 if ((b[0] & 0xE0) != 0xC0 || (b[1] & 0xC0) != 0x80)
197 return false;
198
199 LChar character = ((b[0] & 0x1F) << 6) | (b[1] & 0x3F);
200
201 b += 2;
202
203 if (*a++ != character)
204 return false;
205 }
206
207 return true;
208}
