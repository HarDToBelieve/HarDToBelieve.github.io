34namespace WebCore {
35
36WTF_MAKE_ISO_ALLOCATED_IMPL(WebKitPlaybackTargetAvailabilityEvent);
37
38static const AtomString& stringForPlaybackTargetAvailability(bool available)
39{
40 static MainThreadNeverDestroyed<const AtomString> availableString("available", AtomString::ConstructFromLiteral);
41 static MainThreadNeverDestroyed<const AtomString> notAvailableString("not-available", AtomString::ConstructFromLiteral);
42
43 return available ? availableString : notAvailableString;
44}
45
46WebKitPlaybackTargetAvailabilityEvent::WebKitPlaybackTargetAvailabilityEvent(const AtomString& eventType, bool available)
47 : Event(eventType, CanBubble::No, IsCancelable::No)
48 , m_availability(stringForPlaybackTargetAvailability(available))
49{
50}
51
52WebKitPlaybackTargetAvailabilityEvent::WebKitPlaybackTargetAvailabilityEvent(const AtomString& eventType, const Init& initializer, IsTrusted isTrusted)
53 : Event(eventType, initializer, isTrusted)
54 , m_availability(initializer.availability)
55{
56}
