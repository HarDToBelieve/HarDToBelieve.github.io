31namespace WebCore {
32
33ExceptionOr<ApplePaySessionPaymentRequest::MerchantCapabilities> convertAndValidate(const Vector<ApplePayMerchantCapability>& merchantCapabilities)
34{
35 if (merchantCapabilities.isEmpty())
36 return Exception { TypeError, "At least one merchant capability must be provided." };
37
38 ApplePaySessionPaymentRequest::MerchantCapabilities result;
39
40 for (auto& merchantCapability : merchantCapabilities) {
41 switch (merchantCapability) {
42 case ApplePayMerchantCapability::Supports3DS:
43 result.supports3DS = true;
44 break;
45 case ApplePayMerchantCapability::SupportsEMV:
46 result.supportsEMV = true;
47 break;
48 case ApplePayMerchantCapability::SupportsCredit:
49 result.supportsCredit = true;
50 break;
51 case ApplePayMerchantCapability::SupportsDebit:
52 result.supportsDebit = true;
53 break;
54 }
55 }
56
57 return WTFMove(result);
58}
