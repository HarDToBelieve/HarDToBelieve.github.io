79static ExceptionOr<ApplePayLineItem> convertAndValidateTotal(ApplePayLineItem&& lineItem)
80{
81 if (!isValidDecimalMonetaryValue(lineItem.amount))
82 return Exception { TypeError, makeString("\"" + lineItem.amount, "\" is not a valid amount.") };
83
84 auto validatedTotal = PaymentRequestValidator::validateTotal(lineItem);
85 if (validatedTotal.hasException())
86 return validatedTotal.releaseException();
87
88 return WTFMove(lineItem);
89}
90
91static ExceptionOr<ApplePayLineItem> convertAndValidate(ApplePayLineItem&& lineItem)
92{
93 if (lineItem.type == ApplePayLineItem::Type::Pending) {
94 // It is OK for pending types to not have an amount.
95 lineItem.amount = nullString();
96 } else {
97 if (!isValidDecimalMonetaryValue(lineItem.amount))
98 return Exception { TypeError, makeString("\"" + lineItem.amount, "\" is not a valid amount.") };
99 }
100
101 return WTFMove(lineItem);
102}
103
104static ExceptionOr<Vector<ApplePayLineItem>> convertAndValidate(std::optional<Vector<ApplePayLineItem>>&& lineItems)
105{
106 Vector<ApplePayLineItem> result;
107 if (!lineItems)
108 return WTFMove(result);
109
110 result.reserveInitialCapacity(lineItems->size());
111
112 for (auto lineItem : lineItems.value()) {
113 auto convertedLineItem = convertAndValidate(WTFMove(lineItem));
114 if (convertedLineItem.hasException())
115 return convertedLineItem.releaseException();
116 result.uncheckedAppend(convertedLineItem.releaseReturnValue());
117 }
118
119 return WTFMove(result);
120}

145static ExceptionOr<ApplePaySessionPaymentRequest> convertAndValidate(Document& document, unsigned version, ApplePayPaymentRequest&& paymentRequest, const PaymentCoordinator& paymentCoordinator)
146{
147 auto convertedRequest = convertAndValidate(document, version, paymentRequest, paymentCoordinator);
148 if (convertedRequest.hasException())
149 return convertedRequest.releaseException();
150
151 auto result = convertedRequest.releaseReturnValue();
152 result.setRequester(ApplePaySessionPaymentRequest::Requester::ApplePayJS);
153 result.setCurrencyCode(paymentRequest.currencyCode);
154
155 auto total = convertAndValidateTotal(WTFMove(paymentRequest.total));
156 if (total.hasException())
157 return total.releaseException();
158 result.setTotal(total.releaseReturnValue());
159
160 auto lineItems = convertAndValidate(WTFMove(paymentRequest.lineItems));
161 if (lineItems.hasException())
162 return lineItems.releaseException();
163 result.setLineItems(lineItems.releaseReturnValue());
164
165 result.setShippingType(paymentRequest.shippingType);
166
167 if (paymentRequest.shippingMethods) {
168 auto shippingMethods = convertAndValidate(WTFMove(*paymentRequest.shippingMethods));
169 if (shippingMethods.hasException())
170 return shippingMethods.releaseException();
171 result.setShippingMethods(shippingMethods.releaseReturnValue());
172 }
173
174#if defined(ApplePaySessionAdditions_convertAndValidate_request)
175 ApplePaySessionAdditions_convertAndValidate_request
176#endif
177
178 // FIXME: Merge this validation into the validation we are doing above.
179 auto validatedPaymentRequest = PaymentRequestValidator::validate(result);
180 if (validatedPaymentRequest.hasException())
181 return validatedPaymentRequest.releaseException();
182
183 return WTFMove(result);
184}

186static ExceptionOr<PaymentAuthorizationResult> convertAndValidate(ApplePayPaymentAuthorizationResult&& result)
187{
188 PaymentAuthorizationResult convertedResult;
189
190 switch (result.status) {
191 case ApplePaySession::STATUS_SUCCESS:
192 convertedResult.status = PaymentAuthorizationStatus::Success;
193 break;
194
195 case ApplePaySession::STATUS_FAILURE:
196 convertedResult.status = PaymentAuthorizationStatus::Failure;
197 break;
198
199 case ApplePaySession::STATUS_INVALID_BILLING_POSTAL_ADDRESS:
200 convertedResult.status = PaymentAuthorizationStatus::Failure;
201 convertedResult.errors.append(ApplePayError::create(ApplePayErrorCode::BillingContactInvalid, std::nullopt, nullString()));
202 break;
203
204 case ApplePaySession::STATUS_INVALID_SHIPPING_POSTAL_ADDRESS:
205 convertedResult.status = PaymentAuthorizationStatus::Failure;
206 convertedResult.errors.append(ApplePayError::create(ApplePayErrorCode::ShippingContactInvalid, ApplePayErrorContactField::PostalAddress, nullString()));
207 break;
208
209 case ApplePaySession::STATUS_INVALID_SHIPPING_CONTACT:
210 convertedResult.status = PaymentAuthorizationStatus::Failure;
211 convertedResult.errors.append(ApplePayError::create(ApplePayErrorCode::ShippingContactInvalid, std::nullopt, nullString()));
212 break;
213
214 case ApplePaySession::STATUS_PIN_REQUIRED:
215 convertedResult.status = PaymentAuthorizationStatus::PINRequired;
216 break;
217
218 case ApplePaySession::STATUS_PIN_INCORRECT:
219 convertedResult.status = PaymentAuthorizationStatus::PINIncorrect;
220 break;
221
222 case ApplePaySession::STATUS_PIN_LOCKOUT:
223 convertedResult.status = PaymentAuthorizationStatus::PINLockout;
224 break;
225
226 default:
227 return Exception { InvalidAccessError };
228 }
229
230 convertedResult.errors.appendVector(WTFMove(result.errors));
231
232 return WTFMove(convertedResult);
233}

309ExceptionOr<Ref<ApplePaySession>> ApplePaySession::create(Document& document, unsigned version, ApplePayPaymentRequest&& paymentRequest)
310{
311 auto canCall = canCreateSession(document);
312 if (canCall.hasException())
313 return canCall.releaseException();
314
315 if (!UserGestureIndicator::processingUserGesture())
316 return Exception { InvalidAccessError, "Must create a new ApplePaySession from a user gesture handler." };
317
318 if (!document.page())
319 return Exception { InvalidAccessError, "Frame is detached" };
320
321 auto convertedPaymentRequest = convertAndValidate(document, version, WTFMove(paymentRequest), document.page()->paymentCoordinator());
322 if (convertedPaymentRequest.hasException())
323 return convertedPaymentRequest.releaseException();
324
325 return adoptRef(*new ApplePaySession(document, version, convertedPaymentRequest.releaseReturnValue()));
326}

377ExceptionOr<void> ApplePaySession::canMakePaymentsWithActiveCard(Document& document, const String& merchantIdentifier, Ref<DeferredPromise>&& passedPromise)
378{
379 auto canCall = canCreateSession(document);
380 if (canCall.hasException())
381 return canCall.releaseException();
382
383 RefPtr<DeferredPromise> promise(WTFMove(passedPromise));
384 if (!shouldDiscloseApplePayCapability(document)) {
385 auto* page = document.page();
386 if (!page)
387 return Exception { InvalidAccessError };
388
389 auto& paymentCoordinator = page->paymentCoordinator();
390 bool canMakePayments = paymentCoordinator.canMakePayments(document);
391
392 RunLoop::main().dispatch([promise, canMakePayments]() mutable {
393 promise->resolve<IDLBoolean>(canMakePayments);
394 });
395 return { };
396 }
397
398 auto* page = document.page();
399 if (!page)
400 return Exception { InvalidAccessError };
401
402 auto& paymentCoordinator = page->paymentCoordinator();
403
404 paymentCoordinator.canMakePaymentsWithActiveCard(document, merchantIdentifier, [promise](bool canMakePayments) mutable {
405 promise->resolve<IDLBoolean>(canMakePayments);
406 });
407 return { };
408}

433ExceptionOr<void> ApplePaySession::begin(Document& document)
434{
435 if (!canBegin())
436 return Exception { InvalidAccessError, "Payment session is already active." };
437
438 if (paymentCoordinator().hasActiveSession())
439 return Exception { InvalidAccessError, "Page already has an active payment session." };
440
441 if (!paymentCoordinator().beginPaymentSession(document, *this, m_paymentRequest))
442 return Exception { InvalidAccessError, "There is already has an active payment session." };
443
444 m_state = State::Active;
445
446 return { };
447}

460ExceptionOr<void> ApplePaySession::completeMerchantValidation(JSC::JSGlobalObject& state, JSC::JSValue merchantSessionValue)
461{
462 if (!canCompleteMerchantValidation())
463 return Exception { InvalidAccessError };
464
465 if (!merchantSessionValue.isObject())
466 return Exception { TypeError };
467
468 auto& document = *downcast<Document>(scriptExecutionContext());
469 auto& window = *document.domWindow();
470
471 String errorMessage;
472 auto merchantSession = PaymentMerchantSession::fromJS(state, asObject(merchantSessionValue), errorMessage);
473 if (!merchantSession) {
474 window.printErrorMessage(errorMessage);
475 return Exception { InvalidAccessError };
476 }
477
478 m_merchantValidationState = MerchantValidationState::ValidationComplete;
479 paymentCoordinator().completeMerchantValidation(*merchantSession);
480
481 return { };
482}

548ExceptionOr<void> ApplePaySession::completePayment(ApplePayPaymentAuthorizationResult&& result)
549{
550 if (!canCompletePayment())
551 return Exception { InvalidAccessError };
552
553 auto convertedResultOrException = convertAndValidate(WTFMove(result));
554 if (convertedResultOrException.hasException())
555 return convertedResultOrException.releaseException();
556
557 auto&& convertedResult = convertedResultOrException.releaseReturnValue();
558 bool isFinalState = isFinalStateResult(convertedResult);
559
560 paymentCoordinator().completePaymentSession(WTFMove(convertedResult));
561
562 if (!isFinalState) {
563 m_state = State::Active;
564 return { };
565 }
566
567 m_state = State::Completed;
568
569 return { };
570}

572ExceptionOr<void> ApplePaySession::completeShippingMethodSelection(unsigned short status, ApplePayLineItem&& newTotal, Vector<ApplePayLineItem>&& newLineItems)
573{
574 ApplePayShippingMethodUpdate update;
575
576 switch (status) {
577 case ApplePaySession::STATUS_SUCCESS:
578 break;
579
580 case ApplePaySession::STATUS_FAILURE:
581 case ApplePaySession::STATUS_INVALID_BILLING_POSTAL_ADDRESS:
582 case ApplePaySession::STATUS_INVALID_SHIPPING_POSTAL_ADDRESS:
583 case ApplePaySession::STATUS_INVALID_SHIPPING_CONTACT:
584 case ApplePaySession::STATUS_PIN_REQUIRED:
585 case ApplePaySession::STATUS_PIN_INCORRECT:
586 case ApplePaySession::STATUS_PIN_LOCKOUT:
587 // This is a fatal error. Cancel the request.
588 m_state = State::CancelRequested;
589 paymentCoordinator().cancelPaymentSession();
590 return { };
591
592 default:
593 return Exception { InvalidAccessError };
594 }
595
596 update.newTotal = WTFMove(newTotal);
597 update.newLineItems = WTFMove(newLineItems);
598
599 return completeShippingMethodSelection(WTFMove(update));
600}

602ExceptionOr<void> ApplePaySession::completeShippingContactSelection(unsigned short status, Vector<ApplePayShippingMethod>&& newShippingMethods, ApplePayLineItem&& newTotal, Vector<ApplePayLineItem>&& newLineItems)
603{
604 ApplePayShippingContactUpdate update;
605
606 std::optional<ApplePayErrorCode> errorCode;
607 std::optional<ApplePayErrorContactField> contactField;
608
609 switch (status) {
610 case ApplePaySession::STATUS_SUCCESS:
611 break;
612
613 case ApplePaySession::STATUS_FAILURE:
614 case ApplePaySession::STATUS_PIN_REQUIRED:
615 case ApplePaySession::STATUS_PIN_INCORRECT:
616 case ApplePaySession::STATUS_PIN_LOCKOUT:
617 errorCode = ApplePayErrorCode::Unknown;
618 break;
619
620 case ApplePaySession::STATUS_INVALID_BILLING_POSTAL_ADDRESS:
621 errorCode = ApplePayErrorCode::BillingContactInvalid;
622 break;
623
624 case ApplePaySession::STATUS_INVALID_SHIPPING_POSTAL_ADDRESS:
625 errorCode = ApplePayErrorCode::ShippingContactInvalid;
626 contactField = ApplePayErrorContactField::PostalAddress;
627 break;
628
629 case ApplePaySession::STATUS_INVALID_SHIPPING_CONTACT:
630 errorCode = ApplePayErrorCode::ShippingContactInvalid;
631 break;
632
633 default:
634 return Exception { InvalidAccessError };
635 }
636
637 if (errorCode)
638 update.errors = { ApplePayError::create(*errorCode, contactField, { }) };
639
640 update.newShippingMethods = WTFMove(newShippingMethods);
641 update.newTotal = WTFMove(newTotal);
642 update.newLineItems = WTFMove(newLineItems);
643
644 return completeShippingContactSelection(WTFMove(update));
645}

670void ApplePaySession::validateMerchant(URL&& validationURL)
671{
672 if (m_state == State::Aborted) {
673 // ApplePaySession::abort has been called.
674 return;
675 }
676
677 ASSERT(m_merchantValidationState == MerchantValidationState::Idle);
678 ASSERT(m_state == State::Active);
679
680 if (validationURL.isNull()) {
681 // Something went wrong when getting the validation URL.
682 // FIXME: Maybe we should send an error event here instead?
683 return;
684 }
685
686 m_merchantValidationState = MerchantValidationState::ValidatingMerchant;
687
688 auto event = ApplePayValidateMerchantEvent::create(eventNames().validatemerchantEvent, WTFMove(validationURL));
689 dispatchEvent(event.get());
690}

809{
