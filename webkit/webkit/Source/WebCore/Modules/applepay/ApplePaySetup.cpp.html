43static bool shouldDiscloseFeatures(Document& document)
44{
45 auto* page = document.page();
46 if (!page || page->usesEphemeralSession())
47 return false;
48
49 return document.settings().applePayCapabilityDisclosureAllowed();
50}
51
52void ApplePaySetup::getSetupFeatures(Document& document, SetupFeaturesPromise&& promise)
53{
54 auto canCall = PaymentSession::canCreateSession(document);
55 if (canCall.hasException()) {
56 promise.settle(canCall.releaseException());
57 return;
58 }
59
60 auto page = document.page();
61 if (!page) {
62 promise.settle(Exception { InvalidStateError });
63 return;
64 }
65
66 if (m_setupFeaturesPromise) {
67 promise.settle(Exception { InvalidStateError });
68 return;
69 }
70
71 // Resolve with an empty sequence of features if Apple Pay capability disclosure is not allowed.
72 if (!shouldDiscloseFeatures(document)) {
73 promise.resolve({ });
74 return;
75 }
76
77 m_setupFeaturesPromise = WTFMove(promise);
78
79 page->paymentCoordinator().getSetupFeatures(m_configuration, document.url(), [this, pendingActivity = makePendingActivity(*this)](Vector<Ref<ApplePaySetupFeature>>&& setupFeatures) {
80 if (m_setupFeaturesPromise)
81 std::exchange(m_setupFeaturesPromise, std::nullopt)->settle(WTFMove(setupFeatures));
82 });
83}
84
85void ApplePaySetup::begin(Document& document, Vector<RefPtr<ApplePaySetupFeature>>&& features, BeginPromise&& promise)
86{
87 auto canCall = PaymentSession::canCreateSession(document);
88 if (canCall.hasException()) {
89 promise.settle(canCall.releaseException());
90 return;
91 }
92
93 if (!UserGestureIndicator::processingUserGesture()) {
94 promise.settle(Exception { InvalidAccessError, "Must call ApplePaySetup.begin from a user gesture handler." });
95 return;
96 }
97
98 auto page = document.page();
99 if (!page) {
100 promise.settle(Exception { InvalidStateError });
101 return;
102 }
103
104 if (m_beginPromise) {
105 promise.settle(Exception { InvalidStateError });
106 return;
107 }
108
109 m_beginPromise = WTFMove(promise);
110 m_pendingActivity = makePendingActivity(*this);
111
112 page->paymentCoordinator().beginApplePaySetup(m_configuration, document.url(), WTFMove(features), [this](bool result) {
113 if (m_beginPromise)
114 std::exchange(m_beginPromise, std::nullopt)->settle(result);
115 });
116}

118ApplePaySetup::ApplePaySetup(ScriptExecutionContext& context, ApplePaySetupConfiguration&& configuration)
119 : ActiveDOMObject(&context)
120 , m_configuration(WTFMove(configuration))
121{
122 suspendIfNeeded();
123}
124
125void ApplePaySetup::stop()
126{
127 if (m_setupFeaturesPromise)
128 std::exchange(m_setupFeaturesPromise, std::nullopt)->settle(Exception { AbortError });
129
130 if (m_beginPromise)
131 std::exchange(m_beginPromise, std::nullopt)->settle(Exception { AbortError });
132
133 if (auto page = downcast<Document>(*scriptExecutionContext()).page())
134 page->paymentCoordinator().endApplePaySetup();
135
136 m_pendingActivity = nullptr;
137}
138
139void ApplePaySetup::suspend(ReasonForSuspension)
140{
141 stop();
142}
