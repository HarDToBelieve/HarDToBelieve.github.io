37ApplePaySetupFeature::ApplePaySetupFeature() = default;
38ApplePaySetupFeature::~ApplePaySetupFeature() = default;

40ApplePaySetupFeatureType ApplePaySetupFeature::type() const
41{
42 switch ([m_feature type]) {
43 case PKPaymentSetupFeatureTypeApplePay:
44 return ApplePaySetupFeatureType::ApplePay;
45 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
46 case PKPaymentSetupFeatureTypeApplePay_X:
47 ALLOW_DEPRECATED_DECLARATIONS_END
48 return ApplePaySetupFeatureType::AppleCard;
49 }
50}
51
52ApplePaySetupFeatureState ApplePaySetupFeature::state() const
53{
54 switch ([m_feature state]) {
55 case PKPaymentSetupFeatureStateUnsupported:
56 return ApplePaySetupFeatureState::Unsupported;
57 case PKPaymentSetupFeatureStateSupported:
58 return ApplePaySetupFeatureState::Supported;
59 case PKPaymentSetupFeatureStateSupplementarySupported:
60 return ApplePaySetupFeatureState::SupplementarySupported;
61 case PKPaymentSetupFeatureStateCompleted:
62 return ApplePaySetupFeatureState::Completed;
63 }
64}

66#if ENABLE(APPLE_PAY_INSTALLMENTS)
67bool ApplePaySetupFeature::supportsInstallments() const
68{
69#if PLATFORM(MAC)
70 if (![m_feature respondsToSelector:@selector(supportedOptions)])
71 return false;
72#endif
73 return [m_feature supportedOptions] & PKPaymentSetupFeatureSupportedOptionsInstallments;
74}
75#endif
