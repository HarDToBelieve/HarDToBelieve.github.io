28#if ENABLE(APPLE_PAY)
29
30#include "ApplePayDetailsUpdateBase.h"
31#include "ApplePayError.h"
32#include "ApplePayShippingMethod.h"
33#include <wtf/RefPtr.h>
34#include <wtf/Vector.h>
35
36namespace WebCore {
37
38struct ApplePayShippingContactUpdate final : public ApplePayDetailsUpdateBase {
39 Vector<RefPtr<ApplePayError>> errors;
40
41 Vector<ApplePayShippingMethod> newShippingMethods;
42
43 template<class Encoder> void encode(Encoder&) const;
44 template<class Decoder> static std::optional<ApplePayShippingContactUpdate> decode(Decoder&);
45};
46
47template<class Encoder>
48void ApplePayShippingContactUpdate::encode(Encoder& encoder) const
49{
50 ApplePayDetailsUpdateBase::encode(encoder);
51 encoder << errors;
52 encoder << newShippingMethods;
53}
54
55template<class Decoder>
56std::optional<ApplePayShippingContactUpdate> ApplePayShippingContactUpdate::decode(Decoder& decoder)
57{
58 ApplePayShippingContactUpdate result;
59
60 if (!result.decodeBase(decoder))
61 return std::nullopt;
62
63#define DECODE(name, type) \
64 std::optional<type> name; \
65 decoder >> name; \
66 if (!name) \
67 return std::nullopt; \
68 result.name = WTFMove(*name); \
69
70 DECODE(errors, Vector<RefPtr<ApplePayError>>)
71 DECODE(newShippingMethods, Vector<ApplePayShippingMethod>)
72
73#undef DECODE
74
75 return result;
76}

78} // namespace WebCore
