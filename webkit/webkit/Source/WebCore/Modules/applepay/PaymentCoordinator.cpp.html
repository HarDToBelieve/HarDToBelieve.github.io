69bool PaymentCoordinator::canMakePayments(Document& document)
70{
71 auto canMakePayments = m_client.canMakePayments();
72 PAYMENT_COORDINATOR_RELEASE_LOG("canMakePayments() -> %d", canMakePayments);
73
74 if (!canMakePayments)
75 return false;
76
77 if (!setApplePayIsActiveIfAllowed(document))
78 return false;
79
80 return true;
81}

107bool PaymentCoordinator::beginPaymentSession(Document& document, PaymentSession& paymentSession, const ApplePaySessionPaymentRequest& paymentRequest)
108{
109 ASSERT(!m_activeSession);
110
111 if (!setApplePayIsActiveIfAllowed(document))
112 return false;
113
114 Vector<URL> linkIconURLs;
115 for (auto& icon : LinkIconCollector { document }.iconsOfTypes({ LinkIconType::TouchIcon, LinkIconType::TouchPrecomposedIcon }))
116 linkIconURLs.append(icon.url);
117
118 auto showPaymentUI = m_client.showPaymentUI(document.url(), linkIconURLs, paymentRequest);
119 PAYMENT_COORDINATOR_RELEASE_LOG("beginPaymentSession() -> %d", showPaymentUI);
120 if (!showPaymentUI)
121 return false;
122
123 m_activeSession = &paymentSession;
124 return true;
125}

277std::optional<String> PaymentCoordinator::validatedPaymentNetwork(Document&, unsigned version, const String& paymentNetwork) const
278{
279 if (version < 2 && equalIgnoringASCIICase(paymentNetwork, "jcb"))
280 return std::nullopt;
281
282 if (version < 3 && equalIgnoringASCIICase(paymentNetwork, "carteBancaire"))
283 return std::nullopt;
284
285 return m_client.validatedPaymentNetwork(paymentNetwork);
286}

288bool PaymentCoordinator::shouldEnableApplePayAPIs(Document& document) const
289{
290 if (m_client.supportsUnrestrictedApplePay())
291 return true;
292
293 bool shouldEnableAPIs = true;
294 document.page()->userContentProvider().forEachUserScript([&](DOMWrapperWorld&, const UserScript&) {
295 shouldEnableAPIs = false;
296 });
297
298 if (!shouldEnableAPIs)
299 PAYMENT_COORDINATOR_RELEASE_LOG("shouldEnableApplePayAPIs() -> false (user scripts)");
300
301 return shouldEnableAPIs;
302}

304bool PaymentCoordinator::setApplePayIsActiveIfAllowed(Document& document) const
305{
306 auto hasEvaluatedUserAgentScripts = document.hasEvaluatedUserAgentScripts();
307 auto isRunningUserScripts = document.isRunningUserScripts();
308 auto supportsUnrestrictedApplePay = m_client.supportsUnrestrictedApplePay();
309
310 if (!supportsUnrestrictedApplePay && (hasEvaluatedUserAgentScripts || isRunningUserScripts)) {
311 ASSERT(!document.isApplePayActive());
312 PAYMENT_COORDINATOR_RELEASE_LOG("setApplePayIsActiveIfAllowed() -> false (hasEvaluatedUserAgentScripts: %d, isRunningUserScripts: %d)", hasEvaluatedUserAgentScripts, isRunningUserScripts);
313 return false;
314 }
315
316 document.setApplePayIsActive();
317 return true;
318}
319
320Expected<void, ExceptionDetails> PaymentCoordinator::shouldAllowUserAgentScripts(Document& document) const
321{
322 if (m_client.supportsUnrestrictedApplePay() || !document.isApplePayActive())
323 return { };
324
325 ASSERT(!document.hasEvaluatedUserAgentScripts());
326 ASSERT(!document.isRunningUserScripts());
327 PAYMENT_COORDINATOR_RELEASE_LOG_ERROR("shouldAllowUserAgentScripts() -> false (active session)");
328 return makeUnexpected(ExceptionDetails { m_client.userAgentScriptsBlockedErrorMessage() });
329}
