41// FIXME: Merge with toDecimalNumber() in WebPaymentCoordinatorProxyCocoa.
42static NSDecimalNumber *toDecimalNumber(const String& amount)
43{
44 if (!amount)
45 return [NSDecimalNumber zero];
46 return [NSDecimalNumber decimalNumberWithString:amount locale:@{ NSLocaleDecimalSeparator : @"." }];
47}
48
49static String fromDecimalNumber(NSDecimalNumber *number)
50{
51 auto numberFormatter = adoptNS([[NSNumberFormatter alloc] init]);
52 [numberFormatter setNumberStyle:NSNumberFormatterNoStyle];
53 [numberFormatter setMinimumIntegerDigits:1];
54 [numberFormatter setMinimumFractionDigits:2];
55 [numberFormatter setMaximumFractionDigits:[numberFormatter maximumIntegerDigits]];
56 return [numberFormatter stringFromNumber:number];
57}

143static RetainPtr<id> makeNSArrayElement(const ApplePayInstallmentItem& item)
144{
145 ASSERT(PAL::getPKPaymentInstallmentItemClass());
146 auto installmentItem = adoptNS([PAL::allocPKPaymentInstallmentItemInstance() init]);
147 [installmentItem setInstallmentItemType:platformItemType(item.type)];
148 [installmentItem setAmount:toDecimalNumber(item.amount)];
149 [installmentItem setCurrencyCode:item.currencyCode];
150 [installmentItem setProgramIdentifier:item.programIdentifier];
151 [installmentItem setApr:toDecimalNumber(item.apr)];
152 [installmentItem setProgramTerms:item.programTerms];
153 return installmentItem;
154}

156static std::optional<ApplePayInstallmentItem> makeVectorElement(const ApplePayInstallmentItem*, id arrayElement)
157{
158 if (![arrayElement isKindOfClass:PAL::getPKPaymentInstallmentItemClass()])
159 return std::nullopt;
160
161 PKPaymentInstallmentItem *item = arrayElement;
162 return ApplePayInstallmentItem {
163 applePayItemType([item installmentItemType]),
164 fromDecimalNumber([item amount]),
165 [item currencyCode],
166 [item programIdentifier],
167 fromDecimalNumber([item apr]),
168 [item programTerms],
169 };
170}

172static RetainPtr<PKPaymentInstallmentConfiguration> createPlatformConfiguration(const ApplePayInstallmentConfiguration& coreConfiguration, NSDictionary *applicationMetadata)
173{
174 if (!PAL::getPKPaymentInstallmentConfigurationClass())
175 return nil;
176
177 auto configuration = adoptNS([PAL::allocPKPaymentInstallmentConfigurationInstance() init]);
178
179 [configuration setFeature:platformFeatureType(coreConfiguration.featureType)];
180
181 [configuration setBindingTotalAmount:toDecimalNumber(coreConfiguration.bindingTotalAmount)];
182 [configuration setCurrencyCode:coreConfiguration.currencyCode];
183 [configuration setInStorePurchase:coreConfiguration.isInStorePurchase];
184 [configuration setOpenToBuyThresholdAmount:toDecimalNumber(coreConfiguration.openToBuyThresholdAmount)];
185
186 auto merchandisingImageData = adoptNS([[NSData alloc] initWithBase64EncodedString:coreConfiguration.merchandisingImageData options:0]);
187 [configuration setMerchandisingImageData:merchandisingImageData.get()];
188 [configuration setInstallmentMerchantIdentifier:coreConfiguration.merchantIdentifier];
189 [configuration setReferrerIdentifier:coreConfiguration.referrerIdentifier];
190
191 if (!PAL::getPKPaymentInstallmentItemClass())
192 return configuration;
193
194 [configuration setInstallmentItems:createNSArray(coreConfiguration.items).get()];
195 [configuration setApplicationMetadata:applicationMetadata];
196 [configuration setRetailChannel:platformRetailChannel(coreConfiguration.retailChannel)];
197
198 return configuration;
199}

201ExceptionOr<PaymentInstallmentConfiguration> PaymentInstallmentConfiguration::create(const ApplePayInstallmentConfiguration& configuration)
202{
203 NSDictionary *applicationMetadataDictionary = nil;
204 if (!configuration.applicationMetadata.isNull()) {
205 NSData *applicationMetadata = [configuration.applicationMetadata dataUsingEncoding:NSUTF8StringEncoding];
206 applicationMetadataDictionary = dynamic_objc_cast<NSDictionary>([NSJSONSerialization JSONObjectWithData:applicationMetadata options:0 error:nil]);
207 if (!applicationMetadataDictionary)
208 return Exception { TypeError, "applicationMetadata must be a JSON object" };
209 }
210
211 return PaymentInstallmentConfiguration(configuration, applicationMetadataDictionary);
212}

229ApplePayInstallmentConfiguration PaymentInstallmentConfiguration::applePayInstallmentConfiguration() const
230{
231 ApplePayInstallmentConfiguration installmentConfiguration;
232 if (!PAL::getPKPaymentInstallmentConfigurationClass())
233 return installmentConfiguration;
234
235 installmentConfiguration.featureType = applePaySetupFeatureType([m_configuration feature]);
236
237 installmentConfiguration.bindingTotalAmount = fromDecimalNumber([m_configuration bindingTotalAmount]);
238 installmentConfiguration.currencyCode = [m_configuration currencyCode];
239 installmentConfiguration.isInStorePurchase = [m_configuration isInStorePurchase];
240 installmentConfiguration.openToBuyThresholdAmount = fromDecimalNumber([m_configuration openToBuyThresholdAmount]);
241
242 installmentConfiguration.merchandisingImageData = [[m_configuration merchandisingImageData] base64EncodedStringWithOptions:0];
243 installmentConfiguration.merchantIdentifier = [m_configuration installmentMerchantIdentifier];
244 installmentConfiguration.referrerIdentifier = [m_configuration referrerIdentifier];
245
246 if (!PAL::getPKPaymentInstallmentItemClass())
247 return installmentConfiguration;
248
249 RetainPtr<NSString> applicationMetadataString;
250 if (NSDictionary *applicationMetadataDictionary = [m_configuration applicationMetadata]) {
251 if (NSData *applicationMetadata = [NSJSONSerialization dataWithJSONObject:applicationMetadataDictionary options:NSJSONWritingSortedKeys error:nil])
252 applicationMetadataString = adoptNS([[NSString alloc] initWithData:applicationMetadata encoding:NSUTF8StringEncoding]);
253 }
254
255 installmentConfiguration.items = makeVector<ApplePayInstallmentItem>([m_configuration installmentItems]);
256 installmentConfiguration.applicationMetadata = applicationMetadataString.get();
257 installmentConfiguration.retailChannel = applePayRetailChannel([m_configuration retailChannel]);
258
259 return installmentConfiguration;
260}
