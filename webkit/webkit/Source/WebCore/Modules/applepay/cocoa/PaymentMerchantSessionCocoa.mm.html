36std::optional<PaymentMerchantSession> PaymentMerchantSession::fromJS(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, String&)
37{
38 // FIXME: Don't round-trip using NSString.
39 auto jsonString = JSONStringify(&lexicalGlobalObject, value, 0);
40 if (!jsonString)
41 return std::nullopt;
42
43 auto dictionary = dynamic_objc_cast<NSDictionary>([NSJSONSerialization JSONObjectWithData:[(NSString *)jsonString dataUsingEncoding:NSUTF8StringEncoding] options:0 error:nil]);
44 if (!dictionary || ![dictionary isKindOfClass:[NSDictionary class]])
45 return std::nullopt;
46
47 auto pkPaymentMerchantSession = adoptNS([PAL::allocPKPaymentMerchantSessionInstance() initWithDictionary:dictionary]);
48
49 return PaymentMerchantSession(pkPaymentMerchantSession.get());
50}
