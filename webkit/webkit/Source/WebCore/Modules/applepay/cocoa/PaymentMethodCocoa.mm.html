37static void finishConverting(PKPaymentMethod *paymentMethod, ApplePayPaymentMethod& result)
38{

48static ApplePayPaymentPass::ActivationState convert(PKPaymentPassActivationState paymentPassActivationState)
49{
50 switch (paymentPassActivationState) {
51 case PKPaymentPassActivationStateActivated:
52 return ApplePayPaymentPass::ActivationState::Activated;
53 case PKPaymentPassActivationStateRequiresActivation:
54 return ApplePayPaymentPass::ActivationState::RequiresActivation;
55 case PKPaymentPassActivationStateActivating:
56 return ApplePayPaymentPass::ActivationState::Activating;
57 case PKPaymentPassActivationStateSuspended:
58 return ApplePayPaymentPass::ActivationState::Suspended;
59 case PKPaymentPassActivationStateDeactivated:
60 return ApplePayPaymentPass::ActivationState::Deactivated;
61 }
62}

64static std::optional<ApplePayPaymentPass> convert(PKPaymentPass *paymentPass)
65{
66 if (!paymentPass)
67 return std::nullopt;
68
69 ApplePayPaymentPass result;
70
71 result.primaryAccountIdentifier = paymentPass.primaryAccountIdentifier;
72 result.primaryAccountNumberSuffix = paymentPass.primaryAccountNumberSuffix;
73
74 if (NSString *deviceAccountIdentifier = paymentPass.deviceAccountIdentifier)
75 result.deviceAccountIdentifier = deviceAccountIdentifier;
76 if (NSString *deviceAccountNumberSuffix = paymentPass.deviceAccountNumberSuffix)
77 result.deviceAccountNumberSuffix = deviceAccountNumberSuffix;
78
79 result.activationState = convert(paymentPass.activationState);
80
81 return result;
82}

84static std::optional<ApplePayPaymentMethod::Type> convert(PKPaymentMethodType paymentMethodType)
85{
86 switch (paymentMethodType) {
87 case PKPaymentMethodTypeDebit:
88 return ApplePayPaymentMethod::Type::Debit;
89 case PKPaymentMethodTypeCredit:
90 return ApplePayPaymentMethod::Type::Credit;
91 case PKPaymentMethodTypePrepaid:
92 return ApplePayPaymentMethod::Type::Prepaid;
93 case PKPaymentMethodTypeStore:
94 return ApplePayPaymentMethod::Type::Store;
95 case PKPaymentMethodTypeUnknown:
96 default:
97 return std::nullopt;
98 }
99}

101#if HAVE(PASSKIT_PAYMENT_METHOD_BILLING_ADDRESS)
102static void convert(CNLabeledValue<CNPostalAddress*> *postalAddress, ApplePayPaymentContact &result)
103{
104 Vector<String> addressLine;
105 if (NSString *street = postalAddress.value.street) {
106 addressLine.append(street);
107 result.addressLines = WTFMove(addressLine);
108 }
109 result.subLocality = postalAddress.value.subLocality;
110 result.locality = postalAddress.value.city;
111 result.subAdministrativeArea = postalAddress.value.subAdministrativeArea;
112 result.administrativeArea = postalAddress.value.state;
113 result.postalCode = postalAddress.value.postalCode;
114 result.country = postalAddress.value.country;
115 result.countryCode = postalAddress.value.ISOCountryCode;
116}

118static std::optional<ApplePayPaymentContact> convert(CNContact *billingContact)
119{
120 if (!billingContact)
121 return std::nullopt;
122
123 ApplePayPaymentContact result;
124
125 if (auto firstPhoneNumber = billingContact.phoneNumbers.firstObject)
126 result.phoneNumber = firstPhoneNumber.value.stringValue;
127
128 if (auto firstEmailAddress = billingContact.emailAddresses.firstObject)
129 result.emailAddress = firstEmailAddress.value;
130
131 result.givenName = billingContact.givenName;
132 result.familyName = billingContact.familyName;
133
134 result.phoneticGivenName = billingContact.phoneticGivenName;
135 result.phoneticFamilyName = billingContact.phoneticFamilyName;
136
137 if (CNLabeledValue<CNPostalAddress*> *firstPostalAddress = billingContact.postalAddresses.firstObject)
138 convert(firstPostalAddress, result);
139
140 return result;
141}
142#endif

144static ApplePayPaymentMethod convert(PKPaymentMethod *paymentMethod)
145{
146 ApplePayPaymentMethod result;
147
148 if (NSString *displayName = paymentMethod.displayName)
149 result.displayName = displayName;
150 if (NSString *network = paymentMethod.network)
151 result.network = network;
152#if HAVE(PASSKIT_PAYMENT_METHOD_BILLING_ADDRESS)
153 result.billingContact = convert(paymentMethod.billingAddress);
154#endif
155 result.type = convert(paymentMethod.type);
156 result.paymentPass = convert(paymentMethod.paymentPass);
157
158 finishConverting(paymentMethod, result);
159
160 return result;
161}
