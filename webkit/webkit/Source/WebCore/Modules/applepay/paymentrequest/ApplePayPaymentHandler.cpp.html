81static ExceptionOr<ApplePayRequest> convertAndValidate(ScriptExecutionContext& context, JSC::JSValue data)
82{
83 if (data.isEmpty())
84 return Exception { TypeError, "Missing payment method data." };
85
86 auto throwScope = DECLARE_THROW_SCOPE(context.vm());
87 auto applePayRequest = convertDictionary<ApplePayRequest>(*context.globalObject(), data);
88 if (throwScope.exception())
89 return Exception { ExistingExceptionError };
90
91 return WTFMove(applePayRequest);
92}

103bool ApplePayPaymentHandler::handlesIdentifier(const PaymentRequest::MethodIdentifier& identifier)
104{
105 if (!WTF::holds_alternative<URL>(identifier))
106 return false;
107
108 auto& url = WTF::get<URL>(identifier);
109 return url.host() == "apple.com" && url.path() == "/apple-pay";
110}

141static ExceptionOr<void> validate(const PaymentCurrencyAmount& amount, const String& expectedCurrency)
142{
143 if (amount.currency != expectedCurrency)
144 return Exception { TypeError, makeString("\"", amount.currency, "\" does not match the expected currency of \"", expectedCurrency, "\". Apple Pay requires all PaymentCurrencyAmounts to use the same currency code.") };
145 return { };
146}

148static ExceptionOr<ApplePayLineItem> convertAndValidate(const PaymentItem& item, const String& expectedCurrency)
149{
150 auto exception = validate(item.amount, expectedCurrency);
151 if (exception.hasException())
152 return exception.releaseException();
153
154
155 ApplePayLineItem lineItem;
156 lineItem.amount = item.amount.value;
157 lineItem.type = item.pending ? ApplePayLineItem::Type::Pending : ApplePayLineItem::Type::Final;
158 lineItem.label = item.label;
159 return { WTFMove(lineItem) };
160}
161
162static ExceptionOr<Vector<ApplePayLineItem>> convertAndValidate(const Vector<PaymentItem>& lineItems, const String& expectedCurrency)
163{
164 Vector<ApplePayLineItem> result;
165 result.reserveInitialCapacity(lineItems.size());
166 for (auto& lineItem : lineItems) {
167 auto convertedLineItem = convertAndValidate(lineItem, expectedCurrency);
168 if (convertedLineItem.hasException())
169 return convertedLineItem.releaseException();
170 result.uncheckedAppend(convertedLineItem.releaseReturnValue());
171 }
172 return { WTFMove(result) };
173}

175static ApplePaySessionPaymentRequest::ShippingType convert(PaymentShippingType type)
176{
177 switch (type) {
178 case PaymentShippingType::Shipping:
179 return ApplePaySessionPaymentRequest::ShippingType::Shipping;
180 case PaymentShippingType::Delivery:
181 return ApplePaySessionPaymentRequest::ShippingType::Delivery;
182 case PaymentShippingType::Pickup:
183 return ApplePaySessionPaymentRequest::ShippingType::StorePickup;
184 }
185
186 ASSERT_NOT_REACHED();
187 return ApplePaySessionPaymentRequest::ShippingType::Shipping;
188}

213static void mergePaymentOptions(const PaymentOptions& options, ApplePaySessionPaymentRequest& request)
214{
215 auto requiredShippingContactFields = request.requiredShippingContactFields();
216 requiredShippingContactFields.email |= options.requestPayerEmail;
217 requiredShippingContactFields.name |= options.requestPayerName;
218 requiredShippingContactFields.phone |= options.requestPayerPhone;
219 requiredShippingContactFields.postalAddress |= options.requestShipping;
220 request.setRequiredShippingContactFields(requiredShippingContactFields);
221
222 auto requiredBillingContactFields = request.requiredBillingContactFields();
223 requiredBillingContactFields.postalAddress |= options.requestBillingAddress;
224 request.setRequiredBillingContactFields(requiredBillingContactFields);
225
226 if (options.requestShipping)
227 request.setShippingType(convert(options.shippingType));
228}

234ExceptionOr<void> ApplePayPaymentHandler::show(Document& document)
235{
236 auto validatedRequest = convertAndValidate(document, m_applePayRequest->version, *m_applePayRequest, paymentCoordinator());
237 if (validatedRequest.hasException())
238 return validatedRequest.releaseException();
239
240 ApplePaySessionPaymentRequest request = validatedRequest.releaseReturnValue();
241 request.setRequester(ApplePaySessionPaymentRequest::Requester::PaymentRequest);
242
243 auto& details = m_paymentRequest->paymentDetails();
244
245 String expectedCurrency = details.total.amount.currency;
246 request.setCurrencyCode(expectedCurrency);
247
248 auto total = convertAndValidate(details.total, expectedCurrency);
249 ASSERT(!total.hasException());
250 request.setTotal(total.releaseReturnValue());
251
252 if (details.displayItems) {
253 auto convertedLineItems = convertAndValidate(*details.displayItems, expectedCurrency);
254 if (convertedLineItems.hasException())
255 return convertedLineItems.releaseException();
256 request.setLineItems(convertedLineItems.releaseReturnValue());
257 }
258
259 mergePaymentOptions(m_paymentRequest->paymentOptions(), request);
260
261 auto shippingMethods = computeShippingMethods();
262 if (shippingMethods.hasException())
263 return shippingMethods.releaseException();
264 request.setShippingMethods(shippingMethods.releaseReturnValue());
265
266 auto modifierException = firstApplicableModifier();
267 if (modifierException.hasException())
268 return modifierException.releaseException();
269 if (auto modifierData = modifierException.releaseReturnValue())
270 merge(request, WTFMove(std::get<1>(*modifierData)));
271
272 auto exception = PaymentRequestValidator::validate(request);
273 if (exception.hasException())
274 return exception.releaseException();
275
276 if (!paymentCoordinator().beginPaymentSession(document, *this, request))
277 return Exception { AbortError };
278
279 return { };
280}

292ExceptionOr<Vector<ApplePayShippingMethod>> ApplePayPaymentHandler::computeShippingMethods() const
293{
294 auto& details = m_paymentRequest->paymentDetails();
295
296 Vector<ApplePayShippingMethod> shippingOptions;
297 if (details.shippingOptions) {
298 auto& currency = details.total.amount.currency;
299
300 shippingOptions.reserveInitialCapacity(details.shippingOptions->size());
301 for (auto& shippingOption : *details.shippingOptions) {
302 auto shippingMethod = convertAndValidate(shippingOption, currency);
303 if (shippingMethod.hasException())
304 return shippingMethod.releaseException();
305 shippingOptions.uncheckedAppend(shippingMethod.releaseReturnValue());
306 }
307 }
308
309 auto modifierException = firstApplicableModifier();
310 if (modifierException.hasException())
311 return modifierException.releaseException();
312 if (auto modifierData = modifierException.releaseReturnValue())
313 shippingOptions.appendVector(WTFMove(std::get<1>(*modifierData).additionalShippingMethods));
314
315 return WTFMove(shippingOptions);
316}

318ExceptionOr<std::tuple<ApplePayLineItem, Vector<ApplePayLineItem>>> ApplePayPaymentHandler::computeTotalAndLineItems() const
319{
320 auto& details = m_paymentRequest->paymentDetails();
321 auto& currency = details.total.amount.currency;
322
323 auto convertedTotal = convertAndValidate(details.total, currency);
324 if (convertedTotal.hasException())
325 return convertedTotal.releaseException();
326 auto total = convertedTotal.releaseReturnValue();
327
328 Vector<ApplePayLineItem> lineItems;
329 if (details.displayItems) {
330 auto convertedLineItems = convertAndValidate(*details.displayItems, currency);
331 if (convertedLineItems.hasException())
332 return convertedLineItems.releaseException();
333 lineItems = convertedLineItems.releaseReturnValue();
334 }
335
336 auto modifierException = firstApplicableModifier();
337 if (modifierException.hasException())
338 return modifierException.releaseException();
339 if (auto modifierData = modifierException.releaseReturnValue()) {
340 auto& [modifier, applePayModifier] = *modifierData;
341
342 if (modifier.total) {
343 auto totalOverride = convertAndValidate(*modifier.total, currency);
344 if (totalOverride.hasException())
345 return totalOverride.releaseException();
346 total = totalOverride.releaseReturnValue();
347 }
348
349 auto additionalDisplayItems = convertAndValidate(modifier.additionalDisplayItems, currency);
350 if (additionalDisplayItems.hasException())
351 return additionalDisplayItems.releaseException();
352 lineItems.appendVector(additionalDisplayItems.releaseReturnValue());
353
354 if (applePayModifier.total)
355 total = *applePayModifier.total;
356
357 lineItems.appendVector(applePayModifier.additionalLineItems);
358 }
359
360 return {{ WTFMove(total), WTFMove(lineItems) }};
361}

369Vector<RefPtr<ApplePayError>> ApplePayPaymentHandler::computeErrors(String&& error, AddressErrors&& addressErrors, PayerErrorFields&& payerErrors, JSC::JSObject* paymentMethodErrors) const
370{
371 Vector<RefPtr<ApplePayError>> errors;
372
373 auto& details = m_paymentRequest->paymentDetails();
374
375 if (!details.shippingOptions || details.shippingOptions->isEmpty())
376 computeAddressErrors(WTFMove(error), WTFMove(addressErrors), errors);
377
378 computePayerErrors(WTFMove(payerErrors), errors);
379
380 auto scope = DECLARE_CATCH_SCOPE(scriptExecutionContext()->vm());
381 auto exception = computePaymentMethodErrors(paymentMethodErrors, errors);
382 if (exception.hasException()) {
383 ASSERT(scope.exception());
384 scope.clearException();
385 }
386
387 return errors;
388}

404void ApplePayPaymentHandler::computeAddressErrors(String&& error, AddressErrors&& addressErrors, Vector<RefPtr<ApplePayError>>& errors) const
405{
406 if (!m_paymentRequest->paymentOptions().requestShipping)
407 return;
408
409 appendShippingContactInvalidError(WTFMove(error), std::nullopt, errors);
410 appendShippingContactInvalidError(WTFMove(addressErrors.addressLine), ApplePayErrorContactField::AddressLines, errors);
411 appendShippingContactInvalidError(WTFMove(addressErrors.city), ApplePayErrorContactField::Locality, errors);
412 appendShippingContactInvalidError(WTFMove(addressErrors.country), ApplePayErrorContactField::Country, errors);
413 appendShippingContactInvalidError(WTFMove(addressErrors.dependentLocality), ApplePayErrorContactField::SubLocality, errors);
414 appendShippingContactInvalidError(WTFMove(addressErrors.phone), ApplePayErrorContactField::PhoneNumber, errors);
415 appendShippingContactInvalidError(WTFMove(addressErrors.postalCode), ApplePayErrorContactField::PostalCode, errors);
416 appendShippingContactInvalidError(WTFMove(addressErrors.recipient), ApplePayErrorContactField::Name, errors);
417 appendShippingContactInvalidError(WTFMove(addressErrors.region), ApplePayErrorContactField::AdministrativeArea, errors);
418}

434ExceptionOr<void> ApplePayPaymentHandler::computePaymentMethodErrors(JSC::JSObject* paymentMethodErrors, Vector<RefPtr<ApplePayError>>& errors) const
435{
436 if (!paymentMethodErrors)
437 return { };
438
439 auto& context = *scriptExecutionContext();
440 auto throwScope = DECLARE_THROW_SCOPE(context.vm());
441 auto applePayErrors = convert<IDLSequence<IDLInterface<ApplePayError>>>(*context.globalObject(), paymentMethodErrors);
442 if (throwScope.exception())
443 return Exception { ExistingExceptionError };
444
445 for (auto&& applePayError : WTFMove(applePayErrors)) {
446 if (applePayError)
447 errors.append(WTFMove(applePayError));
448 }
449
450 return { };
451}

457ExceptionOr<std::optional<std::tuple<PaymentDetailsModifier, ApplePayModifier>>> ApplePayPaymentHandler::firstApplicableModifier() const
458{
459 auto& details = m_paymentRequest->paymentDetails();
460 if (!details.modifiers)
461 return { std::nullopt };
462
463 auto& lexicalGlobalObject = *document().globalObject();
464
465 auto& serializedModifierData = m_paymentRequest->serializedModifierData();
466 ASSERT(details.modifiers->size() == serializedModifierData.size());
467 for (size_t i = 0; i < details.modifiers->size(); ++i) {
468 auto& modifier = details.modifiers->at(i);
469
470 auto convertedIdentifier = convertAndValidatePaymentMethodIdentifier(modifier.supportedMethods);
471 if (!convertedIdentifier || !handlesIdentifier(*convertedIdentifier))
472 continue;
473
474 if (serializedModifierData[i].isEmpty())
475 continue;
476
477 auto scope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
478 JSC::JSValue data;
479 {
480 JSC::JSLockHolder lock(&lexicalGlobalObject);
481 data = JSONParse(&lexicalGlobalObject, serializedModifierData[i]);
482 if (scope.exception())
483 return Exception(ExistingExceptionError);
484 }
485
486 auto applePayModifier = convertDictionary<ApplePayModifier>(lexicalGlobalObject, WTFMove(data));
487 if (scope.exception())
488 return Exception(ExistingExceptionError);
489
490 auto validateApplePayModifierResult = validate(applePayModifier);
491 if (validateApplePayModifierResult.hasException())
492 return validateApplePayModifierResult.releaseException();
493
494 if (applePayModifier.paymentMethodType && *applePayModifier.paymentMethodType != m_selectedPaymentMethodType)
495 continue;
496
497 return { { { modifier, WTFMove(applePayModifier) } } };
498 }
499
500 return { std::nullopt };
501}

503ExceptionOr<void> ApplePayPaymentHandler::detailsUpdated(PaymentRequest::UpdateReason reason, String&& error, AddressErrors&& addressErrors, PayerErrorFields&& payerErrors, JSC::JSObject* paymentMethodErrors)
504{
505 using Reason = PaymentRequest::UpdateReason;
506 switch (reason) {
507 case Reason::ShowDetailsResolved:
508 return { };
509 case Reason::ShippingAddressChanged:
510 return shippingAddressUpdated(computeErrors(WTFMove(error), WTFMove(addressErrors), WTFMove(payerErrors), paymentMethodErrors));
511 case Reason::ShippingOptionChanged:
512 return shippingOptionUpdated();
513 case Reason::PaymentMethodChanged:
514 return paymentMethodUpdated(computeErrors(paymentMethodErrors));
515 }
516
517 ASSERT_NOT_REACHED();
518 return { };
519}

542ExceptionOr<void> ApplePayPaymentHandler::shippingAddressUpdated(Vector<RefPtr<ApplePayError>>&& errors)
543{
544 ASSERT(m_updateState == UpdateState::ShippingAddress);
545 m_updateState = UpdateState::None;
546
547 ApplePayShippingContactUpdate update;
548 update.errors = WTFMove(errors);
549
550 auto newShippingMethods = computeShippingMethods();
551 if (newShippingMethods.hasException())
552 return newShippingMethods.releaseException();
553 update.newShippingMethods = newShippingMethods.releaseReturnValue();
554
555 auto newTotalAndLineItems = computeTotalAndLineItems();
556 if (newTotalAndLineItems.hasException())
557 return newTotalAndLineItems.releaseException();
558 std::tie(update.newTotal, update.newLineItems) = newTotalAndLineItems.releaseReturnValue();
559
560 auto modifierException = firstApplicableModifier();
561 if (modifierException.hasException())
562 return modifierException.releaseException();
563 if (auto modifierData = modifierException.releaseReturnValue())
564 merge(update, WTFMove(std::get<1>(*modifierData)));
565
566 paymentCoordinator().completeShippingContactSelection(WTFMove(update));
567 return { };
568}

644void ApplePayPaymentHandler::complete(std::optional<PaymentComplete>&& result)
645{
646 if (!result) {
647 ASSERT(isFinalStateResult(std::nullopt));
648 paymentCoordinator().completePaymentSession(std::nullopt);
649 return;
650 }
651
652 PaymentAuthorizationResult authorizationResult;
653 switch (*result) {
654 case PaymentComplete::Fail:
655 case PaymentComplete::Unknown:
656 authorizationResult.status = PaymentAuthorizationStatus::Failure;
657 break;
658 case PaymentComplete::Success:
659 authorizationResult.status = PaymentAuthorizationStatus::Success;
660 break;
661 }
662
663 ASSERT(isFinalStateResult(authorizationResult));
664 paymentCoordinator().completePaymentSession(WTFMove(authorizationResult));
665}

667ExceptionOr<void> ApplePayPaymentHandler::retry(PaymentValidationErrors&& validationErrors)
668{
669 Vector<RefPtr<ApplePayError>> errors;
670
671 computeAddressErrors(WTFMove(validationErrors.error), WTFMove(validationErrors.shippingAddress), errors);
672 computePayerErrors(WTFMove(validationErrors.payer), errors);
673
674 auto exception = computePaymentMethodErrors(validationErrors.paymentMethod.get(), errors);
675 if (exception.hasException())
676 return exception.releaseException();
677
678 // Ensure there is always at least one error to avoid having a final result.
679 if (errors.isEmpty())
680 errors.append(ApplePayError::create(ApplePayErrorCode::Unknown, std::nullopt, nullString()));
681
682 PaymentAuthorizationResult authorizationResult { PaymentAuthorizationStatus::Failure, WTFMove(errors) };
