37ClipboardItemPasteboardDataSource::ClipboardItemPasteboardDataSource(ClipboardItem& item, const PasteboardItemInfo& info)
38 : ClipboardItemDataSource(item)
39 , m_types(info.webSafeTypesByFidelity)
40{
41}
42
43ClipboardItemPasteboardDataSource::~ClipboardItemPasteboardDataSource() = default;
44
45Vector<String> ClipboardItemPasteboardDataSource::types() const
46{
47 return m_types;
48}
49
50void ClipboardItemPasteboardDataSource::getType(const String& type, Ref<DeferredPromise>&& promise)
51{
52 if (auto clipboard = makeRefPtr(m_item.clipboard()))
53 clipboard->getType(m_item, type, WTFMove(promise));
54 else
55 promise->reject(NotAllowedError);
56}
