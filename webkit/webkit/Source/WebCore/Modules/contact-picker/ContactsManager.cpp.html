77
78void ContactsManager::select(const Vector<ContactProperty>& properties, const ContactsSelectOptions& options, Ref<DeferredPromise>&& promise)
79{
80 auto* frame = this->frame();
81 if (!frame || !frame->isMainFrame() || !frame->document() || !frame->page()) {
82 promise->reject(InvalidStateError);
83 return;
84 }
85
86 if (!UserGestureIndicator::processingUserGesture()) {
87 promise->reject(SecurityError);
88 return;
89 }
90
91 if (m_contactPickerIsShowing) {
92 promise->reject(InvalidStateError);
93 return;
94 }
95
96 if (properties.isEmpty()) {
97 promise->reject(TypeError);
98 return;
99 }
100
101 ContactsRequestData requestData;
102 requestData.properties = properties;
103 requestData.multiple = options.multiple;
104 requestData.url = frame->document()->url().truncatedForUseAsBase().string();
105
106 m_contactPickerIsShowing = true;
107
108 frame->page()->chrome().showContactPicker(requestData, [promise = WTFMove(promise), this] (std::optional<Vector<ContactInfo>>&& info) {
109 m_contactPickerIsShowing = false;
110 if (info) {
111 promise->resolve<IDLSequence<IDLDictionary<ContactInfo>>>(*info);
112 return;
113 }
114
115 promise->reject(UnknownError);
116 });
117}
