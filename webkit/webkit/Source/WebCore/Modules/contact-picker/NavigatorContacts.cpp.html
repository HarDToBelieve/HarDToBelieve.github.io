35NavigatorContacts::NavigatorContacts(Navigator& navigator)
36 : m_navigator(navigator)
37{
38}
39
40NavigatorContacts::~NavigatorContacts() = default;
41
42RefPtr<ContactsManager> NavigatorContacts::contacts(Navigator& navigator)
43{
44 return NavigatorContacts::from(navigator)->contacts();
45}
46
47RefPtr<ContactsManager> NavigatorContacts::contacts()
48{
49 if (!m_contactsManager)
50 m_contactsManager = ContactsManager::create(m_navigator);
51 return m_contactsManager;
52}
53
54NavigatorContacts* NavigatorContacts::from(Navigator& navigator)
55{
56 auto* supplement = static_cast<NavigatorContacts*>(Supplement<Navigator>::from(&navigator, supplementName()));
57 if (!supplement) {
58 auto newSupplement = makeUnique<NavigatorContacts>(navigator);
59 supplement = newSupplement.get();
60 provideTo(&navigator, supplementName(), WTFMove(newSupplement));
61 }
62 return supplement;
63}
64
65const char* NavigatorContacts::supplementName()
66{
67 return "NavigatorContacts";
68}
