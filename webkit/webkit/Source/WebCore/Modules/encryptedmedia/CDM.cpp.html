51bool CDM::supportsKeySystem(const String& keySystem)
52{
53 for (auto* factory : CDMFactory::registeredFactories()) {
54 if (factory->supportsKeySystem(keySystem))
55 return true;
56 }
57 return false;
58}

61{
62 return adoptRef(*new CDM(document, keySystem));
63}
64
65CDM::CDM(Document& document, const String& keySystem)
66 : ContextDestructionObserver(&document)
67#if !RELEASE_LOG_DISABLED
68 , m_logger(document.logger())
69 , m_logIdentifier(LoggerHelper::uniqueLogIdentifier())
70#endif
71 , m_keySystem(keySystem)
72{
73 ASSERT(supportsKeySystem(keySystem));
74 for (auto* factory : CDMFactory::registeredFactories()) {
75 if (factory->supportsKeySystem(keySystem)) {
76 m_private = factory->createCDM(keySystem);
77#if !RELEASE_LOG_DISABLED
78 m_private->setLogger(m_logger, m_logIdentifier);
79#endif
80 break;
81 }
82 }
83}

87void CDM::getSupportedConfiguration(MediaKeySystemConfiguration&& candidateConfiguration, SupportedConfigurationCallback&& callback)
88{
89 // https://w3c.github.io/encrypted-media/#get-supported-configuration
90 // W3C Editor's Draft 09 November 2016
91 // Implemented in CDMPrivate::getSupportedConfiguration()
92
93 Document* document = downcast<Document>(m_scriptExecutionContext);
94 if (!document || !m_private) {
95 callback(std::nullopt);
96 return;
97 }
98
99 bool isEphemeral = !document->page() || document->page()->sessionID().isEphemeral();
100
101 SecurityOrigin& origin = document->securityOrigin();
102 SecurityOrigin& topOrigin = document->topOrigin();
103 CDMPrivate::LocalStorageAccess access = !isEphemeral && origin.canAccessLocalStorage(&topOrigin) ? CDMPrivate::LocalStorageAccess::Allowed : CDMPrivate::LocalStorageAccess::NotAllowed;
104 m_private->getSupportedConfiguration(WTFMove(candidateConfiguration), access, WTFMove(callback));
105}

164String CDM::storageDirectory() const
165{
166 auto* document = downcast<Document>(scriptExecutionContext());
167 if (!document)
168 return emptyString();
169
170 auto* page = document->page();
171 if (!page || page->usesEphemeralSession())
172 return emptyString();
173
174 auto storageDirectory = document->settings().mediaKeysStorageDirectory();
175 if (storageDirectory.isEmpty())
176 return emptyString();
177
178 return FileSystem::pathByAppendingComponent(storageDirectory, document->securityOrigin().data().databaseIdentifier());
179}
