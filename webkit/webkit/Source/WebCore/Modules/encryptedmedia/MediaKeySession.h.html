53namespace WebCore {
54
55class BufferSource;
56class CDM;
57class DeferredPromise;
58class MediaKeyStatusMap;
59class MediaKeys;
60class SharedBuffer;
61
62template<typename IDLType> class DOMPromiseProxy;
63
64class MediaKeySession final : public RefCounted<MediaKeySession>, public EventTargetWithInlineData, public ActiveDOMObject, public CDMInstanceSessionClient {
65 WTF_MAKE_ISO_ALLOCATED(MediaKeySession);
66public:
67 static Ref<MediaKeySession> create(Document&, WeakPtr<MediaKeys>&&, MediaKeySessionType, bool useDistinctiveIdentifier, Ref<CDM>&&, Ref<CDMInstanceSession>&&);
68 virtual ~MediaKeySession();
69
70 using CDMInstanceSessionClient::weakPtrFactory;
71 using WeakValueType = CDMInstanceSessionClient::WeakValueType;
72 using RefCounted<MediaKeySession>::ref;
73 using RefCounted<MediaKeySession>::deref;
74
75 bool isClosed() const { return m_closed; }
76
77 const String& sessionId() const;
78 double expiration() const;
79 Ref<MediaKeyStatusMap> keyStatuses() const;
80
81 void generateRequest(const AtomString&, const BufferSource&, Ref<DeferredPromise>&&);
82 void load(const String&, Ref<DeferredPromise>&&);
83 void update(const BufferSource&, Ref<DeferredPromise>&&);
84 void close(Ref<DeferredPromise>&&);
85 void remove(Ref<DeferredPromise>&&);
86
87 using ClosedPromise = DOMPromiseProxy<IDLUndefined>;
88 ClosedPromise& closed() { return m_closedPromise.get(); }
89
90 const Vector<std::pair<Ref<SharedBuffer>, MediaKeyStatus>>& statuses() const { return m_statuses; }
91
92 unsigned internalInstanceSessionObjectRefCount() const { return m_instanceSession->refCount(); }
93
94private:
95 MediaKeySession(Document&, WeakPtr<MediaKeys>&&, MediaKeySessionType, bool useDistinctiveIdentifier, Ref<CDM>&&, Ref<CDMInstanceSession>&&);
96 void enqueueMessage(MediaKeyMessageType, const SharedBuffer&);
97 void updateExpiration(double);
98 void sessionClosed();
99 String mediaKeysStorageDirectory() const;
100
101 // CDMInstanceSessionClient
102 void updateKeyStatuses(CDMInstanceSessionClient::KeyStatusVector&&) override;
103 void sendMessage(CDMMessageType, Ref<SharedBuffer>&& message) final;
104 void sessionIdChanged(const String&) final;
105 PlatformDisplayID displayID() final;
106
107 // EventTarget
108 EventTargetInterface eventTargetInterface() const override { return MediaKeySessionEventTargetInterfaceType; }
109 ScriptExecutionContext* scriptExecutionContext() const override { return ActiveDOMObject::scriptExecutionContext(); }
110 void refEventTarget() override { ref(); }
111 void derefEventTarget() override { deref(); }
112
113 // ActiveDOMObject
114 const char* activeDOMObjectName() const final;
115 bool virtualHasPendingActivity() const final;
116
117 // DisplayChangedObserver
118 void displayChanged(PlatformDisplayID);
119
120#if !RELEASE_LOG_DISABLED
121 // LoggerHelper
122 const WTF::Logger& logger() const { return m_logger; }
123 const char* logClassName() const { return "MediaKeySession"; }
124 WTFLogChannel& logChannel() const;
125 const void* logIdentifier() const { return m_logIdentifier; }
126
127 Ref<WTF::Logger> m_logger;
128 const void* m_logIdentifier;
129#endif
130
131 WeakPtr<MediaKeys> m_keys;
132 String m_sessionId;
133 double m_expiration;
134 UniqueRef<ClosedPromise> m_closedPromise;
135 Ref<MediaKeyStatusMap> m_keyStatuses;
136 bool m_closed { false };
137 bool m_uninitialized { true };
138 bool m_callable { false };
139 bool m_useDistinctiveIdentifier;
140 MediaKeySessionType m_sessionType;
141 Ref<CDM> m_implementation;
142 Ref<CDMInstanceSession> m_instanceSession;
143 Vector<Ref<SharedBuffer>> m_recordOfKeyUsage;
144 double m_firstDecryptTime { 0 };
145 double m_latestDecryptTime { 0 };
146 Vector<std::pair<Ref<SharedBuffer>, MediaKeyStatus>> m_statuses;
147
148 using DisplayChangedObserver = Observer<void(PlatformDisplayID)>;
149 DisplayChangedObserver m_displayChangedObserver;
150};
