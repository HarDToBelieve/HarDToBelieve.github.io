59void MediaKeySystemAccess::createMediaKeys(Document& document, Ref<DeferredPromise>&& promise)
60{
61 // https://w3c.github.io/encrypted-media/#dom-mediakeysystemaccess-createmediakeys
62 // W3C Editor's Draft 09 November 2016
63
64 // When this method is invoked, the user agent must run the following steps:
65 // 1. Let promise be a new promise.
66 // 2. Run the following steps in parallel:
67 document.eventLoop().queueTask(TaskSource::MediaElement, [this, weakDocument = makeWeakPtr(document), promise = WTFMove(promise)] () mutable {
68 // 2.1. Let configuration be the value of this object's configuration value.
69 // 2.2. Let use distinctive identifier be true if the value of configuration's distinctiveIdentifier member is "required" and false otherwise.
70 bool useDistinctiveIdentifier = m_configuration->distinctiveIdentifier == MediaKeysRequirement::Required;
71
72 // 2.3. Let persistent state allowed be true if the value of configuration's persistentState member is "required" and false otherwise.
73 bool persistentStateAllowed = m_configuration->persistentState == MediaKeysRequirement::Required;
74
75 // 2.4. Load and initialize the Key System implementation represented by this object's cdm implementation value if necessary.
76 m_implementation->loadAndInitialize();
77
78 // 2.5. Let instance be a new instance of the Key System implementation represented by this object's cdm implementation value.
79 auto instance = m_implementation->createInstance();
80 if (!instance) {
81 promise->reject(InvalidStateError);
82 return;
83 }
84
85 // 2.6. Initialize instance to enable, disable and/or select Key System features using configuration.
86 // 2.7. If use distinctive identifier is false, prevent instance from using Distinctive Identifier(s) and Distinctive Permanent Identifier(s).
87 // 2.8. If persistent state allowed is false, prevent instance from persisting any state related to the application or origin of this object's Document.
88 auto allowDistinctiveIdentifiers = useDistinctiveIdentifier ? CDMInstance::AllowDistinctiveIdentifiers::Yes : CDMInstance::AllowDistinctiveIdentifiers::No;
89 auto allowPersistentState = persistentStateAllowed ? CDMInstance::AllowPersistentState::Yes : CDMInstance::AllowPersistentState::No;
90
91 instance->initializeWithConfiguration(*m_configuration, allowDistinctiveIdentifiers, allowPersistentState, [weakDocument = WTFMove(weakDocument), sessionTypes = m_configuration->sessionTypes, implementation = m_implementation.copyRef(), useDistinctiveIdentifier, persistentStateAllowed, instance = instance.releaseNonNull(), promise = WTFMove(promise)] (auto successValue) mutable {
92 if (successValue == CDMInstance::Failed || !weakDocument) {
93 promise->reject(NotAllowedError);
94 return;
95 }
96
97 // 2.9. If any of the preceding steps failed, reject promise with a new DOMException whose name is the appropriate error name.
98 // 2.10. Let media keys be a new MediaKeys object, and initialize it as follows:
99 // 2.10.1. Let the use distinctive identifier value be use distinctive identifier.
100 // 2.10.2. Let the persistent state allowed value be persistent state allowed.
101 // 2.10.3. Let the supported session types value be be the value of configuration's sessionTypes member.
102 // 2.10.4. Let the cdm implementation value be this object's cdm implementation value.
103 // 2.10.5. Let the cdm instance value be instance.
104 auto mediaKeys = MediaKeys::create(*weakDocument, useDistinctiveIdentifier, persistentStateAllowed, sessionTypes, WTFMove(implementation), WTFMove(instance));
105
106 // 2.11. Resolve promise with media keys.
107 promise->resolveWithNewlyCreated<IDLInterface<MediaKeys>>(WTFMove(mediaKeys));
108 });
109 });
110
111 // 3. Return promise.
112}
