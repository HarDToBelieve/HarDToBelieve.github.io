38namespace WebCore {
39
40const char* MediaKeySystemController::supplementName()
41{
42 return "MediaKeySystemController";
43}
44
45MediaKeySystemController* MediaKeySystemController::from(Page* page)
46{
47 return static_cast<MediaKeySystemController*>(Supplement<Page>::from(page, MediaKeySystemController::supplementName()));
48}
49
50MediaKeySystemController::MediaKeySystemController(MediaKeySystemClient& client)
51 : m_client(makeWeakPtr(client))
52{
53}
54
55MediaKeySystemController::~MediaKeySystemController()
56{
57 if (m_client)
58 m_client->pageDestroyed();
59}
60
61void provideMediaKeySystemTo(Page& page, MediaKeySystemClient& client)
62{
63 MediaKeySystemController::provideTo(&page, MediaKeySystemController::supplementName(), makeUnique<MediaKeySystemController>(client));
64}
65
66void MediaKeySystemController::logRequestMediaKeySystemDenial(Document& document)
67{
68 if (auto* window = document.domWindow())
69 window->printErrorMessage(makeString("Not allowed to access MediaKeySystem."));
70}
