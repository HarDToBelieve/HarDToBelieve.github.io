42Ref<MediaKeySystemRequest> MediaKeySystemRequest::create(Document& document, const String& keySystem, Ref<DeferredPromise>&& promise)
43{
44 auto result = adoptRef(*new MediaKeySystemRequest(document, keySystem, WTFMove(promise)));
45 result->suspendIfNeeded();
46 return result;
47}
48
49MediaKeySystemRequest::MediaKeySystemRequest(Document& document, const String& keySystem, Ref<DeferredPromise>&& promise)
50 : ActiveDOMObject(document)
51 , m_identifier(MediaKeySystemRequestIdentifier::generate())
52 , m_keySystem(keySystem)
53 , m_promise(WTFMove(promise))
54{
55}
56
57MediaKeySystemRequest::~MediaKeySystemRequest()
58{
59 if (m_allowCompletionHandler)
60 m_allowCompletionHandler(WTFMove(m_promise));
61}
62
63SecurityOrigin* MediaKeySystemRequest::topLevelDocumentOrigin() const
64{
65 if (!m_scriptExecutionContext)
66 return nullptr;
67 return &m_scriptExecutionContext->topOrigin();
68}
69
70void MediaKeySystemRequest::start()
71{
72 ASSERT(m_scriptExecutionContext);
73 if (!m_scriptExecutionContext) {
74 deny();
75 return;
76 }
77
78 auto& document = downcast<Document>(*m_scriptExecutionContext);
79 auto* controller = MediaKeySystemController::from(document.page());
80 if (!controller) {
81 deny();
82 return;
83 }
84
85 controller->requestMediaKeySystem(*this);
86}
87
88void MediaKeySystemRequest::allow(CompletionHandler<void()>&& completionHandler)
89{
90 queueTaskKeepingObjectAlive(*this, TaskSource::UserInteraction, [this, handler = WTFMove(completionHandler)]() mutable {
91 auto completionHandler = WTFMove(m_allowCompletionHandler);
92 completionHandler(WTFMove(m_promise));
93 handler();
94 });
95}
96
97void MediaKeySystemRequest::deny(const String& message)
98{
99 if (!m_scriptExecutionContext)
100 return;
101
102 ExceptionCode code = NotSupportedError;
103 if (!message.isEmpty())
104 m_promise->reject(code, message);
105 else
106 m_promise->reject(code);
107}
108
109void MediaKeySystemRequest::stop()
110{
111 auto& document = downcast<Document>(*m_scriptExecutionContext);
112 if (auto* controller = MediaKeySystemController::from(document.page()))
113 controller->cancelMediaKeySystemRequest(*this);
114}
115
116const char* MediaKeySystemRequest::activeDOMObjectName() const
117{
118 return "MediaKeySystemRequest";
119}
120
121Document* MediaKeySystemRequest::document() const
122{
123 return downcast<Document>(m_scriptExecutionContext);
124}
