74ExceptionOr<Ref<MediaKeySession>> MediaKeys::createSession(Document& document, MediaKeySessionType sessionType)
75{
76 // https://w3c.github.io/encrypted-media/#dom-mediakeys-setservercertificate
77 // W3C Editor's Draft 09 November 2016
78 auto identifier = LOGIDENTIFIER;
79 INFO_LOG(identifier, "EME - check if a new session can be created");
80
81 // When this method is invoked, the user agent must run the following steps:
82 // 1. If this object's supported session types value does not contain sessionType, throw [WebIDL] a NotSupportedError.
83 if (!m_supportedSessionTypes.contains(sessionType)) {
84 INFO_LOG(identifier, "Exception: unsupported sessionType: ", sessionType);
85 return Exception(NotSupportedError);
86 }
87
88 // 2. If the implementation does not support MediaKeySession operations in the current state, throw [WebIDL] an InvalidStateError.
89 if (!m_implementation->supportsSessions()) {
90 INFO_LOG(identifier, "Exception: implementation does not support sessions");
91 return Exception(InvalidStateError);
92 }
93
94 auto instanceSession = m_instance->createSession();
95 if (!instanceSession) {
96 INFO_LOG(identifier, "Exception: could not create session");
97 return Exception(InvalidStateError);
98 }
99
100 // 3. Let session be a new MediaKeySession object, and initialize it as follows:
101 // NOTE: Continued in MediaKeySession.
102 // 4. Return session.
103 auto session = MediaKeySession::create(document, makeWeakPtr(*this), sessionType, m_useDistinctiveIdentifier, m_implementation.copyRef(), instanceSession.releaseNonNull());
104 INFO_LOG(identifier, "Created session");
105 m_sessions.append(session.copyRef());
106 return session;
107}

109void MediaKeys::setServerCertificate(const BufferSource& serverCertificate, Ref<DeferredPromise>&& promise)
110{
111 // https://w3c.github.io/encrypted-media/#dom-mediakeys-setservercertificate
112 // W3C Editor's Draft 09 November 2016
113 auto identifier = LOGIDENTIFIER;
114
115 // When this method is invoked, the user agent must run the following steps:
116 // 1. If the Key System implementation represented by this object's cdm implementation value does not support
117 // server certificates, return a promise resolved with false.
118 if (!m_implementation->supportsServerCertificates()) {
119 INFO_LOG(identifier, "Rejected: !supportsServerCertificates()");
120 promise->resolve<IDLBoolean>(false);
121 return;
122 }
123
124 // 2. If serverCertificate is an empty array, return a promise rejected with a new a newly created TypeError.
125 if (!serverCertificate.length()) {
126 INFO_LOG(identifier, "Rejected: empty serverCertificate");
127 promise->reject(TypeError);
128 return;
129 }
130
131 // 3. Let certificate be a copy of the contents of the serverCertificate parameter.
132 auto certificate = SharedBuffer::create(serverCertificate.data(), serverCertificate.length());
133
134 // 4. Let promise be a new promise.
135 // 5. Run the following steps in parallel:
136
137 // 5.1. Use this object's cdm instance to process certificate.
138 m_instance->setServerCertificate(WTFMove(certificate), [this, protectedThis = makeRef(*this), promise = WTFMove(promise), identifier = WTFMove(identifier)] (auto success) {
139 // 5.2. If the preceding step failed, resolve promise with a new DOMException whose name is the appropriate error name.
140 // 5.1. [Else,] Resolve promise with true.
141 if (success == CDMInstance::Failed) {
142 INFO_LOG(identifier, "::task() - Rejected, setServerCertificate() failed");
143 promise->reject(InvalidStateError);
144 return;
145 }
146
147 INFO_LOG(identifier, "::task() - Resolved");
148 promise->resolve<IDLBoolean>(true);
149 });
150
151 // 6. Return promise.
152}

154void MediaKeys::attachCDMClient(CDMClient& client)
155{
156 ASSERT(!m_cdmClients.contains(client));
157 m_cdmClients.add(client);
158}
159
160void MediaKeys::detachCDMClient(CDMClient& client)
161{
162 ASSERT(m_cdmClients.contains(client));
163 m_cdmClients.remove(client);
164}

166void MediaKeys::attemptToResumePlaybackOnClients()
167{
168 for (auto& cdmClient : m_cdmClients)
169 cdmClient.cdmClientAttemptToResumePlaybackIfNecessary();
170}

172bool MediaKeys::hasOpenSessions() const
173{
174 return std::any_of(m_sessions.begin(), m_sessions.end(),
175 [](auto& session) {
176 return !session->isClosed();
177 });
178}

180void MediaKeys::unrequestedInitializationDataReceived(const String& initDataType, Ref<SharedBuffer>&& initData)
181{
182 for (auto& cdmClient : m_cdmClients)
183 cdmClient.cdmClientUnrequestedInitializationDataReceived(initDataType, initData.copyRef());
184}
