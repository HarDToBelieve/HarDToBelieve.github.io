45namespace WebCore {

47class CDM;
48class CDMClient;
49class CDMInstance;
50class BufferSource;
51class DeferredPromise;
52class Document;
53class MediaKeySession;
54
55class MediaKeys final
56 : public RefCounted<MediaKeys>
57 , public CDMInstanceClient {
58public:
59 using KeySessionType = MediaKeySessionType;
60
61 static Ref<MediaKeys> create(Document& document, bool useDistinctiveIdentifier, bool persistentStateAllowed, const Vector<MediaKeySessionType>& supportedSessionTypes, Ref<CDM>&& implementation, Ref<CDMInstance>&& instance)
62 {
63 return adoptRef(*new MediaKeys(document, useDistinctiveIdentifier, persistentStateAllowed, supportedSessionTypes, WTFMove(implementation), WTFMove(instance)));
64 }
65
66 ~MediaKeys();
67
68 ExceptionOr<Ref<MediaKeySession>> createSession(Document&, MediaKeySessionType);
69 void setServerCertificate(const BufferSource&, Ref<DeferredPromise>&&);
70
71 void attachCDMClient(CDMClient&);
72 void detachCDMClient(CDMClient&);
73 void attemptToResumePlaybackOnClients();
74
75 bool hasOpenSessions() const;
76 CDMInstance& cdmInstance() { return m_instance; }
77 const CDMInstance& cdmInstance() const { return m_instance; }
78
79#if !RELEASE_LOG_DISABLED
80 const void* nextChildIdentifier() const;
81#endif
82
83 unsigned internalInstanceObjectRefCount() const { return m_instance->refCount(); }
84
85protected:
86 MediaKeys(Document&, bool useDistinctiveIdentifier, bool persistentStateAllowed, const Vector<MediaKeySessionType>&, Ref<CDM>&&, Ref<CDMInstance>&&);
87
88 // CDMInstanceClient
89 void unrequestedInitializationDataReceived(const String&, Ref<SharedBuffer>&&) final;
90
91#if !RELEASE_LOG_DISABLED
92 const WTF::Logger& logger() const { return m_logger; }
93 const void* logIdentifier() const { return m_logIdentifier; }
94#endif
95
96 bool m_useDistinctiveIdentifier;
97 bool m_persistentStateAllowed;
98 Vector<MediaKeySessionType> m_supportedSessionTypes;
99 Ref<CDM> m_implementation;
100 Ref<CDMInstance> m_instance;
101
102 Vector<Ref<MediaKeySession>> m_sessions;
103 WeakHashSet<CDMClient> m_cdmClients;
104
105#if !RELEASE_LOG_DISABLED
106 Ref<WTF::Logger> m_logger;
107 const void* m_logIdentifier;
108 mutable uint64_t m_childIdentifierSeed { 0 };
109#endif
110};
