36class CDMSessionClearKey final : public LegacyCDMSession {

48 bool update(Uint8Array*, RefPtr<Uint8Array>&, unsigned short&, uint32_t&) override;

51private:
52 LegacyCDMSessionClient* m_client;
53 RefPtr<Uint8Array> m_initData;

55 String m_sessionId;
56};
