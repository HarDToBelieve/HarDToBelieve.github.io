34namespace WebCore {
35
36class WebKitMediaKeyMessageEvent final : public Event {
37 WTF_MAKE_ISO_ALLOCATED(WebKitMediaKeyMessageEvent);
38public:
39 virtual ~WebKitMediaKeyMessageEvent();
40
41 static Ref<WebKitMediaKeyMessageEvent> create(const AtomString& type, Uint8Array* message, const String& destinationURL)
42 {
43 return adoptRef(*new WebKitMediaKeyMessageEvent(type, message, destinationURL));
44 }
45
46 struct Init : EventInit {
47 RefPtr<Uint8Array> message;
48 String destinationURL;
49 };
50
51 static Ref<WebKitMediaKeyMessageEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
52 {
53 return adoptRef(*new WebKitMediaKeyMessageEvent(type, initializer, isTrusted));
54 }
55
56 EventInterface eventInterface() const override;
57
58 Uint8Array* message() const { return m_message.get(); }
59 String destinationURL() const { return m_destinationURL; }
60
61private:
62 WebKitMediaKeyMessageEvent(const AtomString& type, Uint8Array* message, const String& destinationURL);
63 WebKitMediaKeyMessageEvent(const AtomString& type, const Init&, IsTrusted);
64
65 RefPtr<Uint8Array> m_message;
66 String m_destinationURL;
67};
