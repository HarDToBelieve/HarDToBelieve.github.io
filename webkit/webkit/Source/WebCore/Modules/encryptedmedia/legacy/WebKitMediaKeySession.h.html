38namespace WebCore {
39
40class WebKitMediaKeyError;
41class WebKitMediaKeys;
42
43class WebKitMediaKeySession final : public RefCounted<WebKitMediaKeySession>, public EventTargetWithInlineData, public ActiveDOMObject, private LegacyCDMSessionClient {
44 WTF_MAKE_ISO_ALLOCATED(WebKitMediaKeySession);
45public:
46 static Ref<WebKitMediaKeySession> create(ScriptExecutionContext&, WebKitMediaKeys&, const String& keySystem);
47 ~WebKitMediaKeySession();
48
49 WebKitMediaKeyError* error() { return m_error.get(); }
50 const String& keySystem() const { return m_keySystem; }
51 const String& sessionId() const { return m_sessionId; }
52 ExceptionOr<void> update(Ref<Uint8Array>&& key);
53 void close();
54
55 LegacyCDMSession* session() { return m_session.get(); }
56
57 void detachKeys() { m_keys = nullptr; }
58
59 void generateKeyRequest(const String& mimeType, Ref<Uint8Array>&& initData);
60 RefPtr<ArrayBuffer> cachedKeyForKeyId(const String& keyId) const;
61
62 using RefCounted::ref;
63 using RefCounted::deref;
64
65private:
66 WebKitMediaKeySession(ScriptExecutionContext&, WebKitMediaKeys&, const String& keySystem);
67 void keyRequestTimerFired();
68 void addKeyTimerFired();
69
70 void sendMessage(Uint8Array*, String destinationURL) final;
71 void sendError(MediaKeyErrorCode, uint32_t systemCode) final;
72 String mediaKeysStorageDirectory() const final;
73
74 void refEventTarget() final { ref(); }
75 void derefEventTarget() final { deref(); }
76
77 // ActiveDOMObject.
78 void stop() final;
79 const char* activeDOMObjectName() const final;
80 bool virtualHasPendingActivity() const final;
81
82 EventTargetInterface eventTargetInterface() const final { return WebKitMediaKeySessionEventTargetInterfaceType; }
83 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
84
85 WebKitMediaKeys* m_keys;
86 String m_keySystem;
87 String m_sessionId;
88 RefPtr<WebKitMediaKeyError> m_error;
89 std::unique_ptr<LegacyCDMSession> m_session;
90
91 struct PendingKeyRequest {
92 String mimeType;
93 Ref<Uint8Array> initData;
94 };
95 Deque<PendingKeyRequest> m_pendingKeyRequests;
96 Timer m_keyRequestTimer;
97
98 Deque<Ref<Uint8Array>> m_pendingKeys;
99 Timer m_addKeyTimer;
100};

102}
