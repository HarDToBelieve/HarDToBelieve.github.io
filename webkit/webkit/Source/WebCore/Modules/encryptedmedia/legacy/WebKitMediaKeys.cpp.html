36ExceptionOr<Ref<WebKitMediaKeys>> WebKitMediaKeys::create(const String& keySystem)
37{
38 // From <http://dvcs.w3.org/hg/html-media/raw-file/tip/encrypted-media/encrypted-media.html#dom-media-keys-constructor>:
39 // The MediaKeys(keySystem) constructor must run the following steps:
40
41 // 1. If keySystem is null or an empty string, throw an InvalidAccessError exception and abort these steps.
42 if (keySystem.isEmpty())
43 return Exception { InvalidAccessError };
44
45 // 2. If keySystem is not one of the user agent's supported Key Systems, throw a NotSupportedError and abort these steps.
46 if (!LegacyCDM::supportsKeySystem(keySystem))
47 return Exception { NotSupportedError };
48
49 // 3. Let cdm be the content decryption module corresponding to keySystem.
50 // 4. Load cdm if necessary.
51 auto cdm = LegacyCDM::create(keySystem);
52
53 // 5. Create a new MediaKeys object.
54 // 5.1 Let the keySystem attribute be keySystem.
55 // 6. Return the new object to the caller.
56 return adoptRef(*new WebKitMediaKeys(keySystem, WTFMove(cdm)));
57}

66WebKitMediaKeys::~WebKitMediaKeys()
67{
68 // From <http://dvcs.w3.org/hg/html-media/raw-file/tip/encrypted-media/encrypted-media.html#dom-media-keys-constructor>:

76ExceptionOr<Ref<WebKitMediaKeySession>> WebKitMediaKeys::createSession(ScriptExecutionContext& context, const String& type, Ref<Uint8Array>&& initData)
77{
78 // From <http://www.w3.org/TR/2014/WD-encrypted-media-20140218/#dom-createsession>:
79 // The createSession(type, initData) method must run the following steps:
80 // Note: The contents of initData are container-specific Initialization Data.
81
82 // 1. If contentType is null or an empty string, throw an InvalidAccessError exception and abort these steps.
83 if (type.isEmpty())
84 return Exception { InvalidAccessError };
85
86 // 2. If initData is an empty array, throw an InvalidAccessError exception and abort these steps.
87 if (!initData->length())
88 return Exception { InvalidAccessError };
89
90 // 3. If type contains a MIME type that is not supported or is not supported by the keySystem, throw
91 // a NotSupportedError exception and abort these steps.
92 if (!m_cdm->supportsMIMEType(type))
93 return Exception { NotSupportedError };
94
95 // 4. Create a new MediaKeySession object.
96 // 4.1 Let the keySystem attribute be keySystem.
97 // 4.2 Let the sessionId attribute be a unique Session ID string. It may be generated by cdm.
98 auto session = WebKitMediaKeySession::create(context, *this, m_keySystem);
99
100 m_sessions.append(session.copyRef());
101
102 // 5. Schedule a task to initialize the session, providing contentType, initData, and the new object.
103 session->generateKeyRequest(type, WTFMove(initData));
104
105 // 6. Return the new object to the caller.
106 return WTFMove(session);
107}

109bool WebKitMediaKeys::isTypeSupported(const String& keySystem, const String& mimeType)
110{
111 // 1. If keySystem contains an unrecognized or unsupported Key System, return false and abort these steps.
112 // Key system string comparison is case-sensitive.
113 if (keySystem.isEmpty() || !LegacyCDM::supportsKeySystem(keySystem))
114 return false;
115
116 // 2. If type is null or an empty string, return true and abort these steps.
117 if (mimeType.isEmpty())
118 return true;
119
120 // 3. If the Key System specified by keySystem does not support decrypting the container and/or codec
121 // specified by type, return false and abort these steps.
122 if (!LegacyCDM::keySystemSupportsMimeType(keySystem, mimeType))
123 return false;
124
125 // 4. Return true;
126 return true;
127}

129void WebKitMediaKeys::setMediaElement(HTMLMediaElement* element)
130{
131 if (m_mediaElement && m_mediaElement->player()) {
132 m_mediaElement->player()->setCDMSession(nullptr);
133 m_mediaElement->player()->setCDM(nullptr);
134 }
135
136 m_mediaElement = makeWeakPtr(element);
137
138 if (m_mediaElement && m_mediaElement->player()) {
139 m_mediaElement->player()->setCDM(m_cdm.get());
140 if (!m_sessions.isEmpty())
141 m_mediaElement->player()->setCDMSession(m_sessions.last()->session());
142 }
143}
144
145RefPtr<MediaPlayer> WebKitMediaKeys::cdmMediaPlayer(const LegacyCDM*) const
146{
147 if (!m_mediaElement)
148 return nullptr;
149 return m_mediaElement->player();
150}

152void WebKitMediaKeys::keyAdded()

154 if (m_mediaElement)
155 m_mediaElement->keyAdded();
156}

158RefPtr<ArrayBuffer> WebKitMediaKeys::cachedKeyForKeyId(const String& keyId) const
159{
160 for (auto& session : m_sessions) {
161 if (auto key = session->cachedKeyForKeyId(keyId))
162 return key;
163 }
164 return nullptr;
165}
