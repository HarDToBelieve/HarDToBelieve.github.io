43struct ListedChild {
44 String filename;
45 FileSystem::FileType type;
46
47 ListedChild isolatedCopy() const { return { filename.isolatedCopy(), type }; }
48};

50static std::optional<FileSystem::FileType> fileTypeIgnoringHiddenFiles(const String& fullPath)
51{
52 if (FileSystem::isHiddenFile(fullPath))
53 return std::nullopt;
54 return FileSystem::fileType(fullPath);
55}
56
57static ExceptionOr<Vector<ListedChild>> listDirectoryWithMetadata(const String& fullPath)
58{
59 ASSERT(!isMainThread());
60 if (FileSystem::fileType(fullPath) != FileSystem::FileType::Directory)
61 return Exception { NotFoundError, "Path no longer exists or is no longer a directory" };
62
63 auto childNames = FileSystem::listDirectory(fullPath);
64 Vector<ListedChild> listedChildren;
65 listedChildren.reserveInitialCapacity(childNames.size());
66 for (auto& childName : childNames) {
67 auto childPath = FileSystem::pathByAppendingComponent(fullPath, childName);
68 auto fileType = fileTypeIgnoringHiddenFiles(childPath);
69 if (!fileType)
70 continue;
71 listedChildren.uncheckedAppend(ListedChild { childName, *fileType });
72 }
73 return listedChildren;
74}

76static ExceptionOr<Vector<Ref<FileSystemEntry>>> toFileSystemEntries(ScriptExecutionContext& context, DOMFileSystem& fileSystem, ExceptionOr<Vector<ListedChild>>&& listedChildren, const String& parentVirtualPath)
77{
78 ASSERT(isMainThread());
79 if (listedChildren.hasException())
80 return listedChildren.releaseException();
81
82 Vector<Ref<FileSystemEntry>> entries;
83 entries.reserveInitialCapacity(listedChildren.returnValue().size());
84 for (auto& child : listedChildren.returnValue()) {
85 String virtualPath = parentVirtualPath + "/" + child.filename;
86 switch (child.type) {
87 case FileSystem::FileType::Regular:
88 entries.uncheckedAppend(FileSystemFileEntry::create(context, fileSystem, virtualPath));
89 break;
90 case FileSystem::FileType::Directory:
91 entries.uncheckedAppend(FileSystemDirectoryEntry::create(context, fileSystem, virtualPath));
92 break;
93 default:
94 break;
95 }
96 }
97 return entries;
98}

100// https://wicg.github.io/entries-api/#name
101static bool isValidPathNameCharacter(UChar c)
102{
103 return c != '\0' && c != '/' && c != '\\';
104}

106// https://wicg.github.io/entries-api/#path-segment
107static bool isValidPathSegment(StringView segment)
108{
109 if (segment.isEmpty() || segment == "." || segment == "..")
110 return true;
111
112 for (unsigned i = 0; i < segment.length(); ++i) {
113 if (!isValidPathNameCharacter(segment[i]))
114 return false;
115 }
116 return true;
117}
118
119static bool isZeroOrMorePathSegmentsSeparatedBySlashes(StringView string)
120{
121 auto segments = string.split('/');
122 for (auto segment : segments) {
123 if (!isValidPathSegment(segment))
124 return false;
125 }
126 return true;
127}

129// https://wicg.github.io/entries-api/#relative-path
130static bool isValidRelativeVirtualPath(StringView virtualPath)
131{
132 if (virtualPath.isEmpty())
133 return false;
134
135 if (virtualPath[0] == '/')
136 return false;
137
138 return isZeroOrMorePathSegmentsSeparatedBySlashes(virtualPath);
139}
140
141// https://wicg.github.io/entries-api/#valid-path
142static bool isValidVirtualPath(StringView virtualPath)
143{
144 if (virtualPath.isEmpty())
145 return true;
146 if (virtualPath[0] == '/') {
147 // An absolute path is a string consisting of '/' (U+002F SOLIDUS) followed by one or more path segments joined by '/' (U+002F SOLIDUS).
148 return isZeroOrMorePathSegmentsSeparatedBySlashes(virtualPath.substring(1));
149 }
150 return isValidRelativeVirtualPath(virtualPath);
151}

169Ref<FileSystemEntry> DOMFileSystem::fileAsEntry(ScriptExecutionContext& context)
170{
171 if (m_file->isDirectory())

176static ExceptionOr<String> validatePathIsExpectedType(const String& fullPath, String&& virtualPath, FileSystem::FileType expectedType)
177{
178 ASSERT(!isMainThread());
179
180 auto fileType = fileTypeIgnoringHiddenFiles(fullPath);
181 if (!fileType)
182 return Exception { NotFoundError, "Path does not exist"_s };
183
184 if (*fileType != expectedType)
185 return Exception { TypeMismatchError, "Entry at path does not have expected type" };
186
187 return WTFMove(virtualPath);
188}

190// https://wicg.github.io/entries-api/#resolve-a-relative-path
191static String resolveRelativeVirtualPath(StringView baseVirtualPath, StringView relativeVirtualPath)
192{
193 ASSERT(baseVirtualPath[0] == '/');
194 if (!relativeVirtualPath.isEmpty() && relativeVirtualPath[0] == '/')
195 return relativeVirtualPath.length() == 1 ? relativeVirtualPath.toString() : resolveRelativeVirtualPath("/", relativeVirtualPath.substring(1));
196
197 Vector<StringView> virtualPathSegments;
198 for (auto segment : baseVirtualPath.split('/'))
199 virtualPathSegments.append(segment);
200
201 for (auto segment : relativeVirtualPath.split('/')) {
202 ASSERT(!segment.isEmpty());
203 if (segment == ".")
204 continue;
205 if (segment == "..") {
206 if (!virtualPathSegments.isEmpty())
207 virtualPathSegments.removeLast();
208 continue;
209 }
210 virtualPathSegments.append(segment);
211 }
212
213 if (virtualPathSegments.isEmpty())
214 return "/"_s;
215
216 StringBuilder builder;
217 for (auto& segment : virtualPathSegments) {
218 builder.append('/');
219 builder.append(segment);
220 }
221 return builder.toString();
222}

224// https://wicg.github.io/entries-api/#evaluate-a-path
225String DOMFileSystem::evaluatePath(StringView virtualPath)
226{
227 ASSERT(virtualPath[0] == '/');
228
229 Vector<StringView> resolvedComponents;
230 for (auto component : virtualPath.split('/')) {
231 if (component == ".")
232 continue;
233 if (component == "..") {
234 if (!resolvedComponents.isEmpty())
235 resolvedComponents.removeLast();
236 continue;
237 }
238 resolvedComponents.append(component);
239 }
240
241 return FileSystem::pathByAppendingComponents(m_rootPath, resolvedComponents);
242}

244void DOMFileSystem::listDirectory(ScriptExecutionContext& context, FileSystemDirectoryEntry& directory, DirectoryListingCallback&& completionHandler)
245{
246 ASSERT(&directory.filesystem() == this);
247
248 auto directoryVirtualPath = directory.virtualPath();
249 auto fullPath = evaluatePath(directoryVirtualPath);
250 if (fullPath == m_rootPath) {
251 Vector<Ref<FileSystemEntry>> children;
252 children.append(fileAsEntry(context));
253 completionHandler(WTFMove(children));
254 return;
255 }
256
257 m_workQueue->dispatch([protectedThis = makeRef(*this), context = makeRef(context), completionHandler = WTFMove(completionHandler), fullPath = crossThreadCopy(fullPath), directoryVirtualPath = crossThreadCopy(directoryVirtualPath)]() mutable {
258 auto listedChildren = listDirectoryWithMetadata(fullPath);
259 callOnMainThread([protectedThis = WTFMove(protectedThis), context = WTFMove(context), completionHandler = WTFMove(completionHandler), listedChildren = crossThreadCopy(listedChildren), directoryVirtualPath = directoryVirtualPath.isolatedCopy()]() mutable {
260 completionHandler(toFileSystemEntries(context, protectedThis, WTFMove(listedChildren), directoryVirtualPath));
261 });
262 });
263}

265void DOMFileSystem::getParent(ScriptExecutionContext& context, FileSystemEntry& entry, GetParentCallback&& completionCallback)
266{
267 ASSERT(&entry.filesystem() == this);
268
269 auto virtualPath = resolveRelativeVirtualPath(entry.virtualPath(), "..");
270 ASSERT(virtualPath[0] == '/');
271 auto fullPath = evaluatePath(virtualPath);
272 m_workQueue->dispatch([protectedThis = makeRef(*this), context = makeRef(context), fullPath = crossThreadCopy(fullPath), virtualPath = crossThreadCopy(virtualPath), completionCallback = WTFMove(completionCallback)]() mutable {
273 auto validatedVirtualPath = validatePathIsExpectedType(fullPath, WTFMove(virtualPath), FileSystem::FileType::Directory);
274 callOnMainThread([protectedThis = WTFMove(protectedThis), context = WTFMove(context), validatedVirtualPath = crossThreadCopy(validatedVirtualPath), completionCallback = WTFMove(completionCallback)]() mutable {
275 if (validatedVirtualPath.hasException())
276 completionCallback(validatedVirtualPath.releaseException());
277 else
278 completionCallback(FileSystemDirectoryEntry::create(context, protectedThis.get(), validatedVirtualPath.releaseReturnValue()));
279 });
280 });
281}

283// https://wicg.github.io/entries-api/#dom-filesystemdirectoryentry-getfile
284// https://wicg.github.io/entries-api/#dom-filesystemdirectoryentry-getdirectory
285void DOMFileSystem::getEntry(ScriptExecutionContext& context, FileSystemDirectoryEntry& directory, const String& virtualPath, const FileSystemDirectoryEntry::Flags& flags, GetEntryCallback&& completionCallback)
286{
287 ASSERT(&directory.filesystem() == this);
288
289 if (!isValidVirtualPath(virtualPath)) {
290 callOnMainThread([completionCallback = WTFMove(completionCallback)] {
291 completionCallback(Exception { TypeMismatchError, "Path is invalid"_s });
292 });
293 return;
294 }
295
296 if (flags.create) {
297 callOnMainThread([completionCallback = WTFMove(completionCallback)] {
298 completionCallback(Exception { SecurityError, "create flag cannot be true"_s });
299 });
300 return;
301 }
302
303 auto resolvedVirtualPath = resolveRelativeVirtualPath(directory.virtualPath(), virtualPath);
304 ASSERT(resolvedVirtualPath[0] == '/');
305 auto fullPath = evaluatePath(resolvedVirtualPath);
306 if (fullPath == m_rootPath) {
307 callOnMainThread([this, context = makeRef(context), completionCallback = WTFMove(completionCallback)]() mutable {
308 completionCallback(Ref<FileSystemEntry> { root(context) });
309 });
310 return;
311 }
312
313 m_workQueue->dispatch([protectedThis = makeRef(*this), context = makeRef(context), fullPath = crossThreadCopy(fullPath), resolvedVirtualPath = crossThreadCopy(resolvedVirtualPath), completionCallback = WTFMove(completionCallback)]() mutable {
314 auto entryType = fileTypeIgnoringHiddenFiles(fullPath);
315 callOnMainThread([protectedThis = WTFMove(protectedThis), context = WTFMove(context), resolvedVirtualPath = crossThreadCopy(resolvedVirtualPath), entryType, completionCallback = WTFMove(completionCallback)]() mutable {
316 if (!entryType) {
317 completionCallback(Exception { NotFoundError, "Cannot find entry at given path"_s });
318 return;
319 }
320 switch (entryType.value()) {
321 case FileSystem::FileType::Directory:
322 completionCallback(Ref<FileSystemEntry> { FileSystemDirectoryEntry::create(context, protectedThis.get(), resolvedVirtualPath) });
323 break;
324 case FileSystem::FileType::Regular:
325 completionCallback(Ref<FileSystemEntry> { FileSystemFileEntry::create(context, protectedThis.get(), resolvedVirtualPath) });
326 break;
327 default:
328 completionCallback(Exception { NotFoundError, "Cannot find entry at given path"_s });
329 break;
330 }
331 });
332 });
333}

335void DOMFileSystem::getFile(ScriptExecutionContext& context, FileSystemFileEntry& fileEntry, GetFileCallback&& completionCallback)
336{
337 auto virtualPath = fileEntry.virtualPath();
338 auto fullPath = evaluatePath(virtualPath);
339 m_workQueue->dispatch([fullPath = crossThreadCopy(fullPath), virtualPath = crossThreadCopy(virtualPath), context = makeRef(context), completionCallback = WTFMove(completionCallback)]() mutable {
340 auto validatedVirtualPath = validatePathIsExpectedType(fullPath, WTFMove(virtualPath), FileSystem::FileType::Regular);
341 callOnMainThread([fullPath = crossThreadCopy(fullPath), validatedVirtualPath = crossThreadCopy(validatedVirtualPath), context = WTFMove(context), completionCallback = WTFMove(completionCallback)]() mutable {
342 if (validatedVirtualPath.hasException())
343 completionCallback(validatedVirtualPath.releaseException());
344 else
345 completionCallback(File::create(context.ptr(), fullPath));
346 });
347 });
348}
