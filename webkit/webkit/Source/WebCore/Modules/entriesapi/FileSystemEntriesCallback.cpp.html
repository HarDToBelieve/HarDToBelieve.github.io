34static Vector<Ref<FileSystemEntry>> copyVector(const Vector<Ref<FileSystemEntry>>& entries)
35{
36 Vector<Ref<FileSystemEntry>> entriesCopy;
37 entriesCopy.reserveInitialCapacity(entries.size());
38 for (auto& entry : entries)
39 entriesCopy.uncheckedAppend(entry.copyRef());
40 return entriesCopy;
41}
42
43void FileSystemEntriesCallback::scheduleCallback(ScriptExecutionContext& context, const Vector<Ref<FileSystemEntry>>& entries)
44{
45 context.postTask([protectedThis = makeRef(*this), entries = copyVector(entries)] (ScriptExecutionContext&) {
46 protectedThis->handleEvent(entries);
47 });
48}
