37Vector<Ref<FileSystemEntry>> HTMLInputElementEntriesAPI::webkitEntries(ScriptExecutionContext& context, HTMLInputElement& input)
38{
39 // As of Sept 2017, Chrome and Firefox both only populate webkitEntries when the webkitdirectory flag is unset.
40 // We do the same for consistency.
41 if (input.hasAttributeWithoutSynchronization(webkitdirectoryAttr))
42 return { };
43
44 auto* fileList = input.files();
45 if (!fileList)
46 return { };
47
48 Vector<Ref<FileSystemEntry>> entries;
49 entries.reserveInitialCapacity(fileList->files().size());
50 for (auto& file : fileList->files())
51 entries.uncheckedAppend(DOMFileSystem::createEntryForFile(context, file.copyRef()));
52 return entries;
53}
