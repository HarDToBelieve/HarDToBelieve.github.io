29#pragma once
30
31#include "DOMFormData.h"
32#include "ExceptionOr.h"
33#include "FetchBodyConsumer.h"
34#include "FormData.h"
35#include "ReadableStream.h"
36#include "URLSearchParams.h"
37#include <wtf/Variant.h>
38
39namespace WebCore {
40
41class DeferredPromise;
42class FetchBodyOwner;
43class FetchBodySource;
44class ScriptExecutionContext;
45
46class FetchBody {
47public:
48 void arrayBuffer(FetchBodyOwner&, Ref<DeferredPromise>&&);
49 void blob(FetchBodyOwner&, Ref<DeferredPromise>&&, const String&);
50 void json(FetchBodyOwner&, Ref<DeferredPromise>&&);
51 void text(FetchBodyOwner&, Ref<DeferredPromise>&&);
52 void formData(FetchBodyOwner&, Ref<DeferredPromise>&&);
53
54 void consumeAsStream(FetchBodyOwner&, FetchBodySource&);
55
56 using Init = Variant<RefPtr<Blob>, RefPtr<ArrayBufferView>, RefPtr<ArrayBuffer>, RefPtr<DOMFormData>, RefPtr<URLSearchParams>, RefPtr<ReadableStream>, String>;
57 static ExceptionOr<FetchBody> extract(Init&&, String&);
58 FetchBody() = default;
59
60 WEBCORE_EXPORT static std::optional<FetchBody> fromFormData(ScriptExecutionContext&, FormData&);
61
62 void loadingFailed(const Exception&);
63 void loadingSucceeded(const String& contentType);
64
65 RefPtr<FormData> bodyAsFormData() const;
66
67 using TakenData = Variant<std::nullptr_t, Ref<FormData>, Ref<SharedBuffer>>;
68 TakenData take();
69
70 void setAsFormData(Ref<FormData>&& data) { m_data = WTFMove(data); }
71 FetchBodyConsumer& consumer() { return m_consumer; }
72
73 void consumeOnceLoadingFinished(FetchBodyConsumer::Type, Ref<DeferredPromise>&&, const String&);
74 void cleanConsumer() { m_consumer.clean(); }
75
76 FetchBody clone();
77
78 bool hasReadableStream() const { return !!m_readableStream; }
79 const ReadableStream* readableStream() const { return m_readableStream.get(); }
80 ReadableStream* readableStream() { return m_readableStream.get(); }
81 void setReadableStream(Ref<ReadableStream>&& stream)
82 {
83 ASSERT(!m_readableStream);
84 m_readableStream = WTFMove(stream);
85 }
86
87 bool isBlob() const { return WTF::holds_alternative<Ref<const Blob>>(m_data); }
88 bool isFormData() const { return WTF::holds_alternative<Ref<FormData>>(m_data); }
89
90private:
91 explicit FetchBody(Ref<const Blob>&& data) : m_data(WTFMove(data)) { }
92 explicit FetchBody(Ref<const ArrayBuffer>&& data) : m_data(WTFMove(data)) { }
93 explicit FetchBody(Ref<const ArrayBufferView>&& data) : m_data(WTFMove(data)) { }
94 explicit FetchBody(Ref<FormData>&& data) : m_data(WTFMove(data)) { }
95 explicit FetchBody(String&& data) : m_data(WTFMove(data)) { }
96 explicit FetchBody(Ref<const URLSearchParams>&& data) : m_data(WTFMove(data)) { }
97 explicit FetchBody(const FetchBodyConsumer& consumer) : m_consumer(consumer) { }
98 explicit FetchBody(Ref<ReadableStream>&& stream) : m_readableStream(WTFMove(stream)) { }
99
100 void consume(FetchBodyOwner&, Ref<DeferredPromise>&&);
101
102 void consumeArrayBuffer(FetchBodyOwner&, Ref<DeferredPromise>&&);
103 void consumeArrayBufferView(FetchBodyOwner&, Ref<DeferredPromise>&&);
104 void consumeText(FetchBodyOwner&, Ref<DeferredPromise>&&, const String&);
105 void consumeBlob(FetchBodyOwner&, Ref<DeferredPromise>&&);
106 void consumeFormData(FetchBodyOwner&, Ref<DeferredPromise>&&);
107
108 bool isArrayBuffer() const { return WTF::holds_alternative<Ref<const ArrayBuffer>>(m_data); }
109 bool isArrayBufferView() const { return WTF::holds_alternative<Ref<const ArrayBufferView>>(m_data); }
110 bool isURLSearchParams() const { return WTF::holds_alternative<Ref<const URLSearchParams>>(m_data); }
111 bool isText() const { return WTF::holds_alternative<String>(m_data); }
112
113 const Blob& blobBody() const { return WTF::get<Ref<const Blob>>(m_data).get(); }
114 FormData& formDataBody() { return WTF::get<Ref<FormData>>(m_data).get(); }
115 const FormData& formDataBody() const { return WTF::get<Ref<FormData>>(m_data).get(); }
116 const ArrayBuffer& arrayBufferBody() const { return WTF::get<Ref<const ArrayBuffer>>(m_data).get(); }
117 const ArrayBufferView& arrayBufferViewBody() const { return WTF::get<Ref<const ArrayBufferView>>(m_data).get(); }
118 String& textBody() { return WTF::get<String>(m_data); }
119 const String& textBody() const { return WTF::get<String>(m_data); }
120 const URLSearchParams& urlSearchParamsBody() const { return WTF::get<Ref<const URLSearchParams>>(m_data).get(); }
121
122 using Data = Variant<std::nullptr_t, Ref<const Blob>, Ref<FormData>, Ref<const ArrayBuffer>, Ref<const ArrayBufferView>, Ref<const URLSearchParams>, String>;
123 Data m_data { nullptr };
124
125 FetchBodyConsumer m_consumer { FetchBodyConsumer::Type::None };
126 RefPtr<ReadableStream> m_readableStream;
127};
128
129} // namespace WebCore
