48// https://mimesniff.spec.whatwg.org/#http-quoted-string-token-code-point
49static bool isHTTPQuotedStringTokenCodePoint(UChar c)
50{
51 return c == 0x09
52 || (c >= 0x20 && c <= 0x7E)
53 || (c >= 0x80 && c <= 0xFF);
54}

56struct MimeType {
57 String type;
58 String subtype;
59 HashMap<String, String> parameters;
60};

62static HashMap<String, String> parseParameters(StringView input, size_t position)
63{
64 HashMap<String, String> parameters;
65 while (position < input.length()) {
66 while (position < input.length() && RFC7230::isWhitespace(input[position]))
67 position++;
68 size_t nameBegin = position;
69 while (position < input.length() && input[position] != '=' && input[position] != ';')
70 position++;
71 if (position >= input.length())
72 break;
73 if (input[position] == ';') {
74 position++;
75 continue;
76 }
77 StringView parameterName = input.substring(nameBegin, position - nameBegin);
78 position++;
79 if (position >= input.length())
80 break;
81
82 StringView parameterValue;
83 if (position < input.length() && input[position] == '"') {
84 position++;
85 size_t valueBegin = position;
86 while (position < input.length() && input[position] != '"')
87 position++;
88 parameterValue = input.substring(valueBegin, position - valueBegin);
89 position++;
90 } else {
91 size_t valueBegin = position;
92 while (position < input.length() && input[position] != ';')
93 position++;
94 parameterValue = stripLeadingAndTrailingHTTPSpaces(input.substring(valueBegin, position - valueBegin));
95 }
96
97 if (parameterName.length()
98 && isValidHTTPToken(parameterName)
99 && parameterValue.isAllSpecialCharacters<isHTTPQuotedStringTokenCodePoint>()) {
100 parameters.ensure(parameterName.toString(), [&] { return parameterValue.toString(); });
101 }
102 }
103 return parameters;
104}

106// https://mimesniff.spec.whatwg.org/#parsing-a-mime-type
107static std::optional<MimeType> parseMIMEType(const String& contentType)
108{
109 String input = stripLeadingAndTrailingHTTPSpaces(contentType);
110 size_t slashIndex = input.find('/');
111 if (slashIndex == notFound)
112 return std::nullopt;
113
114 String type = input.substring(0, slashIndex);
115 if (!type.length() || !isValidHTTPToken(type))
116 return std::nullopt;
117
118 size_t semicolonIndex = input.find(';', slashIndex);
119 String subtype = stripLeadingAndTrailingHTTPSpaces(input.substring(slashIndex + 1, semicolonIndex - slashIndex - 1));
120 if (!subtype.length() || !isValidHTTPToken(subtype))
121 return std::nullopt;
122
123 return {{ WTFMove(type), WTFMove(subtype), parseParameters(StringView(input), semicolonIndex + 1) }};
124}

126// https://fetch.spec.whatwg.org/#concept-body-package-data
127static RefPtr<DOMFormData> packageFormData(ScriptExecutionContext* context, const String& contentType, const uint8_t* data, size_t length)
128{
129 auto parseMultipartPart = [context] (const uint8_t* part, size_t partLength, DOMFormData& form) -> bool {
130 const uint8_t* headerEnd = static_cast<const uint8_t*>(memmem(part, partLength, "\r\n\r\n", 4));
131 if (!headerEnd)
132 return false;
133 const uint8_t* headerBegin = part;
134 size_t headerLength = headerEnd - headerBegin;
135
136 const uint8_t* bodyBegin = headerEnd + strlen("\r\n\r\n");
137 size_t bodyLength = partLength - (bodyBegin - headerBegin);
138
139 String header = String::fromUTF8(headerBegin, headerLength);
140
141 const char* contentDispositionCharacters = "Content-Disposition:";
142 size_t contentDispositionBegin = header.find(contentDispositionCharacters);
143 if (contentDispositionBegin == notFound)
144 return false;
145 size_t contentDispositionEnd = header.find("\r\n", contentDispositionBegin);
146 size_t contentDispositionParametersBegin = header.find(';', contentDispositionBegin + strlen(contentDispositionCharacters));
147 if (contentDispositionParametersBegin != notFound)
148 contentDispositionParametersBegin++;
149
150 auto parameters = parseParameters(StringView(header).substring(contentDispositionParametersBegin, contentDispositionEnd - contentDispositionParametersBegin), 0);
151 String name = parameters.get("name");
152 if (!name)
153 return false;
154 String filename = parameters.get("filename");
155 if (!filename)
156 form.append(name, String::fromUTF8(bodyBegin, bodyLength));
157 else {
158 String contentType = "text/plain"_s;
159
160 const char* contentTypeCharacters = "Content-Type:";
161 size_t contentTypePrefixLength = strlen(contentTypeCharacters);
162 size_t contentTypeBegin = header.find(contentTypeCharacters);
163 if (contentTypeBegin != notFound) {
164 size_t contentTypeEnd = header.find("\r\n", contentTypeBegin);
165 contentType = stripLeadingAndTrailingHTTPSpaces(header.substring(contentTypeBegin + contentTypePrefixLength, contentTypeEnd - contentTypeBegin - contentTypePrefixLength));
166 }
167
168 form.append(name, File::create(context, Blob::create(context, Vector { bodyBegin, bodyLength }, Blob::normalizedContentType(contentType)).get(), filename).get(), filename);
169 }
170 return true;
171 };
172
173 auto parseMultipartBoundary = [] (const std::optional<MimeType>& mimeType) -> std::optional<String> {
174 if (!mimeType)
175 return std::nullopt;
176 if (equalIgnoringASCIICase(mimeType->type, "multipart") && equalIgnoringASCIICase(mimeType->subtype, "form-data")) {
177 auto iterator = mimeType->parameters.find("boundary"_s);
178 if (iterator != mimeType->parameters.end())
179 return iterator->value;
180 }
181 return std::nullopt;
182 };
183
184 auto form = DOMFormData::create(UTF8Encoding());
185 auto mimeType = parseMIMEType(contentType);
186 if (auto multipartBoundary = parseMultipartBoundary(mimeType)) {
187 auto boundaryWithDashes = makeString("--", *multipartBoundary);
188 CString boundary = boundaryWithDashes.utf8();
189 size_t boundaryLength = boundary.length();
190
191 const uint8_t* currentBoundary = static_cast<const uint8_t*>(memmem(data, length, boundary.data(), boundaryLength));
192 if (!currentBoundary)
193 return nullptr;
194 const uint8_t* nextBoundary = static_cast<const uint8_t*>(memmem(currentBoundary + boundaryLength, length - (currentBoundary + boundaryLength - data), boundary.data(), boundaryLength));
195 if (!nextBoundary)
196 return nullptr;
197 while (nextBoundary) {
198 parseMultipartPart(currentBoundary + boundaryLength, nextBoundary - currentBoundary - boundaryLength - strlen("\r\n"), form.get());
199 currentBoundary = nextBoundary;
200 nextBoundary = static_cast<const uint8_t*>(memmem(nextBoundary + boundaryLength, length - (nextBoundary + boundaryLength - data), boundary.data(), boundaryLength));
201 }
202 } else if (mimeType && equalIgnoringASCIICase(mimeType->type, "application") && equalIgnoringASCIICase(mimeType->subtype, "x-www-form-urlencoded")) {
203 auto dataString = String::fromUTF8(data, length);
204 for (auto& pair : WTF::URLParser::parseURLEncodedForm(dataString))
205 form->append(pair.key, pair.value);
206 } else
207 return nullptr;
208
209 return form;
210}

212static void resolveWithTypeAndData(Ref<DeferredPromise>&& promise, FetchBodyConsumer::Type type, const String& contentType, const unsigned char* data, unsigned length)
213{
214 auto* context = promise->scriptExecutionContext();
215
216 switch (type) {
217 case FetchBodyConsumer::Type::ArrayBuffer:
218 fulfillPromiseWithArrayBuffer(WTFMove(promise), data, length);
219 return;
220 case FetchBodyConsumer::Type::Blob:
221 promise->resolveCallbackValueWithNewlyCreated<IDLInterface<Blob>>([&data, &length, &contentType, context](auto&) {
222 return blobFromData(context, { data, length }, contentType);
223 });
224 return;
225 case FetchBodyConsumer::Type::JSON:
226 fulfillPromiseWithJSON(WTFMove(promise), TextResourceDecoder::textFromUTF8(data, length));
227 return;
228 case FetchBodyConsumer::Type::Text:
229 promise->resolve<IDLDOMString>(TextResourceDecoder::textFromUTF8(data, length));
230 return;
231 case FetchBodyConsumer::Type::FormData:
232 if (auto formData = packageFormData(context, contentType, data, length))
233 promise->resolve<IDLInterface<DOMFormData>>(*formData);
234 else
235 promise->reject(TypeError);
236 return;
237 case FetchBodyConsumer::Type::None:
238 ASSERT_NOT_REACHED();
239 return;
240 }
241}

258void FetchBodyConsumer::extract(ReadableStream& stream, ReadableStreamToSharedBufferSink::Callback&& callback)
259{
260 ASSERT(!m_sink);

262 m_sink->pipeFrom(stream);
263}

265void FetchBodyConsumer::resolve(Ref<DeferredPromise>&& promise, const String& contentType, ReadableStream* stream)
266{
267 if (stream) {
268 ASSERT(!m_sink);
269 m_sink = ReadableStreamToSharedBufferSink::create([promise = WTFMove(promise), data = SharedBuffer::create(), type = m_type, contentType](auto&& result) mutable {
270 if (result.hasException()) {
271 promise->reject(result.releaseException());
272 return;
273 }
274
275 if (auto* chunk = result.returnValue())
276 data->append(chunk->data(), chunk->size());
277 else
278 resolveWithTypeAndData(WTFMove(promise), type, contentType, data->data(), data->size());
279 });
280 m_sink->pipeFrom(*stream);
281 return;
282 }
283
284 if (m_isLoading) {
285 setConsumePromise(WTFMove(promise));
286 return;
287 }
288
289 auto* context = promise->scriptExecutionContext();
290
291 ASSERT(m_type != Type::None);
292 switch (m_type) {
293 case Type::ArrayBuffer:
294 fulfillPromiseWithArrayBuffer(WTFMove(promise), takeAsArrayBuffer().get());
295 return;
296 case Type::Blob:
297 promise->resolveCallbackValueWithNewlyCreated<IDLInterface<Blob>>([this, context](auto&) {
298 return takeAsBlob(context);
299 });
300 return;
301 case Type::JSON:
302 fulfillPromiseWithJSON(WTFMove(promise), takeAsText());
303 return;
304 case Type::Text:
305 promise->resolve<IDLDOMString>(takeAsText());
306 return;
307 case FetchBodyConsumer::Type::FormData: {
308 auto buffer = takeData();
309 if (auto formData = packageFormData(context, contentType, buffer ? buffer->data() : nullptr, buffer ? buffer->size() : 0))
310 promise->resolve<IDLInterface<DOMFormData>>(*formData);
311 else
312 promise->reject(TypeError);
313 return;
314 }
315 case Type::None:
316 ASSERT_NOT_REACHED();
317 return;
318 }
319}

321void FetchBodyConsumer::append(const uint8_t* data, unsigned size)
322{
323 if (m_source) {
324 m_source->enqueue(ArrayBuffer::tryCreate(data, size));
325 return;
326 }
327 if (!m_buffer) {
328 m_buffer = SharedBuffer::create(data, size);
329 return;
330 }
331 m_buffer->append(data, size);
332}

334RefPtr<SharedBuffer> FetchBodyConsumer::takeData()
335{
336 return WTFMove(m_buffer);

339RefPtr<JSC::ArrayBuffer> FetchBodyConsumer::takeAsArrayBuffer()
340{
341 if (!m_buffer)
342 return ArrayBuffer::tryCreate(nullptr, 0);
343
344 auto arrayBuffer = m_buffer->tryCreateArrayBuffer();
345 m_buffer = nullptr;
346 return arrayBuffer;
347}

349Ref<Blob> FetchBodyConsumer::takeAsBlob(ScriptExecutionContext* context)
350{
351 if (!m_buffer)
352 return Blob::create(context, Vector<uint8_t>(), Blob::normalizedContentType(m_contentType));
353
354 auto data = std::exchange(m_buffer, nullptr)->takeData();
355 return blobFromData(context, WTFMove(data), m_contentType);
356}

358String FetchBodyConsumer::takeAsText()
359{
360 // FIXME: We could probably text decode on the fly as soon as m_type is set to JSON or Text.
361 if (!m_buffer)
362 return String();
363
364 auto text = TextResourceDecoder::textFromUTF8(m_buffer->data(), m_buffer->size());
365 m_buffer = nullptr;
366 return text;
367}

370{

391void FetchBodyConsumer::loadingFailed(const Exception& exception)
392{
393 m_isLoading = false;
394 if (m_consumePromise) {
395 m_consumePromise->reject(exception);
396 resetConsumePromise();
397 }
398 if (m_source) {
399 m_source->error(exception);
400 m_source = nullptr;
401 }
402}
403
404void FetchBodyConsumer::loadingSucceeded(const String& contentType)
405{
406 m_isLoading = false;
407
408 if (m_consumePromise) {
409 if (!m_userGestureToken || m_userGestureToken->hasExpired(UserGestureToken::maximumIntervalForUserGestureForwardingForFetch()) || !m_userGestureToken->processingUserGesture())
410 resolve(m_consumePromise.releaseNonNull(), contentType, nullptr);
411 else {
412 UserGestureIndicator gestureIndicator(m_userGestureToken, UserGestureToken::GestureScope::MediaOnly, UserGestureToken::IsPropagatedFromFetch::Yes);
413 resolve(m_consumePromise.releaseNonNull(), contentType, nullptr);
414 }
415 }
416 if (m_source) {
417 m_source->close();
418 m_source = nullptr;
419 }
420}
