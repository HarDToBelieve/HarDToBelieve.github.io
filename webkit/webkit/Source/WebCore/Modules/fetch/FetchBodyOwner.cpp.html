57void FetchBodyOwner::stop()
58{
59 m_readableStreamSource = nullptr;
60 if (m_body)
61 m_body->cleanConsumer();
62
63 if (m_blobLoader) {
64 bool isUniqueReference = hasOneRef();
65 if (m_blobLoader->loader)
66 m_blobLoader->loader->stop();
67 // After that point, 'this' may be destroyed, since unsetPendingActivity should have been called.
68 ASSERT_UNUSED(isUniqueReference, isUniqueReference || !m_blobLoader);
69 }
70}

72bool FetchBodyOwner::isDisturbed() const
73{
74 if (isBodyNull())
75 return false;
76
77 if (m_isDisturbed)
78 return true;
79
80 if (body().readableStream())
81 return body().readableStream()->isDisturbed();
82
83 return false;
84}

86bool FetchBodyOwner::isDisturbedOrLocked() const
87{
88 if (isBodyNull())
89 return false;
90
91 if (m_isDisturbed)
92 return true;
93
94 if (body().readableStream())
95 return body().readableStream()->isDisturbed() || body().readableStream()->isLocked();
96
97 return false;
98}

100void FetchBodyOwner::arrayBuffer(Ref<DeferredPromise>&& promise)
101{
102 if (auto exception = loadingException()) {
103 promise->reject(*exception);
104 return;
105 }
106
107 if (isBodyNullOrOpaque()) {
108 fulfillPromiseWithArrayBuffer(WTFMove(promise), nullptr, 0);
109 return;
110 }
111 if (isDisturbedOrLocked()) {
112 promise->reject(Exception { TypeError, "Body is disturbed or locked"_s });
113 return;
114 }
115 m_isDisturbed = true;
116 m_body->arrayBuffer(*this, WTFMove(promise));
117}

119void FetchBodyOwner::blob(Ref<DeferredPromise>&& promise)
120{
121 if (auto exception = loadingException()) {
122 promise->reject(*exception);
123 return;
124 }
125
126 if (isBodyNullOrOpaque()) {
127 auto* context = promise->scriptExecutionContext();
128 promise->resolveCallbackValueWithNewlyCreated<IDLInterface<Blob>>([this, context](auto&) {
129 return Blob::create(context, Vector<uint8_t> { }, Blob::normalizedContentType(extractMIMETypeFromMediaType(m_contentType)));
130 });
131 return;
132 }
133 if (isDisturbedOrLocked()) {
134 promise->reject(Exception { TypeError, "Body is disturbed or locked"_s });
135 return;
136 }
137 m_isDisturbed = true;
138 m_body->blob(*this, WTFMove(promise), m_contentType);
139}

238void FetchBodyOwner::loadBlob(const Blob& blob, FetchBodyConsumer* consumer)
239{
240 // Can only be called once for a body instance.
241 ASSERT(!m_blobLoader);
242 ASSERT(!isBodyNull());
243
244 if (!scriptExecutionContext()) {
245 m_body->loadingFailed(Exception { TypeError, "Blob loading failed"_s});
246 return;
247 }
248
249 m_blobLoader.emplace(*this);
250 m_blobLoader->loader = makeUnique<FetchLoader>(*m_blobLoader, consumer);
251
252 m_blobLoader->loader->start(*scriptExecutionContext(), blob);
253 if (!m_blobLoader->loader->isStarted()) {
254 m_body->loadingFailed(Exception { TypeError, "Blob loading failed"_s});
255 m_blobLoader = std::nullopt;
256 return;
257 }
258}

260void FetchBodyOwner::finishBlobLoading()
261{
262 ASSERT(m_blobLoader);
263
264 m_blobLoader = std::nullopt;
265}

267void FetchBodyOwner::blobLoadingSucceeded()
268{
269 ASSERT(!isBodyNull());
270 if (m_readableStreamSource) {
271 m_readableStreamSource->close();
272 m_readableStreamSource = nullptr;
273 }
274
275 m_body->loadingSucceeded(contentType());
276 finishBlobLoading();
277}
278
279void FetchBodyOwner::blobLoadingFailed()
280{
281 ASSERT(!isBodyNull());
282 if (m_readableStreamSource) {
283 if (!m_readableStreamSource->isCancelling())
284 m_readableStreamSource->error(Exception { TypeError, "Blob loading failed"_s});
285 m_readableStreamSource = nullptr;
286 } else
287 m_body->loadingFailed(Exception { TypeError, "Blob loading failed"_s});
288 finishBlobLoading();
289}

317ExceptionOr<RefPtr<ReadableStream>> FetchBodyOwner::readableStream(JSC::JSGlobalObject& state)
318{
319 if (isBodyNullOrOpaque())
320 return nullptr;
321
322 if (!m_body->hasReadableStream()) {
323 auto voidOrException = createReadableStream(state);
324 if (UNLIKELY(voidOrException.hasException()))
325 return voidOrException.releaseException();
326 }
327
328 return m_body->readableStream();
329}
330
331ExceptionOr<void> FetchBodyOwner::createReadableStream(JSC::JSGlobalObject& state)
332{
333 ASSERT(!m_readableStreamSource);
334 if (isDisturbed()) {
335 auto streamOrException = ReadableStream::create(state, nullptr);
336 if (UNLIKELY(streamOrException.hasException()))
337 return streamOrException.releaseException();
338 m_body->setReadableStream(streamOrException.releaseReturnValue());
339 m_body->readableStream()->lock();
340 return { };
341 }
342
343 m_readableStreamSource = adoptRef(*new FetchBodySource(*this));
344 auto streamOrException = ReadableStream::create(state, m_readableStreamSource);
345 if (UNLIKELY(streamOrException.hasException())) {
346 m_readableStreamSource = nullptr;
347 return streamOrException.releaseException();
348 }
349 m_body->setReadableStream(streamOrException.releaseReturnValue());
350 return { };
351}

353void FetchBodyOwner::consumeBodyAsStream()
354{
355 ASSERT(m_readableStreamSource);
356
357 if (auto exception = loadingException()) {
358 m_readableStreamSource->error(*exception);
359 return;
360 }
361
362 body().consumeAsStream(*this, *m_readableStreamSource);
363 if (!m_readableStreamSource->isPulling())
364 m_readableStreamSource = nullptr;
365}

367ResourceError FetchBodyOwner::loadingError() const
368{
369 return WTF::switchOn(m_loadingError, [](const ResourceError& error) {
370 return ResourceError { error };
371 }, [](const Exception& exception) {
372 return ResourceError { errorDomainWebKitInternal, 0, { }, exception.message() };
373 }, [](auto&&) {
374 return ResourceError { };
375 });
376}
377
378std::optional<Exception> FetchBodyOwner::loadingException() const
379{
380 return WTF::switchOn(m_loadingError, [](const ResourceError& error) {
381 return Exception { TypeError, error.localizedDescription().isEmpty() ? "Loading failed"_s : error.localizedDescription() };
382 }, [](const Exception& exception) {
383 return Exception { exception };
384 }, [](auto&&) -> std::optional<Exception> {
385 return std::nullopt;
386 });
387}

394bool FetchBodyOwner::hasLoadingError() const
395{

399 return true;
400 }, [](auto&&) {
401 return false;
402 });
403}
