36// https://fetch.spec.whatwg.org/#concept-headers-remove-privileged-no-cors-request-headers
37static void removePrivilegedNoCORSRequestHeaders(HTTPHeaderMap& headers)
38{
39 headers.remove(HTTPHeaderName::Range);
40}

42static ExceptionOr<bool> canWriteHeader(const String& name, const String& value, const String& combinedValue, FetchHeaders::Guard guard)
43{
44 if (!isValidHTTPToken(name))
45 return Exception { TypeError, makeString("Invalid header name: '", name, "'") };
46 ASSERT(value.isEmpty() || (!isHTTPSpace(value[0]) && !isHTTPSpace(value[value.length() - 1])));
47 if (!isValidHTTPHeaderValue((value)))
48 return Exception { TypeError, makeString("Header '", name, "' has invalid value: '", value, "'") };
49 if (guard == FetchHeaders::Guard::Immutable)
50 return Exception { TypeError, "Headers object's guard is 'immutable'"_s };
51 if (guard == FetchHeaders::Guard::Request && isForbiddenHeaderName(name))
52 return false;
53 if (guard == FetchHeaders::Guard::RequestNoCors && !combinedValue.isEmpty() && !isSimpleHeader(name, combinedValue))
54 return false;
55 if (guard == FetchHeaders::Guard::Response && isForbiddenResponseHeaderName(name))
56 return false;
57 return true;
58}

60static ExceptionOr<void> appendToHeaderMap(const String& name, const String& value, HTTPHeaderMap& headers, FetchHeaders::Guard guard)
61{
62 String normalizedValue = stripLeadingAndTrailingHTTPSpaces(value);
63 String combinedValue = normalizedValue;
64 if (headers.contains(name))
65 combinedValue = makeString(headers.get(name), ", ", normalizedValue);
66 auto canWriteResult = canWriteHeader(name, normalizedValue, combinedValue, guard);
67 if (canWriteResult.hasException())
68 return canWriteResult.releaseException();
69 if (!canWriteResult.releaseReturnValue())
70 return { };
71 headers.set(name, combinedValue);
72
73 if (guard == FetchHeaders::Guard::RequestNoCors)
74 removePrivilegedNoCORSRequestHeaders(headers);
75
76 return { };
77}

79static ExceptionOr<void> appendToHeaderMap(const HTTPHeaderMap::HTTPHeaderMapConstIterator::KeyValue& header, HTTPHeaderMap& headers, FetchHeaders::Guard guard)
80{
81 String normalizedValue = stripLeadingAndTrailingHTTPSpaces(header.value);
82 auto canWriteResult = canWriteHeader(header.key, normalizedValue, header.value, guard);
83 if (canWriteResult.hasException())
84 return canWriteResult.releaseException();
85 if (!canWriteResult.releaseReturnValue())
86 return { };
87 if (header.keyAsHTTPHeaderName)
88 headers.add(header.keyAsHTTPHeaderName.value(), header.value);
89 else
90 headers.add(header.key, header.value);
91
92 if (guard == FetchHeaders::Guard::RequestNoCors)
93 removePrivilegedNoCORSRequestHeaders(headers);
94
95 return { };
96}

98// https://fetch.spec.whatwg.org/#concept-headers-fill
99static ExceptionOr<void> fillHeaderMap(HTTPHeaderMap& headers, const FetchHeaders::Init& headersInit, FetchHeaders::Guard guard)
100{
101 if (WTF::holds_alternative<Vector<Vector<String>>>(headersInit)) {
102 auto& sequence = WTF::get<Vector<Vector<String>>>(headersInit);
103 for (auto& header : sequence) {
104 if (header.size() != 2)
105 return Exception { TypeError, "Header sub-sequence must contain exactly two items" };
106 auto result = appendToHeaderMap(header[0], header[1], headers, guard);
107 if (result.hasException())
108 return result.releaseException();
109 }
110 } else {
111 auto& record = WTF::get<Vector<WTF::KeyValuePair<String, String>>>(headersInit);
112 for (auto& header : record) {
113 auto result = appendToHeaderMap(header.key, header.value, headers, guard);
114 if (result.hasException())
115 return result.releaseException();
116 }
117 }
118
119 return { };
120}

122ExceptionOr<Ref<FetchHeaders>> FetchHeaders::create(std::optional<Init>&& headersInit)
123{
124 HTTPHeaderMap headers;
125
126 if (headersInit) {
127 auto result = fillHeaderMap(headers, *headersInit, Guard::None);
128 if (result.hasException())
129 return result.releaseException();
130 }
131
132 return adoptRef(*new FetchHeaders { Guard::None, WTFMove(headers) });
133}

135ExceptionOr<void> FetchHeaders::fill(const Init& headerInit)
136{
137 return fillHeaderMap(m_headers, headerInit, m_guard);
138}

140ExceptionOr<void> FetchHeaders::fill(const FetchHeaders& otherHeaders)
141{
142 for (auto& header : otherHeaders.m_headers) {
143 auto result = appendToHeaderMap(header, m_headers, m_guard);
144 if (result.hasException())
145 return result.releaseException();
146 }
147
148 return { };
149}

156// https://fetch.spec.whatwg.org/#dom-headers-delete
157ExceptionOr<void> FetchHeaders::remove(const String& name)
158{
159 if (!isValidHTTPToken(name))
160 return Exception { TypeError, makeString("Invalid header name: '", name, "'") };
161 if (m_guard == FetchHeaders::Guard::Immutable)
162 return Exception { TypeError, "Headers object's guard is 'immutable'"_s };
163 if (m_guard == FetchHeaders::Guard::Request && isForbiddenHeaderName(name))
164 return { };
165 if (m_guard == FetchHeaders::Guard::RequestNoCors && !isNoCORSSafelistedRequestHeaderName(name) && !isPriviledgedNoCORSRequestHeaderName(name))
166 return { };
167 if (m_guard == FetchHeaders::Guard::Response && isForbiddenResponseHeaderName(name))
168 return { };
169
170 m_headers.remove(name);
171
172 if (m_guard == FetchHeaders::Guard::RequestNoCors)
173 removePrivilegedNoCORSRequestHeaders(m_headers);
174
175 return { };
176}

192ExceptionOr<void> FetchHeaders::set(const String& name, const String& value)
193{
194 String normalizedValue = stripLeadingAndTrailingHTTPSpaces(value);
195 auto canWriteResult = canWriteHeader(name, normalizedValue, normalizedValue, m_guard);
196 if (canWriteResult.hasException())
197 return canWriteResult.releaseException();
198 if (!canWriteResult.releaseReturnValue())
199 return { };
200
201 m_headers.set(name, normalizedValue);
202
203 if (m_guard == FetchHeaders::Guard::RequestNoCors)
204 removePrivilegedNoCORSRequestHeaders(m_headers);
205
206 return { };
207}

209void FetchHeaders::filterAndFill(const HTTPHeaderMap& headers, Guard guard)
210{
211 for (auto& header : headers) {
212 String normalizedValue = stripLeadingAndTrailingHTTPSpaces(header.value);
213 auto canWriteResult = canWriteHeader(header.key, normalizedValue, header.value, guard);
214 if (canWriteResult.hasException())
215 continue;
216 if (!canWriteResult.releaseReturnValue())
217 continue;
218 if (header.keyAsHTTPHeaderName)
219 m_headers.add(header.keyAsHTTPHeaderName.value(), header.value);
220 else
221 m_headers.add(header.key, header.value);
222 }
223}

225std::optional<WTF::KeyValuePair<String, String>> FetchHeaders::Iterator::next()
226{
227 while (m_currentIndex < m_keys.size()) {
228 auto key = m_keys[m_currentIndex++];
229 auto value = m_headers->m_headers.get(key);
230 if (!value.isNull())
231 return WTF::KeyValuePair<String, String> { WTFMove(key), WTFMove(value) };
232 }
233 return std::nullopt;
234}
235
236FetchHeaders::Iterator::Iterator(FetchHeaders& headers)
237 : m_headers(headers)
238{
239 m_keys.reserveInitialCapacity(headers.m_headers.size());
240 for (auto& header : headers.m_headers)
241 m_keys.uncheckedAppend(header.key.convertToASCIILowercase());
242 std::sort(m_keys.begin(), m_keys.end(), WTF::codePointCompareLessThan);
243}
