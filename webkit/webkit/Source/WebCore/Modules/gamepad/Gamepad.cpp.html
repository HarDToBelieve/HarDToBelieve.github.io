35namespace WebCore {
36
37Gamepad::Gamepad(const PlatformGamepad& platformGamepad)
38 : m_id(platformGamepad.id())
39 , m_index(platformGamepad.index())
40 , m_connected(true)
41 , m_timestamp(platformGamepad.lastUpdateTime())
42 , m_mapping(platformGamepad.mapping())
43 , m_axes(platformGamepad.axisValues().size(), 0.0)
44{
45 unsigned buttonCount = platformGamepad.buttonValues().size();
46 for (unsigned i = 0; i < buttonCount; ++i)
47 m_buttons.append(GamepadButton::create());
48}
49
50Gamepad::~Gamepad() = default;
51
52const Vector<double>& Gamepad::axes() const
53{
54 return m_axes;
55}
56
57const Vector<Ref<GamepadButton>>& Gamepad::buttons() const
58{
59 return m_buttons;
60}
61
62void Gamepad::updateFromPlatformGamepad(const PlatformGamepad& platformGamepad)
63{
64 for (unsigned i = 0; i < m_axes.size(); ++i)
65 m_axes[i] = platformGamepad.axisValues()[i].value();
66 for (unsigned i = 0; i < m_buttons.size(); ++i)
67 m_buttons[i]->setValue(platformGamepad.buttonValues()[i].value());
68
69 m_timestamp = platformGamepad.lastUpdateTime();
70}
