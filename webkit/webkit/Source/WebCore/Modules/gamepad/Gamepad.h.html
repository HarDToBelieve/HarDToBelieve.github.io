26#pragma once
27
28#if ENABLE(GAMEPAD)
29
30#include <wtf/MonotonicTime.h>
31#include <wtf/RefCounted.h>
32#include <wtf/Vector.h>
33#include <wtf/text/WTFString.h>
34
35namespace WebCore {
36
37class GamepadButton;
38class PlatformGamepad;
39
40class Gamepad: public RefCounted<Gamepad> {
41public:
42 static Ref<Gamepad> create(const PlatformGamepad& platformGamepad)
43 {
44 return adoptRef(*new Gamepad(platformGamepad));
45 }
46 ~Gamepad();
47
48 const String& id() const { return m_id; }
49 unsigned index() const { return m_index; }
50 const String& mapping() const { return m_mapping; }
51
52 bool connected() const { return m_connected; }
53 double timestamp() const { return m_timestamp.secondsSinceEpoch().seconds(); }
54 const Vector<double>& axes() const;
55 const Vector<Ref<GamepadButton>>& buttons() const;
56
57 void updateFromPlatformGamepad(const PlatformGamepad&);
58 void setConnected(bool connected) { m_connected = connected; }
59
60private:
61 explicit Gamepad(const PlatformGamepad&);
62 String m_id;
63 unsigned m_index;
64 bool m_connected;
65 MonotonicTime m_timestamp;
66 String m_mapping;
67
68 Vector<double> m_axes;
69 Vector<Ref<GamepadButton>> m_buttons;
70};
71
72} // namespace WebCore
73
74#endif // ENABLE(GAMEPAD)
