50const char* NavigatorGamepad::supplementName()
51{
52 return "NavigatorGamepad";
53}

55NavigatorGamepad* NavigatorGamepad::from(Navigator* navigator)
56{
57 NavigatorGamepad* supplement = static_cast<NavigatorGamepad*>(Supplement<Navigator>::from(navigator, supplementName()));
58 if (!supplement) {
59 auto newSupplement = makeUnique<NavigatorGamepad>();
60 supplement = newSupplement.get();
61 provideTo(navigator, supplementName(), WTFMove(newSupplement));
62 }
63 return supplement;
64}

66Ref<Gamepad> NavigatorGamepad::gamepadFromPlatformGamepad(PlatformGamepad& platformGamepad)
67{
68 unsigned index = platformGamepad.index();
69 if (index >= m_gamepads.size() || !m_gamepads[index])
70 return Gamepad::create(platformGamepad);
71
72 return *m_gamepads[index];
73}

75const Vector<RefPtr<Gamepad>>& NavigatorGamepad::getGamepads(Navigator& navigator)
76{
77 auto* domWindow = navigator.window();
78 Document* document = domWindow ? domWindow->document() : nullptr;
79 if (!document) {
80 static NeverDestroyed<Vector<RefPtr<Gamepad>>> emptyGamepads;
81 return emptyGamepads;
82 }
83
84 if (!document->isSecureContext()) {
85 static std::once_flag onceFlag;
86 std::call_once(onceFlag, [document] {
87 document->addConsoleMessage(MessageSource::Security, MessageLevel::Warning, "Navigator.getGamepads() will be removed from insecure contexts in a future release"_s);
88 });
89
90 }
91
92 return NavigatorGamepad::from(&navigator)->gamepads();
93}
94
95const Vector<RefPtr<Gamepad>>& NavigatorGamepad::gamepads()
96{
97 if (m_gamepads.isEmpty())
98 return m_gamepads;
99
100 const Vector<PlatformGamepad*>& platformGamepads = GamepadProvider::singleton().platformGamepads();
101
102 for (unsigned i = 0; i < platformGamepads.size(); ++i) {
103 if (!platformGamepads[i]) {
104 ASSERT(!m_gamepads[i]);
105 continue;
106 }
107
108 ASSERT(m_gamepads[i]);
109 m_gamepads[i]->updateFromPlatformGamepad(*platformGamepads[i]);
110 }
111
112 return m_gamepads;
113}

115void NavigatorGamepad::gamepadsBecameVisible()
116{
117 const Vector<PlatformGamepad*>& platformGamepads = GamepadProvider::singleton().platformGamepads();
118 m_gamepads.resize(platformGamepads.size());
119
120 for (unsigned i = 0; i < platformGamepads.size(); ++i) {
121 if (!platformGamepads[i])
122 continue;
123
124 m_gamepads[i] = Gamepad::create(*platformGamepads[i]);
125 }
126}

128void NavigatorGamepad::gamepadConnected(PlatformGamepad& platformGamepad)
129{
130 // If this is the first gamepad this Navigator object has seen, then all gamepads just became visible.
131 if (m_gamepads.isEmpty()) {
132 gamepadsBecameVisible();
133 return;
134 }
135
136 unsigned index = platformGamepad.index();
137 ASSERT(GamepadProvider::singleton().platformGamepads()[index] == &platformGamepad);
138
139 // The new index should already fit in the existing array, or should be exactly one past-the-end of the existing array.
140 ASSERT(index <= m_gamepads.size());
141
142 if (index < m_gamepads.size())
143 m_gamepads[index] = Gamepad::create(platformGamepad);
144 else if (index == m_gamepads.size())
145 m_gamepads.append(Gamepad::create(platformGamepad));
146}
147
148void NavigatorGamepad::gamepadDisconnected(PlatformGamepad& platformGamepad)
149{
150 // If this Navigator hasn't seen any gamepads yet its Vector will still be empty.
151 if (!m_gamepads.size())
152 return;
153
154 ASSERT(platformGamepad.index() < m_gamepads.size());
155 ASSERT(m_gamepads[platformGamepad.index()]);
156
157 m_gamepads[platformGamepad.index()] = nullptr;
158}
