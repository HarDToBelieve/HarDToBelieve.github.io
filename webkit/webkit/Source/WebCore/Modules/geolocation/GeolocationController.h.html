26#pragma once
27
28#if ENABLE(GEOLOCATION)
29
30#include "ActivityStateChangeObserver.h"
31#include "Geolocation.h"
32#include "Page.h"
33#include <wtf/HashSet.h>
34#include <wtf/Noncopyable.h>
35#include <wtf/RefPtr.h>
36
37namespace WebCore {
38
39class GeolocationClient;
40class GeolocationError;
41class GeolocationPositionData;
42
43class GeolocationController : public Supplement<Page>, private ActivityStateChangeObserver {
44 WTF_MAKE_FAST_ALLOCATED;
45 WTF_MAKE_NONCOPYABLE(GeolocationController);
46public:
47 GeolocationController(Page&, GeolocationClient&);
48 ~GeolocationController();
49
50 void addObserver(Geolocation&, bool enableHighAccuracy);
51 void removeObserver(Geolocation&);
52
53 void requestPermission(Geolocation&);
54 void cancelPermissionRequest(Geolocation&);
55
56 WEBCORE_EXPORT void positionChanged(const std::optional<GeolocationPositionData>&);
57 WEBCORE_EXPORT void errorOccurred(GeolocationError&);
58
59 std::optional<GeolocationPositionData> lastPosition();
60
61 GeolocationClient& client() { return m_client; }
62
63 WEBCORE_EXPORT static const char* supplementName();
64 static GeolocationController* from(Page* page) { return static_cast<GeolocationController*>(Supplement<Page>::from(page, supplementName())); }
65
66 void revokeAuthorizationToken(const String&);
67
68private:
69 Page& m_page;
70 GeolocationClient& m_client;
71
72 void activityStateDidChange(OptionSet<ActivityState::Flag> oldActivityState, OptionSet<ActivityState::Flag> newActivityState) override;
73
74 std::optional<GeolocationPositionData> m_lastPosition;
75
76 typedef HashSet<Ref<Geolocation>> ObserversSet;
77 // All observers; both those requesting high accuracy and those not.
78 ObserversSet m_observers;
79 ObserversSet m_highAccuracyObservers;
80
81 // While the page is not visible, we pend permission requests.
82 HashSet<Ref<Geolocation>> m_pendingPermissionRequest;
83};
84
85} // namespace WebCore
86
87#endif // ENABLE(GEOLOCATION)
