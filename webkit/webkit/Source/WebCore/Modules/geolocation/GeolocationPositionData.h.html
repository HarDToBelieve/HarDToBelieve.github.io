26#pragma once
27
28#include <cmath>
29#include <wtf/Ref.h>
30#include <wtf/RefCounted.h>
31
32#if PLATFORM(IOS_FAMILY)
33OBJC_CLASS CLLocation;
34#endif
35
36namespace WebCore {
37
38class GeolocationPositionData {
39public:
40 GeolocationPositionData() = default;
41
42 GeolocationPositionData(double timestamp, double latitude, double longitude, double accuracy)
43 : timestamp(timestamp)
44 , latitude(latitude)
45 , longitude(longitude)
46 , accuracy(accuracy)
47 {
48 }
49
50#if PLATFORM(IOS_FAMILY)
51 WEBCORE_EXPORT explicit GeolocationPositionData(CLLocation*);
52#endif
53
54 double timestamp { std::numeric_limits<double>::quiet_NaN() };
55
56 double latitude { std::numeric_limits<double>::quiet_NaN() };
57 double longitude { std::numeric_limits<double>::quiet_NaN() };
58 double accuracy { std::numeric_limits<double>::quiet_NaN() };
59
60 std::optional<double> altitude;
61 std::optional<double> altitudeAccuracy;
62 std::optional<double> heading;
63 std::optional<double> speed;
64 std::optional<double> floorLevel;
65
66 bool isValid() const;
67
68 template<class Encoder> void encode(Encoder&) const;
69 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, GeolocationPositionData&);
70};
71
72template<class Encoder>
73void GeolocationPositionData::encode(Encoder& encoder) const
74{
75 encoder << timestamp;
76 encoder << latitude;
77 encoder << longitude;
78 encoder << accuracy;
79 encoder << altitude;
80 encoder << altitudeAccuracy;
81 encoder << heading;
82 encoder << speed;
83 encoder << floorLevel;
84}
85
86template<class Decoder>
87bool GeolocationPositionData::decode(Decoder& decoder, GeolocationPositionData& position)
88{
89 if (!decoder.decode(position.timestamp))
90 return false;
91 if (!decoder.decode(position.latitude))
92 return false;
93 if (!decoder.decode(position.longitude))
94 return false;
95 if (!decoder.decode(position.accuracy))
96 return false;
97 if (!decoder.decode(position.altitude))
98 return false;
99 if (!decoder.decode(position.altitudeAccuracy))
100 return false;
101 if (!decoder.decode(position.heading))
102 return false;
103 if (!decoder.decode(position.speed))
104 return false;
105 if (!decoder.decode(position.floorLevel))
106 return false;
107
108 return true;
109}
110
111inline bool GeolocationPositionData::isValid() const
112{
113 return !std::isnan(timestamp) && !std::isnan(latitude) && !std::isnan(longitude) && !std::isnan(accuracy);
114}
