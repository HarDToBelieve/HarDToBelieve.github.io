78bool IDBCursor::sourcesDeleted() const
79{
80 ASSERT(canCurrentThreadAccessThreadLocalData(effectiveObjectStore().transaction().database().originThread()));
81
82 return WTF::switchOn(m_source,
83 [] (const RefPtr<IDBObjectStore>& objectStore) { return objectStore->isDeleted(); },

85 );

102ExceptionOr<Ref<IDBRequest>> IDBCursor::update(JSGlobalObject& state, JSValue value)
103{
104 LOG(IndexedDB, "IDBCursor::update");
105 ASSERT(canCurrentThreadAccessThreadLocalData(effectiveObjectStore().transaction().database().originThread()));
106
107 if (sourcesDeleted())
108 return Exception { InvalidStateError, "Failed to execute 'update' on 'IDBCursor': The cursor's source or effective object store has been deleted."_s };
109
110 if (!transaction().isActive())
111 return Exception { TransactionInactiveError, "Failed to execute 'update' on 'IDBCursor': The transaction is inactive or finished."_s };
112
113 if (transaction().isReadOnly())
114 return Exception { ReadonlyError, "Failed to execute 'update' on 'IDBCursor': The record may not be updated inside a read-only transaction."_s };
115
116 if (!m_gotValue)
117 return Exception { InvalidStateError, "Failed to execute 'update' on 'IDBCursor': The cursor is being iterated or has iterated past its end."_s };
118
119 if (!isKeyCursorWithValue())
120 return Exception { InvalidStateError, "Failed to execute 'update' on 'IDBCursor': The cursor is a key cursor."_s };
121
122 auto& objectStore = effectiveObjectStore();
123 auto& optionalKeyPath = objectStore.info().keyPath();
124 const bool usesInLineKeys = !!optionalKeyPath;
125 if (usesInLineKeys) {
126 RefPtr<IDBKey> keyPathKey = maybeCreateIDBKeyFromScriptValueAndKeyPath(state, value, optionalKeyPath.value());
127 IDBKeyData keyPathKeyData(keyPathKey.get());
128 if (!keyPathKey || keyPathKeyData != m_primaryKeyData)
129 return Exception { DataError, "Failed to execute 'update' on 'IDBCursor': The effective object store of this cursor uses in-line keys and evaluating the key path of the value parameter results in a different value than the cursor's effective key."_s };
130 }
131
132 auto putResult = effectiveObjectStore().putForCursorUpdate(state, value, m_primaryKey.copyRef());
133 if (putResult.hasException())
134 return putResult.releaseException();
135
136 auto request = putResult.releaseReturnValue();
137 request->setSource(*this);
138
139 return request;
140}

142ExceptionOr<void> IDBCursor::advance(unsigned count)
143{
144 LOG(IndexedDB, "IDBCursor::advance");
145 ASSERT(canCurrentThreadAccessThreadLocalData(effectiveObjectStore().transaction().database().originThread()));
146
147 if (!m_request)
148 return Exception { InvalidStateError };
149
150 if (!count)
151 return Exception { TypeError, "Failed to execute 'advance' on 'IDBCursor': A count argument with value 0 (zero) was supplied, must be greater than 0."_s };
152
153 if (!transaction().isActive())
154 return Exception { TransactionInactiveError, "Failed to execute 'advance' on 'IDBCursor': The transaction is inactive or finished."_s };
155
156 if (sourcesDeleted())
157 return Exception { InvalidStateError, "Failed to execute 'advance' on 'IDBCursor': The cursor's source or effective object store has been deleted."_s };
158
159 if (!m_gotValue)
160 return Exception { InvalidStateError, "Failed to execute 'advance' on 'IDBCursor': The cursor is being iterated or has iterated past its end."_s };
161
162 m_gotValue = false;
163
164 uncheckedIterateCursor(IDBKeyData(), count);
165
166 return { };
167}

169ExceptionOr<void> IDBCursor::continuePrimaryKey(JSGlobalObject& state, JSValue keyValue, JSValue primaryKeyValue)
170{
171 if (!m_request)
172 return Exception { InvalidStateError };
173
174 if (!transaction().isActive())
175 return Exception { TransactionInactiveError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The transaction is inactive or finished."_s };
176
177 if (sourcesDeleted())
178 return Exception { InvalidStateError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The cursor's source or effective object store has been deleted."_s };
179
180 if (!WTF::holds_alternative<RefPtr<IDBIndex>>(m_source))
181 return Exception { InvalidAccessError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The cursor's source is not an index."_s };
182
183 auto direction = m_info.cursorDirection();
184 if (direction != IndexedDB::CursorDirection::Next && direction != IndexedDB::CursorDirection::Prev)
185 return Exception { InvalidAccessError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The cursor's direction must be either \"next\" or \"prev\"."_s };
186
187 if (!m_gotValue)
188 return Exception { InvalidStateError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The cursor is being iterated or has iterated past its end."_s };
189
190 RefPtr<IDBKey> key = scriptValueToIDBKey(state, keyValue);
191 if (!key->isValid())
192 return Exception { DataError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The first parameter is not a valid key."_s };
193
194 RefPtr<IDBKey> primaryKey = scriptValueToIDBKey(state, primaryKeyValue);
195 if (!primaryKey->isValid())
196 return Exception { DataError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The second parameter is not a valid key."_s };
197
198 IDBKeyData keyData = { key.get() };
199 IDBKeyData primaryKeyData = { primaryKey.get() };
200
201 if (keyData < m_keyData && direction == IndexedDB::CursorDirection::Next)
202 return Exception { DataError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The first parameter is less than this cursor's position and this cursor's direction is \"next\"."_s };
203
204 if (keyData > m_keyData && direction == IndexedDB::CursorDirection::Prev)
205 return Exception { DataError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The first parameter is greater than this cursor's position and this cursor's direction is \"prev\"."_s };
206
207 if (keyData == m_keyData) {
208 if (primaryKeyData <= m_primaryKeyData && direction == IndexedDB::CursorDirection::Next)
209 return Exception { DataError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The key parameters represent a position less-than-or-equal-to this cursor's position and this cursor's direction is \"next\"."_s };
210 if (primaryKeyData >= m_primaryKeyData && direction == IndexedDB::CursorDirection::Prev)
211 return Exception { DataError, "Failed to execute 'continuePrimaryKey' on 'IDBCursor': The key parameters represent a position greater-than-or-equal-to this cursor's position and this cursor's direction is \"prev\"."_s };
212 }
213
214 m_gotValue = false;
215
216 uncheckedIterateCursor(keyData, primaryKeyData);
217
218 return { };
219}

230ExceptionOr<void> IDBCursor::continueFunction(const IDBKeyData& key)
231{
232 LOG(IndexedDB, "IDBCursor::continueFunction (to key %s)", key.loggingString().utf8().data());
233 ASSERT(canCurrentThreadAccessThreadLocalData(effectiveObjectStore().transaction().database().originThread()));
234
235 if (!m_request)
236 return Exception { InvalidStateError };
237
238 if (!transaction().isActive())
239 return Exception { TransactionInactiveError, "Failed to execute 'continue' on 'IDBCursor': The transaction is inactive or finished."_s };
240
241 if (sourcesDeleted())
242 return Exception { InvalidStateError, "Failed to execute 'continue' on 'IDBCursor': The cursor's source or effective object store has been deleted."_s };
243
244 if (!m_gotValue)
245 return Exception { InvalidStateError, "Failed to execute 'continue' on 'IDBCursor': The cursor is being iterated or has iterated past its end."_s };
246
247 if (!key.isNull() && !key.isValid())
248 return Exception { DataError, "Failed to execute 'continue' on 'IDBCursor': The parameter is not a valid key."_s };
249
250 if (m_info.isDirectionForward()) {
251 if (!key.isNull() && key.compare(m_keyData) <= 0)
252 return Exception { DataError, "Failed to execute 'continue' on 'IDBCursor': The parameter is less than or equal to this cursor's position."_s };
253 } else {
254 if (!key.isNull() && key.compare(m_keyData) >= 0)
255 return Exception { DataError, "Failed to execute 'continue' on 'IDBCursor': The parameter is greater than or equal to this cursor's position."_s };
256 }
257
258 m_gotValue = false;
259
260 uncheckedIterateCursor(key, 0);
261
262 return { };
263}

283ExceptionOr<Ref<WebCore::IDBRequest>> IDBCursor::deleteFunction(JSGlobalObject& state)
284{
285 LOG(IndexedDB, "IDBCursor::deleteFunction");
286 ASSERT(canCurrentThreadAccessThreadLocalData(effectiveObjectStore().transaction().database().originThread()));
287
288 if (sourcesDeleted())
289 return Exception { InvalidStateError, "Failed to execute 'delete' on 'IDBCursor': The cursor's source or effective object store has been deleted."_s };
290
291 if (!transaction().isActive())
292 return Exception { TransactionInactiveError, "Failed to execute 'delete' on 'IDBCursor': The transaction is inactive or finished."_s };
293
294 if (transaction().isReadOnly())
295 return Exception { ReadonlyError, "Failed to execute 'delete' on 'IDBCursor': The record may not be deleted inside a read-only transaction."_s };
296
297 if (!m_gotValue)
298 return Exception { InvalidStateError, "Failed to execute 'delete' on 'IDBCursor': The cursor is being iterated or has iterated past its end."_s };
299
300 if (!isKeyCursorWithValue())
301 return Exception { InvalidStateError, "Failed to execute 'delete' on 'IDBCursor': The cursor is a key cursor."_s };
302
303 auto result = effectiveObjectStore().deleteFunction(state, IDBKeyRange::create(m_primaryKey.copyRef()).ptr());
304 if (result.hasException())
305 return result.releaseException();
306
307 auto request = result.releaseReturnValue();
308 request->setSource(*this);
309
310 return request;
311}

313bool IDBCursor::setGetResult(IDBRequest& request, const IDBGetResult& getResult, uint64_t operationID)
314{
315 LOG(IndexedDB, "IDBCursor::setGetResult - current key %s", getResult.keyData().loggingString().substring(0, 100).utf8().data());
316 ASSERT(canCurrentThreadAccessThreadLocalData(effectiveObjectStore().transaction().database().originThread()));
317
318 auto* context = request.scriptExecutionContext();
319 if (!context)
320 return false;
321
322 VM& vm = context->vm();
323 JSLockHolder lock(vm);
324
325 m_keyWrapper = { };
326 m_primaryKeyWrapper = { };
327 m_valueWrapper = { };
328
329 if (!getResult.isDefined()) {
330 m_keyData = { };
331 m_key = nullptr;
332 m_primaryKeyData = { };
333 m_primaryKey = nullptr;
334 m_value = { };
335
336 m_gotValue = false;
337 return false;
338 }
339
340 m_keyData = getResult.keyData();
341 m_key = m_keyData.maybeCreateIDBKey();
342 m_primaryKeyData = getResult.primaryKeyData();
343 m_primaryKey = m_primaryKeyData.maybeCreateIDBKey();
344
345 if (isKeyCursorWithValue()) {
346 m_value = getResult.value();
347 m_keyPath = getResult.keyPath();
348 }
349
350 auto prefetchedRecords = getResult.prefetchedRecords();
351 if (!prefetchedRecords.isEmpty()) {
352 for (auto& record : prefetchedRecords)
353 m_prefetchedRecords.append(record);
354 m_prefetchOperationID = operationID;
355 }
356
357 m_gotValue = true;
358 return true;
359}

368std::optional<IDBGetResult> IDBCursor::iterateWithPrefetchedRecords(unsigned count, uint64_t lastWriteOperationID)
369{
370 unsigned step = count > 0 ? count : 1;
371 if (step > m_prefetchedRecords.size() || m_prefetchOperationID <= lastWriteOperationID)
372 return std::nullopt;
373
374 while (--step)
375 m_prefetchedRecords.removeFirst();
376
377 auto record = m_prefetchedRecords.takeFirst();
378
379 LOG(IndexedDB, "IDBTransaction::iterateWithPrefetchedRecords consumes %u records", count > 0 ? count : 1);
380 return IDBGetResult(record.key, record.primaryKey, IDBValue(record.value), effectiveObjectStore().keyPath());
381}
