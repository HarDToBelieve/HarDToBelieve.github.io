33namespace WebCore {
34
35WTF_MAKE_ISO_ALLOCATED_IMPL(IDBDatabaseNameAndVersionRequest);
36
37Ref<IDBDatabaseNameAndVersionRequest> IDBDatabaseNameAndVersionRequest::create(ScriptExecutionContext& context, IDBClient::IDBConnectionProxy& connectionProxy, InfoCallback&& callback)
38{
39 return adoptRef(*new IDBDatabaseNameAndVersionRequest(context, connectionProxy, WTFMove(callback)));
40}
41
42IDBDatabaseNameAndVersionRequest::IDBDatabaseNameAndVersionRequest(ScriptExecutionContext& context, IDBClient::IDBConnectionProxy& connectionProxy, InfoCallback&& callback)
43 : IDBActiveDOMObject(&context)
44 , m_connectionProxy(connectionProxy)
45 , m_resourceIdentifier(connectionProxy)
46 , m_callback(WTFMove(callback))
47{
48 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
49
50 suspendIfNeeded();
51}
52
53void IDBDatabaseNameAndVersionRequest::complete(std::optional<Vector<IDBDatabaseNameAndVersion>>&& databases)
54{
55 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
56
57 if (auto callback = WTFMove(m_callback))
58 callback(WTFMove(databases));
59}
60
61const char* IDBDatabaseNameAndVersionRequest::activeDOMObjectName() const
62{
63 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
64 return "IDBDatabaseNameAndVersionRequest";
65}
66
67bool IDBDatabaseNameAndVersionRequest::virtualHasPendingActivity() const
68{
69 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()) || Thread::mayBeGCThread());
70 return true;
71}
72
73void IDBDatabaseNameAndVersionRequest::stop()
74{
75 ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
76}
