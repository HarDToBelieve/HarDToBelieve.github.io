46static bool shouldThrowSecurityException(ScriptExecutionContext& context)
47{
48 ASSERT(is<Document>(context) || context.isWorkerGlobalScope());
49 if (is<Document>(context)) {
50 Document& document = downcast<Document>(context);
51 if (!document.frame())
52 return true;
53 if (!document.page())
54 return true;
55 }
56
57 if (!context.securityOrigin()->canAccessDatabase(nullptr))
58 return true;
59
60 return false;
61}

75ExceptionOr<Ref<IDBOpenDBRequest>> IDBFactory::open(ScriptExecutionContext& context, const String& name, std::optional<uint64_t> version)
76{
77 LOG(IndexedDB, "IDBFactory::open");
78
79 if (version && !version.value())
80 return Exception { TypeError, "IDBFactory.open() called with a version of 0"_s };
81
82 return openInternal(context, name, version.value_or(0));
83}
84
85ExceptionOr<Ref<IDBOpenDBRequest>> IDBFactory::openInternal(ScriptExecutionContext& context, const String& name, uint64_t version)
86{
87 if (name.isNull())
88 return Exception { TypeError, "IDBFactory.open() called without a database name"_s };
89
90 if (shouldThrowSecurityException(context))
91 return Exception { SecurityError, "IDBFactory.open() called in an invalid security context"_s };
92
93 ASSERT(context.securityOrigin());
94 bool isTransient = !context.securityOrigin()->canAccessDatabase(&context.topOrigin());
95 IDBDatabaseIdentifier databaseIdentifier(name, SecurityOriginData { context.securityOrigin()->data() }, SecurityOriginData { context.topOrigin().data() }, isTransient);
96 if (!databaseIdentifier.isValid())
97 return Exception { TypeError, "IDBFactory.open() called with an invalid security origin"_s };
98
99 LOG(IndexedDBOperations, "IDB opening database: %s %" PRIu64, name.utf8().data(), version);
100
101 return m_connectionProxy->openDatabase(context, databaseIdentifier, version);
102}

104ExceptionOr<Ref<IDBOpenDBRequest>> IDBFactory::deleteDatabase(ScriptExecutionContext& context, const String& name)
105{
106 LOG(IndexedDB, "IDBFactory::deleteDatabase - %s", name.utf8().data());
107
108 if (name.isNull())
109 return Exception { TypeError, "IDBFactory.deleteDatabase() called without a database name"_s };
110
111 if (shouldThrowSecurityException(context))
112 return Exception { SecurityError, "IDBFactory.deleteDatabase() called in an invalid security context"_s };
113
114 ASSERT(context.securityOrigin());
115 bool isTransient = !context.securityOrigin()->canAccessDatabase(&context.topOrigin());
116 IDBDatabaseIdentifier databaseIdentifier(name, SecurityOriginData { context.securityOrigin()->data() }, SecurityOriginData { context.topOrigin().data() }, isTransient);
117 if (!databaseIdentifier.isValid())
118 return Exception { TypeError, "IDBFactory.deleteDatabase() called with an invalid security origin"_s };
119
120 LOG(IndexedDBOperations, "IDB deleting database: %s", name.utf8().data());
121
122 return m_connectionProxy->deleteDatabase(context, databaseIdentifier);
123}

125ExceptionOr<short> IDBFactory::cmp(JSGlobalObject& execState, JSValue firstValue, JSValue secondValue)
126{
127 auto first = scriptValueToIDBKey(execState, firstValue);
128 if (!first->isValid())
129 return Exception { DataError, "Failed to execute 'cmp' on 'IDBFactory': The parameter is not a valid key."_s };
130
131 auto second = scriptValueToIDBKey(execState, secondValue);
132 if (!second->isValid())
133 return Exception { DataError, "Failed to execute 'cmp' on 'IDBFactory': The parameter is not a valid key."_s };
134
135 return first->compare(second.get());
136}

138void IDBFactory::databases(ScriptExecutionContext& context, IDBDatabasesResponsePromise&& promise)
139{
140 LOG(IndexedDB, "IDBFactory::databases");
141
142 if (shouldThrowSecurityException(context)) {
143 promise.reject(SecurityError);
144 return;
145 }
146
147 ASSERT(context.securityOrigin());
148
149 m_connectionProxy->getAllDatabaseNamesAndVersions(context, [promise = WTFMove(promise)](auto&& result) mutable {
150 if (!result) {
151 promise.reject(Exception { UnknownError });
152 return;
153 }
154
155 promise.resolve(WTF::map(*result, [](auto&& info) {
156 return IDBFactory::DatabaseInfo { WTFMove(info.name), info.version };
157 }));
158 });
159}
160
161void IDBFactory::getAllDatabaseNames(ScriptExecutionContext& context, Function<void(const Vector<String>&)>&& callback)
162{
163 m_connectionProxy->getAllDatabaseNamesAndVersions(context, [callback = WTFMove(callback)](auto&& result) mutable {
164 if (!result) {
165 callback({ });
166 return;
167 }
168
169 callback(WTF::map(*result, [](auto&& info) {
170 return WTFMove(info.name);
171 }));
172 });
173}
