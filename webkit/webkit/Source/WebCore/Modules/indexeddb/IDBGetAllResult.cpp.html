35WTF_MAKE_ISO_ALLOCATED_IMPL(IDBGetAllResult);

37IDBGetAllResult::IDBGetAllResult(const IDBGetAllResult& that, IsolatedCopyTag)
38{
39 isolatedCopy(that, *this);
40}
41
42IDBGetAllResult IDBGetAllResult::isolatedCopy() const
43{
44 return { *this, IsolatedCopy };
45}
46
47void IDBGetAllResult::isolatedCopy(const IDBGetAllResult& source, IDBGetAllResult& destination)
48{
49 destination.m_type = source.m_type;
50 destination.m_keys = crossThreadCopy(source.m_keys);
51 destination.m_values = crossThreadCopy(source.m_values);
52 destination.m_keyPath = WebCore::isolatedCopy(source.m_keyPath);
53}
54
55void IDBGetAllResult::addKey(IDBKeyData&& key)
56{
57 m_keys.append(WTFMove(key));
58}
59
60void IDBGetAllResult::addValue(IDBValue&& value)
61{
62 m_values.append(WTFMove(value));
63}
64
65const Vector<IDBKeyData>& IDBGetAllResult::keys() const
66{
67 return m_keys;
68}
69
70const Vector<IDBValue>& IDBGetAllResult::values() const
71{
72 return m_values;
73}
74
75Vector<String> IDBGetAllResult::allBlobFilePaths() const
76{
77 ASSERT(m_type == IndexedDB::GetAllType::Values);
78
79 HashSet<String> pathSet;
80 for (auto& value : m_values) {
81 for (auto& path : value.blobFilePaths())
82 pathSet.add(path);
83 }
84
85 return copyToVector(pathSet);
86}
