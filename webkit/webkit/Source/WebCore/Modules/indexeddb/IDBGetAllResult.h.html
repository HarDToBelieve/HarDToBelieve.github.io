27
28#include "IDBKeyData.h"
29#include "IDBKeyPath.h"

31#include "IndexedDB.h"
32
33#include <wtf/IsoMalloc.h>
34#include <wtf/Variant.h>

36namespace WebCore {
37
38class IDBGetAllResult {
39 WTF_MAKE_ISO_ALLOCATED_EXPORT(IDBGetAllResult, WEBCORE_EXPORT);
40public:
41 IDBGetAllResult()
42 {
43 }
44
45 IDBGetAllResult(IndexedDB::GetAllType type, const std::optional<IDBKeyPath>& keyPath)
46 : m_type(type)
47 , m_keyPath(keyPath)
48 {
49 }
50
51 enum IsolatedCopyTag { IsolatedCopy };
52 IDBGetAllResult(const IDBGetAllResult&, IsolatedCopyTag);
53 IDBGetAllResult isolatedCopy() const;
54
55 IndexedDB::GetAllType type() const { return m_type; }
56 const std::optional<IDBKeyPath>& keyPath() const { return m_keyPath; }
57 const Vector<IDBKeyData>& keys() const;
58 const Vector<IDBValue>& values() const;
59
60 void addKey(IDBKeyData&&);
61 void addValue(IDBValue&&);
62
63 template<class Encoder> void encode(Encoder&) const;
64 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBGetAllResult&);
65
66 WEBCORE_EXPORT Vector<String> allBlobFilePaths() const;
67
68private:
69 static void isolatedCopy(const IDBGetAllResult& source, IDBGetAllResult& destination);
70
71 IndexedDB::GetAllType m_type { IndexedDB::GetAllType::Keys };
72 Vector<IDBKeyData> m_keys;
73 Vector<IDBValue> m_values;
74 std::optional<IDBKeyPath> m_keyPath;
75};
76
77template<class Encoder>
78void IDBGetAllResult::encode(Encoder& encoder) const
79{
80 encoder << m_type << m_keys << m_values << m_keyPath;
81}
82
83template<class Decoder>
84bool IDBGetAllResult::decode(Decoder& decoder, IDBGetAllResult& result)
85{
86 if (!decoder.decode(result.m_type))
87 return false;
88
89 if (!decoder.decode(result.m_keys))
90 return false;
91
92 if (!decoder.decode(result.m_values))
93 return false;
94
95 if (!decoder.decode(result.m_keyPath))
96 return false;
97
98 return true;
99}
