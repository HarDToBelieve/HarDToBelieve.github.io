33WTF_MAKE_ISO_ALLOCATED_IMPL(IDBGetResult);

35IDBGetResult::IDBGetResult(const IDBGetResult& that, IsolatedCopyTag)
36{
37 isolatedCopy(that, *this);
38}
39
40IDBGetResult IDBGetResult::isolatedCopy() const
41{
42 return { *this, IsolatedCopy };
43}
44
45void IDBGetResult::isolatedCopy(const IDBGetResult& source, IDBGetResult& destination)
46{
47 destination.m_value = source.m_value.isolatedCopy();
48 destination.m_keyData = source.m_keyData.isolatedCopy();
49 destination.m_primaryKeyData = source.m_primaryKeyData.isolatedCopy();
50 destination.m_keyPath = WebCore::isolatedCopy(source.m_keyPath);
51 destination.m_isDefined = source.m_isDefined;
52 destination.m_prefetchedRecords = source.m_prefetchedRecords.isolatedCopy();
53}
54
55void IDBGetResult::setValue(IDBValue&& value)
56{
57 m_value = WTFMove(value);
58}
