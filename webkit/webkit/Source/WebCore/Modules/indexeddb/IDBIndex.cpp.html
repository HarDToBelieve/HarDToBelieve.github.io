77ExceptionOr<void> IDBIndex::setName(const String& name)
78{
79 ASSERT(canCurrentThreadAccessThreadLocalData(m_objectStore.transaction().database().originThread()));
80
81 if (m_deleted)
82 return Exception { InvalidStateError, "Failed set property 'name' on 'IDBIndex': The index has been deleted."_s };
83
84 if (m_objectStore.isDeleted())
85 return Exception { InvalidStateError, "Failed set property 'name' on 'IDBIndex': The index's object store has been deleted."_s };
86
87 if (!m_objectStore.transaction().isVersionChange())
88 return Exception { InvalidStateError, "Failed set property 'name' on 'IDBIndex': The index's transaction is not a version change transaction."_s };
89
90 if (!m_objectStore.transaction().isActive())
91 return Exception { TransactionInactiveError, "Failed set property 'name' on 'IDBIndex': The index's transaction is not active."_s };
92
93 if (m_info.name() == name)
94 return { };
95
96 if (m_objectStore.info().hasIndex(name))
97 return Exception { ConstraintError, makeString("Failed set property 'name' on 'IDBIndex': The owning object store already has an index named '", name, "'.") };
98
99 m_objectStore.transaction().database().renameIndex(*this, name);
100 m_info.rename(name);
101
102 return { };
103}

129void IDBIndex::rollbackInfoForVersionChangeAbort()
130{
131 ASSERT(canCurrentThreadAccessThreadLocalData(m_objectStore.transaction().database().originThread()));
132
133 // Only rollback to the original info if this index still exists in the rolled-back database info.
134 auto* objectStoreInfo = m_objectStore.transaction().database().info().infoForExistingObjectStore(m_objectStore.info().identifier());
135 if (!objectStoreInfo)
136 return;
137
138 if (!objectStoreInfo->hasIndex(m_info.identifier())) {
139 m_deleted = true;
140 return;
141 }
142
143 m_info = m_originalInfo;
144 m_deleted = false;
145}

147ExceptionOr<Ref<IDBRequest>> IDBIndex::doOpenCursor(JSGlobalObject& execState, IDBCursorDirection direction, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()>&& function)
148{
149 LOG(IndexedDB, "IDBIndex::openCursor");
150 ASSERT(canCurrentThreadAccessThreadLocalData(m_objectStore.transaction().database().originThread()));
151
152 if (m_deleted || m_objectStore.isDeleted())
153 return Exception { InvalidStateError, "Failed to execute 'openCursor' on 'IDBIndex': The index or its object store has been deleted."_s };
154
155 if (!m_objectStore.transaction().isActive())
156 return Exception { TransactionInactiveError, "Failed to execute 'openCursor' on 'IDBIndex': The transaction is inactive or finished."_s };
157
158 auto keyRange = function();
159 if (keyRange.hasException())
160 return keyRange.releaseException();
161
162 IDBKeyRangeData rangeData = keyRange.returnValue().get();
163 if (rangeData.lowerKey.isNull())
164 rangeData.lowerKey = IDBKeyData::minimum();
165 if (rangeData.upperKey.isNull())
166 rangeData.upperKey = IDBKeyData::maximum();
167
168 auto info = IDBCursorInfo::indexCursor(m_objectStore.transaction(), m_objectStore.info().identifier(), m_info.identifier(), rangeData, direction, IndexedDB::CursorType::KeyAndValue);
169 return m_objectStore.transaction().requestOpenCursor(execState, *this, info);
170}

245ExceptionOr<Ref<IDBRequest>> IDBIndex::doCount(JSGlobalObject& execState, const IDBKeyRangeData& range)
246{
247 ASSERT(canCurrentThreadAccessThreadLocalData(m_objectStore.transaction().database().originThread()));
248
249 if (m_deleted || m_objectStore.isDeleted())
250 return Exception { InvalidStateError, "Failed to execute 'count' on 'IDBIndex': The index or its object store has been deleted."_s };
251
252 auto& transaction = m_objectStore.transaction();
253 if (!transaction.isActive())
254 return Exception { TransactionInactiveError, "Failed to execute 'count' on 'IDBIndex': The transaction is inactive or finished."_s };
255
256 if (!range.isValid())
257 return Exception { DataError, "Failed to execute 'count' on 'IDBIndex': The parameter is not a valid key."_s };
258
259 return transaction.requestCount(execState, *this, range);
260}

280ExceptionOr<Ref<IDBRequest>> IDBIndex::doGet(JSGlobalObject& execState, ExceptionOr<IDBKeyRangeData> range)
281{
282 ASSERT(canCurrentThreadAccessThreadLocalData(m_objectStore.transaction().database().originThread()));
283
284 if (m_deleted || m_objectStore.isDeleted())
285 return Exception { InvalidStateError, "Failed to execute 'get' on 'IDBIndex': The index or its object store has been deleted."_s };
