48{
49 auto* buffer = arrayBuffer.impl();

53Ref<IDBKey> IDBKey::createBinary(JSC::JSArrayBufferView& arrayBufferView)
54{
55 auto bufferView = arrayBufferView.possiblySharedImpl();
56 if (!bufferView)
57 return createInvalid();
58 return adoptRef(*new IDBKey(ThreadSafeDataBuffer::copyData(bufferView->data(), bufferView->byteLength())));
59}

91bool IDBKey::isValid() const
92{
93 if (m_type == IndexedDB::KeyType::Invalid)
94 return false;
95
96 if (m_type == IndexedDB::KeyType::Array) {
97 for (auto& key : WTF::get<IDBKeyVector>(m_value)) {
98 if (!key->isValid())
99 return false;
100 }
101 }
102
103 return true;
104}

106int IDBKey::compare(const IDBKey& other) const
107{
108 if (m_type != other.m_type)
109 return m_type > other.m_type ? -1 : 1;
110
111 switch (m_type) {
112 case IndexedDB::KeyType::Array: {
113 auto& array = WTF::get<IDBKeyVector>(m_value);
114 auto& otherArray = WTF::get<IDBKeyVector>(other.m_value);
115 for (size_t i = 0; i < array.size() && i < otherArray.size(); ++i) {
116 if (int result = array[i]->compare(*otherArray[i]))
117 return result;
118 }
119 if (array.size() < otherArray.size())
120 return -1;
121 if (array.size() > otherArray.size())
122 return 1;
123 return 0;
124 }
125 case IndexedDB::KeyType::Binary:
126 return compareBinaryKeyData(WTF::get<ThreadSafeDataBuffer>(m_value), WTF::get<ThreadSafeDataBuffer>(other.m_value));
127 case IndexedDB::KeyType::String:
128 return -codePointCompare(WTF::get<String>(other.m_value), WTF::get<String>(m_value));
129 case IndexedDB::KeyType::Date:
130 case IndexedDB::KeyType::Number: {
131 auto number = WTF::get<double>(m_value);
132 auto otherNumber = WTF::get<double>(other.m_value);
133 return (number < otherNumber) ? -1 : ((number > otherNumber) ? 1 : 0);
134 }
135 case IndexedDB::KeyType::Invalid:
136 case IndexedDB::KeyType::Min:
137 case IndexedDB::KeyType::Max:
138 ASSERT_NOT_REACHED();
139 return 0;
140 }
141
142 ASSERT_NOT_REACHED();
143 return 0;
144}
145
146bool IDBKey::isLessThan(const IDBKey& other) const
147{
148 return compare(other) == -1;
149}
150
151bool IDBKey::isEqual(const IDBKey& other) const
152{
153 return !compare(other);
154}
