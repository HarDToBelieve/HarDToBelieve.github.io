35IDBKeyData::IDBKeyData(const IDBKey* key)
36 : m_type(IndexedDB::KeyType::Invalid)
37{
38 if (!key) {
39 m_isNull = true;
40 return;
41 }
42
43 m_type = key->type();
44
45 switch (m_type) {
46 case IndexedDB::KeyType::Invalid:
47 break;
48 case IndexedDB::KeyType::Array: {
49 m_value = Vector<IDBKeyData>();
50 auto& array = WTF::get<Vector<IDBKeyData>>(m_value);
51 for (auto& key2 : key->array())
52 array.append(IDBKeyData(key2.get()));
53 break;
54 }
55 case IndexedDB::KeyType::Binary:
56 m_value = key->binary();
57 break;
58 case IndexedDB::KeyType::String:
59 m_value = key->string();
60 break;
61 case IndexedDB::KeyType::Date:
62 m_value = key->date();
63 break;
64 case IndexedDB::KeyType::Number:
65 m_value = key->number();
66 break;
67 case IndexedDB::KeyType::Max:
68 case IndexedDB::KeyType::Min:
69 break;
70 }
71}

73RefPtr<IDBKey> IDBKeyData::maybeCreateIDBKey() const
74{
75 if (m_isNull)
76 return nullptr;
77
78 switch (m_type) {
79 case IndexedDB::KeyType::Invalid:
80 return IDBKey::createInvalid();
81 case IndexedDB::KeyType::Array: {
82 Vector<RefPtr<IDBKey>> array;
83 for (auto& keyData : WTF::get<Vector<IDBKeyData>>(m_value)) {
84 array.append(keyData.maybeCreateIDBKey());
85 ASSERT(array.last());
86 }
87 return IDBKey::createArray(array);
88 }
89 case IndexedDB::KeyType::Binary:
90 return IDBKey::createBinary(WTF::get<ThreadSafeDataBuffer>(m_value));
91 case IndexedDB::KeyType::String:
92 return IDBKey::createString(WTF::get<String>(m_value));
93 case IndexedDB::KeyType::Date:
94 return IDBKey::createDate(WTF::get<double>(m_value));
95 case IndexedDB::KeyType::Number:
96 return IDBKey::createNumber(WTF::get<double>(m_value));
97 case IndexedDB::KeyType::Max:
98 case IndexedDB::KeyType::Min:
99 ASSERT_NOT_REACHED();
100 return nullptr;
101 }
102
103 ASSERT_NOT_REACHED();
104 return nullptr;
105}

117void IDBKeyData::isolatedCopy(const IDBKeyData& source, IDBKeyData& destination)
118{
119 destination.m_type = source.m_type;
120 destination.m_isNull = source.m_isNull;
121
122 switch (source.m_type) {
123 case IndexedDB::KeyType::Invalid:
124 return;
125 case IndexedDB::KeyType::Array: {
126 destination.m_value = Vector<IDBKeyData>();
127 auto& destinationArray = WTF::get<Vector<IDBKeyData>>(destination.m_value);
128 for (auto& key : WTF::get<Vector<IDBKeyData>>(source.m_value))
129 destinationArray.append(key.isolatedCopy());
130 return;
131 }
132 case IndexedDB::KeyType::Binary:
133 destination.m_value = WTF::get<ThreadSafeDataBuffer>(source.m_value);
134 return;
135 case IndexedDB::KeyType::String:
136 destination.m_value = WTF::get<String>(source.m_value).isolatedCopy();
137 return;
138 case IndexedDB::KeyType::Date:
139 case IndexedDB::KeyType::Number:
140 destination.m_value = WTF::get<double>(source.m_value);
141 return;
142 case IndexedDB::KeyType::Max:
143 case IndexedDB::KeyType::Min:
144 return;
145 }
146
147 ASSERT_NOT_REACHED();
148}

150void IDBKeyData::encode(KeyedEncoder& encoder) const
151{
152 encoder.encodeBool("null", m_isNull);
153 if (m_isNull)
154 return;
155
156 encoder.encodeEnum("type", m_type);
157
158 switch (m_type) {
159 case IndexedDB::KeyType::Invalid:
160 return;
161 case IndexedDB::KeyType::Array: {
162 auto& array = WTF::get<Vector<IDBKeyData>>(m_value);
163 encoder.encodeObjects("array", array.begin(), array.end(), [](KeyedEncoder& encoder, const IDBKeyData& key) {
164 key.encode(encoder);
165 });
166 return;
167 }
168 case IndexedDB::KeyType::Binary: {
169 auto* data = WTF::get<ThreadSafeDataBuffer>(m_value).data();
170 encoder.encodeBool("hasBinary", !!data);
171 if (data)
172 encoder.encodeBytes("binary", data->data(), data->size());
173 return;
174 }
175 case IndexedDB::KeyType::String:
176 encoder.encodeString("string", WTF::get<String>(m_value));
177 return;
178 case IndexedDB::KeyType::Date:
179 case IndexedDB::KeyType::Number:
180 encoder.encodeDouble("number", WTF::get<double>(m_value));
181 return;
182 case IndexedDB::KeyType::Max:
183 case IndexedDB::KeyType::Min:
184 return;
185 }
186
187 ASSERT_NOT_REACHED();
188}

190bool IDBKeyData::decode(KeyedDecoder& decoder, IDBKeyData& result)
191{
192 if (!decoder.decodeBool("null", result.m_isNull))
193 return false;
194
195 if (result.m_isNull)
196 return true;
197
198 auto enumFunction = [](int64_t value) {
199 return value == IndexedDB::KeyType::Max
200 || value == IndexedDB::KeyType::Invalid
201 || value == IndexedDB::KeyType::Array
202 || value == IndexedDB::KeyType::Binary
203 || value == IndexedDB::KeyType::String
204 || value == IndexedDB::KeyType::Date
205 || value == IndexedDB::KeyType::Number
206 || value == IndexedDB::KeyType::Min;
207 };
208 if (!decoder.decodeEnum("type", result.m_type, enumFunction))
209 return false;
210
211 if (result.m_type == IndexedDB::KeyType::Invalid)
212 return true;
213
214 if (result.m_type == IndexedDB::KeyType::Max)
215 return true;
216
217 if (result.m_type == IndexedDB::KeyType::Min)
218 return true;
219
220 if (result.m_type == IndexedDB::KeyType::String) {
221 result.m_value = String();
222 return decoder.decodeString("string", WTF::get<String>(result.m_value));
223 }
224
225 if (result.m_type == IndexedDB::KeyType::Number || result.m_type == IndexedDB::KeyType::Date) {
226 result.m_value = 0.0;
227 return decoder.decodeDouble("number", WTF::get<double>(result.m_value));
228 }
229
230 if (result.m_type == IndexedDB::KeyType::Binary) {
231 result.m_value = ThreadSafeDataBuffer();
232
233 bool hasBinaryData;
234 if (!decoder.decodeBool("hasBinary", hasBinaryData))
235 return false;
236
237 if (!hasBinaryData)
238 return true;
239
240 Vector<uint8_t> bytes;
241 if (!decoder.decodeBytes("binary", bytes))
242 return false;
243
244 result.m_value = ThreadSafeDataBuffer::create(WTFMove(bytes));
245 return true;
246 }
247
248 ASSERT(result.m_type == IndexedDB::KeyType::Array);
249
250 auto arrayFunction = [](KeyedDecoder& decoder, IDBKeyData& result) {
251 return decode(decoder, result);
252 };
253
254 result.m_value = Vector<IDBKeyData>();
255 return decoder.decodeObjects("array", WTF::get<Vector<IDBKeyData>>(result.m_value), arrayFunction);
256}

258int IDBKeyData::compare(const IDBKeyData& other) const
259{
260 if (m_type == IndexedDB::KeyType::Invalid) {
261 if (other.m_type != IndexedDB::KeyType::Invalid)
262 return -1;
263 if (other.m_type == IndexedDB::KeyType::Invalid)
264 return 0;
265 } else if (other.m_type == IndexedDB::KeyType::Invalid)
266 return 1;
267
268 // The IDBKey::m_type enum is in reverse sort order.
269 if (m_type != other.m_type)
270 return m_type < other.m_type ? 1 : -1;
271
272 // The types are the same, so handle actual value comparison.
273 switch (m_type) {
274 case IndexedDB::KeyType::Invalid:
275 // Invalid type should have been fully handled above
276 ASSERT_NOT_REACHED();
277 return 0;
278 case IndexedDB::KeyType::Array: {
279 auto& array = WTF::get<Vector<IDBKeyData>>(m_value);
280 auto& otherArray = WTF::get<Vector<IDBKeyData>>(other.m_value);
281 for (size_t i = 0; i < array.size() && i < otherArray.size(); ++i) {
282 if (int result = array[i].compare(otherArray[i]))
283 return result;
284 }
285 if (array.size() < otherArray.size())
286 return -1;
287 if (array.size() > otherArray.size())
288 return 1;
289 return 0;
290 }
291 case IndexedDB::KeyType::Binary:
292 return compareBinaryKeyData(WTF::get<ThreadSafeDataBuffer>(m_value), WTF::get<ThreadSafeDataBuffer>(other.m_value));
293 case IndexedDB::KeyType::String:
294 return codePointCompare(WTF::get<String>(m_value), WTF::get<String>(other.m_value));
295 case IndexedDB::KeyType::Date:
296 case IndexedDB::KeyType::Number: {
297 auto number = WTF::get<double>(m_value);
298 auto otherNumber = WTF::get<double>(other.m_value);
299
300 if (number == otherNumber)
301 return 0;
302 return number > otherNumber ? 1 : -1;
303 }
304 case IndexedDB::KeyType::Max:
305 case IndexedDB::KeyType::Min:
306 return 0;
307 }
308
309 ASSERT_NOT_REACHED();
310 return 0;
311}

313#if !LOG_DISABLED
314String IDBKeyData::loggingString() const
315{
316 if (m_isNull)
317 return "<null>"_s;
318
319 String result;
320
321 switch (m_type) {
322 case IndexedDB::KeyType::Invalid:
323 return "<invalid>"_s;
324 case IndexedDB::KeyType::Array: {
325 StringBuilder builder;
326 builder.append("<array> - { ");
327 auto& array = WTF::get<Vector<IDBKeyData>>(m_value);
328 for (size_t i = 0; i < array.size(); ++i) {
329 builder.append(array[i].loggingString());
330 if (i < array.size() - 1)
331 builder.append(", ");
332 }
333 builder.append(" }");
334 result = builder.toString();
335 break;
336 }
337 case IndexedDB::KeyType::Binary: {
338 StringBuilder builder;
339 builder.append("<binary> - ");
340
341 auto* data = WTF::get<ThreadSafeDataBuffer>(m_value).data();
342 if (!data) {
343 builder.append("(null)");
344 result = builder.toString();
345 break;
346 }
347
348 size_t i = 0;
349 for (; i < 8 && i < data->size(); ++i) {
350 uint8_t byte = data->at(i);
351 builder.append(upperNibbleToLowercaseASCIIHexDigit(byte));
352 builder.append(lowerNibbleToLowercaseASCIIHexDigit(byte));
353 }
354
355 if (data->size() > 8)
356 builder.append("...");
357
358 result = builder.toString();
359 break;
360 }
361 case IndexedDB::KeyType::String:
362 result = "<string> - " + WTF::get<String>(m_value);
363 break;
364 case IndexedDB::KeyType::Date:
365 return makeString("<date> - ", WTF::get<double>(m_value));
366 case IndexedDB::KeyType::Number:
367 return makeString("<number> - ", WTF::get<double>(m_value));
368 case IndexedDB::KeyType::Max:
369 return "<maximum>"_s;
370 case IndexedDB::KeyType::Min:
371 return "<minimum>"_s;
372 }
373
374 if (result.length() > 150) {
375 result.truncate(147);
376 result.append("..."_s);
377 }
378
379 return result;
380}

423bool IDBKeyData::isValid() const
424{
425 if (m_type == IndexedDB::KeyType::Invalid)
426 return false;
427
428 if (m_type == IndexedDB::KeyType::Array) {
429 for (auto& key : array()) {
430 if (!key.isValid())
431 return false;
432 }
433 }
434
435 return true;
436}

443bool IDBKeyData::operator==(const IDBKeyData& other) const
444{
445 if (m_type != other.m_type || m_isNull != other.m_isNull || m_isDeletedValue != other.m_isDeletedValue)
446 return false;
447 switch (m_type) {
448 case IndexedDB::KeyType::Invalid:
449 case IndexedDB::KeyType::Max:
450 case IndexedDB::KeyType::Min:
451 return true;
452 case IndexedDB::KeyType::Number:
453 case IndexedDB::KeyType::Date:
454 return WTF::get<double>(m_value) == WTF::get<double>(other.m_value);
455 case IndexedDB::KeyType::String:
456 return WTF::get<String>(m_value) == WTF::get<String>(other.m_value);
457 case IndexedDB::KeyType::Binary:
458 return WTF::get<ThreadSafeDataBuffer>(m_value) == WTF::get<ThreadSafeDataBuffer>(other.m_value);
459 case IndexedDB::KeyType::Array:
460 return WTF::get<Vector<IDBKeyData>>(m_value) == WTF::get<Vector<IDBKeyData>>(other.m_value);
461 }
462 RELEASE_ASSERT_NOT_REACHED();
463}

465size_t IDBKeyData::size() const
466{
467 if (m_isNull)
468 return 0;
469
470 switch (m_type) {
471 case IndexedDB::KeyType::Invalid:
472 return 0;
473 case IndexedDB::KeyType::Array: {
474 Vector<RefPtr<IDBKey>> array;
475 size_t totalSize = 0;
476 for (auto& keyData : WTF::get<Vector<IDBKeyData>>(m_value))
477 totalSize += keyData.size();
478 return totalSize;
479 }
480 case IndexedDB::KeyType::Binary:
481 return WTF::get<ThreadSafeDataBuffer>(m_value).size();
482 case IndexedDB::KeyType::String:
483 return WTF::get<String>(m_value).sizeInBytes();
484 case IndexedDB::KeyType::Date:
485 case IndexedDB::KeyType::Number:
486 return sizeof(double);
487 case IndexedDB::KeyType::Max:
488 case IndexedDB::KeyType::Min:
489 return 0;
490 }
491
492 ASSERT_NOT_REACHED();
493 return 0;
494}
