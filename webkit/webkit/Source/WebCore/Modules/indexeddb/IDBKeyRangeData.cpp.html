33IDBKeyRangeData::IDBKeyRangeData(IDBKey* key)
34 : lowerKey(key)
35 , upperKey(key)
36 , isNull(!key)
37{
38}
39
40IDBKeyRangeData::IDBKeyRangeData(const IDBKeyData& keyData)
41 : lowerKey(keyData)
42 , upperKey(keyData)
43 , isNull(keyData.isNull())
44{
45}
46
47IDBKeyRangeData IDBKeyRangeData::isolatedCopy() const
48{
49 IDBKeyRangeData result;
50
51 result.isNull = isNull;
52 result.lowerKey = lowerKey.isolatedCopy();
53 result.upperKey = upperKey.isolatedCopy();
54 result.lowerOpen = lowerOpen;
55 result.upperOpen = upperOpen;
56
57 return result;
58}
59
60RefPtr<IDBKeyRange> IDBKeyRangeData::maybeCreateIDBKeyRange() const
61{
62 if (isNull)
63 return nullptr;
64
65 return IDBKeyRange::create(lowerKey.maybeCreateIDBKey(), upperKey.maybeCreateIDBKey(), lowerOpen, upperOpen);
66}
67
68bool IDBKeyRangeData::isExactlyOneKey() const
69{
70 if (isNull || lowerOpen || upperOpen || !upperKey.isValid() || !lowerKey.isValid())
71 return false;
72
73 return !lowerKey.compare(upperKey);
74}
75
76bool IDBKeyRangeData::containsKey(const IDBKeyData& key) const
77{
78 if (lowerKey.isValid()) {
79 auto compare = lowerKey.compare(key);
80 if (compare > 0)
81 return false;
82 if (lowerOpen && !compare)
83 return false;
84 }
85 if (upperKey.isValid()) {
86 auto compare = upperKey.compare(key);
87 if (compare < 0)
88 return false;
89 if (upperOpen && !compare)
90 return false;
91 }
92
93 return true;
94}
95
96bool IDBKeyRangeData::isValid() const
97{
98 if (isNull)
99 return false;
100
101 if (!lowerKey.isValid() && !lowerKey.isNull())
102 return false;
103
104 if (!upperKey.isValid() && !upperKey.isNull())
105 return false;
106
107 return true;
108}
