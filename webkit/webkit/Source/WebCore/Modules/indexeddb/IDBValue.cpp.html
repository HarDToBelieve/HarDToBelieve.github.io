49IDBValue::IDBValue(const SerializedScriptValue& scriptValue, const Vector<String>& blobURLs, const Vector<String>& blobFilePaths)

71void IDBValue::setAsIsolatedCopy(const IDBValue& other)
72{
73 ASSERT(m_blobURLs.isEmpty() && m_blobFilePaths.isEmpty());
74
75 m_data = other.m_data;
76 m_blobURLs = crossThreadCopy(other.m_blobURLs);
77 m_blobFilePaths = crossThreadCopy(other.m_blobFilePaths);
78}
79
80IDBValue IDBValue::isolatedCopy() const
81{
82 IDBValue result;
83 result.setAsIsolatedCopy(*this);
84 return result;
85}
86
87size_t IDBValue::size() const
88{
89 size_t totalSize = 0;
90
91 for (auto& url : m_blobURLs)
92 totalSize += url.sizeInBytes();
93
94 for (auto& path : m_blobFilePaths)
95 totalSize += path.sizeInBytes();
96
97 totalSize += m_data.size();
98
99 return totalSize;
100}
