26#pragma once
27
28#include "ThreadSafeDataBuffer.h"
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class SerializedScriptValue;
34
35class IDBValue {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 WEBCORE_EXPORT IDBValue();
39 IDBValue(const SerializedScriptValue&);
40 IDBValue(const ThreadSafeDataBuffer&);
41 IDBValue(const SerializedScriptValue&, const Vector<String>& blobURLs, const Vector<String>& blobFilePaths);
42 IDBValue(const ThreadSafeDataBuffer&, Vector<String>&& blobURLs, Vector<String>&& blobFilePaths);
43 IDBValue(const ThreadSafeDataBuffer&, const Vector<String>& blobURLs, const Vector<String>& blobFilePaths);
44
45 void setAsIsolatedCopy(const IDBValue&);
46 WEBCORE_EXPORT IDBValue isolatedCopy() const;
47
48 const ThreadSafeDataBuffer& data() const { return m_data; }
49 const Vector<String>& blobURLs() const { return m_blobURLs; }
50 const Vector<String>& blobFilePaths() const { return m_blobFilePaths; }
51
52 template<class Encoder> void encode(Encoder&) const;
53 template<class Decoder> static std::optional<IDBValue> decode(Decoder&);
54
55 size_t size() const;
56private:
57 ThreadSafeDataBuffer m_data;
58 Vector<String> m_blobURLs;
59 Vector<String> m_blobFilePaths;
60};
61
62template<class Encoder>
63void IDBValue::encode(Encoder& encoder) const
64{
65 encoder << m_data;
66 encoder << m_blobURLs;
67 encoder << m_blobFilePaths;
68}
69
70template<class Decoder>
71std::optional<IDBValue> IDBValue::decode(Decoder& decoder)
72{
73 IDBValue result;
74 if (!decoder.decode(result.m_data))
75 return std::nullopt;
76
77 if (!decoder.decode(result.m_blobURLs))
78 return std::nullopt;
79
80 if (!decoder.decode(result.m_blobFilePaths))
81 return std::nullopt;
82
83 return result;
84}
85
86} // namespace WebCore
