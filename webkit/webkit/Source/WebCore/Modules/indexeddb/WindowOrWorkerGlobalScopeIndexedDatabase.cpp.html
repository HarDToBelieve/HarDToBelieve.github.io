80DOMWindowIndexedDatabase* DOMWindowIndexedDatabase::from(DOMWindow& window)
81{
82 auto* supplement = static_cast<DOMWindowIndexedDatabase*>(Supplement<DOMWindow>::from(&window, supplementName()));
83 if (!supplement) {
84 auto newSupplement = makeUnique<DOMWindowIndexedDatabase>(window);
85 supplement = newSupplement.get();
86 provideTo(&window, supplementName(), WTFMove(newSupplement));
87 }
88 return supplement;
89}
90
91IDBFactory* DOMWindowIndexedDatabase::indexedDB()
92{
93 auto* window = this->window();
94 if (!window)
95 return nullptr;
96
97 auto* document = window->document();
98 if (!document)
99 return nullptr;
100
101 auto* page = document->page();
102 if (!page)
103 return nullptr;
104
105 if (!window->isCurrentlyDisplayedInFrame())
106 return nullptr;
107
108 if (!m_idbFactory) {
109 auto* connectionProxy = document->idbConnectionProxy();
110 if (!connectionProxy)
111 return nullptr;
112
113 m_idbFactory = IDBFactory::create(*connectionProxy);
114 }
115
116 return m_idbFactory.get();
117}

126WorkerGlobalScopeIndexedDatabase* WorkerGlobalScopeIndexedDatabase::from(WorkerGlobalScope& scope)
127{
128 auto* supplement = static_cast<WorkerGlobalScopeIndexedDatabase*>(Supplement<WorkerGlobalScope>::from(&scope, supplementName()));
129 if (!supplement) {
130 auto* connectionProxy = scope.idbConnectionProxy();
131 if (!connectionProxy)
132 return nullptr;
133
134 auto newSupplement = makeUnique<WorkerGlobalScopeIndexedDatabase>(*connectionProxy);
135 supplement = newSupplement.get();
136 provideTo(&scope, supplementName(), WTFMove(newSupplement));
137 }
138 return supplement;
139}
140
141IDBFactory* WorkerGlobalScopeIndexedDatabase::indexedDB()
142{
143 if (!m_idbFactory)
144 m_idbFactory = IDBFactory::create(m_connectionProxy.get());
145 return m_idbFactory.get();
146}

149IDBFactory* WindowOrWorkerGlobalScopeIndexedDatabase::indexedDB(WorkerGlobalScope& scope)
150{
151 auto* scopeIDB = WorkerGlobalScopeIndexedDatabase::from(scope);
152 return scopeIDB ? scopeIDB->indexedDB() : nullptr;
153}
154
155IDBFactory* WindowOrWorkerGlobalScopeIndexedDatabase::indexedDB(DOMWindow& window)
156{
157 return DOMWindowIndexedDatabase::from(window)->indexedDB();
158}
