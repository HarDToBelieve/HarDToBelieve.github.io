135 template<typename... Parameters, typename... Arguments>
136 void callConnectionOnMainThread(void (IDBConnectionToServer::*method)(Parameters...), Arguments&&... arguments)
137 {
138 if (isMainThread())
139 (m_connectionToServer.*method)(std::forward<Arguments>(arguments)...);
140 else
141 postMainThreadTask(m_connectionToServer, method, arguments...);
142 }
143
144 template<typename... Arguments>
145 void postMainThreadTask(Arguments&&... arguments)
146 {
147 auto task = createCrossThreadTask(arguments...);
148 m_mainThreadQueue.append(WTFMove(task));
149
150 scheduleMainThreadTasks();
151 }
