189void IDBConnectionToClient::registerDatabaseConnection(UniqueIDBDatabaseConnection& connection)
190{
191 ASSERT(!m_databaseConnections.contains(&connection));
192 m_databaseConnections.add(&connection);
193}
194
195void IDBConnectionToClient::unregisterDatabaseConnection(UniqueIDBDatabaseConnection& connection)
196{
197 m_databaseConnections.remove(&connection);
198}
199
200void IDBConnectionToClient::connectionToClientClosed()
201{
202 m_isClosed = true;
203 auto databaseConnections = m_databaseConnections;
204
205 for (auto* connection : databaseConnections) {
206 ASSERT(m_databaseConnections.contains(connection));
207 connection->connectionClosedFromClient();
208 }
209
210 ASSERT(m_databaseConnections.isEmpty());
211}
