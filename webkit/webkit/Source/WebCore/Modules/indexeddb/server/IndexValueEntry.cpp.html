34IndexValueEntry::IndexValueEntry(bool unique)
35 : m_unique(unique)
36{
37 if (m_unique)
38 m_key = nullptr;
39 else
40 m_orderedKeys = new IDBKeyDataSet;
41}
42
43IndexValueEntry::~IndexValueEntry()
44{
45 if (m_unique)
46 delete m_key;
47 else
48 delete m_orderedKeys;
49}
50
51void IndexValueEntry::addKey(const IDBKeyData& key)
52{
53 if (m_unique) {
54 delete m_key;
55 m_key = new IDBKeyData(key);
56 return;
57 }
58
59 m_orderedKeys->insert(key);
60}
61
62bool IndexValueEntry::removeKey(const IDBKeyData& key)
63{
64 if (m_unique) {
65 if (m_key && *m_key == key) {
66 delete m_key;
67 m_key = nullptr;
68 return true;
69 }
70
71 return false;
72 }
73
74 return m_orderedKeys->erase(key);
75}
76
77const IDBKeyData* IndexValueEntry::getLowest() const
78{
79 if (m_unique)
80 return m_key;
81
82 if (m_orderedKeys->empty())
83 return nullptr;
84
85 return &(*m_orderedKeys->begin());
86}
87
88uint64_t IndexValueEntry::getCount() const
89{
90 if (m_unique)
91 return m_key ? 1 : 0;
92
93 return m_orderedKeys->size();
94}

115const IDBKeyData& IndexValueEntry::Iterator::key() const
116{
117 ASSERT(isValid());
118 if (m_entry->unique()) {
119 ASSERT(m_entry->m_key);
120 return *m_entry->m_key;
121 }
122
123 return m_forward ? *m_forwardIterator : *m_reverseIterator;
124}
125
126bool IndexValueEntry::Iterator::isValid() const
127{
128#if !LOG_DISABLED
129 if (m_entry) {
130 if (m_entry->m_unique)
131 ASSERT(m_entry->m_key);
132 else
133 ASSERT(m_entry->m_orderedKeys);
134 }
135#endif
136
137 return m_entry;
138}
139
140void IndexValueEntry::Iterator::invalidate()
141{
142 m_entry = nullptr;
143}
144
145IndexValueEntry::Iterator& IndexValueEntry::Iterator::operator++()
146{
147 if (!isValid())
148 return *this;
149
150 if (m_entry->m_unique) {
151 invalidate();
152 return *this;
153 }
154
155 if (m_forward) {
156 ++m_forwardIterator;
157 if (m_forwardIterator == m_entry->m_orderedKeys->end())
158 invalidate();
159 } else {
160 ++m_reverseIterator;
161 if (m_reverseIterator == m_entry->m_orderedKeys->rend())
162 invalidate();
163 }
164
165 return *this;
166}

168IndexValueEntry::Iterator IndexValueEntry::begin()
169{
170 if (m_unique) {
171 ASSERT(m_key);
172 return { *this };
173 }
174
175 ASSERT(m_orderedKeys);
176 return { *this, m_orderedKeys->begin() };
177}
178
179IndexValueEntry::Iterator IndexValueEntry::reverseBegin(CursorDuplicity duplicity)
180{
181 if (m_unique) {
182 ASSERT(m_key);
183 return { *this };
184 }
185
186 ASSERT(m_orderedKeys);
187
188 if (duplicity == CursorDuplicity::Duplicates)
189 return { *this, m_orderedKeys->rbegin() };
190
191 auto iterator = m_orderedKeys->rend();
192 --iterator;
193 return { *this, iterator };
194}
195
196IndexValueEntry::Iterator IndexValueEntry::find(const IDBKeyData& key)
197{
198 if (m_unique) {
199 ASSERT(m_key);
200 return *m_key == key ? IndexValueEntry::Iterator(*this) : IndexValueEntry::Iterator();
201 }
202
203 ASSERT(m_orderedKeys);
204 auto iterator = m_orderedKeys->lower_bound(key);
205 if (iterator == m_orderedKeys->end())
206 return { };
207
208 return { *this, iterator };
209}
210
211IndexValueEntry::Iterator IndexValueEntry::reverseFind(const IDBKeyData& key, CursorDuplicity)
212{
213 if (m_unique) {
214 ASSERT(m_key);
215 return *m_key == key ? IndexValueEntry::Iterator(*this) : IndexValueEntry::Iterator();
216 }
217
218 ASSERT(m_orderedKeys);
219 auto iterator = IDBKeyDataSet::reverse_iterator(m_orderedKeys->upper_bound(key));
220 if (iterator == m_orderedKeys->rend())
221 return { };
222
223 return { *this, iterator };
224}
