66void MemoryIndex::objectStoreCleared()
67{
68 auto transaction = m_objectStore.writeTransaction();
69 ASSERT(transaction);
70
71 transaction->indexCleared(*this, WTFMove(m_records));
72
73 notifyCursorsOfAllRecordsChanged();
74}
75
76void MemoryIndex::notifyCursorsOfValueChange(const IDBKeyData& indexKey, const IDBKeyData& primaryKey)
77{
78 for (auto* cursor : copyToVector(m_cleanCursors))
79 cursor->indexValueChanged(indexKey, primaryKey);
80}
81
82void MemoryIndex::notifyCursorsOfAllRecordsChanged()
83{
84 for (auto* cursor : copyToVector(m_cleanCursors))
85 cursor->indexRecordsAllChanged();
86
87 ASSERT(m_cleanCursors.isEmpty());
88}

106IDBGetResult MemoryIndex::getResultForKeyRange(IndexedDB::IndexRecordType type, const IDBKeyRangeData& range) const
107{
108 LOG(IndexedDB, "MemoryIndex::getResultForKeyRange - %s", range.loggingString().utf8().data());
109
110 if (!m_records)
111 return { };
112
113 IDBKeyData keyToLookFor;
114 if (range.isExactlyOneKey())
115 keyToLookFor = range.lowerKey;
116 else
117 keyToLookFor = m_records->lowestKeyWithRecordInRange(range);
118
119 if (keyToLookFor.isNull())
120 return { };
121
122 const IDBKeyData* keyValue = m_records->lowestValueForKey(keyToLookFor);
123
124 if (!keyValue)
125 return { };
126
127 return type == IndexedDB::IndexRecordType::Key ? IDBGetResult(*keyValue) : IDBGetResult(*keyValue, m_objectStore.valueForKeyRange(*keyValue), m_objectStore.info().keyPath());
128}

130uint64_t MemoryIndex::countForKeyRange(const IDBKeyRangeData& inRange)
131{
132 LOG(IndexedDB, "MemoryIndex::countForKeyRange");
133
134 if (!m_records)
135 return 0;
136
137 uint64_t count = 0;
138 IDBKeyRangeData range = inRange;
139 while (true) {
140 auto key = m_records->lowestKeyWithRecordInRange(range);
141 if (key.isNull())
142 break;
143
144 count += m_records->countForKey(key);
145
146 range.lowerKey = key;
147 range.lowerOpen = true;
148 }
149
150 return count;
151}

153void MemoryIndex::getAllRecords(const IDBKeyRangeData& keyRangeData, std::optional<uint32_t> count, IndexedDB::GetAllType type, IDBGetAllResult& result) const
154{
155 LOG(IndexedDB, "MemoryIndex::getAllRecords");
156
157 result = { type, m_objectStore.info().keyPath() };
158
159 if (!m_records)
160 return;
161
162 uint32_t targetCount;
163 if (count && count.value())
164 targetCount = count.value();
165 else
166 targetCount = std::numeric_limits<uint32_t>::max();
167
168 IDBKeyRangeData range = keyRangeData;
169 uint32_t currentCount = 0;
170 while (currentCount < targetCount) {
171 auto key = m_records->lowestKeyWithRecordInRange(range);
172 if (key.isNull())
173 return;
174
175 range.lowerKey = key;
176 range.lowerOpen = true;
177
178 auto allValues = m_records->allValuesForKey(key, targetCount - currentCount);
179 for (auto& keyValue : allValues) {
180 result.addKey(IDBKeyData(keyValue));
181 if (type == IndexedDB::GetAllType::Values)
182 result.addValue(m_objectStore.valueForKeyRange(keyValue));
183 }
184
185 currentCount += allValues.size();
186 }
187}

190IDBError MemoryIndex::putIndexKey(const IDBKeyData& valueKey, const IndexKey& indexKey)
191{
192 LOG(IndexedDB, "MemoryIndex::provisionalPutIndexKey");
193
194 if (!m_records) {
195 m_records = makeUnique<IndexValueStore>(m_info.unique());
196 notifyCursorsOfAllRecordsChanged();
197 }
198
199 if (!m_info.multiEntry()) {
200 IDBKeyData key = indexKey.asOneKey();
201 IDBError result = m_records->addRecord(key, valueKey);
202 notifyCursorsOfValueChange(key, valueKey);
203 return result;
204 }
205
206 Vector<IDBKeyData> keys = indexKey.multiEntry();
207
208 if (m_info.unique()) {
209 for (auto& key : keys) {
210 if (m_records->contains(key))
211 return IDBError(ConstraintError);
212 }
213 }
214
215 for (auto& key : keys) {
216 auto error = m_records->addRecord(key, valueKey);
217 ASSERT_UNUSED(error, error.isNull());
218 notifyCursorsOfValueChange(key, valueKey);
219 }
220
221 return IDBError { };
222}

224void MemoryIndex::removeRecord(const IDBKeyData& valueKey, const IndexKey& indexKey)
225{
226 LOG(IndexedDB, "MemoryIndex::removeRecord");
227
228 ASSERT(m_records);
229
230 if (!m_info.multiEntry()) {
231 IDBKeyData key = indexKey.asOneKey();
232 m_records->removeRecord(key, valueKey);
233 notifyCursorsOfValueChange(key, valueKey);
234 return;
235 }
236
237 Vector<IDBKeyData> keys = indexKey.multiEntry();
238 for (auto& key : keys) {
239 m_records->removeRecord(key, valueKey);
240 notifyCursorsOfValueChange(key, valueKey);
241 }
242}

254MemoryIndexCursor* MemoryIndex::maybeOpenCursor(const IDBCursorInfo& info)
255{
256 auto result = m_cursors.add(info.identifier(), nullptr);
257 if (!result.isNewEntry)
258 return nullptr;
259
260 result.iterator->value = makeUnique<MemoryIndexCursor>(*this, info);
261 return result.iterator->value.get();
262}
