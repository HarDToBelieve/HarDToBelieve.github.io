40MemoryIndexCursor::MemoryIndexCursor(MemoryIndex& index, const IDBCursorInfo& info)
41 : MemoryCursor(info)
42 , m_index(index)
43{
44 LOG(IndexedDB, "MemoryIndexCursor::MemoryIndexCursor %s", info.range().loggingString().utf8().data());
45
46 auto* valueStore = m_index.valueStore();
47 if (!valueStore)
48 return;
49
50 if (m_info.isDirectionForward())
51 m_currentIterator = valueStore->find(m_info.range().lowerKey, m_info.range().lowerOpen);
52 else
53 m_currentIterator = valueStore->reverseFind(m_info.range().upperKey, m_info.duplicity(), m_info.range().upperOpen);
54
55 if (m_currentIterator.isValid() && m_info.range().containsKey(m_currentIterator.key())) {
56 m_currentKey = m_currentIterator.key();
57 m_currentPrimaryKey = m_currentIterator.primaryKey();
58 m_index.cursorDidBecomeClean(*this);
59 } else
60 m_currentIterator.invalidate();
61}

63MemoryIndexCursor::~MemoryIndexCursor() = default;

65void MemoryIndexCursor::currentData(IDBGetResult& getResult)
66{
67 if (!m_currentIterator.isValid()) {
68 getResult = { };
69 return;
70 }
71
72 if (m_info.cursorType() == IndexedDB::CursorType::KeyOnly)
73 getResult = { m_currentKey, m_currentPrimaryKey };
74 else {
75 IDBValue value = { m_index.objectStore().valueForKey(m_currentPrimaryKey), { }, { } };
76 getResult = { m_currentKey, m_currentPrimaryKey, WTFMove(value), m_index.objectStore().info().keyPath() };
77 }
78}

80void MemoryIndexCursor::iterate(const IDBKeyData& key, const IDBKeyData& primaryKey, uint32_t count, IDBGetResult& getResult)
81{
82 LOG(IndexedDB, "MemoryIndexCursor::iterate to key %s, %u count", key.loggingString().utf8().data(), count);
83
84#ifndef NDEBUG
85 if (primaryKey.isValid())
86 ASSERT(key.isValid());
87#endif
88
89 if (key.isValid()) {
90 // Cannot iterate by both a count and to a key
91 ASSERT(!count);
92
93 auto* valueStore = m_index.valueStore();
94 if (!valueStore) {
95 m_currentKey = { };
96 m_currentPrimaryKey = { };
97 getResult = { };
98 return;
99 }
100
101 if (primaryKey.isValid()) {
102 if (m_info.isDirectionForward())
103 m_currentIterator = valueStore->find(key, primaryKey);
104 else
105 m_currentIterator = valueStore->reverseFind(key, primaryKey, m_info.duplicity());
106 } else {
107 if (m_info.isDirectionForward())
108 m_currentIterator = valueStore->find(key);
109 else
110 m_currentIterator = valueStore->reverseFind(key, m_info.duplicity());
111 }
112
113 if (m_currentIterator.isValid() && !m_info.range().containsKey(m_currentIterator.key()))
114 m_currentIterator.invalidate();
115
116 if (!m_currentIterator.isValid()) {
117 m_currentKey = { };
118 m_currentPrimaryKey = { };
119 getResult = { };
120 return;
121 }
122
123 m_index.cursorDidBecomeClean(*this);
124
125 m_currentKey = m_currentIterator.key();
126 m_currentPrimaryKey = m_currentIterator.primaryKey();
127 currentData(getResult);
128
129 return;
130 }
131
132 // If there was not a valid key argument and no positive count argument
133 // that means the default iteration count of "1"
134 if (!count)
135 count = 1;
136
137 if (!m_currentIterator.isValid()) {
138 auto* valueStore = m_index.valueStore();
139 if (!valueStore) {
140 m_currentKey = { };
141 m_currentPrimaryKey = { };
142 getResult = { };
143 return;
144 }
145
146 switch (m_info.cursorDirection()) {
147 case IndexedDB::CursorDirection::Next:
148 m_currentIterator = valueStore->find(m_currentKey, m_currentPrimaryKey);
149 break;
150 case IndexedDB::CursorDirection::Nextunique:
151 m_currentIterator = valueStore->find(m_currentKey, true);
152 break;
153 case IndexedDB::CursorDirection::Prev:
154 m_currentIterator = valueStore->reverseFind(m_currentKey, m_currentPrimaryKey, m_info.duplicity());
155 break;
156 case IndexedDB::CursorDirection::Prevunique:
157 m_currentIterator = valueStore->reverseFind(m_currentKey, m_info.duplicity(), true);
158 break;
159 }
160
161 if (!m_currentIterator.isValid()) {
162 m_currentKey = { };
163 m_currentPrimaryKey = { };
164 getResult = { };
165 return;
166 }
167
168 m_index.cursorDidBecomeClean(*this);
169
170 // If we restored the current iterator and it does *not* match the current key/primaryKey,
171 // then it is the next record in line and we should consider that an iteration.
172 if (m_currentKey != m_currentIterator.key() || m_currentPrimaryKey != m_currentIterator.primaryKey())
173 --count;
174 }
175
176 ASSERT(m_currentIterator.isValid());
177
178 while (count) {
179 if (m_info.duplicity() == CursorDuplicity::NoDuplicates)
180 m_currentIterator.nextIndexEntry();
181 else
182 ++m_currentIterator;
183
184 if (!m_currentIterator.isValid())
185 break;
186
187 --count;
188 }
189
190 if (m_currentIterator.isValid() && !m_info.range().containsKey(m_currentIterator.key()))
191 m_currentIterator.invalidate();
192
193 // Not having a valid iterator after finishing any iteration means we've reached the end of the cursor.
194 if (!m_currentIterator.isValid()) {
195 m_currentKey = { };
196 m_currentPrimaryKey = { };
197 getResult = { };
198 return;
199 }
200
201 m_currentKey = m_currentIterator.key();
202 m_currentPrimaryKey = m_currentIterator.primaryKey();
203 currentData(getResult);
204}
205
206void MemoryIndexCursor::indexRecordsAllChanged()
207{
208 m_currentIterator.invalidate();
209 m_index.cursorDidBecomeDirty(*this);
210}
211
212void MemoryIndexCursor::indexValueChanged(const IDBKeyData& key, const IDBKeyData& primaryKey)
213{
214 if (m_currentKey != key || m_currentPrimaryKey != primaryKey)
215 return;
216
217 m_currentIterator.invalidate();
218 m_index.cursorDidBecomeDirty(*this);
219}
