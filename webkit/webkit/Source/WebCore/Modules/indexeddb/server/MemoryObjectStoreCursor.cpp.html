36MemoryObjectStoreCursor::MemoryObjectStoreCursor(MemoryObjectStore& objectStore, const IDBCursorInfo& info)
37 : MemoryCursor(info)
38 , m_objectStore(objectStore)
39 , m_remainingRange(info.range())
40{
41 LOG(IndexedDB, "MemoryObjectStoreCursor::MemoryObjectStoreCursor %s", info.range().loggingString().utf8().data());
42
43 auto* orderedKeys = objectStore.orderedKeys();
44 if (!orderedKeys)
45 return;
46
47 setFirstInRemainingRange(*orderedKeys);
48}

55void MemoryObjectStoreCursor::keyDeleted(const IDBKeyData& key)

63void MemoryObjectStoreCursor::keyAdded(IDBKeyDataSet::iterator iterator)
64{
65 if (m_iterator)

72void MemoryObjectStoreCursor::setFirstInRemainingRange(IDBKeyDataSet& set)
73{
74 m_iterator = std::nullopt;
75
76 if (m_info.isDirectionForward()) {
77 setForwardIteratorFromRemainingRange(set);
78 if (m_iterator) {
79 m_remainingRange.lowerKey = **m_iterator;
80 m_remainingRange.lowerOpen = true;
81 }
82 } else {
83 setReverseIteratorFromRemainingRange(set);
84 if (m_iterator) {
85 m_remainingRange.upperKey = **m_iterator;
86 m_remainingRange.upperOpen = true;
87 }
88 }
89
90 ASSERT(!m_iterator || *m_iterator != set.end());
91}

93void MemoryObjectStoreCursor::setForwardIteratorFromRemainingRange(IDBKeyDataSet& set)
94{
95 if (!set.size()) {
96 m_iterator = std::nullopt;
97 return;
98 }
99
100 if (m_remainingRange.isExactlyOneKey()) {
101 m_iterator = set.find(m_remainingRange.lowerKey);
102 if (*m_iterator == set.end())
103 m_iterator = std::nullopt;
104
105 return;
106 }
107
108 m_iterator = std::nullopt;
109
110 auto lowest = set.lower_bound(m_remainingRange.lowerKey);
111 if (lowest == set.end())
112 return;
113
114 if (m_remainingRange.lowerOpen && *lowest == m_remainingRange.lowerKey) {
115 ++lowest;
116 if (lowest == set.end())
117 return;
118 }
119
120 if (!m_remainingRange.upperKey.isNull()) {
121 if (lowest->compare(m_remainingRange.upperKey) > 0)
122 return;
123
124 if (m_remainingRange.upperOpen && *lowest == m_remainingRange.upperKey)
125 return;
126 }
127
128 m_iterator = lowest;
129}

131void MemoryObjectStoreCursor::setReverseIteratorFromRemainingRange(IDBKeyDataSet& set)
132{
133 if (!set.size()) {
134 m_iterator = std::nullopt;
135 return;
136 }
137
138 if (m_remainingRange.isExactlyOneKey()) {
139 m_iterator = set.find(m_remainingRange.lowerKey);
140 if (*m_iterator == set.end())
141 m_iterator = std::nullopt;
142
143 return;
144 }
145
146 if (!m_remainingRange.upperKey.isValid()) {
147 m_iterator = --set.end();
148 if (!m_remainingRange.containsKey(**m_iterator))
149 m_iterator = std::nullopt;
150
151 return;
152 }
153
154 m_iterator = std::nullopt;
155
156 // This is one record past the actual key we're looking for.
157 auto highest = set.upper_bound(m_remainingRange.upperKey);
158
159 if (highest == set.begin())
160 return;
161
162 // This is one record before that, which *is* the key we're looking for.
163 --highest;
164
165 if (m_remainingRange.upperOpen && *highest == m_remainingRange.upperKey) {
166 if (highest == set.begin())
167 return;
168 --highest;
169 }
170
171 if (!m_remainingRange.lowerKey.isNull()) {
172 if (highest->compare(m_remainingRange.lowerKey) < 0)
173 return;
174
175 if (m_remainingRange.lowerOpen && *highest == m_remainingRange.lowerKey)
176 return;
177 }
178
179 m_iterator = highest;
180}

182void MemoryObjectStoreCursor::currentData(IDBGetResult& data)
183{
184 if (!m_iterator) {
185 m_currentPositionKey = { };
186 data = { };
187 return;
188 }
189
190 m_currentPositionKey = **m_iterator;
191 if (m_info.cursorType() == IndexedDB::CursorType::KeyOnly)
192 data = { m_currentPositionKey, m_currentPositionKey };
193 else {
194 IDBValue value = { m_objectStore.valueForKeyRange(m_currentPositionKey), { }, { } };
195 data = { m_currentPositionKey, m_currentPositionKey, WTFMove(value), m_objectStore.info().keyPath() };
196 }
197}

199void MemoryObjectStoreCursor::incrementForwardIterator(IDBKeyDataSet& set, const IDBKeyData& key, uint32_t count)
200{
201 // We might need to re-grab the current iterator.
202 // e.g. If the record it was pointed to had been deleted.
203 bool didResetIterator = false;
204 if (!m_iterator) {
205 if (!m_currentPositionKey.isValid())
206 return;
207
208 m_remainingRange.lowerKey = m_currentPositionKey;
209 m_remainingRange.lowerOpen = false;
210 setFirstInRemainingRange(set);
211
212 didResetIterator = true;
213 }
214
215 if (!m_iterator)
216 return;
217
218 ASSERT(*m_iterator != set.end());
219
220 if (key.isValid()) {
221 // If iterating to a key, the count passed in must be zero, as there is no way to iterate by both a count and to a key.
222 ASSERT(!count);
223
224 if (!m_info.range().containsKey(key))
225 return;
226
227 if ((*m_iterator)->compare(key) < 0) {
228 m_remainingRange.lowerKey = key;
229 m_remainingRange.lowerOpen = false;
230 setFirstInRemainingRange(set);
231 }
232
233 return;
234 }
235
236 if (!count)
237 count = 1;
238
239 // If the forwardIterator was reset because it's previous record had been deleted,
240 // we might have already advanced past the current position, eating one one of the iteration counts.
241 if (didResetIterator && (*m_iterator)->compare(m_currentPositionKey) > 0)
242 --count;
243
244 while (count) {
245 --count;
246 ++*m_iterator;
247
248 if (*m_iterator == set.end() || !m_info.range().containsKey(**m_iterator)) {
249 m_iterator = std::nullopt;
250 return;
251 }
252 }
253}

255void MemoryObjectStoreCursor::incrementReverseIterator(IDBKeyDataSet& set, const IDBKeyData& key, uint32_t count)
256{
257 // We might need to re-grab the current iterator.
258 // e.g. If the record it was pointed to had been deleted.
259 bool didResetIterator = false;
260 if (!m_iterator) {
261 if (!m_currentPositionKey.isValid())
262 return;
263
264 m_remainingRange.upperKey = m_currentPositionKey;
265 m_remainingRange.upperOpen = false;
266 setFirstInRemainingRange(set);
267
268 didResetIterator = true;
269 }
270
271 if (!m_iterator || *m_iterator == set.end())
272 return;
273
274 if (key.isValid()) {
275 // If iterating to a key, the count passed in must be zero, as there is no way to iterate by both a count and to a key.
276 ASSERT(!count);
277
278 if (!m_info.range().containsKey(key))
279 return;
280
281 if ((*m_iterator)->compare(key) > 0) {
282 m_remainingRange.upperKey = key;
283 m_remainingRange.upperOpen = false;
284
285 setFirstInRemainingRange(set);
286 }
287
288 return;
289 }
290
291 if (!count)
292 count = 1;
293
294 // If the reverseIterator was reset because it's previous record had been deleted,
295 // we might have already advanced past the current position, eating one one of the iteration counts.
296 if (didResetIterator && (*m_iterator)->compare(m_currentPositionKey) < 0)
297 --count;
298
299 while (count) {
300 if (*m_iterator == set.begin()) {
301 m_iterator = std::nullopt;
302 return;
303 }
304
305 --count;
306 --*m_iterator;
307
308 if (!m_info.range().containsKey(**m_iterator)) {
309 m_iterator = std::nullopt;
310 return;
311 }
312 }

315void MemoryObjectStoreCursor::iterate(const IDBKeyData& key, const IDBKeyData& primaryKeyData, uint32_t count, IDBGetResult& outData)
316{
317 LOG(IndexedDB, "MemoryObjectStoreCursor::iterate to key %s", key.loggingString().utf8().data());
318
319 ASSERT_UNUSED(primaryKeyData, primaryKeyData.isNull());
320
321 if (!m_objectStore.orderedKeys()) {
322 m_currentPositionKey = { };
323 outData = { };
324 return;
325 }
326
327 if (key.isValid() && !m_info.range().containsKey(key)) {
328 m_currentPositionKey = { };
329 outData = { };
330 return;
331 }
332
333 auto* set = m_objectStore.orderedKeys();
334 if (set) {
335 if (m_info.isDirectionForward())
336 incrementForwardIterator(*set, key, count);
337 else
338 incrementReverseIterator(*set, key, count);
339 }
340
341 m_currentPositionKey = { };
342
343 if (!m_iterator) {
344 outData = { };
345 return;
346 }
347
348 currentData(outData);
349}
