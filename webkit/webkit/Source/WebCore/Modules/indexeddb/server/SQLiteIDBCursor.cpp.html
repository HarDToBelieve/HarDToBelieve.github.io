102void SQLiteIDBCursor::currentData(IDBGetResult& result, const std::optional<IDBKeyPath>& keyPath, ShouldIncludePrefetchedRecords shouldIncludePrefetchedRecords)
103{
104 ASSERT(!m_fetchedRecords.isEmpty());
105
106 auto& currentRecord = m_fetchedRecords.first();
107 if (currentRecord.completed) {
108 ASSERT(!currentRecord.errored);
109 result = { };
110 return;
111 }
112
113 if (shouldIncludePrefetchedRecords == ShouldIncludePrefetchedRecords::No) {
114 result = { currentRecord.record.key, currentRecord.record.primaryKey, IDBValue(currentRecord.record.value), keyPath };
115 return;
116 }
117
118 Vector<IDBCursorRecord> prefetchedRecords;
119 prefetchedRecords.reserveCapacity(m_fetchedRecords.size());
120 for (auto& record : m_fetchedRecords) {
121 if (record.isTerminalRecord())
122 break;
123
124 prefetchedRecords.append(record.record);
125 }
126
127 // First record will be returned as current record.
128 if (!prefetchedRecords.isEmpty())
129 prefetchedRecords.remove(0);
130
131 result = { currentRecord.record.key, currentRecord.record.primaryKey, IDBValue(currentRecord.record.value), keyPath, WTFMove(prefetchedRecords) };
132}

134static String buildPreIndexStatement(bool isDirectionNext)
135{
136 return makeString("SELECT rowid, key, value FROM IndexRecords WHERE indexID = ? AND key = CAST(? AS TEXT) AND value ",
137 isDirectionNext ? '>' : '<', " CAST(? AS TEXT) ORDER BY value", isDirectionNext ? "" : " DESC", ';');
138}
139
140static String buildIndexStatement(const IDBKeyRangeData& keyRange, IndexedDB::CursorDirection cursorDirection)
141{
142 return makeString("SELECT rowid, key, value FROM IndexRecords WHERE indexID = ? AND key ",
143 !keyRange.lowerKey.isNull() && !keyRange.lowerOpen ? ">=" : ">",
144 " CAST(? AS TEXT) AND key ",
145 !keyRange.upperKey.isNull() && !keyRange.upperOpen ? "<=" : "<",
146 " CAST(? AS TEXT) ORDER BY key",
147 cursorDirection == IndexedDB::CursorDirection::Prev || cursorDirection == IndexedDB::CursorDirection::Prevunique ? " DESC" : "",
148 ", value",
149 cursorDirection == IndexedDB::CursorDirection::Prev ? " DESC" : "",
150 ';');
151}
152
153static String buildObjectStoreStatement(const IDBKeyRangeData& keyRange, IndexedDB::CursorDirection cursorDirection)
154{
155 return makeString("SELECT rowid, key, value FROM Records WHERE objectStoreID = ? AND key ",
156 !keyRange.lowerKey.isNull() && !keyRange.lowerOpen ? ">=" : ">",
157 " CAST(? AS TEXT) AND key ",
158 !keyRange.upperKey.isNull() && !keyRange.upperOpen ? "<=" : "<",
159 " CAST(? AS TEXT) ORDER BY key",
160 cursorDirection == IndexedDB::CursorDirection::Prev || cursorDirection == IndexedDB::CursorDirection::Prevunique ? " DESC" : "",
161 ';');
162}

164bool SQLiteIDBCursor::establishStatement()
165{
166 ASSERT(!m_statement);
167 String sql;
168
169 if (m_indexID != IDBIndexInfo::InvalidId) {
170 sql = buildIndexStatement(m_keyRange, m_cursorDirection);
171 m_boundID = m_indexID;
172 } else {
173 sql = buildObjectStoreStatement(m_keyRange, m_cursorDirection);
174 m_boundID = m_objectStoreID;
175 }
176
177 m_currentLowerKey = m_keyRange.lowerKey.isNull() ? IDBKeyData::minimum() : m_keyRange.lowerKey;
178 m_currentUpperKey = m_keyRange.upperKey.isNull() ? IDBKeyData::maximum() : m_keyRange.upperKey;
179
180 return createSQLiteStatement(sql);
181}

183bool SQLiteIDBCursor::createSQLiteStatement(const String& sql)
184{
185 LOG(IndexedDB, "Creating cursor with SQL query: \"%s\"", sql.utf8().data());
186
187 ASSERT(!m_currentLowerKey.isNull());
188 ASSERT(!m_currentUpperKey.isNull());
189 ASSERT(m_transaction->sqliteTransaction());
190
191 auto statement = m_transaction->sqliteTransaction()->database().prepareHeapStatementSlow(sql);
192 if (!statement) {
193 LOG_ERROR("Could not create cursor statement (prepare/id) - '%s'", m_transaction->sqliteTransaction()->database().lastErrorMsg());
194 return false;
195 }
196 m_statement = statement.value().moveToUniquePtr();
197
198 return bindArguments();
199}

201void SQLiteIDBCursor::objectStoreRecordsChanged()
202{
203 if (m_statementNeedsReset)
204 return;
205
206 ASSERT(!m_fetchedRecords.isEmpty());
207
208 m_currentKeyForUniqueness = m_fetchedRecords.first().record.key;
209 if (m_indexID != IDBIndexInfo::InvalidId)
210 m_currentIndexRecordValue = m_fetchedRecords.first().record.primaryKey;
211
212 // If ObjectStore or Index contents changed, we need to reset the statement and bind new parameters to it.
213 // This is to pick up any changes that might exist.
214 m_statementNeedsReset = true;
215
216 if (isDirectionNext()) {
217 m_currentLowerKey = m_currentKeyForUniqueness;
218 if (!m_keyRange.lowerOpen) {
219 m_keyRange.lowerOpen = true;
220 m_keyRange.lowerKey = m_currentLowerKey;
221 m_statement = nullptr;
222 }
223 } else {
224 m_currentUpperKey = m_currentKeyForUniqueness;
225 if (!m_keyRange.upperOpen) {
226 m_keyRange.upperOpen = true;
227 m_keyRange.upperKey = m_currentUpperKey;
228 m_statement = nullptr;
229 }
230 }
231
232 // We also need to throw away any fetched records as they may no longer be valid.
233 m_fetchedRecords.clear();
234 m_fetchedRecordsSize = 0;
235
236 m_prefetchCount = 0;
237}

239void SQLiteIDBCursor::resetAndRebindStatement()
240{
241 ASSERT(!m_currentLowerKey.isNull());
242 ASSERT(!m_currentUpperKey.isNull());
243 ASSERT(m_transaction->sqliteTransaction());
244 ASSERT(m_statementNeedsReset);
245
246 m_statementNeedsReset = false;
247
248 if (!m_statement && !establishStatement()) {
249 LOG_ERROR("Unable to establish new statement for cursor iteration");
250 return;
251 }
252
253 if (m_statement->reset() != SQLITE_OK) {
254 LOG_ERROR("Could not reset cursor statement to respond to object store changes");
255 return;
256 }
257
258 bindArguments();
259}

261bool SQLiteIDBCursor::bindArguments()
262{
263 LOG(IndexedDB, "Cursor is binding lower key '%s' and upper key '%s'", m_currentLowerKey.loggingString().utf8().data(), m_currentUpperKey.loggingString().utf8().data());
264
265 int currentBindArgument = 1;
266
267 if (m_statement->bindInt64(currentBindArgument++, m_boundID) != SQLITE_OK) {
268 LOG_ERROR("Could not bind id argument (bound ID)");
269 return false;
270 }
271
272 RefPtr<SharedBuffer> buffer = serializeIDBKeyData(m_currentLowerKey);
273 if (m_statement->bindBlob(currentBindArgument++, *buffer) != SQLITE_OK) {
274 LOG_ERROR("Could not create cursor statement (lower key)");
275 return false;
276 }
277
278 buffer = serializeIDBKeyData(m_currentUpperKey);
279 if (m_statement->bindBlob(currentBindArgument++, *buffer) != SQLITE_OK) {
280 LOG_ERROR("Could not create cursor statement (upper key)");
281 return false;
282 }
283
284 return true;
285}

287bool SQLiteIDBCursor::resetAndRebindPreIndexStatementIfNecessary()
288{
289 if (m_indexID == IDBIndexInfo::InvalidId)
290 return true;
291
292 if (m_currentIndexRecordValue.isNull())
293 return true;
294
295 auto& database = m_transaction->sqliteTransaction()->database();
296 if (!m_preIndexStatement) {
297 auto preIndexStatement = database.prepareHeapStatementSlow(buildPreIndexStatement(isDirectionNext()));
298 if (!preIndexStatement) {
299 LOG_ERROR("Could not prepare pre statement - '%s'", database.lastErrorMsg());
300 return false;
301 }
302 m_preIndexStatement = preIndexStatement.value().moveToUniquePtr();
303 }
304
305 if (m_preIndexStatement->reset() != SQLITE_OK) {
306 LOG_ERROR("Could not reset pre statement - '%s'", database.lastErrorMsg());
307 return false;
308 }
309
310 auto key = isDirectionNext() ? m_currentLowerKey : m_currentUpperKey;
311 int currentBindArgument = 1;
312
313 if (m_preIndexStatement->bindInt64(currentBindArgument++, m_boundID) != SQLITE_OK) {
314 LOG_ERROR("Could not bind id argument to pre statement (bound ID)");
315 return false;
316 }
317
318 RefPtr<SharedBuffer> buffer = serializeIDBKeyData(key);
319 if (m_preIndexStatement->bindBlob(currentBindArgument++, *buffer) != SQLITE_OK) {
320 LOG_ERROR("Could not bind id argument to pre statement (key)");
321 return false;
322 }
323
324 buffer = serializeIDBKeyData(m_currentIndexRecordValue);
325 if (m_preIndexStatement->bindBlob(currentBindArgument++, *buffer) != SQLITE_OK) {
326 LOG_ERROR("Could not bind id argument to pre statement (value)");
327 return false;
328 }
329
330 return true;
331}

333bool SQLiteIDBCursor::prefetchOneRecord()
334{
335 LOG(IndexedDB, "SQLiteIDBCursor::prefetchOneRecord() - Cursor already has %zu fetched records", m_fetchedRecords.size());
336
337 if (m_fetchedRecordsSize >= prefetchSizeLimit || m_fetchedRecords.isEmpty() || m_fetchedRecords.size() >= prefetchLimit || m_fetchedRecords.last().isTerminalRecord())
338 return false;
339
340 m_currentKeyForUniqueness = m_fetchedRecords.last().record.key;
341
342 return fetch() && m_fetchedRecords.size() < prefetchLimit && m_fetchedRecordsSize < prefetchSizeLimit;
343}

350bool SQLiteIDBCursor::prefetch()
351{
352 for (unsigned i = 0; i < m_prefetchCount; ++i) {
353 if (!prefetchOneRecord())
354 return false;
355 }
356
357 increaseCountToPrefetch();
358 return true;

361bool SQLiteIDBCursor::advance(uint64_t count)
362{
363 LOG(IndexedDB, "SQLiteIDBCursor::advance() - Count %" PRIu64 ", %zu fetched records", count, m_fetchedRecords.size());
364 ASSERT(count);
365
366 if (!m_fetchedRecords.isEmpty() && m_fetchedRecords.first().isTerminalRecord()) {
367 LOG_ERROR("Attempt to advance a completed cursor");
368 return false;
369 }
370
371 if (!m_fetchedRecords.isEmpty())
372 m_currentKeyForUniqueness = m_fetchedRecords.last().record.key;
373
374 // Drop already-fetched records up to `count` to see if we've already fetched the record we're looking for.
375 bool hadCurrentRecord = !m_fetchedRecords.isEmpty();
376 for (; count && !m_fetchedRecords.isEmpty(); --count) {
377 if (m_fetchedRecords.first().isTerminalRecord())
378 break;
379
380 ASSERT(m_fetchedRecordsSize >= m_fetchedRecords.first().record.size());
381 m_fetchedRecordsSize -= m_fetchedRecords.first().record.size();
382 m_fetchedRecords.removeFirst();
383 }
384
385 // If we still have any records left, the first record is our new current record.
386 if (!m_fetchedRecords.isEmpty())
387 return true;
388
389 ASSERT(m_fetchedRecords.isEmpty());
390
391 // If we started out with a current record, we burnt a count on removing it.
392 // Replace that count now.
393 if (hadCurrentRecord)
394 ++count;
395
396 for (; count; --count) {
397 if (!m_fetchedRecords.isEmpty()) {
398 ASSERT(m_fetchedRecords.size() == 1);
399 m_currentKeyForUniqueness = m_fetchedRecords.first().record.key;
400
401 ASSERT(m_fetchedRecordsSize >= m_fetchedRecords.first().record.size());
402 m_fetchedRecordsSize -= m_fetchedRecords.first().record.size();
403 m_fetchedRecords.removeFirst();
404 }
405
406 if (!fetch())
407 return false;
408
409 ASSERT(!m_fetchedRecords.isEmpty());
410 ASSERT(!m_fetchedRecords.first().errored);
411 if (m_fetchedRecords.first().completed)
412 break;
413 }
414
415 return true;
416}

418bool SQLiteIDBCursor::fetch()
419{
420 ASSERT(m_fetchedRecords.isEmpty() || !m_fetchedRecords.last().isTerminalRecord());
421
422 m_fetchedRecords.append({ });
423
424 bool isUnique = m_cursorDirection == IndexedDB::CursorDirection::Nextunique || m_cursorDirection == IndexedDB::CursorDirection::Prevunique;
425 if (!isUnique) {
426 bool fetchSucceeded = fetchNextRecord(m_fetchedRecords.last());
427 if (fetchSucceeded)
428 m_fetchedRecordsSize += m_fetchedRecords.last().record.size();
429 return fetchSucceeded;
430 }
431
432 while (fetchNextRecord(m_fetchedRecords.last())) {
433 m_fetchedRecordsSize += m_fetchedRecords.last().record.size();
434
435 if (m_currentKeyForUniqueness.compare(m_fetchedRecords.last().record.key))
436 return true;
437
438 if (m_fetchedRecords.last().completed)
439 return false;
440
441 m_fetchedRecordsSize -= m_fetchedRecords.last().record.size();
442 }
443
444 return false;
445}

447bool SQLiteIDBCursor::fetchNextRecord(SQLiteCursorRecord& record)
448{
449 if (m_statementNeedsReset) {
450 resetAndRebindPreIndexStatementIfNecessary();
451 resetAndRebindStatement();
452 }
453
454 FetchResult result;
455 do {
456 result = internalFetchNextRecord(record);
457 } while (result == FetchResult::ShouldFetchAgain);
458
459 return result == FetchResult::Success;
460}

462void SQLiteIDBCursor::markAsErrored(SQLiteCursorRecord& record)
463{
464 record.record = { };
465 record.completed = true;
466 record.errored = true;
467 record.rowID = 0;

470SQLiteIDBCursor::FetchResult SQLiteIDBCursor::internalFetchNextRecord(SQLiteCursorRecord& record)
471{
472 ASSERT(m_transaction->sqliteTransaction());
473 ASSERT(m_statement);
474 ASSERT(!m_fetchedRecords.isEmpty());
475 ASSERT(!m_fetchedRecords.last().isTerminalRecord());
476
477 record.record.value = { };
478
479 auto& database = m_transaction->sqliteTransaction()->database();
480 SQLiteStatement* statement = nullptr;
481
482 int result;
483 if (m_preIndexStatement) {
484 ASSERT(m_indexID != IDBIndexInfo::InvalidId);
485
486 result = m_preIndexStatement->step();
487 if (result == SQLITE_ROW)
488 statement = m_preIndexStatement.get();
489 else if (result != SQLITE_DONE)
490 LOG_ERROR("Error advancing with pre statement - (%i) %s", result, database.lastErrorMsg());
491 }
492
493 if (!statement) {
494 result = m_statement->step();
495 if (result == SQLITE_DONE) {
496 record = { };
497 record.completed = true;
498 return FetchResult::Success;
499 }
500 if (result != SQLITE_ROW) {
501 LOG_ERROR("Error advancing cursor - (%i) %s", result, database.lastErrorMsg());
502 markAsErrored(record);
503 return FetchResult::Failure;
504 }
505 statement = m_statement.get();
506 }
507
508 record.rowID = statement->columnInt64(0);
509 ASSERT(record.rowID);
510 auto keyDataSpan = statement->columnBlobAsSpan(1);
511
512 if (!deserializeIDBKeyData(keyDataSpan.data(), keyDataSpan.size(), record.record.key)) {
513 LOG_ERROR("Unable to deserialize key data from database while advancing cursor");
514 markAsErrored(record);
515 return FetchResult::Failure;
516 }
517
518 auto keyData = statement->columnBlob(2);
519
520 // The primaryKey of an ObjectStore cursor is the same as its key.
521 if (m_indexID == IDBIndexInfo::InvalidId) {
522 record.record.primaryKey = record.record.key;
523
524 Vector<String> blobURLs, blobFilePaths;
525 auto error = m_transaction->backingStore().getBlobRecordsForObjectStoreRecord(record.rowID, blobURLs, blobFilePaths);
526 if (!error.isNull()) {
527 LOG_ERROR("Unable to fetch blob records from database while advancing cursor");
528 markAsErrored(record);
529 return FetchResult::Failure;
530 }
531
532 if (m_cursorType == IndexedDB::CursorType::KeyAndValue)
533 record.record.value = { ThreadSafeDataBuffer::create(WTFMove(keyData)), blobURLs, blobFilePaths };
534 } else {
535 if (!deserializeIDBKeyData(keyData.data(), keyData.size(), record.record.primaryKey)) {
536 LOG_ERROR("Unable to deserialize value data from database while advancing index cursor");
537 markAsErrored(record);
538 return FetchResult::Failure;
539 }
540
541 if (!m_cachedObjectStoreStatement || m_cachedObjectStoreStatement->reset() != SQLITE_OK) {
542 if (auto cachedObjectStoreStatement = database.prepareHeapStatement("SELECT value FROM Records WHERE key = CAST(? AS TEXT) and objectStoreID = ?;"_s))
543 m_cachedObjectStoreStatement = cachedObjectStoreStatement.value().moveToUniquePtr();
544 }
545
546 if (!m_cachedObjectStoreStatement
547 || m_cachedObjectStoreStatement->bindBlob(1, keyData) != SQLITE_OK
548 || m_cachedObjectStoreStatement->bindInt64(2, m_objectStoreID) != SQLITE_OK) {
549 LOG_ERROR("Could not create index cursor statement into object store records (%i) '%s'", database.lastError(), database.lastErrorMsg());
550 markAsErrored(record);
551 return FetchResult::Failure;
552 }
553
554 int result = m_cachedObjectStoreStatement->step();
555
556 if (result == SQLITE_ROW)
557 record.record.value = { ThreadSafeDataBuffer::create(m_cachedObjectStoreStatement->columnBlob(0)) };
558 else if (result == SQLITE_DONE) {
559 // This indicates that the record we're trying to retrieve has been removed from the object store.
560 // Skip over it.
561 return FetchResult::ShouldFetchAgain;
562 } else {
563 LOG_ERROR("Could not step index cursor statement into object store records (%i) '%s'", database.lastError(), database.lastErrorMsg());
564 markAsErrored(record);
565 return FetchResult::Failure;
566
567 }
568 }
569
570 return FetchResult::Success;
571}

573bool SQLiteIDBCursor::iterate(const IDBKeyData& targetKey, const IDBKeyData& targetPrimaryKey)
574{
575 ASSERT(m_transaction->sqliteTransaction());
576 ASSERT(m_statement);
577
578 bool result = advance(1);
579 ASSERT(!m_fetchedRecords.isEmpty());
580
581 // Iterating with no key is equivalent to advancing 1 step.
582 if (targetKey.isNull() || !result)
583 return result;
584
585 while (!m_fetchedRecords.first().isTerminalRecord()) {
586 if (!result)
587 return false;
588
589 // Search for the next key >= the target if the cursor is a Next cursor, or the next key <= if the cursor is a Previous cursor.
590 if (m_cursorDirection == IndexedDB::CursorDirection::Next || m_cursorDirection == IndexedDB::CursorDirection::Nextunique) {
591 if (m_fetchedRecords.first().record.key.compare(targetKey) >= 0)
592 break;
593 } else if (m_fetchedRecords.first().record.key.compare(targetKey) <= 0)
594 break;
595
596 result = advance(1);
597 }
598
599 if (targetPrimaryKey.isValid()) {
600 while (!m_fetchedRecords.first().isTerminalRecord() && !m_fetchedRecords.first().record.key.compare(targetKey)) {
601 if (!result)
602 return false;
603
604 // Search for the next primary key >= the primary target if the cursor is a Next cursor, or the next key <= if the cursor is a Previous cursor.
605 if (m_cursorDirection == IndexedDB::CursorDirection::Next || m_cursorDirection == IndexedDB::CursorDirection::Nextunique) {
606 if (m_fetchedRecords.first().record.primaryKey.compare(targetPrimaryKey) >= 0)
607 break;
608 } else if (m_fetchedRecords.first().record.primaryKey.compare(targetPrimaryKey) <= 0)
609 break;
610
611 result = advance(1);
612 }
613 }
614
615 return result;
616}
