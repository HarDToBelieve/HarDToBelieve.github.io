39SQLiteIDBTransaction::SQLiteIDBTransaction(SQLiteIDBBackingStore& backingStore, const IDBTransactionInfo& info)
40 : m_info(info)
41 , m_backingStore(backingStore)
42{
43}

45SQLiteIDBTransaction::~SQLiteIDBTransaction()
46{
47 if (inProgress())
48 m_sqliteTransaction->rollback();
49
50 // Explicitly clear cursors, as that also unregisters them from the backing store.
51 clearCursors();
52}

55IDBError SQLiteIDBTransaction::begin(SQLiteDatabase& database)
56{
57 ASSERT(!m_sqliteTransaction);
58
59 m_sqliteTransaction = makeUnique<SQLiteTransaction>(database, m_info.mode() == IDBTransactionMode::Readonly);
60 m_sqliteTransaction->begin();
61
62 if (m_sqliteTransaction->inProgress())
63 return IDBError { };
64
65 return IDBError { UnknownError, "Could not start SQLite transaction in database backend"_s };
66}
67
68IDBError SQLiteIDBTransaction::commit()
69{
70 LOG(IndexedDB, "SQLiteIDBTransaction::commit");
71 if (!m_sqliteTransaction || !m_sqliteTransaction->inProgress())
72 return IDBError { UnknownError, "No SQLite transaction in progress to commit"_s };
73
74 m_sqliteTransaction->commit();
75
76 if (m_sqliteTransaction->inProgress())
77 return IDBError { UnknownError, "Unable to commit SQLite transaction in database backend"_s };
78
79 deleteBlobFilesIfNecessary();
80 moveBlobFilesIfNecessary();
81
82 reset();
83 return IDBError { };
84}
85
86void SQLiteIDBTransaction::moveBlobFilesIfNecessary()
87{
88 String databaseDirectory = m_backingStore.databaseDirectory();
89 for (auto& entry : m_blobTemporaryAndStoredFilenames) {
90 if (!FileSystem::hardLinkOrCopyFile(entry.first, FileSystem::pathByAppendingComponent(databaseDirectory, entry.second)))
91 LOG_ERROR("Failed to link/copy temporary blob file '%s' to location '%s'", entry.first.utf8().data(), FileSystem::pathByAppendingComponent(databaseDirectory, entry.second).utf8().data());
92
93 FileSystem::deleteFile(entry.first);
94 }
95
96 m_blobTemporaryAndStoredFilenames.clear();
97}
98
99void SQLiteIDBTransaction::deleteBlobFilesIfNecessary()
100{
101 if (m_blobRemovedFilenames.isEmpty())
102 return;
103
104 String databaseDirectory = m_backingStore.databaseDirectory();
105 for (auto& entry : m_blobRemovedFilenames) {
106 String fullPath = FileSystem::pathByAppendingComponent(databaseDirectory, entry);
107
108 FileSystem::deleteFile(fullPath);
109 }
110
111 m_blobRemovedFilenames.clear();
112}
113
114IDBError SQLiteIDBTransaction::abort()
115{
116 for (auto& entry : m_blobTemporaryAndStoredFilenames)
117 FileSystem::deleteFile(entry.first);
118
119 m_blobTemporaryAndStoredFilenames.clear();
120
121 if (!m_sqliteTransaction || !m_sqliteTransaction->inProgress())
122 return IDBError { UnknownError, "No SQLite transaction in progress to abort"_s };
123
124 m_sqliteTransaction->rollback();
125
126 if (m_sqliteTransaction->inProgress())
127 return IDBError { UnknownError, "Unable to abort SQLite transaction in database backend"_s };
128
129 reset();
130 return IDBError { };
131}

133void SQLiteIDBTransaction::reset()
134{
135 m_sqliteTransaction = nullptr;
136 clearCursors();
137 ASSERT(m_blobTemporaryAndStoredFilenames.isEmpty());
138}

140std::unique_ptr<SQLiteIDBCursor> SQLiteIDBTransaction::maybeOpenBackingStoreCursor(uint64_t objectStoreID, uint64_t indexID, const IDBKeyRangeData& range)
141{
142 ASSERT(m_sqliteTransaction);
143 ASSERT(m_sqliteTransaction->inProgress());
144
145 auto cursor = SQLiteIDBCursor::maybeCreateBackingStoreCursor(*this, objectStoreID, indexID, range);
146
147 if (cursor)
148 m_backingStoreCursors.add(cursor.get());
149
150 return cursor;
151}
152
153SQLiteIDBCursor* SQLiteIDBTransaction::maybeOpenCursor(const IDBCursorInfo& info)
154{
155 ASSERT(m_sqliteTransaction);
156 if (!m_sqliteTransaction->inProgress())
157 return nullptr;
158
159 auto addResult = m_cursors.add(info.identifier(), SQLiteIDBCursor::maybeCreate(*this, info));
160
161 ASSERT(addResult.isNewEntry);
162
163 // It is possible the cursor failed to create and we just stored a null value.
164 if (!addResult.iterator->value) {
165 m_cursors.remove(addResult.iterator);
166 return nullptr;
167 }
168
169 return addResult.iterator->value.get();
170}

172void SQLiteIDBTransaction::closeCursor(SQLiteIDBCursor& cursor)
173{
174 auto backingStoreTake = m_backingStoreCursors.take(&cursor);
175 if (backingStoreTake) {
176 ASSERT(!m_cursors.contains(cursor.identifier()));
177 return;
178 }
179
180 ASSERT(m_cursors.contains(cursor.identifier()));
181
182 m_backingStore.unregisterCursor(cursor);
183 m_cursors.remove(cursor.identifier());
184}

186void SQLiteIDBTransaction::notifyCursorsOfChanges(int64_t objectStoreID)
187{
188 for (auto& i : m_cursors) {
189 if (i.value->objectStoreID() == objectStoreID)
190 i.value->objectStoreRecordsChanged();
191 }
192
193 for (auto* cursor : m_backingStoreCursors) {
194 if (cursor->objectStoreID() == objectStoreID)
195 cursor->objectStoreRecordsChanged();
196 }
197}
198
199void SQLiteIDBTransaction::clearCursors()
200{
201 for (auto& cursor : m_cursors.values())
202 m_backingStore.unregisterCursor(*cursor);
203
204 m_cursors.clear();
205}

207bool SQLiteIDBTransaction::inProgress() const
208{
209 return m_sqliteTransaction && m_sqliteTransaction->inProgress();
210}

214 m_blobTemporaryAndStoredFilenames.append({ temporaryPath, storedFilename });

217void SQLiteIDBTransaction::addRemovedBlobFile(const String& removedFilename)

220 m_blobRemovedFilenames.add(removedFilename);
