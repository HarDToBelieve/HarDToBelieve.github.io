48class SQLiteIDBTransaction {
49 WTF_MAKE_FAST_ALLOCATED;
50 WTF_MAKE_NONCOPYABLE(SQLiteIDBTransaction);
51public:
52 SQLiteIDBTransaction(SQLiteIDBBackingStore&, const IDBTransactionInfo&);
53 ~SQLiteIDBTransaction();
54
55 const IDBResourceIdentifier& transactionIdentifier() const { return m_info.identifier(); }
56
57 IDBError begin(SQLiteDatabase&);
58 IDBError commit();
59 IDBError abort();
60
61 std::unique_ptr<SQLiteIDBCursor> maybeOpenBackingStoreCursor(uint64_t objectStoreID, uint64_t indexID, const IDBKeyRangeData&);
62 SQLiteIDBCursor* maybeOpenCursor(const IDBCursorInfo&);
63
64 void closeCursor(SQLiteIDBCursor&);
65 void notifyCursorsOfChanges(int64_t objectStoreID);
66
67 IDBTransactionMode mode() const { return m_info.mode(); }
68 bool inProgress() const;
69
70 SQLiteTransaction* sqliteTransaction() const { return m_sqliteTransaction.get(); }
71 SQLiteIDBBackingStore& backingStore() { return m_backingStore; }
72
73 void addBlobFile(const String& temporaryPath, const String& storedFilename);
74 void addRemovedBlobFile(const String& removedFilename);
75
76private:
77 void clearCursors();
78 void reset();
79
80 void moveBlobFilesIfNecessary();
81 void deleteBlobFilesIfNecessary();
82
83 IDBTransactionInfo m_info;
84
85 SQLiteIDBBackingStore& m_backingStore;
86 std::unique_ptr<SQLiteTransaction> m_sqliteTransaction;
87 HashMap<IDBResourceIdentifier, std::unique_ptr<SQLiteIDBCursor>> m_cursors;
88 HashSet<SQLiteIDBCursor*> m_backingStoreCursors;
89 Vector<std::pair<String, String>> m_blobTemporaryAndStoredFilenames;
90 HashSet<String> m_blobRemovedFilenames;
91};
