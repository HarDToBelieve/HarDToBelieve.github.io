34namespace WebCore {
35
36class IDBDatabaseInfo;
37
38namespace IDBServer {
39
40class ServerOpenDBRequest : public RefCounted<ServerOpenDBRequest> {
41public:
42 static Ref<ServerOpenDBRequest> create(IDBConnectionToClient&, const IDBRequestData&);
43
44 IDBConnectionToClient& connection() { return m_connection; }
45 const IDBRequestData& requestData() const { return m_requestData; }
46
47 bool isOpenRequest() const;
48 bool isDeleteRequest() const;
49
50 void maybeNotifyRequestBlocked(uint64_t currentVersion);
51 void notifyDidDeleteDatabase(const IDBDatabaseInfo&);
52
53 uint64_t versionChangeID() const;
54
55 void notifiedConnectionsOfVersionChange(HashSet<uint64_t>&& connectionIdentifiers);
56 void connectionClosedOrFiredVersionChangeEvent(uint64_t connectionIdentifier);
57 bool hasConnectionsPendingVersionChangeEvent() const { return !m_connectionsPendingVersionChangeEvent.isEmpty(); }
58 bool hasNotifiedConnectionsOfVersionChange() const { return m_notifiedConnectionsOfVersionChange; }
59
60
61private:
62 ServerOpenDBRequest(IDBConnectionToClient&, const IDBRequestData&);
63
64 Ref<IDBConnectionToClient> m_connection;
65 IDBRequestData m_requestData;
66
67 bool m_notifiedBlocked { false };
68
69 bool m_notifiedConnectionsOfVersionChange { false };
70 HashSet<uint64_t> m_connectionsPendingVersionChangeEvent;
71};
