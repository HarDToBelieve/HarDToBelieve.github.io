51static const uint64_t defaultWriteOperationCost = 4;
52
53static inline uint64_t estimateSize(const IDBKeyData& keyData)
54{
55 uint64_t size = 4;
56 switch (keyData.type()) {
57 case IndexedDB::KeyType::String:
58 size += keyData.string().sizeInBytes();
59 break;
60 case IndexedDB::KeyType::Binary: {
61 size += keyData.binary().size();
62 break;
63 }
64 case IndexedDB::KeyType::Array:
65 for (auto& data : keyData.array())
66 size += estimateSize(data);
67 break;
68 default:
69 break;
70 }
71 return size;
72}

74static inline uint64_t estimateSize(const IDBObjectStoreInfo& info, const IndexIDToIndexKeyMap& indexKeys, uint64_t primaryKeySize)
75{
76 // Each IndexRecord has 5 columns:
77 // - indexID, objectStoreID, recordID: these are varints, estimate 4 bytes each = 12 bytes
78 // - primary key: use primaryKeySize
79 // - secondary index key: use estimateSize(secondary key)
80 static constexpr uint64_t baseIndexRowSize = 12;
81 uint64_t size = 0;
82
83 for (const auto& entry : indexKeys) {
84 auto indexIterator = info.indexMap().find(entry.key);
85 ASSERT(indexIterator != info.indexMap().end());
86
87 if (indexIterator != info.indexMap().end() && indexIterator->value.multiEntry()) {
88 for (const auto& secondaryKey : entry.value.multiEntry())
89 size += (baseIndexRowSize + primaryKeySize + estimateSize(secondaryKey));
90 } else
91 size += (baseIndexRowSize + primaryKeySize + estimateSize(entry.value.asOneKey()));
92 }
93
94 return size;
95}

97static inline uint64_t estimateSize(const IDBValue& value)
98{
99 uint64_t size = 4;
100 size += value.data().size();
101 for (auto& url : value.blobURLs())
102 size += url.sizeInBytes();
103 for (auto& path : value.blobFilePaths())
104 size += path.sizeInBytes();
105 return size;
106}
107
108static inline uint64_t estimateSize(const IDBKeyPath& keyPath)
109{
110 return WTF::switchOn(keyPath, [](const String& path) {
111 return static_cast<uint64_t>(path.sizeInBytes());
112 }, [](const Vector<String>& paths) {
113 uint64_t size = 0;
114 for (auto path : paths)
115 size += path.sizeInBytes();
116 return size;
117 });
118}
119
120static inline uint64_t estimateSize(const IDBIndexInfo& info)
121{
122 uint64_t size = 4;
123 size += info.name().sizeInBytes();
124 size += estimateSize(info.keyPath());
125 return size;
126}
127
128static inline uint64_t estimateSize(const IDBObjectStoreInfo& info)
129{
130 uint64_t size = 4;
131 size += info.name().sizeInBytes();
132 if (auto keyPath = info.keyPath())
133 size += estimateSize(*keyPath);
134 return size;
135}

181void UniqueIDBDatabase::performCurrentOpenOperation()
182{
183 LOG(IndexedDB, "UniqueIDBDatabase::performCurrentOpenOperation (%p)", this);
184
185 ASSERT(m_currentOpenDBRequest);
186 ASSERT(m_currentOpenDBRequest->isOpenRequest());
187
188 IDBError backingStoreOpenError;
189 if (!m_backingStore) {
190 // Quota check.
191 auto decision = m_server.requestSpace(m_identifier.origin(), defaultWriteOperationCost);
192 if (decision == StorageQuotaManager::Decision::Deny)
193 backingStoreOpenError = IDBError { QuotaExceededError, quotaErrorMessageName("OpenBackingStore") };
194 else {
195 m_backingStore = m_server.createBackingStore(m_identifier);
196 IDBDatabaseInfo databaseInfo;
197 backingStoreOpenError = m_backingStore->getOrEstablishDatabaseInfo(databaseInfo);
198 if (backingStoreOpenError.isNull())
199 m_databaseInfo = makeUnique<IDBDatabaseInfo>(databaseInfo);
200 else {
201 LOG_ERROR("Failed to get database info '%s'", backingStoreOpenError.message().utf8().data());
202 m_backingStore = nullptr;
203 }
204 }
205 }
206
207 if (!backingStoreOpenError.isNull()) {
208 auto result = IDBResultData::error(m_currentOpenDBRequest->requestData().requestIdentifier(), backingStoreOpenError);
209 m_currentOpenDBRequest->connection().didOpenDatabase(result);
210 m_currentOpenDBRequest = nullptr;
211
212 return;
213 }
214
215 // If we previously started a version change operation but were blocked by having open connections,
216 // we might now be unblocked.
217 if (m_versionChangeDatabaseConnection) {
218 if (!m_versionChangeTransaction && !hasAnyOpenConnections())
219 startVersionChangeTransaction();
220 return;
221 }
222
223 // 3.3.1 Opening a database
224 // If requested version is undefined, then let requested version be 1 if db was created in the previous step,
225 // or the current version of db otherwise.
226 uint64_t requestedVersion = m_currentOpenDBRequest->requestData().requestedVersion();
227 if (!requestedVersion)
228 requestedVersion = m_databaseInfo->version() ? m_databaseInfo->version() : 1;
229
230 // 3.3.1 Opening a database
231 // If the database version higher than the requested version, abort these steps and return a VersionError.
232 if (requestedVersion < m_databaseInfo->version()) {
233 auto result = IDBResultData::error(m_currentOpenDBRequest->requestData().requestIdentifier(), IDBError(VersionError));
234 m_currentOpenDBRequest->connection().didOpenDatabase(result);
235 m_currentOpenDBRequest = nullptr;
236
237 return;
238 }
239
240 Ref<UniqueIDBDatabaseConnection> connection = UniqueIDBDatabaseConnection::create(*this, *m_currentOpenDBRequest);
241
242 if (requestedVersion == m_databaseInfo->version()) {
243 auto* rawConnection = &connection.get();
244 addOpenDatabaseConnection(WTFMove(connection));
245
246 auto result = IDBResultData::openDatabaseSuccess(m_currentOpenDBRequest->requestData().requestIdentifier(), *rawConnection);
247 m_currentOpenDBRequest->connection().didOpenDatabase(result);
248 m_currentOpenDBRequest = nullptr;
249
250 return;
251 }
252
253 ASSERT(!m_versionChangeDatabaseConnection);
254 m_versionChangeDatabaseConnection = WTFMove(connection);
255
256 // 3.3.7 "versionchange" transaction steps
257 // If there's no other open connections to this database, the version change process can begin immediately.
258 if (!hasAnyOpenConnections()) {
259 startVersionChangeTransaction();
260 return;
261 }
262
263 // Otherwise we have to notify all those open connections and wait for them to close.
264 maybeNotifyConnectionsOfVersionChange();
265}

267void UniqueIDBDatabase::performCurrentDeleteOperation()
268{
269 LOG(IndexedDB, "UniqueIDBDatabase::performCurrentDeleteOperation - %s", m_identifier.loggingString().utf8().data());
270
271 ASSERT(m_currentOpenDBRequest);
272 ASSERT(m_currentOpenDBRequest->isDeleteRequest());
273
274 if (hasAnyOpenConnections()) {
275 maybeNotifyConnectionsOfVersionChange();
276 return;
277 }
278
279 ASSERT(m_pendingTransactions.isEmpty());
280 ASSERT(m_openDatabaseConnections.isEmpty());
281
282 // It's possible to have multiple delete requests queued up in a row.
283 // In that scenario only the first request will actually have to delete the database.
284 // Subsequent requests can immediately notify their completion.
285
286 if (!m_databaseInfo && m_mostRecentDeletedDatabaseInfo)
287 didDeleteBackingStore(0);
288 else
289 deleteBackingStore();
290}

292void UniqueIDBDatabase::deleteBackingStore()
293{
294 ASSERT(!isMainThread());
295 LOG(IndexedDB, "UniqueIDBDatabase::deleteBackingStore");
296
297 auto backingStore = m_backingStore ? std::exchange(m_backingStore, nullptr) : m_server.createBackingStore(m_identifier);
298 uint64_t deletedVersion = backingStore->databaseVersion();
299 backingStore->deleteBackingStore();
300 didDeleteBackingStore(deletedVersion);
301}
302
303void UniqueIDBDatabase::didDeleteBackingStore(uint64_t deletedVersion)
304{
305 LOG(IndexedDB, "UniqueIDBDatabase::didDeleteBackingStore");
306
307 ASSERT(m_inProgressTransactions.isEmpty());
308 ASSERT(m_pendingTransactions.isEmpty());
309 ASSERT(m_openDatabaseConnections.isEmpty());
310 ASSERT(!m_backingStore);
311
312 ASSERT(m_currentOpenDBRequest->isDeleteRequest());
313
314 if (m_databaseInfo)
315 m_mostRecentDeletedDatabaseInfo = WTFMove(m_databaseInfo);
316
317 // If this UniqueIDBDatabase was brought into existence for the purpose of deleting the file on disk,
318 // we won't have a m_mostRecentDeletedDatabaseInfo. In that case, we'll manufacture one using the
319 // passed in deletedVersion argument.
320 if (!m_mostRecentDeletedDatabaseInfo)
321 m_mostRecentDeletedDatabaseInfo = makeUnique<IDBDatabaseInfo>(m_identifier.databaseName(), deletedVersion, 0);
322
323 if (m_currentOpenDBRequest) {
324 m_currentOpenDBRequest->notifyDidDeleteDatabase(*m_mostRecentDeletedDatabaseInfo);
325 m_currentOpenDBRequest = nullptr;
326 }
327}

335void UniqueIDBDatabase::handleDatabaseOperations()
336{
337 ASSERT(!isMainThread());
338 LOG(IndexedDB, "UniqueIDBDatabase::handleDatabaseOperations - There are %u pending", m_pendingOpenDBRequests.size());
339
340 if (m_versionChangeDatabaseConnection || m_versionChangeTransaction) {
341 // We can't start any new open-database operations right now, but we might be able to start handling a delete operation.
342 if (!m_currentOpenDBRequest)
343 m_currentOpenDBRequest = takeNextRunnableRequest(RequestType::Delete);
344 } else if (!m_currentOpenDBRequest || m_currentOpenDBRequest->connection().isClosed())
345 m_currentOpenDBRequest = takeNextRunnableRequest();
346
347 while (m_currentOpenDBRequest) {
348 handleCurrentOperation();
349 if (!m_currentOpenDBRequest) {
350 if (m_versionChangeTransaction)
351 m_currentOpenDBRequest = takeNextRunnableRequest(RequestType::Delete);
352 else
353 m_currentOpenDBRequest = takeNextRunnableRequest();
354 } else // Request need multiple attempts to handle.
355 break;
356 }
357 LOG(IndexedDB, "UniqueIDBDatabase::handleDatabaseOperations - There are %u pending after this round of handling", m_pendingOpenDBRequests.size());
358}

360void UniqueIDBDatabase::handleCurrentOperation()
361{
362 LOG(IndexedDB, "UniqueIDBDatabase::handleCurrentOperation");
363 ASSERT(m_currentOpenDBRequest);
364
365 if (m_currentOpenDBRequest->isOpenRequest())
366 performCurrentOpenOperation();
367 else if (m_currentOpenDBRequest->isDeleteRequest())
368 performCurrentDeleteOperation();
369 else
370 ASSERT_NOT_REACHED();
371}

396void UniqueIDBDatabase::startVersionChangeTransaction()
397{
398 LOG(IndexedDB, "UniqueIDBDatabase::startVersionChangeTransaction");
399
400 ASSERT(!m_versionChangeTransaction);
401 ASSERT(m_currentOpenDBRequest);
402 ASSERT(m_currentOpenDBRequest->isOpenRequest());
403 ASSERT(m_versionChangeDatabaseConnection);
404
405 uint64_t requestedVersion = m_currentOpenDBRequest->requestData().requestedVersion();
406 if (!requestedVersion)
407 requestedVersion = m_databaseInfo->version() ? m_databaseInfo->version() : 1;
408
409 m_versionChangeTransaction = &m_versionChangeDatabaseConnection->createVersionChangeTransaction(requestedVersion);
410 auto versionChangeTransactionInfo = m_versionChangeTransaction->info();
411 m_inProgressTransactions.set(versionChangeTransactionInfo.identifier(), m_versionChangeTransaction);
412
413 auto error = m_backingStore->beginTransaction(versionChangeTransactionInfo);
414 auto operation = WTFMove(m_currentOpenDBRequest);
415 IDBResultData result;
416 if (error.isNull()) {
417 addOpenDatabaseConnection(*m_versionChangeDatabaseConnection);
418 m_databaseInfo->setVersion(versionChangeTransactionInfo.newVersion());
419 result = IDBResultData::openDatabaseUpgradeNeeded(operation->requestData().requestIdentifier(), *m_versionChangeTransaction);
420 operation->connection().didOpenDatabase(result);
421 } else {
422 m_versionChangeDatabaseConnection->abortTransactionWithoutCallback(*m_versionChangeTransaction);
423 m_versionChangeDatabaseConnection = nullptr;
424 result = IDBResultData::error(operation->requestData().requestIdentifier(), error);
425 operation->connection().didOpenDatabase(result);
426 }
427}

429void UniqueIDBDatabase::maybeNotifyConnectionsOfVersionChange()
430{
431 ASSERT(m_currentOpenDBRequest);
432
433 if (m_currentOpenDBRequest->hasNotifiedConnectionsOfVersionChange())
434 return;
435
436 uint64_t newVersion = m_currentOpenDBRequest->isOpenRequest() ? m_currentOpenDBRequest->requestData().requestedVersion() : 0;
437 auto requestIdentifier = m_currentOpenDBRequest->requestData().requestIdentifier();
438
439 LOG(IndexedDB, "UniqueIDBDatabase::notifyConnectionsOfVersionChange - %" PRIu64, newVersion);
440
441 // 3.3.7 "versionchange" transaction steps
442 // Fire a versionchange event at each connection in m_openDatabaseConnections that is open.
443 // The event must not be fired on connections which has the closePending flag set.
444 HashSet<uint64_t> connectionIdentifiers;
445 for (const auto& connection : m_openDatabaseConnections) {
446 if (connection->closePending())
447 continue;
448
449 connection->fireVersionChangeEvent(requestIdentifier, newVersion);
450 connectionIdentifiers.add(connection->identifier());
451 }
452
453 if (!connectionIdentifiers.isEmpty())
454 m_currentOpenDBRequest->notifiedConnectionsOfVersionChange(WTFMove(connectionIdentifiers));
455 else
456 m_currentOpenDBRequest->maybeNotifyRequestBlocked(m_databaseInfo->version());
457}

459void UniqueIDBDatabase::notifyCurrentRequestConnectionClosedOrFiredVersionChangeEvent(uint64_t connectionIdentifier)
460{
461 LOG(IndexedDB, "UniqueIDBDatabase::notifyCurrentRequestConnectionClosedOrFiredVersionChangeEvent - %" PRIu64, connectionIdentifier);
462
463 ASSERT(m_currentOpenDBRequest);
464
465 m_currentOpenDBRequest->connectionClosedOrFiredVersionChangeEvent(connectionIdentifier);
466
467 if (m_currentOpenDBRequest->hasConnectionsPendingVersionChangeEvent())
468 return;
469
470 if (!hasAnyOpenConnections() || allConnectionsAreClosedOrClosing()) {
471 handleDatabaseOperations();
472 return;
473 }
474
475 // Since all open connections have fired their version change events but not all of them have closed,
476 // this request is officially blocked.
477 m_currentOpenDBRequest->maybeNotifyRequestBlocked(m_databaseInfo->version());
478}

480void UniqueIDBDatabase::clearTransactionsOnConnection(UniqueIDBDatabaseConnection& connection)
481{
482 Deque<RefPtr<UniqueIDBDatabaseTransaction>> pendingTransactions;
483 while (!m_pendingTransactions.isEmpty()) {
484 auto transaction = m_pendingTransactions.takeFirst();
485 if (&transaction->databaseConnection() != &connection)
486 pendingTransactions.append(WTFMove(transaction));
487 else
488 connection.deleteTransaction(*transaction);
489 }
490 if (!pendingTransactions.isEmpty())
491 m_pendingTransactions.swap(pendingTransactions);
492
493 Deque<RefPtr<UniqueIDBDatabaseTransaction>> transactionsToAbort;
494 for (auto& transaction : m_inProgressTransactions.values()) {
495 if (&transaction->databaseConnection() == &connection)
496 transactionsToAbort.append(transaction);
497 }
498 for (auto& transaction : transactionsToAbort)
499 transaction->abortWithoutCallback();
500}

502void UniqueIDBDatabase::didFireVersionChangeEvent(UniqueIDBDatabaseConnection& connection, const IDBResourceIdentifier& requestIdentifier, IndexedDB::ConnectionClosedOnBehalfOfServer connectionClosedOnBehalfOfServer)
503{
504 LOG(IndexedDB, "UniqueIDBDatabase::didFireVersionChangeEvent");
505
506 if (!m_currentOpenDBRequest)
507 return;
508
509 ASSERT_UNUSED(requestIdentifier, m_currentOpenDBRequest->requestData().requestIdentifier() == requestIdentifier);
510
511 if (connectionClosedOnBehalfOfServer == IndexedDB::ConnectionClosedOnBehalfOfServer::Yes) {
512 if (m_openDatabaseConnections.contains(&connection)) {
513 clearTransactionsOnConnection(connection);
514 m_openDatabaseConnections.remove(&connection);
515 }
516 }
517
518 notifyCurrentRequestConnectionClosedOrFiredVersionChangeEvent(connection.identifier());
519}

521void UniqueIDBDatabase::openDBRequestCancelled(const IDBResourceIdentifier& requestIdentifier)
522{
523 LOG(IndexedDB, "UniqueIDBDatabase::openDBRequestCancelled - %s", requestIdentifier.loggingString().utf8().data());
524
525 if (m_currentOpenDBRequest && m_currentOpenDBRequest->requestData().requestIdentifier() == requestIdentifier)
526 m_currentOpenDBRequest = nullptr;
527
528 if (m_versionChangeDatabaseConnection && m_versionChangeDatabaseConnection->openRequestIdentifier() == requestIdentifier) {
529 ASSERT(!m_versionChangeTransaction || m_versionChangeTransaction->databaseConnection().openRequestIdentifier() == requestIdentifier);
530 ASSERT(!m_versionChangeTransaction || &m_versionChangeTransaction->databaseConnection() == m_versionChangeDatabaseConnection);
531
532 connectionClosedFromClient(*m_versionChangeDatabaseConnection);
533 }
534
535 for (auto& request : m_pendingOpenDBRequests) {
536 if (request->requestData().requestIdentifier() == requestIdentifier) {
537 m_pendingOpenDBRequests.remove(request);
538 return;
539 }
540 }

549void UniqueIDBDatabase::createObjectStore(UniqueIDBDatabaseTransaction& transaction, const IDBObjectStoreInfo& info, ErrorCallback callback)
550{
551 ASSERT(!isMainThread());
552 LOG(IndexedDB, "UniqueIDBDatabase::createObjectStore");
553
554 ASSERT(m_backingStore);
555
556 // Quota check.
557 auto taskSize = defaultWriteOperationCost + estimateSize(info);
558 if (m_server.requestSpace(m_identifier.origin(), taskSize) == StorageQuotaManager::Decision::Deny) {
559 callback(IDBError { QuotaExceededError, quotaErrorMessageName("CreateObjectStore") });
560 return;
561 }
562
563 auto error = m_backingStore->createObjectStore(transaction.info().identifier(), info);
564 if (error.isNull())
565 m_databaseInfo->addExistingObjectStore(info);
566
567 callback(error);
568}

677void UniqueIDBDatabase::renameIndex(UniqueIDBDatabaseTransaction& transaction, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const String& newName, ErrorCallback callback)
678{
679 ASSERT(!isMainThread());
680 LOG(IndexedDB, "UniqueIDBDatabase::renameIndex");
681
682 // Quota check.
683 auto taskSize = defaultWriteOperationCost + newName.sizeInBytes();
684 if (m_server.requestSpace(m_identifier.origin(), taskSize) == StorageQuotaManager::Decision::Deny) {
685 callback(IDBError { QuotaExceededError, quotaErrorMessageName("RenameIndex") });
686 return;
687 }
688
689 auto* objectStoreInfo = m_databaseInfo->infoForExistingObjectStore(objectStoreIdentifier);
690 if (!objectStoreInfo) {
691 callback(IDBError { UnknownError, "Attempt to rename index in non-existant object store"_s });
692 return;
693 }
694
695 auto* indexInfo = objectStoreInfo->infoForExistingIndex(indexIdentifier);
696 if (!indexInfo) {
697 callback(IDBError { UnknownError, "Attempt to rename non-existant index"_s });
698 return;
699 }
700
701 ASSERT(m_backingStore);
702 auto error = m_backingStore->renameIndex(transaction.info().identifier(), objectStoreIdentifier, indexIdentifier, newName);
703 if (error.isNull())
704 indexInfo->rename(newName);
705
706 callback(error);
707}

709void UniqueIDBDatabase::putOrAdd(const IDBRequestData& requestData, const IDBKeyData& keyData, const IDBValue& value, IndexedDB::ObjectStoreOverwriteMode overwriteMode, KeyDataCallback callback)
710{
711 ASSERT(!isMainThread());
712 LOG(IndexedDB, "UniqueIDBDatabase::putOrAdd");
713
714 ASSERT(m_databaseInfo);
715
716 IDBKeyData usedKey;
717 IDBError error;
718
719 ASSERT(m_backingStore);
720 auto objectStoreIdentifier = requestData.objectStoreIdentifier();
721 auto* objectStoreInfo = m_backingStore->infoForObjectStore(objectStoreIdentifier);
722 if (!objectStoreInfo) {
723 callback(IDBError { InvalidStateError, "Object store cannot be found in the backing store"_s }, usedKey);
724 return;
725 }
726
727 bool usedKeyIsGenerated = false;
728 uint64_t keyNumber;
729 auto transactionIdentifier = requestData.transactionIdentifier();
730 auto generatedKeyResetter = WTF::makeScopeExit([this, transactionIdentifier, objectStoreIdentifier, &keyNumber, &usedKeyIsGenerated]() {
731 if (usedKeyIsGenerated)
732 m_backingStore->revertGeneratedKeyNumber(transactionIdentifier, objectStoreIdentifier, keyNumber);
733 });
734
735 if (objectStoreInfo->autoIncrement() && !keyData.isValid()) {
736 error = m_backingStore->generateKeyNumber(transactionIdentifier, objectStoreIdentifier, keyNumber);
737 if (!error.isNull()) {
738 callback(error, usedKey);
739 return;
740 }
741 usedKey.setNumberValue(keyNumber);
742 usedKeyIsGenerated = true;
743 } else
744 usedKey = keyData;
745
746 // Generate index keys up front for more accurate quota check.
747 IndexIDToIndexKeyMap indexKeys;
748 callOnIDBSerializationThreadAndWait([objectStoreInfo = objectStoreInfo->isolatedCopy(), key = usedKey.isolatedCopy(), value = value.isolatedCopy(), &indexKeys](auto& globalObject) {
749 indexKeys = generateIndexKeyMapForValue(globalObject, objectStoreInfo, key, value);
750 });
751
752 if (overwriteMode == IndexedDB::ObjectStoreOverwriteMode::NoOverwrite) {
753 bool keyExists;
754 error = m_backingStore->keyExistsInObjectStore(transactionIdentifier, objectStoreIdentifier, usedKey, keyExists);
755 if (error.isNull() && keyExists)
756 error = IDBError { ConstraintError, "Key already exists in the object store"_s };
757
758 if (!error.isNull()) {
759 callback(error, usedKey);
760 return;
761 }
762 }
763
764 // Quota check.
765 auto keySize = estimateSize(usedKey);
766 auto valueSize = estimateSize(value);
767 auto indexSize = estimateSize(*objectStoreInfo, indexKeys, keySize);
768 auto taskSize = defaultWriteOperationCost + keySize + valueSize + indexSize;
769
770 LOG(IndexedDB, "UniqueIDBDatabase::putOrAdd quota check with task size: %" PRIu64 " key size: %" PRIu64 " value size: %" PRIu64 " index size: %" PRIu64, taskSize, keySize, valueSize, indexSize);
771
772 if (m_server.requestSpace(m_identifier.origin(), taskSize) == StorageQuotaManager::Decision::Deny) {
773 callback(IDBError { QuotaExceededError, quotaErrorMessageName("PutOrAdd") }, usedKey);
774 return;
775 }
776
777 // If a record already exists in store, then remove the record from store using the steps for deleting records from an object store.
778 // This is important because formally deleting it from from the object store also removes it from the appropriate indexes.
779 error = m_backingStore->deleteRange(transactionIdentifier, objectStoreIdentifier, usedKey);
780 if (!error.isNull()) {
781 callback(error, usedKey);
782 return;
783 }
784
785 error = m_backingStore->addRecord(transactionIdentifier, *objectStoreInfo, usedKey, indexKeys, value);
786 if (!error.isNull()) {
787 callback(error, usedKey);
788 return;
789 }
790
791 if (overwriteMode != IndexedDB::ObjectStoreOverwriteMode::OverwriteForCursor && objectStoreInfo->autoIncrement() && keyData.type() == IndexedDB::KeyType::Number)
792 error = m_backingStore->maybeUpdateKeyGeneratorNumber(transactionIdentifier, objectStoreIdentifier, keyData.number());
793
794 generatedKeyResetter.release();
795 callback(error, usedKey);
796}

900void UniqueIDBDatabase::abortTransaction(UniqueIDBDatabaseTransaction& transaction, ErrorCallback callback)
901{
902 ASSERT(!isMainThread());
903 LOG(IndexedDB, "UniqueIDBDatabase::abortTransaction - %s", transaction.info().identifier().loggingString().utf8().data());
904
905 ASSERT(transaction.databaseConnection().database() == this);
906
907 auto takenTransaction = m_inProgressTransactions.take(transaction.info().identifier());
908 if (!takenTransaction) {
909 if (!m_openDatabaseConnections.contains(&transaction.databaseConnection()))
910 return;
911
912 callback(IDBError { UnknownError, "Attempt to abort transaction that is not running"_s });
913 return;
914 }
915
916 // If transaction is already aborted on the main thread for suspension,
917 // return the result of that abort.
918 if (auto existingAbortResult = takenTransaction->mainThreadAbortResult()) {
919 callback(*existingAbortResult);
920 transactionCompleted(WTFMove(takenTransaction));
921 return;
922 }
923
924 auto transactionIdentifier = transaction.info().identifier();
925 if (m_versionChangeTransaction && m_versionChangeTransaction->info().identifier() == transactionIdentifier) {
926 ASSERT(m_versionChangeTransaction == &transaction);
927 ASSERT(!m_versionChangeDatabaseConnection || &m_versionChangeTransaction->databaseConnection() == m_versionChangeDatabaseConnection);
928 ASSERT(m_versionChangeTransaction->originalDatabaseInfo());
929 m_databaseInfo = makeUnique<IDBDatabaseInfo>(*m_versionChangeTransaction->originalDatabaseInfo());
930 }
931
932 auto error = m_backingStore->abortTransaction(transactionIdentifier);
933
934 callback(error);
935 transactionCompleted(WTFMove(takenTransaction));

1009void UniqueIDBDatabase::handleTransactions()
1010{
1011 LOG(IndexedDB, "UniqueIDBDatabase::handleTransactions - There are %zu pending", m_pendingTransactions.size());
1012
1013 bool hadDeferredTransactions = false;
1014 auto transaction = takeNextRunnableTransaction(hadDeferredTransactions);
1015
1016 while (transaction) {
1017 m_inProgressTransactions.set(transaction->info().identifier(), transaction);
1018 for (auto objectStore : transaction->objectStoreIdentifiers()) {
1019 m_objectStoreTransactionCounts.add(objectStore);
1020 if (!transaction->isReadOnly()) {
1021 m_objectStoreWriteTransactions.add(objectStore);
1022 ASSERT(m_objectStoreTransactionCounts.count(objectStore) == 1);
1023 }
1024 }
1025
1026 activateTransactionInBackingStore(*transaction);
1027 if (hadDeferredTransactions)
1028 break;
1029 transaction = takeNextRunnableTransaction(hadDeferredTransactions);
1030 }
1031 LOG(IndexedDB, "UniqueIDBDatabase::handleTransactions - There are %zu pending after this round of handling", m_pendingTransactions.size());
1032}

1046{

1055RefPtr<UniqueIDBDatabaseTransaction> UniqueIDBDatabase::takeNextRunnableTransaction(bool& hadDeferredTransactions)
1056{
1057 hadDeferredTransactions = false;
1058
1059 if (m_pendingTransactions.isEmpty())
1060 return nullptr;
1061
1062 if (!m_backingStore->supportsSimultaneousTransactions() && !m_inProgressTransactions.isEmpty()) {
1063 LOG(IndexedDB, "UniqueIDBDatabase::takeNextRunnableTransaction - Backing store only supports 1 transaction, and we already have 1");
1064 return nullptr;
1065 }
1066
1067 Deque<RefPtr<UniqueIDBDatabaseTransaction>> deferredTransactions;
1068 RefPtr<UniqueIDBDatabaseTransaction> currentTransaction;
1069
1070 HashSet<uint64_t> deferredReadWriteScopes;
1071
1072 while (!m_pendingTransactions.isEmpty()) {
1073 currentTransaction = m_pendingTransactions.takeFirst();
1074
1075 switch (currentTransaction->info().mode()) {
1076 case IDBTransactionMode::Readonly: {
1077 bool hasOverlappingScopes = scopesOverlap(deferredReadWriteScopes, currentTransaction->objectStoreIdentifiers());
1078 hasOverlappingScopes |= scopesOverlap(m_objectStoreWriteTransactions, currentTransaction->objectStoreIdentifiers());
1079
1080 if (hasOverlappingScopes)
1081 deferredTransactions.append(WTFMove(currentTransaction));
1082
1083 break;
1084 }
1085 case IDBTransactionMode::Readwrite: {
1086 bool hasOverlappingScopes = scopesOverlap(m_objectStoreTransactionCounts, currentTransaction->objectStoreIdentifiers());
1087 hasOverlappingScopes |= scopesOverlap(deferredReadWriteScopes, currentTransaction->objectStoreIdentifiers());
1088
1089 if (hasOverlappingScopes) {
1090 for (auto objectStore : currentTransaction->objectStoreIdentifiers())
1091 deferredReadWriteScopes.add(objectStore);
1092 deferredTransactions.append(WTFMove(currentTransaction));
1093 }
1094
1095 break;
1096 }
1097 case IDBTransactionMode::Versionchange:
1098 // Version change transactions should never be scheduled in the traditional manner.
1099 RELEASE_ASSERT_NOT_REACHED();
1100 }
1101
1102 // If we didn't defer the currentTransaction above, it can be run now.
1103 if (currentTransaction)
1104 break;
1105 }
1106
1107 hadDeferredTransactions = !deferredTransactions.isEmpty();
1108 if (!hadDeferredTransactions)
1109 return currentTransaction;
1110
1111 // Prepend the deferred transactions back on the beginning of the deque for future scheduling passes.
1112 while (!deferredTransactions.isEmpty())
1113 m_pendingTransactions.prepend(deferredTransactions.takeLast());
1114
1115 return currentTransaction;
1116}

1119{

1149void UniqueIDBDatabase::immediateClose()
1150{
1151 LOG(IndexedDB, "UniqueIDBDatabase::immediateClose");
1152
1153 // Error out all transactions.
1154 // Pending transactions must be cleared before in-progress transactions,
1155 // or they may get started right away after aborting in-progress transactions.
1156 for (auto& transaction : m_pendingTransactions)
1157 transaction->databaseConnection().deleteTransaction(*transaction);
1158 m_pendingTransactions.clear();
1159
1160 for (auto& identifier : copyToVector(m_inProgressTransactions.keys()))
1161 m_inProgressTransactions.get(identifier)->abortWithoutCallback();
1162
1163 ASSERT(m_inProgressTransactions.isEmpty());
1164
1165 m_objectStoreTransactionCounts.clear();
1166 m_objectStoreWriteTransactions.clear();
1167
1168 // Error out all IDBOpenDBRequests
1169 if (m_currentOpenDBRequest) {
1170 errorOpenDBRequestForUserDelete(*m_currentOpenDBRequest);
1171 m_currentOpenDBRequest = nullptr;
1172 }
1173
1174 for (auto& request : m_pendingOpenDBRequests)
1175 errorOpenDBRequestForUserDelete(*request);
1176
1177 m_pendingOpenDBRequests.clear();
1178
1179 // Close all open connections
1180 auto openDatabaseConnections = m_openDatabaseConnections;
1181 for (auto& connection : openDatabaseConnections)
1182 connectionClosedFromServer(*connection);
1183
1184 if (m_versionChangeDatabaseConnection) {
1185 connectionClosedFromServer(*m_versionChangeDatabaseConnection);
1186 m_versionChangeDatabaseConnection = nullptr;
1187 }
1188
1189 ASSERT(!hasAnyOpenConnections());
1190
1191 close();
1192}
