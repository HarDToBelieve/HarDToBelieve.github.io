38Ref<UniqueIDBDatabaseTransaction> UniqueIDBDatabaseTransaction::create(UniqueIDBDatabaseConnection& connection, const IDBTransactionInfo& info)

43UniqueIDBDatabaseTransaction::UniqueIDBDatabaseTransaction(UniqueIDBDatabaseConnection& connection, const IDBTransactionInfo& info)
44 : m_databaseConnection(&connection)
45 , m_transactionInfo(info)
46{
47 auto database = m_databaseConnection->database();
48 ASSERT(database);
49
50 if (m_transactionInfo.mode() == IDBTransactionMode::Versionchange)
51 m_originalDatabaseInfo = makeUnique<IDBDatabaseInfo>(database->info());
52
53 m_databaseConnection->server()->registerTransaction(*this);
54}
55
56UniqueIDBDatabaseTransaction::~UniqueIDBDatabaseTransaction()
57{
58 m_databaseConnection->server()->unregisterTransaction(*this);
59}

61IDBDatabaseInfo* UniqueIDBDatabaseTransaction::originalDatabaseInfo() const
62{
63 ASSERT(m_transactionInfo.mode() == IDBTransactionMode::Versionchange);
64 return m_originalDatabaseInfo.get();

112{
113 LOG(IndexedDB, "UniqueIDBDatabaseTransaction::createObjectStore");
114

366void UniqueIDBDatabaseTransaction::iterateCursor(const IDBRequestData& requestData, const IDBIterateCursorData& data)
367{
368 LOG(IndexedDB, "UniqueIDBDatabaseTransaction::iterateCursor");
369
370 ASSERT(m_transactionInfo.identifier() == requestData.transactionIdentifier());
371
372 auto database = m_databaseConnection->database();
373 ASSERT(database);
374
375 database->iterateCursor(requestData, data, [this, requestData, option = data.option](auto& error, const IDBGetResult& result) {
376 LOG(IndexedDB, "UniqueIDBDatabaseTransaction::iterateCursor (callback)");
377
378 if (option == IndexedDB::CursorIterateOption::DoNotReply)
379 return;
380
381 if (error.isNull())
382 m_databaseConnection->connectionToClient().didIterateCursor(IDBResultData::iterateCursorSuccess(requestData.requestIdentifier(), result));
383 else
384 m_databaseConnection->connectionToClient().didIterateCursor(IDBResultData::error(requestData.requestIdentifier(), error));
385 });

388const Vector<uint64_t>& UniqueIDBDatabaseTransaction::objectStoreIdentifiers()
389{
390 if (!m_objectStoreIdentifiers.isEmpty())
391 return m_objectStoreIdentifiers;
392
393 auto& info = m_databaseConnection->database()->info();
394 for (const auto& objectStoreName : info.objectStoreNames()) {
395 auto objectStoreInfo = info.infoForExistingObjectStore(objectStoreName);
396 ASSERT(objectStoreInfo);
397 if (!objectStoreInfo)
398 continue;
399
400 if (m_transactionInfo.objectStores().contains(objectStoreName))
401 m_objectStoreIdentifiers.append(objectStoreInfo->identifier());
402 }
403
404 return m_objectStoreIdentifiers;
405}
