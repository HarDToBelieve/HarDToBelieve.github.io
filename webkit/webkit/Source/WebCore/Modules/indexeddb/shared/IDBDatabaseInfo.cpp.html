40IDBDatabaseInfo::IDBDatabaseInfo(const String& name, uint64_t version, uint64_t maxIndexID)
41 : m_name(name)
42 , m_version(version)
43 , m_maxIndexID(maxIndexID)
44{
45}
46
47IDBDatabaseInfo::IDBDatabaseInfo(const IDBDatabaseInfo& other, IsolatedCopyTag)
48 : m_name(other.m_name.isolatedCopy())
49 , m_version(other.m_version)
50 , m_maxObjectStoreID(other.m_maxObjectStoreID)
51 , m_maxIndexID(other.m_maxIndexID)
52{
53 for (const auto& entry : other.m_objectStoreMap)
54 m_objectStoreMap.set(entry.key, entry.value.isolatedCopy());
55}

57IDBDatabaseInfo IDBDatabaseInfo::isolatedCopy() const
58{
59 return { *this, IDBDatabaseInfo::IsolatedCopy };
60}
61
62bool IDBDatabaseInfo::hasObjectStore(const String& name) const
63{
64 for (auto& objectStore : m_objectStoreMap.values()) {
65 if (objectStore.name() == name)
66 return true;
67 }
68
69 return false;
70}
71
72IDBObjectStoreInfo IDBDatabaseInfo::createNewObjectStore(const String& name, std::optional<IDBKeyPath>&& keyPath, bool autoIncrement)
73{
74 IDBObjectStoreInfo info(++m_maxObjectStoreID, name, WTFMove(keyPath), autoIncrement);
75 m_objectStoreMap.set(info.identifier(), info);
76 return info;
77}
78
79void IDBDatabaseInfo::addExistingObjectStore(const IDBObjectStoreInfo& info)
80{
81 ASSERT(!m_objectStoreMap.contains(info.identifier()));
82
83 if (info.identifier() > m_maxObjectStoreID)
84 m_maxObjectStoreID = info.identifier();
85
86 m_objectStoreMap.set(info.identifier(), info);
87}
88
89IDBObjectStoreInfo* IDBDatabaseInfo::getInfoForExistingObjectStore(uint64_t objectStoreIdentifier)
90{
91 auto iterator = m_objectStoreMap.find(objectStoreIdentifier);
92 if (iterator == m_objectStoreMap.end())
93 return nullptr;
94
95 return &iterator->value;
96}
97
98IDBObjectStoreInfo* IDBDatabaseInfo::getInfoForExistingObjectStore(const String& name)
99{
100 for (auto& objectStore : m_objectStoreMap.values()) {
101 if (objectStore.name() == name)
102 return &objectStore;
103 }
104
105 return nullptr;
106}

108const IDBObjectStoreInfo* IDBDatabaseInfo::infoForExistingObjectStore(uint64_t objectStoreIdentifier) const
109{
110 return const_cast<IDBDatabaseInfo*>(this)->getInfoForExistingObjectStore(objectStoreIdentifier);
111}

113IDBObjectStoreInfo* IDBDatabaseInfo::infoForExistingObjectStore(uint64_t objectStoreIdentifier)

115 return getInfoForExistingObjectStore(objectStoreIdentifier);

118const IDBObjectStoreInfo* IDBDatabaseInfo::infoForExistingObjectStore(const String& name) const

120 return const_cast<IDBDatabaseInfo*>(this)->getInfoForExistingObjectStore(name);
121}
122
123IDBObjectStoreInfo* IDBDatabaseInfo::infoForExistingObjectStore(const String& name)
124{
125 return getInfoForExistingObjectStore(name);
126}
127
128void IDBDatabaseInfo::renameObjectStore(uint64_t objectStoreIdentifier, const String& newName)
129{
130 auto* info = infoForExistingObjectStore(objectStoreIdentifier);
131 if (!info)
132 return;
133
134 info->rename(newName);
135}
136
137Vector<String> IDBDatabaseInfo::objectStoreNames() const
138{
139 Vector<String> names;
140 names.reserveCapacity(m_objectStoreMap.size());
141 for (auto& objectStore : m_objectStoreMap.values())
142 names.uncheckedAppend(objectStore.name());
143
144 return names;
145}
146
147void IDBDatabaseInfo::deleteObjectStore(const String& objectStoreName)
148{
149 auto* info = infoForExistingObjectStore(objectStoreName);
150 if (!info)
151 return;
152
153 m_objectStoreMap.remove(info->identifier());
154}
155
156void IDBDatabaseInfo::deleteObjectStore(uint64_t objectStoreIdentifier)
157{
158 m_objectStoreMap.remove(objectStoreIdentifier);
159}

163String IDBDatabaseInfo::loggingString() const
164{
165 StringBuilder builder;
166 builder.append("Database:", m_name, " version ", m_version, '\n');
167 for (auto& objectStore : m_objectStoreMap.values())
168 builder.append(objectStore.loggingString(1), '\n');
169 return builder.toString();
170}

174void IDBDatabaseInfo::setMaxIndexID(uint64_t maxIndexID)
175{
176 ASSERT(maxIndexID > m_maxIndexID || (!maxIndexID && !m_maxIndexID));
177 m_maxIndexID = maxIndexID;
178}
