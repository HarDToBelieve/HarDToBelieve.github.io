33IDBIndexInfo::IDBIndexInfo()

37IDBIndexInfo::IDBIndexInfo(uint64_t identifier, uint64_t objectStoreIdentifier, const String& name, IDBKeyPath&& keyPath, bool unique, bool multiEntry)
38 : m_identifier(identifier)
39 , m_objectStoreIdentifier(objectStoreIdentifier)
40 , m_name(name)
41 , m_keyPath(WTFMove(keyPath))
42 , m_unique(unique)
43 , m_multiEntry(multiEntry)
44{
45}
46
47IDBIndexInfo IDBIndexInfo::isolatedCopy() const
48{
49 return { m_identifier, m_objectStoreIdentifier, m_name.isolatedCopy(), WebCore::isolatedCopy(m_keyPath), m_unique, m_multiEntry };
50}
51
52#if !LOG_DISABLED
53
54String IDBIndexInfo::loggingString(int indent) const
55{
56 String indentString;
57 for (int i = 0; i < indent; ++i)
58 indentString.append(" ");
59 return makeString(indentString, "Index: ", m_name, " (", m_identifier, ") keyPath: ", WebCore::loggingString(m_keyPath), '\n');
60}
61
62String IDBIndexInfo::condensedLoggingString() const
63{
64 return makeString("<Idx: ", m_name, " (", m_identifier, "), OS (", m_objectStoreIdentifier, ")>");
65}
