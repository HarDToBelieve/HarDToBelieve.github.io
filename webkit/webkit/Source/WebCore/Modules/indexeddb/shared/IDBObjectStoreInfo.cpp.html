32IDBObjectStoreInfo::IDBObjectStoreInfo()
33{
34}

36IDBObjectStoreInfo::IDBObjectStoreInfo(uint64_t identifier, const String& name, std::optional<IDBKeyPath>&& keyPath, bool autoIncrement)
37 : m_identifier(identifier)
38 , m_name(name)
39 , m_keyPath(WTFMove(keyPath))
40 , m_autoIncrement(autoIncrement)
41{
42}
43
44IDBIndexInfo IDBObjectStoreInfo::createNewIndex(uint64_t indexID, const String& name, IDBKeyPath&& keyPath, bool unique, bool multiEntry)
45{
46 IDBIndexInfo info(indexID, m_identifier, name, WTFMove(keyPath), unique, multiEntry);
47 m_indexMap.set(info.identifier(), info);
48 return info;
49}
50
51void IDBObjectStoreInfo::addExistingIndex(const IDBIndexInfo& info)
52{
53 if (m_indexMap.contains(info.identifier()))
54 LOG_ERROR("Adding an index '%s' with existing Index ID", info.name().utf8().data());
55
56 m_indexMap.set(info.identifier(), info);
57}
58
59bool IDBObjectStoreInfo::hasIndex(const String& name) const
60{
61 for (auto& index : m_indexMap.values()) {
62 if (index.name() == name)
63 return true;
64 }
65
66 return false;
67}
68
69bool IDBObjectStoreInfo::hasIndex(uint64_t indexIdentifier) const
70{
71 return m_indexMap.contains(indexIdentifier);
72}
73
74IDBIndexInfo* IDBObjectStoreInfo::infoForExistingIndex(const String& name)
75{
76 for (auto& index : m_indexMap.values()) {
77 if (index.name() == name)
78 return &index;
79 }
80
81 return nullptr;
82}
83
84IDBIndexInfo* IDBObjectStoreInfo::infoForExistingIndex(uint64_t identifier)
85{
86 auto iterator = m_indexMap.find(identifier);
87 if (iterator == m_indexMap.end())
88 return nullptr;
89
90 return &iterator->value;
91}
92
93IDBObjectStoreInfo IDBObjectStoreInfo::isolatedCopy() const
94{
95 IDBObjectStoreInfo result = { m_identifier, m_name.isolatedCopy(), WebCore::isolatedCopy(m_keyPath), m_autoIncrement };
96
97 for (auto& iterator : m_indexMap)
98 result.m_indexMap.set(iterator.key, iterator.value.isolatedCopy());
99
100 return result;
101}
102
103Vector<String> IDBObjectStoreInfo::indexNames() const
104{
105 Vector<String> names;
106 names.reserveCapacity(m_indexMap.size());
107 for (auto& index : m_indexMap.values())
108 names.uncheckedAppend(index.name());
109
110 return names;
111}
112
113void IDBObjectStoreInfo::deleteIndex(const String& indexName)
114{
115 auto* info = infoForExistingIndex(indexName);
116 if (!info)
117 return;
118
119 m_indexMap.remove(info->identifier());
120}
121
122void IDBObjectStoreInfo::deleteIndex(uint64_t indexIdentifier)
123{
124 m_indexMap.remove(indexIdentifier);
125}

129String IDBObjectStoreInfo::loggingString(int indent) const
130{
131 StringBuilder builder;
132 for (int i = 0; i < indent; ++i)
133 builder.append(' ');
134 builder.append("Object store: ", m_name, m_identifier);
135 for (auto index : m_indexMap.values())
136 builder.append(index.loggingString(indent + 1), '\n');
137 return builder.toString();
138}

140String IDBObjectStoreInfo::condensedLoggingString() const
