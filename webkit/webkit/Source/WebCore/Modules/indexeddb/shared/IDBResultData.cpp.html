50IDBResultData::IDBResultData(const IDBResultData& other)
51 : m_type(other.m_type)
52 , m_requestIdentifier(other.m_requestIdentifier)
53 , m_error(other.m_error)
54 , m_databaseConnectionIdentifier(other.m_databaseConnectionIdentifier)
55 , m_resultInteger(other.m_resultInteger)
56{
57 if (other.m_databaseInfo)
58 m_databaseInfo = makeUnique<IDBDatabaseInfo>(*other.m_databaseInfo);
59 if (other.m_transactionInfo)
60 m_transactionInfo = makeUnique<IDBTransactionInfo>(*other.m_transactionInfo);
61 if (other.m_resultKey)
62 m_resultKey = makeUnique<IDBKeyData>(*other.m_resultKey);
63 if (other.m_getResult)
64 m_getResult = makeUnique<IDBGetResult>(*other.m_getResult);
65 if (other.m_getAllResult)
66 m_getAllResult = makeUnique<IDBGetAllResult>(*other.m_getAllResult);
67}

79void IDBResultData::isolatedCopy(const IDBResultData& source, IDBResultData& destination)
80{
81 destination.m_type = source.m_type;
82 destination.m_requestIdentifier = source.m_requestIdentifier.isolatedCopy();
83 destination.m_error = source.m_error.isolatedCopy();
84 destination.m_databaseConnectionIdentifier = source.m_databaseConnectionIdentifier;
85 destination.m_resultInteger = source.m_resultInteger;
86
87 if (source.m_databaseInfo)
88 destination.m_databaseInfo = makeUnique<IDBDatabaseInfo>(*source.m_databaseInfo, IDBDatabaseInfo::IsolatedCopy);
89 if (source.m_transactionInfo)
90 destination.m_transactionInfo = makeUnique<IDBTransactionInfo>(*source.m_transactionInfo, IDBTransactionInfo::IsolatedCopy);
91 if (source.m_resultKey)
92 destination.m_resultKey = makeUnique<IDBKeyData>(*source.m_resultKey, IDBKeyData::IsolatedCopy);
93 if (source.m_getResult)
94 destination.m_getResult = makeUnique<IDBGetResult>(*source.m_getResult, IDBGetResult::IsolatedCopy);
95 if (source.m_getAllResult)
96 destination.m_getAllResult = makeUnique<IDBGetAllResult>(*source.m_getAllResult, IDBGetAllResult::IsolatedCopy);
97}

117IDBResultData IDBResultData::openDatabaseUpgradeNeeded(const IDBResourceIdentifier& requestIdentifier, IDBServer::UniqueIDBDatabaseTransaction& transaction)
118{
119 IDBResultData result { requestIdentifier };
