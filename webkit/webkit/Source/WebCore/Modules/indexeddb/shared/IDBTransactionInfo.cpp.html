42IDBTransactionInfo IDBTransactionInfo::clientTransaction(const IDBClient::IDBConnectionProxy& connectionProxy, const Vector<String>& objectStores, IDBTransactionMode mode)

51IDBTransactionInfo IDBTransactionInfo::versionChange(const IDBServer::IDBConnectionToClient& connection, const IDBDatabaseInfo& originalDatabaseInfo, uint64_t newVersion)
52{
53 IDBTransactionInfo result((IDBResourceIdentifier(connection)));
54 result.m_mode = IDBTransactionMode::Versionchange;
55 result.m_newVersion = newVersion;
56 result.m_originalDatabaseInfo = makeUnique<IDBDatabaseInfo>(originalDatabaseInfo);
57
58 return result;
59}

76IDBTransactionInfo IDBTransactionInfo::isolatedCopy() const
77{
78 return { *this, IsolatedCopy };
79}
80
81void IDBTransactionInfo::isolatedCopy(const IDBTransactionInfo& source, IDBTransactionInfo& destination)
82{
83 destination.m_identifier = source.m_identifier.isolatedCopy();
84 destination.m_mode = source.m_mode;
85 destination.m_newVersion = source.m_newVersion;
86
87 destination.m_objectStores.reserveCapacity(source.m_objectStores.size());
88 for (auto& objectStore : source.m_objectStores)
89 destination.m_objectStores.uncheckedAppend(objectStore.isolatedCopy());
90
91 if (source.m_originalDatabaseInfo)
92 destination.m_originalDatabaseInfo = makeUnique<IDBDatabaseInfo>(*source.m_originalDatabaseInfo, IDBDatabaseInfo::IsolatedCopy);
93}

97String IDBTransactionInfo::loggingString() const
98{
99 String modeString;
100 switch (m_mode) {
101 case IDBTransactionMode::Readonly:
102 modeString = "readonly"_s;
103 break;
104 case IDBTransactionMode::Readwrite:
105 modeString = "readwrite"_s;
106 break;
107 case IDBTransactionMode::Versionchange:
108 modeString = "versionchange"_s;
109 break;
110 default:
111 ASSERT_NOT_REACHED();
112 }
113
114 return makeString("Transaction: ", m_identifier.loggingString(), " mode ", modeString, " newVersion ", m_newVersion);
115}
