48static const MemoryCompactLookupOnlyRobinHoodHashSet<String>& bucketMIMETypes()
49{
50 // A "bucket" MIME types is one whose container type does not uniquely specify a codec.
51 // See: https://tools.ietf.org/html/rfc6381
52 static NeverDestroyed<MemoryCompactLookupOnlyRobinHoodHashSet<String>> bucketMIMETypes(std::initializer_list<String> {
53 "audio/3gpp"_s,
54 "video/3gpp"_s,
55 "audio/3gpp2"_s,
56 "video/3gpp2"_s,
57 "audio/mp4"_s,
58 "video/mp4"_s,
59 "application/mp4"_s,
60 "video/quicktime"_s,
61 "application/mp21"_s,
62 "audio/vnd.apple.mpegurl"_s,
63 "video/vnd.apple.mpegurl"_s,
64 "audio/ogg"_s,
65 "video/ogg"_s,
66 "video/webm"_s,
67 "audio/webm"_s,
68 });
69 return bucketMIMETypes;
70}

72static bool isValidMediaMIMEType(const ContentType& contentType)
73{
74 // 2.1.4. MIME types
75 // https://wicg.github.io/media-capabilities/#valid-media-mime-type
76 // A valid media MIME type is a string that is a valid MIME type per [mimesniff]. If the MIME type does
77 // not imply a codec, the string MUST also have one and only one parameter that is named codecs with a
78 // value describing a single media codec. Otherwise, it MUST contain no parameters.
79 if (contentType.isEmpty())
80 return false;
81
82 auto codecs = contentType.codecs();
83
84 // FIXME: The spec requires that the "codecs" parameter is the only parameter present.
85 if (bucketMIMETypes().contains(contentType.containerType()))
86 return codecs.size() == 1;
87 return !codecs.size();
88}

90static bool isValidVideoMIMEType(const ContentType& contentType)
91{
92 // 2.1.4 MIME Types
93 // https://wicg.github.io/media-capabilities/#valid-video-mime-type
94 // A valid video MIME type is a string that is a valid media MIME type and for which the type per [RFC7231]
95 // is either video or application.
96 if (!isValidMediaMIMEType(contentType))
97 return false;
98
99 auto containerType = contentType.containerType();
100 if (!startsWithLettersIgnoringASCIICase(containerType, "video/") && !startsWithLettersIgnoringASCIICase(containerType, "application/"))
101 return false;
102
103 return true;
104}

122static bool isValidVideoConfiguration(const VideoConfiguration& configuration)
123{
124 // 2.1.5. VideoConfiguration
125 // https://wicg.github.io/media-capabilities/#valid-video-configuration
126 // 1. If configuration’s contentType is not a valid video MIME type, return false and abort these steps.
127 if (!isValidVideoMIMEType(ContentType(configuration.contentType)))
128 return false;
129
130 // 2. If none of the following is true, return false and abort these steps:
131 // o. Applying the rules for parsing floating-point number values to configuration’s framerate
132 // results in a number that is finite and greater than 0.
133 if (!std::isfinite(configuration.framerate) || configuration.framerate <= 0)
134 return false;
135
136 // 3. Return true.
137 return true;
138}

140static bool isValidAudioConfiguration(const AudioConfiguration& configuration)
141{
142 // 2.1.6. AudioConfiguration
143 // https://wicg.github.io/media-capabilities/#audioconfiguration
144 // 1. If configuration’s contentType is not a valid audio MIME type, return false and abort these steps.
145 if (!isValidAudioMIMEType(ContentType(configuration.contentType)))
146 return false;
147
148 // 2. Return true.
149 return true;
150}

152static bool isValidMediaConfiguration(const MediaConfiguration& configuration)
153{
154 // 2.1.1. MediaConfiguration
155 // https://wicg.github.io/media-capabilities/#mediaconfiguration
156 // For a MediaConfiguration to be a valid MediaConfiguration, audio or video MUST be present.
157 if (!configuration.video && !configuration.audio)
158 return false;
159
160 if (configuration.video && !isValidVideoConfiguration(configuration.video.value()))
161 return false;
162
163 if (configuration.audio && !isValidAudioConfiguration(configuration.audio.value()))
164 return false;
165
166 return true;
167}

169void MediaCapabilities::decodingInfo(Document& document, MediaDecodingConfiguration&& configuration, Ref<DeferredPromise>&& promise)
170{
171 // 2.4 Media Capabilities Interface
172 // https://wicg.github.io/media-capabilities/#media-capabilities-interface
173
174 auto identifier = WTF::Logger::LogSiteIdentifier("MediaCapabilities", __func__, this);
175 Ref<Logger> logger = document.logger();
176
177 // 1. If configuration is not a valid MediaConfiguration, return a Promise rejected with a TypeError.
178 // 2. If configuration.video is present and is not a valid video configuration, return a Promise rejected with a TypeError.
179 // 3. If configuration.audio is present and is not a valid audio configuration, return a Promise rejected with a TypeError.
180 if (!isValidMediaConfiguration(configuration)) {
181#if !RELEASE_LOG_DISABLED
182 logger->info(LogMedia, identifier, " - Rejected. configuration: ", configuration);
183#endif
184 promise->reject(TypeError);
185 return;
186 }
187
188 if (!document.settings().mediaCapabilitiesExtensionsEnabled() && configuration.video)
189 configuration.video.value().alphaChannel.reset();
190
191#if ENABLE(VP9)
192 configuration.canExposeVP9 = document.settings().vp9DecoderEnabled();
193#endif
194
195 // 4. Let p be a new promise.
196 // 5. In parallel, run the create a MediaCapabilitiesInfo algorithm with configuration and resolve p with its result.
197 // 6. Return p.
198 document.eventLoop().queueTask(TaskSource::MediaElement, [configuration = WTFMove(configuration), promise = WTFMove(promise), logger = WTFMove(logger), identifier = WTFMove(identifier)] () mutable {
199
200 // 2.2.3 If configuration is of type MediaDecodingConfiguration, run the following substeps:
201 MediaEngineConfigurationFactory::DecodingConfigurationCallback callback = [promise = WTFMove(promise), logger = WTFMove(logger), identifier = WTFMove(identifier)] (auto info) mutable {
202 // 2.2.3.1. If the user agent is able to decode the media represented by
203 // configuration, set supported to true. Otherwise set it to false.
204 // 2.2.3.2. If the user agent is able to decode the media represented by
205 // configuration at a pace that allows a smooth playback, set smooth to
206 // true. Otherwise set it to false.
207 // 2.2.3.3. If the user agent is able to decode the media represented by
208 // configuration in a power efficient manner, set powerEfficient to
209 // true. Otherwise set it to false. The user agent SHOULD NOT take into
210 // consideration the current power source in order to determine the
211 // decoding power efficiency unless the device’s power source has side
212 // effects such as enabling different decoding modules.
213#if !RELEASE_LOG_DISABLED
214 logger->info(LogMedia, identifier, "::callback() - Resolved. info: ", info);
215#endif
216 promise->resolve<IDLDictionary<MediaCapabilitiesDecodingInfo>>(WTFMove(info));
217 };
218
219 MediaEngineConfigurationFactory::createDecodingConfiguration(WTFMove(configuration), WTFMove(callback));
220 });
221}

223void MediaCapabilities::encodingInfo(Document& document, MediaEncodingConfiguration&& configuration, Ref<DeferredPromise>&& promise)
224{
225 // 2.4 Media Capabilities Interface
226 // https://wicg.github.io/media-capabilities/#media-capabilities-interface
227
228 // 1. If configuration is not a valid MediaConfiguration, return a Promise rejected with a TypeError.
229 // 2. If configuration.video is present and is not a valid video configuration, return a Promise rejected with a TypeError.
230 // 3. If configuration.audio is present and is not a valid audio configuration, return a Promise rejected with a TypeError.
231 if (!isValidMediaConfiguration(configuration)) {
232 promise->reject(TypeError);
233 return;
234 }
235
236 // 4. Let p be a new promise.
237 // 5. In parallel, run the create a MediaCapabilitiesInfo algorithm with configuration and resolve p with its result.
238 // 6. Return p.
239 document.eventLoop().queueTask(TaskSource::MediaElement, [configuration = WTFMove(configuration), promise = WTFMove(promise)] () mutable {
240
241 // 2.2.4. If configuration is of type MediaEncodingConfiguration, run the following substeps:
242 MediaEngineConfigurationFactory::EncodingConfigurationCallback callback = [promise = WTFMove(promise)] (auto info) mutable {
243 // 2.2.4.1. If the user agent is able to encode the media
244 // represented by configuration, set supported to true. Otherwise
245 // set it to false.
246 // 2.2.4.2. If the user agent is able to encode the media
247 // represented by configuration at a pace that allows encoding
248 // frames at the same pace as they are sent to the encoder, set
249 // smooth to true. Otherwise set it to false.
250 // 2.2.4.3. If the user agent is able to encode the media
251 // represented by configuration in a power efficient manner, set
252 // powerEfficient to true. Otherwise set it to false. The user agent
253 // SHOULD NOT take into consideration the current power source in
254 // order to determine the encoding power efficiency unless the
255 // device’s power source has side effects such as enabling different
256 // encoding modules.
257 promise->resolve<IDLDictionary<MediaCapabilitiesEncodingInfo>>(WTFMove(info));
258 };
259
260 MediaEngineConfigurationFactory::createEncodingConfiguration(WTFMove(configuration), WTFMove(callback));
261
262 });
263}
