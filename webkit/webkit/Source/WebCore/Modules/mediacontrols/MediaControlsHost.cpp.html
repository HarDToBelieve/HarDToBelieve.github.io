184AtomString MediaControlsHost::captionDisplayMode() const
185{
186 if (!m_mediaElement)
187 return emptyAtom();
188
189 Page* page = m_mediaElement->document().page();
190 if (!page)
191 return emptyAtom();
192
193 switch (page->group().captionPreferences().captionDisplayMode()) {
194 case CaptionUserPreferences::Automatic:
195 return automaticKeyword();
196 case CaptionUserPreferences::ForcedOnly:
197 return forcedOnlyKeyword();
198 case CaptionUserPreferences::AlwaysOn:
199 return alwaysOnKeyword();
200 case CaptionUserPreferences::Manual:
201 return manualKeyword();
202 default:
203 ASSERT_NOT_REACHED();
204 return emptyAtom();
205 }
206}

302auto MediaControlsHost::externalDeviceType() const -> DeviceType
303{
304#if !ENABLE(WIRELESS_PLAYBACK_TARGET)
305 return DeviceType::None;
306#else
307 if (!m_mediaElement)
308 return DeviceType::None;
309
310 auto player = m_mediaElement->player();
311 if (!player) {
312 LOG(Media, "MediaControlsHost::externalDeviceType - returning \"none\" because player is NULL");
313 return DeviceType::None;
314 }
315
316 switch (player->wirelessPlaybackTargetType()) {
317 case MediaPlayer::WirelessPlaybackTargetType::TargetTypeNone:
318 return DeviceType::None;
319 case MediaPlayer::WirelessPlaybackTargetType::TargetTypeAirPlay:
320 return DeviceType::Airplay;
321 case MediaPlayer::WirelessPlaybackTargetType::TargetTypeTVOut:
322 return DeviceType::Tvout;
323 }
324
325 ASSERT_NOT_REACHED();
326 return DeviceType::None;
327#endif
328}

379
380 ~MediaControlsContextMenuProvider() override

383 }
384

392 {
393 m_callback(action - ContextMenuItemBaseCustomTag);
394 }
395

397 {
398 if (m_callback)
399 m_callback(ContextMenuItemTagNoAction);
400 m_items.clear();
401 }

406 }

408 Vector<ContextMenuItem> m_items;
409 CompletionHandler<void(uint64_t)> m_callback;
410};

412class MediaControlsContextMenuEventListener final : public EventListener {
413public:
414 static Ref<MediaControlsContextMenuEventListener> create(Ref<MediaControlsContextMenuProvider>&& contextMenuProvider)
415 {
416 return adoptRef(*new MediaControlsContextMenuEventListener(WTFMove(contextMenuProvider)));
417 }
418
419 bool operator==(const EventListener& other) const override
420 {
421 return this == &other;
422 }
423
424 void handleEvent(ScriptExecutionContext&, Event& event) override
425 {
426 ASSERT(event.type() == eventNames().contextmenuEvent);
427
428 auto* target = event.target();
429 if (!is<Node>(target))
430 return;
431 auto& node = downcast<Node>(*target);
432
433 auto* page = node.document().page();
434 if (!page)
435 return;
436
437 page->contextMenuController().showContextMenu(event, m_contextMenuProvider);
438 event.preventDefault();
439 event.stopPropagation();
440 event.stopImmediatePropagation();
441 }
442
443private:
444 MediaControlsContextMenuEventListener(Ref<MediaControlsContextMenuProvider>&& contextMenuProvider)
445 : EventListener(EventListener::CPPEventListenerType)
446 , m_contextMenuProvider(WTFMove(contextMenuProvider))
447 {
448 }
449
450 Ref<MediaControlsContextMenuProvider> m_contextMenuProvider;
451};

454
455bool MediaControlsHost::showMediaControlsContextMenu(HTMLElement& target, String&& optionsJSONString, Ref<VoidCallback>&& callback)
456{
457#if USE(UICONTEXTMENU) || (ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS))
458 if (m_showMediaControlsContextMenuCallback) {
459#if USE(UICONTEXTMENU)
460 return false;
461#elif (ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS))
462 // FIXME: `contextMenuCleared` is invoked between show and item selected so we may have a pending callback.
463 std::exchange(m_showMediaControlsContextMenuCallback, nullptr)->handleEvent();
464#endif
465 }
466
467 m_showMediaControlsContextMenuCallback = WTFMove(callback);
468
469 auto invokeCallbackAtScopeExit = makeScopeExit([&, protectedThis = makeRef(*this)] {
470 if (m_showMediaControlsContextMenuCallback)
471 std::exchange(m_showMediaControlsContextMenuCallback, nullptr)->handleEvent();
472 });
473
474 if (!m_mediaElement)
475 return false;
476
477 auto& mediaElement = *m_mediaElement;
478
479 auto* page = mediaElement.document().page();
480 if (!page)
481 return false;
482
483 auto optionsJSON = JSON::Value::parseJSON(optionsJSONString);
484 if (!optionsJSON)
485 return false;
486
487 auto optionsJSONObject = optionsJSON->asObject();
488 if (!optionsJSONObject)
489 return false;
490
491#if USE(UICONTEXTMENU)
492 using MenuItem = MediaControlsContextMenuItem;
493 using MenuItemIdentifier = MediaControlsContextMenuItem::ID;
494 constexpr auto invalidMenuItemIdentifier = MediaControlsContextMenuItem::invalidID;
495#elif ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
496 using MenuItem = ContextMenuItem;
497 using MenuItemIdentifier = uint64_t;
498 constexpr auto invalidMenuItemIdentifier = ContextMenuItemTagNoAction;
499#endif
500
501#if ENABLE(VIDEO_PRESENTATION_MODE)
502 enum class PictureInPictureTag { IncludePictureInPicture };
503#endif // ENABLE(VIDEO_PRESENTATION_MODE)
504
505 enum class PlaybackSpeed {
506 x0_5,
507 x1_0,
508 x1_25,
509 x1_5,
510 x2_0,
511 };
512
513 using MenuData = Variant<
514#if ENABLE(VIDEO_PRESENTATION_MODE)
515 PictureInPictureTag,
516#endif // ENABLE(VIDEO_PRESENTATION_MODE)
517 RefPtr<AudioTrack>,
518 RefPtr<TextTrack>,
519 RefPtr<VTTCue>,
520 PlaybackSpeed
521 >;
522 HashMap<MenuItemIdentifier, MenuData> idMap;
523
524 auto createSubmenu = [] (const String& title, const String& icon, Vector<MenuItem>&& children) -> MenuItem {
525#if USE(UICONTEXTMENU)
526 return { MediaControlsContextMenuItem::invalidID, title, icon, /* checked */ false, WTFMove(children) };
527#elif ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
528 UNUSED_PARAM(icon);
529 return { ContextMenuItemTagNoAction, title, /* enabled */ true, /* checked */ false, WTFMove(children) };
530#endif
531 };
532
533 auto createMenuItem = [&] (MenuData data, const String& title, bool checked = false, const String& icon = nullString()) -> MenuItem {
534 auto id = idMap.size() + 1;
535 idMap.add(id, data);
536
537#if USE(UICONTEXTMENU)
538 return { id, title, icon, checked, /* children */ { } };
539#elif ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
540 UNUSED_PARAM(icon);
541 return { CheckableActionType, static_cast<ContextMenuAction>(ContextMenuItemBaseCustomTag + id), title, /* enabled */ true, checked };
542#endif
543 };
544
545 Vector<MenuItem> items;
546
547#if ENABLE(VIDEO_PRESENTATION_MODE)
548 if (optionsJSONObject->getBoolean("includePictureInPicture"_s).value_or(false)) {
549 ASSERT(is<HTMLVideoElement>(mediaElement));
550 ASSERT(downcast<HTMLVideoElement>(mediaElement).webkitSupportsPresentationMode(HTMLVideoElement::VideoPresentationMode::PictureInPicture));
551 items.append(createMenuItem(PictureInPictureTag::IncludePictureInPicture, WEB_UI_STRING_KEY("Picture in Picture", "Picture in Picture (Media Controls Menu)", "Picture in Picture media controls context menu title"), false, "pip.enter"_s));
552 }
553#endif // ENABLE(VIDEO_PRESENTATION_MODE)
554
555 if (optionsJSONObject->getBoolean("includeLanguages"_s).value_or(false)) {
556 if (auto* audioTracks = mediaElement.audioTracks(); audioTracks && audioTracks->length() > 1) {
557 Vector<MenuItem> languageMenuItems;
558
559 auto& captionPreferences = page->group().captionPreferences();
560 for (auto& audioTrack : captionPreferences.sortedTrackListForMenu(audioTracks))
561 languageMenuItems.append(createMenuItem(audioTrack, captionPreferences.displayNameForTrack(audioTrack.get()), audioTrack->enabled()));
562
563 if (!languageMenuItems.isEmpty())
564 items.append(createSubmenu(WEB_UI_STRING_KEY("Languages", "Languages (Media Controls Menu)", "Languages media controls context menu title"), "globe"_s, WTFMove(languageMenuItems)));
565 }
566 }
567
568 if (optionsJSONObject->getBoolean("includeSubtitles"_s).value_or(false)) {
569 if (auto* textTracks = mediaElement.textTracks(); textTracks && textTracks->length()) {
570 Vector<MenuItem> subtitleMenuItems;
571
572 auto& captionPreferences = page->group().captionPreferences();
573 auto sortedTextTracks = captionPreferences.sortedTrackListForMenu(textTracks, { TextTrack::Kind::Subtitles, TextTrack::Kind::Captions, TextTrack::Kind::Descriptions });
574 bool allTracksDisabled = notFound == sortedTextTracks.findMatching([] (const auto& textTrack) {
575 return textTrack->mode() == TextTrack::Mode::Showing;
576 });
577 bool usesAutomaticTrack = captionPreferences.captionDisplayMode() == CaptionUserPreferences::Automatic && allTracksDisabled;
578 for (auto& textTrack : sortedTextTracks) {
579 bool checked = false;
580 if (allTracksDisabled && textTrack == &TextTrack::captionMenuOffItem() && (captionPreferences.captionDisplayMode() == CaptionUserPreferences::ForcedOnly || captionPreferences.captionDisplayMode() == CaptionUserPreferences::Manual))
581 checked = true;
582 else if (usesAutomaticTrack && textTrack == &TextTrack::captionMenuAutomaticItem())
583 checked = true;
584 else if (!usesAutomaticTrack && textTrack->mode() == TextTrack::Mode::Showing)
585 checked = true;
586 subtitleMenuItems.append(createMenuItem(textTrack, captionPreferences.displayNameForTrack(textTrack.get()), checked));
587 }
588
589 if (!subtitleMenuItems.isEmpty())
590 items.append(createSubmenu(WEB_UI_STRING_KEY("Subtitles", "Subtitles (Media Controls Menu)", "Subtitles media controls context menu title"), "captions.bubble"_s, WTFMove(subtitleMenuItems)));
591 }
592 }
593
594 if (optionsJSONObject->getBoolean("includeChapters"_s).value_or(false)) {
595 if (auto* textTracks = mediaElement.textTracks(); textTracks && textTracks->length()) {
596 auto& captionPreferences = page->group().captionPreferences();
597
598 for (auto& textTrack : captionPreferences.sortedTrackListForMenu(textTracks, { TextTrack::Kind::Chapters })) {
599 Vector<MenuItem> chapterMenuItems;
600
601 if (auto* cues = textTrack->cues()) {
602 for (unsigned i = 0; i < cues->length(); ++i) {
603 auto* cue = cues->item(i);
604 if (!is<VTTCue>(cue))
605 continue;
606
607 auto& vttCue = downcast<VTTCue>(*cue);
608 chapterMenuItems.append(createMenuItem(makeRefPtr(vttCue), vttCue.text()));
609 }
610 }
611
612 if (!chapterMenuItems.isEmpty()) {
613 items.append(createSubmenu(captionPreferences.displayNameForTrack(textTrack.get()), "list.bullet"_s, WTFMove(chapterMenuItems)));
614
615 /* Only show the first valid chapters track. */
616 break;
617 }
618 }
619 }
620 }
621
622 if (optionsJSONObject->getBoolean("includePlaybackRates"_s).value_or(false)) {
623 auto playbackRate = mediaElement.playbackRate();
624
625 items.append(createSubmenu(WEB_UI_STRING_KEY("Playback Speed", "Playback Speed (Media Controls Menu)", "Playback Speed media controls context menu title"), "speedometer"_s, {
626 createMenuItem(PlaybackSpeed::x0_5, WEB_UI_STRING_KEY("0.5", "0.5 (Media Controls Menu Playback Speed)", "0.5 media controls context menu playback speed label"), playbackRate == 0.5),
627 createMenuItem(PlaybackSpeed::x1_0, WEB_UI_STRING_KEY("1", "1 (Media Controls Menu Playback Speed)", "1 media controls context menu playback speed label"), playbackRate == 1.0),
628 createMenuItem(PlaybackSpeed::x1_25, WEB_UI_STRING_KEY("1.25", "1.25 (Media Controls Menu Playback Speed)", "1.25 media controls context menu playback speed label"), playbackRate == 1.25),
629 createMenuItem(PlaybackSpeed::x1_5, WEB_UI_STRING_KEY("1.5", "1.5 (Media Controls Menu Playback Speed)", "1.5 media controls context menu playback speed label"), playbackRate == 1.5),
630 createMenuItem(PlaybackSpeed::x2_0, WEB_UI_STRING_KEY("2", "2 (Media Controls Menu Playback Speed)", "2 media controls context menu playback speed label"), playbackRate == 2.0),
631 }));
632 }
633
634#if ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
635 if ((items.size() == 1 && items[0].type() == SubmenuType) || optionsJSONObject->getBoolean("promoteSubMenus"_s).value_or(false)) {
636 for (auto&& item : std::exchange(items, { })) {
637 if (!items.isEmpty())
638 items.append({ SeparatorType, invalidMenuItemIdentifier, /* title */ nullString() });
639
640 ASSERT(item.type() == SubmenuType);
641 items.append({ ActionType, invalidMenuItemIdentifier, item.title(), /* enabled */ false, /* checked */ false });
642 items.appendVector(WTF::map(item.subMenuItems(), [] (const auto& item) -> ContextMenuItem {
643 // The disabled inline item used instead of an actual submenu should be indented less than the submenu items.
644 constexpr unsigned indentationLevel = 1;
645 if (item.type() == SubmenuType)
646 return { item.action(), item.title(), item.enabled(), item.checked(), item.subMenuItems(), indentationLevel };
647 return { item.type(), item.action(), item.title(), item.enabled(), item.checked(), indentationLevel };
648 }));
649 }
650 }
651#endif // ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
652
653 if (items.isEmpty())
654 return false;
655
656 ASSERT(!idMap.isEmpty());
657
658 auto handleItemSelected = [weakMediaElement = makeWeakPtr(mediaElement), idMap = WTFMove(idMap), invokeCallbackAtScopeExit = WTFMove(invokeCallbackAtScopeExit)] (MenuItemIdentifier selectedItemID) {
659 if (selectedItemID == invalidMenuItemIdentifier)
660 return;
661
662 if (!weakMediaElement)
663 return;
664
665 auto& mediaElement = *weakMediaElement;
666
667 UserGestureIndicator gestureIndicator(ProcessingUserGesture, &mediaElement.document());
668
669 auto selectedItem = idMap.get(selectedItemID);
670 WTF::switchOn(selectedItem,
671#if ENABLE(VIDEO_PRESENTATION_MODE)
672 [&] (PictureInPictureTag) {
673 // Media controls are not shown when in PiP so we can assume that we're not in PiP.
674 downcast<HTMLVideoElement>(mediaElement).webkitSetPresentationMode(HTMLVideoElement::VideoPresentationMode::PictureInPicture);
675 },
676#endif // ENABLE(VIDEO_PRESENTATION_MODE)
677 [&] (RefPtr<AudioTrack>& selectedAudioTrack) {
678 for (auto& track : idMap.values()) {
679 if (auto* audioTrack = WTF::get_if<RefPtr<AudioTrack>>(track))
680 (*audioTrack)->setEnabled(*audioTrack == selectedAudioTrack);
681 }
682 },
683 [&] (RefPtr<TextTrack>& selectedTextTrack) {
684 for (auto& track : idMap.values()) {
