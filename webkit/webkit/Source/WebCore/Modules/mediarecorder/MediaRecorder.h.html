29#include "ActiveDOMObject.h"
30#include "EventTarget.h"
31#include "ExceptionOr.h"
32#include "MediaRecorderPrivateOptions.h"
33#include "MediaStream.h"

36#include <wtf/Deque.h>
37#include <wtf/UniqueRef.h>
38
39namespace WebCore {
40
41class Blob;
42class Document;
43class MediaRecorderPrivate;
44
45class MediaRecorder final
46 : public ActiveDOMObject
47 , public RefCounted<MediaRecorder>
48 , public EventTargetWithInlineData
49 , private MediaStreamPrivate::Observer
50 , private MediaStreamTrackPrivate::Observer {
51 WTF_MAKE_ISO_ALLOCATED(MediaRecorder);
52public:
53 enum class RecordingState { Inactive, Recording, Paused };
54
55 ~MediaRecorder();
56
57 static bool isTypeSupported(Document&, const String&);
58
59 using Options = MediaRecorderPrivateOptions;
60 static ExceptionOr<Ref<MediaRecorder>> create(Document&, Ref<MediaStream>&&, Options&& = { });
61
62 using CreatorFunction = ExceptionOr<std::unique_ptr<MediaRecorderPrivate>> (*)(MediaStreamPrivate&, const Options&);
63
64 WEBCORE_EXPORT static void setCustomPrivateRecorderCreator(CreatorFunction);
65
66 RecordingState state() const { return m_state; }
67 const String& mimeType() const { return m_options.mimeType; }
68
69 using RefCounted::ref;
70 using RefCounted::deref;
71
72 ExceptionOr<void> startRecording(std::optional<unsigned>);
73 void stopRecording();
74 ExceptionOr<void> requestData();
75 ExceptionOr<void> pauseRecording();
76 ExceptionOr<void> resumeRecording();
77
78 unsigned videoBitsPerSecond() const { return m_options.videoBitsPerSecond.value_or(0); }
79 unsigned audioBitsPerSecond() const { return m_options.audioBitsPerSecond.value_or(0); }
80
81 MediaStream& stream() { return m_stream.get(); }
82
83private:
84 MediaRecorder(Document&, Ref<MediaStream>&&, Options&&);
85
86 static ExceptionOr<std::unique_ptr<MediaRecorderPrivate>> createMediaRecorderPrivate(Document&, MediaStreamPrivate&, const Options&);
87
88 Document* document() const;
89
90 // EventTarget
91 void refEventTarget() final { ref(); }
92 void derefEventTarget() final { deref(); }
93 EventTargetInterface eventTargetInterface() const final { return MediaRecorderEventTargetInterfaceType; }
94 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
95
96 // ActiveDOMObject API.
97 void suspend(ReasonForSuspension) final;
98 void stop() final;
99 const char* activeDOMObjectName() const final;
100 bool virtualHasPendingActivity() const final;
101
102 void stopRecordingInternal(CompletionHandler<void()>&& = [] { });
103 void dispatchError(Exception&&);
104
105 enum class TakePrivateRecorder { No, Yes };
106 using FetchDataCallback = Function<void(RefPtr<SharedBuffer>&&, const String& mimeType, double)>;
107 void fetchData(FetchDataCallback&&, TakePrivateRecorder);
108
109 // MediaStream::Observer
110 void didAddTrack(MediaStreamTrackPrivate&) final { handleTrackChange(); }
111 void didRemoveTrack(MediaStreamTrackPrivate&) final { handleTrackChange(); }
112
113 void handleTrackChange();
114
115 // MediaStreamTrackPrivate::Observer
116 void trackEnded(MediaStreamTrackPrivate&) final;
117 void trackMutedChanged(MediaStreamTrackPrivate&) final;
118 void trackEnabledChanged(MediaStreamTrackPrivate&) final;
119 void trackSettingsChanged(MediaStreamTrackPrivate&) final { };
120
121 static CreatorFunction m_customCreator;
122
123 Options m_options;
124 Ref<MediaStream> m_stream;
125 std::unique_ptr<MediaRecorderPrivate> m_private;
126 RecordingState m_state { RecordingState::Inactive };
127 Vector<Ref<MediaStreamTrackPrivate>> m_tracks;
128 std::optional<unsigned> m_timeSlice;
129 Timer m_timeSliceTimer;
130
131 bool m_isActive { true };
132 bool m_isFetchingData { false };
133 Deque<FetchDataCallback> m_pendingFetchDataTasks;
134};
135
136} // namespace WebCore

138#endif // ENABLE(MEDIA_STREAM)
