38Ref<MediaRecorderErrorEvent> MediaRecorderErrorEvent::create(const AtomString& type, Exception&& exception)
39{
40 return adoptRef(*new MediaRecorderErrorEvent(type, WTFMove(exception)));
41}
42
43Ref<MediaRecorderErrorEvent> MediaRecorderErrorEvent::create(const AtomString& type, Init&& init, IsTrusted isTrusted)
44{
45 auto domError = init.error.releaseNonNull();
46 return adoptRef(*new MediaRecorderErrorEvent(type, WTFMove(init), WTFMove(domError), isTrusted));
47}
48
49MediaRecorderErrorEvent::MediaRecorderErrorEvent(const AtomString& type, Init&& init, Ref<DOMException>&& exception, IsTrusted isTrusted)
50 : Event(type, WTFMove(init), isTrusted)
51 , m_domError(WTFMove(exception))
52{
53}
54
55MediaRecorderErrorEvent::MediaRecorderErrorEvent(const AtomString& type, Exception&& exception)
56 : Event(type, Event::CanBubble::No, Event::IsCancelable::No)
57 , m_domError(DOMException::create(exception))
58{
59}

61EventInterface MediaRecorderErrorEvent::eventInterface() const

63 return MediaRecorderErrorEventInterfaceType;
64}
