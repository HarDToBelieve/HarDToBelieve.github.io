58void ArtworkImageLoader::requestImageResource()
59{
60 ASSERT(!m_cachedImage, "Can only call requestImageResource once");
61 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
62 options.contentSecurityPolicyImposition = m_document.isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
63
64 CachedResourceRequest request(ResourceRequest(m_document.completeURL(m_src)), options);
65 request.setInitiator(m_document.documentURI());
66 m_cachedImage = m_document.cachedResourceLoader().requestImage(WTFMove(request)).value_or(nullptr);
67
68 if (m_cachedImage)
69 m_cachedImage->addClient(*this);
70}

72void ArtworkImageLoader::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
73{
74 ASSERT_UNUSED(resource, &resource == m_cachedImage);
75 if (m_cachedImage->loadFailedOrCanceled() || m_cachedImage->errorOccurred() || !m_cachedImage->image() || !m_cachedImage->image()->data() || m_cachedImage->image()->data()->isEmpty()) {
76 m_callback(nullptr);
77 return;
78 }
79 // Sanitize the image by decoding it into a BitmapImage.
80 RefPtr<SharedBuffer> bufferToSanitize = m_cachedImage->image()->data();
81 auto bitmapImage = BitmapImage::create();
82 bitmapImage->setData(WTFMove(bufferToSanitize), true);
83 auto imageBuffer = ImageBuffer::create(bitmapImage->size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
84 if (!imageBuffer) {
85 m_callback(nullptr);
86 return;
87 }
88 imageBuffer->context().drawImage(bitmapImage.get(), FloatPoint::zero());
89 m_callback(bitmapImage.ptr());
90}

92ExceptionOr<Ref<MediaMetadata>> MediaMetadata::create(ScriptExecutionContext& context, std::optional<MediaMetadataInit>&& init)
93{
94 auto metadata = adoptRef(*new MediaMetadata);
95 if (init) {
96 metadata->setTitle(init->title);
97 metadata->setArtist(init->artist);
98 metadata->setAlbum(init->album);
99#if ENABLE(MEDIA_SESSION_PLAYLIST)
100 metadata->setTrackIdentifier(init->trackIdentifier);
101#endif
102 auto possibleException = metadata->setArtwork(context, WTFMove(init->artwork));
103 if (possibleException.hasException())
104 return Exception { possibleException.exception() };
105 }
106 return metadata;
107}

152ExceptionOr<void> MediaMetadata::setArtwork(ScriptExecutionContext& context, Vector<MediaImage>&& artwork)
153{
154 Vector<MediaImage> resolvedArtwork;
155 resolvedArtwork.reserveInitialCapacity(artwork.size());
156 for (auto& image : artwork) {
157 auto resolvedSrc = context.completeURL(image.src);
158 if (!resolvedSrc.isValid())
159 return Exception { TypeError };
160 resolvedArtwork.uncheckedAppend(MediaImage { resolvedSrc.string(), image.sizes, image.type });
161 }
162
163 m_metadata.artwork = WTFMove(resolvedArtwork);
164
165 refreshArtworkImage();
166
167 metadataUpdated();
168 return { };
169}

171void MediaMetadata::refreshArtworkImage()
172{
173 const auto& mediaImages = m_metadata.artwork;
174 if (mediaImages.isEmpty()) {
175 m_artworkLoader = nullptr;
176 return;
177 }
178 if (!m_session || !m_session->document() || mediaImages[0].src == m_artworkImageSrc)
179 return;
180 // FIXME: Implement a heuristic to retrieve the "best" image.
181 m_artworkImageSrc = mediaImages[0].src;
182 m_artworkLoader = makeUnique<ArtworkImageLoader>(*m_session->document(), m_artworkImageSrc, [this](Image* image) {
183 if (!image || !image->data())
184 return;
185 setArtworkImage(image);
186 metadataUpdated();
187 });
188 m_artworkLoader->requestImageResource();
189}
