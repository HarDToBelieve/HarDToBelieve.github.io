38namespace WebCore {
39
40class CachedImage;
41class Document;
42class Image;
43struct MediaImage;
44
45using MediaSessionMetadata = MediaMetadataInit;
46
47class ArtworkImageLoader final : public CachedImageClient {
48 WTF_MAKE_FAST_ALLOCATED;
49public:
50 using ArtworkImageLoaderCallback = Function<void(Image*)>;
51 // The callback will only be called upon success or explicit failure to retrieve the image. If the operation is interrupted following the
52 // destruction of the ArtworkImageLoader, the callback won't be called.
53 WEBCORE_EXPORT ArtworkImageLoader(Document&, const String& src, ArtworkImageLoaderCallback&&);
54 WEBCORE_EXPORT ~ArtworkImageLoader();
55
56 WEBCORE_EXPORT void requestImageResource();
57
58protected:
59 void notifyFinished(CachedResource&, const NetworkLoadMetrics&) override;
60
61private:
62 Document& m_document;
63 const String m_src;
64 ArtworkImageLoaderCallback m_callback;
65 CachedResourceHandle<CachedImage> m_cachedImage;
66};
67
68class MediaMetadata : public RefCounted<MediaMetadata> {
69public:
70 static ExceptionOr<Ref<MediaMetadata>> create(ScriptExecutionContext&, std::optional<MediaMetadataInit>&&);
71 ~MediaMetadata();
72
73 void setMediaSession(MediaSession&);
74 void resetMediaSession();
75
76 const String& title() const { return m_metadata.title; }
77 void setTitle(const String&);
78
79 const String& artist() const { return m_metadata.artist; }
80 void setArtist(const String&);
81
82 const String& album() const { return m_metadata.album; }
83 void setAlbum(const String&);
84
85 const Vector<MediaImage>& artwork() const { return m_metadata.artwork; }
86 ExceptionOr<void> setArtwork(ScriptExecutionContext&, Vector<MediaImage>&&);
87
88 const String& artworkSrc() const { return m_artworkImageSrc; }
89 const RefPtr<Image>& artworkImage() const { return m_artworkImage; }
90
91 const MediaSessionMetadata& metadata() const { return m_metadata; }
92
93#if ENABLE(MEDIA_SESSION_PLAYLIST)
94 const String& trackIdentifier() const { return m_metadata.trackIdentifier; }
95 void setTrackIdentifier(const String&);
96#endif
97
98private:
99 MediaMetadata();
100 void setArtworkImage(Image*);
101 void metadataUpdated();
102 void refreshArtworkImage();
103
104 WeakPtr<MediaSession> m_session;
105 MediaSessionMetadata m_metadata;
106 std::unique_ptr<ArtworkImageLoader> m_artworkLoader;
107 String m_artworkImageSrc;
108 RefPtr<Image> m_artworkImage;
109};
