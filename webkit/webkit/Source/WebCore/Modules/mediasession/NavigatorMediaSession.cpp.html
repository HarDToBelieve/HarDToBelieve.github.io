37NavigatorMediaSession::NavigatorMediaSession(Navigator& navigator)
38 : m_navigator(navigator)
39{
40}
41
42NavigatorMediaSession::~NavigatorMediaSession() = default;
43
44MediaSession& NavigatorMediaSession::mediaSession(Navigator& navigator)
45{
46 return NavigatorMediaSession::from(navigator)->mediaSession();
47}
48
49MediaSession& NavigatorMediaSession::mediaSession()
50{
51 if (!m_mediaSession)
52 m_mediaSession = MediaSession::create(m_navigator);
53 return *m_mediaSession;
54}
55
56NavigatorMediaSession* NavigatorMediaSession::from(Navigator& navigator)
57{
58 auto* supplement = static_cast<NavigatorMediaSession*>(Supplement<Navigator>::from(&navigator, supplementName()));
59 if (!supplement) {
60 auto newSupplement = makeUnique<NavigatorMediaSession>(navigator);
61 supplement = newSupplement.get();
62 provideTo(&navigator, supplementName(), WTFMove(newSupplement));
63 }
64 return supplement;
65}
66
67const char* NavigatorMediaSession::supplementName()
68{
69 return "NavigatorMediaSession";
70}
