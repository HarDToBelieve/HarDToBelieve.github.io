182void MediaSource::seekToTime(const MediaTime& time)
183{
184 if (isClosed())
185 return;
186
187 ALWAYS_LOG(LOGIDENTIFIER, time);
188
189 // 2.4.3 Seeking
190 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#mediasource-seeking
191
192 m_pendingSeekTime = time;
193 m_private->setIsSeeking(true);
194
195 // Run the following steps as part of the "Wait until the user agent has established whether or not the
196 // media data for the new playback position is available, and, if it is, until it has decoded enough data
197 // to play back that position" step of the seek algorithm:
198 // If new playback position is not in any TimeRange of HTMLMediaElement.buffered
199 if (!hasBufferedTime(time)) {
200 // 1. If the HTMLMediaElement.readyState attribute is greater than HAVE_METADATA,
201 // then set the HTMLMediaElement.readyState attribute to HAVE_METADATA.
202 m_private->setReadyState(MediaPlayer::ReadyState::HaveMetadata);
203
204 // 2. The media element waits until an appendBuffer() or an appendStream() call causes the coded
205 // frame processing algorithm to set the HTMLMediaElement.readyState attribute to a value greater
206 // than HAVE_METADATA.
207 m_private->waitForSeekCompleted();
208 return;
209 }
210 // Otherwise
211 // Continue
212
213// https://bugs.webkit.org/show_bug.cgi?id=125157 broke seek on MediaPlayerPrivateGStreamerMSE
214#if !USE(GSTREAMER)
215 m_private->waitForSeekCompleted();
216#endif
217 completeSeek();
218}

220void MediaSource::completeSeek()
221{
222 if (isClosed())
223 return;
224
225 // 2.4.3 Seeking, ctd.
226 // https://dvcs.w3.org/hg/html-media/raw-file/tip/media-source/media-source.html#mediasource-seeking
227
228 ASSERT(m_pendingSeekTime.isValid());
229
230 ALWAYS_LOG(LOGIDENTIFIER, m_pendingSeekTime);
231
232 // 2. The media element resets all decoders and initializes each one with data from the appropriate
233 // initialization segment.
234 // 3. The media element feeds coded frames from the active track buffers into the decoders starting
235 // with the closest random access point before the new playback position.
236 MediaTime pendingSeekTime = m_pendingSeekTime;
237 m_pendingSeekTime = MediaTime::invalidTime();
238 m_private->setIsSeeking(false);
239 for (auto& sourceBuffer : *m_activeSourceBuffers)
240 sourceBuffer->seekToTime(pendingSeekTime);
241
242 // 4. Resume the seek algorithm at the "Await a stable state" step.
243 m_private->seekCompleted();
244
245 monitorSourceBuffers();
246}

248Ref<TimeRanges> MediaSource::seekable()
249{
250 // 6. HTMLMediaElement Extensions, seekable
251 // W3C Editor's Draft 16 September 2016
252 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#htmlmediaelement-extensions
253
254 // If duration equals NaN:
255 // Return an empty TimeRanges object.
256 if (m_duration.isInvalid())
257 return TimeRanges::create();
258
259 // If duration equals positive Infinity:
260 if (m_duration.isPositiveInfinite()) {
261 auto buffered = this->buffered();
262 // If live seekable range is not empty:
263 if (m_liveSeekable && m_liveSeekable->length()) {
264 // Let union ranges be the union of live seekable range and the HTMLMediaElement.buffered attribute.
265 buffered->unionWith(*m_liveSeekable);
266 // Return a single range with a start time equal to the earliest start time in union ranges
267 // and an end time equal to the highest end time in union ranges and abort these steps.
268 buffered->add(buffered->start(0), buffered->maximumBufferedTime());
269 return TimeRanges::create(*buffered);
270 }
271
272 // If the HTMLMediaElement.buffered attribute returns an empty TimeRanges object, then return
273 // an empty TimeRanges object and abort these steps.
274 if (!buffered->length())
275 return TimeRanges::create();
276
277 // Return a single range with a start time of 0 and an end time equal to the highest end time
278 // reported by the HTMLMediaElement.buffered attribute.
279 return TimeRanges::create({MediaTime::zeroTime(), buffered->maximumBufferedTime()});
280 }
281
282 // Otherwise:
283 // Return a single range with a start time of 0 and an end time equal to duration.
284 return TimeRanges::create({MediaTime::zeroTime(), m_duration});
285}

287ExceptionOr<void> MediaSource::setLiveSeekableRange(double start, double end)
288{
289 // W3C Editor's Draft 16 September 2016
290 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#dom-mediasource-setliveseekablerange
291
292 ALWAYS_LOG(LOGIDENTIFIER, "start = ", start, ", end = ", end);
293
294 // If the readyState attribute is not "open" then throw an InvalidStateError exception and abort these steps.
295 if (!isOpen())
296 return Exception { InvalidStateError };
297
298 // If start is negative or greater than end, then throw a TypeError exception and abort these steps.
299 if (start < 0 || start > end)
300 return Exception { TypeError };
301
302 // Set live seekable range to be a new normalized TimeRanges object containing a single range
303 // whose start position is start and end position is end.
304 m_liveSeekable = makeUnique<PlatformTimeRanges>(MediaTime::createWithDouble(start), MediaTime::createWithDouble(end));
305
306 return { };
307}

335bool MediaSource::hasBufferedTime(const MediaTime& time)
336{
337 if (time > duration())
338 return false;
339
340 auto ranges = buffered();
341 if (!ranges->length())
342 return false;
343
344 return abs(ranges->nearest(time) - time) <= currentTimeFudgeFactor();
345}
346
347bool MediaSource::hasCurrentTime()
348{
349 return hasBufferedTime(currentTime());
350}
351
352bool MediaSource::hasFutureTime()
353{
354 MediaTime currentTime = this->currentTime();
355 MediaTime duration = this->duration();
356
357 if (currentTime >= duration)
358 return true;
359
360 auto ranges = buffered();
361 MediaTime nearest = ranges->nearest(currentTime);
362 if (abs(nearest - currentTime) > currentTimeFudgeFactor())
363 return false;
364
365 size_t found = ranges->find(nearest);
366 if (found == notFound)
367 return false;
368
369 MediaTime localEnd = ranges->end(found);
370 if (localEnd == duration)
371 return true;
372
373 return localEnd - currentTime > currentTimeFudgeFactor();
374}

376void MediaSource::monitorSourceBuffers()
377{
378 if (isClosed())
379 return;
380
381 // 2.4.4 SourceBuffer Monitoring
382 // https://rawgit.com/w3c/media-source/45627646344eea0170dd1cbc5a3d508ca751abb8/media-source-respec.html#buffer-monitoring
383
384 // Note, the behavior if activeSourceBuffers is empty is undefined.
385 if (!m_activeSourceBuffers) {
386 m_private->setReadyState(MediaPlayer::ReadyState::HaveNothing);
387 return;
388 }
389
390 // If the HTMLMediaElement.readyState attribute equals HAVE_NOTHING:
391 if (mediaElement()->readyState() == HTMLMediaElement::HAVE_NOTHING) {
392 // 1. Abort these steps.
393 return;
394 }
395
396 // If HTMLMediaElement.buffered does not contain a TimeRange for the current playback position:
397 if (!hasCurrentTime()) {
398 // 1. Set the HTMLMediaElement.readyState attribute to HAVE_METADATA.
399 // 2. If this is the first transition to HAVE_METADATA, then queue a task to fire a simple event
400 // named loadedmetadata at the media element.
401 m_private->setReadyState(MediaPlayer::ReadyState::HaveMetadata);
402
403 // 3. Abort these steps.
404 return;
405 }
406
407 // If HTMLMediaElement.buffered contains a TimeRange that includes the current
408 // playback position and enough data to ensure uninterrupted playback:
409 auto ranges = buffered();
410 if (std::all_of(m_activeSourceBuffers->begin(), m_activeSourceBuffers->end(), [&](auto& sourceBuffer) {
411 return sourceBuffer->canPlayThroughRange(*ranges);
412 })) {
413 // 1. Set the HTMLMediaElement.readyState attribute to HAVE_ENOUGH_DATA.
414 // 2. Queue a task to fire a simple event named canplaythrough at the media element.
415 // 3. Playback may resume at this point if it was previously suspended by a transition to HAVE_CURRENT_DATA.
416 m_private->setReadyState(MediaPlayer::ReadyState::HaveEnoughData);
417
418 if (m_pendingSeekTime.isValid())
419 completeSeek();
420
421 // 4. Abort these steps.
422 return;
423 }
424
425 // If HTMLMediaElement.buffered contains a TimeRange that includes the current playback
426 // position and some time beyond the current playback position, then run the following steps:
427 if (hasFutureTime()) {
428 // 1. Set the HTMLMediaElement.readyState attribute to HAVE_FUTURE_DATA.
429 // 2. If the previous value of HTMLMediaElement.readyState was less than HAVE_FUTURE_DATA, then queue a task to fire a simple event named canplay at the media element.
430 // 3. Playback may resume at this point if it was previously suspended by a transition to HAVE_CURRENT_DATA.
431 m_private->setReadyState(MediaPlayer::ReadyState::HaveFutureData);
432
433 if (m_pendingSeekTime.isValid())
434 completeSeek();
435
436 // 4. Abort these steps.
437 return;
438 }
439
440 // If HTMLMediaElement.buffered contains a TimeRange that ends at the current playback position and does not have a range covering the time immediately after the current position:
441 // NOTE: Logically, !(all objects do not contain currentTime) == (some objects contain current time)
442
443 // 1. Set the HTMLMediaElement.readyState attribute to HAVE_CURRENT_DATA.
444 // 2. If this is the first transition to HAVE_CURRENT_DATA, then queue a task to fire a simple
445 // event named loadeddata at the media element.
446 // 3. Playback is suspended at this point since the media element doesn't have enough data to
447 // advance the media timeline.
448 m_private->setReadyState(MediaPlayer::ReadyState::HaveCurrentData);
449
450 if (m_pendingSeekTime.isValid())
451 completeSeek();
452
453 // 4. Abort these steps.
454}

456ExceptionOr<void> MediaSource::setDuration(double duration)
457{
458 // 2.1 Attributes - Duration
459 // https://www.w3.org/TR/2016/REC-media-source-20161117/#attributes
460
461 ALWAYS_LOG(LOGIDENTIFIER, duration);
462
463 // On setting, run the following steps:
464 // 1. If the value being set is negative or NaN then throw a TypeError exception and abort these steps.
465 if (duration < 0.0 || std::isnan(duration))
466 return Exception { TypeError };
467
468 // 2. If the readyState attribute is not "open" then throw an InvalidStateError exception and abort these steps.
469 if (!isOpen())
470 return Exception { InvalidStateError };
471
472 // 3. If the updating attribute equals true on any SourceBuffer in sourceBuffers, then throw an InvalidStateError
473 // exception and abort these steps.
474 for (auto& sourceBuffer : *m_sourceBuffers) {
475 if (sourceBuffer->updating())
476 return Exception { InvalidStateError };
477 }
478
479 // 4. Run the duration change algorithm with new duration set to the value being assigned to this attribute.
480 return setDurationInternal(MediaTime::createWithDouble(duration));
481}
482
483ExceptionOr<void> MediaSource::setDurationInternal(const MediaTime& duration)
484{
485 // 2.4.6 Duration Change
486 // https://www.w3.org/TR/2016/REC-media-source-20161117/#duration-change-algorithm
487
488 MediaTime newDuration = duration;
489
490 // 1. If the current value of duration is equal to new duration, then return.
491 if (newDuration == m_duration)
492 return { };
493
494 // 2. If new duration is less than the highest presentation timestamp of any buffered coded frames
495 // for all SourceBuffer objects in sourceBuffers, then throw an InvalidStateError exception and
496 // abort these steps.
497 // 3. Let highest end time be the largest track buffer ranges end time across all the track buffers
498 // across all SourceBuffer objects in sourceBuffers.
499 MediaTime highestPresentationTimestamp;
500 MediaTime highestEndTime;
501 for (auto& sourceBuffer : *m_sourceBuffers) {
502 highestPresentationTimestamp = std::max(highestPresentationTimestamp, sourceBuffer->highestPresentationTimestamp());
503 highestEndTime = std::max(highestEndTime, sourceBuffer->bufferedInternal().ranges().maximumBufferedTime());
504 }
505 if (highestPresentationTimestamp.isValid() && newDuration < highestPresentationTimestamp)
506 return Exception { InvalidStateError };
507
508 // 4. If new duration is less than highest end time, then
509 // 4.1. Update new duration to equal highest end time.
510 if (highestEndTime.isValid() && newDuration < highestEndTime)
511 newDuration = highestEndTime;
512
513 // 5. Update duration to new duration.
514 m_duration = newDuration;
515 ALWAYS_LOG(LOGIDENTIFIER, duration);
516
517 // 6. Update the media duration to new duration and run the HTMLMediaElement duration change algorithm.
518 m_private->durationChanged(duration);
519
520 return { };
521}

534ExceptionOr<void> MediaSource::endOfStream(std::optional<EndOfStreamError> error)
535{
536 ALWAYS_LOG(LOGIDENTIFIER);
537
538 // 2.2 https://dvcs.w3.org/hg/html-media/raw-file/tip/media-source/media-source.html#widl-MediaSource-endOfStream-void-EndOfStreamError-error
539 // 1. If the readyState attribute is not in the "open" state then throw an
540 // InvalidStateError exception and abort these steps.
541 if (!isOpen())
542 return Exception { InvalidStateError };
543
544 // 2. If the updating attribute equals true on any SourceBuffer in sourceBuffers, then throw an
545 // InvalidStateError exception and abort these steps.
546 if (std::any_of(m_sourceBuffers->begin(), m_sourceBuffers->end(), [](auto& sourceBuffer) { return sourceBuffer->updating(); }))
547 return Exception { InvalidStateError };
548
549 // 3. Run the end of stream algorithm with the error parameter set to error.
550 streamEndedWithError(error);
551
552 return { };
553}
554
555void MediaSource::streamEndedWithError(std::optional<EndOfStreamError> error)
556{
557#if !RELEASE_LOG_DISABLED
558 if (error)
559 ALWAYS_LOG(LOGIDENTIFIER, error.value());
560 else
561 ALWAYS_LOG(LOGIDENTIFIER);
562#endif
563
564 if (isClosed())
565 return;
566
567 // 2.4.7 https://dvcs.w3.org/hg/html-media/raw-file/tip/media-source/media-source.html#end-of-stream-algorithm
568
569 // 1. Change the readyState attribute value to "ended".
570 // 2. Queue a task to fire a simple event named sourceended at the MediaSource.
571 setReadyState(ReadyState::Ended);
572
573 // 3.
574 if (!error) {
575 // If error is not set, is null, or is an empty string
576 // 1. Run the duration change algorithm with new duration set to the highest end time reported by
577 // the buffered attribute across all SourceBuffer objects in sourceBuffers.
578 MediaTime maxEndTime;
579 for (auto& sourceBuffer : *m_sourceBuffers) {
580 if (auto length = sourceBuffer->bufferedInternal().length())
581 maxEndTime = std::max(sourceBuffer->bufferedInternal().ranges().end(length - 1), maxEndTime);
582 }
583 setDurationInternal(maxEndTime);
584
585 // 2. Notify the media element that it now has all of the media data.
586 for (auto& sourceBuffer : *m_sourceBuffers)
587 sourceBuffer->setMediaSourceEnded(true);
588 m_private->markEndOfStream(MediaSourcePrivate::EosNoError);
589 } else if (error == EndOfStreamError::Network) {
590 m_private->markEndOfStream(MediaSourcePrivate::EosNetworkError);
591 // If error is set to "network"
592 ASSERT(m_mediaElement);
593 if (m_mediaElement->readyState() == HTMLMediaElement::HAVE_NOTHING) {
594 // If the HTMLMediaElement.readyState attribute equals HAVE_NOTHING
595 // Run the "If the media data cannot be fetched at all, due to network errors, causing
596 // the user agent to give up trying to fetch the resource" steps of the resource fetch algorithm.
597 // NOTE: This step is handled by HTMLMediaElement::mediaLoadingFailed().
598 m_mediaElement->mediaLoadingFailed(MediaPlayer::NetworkState::NetworkError);
599 } else {
600 // If the HTMLMediaElement.readyState attribute is greater than HAVE_NOTHING
601 // Run the "If the connection is interrupted after some media data has been received, causing the
602 // user agent to give up trying to fetch the resource" steps of the resource fetch algorithm.
603 // NOTE: This step is handled by HTMLMediaElement::mediaLoadingFailedFatally().
604 m_mediaElement->mediaLoadingFailedFatally(MediaPlayer::NetworkState::NetworkError);
605 }
606 } else {
607 // If error is set to "decode"
608 ASSERT(error == EndOfStreamError::Decode);
609 m_private->markEndOfStream(MediaSourcePrivate::EosDecodeError);
610
611 ASSERT(m_mediaElement);
612 if (m_mediaElement->readyState() == HTMLMediaElement::HAVE_NOTHING) {
613 // If the HTMLMediaElement.readyState attribute equals HAVE_NOTHING
614 // Run the "If the media data can be fetched but is found by inspection to be in an unsupported
615 // format, or can otherwise not be rendered at all" steps of the resource fetch algorithm.
616 // NOTE: This step is handled by HTMLMediaElement::mediaLoadingFailed().
617 m_mediaElement->mediaLoadingFailed(MediaPlayer::NetworkState::FormatError);
618 } else {
619 // If the HTMLMediaElement.readyState attribute is greater than HAVE_NOTHING
620 // Run the media data is corrupted steps of the resource fetch algorithm.
621 // NOTE: This step is handled by HTMLMediaElement::mediaLoadingFailedFatally().
622 m_mediaElement->mediaLoadingFailedFatally(MediaPlayer::NetworkState::DecodeError);
623 }
624 }

627static ContentType addVP9FullRangeVideoFlagToContentType(const ContentType& type)
628{
629 auto countPeriods = [] (const String& codec) {
630 unsigned count = 0;
631 unsigned position = 0;
632
633 while (codec.find('.', position) != notFound) {
634 ++count;
635 ++position;
636 }
637
638 return count;
639 };
640
641 for (auto codec : type.codecs()) {
642 if (!codec.startsWith("vp09") || countPeriods(codec) != 7)
643 continue;
644
645 auto rawType = type.raw();
646 auto position = rawType.find(codec);
647 ASSERT(position != notFound);
648 if (position == notFound)
649 continue;
650

1112void MediaSource::updateBufferedIfNeeded()
1113{
1114 if (m_buffered && m_activeSourceBuffers->length() && std::all_of(m_activeSourceBuffers->begin(), m_activeSourceBuffers->end(), [](auto& buffer) { return !buffer->isBufferedDirty(); }))
1115 return;
1116
1117 m_buffered = makeUnique<PlatformTimeRanges>();
1118 for (auto& sourceBuffer : *m_activeSourceBuffers)
1119 sourceBuffer->setBufferedDirty(false);
1120
1121 // Implements MediaSource algorithm for HTMLMediaElement.buffered.
1122 // https://dvcs.w3.org/hg/html-media/raw-file/default/media-source/media-source.html#htmlmediaelement-extensions
1123 Vector<PlatformTimeRanges> activeRanges = this->activeRanges();
1124
1125 // 1. If activeSourceBuffers.length equals 0 then return an empty TimeRanges object and abort these steps.
1126 if (activeRanges.isEmpty())
1127 return;
1128
1129 // 2. Let active ranges be the ranges returned by buffered for each SourceBuffer object in activeSourceBuffers.
1130 // 3. Let highest end time be the largest range end time in the active ranges.
1131 MediaTime highestEndTime = MediaTime::zeroTime();
1132 for (auto& ranges : activeRanges) {
1133 unsigned length = ranges.length();
1134 if (length)
1135 highestEndTime = std::max(highestEndTime, ranges.end(length - 1));
1136 }
1137
1138 // Return an empty range if all ranges are empty.
1139 if (!highestEndTime)
1140 return;
1141
1142 // 4. Let intersection ranges equal a TimeRange object containing a single range from 0 to highest end time.
1143 m_buffered->add(MediaTime::zeroTime(), highestEndTime);
1144
1145 // 5. For each SourceBuffer object in activeSourceBuffers run the following steps:
1146 bool ended = readyState() == ReadyState::Ended;
1147 for (auto& sourceRanges : activeRanges) {
1148 // 5.1 Let source ranges equal the ranges returned by the buffered attribute on the current SourceBuffer.
1149 // 5.2 If readyState is "ended", then set the end time on the last range in source ranges to highest end time.
1150 if (ended && sourceRanges.length())
1151 sourceRanges.add(sourceRanges.start(sourceRanges.length() - 1), highestEndTime);
1152
1153 // 5.3 Let new intersection ranges equal the intersection between the intersection ranges and the source ranges.
1154 // 5.4 Replace the ranges in intersection ranges with the new intersection ranges.
1155 m_buffered->intersectWith(sourceRanges);
1156 }
1157
1158 if (m_private)
1159 m_private->bufferedChanged(*m_buffered);
1160}
