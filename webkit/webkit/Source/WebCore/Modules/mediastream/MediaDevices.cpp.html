72void MediaDevices::stop()
73{
74 if (m_deviceChangeToken) {
75 auto* controller = UserMediaController::from(document()->page());
76 if (controller)
77 controller->removeDeviceChangeObserver(m_deviceChangeToken);
78 }
79 m_scheduledEventTimer.stop();
80}

92static MediaConstraints createMediaConstraints(const Variant<bool, MediaTrackConstraints>& constraints)
93{
94 return WTF::switchOn(constraints,
95 [&] (bool isValid) {
96 MediaConstraints constraints;
97 constraints.isValid = isValid;
98 return constraints;
99 },
100 [&] (const MediaTrackConstraints& trackConstraints) {
101 return createMediaConstraints(trackConstraints);
102 }
103 );
104}

106bool MediaDevices::computeUserGesturePriviledge(GestureAllowedRequest requestType)
107{
108 auto* currentGestureToken = UserGestureIndicator::currentUserGesture().get();
109 if (m_currentGestureToken.get() != currentGestureToken) {
110 m_currentGestureToken = makeWeakPtr(currentGestureToken);
111 m_requestTypesForCurrentGesture = { };
112 }
113
114 bool isUserGesturePriviledged = m_currentGestureToken && !m_requestTypesForCurrentGesture.contains(requestType);
115 m_requestTypesForCurrentGesture.add(requestType);
116 return isUserGesturePriviledged;
117}

119void MediaDevices::getUserMedia(const StreamConstraints& constraints, Promise&& promise)
120{
121 auto* document = this->document();
122 if (!document || !document->isFullyActive()) {
123 promise.reject(Exception { InvalidStateError, "Document is not fully active"_s });
124 return;
125 }
126
127 auto audioConstraints = createMediaConstraints(constraints.audio);
128 auto videoConstraints = createMediaConstraints(constraints.video);
129
130 bool isUserGesturePriviledged = false;
131
132 if (audioConstraints.isValid)
133 isUserGesturePriviledged |= computeUserGesturePriviledge(GestureAllowedRequest::Microphone);
134
135 if (videoConstraints.isValid) {
136 isUserGesturePriviledged |= computeUserGesturePriviledge(GestureAllowedRequest::Camera);
137 videoConstraints.setDefaultVideoConstraints();
138 }
139
140 auto request = UserMediaRequest::create(*document, { MediaStreamRequest::Type::UserMedia, WTFMove(audioConstraints), WTFMove(videoConstraints), isUserGesturePriviledged }, WTFMove(promise));
141 request->start();
142}

144void MediaDevices::getDisplayMedia(const DisplayMediaStreamConstraints& constraints, Promise&& promise)
145{
146 auto* document = this->document();
147 if (!document)
148 return;
149
150 bool isUserGesturePriviledged = computeUserGesturePriviledge(GestureAllowedRequest::Display);
151 if (!isUserGesturePriviledged) {
152 promise.reject(Exception { InvalidAccessError, "getDisplayMedia must be called from a user gesture handler."_s });
153 return;
154 }
155
156 auto request = UserMediaRequest::create(*document, { MediaStreamRequest::Type::DisplayMedia, { }, createMediaConstraints(constraints.video), isUserGesturePriviledged }, WTFMove(promise));
157 request->start();
158}

160static inline bool checkCameraAccess(const Document& document)
161{
162 return isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::Camera, document, LogFeaturePolicyFailure::No);
163}
164
165static inline bool checkMicrophoneAccess(const Document& document)
166{
167 return isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::Microphone, document, LogFeaturePolicyFailure::No);
168}
169
170static inline bool checkSpeakerAccess(const Document& document)
171{
172 return document.frame()
173 && document.frame()->settings().exposeSpeakersEnabled()
174 && isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::SpeakerSelection, document, LogFeaturePolicyFailure::No);
175}

177static inline MediaDeviceInfo::Kind toMediaDeviceInfoKind(CaptureDevice::DeviceType type)
178{
179 switch (type) {
180 case CaptureDevice::DeviceType::Microphone:
181 return MediaDeviceInfo::Kind::Audioinput;
182 case CaptureDevice::DeviceType::Speaker:
183 return MediaDeviceInfo::Kind::Audiooutput;
184 case CaptureDevice::DeviceType::Camera:
185 case CaptureDevice::DeviceType::Screen:
186 case CaptureDevice::DeviceType::Window:
187 return MediaDeviceInfo::Kind::Videoinput;
188 case CaptureDevice::DeviceType::Unknown:
189 ASSERT_NOT_REACHED();
190 }
191 return MediaDeviceInfo::Kind::Audioinput;
192}

194void MediaDevices::exposeDevices(const Vector<CaptureDevice>& newDevices, const String& deviceIDHashSalt, EnumerateDevicesPromise&& promise)
195{
196 if (isContextStopped())
197 return;
198
199 auto& document = *this->document();
200
201 bool canAccessCamera = checkCameraAccess(document);
202 bool canAccessMicrophone = checkMicrophoneAccess(document);
203 bool canAccessSpeaker = checkSpeakerAccess(document);
204
205 m_audioOutputDeviceIdToPersistentId.clear();
206
207 Vector<Ref<MediaDeviceInfo>> devices;
208 for (auto& newDevice : newDevices) {
209 if (!canAccessMicrophone && newDevice.type() == CaptureDevice::DeviceType::Microphone)
210 continue;
211 if (!canAccessCamera && newDevice.type() == CaptureDevice::DeviceType::Camera)
212 continue;
213 if (!canAccessSpeaker && newDevice.type() == CaptureDevice::DeviceType::Speaker)
214 continue;
215
216 auto deviceId = RealtimeMediaSourceCenter::singleton().hashStringWithSalt(newDevice.persistentId(), deviceIDHashSalt);
217 auto groupId = RealtimeMediaSourceCenter::singleton().hashStringWithSalt(newDevice.groupId(), m_groupIdHashSalt);
218
219 if (newDevice.type() == CaptureDevice::DeviceType::Speaker)
220 m_audioOutputDeviceIdToPersistentId.add(deviceId, newDevice.persistentId());
221
222 devices.append(MediaDeviceInfo::create(newDevice.label(), WTFMove(deviceId), WTFMove(groupId), toMediaDeviceInfoKind(newDevice.type())));
223 }
224 promise.resolve(devices);
225}

227void MediaDevices::enumerateDevices(EnumerateDevicesPromise&& promise)
228{
229 auto* document = this->document();
230 if (!document)
231 return;
232
233 auto* controller = UserMediaController::from(document->page());
234 if (!controller) {
235 promise.resolve({ });
236 return;
237 }
238
239 if (!checkCameraAccess(*document) && !checkMicrophoneAccess(*document)) {
240 controller->logEnumerateDevicesDenial(*document);
241 promise.resolve({ });
242 return;
243 }
244
245 controller->enumerateMediaDevices(*document, [this, weakThis = makeWeakPtr(this), promise = WTFMove(promise)](const auto& newDevices, const auto& deviceIDHashSalt) mutable {
246 if (!weakThis)
247 return;
248 exposeDevices(newDevices, deviceIDHashSalt, WTFMove(promise));
249 });
250}

252MediaTrackSupportedConstraints MediaDevices::getSupportedConstraints()
253{
254 auto& supported = RealtimeMediaSourceCenter::singleton().supportedConstraints();
255 MediaTrackSupportedConstraints result;
256 result.width = supported.supportsWidth();
257 result.height = supported.supportsHeight();
258 result.aspectRatio = supported.supportsAspectRatio();
259 result.frameRate = supported.supportsFrameRate();
260 result.facingMode = supported.supportsFacingMode();
261 result.volume = supported.supportsVolume();
262 result.sampleRate = supported.supportsSampleRate();
263 result.sampleSize = supported.supportsSampleSize();
264 result.echoCancellation = supported.supportsEchoCancellation();
265 result.deviceId = supported.supportsDeviceId();
266 result.groupId = supported.supportsGroupId();
267
268 return result;
269}

287void MediaDevices::listenForDeviceChanges()
288{
289 auto* document = this->document();
290 auto* controller = document ? UserMediaController::from(document->page()) : nullptr;
291 if (!controller)
292 return;
293
294 bool canAccessCamera = isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::Camera, *document, LogFeaturePolicyFailure::No);
295 bool canAccessMicrophone = isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::Microphone, *document, LogFeaturePolicyFailure::No);
296
297 if (m_listeningForDeviceChanges || (!canAccessCamera && !canAccessMicrophone))
298 return;
299
300 m_listeningForDeviceChanges = true;
301
302 m_deviceChangeToken = controller->addDeviceChangeObserver([weakThis = makeWeakPtr(*this), this]() {
303 if (!weakThis || isContextStopped() || m_scheduledEventTimer.isActive())
304 return;
305
306 m_scheduledEventTimer.startOneShot(Seconds(randomNumber() / 2));
307 });
308}

310bool MediaDevices::addEventListener(const AtomString& eventType, Ref<EventListener>&& listener, const AddEventListenerOptions& options)
311{
312 if (eventType == eventNames().devicechangeEvent)
313 listenForDeviceChanges();
314
315 return EventTargetWithInlineData::addEventListener(eventType, WTFMove(listener), options);
316}
