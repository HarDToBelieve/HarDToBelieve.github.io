66static inline MediaStreamTrackPrivateVector createTrackPrivateVector(const MediaStreamTrackVector& tracks)
67{
68 return map(tracks, [](auto& track) {
69 return makeRefPtr(&track->privateTrack());
70 });
71}
72
73MediaStream::MediaStream(Document& document, const MediaStreamTrackVector& tracks)
74 : ActiveDOMObject(document)
75 , m_private(MediaStreamPrivate::create(document.logger(), createTrackPrivateVector(tracks)))
76{
77 // This constructor preserves MediaStreamTrack instances and must be used by calls originating
78 // from the JavaScript MediaStream constructor.
79
80 for (auto& track : tracks)
81 m_trackSet.add(track->id(), track);
82
83 setIsActive(m_private->active());
84 m_private->addObserver(*this);
85 suspendIfNeeded();
86}
87
88MediaStream::MediaStream(Document& document, Ref<MediaStreamPrivate>&& streamPrivate)
89 : ActiveDOMObject(document)
90 , m_private(WTFMove(streamPrivate))
91{
92 ALWAYS_LOG(LOGIDENTIFIER);
93
94 for (auto& trackPrivate : m_private->tracks())
95 m_trackSet.add(trackPrivate->id(), MediaStreamTrack::create(document, *trackPrivate));
96
97 setIsActive(m_private->active());
98 m_private->addObserver(*this);
99 suspendIfNeeded();
100}
101
102MediaStream::~MediaStream()
103{
104 // Set isActive to false immediately so any callbacks triggered by shutting down, e.g.
105 // mediaState(), are short circuited.
106 m_isActive = false;
107 m_private->removeObserver(*this);
108 if (auto* document = this->document()) {
109 if (m_isWaitingUntilMediaCanStart)
110 document->removeMediaCanStartListener(*this);
111 }
112}

114RefPtr<MediaStream> MediaStream::clone()
115{
116 ALWAYS_LOG(LOGIDENTIFIER);
117
118 MediaStreamTrackVector clonedTracks;
119 clonedTracks.reserveInitialCapacity(m_trackSet.size());
120
121 for (auto& track : m_trackSet.values())
122 clonedTracks.uncheckedAppend(track->clone());
123
124 return MediaStream::create(*document(), clonedTracks);
125}

127void MediaStream::addTrack(MediaStreamTrack& track)
128{
129 ALWAYS_LOG(LOGIDENTIFIER, track.logIdentifier());
130 if (getTrackById(track.privateTrack().id()))
131 return;
132
133 internalAddTrack(track);
134 m_private->addTrack(track.privateTrack());
135}
136
137void MediaStream::removeTrack(MediaStreamTrack& track)
138{
139 ALWAYS_LOG(LOGIDENTIFIER, track.logIdentifier());
140 if (auto taken = internalTakeTrack(track.id())) {
141 ASSERT(taken.get() == &track);
142 m_private->removeTrack(track.privateTrack());
143 }
144}

146MediaStreamTrack* MediaStream::getTrackById(String id)
147{
148 auto iterator = m_trackSet.find(id);
149 if (iterator != m_trackSet.end())
150 return iterator->value.get();
151
152 return nullptr;
153}

175void MediaStream::didAddTrack(MediaStreamTrackPrivate& trackPrivate)
176{
177 ScriptExecutionContext* context = scriptExecutionContext();
178 if (!context)
179 return;
180
181 if (getTrackById(trackPrivate.id()))
182 return;
183
184 auto track = MediaStreamTrack::create(*context, trackPrivate);
185 internalAddTrack(track.copyRef());
186 dispatchEvent(MediaStreamTrackEvent::create(eventNames().addtrackEvent, Event::CanBubble::No, Event::IsCancelable::No, WTFMove(track)));
187}
188
189void MediaStream::didRemoveTrack(MediaStreamTrackPrivate& trackPrivate)
190{
191 if (auto track = internalTakeTrack(trackPrivate.id()))
192 dispatchEvent(MediaStreamTrackEvent::create(eventNames().removetrackEvent, Event::CanBubble::No, Event::IsCancelable::No, WTFMove(track)));
193}

195void MediaStream::addTrackFromPlatform(Ref<MediaStreamTrack>&& track)
196{
197 ALWAYS_LOG(LOGIDENTIFIER, track->logIdentifier());
198
199 auto& privateTrack = track->privateTrack();
200 internalAddTrack(track.copyRef());
201 m_private->addTrack(privateTrack);
202 dispatchEvent(MediaStreamTrackEvent::create(eventNames().addtrackEvent, Event::CanBubble::No, Event::IsCancelable::No, WTFMove(track)));
203}
204
205void MediaStream::internalAddTrack(Ref<MediaStreamTrack>&& trackToAdd)
206{
207 ASSERT(!m_trackSet.contains(trackToAdd->id()));
208 m_trackSet.add(trackToAdd->id(), WTFMove(trackToAdd));
209 updateActiveState();
210}
211
212RefPtr<MediaStreamTrack> MediaStream::internalTakeTrack(const String& trackId)
213{
214 auto track = m_trackSet.take(trackId);
215 if (track)
216 updateActiveState();
217
218 return track;
219}

221void MediaStream::setIsActive(bool active)
222{
223 if (m_isActive == active)
224 return;
225
226 ALWAYS_LOG(LOGIDENTIFIER, active);
227
228 m_isActive = active;
229 statusDidChange();
230}

232void MediaStream::mediaCanStart(Document& document)
233{
234 ALWAYS_LOG(LOGIDENTIFIER);
235
236 ASSERT_UNUSED(document, &document == this->document());
237 ASSERT(m_isWaitingUntilMediaCanStart);
238 if (m_isWaitingUntilMediaCanStart) {
239 m_isWaitingUntilMediaCanStart = false;
240 startProducingData();
241 }
242}
243
244void MediaStream::startProducingData()
245{
246 Document* document = this->document();
247 if (!document || !document->page())
248 return;
249
250 ALWAYS_LOG(LOGIDENTIFIER);
251
252 // If we can't start a load right away, start it later.
253 if (!document->page()->canStartMedia()) {
254 ALWAYS_LOG(LOGIDENTIFIER, "not allowed to start in background, waiting");
255 if (m_isWaitingUntilMediaCanStart)
256 return;
257
258 m_isWaitingUntilMediaCanStart = true;
259 document->addMediaCanStartListener(*this);
260 return;
261 }
262
263 if (m_isProducingData)
264 return;
265 m_isProducingData = true;
266 m_private->startProducingData();
267}
268
269void MediaStream::stopProducingData()
270{
271 ALWAYS_LOG(LOGIDENTIFIER);
272
273 if (!m_isProducingData)
274 return;
275
276 m_isProducingData = false;
277 m_private->stopProducingData();
278}

280MediaProducer::MediaStateFlags MediaStream::mediaState() const
281{
282 MediaProducer::MediaStateFlags state;
283
284 if (!m_isActive || !document() || !document()->page())
285 return state;
286
287 for (const auto& track : m_trackSet.values())
288 state.add(track->mediaState());
289
290 return state;
291}
292
293void MediaStream::statusDidChange()
294{
295 if (auto* document = this->document()) {
296 if (!m_isActive)
297 return;
298 document->updateIsPlayingMedia();
299 }
300}
301
302void MediaStream::characteristicsChanged()
303{
304 auto state = mediaState();
305 if (m_state != state) {
306 m_state = state;
307 statusDidChange();
308 }
309}
310
311void MediaStream::updateActiveState()
312{
313 bool active = false;
314 for (auto& track : m_trackSet.values()) {
315 if (!track->ended()) {
316 active = true;
317 break;
318 }
319 }
320
321 if (m_isActive == active)
322 return;
323
324 setIsActive(active);
325}

327MediaStreamTrackVector MediaStream::trackVectorForType(RealtimeMediaSource::Type filterType) const
328{
329 MediaStreamTrackVector tracks;
330 for (auto& track : m_trackSet.values()) {
331 if (track->source().type() == filterType)
332 tracks.append(track);
333 }
334
335 return tracks;
336}
