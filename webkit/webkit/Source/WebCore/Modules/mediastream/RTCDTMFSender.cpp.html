41static const size_t minToneDurationMs = 40;
42static const size_t maxToneDurationMs = 6000;
43static const size_t minInterToneGapMs = 30;
44
45RTCDTMFSender::RTCDTMFSender(ScriptExecutionContext& context, RTCRtpSender& sender, std::unique_ptr<RTCDTMFSenderBackend>&& backend)
46 : ActiveDOMObject(&context)
47 , m_toneTimer(*this, &RTCDTMFSender::toneTimerFired)
48 , m_sender(makeWeakPtr(sender))
49 , m_backend(WTFMove(backend))
50{
51 m_backend->onTonePlayed([this](const String&) {
52 onTonePlayed();
53 });
54 suspendIfNeeded();
55}

57RTCDTMFSender::~RTCDTMFSender() = default;

59bool RTCDTMFSender::canInsertDTMF() const
60{
61 if (!m_sender || m_sender->isStopped())
62 return false;
63
64 auto currentDirection = m_sender->currentTransceiverDirection();
65 if (!currentDirection)
66 return false;
67 if (*currentDirection != RTCRtpTransceiverDirection::Sendrecv && *currentDirection != RTCRtpTransceiverDirection::Sendonly)
68 return false;
69
70 return m_backend && m_backend->canInsertDTMF();
71}

78static inline bool isToneCharacterInvalid(UChar character)
79{
80 if (character >= '0' && character <= '9')
81 return false;
82 if (character >= 'A' && character <= 'D')
83 return false;
84 return character != '#' && character != '*' && character != ',';
85}
86
87ExceptionOr<void> RTCDTMFSender::insertDTMF(const String& tones, size_t duration, size_t interToneGap)
88{
89 if (!canInsertDTMF())
90 return Exception { InvalidStateError, "Cannot insert DTMF"_s };
91
92 auto normalizedTones = tones.convertToUppercaseWithoutLocale();
93 if (normalizedTones.find(isToneCharacterInvalid) != notFound)
94 return Exception { InvalidCharacterError, "Tones are not valid"_s };
95
96 m_tones = WTFMove(normalizedTones);
97 m_duration = clampTo(duration, minToneDurationMs, maxToneDurationMs);
98 m_interToneGap = std::max(interToneGap, minInterToneGapMs);
99
100 if (m_tones.isEmpty() || m_isPendingPlayoutTask)
101 return { };
102
103 m_isPendingPlayoutTask = true;
104 scriptExecutionContext()->postTask([this, protectedThis = makeRef(*this)](auto&) {
105 playNextTone();
106 });
107 return { };
108}
109
110void RTCDTMFSender::playNextTone()
111{
112 if (m_tones.isEmpty()) {
113 m_isPendingPlayoutTask = false;
114 dispatchEvent(RTCDTMFToneChangeEvent::create({ }));
115 return;
116 }
117
118 if (!canInsertDTMF()) {
119 m_isPendingPlayoutTask = false;
120 return;
121 }
122
123 auto currentTone = m_tones.substring(0, 1);
124 m_tones.remove(0);
125
126 m_backend->playTone(currentTone, m_duration, m_interToneGap);
127 dispatchEvent(RTCDTMFToneChangeEvent::create(currentTone));
128}
129
130void RTCDTMFSender::onTonePlayed()
131{
132 m_toneTimer.startOneShot(1_ms * m_interToneGap);
133}
134
135void RTCDTMFSender::toneTimerFired()
136{
137 playNextTone();
138}
139
140void RTCDTMFSender::stop()
141{
142 m_backend = nullptr;
143 m_toneTimer.stop();
144}
