60Ref<RTCDataChannel> RTCDataChannel::create(ScriptExecutionContext& context, std::unique_ptr<RTCDataChannelHandler>&& handler, String&& label, RTCDataChannelInit&& options)
61{
62 ASSERT(handler);
63 auto channel = adoptRef(*new RTCDataChannel(context, WTFMove(handler), WTFMove(label), WTFMove(options)));
64 channel->suspendIfNeeded();
65 queueTaskKeepingObjectAlive(channel.get(), TaskSource::Networking, [channel = channel.ptr()] {
66 if (!channel->m_isDetachable)
67 return;
68 channel->m_isDetachable = false;
69 if (!channel->m_handler)
70 return;
71 if (auto* context = channel->scriptExecutionContext())
72 channel->m_handler->setClient(*channel, context->contextIdentifier());
73 });
74 return channel;
75}

77NetworkSendQueue RTCDataChannel::createMessageQueue(ScriptExecutionContext& context, RTCDataChannel& channel)
78{
79 return { context, [&channel](auto& utf8) {
80 if (!channel.m_handler->sendStringData(utf8))
81 channel.scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Error, "Error sending string through RTCDataChannel."_s);
82 }, [&channel](auto& span) {
83 if (!channel.m_handler->sendRawData(span.data(), span.size()))
84 channel.scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Error, "Error sending binary data through RTCDataChannel."_s);
85 }, [&channel](ExceptionCode errorCode) {
86 if (auto* context = channel.scriptExecutionContext()) {
87 auto code = static_cast<int>(errorCode);
88 context->addConsoleMessage(MessageSource::JS, MessageLevel::Error, makeString("Error ", code, " in retrieving a blob data to be sent through RTCDataChannel."));
89 }
90 return NetworkSendQueue::Continue::Yes;
91 } };
92}

105const AtomString& RTCDataChannel::binaryType() const
106{
107 switch (m_binaryType) {
108 case BinaryType::Blob:
109 return blobKeyword();
110 case BinaryType::ArrayBuffer:
111 return arraybufferKeyword();
112 }
113
114 ASSERT_NOT_REACHED();
115 return emptyAtom();
116}

118ExceptionOr<void> RTCDataChannel::setBinaryType(const AtomString& binaryType)
119{
120 if (binaryType == blobKeyword()) {
121 m_binaryType = BinaryType::Blob;
122 return { };
123 }
124 if (binaryType == arraybufferKeyword()) {
125 m_binaryType = BinaryType::ArrayBuffer;
126 return { };
127 }
128 return Exception { TypeMismatchError };
129}

131ExceptionOr<void> RTCDataChannel::send(const String& data)
132{
133 if (m_readyState != RTCDataChannelState::Open)
134 return Exception { InvalidStateError };
135
136 // FIXME: We might want to use strict conversion like WebSocket.
137 auto utf8 = data.utf8();
138 m_bufferedAmount += utf8.length();
139 m_messageQueue.enqueue(WTFMove(utf8));
140 return { };
141}
142
143ExceptionOr<void> RTCDataChannel::send(ArrayBuffer& data)
144{
145 if (m_readyState != RTCDataChannelState::Open)
146 return Exception { InvalidStateError };
147
148 m_bufferedAmount += data.byteLength();
149 m_messageQueue.enqueue(data, 0, data.byteLength());
150 return { };
151}
152
153ExceptionOr<void> RTCDataChannel::send(ArrayBufferView& data)
154{
155 if (m_readyState != RTCDataChannelState::Open)
156 return Exception { InvalidStateError };
157
158 m_bufferedAmount += data.byteLength();
159 m_messageQueue.enqueue(*data.unsharedBuffer(), data.byteOffset(), data.byteLength());
160 return { };
161}
162
163ExceptionOr<void> RTCDataChannel::send(Blob& blob)
164{
165 if (m_readyState != RTCDataChannelState::Open)
166 return Exception { InvalidStateError };
167
168 m_bufferedAmount += blob.size();
169 m_messageQueue.enqueue(blob);
170 return { };
171}

173void RTCDataChannel::close()
174{
175 if (m_stopped)
176 return;
177
178 m_stopped = true;
179 m_readyState = RTCDataChannelState::Closed;
180
181 m_messageQueue.clear();
182
183 if (m_handler)
184 m_handler->close();
185 m_handler = nullptr;
186}

188bool RTCDataChannel::virtualHasPendingActivity() const
189{
190 return !m_stopped;
191}

193void RTCDataChannel::didChangeReadyState(RTCDataChannelState newState)
194{
195 if (m_stopped || m_readyState == RTCDataChannelState::Closed || m_readyState == newState)
196 return;
197
198 m_readyState = newState;
199
200 switch (m_readyState) {
201 case RTCDataChannelState::Open:
202 scheduleDispatchEvent(Event::create(eventNames().openEvent, Event::CanBubble::No, Event::IsCancelable::No));
203 break;
204 case RTCDataChannelState::Closed:
205 scheduleDispatchEvent(Event::create(eventNames().closeEvent, Event::CanBubble::No, Event::IsCancelable::No));
206 break;
207 default:
208 break;
209 }
210}

212void RTCDataChannel::didReceiveStringData(const String& text)
213{
214 scheduleDispatchEvent(MessageEvent::create(text));
215}
216
217void RTCDataChannel::didReceiveRawData(const uint8_t* data, size_t dataLength)
218{
219 switch (m_binaryType) {
220 case BinaryType::Blob:
221 scheduleDispatchEvent(MessageEvent::create(Blob::create(scriptExecutionContext(), Vector { data, dataLength }, emptyString()), { }));
222 return;
223 case BinaryType::ArrayBuffer:
224 scheduleDispatchEvent(MessageEvent::create(ArrayBuffer::create(data, dataLength)));
225 return;
226 }
227 ASSERT_NOT_REACHED();
228}

230void RTCDataChannel::didDetectError()
231{
232 scheduleDispatchEvent(Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::No));
233}

235void RTCDataChannel::bufferedAmountIsDecreasing(size_t amount)
236{
237 auto previousBufferedAmount = m_bufferedAmount;
238 m_bufferedAmount -= amount;
239 if (previousBufferedAmount > m_bufferedAmountLowThreshold && m_bufferedAmount <= m_bufferedAmountLowThreshold)
240 scheduleDispatchEvent(Event::create(eventNames().bufferedamountlowEvent, Event::CanBubble::No, Event::IsCancelable::No));
241}

250void RTCDataChannel::scheduleDispatchEvent(Ref<Event>&& event)
251{
252 if (m_stopped)
253 return;
254
255 // https://w3c.github.io/webrtc-pc/#operation
256 queueTaskToDispatchEvent(*this, TaskSource::Networking, WTFMove(event));
257}

260static HashMap<RTCDataChannelLocalIdentifier, std::unique_ptr<RTCDataChannelHandler>>& rtcDataChannelLocalMap() WTF_REQUIRES_LOCK(s_rtcDataChannelLocalMapLock)
261{
262 ASSERT(s_rtcDataChannelLocalMapLock.isHeld());

271bool RTCDataChannel::canDetach() const
272{
273 return m_isDetachable && !m_isDetached && !m_bufferedAmount && m_readyState != RTCDataChannelState::Closed;
274}

276std::unique_ptr<DetachedRTCDataChannel> RTCDataChannel::detach()
277{
278 ASSERT(canDetach());
279
280 auto state = m_readyState;
281
282 m_isDetachable = false;
283 m_isDetached = true;
284 m_readyState = RTCDataChannelState::Closed;
285
286 Locker locker { s_rtcDataChannelLocalMapLock };
287 rtcDataChannelLocalMap().add(identifier().channelIdentifier, WTFMove(m_handler));
288
289 return makeUnique<DetachedRTCDataChannel>(identifier(), label().isolatedCopy(), options(), state);
290}
291
292void RTCDataChannel::removeFromDataChannelLocalMapIfNeeded()
293{
294 if (!m_isDetached)
295 return;
296
297 Locker locker { s_rtcDataChannelLocalMapLock };
298 rtcDataChannelLocalMap().remove(identifier().channelIdentifier);
299}

301std::unique_ptr<RTCDataChannelHandler> RTCDataChannel::handlerFromIdentifier(RTCDataChannelLocalIdentifier channelIdentifier)
302{
303 Locker locker { s_rtcDataChannelLocalMapLock };
304 return rtcDataChannelLocalMap().take(channelIdentifier);

307static Ref<RTCDataChannel> createClosedChannel(ScriptExecutionContext& context, String&& label, RTCDataChannelInit&& options)
308{
309 auto channel = RTCDataChannel::create(context, nullptr, WTFMove(label), WTFMove(options));
310 channel->close();
311 return channel;

314Ref<RTCDataChannel> RTCDataChannel::create(ScriptExecutionContext& context, RTCDataChannelIdentifier identifier, String&& label, RTCDataChannelInit&& options, RTCDataChannelState state)
315{
316 RTCDataChannelRemoteHandler* remoteHandlerPtr = nullptr;
317 std::unique_ptr<RTCDataChannelHandler> handler;
318 if (identifier.processIdentifier == Process::identifier())
319 handler = RTCDataChannel::handlerFromIdentifier(identifier.channelIdentifier);
320 else {
321 auto remoteHandler = RTCDataChannelRemoteHandler::create(identifier, context.createRTCDataChannelRemoteHandlerConnection());
322 remoteHandlerPtr = remoteHandler.get();
323 handler = WTFMove(remoteHandler);
324 }
325
326 if (!handler)
327 return createClosedChannel(context, WTFMove(label), WTFMove(options));
328
329 auto channel = RTCDataChannel::create(context, WTFMove(handler), WTFMove(label), WTFMove(options));
330 channel->m_readyState = state;
331
332 if (remoteHandlerPtr)
333 remoteHandlerPtr->setLocalIdentifier(channel->identifier());
334
335 return channel;
336}
