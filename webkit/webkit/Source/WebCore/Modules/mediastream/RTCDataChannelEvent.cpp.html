38Ref<RTCDataChannelEvent> RTCDataChannelEvent::create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Ref<RTCDataChannel>&& channel)

40 return adoptRef(*new RTCDataChannelEvent(type, canBubble, cancelable, WTFMove(channel)));

43Ref<RTCDataChannelEvent> RTCDataChannelEvent::create(const AtomString& type, Init&& initializer, IsTrusted isTrusted)

46}

48RTCDataChannelEvent::RTCDataChannelEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Ref<RTCDataChannel>&& channel)
49 : Event(type, canBubble, cancelable)
50 , m_channel(WTFMove(channel))

52}

54RTCDataChannelEvent::RTCDataChannelEvent(const AtomString& type, Init&& initializer, IsTrusted isTrusted)
55 : Event(type, initializer, isTrusted)
56 , m_channel(initializer.channel.releaseNonNull())
57{
58}

60RTCDataChannel& RTCDataChannelEvent::channel()
