78void RTCDataChannelRemoteHandler::readyToSend()
79{
80 m_isReadyToSend = true;
81
82 for (auto& message : m_pendingMessages)
83 m_connection->sendData(m_remoteIdentifier, message.isRaw, message.buffer->data(), message.buffer->size());
84 m_pendingMessages.clear();
85
86 if (m_isPendingClose)
87 m_connection->close(m_remoteIdentifier);
88}
89
90void RTCDataChannelRemoteHandler::setClient(RTCDataChannelHandlerClient& client, ScriptExecutionContextIdentifier contextIdentifier)
91{
92 m_client = &client;
93 ASSERT(m_localIdentifier.channelIdentifier);
94 m_connection->connectToSource(*this, contextIdentifier, m_localIdentifier, m_remoteIdentifier);
95}
96
97bool RTCDataChannelRemoteHandler::sendStringData(const CString& text)
98{
99 if (!m_isReadyToSend) {
100 m_pendingMessages.append(Message { false, SharedBuffer::create(text.data(), text.length()) });
101 return true;
102 }
103 m_connection->sendData(m_remoteIdentifier, false, text.dataAsUInt8Ptr(), text.length());
104 return true;
105}
106
107bool RTCDataChannelRemoteHandler::sendRawData(const uint8_t* data, size_t size)
108{
109 if (!m_isReadyToSend) {
110 m_pendingMessages.append(Message { true, SharedBuffer::create(data, size) });
111 return true;
112 }
113 m_connection->sendData(m_remoteIdentifier, true, data, size);
114 return true;
115}
116
117void RTCDataChannelRemoteHandler::close()
118{
119 if (!m_isReadyToSend) {
120 m_isPendingClose = true;
121 return;
122 }
123 m_connection->close(m_remoteIdentifier);
124}
