45RTCIceCandidate::RTCIceCandidate(const String& candidate, const String& sdpMid, std::optional<unsigned short> sdpMLineIndex, Fields&& fields)
46 : m_candidate(candidate)
47 , m_sdpMid(sdpMid)
48 , m_sdpMLineIndex(sdpMLineIndex)
49 , m_fields(WTFMove(fields))
50{
51 ASSERT(!sdpMid.isNull() || sdpMLineIndex);
52}
53
54ExceptionOr<Ref<RTCIceCandidate>> RTCIceCandidate::create(const RTCIceCandidateInit& dictionary)
55{
56 if (dictionary.sdpMid.isNull() && !dictionary.sdpMLineIndex)
57 return Exception { TypeError, "Candidate must not have both null sdpMid and sdpMLineIndex" };
58
59 auto fields = parseIceCandidateSDP(dictionary.candidate).value_or(RTCIceCandidate::Fields { });
60 fields.usernameFragment = dictionary.usernameFragment;
61 return adoptRef(*new RTCIceCandidate(dictionary.candidate, dictionary.sdpMid, dictionary.sdpMLineIndex, WTFMove(fields)));
62}
63
64Ref<RTCIceCandidate> RTCIceCandidate::create(const String& candidate, const String& sdpMid, std::optional<unsigned short> sdpMLineIndex)
65{
66 auto fields = parseIceCandidateSDP(candidate);
67 return adoptRef(*new RTCIceCandidate(candidate, sdpMid, sdpMLineIndex, WTFMove(*fields)));
68}

70RTCIceCandidateInit RTCIceCandidate::toJSON() const
71{
72 RTCIceCandidateInit result;
73 result.candidate = m_candidate;
74 result.sdpMid = m_sdpMid;
75 result.sdpMLineIndex = m_sdpMLineIndex;
76 return result;
77}
