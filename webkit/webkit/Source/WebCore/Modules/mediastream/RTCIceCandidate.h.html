42namespace WebCore {
43
44struct RTCIceCandidateInit;
45
46class RTCIceCandidate final : public RefCounted<RTCIceCandidate>, public ScriptWrappable {
47 WTF_MAKE_ISO_ALLOCATED(RTCIceCandidate);
48public:
49 static ExceptionOr<Ref<RTCIceCandidate>> create(const RTCIceCandidateInit&);
50 static Ref<RTCIceCandidate> create(const String& candidate, const String& sdpMid, std::optional<unsigned short> sdpMLineIndex);
51
52 const String& candidate() const { return m_candidate; }
53 const String& sdpMid() const { return m_sdpMid; }
54 std::optional<unsigned short> sdpMLineIndex() const { return m_sdpMLineIndex; }
55
56 struct Fields {
57 String foundation;
58 std::optional<RTCIceComponent> component;
59 std::optional<unsigned> priority;
60 String address;
61 std::optional<RTCIceProtocol> protocol;
62 std::optional<unsigned short> port;
63 std::optional<RTCIceCandidateType> type;
64 std::optional<RTCIceTcpCandidateType> tcpType;
65 String relatedAddress;
66 std::optional<unsigned short> relatedPort;
67 String usernameFragment;
68 };
69
70 String foundation() const { return m_fields.foundation; }
71 std::optional<RTCIceComponent> component() const { return m_fields.component; }
72 std::optional<unsigned> priority() const { return m_fields.priority; }
73 String address() const { return m_fields.address; }
74 std::optional<RTCIceProtocol> protocol() const { return m_fields.protocol; }
75 std::optional<unsigned short> port() const { return m_fields.port; }
76 std::optional<RTCIceCandidateType> type() const { return m_fields.type; }
77 std::optional<RTCIceTcpCandidateType> tcpType() const { return m_fields.tcpType; }
78 String relatedAddress() const { return m_fields.relatedAddress; }
79 std::optional<unsigned short> relatedPort() const { return m_fields.relatedPort; }
80 String usernameFragment() const { return m_fields.usernameFragment; }
81
82 RTCIceCandidateInit toJSON() const;
83
84private:
85 RTCIceCandidate(const String& candidate, const String& sdpMid, std::optional<unsigned short> sdpMLineIndex, Fields&&);
86
87 String m_candidate;
88 String m_sdpMid;
89 std::optional<unsigned short> m_sdpMLineIndex;
90 Fields m_fields;
91};
