74ExceptionOr<Ref<RTCPeerConnection>> RTCPeerConnection::create(Document& document, RTCConfiguration&& configuration)
75{
76 if (!document.frame())
77 return Exception { NotSupportedError };
78
79 auto peerConnection = adoptRef(*new RTCPeerConnection(document));
80 peerConnection->suspendIfNeeded();
81
82 if (!peerConnection->m_backend)
83 return Exception { NotSupportedError };
84
85 auto exception = peerConnection->initializeConfiguration(WTFMove(configuration));
86 if (exception.hasException())
87 return exception.releaseException();
88
89 if (!peerConnection->isClosed()) {
90 if (auto* page = document.page()) {
91 peerConnection->registerToController(page->rtcController());
92#if !LOG_DISABLED || !RELEASE_LOG_DISABLED
93 if (!page->sessionID().isEphemeral())
94 page->libWebRTCProvider().setLoggingLevel(LogWebRTC.level);
95#endif
96 }
97 }
98 return peerConnection;
99}

128ExceptionOr<Ref<RTCRtpSender>> RTCPeerConnection::addTrack(Ref<MediaStreamTrack>&& track, const Vector<std::reference_wrapper<MediaStream>>& streams)
129{
130 INFO_LOG(LOGIDENTIFIER);
131
132 if (isClosed())
133 return Exception { InvalidStateError };
134
135 for (auto& transceiver : m_transceiverSet->list()) {
136 if (transceiver->sender().trackId() == track->id())
137 return Exception { InvalidAccessError };
138 }
139
140 Vector<String> mediaStreamIds;
141 for (auto stream : streams)
142 mediaStreamIds.append(stream.get().id());
143
144 return m_backend->addTrack(track.get(), WTFMove(mediaStreamIds));
145}

147ExceptionOr<void> RTCPeerConnection::removeTrack(RTCRtpSender& sender)
148{
149 INFO_LOG(LOGIDENTIFIER);
150
151 if (isClosed())
152 return Exception { InvalidStateError, "RTCPeerConnection is closed"_s };
153
154 if (!sender.isCreatedBy(*this))
155 return Exception { InvalidAccessError, "RTCPeerConnection did not create the given sender"_s };
156
157 bool shouldAbort = true;
158 RTCRtpTransceiver* senderTransceiver = nullptr;
159 for (auto& transceiver : m_transceiverSet->list()) {
160 if (&sender == &transceiver->sender()) {
161 senderTransceiver = transceiver.get();
162 shouldAbort = sender.isStopped() || !sender.track();
163 break;
164 }
165 }
166 if (shouldAbort)
167 return { };
168
169 sender.setTrackToNull();
170 senderTransceiver->disableSendingDirection();
171 m_backend->removeTrack(sender);
172 return { };
173}

299void RTCPeerConnection::addIceCandidate(Candidate&& rtcCandidate, Ref<DeferredPromise>&& promise)
300{
301 std::optional<Exception> exception;
302 RefPtr<RTCIceCandidate> candidate;
303 if (rtcCandidate) {
304 candidate = switchOn(*rtcCandidate, [&exception](RTCIceCandidateInit& init) -> RefPtr<RTCIceCandidate> {
305 if (init.candidate.isEmpty())
306 return nullptr;
307
308 auto result = RTCIceCandidate::create(WTFMove(init));
309 if (result.hasException()) {
310 exception = result.releaseException();
311 return nullptr;
312 }
313 return result.releaseReturnValue();
314 }, [](RefPtr<RTCIceCandidate>& iceCandidate) {
315 return WTFMove(iceCandidate);
316 });
317 }
318
319 if (exception) {
320 promise->reject(*exception);
321 return;
322 }
323
324 if (isClosed()) {
325 promise->reject(InvalidStateError);
326 return;
327 }
328
329 ALWAYS_LOG(LOGIDENTIFIER, "Received ice candidate:\n", candidate ? candidate->candidate() : "null");
330 chainOperation(WTFMove(promise), [this, candidate = WTFMove(candidate)](auto&& promise) mutable {
331 m_backend->addIceCandidate(candidate.get(), WTFMove(promise));
332 });
333}

335// Implementation of https://w3c.github.io/webrtc-pc/#set-pc-configuration
336ExceptionOr<Vector<MediaEndpointConfiguration::IceServerInfo>> RTCPeerConnection::iceServersFromConfiguration(RTCConfiguration& newConfiguration, const RTCConfiguration* existingConfiguration, bool isLocalDescriptionSet)
337{
338 if (existingConfiguration && newConfiguration.bundlePolicy != existingConfiguration->bundlePolicy)
339 return Exception { InvalidModificationError, "BundlePolicy does not match existing policy" };
340
341 if (existingConfiguration && newConfiguration.rtcpMuxPolicy != existingConfiguration->rtcpMuxPolicy)
342 return Exception { InvalidModificationError, "RTCPMuxPolicy does not match existing policy" };
343
344 if (existingConfiguration && newConfiguration.iceCandidatePoolSize != existingConfiguration->iceCandidatePoolSize && isLocalDescriptionSet)
345 return Exception { InvalidModificationError, "IceTransportPolicy pool size does not match existing pool size" };
346
347 Vector<MediaEndpointConfiguration::IceServerInfo> servers;
348 if (newConfiguration.iceServers) {
349 servers.reserveInitialCapacity(newConfiguration.iceServers->size());
350 for (auto& server : newConfiguration.iceServers.value()) {
351 Vector<String> urls;
352 WTF::switchOn(server.urls, [&urls] (String& url) {
353 urls = { WTFMove(url) };
354 }, [&urls] (Vector<String>& vector) {
355 urls = WTFMove(vector);
356 });
357
358 urls.removeAllMatching([&](auto& urlString) {
359 URL url { URL { }, urlString };
360 if (url.path().endsWithIgnoringASCIICase(".local") || !portAllowed(url)) {
361 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, RTCPeerConnectionIceErrorEvent::create(Event::CanBubble::No, Event::IsCancelable::No, { }, { }, WTFMove(urlString), 701, "URL is not allowed"_s));
362 return true;
363 }
364 return false;
365 });
366
367 auto serverURLs = WTF::map(urls, [](auto& url) -> URL {
368 return { URL { }, url };
369 });
370 server.urls = WTFMove(urls);
371
372 for (auto& serverURL : serverURLs) {
373 if (serverURL.isNull())
374 return Exception { TypeError, "Bad ICE server URL" };
375 if (serverURL.protocolIs("turn") || serverURL.protocolIs("turns")) {
376 if (server.credential.isNull() || server.username.isNull())
377 return Exception { InvalidAccessError, "TURN/TURNS server requires both username and credential" };
378 // https://tools.ietf.org/html/rfc8489#section-14.3
379 if (server.credential.length() > 64 || server.username.length() > 64) {
380 constexpr size_t MaxTurnUsernameLength = 509;
381 if (server.credential.utf8().length() > MaxTurnUsernameLength || server.username.utf8().length() > MaxTurnUsernameLength)
382 return Exception { TypeError, "TURN/TURNS username and/or credential are too long" };
383 }
384 } else if (!serverURL.protocolIs("stun"))
385 return Exception { NotSupportedError, "ICE server protocol not supported" };
386 }
387 if (serverURLs.size())
388 servers.uncheckedAppend({ WTFMove(serverURLs), server.credential, server.username });
389 }
390 }
391 return servers;
392}

394ExceptionOr<Vector<MediaEndpointConfiguration::CertificatePEM>> RTCPeerConnection::certificatesFromConfiguration(const RTCConfiguration& configuration)
395{
396 auto currentMilliSeconds = WallTime::now().secondsSinceEpoch().milliseconds();
397 auto& origin = document()->securityOrigin();
398
399 Vector<MediaEndpointConfiguration::CertificatePEM> certificates;
400 certificates.reserveInitialCapacity(configuration.certificates.size());
401 for (auto& certificate : configuration.certificates) {
402 if (!origin.isSameOriginAs(certificate->origin()))
403 return Exception { InvalidAccessError, "Certificate does not have a valid origin" };
404
405 if (currentMilliSeconds > certificate->expires())
406 return Exception { InvalidAccessError, "Certificate has expired"_s };
407
408 certificates.uncheckedAppend(MediaEndpointConfiguration::CertificatePEM { certificate->pemCertificate(), certificate->pemPrivateKey(), });
409 }
410 return certificates;
411}

413ExceptionOr<void> RTCPeerConnection::initializeConfiguration(RTCConfiguration&& configuration)
414{
415 INFO_LOG(LOGIDENTIFIER);
416
417 auto servers = iceServersFromConfiguration(configuration, nullptr, false);
418 if (servers.hasException())
419 return servers.releaseException();
420
421 auto certificates = certificatesFromConfiguration(configuration);
422 if (certificates.hasException())
423 return certificates.releaseException();
424
425 if (!m_backend->setConfiguration({ servers.releaseReturnValue(), configuration.iceTransportPolicy, configuration.bundlePolicy, configuration.rtcpMuxPolicy, configuration.iceCandidatePoolSize, certificates.releaseReturnValue() }))
426 return Exception { InvalidAccessError, "Bad Configuration Parameters" };
427
428 m_configuration = WTFMove(configuration);
429 return { };
430}
431
432ExceptionOr<void> RTCPeerConnection::setConfiguration(RTCConfiguration&& configuration)
433{
434 if (isClosed())
435 return Exception { InvalidStateError };
436
437 INFO_LOG(LOGIDENTIFIER);
438
439 auto servers = iceServersFromConfiguration(configuration, &m_configuration, m_backend->isLocalDescriptionSet());
440 if (servers.hasException())
441 return servers.releaseException();
442
443 if (configuration.certificates.size()) {
444 if (configuration.certificates.size() != m_configuration.certificates.size())
445 return Exception { InvalidModificationError, "Certificates parameters are different" };
446
447 for (auto& certificate : configuration.certificates) {
448 bool isThere = m_configuration.certificates.findMatching([&certificate](const auto& item) {
449 return item.get() == certificate.get();
450 }) != notFound;
451 if (!isThere)
452 return Exception { InvalidModificationError, "A certificate given in constructor is not present" };
453 }
454 }
455
456 if (!m_backend->setConfiguration({ servers.releaseReturnValue(), configuration.iceTransportPolicy, configuration.bundlePolicy, configuration.rtcpMuxPolicy, configuration.iceCandidatePoolSize, { } }))
457 return Exception { InvalidAccessError, "Bad Configuration Parameters" };
458
459 m_configuration = WTFMove(configuration);
460 return { };
461}

481ExceptionOr<Ref<RTCDataChannel>> RTCPeerConnection::createDataChannel(String&& label, RTCDataChannelInit&& options)
482{
483 ALWAYS_LOG(LOGIDENTIFIER);
484
485 if (isClosed())
486 return Exception { InvalidStateError };
487
488 if (options.negotiated && !options.negotiated.value() && (label.length() > 65535 || options.protocol.length() > 65535))
489 return Exception { TypeError };
490
491 if (options.maxPacketLifeTime && options.maxRetransmits)
492 return Exception { TypeError };
493
494 if (options.id && options.id.value() > 65534)
495 return Exception { TypeError };
496
497 auto channelHandler = m_backend->createDataChannelHandler(label, options);
498 if (!channelHandler)
499 return Exception { NotSupportedError };
500
501 return RTCDataChannel::create(*document(), WTFMove(channelHandler), WTFMove(label), WTFMove(options));
502}

504bool RTCPeerConnection::doClose()
505{
506 if (isClosed())
507 return false;
508
509 m_shouldDelayTasks = false;
510 m_connectionState = RTCPeerConnectionState::Closed;
511 m_iceConnectionState = RTCIceConnectionState::Closed;
512 m_signalingState = RTCSignalingState::Closed;
513
514 for (auto& transceiver : m_transceiverSet->list()) {
515 transceiver->stop();
516 transceiver->sender().stop();
517 transceiver->receiver().stop();
518 }
519 m_operations.clear();
520
521 return true;
522}

531 m_backend->close();

648void RTCPeerConnection::updateConnectionState()
649{
650 RTCPeerConnectionState state;
651
652 if (m_iceConnectionState == RTCIceConnectionState::Closed)
653 state = RTCPeerConnectionState::Closed;
654 else if (m_iceConnectionState == RTCIceConnectionState::Disconnected)
655 state = RTCPeerConnectionState::Disconnected;
656 else if (m_iceConnectionState == RTCIceConnectionState::Failed)
657 state = RTCPeerConnectionState::Failed;
658 else if (m_iceConnectionState == RTCIceConnectionState::New && m_iceGatheringState == RTCIceGatheringState::New)
659 state = RTCPeerConnectionState::New;
660 else if (m_iceConnectionState == RTCIceConnectionState::Checking || m_iceGatheringState == RTCIceGatheringState::Gathering)
661 state = RTCPeerConnectionState::Connecting;
662 else if ((m_iceConnectionState == RTCIceConnectionState::Completed || m_iceConnectionState == RTCIceConnectionState::Connected) && m_iceGatheringState == RTCIceGatheringState::Complete)
663 state = RTCPeerConnectionState::Connected;
664 else
665 return;
666
667 if (state == m_connectionState)
668 return;
669
670 INFO_LOG(LOGIDENTIFIER, "state changed from: " , m_connectionState, " to ", state);
671
672 m_connectionState = state;
673 dispatchEventWhenFeasible(Event::create(eventNames().connectionstatechangeEvent, Event::CanBubble::No, Event::IsCancelable::No));
674}

676void RTCPeerConnection::scheduleNegotiationNeededEvent()

691 m_backend->clearNegotiationNeededState();
692 dispatchEventWhenFeasible(Event::create(eventNames().negotiationneededEvent, Event::CanBubble::No, Event::IsCancelable::No));
693 });
694}

718static inline ExceptionOr<PeerConnectionBackend::CertificateInformation> certificateTypeFromAlgorithmIdentifier(JSC::JSGlobalObject& lexicalGlobalObject, RTCPeerConnection::AlgorithmIdentifier&& algorithmIdentifier)
719{
720 if (WTF::holds_alternative<String>(algorithmIdentifier))
721 return Exception { NotSupportedError, "Algorithm is not supported"_s };
722
723 auto& value = WTF::get<JSC::Strong<JSC::JSObject>>(algorithmIdentifier);
724
725 JSC::VM& vm = lexicalGlobalObject.vm();
726 auto scope = DECLARE_CATCH_SCOPE(vm);
727
728 auto parameters = convertDictionary<RTCPeerConnection::CertificateParameters>(lexicalGlobalObject, value.get());
729 if (UNLIKELY(scope.exception())) {
730 scope.clearException();
731 return Exception { TypeError, "Unable to read certificate parameters"_s };
732 }
733
734 if (parameters.expires && *parameters.expires < 0)
735 return Exception { TypeError, "Expire value is invalid"_s };
736
737 if (parameters.name == "RSASSA-PKCS1-v1_5"_s) {
738 if (!parameters.hash.isNull() && parameters.hash != "SHA-256"_s)
739 return Exception { NotSupportedError, "Only SHA-256 is supported for RSASSA-PKCS1-v1_5"_s };
740
741 auto result = PeerConnectionBackend::CertificateInformation::RSASSA_PKCS1_v1_5();
742 if (parameters.modulusLength && parameters.publicExponent) {
743 int publicExponent = 0;
744 int value = 1;
745 for (unsigned counter = 0; counter < parameters.publicExponent->byteLength(); ++counter) {
746 publicExponent += parameters.publicExponent->data()[counter] * value;
747 value <<= 8;
748 }
749
750 result.rsaParameters = PeerConnectionBackend::CertificateInformation::RSA { *parameters.modulusLength, publicExponent };
751 }
752 result.expires = parameters.expires;
753 return result;
754 }
755 if (parameters.name == "ECDSA"_s && parameters.namedCurve == "P-256"_s) {
756 auto result = PeerConnectionBackend::CertificateInformation::ECDSA_P256();
757 result.expires = parameters.expires;
758 return result;
759 }
760
761 return Exception { NotSupportedError, "Algorithm is not supported"_s };
762}

793void RTCPeerConnection::chainOperation(Ref<DeferredPromise>&& promise, Function<void(Ref<DeferredPromise>&&)>&& operation)
794{
795 if (isClosed()) {
796 promise->reject(InvalidStateError, "RTCPeerConnection is closed");
797 return;
798 }
799
800 promise->whenSettled([this, pendingActivity = makePendingActivity(*this)] {
801 ASSERT(m_hasPendingOperation);
802 if (isClosed()) {
803 for (auto& operation : std::exchange(m_operations, { }))
804 operation.first->reject(InvalidStateError, "RTCPeerConnection is closed");
805 m_hasPendingOperation = false;
806 return;
807 }
808
809 if (m_operations.isEmpty()) {
810 m_hasPendingOperation = false;
811 if (m_shouldFireNegotiationNeededOnceOperationChainIsEmpty) {
812 m_shouldFireNegotiationNeededOnceOperationChainIsEmpty = false;
813 scheduleNegotiationNeededEvent();
814 }
815 return;
816 }
817
818 auto promiseOperation = m_operations.takeFirst();
819 promiseOperation.second(WTFMove(promiseOperation.first));
820 });
821
822 if (m_hasPendingOperation || !m_operations.isEmpty()) {
823 m_operations.append(std::make_pair(WTFMove(promise), WTFMove(operation)));
824 return;
825 }
826
827 m_hasPendingOperation = true;
828 operation(WTFMove(promise));
829}
