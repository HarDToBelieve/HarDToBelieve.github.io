31#pragma once
32
33#if ENABLE(WEB_RTC)
34
35#include "MediaStreamTrack.h"
36#include "RTCRtpReceiverBackend.h"
37#include "RTCRtpSynchronizationSource.h"
38#include "RTCRtpTransform.h"
39#include "ScriptWrappable.h"
40
41namespace WebCore {
42
43class DeferredPromise;
44class PeerConnectionBackend;
45struct RTCRtpCapabilities;
46
47class RTCRtpReceiver final : public RefCounted<RTCRtpReceiver>
48 , public ScriptWrappable
49#if !RELEASE_LOG_DISABLED
50 , private LoggerHelper
51#endif
52 {
53 WTF_MAKE_ISO_ALLOCATED(RTCRtpReceiver);
54public:
55 static Ref<RTCRtpReceiver> create(PeerConnectionBackend& connection, Ref<MediaStreamTrack>&& track, std::unique_ptr<RTCRtpReceiverBackend>&& backend)
56 {
57 return adoptRef(*new RTCRtpReceiver(connection, WTFMove(track), WTFMove(backend)));
58 }
59 ~RTCRtpReceiver();
60
61 static std::optional<RTCRtpCapabilities> getCapabilities(ScriptExecutionContext&, const String& kind);
62
63 void stop();
64
65 void setBackend(std::unique_ptr<RTCRtpReceiverBackend>&& backend) { m_backend = WTFMove(backend); }
66 RTCRtpParameters getParameters() { return m_backend ? m_backend->getParameters() : RTCRtpParameters(); }
67 Vector<RTCRtpContributingSource> getContributingSources() const { return m_backend ? m_backend->getContributingSources() : Vector<RTCRtpContributingSource> { }; }
68 Vector<RTCRtpSynchronizationSource> getSynchronizationSources() const { return m_backend ? m_backend->getSynchronizationSources() : Vector<RTCRtpSynchronizationSource> { }; }
69
70 MediaStreamTrack& track() { return m_track.get(); }
71
72 RTCRtpReceiverBackend* backend() { return m_backend.get(); }
73 void getStats(Ref<DeferredPromise>&&);
74
75 std::optional<RTCRtpTransform::Internal> transform();
76 ExceptionOr<void> setTransform(std::unique_ptr<RTCRtpTransform>&&);
77
78private:
79 RTCRtpReceiver(PeerConnectionBackend&, Ref<MediaStreamTrack>&&, std::unique_ptr<RTCRtpReceiverBackend>&&);
80
81#if !RELEASE_LOG_DISABLED
82 const Logger& logger() const final { return m_logger.get(); }
83 const void* logIdentifier() const final { return m_logIdentifier; }
84 const char* logClassName() const final { return "RTCRtpReceiver"; }
85 WTFLogChannel& logChannel() const final;
86#endif
87
88 Ref<MediaStreamTrack> m_track;
89 std::unique_ptr<RTCRtpReceiverBackend> m_backend;
90 WeakPtr<PeerConnectionBackend> m_connection;
91 std::unique_ptr<RTCRtpTransform> m_transform;
92#if !RELEASE_LOG_DISABLED
93 Ref<const Logger> m_logger;
94 const void* m_logIdentifier { nullptr };
95#endif
96};
97
98} // namespace WebCore
99
100#endif // ENABLE(WEB_RTC)
