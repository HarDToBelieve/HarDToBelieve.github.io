35static constexpr unsigned AES_CM_128_HMAC_SHA256_NONCE_SIZE = 12;
36
37static inline void writeUInt64(uint8_t* data, uint64_t value, uint8_t valueLength)
38{
39 for (unsigned i = 0; i < valueLength; ++i)
40 *data++ = (value >> ((valueLength - 1 - i) * 8)) & 0xff;
41}
42
43static inline uint64_t readUInt64(const uint8_t* data, size_t size)
44{
45 uint64_t value = 0;
46 while (size--)
47 value = (value << 8) | *data++;
48 return value;
49}
50
51static inline uint8_t lengthOfUInt64(uint64_t value)
52{
53 uint8_t length = 0;
54 do {
55 ++length;
56 value = value >> 8;
57 } while (value);
58 return length;
59}
60
61static inline uint8_t computeFirstHeaderByte(uint64_t keyId, uint64_t counter)
62{
63 uint8_t value = 0;
64 value |= (lengthOfUInt64(counter) - 1) << 4;
65 if (keyId < 8)
66 value |= keyId;
67 else {
68 value |= (lengthOfUInt64(keyId) - 1);
69 value |= 1 << 3;
70 }
71 return value;
72}
73
74static inline Vector<uint8_t> computeIV(uint64_t counter, const Vector<uint8_t>& saltKey)
75{
76 // The saltKey is 12 bytes (AES_CM_128_HMAC_SHA256_NONCE_SIZE), we XOR with the counter as 12 bytes.
77 // We then extend it to 16 bytes since that is what is expected by the crypto routines.
78 Vector<uint8_t> iv(16);
79 for (unsigned i = 0; i < 4; ++i)
80 iv[i] = saltKey[i];
81 for (unsigned i = 11; i >= 4; --i) {
82 auto value = counter & 0xff;
83 counter = counter >> 8;
84 iv[i] = value ^ saltKey[i];
85 }
86 for (unsigned i = 12; i < 16; ++i)
87 iv[i] = 0;
88 return iv;
89}

91static inline bool hasSignature(uint8_t firstByte)
92{
93 return firstByte & 0x80;
94}
95
96static inline bool hasLongKeyLength(uint8_t firstByte)
97{
98 return firstByte & 0x08;
99}

101struct SFrameHeaderInfo {
102 uint8_t size;
103 uint64_t keyId;
104 uint64_t counter;
105};
106static inline std::optional<SFrameHeaderInfo> parseSFrameHeader(const uint8_t* data, size_t size)
107{
108 auto* start = data;
109
110 uint64_t keyId = 0;
111 uint64_t counter = 0;
112
113 auto firstByte = *data++;
114
115 // Signature bit.
116 if (hasSignature(firstByte))
117 return { };
118
119 size_t counterLength = ((firstByte >> 4) & 0x07) + 1;
120
121 if (size < counterLength + 1)
122 return { };
123
124 if (hasLongKeyLength(firstByte)) {
125 size_t keyLength = (firstByte & 0x07) + 1;
126 if (size < counterLength + keyLength + 1)
127 return { };
128
129 keyId = readUInt64(data, keyLength);
130 data += keyLength;
131
132 counter = readUInt64(data, counterLength);
133 data += counterLength;
134 } else {
135 keyId = firstByte & 0x07;
136 counter = readUInt64(data, counterLength);
137 data += counterLength;
138 }
139 uint8_t headerSize = data - start;
140 return SFrameHeaderInfo { headerSize, keyId, counter };
141}

163ExceptionOr<void> RTCRtpSFrameTransformer::updateEncryptionKey(const Vector<uint8_t>& rawKey, std::optional<uint64_t> keyId, ShouldUpdateKeys shouldUpdateKeys)
164{
165 ASSERT(m_keyLock.isLocked());
166
167 auto saltKeyResult = computeSaltKey(rawKey);
168 if (saltKeyResult.hasException())
169 return saltKeyResult.releaseException();
170
171 ASSERT(saltKeyResult.returnValue().size() >= AES_CM_128_HMAC_SHA256_NONCE_SIZE);
172
173 auto authenticationKeyResult = computeAuthenticationKey(rawKey);
174 if (authenticationKeyResult.hasException())
175 return authenticationKeyResult.releaseException();
176
177 auto encryptionKeyResult = computeEncryptionKey(rawKey);
178 if (encryptionKeyResult.hasException())
179 return encryptionKeyResult.releaseException();
180
181 if (shouldUpdateKeys == ShouldUpdateKeys::No)
182 m_keyId = *keyId;
183 else {
184 // FIXME: In case keyId is not set, it might be best to use the first non used ID.
185 if (!keyId)
186 keyId = m_keys.size();
187
188 m_keyId = *keyId;
189 m_keys.append({ m_keyId, rawKey });
190 }
191
192 m_saltKey = saltKeyResult.releaseReturnValue();
193 m_authenticationKey = authenticationKeyResult.releaseReturnValue();
194 m_encryptionKey = encryptionKeyResult.releaseReturnValue();
195
196 updateAuthenticationSize();
197 m_hasKey = true;
198
199 return { };
200}

202ExceptionOr<Vector<uint8_t>> RTCRtpSFrameTransformer::decryptFrame(const uint8_t* frameData, size_t frameSize)
203{
204 Vector<uint8_t> buffer;
205 switch (m_compatibilityMode) {
206 case CompatibilityMode::H264: {
207 auto offset = computeH264PrefixOffset(frameData, frameSize);
208 frameData += offset;
209 frameSize -= offset;
210 if (needsRbspUnescaping(frameData, frameSize)) {
211 buffer = fromRbsp(frameData, frameSize);
212 frameData = buffer.data();
213 frameSize = buffer.size();
214 }
215 break;
216 }
217 case CompatibilityMode::VP8: {
218 auto offset = computeVP8PrefixOffset(frameData, frameSize);
219 frameData += offset;
220 frameSize -= offset;
221 break;
222 }
223 case CompatibilityMode::None:
224 break;
225 }
226
227 Locker locker { m_keyLock };
228
229 auto header = parseSFrameHeader(frameData, frameSize);
230
231 if (!header)
232 return Exception { NotSupportedError };
233
234 if (header->counter <= m_counter && m_counter)
235 return Exception { InvalidStateError };
236 m_counter = header->counter;
237
238 if (header->keyId != m_keyId) {
239 auto position = m_keys.findMatching([keyId = header->keyId](auto& item) { return item.keyId == keyId; });
240 if (position == notFound)
241 return Exception { DataError, "Key ID is unknown" };
242 auto result = updateEncryptionKey(m_keys[position].keyData, header->keyId, ShouldUpdateKeys::No);
243 if (result.hasException())
244 return result.releaseException();
245 }
246
247 if (frameSize < (header->size + m_authenticationSize))
248 return Exception { DataError, "Chunk is too small for authentication size" };
249
250 auto iv = computeIV(m_counter, m_saltKey);
251
252 // Compute signature
253 auto* transmittedSignature = frameData + frameSize - m_authenticationSize;
254 auto signature = computeEncryptedDataSignature(iv, frameData, header->size, frameData + header->size, frameSize - m_authenticationSize - header->size, m_authenticationKey);
255 for (size_t cptr = 0; cptr < m_authenticationSize; ++cptr) {
256 if (signature[cptr] != transmittedSignature[cptr]) {
257 // FIXME: We should try ratcheting.
258 return Exception { NotSupportedError };
259 }
260 }
261
262 // Decrypt data
263 auto dataSize = frameSize - header->size - m_authenticationSize;
264 auto result = decryptData(frameData + header->size, dataSize, iv, m_encryptionKey);
265
266 if (result.hasException())
267 return result.releaseException();
268
269 return result.releaseReturnValue();
270}

272ExceptionOr<Vector<uint8_t>> RTCRtpSFrameTransformer::encryptFrame(const uint8_t* frameData, size_t frameSize)
273{
274 static const unsigned MaxHeaderSize = 17;
275
276 Vector<uint8_t> transformedData;
277 SFrameCompatibilityPrefixBuffer prefixBuffer;
278 switch (m_compatibilityMode) {
279 case CompatibilityMode::H264:
280 prefixBuffer = computeH264PrefixBuffer(frameData, frameSize);
281 break;
282 case CompatibilityMode::VP8:
283 prefixBuffer = computeVP8PrefixBuffer(frameData, frameSize);
284 break;
285 case CompatibilityMode::None:
286 break;
287 }
288
289 Locker locker { m_keyLock };
290
291 auto iv = computeIV(m_counter, m_saltKey);
292
293 transformedData.resize(prefixBuffer.size + frameSize + MaxHeaderSize + m_authenticationSize);
294
295 if (prefixBuffer.data)
296 std::memcpy(transformedData.data(), prefixBuffer.data, prefixBuffer.size);
297
298 auto* newDataPointer = transformedData.data() + prefixBuffer.size;
299 // Fill header.
300 size_t headerSize = 1;
301 *newDataPointer = computeFirstHeaderByte(m_keyId, m_counter);
302 if (m_keyId >= 8) {
303 auto keyIdLength = lengthOfUInt64(m_keyId);
304 writeUInt64(newDataPointer + headerSize, m_keyId, keyIdLength);
305 headerSize += keyIdLength;
306 }
307 auto counterLength = lengthOfUInt64(m_counter);
308 writeUInt64(newDataPointer + headerSize, m_counter, counterLength);
309 headerSize += counterLength;
310
311 transformedData.resize(prefixBuffer.size + frameSize + headerSize + m_authenticationSize);
312
313 // Fill encrypted data
314 auto encryptedData = encryptData(frameData, frameSize, iv, m_encryptionKey);
315 ASSERT(!encryptedData.hasException());
316 if (encryptedData.hasException())
317 return encryptedData.releaseException();
318
319 std::memcpy(newDataPointer + headerSize, encryptedData.returnValue().data(), frameSize);
320
321 // Fill signature
322 auto signature = computeEncryptedDataSignature(iv, newDataPointer, headerSize, newDataPointer + headerSize, frameSize, m_authenticationKey);
323 std::memcpy(newDataPointer + frameSize + headerSize, signature.data(), m_authenticationSize);
324
325 if (m_compatibilityMode == CompatibilityMode::H264)
326 toRbsp(transformedData, prefixBuffer.size);
327
328 ++m_counter;
329
330 return transformedData;
331}

333ExceptionOr<Vector<uint8_t>> RTCRtpSFrameTransformer::transform(const uint8_t* data, size_t size)
334{
335 if (!m_hasKey)
336 return Exception { InvalidStateError, "Key is not initialized"_s };
337
338 return m_isEncrypting ? encryptFrame(data, size) : decryptFrame(data, size);
339}
