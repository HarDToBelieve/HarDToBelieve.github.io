27
28#if ENABLE(WEB_RTC)
29
30#include "ExceptionOr.h"
31#include "RTCRtpTransformBackend.h"
32#include <wtf/Lock.h>
33#include <wtf/ThreadSafeRefCounted.h>
34
35namespace WebCore {
36
37class CryptoKey;
38
39class RTCRtpSFrameTransformer : public ThreadSafeRefCounted<RTCRtpSFrameTransformer, WTF::DestructionThread::Main> {
40public:
41 enum class CompatibilityMode { None, H264, VP8 };
42
43 WEBCORE_EXPORT static Ref<RTCRtpSFrameTransformer> create(CompatibilityMode = CompatibilityMode::None);
44 WEBCORE_EXPORT ~RTCRtpSFrameTransformer();
45
46 void setIsEncrypting(bool);
47 void setAuthenticationSize(uint64_t);
48 void setMediaType(RTCRtpTransformBackend::MediaType);
49
50 WEBCORE_EXPORT ExceptionOr<void> setEncryptionKey(const Vector<uint8_t>& rawKey, std::optional<uint64_t>);
51 WEBCORE_EXPORT ExceptionOr<Vector<uint8_t>> transform(const uint8_t*, size_t);
52
53 const Vector<uint8_t>& authenticationKey() const { return m_authenticationKey; }
54 const Vector<uint8_t>& encryptionKey() const { return m_encryptionKey; }
55 const Vector<uint8_t>& saltKey() const { return m_saltKey; }
56
57 uint64_t keyId() const { return m_keyId; }
58 uint64_t counter() const { return m_counter; }
59 void setCounter(uint64_t counter) { m_counter = counter; }
60
61private:
62 WEBCORE_EXPORT explicit RTCRtpSFrameTransformer(CompatibilityMode);
63
64 ExceptionOr<Vector<uint8_t>> decryptFrame(const uint8_t*, size_t);
65 ExceptionOr<Vector<uint8_t>> encryptFrame(const uint8_t*, size_t);
66
67 enum class ShouldUpdateKeys { No, Yes };
68 ExceptionOr<void> updateEncryptionKey(const Vector<uint8_t>& rawKey, std::optional<uint64_t>, ShouldUpdateKeys = ShouldUpdateKeys::Yes) WTF_REQUIRES_LOCK(m_keyLock);
69
70 ExceptionOr<Vector<uint8_t>> computeSaltKey(const Vector<uint8_t>&);
71 ExceptionOr<Vector<uint8_t>> computeAuthenticationKey(const Vector<uint8_t>&);
72 ExceptionOr<Vector<uint8_t>> computeEncryptionKey(const Vector<uint8_t>&);
73
74 ExceptionOr<Vector<uint8_t>> encryptData(const uint8_t*, size_t, const Vector<uint8_t>& iv, const Vector<uint8_t>& key);
75 ExceptionOr<Vector<uint8_t>> decryptData(const uint8_t*, size_t, const Vector<uint8_t>& iv, const Vector<uint8_t>& key);
76 Vector<uint8_t> computeEncryptedDataSignature(const Vector<uint8_t>& nonce, const uint8_t* header, size_t headerSize, const uint8_t* data, size_t dataSize, const Vector<uint8_t>& key);
77 void updateAuthenticationSize();
78
79 Lock m_keyLock;
80 bool m_hasKey { false };
81 Vector<uint8_t> m_authenticationKey;
82 Vector<uint8_t> m_encryptionKey;
83 Vector<uint8_t> m_saltKey;
84
85 struct IdentifiedKey {
86 uint64_t keyId { 0 };
87 Vector<uint8_t> keyData;
88 };
89 Vector<IdentifiedKey> m_keys WTF_GUARDED_BY_LOCK(m_keyLock);
90
91 bool m_isEncrypting { false };
92 uint64_t m_authenticationSize { 10 };
93 uint64_t m_keyId { 0 };
94 uint64_t m_counter { 0 };
95 CompatibilityMode m_compatibilityMode { CompatibilityMode::None };
96};
97
98inline void RTCRtpSFrameTransformer::setIsEncrypting(bool isEncrypting)
99{
100 m_isEncrypting = isEncrypting;
101}
102
103inline void RTCRtpSFrameTransformer::setAuthenticationSize(uint64_t size)
104{
105 m_authenticationSize = size;
106}
107
108inline void RTCRtpSFrameTransformer::setMediaType(RTCRtpTransformBackend::MediaType mediaType)
109{
110 if (mediaType == RTCRtpTransformBackend::MediaType::Video) {
111 m_authenticationSize = 10;
112 return;
113 }
114 m_authenticationSize = 4;
115 m_compatibilityMode = CompatibilityMode::None;
116}

118} // namespace WebCore
