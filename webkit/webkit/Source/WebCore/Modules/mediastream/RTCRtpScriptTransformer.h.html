37namespace WebCore {
38
39class MessagePort;
40class ReadableStream;
41class ScriptExecutionContext;
42class RTCRtpTransformBackend;
43class SerializedScriptValue;
44class SimpleReadableStreamSource;
45class WritableStream;
46
47struct MessageWithMessagePorts;
48
49class RTCRtpScriptTransformer
50 : public RefCounted<RTCRtpScriptTransformer>
51 , public ActiveDOMObject
52 , public CanMakeWeakPtr<RTCRtpScriptTransformer> {
53public:
54 static ExceptionOr<Ref<RTCRtpScriptTransformer>> create(ScriptExecutionContext&, MessageWithMessagePorts&&);
55 ~RTCRtpScriptTransformer();
56
57 ReadableStream& readable();
58 ExceptionOr<Ref<WritableStream>> writable();
59 JSC::JSValue options(JSC::JSGlobalObject&);
60
61 ExceptionOr<void> requestKeyFrame();
62
63 void startPendingActivity() { m_pendingActivity = makePendingActivity(*this); }
64 void start(Ref<RTCRtpTransformBackend>&&);
65
66 enum class ClearCallback { No, Yes};
67 void clear(ClearCallback);
68
69private:
70 RTCRtpScriptTransformer(ScriptExecutionContext&, Ref<SerializedScriptValue>&&, Vector<RefPtr<MessagePort>>&&, Ref<ReadableStream>&&, Ref<SimpleReadableStreamSource>&&);
71
72 // ActiveDOMObject
73 const char* activeDOMObjectName() const final { return "RTCRtpScriptTransformer"; }
74 void stop() final { stopPendingActivity(); }
75
76 void stopPendingActivity() { auto pendingActivity = WTFMove(m_pendingActivity); }
77
78 Ref<SerializedScriptValue> m_options;
79 Vector<RefPtr<MessagePort>> m_ports;
80
81 Ref<SimpleReadableStreamSource> m_readableSource;
82 Ref<ReadableStream> m_readable;
83 RefPtr<WritableStream> m_writable;
84
85 RefPtr<RTCRtpTransformBackend> m_backend;
86 RefPtr<PendingActivity<RTCRtpScriptTransformer>> m_pendingActivity;
87};
