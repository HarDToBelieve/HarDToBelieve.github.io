36#include "RTCIceTransport.h"
37#include "RTCRtpReceiver.h"
38#include "RTCRtpSender.h"
39#include "RTCRtpTransceiverBackend.h"
40#include "RTCRtpTransceiverDirection.h"
41#include "ScriptWrappable.h"
42#include <wtf/RefCounted.h>
43#include <wtf/RefPtr.h>
44#include <wtf/text/WTFString.h>
45
46namespace WebCore {
47
48class RTCPeerConnection;
49struct RTCRtpCodecCapability;
50
51class RTCRtpTransceiver final : public RefCounted<RTCRtpTransceiver>, public ScriptWrappable {
52 WTF_MAKE_ISO_ALLOCATED(RTCRtpTransceiver);
53public:
54 static Ref<RTCRtpTransceiver> create(Ref<RTCRtpSender>&& sender, Ref<RTCRtpReceiver>&& receiver, std::unique_ptr<RTCRtpTransceiverBackend>&& backend) { return adoptRef(*new RTCRtpTransceiver(WTFMove(sender), WTFMove(receiver), WTFMove(backend))); }
55 virtual ~RTCRtpTransceiver() = default;
56
57 bool hasSendingDirection() const;
58 void enableSendingDirection();
59 void disableSendingDirection();
60
61 RTCRtpTransceiverDirection direction() const;
62 std::optional<RTCRtpTransceiverDirection> currentDirection() const;
63 void setDirection(RTCRtpTransceiverDirection);
64 String mid() const;
65
66 RTCRtpSender& sender() { return m_sender.get(); }
67 RTCRtpReceiver& receiver() { return m_receiver.get(); }
68
69 bool stopped() const;
70 ExceptionOr<void> stop();
71 ExceptionOr<void> setCodecPreferences(const Vector<RTCRtpCodecCapability>&);
72
73 // FIXME: Temporary solution to keep track of ICE states for this transceiver. Later, each
74 // sender and receiver will have up to two DTLS transports, which in turn will have an ICE
75 // transport each.
76 RTCIceTransport& iceTransport() { return m_iceTransport.get(); }
77
78 RTCRtpTransceiverBackend* backend() { return m_backend.get(); }
79 void setConnection(RTCPeerConnection&);
80
81private:
82 RTCRtpTransceiver(Ref<RTCRtpSender>&&, Ref<RTCRtpReceiver>&&, std::unique_ptr<RTCRtpTransceiverBackend>&&);
83
84 RTCRtpTransceiverDirection m_direction;
85
86 Ref<RTCRtpSender> m_sender;
87 Ref<RTCRtpReceiver> m_receiver;
88
89 bool m_stopped { false };
90
91 Ref<RTCIceTransport> m_iceTransport;
92 std::unique_ptr<RTCRtpTransceiverBackend> m_backend;
93 WeakPtr<RTCPeerConnection> m_connection;
94};
95
96class RtpTransceiverSet {
97public:
98 const Vector<RefPtr<RTCRtpTransceiver>>& list() const { return m_transceivers; }
99 void append(Ref<RTCRtpTransceiver>&&);
100
101 Vector<std::reference_wrapper<RTCRtpSender>> senders() const;
102 Vector<std::reference_wrapper<RTCRtpReceiver>> receivers() const;
103
104private:
105 Vector<RefPtr<RTCRtpTransceiver>> m_transceivers;
106};

108} // namespace WebCore
