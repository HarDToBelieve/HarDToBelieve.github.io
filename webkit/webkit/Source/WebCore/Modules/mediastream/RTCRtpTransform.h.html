26#pragma once
27
28#if ENABLE(WEB_RTC)
29
30#include "RTCRtpSFrameTransform.h"
31#include "RTCRtpScriptTransform.h"
32#include "RTCRtpTransformBackend.h"
33#include <wtf/FastMalloc.h>
34
35namespace WebCore {
36
37class RTCRtpReceiver;
38class RTCRtpSender;
39
40class RTCRtpTransform {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 using Internal = Variant<RefPtr<RTCRtpSFrameTransform>, RefPtr<RTCRtpScriptTransform>>;
44 static std::unique_ptr<RTCRtpTransform> from(std::optional<Internal>&&);
45
46 explicit RTCRtpTransform(Internal&&);
47 ~RTCRtpTransform();
48
49 bool isAttached() const;
50 void attachToReceiver(RTCRtpReceiver&, RTCRtpTransform*);
51 void attachToSender(RTCRtpSender&, RTCRtpTransform*);
52 void detachFromReceiver(RTCRtpReceiver&);
53 void detachFromSender(RTCRtpSender&);
54
55 RefPtr<RTCRtpTransformBackend> takeBackend() { return WTFMove(m_backend); }
56 Internal internalTransform() { return m_transform; }
57
58 friend bool operator==(const RTCRtpTransform&, const RTCRtpTransform&);
59
60private:
61 void clearBackend();
62 void backendTransferedToNewTransform();
63
64 RefPtr<RTCRtpTransformBackend> m_backend;
65 Internal m_transform;
66};
67
68bool operator==(const RTCRtpTransform&, const RTCRtpTransform&);
69inline bool operator!=(const RTCRtpTransform& a, const RTCRtpTransform& b)
70{
71 return !(a == b);
72}
73
74} // namespace WebCore
75
76#endif // ENABLE(WEB_RTC)
