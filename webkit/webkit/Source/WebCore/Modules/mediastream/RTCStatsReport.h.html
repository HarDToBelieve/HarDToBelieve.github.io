33class DOMMapAdapter;

35class RTCStatsReport : public RefCounted<RTCStatsReport> {
36public:
37 using MapInitializer = Function<void(DOMMapAdapter&)>;
38 static Ref<RTCStatsReport> create(MapInitializer&& mapInitializer) { return adoptRef(*new RTCStatsReport(WTFMove(mapInitializer))); }
39
40 void initializeMapLike(DOMMapAdapter& adapter) { m_mapInitializer(adapter); }

42 enum class Type {
43 CandidatePair,
44 Certificate,
45 Codec,
46 DataChannel,
47 InboundRtp,
48 LocalCandidate,
49 MediaSource,
50 OutboundRtp,
51 PeerConnection,
52 RemoteCandidate,
53 RemoteInboundRtp,
54 Track,
55 Transport
56 };
57 struct Stats {
58 double timestamp;
59 Type type;
60 String id;
61 };
