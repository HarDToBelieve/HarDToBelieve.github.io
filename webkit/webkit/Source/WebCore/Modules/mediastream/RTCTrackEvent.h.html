31#pragma once
32
33#if ENABLE(WEB_RTC)
34
35#include "Event.h"
36#include <wtf/text/AtomString.h>
37
38namespace WebCore {
39
40class MediaStream;
41class MediaStreamTrack;
42class RTCRtpReceiver;
43class RTCRtpTransceiver;
44
45typedef Vector<RefPtr<MediaStream>> MediaStreamArray;
46
47class RTCTrackEvent final : public Event {
48 WTF_MAKE_ISO_ALLOCATED(RTCTrackEvent);
49public:
50 static Ref<RTCTrackEvent> create(const AtomString& type, CanBubble, IsCancelable, RefPtr<RTCRtpReceiver>&&, RefPtr<MediaStreamTrack>&&, MediaStreamArray&&, RefPtr<RTCRtpTransceiver>&&);
51
52 struct Init : EventInit {
53 RefPtr<RTCRtpReceiver> receiver;
54 RefPtr<MediaStreamTrack> track;
55 MediaStreamArray streams;
56 RefPtr<RTCRtpTransceiver> transceiver;
57 };
58 static Ref<RTCTrackEvent> create(const AtomString& type, const Init&, IsTrusted = IsTrusted::No);
59
60 RTCRtpReceiver* receiver() const { return m_receiver.get(); }
61 MediaStreamTrack* track() const { return m_track.get(); }
62 const MediaStreamArray& streams() const { return m_streams; }
63 RTCRtpTransceiver* transceiver() const { return m_transceiver.get(); }
64
65 virtual EventInterface eventInterface() const { return RTCTrackEventInterfaceType; }
66
67private:
68 RTCTrackEvent(const AtomString& type, CanBubble, IsCancelable, RefPtr<RTCRtpReceiver>&&, RefPtr<MediaStreamTrack>&&, MediaStreamArray&&, RefPtr<RTCRtpTransceiver>&&);
69 RTCTrackEvent(const AtomString& type, const Init&, IsTrusted);
70
71 RefPtr<RTCRtpReceiver> m_receiver;
72 RefPtr<MediaStreamTrack> m_track;
73 MediaStreamArray m_streams;
74 RefPtr<RTCRtpTransceiver> m_transceiver;
75};
76
77} // namespace WebCore
