36Ref<RTCTransformEvent> RTCTransformEvent::create(const AtomString& type, Ref<RTCRtpScriptTransformer>&& transformer, IsTrusted isTrusted)
37{
38 return adoptRef(*new RTCTransformEvent(type, WTFMove(transformer), isTrusted));
39}
40
41RTCTransformEvent::RTCTransformEvent(const AtomString& type, Ref<RTCRtpScriptTransformer>&& transformer, IsTrusted isTrusted)
42 : Event(type, { }, isTrusted)
43 , m_transformer(WTFMove(transformer))
44{
45}
46
47RTCRtpScriptTransformer& RTCTransformEvent::transformer()
48{
49 return m_transformer.get();
50}
51
52EventInterface RTCTransformEvent::eventInterface() const
53{
54 return RTCTransformEventInterfaceType;
55}
