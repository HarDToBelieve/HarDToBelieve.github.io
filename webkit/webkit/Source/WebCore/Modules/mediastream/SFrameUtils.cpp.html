35static inline bool isSliceNALU(uint8_t data)
36{
37 return (data & 0x1F) == 1;
38}
39
40static inline bool isSPSNALU(uint8_t data)
41{
42 return (data & 0x1F) == 7;
43}
44
45static inline bool isPPSNALU(uint8_t data)
46{
47 return (data & 0x1F) == 8;
48}
49
50static inline bool isIDRNALU(uint8_t data)
51{
52 return (data & 0x1F) == 5;
53}
54
55static inline void findNalus(const uint8_t* frameData, size_t frameSize, size_t offset, const Function<bool(size_t)>& callback)
56{
57 for (size_t i = 4 + offset; i < frameSize; ++i) {
58 if (frameData[i - 4] == 0 && frameData[i - 3] == 0 && frameData[i - 2] == 0 && frameData[i - 1] == 1) {
59 if (callback(i))
60 return;
61 }
62 }
63}
64
65size_t computeH264PrefixOffset(const uint8_t* frameData, size_t frameSize)
66{
67 size_t offset = 0;
68 findNalus(frameData, frameSize, 0, [&offset, &frameData](auto position) {
69 if (isIDRNALU(frameData[position]) || isSliceNALU(frameData[position])) {
70 // Skip 00 00 00 01, nalu type byte and the next byte.
71 offset = position + 2;
72 return true;
73 }
74 return false;
75 });
76 return offset;
77}

79bool needsRbspUnescaping(const uint8_t* frameData, size_t frameSize)
80{
81 for (size_t i = 0; i < frameSize - 3; ++i) {
82 if (frameData[i] == 0 && frameData[i + 1] == 0 && frameData[i + 2] == 3)
83 return true;
84 }
85 return false;
86}

88Vector<uint8_t> fromRbsp(const uint8_t* frameData, size_t frameSize)
89{
90 Vector<uint8_t> buffer;
91 buffer.reserveInitialCapacity(frameSize);
92
93 size_t i;
94 for (i = 0; i < frameSize - 3; ++i) {
95 if (frameData[i] == 0 && frameData[i + 1] == 0 && frameData[i + 2] == 3) {
96 buffer.uncheckedAppend(frameData[i]);
97 buffer.uncheckedAppend(frameData[i + 1]);
98 // Skip next byte which is delimiter.
99 i += 2;
100 } else
101 buffer.uncheckedAppend(frameData[i]);
102 }
103 for (; i < frameSize; ++i)
104 buffer.uncheckedAppend(frameData[i]);
105
106 return buffer;
107}

109SFrameCompatibilityPrefixBuffer computeH264PrefixBuffer(const uint8_t* frameData, size_t frameSize)
110{
111 // Delta and key prefixes assume SPS/PPS with IDs equal to 0 have been transmitted.
112 static const uint8_t prefixDeltaFrame[6] = { 0x00, 0x00, 0x00, 0x01, 0x21, 0xe0 };
113
114 if (frameSize < 5)
115 return { };
116
117 // We assume a key frame starts with SPS, then PPS. Otherwise we wrap it as a delta frame.
118 if (!isSPSNALU(frameData[4]))
119 return SFrameCompatibilityPrefixBuffer { prefixDeltaFrame, sizeof(prefixDeltaFrame), { } };
120
121 // Search for PPS
122 size_t spsPpsLength = 0;
123 findNalus(frameData, frameSize, 5, [frameData, &spsPpsLength](auto position) {
124 if (isPPSNALU(frameData[position]))
125 spsPpsLength = position;
126 return true;
127 });
128 if (!spsPpsLength)
129 return SFrameCompatibilityPrefixBuffer { prefixDeltaFrame, sizeof(prefixDeltaFrame), { } };
130
131 // Search for next NALU to compute the real spsPpsLength, including the next 00 00 00 01.
132 findNalus(frameData, frameSize, spsPpsLength + 1, [&spsPpsLength](auto position) {
133 spsPpsLength = position;
134 return true;
135 });
136
137 Vector<uint8_t> buffer;
138 buffer.resize(spsPpsLength + 2);
139 std::memcpy(buffer.data(), frameData, spsPpsLength);
140 buffer[spsPpsLength] = 0x25;
141 buffer[spsPpsLength + 1] = 0xb8;
142 return { buffer.data(), buffer.size(), WTFMove(buffer) };
143}

145static inline void findEscapeRbspPatterns(const Vector<uint8_t>& frame, size_t offset, const Function<void(size_t, bool)>& callback)
146{
147 size_t numConsecutiveZeros = 0;
148 auto* data = frame.data();
149 for (size_t i = offset; i < frame.size(); ++i) {
150 bool shouldEscape = data[i] <= 3 && numConsecutiveZeros >= 2;
151 if (shouldEscape)
152 numConsecutiveZeros = 0;
153
154 if (data[i] == 0)
155 ++numConsecutiveZeros;
156 else
157 numConsecutiveZeros = 0;
158
159 callback(i, shouldEscape);
160 }
161}
162
163void toRbsp(Vector<uint8_t>& frame, size_t offset)
164{
165 size_t count = 0;
166 findEscapeRbspPatterns(frame, offset, [&count](size_t, bool shoudlBeEscaped) {
167 if (shoudlBeEscaped)
168 count++;
169 });
170 if (!count)
171 return;
172
173 Vector<uint8_t> newFrame;
174 newFrame.reserveInitialCapacity(frame.size() + count);
175 newFrame.append(frame.data(), offset);
176
177 findEscapeRbspPatterns(frame, offset, [data = frame.data(), &newFrame](size_t position, bool shoudlBeEscaped) {
178 if (shoudlBeEscaped)
179 newFrame.uncheckedAppend(3);
180 newFrame.uncheckedAppend(data[position]);
181 });
182
183 frame = WTFMove(newFrame);
184}

186static inline bool isVP8KeyFrame(const uint8_t* frame, size_t size)
187{
188 ASSERT_UNUSED(size, size);
189 return !(*frame & 0x01);
190}
191
192size_t computeVP8PrefixOffset(const uint8_t* frame, size_t size)
193{
194 return isVP8KeyFrame(frame, size) ? 10 : 3;
195}
196
197SFrameCompatibilityPrefixBuffer computeVP8PrefixBuffer(const uint8_t* frame, size_t size)
198{
199 Vector<uint8_t> prefix;
200 prefix.append(frame, isVP8KeyFrame(frame, size) ? 10 : 3);
201 return { prefix.data(), prefix.size(), WTFMove(prefix) };
202}
