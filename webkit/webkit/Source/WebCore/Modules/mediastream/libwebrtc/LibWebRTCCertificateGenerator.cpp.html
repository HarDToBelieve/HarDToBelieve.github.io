45static inline String fromStdString(const std::string& value)
46{
47 return String::fromUTF8(value.data(), value.length());
48}
49
50class RTCCertificateGeneratorCallback : public ThreadSafeRefCounted<RTCCertificateGeneratorCallback, WTF::DestructionThread::Main>, public rtc::RTCCertificateGeneratorCallback {
51public:
52 RTCCertificateGeneratorCallback(Ref<SecurityOrigin>&& origin, DOMPromiseDeferred<IDLInterface<RTCCertificate>>&& promise)
53 : m_origin(WTFMove(origin))
54 , m_promise(WTFMove(promise))
55 {
56 }
57
58 void AddRef() const { ref(); }
59 rtc::RefCountReleaseStatus Release() const
60 {
61 auto result = refCount() - 1;
62 deref();
63 return result ? rtc::RefCountReleaseStatus::kOtherRefsRemained : rtc::RefCountReleaseStatus::kDroppedLastRef;
64 }
65
66private:
67 void OnSuccess(const rtc::scoped_refptr<rtc::RTCCertificate>& certificate) final
68 {
69 callOnMainThread([origin = m_origin.releaseNonNull(), promise = WTFMove(m_promise.value()), certificate]() mutable {
70 Vector<RTCCertificate::DtlsFingerprint> fingerprints;
71 auto stats = certificate->ssl_certificate().GetStats();
72 auto* info = stats.get();
73 while (info) {
74 StringView fingerprint { reinterpret_cast<const unsigned char*>(info->fingerprint.data()), static_cast<unsigned>(info->fingerprint.length()) };
75 fingerprints.append({ fromStdString(info->fingerprint_algorithm), fingerprint.convertToASCIILowercase() });
76 info = info->issuer.get();
77 };
78
79 auto pem = certificate->ToPEM();
80 promise.resolve(RTCCertificate::create(WTFMove(origin), certificate->Expires(), WTFMove(fingerprints), fromStdString(pem.certificate()), fromStdString(pem.private_key())));
81 });
82 }
83
84 void OnFailure() final
85 {
86 callOnMainThread([promise = WTFMove(m_promise.value())]() mutable {
87 promise.reject(Exception { TypeError, "Unable to create a certificate"_s});
88 });
89 }
90
91 RefPtr<SecurityOrigin> m_origin;
92 std::optional<DOMPromiseDeferred<IDLInterface<RTCCertificate>>> m_promise;
93};
94
95static inline rtc::KeyParams keyParamsFromCertificateType(const PeerConnectionBackend::CertificateInformation& info)
96{
97 switch (info.type) {
98 case PeerConnectionBackend::CertificateInformation::Type::ECDSAP256:
99 return rtc::KeyParams::ECDSA();
100 case PeerConnectionBackend::CertificateInformation::Type::RSASSAPKCS1v15:
101 if (info.rsaParameters)
102 return rtc::KeyParams::RSA(info.rsaParameters->modulusLength, info.rsaParameters->publicExponent);
103 return rtc::KeyParams::RSA(2048, 65537);
104 }
105
106 RELEASE_ASSERT_NOT_REACHED();
107}
108
109void generateCertificate(Ref<SecurityOrigin>&& origin, LibWebRTCProvider& provider, const PeerConnectionBackend::CertificateInformation& info, DOMPromiseDeferred<IDLInterface<RTCCertificate>>&& promise)
110{
111 rtc::scoped_refptr<RTCCertificateGeneratorCallback> callback(new rtc::RefCountedObject<RTCCertificateGeneratorCallback>(WTFMove(origin), WTFMove(promise)));
112
113 absl::optional<uint64_t> expiresMs;
114 if (info.expires)
115 expiresMs = static_cast<uint64_t>(*info.expires);
116
117 provider.prepareCertificateGenerator([info, expiresMs, callback = WTFMove(callback)](auto& generator) mutable {
118 generator.GenerateCertificateAsync(keyParamsFromCertificateType(info), expiresMs, WTFMove(callback));
119 });
120}
