38webrtc::DataChannelInit LibWebRTCDataChannelHandler::fromRTCDataChannelInit(const RTCDataChannelInit& options)
39{
40 webrtc::DataChannelInit init;
41 if (options.ordered)
42 init.ordered = *options.ordered;
43 if (options.maxPacketLifeTime)
44 init.maxRetransmitTime = *options.maxPacketLifeTime;
45 if (options.maxRetransmits)
46 init.maxRetransmits = *options.maxRetransmits;
47 init.protocol = options.protocol.utf8().data();
48 if (options.negotiated)
49 init.negotiated = *options.negotiated;
50 if (options.id)
51 init.id = *options.id;
52 init.priority = fromRTCPriorityType(options.priority);
53 return init;
54}

56Ref<RTCDataChannelEvent> LibWebRTCDataChannelHandler::channelEvent(Document& document, rtc::scoped_refptr<webrtc::DataChannelInterface>&& dataChannel)
57{
58 auto protocol = dataChannel->protocol();
59 auto label = dataChannel->label();
60
61 RTCDataChannelInit init;
62 init.ordered = dataChannel->ordered();
63 init.maxPacketLifeTime = dataChannel->maxRetransmitTime();
64 init.maxRetransmits = dataChannel->maxRetransmits();
65 init.protocol = fromStdString(protocol);
66 init.negotiated = dataChannel->negotiated();
67 init.id = dataChannel->id();
68 init.priority = toRTCPriorityType(dataChannel->priority());
69
70 auto handler = makeUnique<LibWebRTCDataChannelHandler>(WTFMove(dataChannel));
71 auto channel = RTCDataChannel::create(document, WTFMove(handler), fromStdString(label), WTFMove(init));
72
73 return RTCDataChannelEvent::create(eventNames().datachannelEvent, Event::CanBubble::No, Event::IsCancelable::No, WTFMove(channel));
74}

88void LibWebRTCDataChannelHandler::setClient(RTCDataChannelHandlerClient& client, ScriptExecutionContextIdentifier contextIdentifier)
89{
90 {
91 Locker locker { m_clientLock };
92 ASSERT(!m_client);
93 m_client = &client;
94 m_contextIdentifier = contextIdentifier;
95
96 for (auto& message : m_bufferedMessages) {
97 switchOn(message, [&](Ref<SharedBuffer>& data) {
98 client.didReceiveRawData(data->data(), data->size());
99 }, [&](String& text) {
100 client.didReceiveStringData(text);
101 }, [&](RTCDataChannelState state) {
102 client.didChangeReadyState(state);
103 });
104 }
105 m_bufferedMessages.clear();
106 }
107 checkState();
108}

130void LibWebRTCDataChannelHandler::checkState()
131{
132 RTCDataChannelState state;
133 switch (m_channel->state()) {
134 case webrtc::DataChannelInterface::kConnecting:
135 state = RTCDataChannelState::Connecting;
136 break;
137 case webrtc::DataChannelInterface::kOpen:
138 state = RTCDataChannelState::Open;
139 break;
140 case webrtc::DataChannelInterface::kClosing:
141 state = RTCDataChannelState::Closing;
142 break;
143 case webrtc::DataChannelInterface::kClosed:
144 state = RTCDataChannelState::Closed;
145 break;
146 }
147
148 Locker locker { m_clientLock };
149 if (!m_client) {
150 m_bufferedMessages.append(state);
151 return;
152 }
153 postTask([protectedClient = makeRef(*m_client), state] {
154 protectedClient->didChangeReadyState(state);
155 });
156}
157
158void LibWebRTCDataChannelHandler::OnMessage(const webrtc::DataBuffer& buffer)
159{
160 Locker locker { m_clientLock };
161 if (!m_client) {
162 auto* data = buffer.data.data<uint8_t>();
163 if (buffer.binary)
164 m_bufferedMessages.append(SharedBuffer::create(data, buffer.size()));
165 else
166 m_bufferedMessages.append(String::fromUTF8(data, buffer.size()));
167 return;
168 }
169
170 std::unique_ptr<webrtc::DataBuffer> protectedBuffer(new webrtc::DataBuffer(buffer));
171 postTask([protectedClient = makeRef(*m_client), buffer = WTFMove(protectedBuffer)] {
172 auto* data = buffer->data.data<uint8_t>();
173 if (buffer->binary)
174 protectedClient->didReceiveRawData(data, buffer->size());
175 else
176 protectedClient->didReceiveStringData(String::fromUTF8(data, buffer->size()));
177 });
178}

180void LibWebRTCDataChannelHandler::OnBufferedAmountChange(uint64_t amount)
181{
182 Locker locker { m_clientLock };
183 if (!m_client)
184 return;
185
186 postTask([protectedClient = makeRef(*m_client), amount] {
187 protectedClient->bufferedAmountIsDecreasing(static_cast<size_t>(amount));
188 });
189}
190
191void LibWebRTCDataChannelHandler::postTask(Function<void()>&& function)
192{
193 ASSERT(m_clientLock.isHeld());
194
195 if (!m_contextIdentifier) {
196 callOnMainThread(WTFMove(function));
197 return;
198 }
199 ScriptExecutionContext::postTaskTo(m_contextIdentifier, WTFMove(function));
200}
