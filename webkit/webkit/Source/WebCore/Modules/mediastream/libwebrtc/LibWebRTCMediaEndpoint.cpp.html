90bool LibWebRTCMediaEndpoint::setConfiguration(LibWebRTCProvider& client, webrtc::PeerConnectionInterface::RTCConfiguration&& configuration)
91{
92 configuration.sdp_semantics = webrtc::SdpSemantics::kUnifiedPlan;
93
94 if (!m_backend) {
95 if (!m_rtcSocketFactory) {
96 auto& document = downcast<Document>(*m_peerConnectionBackend.connection().scriptExecutionContext());
97 RegistrableDomain domain { document.url() };
98 bool isFirstParty = domain == RegistrableDomain(document.firstPartyForCookies());
99 m_rtcSocketFactory = client.createSocketFactory(document.userAgent(document.url()), isFirstParty, WTFMove(domain));
100 if (!m_peerConnectionBackend.shouldFilterICECandidates())
101 m_rtcSocketFactory->disableRelay();
102 }
103 m_backend = client.createPeerConnection(*this, m_rtcSocketFactory.get(), WTFMove(configuration));
104 return !!m_backend;
105 }
106 auto oldConfiguration = m_backend->GetConfiguration();
107 configuration.certificates = oldConfiguration.certificates;
108 return m_backend->SetConfiguration(WTFMove(configuration)).ok();
109}

123static inline const char* sessionDescriptionType(RTCSdpType sdpType)
124{
125 switch (sdpType) {
126 case RTCSdpType::Offer:
127 return "offer";
128 case RTCSdpType::Pranswer:
129 return "pranswer";
130 case RTCSdpType::Answer:
131 return "answer";
132 case RTCSdpType::Rollback:
133 return "rollback";
134 }
135
136 ASSERT_NOT_REACHED();
137 return "";
138}

140static inline RTCSdpType fromSessionDescriptionType(const webrtc::SessionDescriptionInterface& description)
141{
142 auto type = description.type();
143 if (type == webrtc::SessionDescriptionInterface::kOffer)
144 return RTCSdpType::Offer;
145 if (type == webrtc::SessionDescriptionInterface::kAnswer)
146 return RTCSdpType::Answer;
147 ASSERT(type == webrtc::SessionDescriptionInterface::kPrAnswer);
148 return RTCSdpType::Pranswer;
149}

193void LibWebRTCMediaEndpoint::doSetLocalDescription(const RTCSessionDescription* description)
194{
195 ASSERT(m_backend);
196
197 if (!description) {
198 m_backend->SetLocalDescription(&m_setLocalSessionDescriptionObserver);
199 return;
200 }
201
202 webrtc::SdpParseError error;
203 std::unique_ptr<webrtc::SessionDescriptionInterface> sessionDescription(webrtc::CreateSessionDescription(sessionDescriptionType(description->type()), description->sdp().utf8().data(), &error));
204
205 if (!sessionDescription) {
206 m_peerConnectionBackend.setLocalDescriptionFailed(Exception { OperationError, fromStdString(error.description) });
207 return;
208 }
209
210 // FIXME: See https://bugs.webkit.org/show_bug.cgi?id=173783. Remove this test once fixed at LibWebRTC level.
211 if (description->type() == RTCSdpType::Answer && !m_backend->pending_remote_description()) {
212 m_peerConnectionBackend.setLocalDescriptionFailed(Exception { InvalidStateError, "Failed to set local answer sdp: no pending remote description."_s });
213 return;
214 }
215
216 m_backend->SetLocalDescription(&m_setLocalSessionDescriptionObserver, sessionDescription.release());
217}

219void LibWebRTCMediaEndpoint::doSetRemoteDescription(const RTCSessionDescription& description)
220{
221 ASSERT(m_backend);
222
223 webrtc::SdpParseError error;
224 std::unique_ptr<webrtc::SessionDescriptionInterface> sessionDescription(webrtc::CreateSessionDescription(sessionDescriptionType(description.type()), description.sdp().utf8().data(), &error));
225 if (!sessionDescription) {
226 m_peerConnectionBackend.setRemoteDescriptionFailed(Exception { SyntaxError, fromStdString(error.description) });
227 return;

229 m_backend->SetRemoteDescription(&m_setRemoteSessionDescriptionObserver, sessionDescription.release());
230
231 startLoggingStats();
232}

234bool LibWebRTCMediaEndpoint::addTrack(LibWebRTCRtpSenderBackend& sender, MediaStreamTrack& track, const Vector<String>& mediaStreamIds)
235{
236 ASSERT(m_backend);
237
238 LibWebRTCRtpSenderBackend::Source source;
239 rtc::scoped_refptr<webrtc::MediaStreamTrackInterface> rtcTrack;
240 switch (track.privateTrack().type()) {
241 case RealtimeMediaSource::Type::Audio: {
242 auto audioSource = RealtimeOutgoingAudioSource::create(track.privateTrack());
243 rtcTrack = m_peerConnectionFactory->CreateAudioTrack(track.id().utf8().data(), audioSource.ptr());
244 source = WTFMove(audioSource);
245 break;
246 }
247 case RealtimeMediaSource::Type::Video: {
248 auto videoSource = RealtimeOutgoingVideoSource::create(track.privateTrack());
249 rtcTrack = m_peerConnectionFactory->CreateVideoTrack(track.id().utf8().data(), videoSource.ptr());
250 source = WTFMove(videoSource);
251 break;
252 }
253 case RealtimeMediaSource::Type::None:
254 ASSERT_NOT_REACHED();
255 return false;
256 }
257
258 sender.setSource(WTFMove(source));
259 if (auto rtpSender = sender.rtcSender()) {
260 rtpSender->SetTrack(rtcTrack.get());
261 return true;
262 }
263
264 std::vector<std::string> ids;
265 for (auto& id : mediaStreamIds)
266 ids.push_back(id.utf8().data());
267
268 auto newRTPSender = m_backend->AddTrack(rtcTrack.get(), WTFMove(ids));
269 if (!newRTPSender.ok())
270 return false;
271 sender.setRTCSender(newRTPSender.MoveValue());
272 return true;
273}

331static RTCSignalingState signalingState(webrtc::PeerConnectionInterface::SignalingState state)
332{
333 switch (state) {
334 case webrtc::PeerConnectionInterface::kStable:
335 return RTCSignalingState::Stable;
336 case webrtc::PeerConnectionInterface::kHaveLocalOffer:
337 return RTCSignalingState::HaveLocalOffer;
338 case webrtc::PeerConnectionInterface::kHaveLocalPrAnswer:
339 return RTCSignalingState::HaveLocalPranswer;
340 case webrtc::PeerConnectionInterface::kHaveRemoteOffer:
341 return RTCSignalingState::HaveRemoteOffer;
342 case webrtc::PeerConnectionInterface::kHaveRemotePrAnswer:
343 return RTCSignalingState::HaveRemotePranswer;
344 case webrtc::PeerConnectionInterface::kClosed:
345 return RTCSignalingState::Stable;
346 }
347
348 ASSERT_NOT_REACHED();
349 return RTCSignalingState::Stable;
350}

372void LibWebRTCMediaEndpoint::addPendingTrackEvent(Ref<RTCRtpReceiver>&& receiver, MediaStreamTrack& track, const std::vector<rtc::scoped_refptr<webrtc::MediaStreamInterface>>& rtcStreams, RefPtr<RTCRtpTransceiver>&& transceiver)
373{
374 Vector<RefPtr<MediaStream>> streams;
375 for (auto& rtcStream : rtcStreams) {
376 auto& mediaStream = mediaStreamFromRTCStream(*rtcStream.get());
377 streams.append(&mediaStream);
378 mediaStream.addTrackFromPlatform(track);
379 }
380 auto streamIds = WTF::map(streams, [](auto& stream) -> String {
381 return stream->id();
382 });
383 m_remoteStreamsFromRemoteTrack.add(&track, WTFMove(streamIds));
384
385 m_peerConnectionBackend.addPendingTrackEvent({ WTFMove(receiver), makeRef(track), WTFMove(streams), WTFMove(transceiver) });
386}

388void LibWebRTCMediaEndpoint::collectTransceivers()
389{
390 if (!m_backend)
391 return;
392
393 for (auto& rtcTransceiver : m_backend->GetTransceivers()) {
394 auto* existingTransceiver = m_peerConnectionBackend.existingTransceiver([&](auto& transceiverBackend) {
395 return rtcTransceiver.get() == transceiverBackend.rtcTransceiver();
396 });
397 if (existingTransceiver)
398 continue;
399
400 auto rtcReceiver = rtcTransceiver->receiver();
401 m_peerConnectionBackend.newRemoteTransceiver(makeUnique<LibWebRTCRtpTransceiverBackend>(WTFMove(rtcTransceiver)), rtcReceiver->media_type() == cricket::MEDIA_TYPE_AUDIO ? RealtimeMediaSource::Type::Audio : RealtimeMediaSource::Type::Video);
402 }
403}

405void LibWebRTCMediaEndpoint::newTransceiver(rtc::scoped_refptr<webrtc::RtpTransceiverInterface>&& rtcTransceiver)
406{
407 auto rtcReceiver = rtcTransceiver->receiver();
408 auto* transceiver = m_peerConnectionBackend.existingTransceiver([&](auto& transceiverBackend) {
409 return rtcTransceiver.get() == transceiverBackend.rtcTransceiver();
410 });
411 if (!transceiver)
412 transceiver = &m_peerConnectionBackend.newRemoteTransceiver(makeUnique<LibWebRTCRtpTransceiverBackend>(WTFMove(rtcTransceiver)), rtcReceiver->media_type() == cricket::MEDIA_TYPE_AUDIO ? RealtimeMediaSource::Type::Audio : RealtimeMediaSource::Type::Video);
413
414 addPendingTrackEvent(makeRef(transceiver->receiver()), transceiver->receiver().track(), rtcReceiver->streams(), makeRef(*transceiver));
415}

417void LibWebRTCMediaEndpoint::removeRemoteTrack(rtc::scoped_refptr<webrtc::RtpReceiverInterface>&& receiver)
418{
419 auto* transceiver = m_peerConnectionBackend.existingTransceiver([&receiver](auto& transceiverBackend) {
420 auto* rtcTransceiver = transceiverBackend.rtcTransceiver();
421 return rtcTransceiver && receiver.get() == rtcTransceiver->receiver().get();
422 });
423 if (!transceiver)
424 return;
425
426 auto& track = transceiver->receiver().track();
427
428 for (auto& id : m_remoteStreamsFromRemoteTrack.get(&track)) {
429 if (auto stream = m_remoteStreamsById.get(id))
430 stream->privateStream().removeTrack(track.privateTrack());
431 }
432
433 track.source().setMuted(true);
434}

611void LibWebRTCMediaEndpoint::OnIceCandidate(const webrtc::IceCandidateInterface *rtcCandidate)
612{
613 ASSERT(rtcCandidate);
614
615 std::string sdp;
616 rtcCandidate->ToString(&sdp);
617
618 auto sdpMLineIndex = safeCast<unsigned short>(rtcCandidate->sdp_mline_index());
619

713void LibWebRTCMediaEndpoint::OnStatsDelivered(const rtc::scoped_refptr<const webrtc::RTCStatsReport>& report)
714{
715#if !RELEASE_LOG_DISABLED
716 int64_t timestamp = report->timestamp_us();
717 if (!m_statsFirstDeliveredTimestamp)
718 m_statsFirstDeliveredTimestamp = timestamp;
719
720 callOnMainThread([protectedThis = makeRef(*this), this, timestamp, report] {
721 if (m_backend && m_statsLogTimer.repeatInterval() != statsLogInterval(timestamp)) {
722 m_statsLogTimer.stop();
723 m_statsLogTimer.startRepeating(statsLogInterval(timestamp));
724 }
725
726 for (auto iterator = report->begin(); iterator != report->end(); ++iterator) {
727 if (logger().willLog(logChannel(), WTFLogLevel::Debug)) {
728 // Stats are very verbose, let's only display them in inspector console in verbose mode.
729 logger().debug(LogWebRTC,
730 Logger::LogSiteIdentifier("LibWebRTCMediaEndpoint", "OnStatsDelivered", logIdentifier()),
731 RTCStatsLogger { *iterator });
732 } else {
733 logger().logAlways(LogWebRTCStats,
734 Logger::LogSiteIdentifier("LibWebRTCMediaEndpoint", "OnStatsDelivered", logIdentifier()),
735 RTCStatsLogger { *iterator });
736 }
737 }
738 });
739#else
740 UNUSED_PARAM(report);
741#endif
742}

748 m_statsLogTimer.stop();

750#endif
751}

764Seconds LibWebRTCMediaEndpoint::statsLogInterval(int64_t reportTimestamp) const
765{
766 if (logger().willLog(logChannel(), WTFLogLevel::Info))
767 return 2_s;
768
769 if (reportTimestamp - m_statsFirstDeliveredTimestamp > 15000000)
770 return 10_s;

772 return 4_s;
