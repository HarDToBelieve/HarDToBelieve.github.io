41namespace WebCore {
42
43template<typename Endpoint>
44class CreateSessionDescriptionObserver final : public webrtc::CreateSessionDescriptionObserver {
45public:
46 explicit CreateSessionDescriptionObserver(Endpoint &endpoint)
47 : m_endpoint(endpoint)
48 {
49 }
50
51 void OnSuccess(webrtc::SessionDescriptionInterface* sessionDescription) final { m_endpoint.createSessionDescriptionSucceeded(std::unique_ptr<webrtc::SessionDescriptionInterface>(sessionDescription)); }
52 void OnFailure(webrtc::RTCError error) final { m_endpoint.createSessionDescriptionFailed(toExceptionCode(error.type()), error.message()); }
53
54 void AddRef() const { m_endpoint.AddRef(); }
55 rtc::RefCountReleaseStatus Release() const { return m_endpoint.Release(); }
56
57private:
58 Endpoint& m_endpoint;
59};
