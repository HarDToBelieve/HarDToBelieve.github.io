32namespace WebCore {

34void LibWebRTCRtpTransformBackend::setInputCallback(Callback&& callback)
35{
36 Locker locker { m_inputCallbackLock };
37 m_inputCallback = WTFMove(callback);
38}
39
40void LibWebRTCRtpTransformBackend::clearTransformableFrameCallback()
41{
42 setInputCallback({ });
43}
44
45void LibWebRTCRtpTransformBackend::setOutputCallback(rtc::scoped_refptr<webrtc::TransformedFrameCallback>&& callback)
46{
47 Locker locker { m_outputCallbackLock };
48 m_outputCallback = WTFMove(callback);
49}
50
51void LibWebRTCRtpTransformBackend::processTransformedFrame(RTCRtpTransformableFrame& frame)
52{
53 auto rtcFrame = static_cast<LibWebRTCRtpTransformableFrame&>(frame).takeRTCFrame();
54 if (!rtcFrame)
55 return;
56 Locker locker { m_outputCallbackLock };
57 if (m_outputCallback)
58 m_outputCallback->OnTransformedFrame(WTFMove(rtcFrame));
59}
60
61void LibWebRTCRtpTransformBackend::Transform(std::unique_ptr<webrtc::TransformableFrameInterface> rtcFrame)
62{
63 {
64 Locker locker { m_inputCallbackLock };
65 if (m_inputCallback) {
66 m_inputCallback(LibWebRTCRtpTransformableFrame::create(WTFMove(rtcFrame), m_mediaType == MediaType::Audio && m_side == Side::Sender));
67 return;
68 }
69 }
70 // In case of no input callback, make the transform a no-op.
71 Locker locker { m_outputCallbackLock };
72 if (m_outputCallback)
73 m_outputCallback->OnTransformedFrame(WTFMove(rtcFrame));
74}
75
76void LibWebRTCRtpTransformBackend::RegisterTransformedFrameCallback(rtc::scoped_refptr<webrtc::TransformedFrameCallback> callback)
77{
78 setOutputCallback(WTFMove(callback));
79}
80
81void LibWebRTCRtpTransformBackend::RegisterTransformedFrameSinkCallback(rtc::scoped_refptr<webrtc::TransformedFrameCallback> callback, uint32_t /* ssrc */)
82{
83 RegisterTransformedFrameCallback(WTFMove(callback));
84}
85
86void LibWebRTCRtpTransformBackend::UnregisterTransformedFrameCallback()
87{
88 setOutputCallback(nullptr);
89}
90
91void LibWebRTCRtpTransformBackend::UnregisterTransformedFrameSinkCallback(uint32_t /* ssrc */)
92{
93 UnregisterTransformedFrameCallback();
94}
