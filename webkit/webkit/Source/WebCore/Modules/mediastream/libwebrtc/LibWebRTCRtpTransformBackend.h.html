42namespace WebCore {
43
44class LibWebRTCRtpTransformBackend : public RTCRtpTransformBackend, public webrtc::FrameTransformerInterface {
45protected:
46 LibWebRTCRtpTransformBackend(MediaType, Side);
47 void setInputCallback(Callback&&);
48
49protected:
50 MediaType mediaType() const final { return m_mediaType; }
51
52private:
53 void setOutputCallback(rtc::scoped_refptr<webrtc::TransformedFrameCallback>&&);
54
55 // RTCRtpTransformBackend
56 void processTransformedFrame(RTCRtpTransformableFrame&) final;
57 void clearTransformableFrameCallback() final;
58 Side side() const final { return m_side; }
59
60 // webrtc::FrameTransformerInterface
61 void Transform(std::unique_ptr<webrtc::TransformableFrameInterface>) final;
62 void RegisterTransformedFrameCallback(rtc::scoped_refptr<webrtc::TransformedFrameCallback>) final;
63 void RegisterTransformedFrameSinkCallback(rtc::scoped_refptr<webrtc::TransformedFrameCallback>, uint32_t ssrc) final;
64 void UnregisterTransformedFrameCallback() final;
65 void UnregisterTransformedFrameSinkCallback(uint32_t ssrc) final;
66 void AddRef() const final { ref(); }
67 rtc::RefCountReleaseStatus Release() const final;
68
69 MediaType m_mediaType;
70 Side m_side;
71
72 Lock m_inputCallbackLock;
73 Callback m_inputCallback WTF_GUARDED_BY_LOCK(m_inputCallbackLock);
74
75 Lock m_outputCallbackLock;
76 rtc::scoped_refptr<webrtc::TransformedFrameCallback> m_outputCallback WTF_GUARDED_BY_LOCK(m_outputCallbackLock);
77};
78
79inline LibWebRTCRtpTransformBackend::LibWebRTCRtpTransformBackend(MediaType mediaType, Side side)
80 : m_mediaType(mediaType)
81 , m_side(side)
82{
83}
84
85inline rtc::RefCountReleaseStatus LibWebRTCRtpTransformBackend::Release() const
86{
87 deref();
88 return rtc::RefCountReleaseStatus::kOtherRefsRemained;
89}
