69bool LibWebRTCRtpTransformableFrame::isKeyFrame() const
70{
71 ASSERT(m_rtcFrame);

81RTCEncodedAudioFrameMetadata LibWebRTCRtpTransformableFrame::audioMetadata() const
82{
83 if (!m_rtcFrame)
84 return { };
85
86 Vector<uint32_t> cssrcs;
87 if (!m_isAudioSenderFrame) {
88 auto* audioFrame = static_cast<webrtc::TransformableAudioFrameInterface*>(m_rtcFrame.get());
89 auto& header = audioFrame->GetHeader();
90 if (header.numCSRCs) {
91 cssrcs.reserveInitialCapacity(header.numCSRCs);
92 for (size_t cptr = 0; cptr < header.numCSRCs; ++cptr)
93 cssrcs.uncheckedAppend(header.arrOfCSRCs[cptr]);
94 }
95 }
96 return { m_rtcFrame->GetSsrc(), WTFMove(cssrcs) };
97}
98
99RTCEncodedVideoFrameMetadata LibWebRTCRtpTransformableFrame::videoMetadata() const
100{
101 if (!m_rtcFrame)
102 return { };
103 auto* videoFrame = static_cast<webrtc::TransformableVideoFrameInterface*>(m_rtcFrame.get());
104 auto& metadata = videoFrame->GetMetadata();
105
106 std::optional<int64_t> frameId;
107 if (metadata.GetFrameId())
108 frameId = *metadata.GetFrameId();
109
110 Vector<int64_t> dependencies;
111 for (auto value : metadata.GetFrameDependencies())
112 dependencies.append(value);
113
114 return { frameId, WTFMove(dependencies), metadata.GetWidth(), metadata.GetHeight(), metadata.GetSpatialIndex(), metadata.GetTemporalIndex(), m_rtcFrame->GetSsrc() };
115}
