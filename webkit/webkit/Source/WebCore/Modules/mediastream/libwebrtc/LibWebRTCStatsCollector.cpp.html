57static inline void fillRTCStats(RTCStatsReport::Stats& stats, const webrtc::RTCStats& rtcStats)
58{
59 stats.timestamp = Performance::reduceTimeResolution(Seconds::fromMicroseconds(rtcStats.timestamp_us())).milliseconds();
60 stats.id = fromStdString(rtcStats.id());
61}

357static inline RTCStatsReport::IceCandidatePairState iceCandidatePairState(const std::string& state)
358{
359 if (state == "frozen")
360 return RTCStatsReport::IceCandidatePairState::Frozen;
361 if (state == "waiting")
362 return RTCStatsReport::IceCandidatePairState::Waiting;
363 if (state == "in-progress")
364 return RTCStatsReport::IceCandidatePairState::Inprogress;
365 if (state == "failed")
366 return RTCStatsReport::IceCandidatePairState::Failed;
367 if (state == "succeeded")
368 return RTCStatsReport::IceCandidatePairState::Succeeded;
369 if (state == "cancelled")
370 return RTCStatsReport::IceCandidatePairState::Cancelled;
371 ASSERT_NOT_REACHED();
372 return RTCStatsReport::IceCandidatePairState::Frozen;
373}

433static inline std::optional<RTCIceCandidateType> iceCandidateState(const std::string& state)
434{
435 if (state == "host")
436 return RTCIceCandidateType::Host;
437 if (state == "srflx")
438 return RTCIceCandidateType::Srflx;
439 if (state == "prflx")
440 return RTCIceCandidateType::Prflx;
441 if (state == "relay")
442 return RTCIceCandidateType::Relay;
443
444 return { };
445}

447static inline void fillRTCIceCandidateStats(RTCStatsReport::IceCandidateStats& stats, const webrtc::RTCIceCandidateStats& rtcStats)
448{
449 stats.type = rtcStats.type() == webrtc::RTCRemoteIceCandidateStats::kType ? RTCStatsReport::Type::RemoteCandidate : RTCStatsReport::Type::LocalCandidate;
450
451 fillRTCStats(stats, rtcStats);
452
453 if (rtcStats.transport_id.is_defined())
454 stats.transportId = fromStdString(*rtcStats.transport_id);
455 if (rtcStats.ip.is_defined())
456 stats.address = fromStdString(*rtcStats.ip);
457 if (rtcStats.port.is_defined())
458 stats.port = *rtcStats.port;
459 if (rtcStats.protocol.is_defined())
460 stats.protocol = fromStdString(*rtcStats.protocol);
461
462 if (rtcStats.candidate_type.is_defined())
463 stats.candidateType = iceCandidateState(*rtcStats.candidate_type);
464
465 if (!stats.candidateType || stats.candidateType == RTCIceCandidateType::Prflx || stats.candidateType == RTCIceCandidateType::Host)
466 stats.address = { };
467
468 if (rtcStats.priority.is_defined())
469 stats.priority = *rtcStats.priority;
470 if (rtcStats.url.is_defined())
471 stats.url = fromStdString(*rtcStats.url);
472}

575static inline void initializeRTCStatsReportBackingMap(DOMMapAdapter& report, const webrtc::RTCStatsReport& rtcReport)
576{
577 for (const auto& rtcStats : rtcReport) {
578 if (rtcStats.type() == webrtc::RTCInboundRTPStreamStats::kType) {
579 RTCStatsReport::InboundRtpStreamStats stats;
580 fillInboundRtpStreamStats(stats, static_cast<const webrtc::RTCInboundRTPStreamStats&>(rtcStats));
581 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::InboundRtpStreamStats>>(stats.id, WTFMove(stats));
582 } else if (rtcStats.type() == webrtc::RTCOutboundRTPStreamStats::kType) {
583 RTCStatsReport::OutboundRtpStreamStats stats;
584 fillOutboundRtpStreamStats(stats, static_cast<const webrtc::RTCOutboundRTPStreamStats&>(rtcStats));
585 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::OutboundRtpStreamStats>>(stats.id, WTFMove(stats));
586 } else if (rtcStats.type() == webrtc::RTCMediaStreamTrackStats::kType) {
587 RTCStatsReport::MediaStreamTrackStats stats;
588 fillRTCMediaStreamTrackStats(stats, static_cast<const webrtc::RTCMediaStreamTrackStats&>(rtcStats));
589 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::MediaStreamTrackStats>>(stats.id, WTFMove(stats));
590 } else if (rtcStats.type() == webrtc::RTCDataChannelStats::kType) {
591 RTCStatsReport::DataChannelStats stats;
592 fillRTCDataChannelStats(stats, static_cast<const webrtc::RTCDataChannelStats&>(rtcStats));
593 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::DataChannelStats>>(stats.id, WTFMove(stats));
594 } else if (rtcStats.type() == webrtc::RTCIceCandidatePairStats::kType) {
595 RTCStatsReport::IceCandidatePairStats stats;
596 fillRTCIceCandidatePairStats(stats, static_cast<const webrtc::RTCIceCandidatePairStats&>(rtcStats));
597 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::IceCandidatePairStats>>(stats.id, WTFMove(stats));
598 } else if (rtcStats.type() == webrtc::RTCRemoteIceCandidateStats::kType || rtcStats.type() == webrtc::RTCLocalIceCandidateStats::kType) {
599 RTCStatsReport::IceCandidateStats stats;
600 fillRTCIceCandidateStats(stats, static_cast<const webrtc::RTCIceCandidateStats&>(rtcStats));
601 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::IceCandidateStats>>(stats.id, WTFMove(stats));
602 } else if (rtcStats.type() == webrtc::RTCCertificateStats::kType) {
603 RTCStatsReport::CertificateStats stats;
604 fillRTCCertificateStats(stats, static_cast<const webrtc::RTCCertificateStats&>(rtcStats));
605 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::CertificateStats>>(stats.id, WTFMove(stats));
606 } else if (rtcStats.type() == webrtc::RTCCodecStats::kType) {
607 RTCStatsReport::CodecStats stats;
608 fillRTCCodecStats(stats, static_cast<const webrtc::RTCCodecStats&>(rtcStats));
609 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::CodecStats>>(stats.id, WTFMove(stats));

611 RTCStatsReport::TransportStats stats;
612 fillRTCTransportStats(stats, static_cast<const webrtc::RTCTransportStats&>(rtcStats));
613 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::TransportStats>>(stats.id, WTFMove(stats));
614 } else if (rtcStats.type() == webrtc::RTCPeerConnectionStats::kType) {
615 RTCStatsReport::PeerConnectionStats stats;
616 fillRTCPeerConnectionStats(stats, static_cast<const webrtc::RTCPeerConnectionStats&>(rtcStats));
617 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::PeerConnectionStats>>(stats.id, WTFMove(stats));

627 RTCStatsReport::RemoteInboundRtpStreamStats stats;

629 report.set<IDLDOMString, IDLDictionary<RTCStatsReport::RemoteInboundRtpStreamStats>>(stats.id, WTFMove(stats));
