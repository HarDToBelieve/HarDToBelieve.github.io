99void HTMLModelElement::sourcesChanged()
100{
101 if (!document().hasBrowsingContext()) {
102 setSourceURL(URL { });
103 return;
104 }
105
106 for (auto& element : childrenOfType<HTMLSourceElement>(*this)) {
107 // FIXME: for now we use the first valid URL without looking at the mime-type.
108 auto url = element.getNonEmptyURLAttribute(HTMLNames::srcAttr);
109 if (url.isValid()) {
110 setSourceURL(url);
111 return;
112 }
113 }
114
115 setSourceURL(URL { });
116}
117
118void HTMLModelElement::setSourceURL(const URL& url)
119{
120 if (url == m_sourceURL)
121 return;
122
123 m_sourceURL = url;
124
125 m_data = nullptr;
126 m_dataComplete = false;
127
128 if (m_resource) {
129 m_resource->removeClient(*this);
130 m_resource = nullptr;
131 }
132
133 if (!m_readyPromise->isFulfilled())
134 m_readyPromise->reject(Exception { AbortError });
135
136 m_readyPromise = makeUniqueRef<ReadyPromise>(*this, &HTMLModelElement::readyPromiseResolve);
137
138 if (m_sourceURL.isEmpty()) {
139#if HAVE(ARKIT_INLINE_PREVIEW_MAC)
140 clearFile();
141#endif
142 return;
143 }
144
145 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
146 options.destination = FetchOptions::Destination::Model;
147 // FIXME: Set other options.
148
149 auto crossOriginAttribute = parseCORSSettingsAttribute(attributeWithoutSynchronization(HTMLNames::crossoriginAttr));
150 auto request = createPotentialAccessControlRequest(ResourceRequest { m_sourceURL }, WTFMove(options), document(), crossOriginAttribute);
151 request.setInitiator(*this);
152
153 auto resource = document().cachedResourceLoader().requestModelResource(WTFMove(request));
154 if (!resource.has_value()) {
155 m_readyPromise->reject(Exception { NetworkError });
156 return;
157 }
158
159 m_data = SharedBuffer::create();
160
161 m_resource = resource.value();
162 m_resource->addClient(*this);
163}

212void HTMLModelElement::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
213{
214 auto invalidateResourceHandleAndUpdateRenderer = [&] {
215 m_resource->removeClient(*this);
216 m_resource = nullptr;
217
218 if (auto* renderer = this->renderer())
219 renderer->updateFromElement();
220 };
221
222 if (resource.loadFailedOrCanceled()) {
223 m_data = nullptr;
224
225 invalidateResourceHandleAndUpdateRenderer();
226
227 m_readyPromise->reject(Exception { NetworkError });
228 return;
229 }
230
231 m_dataComplete = true;
232 m_model = Model::create(*m_data);
233
234 invalidateResourceHandleAndUpdateRenderer();
235
236 m_readyPromise->resolve(*this);
237
238#if HAVE(ARKIT_INLINE_PREVIEW_MAC)
239 modelDidChange();
240#endif
241}

243void HTMLModelElement::enterFullscreen()
244{
245#if HAVE(ARKIT_INLINE_PREVIEW_IOS)
246 auto* page = document().page();
247 if (!page)
248 return;
249
250 if (!is<RenderLayerModelObject>(this->renderer()))
251 return;
252
253 auto& renderLayerModelObject = downcast<RenderLayerModelObject>(*this->renderer());
254 if (!renderLayerModelObject.isComposited() || !renderLayerModelObject.layer() || !renderLayerModelObject.layer()->backing())
255 return;
256
257 auto* graphicsLayer = renderLayerModelObject.layer()->backing()->graphicsLayer();
258 if (!graphicsLayer)
259 return;
260
261 if (auto contentLayerId = graphicsLayer->contentsLayerIDForModel())
262 page->chrome().client().takeModelElementFullscreen(contentLayerId);
263#endif
264}
