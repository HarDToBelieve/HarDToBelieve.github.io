54Ref<Notification> Notification::create(Document& context, const String& title, const Options& options)
55{
56 auto notification = adoptRef(*new Notification(context, title, options));
57 notification->suspendIfNeeded();
58 notification->showSoon();
59 return notification;
60}

62Notification::Notification(Document& document, const String& title, const Options& options)
63 : ActiveDOMObject(document)
64 , m_title(title)
65 , m_direction(options.dir)
66 , m_lang(options.lang)
67 , m_body(options.body)
68 , m_tag(options.tag)
69 , m_state(Idle)
70{
71 if (!options.icon.isEmpty()) {
72 auto iconURL = document.completeURL(options.icon);
73 if (iconURL.isValid())
74 m_icon = iconURL;
75 }
76}

81void Notification::showSoon()
82{
83 queueTaskKeepingObjectAlive(*this, TaskSource::UserInteraction, [this] {
84 show();
85 });
86}
87
88void Notification::show()
89{
90 // prevent double-showing
91 if (m_state != Idle)
92 return;
93
94 auto* page = document()->page();
95 if (!page)
96 return;
97
98 auto& client = NotificationController::from(page)->client();
99
100 if (client.checkPermission(scriptExecutionContext()) != Permission::Granted) {
101 dispatchErrorEvent();
102 return;
103 }
104 if (client.show(this))
105 m_state = Showing;
106}
107
108void Notification::close()
109{
110 switch (m_state) {
111 case Idle:
112 break;
113 case Showing: {
114 if (auto* page = document()->page())
115 NotificationController::from(page)->client().cancel(this);
116 break;
117 }
118 case Closed:
119 break;
120 }
121}

133void Notification::stop()
134{
135 ActiveDOMObject::stop();
136
137 if (auto* page = document()->page())
138 NotificationController::from(page)->client().notificationObjectDestroyed(this);
139}

141void Notification::suspend(ReasonForSuspension)
142{
143 close();
144}
145
146void Notification::finalize()
147{
148 if (m_state == Closed)
149 return;
150 m_state = Closed;
151}

153void Notification::dispatchShowEvent()
154{
155 queueTaskToDispatchEvent(*this, TaskSource::UserInteraction, Event::create(eventNames().showEvent, Event::CanBubble::No, Event::IsCancelable::No));
156}

158void Notification::dispatchClickEvent()
159{
160 queueTaskKeepingObjectAlive(*this, TaskSource::UserInteraction, [this] {
161 WindowFocusAllowedIndicator windowFocusAllowed;
162 dispatchEvent(Event::create(eventNames().clickEvent, Event::CanBubble::No, Event::IsCancelable::No));
163 });
164}
165
166void Notification::dispatchCloseEvent()
167{
168 queueTaskToDispatchEvent(*this, TaskSource::UserInteraction, Event::create(eventNames().closeEvent, Event::CanBubble::No, Event::IsCancelable::No));
169 finalize();
170}

172void Notification::dispatchErrorEvent()
173{
174 queueTaskToDispatchEvent(*this, TaskSource::UserInteraction, Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::No));
175}

177auto Notification::permission(Document& document) -> Permission
178{
179 auto* page = document.page();
180 if (!page)
181 return Permission::Default;
182
183 if (!document.isSecureContext())
184 return Permission::Denied;
185
186 return NotificationController::from(document.page())->client().checkPermission(&document);
187}
188
189void Notification::requestPermission(Document& document, RefPtr<NotificationPermissionCallback>&& callback, Ref<DeferredPromise>&& promise)
190{
191 auto resolvePromiseAndCallback = [document = makeRef(document), callback = WTFMove(callback), promise = WTFMove(promise)](Permission permission) mutable {
192 document->eventLoop().queueTask(TaskSource::DOMManipulation, [callback = WTFMove(callback), promise = WTFMove(promise), permission]() mutable {
193 if (callback)
194 callback->handleEvent(permission);
195 promise->resolve<IDLEnumeration<NotificationPermission>>(permission);
196 });
197 };
198
199 auto* page = document.page();
200 if (!page)
201 return resolvePromiseAndCallback(Permission::Denied);
202
203 if (!document.isSecureContext()) {
204 document.addConsoleMessage(MessageSource::Security, MessageLevel::Error, "The Notification permission may only be requested in a secure context."_s);
205 return resolvePromiseAndCallback(Permission::Denied);
206 }
207
208 NotificationController::from(page)->client().requestPermission(document, WTFMove(resolvePromiseAndCallback));
209}

211void Notification::eventListenersDidChange()
212{
213 m_hasRelevantEventListener = hasEventListeners(eventNames().clickEvent)
214 || hasEventListeners(eventNames().closeEvent)
215 || hasEventListeners(eventNames().errorEvent)
216 || hasEventListeners(eventNames().showEvent);
217}

219// A Notification object must not be garbage collected while the list of notifications contains its corresponding
220// notification and it has an event listener whose type is click, show, close, or error.
221// See https://notifications.spec.whatwg.org/#garbage-collection
222bool Notification::virtualHasPendingActivity() const
223{
224 return m_state == Showing && m_hasRelevantEventListener;
225}
