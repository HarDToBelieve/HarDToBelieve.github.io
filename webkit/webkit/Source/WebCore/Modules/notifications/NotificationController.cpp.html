26#include "config.h"
27#include "NotificationController.h"
28
29#if ENABLE(NOTIFICATIONS)
30
31#include "NotificationClient.h"
32
33namespace WebCore {
34
35NotificationController::NotificationController(NotificationClient* client)
36 : m_client(*client)
37{
38 ASSERT(client);
39}
40
41NotificationController::~NotificationController()
42{
43 m_client.notificationControllerDestroyed();
44}
45
46NotificationClient* NotificationController::clientFrom(Page& page)
47{
48 auto* controller = NotificationController::from(&page);
49 if (!controller)
50 return nullptr;
51 return &controller->client();
52}
53
54const char* NotificationController::supplementName()
55{
56 return "NotificationController";
57}
58
59void provideNotification(Page* page, NotificationClient* client)
60{
61 NotificationController::provideTo(page, NotificationController::supplementName(), makeUnique<NotificationController>(client));
62}

64} // namespace WebCore

66#endif // ENABLE(NOTIFICATIONS)
