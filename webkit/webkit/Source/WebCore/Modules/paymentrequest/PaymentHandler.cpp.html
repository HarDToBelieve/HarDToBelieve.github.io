37RefPtr<PaymentHandler> PaymentHandler::create(Document& document, PaymentRequest& paymentRequest, const PaymentRequest::MethodIdentifier& identifier)
38{
39#if ENABLE(APPLE_PAY)
40 if (ApplePayPaymentHandler::handlesIdentifier(identifier))
41 return adoptRef(new ApplePayPaymentHandler(document, identifier, paymentRequest));
42#else
43 UNUSED_PARAM(document);
44 UNUSED_PARAM(paymentRequest);
45 UNUSED_PARAM(identifier);
46#endif
47
48 return nullptr;
49}
50
51ExceptionOr<void> PaymentHandler::canCreateSession(Document& document)
52{
53#if ENABLE(APPLE_PAY)
54 auto result = PaymentSession::canCreateSession(document);
55 if (result.hasException())
56 return Exception { SecurityError, result.releaseException().releaseMessage() };
57#else
58 UNUSED_PARAM(document);
59#endif
60
61 return { };
62}
63
64ExceptionOr<void> PaymentHandler::validateData(Document& document, JSC::JSValue data, const PaymentRequest::MethodIdentifier& identifier)
65{
66#if ENABLE(APPLE_PAY)
67 if (ApplePayPaymentHandler::handlesIdentifier(identifier))
68 return ApplePayPaymentHandler::validateData(document, data);
69#else
70 UNUSED_PARAM(document);
71 UNUSED_PARAM(data);
72 UNUSED_PARAM(identifier);
73#endif
74
75 return { };
76}
77
78bool PaymentHandler::enabledForContext(ScriptExecutionContext& context)
79{
80#if ENABLE(APPLE_PAY)
81 return PaymentSession::enabledForContext(context);
82#else
83 UNUSED_PARAM(context);
84 return false;
85#endif
86}
87
88bool PaymentHandler::hasActiveSession(Document& document)
89{
90#if ENABLE(APPLE_PAY)
91 return ApplePayPaymentHandler::hasActiveSession(document);
92#else
93 UNUSED_PARAM(document);
94 return false;
95#endif
96}
