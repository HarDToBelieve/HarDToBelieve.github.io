53ExceptionOr<void> PaymentRequestUpdateEvent::updateWith(Ref<DOMPromise>&& detailsPromise)
54{
55 if (!isTrusted())
56 return Exception { InvalidStateError };
57
58 if (m_waitForUpdate)
59 return Exception { InvalidStateError };
60
61 stopPropagation();
62 stopImmediatePropagation();
63
64 PaymentRequest::UpdateReason reason;
65 if (type() == eventNames().shippingaddresschangeEvent)
66 reason = PaymentRequest::UpdateReason::ShippingAddressChanged;
67 else if (type() == eventNames().shippingoptionchangeEvent)
68 reason = PaymentRequest::UpdateReason::ShippingOptionChanged;
69 else if (type() == eventNames().paymentmethodchangeEvent)
70 reason = PaymentRequest::UpdateReason::PaymentMethodChanged;
71 else {
72 ASSERT_NOT_REACHED();
73 return Exception { TypeError };
74 }
75
76 auto exception = downcast<PaymentRequest>(target())->updateWith(reason, WTFMove(detailsPromise));
77 if (exception.hasException())
78 return exception.releaseException();
79
80 m_waitForUpdate = true;
81 return { };
82}
