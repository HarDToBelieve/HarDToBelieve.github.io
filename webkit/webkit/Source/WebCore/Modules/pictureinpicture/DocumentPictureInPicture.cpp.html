40WTF_MAKE_ISO_ALLOCATED_IMPL(DocumentPictureInPicture);
41
42DocumentPictureInPicture::~DocumentPictureInPicture() = default;
43
44bool DocumentPictureInPicture::pictureInPictureEnabled(Document&)
45{
46 return supportsPictureInPicture();
47}
48
49void DocumentPictureInPicture::exitPictureInPicture(Document& document, Ref<DeferredPromise>&& promise)
50{
51 auto element = document.pictureInPictureElement();
52
53 if (!element) {
54 promise->reject(InvalidStateError);
55 return;
56 }
57
58 HTMLVideoElementPictureInPicture::from(*element)->exitPictureInPicture(WTFMove(promise));
59}
60
61DocumentPictureInPicture* DocumentPictureInPicture::from(Document& document)
62{
63 DocumentPictureInPicture* supplement = static_cast<DocumentPictureInPicture*>(Supplement<Document>::from(&document, supplementName()));
64 if (!supplement) {
65 auto newSupplement = makeUnique<DocumentPictureInPicture>();
66 supplement = newSupplement.get();
67 provideTo(&document, supplementName(), WTFMove(newSupplement));
68 }
69 return supplement;
70}
