39class HTMLPlugInElement;
40class RenderElement;
41class RenderStyle;

46class PluginReplacement : public RefCounted<PluginReplacement> {
47public:
48 virtual ~PluginReplacement() = default;
49
50 struct InstallResult {
51 bool success;
52#if PLATFORM(COCOA)
53 JSC::JSValue scriptObject { };
54#endif
55 };
56
57 virtual InstallResult installReplacement(ShadowRoot&) = 0;
58
59 virtual bool willCreateRenderer() { return false; }
60 virtual RenderPtr<RenderElement> createElementRenderer(HTMLPlugInElement&, RenderStyle&&, const RenderTreePosition&) = 0;
61};

63typedef Ref<PluginReplacement> (*CreatePluginReplacement)(HTMLPlugInElement&, const Vector<String>& paramNames, const Vector<String>& paramValues);
64typedef bool (*PluginReplacementSupportsType)(const String&);
65typedef bool (*PluginReplacementSupportsFileExtension)(const String&);
66typedef bool (*PluginReplacementSupportsURL)(const URL&);
67typedef bool (*PluginReplacementEnabledForSettings)(const Settings&);
68
69class ReplacementPlugin {
70public:
71 ReplacementPlugin(CreatePluginReplacement constructor, PluginReplacementSupportsType supportsType, PluginReplacementSupportsFileExtension supportsFileExtension, PluginReplacementSupportsURL supportsURL, PluginReplacementEnabledForSettings isEnabledBySettings)
72 : m_constructor(constructor)
73 , m_supportsType(supportsType)
74 , m_supportsFileExtension(supportsFileExtension)
75 , m_supportsURL(supportsURL)
76 , m_isEnabledBySettings(isEnabledBySettings)
77 {
78 }
79
80 explicit ReplacementPlugin(const ReplacementPlugin& other)
81 : m_constructor(other.m_constructor)
82 , m_supportsType(other.m_supportsType)
83 , m_supportsFileExtension(other.m_supportsFileExtension)
84 , m_supportsURL(other.m_supportsURL)
85 , m_isEnabledBySettings(other.m_isEnabledBySettings)
86 {
87 }
88
89 Ref<PluginReplacement> create(HTMLPlugInElement& element, const Vector<String>& paramNames, const Vector<String>& paramValues) const { return m_constructor(element, paramNames, paramValues); }
90 bool supportsType(const String& mimeType) const { return m_supportsType(mimeType); }
91 bool supportsFileExtension(const String& extension) const { return m_supportsFileExtension(extension); }
92 bool supportsURL(const URL& url) const { return m_supportsURL(url); }
93 bool isEnabledBySettings(const Settings& settings) const { return m_isEnabledBySettings(settings); };
94
95private:
96 CreatePluginReplacement m_constructor;
97 PluginReplacementSupportsType m_supportsType;
98 PluginReplacementSupportsFileExtension m_supportsFileExtension;
99 PluginReplacementSupportsURL m_supportsURL;
100 PluginReplacementEnabledForSettings m_isEnabledBySettings;
101};
