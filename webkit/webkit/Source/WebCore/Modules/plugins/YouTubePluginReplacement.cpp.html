62YouTubePluginReplacement::YouTubePluginReplacement(HTMLPlugInElement& plugin, const Vector<String>& paramNames, const Vector<String>& paramValues)
63 : m_parentElement(makeWeakPtr(plugin))
64{
65 ASSERT(paramNames.size() == paramValues.size());
66 for (size_t i = 0; i < paramNames.size(); ++i)
67 m_attributes.add(paramNames[i], paramValues[i]);
68}

80auto YouTubePluginReplacement::installReplacement(ShadowRoot& root) -> InstallResult
81{
82 m_embedShadowElement = YouTubeEmbedShadowElement::create(m_parentElement->document());
83
84 root.appendChild(*m_embedShadowElement);
85
86 auto iframeElement = HTMLIFrameElement::create(HTMLNames::iframeTag, m_parentElement->document());
87 if (m_attributes.contains("width"))
88 iframeElement->setAttributeWithoutSynchronization(HTMLNames::widthAttr, AtomString("100%", AtomString::ConstructFromLiteral));
89
90 const auto& heightValue = m_attributes.find("height");
91 if (heightValue != m_attributes.end()) {
92 iframeElement->setAttribute(HTMLNames::styleAttr, AtomString("max-height: 100%", AtomString::ConstructFromLiteral));
93 iframeElement->setAttributeWithoutSynchronization(HTMLNames::heightAttr, heightValue->value);
94 }
95
96 iframeElement->setAttributeWithoutSynchronization(HTMLNames::srcAttr, youTubeURL(m_attributes.get("src")));
97 iframeElement->setAttributeWithoutSynchronization(HTMLNames::frameborderAttr, AtomString("0", AtomString::ConstructFromLiteral));
98
99 // Disable frame flattening for this iframe.
100 iframeElement->setAttributeWithoutSynchronization(HTMLNames::scrollingAttr, AtomString("no", AtomString::ConstructFromLiteral));
101 m_embedShadowElement->appendChild(iframeElement);
102
103 return { true };
104}

106static URL createYouTubeURL(StringView videoID, StringView timeID)
107{
108 ASSERT(!videoID.isEmpty());
109 ASSERT(videoID != "/");
110 return URL(URL(), makeString("youtube:", videoID, timeID.isEmpty() ? "" : "t=", timeID));
111}

113static YouTubePluginReplacement::KeyValueMap queryKeysAndValues(StringView queryString)
114{
115 YouTubePluginReplacement::KeyValueMap queryDictionary;
116
117 size_t queryLength = queryString.length();
118 if (!queryLength)
119 return queryDictionary;
120
121 size_t equalSearchLocation = 0;
122 size_t equalSearchLength = queryLength;
123
124 while (equalSearchLocation < queryLength - 1 && equalSearchLength) {
125
126 // Search for "=".
127 size_t equalLocation = queryString.find('=', equalSearchLocation);
128 if (equalLocation == notFound)
129 break;
130
131 size_t indexAfterEqual = equalLocation + 1;
132 if (indexAfterEqual > queryLength - 1)
133 break;
134
135 // Get the key before the "=".
136 size_t keyLocation = equalSearchLocation;
137 size_t keyLength = equalLocation - equalSearchLocation;
138
139 // Seach for the ampersand.
140 size_t ampersandLocation = queryString.find('&', indexAfterEqual);
141
142 // Get the value after the "=", before the ampersand.
143 size_t valueLocation = indexAfterEqual;
144 size_t valueLength;
145 if (ampersandLocation != notFound)
146 valueLength = ampersandLocation - indexAfterEqual;
147 else
148 valueLength = queryLength - indexAfterEqual;
149
150 // Save the key and the value.
151 if (keyLength && valueLength) {
152 String key = queryString.substring(keyLocation, keyLength).convertToASCIILowercase();
153 String value = queryString.substring(valueLocation, valueLength).toString();
154 value.replace('+', ' ');
155
156 if (!key.isEmpty() && !value.isEmpty())
157 queryDictionary.add(key, value);
158 }
159
160 if (ampersandLocation == notFound)
161 break;
162
163 // Continue searching after the ampersand.
164 size_t indexAfterAmpersand = ampersandLocation + 1;
165 equalSearchLocation = indexAfterAmpersand;
166 equalSearchLength = queryLength - indexAfterAmpersand;
167 }
168
169 return queryDictionary;
170}

172static bool isYouTubeURL(const URL& url)
173{
174 auto hostName = url.host();
175 return equalLettersIgnoringASCIICase(hostName, "m.youtube.com")
176 || equalLettersIgnoringASCIICase(hostName, "youtu.be")
177 || equalLettersIgnoringASCIICase(hostName, "www.youtube.com")
178 || equalLettersIgnoringASCIICase(hostName, "youtube.com")
179 || equalLettersIgnoringASCIICase(hostName, "www.youtube-nocookie.com")
180 || equalLettersIgnoringASCIICase(hostName, "youtube-nocookie.com");
181}

183static const String& valueForKey(const YouTubePluginReplacement::KeyValueMap& dictionary, const String& key)
184{
185 const auto& value = dictionary.find(key);
186 if (value == dictionary.end())
187 return emptyString();
188
189 return value->value;
190}

192static URL processAndCreateYouTubeURL(const URL& url, bool& isYouTubeShortenedURL, String& outPathAfterFirstAmpersand)
193{
194 if (!url.protocolIsInHTTPFamily())
195 return URL();
196
197 // Bail out early if we aren't even on www.youtube.com or youtube.com.
198 if (!isYouTubeURL(url))
199 return URL();
200
201 auto hostName = url.host();
202 bool isYouTubeMobileWebAppURL = equalLettersIgnoringASCIICase(hostName, "m.youtube.com");
203 isYouTubeShortenedURL = equalLettersIgnoringASCIICase(hostName, "youtu.be");
204
205 // Short URL of the form: http://youtu.be/v1d301D
206 if (isYouTubeShortenedURL) {
207 auto videoID = url.lastPathComponent();
208 if (videoID.isEmpty() || videoID == "/")
209 return URL();
210 return createYouTubeURL(videoID, { });
211 }
212
213 auto path = url.path();
214 auto query = url.query();
215 auto fragment = url.fragmentIdentifier();
216
217 // On the YouTube mobile web app, the path and query string are put into the
218 // fragment so that one web page is only ever loaded (see <rdar://problem/9550639>).
219 if (isYouTubeMobileWebAppURL) {
220 size_t location = fragment.find('?');
221 if (location == notFound) {
222 path = fragment;
223 query = emptyString();
224 } else {
225 path = fragment.substring(0, location);
226 query = fragment.substring(location + 1);
227 }
228 fragment = emptyString();
229 }
230
231 if (equalLettersIgnoringASCIICase(path, "/watch")) {
232 if (!query.isEmpty()) {
233 const auto& queryDictionary = queryKeysAndValues(query);
234 String videoID = valueForKey(queryDictionary, "v");
235
236 if (!videoID.isEmpty()) {
237 const auto& fragmentDictionary = queryKeysAndValues(url.fragmentIdentifier());
238 String timeID = valueForKey(fragmentDictionary, "t");
239 return createYouTubeURL(videoID, timeID);
240 }
241 }
242
243 // May be a new-style link (see <rdar://problem/7733692>).
244 if (fragment.startsWith('!')) {
245 query = fragment.substring(1);
246
247 if (!query.isEmpty()) {
248 const auto& queryDictionary = queryKeysAndValues(query);
249 String videoID = valueForKey(queryDictionary, "v");
250
251 if (!videoID.isEmpty()) {
252 String timeID = valueForKey(queryDictionary, "t");
253 return createYouTubeURL(videoID, timeID);
254 }
255 }
256 }
257 } else if (startsWithLettersIgnoringASCIICase(path, "/v/") || startsWithLettersIgnoringASCIICase(path, "/e/")) {
258 StringView videoID;
259 StringView pathAfterFirstAmpersand;
260
261 auto lastPathComponent = url.lastPathComponent();
262 size_t ampersandLocation = lastPathComponent.find('&');
263 if (ampersandLocation != notFound) {
264 // Some URLs we care about use & in place of ? for the first query parameter.
265 videoID = lastPathComponent.substring(0, ampersandLocation);
266 pathAfterFirstAmpersand = lastPathComponent.substring(ampersandLocation + 1, lastPathComponent.length() - ampersandLocation);
267 } else
268 videoID = lastPathComponent;
269
270 if (!videoID.isEmpty()) {
271 outPathAfterFirstAmpersand = pathAfterFirstAmpersand.toString();
272 return createYouTubeURL(videoID, emptyString());
273 }
274 }
275
276 return URL();
277}

279String YouTubePluginReplacement::youTubeURL(const String& srcString)
280{
281 URL srcURL = m_parentElement->document().completeURL(stripLeadingAndTrailingHTMLSpaces(srcString));
282 return youTubeURLFromAbsoluteURL(srcURL, srcString);
283}

285String YouTubePluginReplacement::youTubeURLFromAbsoluteURL(const URL& srcURL, const String& srcString)
286{
287 bool isYouTubeShortenedURL = false;
288 String possiblyMalformedQuery;
289 URL youTubeURL = processAndCreateYouTubeURL(srcURL, isYouTubeShortenedURL, possiblyMalformedQuery);
290 if (srcURL.isEmpty() || youTubeURL.isEmpty())
291 return srcString;
292
293 // Transform the youtubeURL (youtube:VideoID) to iframe embed url which has the format: http://www.youtube.com/embed/VideoID
294 auto srcPath = srcURL.path();
295 const String& videoID = youTubeURL.string().substring(youTubeURL.protocol().length() + 1);
296 size_t locationOfVideoIDInPath = srcPath.find(videoID);
297
298 size_t locationOfPathBeforeVideoID = notFound;
299 if (locationOfVideoIDInPath != notFound) {
300 ASSERT(locationOfVideoIDInPath);
301
302 // From the original URL, we need to get the part before /path/VideoId.
303 locationOfPathBeforeVideoID = StringView(srcString).find(srcPath.substring(0, locationOfVideoIDInPath));
304 } else if (equalLettersIgnoringASCIICase(srcPath, "/watch")) {
305 // From the original URL, we need to get the part before /watch/#!v=VideoID
306 // FIXME: Shouldn't this be ASCII case-insensitive?
307 locationOfPathBeforeVideoID = srcString.find("/watch");
308 } else
309 return srcString;
310
311 ASSERT(locationOfPathBeforeVideoID != notFound);
312
313 auto srcURLPrefix = StringView(srcString).substring(0, locationOfPathBeforeVideoID);
314 auto query = srcURL.query();
315
316 // If the URL has no query, use the possibly malformed query we found.
317 if (query.isEmpty())
318 query = possiblyMalformedQuery;
319
320 // Append the query string if it is valid.
321 return makeString(
322 isYouTubeShortenedURL ? "http://www.youtube.com" : srcURLPrefix,
323 "/embed/",
324 videoID,
325 query.isEmpty() ? "" : "?",
326 query
327 );
328}
