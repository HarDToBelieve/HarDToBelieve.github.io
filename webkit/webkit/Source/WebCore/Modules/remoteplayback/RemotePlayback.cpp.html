73void RemotePlayback::watchAvailability(Ref<RemotePlaybackAvailabilityCallback>&& callback, Ref<DeferredPromise>&& promise)
74{
75 // 6.2.1.3 Getting the remote playback devices availability information
76 // https://w3c.github.io/remote-playback/#monitoring-the-list-of-available-remote-playback-devices
77 // W3C Editor's Draft 15 July 2016
78
79 // 1. Let promise be a new promise->
80 // 2. Return promise, and run the following steps below:
81
82 queueTaskKeepingObjectAlive(*this, TaskSource::MediaElement, [this, callback = WTFMove(callback), promise = WTFMove(promise)] () mutable {
83 if (isContextStopped())
84 return;
85
86 // 3. If the disableRemotePlayback attribute is present for the media element, reject the promise with
87 // InvalidStateError and abort all the remaining steps.
88 if (!m_mediaElement
89 || m_mediaElement->hasAttributeWithoutSynchronization(HTMLNames::webkitwirelessvideoplaybackdisabledAttr)
90 || m_mediaElement->hasAttributeWithoutSynchronization(HTMLNames::disableremoteplaybackAttr)) {
91 WTFLogAlways("RemotePlayback::watchAvailability()::task - promise rejected");
92 promise->reject(InvalidStateError);
93 return;
94 }
95
96 // 4. If the user agent is unable to monitor the list of available remote playback devices for the entire
97 // lifetime of the browsing context (for instance, because the user has disabled this feature), then run
98 // the following steps in parallel:
99 // 5. If the user agent is unable to continuously monitor the list of available remote playback devices but
100 // can do it for a short period of time when initiating remote playback, then:
101 // NOTE: Unimplemented; all current ports can support continuous device monitoring
102
103 // 6. Let callbackId be a number unique to the media element that will identify the callback.
104 int32_t callbackId = ++m_nextId;
105
106 // 7. Create a tuple (callbackId, callback) and add it to the set of availability callbacks for this media element.
107 ASSERT(!m_callbackMap.contains(callbackId));
108 m_callbackMap.add(callbackId, WTFMove(callback));
109
110 // 8. Fulfill promise with the callbackId and run the following steps in parallel:
111 promise->whenSettled([this, protectedThis = makeRefPtr(this), callbackId] {
112 // 8.1 Queue a task to invoke the callback with the current availability for the media element.
113 queueTaskKeepingObjectAlive(*this, TaskSource::MediaElement, [this, callbackId, available = m_available] {
114 if (isContextStopped())
115 return;
116 auto foundCallback = m_callbackMap.find(callbackId);
117 if (foundCallback == m_callbackMap.end())
118 return;
119
120 foundCallback->value->handleEvent(available);
121 });
122
123 // 8.2 Run the algorithm to monitor the list of available remote playback devices.
124 if (m_mediaElement) {
125 availabilityChanged(m_mediaElement->mediaSession().hasWirelessPlaybackTargets());
126 m_mediaElement->remoteHasAvailabilityCallbacksChanged();
127 }
128 });
129 promise->resolve<IDLLong>(callbackId);
130 });
131}

133void RemotePlayback::cancelWatchAvailability(std::optional<int32_t> id, Ref<DeferredPromise>&& promise)
134{
135 // 6.2.1.5 Stop observing remote playback devices availability
136 // https://w3c.github.io/remote-playback/#stop-observing-remote-playback-devices-availability
137 // W3C Editor's Draft 15 July 2016
138
139 // 1. Let promise be a new promise->
140 // 2. Return promise, and run the following steps below:
141
142 queueTaskKeepingObjectAlive(*this, TaskSource::MediaElement, [this, id = WTFMove(id), promise = WTFMove(promise)] {
143 if (isContextStopped())
144 return;
145 // 3. If the disableRemotePlayback attribute is present for the media element, reject promise with
146 // InvalidStateError and abort all the remaining steps.
147 if (!m_mediaElement
148 || m_mediaElement->hasAttributeWithoutSynchronization(HTMLNames::webkitwirelessvideoplaybackdisabledAttr)
149 || m_mediaElement->hasAttributeWithoutSynchronization(HTMLNames::disableremoteplaybackAttr)) {
150 promise->reject(InvalidStateError);
151 return;
152 }
153
154 // 4. If the parameter id is undefined, clear the set of availability callbacks.
155 if (!id)
156 m_callbackMap.clear();
157 else {
158 // 5. Otherwise, if id matches the callbackId for any entry in the set of availability callbacks,
159 // remove the entry from the set.
160 if (auto it = m_callbackMap.find(id.value()) != m_callbackMap.end())
161 m_callbackMap.remove(it);
162 // 6. Otherwise, reject promise with NotFoundError and abort all the remaining steps.
163 else {
164 promise->reject(NotFoundError);
165 return;
166 }
167 }
168 // 7. If the set of availability callbacks is now empty and there is no pending request to initiate remote
169 // playback, cancel any pending task to monitor the list of available remote playback devices for power
170 // saving purposes.
171 m_mediaElement->remoteHasAvailabilityCallbacksChanged();
172
173 // 8. Fulfill promise.
174 promise->resolve();
175 });
176}

178void RemotePlayback::prompt(Ref<DeferredPromise>&& promise)
179{
180 // 6.2.2 Prompt user for changing remote playback statee
181 // https://w3c.github.io/remote-playback/#stop-observing-remote-playback-devices-availability
182 // W3C Editor's Draft 15 July 2016
183
184 // 1. Let promise be a new promise->
185 // 2. Return promise, and run the following steps below:
186
187 queueTaskKeepingObjectAlive(*this, TaskSource::MediaElement, [this, promise = WTFMove(promise), processingUserGesture = UserGestureIndicator::processingUserGesture()] () mutable {
188 if (isContextStopped())
189 return;
190
191 // 3. If the disableRemotePlayback attribute is present for the media element, reject the promise with
192 // InvalidStateError and abort all the remaining steps.
193 if (!m_mediaElement
194 || m_mediaElement->hasAttributeWithoutSynchronization(HTMLNames::webkitwirelessvideoplaybackdisabledAttr)
195 || m_mediaElement->hasAttributeWithoutSynchronization(HTMLNames::disableremoteplaybackAttr)) {
196 promise->reject(InvalidStateError);
197 return;
198 }
199
200 // 4. If there is already an unsettled promise from a previous call to prompt for the same media element
201 // or even for the same browsing context, the user agent may reject promise with an OperationError
202 // exception and abort all remaining steps.
203 // NOTE: consider implementing
204
205 // 5. OPTIONALLY, if the user agent knows a priori that showing the UI for this particular media element
206 // is not feasible, reject promise with a NotSupportedError and abort all remaining steps.
207#if !PLATFORM(IOS)
208 if (m_mediaElement->readyState() < HTMLMediaElementEnums::HAVE_METADATA) {
209 promise->reject(NotSupportedError);
210 return;
211 }
212#endif
213
214 // 6. If the algorithm isn't allowed to show a popup, reject promise with an InvalidAccessError exception
215 // and abort these steps.
216 if (!processingUserGesture) {
217 promise->reject(InvalidAccessError);
218 return;
219 }
220
221 // 7. If the user agent needs to show the list of available remote playback devices and is not monitoring
222 // the list of available remote playback devices, run the steps to monitor the list of available remote
223 // playback devices in parallel.
224 // NOTE: Monitoring enabled by adding to m_promptPromises and calling remoteHasAvailabilityCallbacksChanged().
225
226 // 8. If the list of available remote playback devices is empty and will remain so before the request for
227 // user permission is completed, reject promise with a NotFoundError exception and abort all remaining steps.
228 // NOTE: consider implementing (no network?)
229
230 // 9. If the state is disconnected and availability for the media element is false, reject promise with a
231 // NotSupportedError exception and abort all remaining steps.
232 if (m_state == State::Disconnected && !m_available) {
233 promise->reject(NotSupportedError);
234 return;
235 }
236
237 m_promptPromises.append(WTFMove(promise));
238 availabilityChanged(m_mediaElement->mediaSession().hasWirelessPlaybackTargets());
239 m_mediaElement->remoteHasAvailabilityCallbacksChanged();
240 m_mediaElement->webkitShowPlaybackTargetPicker();
241
242 // NOTE: Steps 10-12 are implemented in the following methods:
243 });

246void RemotePlayback::shouldPlayToRemoteTargetChanged(bool shouldPlayToRemoteTarget)
247{
248 // 6.2.2 Prompt user for changing remote playback state [Ctd]
249 // https://w3c.github.io/remote-playback/#prompt-user-for-changing-remote-playback-statee
250 // W3C Editor's Draft 15 July 2016
251
252 LOG(Media, "RemotePlayback::shouldPlayToRemoteTargetChanged(%p), shouldPlay(%d), promise count(%lu)", this, shouldPlayToRemoteTarget, m_promptPromises.size());
253
254 // 10. If the user picked a remote playback device device to initiate remote playback with, the user agent
255 // must run the following steps:
256 if (shouldPlayToRemoteTarget) {
257 // 10.1 Set the state of the remote object to connecting.
258 // 10.3 Queue a task to fire a simple event with the name connecting at the remote property of the media element.
259 // The event must not bubble, must not be cancelable, and has no default action.
260 setState(State::Connecting);
261 }
262
263 for (auto& promise : std::exchange(m_promptPromises, { })) {
264 // 10.2 Fulfill promise.
265 // 10.4 Establish a connection with the remote playback device device for the media element.
266 // NOTE: Implemented in establishConnection().
267
268 // 11. Otherwise, if the user chose to disconnect from the remote playback device device, the user agent
269 // must run the following steps:
270 // 11.1. Fulfill promise.
271 // 11.2. Run the disconnect from remote playback device algorithm for the device.
272 // NOTE: Implemented in disconnect().
273
274 promise->resolve();
275 }
276
277 if (shouldPlayToRemoteTarget)
278 establishConnection();
279 else
280 disconnect();
281
282 if (m_mediaElement)
283 m_mediaElement->remoteHasAvailabilityCallbacksChanged();
284}

286void RemotePlayback::setState(State state)
287{
288 if (m_state == state)
289 return;
290
291 m_state = state;
292
293 auto eventName = [](State state) {
294 switch (state) {
295 case State::Connected:
296 return eventNames().connectEvent;
297 case State::Connecting:
298 return eventNames().connectingEvent;
299 case State::Disconnected:
300 return eventNames().disconnectEvent;
301 }
302 };
303
304 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, Event::create(eventName(state), Event::CanBubble::No, Event::IsCancelable::No));
305}

350void RemotePlayback::playbackTargetPickerWasDismissed()
351{
352 if (m_promptPromises.isEmpty())
353 return;
354
355 // 6.2.2 Prompt user for changing remote playback state [Ctd]
356 // https://w3c.github.io/remote-playback/#stop-observing-remote-playback-devices-availability
357 // W3C Editor's Draft 15 July 2016
358
359 // 12. Otherwise, the user is considered to deny permission to use the device, so reject promise with NotAllowedError
360 // exception and hide the UI shown by the user agent
361 ASSERT(!m_promptPromises.isEmpty());
362
363 for (auto& promise : std::exchange(m_promptPromises, { }))
364 promise->reject(NotAllowedError);
365
366 if (m_mediaElement)
367 m_mediaElement->remoteHasAvailabilityCallbacksChanged();

399void RemotePlayback::availabilityChanged(bool available)
400{
401 if (available == m_available)
402 return;
403 m_available = available;
404
405 queueTaskKeepingObjectAlive(*this, TaskSource::MediaElement, [this, available] {
406 if (isContextStopped())
407 return;
408
409 // Protect m_callbackMap against mutation while it's being iterated over.
410 Vector<Ref<RemotePlaybackAvailabilityCallback>> callbacks;
411 callbacks.reserveInitialCapacity(m_callbackMap.size());
412
413 // Can't use copyValuesToVector() here because Ref<> has a deleted assignment operator.
414 for (auto& callback : m_callbackMap.values())
415 callbacks.uncheckedAppend(callback.copyRef());
416 for (auto& callback : callbacks)
417 callback->handleEvent(available);
418 });
419}
