41DOMWindowSpeechSynthesis::DOMWindowSpeechSynthesis(DOMWindow* window)
42 : DOMWindowProperty(window)
43{
44}

46DOMWindowSpeechSynthesis::~DOMWindowSpeechSynthesis() = default;
47
48const char* DOMWindowSpeechSynthesis::supplementName()
49{
50 return "DOMWindowSpeechSynthesis";
51}
52
53// static
54DOMWindowSpeechSynthesis* DOMWindowSpeechSynthesis::from(DOMWindow* window)
55{
56 DOMWindowSpeechSynthesis* supplement = static_cast<DOMWindowSpeechSynthesis*>(Supplement<DOMWindow>::from(window, supplementName()));
57 if (!supplement) {
58 auto newSupplement = makeUnique<DOMWindowSpeechSynthesis>(window);
59 supplement = newSupplement.get();
60 provideTo(window, supplementName(), WTFMove(newSupplement));
61 }
62 return supplement;
63}
64
65// static
66SpeechSynthesis* DOMWindowSpeechSynthesis::speechSynthesis(DOMWindow& window)
67{
68 return DOMWindowSpeechSynthesis::from(&window)->speechSynthesis();
69}
70
71SpeechSynthesis* DOMWindowSpeechSynthesis::speechSynthesis()
72{
73 if (!m_speechSynthesis && frame() && frame()->document())
74 m_speechSynthesis = SpeechSynthesis::create(makeWeakPtr(frame()->page()->speechSynthesisClient()), *frame()->document());
75 return m_speechSynthesis.get();
76}
