27
28#include "ActiveDOMObject.h"
29#include "EventTarget.h"
30#include "SpeechRecognitionConnection.h"
31#include "SpeechRecognitionConnectionClient.h"
32#include "SpeechRecognitionResult.h"

34namespace WebCore {
35
36class Document;
37class SpeechRecognitionResult;
38
39class SpeechRecognition : public SpeechRecognitionConnectionClient, public ActiveDOMObject, public RefCounted<SpeechRecognition>, public EventTargetWithInlineData {
40 WTF_MAKE_ISO_ALLOCATED(SpeechRecognition);
41public:
42 static Ref<SpeechRecognition> create(Document&);
43
44 using SpeechRecognitionConnectionClient::weakPtrFactory;
45 using WeakValueType = SpeechRecognitionConnectionClient::WeakValueType;
46
47 const String& lang() const { return m_lang; }
48 void setLang(String&& lang) { m_lang = WTFMove(lang); }
49
50 bool continuous() const { return m_continuous; }
51 void setContinuous(bool continuous) { m_continuous = continuous; }
52
53 bool interimResults() const { return m_interimResults; }
54 void setInterimResults(bool interimResults) { m_interimResults = interimResults; }
55
56 uint64_t maxAlternatives() const { return m_maxAlternatives; }
57 void setMaxAlternatives(unsigned maxAlternatives) { m_maxAlternatives = maxAlternatives; }
58
59 ExceptionOr<void> startRecognition();
60 void stopRecognition();
61 void abortRecognition();
62
63 using RefCounted::ref;
64 using RefCounted::deref;
65
66private:
67 enum class State {
68 Inactive,
69 Starting,
70 Running,
71 Stopping,
72 Aborting,
73 };
74
75 explicit SpeechRecognition(Document&);
76
77 // SpeechRecognitionConnectionClient
78 void didStart() final;
79 void didStartCapturingAudio() final;
80 void didStartCapturingSound() final;
81 void didStartCapturingSpeech() final;
82 void didStopCapturingSpeech() final;
83 void didStopCapturingSound() final;
84 void didStopCapturingAudio() final;
85 void didFindNoMatch() final;
86 void didReceiveResult(Vector<SpeechRecognitionResultData>&& resultDatas) final;
87 void didError(const SpeechRecognitionError&) final;
88 void didEnd() final;
89
90 // ActiveDOMObject
91 const char* activeDOMObjectName() const final;
92 void suspend(ReasonForSuspension) final;
93 void stop() final;
94
95 // EventTarget
96 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
97 EventTargetInterface eventTargetInterface() const final { return SpeechRecognitionEventTargetInterfaceType; }
98 void refEventTarget() final { ref(); }
99 void derefEventTarget() final { deref(); }
100
101 String m_lang;
102 bool m_continuous { false };
103 bool m_interimResults { false };
104 uint64_t m_maxAlternatives { 1 };
105
106 State m_state { State::Inactive };
107 Vector<Ref<SpeechRecognitionResult>> m_finalResults;
108 RefPtr<SpeechRecognitionConnection> m_connection;
109};

111} // namespace WebCore
