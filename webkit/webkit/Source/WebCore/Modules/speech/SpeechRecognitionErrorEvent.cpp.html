31namespace WebCore {

33WTF_MAKE_ISO_ALLOCATED_IMPL(SpeechRecognitionErrorEvent);
34
35Ref<SpeechRecognitionErrorEvent> SpeechRecognitionErrorEvent::create(const AtomString& type, Init&& init, IsTrusted isTrusted)
36{
37 return adoptRef(*new SpeechRecognitionErrorEvent(type, WTFMove(init), isTrusted));
38}
39
40Ref<SpeechRecognitionErrorEvent> SpeechRecognitionErrorEvent::create(const AtomString& type, SpeechRecognitionErrorCode error, const String& message)
41{
42 return adoptRef(*new SpeechRecognitionErrorEvent(type, error, message));
43}
44
45SpeechRecognitionErrorEvent::SpeechRecognitionErrorEvent(const AtomString& type, Init&& init, IsTrusted isTrusted)
46 : Event(type, init, isTrusted)
47 , m_error(init.error)
48 , m_message(WTFMove(init.message))
49{
50}
51
52SpeechRecognitionErrorEvent::SpeechRecognitionErrorEvent(const AtomString& type, SpeechRecognitionErrorCode error, const String& message)
53 : Event(type, Event::CanBubble::No, Event::IsCancelable::No)
54 , m_error(error)
55 , m_message(message)
56{
57}

59EventInterface SpeechRecognitionErrorEvent::eventInterface() const
60{
61 return SpeechRecognitionErrorEventInterfaceType;
62}
