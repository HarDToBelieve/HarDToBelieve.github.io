36Ref<SpeechRecognitionEvent> SpeechRecognitionEvent::create(const AtomString& type, Init&& init, IsTrusted isTrusted)
37{
38 return adoptRef(*new SpeechRecognitionEvent(type, WTFMove(init), isTrusted));
39}
40
41Ref<SpeechRecognitionEvent> SpeechRecognitionEvent::create(const AtomString& type, uint64_t resultIndex, RefPtr<SpeechRecognitionResultList>&& results)
42{
43 return adoptRef(*new SpeechRecognitionEvent(type, resultIndex, WTFMove(results)));
44}
45
46SpeechRecognitionEvent::SpeechRecognitionEvent(const AtomString& type, Init&& init, IsTrusted isTrusted)
47 : Event(type, init, isTrusted)
48 , m_resultIndex(init.resultIndex)
49 , m_results(WTFMove(init.results))
50{
51}
52
53SpeechRecognitionEvent::SpeechRecognitionEvent(const AtomString& type, uint64_t resultIndex, RefPtr<SpeechRecognitionResultList>&& results)
54 : Event(type, Event::CanBubble::No, Event::IsCancelable::No)
55 , m_resultIndex(resultIndex)
56 , m_results(WTFMove(results))
57{
58}

60EventInterface SpeechRecognitionEvent::eventInterface() const
