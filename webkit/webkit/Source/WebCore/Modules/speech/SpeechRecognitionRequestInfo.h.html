26#pragma once
27
28#include "ClientOrigin.h"
29#include "FrameIdentifier.h"
30#include "SpeechRecognitionConnectionClientIdentifier.h"
31
32namespace WebCore {
33
34struct SpeechRecognitionRequestInfo {
35 SpeechRecognitionConnectionClientIdentifier clientIdentifier;
36 String lang;
37 bool continuous { false };
38 bool interimResults { false };
39 uint64_t maxAlternatives { 1 };
40 ClientOrigin clientOrigin;
41 FrameIdentifier frameIdentifier;
42
43 template<class Encoder> void encode(Encoder&) const;
44 template<class Decoder> static std::optional<SpeechRecognitionRequestInfo> decode(Decoder&);
45};
46
47template<class Encoder>
48void SpeechRecognitionRequestInfo::encode(Encoder& encoder) const
49{
50 encoder << clientIdentifier << lang << continuous << interimResults << maxAlternatives << clientOrigin << frameIdentifier;
51}
52
53template<class Decoder>
54std::optional<SpeechRecognitionRequestInfo> SpeechRecognitionRequestInfo::decode(Decoder& decoder)
55{
56 std::optional<SpeechRecognitionConnectionClientIdentifier> clientIdentifier;
57 decoder >> clientIdentifier;
58 if (!clientIdentifier)
59 return std::nullopt;
60
61 std::optional<String> lang;
62 decoder >> lang;
63 if (!lang)
64 return std::nullopt;
65
66 std::optional<bool> continuous;
67 decoder >> continuous;
68 if (!continuous)
69 return std::nullopt;
70
71 std::optional<bool> interimResults;
72 decoder >> interimResults;
73 if (!interimResults)
74 return std::nullopt;
75
76 std::optional<uint64_t> maxAlternatives;
77 decoder >> maxAlternatives;
78 if (!maxAlternatives)
79 return std::nullopt;
80
81 std::optional<ClientOrigin> clientOrigin;
82 decoder >> clientOrigin;
83 if (!clientOrigin)
84 return std::nullopt;
85
86 std::optional<FrameIdentifier> frameIdentifier;
87 decoder >> frameIdentifier;
88 if (!frameIdentifier)
89 return std::nullopt;
90
91 return {{
92 WTFMove(*clientIdentifier),
93 WTFMove(*lang),
94 WTFMove(*continuous),
95 WTFMove(*interimResults),
96 WTFMove(*maxAlternatives),
97 WTFMove(*clientOrigin),
98 WTFMove(*frameIdentifier)
99 }};
100}
