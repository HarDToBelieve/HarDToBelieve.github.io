29#include <wtf/IsoMallocInlines.h>

31namespace WebCore {
32
33WTF_MAKE_ISO_ALLOCATED_IMPL(SpeechRecognitionResult);
34
35Ref<SpeechRecognitionResult> SpeechRecognitionResult::create(Vector<Ref<SpeechRecognitionAlternative>>&& alternatives, bool isFinal)
36{
37 return adoptRef(*new SpeechRecognitionResult(WTFMove(alternatives), isFinal));
38}
39
40SpeechRecognitionResult::SpeechRecognitionResult(Vector<Ref<SpeechRecognitionAlternative>>&& alternatives, bool isFinal)
41 : m_alternatives(WTFMove(alternatives))
42 , m_isFinal(isFinal)
43{
44}
45
46SpeechRecognitionAlternative* SpeechRecognitionResult::item(uint64_t index) const
47{
48 return (index < m_alternatives.size()) ? m_alternatives[index].ptr() : nullptr;
49}
