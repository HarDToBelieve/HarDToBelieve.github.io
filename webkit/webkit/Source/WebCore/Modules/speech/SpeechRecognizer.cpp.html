39SpeechRecognizer::SpeechRecognizer(DelegateCallback&& delegateCallback, UniqueRef<SpeechRecognitionRequest>&& request)
40 : m_delegateCallback(WTFMove(delegateCallback))
41 , m_request(WTFMove(request))
42{
43}

45void SpeechRecognizer::abort(std::optional<SpeechRecognitionError>&& error)
46{
47 if (m_state == State::Aborting || m_state == State::Inactive)
48 return;
49 m_state = State::Aborting;
50
51 if (error)
52 m_delegateCallback(SpeechRecognitionUpdate::createError(clientIdentifier(), *error));
53
54 stopCapture();
55 abortRecognition();
56}
57
58void SpeechRecognizer::stop()
59{
60 if (m_state == State::Aborting || m_state == State::Inactive)
61 return;
62 m_state = State::Stopping;
63
64 stopCapture();
65 stopRecognition();
66}

68SpeechRecognitionConnectionClientIdentifier SpeechRecognizer::clientIdentifier() const
69{
70 return m_request->clientIdentifier();
71}
72
73void SpeechRecognizer::prepareForDestruction()
74{
75 if (m_state == State::Inactive)
76 return;
77
78 auto delegateCallback = std::exchange(m_delegateCallback, [](const SpeechRecognitionUpdate&) { });
79 delegateCallback(SpeechRecognitionUpdate::create(clientIdentifier(), SpeechRecognitionUpdateType::End));
80 m_state = State::Inactive;
81}

84
85void SpeechRecognizer::start(Ref<RealtimeMediaSource>&& source, bool mockSpeechRecognitionEnabled)
86{
87 if (!startRecognition(mockSpeechRecognitionEnabled, clientIdentifier(), m_request->lang(), m_request->continuous(), m_request->interimResults(), m_request->maxAlternatives())) {
88 auto error = SpeechRecognitionError { SpeechRecognitionErrorType::ServiceNotAllowed, "Failed to start recognition"_s };
89 m_delegateCallback(SpeechRecognitionUpdate::createError(clientIdentifier(), WTFMove(error)));
90 return;
91 }
92
93 m_state = State::Running;
94 m_delegateCallback(SpeechRecognitionUpdate::create(clientIdentifier(), SpeechRecognitionUpdateType::Start));
95 startCapture(WTFMove(source));
96}
97
98void SpeechRecognizer::startCapture(Ref<RealtimeMediaSource>&& source)
99{
100 auto dataCallback = [weakThis = makeWeakPtr(this)](const auto& time, const auto& data, const auto& description, auto sampleCount) {
101 if (weakThis)
102 weakThis->dataCaptured(time, data, description, sampleCount);
103 };
104
105 auto stateUpdateCallback = [weakThis = makeWeakPtr(this)](const auto& update) {
106 if (weakThis)
107 weakThis->m_delegateCallback(update);
108 };
109
110 m_source = makeUnique<SpeechRecognitionCaptureSource>(clientIdentifier(), WTFMove(dataCallback), WTFMove(stateUpdateCallback), WTFMove(source));
111}
112
113#endif
114
115void SpeechRecognizer::stopCapture()
116{
117 if (!m_source)
118 return;
119
120 m_source = nullptr;
121 m_delegateCallback(SpeechRecognitionUpdate::create(clientIdentifier(), SpeechRecognitionUpdateType::AudioEnd));
122}
