26#pragma once
27
28#include "SpeechRecognitionCaptureSource.h"
29#include "SpeechRecognitionConnectionClientIdentifier.h"
30#include "SpeechRecognitionError.h"
31#include <wtf/UniqueRef.h>
32
33#if HAVE(SPEECHRECOGNIZER)
34#include <wtf/RetainPtr.h>
35OBJC_CLASS WebSpeechRecognizerTask;
36#endif
37
38namespace WebCore {
39
40class SpeechRecognitionRequest;
41class SpeechRecognitionUpdate;
42
43class SpeechRecognizer : public CanMakeWeakPtr<SpeechRecognizer> {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 using DelegateCallback = Function<void(const SpeechRecognitionUpdate&)>;
47 WEBCORE_EXPORT explicit SpeechRecognizer(DelegateCallback&&, UniqueRef<SpeechRecognitionRequest>&&);
48
49#if ENABLE(MEDIA_STREAM)
50 WEBCORE_EXPORT void start(Ref<RealtimeMediaSource>&&, bool mockSpeechRecognitionEnabled);
51#endif
52 WEBCORE_EXPORT void abort(std::optional<SpeechRecognitionError>&& = std::nullopt);
53 WEBCORE_EXPORT void stop();
54 WEBCORE_EXPORT void prepareForDestruction();
55
56 WEBCORE_EXPORT SpeechRecognitionConnectionClientIdentifier clientIdentifier() const;
57 SpeechRecognitionCaptureSource* source() { return m_source.get(); }
58
59 void setInactive() { m_state = State::Inactive; }
60
61private:
62 enum class State {
63 Inactive,
64 Running,
65 Stopping,
66 Aborting
67 };
68
69#if ENABLE(MEDIA_STREAM)
70 void startCapture(Ref<RealtimeMediaSource>&&);
71#endif
72 void stopCapture();
73 void dataCaptured(const WTF::MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t sampleCount);
74 bool startRecognition(bool mockSpeechRecognitionEnabled, SpeechRecognitionConnectionClientIdentifier, const String& localeIdentifier, bool continuous, bool interimResults, uint64_t alternatives);
75 void abortRecognition();
76 void stopRecognition();
77
78 DelegateCallback m_delegateCallback;
79 UniqueRef<SpeechRecognitionRequest> m_request;
80 std::unique_ptr<SpeechRecognitionCaptureSource> m_source;
81 State m_state { State::Inactive };
82
83#if HAVE(SPEECHRECOGNIZER)
84 RetainPtr<WebSpeechRecognizerTask> m_task;
85#endif
86};
87
88} // namespace WebCore
