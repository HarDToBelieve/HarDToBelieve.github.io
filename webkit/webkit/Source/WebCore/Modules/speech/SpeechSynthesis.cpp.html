50SpeechSynthesis::SpeechSynthesis(WeakPtr<SpeechSynthesisClient> client, Document& document)
51 : m_currentSpeechUtterance(nullptr)
52 , m_isPaused(false)
53#if PLATFORM(IOS_FAMILY)
54 , m_restrictions(document.audioPlaybackRequiresUserGesture() ? RequireUserGestureForSpeechStartRestriction : NoRestrictions)
55#endif
56 , m_speechSynthesisClient(client)
57{
58#if !PLATFORM(IOS_FAMILY)
59 UNUSED_PARAM(document);
60#endif
61
62 if (m_speechSynthesisClient)
63 m_speechSynthesisClient->setObserver(makeWeakPtr(this));
64}

66void SpeechSynthesis::setPlatformSynthesizer(std::unique_ptr<PlatformSpeechSynthesizer> synthesizer)

88const Vector<Ref<SpeechSynthesisVoice>>& SpeechSynthesis::getVoices()
89{
90 if (!m_voiceList.isEmpty())
91 return m_voiceList;
92
93 // If the voiceList is empty, that's the cue to get the voices from the platform again.
94 for (auto& voice : m_speechSynthesisClient ? m_speechSynthesisClient->voiceList() : ensurePlatformSpeechSynthesizer().voiceList())
95 m_voiceList.append(SpeechSynthesisVoice::create(*voice));
96
97 return m_voiceList;
98}

107bool SpeechSynthesis::pending() const
108{
109 // This is true if there are any utterances that have not started.
110 // That means there will be more than one in the queue.
111 return m_utteranceQueue.size() > 1;
112}

119void SpeechSynthesis::startSpeakingImmediately(SpeechSynthesisUtterance& utterance)
120{
121 utterance.setStartTime(MonotonicTime::now());
122 m_currentSpeechUtterance = &utterance;
123 m_isPaused = false;
124
125 // Zero lengthed strings should immediately notify that the event is complete.
126 if (utterance.text().isEmpty()) {
127 handleSpeakingCompleted(utterance, false);
128 return;
129 }
130
131 if (m_speechSynthesisClient)
132 m_speechSynthesisClient->speak(utterance.platformUtterance());
133 else
134 ensurePlatformSpeechSynthesizer().speak(utterance.platformUtterance());
135}
136
137void SpeechSynthesis::speak(SpeechSynthesisUtterance& utterance)
138{
139 // Like Audio, we should require that the user interact to start a speech synthesis session.
140#if PLATFORM(IOS_FAMILY)
141 if (UserGestureIndicator::processingUserGesture())
142 removeBehaviorRestriction(RequireUserGestureForSpeechStartRestriction);
143 else if (userGestureRequiredForSpeechStart())
144 return;
145#endif
146
147 m_utteranceQueue.append(utterance);
148
149 // If the queue was empty, speak this immediately and add it to the queue.
150 if (m_utteranceQueue.size() == 1)
151 startSpeakingImmediately(m_utteranceQueue.first());
152}

154void SpeechSynthesis::cancel()
155{
156 // Remove all the items from the utterance queue.
157 // Hold on to the current utterance so the platform synthesizer can have a chance to clean up.
158 RefPtr<SpeechSynthesisUtterance> current = m_currentSpeechUtterance;
159 m_utteranceQueue.clear();
160 if (m_speechSynthesisClient) {
161 m_speechSynthesisClient->cancel();
162 // If we wait for cancel to callback speakingErrorOccurred, then m_currentSpeechUtterance will be null
163 // and the event won't be processed. Instead we process the error immediately.
164 speakingErrorOccurred();
165 m_currentSpeechUtterance = nullptr;
166 } else if (m_platformSpeechSynthesizer) {
167 m_platformSpeechSynthesizer->cancel();
168 // The platform should have called back immediately and cleared the current utterance.
169 ASSERT(!m_currentSpeechUtterance);
170 }
171 current = nullptr;
172}

174void SpeechSynthesis::pause()
175{
176 if (!m_isPaused) {
177 if (m_speechSynthesisClient)
178 m_speechSynthesisClient->pause();
179 else if (m_platformSpeechSynthesizer)

181 }

194void SpeechSynthesis::fireEvent(const AtomString& type, SpeechSynthesisUtterance& utterance, unsigned long charIndex, const String& name)
195{
196 utterance.dispatchEvent(SpeechSynthesisEvent::create(type, charIndex, (MonotonicTime::now() - utterance.startTime()).seconds(), name));
197}

199void SpeechSynthesis::handleSpeakingCompleted(SpeechSynthesisUtterance& utterance, bool errorOccurred)
200{
201 ASSERT(m_currentSpeechUtterance);
202 Ref<SpeechSynthesisUtterance> protect(utterance);
203
204 m_currentSpeechUtterance = nullptr;
205
206 fireEvent(errorOccurred ? eventNames().errorEvent : eventNames().endEvent, utterance, 0, String());
207
208 if (m_utteranceQueue.size()) {
209 Ref<SpeechSynthesisUtterance> firstUtterance = m_utteranceQueue.takeFirst();
210 ASSERT(&utterance == firstUtterance.ptr());
211
212 // Start the next job if there is one pending.
213 if (!m_utteranceQueue.isEmpty())
214 startSpeakingImmediately(m_utteranceQueue.first());
215 }
216}

218void SpeechSynthesis::boundaryEventOccurred(PlatformSpeechSynthesisUtterance& utterance, SpeechBoundary boundary, unsigned charIndex)
219{
220 static NeverDestroyed<const String> wordBoundaryString(MAKE_STATIC_STRING_IMPL("word"));
221 static NeverDestroyed<const String> sentenceBoundaryString(MAKE_STATIC_STRING_IMPL("sentence"));
222
223 ASSERT(utterance.client());
224
225 switch (boundary) {
226 case SpeechBoundary::SpeechWordBoundary:
227 fireEvent(eventNames().boundaryEvent, static_cast<SpeechSynthesisUtterance&>(*utterance.client()), charIndex, wordBoundaryString);
228 break;
229 case SpeechBoundary::SpeechSentenceBoundary:
230 fireEvent(eventNames().boundaryEvent, static_cast<SpeechSynthesisUtterance&>(*utterance.client()), charIndex, sentenceBoundaryString);
231 break;
232 default:
233 ASSERT_NOT_REACHED();
234 }
235}

284void SpeechSynthesis::didStartSpeaking(PlatformSpeechSynthesisUtterance& utterance)
285{
286 if (utterance.client())
287 fireEvent(eventNames().startEvent, static_cast<SpeechSynthesisUtterance&>(*utterance.client()), 0, String());
288}
289
290void SpeechSynthesis::didPauseSpeaking(PlatformSpeechSynthesisUtterance& utterance)
291{
292 m_isPaused = true;
293 if (utterance.client())
294 fireEvent(eventNames().pauseEvent, static_cast<SpeechSynthesisUtterance&>(*utterance.client()), 0, String());
295}
296
297void SpeechSynthesis::didResumeSpeaking(PlatformSpeechSynthesisUtterance& utterance)
298{
299 m_isPaused = false;
300 if (utterance.client())
301 fireEvent(eventNames().resumeEvent, static_cast<SpeechSynthesisUtterance&>(*utterance.client()), 0, String());
302}
303
304void SpeechSynthesis::didFinishSpeaking(PlatformSpeechSynthesisUtterance& utterance)
305{
306 if (utterance.client())
307 handleSpeakingCompleted(static_cast<SpeechSynthesisUtterance&>(*utterance.client()), false);
308}

310void SpeechSynthesis::speakingErrorOccurred(PlatformSpeechSynthesisUtterance& utterance)
311{
312 if (utterance.client())
313 handleSpeakingCompleted(static_cast<SpeechSynthesisUtterance&>(*utterance.client()), true);
314}
