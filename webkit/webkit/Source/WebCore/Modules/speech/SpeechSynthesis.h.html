38namespace WebCore {
39
40class Document;
41class PlatformSpeechSynthesizerClient;
42class SpeechSynthesisVoice;
43
44class SpeechSynthesis : public PlatformSpeechSynthesizerClient, public SpeechSynthesisClientObserver, public RefCounted<SpeechSynthesis> {
45public:
46 static Ref<SpeechSynthesis> create(WeakPtr<SpeechSynthesisClient>, Document&);
47
48 bool pending() const;
49 bool speaking() const;
50 bool paused() const;
51
52 void speak(SpeechSynthesisUtterance&);
53 void cancel();
54 void pause();
55 void resume();
56
57 const Vector<Ref<SpeechSynthesisVoice>>& getVoices();
58
59 // Used in testing to use a mock platform synthesizer
60 WEBCORE_EXPORT void setPlatformSynthesizer(std::unique_ptr<PlatformSpeechSynthesizer>);
61
62private:
63 SpeechSynthesis(WeakPtr<SpeechSynthesisClient>, Document&);
64
65 // PlatformSpeechSynthesizerClient override methods.
66 void voicesDidChange() override;
67 void didStartSpeaking(PlatformSpeechSynthesisUtterance&) override;
68 void didPauseSpeaking(PlatformSpeechSynthesisUtterance&) override;
69 void didResumeSpeaking(PlatformSpeechSynthesisUtterance&) override;
70 void didFinishSpeaking(PlatformSpeechSynthesisUtterance&) override;
71 void speakingErrorOccurred(PlatformSpeechSynthesisUtterance&) override;
72 void boundaryEventOccurred(PlatformSpeechSynthesisUtterance&, SpeechBoundary, unsigned charIndex) override;
73
74 // SpeechSynthesisClient override methods
75 void didStartSpeaking() override;
76 void didFinishSpeaking() override;
77 void didPauseSpeaking() override;
78 void didResumeSpeaking() override;
79 void speakingErrorOccurred() override;
80 void boundaryEventOccurred(bool wordBoundary, unsigned charIndex) override;
81 void voicesChanged() override;
82
83 void startSpeakingImmediately(SpeechSynthesisUtterance&);
84 void handleSpeakingCompleted(SpeechSynthesisUtterance&, bool errorOccurred);
85 void fireEvent(const AtomString& type, SpeechSynthesisUtterance&, unsigned long charIndex, const String& name);
86
87#if PLATFORM(IOS_FAMILY)
88 // Restrictions to change default behaviors.
89 enum BehaviorRestrictionFlags {
90 NoRestrictions = 0,
91 RequireUserGestureForSpeechStartRestriction = 1 << 0,
92 };
93 typedef unsigned BehaviorRestrictions;
94
95 bool userGestureRequiredForSpeechStart() const { return m_restrictions & RequireUserGestureForSpeechStartRestriction; }
96 void removeBehaviorRestriction(BehaviorRestrictions restriction) { m_restrictions &= ~restriction; }
97#endif
98 PlatformSpeechSynthesizer& ensurePlatformSpeechSynthesizer();
99
100 std::unique_ptr<PlatformSpeechSynthesizer> m_platformSpeechSynthesizer;
101 Vector<Ref<SpeechSynthesisVoice>> m_voiceList;
102 RefPtr<SpeechSynthesisUtterance> m_currentSpeechUtterance;
103 Deque<Ref<SpeechSynthesisUtterance>> m_utteranceQueue;
104 bool m_isPaused;
105#if PLATFORM(IOS_FAMILY)
106 BehaviorRestrictions m_restrictions;
107#endif
108 WeakPtr<SpeechSynthesisClient> m_speechSynthesisClient;
109};
