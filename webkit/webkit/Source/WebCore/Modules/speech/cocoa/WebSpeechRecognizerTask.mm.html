69@implementation WebSpeechRecognizerTaskImpl

71- (instancetype)initWithIdentifier:(SpeechRecognitionConnectionClientIdentifier)identifier locale:(NSString*)localeIdentifier doMultipleRecognitions:(BOOL)continuous reportInterimResults:(BOOL)interimResults maxAlternatives:(unsigned long)alternatives delegateCallback:(void(^)(const SpeechRecognitionUpdate&))callback
72{
73 if (!(self = [super init]))
74 return nil;
75
76 _identifier = identifier;
77 _doMultipleRecognitions = continuous;
78 _delegateCallback = callback;
79 _hasSentSpeechStart = false;
80 _hasSentSpeechEnd = false;
81 _hasSentEnd = false;
82
83 _maxAlternatives = alternatives ? alternatives : 1;
84
85 if (![localeIdentifier length])
86 _recognizer = adoptNS([PAL::allocSFSpeechRecognizerInstance() init]);
87 else
88 _recognizer = adoptNS([PAL::allocSFSpeechRecognizerInstance() initWithLocale:[NSLocale localeWithLocaleIdentifier:localeIdentifier]]);
89 if (!_recognizer) {
90 [self release];
91 return nil;
92 }
93
94 if (![_recognizer isAvailable]) {
95 [self release];
96 return nil;
97 }
98
99 [_recognizer setDelegate:self];
100
101 _request = adoptNS([PAL::allocSFSpeechAudioBufferRecognitionRequestInstance() init]);
102 if ([_recognizer supportsOnDeviceRecognition])
103 [_request setRequiresOnDeviceRecognition:YES];
104 [_request setShouldReportPartialResults:interimResults];
105 [_request setTaskHint:SFSpeechRecognitionTaskHintDictation];
106
107#if USE(APPLE_INTERNAL_SDK)
108 [_request setDetectMultipleUtterances:YES];
109 [_request _setMaximumRecognitionDuration:maximumRecognitionDuration];
110#endif
111
112 _task = [_recognizer recognitionTaskWithRequest:_request.get() delegate:self];
113 return self;
114}
115
116- (void)callbackWithTranscriptions:(NSArray<SFTranscription *> *)transcriptions isFinal:(BOOL)isFinal
117{
118 Vector<SpeechRecognitionAlternativeData> alternatives;
119 alternatives.reserveInitialCapacity(_maxAlternatives);
120 for (SFTranscription* transcription in transcriptions) {
121 // FIXME: <rdar://73629573> get confidence of SFTranscription when possible.
122 double maxConfidence = 0.0;
123 for (SFTranscriptionSegment* segment in [transcription segments]) {
124 double confidence = [segment confidence];
125 maxConfidence = maxConfidence < confidence ? confidence : maxConfidence;
126 }
127 alternatives.uncheckedAppend(SpeechRecognitionAlternativeData { [transcription formattedString], maxConfidence });
128 if (alternatives.size() == _maxAlternatives)
129 break;
130 }
131 Vector<SpeechRecognitionResultData> datas;
132 datas.append(SpeechRecognitionResultData { WTFMove(alternatives), bool(isFinal) });
133 _delegateCallback(SpeechRecognitionUpdate::createResult(_identifier, WTFMove(datas)));
134}

136- (void)audioSamplesAvailable:(CMSampleBufferRef)sampleBuffer
137{
138 ASSERT(isMainThread());
139 [_request appendAudioSampleBuffer:sampleBuffer];
140}
141
142- (void)abort
143{
144 if (!_task || [_task state] == SFSpeechRecognitionTaskStateCanceling)
145 return;
146
147 if ([_task state] == SFSpeechRecognitionTaskStateCompleted) {
148 [self sendSpeechEndIfNeeded];
149 [self sendEndIfNeeded];
150 return;
151 }
152
153 [self sendSpeechEndIfNeeded];
154 [_request endAudio];
155 [_task cancel];
156}
157
158- (void)stop
159{
160 if (!_task || [_task state] == SFSpeechRecognitionTaskStateCanceling)
161 return;
162
163 if ([_task state] == SFSpeechRecognitionTaskStateCompleted) {
164 [self sendSpeechEndIfNeeded];
165 [self sendEndIfNeeded];
166 return;
167 }
168
169 [self sendSpeechEndIfNeeded];
170 [_request endAudio];
171 [_task finish];
172}
173
174- (void)sendSpeechStartIfNeeded
175{
176 if (_hasSentSpeechStart)
177 return;
178
179 _hasSentSpeechStart = true;
180 _delegateCallback(SpeechRecognitionUpdate::create(_identifier, SpeechRecognitionUpdateType::SpeechStart));
181}
182
183- (void)sendSpeechEndIfNeeded
184{
185 if (!_hasSentSpeechStart || _hasSentSpeechEnd)
186 return;
187
188 _hasSentSpeechEnd = true;
189 _delegateCallback(SpeechRecognitionUpdate::create(_identifier, SpeechRecognitionUpdateType::SpeechEnd));
190}
191
192- (void)sendEndIfNeeded
193{
194 if (_hasSentEnd)
195 return;
196
197 _hasSentEnd = true;
198 _delegateCallback(SpeechRecognitionUpdate::create(_identifier, SpeechRecognitionUpdateType::End));
199}

201#pragma mark SFSpeechRecognizerDelegate
202
203- (void)speechRecognizer:(SFSpeechRecognizer *)speechRecognizer availabilityDidChange:(BOOL)available
204{
205 ASSERT(isMainThread());
206
207 if (available || !_task)
208 return;
209
210 auto error = SpeechRecognitionError { SpeechRecognitionErrorType::ServiceNotAllowed, "Speech recognition service becomes unavailable"_s };
211 _delegateCallback(SpeechRecognitionUpdate::createError(_identifier, WTFMove(error)));
212}
213
214#pragma mark SFSpeechRecognitionTaskDelegate
215
216- (void)speechRecognitionTask:(SFSpeechRecognitionTask *)task didHypothesizeTranscription:(SFTranscription *)transcription
217{
218 ASSERT(isMainThread());
219
220 [self sendSpeechStartIfNeeded];
221 [self callbackWithTranscriptions:[NSArray arrayWithObjects:transcription, nil] isFinal:NO];
222}
223
224- (void)speechRecognitionTask:(SFSpeechRecognitionTask *)task didFinishRecognition:(SFSpeechRecognitionResult *)recognitionResult
225{
226 ASSERT(isMainThread());
227 [self callbackWithTranscriptions:recognitionResult.transcriptions isFinal:YES];
228
229 if (!_doMultipleRecognitions) {
230 [self sendSpeechEndIfNeeded];
231 [self sendEndIfNeeded];
232 }
233}
234
235- (void)speechRecognitionTaskWasCancelled:(SFSpeechRecognitionTask *)task
236{
237 ASSERT(isMainThread());
238
239 [self sendSpeechEndIfNeeded];
240 [self sendEndIfNeeded];
241}
242
243- (void)speechRecognitionTask:(SFSpeechRecognitionTask *)task didFinishSuccessfully:(BOOL)successfully
244{
245 ASSERT(isMainThread());
246
247 if (!successfully) {
248 auto error = SpeechRecognitionError { SpeechRecognitionErrorType::Aborted, task.error.localizedDescription };
249 _delegateCallback(SpeechRecognitionUpdate::createError(_identifier, WTFMove(error)));
250 }
251
252 [self sendEndIfNeeded];
253}
