26#import "config.h"
27#import "WebSpeechRecognizerTaskMock.h"
28
29#if HAVE(SPEECHRECOGNIZER)
30
31using namespace WebCore;
32
33NS_ASSUME_NONNULL_BEGIN
34
35@implementation WebSpeechRecognizerTaskMock
36
37- (instancetype)initWithIdentifier:(SpeechRecognitionConnectionClientIdentifier)identifier locale:(NSString*)localeIdentifier doMultipleRecognitions:(BOOL)continuous reportInterimResults:(BOOL)interimResults maxAlternatives:(unsigned long)alternatives delegateCallback:(void(^)(const SpeechRecognitionUpdate&))callback
38{
39 if (!(self = [super init]))
40 return nil;
41
42 _doMultipleRecognitions = continuous;
43 _identifier = identifier;
44 _delegateCallback = callback;
45 _completed = false;
46 _hasSentSpeechStart = false;
47 _hasSentSpeechEnd = false;
48
49 return self;
50}
51
52- (void)audioSamplesAvailable:(CMSampleBufferRef)sampleBuffer
53{
54 if (!_hasSentSpeechStart) {
55 _hasSentSpeechStart = true;
56 _delegateCallback(SpeechRecognitionUpdate::create(_identifier, SpeechRecognitionUpdateType::SpeechStart));
57 }
58
59 // Fake some recognition results.
60 Vector<SpeechRecognitionAlternativeData> alternatives;
61 alternatives.append(SpeechRecognitionAlternativeData { "Test", 1.0 });
62 Vector<SpeechRecognitionResultData> datas;
63 datas.append(SpeechRecognitionResultData { WTFMove(alternatives), true });
64 _delegateCallback(SpeechRecognitionUpdate::createResult(_identifier, WTFMove(datas)));
65
66 if (!_doMultipleRecognitions)
67 [self abort];
68}
69
70- (void)abort
71{
72 if (_completed)
73 return;
74 _completed = true;
75
76 if (!_hasSentSpeechEnd && _hasSentSpeechStart) {
77 _hasSentSpeechEnd = true;
78 _delegateCallback(SpeechRecognitionUpdate::create(_identifier, SpeechRecognitionUpdateType::SpeechEnd));
79 }
80
81 _delegateCallback(SpeechRecognitionUpdate::create(_identifier, SpeechRecognitionUpdateType::End));
82}
83
84- (void)stop
85{
86 [self abort];
87}
88
89@end
90
91NS_ASSUME_NONNULL_END
92
93#endif
