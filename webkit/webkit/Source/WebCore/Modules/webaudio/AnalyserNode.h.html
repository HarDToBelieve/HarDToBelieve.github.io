25#pragma once
26
27#include "AnalyserOptions.h"
28#include "AudioBasicInspectorNode.h"
29#include "RealtimeAnalyser.h"
30
31namespace WebCore {
32
33class AnalyserNode final : public AudioBasicInspectorNode {
34 WTF_MAKE_ISO_ALLOCATED(AnalyserNode);
35public:
36 static ExceptionOr<Ref<AnalyserNode>> create(BaseAudioContext&, const AnalyserOptions& = { });
37
38 virtual ~AnalyserNode();
39
40 unsigned fftSize() const { return m_analyser.fftSize(); }
41 ExceptionOr<void> setFftSize(unsigned);
42
43 unsigned frequencyBinCount() const { return m_analyser.frequencyBinCount(); }
44
45 ExceptionOr<void> setMinMaxDecibels(double minDecibels, double maxDecibels);
46
47 ExceptionOr<void> setMinDecibels(double);
48 double minDecibels() const { return m_analyser.minDecibels(); }
49
50 ExceptionOr<void> setMaxDecibels(double);
51 double maxDecibels() const { return m_analyser.maxDecibels(); }
52
53 ExceptionOr<void> setSmoothingTimeConstant(double);
54 double smoothingTimeConstant() const { return m_analyser.smoothingTimeConstant(); }
55
56 void getFloatFrequencyData(const Ref<JSC::Float32Array>& array) { m_analyser.getFloatFrequencyData(array.get()); }
57 void getByteFrequencyData(const Ref<JSC::Uint8Array>& array) { m_analyser.getByteFrequencyData(array.get()); }
58 void getFloatTimeDomainData(const Ref<JSC::Float32Array>& array) { m_analyser.getFloatTimeDomainData(array.get()); }
59 void getByteTimeDomainData(const Ref<JSC::Uint8Array>& array) { m_analyser.getByteTimeDomainData(array.get()); }
60
61private:
62 AnalyserNode(BaseAudioContext&);
63
64 void process(size_t framesToProcess) final;
65 void updatePullStatus() final;
66
67 double tailTime() const final;
68 double latencyTime() const final { return 0; }
69 bool requiresTailProcessing() const final;
70 bool propagatesSilence() const final;
71
72 RealtimeAnalyser m_analyser;
73};
74
75} // namespace WebCore
