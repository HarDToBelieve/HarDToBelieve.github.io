38AsyncAudioDecoder::AsyncAudioDecoder()
39{
40 // Start worker thread.
41 Locker locker { m_threadCreationMutex };
42 m_thread = Thread::create("Audio Decoder", [this] {
43 runLoop();
44 }, ThreadType::Audio);
45}
46
47AsyncAudioDecoder::~AsyncAudioDecoder()
48{
49 m_queue.kill();
50
51 // Stop thread.
52 m_thread->waitForCompletion();
53}
54
55void AsyncAudioDecoder::decodeAsync(Ref<ArrayBuffer>&& audioData, float sampleRate, Function<void(ExceptionOr<Ref<AudioBuffer>>&&)>&& callback)
56{
57 ASSERT(isMainThread());
58
59 auto decodingTask = makeUnique<DecodingTask>(WTFMove(audioData), sampleRate, WTFMove(callback));
60 m_queue.append(WTFMove(decodingTask)); // note that ownership of the task is effectively taken by the queue.
61}
62
63void AsyncAudioDecoder::runLoop()
64{
65 ASSERT(!isMainThread());
66
67 {
68 // Wait for until we have m_thread established before starting the run loop.
69 Locker locker { m_threadCreationMutex };
70 }
71
72 // Keep running decoding tasks until we're killed.
73 while (auto decodingTask = m_queue.waitForMessage()) {
74 // Let the task take care of its own ownership.
75 // See DecodingTask::notifyComplete() for cleanup.
76 decodingTask.release()->decode();
77 }
78}
79
80AsyncAudioDecoder::DecodingTask::DecodingTask(Ref<ArrayBuffer>&& audioData, float sampleRate, Function<void(ExceptionOr<Ref<AudioBuffer>>&&)>&& callback)
81 : m_audioData(WTFMove(audioData))
82 , m_sampleRate(sampleRate)
83 , m_callback(WTFMove(callback))
84{
85}
86
87void AsyncAudioDecoder::DecodingTask::decode()
88{
89 // Do the actual decoding and invoke the callback.
90 m_audioBuffer = AudioBuffer::createFromAudioFileData(m_audioData->data(), m_audioData->byteLength(), false, sampleRate());
91
92 // Decoding is finished, but we need to do the callbacks on the main thread.
93 callOnMainThread([this] {
94 notifyComplete();
95 });
96}
97
98void AsyncAudioDecoder::DecodingTask::notifyComplete()
99{
100 if (auto* audioBuffer = this->audioBuffer())
101 callback()(makeRef(*audioBuffer));
102 else
103 callback()(Exception { EncodingError, "Decoding failed"_s });
104
105 // Our ownership was given up in AsyncAudioDecoder::runLoop()
106 // Make sure to clean up here.
107 delete this;
108}
