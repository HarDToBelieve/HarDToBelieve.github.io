43RefPtr<AudioBuffer> AudioBuffer::create(unsigned numberOfChannels, size_t numberOfFrames, float sampleRate, LegacyPreventDetaching preventDetaching)
44{
45 if (!BaseAudioContext::isSupportedSampleRate(sampleRate) || !numberOfChannels || numberOfChannels > AudioContext::maxNumberOfChannels || !numberOfFrames)
46 return nullptr;
47
48 auto buffer = adoptRef(*new AudioBuffer(numberOfChannels, numberOfFrames, sampleRate, preventDetaching));
49 if (!buffer->originalLength())
50 return nullptr;
51
52 return buffer;
53}
54
55ExceptionOr<Ref<AudioBuffer>> AudioBuffer::create(const AudioBufferOptions& options)
56{
57 if (!options.numberOfChannels)
58 return Exception { NotSupportedError, "Number of channels cannot be 0."_s };
59
60 if (options.numberOfChannels > AudioContext::maxNumberOfChannels)
61 return Exception { NotSupportedError, "Number of channels cannot be more than max supported."_s };
62
63 if (!options.length)
64 return Exception { NotSupportedError, "Length must be at least 1."_s };
65
66 if (!BaseAudioContext::isSupportedSampleRate(options.sampleRate))
67 return Exception { NotSupportedError, "Sample rate is not in the supported range."_s };
68
69 auto buffer = adoptRef(*new AudioBuffer(options.numberOfChannels, options.length, options.sampleRate));
70 if (!buffer->originalLength())
71 return Exception { NotSupportedError, "Channel was not able to be created."_s };
72
73 return buffer;
74}

84AudioBuffer::AudioBuffer(unsigned numberOfChannels, size_t length, float sampleRate, LegacyPreventDetaching preventDetaching)
85 : m_sampleRate(sampleRate)
86 , m_originalLength(length)
87 , m_isDetachable(preventDetaching == LegacyPreventDetaching::No)
88{
89 m_channels.reserveCapacity(numberOfChannels);
90
91 for (unsigned i = 0; i < numberOfChannels; ++i) {
92 auto channelDataArray = Float32Array::tryCreate(m_originalLength);
93 if (!channelDataArray) {
94 invalidate();
95 break;
96 }
97
98 if (preventDetaching == LegacyPreventDetaching::Yes)
99 channelDataArray->setDetachable(false);
100
101 m_channels.append(WTFMove(channelDataArray));
102 }
103 m_channelWrappers.resize(m_channels.size());
104}
105
106AudioBuffer::AudioBuffer(AudioBus& bus)
107 : m_sampleRate(bus.sampleRate())
108 , m_originalLength(bus.length())
109{
110 // Copy audio data from the bus to the Float32Arrays we manage.
111 unsigned numberOfChannels = bus.numberOfChannels();
112 m_channels.reserveCapacity(numberOfChannels);
113 for (unsigned i = 0; i < numberOfChannels; ++i) {
114 auto channelDataArray = Float32Array::tryCreate(m_originalLength);
115 if (!channelDataArray) {
116 invalidate();
117 break;
118 }
119
120 channelDataArray->setRange(bus.channel(i)->data(), m_originalLength, 0);
121 m_channels.append(WTFMove(channelDataArray));
122 }
123 m_channelWrappers.resize(m_channels.size());
124}

126void AudioBuffer::invalidate()
127{
128 releaseMemory();
129 m_originalLength = 0;
130}

132void AudioBuffer::releaseMemory()
133{
134 Locker locker { m_channelsLock };
135 m_channels.clear();
136 m_channelWrappers.clear();
137}

139ExceptionOr<JSC::JSValue> AudioBuffer::getChannelData(JSDOMGlobalObject& globalObject, unsigned channelIndex)
140{
141 ASSERT(m_channelWrappers.size() == m_channels.size());
142 if (channelIndex >= m_channelWrappers.size())
143 return Exception { IndexSizeError, "Index must be less than number of channels."_s };
144
145 auto& channelData = m_channels[channelIndex];
146 auto constructJSArray = [&] {
147 return JSC::JSFloat32Array::create(globalObject.vm(), globalObject.typedArrayStructure(JSC::TypeFloat32), channelData.copyRef());
148 };
149
150 if (globalObject.worldIsNormal()) {
151 if (!m_channelWrappers[channelIndex])
152 m_channelWrappers[channelIndex] = { constructJSArray() };
153 return static_cast<JSC::JSValue>(m_channelWrappers[channelIndex]);
154 }
155 return constructJSArray();
156}

168RefPtr<Float32Array> AudioBuffer::channelData(unsigned channelIndex)
169{
170 if (channelIndex >= m_channels.size())
171 return nullptr;
172 if (hasDetachedChannelBuffer())
173 return Float32Array::create(0);
174 return m_channels[channelIndex].copyRef();
175}

177float* AudioBuffer::rawChannelData(unsigned channelIndex)
178{
179 if (channelIndex >= m_channels.size())
180 return nullptr;
181 if (hasDetachedChannelBuffer())
182 return nullptr;
183 return m_channels[channelIndex]->data();

186ExceptionOr<void> AudioBuffer::copyFromChannel(Ref<Float32Array>&& destination, unsigned channelNumber, unsigned bufferOffset)
187{
188 if (destination->isShared())
189 return Exception { TypeError, "Destination may not be a shared buffer."_s };
190
191 if (channelNumber >= m_channels.size())
192 return Exception { IndexSizeError, "Not a valid channelNumber."_s };
193
194 Float32Array* channelData = m_channels[channelNumber].get();
195
196 size_t dataLength = channelData->length();
197
198 if (bufferOffset >= dataLength)
199 return { };
200
201 unsigned count = dataLength - bufferOffset;
202 count = std::min(destination.get().length(), count);
203
204 const float* src = channelData->data();
205 float* dst = destination->data();
206
207 ASSERT(src);
208 ASSERT(dst);
209
210 memmove(dst, src + bufferOffset, count * sizeof(*src));
211 return { };
212}

214ExceptionOr<void> AudioBuffer::copyToChannel(Ref<Float32Array>&& source, unsigned channelNumber, unsigned bufferOffset)
215{
216 if (source->isShared())
217 return Exception { TypeError, "Source may not be a shared buffer."_s };
218
219 if (channelNumber >= m_channels.size())
220 return Exception { IndexSizeError, "Not a valid channelNumber."_s };
221
222 Float32Array* channelData = m_channels[channelNumber].get();
223
224 size_t dataLength = channelData->length();
225
226 if (bufferOffset >= dataLength)
227 return { };
228
229 unsigned count = dataLength - bufferOffset;
230 count = std::min(source.get().length(), count);
231
232 const float* src = source->data();
233 float* dst = channelData->data();
234
235 ASSERT(src);
236 ASSERT(dst);
237
238 memmove(dst + bufferOffset, src, count * sizeof(*dst));
239 return { };
240}

248size_t AudioBuffer::memoryCost() const
249{
250 // memoryCost() may be invoked concurrently from a GC thread, and we need to be careful
251 // about what data we access here and how. We need to hold a lock to prevent m_channels
252 // from being changed while we iterate it, but calling channel->byteLength() is safe
253 // because it doesn't involve chasing any pointers that can be nullified while the
254 // AudioBuffer is alive.
255 Locker locker { m_channelsLock };
256 size_t cost = 0;
257 for (auto& channel : m_channels)
258 cost += channel->byteLength();
259 return cost;
260}

262bool AudioBuffer::hasDetachedChannelBuffer() const
263{
264 for (auto& channel : m_channels) {
265 if (channel->isDetached())
266 return true;
267 }
268 return false;
269}

271bool AudioBuffer::topologyMatches(const AudioBuffer& other) const
272{
273 return numberOfChannels() == other.numberOfChannels() && length() == other.length() && sampleRate() == other.sampleRate();
274}
275
276bool AudioBuffer::copyTo(AudioBuffer& other) const
277{
278 if (!topologyMatches(other))
279 return false;
280
281 if (hasDetachedChannelBuffer() || other.hasDetachedChannelBuffer())
282 return false;
283
284 for (unsigned channelIndex = 0; channelIndex < numberOfChannels(); ++channelIndex)
285 memcpy(other.rawChannelData(channelIndex), m_channels[channelIndex]->data(), length() * sizeof(float));
286
287 return true;
288}

290Ref<AudioBuffer> AudioBuffer::clone(ShouldCopyChannelData shouldCopyChannelData) const
291{
292 auto clone = AudioBuffer::create(numberOfChannels(), length(), sampleRate(), m_isDetachable ? LegacyPreventDetaching::No : LegacyPreventDetaching::Yes);
293 ASSERT(clone);
294 if (shouldCopyChannelData == ShouldCopyChannelData::Yes)
295 copyTo(*clone);
296 return clone.releaseNonNull();
297}
