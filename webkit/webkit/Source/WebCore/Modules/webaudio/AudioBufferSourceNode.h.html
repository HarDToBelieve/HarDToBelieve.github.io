26#pragma once
27
28#include "AudioBufferSourceOptions.h"
29#include "AudioScheduledSourceNode.h"
30#include <wtf/Lock.h>
31#include <wtf/UniqueArray.h>
32
33namespace WebCore {
34
35class AudioBuffer;
36struct AudioBufferSourceOptions;
37
38// AudioBufferSourceNode is an AudioNode representing an audio source from an in-memory audio asset represented by an AudioBuffer.
39// It generally will be used for short sounds which require a high degree of scheduling flexibility (can playback in rhythmically perfect ways).
40
41class AudioBufferSourceNode final : public AudioScheduledSourceNode {
42 WTF_MAKE_ISO_ALLOCATED(AudioBufferSourceNode);
43public:
44 static Ref<AudioBufferSourceNode> create(BaseAudioContext&, float sampleRate);
45 static ExceptionOr<Ref<AudioBufferSourceNode>> create(BaseAudioContext&, AudioBufferSourceOptions&& = { });
46
47 virtual ~AudioBufferSourceNode();
48
49 // AudioNode
50 void process(size_t framesToProcess) final;
51
52 // setBufferForBindings() is called on the main thread. This is the buffer we use for playback.
53 ExceptionOr<void> setBufferForBindings(RefPtr<AudioBuffer>&&);
54
55 // This function does not lock before accessing the buffer and should therefore only be called on the main thread.
56 AudioBuffer* bufferForBindings() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_buffer.get(); }
57
58 // numberOfChannels() returns the number of output channels. This value equals the number of channels from the buffer.
59 // If a new buffer is set with a different number of channels, then this value will dynamically change.
60 unsigned numberOfChannels();
61
62 // Play-state
63 ExceptionOr<void> startLater(double when, double grainOffset, std::optional<double> grainDuration);
64
65 // This function doesn't grab the lock before accessing m_isLooping and is thus only safe to call on the main thread.
66 bool loopForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_isLooping; }
67 void setLoopForBindings(bool);
68
69 // Loop times in seconds.
70 double loopStartForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_loopStart; } // This function doesn't grab the lock and is thus only safe to call on the main thread.
71 double loopEndForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_loopEnd; } // This function doesn't grab the lock and is thus only safe to call on the main thread.
72 void setLoopStartForBindings(double);
73 void setLoopEndForBindings(double);
74
75 AudioParam& detune() { return m_detune.get(); }
76 AudioParam& playbackRate() { return m_playbackRate.get(); }
77
78 // If we are no longer playing, propogate silence ahead to downstream nodes.
79 bool propagatesSilence() const final;
80
81 const char* activeDOMObjectName() const override { return "AudioBufferSourceNode"; }
82
83protected:
84 explicit AudioBufferSourceNode(BaseAudioContext&);
85
86private:
87 double tailTime() const final { return 0; }
88 double latencyTime() const final { return 0; }
89
90 ExceptionOr<void> startPlaying(double when, double grainOffset, std::optional<double> grainDuration);
91 void adjustGrainParameters() WTF_REQUIRES_LOCK(m_processLock);
92
93 // Returns true on success.
94 bool renderFromBuffer(AudioBus*, unsigned destinationFrameOffset, size_t numberOfFrames, double startFrameOffset) WTF_REQUIRES_LOCK(m_processLock);
95
96 // Render silence starting from "index" frame in AudioBus.
97 inline bool renderSilenceAndFinishIfNotLooping(AudioBus*, unsigned index, size_t framesToProcess) WTF_REQUIRES_LOCK(m_processLock);
98
99 // m_buffer holds the sample data which this node outputs.
100 RefPtr<AudioBuffer> m_buffer WTF_GUARDED_BY_LOCK(m_processLock); // Only modified on the main thread but used on the audio thread.
101
102 // Pointers for the buffer and destination.
103 UniqueArray<const float*> m_sourceChannels;
104 UniqueArray<float*> m_destinationChannels;
105
106 Ref<AudioParam> m_detune;
107 Ref<AudioParam> m_playbackRate;
108
109 // If m_isLooping is false, then this node will be done playing and become inactive after it reaches the end of the sample data in the buffer.
110 // If true, it will wrap around to the start of the buffer each time it reaches the end.
111 bool m_isLooping WTF_GUARDED_BY_LOCK(m_processLock) { false }; // Only modified on the main thread but queried on the audio thread.
112
113 bool m_wasBufferSet { false };
114
115 double m_loopStart WTF_GUARDED_BY_LOCK(m_processLock) { 0 }; // Only modified on the main thread but queried on the audio thread.
116 double m_loopEnd WTF_GUARDED_BY_LOCK(m_processLock) { 0 }; // Only modified on the main thread but queried on the audio thread.
117
118 // m_virtualReadIndex is a sample-frame index into our buffer representing the current playback position.
119 // Since it's floating-point, it has sub-sample accuracy.
120 double m_virtualReadIndex { 0 };
121
122 // Granular playback
123 bool m_isGrain WTF_GUARDED_BY_LOCK(m_processLock) { false };
124 double m_grainOffset WTF_GUARDED_BY_LOCK(m_processLock) { 0 }; // in seconds
125 double m_grainDuration WTF_GUARDED_BY_LOCK(m_processLock); // in seconds
126 double m_wasGrainDurationGiven WTF_GUARDED_BY_LOCK(m_processLock) { false };
127
128 // totalPitchRate() returns the instantaneous pitch rate (non-time preserving).
129 // It incorporates the base pitch rate, any sample-rate conversion factor from the buffer.
130 double totalPitchRate() WTF_REQUIRES_LOCK(m_processLock);
131
132 // This synchronizes process() with setBuffer() which can cause dynamic channel count changes.
133 mutable Lock m_processLock;
134};
135
136} // namespace WebCore
