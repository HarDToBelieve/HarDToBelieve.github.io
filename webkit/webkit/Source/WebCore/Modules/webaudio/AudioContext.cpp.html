77static bool shouldDocumentAllowWebAudioToAutoPlay(const Document& document)
78{
79 if (document.processingUserGestureForMedia() || document.isCapturing())
80 return true;
81 return document.quirks().shouldAutoplayWebAudioForArbitraryUserGesture() && document.topDocument().hasHadUserInteraction();
82}

89ExceptionOr<Ref<AudioContext>> AudioContext::create(Document& document, AudioContextOptions&& contextOptions)
90{
91 ASSERT(isMainThread());
92#if OS(WINDOWS)
93 if (hardwareContextCount >= maxHardwareContexts)
94 return Exception { QuotaExceededError, "Reached maximum number of hardware contexts on this platform"_s };
95#endif
96
97 if (!document.isFullyActive())
98 return Exception { InvalidStateError, "Document is not fully active"_s };
99
100 // FIXME: Figure out where latencyHint should go.
101
102 if (!contextOptions.sampleRate && defaultSampleRateForTesting())
103 contextOptions.sampleRate = *defaultSampleRateForTesting();
104
105 if (contextOptions.sampleRate && !isSupportedSampleRate(*contextOptions.sampleRate))
106 return Exception { SyntaxError, "sampleRate is not in range"_s };
107
108 auto audioContext = adoptRef(*new AudioContext(document, contextOptions));
109 audioContext->suspendIfNeeded();
110 return audioContext;
111}

113AudioContext::AudioContext(Document& document, const AudioContextOptions& contextOptions)
114 : BaseAudioContext(document)
115 , m_destinationNode(makeUniqueRef<DefaultAudioDestinationNode>(*this, contextOptions.sampleRate))
116 , m_mediaSession(PlatformMediaSession::create(PlatformMediaSessionManager::sharedManager(), *this))
117{
118 constructCommon();
119
120 // Initialize the destination node's muted state to match the page's current muted state.
121 pageMutedStateDidChange();
122
123 document.addAudioProducer(*this);
124 document.registerForVisibilityStateChangedCallbacks(*this);
125
126 // Unlike OfflineAudioContext, AudioContext does not require calling resume() to start rendering.
127 // Lazy initialization starts rendering so we schedule a task here to make sure lazy initialization
128 // ends up happening, even if no audio node gets constructed.
129 postTask([this, pendingActivity = makePendingActivity(*this)] {
130 if (!isStopped())
131 lazyInitialize();
132 });
133}

135void AudioContext::constructCommon()
136{
137 ASSERT(document());
138 if (document()->topDocument().audioPlaybackRequiresUserGesture())
139 addBehaviorRestriction(RequireUserGestureForAudioStartRestriction);
140 else
141 m_restrictions = NoRestrictions;
142
143#if PLATFORM(COCOA)
144 addBehaviorRestriction(RequirePageConsentForAudioStartRestriction);
145#endif
146}

151 document()->removeAudioProducer(*this);
152 document()->unregisterForVisibilityStateChangedCallbacks(*this);

178AudioTimestamp AudioContext::getOutputTimestamp(DOMWindow& window)
179{
180 auto& performance = window.performance();
181
182 auto position = outputPosition();
183
184 // The timestamp of what is currently being played (contextTime) cannot be
185 // later than what is being rendered. (currentTime)
186 position.position = Seconds { std::min(position.position.seconds(), currentTime()) };
187
188 auto performanceTime = performance.relativeTimeFromTimeOriginInReducedResolution(position.timestamp);
189 performanceTime = std::max(performanceTime, 0.0);
190
191 return { position.position.seconds(), performanceTime };
192}

194void AudioContext::close(DOMPromiseDeferred<void>&& promise)
195{
196 if (isOfflineContext() || isStopped()) {
197 promise.reject(InvalidStateError);
198 return;
199 }
200
201 if (state() == State::Closed) {
202 promise.resolve();
203 return;
204 }
205
206 addReaction(State::Closed, WTFMove(promise));
207
208 lazyInitialize();
209
210 destination().close([this, activity = makePendingActivity(*this)] {
211 setState(State::Closed);
212 uninitialize();
213 m_mediaSession->setActive(false);
214 });
215}

217void AudioContext::suspendRendering(DOMPromiseDeferred<void>&& promise)
218{
219 if (isOfflineContext()) {
220 promise.reject(Exception { InvalidStateError, "Cannot call suspend() on an OfflineAudioContext"_s });
221 return;
222 }
223
224 if (isStopped() || state() == State::Closed) {
225 promise.reject(Exception { InvalidStateError, "Context is closed"_s });
226 return;
227 }
228
229 m_wasSuspendedByScript = true;
230
231 if (!willPausePlayback()) {
232 addReaction(State::Suspended, WTFMove(promise));
233 return;
234 }
235
236 lazyInitialize();
237
238 destination().suspend([this, activity = makePendingActivity(*this), promise = WTFMove(promise)](std::optional<Exception>&& exception) mutable {
239 if (exception) {
240 promise.reject(WTFMove(*exception));
241 return;
242 }
243 setState(State::Suspended);
244 promise.resolve();
245 });
246}
247
248void AudioContext::resumeRendering(DOMPromiseDeferred<void>&& promise)
249{
250 if (isOfflineContext()) {
251 promise.reject(Exception { InvalidStateError, "Cannot call resume() on an OfflineAudioContext"_s });
252 return;
253 }
254
255 if (isStopped() || state() == State::Closed) {
256 promise.reject(Exception { InvalidStateError, "Context is closed"_s });
257 return;
258 }
259
260 m_wasSuspendedByScript = false;
261
262 if (!willBeginPlayback()) {
263 addReaction(State::Running, WTFMove(promise));
264 return;
265 }
266
267 lazyInitialize();
268
269 destination().resume([this, activity = makePendingActivity(*this), promise = WTFMove(promise)](std::optional<Exception>&& exception) mutable {
270 if (exception) {
271 promise.reject(WTFMove(*exception));
272 return;
273 }
274
275 // Since we update the state asynchronously, we may have been interrupted after the
276 // call to resume() and before this lambda runs. In this case, we don't want to
277 // reset the state to running.
278 bool interrupted = m_mediaSession->state() == PlatformMediaSession::Interrupted;
279 setState(interrupted ? State::Interrupted : State::Running);
280 if (interrupted)
281 addReaction(State::Running, WTFMove(promise));
282 else
283 promise.resolve();
284 });
285}

287void AudioContext::sourceNodeWillBeginPlayback(AudioNode& audioNode)
288{
289 BaseAudioContext::sourceNodeWillBeginPlayback(audioNode);
290
291 // Called by scheduled AudioNodes when clients schedule their start times.
292 // Prior to the introduction of suspend(), resume(), and stop(), starting
293 // a scheduled AudioNode would remove the user-gesture restriction, if present,
294 // and would thus unmute the context. Now that AudioContext stays in the
295 // "suspended" state if a user-gesture restriction is present, starting a
296 // schedule AudioNode should set the state to "running", but only if the
297 // user-gesture restriction is set.
298 if (userGestureRequiredForAudioStart())
299 startRendering();
300}
301
302void AudioContext::startRendering()
303{
304 ALWAYS_LOG(LOGIDENTIFIER);
305 if (isStopped() || !willBeginPlayback() || m_wasSuspendedByScript)
306 return;
307
308 lazyInitialize();
309 destination().startRendering([this, protectedThis = makeRef(*this), pendingActivity = makePendingActivity(*this)](std::optional<Exception>&& exception) {
310 if (!exception)
311 setState(State::Running);
312 });
313}

315void AudioContext::lazyInitialize()
316{
317 if (isInitialized())
318 return;
319
320 BaseAudioContext::lazyInitialize();
321 if (isInitialized()) {
322 if (state() != State::Running) {
323 // This starts the audio thread. The destination node's provideInput() method will now be called repeatedly to render audio.
324 // Each time provideInput() is called, a portion of the audio stream is rendered. Let's call this time period a "render quantum".
325 // NOTE: for now default AudioContext does not need an explicit startRendering() call from JavaScript.
326 // We may want to consider requiring it for symmetry with OfflineAudioContext.
327 startRendering();
328#if OS(WINDOWS)
329 ++hardwareContextCount;
330#endif
331 }
332 }
333}

335bool AudioContext::willPausePlayback()
336{
337 auto* document = this->document();
338 if (!document)
339 return false;
340
341 if (userGestureRequiredForAudioStart()) {
342 if (!document->processingUserGestureForMedia())
343 return false;
344 removeBehaviorRestriction(RequireUserGestureForAudioStartRestriction);
345 }
346
347 if (pageConsentRequiredForAudioStart()) {
348 auto* page = document->page();
349 if (page && !page->canStartMedia()) {
350 document->addMediaCanStartListener(*this);
351 return false;
352 }
353 removeBehaviorRestriction(RequirePageConsentForAudioStartRestriction);
354 }
355
356 return m_mediaSession->clientWillPausePlayback();
357}

367void AudioContext::mayResumePlayback(bool shouldResume)
368{
369 if (state() == State::Closed || !isInitialized() || state() == State::Running)
370 return;
371
372 if (!shouldResume) {
373 setState(State::Suspended);
374 return;
375 }
376
377 if (!willBeginPlayback())
378 return;
379
380 lazyInitialize();
381
382 destination().resume([this, protectedThis = makeRef(*this), pendingActivity = makePendingActivity(*this)](std::optional<Exception>&& exception) {
383 setState(exception ? State::Suspended : State::Running);
384 });
385}
386
387bool AudioContext::willBeginPlayback()
388{
389 auto* document = this->document();
390 if (!document)
391 return false;
392
393 if (userGestureRequiredForAudioStart()) {
394 if (!shouldDocumentAllowWebAudioToAutoPlay(*document)) {
395 ALWAYS_LOG(LOGIDENTIFIER, "returning false, not processing user gesture or capturing");
396 return false;
397 }
398 removeBehaviorRestriction(RequireUserGestureForAudioStartRestriction);
399 }
400
401 if (pageConsentRequiredForAudioStart()) {
402 auto* page = document->page();
403 if (page && !page->canStartMedia()) {
404 document->addMediaCanStartListener(*this);
405 ALWAYS_LOG(LOGIDENTIFIER, "returning false, page doesn't allow media to start");
406 return false;
407 }
408 removeBehaviorRestriction(RequirePageConsentForAudioStartRestriction);
409 }
410
411 m_mediaSession->setActive(true);
412
413 auto willBegin = m_mediaSession->clientWillBeginPlayback();
414 ALWAYS_LOG(LOGIDENTIFIER, "returning ", willBegin);
415
416 return willBegin;
417}

419void AudioContext::visibilityStateChanged()
420{
421 // Do not suspend if audio is audible.
422 if (!document() || mediaState() == MediaProducer::MediaState::IsPlayingAudio || isStopped())
423 return;
424
425 if (document()->hidden()) {
426 if (state() == State::Running) {
427 ALWAYS_LOG(LOGIDENTIFIER, "Suspending playback after going to the background");
428 m_mediaSession->beginInterruption(PlatformMediaSession::EnteringBackground);
429 }
430 } else {
431 if (state() == State::Interrupted) {
432 ALWAYS_LOG(LOGIDENTIFIER, "Resuming playback after entering foreground");
433 m_mediaSession->endInterruption(PlatformMediaSession::MayResumePlaying);
434 }
435 }
436}

438void AudioContext::suspend(ReasonForSuspension)
439{
440 if (isClosed() || m_wasSuspendedByScript)
441 return;
442
443 m_mediaSession->beginInterruption(PlatformMediaSession::PlaybackSuspended);
444 document()->updateIsPlayingMedia();
445}
446
447void AudioContext::resume()
448{
449 if (isClosed() || m_wasSuspendedByScript)
450 return;
451
452 m_mediaSession->endInterruption(PlatformMediaSession::MayResumePlaying);
453 document()->updateIsPlayingMedia();
454}

461void AudioContext::suspendPlayback()
462{
463 if (state() == State::Closed || !isInitialized())
464 return;
465
466 lazyInitialize();
467
468 destination().suspend([this, protectedThis = makeRef(*this), pendingActivity = makePendingActivity(*this)](std::optional<Exception>&& exception) {
469 if (exception)
470 return;
471
472 bool interrupted = m_mediaSession->state() == PlatformMediaSession::Interrupted;
473 setState(interrupted ? State::Interrupted : State::Suspended);
474 });
475}

501void AudioContext::isPlayingAudioDidChange()
502{
503 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
504 // explicitly allow the following allocation(s).
505 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
506
507 // Make sure to call Document::updateIsPlayingMedia() on the main thread, since
508 // we could be on the audio I/O thread here and the call into WebCore could block.
509 callOnMainThread([protectedThis = makeRef(*this)] {
510 if (auto* document = protectedThis->document())
511 document->updateIsPlayingMedia();
512 });
513}
