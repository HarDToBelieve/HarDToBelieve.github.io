42static void replaceNaNValues(float* values, unsigned numberOfValues, float defaultValue)
43{
44 for (unsigned i = 0; i < numberOfValues; ++i) {
45 if (std::isnan(values[i]))
46 values[i] = defaultValue;
47 }
48}

69float AudioParam::value()
70{
71 // Update value for timeline.
72 if (context() && context()->isAudioThread()) {
73 auto timelineValue = m_timeline.valueForContextTime(*context(), m_value, minValue(), maxValue());
74 if (timelineValue)
75 m_value = *timelineValue;
76 }
77
78 return m_value;

123bool AudioParam::smooth()
124{
125 if (!context())
126 return true;
127
128 // If values have been explicitly scheduled on the timeline, then use the exact value.
129 // Smoothing effectively is performed by the timeline.
130 auto timelineValue = m_timeline.valueForContextTime(*context(), m_value, minValue(), maxValue());
131 if (timelineValue)
132 m_value = *timelineValue;
133
134 if (m_smoothedValue == m_value) {
135 // Smoothed value has already approached and snapped to value.
136 return true;
137 }
138
139 if (timelineValue)
140 m_smoothedValue = m_value;
141 else {
142 // Dezipper - exponential approach.
143 m_smoothedValue += (m_value - m_smoothedValue) * SmoothingConstant;
144
145 // If we get close enough then snap to actual value.
146 if (fabs(m_smoothedValue - m_value) < SnapThreshold) // FIXME: the threshold needs to be adjustable depending on range - but this is OK general purpose value.
147 m_smoothedValue = m_value;
148 }
149
150 return false;
151}

275void AudioParam::calculateSampleAccurateValues(float* values, unsigned numberOfValues)
276{
277 bool isSafe = context() && context()->isAudioThread() && values && numberOfValues;
278 ASSERT(isSafe);
279 if (!isSafe)
280 return;
281
282 calculateFinalValues(values, numberOfValues, automationRate() == AutomationRate::ARate);
283}

285void AudioParam::calculateFinalValues(float* values, unsigned numberOfValues, bool sampleAccurate)
286{
287 bool isGood = context() && context()->isAudioThread() && values && numberOfValues;
288 ASSERT(isGood);
289 if (!isGood)
290 return;
291
292 // The calculated result will be the "intrinsic" value summed with all audio-rate connections.
293
294 if (sampleAccurate) {
295 // Calculate sample-accurate (a-rate) intrinsic values.
296 calculateTimelineValues(values, numberOfValues);
297 } else {
298 // Calculate control-rate (k-rate) intrinsic value.
299 auto timelineValue = m_timeline.valueForContextTime(*context(), m_value, minValue(), maxValue());
300
301 if (timelineValue)
302 m_value = *timelineValue;
303 std::fill_n(values, numberOfValues, m_value);
304 }
305
306 if (!numberOfRenderingConnections())
307 return;
308
309 // Now sum all of the audio-rate connections together (unity-gain summing junction).
310 // Note that connections would normally be mono, but we mix down to mono if necessary.
311 // If we're not sample accurate, we only need one value, so make the summing
312 // bus have length 1. When the connections are added in, only the first
313 // value will be added. Which is exactly what we want.
314 ASSERT(numberOfValues <= AudioUtilities::renderQuantumSize);
315 m_summingBus->setChannelMemory(0, values, sampleAccurate ? numberOfValues : 1);
316
317 for (auto& output : m_renderingOutputs) {
318 ASSERT(output);
319
320 // Render audio from this output.
321 AudioBus* connectionBus = output->pull(0, AudioUtilities::renderQuantumSize);
322
323 // Sum, with unity-gain.
324 m_summingBus->sumFrom(*connectionBus);
325 }
326
327 // If we're not sample accurate, duplicate the first element of |values| to all of the elements.
328 if (!sampleAccurate)
329 std::fill_n(values + 1, numberOfValues - 1, values[0]);
330
331 // As per https://webaudio.github.io/web-audio-api/#computation-of-value, we should replace NaN values
332 // with the default value.
333 replaceNaNValues(values, numberOfValues, m_defaultValue);
334
335 // Clamp values based on range allowed by AudioParam's min and max values.
336 VectorMath::clamp(values, minValue(), maxValue(), values, numberOfValues);
337}
338
339void AudioParam::calculateTimelineValues(float* values, unsigned numberOfValues)
340{
341 if (!context())
342 return;
343
344 // Calculate values for this render quantum.
345 // Normally numberOfValues will equal AudioUtilities::renderQuantumSize (the render quantum size).
346 double sampleRate = context()->sampleRate();
347 size_t startFrame = context()->currentSampleFrame();
348 size_t endFrame = startFrame + numberOfValues;
349
350 // Note we're running control rate at the sample-rate.
351 // Pass in the current value as default value.
352 m_value = m_timeline.valuesForFrameRange(startFrame, endFrame, m_value, minValue(), maxValue(), values, numberOfValues, sampleRate, sampleRate);
353}

355void AudioParam::connect(AudioNodeOutput* output)
356{
357 ASSERT(context());
358 ASSERT(context()->isGraphOwner());
359
360 ASSERT(output);
361 if (!output)
362 return;
363
364 if (!addOutput(*output))
365 return;
366
367 INFO_LOG(LOGIDENTIFIER, output->node()->nodeType());
368 output->addParam(this);
369}
370
371void AudioParam::disconnect(AudioNodeOutput* output)
372{
373 ASSERT(context());
374 ASSERT(context()->isGraphOwner());
375
376 ASSERT(output);
377 if (!output)
378 return;
379
380 INFO_LOG(LOGIDENTIFIER, output->node()->nodeType());
381
382 if (removeOutput((*output)))
383 output->removeParam(this);
384}
