29#pragma once
30
31#include "AudioContext.h"
32#include "AudioParamTimeline.h"
33#include "AudioSummingJunction.h"
34#include "AutomationRate.h"
35#include <JavaScriptCore/Float32Array.h>
36#include <sys/types.h>
37#include <wtf/LoggerHelper.h>
38#include <wtf/RefCounted.h>
39#include <wtf/text/WTFString.h>
40
41namespace WebCore {
42
43class AudioNodeOutput;
44
45enum class AutomationRateMode : bool { Fixed, Variable };
46
47class AudioParam final
48 : public AudioSummingJunction
49 , public RefCounted<AudioParam>
50#if !RELEASE_LOG_DISABLED
51 , private LoggerHelper
52#endif
53{
54public:
55 static constexpr double SmoothingConstant = 0.05;
56 static constexpr double SnapThreshold = 0.001;
57
58 static Ref<AudioParam> create(BaseAudioContext& context, const String& name, float defaultValue, float minValue, float maxValue, AutomationRate automationRate, AutomationRateMode automationRateMode = AutomationRateMode::Variable)
59 {
60 return adoptRef(*new AudioParam(context, name, defaultValue, minValue, maxValue, automationRate, automationRateMode));
61 }
62
63 // AudioSummingJunction
64 bool canUpdateState() override { return true; }
65 void didUpdate() override { }
66
67 // Intrinsic value.
68 float value();
69 void setValue(float);
70
71 float valueForBindings() const;
72 ExceptionOr<void> setValueForBindings(float);
73
74 AutomationRate automationRate() const { return m_automationRate; }
75 ExceptionOr<void> setAutomationRate(AutomationRate);
76
77 // Final value for k-rate parameters, otherwise use calculateSampleAccurateValues() for a-rate.
78 // Must be called in the audio thread.
79 float finalValue();
80
81 const String& name() const { return m_name; }
82
83 float minValue() const { return m_minValue; }
84 float maxValue() const { return m_maxValue; }
85 float defaultValue() const { return m_defaultValue; }
86
87 // Value smoothing:
88
89 // When a new value is set with setValue(), in our internal use of the parameter we don't immediately jump to it.
90 // Instead we smoothly approach this value to avoid glitching.
91 float smoothedValue();
92
93 // Smoothly exponentially approaches to (de-zippers) the desired value.
94 // Returns true if smoothed value has already snapped exactly to value.
95 bool smooth();
96
97 void resetSmoothedValue() { m_smoothedValue = m_value; }
98
99 // Parameter automation.
100 ExceptionOr<AudioParam&> setValueAtTime(float value, double startTime);
101 ExceptionOr<AudioParam&> linearRampToValueAtTime(float value, double endTime);
102 ExceptionOr<AudioParam&> exponentialRampToValueAtTime(float value, double endTime);
103 ExceptionOr<AudioParam&> setTargetAtTime(float target, double startTime, float timeConstant);
104 ExceptionOr<AudioParam&> setValueCurveAtTime(Vector<float>&& curve, double startTime, double duration);
105 ExceptionOr<AudioParam&> cancelScheduledValues(double cancelTime);
106 ExceptionOr<AudioParam&> cancelAndHoldAtTime(double cancelTime);
107
108 bool hasSampleAccurateValues() const;
109
110 // Calculates numberOfValues parameter values starting at the context's current time.
111 // Must be called in the context's render thread.
112 void calculateSampleAccurateValues(float* values, unsigned numberOfValues);
113
114 // Connect an audio-rate signal to control this parameter.
115 void connect(AudioNodeOutput*);
116 void disconnect(AudioNodeOutput*);
117
118protected:
119 AudioParam(BaseAudioContext&, const String&, float defaultValue, float minValue, float maxValue, AutomationRate, AutomationRateMode);
120
121private:
122 // sampleAccurate corresponds to a-rate (audio rate) vs. k-rate in the Web Audio specification.
123 void calculateFinalValues(float* values, unsigned numberOfValues, bool sampleAccurate);
124 void calculateTimelineValues(float* values, unsigned numberOfValues);
125
126#if !RELEASE_LOG_DISABLED
127 const Logger& logger() const final { return m_logger.get(); }
128 const void* logIdentifier() const final { return m_logIdentifier; }
129 const char* logClassName() const final { return "AudioParam"; }
130 WTFLogChannel& logChannel() const final;
131#endif
132
133 String m_name;
134 float m_value;
135 float m_defaultValue;
136 float m_minValue;
137 float m_maxValue;
138 AutomationRate m_automationRate;
139 AutomationRateMode m_automationRateMode;
140
141 // Smoothing (de-zippering)
142 float m_smoothedValue;
143
144 AudioParamTimeline m_timeline;
145 Ref<AudioBus> m_summingBus;
146
147#if !RELEASE_LOG_DISABLED
148 mutable Ref<const Logger> m_logger;
149 const void* m_logIdentifier;
150#endif
151};
152
153} // namespace WebCore
