37class AudioParam;
38class DOMMapAdapter;
39
40class AudioParamMap : public RefCounted<AudioParamMap> {

42 static Ref<AudioParamMap> create()
43 {
44 return adoptRef(*new AudioParamMap);
45 }
46 ~AudioParamMap();
47
48 void initializeMapLike(DOMMapAdapter&);
49 void add(const String&, Ref<AudioParam>&&);
50
51 const HashMap<String, Ref<AudioParam>>& map() const { return m_map; }
52
53private:

55
56 HashMap<String, Ref<AudioParam>> m_map;
57};

59} // namespace WebCore
