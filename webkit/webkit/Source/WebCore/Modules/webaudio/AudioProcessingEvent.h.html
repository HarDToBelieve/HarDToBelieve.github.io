25#pragma once
26
27#include "AudioBuffer.h"
28#include "Event.h"
29#include <wtf/RefPtr.h>
30
31namespace WebCore {
32
33class AudioBuffer;
34struct AudioProcessingEventInit;
35
36class AudioProcessingEvent final : public Event {
37 WTF_MAKE_ISO_ALLOCATED(AudioProcessingEvent);
38public:
39 static Ref<AudioProcessingEvent> create(RefPtr<AudioBuffer>&& inputBuffer, RefPtr<AudioBuffer>&& outputBuffer, double playbackTime)
40 {
41 return adoptRef(*new AudioProcessingEvent(WTFMove(inputBuffer), WTFMove(outputBuffer), playbackTime));
42 }
43
44 static Ref<AudioProcessingEvent> create(const AtomString&, AudioProcessingEventInit&&);
45
46 virtual ~AudioProcessingEvent();
47
48 AudioBuffer* inputBuffer() { return m_inputBuffer.get(); }
49 AudioBuffer* outputBuffer() { return m_outputBuffer.get(); }
50 double playbackTime() const { return m_playbackTime; }
51
52 EventInterface eventInterface() const override;
53
54private:
55 AudioProcessingEvent(RefPtr<AudioBuffer>&& inputBuffer, RefPtr<AudioBuffer>&& outputBuffer, double playbackTime);
56 AudioProcessingEvent(const AtomString&, AudioProcessingEventInit&&);
57
58 RefPtr<AudioBuffer> m_inputBuffer;
59 RefPtr<AudioBuffer> m_outputBuffer;
60 double m_playbackTime;
61};
62
63} // namespace WebCore
