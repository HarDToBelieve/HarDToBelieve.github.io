57void AudioScheduledSourceNode::updateSchedulingInfo(size_t quantumFrameSize, AudioBus& outputBus, size_t& quantumFrameOffset, size_t& nonSilentFramesToProcess, double& startFrameOffset)
58{
59 nonSilentFramesToProcess = 0;
60 quantumFrameOffset = 0;
61 startFrameOffset = 0;
62
63 ASSERT(quantumFrameSize == AudioUtilities::renderQuantumSize);
64 if (quantumFrameSize != AudioUtilities::renderQuantumSize)
65 return;
66
67 double sampleRate = this->sampleRate();
68
69 // quantumStartFrame : Start frame of the current time quantum.
70 // quantumEndFrame : End frame of the current time quantum.
71 // startFrame : Start frame for this source.
72 // endFrame : End frame for this source.
73 size_t quantumStartFrame = context().currentSampleFrame();
74 size_t quantumEndFrame = quantumStartFrame + quantumFrameSize;
75
76 // Round up if the start time isn't on a frame boundary so we don't start too early.
77 size_t startFrame = AudioUtilities::timeToSampleFrame(m_startTime, sampleRate, AudioUtilities::SampleFrameRounding::Up);
78 size_t endFrame = 0;
79 if (m_endTime) {
80 // The end frame is the end time rounded up because it is an exclusive upper
81 // bound of the end time. We also need to take care to handle huge end
82 // times and clamp the corresponding frame to the largest size_t value.
83 endFrame = AudioUtilities::timeToSampleFrame(*m_endTime, sampleRate, AudioUtilities::SampleFrameRounding::Up);
84 }
85
86 // If we know the end time and it's already passed, then don't bother doing any more rendering this cycle.
87 if (m_endTime && endFrame <= quantumStartFrame)
88 finish();
89
90 if (m_playbackState == UNSCHEDULED_STATE || m_playbackState == FINISHED_STATE || startFrame >= quantumEndFrame) {
91 // Output silence.
92 outputBus.zero();
93 return;
94 }
95
96 // Check if it's time to start playing.
97 if (m_playbackState == SCHEDULED_STATE) {
98 // Increment the active source count only if we're transitioning from SCHEDULED_STATE to PLAYING_STATE.
99 m_playbackState = PLAYING_STATE;
100 // NOTE: startFrameOffset is usually negative, but may not be because of
101 // the rounding that may happen in computing |startFrame| above.
102 startFrameOffset = m_startTime * sampleRate - startFrame;
103 }
104
105 quantumFrameOffset = startFrame > quantumStartFrame ? startFrame - quantumStartFrame : 0;
106 quantumFrameOffset = std::min(quantumFrameOffset, quantumFrameSize); // clamp to valid range
107 nonSilentFramesToProcess = quantumFrameSize - quantumFrameOffset;
108
109 if (!nonSilentFramesToProcess) {
110 // Output silence.
111 outputBus.zero();
112 return;
113 }
114
115 // Handle silence before we start playing.
116 // Zero any initial frames representing silence leading up to a rendering start time in the middle of the quantum.
117 if (quantumFrameOffset) {
118 for (unsigned i = 0; i < outputBus.numberOfChannels(); ++i)
119 memset(outputBus.channel(i)->mutableData(), 0, sizeof(float) * quantumFrameOffset);
120 }
121
122 // Handle silence after we're done playing.
123 // If the end time is somewhere in the middle of this time quantum, then zero out the
124 // frames from the end time to the very end of the quantum.
125 if (m_endTime && endFrame >= quantumStartFrame && endFrame < quantumEndFrame) {
126 size_t zeroStartFrame = endFrame - quantumStartFrame;
127 size_t framesToZero = quantumFrameSize - zeroStartFrame;
128
129 ASSERT(zeroStartFrame < quantumFrameSize);
130 ASSERT(framesToZero <= quantumFrameSize);
131 ASSERT(zeroStartFrame + framesToZero <= quantumFrameSize);
132
133 bool isSafe = zeroStartFrame < quantumFrameSize && framesToZero <= quantumFrameSize && zeroStartFrame + framesToZero <= quantumFrameSize;
134 if (isSafe) {
135 if (framesToZero > nonSilentFramesToProcess)
136 nonSilentFramesToProcess = 0;
137 else
138 nonSilentFramesToProcess -= framesToZero;
139
140 for (unsigned i = 0; i < outputBus.numberOfChannels(); ++i)
141 memset(outputBus.channel(i)->mutableData() + zeroStartFrame, 0, sizeof(float) * framesToZero);
142 }
143
144 finish();
145 }
146}

148ExceptionOr<void> AudioScheduledSourceNode::startLater(double when)
149{
150 ASSERT(isMainThread());
151 ALWAYS_LOG(LOGIDENTIFIER, when);
152
153 if (m_playbackState != UNSCHEDULED_STATE)
154 return Exception { InvalidStateError, "Cannot call start() more than once"_s };
155
156 if (!std::isfinite(when) || when < 0)
157 return Exception { RangeError, "when value should be positive"_s };
158
159 context().sourceNodeWillBeginPlayback(*this);
160
161 m_startTime = when;
162 m_playbackState = SCHEDULED_STATE;
163
164 return { };
165}
166
167ExceptionOr<void> AudioScheduledSourceNode::stopLater(double when)
168{
169 ASSERT(isMainThread());
170 ALWAYS_LOG(LOGIDENTIFIER, when);
171
172 if (m_playbackState == UNSCHEDULED_STATE)
173 return Exception { InvalidStateError, "cannot call stop without calling start first."_s };
174
175 if (!std::isfinite(when) || when < 0)
176 return Exception { RangeError, "when value should be positive"_s };
177
178 m_endTime = when;
179
180 return { };
181}

183bool AudioScheduledSourceNode::virtualHasPendingActivity() const
184{
185 return m_hasEndedEventListener && m_playbackState != FINISHED_STATE && !isMarkedForDeletion() && !context().isClosed();
186}
187
188void AudioScheduledSourceNode::eventListenersDidChange()
189{
190 m_hasEndedEventListener = hasEventListeners(eventNames().endedEvent);
191}
192
193void AudioScheduledSourceNode::finish()
194{
195 ASSERT(!hasFinished());
196 // Let the context dereference this AudioNode.
197 context().sourceNodeDidFinishPlayback(*this);
198 m_playbackState = FINISHED_STATE;
199
200 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
201 // explicitly allow the following allocation(s).
202 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
203 callOnMainThread([this, protectedThis = makeRef(*this), pendingActivity = makePendingActivity(*this)] {
204 if (context().isStopped())
205 return;
206 this->dispatchEvent(Event::create(eventNames().endedEvent, Event::CanBubble::No, Event::IsCancelable::No));
207 });
208}
