29#pragma once
30
31#include "ActiveDOMObject.h"
32#include "AudioNode.h"
33
34namespace WebCore {
35
36class AudioScheduledSourceNode : public AudioNode, public ActiveDOMObject {
37 WTF_MAKE_ISO_ALLOCATED(AudioScheduledSourceNode);
38public:
39 // These are the possible states an AudioScheduledSourceNode can be in:
40 //
41 // UNSCHEDULED_STATE - Initial playback state. Created, but not yet scheduled.
42 // SCHEDULED_STATE - Scheduled to play but not yet playing.
43 // PLAYING_STATE - Generating sound.
44 // FINISHED_STATE - Finished generating sound.
45 //
46 // The state can only transition to the next state, except for the FINISHED_STATE which can
47 // never be changed.
48 enum PlaybackState {
49 // These must be defined with the same names and values as in the .idl file.
50 UNSCHEDULED_STATE = 0,
51 SCHEDULED_STATE = 1,
52 PLAYING_STATE = 2,
53 FINISHED_STATE = 3
54 };
55
56 AudioScheduledSourceNode(BaseAudioContext&, NodeType);
57
58 ExceptionOr<void> startLater(double when);
59 ExceptionOr<void> stopLater(double when);
60
61 unsigned short playbackState() const { return static_cast<unsigned short>(m_playbackState); }
62 bool isPlayingOrScheduled() const { return m_playbackState == PLAYING_STATE || m_playbackState == SCHEDULED_STATE; }
63 bool hasFinished() const { return m_playbackState == FINISHED_STATE; }
64
65protected:
66 // Get frame information for the current time quantum.
67 // We handle the transition into PLAYING_STATE and FINISHED_STATE here,
68 // zeroing out portions of the outputBus which are outside the range of startFrame and endFrame.
69 // Each frame time is relative to the context's currentSampleFrame().
70 // quantumFrameOffset: Offset frame in this time quantum to start rendering.
71 // nonSilentFramesToProcess: Number of frames rendering non-silence (will be <= quantumFrameSize).
72 // startFrameOffset : The fractional frame offset from quantumFrameOffset and the actual starting
73 // time of the source. This is non-zero only when transitioning from the
74 // SCHEDULED_STATE to the PLAYING_STATE.
75 void updateSchedulingInfo(size_t quantumFrameSize, AudioBus& outputBus, size_t& quantumFrameOffset, size_t& nonSilentFramesToProcess, double& startFrameOffset);
76
77 // Called when we have no more sound to play or the noteOff() time has been reached.
78 virtual void finish();
79
80 bool virtualHasPendingActivity() const final;
81 void eventListenersDidChange() final;
82
83 bool requiresTailProcessing() const final { return false; }
84
85 PlaybackState m_playbackState { UNSCHEDULED_STATE };
86
87 // m_startTime is the time to start playing based on the context's timeline (0 or a time less than the context's current time means "now").
88 double m_startTime { 0 }; // in seconds
89
90 // m_endTime is the time to stop playing based on the context's timeline (0 or a time less than the context's current time means "now").
91 // If it hasn't been set explicitly, then the sound will not stop playing (if looping) or will stop when the end of the AudioBuffer
92 // has been reached.
93 std::optional<double> m_endTime; // in seconds
94 bool m_hasEndedEventListener { false };
95};

97} // namespace WebCore
