25#pragma once
26
27#include "AudioBus.h"
28#include <wtf/HashSet.h>
29#include <wtf/Vector.h>
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class BaseAudioContext;
35class AudioNodeOutput;
36
37// An AudioSummingJunction represents a point where zero, one, or more AudioNodeOutputs connect.
38
39class AudioSummingJunction {
40public:
41 explicit AudioSummingJunction(BaseAudioContext&);
42 virtual ~AudioSummingJunction();
43
44 // Can be called from any thread.
45 BaseAudioContext* context() { return m_context.get(); }
46 const BaseAudioContext* context() const { return m_context.get(); }
47
48 // This copies m_outputs to m_renderingOutputs. Please see comments for these lists below.
49 // This must be called when we own the context's graph lock in the audio thread at the very start or end of the render quantum.
50 void updateRenderingState();
51
52 // Rendering code accesses its version of the current connections here.
53 unsigned numberOfRenderingConnections() const { return m_renderingOutputs.size(); }
54 AudioNodeOutput* renderingOutput(unsigned i) { return m_renderingOutputs[i]; }
55 const AudioNodeOutput* renderingOutput(unsigned i) const { return m_renderingOutputs[i]; }
56 bool isConnected() const { return numberOfRenderingConnections() > 0; }
57
58 virtual bool canUpdateState() = 0;
59 virtual void didUpdate() = 0;
60
61 bool addOutput(AudioNodeOutput&);
62 bool removeOutput(AudioNodeOutput&);
63
64 void markRenderingStateAsDirty();
65
66protected:
67 WeakPtr<BaseAudioContext> m_context;
68
69 // numberOfConnections() should never be called from the audio rendering thread.
70 // Instead numberOfRenderingConnections() and renderingOutput() should be used.
71 unsigned numberOfConnections() const { return m_outputs.size(); }
72
73 unsigned maximumNumberOfChannels() const;
74
75 // m_renderingOutputs is a copy of m_outputs which will never be modified during the graph rendering on the audio thread.
76 // This is the list which is used by the rendering code.
77 // Whenever m_outputs is modified, the context is told so it can later update m_renderingOutputs from m_outputs at a safe time.
78 // Most of the time, m_renderingOutputs is identical to m_outputs.
79 Vector<AudioNodeOutput*> m_renderingOutputs;
80
81 // m_renderingStateNeedUpdating keeps track if m_outputs is modified.
82 bool m_renderingStateNeedUpdating { false };
83
84private:
85 // m_outputs contains the AudioNodeOutputs representing current connections which are not disabled.
86 // The rendering code should never use this directly, but instead uses m_renderingOutputs.
87 HashSet<AudioNodeOutput*> m_outputs;
88 Vector<AudioNodeOutput*> m_pendingRenderingOutputs;
89};
