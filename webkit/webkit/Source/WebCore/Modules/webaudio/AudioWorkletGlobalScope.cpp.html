68// https://www.w3.org/TR/webaudio/#dom-audioworkletglobalscope-registerprocessor
69ExceptionOr<void> AudioWorkletGlobalScope::registerProcessor(String&& name, Ref<JSAudioWorkletProcessorConstructor>&& processorContructor)
70{
71 ASSERT(!isMainThread());
72
73 if (name.isEmpty())
74 return Exception { NotSupportedError, "Name cannot be the empty string"_s };
75
76 if (m_processorConstructorMap.contains(name))
77 return Exception { NotSupportedError, "A processor was already registered with this name"_s };
78
79 JSC::JSObject* jsConstructor = processorContructor->callbackData()->callback();
80 auto* globalObject = jsConstructor->globalObject();
81 auto& vm = globalObject->vm();
82 auto scope = DECLARE_THROW_SCOPE(vm);
83
84 if (!jsConstructor->isConstructor(vm))
85 return Exception { TypeError, "Class definition passed to registerProcessor() is not a constructor"_s };
86
87 auto prototype = jsConstructor->getPrototype(vm, globalObject);
88 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
89
90 if (!prototype.isObject())
91 return Exception { TypeError, "Class definition passed to registerProcessor() has invalid prototype"_s };
92
93 auto parameterDescriptorsValue = jsConstructor->get(globalObject, JSC::Identifier::fromString(vm, "parameterDescriptors"));
94 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
95
96 Vector<AudioParamDescriptor> parameterDescriptors;
97 if (!parameterDescriptorsValue.isUndefined()) {
98 parameterDescriptors = convert<IDLSequence<IDLDictionary<AudioParamDescriptor>>>(*globalObject, parameterDescriptorsValue);
99 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
100 UNUSED_PARAM(parameterDescriptors);
101 HashSet<String> paramNames;
102 for (auto& descriptor : parameterDescriptors) {
103 auto addResult = paramNames.add(descriptor.name);
104 if (!addResult.isNewEntry)
105 return Exception { NotSupportedError, makeString("parameterDescriptors contain duplicate AudioParam name: ", name) };
106 if (descriptor.defaultValue < descriptor.minValue)
107 return Exception { InvalidStateError, makeString("AudioParamDescriptor with name '", name, "' has a defaultValue that is less than the minValue") };
108 if (descriptor.defaultValue > descriptor.maxValue)
109 return Exception { InvalidStateError, makeString("AudioParamDescriptor with name '", name, "' has a defaultValue that is greater than the maxValue") };
110 }
111 }
112
113 auto addResult = m_processorConstructorMap.add(name, WTFMove(processorContructor));
114
115 // We've already checked at the beginning of this function but then we ran some JS so we need to check again.
116 if (!addResult.isNewEntry)
117 return Exception { NotSupportedError, "A processor was already registered with this name"_s };
118
119 thread().messagingProxy().postTaskToAudioWorklet([name = name.isolatedCopy(), parameterDescriptors = crossThreadCopy(parameterDescriptors)](AudioWorklet& worklet) mutable {
120 ASSERT(isMainThread());
121 if (auto* audioContext = worklet.audioContext())
122 audioContext->addAudioParamDescriptors(name, WTFMove(parameterDescriptors));
123 });
124
125 return { };
126}

128RefPtr<AudioWorkletProcessor> AudioWorkletGlobalScope::createProcessor(const String& name, TransferredMessagePort port, Ref<SerializedScriptValue>&& options)
129{
130 auto constructor = m_processorConstructorMap.get(name);
131 ASSERT(constructor);
132 if (!constructor)
133 return nullptr;
134
135 JSC::JSObject* jsConstructor = constructor->callbackData()->callback();
136 auto* globalObject = constructor->callbackData()->globalObject();
137 JSC::VM& vm = globalObject->vm();
138 auto scope = DECLARE_THROW_SCOPE(vm);
139 JSC::JSLockHolder lock { globalObject };
140
141 m_pendingProcessorConstructionData = makeUnique<AudioWorkletProcessorConstructionData>(String { name }, MessagePort::entangle(*this, WTFMove(port)));
142
143 JSC::MarkedArgumentBuffer args;
144 auto arg = options->deserialize(*globalObject, globalObject, SerializationErrorMode::NonThrowing);
145 RETURN_IF_EXCEPTION(scope, nullptr);
146 args.append(arg);
147 ASSERT(!args.hasOverflowed());
148
149 auto* object = JSC::construct(globalObject, jsConstructor, args, "Failed to construct AudioWorkletProcessor");
150 ASSERT(!!scope.exception() == !object);
151 RETURN_IF_EXCEPTION(scope, nullptr);
152
153 auto* jsProcessor = JSC::jsDynamicCast<JSAudioWorkletProcessor*>(vm, object);
154 if (!jsProcessor)
155 return nullptr;
156
157 jsProcessor->wrapped().setProcessCallback(makeUnique<JSCallbackDataStrong>(jsProcessor, globalObject));
158
159 return &jsProcessor->wrapped();
160}

179void AudioWorkletGlobalScope::handlePreRenderTasks()
180{
181 // We grab the JS API lock at the beginning of rendering and release it at the end of rendering.
182 // This makes sure that we only drain the MicroTask queue after each render quantum.
183 // It is only safe to grab the lock if we are on the context thread. We might get called on
184 // another thread if audio rendering started before the audio worklet got started.
185 if (isContextThread())
186 m_lockDuringRendering.emplace(script()->vm());
187}
188
189void AudioWorkletGlobalScope::handlePostRenderTasks(size_t currentFrame)
190{
191 m_currentFrame = currentFrame;
192
193 {
194 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
195 // explicitly allow the following allocation(s).
196 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
197 // This takes care of processing the MicroTask queue after rendering.
198 m_lockDuringRendering = std::nullopt;
199 }
200}
