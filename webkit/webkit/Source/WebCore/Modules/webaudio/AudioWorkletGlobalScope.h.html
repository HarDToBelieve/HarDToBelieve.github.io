31#if ENABLE(WEB_AUDIO)
32#include "AudioWorkletThread.h"
33#include "MessagePort.h"

36namespace JSC {
37class VM;
38}
39
40namespace WebCore {
41
42class AudioWorkletProcessorConstructionData;
43class AudioWorkletProcessor;
44class AudioWorkletThread;
45class JSAudioWorkletProcessorConstructor;
46
47struct WorkletParameters;
48
49class AudioWorkletGlobalScope final : public WorkletGlobalScope {
50 WTF_MAKE_ISO_ALLOCATED(AudioWorkletGlobalScope);
51public:
52 static RefPtr<AudioWorkletGlobalScope> tryCreate(AudioWorkletThread&, const WorkletParameters&);
53 ~AudioWorkletGlobalScope();
54
55 ExceptionOr<void> registerProcessor(String&& name, Ref<JSAudioWorkletProcessorConstructor>&&);
56 RefPtr<AudioWorkletProcessor> createProcessor(const String& name, TransferredMessagePort, Ref<SerializedScriptValue>&& options);
57
58 size_t currentFrame() const { return m_currentFrame; }
59
60 float sampleRate() const { return m_sampleRate; }
61
62 double currentTime() const { return m_sampleRate > 0.0 ? m_currentFrame / static_cast<double>(m_sampleRate) : 0.0; }
63
64 AudioWorkletThread& thread() const;
65 void prepareForDestruction() final;
66
67 std::unique_ptr<AudioWorkletProcessorConstructionData> takePendingProcessorConstructionData();
68
69 void handlePreRenderTasks();
70 void handlePostRenderTasks(size_t currentFrame);
71
72 FetchOptions::Destination destination() const final { return FetchOptions::Destination::Audioworklet; }
73
74private:
75 AudioWorkletGlobalScope(AudioWorkletThread&, Ref<JSC::VM>&&, const WorkletParameters&);
76
77 bool isAudioWorkletGlobalScope() const final { return true; }
78
79 size_t m_currentFrame { 0 };
80 const float m_sampleRate;
81 HashMap<String, RefPtr<JSAudioWorkletProcessorConstructor>> m_processorConstructorMap;
82 std::unique_ptr<AudioWorkletProcessorConstructionData> m_pendingProcessorConstructionData;
83 std::optional<JSC::JSLockHolder> m_lockDuringRendering;
84};
85
86} // namespace WebCore

88SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::AudioWorkletGlobalScope)
89static bool isType(const WebCore::ScriptExecutionContext& context) { return is<WebCore::WorkletGlobalScope>(context) && downcast<WebCore::WorkletGlobalScope>(context).isAudioWorkletGlobalScope(); }
90static bool isType(const WebCore::WorkletGlobalScope& context) { return context.isAudioWorkletGlobalScope(); }
91SPECIALIZE_TYPE_TRAITS_END()
