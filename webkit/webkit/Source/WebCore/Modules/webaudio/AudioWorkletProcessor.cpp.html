68static JSFloat32Array* constructJSFloat32Array(JSGlobalObject& globalObject, unsigned length, const float* data = nullptr)
69{
70 auto* jsArray = JSFloat32Array::create(&globalObject, globalObject.typedArrayStructure(TypeFloat32), length);
71 if (data)
72 memcpy(jsArray->typedVector(), data, sizeof(float) * length);
73 return jsArray;
74}

76static JSObject* constructFrozenKeyValueObject(VM& vm, JSGlobalObject& globalObject, const HashMap<String, std::unique_ptr<AudioFloatArray>>& paramValuesMap)
77{
78 auto scope = DECLARE_THROW_SCOPE(vm);
79 auto* plainObjectStructure = JSFinalObject::createStructure(vm, &globalObject, globalObject.objectPrototype(), 0);
80 auto* object = JSFinalObject::create(vm, plainObjectStructure);
81 for (auto& pair : paramValuesMap) {
82 PutPropertySlot slot(object, false, PutPropertySlot::PutById);
83 // Per the specification, if the value is constant, we pass the JS an array with length 1, with the array item being the constant.
84 unsigned jsArraySize = pair.value->containsConstantValue() ? 1 : pair.value->size();
85 object->putInline(&globalObject, Identifier::fromString(vm, pair.key), constructJSFloat32Array(globalObject, jsArraySize, pair.value->data()), slot);
86 }
87 JSC::objectConstructorFreeze(&globalObject, object);
88 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
89 return object;
90}

92enum class ShouldPopulateWithBusData : bool { No, Yes };
93
94template <typename T>
95static JSArray* constructFrozenJSArray(VM& vm, JSGlobalObject& globalObject, JSC::ThrowScope& scope, const T& bus, ShouldPopulateWithBusData shouldPopulateWithBusData)
96{
97 unsigned numberOfChannels = busChannelCount(bus.get());
98 auto* channelsData = JSArray::create(vm, globalObject.originalArrayStructureForIndexingType(ArrayWithContiguous), numberOfChannels);
99 for (unsigned j = 0; j < numberOfChannels; ++j) {
100 auto* channel = bus->channel(j);
101 channelsData->setIndexQuickly(vm, j, constructJSFloat32Array(globalObject, channel->length(), shouldPopulateWithBusData == ShouldPopulateWithBusData::Yes ? channel->data() : nullptr));
102 }
103 JSC::objectConstructorFreeze(&globalObject, channelsData);
104 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
105 return channelsData;
106}
107
108template <typename T>
109static JSArray* constructFrozenJSArray(VM& vm, JSGlobalObject& globalObject, const Vector<T>& buses, ShouldPopulateWithBusData shouldPopulateWithBusData)
110{
111 auto scope = DECLARE_THROW_SCOPE(vm);
112 auto* array = JSArray::create(vm, globalObject.originalArrayStructureForIndexingType(ArrayWithContiguous), buses.size());
113 for (unsigned i = 0; i < buses.size(); ++i)
114 array->setIndexQuickly(vm, i, constructFrozenJSArray(vm, globalObject, scope, buses[i], shouldPopulateWithBusData));
115 JSC::objectConstructorFreeze(&globalObject, array);
116 EXCEPTION_ASSERT(!scope.exception());
117 return array;
118}

120static void copyDataFromJSArrayToBuses(JSGlobalObject& globalObject, const JSArray& jsArray, Vector<Ref<AudioBus>>& buses)
121{
122 // We can safely make assumptions about the structure of the JSArray since we use frozen arrays.
123 for (unsigned i = 0; i < buses.size(); ++i) {
124 auto& bus = buses[i];
125 auto* channelsArray = jsCast<JSArray*>(jsArray.getIndex(&globalObject, i));
126 for (unsigned j = 0; j < bus->numberOfChannels(); ++j) {
127 auto* channel = bus->channel(j);
128 auto* jsChannelData = jsCast<JSFloat32Array*>(channelsArray->getIndex(&globalObject, j));
129 if (jsChannelData->length() == channel->length())
130 memcpy(channel->mutableData(), jsChannelData->typedVector(), sizeof(float) * channel->length());
131 else
132 channel->zero();
133 }
134 }
135}

137static bool copyDataFromBusesToJSArray(VM& vm, JSGlobalObject& globalObject, const Vector<RefPtr<AudioBus>>& buses, JSArray* jsArray)
138{
139 if (!jsArray)
140 return false;
141
142 for (size_t busIndex = 0; busIndex < buses.size(); ++busIndex) {
143 auto& bus = buses[busIndex];
144 auto* jsChannelsArray = jsDynamicCast<JSArray*>(vm, jsArray->getIndex(&globalObject, busIndex));
145 unsigned numberOfChannels = busChannelCount(bus.get());
146 if (!jsChannelsArray || jsChannelsArray->length() != numberOfChannels)
147 return false;
148 for (unsigned channelIndex = 0; channelIndex < numberOfChannels; ++channelIndex) {
149 auto* channel = bus->channel(channelIndex);
150 auto* jsChannelArray = jsDynamicCast<JSFloat32Array*>(vm, jsChannelsArray->getIndex(&globalObject, channelIndex));
151 if (!jsChannelArray || jsChannelArray->length() != channel->length())
152 return false;
153 memcpy(jsChannelArray->typedVector(), channel->mutableData(), sizeof(float) * jsChannelArray->length());
154 }
155 }
156 return true;
157}

159static bool copyDataFromParameterMapToJSObject(VM& vm, JSGlobalObject& globalObject, const HashMap<String, std::unique_ptr<AudioFloatArray>>& paramValuesMap, JSObject* jsObject)
160{
161 if (!jsObject)
162 return false;
163
164 for (auto& pair : paramValuesMap) {
165 auto* jsTypedArray = jsDynamicCast<JSFloat32Array*>(vm, jsObject->get(&globalObject, Identifier::fromString(vm, pair.key)));
166 if (!jsTypedArray)
167 return false;
168 unsigned expectedLength = pair.value->containsConstantValue() ? 1 : pair.value->size();
169 if (jsTypedArray->length() != expectedLength)
170 return false;
171 memcpy(jsTypedArray->typedVector(), pair.value->data(), sizeof(float) * jsTypedArray->length());
172 }
173 return true;
174}

176static bool zeroJSArray(VM& vm, JSGlobalObject& globalObject, const Vector<Ref<AudioBus>>& outputs, JSArray* jsArray)
177{
178 if (!jsArray)
179 return false;
180
181 for (size_t busIndex = 0; busIndex < outputs.size(); ++busIndex) {
182 auto& bus = outputs[busIndex];
183 auto* jsChannelsArray = jsDynamicCast<JSArray*>(vm, jsArray->getIndex(&globalObject, busIndex));
184 unsigned numberOfChannels = busChannelCount(bus.get());
185 if (!jsChannelsArray || jsChannelsArray->length() != numberOfChannels)
186 return false;
187 for (unsigned channelIndex = 0; channelIndex < numberOfChannels; ++channelIndex) {
188 auto* channel = bus->channel(channelIndex);
189 auto* jsChannelArray = jsDynamicCast<JSFloat32Array*>(vm, jsChannelsArray->getIndex(&globalObject, channelIndex));
190 if (!jsChannelArray || jsChannelArray->length() != channel->length())
191 return false;
192 memset(jsChannelArray->typedVector(), 0, sizeof(float) * jsChannelArray->length());
193 }
194 }
195 return true;
196}

216void AudioWorkletProcessor::buildJSArguments(VM& vm, JSGlobalObject& globalObject, MarkedArgumentBuffer& args, const Vector<RefPtr<AudioBus>>& inputs, Vector<Ref<AudioBus>>& outputs, const HashMap<String, std::unique_ptr<AudioFloatArray>>& paramValuesMap)
217{
218 // For performance reasons, we cache the arrays passed to JS and reconstruct them only when the topology changes.
219 if (!copyDataFromBusesToJSArray(vm, globalObject, inputs, toJSArray(m_jsInputs)))
220 m_jsInputs = { constructFrozenJSArray(vm, globalObject, inputs, ShouldPopulateWithBusData::Yes) };
221 args.append(m_jsInputs);
222
223 if (!zeroJSArray(vm, globalObject, outputs, toJSArray(m_jsOutputs)))
224 m_jsOutputs = { constructFrozenJSArray(vm, globalObject, outputs, ShouldPopulateWithBusData::No) };
225 args.append(m_jsOutputs);
226
227 if (!copyDataFromParameterMapToJSObject(vm, globalObject, paramValuesMap, toJSObject(m_jsParamValues)))
228 m_jsParamValues = { constructFrozenKeyValueObject(vm, globalObject, paramValuesMap) };
229 args.append(m_jsParamValues);
230}

232bool AudioWorkletProcessor::process(const Vector<RefPtr<AudioBus>>& inputs, Vector<Ref<AudioBus>>& outputs, const HashMap<String, std::unique_ptr<AudioFloatArray>>& paramValuesMap, bool& threwException)
233{
234 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
235 // explicitly allow the following allocation(s).
236 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
237
238 ASSERT(m_processCallback);
239 auto& globalObject = *m_processCallback->globalObject();
240 ASSERT(globalObject.scriptExecutionContext());
241 ASSERT(globalObject.scriptExecutionContext()->isContextThread());
242
243 auto& vm = globalObject.vm();
244 JSLockHolder lock(vm);
245
246 MarkedArgumentBuffer args;
247 buildJSArguments(vm, globalObject, args, inputs, outputs, paramValuesMap);
248
249 NakedPtr<JSC::Exception> returnedException;
250 auto result = m_processCallback->invokeCallback(jsUndefined(), args, JSCallbackData::CallbackType::Object, Identifier::fromString(vm, "process"), returnedException);
251 if (returnedException) {
252 reportException(&globalObject, returnedException);
253 threwException = true;
254 return false;
255 }
256
257 copyDataFromJSArrayToBuses(globalObject, *toJSArray(m_jsOutputs), outputs);
258
259 return result.toBoolean(&globalObject);
260}
