36ExceptionOr<Ref<BiquadFilterNode>> BiquadFilterNode::create(BaseAudioContext& context, const BiquadFilterOptions& options)
37{
38 auto node = adoptRef(*new BiquadFilterNode(context));
39
40 auto result = node->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
41 if (result.hasException())
42 return result.releaseException();
43
44 node->setType(options.type);
45 node->q().setValue(options.Q);
46 node->detune().setValue(options.detune);
47 node->frequency().setValue(options.frequency);
48 node->gain().setValue(options.gain);
49
50 return node;
51}

53BiquadFilterNode::BiquadFilterNode(BaseAudioContext& context)
54 : AudioBasicProcessorNode(context, NodeTypeBiquadFilter)
55{
56 // Initially setup as lowpass filter.
57 m_processor = makeUnique<BiquadProcessor>(context, context.sampleRate(), 1, false);
58
59 // Initialize so that AudioParams can be processed.
60 initialize();
61}
62
63BiquadFilterType BiquadFilterNode::type() const
64{
65 return const_cast<BiquadFilterNode*>(this)->biquadProcessor()->type();
66}
67
68void BiquadFilterNode::setType(BiquadFilterType type)
69{
70 biquadProcessor()->setType(type);
71}
72
73ExceptionOr<void> BiquadFilterNode::getFrequencyResponse(const Ref<Float32Array>& frequencyHz, const Ref<Float32Array>& magResponse, const Ref<Float32Array>& phaseResponse)
74{
75 unsigned length = frequencyHz->length();
76 if (magResponse->length() != length || phaseResponse->length() != length)
77 return Exception { InvalidStateError, "The arrays passed as arguments must have the same length" };
78
79 if (length)
80 biquadProcessor()->getFrequencyResponse(length, frequencyHz->data(), magResponse->data(), phaseResponse->data());
81 return { };
82}
