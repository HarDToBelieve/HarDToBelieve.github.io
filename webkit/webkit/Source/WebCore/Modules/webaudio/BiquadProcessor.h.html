26
27#include "AudioDSPKernel.h"
28#include "AudioDSPKernelProcessor.h"
29#include "AudioNode.h"
30#include "AudioParam.h"
31#include "Biquad.h"
32#include "BiquadFilterType.h"
33#include <memory>
34#include <wtf/RefPtr.h>
35
36namespace WebCore {
37
38// BiquadProcessor is an AudioDSPKernelProcessor which uses Biquad objects to implement several common filters.
39
40class BiquadProcessor final : public AudioDSPKernelProcessor {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 BiquadProcessor(BaseAudioContext&, float sampleRate, size_t numberOfChannels, bool autoInitialize);
44
45 virtual ~BiquadProcessor();
46
47 std::unique_ptr<AudioDSPKernel> createKernel() override;
48
49 void process(const AudioBus* source, AudioBus* destination, size_t framesToProcess) override;
50 void processOnlyAudioParams(size_t framesToProcess) final;
51
52 // Get the magnitude and phase response of the filter at the given
53 // set of frequencies (in Hz). The phase response is in radians.
54 void getFrequencyResponse(unsigned nFrequencies, const float* frequencyHz, float* magResponse, float* phaseResponse);
55
56 void checkForDirtyCoefficients();
57
58 bool filterCoefficientsDirty() const { return m_filterCoefficientsDirty; }
59 bool hasSampleAccurateValues() const { return m_hasSampleAccurateValues; }
60
61 AudioParam& parameter1() { return m_parameter1.get(); }
62 AudioParam& parameter2() { return m_parameter2.get(); }
63 AudioParam& parameter3() { return m_parameter3.get(); }
64 AudioParam& parameter4() { return m_parameter4.get(); }
65
66 BiquadFilterType type() const { return m_type; }
67 void setType(BiquadFilterType);
68
69 bool shouldUseARate() const { return m_shouldUseARate; }
70
71private:
72 BiquadFilterType m_type { BiquadFilterType::Lowpass };
73
74 Ref<AudioParam> m_parameter1;
75 Ref<AudioParam> m_parameter2;
76 Ref<AudioParam> m_parameter3;
77 Ref<AudioParam> m_parameter4;
78
79 // so DSP kernels know when to re-compute coefficients
80 bool m_filterCoefficientsDirty { true };
81
82 // Set to true if any of the filter parameters are sample-accurate.
83 bool m_hasSampleAccurateValues { false };
84
85 bool m_shouldUseARate { true };
86};
87
88} // namespace WebCore
