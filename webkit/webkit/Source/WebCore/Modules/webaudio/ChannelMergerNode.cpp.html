44ExceptionOr<Ref<ChannelMergerNode>> ChannelMergerNode::create(BaseAudioContext& context, const ChannelMergerOptions& options)
45{
46 if (options.numberOfInputs > AudioContext::maxNumberOfChannels || !options.numberOfInputs)
47 return Exception { IndexSizeError, "Number of inputs is not in the allowed range."_s };
48
49 auto merger = adoptRef(*new ChannelMergerNode(context, options.numberOfInputs));
50
51 auto result = merger->handleAudioNodeOptions(options, { 1, ChannelCountMode::Explicit, ChannelInterpretation::Speakers });
52 if (result.hasException())
53 return result.releaseException();
54
55 return merger;
56}

58ChannelMergerNode::ChannelMergerNode(BaseAudioContext& context, unsigned numberOfInputs)
59 : AudioNode(context, NodeTypeChannelMerger)
60{
61 // Create the requested number of inputs.
62 for (unsigned i = 0; i < numberOfInputs; ++i)
63 addInput();
64
65 addOutput(numberOfInputs);
66
67 initialize();
68
69 Locker contextLocker { context.graphLock() };
70 disableOutputs();
71}

73void ChannelMergerNode::process(size_t framesToProcess)
74{
75 AudioNodeOutput* output = this->output(0);
76 ASSERT(output);
77 ASSERT_UNUSED(framesToProcess, framesToProcess == output->bus()->length());
78 ASSERT(numberOfInputs() == output->numberOfChannels());
79
80 // Merge all the channels from all the inputs into one output.
81 for (unsigned i = 0; i < numberOfInputs(); ++i) {
82 AudioNodeInput* input = this->input(i);
83 ASSERT(input->numberOfChannels() == 1u);
84 auto* outputChannel = output->bus()->channel(i);
85 if (input->isConnected()) {
86 // The mixing rules will be applied so multiple channels are down-
87 // mixed to mono (when the mixing rule is defined). Note that only
88 // the first channel will be taken for the undefined input channel
89 // layout.
90 //
91 // See:
92 // http://webaudio.github.io/web-audio-api/#channel-up-mixing-and-down-mixing
93 auto* inputChannel = input->bus()->channel(0);
94 outputChannel->copyFrom(inputChannel);
95 } else {
96 // If input is unconnected, fill zeros in the channel.
97 outputChannel->zero();
98 }
99 }
100}

102ExceptionOr<void> ChannelMergerNode::setChannelCount(unsigned channelCount)
103{
104 if (channelCount != 1)
105 return Exception { InvalidStateError, "Channel count cannot be changed from 1."_s };
106
107 return AudioNode::setChannelCount(channelCount);
108}
109
110ExceptionOr<void> ChannelMergerNode::setChannelCountMode(ChannelCountMode mode)
111{
112 if (mode != ChannelCountMode::Explicit)
113 return Exception { InvalidStateError, "Channel count mode cannot be changed from explicit."_s };
114
115 return AudioNode::setChannelCountMode(mode);
116}
