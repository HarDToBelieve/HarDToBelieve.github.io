47ConstantSourceNode::ConstantSourceNode(BaseAudioContext& context, float offset)
48 : AudioScheduledSourceNode(context, NodeTypeConstant)
49 , m_offset(AudioParam::create(context, "offset"_s, offset, -FLT_MAX, FLT_MAX, AutomationRate::ARate))
50 , m_sampleAccurateValues(AudioUtilities::renderQuantumSize)
51{
52 addOutput(1);
53 initialize();
54}
55
56ConstantSourceNode::~ConstantSourceNode()
57{
58 uninitialize();
59}

61void ConstantSourceNode::process(size_t framesToProcess)
62{
63 auto& outputBus = *output(0)->bus();
64
65 if (!isInitialized() || !outputBus.numberOfChannels()) {
66 outputBus.zero();
67 return;
68 }
69
70 size_t quantumFrameOffset = 0;
71 size_t nonSilentFramesToProcess = 0;
72 double startFrameOffset = 0;
73 updateSchedulingInfo(framesToProcess, outputBus, quantumFrameOffset, nonSilentFramesToProcess, startFrameOffset);
74
75 if (!nonSilentFramesToProcess) {
76 outputBus.zero();
77 return;
78 }
79
80 bool isSampleAccurate = m_offset->hasSampleAccurateValues();
81 if (isSampleAccurate && m_offset->automationRate() == AutomationRate::ARate) {
82 float* offsets = m_sampleAccurateValues.data();
83 m_offset->calculateSampleAccurateValues(offsets, framesToProcess);
84 if (nonSilentFramesToProcess > 0) {
85 memcpy(outputBus.channel(0)->mutableData() + quantumFrameOffset, offsets + quantumFrameOffset, nonSilentFramesToProcess * sizeof(*offsets));
86 outputBus.clearSilentFlag();
87 } else
88 outputBus.zero();
89 return;
90 }
91
92 float value = isSampleAccurate ? m_offset->finalValue() : m_offset->value();
93 if (!value)
94 outputBus.zero();
95 else {
96 float* dest = outputBus.channel(0)->mutableData();
97 std::fill_n(dest + quantumFrameOffset, nonSilentFramesToProcess, value);
98 outputBus.clearSilentFlag();
99 }
100}
101
102bool ConstantSourceNode::propagatesSilence() const
103{
104 return !isPlayingOrScheduled() || hasFinished();
105}
