26#pragma once
27
28#include "AudioDestinationNode.h"
29#include "AudioIOCallback.h"
30
31namespace WebCore {
32
33class AudioContext;
34class AudioDestination;
35
36class DefaultAudioDestinationNode final : public AudioDestinationNode, public AudioIOCallback {
37 WTF_MAKE_ISO_ALLOCATED(DefaultAudioDestinationNode);
38public:
39 explicit DefaultAudioDestinationNode(AudioContext&, std::optional<float> sampleRate = std::nullopt);
40 ~DefaultAudioDestinationNode();
41
42 AudioContext& context();
43 const AudioContext& context() const;
44
45 unsigned framesPerBuffer() const;
46
47 void startRendering(CompletionHandler<void(std::optional<Exception>&&)>&&) final;
48 void resume(CompletionHandler<void(std::optional<Exception>&&)>&&);
49 void suspend(CompletionHandler<void(std::optional<Exception>&&)>&&);
50 void close(CompletionHandler<void()>&&);
51
52 void setMuted(bool muted) { m_muted = muted; }
53 bool isPlayingAudio() const { return m_isEffectivelyPlayingAudio; }
54
55private:
56 void createDestination();
57 void clearDestination();
58 void recreateDestination();
59
60 // AudioIOCallback
61 // The audio hardware calls render() to get the next render quantum of audio into destinationBus.
62 // It will optionally give us local/live audio input in sourceBus (if it's not 0).
63 void render(AudioBus* sourceBus, AudioBus* destinationBus, size_t numberOfFrames, const AudioIOPosition& outputPosition) final;
64 void isPlayingDidChange() final;
65
66 void setIsSilent(bool);
67 void updateIsEffectivelyPlayingAudio();
68
69 Function<void(Function<void()>&&)> dispatchToRenderThreadFunction();
70
71 void initialize() final;
72 void uninitialize() final;
73 ExceptionOr<void> setChannelCount(unsigned) final;
74
75 bool requiresTailProcessing() const final { return false; }
76
77 void enableInput(const String& inputDeviceId) final;
78 void restartRendering() final;
79 unsigned maxChannelCount() const final;
80
81 RefPtr<AudioDestination> m_destination;
82 String m_inputDeviceId;
83 unsigned m_numberOfInputChannels { 0 };
84 bool m_wasDestinationStarted { false };
85 bool m_isEffectivelyPlayingAudio { false };
86 bool m_isSilent { true };
87 bool m_muted { false };
88};
89
90} // namespace WebCore
