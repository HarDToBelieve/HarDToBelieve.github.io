37WTF_MAKE_ISO_ALLOCATED_IMPL(DelayNode);
38
39constexpr double maximumAllowedDelayTime = 180;
40
41inline DelayNode::DelayNode(BaseAudioContext& context, double maxDelayTime)
42 : AudioBasicProcessorNode(context, NodeTypeDelay)
43{
44 m_processor = makeUnique<DelayProcessor>(context, context.sampleRate(), 1, maxDelayTime);
45
46 // Initialize so that AudioParams can be processed.
47 initialize();
48}
49
50ExceptionOr<Ref<DelayNode>> DelayNode::create(BaseAudioContext& context, const DelayOptions& options)
51{
52 if (options.maxDelayTime <= 0)
53 return Exception { NotSupportedError, "maxDelayTime should be a positive value"_s };
54
55 if (options.maxDelayTime >= maximumAllowedDelayTime)
56 return Exception { NotSupportedError, makeString("maxDelayTime should be less than ", maximumAllowedDelayTime) };
57
58 auto delayNode = adoptRef(*new DelayNode(context, options.maxDelayTime));
59
60 auto result = delayNode->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
61 if (result.hasException())
62 return result.releaseException();
63
64 delayNode->delayTime().setValue(options.delayTime);
65
66 return delayNode;
67}
68
69AudioParam& DelayNode::delayTime()
70{
71 return static_cast<DelayProcessor&>(*m_processor).delayTime();
72}
