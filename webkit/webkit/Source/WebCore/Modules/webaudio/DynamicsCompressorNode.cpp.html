46ExceptionOr<Ref<DynamicsCompressorNode>> DynamicsCompressorNode::create(BaseAudioContext& context, const DynamicsCompressorOptions& options)
47{
48 auto node = adoptRef(*new DynamicsCompressorNode(context, options));
49
50 auto result = node->handleAudioNodeOptions(options, { 2, ChannelCountMode::ClampedMax, ChannelInterpretation::Speakers });
51 if (result.hasException())
52 return result.releaseException();
53
54 return node;
55}

57DynamicsCompressorNode::DynamicsCompressorNode(BaseAudioContext& context, const DynamicsCompressorOptions& options)
58 : AudioNode(context, NodeTypeDynamicsCompressor)
59 , m_threshold(AudioParam::create(context, "threshold"_s, options.threshold, -100, 0, AutomationRate::KRate, AutomationRateMode::Fixed))
60 , m_knee(AudioParam::create(context, "knee"_s, options.knee, 0, 40, AutomationRate::KRate, AutomationRateMode::Fixed))
61 , m_ratio(AudioParam::create(context, "ratio"_s, options.ratio, 1, 20, AutomationRate::KRate, AutomationRateMode::Fixed))
62 , m_attack(AudioParam::create(context, "attack"_s, options.attack, 0, 1, AutomationRate::KRate, AutomationRateMode::Fixed))
63 , m_release(AudioParam::create(context, "release"_s, options.release, 0, 1, AutomationRate::KRate, AutomationRateMode::Fixed))
64{
65 addInput();
66 addOutput(defaultNumberOfOutputChannels);
67
68 initialize();
69}
70
71DynamicsCompressorNode::~DynamicsCompressorNode()
72{
73 uninitialize();
74}

76void DynamicsCompressorNode::process(size_t framesToProcess)
77{
78 AudioBus* outputBus = output(0)->bus();
79 ASSERT(outputBus);
80
81 float threshold = m_threshold->finalValue();
82 float knee = m_knee->finalValue();
83 float ratio = m_ratio->finalValue();
84 float attack = m_attack->finalValue();
85 float release = m_release->finalValue();
86
87 m_dynamicsCompressor->setParameterValue(DynamicsCompressor::ParamThreshold, threshold);
88 m_dynamicsCompressor->setParameterValue(DynamicsCompressor::ParamKnee, knee);
89 m_dynamicsCompressor->setParameterValue(DynamicsCompressor::ParamRatio, ratio);
90 m_dynamicsCompressor->setParameterValue(DynamicsCompressor::ParamAttack, attack);
91 m_dynamicsCompressor->setParameterValue(DynamicsCompressor::ParamRelease, release);
92
93 m_dynamicsCompressor->process(input(0)->bus(), outputBus, framesToProcess);
94
95 setReduction(m_dynamicsCompressor->parameterValue(DynamicsCompressor::ParamReduction));
96}
97
98void DynamicsCompressorNode::processOnlyAudioParams(size_t framesToProcess)
99{
100 float values[AudioUtilities::renderQuantumSize];
101 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
102
103 m_threshold->calculateSampleAccurateValues(values, framesToProcess);
104 m_knee->calculateSampleAccurateValues(values, framesToProcess);
105 m_ratio->calculateSampleAccurateValues(values, framesToProcess);
106 m_attack->calculateSampleAccurateValues(values, framesToProcess);
107 m_release->calculateSampleAccurateValues(values, framesToProcess);
108}

110void DynamicsCompressorNode::initialize()
111{
112 if (isInitialized())
113 return;
114
115 AudioNode::initialize();
116 m_dynamicsCompressor = makeUnique<DynamicsCompressor>(sampleRate(), defaultNumberOfOutputChannels);
117}
118
119void DynamicsCompressorNode::uninitialize()
120{
121 if (!isInitialized())
122 return;
123
124 m_dynamicsCompressor = nullptr;
125 AudioNode::uninitialize();
126}

144ExceptionOr<void> DynamicsCompressorNode::setChannelCount(unsigned count)
145{
146 if (count > 2)
147 return Exception { NotSupportedError, "DynamicsCompressorNode's channel count cannot be greater than 2"_s };
148 return AudioNode::setChannelCount(count);
149}
150
151ExceptionOr<void> DynamicsCompressorNode::setChannelCountMode(ChannelCountMode mode)
152{
153 if (mode == ChannelCountMode::Max)
154 return Exception { NotSupportedError, "DynamicsCompressorNode's channel count mode cannot be set to 'max'"_s };
155 return AudioNode::setChannelCountMode(mode);
156}
