33IIRDSPKernel::IIRDSPKernel(IIRProcessor& processor)
34 : AudioDSPKernel(&processor)
35 , m_iirFilter(processor.feedforward(), processor.feedback())
36 , m_tailTime(m_iirFilter.tailTime(processor.sampleRate(), processor.isFilterStable()))
37{
38}
39
40void IIRDSPKernel::getFrequencyResponse(unsigned length, const float* frequencyHz, float* magResponse, float* phaseResponse)
41{
42 ASSERT(frequencyHz);
43 ASSERT(magResponse);
44 ASSERT(phaseResponse);
45
46 Vector<float> frequency(length);
47 double nyquist = this->nyquist();
48
49 // Convert from frequency in Hz to normalized frequency (0 -> 1), with 1 equal to the Nyquist frequency.
50 for (unsigned k = 0; k < length; ++k)
51 frequency[k] = frequencyHz[k] / nyquist;
52
53 m_iirFilter.getFrequencyResponse(length, frequency.data(), magResponse, phaseResponse);
54}
55
56void IIRDSPKernel::process(const float* source, float* destination, size_t framesToProcess)
57{
58 ASSERT(source);
59 ASSERT(destination);
60
61 m_iirFilter.process(source, destination, framesToProcess);
62}
63
64void IIRDSPKernel::reset()
65{
66 m_iirFilter.reset();
67}

69bool IIRDSPKernel::requiresTailProcessing() const
70{
71 // Always return true even if the tail time and latency might both be zero.
72 return true;
73}
