40// Determine if filter is stable based on the feedback coefficients.
41// We compute the reflection coefficients for the filter. If, at any
42// point, the magnitude of the reflection coefficient is greater than
43// or equal to 1, the filter is declared unstable.
44//
45// Let A(z) be the feedback polynomial given by
46// A[n](z) = 1 + a[1]/z + a[2]/z^2 + ... + a[n]/z^n
47//
48// The first reflection coefficient k[n] = a[n]. Then, recursively compute
49//
50// A[n-1](z) = (A[n](z) - k[n]*A[n](1/z)/z^n)/(1-k[n]^2);
51//
52// stopping at A[1](z). If at any point |k[n]| >= 1, the filter is unstable.
53static bool isFilterStable(const Vector<double>& feedback)
54{
55 // Make a copy of the feedback coefficients
56 Vector<double> coefficients(feedback);
57 int order = coefficients.size() - 1;
58
59 // If necessary, normalize filter coefficients so that constant term is 1.
60 if (coefficients[0] != 1) {
61 for (int m = 1; m <= order; ++m)
62 coefficients[m] /= coefficients[0];
63 coefficients[0] = 1;
64 }
65
66 // Begin recursion, using a work array to hold intermediate results.
67 Vector<double> work(order + 1);
68 for (int n = order; n >= 1; --n) {
69 double k = coefficients[n];
70
71 if (std::fabs(k) >= 1)
72 return false;
73
74 // Note that A[n](1/z)/z^n is basically the coefficients of A[n]
75 // in reverse order.
76 double factor = 1 - k * k;
77 for (int m = 0; m <= n; ++m)
78 work[m] = (coefficients[m] - k * coefficients[n - m]) / factor;
79 coefficients.swap(work);
80 }
81
82 return true;
83}

85ExceptionOr<Ref<IIRFilterNode>> IIRFilterNode::create(ScriptExecutionContext& scriptExecutionContext, BaseAudioContext& context, IIRFilterOptions&& options)
86{
87 if (!options.feedforward.size() || options.feedforward.size() > IIRFilter::maxOrder)
88 return Exception { NotSupportedError, "feedforward array must have a length between 1 and 20"_s };
89
90 auto nonZeroValueIndex = options.feedforward.findMatching([](auto& value) { return !!value; });
91 if (nonZeroValueIndex == notFound)
92 return Exception { InvalidStateError, "feedforward array must contain a non-zero value"_s };
93
94 if (!options.feedback.size() || options.feedback.size() > IIRFilter::maxOrder)
95 return Exception { NotSupportedError, "feedback array must have a length between 1 and 20"_s };
96
97 if (!options.feedback[0])
98 return Exception { InvalidStateError, "first value of feedback array cannot be zero"_s };
99
100 bool isFilterStable = WebCore::isFilterStable(options.feedback);
101 if (!isFilterStable)
102 scriptExecutionContext.addConsoleMessage(MessageSource::JS, MessageLevel::Warning, "IIRFilter is unstable with provided feedback coefficients"_s);
103
104 auto node = adoptRef(*new IIRFilterNode(context, options.feedforward, options.feedback, isFilterStable));
105
106 auto result = node->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
107 if (result.hasException())
108 return result.releaseException();
109
110 return node;
111}

121ExceptionOr<void> IIRFilterNode::getFrequencyResponse(Float32Array& frequencyHz, Float32Array& magResponse, Float32Array& phaseResponse)
122{
123 auto expectedLength = frequencyHz.length();
124 if (magResponse.length() != expectedLength || phaseResponse.length() != expectedLength)
125 return Exception { InvalidAccessError, "Arrays must have the same length"_s };
126
127 // Nothing to do if the length is 0.
128 if (expectedLength > 0)
129 iirProcessor()->getFrequencyResponse(expectedLength, frequencyHz.data(), magResponse.data(), phaseResponse.data());
130
131 return { };
132}
