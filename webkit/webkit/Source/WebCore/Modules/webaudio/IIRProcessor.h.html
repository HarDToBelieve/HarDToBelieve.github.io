35class IIRProcessor final : public AudioDSPKernelProcessor {
36public:
37 IIRProcessor(float sampleRate, unsigned numberOfChannels, const Vector<double>& feedforward, const Vector<double>& feedback, bool isFilterStable);
38 ~IIRProcessor();
39
40 // AudioDSPKernelProcessor.
41 void process(const AudioBus* source, AudioBus* destination, size_t framesToProcess) final;
42 std::unique_ptr<AudioDSPKernel> createKernel() final;
43
44 // Get the magnitude and phase response of the filter at the given set of frequencies (in Hz). The phase response is in radians.
45 void getFrequencyResponse(unsigned length, const float* frequencyHz, float* magResponse, float* phaseResponse);
46
47 const Vector<double>& feedforward() const { return m_feedforward; }
48 const Vector<double>& feedback() const { return m_feedback; }
49 bool isFilterStable() const { return m_isFilterStable; }

51private:
52 Vector<double> m_feedforward;
53 Vector<double> m_feedback;
54 bool m_isFilterStable;
55
56 std::unique_ptr<IIRDSPKernel> m_responseKernel;
57};
