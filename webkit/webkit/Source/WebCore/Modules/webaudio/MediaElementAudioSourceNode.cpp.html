42// These are somewhat arbitrary limits, but we need to do some kind of sanity-checking.
43constexpr unsigned minSampleRate = 8000;
44constexpr unsigned maxSampleRate = 192000;

50ExceptionOr<Ref<MediaElementAudioSourceNode>> MediaElementAudioSourceNode::create(BaseAudioContext& context, MediaElementAudioSourceOptions&& options)
51{
52 RELEASE_ASSERT(options.mediaElement);
53
54 if (options.mediaElement->audioSourceNode())
55 return Exception { InvalidStateError, "Media element is already associated with an audio source node"_s };
56
57 auto node = adoptRef(*new MediaElementAudioSourceNode(context, *options.mediaElement));
58
59 options.mediaElement->setAudioSourceNode(node.ptr());
60
61 // context keeps reference until node is disconnected.
62 context.sourceNodeWillBeginPlayback(node);
63
64 return node;
65}

67MediaElementAudioSourceNode::MediaElementAudioSourceNode(BaseAudioContext& context, Ref<HTMLMediaElement>&& mediaElement)
68 : AudioNode(context, NodeTypeMediaElementAudioSource)
69 , m_mediaElement(WTFMove(mediaElement))
70{
71 // Default to stereo. This could change depending on what the media element .src is set to.
72 addOutput(2);
73
74 initialize();
75}

77MediaElementAudioSourceNode::~MediaElementAudioSourceNode()
78{
79 m_mediaElement->setAudioSourceNode(nullptr);
80 uninitialize();
81}

83void MediaElementAudioSourceNode::setFormat(size_t numberOfChannels, float sourceSampleRate)
84{
85 auto protectedThis = makeRef(*this);
86
87 // Synchronize with process().
88 Locker locker { m_processLock };
89
90 m_muted = wouldTaintOrigin();
91
92 if (numberOfChannels != m_sourceNumberOfChannels || sourceSampleRate != m_sourceSampleRate) {
93 if (!numberOfChannels || numberOfChannels > AudioContext::maxNumberOfChannels || sourceSampleRate < minSampleRate || sourceSampleRate > maxSampleRate) {
94 // process() will generate silence for these uninitialized values.
95 LOG(Media, "MediaElementAudioSourceNode::setFormat(%u, %f) - unhandled format change", static_cast<unsigned>(numberOfChannels), sourceSampleRate);
96 m_sourceNumberOfChannels = 0;
97 m_sourceSampleRate = 0;
98 return;
99 }
100
101 m_sourceNumberOfChannels = numberOfChannels;
102 m_sourceSampleRate = sourceSampleRate;
103
104 if (sourceSampleRate != sampleRate()) {
105 double scaleFactor = sourceSampleRate / sampleRate();
106 m_multiChannelResampler = makeUnique<MultiChannelResampler>(scaleFactor, numberOfChannels, AudioUtilities::renderQuantumSize, std::bind(&MediaElementAudioSourceNode::provideInput, this, std::placeholders::_1, std::placeholders::_2));
107 } else {
108 // Bypass resampling.
109 m_multiChannelResampler = nullptr;
110 }
111
112 {
113 // The context must be locked when changing the number of output channels.
114 Locker contextLocker { context().graphLock() };
115
116 // Do any necesssary re-configuration to the output's number of channels.
117 output(0)->setNumberOfChannels(numberOfChannels);
118 }
119 }
120}

122void MediaElementAudioSourceNode::provideInput(AudioBus* bus, size_t framesToProcess)
123{
124 ASSERT(bus);
125 if (auto* provider = mediaElement().audioSourceProvider())
126 provider->provideInput(bus, framesToProcess);
127 else {
128 // Either this port doesn't yet support HTMLMediaElement audio stream access,
129 // or the stream is not yet available.
130 bus->zero();
131 }
132}

134bool MediaElementAudioSourceNode::wouldTaintOrigin()
135{
136 // If the resource is redirected to another origin, treat it as tainted if the crossorigin attribute
137 // is not set. This is done for consistency with Blink.
138 if (!m_mediaElement->hasSingleSecurityOrigin() && m_mediaElement->crossOrigin().isNull())
139 return true;
140
141 if (m_mediaElement->didPassCORSAccessCheck())
142 return false;
143
144 if (auto* origin = context().origin())
145 return m_mediaElement->wouldTaintOrigin(*origin);
146
147 return true;
148}
149
150void MediaElementAudioSourceNode::process(size_t numberOfFrames)
151{
152 AudioBus* outputBus = output(0)->bus();
153
154 // Use tryLock() to avoid contention in the real-time audio thread.
155 // If we fail to acquire the lock then the HTMLMediaElement must be in the middle of
156 // reconfiguring its playback engine, so we output silence in this case.
157 if (!m_processLock.tryLock()) {
158 // We failed to acquire the lock.
159 outputBus->zero();
160 return;
161 }
162
163 Locker locker { AdoptLock, m_processLock };
164
165 if (m_muted || !m_sourceNumberOfChannels || !m_sourceSampleRate || m_sourceNumberOfChannels != outputBus->numberOfChannels()) {
166 outputBus->zero();
167 return;
168 }
169
170 if (m_multiChannelResampler) {
171 ASSERT(m_sourceSampleRate != sampleRate());
172 m_multiChannelResampler->process(outputBus, numberOfFrames);
173 } else {
174 // Bypass the resampler completely if the source is at the context's sample-rate.
175 ASSERT(m_sourceSampleRate == sampleRate());
176 provideInput(outputBus, numberOfFrames);
177 }
178}
