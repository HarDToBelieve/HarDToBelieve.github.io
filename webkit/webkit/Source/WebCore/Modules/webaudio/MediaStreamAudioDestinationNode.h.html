35
36class BaseAudioContext;
37class MediaStreamAudioSource;
38
39class MediaStreamAudioDestinationNode final : public AudioBasicInspectorNode {
40 WTF_MAKE_ISO_ALLOCATED(MediaStreamAudioDestinationNode);
41public:
42 static ExceptionOr<Ref<MediaStreamAudioDestinationNode>> create(BaseAudioContext&, const AudioNodeOptions& = { });
43
44 virtual ~MediaStreamAudioDestinationNode();
45
46 MediaStream& stream() { return m_stream.get(); }
47
48 // AudioNode.
49 void process(size_t framesToProcess) final;
50
51private:
52 explicit MediaStreamAudioDestinationNode(BaseAudioContext&);
53
54 double tailTime() const final { return 0; }

56 bool requiresTailProcessing() const final { return false; }
57
58 // As an audio source, we will never propagate silence.
59 bool propagatesSilence() const final { return false; }
60
61 Ref<MediaStreamAudioSource> m_source;
62 Ref<MediaStream> m_stream;
63};
