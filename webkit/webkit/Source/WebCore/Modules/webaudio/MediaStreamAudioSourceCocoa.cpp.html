42static inline CAAudioStreamDescription streamDescription(size_t sampleRate, size_t channelCount)
43{
44 bool isFloat = true;
45 bool isBigEndian = false;
46 bool isNonInterleaved = true;
47 static const size_t sampleSize = 8 * sizeof(float);
48
49 AudioStreamBasicDescription streamFormat;
50 FillOutASBDForLPCM(streamFormat, sampleRate, channelCount, sampleSize, sampleSize, isFloat, isBigEndian, isNonInterleaved);
51 return streamFormat;
52}
53
54static inline void copyChannelData(AudioChannel& channel, AudioBuffer& buffer, size_t numberOfFrames, bool isMuted)
55{
56 buffer.mDataByteSize = numberOfFrames * sizeof(float);
57 buffer.mNumberChannels = 1;
58 if (isMuted) {
59 memset(buffer.mData, 0, buffer.mDataByteSize);
60 return;
61 }
62 memcpy(buffer.mData, channel.data(), buffer.mDataByteSize);
63}
64
65void MediaStreamAudioSource::consumeAudio(AudioBus& bus, size_t numberOfFrames)
66{
67 if (bus.numberOfChannels() != 1 && bus.numberOfChannels() != 2) {
68 RELEASE_LOG_ERROR(Media, "MediaStreamAudioSource::consumeAudio(%p) trying to consume bus with %u channels", this, bus.numberOfChannels());
69 return;
70 }
71
72 CMTime startTime = PAL::CMTimeMake(m_numberOfFrames, m_currentSettings.sampleRate());
73 auto mediaTime = PAL::toMediaTime(startTime);
74 m_numberOfFrames += numberOfFrames;
75
76 auto* audioBuffer = m_audioBuffer ? &downcast<WebAudioBufferList>(*m_audioBuffer) : nullptr;
77
78 auto description = streamDescription(m_currentSettings.sampleRate(), bus.numberOfChannels());
79 if (!audioBuffer || audioBuffer->channelCount() != bus.numberOfChannels()) {
80 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
81 // explicitly allow the following allocation(s).
82 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
83 m_audioBuffer = makeUnique<WebAudioBufferList>(description, WTF::safeCast<uint32_t>(numberOfFrames));
84 audioBuffer = &downcast<WebAudioBufferList>(*m_audioBuffer);
85 } else
86 audioBuffer->setSampleCount(numberOfFrames);
87
88 for (size_t cptr = 0; cptr < bus.numberOfChannels(); ++cptr)
89 copyChannelData(*bus.channel(cptr), *audioBuffer->buffer(cptr), numberOfFrames, muted());
90
91 audioSamplesAvailable(mediaTime, *m_audioBuffer, description, numberOfFrames);
92}
