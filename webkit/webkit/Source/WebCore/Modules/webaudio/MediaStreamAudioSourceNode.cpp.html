45ExceptionOr<Ref<MediaStreamAudioSourceNode>> MediaStreamAudioSourceNode::create(BaseAudioContext& context, MediaStreamAudioSourceOptions&& options)
46{
47 RELEASE_ASSERT(options.mediaStream);
48
49 auto audioTracks = options.mediaStream->getAudioTracks();
50 if (audioTracks.isEmpty())
51 return Exception { InvalidStateError, "Media stream has no audio tracks"_s };
52
53 RefPtr<WebAudioSourceProvider> provider;
54 for (auto& track : audioTracks) {
55 provider = track->createAudioSourceProvider();
56 if (provider)
57 break;
58 }
59 if (!provider)
60 return Exception { InvalidStateError, "Could not find an audio track with an audio source provider"_s };
61
62 auto node = adoptRef(*new MediaStreamAudioSourceNode(context, *options.mediaStream, provider.releaseNonNull()));
63 node->setFormat(2, context.sampleRate());
64
65 // Context keeps reference until node is disconnected.
66 context.sourceNodeWillBeginPlayback(node);
67
68 return node;
69}

71MediaStreamAudioSourceNode::MediaStreamAudioSourceNode(BaseAudioContext& context, MediaStream& mediaStream, Ref<WebAudioSourceProvider>&& provider)
72 : AudioNode(context, NodeTypeMediaStreamAudioSource)
73 , m_mediaStream(mediaStream)
74 , m_provider(provider)
75{
76 m_provider->setClient(this);
77
78 // Default to stereo. This could change depending on the format of the MediaStream's audio track.
79 addOutput(2);
80
81 initialize();
82}

84MediaStreamAudioSourceNode::~MediaStreamAudioSourceNode()

86 m_provider->setClient(nullptr);
87 uninitialize();

90void MediaStreamAudioSourceNode::setFormat(size_t numberOfChannels, float sourceSampleRate)
91{
92 // Synchronize with process().
93 Locker locker { m_processLock };
94
95 if (numberOfChannels == m_sourceNumberOfChannels && sourceSampleRate == m_sourceSampleRate)
96 return;
97
98 // The sample-rate must be equal to the context's sample-rate.
99 if (!numberOfChannels || numberOfChannels > AudioContext::maxNumberOfChannels) {
100 // process() will generate silence for these uninitialized values.
101 LOG(Media, "MediaStreamAudioSourceNode::setFormat(%u, %f) - unhandled format change", static_cast<unsigned>(numberOfChannels), sourceSampleRate);
102 m_sourceNumberOfChannels = 0;
103 return;
104 }
105
106 m_sourceNumberOfChannels = numberOfChannels;
107 m_sourceSampleRate = sourceSampleRate;
108
109 float sampleRate = this->sampleRate();
110 if (sourceSampleRate == sampleRate)
111 m_multiChannelResampler = nullptr;
112 else {
113 double scaleFactor = sourceSampleRate / sampleRate;
114 m_multiChannelResampler = makeUnique<MultiChannelResampler>(scaleFactor, numberOfChannels, AudioUtilities::renderQuantumSize, std::bind(&MediaStreamAudioSourceNode::provideInput, this, std::placeholders::_1, std::placeholders::_2));
115 }
116
117 m_sourceNumberOfChannels = numberOfChannels;
118
119 {
120 // The context must be locked when changing the number of output channels.
121 Locker contextLocker { context().graphLock() };
122
123 // Do any necesssary re-configuration to the output's number of channels.
124 output(0)->setNumberOfChannels(numberOfChannels);
125 }
126}

133void MediaStreamAudioSourceNode::process(size_t numberOfFrames)
134{
135 AudioBus* outputBus = output(0)->bus();
136
137 // Use tryLock() to avoid contention in the real-time audio thread.
138 // If we fail to acquire the lock then the MediaStream must be in the middle of
139 // a format change, so we output silence in this case.
140 if (!m_processLock.tryLock()) {
141 // We failed to acquire the lock.
142 outputBus->zero();
143 return;
144 }
145 Locker locker { AdoptLock, m_processLock };
146
147 if (!m_sourceNumberOfChannels || !m_sourceSampleRate || m_sourceNumberOfChannels != outputBus->numberOfChannels()) {
148 outputBus->zero();
149 return;
150 }
151
152 if (numberOfFrames > outputBus->length())
153 numberOfFrames = outputBus->length();
154
155 if (m_multiChannelResampler) {
156 ASSERT(m_sourceSampleRate != sampleRate());
157 m_multiChannelResampler->process(outputBus, numberOfFrames);
158 } else {
159 // Bypass the resampler completely if the source is at the context's sample-rate.
160 ASSERT(m_sourceSampleRate == sampleRate());
161 provideInput(outputBus, numberOfFrames);
162 }
163}
