26
27#if ENABLE(WEB_AUDIO) && ENABLE(MEDIA_STREAM)
28
29#include "AudioNode.h"
30#include "AudioSourceProviderClient.h"
31#include "MediaStream.h"
32#include "MultiChannelResampler.h"
33#include <wtf/Lock.h>
34
35namespace WebCore {
36
37class AudioContext;
38struct MediaStreamAudioSourceOptions;
39class MultiChannelResampler;
40class WebAudioSourceProvider;
41
42class MediaStreamAudioSourceNode final : public AudioNode, public AudioSourceProviderClient {
43 WTF_MAKE_ISO_ALLOCATED(MediaStreamAudioSourceNode);
44public:
45 static ExceptionOr<Ref<MediaStreamAudioSourceNode>> create(BaseAudioContext&, MediaStreamAudioSourceOptions&&);
46
47 ~MediaStreamAudioSourceNode();
48
49 MediaStream& mediaStream() { return m_mediaStream; }
50
51private:
52 MediaStreamAudioSourceNode(BaseAudioContext&, MediaStream&, Ref<WebAudioSourceProvider>&&);
53
54 // AudioNode
55 void process(size_t framesToProcess) final;
56 // AudioSourceProviderClient
57 void setFormat(size_t numberOfChannels, float sampleRate) final;
58
59 void provideInput(AudioBus*, size_t framesToProcess);
60
61 double tailTime() const override { return 0; }
62 double latencyTime() const override { return 0; }
63 bool requiresTailProcessing() const final { return false; }
64
65 // As an audio source, we will never propagate silence.
66 bool propagatesSilence() const override { return false; }
67
68 Ref<MediaStream> m_mediaStream;
69 Ref<WebAudioSourceProvider> m_provider;
70 std::unique_ptr<MultiChannelResampler> m_multiChannelResampler WTF_GUARDED_BY_LOCK(m_processLock);
71
72 Lock m_processLock;
73
74 unsigned m_sourceNumberOfChannels WTF_GUARDED_BY_LOCK(m_processLock) { 0 };
75 double m_sourceSampleRate WTF_GUARDED_BY_LOCK(m_processLock) { 0 };
76};
77
78} // namespace WebCore

80#endif // ENABLE(WEB_AUDIO) && ENABLE(MEDIA_STREAM)
