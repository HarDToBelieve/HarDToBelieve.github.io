45OfflineAudioContext::OfflineAudioContext(Document& document, const OfflineAudioContextOptions& options)
46 : BaseAudioContext(document)
47 , m_destinationNode(makeUniqueRef<OfflineAudioDestinationNode>(*this, options.numberOfChannels, options.sampleRate, AudioBuffer::create(options.numberOfChannels, options.length, options.sampleRate)))
48 , m_length(options.length)
49{
50 if (!renderTarget())
51 document.addConsoleMessage(MessageSource::JS, MessageLevel::Warning, makeString("Failed to construct internal AudioBuffer with ", options.numberOfChannels, " channel(s), a sample rate of ", options.sampleRate, " and a length of ", options.length, "."));
52}

54ExceptionOr<Ref<OfflineAudioContext>> OfflineAudioContext::create(ScriptExecutionContext& context, const OfflineAudioContextOptions& options)
55{
56 if (!is<Document>(context))
57 return Exception { NotSupportedError, "OfflineAudioContext is only supported in Document contexts"_s };
58 if (!options.numberOfChannels || options.numberOfChannels > maxNumberOfChannels)
59 return Exception { SyntaxError, "Number of channels is not in range"_s };
60 if (!options.length)
61 return Exception { SyntaxError, "length cannot be 0"_s };
62 if (!isSupportedSampleRate(options.sampleRate))
63 return Exception { SyntaxError, "sampleRate is not in range"_s };
64
65 auto audioContext = adoptRef(*new OfflineAudioContext(downcast<Document>(context), options));
66 audioContext->suspendIfNeeded();
67 return audioContext;
68}

75void OfflineAudioContext::uninitialize()
76{
77 if (!isInitialized())
78 return;
79
80 BaseAudioContext::uninitialize();
81
82 if (auto promise = std::exchange(m_pendingRenderingPromise, nullptr))
83 promise->reject(Exception { InvalidStateError, "Context is going away"_s });
84}

91void OfflineAudioContext::startRendering(Ref<DeferredPromise>&& promise)
92{
93 if (isStopped()) {
94 promise->reject(Exception { InvalidStateError, "Context is stopped"_s });
95 return;
96 }
97
98 if (m_didStartRendering) {
99 promise->reject(Exception { InvalidStateError, "Rendering was already started"_s });
100 return;
101 }
102
103 if (!renderTarget()) {
104 promise->reject(Exception { NotSupportedError, "Failed to create audio buffer"_s });
105 return;
106 }
107
108 lazyInitialize();
109
110 destination().startRendering([this, promise = WTFMove(promise), pendingActivity = makePendingActivity(*this)](std::optional<Exception>&& exception) mutable {
111 if (exception) {
112 promise->reject(WTFMove(*exception));
113 return;
114 }
115
116 m_pendingRenderingPromise = WTFMove(promise);
117 m_didStartRendering = true;
118 setState(State::Running);
119 });
120}

122void OfflineAudioContext::suspendRendering(double suspendTime, Ref<DeferredPromise>&& promise)
123{
124 if (isStopped()) {
125 promise->reject(Exception { InvalidStateError, "Context is stopped"_s });
126 return;
127 }
128
129 if (suspendTime < 0) {
130 promise->reject(Exception { InvalidStateError, "suspendTime cannot be negative"_s });
131 return;
132 }
133
134 double totalRenderDuration = length() / sampleRate();
135 if (totalRenderDuration <= suspendTime) {
136 promise->reject(Exception { InvalidStateError, "suspendTime cannot be greater than total rendering duration"_s });
137 return;
138 }
139
140 size_t frame = AudioUtilities::timeToSampleFrame(suspendTime, sampleRate());
141 frame = AudioUtilities::renderQuantumSize * ((frame + AudioUtilities::renderQuantumSize - 1) / AudioUtilities::renderQuantumSize);
142 if (frame < currentSampleFrame()) {
143 promise->reject(Exception { InvalidStateError, "Suspension frame is earlier than current frame"_s });
144 return;
145 }
146
147 Locker locker { graphLock() };
148 auto addResult = m_suspendRequests.add(frame, promise.ptr());
149 if (!addResult.isNewEntry) {
150 promise->reject(Exception { InvalidStateError, "There is already a pending suspend request at this frame"_s });
151 return;
152 }
153}

155void OfflineAudioContext::resumeRendering(Ref<DeferredPromise>&& promise)
156{
157 if (!m_didStartRendering) {
158 promise->reject(Exception { InvalidStateError, "Cannot resume an offline audio context that has not started"_s });
159 return;
160 }
161 if (isClosed()) {
162 promise->reject(Exception { InvalidStateError, "Cannot resume an offline audio context that is closed"_s });
163 return;
164 }
165 if (state() == AudioContextState::Running) {
166 promise->resolve();
167 return;
168 }
169 ASSERT(state() == AudioContextState::Suspended);
170
171 destination().startRendering([this, promise = WTFMove(promise), pendingActivity = makePendingActivity(*this)](std::optional<Exception>&& exception) mutable {
172 if (exception) {
173 promise->reject(WTFMove(*exception));
174 return;
175 }
176
177 setState(State::Running);
178 promise->resolve();
179 });
180}

182bool OfflineAudioContext::shouldSuspend()
183{
184 ASSERT(!isMainThread());
185 // Note that we are not using a tryLock() here. We usually avoid blocking the AudioThread
186 // on lock() but we don't have a choice here since the suspension need to be exact.
187 // Also, this not a real-time AudioContext so blocking the AudioThread is not as harmful.
188 Locker locker { graphLock() };
189 return m_suspendRequests.contains(currentSampleFrame());
190}
191
192void OfflineAudioContext::didSuspendRendering(size_t frame)
193{
194 setState(State::Suspended);
195
196 RefPtr<DeferredPromise> promise;
197 {
198 Locker locker { graphLock() };
199 promise = m_suspendRequests.take(frame);
200 }
201 ASSERT(promise);
202 if (promise)
203 promise->resolve();
204}

206void OfflineAudioContext::finishedRendering(bool didRendering)
207{
208 ASSERT(isMainThread());
209 ALWAYS_LOG(LOGIDENTIFIER);
210
211 auto uninitializeOnExit = WTF::makeScopeExit([this] {
212 uninitialize();
213 clear();
214 });
215
216 // Make sure our JSwrapper stays alive long enough to resolve the promise and queue the completion event.
217 // Otherwise, setting the state to Closed may cause our JS wrapper to get collected early.
218 auto protectedJSWrapper = makePendingActivity(*this);
219 setState(State::Closed);
220
221 // Avoid firing the event if the document has already gone away.
222 if (isStopped())
223 return;
224
225 RefPtr<AudioBuffer> renderedBuffer = renderTarget();
226 ASSERT(renderedBuffer);
227
228 if (didRendering) {
229 queueTaskToDispatchEvent(*this, TaskSource::MediaElement, OfflineAudioCompletionEvent::create(*renderedBuffer));
230 settleRenderingPromise(renderedBuffer.releaseNonNull());
231 } else
232 settleRenderingPromise(Exception { InvalidStateError, "Offline rendering failed"_s });
233}
234
235void OfflineAudioContext::settleRenderingPromise(ExceptionOr<Ref<AudioBuffer>>&& result)
236{
237 auto promise = std::exchange(m_pendingRenderingPromise, nullptr);
238 if (!promise)
239 return;
240
241 if (result.hasException()) {
242 promise->reject(result.releaseException());
243 return;
244 }
245 promise->resolve<IDLInterface<AudioBuffer>>(result.releaseReturnValue());
246}

248bool OfflineAudioContext::virtualHasPendingActivity() const
249{
250 return state() == State::Running;
251}
