47// PannerNode is an AudioNode with one input and one output.
48// It positions a sound in 3D space, with the exact effect dependent on the panning model.
49// It has a position and an orientation in 3D space which is relative to the position and orientation of the context's AudioListener.
50// A distance effect will attenuate the gain as the position moves away from the listener.
51// A cone effect will attenuate the gain as the orientation moves away from the listener.
52// All of these effects follow the OpenAL specification very closely.
53
54class PannerNode final : public AudioNode {
55 WTF_MAKE_ISO_ALLOCATED(PannerNode);
56public:
57 static ExceptionOr<Ref<PannerNode>> create(BaseAudioContext&, const PannerOptions& = { });
58
59 virtual ~PannerNode();
60
61 // AudioNode
62 void process(size_t framesToProcess) override;
63 void processOnlyAudioParams(size_t framesToProcess) final;
64
65 // Listener
66 AudioListener& listener();
67
68 // Panning model
69 PanningModelType panningModelForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_panningModel; }
70 void setPanningModelForBindings(PanningModelType);
71
72 // Position
73 ExceptionOr<void> setPosition(float x, float y, float z);
74 AudioParam& positionX() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_positionX.get(); }
75 AudioParam& positionY() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_positionY.get(); }
76 AudioParam& positionZ() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_positionZ.get(); }
77
78 // Orientation
79 ExceptionOr<void> setOrientation(float x, float y, float z);
80 AudioParam& orientationX() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_orientationX.get(); }
81 AudioParam& orientationY() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_orientationY.get(); }
82 AudioParam& orientationZ() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_orientationZ.get(); }
83
84 // Distance parameters
85 DistanceModelType distanceModelForBindings() const;
86 void setDistanceModelForBindings(DistanceModelType);
87
88 double refDistanceForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_distanceEffect.refDistance(); }
89 ExceptionOr<void> setRefDistanceForBindings(double);
90
91 double maxDistanceForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_distanceEffect.maxDistance(); }
92 ExceptionOr<void> setMaxDistanceForBindings(double);
93
94 double rolloffFactorForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_distanceEffect.rolloffFactor(); }
95 ExceptionOr<void> setRolloffFactorForBindings(double);
96
97 // Sound cones - angles in degrees
98 double coneInnerAngleForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_coneEffect.innerAngle(); }
99 void setConeInnerAngleForBindings(double);
100
101 double coneOuterAngleForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_coneEffect.outerAngle(); }
102 void setConeOuterAngleForBindings(double);
103
104 double coneOuterGainForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_coneEffect.outerGain(); }
105 ExceptionOr<void> setConeOuterGainForBindings(double);
106
107 ExceptionOr<void> setChannelCount(unsigned) final;
108 ExceptionOr<void> setChannelCountMode(ChannelCountMode) final;
109
110 double tailTime() const final;
111 double latencyTime() const final;
112
113private:
114 PannerNode(BaseAudioContext&, const PannerOptions&);
115
116 void calculateAzimuthElevation(double* outAzimuth, double* outElevation, const FloatPoint3D& position, const FloatPoint3D& listenerPosition, const FloatPoint3D& listenerForward, const FloatPoint3D& listenerUp) WTF_REQUIRES_LOCK(m_processLock);
117 float calculateDistanceConeGain(const FloatPoint3D& position, const FloatPoint3D& orientation, const FloatPoint3D& listenerPosition) WTF_REQUIRES_LOCK(m_processLock);
118
119 // Returns the combined distance and cone gain attenuation.
120 float distanceConeGain() WTF_REQUIRES_LOCK(m_processLock);
121
122 bool requiresTailProcessing() const final;
123
124 void azimuthElevation(double* outAzimuth, double* outElevation) WTF_REQUIRES_LOCK(m_processLock);
125 void processSampleAccurateValues(AudioBus* destination, const AudioBus* source, size_t framesToProcess) WTF_REQUIRES_LOCK(m_processLock);
126 bool hasSampleAccurateValues() const WTF_REQUIRES_LOCK(m_processLock);
127 bool shouldUseARate() const WTF_REQUIRES_LOCK(m_processLock);
128
129 FloatPoint3D position() const WTF_REQUIRES_LOCK(m_processLock);
130 FloatPoint3D orientation() const WTF_REQUIRES_LOCK(m_processLock);
131
132 Ref<HRTFDatabaseLoader> m_hrtfDatabaseLoader;
133 PanningModelType m_panningModel WTF_GUARDED_BY_LOCK(m_processLock);
134 std::unique_ptr<Panner> m_panner WTF_GUARDED_BY_LOCK(m_processLock);
135
136 // Gain
137 DistanceEffect m_distanceEffect WTF_GUARDED_BY_LOCK(m_processLock);
138 ConeEffect m_coneEffect WTF_GUARDED_BY_LOCK(m_processLock);
139
140 Ref<AudioParam> m_positionX WTF_GUARDED_BY_LOCK(m_processLock);
141 Ref<AudioParam> m_positionY WTF_GUARDED_BY_LOCK(m_processLock);
142 Ref<AudioParam> m_positionZ WTF_GUARDED_BY_LOCK(m_processLock);
143
144 Ref<AudioParam> m_orientationX WTF_GUARDED_BY_LOCK(m_processLock);
145 Ref<AudioParam> m_orientationY WTF_GUARDED_BY_LOCK(m_processLock);
146 Ref<AudioParam> m_orientationZ WTF_GUARDED_BY_LOCK(m_processLock);
147
148 // Synchronize process() with setting of the panning model, source's location
149 // information, listener, distance parameters and sound cones.
150 mutable Lock m_processLock;
151};
