29#pragma once
30
31#include "AudioArray.h"
32#include "ExceptionOr.h"
33#include "PeriodicWaveOptions.h"
34#include <JavaScriptCore/Float32Array.h>
35#include <memory>
36#include <wtf/RefCounted.h>
37#include <wtf/Vector.h>
38
39namespace WebCore {
40
41class BaseAudioContext;
42
43enum class ShouldDisableNormalization : bool { No, Yes };
44
45class PeriodicWave final : public RefCounted<PeriodicWave> {
46public:
47 static Ref<PeriodicWave> createSine(float sampleRate);
48 static Ref<PeriodicWave> createSquare(float sampleRate);
49 static Ref<PeriodicWave> createSawtooth(float sampleRate);
50 static Ref<PeriodicWave> createTriangle(float sampleRate);
51
52 // Creates an arbitrary wave given the frequency components (Fourier coefficients).
53 // FIXME: Remove once old constructor is phased out
54 static Ref<PeriodicWave> create(float sampleRate, Float32Array& real, Float32Array& imag);
55
56 static ExceptionOr<Ref<PeriodicWave>> create(BaseAudioContext&, PeriodicWaveOptions&& = { });
57
58 // Returns pointers to the lower and higher wave data for the pitch range containing
59 // the given fundamental frequency. These two tables are in adjacent "pitch" ranges
60 // where the higher table will have the maximum number of partials which won't alias when played back
61 // at this fundamental frequency. The lower wave is the next range containing fewer partials than the higher wave.
62 // Interpolation between these two tables can be made according to tableInterpolationFactor.
63 // Where values from 0 -> 1 interpolate between lower -> higher.
64 void waveDataForFundamentalFrequency(float, float* &lowerWaveData, float* &higherWaveData, float& tableInterpolationFactor);
65
66 // Returns the scalar multiplier to the oscillator frequency to calculate wave table phase increment.
67 float rateScale() const { return m_rateScale; }
68
69 unsigned periodicWaveSize() const;
70 float sampleRate() const { return m_sampleRate; }
71
72private:
73 enum class Type {
74 Sine,
75 Square,
76 Sawtooth,
77 Triangle,
78 };
79
80 explicit PeriodicWave(float sampleRate);
81
82 void generateBasicWaveform(Type);
83
84 float m_sampleRate;
85 unsigned m_numberOfRanges;
86
87 // The lowest frequency (in Hertz) where playback will include all of the partials.
88 // Playing back lower than this frequency will gradually lose more high-frequency information.
89 // This frequency is quite low (~10Hz @ 44.1KHz)
90 float m_lowestFundamentalFrequency;
91
92 float m_rateScale;
93
94 unsigned numberOfRanges() const { return m_numberOfRanges; }
95
96 // Maximum possible number of partials (before culling).
97 unsigned maxNumberOfPartials() const;
98
99 unsigned numberOfPartialsForRange(unsigned rangeIndex) const;
100
101 // Creates tables based on numberOfComponents Fourier coefficients.
102 void createBandLimitedTables(const float* real, const float* imag, unsigned numberOfComponents, ShouldDisableNormalization = ShouldDisableNormalization::No);
103 Vector<std::unique_ptr<AudioFloatArray>> m_bandLimitedTables;
104};
105
106} // namespace WebCore
