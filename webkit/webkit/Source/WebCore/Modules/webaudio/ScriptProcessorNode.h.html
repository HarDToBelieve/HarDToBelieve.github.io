38class AudioBuffer;
39class AudioContext;
40class AudioProcessingEvent;

42// ScriptProcessorNode is an AudioNode which allows for arbitrary synthesis or processing directly using JavaScript.
43// The API allows for a variable number of inputs and outputs, although it must have at least one input or output.
44// This basic implementation supports no more than one input and output.
45// The "onaudioprocess" attribute is an event listener which will get called periodically with an AudioProcessingEvent which has
46// AudioBuffers for each input and output.
47
48class ScriptProcessorNode final : public AudioNode, public ActiveDOMObject {
49 WTF_MAKE_ISO_ALLOCATED(ScriptProcessorNode);
50public:
51 // bufferSize must be one of the following values: 256, 512, 1024, 2048, 4096, 8192, 16384.
52 // This value controls how frequently the onaudioprocess event handler is called and how many sample-frames need to be processed each call.
53 // Lower numbers for bufferSize will result in a lower (better) latency. Higher numbers will be necessary to avoid audio breakup and glitches.
54 // The value chosen must carefully balance between latency and audio quality.
55 static Ref<ScriptProcessorNode> create(BaseAudioContext&, size_t bufferSize, unsigned numberOfInputChannels, unsigned numberOfOutputChannels);
56
57 virtual ~ScriptProcessorNode();
58
59 const char* activeDOMObjectName() const override { return "ScriptProcessorNode"; }
60
61 // AudioNode
62 void process(size_t framesToProcess) override;
63 void initialize() override;
64 void uninitialize() override;
65
66 size_t bufferSize() const { return m_bufferSize; }
67
68 ExceptionOr<void> setChannelCount(unsigned) final;
69 ExceptionOr<void> setChannelCountMode(ChannelCountMode) final;
70
71private:
72 double tailTime() const override;
73 double latencyTime() const override;
74 bool requiresTailProcessing() const final;
75
76 ScriptProcessorNode(BaseAudioContext&, size_t bufferSize, unsigned numberOfInputChannels, unsigned numberOfOutputChannels);
77
78 bool virtualHasPendingActivity() const final;
79 void eventListenersDidChange() final;
80 void fireProcessEvent(unsigned bufferIndex);
81
82 RefPtr<AudioBuffer> createInputBufferForJS(AudioBuffer*) const;
83 RefPtr<AudioBuffer> createOutputBufferForJS(AudioBuffer&) const;
84
85 // Double buffering.
86 static constexpr unsigned bufferCount = 2;
87 unsigned bufferIndex() const { return m_bufferIndex; }
88 void swapBuffers() { m_bufferIndex = (m_bufferIndex + 1) % bufferCount; }
89
90 unsigned m_bufferIndex { 0 };
91 std::array<Lock, bufferCount> m_bufferLocks;
92 std::array<RefPtr<AudioBuffer>, bufferCount> m_inputBuffers;
93 std::array<RefPtr<AudioBuffer>, bufferCount> m_outputBuffers;
94 mutable RefPtr<AudioBuffer> m_cachedInputBufferForJS;
95 mutable RefPtr<AudioBuffer> m_cachedOutputBufferForJS;
96
97 size_t m_bufferSize;
98 unsigned m_bufferReadWriteIndex { 0 };
99
100 unsigned m_numberOfInputChannels;
101 unsigned m_numberOfOutputChannels;
102
103 RefPtr<AudioBus> m_internalInputBus;
104 bool m_hasAudioProcessEventListener { false };
105};
