42ExceptionOr<Ref<StereoPannerNode>> StereoPannerNode::create(BaseAudioContext& context, const StereoPannerOptions& options)
43{
44 auto stereo = adoptRef(*new StereoPannerNode(context, options.pan));
45
46 auto result = stereo->handleAudioNodeOptions(options, { 2, ChannelCountMode::ClampedMax, ChannelInterpretation::Speakers });
47 if (result.hasException())
48 return result.releaseException();
49
50 return stereo;
51}

53StereoPannerNode::StereoPannerNode(BaseAudioContext& context, float pan)
54 : AudioNode(context, NodeTypeStereoPanner)
55 , m_pan(AudioParam::create(context, "pan"_s, pan, -1, 1, AutomationRate::ARate))
56 , m_sampleAccurateValues(AudioUtilities::renderQuantumSize)
57{
58 addInput();
59 addOutput(2);
60
61 initialize();
62}
63
64StereoPannerNode::~StereoPannerNode()
65{
66 uninitialize();
67}

69void StereoPannerNode::process(size_t framesToProcess)
70{
71 AudioBus* destination = output(0)->bus();
72
73 if (!isInitialized() || !input(0)->isConnected()) {
74 destination->zero();
75 return;
76 }
77
78 AudioBus* source = input(0)->bus();
79 if (!source) {
80 destination->zero();
81 return;
82 }
83
84 if (m_pan->hasSampleAccurateValues() && m_pan->automationRate() == AutomationRate::ARate) {
85 float* panValues = m_sampleAccurateValues.data();
86 m_pan->calculateSampleAccurateValues(panValues, framesToProcess);
87 StereoPanner::panWithSampleAccurateValues(source, destination, panValues, framesToProcess);
88 return;
89 }
90
91 // The pan value is not sample-accurate or not a-rate. In this case, we have
92 // a fixed pan value for the render and just need to incorporate any inputs to
93 // the value, if any.
94 float panValue = m_pan->hasSampleAccurateValues() ? m_pan->finalValue() : m_pan->value();
95 StereoPanner::panToTargetValue(source, destination, panValue, framesToProcess);
96}
97
98void StereoPannerNode::processOnlyAudioParams(size_t framesToProcess)
99{
100 float values[AudioUtilities::renderQuantumSize];
101 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
102
103 m_pan->calculateSampleAccurateValues(values, framesToProcess);
104}

106ExceptionOr<void> StereoPannerNode::setChannelCount(unsigned channelCount)
107{
108 if (channelCount > 2)
109 return Exception { NotSupportedError, "StereoPannerNode's channelCount cannot be greater than 2."_s };
110
111 return AudioNode::setChannelCount(channelCount);
112}
113
114ExceptionOr<void> StereoPannerNode::setChannelCountMode(ChannelCountMode mode)
115{
116 if (mode == ChannelCountMode::Max)
117 return Exception { NotSupportedError, "StereoPannerNode's channelCountMode cannot be max."_s };
118
119 return AudioNode::setChannelCountMode(mode);
120}
