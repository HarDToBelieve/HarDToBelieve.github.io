39WaveShaperDSPKernel::WaveShaperDSPKernel(WaveShaperProcessor* processor)
40 : AudioDSPKernel(processor)
41{
42 if (processor->oversample() != WaveShaperProcessor::OverSampleNone)
43 lazyInitializeOversampling();
44}

46void WaveShaperDSPKernel::lazyInitializeOversampling()
47{
48 ASSERT(isMainThread());
49
50 if (!m_tempBuffer) {
51 m_tempBuffer = makeUnique<AudioFloatArray>(AudioUtilities::renderQuantumSize * 2);
52 m_tempBuffer2 = makeUnique<AudioFloatArray>(AudioUtilities::renderQuantumSize * 4);
53 m_upSampler = makeUnique<UpSampler>(AudioUtilities::renderQuantumSize);
54 m_downSampler = makeUnique<DownSampler>(AudioUtilities::renderQuantumSize * 2);
55 m_upSampler2 = makeUnique<UpSampler>(AudioUtilities::renderQuantumSize * 2);
56 m_downSampler2 = makeUnique<DownSampler>(AudioUtilities::renderQuantumSize * 4);
57 }
58}

60void WaveShaperDSPKernel::process(const float* source, float* destination, size_t framesToProcess)
61{
62 assertIsHeld(waveShaperProcessor()->processLock());
63 switch (waveShaperProcessor()->oversample()) {
64 case WaveShaperProcessor::OverSampleNone:
65 processCurve(source, destination, framesToProcess);
66 break;
67 case WaveShaperProcessor::OverSample2x:
68 processCurve2x(source, destination, framesToProcess);
69 break;
70 case WaveShaperProcessor::OverSample4x:
71 processCurve4x(source, destination, framesToProcess);
72 break;
73
74 default:
75 ASSERT_NOT_REACHED();
76 }
77}
78
79void WaveShaperDSPKernel::processCurve(const float* source, float* destination, size_t framesToProcess)
80{
81 ASSERT(source && destination && waveShaperProcessor());
82
83 assertIsHeld(waveShaperProcessor()->processLock());
84 Float32Array* curve = waveShaperProcessor()->curve();
85 if (!curve) {
86 // Act as "straight wire" pass-through if no curve is set.
87 memcpy(destination, source, sizeof(float) * framesToProcess);
88 return;
89 }
90
91 float* curveData = curve->data();
92 int curveLength = curve->length();
93
94 ASSERT(curveData);
95
96 if (!curveData || !curveLength) {
97 memcpy(destination, source, sizeof(float) * framesToProcess);
98 return;
99 }
100
101 // Apply waveshaping curve.
102 for (unsigned i = 0; i < framesToProcess; ++i) {
103 const float input = source[i];
104
105 float v = (curveLength - 1) * 0.5f * (input + 1);
106 if (v < 0)
107 destination[i] = curveData[0];
108 else if (v >= curveLength - 1)
109 destination[i] = curveData[curveLength - 1];
110 else {
111 float k = std::floor(v);
112 float f = v - k;
113 unsigned kIndex = k;
114 destination[i] = (1 - f) * curveData[kIndex] + f * curveData[kIndex + 1];
115 }
116 }
117}

119void WaveShaperDSPKernel::processCurve2x(const float* source, float* destination, size_t framesToProcess)
120{
121 bool isSafe = framesToProcess == AudioUtilities::renderQuantumSize;
122 ASSERT(isSafe);
123 if (!isSafe)
124 return;
125
126 float* tempP = m_tempBuffer->data();
127
128 m_upSampler->process(source, tempP, framesToProcess);
129
130 // Process at 2x up-sampled rate.
131 processCurve(tempP, tempP, framesToProcess * 2);
132
133 m_downSampler->process(tempP, destination, framesToProcess * 2);
134}
135
136void WaveShaperDSPKernel::processCurve4x(const float* source, float* destination, size_t framesToProcess)
137{
138 bool isSafe = framesToProcess == AudioUtilities::renderQuantumSize;
139 ASSERT(isSafe);
140 if (!isSafe)
141 return;
142
143 float* tempP = m_tempBuffer->data();
144 float* tempP2 = m_tempBuffer2->data();
145
146 m_upSampler->process(source, tempP, framesToProcess);
147 m_upSampler2->process(tempP, tempP2, framesToProcess * 2);
148
149 // Process at 4x up-sampled rate.
150 processCurve(tempP2, tempP2, framesToProcess * 4);
151
152 m_downSampler2->process(tempP2, tempP, framesToProcess * 4);
153 m_downSampler->process(tempP, destination, framesToProcess * 2);
154}

156void WaveShaperDSPKernel::reset()
157{
158 if (m_upSampler) {
159 m_upSampler->reset();
160 m_downSampler->reset();
161 m_upSampler2->reset();
162 m_downSampler2->reset();
163 }
164}

166double WaveShaperDSPKernel::latencyTime() const
167{
168 if (!waveShaperProcessor()->processLock().tryLock())
169 return std::numeric_limits<double>::infinity();
170
171 Locker locker { AdoptLock, waveShaperProcessor()->processLock() };
172
173 size_t latencyFrames = 0;
174 switch (waveShaperProcessor()->oversample()) {
175 case WaveShaperProcessor::OverSampleNone:
176 break;
177 case WaveShaperProcessor::OverSample2x:
178 latencyFrames += m_upSampler->latencyFrames();
179 latencyFrames += m_downSampler->latencyFrames();
180 break;
181 case WaveShaperProcessor::OverSample4x:
182 {
183 // Account for first stage upsampling.
184 latencyFrames += m_upSampler->latencyFrames();
185 latencyFrames += m_downSampler->latencyFrames();
186
187 // Account for second stage upsampling.
188 // and divide by 2 to get back down to the regular sample-rate.
189 size_t latencyFrames2 = (m_upSampler2->latencyFrames() + m_downSampler2->latencyFrames()) / 2;
190 latencyFrames += latencyFrames2;
191 break;
192 }
193 default:
194 ASSERT_NOT_REACHED();
195 }
196
197 return static_cast<double>(latencyFrames) / sampleRate();
198}

200bool WaveShaperDSPKernel::requiresTailProcessing() const
201{
202 // Always return true even if the tail time and latency might both be zero.
203 return true;
204}
