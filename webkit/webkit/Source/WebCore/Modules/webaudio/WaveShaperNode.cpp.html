40ExceptionOr<Ref<WaveShaperNode>> WaveShaperNode::create(BaseAudioContext& context, const WaveShaperOptions& options)
41{
42 RefPtr<Float32Array> curve;
43 if (options.curve) {
44 curve = Float32Array::tryCreate(options.curve->data(), options.curve->size());
45 if (!curve)
46 return Exception { InvalidStateError, "Invalid curve parameter" };
47 }
48
49 auto node = adoptRef(*new WaveShaperNode(context));
50
51 auto result = node->handleAudioNodeOptions(options, { 2, ChannelCountMode::Max, ChannelInterpretation::Speakers });
52 if (result.hasException())
53 return result.releaseException();
54
55 if (curve) {
56 result = node->setCurveForBindings(WTFMove(curve));
57 if (result.hasException())
58 return result.releaseException();
59 }
60
61 node->setOversampleForBindings(options.oversample);
62
63 return node;
64}

66WaveShaperNode::WaveShaperNode(BaseAudioContext& context)
67 : AudioBasicProcessorNode(context, NodeTypeWaveShaper)
68{
69 m_processor = makeUnique<WaveShaperProcessor>(context.sampleRate(), 1);
70
71 initialize();
72}
73
74ExceptionOr<void> WaveShaperNode::setCurveForBindings(RefPtr<Float32Array>&& curve)
75{
76 ASSERT(isMainThread());
77 DEBUG_LOG(LOGIDENTIFIER);
78 if (curve && curve->length() < 2)
79 return Exception { InvalidStateError, "Length of curve array cannot be less than 2" };
80
81 if (curve) {
82 // The specification states that we should maintain an internal copy of the curve so that
83 // subsequent modifications of the contents of the array have no effect.
84 auto clonedCurve = Float32Array::create(curve->data(), curve->length());
85 curve = WTFMove(clonedCurve);
86 }
87
88 waveShaperProcessor()->setCurveForBindings(curve.get());
89 return { };
90}
91
92Float32Array* WaveShaperNode::curveForBindings()
93{
94 ASSERT(isMainThread());
95 return waveShaperProcessor()->curveForBindings();
96}
97
98static inline WaveShaperProcessor::OverSampleType processorType(OverSampleType type)
99{
100 switch (type) {
101 case OverSampleType::None:
102 return WaveShaperProcessor::OverSampleNone;
103 case OverSampleType::_2x:
104 return WaveShaperProcessor::OverSample2x;
105 case OverSampleType::_4x:
106 return WaveShaperProcessor::OverSample4x;
107 }
108 ASSERT_NOT_REACHED();
109 return WaveShaperProcessor::OverSampleNone;
110}
111
112void WaveShaperNode::setOversampleForBindings(OverSampleType type)
113{
114 ASSERT(isMainThread());
115 INFO_LOG(LOGIDENTIFIER, type);
116
117 // Synchronize with any graph changes or changes to channel configuration.
118 Locker contextLocker { context().graphLock() };
119 waveShaperProcessor()->setOversampleForBindings(processorType(type));
120}
121
122auto WaveShaperNode::oversampleForBindings() const -> OverSampleType
123{
124 ASSERT(isMainThread());
125 switch (waveShaperProcessor()->oversampleForBindings()) {
126 case WaveShaperProcessor::OverSampleNone:
127 return OverSampleType::None;
128 case WaveShaperProcessor::OverSample2x:
129 return OverSampleType::_2x;
130 case WaveShaperProcessor::OverSample4x:
131 return OverSampleType::_4x;
132 }
133 ASSERT_NOT_REACHED();
134 return OverSampleType::None;
135}
136
137bool WaveShaperNode::propagatesSilence() const
138{
139 if (!waveShaperProcessor()->processLock().tryLock())
140 return false;
141
142 Locker locker { AdoptLock, waveShaperProcessor()->processLock() };
143 auto curve = waveShaperProcessor()->curve();
144 return !curve || !curve->length();
145}
