51void WaveShaperProcessor::setCurveForBindings(Float32Array* curve)
52{
53 ASSERT(isMainThread());
54 // This synchronizes with process().
55 Locker locker { m_processLock };
56
57 m_curve = curve;
58}
59
60void WaveShaperProcessor::setOversampleForBindings(OverSampleType oversample)
61{
62 ASSERT(isMainThread());
63 // This synchronizes with process().
64 Locker locker { m_processLock };
65
66 m_oversample = oversample;
67
68 if (oversample == OverSampleNone)
69 return;
70
71 for (auto& audioDSPKernel : m_kernels)
72 static_cast<WaveShaperDSPKernel&>(*audioDSPKernel).lazyInitializeOversampling();
73}
74
75void WaveShaperProcessor::process(const AudioBus* source, AudioBus* destination, size_t framesToProcess)
76{
77 if (!isInitialized()) {
78 destination->zero();
79 return;
80 }
81
82 bool channelCountMatches = source->numberOfChannels() == destination->numberOfChannels() && source->numberOfChannels() == m_kernels.size();
83 ASSERT(channelCountMatches);
84 if (!channelCountMatches)
85 return;
86
87 // The audio thread can't block on this lock, so we use tryLock() instead.
88 if (!m_processLock.tryLock()) {
89 // Too bad - tryLock() failed. We must be in the middle of a setCurve() call.
90 destination->zero();
91 return;
92 }
93 Locker locker { AdoptLock, m_processLock };
94
95 // For each channel of our input, process using the corresponding WaveShaperDSPKernel into the output channel.
96 for (size_t i = 0; i < m_kernels.size(); ++i)
97 static_cast<WaveShaperDSPKernel&>(*m_kernels[i]).process(source->channel(i)->data(), destination->channel(i)->mutableData(), framesToProcess);
98}
