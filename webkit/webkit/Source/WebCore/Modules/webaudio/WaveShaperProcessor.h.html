27#include "AudioDSPKernel.h"
28#include "AudioDSPKernelProcessor.h"
29#include "AudioNode.h"
30#include <JavaScriptCore/Float32Array.h>
31#include <memory>
32#include <wtf/Lock.h>
33#include <wtf/RefPtr.h>
34
35namespace WebCore {
36
37// WaveShaperProcessor is an AudioDSPKernelProcessor which uses WaveShaperDSPKernel objects to implement non-linear distortion effects.
38
39class WaveShaperProcessor final : public AudioDSPKernelProcessor {
40public:
41 enum OverSampleType {
42 OverSampleNone,
43 OverSample2x,
44 OverSample4x
45 };
46
47 WaveShaperProcessor(float sampleRate, size_t numberOfChannels);
48
49 virtual ~WaveShaperProcessor();
50
51 std::unique_ptr<AudioDSPKernel> createKernel() final;
52
53 void process(const AudioBus* source, AudioBus* destination, size_t framesToProcess) final;
54
55 void setCurveForBindings(Float32Array*);
56 Float32Array* curveForBindings() WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_curve.get(); } // Doesn't grab the lock, only safe to call on the main thread.
57 Float32Array* curve() const WTF_REQUIRES_LOCK(m_processLock) { return m_curve.get(); }
58
59 void setOversampleForBindings(OverSampleType);
60 OverSampleType oversampleForBindings() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS { ASSERT(isMainThread()); return m_oversample; } // Doesn't grab the lock, only safe to call on the main thread.
61 OverSampleType oversample() const WTF_REQUIRES_LOCK(m_processLock) { return m_oversample; }
62
63 Lock& processLock() const WTF_RETURNS_LOCK(m_processLock) { return m_processLock; }
64
65private:
66 // m_curve represents the non-linear shaping curve.
67 RefPtr<Float32Array> m_curve WTF_GUARDED_BY_LOCK(m_processLock);
68
69 OverSampleType m_oversample WTF_GUARDED_BY_LOCK(m_processLock) { OverSampleNone };
70
71 // This synchronizes process() with setCurve().
72 mutable Lock m_processLock;
73};
74
75} // namespace WebCore
