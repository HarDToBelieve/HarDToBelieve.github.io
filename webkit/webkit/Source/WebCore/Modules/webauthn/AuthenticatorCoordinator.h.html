36class AbortSignal;

41struct PublicKeyCredentialCreationOptions;

46using CredentialPromise = DOMPromiseDeferred<IDLNullable<IDLInterface<BasicCredential>>>;

48class AuthenticatorCoordinator final {

55 // The following methods implement static methods of PublicKeyCredential.
56 void create(const Document&, const PublicKeyCredentialCreationOptions&, bool sameOriginWithAncestors, RefPtr<AbortSignal>&&, CredentialPromise&&) const;
57 void discoverFromExternalSource(const Document&, const PublicKeyCredentialRequestOptions&, bool sameOriginWithAncestors, RefPtr<AbortSignal>&&, CredentialPromise&&) const;
58 void isUserVerifyingPlatformAuthenticatorAvailable(DOMPromiseDeferred<IDLBoolean>&&) const;
59
60 void resetUserGestureRequirement();
61
62private:
63 AuthenticatorCoordinator() = default;
64
65 std::unique_ptr<AuthenticatorCoordinatorClient> m_client;
66};
