37RefPtr<AuthenticatorResponse> AuthenticatorResponse::tryCreate(AuthenticatorResponseData&& data, AuthenticatorAttachment attachment)
38{
39 if (!data.rawId)
40 return nullptr;
41
42 if (data.isAuthenticatorAttestationResponse) {
43 if (!data.attestationObject)
44 return nullptr;
45
46 return AuthenticatorAttestationResponse::create(data.rawId.releaseNonNull(), data.attestationObject.releaseNonNull(), attachment);
47 }
48
49 if (!data.authenticatorData || !data.signature)
50 return nullptr;
51
52 return AuthenticatorAssertionResponse::create(data.rawId.releaseNonNull(), data.authenticatorData.releaseNonNull(), data.signature.releaseNonNull(), WTFMove(data.userHandle), AuthenticationExtensionsClientOutputs { data.appid }, attachment);
53}
54
55AuthenticatorResponseData AuthenticatorResponse::data() const
56{
57 AuthenticatorResponseData data;
58 data.rawId = m_rawId.copyRef();
59 data.appid = m_extensions.appid;
60 return data;
61}
