40// Authenticator response for AuthenticatorGetInfo request that encapsulates
41// versions, options, AAGUID(Authenticator Attestation GUID), other
42// authenticator device information.
43// https://fidoalliance.org/specs/fido-v2.0-ps-20170927/fido-client-to-authenticator-protocol-v2.0-ps-20170927.html#authenticatorgetinfo-0x04
44class WEBCORE_EXPORT AuthenticatorGetInfoResponse {
45 WTF_MAKE_NONCOPYABLE(AuthenticatorGetInfoResponse);
46public:
47 AuthenticatorGetInfoResponse(StdSet<ProtocolVersion>&& versions, Vector<uint8_t>&& aaguid);
48 AuthenticatorGetInfoResponse(AuthenticatorGetInfoResponse&& that) = default;
49 AuthenticatorGetInfoResponse& operator=(AuthenticatorGetInfoResponse&& other) = default;
50
51 AuthenticatorGetInfoResponse& setMaxMsgSize(uint32_t);
52 AuthenticatorGetInfoResponse& setPinProtocols(Vector<uint8_t>&&);
53 AuthenticatorGetInfoResponse& setExtensions(Vector<String>&&);
54 AuthenticatorGetInfoResponse& setOptions(AuthenticatorSupportedOptions&&);
55
56 const StdSet<ProtocolVersion>& versions() const { return m_versions; }
57 const Vector<uint8_t>& aaguid() const { return m_aaguid; }
58 const std::optional<uint32_t>& maxMsgSize() const { return m_maxMsgSize; }
59 const std::optional<Vector<uint8_t>>& pinProtocol() const { return m_pinProtocols; }
60 const std::optional<Vector<String>>& extensions() const { return m_extensions; }
61 const AuthenticatorSupportedOptions& options() const { return m_options; }
62
63private:
64 StdSet<ProtocolVersion> m_versions;
65 Vector<uint8_t> m_aaguid;
66 std::optional<uint32_t> m_maxMsgSize;
67 std::optional<Vector<uint8_t>> m_pinProtocols;
68 std::optional<Vector<String>> m_extensions;
69 AuthenticatorSupportedOptions m_options;
70};
71
72WEBCORE_EXPORT Vector<uint8_t> encodeAsCBOR(const AuthenticatorGetInfoResponse&);
73
74} // namespace fido
