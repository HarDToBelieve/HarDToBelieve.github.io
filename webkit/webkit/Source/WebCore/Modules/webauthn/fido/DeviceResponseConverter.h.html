40// Converts response from authenticators to CTAPResponse objects. If the
41// response of the authenticator does not conform to format specified by the
42// CTAP protocol, null optional is returned.
43namespace fido {
44
45std::optional<cbor::CBORValue> decodeResponseMap(const Vector<uint8_t>&);
46
47// Parses response code from response received from the authenticator. If
48// unknown response code value is received, then CTAP2_ERR_OTHER is returned.
49WEBCORE_EXPORT CtapDeviceResponseCode getResponseCode(const Vector<uint8_t>&);
50
51// De-serializes CBOR encoded response, checks for valid CBOR map formatting,
52// and converts response to AuthenticatorMakeCredentialResponse object with
53// CBOR map keys that conform to format of attestation object defined by the
54// WebAuthN spec : https://w3c.github.io/webauthn/#fig-attStructs
55WEBCORE_EXPORT RefPtr<WebCore::AuthenticatorAttestationResponse> readCTAPMakeCredentialResponse(const Vector<uint8_t>&, WebCore::AuthenticatorAttachment, const WebCore::AttestationConveyancePreference& = WebCore::AttestationConveyancePreference::Direct);
56
57// De-serializes CBOR encoded response to AuthenticatorGetAssertion /
58// AuthenticatorGetNextAssertion request to AuthenticatorGetAssertionResponse
59// object.
60WEBCORE_EXPORT RefPtr<WebCore::AuthenticatorAssertionResponse> readCTAPGetAssertionResponse(const Vector<uint8_t>&, WebCore::AuthenticatorAttachment);
61
62// De-serializes CBOR encoded response to AuthenticatorGetInfo request to
63// AuthenticatorGetInfoResponse object.
64WEBCORE_EXPORT std::optional<AuthenticatorGetInfoResponse> readCTAPGetInfoResponse(const Vector<uint8_t>&);
65
66} // namespace fido
