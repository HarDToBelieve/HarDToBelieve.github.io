35Vector<uint8_t> getInitPacketData(const Vector<uint8_t>& data)
36{
37 Vector<uint8_t> result;
38 result.append(data.data(), data.size() > kHidInitPacketDataSize ? kHidInitPacketDataSize : data.size());
39 return result;
40}
41
42Vector<uint8_t> getContinuationPacketData(const Vector<uint8_t>& data, size_t beginPosition)
43{
44 if (beginPosition > data.size())
45 return { };
46
47 Vector<uint8_t> result;
48 result.append(data.data() + beginPosition, data.size() > kHidContinuationPacketDataSize + beginPosition ? kHidContinuationPacketDataSize : data.size() - beginPosition);
49 return result;
50}
