45// In a U2F registration response, the key is in X9.62 format:
46// - a constant 0x04 prefix to indicate an uncompressed key
47// - the 32-byte x coordinate
48// - the 32-byte y coordinate.
49const uint8_t uncompressedKey = 0x04;
50// https://www.w3.org/TR/webauthn/#flags
51const uint8_t makeCredentialFlags = 0b01000001; // UP and AT are set.
52// https://fidoalliance.org/specs/fido-u2f-v1.2-ps-20170411/fido-u2f-raw-message-formats-v1.2-ps-20170411.html#authentication-response-message-success
53const size_t flagIndex = 0;
54const size_t counterIndex = 1;
55const size_t signatureIndex = 5;
56
57static Vector<uint8_t> extractECPublicKeyFromU2fRegistrationResponse(const Vector<uint8_t>& u2fData)
58{
59 size_t pos = kReservedLength;
60 if (u2fData.size() <= pos || u2fData[pos] != uncompressedKey)
61 return { };
62 pos++;
63
64 if (u2fData.size() < pos + 2 * ES256FieldElementLength)
65 return { };
66
67 Vector<uint8_t> x { u2fData.data() + pos, ES256FieldElementLength };
68 pos += ES256FieldElementLength;
69
70 Vector<uint8_t> y { u2fData.data() + pos, ES256FieldElementLength };
71 return encodeES256PublicKeyAsCBOR(WTFMove(x), WTFMove(y));
72}
73
74static Vector<uint8_t> extractCredentialIdFromU2fRegistrationResponse(const Vector<uint8_t>& u2fData)
75{
76 size_t pos = kU2fKeyHandleLengthOffset;
77 if (u2fData.size() <= pos)
78 return { };
79 size_t credentialIdLength = u2fData[pos];
80 pos++;
81
82 if (u2fData.size() < pos + credentialIdLength)
83 return { };
84 return { u2fData.data() + pos, credentialIdLength };
85}

87static Vector<uint8_t> createAttestedCredentialDataFromU2fRegisterResponse(const Vector<uint8_t>& u2fData, const Vector<uint8_t>& publicKey)
88{
89 auto credentialId = extractCredentialIdFromU2fRegistrationResponse(u2fData);
90 if (credentialId.isEmpty())
91 return { };
92
93 return buildAttestedCredentialData(Vector<uint8_t>(aaguidLength, 0), credentialId, publicKey);
94}

96static size_t parseX509Length(const Vector<uint8_t>& u2fData, size_t offset)
97{
98 if (u2fData.size() <= offset || u2fData[offset] != SequenceMark)
99 return 0;
100 offset++;
101
102 if (u2fData.size() <= offset)
103 return 0;
104 const auto sequenceLengthLength = bytesUsedToEncodedLength(u2fData[offset]);
105
106 if (sequenceLengthLength > sizeof(size_t) || (u2fData.size() < offset + sequenceLengthLength))
107 return 0;
108 size_t sequenceLength = sequenceLengthLength == 1 ? u2fData[offset] : 0;
109 offset++;
110 for (auto i = sequenceLengthLength - 1; i; i--, offset++)
111 sequenceLength += u2fData[offset] << (i - 1) * 8;
112
113 return sequenceLength + sequenceLengthLength + sizeof(SequenceMark);
114}

116static cbor::CBORValue::MapValue createFidoAttestationStatementFromU2fRegisterResponse(const Vector<uint8_t>& u2fData, size_t offset)
117{
118 auto x509Length = parseX509Length(u2fData, offset);
119 if (!x509Length || u2fData.size() < offset + x509Length)
120 return { };
121
122 Vector<uint8_t> x509 { u2fData.data() + offset, x509Length };
123 offset += x509Length;
124
125 Vector<uint8_t> signature { u2fData.data() + offset, u2fData.size() - offset };
126 if (signature.isEmpty())
127 return { };
128
129 cbor::CBORValue::MapValue attestationStatementMap;
130 attestationStatementMap[cbor::CBORValue("sig")] = cbor::CBORValue(WTFMove(signature));
131 Vector<cbor::CBORValue> cborArray;
132 cborArray.append(cbor::CBORValue(WTFMove(x509)));
133 attestationStatementMap[cbor::CBORValue("x5c")] = cbor::CBORValue(WTFMove(cborArray));
134
135 return attestationStatementMap;
136}

140RefPtr<AuthenticatorAttestationResponse> readU2fRegisterResponse(const String& rpId, const Vector<uint8_t>& u2fData, AuthenticatorAttachment attachment, const AttestationConveyancePreference& attestation)
141{
142 auto publicKey = extractECPublicKeyFromU2fRegistrationResponse(u2fData);
143 if (publicKey.isEmpty())
144 return nullptr;
145
146 auto attestedCredentialData = createAttestedCredentialDataFromU2fRegisterResponse(u2fData, publicKey);
147 if (attestedCredentialData.isEmpty())
148 return nullptr;
149
150 // Extract the credentialId for packing into the response data.
151 auto credentialId = extractCredentialIdFromU2fRegistrationResponse(u2fData);
152 ASSERT(!credentialId.isEmpty());
153
154 // The counter is zeroed out for Register requests.
155 auto authData = buildAuthData(rpId, makeCredentialFlags, 0, attestedCredentialData);
156
157 auto fidoAttestationStatement = createFidoAttestationStatementFromU2fRegisterResponse(u2fData, kU2fKeyHandleOffset + credentialId.size());
158 if (fidoAttestationStatement.empty())
159 return nullptr;
160
161 auto attestationObject = buildAttestationObject(WTFMove(authData), "fido-u2f", WTFMove(fidoAttestationStatement), attestation);
162
163 return AuthenticatorAttestationResponse::create(credentialId, attestationObject, attachment);
164}

166RefPtr<AuthenticatorAssertionResponse> readU2fSignResponse(const String& rpId, const Vector<uint8_t>& keyHandle, const Vector<uint8_t>& u2fData, AuthenticatorAttachment attachment)
167{
168 if (keyHandle.isEmpty() || u2fData.size() <= signatureIndex)
169 return nullptr;
170
171 // 1 byte flags, 4 bytes counter
172 auto flags = u2fData[flagIndex];
173 uint32_t counter = u2fData[counterIndex] << 24;
174 counter += u2fData[counterIndex + 1] << 16;
175 counter += u2fData[counterIndex + 2] << 8;
176 counter += u2fData[counterIndex + 3];
177 auto authData = buildAuthData(rpId, flags, counter, { });
178
179 // FIXME: Find a way to remove the need of constructing a vector here.
180 Vector<uint8_t> signature { u2fData.data() + signatureIndex, u2fData.size() - signatureIndex };
181 return AuthenticatorAssertionResponse::create(keyHandle, authData, signature, { }, attachment);
182}
