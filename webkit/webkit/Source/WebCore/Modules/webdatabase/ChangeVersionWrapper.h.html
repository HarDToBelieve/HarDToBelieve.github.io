31#include "SQLTransaction.h"

34namespace WebCore {
35
36class SQLError;
37
38class ChangeVersionWrapper : public SQLTransactionWrapper {
39public:
40 static Ref<ChangeVersionWrapper> create(const String& oldVersion, const String& newVersion) { return adoptRef(*new ChangeVersionWrapper(oldVersion, newVersion)); }
41
42 bool performPreflight(SQLTransaction&) override;
43 bool performPostflight(SQLTransaction&) override;
44 SQLError* sqlError() const override { return m_sqlError.get(); };
45 void handleCommitFailedAfterPostflight(SQLTransaction&) override;
46
47private:
48 ChangeVersionWrapper(const String& oldVersion, const String& newVersion);
49
50 String m_oldVersion;
51 String m_newVersion;
52 RefPtr<SQLError> m_sqlError;
53};
