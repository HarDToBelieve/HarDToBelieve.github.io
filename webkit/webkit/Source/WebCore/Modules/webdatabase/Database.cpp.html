64// Registering "opened" databases with the DatabaseTracker
65// =======================================================
66// The DatabaseTracker maintains a list of databases that have been
67// "opened" so that the client can call interrupt or delete on every database
68// associated with a DatabaseContext.
69//
70// We will only call DatabaseTracker::addOpenDatabase() to add the database
71// to the tracker as opened when we've succeeded in opening the database,
72// and will set m_opened to true. Similarly, we only call
73// DatabaseTracker::removeOpenDatabase() to remove the database from the
74// tracker when we set m_opened to false in closeDatabase(). This sets up
75// a simple symmetry between open and close operations, and a direct
76// correlation to adding and removing databases from the tracker's list,
77// thus ensuring that we have a correct list for the interrupt and
78// delete operations to work on.
79//
80// The only databases instances not tracked by the tracker's open database
81// list are the ones that have not been added yet, or the ones that we
82// attempted an open on but failed to. Such instances only exist in the
83// factory functions for creating database backends.
84//
85// The factory functions will either call openAndVerifyVersion() or
86// performOpenAndVerify(). These methods will add the newly instantiated
87// database backend if they succeed in opening the requested database.
88// In the case of failure to open the database, the factory methods will
89// simply discard the newly instantiated database backend when they return.
90// The ref counting mechanims will automatically destruct the un-added
91// (and un-returned) databases instances.

115static bool setTextValueInDatabase(SQLiteDatabase& db, const String& query, const String& value)
116{
117 auto statement = db.prepareStatementSlow(query);
118 if (!statement) {
119 LOG_ERROR("Failed to prepare statement to set value in database (%s)", query.ascii().data());
120 return false;
121 }
122
123 statement->bindText(1, value);
124
125 if (statement->step() != SQLITE_DONE) {
126 LOG_ERROR("Failed to step statement to set value in database (%s)", query.ascii().data());
127 return false;
128 }
129
130 return true;
131}
132
133static bool retrieveTextResultFromDatabase(SQLiteDatabase& db, const String& query, String& resultString)
134{
135 auto statement = db.prepareStatementSlow(query);
136 if (!statement) {
137 LOG_ERROR("Error (%i) preparing statement to read text result from database (%s)", statement.error(), query.ascii().data());
138 return false;
139 }
140
141 int result = statement->step();
142 if (result == SQLITE_ROW) {
143 resultString = statement->columnText(0);
144 return true;
145 }
146 if (result == SQLITE_DONE) {
147 resultString = String();
148 return true;
149 }
150
151 LOG_ERROR("Error (%i) reading text result from database (%s)", result, query.ascii().data());
152 return false;
153}

164static inline void updateGUIDVersionMap(DatabaseGUID guid, const String& newVersion) WTF_REQUIRES_LOCK(guidLock)
165{
166 // Note: It is not safe to put an empty string into the guidToVersionMap() map.
167 // That's because the map is cross-thread, but empty strings are per-thread.
168 // The copy() function makes a version of the string you can use on the current
169 // thread, but we need a string we can keep in a cross-thread data structure.
170 // FIXME: This is a quite-awkward restriction to have to program with.
171
172 // Map empty string to null string (see comment above).
173 guidToVersionMap().set(guid, newVersion.isEmpty() ? String() : newVersion.isolatedCopy());
174}

182static inline DatabaseGUID guidForOriginAndName(const String& origin, const String& name) WTF_REQUIRES_LOCK(guidLock)
183{
184 static NeverDestroyed<HashMap<String, DatabaseGUID>> map;

187 return ++lastUsedGUID;
188 }).iterator->value;

223Database::~Database()
224{
225 // The reference to the Document needs to be cleared on the JavaScript thread. If we're on that thread already, we can just let the RefPtr's destruction do the dereffing.
226 if (!isMainThread())
227 callOnMainThread([document = WTFMove(m_document), databaseContext = WTFMove(m_databaseContext)] { });
228
229 // SQLite is "multi-thread safe", but each database handle can only be used
230 // on a single thread at a time.
231 //
232 // For DatabaseBackend, we open the SQLite database on the DatabaseThread,
233 // and hence we should also close it on that same thread. This means that the
234 // SQLite database need to be closed by another mechanism (see
235 // DatabaseContext::stopDatabases()). By the time we get here, the SQLite
236 // database should have already been closed.
237
238 ASSERT(!m_opened);
239}

241ExceptionOr<void> Database::openAndVerifyVersion(bool setVersionInNewDatabase)
242{
243 DatabaseTaskSynchronizer synchronizer;
244 auto& thread = databaseThread();
245 if (thread.terminationRequested(&synchronizer))
246 return Exception { InvalidStateError };
247
248 ExceptionOr<void> result;
249 auto task = makeUnique<DatabaseOpenTask>(*this, setVersionInNewDatabase, synchronizer, result);
250 thread.scheduleImmediateTask(WTFMove(task));
251 synchronizer.waitForTaskCompletion();
252
253 return result;
254}

262void Database::close()
263{
264 auto& thread = databaseThread();
265
266 DatabaseTaskSynchronizer synchronizer;
267 if (thread.terminationRequested(&synchronizer)) {
268 LOG(StorageAPI, "Database handle %p is on a terminated DatabaseThread, cannot be marked for normal closure\n", this);
269 return;
270 }
271
272 thread.scheduleImmediateTask(makeUnique<DatabaseCloseTask>(*this, synchronizer));
273
274 // FIXME: iOS depends on this function blocking until the database is closed as part
275 // of closing all open databases from a process assertion expiration handler.
276 // See <https://bugs.webkit.org/show_bug.cgi?id=157184>.
277 synchronizer.waitForTaskCompletion();
278}

280void Database::performClose()
281{
282 ASSERT(databaseThread().getThread() == &Thread::current());
283
284 {
285 Locker locker { m_transactionInProgressLock };
286
287 // Clean up transactions that have not been scheduled yet:
288 // Transaction phase 1 cleanup. See comment on "What happens if a
289 // transaction is interrupted?" at the top of SQLTransactionBackend.cpp.
290 while (!m_transactionQueue.isEmpty())
291 m_transactionQueue.takeFirst()->notifyDatabaseThreadIsShuttingDown();
292
293 m_isTransactionQueueEnabled = false;
294 m_transactionInProgress = false;
295 }
296
297 closeDatabase();
298
299 // DatabaseThread keeps databases alive by referencing them in its
300 // m_openDatabaseSet. DatabaseThread::recordDatabaseClose() will remove
301 // this database from that set (which effectively deref's it). We hold on
302 // to it with a local pointer here for a liitle longer, so that we can
303 // unschedule any DatabaseTasks that refer to it before the database gets
304 // deleted.
305 Ref<Database> protectedThis(*this);
306 auto& thread = databaseThread();
307 thread.recordDatabaseClosed(*this);
308 thread.unscheduleDatabaseTasks(*this);
309}

327ExceptionOr<void> Database::performOpenAndVerify(bool shouldSetVersionInNewDatabase)
328{
329 DoneCreatingDatabaseOnExitCaller onExitCaller(*this);
330
331 const int maxSqliteBusyWaitTime = 30000;
332
333#if PLATFORM(IOS_FAMILY)
334 {
335 // Make sure we wait till the background removal of the empty database files finished before trying to open any database.
336 Locker locker { DatabaseTracker::openDatabaseMutex() };
337 }
338#endif
339
340 SQLiteTransactionInProgressAutoCounter transactionCounter;
341
342 if (!m_sqliteDatabase.open(m_filename))
343 return Exception { InvalidStateError, formatErrorMessage("unable to open database", m_sqliteDatabase.lastError(), m_sqliteDatabase.lastErrorMsg()) };
344 if (!m_sqliteDatabase.turnOnIncrementalAutoVacuum())
345 LOG_ERROR("Unable to turn on incremental auto-vacuum (%d %s)", m_sqliteDatabase.lastError(), m_sqliteDatabase.lastErrorMsg());
346
347 m_sqliteDatabase.setBusyTimeout(maxSqliteBusyWaitTime);
348
349 String currentVersion;
350 {
351 Locker locker { guidLock };
352
353 auto entry = guidToVersionMap().find(m_guid);
354 if (entry != guidToVersionMap().end()) {
355 // Map null string to empty string (see updateGUIDVersionMap()).
356 currentVersion = entry->value.isNull() ? emptyString() : entry->value.isolatedCopy();
357 LOG(StorageAPI, "Current cached version for guid %i is %s", m_guid, currentVersion.ascii().data());
358 } else {
359 LOG(StorageAPI, "No cached version for guid %i", m_guid);
360
361 SQLiteTransaction transaction(m_sqliteDatabase);
362 transaction.begin();
363 if (!transaction.inProgress()) {
364 String message = formatErrorMessage("unable to open database, failed to start transaction", m_sqliteDatabase.lastError(), m_sqliteDatabase.lastErrorMsg());
365 m_sqliteDatabase.close();
366 return Exception { InvalidStateError, WTFMove(message) };
367 }
368
369 String tableName(unqualifiedInfoTableName);
370 if (!m_sqliteDatabase.tableExists(tableName)) {
371 m_new = true;
372
373 if (!m_sqliteDatabase.executeCommandSlow("CREATE TABLE " + tableName + " (key TEXT NOT NULL ON CONFLICT FAIL UNIQUE ON CONFLICT REPLACE,value TEXT NOT NULL ON CONFLICT FAIL);")) {
374 String message = formatErrorMessage("unable to open database, failed to create 'info' table", m_sqliteDatabase.lastError(), m_sqliteDatabase.lastErrorMsg());
375 transaction.rollback();
376 m_sqliteDatabase.close();
377 return Exception { InvalidStateError, WTFMove(message) };
378 }
379 } else if (!getVersionFromDatabase(currentVersion, false)) {
380 String message = formatErrorMessage("unable to open database, failed to read current version", m_sqliteDatabase.lastError(), m_sqliteDatabase.lastErrorMsg());
381 transaction.rollback();
382 m_sqliteDatabase.close();
383 return Exception { InvalidStateError, WTFMove(message) };
384 }
385
386 if (currentVersion.length()) {
387 LOG(StorageAPI, "Retrieved current version %s from database %s", currentVersion.ascii().data(), databaseDebugName().ascii().data());
388 } else if (!m_new || shouldSetVersionInNewDatabase) {
389 LOG(StorageAPI, "Setting version %s in database %s that was just created", m_expectedVersion.ascii().data(), databaseDebugName().ascii().data());
390 if (!setVersionInDatabase(m_expectedVersion, false)) {
391 String message = formatErrorMessage("unable to open database, failed to write current version", m_sqliteDatabase.lastError(), m_sqliteDatabase.lastErrorMsg());
392 transaction.rollback();
393 m_sqliteDatabase.close();
394 return Exception { InvalidStateError, WTFMove(message) };
395 }
396 currentVersion = m_expectedVersion;
397 }
398 updateGUIDVersionMap(m_guid, currentVersion);
399 transaction.commit();
400 }
401 }
402
403 if (currentVersion.isNull()) {
404 LOG(StorageAPI, "Database %s does not have its version set", databaseDebugName().ascii().data());
405 currentVersion = emptyString();
406 }
407
408 // If the expected version isn't the empty string, ensure that the current database version we have matches that version. Otherwise, set an exception.
409 // If the expected version is the empty string, then we always return with whatever version of the database we have.
410 if ((!m_new || shouldSetVersionInNewDatabase) && m_expectedVersion.length() && m_expectedVersion != currentVersion) {
411 m_sqliteDatabase.close();
412 return Exception { InvalidStateError, "unable to open database, version mismatch, '" + m_expectedVersion + "' does not match the currentVersion of '" + currentVersion + "'" };
413 }
414
415 m_sqliteDatabase.setAuthorizer(m_databaseAuthorizer.get());
416
417 DatabaseTracker::singleton().addOpenDatabase(*this);
418 m_opened = true;
419
420 if (m_new && !shouldSetVersionInNewDatabase)
421 m_expectedVersion = emptyString(); // The caller provided a creationCallback which will set the expected version.
422
423 databaseThread().recordDatabaseOpen(*this);
424
425 return { };
426}

428void Database::closeDatabase()
429{
430 if (!m_opened)
431 return;
432
433 m_sqliteDatabase.close();
434 m_opened = false;
435
436 // See comment at the top this file regarding calling removeOpenDatabase().
437 DatabaseTracker::singleton().removeOpenDatabase(*this);
438
439 {
440 Locker locker { guidLock };
441
442 auto it = guidToDatabaseMap().find(m_guid);
443 ASSERT(it != guidToDatabaseMap().end());
444 ASSERT(it->value.contains(this));
445 it->value.remove(this);
446 if (it->value.isEmpty()) {
447 guidToDatabaseMap().remove(it);
448 guidToVersionMap().remove(m_guid);
449 }
450 }
451}

453bool Database::getVersionFromDatabase(String& version, bool shouldCacheVersion)
454{
455 String query(String("SELECT value FROM ") + fullyQualifiedInfoTableName() + " WHERE key = '" + versionKey + "';");
456
457 m_databaseAuthorizer->disable();
458
459 bool result = retrieveTextResultFromDatabase(m_sqliteDatabase, query, version);
460 if (result) {
461 if (shouldCacheVersion)
462 setCachedVersion(version);
463 } else
464 LOG_ERROR("Failed to retrieve version from database %s", databaseDebugName().ascii().data());
465
466 m_databaseAuthorizer->enable();
467
468 return result;
469}

471bool Database::setVersionInDatabase(const String& version, bool shouldCacheVersion)
472{
473 // The INSERT will replace an existing entry for the database with the new version number, due to the UNIQUE ON CONFLICT REPLACE
474 // clause in the CREATE statement (see Database::performOpenAndVerify()).
475 String query(String("INSERT INTO ") + fullyQualifiedInfoTableName() + " (key, value) VALUES ('" + versionKey + "', ?);");
476
477 m_databaseAuthorizer->disable();
478
479 bool result = setTextValueInDatabase(m_sqliteDatabase, query, version);
480 if (result) {
481 if (shouldCacheVersion)
482 setCachedVersion(version);
483 } else
484 LOG_ERROR("Failed to set version %s in database (%s)", version.ascii().data(), query.ascii().data());
485
486 m_databaseAuthorizer->enable();
487
488 return result;
489}

519void Database::scheduleTransaction()
520{
521 ASSERT(m_transactionInProgressLock.isHeld());
522
523 if (!m_isTransactionQueueEnabled || m_transactionQueue.isEmpty()) {
524 m_transactionInProgress = false;
525 return;
526 }
527
528 m_transactionInProgress = true;
529
530 auto transaction = m_transactionQueue.takeFirst();
531 auto task = makeUnique<DatabaseTransactionTask>(WTFMove(transaction));
532 LOG(StorageAPI, "Scheduling DatabaseTransactionTask %p for transaction %p\n", task.get(), task->transaction());
533 databaseThread().scheduleTask(WTFMove(task));
534}

681void Database::runTransaction(RefPtr<SQLTransactionCallback>&& callback, RefPtr<SQLTransactionErrorCallback>&& errorCallback, RefPtr<VoidCallback>&& successCallback, RefPtr<SQLTransactionWrapper>&& wrapper, bool readOnly)
682{
683 ASSERT(isMainThread());
684 Locker locker { m_transactionInProgressLock };
685 if (!m_isTransactionQueueEnabled) {
686 if (errorCallback) {
687 m_document->eventLoop().queueTask(TaskSource::Networking, [errorCallback = makeRef(*errorCallback)]() {
688 errorCallback->handleEvent(SQLError::create(SQLError::UNKNOWN_ERR, "database has been closed"));
689 });
690 }
691 return;
692 }
693
694 m_transactionQueue.append(SQLTransaction::create(*this, WTFMove(callback), WTFMove(successCallback), errorCallback.copyRef(), WTFMove(wrapper), readOnly));
695 if (!m_transactionInProgress)
696 scheduleTransaction();
697}

699void Database::scheduleTransactionCallback(SQLTransaction* transaction)
700{
701 callOnMainThread([this, protectedThis = makeRef(*this), transaction = makeRefPtr(transaction)]() mutable {
702 m_document->eventLoop().queueTask(TaskSource::Networking, [transaction = WTFMove(transaction)] {
703 transaction->performPendingCallback();
704 });
705 });
706}

708Vector<String> Database::performGetTableNames()
709{
710 disableAuthorizer();
711
712 auto statement = sqliteDatabase().prepareStatement("SELECT name FROM sqlite_master WHERE type='table';"_s);
713 if (!statement) {
714 LOG_ERROR("Unable to retrieve list of tables for database %s", databaseDebugName().ascii().data());
715 enableAuthorizer();
716 return Vector<String>();
717 }
718
719 Vector<String> tableNames;
720 int result;
721 while ((result = statement->step()) == SQLITE_ROW) {
722 String name = statement->columnText(0);
723 if (name != unqualifiedInfoTableName)
724 tableNames.append(name);
725 }
726
727 enableAuthorizer();
728
729 if (result != SQLITE_DONE) {
730 LOG_ERROR("Error getting tables for database %s", databaseDebugName().ascii().data());
731 return Vector<String>();
732 }
733
734 return tableNames;
735}

791bool Database::didExceedQuota()
792{
793 ASSERT(isMainThread());
794 auto& tracker = DatabaseTracker::singleton();
795 auto oldQuota = tracker.quota(securityOrigin());
796 if (estimatedSize() <= oldQuota) {
797 // The expected usage provided by the page is now smaller than the actual database size so we bump the expected usage to
798 // oldQuota + 5MB so that the client actually increases the quota.
799 setEstimatedSize(oldQuota + quotaIncreaseSize);
800 }
801 databaseContext().databaseExceededQuota(stringIdentifierIsolatedCopy(), details());
802 return tracker.quota(securityOrigin()) > oldQuota;
803}
