59static Vector<String> isolatedCopy(const Vector<String>& original)
60{
61 Vector<String> copy;
62 copy.reserveInitialCapacity(original.size());
63 for (auto& string : original)
64 copy.uncheckedAppend(string.isolatedCopy());
65 return copy;
66}

105void DatabaseTracker::openTrackerDatabase(TrackerCreationAction createAction)
106{
107 ASSERT(m_databaseGuard.isHeld());
108
109 if (m_database.isOpen())
110 return;
111
112 // If createIfDoesNotExist is false, SQLiteFileSystem::ensureDatabaseFileExists()
113 // will return false if the database file does not exist.
114 // If createIfDoesNotExist is true, SQLiteFileSystem::ensureDatabaseFileExists()
115 // will attempt to create the path to the database file if it does not
116 // exists yet. It'll return true if the path already exists, or if it
117 // successfully creates the path. Else, it will return false.
118 String databasePath = trackerDatabasePath();
119 if (!SQLiteFileSystem::ensureDatabaseFileExists(databasePath, createAction == CreateIfDoesNotExist))
120 return;
121
122 if (!m_database.open(databasePath)) {
123 // FIXME: What do do here?
124 LOG_ERROR("Failed to open databasePath %s.", databasePath.utf8().data());
125 return;
126 }
127 m_database.disableThreadingChecks();
128
129 if (!m_database.tableExists("Origins")) {
130 if (!m_database.executeCommand("CREATE TABLE Origins (origin TEXT UNIQUE ON CONFLICT REPLACE, quota INTEGER NOT NULL ON CONFLICT FAIL);"_s)) {
131 // FIXME: and here
132 LOG_ERROR("Failed to create Origins table");
133 }
134 }
135
136 if (!m_database.tableExists("Databases")) {
137 if (!m_database.executeCommand("CREATE TABLE Databases (guid INTEGER PRIMARY KEY AUTOINCREMENT, origin TEXT, name TEXT, displayName TEXT, estimatedSize INTEGER, path TEXT);"_s)) {
138 // FIXME: and here
139 LOG_ERROR("Failed to create Databases table");
140 }
141 }
142}

144ExceptionOr<void> DatabaseTracker::hasAdequateQuotaForOrigin(const SecurityOriginData& origin, uint64_t estimatedSize)
145{
146 ASSERT(m_databaseGuard.isHeld());
147 auto usage = this->usage(origin);
148
149 // If the database will fit, allow its creation.
150 auto requirement = usage + std::max<uint64_t>(1u, estimatedSize);
151 if (requirement < usage) {
152 // The estimated size is so big it causes an overflow; don't allow creation.
153 return Exception { SecurityError };
154 }
155 if (requirement > quotaNoLock(origin))
156 return Exception { QuotaExceededError };
157 return { };
158}

160ExceptionOr<void> DatabaseTracker::canEstablishDatabase(DatabaseContext& context, const String& name, uint64_t estimatedSize)
161{
162 Locker lockDatabase { m_databaseGuard };
163
164 // FIXME: What guarantees this context.securityOrigin() is non-null?
165 auto origin = context.securityOrigin();
166
167 if (isDeletingDatabaseOrOriginFor(origin, name))
168 return Exception { SecurityError };
169
170 recordCreatingDatabase(origin, name);
171
172 // If a database already exists, ignore the passed-in estimated size and say it's OK.
173 if (hasEntryForDatabase(origin, name))
174 return { };
175
176 auto result = hasAdequateQuotaForOrigin(origin, estimatedSize);
177 if (!result.hasException())
178 return { };
179
180 // If we get here, then we do not have enough quota for one of the
181 // following reasons as indicated by the set error:
182 //
183 // If the error is DatabaseSizeOverflowed, then this means the requested
184 // estimatedSize if so unreasonably large that it can cause an overflow in
185 // the usage budget computation. In that case, there's nothing more we can
186 // do, and there's no need for a retry. Hence, we should indicate that
187 // we're done with our attempt to create the database.
188 //
189 // If the error is DatabaseSizeExceededQuota, then we'll give the client
190 // a chance to update the quota and call retryCanEstablishDatabase() to try
191 // again. Hence, we don't call doneCreatingDatabase() yet in that case.
192
193 auto exception = result.releaseException();
194 if (exception.code() != QuotaExceededError)
195 doneCreatingDatabase(origin, name);
196
197 return exception;
198}
199
200// Note: a thought about performance: hasAdequateQuotaForOrigin() was also
201// called in canEstablishDatabase(), and hence, we're repeating some work within
202// hasAdequateQuotaForOrigin(). However, retryCanEstablishDatabase() should only
203// be called in the rare even if canEstablishDatabase() fails. Since it is rare,
204// we should not bother optimizing it. It is more beneficial to keep
205// hasAdequateQuotaForOrigin() simple and correct (i.e. bug free), and just
206// re-use it. Also note that the path for opening a database involves IO, and
207// hence should not be a performance critical path anyway.
208ExceptionOr<void> DatabaseTracker::retryCanEstablishDatabase(DatabaseContext& context, const String& name, uint64_t estimatedSize)
209{
210 Locker lockDatabase { m_databaseGuard };
211
212 // FIXME: What guarantees context.securityOrigin() is non-null?
213 auto origin = context.securityOrigin();
214
215 // We have already eliminated other types of errors in canEstablishDatabase().
216 // The only reason we're in retryCanEstablishDatabase() is because we gave
217 // the client a chance to update the quota and are rechecking it here.
218 // If we fail this check, the only possible reason this time should be due
219 // to inadequate quota.
220 auto result = hasAdequateQuotaForOrigin(origin, estimatedSize);
221 if (!result.hasException())
222 return { };
223
224 auto exception = result.releaseException();
225 ASSERT(exception.code() == QuotaExceededError);
226 doneCreatingDatabase(origin, name);
227
228 return exception;
229}

231bool DatabaseTracker::hasEntryForOriginNoLock(const SecurityOriginData& origin)
232{
233 ASSERT(m_databaseGuard.isHeld());
234 openTrackerDatabase(DontCreateIfDoesNotExist);
235 if (!m_database.isOpen())
236 return false;
237
238 auto statement = m_database.prepareStatement("SELECT origin FROM Origins where origin=?;"_s);
239 if (!statement) {
240 LOG_ERROR("Failed to prepare statement.");
241 return false;
242 }
243
244 statement->bindText(1, origin.databaseIdentifier());
245
246 return statement->step() == SQLITE_ROW;
247}

249bool DatabaseTracker::hasEntryForDatabase(const SecurityOriginData& origin, const String& databaseIdentifier)
250{
251 ASSERT(m_databaseGuard.isHeld());
252 openTrackerDatabase(DontCreateIfDoesNotExist);
253 if (!m_database.isOpen()) {
254 // No "tracker database". Hence, no entry for the database of interest.
255 return false;
256 }
257
258 // We've got a tracker database. Set up a query to ask for the db of interest:
259 auto statement = m_database.prepareStatement("SELECT guid FROM Databases WHERE origin=? AND name=?;"_s);
260
261 if (!statement)
262 return false;
263
264 statement->bindText(1, origin.databaseIdentifier());
265 statement->bindText(2, databaseIdentifier);
266
267 return statement->step() == SQLITE_ROW;
268}

270uint64_t DatabaseTracker::maximumSize(Database& database)
271{
272 // The maximum size for a database is the full quota for its origin, minus the current usage within the origin,
273 // plus the current usage of the given database
274 Locker lockDatabase { m_databaseGuard };
275 auto origin = database.securityOrigin();
276
277 auto quota = quotaNoLock(origin);
278 auto diskUsage = usage(origin);
279 auto databaseFileSize = SQLiteFileSystem::databaseFileSize(database.fileNameIsolatedCopy());
280 ASSERT(databaseFileSize <= diskUsage);
281
282 if (diskUsage > quota)
283 return databaseFileSize;
284
285 // A previous error may have allowed the origin to exceed its quota, or may
286 // have allowed this database to exceed our cached estimate of the origin
287 // disk usage. Don't multiply that error through integer underflow, or the
288 // effective quota will permanently become 2^64.
289 uint64_t maxSize = quota - diskUsage + databaseFileSize;
290 if (maxSize > quota)
291 maxSize = databaseFileSize;
292 return maxSize;
293}

314String DatabaseTracker::fullPathForDatabaseNoLock(const SecurityOriginData& origin, const String& name, bool createIfNotExists)
315{
316 ASSERT(m_databaseGuard.isHeld());
317
318 String originIdentifier = origin.databaseIdentifier();
319 String originPath = this->originPath(origin);
320
321 // Make sure the path for this SecurityOrigin exists
322 if (createIfNotExists && !SQLiteFileSystem::ensureDatabaseDirectoryExists(originPath))
323 return String();
324
325 // See if we have a path for this database yet
326 if (!m_database.isOpen())
327 return String();
328
329 {
330 auto statement = m_database.prepareStatement("SELECT path FROM Databases WHERE origin=? AND name=?;"_s);
331 if (!statement)
332 return String();
333
334 statement->bindText(1, originIdentifier);
335 statement->bindText(2, name);
336
337 int result = statement->step();
338 if (result == SQLITE_ROW)
339 return SQLiteFileSystem::appendDatabaseFileNameToPath(originPath, statement->columnText(0));
340 if (!createIfNotExists)
341 return String();
342
343 if (result != SQLITE_DONE) {
344 LOG_ERROR("Failed to retrieve filename from Database Tracker for origin %s, name %s", originIdentifier.utf8().data(), name.utf8().data());
345 return String();
346 }
347 }
348
349 String fileName = generateDatabaseFileName();
350
351 if (!addDatabase(origin, name, fileName))
352 return String();
353
354 // If this origin's quota is being tracked (open handle to a database in this origin), add this new database
355 // to the quota manager now
356 String fullFilePath = SQLiteFileSystem::appendDatabaseFileNameToPath(originPath, fileName);
357
358 return fullFilePath;
359}

361String DatabaseTracker::fullPathForDatabase(const SecurityOriginData& origin, const String& name, bool createIfNotExists)
362{
363 Locker lockDatabase { m_databaseGuard };
364 return fullPathForDatabaseNoLock(origin, name, createIfNotExists).isolatedCopy();

367Vector<SecurityOriginData> DatabaseTracker::origins()
368{
369 Locker lockDatabase { m_databaseGuard };
370
371 openTrackerDatabase(DontCreateIfDoesNotExist);
372 if (!m_database.isOpen())
373 return { };
374
375 auto statement = m_database.prepareStatement("SELECT origin FROM Origins"_s);
376 if (!statement) {
377 LOG_ERROR("Failed to prepare statement.");
378 return { };
379 }
380
381 Vector<SecurityOriginData> origins;
382 int stepResult;
383 while ((stepResult = statement->step()) == SQLITE_ROW)
384 origins.append(SecurityOriginData::fromDatabaseIdentifier(statement->columnText(0))->isolatedCopy());
385 origins.shrinkToFit();
386
387 if (stepResult != SQLITE_DONE)
388 LOG_ERROR("Failed to read in all origins from the database.");
389
390 return origins;
391}

430DatabaseDetails DatabaseTracker::detailsForNameAndOrigin(const String& name, const SecurityOriginData& origin)
431{
432 String originIdentifier = origin.databaseIdentifier();
433 String displayName;
434 int64_t expectedUsage;
435
436 {
437 Locker lockDatabase { m_databaseGuard };
438
439 openTrackerDatabase(DontCreateIfDoesNotExist);
440 if (!m_database.isOpen())
441 return DatabaseDetails();
442 auto statement = m_database.prepareStatement("SELECT displayName, estimatedSize FROM Databases WHERE origin=? AND name=?"_s);
443 if (!statement)
444 return DatabaseDetails();
445
446 statement->bindText(1, originIdentifier);
447 statement->bindText(2, name);
448
449 int result = statement->step();
450 if (result == SQLITE_DONE)
451 return DatabaseDetails();
452
453 if (result != SQLITE_ROW) {
454 LOG_ERROR("Error retrieving details for database %s in origin %s from tracker database", name.utf8().data(), originIdentifier.utf8().data());
455 return DatabaseDetails();
456 }
457 displayName = statement->columnText(0);
458 expectedUsage = statement->columnInt64(1);
459 }
460
461 String path = fullPathForDatabase(origin, name, false);
462 if (path.isEmpty())
463 return DatabaseDetails(name, displayName, expectedUsage, 0, std::nullopt, std::nullopt);
464 return DatabaseDetails(name, displayName, expectedUsage, SQLiteFileSystem::databaseFileSize(path), SQLiteFileSystem::databaseCreationTime(path), SQLiteFileSystem::databaseModificationTime(path));
465}

467void DatabaseTracker::setDatabaseDetails(const SecurityOriginData& origin, const String& name, const String& displayName, uint64_t estimatedSize)
468{
469 String originIdentifier = origin.databaseIdentifier();
470 int64_t guid = 0;
471
472 Locker lockDatabase { m_databaseGuard };
473
474 openTrackerDatabase(CreateIfDoesNotExist);
475 if (!m_database.isOpen())
476 return;
477
478 {
479 auto statement = m_database.prepareStatement("SELECT guid FROM Databases WHERE origin=? AND name=?"_s);
480 if (!statement)
481 return;
482
483 statement->bindText(1, originIdentifier);
484 statement->bindText(2, name);
485
486 int result = statement->step();
487 if (result == SQLITE_ROW)
488 guid = statement->columnInt64(0);
489
490 if (!guid) {
491 if (result != SQLITE_DONE)
492 LOG_ERROR("Error to determing existence of database %s in origin %s in tracker database", name.utf8().data(), originIdentifier.utf8().data());
493 else {
494 // This case should never occur - we should never be setting database details for a database that doesn't already exist in the tracker
495 // But since the tracker file is an external resource not under complete control of our code, it's somewhat invalid to make this an ASSERT case
496 // So we'll print an error instead
497 LOG_ERROR("Could not retrieve guid for database %s in origin %s from the tracker database - it is invalid to set database details on a database that doesn't already exist in the tracker", name.utf8().data(), originIdentifier.utf8().data());
498 }
499 return;
500 }
501 }
502
503 auto updateStatement = m_database.prepareStatement("UPDATE Databases SET displayName=?, estimatedSize=? WHERE guid=?"_s);
504 if (!updateStatement)
505 return;
506
507 updateStatement->bindText(1, displayName);
508 updateStatement->bindInt64(2, estimatedSize);
509 updateStatement->bindInt64(3, guid);
510
511 if (updateStatement->step() != SQLITE_DONE) {
512 LOG_ERROR("Failed to update details for database %s in origin %s", name.utf8().data(), originIdentifier.utf8().data());
513 return;
514 }
515
516 if (m_client)
517 m_client->dispatchDidModifyDatabase(origin, name);
518}

544void DatabaseTracker::addOpenDatabase(Database& database)
545{
546 Locker openDatabaseMapLock { m_openDatabaseMapGuard };
547
548 if (!m_openDatabaseMap)
549 m_openDatabaseMap = makeUnique<DatabaseOriginMap>();
550
551 auto origin = database.securityOrigin();
552
553 auto* nameMap = m_openDatabaseMap->get(origin);
554 if (!nameMap) {
555 nameMap = new DatabaseNameMap;
556 m_openDatabaseMap->add(origin.isolatedCopy(), nameMap);
557 }
558
559 String name = database.stringIdentifierIsolatedCopy();
560 auto* databaseSet = nameMap->get(name);
561 if (!databaseSet) {
562 databaseSet = new DatabaseSet;
563 nameMap->set(name.isolatedCopy(), databaseSet);
564 }
565
566 databaseSet->add(&database);
567
568 LOG(StorageAPI, "Added open Database %s (%p)\n", database.stringIdentifierIsolatedCopy().utf8().data(), &database);
569}

571void DatabaseTracker::removeOpenDatabase(Database& database)
572{
573 Locker openDatabaseMapLock { m_openDatabaseMapGuard };
574
575 if (!m_openDatabaseMap) {
576 ASSERT_NOT_REACHED();
577 return;
578 }
579
580 DatabaseNameMap* nameMap = m_openDatabaseMap->get(database.securityOrigin());
581 if (!nameMap) {
582 ASSERT_NOT_REACHED();
583 return;
584 }
585
586 String name = database.stringIdentifierIsolatedCopy();
587 auto* databaseSet = nameMap->get(name);
588 if (!databaseSet) {
589 ASSERT_NOT_REACHED();
590 return;
591 }
592
593 databaseSet->remove(&database);
594
595 LOG(StorageAPI, "Removed open Database %s (%p)\n", database.stringIdentifierIsolatedCopy().utf8().data(), &database);
596
597 if (!databaseSet->isEmpty())
598 return;
599
600 nameMap->remove(name);
601 delete databaseSet;
602
603 if (!nameMap->isEmpty())
604 return;
605
606 m_openDatabaseMap->remove(database.securityOrigin());
607 delete nameMap;
608}

610RefPtr<OriginLock> DatabaseTracker::originLockFor(const SecurityOriginData& origin)
611{
612 Locker lockDatabase { m_databaseGuard };
613 String databaseIdentifier = origin.databaseIdentifier();
614
615 // The originLockMap is accessed from multiple DatabaseThreads since
616 // different script contexts can be writing to different databases from
617 // the same origin. Hence, the databaseIdentifier key needs to be an
618 // isolated copy. An isolated copy gives us a value whose refCounting is
619 // thread-safe, since our copy is guarded by the m_databaseGuard mutex.
620 databaseIdentifier = databaseIdentifier.isolatedCopy();
621
622 OriginLockMap::AddResult addResult =
623 m_originLockMap.add(databaseIdentifier, RefPtr<OriginLock>());
624 if (!addResult.isNewEntry)
625 return addResult.iterator->value;
626
627 String path = originPath(origin);
628 RefPtr<OriginLock> lock = adoptRef(*new OriginLock(path));
629 ASSERT(lock);
630 addResult.iterator->value = lock;
631
632 return lock;
633}

635void DatabaseTracker::deleteOriginLockFor(const SecurityOriginData& origin)
636{
637 ASSERT(m_databaseGuard.isHeld());
638
639 // There is not always an instance of an OriginLock associated with an origin.
640 // For example, if the OriginLock lock file was created by a previous run of
641 // the browser which has now terminated, and the current browser process
642 // has not executed any database transactions from this origin that would
643 // have created the OriginLock instance in memory. In this case, we will have
644 // a lock file but not an OriginLock instance in memory.
645
646 // This function is only called if we are already deleting all the database
647 // files in this origin. We'll give the OriginLock one chance to do an
648 // orderly clean up first when we remove its ref from the m_originLockMap.
649 // This may or may not be possible depending on whether other threads are
650 // also using the OriginLock at the same time. After that, we will delete the lock file.
651
652 m_originLockMap.remove(origin.databaseIdentifier());
653 OriginLock::deleteLockFile(originPath(origin));
654}

656uint64_t DatabaseTracker::usage(const SecurityOriginData& origin)
657{
658 String originPath = this->originPath(origin);
659 uint64_t diskUsage = 0;
660 for (auto& fileName : FileSystem::listDirectory(originPath)) {
661 if (fileName.endsWith(".db"))
662 diskUsage += SQLiteFileSystem::databaseFileSize(FileSystem::pathByAppendingComponent(originPath, fileName));
663 }
664 return diskUsage;

667uint64_t DatabaseTracker::quotaNoLock(const SecurityOriginData& origin)
668{
669 ASSERT(m_databaseGuard.isHeld());
670 uint64_t quota = 0;
671
672 openTrackerDatabase(DontCreateIfDoesNotExist);
673 if (!m_database.isOpen())
674 return quota;
675
676 auto statement = m_database.prepareStatement("SELECT quota FROM Origins where origin=?;"_s);
677 if (!statement) {
678 LOG_ERROR("Failed to prepare statement.");
679 return quota;
680 }
681 statement->bindText(1, origin.databaseIdentifier());
682
683 if (statement->step() == SQLITE_ROW)
684 quota = statement->columnInt64(0);
685
686 return quota;
687}

694
695void DatabaseTracker::setQuota(const SecurityOriginData& origin, uint64_t quota)
696{
697 Locker lockDatabase { m_databaseGuard };
698
699 if (quotaNoLock(origin) == quota)
700 return;
701
702 openTrackerDatabase(CreateIfDoesNotExist);
703 if (!m_database.isOpen())
704 return;
705
706 bool insertedNewOrigin = false;
707
708 bool originEntryExists = hasEntryForOriginNoLock(origin);
709 if (!originEntryExists) {
710 auto statement = m_database.prepareStatement("INSERT INTO Origins VALUES (?, ?)"_s);
711 if (!statement) {
712 LOG_ERROR("Unable to establish origin %s in the tracker", origin.databaseIdentifier().utf8().data());
713 } else {
714 statement->bindText(1, origin.databaseIdentifier());
715 statement->bindInt64(2, quota);
716
717 if (statement->step() != SQLITE_DONE)
718 LOG_ERROR("Unable to establish origin %s in the tracker", origin.databaseIdentifier().utf8().data());
719 else
720 insertedNewOrigin = true;
721 }
722 } else {
723 auto statement = m_database.prepareStatement("UPDATE Origins SET quota=? WHERE origin=?"_s);
724 bool error = !statement;
725 if (!error) {
726 statement->bindInt64(1, quota);
727 statement->bindText(2, origin.databaseIdentifier());
728
729 error = !statement->executeCommand();
730 }
731
732 if (error)
733 LOG_ERROR("Failed to set quota %llu in tracker database for origin %s", quota, origin.databaseIdentifier().utf8().data());
734 }
735
736 if (m_client) {
737 if (insertedNewOrigin)
738 m_client->dispatchDidAddNewOrigin();
739 m_client->dispatchDidModifyOrigin(origin);
740 }
741}

783void DatabaseTracker::deleteDatabasesModifiedSince(WallTime time)
784{
785 for (auto& origin : origins()) {
786 Vector<String> databaseNames = this->databaseNames(origin);
787 Vector<String> databaseNamesToDelete;
788 databaseNamesToDelete.reserveInitialCapacity(databaseNames.size());
789 for (const auto& databaseName : databaseNames) {
790 auto fullPath = fullPathForDatabase(origin, databaseName, false);
791
792 // If the file doesn't exist, we previously deleted it but failed to remove the information
793 // from the tracker database. We want to delete all of the information associated with this
794 // database from the tracker database, so still add its name to databaseNamesToDelete.
795 if (FileSystem::fileExists(fullPath)) {
796 auto modificationTime = FileSystem::fileModificationTime(fullPath);
797 if (!modificationTime)
798 continue;
799
800 if (modificationTime.value() < time)
801 continue;
802 }
803
804 databaseNamesToDelete.uncheckedAppend(databaseName);
805 }
806
807 if (databaseNames.size() == databaseNamesToDelete.size())
808 deleteOrigin(origin);
809 else {
810 for (const auto& databaseName : databaseNamesToDelete)
811 deleteDatabase(origin, databaseName);
812 }
813 }
814}

823bool DatabaseTracker::deleteOrigin(const SecurityOriginData& origin, DeletionMode deletionMode)
824{
825 Vector<String> databaseNames;
826 {
827 Locker lockDatabase { m_databaseGuard };
828 openTrackerDatabase(DontCreateIfDoesNotExist);
829 if (!m_database.isOpen())
830 return false;
831
832 databaseNames = databaseNamesNoLock(origin);
833 if (databaseNames.isEmpty())
834 LOG_ERROR("Unable to retrieve list of database names for origin %s", origin.databaseIdentifier().utf8().data());
835
836 if (!canDeleteOrigin(origin)) {
837 LOG_ERROR("Tried to delete an origin (%s) while either creating database in it or already deleting it", origin.databaseIdentifier().utf8().data());
838 ASSERT_NOT_REACHED();
839 return false;
840 }
841 recordDeletingOrigin(origin);
842 }
843
844 // We drop the lock here because holding locks during a call to deleteDatabaseFile will deadlock.
845 bool failedToDeleteAnyDatabaseFile = false;
846 for (auto& name : databaseNames) {
847 if (FileSystem::fileExists(fullPathForDatabase(origin, name, false)) && !deleteDatabaseFile(origin, name, deletionMode)) {
848 // Even if the file can't be deleted, we want to try and delete the rest, don't return early here.
849 LOG_ERROR("Unable to delete file for database %s in origin %s", name.utf8().data(), origin.databaseIdentifier().utf8().data());
850 failedToDeleteAnyDatabaseFile = true;
851 }
852 }
853
854 // If databaseNames is empty, delete everything in the directory containing the databases for this origin.
855 // This condition indicates that we previously tried to remove the origin but didn't get all of the way
856 // through the deletion process. Because we have lost track of the databases for this origin,
857 // we can assume that no other process is accessing them. This means it should be safe to delete them outright.
858 if (databaseNames.isEmpty()) {
859#if PLATFORM(COCOA)
860 RELEASE_LOG_ERROR(DatabaseTracker, "Unable to retrieve list of database names for origin");
861#endif
862 auto originPath = this->originPath(origin);
863 for (const auto& fileName : FileSystem::listDirectory(originPath)) {
864 if (!FileSystem::deleteFile(FileSystem::pathByAppendingComponent(originPath, fileName)))
865 failedToDeleteAnyDatabaseFile = true;
866 }
867 }
868
869 // If we failed to delete any database file, don't remove the origin from the tracker
870 // database because we didn't successfully remove all of its data.
871 if (failedToDeleteAnyDatabaseFile) {
872#if PLATFORM(COCOA)
873 RELEASE_LOG_ERROR(DatabaseTracker, "Failed to delete database for origin");
874#endif
875 return false;
876 }
877
878 {
879 Locker lockDatabase { m_databaseGuard };
880 deleteOriginLockFor(origin);
881 doneDeletingOrigin(origin);
882
883 SQLiteTransaction transaction(m_database);
884 transaction.begin();
885
886 {
887 auto statement = m_database.prepareStatement("DELETE FROM Databases WHERE origin=?"_s);
888 if (!statement) {
889 LOG_ERROR("Unable to prepare deletion of databases from origin %s from tracker", origin.databaseIdentifier().utf8().data());
890 return false;
891 }
892
893 statement->bindText(1, origin.databaseIdentifier());
894
895 if (!statement->executeCommand()) {
896 LOG_ERROR("Unable to execute deletion of databases from origin %s from tracker", origin.databaseIdentifier().utf8().data());
897 return false;
898 }
899
900 auto originStatement = m_database.prepareStatement("DELETE FROM Origins WHERE origin=?"_s);
901 if (!originStatement) {
902 LOG_ERROR("Unable to prepare deletion of origin %s from tracker", origin.databaseIdentifier().utf8().data());
903 return false;
904 }
905
906 originStatement->bindText(1, origin.databaseIdentifier());
907
908 if (!originStatement->executeCommand()) {
909 LOG_ERROR("Unable to execute deletion of databases from origin %s from tracker", origin.databaseIdentifier().utf8().data());
910 return false;
911 }
912 }
913
914 transaction.commit();
915
916 SQLiteFileSystem::deleteEmptyDatabaseDirectory(originPath(origin));
917
918 bool isEmpty = true;
919
920 openTrackerDatabase(DontCreateIfDoesNotExist);
921 if (m_database.isOpen()) {
922 auto statement = m_database.prepareStatement("SELECT origin FROM Origins"_s);
923 if (!statement)
924 LOG_ERROR("Failed to prepare statement.");
925 else if (statement->step() == SQLITE_ROW)
926 isEmpty = false;
927 }
928
929 // If we removed the last origin, do some additional deletion.
930 if (isEmpty) {
931 if (m_database.isOpen())
932 m_database.close();
933 SQLiteFileSystem::deleteDatabaseFile(trackerDatabasePath());
934 SQLiteFileSystem::deleteEmptyDatabaseDirectory(m_databaseDirectoryPath);
935 }
936
937 if (m_client) {
938 m_client->dispatchDidModifyOrigin(origin);
939 m_client->dispatchDidDeleteDatabaseOrigin();
940 for (auto& name : databaseNames)
941 m_client->dispatchDidModifyDatabase(origin, name);
942 }
943 }
944 return true;
945}

1079 if (FileSystem::fileExists(fullPathForDatabase(origin, name, false)) && !deleteDatabaseFile(origin, name, DeletionMode::Default)) {
1080 LOG_ERROR("Unable to delete file for database %s in origin %s", name.utf8().data(), origin.databaseIdentifier().utf8().data());
1081 Locker lockDatabase { m_databaseGuard };
1082 doneDeletingDatabase(origin, name);
1083 return false;
1084 }
1085
1086 Locker lockDatabase { m_databaseGuard };
1087
1088 auto statement = m_database.prepareStatement("DELETE FROM Databases WHERE origin=? AND name=?"_s);
1089 if (!statement) {
1090 LOG_ERROR("Unable to prepare deletion of database %s from origin %s from tracker", name.utf8().data(), origin.databaseIdentifier().utf8().data());
1091 doneDeletingDatabase(origin, name);
1092 return false;
1093 }
1094
1095 statement->bindText(1, origin.databaseIdentifier());
1096 statement->bindText(2, name);
1097
1098 if (!statement->executeCommand()) {
1099 LOG_ERROR("Unable to execute deletion of database %s from origin %s from tracker", name.utf8().data(), origin.databaseIdentifier().utf8().data());
1100 doneDeletingDatabase(origin, name);
1101 return false;
1102 }
1103
1104 if (m_client) {
1105 m_client->dispatchDidModifyOrigin(origin);
1106 m_client->dispatchDidModifyDatabase(origin, name);
1107 m_client->dispatchDidDeleteDatabase();

1114// deleteDatabaseFile has to release locks between looking up the list of databases to close and closing them. While this is in progress, the caller
1115// is responsible for making sure no new databases are opened in the file to be deleted.
1116bool DatabaseTracker::deleteDatabaseFile(const SecurityOriginData& origin, const String& name, DeletionMode deletionMode)
1117{
1118 String fullPath = fullPathForDatabase(origin, name, false);
1119 if (fullPath.isEmpty())
1120 return true;
1121
1122#ifndef NDEBUG
1123 {
1124 Locker lockDatabase { m_databaseGuard };
1125 ASSERT(isDeletingDatabaseOrOriginFor(origin, name));
1126 }
1127#endif
1128
1129 Vector<Ref<Database>> deletedDatabases;
1130
1131 // Make sure not to hold the any locks when calling
1132 // Database::markAsDeletedAndClose(), since that can cause a deadlock
1133 // during the synchronous DatabaseThread call it triggers.
1134 {
1135 Locker openDatabaseMapLock { m_openDatabaseMapGuard };
1136 if (m_openDatabaseMap) {
1137 if (auto* nameMap = m_openDatabaseMap->get(origin)) {
1138 if (auto* databaseSet = nameMap->get(name)) {
1139 for (auto& database : *databaseSet)
1140 deletedDatabases.append(*database);
1141 }
1142 }
1143 }
1144 }
1145
1146 for (auto& database : deletedDatabases)
1147 database->markAsDeletedAndClose();
1148
1149#if PLATFORM(IOS_FAMILY)
1150 if (deletionMode == DeletionMode::Deferred) {
1151 // Other background processes may still be accessing this database. Deleting the database directly
1152 // would nuke the POSIX file locks, potentially causing Safari/WebApp to corrupt the new db if it's running in the background.
1153 // We'll instead truncate the database file to 0 bytes. If another process is operating on this same database file after
1154 // the truncation, it should get an error since the database file is no longer valid. When Safari is launched
1155 // next time, it'll go through the database files and clean up any zero-bytes ones.
1156 SQLiteDatabase database;
1157 if (!database.open(fullPath))
1158 return false;
1159 return SQLiteFileSystem::truncateDatabaseFile(database.sqlite3Handle());
1160 }
1161#else
1162 UNUSED_PARAM(deletionMode);
1163#endif
1164
1165 return SQLiteFileSystem::deleteDatabaseFile(fullPath);
1166}

1170// FIXME: This uses m_database without locking m_databaseGuard.
1171void DatabaseTracker::removeDeletedOpenedDatabases() WTF_IGNORES_THREAD_SAFETY_ANALYSIS
1172{
1173 // This is called when another app has deleted a database. Go through all opened databases in this
1174 // tracker and close any that's no longer being tracked in the database.
1175
1176 {
1177 // Acquire the lock before calling openTrackerDatabase.
1178 Locker lockDatabase { m_databaseGuard };
1179 openTrackerDatabase(DontCreateIfDoesNotExist);
1180
1181 if (!m_database.isOpen())
1182 return;
1183 }
1184
1185 // Keep track of which opened databases have been deleted.
1186 Vector<RefPtr<Database>> deletedDatabases;
1187 Vector<std::pair<SecurityOriginData, Vector<String>>> deletedDatabaseNames;
1188
1189 // Make sure not to hold the m_openDatabaseMapGuard mutex when calling
1190 // Database::markAsDeletedAndClose(), since that can cause a deadlock
1191 // during the synchronous DatabaseThread call it triggers.
1192 {
1193 Locker openDatabaseMapLock { m_openDatabaseMapGuard };
1194 if (m_openDatabaseMap) {
1195 for (auto& openDatabase : *m_openDatabaseMap) {
1196 auto& origin = openDatabase.key;
1197 DatabaseNameMap* databaseNameMap = openDatabase.value;
1198 Vector<String> deletedDatabaseNamesForThisOrigin;
1199
1200 // Loop through all opened databases in this origin. Get the current database file path of each database and see if
1201 // it still matches the path stored in the opened database object.
1202 for (auto& databases : *databaseNameMap) {
1203 String databaseName = databases.key;
1204 String databaseFileName;
1205 if (auto statement = m_database.prepareStatement("SELECT path FROM Databases WHERE origin=? AND name=?;"_s)) {
1206 statement->bindText(1, origin.databaseIdentifier());
1207 statement->bindText(2, databaseName);
1208 if (statement->step() == SQLITE_ROW)
1209 databaseFileName = statement->columnText(0);
1210 }
1211
1212 bool foundDeletedDatabase = false;
1213 for (auto& db : *databases.value) {
1214 // We are done if this database has already been marked as deleted.
1215 if (db->deleted())
1216 continue;
1217
1218 // If this database has been deleted or if its database file no longer matches the current version, this database is no longer valid and it should be marked as deleted.
1219 if (databaseFileName.isNull() || databaseFileName != FileSystem::pathFileName(db->fileNameIsolatedCopy())) {
1220 deletedDatabases.append(db);
1221 foundDeletedDatabase = true;
1222 }
1223 }
1224
1225 // If the database no longer exists, we should remember to send that information to the client later.
1226 if (m_client && foundDeletedDatabase && databaseFileName.isNull())
1227 deletedDatabaseNamesForThisOrigin.append(databaseName);
1228 }
1229
1230 if (!deletedDatabaseNamesForThisOrigin.isEmpty())
1231 deletedDatabaseNames.append({ origin, WTFMove(deletedDatabaseNamesForThisOrigin) });
1232 }
1233 }
1234 }
1235
1236 for (auto& deletedDatabase : deletedDatabases)
1237 deletedDatabase->markAsDeletedAndClose();

1239 for (auto& deletedDatabase : deletedDatabaseNames) {
1240 auto& origin = deletedDatabase.first;
1241 m_client->dispatchDidModifyOrigin(origin);
1242 for (auto& databaseName : deletedDatabase.second)
1243 m_client->dispatchDidModifyDatabase(origin, databaseName);

1248{
1249 auto size = FileSystem::fileSize(path);

1253bool DatabaseTracker::deleteDatabaseFileIfEmpty(const String& path)
1254{
1255 if (!isZeroByteFile(path))
1256 return false;
1257
1258 SQLiteDatabase database;
1259 if (!database.open(path))
1260 return false;
1261
1262 // Specify that we want the exclusive locking mode, so after the next write,
1263 // we'll be holding the lock to this database file.
1264 {
1265 auto lockStatement = database.prepareStatement("PRAGMA locking_mode=EXCLUSIVE;"_s);
1266 if (!lockStatement)
1267 return false;
1268 int result = lockStatement->step();
1269 if (result != SQLITE_ROW && result != SQLITE_DONE)
1270 return false;
1271 }
1272
1273 if (!database.executeCommand("BEGIN EXCLUSIVE TRANSACTION;"_s))
1274 return false;
1275
1276 // At this point, we hold the exclusive lock to this file.
1277 // Check that the database doesn't contain any tables.
1278 if (!database.executeCommand("SELECT name FROM sqlite_master WHERE type='table';"_s))
1279 return false;
1280
1281 database.executeCommand("COMMIT TRANSACTION;"_s);
1282
1283 database.close();
1284
1285 return SQLiteFileSystem::deleteDatabaseFile(path);
1286}
