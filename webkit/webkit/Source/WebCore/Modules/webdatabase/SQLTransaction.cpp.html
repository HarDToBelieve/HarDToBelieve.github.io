75ExceptionOr<void> SQLTransaction::executeSql(const String& sqlStatement, std::optional<Vector<SQLValue>>&& arguments, RefPtr<SQLStatementCallback>&& callback, RefPtr<SQLStatementErrorCallback>&& callbackError)
76{
77 if (!m_executeSqlAllowed || !m_database->opened())
78 return Exception { InvalidStateError };
79
80 int permissions = DatabaseAuthorizer::ReadWriteMask;
81 if (!m_database->databaseContext().allowDatabaseAccess())
82 permissions |= DatabaseAuthorizer::NoAccessMask;
83 else if (m_readOnly)
84 permissions |= DatabaseAuthorizer::ReadOnlyMask;
85
86 auto statement = makeUnique<SQLStatement>(m_database, sqlStatement, arguments.value_or(Vector<SQLValue> { }), WTFMove(callback), WTFMove(callbackError), permissions);
87
88 if (m_database->deleted())
89 statement->setDatabaseDeletedError();
90
91 enqueueStatement(WTFMove(statement));
92
93 return { };
94}

110void SQLTransaction::performPendingCallback()
111{
112 ASSERT(isMainThread());
113 LOG(StorageAPI, "Callback %s\n", debugStepName(m_nextStep));
114
115 ASSERT(m_nextStep == &SQLTransaction::deliverTransactionCallback
116 || m_nextStep == &SQLTransaction::deliverTransactionErrorCallback
117 || m_nextStep == &SQLTransaction::deliverStatementCallback
118 || m_nextStep == &SQLTransaction::deliverQuotaIncreaseCallback
119 || m_nextStep == &SQLTransaction::deliverSuccessCallback);
120
121 checkAndHandleClosedDatabase();
122
123 if (m_nextStep)
124 (this->*m_nextStep)();
125}

150SQLTransaction::StateFunction SQLTransaction::stateFunctionFor(SQLTransactionState state)
151{
152 static const StateFunction stateFunctions[] = {
153 &SQLTransaction::unreachableState, // 0. illegal
154 &SQLTransaction::unreachableState, // 1. idle
155 &SQLTransaction::unreachableState, // 2. acquireLock
156 &SQLTransaction::unreachableState, // 3. openTransactionAndPreflight
157 &SQLTransaction::unreachableState, // 4. runStatements
158 &SQLTransaction::unreachableState, // 5. postflightAndCommit
159 &SQLTransaction::unreachableState, // 6. cleanupAndTerminate
160 &SQLTransaction::unreachableState, // 7. cleanupAfterTransactionErrorCallback
161 &SQLTransaction::deliverTransactionCallback, // 8.
162 &SQLTransaction::deliverTransactionErrorCallback, // 9.
163 &SQLTransaction::deliverStatementCallback, // 10.
164 &SQLTransaction::deliverQuotaIncreaseCallback, // 11.
165 &SQLTransaction::deliverSuccessCallback // 12.
166 };
167
168 ASSERT(WTF_ARRAY_LENGTH(stateFunctions) == static_cast<int>(SQLTransactionState::NumberOfStates));
169 ASSERT(state < SQLTransactionState::NumberOfStates);
170
171 return stateFunctions[static_cast<int>(state)];
172}

184void SQLTransaction::checkAndHandleClosedDatabase()
185{
186 if (m_database->opened())
187 return;
188
189 // If the database was stopped, don't do anything and cancel queued work
190 LOG(StorageAPI, "Database was stopped or interrupted - cancelling work for this transaction");
191
192 Locker locker { m_statementLock };
193 m_statementQueue.clear();
194 m_nextStep = nullptr;
195
196 callErrorCallbackDueToInterruption();
197
198 // Release the unneeded callbacks, to break reference cycles.
199 m_callbackWrapper.clear();
200 m_successCallbackWrapper.clear();
201 m_errorCallbackWrapper.clear();
202
203 // The next steps should be executed only if we're on the DB thread.
204 if (m_database->databaseThread().getThread() != &Thread::current())
205 return;
206
207 // The current SQLite transaction should be stopped, as well
208 if (m_sqliteTransaction) {
209 m_sqliteTransaction->stop();
210 m_sqliteTransaction = nullptr;
211 }
212
213 if (m_lockAcquired)
214 m_database->transactionCoordinator()->releaseLock(*this);
215}

230void SQLTransaction::openTransactionAndPreflight()
231{
232 ASSERT(!m_database->sqliteDatabase().transactionInProgress());
233 ASSERT(m_lockAcquired);
234
235 LOG(StorageAPI, "Opening and preflighting transaction %p", this);
236
237 // If the database was deleted, jump to the error callback
238 if (m_database->deleted()) {
239 m_transactionError = SQLError::create(SQLError::UNKNOWN_ERR, "unable to open a transaction, because the user deleted the database");
240
241 handleTransactionError();
242 return;
243 }
244
245 // Set the maximum usage for this transaction if this transactions is not read-only
246 if (!m_readOnly) {
247 acquireOriginLock();
248 m_database->sqliteDatabase().setMaximumSize(m_database->maximumSize());
249 }
250
251 ASSERT(!m_sqliteTransaction);
252 m_sqliteTransaction = makeUnique<SQLiteTransaction>(m_database->sqliteDatabase(), m_readOnly);
253
254 m_database->resetDeletes();
255 m_database->disableAuthorizer();
256 m_sqliteTransaction->begin();
257 m_database->enableAuthorizer();
258
259 // Spec 4.3.2.1+2: Open a transaction to the database, jumping to the error callback if that fails
260 if (!m_sqliteTransaction->inProgress()) {
261 ASSERT(!m_database->sqliteDatabase().transactionInProgress());
262 m_transactionError = SQLError::create(SQLError::DATABASE_ERR, "unable to begin transaction", m_database->sqliteDatabase().lastError(), m_database->sqliteDatabase().lastErrorMsg());
263 m_sqliteTransaction = nullptr;
264
265 handleTransactionError();
266 return;
267 }
268
269 // Note: We intentionally retrieve the actual version even with an empty expected version.
270 // In multi-process browsers, we take this opportinutiy to update the cached value for
271 // the actual version. In single-process browsers, this is just a map lookup.
272 String actualVersion;
273 if (!m_database->getActualVersionForTransaction(actualVersion)) {
274 m_transactionError = SQLError::create(SQLError::DATABASE_ERR, "unable to read version", m_database->sqliteDatabase().lastError(), m_database->sqliteDatabase().lastErrorMsg());
275 m_database->disableAuthorizer();
276 m_sqliteTransaction = nullptr;
277 m_database->enableAuthorizer();
278
279 handleTransactionError();
280 return;
281 }
282
283 auto expectedVersion = m_database->expectedVersionIsolatedCopy();
284 m_hasVersionMismatch = !expectedVersion.isEmpty() && expectedVersion != actualVersion;
285
286 // Spec 4.3.2.3: Perform preflight steps, jumping to the error callback if they fail
287 if (m_wrapper && !m_wrapper->performPreflight(*this)) {
288 m_database->disableAuthorizer();
289 m_sqliteTransaction = nullptr;
290 m_database->enableAuthorizer();
291 m_transactionError = m_wrapper->sqlError();
292 if (!m_transactionError)
293 m_transactionError = SQLError::create(SQLError::UNKNOWN_ERR, "unknown error occurred during transaction preflight");
294
295 handleTransactionError();
296 return;
297 }
298
299 // Spec 4.3.2.4: Invoke the transaction callback with the new SQLTransaction object
300 if (m_callbackWrapper.hasCallback()) {
301 scheduleCallback(&SQLTransaction::deliverTransactionCallback);
302 return;
303 }
304
305 // If we have no callback to make, skip pass to the state after:
306 runStatements();
307}

309void SQLTransaction::runStatements()
310{
311 ASSERT(m_lockAcquired);
312
313 // If there is a series of statements queued up that are all successful and have no associated
314 // SQLStatementCallback objects, then we can burn through the queue
315 do {
316 if (m_shouldRetryCurrentStatement && !m_sqliteTransaction->wasRolledBackBySqlite()) {
317 m_shouldRetryCurrentStatement = false;
318 // FIXME - Another place that needs fixing up after <rdar://problem/5628468> is addressed.
319 // See ::openTransactionAndPreflight() for discussion
320
321 // Reset the maximum size here, as it was increased to allow us to retry this statement.
322 // m_shouldRetryCurrentStatement is set to true only when a statement exceeds
323 // the quota, which can happen only in a read-write transaction. Therefore, there
324 // is no need to check here if the transaction is read-write.
325 m_database->sqliteDatabase().setMaximumSize(m_database->maximumSize());
326 } else {
327 // If the current statement has already been run, failed due to quota constraints, and we're not retrying it,
328 // that means it ended in an error. Handle it now
329 if (m_currentStatement && m_currentStatement->lastExecutionFailedDueToQuota()) {
330 handleCurrentStatementError();
331 break;
332 }
333
334 // Otherwise, advance to the next statement
335 getNextStatement();
336 }
337 } while (runCurrentStatement());
338
339 // If runCurrentStatement() returned false, that means either there was no current statement to run,
340 // or the current statement requires a callback to complete. In the later case, it also scheduled
341 // the callback or performed any other additional work so we can return.
342 if (!m_currentStatement)
343 postflightAndCommit();
344}

346void SQLTransaction::cleanupAndTerminate()
347{
348 ASSERT(m_lockAcquired);
349
350 // Spec 4.3.2.9: End transaction steps. There is no next step.
351 LOG(StorageAPI, "Transaction %p is complete\n", this);
352 ASSERT(!m_database->sqliteDatabase().transactionInProgress());
353
354 // Phase 5 cleanup. See comment on the SQLTransaction life-cycle above.
355 m_backend.doCleanup();
356 m_database->inProgressTransactionCompleted();
357}
358
359void SQLTransaction::cleanupAfterTransactionErrorCallback()
360{
361 ASSERT(m_lockAcquired);
362
363 LOG(StorageAPI, "Transaction %p is complete with an error\n", this);
364 m_database->disableAuthorizer();
365 if (m_sqliteTransaction) {
366 // Spec 4.3.2.10: Rollback the transaction.
367 m_sqliteTransaction->rollback();
368
369 ASSERT(!m_database->sqliteDatabase().transactionInProgress());
370 m_sqliteTransaction = nullptr;
371 }
372 m_database->enableAuthorizer();
373
374 releaseOriginLockIfNeeded();
375
376 ASSERT(!m_database->sqliteDatabase().transactionInProgress());
377
378 cleanupAndTerminate();
379}

381void SQLTransaction::deliverTransactionCallback()
382{
383 bool shouldDeliverErrorCallback = false;
384
385 // Spec 4.3.2 4: Invoke the transaction callback with the new SQLTransaction object
386 RefPtr<SQLTransactionCallback> callback = m_callbackWrapper.unwrap();
387 if (callback) {
388 m_executeSqlAllowed = true;
389
390 auto result = callback->handleEvent(*this);
391 shouldDeliverErrorCallback = result.type() == CallbackResultType::ExceptionThrown;
392
393 m_executeSqlAllowed = false;
394 }
395
396 // Spec 4.3.2 5: If the transaction callback was null or raised an exception, jump to the error callback
397 if (shouldDeliverErrorCallback) {
398 m_transactionError = SQLError::create(SQLError::UNKNOWN_ERR, "the SQLTransactionCallback was null or threw an exception");
399 return deliverTransactionErrorCallback();
400 }
401
402 m_backend.requestTransitToState(SQLTransactionState::RunStatements);
403}
404
405void SQLTransaction::deliverTransactionErrorCallback()
406{
407 ASSERT(m_transactionError);
408
409 // Spec 4.3.2.10: If exists, invoke error callback with the last
410 // error to have occurred in this transaction.
411 RefPtr<SQLTransactionErrorCallback> errorCallback = m_errorCallbackWrapper.unwrap();
412 if (errorCallback) {
413 m_database->document().eventLoop().queueTask(TaskSource::Networking, [errorCallback = WTFMove(errorCallback), transactionError = m_transactionError]() mutable {
414 errorCallback->handleEvent(*transactionError);
415 });
416 }
417
418 clearCallbackWrappers();
419
420 // Spec 4.3.2.10: Rollback the transaction.
421 m_backend.requestTransitToState(SQLTransactionState::CleanupAfterTransactionErrorCallback);
422}
423
424void SQLTransaction::deliverStatementCallback()
425{
426 ASSERT(m_currentStatement);
427
428 // Spec 4.3.2.6.6 and 4.3.2.6.3: If the statement callback went wrong, jump to the transaction error callback
429 // Otherwise, continue to loop through the statement queue
430 m_executeSqlAllowed = true;
431 bool result = m_currentStatement->performCallback(*this);
432 m_executeSqlAllowed = false;
433
434 if (result) {
435 m_transactionError = SQLError::create(SQLError::UNKNOWN_ERR, "the statement callback raised an exception or statement error callback did not return false");
436
437 if (m_errorCallbackWrapper.hasCallback())
438 return deliverTransactionErrorCallback();
439
440 // No error callback, so fast-forward to:
441 // Transaction Step 11 - Rollback the transaction.
442 m_backend.requestTransitToState(SQLTransactionState::CleanupAfterTransactionErrorCallback);
443 return;
444 }
445
446 m_backend.requestTransitToState(SQLTransactionState::RunStatements);
447}

449void SQLTransaction::deliverQuotaIncreaseCallback()
450{
451 ASSERT(m_currentStatement);
452 ASSERT(!m_shouldRetryCurrentStatement);
453
454 m_shouldRetryCurrentStatement = m_database->didExceedQuota();
455
456 m_backend.requestTransitToState(SQLTransactionState::RunStatements);
457}

459void SQLTransaction::deliverSuccessCallback()
460{
461 // Spec 4.3.2.8: Deliver success callback.
462 RefPtr<VoidCallback> successCallback = m_successCallbackWrapper.unwrap();
463 if (successCallback) {
464 m_database->document().eventLoop().queueTask(TaskSource::Networking, [successCallback = WTFMove(successCallback)]() mutable {
465 successCallback->handleEvent();
466 });
467 }
468
469 clearCallbackWrappers();
470
471 // Schedule a "post-success callback" step to return control to the database thread in case there
472 // are further transactions queued up for this Database
473 m_backend.requestTransitToState(SQLTransactionState::CleanupAndTerminate);
474}

484void SQLTransaction::computeNextStateAndCleanupIfNeeded()
485{
486 // Only honor the requested state transition if we're not supposed to be
487 // cleaning up and shutting down:
488 if (m_database->opened()) {
489 setStateToRequestedState();
490 ASSERT(m_nextState == SQLTransactionState::End
491 || m_nextState == SQLTransactionState::DeliverTransactionCallback
492 || m_nextState == SQLTransactionState::DeliverTransactionErrorCallback
493 || m_nextState == SQLTransactionState::DeliverStatementCallback
494 || m_nextState == SQLTransactionState::DeliverQuotaIncreaseCallback
495 || m_nextState == SQLTransactionState::DeliverSuccessCallback);
496
497 LOG(StorageAPI, "Callback %s\n", nameForSQLTransactionState(m_nextState));
498 return;
499 } else
500 callErrorCallbackDueToInterruption();
501
502 clearCallbackWrappers();
503 m_backend.requestTransitToState(SQLTransactionState::CleanupAndTerminate);
504}

514void SQLTransaction::getNextStatement()
515{
516 m_currentStatement = nullptr;

520 m_currentStatement = m_statementQueue.takeFirst();
521}
522
523bool SQLTransaction::runCurrentStatement()
524{
525 if (!m_currentStatement) {
526 // No more statements to run. So move on to the next state.
527 return false;
528 }
529
530 m_database->resetAuthorizer();
531
532 if (m_hasVersionMismatch)
533 m_currentStatement->setVersionMismatchedError();
534
535 if (m_currentStatement->execute(m_database)) {
536 if (m_database->lastActionChangedDatabase()) {
537 // Flag this transaction as having changed the database for later delegate notification
538 m_modifiedDatabase = true;
539 }
540
541 if (m_currentStatement->hasStatementCallback()) {
542 scheduleCallback(&SQLTransaction::deliverStatementCallback);
543 return false;
544 }
545
546 // If we get here, then the statement doesn't have a callback to invoke.
547 // We can move on to the next statement. Hence, stay in this state.
548 return true;
549 }
550
551 if (m_currentStatement->lastExecutionFailedDueToQuota()) {
552 scheduleCallback(&SQLTransaction::deliverQuotaIncreaseCallback);
553 return false;
554 }
555
556 handleCurrentStatementError();
557 return false;
558}

560void SQLTransaction::handleCurrentStatementError()
561{
562 // Spec 4.3.2.6.6: error - Call the statement's error callback, but if there was no error callback,
563 // or the transaction was rolled back, jump to the transaction error callback
564 if (m_currentStatement->hasStatementErrorCallback() && !m_sqliteTransaction->wasRolledBackBySqlite()) {
565 scheduleCallback(&SQLTransaction::deliverStatementCallback);
566 return;
567 }
568
569 m_transactionError = m_currentStatement->sqlError();
570 if (!m_transactionError)
571 m_transactionError = SQLError::create(SQLError::DATABASE_ERR, "the statement failed to execute");
572
573 handleTransactionError();
574}
575
576void SQLTransaction::handleTransactionError()
577{
578 ASSERT(m_transactionError);
579 if (m_errorCallbackWrapper.hasCallback()) {
580 scheduleCallback(&SQLTransaction::deliverTransactionErrorCallback);
581 return;
582 }
583
584 // No error callback, so fast-forward to the next state and rollback the
585 // transaction.
586 m_backend.cleanupAfterTransactionErrorCallback();
587}
588
589void SQLTransaction::postflightAndCommit()
590{
591 ASSERT(m_lockAcquired);
592
593 // Spec 4.3.2.7: Perform postflight steps, jumping to the error callback if they fail.
594 if (m_wrapper && !m_wrapper->performPostflight(*this)) {
595 m_transactionError = m_wrapper->sqlError();
596 if (!m_transactionError)
597 m_transactionError = SQLError::create(SQLError::UNKNOWN_ERR, "unknown error occurred during transaction postflight");
598
599 handleTransactionError();
600 return;
601 }
602
603 // Spec 4.3.2.7: Commit the transaction, jumping to the error callback if that fails.
604 ASSERT(m_sqliteTransaction);
605
606 m_database->disableAuthorizer();
607 m_sqliteTransaction->commit();
608 m_database->enableAuthorizer();
609
610 releaseOriginLockIfNeeded();
611
612 // If the commit failed, the transaction will still be marked as "in progress"
613 if (m_sqliteTransaction->inProgress()) {
614 if (m_wrapper)
615 m_wrapper->handleCommitFailedAfterPostflight(*this);
616 m_transactionError = SQLError::create(SQLError::DATABASE_ERR, "unable to commit transaction", m_database->sqliteDatabase().lastError(), m_database->sqliteDatabase().lastErrorMsg());
617
618 handleTransactionError();
619 return;
620 }
621
622 // Vacuum the database if anything was deleted.
623 if (m_database->hadDeletes())
624 m_database->incrementalVacuumIfNeeded();
625
626 // The commit was successful. If the transaction modified this database, notify the delegates.
627 if (m_modifiedDatabase)
628 m_database->didCommitWriteTransaction();
629
630 // Spec 4.3.2.8: Deliver success callback, if there is one.
631 scheduleCallback(&SQLTransaction::deliverSuccessCallback);
632}
