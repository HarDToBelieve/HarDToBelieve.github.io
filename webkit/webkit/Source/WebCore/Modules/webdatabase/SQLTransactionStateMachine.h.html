26#pragma once
27
28#include "SQLTransactionState.h"
29
30namespace WebCore {
31
32template<typename T>
33class SQLTransactionStateMachine {
34public:
35 virtual ~SQLTransactionStateMachine() = default;
36
37protected:
38 SQLTransactionStateMachine();
39
40 typedef void (T::*StateFunction)();
41 virtual StateFunction stateFunctionFor(SQLTransactionState) = 0;
42
43 void setStateToRequestedState();
44 void runStateMachine();
45
46 SQLTransactionState m_nextState;
47 SQLTransactionState m_requestedState;
48
49#ifndef NDEBUG
50 // The state audit trail (i.e. bread crumbs) keeps track of up to the last
51 // s_sizeOfStateAuditTrail states that the state machine enters. The audit
52 // trail is updated before entering each state. This is for debugging use
53 // only.
54 static const int s_sizeOfStateAuditTrail = 20;
55 int m_nextStateAuditEntry;
56 SQLTransactionState m_stateAuditTrail[s_sizeOfStateAuditTrail];
57#endif
58};
59
60#if !LOG_DISABLED
61extern const char* nameForSQLTransactionState(SQLTransactionState);
62#endif
63
64template<typename T>
65SQLTransactionStateMachine<T>::SQLTransactionStateMachine()
66 : m_nextState(SQLTransactionState::Idle)
67 , m_requestedState(SQLTransactionState::Idle)
68#ifndef NDEBUG
69 , m_nextStateAuditEntry(0)
70#endif
71{
72#ifndef NDEBUG
73 for (int i = 0; i < s_sizeOfStateAuditTrail; i++)
74 m_stateAuditTrail[i] = SQLTransactionState::NumberOfStates;
75#endif
76}
77
78template<typename T>
79void SQLTransactionStateMachine<T>::setStateToRequestedState()
80{
81 ASSERT(m_nextState == SQLTransactionState::Idle);
82 ASSERT(m_requestedState != SQLTransactionState::Idle);
83 m_nextState = m_requestedState;
84 m_requestedState = SQLTransactionState::Idle;
85}
86
87template<typename T>
88void SQLTransactionStateMachine<T>::runStateMachine()
89{
90 ASSERT(SQLTransactionState::End < SQLTransactionState::Idle);
91
92 if (m_nextState <= SQLTransactionState::Idle)
93 return;
94
95 ASSERT(m_nextState < SQLTransactionState::NumberOfStates);
96
97 StateFunction stateFunction = stateFunctionFor(m_nextState);
98 ASSERT(stateFunction);
99
100#ifndef NDEBUG
101 m_stateAuditTrail[m_nextStateAuditEntry] = m_nextState;
102 m_nextStateAuditEntry = (m_nextStateAuditEntry + 1) % s_sizeOfStateAuditTrail;
103#endif
104
105 (static_cast<T*>(this)->*stateFunction)();
106 m_nextState = SQLTransactionState::Idle;
107}
