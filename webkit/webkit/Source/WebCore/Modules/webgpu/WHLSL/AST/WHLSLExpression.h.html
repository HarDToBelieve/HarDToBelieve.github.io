30#include "WHLSLAddressSpace.h"
31#include "WHLSLCodeLocation.h"
32#include "WHLSLDefaultDelete.h"

35#include <wtf/UniqueRef.h>

37namespace WebCore {
38
39namespace WHLSL {
40
41namespace AST {
42
43class Expression {
44 WTF_MAKE_FAST_ALLOCATED;
45
46protected:
47 ~Expression() = default;
48
49public:
50
51 enum class Kind : uint8_t {
52 Assignment,
53 BooleanLiteral,
54 Call,
55 Comma,
56 Dereference,
57 Dot,
58 GlobalVariableReference,
59 FloatLiteral,
60 Index,
61 IntegerLiteral,
62 Logical,
63 LogicalNot,
64 MakeArrayReference,
65 MakePointer,
66 ReadModifyWrite,
67 Ternary,
68 UnsignedIntegerLiteral,
69 VariableReference,
70 EnumerationMemberLiteral,
71 };
72
73 Expression(CodeLocation codeLocation, Kind kind)
74 : m_codeLocation(codeLocation)
75 , m_kind(kind)
76 {
77 }
78
79 static void destroy(Expression&);
80 static void destruct(Expression&);
81
82 Expression(const Expression&) = delete;
83 Expression(Expression&&) = default;
84
85 Expression& operator=(const Expression&) = delete;
86 Expression& operator=(Expression&&) = default;
87
88 UnnamedType* maybeResolvedType() { return m_type ? &*m_type : nullptr; }
89
90 UnnamedType& resolvedType()
91 {
92 ASSERT(m_type);
93 return *m_type;
94 }
95
96 void setType(Ref<UnnamedType> type)
97 {
98 ASSERT(!m_type);
99 m_type = WTFMove(type);
100 }
101
102 const TypeAnnotation* maybeTypeAnnotation() const { return m_typeAnnotation ? &*m_typeAnnotation : nullptr; }
103
104 const TypeAnnotation& typeAnnotation() const
105 {
106 ASSERT(m_typeAnnotation);
107 return *m_typeAnnotation;
108 }
109
110 void setTypeAnnotation(TypeAnnotation&& typeAnnotation)
111 {
112 ASSERT(!m_typeAnnotation);
113 m_typeAnnotation = WTFMove(typeAnnotation);
114 }
115
116 void copyTypeTo(Expression& other) const
117 {
118 if (auto* resolvedType = const_cast<Expression*>(this)->maybeResolvedType())
119 other.setType(*resolvedType);
120 }
121
122 bool mayBeEffectful() const;
123
124 Kind kind() const { return m_kind; }
125 bool isAssignmentExpression() const { return kind() == Kind::Assignment; }
126 bool isBooleanLiteral() const { return kind() == Kind::BooleanLiteral; }
127 bool isCallExpression() const { return kind() == Kind::Call; }
128 bool isCommaExpression() const { return kind() == Kind::Comma; }
129 bool isDereferenceExpression() const { return kind() == Kind::Dereference; }
130 bool isDotExpression() const { return kind() == Kind::Dot; }
131 bool isGlobalVariableReference() const { return kind() == Kind::GlobalVariableReference; }
132 bool isFloatLiteral() const { return kind() == Kind::FloatLiteral; }
133 bool isIndexExpression() const { return kind() == Kind::Index; }
134 bool isIntegerLiteral() const { return kind() == Kind::IntegerLiteral; }
135 bool isLogicalExpression() const { return kind() == Kind::Logical; }
136 bool isLogicalNotExpression() const { return kind() == Kind::LogicalNot; }
137 bool isMakeArrayReferenceExpression() const { return kind() == Kind::MakeArrayReference; }
138 bool isMakePointerExpression() const { return kind() == Kind::MakePointer; }
139 bool isPropertyAccessExpression() const { return isDotExpression() || isIndexExpression(); }
140 bool isReadModifyWriteExpression() const { return kind() == Kind::ReadModifyWrite; }
141 bool isTernaryExpression() const { return kind() == Kind::Ternary; }
142 bool isUnsignedIntegerLiteral() const { return kind() == Kind::UnsignedIntegerLiteral; }
143 bool isVariableReference() const { return kind() == Kind::VariableReference; }
144 bool isEnumerationMemberLiteral() const { return kind() == Kind::EnumerationMemberLiteral; }
145
146 CodeLocation codeLocation() const { return m_codeLocation; }
147 void updateCodeLocation(CodeLocation location) { m_codeLocation = location; }
148
149private:
150 CodeLocation m_codeLocation;
151 RefPtr<UnnamedType> m_type;
152 std::optional<TypeAnnotation> m_typeAnnotation;
153 Kind m_kind;
154};
155
156} // namespace AST
157
158}
159
160}
