50static String attributeForSemantic(AST::BuiltInSemantic& builtInSemantic)
51{
52 switch (builtInSemantic.variable()) {
53 case AST::BuiltInSemantic::Variable::SVInstanceID:
54 return "[[instance_id]]"_str;
55 case AST::BuiltInSemantic::Variable::SVVertexID:
56 return "[[vertex_id]]"_str;
57 case AST::BuiltInSemantic::Variable::PSize:
58 return "[[point_size]]"_str;
59 case AST::BuiltInSemantic::Variable::SVPosition:
60 return "[[position]]"_str;
61 case AST::BuiltInSemantic::Variable::SVIsFrontFace:
62 return "[[front_facing]]"_str;
63 case AST::BuiltInSemantic::Variable::SVSampleIndex:
64 return "[[sample_id]]"_str;
65 case AST::BuiltInSemantic::Variable::SVInnerCoverage:
66 return "[[sample_mask]]"_str;
67 case AST::BuiltInSemantic::Variable::SVTarget:
68 return makeString("[[color(", *builtInSemantic.targetIndex(), ")]]");
69 case AST::BuiltInSemantic::Variable::SVDepth:
70 return "[[depth(any)]]"_str;
71 case AST::BuiltInSemantic::Variable::SVCoverage:
72 return "[[sample_mask]]"_str;
73 case AST::BuiltInSemantic::Variable::SVDispatchThreadID:
74 return "[[thread_position_in_grid]]"_str;
75 case AST::BuiltInSemantic::Variable::SVGroupID:
76 return "[[threadgroup_position_in_grid]]"_str;
77 case AST::BuiltInSemantic::Variable::SVGroupIndex:
78 return "[[thread_index_in_threadgroup]]"_str;
79 default:
80 ASSERT(builtInSemantic.variable() == AST::BuiltInSemantic::Variable::SVGroupThreadID);
81 return "[[thread_position_in_threadgroup]]"_str;
82 }
83}

85static String attributeForSemantic(AST::Semantic& semantic)
86{
87 if (WTF::holds_alternative<AST::BuiltInSemantic>(semantic))
88 return attributeForSemantic(WTF::get<AST::BuiltInSemantic>(semantic));
89 auto& stageInOutSemantic = WTF::get<AST::StageInOutSemantic>(semantic);
90 return makeString("[[user(user", stageInOutSemantic.index(), ")]]");
91}

93EntryPointScaffolding::EntryPointScaffolding(AST::FunctionDefinition& functionDefinition, Intrinsics& intrinsics, TypeNamer& typeNamer, EntryPointItems& entryPointItems, HashMap<Binding*, size_t>& resourceMap, Layout& layout, std::function<MangledVariableName()>&& generateNextVariableName)
94 : m_functionDefinition(functionDefinition)
95 , m_intrinsics(intrinsics)
96 , m_typeNamer(typeNamer)
97 , m_entryPointItems(entryPointItems)
98 , m_resourceMap(resourceMap)
99 , m_layout(layout)
100 , m_generateNextVariableName(generateNextVariableName)
101{
102 m_namedBindGroups.reserveInitialCapacity(m_layout.size());
103 for (size_t i = 0; i < m_layout.size(); ++i) {
104 NamedBindGroup namedBindGroup;
105 namedBindGroup.structName = m_typeNamer.generateNextTypeName();
106 namedBindGroup.variableName = m_generateNextVariableName();
107 namedBindGroup.argumentBufferIndex = m_layout[i].name; // convertLayout() in GPURenderPipelineMetal.mm makes sure these don't collide.
108 namedBindGroup.namedBindings.reserveInitialCapacity(m_layout[i].bindings.size());
109 for (size_t j = 0; j < m_layout[i].bindings.size(); ++j) {
110 NamedBinding namedBinding;
111 namedBinding.elementName = m_typeNamer.generateNextStructureElementName();
112 namedBinding.index = m_layout[i].bindings[j].internalName;
113 WTF::visit(WTF::makeVisitor([&](UniformBufferBinding& uniformBufferBinding) {
114 LengthInformation lengthInformation { m_typeNamer.generateNextStructureElementName(), m_generateNextVariableName(), uniformBufferBinding.lengthName };
115 namedBinding.lengthInformation = lengthInformation;
116 }, [&](SamplerBinding&) {
117 }, [&](TextureBinding&) {
118 }, [&](StorageBufferBinding& storageBufferBinding) {
119 LengthInformation lengthInformation { m_typeNamer.generateNextStructureElementName(), m_generateNextVariableName(), storageBufferBinding.lengthName };
120 namedBinding.lengthInformation = lengthInformation;
121 }), m_layout[i].bindings[j].binding);
122 namedBindGroup.namedBindings.uncheckedAppend(WTFMove(namedBinding));
123 }
124 m_namedBindGroups.uncheckedAppend(WTFMove(namedBindGroup));
125 }
126
127 for (size_t i = 0; i < m_entryPointItems.inputs.size(); ++i) {
128 if (!WTF::holds_alternative<AST::BuiltInSemantic>(*m_entryPointItems.inputs[i].semantic))
129 continue;
130 NamedBuiltIn namedBuiltIn;
131 namedBuiltIn.indexInEntryPointItems = i;
132 namedBuiltIn.variableName = m_generateNextVariableName();
133 m_namedBuiltIns.append(WTFMove(namedBuiltIn));
134 }
135
136 m_parameterVariables.reserveInitialCapacity(m_functionDefinition.parameters().size());
137 for (size_t i = 0; i < m_functionDefinition.parameters().size(); ++i)
138 m_parameterVariables.uncheckedAppend(m_generateNextVariableName());
139}

141void EntryPointScaffolding::emitResourceHelperTypes(StringBuilder& stringBuilder, Indentation<4> indent, ShaderStage shaderStage)
142{
143 for (size_t i = 0; i < m_layout.size(); ++i) {
144 stringBuilder.append(indent, "struct ", m_namedBindGroups[i].structName, " {\n");
145 {
146 IndentationScope scope(indent);
147 Vector<std::pair<unsigned, String>> structItems;
148 for (size_t j = 0; j < m_layout[i].bindings.size(); ++j) {
149 auto& binding = m_layout[i].bindings[j];
150 if (!binding.visibility.contains(shaderStage))
151 continue;
152
153 auto elementName = m_namedBindGroups[i].namedBindings[j].elementName;
154 auto index = m_namedBindGroups[i].namedBindings[j].index;
155 if (auto lengthInformation = m_namedBindGroups[i].namedBindings[j].lengthInformation)
156 structItems.append(std::make_pair(lengthInformation->index, makeString("uint2 ", lengthInformation->elementName, " [[id(", lengthInformation->index, ")]];")));
157
158 auto iterator = m_resourceMap.find(&m_layout[i].bindings[j]);
159 if (iterator != m_resourceMap.end()) {
160 auto& type = m_entryPointItems.inputs[iterator->value].unnamedType->unifyNode();
161 if (is<AST::UnnamedType>(type) && is<AST::ReferenceType>(downcast<AST::UnnamedType>(type))) {
162 auto& referenceType = downcast<AST::ReferenceType>(downcast<AST::UnnamedType>(type));
163 auto mangledTypeName = m_typeNamer.mangledNameForType(referenceType.elementType());
164 auto addressSpace = toString(referenceType.addressSpace());
165 structItems.append(std::make_pair(index, makeString(addressSpace, " ", mangledTypeName, "* ", elementName, " [[id(", index, ")]];")));
166 } else if (is<AST::NamedType>(type) && is<AST::NativeTypeDeclaration>(downcast<AST::NamedType>(type))) {
167 auto& namedType = downcast<AST::NativeTypeDeclaration>(downcast<AST::NamedType>(type));
168 auto mangledTypeName = m_typeNamer.mangledNameForType(namedType);
169 structItems.append(std::make_pair(index, makeString(mangledTypeName, ' ', elementName, " [[id(", index, ")]];")));
170 }
171 } else {
172 // The binding doesn't appear in the shader source.
173 // However, we must still emit a placeholder, so successive items in the argument buffer struct have the correct offset.
174 // Because the binding doesn't appear in the shader source, we don't know which exact type the bind point should have.
175 // Therefore, we must synthesize a type out of thin air.
176 WTF::visit(WTF::makeVisitor([&](UniformBufferBinding) {
177 structItems.append(std::make_pair(index, makeString("constant void* ", elementName, " [[id(", index, ")]];")));
178 }, [&](SamplerBinding) {
179 structItems.append(std::make_pair(index, makeString("sampler ", elementName, " [[id(", index, ")]];")));
180 }, [&](TextureBinding) {
181 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=201384 We don't know which texture type the binding represents. This is no good very bad.
182 structItems.append(std::make_pair(index, makeString("texture2d<float> ", elementName, " [[id(", index, ")]];")));
183 }, [&](StorageBufferBinding) {
184 structItems.append(std::make_pair(index, makeString("device void* ", elementName, " [[id(", index, ")]];")));
185 }), binding.binding);
186 }
187 }
188 std::sort(structItems.begin(), structItems.end(), [](const std::pair<unsigned, String>& left, const std::pair<unsigned, String>& right) {
189 return left.first < right.first;
190 });
191 for (const auto& structItem : structItems)
192 stringBuilder.append(indent, structItem.second, '\n');
193 }
194 stringBuilder.append(indent, "};\n\n");
195 }
196}

198bool EntryPointScaffolding::emitResourceSignature(StringBuilder& stringBuilder, IncludePrecedingComma includePrecedingComma)
199{
200 if (!m_layout.size())
201 return false;
202
203 if (includePrecedingComma == IncludePrecedingComma::Yes)
204 stringBuilder.append(", ");
205
206 for (size_t i = 0; i < m_layout.size(); ++i) {
207 if (i)
208 stringBuilder.append(", ");
209 auto& namedBindGroup = m_namedBindGroups[i];
210 stringBuilder.append("device ", namedBindGroup.structName, "& ", namedBindGroup.variableName, " [[buffer(", namedBindGroup.argumentBufferIndex, ")]]");
211 }
212 return true;
213}

215static StringView internalTypeForSemantic(const AST::BuiltInSemantic& builtInSemantic)
216{
217 switch (builtInSemantic.variable()) {
218 case AST::BuiltInSemantic::Variable::SVInstanceID:
219 return "uint";
220 case AST::BuiltInSemantic::Variable::SVVertexID:
221 return "uint";
222 case AST::BuiltInSemantic::Variable::PSize:
223 return "float";
224 case AST::BuiltInSemantic::Variable::SVPosition:
225 return "float4";
226 case AST::BuiltInSemantic::Variable::SVIsFrontFace:
227 return "bool";
228 case AST::BuiltInSemantic::Variable::SVSampleIndex:
229 return "uint";
230 case AST::BuiltInSemantic::Variable::SVInnerCoverage:
231 return "uint";
232 case AST::BuiltInSemantic::Variable::SVTarget:
233 return { };
234 case AST::BuiltInSemantic::Variable::SVDepth:
235 return "float";
236 case AST::BuiltInSemantic::Variable::SVCoverage:
237 return "uint";
238 case AST::BuiltInSemantic::Variable::SVDispatchThreadID:
239 return "uint3";
240 case AST::BuiltInSemantic::Variable::SVGroupID:
241 return "uint3";
242 case AST::BuiltInSemantic::Variable::SVGroupIndex:
243 return "uint";
244 default:
245 ASSERT(builtInSemantic.variable() == AST::BuiltInSemantic::Variable::SVGroupThreadID);
246 return "uint3";
247 }

250bool EntryPointScaffolding::emitBuiltInsSignature(StringBuilder& stringBuilder, IncludePrecedingComma includePrecedingComma)
251{
252 if (!m_namedBuiltIns.size())
253 return false;
254
255 if (includePrecedingComma == IncludePrecedingComma::Yes)
256 stringBuilder.append(", ");
257
258 for (size_t i = 0; i < m_namedBuiltIns.size(); ++i) {
259 if (i)
260 stringBuilder.append(", ");
261 auto& namedBuiltIn = m_namedBuiltIns[i];
262 auto& item = m_entryPointItems.inputs[namedBuiltIn.indexInEntryPointItems];
263 auto& builtInSemantic = WTF::get<AST::BuiltInSemantic>(*item.semantic);
264 auto internalType = internalTypeForSemantic(builtInSemantic);
265 if (!internalType.isNull())
266 stringBuilder.append(internalType);
267 else
268 stringBuilder.append(m_typeNamer.mangledNameForType(*item.unnamedType));
269 stringBuilder.append(' ', namedBuiltIn.variableName, ' ', attributeForSemantic(builtInSemantic));
270 }
271 return true;
272}

274void EntryPointScaffolding::emitMangledInputPath(StringBuilder& stringBuilder, Vector<String>& path)
275{
276 ASSERT(!path.isEmpty());
277 bool found = false;
278 AST::StructureDefinition* structureDefinition = nullptr;
279 for (size_t i = 0; i < m_functionDefinition.parameters().size(); ++i) {
280 if (m_functionDefinition.parameters()[i]->name() == path[0]) {
281 stringBuilder.append(m_parameterVariables[i]);
282 auto& unifyNode = m_functionDefinition.parameters()[i]->type()->unifyNode();
283 if (is<AST::NamedType>(unifyNode)) {
284 auto& namedType = downcast<AST::NamedType>(unifyNode);
285 if (is<AST::StructureDefinition>(namedType))
286 structureDefinition = &downcast<AST::StructureDefinition>(namedType);
287 }
288 found = true;
289 break;
290 }
291 }
292 ASSERT(found);
293 for (size_t i = 1; i < path.size(); ++i) {
294 ASSERT(structureDefinition);
295 auto* next = structureDefinition->find(path[i]);
296 ASSERT(next);
297 stringBuilder.append('.', m_typeNamer.mangledNameForStructureElement(*next));
298 structureDefinition = nullptr;
299 auto& unifyNode = next->type().unifyNode();
300 if (is<AST::NamedType>(unifyNode)) {
301 auto& namedType = downcast<AST::NamedType>(unifyNode);
302 if (is<AST::StructureDefinition>(namedType))
303 structureDefinition = &downcast<AST::StructureDefinition>(namedType);
304 }
305 }
306}

308void EntryPointScaffolding::emitMangledOutputPath(StringBuilder& stringBuilder, Vector<String>& path)
309{
310 AST::StructureDefinition* structureDefinition = nullptr;
311 auto& unifyNode = m_functionDefinition.type().unifyNode();
312 structureDefinition = &downcast<AST::StructureDefinition>(downcast<AST::NamedType>(unifyNode));
313 for (auto& component : path) {
314 ASSERT(structureDefinition);
315 auto* next = structureDefinition->find(component);
316 ASSERT(next);
317 stringBuilder.append('.', m_typeNamer.mangledNameForStructureElement(*next));
318 structureDefinition = nullptr;
319 auto& unifyNode = next->type().unifyNode();
320 if (is<AST::NamedType>(unifyNode)) {
321 auto& namedType = downcast<AST::NamedType>(unifyNode);
322 if (is<AST::StructureDefinition>(namedType))
323 structureDefinition = &downcast<AST::StructureDefinition>(namedType);
324 }
325 }

328void EntryPointScaffolding::emitUnpackResourcesAndNamedBuiltIns(StringBuilder& stringBuilder, Indentation<4> indent)
329{
330 for (size_t i = 0; i < m_functionDefinition.parameters().size(); ++i)
331 stringBuilder.append(indent, m_typeNamer.mangledNameForType(*m_functionDefinition.parameters()[i]->type()), ' ', m_parameterVariables[i], ";\n");
332
333 for (size_t i = 0; i < m_layout.size(); ++i) {
334 auto variableName = m_namedBindGroups[i].variableName;
335 for (size_t j = 0; j < m_layout[i].bindings.size(); ++j) {
336 auto iterator = m_resourceMap.find(&m_layout[i].bindings[j]);
337 if (iterator == m_resourceMap.end())
338 continue;
339 if (m_namedBindGroups[i].namedBindings[j].lengthInformation) {
340 auto& path = m_entryPointItems.inputs[iterator->value].path;
341 auto elementName = m_namedBindGroups[i].namedBindings[j].elementName;
342 auto lengthElementName = m_namedBindGroups[i].namedBindings[j].lengthInformation->elementName;
343 auto lengthTemporaryName = m_namedBindGroups[i].namedBindings[j].lengthInformation->temporaryName;
344
345 auto& unnamedType = *m_entryPointItems.inputs[iterator->value].unnamedType;
346 auto mangledTypeName = m_typeNamer.mangledNameForType(downcast<AST::ReferenceType>(unnamedType).elementType());
347
348 stringBuilder.append(
349 indent, "size_t ", lengthTemporaryName, " = ", variableName, '.', lengthElementName, ".y;\n",
350 indent, lengthTemporaryName, " = ", lengthTemporaryName, " << 32;\n",
351 indent, lengthTemporaryName, " = ", lengthTemporaryName, " | ", variableName, '.', lengthElementName, ".x;\n",
352 indent, lengthTemporaryName, " = ", lengthTemporaryName, " / sizeof(", mangledTypeName, ");\n",
353 indent, "if (", lengthTemporaryName, " > 0xFFFFFFFF)\n",
354 indent, " ", lengthTemporaryName, " = 0xFFFFFFFF;\n"
355 );
356
357 stringBuilder.append(indent);
358 emitMangledInputPath(stringBuilder, path);
359 stringBuilder.append(
360 " = { ", variableName, '.', elementName, ", static_cast<uint32_t>(", lengthTemporaryName, ") };\n"
361 );
362 } else {
363 auto& path = m_entryPointItems.inputs[iterator->value].path;
364 auto elementName = m_namedBindGroups[i].namedBindings[j].elementName;
365
366 stringBuilder.append(indent);
367 emitMangledInputPath(stringBuilder, path);
368 stringBuilder.append(" = ", variableName, '.', elementName, ";\n");
369 }
370 }
371 }
372
373 for (auto& namedBuiltIn : m_namedBuiltIns) {
374 auto& item = m_entryPointItems.inputs[namedBuiltIn.indexInEntryPointItems];
375 auto& path = item.path;
376 auto& variableName = namedBuiltIn.variableName;
377 auto mangledTypeName = m_typeNamer.mangledNameForType(*item.unnamedType);
378
379 stringBuilder.append(indent);
380 emitMangledInputPath(stringBuilder, path);
381 stringBuilder.append(" = ", mangledTypeName, '(', variableName, ");\n");
382 }
383}

385VertexEntryPointScaffolding::VertexEntryPointScaffolding(AST::FunctionDefinition& functionDefinition, Intrinsics& intrinsics, TypeNamer& typeNamer, EntryPointItems& entryPointItems, HashMap<Binding*, size_t>& resourceMap, Layout& layout, std::function<MangledVariableName()>&& generateNextVariableName, HashMap<VertexAttribute*, size_t>& matchedVertexAttributes)
386 : EntryPointScaffolding(functionDefinition, intrinsics, typeNamer, entryPointItems, resourceMap, layout, WTFMove(generateNextVariableName))
387 , m_matchedVertexAttributes(matchedVertexAttributes)
388 , m_stageInStructName(typeNamer.generateNextTypeName())
389 , m_returnStructName(typeNamer.generateNextTypeName())
390 , m_stageInParameterName(m_generateNextVariableName())
391{
392 m_namedStageIns.reserveInitialCapacity(m_matchedVertexAttributes.size());
393 for (auto& keyValuePair : m_matchedVertexAttributes) {
394 NamedStageIn namedStageIn;
395 namedStageIn.indexInEntryPointItems = keyValuePair.value;
396 namedStageIn.elementName = m_typeNamer.generateNextStructureElementName();
397 namedStageIn.attributeIndex = keyValuePair.key->metalLocation;
398 m_namedStageIns.uncheckedAppend(WTFMove(namedStageIn));
399 }
400
401 m_namedOutputs.reserveInitialCapacity(m_entryPointItems.outputs.size());
402 for (size_t i = 0; i < m_entryPointItems.outputs.size(); ++i) {
403 auto& outputItem = m_entryPointItems.outputs[i];
404 NamedOutput namedOutput;
405 namedOutput.elementName = m_typeNamer.generateNextStructureElementName();
406 StringView internalType;
407 if (WTF::holds_alternative<AST::BuiltInSemantic>(*outputItem.semantic))
408 internalType = internalTypeForSemantic(WTF::get<AST::BuiltInSemantic>(*outputItem.semantic));
409 if (!internalType.isNull())
410 namedOutput.internalTypeName = internalType.toString();
411 else
412 namedOutput.internalTypeName = m_typeNamer.mangledNameForType(*outputItem.unnamedType);
413 m_namedOutputs.uncheckedAppend(WTFMove(namedOutput));
414 }
415}

417void VertexEntryPointScaffolding::emitHelperTypes(StringBuilder& stringBuilder, Indentation<4> indent)
418{
419 stringBuilder.append(indent, "struct ", m_stageInStructName, " {\n");
420 {
421 IndentationScope scope(indent);
422 for (auto& namedStageIn : m_namedStageIns) {
423 auto mangledTypeName = m_typeNamer.mangledNameForType(*m_entryPointItems.inputs[namedStageIn.indexInEntryPointItems].unnamedType);
424 auto elementName = namedStageIn.elementName;
425 auto attributeIndex = namedStageIn.attributeIndex;
426 stringBuilder.append(indent, mangledTypeName, ' ', elementName, " [[attribute(", attributeIndex, ")]];\n");
427 }
428 }
429 stringBuilder.append(
430 indent, "};\n\n",
431 indent, "struct ", m_returnStructName, " {\n"
432 );
433 {
434 IndentationScope scope(indent);
435 for (size_t i = 0; i < m_entryPointItems.outputs.size(); ++i) {
436 auto& outputItem = m_entryPointItems.outputs[i];
437 auto& internalTypeName = m_namedOutputs[i].internalTypeName;
438 auto elementName = m_namedOutputs[i].elementName;
439 auto attribute = attributeForSemantic(*outputItem.semantic);
440 stringBuilder.append(indent, internalTypeName, ' ', elementName, ' ', attribute, ";\n");
441 }
442 }
443 stringBuilder.append(indent, "};\n\n");
444
445 emitResourceHelperTypes(stringBuilder, indent, ShaderStage::Vertex);
446}

470void VertexEntryPointScaffolding::emitPack(StringBuilder& stringBuilder, MangledVariableName inputVariableName, MangledVariableName outputVariableName, Indentation<4> indent)
471{
472 stringBuilder.append(indent, m_returnStructName, ' ', outputVariableName, ";\n");
473 if (m_entryPointItems.outputs.size() == 1 && !m_entryPointItems.outputs[0].path.size()) {
474 auto& elementName = m_namedOutputs[0].elementName;
475 stringBuilder.append(indent, outputVariableName, '.', elementName, " = ", inputVariableName, ";\n");
476 return;
477 }
478 for (size_t i = 0; i < m_entryPointItems.outputs.size(); ++i) {
479 auto& elementName = m_namedOutputs[i].elementName;
480 auto& internalTypeName = m_namedOutputs[i].internalTypeName;
481 auto& path = m_entryPointItems.outputs[i].path;
482 stringBuilder.append(indent, outputVariableName, '.', elementName, " = ", internalTypeName, '(', inputVariableName);
483 emitMangledOutputPath(stringBuilder, path);
484 stringBuilder.append(");\n");
485 }

488FragmentEntryPointScaffolding::FragmentEntryPointScaffolding(AST::FunctionDefinition& functionDefinition, Intrinsics& intrinsics, TypeNamer& typeNamer, EntryPointItems& entryPointItems, HashMap<Binding*, size_t>& resourceMap, Layout& layout, std::function<MangledVariableName()>&& generateNextVariableName, HashMap<AttachmentDescriptor*, size_t>&)
489 : EntryPointScaffolding(functionDefinition, intrinsics, typeNamer, entryPointItems, resourceMap, layout, WTFMove(generateNextVariableName))
490 , m_stageInStructName(typeNamer.generateNextTypeName())
491 , m_returnStructName(typeNamer.generateNextTypeName())
492 , m_stageInParameterName(m_generateNextVariableName())
493{
494 for (size_t i = 0; i < m_entryPointItems.inputs.size(); ++i) {
495 auto& inputItem = m_entryPointItems.inputs[i];
496 if (!WTF::holds_alternative<AST::StageInOutSemantic>(*inputItem.semantic))
497 continue;
498 auto& stageInOutSemantic = WTF::get<AST::StageInOutSemantic>(*inputItem.semantic);
499 NamedStageIn namedStageIn;
500 namedStageIn.indexInEntryPointItems = i;
501 namedStageIn.elementName = m_typeNamer.generateNextStructureElementName();
502 namedStageIn.attributeIndex = stageInOutSemantic.index();
503 m_namedStageIns.append(WTFMove(namedStageIn));
504 }
505
506 m_namedOutputs.reserveInitialCapacity(m_entryPointItems.outputs.size());
507 for (size_t i = 0; i < m_entryPointItems.outputs.size(); ++i) {
508 auto& outputItem = m_entryPointItems.outputs[i];
509 NamedOutput namedOutput;
510 namedOutput.elementName = m_typeNamer.generateNextStructureElementName();
511 StringView internalType;
512 if (WTF::holds_alternative<AST::BuiltInSemantic>(*outputItem.semantic))
513 internalType = internalTypeForSemantic(WTF::get<AST::BuiltInSemantic>(*outputItem.semantic));
514 if (!internalType.isNull())
515 namedOutput.internalTypeName = internalType.toString();
516 else
