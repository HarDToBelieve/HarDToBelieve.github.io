72void TypeNamer::visit(AST::EnumerationDefinition& enumerationDefinition)
73{
74 {
75 auto addResult = m_namedTypeMapping.add(&enumerationDefinition, generateNextTypeName());
76 ASSERT_UNUSED(addResult, addResult.isNewEntry);
77 }
78
79 for (auto& enumerationMember : enumerationDefinition.enumerationMembers()) {
80 auto addResult = m_enumerationMemberMapping.add(&static_cast<AST::EnumerationMember&>(enumerationMember), generateNextEnumerationMemberName());
81 ASSERT_UNUSED(addResult, addResult.isNewEntry);
82 }
83
84 Visitor::visit(enumerationDefinition);
85
86 {
87 auto addResult = m_dependencyGraph.add(&enumerationDefinition, Vector<std::reference_wrapper<AST::UnnamedType>> { enumerationDefinition.type() });
88 ASSERT_UNUSED(addResult, addResult.isNewEntry);
89 }
90}

92void TypeNamer::visit(AST::NativeTypeDeclaration& nativeTypeDeclaration)
93{
94 // Native type declarations already have names, and are already declared in Metal.
95 auto addResult = m_dependencyGraph.add(&nativeTypeDeclaration, Vector<std::reference_wrapper<AST::UnnamedType>> { });
96 ASSERT_UNUSED(addResult, addResult.isNewEntry);
97}

99void TypeNamer::visit(AST::StructureDefinition& structureDefinition)
100{
101 {
102 auto addResult = m_namedTypeMapping.add(&structureDefinition, generateNextTypeName());
103 ASSERT_UNUSED(addResult, addResult.isNewEntry);
104 }
105 Visitor::visit(structureDefinition);
106 {
107 Vector<std::reference_wrapper<AST::UnnamedType>> neighbors;
108 for (auto& structureElement : structureDefinition.structureElements()) {
109 auto addResult = m_structureElementMapping.add(&structureElement, generateNextStructureElementName());
110 ASSERT_UNUSED(addResult, addResult.isNewEntry);
111 neighbors.append(structureElement.type());
112 }
113 auto addResult = m_dependencyGraph.add(&structureDefinition, WTFMove(neighbors));
114 ASSERT_UNUSED(addResult, addResult.isNewEntry);
115 }
116}

148static AST::UnnamedType* parent(AST::UnnamedType& unnamedType)
149{
150 switch (unnamedType.kind()) {
151 case AST::UnnamedType::Kind::TypeReference:
152 return nullptr;
153 case AST::UnnamedType::Kind::Pointer:
154 return &downcast<AST::PointerType>(unnamedType).elementType();
155 case AST::UnnamedType::Kind::ArrayReference:
156 return &downcast<AST::ArrayReferenceType>(unnamedType).elementType();
157 case AST::UnnamedType::Kind::Array:
158 return &downcast<AST::ArrayType>(unnamedType).type();
159 default:
160 RELEASE_ASSERT_NOT_REACHED();
161 }
162}
163
164void TypeNamer::generateUniquedTypeName(AST::UnnamedType& unnamedType)
165{
166 auto* parentUnnamedType = parent(unnamedType);
167 if (parentUnnamedType)
168 generateUniquedTypeName(*parentUnnamedType);
169
170 m_unnamedTypeMapping.ensure(UnnamedTypeKey { unnamedType }, [&] {
171 return generateNextTypeName();
172 });
173}

175class MetalTypeDeclarationWriter final : public Visitor {
176 WTF_MAKE_FAST_ALLOCATED;
177public:
178 MetalTypeDeclarationWriter(StringBuilder& stringBuilder, std::function<MangledOrNativeTypeName(AST::NamedType&)>&& mangledNameForNamedType)
179 : m_mangledNameForNamedType(WTFMove(mangledNameForNamedType))
180 , m_stringBuilder(stringBuilder)
181 {
182 }
183
184private:
185 void visit(AST::StructureDefinition& structureDefinition) override
186 {
187 m_stringBuilder.append("struct ", m_mangledNameForNamedType(structureDefinition), ";\n");
188 }
189
190 std::function<MangledOrNativeTypeName(AST::NamedType&)> m_mangledNameForNamedType;
191 StringBuilder& m_stringBuilder;
192};

202void TypeNamer::emitUnnamedTypeDefinition(StringBuilder& stringBuilder, AST::UnnamedType& unnamedType, MangledTypeName mangledName, HashSet<AST::NamedType*>& emittedNamedTypes, HashSet<UnnamedTypeKey>& emittedUnnamedTypes)
203{
204 if (emittedUnnamedTypes.contains(UnnamedTypeKey { unnamedType }))
205 return;
206
207 switch (unnamedType.kind()) {
208 case AST::UnnamedType::Kind::TypeReference: {
209 auto& typeReference = downcast<AST::TypeReference>(unnamedType);
210
211 auto& parent = typeReference.resolvedType();
212 auto parentMangledName = mangledNameForType(typeReference.resolvedType());
213 auto iterator = m_dependencyGraph.find(&parent);
214 ASSERT(iterator != m_dependencyGraph.end());
215 emitNamedTypeDefinition(stringBuilder, parent, iterator->value, emittedNamedTypes, emittedUnnamedTypes);
216
217 stringBuilder.append("typedef ", parentMangledName, ' ', mangledName, ";\n");
218 break;
219 }
220 case AST::UnnamedType::Kind::Pointer: {
221 auto& pointerType = downcast<AST::PointerType>(unnamedType);
222
223 auto& parent = pointerType.elementType();
224 auto parentMangledName = mangledNameForType(parent);
225 emitUnnamedTypeDefinition(stringBuilder, parent, parentMangledName, emittedNamedTypes, emittedUnnamedTypes);
226
227 stringBuilder.append("typedef ", toString(pointerType.addressSpace()), ' ', parentMangledName, "* ", mangledName, ";\n");
228 break;
229 }
230 case AST::UnnamedType::Kind::ArrayReference: {
231 auto& arrayReferenceType = downcast<AST::ArrayReferenceType>(unnamedType);
232
233 auto& parent = arrayReferenceType.elementType();
234 auto parentMangledName = mangledNameForType(parent);
235 emitUnnamedTypeDefinition(stringBuilder, parent, parentMangledName, emittedNamedTypes, emittedUnnamedTypes);
236
237 stringBuilder.append(
238 "struct ", mangledName, " {\n"
239 " ", toString(arrayReferenceType.addressSpace()), ' ', parentMangledName, "* pointer;\n"
240 " uint32_t length;\n"
241 "};\n"
242 );
243 break;
244 }
245 case AST::UnnamedType::Kind::Array: {
246 auto& arrayType = downcast<AST::ArrayType>(unnamedType);
247
248 auto& parent = arrayType.type();
249 auto parentMangledName = mangledNameForType(parent);
250 emitUnnamedTypeDefinition(stringBuilder, parent, parentMangledName, emittedNamedTypes, emittedUnnamedTypes);
251
252 stringBuilder.append("typedef array<", parentMangledName, ", ", arrayType.numElements(), "> ", mangledName, ";\n");
253 break;
254 }
255 default:
256 RELEASE_ASSERT_NOT_REACHED();
257 }
258
259 emittedUnnamedTypes.add(UnnamedTypeKey { unnamedType });
260}

262void TypeNamer::emitNamedTypeDefinition(StringBuilder& stringBuilder, AST::NamedType& namedType, Vector<std::reference_wrapper<AST::UnnamedType>>& neighbors, HashSet<AST::NamedType*>& emittedNamedTypes, HashSet<UnnamedTypeKey>& emittedUnnamedTypes)
263{
264 if (emittedNamedTypes.contains(&namedType))
265 return;
266
267 for (auto& unnameType : neighbors)
268 emitUnnamedTypeDefinition(stringBuilder, unnameType, mangledNameForType(unnameType), emittedNamedTypes, emittedUnnamedTypes);
269
270 switch (namedType.kind()) {
271 case AST::NamedType::Kind::EnumerationDefinition: {
272 auto& enumerationDefinition = downcast<AST::EnumerationDefinition>(namedType);
273 auto& baseType = enumerationDefinition.type().unifyNode();
274
275 stringBuilder.append("enum class ", mangledNameForType(enumerationDefinition), " : ", mangledNameForType(downcast<AST::NamedType>(baseType)), " {\n");
276 for (auto& enumerationMember : enumerationDefinition.enumerationMembers())
277 stringBuilder.append(" ", mangledNameForEnumerationMember(enumerationMember), " = ", enumerationMember.get().value(), ",\n");
278 stringBuilder.append("};\n");
279 break;
280 }
281 case AST::NamedType::Kind::NativeTypeDeclaration: {
282 // Native types already have definitions. There's nothing to do.
283 break;
284 }
285 case AST::NamedType::Kind::StructureDefinition: {
286 auto& structureDefinition = downcast<AST::StructureDefinition>(namedType);
287
288 stringBuilder.append("struct ", mangledNameForType(structureDefinition), " {\n");
289 for (auto& structureElement : structureDefinition.structureElements())
290 stringBuilder.append(" ", mangledNameForType(structureElement.type()), ' ', mangledNameForStructureElement(structureElement), ";\n");
291 stringBuilder.append("};\n");
292 break;
293 }
294 case AST::NamedType::Kind::TypeDefinition: {
295 auto& typeDefinition = downcast<AST::TypeDefinition>(namedType);
296
297 stringBuilder.append("typedef ", mangledNameForType(typeDefinition.type()), ' ', mangledNameForType(typeDefinition), ";\n");
298 break;
299 }
300 default:
301 RELEASE_ASSERT_NOT_REACHED();
302 }
303
304 emittedNamedTypes.add(&namedType);
305}

307void TypeNamer::emitMetalTypeDefinitions(StringBuilder& stringBuilder)
308{
309 HashSet<AST::NamedType*> emittedNamedTypes;
310 HashSet<UnnamedTypeKey> emittedUnnamedTypes;
311 for (auto& [namedType, neighbors] : m_dependencyGraph)
312 emitNamedTypeDefinition(stringBuilder, *namedType, neighbors, emittedNamedTypes, emittedUnnamedTypes);
313 for (auto& [unnamedTypeKey, mangledName] : m_unnamedTypeMapping)
314 emitUnnamedTypeDefinition(stringBuilder, unnamedTypeKey.unnamedType(), mangledName, emittedNamedTypes, emittedUnnamedTypes);
315}

317MangledTypeName TypeNamer::mangledNameForType(AST::UnnamedType& unnamedType)
318{
319 auto iterator = m_unnamedTypeMapping.find(UnnamedTypeKey { unnamedType });
320 ASSERT(iterator != m_unnamedTypeMapping.end());
321 return iterator->value;
322}

324MangledOrNativeTypeName TypeNamer::mangledNameForType(AST::NamedType& namedType)
325{
326 if (is<AST::NativeTypeDeclaration>(namedType))
327 return mangledNameForType(downcast<AST::NativeTypeDeclaration>(namedType));
328 auto iterator = m_namedTypeMapping.find(&namedType);
329 ASSERT(iterator != m_namedTypeMapping.end());
330 return iterator->value;
331}

347void TypeNamer::emitMetalTypes(StringBuilder& stringBuilder)
348{
349 Visitor::visit(m_program);
350
351 emitMetalTypeDeclarations(stringBuilder);
352 stringBuilder.append('\n');
353 emitMetalTypeDefinitions(stringBuilder);
