130template <typename T>
131ALWAYS_INLINE void dumpASTNode(PrintStream& out, T& value)
132{
133 ASTDumper dumper;
134 dumper.visit(value);
135 out.print(dumper.toString());
136}
137MAKE_PRINT_ADAPTOR(ExpressionDumper, AST::Expression&, dumpASTNode);
138MAKE_PRINT_ADAPTOR(StatementDumper, AST::Statement&, dumpASTNode);
139MAKE_PRINT_ADAPTOR(ProgramDumper, Program&, dumpASTNode);
140MAKE_PRINT_ADAPTOR(StructureDefinitionDumper, AST::StructureDefinition&, dumpASTNode);
141MAKE_PRINT_ADAPTOR(FunctionDefinitionDumper, AST::FunctionDefinition&, dumpASTNode);
142MAKE_PRINT_ADAPTOR(NativeFunctionDeclarationDumper, AST::NativeFunctionDeclaration&, dumpASTNode);
143MAKE_PRINT_ADAPTOR(TypeDumper, AST::UnnamedType&, dumpASTNode);

146static ALWAYS_INLINE void dumpAST(Program& program)
147{
148 dataLogLn(ProgramDumper(program));
