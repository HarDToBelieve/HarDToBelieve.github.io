42class DuplicateFunctionKey {
43public:
44 DuplicateFunctionKey() = default;
45 DuplicateFunctionKey(WTF::HashTableDeletedValueType)
46 {
47 m_function = bitwise_cast<AST::FunctionDeclaration*>(static_cast<uintptr_t>(1));
48 }
49
50 DuplicateFunctionKey(const AST::FunctionDeclaration& function)
51 : m_function(&function)
52 { }
53
54 bool isEmptyValue() const { return !m_function; }
55 bool isHashTableDeletedValue() const { return m_function == bitwise_cast<AST::FunctionDeclaration*>(static_cast<uintptr_t>(1)); }
56
57 unsigned hash() const
58 {
59 unsigned hash = IntHash<size_t>::hash(m_function->parameters().size());
60 hash ^= m_function->name().hash();
61 for (size_t i = 0; i < m_function->parameters().size(); ++i)
62 hash ^= m_function->parameters()[i]->type()->hash();
63
64 if (m_function->isCast())
65 hash ^= m_function->type().hash();
66
67 return hash;
68 }
69
70 bool operator==(const DuplicateFunctionKey& other) const
71 {
72 if (m_function->parameters().size() != other.m_function->parameters().size())
73 return false;
74
75 if (m_function->name() != other.m_function->name())
76 return false;
77
78 if (m_function->nameSpace() != AST::NameSpace::StandardLibrary
79 && other.m_function->nameSpace() != AST::NameSpace::StandardLibrary
80 && m_function->nameSpace() != other.m_function->nameSpace())
81 return false;
82
83 ASSERT(m_function->isCast() == other.m_function->isCast());
84
85 for (size_t i = 0; i < m_function->parameters().size(); ++i) {
86 if (!matches(*m_function->parameters()[i]->type(), *other.m_function->parameters()[i]->type()))
87 return false;
88 }
89
90 if (!m_function->isCast())
91 return true;
92
93 if (matches(m_function->type(), other.m_function->type()))
94 return true;
95
96 return false;
97 }
98
99 struct Hash {
100 static unsigned hash(const DuplicateFunctionKey& key)
101 {
102 return key.hash();
103 }
104
105 static bool equal(const DuplicateFunctionKey& a, const DuplicateFunctionKey& b)
106 {
107 return a == b;
108 }
109
110 static const bool safeToCompareToEmptyOrDeleted = false;
111 };
112
113 struct Traits : public WTF::SimpleClassHashTraits<DuplicateFunctionKey> {
114 static const bool hasIsEmptyValueFunction = true;
115 static bool isEmptyValue(const DuplicateFunctionKey& key) { return key.isEmptyValue(); }
116 };
117
118private:
119 const AST::FunctionDeclaration* m_function { nullptr };
120};

122Expected<void, Error> checkDuplicateFunctions(const Program& program)
123{
124 auto passesStaticChecks = [&] (const AST::FunctionDeclaration& function) -> Expected<void, Error> {
125 if (function.name() == "operator=="
126 && function.parameters().size() == 2
127 && is<AST::ReferenceType>(static_cast<const AST::UnnamedType&>(*function.parameters()[0]->type()))
128 && is<AST::ReferenceType>(static_cast<const AST::UnnamedType&>(*function.parameters()[1]->type()))
129 && matches(*function.parameters()[0]->type(), *function.parameters()[1]->type()))
130 return makeUnexpected(Error("Cannot define operator== on two reference types."));
131 else if (function.isCast() && function.parameters().isEmpty()) {
132 auto& unifyNode = function.type().unifyNode();
133 if (is<AST::NamedType>(unifyNode) && is<AST::NativeTypeDeclaration>(downcast<AST::NamedType>(unifyNode))) {
134 auto& nativeTypeDeclaration = downcast<AST::NativeTypeDeclaration>(downcast<AST::NamedType>(unifyNode));
135 if (nativeTypeDeclaration.isOpaqueType())
136 return makeUnexpected(Error("Cannot define a cast on an opaque type."));
137 }
138 }
139
140 return { };
141 };
142
143 HashSet<DuplicateFunctionKey, DuplicateFunctionKey::Hash, DuplicateFunctionKey::Traits> functions;
144
145 auto add = [&] (const AST::FunctionDeclaration& function) -> Expected<void, Error> {
146 auto addResult = functions.add(DuplicateFunctionKey { function });
147 if (!addResult.isNewEntry)
148 return makeUnexpected(Error("Found duplicate function"));
149 return passesStaticChecks(function);
150 };
151
152 for (auto& functionDefinition : program.functionDefinitions()) {
153 auto addResult = add(functionDefinition.get());
154 if (!addResult)
155 return addResult;
156 }
157
158 for (auto& nativeFunctionDeclaration : program.nativeFunctionDeclarations()) {
159 // We generate duplicate native function declarations in synthesize constructors.
160 // FIXME: is this right?
161 // https://bugs.webkit.org/show_bug.cgi?id=198580
162 //
163 // Since we do that, we just need to make sure no native function is a duplicate
164 // of a user-defined function.
165
166 // FIXME: Add back this assert once we begin to auto generate these in the compiler
167 // instead of having them in the stdlib
168 // https://bugs.webkit.org/show_bug.cgi?id=198861
169 // ASSERT(passesStaticChecks(nativeFunctionDeclaration.get()));
170 if (functions.contains(DuplicateFunctionKey { nativeFunctionDeclaration.get() }))
171 return makeUnexpected(Error("Duplicate native function."));
172 }
173
174 return { };
175}
