43class TextureReferencesChecker final : public Visitor {
44public:
45 TextureReferencesChecker() = default;
46
47 virtual ~TextureReferencesChecker() = default;
48
49private:
50 void visit(AST::PointerType&) override;
51 void visit(AST::ArrayReferenceType&) override;
52 void visit(AST::ArrayType&) override;
53 void visit(AST::Expression&) override;
54 void visit(AST::FunctionDefinition&) override;
55 void visit(AST::NativeFunctionDeclaration&) override;
56
57 bool containsTextureOrSampler(AST::UnnamedType&);
58};
59
60class Searcher : public Visitor {
61public:
62 Searcher() = default;
63
64 virtual ~Searcher() = default;
65
66 bool found() const { return m_found; }
67
68private:
69 void visit(AST::NativeTypeDeclaration&);
70
71 bool m_found { false };
72};
73
74void Searcher::visit(AST::NativeTypeDeclaration& nativeTypeDeclaration)
75{
76 if (nativeTypeDeclaration.isOpaqueType())
77 m_found = true;
78}
79
80bool TextureReferencesChecker::containsTextureOrSampler(AST::UnnamedType& unnamedType)
81{
82 Searcher searcher;
83 searcher.checkErrorAndVisit(unnamedType);
84 return searcher.found();
85}
86
87void TextureReferencesChecker::visit(AST::PointerType& pointerType)
88{
89 Visitor::visit(pointerType);
90 if (hasError())
91 return;
92 if (containsTextureOrSampler(pointerType.elementType()))
93 setError(Error("Cannot have a pointer to a texture or sampler.", pointerType.codeLocation()));
94}
95
96void TextureReferencesChecker::visit(AST::ArrayReferenceType& arrayReferenceType)
97{
98 Visitor::visit(arrayReferenceType);
99 if (hasError())
100 return;
101 if (containsTextureOrSampler(arrayReferenceType.elementType()))
102 setError(Error("Cannot have an array reference to a texture or sampler.", arrayReferenceType.codeLocation()));
103}
104
105void TextureReferencesChecker::visit(AST::ArrayType& arrayType)
106{
107 Visitor::visit(arrayType);
108 if (hasError())
109 return;
110 if (containsTextureOrSampler(arrayType.type()))
111 setError(Error("Cannot have an array of textures or samplers.", arrayType.codeLocation()));
112}
113
114void TextureReferencesChecker::visit(AST::Expression& expression)
115{
116 Visitor::visit(expression);
117 checkErrorAndVisit(expression.resolvedType());
118}
119
120void TextureReferencesChecker::visit(AST::FunctionDefinition& functionDefinition)
121{
122 if (functionDefinition.parsingMode() != ParsingMode::StandardLibrary)
123 Visitor::visit(functionDefinition);
124}
125
126void TextureReferencesChecker::visit(AST::NativeFunctionDeclaration&)
127{
128}
129
130Expected<void, Error> checkTextureReferences(Program& program)
131{
132 TextureReferencesChecker textureReferencesChecker;
133 textureReferencesChecker.checkErrorAndVisit(program);
134 return textureReferencesChecker.result();
135}
