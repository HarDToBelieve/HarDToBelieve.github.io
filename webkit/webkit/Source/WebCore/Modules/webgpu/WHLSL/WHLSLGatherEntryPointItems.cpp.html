47class Gatherer : public Visitor {
48public:
49 Gatherer(const Intrinsics& intrinsics, AST::Semantic* semantic = nullptr)
50 : m_intrinsics(intrinsics)
51 , m_currentSemantic(semantic)
52 {
53 }
54
55 virtual ~Gatherer() = default;
56
57 void reset()
58 {
59 m_currentSemantic = nullptr;
60 m_currentVariableLocation = { };
61 }
62
63 Vector<EntryPointItem> takeEntryPointItems()
64 {
65 return WTFMove(m_entryPointItems);
66 }
67
68 void visit(AST::EnumerationDefinition&)
69 {
70 if (!m_currentSemantic) {
71 setError(Error("Expected semantic for entrypoint argument.", m_currentVariableLocation));
72 return;
73 }
74 m_entryPointItems.append(EntryPointItem(m_typeReferences.last().get(), *m_currentSemantic, m_path));
75 }
76
77 void visit(AST::NativeTypeDeclaration& nativeTypeDeclaration)
78 {
79 if (!m_currentSemantic) {
80 setError(Error("Expected semantic for entrypoint argument.", m_currentVariableLocation));
81 return;
82 }
83 if (matches(nativeTypeDeclaration, m_intrinsics.voidType())) {
84 setError(Error("Unexpected void type for entrypoint argument.", m_currentVariableLocation));
85 return;
86 }
87
88 m_entryPointItems.append(EntryPointItem(m_typeReferences.last().get(), *m_currentSemantic, m_path));
89 }
90
91 void visit(AST::StructureDefinition& structureDefinition)
92 {
93 if (m_currentSemantic) {
94 setError(Error("Unexpected semantic for struct entrypoint argument.", m_currentVariableLocation));
95 return;
96 }
97
98 for (auto& structureElement : structureDefinition.structureElements()) {
99 if (structureElement.semantic())
100 m_currentSemantic = structureElement.semantic();
101 m_path.append(structureElement.name());
102 checkErrorAndVisit(structureElement);
103 m_path.takeLast();
104 }
105 }
106
107 void visit(AST::TypeDefinition& typeDefinition)
108 {
109 checkErrorAndVisit(typeDefinition.type());
110 }
111
112 void visit(AST::TypeReference& typeReference)
113 {
114 m_typeReferences.append(typeReference);
115 auto depth = m_typeReferences.size();
116 checkErrorAndVisit(typeReference.resolvedType());
117 ASSERT_UNUSED(depth, m_typeReferences.size() == depth);
118 m_typeReferences.removeLast();
119 }
120
121 void visit(AST::PointerType& pointerType)
122 {
123 if (!m_currentSemantic) {
124 setError(Error("Expected semantic for entrypoint argument.", m_currentVariableLocation));
125 return;
126 }
127 m_entryPointItems.append(EntryPointItem(pointerType, *m_currentSemantic, m_path));
128 }
129
130 void visit(AST::ArrayReferenceType& arrayReferenceType)
131 {
132 if (!m_currentSemantic) {
133 setError(Error("Expected semantic for entrypoint argument.", m_currentVariableLocation));
134 return;
135 }
136 m_entryPointItems.append(EntryPointItem(arrayReferenceType, *m_currentSemantic, m_path));
137 }
138
139 void visit(AST::ArrayType& arrayType)
140 {
141 if (!m_currentSemantic) {
142 setError(Error("Expected semantic for entrypoint argument.", m_currentVariableLocation));
143 return;
144 }
145 m_entryPointItems.append(EntryPointItem(arrayType, *m_currentSemantic, m_path));
146 }
147
148 void visit(AST::VariableDeclaration& variableDeclaration)
149 {
150 ASSERT(!m_currentSemantic);
151 m_currentVariableLocation = variableDeclaration.codeLocation();
152 if (variableDeclaration.semantic())
153 m_currentSemantic = variableDeclaration.semantic();
154 ASSERT(variableDeclaration.type());
155 m_path.append(variableDeclaration.name());
156 checkErrorAndVisit(*variableDeclaration.type());
157 m_path.takeLast();
158 }
159
160private:
161 Vector<String> m_path;
162 const Intrinsics& m_intrinsics;
163 AST::Semantic* m_currentSemantic { nullptr };
164 CodeLocation m_currentVariableLocation;
165 Vector<std::reference_wrapper<AST::TypeReference>> m_typeReferences;
166 Vector<EntryPointItem> m_entryPointItems;
167};
168
169Expected<EntryPointItems, Error> gatherEntryPointItems(const Intrinsics& intrinsics, AST::FunctionDefinition& functionDefinition)
170{
171 ASSERT(functionDefinition.entryPointType());
172 Gatherer inputGatherer(intrinsics);
173 for (auto& parameter : functionDefinition.parameters()) {
174 inputGatherer.reset();
175 inputGatherer.checkErrorAndVisit(parameter);
176 if (inputGatherer.hasError())
177 return makeUnexpected(inputGatherer.result().error());
178 }
179 Gatherer outputGatherer(intrinsics, functionDefinition.semantic());
180 if (*functionDefinition.entryPointType() != AST::EntryPointType::Compute)
181 outputGatherer.checkErrorAndVisit(functionDefinition.type());
182 if (outputGatherer.hasError())
183 return makeUnexpected(outputGatherer.result().error());
184
185 return {{ inputGatherer.takeEntryPointItems(), outputGatherer.takeEntryPointItems() }};
186}
