172class Lexer {
173public:
174 Lexer() = default;
175
176 Lexer(StringView stringView, AST::NameSpace nameSpace)
177 : m_stringView(stringView)
178 , m_nameSpace(nameSpace)
179 {
180 skipWhitespaceAndComments();
181 m_ringBuffer[0] = consumeTokenFromStream();
182 m_ringBuffer[1] = consumeTokenFromStream();
183 }
184
185 Lexer(const Lexer&) = delete;
186 Lexer(Lexer&&) = default;
187
188 Lexer& operator=(const Lexer&) = delete;
189 Lexer& operator=(Lexer&&) = default;
190
191
192 Token consumeToken()
193 {
194 auto result = m_ringBuffer[m_ringBufferIndex];
195 m_ringBuffer[m_ringBufferIndex] = consumeTokenFromStream();
196 m_ringBufferIndex = (m_ringBufferIndex + 1) % 2;
197 return result;
198 }
199
200 Token peek() const
201 {
202 return m_ringBuffer[m_ringBufferIndex];
203 }
204
205 Token peekFurther() const
206 {
207 return m_ringBuffer[(m_ringBufferIndex + 1) % 2];
208 }
209
210 // FIXME: We should not need this
211 // https://bugs.webkit.org/show_bug.cgi?id=198357
212 struct State {
213 Token ringBuffer[2];
214 unsigned ringBufferIndex;
215 unsigned offset;
216 };
217
218 State state() const
219 {
220 State state;
221 state.ringBuffer[0] = m_ringBuffer[0];
222 state.ringBuffer[1] = m_ringBuffer[1];
223 state.ringBufferIndex = m_ringBufferIndex;
224 state.offset = m_offset;
225 return state;
226 }
227
228 void setState(const State& state)
229 {
230 m_ringBuffer[0] = state.ringBuffer[0];
231 m_ringBuffer[1] = state.ringBuffer[1];
232 m_ringBufferIndex = state.ringBufferIndex;
233 m_offset = state.offset;
234
235 }
236
237 void setState(State&& state)
238 {
239 m_ringBuffer[0] = WTFMove(state.ringBuffer[0]);
240 m_ringBuffer[1] = WTFMove(state.ringBuffer[1]);
241 m_ringBufferIndex = WTFMove(state.ringBufferIndex);
242 m_offset = WTFMove(state.offset);
243 }
244
245 bool isFullyConsumed() const
246 {
247 return peek().type == Token::Type::EndOfFile;
248 }
249
250 String errorString(const Token& token, const String& message)
251 {
252 return makeString("Parse error at line ", lineAndColumnNumberFromOffset(m_stringView, token.startOffset()).line, ": ", message);
253 }
254
255 static String errorString(Error, const String& source1, const String* source2 = nullptr);
256
257 AST::NameSpace nameSpace() const { return m_nameSpace; }
258
259private:
260 friend struct Token;
261 Token consumeTokenFromStream();
262
263 void skipWhitespaceAndComments();
264
265 struct LineAndColumn {
266 unsigned line;
267 unsigned column;
268 };
269 static LineAndColumn lineAndColumnNumberFromOffset(const StringView&, unsigned offset);
270
271 StringView m_stringView;
272 Token m_ringBuffer[2];
273 unsigned m_ringBufferIndex { 0 };
274 unsigned m_offset { 0 };
275 AST::NameSpace m_nameSpace { AST::NameSpace::StandardLibrary };
276};
277
278StringView Token::stringView(const Lexer& lexer) const
279{
280 return lexer.m_stringView.substring(codeLocation.startOffset(), codeLocation.endOffset() - codeLocation.startOffset());
281}
