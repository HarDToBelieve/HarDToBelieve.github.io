41#if ASSERT_ENABLED
42static AST::NativeTypeDeclaration* getNativeTypeDeclaration(AST::ResolvableType& resolvableType)
43{
44 if (!is<AST::TypeReference>(resolvableType.resolvedType()))
45 return nullptr;
46 auto& typeReference = downcast<AST::TypeReference>(resolvableType.resolvedType());
47 if (!is<AST::NativeTypeDeclaration>(typeReference.resolvedType()))
48 return nullptr;
49 return &downcast<AST::NativeTypeDeclaration>(typeReference.resolvedType());
50}
51
52class LiteralTypeChecker : public Visitor {
53public:
54private:
55 void visit(AST::FloatLiteralType& floatLiteralType) override
56 {
57 auto* nativeTypeDeclaration = getNativeTypeDeclaration(floatLiteralType);
58 ASSERT(nativeTypeDeclaration);
59 ASSERT(nativeTypeDeclaration->canRepresentFloat()(floatLiteralType.value()));
60 }
61
62 void visit(AST::IntegerLiteralType& integerLiteralType) override
63 {
64 auto* nativeTypeDeclaration = getNativeTypeDeclaration(integerLiteralType);
65 ASSERT(nativeTypeDeclaration);
66 ASSERT(nativeTypeDeclaration->canRepresentInteger()(integerLiteralType.value()));
67 }
68
69 void visit(AST::UnsignedIntegerLiteralType& unsignedIntegerLiteralType) override
70 {
71 auto* nativeTypeDeclaration = getNativeTypeDeclaration(unsignedIntegerLiteralType);
72 ASSERT(nativeTypeDeclaration);
73 ASSERT(nativeTypeDeclaration->canRepresentUnsignedInteger()(unsignedIntegerLiteralType.value()));
74 }
75};
76#endif // ASSERT_ENABLED
77
78void checkLiteralTypes(Program& program)
79{
80#if ASSERT_ENABLED
81 LiteralTypeChecker().Visitor::visit(program);
82#else
83 UNUSED_PARAM(program);
84#endif
85}
