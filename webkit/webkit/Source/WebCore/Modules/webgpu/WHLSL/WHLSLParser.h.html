49 // https://bugs.webkit.org/show_bug.cgi?id=198357
50 template<typename T> T backtrackingScope(std::function<T()> callback)
51 {
52 auto state = m_lexer.state();
53 auto result = callback();
54 if (result)
55 return result;
56 m_lexer.setState(WTFMove(state));
57 return result;
58 }
