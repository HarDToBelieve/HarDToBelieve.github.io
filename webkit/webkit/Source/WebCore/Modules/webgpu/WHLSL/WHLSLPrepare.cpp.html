90static constexpr bool dumpASTBeforeEachPass = false;
91static constexpr bool dumpASTAfterParsing = false;
92static constexpr bool dumpASTAtEnd = false;
93static constexpr bool alwaysDumpPassFailures = false;
94static constexpr bool dumpPassFailure = dumpASTBeforeEachPass || dumpASTAfterParsing || dumpASTAtEnd || alwaysDumpPassFailures;
95static constexpr bool dumpPhaseTimes = false;
96
97static constexpr bool parseFullStandardLibrary = false;
98
99static bool dumpASTIfNeeded(bool shouldDump, Program& program, const char* message)
100{
101 if (shouldDump) {
102 dataLogLn(message);
103 dumpAST(program);
104 return true;
105 }
106
107 return false;
108}
109
110static bool dumpASTAfterParsingIfNeeded(Program& program)
111{
112 return dumpASTIfNeeded(dumpASTAfterParsing, program, "AST after parsing");
113}
114
115static bool dumpASTBetweenEachPassIfNeeded(Program& program, const char* message)
116{
117 return dumpASTIfNeeded(dumpASTBeforeEachPass, program, message);
118}
119
120static bool dumpASTAtEndIfNeeded(Program& program)
121{
122 return dumpASTIfNeeded(dumpASTAtEnd, program, "AST at end");
123}

125using PhaseTimes = Vector<std::pair<String, Seconds>>;
126
127static void logPhaseTimes(PhaseTimes& phaseTimes)
128{
129 if (!dumpPhaseTimes)
130 return;
131
132 for (auto& entry : phaseTimes)
133 dataLogLn(entry.first, ": ", entry.second.milliseconds(), " ms");
134}
135
136class PhaseTimer {
137public:
138 PhaseTimer(const char* phaseName, PhaseTimes& phaseTimes)
139 : m_phaseTimes(phaseTimes)
140 {
141 if (dumpPhaseTimes) {
142 m_phaseName = phaseName;
143 m_start = MonotonicTime::now();
144 }
145 }
146
147 ~PhaseTimer()
148 {
149 if (dumpPhaseTimes) {
150 auto totalTime = MonotonicTime::now() - m_start;
151 m_phaseTimes.append({ m_phaseName, totalTime });
152 }
153 }
154
155private:
156 String m_phaseName;
157 PhaseTimes& m_phaseTimes;
158 MonotonicTime m_start;
159};

167#define CHECK_PASS(pass, ...) \
168 do { \
169 dumpASTBetweenEachPassIfNeeded(program, "AST before " # pass); \
170 PhaseTimer phaseTimer(#pass, phaseTimes); \
171 auto result = pass(__VA_ARGS__); \
172 if (!result) { \
173 if (dumpPassFailure) \
174 dataLogLn("failed pass: " # pass, Lexer::errorString(result.error(), whlslSource1, whlslSource2)); \
175 return makeUnexpected(Lexer::errorString(result.error(), whlslSource1, whlslSource2)); \
176 } \
177 } while (0)
178
179#define RUN_PASS(pass, ...) \
180 do { \
181 PhaseTimer phaseTimer(#pass, phaseTimes); \
182 dumpASTBetweenEachPassIfNeeded(program, "AST before " # pass); \
183 pass(__VA_ARGS__); \
184 } while (0)

186static Expected<Program, String> prepareShared(PhaseTimes& phaseTimes, const String& whlslSource1, const String* whlslSource2 = nullptr)
187{
188 Program program;
189 Parser parser;
190
191 {
192 program.nameContext().setCurrentNameSpace(AST::NameSpace::NameSpace1);
193
194 PhaseTimer phaseTimer("parse", phaseTimes);
195 auto parseResult = parser.parse(program, whlslSource1, ParsingMode::User, AST::NameSpace::NameSpace1);
196 if (!parseResult) {
197 if (dumpPassFailure)
198 dataLogLn("failed to parse the program: ", Lexer::errorString(parseResult.error(), whlslSource1, whlslSource2));
199 return makeUnexpected(Lexer::errorString(parseResult.error(), whlslSource1, whlslSource2));
200 }
201 if (whlslSource2) {
202 program.nameContext().setCurrentNameSpace(AST::NameSpace::NameSpace2);
203 auto parseResult = parser.parse(program, *whlslSource2, ParsingMode::User, AST::NameSpace::NameSpace2);
204 if (!parseResult) {
205 if (dumpPassFailure)
206 dataLogLn("failed to parse the program: ", Lexer::errorString(parseResult.error(), whlslSource1, whlslSource2));
207 return makeUnexpected(Lexer::errorString(parseResult.error(), whlslSource1, whlslSource2));
208 }
209 }
210 program.nameContext().setCurrentNameSpace(AST::NameSpace::StandardLibrary);
211 }
212
213 {
214 PhaseTimer phaseTimer("includeStandardLibrary", phaseTimes);
215 includeStandardLibrary(program, parser, parseFullStandardLibrary);
216 }
217
218 if (!dumpASTBetweenEachPassIfNeeded(program, "AST after parsing"))
219 dumpASTAfterParsingIfNeeded(program);
220
221 NameResolver nameResolver(program.nameContext());
222 CHECK_PASS(resolveNamesInTypes, program, nameResolver);
223 CHECK_PASS(checkRecursiveTypes, program);
224 CHECK_PASS(synthesizeEnumerationFunctions, program);
225 CHECK_PASS(resolveTypeNamesInFunctions, program, nameResolver);
226 CHECK_PASS(synthesizeConstructors, program);
227 CHECK_PASS(checkDuplicateFunctions, program);
228
229 CHECK_PASS(check, program);
230 RUN_PASS(pruneUnreachableStandardLibraryFunctions, program);
231
232 RUN_PASS(checkLiteralTypes, program);
233 CHECK_PASS(checkTextureReferences, program);
234 CHECK_PASS(checkReferenceTypes, program);
235 RUN_PASS(resolveProperties, program);
236 RUN_PASS(findHighZombies, program);
237 CHECK_PASS(checkStatementBehavior, program);
238 CHECK_PASS(checkRecursion, program);
239 CHECK_PASS(checkFunctionStages, program);
240 RUN_PASS(preserveVariableLifetimes, program);
241
242 dumpASTAtEndIfNeeded(program);
243
244 return program;
245}

247Expected<RenderPrepareResult, String> prepare(const ShaderModule& vertexShaderModule, const ShaderModule* fragmentShaderModule, RenderPipelineDescriptor& renderPipelineDescriptor)
248{
249 PhaseTimes phaseTimes;
250 Metal::RenderMetalCode generatedCode;
251
252 {
253 PhaseTimer phaseTimer("prepare total", phaseTimes);
254 const String* secondShader = nullptr;
255 bool distinctFragmentShader = false;
256 if (fragmentShaderModule && fragmentShaderModule != &vertexShaderModule) {
257 secondShader = &fragmentShaderModule->whlslSource;
258 distinctFragmentShader = true;
259 }
260 auto program = prepareShared(phaseTimes, vertexShaderModule.whlslSource, secondShader);
261 if (!program)
262 return makeUnexpected(program.error());
263
264 std::optional<MatchedRenderSemantics> matchedSemantics;
265 {
266 PhaseTimer phaseTimer("matchSemantics", phaseTimes);
267 matchedSemantics = matchSemantics(*program, renderPipelineDescriptor, distinctFragmentShader, fragmentShaderModule);
268 if (!matchedSemantics)
269 return makeUnexpected(Lexer::errorString(Error("Could not match semantics"_str), vertexShaderModule.whlslSource, secondShader));
270 }
271
272 {
273 PhaseTimer phaseTimer("generateMetalCode", phaseTimes);
274 generatedCode = Metal::generateMetalCode(*program, WTFMove(*matchedSemantics), renderPipelineDescriptor.layout);
275 }
276 }
277
278 logPhaseTimes(phaseTimes);
279
280 RenderPrepareResult result;
281 result.metalSource = WTFMove(generatedCode.metalSource);
282 result.mangledVertexEntryPointName = WTFMove(generatedCode.mangledVertexEntryPointName);
283 result.mangledFragmentEntryPointName = WTFMove(generatedCode.mangledFragmentEntryPointName);
284 return result;
285}

287Expected<ComputePrepareResult, String> prepare(const ShaderModule& shaderModule, ComputePipelineDescriptor& computePipelineDescriptor)
288{
289 PhaseTimes phaseTimes;
290 Metal::ComputeMetalCode generatedCode;
291 std::optional<ComputeDimensions> computeDimensions;
292
293 {
294 PhaseTimer phaseTimer("prepare total", phaseTimes);
295 auto program = prepareShared(phaseTimes, shaderModule.whlslSource);
296 if (!program)
297 return makeUnexpected(program.error());
298
299 std::optional<MatchedComputeSemantics> matchedSemantics;
300 {
301 PhaseTimer phaseTimer("matchSemantics", phaseTimes);
302 matchedSemantics = matchSemantics(*program, computePipelineDescriptor);
303 if (!matchedSemantics)
304 return makeUnexpected(Lexer::errorString(Error("Could not match semantics"_str), shaderModule.whlslSource));
305 }
306
307 {
308 PhaseTimer phaseTimer("computeDimensions", phaseTimes);
309 computeDimensions = WHLSL::computeDimensions(*program, *matchedSemantics->shader);
310 if (!computeDimensions)
311 return makeUnexpected(Lexer::errorString(Error("Could not match compute dimensions"_str), shaderModule.whlslSource));
312 }
313
314 {
315 PhaseTimer phaseTimer("generateMetalCode", phaseTimes);
316 generatedCode = Metal::generateMetalCode(*program, WTFMove(*matchedSemantics), computePipelineDescriptor.layout);
317 }
318 }
319
320 logPhaseTimes(phaseTimes);
321
322 ComputePrepareResult result;
323 result.metalSource = WTFMove(generatedCode.metalSource);
324 result.mangledEntryPointName = WTFMove(generatedCode.mangledEntryPointName);
325 result.computeDimensions = WTFMove(*computeDimensions);
326 return result;
327}
