35bool Program::isValidVectorProperty(String property)
36{
37 if (property.length() < 1 || property.length() > 4)
38 return false;
39
40 for (size_t i = 0; i < property.length(); ++i) {
41 switch (property[i]) {
42 case 'x':
43 case 'y':
44 case 'z':
45 case 'w':
46 break;
47 default:
48 return false;
49 }
50 }
51
52 return true;
53}
