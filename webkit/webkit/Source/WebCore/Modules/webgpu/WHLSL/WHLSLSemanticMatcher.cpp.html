47static bool matchMode(Binding::BindingDetails bindingType, AST::ResourceSemantic::Mode mode)
48{
49 return WTF::visit(WTF::makeVisitor([&](UniformBufferBinding) -> bool {
50 return mode == AST::ResourceSemantic::Mode::Buffer;
51 }, [&](SamplerBinding) -> bool {
52 return mode == AST::ResourceSemantic::Mode::Sampler;
53 }, [&](TextureBinding) -> bool {
54 return mode == AST::ResourceSemantic::Mode::Texture;
55 }, [&](StorageBufferBinding) -> bool {
56 return mode == AST::ResourceSemantic::Mode::UnorderedAccessView;
57 }), bindingType);
58}

60static std::optional<HashMap<Binding*, size_t>> matchResources(Vector<EntryPointItem>& entryPointItems, Layout& layout, ShaderStage shaderStage)
61{
62 HashMap<Binding*, size_t> result;
63 HashSet<size_t, DefaultHash<size_t>, WTF::UnsignedWithZeroKeyHashTraits<size_t>> itemIndices;
64 for (auto& bindGroup : layout) {
65 auto space = bindGroup.name;
66 for (auto& binding : bindGroup.bindings) {
67 if (!binding.visibility.contains(shaderStage))
68 continue;
69 for (size_t i = 0; i < entryPointItems.size(); ++i) {
70 auto& item = entryPointItems[i];
71 auto& semantic = *item.semantic;
72 if (!WTF::holds_alternative<AST::ResourceSemantic>(semantic))
73 continue;
74 auto& resourceSemantic = WTF::get<AST::ResourceSemantic>(semantic);
75 if (!matchMode(binding.binding, resourceSemantic.mode()))
76 continue;
77 if (binding.externalName != resourceSemantic.index())
78 continue;
79 if (space != resourceSemantic.space())
80 continue;
81 result.add(&binding, i);
82 itemIndices.add(i);
83 }
84 }
85 }
86
87 for (size_t i = 0; i < entryPointItems.size(); ++i) {
88 auto& item = entryPointItems[i];
89 auto& semantic = *item.semantic;
90 if (!WTF::holds_alternative<AST::ResourceSemantic>(semantic))
91 continue;
92 if (!itemIndices.contains(i))
93 return std::nullopt;
94 }
95
96 return result;
97}

99static bool matchInputsOutputs(Vector<EntryPointItem>& vertexOutputs, Vector<EntryPointItem>& fragmentInputs)
100{
101 for (auto& fragmentInput : fragmentInputs) {
102 if (!WTF::holds_alternative<AST::StageInOutSemantic>(*fragmentInput.semantic))
103 continue;
104 auto& fragmentInputStageInOutSemantic = WTF::get<AST::StageInOutSemantic>(*fragmentInput.semantic);
105 bool found = false;
106 for (auto& vertexOutput : vertexOutputs) {
107 if (!WTF::holds_alternative<AST::StageInOutSemantic>(*vertexOutput.semantic))
108 continue;
109 auto& vertexOutputStageInOutSemantic = WTF::get<AST::StageInOutSemantic>(*vertexOutput.semantic);
110 if (fragmentInputStageInOutSemantic.index() == vertexOutputStageInOutSemantic.index()) {
111 if (matches(*fragmentInput.unnamedType, *vertexOutput.unnamedType)) {
112 found = true;
113 break;
114 }
115 return false;
116 }
117 }
118 if (!found)
119 return false;
120 }
121 return true;
122}

124static bool isAcceptableFormat(VertexFormat vertexFormat, AST::UnnamedType& unnamedType, Intrinsics& intrinsics)
125{
126 switch (vertexFormat) {
127 case VertexFormat::FloatR32G32B32A32:
128 return matches(unnamedType, intrinsics.float4Type());
129 case VertexFormat::FloatR32G32B32:
130 return matches(unnamedType, intrinsics.float3Type());
131 case VertexFormat::FloatR32G32:
132 return matches(unnamedType, intrinsics.float2Type());
133 default:
134 ASSERT(vertexFormat == VertexFormat::FloatR32);
135 return matches(unnamedType, intrinsics.floatType());
136 }
137}

139static std::optional<HashMap<VertexAttribute*, size_t>> matchVertexAttributes(Vector<EntryPointItem>& vertexInputs, VertexAttributes& vertexAttributes, Intrinsics& intrinsics)
140{
141 HashMap<VertexAttribute*, size_t> result;
142 HashSet<size_t, DefaultHash<size_t>, WTF::UnsignedWithZeroKeyHashTraits<size_t>> itemIndices;
143 for (auto& vertexAttribute : vertexAttributes) {
144 for (size_t i = 0; i < vertexInputs.size(); ++i) {
145 auto& item = vertexInputs[i];
146 auto& semantic = *item.semantic;
147 if (!WTF::holds_alternative<AST::StageInOutSemantic>(semantic))
148 continue;
149 auto& stageInOutSemantic = WTF::get<AST::StageInOutSemantic>(semantic);
150 if (stageInOutSemantic.index() != vertexAttribute.shaderLocation)
151 continue;
152 if (!isAcceptableFormat(vertexAttribute.vertexFormat, *item.unnamedType, intrinsics))
153 return std::nullopt;
154 result.add(&vertexAttribute, i);
155 itemIndices.add(i);
156 }
157 }
158
159 for (size_t i = 0; i < vertexInputs.size(); ++i) {
160 auto& item = vertexInputs[i];
161 auto& semantic = *item.semantic;
162 if (!WTF::holds_alternative<AST::StageInOutSemantic>(semantic))
163 continue;
164 if (!itemIndices.contains(i))
165 return std::nullopt;
166 }
167
168 return result;
169}

171static bool isAcceptableFormat(TextureFormat textureFormat, AST::UnnamedType& unnamedType, Intrinsics& intrinsics, bool isColor)
172{
173 if (isColor) {
174 switch (textureFormat) {
175 case TextureFormat::R8Unorm:
176 case TextureFormat::R8UnormSrgb:
177 case TextureFormat::R8Snorm:
178 case TextureFormat::R16Unorm:
179 case TextureFormat::R16Snorm:
180 case TextureFormat::R16Float:
181 case TextureFormat::R32Float:
182 return matches(unnamedType, intrinsics.floatType());
183 case TextureFormat::RG8Unorm:
184 case TextureFormat::RG8UnormSrgb:
185 case TextureFormat::RG8Snorm:
186 case TextureFormat::RG16Unorm:
187 case TextureFormat::RG16Snorm:
188 case TextureFormat::RG16Float:
189 case TextureFormat::RG32Float:
190 return matches(unnamedType, intrinsics.float2Type());
191 case TextureFormat::B5G6R5Unorm:
192 case TextureFormat::RG11B10Float:
193 return matches(unnamedType, intrinsics.float3Type());
194 case TextureFormat::RGBA8Unorm:
195 case TextureFormat::RGBA8UnormSrgb:
196 case TextureFormat::BGRA8Unorm:
197 case TextureFormat::BGRA8UnormSrgb:
198 case TextureFormat::RGBA8Snorm:
199 case TextureFormat::RGB10A2Unorm:
200 case TextureFormat::RGBA16Unorm:
201 case TextureFormat::RGBA16Snorm:
202 case TextureFormat::RGBA16Float:
203 case TextureFormat::RGBA32Float:
204 return matches(unnamedType, intrinsics.float4Type());
205 case TextureFormat::R32Uint:
206 return matches(unnamedType, intrinsics.uintType());
207 case TextureFormat::R32Sint:
208 return matches(unnamedType, intrinsics.intType());
209 case TextureFormat::RG32Uint:
210 return matches(unnamedType, intrinsics.uint2Type());
211 case TextureFormat::RG32Sint:
212 return matches(unnamedType, intrinsics.int2Type());
213 case TextureFormat::RGBA32Uint:
214 return matches(unnamedType, intrinsics.uint4Type());
215 case TextureFormat::RGBA32Sint:
216 return matches(unnamedType, intrinsics.int4Type());
217 default:
218 ASSERT_NOT_REACHED();
219 return false;
220 }
221 }
222 return false;
223}

225static std::optional<HashMap<AttachmentDescriptor*, size_t>> matchColorAttachments(Vector<EntryPointItem>& fragmentOutputs, Vector<AttachmentDescriptor>& attachmentDescriptors, Intrinsics& intrinsics)
226{
227 HashMap<AttachmentDescriptor*, size_t> result;
228 HashSet<size_t, DefaultHash<size_t>, WTF::UnsignedWithZeroKeyHashTraits<size_t>> itemIndices;
229 for (auto& attachmentDescriptor : attachmentDescriptors) {
230 for (size_t i = 0; i < fragmentOutputs.size(); ++i) {
231 auto& item = fragmentOutputs[i];
232 auto& semantic = *item.semantic;
233 if (!WTF::holds_alternative<AST::StageInOutSemantic>(semantic))
234 continue;
235 auto& stageInOutSemantic = WTF::get<AST::StageInOutSemantic>(semantic);
236 if (stageInOutSemantic.index() != attachmentDescriptor.name)
237 continue;
238 if (!isAcceptableFormat(attachmentDescriptor.textureFormat, *item.unnamedType, intrinsics, true))
239 return std::nullopt;
240 result.add(&attachmentDescriptor, i);
241 itemIndices.add(i);
242 }
243 }
244
245 for (size_t i = 0; i < fragmentOutputs.size(); ++i) {
246 auto& item = fragmentOutputs[i];
247 auto& semantic = *item.semantic;
248 if (!WTF::holds_alternative<AST::StageInOutSemantic>(semantic))
249 continue;
250 if (!itemIndices.contains(i))
251 return std::nullopt;
252 }
253
254 return result;
255}

257static bool matchDepthAttachment(Vector<EntryPointItem>& fragmentOutputs, std::optional<AttachmentDescriptor>& depthStencilAttachmentDescriptor, Intrinsics& intrinsics)
258{
259 auto iterator = std::find_if(fragmentOutputs.begin(), fragmentOutputs.end(), [&](EntryPointItem& item) {
260 auto& semantic = *item.semantic;
261 if (!WTF::holds_alternative<AST::BuiltInSemantic>(semantic))
262 return false;
263 auto& builtInSemantic = WTF::get<AST::BuiltInSemantic>(semantic);
264 return builtInSemantic.variable() == AST::BuiltInSemantic::Variable::SVDepth;
265 });
266 if (iterator == fragmentOutputs.end())
267 return true;
268
269 if (depthStencilAttachmentDescriptor) {
270 ASSERT(!depthStencilAttachmentDescriptor->name);
271 return isAcceptableFormat(depthStencilAttachmentDescriptor->textureFormat, *iterator->unnamedType, intrinsics, false);
272 }
273 return false;
274}

276std::optional<MatchedRenderSemantics> matchSemantics(Program& program, RenderPipelineDescriptor& renderPipelineDescriptor, bool distinctFragmentShader, bool fragmentShaderExists)
277{
278 auto vertexFunctions = program.nameContext().getFunctions(renderPipelineDescriptor.vertexEntryPointName, AST::NameSpace::NameSpace1);
279 if (vertexFunctions.size() != 1 || !vertexFunctions[0].get().entryPointType() || !is<AST::FunctionDefinition>(vertexFunctions[0].get()))
280 return std::nullopt;
281 auto& vertexShaderEntryPoint = downcast<AST::FunctionDefinition>(vertexFunctions[0].get());
282 auto vertexShaderEntryPointItems = gatherEntryPointItems(program.intrinsics(), vertexShaderEntryPoint);
283 if (!vertexShaderEntryPointItems)
284 return std::nullopt;
285 auto vertexShaderResourceMap = matchResources(vertexShaderEntryPointItems->inputs, renderPipelineDescriptor.layout, ShaderStage::Vertex);
286 if (!vertexShaderResourceMap)
287 return std::nullopt;
288 auto matchedVertexAttributes = matchVertexAttributes(vertexShaderEntryPointItems->inputs, renderPipelineDescriptor.vertexAttributes, program.intrinsics());
289 if (!matchedVertexAttributes)
290 return std::nullopt;
291 if (!fragmentShaderExists)
292 return {{ &vertexShaderEntryPoint, nullptr, *vertexShaderEntryPointItems, EntryPointItems(), *vertexShaderResourceMap, HashMap<Binding*, size_t>(), *matchedVertexAttributes, HashMap<AttachmentDescriptor*, size_t>() }};
293
294 auto fragmentNameSpace = distinctFragmentShader ? AST::NameSpace::NameSpace2 : AST::NameSpace::NameSpace1;
295 auto fragmentFunctions = program.nameContext().getFunctions(renderPipelineDescriptor.fragmentEntryPointName, fragmentNameSpace);
296 if (fragmentFunctions.size() != 1 || !fragmentFunctions[0].get().entryPointType() || !is<AST::FunctionDefinition>(fragmentFunctions[0].get()))
297 return std::nullopt;
298 auto& fragmentShaderEntryPoint = downcast<AST::FunctionDefinition>(fragmentFunctions[0].get());
299 auto fragmentShaderEntryPointItems = gatherEntryPointItems(program.intrinsics(), fragmentShaderEntryPoint);
300 if (!fragmentShaderEntryPointItems)
301 return std::nullopt;
302 auto fragmentShaderResourceMap = matchResources(fragmentShaderEntryPointItems->inputs, renderPipelineDescriptor.layout, ShaderStage::Fragment);
303 if (!fragmentShaderResourceMap)
304 return std::nullopt;
305 if (!matchInputsOutputs(vertexShaderEntryPointItems->outputs, fragmentShaderEntryPointItems->inputs))
306 return std::nullopt;
307 auto matchedColorAttachments = matchColorAttachments(fragmentShaderEntryPointItems->outputs, renderPipelineDescriptor.attachmentsStateDescriptor.attachmentDescriptors, program.intrinsics());
308 if (!matchedColorAttachments)
309 return std::nullopt;
310 if (!matchDepthAttachment(fragmentShaderEntryPointItems->outputs, renderPipelineDescriptor.attachmentsStateDescriptor.depthStencilAttachmentDescriptor, program.intrinsics()))
311 return std::nullopt;
312 return {{ &vertexShaderEntryPoint, &fragmentShaderEntryPoint, *vertexShaderEntryPointItems, *fragmentShaderEntryPointItems, *vertexShaderResourceMap, *fragmentShaderResourceMap, *matchedVertexAttributes, *matchedColorAttachments }};
313}

315std::optional<MatchedComputeSemantics> matchSemantics(Program& program, ComputePipelineDescriptor& computePipelineDescriptor)
316{
317 auto functions = program.nameContext().getFunctions(computePipelineDescriptor.entryPointName, AST::NameSpace::NameSpace1);
318 if (functions.size() != 1 || !functions[0].get().entryPointType() || !is<AST::FunctionDefinition>(functions[0].get()))
319 return std::nullopt;
320 auto& entryPoint = downcast<AST::FunctionDefinition>(functions[0].get());
321 auto entryPointItems = gatherEntryPointItems(program.intrinsics(), entryPoint);
322 if (!entryPointItems)
323 return std::nullopt;
324 ASSERT(entryPointItems->outputs.isEmpty());
325 auto resourceMap = matchResources(entryPointItems->inputs, computePipelineDescriptor.layout, ShaderStage::Compute);
326 if (!resourceMap)
327 return std::nullopt;
328 return {{ &entryPoint, *entryPointItems, *resourceMap }};
329}
