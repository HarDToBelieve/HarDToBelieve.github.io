180 bool hasError() const { return !m_expectedError; }
181 Expected<void, Error> result() { return m_expectedError; }
182
183 template<typename T> void checkErrorAndVisit(T& x)
184 {
185 if (!hasError())
186 visit(x);
187 }
188
189protected:
190 void setError(Error error)
191 {
192 ASSERT(!hasError());
193 m_expectedError = makeUnexpected(error);
194 }
195
196private:
197 Expected<void, Error> m_expectedError;
