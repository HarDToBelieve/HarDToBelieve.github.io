37Ref<WebGPUBuffer> WebGPUBuffer::create(RefPtr<GPUBuffer>&& buffer, GPUErrorScopes& errorScopes)
38{
39 return adoptRef(*new WebGPUBuffer(WTFMove(buffer), errorScopes));
40}
41
42WebGPUBuffer::WebGPUBuffer(RefPtr<GPUBuffer>&& buffer, GPUErrorScopes& errorScopes)
43 : GPUObjectBase(makeRef(errorScopes))
44 , m_buffer(WTFMove(buffer))
45{
46}
47
48void WebGPUBuffer::mapReadAsync(BufferMappingPromise&& promise)
49{
50 rejectOrRegisterPromiseCallback(WTFMove(promise), true);
51}
52
53void WebGPUBuffer::mapWriteAsync(BufferMappingPromise&& promise)
54{
55 rejectOrRegisterPromiseCallback(WTFMove(promise), false);
56}
57
58void WebGPUBuffer::unmap()
59{
60 errorScopes().setErrorPrefix("GPUBuffer.unmap(): ");
61
62 if (!m_buffer)
63 errorScopes().generatePrefixedError("Invalid operation: invalid GPUBuffer!");
64 else
65 m_buffer->unmap(&errorScopes());
66}
67
68void WebGPUBuffer::destroy()
69{
70 errorScopes().setErrorPrefix("GPUBuffer.destroy(): ");
71
72 if (!m_buffer)
73 errorScopes().generatePrefixedError("Invalid operation!");
74 else {
75 m_buffer->destroy(&errorScopes());
76 m_buffer = nullptr;
77 }
78}
79
80void WebGPUBuffer::rejectOrRegisterPromiseCallback(BufferMappingPromise&& promise, bool isRead)
81{
82 errorScopes().setErrorPrefix(makeString("GPUBuffer.map", isRead ? "Read" : "Write", "Async(): "));
83
84 if (!m_buffer) {
85 errorScopes().generatePrefixedError("Invalid operation: invalid GPUBuffer!");
86 promise.reject();
87 return;
88 }
89
90 m_buffer->registerMappingCallback([promise = WTFMove(promise), protectedErrorScopes = makeRef(errorScopes())] (JSC::ArrayBuffer* arrayBuffer) mutable {
91 if (arrayBuffer)
92 promise.resolve(*arrayBuffer);
93 else {
94 // FIXME: It's possible to hit this code path in response to an unmap call, where
95 // the ArrayBuffer will definitely be null, and thus an out of memory error makes no sense.
96 // https://bugs.webkit.org/show_bug.cgi?id=204166
97 protectedErrorScopes->generateError("", GPUErrorFilter::OutOfMemory);
98 promise.reject();
99 }
100 }, isRead, errorScopes());
101}
