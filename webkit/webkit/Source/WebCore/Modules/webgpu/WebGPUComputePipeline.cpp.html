43Ref<WebGPUComputePipeline> WebGPUComputePipeline::create(WebGPUDevice& device, RefPtr<GPUComputePipeline>&& pipeline, GPUErrorScopes& errorScopes, WebGPUPipeline::ShaderData&& computeShader)
44{
45 return adoptRef(*new WebGPUComputePipeline(device, WTFMove(pipeline), errorScopes, WTFMove(computeShader)));
46}
47
48WebGPUComputePipeline::WebGPUComputePipeline(WebGPUDevice& device, RefPtr<GPUComputePipeline>&& pipeline, GPUErrorScopes& errorScopes, WebGPUPipeline::ShaderData&& computeShader)
49 : WebGPUPipeline(device, errorScopes)
50 , m_computePipeline(WTFMove(pipeline))
51 , m_computeShader(WTFMove(computeShader))
52{
53}
54
55WebGPUComputePipeline::~WebGPUComputePipeline() = default;
56
57bool WebGPUComputePipeline::cloneShaderModules(const WebGPUDevice& device)
58{
59 if (m_computeShader.module) {
60 const auto& computeSource = m_computeShader.module->source();
61 m_computeShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { computeSource }), computeSource);
62 return true;
63 }
64 return false;
65}
66
67bool WebGPUComputePipeline::recompile(const WebGPUDevice& device)
68{
69 if (m_computePipeline && m_computeShader.module) {
70 if (auto* gpuComputeShaderModule = m_computeShader.module->module()) {
71 GPUProgrammableStageDescriptor computeStage(makeRef(*gpuComputeShaderModule), { m_computeShader.entryPoint });
72 return m_computePipeline->recompile(device.device(), WTFMove(computeStage));
73 }
74 }
75 return false;
76}
