36std::optional<GPUPipelineLayoutDescriptor> WebGPUPipelineLayoutDescriptor::tryCreateGPUPipelineLayoutDescriptor() const
37{
38 Vector<Ref<const GPUBindGroupLayout>> gpuLayouts;
39 gpuLayouts.reserveCapacity(bindGroupLayouts.size());
40
41 for (const auto& layout : bindGroupLayouts) {
42 if (!layout || !layout->bindGroupLayout()) {
43 LOG(WebGPU, "GPUDevice::createPipelineLayout(): Invalid GPUBindGroupLayout!");
44 return std::nullopt;
45 }
46
47 gpuLayouts.uncheckedAppend(makeRef(*layout->bindGroupLayout()));
48 }
49
50 return GPUPipelineLayoutDescriptor { WTFMove(gpuLayouts) };
51}
