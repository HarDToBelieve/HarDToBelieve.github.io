37WorkerNavigatorGPU* WorkerNavigatorGPU::from(WorkerNavigator* navigator)
38{
39 WorkerNavigatorGPU* supplement = static_cast<WorkerNavigatorGPU*>(Supplement<WorkerNavigator>::from(navigator, supplementName()));
40 if (!supplement) {
41 auto newSupplement = makeUnique<WorkerNavigatorGPU>();
42 supplement = newSupplement.get();
43 provideTo(navigator, supplementName(), WTFMove(newSupplement));
44 }
45 return supplement;
46}
47
48const char* WorkerNavigatorGPU::supplementName()
49{
50 return "WorkerNavigatorGPU";
51}
52
53const WebGPU* WorkerNavigatorGPU::gpu(WorkerNavigator& navigator)
54{
55 auto scriptExecutionContext = navigator.scriptExecutionContext();
56 if (!scriptExecutionContext)
57 return nullptr;
58
59 ASSERT(scriptExecutionContext->settingsValues().webGPUEnabled);
60 return WorkerNavigatorGPU::from(&navigator)->gpu();
61}
62
63const WebGPU* WorkerNavigatorGPU::gpu() const
64{
65 if (!m_gpu)
66 m_gpu = WebGPU::create();
67 return m_gpu.get();
68}
