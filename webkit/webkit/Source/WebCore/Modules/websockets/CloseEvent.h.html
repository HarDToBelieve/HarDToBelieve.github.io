31#pragma once
32
33#include "Event.h"
34#include "EventNames.h"
35
36namespace WebCore {
37
38class CloseEvent final : public Event {
39 WTF_MAKE_ISO_ALLOCATED(CloseEvent);
40public:
41 static Ref<CloseEvent> create(bool wasClean, unsigned short code, const String& reason)
42 {
43 return adoptRef(*new CloseEvent(wasClean, code, reason));
44 }
45
46 struct Init : EventInit {
47 bool wasClean { false };
48 unsigned short code { 0 };
49 String reason;
50 };
51
52 static Ref<CloseEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
53 {
54 return adoptRef(*new CloseEvent(type, initializer, isTrusted));
55 }
56
57 bool wasClean() const { return m_wasClean; }
58 unsigned short code() const { return m_code; }
59 String reason() const { return m_reason; }
60
61 // Event function.
62 EventInterface eventInterface() const override { return CloseEventInterfaceType; }
63
64private:
65 CloseEvent(bool wasClean, int code, const String& reason)
66 : Event(eventNames().closeEvent, CanBubble::No, IsCancelable::No)
67 , m_wasClean(wasClean)
68 , m_code(code)
69 , m_reason(reason)
70 {
71 }
72
73 CloseEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
74 : Event(type, initializer, isTrusted)
75 , m_wasClean(initializer.wasClean)
76 , m_code(initializer.code)
77 , m_reason(initializer.reason)
78 {
79 }
80
81 bool m_wasClean;
82 unsigned short m_code;
83 String m_reason;
84};
85
86} // namespace WebCore
