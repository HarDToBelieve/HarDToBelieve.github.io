31#pragma once
32
33#include "WebSocketIdentifier.h"
34#include <wtf/Forward.h>
35#include <wtf/Noncopyable.h>
36#include <wtf/URL.h>
37
38namespace JSC {
39class ArrayBuffer;
40}
41
42namespace WebCore {
43
44class Blob;
45class Document;
46class ResourceRequest;
47class ResourceResponse;
48class ScriptExecutionContext;
49class SocketProvider;
50class WebSocketChannelClient;
51
52class ThreadableWebSocketChannel {
53 WTF_MAKE_NONCOPYABLE(ThreadableWebSocketChannel);
54public:
55 static Ref<ThreadableWebSocketChannel> create(Document&, WebSocketChannelClient&, SocketProvider&);
56 static Ref<ThreadableWebSocketChannel> create(ScriptExecutionContext&, WebSocketChannelClient&, SocketProvider&);
57 WEBCORE_EXPORT ThreadableWebSocketChannel();
58
59 void ref() { refThreadableWebSocketChannel(); }
60 void deref() { derefThreadableWebSocketChannel(); }
61
62 WebSocketIdentifier identifier() const { return m_identifier; };
63
64 enum class ConnectStatus { KO, OK };
65 virtual ConnectStatus connect(const URL&, const String& protocol) = 0;
66 virtual String subprotocol() = 0; // Will be available after didConnect() callback is invoked.
67 virtual String extensions() = 0; // Will be available after didConnect() callback is invoked.
68
69 enum SendResult { SendSuccess, SendFail };
70 virtual SendResult send(const String& message) = 0;
71 virtual SendResult send(const JSC::ArrayBuffer&, unsigned byteOffset, unsigned byteLength) = 0;
72 virtual SendResult send(Blob&) = 0;
73
74 virtual unsigned bufferedAmount() const = 0;
75 virtual void close(int code, const String& reason) = 0;
76 // Log the reason text and close the connection. Will call didClose().
77 virtual void fail(const String& reason) = 0;
78 virtual void disconnect() = 0; // Will suppress didClose().
79
80 virtual void suspend() = 0;
81 virtual void resume() = 0;
82
83 virtual unsigned long progressIdentifier() const = 0;
84 virtual bool hasCreatedHandshake() const = 0;
85 virtual bool isConnected() const = 0;
86 using CookieGetter = Function<String(const URL&)>;
87 virtual ResourceRequest clientHandshakeRequest(const CookieGetter&) const = 0;
88 virtual const ResourceResponse& serverHandshakeResponse() const = 0;
89
90protected:
91 virtual ~ThreadableWebSocketChannel() = default;
92 virtual void refThreadableWebSocketChannel() = 0;
93 virtual void derefThreadableWebSocketChannel() = 0;
94
95 struct ValidatedURL {
96 URL url;
97 bool areCookiesAllowed { true };
98 };
99 static std::optional<ValidatedURL> validateURL(Document&, const URL&);
100 WEBCORE_EXPORT static std::optional<ResourceRequest> webSocketConnectRequest(Document&, const URL&);
101
102 WebSocketIdentifier m_identifier;
103};
104
105} // namespace WebCore
