83static inline bool isValidProtocolCharacter(UChar character)
84{
85 // Hybi-10 says "(Subprotocol string must consist of) characters in the range U+0021 to U+007E not including
86 // separator characters as defined in [RFC2616]."
87 const UChar minimumProtocolCharacter = '!'; // U+0021.
88 const UChar maximumProtocolCharacter = '~'; // U+007E.
89 return character >= minimumProtocolCharacter && character <= maximumProtocolCharacter
90 && character != '"' && character != '(' && character != ')' && character != ',' && character != '/'
91 && !(character >= ':' && character <= '@') // U+003A - U+0040 (':', ';', '<', '=', '>', '?', '@').
92 && !(character >= '[' && character <= ']') // U+005B - U+005D ('[', '\\', ']').
93 && character != '{' && character != '}';
94}
95
96static bool isValidProtocolString(StringView protocol)
97{
98 if (protocol.isEmpty())
99 return false;
100 for (auto codeUnit : protocol.codeUnits()) {
101 if (!isValidProtocolCharacter(codeUnit))
102 return false;
103 }
104 return true;
105}

107static String encodeProtocolString(const String& protocol)
108{
109 StringBuilder builder;
110 for (size_t i = 0; i < protocol.length(); i++) {
111 if (protocol[i] < 0x20 || protocol[i] > 0x7E)
112 builder.append("\\u", hex(protocol[i], 4));
113 else if (protocol[i] == 0x5c)
114 builder.append("\\\\");
115 else
116 builder.append(protocol[i]);
117 }
118 return builder.toString();
119}

121static String joinStrings(const Vector<String>& strings, const char* separator)
122{
123 StringBuilder builder;
124 for (size_t i = 0; i < strings.size(); ++i) {
125 if (i)
126 builder.append(separator);
127 builder.append(strings[i]);
128 }
129 return builder.toString();
130}

132static unsigned saturateAdd(unsigned a, unsigned b)
133{
134 if (std::numeric_limits<unsigned>::max() - a < b)
135 return std::numeric_limits<unsigned>::max();
136 return a + b;

170ExceptionOr<Ref<WebSocket>> WebSocket::create(ScriptExecutionContext& context, const String& url, const Vector<String>& protocols)
171{
172 if (url.isNull())
173 return Exception { SyntaxError };
174
175 auto socket = adoptRef(*new WebSocket(context));
176 socket->suspendIfNeeded();
177
178 auto result = socket->connect(context.completeURL(url).string(), protocols);
179 if (result.hasException())
180 return result.releaseException();
181
182 return socket;
183}

211void WebSocket::failAsynchronously()
212{
213 m_pendingActivity = makePendingActivity(*this);
214
215 // We must block this connection. Instead of throwing an exception, we indicate this
216 // using the error event. But since this code executes as part of the WebSocket's
217 // constructor, we have to wait until the constructor has completed before firing the
218 // event; otherwise, users can't connect to the event.
219
220 scriptExecutionContext()->postTask([this, protectedThis = makeRef(*this)](auto&) {
221 this->dispatchOrQueueErrorEvent();
222 this->stop();
223 });
224}

226ExceptionOr<void> WebSocket::connect(const String& url, const Vector<String>& protocols)
227{
228 LOG(Network, "WebSocket %p connect() url='%s'", this, url.utf8().data());
229 m_url = URL(URL(), url);
230
231 ASSERT(scriptExecutionContext());
232 auto& context = *scriptExecutionContext();
233
234 if (!m_url.isValid()) {
235 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, "Invalid url for WebSocket " + m_url.stringCenterEllipsizedToLength());
236 m_state = CLOSED;
237 return Exception { SyntaxError };
238 }
239
240 if (!m_url.protocolIs("ws") && !m_url.protocolIs("wss")) {
241 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, "Wrong url scheme for WebSocket " + m_url.stringCenterEllipsizedToLength());
242 m_state = CLOSED;
243 return Exception { SyntaxError };
244 }
245 if (m_url.hasFragmentIdentifier()) {
246 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, "URL has fragment component " + m_url.stringCenterEllipsizedToLength());
247 m_state = CLOSED;
248 return Exception { SyntaxError };
249 }
250
251 ASSERT(context.contentSecurityPolicy());
252 auto& contentSecurityPolicy = *context.contentSecurityPolicy();
253
254 contentSecurityPolicy.upgradeInsecureRequestIfNeeded(m_url, ContentSecurityPolicy::InsecureRequestType::Load);
255
256 if (!portAllowed(m_url)) {
257 String message;
258 if (m_url.port())
259 message = makeString("WebSocket port ", m_url.port().value(), " blocked");
260 else
261 message = "WebSocket without port blocked"_s;
262 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, message);
263 failAsynchronously();
264 return { };
265 }
266
267 // FIXME: Convert this to check the isolated world's Content Security Policy once webkit.org/b/104520 is solved.
268 if (!context.shouldBypassMainWorldContentSecurityPolicy() && !contentSecurityPolicy.allowConnectToSource(m_url)) {
269 m_state = CLOSED;
270
271 // FIXME: Should this be throwing an exception?
272 return Exception { SecurityError };
273 }
274
275 if (auto* provider = context.socketProvider())
276 m_channel = ThreadableWebSocketChannel::create(*scriptExecutionContext(), *this, *provider);
277
278 // Every ScriptExecutionContext should have a SocketProvider.
279 RELEASE_ASSERT(m_channel);
280
281 // FIXME: There is a disagreement about restriction of subprotocols between WebSocket API and hybi-10 protocol
282 // draft. The former simply says "only characters in the range U+0021 to U+007E are allowed," while the latter
283 // imposes a stricter rule: "the elements MUST be non-empty strings with characters as defined in [RFC2616],
284 // and MUST all be unique strings."
285 //
286 // Here, we throw SyntaxError if the given protocols do not meet the latter criteria. This behavior does not
287 // comply with WebSocket API specification, but it seems to be the only reasonable way to handle this conflict.
288 for (auto& protocol : protocols) {
289 if (!isValidProtocolString(protocol)) {
290 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, "Wrong protocol for WebSocket '" + encodeProtocolString(protocol) + "'");
291 m_state = CLOSED;
292 return Exception { SyntaxError };
293 }
294 }
295 HashSet<String> visited;
296 for (auto& protocol : protocols) {
297 if (!visited.add(protocol).isNewEntry) {
298 context.addConsoleMessage(MessageSource::JS, MessageLevel::Error, "WebSocket protocols contain duplicates: '" + encodeProtocolString(protocol) + "'");
299 m_state = CLOSED;
300 return Exception { SyntaxError };
301 }
302 }
303
304 RunLoop::main().dispatch([targetURL = m_url.isolatedCopy(), mainFrameURL = context.url().isolatedCopy()]() {
305 ResourceLoadObserver::shared().logWebSocketLoading(targetURL, mainFrameURL);
306 });
307
308 if (is<Document>(context)) {
309 Document& document = downcast<Document>(context);
310 RefPtr<Frame> frame = document.frame();
311 // FIXME: make the mixed content check equivalent to the non-document mixed content check currently in WorkerThreadableWebSocketChannel::Bridge::connect()
312 if (!frame || !MixedContentChecker::canRunInsecureContent(*frame, document.securityOrigin(), m_url)) {
313 failAsynchronously();
314 return { };
315 }
316 }
317
318 String protocolString;
319 if (!protocols.isEmpty())
320 protocolString = joinStrings(protocols, subprotocolSeparator());
321
322 if (m_channel->connect(m_url, protocolString) == ThreadableWebSocketChannel::ConnectStatus::KO) {
323 failAsynchronously();
324 return { };
325 }
326
327#if ENABLE(RESOURCE_LOAD_STATISTICS)
328 auto reportRegistrableDomain = [domain = RegistrableDomain(m_url).isolatedCopy()](auto& context) mutable {
329 if (auto* frame = downcast<Document>(context).frame())
330 frame->loader().client().didLoadFromRegistrableDomain(WTFMove(domain));
331 };
332 if (is<Document>(context))
333 reportRegistrableDomain(context);
334 else
335 downcast<WorkerGlobalScope>(context).thread().workerLoaderProxy().postTaskToLoader(WTFMove(reportRegistrableDomain));
336#endif
337
338 m_pendingActivity = makePendingActivity(*this);
339
340 return { };
341}

343ExceptionOr<void> WebSocket::send(const String& message)
344{
345 LOG(Network, "WebSocket %p send() Sending String '%s'", this, message.utf8().data());
346 if (m_state == CONNECTING)
347 return Exception { InvalidStateError };
348 // No exception is raised if the connection was once established but has subsequently been closed.
349 if (m_state == CLOSING || m_state == CLOSED) {
350 size_t payloadSize = message.utf8().length();
351 m_bufferedAmountAfterClose = saturateAdd(m_bufferedAmountAfterClose, payloadSize);
352 m_bufferedAmountAfterClose = saturateAdd(m_bufferedAmountAfterClose, getFramingOverhead(payloadSize));
353 return { };
354 }
355 ASSERT(m_channel);
356 m_channel->send(message);
357 return { };
358}

409ExceptionOr<void> WebSocket::close(std::optional<unsigned short> optionalCode, const String& reason)
410{
411 int code = optionalCode ? optionalCode.value() : static_cast<int>(WebSocketChannel::CloseEventCodeNotSpecified);
412 if (code == WebSocketChannel::CloseEventCodeNotSpecified)
413 LOG(Network, "WebSocket %p close() without code and reason", this);
414 else {
415 LOG(Network, "WebSocket %p close() code=%d reason='%s'", this, code, reason.utf8().data());
416 if (!(code == WebSocketChannel::CloseEventCodeNormalClosure || (WebSocketChannel::CloseEventCodeMinimumUserDefined <= code && code <= WebSocketChannel::CloseEventCodeMaximumUserDefined)))
417 return Exception { InvalidAccessError };
418 CString utf8 = reason.utf8(StrictConversionReplacingUnpairedSurrogatesWithFFFD);
419 if (utf8.length() > maxReasonSizeInBytes) {
420 scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Error, "WebSocket close message is too long."_s);
421 return Exception { SyntaxError };
422 }
423 }
424
425 if (m_state == CLOSING || m_state == CLOSED)
426 return { };
427 if (m_state == CONNECTING) {
428 m_state = CLOSING;
429 m_channel->fail("WebSocket is closed before the connection is established.");
430 return { };
431 }
432 m_state = CLOSING;
433 if (m_channel)
434 m_channel->close(code, reason);
435 return { };
436}

512void WebSocket::suspend(ReasonForSuspension reason)
513{
514 if (m_resumeTimer.isActive())
515 m_resumeTimer.stop();
516
517 m_shouldDelayEventFiring = true;
518
519 if (m_channel) {
520 if (reason == ReasonForSuspension::BackForwardCache) {
521 // This will cause didClose() to be called.
522 m_channel->fail("WebSocket is closed due to suspension.");
523 } else
524 m_channel->suspend();
525 }
526}
527
528void WebSocket::resume()
529{
530 if (m_channel)
531 m_channel->resume();
532
533 if (!m_pendingEvents.isEmpty() && !m_resumeTimer.isActive()) {
534 // Fire the pending events in a timer as we are not allowed to execute arbitrary JS from resume().
535 m_resumeTimer.startOneShot(0_s);
536 }
537
538 m_shouldDelayEventFiring = false;
539}

541void WebSocket::resumeTimerFired()
542{
543 Ref<WebSocket> protectedThis(*this);
544
545 ASSERT(!m_pendingEvents.isEmpty());
546
547 // Check m_shouldDelayEventFiring when iterating in case firing an event causes
548 // suspend() to be called.
549 while (!m_pendingEvents.isEmpty() && !m_shouldDelayEventFiring)
550 dispatchEvent(m_pendingEvents.takeFirst());
551}

570{

594 LOG(Network, "WebSocket %p didReceiveBinaryData() %u byte binary message", this, static_cast<unsigned>(binaryData.size()));

628void WebSocket::didClose(unsigned unhandledBufferedAmount, ClosingHandshakeCompletionStatus closingHandshakeCompletion, unsigned short code, const String& reason)
629{
630 LOG(Network, "WebSocket %p didClose()", this);
631 if (!m_channel)
632 return;
633 bool wasClean = m_state == CLOSING && !unhandledBufferedAmount && closingHandshakeCompletion == ClosingHandshakeComplete && code != WebSocketChannel::CloseEventCodeAbnormalClosure;
634 m_state = CLOSED;
635 m_bufferedAmount = unhandledBufferedAmount;
636 ASSERT(scriptExecutionContext());
637
638 dispatchOrQueueEvent(CloseEvent::create(wasClean, code, reason));
639
640 if (m_channel) {
641 m_channel->disconnect();
642 m_channel = nullptr;
643 }
644 m_pendingActivity = nullptr;
645}

653size_t WebSocket::getFramingOverhead(size_t payloadSize)
654{
655 static const size_t hybiBaseFramingOverhead = 2; // Every frame has at least two-byte header.
656 static const size_t hybiMaskingKeyLength = 4; // Every frame from client must have masking key.
657 static const size_t minimumPayloadSizeWithTwoByteExtendedPayloadLength = 126;
658 static const size_t minimumPayloadSizeWithEightByteExtendedPayloadLength = 0x10000;
659 size_t overhead = hybiBaseFramingOverhead + hybiMaskingKeyLength;
660 if (payloadSize >= minimumPayloadSizeWithEightByteExtendedPayloadLength)
661 overhead += 8;
662 else if (payloadSize >= minimumPayloadSizeWithTwoByteExtendedPayloadLength)
663 overhead += 2;
664 return overhead;
665}

676void WebSocket::dispatchOrQueueEvent(Ref<Event>&& event)
677{
