69bool WebSocketExtensionDeflateFrame::processResponse(const HashMap<String, String>& serverParameters)
70{
71 if (m_responseProcessed) {
72 m_failureReason = "Received duplicate deflate-frame response"_s;
73 return false;
74 }
75 m_responseProcessed = true;
76
77 unsigned expectedNumParameters = 0;
78 int windowBits = 15;
79 auto parameter = serverParameters.find("max_window_bits");
80 if (parameter != serverParameters.end()) {
81 windowBits = parseIntegerAllowingTrailingJunk<int>(parameter->value).value_or(0);
82 if (windowBits < 8 || windowBits > 15) {
83 m_failureReason = "Received invalid max_window_bits parameter"_s;
84 return false;
85 }
86 expectedNumParameters++;
87 }
88
89 WebSocketDeflater::ContextTakeOverMode mode = WebSocketDeflater::TakeOverContext;
90 parameter = serverParameters.find("no_context_takeover");
91 if (parameter != serverParameters.end()) {
92 if (!parameter->value.isNull()) {
93 m_failureReason = "Received invalid no_context_takeover parameter"_s;
94 return false;
95 }
96 mode = WebSocketDeflater::DoNotTakeOverContext;
97 expectedNumParameters++;
98 }
99
100 if (expectedNumParameters != serverParameters.size()) {
101 m_failureReason = "Received unexpected deflate-frame parameter"_s;
102 return false;
103 }
104
105 m_framer.enableDeflate(windowBits, mode);
106 return true;
107}

146void WebSocketDeflateFramer::enableDeflate(int windowBits, WebSocketDeflater::ContextTakeOverMode mode)
147{
148 m_deflater = makeUnique<WebSocketDeflater>(windowBits, mode);
149 m_inflater = makeUnique<WebSocketInflater>();
150 if (!m_deflater->initialize() || !m_inflater->initialize()) {
151 m_deflater = nullptr;
152 m_inflater = nullptr;
153 return;
154 }
155 m_enabled = true;
156}
157
158std::unique_ptr<DeflateResultHolder> WebSocketDeflateFramer::deflate(WebSocketFrame& frame)
159{
160 auto result = makeUnique<DeflateResultHolder>(*this);
161 if (!enabled() || !WebSocketFrame::isNonControlOpCode(frame.opCode) || !frame.payloadLength)
162 return result;
163 if (!m_deflater->addBytes(frame.payload, frame.payloadLength) || !m_deflater->finish()) {
164 result->fail("Failed to compress frame");
165 return result;
166 }
167 frame.compress = true;
168 frame.payload = m_deflater->data();
169 frame.payloadLength = m_deflater->size();
170 return result;
171}

173void WebSocketDeflateFramer::resetDeflateContext()
174{
175 if (m_deflater)
176 m_deflater->reset();
177}

179std::unique_ptr<InflateResultHolder> WebSocketDeflateFramer::inflate(WebSocketFrame& frame)
180{
181 auto result = makeUnique<InflateResultHolder>(*this);
182 if (!enabled() && frame.compress) {
183 result->fail("Compressed bit must be 0 if no negotiated deflate-frame extension");
184 return result;
185 }
186 if (!frame.compress)
187 return result;
188 if (!WebSocketFrame::isNonControlOpCode(frame.opCode)) {
189 result->fail("Received unexpected compressed frame");
190 return result;
191 }
192 if (!m_inflater->addBytes(frame.payload, frame.payloadLength) || !m_inflater->finish()) {
193 result->fail("Failed to decompress frame");
194 return result;
195 }
196 frame.compress = false;
197 frame.payload = m_inflater->data();
198 frame.payloadLength = m_inflater->size();
199 return result;
200}
201
202void WebSocketDeflateFramer::resetInflateContext()
203{
204 if (m_inflater)
205 m_inflater->reset();
206}
207
208void WebSocketDeflateFramer::didFail()
209{
210 resetDeflateContext();
211 resetInflateContext();
212}
