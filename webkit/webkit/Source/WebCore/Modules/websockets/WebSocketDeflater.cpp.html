41static const int defaultMemLevel = 8;
42static const size_t bufferIncrementUnit = 4096;
43
44WebSocketDeflater::WebSocketDeflater(int windowBits, ContextTakeOverMode contextTakeOverMode)
45 : m_windowBits(windowBits)
46 , m_contextTakeOverMode(contextTakeOverMode)
47{
48 ASSERT(m_windowBits >= 8);
49 ASSERT(m_windowBits <= 15);
50 m_stream = makeUniqueWithoutFastMallocCheck<z_stream>();
51 memset(m_stream.get(), 0, sizeof(z_stream));
52}
53
54bool WebSocketDeflater::initialize()
55{
56 return deflateInit2(m_stream.get(), Z_DEFAULT_COMPRESSION, Z_DEFLATED, -m_windowBits, defaultMemLevel, Z_DEFAULT_STRATEGY) == Z_OK;
57}
58
59WebSocketDeflater::~WebSocketDeflater()
60{
61 int result = deflateEnd(m_stream.get());
62 if (result != Z_OK)
63 LOG(Network, "WebSocketDeflater %p Destructor deflateEnd() failed: %d is returned", this, result);
64}

66static void setStreamParameter(z_stream* stream, const uint8_t* inputData, size_t inputLength, uint8_t* outputData, size_t outputLength)
67{
68 stream->next_in = const_cast<uint8_t*>(inputData);
69 stream->avail_in = inputLength;
70 stream->next_out = outputData;
71 stream->avail_out = outputLength;
72}
73
74bool WebSocketDeflater::addBytes(const uint8_t* data, size_t length)
75{
76 if (!length)
77 return false;
78
79 size_t maxLength = deflateBound(m_stream.get(), length);
80 size_t writePosition = m_buffer.size();
81 m_buffer.grow(writePosition + maxLength);
82 setStreamParameter(m_stream.get(), data, length, m_buffer.data() + writePosition, maxLength);
83 int result = deflate(m_stream.get(), Z_NO_FLUSH);
84 if (result != Z_OK || m_stream->avail_in > 0)
85 return false;
86
87 m_buffer.shrink(writePosition + maxLength - m_stream->avail_out);
88 return true;
89}
90
91bool WebSocketDeflater::finish()
92{
93 while (true) {
94 size_t writePosition = m_buffer.size();
95 m_buffer.grow(writePosition + bufferIncrementUnit);
96 size_t availableCapacity = m_buffer.size() - writePosition;
97 setStreamParameter(m_stream.get(), 0, 0, m_buffer.data() + writePosition, availableCapacity);
98 int result = deflate(m_stream.get(), Z_SYNC_FLUSH);
99 if (m_stream->avail_out) {
100 m_buffer.shrink(writePosition + availableCapacity - m_stream->avail_out);
101 if (result == Z_OK)
102 break;
103 if (result != Z_BUF_ERROR)
104 return false;
105 }
106 }
107 // Remove 4 octets from the tail as the specification requires.
108 if (m_buffer.size() <= 4)
109 return false;
110 m_buffer.shrink(m_buffer.size() - 4);
111 return true;
112}

114void WebSocketDeflater::reset()
115{
116 m_buffer.clear();
117 if (m_contextTakeOverMode == DoNotTakeOverContext)
118 deflateReset(m_stream.get());
119}

140bool WebSocketInflater::addBytes(const uint8_t* data, size_t length)
141{
142 if (!length)
143 return false;
144
145 size_t consumedSoFar = 0;
146 while (consumedSoFar < length) {
147 size_t writePosition = m_buffer.size();
148 m_buffer.grow(writePosition + bufferIncrementUnit);
149 size_t availableCapacity = m_buffer.size() - writePosition;
150 size_t remainingLength = length - consumedSoFar;
151 setStreamParameter(m_stream.get(), data + consumedSoFar, remainingLength, m_buffer.data() + writePosition, availableCapacity);
152 int result = inflate(m_stream.get(), Z_NO_FLUSH);
153 consumedSoFar += remainingLength - m_stream->avail_in;
154 m_buffer.shrink(writePosition + availableCapacity - m_stream->avail_out);
155 if (result == Z_BUF_ERROR)
156 continue;
157 if (result == Z_STREAM_END) {
158 // Received a block with BFINAL set to 1. Reset decompression state.
159 if (inflateReset(m_stream.get()) != Z_OK)
160 return false;
161 continue;
162 }
163 if (result != Z_OK)
164 return false;
165 ASSERT(remainingLength > m_stream->avail_in);
166 }
167 ASSERT(consumedSoFar == length);
168 return true;
169}
170
171bool WebSocketInflater::finish()
172{
173 constexpr uint8_t strippedFields[] = { 0, 0, 0xFF, 0xFF };
174 constexpr auto strippedLength = std::size(strippedFields);
175
176 // Appends 4 octests of 0x00 0x00 0xff 0xff
177 size_t consumedSoFar = 0;
178 while (consumedSoFar < strippedLength) {
179 size_t writePosition = m_buffer.size();
180 m_buffer.grow(writePosition + bufferIncrementUnit);
181 size_t availableCapacity = m_buffer.size() - writePosition;
182 size_t remainingLength = strippedLength - consumedSoFar;
183 setStreamParameter(m_stream.get(), strippedFields + consumedSoFar, remainingLength, m_buffer.data() + writePosition, availableCapacity);
184 int result = inflate(m_stream.get(), Z_FINISH);
185 consumedSoFar += remainingLength - m_stream->avail_in;
186 m_buffer.shrink(writePosition + availableCapacity - m_stream->avail_out);
187 if (result == Z_BUF_ERROR)
188 continue;
189 if (result != Z_OK && result != Z_STREAM_END)
190 return false;
191 ASSERT(remainingLength > m_stream->avail_in);
192 }
193 ASSERT(consumedSoFar == strippedLength);
194
195 return true;
196}

199{
200 m_buffer.clear();
