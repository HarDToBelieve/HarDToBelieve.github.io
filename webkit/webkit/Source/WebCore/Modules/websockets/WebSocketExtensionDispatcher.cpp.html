42void WebSocketExtensionDispatcher::reset()
43{
44 m_processors.clear();
45}
46
47void WebSocketExtensionDispatcher::addProcessor(std::unique_ptr<WebSocketExtensionProcessor> processor)
48{
49 for (auto& extensionProcessor : m_processors) {
50 if (extensionProcessor->extensionToken() == processor->extensionToken())
51 return;
52 }
53 ASSERT(processor->handshakeString().length());
54 ASSERT(!processor->handshakeString().contains('\n'));
55 ASSERT(!processor->handshakeString().contains(static_cast<UChar>('\0')));
56 m_processors.append(WTFMove(processor));
57}

59const String WebSocketExtensionDispatcher::createHeaderValue() const
60{
61 size_t numProcessors = m_processors.size();
62 if (!numProcessors)
63 return String();
64
65 StringBuilder builder;
66 builder.append(m_processors[0]->handshakeString());
67 for (size_t i = 1; i < numProcessors; ++i)
68 builder.append(", ", m_processors[i]->handshakeString());
69 return builder.toString();
70}

72void WebSocketExtensionDispatcher::appendAcceptedExtension(const String& extensionToken, HashMap<String, String>& extensionParameters)
73{
74 m_acceptedExtensionsBuilder.append(m_acceptedExtensionsBuilder.isEmpty() ? "" : ", ", extensionToken);
75 // FIXME: Should use ListHashSet to keep the order of the parameters.
76 for (auto& parameter : extensionParameters) {
77 m_acceptedExtensionsBuilder.append("; ", parameter.key);
78 if (!parameter.value.isNull())
79 m_acceptedExtensionsBuilder.append('=', parameter.value);
80 }
81}

83void WebSocketExtensionDispatcher::fail(const String& reason)
84{
85 m_failureReason = reason;
86 m_acceptedExtensionsBuilder.clear();

89bool WebSocketExtensionDispatcher::processHeaderValue(const String& headerValue)
90{
91 if (!headerValue.length())
92 return true;
93
94 // If we don't send Sec-WebSocket-Extensions header, the server should not return the header.
95 if (!m_processors.size()) {
96 fail("Received unexpected Sec-WebSocket-Extensions header");
97 return false;
98 }
99
100 const CString headerValueData = headerValue.utf8();
101 WebSocketExtensionParser parser(headerValueData.data(), headerValueData.data() + headerValueData.length());
102 while (!parser.finished()) {
103 String extensionToken;
104 HashMap<String, String> extensionParameters;
105 if (!parser.parseExtension(extensionToken, extensionParameters)) {
106 fail("Sec-WebSocket-Extensions header is invalid");
107 return false;
108 }
109
110 size_t index = 0;
111 for (auto& processor : m_processors) {
112 if (extensionToken == processor->extensionToken()) {
113 if (processor->processResponse(extensionParameters)) {
114 appendAcceptedExtension(extensionToken, extensionParameters);
115 break;
116 }
117 fail(processor->failureReason());
118 return false;
119 }
120 ++index;
121 }
122 // There is no extension which can process the response.
123 if (index == m_processors.size()) {
124 fail("Received unexpected extension: " + extensionToken);
125 return false;
126 }
127 }
128 return parser.parsedSuccessfully();
129}

131String WebSocketExtensionDispatcher::acceptedExtensions() const
132{
133 if (m_acceptedExtensionsBuilder.isEmpty())
134 return String();
135 return m_acceptedExtensionsBuilder.toStringPreserveCapacity();
136}
