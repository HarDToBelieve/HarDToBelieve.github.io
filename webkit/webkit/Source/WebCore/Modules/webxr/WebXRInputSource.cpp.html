50WebXRInputSource::WebXRInputSource(Document& document, WebXRSession& session, double timestamp, const PlatformXR::Device::FrameData::InputSource& source)
51 : m_session(makeWeakPtr(session))
52 , m_targetRaySpace(WebXRInputSpace::create(document, session, source.pointerOrigin))
53 , m_connectTime(timestamp)
54#if ENABLE(GAMEPAD)
55 , m_gamepad(Gamepad::create(WebXRGamepad(timestamp, timestamp, source)))
56#endif
57{
58 update(timestamp, source);
59}

68void WebXRInputSource::update(double timestamp, const PlatformXR::Device::FrameData::InputSource& source)
69{
70 auto session = makeRefPtr(m_session.get());
71 if (!session)
72 return;
73
74#if !ENABLE(GAMEPAD)
75 UNUSED_PARAM(timestamp);
76#endif
77 m_source = source;
78 m_targetRaySpace->setPose(source.pointerOrigin);
79 if (auto gripOrigin = source.gripOrigin) {
80 if (m_gripSpace)
81 m_gripSpace->setPose(*gripOrigin);
82 else if (auto* document = downcast<Document>(session->scriptExecutionContext()))
83 m_gripSpace = WebXRInputSpace::create(*document, *session, *gripOrigin);
84 } else
85 m_gripSpace = nullptr;
86#if ENABLE(GAMEPAD)
87 m_gamepad->updateFromPlatformGamepad(WebXRGamepad(timestamp, m_connectTime, source));
88#endif
89}
90
91bool WebXRInputSource::requiresInputSourceChange(const InputSource& source)
92{
93 return m_source.handeness != source.handeness
94 || m_source.targetRayMode != source.targetRayMode
95 || m_source.profiles != source.profiles
96 || static_cast<bool>(m_gripSpace) != source.gripOrigin.has_value();
97}

99void WebXRInputSource::disconnect()
100{
101 m_connected = false;
102#if ENABLE(GAMEPAD)
103 m_gamepad->setConnected(false);
104#endif
105}

107void WebXRInputSource::pollEvents(Vector<Ref<XRInputSourceEvent>>& events)
108{
109 auto session = makeRefPtr(m_session.get());
110 if (!session)
111 return;
112
113 auto createEvent = [this, session](const AtomString& name) -> Ref<XRInputSourceEvent> {
114 XRInputSourceEvent::Init init;
115 init.frame = WebXRFrame::create(*session, WebXRFrame::IsAnimationFrame::No);
116 init.inputSource = makeRefPtr(*this);
117
118 return XRInputSourceEvent::create(name, init);
119 };
120
121 if (!m_connected) {
122 // A user agent MUST dispatch a selectend event on an XRSession when one of its XRInputSources ends
123 // when an XRInputSource that has begun a primary select action is disconnected.
124 if (m_selectStarted)
125 events.append(createEvent(eventNames().selectendEvent));
126
127 // A user agent MUST dispatch a squeezeend event on an XRSession when one of its XRInputSources ends
128 // when an XRInputSource that has begun a primary squeeze action is disconnected.
129 if (m_squeezeStarted)
130 events.append(createEvent(eventNames().squeezeendEvent));
131
132 return;
133 }
134
135 // https://immersive-web.github.io/webxr-gamepads-module/#xr-standard-gamepad-mapping
136 // Button indices are based on the xr-standard layout.
137 bool selectStarted = !m_source.buttons.isEmpty() && m_source.buttons.first().pressed;
138 bool squeezeStarted = m_source.buttons.size() > 1 && m_source.buttons[1].pressed;
139
140 if (selectStarted != m_selectStarted) {
141 if (selectStarted) {
142 // A user agent MUST dispatch a selectstart event on an XRSession when one of its XRInputSources begins its primary action.
143 events.append(createEvent(eventNames().selectstartEvent));
144 } else {
145 // A user agent MUST dispatch a selectend event on an XRSession when one of its XRInputSources ends its primary action.
146 // A user agent MUST dispatch a select event on an XRSession when one of its XRInputSources has fully completed a primary action
147 events.append(createEvent(eventNames().selectEvent));
148 events.append(createEvent(eventNames().selectendEvent));
149 }
150 m_selectStarted = selectStarted;
151 }
152
153 if (squeezeStarted != m_squeezeStarted) {
154 if (squeezeStarted) {
155 // A user agent MUST dispatch a squeezestart event on an XRSession when one of its XRInputSources begins its primary squeeze action.
156 events.append(createEvent(eventNames().squeezestartEvent));
157 } else {
158 // A user agent MUST dispatch a selectend event on an XRSession when one of its XRInputSources ends its primary squeeze action.
159 // A user agent MUST dispatch a select event on an XRSession when one of its XRInputSources has fully completed a primary squeeze action.
160 events.append(createEvent(eventNames().squeezeEvent));
161 events.append(createEvent(eventNames().squeezeendEvent));
162 }
163 m_squeezeStarted = squeezeStarted;
164 }
165}
