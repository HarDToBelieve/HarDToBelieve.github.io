78// https://immersive-web.github.io/webxr/#list-of-active-xr-input-sources
79void WebXRInputSourceArray::update(double timestamp, const InputSourceList& inputSources)
80{
81 Vector<RefPtr<WebXRInputSource>> added;
82 Vector<RefPtr<WebXRInputSource>> removed;
83 Vector<Ref<XRInputSourceEvent>> inputEvents;
84
85 handleRemovedInputSources(inputSources, removed, inputEvents);
86 handleAddedOrUpdatedInputSources(timestamp, inputSources, added, removed, inputEvents);
87
88 if (!added.isEmpty() || !removed.isEmpty()) {
89 // A user agent MUST dispatch an inputsourceschange event on an XRSession when the session’s list of active XR input sources has changed.
90 XRInputSourcesChangeEvent::Init init;
91 init.session = makeRef(m_session);
92 init.added = WTFMove(added);
93 init.removed = WTFMove(removed);
94
95 auto event = XRInputSourcesChangeEvent::create(eventNames().inputsourceschangeEvent, init);
96 ActiveDOMObject::queueTaskToDispatchEvent(m_session, TaskSource::WebXR, WTFMove(event));
97 }
98
99 if (!inputEvents.isEmpty()) {
100 // When the user agent has to fire an input source event with name name, XRFrame frame, and XRInputSource source it MUST run the following steps:
101 // 1. Create an XRInputSourceEvent event with type name, frame frame, and inputSource source.
102 // 2. Set frame’s active boolean to true.
103 // 3. Apply frame updates for frame.
104 // 4. Dispatch event on frame’s session
105 // 5. Set frame’s active boolean to false.
106
107 for (auto& event : inputEvents) {
108 ActiveDOMObject::queueTaskKeepingObjectAlive(m_session, TaskSource::WebXR, [session = makeRefPtr(m_session), event = WTFMove(event)]() {
109 event->setFrameActive(true);
110 session->dispatchEvent(event.copyRef());
111 event->setFrameActive(false);
112 });
113 }
114 }
115}

117// https://immersive-web.github.io/webxr/#list-of-active-xr-input-sources
118void WebXRInputSourceArray::handleRemovedInputSources(const InputSourceList& inputSources, Vector<RefPtr<WebXRInputSource>>& removed, Vector<Ref<XRInputSourceEvent>>& inputEvents)
119{
120 // When any previously added XR input sources are no longer available for XRSession session, the user agent MUST run the following steps:
121 // 1. If session's promise resolved flag is not set, abort these steps.
122 // 2. Let removed be a new list.
123 // 3. For each XR input source that is no longer available:
124 // 3.1 Let inputSource be the XRInputSource in session's list of active XR input sources associated with the XR input source.
125 // 3.2 Add inputSource to removed.
126 m_inputSources.removeAllMatching([&inputSources, &removed, &inputEvents](auto& source) {
127 if (!WTF::anyOf(inputSources, [&source](auto& item) { return item.handle == source->handle(); })) {
128 removed.append(source.copyRef());
129 source->disconnect();
130 source->pollEvents(inputEvents);
131 return true;
132 }
133 return false;
134 });
135}

137// https://immersive-web.github.io/webxr/#list-of-active-xr-input-sources
138void WebXRInputSourceArray::handleAddedOrUpdatedInputSources(double timestamp, const InputSourceList& inputSources, Vector<RefPtr<WebXRInputSource>>& added, Vector<RefPtr<WebXRInputSource>>& removed, Vector<Ref<XRInputSourceEvent>>& inputEvents)
139{
140 auto* document = downcast<Document>(m_session.scriptExecutionContext());
141 if (!document)
142 return;
143
144 for (auto& inputSource : inputSources) {
145 auto index = m_inputSources.findMatching([&inputSource](auto& item) { return item->handle() == inputSource.handle; });
146 if (index == WTF::notFound) {
147 // When new XR input sources become available for XRSession session, the user agent MUST run the following steps:
148 // 1. If session's promise resolved flag is not set, abort these steps.
149 // 2. Let added be a new list.
150 // 3. For each new XR input source:
151 // 3.1 Let inputSource be a new XRInputSource in the relevant realm of this XRSession.
152 // 3.2 Add inputSource to added.
153
154 auto input = WebXRInputSource::create(*document, m_session, timestamp, inputSource);
155 added.append(input.copyRef());
156 input->pollEvents(inputEvents);
157 m_inputSources.append(WTFMove(input));
158 continue;
159 }
160
161 // When the handedness, targetRayMode, profiles, or presence of a gripSpace for any XR input sources change for XRSession session, the user agent MUST run the following steps
162 // 1. If session’s promise resolved flag is not set, abort these steps.
163 // 2. Let added be a new list.
164 // 3. Let removed be a new list.
165 // 4. For each changed XR input source:
166 // 4.1 Let oldInputSource be the XRInputSource in session's list of active XR input sources previously associated with the XR input source.
167 // 4.1 Let newInputSource be a new XRInputSource in the relevant realm of session.
168 // 4.1 Add oldInputSource to removed.
169 // 4.1 Add newInputSource to added.
170 auto& input = m_inputSources[index];
171
172 if (input->requiresInputSourceChange(inputSource)) {
173 removed.append(input.copyRef());
174 input->disconnect();
175 input->pollEvents(inputEvents);
176 m_inputSources.remove(index);
177
178 auto newInputSource = WebXRInputSource::create(*document, m_session, timestamp, inputSource);
179 added.append(newInputSource.copyRef());
180 newInputSource->pollEvents(inputEvents);
181 m_inputSources.append(WTFMove(newInputSource));
182 } else {
183 input->update(timestamp, inputSource);
184 input->pollEvents(inputEvents);
185 }
186 }
187}
