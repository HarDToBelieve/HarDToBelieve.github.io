36namespace WebCore {
37
38class ScriptExecutionContext;
39
40class WebXRLayer : public RefCounted<WebXRLayer>, public EventTargetWithInlineData, public ContextDestructionObserver {
41 WTF_MAKE_ISO_ALLOCATED(WebXRLayer);
42public:
43 ~WebXRLayer();
44
45 using RefCounted<WebXRLayer>::ref;
46 using RefCounted<WebXRLayer>::deref;
47
48 virtual void startFrame(const PlatformXR::Device::FrameData&) = 0;
49 virtual PlatformXR::Device::Layer endFrame() = 0;
50
51protected:
52 explicit WebXRLayer(ScriptExecutionContext*);
53
54 // EventTarget
55 ScriptExecutionContext* scriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
56
57private:
58 // EventTarget
59 EventTargetInterface eventTargetInterface() const final { return WebXRLayerEventTargetInterfaceType; }
60 void refEventTarget() final { ref(); }
61 void derefEventTarget() final { deref(); }
62};
63
64} // namespace WebCore
65
66#endif // ENABLE(WEBXR)
