37WTF_MAKE_ISO_ALLOCATED_IMPL(WebXRRenderState);

39Ref<WebXRRenderState> WebXRRenderState::create(XRSessionMode mode)
40{
41 // https://immersive-web.github.io/webxr/#initialize-the-render-state
42 // depthNear, depthFar and baseLayer are initialized in the class definition
43 return adoptRef(*new WebXRRenderState(mode == XRSessionMode::Inline ? std::make_optional(piOverTwoDouble) : std::nullopt));
44}
45
46WebXRRenderState::WebXRRenderState(std::optional<double> inlineVerticalFieldOfView)
47 : m_inlineVerticalFieldOfView(inlineVerticalFieldOfView)
48{
49}
50
51WebXRRenderState::~WebXRRenderState() = default;
52
53Ref<WebXRRenderState> WebXRRenderState::clone() const
54{
55 return adoptRef(*new WebXRRenderState(*this));
56}
57
58WebXRRenderState::WebXRRenderState(const WebXRRenderState& other)
59 : m_depth(other.m_depth)
60 , m_inlineVerticalFieldOfView(other.m_inlineVerticalFieldOfView)
61 , m_baseLayer(other.baseLayer())
62{
63}
