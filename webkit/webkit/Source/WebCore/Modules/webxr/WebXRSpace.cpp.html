39WTF_MAKE_ISO_ALLOCATED_IMPL(WebXRSpace);
40
41WebXRSpace::WebXRSpace(Document& document, Ref<WebXRRigidTransform>&& offset)
42 : ContextDestructionObserver(&document)
43 , m_originOffset(WTFMove(offset))
44{
45}
46
47WebXRSpace::~WebXRSpace() = default;
48
49TransformationMatrix WebXRSpace::effectiveOrigin() const
50{
51 // https://immersive-web.github.io/webxr/#xrspace-effective-origin
52 // The effective origin can be obtained by multiplying origin offset and the native origin.
53 return nativeOrigin() * m_originOffset->rawTransform();
54}
55
56
57bool WebXRSpace::isPositionEmulated() const
58{
59 return session().isPositionEmulated();
60}
61
62WTF_MAKE_ISO_ALLOCATED_IMPL(WebXRViewerSpace);
63
64WebXRViewerSpace::WebXRViewerSpace(Document& document, WebXRSession& session)
65 : WebXRSpace(document, WebXRRigidTransform::create())
66 , m_session(session)
67{
68}
69
70WebXRViewerSpace::~WebXRViewerSpace() = default;
71
72TransformationMatrix WebXRViewerSpace::nativeOrigin() const
73{
74 return WebXRFrame::matrixFromPose(m_session.frameData().origin);
75}
