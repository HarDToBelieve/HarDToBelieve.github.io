35namespace WebCore {
36
37class Document;
38class ScriptExecutionContext;
39class WebXRRigidTransform;
40class WebXRSession;
41
42class WebXRSpace : public EventTargetWithInlineData, public ContextDestructionObserver {
43 WTF_MAKE_ISO_ALLOCATED(WebXRSpace);
44public:
45 virtual ~WebXRSpace();
46
47 virtual WebXRSession& session() const = 0;
48 virtual TransformationMatrix nativeOrigin() const = 0;
49 TransformationMatrix effectiveOrigin() const;
50 virtual bool isPositionEmulated() const;
51
52 virtual bool isReferenceSpace() const { return false; }
53 virtual bool isBoundedReferenceSpace() const { return false; }
54
55protected:
56 WebXRSpace(Document&, Ref<WebXRRigidTransform>&&);
57
58 const WebXRRigidTransform& originOffset() const { return m_originOffset.get(); }
59
60 // EventTarget
61 ScriptExecutionContext* scriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
62
63private:
64 // EventTarget
65 EventTargetInterface eventTargetInterface() const final { return WebXRSpaceEventTargetInterfaceType; }
66
67 Ref<WebXRRigidTransform> m_originOffset;
68};

70// https://immersive-web.github.io/webxr/#xrsession-viewer-reference-space
71// This is a helper class to implement the viewer space owned by a WebXRSession.
72// It avoids a circular reference between the session and the reference space.
73class WebXRViewerSpace : public WebXRSpace {
74 WTF_MAKE_ISO_ALLOCATED(WebXRViewerSpace);
75public:
76 WebXRViewerSpace(Document&, WebXRSession&);
77 virtual ~WebXRViewerSpace();
78
79private:
80 WebXRSession& session() const final { return m_session; }
81 TransformationMatrix nativeOrigin() const final;
82
83 void refEventTarget() final { RELEASE_ASSERT_NOT_REACHED(); }
84 void derefEventTarget() final { RELEASE_ASSERT_NOT_REACHED(); }
85
86 WebXRSession& m_session;
87};

89} // namespace WebCore
