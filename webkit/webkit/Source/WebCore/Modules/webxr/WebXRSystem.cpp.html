78// https://immersive-web.github.io/webxr/#ensures-an-immersive-xr-device-is-selected
79void WebXRSystem::ensureImmersiveXRDeviceIsSelected(CompletionHandler<void()>&& callback)
80{
81 // Don't ask platform code for XR devices, we're using simulated ones.
82 if (UNLIKELY(m_testingDevices)) {
83 callback();
84 return;
85 }
86
87 if (m_activeImmersiveDevice) {
88 callback();
89 return;
90 }
91
92 // https://immersive-web.github.io/webxr/#enumerate-immersive-xr-devices
93 auto document = downcast<Document>(scriptExecutionContext());
94 if (!document || !document->page()) {
95 callback();
96 return;
97 }
98
99 bool isFirstXRDevicesEnumeration = !m_immersiveXRDevicesHaveBeenEnumerated;
100 document->page()->chrome().client().enumerateImmersiveXRDevices([this, protectedThis = makeRef(*this), isFirstXRDevicesEnumeration, callback = WTFMove(callback)](auto& immersiveXRDevices) mutable {
101 m_immersiveXRDevicesHaveBeenEnumerated = true;
102
103 auto callbackOnExit = makeScopeExit([&]() {
104 callOnMainThread(WTFMove(callback));
105 });
106
107 // https://immersive-web.github.io/webxr/#select-an-immersive-xr-device
108 auto* oldDevice = m_activeImmersiveDevice.get();
109 if (immersiveXRDevices.isEmpty()) {
110 m_activeImmersiveDevice = nullptr;
111 return;
112 }
113 if (immersiveXRDevices.size() == 1) {
114 m_activeImmersiveDevice = makeWeakPtr(immersiveXRDevices.first().get());
115 return;
116 }
117
118 if (m_activeImmersiveSession && oldDevice && immersiveXRDevices.findMatching([&](auto& entry) { return entry.ptr() == oldDevice; }) != notFound)
119 ASSERT(m_activeImmersiveDevice.get() == oldDevice);
120 else {
121 // FIXME: implement a better UA selection mechanism if required.
122 m_activeImmersiveDevice = makeWeakPtr(immersiveXRDevices.first().get());
123 }
124
125 if (isFirstXRDevicesEnumeration || m_activeImmersiveDevice.get() == oldDevice) {
126 return;
127 }
128
129 // FIXME: 7. Shut down any active XRSessions.
130 // FIXME: 8. Set the XR compatible boolean of all WebGLRenderingContextBase instances to false.
131 // FIXME: 9. Queue a task to fire an event named devicechange on the context object.
132 });
133}

135void WebXRSystem::obtainCurrentDevice(XRSessionMode mode, const JSFeatureList& requiredFeatures, const JSFeatureList& optionalFeatures, CompletionHandler<void(PlatformXR::Device*)>&& callback)
136{
137 if (mode == XRSessionMode::ImmersiveAr || mode == XRSessionMode::ImmersiveVr) {
138 ensureImmersiveXRDeviceIsSelected([this, callback = WTFMove(callback)]() mutable {
139 callback(m_activeImmersiveDevice.get());
140 });
141 return;
142 }
143 if (!requiredFeatures.isEmpty() || !optionalFeatures.isEmpty()) {
144 callback(m_inlineXRDevice.get());
145 return;
146 }
147
148 callback(&m_defaultInlineDevice);
149}

152// https://immersive-web.github.io/webxr/#dom-xrsystem-issessionsupported
153void WebXRSystem::isSessionSupported(XRSessionMode mode, IsSessionSupportedPromise&& promise)
154{
155 // 1. Let promise be a new Promise.
156 // 2. If mode is "inline", resolve promise with true and return it.
157 if (mode == XRSessionMode::Inline) {
158 promise.resolve(true);
159 return;
160 }
161
162 // 3. If the requesting document's origin is not allowed to use the "xr-spatial-tracking" feature policy,
163 // reject promise with a "SecurityError" DOMException and return it.
164 auto document = downcast<Document>(scriptExecutionContext());
165 if (!isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::XRSpatialTracking, *document, LogFeaturePolicyFailure::Yes)) {
166 promise.reject(Exception { SecurityError });
167 return;
168 }
169
170 // 4. Run the following steps in parallel:
171 // 4.1 Ensure an immersive XR device is selected.
172 ensureImmersiveXRDeviceIsSelected([this, promise = WTFMove(promise), mode]() mutable {
173 // 4.2 If the immersive XR device is null, resolve promise with false and abort these steps.
174 if (!m_activeImmersiveDevice) {
175 promise.resolve(false);
176 return;
177 }
178
179 // 4.3 If the immersive XR device's list of supported modes does not contain mode, resolve promise with false and abort these steps.
180 if (!m_activeImmersiveDevice->supports(mode)) {
181 promise.resolve(false);
182 return;
183 }
184
185 // 4.4 Resolve promise with true.
186 promise.resolve(true);
187 });
188}

190// https://immersive-web.github.io/webxr/#immersive-session-request-is-allowed
191bool WebXRSystem::immersiveSessionRequestIsAllowedForGlobalObject(DOMWindow& globalObject, Document& document) const
192{
193 // 1. If the request was not made while the global object has transient
194 // activation or when launching a web application, return false
195 // TODO: add a check for "not launching a web application".
196 if (!globalObject.hasTransientActivation())
197 return false;
198
199 // 2. If the requesting document is not considered trustworthy, return false.
200 // https://immersive-web.github.io/webxr/#trustworthy.
201 if (&document != globalObject.document())
202 return false;
203
204 // https://immersive-web.github.io/webxr/#active-and-focused
205 if (!document.hasFocus() || !document.securityOrigin().isSameOriginAs(globalObject.document()->securityOrigin()))
206 return false;
207
208 // 3. If user intent to begin an immersive session is not well understood,
209 // either via explicit consent or implicit consent, return false
210 if (!UserGestureIndicator::processingUserGesture())
211 return false;
212
213 return true;
214}

216// https://immersive-web.github.io/webxr/#inline-session-request-is-allowed
217bool WebXRSystem::inlineSessionRequestIsAllowedForGlobalObject(DOMWindow& globalObject, Document& document, const XRSessionInit& init) const
218{
219 auto isEmptyOrViewer = [&document](const JSFeatureList& features) {
220 if (features.isEmpty())
221 return true;
222 if (features.size() == 1) {
223 auto feature = parseEnumeration<XRReferenceSpaceType>(*document.globalObject(), features.first());
224 if (feature == XRReferenceSpaceType::Viewer)
225 return true;
226 }
227 return false;
228 };
229
230 // 1. If the session request contained any required features or optional features (besides 'viewer') and the request was not made
231 // while the global object has transient activation or when launching a web application, return false.
232 bool sessionRequestContainedAnyFeature = !isEmptyOrViewer(init.optionalFeatures) || !isEmptyOrViewer(init.requiredFeatures);
233 if (sessionRequestContainedAnyFeature && !globalObject.hasTransientActivation() /* TODO: || !launching a web app */)
234 return false;
235
236 // 2. If the requesting document is not responsible, return false.
237 if (&document != globalObject.document())
238 return false;
239
240 return true;
241}

244struct WebXRSystem::ResolvedRequestedFeatures {
245 FeatureList granted;
246 FeatureList consentRequired;
247 FeatureList consentOptional;
248 FeatureList requested;
249};

251#define RETURN_FALSE_OR_CONTINUE(mustReturn) { \
252 if (mustReturn) {\
253 return false; \
254 } \
255 continue; \
256}

258// https://immersive-web.github.io/webxr/#resolve-the-requested-features
259std::optional<WebXRSystem::ResolvedRequestedFeatures> WebXRSystem::resolveRequestedFeatures(XRSessionMode mode, const XRSessionInit& init, PlatformXR::Device* device, JSC::JSGlobalObject& globalObject) const
260{
261 // 1. Let consentRequired be an empty list of DOMString.
262 // 2. Let consentOptional be an empty list of DOMString.
263 ResolvedRequestedFeatures resolvedFeatures;
264
265 // 3. Let device be the result of obtaining the current device for mode, requiredFeatures, and optionalFeatures.
266 // 4. Let granted be a list of DOMString initialized to device's list of enabled features for mode.
267 // 5. If device is null or device's list of supported modes does not contain mode, run the following steps:
268 // 5.1 Return the tuple (consentRequired, consentOptional, granted)
269 if (!device || !device->supports(mode))
270 return resolvedFeatures;
271
272 resolvedFeatures.granted = device->enabledFeatures(mode);
273
274 // 6. Add every feature descriptor in the default features table associated
275 // with mode to the indicated feature list if it is not already present.
276 // https://immersive-web.github.io/webxr/#default-features
277 auto requiredFeaturesWithDefaultFeatures = init.requiredFeatures;
278 requiredFeaturesWithDefaultFeatures.append(convertEnumerationToJS(globalObject, XRReferenceSpaceType::Viewer));
279 if (mode == XRSessionMode::ImmersiveAr || mode == XRSessionMode::ImmersiveVr)
280 requiredFeaturesWithDefaultFeatures.append(convertEnumerationToJS(globalObject, XRReferenceSpaceType::Local));
281
282 // 7. For each feature in requiredFeatures|optionalFeatures perform the following steps:
283 // 8. For each feature in optionalFeatures perform the following steps:
284 // We're merging both loops in a single lambda. The only difference is that a failure on any required features
285 // implies cancelling the whole process while failures in optional features are just skipped.
286 enum class ParsingMode { Strict, Loose };
287 auto parseFeatures = [&device, &globalObject, mode, &resolvedFeatures] (const JSFeatureList& sessionFeatures, ParsingMode parsingMode) -> bool {
288 bool returnOnFailure = parsingMode == ParsingMode::Strict;
289 for (const auto& sessionFeature : sessionFeatures) {
290 // 1. If the feature is null, continue to the next entry.
291 if (sessionFeature.isNull())
292 continue;
293
294 // 2. If feature is not a valid feature descriptor, perform the following steps
295 // 2.1. Let s be the result of calling ? ToString(feature).
296 // 2.2. If s is not a valid feature descriptor or is undefined, (return null|continue to next entry).
297 // 2.3. Set feature to s.
298 auto feature = parseEnumeration<XRReferenceSpaceType>(globalObject, sessionFeature);
299 if (!feature)
300 RETURN_FALSE_OR_CONTINUE(returnOnFailure);
301
302 // 3. If feature is already in granted, continue to the next entry.
303 if (resolvedFeatures.granted.contains(feature.value())) {
304 resolvedFeatures.requested.append(feature.value());
305 continue;
306 }
307
308 // 4. If the requesting document's origin is not allowed to use any feature policy required by feature
309 // as indicated by the feature requirements table, (return null|continue to next entry).
310
311 // 5. If session's XR device is not capable of supporting the functionality described by feature or the
312 // user agent has otherwise determined to reject the feature, (return null|continue to next entry).
313 if (!device->supportedFeatures(mode).contains(feature.value()))
314 RETURN_FALSE_OR_CONTINUE(returnOnFailure);
315
316 // 6. If the functionality described by feature requires explicit consent, append it to (consentRequired|consentOptional).
317 // 7. Else append feature to granted.
318 resolvedFeatures.granted.append(feature.value());
319
320 // https://immersive-web.github.io/webxr/#requested-features
321 // The combined list of feature descriptors given by the requiredFeatures and optionalFeatures are collectively considered the requested features for an XRSession.
322 resolvedFeatures.requested.append(feature.value());
323 }
324 return true;
325 };
326
327 if (!parseFeatures(requiredFeaturesWithDefaultFeatures, ParsingMode::Strict))
328 return std::nullopt;
329
330 parseFeatures(init.optionalFeatures, ParsingMode::Loose);
331 return resolvedFeatures;
332}

334// https://immersive-web.github.io/webxr/#request-the-xr-permission
335std::optional<WebXRSystem::FeatureList> WebXRSystem::resolveFeaturePermissions(XRSessionMode mode, const XRSessionInit& init, PlatformXR::Device* device, JSC::JSGlobalObject& globalObject) const
336{
337 // 1. Set status's granted to an empty FrozenArray.
338 // 2. Let requiredFeatures be descriptor's requiredFeatures.
339 // 3. Let optionalFeatures be descriptor's optionalFeatures.
340 // 4. Let device be the result of obtaining the current device for mode, requiredFeatures, and optionalFeatures.
341
342 // 5. Let result be the result of resolving the requested features given requiredFeatures,optionalFeatures, and mode.
343 auto resolvedFeatures = resolveRequestedFeatures(mode, init, device, globalObject);
344
345 // 6. If result is null, run the following steps:
346 // 6.1. Set status's state to "denied".
347 // 6.2. Abort these steps.
348 if (!resolvedFeatures)
349 return std::nullopt;
350
351 // 7. Let (consentRequired, consentOptional, granted) be the fields of result.
352 // 8. The user agent MAY at this point ask the user's permission for the calling algorithm to use any of the features
353 // in consentRequired and consentOptional. The results of these prompts should be included when determining if there
354 // is a clear signal of user intent for enabling these features.
355 // 9. For each feature in consentRequired perform the following steps:
356 // 9.1. The user agent MAY at this point ask the user's permission for the calling algorithm to use feature. The results
357 // of these prompts should be included when determining if there is a clear signal of user intent to enable feature.
358 // 9.2. If a clear signal of user intent to enable feature has not been determined, set status's state to "denied" and
359 // abort these steps.
360 // 9.3. If feature is not in granted, append feature to granted.
361 // 10. For each feature in consentOptional perform the following steps:
362 // 10.1. The user agent MAY at this point ask the user's permission for the calling algorithm to use feature. The results
363 // of these prompts should be included when determining if there is a clear signal of user intent to enable feature.
364 // 10.2. If a clear signal of user intent to enable feature has not been determined, continue to the next entry.
365 // 10.3. If feature is not in granted, append feature to granted.
366 // 11. Set status's granted to granted.
367 // 12. Set device's list of enabled features for mode to granted.
368 // 13. Set status's state to "granted".
369 device->setEnabledFeatures(mode, resolvedFeatures->granted);
370
371 return resolvedFeatures->requested;
372}

375// https://immersive-web.github.io/webxr/#dom-xrsystem-requestsession
376void WebXRSystem::requestSession(Document& document, XRSessionMode mode, const XRSessionInit& init, RequestSessionPromise&& promise)
377{
378 // 1. Let promise be a new Promise.
379 // 2. Let immersive be true if mode is an immersive session mode, and false otherwise.
380 // 3. Let global object be the relevant Global object for the XRSystem on which this method was invoked.
381 bool immersive = mode == XRSessionMode::ImmersiveAr || mode == XRSessionMode::ImmersiveVr;
382 auto* globalObject = document.domWindow();
383 if (!globalObject) {
384 promise.reject(Exception { InvalidAccessError});
385 return;
386 }
387
388 // 4. Check whether the session request is allowed as follows:
389 if (immersive) {
390 if (!immersiveSessionRequestIsAllowedForGlobalObject(*globalObject, document)) {
391 promise.reject(Exception { SecurityError });
392 return;
393 }
394 if (m_pendingImmersiveSession || m_activeImmersiveSession) {
395 promise.reject(Exception { InvalidStateError });
396 return;
397 }
398 m_pendingImmersiveSession = true;
399 } else if (!inlineSessionRequestIsAllowedForGlobalObject(*globalObject, document, init)) {
400 promise.reject(Exception { SecurityError });
401 return;
402 }
403
404 // 5. Run the following steps in parallel:
405 // 5.1 Let requiredFeatures be options' requiredFeatures.
406 // 5.2 Let optionalFeatures be options' optionalFeatures.
407 // 5.3 Set device to the result of obtaining the current device for mode, requiredFeatures, and optionalFeatures.
408 // 5.4 Queue a task to perform the following steps:
409 obtainCurrentDevice(mode, init.requiredFeatures, init.optionalFeatures, [this, protectedDocument = makeRef(document), immersive, init, mode, promise = WTFMove(promise)](auto* device) mutable {
410 auto rejectPromiseWithNotSupportedError = makeScopeExit([&]() {
411 promise.reject(Exception { NotSupportedError });
412 m_pendingImmersiveSession = false;
413 });
414
415 // 5.4.1 If device is null or device's list of supported modes does not contain mode, run the following steps:
416 // - Reject promise with a "NotSupportedError" DOMException.
417 // - If immersive is true, set pending immersive session to false.
418 // - Abort these steps.
419 if (!device || !device->supports(mode))
420 return;
421
422 auto* globalObject = protectedDocument->globalObject();
423 if (!globalObject)
424 return;
425
426 // WebKit does not currently support the Permissions API. https://w3c.github.io/permissions/
427 // However we do implement here the permission request algorithm without the
428 // Permissions API bits as it handles, among others, the session features parsing. We also
429 // do it here before creating the session as there is no need to do it on advance.
430 // FIXME: we just perform basic checks without asking any permission to the user so far. Maybe we should implement
431 // a mechanism similar to what others do involving passing a message to the UI process.
432
433 // 5.4.4 Let descriptor be an XRPermissionDescriptor initialized with session, requiredFeatures, and optionalFeatures
434 // 5.4.5 Let status be an XRPermissionStatus, initially null
435 // 5.4.6 Request the xr permission with descriptor and status.
436 // 5.4.7 If status' state is "denied" run the following steps: (same as above in 5.4.1)
437 auto requestedFeatures = resolveFeaturePermissions(mode, init, device, *globalObject);
438 if (!requestedFeatures)
439 return;
440
441 // 5.4.2 Let session be a new XRSession object.
442 // 5.4.3 Initialize the session with session, mode, and device.
443 auto session = WebXRSession::create(protectedDocument.get(), *this, mode, *device, WTFMove(*requestedFeatures));
444
445 // 5.4.8 Potentially set the active immersive session as follows:
446 if (immersive) {
447 m_activeImmersiveSession = session.copyRef();
448 m_pendingImmersiveSession = false;
449 } else
450 m_inlineSessions.add(session.copyRef());
451
452 // 5.4.9 Resolve promise with session.
453 promise.resolve(WTFMove(session));
454 rejectPromiseWithNotSupportedError.release();
455
456 // 5.4.10 is handled in WebXRSession::sessionDidInitializeInputSources.
457 });
458}

469void WebXRSystem::registerSimulatedXRDeviceForTesting(PlatformXR::Device& device)
470{
471 auto scriptExecutionContext = this->scriptExecutionContext();
472 if (!scriptExecutionContext || !scriptExecutionContext->settingsValues().webXREnabled)
473 return;
474
475 m_testingDevices++;
476 if (device.supports(XRSessionMode::ImmersiveVr) || device.supports(XRSessionMode::ImmersiveAr)) {
477 m_immersiveDevices.add(device);
478 m_activeImmersiveDevice = makeWeakPtr(device);
479 }
480 if (device.supports(XRSessionMode::Inline))
481 m_inlineXRDevice = makeWeakPtr(device);
482}
483
484void WebXRSystem::unregisterSimulatedXRDeviceForTesting(PlatformXR::Device& device)
485{
486 auto scriptExecutionContext = this->scriptExecutionContext();
487 if (!scriptExecutionContext || !scriptExecutionContext->settingsValues().webXREnabled)
488 return;
489
490 ASSERT(m_testingDevices);
491 bool removed = m_immersiveDevices.remove(device);
492 ASSERT_UNUSED(removed, removed || m_inlineXRDevice == &device);
493 if (m_activeImmersiveDevice == &device)
494 m_activeImmersiveDevice = nullptr;
495 if (m_inlineXRDevice == &device)
496 m_inlineXRDevice = makeWeakPtr(m_defaultInlineDevice);
497 m_testingDevices--;
498}

505 m_inlineSessions.remove(session);

535
536void WebXRSystem::DummyInlineDevice::requestFrame(PlatformXR::Device::RequestFrameCallback&& callback)
537{
538 if (!scriptExecutionContext())
539 return;
540 // Inline XR sessions rely on document.requestAnimationFrame to perform the render loop.
541 auto document = downcast<Document>(scriptExecutionContext());
542 if (!document)
543 return;
544
545 auto raf = InlineRequestAnimationFrameCallback::create(*m_scriptExecutionContext, [callback = WTFMove(callback)]() mutable {
546 PlatformXR::Device::FrameData data;
547 data.isTrackingValid = true;
548 data.isPositionValid = true;
549 data.shouldRender = true;
550 data.views.append({ });
551 callback(WTFMove(data));
552 });
553
554 document->requestAnimationFrame(raf);
555}
