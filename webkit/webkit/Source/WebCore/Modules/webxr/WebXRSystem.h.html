48namespace WebCore {
49
50class DOMWindow;
51class Navigator;
52class ScriptExecutionContext;
53class WebXRSession;
54struct XRSessionInit;
55
56class WebXRSystem final : public RefCounted<WebXRSystem>, public EventTargetWithInlineData, public ActiveDOMObject {
57 WTF_MAKE_ISO_ALLOCATED(WebXRSystem);
58public:
59 using IsSessionSupportedPromise = DOMPromiseDeferred<IDLBoolean>;
60 using RequestSessionPromise = DOMPromiseDeferred<IDLInterface<WebXRSession>>;
61
62 static Ref<WebXRSystem> create(Navigator&);
63 ~WebXRSystem();
64
65 using RefCounted<WebXRSystem>::ref;
66 using RefCounted<WebXRSystem>::deref;
67
68 void isSessionSupported(XRSessionMode, IsSessionSupportedPromise&&);
69 void requestSession(Document&, XRSessionMode, const XRSessionInit&, RequestSessionPromise&&);
70
71 // This is also needed by WebGLRenderingContextBase::makeXRCompatible() and HTMLCanvasElement::createContextWebGL().
72 void ensureImmersiveXRDeviceIsSelected(CompletionHandler<void()>&&);
73 bool hasActiveImmersiveXRDevice() { return !!m_activeImmersiveDevice; }
74
75 void sessionEnded(WebXRSession&);
76
77 // For testing purpouses only.
78 WEBCORE_EXPORT void registerSimulatedXRDeviceForTesting(PlatformXR::Device&);
79 WEBCORE_EXPORT void unregisterSimulatedXRDeviceForTesting(PlatformXR::Device&);
80
81 Navigator* navigator();
82
83protected:
84 // EventTarget
85 EventTargetInterface eventTargetInterface() const override { return WebXRSystemEventTargetInterfaceType; }
86 ScriptExecutionContext* scriptExecutionContext() const override { return ActiveDOMObject::scriptExecutionContext(); }
87 void refEventTarget() override { ref(); }
88 void derefEventTarget() override { deref(); }
89
90 // ActiveDOMObject
91 const char* activeDOMObjectName() const override;
92 void stop() override;
93
94private:
95 WebXRSystem(Navigator&);
96
97 using FeatureList = PlatformXR::Device::FeatureList;
98 using JSFeatureList = Vector<JSC::JSValue>;
99 void obtainCurrentDevice(XRSessionMode, const JSFeatureList& requiredFeatures, const JSFeatureList& optionalFeatures, CompletionHandler<void(PlatformXR::Device*)>&&);
100
101 bool immersiveSessionRequestIsAllowedForGlobalObject(DOMWindow&, Document&) const;
102 bool inlineSessionRequestIsAllowedForGlobalObject(DOMWindow&, Document&, const XRSessionInit&) const;
103
104 struct ResolvedRequestedFeatures;
105 std::optional<ResolvedRequestedFeatures> resolveRequestedFeatures(XRSessionMode, const XRSessionInit&, PlatformXR::Device*, JSC::JSGlobalObject&) const;
106 std::optional<FeatureList> resolveFeaturePermissions(XRSessionMode, const XRSessionInit&, PlatformXR::Device*, JSC::JSGlobalObject&) const;
107
108 // https://immersive-web.github.io/webxr/#default-inline-xr-device
109 class DummyInlineDevice final : public PlatformXR::Device, private ContextDestructionObserver {
110 public:
111 explicit DummyInlineDevice(ScriptExecutionContext&);
112
113 private:
114 void initializeTrackingAndRendering(PlatformXR::SessionMode) final { }
115 void shutDownTrackingAndRendering() final { }
116 void initializeReferenceSpace(PlatformXR::ReferenceSpaceType) final { }
117
118 void requestFrame(PlatformXR::Device::RequestFrameCallback&&) final;
119 Vector<Device::ViewData> views(XRSessionMode) const final;
120 std::optional<PlatformXR::LayerHandle> createLayerProjection(uint32_t, uint32_t, bool) final { return std::nullopt; }
121 void deleteLayer(PlatformXR::LayerHandle) final { }
122 };
123
124 WeakPtr<Navigator> m_navigator;
125 DummyInlineDevice m_defaultInlineDevice;
126
127 bool m_immersiveXRDevicesHaveBeenEnumerated { false };
128 uint m_testingDevices { 0 };
129
130 bool m_pendingImmersiveSession { false };
131 RefPtr<WebXRSession> m_activeImmersiveSession;
132
133 // We use a set here although the specs talk about a list of inline sessions
134 // https://immersive-web.github.io/webxr/#list-of-inline-sessions.
135 HashSet<Ref<WebXRSession>> m_inlineSessions;
136
137 WeakPtr<PlatformXR::Device> m_activeImmersiveDevice;
138 WeakHashSet<PlatformXR::Device> m_immersiveDevices;
139 WeakPtr<PlatformXR::Device> m_inlineXRDevice;
140};
