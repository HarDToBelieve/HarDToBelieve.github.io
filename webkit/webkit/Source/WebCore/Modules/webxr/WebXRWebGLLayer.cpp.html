59// Arbitrary value for minimum framebuffer scaling.
60// Below this threshold the resulting framebuffer would be too small to see.
61constexpr double MinFramebufferScalingFactor = 0.2;
62
63static ExceptionOr<std::unique_ptr<WebXROpaqueFramebuffer>> createOpaqueFramebuffer(WebXRSession& session, WebGLRenderingContextBase& context, const XRWebGLLayerInit& init)
64{
65 auto device = session.device();
66 if (!device)
67 return Exception { OperationError, "Cannot create an XRWebGLLayer with an XRSession that has ended." };
68
69 // 9.1. Initialize layer’s antialias to layerInit’s antialias value.
70 // 9.2. Let framebufferSize be the recommended WebGL framebuffer resolution multiplied by layerInit's framebufferScaleFactor.
71
72 float scaleFactor = std::clamp(init.framebufferScaleFactor, MinFramebufferScalingFactor, device->maxFramebufferScalingFactor());
73
74 IntSize recommendedSize = session.recommendedWebGLFramebufferResolution();
75 auto width = static_cast<uint32_t>(std::ceil(recommendedSize.width() * scaleFactor));
76 auto height = static_cast<uint32_t>(std::ceil(recommendedSize.height() * scaleFactor));
77
78 // 9.3. Initialize layer’s framebuffer to a new opaque framebuffer with the dimensions framebufferSize
79 // created with context, session initialized to session, and layerInit’s depth, stencil, and alpha values.
80 // 9.4. Allocate and initialize resources compatible with session’s XR device, including GPU accessible memory buffers,
81 // as required to support the compositing of layer.
82 // 9.5. If layer’s resources were unable to be created for any reason, throw an OperationError and abort these steps.
83 auto layerHandle = device->createLayerProjection(width, height, init.alpha);
84 if (!layerHandle)
85 return Exception { OperationError, "Unable to allocate XRWebGLLayer GPU resources."};
86
87 WebXROpaqueFramebuffer::Attributes attributes {
88 .alpha = init.alpha,
89 .antialias = init.antialias,
90 .depth = init.depth,
91 .stencil = init.stencil
92 };
93
94 auto framebuffer = WebXROpaqueFramebuffer::create(*layerHandle, context, WTFMove(attributes), width, height);
95 if (!framebuffer)
96 return Exception { OperationError, "Unable to create a framebuffer." };
97
98 return framebuffer;
99}

101// https://immersive-web.github.io/webxr/#dom-xrwebgllayer-xrwebgllayer
102ExceptionOr<Ref<WebXRWebGLLayer>> WebXRWebGLLayer::create(Ref<WebXRSession>&& session, WebXRRenderingContext&& context, const XRWebGLLayerInit& init)
103{
104 // 1. Let layer be a new XRWebGLLayer
105 // 2. If session’s ended value is true, throw an InvalidStateError and abort these steps.
106 if (session->ended())
107 return Exception { InvalidStateError, "Cannot create an XRWebGLLayer with an XRSession that has ended." };
108
109 // 3. If context is lost, throw an InvalidStateError and abort these steps.
110 // 4. If session is an immersive session and context’s XR compatible boolean is false, throw
111 // an InvalidStateError and abort these steps.
112 return WTF::switchOn(context,
113 [&](const RefPtr<WebGLRenderingContextBase>& baseContext) -> ExceptionOr<Ref<WebXRWebGLLayer>>
114 {
115 if (baseContext->isContextLost())
116 return Exception { InvalidStateError, "Cannot create an XRWebGLLayer with a lost WebGL context." };
117
118 auto mode = session->mode();
119 if ((mode == XRSessionMode::ImmersiveAr || mode == XRSessionMode::ImmersiveVr) && !baseContext->isXRCompatible())
120 return Exception { InvalidStateError, "Cannot create an XRWebGLLayer with WebGL context not marked as XR compatible." };
121
122
123 // 5. Initialize layer’s context to context. (see constructor)
124 // 6. Initialize layer’s session to session. (see constructor)
125 // 7. Initialize layer’s ignoreDepthValues as follows. (see constructor)
126 // 7.1 If layerInit’s ignoreDepthValues value is false and the XR Compositor will make use of depth values,
127 // Initialize layer’s ignoreDepthValues to false.
128 // 7.2. Else Initialize layer’s ignoreDepthValues to true
129 // TODO: ask XR compositor for depth value usages support
130 bool ignoreDepthValues = true;
131 // 8. Initialize layer’s composition disabled boolean as follows.
132 // If session is an inline session -> Initialize layer's composition disabled to true
133 // Otherwise -> Initialize layer's composition disabled boolean to false
134 bool isCompositionEnabled = session->mode() != XRSessionMode::Inline;
135 bool antialias = false;
136 std::unique_ptr<WebXROpaqueFramebuffer> framebuffer;
137
138 // 9. If layer's composition enabled boolean is true:
139 if (isCompositionEnabled) {
140 auto createResult = createOpaqueFramebuffer(session.get(), *baseContext, init);
141 if (createResult.hasException())
142 return createResult.releaseException();
143 framebuffer = createResult.releaseReturnValue();
144 } else {
145 // Otherwise
146 // 9.1. Initialize layer’s antialias to layer’s context's actual context parameters antialias value.
147 if (auto attributes = baseContext->getContextAttributes())
148 antialias = attributes->antialias;
149 // 9.2 Initialize layer's framebuffer to null.
150 }
151
152 // 10. Return layer.
153 return adoptRef(*new WebXRWebGLLayer(WTFMove(session), WTFMove(context), WTFMove(framebuffer), antialias, ignoreDepthValues, isCompositionEnabled));
154 },
155 [](WTF::Monostate) {
156 ASSERT_NOT_REACHED();
157 return Exception { InvalidStateError };
158 }
159 );
160}

162WebXRWebGLLayer::WebXRWebGLLayer(Ref<WebXRSession>&& session, WebXRRenderingContext&& context, std::unique_ptr<WebXROpaqueFramebuffer>&& framebuffer,
163 bool antialias, bool ignoreDepthValues, bool isCompositionEnabled)
164 : WebXRLayer(session->scriptExecutionContext())
165 , m_session(WTFMove(session))
166 , m_context(WTFMove(context))
167 , m_leftViewportData({ WebXRViewport::create({ }) })
168 , m_rightViewportData({ WebXRViewport::create({ }) })
169 , m_framebuffer(WTFMove(framebuffer))
170 , m_antialias(antialias)
171 , m_ignoreDepthValues(ignoreDepthValues)
172 , m_isCompositionEnabled(isCompositionEnabled)
173{
174}

176WebXRWebGLLayer::~WebXRWebGLLayer()
177{
178 auto canvasElement = canvas();
179 if (canvasElement)
180 canvasElement->removeObserver(*this);
181 if (m_framebuffer) {
182 auto device = m_session->device();
183 if (device)
184 device->deleteLayer(m_framebuffer->handle());
185 }
186}

223// https://immersive-web.github.io/webxr/#dom-xrwebgllayer-getviewport
224ExceptionOr<RefPtr<WebXRViewport>> WebXRWebGLLayer::getViewport(WebXRView& view)
225{
226 // 1. Let session be view’s session.
227 // 2. Let frame be session’s animation frame.
228 // 3. If session is not equal to layer’s session, throw an InvalidStateError and abort these steps.
229 if (&view.frame().session() != m_session.ptr())
230 return Exception { InvalidStateError };
231
232 // 4. If frame’s active boolean is false, throw an InvalidStateError and abort these steps.
233 // 5. If view’s frame is not equal to frame, throw an InvalidStateError and abort these steps.
234 if (!view.frame().isActive() || !view.frame().isAnimationFrame())
235 return Exception { InvalidStateError };
236
237 auto& viewportData = view.eye() == XREye::Right ? m_rightViewportData : m_leftViewportData;
238
239 // 6. If the viewport modifiable flag is true and view’s requested viewport scale is not equal to current viewport scale:
240 // 6.1 Set current viewport scale to requested viewport scale.
241 // 6.2 Compute the scaled viewport.
242 if (view.isViewportModifiable() && view.requestedViewportScale() != viewportData.currentScale) {
243 viewportData.currentScale = view.requestedViewportScale();
244 m_viewportsDirty = true;
245 }
246
247 // 7. Set the view’s viewport modifiable flag to false.
248 view.setViewportModifiable(false);
249
250 if (m_viewportsDirty)
251 computeViewports();
252
253 // 8. Let viewport be the XRViewport from the list of viewport objects associated with view.
254 // 9. Return viewport.
255 return RefPtr<WebXRViewport>(viewportData.viewport.copyRef());
256}

258double WebXRWebGLLayer::getNativeFramebufferScaleFactor(const WebXRSession& session)
259{
260 if (session.ended())
261 return 0.0;
262
263 IntSize nativeSize = session.nativeWebGLFramebufferResolution();
264 IntSize recommendedSize = session.recommendedWebGLFramebufferResolution();
265
266 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!recommendedSize.isZero());
267 return (nativeSize / recommendedSize).width();
268}

284void WebXRWebGLLayer::startFrame(const PlatformXR::Device::FrameData& data)
285{
286 ASSERT(m_framebuffer);
287
288 auto it = data.layers.find(m_framebuffer->handle());
289 if (it == data.layers.end()) {
290 // For some reason the device didn't provide a texture for this frame.
291 // The frame is ignored and the device can recover the texture in future frames;
292 return;
293 }
294
295 m_framebuffer->startFrame(it->value);
296}
297
298PlatformXR::Device::Layer WebXRWebGLLayer::endFrame()
299{
300 ASSERT(m_framebuffer);
301 m_framebuffer->endFrame();
302
303 Vector<PlatformXR::Device::LayerView> views {
304 { PlatformXR::Eye::Left, m_leftViewportData.viewport->rect() },
305 { PlatformXR::Eye::Right, m_rightViewportData.viewport->rect() }
306 };
307
308 return PlatformXR::Device::Layer { .handle = m_framebuffer->handle(), .visible = true, .views = WTFMove(views) };
309}

316// https://immersive-web.github.io/webxr/#xrview-obtain-a-scaled-viewport
317void WebXRWebGLLayer::computeViewports()
318{
319 auto roundDown = [](double value) -> int {
320 // Round down to integer value and ensure that the value is not zero.
321 return std::max(1, static_cast<int>(std::floor(value)));
322 };
323
324 auto width = framebufferWidth();
325 auto height = framebufferHeight();
326
327 if (m_session->mode() == XRSessionMode::ImmersiveVr) {
328 // Update left viewport
329 auto scale = m_leftViewportData.currentScale;
330 m_leftViewportData.viewport->updateViewport(IntRect(0, 0, roundDown(width * 0.5 * scale), roundDown(height * scale)));
331
332 // Update right viewport
333 scale = m_rightViewportData.currentScale;
334 m_rightViewportData.viewport->updateViewport(IntRect(width * 0.5, 0, roundDown(width * 0.5 * scale), roundDown(height * scale)));
335 } else {
336 // We reuse m_leftViewport for XREye::None.
337 m_leftViewportData.viewport->updateViewport(IntRect(0, 0, width, height));
338 }
339
340 m_viewportsDirty = false;
341}
