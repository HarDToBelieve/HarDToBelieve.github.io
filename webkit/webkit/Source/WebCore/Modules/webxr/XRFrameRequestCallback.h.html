26#pragma once
27
28#if ENABLE(WEBXR)
29
30#include "ActiveDOMCallback.h"
31#include "CallbackResult.h"
32#include <wtf/RefCounted.h>
33
34namespace WebCore {
35
36class WebXRFrame;
37
38class XRFrameRequestCallback : public RefCounted<XRFrameRequestCallback>, public ActiveDOMCallback {
39public:
40 using ActiveDOMCallback::ActiveDOMCallback;
41
42 virtual CallbackResult<void> handleEvent(double highResTimeMs, WebXRFrame&) = 0;
43
44 unsigned callbackId() { ASSERT(m_id); return m_id; }
45 void setCallbackId(unsigned id) { ASSERT(!m_id); m_id = id; }
46 void setFiredOrCancelled() { m_firedOrCancelled = true; }
47 bool isFiredOrCancelled() const { return m_firedOrCancelled; }
48
49private:
50 unsigned m_id { 0 };
51 bool m_firedOrCancelled { false };
52};
53
54} // namespace WebCore
55
56#endif // ENABLE(WEBXR)
