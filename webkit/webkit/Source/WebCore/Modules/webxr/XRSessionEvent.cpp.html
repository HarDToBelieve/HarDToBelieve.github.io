31#include <wtf/IsoMallocInlines.h>
32#include "WebXRSession.h"

34namespace WebCore {
35
36WTF_MAKE_ISO_ALLOCATED_IMPL(XRSessionEvent);
37
38Ref<XRSessionEvent> XRSessionEvent::create(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
39{
40 return adoptRef(*new XRSessionEvent(type, initializer, isTrusted));
41}
42
43XRSessionEvent::XRSessionEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
44 : Event(type, initializer, isTrusted)
45 , m_session(initializer.session)
46{
47 ASSERT(m_session);
48}
49
50XRSessionEvent::~XRSessionEvent() = default;
51
52EventInterface XRSessionEvent::eventInterface() const
53{
54 return XRSessionEventInterfaceType;
55}
56
57const WebXRSession& XRSessionEvent::session() const
58{
59 ASSERT(m_session);
60 return *m_session;
61}
