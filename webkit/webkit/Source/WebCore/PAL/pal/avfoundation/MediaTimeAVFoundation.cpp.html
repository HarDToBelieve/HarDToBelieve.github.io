35static bool CMTimeHasFlags(const CMTime& cmTime, uint32_t flags)
36{
37 return (cmTime.flags & flags) == flags;
38}
39
40MediaTime toMediaTime(const CMTime& cmTime)
41{
42 uint32_t flags = 0;
43 if (CMTimeHasFlags(cmTime, kCMTimeFlags_Valid))
44 flags |= MediaTime::Valid;
45 if (CMTimeHasFlags(cmTime, kCMTimeFlags_Valid | kCMTimeFlags_HasBeenRounded))
46 flags |= MediaTime::HasBeenRounded;
47 if (CMTimeHasFlags(cmTime, kCMTimeFlags_Valid | kCMTimeFlags_PositiveInfinity))
48 flags |= MediaTime::PositiveInfinite;
49 if (CMTimeHasFlags(cmTime, kCMTimeFlags_Valid | kCMTimeFlags_NegativeInfinity))
50 flags |= MediaTime::NegativeInfinite;
51 if (CMTimeHasFlags(cmTime, kCMTimeFlags_Valid | kCMTimeFlags_Indefinite))
52 flags |= MediaTime::Indefinite;
53
54 return MediaTime(cmTime.value, cmTime.timescale, flags);
55}
56
57CMTime toCMTime(const MediaTime& mediaTime)
58{
59 CMTime time;
60
61 if (mediaTime.hasDoubleValue())
62 time = CMTimeMakeWithSeconds(mediaTime.toDouble(), mediaTime.timeScale());
63 else
64 time = CMTimeMake(mediaTime.timeValue(), mediaTime.timeScale());
65
66 if (mediaTime.isValid())
67 time.flags |= kCMTimeFlags_Valid;
68 if (mediaTime.hasBeenRounded())
69 time.flags |= kCMTimeFlags_HasBeenRounded;
70 if (mediaTime.isPositiveInfinite())
71 time.flags |= kCMTimeFlags_PositiveInfinity;
72 if (mediaTime.isNegativeInfinite())
73 time.flags |= kCMTimeFlags_NegativeInfinity;
74
75 return time;
76}
