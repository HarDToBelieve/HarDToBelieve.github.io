33struct CryptoDigestContext {
34 CryptoDigest::Algorithm algorithm;
35 void* ccContext;
36};
37
38inline CC_SHA1_CTX* toSHA1Context(CryptoDigestContext* context)
39{
40 ASSERT(context->algorithm == CryptoDigest::Algorithm::SHA_1);
41 return static_cast<CC_SHA1_CTX*>(context->ccContext);
42}
43inline CC_SHA256_CTX* toSHA224Context(CryptoDigestContext* context)
44{
45 ASSERT(context->algorithm == CryptoDigest::Algorithm::SHA_224);
46 return static_cast<CC_SHA256_CTX*>(context->ccContext);
47}
48inline CC_SHA256_CTX* toSHA256Context(CryptoDigestContext* context)
49{
50 ASSERT(context->algorithm == CryptoDigest::Algorithm::SHA_256);
51 return static_cast<CC_SHA256_CTX*>(context->ccContext);
52}
53inline CC_SHA512_CTX* toSHA384Context(CryptoDigestContext* context)
54{
55 ASSERT(context->algorithm == CryptoDigest::Algorithm::SHA_384);
56 return static_cast<CC_SHA512_CTX*>(context->ccContext);
57}
58inline CC_SHA512_CTX* toSHA512Context(CryptoDigestContext* context)
59{
60 ASSERT(context->algorithm == CryptoDigest::Algorithm::SHA_512);
61 return static_cast<CC_SHA512_CTX*>(context->ccContext);
62}

64CryptoDigest::CryptoDigest()
65 : m_context(new CryptoDigestContext)
66{
67}
68
69CryptoDigest::~CryptoDigest()
70{
71 switch (m_context->algorithm) {
72 case CryptoDigest::Algorithm::SHA_1:
73 delete toSHA1Context(m_context.get());
74 return;
75 case CryptoDigest::Algorithm::SHA_224:
76 delete toSHA224Context(m_context.get());
77 return;
78 case CryptoDigest::Algorithm::SHA_256:
79 delete toSHA256Context(m_context.get());
80 return;
81 case CryptoDigest::Algorithm::SHA_384:
82 delete toSHA384Context(m_context.get());
83 return;
84 case CryptoDigest::Algorithm::SHA_512:
85 delete toSHA512Context(m_context.get());
86 return;
87 }
88}

91std::unique_ptr<CryptoDigest> CryptoDigest::create(CryptoDigest::Algorithm algorithm)
92{
93 std::unique_ptr<CryptoDigest> digest(new CryptoDigest);
94 digest->m_context->algorithm = algorithm;
95
96 switch (algorithm) {
97 case CryptoDigest::Algorithm::SHA_1: {
98 CC_SHA1_CTX* context = new CC_SHA1_CTX;
99 digest->m_context->ccContext = context;
100 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
101 CC_SHA1_Init(context);
102 ALLOW_DEPRECATED_DECLARATIONS_END
103 return digest;
104 }
105 case CryptoDigest::Algorithm::SHA_224: {
106 CC_SHA256_CTX* context = new CC_SHA256_CTX;
107 digest->m_context->ccContext = context;
108 CC_SHA224_Init(context);
109 return digest;
110 }
111 case CryptoDigest::Algorithm::SHA_256: {
112 CC_SHA256_CTX* context = new CC_SHA256_CTX;
113 digest->m_context->ccContext = context;
114 CC_SHA256_Init(context);
115 return digest;
116 }
117 case CryptoDigest::Algorithm::SHA_384: {
118 CC_SHA512_CTX* context = new CC_SHA512_CTX;
119 digest->m_context->ccContext = context;
120 CC_SHA384_Init(context);
121 return digest;
122 }
123 case CryptoDigest::Algorithm::SHA_512: {
124 CC_SHA512_CTX* context = new CC_SHA512_CTX;
125 digest->m_context->ccContext = context;
126 CC_SHA512_Init(context);
127 return digest;
128 }
129 }

132void CryptoDigest::addBytes(const void* input, size_t length)
133{
134 switch (m_context->algorithm) {
135 case CryptoDigest::Algorithm::SHA_1:
136 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
137 CC_SHA1_Update(toSHA1Context(m_context.get()), input, length);
138 ALLOW_DEPRECATED_DECLARATIONS_END
139 return;
140 case CryptoDigest::Algorithm::SHA_224:
141 CC_SHA224_Update(toSHA224Context(m_context.get()), input, length);
142 return;
143 case CryptoDigest::Algorithm::SHA_256:
144 CC_SHA256_Update(toSHA256Context(m_context.get()), input, length);
145 return;
146 case CryptoDigest::Algorithm::SHA_384:
147 CC_SHA384_Update(toSHA384Context(m_context.get()), input, length);
148 return;
149 case CryptoDigest::Algorithm::SHA_512:
150 CC_SHA512_Update(toSHA512Context(m_context.get()), input, length);
151 return;
152 }
153}

155Vector<uint8_t> CryptoDigest::computeHash()
156{
157 Vector<uint8_t> result;
158 switch (m_context->algorithm) {
159 case CryptoDigest::Algorithm::SHA_1:
160 result.resize(CC_SHA1_DIGEST_LENGTH);
161 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
162 CC_SHA1_Final(result.data(), toSHA1Context(m_context.get()));
163 ALLOW_DEPRECATED_DECLARATIONS_END
164 break;
165 case CryptoDigest::Algorithm::SHA_224:
166 result.resize(CC_SHA224_DIGEST_LENGTH);
167 CC_SHA224_Final(result.data(), toSHA224Context(m_context.get()));
168 break;
169 case CryptoDigest::Algorithm::SHA_256:
170 result.resize(CC_SHA256_DIGEST_LENGTH);
171 CC_SHA256_Final(result.data(), toSHA256Context(m_context.get()));
172 break;
173 case CryptoDigest::Algorithm::SHA_384:
174 result.resize(CC_SHA384_DIGEST_LENGTH);
175 CC_SHA384_Final(result.data(), toSHA384Context(m_context.get()));
176 break;
177 case CryptoDigest::Algorithm::SHA_512:
178 result.resize(CC_SHA512_DIGEST_LENGTH);
179 CC_SHA512_Final(result.data(), toSHA512Context(m_context.get()));
180 break;
181 }
182 return result;
183}
