105// FIXME: The SPIs above can be declared by WebKit without causing redeclaration errors on Catalina
106// internal SDKs because we can avoid including the SPIs' private headers from PassKit, but we can't
107// avoid importing PKPaymentSetupFeature.h due to how many other private headers include it. To avoid
108// redeclaration errors while continuing to support all Catalina SDKs, declare -supportedOptions
109// only when building against an internal SDK without PKPaymentInstallmentConfiguration.h (so that we
110// can implement a -respondsToSelector: check). The __has_include portion of this check can be
111// removed once the minimum supported Catalina internal SDK is known to contain this private header.
112#if !USE(APPLE_INTERNAL_SDK) || !__has_include(<PassKitCore/PKPaymentInstallmentConfiguration.h>)
113
114typedef NS_OPTIONS(NSInteger, PKPaymentSetupFeatureSupportedOptions) {
115 PKPaymentSetupFeatureSupportedOptionsInstallments = 1 << 0,
116};
117
118@interface PKPaymentSetupFeature ()
119@property (nonatomic, assign, readonly) PKPaymentSetupFeatureSupportedOptions supportedOptions;
120@end
