42namespace PAL {
43
44void popUpMenu(NSMenu *menu, NSPoint location, float width, NSView *view, int selectedItem, NSFont *font, NSControlSize controlSize, bool usesCustomAppearance)
45{
46 NSRect adjustedPopupBounds = [view.window convertRectToScreen:[view convertRect:view.bounds toView:nil]];
47 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
48 if (controlSize != NSMiniControlSize) {
49 ALLOW_DEPRECATED_DECLARATIONS_END
50 adjustedPopupBounds.origin.x -= 3;
51 adjustedPopupBounds.origin.y -= 1;
52 adjustedPopupBounds.size.width += 6;
53 }
54
55 // These numbers were extracted from visual inspection as the menu animates shut.
56 NSSize labelOffset = NSMakeSize(11, 1);
57 if (menu.userInterfaceLayoutDirection == NSUserInterfaceLayoutDirectionRightToLeft)
58 labelOffset = NSMakeSize(24, 1);
59
60 auto options = adoptNS([@{
61 NSPopUpMenuPopupButtonBounds : [NSValue valueWithRect:adjustedPopupBounds],
62 NSPopUpMenuPopupButtonLabelOffset : [NSValue valueWithSize:labelOffset],
63 NSPopUpMenuPopupButtonSize : @(controlSize)
64 } mutableCopy]);
65
66 if (usesCustomAppearance)
67 [options setObject:@"" forKey:NSPopUpMenuPopupButtonWidget];
68
69 [[menu _menuImpl] popUpMenu:menu atLocation:location width:width forView:view withSelectedItem:selectedItem withFont:font withFlags:(usesCustomAppearance ? 0 : NSPopUpMenuIsPopupButton) withOptions:options.get()];
70}
