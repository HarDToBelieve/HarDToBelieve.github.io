26#import "config.h"
27#import "KillRing.h"
28
29#if PLATFORM(MAC)
30
31namespace PAL {
32
33extern "C" {
34
35// Kill ring calls. Would be better to use NSKillRing.h, but that's not available as API or SPI.
36
37void _NSInitializeKillRing();
38void _NSAppendToKillRing(NSString *);
39void _NSPrependToKillRing(NSString *);
40NSString *_NSYankFromKillRing();
41void _NSNewKillRingSequence();
42void _NSSetKillRingToYankedState();
43void _NSResetKillRingOperationFlag();
44
45}
46
47static void initializeKillRingIfNeeded()
48{
49 static bool initializedKillRing = false;
50 if (!initializedKillRing) {
51 initializedKillRing = true;
52 _NSInitializeKillRing();
53 }
54}
55
56void KillRing::append(const String& string)
57{
58 initializeKillRingIfNeeded();
59 // Necessary to prevent an implicit new sequence if the previous command was NSPrependToKillRing.
60 _NSResetKillRingOperationFlag();
61 _NSAppendToKillRing(string);
62}
63
64void KillRing::prepend(const String& string)
65{
66 initializeKillRingIfNeeded();
67 // Necessary to prevent an implicit new sequence if the previous command was NSAppendToKillRing.
68 _NSResetKillRingOperationFlag();
69 _NSPrependToKillRing(string);
70}
71
72String KillRing::yank()
73{
74 initializeKillRingIfNeeded();
75 return _NSYankFromKillRing();
76}
77
78void KillRing::startNewSequence()
79{
80 initializeKillRingIfNeeded();
81 _NSNewKillRingSequence();
82}
83
84void KillRing::setToYankedState()
85{
86 initializeKillRingIfNeeded();
87 _NSSetKillRingToYankedState();
88}
89
90} // namespace PAL
91
92#endif // PLATFORM(MAC)
