44AXLogger::AXLogger(const String& methodName)
45 : m_methodName(methodName)
46{
47 if (auto* channel = getLogChannel("Accessibility"))
48 channel->level = WTFLogLevel::Debug;
49
50 if (!m_methodName.isEmpty())
51 LOG_WITH_STREAM(Accessibility, stream << m_methodName << " {");
52}
53
54AXLogger::~AXLogger()
55{
56 if (!m_methodName.isEmpty())
57 LOG_WITH_STREAM(Accessibility, stream << "} " << m_methodName);
58}

77void AXLogger::log(const Vector<RefPtr<AXCoreObject>>& objects)
78{
79 TextStream stream(TextStream::LineMode::MultipleLine);
80
81 stream << "[";
82 for (auto object : objects) {
83 if (object)
84 stream << *object;
85 else
86 stream << "null";
87 }
88 stream << "]";
89
90 LOG(Accessibility, "%s", stream.release().utf8().data());
91}
92
93void AXLogger::add(TextStream& stream, const RefPtr<AXCoreObject>& object, bool recursive)
94{
95 if (!object)
96 return;
97
98 stream.increaseIndent();
99 stream << *object;
100
101 if (recursive) {
102 for (auto& child : object->children())
103 add(stream, child, true);
104 }
105 stream.decreaseIndent();
106}

300TextStream& operator<<(TextStream& stream, const AccessibilitySearchCriteria& criteria)
301{
302 TextStream::GroupScope groupScope(stream);
303 stream << "SearchCriteria " << &criteria;
304 stream.dumpProperty("anchorObject", criteria.anchorObject);
305 stream.dumpProperty("startObject", criteria.startObject);
306 stream.dumpProperty("searchDirection", criteria.searchDirection);
307
308 stream.nextLine();
309 stream << "(searchKeys [";
310 for (auto searchKey : criteria.searchKeys)
311 stream << searchKey << ", ";
312 stream << "])";
313
314 stream.dumpProperty("searchText", criteria.searchText);
315 stream.dumpProperty("resultsLimit", criteria.resultsLimit);
316 stream.dumpProperty("visibleOnly", criteria.visibleOnly);
317 stream.dumpProperty("immediateDescendantsOnly", criteria.immediateDescendantsOnly);
318
319 return stream;
320}

482TextStream& operator<<(TextStream& stream, const AXCoreObject& object)
483{
484 TextStream::GroupScope groupScope(stream);
485 stream << "objectID " << object.objectID();
486 stream.dumpProperty("identifierAttribute", object.identifierAttribute());
487 auto role = object.roleValue();
488 stream.dumpProperty("roleValue", role);
489
490 auto* objectWithInterestingHTML = role == AccessibilityRole::Button ? // Add here other roles of interest.
491 &object : nullptr;
492 auto* parent = object.parentObject();
493 if (role == AccessibilityRole::StaticText && parent)
494 objectWithInterestingHTML = parent;
495 if (objectWithInterestingHTML)
496 stream.dumpProperty("outerHTML", objectWithInterestingHTML->outerHTML());
497
498 stream.dumpProperty("address", &object);
499 stream.dumpProperty("wrapper", object.wrapper());
500
501 stream.dumpProperty("parentObject", parent ? parent->objectID() : 0);
502#if PLATFORM(COCOA)
503 stream.dumpProperty("remoteParentObject", object.remoteParentObject());
504#endif
505
506 return stream;
507}

515 stream.dumpProperty("focusedNodeID", tree.m_focusedNodeID);
516 AXLogger::add(stream, tree.rootNode(), true);

521TextStream& operator<<(TextStream& stream, AXObjectCache& axObjectCache)
522{
523 TextStream::GroupScope groupScope(stream);
524 stream << "AXObjectCache " << &axObjectCache;
525
526 if (auto* root = axObjectCache.get(axObjectCache.document().view()))
527 AXLogger::add(stream, root, true);
528 else
529 stream << "No root!";
530
531 return stream;
532}
