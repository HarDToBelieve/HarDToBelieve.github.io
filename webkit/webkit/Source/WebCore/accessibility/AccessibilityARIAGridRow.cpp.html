49bool AccessibilityARIAGridRow::isARIATreeGridRow() const
50{
51 AccessibilityObject* parent = parentTable();
52 if (!parent)
53 return false;
54
55 return parent->isTreeGrid();
56}

58AXCoreObject::AccessibilityChildrenVector AccessibilityARIAGridRow::disclosedRows()
59{
60 AccessibilityChildrenVector disclosedRows;
61 // The contiguous disclosed rows will be the rows in the table that
62 // have an aria-level of plus 1 from this row.
63 AccessibilityObject* parent = parentObjectUnignored();
64 if (!is<AccessibilityTable>(*parent) || !downcast<AccessibilityTable>(*parent).isExposable())
65 return disclosedRows;
66
67 // Search for rows that match the correct level.
68 // Only take the subsequent rows from this one that are +1 from this row's level.
69 int index = rowIndex();
70 if (index < 0)
71 return disclosedRows;
72
73 unsigned level = hierarchicalLevel();
74 auto allRows = parent->rows();
75 int rowCount = allRows.size();
76 for (int k = index + 1; k < rowCount; ++k) {
77 auto* row = allRows[k].get();
78 // Stop at the first row that doesn't match the correct level.
79 if (row->hierarchicalLevel() != level + 1)
80 break;
81
82 disclosedRows.append(row);
83 }
84
85 return disclosedRows;
86}

88AXCoreObject* AccessibilityARIAGridRow::disclosedByRow() const
89{
90 // The row that discloses this one is the row in the table
91 // that is aria-level subtract 1 from this row.
92 AccessibilityObject* parent = parentObjectUnignored();
93 if (!is<AccessibilityTable>(*parent) || !downcast<AccessibilityTable>(*parent).isExposable())
94 return nullptr;
95
96 // If the level is 1 or less, than nothing discloses this row.
97 unsigned level = hierarchicalLevel();
98 if (level <= 1)
99 return nullptr;
100
101 // Search for the previous row that matches the correct level.
102 int index = rowIndex();
103 auto allRows = parent->rows();
104 int rowCount = allRows.size();
105 if (index >= rowCount)
106 return nullptr;
107
108 for (int k = index - 1; k >= 0; --k) {
109 auto* row = allRows[k].get();
110 if (row->hierarchicalLevel() == level - 1)
111 return row;
112 }
113
114 return nullptr;
115}

122AccessibilityTable* AccessibilityARIAGridRow::parentTable() const
123{
124 // The parent table might not be the direct ancestor of the row unfortunately. ARIA states that role="grid" should
125 // only have "row" elements, but if not, we still should handle it gracefully by finding the right table.
126 for (AccessibilityObject* parent = parentObject(); parent; parent = parent->parentObject()) {
127 // The parent table for an ARIA grid row should be an ARIA table.
128 // Unless the row is a native tr element.
129 if (is<AccessibilityTable>(*parent)) {
130 AccessibilityTable& tableParent = downcast<AccessibilityTable>(*parent);
131 if (tableParent.isExposable() && (tableParent.isAriaTable() || node()->hasTagName(HTMLNames::trTag)))
132 return &tableParent;
133 }
134 }
135
136 return nullptr;
137}

139AXCoreObject* AccessibilityARIAGridRow::headerObject()
140{
141 for (const auto& child : children()) {
142 if (child->ariaRoleAttribute() == AccessibilityRole::RowHeader)
143 return child.get();
144 }
145
146 return nullptr;
147}
