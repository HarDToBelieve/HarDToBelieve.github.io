56bool AccessibilityListBox::canSetSelectedChildren() const
57{
58 Node* selectNode = m_renderer->node();
59 if (!selectNode)
60 return false;
61
62 return !downcast<HTMLSelectElement>(*selectNode).isDisabledFormControl();
63}
64
65void AccessibilityListBox::addChildren()
66{
67 if (!m_renderer)
68 return;
69
70 Node* selectNode = m_renderer->node();
71 if (!selectNode)
72 return;
73
74 m_haveChildren = true;
75
76 for (const auto& listItem : downcast<HTMLSelectElement>(*selectNode).listItems()) {
77 AccessibilityObject* listOption = listBoxOptionAccessibilityObject(listItem);
78 if (listOption && !listOption->accessibilityIsIgnored())
79 m_children.append(listOption);
80 }
81}
82
83void AccessibilityListBox::setSelectedChildren(const AccessibilityChildrenVector& children)
84{
85 if (!canSetSelectedChildren())
86 return;
87
88 Node* selectNode = m_renderer->node();
89 if (!selectNode)
90 return;
91
92 // disable any selected options
93 for (const auto& child : m_children) {
94 auto& listBoxOption = downcast<AccessibilityListBoxOption>(*child);
95 if (listBoxOption.isSelected())
96 listBoxOption.setSelected(false);
97 }
98
99 for (const auto& obj : children) {
100 if (obj->roleValue() != AccessibilityRole::ListBoxOption)
101 continue;
102
103 downcast<AccessibilityListBoxOption>(*obj).setSelected(true);
104 }
105}

107void AccessibilityListBox::selectedChildren(AccessibilityChildrenVector& result)
108{
109 ASSERT(result.isEmpty());
110
111 if (!hasChildren())
112 addChildren();
113
114 for (const auto& child : m_children) {
115 if (downcast<AccessibilityListBoxOption>(*child).isSelected())
116 result.append(child.get());
117 }
118}
119
120void AccessibilityListBox::visibleChildren(AccessibilityChildrenVector& result)
121{
122 ASSERT(result.isEmpty());
123
124 if (!hasChildren())
125 addChildren();
126
127 unsigned length = m_children.size();
128 for (unsigned i = 0; i < length; i++) {
129 if (downcast<RenderListBox>(*m_renderer).listIndexIsVisible(i))
130 result.append(m_children[i]);
131 }
132}

140AXCoreObject* AccessibilityListBox::elementAccessibilityHitTest(const IntPoint& point) const
141{
142 // the internal HTMLSelectElement methods for returning a listbox option at a point
143 // ignore optgroup elements.
144 if (!m_renderer)
145 return nullptr;
146
147 Node* node = m_renderer->node();
148 if (!node)
149 return nullptr;
150
151 LayoutRect parentRect = boundingBoxRect();
152
153 AXCoreObject* listBoxOption = nullptr;
154 unsigned length = m_children.size();
155 for (unsigned i = 0; i < length; ++i) {
156 LayoutRect rect = downcast<RenderListBox>(*m_renderer).itemBoundingBoxRect(parentRect.location(), i);
157 // The cast to HTMLElement below is safe because the only other possible listItem type
158 // would be a WMLElement, but WML builds don't use accessibility features at all.
159 if (rect.contains(point)) {
160 listBoxOption = m_children[i].get();
161 break;
162 }
163 }
164
165 if (listBoxOption && !listBoxOption->accessibilityIsIgnored())
166 return listBoxOption;
167
168 return axObjectCache()->getOrCreate(renderer());
169}
