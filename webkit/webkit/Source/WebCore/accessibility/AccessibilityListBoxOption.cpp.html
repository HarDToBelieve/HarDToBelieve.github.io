54 return adoptRef(*new AccessibilityListBoxOption(element));

57bool AccessibilityListBoxOption::isEnabled() const
58{
59 if (is<HTMLOptGroupElement>(m_optionElement.get()))
60 return false;
61
62 if (equalLettersIgnoringASCIICase(getAttribute(aria_disabledAttr), "true"))
63 return false;
64
65 if (m_optionElement->hasAttributeWithoutSynchronization(disabledAttr))
66 return false;
67
68 return true;
69}
70
71bool AccessibilityListBoxOption::isSelected() const
72{
73 if (!is<HTMLOptionElement>(m_optionElement.get()))
74 return false;
75
76 return downcast<HTMLOptionElement>(*m_optionElement).selected();
77}

79bool AccessibilityListBoxOption::isSelectedOptionActive() const
80{
81 HTMLSelectElement* listBoxParentNode = listBoxOptionParentNode();
82 if (!listBoxParentNode)
83 return false;
84
85 return listBoxParentNode->activeSelectionEndListIndex() == listBoxOptionIndex();
86}
87
88LayoutRect AccessibilityListBoxOption::elementRect() const
89{
90 LayoutRect rect;
91 if (!m_optionElement)
92 return rect;
93
94 HTMLSelectElement* listBoxParentNode = listBoxOptionParentNode();
95 if (!listBoxParentNode)
96 return rect;
97
98 RenderElement* listBoxRenderer = listBoxParentNode->renderer();
99 if (!listBoxRenderer)
100 return rect;
101
102 LayoutRect parentRect = listBoxRenderer->document().axObjectCache()->getOrCreate(listBoxRenderer)->boundingBoxRect();
103 int index = listBoxOptionIndex();
104 if (index != -1)
105 rect = downcast<RenderListBox>(*listBoxRenderer).itemBoundingBoxRect(parentRect.location(), index);
106
107 return rect;
108}

110bool AccessibilityListBoxOption::computeAccessibilityIsIgnored() const
111{
112 if (!m_optionElement)
113 return true;
114
115 if (accessibilityIsIgnoredByDefault())
116 return true;
117
118 auto* parent = parentObject();
119 return parent ? parent->accessibilityIsIgnored() : true;
120}

122bool AccessibilityListBoxOption::canSetSelectedAttribute() const
123{
124 if (!is<HTMLOptionElement>(m_optionElement.get()))
125 return false;
126
127 if (m_optionElement->isDisabledFormControl())
128 return false;
129
130 HTMLSelectElement* selectElement = listBoxOptionParentNode();
131 if (selectElement && selectElement->isDisabledFormControl())
132 return false;
133
134 return true;
135}

137String AccessibilityListBoxOption::stringValue() const
138{
139 if (!m_optionElement)
140 return String();
141
142 const AtomString& ariaLabel = getAttribute(aria_labelAttr);
143 if (!ariaLabel.isNull())
144 return ariaLabel;
145
146 if (is<HTMLOptionElement>(*m_optionElement))
147 return downcast<HTMLOptionElement>(*m_optionElement).label();
148
149 if (is<HTMLOptGroupElement>(*m_optionElement))
150 return downcast<HTMLOptGroupElement>(*m_optionElement).groupLabelText();
151
152 return String();
153}

174void AccessibilityListBoxOption::setSelected(bool selected)
175{
176 HTMLSelectElement* selectElement = listBoxOptionParentNode();
177 if (!selectElement)
178 return;
179
180 if (!canSetSelectedAttribute())
181 return;
182
183 bool isOptionSelected = isSelected();
184 if ((isOptionSelected && selected) || (!isOptionSelected && !selected))
185 return;
186
187 // Convert from the entire list index to the option index.
188 int optionIndex = selectElement->listToOptionIndex(listBoxOptionIndex());
189 selectElement->accessKeySetSelectedIndex(optionIndex);
190}
191
192HTMLSelectElement* AccessibilityListBoxOption::listBoxOptionParentNode() const
193{
194 if (!m_optionElement)
195 return nullptr;
196
197 if (is<HTMLOptionElement>(*m_optionElement))
198 return downcast<HTMLOptionElement>(*m_optionElement).ownerSelectElement();
199
200 if (is<HTMLOptGroupElement>(*m_optionElement))
201 return downcast<HTMLOptGroupElement>(*m_optionElement).ownerSelectElement();
202
203 return nullptr;
204}
205
206int AccessibilityListBoxOption::listBoxOptionIndex() const
207{
208 if (!m_optionElement)
209 return -1;
210
211 HTMLSelectElement* selectElement = listBoxOptionParentNode();
212 if (!selectElement)
213 return -1;
214
215 const auto& listItems = selectElement->listItems();
216 unsigned length = listItems.size();
217 for (unsigned i = 0; i < length; i++)
218 if (listItems[i] == m_optionElement)
219 return i;
220
221 return -1;
222}
