28#include "AccessibilityRenderObject.h"

30namespace WebCore {
31
32class RenderMenuList;
33
34class AccessibilityMenuList final : public AccessibilityRenderObject {
35public:
36 static Ref<AccessibilityMenuList> create(RenderMenuList* renderer);
37
38 bool isCollapsed() const override;
39 bool press() override;
40
41 void didUpdateActiveOption(int optionIndex);
42
43private:
44 explicit AccessibilityMenuList(RenderMenuList*);
45
46 bool isMenuList() const override { return true; }
47 AccessibilityRole roleValue() const override { return AccessibilityRole::PopUpButton; }
48 bool canSetFocusAttribute() const override;
49
50 void addChildren() override;
51 void childrenChanged() override;
52};
53
54} // namespace WebCore
