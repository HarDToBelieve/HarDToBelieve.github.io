39AccessibilityMenuListOption::AccessibilityMenuListOption(HTMLOptionElement& element)
40 : m_element(makeWeakPtr(element))
41{
42}
43
44Ref<AccessibilityMenuListOption> AccessibilityMenuListOption::create(HTMLOptionElement& element)
45{
46 return adoptRef(*new AccessibilityMenuListOption(element));
47}
48
49Element* AccessibilityMenuListOption::actionElement() const
50{
51 return m_element.get();
52}
53
54Node* AccessibilityMenuListOption::node() const
55{
56 return m_element.get();
57}
58
59bool AccessibilityMenuListOption::isEnabled() const
60{
61 return m_element && !m_element->ownElementDisabled();
62}
63
64bool AccessibilityMenuListOption::isVisible() const
65{
66 if (!m_element)
67 return false;
68
69 // In a single-option select with the popup collapsed, only the selected item is considered visible.
70 auto parent = m_element->document().axObjectCache()->getOrCreate(m_element->ownerSelectElement());
71 return parent && (!parent->isOffScreen() || isSelected());
72}
73
74bool AccessibilityMenuListOption::isOffScreen() const
75{
76 // Invisible list options are considered to be offscreen.
77 return !isVisible();
78}
79
80bool AccessibilityMenuListOption::isSelected() const
81{
82 return m_element && m_element->selected();
83}
84
85void AccessibilityMenuListOption::setSelected(bool selected)
86{
87 if (!canSetSelectedAttribute())
88 return;
89
90 m_element->setSelected(selected);
91}
92
93String AccessibilityMenuListOption::nameForMSAA() const
94{
95 return stringValue();
96}
97
98bool AccessibilityMenuListOption::canSetSelectedAttribute() const
99{
100 return isEnabled();
101}
102
103bool AccessibilityMenuListOption::computeAccessibilityIsIgnored() const
104{
105 return accessibilityIsIgnoredByDefault();
106}
107
108LayoutRect AccessibilityMenuListOption::elementRect() const
109{
110 AccessibilityObject* parent = parentObject();
111 if (!parent)
112 return boundingBoxRect();
113 ASSERT(parent->isMenuListPopup());
114
115 AccessibilityObject* grandparent = parent->parentObject();
116 if (!grandparent)
117 return boundingBoxRect();
118 ASSERT(grandparent->isMenuList());
119
120 return grandparent->elementRect();
121}
122
123String AccessibilityMenuListOption::stringValue() const
124{
125 return m_element ? m_element->label() : String();
126}
