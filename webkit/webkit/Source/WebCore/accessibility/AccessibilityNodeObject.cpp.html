120void AccessibilityNodeObject::childrenChanged()
121{
122 // This method is meant as a quick way of marking a portion of the accessibility tree dirty.
123 if (!node() && !renderer())
124 return;
125
126 AXObjectCache* cache = axObjectCache();
127 if (!cache)
128 return;
129 cache->postNotification(this, document(), AXObjectCache::AXChildrenChanged);
130
131 // Should make the sub tree dirty so that everything below will be updated correctly.
132 this->setNeedsToUpdateSubtree();
133 bool shouldStopUpdatingParent = false;
134
135 // Go up the accessibility parent chain, but only if the element already exists. This method is
136 // called during render layouts, minimal work should be done.
137 // If AX elements are created now, they could interrogate the render tree while it's in a funky state.
138 // At the same time, process ARIA live region changes.
139 for (AccessibilityObject* parent = this; parent; parent = parent->parentObjectIfExists()) {
140 if (!shouldStopUpdatingParent)
141 parent->setNeedsToUpdateChildren();
142
143
144 // These notifications always need to be sent because screenreaders are reliant on them to perform.
145 // In other words, they need to be sent even when the screen reader has not accessed this live region since the last update.
146
147 // If this element supports ARIA live regions, then notify the AT of changes.
148 // Sometimes this function can be called many times within a short period of time, leading to posting too many AXLiveRegionChanged
149 // notifications. To fix this, we used a timer to make sure we only post one notification for the children changes within a pre-defined
150 // time interval.
151 if (parent->supportsLiveRegion())
152 cache->postLiveRegionChangeNotification(parent);
153
154 // If this element is an ARIA text control, notify the AT of changes.
155 if (parent->isNonNativeTextControl()) {
156 cache->postNotification(parent, parent->document(), AXObjectCache::AXValueChanged);
157
158 // Do not let the parent that's above the editable ancestor update its children
159 // since we already notify the AT of changes.
160 shouldStopUpdatingParent = true;
161 }
162 }
163}

165void AccessibilityNodeObject::updateAccessibilityRole()
166{
167 bool ignoredStatus = accessibilityIsIgnored();
168 m_role = determineAccessibilityRole();
169
170 // The AX hierarchy only needs to be updated if the ignored status of an element has changed.
171 if (ignoredStatus != accessibilityIsIgnored())
172 childrenChanged();
173}

253LayoutRect AccessibilityNodeObject::boundingBoxRect() const
254{
255 // AccessibilityNodeObjects have no mechanism yet to return a size or position.
256 // For now, let's return the position of the ancestor that does have a position,
257 // and make it the width of that parent, and about the height of a line of text, so that it's clear the object is a child of the parent.
258
259 LayoutRect boundingBox;
260
261 for (AccessibilityObject* positionProvider = parentObject(); positionProvider; positionProvider = positionProvider->parentObject()) {
262 if (positionProvider->isAccessibilityRenderObject()) {
263 LayoutRect parentRect = positionProvider->elementRect();
264 boundingBox.setSize(LayoutSize(parentRect.width(), LayoutUnit(std::min(10.0f, parentRect.height().toFloat()))));
265 boundingBox.setLocation(parentRect.location());
266 break;
267 }
268 }
269
270 return boundingBox;
271}

285AccessibilityRole AccessibilityNodeObject::determineAccessibilityRole()
286{
287 AXTRACE("AccessibilityNodeObject::determineAccessibilityRole");
288 if (!node())
289 return AccessibilityRole::Unknown;
290
291 if ((m_ariaRole = determineAriaRoleAttribute()) != AccessibilityRole::Unknown)
292 return m_ariaRole;
293
294 if (node()->isLink())
295 return AccessibilityRole::WebCoreLink;
296 if (node()->isTextNode())
297 return AccessibilityRole::StaticText;
298 if (node()->hasTagName(buttonTag))
299 return buttonRoleType();
300 if (is<HTMLInputElement>(*node())) {
301 HTMLInputElement& input = downcast<HTMLInputElement>(*node());
302 if (input.isCheckbox())
303 return AccessibilityRole::CheckBox;
304 if (input.isRadioButton())
305 return AccessibilityRole::RadioButton;
306 if (input.isTextButton())
307 return buttonRoleType();
308 if (input.isRangeControl())
309 return AccessibilityRole::Slider;
310 if (input.isInputTypeHidden())
311 return AccessibilityRole::Ignored;
312 if (input.isSearchField())
313 return AccessibilityRole::SearchField;
314#if ENABLE(INPUT_TYPE_COLOR)
315 if (input.isColorControl())
316 return AccessibilityRole::ColorWell;
317#endif
318 return AccessibilityRole::TextField;
319 }
320 if (node()->hasTagName(selectTag)) {
321 HTMLSelectElement& selectElement = downcast<HTMLSelectElement>(*node());
322 return selectElement.multiple() ? AccessibilityRole::ListBox : AccessibilityRole::PopUpButton;
323 }
324 if (is<HTMLTextAreaElement>(*node()))
325 return AccessibilityRole::TextArea;
326 if (headingLevel())
327 return AccessibilityRole::Heading;
328 if (node()->hasTagName(blockquoteTag))
329 return AccessibilityRole::Blockquote;
330 if (node()->hasTagName(divTag))
331 return AccessibilityRole::Div;
332 if (node()->hasTagName(pTag))
333 return AccessibilityRole::Paragraph;
334 if (is<HTMLLabelElement>(*node()))
335 return AccessibilityRole::Label;
336 if (is<Element>(*node()) && downcast<Element>(*node()).isFocusable())
337 return AccessibilityRole::Group;
338
339 return AccessibilityRole::Unknown;
340}

342void AccessibilityNodeObject::addChildren()
343{
344 // If the need to add more children in addition to existing children arises,
345 // childrenChanged should have been called, leaving the object with no children.
346 ASSERT(!m_haveChildren);
347
348 if (!m_node)
349 return;
350
351 m_haveChildren = true;
352
353 // The only time we add children from the DOM tree to a node with a renderer is when it's a canvas.
354 if (renderer() && !m_node->hasTagName(canvasTag))
355 return;
356
357 auto objectCache = axObjectCache();
358 if (!objectCache)
359 return;
360
361 for (Node* child = m_node->firstChild(); child; child = child->nextSibling())
362 addChild(objectCache->getOrCreate(child));
363
364 m_subtreeDirty = false;
365}

367bool AccessibilityNodeObject::canHaveChildren() const
368{
369 // If this is an AccessibilityRenderObject, then it's okay if this object
370 // doesn't have a node - there are some renderers that don't have associated
371 // nodes, like scroll areas and css-generated text.
372 if (!node() && !isAccessibilityRenderObject())
373 return false;
374
375 // When <noscript> is not being used (its renderer() == 0), ignore its children.
376 if (node() && !renderer() && node()->hasTagName(noscriptTag))
377 return false;
378
379 // Elements that should not have children
380 switch (roleValue()) {
381 case AccessibilityRole::Button:
382 case AccessibilityRole::PopUpButton:
383 case AccessibilityRole::CheckBox:
384 case AccessibilityRole::RadioButton:
385 case AccessibilityRole::Tab:
386 case AccessibilityRole::ToggleButton:
387 case AccessibilityRole::StaticText:
388 case AccessibilityRole::ListBoxOption:
389 case AccessibilityRole::ScrollBar:
390 case AccessibilityRole::ProgressIndicator:
391 case AccessibilityRole::Switch:
392 case AccessibilityRole::MenuItemCheckbox:
393 case AccessibilityRole::MenuItemRadio:
394 case AccessibilityRole::Splitter:
395 case AccessibilityRole::Meter:
396 return false;
397 case AccessibilityRole::DocumentMath:
398#if ENABLE(MATHML)
399 return node()->isMathMLElement();
400#endif
401 return false;
402 default:
403 return true;
404 }
405}

407bool AccessibilityNodeObject::computeAccessibilityIsIgnored() const
408{
409#ifndef NDEBUG
410 // Double-check that an AccessibilityObject is never accessed before
411 // it's been initialized.
412 ASSERT(m_initialized);
413#endif
414
415 // Handle non-rendered text that is exposed through aria-hidden=false.
416 if (m_node && m_node->isTextNode() && !renderer()) {
417 // Fallback content in iframe nodes should be ignored.
418 if (m_node->parentNode() && m_node->parentNode()->hasTagName(iframeTag) && m_node->parentNode()->renderer())
419 return true;
420
421 // Whitespace only text elements should be ignored when they have no renderer.
422 String string = stringValue().stripWhiteSpace().simplifyWhiteSpace();
423 if (!string.length())
424 return true;
425 }
426
427 AccessibilityObjectInclusion decision = defaultObjectInclusion();
428 if (decision == AccessibilityObjectInclusion::IncludeObject)
429 return false;
430 if (decision == AccessibilityObjectInclusion::IgnoreObject)
431 return true;
432 // If this element is within a parent that cannot have children, it should not be exposed.
433 if (isDescendantOfBarrenParent())
434 return true;
435
436 if (roleValue() == AccessibilityRole::Ignored)
437 return true;
438
439 return m_role == AccessibilityRole::Unknown;
440}

442bool AccessibilityNodeObject::canvasHasFallbackContent() const
443{
444 Node* node = this->node();
445 if (!is<HTMLCanvasElement>(node))
446 return false;
447 HTMLCanvasElement& canvasElement = downcast<HTMLCanvasElement>(*node);
448 // If it has any children that are elements, we'll assume it might be fallback
449 // content. If it has no children or its only children are not elements
450 // (e.g. just text nodes), it doesn't have fallback content.
451 return childrenOfType<Element>(canvasElement).first();
452}

476bool AccessibilityNodeObject::isSearchField() const
477{
478 Node* node = this->node();
479 if (!node)
480 return false;
481
482 if (roleValue() == AccessibilityRole::SearchField)
483 return true;
484
485 if (!is<HTMLInputElement>(*node))
486 return false;
487
488 auto& inputElement = downcast<HTMLInputElement>(*node);
489
490 // Some websites don't label their search fields as such. However, they will
491 // use the word "search" in either the form or input type. This won't catch every case,
492 // but it will catch google.com for example.
493
494 // Check the node name of the input type, sometimes it's "search".
495 const AtomString& nameAttribute = getAttribute(nameAttr);
496 if (nameAttribute.containsIgnoringASCIICase("search"))
497 return true;
498
499 // Check the form action and the name, which will sometimes be "search".
500 auto* form = inputElement.form();
501 if (form && (form->name().containsIgnoringASCIICase("search") || form->action().containsIgnoringASCIICase("search")))
502 return true;
503
504 return false;
505}

631bool AccessibilityNodeObject::isEnabled() const
632{
633 // ARIA says that the disabled status applies to the current element and all descendant elements.
634 for (AccessibilityObject* object = const_cast<AccessibilityNodeObject*>(this); object; object = object->parentObject()) {
635 const AtomString& disabledStatus = object->getAttribute(aria_disabledAttr);
636 if (equalLettersIgnoringASCIICase(disabledStatus, "true"))
637 return false;
638 if (equalLettersIgnoringASCIICase(disabledStatus, "false"))
639 break;
640 }
641
642 if (roleValue() == AccessibilityRole::HorizontalRule)
643 return false;
644
645 Node* node = this->node();
646 if (!is<Element>(node))
647 return true;
648
649 return !downcast<Element>(*node).isDisabledFormControl();
650}

675bool AccessibilityNodeObject::isChecked() const
676{
677 Node* node = this->node();
678 if (!node)
679 return false;
680
681 // First test for native checkedness semantics
682 if (is<HTMLInputElement>(*node))
683 return downcast<HTMLInputElement>(*node).shouldAppearChecked();
684
685 // Else, if this is an ARIA checkbox or radio, respect the aria-checked attribute
686 bool validRole = false;
687 switch (ariaRoleAttribute()) {
688 case AccessibilityRole::RadioButton:
689 case AccessibilityRole::CheckBox:
690 case AccessibilityRole::MenuItem:
691 case AccessibilityRole::MenuItemCheckbox:
692 case AccessibilityRole::MenuItemRadio:
693 case AccessibilityRole::Switch:
694 case AccessibilityRole::TreeItem:
695 validRole = true;
696 break;
697 default:
698 break;
699 }
700
701 if (validRole && equalLettersIgnoringASCIICase(getAttribute(aria_checkedAttr), "true"))
702 return true;
703
704 return false;
705}

769int AccessibilityNodeObject::headingLevel() const
770{
771 // headings can be in block flow and non-block flow
772 Node* node = this->node();
773 if (!node)
774 return false;
775
776 if (isHeading()) {
777 if (auto level = getIntegralAttribute(aria_levelAttr); level > 0)
778 return level;
779 }
780
781 if (node->hasTagName(h1Tag))
782 return 1;
783
784 if (node->hasTagName(h2Tag))
785 return 2;
786
787 if (node->hasTagName(h3Tag))
788 return 3;
789
790 if (node->hasTagName(h4Tag))
791 return 4;
792
793 if (node->hasTagName(h5Tag))
794 return 5;
795
796 if (node->hasTagName(h6Tag))
797 return 6;
798
799 // The implicit value of aria-level is 2 for the heading role.
800 // https://www.w3.org/TR/wai-aria-1.1/#heading
801 if (ariaRoleAttribute() == AccessibilityRole::Heading)
802 return 2;
803
804 return 0;
805}

815float AccessibilityNodeObject::valueForRange() const
816{
817 if (is<HTMLInputElement>(node())) {
818 HTMLInputElement& input = downcast<HTMLInputElement>(*node());
819 if (input.isRangeControl())
820 return input.valueAsNumber();
821 }
822
823 if (!isRangeControl())
824 return 0.0f;
825

828 auto& value = getAttribute(aria_valuenowAttr);
829 if (!value.isEmpty())

985{

1099static bool dispatchSimulatedKeyboardUpDownEvent(AccessibilityObject* object, const KeyboardEvent::Init& keyInit)
1100{
1101 // In case the keyboard event causes this element to be removed.
1102 Ref<AccessibilityObject> protectedObject(*object);
1103
1104 bool handled = false;
1105 if (auto* node = object->node()) {
1106 auto event = KeyboardEvent::create(eventNames().keydownEvent, keyInit, Event::IsTrusted::Yes);
1107 node->dispatchEvent(event);
1108 handled |= event->defaultHandled();
1109 }
1110
1111 // Ensure node is still valid and wasn't removed after the keydown.
1112 if (auto* node = object->node()) {
1113 auto event = KeyboardEvent::create(eventNames().keyupEvent, keyInit, Event::IsTrusted::Yes);
1114 node->dispatchEvent(event);
1115 handled |= event->defaultHandled();
1116 }
1117 return handled;
1118}

1142// Fire a keyboard event if we were not able to set this value natively.
1143bool AccessibilityNodeObject::postKeyboardKeysForValueChange(bool increase)
1144{
1145 auto keyInit = KeyboardEvent::Init();
1146 bool vertical = orientation() == AccessibilityOrientation::Vertical;
1147 bool isLTR = page()->userInterfaceLayoutDirection() == UserInterfaceLayoutDirection::LTR;
1148
1149 // The goal is to mimic existing keyboard dispatch completely, so that this is indistinguishable from a real key press.
1150 typedef enum { left = 37, up = 38, right = 39, down = 40 } keyCode;
1151 keyInit.key = increase ? (vertical ? "ArrowUp"_s : (isLTR ? "ArrowRight"_s : "ArrowLeft"_s)) : (vertical ? "ArrowDown"_s : (isLTR ? "ArrowLeft"_s : "ArrowRight"_s));
1152 keyInit.keyCode = increase ? (vertical ? keyCode::up : (isLTR ? keyCode::right : keyCode::left)) : (vertical ? keyCode::down : (isLTR ? keyCode::left : keyCode::right));
1153 keyInit.keyIdentifier = increase ? (vertical ? "Up"_s : (isLTR ? "Right"_s : "Left"_s)) : (vertical ? "Down"_s : (isLTR ? "Left"_s : "Right"_s));
1154
1155 InitializeLegacyKeyInitProperties(keyInit, *this);
1156
1157 return dispatchSimulatedKeyboardUpDownEvent(this, keyInit);
1158}

1180void AccessibilityNodeObject::changeValueByPercent(float percentChange)
1181{
1182 float range = maxValueForRange() - minValueForRange();
1183 float step = range * (percentChange / 100);
1184 float value = valueForRange();

1194bool AccessibilityNodeObject::isGenericFocusableElement() const
1195{
1196 if (!canSetFocusAttribute())
1197 return false;
1198
1199 // If it's a control, it's not generic.
1200 if (isControl())
1201 return false;
1202
1203 AccessibilityRole role = roleValue();
1204 if (role == AccessibilityRole::Video || role == AccessibilityRole::Audio)
1205 return false;
1206
1207 // If it has an aria role, it's not generic.
1208 if (m_ariaRole != AccessibilityRole::Unknown)
1209 return false;
1210
1211 // If the content editable attribute is set on this element, that's the reason
1212 // it's focusable, and existing logic should handle this case already - so it's not a
1213 // generic focusable element.
1214
1215 if (hasContentEditableAttributeSet())
1216 return false;
1217
1218 // The web area and body element are both focusable, but existing logic handles these
1219 // cases already, so we don't need to include them here.
1220 if (role == AccessibilityRole::WebArea)
1221 return false;
1222 if (node() && node()->hasTagName(bodyTag))
1223 return false;
1224
1225 // An SVG root is focusable by default, but it's probably not interactive, so don't
1226 // include it. It can still be made accessible by giving it an ARIA role.
1227 if (role == AccessibilityRole::SVGRoot)
1228 return false;
1229
1230 return true;
1231}

1388void AccessibilityNodeObject::alternativeText(Vector<AccessibilityText>& textOrder) const
1389{
1390 if (isWebArea()) {
1391 String webAreaText = alternativeTextForWebArea();
1392 if (!webAreaText.isEmpty())
1393 textOrder.append(AccessibilityText(webAreaText, AccessibilityTextSource::Alternative));
1394 return;
1395 }
1396
1397 ariaLabeledByText(textOrder);
1398
1399 const AtomString& ariaLabel = getAttribute(aria_labelAttr);
1400 if (!ariaLabel.isEmpty())
1401 textOrder.append(AccessibilityText(ariaLabel, AccessibilityTextSource::Alternative));
1402
1403 if (usesAltTagForTextComputation()) {
1404 if (is<RenderImage>(renderer())) {
1405 String renderAltText = downcast<RenderImage>(*renderer()).altText();
1406
1407 // RenderImage will return title as a fallback from altText, but we don't want title here because we consider that in helpText.
1408 if (!renderAltText.isEmpty() && renderAltText != getAttribute(titleAttr)) {
1409 textOrder.append(AccessibilityText(renderAltText, AccessibilityTextSource::Alternative));
1410 return;
1411 }
1412 }
1413 // Images should use alt as long as the attribute is present, even if empty.
1414 // Otherwise, it should fallback to other methods, like the title attribute.
1415 const AtomString& alt = getAttribute(altAttr);
1416 if (!alt.isEmpty())
1417 textOrder.append(AccessibilityText(alt, AccessibilityTextSource::Alternative));
1418 }
1419
1420 Node* node = this->node();
1421 if (!node)
1422 return;
1423
1424 auto objectCache = axObjectCache();
1425 // The fieldset element derives its alternative text from the first associated legend element if one is available.
1426 if (objectCache && is<HTMLFieldSetElement>(*node)) {
1427 AccessibilityObject* object = objectCache->getOrCreate(downcast<HTMLFieldSetElement>(*node).legend());
1428 if (object && !object->isHidden())
1429 textOrder.append(AccessibilityText(accessibleNameForNode(object->node()), AccessibilityTextSource::Alternative));
1430 }
1431
1432 // The figure element derives its alternative text from the first associated figcaption element if one is available.
1433 if (isFigureElement()) {
1434 AccessibilityObject* captionForFigure = this->captionForFigure();
1435 if (captionForFigure && !captionForFigure->isHidden())
1436 textOrder.append(AccessibilityText(accessibleNameForNode(captionForFigure->node()), AccessibilityTextSource::Alternative));
1437 }
1438
1439 // Tree items missing a label are labeled by all child elements.
1440 if (isTreeItem() && ariaLabel.isEmpty() && ariaLabeledByAttribute().isEmpty())
1441 textOrder.append(AccessibilityText(accessibleNameForNode(node), AccessibilityTextSource::Alternative));
1442
1443#if ENABLE(MATHML)
1444 if (node->isMathMLElement())
1445 textOrder.append(AccessibilityText(getAttribute(MathMLNames::alttextAttr), AccessibilityTextSource::Alternative));
1446#endif

1449void AccessibilityNodeObject::visibleText(Vector<AccessibilityText>& textOrder) const
1450{
1451 Node* node = this->node();
1452 if (!node)
1453 return;
1454
1455 bool isInputTag = is<HTMLInputElement>(*node);
1456 if (isInputTag) {
1457 HTMLInputElement& input = downcast<HTMLInputElement>(*node);
1458 if (input.isTextButton()) {
1459 textOrder.append(AccessibilityText(input.valueWithDefault(), AccessibilityTextSource::Visible));
1460 return;
1461 }
1462 }
1463
1464 // If this node isn't rendered, there's no inner text we can extract from a select element.
1465 if (!isAccessibilityRenderObject() && node->hasTagName(selectTag))
1466 return;
1467
1468 bool useTextUnderElement = false;
1469
1470 switch (roleValue()) {
1471 case AccessibilityRole::PopUpButton:
1472 // Native popup buttons should not use their button children's text as a title. That value is retrieved through stringValue().
1473 if (node->hasTagName(selectTag))
1474 break;
1475 FALLTHROUGH;
1476 case AccessibilityRole::Button:
1477 case AccessibilityRole::ToggleButton:
1478 case AccessibilityRole::CheckBox:
1479 case AccessibilityRole::ListBoxOption:
1480 // MacOS does not expect native <li> elements to expose label information, it only expects leaf node elements to do that.
1481#if !PLATFORM(COCOA)
1482 case AccessibilityRole::ListItem:
1483#endif
1484 case AccessibilityRole::MenuButton:
1485 case AccessibilityRole::MenuItem:
1486 case AccessibilityRole::MenuItemCheckbox:
1487 case AccessibilityRole::MenuItemRadio:
1488 case AccessibilityRole::RadioButton:
1489 case AccessibilityRole::Switch:
1490 case AccessibilityRole::Tab:
1491 useTextUnderElement = true;
1492 break;
1493 default:
1494 break;
1495 }
1496
1497 // If it's focusable but it's not content editable or a known control type, then it will appear to
1498 // the user as a single atomic object, so we should use its text as the default title.
1499 if (isHeading() || isLink())
1500 useTextUnderElement = true;
1501
1502 if (isOutput())
1503 useTextUnderElement = true;
1504
1505 if (useTextUnderElement) {
1506 AccessibilityTextUnderElementMode mode;
1507
1508 // Headings often include links as direct children. Those links need to be included in text under element.
1509 if (isHeading())
1510 mode.includeFocusableContent = true;
1511
1512 String text = textUnderElement(mode);
1513 if (!text.isEmpty())
1514 textOrder.append(AccessibilityText(text, AccessibilityTextSource::Children));
1515 }

1544 // can expose units through this attribute. Therefore, if the element is a meter, change its source

1549 textOrder.append(AccessibilityText(title, AccessibilityTextSource::TitleTag));
1550 else
1551 textOrder.append(AccessibilityText(title, AccessibilityTextSource::Help));
1552 }

1628String AccessibilityNodeObject::accessibilityDescription() const
1629{
1630 // Static text should not have a description, it should only have a stringValue.
1631 if (roleValue() == AccessibilityRole::StaticText)
1632 return String();
1633
1634 String ariaDescription = ariaAccessibilityDescription();
1635 if (!ariaDescription.isEmpty())
1636 return ariaDescription;
1637
1638 if (usesAltTagForTextComputation()) {
1639 // Images should use alt as long as the attribute is present, even if empty.
1640 // Otherwise, it should fallback to other methods, like the title attribute.
1641 const AtomString& alt = getAttribute(altAttr);
1642 if (!alt.isNull())
1643 return alt;
1644 }
1645
1646#if ENABLE(MATHML)
1647 if (is<MathMLElement>(m_node))
1648 return getAttribute(MathMLNames::alttextAttr);
1649#endif
1650
1651 // An element's descriptive text is comprised of title() (what's visible on the screen) and accessibilityDescription() (other descriptive text).
1652 // Both are used to generate what a screen reader speaks.
1653 // If this point is reached (i.e. there's no accessibilityDescription) and there's no title(), we should fallback to using the title attribute.
1654 // The title attribute is normally used as help text (because it is a tooltip), but if there is nothing else available, this should be used (according to ARIA).
1655 // https://bugs.webkit.org/show_bug.cgi?id=170475: An exception is when the element is semantically unimportant. In those cases, title text should remain as help text.
1656 if (!roleIgnoresTitle()) {
1657 // title() can be an expensive operation because it can invoke textUnderElement for all descendants. Thus call it last.
1658 auto titleAttribute = getAttribute(titleAttr);
1659 if (!titleAttribute.isEmpty() && title().isEmpty())
1660 return titleAttribute;
1661 }
1662
1663 return String();
1664}

1725unsigned AccessibilityNodeObject::hierarchicalLevel() const
1726{
1727 Node* node = this->node();
1728 if (!is<Element>(node))
1729 return 0;
1730
1731 auto& element = downcast<Element>(*node);
1732 if (!element.attributeWithoutSynchronization(aria_levelAttr).isEmpty())
1733 return element.getIntegralAttribute(aria_levelAttr);
1734
1735 // Only tree item will calculate its level through the DOM currently.
1736 if (roleValue() != AccessibilityRole::TreeItem)
1737 return 0;
1738

1740 // We measure tree hierarchy by the number of groups that the item is within.
1741 unsigned level = 1;
1742 for (AccessibilityObject* parent = parentObject(); parent; parent = parent->parentObject()) {
1743 AccessibilityRole parentRole = parent->ariaRoleAttribute();
1744 if (parentRole == AccessibilityRole::ApplicationGroup)
1745 level++;
1746 else if (parentRole == AccessibilityRole::Tree)
1747 break;
1748 }
1749
1750 return level;

1762// When building the textUnderElement for an object, determine whether or not
1763// we should include the inner text of this given descendant object or skip it.
1764static bool shouldUseAccessibilityObjectInnerText(AccessibilityObject* obj, AccessibilityTextUnderElementMode mode)
1765{
1766 // Do not use any heuristic if we are explicitly asking to include all the children.
1767 if (mode.childrenInclusion == AccessibilityTextUnderElementMode::TextUnderElementModeIncludeAllChildren)
1768 return true;
1769
1770 // Consider this hypothetical example:
1771 // <div tabindex=0>
1772 // <h2>
1773 // Table of contents
1774 // </h2>
1775 // <a href="#start">Jump to start of book</a>
1776 // <ul>
1777 // <li><a href="#1">Chapter 1</a></li>
1778 // <li><a href="#1">Chapter 2</a></li>
1779 // </ul>
1780 // </div>
1781 //
1782 // The goal is to return a reasonable title for the outer container div, because
1783 // it's focusable - but without making its title be the full inner text, which is
1784 // quite long. As a heuristic, skip links, controls, and elements that are usually
1785 // containers with lots of children.
1786
1787 // ARIA states that certain elements are not allowed to expose their children content for name calculation.
1788 if (mode.childrenInclusion == AccessibilityTextUnderElementMode::TextUnderElementModeIncludeNameFromContentsChildren
1789 && !obj->accessibleNameDerivesFromContent())
1790 return false;
1791
1792 if (equalLettersIgnoringASCIICase(obj->getAttribute(aria_hiddenAttr), "true"))
1793 return false;
1794
1795 // If something doesn't expose any children, then we can always take the inner text content.
1796 // This is what we want when someone puts an <a> inside a <button> for example.
1797 if (obj->isDescendantOfBarrenParent())
1798 return true;
1799
1800 // Skip focusable children, so we don't include the text of links and controls.
1801 if (obj->canSetFocusAttribute() && !mode.includeFocusableContent)
1802 return false;
1803
1804 // Skip big container elements like lists, tables, etc.
1805 if (is<AccessibilityList>(*obj))
1806 return false;
1807
1808 if (is<AccessibilityTable>(*obj) && downcast<AccessibilityTable>(*obj).isExposable())
1809 return false;
1810
1811 if (obj->isTree() || obj->isCanvas())
1812 return false;
1813
1814 return true;
1815}

1833String AccessibilityNodeObject::textUnderElement(AccessibilityTextUnderElementMode mode) const
1834{
1835 Node* node = this->node();
1836 if (is<Text>(node))
1837 return downcast<Text>(*node).wholeText();
1838
1839 bool isAriaVisible = Accessibility::findAncestor<AccessibilityObject>(*this, true, [] (const AccessibilityObject& object) {
1840 return equalLettersIgnoringASCIICase(object.getAttribute(aria_hiddenAttr), "false");
1841 }) != nullptr;
1842
1843 // The Accname specification states that if the current node is hidden, and not directly
1844 // referenced by aria-labelledby or aria-describedby, and is not a host language text
1845 // alternative, the empty string should be returned.
1846 if (isDOMHidden() && !isAriaVisible && !is<HTMLLabelElement>(node) && (node && !ancestorsOfType<HTMLCanvasElement>(*node).first())) {
1847 AccessibilityObject::AccessibilityChildrenVector labelFor;
1848 AccessibilityObject::AccessibilityChildrenVector descriptionFor;
1849 ariaLabelledByReferencingElements(labelFor);
1850 ariaDescribedByReferencingElements(descriptionFor);
1851 if (!labelFor.size() && !descriptionFor.size())
1852 return String();
1853 }
1854
1855 StringBuilder builder;
1856 for (AccessibilityObject* child = firstChild(); child; child = child->nextSibling()) {
1857 if (mode.ignoredChildNode && child->node() == mode.ignoredChildNode)
1858 continue;
1859
1860 bool shouldDeriveNameFromAuthor = (mode.childrenInclusion == AccessibilityTextUnderElementMode::TextUnderElementModeIncludeNameFromContentsChildren && !child->accessibleNameDerivesFromContent());
1861 if (shouldDeriveNameFromAuthor) {
1862 appendNameToStringBuilder(builder, accessibleNameForNode(child->node()));
1863 continue;
1864 }
1865
1866 if (!shouldUseAccessibilityObjectInnerText(child, mode))
1867 continue;
1868
1869 if (is<AccessibilityNodeObject>(*child)) {
1870 // We should ignore the child if it's labeled by this node.
1871 // This could happen when this node labels multiple child nodes and we didn't
1872 // skip in the above ignoredChildNode check.
1873 Vector<Element*> labeledByElements;
1874 downcast<AccessibilityNodeObject>(*child).ariaLabeledByElements(labeledByElements);
1875 if (labeledByElements.contains(node))
1876 continue;
1877
1878 Vector<AccessibilityText> textOrder;
1879 downcast<AccessibilityNodeObject>(*child).alternativeText(textOrder);
1880 if (textOrder.size() > 0 && textOrder[0].text.length()) {
1881 appendNameToStringBuilder(builder, textOrder[0].text);
1882 continue;
1883 }
1884 }
1885
1886 String childText = child->textUnderElement(mode);
1887 if (childText.length())
1888 appendNameToStringBuilder(builder, childText);
1889 }
1890
1891 return builder.toString().stripWhiteSpace().simplifyWhiteSpace(isHTMLSpaceButNotLineBreak);
1892}

1894String AccessibilityNodeObject::title() const
1895{
1896 Node* node = this->node();
1897 if (!node)
1898 return String();
1899
1900 bool isInputTag = is<HTMLInputElement>(*node);
1901 if (isInputTag) {
1902 HTMLInputElement& input = downcast<HTMLInputElement>(*node);
1903 if (input.isTextButton())
1904 return input.valueWithDefault();
1905 }
1906
1907 if (isLabelable()) {
1908 HTMLLabelElement* label = labelForElement(downcast<Element>(node));
1909 // Use the label text as the title if 1) the title element is NOT an exposed element and 2) there's no ARIA override.
1910 if (label && !exposesTitleUIElement() && !ariaAccessibilityDescription().length())
1911 return textForLabelElement(label);
1912 }
1913
1914 // If this node isn't rendered, there's no inner text we can extract from a select element.
1915 if (!isAccessibilityRenderObject() && node->hasTagName(selectTag))
1916 return String();
1917
1918 switch (roleValue()) {
1919 case AccessibilityRole::PopUpButton:
1920 // Native popup buttons should not use their button children's text as a title. That value is retrieved through stringValue().
1921 if (node->hasTagName(selectTag))
1922 return String();
1923 FALLTHROUGH;
1924 case AccessibilityRole::Button:
1925 case AccessibilityRole::ToggleButton:
1926 case AccessibilityRole::CheckBox:
1927 case AccessibilityRole::ListBoxOption:
1928 case AccessibilityRole::ListItem:
1929 case AccessibilityRole::MenuButton:
1930 case AccessibilityRole::MenuItem:
1931 case AccessibilityRole::MenuItemCheckbox:
1932 case AccessibilityRole::MenuItemRadio:
1933 case AccessibilityRole::RadioButton:
1934 case AccessibilityRole::Switch:
1935 case AccessibilityRole::Tab:
1936 return textUnderElement();
1937 // SVGRoots should not use the text under itself as a title. That could include the text of objects like <text>.
1938 case AccessibilityRole::SVGRoot:
1939 return String();
1940 default:
1941 break;
1942 }
1943
1944 if (isLink())
1945 return textUnderElement();
1946 if (isHeading())
1947 return textUnderElement(AccessibilityTextUnderElementMode(AccessibilityTextUnderElementMode::TextUnderElementModeSkipIgnoredChildren, true));
1948
1949 return String();
1950}

2031// This function implements the ARIA accessible name as described by the Mozilla
2032// ARIA Implementer's Guide.
2033static String accessibleNameForNode(Node* node, Node* labelledbyNode)
2034{
2035 ASSERT(node);
2036 if (!is<Element>(node))
2037 return String();
2038
2039 Element& element = downcast<Element>(*node);
2040 const AtomString& ariaLabel = element.attributeWithoutSynchronization(aria_labelAttr);
2041 if (!ariaLabel.isEmpty())
2042 return ariaLabel;
2043
2044 const AtomString& alt = element.attributeWithoutSynchronization(altAttr);
2045 if (!alt.isEmpty())
2046 return alt;
2047
2048 // If the node can be turned into an AX object, we can use standard name computation rules.
2049 // If however, the node cannot (because there's no renderer e.g.) fallback to using the basic text underneath.
2050 auto axObject = element.document().axObjectCache()->getOrCreate(&element);
2051 if (axObject) {
2052 String valueDescription = axObject->valueDescription();
2053 if (!valueDescription.isEmpty())
2054 return valueDescription;
2055
2056 // The Accname specification states that if the name is being calculated for a combobox
2057 // or listbox inside a labeling element, return the text alternative of the chosen option.
2058 AccessibilityObject::AccessibilityChildrenVector children;
2059 if (axObject->isListBox())
2060 axObject->selectedChildren(children);
2061 else if (axObject->isComboBox()) {
2062 for (const auto& child : axObject->children()) {
2063 if (child->isListBox()) {
2064 child->selectedChildren(children);
2065 break;
2066 }
2067 }
2068 }
2069
2070 StringBuilder builder;
2071 String childText;
2072 for (const auto& child : children)
2073 appendNameToStringBuilder(builder, accessibleNameForNode(child->node()));
2074
2075 childText = builder.toString();
2076 if (!childText.isEmpty())
2077 return childText;
2078 }
2079
2080 if (is<HTMLInputElement>(element))
2081 return downcast<HTMLInputElement>(element).value();
2082 if (is<HTMLOptionElement>(element))
2083 return downcast<HTMLOptionElement>(element).value();
2084
2085 String text;
2086 if (axObject) {
2087 if (axObject->accessibleNameDerivesFromContent())
2088 text = axObject->textUnderElement(AccessibilityTextUnderElementMode(AccessibilityTextUnderElementMode::TextUnderElementModeIncludeNameFromContentsChildren, true, labelledbyNode));
2089 } else
2090 text = element.innerText().simplifyWhiteSpace();
2091
2092 if (!text.isEmpty())
2093 return text;
2094
2095 const AtomString& title = element.attributeWithoutSynchronization(titleAttr);
2096 if (!title.isEmpty())
2097 return title;
2098
2099 return String();
2100}
