63String AccessibilityProgressIndicator::valueDescription() const
64{
65 // If the author has explicitly provided a value through aria-valuetext, use it.
66 String description = AccessibilityRenderObject::valueDescription();
67 if (!description.isEmpty())
68 return description;
69
70 if (!m_renderer->isMeter())
71 return description;
72
73 HTMLMeterElement* meter = meterElement();
74 if (!meter)
75 return description;
76
77 // The HTML spec encourages authors to include a textual representation of the meter's state in
78 // the element's contents. We'll fall back on that if there is not a more accessible alternative.
79 AccessibilityObject* axMeter = axObjectCache()->getOrCreate(meter);
80 if (is<AccessibilityNodeObject>(axMeter))
81 description = downcast<AccessibilityNodeObject>(axMeter)->accessibilityDescriptionForChildren();
82
83 if (description.isEmpty())
84 description = meter->textContent();
85
86 String gaugeRegionValue = gaugeRegionValueDescription();
87 if (!gaugeRegionValue.isEmpty())
88 description = description.isEmpty() ? gaugeRegionValue : description + ", " + gaugeRegionValue;
89
90 return description;
91}

93float AccessibilityProgressIndicator::valueForRange() const
94{
95 if (!m_renderer)
96 return 0.0;
97
98 if (m_renderer->isProgress()) {
99 HTMLProgressElement* progress = progressElement();
100 if (progress && progress->position() >= 0)
101 return narrowPrecisionToFloat(progress->value());
102 }
103
104 if (m_renderer->isMeter()) {
105 if (HTMLMeterElement* meter = meterElement())
106 return narrowPrecisionToFloat(meter->value());
107 }
108
109 // Indeterminate progress bar should return 0.
110 return 0.0;
111}

170String AccessibilityProgressIndicator::gaugeRegionValueDescription() const
171{
172#if PLATFORM(COCOA)
173 if (!m_renderer || !m_renderer->isMeter())
174 return String();
175
176 // Only expose this when the author has explicitly specified the following attributes.
177 if (!hasAttribute(lowAttr) && !hasAttribute(highAttr) && !hasAttribute(optimumAttr))
178 return String();
179
180 if (HTMLMeterElement* element = meterElement()) {
181 switch (element->gaugeRegion()) {
182 case HTMLMeterElement::GaugeRegionOptimum:
183 return AXMeterGaugeRegionOptimumText();
184 case HTMLMeterElement::GaugeRegionSuboptimal:
185 return AXMeterGaugeRegionSuboptimalText();
186 case HTMLMeterElement::GaugeRegionEvenLessGood:
187 return AXMeterGaugeRegionLessGoodText();
188 default:
189 break;
190 }
191 }
192#endif
193 return String();
194}
195
196Element* AccessibilityProgressIndicator::element() const
197{
198 if (!m_renderer)
199 return nullptr;
200
201 if (m_renderer->isProgress())
202 return progressElement();
203
204 if (m_renderer->isMeter())
205 return meterElement();
206
207 return AccessibilityObject::element();
208}
