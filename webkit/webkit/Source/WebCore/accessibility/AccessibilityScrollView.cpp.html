62AccessibilityObject* AccessibilityScrollView::scrollBar(AccessibilityOrientation orientation)
63{
64 updateScrollbars();
65
66 switch (orientation) {
67 // ARIA 1.1 Elements with the role scrollbar have an implicit aria-orientation value of vertical.
68 case AccessibilityOrientation::Undefined:
69 case AccessibilityOrientation::Vertical:
70 return m_verticalScrollbar ? m_verticalScrollbar.get() : nullptr;
71 case AccessibilityOrientation::Horizontal:
72 return m_horizontalScrollbar ? m_horizontalScrollbar.get() : nullptr;
73 }
74
75 return nullptr;
76}
77
78// If this is WebKit1 then the native scroll view needs to return the
79// AX information (because there are no scroll bar children in the ScrollView object in WK1).
80// In WebKit2, the ScrollView object will return the AX information (because there are no platform widgets).
81bool AccessibilityScrollView::isAttachment() const
82{
83 return m_scrollView && m_scrollView->platformWidget();
84}

117void AccessibilityScrollView::updateChildrenIfNecessary()
118{
119 // Always update our children when asked for them so that we don't inadvertently cache them after
120 // a new web area has been created for this scroll view (like when moving back and forth through history).
121 // Since a ScrollViews children will always be relatively small and limited this should not be a performance problem.
122 clearChildren();
123 addChildren();
124}
125
126void AccessibilityScrollView::updateScrollbars()
127{
128 if (!m_scrollView)
129 return;
130
131 if (m_scrollView->horizontalScrollbar() && !m_horizontalScrollbar)
132 m_horizontalScrollbar = addChildScrollbar(m_scrollView->horizontalScrollbar());
133 else if (!m_scrollView->horizontalScrollbar() && m_horizontalScrollbar) {
134 removeChildScrollbar(m_horizontalScrollbar.get());
135 m_horizontalScrollbar = nullptr;
136 }
137
138 if (m_scrollView->verticalScrollbar() && !m_verticalScrollbar)
139 m_verticalScrollbar = addChildScrollbar(m_scrollView->verticalScrollbar());
140 else if (!m_scrollView->verticalScrollbar() && m_verticalScrollbar) {
141 removeChildScrollbar(m_verticalScrollbar.get());
142 m_verticalScrollbar = nullptr;
143 }
144}

146void AccessibilityScrollView::removeChildScrollbar(AccessibilityObject* scrollbar)
147{
148 size_t pos = m_children.find(scrollbar);
149 if (pos != WTF::notFound) {
150 m_children[pos]->detachFromParent();
151 m_children.remove(pos);
152 }
153}
154
155AccessibilityScrollbar* AccessibilityScrollView::addChildScrollbar(Scrollbar* scrollbar)
156{
157 if (!scrollbar)
158 return nullptr;
159
160 AXObjectCache* cache = axObjectCache();
161 if (!cache)
162 return nullptr;
163
164 auto& scrollBarObject = downcast<AccessibilityScrollbar>(*cache->getOrCreate(scrollbar));
165 scrollBarObject.setParent(this);
166 m_children.append(&scrollBarObject);
167 return &scrollBarObject;
168}

186void AccessibilityScrollView::addChildren()

195AccessibilityObject* AccessibilityScrollView::webAreaObject() const
196{
197 if (!is<FrameView>(m_scrollView.get()))
198 return nullptr;
199
200 Document* document = downcast<FrameView>(*m_scrollView).frame().document();
201 if (!document || !document->hasLivingRenderTree())
202 return nullptr;
203
204 if (AXObjectCache* cache = axObjectCache())
205 return cache->getOrCreate(document);
206
207 return nullptr;
208}

210AXCoreObject* AccessibilityScrollView::accessibilityHitTest(const IntPoint& point) const
211{
212 AccessibilityObject* webArea = webAreaObject();
213 if (!webArea)
214 return nullptr;
215
216 if (m_horizontalScrollbar && m_horizontalScrollbar->elementRect().contains(point))
217 return m_horizontalScrollbar.get();
218 if (m_verticalScrollbar && m_verticalScrollbar->elementRect().contains(point))
219 return m_verticalScrollbar.get();
220
221 return webArea->accessibilityHitTest(point);
222}
223
224LayoutRect AccessibilityScrollView::elementRect() const
225{
226 if (!m_scrollView)
227 return LayoutRect();
228
229 LayoutRect rect = m_scrollView->frameRect();
230 float topContentInset = m_scrollView->topContentInset();
231 // Top content inset pushes the frame down and shrinks it.
232 rect.move(0, topContentInset);
233 rect.contract(0, topContentInset);
234 return rect;
235}

245AccessibilityObject* AccessibilityScrollView::parentObject() const
246{
247 if (!is<FrameView>(m_scrollView.get()))
248 return nullptr;
249
250 AXObjectCache* cache = axObjectCache();
251 if (!cache)
252 return nullptr;
253
254 HTMLFrameOwnerElement* owner = downcast<FrameView>(*m_scrollView).frame().ownerElement();
255 if (owner && owner->renderer())
256 return cache->getOrCreate(owner);
257
258 return nullptr;
259}
260
261AccessibilityObject* AccessibilityScrollView::parentObjectIfExists() const
262{
263 if (!is<FrameView>(m_scrollView.get()))
264 return nullptr;
265
266 AXObjectCache* cache = axObjectCache();
267 if (!cache)
268 return nullptr;
269
270 HTMLFrameOwnerElement* owner = downcast<FrameView>(*m_scrollView).frame().ownerElement();
271 if (owner && owner->renderer())
272 return cache->get(owner);
273
274 return nullptr;
275}
