26#pragma once
27
28#include "AccessibilityObject.h"
29
30namespace WebCore {
31
32class AccessibilityScrollbar;
33class Scrollbar;
34class ScrollView;
35
36class AccessibilityScrollView final : public AccessibilityObject {
37public:
38 static Ref<AccessibilityScrollView> create(ScrollView*);
39 AccessibilityRole roleValue() const override { return AccessibilityRole::ScrollArea; }
40 ScrollView* scrollView() const override { return m_scrollView.get(); }
41
42 virtual ~AccessibilityScrollView();
43
44 AccessibilityObject* webAreaObject() const override;
45
46private:
47 explicit AccessibilityScrollView(ScrollView*);
48 void detachRemoteParts(AccessibilityDetachmentType) override;
49
50 ScrollableArea* getScrollableAreaIfScrollable() const override;
51 void scrollTo(const IntPoint&) const override;
52 bool computeAccessibilityIsIgnored() const override;
53 bool isAccessibilityScrollViewInstance() const override { return true; }
54 bool isEnabled() const override { return true; }
55
56 bool isAttachment() const override;
57 PlatformWidget platformWidget() const override;
58 Widget* widgetForAttachmentView() const override;
59
60 AccessibilityObject* scrollBar(AccessibilityOrientation) override;
61 void addChildren() override;
62 void clearChildren() override;
63 AXCoreObject* accessibilityHitTest(const IntPoint&) const override;
64 void updateChildrenIfNecessary() override;
65 void setNeedsToUpdateChildren() override { m_childrenDirty = true; }
66 void updateScrollbars();
67 void setFocused(bool) override;
68 bool canSetFocusAttribute() const override;
69 bool isFocused() const override;
70
71 FrameView* documentFrameView() const override;
72 LayoutRect elementRect() const override;
73 AccessibilityObject* parentObject() const override;
74 AccessibilityObject* parentObjectIfExists() const override;
75
76 AccessibilityObject* firstChild() const override { return webAreaObject(); }
77 AccessibilityScrollbar* addChildScrollbar(Scrollbar*);
78 void removeChildScrollbar(AccessibilityObject*);
79
80 WeakPtr<ScrollView> m_scrollView;
81 RefPtr<AccessibilityObject> m_horizontalScrollbar;
82 RefPtr<AccessibilityObject> m_verticalScrollbar;
83 bool m_childrenDirty;
84};
85
86} // namespace WebCore
87
88SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilityScrollView, isAccessibilityScrollViewInstance())
