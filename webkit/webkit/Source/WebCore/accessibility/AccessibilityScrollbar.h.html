29#pragma once
30
31#include "AccessibilityMockObject.h"
32
33namespace WebCore {
34
35class Scrollbar;
36
37class AccessibilityScrollbar final : public AccessibilityMockObject {
38public:
39 static Ref<AccessibilityScrollbar> create(Scrollbar*);
40
41 Scrollbar* scrollbar() const { return m_scrollbar.get(); }
42
43private:
44 explicit AccessibilityScrollbar(Scrollbar*);
45
46 bool canSetValueAttribute() const override { return true; }
47 bool canSetNumericValue() const override { return true; }
48
49 bool isAccessibilityScrollbar() const override { return true; }
50 LayoutRect elementRect() const override;
51
52 AccessibilityRole roleValue() const override { return AccessibilityRole::ScrollBar; }
53 AccessibilityOrientation orientation() const override;
54 Document* document() const override;
55 bool isEnabled() const override;
56
57 // Assumes float [0..1]
58 bool setValue(float) override;
59 float valueForRange() const override;
60
61 RefPtr<Scrollbar> m_scrollbar;
62};
63
64} // namespace WebCore
65
66SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilityScrollbar, isAccessibilityScrollbar())
