53AccessibilityOrientation AccessibilitySlider::orientation() const
54{
55 // Default to horizontal in the unknown case.
56 if (!m_renderer)
57 return AccessibilityOrientation::Horizontal;
58
59 auto ariaOrientation = getAttribute(aria_orientationAttr);
60 if (equalLettersIgnoringASCIICase(ariaOrientation, "horizontal"))
61 return AccessibilityOrientation::Horizontal;
62 if (equalLettersIgnoringASCIICase(ariaOrientation, "vertical"))
63 return AccessibilityOrientation::Vertical;
64 if (equalLettersIgnoringASCIICase(ariaOrientation, "undefined"))
65 return AccessibilityOrientation::Undefined;
66
67 const RenderStyle& style = m_renderer->style();
68
69 ControlPart styleAppearance = style.appearance();
70 switch (styleAppearance) {
71 case SliderThumbHorizontalPart:
72 case SliderHorizontalPart:
73 case MediaSliderPart:
74 case MediaFullScreenVolumeSliderPart:
75 return AccessibilityOrientation::Horizontal;
76
77 case SliderThumbVerticalPart:
78 case SliderVerticalPart:
79 case MediaVolumeSliderPart:
80 return AccessibilityOrientation::Vertical;
81
82 default:
83 return AccessibilityOrientation::Horizontal;
84 }
85}

87void AccessibilitySlider::addChildren()
88{
89 ASSERT(!m_haveChildren);
90
91 m_haveChildren = true;
92
93 AXObjectCache* cache = m_renderer->document().axObjectCache();
94
95 auto& thumb = downcast<AccessibilitySliderThumb>(*cache->create(AccessibilityRole::SliderThumb));
96 thumb.setParent(this);
97
98 // Before actually adding the value indicator to the hierarchy,
99 // allow the platform to make a final decision about it.
100 if (thumb.accessibilityIsIgnored())
101 cache->remove(thumb.objectID());
102 else
103 m_children.append(&thumb);
104}

111AXCoreObject* AccessibilitySlider::elementAccessibilityHitTest(const IntPoint& point) const
112{
113 if (m_children.size()) {
114 ASSERT(m_children.size() == 1);
115 if (m_children[0]->elementRect().contains(point))
116 return m_children[0].get();
117 }
118
119 return axObjectCache()->getOrCreate(renderer());
120}

137bool AccessibilitySlider::setValue(const String& value)
138{
139 HTMLInputElement* input = inputElement();
140
141 if (input->value() == value)
142 return true;
143
144 input->setValue(value, DispatchChangeEvent);
145 return true;
146}

163LayoutRect AccessibilitySliderThumb::elementRect() const
164{
165 if (!m_parent)
166 return LayoutRect();
167
168 RenderObject* sliderRenderer = m_parent->renderer();
169 if (!sliderRenderer || !sliderRenderer->isSlider())
170 return LayoutRect();
171 if (auto* thumbRenderer = downcast<RenderSlider>(*sliderRenderer).element().sliderThumbElement()->renderer())
172 return thumbRenderer->absoluteBoundingBoxRect();
173 return LayoutRect();
174}
