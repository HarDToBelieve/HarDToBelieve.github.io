26#pragma once
27
28#include "AccessibilityMockObject.h"
29#include "SpinButtonElement.h"
30
31namespace WebCore {
32
33class AccessibilitySpinButton final : public AccessibilityMockObject {
34public:
35 static Ref<AccessibilitySpinButton> create();
36 virtual ~AccessibilitySpinButton();
37
38 void setSpinButtonElement(SpinButtonElement* spinButton) { m_spinButtonElement = spinButton; }
39
40 AXCoreObject* incrementButton() override;
41 AXCoreObject* decrementButton() override;
42
43 void step(int amount);
44
45private:
46 AccessibilitySpinButton();
47
48 AccessibilityRole roleValue() const override { return AccessibilityRole::SpinButton; }
49 bool isNativeSpinButton() const override { return true; }
50 void addChildren() override;
51 LayoutRect elementRect() const override;
52
53 SpinButtonElement* m_spinButtonElement;
54};
55
56class AccessibilitySpinButtonPart final : public AccessibilityMockObject {
57public:
58 static Ref<AccessibilitySpinButtonPart> create();
59 virtual ~AccessibilitySpinButtonPart() = default;
60
61 bool isIncrementor() const override { return m_isIncrementor; }
62 void setIsIncrementor(bool value) { m_isIncrementor = value; }
63
64private:
65 AccessibilitySpinButtonPart();
66
67 bool press() override;
68 AccessibilityRole roleValue() const override { return AccessibilityRole::SpinButtonPart; }
69 bool isSpinButtonPart() const override { return true; }
70 LayoutRect elementRect() const override;
71
72 unsigned m_isIncrementor : 1;
73};
74
75} // namespace WebCore
76
77SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilitySpinButton, isNativeSpinButton())
78SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilitySpinButtonPart, isSpinButtonPart())
