29#pragma once
30
31#include "AccessibilityRenderObject.h"
32#include "AccessibilityTableCell.h"
33#include <wtf/Forward.h>
34
35namespace WebCore {
36
37class HTMLTableElement;
38class RenderTableSection;
39
40class AccessibilityTable : public AccessibilityRenderObject {
41public:
42 static Ref<AccessibilityTable> create(RenderObject*);
43 virtual ~AccessibilityTable();
44
45 void init() final;
46
47 AccessibilityRole roleValue() const final;
48 virtual bool isAriaTable() const { return false; }
49
50 void addChildren() override;
51 void clearChildren() final;
52
53 AccessibilityChildrenVector columns() override;
54 AccessibilityChildrenVector rows() override;
55
56 unsigned columnCount() override;
57 unsigned rowCount() override;
58 int tableLevel() const final;
59
60 String title() const final;
61
62 // all the cells in the table
63 AccessibilityChildrenVector cells() override;
64 AXCoreObject* cellForColumnAndRow(unsigned column, unsigned row) override;
65
66 AccessibilityChildrenVector columnHeaders() override;
67 AccessibilityChildrenVector rowHeaders() override;
68 AccessibilityChildrenVector visibleRows() override;
69
70 // Returns an object that contains, as children, all the objects that act as headers.
71 AXCoreObject* headerContainer() override;
72
73 bool isTable() const override { return true; }
74 // Returns whether it is exposed as an AccessibilityTable to the platform.
75 bool isExposable() const override;
76
77 int axColumnCount() const override;
78 int axRowCount() const override;
79
80protected:
81 explicit AccessibilityTable(RenderObject*);
82
83 AccessibilityChildrenVector m_rows;
84 AccessibilityChildrenVector m_columns;
85
86 RefPtr<AccessibilityObject> m_headerContainer;
87 bool m_isExposable;
88
89 bool hasARIARole() const;
90
91 // Used in type checking function is<AccessibilityTable>.
92 bool isAccessibilityTableInstance() const final { return true; }
93
94 // isDataTable is whether it is exposed as an AccessibilityTable because the heuristic
95 // think this "looks" like a data-based table (instead of a table used for layout).
96 bool isDataTable() const final;
97 bool computeAccessibilityIsIgnored() const final;
98
99private:
100 virtual bool computeIsTableExposableThroughAccessibility() const;
101 void titleElementText(Vector<AccessibilityText>&) const final;
102 HTMLTableElement* tableElement() const;
103 void addChildrenFromSection(RenderTableSection*, unsigned& maxColumnCount);
104 void addTableCellChild(AccessibilityObject*, HashSet<AccessibilityObject*>& appendedRows, unsigned& columnCount);
105};
106
107} // namespace WebCore
108
109SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilityTable, isAccessibilityTableInstance())
