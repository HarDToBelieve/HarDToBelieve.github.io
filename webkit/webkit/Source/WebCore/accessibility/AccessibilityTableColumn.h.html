39class AccessibilityTableColumn final : public AccessibilityMockObject {
40public:
41 static Ref<AccessibilityTableColumn> create();
42 virtual ~AccessibilityTableColumn();
43
44 AXCoreObject* columnHeader() override;
45
46 AccessibilityRole roleValue() const override { return AccessibilityRole::Column; }
47
48 void setColumnIndex(unsigned columnIndex) { m_columnIndex = columnIndex; }
49 unsigned columnIndex() const override { return m_columnIndex; }
50
51 void addChildren() override;
52 void setParent(AccessibilityObject*) override;
53
54 LayoutRect elementRect() const override;
55
56private:
57 AccessibilityTableColumn();
58
59 AccessibilityObject* headerObjectForSection(RenderTableSection*, bool thTagRequired);
60 bool computeAccessibilityIsIgnored() const override;
61 bool isTableColumn() const override { return true; }
62
63 bool isAccessibilityTableColumnInstance() const final { return true; }
64 unsigned m_columnIndex;
65};

67} // namespace WebCore
