277static void AXPostNotificationWithUserInfo(AccessibilityObjectWrapper *object, NSString *notification, id userInfo, bool skipSystemNotification = false)
278{
279 if (id associatedPluginParent = [object associatedPluginParent])
280 object = associatedPluginParent;
281
282 // To simplify monitoring for notifications in tests, repost as a simple NSNotification instead of forcing test infrastucture to setup an IPC client and do all the translation between WebCore types and platform specific IPC types and back
283 if (UNLIKELY(axShouldRepostNotificationsForTests))
284 [object accessibilityPostedNotification:notification userInfo:userInfo];
285 else if (skipSystemNotification)
286 return;
287
288 NSAccessibilityPostNotificationWithUserInfo(object, notification, userInfo);
289}

291void AXObjectCache::postPlatformNotification(AXCoreObject* obj, AXNotification notification)
292{
293 if (!obj)
294 return;
295
296 bool skipSystemNotification = false;
297 // Some notifications are unique to Safari and do not have NSAccessibility equivalents.
298 NSString *macNotification;
299 switch (notification) {
300 case AXActiveDescendantChanged:
301 // An active descendant change for trees means a selected rows change.
302 if (obj->isTree() || obj->isTable())
303 macNotification = NSAccessibilitySelectedRowsChangedNotification;
304
305 // When a combobox uses active descendant, it means the selected item in its associated
306 // list has changed. In these cases we should use selected children changed, because
307 // we don't want the focus to change away from the combobox where the user is typing.
308 else if (obj->isComboBox() || obj->isList() || obj->isListBox())
309 macNotification = NSAccessibilitySelectedChildrenChangedNotification;
310 else
311 macNotification = NSAccessibilityFocusedUIElementChangedNotification;
312 break;
313 case AXAutocorrectionOccured:
314 macNotification = @"AXAutocorrectionOccurred";
315 break;
316 case AXCurrentStateChanged:
317 macNotification = NSAccessibilityCurrentStateChangedNotification;
318 break;
319 case AXFocusedUIElementChanged:
320 macNotification = NSAccessibilityFocusedUIElementChangedNotification;
321 break;
322 case AXImageOverlayChanged:
323 macNotification = @"AXImageOverlayChanged";
324 break;
325 case AXLayoutComplete:
326 macNotification = @"AXLayoutComplete";
327 break;
328 case AXLoadComplete:
329 case AXFrameLoadComplete:
330 macNotification = @"AXLoadComplete";
331 // Frame loading events are handled by the UIProcess on macOS to improve reliability.
332 // On macOS, before notifications are allowed by AppKit to be sent to clients, you need to have a client (e.g. VoiceOver)
333 // register for that notification. Because these new processes appear before VO has a chance to register, it will often
334 // miss AXLoadComplete notifications. By moving them to the UIProcess, we can eliminate that issue.
335 skipSystemNotification = true;
336 break;
337 case AXInvalidStatusChanged:
338 macNotification = @"AXInvalidStatusChanged";
339 break;
340 case AXSelectedChildrenChanged:
341 if (obj->isTable() && obj->isExposable())
342 macNotification = NSAccessibilitySelectedRowsChangedNotification;
343 else
344 macNotification = NSAccessibilitySelectedChildrenChangedNotification;
345 break;
346 case AXSelectedTextChanged:
347 macNotification = NSAccessibilitySelectedTextChangedNotification;
348 break;
349 case AXCheckedStateChanged:
350 case AXValueChanged:
351 macNotification = NSAccessibilityValueChangedNotification;
352 break;
353 case AXLiveRegionCreated:
354 macNotification = NSAccessibilityLiveRegionCreatedNotification;
355 break;
356 case AXLiveRegionChanged:
357 macNotification = NSAccessibilityLiveRegionChangedNotification;
358 break;
359 case AXRowCountChanged:
360 macNotification = NSAccessibilityRowCountChangedNotification;
361 break;
362 case AXRowExpanded:
363 macNotification = NSAccessibilityRowExpandedNotification;
364 break;
365 case AXRowCollapsed:
366 macNotification = NSAccessibilityRowCollapsedNotification;
367 break;
368 case AXElementBusyChanged:
369 macNotification = @"AXElementBusyChanged";
370 break;
371 case AXExpandedChanged:
372 macNotification = @"AXExpandedChanged";
373 break;
374 case AXSortDirectionChanged:
375 macNotification = @"AXSortDirectionChanged";
376 break;
377 case AXMenuClosed:
378 macNotification = (id)kAXMenuClosedNotification;
379 break;
380 case AXMenuListItemSelected:
381 case AXMenuListValueChanged:
382 macNotification = (id)kAXMenuItemSelectedNotification;
383 break;
384 case AXPressDidSucceed:
385 macNotification = @"AXPressDidSucceed";
386 break;
387 case AXPressDidFail:
388 macNotification = @"AXPressDidFail";
389 break;
390 case AXMenuOpened:
391 macNotification = (id)kAXMenuOpenedNotification;
392 break;
393 case AXDraggingStarted:
394 macNotification = (id)kAXDraggingSourceDragBeganNotification;
395 break;
396 case AXDraggingEnded:
397 macNotification = (id)kAXDraggingSourceDragEndedNotification;
398 break;
399 case AXDraggingEnteredDropZone:
400 macNotification = (id)kAXDraggingDestinationDropAllowedNotification;
401 break;
402 case AXDraggingDropped:
403 macNotification = (id)kAXDraggingDestinationDragAcceptedNotification;
404 break;
405 case AXDraggingExitedDropZone:
406 macNotification = (id)kAXDraggingDestinationDragNotAcceptedNotification;
407 break;
408 default:
409 return;
410 }
411
412 // NSAccessibilityPostNotification will call this method, (but not when running DRT), so ASSERT here to make sure it does not crash.
413 // https://bugs.webkit.org/show_bug.cgi?id=46662
414 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
415 ASSERT([obj->wrapper() accessibilityIsIgnored] || true);
416 ALLOW_DEPRECATED_DECLARATIONS_END
417
418 AXPostNotificationWithUserInfo(obj->wrapper(), macNotification, nil, skipSystemNotification);
419}

421void AXObjectCache::postTextStateChangePlatformNotification(AXCoreObject* object, const AXTextStateChangeIntent& intent, const VisibleSelection& selection)
422{
423 if (!object)
424 object = rootWebArea();
425
426 if (!object)
427 return;
428
429 auto userInfo = adoptNS([[NSMutableDictionary alloc] initWithCapacity:5]);
430 if (m_isSynchronizingSelection)
431 [userInfo setObject:@YES forKey:NSAccessibilityTextStateSyncKey];
432 if (intent.type != AXTextStateChangeTypeUnknown) {
433 [userInfo setObject:@(platformChangeTypeForWebCoreChangeType(intent.type)) forKey:NSAccessibilityTextStateChangeTypeKey];
434 switch (intent.type) {
435 case AXTextStateChangeTypeSelectionMove:
436 case AXTextStateChangeTypeSelectionExtend:
437 case AXTextStateChangeTypeSelectionBoundary:
438 [userInfo setObject:@(platformDirectionForWebCoreDirection(intent.selection.direction)) forKey:NSAccessibilityTextSelectionDirection];
439 switch (intent.selection.direction) {
440 case AXTextSelectionDirectionUnknown:
441 break;
442 case AXTextSelectionDirectionBeginning:
443 case AXTextSelectionDirectionEnd:
444 case AXTextSelectionDirectionPrevious:
445 case AXTextSelectionDirectionNext:
446 [userInfo setObject:@(platformGranularityForWebCoreGranularity(intent.selection.granularity)) forKey:NSAccessibilityTextSelectionGranularity];
447 break;
448 case AXTextSelectionDirectionDiscontiguous:
449 break;
450 }
451 if (intent.selection.focusChange)
452 [userInfo setObject:@(intent.selection.focusChange) forKey:NSAccessibilityTextSelectionChangedFocus];
453 break;
454 case AXTextStateChangeTypeUnknown:
455 case AXTextStateChangeTypeEdit:
456 break;
457 }
458 }
459 if (!selection.isNone()) {
460 if (auto textMarkerRange = textMarkerRangeFromVisiblePositions(this, selection.visibleStart(), selection.visibleEnd()))
461 [userInfo setObject:(id)textMarkerRange forKey:NSAccessibilitySelectedTextMarkerRangeAttribute];
462 }
463
464 if (id wrapper = object->wrapper())
465 [userInfo setObject:wrapper forKey:NSAccessibilityTextChangeElement];
466
467 if (auto root = rootWebArea()) {
468 AXPostNotificationWithUserInfo(rootWebArea()->wrapper(), NSAccessibilitySelectedTextChangedNotification, userInfo.get());
469 if (root->wrapper() != object->wrapper())
470 AXPostNotificationWithUserInfo(object->wrapper(), NSAccessibilitySelectedTextChangedNotification, userInfo.get());
471 }
472}

488template <typename TextMarkerTargetType>
489static NSDictionary *textReplacementChangeDictionary(AXCoreObject& object, AXTextEditType type, const String& string, TextMarkerTargetType& markerTarget)
490{
491 NSString *text = (NSString *)string;
492 NSUInteger length = [text length];
493 if (!length)
494 return nil;
495 auto change = adoptNS([[NSMutableDictionary alloc] initWithCapacity:4]);
496 [change setObject:@(platformEditTypeForWebCoreEditType(type)) forKey:NSAccessibilityTextEditType];
497 if (length > AXValueChangeTruncationLength) {
498 [change setObject:@(length) forKey:NSAccessibilityTextChangeValueLength];
499 text = [text substringToIndex:AXValueChangeTruncationLength];
500 }
501 [change setObject:text forKey:NSAccessibilityTextChangeValue];
502 addTextMarkerFor(change.get(), object, markerTarget);
503 return change.autorelease();
504}

514static void postUserInfoForChanges(AXCoreObject& rootWebArea, AXCoreObject& object, NSMutableArray* changes)
515{
516 auto userInfo = adoptNS([[NSMutableDictionary alloc] initWithCapacity:4]);
517 [userInfo setObject:@(platformChangeTypeForWebCoreChangeType(AXTextStateChangeTypeEdit)) forKey:NSAccessibilityTextStateChangeTypeKey];
518 if (changes.count)
519 [userInfo setObject:changes forKey:NSAccessibilityTextChangeValues];
520
521 if (id wrapper = object.wrapper())
522 [userInfo setObject:wrapper forKey:NSAccessibilityTextChangeElement];
523
524 AXPostNotificationWithUserInfo(rootWebArea.wrapper(), NSAccessibilityValueChangedNotification, userInfo.get());
525 if (rootWebArea.wrapper() != object.wrapper())
526 AXPostNotificationWithUserInfo(object.wrapper(), NSAccessibilityValueChangedNotification, userInfo.get());
527}

529void AXObjectCache::postTextReplacementPlatformNotification(AXCoreObject* object, AXTextEditType deletionType, const String& deletedText, AXTextEditType insertionType, const String& insertedText, const VisiblePosition& position)
530{
531 if (!object)
532 object = rootWebArea();
533
534 if (!object)
535 return;
536
537 auto changes = adoptNS([[NSMutableArray alloc] initWithCapacity:2]);
538 if (NSDictionary *change = textReplacementChangeDictionary(*object, deletionType, deletedText, position))
539 [changes addObject:change];
540 if (NSDictionary *change = textReplacementChangeDictionary(*object, insertionType, insertedText, position))
541 [changes addObject:change];
542
543 if (auto* root = rootWebArea())
544 postUserInfoForChanges(*root, *object, changes.get());
545}

634static TextMarkerData getBytesFromAXTextMarker(AXTextMarkerRef textMarker)
635{
636 TextMarkerData data;
637 if (!textMarker)
638 return data;
639
640 ASSERT(CFGetTypeID(textMarker) == AXTextMarkerGetTypeID());
641 if (CFGetTypeID(textMarker) != AXTextMarkerGetTypeID())
642 return data;
643
644 ASSERT(AXTextMarkerGetLength(textMarker) == sizeof(data));
645 if (AXTextMarkerGetLength(textMarker) != sizeof(data))
646 return data;
647
648 memcpy(&data, AXTextMarkerGetBytePtr(textMarker), sizeof(data));
649 return data;
650}
