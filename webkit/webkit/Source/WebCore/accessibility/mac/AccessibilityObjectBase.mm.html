41String AccessibilityObject::speechHintAttributeValue() const
42{
43 auto speak = speakAsProperty();
44 NSMutableArray<NSString *> *hints = [NSMutableArray array];
45 [hints addObject:(speak & SpeakAs::SpellOut) ? @"spell-out" : @"normal"];
46 if (speak & SpeakAs::Digits)
47 [hints addObject:@"digits"];
48 if (speak & SpeakAs::LiteralPunctuation)
49 [hints addObject:@"literal-punctuation"];
50 if (speak & SpeakAs::NoPunctuation)
51 [hints addObject:@"no-punctuation"];
52 return [hints componentsJoinedByString:@" "];
53}

55static bool isVisibleText(AccessibilityTextSource textSource)
56{
57 switch (textSource) {
58 case AccessibilityTextSource::Visible:
59 case AccessibilityTextSource::Children:
60 case AccessibilityTextSource::LabelByElement:
61 return true;
62 case AccessibilityTextSource::Alternative:
63 case AccessibilityTextSource::Summary:
64 case AccessibilityTextSource::Help:
65 case AccessibilityTextSource::TitleTag:
66 case AccessibilityTextSource::Placeholder:
67 case AccessibilityTextSource::Title:
68 case AccessibilityTextSource::Subtitle:
69 case AccessibilityTextSource::Action:
70 return false;
71 }
72}

74static bool isDescriptiveText(AccessibilityTextSource textSource)
75{
76 switch (textSource) {
77 case AccessibilityTextSource::Alternative:
78 case AccessibilityTextSource::Visible:
79 case AccessibilityTextSource::Children:
80 case AccessibilityTextSource::LabelByElement:
81 return true;
82 case AccessibilityTextSource::Summary:
83 case AccessibilityTextSource::Help:
84 case AccessibilityTextSource::TitleTag:
85 case AccessibilityTextSource::Placeholder:
86 case AccessibilityTextSource::Title:
87 case AccessibilityTextSource::Subtitle:
88 case AccessibilityTextSource::Action:
89 return false;
90 }

93String AccessibilityObject::descriptionAttributeValue() const
94{
95 // Static text objects should not have a description. Its content is communicated in its AXValue.
96 // One exception is the media control labels that have a value and a description. Those are set programatically.
97 if (roleValue() == AccessibilityRole::StaticText && !isMediaControlLabel())
98 return { };
99
100 Vector<AccessibilityText> textOrder;
101 accessibilityText(textOrder);
102
103 // Determine if any visible text is available, which influences our usage of title tag.
104 bool visibleTextAvailable = false;
105 for (const auto& text : textOrder) {
106 if (isVisibleText(text.textSource)) {
107 visibleTextAvailable = true;
108 break;
109 }
110 }
111
112 NSMutableString *returnText = [NSMutableString string];
113 for (const auto& text : textOrder) {
114 if (text.textSource == AccessibilityTextSource::Alternative) {
115 [returnText appendString:text.text];
116 break;
117 }
118
119 switch (text.textSource) {
120 // These are sub-components of one element (Attachment) that are re-combined in OSX and iOS.
121 case AccessibilityTextSource::Title:
122 case AccessibilityTextSource::Subtitle:
123 case AccessibilityTextSource::Action: {
124 if (!text.text.length())
125 break;
126 if ([returnText length])
127 [returnText appendString:@", "];
128 [returnText appendString:text.text];
129 break;
130 }
131 default:
132 break;
133 }
134
135 if (text.textSource == AccessibilityTextSource::TitleTag && !visibleTextAvailable) {
136 [returnText appendString:text.text];
137 break;
138 }
139 }
140
141 return returnText;
142}

144String AccessibilityObject::titleAttributeValue() const
145{
146 // Static text objects should not have a title. Its content is communicated in its AXValue.
147 if (roleValue() == AccessibilityRole::StaticText)
148 return String();
149
150 // Meter elements should communicate their content via AXValueDescription.
151 if (isMeter())
152 return { };
153
154 // Summary element should use its text node as AXTitle.
155 if (isSummary())
156 return textUnderElement();
157
158 // A file upload button presents a challenge because it has button text and a value, but the
159 // API doesn't support this paradigm.
160 // The compromise is to return the button type in the role description and the value of the file path in the title
161 if (isFileUploadButton() && fileUploadButtonReturnsValueInTitle())
162 return stringValue();
163
164 Vector<AccessibilityText> textOrder;
165 accessibilityText(textOrder);
166
167 for (const auto& text : textOrder) {
168 // If we have alternative text, then we should not expose a title.
169 if (text.textSource == AccessibilityTextSource::Alternative)
170 break;
171
172 // Once we encounter visible text, or the text from our children that should be used foremost.
173 if (text.textSource == AccessibilityTextSource::Visible || text.textSource == AccessibilityTextSource::Children)
174 return text.text;
175
176 // If there's an element that labels this object and it's not exposed, then we should use
177 // that text as our title.
178 if (text.textSource == AccessibilityTextSource::LabelByElement && !exposesTitleUIElement())
179 return text.text;
180 }
181
182 return { };
183}
184
185String AccessibilityObject::helpTextAttributeValue() const
186{
187 Vector<AccessibilityText> textOrder;
188 accessibilityText(textOrder);
189
190 // Determine if any descriptive text is available, which influences our usage of title tag.
191 bool descriptiveTextAvailable = false;
192 for (const auto& text : textOrder) {
193 if (isDescriptiveText(text.textSource)) {
194 descriptiveTextAvailable = true;
195 break;
196 }
197 }
198
199 for (const auto& text : textOrder) {
200 if (text.textSource == AccessibilityTextSource::Help || text.textSource == AccessibilityTextSource::Summary)
201 return text.text;
202
203 // If an element does NOT have other descriptive text the title tag should be used as its descriptive text.
204 // But, if those ARE available, then the title tag should be used for help text instead.
205 if (text.textSource == AccessibilityTextSource::TitleTag && descriptiveTextAvailable)
206 return text.text;
207 }
208
209 return { };
210}
