56void AccessibilityObject::overrideAttachmentParent(AXCoreObject* parent)
57{
58 if (!isAttachment())
59 return;
60
61 id parentWrapper = nil;
62 if (parent) {
63 if (parent->accessibilityIsIgnored())
64 parent = parent->parentObjectUnignored();
65 parentWrapper = parent->wrapper();
66 }
67
68 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
69 [[wrapper() attachmentView] accessibilitySetOverrideValue:parentWrapper forAttribute:NSAccessibilityParentAttribute];
70 ALLOW_DEPRECATED_DECLARATIONS_END
71}

73FloatRect AccessibilityObject::convertRectToPlatformSpace(const FloatRect& rect, AccessibilityConversionSpace space) const
74{
75 // WebKit1 code path... platformWidget() exists.
76 auto* frameView = documentFrameView();
77 if (frameView && frameView->platformWidget()) {
78 CGPoint point = CGPointMake(rect.x(), rect.y());
79 CGSize size = CGSizeMake(rect.size().width(), rect.size().height());
80 CGRect cgRect = CGRectMake(point.x, point.y, size.width, size.height);
81
82 NSRect nsRect = NSRectFromCGRect(cgRect);
83 NSView *view = frameView->documentView();
84 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
85 nsRect = [[view window] convertRectToScreen:[view convertRect:nsRect toView:nil]];
86 ALLOW_DEPRECATED_DECLARATIONS_END
87 return NSRectToCGRect(nsRect);
88 }
89
90 return convertFrameToSpace(rect, space);
91}

99bool AccessibilityObject::accessibilityIgnoreAttachment() const
100{
101 // FrameView attachments are now handled by AccessibilityScrollView,
102 // so if this is the attachment, it should be ignored.
103 Widget* widget = nullptr;
104 if (isAttachment() && (widget = widgetForAttachmentView()) && widget->isFrameView())
105 return true;
106
107 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
108 id attachmentView = widget ? NSAccessibilityUnignoredDescendant(widget->platformWidget()) : nil;
109 if (attachmentView)
110 return [attachmentView accessibilityIsIgnored];
111 ALLOW_DEPRECATED_DECLARATIONS_END
112
113 // Attachments are ignored by default (unless we determine that we should expose them).
114 return true;
115}

117AccessibilityObjectInclusion AccessibilityObject::accessibilityPlatformIncludesObject() const
118{
119 if (isMenuListPopup() || isMenuListOption())
120 return AccessibilityObjectInclusion::IgnoreObject;
121
122 if (roleValue() == AccessibilityRole::Caption && ariaRoleAttribute() == AccessibilityRole::Unknown)
123 return AccessibilityObjectInclusion::IgnoreObject;
124
125 if (roleValue() == AccessibilityRole::Mark)
126 return AccessibilityObjectInclusion::IncludeObject;
127
128 // Never expose an unknown object on the Mac. Clients of the AX API will not know what to do with it.
129 // Special case is when the unknown object is actually an attachment.
130 if (roleValue() == AccessibilityRole::Unknown && !isAttachment())
131 return AccessibilityObjectInclusion::IgnoreObject;
132
133 if (roleValue() == AccessibilityRole::Inline && !isStyleFormatGroup())
134 return AccessibilityObjectInclusion::IgnoreObject;
135
136 if (RenderObject* renderer = this->renderer()) {
137 // The legend element is ignored if it lives inside of a fieldset element that uses it to generate alternative text.
138 if (renderer->isLegend()) {
139 Element* element = this->element();
140 if (element && ancestorsOfType<HTMLFieldSetElement>(*element).first())
141 return AccessibilityObjectInclusion::IgnoreObject;
142 }
143 }
144
145 return AccessibilityObjectInclusion::DefaultBehavior;
146}

162String AccessibilityObject::rolePlatformString() const
163{
164 AccessibilityRole role = roleValue();
165
166 // If it is a label with just static text or an anonymous math operator, remap role to StaticText.
167 // The mfenced element creates anonymous RenderMathMLOperators with no RenderText
168 // descendants. These anonymous renderers are the only accessible objects
169 // containing the operator.
170 if ((role == AccessibilityRole::Label && is<AccessibilityLabel>(*this) && downcast<AccessibilityLabel>(*this).containsOnlyStaticText())
171 || isAnonymousMathOperator())
172 role = AccessibilityRole::StaticText;
173 else if (role == AccessibilityRole::Canvas && canvasHasFallbackContent())
174 role = AccessibilityRole::Group;
175
176 return Accessibility::roleToPlatformString(role);
177}

179String AccessibilityObject::rolePlatformDescription() const
180{
181 AccessibilityRole role = roleValue();
182 NSString *axRole = rolePlatformString();
183
184 if ([axRole isEqualToString:NSAccessibilityGroupRole]) {
185 if (isOutput())
186 return AXOutputText();
187
188 String ariaLandmarkRoleDescription = this->ariaLandmarkRoleDescription();
189 if (!ariaLandmarkRoleDescription.isEmpty())
190 return ariaLandmarkRoleDescription;
191
192 switch (role) {
193 case AccessibilityRole::Audio:
194 return localizedMediaControlElementString("AudioElement");
195 case AccessibilityRole::Definition:
196 return AXDefinitionText();
197 case AccessibilityRole::DescriptionListTerm:
198 case AccessibilityRole::Term:
199 return AXDescriptionListTermText();
200 case AccessibilityRole::DescriptionListDetail:
201 return AXDescriptionListDetailText();
202 case AccessibilityRole::Details:
203 return AXDetailsText();
204 case AccessibilityRole::Feed:
205 return AXFeedText();
206 case AccessibilityRole::Footer:
207 return AXFooterRoleDescriptionText();
208 case AccessibilityRole::Mark:
209 return AXMarkText();
210 case AccessibilityRole::Video:
211 return localizedMediaControlElementString("VideoElement");
212 case AccessibilityRole::GraphicsDocument:
213 return AXARIAContentGroupText(@"ARIADocument");
214 default:
215 return String();
216 }
217 }
218
219 if ([axRole isEqualToString:@"AXWebArea"])
220 return AXWebAreaText();
221
222 if ([axRole isEqualToString:@"AXLink"])
223 return AXLinkText();
224
225 if ([axRole isEqualToString:@"AXListMarker"])
226 return AXListMarkerText();
227
228 if ([axRole isEqualToString:@"AXImageMap"])
229 return AXImageMapText();
230
231 if ([axRole isEqualToString:@"AXHeading"])
232 return AXHeadingText();
233
234 if ([axRole isEqualToString:NSAccessibilityTextFieldRole]) {
235 auto* node = this->node();
236 if (is<HTMLInputElement>(node)) {
237 auto& input = downcast<HTMLInputElement>(*node);
238 if (input.isEmailField())
239 return AXEmailFieldText();
240 if (input.isTelephoneField())
241 return AXTelephoneFieldText();
242 if (input.isURLField())
243 return AXURLFieldText();
244 if (input.isNumberField())
245 return AXNumberFieldText();
246
247 // These input types are not enabled on mac yet, we check the type attribute for now.
248 auto& type = input.attributeWithoutSynchronization(HTMLNames::typeAttr);
249 if (equalLettersIgnoringASCIICase(type, "date"))
250 return AXDateFieldText();
251 if (equalLettersIgnoringASCIICase(type, "time"))
252 return AXTimeFieldText();
253 if (equalLettersIgnoringASCIICase(type, "week"))
254 return AXWeekFieldText();
255 if (equalLettersIgnoringASCIICase(type, "month"))
256 return AXMonthFieldText();
257 if (equalLettersIgnoringASCIICase(type, "datetime-local"))
258 return AXDateTimeFieldText();
259 }
260 }
261
262 if (isFileUploadButton())
263 return AXFileUploadButtonText();
264
265 // Only returning for DL (not UL or OL) because description changed with HTML5 from 'definition list' to
266 // superset 'description list' and does not return the same values in AX API on some OS versions.
267 if (isDescriptionList())
268 return AXDescriptionListText();
269
270 if (role == AccessibilityRole::HorizontalRule)
271 return AXHorizontalRuleDescriptionText();
272
273 // AppKit also returns AXTab for the role description for a tab item.
274 if (isTabItem())
275 return NSAccessibilityRoleDescription(@"AXTab", nil);
276
277 if (isSummary())
278 return AXSummaryText();
279
280 return String();
281}
