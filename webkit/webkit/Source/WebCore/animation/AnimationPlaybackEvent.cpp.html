32namespace WebCore {
33
34WTF_MAKE_ISO_ALLOCATED_IMPL(AnimationPlaybackEvent);
35
36AnimationPlaybackEvent::AnimationPlaybackEvent(const AtomString& type, const AnimationPlaybackEventInit& initializer, IsTrusted isTrusted)
37 : AnimationEventBase(type, initializer, isTrusted)
38{
39 if (initializer.currentTime)
40 m_currentTime = Seconds::fromMilliseconds(*initializer.currentTime);
41 else
42 m_currentTime = std::nullopt;
43
44 if (initializer.timelineTime)
45 m_timelineTime = Seconds::fromMilliseconds(*initializer.timelineTime);
46 else
47 m_timelineTime = std::nullopt;
48}
49
50AnimationPlaybackEvent::AnimationPlaybackEvent(const AtomString& type, std::optional<Seconds> currentTime, std::optional<Seconds> timelineTime, WebAnimation* animation)
51 : AnimationEventBase(type, animation, timelineTime)
52 , m_currentTime(currentTime)
53{
54}
55
56AnimationPlaybackEvent::~AnimationPlaybackEvent() = default;
57
58std::optional<double> AnimationPlaybackEvent::bindingsCurrentTime() const
59{
60 if (!m_currentTime)
61 return std::nullopt;
62 return secondsToWebAnimationsAPITime(m_currentTime.value());
63}
64
65std::optional<double> AnimationPlaybackEvent::bindingsTimelineTime() const
66{
67 if (!timelineTime())
68 return std::nullopt;
69 return secondsToWebAnimationsAPITime(timelineTime().value());
70}
