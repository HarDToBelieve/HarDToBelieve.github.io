55void CSSAnimation::syncPropertiesWithBackingAnimation()
56{
57 DeclarativeAnimation::syncPropertiesWithBackingAnimation();
58
59 if (!effect())
60 return;
61
62 suspendEffectInvalidation();
63
64 auto& animation = backingAnimation();
65 auto* animationEffect = effect();
66
67 if (!m_overriddenProperties.contains(Property::FillMode)) {
68 switch (animation.fillMode()) {
69 case AnimationFillMode::None:
70 animationEffect->setFill(FillMode::None);
71 break;
72 case AnimationFillMode::Backwards:
73 animationEffect->setFill(FillMode::Backwards);
74 break;
75 case AnimationFillMode::Forwards:
76 animationEffect->setFill(FillMode::Forwards);
77 break;
78 case AnimationFillMode::Both:
79 animationEffect->setFill(FillMode::Both);
80 break;
81 }
82 }
83
84 if (!m_overriddenProperties.contains(Property::Direction)) {
85 switch (animation.direction()) {
86 case Animation::AnimationDirectionNormal:
87 animationEffect->setDirection(PlaybackDirection::Normal);
88 break;
89 case Animation::AnimationDirectionAlternate:
90 animationEffect->setDirection(PlaybackDirection::Alternate);
91 break;
92 case Animation::AnimationDirectionReverse:
93 animationEffect->setDirection(PlaybackDirection::Reverse);
94 break;
95 case Animation::AnimationDirectionAlternateReverse:
96 animationEffect->setDirection(PlaybackDirection::AlternateReverse);
97 break;
98 }
99 }
100
101 if (!m_overriddenProperties.contains(Property::IterationCount)) {
102 auto iterationCount = animation.iterationCount();
103 animationEffect->setIterations(iterationCount == Animation::IterationCountInfinite ? std::numeric_limits<double>::infinity() : iterationCount);
104 }
105
106 if (!m_overriddenProperties.contains(Property::Delay))
107 animationEffect->setDelay(Seconds(animation.delay()));
108
109 if (!m_overriddenProperties.contains(Property::Duration))
110 animationEffect->setIterationDuration(Seconds(animation.duration()));
111
112 animationEffect->updateStaticTimingProperties();
113 effectTimingDidChange();
114
115 // Synchronize the play state
116 if (!m_overriddenProperties.contains(Property::PlayState)) {
117 if (animation.playState() == AnimationPlayState::Playing && playState() == WebAnimation::PlayState::Paused)
118 play();
119 else if (animation.playState() == AnimationPlayState::Paused && playState() == WebAnimation::PlayState::Running)
120 pause();
121 }
122
123 unsuspendEffectInvalidation();
124}

126ExceptionOr<void> CSSAnimation::bindingsPlay()
127{
128 // https://drafts.csswg.org/css-animations-2/#animations
129
130 // After a successful call to play() or pause() on a CSSAnimation, any subsequent change to the animation-play-state will
131 // no longer cause the CSSAnimation to be played or paused.
132
133 auto retVal = DeclarativeAnimation::bindingsPlay();
134 if (!retVal.hasException())
135 m_overriddenProperties.add(Property::PlayState);
136 return retVal;
137}

152void CSSAnimation::setBindingsEffect(RefPtr<AnimationEffect>&& newEffect)
153{
154 // https://drafts.csswg.org/css-animations-2/#animations
155
156 // After successfully setting the effect of a CSSAnimation to null or some AnimationEffect other than the original KeyframeEffect,
157 // all subsequent changes to animation properties other than animation-name or animation-play-state will not be reflected in that
158 // animation. Similarly, any change to matching @keyframes rules will not be reflected in that animation. However, if the last
159 // matching @keyframes rule is removed the animation must still be canceled.
160
161 auto* previousEffect = effect();
162 DeclarativeAnimation::setBindingsEffect(WTFMove(newEffect));
163 if (effect() != previousEffect) {
164 m_overriddenProperties.add(Property::Duration);
165 m_overriddenProperties.add(Property::TimingFunction);
166 m_overriddenProperties.add(Property::IterationCount);
167 m_overriddenProperties.add(Property::Direction);
168 m_overriddenProperties.add(Property::Delay);
169 m_overriddenProperties.add(Property::FillMode);
170 }
171}

173void CSSAnimation::setBindingsStartTime(std::optional<double> startTime)
174{
175 // https://drafts.csswg.org/css-animations-2/#animations
176
177 // After a successful call to reverse() on a CSSAnimation or after successfully setting the startTime on a CSSAnimation,
178 // if, as a result of that call the play state of the CSSAnimation changes to or from the paused play state, any subsequent
179 // change to the animation-play-state will no longer cause the CSSAnimation to be played or paused.
180
181 auto previousPlayState = playState();
182 DeclarativeAnimation::setBindingsStartTime(startTime);
183 auto currentPlayState = playState();
184 if (currentPlayState != previousPlayState && (currentPlayState == PlayState::Paused || previousPlayState == PlayState::Paused))
185 m_overriddenProperties.add(Property::PlayState);
186}

188ExceptionOr<void> CSSAnimation::bindingsReverse()
189{
190 // https://drafts.csswg.org/css-animations-2/#animations
191
192 // After a successful call to reverse() on a CSSAnimation or after successfully setting the startTime on a CSSAnimation,
193 // if, as a result of that call the play state of the CSSAnimation changes to or from the paused play state, any subsequent
194 // change to the animation-play-state will no longer cause the CSSAnimation to be played or paused.
195
196 auto previousPlayState = playState();
197 auto retVal = DeclarativeAnimation::bindingsReverse();
198 if (!retVal.hasException()) {
199 auto currentPlayState = playState();
200 if (currentPlayState != previousPlayState && (currentPlayState == PlayState::Paused || previousPlayState == PlayState::Paused))
201 m_overriddenProperties.add(Property::PlayState);
202 }
203 return retVal;
204}

206void CSSAnimation::effectTimingWasUpdatedUsingBindings(OptionalEffectTiming timing)
207{
208 // https://drafts.csswg.org/css-animations-2/#animations
209
210 // After a successful call to updateTiming() on the KeyframeEffect associated with a CSSAnimation, for each property
211 // included in the timing parameter, any subsequent change to a corresponding animation property will not be reflected
212 // in that animation.
213
214 if (timing.duration)
215 m_overriddenProperties.add(Property::Duration);
216
217 if (timing.iterations)
218 m_overriddenProperties.add(Property::IterationCount);
219
220 if (timing.delay)
221 m_overriddenProperties.add(Property::Delay);
222
223 if (!timing.easing.isNull())
224 m_overriddenProperties.add(Property::TimingFunction);
225
226 if (timing.fill)
227 m_overriddenProperties.add(Property::FillMode);
228
229 if (timing.direction)
230 m_overriddenProperties.add(Property::Direction);
231}

233void CSSAnimation::effectKeyframesWereSetUsingBindings()
234{
235 // https://drafts.csswg.org/css-animations-2/#animations
236
237 // After a successful call to setKeyframes() on the KeyframeEffect associated with a CSSAnimation, any subsequent change to
238 // matching @keyframes rules or the resolved value of the animation-timing-function property for the target element will not
239 // be reflected in that animation.
240 m_overriddenProperties.add(Property::TimingFunction);
241}
