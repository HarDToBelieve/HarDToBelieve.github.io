40Ref<CSSTransition> CSSTransition::create(const Styleable& owningElement, CSSPropertyID property, MonotonicTime generationTime, const Animation& backingAnimation, const RenderStyle* oldStyle, const RenderStyle& newStyle, Seconds delay, Seconds duration, const RenderStyle& reversingAdjustedStartStyle, double reversingShorteningFactor)
41{
42 ASSERT(oldStyle);
43 auto result = adoptRef(*new CSSTransition(owningElement, property, generationTime, backingAnimation, *oldStyle, newStyle, reversingAdjustedStartStyle, reversingShorteningFactor));
44 result->initialize(oldStyle, newStyle, nullptr);
45 result->setTimingProperties(delay, duration);
46
47 InspectorInstrumentation::didCreateWebAnimation(result.get());
48
49 return result;
50}

70void CSSTransition::setTimingProperties(Seconds delay, Seconds duration)
71{
72 suspendEffectInvalidation();
73
74 // This method is only called from CSSTransition::create() where we're guaranteed to have an effect.
75 auto* animationEffect = effect();
76 ASSERT(animationEffect);
77
78 // In order for CSS Transitions to be seeked backwards, they need to have their fill mode set to backwards
79 // such that the original CSS value applied prior to the transition is used for a negative current time.
80 animationEffect->setFill(FillMode::Backwards);
81 animationEffect->setDelay(delay);
82 animationEffect->setIterationDuration(duration);
83 animationEffect->setTimingFunction(backingAnimation().timingFunction());
84 animationEffect->updateStaticTimingProperties();
85 effectTimingDidChange();
86
87 unsuspendEffectInvalidation();
88}
