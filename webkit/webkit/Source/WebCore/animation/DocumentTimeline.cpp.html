81void DocumentTimeline::detachFromDocument()
82{
83 Ref<DocumentTimeline> protectedThis(*this);
84 if (auto* controller = this->controller())
85 controller->removeTimeline(*this);
86
87 m_pendingAnimationEvents.clear();
88
89 auto& animationsToRemove = m_animations;
90 while (!animationsToRemove.isEmpty())
91 animationsToRemove.first()->remove();
92
93 clearTickScheduleTimer();
94 m_document = nullptr;
95}

161void DocumentTimeline::scheduleAnimationResolution()
162{
163 if (animationsAreSuspended() || m_animationResolutionScheduled || !m_document || !m_document->page())
164 return;
165
166 // We need some relevant animations or pending events to proceed.
167 if (!shouldRunUpdateAnimationsAndSendEventsIgnoringSuspensionState())
168 return;
169
170 m_document->page()->scheduleRenderingUpdate(RenderingUpdateStep::Animations);
171 m_animationResolutionScheduled = true;
172}

184DocumentTimeline::ShouldUpdateAnimationsAndSendEvents DocumentTimeline::documentWillUpdateAnimationsAndSendEvents()
185{
186 // Updating animations and sending events may invalidate the timing of some animations, so we must set the m_animationResolutionScheduled
187 // flag to false prior to running that procedure to allow animation with timing model updates to schedule updates.
188 bool wasAnimationResolutionScheduled = std::exchange(m_animationResolutionScheduled, false);
189
190 if (!wasAnimationResolutionScheduled || animationsAreSuspended() || !shouldRunUpdateAnimationsAndSendEventsIgnoringSuspensionState())
191 return DocumentTimeline::ShouldUpdateAnimationsAndSendEvents::No;
192
193 m_numberOfAnimationTimelineInvalidationsForTesting++;
194
195 // enqueueAnimationEvent() calls scheduleAnimationResolution() to ensure that the "update animations and send events"
196 // procedure is run and enqueued events are dispatched in the next frame. However, events that are enqueued while
197 // this procedure is running should not schedule animation resolution until the event queue has been cleared.
198 m_shouldScheduleAnimationResolutionForNewPendingEvents = false;
199
200 return DocumentTimeline::ShouldUpdateAnimationsAndSendEvents::Yes;
201}

203void DocumentTimeline::documentDidUpdateAnimationsAndSendEvents()
204{
205 applyPendingAcceleratedAnimations();
206
207 if (!m_animationResolutionScheduled)
208 scheduleNextTick();
209}

211bool DocumentTimeline::animationCanBeRemoved(WebAnimation& animation)
212{
213 // https://drafts.csswg.org/web-animations/#removing-replaced-animations
214
215 ASSERT(m_document);
216
217 // - is replaceable, and
218 if (!animation.isReplaceable())
219 return false;
220
221 // - has a replace state of active, and
222 if (animation.replaceState() != WebAnimation::ReplaceState::Active)
223 return false;
224
225 // - has an associated animation effect whose target element is a descendant of doc, and
226 auto* effect = animation.effect();
227 if (!is<KeyframeEffect>(effect))
228 return false;
229
230 auto* keyframeEffect = downcast<KeyframeEffect>(effect);
231 auto target = keyframeEffect->targetStyleable();
232 if (!target || !target->element.isDescendantOf(*m_document))
233 return false;
234
235 HashSet<CSSPropertyID> propertiesToMatch = keyframeEffect->animatedProperties();
236
237 Vector<RefPtr<WebAnimation>> animations;
238 if (auto* keyframeEffectStack = target->keyframeEffectStack()) {
239 for (auto& effect : keyframeEffectStack->sortedEffects()) {
240 if (effect->animation()->isRelevant())
241 animations.append(effect->animation());
242 }
243 }
244
245 for (auto& animationWithHigherCompositeOrder : WTF::makeReversedRange(animations)) {
246 if (&animation == animationWithHigherCompositeOrder)
247 break;
248
249 if (animationWithHigherCompositeOrder && animationWithHigherCompositeOrder->isReplaceable()) {
250 auto* effectWithHigherCompositeOrder = animationWithHigherCompositeOrder->effect();
251 if (is<KeyframeEffect>(effectWithHigherCompositeOrder)) {
252 auto* keyframeEffectWithHigherCompositeOrder = downcast<KeyframeEffect>(effectWithHigherCompositeOrder);
253 for (auto cssPropertyId : keyframeEffectWithHigherCompositeOrder->animatedProperties()) {
254 if (propertiesToMatch.remove(cssPropertyId) && propertiesToMatch.isEmpty())
255 break;
256 }
257 }
258 }
259 }
260
261 return propertiesToMatch.isEmpty();
262}

264void DocumentTimeline::removeReplacedAnimations()
265{
266 // https://drafts.csswg.org/web-animations/#removing-replaced-animations
267
268 Vector<RefPtr<WebAnimation>> animationsToRemove;
269
270 // When asked to remove replaced animations for a Document, doc, then for every animation, animation
271 for (auto& animation : m_allAnimations) {
272 if (animation && animationCanBeRemoved(*animation)) {
273 // perform the following steps:
274 // 1. Set animation's replace state to removed.
275 animation->setReplaceState(WebAnimation::ReplaceState::Removed);
276 // 2. Create an AnimationPlaybackEvent, removeEvent.
277 // 3. Set removeEvent's type attribute to remove.
278 // 4. Set removeEvent's currentTime attribute to the current time of animation.
279 // 5. Set removeEvent's timelineTime attribute to the current time of the timeline with which animation is associated.
280 // 6. If animation has a document for timing, then append removeEvent to its document for timing's pending animation
281 // event queue along with its target, animation. For the scheduled event time, use the result of applying the procedure
282 // to convert timeline time to origin-relative time to the current time of the timeline with which animation is associated.
283 // Otherwise, queue a task to dispatch removeEvent at animation. The task source for this task is the DOM manipulation task source.
284 animation->enqueueAnimationPlaybackEvent(eventNames().removeEvent, animation->currentTime(), currentTime());
285
286 animationsToRemove.append(animation.get());
287 }
288 }
289
290 for (auto& animation : animationsToRemove) {
291 if (auto* timeline = animation->timeline())
292 timeline->removeAnimation(*animation);
293 }
294}

296void DocumentTimeline::transitionDidComplete(RefPtr<CSSTransition> transition)
297{
298 ASSERT(transition);
299 removeAnimation(*transition);
300 if (is<KeyframeEffect>(transition->effect())) {
301 if (auto styleable = downcast<KeyframeEffect>(transition->effect())->targetStyleable()) {
302 auto property = transition->property();
303 if (styleable->hasRunningTransitionForProperty(property))
304 styleable->ensureCompletedTransitionsByProperty().set(property, transition);
305 }
306 }

309void DocumentTimeline::scheduleNextTick()
310{
311 // If we have pending animation events, we need to schedule an update right away.
312 if (!m_pendingAnimationEvents.isEmpty())
313 scheduleAnimationResolution();
314
315 // There is no tick to schedule if we don't have any relevant animations.
316 if (m_animations.isEmpty())
317 return;
318
319 Seconds scheduleDelay = Seconds::infinity();
320
321 for (const auto& animation : m_animations) {
322 if (!animation->isRelevant())
323 continue;
324 auto animationTimeToNextRequiredTick = animation->timeToNextTick();
325 if (animationTimeToNextRequiredTick < animationInterval()) {
326 scheduleAnimationResolution();
327 return;
328 }
329 scheduleDelay = std::min(scheduleDelay, animationTimeToNextRequiredTick);
330 }
331
332 if (scheduleDelay < Seconds::infinity())
333 m_tickScheduleTimer.startOneShot(scheduleDelay);
334}

336bool DocumentTimeline::computeExtentOfAnimation(RenderElement& renderer, LayoutRect& bounds) const
337{
338 auto styleable = Styleable::fromRenderer(renderer);
339 if (!styleable)
340 return false;
341
342 auto* animations = styleable->animations();
343 if (!animations)
344 return false;
345
346 KeyframeEffect* matchingEffect = nullptr;
347 for (const auto& animation : *animations) {
348 auto* effect = animation->effect();
349 if (is<KeyframeEffect>(effect)) {
350 auto* keyframeEffect = downcast<KeyframeEffect>(effect);
351 if (keyframeEffect->animatedProperties().contains(CSSPropertyTransform))
352 matchingEffect = downcast<KeyframeEffect>(effect);
353 }
354 }
355
356 if (matchingEffect)
357 return matchingEffect->computeExtentOfTransformAnimation(bounds);
358
359 return true;
360}

362bool DocumentTimeline::isRunningAnimationOnRenderer(RenderElement& renderer, CSSPropertyID property) const
363{
364 auto styleable = Styleable::fromRenderer(renderer);
365 if (!styleable)
366 return false;
367
368 auto* animations = styleable->animations();
369 if (!animations)
370 return false;
371
372 for (const auto& animation : *animations) {
373 auto playState = animation->playState();
374 if (playState != WebAnimation::PlayState::Running && playState != WebAnimation::PlayState::Paused)
375 continue;
376 auto* effect = animation->effect();
377 if (is<KeyframeEffect>(effect) && downcast<KeyframeEffect>(effect)->animatedProperties().contains(property))
378 return true;
379 }
380
381 return false;
382}

384bool DocumentTimeline::isRunningAcceleratedAnimationOnRenderer(RenderElement& renderer, CSSPropertyID property) const
385{
386 auto styleable = Styleable::fromRenderer(renderer);
387 if (!styleable)
388 return false;
389
390 auto* animations = styleable->animations();
391 if (!animations)
392 return false;
393
394 for (const auto& animation : *animations) {
395 auto playState = animation->playState();
396 if (playState != WebAnimation::PlayState::Running && playState != WebAnimation::PlayState::Paused)
397 continue;
398 auto* effect = animation->effect();
399 if (is<KeyframeEffect>(effect)) {
400 auto* keyframeEffect = downcast<KeyframeEffect>(effect);
401 if (keyframeEffect->isCurrentlyAffectingProperty(property, KeyframeEffect::Accelerated::Yes))
402 return true;
403 }
404 }
405
406 return false;
407}

409std::unique_ptr<RenderStyle> DocumentTimeline::animatedStyleForRenderer(RenderElement& renderer)
410{
411 auto styleable = Styleable::fromRenderer(renderer);
412 if (!styleable)
413 return RenderStyle::clonePtr(renderer.style());
414
415 auto* animations = styleable->animations();
416 if (!animations)
417 return RenderStyle::clonePtr(renderer.style());
418
419 std::unique_ptr<RenderStyle> result;
420 for (const auto& animation : *animations) {
421 if (is<KeyframeEffect>(animation->effect()))
422 downcast<KeyframeEffect>(animation->effect())->getAnimatedStyle(result);
423 }
424
425 if (!result)
426 result = RenderStyle::clonePtr(renderer.style());
427
428 return result;
429}

441void DocumentTimeline::applyPendingAcceleratedAnimations()
442{
443 auto acceleratedAnimationsPendingRunningStateChange = m_acceleratedAnimationsPendingRunningStateChange;
444 m_acceleratedAnimationsPendingRunningStateChange.clear();
445
446 // Animations may fail to run accelerated for reasons private to GraphicsLayerCA. If that happens, and the animation
447 // in question targets a transform-related property, we must prevent all other transform-related animations for this
448 // element to run accelerated since we can't run some transform-related animations accelerated, and some not. To do
449 // this, we keep a list of all KeyframeEffectStack objects containing an effect that failed to start a transform-related
450 // animation so that we can return any transform-related accelerated animation to run non-accelerated.
451 HashSet<KeyframeEffectStack*> effectStacksContainingEffectThatFailedToRunAcceleratedTransformRelatedAnimation;
452
453 bool hasForcedLayout = false;
454 for (auto& animation : acceleratedAnimationsPendingRunningStateChange) {
455 auto* effect = animation->effect();
456 if (!is<KeyframeEffect>(effect))
457 continue;
458
459 auto& keyframeEffect = downcast<KeyframeEffect>(*effect);
460 if (!hasForcedLayout)
461 hasForcedLayout |= keyframeEffect.forceLayoutIfNeeded();
462 auto pendingAccelerationActionResult = keyframeEffect.applyPendingAcceleratedActions();
463 if (pendingAccelerationActionResult.contains(AcceleratedActionApplicationResult::TransformRelatedAnimationCannotBeAccelerated)) {
464 ASSERT(keyframeEffect.targetStyleable());
465 ASSERT(keyframeEffect.targetStyleable()->keyframeEffectStack());
466 effectStacksContainingEffectThatFailedToRunAcceleratedTransformRelatedAnimation.add(keyframeEffect.targetStyleable()->keyframeEffectStack());
467 }
468 }
469
470 for (auto& effectStack : effectStacksContainingEffectThatFailedToRunAcceleratedTransformRelatedAnimation)
471 effectStack->stopAcceleratingTransformRelatedProperties(UseAcceleratedAction::No);
472}

474bool DocumentTimeline::runningAnimationsForRendererAreAllAccelerated(const RenderBoxModelObject& renderer) const
475{
476 auto styleable = Styleable::fromRenderer(renderer);
477 if (!styleable)
478 return false;
479
480 auto* animations = styleable->animations();
481 if (!animations || animations->isEmpty())
482 return false;
483
484 for (const auto& animation : *animations) {
485 if (!animation->isRunningAccelerated())
486 return false;
487 }
488
489 return true;
490}
