45DocumentTimelinesController::DocumentTimelinesController(Document& document)
46 : m_document(document)
47{
48 if (auto* page = document.page()) {
49 if (page->settings().hiddenPageCSSAnimationSuspensionEnabled() && !page->isVisible())
50 suspendAnimations();
51 }
52}

56void DocumentTimelinesController::addTimeline(DocumentTimeline& timeline)
57{
58 m_timelines.add(timeline);
59
60 if (m_isSuspended)
61 timeline.suspendAnimations();
62 else

71void DocumentTimelinesController::detachFromDocument()
72{
73 m_currentTimeClearingTaskCancellationGroup.cancel();
74
75 while (!m_timelines.computesEmpty())

79void DocumentTimelinesController::updateAnimationsAndSendEvents(ReducedResolutionSeconds timestamp)
80{
81 LOG_WITH_STREAM(Animations, stream << "DocumentTimelinesController::updateAnimationsAndSendEvents for time " << timestamp);
82
83 ASSERT(!m_timelines.hasNullReferences());
84
85 // We need to copy m_timelines before iterating over its members since the steps in this procedure may mutate m_timelines.
86 Vector<Ref<DocumentTimeline>> protectedTimelines;
87 for (auto& timeline : m_timelines)
88 protectedTimelines.append(timeline);
89
90 // We need to freeze the current time even if no animation is running.
91 // document.timeline.currentTime may be called from a rAF callback and
92 // it has to match the rAF timestamp.
93 if (!m_isSuspended)
94 cacheCurrentTime(timestamp);
95
96 // 1. Update the current time of all timelines associated with document passing now as the timestamp.
97 Vector<Ref<DocumentTimeline>> timelinesToUpdate;
98 Vector<Ref<WebAnimation>> animationsToRemove;
99 Vector<Ref<CSSTransition>> completedTransitions;
100 for (auto& timeline : protectedTimelines) {
101 auto shouldUpdateAnimationsAndSendEvents = timeline->documentWillUpdateAnimationsAndSendEvents();
102 if (shouldUpdateAnimationsAndSendEvents == DocumentTimeline::ShouldUpdateAnimationsAndSendEvents::No)
103 continue;
104
105 timelinesToUpdate.append(timeline.copyRef());
106
107 for (auto& animation : copyToVector(timeline->relevantAnimations())) {
108 if (animation->timeline() != timeline.ptr()) {
109 ASSERT(!animation->timeline());
110 continue;
111 }
112
113 // This will notify the animation that timing has changed and will call automatically
114 // schedule invalidation if required for this animation.
115 animation->tick();
116
117 if (!animation->isRelevant() && !animation->needsTick())
118 animationsToRemove.append(*animation);
119
120 if (is<CSSTransition>(*animation)) {
121 auto& transition = downcast<CSSTransition>(*animation);
122 if (!transition.needsTick() && transition.playState() == WebAnimation::PlayState::Finished && transition.owningElement())
123 completedTransitions.append(transition);
124 }
125 }
126 }
127
128 if (timelinesToUpdate.isEmpty())
129 return;
130
131 // 2. Remove replaced animations for document.
132 for (auto& timeline : protectedTimelines)
133 timeline->removeReplacedAnimations();
134
135 // 3. Perform a microtask checkpoint.
136 makeRef(m_document)->eventLoop().performMicrotaskCheckpoint();
137
138 // 4. Let events to dispatch be a copy of doc's pending animation event queue.
139 // 5. Clear doc's pending animation event queue.
140 AnimationEvents events;
141 for (auto& timeline : timelinesToUpdate)
142 events.appendVector(timeline->prepareForPendingAnimationEventsDispatch());
143
144 // 6. Perform a stable sort of the animation events in events to dispatch as follows.
145 std::stable_sort(events.begin(), events.end(), [] (const Ref<AnimationEventBase>& lhs, const Ref<AnimationEventBase>& rhs) {
146 // 1. Sort the events by their scheduled event time such that events that were scheduled to occur earlier, sort before events scheduled to occur later
147 // and events whose scheduled event time is unresolved sort before events with a resolved scheduled event time.
148 // 2. Within events with equal scheduled event times, sort by their composite order. FIXME: Need to do this.
149 return lhs->timelineTime() < rhs->timelineTime();
150 });
151
152 // 7. Dispatch each of the events in events to dispatch at their corresponding target using the order established in the previous step.
153 for (auto& event : events)
154 event->target()->dispatchEvent(event);
155
156 // This will cancel any scheduled invalidation if we end up removing all animations.
157 for (auto& animation : animationsToRemove) {
158 // An animation that was initially marked as irrelevant may have changed while
159 // we were sending events, so redo the the check for whether it should be removed.
160 if (auto timeline = animation->timeline()) {
161 if (!animation->isRelevant() && !animation->needsTick())
162 timeline->removeAnimation(animation);
163 }
164 }
165
166 // Now that animations that needed removal have been removed, update the list of completed transitions.
167 // This needs to happen after dealing with the list of animations to remove as the animation may have been
168 // removed from the list of completed transitions otherwise.
169 for (auto& completedTransition : completedTransitions) {
170 if (auto timeline = completedTransition->timeline())
171 downcast<DocumentTimeline>(*timeline).transitionDidComplete(WTFMove(completedTransition));
172 }
173
174 for (auto& timeline : timelinesToUpdate)
175 timeline->documentDidUpdateAnimationsAndSendEvents();
176}

178void DocumentTimelinesController::suspendAnimations()
179{
180 if (m_isSuspended)
181 return;
182
183 if (!m_cachedCurrentTime)
184 m_cachedCurrentTime = liveCurrentTime();
185
186 for (auto& timeline : m_timelines)
187 timeline.suspendAnimations();
188
189 m_isSuspended = true;
190}
191
192void DocumentTimelinesController::resumeAnimations()
193{
194 if (!m_isSuspended)
195 return;
196
197 m_cachedCurrentTime = std::nullopt;
198
199 m_isSuspended = false;
200
201 for (auto& timeline : m_timelines)
202 timeline.resumeAnimations();
203}

215std::optional<Seconds> DocumentTimelinesController::currentTime()
216{
217 if (!m_document.domWindow())
218 return std::nullopt;
219
220 if (!m_cachedCurrentTime)
221 cacheCurrentTime(liveCurrentTime());
222
223 return *m_cachedCurrentTime;
224}

226void DocumentTimelinesController::cacheCurrentTime(ReducedResolutionSeconds newCurrentTime)
227{
228 m_cachedCurrentTime = newCurrentTime;
229 // We want to be sure to keep this time cached until we've both finished running JS and finished updating
230 // animations, so we schedule the invalidation task and register a whenIdle callback on the VM, which will
231 // fire syncronously if no JS is running.
232 m_waitingOnVMIdle = true;
233 if (!m_currentTimeClearingTaskCancellationGroup.hasPendingTask()) {
234 CancellableTask task(m_currentTimeClearingTaskCancellationGroup, std::bind(&DocumentTimelinesController::maybeClearCachedCurrentTime, this));
235 m_document.eventLoop().queueTask(TaskSource::InternalAsyncTask, WTFMove(task));
236 }
237 // We extent the associated Document's lifecycle until the VM became idle since the DocumentTimelinesController
238 // is owned by the Document.
239 m_document.vm().whenIdle([this, protectedDocument = makeRefPtr(m_document)]() {
240 m_waitingOnVMIdle = false;
241 maybeClearCachedCurrentTime();
242 });
243}
244
245void DocumentTimelinesController::maybeClearCachedCurrentTime()
246{
247 // We want to make sure we only clear the cached current time if we're not currently running
248 // JS or waiting on all current animation updating code to have completed. This is so that
249 // we're guaranteed to have a consistent current time reported for all work happening in a given
250 // JS frame or throughout updating animations in WebCore.
251 if (!m_isSuspended && !m_waitingOnVMIdle && !m_currentTimeClearingTaskCancellationGroup.hasPendingTask())
252 m_cachedCurrentTime = std::nullopt;
253}
