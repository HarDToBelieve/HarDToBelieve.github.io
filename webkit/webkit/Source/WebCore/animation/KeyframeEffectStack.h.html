28#include "AnimationList.h"
29#include "CSSPropertyNames.h"
30#include "WebAnimationTypes.h"
31#include <wtf/Vector.h>
32#include <wtf/WeakPtr.h>

34namespace WebCore {
35
36class KeyframeEffect;
37class RenderStyle;
38
39class KeyframeEffectStack {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 explicit KeyframeEffectStack();
43 ~KeyframeEffectStack();
44
45 bool addEffect(KeyframeEffect&);
46 void removeEffect(KeyframeEffect&);
47 bool hasEffects() const { return !m_effects.isEmpty(); }
48 Vector<WeakPtr<KeyframeEffect>> sortedEffects();
49 const AnimationList* cssAnimationList() const { return m_cssAnimationList.get(); }
50 void setCSSAnimationList(RefPtr<const AnimationList>&&);
51 bool isCurrentlyAffectingProperty(CSSPropertyID) const;
52 bool requiresPseudoElement() const;
53 OptionSet<AnimationImpact> applyKeyframeEffects(RenderStyle& targetStyle, const RenderStyle& previousLastStyleChangeEventStyle, const RenderStyle* parentElementStyle);
54 bool hasEffectWithImplicitKeyframes() const;
55
56 void stopAcceleratingTransformRelatedProperties(UseAcceleratedAction);
57
58private:
59 void ensureEffectsAreSorted();
60
61 Vector<WeakPtr<KeyframeEffect>> m_effects;
62 RefPtr<const AnimationList> m_cssAnimationList;
63 bool m_isSorted { true };
64
65};

67} // namespace WebCore
