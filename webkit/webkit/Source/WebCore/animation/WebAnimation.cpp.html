158void WebAnimation::setEffect(RefPtr<AnimationEffect>&& newEffect)
159{
160 // 3.4.3. Setting the target effect of an animation
161 // https://drafts.csswg.org/web-animations-1/#setting-the-target-effect
162
163 // 1. Let old effect be the current target effect of animation, if any.
164 auto oldEffect = m_effect;
165
166 // 2. If new effect is the same object as old effect, abort this procedure.
167 if (newEffect == oldEffect)
168 return;
169
170 // 3. If animation has a pending pause task, reschedule that task to run as soon as animation is ready.
171 if (hasPendingPauseTask())
172 m_timeToRunPendingPauseTask = TimeToRunPendingTask::WhenReady;
173
174 // 4. If animation has a pending play task, reschedule that task to run as soon as animation is ready to play new effect.
175 if (hasPendingPlayTask())
176 m_timeToRunPendingPlayTask = TimeToRunPendingTask::WhenReady;
177
178 // 5. If new effect is not null and if new effect is the target effect of another animation, previous animation, run the
179 // procedure to set the target effect of an animation (this procedure) on previous animation passing null as new effect.
180 if (newEffect && newEffect->animation())
181 newEffect->animation()->setEffect(nullptr);
182
183 // 6. Let the target effect of animation be new effect.
184 // In the case of a declarative animation, we don't want to remove the animation from the relevant maps because
185 // while the effect was set via the API, the element still has a transition or animation set up and we must
186 // not break the timeline-to-animation relationship.
187
188 invalidateEffect();
189
190 // This object could be deleted after clearing the effect relationship.
191 auto protectedThis = makeRef(*this);
192 setEffectInternal(WTFMove(newEffect), isDeclarativeAnimation());
193
194 // 7. Run the procedure to update an animation's finished state for animation with the did seek flag set to false,
195 // and the synchronously notify flag set to false.
196 timingDidChange(DidSeek::No, SynchronouslyNotify::No);
197
198 invalidateEffect();
199}

201void WebAnimation::setEffectInternal(RefPtr<AnimationEffect>&& newEffect, bool doNotRemoveFromTimeline)
202{
203 if (m_effect == newEffect)
204 return;
205
206 auto oldEffect = std::exchange(m_effect, WTFMove(newEffect));
207
208 std::optional<const Styleable> previousTarget = is<KeyframeEffect>(oldEffect) ? downcast<KeyframeEffect>(oldEffect.get())->targetStyleable() : std::nullopt;
209 std::optional<const Styleable> newTarget = is<KeyframeEffect>(m_effect) ? downcast<KeyframeEffect>(m_effect.get())->targetStyleable() : std::nullopt;
210
211 // Update the effect-to-animation relationships and the timeline's animation map.
212 if (oldEffect) {
213 oldEffect->setAnimation(nullptr);
214 if (!doNotRemoveFromTimeline && previousTarget && previousTarget != newTarget)
215 previousTarget->animationWasRemoved(*this);
216 updateRelevance();
217 }
218
219 if (m_effect) {
220 m_effect->setAnimation(this);
221 if (newTarget && previousTarget != newTarget)
222 newTarget->animationWasAdded(*this);
223 }
224
225 InspectorInstrumentation::didSetWebAnimationEffect(*this);
226}

228void WebAnimation::setTimeline(RefPtr<AnimationTimeline>&& timeline)
229{
230 // 3.4.1. Setting the timeline of an animation
231 // https://drafts.csswg.org/web-animations-1/#setting-the-timeline
232
233 // 2. If new timeline is the same object as old timeline, abort this procedure.
234 if (timeline == m_timeline)
235 return;
236
237 // 4. If the animation start time of animation is resolved, make animation's hold time unresolved.
238 if (m_startTime)
239 m_holdTime = std::nullopt;
240
241 if (is<KeyframeEffect>(m_effect)) {
242 if (auto target = downcast<KeyframeEffect>(m_effect.get())->targetStyleable()) {
243 // In the case of a declarative animation, we don't want to remove the animation from the relevant maps because
244 // while the timeline was set via the API, the element still has a transition or animation set up and we must
245 // not break the relationship.
246 if (!isDeclarativeAnimation())
247 target->animationWasRemoved(*this);
248 target->animationWasAdded(*this);
249 }
250 }
251
252 // This object could be deleted after clearing the timeline relationship.
253 auto protectedThis = makeRef(*this);
254 setTimelineInternal(WTFMove(timeline));
255
256 setSuspended(is<DocumentTimeline>(m_timeline) && downcast<DocumentTimeline>(*m_timeline).animationsAreSuspended());
257
258 // 5. Run the procedure to update an animation's finished state for animation with the did seek flag set to false,
259 // and the synchronously notify flag set to false.
260 timingDidChange(DidSeek::No, SynchronouslyNotify::No);
261
262 invalidateEffect();
263}

310void WebAnimation::setStartTime(std::optional<Seconds> newStartTime)
311{
312 // 3.4.6 The procedure to set the start time of animation, animation, to new start time, is as follows:
313 // https://drafts.csswg.org/web-animations/#setting-the-start-time-of-an-animation
314
315 // 1. Let timeline time be the current time value of the timeline that animation is associated with. If
316 // there is no timeline associated with animation or the associated timeline is inactive, let the timeline
317 // time be unresolved.
318 auto timelineTime = m_timeline ? m_timeline->currentTime() : std::nullopt;
319
320 // 2. If timeline time is unresolved and new start time is resolved, make animation's hold time unresolved.
321 if (!timelineTime && newStartTime)
322 m_holdTime = std::nullopt;
323
324 // 3. Let previous current time be animation's current time.
325 auto previousCurrentTime = currentTime();
326
327 // 4. Apply any pending playback rate on animation.
328 applyPendingPlaybackRate();
329
330 // 5. Set animation's start time to new start time.
331 m_startTime = newStartTime;
332
333 // 6. Update animation's hold time based on the first matching condition from the following,
334 if (newStartTime) {
335 // If new start time is resolved,
336 // If animation's playback rate is not zero, make animation's hold time unresolved.
337 if (m_playbackRate)
338 m_holdTime = std::nullopt;
339 } else {
340 // Otherwise (new start time is unresolved),
341 // Set animation's hold time to previous current time even if previous current time is unresolved.
342 m_holdTime = previousCurrentTime;
343 }
344
345 // 7. If animation has a pending play task or a pending pause task, cancel that task and resolve animation's current ready promise with animation.
346 if (pending()) {
347 m_timeToRunPendingPauseTask = TimeToRunPendingTask::NotScheduled;
348 m_timeToRunPendingPlayTask = TimeToRunPendingTask::NotScheduled;
349 m_readyPromise->resolve(*this);
350 }
351
352 // 8. Run the procedure to update an animation's finished state for animation with the did seek flag set to true, and the synchronously notify flag set to false.
353 timingDidChange(DidSeek::Yes, SynchronouslyNotify::No);
354
355 invalidateEffect();
356}

378std::optional<Seconds> WebAnimation::currentTime(RespectHoldTime respectHoldTime, std::optional<Seconds> startTime) const
379{
380 // 3.4.4. The current time of an animation
381 // https://drafts.csswg.org/web-animations-1/#the-current-time-of-an-animation
382
383 // The current time is calculated from the first matching condition from below:
384
385 // If the animation's hold time is resolved, the current time is the animation's hold time.
386 if (respectHoldTime == RespectHoldTime::Yes && m_holdTime)
387 return m_holdTime;
388
389 // If any of the following are true:
390 // 1. the animation has no associated timeline, or
391 // 2. the associated timeline is inactive, or
392 // 3. the animation's start time is unresolved.
393 // The current time is an unresolved time value.
394 if (!m_timeline || !m_timeline->currentTime() || !m_startTime)
395 return std::nullopt;
396
397 // Otherwise, current time = (timeline time - start time) * playback rate
398 return (*m_timeline->currentTime() - startTime.value_or(*m_startTime)) * m_playbackRate;
399}

401ExceptionOr<void> WebAnimation::silentlySetCurrentTime(std::optional<Seconds> seekTime)
402{
403 LOG_WITH_STREAM(Animations, stream << "WebAnimation " << this << " silentlySetCurrentTime " << seekTime);
404
405 // 3.4.5. Setting the current time of an animation
406 // https://drafts.csswg.org/web-animations-1/#setting-the-current-time-of-an-animation
407
408 // 1. If seek time is an unresolved time value, then perform the following steps.
409 if (!seekTime) {
410 // 1. If the current time is resolved, then throw a TypeError.
411 if (currentTime())
412 return Exception { TypeError };
413 // 2. Abort these steps.
414 return { };
415 }
416
417 // 2. Update either animation's hold time or start time as follows:
418 // If any of the following conditions are true:
419 // - animation's hold time is resolved, or
420 // - animation's start time is unresolved, or
421 // - animation has no associated timeline or the associated timeline is inactive, or
422 // - animation's playback rate is 0,
423 // Set animation's hold time to seek time.
424 // Otherwise, set animation's start time to the result of evaluating timeline time - (seek time / playback rate)
425 // where timeline time is the current time value of timeline associated with animation.
426 if (m_holdTime || !m_startTime || !m_timeline || !m_timeline->currentTime() || !m_playbackRate)
427 m_holdTime = seekTime;
428 else
429 m_startTime = m_timeline->currentTime().value() - (seekTime.value() / m_playbackRate);
430
431 // 3. If animation has no associated timeline or the associated timeline is inactive, make animation's start time unresolved.
432 if (!m_timeline || !m_timeline->currentTime())
433 m_startTime = std::nullopt;
434
435 // 4. Make animation's previous current time unresolved.
436 m_previousCurrentTime = std::nullopt;
437
438 return { };
439}

441ExceptionOr<void> WebAnimation::setCurrentTime(std::optional<Seconds> seekTime)
442{
443 LOG_WITH_STREAM(Animations, stream << "WebAnimation " << this << " setCurrentTime " << seekTime);
444
445 // 3.4.5. Setting the current time of an animation
446 // https://drafts.csswg.org/web-animations-1/#setting-the-current-time-of-an-animation
447
448 // 1. Run the steps to silently set the current time of animation to seek time.
449 auto silentResult = silentlySetCurrentTime(seekTime);
450 if (silentResult.hasException())
451 return silentResult.releaseException();
452
453 // 2. If animation has a pending pause task, synchronously complete the pause operation by performing the following steps:
454 if (hasPendingPauseTask()) {
455 // 1. Set animation's hold time to seek time.
456 m_holdTime = seekTime;
457 // 2. Apply any pending playback rate to animation.
458 applyPendingPlaybackRate();
459 // 3. Make animation's start time unresolved.
460 m_startTime = std::nullopt;
461 // 4. Cancel the pending pause task.
462 m_timeToRunPendingPauseTask = TimeToRunPendingTask::NotScheduled;
463 // 5. Resolve animation's current ready promise with animation.
464 m_readyPromise->resolve(*this);
465 }
466
467 // 3. Run the procedure to update an animation's finished state for animation with the did seek flag set to true, and the synchronously notify flag set to false.
468 timingDidChange(DidSeek::Yes, SynchronouslyNotify::No);
469
470 if (m_effect)
471 m_effect->animationDidChangeTimingProperties();
472
473 invalidateEffect();
474
475 return { };
476}

485void WebAnimation::setPlaybackRate(double newPlaybackRate)
486{
487 // 3.4.17.1. Updating the playback rate of an animation
488 // https://drafts.csswg.org/web-animations-1/#updating-the-playback-rate-of-an-animation
489
490 // 1. Clear any pending playback rate on animation.
491 m_pendingPlaybackRate = std::nullopt;
492
493 // 2. Let previous time be the value of the current time of animation before changing the playback rate.
494 auto previousTime = currentTime();
495
496 // 3. Set the playback rate to new playback rate.
497 m_playbackRate = newPlaybackRate;
498

507void WebAnimation::updatePlaybackRate(double newPlaybackRate)
508{
509 // https://drafts.csswg.org/web-animations/#seamlessly-update-the-playback-rate
510
511 // The procedure to seamlessly update the playback rate an animation, animation, to new playback rate preserving its current time is as follows:
512
513 // 1. Let previous play state be animation's play state.
514 // Note: It is necessary to record the play state before updating animation's effective playback rate since, in the following logic,
515 // we want to immediately apply the pending playback rate of animation if it is currently finished regardless of whether or not it will
516 // still be finished after we apply the pending playback rate.
517 auto previousPlayState = playState();
518
519 // 2. Let animation's pending playback rate be new playback rate.
520 m_pendingPlaybackRate = newPlaybackRate;
521
522 // 3. Perform the steps corresponding to the first matching condition from below:
523 if (pending()) {
524 // If animation has a pending play task or a pending pause task,
525 // Abort these steps.
526 // Note: The different types of pending tasks will apply the pending playback rate when they run so there is no further action required in this case.
527 return;
528 }
529
530 if (previousPlayState == PlayState::Idle || previousPlayState == PlayState::Paused) {
531 // If previous play state is idle or paused,
532 // Apply any pending playback rate on animation.
533 applyPendingPlaybackRate();
534 } else if (previousPlayState == PlayState::Finished) {
535 // If previous play state is finished,
536 // 1. Let the unconstrained current time be the result of calculating the current time of animation substituting an unresolved time value for the hold time.
537 auto unconstrainedCurrentTime = currentTime(RespectHoldTime::No);
538 // 2. Let animation's start time be the result of evaluating the following expression:
539 // timeline time - (unconstrained current time / pending playback rate)
540 // Where timeline time is the current time value of the timeline associated with animation.
541 // If pending playback rate is zero, let animation's start time be timeline time.
542
543 // If timeline is inactive abort these steps.
544 if (!m_timeline->currentTime())
545 return;
546
547 auto newStartTime = m_timeline->currentTime().value();
548 if (m_pendingPlaybackRate)
549 newStartTime -= (unconstrainedCurrentTime.value() / m_pendingPlaybackRate.value());
550 m_startTime = newStartTime;
551 // 3. Apply any pending playback rate on animation.
552 applyPendingPlaybackRate();
553 // 4. Run the procedure to update an animation's finished state for animation with the did seek flag set to false, and the synchronously notify flag set to false.
554 timingDidChange(DidSeek::No, SynchronouslyNotify::No);
555
556 invalidateEffect();
557 } else {
558 // Otherwise,
559 // Run the procedure to play an animation for animation with the auto-rewind flag set to false.
560 play(AutoRewind::No);
561 }
562
563 if (m_effect)
564 m_effect->animationDidChangeTimingProperties();
565}

567void WebAnimation::applyPendingPlaybackRate()
568{
569 // https://drafts.csswg.org/web-animations/#apply-any-pending-playback-rate
570
571 // 1. If animation does not have a pending playback rate, abort these steps.
572 if (!m_pendingPlaybackRate)
573 return;
574
575 // 2. Set animation's playback rate to its pending playback rate.
576 m_playbackRate = m_pendingPlaybackRate.value();
577
578 // 3. Clear animation's pending playback rate.
579 m_pendingPlaybackRate = std::nullopt;
580}

582auto WebAnimation::playState() const -> PlayState
583{
584 // 3.5.19 Play states
585 // https://drafts.csswg.org/web-animations/#play-states
586
587 // The play state of animation, animation, at a given moment is the state corresponding to the
588 // first matching condition from the following:
589
590 // The current time of animation is unresolved, and animation does not have either a pending
591 // play task or a pending pause task,
592 // idle
593 auto animationCurrentTime = currentTime();
594 if (!animationCurrentTime && !pending())
595 return PlayState::Idle;
596
597 // Animation has a pending pause task, or both the start time of animation is unresolved and it does not
598 // have a pending play task,
599 // paused
600 if (hasPendingPauseTask() || (!m_startTime && !hasPendingPlayTask()))
601 return PlayState::Paused;
602
603 // For animation, current time is resolved and either of the following conditions are true:
604 // animation's effective playback rate > 0 and current time target effect end; or
605 // animation's effective playback rate < 0 and current time 0,
606 // finished
607 if (animationCurrentTime && ((effectivePlaybackRate() > 0 && (*animationCurrentTime + timeEpsilon) >= effectEndTime()) || (effectivePlaybackRate() < 0 && (*animationCurrentTime - timeEpsilon) <= 0_s)))
608 return PlayState::Finished;
609
610 // Otherwise running
611 return PlayState::Running;
612}

621void WebAnimation::cancel()
622{
623 LOG_WITH_STREAM(Animations, stream << "WebAnimation " << this << " cancel() (current time is " << currentTime() << ")");
624
625 // 3.4.16. Canceling an animation
626 // https://drafts.csswg.org/web-animations-1/#canceling-an-animation-section
627 //
628 // An animation can be canceled which causes the current time to become unresolved hence removing any effects caused by the target effect.
629 //
630 // The procedure to cancel an animation for animation is as follows:
631 //
632 // 1. If animation's play state is not idle, perform the following steps:
633 if (playState() != PlayState::Idle) {
634 // 1. Run the procedure to reset an animation's pending tasks on animation.
635 resetPendingTasks();
636
637 // 2. Reject the current finished promise with a DOMException named "AbortError".
638 // 3. Set the [[PromiseIsHandled]] internal slot of the current finished promise to true.
639 if (!m_finishedPromise->isFulfilled())
640 m_finishedPromise->reject(Exception { AbortError }, RejectAsHandled::Yes);
641
642 // 4. Let current finished promise be a new (pending) Promise object.
643 m_finishedPromise = makeUniqueRef<FinishedPromise>(*this, &WebAnimation::finishedPromiseResolve);
644
645 // 5. Create an AnimationPlaybackEvent, cancelEvent.
646 // 6. Set cancelEvent's type attribute to cancel.
647 // 7. Set cancelEvent's currentTime to null.
648 // 8. Let timeline time be the current time of the timeline with which animation is associated. If animation is not associated with an
649 // active timeline, let timeline time be n unresolved time value.
650 // 9. Set cancelEvent's timelineTime to timeline time. If timeline time is unresolved, set it to null.
651 // 10. If animation has a document for timing, then append cancelEvent to its document for timing's pending animation event queue along
652 // with its target, animation. If animation is associated with an active timeline that defines a procedure to convert timeline times
653 // to origin-relative time, let the scheduled event time be the result of applying that procedure to timeline time. Otherwise, the
654 // scheduled event time is an unresolved time value.
655 // Otherwise, queue a task to dispatch cancelEvent at animation. The task source for this task is the DOM manipulation task source.
656 enqueueAnimationPlaybackEvent(eventNames().cancelEvent, std::nullopt, m_timeline ? m_timeline->currentTime() : std::nullopt);
657 }
658
659 // 2. Make animation's hold time unresolved.
660 m_holdTime = std::nullopt;
661
662 // 3. Make animation's start time unresolved.
663 m_startTime = std::nullopt;
664

704 // https://drafts.csswg.org/web-animations-1/#reset-an-animations-pending-tasks

730ExceptionOr<void> WebAnimation::finish()
731{
732 LOG_WITH_STREAM(Animations, stream << "WebAnimation " << this << " finish (current time is " << currentTime() << ")");
733
734 // 3.4.15. Finishing an animation
735 // https://drafts.csswg.org/web-animations-1/#finishing-an-animation-section
736
737 // An animation can be advanced to the natural end of its current playback direction by using the procedure to finish an animation for animation defined below:
738 //
739 // 1. If animation's effective playback rate is zero, or if animation's effective playback rate > 0 and target effect end is infinity, throw an InvalidStateError and abort these steps.
740 if (!effectivePlaybackRate() || (effectivePlaybackRate() > 0 && effectEndTime() == Seconds::infinity()))
741 return Exception { InvalidStateError };
742
743 // 2. Apply any pending playback rate to animation.
744 applyPendingPlaybackRate();
745
746 // 3. Set limit as follows:
747 // If animation playback rate > 0, let limit be target effect end.
748 // Otherwise, let limit be zero.
749 auto limit = m_playbackRate > 0 ? effectEndTime() : 0_s;
750
751 // 4. Silently set the current time to limit.
752 silentlySetCurrentTime(limit);
753
754 // 5. If animation's start time is unresolved and animation has an associated active timeline, let the start time be the result of
755 // evaluating timeline time - (limit / playback rate) where timeline time is the current time value of the associated timeline.
756 if (!m_startTime && m_timeline && m_timeline->currentTime())
757 m_startTime = m_timeline->currentTime().value() - (limit / m_playbackRate);
758
759 // 6. If there is a pending pause task and start time is resolved,
760 if (hasPendingPauseTask() && m_startTime) {
761 // 1. Let the hold time be unresolved.
762 m_holdTime = std::nullopt;
763 // 2. Cancel the pending pause task.
764 m_timeToRunPendingPauseTask = TimeToRunPendingTask::NotScheduled;
765 // 3. Resolve the current ready promise of animation with animation.
766 m_readyPromise->resolve(*this);
767 }
768
769 // 7. If there is a pending play task and start time is resolved, cancel that task and resolve the current ready promise of animation with animation.
770 if (hasPendingPlayTask() && m_startTime) {
771 m_timeToRunPendingPlayTask = TimeToRunPendingTask::NotScheduled;
772 m_readyPromise->resolve(*this);
773 }
774
775 // 8. Run the procedure to update an animation's finished state animation with the did seek flag set to true, and the synchronously notify flag set to true.
776 timingDidChange(DidSeek::Yes, SynchronouslyNotify::Yes);
777
778 invalidateEffect();
779
780 return { };
781}

803void WebAnimation::updateFinishedState(DidSeek didSeek, SynchronouslyNotify synchronouslyNotify)
804{
805 // 3.4.14. Updating the finished state
806 // https://drafts.csswg.org/web-animations-1/#updating-the-finished-state
807
808 // 1. Let the unconstrained current time be the result of calculating the current time substituting an unresolved time value
809 // for the hold time if did seek is false. If did seek is true, the unconstrained current time is equal to the current time.
810 auto unconstrainedCurrentTime = currentTime(didSeek == DidSeek::Yes ? RespectHoldTime::Yes : RespectHoldTime::No);
811 auto endTime = effectEndTime();
812
813 // 2. If all three of the following conditions are true,
814 // - the unconstrained current time is resolved, and
815 // - animation's start time is resolved, and
816 // - animation does not have a pending play task or a pending pause task,
817 if (unconstrainedCurrentTime && m_startTime && !pending()) {
818 // then update animation's hold time based on the first matching condition for animation from below, if any:
819 if (m_playbackRate > 0 && unconstrainedCurrentTime >= endTime) {
820 // If animation playback rate > 0 and unconstrained current time is greater than or equal to target effect end,
821 // If did seek is true, let the hold time be the value of unconstrained current time.
822 if (didSeek == DidSeek::Yes)
823 m_holdTime = unconstrainedCurrentTime;
824 // If did seek is false, let the hold time be the maximum value of previous current time and target effect end. If the previous current time is unresolved, let the hold time be target effect end.
825 else if (!m_previousCurrentTime)
826 m_holdTime = endTime;
827 else
828 m_holdTime = std::max(m_previousCurrentTime.value(), endTime);
829 } else if (m_playbackRate < 0 && unconstrainedCurrentTime <= 0_s) {
830 // If animation playback rate < 0 and unconstrained current time is less than or equal to 0,
831 // If did seek is true, let the hold time be the value of unconstrained current time.
832 if (didSeek == DidSeek::Yes)
833 m_holdTime = unconstrainedCurrentTime;
834 // If did seek is false, let the hold time be the minimum value of previous current time and zero. If the previous current time is unresolved, let the hold time be zero.
835 else if (!m_previousCurrentTime)
836 m_holdTime = 0_s;
837 else
838 m_holdTime = std::min(m_previousCurrentTime.value(), 0_s);
839 } else if (m_playbackRate && m_timeline && m_timeline->currentTime()) {
840 // If animation playback rate 0, and animation is associated with an active timeline,
841 // Perform the following steps:
842 // 1. If did seek is true and the hold time is resolved, let animation's start time be equal to the result of evaluating timeline time - (hold time / playback rate)
843 // where timeline time is the current time value of timeline associated with animation.
844 if (didSeek == DidSeek::Yes && m_holdTime)
845 m_startTime = m_timeline->currentTime().value() - (m_holdTime.value() / m_playbackRate);
846 // 2. Let the hold time be unresolved.
847 m_holdTime = std::nullopt;
848 }
849 }
850
851 // 3. Set the previous current time of animation be the result of calculating its current time.
852 m_previousCurrentTime = currentTime();
853
854 // 4. Let current finished state be true if the play state of animation is finished. Otherwise, let it be false.
855 auto currentFinishedState = playState() == PlayState::Finished;
856
857 // 5. If current finished state is true and the current finished promise is not yet resolved, perform the following steps:
858 if (currentFinishedState && !m_finishedPromise->isFulfilled()) {
859 if (synchronouslyNotify == SynchronouslyNotify::Yes) {
860 // If synchronously notify is true, cancel any queued microtask to run the finish notification steps for this animation,
861 // and run the finish notification steps immediately.
862 m_finishNotificationStepsMicrotaskPending = false;
863 finishNotificationSteps();
864 } else if (!m_finishNotificationStepsMicrotaskPending) {
865 // Otherwise, if synchronously notify is false, queue a microtask to run finish notification steps for animation unless there
866 // is already a microtask queued to run those steps for animation.
867 m_finishNotificationStepsMicrotaskPending = true;
868 if (auto* context = scriptExecutionContext()) {
869 context->eventLoop().queueMicrotask([this, protectedThis = makeRef(*this)] {
870 if (m_finishNotificationStepsMicrotaskPending) {
871 m_finishNotificationStepsMicrotaskPending = false;
872 finishNotificationSteps();
873 }
874 });
875 }
876 }
877 }
878
879 // 6. If current finished state is false and animation's current finished promise is already resolved, set animation's current
880 // finished promise to a new (pending) Promise object.
881 if (!currentFinishedState && m_finishedPromise->isFulfilled())
882 m_finishedPromise = makeUniqueRef<FinishedPromise>(*this, &WebAnimation::finishedPromiseResolve);
883
884 updateRelevance();
885}
886
887void WebAnimation::finishNotificationSteps()
888{
889 // 3.4.14. Updating the finished state
890 // https://drafts.csswg.org/web-animations-1/#finish-notification-steps
891
892 // Let finish notification steps refer to the following procedure:
893 // 1. If animation's play state is not equal to finished, abort these steps.
894 if (playState() != PlayState::Finished)
895 return;
896
897 // 2. Resolve animation's current finished promise object with animation.
898 m_finishedPromise->resolve(*this);
899
900 // 3. Create an AnimationPlaybackEvent, finishEvent.
901 // 4. Set finishEvent's type attribute to finish.
902 // 5. Set finishEvent's currentTime attribute to the current time of animation.
903 // 6. Set finishEvent's timelineTime attribute to the current time of the timeline with which animation is associated.
904 // If animation is not associated with a timeline, or the timeline is inactive, let timelineTime be null.
905 // 7. If animation has a document for timing, then append finishEvent to its document for timing's pending animation event
906 // queue along with its target, animation. For the scheduled event time, use the result of converting animation's target
907 // effect end to an origin-relative time.
908 // Otherwise, queue a task to dispatch finishEvent at animation. The task source for this task is the DOM manipulation task source.
909 enqueueAnimationPlaybackEvent(eventNames().finishEvent, currentTime(), m_timeline ? m_timeline->currentTime() : std::nullopt);
910
911 if (is<KeyframeEffect>(m_effect)) {
912 if (auto target = makeRefPtr(downcast<KeyframeEffect>(*m_effect).target())) {
913 if (auto* page = target->document().page())
914 page->chrome().client().animationDidFinishForElement(*target);
915 }
916 }
917}

924ExceptionOr<void> WebAnimation::play(AutoRewind autoRewind)
925{
926 LOG_WITH_STREAM(Animations, stream << "WebAnimation " << this << " play(autoRewind " << (autoRewind == AutoRewind::Yes) << ") (current time is " << currentTime() << ")");
927
928 // 3.4.10. Playing an animation
929 // https://drafts.csswg.org/web-animations-1/#play-an-animation
930
931 auto localTime = currentTime();
932 auto endTime = effectEndTime();
933
934 // 1. Let aborted pause be a boolean flag that is true if animation has a pending pause task, and false otherwise.
935 bool abortedPause = hasPendingPauseTask();
936
937 // 2. Let has pending ready promise be a boolean flag that is initially false.
938 bool hasPendingReadyPromise = false;
939
940 // 3. Perform the steps corresponding to the first matching condition from the following, if any:
941 if (effectivePlaybackRate() > 0 && autoRewind == AutoRewind::Yes && (!localTime || localTime.value() < 0_s || (localTime.value() + timeEpsilon) >= endTime)) {
942 // If animation's effective playback rate > 0, the auto-rewind flag is true and either animation's:
943 // - current time is unresolved, or
944 // - current time < zero, or
945 // - current time target effect end,
946 // Set animation's hold time to zero.
947 m_holdTime = 0_s;
948 } else if (effectivePlaybackRate() < 0 && autoRewind == AutoRewind::Yes && (!localTime || localTime.value() <= 0_s || localTime.value() > endTime)) {
949 // If animation's effective playback rate < 0, the auto-rewind flag is true and either animation's:
950 // - current time is unresolved, or
951 // - current time zero, or
952 // - current time > target effect end
953 // If target effect end is positive infinity, throw an InvalidStateError and abort these steps.
954 if (endTime == Seconds::infinity())
955 return Exception { InvalidStateError };
956 m_holdTime = endTime;
957 } else if (!effectivePlaybackRate() && !localTime) {
958 // If animation's effective playback rate = 0 and animation's current time is unresolved,
959 // Set animation's hold time to zero.
960 m_holdTime = 0_s;
961 }
962
963 // 4. If animation has a pending play task or a pending pause task,
964 if (pending()) {
965 // 1. Cancel that task.
966 m_timeToRunPendingPauseTask = TimeToRunPendingTask::NotScheduled;
967 m_timeToRunPendingPlayTask = TimeToRunPendingTask::NotScheduled;
968 // 2. Set has pending ready promise to true.
969 hasPendingReadyPromise = true;
970 }
971
972 // 5. If the following three conditions are all satisfied:
973 // - animation's hold time is unresolved, and
974 // - aborted pause is false, and
975 // - animation does not have a pending playback rate,
976 // abort this procedure.
977 if (!m_holdTime && !abortedPause && !m_pendingPlaybackRate)
978 return { };
979
980 // 6. If animation's hold time is resolved, let its start time be unresolved.
981 if (m_holdTime)
982 m_startTime = std::nullopt;
983
984 // 7. If has pending ready promise is false, let animation's current ready promise be a new (pending) Promise object.
985 if (!hasPendingReadyPromise)
986 m_readyPromise = makeUniqueRef<ReadyPromise>(*this, &WebAnimation::readyPromiseResolve);
987
988 // 8. Schedule a task to run as soon as animation is ready.
989 m_timeToRunPendingPlayTask = TimeToRunPendingTask::WhenReady;
990
991 // 9. Run the procedure to update an animation's finished state for animation with the did seek flag set to false, and the synchronously notify flag set to false.
992 timingDidChange(DidSeek::No, SynchronouslyNotify::No);
993
994 invalidateEffect();
995
996 if (m_effect)
997 m_effect->animationDidPlay();
998
999 return { };
1000}
1001
1002void WebAnimation::runPendingPlayTask()
1003{
1004 LOG_WITH_STREAM(Animations, stream << "WebAnimation " << this << " runPendingPlayTask (current time is " << currentTime() << ")");
1005
1006 // 3.4.10. Playing an animation, step 8.
1007 // https://drafts.csswg.org/web-animations-1/#play-an-animation
1008
1009 m_timeToRunPendingPlayTask = TimeToRunPendingTask::NotScheduled;
1010
1011 // 1. Assert that at least one of animation's start time or hold time is resolved.
1012 ASSERT(m_startTime || m_holdTime);
1013
1014 // 2. Let ready time be the time value of the timeline associated with animation at the moment when animation became ready.
1015 auto readyTime = m_timeline->currentTime();
1016
1017 // 3. Perform the steps corresponding to the first matching condition below, if any:
1018 if (m_holdTime) {
1019 // If animation's hold time is resolved,
1020 // 1. Apply any pending playback rate on animation.
1021 applyPendingPlaybackRate();
1022 // 2. Let new start time be the result of evaluating ready time - hold time / animation playback rate for animation.
1023 // If the animation playback rate is zero, let new start time be simply ready time.
1024 // FIXME: Implementation cannot guarantee an active timeline at the point of this async dispatch.
1025 // Subsequently, the resulting readyTime value can be null. Unify behavior between C++17 and
1026 // C++14 builds (the latter using WTF's Optional) and avoid null Optional dereferencing
1027 // by defaulting to a Seconds(0) value. See https://bugs.webkit.org/show_bug.cgi?id=186189.
1028 auto newStartTime = readyTime.value_or(0_s);
1029 if (m_playbackRate)
1030 newStartTime -= m_holdTime.value() / m_playbackRate;
1031 // 3. Set the start time of animation to new start time.
1032 m_startTime = newStartTime;
1033 // 4. If animation's playback rate is not 0, make animation's hold time unresolved.
1034 if (m_playbackRate)
1035 m_holdTime = std::nullopt;
1036 } else if (m_startTime && m_pendingPlaybackRate) {
1037 // If animation's start time is resolved and animation has a pending playback rate,
1038 // 1. Let current time to match be the result of evaluating (ready time - start time) playback rate for animation.
1039 auto currentTimeToMatch = (readyTime.value_or(0_s) - m_startTime.value()) * m_playbackRate;
1040 // 2. Apply any pending playback rate on animation.
1041 applyPendingPlaybackRate();
1042 // 3. If animation's playback rate is zero, let animation's hold time be current time to match.
1043 if (m_playbackRate)
1044 m_holdTime = currentTimeToMatch;
1045 // 4. Let new start time be the result of evaluating ready time - current time to match / playback rate for animation.
1046 // If the playback rate is zero, let new start time be simply ready time.
1047 auto newStartTime = readyTime.value_or(0_s);
1048 if (m_playbackRate)
1049 newStartTime -= currentTimeToMatch / m_playbackRate;
1050 // 5. Set the start time of animation to new start time.
1051 m_startTime = newStartTime;
1052 }
1053
1054 // 4. Resolve animation's current ready promise with animation.
1055 if (!m_readyPromise->isFulfilled())
1056 m_readyPromise->resolve(*this);
1057
1058 // 5. Run the procedure to update an animation's finished state for animation with the did seek flag set to false, and the synchronously notify flag set to false.
1059 timingDidChange(DidSeek::No, SynchronouslyNotify::No, Silently::Yes);
1060
1061 invalidateEffect();
1062}

1067
1068 // 3.4.11. Pausing an animation
1069 // https://drafts.csswg.org/web-animations-1/#pause-an-animation
1070
1071 // 1. If animation has a pending pause task, abort these steps.
1072 if (hasPendingPauseTask())
1073 return { };
1074
1075 // 2. If the play state of animation is paused, abort these steps.
1076 if (playState() == PlayState::Paused)
1077 return { };
1078
1079 auto localTime = currentTime();
1080
1081 // 3. If the animation's current time is unresolved, perform the steps according to the first matching condition from below:
1082 if (!localTime) {
1083 if (m_playbackRate >= 0) {
1084 // If animation's playback rate is 0, let animation's hold time be zero.
1085 m_holdTime = 0_s;
1086 } else if (effectEndTime() == Seconds::infinity()) {

1088 return Exception { InvalidStateError };
1089 } else {
1090 // Otherwise, let animation's hold time be target effect end.
1091 m_holdTime = effectEndTime();
1092 }
1093 }
1094
1095 // 4. Let has pending ready promise be a boolean flag that is initially false.
1096 bool hasPendingReadyPromise = false;
1097
1098 // 5. If animation has a pending play task, cancel that task and let has pending ready promise be true.
1099 if (hasPendingPlayTask()) {
1100 m_timeToRunPendingPlayTask = TimeToRunPendingTask::NotScheduled;
1101 hasPendingReadyPromise = true;
1102 }
1103
1104 // 6. If has pending ready promise is false, set animation's current ready promise to a new (pending) Promise object.
1105 if (!hasPendingReadyPromise)
1106 m_readyPromise = makeUniqueRef<ReadyPromise>(*this, &WebAnimation::readyPromiseResolve);
1107
1108 // 7. Schedule a task to be executed at the first possible moment after the user agent has performed any processing necessary
1109 // to suspend the playback of animation's target effect, if any.
1110 m_timeToRunPendingPauseTask = TimeToRunPendingTask::ASAP;
1111
1112 // 8. Run the procedure to update an animation's finished state for animation with the did seek flag set to false, and the synchronously notify flag set to false.
1113 timingDidChange(DidSeek::No, SynchronouslyNotify::No);
1114
1115 invalidateEffect();
1116
1117 return { };
1118}

1162{

1190 // 4. Make animation's start time unresolved.

1197 // 6. Run the procedure to update an animation's finished state for animation with the did seek flag set to false, and the

1214void WebAnimation::tick()
1215{
1216 m_hasScheduledEventsDuringTick = false;
1217 updateFinishedState(DidSeek::No, SynchronouslyNotify::Yes);
1218 m_shouldSkipUpdatingFinishedStateWhenResolving = true;
1219
1220 // Run pending tasks, if any.

1299bool WebAnimation::computeRelevance()
1300{
1301 // To be listed in getAnimations() an animation needs a target effect which is current or in effect.
1302 if (!m_effect)
1303 return false;
1304
1305 if (m_replaceState == ReplaceState::Removed)
1306 return false;
1307
1308 auto timing = m_effect->getBasicTiming();
1309
1310 // An animation effect is in effect if its active time is not unresolved.
1311 if (timing.activeTime)
1312 return true;
1313
1314 // An animation effect is current if either of the following conditions is true:
1315 // - the animation effect is in the before phase, or
1316 // - the animation effect is in play.
1317
1318 // An animation effect is in play if all of the following conditions are met:
1319 // - the animation effect is in the active phase, and
1320 // - the animation effect is associated with an animation that is not finished.
1321 return timing.phase == AnimationEffectPhase::Before || (timing.phase == AnimationEffectPhase::Active && playState() != PlayState::Finished);
1322}

1324bool WebAnimation::isReplaceable() const
1325{
1326 // An animation is replaceable if all of the following conditions are true:
1327 // https://drafts.csswg.org/web-animations/#removing-replaced-animations
1328
1329 // The existence of the animation is not prescribed by markup. That is, it is not a CSS animation with an owning element,
1330 // nor a CSS transition with an owning element.
1331 if (isDeclarativeAnimation() && downcast<DeclarativeAnimation>(this)->owningElement())
1332 return false;
1333
1334 // The animation's play state is finished.
1335 if (playState() != PlayState::Finished)
1336 return false;
1337
1338 // The animation's replace state is not removed.
1339 if (m_replaceState == ReplaceState::Removed)
1340 return false;
1341
1342 // The animation is associated with a monotonically increasing timeline.
1343 if (!m_timeline)
1344 return false;
1345
1346 // The animation has an associated target effect.
1347 if (!m_effect)
1348 return false;
1349
1350 // The target effect associated with the animation is in effect.
1351 if (!m_effect->getBasicTiming().activeTime)
1352 return false;
1353
1354 // The target effect has an associated target element.
1355 if (!is<KeyframeEffect>(m_effect) || !downcast<KeyframeEffect>(m_effect.get())->target())
1356 return false;
1357
1358 return true;
1359}

1361void WebAnimation::persist()
1362{
1363 auto previousReplaceState = std::exchange(m_replaceState, ReplaceState::Persisted);
1364
1365 if (previousReplaceState == ReplaceState::Removed && m_timeline) {
1366 if (is<KeyframeEffect>(m_effect)) {
1367 auto& keyframeEffect = downcast<KeyframeEffect>(*m_effect);
1368 auto styleable = keyframeEffect.targetStyleable();

1371 }

1375ExceptionOr<void> WebAnimation::commitStyles()
1376{
1377 // https://drafts.csswg.org/web-animations-1/#commit-computed-styles
1378
1379 // 1. Let targets be the set of all effect targets for animation effects associated with animation.
1380 auto* effect = is<KeyframeEffect>(m_effect) ? downcast<KeyframeEffect>(m_effect.get()) : nullptr;
1381 auto* target = effect ? effect->target() : nullptr;
1382
1383 // 2. For each target in targets:
1384 //
1385 // 2.1 If target is not an element capable of having a style attribute (for example, it is a pseudo-element or is an element in a
1386 // document format for which style attributes are not defined) throw a "NoModificationAllowedError" DOMException and abort these steps.
1387 if (!is<StyledElement>(target) || effect->targetsPseudoElement())
1388 return Exception { NoModificationAllowedError };
1389
1390 auto& styledElement = downcast<StyledElement>(*target);
1391
1392 // 2.2 If, after applying any pending style changes, target is not being rendered, throw an "InvalidStateError" DOMException and abort these steps.
1393 styledElement.document().updateStyleIfNeeded();
1394 auto* renderer = styledElement.renderer();
1395 if (!renderer)
1396 return Exception { InvalidStateError };
1397
1398 // 2.3 Let inline style be the result of getting the CSS declaration block corresponding to targets style attribute. If target does not have a style
1399 // attribute, let inline style be a new empty CSS declaration block with the readonly flag unset and owner node set to target.
1400
1401 // 2.4 Let targeted properties be the set of physical longhand properties that are a target property for at least one animation effect associated with
1402 // animation whose effect target is target.
1403
1404 auto& style = renderer->style();
1405 auto computedStyleExtractor = ComputedStyleExtractor(&styledElement);
1406 auto inlineStyle = styledElement.document().createCSSStyleDeclaration();
1407 inlineStyle->setCssText(styledElement.getAttribute("style"));
1408
1409 auto& keyframeStack = styledElement.ensureKeyframeEffectStack(PseudoId::None);
1410
1411 // 2.5 For each property, property, in targeted properties:
1412 for (auto property : effect->animatedProperties()) {
1413 // 1. Let partialEffectStack be a copy of the effect stack for property on target.
1414 // 2. If animation's replace state is removed, add all animation effects associated with animation whose effect target is target and which include
1415 // property as a target property to partialEffectStack.
1416 // 3. Remove from partialEffectStack any animation effects whose associated animation has a higher composite order than animation.
1417 // 4. Let effect value be the result of calculating the result of partialEffectStack for property using target's computed style (see 5.4.3 Calculating
1418 // the result of an effect stack).
1419 // 5. Set a CSS declaration property for effect value in inline style.
1420 // 6. Update style attribute for inline style.
1421
1422 // We actually perform those steps in a different way: instead of building a copy of the effect stack and then removing stuff, we iterate through the
1423 // effect stack and stop when we've found this animation's effect or when we've found an effect associated with an animation with a higher composite order.
1424 auto animatedStyle = RenderStyle::clonePtr(style);
1425 for (const auto& effectInStack : keyframeStack.sortedEffects()) {
1426 if (effectInStack->animation() != this && !compareAnimationsByCompositeOrder(*effectInStack->animation(), *this))
1427 break;
1428 if (effectInStack->animatedProperties().contains(property))
1429 effectInStack->animation()->resolve(*animatedStyle, nullptr);
1430 if (effectInStack->animation() == this)
1431 break;
1432 }
1433 if (m_replaceState == ReplaceState::Removed)
1434 effect->animation()->resolve(*animatedStyle, nullptr);
1435 if (auto cssValue = computedStyleExtractor.valueForPropertyInStyle(*animatedStyle, property, renderer))
1436 inlineStyle->setPropertyInternal(property, cssValue->cssText(), false);
1437 }
1438
1439 styledElement.setAttribute("style", inlineStyle->cssText());
1440
1441 return { };
1442}

1444Seconds WebAnimation::timeToNextTick() const
1445{
1446 if (pending())
1447 return 0_s;
1448
1449 // If we're not running, or time is not advancing for this animation, there's no telling when we'll end.
1450 auto playbackRate = effectivePlaybackRate();
1451 if (playState() != PlayState::Running || !playbackRate)
1452 return Seconds::infinity();
1453
1454 ASSERT(effect());
1455 return effect()->timeToNextTick() / playbackRate;
1456}
