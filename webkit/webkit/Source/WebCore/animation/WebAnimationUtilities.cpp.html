41static bool compareDeclarativeAnimationOwningElementPositionsInDocumentTreeOrder(const Styleable& a, const Styleable& b)
42{
43 // We should not ever be calling this function with two Elements that are the same. If that were the case,
44 // then comparing objects of this kind would yield inconsistent results when comparing A == B and B == A.
45 // As such, this function should be called with std::stable_sort().
46 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(a != b);
47
48 // With regard to pseudo-elements, the sort order is as follows:
49 //
50 // - element
51 // - ::marker
52 // - ::before
53 // - any other pseudo-elements not mentioned specifically in this list, sorted in ascending order by the Unicode codepoints that make up each selector
54 // - ::after
55 // - element children
56 enum SortingIndex : uint8_t { NotPseudo, Marker, Before, FirstLetter, FirstLine, Highlight, Scrollbar, Selection, After, Other };
57 auto sortingIndex = [](PseudoId pseudoId) -> SortingIndex {
58 switch (pseudoId) {
59 case PseudoId::None:
60 return NotPseudo;
61 case PseudoId::Marker:
62 return Marker;
63 case PseudoId::Before:
64 return Before;
65 case PseudoId::FirstLetter:
66 return FirstLetter;
67 case PseudoId::FirstLine:
68 return FirstLine;
69 case PseudoId::Highlight:
70 return Highlight;
71 case PseudoId::Scrollbar:
72 return Scrollbar;
73 case PseudoId::Selection:
74 return Selection;
75 case PseudoId::After:
76 return After;
77 default:
78 ASSERT_NOT_REACHED();
79 return Other;
80 }
81 };
82
83 auto& aReferenceElement = a.element;
84 int aSortingIndex = sortingIndex(a.pseudoId);
85
86 auto& bReferenceElement = b.element;
87 int bSortingIndex = sortingIndex(b.pseudoId);
88
89 if (&aReferenceElement == &bReferenceElement) {
90 ASSERT(aSortingIndex != bSortingIndex);
91 return aSortingIndex < bSortingIndex;
92 }
93 return aReferenceElement.compareDocumentPosition(bReferenceElement) & Node::DOCUMENT_POSITION_FOLLOWING;
94}

96static bool compareCSSTransitions(const CSSTransition& a, const CSSTransition& b)
97{
98 ASSERT(a.owningElement());
99 ASSERT(b.owningElement());
100 auto& aOwningElement = a.owningElement();
101 auto& bOwningElement = b.owningElement();
102
103 // If the owning element of A and B differs, sort A and B by tree order of their corresponding owning elements.
104 if (*aOwningElement != *bOwningElement)
105 return compareDeclarativeAnimationOwningElementPositionsInDocumentTreeOrder(*aOwningElement, *bOwningElement);
106
107 // Otherwise, if A and B have different transition generation values, sort by their corresponding transition generation in ascending order.
108 if (a.generationTime() != b.generationTime())
109 return a.generationTime() < b.generationTime();
110
111 // Otherwise, sort A and B in ascending order by the Unicode codepoints that make up the expanded transition property name of each transition
112 // (i.e. without attempting case conversion and such that ‘-moz-column-width’ sorts before ‘column-width’).
113 return a.transitionProperty().utf8() < b.transitionProperty().utf8();
114}

116static bool compareCSSAnimations(const CSSAnimation& a, const CSSAnimation& b)
117{
118 // https://drafts.csswg.org/css-animations-2/#animation-composite-order
119 ASSERT(a.owningElement());
120 ASSERT(b.owningElement());
121 auto& aOwningElement = a.owningElement();
122 auto& bOwningElement = b.owningElement();
123
124 // If the owning element of A and B differs, sort A and B by tree order of their corresponding owning elements.
125 if (*aOwningElement != *bOwningElement)
126 return compareDeclarativeAnimationOwningElementPositionsInDocumentTreeOrder(*aOwningElement, *bOwningElement);
127
128 // Sort A and B based on their position in the computed value of the animation-name property of the (common) owning element.
129 auto* cssAnimationList = aOwningElement->ensureKeyframeEffectStack().cssAnimationList();
130 ASSERT(cssAnimationList);
131 ASSERT(!cssAnimationList->isEmpty());
132
133 auto& aBackingAnimation = a.backingAnimation();
134 auto& bBackingAnimation = b.backingAnimation();
135 for (size_t i = 0; i < cssAnimationList->size(); ++i) {
136 auto& animation = cssAnimationList->animation(i);
137 if (animation == aBackingAnimation)
138 return true;
139 if (animation == bBackingAnimation)
140 return false;
141 }
142
143 // We should have found either of those CSS animations in the CSS animations list.
144 RELEASE_ASSERT_NOT_REACHED();
145}

147bool compareAnimationsByCompositeOrder(const WebAnimation& a, const WebAnimation& b)
148{
149 // We should not ever be calling this function with two WebAnimation objects that are the same. If that were the case,
150 // then comparing objects of this kind would yield inconsistent results when comparing A == B and B == A. As such,
151 // this function should be called with std::stable_sort().
152 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(&a != &b);
153
154 bool aHasOwningElement = is<DeclarativeAnimation>(a) && downcast<DeclarativeAnimation>(a).owningElement();
155 bool bHasOwningElement = is<DeclarativeAnimation>(b) && downcast<DeclarativeAnimation>(b).owningElement();
156
157 // CSS Transitions sort first.
158 bool aIsCSSTransition = aHasOwningElement && is<CSSTransition>(a);
159 bool bIsCSSTransition = bHasOwningElement && is<CSSTransition>(b);
160 if (aIsCSSTransition || bIsCSSTransition) {
161 if (aIsCSSTransition == bIsCSSTransition)
162 return compareCSSTransitions(downcast<CSSTransition>(a), downcast<CSSTransition>(b));
163 return !bIsCSSTransition;
164 }
165
166 // CSS Animations sort next.
167 bool aIsCSSAnimation = aHasOwningElement && is<CSSAnimation>(a);
168 bool bIsCSSAnimation = bHasOwningElement && is<CSSAnimation>(b);
169 if (aIsCSSAnimation || bIsCSSAnimation) {
170 if (aIsCSSAnimation == bIsCSSAnimation)
171 return compareCSSAnimations(downcast<CSSAnimation>(a), downcast<CSSAnimation>(b));
172 return !bIsCSSAnimation;
173 }
174
175 // JS-originated animations sort last based on their position in the global animation list.
176 // https://drafts.csswg.org/web-animations-1/#animation-composite-order
177 RELEASE_ASSERT(a.globalPosition() != b.globalPosition());
178 return a.globalPosition() < b.globalPosition();
179}

181String pseudoIdAsString(PseudoId pseudoId)
182{
183 static NeverDestroyed<const String> after(MAKE_STATIC_STRING_IMPL("::after"));
184 static NeverDestroyed<const String> before(MAKE_STATIC_STRING_IMPL("::before"));
185 static NeverDestroyed<const String> firstLetter(MAKE_STATIC_STRING_IMPL("::first-letter"));
186 static NeverDestroyed<const String> firstLine(MAKE_STATIC_STRING_IMPL("::first-line"));
187 static NeverDestroyed<const String> highlight(MAKE_STATIC_STRING_IMPL("::highlight"));
188 static NeverDestroyed<const String> marker(MAKE_STATIC_STRING_IMPL("::marker"));
189 static NeverDestroyed<const String> selection(MAKE_STATIC_STRING_IMPL("::selection"));
190 static NeverDestroyed<const String> scrollbar(MAKE_STATIC_STRING_IMPL("::scrollbar"));
191 switch (pseudoId) {
192 case PseudoId::After:
193 return after;
194 case PseudoId::Before:
195 return before;
196 case PseudoId::FirstLetter:
197 return firstLetter;
198 case PseudoId::FirstLine:
199 return firstLine;
200 case PseudoId::Highlight:
201 return highlight;
202 case PseudoId::Marker:
203 return marker;
204 case PseudoId::Selection:
205 return selection;
206 case PseudoId::Scrollbar:
207 return scrollbar;
208 default:
209 return emptyString();
210 }
211}
