43Ref<CachedModuleScriptLoader> CachedModuleScriptLoader::create(ModuleScriptLoaderClient& client, DeferredPromise& promise, CachedScriptFetcher& scriptFetcher, RefPtr<ModuleFetchParameters>&& parameters)
44{
45 return adoptRef(*new CachedModuleScriptLoader(client, promise, scriptFetcher, WTFMove(parameters)));

48CachedModuleScriptLoader::CachedModuleScriptLoader(ModuleScriptLoaderClient& client, DeferredPromise& promise, CachedScriptFetcher& scriptFetcher, RefPtr<ModuleFetchParameters>&& parameters)

53CachedModuleScriptLoader::~CachedModuleScriptLoader()
54{
55 if (m_cachedScript) {
56 m_cachedScript->removeClient(*this);
57 m_cachedScript = nullptr;
58 }
59}

61bool CachedModuleScriptLoader::load(Document& document, URL&& sourceURL)
62{
63 ASSERT(m_promise);
64 ASSERT(!m_cachedScript);
65 String integrity = m_parameters ? m_parameters->integrity() : String { };
66 m_cachedScript = scriptFetcher().requestModuleScript(document, sourceURL, WTFMove(integrity));
67 if (!m_cachedScript)
68 return false;
69 m_sourceURL = WTFMove(sourceURL);
70
71 // If the content is already cached, this immediately calls notifyFinished.
72 m_cachedScript->addClient(*this);
73 return true;
74}
75
76void CachedModuleScriptLoader::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
77{
78 ASSERT_UNUSED(resource, &resource == m_cachedScript);
79 ASSERT(m_cachedScript);
80 ASSERT(m_promise);
81
82 Ref<CachedModuleScriptLoader> protectedThis(*this);
83 if (m_client)
84 m_client->notifyFinished(*this, WTFMove(m_sourceURL), m_promise.releaseNonNull());
85
86 // Remove the client after calling notifyFinished to keep the data buffer in
87 // CachedResource alive while notifyFinished processes the resource.
88 m_cachedScript->removeClient(*this);
89 m_cachedScript = nullptr;
90}
