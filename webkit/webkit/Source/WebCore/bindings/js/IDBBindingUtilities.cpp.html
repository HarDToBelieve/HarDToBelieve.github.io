65static bool get(JSGlobalObject& lexicalGlobalObject, JSValue object, const String& keyPathElement, JSValue& result)
66{
67 VM& vm = lexicalGlobalObject.vm();
68 auto scope = DECLARE_THROW_SCOPE(vm);
69
70 if (object.isString() && keyPathElement == "length") {
71 result = jsNumber(asString(object)->length());
72 return true;
73 }
74 if (!object.isObject())
75 return false;
76
77 auto* obj = asObject(object);
78 Identifier identifier = Identifier::fromString(vm, keyPathElement);
79 if (obj->inherits<JSArray>(vm) && keyPathElement == "length") {
80 result = obj->get(&lexicalGlobalObject, identifier);
81 RETURN_IF_EXCEPTION(scope, false);
82 return true;
83 }
84 if (obj->inherits<JSBlob>(vm) && (keyPathElement == "size" || keyPathElement == "type")) {
85 if (keyPathElement == "size") {
86 result = jsNumber(jsCast<JSBlob*>(obj)->wrapped().size());
87 return true;
88 }
89 if (keyPathElement == "type") {
90 result = jsString(vm, jsCast<JSBlob*>(obj)->wrapped().type());
91 return true;
92 }
93 }
94 if (obj->inherits<JSFile>(vm)) {
95 if (keyPathElement == "name") {
96 result = jsString(vm, jsCast<JSFile*>(obj)->wrapped().name());
97 return true;
98 }
99 if (keyPathElement == "lastModified") {
100 result = jsNumber(jsCast<JSFile*>(obj)->wrapped().lastModified());
101 return true;
102 }
103 if (keyPathElement == "lastModifiedDate") {
104 result = jsDate(lexicalGlobalObject, jsCast<JSFile*>(obj)->wrapped().lastModified());
105 return true;
106 }
107 }
108
109 PropertyDescriptor descriptor;
110 bool found = obj->getOwnPropertyDescriptor(&lexicalGlobalObject, identifier, descriptor);
111 RETURN_IF_EXCEPTION(scope, false);
112 if (!found)
113 return false;
114 if (!descriptor.enumerable())
115 return false;
116
117 result = obj->get(&lexicalGlobalObject, identifier);
118 RETURN_IF_EXCEPTION(scope, false);
119 return true;
120}

128static bool set(VM& vm, JSValue& object, const String& keyPathElement, JSValue jsValue)
129{
130 if (!canSet(object, keyPathElement))
131 return false;
132 Identifier identifier = Identifier::fromString(vm, keyPathElement);
133 asObject(object)->putDirect(vm, identifier, jsValue);
134 return true;
135}

137JSValue toJS(JSGlobalObject& lexicalGlobalObject, JSGlobalObject& globalObject, IDBKey* key)
138{
139 if (!key) {
140 // This must be undefined, not null.
141 // Spec: http://dvcs.w3.org/hg/IndexedDB/raw-file/tip/Overview.html#idl-def-IDBKeyRange
142 return jsUndefined();
143 }
144
145 VM& vm = lexicalGlobalObject.vm();
146 Locker<JSLock> locker(vm.apiLock());
147 auto scope = DECLARE_THROW_SCOPE(vm);
148
149 switch (key->type()) {
150 case IndexedDB::KeyType::Array: {
151 auto& inArray = key->array();
152 unsigned size = inArray.size();
153 auto outArray = constructEmptyArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), size);
154 RETURN_IF_EXCEPTION(scope, JSValue());
155 for (size_t i = 0; i < size; ++i) {
156 outArray->putDirectIndex(&lexicalGlobalObject, i, toJS(lexicalGlobalObject, globalObject, inArray.at(i).get()));
157 RETURN_IF_EXCEPTION(scope, JSValue());
158 }
159 return outArray;
160 }
161 case IndexedDB::KeyType::Binary: {
162 auto* data = key->binary().data();
163 if (!data) {
164 ASSERT_NOT_REACHED();
165 return jsNull();
166 }
167
168 auto arrayBuffer = ArrayBuffer::create(data->data(), data->size());
169 Structure* structure = globalObject.arrayBufferStructure(arrayBuffer->sharingMode());
170 if (!structure)
171 return jsNull();
172
173 return JSArrayBuffer::create(lexicalGlobalObject.vm(), structure, WTFMove(arrayBuffer));
174 }
175 case IndexedDB::KeyType::String:
176 return jsStringWithCache(vm, key->string());
177 case IndexedDB::KeyType::Date:
178 // FIXME: This should probably be toJS<IDLDate>(...) as per:
179 // http://w3c.github.io/IndexedDB/#request-convert-a-key-to-a-value
180 RELEASE_AND_RETURN(scope, toJS<IDLNullable<IDLDate>>(lexicalGlobalObject, key->date()));
181 case IndexedDB::KeyType::Number:
182 return jsNumber(key->number());
183 case IndexedDB::KeyType::Min:
184 case IndexedDB::KeyType::Max:
185 case IndexedDB::KeyType::Invalid:
186 ASSERT_NOT_REACHED();
187 return jsUndefined();
188 }
189
190 ASSERT_NOT_REACHED();
191 return jsUndefined();
192}

194static const size_t maximumDepth = 2000;
195
196static RefPtr<IDBKey> createIDBKeyFromValue(JSGlobalObject& lexicalGlobalObject, JSValue value, Vector<JSArray*>& stack)
197{
198 VM& vm = lexicalGlobalObject.vm();
199 auto scope = DECLARE_THROW_SCOPE(vm);
200
201 if (value.isNumber() && !std::isnan(value.asNumber()))
202 return IDBKey::createNumber(value.asNumber());
203
204 if (value.isString()) {
205 auto string = asString(value)->value(&lexicalGlobalObject);
206 RETURN_IF_EXCEPTION(scope, { });
207 return IDBKey::createString(WTFMove(string));
208 }
209
210 if (value.inherits<DateInstance>(vm)) {
211 auto dateValue = valueToDate(lexicalGlobalObject, value);
212 RETURN_IF_EXCEPTION(scope, { });
213 if (!std::isnan(dateValue))
214 return IDBKey::createDate(dateValue);
215 }
216
217 if (value.isObject()) {
218 JSObject* object = asObject(value);
219 if (auto* array = jsDynamicCast<JSArray*>(vm, object)) {
220 size_t length = array->length();
221
222 if (stack.contains(array))
223 return nullptr;
224
225 if (stack.size() >= maximumDepth)
226 return nullptr;
227
228 stack.append(array);
229
230 Vector<RefPtr<IDBKey>> subkeys;
231 for (size_t i = 0; i < length; i++) {
232 JSValue item = array->getIndex(&lexicalGlobalObject, i);
233 RETURN_IF_EXCEPTION(scope, { });
234 RefPtr<IDBKey> subkey = createIDBKeyFromValue(lexicalGlobalObject, item, stack);
235 RETURN_IF_EXCEPTION(scope, { });
236 if (!subkey)
237 subkeys.append(IDBKey::createInvalid());
238 else
239 subkeys.append(subkey);
240 }
241
242 stack.removeLast();
243 return IDBKey::createArray(subkeys);
244 }
245
246 if (auto* arrayBuffer = jsDynamicCast<JSArrayBuffer*>(vm, value))
247 return IDBKey::createBinary(*arrayBuffer);
248
249 if (auto* arrayBufferView = jsDynamicCast<JSArrayBufferView*>(vm, value))
250 return IDBKey::createBinary(*arrayBufferView);
251 }
252 return nullptr;
253}

255static Ref<IDBKey> createIDBKeyFromValue(JSGlobalObject& lexicalGlobalObject, JSValue value)
256{
257 Vector<JSArray*> stack;
258 RefPtr<IDBKey> key = createIDBKeyFromValue(lexicalGlobalObject, value, stack);
259 if (key)
260 return *key;
261 return IDBKey::createInvalid();

264static JSValue getNthValueOnKeyPath(JSGlobalObject& lexicalGlobalObject, JSValue rootValue, const Vector<String>& keyPathElements, size_t index)
265{
266 JSValue currentValue(rootValue);
267 ASSERT(index <= keyPathElements.size());
268 for (size_t i = 0; i < index; i++) {
269 JSValue parentValue(currentValue);
270 if (!get(lexicalGlobalObject, parentValue, keyPathElements[i], currentValue))
271 return jsUndefined();
272 }
273 return currentValue;
274}

276static RefPtr<IDBKey> internalCreateIDBKeyFromScriptValueAndKeyPath(JSGlobalObject& lexicalGlobalObject, JSValue value, const String& keyPath)
277{
278 Vector<String> keyPathElements;
279 IDBKeyPathParseError error;
280 IDBParseKeyPath(keyPath, keyPathElements, error);
281 ASSERT(error == IDBKeyPathParseError::None);
282
283 JSValue jsValue = value;
284 jsValue = getNthValueOnKeyPath(lexicalGlobalObject, jsValue, keyPathElements, keyPathElements.size());
285 if (jsValue.isUndefined())
286 return nullptr;
287 return createIDBKeyFromValue(lexicalGlobalObject, jsValue);
288}

290static JSValue ensureNthValueOnKeyPath(JSGlobalObject& lexicalGlobalObject, JSValue rootValue, const Vector<String>& keyPathElements, size_t index)
291{
292 JSValue currentValue(rootValue);
293
294 ASSERT(index <= keyPathElements.size());
295 for (size_t i = 0; i < index; i++) {
296 JSValue parentValue(currentValue);
297 const String& keyPathElement = keyPathElements[i];
298 if (!get(lexicalGlobalObject, parentValue, keyPathElement, currentValue)) {
299 JSObject* object = constructEmptyObject(&lexicalGlobalObject);
300 if (!set(lexicalGlobalObject.vm(), parentValue, keyPathElement, JSValue(object)))
301 return jsUndefined();
302 currentValue = JSValue(object);
303 }
304 }
305
306 return currentValue;
307}

309static bool canInjectNthValueOnKeyPath(JSGlobalObject& lexicalGlobalObject, JSValue rootValue, const Vector<String>& keyPathElements, size_t index)
310{
311 if (!rootValue.isObject())
312 return false;
313
314 JSValue currentValue(rootValue);
315
316 ASSERT(index <= keyPathElements.size());
317 for (size_t i = 0; i <= index; ++i) {
318 JSValue parentValue(currentValue);
319 const String& keyPathElement = keyPathElements[i];
320 if (!get(lexicalGlobalObject, parentValue, keyPathElement, currentValue))
321 return canSet(parentValue, keyPathElement);
322 }
323 return true;
324}

326bool injectIDBKeyIntoScriptValue(JSGlobalObject& lexicalGlobalObject, const IDBKeyData& keyData, JSValue value, const IDBKeyPath& keyPath)
327{
328 LOG(IndexedDB, "injectIDBKeyIntoScriptValue");
329
330 ASSERT(WTF::holds_alternative<String>(keyPath));
331
332 Vector<String> keyPathElements;
333 IDBKeyPathParseError error;
334 IDBParseKeyPath(WTF::get<String>(keyPath), keyPathElements, error);
335 ASSERT(error == IDBKeyPathParseError::None);
336
337 if (keyPathElements.isEmpty())
338 return false;
339
340 JSValue parent = ensureNthValueOnKeyPath(lexicalGlobalObject, value, keyPathElements, keyPathElements.size() - 1);
341 if (parent.isUndefined())
342 return false;
343
344 auto key = keyData.maybeCreateIDBKey();
345 if (!key)
346 return false;
347
348 // Do not set if object already has the correct property value.
349 JSValue existingKey;
350 if (get(lexicalGlobalObject, parent, keyPathElements.last(), existingKey) && !key->compare(createIDBKeyFromValue(lexicalGlobalObject, existingKey)))
351 return true;
352 if (!set(lexicalGlobalObject.vm(), parent, keyPathElements.last(), toJS(lexicalGlobalObject, lexicalGlobalObject, key.get())))
353 return false;
354
355 return true;
356}

359RefPtr<IDBKey> maybeCreateIDBKeyFromScriptValueAndKeyPath(JSGlobalObject& lexicalGlobalObject, JSValue value, const IDBKeyPath& keyPath)
360{
361 if (WTF::holds_alternative<Vector<String>>(keyPath)) {
362 auto& array = WTF::get<Vector<String>>(keyPath);
363 Vector<RefPtr<IDBKey>> result;
364 result.reserveInitialCapacity(array.size());
365 for (auto& string : array) {
366 RefPtr<IDBKey> key = internalCreateIDBKeyFromScriptValueAndKeyPath(lexicalGlobalObject, value, string);
367 if (!key)
368 return nullptr;
369 result.uncheckedAppend(WTFMove(key));
370 }
371 return IDBKey::createArray(WTFMove(result));
372 }
373
374 return internalCreateIDBKeyFromScriptValueAndKeyPath(lexicalGlobalObject, value, WTF::get<String>(keyPath));
375}

377bool canInjectIDBKeyIntoScriptValue(JSGlobalObject& lexicalGlobalObject, JSValue scriptValue, const IDBKeyPath& keyPath)
378{
379 LOG(StorageAPI, "canInjectIDBKeyIntoScriptValue");
380
381 ASSERT(WTF::holds_alternative<String>(keyPath));
382 Vector<String> keyPathElements;
383 IDBKeyPathParseError error;
384 IDBParseKeyPath(WTF::get<String>(keyPath), keyPathElements, error);
385 ASSERT(error == IDBKeyPathParseError::None);
386
387 if (!keyPathElements.size())
388 return false;
389
390 return canInjectNthValueOnKeyPath(lexicalGlobalObject, scriptValue, keyPathElements, keyPathElements.size() - 1);
391}

393static JSValue deserializeIDBValueToJSValue(JSGlobalObject& lexicalGlobalObject, JSC::JSGlobalObject& globalObject, const IDBValue& value)
394{
395 // FIXME: I think it's peculiar to use undefined to mean "null data" and null to mean "empty data".
396 // But I am not changing this at the moment because at least some callers are specifically checking isUndefined.
397
398 if (!value.data().data())
399 return jsUndefined();
400
401 auto& data = *value.data().data();
402 if (data.isEmpty())
403 return jsNull();
404
405 auto serializedValue = SerializedScriptValue::createFromWireBytes(Vector<uint8_t>(data));
406
407 lexicalGlobalObject.vm().apiLock().lock();
408 Vector<RefPtr<MessagePort>> messagePorts;
409 JSValue result = serializedValue->deserialize(lexicalGlobalObject, &globalObject, messagePorts, value.blobURLs(), value.blobFilePaths(), SerializationErrorMode::NonThrowing);
410 lexicalGlobalObject.vm().apiLock().unlock();
411
412 return result;
413}

439static Vector<IDBKeyData> createKeyPathArray(JSGlobalObject& lexicalGlobalObject, JSValue value, const IDBIndexInfo& info, std::optional<IDBKeyPath> objectStoreKeyPath, const IDBKeyData& objectStoreKey)
440{
441 auto visitor = WTF::makeVisitor([&](const String& string) -> Vector<IDBKeyData> {
442 // Value doesn't contain auto-generated key, so we need to manually add key if it is possibly auto-generated.
443 if (objectStoreKeyPath && WTF::holds_alternative<String>(objectStoreKeyPath.value()) && IDBKeyPath(string) == objectStoreKeyPath.value())
444 return { objectStoreKey };
445
446 auto idbKey = internalCreateIDBKeyFromScriptValueAndKeyPath(lexicalGlobalObject, value, string);
447 if (!idbKey)
448 return { };
449
450 Vector<IDBKeyData> keys;
451 if (info.multiEntry() && idbKey->type() == IndexedDB::Array) {
452 for (auto& key : idbKey->array())
453 keys.append(key.get());
454 } else
455 keys.append(idbKey.get());
456 return keys;
457 }, [&](const Vector<String>& vector) -> Vector<IDBKeyData> {
458 Vector<IDBKeyData> keys;
459 for (auto& entry : vector) {
460 if (objectStoreKeyPath && WTF::holds_alternative<String>(objectStoreKeyPath.value()) && IDBKeyPath(entry) == objectStoreKeyPath.value())
461 keys.append(objectStoreKey);
462 else {
463 auto key = internalCreateIDBKeyFromScriptValueAndKeyPath(lexicalGlobalObject, value, entry);
464 if (!key || !key->isValid())
465 return { };
466 keys.append(key.get());
467 }
468 }
469 return keys;
470 });
471
472 return WTF::visit(visitor, info.keyPath());
473}

484IndexIDToIndexKeyMap generateIndexKeyMapForValue(JSC::JSGlobalObject& lexicalGlobalObject, const IDBObjectStoreInfo& storeInfo, const IDBKeyData& key, const IDBValue& value)
485{
486 auto& indexMap = storeInfo.indexMap();
487 auto indexCount = indexMap.size();
488 if (!indexCount)
489 return IndexIDToIndexKeyMap { };
490
491 JSLockHolder locker(lexicalGlobalObject.vm());
492 auto jsValue = deserializeIDBValueToJSValue(lexicalGlobalObject, value);
493 if (jsValue.isUndefinedOrNull())
494 return IndexIDToIndexKeyMap { };
495
496 IndexIDToIndexKeyMap indexKeys;
497 indexKeys.reserveInitialCapacity(indexCount);
498
499 for (const auto& entry : indexMap) {
500 IndexKey indexKey;
501 generateIndexKeyForValue(lexicalGlobalObject, entry.value, jsValue, indexKey, storeInfo.keyPath(), key);
502
503 if (indexKey.isNull())
504 continue;
505
506 indexKeys.add(entry.key, WTFMove(indexKey));
507 }
508
509 return indexKeys;
510}

512std::optional<JSC::JSValue> deserializeIDBValueWithKeyInjection(JSGlobalObject& lexicalGlobalObject, const IDBValue& value, const IDBKeyData& key, const std::optional<IDBKeyPath>& keyPath)
513{
514 auto jsValue = deserializeIDBValueToJSValue(lexicalGlobalObject, value);
515 if (jsValue.isUndefined() || !keyPath || !WTF::holds_alternative<String>(keyPath.value()) || !isIDBKeyPathValid(keyPath.value()))
516 return jsValue;
517
518 JSLockHolder locker(lexicalGlobalObject.vm());
519 if (!injectIDBKeyIntoScriptValue(lexicalGlobalObject, key, jsValue, keyPath.value())) {
520 auto throwScope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
521 propagateException(lexicalGlobalObject, throwScope, Exception(UnknownError, "Cannot inject key into script value"_s));
522 return std::nullopt;
523 }
524
525 return jsValue;
526}

528class IDBSerializationContext {
529public:
530 IDBSerializationContext()
531 : m_thread(Thread::current())
532 {
533 }
534
535 ~IDBSerializationContext()
536 {
537 ASSERT(&m_thread == &Thread::current());
538 if (!m_vm)
539 return;
540
541 JSC::JSLockHolder lock(*m_vm);
542 m_globalObject.clear();
543 m_vm = nullptr;
544 }
545
546 JSC::JSGlobalObject& globalObject()
547 {
548 ASSERT(&m_thread == &Thread::current());
549
550 initializeVM();
551 return *m_globalObject.get();
552 }
553
554private:
555 void initializeVM()
556 {
557 if (m_vm)
558 return;
559
560 ASSERT(!m_globalObject);
561 m_vm = JSC::VM::create();
562 m_vm->heap.acquireAccess();
563 JSVMClientData::initNormalWorld(m_vm.get(), WorkerThreadType::Worklet);
564
565 JSC::JSLockHolder locker(m_vm.get());
566 m_globalObject.set(*m_vm, JSIDBSerializationGlobalObject::create(*m_vm, JSIDBSerializationGlobalObject::createStructure(*m_vm, JSC::jsNull()), normalWorld(*m_vm)));
567 }
568
569 RefPtr<JSC::VM> m_vm;
570 JSC::Strong<JSIDBSerializationGlobalObject> m_globalObject;
571 Thread& m_thread;
572};

574void callOnIDBSerializationThreadAndWait(Function<void(JSC::JSGlobalObject&)>&& function)
575{
576 static NeverDestroyed<MessageQueue<Function<void(JSC::JSGlobalObject&)>>> queue;
577 static std::once_flag createThread;
578
579 std::call_once(createThread, [] {
580 Thread::create("IndexedDB Serialization", [] {
581 IDBSerializationContext serializationContext;
582 while (auto function = queue->waitForMessage()) {
583 AutodrainedPool pool;
584 (*function)(serializationContext.globalObject());
585 }
586 });
587 });
588
589 BinarySemaphore semaphore;
590 auto newFuntion = [&semaphore, function = WTFMove(function)](JSC::JSGlobalObject& globalObject) {
591 function(globalObject);
592 semaphore.signal();
593 };
594 queue->append(makeUnique<Function<void(JSC::JSGlobalObject&)>>(WTFMove(newFuntion)));
595 semaphore.wait();
596}
