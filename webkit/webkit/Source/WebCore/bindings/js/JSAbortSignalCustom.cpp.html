31bool JSAbortSignalOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, JSC::AbstractSlotVisitor& visitor, const char** reason)
32{
33 auto& abortSignal = JSC::jsCast<JSAbortSignal*>(handle.slot()->asCell())->wrapped();
34 if (abortSignal.isFiringEventListeners()) {
35 if (UNLIKELY(reason))
36 *reason = "EventTarget firing event listeners";
37 return true;
38 }
39
40 if (abortSignal.aborted())
41 return false;
42
43 if (abortSignal.isFollowingSignal())
44 return true;
45
46 return visitor.containsOpaqueRoot(&abortSignal);
47}
