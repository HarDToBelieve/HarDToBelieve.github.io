40JSC::JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<AbstractRange>&& range)
41{
42 if (is<StaticRange>(range.get()))
43 return createWrapper<StaticRange>(globalObject, WTFMove(range));
44 return createWrapper<Range>(globalObject, WTFMove(range));
