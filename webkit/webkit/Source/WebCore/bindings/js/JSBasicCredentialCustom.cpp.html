34namespace WebCore {
35using namespace JSC;
36
37JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<BasicCredential>&& credential)
38{
39 if (is<PublicKeyCredential>(credential))
40 return createWrapper<PublicKeyCredential>(globalObject, WTFMove(credential));
41 return createWrapper<BasicCredential>(globalObject, WTFMove(credential));
42}
43
44JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, BasicCredential& credential)
45{
46 return wrap(lexicalGlobalObject, globalObject, credential);
47}
