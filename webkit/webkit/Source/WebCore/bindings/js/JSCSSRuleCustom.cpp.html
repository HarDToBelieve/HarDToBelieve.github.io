56template<typename Visitor>
57void JSCSSRule::visitAdditionalChildren(Visitor& visitor)
58{
59 visitor.addOpaqueRoot(root(&wrapped()));
60}

64JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<CSSRule>&& rule)
65{
66 switch (rule->type()) {
67 case CSSRule::STYLE_RULE:
68 return createWrapper<CSSStyleRule>(globalObject, WTFMove(rule));
69 case CSSRule::MEDIA_RULE:
70 return createWrapper<CSSMediaRule>(globalObject, WTFMove(rule));
71 case CSSRule::FONT_FACE_RULE:
72 return createWrapper<CSSFontFaceRule>(globalObject, WTFMove(rule));
73 case CSSRule::PAGE_RULE:
74 return createWrapper<CSSPageRule>(globalObject, WTFMove(rule));
75 case CSSRule::IMPORT_RULE:
76 return createWrapper<CSSImportRule>(globalObject, WTFMove(rule));
77 case CSSRule::NAMESPACE_RULE:
78 return createWrapper<CSSNamespaceRule>(globalObject, WTFMove(rule));
79 case CSSRule::KEYFRAME_RULE:
80 return createWrapper<CSSKeyframeRule>(globalObject, WTFMove(rule));
81 case CSSRule::KEYFRAMES_RULE:
82 return createWrapper<CSSKeyframesRule>(globalObject, WTFMove(rule));
83 case CSSRule::SUPPORTS_RULE:
84 return createWrapper<CSSSupportsRule>(globalObject, WTFMove(rule));
85 case CSSRule::COUNTER_STYLE_RULE:
86 return createWrapper<CSSCounterStyleRule>(globalObject, WTFMove(rule));
87 default:
88 return createWrapper<CSSRule>(globalObject, WTFMove(rule));
89 }
90}

92JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, CSSRule& object)
93{
94 return wrap(lexicalGlobalObject, globalObject, object);
95}
