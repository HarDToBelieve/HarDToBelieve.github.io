38void printErrorMessageForFrame(Frame* frame, const String& message)
39{
40 if (!frame)
41 return;
42 frame->document()->domWindow()->printErrorMessage(message);
43}
44
45static inline bool canAccessDocument(JSC::JSGlobalObject* lexicalGlobalObject, Document* targetDocument, SecurityReportingOption reportingOption)
46{
47 if (!targetDocument)
48 return false;
49
50 if (auto* templateHost = targetDocument->templateDocumentHost())
51 targetDocument = templateHost;
52
53 DOMWindow& active = activeDOMWindow(*lexicalGlobalObject);
54
55 if (active.document()->securityOrigin().isSameOriginDomain(targetDocument->securityOrigin()))
56 return true;
57
58 switch (reportingOption) {
59 case ThrowSecurityError: {
60 VM& vm = lexicalGlobalObject->vm();
61 auto scope = DECLARE_THROW_SCOPE(vm);
62 throwSecurityError(*lexicalGlobalObject, scope, targetDocument->domWindow()->crossDomainAccessErrorMessage(active, IncludeTargetOrigin::No));
63 break;
64 }
65 case LogSecurityError:
66 printErrorMessageForFrame(targetDocument->frame(), targetDocument->domWindow()->crossDomainAccessErrorMessage(active, IncludeTargetOrigin::Yes));
67 break;
68 case DoNotReportSecurityError:
69 break;
70 }
71
72 return false;
73}

75bool BindingSecurity::shouldAllowAccessToFrame(JSGlobalObject& lexicalGlobalObject, Frame& frame, String& message)
76{
77 if (BindingSecurity::shouldAllowAccessToFrame(&lexicalGlobalObject, &frame, DoNotReportSecurityError))
78 return true;
79 message = frame.document()->domWindow()->crossDomainAccessErrorMessage(activeDOMWindow(lexicalGlobalObject), IncludeTargetOrigin::No);
80 return false;
81}

83bool BindingSecurity::shouldAllowAccessToDOMWindow(JSGlobalObject& lexicalGlobalObject, DOMWindow* globalObject, String& message)
84{
85 return globalObject && shouldAllowAccessToDOMWindow(lexicalGlobalObject, *globalObject, message);
86}
87
88bool BindingSecurity::shouldAllowAccessToDOMWindow(JSGlobalObject& lexicalGlobalObject, DOMWindow& globalObject, String& message)
89{
90 VM& vm = lexicalGlobalObject.vm();
91 auto scope = DECLARE_CATCH_SCOPE(vm);
92
93 bool shouldAllowAccess = BindingSecurity::shouldAllowAccessToDOMWindow(&lexicalGlobalObject, globalObject, DoNotReportSecurityError);
94 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
95 if (shouldAllowAccess)
96 return true;
97 message = globalObject.crossDomainAccessErrorMessage(activeDOMWindow(lexicalGlobalObject), IncludeTargetOrigin::No);
98 return false;
99}

101bool BindingSecurity::shouldAllowAccessToDOMWindow(JSC::JSGlobalObject* lexicalGlobalObject, DOMWindow& target, SecurityReportingOption reportingOption)
102{
103 return canAccessDocument(lexicalGlobalObject, target.document(), reportingOption);
104}

106bool BindingSecurity::shouldAllowAccessToDOMWindow(JSC::JSGlobalObject* lexicalGlobalObject, DOMWindow* target, SecurityReportingOption reportingOption)
107{
108 return target && shouldAllowAccessToDOMWindow(lexicalGlobalObject, *target, reportingOption);
109}
110
111bool BindingSecurity::shouldAllowAccessToFrame(JSC::JSGlobalObject* lexicalGlobalObject, Frame* target, SecurityReportingOption reportingOption)
112{
113 return target && canAccessDocument(lexicalGlobalObject, target->document(), reportingOption);
114}
115
116bool BindingSecurity::shouldAllowAccessToNode(JSC::JSGlobalObject& lexicalGlobalObject, Node* target)
117{
118 return !target || canAccessDocument(&lexicalGlobalObject, &target->document(), LogSecurityError);
119}
