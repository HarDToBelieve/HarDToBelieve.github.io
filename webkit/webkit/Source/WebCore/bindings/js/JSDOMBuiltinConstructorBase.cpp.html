31void JSDOMBuiltinConstructorBase::callFunctionWithCurrentArguments(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& thisObject, JSC::JSFunction& function)
32{
33 JSC::VM& vm = lexicalGlobalObject.vm();
34 auto scope = DECLARE_THROW_SCOPE(vm);
35 auto callData = JSC::getCallData(vm, &function);
36 ASSERT(callData.type != CallData::Type::None);
37
38 JSC::MarkedArgumentBuffer arguments;
39 for (unsigned i = 0; i < callFrame.argumentCount(); ++i)
40 arguments.append(callFrame.uncheckedArgument(i));
41 if (UNLIKELY(arguments.hasOverflowed())) {
42 throwOutOfMemoryError(&lexicalGlobalObject, scope);
43 return;
44 }
45 scope.release();
46 JSC::call(&lexicalGlobalObject, &function, callData, &thisObject, arguments);
47}

49template<typename Visitor>
50void JSDOMBuiltinConstructorBase::visitChildrenImpl(JSC::JSCell* cell, Visitor& visitor)
51{
52 auto* thisObject = jsCast<JSDOMBuiltinConstructorBase*>(cell);
53 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
54 Base::visitChildren(thisObject, visitor);
55 visitor.append(thisObject->m_initializeFunction);
56}

60JSC::IsoSubspace* JSDOMBuiltinConstructorBase::subspaceForImpl(JSC::VM& vm)
61{
62 return &static_cast<JSVMClientData*>(vm.clientData)->domBuiltinConstructorSpace();
63}
