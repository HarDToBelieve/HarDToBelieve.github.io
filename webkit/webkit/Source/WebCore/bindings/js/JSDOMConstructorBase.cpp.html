31STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSDOMConstructorBase);
32
33static JSC_DECLARE_HOST_FUNCTION(callThrowTypeError);
34
35JSC_DEFINE_HOST_FUNCTION(callThrowTypeError, (JSGlobalObject* globalObject, CallFrame*))
36{
37 VM& vm = globalObject->vm();
38 auto scope = DECLARE_THROW_SCOPE(vm);
39 throwTypeError(globalObject, scope, "Constructor requires 'new' operator"_s);
40 return JSValue::encode(jsNull());
41}
42
43JSC_DEFINE_HOST_FUNCTION(callThrowTypeErrorForJSDOMConstructorNotConstructable, (JSC::JSGlobalObject* globalObject, JSC::CallFrame*))
44{
45 JSC::VM& vm = globalObject->vm();
46 auto scope = DECLARE_THROW_SCOPE(vm);
47 JSC::throwTypeError(globalObject, scope, "Illegal constructor"_s);
48 return JSC::JSValue::encode(JSC::jsNull());
49}
50
51CallData JSDOMConstructorBase::getCallData(JSCell*)
52{
53 CallData callData;
54 callData.type = CallData::Type::Native;
55 callData.native.function = callThrowTypeError;
56 return callData;
57}
58
59JSC::IsoSubspace* JSDOMConstructorBase::subspaceForImpl(JSC::VM& vm)
60{
61 return &static_cast<JSVMClientData*>(vm.clientData)->domConstructorSpace();
62}
