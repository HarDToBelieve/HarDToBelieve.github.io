26JSC_DECLARE_HOST_FUNCTION(callThrowTypeErrorForJSDOMConstructorNotConstructable);

28// Base class for all constructor objects in the JSC bindings.
29class JSDOMConstructorBase : public JSDOMObject {
30public:
31 using Base = JSDOMObject;
32
33 static constexpr unsigned StructureFlags = Base::StructureFlags | JSC::ImplementsHasInstance | JSC::ImplementsDefaultHasInstance | JSC::OverridesGetCallData;
34 static constexpr bool needsDestruction = false;
35 static JSC::Structure* createStructure(JSC::VM&, JSC::JSGlobalObject*, JSC::JSValue);
36
37 template<typename CellType, JSC::SubspaceAccess>
38 static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
39 {
40 static_assert(sizeof(CellType) == sizeof(JSDOMConstructorBase));
41 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(CellType, JSDOMConstructorBase);
42 static_assert(CellType::destroy == JSC::JSCell::destroy, "JSDOMConstructor<JSClass> is not destructible actually");
43 return subspaceForImpl(vm);
44 }
45
46 static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
47
48protected:
49 JSDOMConstructorBase(JSC::Structure* structure, JSDOMGlobalObject& globalObject)
50 : JSDOMObject(structure, globalObject)
51 {
52 }
53
54 static JSC::CallData getCallData(JSC::JSCell*);
55};
56
57inline JSC::Structure* JSDOMConstructorBase::createStructure(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
58{
59 return JSC::Structure::create(vm, globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
60}
