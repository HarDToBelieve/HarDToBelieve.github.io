159template<typename T, typename U> inline JSC::JSValue toJS(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& throwScope, U&& valueOrFunctor)
160{
161 if constexpr (std::is_invocable_v<U>) {
162 using FunctorReturnType = std::invoke_result_t<U>;
163
164 if constexpr (std::is_same_v<void, FunctorReturnType>) {
165 valueOrFunctor();
166 return JSC::jsUndefined();
167 } else if constexpr (std::is_same_v<ExceptionOr<void>, FunctorReturnType>) {
168 auto result = valueOrFunctor();
169 if (UNLIKELY(result.hasException())) {
170 propagateException(lexicalGlobalObject, throwScope, result.releaseException());
171 return { };
172 }
173 return JSC::jsUndefined();
174 } else
175 return toJS<T>(lexicalGlobalObject, throwScope, valueOrFunctor());
176 } else {
177 if constexpr (IsExceptionOr<U>) {
178 if (UNLIKELY(valueOrFunctor.hasException())) {
179 propagateException(lexicalGlobalObject, throwScope, valueOrFunctor.releaseException());
180 return { };
181 }
182
183 return toJS<T>(lexicalGlobalObject, valueOrFunctor.releaseReturnValue());
184 } else
185 return toJS<T>(lexicalGlobalObject, std::forward<U>(valueOrFunctor));
186 }
187}

250template<typename T> struct DefaultConverter {
251 using ReturnType = typename T::ImplementationType;
252
253 // We assume the worst, subtypes can override to be less pessimistic.
254 // An example of something that can have side effects
255 // is having a converter that does JSC::JSValue::toNumber.
256 // toNumber() in JavaScript can call arbitrary JS functions.
257 //
258 // An example of something that does not have side effects
259 // is something having a converter that does JSC::JSValue::toBoolean.
260 // toBoolean() in JS can't call arbitrary functions.
261 static constexpr bool conversionHasSideEffects = true;
262};
