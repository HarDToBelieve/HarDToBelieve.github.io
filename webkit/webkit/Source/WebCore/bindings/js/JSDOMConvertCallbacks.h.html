33template<typename T> struct Converter<IDLCallbackFunction<T>> : DefaultConverter<IDLCallbackFunction<T>> {
34
35 static constexpr bool conversionHasSideEffects = false;
36
37 template<typename ExceptionThrower = DefaultExceptionThrower>
38 static RefPtr<T> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, JSDOMGlobalObject& globalObject, ExceptionThrower&& exceptionThrower = ExceptionThrower())
39 {
40 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
41 auto scope = DECLARE_THROW_SCOPE(vm);
42
43 if (!value.isCallable(vm)) {
44 exceptionThrower(lexicalGlobalObject, scope);
45 return nullptr;
46 }
47
48 return T::create(JSC::asObject(value), &globalObject);
49 }
50};
51
52template<typename T> struct JSConverter<IDLCallbackFunction<T>> {
53 static constexpr bool needsState = false;
54 static constexpr bool needsGlobalObject = false;
55
56 template <typename U>
57 static JSC::JSValue convert(const U& value)
58 {
59 return toJS(Detail::getPtrOrRef(value));
60 }
61
62 template<typename U>
63 static JSC::JSValue convertNewlyCreated(U&& value)
64 {
65 return toJSNewlyCreated(std::forward<U>(value));
66 }
67};

70template<typename T> struct Converter<IDLCallbackInterface<T>> : DefaultConverter<IDLCallbackInterface<T>> {
71 template<typename ExceptionThrower = DefaultExceptionThrower>
72 static RefPtr<T> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, JSDOMGlobalObject& globalObject, ExceptionThrower&& exceptionThrower = ExceptionThrower())
73 {
74 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
75 auto scope = DECLARE_THROW_SCOPE(vm);
76
77 if (!value.isObject()) {
78 exceptionThrower(lexicalGlobalObject, scope);
79 return nullptr;
80 }
81
82 return T::create(JSC::asObject(value), &globalObject);
83 }
84};
