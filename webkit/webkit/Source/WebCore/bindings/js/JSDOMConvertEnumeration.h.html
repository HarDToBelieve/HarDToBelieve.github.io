34// Specialized by generated code for IDL enumeration conversion.
35template<typename T> std::optional<T> parseEnumeration(JSC::JSGlobalObject&, JSC::JSValue);
36template<typename T> const char* expectedEnumerationValues();
37
38// Specialized by generated code for IDL enumeration conversion.
39template<typename T> JSC::JSString* convertEnumerationToJS(JSC::JSGlobalObject&, T);
40
41
42template<typename T> struct Converter<IDLEnumeration<T>> : DefaultConverter<IDLEnumeration<T>> {
43 template<typename ExceptionThrower = DefaultExceptionThrower>
44 static T convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, ExceptionThrower&& exceptionThrower = ExceptionThrower())
45 {
46 auto& vm = JSC::getVM(&lexicalGlobalObject);
47 auto throwScope = DECLARE_THROW_SCOPE(vm);
48
49 auto result = parseEnumeration<T>(lexicalGlobalObject, value);
50 RETURN_IF_EXCEPTION(throwScope, { });
51
52 if (UNLIKELY(!result)) {
53 exceptionThrower(lexicalGlobalObject, throwScope);
54 return { };
55 }
56 return result.value();
57 }
58};
59
60template<typename T> struct JSConverter<IDLEnumeration<T>> {
61 static constexpr bool needsState = true;
62 static constexpr bool needsGlobalObject = false;
63
64 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, T value)
65 {
66 return convertEnumerationToJS(lexicalGlobalObject, value);
67 }
68};
