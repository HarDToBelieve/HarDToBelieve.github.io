26#pragma once
27
28#include "IDLTypes.h"
29#include "JSDOMConvertBase.h"
30#include "JSDOMConvertStrings.h"
31#include "ScheduledAction.h"
32
33namespace WebCore {
34
35template<> struct Converter<IDLScheduledAction> : DefaultConverter<IDLScheduledAction> {
36 static std::unique_ptr<ScheduledAction> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, JSDOMGlobalObject& globalObject)
37 {
38 JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
39 auto scope = DECLARE_THROW_SCOPE(vm);
40
41 if (!value.isCallable(vm)) {
42 auto code = Converter<IDLDOMString>::convert(lexicalGlobalObject, value);
43 RETURN_IF_EXCEPTION(scope, nullptr);
44 return ScheduledAction::create(globalObject.world(), WTFMove(code));
45 }
46
47 // The value must be an object at this point because no non-object values are callable.
48 ASSERT(value.isObject());
49 return ScheduledAction::create(globalObject.world(), JSC::Strong<JSC::Unknown> { vm, JSC::asObject(value) });
50 }
51};
