35String identifierToString(JSGlobalObject& lexicalGlobalObject, const Identifier& identifier)
36{
37 if (UNLIKELY(identifier.isSymbol())) {
38 auto scope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
39 throwTypeError(&lexicalGlobalObject, scope, SymbolCoercionError);
40 return { };
41 }
42
43 return identifier.string();
44}
45
46static inline String stringToByteString(JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& scope, String&& string)
47{
48 if (!string.isAllLatin1()) {
49 throwTypeError(&lexicalGlobalObject, scope);
50 return { };
51 }
52
53 return WTFMove(string);
54}
55
56String identifierToByteString(JSGlobalObject& lexicalGlobalObject, const Identifier& identifier)
57{
58 VM& vm = lexicalGlobalObject.vm();
59 auto scope = DECLARE_THROW_SCOPE(vm);
60
61 auto string = identifierToString(lexicalGlobalObject, identifier);
62 RETURN_IF_EXCEPTION(scope, { });
63 return stringToByteString(lexicalGlobalObject, scope, WTFMove(string));
64}

66String valueToByteString(JSGlobalObject& lexicalGlobalObject, JSValue value)
67{
68 VM& vm = lexicalGlobalObject.vm();
69 auto scope = DECLARE_THROW_SCOPE(vm);
70
71 auto string = value.toWTFString(&lexicalGlobalObject);
72 RETURN_IF_EXCEPTION(scope, { });
73
74 return stringToByteString(lexicalGlobalObject, scope, WTFMove(string));
75}

77String stringToUSVString(String&& string)
78{
79 // Fast path for the case where there are no unpaired surrogates.
80 if (!hasUnpairedSurrogate(string))
81 return WTFMove(string);
82
83 // Slow path: http://heycam.github.io/webidl/#dfn-obtain-unicode
84 // Replaces unpaired surrogates with the replacement character.
85 StringBuilder result;
86 result.reserveCapacity(string.length());
87 StringView view { string };
88 for (auto codePoint : view.codePoints()) {
89 if (U_IS_SURROGATE(codePoint))
90 result.append(replacementCharacter);
91 else
92 result.appendCharacter(codePoint);
93 }
94 return result.toString();
95}
96
97String identifierToUSVString(JSGlobalObject& lexicalGlobalObject, const Identifier& identifier)
98{
99 return stringToUSVString(identifierToString(lexicalGlobalObject, identifier));
100}
101
102String valueToUSVString(JSGlobalObject& lexicalGlobalObject, JSValue value)
103{
104 VM& vm = lexicalGlobalObject.vm();
105 auto scope = DECLARE_THROW_SCOPE(vm);
106
107 auto string = value.toWTFString(&lexicalGlobalObject);
108 RETURN_IF_EXCEPTION(scope, { });
109
110 return stringToUSVString(WTFMove(string));
111}
