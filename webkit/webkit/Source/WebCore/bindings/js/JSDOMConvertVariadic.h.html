28#include "IDLTypes.h"
29#include "JSDOMConvertBase.h"

31namespace WebCore {
32
33template<typename IDLType>
34struct VariadicConverter {
35 using Item = typename IDLType::ImplementationType;
36
37 static std::optional<Item> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
38 {
39 auto& vm = JSC::getVM(&lexicalGlobalObject);
40 auto scope = DECLARE_THROW_SCOPE(vm);
41
42 auto result = Converter<IDLType>::convert(lexicalGlobalObject, value);
43 RETURN_IF_EXCEPTION(scope, std::nullopt);
44
45 return result;
46 }
47};
48
49template<typename IDLType> Vector<typename VariadicConverter<IDLType>::Item> convertVariadicArguments(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, size_t startIndex)
50{
51 auto& vm = JSC::getVM(&lexicalGlobalObject);
52 auto scope = DECLARE_THROW_SCOPE(vm);
53
54 size_t length = callFrame.argumentCount();
55 if (startIndex >= length)
56 return { };
57
58 Vector<typename VariadicConverter<IDLType>::Item> result;
59 result.reserveInitialCapacity(length - startIndex);
60
61 for (size_t i = startIndex; i < length; ++i) {
62 auto value = VariadicConverter<IDLType>::convert(lexicalGlobalObject, callFrame.uncheckedArgument(i));
63 EXCEPTION_ASSERT_UNUSED(scope, !!scope.exception() == !value);
64 if (!value)
65 return { };
66 result.uncheckedAppend(WTFMove(*value));
67 }
68
69 return result;
70}
