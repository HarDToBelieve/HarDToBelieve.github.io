41void reportException(JSGlobalObject* lexicalGlobalObject, JSValue exceptionValue, CachedScript* cachedScript)
42{
43 VM& vm = lexicalGlobalObject->vm();
44 RELEASE_ASSERT(vm.currentThreadIsHoldingAPILock());
45 auto* exception = jsDynamicCast<JSC::Exception*>(vm, exceptionValue);
46 if (!exception) {
47 exception = vm.lastException();
48 if (!exception)
49 exception = JSC::Exception::create(lexicalGlobalObject->vm(), exceptionValue, JSC::Exception::DoNotCaptureStack);
50 }
51
52 reportException(lexicalGlobalObject, exception, cachedScript);
53}

55String retrieveErrorMessageWithoutName(JSGlobalObject& lexicalGlobalObject, VM& vm, JSValue exception, CatchScope& catchScope)
56{
57 // FIXME: <http://webkit.org/b/115087> Web Inspector: WebCore::reportException should not evaluate JavaScript handling exceptions
58 // If this is a custom exception object, call toString on it to try and get a nice string representation for the exception.
59 String errorMessage;
60 if (auto* error = jsDynamicCast<ErrorInstance*>(vm, exception))
61 errorMessage = error->sanitizedMessageString(&lexicalGlobalObject);
62 else if (auto* error = jsDynamicCast<JSDOMException*>(vm, exception))
63 errorMessage = error->wrapped().message();
64 else
65 errorMessage = exception.toWTFString(&lexicalGlobalObject);
66
67 // We need to clear any new exception that may be thrown in the toString() call above.
68 // reportException() is not supposed to be making new exceptions.
69 catchScope.clearException();
70 vm.clearLastException();
71 return errorMessage;
72}

74String retrieveErrorMessage(JSGlobalObject& lexicalGlobalObject, VM& vm, JSValue exception, CatchScope& catchScope)
75{
76 // FIXME: <http://webkit.org/b/115087> Web Inspector: WebCore::reportException should not evaluate JavaScript handling exceptions
77 // If this is a custom exception object, call toString on it to try and get a nice string representation for the exception.
78 String errorMessage;
79 if (auto* error = jsDynamicCast<ErrorInstance*>(vm, exception))
80 errorMessage = error->sanitizedToString(&lexicalGlobalObject);
81 else
82 errorMessage = exception.toWTFString(&lexicalGlobalObject);
83
84 // We need to clear any new exception that may be thrown in the toString() call above.
85 // reportException() is not supposed to be making new exceptions.
86 catchScope.clearException();
87 vm.clearLastException();
88 return errorMessage;
89}

91void reportException(JSGlobalObject* lexicalGlobalObject, JSC::Exception* exception, CachedScript* cachedScript, ExceptionDetails* exceptionDetails)
92{
93 VM& vm = lexicalGlobalObject->vm();
94 auto scope = DECLARE_CATCH_SCOPE(vm);
95
96 RELEASE_ASSERT(vm.currentThreadIsHoldingAPILock());
97 if (vm.isTerminationException(exception))
98 return;
99
100 ErrorHandlingScope errorScope(lexicalGlobalObject->vm());
101
102 auto callStack = Inspector::createScriptCallStackFromException(lexicalGlobalObject, exception);
103 scope.clearException();
104 vm.clearLastException();
105
106 auto* globalObject = jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
107 if (auto* window = jsDynamicCast<JSDOMWindow*>(vm, globalObject)) {
108 if (!window->wrapped().isCurrentlyDisplayedInFrame())
109 return;
110 }
111
112 int lineNumber = 0;
113 int columnNumber = 0;
114 String exceptionSourceURL;
115 if (auto* callFrame = callStack->firstNonNativeCallFrame()) {
116 lineNumber = callFrame->lineNumber();
117 columnNumber = callFrame->columnNumber();
118 exceptionSourceURL = callFrame->sourceURL();
119 }
120
121 auto errorMessage = retrieveErrorMessage(*lexicalGlobalObject, vm, exception->value(), scope);
122 globalObject->scriptExecutionContext()->reportException(errorMessage, lineNumber, columnNumber, exceptionSourceURL, exception, callStack->size() ? callStack.ptr() : nullptr, cachedScript);
123
124 if (exceptionDetails) {
125 exceptionDetails->message = errorMessage;
126 exceptionDetails->lineNumber = lineNumber;
127 exceptionDetails->columnNumber = columnNumber;
128 exceptionDetails->sourceURL = exceptionSourceURL;
129 }
130}

132void reportCurrentException(JSGlobalObject* lexicalGlobalObject)
133{
134 VM& vm = lexicalGlobalObject->vm();
135 auto scope = DECLARE_CATCH_SCOPE(vm);
136 auto* exception = scope.exception();
137 scope.clearException();
138 reportException(lexicalGlobalObject, exception);
139}

141JSValue createDOMException(JSGlobalObject* lexicalGlobalObject, ExceptionCode ec, const String& message)
142{
143 switch (ec) {
144 case ExistingExceptionError:
145 return jsUndefined();
146
147 // FIXME: Handle other WebIDL exception types.
148 case TypeError:
149 if (message.isEmpty())
150 return createTypeError(lexicalGlobalObject);
151 return createTypeError(lexicalGlobalObject, message);
152
153 case RangeError:
154 if (message.isEmpty())
155 return createRangeError(lexicalGlobalObject, "Bad value"_s);
156 return createRangeError(lexicalGlobalObject, message);
157
158 case JSSyntaxError:
159 if (message.isEmpty())
160 return createSyntaxError(lexicalGlobalObject);
161 return createSyntaxError(lexicalGlobalObject, message);
162
163 case StackOverflowError:
164 return createStackOverflowError(lexicalGlobalObject);
165
166 case OutOfMemoryError:
167 return createOutOfMemoryError(lexicalGlobalObject);
168
169 default: {
170 // FIXME: All callers to createDOMException need to pass in the correct global object.
171 // For now, we're going to assume the lexicalGlobalObject. Which is wrong in cases like this:
172 // frames[0].document.createElement(null, null); // throws an exception which should have the subframe's prototypes.
173 // https://bugs.webkit.org/show_bug.cgi?id=222229
174 JSDOMGlobalObject* globalObject = deprecatedGlobalObjectForPrototype(lexicalGlobalObject);
175 JSValue errorObject = toJS(lexicalGlobalObject, globalObject, DOMException::create(ec, message));
176
177 ASSERT(errorObject);
178 addErrorInfo(lexicalGlobalObject, asObject(errorObject), true);
179 return errorObject;
180 }
181 }
182 return { };
183}

203 return makeString(

205 functionName ? std::make_tuple(interfaceName, ".", functionName) : std::make_tuple("the ", interfaceName, " constructor"),
206 " must be ", strings...
