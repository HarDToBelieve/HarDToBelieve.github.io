149template<class JSClass>
150JSClass* toJSDOMGlobalObject(JSC::VM& vm, JSC::JSValue value)
151{
152 static_assert(std::is_base_of_v<JSDOMGlobalObject, JSClass>);
153
154 if (auto* object = value.getObject()) {
155 if (object->type() == JSC::PureForwardingProxyType)
156 return JSC::jsDynamicCast<JSClass*>(vm, JSC::jsCast<JSC::JSProxy*>(object)->target());
157 if (object->inherits<JSClass>(vm))
158 return JSC::jsCast<JSClass*>(object);
159 }
160
161 return nullptr;
162}
