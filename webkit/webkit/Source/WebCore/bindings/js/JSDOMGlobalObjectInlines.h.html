34template<class ConstructorClass, DOMConstructorID constructorID>
35inline JSC::JSObject* getDOMConstructor(JSC::VM& vm, const JSDOMGlobalObject& globalObject)
36{
37 // No locking is necessary unless we need to add a new constructor to JSDOMGlobalObject::constructors().
38 if (JSC::JSObject* constructor = globalObject.constructors().array()[static_cast<unsigned>(constructorID)].get())
39 return constructor;
40 JSC::JSObject* constructor = ConstructorClass::create(vm, ConstructorClass::createStructure(vm, const_cast<JSDOMGlobalObject&>(globalObject), ConstructorClass::prototypeForStructure(vm, globalObject)), const_cast<JSDOMGlobalObject&>(globalObject));
41 ASSERT(!globalObject.constructors().array()[static_cast<unsigned>(constructorID)].get());
42 JSDOMGlobalObject& mutableGlobalObject = const_cast<JSDOMGlobalObject&>(globalObject);
43 mutableGlobalObject.constructors().array()[static_cast<unsigned>(constructorID)].set(vm, &globalObject, constructor);
44 return constructor;
45}
