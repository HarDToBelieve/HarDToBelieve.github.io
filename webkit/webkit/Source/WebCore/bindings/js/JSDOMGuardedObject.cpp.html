30namespace WebCore {
31using namespace JSC;
32
33DOMGuardedObject::DOMGuardedObject(JSDOMGlobalObject& globalObject, JSCell& guarded)
34 : ActiveDOMCallback(globalObject.scriptExecutionContext())
35 , m_guarded(&guarded)
36 , m_globalObject(&globalObject)
37{
38 globalObject.vm().heap.writeBarrier(&globalObject, &guarded);
39 if (globalObject.vm().heap.mutatorShouldBeFenced()) {
40 Locker locker { globalObject.gcLock() };
41 globalObject.guardedObjects().add(this);
42 return;
43 }
44 globalObject.guardedObjects(NoLockingNecessary).add(this);
45}
46
47DOMGuardedObject::~DOMGuardedObject()
48{
49 clear();
50}
51
52void DOMGuardedObject::clear()
53{
54 ASSERT(!m_guarded || m_globalObject);
55 removeFromGlobalObject();
56 m_guarded.clear();
57 m_globalObject.clear();
58}
59
60void DOMGuardedObject::removeFromGlobalObject()
61{
62 if (!m_guarded || !m_globalObject)
63 return;
64
65 if (m_globalObject->vm().heap.mutatorShouldBeFenced()) {
66 Locker locker { m_globalObject->gcLock() };
67 m_globalObject->guardedObjects().remove(this);
68 } else
69 m_globalObject->guardedObjects(NoLockingNecessary).remove(this);
70}
71
72void DOMGuardedObject::contextDestroyed()
73{
74 ActiveDOMCallback::contextDestroyed();
75 clear();
76}
