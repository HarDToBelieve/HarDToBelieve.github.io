50public:
51 using Base = JSC::JSNonFinalObject;
52 using DOMWrapped = typename JSWrapper::DOMWrapped;

54 template<typename CellType, JSC::SubspaceAccess>
55 static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
56 {
57 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(JSDOMIteratorPrototype, Base);
58 return &vm.plainObjectSpace;
59 }

61 static JSDOMIteratorPrototype* create(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::Structure* structure)
62 {
63 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(JSDOMIteratorPrototype, JSDOMIteratorPrototype::Base);
64 JSDOMIteratorPrototype* prototype = new (NotNull, JSC::allocateCell<JSDOMIteratorPrototype>(vm.heap)) JSDOMIteratorPrototype(vm, structure);
65 prototype->finishCreation(vm, globalObject);
66 return prototype;
67 }
68
69 DECLARE_INFO;
70
71 static JSC::Structure* createStructure(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
72 {
73 return JSC::Structure::create(vm, globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
74 }
75
76 static JSC::EncodedJSValue JSC_HOST_CALL_ATTRIBUTES next(JSC::JSGlobalObject*, JSC::CallFrame*);
77
78private:
79 JSDOMIteratorPrototype(JSC::VM& vm, JSC::Structure* structure) : Base(vm, structure) { }

86template<typename JSWrapper, typename IteratorTraits> class JSDOMIteratorBase : public JSDOMObject {
87public:
88 using Base = JSDOMObject;
89
90 using Wrapper = JSWrapper;
91 using Traits = IteratorTraits;
92
93 using DOMWrapped = typename Wrapper::DOMWrapped;
94 using Prototype = JSDOMIteratorPrototype<Wrapper, Traits>;
95
96 DECLARE_INFO;
97
98 static Prototype* createPrototype(JSC::VM& vm, JSC::JSGlobalObject& globalObject)
99 {
100 return Prototype::create(vm, &globalObject, Prototype::createStructure(vm, &globalObject, globalObject.iteratorPrototype()));
101 }
102
103 JSC::JSValue next(JSC::JSGlobalObject&);
104
105 static void createStructure(JSC::VM&, JSC::JSGlobalObject*, JSC::JSValue); // Make use of createStructure for this compile-error.
106
107protected:
108 JSDOMIteratorBase(JSC::Structure* structure, JSWrapper& iteratedObject, IterationKind kind)
109 : Base(structure, *iteratedObject.globalObject())
110 , m_iterator(iteratedObject.wrapped().createIterator())
111 , m_kind(kind)
112 {
113 }
114
115 template<typename IteratorValue, typename T = Traits> EnableIfMap<T, JSC::JSValue> asJS(JSC::JSGlobalObject&, IteratorValue&);
116 template<typename IteratorValue, typename T = Traits> EnableIfSet<T, JSC::JSValue> asJS(JSC::JSGlobalObject&, IteratorValue&);
117
118 static void destroy(JSC::JSCell*);
119
120 std::optional<typename DOMWrapped::Iterator> m_iterator;
121 IterationKind m_kind;
122};

124inline JSC::JSValue jsPair(JSC::JSGlobalObject&, JSDOMGlobalObject& globalObject, JSC::JSValue value1, JSC::JSValue value2)
125{
126 JSC::MarkedArgumentBuffer arguments;
127 arguments.append(value1);
128 arguments.append(value2);
129 ASSERT(!arguments.hasOverflowed());
130 return constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), arguments);
131}

133template<typename FirstType, typename SecondType, typename T, typename U>
134inline JSC::JSValue jsPair(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const T& value1, const U& value2)
135{
136 return jsPair(lexicalGlobalObject, globalObject, toJS<FirstType>(lexicalGlobalObject, globalObject, value1), toJS<SecondType>(lexicalGlobalObject, globalObject, value2));

149template<typename JSWrapper, typename IteratorTraits>
150template<typename IteratorValue, typename T> inline EnableIfMap<T, JSC::JSValue> JSDOMIteratorBase<JSWrapper, IteratorTraits>::asJS(JSC::JSGlobalObject& lexicalGlobalObject, IteratorValue& value)
151{
152 ASSERT(value);
153
154 switch (m_kind) {
155 case IterationKind::Keys:
156 return toJS<typename Traits::KeyType>(lexicalGlobalObject, *globalObject(), value->key);
157 case IterationKind::Values:
158 return toJS<typename Traits::ValueType>(lexicalGlobalObject, *globalObject(), value->value);
159 case IterationKind::Entries:
160 return jsPair<typename Traits::KeyType, typename Traits::ValueType>(lexicalGlobalObject, *globalObject(), value->key, value->value);
161 };
162
163 ASSERT_NOT_REACHED();
164 return { };
165}

167template<typename JSWrapper, typename IteratorTraits>
168template<typename IteratorValue, typename T> inline EnableIfSet<T, JSC::JSValue> JSDOMIteratorBase<JSWrapper, IteratorTraits>::asJS(JSC::JSGlobalObject& lexicalGlobalObject, IteratorValue& value)
169{
170 ASSERT(value);
171
172 auto globalObject = this->globalObject();
173 auto result = toJS<typename Traits::ValueType>(lexicalGlobalObject, *globalObject, value);
174
175 switch (m_kind) {
176 case IterationKind::Keys:
177 case IterationKind::Values:
178 return result;
179 case IterationKind::Entries:
180 return jsPair(lexicalGlobalObject, *globalObject, result, result);
181 };
182
183 ASSERT_NOT_REACHED();
184 return { };
185}

187template<typename JSIterator, typename IteratorValue> EnableIfMap<typename JSIterator::Traits> appendForEachArguments(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, JSC::MarkedArgumentBuffer& arguments, IteratorValue& value)
188{
189 ASSERT(value);
190 arguments.append(toJS<typename JSIterator::Traits::ValueType>(lexicalGlobalObject, globalObject, value->value));
191 arguments.append(toJS<typename JSIterator::Traits::KeyType>(lexicalGlobalObject, globalObject, value->key));
192}

194template<typename JSIterator, typename IteratorValue> EnableIfSet<typename JSIterator::Traits> appendForEachArguments(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, JSC::MarkedArgumentBuffer& arguments, IteratorValue& value)
195{
196 ASSERT(value);
197 auto argument = toJS<typename JSIterator::Traits::ValueType>(lexicalGlobalObject, globalObject, value);
198 arguments.append(argument);
199 arguments.append(argument);
200}

202template<typename JSIterator> JSC::JSValue iteratorForEach(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, typename JSIterator::Wrapper& thisObject)
203{
204 auto& vm = JSC::getVM(&lexicalGlobalObject);
205 auto scope = DECLARE_THROW_SCOPE(vm);
206 JSC::JSValue callback = callFrame.argument(0);
207 JSC::JSValue thisValue = callFrame.argument(1);
208
209 auto callData = JSC::getCallData(JSC::getVM(&lexicalGlobalObject), callback);
210 if (callData.type == JSC::CallData::Type::None)
211 return throwTypeError(&lexicalGlobalObject, scope, "Cannot call callback"_s);
212
213 auto iterator = thisObject.wrapped().createIterator();
214 while (auto value = iterator.next()) {
215 JSC::MarkedArgumentBuffer arguments;
216 appendForEachArguments<JSIterator>(lexicalGlobalObject, *thisObject.globalObject(), arguments, value);
217 arguments.append(&thisObject);
218 if (UNLIKELY(arguments.hasOverflowed())) {
219 throwOutOfMemoryError(&lexicalGlobalObject, scope);
220 return { };
221 }
222 JSC::call(&lexicalGlobalObject, callback, callData, thisValue, arguments);
223 if (UNLIKELY(scope.exception()))
224 break;
225 }
226 return JSC::jsUndefined();
227}

236template<typename JSWrapper, typename IteratorTraits>
237JSC::JSValue JSDOMIteratorBase<JSWrapper, IteratorTraits>::next(JSC::JSGlobalObject& lexicalGlobalObject)
238{
239 if (m_iterator) {
240 auto iteratorValue = m_iterator->next();
241 if (iteratorValue)
242 return createIteratorResultObject(&lexicalGlobalObject, asJS(lexicalGlobalObject, iteratorValue), false);
243 m_iterator = std::nullopt;
244 }
245 return createIteratorResultObject(&lexicalGlobalObject, JSC::jsUndefined(), true);
246}

248template<typename JSWrapper, typename IteratorTraits>
249JSC::EncodedJSValue JSC_HOST_CALL_ATTRIBUTES JSDOMIteratorPrototype<JSWrapper, IteratorTraits>::next(JSC::JSGlobalObject* globalObject, JSC::CallFrame* callFrame)
250{
251 JSC::VM& vm = globalObject->vm();
252 auto scope = DECLARE_THROW_SCOPE(vm);
253
254 auto iterator = JSC::jsDynamicCast<JSDOMIteratorBase<JSWrapper, IteratorTraits>*>(vm, callFrame->thisValue());
255 if (!iterator)
256 return JSC::JSValue::encode(throwTypeError(globalObject, scope, "Cannot call next() on a non-Iterator object"_s));
257
258 return JSC::JSValue::encode(iterator->next(*globalObject));
259}
