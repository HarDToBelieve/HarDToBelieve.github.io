41auto DOMPromise::whenSettled(std::function<void()>&& callback) -> IsCallbackRegistered
42{
43 return whenPromiseIsSettled(globalObject(), promise(), WTFMove(callback));
44}
45
46auto DOMPromise::whenPromiseIsSettled(JSDOMGlobalObject* globalObject, JSC::JSObject* promise, Function<void()>&& callback) -> IsCallbackRegistered
47{
48 auto& lexicalGlobalObject = *globalObject;
49 auto& vm = lexicalGlobalObject.vm();
50 JSLockHolder lock(vm);
51 auto* handler = JSC::JSNativeStdFunction::create(vm, globalObject, 1, String { }, [callback = WTFMove(callback)] (JSGlobalObject*, CallFrame*) mutable {
52 callback();
53 return JSC::JSValue::encode(JSC::jsUndefined());
54 });
55
56 auto scope = DECLARE_THROW_SCOPE(vm);
57 const JSC::Identifier& privateName = vm.propertyNames->builtinNames().thenPrivateName();
58 auto thenFunction = promise->get(&lexicalGlobalObject, privateName);
59
60 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
61 if (scope.exception())
62 return IsCallbackRegistered::No;
63
64 ASSERT(thenFunction.isCallable(vm));
65
66 JSC::MarkedArgumentBuffer arguments;
67 arguments.append(handler);
68 arguments.append(handler);
69
70 auto callData = JSC::getCallData(vm, thenFunction);
71 ASSERT(callData.type != JSC::CallData::Type::None);
72 call(&lexicalGlobalObject, thenFunction, callData, promise, arguments);
73
74 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
75 return scope.exception() ? IsCallbackRegistered::No : IsCallbackRegistered::Yes;
76}
77
78JSC::JSValue DOMPromise::result() const
79{
80 return promise()->result(m_globalObject->vm());
81}
82
83DOMPromise::Status DOMPromise::status() const
84{
85 switch (promise()->status(m_globalObject->vm())) {
86 case JSC::JSPromise::Status::Pending:
87 return Status::Pending;
88 case JSC::JSPromise::Status::Fulfilled:
89 return Status::Fulfilled;
90 case JSC::JSPromise::Status::Rejected:
91 return Status::Rejected;
92 };
93 ASSERT_NOT_REACHED();
94 return Status::Rejected;
95}
