45// https://html.spec.whatwg.org/multipage/window-object.html#dom-window-nameditem
46static bool jsDOMWindowPropertiesGetOwnPropertySlotNamedItemGetter(JSDOMWindowProperties* thisObject, DOMWindow& window, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
47{
48 if (auto* frame = window.frame()) {
49 if (auto* scopedChild = frame->tree().scopedChild(propertyNameToAtomString(propertyName))) {
50 slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontEnum), toJS(lexicalGlobalObject, scopedChild->document()->domWindow()));
51 return true;
52 }
53 }
54
55 if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, window, ThrowSecurityError))
56 return false;
57
58 // Allow shortcuts like 'Image1' instead of document.images.Image1
59 auto* document = window.document();
60 if (is<HTMLDocument>(document)) {
61 auto& htmlDocument = downcast<HTMLDocument>(*document);
62 auto* atomicPropertyName = propertyName.publicName();
63 if (atomicPropertyName && htmlDocument.hasWindowNamedItem(*atomicPropertyName)) {
64 JSValue namedItem;
65 if (UNLIKELY(htmlDocument.windowNamedItemContainsMultipleElements(*atomicPropertyName))) {
66 Ref<HTMLCollection> collection = document->windowNamedItems(atomicPropertyName);
67 ASSERT(collection->length() > 1);
68 namedItem = toJS(lexicalGlobalObject, thisObject->globalObject(), collection);
69 } else
70 namedItem = toJS(lexicalGlobalObject, thisObject->globalObject(), htmlDocument.windowNamedItem(*atomicPropertyName));
71 slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontEnum), namedItem);
72 return true;
73 }
74 }
75
76 return false;
77}

88bool JSDOMWindowProperties::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
89{
90 VM& vm = lexicalGlobalObject->vm();
91 auto* thisObject = jsCast<JSDOMWindowProperties*>(object);
92 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
93
94 if (Base::getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot))
95 return true;
96 JSObject* proto = asObject(thisObject->getPrototypeDirect(vm));
97 if (proto->hasProperty(lexicalGlobalObject, propertyName))
98 return false;
99
100 // FIXME: We should probably add support for JSRemoteDOMWindowBase too.
101 auto* jsWindow = jsDynamicCast<JSDOMWindowBase*>(vm, thisObject->globalObject());
102 if (!jsWindow)
103 return false;
104
105 auto& window = jsWindow->wrapped();
106 return jsDOMWindowPropertiesGetOwnPropertySlotNamedItemGetter(thisObject, window, lexicalGlobalObject, propertyName, slot);
107}
