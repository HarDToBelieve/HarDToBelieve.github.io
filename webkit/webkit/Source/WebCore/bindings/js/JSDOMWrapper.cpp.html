41JSDOMObject::JSDOMObject(JSC::Structure* structure, JSC::JSGlobalObject& globalObject)
42 : Base(globalObject.vm(), structure)
43{
44 ASSERT(scriptExecutionContext() || globalObject.classInfo(globalObject.vm()) == JSRemoteDOMWindow::info());
45}

47JSC::JSValue cloneAcrossWorlds(JSC::JSGlobalObject& lexicalGlobalObject, const JSDOMObject& owner, JSC::JSValue value)
48{
49 if (isWorldCompatible(lexicalGlobalObject, value))
50 return value;
51 // FIXME: Is it best to handle errors by returning null rather than throwing an exception?
52 auto serializedValue = SerializedScriptValue::create(lexicalGlobalObject, value, SerializationErrorMode::NonThrowing);
53 if (!serializedValue)
54 return JSC::jsNull();
55 // FIXME: Why is owner->globalObject() better than lexicalGlobalObject.lexicalGlobalObject() here?
56 // Unlike this, isWorldCompatible uses lexicalGlobalObject.lexicalGlobalObject(); should the two match?
57 return serializedValue->deserialize(lexicalGlobalObject, owner.globalObject());
58}
