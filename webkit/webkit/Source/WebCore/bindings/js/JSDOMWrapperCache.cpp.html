30Structure* getCachedDOMStructure(const JSDOMGlobalObject& globalObject, const ClassInfo* classInfo)
31{
32 return globalObject.structures().get(classInfo).get();
33}
34
35Structure* cacheDOMStructure(JSDOMGlobalObject& globalObject, Structure* structure, const ClassInfo* classInfo)
36{
37 auto addToStructures = [](JSDOMStructureMap& structures, JSDOMGlobalObject& globalObject, Structure* structure, const ClassInfo* classInfo) {
38 ASSERT(!structures.contains(classInfo));
39 return structures.set(classInfo, WriteBarrier<Structure>(globalObject.vm(), &globalObject, structure)).iterator->value.get();
40 };
41 if (globalObject.vm().heap.mutatorShouldBeFenced()) {
42 Locker locker { globalObject.gcLock() };
43 return addToStructures(globalObject.structures(), globalObject, structure, classInfo);
44 }
45 return addToStructures(globalObject.structures(NoLockingNecessary), globalObject, structure, classInfo);
46}
