35JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<DocumentFragment>&& impl)
36{
37 if (impl->isShadowRoot())
38 return createWrapper<ShadowRoot>(globalObject, WTFMove(impl));
39 return createWrapper<DocumentFragment>(globalObject, WTFMove(impl));
40}
41
42JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, DocumentFragment& impl)
43{
44 return wrap(lexicalGlobalObject, globalObject, impl);
45}
