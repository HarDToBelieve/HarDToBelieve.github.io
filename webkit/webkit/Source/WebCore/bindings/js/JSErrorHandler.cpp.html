63void JSErrorHandler::handleEvent(ScriptExecutionContext& scriptExecutionContext, Event& event)
64{
65 if (!is<ErrorEvent>(event))
66 return JSEventListener::handleEvent(scriptExecutionContext, event);
67
68 VM& vm = scriptExecutionContext.vm();
69 JSLockHolder lock(vm);
70
71 JSObject* jsFunction = this->ensureJSFunction(scriptExecutionContext);
72 if (!jsFunction)
73 return;
74
75 auto* globalObject = toJSDOMGlobalObject(scriptExecutionContext, isolatedWorld());
76 if (!globalObject)
77 return;
78
79 auto callData = getCallData(vm, jsFunction);
80 if (callData.type != CallData::Type::None) {
81 Ref<JSErrorHandler> protectedThis(*this);
82
83 RefPtr<Event> savedEvent;
84 auto* jsFunctionWindow = jsDynamicCast<JSDOMWindow*>(vm, jsFunction->globalObject());
85 if (jsFunctionWindow) {
86 savedEvent = jsFunctionWindow->currentEvent();
87 jsFunctionWindow->setCurrentEvent(&event);
88 }
89
90 auto& errorEvent = downcast<ErrorEvent>(event);
91
92 MarkedArgumentBuffer args;
93 args.append(toJS<IDLDOMString>(*globalObject, errorEvent.message()));
94 args.append(toJS<IDLUSVString>(*globalObject, errorEvent.filename()));
95 args.append(toJS<IDLUnsignedLong>(errorEvent.lineno()));
96 args.append(toJS<IDLUnsignedLong>(errorEvent.colno()));
97 args.append(errorEvent.error(*globalObject));
98 ASSERT(!args.hasOverflowed());
99
100 VM& vm = globalObject->vm();
101 VMEntryScope entryScope(vm, vm.entryScope ? vm.entryScope->globalObject() : globalObject);
102
103 JSExecState::instrumentFunction(&scriptExecutionContext, callData);
104
105 NakedPtr<JSC::Exception> exception;
106 JSValue returnValue = JSExecState::profiledCall(globalObject, JSC::ProfilingReason::Other, jsFunction, callData, globalObject, args, exception);
107
108 InspectorInstrumentation::didCallFunction(&scriptExecutionContext);
109
110 if (jsFunctionWindow)
111 jsFunctionWindow->setCurrentEvent(savedEvent.get());
112
113 if (exception)
114 reportException(globalObject, exception);
115 else {
116 if (returnValue.isTrue())
117 event.preventDefault();
118 }
119 }
120}
