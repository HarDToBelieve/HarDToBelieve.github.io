36namespace WebCore {
37
38class JSErrorHandler final : public JSEventListener {
39public:
40 static Ref<JSErrorHandler> create(JSC::JSObject& listener, JSC::JSObject& wrapper, bool isAttribute, DOMWrapperWorld&);
41 virtual ~JSErrorHandler();
42
43private:
44 JSErrorHandler(JSC::JSObject& listener, JSC::JSObject& wrapper, bool isAttribute, DOMWrapperWorld&);
45 void handleEvent(ScriptExecutionContext&, Event&) final;
46};
47
48// Creates a listener for "onerror" event handler.
49// It has custom implementation because, unlike other event listeners, it accepts three parameters.
50inline RefPtr<JSErrorHandler> createJSErrorHandler(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue listener, JSC::JSObject& wrapper)
51{
52 if (!listener.isObject())
53 return nullptr;
54 return JSErrorHandler::create(*asObject(listener), wrapper, true, currentWorld(lexicalGlobalObject));
55}
