100inline JSC::JSObject* JSEventListener::ensureJSFunction(ScriptExecutionContext& scriptExecutionContext) const
101{
102 // initializeJSFunction can trigger code that deletes this event listener
103 // before we're done. It should always return null in this case.
104 JSC::VM& vm = m_isolatedWorld->vm();
105 auto protect = makeRef(const_cast<JSEventListener&>(*this));
106 JSC::EnsureStillAliveScope protectedWrapper(m_wrapper.get());
107
108 if (!m_isInitialized) {
109 ASSERT(!m_jsFunction);
110 auto* function = initializeJSFunction(scriptExecutionContext);
111 if (function) {
112 m_jsFunction = JSC::Weak<JSC::JSObject>(function);
113 // When JSFunction is initialized, initializeJSFunction must ensure that m_wrapper should be initialized too.
114 ASSERT(m_wrapper);
115 vm.heap.writeBarrier(m_wrapper.get(), function);
116 m_isInitialized = true;
117 }
118 }
119
120 // m_wrapper and m_jsFunction are Weak<>. nullptr of these fields do not mean that this event-listener is not initialized yet.
121 // If this is initialized once, m_isInitialized should be true, and then m_wrapper and m_jsFunction must be alive. m_wrapper's
122 // liveness should be kept correctly by using ActiveDOMObject, output-constraints, etc. And m_jsFunction must be alive if m_wrapper
123 // is alive since JSEventListener marks m_jsFunction in JSEventListener::visitJSFunction if m_wrapper is alive.
124 // If the event-listener is not initialized yet, we should skip invoking this event-listener.
125 if (!m_isInitialized)
126 return nullptr;
127
128 ASSERT(m_wrapper);
129 ASSERT(m_jsFunction);
130 // Ensure m_jsFunction is live JSObject as a quick sanity check (while it is already ensured by Weak<>). If this fails, this is possibly JSC GC side's bug.
131 ASSERT(static_cast<JSC::JSCell*>(m_jsFunction.get())->isObject());
132
133 return m_jsFunction.get();
134}
