46JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<EventTarget>&& value)
47{
48 return createWrapper<EventTarget>(globalObject, WTFMove(value));
49}
50
51EventTarget* JSEventTarget::toWrapped(VM& vm, JSValue value)
52{
53 if (value.inherits<JSWindowProxy>(vm))
54 return &jsCast<JSWindowProxy*>(asObject(value))->wrapped();
55 if (value.inherits<JSDOMWindow>(vm))
56 return &jsCast<JSDOMWindow*>(asObject(value))->wrapped();
57 if (value.inherits<JSWorkerGlobalScope>(vm))
58 return &jsCast<JSWorkerGlobalScope*>(asObject(value))->wrapped();
59 if (value.inherits<JSEventTarget>(vm))
60 return &jsCast<JSEventTarget*>(asObject(value))->wrapped();
61 return nullptr;
62}
63
64std::unique_ptr<JSEventTargetWrapper> jsEventTargetCast(VM& vm, JSValue thisValue)
65{
66 if (auto* target = jsDynamicCast<JSEventTarget*>(vm, thisValue))
67 return makeUnique<JSEventTargetWrapper>(target->wrapped(), *target);
68 if (auto* window = toJSDOMGlobalObject<JSDOMWindow>(vm, thisValue))
69 return makeUnique<JSEventTargetWrapper>(window->wrapped(), *window);
70 if (auto* scope = toJSDOMGlobalObject<JSWorkerGlobalScope>(vm, thisValue))
71 return makeUnique<JSEventTargetWrapper>(scope->wrapped(), *scope);
72 return nullptr;
73}
