44EncodedJSValue constructJSHTMLElement(JSGlobalObject* lexicalGlobalObject, CallFrame& callFrame)
45{
46 VM& vm = lexicalGlobalObject->vm();
47 auto scope = DECLARE_THROW_SCOPE(vm);
48
49 auto* jsConstructor = jsCast<JSDOMConstructorBase*>(callFrame.jsCallee());
50 ASSERT(jsConstructor);
51
52 auto* context = jsConstructor->scriptExecutionContext();
53 if (!context)
54 return throwConstructorScriptExecutionContextUnavailableError(*lexicalGlobalObject, scope, "HTMLElement");
55 ASSERT(context->isDocument());
56
57 auto* newTarget = callFrame.newTarget().getObject();
58 auto* functionGlobalObject = getFunctionRealm(lexicalGlobalObject, newTarget);
59 RETURN_IF_EXCEPTION(scope, { });
60 auto* newTargetGlobalObject = jsCast<JSDOMGlobalObject*>(functionGlobalObject);
61 JSValue htmlElementConstructorValue = JSHTMLElement::getConstructor(vm, newTargetGlobalObject);
62 if (newTarget == htmlElementConstructorValue)
63 return throwVMTypeError(lexicalGlobalObject, scope, "new.target is not a valid custom element constructor"_s);
64
65 auto& document = downcast<Document>(*context);
66
67 auto* window = document.domWindow();
68 if (!window)
69 return throwVMTypeError(lexicalGlobalObject, scope, "new.target is not a valid custom element constructor"_s);
70
71 auto* registry = window->customElementRegistry();
72 if (!registry)
73 return throwVMTypeError(lexicalGlobalObject, scope, "new.target is not a valid custom element constructor"_s);
74
75 auto* elementInterface = registry->findInterface(newTarget);
76 if (!elementInterface)
77 return throwVMTypeError(lexicalGlobalObject, scope, "new.target does not define a custom element"_s);
78
79 if (!elementInterface->isUpgradingElement()) {
80 Structure* baseStructure = getDOMStructure<JSHTMLElement>(vm, *newTargetGlobalObject);
81 auto* newElementStructure = InternalFunction::createSubclassStructure(lexicalGlobalObject, newTarget, baseStructure);
82 RETURN_IF_EXCEPTION(scope, { });
83
84 Ref<HTMLElement> element = HTMLElement::create(elementInterface->name(), document);
85 element->setIsDefinedCustomElement(*elementInterface);
86 auto* jsElement = JSHTMLElement::create(newElementStructure, newTargetGlobalObject, element.get());
87 cacheWrapper(newTargetGlobalObject->world(), element.ptr(), jsElement);
88 return JSValue::encode(jsElement);
89 }
90
91 Element* elementToUpgrade = elementInterface->lastElementInConstructionStack();
92 if (!elementToUpgrade) {
93 throwTypeError(lexicalGlobalObject, scope, "Cannot instantiate a custom element inside its own constructor during upgrades"_s);
94 return JSValue::encode(jsUndefined());
95 }
96
97 JSValue elementWrapperValue = toJS(lexicalGlobalObject, jsConstructor->globalObject(), *elementToUpgrade);
98 ASSERT(elementWrapperValue.isObject());
99
100 JSValue newPrototype = newTarget->get(lexicalGlobalObject, vm.propertyNames->prototype);
101 RETURN_IF_EXCEPTION(scope, encodedJSValue());
102
103 JSObject* elementWrapperObject = asObject(elementWrapperValue);
104 JSObject::setPrototype(elementWrapperObject, lexicalGlobalObject, newPrototype, true /* shouldThrowIfCantSet */);
105 RETURN_IF_EXCEPTION(scope, encodedJSValue());
106
107 elementInterface->didUpgradeLastElementInConstructionStack();
108
109 return JSValue::encode(elementWrapperValue);
110}

112JSScope* JSHTMLElement::pushEventHandlerScope(JSGlobalObject* lexicalGlobalObject, JSScope* scope) const
113{
114 HTMLElement& element = wrapped();
115
116 // The document is put on first, fall back to searching it only after the element and form.
117 // FIXME: This probably may use the wrong global object. If this is called from a native
118 // function, then it would be correct but not optimal since the native function would *know*
119 // the global object. But, it may be that globalObject() is more correct.
120 // https://bugs.webkit.org/show_bug.cgi?id=134932
121 VM& vm = lexicalGlobalObject->vm();
122
123 scope = JSWithScope::create(vm, lexicalGlobalObject, scope, asObject(toJS(lexicalGlobalObject, globalObject(), element.document())));
124
125 // The form is next, searched before the document, but after the element itself.
126 if (HTMLFormElement* form = element.form())
127 scope = JSWithScope::create(vm, lexicalGlobalObject, scope, asObject(toJS(lexicalGlobalObject, globalObject(), *form)));
128
129 // The element is on top, searched first.
130 return JSWithScope::create(vm, lexicalGlobalObject, scope, asObject(toJS(lexicalGlobalObject, globalObject(), element)));
131}
