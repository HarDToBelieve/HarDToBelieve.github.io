28#include "EmptyScriptExecutionContext.h"
29#include "JSDOMGlobalObject.h"

31namespace WebCore {
32
33class JSIDBSerializationGlobalObject final : public JSDOMGlobalObject {
34public:
35 using Base = JSDOMGlobalObject;
36 static JSIDBSerializationGlobalObject* create(JSC::VM&, JSC::Structure*, Ref<DOMWrapperWorld>&&);
37
38 template<typename, JSC::SubspaceAccess mode>
39 static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
40 {
41 if constexpr (mode == JSC::SubspaceAccess::Concurrently)
42 return nullptr;
43 return subspaceForImpl(vm);
44 }
45 static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
46
47 DECLARE_INFO;
48 static JSC::Structure* createStructure(JSC::VM& vm, JSC::JSValue prototype)
49 {
50 return JSC::Structure::create(vm, 0, prototype, JSC::TypeInfo(JSC::GlobalObjectType, StructureFlags), info());
51 }
52 static void destroy(JSC::JSCell*);
53
54 ScriptExecutionContext* scriptExecutionContext() const { return m_scriptExecutionContext.ptr(); }
55
56private:
57 JSIDBSerializationGlobalObject(JSC::VM&, JSC::Structure*, Ref<DOMWrapperWorld>&&);
58 void finishCreation(JSC::VM&);
59
60 Ref<EmptyScriptExecutionContext> m_scriptExecutionContext;
61};

63} // namespace WebCore
