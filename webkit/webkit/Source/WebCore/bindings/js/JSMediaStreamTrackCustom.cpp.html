36JSC::JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<MediaStreamTrack>&& object)
37{
38 if (is<CanvasCaptureMediaStreamTrack>(object))
39 return createWrapper<CanvasCaptureMediaStreamTrack>(globalObject, WTFMove(object));
40 return createWrapper<MediaStreamTrack>(globalObject, WTFMove(object));
41}

43JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, MediaStreamTrack& track)
44{
45 return wrap(lexicalGlobalObject, globalObject, track);
46}
