46JSC::JSValue JSMessageEvent::ports(JSC::JSGlobalObject& lexicalGlobalObject) const
47{
48 auto throwScope = DECLARE_THROW_SCOPE(lexicalGlobalObject.vm());
49 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().cachedPorts(), [&] {
50 JSC::JSValue ports = toJS<IDLFrozenArray<IDLInterface<MessagePort>>>(lexicalGlobalObject, *globalObject(), throwScope, wrapped().ports());
51 return ports;
52 });
53}
54
55JSC::JSValue JSMessageEvent::data(JSC::JSGlobalObject& lexicalGlobalObject) const
56{
57 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().cachedData(), [this, &lexicalGlobalObject] {
58 return WTF::switchOn(wrapped().data(), [] (JSC::JSValue data) {
59 return data ? data : JSC::jsNull();
60 }, [this, &lexicalGlobalObject] (const Ref<SerializedScriptValue>& data) {
61 // FIXME: Is it best to handle errors by returning null rather than throwing an exception?
62 return data->deserialize(lexicalGlobalObject, globalObject(), wrapped().ports(), SerializationErrorMode::NonThrowing);
63 }, [&lexicalGlobalObject] (const String& data) {
64 return toJS<IDLDOMString>(lexicalGlobalObject, data);
65 }, [this, &lexicalGlobalObject] (const Ref<Blob>& data) {
66 return toJS<IDLInterface<Blob>>(lexicalGlobalObject, *globalObject(), data);
67 }, [this, &lexicalGlobalObject] (const Ref<ArrayBuffer>& data) {
68 return toJS<IDLInterface<ArrayBuffer>>(lexicalGlobalObject, *globalObject(), data);
69 });
70 });
71}

73template<typename Visitor>
74void JSMessageEvent::visitAdditionalChildren(Visitor& visitor)
75{
76 WTF::switchOn(wrapped().data(), [&visitor] (const JSValueInWrappedObject& data) {
77 data.visit(visitor);
78 }, [] (const Ref<SerializedScriptValue>&) {
79 }, [] (const String&) {
80 }, [] (const Ref<Blob>&) {
81 }, [] (const Ref<ArrayBuffer>&) {
82 });
83
84 wrapped().cachedData().visit(visitor);
85 wrapped().cachedPorts().visit(visitor);
86}
