43#if ENABLE(MEDIA_STREAM)
44JSC::JSValue JSNavigator::getUserMedia(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame)
45{
46 auto* function = globalObject()->builtinInternalFunctions().jsDOMBindingInternals().m_getUserMediaShimFunction.get();
47 ASSERT(function);
48
49 auto callData = JSC::getCallData(lexicalGlobalObject.vm(), function);
50 ASSERT(callData.type != JSC::CallData::Type::None);
51 JSC::MarkedArgumentBuffer arguments;
52 for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
53 arguments.append(callFrame.uncheckedArgument(cptr));
54 ASSERT(!arguments.hasOverflowed());
55 JSC::call(&lexicalGlobalObject, function, callData, this, arguments);
56 return JSC::jsUndefined();
57}
58#endif
