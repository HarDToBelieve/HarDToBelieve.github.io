77static inline bool isReachableFromDOM(Node* node, AbstractSlotVisitor& visitor, const char** reason)
78{
79 if (!node->isConnected()) {
80 if (is<Element>(*node)) {
81 auto& element = downcast<Element>(*node);
82
83 // If a wrapper is the last reference to an image element
84 // that is loading but not in the document, the wrapper is observable
85 // because it is the only thing keeping the image element alive, and if
86 // the element is destroyed, its load event will not fire.
87 // FIXME: The DOM should manage this issue without the help of JavaScript wrappers.
88 if (is<HTMLImageElement>(element)) {
89 if (downcast<HTMLImageElement>(element).hasPendingActivity()) {
90 if (UNLIKELY(reason))
91 *reason = "Image element with pending activity";
92 return true;
93 }
94 }
95#if ENABLE(VIDEO)
96 else if (is<HTMLAudioElement>(element)) {
97 if (!downcast<HTMLAudioElement>(element).paused()) {
98 if (UNLIKELY(reason))
99 *reason = "Audio element which is not paused";
100 return true;
101 }
102 }
103#endif
104 }
105
106 // If a node is firing event listeners, its wrapper is observable because
107 // its wrapper is responsible for marking those event listeners.
108 if (node->isFiringEventListeners()) {
109 if (UNLIKELY(reason))
110 *reason = "Node which is firing event listeners";
111 return true;
112 }
113 if (GCReachableRefMap::contains(*node)) {
114 if (UNLIKELY(reason))
115 *reason = "Node is scheduled to be used in an async script invocation)";
116 return true;
117 }
118 }
119
120 if (UNLIKELY(reason))
121 *reason = "Connected node";
122
123 return visitor.containsOpaqueRoot(root(node));
124}

126bool JSNodeOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, AbstractSlotVisitor& visitor, const char** reason)
127{
128 JSNode* jsNode = jsCast<JSNode*>(handle.slot()->asCell());
129 return isReachableFromDOM(&jsNode->wrapped(), visitor, reason);
130}

147static ALWAYS_INLINE JSValue createWrapperInline(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Node>&& node)
148{
149 ASSERT(!getCachedWrapper(globalObject->world(), node));
150
151 JSDOMObject* wrapper;
152 switch (node->nodeType()) {
153 case Node::ELEMENT_NODE:
154 if (is<HTMLElement>(node))
155 wrapper = createJSHTMLWrapper(globalObject, static_reference_cast<HTMLElement>(WTFMove(node)));
156 else if (is<SVGElement>(node))
157 wrapper = createJSSVGWrapper(globalObject, static_reference_cast<SVGElement>(WTFMove(node)));
158#if ENABLE(MATHML)
159 else if (is<MathMLElement>(node))
160 wrapper = createJSMathMLWrapper(globalObject, static_reference_cast<MathMLElement>(WTFMove(node)));
161#endif
162 else
163 wrapper = createWrapper<Element>(globalObject, WTFMove(node));
164 break;
165 case Node::ATTRIBUTE_NODE:
166 wrapper = createWrapper<Attr>(globalObject, WTFMove(node));
167 break;
168 case Node::TEXT_NODE:
169 wrapper = createWrapper<Text>(globalObject, WTFMove(node));
170 break;
171 case Node::CDATA_SECTION_NODE:
172 wrapper = createWrapper<CDATASection>(globalObject, WTFMove(node));
173 break;
174 case Node::PROCESSING_INSTRUCTION_NODE:
175 wrapper = createWrapper<ProcessingInstruction>(globalObject, WTFMove(node));
176 break;
177 case Node::COMMENT_NODE:
178 wrapper = createWrapper<Comment>(globalObject, WTFMove(node));
179 break;
180 case Node::DOCUMENT_NODE:
181 // we don't want to cache the document itself in the per-document dictionary
182 return toJS(lexicalGlobalObject, globalObject, downcast<Document>(node.get()));
183 case Node::DOCUMENT_TYPE_NODE:
184 wrapper = createWrapper<DocumentType>(globalObject, WTFMove(node));
185 break;
186 case Node::DOCUMENT_FRAGMENT_NODE:
187 if (node->isShadowRoot())
188 wrapper = createWrapper<ShadowRoot>(globalObject, WTFMove(node));
189 else
190 wrapper = createWrapper<DocumentFragment>(globalObject, WTFMove(node));
191 break;
192 default:
193 wrapper = createWrapper<Node>(globalObject, WTFMove(node));
194 }
195
196 return wrapper;
197}

199JSValue createWrapper(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Node>&& node)
200{
201 return createWrapperInline(lexicalGlobalObject, globalObject, WTFMove(node));
202}

204JSValue toJSNewlyCreated(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Node>&& node)
205{
206 return createWrapperInline(lexicalGlobalObject, globalObject, WTFMove(node));
207}

210{
211 ASSERT(!globalObject->world().isNormal());

215void willCreatePossiblyOrphanedTreeByRemovalSlowCase(Node* root)
216{
217 JSC::JSGlobalObject* lexicalGlobalObject = mainWorldExecState(root->document().frame());
218 if (!lexicalGlobalObject)
219 return;
220
221 JSLockHolder lock(lexicalGlobalObject);
222 toJS(lexicalGlobalObject, static_cast<JSDOMGlobalObject*>(lexicalGlobalObject), *root);
223}
