34template<typename Visitor>
35void JSPaintWorkletGlobalScope::visitAdditionalChildren(Visitor& visitor)
36{
37 Locker locker { wrapped().paintDefinitionLock() };
38 for (auto& registered : wrapped().paintDefinitionMap().values()) {
39 registered->paintCallback->visitJSFunction(visitor);
40 visitor.appendUnbarriered(registered->paintConstructor);
41 }
42}
