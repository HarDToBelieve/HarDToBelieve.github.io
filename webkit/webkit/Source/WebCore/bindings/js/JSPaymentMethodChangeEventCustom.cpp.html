33JSC::JSValue JSPaymentMethodChangeEvent::methodDetails(JSC::JSGlobalObject& lexicalGlobalObject) const
34{
35 return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().cachedMethodDetails(), [this, &lexicalGlobalObject] {
36 return WTF::switchOn(wrapped().methodDetails(), [](JSC::JSValue methodDetails) {
37 return methodDetails ? methodDetails : JSC::jsNull();
38 }, [&lexicalGlobalObject](const PaymentMethodChangeEvent::MethodDetailsFunction& function) {
39 return function(lexicalGlobalObject).get();
40 });
41 });
42}
43
44template<typename Visitor>
45void JSPaymentMethodChangeEvent::visitAdditionalChildren(Visitor& visitor)
46{
47 WTF::switchOn(wrapped().methodDetails(), [&visitor](const JSValueInWrappedObject& methodDetails) {
48 methodDetails.visit(visitor);
49 }, [](const PaymentMethodChangeEvent::MethodDetailsFunction&) {
50 });
51
52 wrapped().cachedMethodDetails().visit(visitor);
53}
