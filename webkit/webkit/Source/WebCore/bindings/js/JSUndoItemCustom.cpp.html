32template<typename Visitor>
33void JSUndoItem::visitAdditionalChildren(Visitor& visitor)
34{
35 wrapped().undoHandler().visitJSFunction(visitor);
36 wrapped().redoHandler().visitJSFunction(visitor);
37}
38
39DEFINE_VISIT_ADDITIONAL_CHILDREN(JSUndoItem);
40
41bool JSUndoItemOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, JSC::AbstractSlotVisitor& visitor, const char** reason)
42{
43 if (UNLIKELY(reason))
44 *reason = "Document is an opaque root.";
45
46 auto* documentForUndoItem = JSC::jsCast<JSUndoItem*>(handle.slot()->asCell())->wrapped().document();
47 return documentForUndoItem && visitor.containsOpaqueRoot(documentForUndoItem);
48}
