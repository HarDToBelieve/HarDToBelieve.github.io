37JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<WebXRReferenceSpace>&& space)
38{
39 if (space->isBoundedReferenceSpace())
40 return createWrapper<WebXRBoundedReferenceSpace>(globalObject, WTFMove(space));
41 return createWrapper<WebXRReferenceSpace>(globalObject, WTFMove(space));
42}
43
44JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, WebXRReferenceSpace& space)
45{
46 return wrap(lexicalGlobalObject, globalObject, space);
47}
