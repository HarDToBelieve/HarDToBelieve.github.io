136void JSWorkerGlobalScopeBase::queueMicrotaskToEventLoop(JSGlobalObject& object, Ref<JSC::Microtask>&& task)
137{
138 JSWorkerGlobalScopeBase& thisObject = static_cast<JSWorkerGlobalScopeBase&>(object);
139
140 auto callback = JSMicrotaskCallback::create(thisObject, WTFMove(task));
141 auto& context = thisObject.wrapped();
142 context.eventLoop().queueMicrotask([callback = WTFMove(callback)]() mutable {
143 callback->call();
144 });
145}

151
152JSValue toJS(JSGlobalObject*, WorkerGlobalScope& workerGlobalScope)
153{
154 auto* script = workerGlobalScope.script();
155 if (!script)
156 return jsNull();
157 auto* contextWrapper = script->globalScopeWrapper();
158 ASSERT(contextWrapper);
159 return &contextWrapper->proxy();
160}
