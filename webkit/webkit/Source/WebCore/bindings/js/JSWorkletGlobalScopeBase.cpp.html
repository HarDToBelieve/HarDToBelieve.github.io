76void JSWorkletGlobalScopeBase::finishCreation(VM& vm, JSProxy* proxy)

91}

136
137JSValue toJS(JSGlobalObject*, WorkletGlobalScope& workletGlobalScope)
138{
139 if (!workletGlobalScope.script())
140 return jsUndefined();
141 auto* contextWrapper = workletGlobalScope.script()->globalScopeWrapper();
142 if (!contextWrapper)
143 return jsUndefined();
144 return &contextWrapper->proxy();
145}
