37JSValue toJSNewlyCreated(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<XMLDocument>&& document)
38{
39 reportMemoryForDocumentIfFrameless(*lexicalGlobalObject, document.get());
40
41 return createWrapper<XMLDocument>(globalObject, WTFMove(document));
42}
43
44JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, XMLDocument& document)
45{
46 if (auto* wrapper = cachedDocumentWrapper(*lexicalGlobalObject, *globalObject, document))
47 return wrapper;
48 return toJSNewlyCreated(lexicalGlobalObject, globalObject, Ref<XMLDocument>(document));
49}
