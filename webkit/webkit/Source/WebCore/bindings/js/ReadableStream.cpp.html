40ExceptionOr<Ref<ReadableStream>> ReadableStream::create(JSC::JSGlobalObject& lexicalGlobalObject, RefPtr<ReadableStreamSource>&& source)
41{
42 VM& vm = lexicalGlobalObject.vm();
43 auto scope = DECLARE_THROW_SCOPE(vm);
44
45 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
46 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject);
47
48 auto* constructor = JSC::asObject(globalObject.get(&lexicalGlobalObject, clientData.builtinNames().ReadableStreamPrivateName()));
49
50 auto constructData = getConstructData(vm, constructor);
51 ASSERT(constructData.type != CallData::Type::None);
52
53 MarkedArgumentBuffer args;
54 args.append(source ? toJSNewlyCreated(&lexicalGlobalObject, &globalObject, source.releaseNonNull()) : JSC::jsUndefined());
55 ASSERT(!args.hasOverflowed());
56
57 JSObject* object = JSC::construct(&lexicalGlobalObject, constructor, constructData, args);
58 ASSERT(!!scope.exception() == !object);
59 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
60
61 return create(globalObject, *jsCast<JSReadableStream*>(object));
62}

64static inline std::optional<JSC::JSValue> invokeReadableStreamFunction(JSC::JSGlobalObject& lexicalGlobalObject, const JSC::Identifier& identifier, JSC::JSValue thisValue, const JSC::MarkedArgumentBuffer& arguments)
65{
66 JSC::VM& vm = lexicalGlobalObject.vm();
67 JSC::JSLockHolder lock(vm);
68
69 auto function = lexicalGlobalObject.get(&lexicalGlobalObject, identifier);
70 ASSERT(function.isCallable(vm));
71
72 auto scope = DECLARE_CATCH_SCOPE(vm);
73 auto callData = JSC::getCallData(vm, function);
74 auto result = call(&lexicalGlobalObject, function, callData, thisValue, arguments);
75 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
76 if (scope.exception())
77 return { };
78 return result;
79}

81void ReadableStream::pipeTo(ReadableStreamSink& sink)
82{
83 auto& lexicalGlobalObject = *m_globalObject;
84 auto* clientData = static_cast<JSVMClientData*>(lexicalGlobalObject.vm().clientData);
85 auto& privateName = clientData->builtinFunctions().readableStreamInternalsBuiltins().readableStreamPipeToPrivateName();
86
87 MarkedArgumentBuffer arguments;
88 arguments.append(readableStream());
89 arguments.append(toJS(&lexicalGlobalObject, m_globalObject.get(), sink));
90 ASSERT(!arguments.hasOverflowed());
91 invokeReadableStreamFunction(lexicalGlobalObject, privateName, JSC::jsUndefined(), arguments);
92}

94std::optional<std::pair<Ref<ReadableStream>, Ref<ReadableStream>>> ReadableStream::tee()
95{
96 auto& lexicalGlobalObject = *m_globalObject;
97 auto* clientData = static_cast<JSVMClientData*>(lexicalGlobalObject.vm().clientData);
98 auto& privateName = clientData->builtinFunctions().readableStreamInternalsBuiltins().readableStreamTeePrivateName();
99
100 MarkedArgumentBuffer arguments;
101 arguments.append(readableStream());
102 arguments.append(JSC::jsBoolean(true));
103 ASSERT(!arguments.hasOverflowed());
104 auto returnedValue = invokeReadableStreamFunction(lexicalGlobalObject, privateName, JSC::jsUndefined(), arguments);
105 if (!returnedValue)
106 return { };
107
108 auto results = Detail::SequenceConverter<IDLInterface<ReadableStream>>::convert(lexicalGlobalObject, *returnedValue);
109
110 ASSERT(results.size() == 2);
111 return std::make_pair(results[0].releaseNonNull(), results[1].releaseNonNull());
112}

114void ReadableStream::lock()
115{
116 auto& lexicalGlobalObject = *m_globalObject;
117 auto& vm = lexicalGlobalObject.vm();
118#if ENABLE(EXCEPTION_SCOPE_VERIFICATION)
119 auto scope = DECLARE_CATCH_SCOPE(vm);
120#endif
121
122 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
123
124 auto* constructor = JSC::asObject(m_globalObject->get(&lexicalGlobalObject, clientData.builtinNames().ReadableStreamDefaultReaderPrivateName()));
125
126 auto constructData = getConstructData(vm, constructor);
127 ASSERT(constructData.type != CallData::Type::None);
128
129 MarkedArgumentBuffer args;
130 args.append(readableStream());
131 ASSERT(!args.hasOverflowed());
132
133 JSC::construct(&lexicalGlobalObject, constructor, constructData, args);
134 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
135}

137static inline bool checkReadableStream(JSDOMGlobalObject& globalObject, JSReadableStream* readableStream, JSC::JSValue function)
138{
139 auto& lexicalGlobalObject = globalObject;
140
141 ASSERT(function);
142 JSC::MarkedArgumentBuffer arguments;
143 arguments.append(readableStream);
144 ASSERT(!arguments.hasOverflowed());
145
146 auto& vm = lexicalGlobalObject.vm();
147 auto scope = DECLARE_CATCH_SCOPE(vm);
148 auto callData = JSC::getCallData(vm, function);
149 ASSERT(callData.type != JSC::CallData::Type::None);
150
151 auto result = call(&lexicalGlobalObject, function, callData, JSC::jsUndefined(), arguments);
152 EXCEPTION_ASSERT(!scope.exception() || vm.hasPendingTerminationException());
153
154 return result.isTrue() || scope.exception();
155}

157bool ReadableStream::isLocked() const
158{
159 return checkReadableStream(*globalObject(), readableStream(), globalObject()->builtinInternalFunctions().readableStreamInternals().m_isReadableStreamLockedFunction.get());
160}

162bool ReadableStream::isDisturbed() const

165}

167bool ReadableStream::isDisturbed(JSGlobalObject& lexicalGlobalObject, JSValue value)
168{
169 auto& vm = lexicalGlobalObject.vm();
170 auto& globalObject = *jsDynamicCast<JSDOMGlobalObject*>(vm, &lexicalGlobalObject);
171 auto* readableStream = jsDynamicCast<JSReadableStream*>(vm, value);
172
173 return checkReadableStream(globalObject, readableStream, globalObject.builtinInternalFunctions().readableStreamInternals().m_isReadableStreamDisturbedFunction.get());
174}
