32#include "JSDOMConvertBufferSource.h"
33#include "JSReadableStreamDefaultController.h"
34#include <JavaScriptCore/JSCJSValue.h>
35#include <JavaScriptCore/JSCJSValueInlines.h>

37
38namespace WebCore {
39
40class ReadableStreamSource;
41
42class ReadableStreamDefaultController {
43public:
44 explicit ReadableStreamDefaultController(JSReadableStreamDefaultController* controller) : m_jsController(controller) { }
45
46 bool enqueue(RefPtr<JSC::ArrayBuffer>&&);
47 bool enqueue(JSC::JSValue);
48 void error(const Exception&);
49 void close();
50
51private:
52 JSReadableStreamDefaultController& jsController() const;
53
54 JSDOMGlobalObject& globalObject() const;
55
56 // The owner of ReadableStreamDefaultController is responsible to keep uncollected the JSReadableStreamDefaultController.
57 JSReadableStreamDefaultController* m_jsController { nullptr };
58};
59
60inline JSReadableStreamDefaultController& ReadableStreamDefaultController::jsController() const
61{
62 ASSERT(m_jsController);
63 return *m_jsController;
64}
65
66inline JSDOMGlobalObject& ReadableStreamDefaultController::globalObject() const
67{
68 ASSERT(m_jsController);
69 ASSERT(m_jsController->globalObject());
70 return *static_cast<JSDOMGlobalObject*>(m_jsController->globalObject());
71}

73} // namespace WebCore
