91void ScheduledAction::executeFunctionInContext(JSGlobalObject* globalObject, JSValue thisValue, ScriptExecutionContext& context)
92{
93 ASSERT(m_function);
94 VM& vm = context.vm();
95 JSLockHolder lock(vm);
96 auto catchScope = DECLARE_CATCH_SCOPE(vm);
97
98 auto callData = getCallData(vm, m_function.get());
99 if (callData.type == CallData::Type::None)
100 return;
101
102 JSGlobalObject* lexicalGlobalObject = globalObject;
103
104 MarkedArgumentBuffer arguments;
105 for (auto& argument : m_arguments)
106 arguments.append(argument.get());
107 if (UNLIKELY(arguments.hasOverflowed())) {
108 {
109 auto throwScope = DECLARE_THROW_SCOPE(vm);
110 throwOutOfMemoryError(lexicalGlobalObject, throwScope);
111 }
112 NakedPtr<JSC::Exception> exception = catchScope.exception();
113 catchScope.clearException();
114 reportException(lexicalGlobalObject, exception);
115 return;
116 }
117
118 JSExecState::instrumentFunction(&context, callData);
119
120 NakedPtr<JSC::Exception> exception;
121 JSExecState::profiledCall(lexicalGlobalObject, JSC::ProfilingReason::Other, m_function.get(), callData, thisValue, arguments, exception);
122 EXCEPTION_ASSERT(!catchScope.exception());
123
124 InspectorInstrumentation::didCallFunction(&context);
125
126 if (exception)
127 reportException(lexicalGlobalObject, exception);
128}

130void ScheduledAction::execute(Document& document)
131{
132 JSDOMWindow* window = toJSDOMWindow(document.frame(), m_isolatedWorld);
133 if (!window)
134 return;
135
136 RefPtr<Frame> frame = window->wrapped().frame();
137 if (!frame || !frame->script().canExecuteScripts(AboutToExecuteScript))
138 return;
139
140 if (m_function)
141 executeFunctionInContext(window, &window->proxy(), document);
142 else
143 frame->script().executeScriptInWorldIgnoringException(m_isolatedWorld, m_code);
144}
145
146void ScheduledAction::execute(WorkerGlobalScope& workerGlobalScope)
147{
148 // In a Worker, the execution should always happen on a worker thread.
149 ASSERT(workerGlobalScope.thread().thread() == &Thread::current());
150
151 auto* scriptController = workerGlobalScope.script();
152
153 if (m_function) {
154 auto* contextWrapper = scriptController->globalScopeWrapper();
155 executeFunctionInContext(contextWrapper, contextWrapper, workerGlobalScope);
156 } else {
157 ScriptSourceCode code(m_code, URL(workerGlobalScope.url()));
158 scriptController->evaluate(code);
159 }
160}
