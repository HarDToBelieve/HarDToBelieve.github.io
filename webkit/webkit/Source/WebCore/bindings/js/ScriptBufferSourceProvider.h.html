26#pragma once
27
28#include "ScriptBuffer.h"
29#include <JavaScriptCore/SourceProvider.h>
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class ScriptBufferSourceProvider final : public JSC::SourceProvider, public CanMakeWeakPtr<ScriptBufferSourceProvider> {
35 WTF_MAKE_FAST_ALLOCATED;
36public:
37 static Ref<ScriptBufferSourceProvider> create(const ScriptBuffer& scriptBuffer, const JSC::SourceOrigin& sourceOrigin, String sourceURL, const TextPosition& startPosition = TextPosition(), JSC::SourceProviderSourceType sourceType = JSC::SourceProviderSourceType::Program)
38 {
39 return adoptRef(*new ScriptBufferSourceProvider(scriptBuffer, sourceOrigin, WTFMove(sourceURL), startPosition, sourceType));
40 }
41
42 unsigned hash() const final
43 {
44 if (!m_scriptHash)
45 source();
46 return m_scriptHash;
47 }
48
49 StringView source() const final
50 {
51 if (m_scriptBuffer.isEmpty())
52 return emptyString();
53
54 if (!m_containsOnlyASCII) {
55 m_containsOnlyASCII = charactersAreAllASCII(m_scriptBuffer.buffer()->data(), m_scriptBuffer.buffer()->size());
56 if (*m_containsOnlyASCII)
57 m_scriptHash = StringHasher::computeHashAndMaskTop8Bits(m_scriptBuffer.buffer()->data(), m_scriptBuffer.buffer()->size());
58 }
59 if (*m_containsOnlyASCII)
60 return { m_scriptBuffer.buffer()->data(), static_cast<unsigned>(m_scriptBuffer.buffer()->size()) };
61
62 if (!m_cachedScriptString) {
63 m_cachedScriptString = m_scriptBuffer.toString();
64 if (!m_scriptHash)
65 m_scriptHash = m_cachedScriptString.impl()->hash();
66 }
67
68 return m_cachedScriptString;
69 }
70
71 void clearDecodedData()
72 {
73 m_cachedScriptString = String();
74 }
75
76 void tryReplaceScriptBuffer(const ScriptBuffer& scriptBuffer)
77 {
78 // If this new file-mapped script buffer is identical to the one we have, then replace
79 // ours to save dirty memory.
80 if (m_scriptBuffer != scriptBuffer)
81 return;
82
83 m_scriptBuffer = scriptBuffer;
84 }
85
86private:
87 ScriptBufferSourceProvider(const ScriptBuffer& scriptBuffer, const JSC::SourceOrigin& sourceOrigin, String&& sourceURL, const TextPosition& startPosition, JSC::SourceProviderSourceType sourceType)
88 : JSC::SourceProvider(sourceOrigin, WTFMove(sourceURL), startPosition, sourceType)
89 , m_scriptBuffer(scriptBuffer)
90 {
91 }
92
93 ScriptBuffer m_scriptBuffer;
94 mutable unsigned m_scriptHash { 0 };
95 mutable String m_cachedScriptString;
96 mutable std::optional<bool> m_containsOnlyASCII;
97};
98
99} // namespace WebCore
