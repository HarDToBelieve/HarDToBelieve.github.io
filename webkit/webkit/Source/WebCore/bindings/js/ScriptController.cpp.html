125ValueOrException ScriptController::evaluateInWorld(const ScriptSourceCode& sourceCode, DOMWrapperWorld& world)
126{
127 JSLockHolder lock(world.vm());
128
129 const SourceCode& jsSourceCode = sourceCode.jsSourceCode();
130 const URL& sourceURL = jsSourceCode.provider()->sourceOrigin().url();
131
132 // evaluate code. Returns the JS return value or 0
133 // if there was none, an error occurred or the type couldn't be converted.
134
135 // inlineCode is true for <a href="javascript:doSomething()">
136 // and false for <script>doSomething()</script>. Check if it has the
137 // expected value in all cases.
138 // See smart window.open policy for where this is used.
139 auto& proxy = jsWindowProxy(world);
140 auto& globalObject = *proxy.window();
141
142 Ref<Frame> protector(m_frame);
143 SetForScope<const URL*> sourceURLScope(m_sourceURL, &sourceURL);
144
145 InspectorInstrumentation::willEvaluateScript(m_frame, sourceURL.string(), sourceCode.startLine(), sourceCode.startColumn());
146
147 NakedPtr<JSC::Exception> evaluationException;
148 JSValue returnValue = JSExecState::profiledEvaluate(&globalObject, JSC::ProfilingReason::Other, jsSourceCode, &proxy, evaluationException);
149
150 InspectorInstrumentation::didEvaluateScript(m_frame);
151
152 std::optional<ExceptionDetails> optionalDetails;
153 if (evaluationException) {
154 ExceptionDetails details;
155 reportException(&globalObject, evaluationException, sourceCode.cachedScript(), &details);
156 optionalDetails = WTFMove(details);
157 }
158
159 if (optionalDetails)
160 return makeUnexpected(*optionalDetails);
161
162 return returnValue;
163}

202JSC::JSValue ScriptController::linkAndEvaluateModuleScriptInWorld(LoadableModuleScript& moduleScript, DOMWrapperWorld& world)
203{
204 JSC::VM& vm = world.vm();
205 JSLockHolder lock(vm);
206
207 auto& proxy = jsWindowProxy(world);
208 auto& lexicalGlobalObject = *proxy.window();
209
210 // FIXME: Preventing Frame from being destroyed is essentially unnecessary.
211 // https://bugs.webkit.org/show_bug.cgi?id=164763
212 Ref<Frame> protector(m_frame);
213
214 NakedPtr<JSC::Exception> evaluationException;
215 auto returnValue = JSExecState::linkAndEvaluateModule(lexicalGlobalObject, Identifier::fromUid(vm, moduleScript.moduleKey()), jsUndefined(), evaluationException);
216 if (evaluationException) {
217 // FIXME: Give a chance to dump the stack trace if the "crossorigin" attribute allows.
218 // https://bugs.webkit.org/show_bug.cgi?id=164539
219 reportException(&lexicalGlobalObject, evaluationException, nullptr);
220 return jsUndefined();
221 }
222 return returnValue;
223}

230JSC::JSValue ScriptController::evaluateModule(const URL& sourceURL, JSModuleRecord& moduleRecord, DOMWrapperWorld& world, JSC::JSValue awaitedValue, JSC::JSValue resumeMode)
231{
232 JSLockHolder lock(world.vm());
233
234 const auto& jsSourceCode = moduleRecord.sourceCode();
235
236 auto& proxy = jsWindowProxy(world);
237 auto& lexicalGlobalObject = *proxy.window();
238
239 Ref<Frame> protector(m_frame);
240 SetForScope<const URL*> sourceURLScope(m_sourceURL, &sourceURL);
241
242 InspectorInstrumentation::willEvaluateScript(m_frame, sourceURL.string(), jsSourceCode.firstLine().oneBasedInt(), jsSourceCode.startColumn().oneBasedInt());
243 auto returnValue = moduleRecord.evaluate(&lexicalGlobalObject, awaitedValue, resumeMode);
244 InspectorInstrumentation::didEvaluateScript(m_frame);
245
246 return returnValue;
247}

264void ScriptController::initScriptForWindowProxy(JSWindowProxy& windowProxy)
265{
266 auto& world = windowProxy.world();
267 JSC::VM& vm = world.vm();
268 auto scope = DECLARE_CATCH_SCOPE(vm);
269
270 jsCast<JSDOMWindow*>(windowProxy.window())->updateDocument();
271 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
272
273 if (Document* document = m_frame.document())
274 document->contentSecurityPolicy()->didCreateWindowProxy(windowProxy);
275
276 if (Page* page = m_frame.page()) {
277 windowProxy.attachDebugger(page->debugger());
278 windowProxy.window()->setProfileGroup(page->group().identifier());
279 windowProxy.window()->setConsoleClient(makeWeakPtr(page->console()));
280 }
281
282 m_frame.loader().dispatchDidClearWindowObjectInWorld(world);
283}

293void ScriptController::setupModuleScriptHandlers(LoadableModuleScript& moduleScriptRef, JSInternalPromise& promise, DOMWrapperWorld& world)
294{
295 auto& proxy = jsWindowProxy(world);
296 auto& lexicalGlobalObject = *proxy.window();
297
298 // It is not guaranteed that either fulfillHandler or rejectHandler is eventually called.
299 // For example, if the page load is canceled, the DeferredPromise used in the module loader pipeline will stop executing JS code.
300 // Thus the promise returned from this function could remain unresolved.
301
302 RefPtr<LoadableModuleScript> moduleScript(&moduleScriptRef);
303
304 auto& fulfillHandler = *JSNativeStdFunction::create(lexicalGlobalObject.vm(), proxy.window(), 1, String(), [moduleScript](JSGlobalObject* globalObject, CallFrame* callFrame) -> JSC::EncodedJSValue {
305 VM& vm = globalObject->vm();
306 auto scope = DECLARE_THROW_SCOPE(vm);
307 Identifier moduleKey = jsValueToModuleKey(globalObject, callFrame->argument(0));
308 RETURN_IF_EXCEPTION(scope, { });
309 moduleScript->notifyLoadCompleted(*moduleKey.impl());
310 return JSValue::encode(jsUndefined());
311 });
312
313 auto& rejectHandler = *JSNativeStdFunction::create(lexicalGlobalObject.vm(), proxy.window(), 1, String(), [moduleScript](JSGlobalObject* globalObject, CallFrame* callFrame) {
314 VM& vm = globalObject->vm();
315 JSValue errorValue = callFrame->argument(0);
316 if (errorValue.isObject()) {
317 auto* object = JSC::asObject(errorValue);
318 if (JSValue failureKindValue = object->getDirect(vm, static_cast<JSVMClientData&>(*vm.clientData).builtinNames().failureKindPrivateName())) {
319 // This is host propagated error in the module loader pipeline.
320 switch (static_cast<ModuleFetchFailureKind>(failureKindValue.asInt32())) {
321 case ModuleFetchFailureKind::WasErrored:
322 moduleScript->notifyLoadFailed(LoadableScript::Error {
323 LoadableScript::ErrorType::CachedScript,
324 std::nullopt
325 });
326 break;
327 case ModuleFetchFailureKind::WasCanceled:
328 moduleScript->notifyLoadWasCanceled();
329 break;
330 }
331 return JSValue::encode(jsUndefined());
332 }
333 }
334
335 auto scope = DECLARE_CATCH_SCOPE(vm);
336 moduleScript->notifyLoadFailed(LoadableScript::Error {
337 LoadableScript::ErrorType::CachedScript,
338 LoadableScript::ConsoleMessage {
339 MessageSource::JS,
340 MessageLevel::Error,
341 retrieveErrorMessage(*globalObject, vm, errorValue, scope),
342 }
343 });
344 return JSValue::encode(jsUndefined());
345 });
346
347 promise.then(&lexicalGlobalObject, &fulfillHandler, &rejectHandler);
348}

407bool ScriptController::canAccessFromCurrentOrigin(Frame* frame, Document& accessingDocument)
408{
409 auto* lexicalGlobalObject = JSExecState::currentState();
410
411 // If the current lexicalGlobalObject is null we should use the accessing document for the security check.
412 if (!lexicalGlobalObject) {
413 auto* targetDocument = frame ? frame->document() : nullptr;
414 return targetDocument && accessingDocument.securityOrigin().isSameOriginDomain(targetDocument->securityOrigin());
415 }
416
417 return BindingSecurity::shouldAllowAccessToFrame(lexicalGlobalObject, frame);
418}

582ValueOrException ScriptController::executeScriptInWorld(DOMWrapperWorld& world, RunJavaScriptParameters&& parameters)
583{
584#if ENABLE(APP_BOUND_DOMAINS)
585 if (m_frame.loader().client().shouldEnableInAppBrowserPrivacyProtections()) {
586 if (auto* document = m_frame.document())
587 document->addConsoleMessage(MessageSource::Security, MessageLevel::Warning, "Ignoring user script injection for non-app bound domain.");
588 SCRIPTCONTROLLER_RELEASE_LOG_ERROR(Loading, "executeScriptInWorld: Ignoring user script injection for non app-bound domain");
589 return makeUnexpected(ExceptionDetails { "Ignoring user script injection for non-app bound domain"_s });
590 }
591 m_frame.loader().client().notifyPageOfAppBoundBehavior();
592#endif
593
594 UserGestureIndicator gestureIndicator(parameters.forceUserGesture == ForceUserGesture::Yes ? std::optional<ProcessingUserGestureState>(ProcessingUserGesture) : std::nullopt, m_frame.document());
595
596 if (!canExecuteScripts(AboutToExecuteScript) || isPaused())
597 return makeUnexpected(ExceptionDetails { "Cannot execute JavaScript in this document"_s });
598
599 auto sourceURL = parameters.sourceURL;
600 if (!sourceURL.isValid()) {
601 // FIXME: This is gross, but when setTimeout() and setInterval() are passed JS strings, the thrown errors should use the frame document URL (according to WPT).
602 sourceURL = m_frame.document()->url();
603 }
604
605 switch (parameters.runAsAsyncFunction) {
606 case RunAsAsyncFunction::No:
607 return evaluateInWorld({ parameters.source, WTFMove(sourceURL), TextPosition(), JSC::SourceProviderSourceType::Program, CachedScriptFetcher::create(m_frame.document()->charset()) }, world);
608 case RunAsAsyncFunction::Yes:
609 return callInWorld(WTFMove(parameters), world);
610 default:
611 RELEASE_ASSERT_NOT_REACHED();
612 }
613}
614
615ValueOrException ScriptController::callInWorld(RunJavaScriptParameters&& parameters, DOMWrapperWorld& world)
616{
617 ASSERT(parameters.runAsAsyncFunction == RunAsAsyncFunction::Yes);
618 ASSERT(parameters.arguments);
619
620 auto& proxy = jsWindowProxy(world);
621 auto& globalObject = *proxy.window();
622 MarkedArgumentBuffer markedArguments;
623 StringBuilder functionStringBuilder;
624 String errorMessage;
625
626 // Build up a new script string that is an async function with arguments, and deserialize those arguments.
627 functionStringBuilder.append("(async function(");
628 for (auto argument = parameters.arguments->begin(); argument != parameters.arguments->end();) {
629 functionStringBuilder.append(argument->key);
630 auto serializedArgument = SerializedScriptValue::createFromWireBytes(WTFMove(argument->value));
631
632 auto scope = DECLARE_CATCH_SCOPE(globalObject.vm());
633 auto jsArgument = serializedArgument->deserialize(globalObject, &globalObject);
634 if (UNLIKELY(scope.exception())) {
635 errorMessage = "Unable to deserialize argument to execute asynchronous JavaScript function";
636 break;
637 }
638
639 markedArguments.append(jsArgument);
640
641 ++argument;
642 if (argument != parameters.arguments->end())
643 functionStringBuilder.append(',');
644 }
645
646 if (!errorMessage.isEmpty())
647 return makeUnexpected(ExceptionDetails { errorMessage });
648
649 functionStringBuilder.append("){", parameters.source, "})");
650
651 auto sourceCode = ScriptSourceCode { functionStringBuilder.toString(), WTFMove(parameters.sourceURL), TextPosition(), JSC::SourceProviderSourceType::Program, CachedScriptFetcher::create(m_frame.document()->charset()) };
652 const auto& jsSourceCode = sourceCode.jsSourceCode();
653
654 const URL& sourceURL = jsSourceCode.provider()->sourceOrigin().url();
655
656 Ref<Frame> protector(m_frame);
657 SetForScope<const URL*> sourceURLScope(m_sourceURL, &sourceURL);
658
659 InspectorInstrumentation::willEvaluateScript(m_frame, sourceURL.string(), sourceCode.startLine(), sourceCode.startColumn());
660
661 NakedPtr<JSC::Exception> evaluationException;
662 std::optional<ExceptionDetails> optionalDetails;
663 JSValue returnValue;
664 do {
665 JSValue functionObject = JSExecState::profiledEvaluate(&globalObject, JSC::ProfilingReason::Other, jsSourceCode, &proxy, evaluationException);
666
667 if (evaluationException)
668 break;
669
670 if (!functionObject || !functionObject.isCallable(world.vm())) {
671 optionalDetails = { { "Unable to create JavaScript async function to call"_s } };
672 break;
673 }
674
675 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=205562
676 // Getting CallData shouldn't be required to call into JS.
677 auto callData = getCallData(world.vm(), functionObject);
678 if (callData.type == CallData::Type::None) {
679 optionalDetails = { { "Unable to prepare JavaScript async function to be called"_s } };
680 break;
681 }
682
683 returnValue = JSExecState::profiledCall(&globalObject, JSC::ProfilingReason::Other, functionObject, callData, &proxy, markedArguments, evaluationException);
684 } while (false);
685
686 InspectorInstrumentation::didEvaluateScript(m_frame);
687
688 if (evaluationException && !optionalDetails) {
689 ExceptionDetails details;
690 reportException(&globalObject, evaluationException, sourceCode.cachedScript(), &details);
691 optionalDetails = WTFMove(details);
692 }
693
694 if (optionalDetails)
695 return makeUnexpected(*optionalDetails);
696 return returnValue;
697}

719
720void ScriptController::executeAsynchronousUserAgentScriptInWorld(DOMWrapperWorld& world, RunJavaScriptParameters&& parameters, ResolveFunction&& resolveCompletionHandler)
721{
722 auto result = executeUserAgentScriptInWorldInternal(world, WTFMove(parameters));
723
724 if (parameters.runAsAsyncFunction == RunAsAsyncFunction::No || !result || !result.value().isObject()) {
725 resolveCompletionHandler(result);
726 return;
727 }
728
729 // When running JavaScript as an async function, any "thenable" object gets promise-like behavior of deferred completion.
730 auto thenIdentifier = world.vm().propertyNames->then;
731 auto& proxy = jsWindowProxy(world);
732 auto& globalObject = *proxy.window();
733
734 auto thenFunction = result.value().get(&globalObject, thenIdentifier);
735 if (!thenFunction.isObject()) {
736 resolveCompletionHandler(result);
737 return;
738 }
739
740 auto callData = getCallData(world.vm(), thenFunction);
741 if (callData.type == CallData::Type::None) {
742 resolveCompletionHandler(result);
743 return;
744 }
745
746 auto sharedResolveFunction = createSharedTask<void(ValueOrException)>([resolveCompletionHandler = WTFMove(resolveCompletionHandler)](ValueOrException result) mutable {
747 if (resolveCompletionHandler)
748 resolveCompletionHandler(result);
749 resolveCompletionHandler = nullptr;
750 });
751
752 auto* fulfillHandler = JSC::JSNativeStdFunction::create(world.vm(), &globalObject, 1, String { }, [sharedResolveFunction] (JSGlobalObject*, CallFrame* callFrame) mutable {
753 sharedResolveFunction->run(callFrame->argument(0));
754 return JSValue::encode(jsUndefined());
755 });
756
757 auto* rejectHandler = JSC::JSNativeStdFunction::create(world.vm(), &globalObject, 1, String { }, [sharedResolveFunction] (JSGlobalObject* globalObject, CallFrame* callFrame) mutable {
758 sharedResolveFunction->run(makeUnexpected(ExceptionDetails { callFrame->argument(0).toWTFString(globalObject) }));
759 return JSValue::encode(jsUndefined());
760 });
761
762 auto finalizeCount = makeUniqueWithoutFastMallocCheck<unsigned>(0);
763 auto finalizeGuard = createSharedTask<void()>([sharedResolveFunction = WTFMove(sharedResolveFunction), finalizeCount = WTFMove(finalizeCount)]() {
764 if (++(*finalizeCount) == 2)
765 sharedResolveFunction->run(makeUnexpected(ExceptionDetails { "Completion handler for function call is no longer reachable"_s }));
766 });
767
768 world.vm().heap.addFinalizer(fulfillHandler, [finalizeGuard](JSCell*) {
769 finalizeGuard->run();
770 });
771 world.vm().heap.addFinalizer(rejectHandler, [finalizeGuard](JSCell*) {
772 finalizeGuard->run();
773 });
774
775 JSC::MarkedArgumentBuffer arguments;
776 arguments.append(fulfillHandler);
777 arguments.append(rejectHandler);
778
779 call(&globalObject, thenFunction, callData, result.value(), arguments);
780}

793bool ScriptController::canExecuteScripts(ReasonForCallingCanExecuteScripts reason)
794{
795 if (reason == AboutToExecuteScript)
796 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::InMainThread::isScriptAllowed() || !isInWebProcess());
797
798 if (m_frame.document() && m_frame.document()->isSandboxed(SandboxScripts)) {
799 // FIXME: This message should be moved off the console once a solution to https://bugs.webkit.org/show_bug.cgi?id=103274 exists.
800 if (reason == AboutToExecuteScript || reason == AboutToCreateEventListener)
801 m_frame.document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, "Blocked script execution in '" + m_frame.document()->url().stringCenterEllipsizedToLength() + "' because the document's frame is sandboxed and the 'allow-scripts' permission is not set.");
802 return false;
803 }
804
805 if (!m_frame.page())
806 return false;
807
808 return m_frame.loader().client().allowScript(m_frame.settings().isScriptEnabled());
809}

811void ScriptController::executeJavaScriptURL(const URL& url, RefPtr<SecurityOrigin> requesterSecurityOrigin, ShouldReplaceDocumentIfJavaScriptURL shouldReplaceDocumentIfJavaScriptURL)
812{
813 ASSERT(url.protocolIsJavaScript());
814
815 if (requesterSecurityOrigin && !requesterSecurityOrigin->isSameOriginDomain(m_frame.document()->securityOrigin()))
816 return;
817
818 if (!m_frame.page() || !m_frame.document()->contentSecurityPolicy()->allowJavaScriptURLs(m_frame.document()->url().string(), eventHandlerPosition().m_line))
819 return;
820
821 // We need to hold onto the Frame here because executing script can
822 // destroy the frame.
823 Ref<Frame> protector(m_frame);
824 RefPtr<Document> ownerDocument(m_frame.document());
825
826 const int javascriptSchemeLength = sizeof("javascript:") - 1;
827
828 JSDOMGlobalObject* globalObject = jsWindowProxy(mainThreadNormalWorld()).window();
829 VM& vm = globalObject->vm();
830 auto throwScope = DECLARE_THROW_SCOPE(vm);
831
832 String decodedURL = decodeURLEscapeSequences(url.string());
833 auto result = executeScriptIgnoringException(decodedURL.substring(javascriptSchemeLength));
834 RELEASE_ASSERT(&vm == &jsWindowProxy(mainThreadNormalWorld()).window()->vm());
835
836 // If executing script caused this frame to be removed from the page, we
837 // don't want to try to replace its document!
838 if (!m_frame.page())
839 return;
840
841 if (!result)
842 return;
843
844 String scriptResult;
845 bool isString = result.getString(globalObject, scriptResult);
846 RETURN_IF_EXCEPTION(throwScope, void());
847
848 if (!isString)
849 return;
850
851 // FIXME: We should always replace the document, but doing so
852 // synchronously can cause crashes:
853 // http://bugs.webkit.org/show_bug.cgi?id=16782
854 if (shouldReplaceDocumentIfJavaScriptURL == ReplaceDocumentIfJavaScriptURL) {
855 // We're still in a frame, so there should be a DocumentLoader.
856 ASSERT(m_frame.document()->loader());
857
858 // Signal to FrameLoader to disable navigations within this frame while replacing it with the result of executing javascript
859 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=200523
860 // The only reason we do a nestable save/restore of this flag here is because we sometimes nest javascript: url loads as
861 // some will load synchronously. We'd like to remove those synchronous loads and then change this.
862 SetForScope<bool> willBeReplaced(m_willReplaceWithResultOfExecutingJavascriptURL, true);
863
864 // DocumentWriter::replaceDocumentWithResultOfExecutingJavascriptURL can cause the DocumentLoader to get deref'ed and possible destroyed,
865 // so protect it with a RefPtr.
866 if (RefPtr<DocumentLoader> loader = m_frame.document()->loader())
867 loader->writer().replaceDocumentWithResultOfExecutingJavascriptURL(scriptResult, ownerDocument.get());
868 }
869}
