81static Expected<URL, String> resolveModuleSpecifier(ScriptExecutionContext& context, ScriptModuleLoader::OwnerType ownerType, const String& specifier, const URL& baseURL)
82{
83 // https://html.spec.whatwg.org/multipage/webappapis.html#resolve-a-module-specifier
84
85 URL absoluteURL(URL(), specifier);
86 if (absoluteURL.isValid())
87 return absoluteURL;
88
89 if (!specifier.startsWith('/') && !specifier.startsWith("./") && !specifier.startsWith("../"))
90 return makeUnexpected(makeString("Module specifier, '"_s, specifier, "' does not start with \"/\", \"./\", or \"../\". Referenced from "_s, baseURL.string()));
91
92 URL result;
93 if (ownerType == ScriptModuleLoader::OwnerType::Document)
94 result = downcast<Document>(context).completeURL(specifier, baseURL);
95 else
96 result = URL(baseURL, specifier);
97
98 if (!result.isValid())
99 return makeUnexpected(makeString("Module name, '"_s, result.string(), "' does not resolve to a valid URL."_s));
100 return result;
101}

103JSC::Identifier ScriptModuleLoader::resolve(JSC::JSGlobalObject* jsGlobalObject, JSC::JSModuleLoader*, JSC::JSValue moduleNameValue, JSC::JSValue importerModuleKey, JSC::JSValue)
104{
105 JSC::VM& vm = jsGlobalObject->vm();
106 auto scope = DECLARE_THROW_SCOPE(vm);
107
108 // We use a Symbol as a special purpose; It means this module is an inline module.
109 // So there is no correct URL to retrieve the module source code. If the module name
110 // value is a Symbol, it is used directly as a module key.
111 if (moduleNameValue.isSymbol())
112 return JSC::Identifier::fromUid(asSymbol(moduleNameValue)->privateName());
113
114 if (!moduleNameValue.isString()) {
115 JSC::throwTypeError(jsGlobalObject, scope, "Importer module key is not a Symbol or a String."_s);
116 return { };
117 }
118
119 String specifier = asString(moduleNameValue)->value(jsGlobalObject);
120 RETURN_IF_EXCEPTION(scope, { });
121
122 URL baseURL = responseURLFromRequestURL(*jsGlobalObject, importerModuleKey);
123 RETURN_IF_EXCEPTION(scope, { });
124
125 auto result = resolveModuleSpecifier(m_context, m_ownerType, specifier, baseURL);
126 if (!result) {
127 JSC::throwTypeError(jsGlobalObject, scope, result.error());
128 return { };
129 }
130
131 return JSC::Identifier::fromString(vm, result->string());
132}

134static void rejectToPropagateNetworkError(DeferredPromise& deferred, ModuleFetchFailureKind failureKind, ASCIILiteral message)
135{
136 deferred.rejectWithCallback([&] (JSDOMGlobalObject& jsGlobalObject) {
137 // We annotate exception with special private symbol. It allows us to distinguish these errors from the user thrown ones.
138 JSC::VM& vm = jsGlobalObject.vm();
139 // FIXME: Propagate more descriptive error.
140 // https://bugs.webkit.org/show_bug.cgi?id=167553
141 auto* error = JSC::createTypeError(&jsGlobalObject, message);
142 ASSERT(error);
143 error->putDirect(vm, static_cast<JSVMClientData&>(*vm.clientData).builtinNames().failureKindPrivateName(), JSC::jsNumber(static_cast<int32_t>(failureKind)));
144 return error;
145 });
146}

148JSC::JSInternalPromise* ScriptModuleLoader::fetch(JSC::JSGlobalObject* jsGlobalObject, JSC::JSModuleLoader*, JSC::JSValue moduleKeyValue, JSC::JSValue parameters, JSC::JSValue scriptFetcher)
149{
150 JSC::VM& vm = jsGlobalObject->vm();
151 ASSERT(JSC::jsDynamicCast<JSC::JSScriptFetcher*>(vm, scriptFetcher));
152
153 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(jsGlobalObject);
154 auto* jsPromise = JSC::JSInternalPromise::create(vm, globalObject.internalPromiseStructure());
155 RELEASE_ASSERT(jsPromise);
156 auto deferred = DeferredPromise::create(globalObject, *jsPromise);
157 if (moduleKeyValue.isSymbol()) {
158 deferred->reject(TypeError, "Symbol module key should be already fulfilled with the inlined resource."_s);
159 return jsPromise;
160 }
161
162 if (!moduleKeyValue.isString()) {
163 deferred->reject(TypeError, "Module key is not Symbol or String."_s);
164 return jsPromise;
165 }
166
167 // https://html.spec.whatwg.org/multipage/webappapis.html#fetch-a-single-module-script
168
169 URL completedURL(URL(), asString(moduleKeyValue)->value(jsGlobalObject));
170 if (!completedURL.isValid()) {
171 deferred->reject(TypeError, "Module key is a valid URL."_s);
172 return jsPromise;
173 }
174
175 RefPtr<ModuleFetchParameters> topLevelFetchParameters;
176 if (auto* scriptFetchParameters = JSC::jsDynamicCast<JSC::JSScriptFetchParameters*>(vm, parameters))
177 topLevelFetchParameters = static_cast<ModuleFetchParameters*>(&scriptFetchParameters->parameters());
178
179 if (m_ownerType == OwnerType::Document) {
180 auto loader = CachedModuleScriptLoader::create(*this, deferred.get(), *static_cast<CachedScriptFetcher*>(JSC::jsCast<JSC::JSScriptFetcher*>(scriptFetcher)->fetcher()), WTFMove(topLevelFetchParameters));
181 m_loaders.add(loader.copyRef());
182 if (!loader->load(downcast<Document>(m_context), WTFMove(completedURL))) {
183 loader->clearClient();
184 m_loaders.remove(WTFMove(loader));
185 rejectToPropagateNetworkError(deferred.get(), ModuleFetchFailureKind::WasErrored, "Importing a module script failed."_s);
186 return jsPromise;
187 }
188 } else {
189 auto loader = WorkerModuleScriptLoader::create(*this, deferred.get(), *static_cast<WorkerScriptFetcher*>(JSC::jsCast<JSC::JSScriptFetcher*>(scriptFetcher)->fetcher()), WTFMove(topLevelFetchParameters));
190 m_loaders.add(loader.copyRef());
191 loader->load(m_context, WTFMove(completedURL));
192 }
193
194 return jsPromise;
195}

197URL ScriptModuleLoader::moduleURL(JSC::JSGlobalObject& jsGlobalObject, JSC::JSValue moduleKeyValue)
198{
199 if (moduleKeyValue.isSymbol())
200 return m_context.url();
201
202 ASSERT(moduleKeyValue.isString());
203 return URL(URL(), asString(moduleKeyValue)->value(&jsGlobalObject));
204}

206URL ScriptModuleLoader::responseURLFromRequestURL(JSC::JSGlobalObject& jsGlobalObject, JSC::JSValue moduleKeyValue)
207{
208 JSC::VM& vm = jsGlobalObject.vm();
209 auto scope = DECLARE_THROW_SCOPE(vm);
210
211 if (isRootModule(moduleKeyValue)) {
212 if (m_ownerType == OwnerType::Document)
213 return downcast<Document>(m_context).baseURL();
214 return m_context.url();
215 }
216
217 ASSERT(!isRootModule(moduleKeyValue));
218 ASSERT(moduleKeyValue.isString());
219 String requestURL = asString(moduleKeyValue)->value(&jsGlobalObject);
220 RETURN_IF_EXCEPTION(scope, { });
221 ASSERT_WITH_MESSAGE(URL(URL(), requestURL).isValid(), "Invalid module referrer never starts importing dependent modules.");
222
223 auto iterator = m_requestURLToResponseURLMap.find(requestURL);
224 ASSERT_WITH_MESSAGE(iterator != m_requestURLToResponseURLMap.end(), "Module referrer must register itself to the map before starting importing dependent modules.");
225 URL result = iterator->value;
226 ASSERT(result.isValid());
227 return result;
228}

230JSC::JSValue ScriptModuleLoader::evaluate(JSC::JSGlobalObject* jsGlobalObject, JSC::JSModuleLoader*, JSC::JSValue moduleKeyValue, JSC::JSValue moduleRecordValue, JSC::JSValue, JSC::JSValue awaitedValue, JSC::JSValue resumeMode)
231{
232 JSC::VM& vm = jsGlobalObject->vm();
233 auto scope = DECLARE_THROW_SCOPE(vm);
234
235 // FIXME: Currently, we only support JSModuleRecord.
236 // Once the reflective part of the module loader is supported, we will handle arbitrary values.
237 // https://whatwg.github.io/loader/#registry-prototype-provide
238 auto* moduleRecord = JSC::jsDynamicCast<JSC::JSModuleRecord*>(vm, moduleRecordValue);
239 if (!moduleRecord)
240 return JSC::jsUndefined();
241
242 URL sourceURL = moduleURL(*jsGlobalObject, moduleKeyValue);
243 if (!sourceURL.isValid())
244 return JSC::throwTypeError(jsGlobalObject, scope, "Module key is an invalid URL."_s);
245
246 if (m_ownerType == OwnerType::Document) {
247 if (auto* frame = downcast<Document>(m_context).frame())
248 RELEASE_AND_RETURN(scope, frame->script().evaluateModule(sourceURL, *moduleRecord, awaitedValue, resumeMode));
249 } else {
250 ASSERT(is<WorkerOrWorkletGlobalScope>(m_context));
251 if (auto* script = downcast<WorkerOrWorkletGlobalScope>(m_context).script())
252 RELEASE_AND_RETURN(scope, script->evaluateModule(*moduleRecord, awaitedValue, resumeMode));
253 }
254 return JSC::jsUndefined();
255}

257static JSC::JSInternalPromise* rejectPromise(JSDOMGlobalObject& globalObject, ExceptionCode ec, String message)
258{
259 auto* jsPromise = JSC::JSInternalPromise::create(globalObject.vm(), globalObject.internalPromiseStructure());
260 RELEASE_ASSERT(jsPromise);
261 auto deferred = DeferredPromise::create(globalObject, *jsPromise);
262 deferred->reject(ec, WTFMove(message));
263 return jsPromise;
264}

266static bool isWorkletOrServiceWorker(ScriptExecutionContext& context)
267{
268 if (is<WorkletGlobalScope>(context))
269 return true;
270#if ENABLE(SERVICE_WORKER)
271 if (is<ServiceWorkerGlobalScope>(context))
272 return true;
273#endif
274 return false;

277JSC::JSInternalPromise* ScriptModuleLoader::importModule(JSC::JSGlobalObject* jsGlobalObject, JSC::JSModuleLoader*, JSC::JSString* moduleName, JSC::JSValue parameters, const JSC::SourceOrigin& sourceOrigin)
278{
279 JSC::VM& vm = jsGlobalObject->vm();
280 auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(jsGlobalObject);
281
282 // https://html.spec.whatwg.org/multipage/webappapis.html#hostimportmoduledynamically(referencingscriptormodule,-specifier,-promisecapability)
283 // If settings object's global object implements WorkletGlobalScope or ServiceWorkerGlobalScope, then:
284 if (isWorkletOrServiceWorker(m_context))
285 return rejectPromise(globalObject, TypeError, "Dynamic-import is not available in Worklets or ServiceWorkers"_s);
286
287 // If SourceOrigin and/or CachedScriptFetcher is null, we import the module with the default fetcher.
288 // SourceOrigin can be null if the source code is not coupled with the script file.
289 // The examples,
290 // 1. The code evaluated by the inspector.
291 // 2. The other unusual code execution like the evaluation through the NPAPI.
292 // 3. The code from injected bundle's script.
293 // 4. The code from extension script.
294 URL baseURL;
295 RefPtr<JSC::ScriptFetcher> scriptFetcher;
296 if (sourceOrigin.isNull()) {
297 if (m_ownerType == OwnerType::Document) {
298 baseURL = downcast<Document>(m_context).baseURL();
299 scriptFetcher = CachedScriptFetcher::create(downcast<Document>(m_context).charset());
300 } else {
301 // https://html.spec.whatwg.org/multipage/webappapis.html#default-classic-script-fetch-options
302 baseURL = m_context.url();
303 scriptFetcher = WorkerScriptFetcher::create(FetchOptions::Credentials::SameOrigin, FetchOptions::Destination::Script, ReferrerPolicy::EmptyString);
304 }
305 } else {
306 baseURL = URL(URL(), sourceOrigin.string());
307 if (!baseURL.isValid())
308 return rejectPromise(globalObject, TypeError, "Importer module key is not a Symbol or a String."_s);
309
310 if (sourceOrigin.fetcher()) {
311 scriptFetcher = sourceOrigin.fetcher();
312 // https://html.spec.whatwg.org/multipage/webappapis.html#fetch-an-import()-module-script-graph
313 // Destination should be "script" for dynamic-import.
314 if (m_ownerType == OwnerType::WorkerOrWorklet) {
315 auto& fetcher = static_cast<WorkerScriptFetcher&>(*scriptFetcher);
316 scriptFetcher = WorkerScriptFetcher::create(fetcher.credentials(), FetchOptions::Destination::Script, fetcher.referrerPolicy());
317 }
318 }
319
320 if (!scriptFetcher) {
321 if (m_ownerType == OwnerType::Document)
322 scriptFetcher = CachedScriptFetcher::create(downcast<Document>(m_context).charset());
323 else
324 scriptFetcher = WorkerScriptFetcher::create(FetchOptions::Credentials::SameOrigin, FetchOptions::Destination::Script, ReferrerPolicy::EmptyString);
325 }
326 }
327 ASSERT(baseURL.isValid());
328 ASSERT(scriptFetcher);
329
330 auto specifier = moduleName->value(jsGlobalObject);
331 auto result = resolveModuleSpecifier(m_context, m_ownerType, specifier, baseURL);
332 if (!result)
333 return rejectPromise(globalObject, TypeError, result.error());
334
335 return JSC::importModule(jsGlobalObject, JSC::Identifier::fromString(vm, result->string()), parameters, JSC::JSScriptFetcher::create(vm, WTFMove(scriptFetcher) ));
336}

338JSC::JSObject* ScriptModuleLoader::createImportMetaProperties(JSC::JSGlobalObject* jsGlobalObject, JSC::JSModuleLoader*, JSC::JSValue moduleKeyValue, JSC::JSModuleRecord*, JSC::JSValue)
339{
340 auto& vm = jsGlobalObject->vm();
341 auto scope = DECLARE_THROW_SCOPE(vm);
342
343 auto* metaProperties = JSC::constructEmptyObject(vm, jsGlobalObject->nullPrototypeObjectStructure());
344 RETURN_IF_EXCEPTION(scope, nullptr);
345
346 URL responseURL = responseURLFromRequestURL(*jsGlobalObject, moduleKeyValue);
347 RETURN_IF_EXCEPTION(scope, nullptr);
348
349 metaProperties->putDirect(vm, JSC::Identifier::fromString(vm, "url"), JSC::jsString(vm, responseURL.string()));
350 RETURN_IF_EXCEPTION(scope, nullptr);
351
352 return metaProperties;
353}

355void ScriptModuleLoader::notifyFinished(ModuleScriptLoader& moduleScriptLoader, URL&& sourceURL, Ref<DeferredPromise> promise)
356{
357 // https://html.spec.whatwg.org/multipage/webappapis.html#fetch-a-single-module-script
358
359 if (!m_loaders.remove(&moduleScriptLoader))
360 return;
361 moduleScriptLoader.clearClient();
362
363 auto canonicalizeAndRegisterResponseURL = [&] (URL responseURL, bool hasRedirections, ResourceResponse::Source source) {
364 // If we do not have redirection, we must reserve the source URL's fragment explicitly here since ResourceResponse::url() is the one when we first cache it to MemoryCache.
365 // FIXME: We should track fragments through redirections.
366 // https://bugs.webkit.org/show_bug.cgi?id=158420
367 // https://bugs.webkit.org/show_bug.cgi?id=210490
368 if (!hasRedirections && source != ResourceResponse::Source::ServiceWorker) {
369 if (sourceURL.hasFragmentIdentifier())
370 responseURL.setFragmentIdentifier(sourceURL.fragmentIdentifier());
371 }
372 return responseURL;
373 };
374
375 if (m_ownerType == OwnerType::Document) {
376 auto& loader = static_cast<CachedModuleScriptLoader&>(moduleScriptLoader);
377 auto& cachedScript = *loader.cachedScript();
378
379 if (cachedScript.resourceError().isAccessControl()) {
380 rejectToPropagateNetworkError(promise.get(), ModuleFetchFailureKind::WasErrored, "Cross-origin script load denied by Cross-Origin Resource Sharing policy."_s);
381 return;
382 }
383
384 if (cachedScript.errorOccurred()) {
385 rejectToPropagateNetworkError(promise.get(), ModuleFetchFailureKind::WasErrored, "Importing a module script failed."_s);
386 return;
387 }
388
389 if (cachedScript.wasCanceled()) {
390 rejectToPropagateNetworkError(promise.get(), ModuleFetchFailureKind::WasCanceled, "Importing a module script is canceled."_s);
391 return;
392 }
393
394 if (!MIMETypeRegistry::isSupportedJavaScriptMIMEType(cachedScript.response().mimeType())) {
395 // https://html.spec.whatwg.org/multipage/webappapis.html#fetch-a-single-module-script
396 // The result of extracting a MIME type from response's header list (ignoring parameters) is not a JavaScript MIME type.
397 // For historical reasons, fetching a classic script does not include MIME type checking. In contrast, module scripts will fail to load if they are not of a correct MIME type.
398 promise->reject(TypeError, makeString("'", cachedScript.response().mimeType(), "' is not a valid JavaScript MIME type."));
399 return;
400 }
401
402 if (auto* parameters = loader.parameters()) {
403 if (!matchIntegrityMetadata(cachedScript, parameters->integrity())) {
404 promise->reject(TypeError, makeString("Cannot load script ", integrityMismatchDescription(cachedScript, parameters->integrity())));
405 return;
406 }
407 }
408
409 URL responseURL = canonicalizeAndRegisterResponseURL(cachedScript.response().url(), cachedScript.hasRedirections(), cachedScript.response().source());
410 m_requestURLToResponseURLMap.add(sourceURL.string(), WTFMove(responseURL));
411 promise->resolveWithCallback([&] (JSDOMGlobalObject& jsGlobalObject) {
412 return JSC::JSSourceCode::create(jsGlobalObject.vm(),
413 JSC::SourceCode { ScriptSourceCode { &cachedScript, JSC::SourceProviderSourceType::Module, loader.scriptFetcher() }.jsSourceCode() });
414 });
415 } else {
416 auto& loader = static_cast<WorkerModuleScriptLoader&>(moduleScriptLoader);
417
418 if (loader.failed()) {
419 ASSERT(!loader.retrievedFromServiceWorkerCache());
420 auto& workerScriptLoader = loader.scriptLoader();
421 ASSERT(workerScriptLoader.failed());
422 if (workerScriptLoader.error().isAccessControl()) {
423 rejectToPropagateNetworkError(promise.get(), ModuleFetchFailureKind::WasErrored, "Cross-origin script load denied by Cross-Origin Resource Sharing policy."_s);
424 return;
425 }
426
427 if (workerScriptLoader.error().isCancellation()) {
428 rejectToPropagateNetworkError(promise.get(), ModuleFetchFailureKind::WasCanceled, "Importing a module script is canceled."_s);
429 return;
430 }
431
432 rejectToPropagateNetworkError(promise.get(), ModuleFetchFailureKind::WasErrored, "Importing a module script failed."_s);
433 return;
434 }
435
436 if (!MIMETypeRegistry::isSupportedJavaScriptMIMEType(loader.responseMIMEType())) {
437 // https://html.spec.whatwg.org/multipage/webappapis.html#fetch-a-single-module-script
438 // The result of extracting a MIME type from response's header list (ignoring parameters) is not a JavaScript MIME type.
439 // For historical reasons, fetching a classic script does not include MIME type checking. In contrast, module scripts will fail to load if they are not of a correct MIME type.
440 promise->reject(TypeError, makeString("'", loader.responseMIMEType(), "' is not a valid JavaScript MIME type."));
441 return;
442 }
443
444 URL responseURL = loader.responseURL();
445 if (!loader.retrievedFromServiceWorkerCache()) {
446 auto& workerScriptLoader = loader.scriptLoader();
447 if (auto* parameters = loader.parameters()) {
448 // If this is top-level-module, then we extract referrer-policy and apply to the dependent modules.
449 if (parameters->isTopLevelModule())
450 static_cast<WorkerScriptFetcher&>(loader.scriptFetcher()).setReferrerPolicy(loader.referrerPolicy());
451 }
452 responseURL = canonicalizeAndRegisterResponseURL(responseURL, workerScriptLoader.isRedirected(), workerScriptLoader.responseSource());
453#if ENABLE(SERVICE_WORKER)
454 if (is<ServiceWorkerGlobalScope>(m_context))
455 downcast<ServiceWorkerGlobalScope>(m_context).setScriptResource(sourceURL, ServiceWorkerContextData::ImportedScript { loader.script(), responseURL, loader.responseMIMEType() });
456#endif
457 }
458 m_requestURLToResponseURLMap.add(sourceURL.string(), responseURL);
459 promise->resolveWithCallback([&] (JSDOMGlobalObject& jsGlobalObject) {
460 return JSC::JSSourceCode::create(jsGlobalObject.vm(),
461 JSC::SourceCode { ScriptSourceCode { loader.script(), WTFMove(responseURL), { }, JSC::SourceProviderSourceType::Module, loader.scriptFetcher() }.jsSourceCode() });
462 });
463 }
