32#pragma once
33
34#include "JSDOMWrapper.h"
35#include <JavaScriptCore/Weak.h>
36#include <JavaScriptCore/WeakInlines.h>
37
38namespace WebCore {
39
40inline JSDOMObject* ScriptWrappable::wrapper() const
41{
42 return m_wrapper.get();
43}
44
45inline void ScriptWrappable::setWrapper(JSDOMObject* wrapper, JSC::WeakHandleOwner* wrapperOwner, void* context)
46{
47 ASSERT(!m_wrapper);
48 m_wrapper = JSC::Weak<JSDOMObject>(wrapper, wrapperOwner, context);
49}
50
51inline void ScriptWrappable::clearWrapper(JSDOMObject* wrapper)
52{
53 weakClear(m_wrapper, wrapper);
54}
55
56} // namespace WebCore
