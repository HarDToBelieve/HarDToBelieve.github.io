37namespace WebCore {
38
39class DeferredPromise;
40class JSDOMGlobalObject;
41class ModuleFetchParameters;
42class ScriptExecutionContext;
43class ModuleScriptLoaderClient;
44class WorkerScriptLoader;
45
46class WorkerModuleScriptLoader final : public ModuleScriptLoader, private WorkerScriptLoaderClient {
47public:
48 static Ref<WorkerModuleScriptLoader> create(ModuleScriptLoaderClient&, DeferredPromise&, WorkerScriptFetcher&, RefPtr<ModuleFetchParameters>&&);
49
50 virtual ~WorkerModuleScriptLoader();
51
52 bool load(ScriptExecutionContext&, URL&& sourceURL);
53
54 WorkerScriptLoader& scriptLoader() { return m_scriptLoader.get(); }
55
56 static String taskMode();
57 ReferrerPolicy referrerPolicy();
58 bool failed() const { return m_failed; }
59 bool retrievedFromServiceWorkerCache() const { return m_retrievedFromServiceWorkerCache; }
60
61 const ScriptBuffer& script() { return m_script; }
62 const URL& responseURL() const { return m_responseURL; }
63 const String& responseMIMEType() const { return m_responseMIMEType; }
64
65private:
66 WorkerModuleScriptLoader(ModuleScriptLoaderClient&, DeferredPromise&, WorkerScriptFetcher&, RefPtr<ModuleFetchParameters>&&);
67
68 void didReceiveResponse(unsigned long, const ResourceResponse&) final { }
69 void notifyFinished() final;
70
71 void notifyClientFinished();
72
73 Ref<WorkerScriptLoader> m_scriptLoader;
74 URL m_sourceURL;
75 ScriptBuffer m_script;
76 URL m_responseURL;
77 String m_responseMIMEType;
78 bool m_failed { false };
79 bool m_retrievedFromServiceWorkerCache { false };
80};
