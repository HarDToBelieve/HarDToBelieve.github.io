56class ObjectMap {
57public:
58 NPObject* get(RootObject* rootObject, JSObject* jsObject)
59 {
60 return m_map.get(rootObject).get(jsObject);
61 }
62
63 void add(RootObject* rootObject, JSObject* jsObject, NPObject* npObject)
64 {
65 HashMap<RootObject*, JSToNPObjectMap>::iterator iter = m_map.find(rootObject);
66 if (iter == m_map.end()) {
67 rootObject->addInvalidationCallback(&m_invalidationCallback);
68 iter = m_map.add(rootObject, JSToNPObjectMap()).iterator;
69 }
70
71 ASSERT(iter->value.find(jsObject) == iter->value.end());
72 iter->value.add(jsObject, npObject);
73 }
74
75 void remove(RootObject* rootObject)
76 {
77 ASSERT(m_map.contains(rootObject));
78 m_map.remove(rootObject);
79 }
80
81 void remove(RootObject* rootObject, JSObject* jsObject)
82 {
83 HashMap<RootObject*, JSToNPObjectMap>::iterator iter = m_map.find(rootObject);
84 ASSERT(iter != m_map.end());
85 ASSERT(iter->value.find(jsObject) != iter->value.end());
86
87 iter->value.remove(jsObject);
88 }
89
90private:
91 struct RootObjectInvalidationCallback : public RootObject::InvalidationCallback {
92 void operator()(RootObject*) override;
93 };
94 RootObjectInvalidationCallback m_invalidationCallback;
95
96 // JSObjects are protected by RootObject.
97 typedef HashMap<JSObject*, NPObject*> JSToNPObjectMap;
98 HashMap<RootObject*, JSToNPObjectMap> m_map;
99};

113static void getListFromVariantArgs(JSGlobalObject* lexicalGlobalObject, const NPVariant* args, unsigned argCount, RootObject* rootObject, MarkedArgumentBuffer& aList)
114{
115 for (unsigned i = 0; i < argCount; ++i)
116 aList.append(convertNPVariantToValue(lexicalGlobalObject, &args[i], rootObject));
117}

124static void jsDeallocate(NPObject* npObj)
125{
126 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(npObj);
127
128 if (obj->rootObject && obj->rootObject->isValid()) {
129 objectMap().remove(obj->rootObject, obj->imp);
130 obj->rootObject->gcUnprotect(obj->imp);
131 }
132
133 if (obj->rootObject)
134 obj->rootObject->deref();
135
136 free(obj);
137}

146NPObject* _NPN_CreateScriptObject(NPP npp, JSObject* imp, RefPtr<RootObject>&& rootObject)
147{
148 if (NPObject* object = objectMap().get(rootObject.get(), imp))
149 return _NPN_RetainObject(object);
150
151 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(_NPN_CreateObject(npp, NPScriptObjectClass));
152
153 obj->rootObject = rootObject.leakRef();
154
155 if (obj->rootObject) {
156 obj->rootObject->gcProtect(imp);
157 objectMap().add(obj->rootObject, imp, reinterpret_cast<NPObject*>(obj));
158 }
159
160 obj->imp = imp;
161
162 return reinterpret_cast<NPObject*>(obj);
163}

170bool _NPN_InvokeDefault(NPP, NPObject* o, const NPVariant* args, uint32_t argCount, NPVariant* result)
171{
172 if (o->_class == NPScriptObjectClass) {
173 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
174
175 VOID_TO_NPVARIANT(*result);
176
177 // Lookup the function object.
178 RootObject* rootObject = obj->rootObject;
179 if (!rootObject || !rootObject->isValid())
180 return false;
181
182 auto globalObject = rootObject->globalObject();
183 VM& vm = globalObject->vm();
184 JSLockHolder lock(vm);
185 auto scope = DECLARE_CATCH_SCOPE(vm);
186
187 JSGlobalObject* lexicalGlobalObject = globalObject;
188
189 // Call the function object.
190 JSValue function = obj->imp;
191 auto callData = getCallData(vm, function);
192 if (callData.type == CallData::Type::None)
193 return false;
194
195 MarkedArgumentBuffer argList;
196 getListFromVariantArgs(lexicalGlobalObject, args, argCount, rootObject, argList);
197 RELEASE_ASSERT(!argList.hasOverflowed());
198 JSValue resultV = JSC::call(lexicalGlobalObject, function, callData, function, argList);
199
200 // Convert and return the result of the function call.
201 convertValueToNPVariant(lexicalGlobalObject, resultV, result);
202 scope.clearException();
203 return true;
204 }
205
206 if (o->_class->invokeDefault)
207 return o->_class->invokeDefault(o, args, argCount, result);
208 VOID_TO_NPVARIANT(*result);
209 return true;
210}

212bool _NPN_Invoke(NPP npp, NPObject* o, NPIdentifier methodName, const NPVariant* args, uint32_t argCount, NPVariant* result)
213{
214 if (o->_class == NPScriptObjectClass) {
215 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
216
217 IdentifierRep* i = static_cast<IdentifierRep*>(methodName);
218 if (!i->isString())
219 return false;
220
221 // Special case the "eval" method.
222 if (methodName == _NPN_GetStringIdentifier("eval")) {
223 if (argCount != 1)
224 return false;
225 if (args[0].type != NPVariantType_String)
226 return false;
227 return _NPN_Evaluate(npp, o, const_cast<NPString*>(&args[0].value.stringValue), result);
228 }
229
230 // Look up the function object.
231 RootObject* rootObject = obj->rootObject;
232 if (!rootObject || !rootObject->isValid())
233 return false;
234
235 auto globalObject = rootObject->globalObject();
236 VM& vm = globalObject->vm();
237 JSLockHolder lock(vm);
238 auto scope = DECLARE_CATCH_SCOPE(vm);
239
240 JSGlobalObject* lexicalGlobalObject = globalObject;
241 JSValue function = obj->imp->get(lexicalGlobalObject, identifierFromNPIdentifier(lexicalGlobalObject, i->string()));
242 auto callData = getCallData(vm, function);
243 if (callData.type == CallData::Type::None)
244 return false;
245
246 // Call the function object.
247 MarkedArgumentBuffer argList;
248 getListFromVariantArgs(lexicalGlobalObject, args, argCount, rootObject, argList);
249 RELEASE_ASSERT(!argList.hasOverflowed());
250 JSValue resultV = JSC::call(lexicalGlobalObject, function, callData, obj->imp, argList);
251
252 // Convert and return the result of the function call.
253 convertValueToNPVariant(lexicalGlobalObject, resultV, result);
254 scope.clearException();
255 return true;
256 }
257
258 if (o->_class->invoke)
259 return o->_class->invoke(o, methodName, args, argCount, result);
260
261 VOID_TO_NPVARIANT(*result);
262 return true;
263}

293bool _NPN_GetProperty(NPP, NPObject* o, NPIdentifier propertyName, NPVariant* variant)
294{
295 if (o->_class == NPScriptObjectClass) {
296 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
297
298 RootObject* rootObject = obj->rootObject;
299 if (!rootObject || !rootObject->isValid())
300 return false;
301
302 auto globalObject = rootObject->globalObject();
303 VM& vm = globalObject->vm();
304 JSLockHolder lock(vm);
305 auto scope = DECLARE_CATCH_SCOPE(vm);
306
307 JSGlobalObject* lexicalGlobalObject = globalObject;
308 IdentifierRep* i = static_cast<IdentifierRep*>(propertyName);
309
310 JSValue result;
311 if (i->isString())
312 result = obj->imp->get(lexicalGlobalObject, identifierFromNPIdentifier(lexicalGlobalObject, i->string()));
313 else
314 result = obj->imp->get(lexicalGlobalObject, static_cast<uint32_t>(i->number()));
315
316 convertValueToNPVariant(lexicalGlobalObject, result, variant);
317 scope.clearException();
318 return true;
319 }
320
321 if (o->_class->hasProperty && o->_class->getProperty) {
322 if (o->_class->hasProperty(o, propertyName))
323 return o->_class->getProperty(o, propertyName, variant);
324 return false;
325 }
326
327 VOID_TO_NPVARIANT(*variant);
328 return false;
329}

331bool _NPN_SetProperty(NPP, NPObject* o, NPIdentifier propertyName, const NPVariant* variant)
332{
333 if (o->_class == NPScriptObjectClass) {
334 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
335
336 RootObject* rootObject = obj->rootObject;
337 if (!rootObject || !rootObject->isValid())
338 return false;
339
340 auto globalObject = rootObject->globalObject();
341 VM& vm = globalObject->vm();
342 JSLockHolder lock(vm);
343 auto scope = DECLARE_CATCH_SCOPE(vm);
344
345 JSGlobalObject* lexicalGlobalObject = globalObject;
346 IdentifierRep* i = static_cast<IdentifierRep*>(propertyName);
347
348 if (i->isString()) {
349 PutPropertySlot slot(obj->imp);
350 obj->imp->methodTable(vm)->put(obj->imp, lexicalGlobalObject, identifierFromNPIdentifier(lexicalGlobalObject, i->string()), convertNPVariantToValue(lexicalGlobalObject, variant, rootObject), slot);
351 } else
352 obj->imp->methodTable(vm)->putByIndex(obj->imp, lexicalGlobalObject, i->number(), convertNPVariantToValue(lexicalGlobalObject, variant, rootObject), false);
353 scope.clearException();
354 return true;
355 }
356
357 if (o->_class->setProperty)
358 return o->_class->setProperty(o, propertyName, variant);
359
360 return false;
361}

363bool _NPN_RemoveProperty(NPP, NPObject* o, NPIdentifier propertyName)
364{
365 if (o->_class == NPScriptObjectClass) {
366 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
367
368 RootObject* rootObject = obj->rootObject;
369 if (!rootObject || !rootObject->isValid())
370 return false;
371
372 auto globalObject = rootObject->globalObject();
373 VM& vm = globalObject->vm();
374 JSLockHolder lock(vm);
375 auto scope = DECLARE_CATCH_SCOPE(vm);
376
377 JSGlobalObject* lexicalGlobalObject = globalObject;
378
379 IdentifierRep* i = static_cast<IdentifierRep*>(propertyName);
380 if (i->isString()) {
381 if (!obj->imp->hasProperty(lexicalGlobalObject, identifierFromNPIdentifier(lexicalGlobalObject, i->string()))) {
382 scope.clearException();
383 return false;
384 }
385 } else {
386 if (!obj->imp->hasProperty(lexicalGlobalObject, static_cast<uint32_t>(i->number()))) {
387 scope.clearException();
388 return false;
389 }
390 }
391
392 if (i->isString())
393 JSCell::deleteProperty(obj->imp, lexicalGlobalObject, identifierFromNPIdentifier(lexicalGlobalObject, i->string()));
394 else
395 obj->imp->methodTable(vm)->deletePropertyByIndex(obj->imp, lexicalGlobalObject, i->number());
396
397 scope.clearException();
398 return true;
399 }
400 return false;
401}

473bool _NPN_Enumerate(NPP, NPObject* o, NPIdentifier** identifier, uint32_t* count)
474{
475 if (o->_class == NPScriptObjectClass) {
476 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
477
478 RootObject* rootObject = obj->rootObject;
479 if (!rootObject || !rootObject->isValid())
480 return false;
481
482 auto globalObject = rootObject->globalObject();
483 VM& vm = globalObject->vm();
484 JSLockHolder lock(vm);
485 auto scope = DECLARE_CATCH_SCOPE(vm);
486
487 JSGlobalObject* lexicalGlobalObject = globalObject;
488 PropertyNameArray propertyNames(vm, PropertyNameMode::Strings, PrivateSymbolMode::Exclude);
489
490 obj->imp->getPropertyNames(lexicalGlobalObject, propertyNames, DontEnumPropertiesMode::Exclude);
491 unsigned size = static_cast<unsigned>(propertyNames.size());
492 // FIXME: This should really call NPN_MemAlloc but that's in WebKit
493 NPIdentifier* identifiers = static_cast<NPIdentifier*>(malloc(sizeof(NPIdentifier) * size));
494
495 for (unsigned i = 0; i < size; ++i)
496 identifiers[i] = _NPN_GetStringIdentifier(propertyNames[i].string().utf8().data());
497
498 *identifier = identifiers;
499 *count = size;
500
501 scope.clearException();
502 return true;
503 }
504
505 if (NP_CLASS_STRUCT_VERSION_HAS_ENUM(o->_class) && o->_class->enumerate)
506 return o->_class->enumerate(o, identifier, count);
507
508 return false;
509}

511bool _NPN_Construct(NPP, NPObject* o, const NPVariant* args, uint32_t argCount, NPVariant* result)
512{
513 if (o->_class == NPScriptObjectClass) {
514 JavaScriptObject* obj = reinterpret_cast<JavaScriptObject*>(o);
515
516 VOID_TO_NPVARIANT(*result);
517
518 // Lookup the constructor object.
519 RootObject* rootObject = obj->rootObject;
520 if (!rootObject || !rootObject->isValid())
521 return false;
522
523 auto globalObject = rootObject->globalObject();
524 VM& vm = globalObject->vm();
525 JSLockHolder lock(vm);
526 auto scope = DECLARE_CATCH_SCOPE(vm);
527
528 JSGlobalObject* lexicalGlobalObject = globalObject;
529
530 // Call the constructor object.
531 JSValue constructor = obj->imp;
532 auto constructData = getConstructData(vm, constructor);
533 if (constructData.type == CallData::Type::None)
534 return false;
535
536 MarkedArgumentBuffer argList;
537 getListFromVariantArgs(lexicalGlobalObject, args, argCount, rootObject, argList);
538 RELEASE_ASSERT(!argList.hasOverflowed());
539 JSValue resultV = JSC::construct(lexicalGlobalObject, constructor, constructData, argList);
540
541 // Convert and return the result.
542 convertValueToNPVariant(lexicalGlobalObject, resultV, result);
543 scope.clearException();
544 return true;
545 }
546
547 if (NP_CLASS_STRUCT_VERSION_HAS_CTOR(o->_class) && o->_class->construct)
548 return o->_class->construct(o, args, argCount, result);
549
550 return false;
551}
