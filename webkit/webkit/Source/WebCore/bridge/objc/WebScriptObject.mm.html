83NSObject *getJSWrapper(JSObject* impl)
84{
85 ASSERT(isMainThread());
86 Locker locker { wrapperCacheLock };
87
88 NSObject* wrapper = wrapperCache().get(impl);
89 return wrapper ? retainPtr(wrapper).autorelease() : nil;
90}

92void addJSWrapper(NSObject *wrapper, JSObject* impl)
93{
94 ASSERT(isMainThread());
95 Locker locker { wrapperCacheLock };
96
97 wrapperCache().set(impl, wrapper);
98}

100void removeJSWrapper(JSObject* impl)
101{
102 Locker locker { wrapperCacheLock };
103
104 wrapperCache().remove(impl);
105}

107static void removeJSWrapperIfRetainCountOne(NSObject* wrapper, JSObject* impl)
108{
109 Locker locker { wrapperCacheLock };
110
111 if ([wrapper retainCount] == 1)
112 wrapperCache().remove(impl);
113}

122static void addExceptionToConsole(JSC::JSGlobalObject* lexicalGlobalObject, JSC::Exception* exception)
123{
124 JSC::VM& vm = lexicalGlobalObject->vm();
125 JSDOMWindow* window = asJSDOMWindow(vm.deprecatedVMEntryGlobalObject(lexicalGlobalObject));
126 if (!window || !exception)
127 return;
128 reportException(lexicalGlobalObject, exception);
129}

131static void addExceptionToConsole(JSC::JSGlobalObject* lexicalGlobalObject)
132{
133 JSC::VM& vm = lexicalGlobalObject->vm();
134 auto scope = DECLARE_CATCH_SCOPE(vm);
135 JSC::Exception* exception = scope.exception();
136 scope.clearException();
137 addExceptionToConsole(lexicalGlobalObject, exception);
138}

174+ (id)scriptObjectForJSObject:(JSObjectRef)jsObject originRootObject:(RootObject*)originRootObject rootObject:(RootObject*)rootObject
175{
176 ASSERT(jsObject);
177 auto& wrapped = *toJS(jsObject);
178
179 if (WebCore::createDOMWrapperFunction) {
180 if (auto wrapper = WebCore::createDOMWrapperFunction(wrapped)) {
181 if (![wrapper _hasImp]) // new wrapper, not from cache
182 [wrapper _setImp:&wrapped originRootObject:originRootObject rootObject:rootObject];
183 return wrapper;
184 }
185 }
186
187 return WebCore::createJSWrapper(&wrapped, originRootObject, rootObject);
188}

190- (void)_setImp:(JSObject*)imp originRootObject:(RefPtr<RootObject>&&)originRootObject rootObject:(RefPtr<RootObject>&&)rootObject
191{
192 // This function should only be called once, as a (possibly lazy) initializer.
193 ASSERT(!_private->imp);
194 ASSERT(!_private->rootObject);
195 ASSERT(!_private->originRootObject);
196 ASSERT(imp);
197
198 _private->imp = imp;
199 _private->rootObject = rootObject.leakRef();
200 _private->originRootObject = originRootObject.leakRef();
201
202 WebCore::addJSWrapper(self, imp);
203
204 if (_private->rootObject)
205 _private->rootObject->gcProtect(imp);
206}

208- (void)_setOriginRootObject:(RefPtr<RootObject>&&)originRootObject andRootObject:(RefPtr<RootObject>&&)rootObject
209{
210 ASSERT(_private->imp);
211
212 if (rootObject)
213 rootObject->gcProtect(_private->imp);
214
215 if (_private->rootObject && _private->rootObject->isValid())
216 _private->rootObject->gcUnprotect(_private->imp);
217
218 if (_private->rootObject)
219 _private->rootObject->deref();
220
221 if (_private->originRootObject)
222 _private->originRootObject->deref();
223
224 _private->rootObject = rootObject.leakRef();
225 _private->originRootObject = originRootObject.leakRef();
226}

265- (BOOL)_isSafeScript
266{
267 RootObject *root = [self _rootObject];
268 if (!root)
269 return false;
270
271 if (!_private->originRootObject)
272 return true;
273
274 if (!_private->originRootObject->isValid())
275 return false;
276
277 // It's not actually correct to call shouldAllowAccessToFrame in this way because
278 // JSDOMWindowBase* isn't the right object to represent the currently executing
279 // JavaScript. Instead, we should use JSGlobalObject, like we do elsewhere.
280 JSC::JSGlobalObject* globalObject = root->globalObject();
281 JSC::VM& vm = globalObject->vm();
282 auto scope = DECLARE_CATCH_SCOPE(vm);
283
284 auto* target = JSC::jsDynamicCast<JSDOMWindowBase*>(vm, globalObject);
285 if (!target)
286 return false;
287
288 bool isSafe = BindingSecurity::shouldAllowAccessToDOMWindow(_private->originRootObject->globalObject(), target->wrapped());
289 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
290 return isSafe;
291}

344- (id)callWebScriptMethod:(NSString *)name withArguments:(NSArray *)args
345{
346 if (![self _isSafeScript])
347 return nil;
348
349 // Look up the function object.
350 auto globalObject = [self _rootObject]->globalObject();
351 auto& vm = globalObject->vm();
352 JSLockHolder lock(vm);
353 auto scope = DECLARE_CATCH_SCOPE(vm);
354 JSC::JSGlobalObject* lexicalGlobalObject = globalObject;
355 UNUSED_PARAM(scope);
356
357 JSC::JSValue function = [self _imp]->get(lexicalGlobalObject, Identifier::fromString(vm, String(name)));
358 auto callData = getCallData(vm, function);
359 if (callData.type == CallData::Type::None)
360 return nil;
361
362 MarkedArgumentBuffer argList;
363 ASSERT(!argList.hasOverflowed());
364 getListFromNSArray(lexicalGlobalObject, args, [self _rootObject], argList);
365
366 if (![self _isSafeScript])
367 return nil;
368
369 NakedPtr<JSC::Exception> exception;
370 JSC::JSValue result = JSExecState::profiledCall(lexicalGlobalObject, JSC::ProfilingReason::Other, function, callData, [self _imp], argList, exception);
371
372 if (exception) {
373 addExceptionToConsole(lexicalGlobalObject, exception);
374 result = jsUndefined();
375 }
376
377 // Convert and return the result of the function call.
378 id resultObj = [WebScriptObject _convertValueToObjcValue:result originRootObject:[self _originRootObject] rootObject:[self _rootObject]];
379
380 return resultObj;
381}

401- (void)setValue:(id)value forKey:(NSString *)key
402{
403 if (![self _isSafeScript])
404 return;
405
406 auto globalObject = [self _rootObject]->globalObject();
407 auto& vm = globalObject->vm();
408 JSLockHolder lock(vm);
409 auto scope = DECLARE_CATCH_SCOPE(vm);
410 JSC::JSGlobalObject* lexicalGlobalObject = globalObject;
411
412 JSObject* object = JSC::jsDynamicCast<JSObject*>(vm, [self _imp]);
413 PutPropertySlot slot(object);
414 object->methodTable(vm)->put(object, lexicalGlobalObject, Identifier::fromString(vm, String(key)), convertObjcValueToValue(lexicalGlobalObject, &value, ObjcObjectType, [self _rootObject]), slot);
415
416 if (UNLIKELY(scope.exception())) {
417 addExceptionToConsole(lexicalGlobalObject);
418 scope.clearException();
419 }
420}

422- (id)valueForKey:(NSString *)key
423{
424 if (![self _isSafeScript])
425 return nil;
426
427 id resultObj;
428 {
429 auto globalObject = [self _rootObject]->globalObject();
430 auto& vm = globalObject->vm();
431
432 // Need to scope this lock to ensure that we release the lock before calling
433 // [super valueForKey:key] which might throw an exception and bypass the JSLock destructor,
434 // leaving the lock permanently held
435 JSLockHolder lock(vm);
436
437 auto scope = DECLARE_CATCH_SCOPE(vm);
438 JSC::JSGlobalObject* lexicalGlobalObject = globalObject;
439
440 JSC::JSValue result = [self _imp]->get(lexicalGlobalObject, Identifier::fromString(vm, String(key)));
441
442 if (UNLIKELY(scope.exception())) {
443 addExceptionToConsole(lexicalGlobalObject);
444 result = jsUndefined();
445 scope.clearException();
446 }
447
448 resultObj = [WebScriptObject _convertValueToObjcValue:result originRootObject:[self _originRootObject] rootObject:[self _rootObject]];
449 }
450
451 if ([resultObj isKindOfClass:[WebUndefined class]])
452 resultObj = [super valueForKey:key]; // defaults to throwing an exception
453
454 return resultObj;
455}

570+ (id)_convertValueToObjcValue:(JSC::JSValue)value originRootObject:(RootObject*)originRootObject rootObject:(RootObject*)rootObject
571{
572 if (value.isObject()) {
573 JSObject* object = asObject(value);
574 JSC::VM& vm = rootObject->globalObject()->vm();
575 JSLockHolder lock(vm);
576
577 if (object->inherits<JSHTMLElement>(vm)) {
578 // Plugin elements cache the instance internally.
579 if (ObjcInstance* instance = static_cast<ObjcInstance*>(pluginInstance(jsCast<JSHTMLElement*>(object)->wrapped())))
580 return instance->getObject();
581 } else if (object->inherits<ObjCRuntimeObject>(vm)) {
582 ObjCRuntimeObject* runtimeObject = static_cast<ObjCRuntimeObject*>(object);
583 ObjcInstance* instance = runtimeObject->getInternalObjCInstance();
584 if (instance)
585 return instance->getObject();
586 return nil;
587 }
588
589 return [WebScriptObject scriptObjectForJSObject:toRef(object) originRootObject:originRootObject rootObject:rootObject];
590 }
591
592 if (value.isString())
593 return asString(value)->value(rootObject->globalObject());
594
595 if (value.isNumber())
596 return @(value.asNumber());
597
598 if (value.isBoolean())
599 return [NSNumber numberWithBool:value.asBoolean()];
600
601 if (value.isUndefined())
602 return [WebUndefined undefined];
603
604 // jsNull is not returned as NSNull because existing applications do not expect
605 // that return value. Return as nil for compatibility. <rdar://problem/4651318> <rdar://problem/4701626>
606 // Other types (e.g., UnspecifiedType) also return as nil.
607 return nil;
608}
