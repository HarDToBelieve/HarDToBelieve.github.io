99JSValue ObjcField::valueFromInstance(JSGlobalObject* lexicalGlobalObject, const Instance* instance) const
100{
101 VM& vm = lexicalGlobalObject->vm();
102 auto scope = DECLARE_THROW_SCOPE(vm);
103
104 JSValue result = jsUndefined();
105
106 id targetObject = (static_cast<const ObjcInstance*>(instance))->getObject();
107
108 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject); // Can't put this inside the @try scope because it unwinds incorrectly.
109
110 @try {
111 if (id objcValue = [targetObject valueForKey:(__bridge NSString *)_name.get()])
112 result = convertObjcValueToValue(lexicalGlobalObject, &objcValue, ObjcObjectType, instance->rootObject());
113 {
114 JSLockHolder lock(lexicalGlobalObject);
115 ObjcInstance::moveGlobalExceptionToExecState(lexicalGlobalObject);
116 }
117 } @catch(NSException* localException) {
118 JSLockHolder lock(lexicalGlobalObject);
119 throwError(lexicalGlobalObject, scope, [localException reason]);
120 }
121
122 // Work around problem in some versions of GCC where result gets marked volatile and
123 // it can't handle copying from a volatile to non-volatile.
124 return const_cast<JSValue&>(result);
125}

127static id convertValueToObjcObject(JSGlobalObject* lexicalGlobalObject, JSValue value)
128{
129 VM& vm = lexicalGlobalObject->vm();
130 RefPtr<RootObject> rootObject = findRootObject(vm.deprecatedVMEntryGlobalObject(lexicalGlobalObject));
131 if (!rootObject)
132 return nil;
133 return [webScriptObjectClass() _convertValueToObjcValue:value originRootObject:rootObject.get() rootObject:rootObject.get()];
134}

136bool ObjcField::setValueToInstance(JSGlobalObject* lexicalGlobalObject, const Instance* instance, JSValue aValue) const
137{
138 JSC::VM& vm = lexicalGlobalObject->vm();
139 auto scope = DECLARE_THROW_SCOPE(vm);
140
141 id targetObject = (static_cast<const ObjcInstance*>(instance))->getObject();
142 id value = convertValueToObjcObject(lexicalGlobalObject, aValue);
143
144 JSLock::DropAllLocks dropAllLocks(lexicalGlobalObject); // Can't put this inside the @try scope because it unwinds incorrectly.
145
146 @try {
147 [targetObject setValue:value forKey:(__bridge NSString *)_name.get()];
148 {
149 JSLockHolder lock(lexicalGlobalObject);
150 ObjcInstance::moveGlobalExceptionToExecState(lexicalGlobalObject);
151 }
152 return true;
153 } @catch(NSException* localException) {
154 JSLockHolder lock(lexicalGlobalObject);
155 throwError(lexicalGlobalObject, scope, [localException reason]);
156 return false;
157 }
158}

168bool ObjcArray::setValueAt(JSGlobalObject* lexicalGlobalObject, unsigned int index, JSValue aValue) const
169{
170 JSC::VM& vm = lexicalGlobalObject->vm();
171 auto scope = DECLARE_THROW_SCOPE(vm);
172
173 if (![_array.get() respondsToSelector:@selector(insertObject:atIndex:)]) {
174 throwTypeError(lexicalGlobalObject, scope, "Array is not mutable."_s);
175 return false;
176 }
177
178 if (index > [_array.get() count]) {
179 throwException(lexicalGlobalObject, scope, createRangeError(lexicalGlobalObject, "Index exceeds array size."));
180 return false;
181 }
182
183 // Always try to convert the value to an ObjC object, so it can be placed in the
184 // array.
185 ObjcValue oValue = convertValueToObjcValue (lexicalGlobalObject, aValue, ObjcObjectType);
186
187 @try {
188 [_array.get() insertObject:(__bridge id)oValue.objectValue atIndex:index];
189 return true;
190 } @catch(NSException* localException) {
191 throwException(lexicalGlobalObject, scope, createError(lexicalGlobalObject, "Objective-C exception."));
192 return false;
193 }
194}
195
196JSValue ObjcArray::valueAt(JSGlobalObject* lexicalGlobalObject, unsigned int index) const
197{
198 JSC::VM& vm = lexicalGlobalObject->vm();
199 auto scope = DECLARE_THROW_SCOPE(vm);
200
201 if (index > [_array.get() count])
202 return throwException(lexicalGlobalObject, scope, createRangeError(lexicalGlobalObject, "Index exceeds array size."));
203 @try {
204 id obj = [_array.get() objectAtIndex:index];
205 if (obj)
206 return convertObjcValueToValue (lexicalGlobalObject, &obj, ObjcObjectType, m_rootObject.get());
207 } @catch(NSException* localException) {
208 return throwException(lexicalGlobalObject, scope, createError(lexicalGlobalObject, "Objective-C exception."));
209 }
210 return jsUndefined();
211}

257static JSC_DECLARE_HOST_FUNCTION(callObjCFallbackObject);
258
259JSC_DEFINE_HOST_FUNCTION(callObjCFallbackObject, (JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame))
260{
261 JSC::VM& vm = lexicalGlobalObject->vm();
262 auto scope = DECLARE_THROW_SCOPE(vm);
263
264 JSValue thisValue = callFrame->thisValue();
265 if (!thisValue.inherits<ObjCRuntimeObject>(vm))
266 return throwVMTypeError(lexicalGlobalObject, scope);
267
268 JSValue result = jsUndefined();
269
270 ObjCRuntimeObject* runtimeObject = static_cast<ObjCRuntimeObject*>(asObject(thisValue));
271 ObjcInstance* objcInstance = runtimeObject->getInternalObjCInstance();
272
273 if (!objcInstance)
274 return JSValue::encode(throwRuntimeObjectInvalidAccessError(lexicalGlobalObject, scope));
275
276 objcInstance->begin();
277
278 id targetObject = objcInstance->getObject();
279
280 if ([targetObject respondsToSelector:@selector(invokeUndefinedMethodFromWebScript:withArguments:)]){
281 ObjcClass* objcClass = static_cast<ObjcClass*>(objcInstance->getClass());
282 std::unique_ptr<ObjcMethod> fallbackMethod(makeUnique<ObjcMethod>(objcClass->isa(), @selector(invokeUndefinedMethodFromWebScript:withArguments:)));
283 const String& nameIdentifier = static_cast<ObjcFallbackObjectImp*>(callFrame->jsCallee())->propertyName();
284 fallbackMethod->setJavaScriptName(nameIdentifier.createCFString().get());
285 result = objcInstance->invokeObjcMethod(lexicalGlobalObject, callFrame, fallbackMethod.get());
286 }
287
288 objcInstance->end();
289
290 return JSValue::encode(result);
291}

293CallData ObjcFallbackObjectImp::getCallData(JSCell* cell)
294{
295 CallData callData;
296
297 ObjcFallbackObjectImp* thisObject = jsCast<ObjcFallbackObjectImp*>(cell);
298 id targetObject = thisObject->_instance->getObject();
299 if ([targetObject respondsToSelector:@selector(invokeUndefinedMethodFromWebScript:withArguments:)]) {
300 callData.type = CallData::Type::Native;
301 callData.native.function = callObjCFallbackObject;
302 }
303
304 return callData;
305}

312JSC_DEFINE_HOST_FUNCTION(convertObjCFallbackObjectToPrimitive, (JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame))
313{
314 VM& vm = lexicalGlobalObject->vm();
315 auto scope = DECLARE_THROW_SCOPE(vm);
316
317 auto* thisObject = jsDynamicCast<ObjcFallbackObjectImp*>(vm, callFrame->thisValue());
318 if (!thisObject)
319 return throwVMTypeError(lexicalGlobalObject, scope, "ObjcFallbackObject[Symbol.toPrimitive] method called on incompatible |this| value."_s);
320
321 scope.release();
322 return JSValue::encode(thisObject->getInternalObjCInstance()->getValueOfUndefinedField(lexicalGlobalObject, Identifier::fromString(vm, thisObject->propertyName())));
323}

325bool ObjcFallbackObjectImp::toBoolean(JSGlobalObject*) const
326{
327 id targetObject = _instance->getObject();
328
329 if ([targetObject respondsToSelector:@selector(invokeUndefinedMethodFromWebScript:withArguments:)])
330 return true;
331
332 return false;
333}
