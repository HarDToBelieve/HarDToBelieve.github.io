70 JavaScript to ObjC
71 Number coerced to char, short, int, long, float, double, or NSNumber, as appropriate
72 String NSString
73 wrapper id
74 Object WebScriptObject
75 null NSNull
76 [], other exception

80static long long convertDoubleToLongLong(double d)
81{
82 // Ensure that non-x86_64 ports will match x86_64 semantics which allows for overflow.
83 long long potentiallyOverflowed = (unsigned long long)d;
84 if (potentiallyOverflowed < (long long)d)
85 return potentiallyOverflowed;
86 return (long long)d;
87}

89ObjcValue convertValueToObjcValue(JSGlobalObject* lexicalGlobalObject, JSValue value, ObjcValueType type)
90{
91 ObjcValue result;
92 double d = 0;
93
94 if (value.isNumber() || value.isString() || value.isBoolean()) {
95 JSLockHolder lock(lexicalGlobalObject);
96 d = value.toNumber(lexicalGlobalObject);
97 }
98
99 switch (type) {
100 case ObjcObjectType: {
101 VM& vm = lexicalGlobalObject->vm();
102 JSLockHolder lock(vm);
103
104 JSGlobalObject *originGlobalObject = vm.deprecatedVMEntryGlobalObject(lexicalGlobalObject);
105 RootObject* originRootObject = findRootObject(originGlobalObject);
106
107 JSGlobalObject* globalObject = 0;
108 if (value.isObject() && asObject(value)->isGlobalObject())
109 globalObject = jsCast<JSGlobalObject*>(asObject(value));
110
111 if (!globalObject)
112 globalObject = originGlobalObject;
113
114 RootObject* rootObject = findRootObject(globalObject);
115 result.objectValue = rootObject
116 ? (__bridge CFTypeRef)[webScriptObjectClass() _convertValueToObjcValue:value originRootObject:originRootObject rootObject:rootObject]
117 : nil;
118 }
119 break;
120
121 case ObjcCharType:
122 case ObjcUnsignedCharType:
123 result.charValue = (char)d;
124 break;
125 case ObjcShortType:
126 case ObjcUnsignedShortType:
127 result.shortValue = (short)d;
128 break;
129 case ObjcBoolType:
130 result.booleanValue = (bool)d;
131 break;
132 case ObjcIntType:
133 case ObjcUnsignedIntType:
134 result.intValue = (int)d;
135 break;
136 case ObjcLongType:
137 case ObjcUnsignedLongType:
138 result.longValue = (long)d;
139 break;
140 case ObjcLongLongType:
141 case ObjcUnsignedLongLongType:
142 result.longLongValue = convertDoubleToLongLong(d);
143 break;
144 case ObjcFloatType:
145 result.floatValue = (float)d;
146 break;
147 case ObjcDoubleType:
148 result.doubleValue = (double)d;
149 break;
150 case ObjcVoidType:
151 bzero(&result, sizeof(ObjcValue));
152 break;
153
154 case ObjcInvalidType:
155 default:
156 // FIXME: throw an exception?
157 break;
158 }
159
160 return result;
161}

163JSValue convertNSStringToString(JSGlobalObject* lexicalGlobalObject, NSString *nsstring)
164{
165 VM& vm = lexicalGlobalObject->vm();
166 JSLockHolder lock(vm);
167 JSValue aValue = jsString(vm, String(nsstring));
168 return aValue;
169}

171/*
172 ObjC to JavaScript
173 ---- ----------
174 char number
175 short number
176 int number
177 long number
178 float number
179 double number
180 NSNumber boolean or number
181 NSString string
182 NSArray array
183 NSNull null
184 WebScriptObject underlying JavaScript object
185 WebUndefined undefined
186 id object wrapper
187 other should not happen
188*/
189JSValue convertObjcValueToValue(JSGlobalObject* lexicalGlobalObject, void* buffer, ObjcValueType type, RootObject* rootObject)
190{
191 JSLockHolder lock(lexicalGlobalObject);
192
193 switch (type) {
194 case ObjcObjectType: {
195 id obj = *(const id*)buffer;
196 if ([obj isKindOfClass:[NSString class]])
197 return convertNSStringToString(lexicalGlobalObject, (NSString *)obj);
198 if ([obj isKindOfClass:webUndefinedClass()])
199 return jsUndefined();
200 if ((__bridge CFBooleanRef)obj == kCFBooleanTrue)
201 return jsBoolean(true);
202 if ((__bridge CFBooleanRef)obj == kCFBooleanFalse)
203 return jsBoolean(false);
204 if ([obj isKindOfClass:[NSNumber class]])
205 return jsNumber([obj doubleValue]);
206 if ([obj isKindOfClass:[NSArray class]])
207 return RuntimeArray::create(lexicalGlobalObject, new ObjcArray(obj, rootObject));
208 if ([obj isKindOfClass:webScriptObjectClass()]) {
209 JSObject* imp = [obj _imp];
210 return imp ? imp : jsUndefined();
211 }
212 if ([obj isKindOfClass:[NSNull class]])
213 return jsNull();
214 if (obj == 0)
215 return jsUndefined();
216 return ObjcInstance::create(obj, rootObject)->createRuntimeObject(lexicalGlobalObject);
217 }
218 case ObjcCharType:
219 return jsNumber(*(char*)buffer);
220 case ObjcUnsignedCharType:
221 return jsNumber(*(unsigned char*)buffer);
222 case ObjcShortType:
223 return jsNumber(*(short*)buffer);
224 case ObjcUnsignedShortType:
225 return jsNumber(*(unsigned short*)buffer);
226 case ObjcBoolType:
227 return jsBoolean(*(bool*)buffer);
228 case ObjcIntType:
229 return jsNumber(*(int*)buffer);
230 case ObjcUnsignedIntType:
231 return jsNumber(*(unsigned int*)buffer);
232 case ObjcLongType:
233 return jsNumber(*(long*)buffer);
234 case ObjcUnsignedLongType:
235 return jsNumber(*(unsigned long*)buffer);
236 case ObjcLongLongType:
237 return jsNumber(*(long long*)buffer);
238 case ObjcUnsignedLongLongType:
239 return jsNumber(*(unsigned long long*)buffer);
240 case ObjcFloatType:
241 return jsNumber(*(float*)buffer);
242 case ObjcDoubleType:
243 return jsNumber(*(double*)buffer);
244 default:
245 // Should never get here. Argument types are filtered.
246 fprintf(stderr, "%s: invalid type (%d)\n", __PRETTY_FUNCTION__, (int)type);
247 ASSERT_NOT_REACHED();
248 }
249
250 return jsUndefined();
251}

253ObjcValueType objcValueTypeForType(const char *type)
254{
255 int typeLength = strlen(type);
256 ObjcValueType objcValueType = ObjcInvalidType;
257
258 for (int i = 0; i < typeLength; ++i) {
259 char typeChar = type[i];
260 switch (typeChar) {
261 case _C_CONST:
262 case _C_BYCOPY:
263 case _C_BYREF:
264 case _C_ONEWAY:
265 case _C_GCINVISIBLE:
266 // skip these type modifiers
267 break;
268 case _C_ID:
269 objcValueType = ObjcObjectType;
270 break;
271 case _C_CHR:
272 objcValueType = ObjcCharType;
273 break;
274 case _C_UCHR:
275 objcValueType = ObjcUnsignedCharType;
276 break;
277 case _C_SHT:
278 objcValueType = ObjcShortType;
279 break;
280 case _C_USHT:
281 objcValueType = ObjcUnsignedShortType;
282 break;
283 case _C_BOOL:
284 objcValueType = ObjcBoolType;
285 break;
286 case _C_INT:
287 objcValueType = ObjcIntType;
288 break;
289 case _C_UINT:
290 objcValueType = ObjcUnsignedIntType;
291 break;
292 case _C_LNG:
293 objcValueType = ObjcLongType;
294 break;
295 case _C_ULNG:
296 objcValueType = ObjcUnsignedLongType;
297 break;
298 case _C_LNG_LNG:
299 objcValueType = ObjcLongLongType;
300 break;
301 case _C_ULNG_LNG:
302 objcValueType = ObjcUnsignedLongLongType;
303 break;
304 case _C_FLT:
305 objcValueType = ObjcFloatType;
306 break;
307 case _C_DBL:
308 objcValueType = ObjcDoubleType;
309 break;
310 case _C_VOID:
311 objcValueType = ObjcVoidType;
312 break;
313 default:
314 // Unhandled type. We don't handle C structs, unions, etc.
315 // FIXME: throw an exception?
316 WTFLogAlways("Unhandled ObjC type specifier: \"%c\" used in ObjC bridge.", typeChar);
317 RELEASE_ASSERT_NOT_REACHED();
318 }
319
320 if (objcValueType != ObjcInvalidType)
321 break;
322 }
323
324 return objcValueType;
325}
