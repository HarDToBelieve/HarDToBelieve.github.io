62JSC_DEFINE_CUSTOM_GETTER(methodLengthGetter, (JSGlobalObject* exec, EncodedJSValue thisValue, PropertyName))
63{
64 VM& vm = exec->vm();
65 auto scope = DECLARE_THROW_SCOPE(vm);
66
67 RuntimeMethod* thisObject = jsDynamicCast<RuntimeMethod*>(vm, JSValue::decode(thisValue));
68 if (!thisObject)
69 return throwVMTypeError(exec, scope);
70 return JSValue::encode(jsNumber(thisObject->method()->numParameters()));
71}

73bool RuntimeMethod::getOwnPropertySlot(JSObject* object, JSGlobalObject* exec, PropertyName propertyName, PropertySlot &slot)
74{
75 VM& vm = exec->vm();
76 RuntimeMethod* thisObject = jsCast<RuntimeMethod*>(object);
77 if (propertyName == vm.propertyNames->length) {
78 slot.setCacheableCustom(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, methodLengthGetter);
79 return true;
80 }
81
82 return InternalFunction::getOwnPropertySlot(thisObject, exec, propertyName, slot);
83}

90JSC_DEFINE_HOST_FUNCTION(callRuntimeMethod, (JSGlobalObject* globalObject, CallFrame* callFrame))
91{
92 VM& vm = globalObject->vm();
93 auto scope = DECLARE_THROW_SCOPE(vm);
94
95 RuntimeMethod* method = static_cast<RuntimeMethod*>(callFrame->jsCallee());
96
97 if (!method->method())
98 return JSValue::encode(jsUndefined());
99
100 RefPtr<Instance> instance;
101
102 JSValue thisValue = callFrame->thisValue();
103 if (thisValue.inherits<RuntimeObject>(vm)) {
104 RuntimeObject* runtimeObject = static_cast<RuntimeObject*>(asObject(thisValue));
105 instance = runtimeObject->getInternalInstance();
106 if (!instance)
107 return JSValue::encode(throwRuntimeObjectInvalidAccessError(globalObject, scope));
108 } else {
109 // Calling a runtime object of a plugin element?
110 if (thisValue.inherits<JSHTMLElement>(vm))
111 instance = pluginInstance(jsCast<JSHTMLElement*>(asObject(thisValue))->wrapped());
112 if (!instance)
113 return throwVMTypeError(globalObject, scope);
114 }
115 ASSERT(instance);
116
117 instance->begin();
118 JSValue result = instance->invokeMethod(globalObject, callFrame, method);
119 instance->end();
120 return JSValue::encode(result);
121}
