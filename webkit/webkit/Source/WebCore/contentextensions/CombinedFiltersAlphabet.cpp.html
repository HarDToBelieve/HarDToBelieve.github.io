35struct TermCreatorInput {
36 const Term& term;
37 Vector<std::unique_ptr<Term>>& internedTermsStorage;
38};
39
40struct TermCreatorTranslator {
41 static unsigned hash(const TermCreatorInput& input)
42 {
43 return input.term.hash();
44 }
45
46 static inline bool equal(const Term* term, const TermCreatorInput& input)
47 {
48 return *term == input.term;
49 }
50
51 static void translate(const Term*& location, const TermCreatorInput& input, unsigned)
52 {
53 std::unique_ptr<Term> newUniqueTerm(new Term(input.term));
54 location = newUniqueTerm.get();
55 input.internedTermsStorage.append(WTFMove(newUniqueTerm));
56 }
57};
58
59const Term* CombinedFiltersAlphabet::interned(const Term& term)
60{
61 TermCreatorInput input { term, m_internedTermsStorage };
62 auto addResult = m_uniqueTerms.add<TermCreatorTranslator>(input);
63 return *addResult.iterator;
64}
65
66#if CONTENT_EXTENSIONS_PERFORMANCE_REPORTING
67size_t CombinedFiltersAlphabet::memoryUsed() const
68{
69 size_t termsSize = 0;
70 for (const auto& termPointer : m_internedTermsStorage)
71 termsSize += termPointer->memoryUsed();
72 return sizeof(CombinedFiltersAlphabet)
73 + termsSize
74 + m_uniqueTerms.capacity() * sizeof(Term*)
75 + m_internedTermsStorage.capacity() * sizeof(std::unique_ptr<Term>);
76}
