68static size_t recursiveMemoryUsed(const PrefixTreeVertex& vertex)
69{
70 size_t size = sizeof(PrefixTreeVertex)
71 + vertex.edges.capacity() * sizeof(PrefixTreeEdge);
72 for (const auto& edge : vertex.edges) {
73 ASSERT(edge.child);
74 size += recursiveMemoryUsed(*edge.child.get());
75 }
76 return size;
77}

79size_t CombinedURLFilters::memoryUsed() const
80{
81 ASSERT(m_prefixTreeRoot);
82
83 size_t actionsSize = 0;
84 for (const auto& slot : m_actions)
85 actionsSize += slot.value.capacity() * sizeof(uint64_t);
86
87 return sizeof(CombinedURLFilters)
88 + m_alphabet.memoryUsed()
89 + recursiveMemoryUsed(*m_prefixTreeRoot.get())
90 + sizeof(HashMap<PrefixTreeVertex*, ActionList>)
91 + m_actions.capacity() * (sizeof(PrefixTreeVertex*) + sizeof(ActionList))
92 + actionsSize;
93}

97static String prefixTreeVertexToString(const PrefixTreeVertex& vertex, const HashMap<const PrefixTreeVertex*, ActionList>& actions, unsigned depth)
98{
99 StringBuilder builder;
100 while (depth--)
101 builder.append(" ");
102 builder.append("vertex actions: ");
103
104 auto actionsSlot = actions.find(&vertex);
105 if (actionsSlot != actions.end()) {
106 for (uint64_t action : actionsSlot->value)
107 builder.append(action, ',');
108 }
109 builder.append('\n');
110 return builder.toString();
111}

113static void recursivePrint(const PrefixTreeVertex& vertex, const HashMap<const PrefixTreeVertex*, ActionList>& actions, unsigned depth)
114{
115 dataLogF("%s", prefixTreeVertexToString(vertex, actions, depth).utf8().data());
116 for (const auto& edge : vertex.edges) {
117 StringBuilder builder;
118 for (unsigned i = 0; i < depth * 2; ++i)
119 builder.append(' ');
120 builder.append("vertex edge: ", edge.term->toString(), '\n');
121 dataLogF("%s", builder.toString().utf8().data());
122 ASSERT(edge.child);
123 recursivePrint(*edge.child.get(), actions, depth + 1);
124 }

145void CombinedURLFilters::addDomain(uint64_t actionId, const String& domain)
146{
147 unsigned domainLength = domain.length();
148 if (domainLength && domain[0] == '*') {
149 // If domain starts with a '*' then it means match domain and its subdomains, like (^|.)domain$
150 // This way a domain of "*webkit.org" will match "bugs.webkit.org" and "webkit.org".
151 Vector<Term> prependDot;
152 Vector<Term> prependBeginningOfLine;
153 prependDot.reserveInitialCapacity(domainLength + 2);
154 prependBeginningOfLine.reserveInitialCapacity(domainLength); // This is just no .* at the beginning.
155
156 Term canonicalDotStar(Term::UniversalTransition);
157 canonicalDotStar.quantify(AtomQuantifier::ZeroOrMore);
158 prependDot.uncheckedAppend(canonicalDotStar);
159 prependDot.uncheckedAppend(Term('.', true));
160
161 for (unsigned i = 1; i < domainLength; i++) {
162 ASSERT(isASCII(domain[i]));
163 ASSERT(!isASCIIUpper(domain[i]));
164 prependDot.uncheckedAppend(Term(domain[i], true));
165 prependBeginningOfLine.uncheckedAppend(Term(domain[i], true));
166 }
167 prependDot.uncheckedAppend(Term::EndOfLineAssertionTerm);
168 prependBeginningOfLine.uncheckedAppend(Term::EndOfLineAssertionTerm);
169
170 addPattern(actionId, prependDot);
171 addPattern(actionId, prependBeginningOfLine);
172 } else {
173 // This is like adding ^domain$, but interpreting domain as a series of characters, not a regular expression.
174 // "webkit.org" will match "webkit.org" but not "bugs.webkit.org".
175 Vector<Term> prependBeginningOfLine;
176 prependBeginningOfLine.reserveInitialCapacity(domainLength + 1); // This is just no .* at the beginning.
177 for (unsigned i = 0; i < domainLength; i++) {
178 ASSERT(isASCII(domain[i]));
179 ASSERT(!isASCIIUpper(domain[i]));
180 prependBeginningOfLine.uncheckedAppend(Term(domain[i], true));
181 }
182 prependBeginningOfLine.uncheckedAppend(Term::EndOfLineAssertionTerm);
183 addPattern(actionId, prependBeginningOfLine);
184 }
185}

187void CombinedURLFilters::addPattern(uint64_t actionId, const Vector<Term>& pattern)
188{
189 ASSERT_WITH_MESSAGE(!pattern.isEmpty(), "The parser should have excluded empty patterns before reaching CombinedURLFilters.");
190
191 if (pattern.isEmpty())
192 return;
193
194 // Extend the prefix tree with the new pattern.
195 PrefixTreeVertex* lastPrefixTree = m_prefixTreeRoot.get();
196
197 for (const Term& term : pattern) {
198 size_t nextEntryIndex = WTF::notFound;
199 for (size_t i = 0; i < lastPrefixTree->edges.size(); ++i) {
200 if (*lastPrefixTree->edges[i].term == term) {
201 nextEntryIndex = i;
202 break;
203 }
204 }
205 if (nextEntryIndex != WTF::notFound)
206 lastPrefixTree = lastPrefixTree->edges[nextEntryIndex].child.get();
207 else {
208 lastPrefixTree->edges.append(PrefixTreeEdge({m_alphabet.interned(term), makeUnique<PrefixTreeVertex>()}));
209 lastPrefixTree = lastPrefixTree->edges.last().child.get();
210 }
211 }
212
213 auto addResult = m_actions.add(lastPrefixTree, ActionList());
214 ActionList& actions = addResult.iterator->value;
215 if (actions.find(actionId) == WTF::notFound)
216 actions.append(actionId);
217}

219struct ActiveSubtree {
220 ActiveSubtree(PrefixTreeVertex& vertex, ImmutableCharNFANodeBuilder&& nfaNode, unsigned edgeIndex)
221 : vertex(vertex)
222 , nfaNode(WTFMove(nfaNode))
223 , edgeIndex(edgeIndex)
224 {
225 }
226 PrefixTreeVertex& vertex;
227 ImmutableCharNFANodeBuilder nfaNode;
228 unsigned edgeIndex;
229};

231static void generateInfixUnsuitableForReverseSuffixTree(NFA& nfa, Vector<ActiveSubtree>& stack, const HashMap<const PrefixTreeVertex*, ActionList>& actions)
232{
233 // To avoid conflicts, we use the reverse suffix tree for subtrees that do not merge
234 // in the prefix tree.
235 //
236 // We only unify the suffixes to the actions on the leaf.
237 // If there are actions inside the tree, we generate the part of the subtree up to the action.
238 //
239 // If we accidentally insert a node with action inside the reverse-suffix-tree, we would create
240 // new actions on unrelated pattern when unifying their suffixes.
241 for (unsigned i = stack.size() - 1; i--;) {
242 ActiveSubtree& activeSubtree = stack[i];
243 if (activeSubtree.nfaNode.isValid())
244 return;
245
246 RELEASE_ASSERT_WITH_MESSAGE(i > 0, "The bottom of the stack must be the root of our fixed-length subtree. It should have it the isValid() case above.");
247
248 auto actionsIterator = actions.find(&activeSubtree.vertex);
249 bool hasActionInsideTree = actionsIterator != actions.end();
250
251 // Stricto sensu, we should count the number of exit edges with fixed length.
252 // That is costly and unlikely to matter in practice.
253 bool hasSingleOutcome = activeSubtree.vertex.edges.size() == 1;
254
255 if (hasActionInsideTree || !hasSingleOutcome) {
256 // Go back to the end of the subtree that has already been generated.
257 // From there, generate everything up to the vertex we found.
258 unsigned end = i;
259 unsigned beginning = end;
260
261 ActiveSubtree* sourceActiveSubtree = nullptr;
262 while (beginning--) {
263 ActiveSubtree& activeSubtree = stack[beginning];
264 if (activeSubtree.nfaNode.isValid()) {
265 sourceActiveSubtree = &activeSubtree;
266 break;
267 }
268 }
269 ASSERT_WITH_MESSAGE(sourceActiveSubtree, "The root should always have a valid generator.");
270
271 for (unsigned stackIndex = beginning + 1; stackIndex <= end; ++stackIndex) {
272 ImmutableCharNFANodeBuilder& sourceNode = sourceActiveSubtree->nfaNode;
273 ASSERT(sourceNode.isValid());
274 auto& edge = sourceActiveSubtree->vertex.edges[sourceActiveSubtree->edgeIndex];
275
276 ActiveSubtree& destinationActiveSubtree = stack[stackIndex];
277 destinationActiveSubtree.nfaNode = edge.term->generateGraph(nfa, sourceNode, actions.get(&destinationActiveSubtree.vertex));
278
279 sourceActiveSubtree = &destinationActiveSubtree;
280 }
281
282 return;
283 }
284 }

287static void generateSuffixWithReverseSuffixTree(NFA& nfa, Vector<ActiveSubtree>& stack, const HashMap<const PrefixTreeVertex*, ActionList>& actions, ReverseSuffixTreeRoots& reverseSuffixTreeRoots)
288{
289 ActiveSubtree& leafSubtree = stack.last();
290 ASSERT_WITH_MESSAGE(!leafSubtree.nfaNode.isValid(), "The leaf should never be generated by the code above, it should always be inserted into the prefix tree.");
291
292 ActionList actionList = actions.get(&leafSubtree.vertex);
293 ASSERT_WITH_MESSAGE(!actionList.isEmpty(), "Our prefix tree should always have actions on the leaves by construction.");
294
295 HashableActionList hashableActionList(actionList);
296 auto rootAddResult = reverseSuffixTreeRoots.add(hashableActionList, ReverseSuffixTreeVertex());
297 if (rootAddResult.isNewEntry) {
298 ImmutableCharNFANodeBuilder newNode(nfa);
299 newNode.setActions(actionList.begin(), actionList.end());
300 rootAddResult.iterator->value.nodeId = newNode.nodeId();
301 }
302
303 ReverseSuffixTreeVertex* activeReverseSuffixTreeVertex = &rootAddResult.iterator->value;
304 uint32_t destinationNodeId = rootAddResult.iterator->value.nodeId;
305
306 unsigned stackPosition = stack.size() - 2;
307 while (true) {
308 ActiveSubtree& source = stack[stackPosition];
309 auto& edge = source.vertex.edges[source.edgeIndex];
310
311 // This is the end condition: when we meet a node that has already been generated,
312 // we just need to connect our backward tree to the forward tree.
313 //
314 // We *must not* add this last node to the reverse-suffix tree. That node can have
315 // transitions back to earlier part of the prefix tree. If the prefix tree "caches"
316 // such node, it would create new transitions that did not exist in the source language.
317 if (source.nfaNode.isValid()) {
318 stack.shrink(stackPosition + 1);
319 edge.term->generateGraph(nfa, source.nfaNode, destinationNodeId);
320 return;
321 }
322 --stackPosition;
323
324 ASSERT_WITH_MESSAGE(!actions.contains(&source.vertex), "Any node with final actions should have been created before hitting the reverse suffix-tree.");
325
326 ReverseSuffixTreeEdge* existingEdge = nullptr;
327 for (ReverseSuffixTreeEdge& potentialExistingEdge : activeReverseSuffixTreeVertex->edges) {
328 if (edge.term == potentialExistingEdge.term) {
329 existingEdge = &potentialExistingEdge;
330 break;
331 }
332 }
333
334 if (existingEdge)
335 activeReverseSuffixTreeVertex = existingEdge->child.get();
336 else {
337 ImmutableCharNFANodeBuilder newNode(nfa);
338 edge.term->generateGraph(nfa, newNode, destinationNodeId);
339 std::unique_ptr<ReverseSuffixTreeVertex> newVertex(new ReverseSuffixTreeVertex());
340 newVertex->nodeId = newNode.nodeId();
341
342 ReverseSuffixTreeVertex* newVertexAddress = newVertex.get();
343 activeReverseSuffixTreeVertex->edges.append(ReverseSuffixTreeEdge({ edge.term, WTFMove(newVertex) }));
344 activeReverseSuffixTreeVertex = newVertexAddress;
345 }
346 destinationNodeId = activeReverseSuffixTreeVertex->nodeId;
347
348 ASSERT(source.vertex.edges.size() == 1);
349 source.vertex.edges.clear();
350 }
351
352 RELEASE_ASSERT_NOT_REACHED();
353}

355static void clearReverseSuffixTree(ReverseSuffixTreeRoots& reverseSuffixTreeRoots)
356{
357 // We cannot rely on the destructor being called in order from top to bottom as we may overflow
358 // the stack. Instead, we go depth first in the reverse-suffix-tree.
359
360 for (auto& slot : reverseSuffixTreeRoots) {
361 Vector<ReverseSuffixTreeVertex*, 128> stack;
362 stack.append(&slot.value);
363
364 while (true) {
365 ReverseSuffixTreeVertex* top = stack.last();
366 if (top->edges.isEmpty()) {
367 stack.removeLast();
368 if (stack.isEmpty())
369 break;
370 stack.last()->edges.removeLast();
371 } else
372 stack.append(top->edges.last().child.get());
373 }
374 }
375 reverseSuffixTreeRoots.clear();
376}

378static void generateNFAForSubtree(NFA& nfa, ImmutableCharNFANodeBuilder&& subtreeRoot, PrefixTreeVertex& root, const HashMap<const PrefixTreeVertex*, ActionList>& actions, size_t maxNFASize)
379{
380 // This recurses the subtree of the prefix tree.
381 // For each edge that has fixed length (no quantifiers like ?, *, or +) it generates the nfa graph,
382 // recurses into children, and deletes any processed leaf nodes.
383
384 ReverseSuffixTreeRoots reverseSuffixTreeRoots;
385 Vector<ActiveSubtree> stack;
386 if (!root.edges.isEmpty())
387 stack.append(ActiveSubtree(root, WTFMove(subtreeRoot), 0));
388
389 bool nfaTooBig = false;
390
391 // Generate graphs for each subtree that does not contain any quantifiers.
392 while (!stack.isEmpty()) {
393 PrefixTreeVertex& vertex = stack.last().vertex;
394 const unsigned edgeIndex = stack.last().edgeIndex;
395
396 if (edgeIndex < vertex.edges.size()) {
397 auto& edge = vertex.edges[edgeIndex];
398
399 // Clean up any processed leaves and return early if we are past the maxNFASize.
400 if (nfaTooBig) {
401 stack.last().edgeIndex = stack.last().vertex.edges.size();
402 continue;
403 }
404
405 // Quantified edges in the subtree will be a part of another NFA.
406 if (!edge.term->hasFixedLength()) {
407 stack.last().edgeIndex++;
408 continue;
409 }
410
411 ASSERT(edge.child.get());
412 ImmutableCharNFANodeBuilder emptyBuilder;
413 stack.append(ActiveSubtree(*edge.child.get(), WTFMove(emptyBuilder), 0));
414 } else {
415 bool isLeaf = vertex.edges.isEmpty();
416
417 ASSERT(edgeIndex == vertex.edges.size());
418 vertex.edges.removeAllMatching([](PrefixTreeEdge& edge)
419 {
420 return !edge.term;
421 });
422
423 if (isLeaf) {
424 generateInfixUnsuitableForReverseSuffixTree(nfa, stack, actions);
425 generateSuffixWithReverseSuffixTree(nfa, stack, actions, reverseSuffixTreeRoots);
426
427 // Only stop generating an NFA at a leaf to ensure we have a correct NFA. We could go slightly over the maxNFASize.
428 if (nfa.nodes.size() > maxNFASize)
429 nfaTooBig = true;
430 } else
431 stack.removeLast();
432
433 if (!stack.isEmpty()) {
434 auto& activeSubtree = stack.last();
435 auto& edge = activeSubtree.vertex.edges[stack.last().edgeIndex];
436 if (edge.child->edges.isEmpty())
437 edge.term = nullptr; // Mark this leaf for deleting.
438 activeSubtree.edgeIndex++;
439 }
440 }
441 }
442 clearReverseSuffixTree(reverseSuffixTreeRoots);
443}

445bool CombinedURLFilters::processNFAs(size_t maxNFASize, Function<bool(NFA&&)>&& handler)
446{
447#if CONTENT_EXTENSIONS_STATE_MACHINE_DEBUGGING
448 print();
449#endif
450 while (true) {
451 // Traverse out to a leaf.
452 Vector<PrefixTreeVertex*, 128> stack;
453 PrefixTreeVertex* vertex = m_prefixTreeRoot.get();
454 while (true) {
455 ASSERT(vertex);
456 stack.append(vertex);
457 if (vertex->edges.isEmpty())
458 break;
459 vertex = vertex->edges.last().child.get();
460 }
461 if (stack.size() == 1)
462 break; // We're done once we have processed and removed all the edges in the prefix tree.
463
464 // Find the prefix root for this NFA. This is the vertex after the last term with a quantifier if there is one,
465 // or the root if there are no quantifiers left.
466 while (stack.size() > 1) {
467 if (!stack[stack.size() - 2]->edges.last().term->hasFixedLength())
468 break;
469 stack.removeLast();
470 }
471 ASSERT_WITH_MESSAGE(!stack.isEmpty(), "At least the root should be in the stack");
472
473 // Make an NFA with the subtrees for whom this is also the last quantifier (or who also have no quantifier).
474 NFA nfa;
475 {
476 // Put the prefix into the NFA.
477 ImmutableCharNFANodeBuilder lastNode(nfa);
478 for (unsigned i = 0; i < stack.size() - 1; ++i) {
479 const PrefixTreeEdge& edge = stack[i]->edges.last();
480 ImmutableCharNFANodeBuilder newNode = edge.term->generateGraph(nfa, lastNode, m_actions.get(edge.child.get()));
481 lastNode = WTFMove(newNode);
482 }
483
484 // Put the non-quantified vertices in the subtree into the NFA and delete them.
485 ASSERT(stack.last());
486 generateNFAForSubtree(nfa, WTFMove(lastNode), *stack.last(), m_actions, maxNFASize);
487 }
488
489 if (!handler(WTFMove(nfa)))
490 return false;
491
492 // Clean up any processed leaf nodes.
493 while (true) {
494 if (stack.size() > 1) {
495 if (stack[stack.size() - 1]->edges.isEmpty()) {
496 stack[stack.size() - 2]->edges.removeLast();
497 stack.removeLast();
498 } else
499 break; // Vertex is not a leaf.
500 } else
501 break; // Leave the empty root.
502 }
503 }
504 return true;
505}
