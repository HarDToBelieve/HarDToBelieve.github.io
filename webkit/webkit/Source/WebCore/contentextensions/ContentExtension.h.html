30#include "DFABytecodeInterpreter.h"

36namespace WebCore {
37
38namespace ContentExtensions {
39
40class CompiledContentExtension;
41
42class ContentExtension : public RefCounted<ContentExtension> {
43public:
44 enum class ShouldCompileCSS { No, Yes };
45 static Ref<ContentExtension> create(const String& identifier, Ref<CompiledContentExtension>&&, ShouldCompileCSS = ShouldCompileCSS::Yes);
46
47 const String& identifier() const { return m_identifier; }
48 const CompiledContentExtension& compiledExtension() const { return m_compiledExtension.get(); }
49 StyleSheetContents* globalDisplayNoneStyleSheet();
50 const DFABytecodeInterpreter::Actions& topURLActions(const URL& topURL);
51 const Vector<uint32_t>& universalActionsWithoutConditions() { return m_universalActionsWithoutConditions; }
52 const Vector<uint32_t>& universalActionsWithConditions(const URL& topURL);
53
54private:
55 ContentExtension(const String& identifier, Ref<CompiledContentExtension>&&, ShouldCompileCSS);
56 uint32_t findFirstIgnorePreviousRules() const;
57
58 String m_identifier;
59 Ref<CompiledContentExtension> m_compiledExtension;
60
61 RefPtr<StyleSheetContents> m_globalDisplayNoneStyleSheet;
62 void compileGlobalDisplayNoneStyleSheet();
63
64 URL m_cachedTopURL;
65 void populateConditionCacheIfNeeded(const URL& topURL);
66 DFABytecodeInterpreter::Actions m_cachedTopURLActions;
67 Vector<uint32_t> m_cachedUniversalConditionedActions;
68
69 Vector<uint32_t> m_universalActionsWithoutConditions;
70 Vector<uint64_t> m_universalActionsWithConditions;
71};
72
73} // namespace ContentExtensions
74} // namespace WebCore
