50static bool containsOnlyASCIIWithNoUppercase(const String& domain)
51{
52 for (auto character : StringView { domain }.codeUnits()) {
53 if (!isASCII(character) || isASCIIUpper(character))
54 return false;
55 }
56 return true;
57}

59static Expected<Vector<String>, std::error_code> getStringList(JSGlobalObject& lexicalGlobalObject, const JSObject* arrayObject)
60{
61 static const ContentExtensionError error = ContentExtensionError::JSONInvalidConditionList;
62 VM& vm = lexicalGlobalObject.vm();
63 auto scope = DECLARE_THROW_SCOPE(vm);
64
65 if (!arrayObject || !isJSArray(arrayObject))
66 return makeUnexpected(error);
67 const JSArray* array = jsCast<const JSArray*>(arrayObject);
68
69 Vector<String> strings;
70 unsigned length = array->length();
71 for (unsigned i = 0; i < length; ++i) {
72 const JSValue value = array->getIndex(&lexicalGlobalObject, i);
73 if (scope.exception() || !value.isString())
74 return makeUnexpected(error);
75
76 const String& string = asString(value)->value(&lexicalGlobalObject);
77 if (string.isEmpty())
78 return makeUnexpected(error);
79 strings.append(string);
80 }
81 return strings;
82}

84static Expected<Vector<String>, std::error_code> getDomainList(JSGlobalObject& lexicalGlobalObject, const JSObject* arrayObject)
85{
86 auto strings = getStringList(lexicalGlobalObject, arrayObject);
87 if (!strings.has_value())
88 return strings;
89 for (auto& domain : strings.value()) {
90 // Domains should be punycode encoded lower case.
91 if (!containsOnlyASCIIWithNoUppercase(domain))
92 return makeUnexpected(ContentExtensionError::JSONDomainNotLowerCaseASCII);
93 }
94 return strings;
95}

97template<typename T>
98static std::error_code getTypeFlags(JSGlobalObject& lexicalGlobalObject, const JSValue& typeValue, ResourceFlags& flags, std::optional<OptionSet<T>> (*stringToType)(StringView))
99{
100 VM& vm = lexicalGlobalObject.vm();
101 auto scope = DECLARE_THROW_SCOPE(vm);
102
103 if (!typeValue.isObject())
104 return { };
105
106 const JSObject* object = typeValue.toObject(&lexicalGlobalObject);
107 scope.assertNoException();
108 if (!isJSArray(object))
109 return ContentExtensionError::JSONInvalidTriggerFlagsArray;
110
111 const JSArray* array = jsCast<const JSArray*>(object);
112
113 unsigned length = array->length();
114 for (unsigned i = 0; i < length; ++i) {
115 const JSValue value = array->getIndex(&lexicalGlobalObject, i);
116 if (scope.exception() || !value)
117 return ContentExtensionError::JSONInvalidObjectInTriggerFlagsArray;
118
119 String name = value.toWTFString(&lexicalGlobalObject);
120 auto type = stringToType(StringView(name));
121 if (!type)
122 return ContentExtensionError::JSONInvalidStringInTriggerFlagsArray;
123
124 flags |= static_cast<ResourceFlags>(type->toRaw());
125 }
126
127 return { };
128}

130static Expected<Trigger, std::error_code> loadTrigger(JSGlobalObject& lexicalGlobalObject, const JSObject& ruleObject)
131{
132 VM& vm = lexicalGlobalObject.vm();
133 auto scope = DECLARE_THROW_SCOPE(vm);
134
135 const JSValue triggerObject = ruleObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "trigger"));
136 if (!triggerObject || scope.exception() || !triggerObject.isObject())
137 return makeUnexpected(ContentExtensionError::JSONInvalidTrigger);
138
139 const JSValue urlFilterObject = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "url-filter"));
140 if (!urlFilterObject || scope.exception() || !urlFilterObject.isString())
141 return makeUnexpected(ContentExtensionError::JSONInvalidURLFilterInTrigger);
142
143 String urlFilter = asString(urlFilterObject)->value(&lexicalGlobalObject);
144 if (urlFilter.isEmpty())
145 return makeUnexpected(ContentExtensionError::JSONInvalidURLFilterInTrigger);
146
147 Trigger trigger;
148 trigger.urlFilter = urlFilter;
149
150 const JSValue urlFilterCaseValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "url-filter-is-case-sensitive"));
151 if (urlFilterCaseValue && !scope.exception() && urlFilterCaseValue.isBoolean())
152 trigger.urlFilterIsCaseSensitive = urlFilterCaseValue.toBoolean(&lexicalGlobalObject);
153
154 const JSValue topURLFilterCaseValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "top-url-filter-is-case-sensitive"));
155 if (topURLFilterCaseValue && !scope.exception() && topURLFilterCaseValue.isBoolean())
156 trigger.topURLConditionIsCaseSensitive = topURLFilterCaseValue.toBoolean(&lexicalGlobalObject);
157
158 const JSValue resourceTypeValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "resource-type"));
159 if (!scope.exception() && resourceTypeValue.isObject()) {
160 auto typeFlagsError = getTypeFlags(lexicalGlobalObject, resourceTypeValue, trigger.flags, readResourceType);
161 if (typeFlagsError)
162 return makeUnexpected(typeFlagsError);
163 } else if (!resourceTypeValue.isUndefined())
164 return makeUnexpected(ContentExtensionError::JSONInvalidTriggerFlagsArray);
165
166 const JSValue loadTypeValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "load-type"));
167 if (!scope.exception() && loadTypeValue.isObject()) {
168 auto typeFlagsError = getTypeFlags(lexicalGlobalObject, loadTypeValue, trigger.flags, readLoadType);
169 if (typeFlagsError)
170 return makeUnexpected(typeFlagsError);
171 } else if (!loadTypeValue.isUndefined())
172 return makeUnexpected(ContentExtensionError::JSONInvalidTriggerFlagsArray);
173
174 const JSValue loadContextValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "load-context"));
175 if (!scope.exception() && loadContextValue.isObject()) {
176 auto typeFlagsError = getTypeFlags(lexicalGlobalObject, loadContextValue, trigger.flags, readLoadContext);
177 if (typeFlagsError)
178 return makeUnexpected(typeFlagsError);
179 } else if (!loadContextValue.isUndefined())
180 return makeUnexpected(ContentExtensionError::JSONInvalidTriggerFlagsArray);
181
182 const JSValue ifDomainValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "if-domain"));
183 if (!scope.exception() && ifDomainValue.isObject()) {
184 auto ifDomain = getDomainList(lexicalGlobalObject, asObject(ifDomainValue));
185 if (!ifDomain.has_value())
186 return makeUnexpected(ifDomain.error());
187 trigger.conditions = WTFMove(ifDomain.value());
188 if (trigger.conditions.isEmpty())
189 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
190 ASSERT(trigger.conditionType == Trigger::ConditionType::None);
191 trigger.conditionType = Trigger::ConditionType::IfDomain;
192 } else if (!ifDomainValue.isUndefined())
193 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
194
195 const JSValue unlessDomainValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "unless-domain"));
196 if (!scope.exception() && unlessDomainValue.isObject()) {
197 if (trigger.conditionType != Trigger::ConditionType::None)
198 return makeUnexpected(ContentExtensionError::JSONMultipleConditions);
199 auto unlessDomain = getDomainList(lexicalGlobalObject, asObject(unlessDomainValue));
200 if (!unlessDomain.has_value())
201 return makeUnexpected(unlessDomain.error());
202 trigger.conditions = WTFMove(unlessDomain.value());
203 if (trigger.conditions.isEmpty())
204 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
205 trigger.conditionType = Trigger::ConditionType::UnlessDomain;
206 } else if (!unlessDomainValue.isUndefined())
207 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
208
209 const JSValue ifTopURLValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "if-top-url"));
210 if (!scope.exception() && ifTopURLValue.isObject()) {
211 if (trigger.conditionType != Trigger::ConditionType::None)
212 return makeUnexpected(ContentExtensionError::JSONMultipleConditions);
213 auto ifTopURL = getStringList(lexicalGlobalObject, asObject(ifTopURLValue));
214 if (!ifTopURL.has_value())
215 return makeUnexpected(ifTopURL.error());
216 trigger.conditions = WTFMove(ifTopURL.value());
217 if (trigger.conditions.isEmpty())
218 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
219 trigger.conditionType = Trigger::ConditionType::IfTopURL;
220 } else if (!ifTopURLValue.isUndefined())
221 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
222
223 const JSValue unlessTopURLValue = triggerObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "unless-top-url"));
224 if (!scope.exception() && unlessTopURLValue.isObject()) {
225 if (trigger.conditionType != Trigger::ConditionType::None)
226 return makeUnexpected(ContentExtensionError::JSONMultipleConditions);
227 auto unlessTopURL = getStringList(lexicalGlobalObject, asObject(unlessTopURLValue));
228 if (!unlessTopURL.has_value())
229 return makeUnexpected(unlessTopURL.error());
230 trigger.conditions = WTFMove(unlessTopURL.value());
231 if (trigger.conditions.isEmpty())
232 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
233 trigger.conditionType = Trigger::ConditionType::UnlessTopURL;
234 } else if (!unlessTopURLValue.isUndefined())
235 return makeUnexpected(ContentExtensionError::JSONInvalidConditionList);
236
237 return trigger;
238}

240bool isValidCSSSelector(const String& selector)
241{
242 ASSERT(isMainThread());
243 AtomString::init();
244 QualifiedName::init();
245 CSSParserContext context(HTMLQuirksMode);
246 CSSParser parser(context);
247 return !!parser.parseSelector(selector);
248}

250static Expected<std::optional<Action>, std::error_code> loadAction(JSGlobalObject& lexicalGlobalObject, const JSObject& ruleObject)
251{
252 VM& vm = lexicalGlobalObject.vm();
253 auto scope = DECLARE_THROW_SCOPE(vm);
254
255 const JSValue actionObject = ruleObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "action"));
256 if (scope.exception() || !actionObject.isObject())
257 return makeUnexpected(ContentExtensionError::JSONInvalidAction);
258
259 const JSValue typeObject = actionObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "type"));
260 if (scope.exception() || !typeObject.isString())
261 return makeUnexpected(ContentExtensionError::JSONInvalidActionType);
262
263 String actionType = asString(typeObject)->value(&lexicalGlobalObject);
264
265 if (actionType == "block")
266 return { Action(ActionType::BlockLoad) };
267 if (actionType == "ignore-previous-rules")
268 return { Action(ActionType::IgnorePreviousRules) };
269 if (actionType == "block-cookies")
270 return { Action(ActionType::BlockCookies) };
271 if (actionType == "css-display-none") {
272 JSValue selector = actionObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "selector"));
273 if (scope.exception() || !selector.isString())
274 return makeUnexpected(ContentExtensionError::JSONInvalidCSSDisplayNoneActionType);
275
276 String selectorString = asString(selector)->value(&lexicalGlobalObject);
277 if (!isValidCSSSelector(selectorString)) {
278 // Skip rules with invalid selectors to be backwards-compatible.
279 return { std::nullopt };
280 }
281 return { Action(ActionType::CSSDisplayNoneSelector, selectorString) };
282 }
283 if (actionType == "make-https")
284 return { Action(ActionType::MakeHTTPS) };
285 if (actionType == "notify") {
286 JSValue notification = actionObject.get(&lexicalGlobalObject, Identifier::fromString(vm, "notification"));
287 if (scope.exception() || !notification.isString())
288 return makeUnexpected(ContentExtensionError::JSONInvalidNotification);
289 return { Action(ActionType::Notify, asString(notification)->value(&lexicalGlobalObject)) };
290 }
291 return makeUnexpected(ContentExtensionError::JSONInvalidActionType);
292}

294static Expected<std::optional<ContentExtensionRule>, std::error_code> loadRule(JSGlobalObject& lexicalGlobalObject, const JSObject& ruleObject)
295{
296 auto trigger = loadTrigger(lexicalGlobalObject, ruleObject);
297 if (!trigger.has_value())
298 return makeUnexpected(trigger.error());
299
300 auto action = loadAction(lexicalGlobalObject, ruleObject);
301 if (!action.has_value())
302 return makeUnexpected(action.error());
303
304 if (action.value())
305 return {{{ WTFMove(trigger.value()), WTFMove(action.value().value()) }}};
306
307 return { std::nullopt };
308}

310static Expected<Vector<ContentExtensionRule>, std::error_code> loadEncodedRules(JSGlobalObject& lexicalGlobalObject, const String& ruleJSON)
311{
312 VM& vm = lexicalGlobalObject.vm();
313 auto scope = DECLARE_THROW_SCOPE(vm);
314
315 // FIXME: JSONParse should require callbacks instead of an ExecState.
316 const JSValue decodedRules = JSONParse(&lexicalGlobalObject, ruleJSON);
317
318 if (scope.exception() || !decodedRules)
319 return makeUnexpected(ContentExtensionError::JSONInvalid);
320
321 if (!decodedRules.isObject())
322 return makeUnexpected(ContentExtensionError::JSONTopLevelStructureNotAnObject);
323
324 const JSObject* topLevelObject = decodedRules.toObject(&lexicalGlobalObject);
325 if (!topLevelObject || scope.exception())
326 return makeUnexpected(ContentExtensionError::JSONTopLevelStructureNotAnObject);
327
328 if (!isJSArray(topLevelObject))
329 return makeUnexpected(ContentExtensionError::JSONTopLevelStructureNotAnArray);
330
331 const JSArray* topLevelArray = jsCast<const JSArray*>(topLevelObject);
332
333 Vector<ContentExtensionRule> ruleList;
334
335 unsigned length = topLevelArray->length();
336 const unsigned maxRuleCount = 150000;
337 if (length > maxRuleCount)
338 return makeUnexpected(ContentExtensionError::JSONTooManyRules);
339 for (unsigned i = 0; i < length; ++i) {
340 const JSValue value = topLevelArray->getIndex(&lexicalGlobalObject, i);
341 if (scope.exception() || !value)
342 return makeUnexpected(ContentExtensionError::JSONInvalidObjectInTopLevelArray);
343
344 const JSObject* ruleObject = value.toObject(&lexicalGlobalObject);
345 if (!ruleObject || scope.exception())
346 return makeUnexpected(ContentExtensionError::JSONInvalidRule);
347
348 auto rule = loadRule(lexicalGlobalObject, *ruleObject);
349 if (!rule.has_value())
350 return makeUnexpected(rule.error());
351 if (rule.value())
352 ruleList.append(WTFMove(*rule.value()));
353 }
354
355 return ruleList;
356}

358Expected<Vector<ContentExtensionRule>, std::error_code> parseRuleList(const String& ruleJSON)
359{
360#if CONTENT_EXTENSIONS_PERFORMANCE_REPORTING
361 MonotonicTime loadExtensionStartTime = MonotonicTime::now();
362#endif
363 RefPtr<VM> vm = VM::create();
364
365 JSLockHolder locker(vm.get());
366 JSGlobalObject* globalObject = JSGlobalObject::create(*vm, JSGlobalObject::createStructure(*vm, jsNull()));
367
368 JSGlobalObject* lexicalGlobalObject = globalObject;
369 auto ruleList = loadEncodedRules(*lexicalGlobalObject, ruleJSON);
370
371 vm = nullptr;
372
373 if (!ruleList.has_value())
374 return makeUnexpected(ruleList.error());
375
376 if (ruleList->isEmpty())
377 return makeUnexpected(ContentExtensionError::JSONContainsNoRules);
378
379#if CONTENT_EXTENSIONS_PERFORMANCE_REPORTING
380 MonotonicTime loadExtensionEndTime = MonotonicTime::now();
381 dataLogF("Time spent loading extension %f\n", (loadExtensionEndTime - loadExtensionStartTime).seconds());
382#endif
383
384 return ruleList;
385}
