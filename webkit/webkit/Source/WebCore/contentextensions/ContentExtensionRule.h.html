39// A ContentExtensionRule is the smallest unit in a ContentExtension.
40//
41// It is composed of a trigger and an action. The trigger defines on what kind of content this extension should apply.
42// The action defines what to perform on that content.
43
44struct Trigger {
45 String urlFilter;
46 bool urlFilterIsCaseSensitive { false };
47 bool topURLConditionIsCaseSensitive { false };
48 ResourceFlags flags { 0 };
49 Vector<String> conditions;
50 enum class ConditionType {
51 None,
52 IfDomain,
53 UnlessDomain,
54 IfTopURL,
55 UnlessTopURL,
56 };
57 ConditionType conditionType { ConditionType::None };
58
59 WEBCORE_EXPORT Trigger isolatedCopy() const;
60
61 ~Trigger()
62 {
63 ASSERT(conditions.isEmpty() == (conditionType == ConditionType::None));
64 if (topURLConditionIsCaseSensitive)
65 ASSERT(conditionType == ConditionType::IfTopURL || conditionType == ConditionType::UnlessTopURL);
66 }
67
68 bool isEmpty() const
69 {
70 return urlFilter.isEmpty()
71 && !urlFilterIsCaseSensitive
72 && !topURLConditionIsCaseSensitive
73 && !flags
74 && conditions.isEmpty()
75 && conditionType == ConditionType::None;
76 }
77
78 bool operator==(const Trigger& other) const
79 {
80 return urlFilter == other.urlFilter
81 && urlFilterIsCaseSensitive == other.urlFilterIsCaseSensitive
82 && topURLConditionIsCaseSensitive == other.topURLConditionIsCaseSensitive
83 && flags == other.flags
84 && conditions == other.conditions
85 && conditionType == other.conditionType;
86 }
87};
88
89struct TriggerHash {
90 static unsigned hash(const Trigger& trigger)
91 {
92 return computeHash(trigger.urlFilterIsCaseSensitive, trigger.urlFilter, trigger.flags, trigger.conditions, trigger.conditionType);
93 }
94 static bool equal(const Trigger& a, const Trigger& b)
95 {
96 return a == b;
97 }
98 static const bool safeToCompareToEmptyOrDeleted = false;
99};
100
101struct TriggerHashTraits : public WTF::CustomHashTraits<Trigger> {
102 static const bool emptyValueIsZero = false;
103 static const bool hasIsEmptyValueFunction = true;
104
105 static void constructDeletedValue(Trigger& trigger)
106 {
107 new (NotNull, std::addressof(trigger.urlFilter)) String(WTF::HashTableDeletedValue);
108 }
109
110 static bool isDeletedValue(const Trigger& trigger)
111 {
112 return trigger.urlFilter.isHashTableDeletedValue();
113 }
114
115 static Trigger emptyValue()
116 {
117 return Trigger();
118 }
119
120 static bool isEmptyValue(const Trigger& trigger)
121 {
122 return trigger.isEmpty();
123 }
124};

126struct Action {
127 Action(ActionType type, const String& stringArgument, uint32_t actionID = std::numeric_limits<uint32_t>::max())
128 : m_type(type)
129 , m_actionID(actionID)
130 , m_stringArgument(stringArgument)
131 {
132 ASSERT(hasStringArgument(type));
133 }
134
135 Action(ActionType type, uint32_t actionID = std::numeric_limits<uint32_t>::max())
136 : m_type(type)
137 , m_actionID(actionID)
138 {
139 ASSERT(!hasStringArgument(type));
140 }
141 Action(Action&&) = default;
142
143 bool operator==(const Action& other) const
144 {
145 return m_type == other.m_type
146 && m_actionID == other.m_actionID
147 && m_stringArgument == other.m_stringArgument;
148 }
149
150 static Action deserialize(const SerializedActionByte* actions, const uint32_t actionsLength, uint32_t location);
151 static ActionType deserializeType(const SerializedActionByte* actions, const uint32_t actionsLength, uint32_t location);
152 static uint32_t serializedLength(const SerializedActionByte* actions, const uint32_t actionsLength, uint32_t location);
153
154 ActionType type() const { return m_type; }
155 uint32_t actionID() const { return m_actionID; }
156 const String& stringArgument() const { return m_stringArgument; }
157
158 WEBCORE_EXPORT Action isolatedCopy() const;
159
160private:
161 ActionType m_type;
162 uint32_t m_actionID;
163 String m_stringArgument;
164};
165
166class ContentExtensionRule {
167public:
168 WEBCORE_EXPORT ContentExtensionRule(Trigger&&, Action&&);
169
170 const Trigger& trigger() const { return m_trigger; }
171 const Action& action() const { return m_action; }
172
173 ContentExtensionRule isolatedCopy() const
174 {
175 return { m_trigger.isolatedCopy(), m_action.isolatedCopy() };
176 }
177 bool operator==(const ContentExtensionRule& other) const
178 {
179 return m_trigger == other.m_trigger && m_action == other.m_action;
180 }
181
182private:
183 Trigger m_trigger;
184 Action m_action;
185};
