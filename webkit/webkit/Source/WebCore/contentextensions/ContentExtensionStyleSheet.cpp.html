40ContentExtensionStyleSheet::ContentExtensionStyleSheet(Document& document)
41 : m_styleSheet(CSSStyleSheet::create(StyleSheetContents::create(), document))
42{
43 m_styleSheet->contents().setIsUserStyleSheet(true);
44}
45
46ContentExtensionStyleSheet::~ContentExtensionStyleSheet()
47{
48 m_styleSheet->clearOwnerNode();
49}
50
51bool ContentExtensionStyleSheet::addDisplayNoneSelector(const String& selector, uint32_t selectorID)
52{
53 ASSERT(selectorID != std::numeric_limits<uint32_t>::max());
54
55 if (!m_addedSelectorIDs.add(selectorID).isNewEntry)
56 return false;
57
58 StringBuilder css;
59 css.append(selector);
60 css.append('{');
61 css.append(ContentExtensionsBackend::displayNoneCSSRule());
62 css.append('}');
63 m_styleSheet->contents().parseString(css.toString());
64 return true;
65}
