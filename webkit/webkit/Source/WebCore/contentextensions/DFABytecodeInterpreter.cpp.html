38template <typename IntType>
39static inline IntType getBits(const DFABytecode* bytecode, uint32_t bytecodeLength, uint32_t index)
40{
41 ASSERT_UNUSED(bytecodeLength, index + sizeof(IntType) <= bytecodeLength);
42 return *reinterpret_cast<const IntType*>(&bytecode[index]);
43}
44
45static inline DFABytecodeInstruction getInstruction(const DFABytecode* bytecode, uint32_t bytecodeLength, uint32_t index)
46{
47 return static_cast<DFABytecodeInstruction>(getBits<uint8_t>(bytecode, bytecodeLength, index) & DFABytecodeInstructionMask);
48}
49
50static inline size_t jumpSizeInBytes(DFABytecodeJumpSize jumpSize)
51{
52 switch (jumpSize) {
53 case Int8:
54 return sizeof(int8_t);
55 case Int16:
56 return sizeof(int16_t);
57 case Int24:
58 return sizeof(uint16_t) + sizeof(int8_t);
59 case Int32:
60 return sizeof(int32_t);
61 default:
62 RELEASE_ASSERT_NOT_REACHED();
63 }
64}
65
66static inline DFABytecodeJumpSize getJumpSize(const DFABytecode* bytecode, uint32_t bytecodeLength, uint32_t index)
67{
68 DFABytecodeJumpSize jumpSize = static_cast<DFABytecodeJumpSize>(getBits<uint8_t>(bytecode, bytecodeLength, index) & DFABytecodeJumpSizeMask);
69 ASSERT(jumpSize == DFABytecodeJumpSize::Int32 || jumpSize == DFABytecodeJumpSize::Int24 || jumpSize == DFABytecodeJumpSize::Int16 || jumpSize == DFABytecodeJumpSize::Int8);
70 return jumpSize;
71}
72
73static inline int32_t getJumpDistance(const DFABytecode* bytecode, uint32_t bytecodeLength, uint32_t index, DFABytecodeJumpSize jumpSize)
74{
75 switch (jumpSize) {
76 case Int8:
77 return getBits<int8_t>(bytecode, bytecodeLength, index);
78 case Int16:
79 return getBits<int16_t>(bytecode, bytecodeLength, index);
80 case Int24:
81 return getBits<uint16_t>(bytecode, bytecodeLength, index) | (static_cast<int32_t>(getBits<int8_t>(bytecode, bytecodeLength, index + sizeof(uint16_t))) << 16);
82 case Int32:
83 return getBits<int32_t>(bytecode, bytecodeLength, index);
84 default:
85 RELEASE_ASSERT_NOT_REACHED();
86 }
87}

89static inline bool matchesCondition(uint64_t actionAndFlags, const DFABytecodeInterpreter::Actions& conditionActions)
90{
91 bool ifCondition = actionAndFlags & IfConditionFlag;
92 bool condition = conditionActions.contains(actionAndFlags);
93 return ifCondition == condition;
94}

96void DFABytecodeInterpreter::interpretAppendAction(uint32_t& programCounter, Actions& actions, bool ifCondition)
97{
98 ASSERT(getInstruction(m_bytecode, m_bytecodeLength, programCounter) == DFABytecodeInstruction::AppendAction
99 || getInstruction(m_bytecode, m_bytecodeLength, programCounter) == DFABytecodeInstruction::AppendActionWithIfCondition);
100 uint64_t action = (ifCondition ? IfConditionFlag : 0) | static_cast<uint64_t>(getBits<uint32_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction)));
101 if (!m_topURLActions || matchesCondition(action, *m_topURLActions))
102 actions.add(action);
103
104 programCounter += instructionSizeWithArguments(DFABytecodeInstruction::AppendAction);
105 ASSERT(instructionSizeWithArguments(DFABytecodeInstruction::AppendAction) == instructionSizeWithArguments(DFABytecodeInstruction::AppendActionWithIfCondition));
106}

108void DFABytecodeInterpreter::interpretTestFlagsAndAppendAction(uint32_t& programCounter, uint16_t flags, Actions& actions, bool ifCondition)
109{
110 ASSERT(getInstruction(m_bytecode, m_bytecodeLength, programCounter) == DFABytecodeInstruction::TestFlagsAndAppendAction
111 || getInstruction(m_bytecode, m_bytecodeLength, programCounter) == DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition);
112 uint16_t flagsToCheck = getBits<uint16_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction));
113
114 uint16_t loadTypeFlags = flagsToCheck & LoadTypeMask;
115 uint16_t loadContextFlags = flagsToCheck & LoadContextMask;
116 uint16_t resourceTypeFlags = flagsToCheck & ResourceTypeMask;
117
118 bool loadTypeMatches = loadTypeFlags ? (loadTypeFlags & flags) : true;
119 bool loadContextMatches = loadContextFlags ? (loadContextFlags & flags) : true;
120 bool resourceTypeMatches = resourceTypeFlags ? (resourceTypeFlags & flags) : true;
121
122 if (loadTypeMatches && loadContextMatches && resourceTypeMatches) {
123 uint64_t actionAndFlags = (ifCondition ? IfConditionFlag : 0) | (static_cast<uint64_t>(flagsToCheck) << 32) | static_cast<uint64_t>(getBits<uint32_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction) + sizeof(uint16_t)));
124 if (!m_topURLActions || matchesCondition(actionAndFlags, *m_topURLActions))
125 actions.add(actionAndFlags);
126 }
127 programCounter += instructionSizeWithArguments(DFABytecodeInstruction::TestFlagsAndAppendAction);
128 ASSERT(instructionSizeWithArguments(DFABytecodeInstruction::TestFlagsAndAppendAction) == instructionSizeWithArguments(DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition));
129}

131template<bool caseSensitive>
132inline void DFABytecodeInterpreter::interpetJumpTable(const char* url, uint32_t& urlIndex, uint32_t& programCounter, bool& urlIndexIsAfterEndOfString)
133{
134 DFABytecodeJumpSize jumpSize = getJumpSize(m_bytecode, m_bytecodeLength, programCounter);
135
136 char character = caseSensitive ? url[urlIndex] : toASCIILower(url[urlIndex]);
137 uint8_t firstCharacter = getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction));
138 uint8_t lastCharacter = getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction) + sizeof(uint8_t));
139 if (character >= firstCharacter && character <= lastCharacter) {
140 uint32_t startOffset = programCounter + sizeof(DFABytecodeInstruction) + 2 * sizeof(uint8_t);
141 uint32_t jumpLocation = startOffset + (character - firstCharacter) * jumpSizeInBytes(jumpSize);
142 programCounter += getJumpDistance(m_bytecode, m_bytecodeLength, jumpLocation, jumpSize);
143 if (!character)
144 urlIndexIsAfterEndOfString = true;
145 urlIndex++; // This represents an edge in the DFA.
146 } else
147 programCounter += sizeof(DFABytecodeInstruction) + 2 * sizeof(uint8_t) + jumpSizeInBytes(jumpSize) * (lastCharacter - firstCharacter + 1);
148}

150DFABytecodeInterpreter::Actions DFABytecodeInterpreter::actionsMatchingEverything()
151{
152 Actions actions;
153
154 // DFA header.
155 uint32_t dfaBytecodeLength = getBits<uint32_t>(m_bytecode, m_bytecodeLength, 0);
156 uint32_t programCounter = sizeof(uint32_t);
157
158 while (programCounter < dfaBytecodeLength) {
159 DFABytecodeInstruction instruction = getInstruction(m_bytecode, m_bytecodeLength, programCounter);
160 if (instruction == DFABytecodeInstruction::AppendAction)
161 interpretAppendAction(programCounter, actions, false);
162 else if (instruction == DFABytecodeInstruction::AppendActionWithIfCondition)
163 interpretAppendAction(programCounter, actions, true);
164 else if (instruction == DFABytecodeInstruction::TestFlagsAndAppendAction)
165 programCounter += instructionSizeWithArguments(DFABytecodeInstruction::TestFlagsAndAppendAction);
166 else if (instruction == DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition)
167 programCounter += instructionSizeWithArguments(DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition);
168 else
169 break;
170 }
171 return actions;
172}

174DFABytecodeInterpreter::Actions DFABytecodeInterpreter::interpretWithConditions(const CString& urlCString, uint16_t flags, const DFABytecodeInterpreter::Actions& topURLActions)
175{
176 ASSERT(!m_topURLActions);
177 m_topURLActions = &topURLActions;
178 DFABytecodeInterpreter::Actions actions = interpret(urlCString, flags);
179 m_topURLActions = nullptr;
180 return actions;
181}

183DFABytecodeInterpreter::Actions DFABytecodeInterpreter::interpret(const CString& urlCString, uint16_t flags)
184{
185 const char* url = urlCString.data();
186 ASSERT(url);
187
188 Actions actions;
189
190 uint32_t programCounter = 0;
191 while (programCounter < m_bytecodeLength) {
192
193 // DFA header.
194 uint32_t dfaStart = programCounter;
195 uint32_t dfaBytecodeLength = getBits<uint32_t>(m_bytecode, m_bytecodeLength, programCounter);
196 programCounter += sizeof(uint32_t);
197
198 // Skip the actions without flags on the DFA root. These are accessed via actionsMatchingEverything.
199 if (!dfaStart) {
200 while (programCounter < dfaBytecodeLength) {
201 DFABytecodeInstruction instruction = getInstruction(m_bytecode, m_bytecodeLength, programCounter);
202 if (instruction == DFABytecodeInstruction::AppendAction)
203 programCounter += instructionSizeWithArguments(DFABytecodeInstruction::AppendAction);
204 else if (instruction == DFABytecodeInstruction::AppendActionWithIfCondition)
205 programCounter += instructionSizeWithArguments(DFABytecodeInstruction::AppendActionWithIfCondition);
206 else if (instruction == DFABytecodeInstruction::TestFlagsAndAppendAction)
207 interpretTestFlagsAndAppendAction(programCounter, flags, actions, false);
208 else if (instruction == DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition)
209 interpretTestFlagsAndAppendAction(programCounter, flags, actions, true);
210 else
211 break;
212 }
213 if (programCounter >= m_bytecodeLength)
214 return actions;
215 } else {
216 ASSERT_WITH_MESSAGE(getInstruction(m_bytecode, m_bytecodeLength, programCounter) != DFABytecodeInstruction::AppendAction
217 && getInstruction(m_bytecode, m_bytecodeLength, programCounter) != DFABytecodeInstruction::AppendActionWithIfCondition
218 && getInstruction(m_bytecode, m_bytecodeLength, programCounter) != DFABytecodeInstruction::TestFlagsAndAppendAction
219 && getInstruction(m_bytecode, m_bytecodeLength, programCounter) != DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition,
220 "Triggers that match everything should only be in the first DFA.");
221 }
222
223 // Interpret the bytecode from this DFA.
224 // This should always terminate if interpreting correctly compiled bytecode.
225 uint32_t urlIndex = 0;
226 bool urlIndexIsAfterEndOfString = false;
227 while (true) {
228 ASSERT(programCounter <= m_bytecodeLength);
229 switch (getInstruction(m_bytecode, m_bytecodeLength, programCounter)) {
230
231 case DFABytecodeInstruction::Terminate:
232 goto nextDFA;
233
234 case DFABytecodeInstruction::CheckValueCaseSensitive: {
235 if (urlIndexIsAfterEndOfString)
236 goto nextDFA;
237
238 // Check to see if the next character in the url is the value stored with the bytecode.
239 char character = url[urlIndex];
240 DFABytecodeJumpSize jumpSize = getJumpSize(m_bytecode, m_bytecodeLength, programCounter);
241 if (character == getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction))) {
242 uint32_t jumpLocation = programCounter + sizeof(DFABytecodeInstruction) + sizeof(uint8_t);
243 programCounter += getJumpDistance(m_bytecode, m_bytecodeLength, jumpLocation, jumpSize);
244 if (!character)
245 urlIndexIsAfterEndOfString = true;
246 urlIndex++; // This represents an edge in the DFA.
247 } else
248 programCounter += sizeof(DFABytecodeInstruction) + sizeof(uint8_t) + jumpSizeInBytes(jumpSize);
249 break;
250 }
251
252 case DFABytecodeInstruction::CheckValueCaseInsensitive: {
253 if (urlIndexIsAfterEndOfString)
254 goto nextDFA;
255
256 // Check to see if the next character in the url is the value stored with the bytecode.
257 char character = toASCIILower(url[urlIndex]);
258 DFABytecodeJumpSize jumpSize = getJumpSize(m_bytecode, m_bytecodeLength, programCounter);
259 if (character == getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction))) {
260 uint32_t jumpLocation = programCounter + sizeof(DFABytecodeInstruction) + sizeof(uint8_t);
261 programCounter += getJumpDistance(m_bytecode, m_bytecodeLength, jumpLocation, jumpSize);
262 if (!character)
263 urlIndexIsAfterEndOfString = true;
264 urlIndex++; // This represents an edge in the DFA.
265 } else
266 programCounter += sizeof(DFABytecodeInstruction) + sizeof(uint8_t) + jumpSizeInBytes(jumpSize);
267 break;
268 }
269
270 case DFABytecodeInstruction::JumpTableCaseInsensitive:
271 if (urlIndexIsAfterEndOfString)
272 goto nextDFA;
273
274 interpetJumpTable<false>(url, urlIndex, programCounter, urlIndexIsAfterEndOfString);
275 break;
276 case DFABytecodeInstruction::JumpTableCaseSensitive:
277 if (urlIndexIsAfterEndOfString)
278 goto nextDFA;
279
280 interpetJumpTable<true>(url, urlIndex, programCounter, urlIndexIsAfterEndOfString);
281 break;
282
283 case DFABytecodeInstruction::CheckValueRangeCaseSensitive: {
284 if (urlIndexIsAfterEndOfString)
285 goto nextDFA;
286
287 char character = url[urlIndex];
288 DFABytecodeJumpSize jumpSize = getJumpSize(m_bytecode, m_bytecodeLength, programCounter);
289 if (character >= getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction))
290 && character <= getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction) + sizeof(uint8_t))) {
291 uint32_t jumpLocation = programCounter + sizeof(DFABytecodeInstruction) + 2 * sizeof(uint8_t);
292 programCounter += getJumpDistance(m_bytecode, m_bytecodeLength, jumpLocation, jumpSize);
293 if (!character)
294 urlIndexIsAfterEndOfString = true;
295 urlIndex++; // This represents an edge in the DFA.
296 } else
297 programCounter += sizeof(DFABytecodeInstruction) + 2 * sizeof(uint8_t) + jumpSizeInBytes(jumpSize);
298 break;
299 }
300
301 case DFABytecodeInstruction::CheckValueRangeCaseInsensitive: {
302 if (urlIndexIsAfterEndOfString)
303 goto nextDFA;
304
305 char character = toASCIILower(url[urlIndex]);
306 DFABytecodeJumpSize jumpSize = getJumpSize(m_bytecode, m_bytecodeLength, programCounter);
307 if (character >= getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction))
308 && character <= getBits<uint8_t>(m_bytecode, m_bytecodeLength, programCounter + sizeof(DFABytecodeInstruction) + sizeof(uint8_t))) {
309 uint32_t jumpLocation = programCounter + sizeof(DFABytecodeInstruction) + 2 * sizeof(uint8_t);
310 programCounter += getJumpDistance(m_bytecode, m_bytecodeLength, jumpLocation, jumpSize);
311 if (!character)
312 urlIndexIsAfterEndOfString = true;
313 urlIndex++; // This represents an edge in the DFA.
314 } else
315 programCounter += sizeof(DFABytecodeInstruction) + 2 * sizeof(uint8_t) + jumpSizeInBytes(jumpSize);
316 break;
317 }
318
319 case DFABytecodeInstruction::Jump: {
320 if (!url[urlIndex] || urlIndexIsAfterEndOfString)
321 goto nextDFA;
322
323 uint32_t jumpLocation = programCounter + sizeof(DFABytecodeInstruction);
324 DFABytecodeJumpSize jumpSize = getJumpSize(m_bytecode, m_bytecodeLength, programCounter);
325 programCounter += getJumpDistance(m_bytecode, m_bytecodeLength, jumpLocation, jumpSize);
326 urlIndex++; // This represents an edge in the DFA.
327 break;
328 }
329
330 case DFABytecodeInstruction::AppendAction:
331 interpretAppendAction(programCounter, actions, false);
332 break;
333
334 case DFABytecodeInstruction::AppendActionWithIfCondition:
335 interpretAppendAction(programCounter, actions, true);
336 break;
337
338 case DFABytecodeInstruction::TestFlagsAndAppendAction:
339 interpretTestFlagsAndAppendAction(programCounter, flags, actions, false);
340 break;
341
342 case DFABytecodeInstruction::TestFlagsAndAppendActionWithIfCondition:
343 interpretTestFlagsAndAppendAction(programCounter, flags, actions, true);
344 break;
345
346 default:
347 RELEASE_ASSERT_NOT_REACHED(); // Invalid bytecode.
348 }
349 // We should always terminate before or at a null character at the end of a String.
350 ASSERT(urlIndex <= urlCString.length() || (urlIndexIsAfterEndOfString && urlIndex <= urlCString.length() + 1));
351 }
352 RELEASE_ASSERT_NOT_REACHED(); // The while loop can only be exited using goto nextDFA.
353 nextDFA:
354 ASSERT(dfaBytecodeLength);
355 programCounter = dfaStart + dfaBytecodeLength;
356 }
357 return actions;
358}
