38Vector<uint64_t> DFANode::actions(const DFA& dfa) const
39{
40 // FIXME: Use iterators instead of copying the Vector elements.
41 Vector<uint64_t> vector;
42 vector.reserveInitialCapacity(m_actionsLength);
43 for (uint32_t i = m_actionsStart; i < m_actionsStart + m_actionsLength; ++i)
44 vector.uncheckedAppend(dfa.actions[i]);
45 return vector;
46}

48bool DFANode::containsTransition(char transition, const DFA& dfa) const
49{
50 // Called from DFAMinimizer, this loops though a maximum of 128 transitions, so it's not too slow.
51 ASSERT(m_transitionsLength <= 128);
52 for (unsigned i = m_transitionsStart; i < m_transitionsStart + m_transitionsLength; ++i) {
53 if (dfa.transitionRanges[i].first <= transition
54 && dfa.transitionRanges[i].last >= transition)
55 return true;
56 }
57 return false;
58}

60void DFANode::kill(DFA& dfa)
61{
62 ASSERT(m_flags != IsKilled);
63 m_flags = IsKilled; // Killed nodes don't have any other flags.
64
65 // Invalidate the now-unused memory in the DFA to make finding bugs easier.
66 for (unsigned i = m_transitionsStart; i < m_transitionsStart + m_transitionsLength; ++i) {
67 dfa.transitionRanges[i] = { -1, -1 };
68 dfa.transitionDestinations[i] = std::numeric_limits<uint32_t>::max();
69 }
70 for (unsigned i = m_actionsStart; i < m_actionsStart + m_actionsLength; ++i)
71 dfa.actions[i] = std::numeric_limits<uint64_t>::max();
72
73 m_actionsStart = 0;
74 m_actionsLength = 0;
75 m_transitionsStart = 0;
76 m_transitionsLength = 0;
77};

79bool DFANode::canUseFallbackTransition(const DFA& dfa) const
80{
81 // Transitions can contain '\0' if the expression has a end-of-line marker.
82 // Fallback transitions cover 1-127. We have to be careful with the first.
83
84 IterableConstRange iterableTransitions = transitions(dfa);
85 auto iterator = iterableTransitions.begin();
86 auto end = iterableTransitions.end();
87 if (iterator == end)
88 return false;
89
90 char lastSeenCharacter = 0;
91 if (!iterator.first()) {
92 lastSeenCharacter = iterator.last();
93 if (lastSeenCharacter == 127)
94 return true;
95 ++iterator;
96 }
97
98 for (;iterator != end; ++iterator) {
99 ASSERT(iterator.first() > lastSeenCharacter);
100 if (iterator.first() != lastSeenCharacter + 1)
101 return false;
102
103 if (iterator.range().last == 127)
104 return true;
105 lastSeenCharacter = iterator.last();
106 }
107 return false;
108}

110uint32_t DFANode::bestFallbackTarget(const DFA& dfa) const
111{
112 ASSERT(canUseFallbackTransition(dfa));
113
114 HashMap<uint32_t, unsigned, DefaultHash<uint32_t>, WTF::UnsignedWithZeroKeyHashTraits<uint32_t>> histogram;
115
116 IterableConstRange iterableTransitions = transitions(dfa);
117 auto iterator = iterableTransitions.begin();
118 auto end = iterableTransitions.end();
119 ASSERT_WITH_MESSAGE(iterator != end, "An empty range list cannot use a fallback transition.");
120
121 if (!iterator.first() && !iterator.last())
122 ++iterator;
123 ASSERT_WITH_MESSAGE(iterator != end, "An empty range list matching only zero cannot use a fallback transition.");
124
125 uint32_t bestTarget = iterator.target();
126 unsigned bestTargetScore = !iterator.range().first ? iterator.range().size() - 1 : iterator.range().size();
127 histogram.add(bestTarget, bestTargetScore);
128 ++iterator;
129
130 for (;iterator != end; ++iterator) {
131 unsigned rangeSize = iterator.range().size();
132 uint32_t target = iterator.target();
133 auto addResult = histogram.add(target, rangeSize);
134 if (!addResult.isNewEntry)
135 addResult.iterator->value += rangeSize;
136 if (addResult.iterator->value > bestTargetScore) {
137 bestTargetScore = addResult.iterator->value;
138 bestTarget = target;
139 }
140 }
141 return bestTarget;
142}
