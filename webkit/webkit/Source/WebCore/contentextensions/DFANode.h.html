37struct DFA;
38
39struct CharRange {
40 signed char first;
41 signed char last;
42 unsigned size() const { return last - first + 1; }
43};
44
45// A DFANode abstract the transition table out of a DFA state. If a state is accepting, the DFANode also have
46// the actions for that state.
47class DFANode {
48public:
49 struct ConstRangeIterator {
50 const DFA& dfa;
51 uint32_t position;
52
53 const ConstRangeIterator& operator*() const { return *this; }
54
55 bool operator==(const ConstRangeIterator& other) const
56 {
57 ASSERT(&dfa == &other.dfa);
58 return position == other.position;
59 }
60 bool operator!=(const ConstRangeIterator& other) const { return !(*this == other); }
61
62 ConstRangeIterator& operator++()
63 {
64 ++position;
65 return *this;
66 }
67
68 const CharRange& range() const;
69 uint32_t target() const;
70
71 char first() const { return range().first; }
72 char last() const { return range().last; }
73 uint32_t data() const { return target(); }
74 };
75
76 struct IterableConstRange {
77 const DFA& dfa;
78 uint32_t rangesStart;
79 uint32_t rangesEnd;
80
81 ConstRangeIterator begin() const { return { dfa, rangesStart }; }
82 ConstRangeIterator end() const { return { dfa, rangesEnd }; }
83 };
84
85 IterableConstRange transitions(const DFA& dfa) const
86 {
87 return IterableConstRange { dfa, m_transitionsStart, m_transitionsStart + m_transitionsLength };
88 }
89
90 struct RangeIterator {
91 DFA& dfa;
92 uint32_t position;
93
94 RangeIterator& operator*() { return *this; }
95
96 bool operator==(const RangeIterator& other) const
97 {
98 ASSERT(&dfa == &other.dfa);
99 return position == other.position;
100 }
101 bool operator!=(const RangeIterator& other) const { return !(*this == other); }
102
103 RangeIterator& operator++()
104 {
105 ++position;
106 return *this;
107 }
108
109 const CharRange& range() const;
110 uint32_t target() const;
111 void resetTarget(uint32_t);
112
113 char first() const { return range().first; }
114 char last() const { return range().last; }
115 uint32_t data() const { return target(); }
116 };
117
118 struct IterableRange {
119 DFA& dfa;
120 uint32_t rangesStart;
121 uint32_t rangesEnd;
122
123 RangeIterator begin() const { return { dfa, rangesStart }; }
124 RangeIterator end() const { return { dfa, rangesEnd }; }
125 };
126
127 IterableRange transitions(DFA& dfa)
128 {
129 return IterableRange { dfa, m_transitionsStart, m_transitionsStart + m_transitionsLength };
130 }
131
132 // FIXME: Stop minimizing killed nodes and add ASSERT(!isKilled()) in many functions here.
133 void kill(DFA&);
134 Vector<uint64_t> actions(const DFA&) const;
135 bool containsTransition(char, const DFA&) const;
136
137 bool isKilled() const { return m_flags & IsKilled; }
138 bool hasActions() const { return !!m_actionsLength; }
139 uint16_t actionsLength() const { return m_actionsLength; }
140 uint32_t actionsStart() const { return m_actionsStart; }
141
142 bool canUseFallbackTransition(const DFA&) const;
143 uint32_t bestFallbackTarget(const DFA&) const;
144
145 void setActions(uint32_t start, uint16_t length)
146 {
147 ASSERT(!m_actionsLength);
148 m_actionsStart = start;
149 m_actionsLength = length;
150 }
151 void setTransitions(uint32_t start, uint16_t length)
152 {
153 ASSERT(!m_transitionsStart);
154 ASSERT(!m_transitionsLength);
155 m_transitionsStart = start;
156 m_transitionsLength = length;
157 }
158
159private:
160 uint32_t m_actionsStart { 0 };
161 uint32_t m_transitionsStart { 0 };
162 uint16_t m_actionsLength { 0 };
163 uint8_t m_transitionsLength { 0 };
164
165 uint8_t m_flags { 0 };
166 const uint8_t IsKilled = 0x01;
167};
168
169COMPILE_ASSERT(sizeof(DFANode) <= 16, Keep the DFANodes small);
