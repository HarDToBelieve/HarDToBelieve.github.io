26#pragma once
27
28#if ENABLE(CONTENT_EXTENSIONS)
29
30#include "ContentExtensionsDebugging.h"
31#include <wtf/Vector.h>
32
33namespace WebCore {
34
35namespace ContentExtensions {
36
37template <typename CharacterType>
38struct ImmutableRange {
39 uint32_t targetStart;
40 uint32_t targetEnd;
41 CharacterType first;
42 CharacterType last;
43};
44
45struct ImmutableNFANode {
46 uint32_t rangesStart { 0 };
47 uint32_t rangesEnd { 0 };
48 uint32_t epsilonTransitionTargetsStart { 0 };
49 uint32_t epsilonTransitionTargetsEnd { 0 };
50 uint32_t actionStart { 0 };
51 uint32_t actionEnd { 0 };
52};
53
54template <typename CharacterType, typename ActionType>
55struct ImmutableNFA {
56 Vector<ImmutableNFANode, 0, ContentExtensionsOverflowHandler> nodes;
57 Vector<ImmutableRange<CharacterType>, 0, ContentExtensionsOverflowHandler> transitions;
58 Vector<uint32_t, 0, ContentExtensionsOverflowHandler> targets;
59 Vector<uint32_t, 0, ContentExtensionsOverflowHandler> epsilonTransitionsTargets;
60 Vector<ActionType, 0, ContentExtensionsOverflowHandler> actions;
61
62 void clear()
63 {
64 nodes.clear();
65 transitions.clear();
66 targets.clear();
67 epsilonTransitionsTargets.clear();
68 actions.clear();
69 }
70
71 size_t memoryUsed() const
72 {
73 return nodes.capacity() * sizeof(ImmutableNFANode)
74 + transitions.capacity() * sizeof(ImmutableRange<CharacterType>)
75 + targets.capacity() * sizeof(uint32_t)
76 + epsilonTransitionsTargets.capacity() * sizeof(uint32_t)
77 + actions.capacity() * sizeof(ActionType);
78 }
79};
80
81} // namespace ContentExtensions
82} // namespace WebCore
83
84#endif // ENABLE(CONTENT_EXTENSIONS)
