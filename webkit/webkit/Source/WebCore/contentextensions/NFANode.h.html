35namespace WebCore {
36
37namespace ContentExtensions {
38
39// A NFANode abstract the transition table out of a NFA state.
40
41typedef Vector<uint64_t, 0, WTF::CrashOnOverflow, 1> ActionList;
42typedef HashSet<unsigned, DefaultHash<unsigned>, WTF::UnsignedWithZeroKeyHashTraits<unsigned>> NFANodeIndexSet;
43typedef HashMap<uint16_t, NFANodeIndexSet, DefaultHash<uint16_t>, WTF::UnsignedWithZeroKeyHashTraits<uint16_t>> NFANodeTransitions;
44
45class NFANode {
46public:
47 NFANodeTransitions transitions;
48 NFANodeIndexSet transitionsOnAnyCharacter;
49
50 ActionList finalRuleIds;
51};

53} // namespace ContentExtensions
54} // namespace WebCore

56#endif // ENABLE(CONTENT_EXTENSIONS)
