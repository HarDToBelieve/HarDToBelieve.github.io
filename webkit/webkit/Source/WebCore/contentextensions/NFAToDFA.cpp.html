54static inline void epsilonClosureExcludingSelf(const SerializedNFA& nfa, unsigned nodeId, UniqueNodeList& output)
55{
56 NodeIdSet closure({ nodeId });
57 Vector<unsigned, 64, ContentExtensionsOverflowHandler> unprocessedNodes({ nodeId });
58
59 do {
60 unsigned unprocessedNodeId = unprocessedNodes.takeLast();
61 const auto* node = nfa.nodes().pointerAt(unprocessedNodeId);
62
63 for (uint32_t epsilonTargetIndex = node->epsilonTransitionTargetsStart; epsilonTargetIndex < node->epsilonTransitionTargetsEnd; ++epsilonTargetIndex) {
64 uint32_t targetNodeId = nfa.epsilonTransitionsTargets().valueAt(epsilonTargetIndex);
65 auto addResult = closure.add(targetNodeId);
66 if (addResult.isNewEntry) {
67 unprocessedNodes.append(targetNodeId);
68 output.append(targetNodeId);
69 }
70 }
71 } while (!unprocessedNodes.isEmpty());
72
73 output.shrinkToFit();
74}

76static NFANodeClosures resolveEpsilonClosures(const SerializedNFA& nfa)
77{
78 NFANodeClosures nfaNodeClosures;
79 unsigned nfaGraphSize = nfa.nodes().size();
80 nfaNodeClosures.resize(nfaGraphSize);
81
82 for (unsigned nodeId = 0; nodeId < nfaGraphSize; ++nodeId)
83 epsilonClosureExcludingSelf(nfa, nodeId, nfaNodeClosures[nodeId]);
84
85 return nfaNodeClosures;
86}

88static ALWAYS_INLINE void extendSetWithClosure(const NFANodeClosures& nfaNodeClosures, unsigned nodeId, NodeIdSet& set)
89{
90 ASSERT(set.contains(nodeId));
91 const UniqueNodeList& nodeClosure = nfaNodeClosures[nodeId];
92 if (!nodeClosure.isEmpty())
93 set.add(nodeClosure.begin(), nodeClosure.end());
94}

116class UniqueNodeIdSet {
117public:
118 UniqueNodeIdSet() { }
119 enum EmptyValueTag { EmptyValue };
120 enum DeletedValueTag { DeletedValue };
121
122 UniqueNodeIdSet(EmptyValueTag) { }
123 UniqueNodeIdSet(DeletedValueTag)
124 : m_uniqueNodeIdSetBuffer(reinterpret_cast<UniqueNodeIdSetImpl*>(-1))
125 {
126 }
127
128 UniqueNodeIdSet(const NodeIdSet& nodeIdSet, unsigned hash, unsigned dfaNodeId)
129 {
130 ASSERT(nodeIdSet.size());
131
132 unsigned size = nodeIdSet.size();
133 size_t byteSize = sizeof(UniqueNodeIdSetImpl) + (size - 1) * sizeof(unsigned);
134 m_uniqueNodeIdSetBuffer = static_cast<UniqueNodeIdSetImpl*>(fastMalloc(byteSize));
135
136 m_uniqueNodeIdSetBuffer->m_size = size;
137 m_uniqueNodeIdSetBuffer->m_hash = hash;
138 m_uniqueNodeIdSetBuffer->m_dfaNodeId = dfaNodeId;
139
140 unsigned* buffer = m_uniqueNodeIdSetBuffer->buffer();
141 for (unsigned nodeId : nodeIdSet) {
142 *buffer = nodeId;
143 ++buffer;
144 }
145 }
146
147 UniqueNodeIdSet(UniqueNodeIdSet&& other)
148 : m_uniqueNodeIdSetBuffer(other.m_uniqueNodeIdSetBuffer)
149 {
150 other.m_uniqueNodeIdSetBuffer = nullptr;
151 }
152
153 UniqueNodeIdSet& operator=(UniqueNodeIdSet&& other)
154 {
155 m_uniqueNodeIdSetBuffer = other.m_uniqueNodeIdSetBuffer;
156 other.m_uniqueNodeIdSetBuffer = nullptr;
157 return *this;
158 }
159
160 ~UniqueNodeIdSet()
161 {
162 fastFree(m_uniqueNodeIdSetBuffer);
163 }
164
165 bool operator==(const UniqueNodeIdSet& other) const
166 {
167 return m_uniqueNodeIdSetBuffer == other.m_uniqueNodeIdSetBuffer;
168 }
169
170 bool operator==(const NodeIdSet& other) const
171 {
172 if (m_uniqueNodeIdSetBuffer->m_size != static_cast<unsigned>(other.size()))
173 return false;
174 unsigned* buffer = m_uniqueNodeIdSetBuffer->buffer();
175 for (unsigned i = 0; i < m_uniqueNodeIdSetBuffer->m_size; ++i) {
176 if (!other.contains(buffer[i]))
177 return false;
178 }
179 return true;
180 }
181
182 UniqueNodeIdSetImpl* impl() const { return m_uniqueNodeIdSetBuffer; }
183
184 unsigned hash() const { return m_uniqueNodeIdSetBuffer->m_hash; }
185 bool isEmptyValue() const { return !m_uniqueNodeIdSetBuffer; }
186 bool isDeletedValue() const { return m_uniqueNodeIdSetBuffer == reinterpret_cast<UniqueNodeIdSetImpl*>(-1); }
187
188private:
189 UniqueNodeIdSetImpl* m_uniqueNodeIdSetBuffer = nullptr;
190};

215struct NodeIdSetToUniqueNodeIdSetSource {
216 NodeIdSetToUniqueNodeIdSetSource(DFA& dfa, const SerializedNFA& nfa, const NodeIdSet& nodeIdSet)
217 : dfa(dfa)
218 , nfa(nfa)
219 , nodeIdSet(nodeIdSet)
220 {
221 // The hashing operation must be independant of the nodeId.
222 unsigned hash = 4207445155;
223 for (unsigned nodeId : nodeIdSet)
224 hash += nodeId;
225 this->hash = DefaultHash<unsigned>::hash(hash);
226 }
227 DFA& dfa;
228 const SerializedNFA& nfa;
229 const NodeIdSet& nodeIdSet;
230 unsigned hash;
231};

244 static void translate(UniqueNodeIdSet& location, const NodeIdSetToUniqueNodeIdSetSource& source, unsigned hash)
245 {
246 DFANode newDFANode;
247
248 HashSet<uint64_t, DefaultHash<uint64_t>, WTF::UnsignedWithZeroKeyHashTraits<uint64_t>> actions;
249
250 for (unsigned nfaNodeId : source.nodeIdSet) {
251 const auto* nfaNode = source.nfa.nodes().pointerAt(nfaNodeId);
252 for (unsigned actionIndex = nfaNode->actionStart; actionIndex < nfaNode->actionEnd; ++actionIndex)
253 actions.add(source.nfa.actions().valueAt(actionIndex));
254 }
255
256 unsigned actionsStart = source.dfa.actions.size();
257 for (uint64_t action : actions)
258 source.dfa.actions.append(action);
259 unsigned actionsEnd = source.dfa.actions.size();
260 unsigned actionsLength = actionsEnd - actionsStart;
261 RELEASE_ASSERT_WITH_MESSAGE(actionsLength <= std::numeric_limits<uint16_t>::max(), "Too many actions for the current DFANode size.");
262 newDFANode.setActions(actionsStart, static_cast<uint16_t>(actionsLength));
263
264 unsigned dfaNodeId = source.dfa.nodes.size();
265 source.dfa.nodes.append(newDFANode);
266 new (NotNull, &location) UniqueNodeIdSet(source.nodeIdSet, hash, dfaNodeId);
267
268 ASSERT(location.impl());
269 }

272struct DataConverterWithEpsilonClosure {
273 const NFANodeClosures& nfaNodeclosures;
274
275 template<typename Iterable>
276 NFANodeIndexSet convert(const Iterable& iterable)
277 {
278 NFANodeIndexSet result;
279 for (unsigned nodeId : iterable) {
280 result.add(nodeId);
281 const UniqueNodeList& nodeClosure = nfaNodeclosures[nodeId];
282 result.add(nodeClosure.begin(), nodeClosure.end());
283 }
284 return result;
285 }
286
287 template<typename Iterable>
288 void extend(NFANodeIndexSet& destination, const Iterable& iterable)
289 {
290 for (unsigned nodeId : iterable) {
291 auto addResult = destination.add(nodeId);
292 if (addResult.isNewEntry) {
293 const UniqueNodeList& nodeClosure = nfaNodeclosures[nodeId];
294 destination.add(nodeClosure.begin(), nodeClosure.end());
295 }
296 }
297 }

300static inline void createCombinedTransition(PreallocatedNFANodeRangeList& combinedRangeList, const UniqueNodeIdSetImpl& sourceNodeSet, const SerializedNFA& serializedNFA, const NFANodeClosures& nfaNodeclosures)
301{
302 combinedRangeList.clear();
303
304 const unsigned* buffer = sourceNodeSet.buffer();
305
306 DataConverterWithEpsilonClosure converter { nfaNodeclosures };
307 for (unsigned i = 0; i < sourceNodeSet.m_size; ++i) {
308 unsigned nodeId = buffer[i];
309 auto transitions = serializedNFA.transitionsForNode(nodeId);
310 combinedRangeList.extend(transitions.begin(), transitions.end(), converter);
311 }
312}

314static ALWAYS_INLINE unsigned getOrCreateDFANode(const NodeIdSet& nfaNodeSet, const SerializedNFA& serializedNFA, DFA& dfa, UniqueNodeIdSetTable& uniqueNodeIdSetTable, UniqueNodeQueue& unprocessedNodes)
315{
316 NodeIdSetToUniqueNodeIdSetSource nodeIdSetToUniqueNodeIdSetSource(dfa, serializedNFA, nfaNodeSet);
317 auto uniqueNodeIdAddResult = uniqueNodeIdSetTable.add<NodeIdSetToUniqueNodeIdSetTranslator>(nodeIdSetToUniqueNodeIdSetSource);
318 if (uniqueNodeIdAddResult.isNewEntry)
319 unprocessedNodes.append(uniqueNodeIdAddResult.iterator->impl());
320
321 return uniqueNodeIdAddResult.iterator->impl()->m_dfaNodeId;
322}

324std::optional<DFA> NFAToDFA::convert(NFA&& nfa)
325{
326 auto serializedNFA = SerializedNFA::serialize(WTFMove(nfa));
327 if (!serializedNFA)
328 return std::nullopt;
329
330 NFANodeClosures nfaNodeClosures = resolveEpsilonClosures(*serializedNFA);
331 DFA dfa;
332
333 NodeIdSet initialSet({ serializedNFA->root() });
334 extendSetWithClosure(nfaNodeClosures, serializedNFA->root(), initialSet);
335
336 UniqueNodeIdSetTable uniqueNodeIdSetTable;
337
338 NodeIdSetToUniqueNodeIdSetSource initialNodeIdSetToUniqueNodeIdSetSource(dfa, *serializedNFA, initialSet);
339 auto addResult = uniqueNodeIdSetTable.add<NodeIdSetToUniqueNodeIdSetTranslator>(initialNodeIdSetToUniqueNodeIdSetSource);
340
341 UniqueNodeQueue unprocessedNodes;
342 unprocessedNodes.append(addResult.iterator->impl());
343
344 PreallocatedNFANodeRangeList combinedRangeList;
345 do {
346 UniqueNodeIdSetImpl* uniqueNodeIdSetImpl = unprocessedNodes.takeLast();
347 createCombinedTransition(combinedRangeList, *uniqueNodeIdSetImpl, *serializedNFA, nfaNodeClosures);
348
349 unsigned transitionsStart = dfa.transitionRanges.size();
350 for (const NFANodeRange& range : combinedRangeList) {
351 unsigned targetNodeId = getOrCreateDFANode(range.data, *serializedNFA, dfa, uniqueNodeIdSetTable, unprocessedNodes);
352 dfa.transitionRanges.append({ range.first, range.last });
353 dfa.transitionDestinations.append(targetNodeId);
354 }
355 unsigned transitionsEnd = dfa.transitionRanges.size();
356 unsigned transitionsLength = transitionsEnd - transitionsStart;
357
358 unsigned dfaNodeId = uniqueNodeIdSetImpl->m_dfaNodeId;
359 DFANode& dfaSourceNode = dfa.nodes[dfaNodeId];
360 dfaSourceNode.setTransitions(transitionsStart, static_cast<uint8_t>(transitionsLength));
361 } while (!unprocessedNodes.isEmpty());
362
363 return dfa;
364}
