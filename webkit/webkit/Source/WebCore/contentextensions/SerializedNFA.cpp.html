36template<typename T>
37bool writeAllToFile(FileSystem::PlatformFileHandle file, const T& container)
38{
39 auto bytes = reinterpret_cast<const uint8_t*>(container.data());
40 size_t bytesLength = container.size() * sizeof(container[0]);
41 auto end = bytes + bytesLength;
42 while (bytes < end) {
43 auto written = FileSystem::writeToFile(file, bytes, bytesLength);
44 if (written == -1)
45 return false;
46 bytes += written;
47 bytesLength -= written;
48 }
49 return true;
50}

52std::optional<SerializedNFA> SerializedNFA::serialize(NFA&& nfa)
53{
54 auto file = FileSystem::invalidPlatformFileHandle;
55 auto filename = FileSystem::openTemporaryFile("SerializedNFA", file);
56 if (!FileSystem::isHandleValid(file))
57 return std::nullopt;
58
59 bool wroteSuccessfully = writeAllToFile(file, nfa.nodes)
60 && writeAllToFile(file, nfa.transitions)
61 && writeAllToFile(file, nfa.targets)
62 && writeAllToFile(file, nfa.epsilonTransitionsTargets)
63 && writeAllToFile(file, nfa.actions);
64 if (!wroteSuccessfully) {
65 FileSystem::closeFile(file);
66 FileSystem::deleteFile(filename);
67 return std::nullopt;
68 }
69
70 bool mappedSuccessfully = false;
71 FileSystem::MappedFileData mappedFile(file, FileSystem::MappedFileMode::Private, mappedSuccessfully);
72 FileSystem::closeFile(file);
73 FileSystem::deleteFile(filename);
74 if (!mappedSuccessfully)
75 return std::nullopt;
76
77 Metadata metadata {
78 nfa.nodes.size(),
79 nfa.transitions.size(),
80 nfa.targets.size(),
81 nfa.epsilonTransitionsTargets.size(),
82 nfa.actions.size(),
83 0,
84 nfa.nodes.size() * sizeof(nfa.nodes[0]),
85 nfa.nodes.size() * sizeof(nfa.nodes[0])
86 + nfa.transitions.size() * sizeof(nfa.transitions[0]),
87 nfa.nodes.size() * sizeof(nfa.nodes[0])
88 + nfa.transitions.size() * sizeof(nfa.transitions[0])
89 + nfa.targets.size() * sizeof(nfa.targets[0]),
90 nfa.nodes.size() * sizeof(nfa.nodes[0])
91 + nfa.transitions.size() * sizeof(nfa.transitions[0])
92 + nfa.targets.size() * sizeof(nfa.targets[0])
93 + nfa.epsilonTransitionsTargets.size() * sizeof(nfa.epsilonTransitionsTargets[0])
94 };
95
96 nfa.clear();
97
98 return {{ WTFMove(mappedFile), WTFMove(metadata) }};
99}

107template<typename T>
108const T* SerializedNFA::pointerAtOffsetInFile(size_t offset) const
109{
110 return reinterpret_cast<const T*>(static_cast<const uint8_t*>(m_file.data()) + offset);
111}
112
113auto SerializedNFA::nodes() const -> const Range<ImmutableNFANode>
114{
115 return { pointerAtOffsetInFile<ImmutableNFANode>(m_metadata.nodesOffset), m_metadata.nodesSize };
116}
117
118auto SerializedNFA::transitions() const -> const Range<ImmutableRange<char>>
119{
120 return { pointerAtOffsetInFile<ImmutableRange<char>>(m_metadata.transitionsOffset), m_metadata.transitionsSize };
121}
122
123auto SerializedNFA::targets() const -> const Range<uint32_t>
124{

126}
127
128auto SerializedNFA::epsilonTransitionsTargets() const -> const Range<uint32_t>

131}

133auto SerializedNFA::actions() const -> const Range<uint64_t>
