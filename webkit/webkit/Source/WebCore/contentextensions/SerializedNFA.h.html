43 template<typename T>
44 class Range {
45 public:
46 Range(const T* begin, size_t size)
47 : m_begin(begin)
48 , m_size(size) { }
49 const T* begin() const { return m_begin; }
50 const T* end() const { return m_begin + m_size; }
51 size_t size() const { return m_size; }
52 const T* pointerAt(size_t i) const
53 {
54 RELEASE_ASSERT(i < m_size);
55 return begin() + i;
56 }
57 T valueAt(size_t i) const
58 {
59 RELEASE_ASSERT(i < m_size);
60 T value;
61 memcpy(&value, begin() + i, sizeof(T));
62 return value;
63 }
64 private:
65 const T* m_begin { nullptr };
66 size_t m_size { 0 };
67 };

73 const Range<uint64_t> actions() const;

76 {
77 RELEASE_ASSERT(nodes().size());
78 return 0;

81 struct ConstTargetIterator {
82 const SerializedNFA& serializedNFA;
83 uint32_t position;
84
85 uint32_t operator*() const { return serializedNFA.targets().valueAt(position); }
86 const uint32_t* operator->() const { return serializedNFA.targets().pointerAt(position); }
87
88 bool operator==(const ConstTargetIterator& other) const
89 {
90 ASSERT(&serializedNFA == &other.serializedNFA);
91 return position == other.position;
92 }
93 bool operator!=(const ConstTargetIterator& other) const { return !(*this == other); }
94
95 ConstTargetIterator& operator++()
96 {
97 ++position;
98 return *this;
99 }
100 };
101
102 struct IterableConstTargets {
103 const SerializedNFA& serializedNFA;
104 uint32_t targetStart;
105 uint32_t targetEnd;
106
107 ConstTargetIterator begin() const { return { serializedNFA, targetStart }; }
108 ConstTargetIterator end() const { return { serializedNFA, targetEnd }; }
109 };
110
111 struct ConstRangeIterator {
112 const SerializedNFA& serializedNFA;
113 uint32_t position;
114
115 bool operator==(const ConstRangeIterator& other) const
116 {
117 ASSERT(&serializedNFA == &other.serializedNFA);
118 return position == other.position;
119 }
120 bool operator!=(const ConstRangeIterator& other) const { return !(*this == other); }
121
122 ConstRangeIterator& operator++()
123 {
124 ++position;
125 return *this;
126 }
127
128 char first() const
129 {
130 return range()->first;
131 }
132
133 char last() const
134 {
135 return range()->last;
136 }
137
138 IterableConstTargets data() const
139 {
140 const ImmutableRange<char>* range = this->range();
141 return { serializedNFA, range->targetStart, range->targetEnd };
142 };
143
144 private:
145 const ImmutableRange<char>* range() const
146 {
147 return serializedNFA.transitions().pointerAt(position);
148 }
149 };
150
151 struct IterableConstRange {
152 const SerializedNFA& serializedNFA;
153 uint32_t rangesStart;
154 uint32_t rangesEnd;
155
156 ConstRangeIterator begin() const { return { serializedNFA, rangesStart }; }
157 ConstRangeIterator end() const { return { serializedNFA, rangesEnd }; }
158
159#if CONTENT_EXTENSIONS_STATE_MACHINE_DEBUGGING
160 void debugPrint() const
161 {
162 for (const auto& range : *this)
163 WTFLogAlways(" %d-%d", range.first, range.last);
164 }
165#endif
166 };
167
168 IterableConstRange transitionsForNode(uint32_t nodeId) const
169 {
170 const auto* node = nodes().pointerAt(nodeId);
171 return { *this, node->rangesStart, node->rangesEnd };
172 }
