134 class CharacterSet {
135 public:
136 void set(UChar character)
137 {
138 RELEASE_ASSERT(character < 128);
139 m_characters[character / 64] |= (uint64_t(1) << (character % 64));
140 }
141
142 bool get(UChar character) const
143 {
144 RELEASE_ASSERT(character < 128);
145 return m_characters[character / 64] & (uint64_t(1) << (character % 64));
146 }
147
148 void invert()
149 {
150 ASSERT(!m_inverted);
151 m_inverted = true;
152 }
153
154 bool inverted() const { return m_inverted; }
155
156 unsigned bitCount() const
157 {
158 return WTF::bitCount(m_characters[0]) + WTF::bitCount(m_characters[1]);
159 }
160
161 bool operator==(const CharacterSet& other) const
162 {
163 return other.m_inverted == m_inverted
164 && other.m_characters[0] == m_characters[0]
165 && other.m_characters[1] == m_characters[1];
166 }
167
168 unsigned hash() const
169 {
170 return computeHash(m_inverted, m_characters);
171 }
172
173 private:
174 bool m_inverted { false };
175 std::array<uint64_t, 2> m_characters { 0, 0 };
176 };
177
178 struct Group {
179 Vector<Term> terms;
180
181 bool operator==(const Group& other) const
182 {
183 return other.terms == terms;
184 }
185
186 unsigned hash() const
187 {
188 unsigned hash = 6421749;
189 for (const Term& term : terms) {
190 unsigned termHash = term.hash();
191 hash = (hash << 16) ^ ((termHash << 11) ^ hash);
192 hash += hash >> 11;
193 }
194 return hash;
195 }
196 };

199 AtomData()

202 }
203 ~AtomData()
204 {
205 }
206
207 char invalidTerm;
208 CharacterSet characterSet;
209 Group group;
210 } m_atomData;
211};

214inline String quantifierToString(AtomQuantifier quantifier)
215{
216 switch (quantifier) {
217 case AtomQuantifier::One:
218 return "";
219 case AtomQuantifier::ZeroOrOne:
220 return "?";
221 case AtomQuantifier::ZeroOrMore:
222 return "*";
223 case AtomQuantifier::OneOrMore:
224 return "+";
225 }
226}

228inline String Term::toString() const
229{
230 switch (m_termType) {
231 case TermType::Empty:
232 return "(Empty)";
233 case TermType::CharacterSet: {
234 StringBuilder builder;
235 builder.append('[');
236 for (UChar c = 0; c < 128; c++) {
237 if (m_atomData.characterSet.get(c)) {
238 if (isASCIIPrintable(c) && !isASCIISpace(c))
239 builder.append(c);
240 else
241 builder.append("\\u", c);
242 }
243 }
244 builder.append(']');
245 builder.append(quantifierToString(m_quantifier));
246 return builder.toString();
247 }
248 case TermType::Group: {
249 StringBuilder builder;
250 builder.append('(');
251 for (const Term& term : m_atomData.group.terms)
252 builder.append(term.toString());
253 builder.append(')');
254 builder.append(quantifierToString(m_quantifier));
255 return builder.toString();
256 }
257 }
258}

348inline void Term::addCharacter(UChar character, bool isCaseSensitive)
349{
350 ASSERT(isASCII(character));
351
352 ASSERT_WITH_SECURITY_IMPLICATION(m_termType == TermType::CharacterSet);
353 if (m_termType != TermType::CharacterSet)
354 return;
355
356 if (isCaseSensitive || !isASCIIAlpha(character))
357 m_atomData.characterSet.set(character);
358 else {
359 m_atomData.characterSet.set(toASCIIUpper(character));
360 m_atomData.characterSet.set(toASCIILower(character));
361 }
362}

369 m_atomData.group.terms.append(term);

386inline void Term::generateGraph(NFA& nfa, ImmutableCharNFANodeBuilder& source, uint32_t destination) const
387{
388 ASSERT(isValid());
389
390 switch (m_quantifier) {
391 case AtomQuantifier::One: {
392 generateSubgraphForAtom(nfa, source, destination);
393 break;
394 }
395 case AtomQuantifier::ZeroOrOne: {
396 generateSubgraphForAtom(nfa, source, destination);
397 source.addEpsilonTransition(destination);
398 break;
399 }
400 case AtomQuantifier::ZeroOrMore: {
401 ImmutableCharNFANodeBuilder repeatStart(nfa);
402 source.addEpsilonTransition(repeatStart);
403
404 ImmutableCharNFANodeBuilder repeatEnd(nfa);
405 generateSubgraphForAtom(nfa, repeatStart, repeatEnd);
406 repeatEnd.addEpsilonTransition(repeatStart);
407
408 repeatEnd.addEpsilonTransition(destination);
409 source.addEpsilonTransition(destination);
410 break;
411 }
412 case AtomQuantifier::OneOrMore: {
413 ImmutableCharNFANodeBuilder repeatStart(nfa);
414 source.addEpsilonTransition(repeatStart);
415
416 ImmutableCharNFANodeBuilder repeatEnd(nfa);
417 generateSubgraphForAtom(nfa, repeatStart, repeatEnd);
418 repeatEnd.addEpsilonTransition(repeatStart);
419
420 repeatEnd.addEpsilonTransition(destination);
421 break;
422 }
423 }

431inline bool Term::matchesAtLeastOneCharacter() const
432{
433 ASSERT(isValid());
434
435 if (m_quantifier == AtomQuantifier::ZeroOrOne || m_quantifier == AtomQuantifier::ZeroOrMore)
436 return false;
437 if (isEndOfLineAssertion())
438 return false;
439
440 if (m_termType == TermType::Group) {
441 for (const Term& term : m_atomData.group.terms) {
442 if (term.matchesAtLeastOneCharacter())
443 return true;
444 }
445 return false;
446 }
447 return true;
448}

450inline bool Term::isKnownToMatchAnyString() const
451{
452 ASSERT(isValid());
453
454 switch (m_termType) {
455 case TermType::Empty:
456 ASSERT_NOT_REACHED();
457 break;
458 case TermType::CharacterSet:
459 // ".*" is the only simple term matching any string.
460 return isUniversalTransition() && m_quantifier == AtomQuantifier::ZeroOrMore;
461 break;
462 case TermType::Group: {
463 // There are infinitely many ways to match anything with groups, we just handle simple cases
464 if (m_atomData.group.terms.size() != 1)
465 return false;
466
467 const Term& firstTermInGroup = m_atomData.group.terms.first();
468 // -(.*) with any quantifier.
469 if (firstTermInGroup.isKnownToMatchAnyString())
470 return true;
471
472 if (firstTermInGroup.isUniversalTransition()) {
473 // -(.)*, (.+)*, (.?)* etc.
474 if (m_quantifier == AtomQuantifier::ZeroOrMore)
475 return true;
476
477 // -(.+)?.
478 if (m_quantifier == AtomQuantifier::ZeroOrOne && firstTermInGroup.m_quantifier == AtomQuantifier::OneOrMore)
479 return true;
480
481 // -(.?)+.
482 if (m_quantifier == AtomQuantifier::OneOrMore && firstTermInGroup.m_quantifier == AtomQuantifier::ZeroOrOne)
483 return true;
484 }
485 break;
486 }
487 }
488 return false;
489}

491inline bool Term::hasFixedLength() const
492{
493 ASSERT(isValid());
494
495 switch (m_termType) {
496 case TermType::Empty:
497 ASSERT_NOT_REACHED();
498 break;
499 case TermType::CharacterSet:
500 return m_quantifier == AtomQuantifier::One;
501 case TermType::Group: {
502 if (m_quantifier != AtomQuantifier::One)
503 return false;
504 for (const Term& term : m_atomData.group.terms) {
505 if (!term.hasFixedLength())
506 return false;
507 }
508 return true;
509 }
510 }
511 return false;
512}

528inline bool Term::operator==(const Term& other) const
529{
530 if (other.m_termType != m_termType || other.m_quantifier != m_quantifier)
531 return false;
532
533 switch (m_termType) {
534 case TermType::Empty:
535 return true;
536 case TermType::CharacterSet:
537 return m_atomData.characterSet == other.m_atomData.characterSet;
538 case TermType::Group:
539 return m_atomData.group == other.m_atomData.group;
540 }
541 ASSERT_NOT_REACHED();
542 return false;
543}

545inline unsigned Term::hash() const
546{
547 unsigned primary = static_cast<unsigned>(m_termType) << 16 | static_cast<unsigned>(m_quantifier);
548 unsigned secondary = 0;
549 switch (m_termType) {
550 case TermType::Empty:
551 secondary = 52184393;
552 break;
553 case TermType::CharacterSet:
554 secondary = m_atomData.characterSet.hash();
555 break;
556 case TermType::Group:
557 secondary = m_atomData.group.hash();
558 break;
559 }
560 return WTF::pairIntHash(primary, secondary);
561}

568inline bool Term::isUniversalTransition() const
569{
570 ASSERT(isValid());
571
572 switch (m_termType) {
573 case TermType::Empty:
574 ASSERT_NOT_REACHED();
575 break;
576 case TermType::CharacterSet:
577 return (m_atomData.characterSet.inverted() && !m_atomData.characterSet.bitCount())
578 || (!m_atomData.characterSet.inverted() && m_atomData.characterSet.bitCount() == 127 && !m_atomData.characterSet.get(0));
579 case TermType::Group:
580 return m_atomData.group.terms.size() == 1 && m_atomData.group.terms.first().isUniversalTransition();
581 }
582 return false;
583}

590inline void Term::generateSubgraphForAtom(NFA& nfa, ImmutableCharNFANodeBuilder& source, uint32_t destination) const
591{
592 switch (m_termType) {
593 case TermType::Empty:
594 ASSERT_NOT_REACHED();
595 source.addEpsilonTransition(destination);
596 break;
597 case TermType::CharacterSet: {
598 if (!m_atomData.characterSet.inverted()) {
599 UChar i = 0;
600 while (true) {
601 while (i < 128 && !m_atomData.characterSet.get(i))
602 ++i;
603 if (i == 128)
604 break;
605
606 UChar start = i;
607 ++i;
608 while (i < 128 && m_atomData.characterSet.get(i))
609 ++i;
610 source.addTransition(start, i - 1, destination);
611 }
612 } else {
613 UChar i = 1;
614 while (true) {
615 while (i < 128 && m_atomData.characterSet.get(i))
616 ++i;
617 if (i == 128)
618 break;
619
620 UChar start = i;
621 ++i;
622 while (i < 128 && !m_atomData.characterSet.get(i))
623 ++i;
624 source.addTransition(start, i - 1, destination);
625 }
626 }
627 break;
628 }
629 case TermType::Group: {
630 if (m_atomData.group.terms.isEmpty()) {
631 // FIXME: any kind of empty term could be avoided in the parser. This case should turned into an assertion.
632 source.addEpsilonTransition(destination);
633 return;
634 }
635
636 if (m_atomData.group.terms.size() == 1) {
637 m_atomData.group.terms.first().generateGraph(nfa, source, destination);
638 return;
639 }
640
641 ImmutableCharNFANodeBuilder lastTarget = m_atomData.group.terms.first().generateGraph(nfa, source, ActionList());
642 for (unsigned i = 1; i < m_atomData.group.terms.size() - 1; ++i) {
643 const Term& currentTerm = m_atomData.group.terms[i];
644 ImmutableCharNFANodeBuilder newNode = currentTerm.generateGraph(nfa, lastTarget, ActionList());
645 lastTarget = WTFMove(newNode);
646 }
647 const Term& lastTerm = m_atomData.group.terms.last();
648 lastTerm.generateGraph(nfa, lastTarget, destination);
649 break;
650 }
651 }
652}

654inline void Term::destroy()
655{
656 switch (m_termType) {
657 case TermType::Empty:
658 break;
659 case TermType::CharacterSet:
660 m_atomData.characterSet.~CharacterSet();
661 break;
662 case TermType::Group:
663 m_atomData.group.~Group();
664 break;
665 }
666 m_termType = TermType::Empty;
667}
