27
28#include "CryptoAlgorithmIdentifier.h"
29#include "CryptoKeyFormat.h"
30#include "CryptoKeyPair.h"
31#include "CryptoKeyUsage.h"
32#include "ExceptionOr.h"
33#include "JsonWebKey.h"
34#include <wtf/Function.h>
35#include <wtf/ThreadSafeRefCounted.h>
36#include <wtf/Variant.h>
37#include <wtf/Vector.h>
38#include <wtf/WorkQueue.h>
39
40#if ENABLE(WEB_CRYPTO)
41
42namespace WebCore {
43
44class CryptoAlgorithmParameters;
45class CryptoKey;
46class ScriptExecutionContext;
47
48using KeyData = Variant<Vector<uint8_t>, JsonWebKey>;
49using KeyOrKeyPair = Variant<RefPtr<CryptoKey>, CryptoKeyPair>;
50
51class CryptoAlgorithm : public ThreadSafeRefCounted<CryptoAlgorithm> {
52public:
53 virtual ~CryptoAlgorithm() = default;
54
55 virtual CryptoAlgorithmIdentifier identifier() const = 0;
56
57 using BoolCallback = WTF::Function<void(bool)>;
58 using KeyCallback = WTF::Function<void(CryptoKey&)>;
59 using KeyOrKeyPairCallback = WTF::Function<void(KeyOrKeyPair&&)>;
60 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=169395
61 using VectorCallback = WTF::Function<void(const Vector<uint8_t>&)>;
62 using VoidCallback = WTF::Function<void()>;
63 using ExceptionCallback = WTF::Function<void(ExceptionCode)>;
64 using KeyDataCallback = WTF::Function<void(CryptoKeyFormat, KeyData&&)>;
65
66 virtual void encrypt(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&);
67 virtual void decrypt(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&);
68 virtual void sign(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&);
69 virtual void verify(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&& signature, Vector<uint8_t>&&, BoolCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&);
70 virtual void digest(Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&);
71 virtual void generateKey(const CryptoAlgorithmParameters&, bool extractable, CryptoKeyUsageBitmap, KeyOrKeyPairCallback&&, ExceptionCallback&&, ScriptExecutionContext&);
72 virtual void deriveBits(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, size_t length, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&);
73 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=169262
74 virtual void importKey(CryptoKeyFormat, KeyData&&, const CryptoAlgorithmParameters&, bool extractable, CryptoKeyUsageBitmap, KeyCallback&&, ExceptionCallback&&);
75 virtual void exportKey(CryptoKeyFormat, Ref<CryptoKey>&&, KeyDataCallback&&, ExceptionCallback&&);
76 virtual void wrapKey(Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&);
77 virtual void unwrapKey(Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&);
78 virtual ExceptionOr<size_t> getKeyLength(const CryptoAlgorithmParameters&);
79
80 static void dispatchOperationInWorkQueue(WorkQueue&, ScriptExecutionContext&, VectorCallback&&, ExceptionCallback&&, WTF::Function<ExceptionOr<Vector<uint8_t>>()>&&);
81 static void dispatchOperationInWorkQueue(WorkQueue&, ScriptExecutionContext&, BoolCallback&&, ExceptionCallback&&, WTF::Function<ExceptionOr<bool>()>&&);
82};
83
84} // namespace WebCore
85
86#endif // ENABLE(WEB_CRYPTO)
