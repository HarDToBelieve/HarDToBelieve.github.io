35#if ENABLE(WEB_CRYPTO)
36
37namespace WebCore {
38
39class CryptoAlgorithm;
40
41class CryptoAlgorithmRegistry {
42 WTF_MAKE_NONCOPYABLE(CryptoAlgorithmRegistry);
43 friend class LazyNeverDestroyed<CryptoAlgorithmRegistry>;
44
45public:
46 static CryptoAlgorithmRegistry& singleton();
47
48 std::optional<CryptoAlgorithmIdentifier> identifier(const String&);
49 String name(CryptoAlgorithmIdentifier);
50
51 RefPtr<CryptoAlgorithm> create(CryptoAlgorithmIdentifier);
52
53private:
54 CryptoAlgorithmRegistry();
55 void platformRegisterAlgorithms();
56
57 using CryptoAlgorithmConstructor = Ref<CryptoAlgorithm> (*)();
58
59 template<typename AlgorithmClass> void registerAlgorithm()
60 {
61 registerAlgorithm(AlgorithmClass::s_name, AlgorithmClass::s_identifier, AlgorithmClass::create);
62 }
63
64 void registerAlgorithm(const String& name, CryptoAlgorithmIdentifier, CryptoAlgorithmConstructor);
65
66 Lock m_lock;
67 HashMap<String, CryptoAlgorithmIdentifier, ASCIICaseInsensitiveHash> m_identifiers WTF_GUARDED_BY_LOCK(m_lock);
68 HashMap<unsigned, std::pair<String, CryptoAlgorithmConstructor>> m_constructors WTF_GUARDED_BY_LOCK(m_lock);
69};

71} // namespace WebCore
