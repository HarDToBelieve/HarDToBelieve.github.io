33namespace WebCore {
34
35class CryptoAlgorithmAesCtrParams;
36class CryptoKeyAES;
37
38class CryptoAlgorithmAES_CTR final : public CryptoAlgorithm {
39public:
40 class CounterBlockHelper {
41 public:
42 CounterBlockHelper(const Vector<uint8_t>& counterVector, size_t counterLength);
43
44 size_t countToOverflowSaturating() const;
45 Vector<uint8_t> counterVectorAfterOverflow() const;
46
47 private:
48 // 128 bits integer with miminum required operators.
49 struct CounterBlockBits {
50 void set();
51 bool all() const;
52 bool any() const;
53
54 CounterBlockBits operator&(const CounterBlockBits&) const;
55 CounterBlockBits operator~() const;
56 CounterBlockBits& operator <<=(unsigned);
57 CounterBlockBits& operator &=(const CounterBlockBits&);
58
59 uint64_t m_hi { 0 };
60 uint64_t m_lo { 0 };
61 };
62
63 CounterBlockBits m_bits;
64 const size_t m_counterLength;
65 };
66
67 static constexpr const char* s_name = "AES-CTR";
68 static constexpr CryptoAlgorithmIdentifier s_identifier = CryptoAlgorithmIdentifier::AES_CTR;
69 static Ref<CryptoAlgorithm> create();
70
71private:
72 CryptoAlgorithmAES_CTR() = default;
73 CryptoAlgorithmIdentifier identifier() const final;
74
75 void encrypt(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&) final;
76 void decrypt(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&) final;
77 void generateKey(const CryptoAlgorithmParameters&, bool extractable, CryptoKeyUsageBitmap, KeyOrKeyPairCallback&&, ExceptionCallback&&, ScriptExecutionContext&) final;
78 void importKey(CryptoKeyFormat, KeyData&&, const CryptoAlgorithmParameters&, bool extractable, CryptoKeyUsageBitmap, KeyCallback&&, ExceptionCallback&&) final;
79 void exportKey(CryptoKeyFormat, Ref<CryptoKey>&&, KeyDataCallback&&, ExceptionCallback&&) final;
80 ExceptionOr<size_t> getKeyLength(const CryptoAlgorithmParameters&) final;
81
82 static ExceptionOr<Vector<uint8_t>> platformEncrypt(const CryptoAlgorithmAesCtrParams&, const CryptoKeyAES&, const Vector<uint8_t>&);
83 static ExceptionOr<Vector<uint8_t>> platformDecrypt(const CryptoAlgorithmAesCtrParams&, const CryptoKeyAES&, const Vector<uint8_t>&);
84};
