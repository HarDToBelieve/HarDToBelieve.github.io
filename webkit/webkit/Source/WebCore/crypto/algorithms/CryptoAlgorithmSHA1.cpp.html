46void CryptoAlgorithmSHA1::digest(Vector<uint8_t>&& message, VectorCallback&& callback, ExceptionCallback&& exceptionCallback, ScriptExecutionContext& context, WorkQueue& workQueue)
47{
48 auto digest = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_1);
49 if (!digest) {
50 exceptionCallback(OperationError);
51 return;
52 }
53
54 workQueue.dispatch([digest = WTFMove(digest), message = WTFMove(message), callback = WTFMove(callback), contextIdentifier = context.contextIdentifier()]() mutable {
55 digest->addBytes(message.data(), message.size());
56 auto result = digest->computeHash();
57 ScriptExecutionContext::postTaskTo(contextIdentifier, [callback = WTFMove(callback), result = WTFMove(result)](auto&) {
58 callback(result);
59 });
60 });
61}
