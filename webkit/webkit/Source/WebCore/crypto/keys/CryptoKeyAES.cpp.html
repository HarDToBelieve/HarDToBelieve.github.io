41static inline bool lengthIsValid(size_t length)
42{
43 return (length == CryptoKeyAES::s_length128) || (length == CryptoKeyAES::s_length192) || (length == CryptoKeyAES::s_length256);
44}

46CryptoKeyAES::CryptoKeyAES(CryptoAlgorithmIdentifier algorithm, const Vector<uint8_t>& key, bool extractable, CryptoKeyUsageBitmap usage)
47 : CryptoKey(algorithm, CryptoKeyType::Secret, extractable, usage)
48 , m_key(key)
49{
50 ASSERT(isValidAESAlgorithm(algorithm));

53CryptoKeyAES::CryptoKeyAES(CryptoAlgorithmIdentifier algorithm, Vector<uint8_t>&& key, bool extractable, CryptoKeyUsageBitmap usage)
54 : CryptoKey(algorithm, CryptoKeyType::Secret, extractable, usage)
55 , m_key(WTFMove(key))

62bool CryptoKeyAES::isValidAESAlgorithm(CryptoAlgorithmIdentifier algorithm)
63{
64 return algorithm == CryptoAlgorithmIdentifier::AES_CTR
65 || algorithm == CryptoAlgorithmIdentifier::AES_CBC
66 || algorithm == CryptoAlgorithmIdentifier::AES_GCM
67 || algorithm == CryptoAlgorithmIdentifier::AES_CFB
68 || algorithm == CryptoAlgorithmIdentifier::AES_KW;
69}

71RefPtr<CryptoKeyAES> CryptoKeyAES::generate(CryptoAlgorithmIdentifier algorithm, size_t lengthBits, bool extractable, CryptoKeyUsageBitmap usages)
72{
73 if (!lengthIsValid(lengthBits))
74 return nullptr;
75 return adoptRef(new CryptoKeyAES(algorithm, randomData(lengthBits / 8), extractable, usages));
76}
77
78RefPtr<CryptoKeyAES> CryptoKeyAES::importRaw(CryptoAlgorithmIdentifier algorithm, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap usages)
79{
80 if (!lengthIsValid(keyData.size() * 8))
81 return nullptr;
82 return adoptRef(new CryptoKeyAES(algorithm, WTFMove(keyData), extractable, usages));
83}
84
85RefPtr<CryptoKeyAES> CryptoKeyAES::importJwk(CryptoAlgorithmIdentifier algorithm, JsonWebKey&& keyData, bool extractable, CryptoKeyUsageBitmap usages, CheckAlgCallback&& callback)
86{
87 if (keyData.kty != "oct")
88 return nullptr;
89 if (keyData.k.isNull())
90 return nullptr;
91 auto octetSequence = base64URLDecode(keyData.k);
92 if (!octetSequence)
93 return nullptr;
94 if (!callback(octetSequence->size() * 8, keyData.alg))
95 return nullptr;
96 if (usages && !keyData.use.isNull() && keyData.use != "enc")
97 return nullptr;
98 if (keyData.key_ops && ((keyData.usages & usages) != usages))
99 return nullptr;
100 if (keyData.ext && !keyData.ext.value() && extractable)
101 return nullptr;
102
103 return adoptRef(new CryptoKeyAES(algorithm, WTFMove(*octetSequence), extractable, usages));
104}

106JsonWebKey CryptoKeyAES::exportJwk() const
107{
108 JsonWebKey result;
109 result.kty = "oct";
110 result.k = base64URLEncodeToString(m_key);
111 result.key_ops = usages();
112 result.ext = extractable();
113 return result;
114}

116ExceptionOr<size_t> CryptoKeyAES::getKeyLength(const CryptoAlgorithmParameters& parameters)
117{
118 auto& aesParameters = downcast<CryptoAlgorithmAesKeyParams>(parameters);
119 if (!lengthIsValid(aesParameters.length))
120 return Exception { OperationError };
121 return aesParameters.length;
122}
123
124auto CryptoKeyAES::algorithm() const -> KeyAlgorithm
125{
126 CryptoAesKeyAlgorithm result;
127 result.name = CryptoAlgorithmRegistry::singleton().name(algorithmIdentifier());
128 result.length = m_key.size() * 8;
129 return result;
130}
