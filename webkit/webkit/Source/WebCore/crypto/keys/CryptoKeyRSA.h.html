68class CryptoKeyRSA final : public CryptoKey {
69public:
70 static Ref<CryptoKeyRSA> create(CryptoAlgorithmIdentifier identifier, CryptoAlgorithmIdentifier hash, bool hasHash, CryptoKeyType type, PlatformRSAKeyContainer&& platformKey, bool extractable, CryptoKeyUsageBitmap usage)
71 {
72 return adoptRef(*new CryptoKeyRSA(identifier, hash, hasHash, type, WTFMove(platformKey), extractable, usage));
73 }
74 static RefPtr<CryptoKeyRSA> create(CryptoAlgorithmIdentifier, CryptoAlgorithmIdentifier hash, bool hasHash, const CryptoKeyRSAComponents&, bool extractable, CryptoKeyUsageBitmap);
75 virtual ~CryptoKeyRSA() = default;
76
77 bool isRestrictedToHash(CryptoAlgorithmIdentifier&) const;
78
79 size_t keySizeInBits() const;
80
81 using KeyPairCallback = WTF::Function<void(CryptoKeyPair&&)>;
82 using VoidCallback = WTF::Function<void()>;
83 static void generatePair(CryptoAlgorithmIdentifier, CryptoAlgorithmIdentifier hash, bool hasHash, unsigned modulusLength, const Vector<uint8_t>& publicExponent, bool extractable, CryptoKeyUsageBitmap, KeyPairCallback&&, VoidCallback&& failureCallback, ScriptExecutionContext*);
84 static RefPtr<CryptoKeyRSA> importJwk(CryptoAlgorithmIdentifier, std::optional<CryptoAlgorithmIdentifier> hash, JsonWebKey&&, bool extractable, CryptoKeyUsageBitmap);
85 static RefPtr<CryptoKeyRSA> importSpki(CryptoAlgorithmIdentifier, std::optional<CryptoAlgorithmIdentifier> hash, Vector<uint8_t>&&, bool extractable, CryptoKeyUsageBitmap);
86 static RefPtr<CryptoKeyRSA> importPkcs8(CryptoAlgorithmIdentifier, std::optional<CryptoAlgorithmIdentifier> hash, Vector<uint8_t>&&, bool extractable, CryptoKeyUsageBitmap);
87
88 PlatformRSAKey platformKey() const { return m_platformKey.get(); }
89 JsonWebKey exportJwk() const;
90 ExceptionOr<Vector<uint8_t>> exportSpki() const;
91 ExceptionOr<Vector<uint8_t>> exportPkcs8() const;
92
93 std::unique_ptr<CryptoKeyRSAComponents> exportData() const;
94
95 CryptoAlgorithmIdentifier hashAlgorithmIdentifier() const { return m_hash; }
96
97private:
98 CryptoKeyRSA(CryptoAlgorithmIdentifier, CryptoAlgorithmIdentifier hash, bool hasHash, CryptoKeyType, PlatformRSAKeyContainer&&, bool extractable, CryptoKeyUsageBitmap);
99
100 CryptoKeyClass keyClass() const final { return CryptoKeyClass::RSA; }
101
102 KeyAlgorithm algorithm() const final;
103
104 PlatformRSAKeyContainer m_platformKey;
105
106 bool m_restrictedToSpecificHash;
107 CryptoAlgorithmIdentifier m_hash;
108};

110} // namespace WebCore
