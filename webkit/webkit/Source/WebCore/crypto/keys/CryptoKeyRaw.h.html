26#pragma once
27
28#include "CryptoKey.h"
29
30#if ENABLE(WEB_CRYPTO)
31
32namespace WebCore {
33
34class CryptoKeyRaw final : public CryptoKey {
35public:
36 static Ref<CryptoKeyRaw> create(CryptoAlgorithmIdentifier identifier, Vector<uint8_t>&& keyData, CryptoKeyUsageBitmap usages)
37 {
38 return adoptRef(*new CryptoKeyRaw(identifier, WTFMove(keyData), usages));
39 }
40
41 const Vector<uint8_t>& key() const { return m_key; }
42
43private:
44 CryptoKeyRaw(CryptoAlgorithmIdentifier, Vector<uint8_t>&& keyData, CryptoKeyUsageBitmap);
45
46 CryptoKeyClass keyClass() const final { return CryptoKeyClass::Raw; }
47
48 KeyAlgorithm algorithm() const final;
49
50 Vector<uint8_t> m_key;
51};
52
53} // namespace WebCore
54
55SPECIALIZE_TYPE_TRAITS_CRYPTO_KEY(CryptoKeyRaw, CryptoKeyClass::Raw)
56
57#endif // ENABLE(WEB_CRYPTO)
