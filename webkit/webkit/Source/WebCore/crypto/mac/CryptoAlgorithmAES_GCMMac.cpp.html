38static ExceptionOr<Vector<uint8_t>> encryptAES_GCM(const Vector<uint8_t>& iv, const Vector<uint8_t>& key, const Vector<uint8_t>& plainText, const Vector<uint8_t>& additionalData, size_t desiredTagLengthInBytes)
39{
40 Vector<uint8_t> cipherText(plainText.size() + desiredTagLengthInBytes); // Per section 5.2.1.2: http://nvlpubs.nist.gov/nistpubs/Legacy/SP/nistspecialpublication800-38d.pdf
41 Vector<uint8_t> tag(desiredTagLengthInBytes);
42 // tagLength is actual an input <rdar://problem/30660074>
43 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
44 CCCryptorStatus status = CCCryptorGCM(kCCEncrypt, kCCAlgorithmAES, key.data(), key.size(), iv.data(), iv.size(), additionalData.data(), additionalData.size(), plainText.data(), plainText.size(), cipherText.data(), tag.data(), &desiredTagLengthInBytes);
45 ALLOW_DEPRECATED_DECLARATIONS_END
46 if (status)
47 return Exception { OperationError };
48
49 memcpy(cipherText.data() + plainText.size(), tag.data(), desiredTagLengthInBytes);
50
51 return WTFMove(cipherText);
52}

54static ExceptionOr<Vector<uint8_t>> decyptAES_GCM(const Vector<uint8_t>& iv, const Vector<uint8_t>& key, const Vector<uint8_t>& cipherText, const Vector<uint8_t>& additionalData, size_t desiredTagLengthInBytes)
55{
56 Vector<uint8_t> plainText(cipherText.size()); // Per section 5.2.1.2: http://nvlpubs.nist.gov/nistpubs/Legacy/SP/nistspecialpublication800-38d.pdf
57 Vector<uint8_t> tag(desiredTagLengthInBytes);
58 size_t offset = cipherText.size() - desiredTagLengthInBytes;
59 // tagLength is actual an input <rdar://problem/30660074>
60 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
61 CCCryptorStatus status = CCCryptorGCM(kCCDecrypt, kCCAlgorithmAES, key.data(), key.size(), iv.data(), iv.size(), additionalData.data(), additionalData.size(), cipherText.data(), offset, plainText.data(), tag.data(), &desiredTagLengthInBytes);
62 ALLOW_DEPRECATED_DECLARATIONS_END
63 if (status)
64 return Exception { OperationError };
65
66 // Using a constant time comparison to prevent timing attacks.
67 if (constantTimeMemcmp(tag.data(), cipherText.data() + offset, desiredTagLengthInBytes))
68 return Exception { OperationError };
69
70 plainText.shrink(offset);
71 return WTFMove(plainText);
72}

74ExceptionOr<Vector<uint8_t>> CryptoAlgorithmAES_GCM::platformEncrypt(const CryptoAlgorithmAesGcmParams& parameters, const CryptoKeyAES& key, const Vector<uint8_t>& plainText)
75{
76 return encryptAES_GCM(parameters.ivVector(), key.key(), plainText, parameters.additionalDataVector(), parameters.tagLength.value_or(0) / 8);
77}
78
79ExceptionOr<Vector<uint8_t>> CryptoAlgorithmAES_GCM::platformDecrypt(const CryptoAlgorithmAesGcmParams& parameters, const CryptoKeyAES& key, const Vector<uint8_t>& cipherText)
80{
81 return decyptAES_GCM(parameters.ivVector(), key.key(), cipherText, parameters.additionalDataVector(), parameters.tagLength.value_or(0) / 8);
82}
