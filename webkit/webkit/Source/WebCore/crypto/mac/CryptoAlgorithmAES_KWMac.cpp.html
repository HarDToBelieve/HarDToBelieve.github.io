36static ExceptionOr<Vector<uint8_t>> wrapKeyAES_KW(const Vector<uint8_t>& key, const Vector<uint8_t>& data)
37{
38 Vector<uint8_t> result(CCSymmetricWrappedSize(kCCWRAPAES, data.size()));
39 size_t resultSize = result.size();
40 if (CCSymmetricKeyWrap(kCCWRAPAES, CCrfc3394_iv, CCrfc3394_ivLen, key.data(), key.size(), data.data(), data.size(), result.data(), &resultSize))
41 return Exception { OperationError };
42
43 result.shrink(resultSize);
44 return WTFMove(result);
45}
46
47static ExceptionOr<Vector<uint8_t>> unwrapKeyAES_KW(const Vector<uint8_t>& key, const Vector<uint8_t>& data)
48{
49 Vector<uint8_t> result(CCSymmetricUnwrappedSize(kCCWRAPAES, data.size()));
50 size_t resultSize = result.size();
51
52 if (resultSize % 8)
53 return Exception { OperationError };
54
55 if (CCSymmetricKeyUnwrap(kCCWRAPAES, CCrfc3394_iv, CCrfc3394_ivLen, key.data(), key.size(), data.data(), data.size(), result.data(), &resultSize))
56 return Exception { OperationError };
57
58 result.shrink(resultSize);
59 return WTFMove(result);
60}
61
62ExceptionOr<Vector<uint8_t>> CryptoAlgorithmAES_KW::platformWrapKey(const CryptoKeyAES& key, const Vector<uint8_t>& data)
63{
64 return wrapKeyAES_KW(key.key(), data);
65}
66
67ExceptionOr<Vector<uint8_t>> CryptoAlgorithmAES_KW::platformUnwrapKey(const CryptoKeyAES& key, const Vector<uint8_t>& data)
68{
69 return unwrapKeyAES_KW(key.key(), data);
70}
