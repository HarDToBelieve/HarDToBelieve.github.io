39static ExceptionOr<Vector<uint8_t>> signECDSA(CryptoAlgorithmIdentifier hash, const PlatformECKey key, size_t keyLengthInBytes, const Vector<uint8_t>& data)
40{
41 CCDigestAlgorithm digestAlgorithm;
42 if (!getCommonCryptoDigestAlgorithm(hash, digestAlgorithm))
43 return Exception { OperationError };
44
45 auto cryptoDigestAlgorithm = WebCore::cryptoDigestAlgorithm(hash);
46 if (!cryptoDigestAlgorithm)
47 return Exception { OperationError };
48 auto digest = PAL::CryptoDigest::create(*cryptoDigestAlgorithm);
49 if (!digest)
50 return Exception { OperationError };
51 digest->addBytes(data.data(), data.size());
52 auto digestData = digest->computeHash();
53
54 // The signature produced by CCECCryptorSignHash is in DER format.
55 // tag + length(1) + tag + length(1) + InitialOctet(?) + keyLength in bytes + tag + length(1) + InitialOctet(?) + keyLength in bytes
56 Vector<uint8_t> signature(8 + keyLengthInBytes * 2);
57 size_t signatureSize = signature.size();
58
59 CCCryptorStatus status = CCECCryptorSignHash(key, digestData.data(), digestData.size(), signature.data(), &signatureSize);
60 if (status)
61 return Exception { OperationError };
62
63 // FIXME: <rdar://problem/31618371>
64 // convert the DER binary into r + s
65 Vector<uint8_t> newSignature;
66 newSignature.reserveCapacity(keyLengthInBytes * 2);
67 size_t offset = 3; // skip tag, length, tag
68
69 // If r < keyLengthInBytes, fill the head of r with 0s.
70 size_t bytesToCopy = keyLengthInBytes;
71 if (signature[offset] < keyLengthInBytes) {
72 newSignature.resize(keyLengthInBytes - signature[offset]);
73 memset(newSignature.data(), InitialOctet, keyLengthInBytes - signature[offset]);
74 bytesToCopy = signature[offset];
75 } else if (signature[offset] > keyLengthInBytes) // Otherwise skip the leading 0s of r.
76 offset += signature[offset] - keyLengthInBytes;
77 offset++; // skip length
78 ASSERT_WITH_SECURITY_IMPLICATION(signature.size() > offset + bytesToCopy);
79 newSignature.append(signature.data() + offset, bytesToCopy);
80 offset += bytesToCopy + 1; // skip r, tag
81
82 // If s < keyLengthInBytes, fill the head of s with 0s.
83 bytesToCopy = keyLengthInBytes;
84 if (signature[offset] < keyLengthInBytes) {
85 size_t pos = newSignature.size();
86 newSignature.resize(pos + keyLengthInBytes - signature[offset]);
87 memset(newSignature.data() + pos, InitialOctet, keyLengthInBytes - signature[offset]);
88 bytesToCopy = signature[offset];
89 } else if (signature[offset] > keyLengthInBytes) // Otherwise skip the leading 0s of s.
90 offset += signature[offset] - keyLengthInBytes;
91 offset++; // skip length
92 ASSERT_WITH_SECURITY_IMPLICATION(signature.size() >= offset + bytesToCopy);
93 newSignature.append(signature.data() + offset, bytesToCopy);
94
95 return WTFMove(newSignature);
96}

98static ExceptionOr<bool> verifyECDSA(CryptoAlgorithmIdentifier hash, const PlatformECKey key, size_t keyLengthInBytes, const Vector<uint8_t>& signature, const Vector<uint8_t> data)
99{
100 CCDigestAlgorithm digestAlgorithm;
101 if (!getCommonCryptoDigestAlgorithm(hash, digestAlgorithm))
102 return Exception { OperationError };
103
104 auto cryptoDigestAlgorithm = WebCore::cryptoDigestAlgorithm(hash);
105 if (!cryptoDigestAlgorithm)
106 return Exception { OperationError };
107 auto digest = PAL::CryptoDigest::create(*cryptoDigestAlgorithm);
108 if (!digest)
109 return Exception { OperationError };
110 digest->addBytes(data.data(), data.size());
111 auto digestData = digest->computeHash();
112
113 if (signature.size() != keyLengthInBytes * 2)
114 return false;
115
116 // FIXME: <rdar://problem/31618371>
117 // Convert the signature into DER format.
118 // tag + length(1) + tag + length(1) + InitialOctet(?) + r + tag + length(1) + InitialOctet(?) + s
119 // Skip any heading 0s of r and s.
120 size_t rStart = 0;
121 while (rStart < keyLengthInBytes && !signature[rStart])
122 rStart++;
123 size_t sStart = keyLengthInBytes;
124 while (sStart < signature.size() && !signature[sStart])
125 sStart++;
126 if (rStart >= keyLengthInBytes || sStart >= signature.size())
127 return false;
128
129 // InitialOctet is needed when the first byte of r/s is larger than or equal to 128.
130 bool rNeedsInitialOctet = signature[rStart] >= 128;
131 bool sNeedsInitialOctet = signature[sStart] >= 128;
132
133 // Construct the DER signature.
134 Vector<uint8_t> newSignature;
135 newSignature.reserveCapacity(6 + keyLengthInBytes * 3 + rNeedsInitialOctet + sNeedsInitialOctet - rStart - sStart);
136 newSignature.append(SequenceMark);
137 addEncodedASN1Length(newSignature, 4 + keyLengthInBytes * 3 + rNeedsInitialOctet + sNeedsInitialOctet - rStart - sStart);
138 newSignature.append(IntegerMark);
139 addEncodedASN1Length(newSignature, keyLengthInBytes + rNeedsInitialOctet - rStart);
140 if (rNeedsInitialOctet)
141 newSignature.append(InitialOctet);
142 newSignature.append(signature.data() + rStart, keyLengthInBytes - rStart);
143 newSignature.append(IntegerMark);
144 addEncodedASN1Length(newSignature, keyLengthInBytes * 2 + sNeedsInitialOctet - sStart);
145 if (sNeedsInitialOctet)
146 newSignature.append(InitialOctet);
147 newSignature.append(signature.data() + sStart, keyLengthInBytes * 2 - sStart);
148
149 uint32_t valid;
150 CCCryptorStatus status = CCECCryptorVerifyHash(key, digestData.data(), digestData.size(), newSignature.data(), newSignature.size(), &valid);
151 if (status)
152 return Exception { OperationError };
153 return valid;
154}

156ExceptionOr<Vector<uint8_t>> CryptoAlgorithmECDSA::platformSign(const CryptoAlgorithmEcdsaParams& parameters, const CryptoKeyEC& key, const Vector<uint8_t>& data)
157{
158 return signECDSA(parameters.hashIdentifier, key.platformKey(), key.keySizeInBits() / 8, data);
159}
160
161ExceptionOr<bool> CryptoAlgorithmECDSA::platformVerify(const CryptoAlgorithmEcdsaParams& parameters, const CryptoKeyEC& key, const Vector<uint8_t>& signature, const Vector<uint8_t>& data)
162{
163 return verifyECDSA(parameters.hashIdentifier, key.platformKey(), key.keySizeInBits() / 8, signature, data);
164}
