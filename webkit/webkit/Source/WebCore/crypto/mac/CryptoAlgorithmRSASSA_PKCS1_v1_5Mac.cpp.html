37static ExceptionOr<Vector<uint8_t>> signRSASSA_PKCS1_v1_5(CryptoAlgorithmIdentifier hash, const PlatformRSAKey key, size_t keyLength, const Vector<uint8_t>& data)
38{
39 CCDigestAlgorithm digestAlgorithm;
40 if (!getCommonCryptoDigestAlgorithm(hash, digestAlgorithm))
41 return Exception { OperationError };
42
43 auto cryptoDigestAlgorithm = WebCore::cryptoDigestAlgorithm(hash);
44 if (!cryptoDigestAlgorithm)
45 return Exception { OperationError };
46 auto digest = PAL::CryptoDigest::create(*cryptoDigestAlgorithm);
47 if (!digest)
48 return Exception { OperationError };
49 digest->addBytes(data.data(), data.size());
50 auto digestData = digest->computeHash();
51
52 Vector<uint8_t> signature(keyLength / 8); // Per https://tools.ietf.org/html/rfc3447#section-8.2.1
53 size_t signatureSize = signature.size();
54
55 CCCryptorStatus status = CCRSACryptorSign(key, ccPKCS1Padding, digestData.data(), digestData.size(), digestAlgorithm, 0, signature.data(), &signatureSize);
56 if (status)
57 return Exception { OperationError };
58
59 return WTFMove(signature);
60}
61
62static ExceptionOr<bool> verifyRSASSA_PKCS1_v1_5(CryptoAlgorithmIdentifier hash, const PlatformRSAKey key, const Vector<uint8_t>& signature, const Vector<uint8_t>& data)
63{
64 CCDigestAlgorithm digestAlgorithm;
65 if (!getCommonCryptoDigestAlgorithm(hash, digestAlgorithm))
66 return Exception { OperationError };
67
68 auto cryptoDigestAlgorithm = WebCore::cryptoDigestAlgorithm(hash);
69 if (!cryptoDigestAlgorithm)
70 return Exception { OperationError };
71 auto digest = PAL::CryptoDigest::create(*cryptoDigestAlgorithm);
72 if (!digest)
73 return Exception { OperationError };
74 digest->addBytes(data.data(), data.size());
75 auto digestData = digest->computeHash();
76
77 auto status = CCRSACryptorVerify(key, ccPKCS1Padding, digestData.data(), digestData.size(), digestAlgorithm, 0, signature.data(), signature.size());
78 if (!status)
79 return true;
80 if (status == kCCDecodeError)
81 return false;
82
83 return Exception { OperationError };
84}
85
86ExceptionOr<Vector<uint8_t>> CryptoAlgorithmRSASSA_PKCS1_v1_5::platformSign(const CryptoKeyRSA& key, const Vector<uint8_t>& data)
87{
88 return signRSASSA_PKCS1_v1_5(key.hashAlgorithmIdentifier(), key.platformKey(), key.keySizeInBits(), data);
89}
90
91ExceptionOr<bool> CryptoAlgorithmRSASSA_PKCS1_v1_5::platformVerify(const CryptoKeyRSA& key, const Vector<uint8_t>& signature, const Vector<uint8_t>& data)
92{
93 return verifyRSASSA_PKCS1_v1_5(key.hashAlgorithmIdentifier(), key.platformKey(), signature, data);
94}
