38static ExceptionOr<Vector<uint8_t>> signRSA_PSS(CryptoAlgorithmIdentifier hash, const PlatformRSAKey key, size_t keyLength, const Vector<uint8_t>& data, size_t saltLength)
39{
40 CCDigestAlgorithm digestAlgorithm;
41 if (!getCommonCryptoDigestAlgorithm(hash, digestAlgorithm))
42 return Exception { OperationError };
43
44 auto cryptoDigestAlgorithm = WebCore::cryptoDigestAlgorithm(hash);
45 if (!cryptoDigestAlgorithm)
46 return Exception { OperationError };
47 auto digest = PAL::CryptoDigest::create(*cryptoDigestAlgorithm);
48 if (!digest)
49 return Exception { OperationError };
50 digest->addBytes(data.data(), data.size());
51 auto digestData = digest->computeHash();
52
53 Vector<uint8_t> signature(keyLength / 8); // Per https://tools.ietf.org/html/rfc3447#section-8.1.1
54 size_t signatureSize = signature.size();
55
56 CCCryptorStatus status = CCRSACryptorSign(key, ccRSAPSSPadding, digestData.data(), digestData.size(), digestAlgorithm, saltLength, signature.data(), &signatureSize);
57 if (status)
58 return Exception { OperationError };
59
60 return WTFMove(signature);
61}
62
63static ExceptionOr<bool> verifyRSA_PSS(CryptoAlgorithmIdentifier hash, const PlatformRSAKey key, const Vector<uint8_t>& signature, const Vector<uint8_t>& data, size_t saltLength)
64{
65 CCDigestAlgorithm digestAlgorithm;
66 if (!getCommonCryptoDigestAlgorithm(hash, digestAlgorithm))
67 return Exception { OperationError };
68
69 auto cryptoDigestAlgorithm = WebCore::cryptoDigestAlgorithm(hash);
70 if (!cryptoDigestAlgorithm)
71 return Exception { OperationError };
72 auto digest = PAL::CryptoDigest::create(*cryptoDigestAlgorithm);
73 if (!digest)
74 return Exception { OperationError };
75 digest->addBytes(data.data(), data.size());
76 auto digestData = digest->computeHash();
77
78 auto status = CCRSACryptorVerify(key, ccRSAPSSPadding, digestData.data(), digestData.size(), digestAlgorithm, saltLength, signature.data(), signature.size());
79 if (!status)
80 return true;
81 return false;
82}
83
84ExceptionOr<Vector<uint8_t>> CryptoAlgorithmRSA_PSS::platformSign(const CryptoAlgorithmRsaPssParams& parameters, const CryptoKeyRSA& key, const Vector<uint8_t>& data)
85{
86 return signRSA_PSS(key.hashAlgorithmIdentifier(), key.platformKey(), key.keySizeInBits(), data, parameters.saltLength);
87}
88
89ExceptionOr<bool> CryptoAlgorithmRSA_PSS::platformVerify(const CryptoAlgorithmRsaPssParams& parameters, const CryptoKeyRSA& key, const Vector<uint8_t>& signature, const Vector<uint8_t>& data)
90{
91 return verifyRSA_PSS(key.hashAlgorithmIdentifier(), key.platformKey(), signature, data, parameters.saltLength);
92}
