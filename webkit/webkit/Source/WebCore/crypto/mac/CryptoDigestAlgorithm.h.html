26#pragma once
27
28#include <pal/crypto/CryptoDigest.h>
29
30#if ENABLE(WEB_CRYPTO)
31
32#include "CryptoAlgorithm.h"
33
34namespace WebCore {
35
36static std::optional<PAL::CryptoDigest::Algorithm> cryptoDigestAlgorithm(CryptoAlgorithmIdentifier hashFunction)
37{
38 switch (hashFunction) {
39 case CryptoAlgorithmIdentifier::SHA_1:
40 return PAL::CryptoDigest::Algorithm::SHA_1;
41 case CryptoAlgorithmIdentifier::SHA_224:
42 return PAL::CryptoDigest::Algorithm::SHA_224;
43 case CryptoAlgorithmIdentifier::SHA_256:
44 return PAL::CryptoDigest::Algorithm::SHA_256;
45 case CryptoAlgorithmIdentifier::SHA_384:
46 return PAL::CryptoDigest::Algorithm::SHA_384;
47 case CryptoAlgorithmIdentifier::SHA_512:
48 return PAL::CryptoDigest::Algorithm::SHA_512;
49 default:
50 return std::nullopt;
51 }
52}
53
54} // namespace WebCore
55
56#endif // ENABLE(WEB_CRYPTO)
