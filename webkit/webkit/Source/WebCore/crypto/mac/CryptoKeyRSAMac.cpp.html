44// OID rsaEncryption: 1.2.840.113549.1.1.1. Per https://tools.ietf.org/html/rfc3279#section-2.3.1
45static const unsigned char RSAOIDHeader[] = {0x30, 0x0d, 0x06, 0x09, 0x2a, 0x86, 0x48, 0x86, 0xf7, 0x0d, 0x01, 0x01, 0x01, 0x05, 0x00};

47// FIXME: We should get rid of magic number 16384. It assumes that the length of provided key will not exceed 16KB.
48// https://bugs.webkit.org/show_bug.cgi?id=164942
49static CCCryptorStatus getPublicKeyComponents(const PlatformRSAKeyContainer& rsaKey, Vector<uint8_t>& modulus, Vector<uint8_t>& publicExponent)
50{
51 ASSERT(CCRSAGetKeyType(rsaKey.get()) == ccRSAKeyPublic || CCRSAGetKeyType(rsaKey.get()) == ccRSAKeyPrivate);
52 bool keyIsPublic = CCRSAGetKeyType(rsaKey.get()) == ccRSAKeyPublic;
53 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
54 PlatformRSAKeyContainer publicKeyFromPrivateKey(keyIsPublic ? nullptr : CCRSACryptorGetPublicKeyFromPrivateKey(rsaKey.get()));
55 ALLOW_DEPRECATED_DECLARATIONS_END
56
57 modulus.resize(16384);
58 size_t modulusLength = modulus.size();
59 publicExponent.resize(16384);
60 size_t exponentLength = publicExponent.size();
61 CCCryptorStatus status = CCRSAGetKeyComponents(keyIsPublic ? rsaKey.get() : publicKeyFromPrivateKey.get(), modulus.data(), &modulusLength, publicExponent.data(), &exponentLength, 0, 0, 0, 0);
62 if (status)
63 return status;
64
65 modulus.shrink(modulusLength);
66 publicExponent.shrink(exponentLength);
67 return status;
68}

70static CCCryptorStatus getPrivateKeyComponents(const PlatformRSAKeyContainer& rsaKey, Vector<uint8_t>& privateExponent, CryptoKeyRSAComponents::PrimeInfo& firstPrimeInfo, CryptoKeyRSAComponents::PrimeInfo& secondPrimeInfo)
71{
72 ASSERT(CCRSAGetKeyType(rsaKey.get()) == ccRSAKeyPrivate);
73
74 Vector<uint8_t> unusedModulus(16384);
75 size_t modulusLength = unusedModulus.size();
76 privateExponent.resize(16384);
77 size_t exponentLength = privateExponent.size();
78 firstPrimeInfo.primeFactor.resize(16384);
79 size_t pLength = firstPrimeInfo.primeFactor.size();
80 secondPrimeInfo.primeFactor.resize(16384);
81 size_t qLength = secondPrimeInfo.primeFactor.size();
82
83 CCCryptorStatus status = CCRSAGetKeyComponents(rsaKey.get(), unusedModulus.data(), &modulusLength, privateExponent.data(), &exponentLength, firstPrimeInfo.primeFactor.data(), &pLength, secondPrimeInfo.primeFactor.data(), &qLength);
84 if (status)
85 return status;
86
87 privateExponent.shrink(exponentLength);
88 firstPrimeInfo.primeFactor.shrink(pLength);
89 secondPrimeInfo.primeFactor.shrink(qLength);
90
91 size_t dpSize;
92 size_t dqSize;
93 size_t qinvSize;
94 if (auto status = CCRSAGetCRTComponentsSizes(rsaKey.get(), &dpSize, &dqSize, &qinvSize))
95 return status;
96
97 Vector<uint8_t> dp(dpSize);
98 Vector<uint8_t> dq(dqSize);
99 Vector<uint8_t> qinv(qinvSize);
100 if (auto status = CCRSAGetCRTComponents(rsaKey.get(), dp.data(), dpSize, dq.data(), dqSize, qinv.data(), qinvSize))
101 return status;
102
103 firstPrimeInfo.factorCRTExponent = WTFMove(dp);
104 secondPrimeInfo.factorCRTExponent = WTFMove(dq);
105 secondPrimeInfo.factorCRTCoefficient = WTFMove(qinv);
106
107 return status;
108}

118RefPtr<CryptoKeyRSA> CryptoKeyRSA::create(CryptoAlgorithmIdentifier identifier, CryptoAlgorithmIdentifier hash, bool hasHash, const CryptoKeyRSAComponents& keyData, bool extractable, CryptoKeyUsageBitmap usage)
119{
120 if (keyData.type() == CryptoKeyRSAComponents::Type::Private && !keyData.hasAdditionalPrivateKeyParameters()) {
121 // <rdar://problem/15452324> tracks adding support.
122 WTFLogAlways("Private keys without additional data are not supported");
123 return nullptr;
124 }
125 if (keyData.otherPrimeInfos().size()) {
126 // <rdar://problem/15444074> tracks adding support.
127 WTFLogAlways("Keys with more than two primes are not supported");
128 return nullptr;
129 }
130 // When an empty vector p is provided to CCRSACryptorCreateFromData to create a private key, it crashes.
131 // <rdar://problem/30550228> tracks the issue.
132 if (keyData.type() == CryptoKeyRSAComponents::Type::Private && keyData.firstPrimeInfo().primeFactor.isEmpty())
133 return nullptr;
134
135 CCRSACryptorRef cryptor = nullptr;
136 // FIXME: It is so weired that we recaculate the private exponent from first prime factor and second prime factor,
137 // given the fact that we have already had it. Also, the re-caculated private exponent may not match the given one.
138 // See <rdar://problem/15452324>.
139 CCCryptorStatus status = CCRSACryptorCreateFromData(
140 keyData.type() == CryptoKeyRSAComponents::Type::Public ? ccRSAKeyPublic : ccRSAKeyPrivate,
141 keyData.modulus().data(), keyData.modulus().size(),
142 keyData.exponent().data(), keyData.exponent().size(),
143 keyData.firstPrimeInfo().primeFactor.data(), keyData.firstPrimeInfo().primeFactor.size(),
144 keyData.secondPrimeInfo().primeFactor.data(), keyData.secondPrimeInfo().primeFactor.size(),
145 &cryptor);
146
147 if (status) {
148 LOG_ERROR("Couldn't create RSA key from data, error %d", status);
149 return nullptr;
150 }
151
152 return adoptRef(new CryptoKeyRSA(identifier, hash, hasHash, keyData.type() == CryptoKeyRSAComponents::Type::Public ? CryptoKeyType::Public : CryptoKeyType::Private, PlatformRSAKeyContainer(cryptor), extractable, usage));
153}

155bool CryptoKeyRSA::isRestrictedToHash(CryptoAlgorithmIdentifier& identifier) const
156{
157 if (!m_restrictedToSpecificHash)
158 return false;
159
160 identifier = m_hash;
161 return true;

164size_t CryptoKeyRSA::keySizeInBits() const
165{
166 Vector<uint8_t> modulus;
167 Vector<uint8_t> publicExponent;
168 CCCryptorStatus status = getPublicKeyComponents(m_platformKey, modulus, publicExponent);
169 if (status) {
170 WTFLogAlways("Couldn't get RSA key components, status %d", status);
171 return 0;
172 }
173
174 return modulus.size() * 8;
175}

177auto CryptoKeyRSA::algorithm() const -> KeyAlgorithm
178{
179 // FIXME: Add a version of getPublicKeyComponents that returns Uint8Array, rather
180 // than Vector<uint8_t>, to avoid a copy of the data.
181
182 Vector<uint8_t> modulus;
183 Vector<uint8_t> publicExponent;
184 CCCryptorStatus status = getPublicKeyComponents(m_platformKey, modulus, publicExponent);
185 if (status) {
186 WTFLogAlways("Couldn't get RSA key components, status %d", status);
187 publicExponent.clear();
188
189 CryptoRsaKeyAlgorithm result;
190 result.name = CryptoAlgorithmRegistry::singleton().name(algorithmIdentifier());
191 result.modulusLength = 0;
192 result.publicExponent = Uint8Array::tryCreate(0);
193 return result;
194 }
195
196 size_t modulusLength = modulus.size() * 8;
197
198 if (m_restrictedToSpecificHash) {
199 CryptoRsaHashedKeyAlgorithm result;
200 result.name = CryptoAlgorithmRegistry::singleton().name(algorithmIdentifier());
201 result.modulusLength = modulusLength;
202 result.publicExponent = Uint8Array::tryCreate(publicExponent.data(), publicExponent.size());
203 result.hash.name = CryptoAlgorithmRegistry::singleton().name(m_hash);
204 return result;
205 }
206
207 CryptoRsaKeyAlgorithm result;
208 result.name = CryptoAlgorithmRegistry::singleton().name(algorithmIdentifier());
209 result.modulusLength = modulusLength;
210 result.publicExponent = Uint8Array::tryCreate(publicExponent.data(), publicExponent.size());
211 return result;
212}

214std::unique_ptr<CryptoKeyRSAComponents> CryptoKeyRSA::exportData() const
215{
216 switch (CCRSAGetKeyType(m_platformKey.get())) {
217 case ccRSAKeyPublic: {
218 Vector<uint8_t> modulus;
219 Vector<uint8_t> publicExponent;
220 CCCryptorStatus status = getPublicKeyComponents(m_platformKey, modulus, publicExponent);
221 if (status) {
222 WTFLogAlways("Couldn't get RSA key components, status %d", status);
223 return nullptr;
224 }
225 return CryptoKeyRSAComponents::createPublic(modulus, publicExponent);
226 }
227 case ccRSAKeyPrivate: {
228 Vector<uint8_t> modulus;
229 Vector<uint8_t> publicExponent;
230 CCCryptorStatus status = getPublicKeyComponents(m_platformKey, modulus, publicExponent);
231 if (status) {
232 WTFLogAlways("Couldn't get RSA key components, status %d", status);
233 return nullptr;
234 }
235 Vector<uint8_t> privateExponent;
236 CryptoKeyRSAComponents::PrimeInfo firstPrimeInfo;
237 CryptoKeyRSAComponents::PrimeInfo secondPrimeInfo;
238 Vector<CryptoKeyRSAComponents::PrimeInfo> otherPrimeInfos; // Always empty, CommonCrypto only supports two primes (cf. <rdar://problem/15444074>).
239 status = getPrivateKeyComponents(m_platformKey, privateExponent, firstPrimeInfo, secondPrimeInfo);
240 if (status) {
241 WTFLogAlways("Couldn't get RSA key components, status %d", status);
242 return nullptr;
243 }
244 return CryptoKeyRSAComponents::createPrivateWithAdditionalData(modulus, publicExponent, privateExponent, firstPrimeInfo, secondPrimeInfo, otherPrimeInfos);
245 }
246 default:
247 return nullptr;
248 }
249}

251static bool bigIntegerToUInt32(const Vector<uint8_t>& bigInteger, uint32_t& result)
252{
253 result = 0;
254 for (size_t i = 0; i + 4 < bigInteger.size(); ++i) {
255 if (bigInteger[i])
256 return false; // Too big to fit in 32 bits.
257 }
258
259 for (size_t i = bigInteger.size() > 4 ? bigInteger.size() - 4 : 0; i < bigInteger.size(); ++i) {
260 result <<= 8;
261 result += bigInteger[i];
262 }
263 return true;
264}

266// FIXME: We should use WorkQueue here instead of dispatch_async once WebKitSubtleCrypto is deprecated.
267// https://bugs.webkit.org/show_bug.cgi?id=164943
268void CryptoKeyRSA::generatePair(CryptoAlgorithmIdentifier algorithm, CryptoAlgorithmIdentifier hash, bool hasHash, unsigned modulusLength, const Vector<uint8_t>& publicExponent, bool extractable, CryptoKeyUsageBitmap usage, KeyPairCallback&& callback, VoidCallback&& failureCallback, ScriptExecutionContext* context)
269{
270 uint32_t e;
271 if (!bigIntegerToUInt32(publicExponent, e)) {
272 // Adding support is tracked as <rdar://problem/15444034>.
273 WTFLogAlways("Public exponent is too big, not supported");
274 failureCallback();
275 return;
276 }
277
278 __block auto blockCallback(WTFMove(callback));
279 __block auto blockFailureCallback(WTFMove(failureCallback));
280 auto contextIdentifier = context->contextIdentifier();
281 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
282 CCRSACryptorRef ccPublicKey = nullptr;
283 CCRSACryptorRef ccPrivateKey = nullptr;
284 CCCryptorStatus status = CCRSACryptorGeneratePair(modulusLength, e, &ccPublicKey, &ccPrivateKey);
285 if (status) {
286 WTFLogAlways("Could not generate a key pair, status %d", status);
287 ScriptExecutionContext::postTaskTo(contextIdentifier, [callback = WTFMove(blockCallback), failureCallback = WTFMove(blockFailureCallback)](auto&) {
288 failureCallback();
289 });
290 return;
291 }
292 ScriptExecutionContext::postTaskTo(contextIdentifier, [algorithm, hash, hasHash, extractable, usage, callback = WTFMove(blockCallback), failureCallback = WTFMove(blockFailureCallback), ccPublicKey = PlatformRSAKeyContainer(ccPublicKey), ccPrivateKey = PlatformRSAKeyContainer(ccPrivateKey)](auto&) mutable {
293 auto publicKey = CryptoKeyRSA::create(algorithm, hash, hasHash, CryptoKeyType::Public, WTFMove(ccPublicKey), true, usage);
294 auto privateKey = CryptoKeyRSA::create(algorithm, hash, hasHash, CryptoKeyType::Private, WTFMove(ccPrivateKey), extractable, usage);
295
296 callback(CryptoKeyPair { WTFMove(publicKey), WTFMove(privateKey) });
297 });
298 });
299}

301RefPtr<CryptoKeyRSA> CryptoKeyRSA::importSpki(CryptoAlgorithmIdentifier identifier, std::optional<CryptoAlgorithmIdentifier> hash, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap usages)
302{
303 // The current SecLibrary cannot import a SPKI format binary. Hence, we need to strip out the SPKI header.
304 // This hack can be removed when <rdar://problem/29523286> is resolved.
305 // The header format we assume is: SequenceMark(1) + Length(?) + rsaEncryption(15) + BitStringMark(1) + Length(?) + InitialOctet(1).
306 // The header format could be varied. However since we don't have a full-fledged ASN.1 encoder/decoder, we want to restrict it to
307 // the most common one for now.
308 // Per https://tools.ietf.org/html/rfc5280#section-4.1. subjectPublicKeyInfo.
309 size_t headerSize = 1;
310 if (keyData.size() < headerSize + 1)
311 return nullptr;
312 headerSize += bytesUsedToEncodedLength(keyData[headerSize]) + sizeof(RSAOIDHeader) + sizeof(BitStringMark);
313 if (keyData.size() < headerSize + 1)
314 return nullptr;
315 headerSize += bytesUsedToEncodedLength(keyData[headerSize]) + sizeof(InitialOctet);
316
317 CCRSACryptorRef ccPublicKey = nullptr;
318 if (CCRSACryptorImport(keyData.data() + headerSize, keyData.size() - headerSize, &ccPublicKey))
319 return nullptr;
320
321 // Notice: CryptoAlgorithmIdentifier::SHA_1 is just a placeholder. It should not have any effect if hash is std::nullopt.
322 return adoptRef(new CryptoKeyRSA(identifier, hash.value_or(CryptoAlgorithmIdentifier::SHA_1), !!hash, CryptoKeyType::Public, PlatformRSAKeyContainer(ccPublicKey), extractable, usages));
323}

325ExceptionOr<Vector<uint8_t>> CryptoKeyRSA::exportSpki() const
326{
327 if (type() != CryptoKeyType::Public)
328 return Exception { InvalidAccessError };
329
330 // The current SecLibrary cannot output a valid SPKI format binary. Hence, we need the following hack.
331 // This hack can be removed when <rdar://problem/29523286> is resolved.
332 // Estimated size in produced bytes format. Per https://tools.ietf.org/html/rfc3279#section-2.3.1. RSAPublicKey.
333 // O(size) = Sequence(1) + Length(3) + Integer(1) + Length(3) + Modulus + Integer(1) + Length(3) + Exponent
334 Vector<uint8_t> keyBytes(keySizeInBits() / 4);
335 size_t keySize = keyBytes.size();
336 if (CCRSACryptorExport(platformKey(), keyBytes.data(), &keySize))
337 return Exception { OperationError };
338 keyBytes.shrink(keySize);
339
340 // RSAOIDHeader + BitStringMark + Length + keySize + InitialOctet
341 size_t totalSize = sizeof(RSAOIDHeader) + bytesNeededForEncodedLength(keySize + 1) + keySize + 2;
342
343 // Per https://tools.ietf.org/html/rfc5280#section-4.1. subjectPublicKeyInfo.
344 Vector<uint8_t> result;
345 result.reserveCapacity(totalSize + bytesNeededForEncodedLength(totalSize) + 1);
346 result.append(SequenceMark);
347 addEncodedASN1Length(result, totalSize);
348 result.append(RSAOIDHeader, sizeof(RSAOIDHeader));
349 result.append(BitStringMark);
350 addEncodedASN1Length(result, keySize + 1);
351 result.append(InitialOctet);
352 result.append(keyBytes.data(), keyBytes.size());
353
354 return WTFMove(result);
355}

357RefPtr<CryptoKeyRSA> CryptoKeyRSA::importPkcs8(CryptoAlgorithmIdentifier identifier, std::optional<CryptoAlgorithmIdentifier> hash, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap usages)
358{
359 // The current SecLibrary cannot import a PKCS8 format binary. Hence, we need to strip out the PKCS8 header.
360 // This hack can be removed when <rdar://problem/29523286> is resolved.
361 // The header format we assume is: SequenceMark(1) + Length(?) + Version(3) + rsaEncryption(15) + OctetStringMark(1) + Length(?).
362 // The header format could be varied. However since we don't have a full-fledged ASN.1 encoder/decoder, we want to restrict it to
363 // the most common one for now.
364 // Per https://tools.ietf.org/html/rfc5208#section-5. PrivateKeyInfo.
365 // We also assume there is no optional parameters.
366 size_t headerSize = 1;
367 if (keyData.size() < headerSize + 1)
368 return nullptr;
369 headerSize += bytesUsedToEncodedLength(keyData[headerSize]) + sizeof(Version) + sizeof(RSAOIDHeader) + sizeof(OctetStringMark);
370 if (keyData.size() < headerSize + 1)
371 return nullptr;
372 headerSize += bytesUsedToEncodedLength(keyData[headerSize]);
373
374 CCRSACryptorRef ccPrivateKey = nullptr;
375 if (CCRSACryptorImport(keyData.data() + headerSize, keyData.size() - headerSize, &ccPrivateKey))
376 return nullptr;
377
378 // Notice: CryptoAlgorithmIdentifier::SHA_1 is just a placeholder. It should not have any effect if hash is std::nullopt.
379 return adoptRef(new CryptoKeyRSA(identifier, hash.value_or(CryptoAlgorithmIdentifier::SHA_1), !!hash, CryptoKeyType::Private, PlatformRSAKeyContainer(ccPrivateKey), extractable, usages));
380}

382ExceptionOr<Vector<uint8_t>> CryptoKeyRSA::exportPkcs8() const
383{
384 if (type() != CryptoKeyType::Private)
385 return Exception { InvalidAccessError };
386
387 // The current SecLibrary cannot output a valid PKCS8 format binary. Hence, we need the following hack.
388 // This hack can be removed when <rdar://problem/29523286> is resolved.
389 // Estimated size in produced bytes format. Per https://tools.ietf.org/html/rfc3447#appendix-A.1.2. RSAPrivateKey.
390 // O(size) = Sequence(1) + Length(3) + Integer(1) + Length(3) + Modulus + Integer(1) + Length(3) + publicExponent + Integer(1) + Length(3) +
391 // privateExponent + Integer(1) + Length(3) + prime1 + Integer(1) + Length(3) + prime2 + Integer(1) + Length(3) + exponent1 + Integer(1) +
392 // Length(3) + exponent2 + Integer(1) + Length(3) + coefficient.
393 Vector<uint8_t> keyBytes(keySizeInBits());
394 size_t keySize = keyBytes.size();
395 if (CCRSACryptorExport(platformKey(), keyBytes.data(), &keySize))
396 return Exception { OperationError };
397 keyBytes.shrink(keySize);
398
399 // Version + RSAOIDHeader + OctetStringMark + Length + keySize
400 size_t totalSize = sizeof(Version) + sizeof(RSAOIDHeader) + bytesNeededForEncodedLength(keySize) + keySize + 1;
401
402 // Per https://tools.ietf.org/html/rfc5208#section-5. PrivateKeyInfo.
403 Vector<uint8_t> result;
404 result.reserveCapacity(totalSize + bytesNeededForEncodedLength(totalSize) + 1);
405 result.append(SequenceMark);
406 addEncodedASN1Length(result, totalSize);
407 result.append(Version, sizeof(Version));
408 result.append(RSAOIDHeader, sizeof(RSAOIDHeader));
409 result.append(OctetStringMark);
410 addEncodedASN1Length(result, keySize);
411 result.append(keyBytes.data(), keyBytes.size());
412
413 return WTFMove(result);
414}
