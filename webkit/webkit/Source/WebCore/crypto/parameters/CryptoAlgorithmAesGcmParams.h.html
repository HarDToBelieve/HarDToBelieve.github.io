27
28#include "BufferSource.h"
29#include "CryptoAlgorithmParameters.h"
30#include <wtf/Vector.h>
31
32#if ENABLE(WEB_CRYPTO)
33
34namespace WebCore {
35
36class CryptoAlgorithmAesGcmParams final : public CryptoAlgorithmParameters {
37public:
38 BufferSource iv;
39 // Use additionalDataVector() instead of additionalData. The label will be gone once additionalDataVector() is called.
40 mutable std::optional<BufferSource::VariantType> additionalData;
41 mutable std::optional<uint8_t> tagLength;
42
43 Class parametersClass() const final { return Class::AesGcmParams; }
44
45 const Vector<uint8_t>& ivVector() const
46 {
47 if (!m_ivVector.isEmpty() || !iv.length())
48 return m_ivVector;
49
50 m_ivVector.append(iv.data(), iv.length());
51 return m_ivVector;
52 }
53
54 const Vector<uint8_t>& additionalDataVector() const
55 {
56 if (!m_additionalDataVector.isEmpty() || !additionalData)
57 return m_additionalDataVector;
58
59 BufferSource additionalDataBuffer = WTFMove(*additionalData);
60 additionalData = std::nullopt;
61 if (!additionalDataBuffer.length())
62 return m_additionalDataVector;
63
64 m_additionalDataVector.append(additionalDataBuffer.data(), additionalDataBuffer.length());
65 return m_additionalDataVector;
66 }
67
68 CryptoAlgorithmAesGcmParams isolatedCopy() const
69 {
70 CryptoAlgorithmAesGcmParams result;
71 result.identifier = identifier;
72 result.m_ivVector = ivVector();
73 result.m_additionalDataVector = additionalDataVector();
74 result.tagLength = tagLength;
75
76 return result;
77 }
78
79private:
80 mutable Vector<uint8_t> m_ivVector;
81 mutable Vector<uint8_t> m_additionalDataVector;
82};
83
84} // namespace WebCore
