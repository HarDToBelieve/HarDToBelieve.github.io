26#pragma once
27
28#include "CryptoAlgorithmParameters.h"
29#include <JavaScriptCore/Uint8Array.h>
30#include <wtf/Vector.h>
31
32#if ENABLE(WEB_CRYPTO)
33
34namespace WebCore {
35
36class CryptoAlgorithmRsaKeyGenParams : public CryptoAlgorithmParameters {
37public:
38 size_t modulusLength;
39 RefPtr<Uint8Array> publicExponent;
40
41 Class parametersClass() const override { return Class::RsaKeyGenParams; }
42
43 const Vector<uint8_t>& publicExponentVector() const
44 {
45 if (!m_publicExponentVector.isEmpty() || !publicExponent->byteLength())
46 return m_publicExponentVector;
47
48 m_publicExponentVector.append(publicExponent->data(), publicExponent->byteLength());
49 return m_publicExponentVector;
50 }
51private:
52 mutable Vector<uint8_t> m_publicExponentVector;
53};
54
55} // namespace WebCore
56
57SPECIALIZE_TYPE_TRAITS_CRYPTO_ALGORITHM_PARAMETERS(RsaKeyGenParams)
58
59#endif // ENABLE(WEB_CRYPTO)
