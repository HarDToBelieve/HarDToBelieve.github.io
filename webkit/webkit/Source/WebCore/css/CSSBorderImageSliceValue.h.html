26#pragma once
27
28#include "CSSPrimitiveValue.h"
29#include <wtf/RefPtr.h>
30
31namespace WebCore {
32
33class Rect;
34
35class CSSBorderImageSliceValue final : public CSSValue {
36public:
37 static Ref<CSSBorderImageSliceValue> create(RefPtr<CSSPrimitiveValue>&& slices, bool fill)
38 {
39 return adoptRef(*new CSSBorderImageSliceValue(WTFMove(slices), fill));
40 }
41
42 String customCSSText() const;
43
44 Quad* slices() const { return m_slices ? m_slices->quadValue() : nullptr; }
45
46 bool equals(const CSSBorderImageSliceValue&) const;
47
48 // These four values are used to make "cuts" in the border image. They can be numbers
49 // or percentages.
50 RefPtr<CSSPrimitiveValue> m_slices;
51 bool m_fill;
52
53private:
54 CSSBorderImageSliceValue(RefPtr<CSSPrimitiveValue>&& slices, bool fill);
55};
56
57} // namespace WebCore
