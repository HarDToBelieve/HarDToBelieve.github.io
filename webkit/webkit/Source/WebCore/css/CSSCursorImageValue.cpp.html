36CSSCursorImageValue::CSSCursorImageValue(Ref<CSSValue>&& imageValue, const std::optional<IntPoint>& hotSpot, LoadedFromOpaqueSource loadedFromOpaqueSource)
37 : CSSValue(CursorImageClass)
38 , m_imageValue(WTFMove(imageValue))
39 , m_hotSpot(hotSpot)
40 , m_loadedFromOpaqueSource(loadedFromOpaqueSource)
41{
42 if (is<CSSImageValue>(m_imageValue.get()))
43 m_originalURL = downcast<CSSImageValue>(m_imageValue.get()).imageURL();
44}

46Ref<CSSCursorImageValue> CSSCursorImageValue::create(Ref<CSSValue>&& imageValue, const std::optional<IntPoint>& hotSpot, LoadedFromOpaqueSource loadedFromOpaqueSource)
47{
48 return adoptRef(*new CSSCursorImageValue(WTFMove(imageValue), hotSpot, loadedFromOpaqueSource));
49}
50
51CSSCursorImageValue::~CSSCursorImageValue()
52{
53 for (auto* element : m_cursorElements)
54 element->removeClient(*this);
55}

57String CSSCursorImageValue::customCSSText() const
58{
59 String text = m_imageValue.get().cssText();
60 if (!m_hotSpot)
61 return text;
62 return makeString(text, ' ', m_hotSpot->x(), ' ', m_hotSpot->y());
63}

65// FIXME: Should this function take a TreeScope instead?
66SVGCursorElement* CSSCursorImageValue::updateCursorElement(const Document& document)
67{
68 auto element = SVGURIReference::targetElementFromIRIString(m_originalURL.string(), document).element;
69 if (!is<SVGCursorElement>(element))
70 return nullptr;
71
72 // FIXME: Not right to keep old cursor elements as clients. The new one should replace the old, not join it in a set.
73 auto& cursorElement = downcast<SVGCursorElement>(*element);
74 if (m_cursorElements.add(&cursorElement).isNewEntry) {
75 cursorElementChanged(cursorElement);
76 cursorElement.addClient(*this);
77 }
78 return &cursorElement;
79}

81void CSSCursorImageValue::cursorElementRemoved(SVGCursorElement& cursorElement)
82{
83 // FIXME: Not right to stay a client of a cursor element until the element is destroyed. We'd want to stop being a client once it's no longer a valid target, like when it's disconnected.
84 m_cursorElements.remove(&cursorElement);
85}

87void CSSCursorImageValue::cursorElementChanged(SVGCursorElement& cursorElement)
88{
89 // FIXME: Seems wrong that changing an old cursor element, one that that is no longer the target, changes the hot spot.
90 // FIXME: This will override a hot spot that was specified in CSS, which is probably incorrect.
91 // FIXME: Should we clamp from float to int instead of just casting here?
92 SVGLengthContext lengthContext(nullptr);
93 m_hotSpot = IntPoint {
94 static_cast<int>(std::round(cursorElement.x().value(lengthContext))),
95 static_cast<int>(std::round(cursorElement.y().value(lengthContext)))
96 };
97
98 // FIXME: Why doesn't this funtion check for a change to the href of the cursor element? Why would we dynamically track changes to x/y but not href?
99}

101ImageWithScale CSSCursorImageValue::selectBestFitImage(const Document& document)
102{
103 if (is<CSSImageSetValue>(m_imageValue.get()))
104 return downcast<CSSImageSetValue>(m_imageValue.get()).selectBestFitImage(document);
105
106 if (auto* cursorElement = updateCursorElement(document)) {
107 auto location = document.completeURL(cursorElement->href());
108 if (location != downcast<CSSImageValue>(m_imageValue.get()).imageURL())
109 m_imageValue = CSSImageValue::create(WTFMove(location), m_loadedFromOpaqueSource);
110 }
111
112 return { m_imageValue.ptr() , 1 };
113}

115bool CSSCursorImageValue::equals(const CSSCursorImageValue& other) const
116{
117 return m_hotSpot == other.m_hotSpot && compareCSSValue(m_imageValue, other.m_imageValue);
118}

120Ref<CSSCursorImageValue> CSSCursorImageValue::valueWithStylesResolved(Style::BuilderState& state)
121{
122 auto imageValue = state.resolveImageStyles(m_imageValue.get());
123 if (imageValue.ptr() == m_imageValue.ptr())
124 return *this;
125 return create(WTFMove(imageValue), m_hotSpot, m_loadedFromOpaqueSource);
126}
