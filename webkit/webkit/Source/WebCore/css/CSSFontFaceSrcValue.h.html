34namespace WebCore {
35
36class CachedFont;
37class FontLoadRequest;
38class SVGFontFaceElement;
39class ScriptExecutionContext;
40
41class CSSFontFaceSrcValue final : public CSSValue {
42public:
43 static Ref<CSSFontFaceSrcValue> create(const String& resource, LoadedFromOpaqueSource loadedFromOpaqueSource)
44 {
45 return adoptRef(*new CSSFontFaceSrcValue(resource, false, loadedFromOpaqueSource));
46 }
47 static Ref<CSSFontFaceSrcValue> createLocal(const String& resource)
48 {
49 return adoptRef(*new CSSFontFaceSrcValue(resource, true, LoadedFromOpaqueSource::No));
50 }
51
52 const String& resource() const { return m_resource; }
53 const String& format() const { return m_format; }
54 bool isLocal() const { return m_isLocal; }
55
56 void setFormat(const String& format) { m_format = format; }
57
58 bool isSupportedFormat() const;
59
60 bool isSVGFontFaceSrc() const;
61 bool isSVGFontTarget() const;
62
63 SVGFontFaceElement* svgFontFaceElement() const { return m_svgFontFaceElement; }
64 void setSVGFontFaceElement(SVGFontFaceElement* element) { m_svgFontFaceElement = element; }
65
66 String customCSSText() const;
67
68 bool traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const;
69
70 std::unique_ptr<FontLoadRequest> fontLoadRequest(ScriptExecutionContext*, bool isSVG, bool isInitiatingElementInUserAgentShadowTree);
71
72 bool equals(const CSSFontFaceSrcValue&) const;
73
74private:
75 CSSFontFaceSrcValue(const String& resource, bool local, LoadedFromOpaqueSource loadedFromOpaqueSource)
76 : CSSValue(FontFaceSrcClass)
77 , m_resource(resource)
78 , m_isLocal(local)
79 , m_loadedFromOpaqueSource(loadedFromOpaqueSource)
80 , m_svgFontFaceElement(0)
81 {
82 }
83
84 String m_resource;
85 String m_format;
86 bool m_isLocal;
87 LoadedFromOpaqueSource m_loadedFromOpaqueSource { LoadedFromOpaqueSource::No };
88
89 CachedResourceHandle<CachedFont> m_cachedFont;
90
91 SVGFontFaceElement* m_svgFontFaceElement;
92};
93
94} // namespace WebCore
