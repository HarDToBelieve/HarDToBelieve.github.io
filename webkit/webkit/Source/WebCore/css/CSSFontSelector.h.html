52class CSSFontSelector final : public FontSelector, public CSSFontFace::Client, public CanMakeWeakPtr<CSSFontSelector>, public ActiveDOMObject {
53public:
54 static Ref<CSSFontSelector> create(ScriptExecutionContext& context)
55 {
56 return adoptRef(*new CSSFontSelector(context));
57 }
58 virtual ~CSSFontSelector();
59
60 unsigned version() const final { return m_version; }
61 unsigned uniqueId() const final { return m_uniqueId; }
62
63 FontRanges fontRangesForFamily(const FontDescription&, const AtomString&) final;
64 size_t fallbackFontCount() final;
65 RefPtr<Font> fallbackFontAt(const FontDescription&, size_t) final;
66
67 void clearFonts();
68 void emptyCaches();
69 void buildStarted();
70 void buildCompleted();
71
72 void addFontFaceRule(StyleRuleFontFace&, bool isInitiatingElementInUserAgentShadowTree);
73
74 FontCache& fontCache() const final { return m_fontCache.get(); }
75 void fontCacheInvalidated() final;
76
77 bool isEmpty() const;
78
79 void registerForInvalidationCallbacks(FontSelectorClient&) final;
80 void unregisterForInvalidationCallbacks(FontSelectorClient&) final;
81
82 ScriptExecutionContext* scriptExecutionContext() const { return m_context.get(); }
83
84 FontFaceSet* fontFaceSetIfExists();
85 FontFaceSet& fontFaceSet();
86
87 void incrementIsComputingRootStyleFont() { ++m_computingRootStyleFontCount; }
88 void decrementIsComputingRootStyleFont() { --m_computingRootStyleFontCount; }
89
90 void loadPendingFonts();
91
92 // CSSFontFace::Client needs to be able to be held in a RefPtr.
93 void ref() final { FontSelector::ref(); }
94 void deref() final { FontSelector::deref(); }
95
96private:
97 explicit CSSFontSelector(ScriptExecutionContext&);
98
99 void dispatchInvalidationCallbacks();

107 void fontStyleUpdateNeeded(CSSFontFace&) final;
108
109 void fontModified();
110
111 // ActiveDOMObject
112 const char* activeDOMObjectName() const final { return "CSSFontSelector"_s; }

120 WeakPtr<ScriptExecutionContext> m_context;
121 Ref<FontCache> m_fontCache;

137
138 WebKitFontFamilyNames::FamilyNamesList<AtomString> m_fontFamilyNames;
139};
