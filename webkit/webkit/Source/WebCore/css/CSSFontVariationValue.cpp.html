31namespace WebCore {
32
33CSSFontVariationValue::CSSFontVariationValue(FontTag tag, float value)
34 : CSSValue(FontVariationClass)
35 , m_tag(tag)
36 , m_value(value)
37{
38}
39
40String CSSFontVariationValue::customCSSText() const
41{
42 return makeString('"', m_tag[0], m_tag[1], m_tag[2], m_tag[3], "\" ", m_value);
43}
44
45bool CSSFontVariationValue::equals(const CSSFontVariationValue& other) const
46{
47 return m_tag == other.m_tag && m_value == other.m_value;
48}
