44static inline Ref<Gradient> createGradient(CSSGradientValue& value, RenderElement& renderer, FloatSize size)
45{
46 if (is<CSSLinearGradientValue>(value))
47 return downcast<CSSLinearGradientValue>(value).createGradient(renderer, size);
48 if (is<CSSRadialGradientValue>(value))
49 return downcast<CSSRadialGradientValue>(value).createGradient(renderer, size);
50 return downcast<CSSConicGradientValue>(value).createGradient(renderer, size);
51}
52
53RefPtr<Image> CSSGradientValue::image(RenderElement& renderer, const FloatSize& size)
54{
55 if (size.isEmpty())
56 return nullptr;
57 bool cacheable = isCacheable() && !renderer.style().hasAppleColorFilter();
58 if (cacheable) {
59 if (!clients().contains(&renderer))
60 return nullptr;
61 if (auto* result = cachedImageForSize(size))
62 return result;
63 }
64 auto newImage = GradientImage::create(createGradient(*this, renderer, size), size);
65 if (cacheable)
66 saveCachedImageForSize(size, newImage);
67 return newImage;
68}

96bool CSSGradientValue::hasColorDerivedFromElement() const
97{
98 if (!m_hasColorDerivedFromElement) {
99 m_hasColorDerivedFromElement = false;
100 for (auto& stop : m_stops) {
101 if (stop.color && Style::BuilderState::isColorFromPrimitiveValueDerivedFromElement(*stop.color)) {
102 m_hasColorDerivedFromElement = true;
103 break;
104 }
105 }
106 }
107 return *m_hasColorDerivedFromElement;
108}

127class LinearGradientAdapter {
128public:
129 explicit LinearGradientAdapter(Gradient::LinearData& data)
130 : m_data(data)
131 {
132 }
133
134 float gradientLength() const
135 {
136 auto gradientSize = m_data.point0 - m_data.point1;
137 return gradientSize.diagonalLength();
138 }
139 float maxExtent(float, float) const { return 1; }
140
141 void normalizeStopsAndEndpointsOutsideRange(Vector<GradientStop>& stops)
142 {
143 float firstOffset = *stops.first().offset;
144 float lastOffset = *stops.last().offset;
145 if (firstOffset != lastOffset) {
146 float scale = lastOffset - firstOffset;
147
148 for (auto& stop : stops)
149 stop.offset = (*stop.offset - firstOffset) / scale;
150
151 auto p0 = m_data.point0;
152 auto p1 = m_data.point1;
153 m_data.point0 = { p0.x() + firstOffset * (p1.x() - p0.x()), p0.y() + firstOffset * (p1.y() - p0.y()) };
154 m_data.point1 = { p1.x() + (lastOffset - 1) * (p1.x() - p0.x()), p1.y() + (lastOffset - 1) * (p1.y() - p0.y()) };
155 } else {
156 // There's a single position that is outside the scale, clamp the positions to 1.
157 for (auto& stop : stops)
158 stop.offset = 1;
159 }
160 }
161
162private:
163 Gradient::LinearData& m_data;
164};

166class RadialGradientAdapter {
167public:
168 explicit RadialGradientAdapter(Gradient::RadialData& data)
169 : m_data(data)
170 {
171 }
172
173 float gradientLength() const { return m_data.endRadius; }
174
175 // Radial gradients may need to extend further than the endpoints, because they have
176 // to repeat out to the corners of the box.
177 float maxExtent(float maxLengthForRepeat, float gradientLength) const
178 {
179 if (maxLengthForRepeat > gradientLength)
180 return gradientLength > 0 ? maxLengthForRepeat / gradientLength : 0;
181 return 1;
182 }
183
184 void normalizeStopsAndEndpointsOutsideRange(Vector<GradientStop>& stops)
185 {
186 auto numStops = stops.size();
187
188 // Rather than scaling the points < 0, we truncate them, so only scale according to the largest point.
189 float firstOffset = 0;
190 float lastOffset = *stops.last().offset;
191 float scale = lastOffset - firstOffset;
192
193 // Reset points below 0 to the first visible color.
194 size_t firstZeroOrGreaterIndex = numStops;
195 for (size_t i = 0; i < numStops; ++i) {
196 if (*stops[i].offset >= 0) {
197 firstZeroOrGreaterIndex = i;
198 break;
199 }
200 }
201
202 if (firstZeroOrGreaterIndex > 0) {
203 if (firstZeroOrGreaterIndex < numStops && *stops[firstZeroOrGreaterIndex].offset > 0) {
204 float prevOffset = *stops[firstZeroOrGreaterIndex - 1].offset;
205 float nextOffset = *stops[firstZeroOrGreaterIndex].offset;
206
207 float interStopProportion = -prevOffset / (nextOffset - prevOffset);
208 // FIXME: when we interpolate gradients using premultiplied colors, this should do premultiplication.
209 Color blendedColor = blend(stops[firstZeroOrGreaterIndex - 1].color, stops[firstZeroOrGreaterIndex].color, { interStopProportion });
210
211 // Clamp the positions to 0 and set the color.
212 for (size_t i = 0; i < firstZeroOrGreaterIndex; ++i) {
213 stops[i].offset = 0;
214 stops[i].color = blendedColor;
215 }
216 } else {
217 // All stops are below 0; just clamp them.
218 for (size_t i = 0; i < firstZeroOrGreaterIndex; ++i)
219 stops[i].offset = 0;
220 }
221 }
222
223 for (auto& stop : stops)
224 *stop.offset /= scale;
225
226 m_data.startRadius *= scale;
227 m_data.endRadius *= scale;
228 }
229
230private:
231 Gradient::RadialData& m_data;
232};

234class ConicGradientAdapter {
235public:
236 float gradientLength() const { return 1; }
237 float maxExtent(float, float) const { return 1; }
238
239 void normalizeStopsAndEndpointsOutsideRange(Vector<GradientStop>& stops)
240 {
241 size_t numStops = stops.size();
242 size_t lastStopIndex = numStops - 1;
243
244 std::optional<size_t> firstZeroOrGreaterIndex;
245 for (size_t i = 0; i < numStops; ++i) {
246 if (*stops[i].offset >= 0) {
247 firstZeroOrGreaterIndex = i;
248 break;
249 }
250 }
251
252 if (firstZeroOrGreaterIndex) {
253 size_t index = *firstZeroOrGreaterIndex;
254 if (index > 0) {
255 float previousOffset = *stops[index - 1].offset;
256 float nextOffset = *stops[index].offset;
257
258 float interStopProportion = -previousOffset / (nextOffset - previousOffset);
259 // FIXME: when we interpolate gradients using premultiplied colors, this should do premultiplication.
260 Color blendedColor = blend(stops[index - 1].color, stops[index].color, { interStopProportion });
261
262 // Clamp the positions to 0 and set the color.
263 for (size_t i = 0; i < index; ++i) {
264 stops[i].offset = 0;
265 stops[i].color = blendedColor;
266 }
267 }
268 } else {
269 // All stop offsets below 0, clamp them.
270 for (auto& stop : stops)
271 stop.offset = 0;
272 }
273
274 std::optional<size_t> lastOneOrLessIndex;
275 for (int i = lastStopIndex; i >= 0; --i) {
276 if (*stops[i].offset <= 1) {
277 lastOneOrLessIndex = i;
278 break;
279 }
280 }
281
282 if (lastOneOrLessIndex) {
283 size_t index = *lastOneOrLessIndex;
284 if (index < lastStopIndex) {
285 float previousOffset = *stops[index].offset;
286 float nextOffset = *stops[index + 1].offset;
287
288 float interStopProportion = (1 - previousOffset) / (nextOffset - previousOffset);
289 // FIXME: when we interpolate gradients using premultiplied colors, this should do premultiplication.
290 Color blendedColor = blend(stops[index].color, stops[index + 1].color, { interStopProportion });
291
292 // Clamp the positions to 1 and set the color.
293 for (size_t i = index + 1; i < numStops; ++i) {
294 stops[i].offset = 1;
295 stops[i].color = blendedColor;
296 }
297 }
298 } else {
299 // All stop offsets above 1, clamp them.
300 for (auto& stop : stops)
301 stop.offset = 1;
302 }
303 }
304};

306template<typename GradientAdapter>
307Gradient::ColorStopVector CSSGradientValue::computeStops(GradientAdapter& gradientAdapter, const CSSToLengthConversionData& conversionData, const RenderStyle& style, float maxLengthForRepeat)
308{
309 if (m_gradientType == CSSDeprecatedLinearGradient || m_gradientType == CSSDeprecatedRadialGradient) {
310 Gradient::ColorStopVector result;
311 result.reserveInitialCapacity(m_stops.size());
312
313 for (auto& stop : m_stops) {
314 float offset;
315 if (stop.position->isPercentage())
316 offset = stop.position->floatValue(CSSUnitType::CSS_PERCENTAGE) / 100;
317 else
318 offset = stop.position->floatValue(CSSUnitType::CSS_NUMBER);
319
320 Color color = stop.resolvedColor;
321 if (style.hasAppleColorFilter())
322 style.appleColorFilter().transformColor(color);
323 result.uncheckedAppend({ offset, color });
324 }
325
326 std::stable_sort(result.begin(), result.end(), [] (const Gradient::ColorStop& a, const Gradient::ColorStop& b) {
327 return a.offset < b.offset;
328 });
329
330 return result;
331 }
332
333 size_t numStops = m_stops.size();
334 Vector<GradientStop> stops(numStops);
335
336 float gradientLength = gradientAdapter.gradientLength();
337
338 for (size_t i = 0; i < numStops; ++i) {
339 auto& stop = m_stops[i];
340
341 Color color = stop.resolvedColor;
342 if (style.hasAppleColorFilter())
343 style.appleColorFilter().transformColor(color);
344
345 stops[i].color = color;
346
347 if (stop.position) {
348 auto& positionValue = *stop.position;
349 if (positionValue.isPercentage())
350 stops[i].offset = positionValue.floatValue(CSSUnitType::CSS_PERCENTAGE) / 100;
351 else if (positionValue.isLength() || positionValue.isViewportPercentageLength() || positionValue.isCalculatedPercentageWithLength()) {
352 float length;
353 if (positionValue.isLength())
354 length = positionValue.computeLength<float>(conversionData);
355 else {
356 Ref<CalculationValue> calculationValue { positionValue.cssCalcValue()->createCalculationValue(conversionData) };
357 length = calculationValue->evaluate(gradientLength);
358 }
359 stops[i].offset = (gradientLength > 0) ? length / gradientLength : 0;
360 } else if (positionValue.isAngle())
361 stops[i].offset = positionValue.floatValue(CSSUnitType::CSS_DEG) / 360;
362 else {
363 ASSERT_NOT_REACHED();
364 stops[i].offset = 0;
365 }
366 } else {
367 // If the first color-stop does not have a position, its position defaults to 0%.
368 // If the last color-stop does not have a position, its position defaults to 100%.
369 if (!i)
370 stops[i].offset = 0;
371 else if (numStops > 1 && i == numStops - 1)
372 stops[i].offset = 1;
373 }
374
375 // If a color-stop has a position that is less than the specified position of any
376 // color-stop before it in the list, its position is changed to be equal to the
377 // largest specified position of any color-stop before it.
378 if (stops[i].isSpecified() && i > 0) {
379 size_t prevSpecifiedIndex;
380 for (prevSpecifiedIndex = i - 1; prevSpecifiedIndex; --prevSpecifiedIndex) {
381 if (stops[prevSpecifiedIndex].isSpecified())
382 break;
383 }
384
385 if (*stops[i].offset < *stops[prevSpecifiedIndex].offset)
386 stops[i].offset = stops[prevSpecifiedIndex].offset;
387 }
388 }
389
390 ASSERT(stops[0].isSpecified() && stops[numStops - 1].isSpecified());
391
392 // If any color-stop still does not have a position, then, for each run of adjacent
393 // color-stops without positions, set their positions so that they are evenly spaced
394 // between the preceding and following color-stops with positions.
395 if (numStops > 2) {
396 size_t unspecifiedRunStart = 0;
397 bool inUnspecifiedRun = false;
398
399 for (size_t i = 0; i < numStops; ++i) {
400 if (!stops[i].isSpecified() && !inUnspecifiedRun) {
401 unspecifiedRunStart = i;
402 inUnspecifiedRun = true;
403 } else if (stops[i].isSpecified() && inUnspecifiedRun) {
404 size_t unspecifiedRunEnd = i;
405
406 if (unspecifiedRunStart < unspecifiedRunEnd) {
407 float lastSpecifiedOffset = *stops[unspecifiedRunStart - 1].offset;
408 float nextSpecifiedOffset = *stops[unspecifiedRunEnd].offset;
409 float delta = (nextSpecifiedOffset - lastSpecifiedOffset) / (unspecifiedRunEnd - unspecifiedRunStart + 1);
410
411 for (size_t j = unspecifiedRunStart; j < unspecifiedRunEnd; ++j)
412 stops[j].offset = lastSpecifiedOffset + (j - unspecifiedRunStart + 1) * delta;
413 }
414
415 inUnspecifiedRun = false;
416 }
417 }
418 }
419
420 // Walk over the color stops, look for midpoints and add stops as needed.
421 // If mid < 50%, add 2 stops to the left and 6 to the right
422 // else add 6 stops to the left and 2 to the right.
423 // Stops on the side with the most stops start midway because the curve approximates
424 // a line in that region. We then add 5 more color stops on that side to minimize the change
425 // how the luminance changes at each of the color stops. We don't have to add as many on the other side
426 // since it becomes small which increases the differentation of luminance which hides the color stops.
427 // Even with 4 extra color stops, it *is* possible to discern the steps when the gradient is large and has
428 // large luminance differences between midpoint and color stop. If this becomes an issue, we can consider
429 // making this algorithm a bit smarter.
430
431 // Midpoints that coincide with color stops are treated specially since they don't require
432 // extra stops and generate hard lines.
433 for (size_t x = 1; x < stops.size() - 1;) {
434 if (!stops[x].isMidpoint()) {
435 ++x;
436 continue;
437 }
438
439 // Find previous and next color so we know what to interpolate between.
440 // We already know they have a color since we checked for that earlier.
441 Color color1 = stops[x - 1].color;
442 Color color2 = stops[x + 1].color;
443 // Likewise find the position of previous and next color stop.
444 float offset1 = *stops[x - 1].offset;
445 float offset2 = *stops[x + 1].offset;
446 float offset = *stops[x].offset;
447
448 // Check if everything coincides or the midpoint is exactly in the middle.
449 // If so, ignore the midpoint.
450 if (offset - offset1 == offset2 - offset) {
451 stops.remove(x);
452 continue;
453 }
454
455 // Check if we coincide with the left color stop.
456 if (offset1 == offset) {
457 // Morph the midpoint to a regular stop with the color of the next color stop.
458 stops[x].color = color2;
459 continue;
460 }
461
462 // Check if we coincide with the right color stop.
463 if (offset2 == offset) {
464 // Morph the midpoint to a regular stop with the color of the previous color stop.
465 stops[x].color = color1;
466 continue;
467 }
468
469 float midpoint = (offset - offset1) / (offset2 - offset1);
470 GradientStop newStops[9];
471 if (midpoint > .5f) {
472 for (size_t y = 0; y < 7; ++y)
473 newStops[y].offset = offset1 + (offset - offset1) * (7 + y) / 13;
474
475 newStops[7].offset = offset + (offset2 - offset) / 3;
476 newStops[8].offset = offset + (offset2 - offset) * 2 / 3;
477 } else {
478 newStops[0].offset = offset1 + (offset - offset1) / 3;
479 newStops[1].offset = offset1 + (offset - offset1) * 2 / 3;
480
481 for (size_t y = 0; y < 7; ++y)
482 newStops[y + 2].offset = offset + (offset2 - offset) * y / 13;
483 }
484 // calculate colors
485 for (size_t y = 0; y < 9; ++y) {
486 float relativeOffset = (*newStops[y].offset - offset1) / (offset2 - offset1);
487 float multiplier = std::pow(relativeOffset, std::log(.5f) / std::log(midpoint));
488 // FIXME: Why not premultiply here?
489 newStops[y].color = blendWithoutPremultiply(color1, color2, { multiplier });
490 }
491
492 stops.remove(x);
493 stops.insert(x, newStops, 9);
494 x += 9;
495 }
496
497 numStops = stops.size();
498
499 // If the gradient is repeating, repeat the color stops.
500 // We can't just push this logic down into the platform-specific Gradient code,
501 // because we have to know the extent of the gradient, and possible move the end points.
502 if (m_repeating && numStops > 1) {
503 // If the difference in the positions of the first and last color-stops is 0,
504 // the gradient defines a solid-color image with the color of the last color-stop in the rule.
505 float gradientRange = *stops.last().offset - *stops.first().offset;
506 if (!gradientRange) {
507 stops.first().offset = 0;
508 stops.first().color = stops.last().color;
509 stops.shrink(1);
510 numStops = 1;
511 } else {
512 float maxExtent = gradientAdapter.maxExtent(maxLengthForRepeat, gradientLength);
513
514 size_t originalNumStops = numStops;
515 size_t originalFirstStopIndex = 0;
516
517 // Work backwards from the first, adding stops until we get one before 0.
518 float firstOffset = *stops[0].offset;
519 if (firstOffset > 0) {
520 float currOffset = firstOffset;
521 size_t srcStopOrdinal = originalNumStops - 1;
522
523 while (true) {
524 GradientStop newStop = stops[originalFirstStopIndex + srcStopOrdinal];
525 newStop.offset = currOffset;
526 stops.insert(0, newStop);
527 ++originalFirstStopIndex;
528 if (currOffset < 0)
529 break;
530
531 if (srcStopOrdinal)
532 currOffset -= *stops[originalFirstStopIndex + srcStopOrdinal].offset - *stops[originalFirstStopIndex + srcStopOrdinal - 1].offset;
533 srcStopOrdinal = (srcStopOrdinal + originalNumStops - 1) % originalNumStops;
534 }
535 }
536
537 // Work forwards from the end, adding stops until we get one after 1.
538 float lastOffset = *stops[stops.size() - 1].offset;
539 if (lastOffset < maxExtent) {
540 float currOffset = lastOffset;
541 size_t srcStopOrdinal = 0;
542
543 while (true) {
544 size_t srcStopIndex = originalFirstStopIndex + srcStopOrdinal;
545 GradientStop newStop = stops[srcStopIndex];
546 newStop.offset = currOffset;
547 stops.append(newStop);
548 if (currOffset > maxExtent)
549 break;
550 if (srcStopOrdinal < originalNumStops - 1)
551 currOffset += *stops[srcStopIndex + 1].offset - *stops[srcStopIndex].offset;
552 srcStopOrdinal = (srcStopOrdinal + 1) % originalNumStops;
553 }
554 }
555 }
556 }
557
558 // If the gradient goes outside the 0-1 range, normalize it by moving the endpoints, and adjusting the stops.
559 if (stops.size() > 1 && (*stops.first().offset < 0 || *stops.last().offset > 1))
560 gradientAdapter.normalizeStopsAndEndpointsOutsideRange(stops);
561
562 Gradient::ColorStopVector result;
563 result.reserveInitialCapacity(stops.size());
564 for (auto& stop : stops)
565 result.uncheckedAppend({ *stop.offset, stop.color });
566
567 return result;
568}

570static float positionFromValue(const CSSPrimitiveValue* value, const CSSToLengthConversionData& conversionData, const FloatSize& size, bool isHorizontal)
571{
572 if (!value)
573 return 0;
574
575 float origin = 0;
576 float sign = 1;
577 float edgeDistance = isHorizontal ? size.width() : size.height();
578
579 // In this case the center of the gradient is given relative to an edge in the
580 // form of: [ top | bottom | right | left ] [ <percentage> | <length> ].
581 if (value->isPair()) {
582 CSSValueID originID = value->pairValue()->first()->valueID();
583 if (originID == CSSValueRight || originID == CSSValueBottom) {
584 // For right/bottom, the offset is relative to the far edge.
585 origin = edgeDistance;
586 sign = -1;
587 }
588 value = value->pairValue()->second();
589 }
590
591 if (value->isNumber())
592 return origin + sign * value->floatValue() * conversionData.zoom();
593
594 if (value->isPercentage())
595 return origin + sign * value->floatValue() / 100 * edgeDistance;
596
597 if (value->isCalculatedPercentageWithLength())
598 return origin + sign * value->cssCalcValue()->createCalculationValue(conversionData)->evaluate(edgeDistance);
599
600 switch (value->valueID()) {
601 case CSSValueTop:
602 ASSERT(!isHorizontal);
603 return 0;
604 case CSSValueLeft:
605 ASSERT(isHorizontal);
606 return 0;
607 case CSSValueBottom:
608 ASSERT(!isHorizontal);
609 return edgeDistance;
610 case CSSValueRight:
611 ASSERT(isHorizontal);
612 return edgeDistance;
613 case CSSValueCenter:
614 return origin + sign * .5f * edgeDistance;
615 default:
616 break;
617 }
618
619 return origin + sign * value->computeLength<float>(conversionData);
620}

623static FloatPoint computeEndPoint(const CSSPrimitiveValue* horizontal, const CSSPrimitiveValue* vertical, const CSSToLengthConversionData& conversionData, const FloatSize& size)
624{
625 return { positionFromValue(horizontal, conversionData, size, true), positionFromValue(vertical, conversionData, size, false) };

639bool CSSGradientValue::knownToBeOpaque(const RenderElement& renderer) const
640{
641 bool hasColorFilter = renderer.style().hasAppleColorFilter();
642 for (auto& stop : m_stops) {
643 Color color = stop.resolvedColor;
644 if (hasColorFilter)
645 renderer.style().appleColorFilter().transformColor(color);
646 if (!color.isOpaque())
647 return false;
648 }
649 return true;
650}

652bool CSSGradientValue::equals(const CSSGradientValue& other) const
653{
654 return compareCSSValuePtr(m_firstX, other.m_firstX)
655 && compareCSSValuePtr(m_firstY, other.m_firstY)
656 && compareCSSValuePtr(m_secondX, other.m_secondX)
657 && compareCSSValuePtr(m_secondY, other.m_secondY)
658 && m_stops == other.m_stops
659 && m_gradientType == other.m_gradientType
660 && m_repeating == other.m_repeating;
661}

663static void appendGradientStops(StringBuilder& builder, const Vector<CSSGradientColorStop, 2>& stops)
664{
665 for (auto& stop : stops) {
666 double position = stop.position->doubleValue(CSSUnitType::CSS_NUMBER);
667 if (!position)
668 builder.append(", from(", stop.color->cssText(), ')');
669 else if (position == 1)
670 builder.append(", to(", stop.color->cssText(), ')');
671 else
672 builder.append(", color-stop(", position, ", ", stop.color->cssText(), ')');
673 }

741// Compute the endpoints so that a gradient of the given angle covers a box of the given size.
742static void endPointsFromAngle(float angleDeg, const FloatSize& size, FloatPoint& firstPoint, FloatPoint& secondPoint, CSSGradientType type)
743{
744 // Prefixed gradients use "polar coordinate" angles, rather than "bearing" angles.
745 if (type == CSSPrefixedLinearGradient)
746 angleDeg = 90 - angleDeg;
747
748 angleDeg = fmodf(angleDeg, 360);
749 if (angleDeg < 0)
750 angleDeg += 360;
751
752 if (!angleDeg) {
753 firstPoint.set(0, size.height());
754 secondPoint.set(0, 0);
755 return;
756 }
757
758 if (angleDeg == 90) {
759 firstPoint.set(0, 0);
760 secondPoint.set(size.width(), 0);
761 return;
762 }
763
764 if (angleDeg == 180) {
765 firstPoint.set(0, 0);
766 secondPoint.set(0, size.height());
767 return;
768 }
769
770 if (angleDeg == 270) {
771 firstPoint.set(size.width(), 0);
772 secondPoint.set(0, 0);
773 return;
774 }
775
776 // angleDeg is a "bearing angle" (0deg = N, 90deg = E),
777 // but tan expects 0deg = E, 90deg = N.
778 float slope = tan(deg2rad(90 - angleDeg));
779
780 // We find the endpoint by computing the intersection of the line formed by the slope,
781 // and a line perpendicular to it that intersects the corner.
782 float perpendicularSlope = -1 / slope;
783
784 // Compute start corner relative to center, in Cartesian space (+y = up).
785 float halfHeight = size.height() / 2;
786 float halfWidth = size.width() / 2;
787 FloatPoint endCorner;
788 if (angleDeg < 90)
789 endCorner.set(halfWidth, halfHeight);
790 else if (angleDeg < 180)
791 endCorner.set(halfWidth, -halfHeight);
792 else if (angleDeg < 270)
793 endCorner.set(-halfWidth, -halfHeight);
794 else
795 endCorner.set(-halfWidth, halfHeight);
796
797 // Compute c (of y = mx + c) using the corner point.
798 float c = endCorner.y() - perpendicularSlope * endCorner.x();
799 float endX = c / (slope - perpendicularSlope);
800 float endY = perpendicularSlope * endX + c;
801
802 // We computed the end point, so set the second point,
803 // taking into account the moved origin and the fact that we're in drawing space (+y = down).
804 secondPoint.set(halfWidth + endX, halfHeight - endY);
805 // Reflect around the center for the start point.
806 firstPoint.set(halfWidth - endX, halfHeight + endY);
807}

809Ref<Gradient> CSSLinearGradientValue::createGradient(RenderElement& renderer, const FloatSize& size)
810{
811 ASSERT(!size.isEmpty());
812
813 const RenderStyle* rootStyle = nullptr;
814 if (auto* documentElement = renderer.document().documentElement())
815 rootStyle = documentElement->renderStyle();
816
817 CSSToLengthConversionData conversionData(&renderer.style(), rootStyle, renderer.parentStyle(), &renderer.view());
818
819 FloatPoint firstPoint;
820 FloatPoint secondPoint;
821 if (m_angle) {
822 float angle = m_angle->floatValue(CSSUnitType::CSS_DEG);
823 endPointsFromAngle(angle, size, firstPoint, secondPoint, gradientType());
824 } else {
825 switch (gradientType()) {
826 case CSSDeprecatedLinearGradient:
827 firstPoint = computeEndPoint(firstX(), firstY(), conversionData, size);
828 if (secondX() || secondY())
829 secondPoint = computeEndPoint(secondX(), secondY(), conversionData, size);
830 else {
831 if (firstX())
832 secondPoint.setX(size.width() - firstPoint.x());
833 if (firstY())
834 secondPoint.setY(size.height() - firstPoint.y());
835 }
836 break;
837 case CSSPrefixedLinearGradient:
838 firstPoint = computeEndPoint(firstX(), firstY(), conversionData, size);
839 if (firstX())
840 secondPoint.setX(size.width() - firstPoint.x());
841 if (firstY())
842 secondPoint.setY(size.height() - firstPoint.y());
843 break;
844 case CSSLinearGradient:
845 if (firstX() && firstY()) {
846 // "Magic" corners, so the 50% line touches two corners.
847 float rise = size.width();
848 float run = size.height();
849 if (firstX() && firstX()->valueID() == CSSValueLeft)
850 run *= -1;
851 if (firstY() && firstY()->valueID() == CSSValueBottom)
852 rise *= -1;
853 // Compute angle, and flip it back to "bearing angle" degrees.
854 float angle = 90 - rad2deg(atan2(rise, run));
855 endPointsFromAngle(angle, size, firstPoint, secondPoint, gradientType());
856 } else if (firstX() || firstY()) {
857 secondPoint = computeEndPoint(firstX(), firstY(), conversionData, size);
858 if (firstX())
859 firstPoint.setX(size.width() - secondPoint.x());
860 if (firstY())
861 firstPoint.setY(size.height() - secondPoint.y());
862 } else
863 secondPoint.setY(size.height());
864 break;
865 default:
866 ASSERT_NOT_REACHED();
867 }
868 }
869
870 Gradient::LinearData data { firstPoint, secondPoint };
871 LinearGradientAdapter adapter { data };
872 auto stops = computeStops(adapter, conversionData, renderer.style(), 1);
873
874 auto gradient = Gradient::create(WTFMove(data));
875 gradient->setSortedColorStops(WTFMove(stops));
876 return gradient;
877}

985static float distanceToClosestCorner(const FloatPoint& p, const FloatSize& size, FloatPoint& corner)
986{
987 FloatPoint topLeft;
988 float topLeftDistance = FloatSize(p - topLeft).diagonalLength();
989
990 FloatPoint topRight(size.width(), 0);
991 float topRightDistance = FloatSize(p - topRight).diagonalLength();
992
993 FloatPoint bottomLeft(0, size.height());
994 float bottomLeftDistance = FloatSize(p - bottomLeft).diagonalLength();
995
996 FloatPoint bottomRight(size.width(), size.height());
997 float bottomRightDistance = FloatSize(p - bottomRight).diagonalLength();
998
999 corner = topLeft;
1000 float minDistance = topLeftDistance;
1001 if (topRightDistance < minDistance) {
1002 minDistance = topRightDistance;
1003 corner = topRight;
1004 }
1005
1006 if (bottomLeftDistance < minDistance) {
1007 minDistance = bottomLeftDistance;
1008 corner = bottomLeft;
1009 }
1010
1011 if (bottomRightDistance < minDistance) {
1012 minDistance = bottomRightDistance;
1013 corner = bottomRight;
1014 }
1015 return minDistance;
1016}

1051// Compute horizontal radius of ellipse with center at 0,0 which passes through p, and has
1052// width/height given by aspectRatio.
1053static inline float horizontalEllipseRadius(const FloatSize& p, float aspectRatio)
1054{
1055 // x^2/a^2 + y^2/b^2 = 1
1056 // a/b = aspectRatio, b = a/aspectRatio
1057 // a = sqrt(x^2 + y^2/(1/r^2))
1058 return std::hypot(p.width(), p.height() * aspectRatio);
1059}

1062Ref<Gradient> CSSRadialGradientValue::createGradient(RenderElement& renderer, const FloatSize& size)
1063{
1064 ASSERT(!size.isEmpty());
1065
1066 const RenderStyle* rootStyle = nullptr;
1067 if (auto* documentElement = renderer.document().documentElement())
1068 rootStyle = documentElement->renderStyle();
1069
1070 CSSToLengthConversionData conversionData(&renderer.style(), rootStyle, renderer.parentStyle(), &renderer.view());
1071
1072 FloatPoint firstPoint = computeEndPoint(firstX(), firstY(), conversionData, size);
1073 if (!firstX())
1074 firstPoint.setX(size.width() / 2);
1075 if (!firstY())
1076 firstPoint.setY(size.height() / 2);
1077
1078 FloatPoint secondPoint = computeEndPoint(secondX(), secondY(), conversionData, size);
1079 if (!secondX())
1080 secondPoint.setX(size.width() / 2);
1081 if (!secondY())
1082 secondPoint.setY(size.height() / 2);
1083
1084 float firstRadius = 0;
1085 if (m_firstRadius)
1086 firstRadius = resolveRadius(*m_firstRadius, conversionData);
1087
1088 float secondRadius = 0;
1089 float aspectRatio = 1; // width / height.
1090 if (m_secondRadius)
1091 secondRadius = resolveRadius(*m_secondRadius, conversionData);
1092 else if (m_endHorizontalSize) {
1093 float width = size.width();
1094 float height = size.height();
1095 secondRadius = resolveRadius(*m_endHorizontalSize, conversionData, &width);
1096 if (m_endVerticalSize)
1097 aspectRatio = secondRadius / resolveRadius(*m_endVerticalSize, conversionData, &height);
1098 else
1099 aspectRatio = 1;
1100 } else {
1101 enum GradientShape { Circle, Ellipse };
1102 GradientShape shape = Ellipse;
1103 if ((m_shape && m_shape->valueID() == CSSValueCircle)
1104 || (!m_shape && !m_sizingBehavior && m_endHorizontalSize && !m_endVerticalSize))
1105 shape = Circle;
1106
1107 enum GradientFill { ClosestSide, ClosestCorner, FarthestSide, FarthestCorner };
1108 GradientFill fill = FarthestCorner;
1109
1110 switch (m_sizingBehavior ? m_sizingBehavior->valueID() : 0) {
1111 case CSSValueContain:
1112 case CSSValueClosestSide:
1113 fill = ClosestSide;
1114 break;
1115 case CSSValueClosestCorner:
1116 fill = ClosestCorner;
1117 break;
1118 case CSSValueFarthestSide:
1119 fill = FarthestSide;
1120 break;
1121 case CSSValueCover:
1122 case CSSValueFarthestCorner:
1123 fill = FarthestCorner;
1124 break;
1125 default:
1126 break;
1127 }
1128
1129 // Now compute the end radii based on the second point, shape and fill.
1130
1131 // Horizontal
1132 switch (fill) {
1133 case ClosestSide: {
1134 float xDist = std::min(secondPoint.x(), size.width() - secondPoint.x());
1135 float yDist = std::min(secondPoint.y(), size.height() - secondPoint.y());
1136 if (shape == Circle) {
1137 float smaller = std::min(xDist, yDist);
1138 xDist = smaller;
1139 yDist = smaller;
1140 }
1141 secondRadius = xDist;
1142 aspectRatio = xDist / yDist;
1143 break;
1144 }
1145 case FarthestSide: {
1146 float xDist = std::max(secondPoint.x(), size.width() - secondPoint.x());
1147 float yDist = std::max(secondPoint.y(), size.height() - secondPoint.y());
1148 if (shape == Circle) {
1149 float larger = std::max(xDist, yDist);
1150 xDist = larger;
1151 yDist = larger;
1152 }
1153 secondRadius = xDist;
1154 aspectRatio = xDist / yDist;
1155 break;
1156 }
1157 case ClosestCorner: {
1158 FloatPoint corner;
1159 float distance = distanceToClosestCorner(secondPoint, size, corner);
1160 if (shape == Circle)
1161 secondRadius = distance;
1162 else {
1163 // If <shape> is ellipse, the gradient-shape has the same ratio of width to height
1164 // that it would if closest-side or farthest-side were specified, as appropriate.
1165 float xDist = std::min(secondPoint.x(), size.width() - secondPoint.x());
1166 float yDist = std::min(secondPoint.y(), size.height() - secondPoint.y());
1167
1168 secondRadius = horizontalEllipseRadius(corner - secondPoint, xDist / yDist);
1169 aspectRatio = xDist / yDist;
1170 }
1171 break;
1172 }
1173
1174 case FarthestCorner: {
1175 FloatPoint corner;
1176 float distance = distanceToFarthestCorner(secondPoint, size, corner);
1177 if (shape == Circle)
1178 secondRadius = distance;
1179 else {
1180 // If <shape> is ellipse, the gradient-shape has the same ratio of width to height
1181 // that it would if closest-side or farthest-side were specified, as appropriate.
1182 float xDist = std::max(secondPoint.x(), size.width() - secondPoint.x());
1183 float yDist = std::max(secondPoint.y(), size.height() - secondPoint.y());
1184
1185 secondRadius = horizontalEllipseRadius(corner - secondPoint, xDist / yDist);
1186 aspectRatio = xDist / yDist;
1187 }
1188 break;
1189 }
1190 }
1191 }
1192
1193 // computeStops() only uses maxExtent for repeating gradients.
1194 float maxExtent = 0;
1195 if (isRepeating()) {
1196 FloatPoint corner;
1197 maxExtent = distanceToFarthestCorner(secondPoint, size, corner);
1198 }
1199
1200 Gradient::RadialData data { firstPoint, secondPoint, firstRadius, secondRadius, aspectRatio };
1201 RadialGradientAdapter adapter { data };
1202 auto stops = computeStops(adapter, conversionData, renderer.style(), maxExtent);
1203
1204 auto gradient = Gradient::create(WTFMove(data));
1205 gradient->setSortedColorStops(WTFMove(stops));
1206 return gradient;
1207}
