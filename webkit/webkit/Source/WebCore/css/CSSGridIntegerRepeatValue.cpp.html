37String CSSGridIntegerRepeatValue::customCSSText() const
38{
39 return makeString("repeat(", repetitions(), ", ", CSSValueList::customCSSText(), ')');
40}
41
42bool CSSGridIntegerRepeatValue::equals(const CSSGridIntegerRepeatValue& other) const
43{
44 return m_repetitions == other.m_repetitions && CSSValueList::equals(other);
45}
