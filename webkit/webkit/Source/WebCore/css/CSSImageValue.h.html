42 String specifiedURLString;
43 URL resolvedURL;

46class CSSImageValue final : public CSSValue {
47public:
48 static Ref<CSSImageValue> create(ResolvedURL&&, LoadedFromOpaqueSource);
49 static Ref<CSSImageValue> create(URL&&, LoadedFromOpaqueSource);
50 static Ref<CSSImageValue> create(CachedImage&);
51 ~CSSImageValue();
52
53 bool isPending() const;
54 CachedImage* loadImage(CachedResourceLoader&, const ResourceLoaderOptions&);
55 CachedImage* cachedImage() const { return m_cachedImage ? m_cachedImage.value().get() : nullptr; }
56
57 const URL& imageURL() const { return m_location.resolvedURL; }
58
59 String customCSSText() const;
60
61 Ref<DeprecatedCSSOMValue> createDeprecatedCSSOMWrapper(CSSStyleDeclaration&) const;
62
63 bool traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const;
64
65 bool equals(const CSSImageValue&) const;
66
67 bool knownToBeOpaque(const RenderElement&) const;
68
69 void setInitiator(const AtomString& name) { m_initiatorName = name; }
70
71 Ref<CSSImageValue> valueWithStylesResolved(Style::BuilderState&);
72
73private:
74 CSSImageValue(ResolvedURL&&, LoadedFromOpaqueSource);
75 explicit CSSImageValue(CachedImage&);
76
77 URL reresolvedURL(const Document&) const;
78
79 ResolvedURL m_location;
80 std::optional<CachedResourceHandle<CachedImage>> m_cachedImage;
81 AtomString m_initiatorName;
82 LoadedFromOpaqueSource m_loadedFromOpaqueSource { LoadedFromOpaqueSource::No };
83 RefPtr<CSSImageValue> m_unresolvedValue;
84};
