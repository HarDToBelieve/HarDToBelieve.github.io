37template <typename CharacterType>
38static inline bool isCSSTokenizerIdentifier(const CharacterType* characters, unsigned length)
39{
40 const CharacterType* end = characters + length;
41
42 // -?
43 if (characters != end && characters[0] == '-')
44 ++characters;
45
46 // {nmstart}
47 if (characters == end || !isNameStartCodePoint(characters[0]))
48 return false;
49 ++characters;
50
51 // {nmchar}*
52 for (; characters != end; ++characters) {
53 if (!isNameCodePoint(characters[0]))
54 return false;
55 }
56
57 return true;
58}

60// "ident" from the CSS tokenizer, minus backslash-escape sequences
61static bool isCSSTokenizerIdentifier(const String& string)
62{
63 unsigned length = string.length();
64
65 if (!length)
66 return false;
67
68 if (string.is8Bit())
69 return isCSSTokenizerIdentifier(string.characters8(), length);
70 return isCSSTokenizerIdentifier(string.characters16(), length);
71}

73static void serializeCharacter(UChar32 c, StringBuilder& appendTo)
74{
75 appendTo.append('\\');
76 appendTo.appendCharacter(c);
77}
78
79static void serializeCharacterAsCodePoint(UChar32 c, StringBuilder& appendTo)
80{
81 appendTo.append('\\', hex(c, Lowercase), ' ');
82}
83
84void serializeIdentifier(const String& identifier, StringBuilder& appendTo, bool skipStartChecks)
85{
86 bool isFirst = !skipStartChecks;
87 bool isSecond = false;
88 bool isFirstCharHyphen = false;
89 unsigned index = 0;
90 while (index < identifier.length()) {
91 UChar32 c = identifier.characterStartingAt(index);
92 if (!c) {
93 // Check for lone surrogate which characterStartingAt does not return.
94 c = identifier[index];
95 }
96
97 index += U16_LENGTH(c);
98
99 if (!c)
100 appendTo.append(replacementCharacter);
101 else if (c <= 0x1f || c == 0x7f || (0x30 <= c && c <= 0x39 && (isFirst || (isSecond && isFirstCharHyphen))))
102 serializeCharacterAsCodePoint(c, appendTo);
103 else if (c == 0x2d && isFirst && index == identifier.length())
104 serializeCharacter(c, appendTo);
105 else if (0x80 <= c || c == 0x2d || c == 0x5f || (0x30 <= c && c <= 0x39) || (0x41 <= c && c <= 0x5a) || (0x61 <= c && c <= 0x7a))
106 appendTo.appendCharacter(c);
107 else
108 serializeCharacter(c, appendTo);
109
110 if (isFirst) {
111 isFirst = false;
112 isSecond = true;
113 isFirstCharHyphen = (c == 0x2d);
114 } else if (isSecond)
115 isSecond = false;
116 }
117}

119void serializeString(const String& string, StringBuilder& appendTo)
120{
121 appendTo.append('"');
122
123 unsigned index = 0;
124 while (index < string.length()) {
125 UChar32 c = string.characterStartingAt(index);
126 index += U16_LENGTH(c);
127
128 if (c <= 0x1f || c == 0x7f)
129 serializeCharacterAsCodePoint(c, appendTo);
130 else if (c == 0x22 || c == 0x5c)
131 serializeCharacter(c, appendTo);
132 else
133 appendTo.appendCharacter(c);
134 }
135
136 appendTo.append('"');
137}
138
139String serializeString(const String& string)
140{
141 StringBuilder builder;
142 serializeString(string, builder);
143 return builder.toString();
144}
145
146String serializeURL(const String& string)
147{
148 return "url(" + serializeString(string) + ")";
149}
150
151String serializeFontFamily(const String& string)
152{
153 return isCSSTokenizerIdentifier(string) ? string : serializeString(string);
154}
