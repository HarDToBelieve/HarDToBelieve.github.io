35CSSPageRule::CSSPageRule(StyleRulePage& pageRule, CSSStyleSheet* parent)
36 : CSSRule(parent)
37 , m_pageRule(pageRule)
38{
39}
40
41CSSPageRule::~CSSPageRule()
42{
43 if (m_propertiesCSSOMWrapper)
44 m_propertiesCSSOMWrapper->clearParentRule();
45}
46
47CSSStyleDeclaration& CSSPageRule::style()
48{
49 if (!m_propertiesCSSOMWrapper)
50 m_propertiesCSSOMWrapper = StyleRuleCSSStyleDeclaration::create(m_pageRule->mutableProperties(), *this);
51 return *m_propertiesCSSOMWrapper;
52}
53
54String CSSPageRule::selectorText() const
55{
56 if (auto* selector = m_pageRule->selector()) {
57 String pageSpecification = selector->selectorText();
58 if (!pageSpecification.isEmpty() && pageSpecification != starAtom())
59 return makeString("@page ", pageSpecification);
60 }
61 return "@page"_s;
62}
63
64void CSSPageRule::setSelectorText(const String& selectorText)
65{
66 CSSParser parser(parserContext());
67 auto selectorList = parser.parseSelector(selectorText);
68 if (!selectorList)
69 return;
70
71 CSSStyleSheet::RuleMutationScope mutationScope(this);
72
73 m_pageRule->wrapperAdoptSelectorList(WTFMove(*selectorList));
74}
75
76String CSSPageRule::cssText() const
77{
78 if (auto declarations = m_pageRule->properties().asText(); !declarations.isEmpty())
79 return makeString(selectorText(), " { ", declarations, " }");
80 return makeString(selectorText(), " { }");
81}
82
83void CSSPageRule::reattach(StyleRuleBase& rule)
84{
85 m_pageRule = downcast<StyleRulePage>(rule);
86 if (m_propertiesCSSOMWrapper)
87 m_propertiesCSSOMWrapper->reattach(m_pageRule.get().mutableProperties());
88}
