30#pragma once
31
32#include "CSSPropertyNames.h"
33#include "CSSValue.h"
34#include "CSSVariableReferenceValue.h"
35
36namespace WebCore {
37
38class CSSPendingSubstitutionValue : public CSSValue {
39public:
40 static Ref<CSSPendingSubstitutionValue> create(CSSPropertyID shorthandPropertyId, Ref<CSSVariableReferenceValue>&& shorthandValue, const URL& baseURL)
41 {
42 return adoptRef(*new CSSPendingSubstitutionValue(shorthandPropertyId, WTFMove(shorthandValue), baseURL));
43 }
44
45 CSSVariableReferenceValue& shorthandValue() const { return m_shorthandValue; }
46 CSSPropertyID shorthandPropertyId() const { return m_shorthandPropertyId; }
47 const URL& baseURL() const { return m_baseURL; }
48
49 bool equals(const CSSPendingSubstitutionValue& other) const { return m_shorthandValue.ptr() == other.m_shorthandValue.ptr(); }
50 static String customCSSText() { return emptyString(); }
51
52private:
53 CSSPendingSubstitutionValue(CSSPropertyID shorthandPropertyId, Ref<CSSVariableReferenceValue>&& shorthandValue, const URL& baseURL)
54 : CSSValue(PendingSubstitutionValueClass)
55 , m_shorthandPropertyId(shorthandPropertyId)
56 , m_shorthandValue(WTFMove(shorthandValue))
57 , m_baseURL(baseURL)
58 {
59 }
60
61 const CSSPropertyID m_shorthandPropertyId;
62 Ref<CSSVariableReferenceValue> m_shorthandValue;
63 // This is the base URL that should be used in resolution of this value when necessary, e.g. for the `url()` function.
64 const URL m_baseURL;
65};
66
67} // namespace WebCore
68
69SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSPendingSubstitutionValue, isPendingSubstitutionValue())
