60// Dimension calculations are imprecise, often resulting in values of e.g.
61// 44.99998. We need to round if we're really close to the next integer value.
62template<typename T> inline T roundForImpreciseConversion(double value)
63{
64 value += (value < 0) ? -0.01 : +0.01;
65 return ((value > std::numeric_limits<T>::max()) || (value < std::numeric_limits<T>::min())) ? 0 : static_cast<T>(value);
66}
67
68template<> inline float roundForImpreciseConversion(double value)
69{
70 double ceiledValue = ceil(value);
71 double proximityToNextInt = ceiledValue - value;
72 if (proximityToNextInt <= 0.01 && value > 0)
73 return static_cast<float>(ceiledValue);
74 if (proximityToNextInt >= 0.99 && value < 0)
75 return static_cast<float>(floor(value));
76 return static_cast<float>(value);
77}

264constexpr bool CSSPrimitiveValue::isFontRelativeLength(CSSUnitType type)
265{
266 return type == CSSUnitType::CSS_EMS
267 || type == CSSUnitType::CSS_EXS
268 || type == CSSUnitType::CSS_LHS
269 || type == CSSUnitType::CSS_RLHS
270 || type == CSSUnitType::CSS_REMS
271 || type == CSSUnitType::CSS_CHS
272 || type == CSSUnitType::CSS_QUIRKY_EMS;
273}
274
275constexpr bool CSSPrimitiveValue::isLength(CSSUnitType type)
276{
277 return (type >= CSSUnitType::CSS_EMS && type <= CSSUnitType::CSS_PC)
278 || type == CSSUnitType::CSS_REMS
279 || type == CSSUnitType::CSS_CHS
280 || type == CSSUnitType::CSS_Q
281 || type == CSSUnitType::CSS_LHS
282 || type == CSSUnitType::CSS_RLHS
283 || isViewportPercentageLength(type)
284 || type == CSSUnitType::CSS_QUIRKY_EMS;
285}

297template<typename T, CSSPrimitiveValue::TimeUnit timeUnit> inline T CSSPrimitiveValue::computeTime() const
298{
299 if (timeUnit == Seconds && primitiveType() == CSSUnitType::CSS_S)
300 return value<T>();
301 if (timeUnit == Seconds && primitiveType() == CSSUnitType::CSS_MS)
302 return value<T>() / 1000;
303 if (timeUnit == Milliseconds && primitiveType() == CSSUnitType::CSS_MS)
304 return value<T>();
305 if (timeUnit == Milliseconds && primitiveType() == CSSUnitType::CSS_S)
306 return value<T>() * 1000;
307 ASSERT_NOT_REACHED();
308 return 0;
309}

323inline double CSSPrimitiveValue::computeDegrees(CSSUnitType type, double angle)
324{
325 switch (type) {
326 case CSSUnitType::CSS_DEG:
327 return angle;
328 case CSSUnitType::CSS_RAD:
329 return rad2deg(angle);
330 case CSSUnitType::CSS_GRAD:
331 return grad2deg(angle);
332 case CSSUnitType::CSS_TURN:
333 return turn2deg(angle);
334 default:
335 ASSERT_NOT_REACHED();
336 return 0;
337 }
338}
