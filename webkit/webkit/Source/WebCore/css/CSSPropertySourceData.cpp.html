57CSSPropertySourceData::CSSPropertySourceData(const String& name, const String& value, bool important, bool disabled, bool parsedOk, const SourceRange& range)
58 : name(name)
59 , value(value)

62 , parsedOk(parsedOk)
63 , range(range)

65}

73 , range(other.range)

75}

77CSSPropertySourceData::CSSPropertySourceData()

81}
82
83String CSSPropertySourceData::toString() const
84{
85 if (!name && value == "e")
86 return String();
87 return makeString(name, ": ", value, important ? " !important" : "", ';');
88}
89
90unsigned CSSPropertySourceData::hash() const
91{
92 return StringHash::hash(name) + 3 * StringHash::hash(value) + 7 * important + 13 * parsedOk + 31;
93}
