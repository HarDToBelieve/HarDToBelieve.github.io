26#pragma once
27
28#include "CSSValue.h"
29#include <wtf/RefPtr.h>
30
31namespace WebCore {
32
33class CSSPrimitiveValue;
34
35class CSSReflectValue final : public CSSValue {
36public:
37 static Ref<CSSReflectValue> create(Ref<CSSPrimitiveValue>&& direction, Ref<CSSPrimitiveValue>&& offset, RefPtr<CSSValue>&& mask)
38 {
39 return adoptRef(*new CSSReflectValue(WTFMove(direction), WTFMove(offset), WTFMove(mask)));
40 }
41
42 CSSPrimitiveValue& direction() { return m_direction.get(); }
43 CSSPrimitiveValue& offset() { return m_offset.get(); }
44 const CSSPrimitiveValue& direction() const { return m_direction.get(); }
45 const CSSPrimitiveValue& offset() const { return m_offset.get(); }
46 CSSValue* mask() const { return m_mask.get(); }
47
48 String customCSSText() const;
49
50 bool equals(const CSSReflectValue&) const;
51
52private:
53 CSSReflectValue(Ref<CSSPrimitiveValue>&& direction, Ref<CSSPrimitiveValue>&& offset, RefPtr<CSSValue>&& mask)
54 : CSSValue(ReflectClass)
55 , m_direction(WTFMove(direction))
56 , m_offset(WTFMove(offset))
57 , m_mask(WTFMove(mask))
58 {
59 }
60
61 Ref<CSSPrimitiveValue> m_direction;
62 Ref<CSSPrimitiveValue> m_offset;
63 RefPtr<CSSValue> m_mask;
64};
65
66} // namespace WebCore

68SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSReflectValue, isReflectValue())
