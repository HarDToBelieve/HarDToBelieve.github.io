23#pragma once
24
25#include "ExceptionOr.h"
26#include <wtf/TypeCasts.h>
27
28namespace WebCore {
29
30class CSSStyleSheet;
31class StyleRuleBase;
32
33struct CSSParserContext;
34
35class CSSRule : public RefCounted<CSSRule> {
36public:
37 virtual ~CSSRule() = default;
38
39 // WebIDL enum
40 enum Type {
41 UNKNOWN_RULE,
42 STYLE_RULE,
43 CHARSET_RULE,
44 IMPORT_RULE,
45 MEDIA_RULE,
46 FONT_FACE_RULE,
47 PAGE_RULE,
48 KEYFRAMES_RULE,
49 KEYFRAME_RULE,
50 MARGIN_RULE,
51 NAMESPACE_RULE,
52 COUNTER_STYLE_RULE,
53 SUPPORTS_RULE
54 };
55
56 enum DeprecatedType {
57 WEBKIT_KEYFRAMES_RULE = KEYFRAMES_RULE,
58 WEBKIT_KEYFRAME_RULE = KEYFRAME_RULE
59 };
60
61 virtual Type type() const = 0;
62 virtual String cssText() const = 0;
63 virtual void reattach(StyleRuleBase&) = 0;
64
65 void setParentStyleSheet(CSSStyleSheet*);
66 void setParentRule(CSSRule*);
67 CSSStyleSheet* parentStyleSheet() const;
68 CSSRule* parentRule() const { return m_parentIsRule ? m_parentRule : nullptr; }
69
70 WEBCORE_EXPORT ExceptionOr<void> setCssText(const String&);
71
72protected:
73 explicit CSSRule(CSSStyleSheet*);
74
75 bool hasCachedSelectorText() const { return m_hasCachedSelectorText; }
76 void setHasCachedSelectorText(bool hasCachedSelectorText) const { m_hasCachedSelectorText = hasCachedSelectorText; }
77
78 const CSSParserContext& parserContext() const;
79
80private:
81 mutable unsigned char m_hasCachedSelectorText : 1;
82 unsigned char m_parentIsRule : 1;
83 union {
84 CSSRule* m_parentRule;
85 CSSStyleSheet* m_parentStyleSheet;
86 };
87};
88
89inline CSSRule::CSSRule(CSSStyleSheet* parent)
90 : m_hasCachedSelectorText(false)
91 , m_parentIsRule(false)
92 , m_parentStyleSheet(parent)
93{
94}
95
96inline void CSSRule::setParentStyleSheet(CSSStyleSheet* styleSheet)
97{
98 m_parentIsRule = false;
99 m_parentStyleSheet = styleSheet;
100}
101
102inline void CSSRule::setParentRule(CSSRule* rule)
103{
104 m_parentIsRule = true;
105 m_parentRule = rule;
106}
107
108inline CSSStyleSheet* CSSRule::parentStyleSheet() const
109{
110 if (m_parentIsRule)
111 return m_parentRule ? m_parentRule->parentStyleSheet() : nullptr;
112 return m_parentStyleSheet;
113}
114
115} // namespace WebCore

117#define SPECIALIZE_TYPE_TRAITS_CSS_RULE(ToValueTypeName, predicate) \
118SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::ToValueTypeName) \
119 static bool isType(const WebCore::CSSRule& rule) { return rule.type() == WebCore::predicate; } \
120SPECIALIZE_TYPE_TRAITS_END()
