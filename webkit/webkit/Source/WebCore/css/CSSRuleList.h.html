22#pragma once
23
24#include <wtf/RefPtr.h>
25#include <wtf/Vector.h>
26
27namespace WebCore {
28
29class CSSRule;
30class CSSStyleSheet;
31
32class CSSRuleList {
33 WTF_MAKE_NONCOPYABLE(CSSRuleList); WTF_MAKE_FAST_ALLOCATED;
34public:
35 virtual ~CSSRuleList();
36
37 virtual void ref() = 0;
38 virtual void deref() = 0;
39
40 virtual unsigned length() const = 0;
41 virtual CSSRule* item(unsigned index) const = 0;
42
43 virtual CSSStyleSheet* styleSheet() const = 0;
44
45protected:
46 CSSRuleList();
47};
48
49class StaticCSSRuleList final : public CSSRuleList {
50public:
51 static Ref<StaticCSSRuleList> create() { return adoptRef(*new StaticCSSRuleList); }
52
53 void ref() final { ++m_refCount; }
54 void deref() final;
55
56 Vector<RefPtr<CSSRule>>& rules() { return m_rules; }
57
58 CSSStyleSheet* styleSheet() const final { return nullptr; }
59
60private:
61 StaticCSSRuleList();
62 ~StaticCSSRuleList();
63
64 unsigned length() const final { return m_rules.size(); }
65 CSSRule* item(unsigned index) const final { return index < m_rules.size() ? m_rules[index].get() : nullptr; }
66
67 Vector<RefPtr<CSSRule>> m_rules;
68 unsigned m_refCount;
69};
70
71// The rule owns the live list.
72template <class Rule>
73class LiveCSSRuleList final : public CSSRuleList {
74public:
75 LiveCSSRuleList(Rule& rule)
76 : m_rule(rule)
77 {
78 }
79
80 void ref() final { m_rule.ref(); }
81 void deref() final { m_rule.deref(); }
82
83private:
84 unsigned length() const final { return m_rule.length(); }
85 CSSRule* item(unsigned index) const final { return m_rule.item(index); }
86 CSSStyleSheet* styleSheet() const final { return m_rule.parentStyleSheet(); }
87
88 Rule& m_rule;
89};
90
91} // namespace WebCore
