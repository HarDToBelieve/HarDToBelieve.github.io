54CSSSelector::CSSSelector(const QualifiedName& tagQName, bool tagIsForNamespaceRule)
55 : m_relation(DescendantSpace)
56 , m_match(Tag)
57 , m_pseudoType(0)
58 , m_isLastInSelectorList(false)
59 , m_isLastInTagHistory(true)
60 , m_hasRareData(false)
61 , m_hasNameWithCase(false)
62 , m_isForPage(false)
63 , m_tagIsForNamespaceRule(tagIsForNamespaceRule)
64 , m_caseInsensitiveAttributeValueMatching(false)
65#if !ASSERT_WITH_SECURITY_IMPLICATION_DISABLED
66 , m_destructorHasBeenCalled(false)
67#endif
68{
69 const AtomString& tagLocalName = tagQName.localName();
70 const AtomString tagLocalNameASCIILowercase = tagLocalName.convertToASCIILowercase();
71
72 if (tagLocalName == tagLocalNameASCIILowercase) {
73 m_data.m_tagQName = tagQName.impl();
74 m_data.m_tagQName->ref();
75 } else {
76 m_data.m_nameWithCase = adoptRef(new NameWithCase(tagQName, tagLocalNameASCIILowercase)).leakRef();
77 m_hasNameWithCase = true;
78 }
79}

81void CSSSelector::createRareData()
82{
83 ASSERT(match() != Tag);
84 ASSERT(!m_hasNameWithCase);
85 if (m_hasRareData)
86 return;
87 // Move the value to the rare data stucture.
88 AtomString value { adoptRef(m_data.m_value) };
89 m_data.m_rareData = &RareData::create(WTFMove(value)).leakRef();
90 m_hasRareData = true;
91}

95static unsigned selectorSpecificity(const CSSSelector& firstSimpleSelector)
96{
97 unsigned total = 0;
98 for (const CSSSelector* selector = &firstSimpleSelector; selector; selector = selector->tagHistory())
99 total = CSSSelector::addSpecificities(total, simpleSelectorSpecificityInternal(*selector));
100 return total;
101}

103static unsigned maxSpecificity(const CSSSelectorList& selectorList)
104{
105 unsigned maxSpecificity = 0;
106 for (const CSSSelector* subSelector = selectorList.first(); subSelector; subSelector = CSSSelectorList::next(subSelector))
107 maxSpecificity = std::max(maxSpecificity, selectorSpecificity(*subSelector));
108 return maxSpecificity;
109}

111static unsigned simpleSelectorSpecificityInternal(const CSSSelector& simpleSelector)
112{
113 ASSERT_WITH_MESSAGE(!simpleSelector.isForPage(), "At the time of this writing, page selectors are not treated as real selectors that are matched. The value computed here only account for real selectors.");
114
115 switch (simpleSelector.match()) {
116 case CSSSelector::Id:
117 return static_cast<unsigned>(SelectorSpecificityIncrement::ClassA);
118
119 case CSSSelector::PagePseudoClass:
120 break;
121 case CSSSelector::PseudoClass:
122 switch (simpleSelector.pseudoClassType()) {
123 case CSSSelector::PseudoClassIs:
124 case CSSSelector::PseudoClassMatches:
125 case CSSSelector::PseudoClassNot:
126 return maxSpecificity(*simpleSelector.selectorList());
127 case CSSSelector::PseudoClassWhere:
128 return 0;
129 case CSSSelector::PseudoClassNthChild:
130 case CSSSelector::PseudoClassNthLastChild:
131 case CSSSelector::PseudoClassHost:
132 return CSSSelector::addSpecificities(static_cast<unsigned>(SelectorSpecificityIncrement::ClassB), simpleSelector.selectorList() ? maxSpecificity(*simpleSelector.selectorList()) : 0);
133 default:
134 break;
135 }
136 return static_cast<unsigned>(SelectorSpecificityIncrement::ClassB);
137 case CSSSelector::Exact:
138 case CSSSelector::Class:
139 case CSSSelector::Set:
140 case CSSSelector::List:
141 case CSSSelector::Hyphen:
142 case CSSSelector::Contain:
143 case CSSSelector::Begin:
144 case CSSSelector::End:
145 return static_cast<unsigned>(SelectorSpecificityIncrement::ClassB);
146 case CSSSelector::Tag:
147 return (simpleSelector.tagQName().localName() != starAtom()) ? static_cast<unsigned>(SelectorSpecificityIncrement::ClassC) : 0;
148 case CSSSelector::PseudoElement:
149 return static_cast<unsigned>(SelectorSpecificityIncrement::ClassC);
150 case CSSSelector::Unknown:
151 return 0;
152 }
153 ASSERT_NOT_REACHED();
154 return 0;
155}

167unsigned CSSSelector::addSpecificities(unsigned a, unsigned b)
168{
169 unsigned total = a;
170
171 unsigned newIdValue = (b & idMask);
172 if (((total & idMask) + newIdValue) & ~idMask)
173 total |= idMask;
174 else
175 total += newIdValue;
176
177 unsigned newClassValue = (b & classMask);
178 if (((total & classMask) + newClassValue) & ~classMask)
179 total |= classMask;
180 else
181 total += newClassValue;
182
183 unsigned newElementValue = (b & elementMask);
184 if (((total & elementMask) + newElementValue) & ~elementMask)
185 total |= elementMask;
186 else
187 total += newElementValue;
188
189 return total;
190}

192unsigned CSSSelector::specificityForPage() const
193{
194 ASSERT(isForPage());
195
196 // See http://dev.w3.org/csswg/css3-page/#cascading-and-page-context
197 unsigned s = 0;
198
199 for (const CSSSelector* component = this; component; component = component->tagHistory()) {
200 switch (component->match()) {
201 case Tag:
202 s += tagQName().localName() == starAtom() ? 0 : 4;
203 break;
204 case PagePseudoClass:
205 switch (component->pagePseudoClassType()) {
206 case PagePseudoClassFirst:
207 s += 2;
208 break;
209 case PagePseudoClassLeft:
210 case PagePseudoClassRight:
211 s += 1;
212 break;
213 }
214 break;
215 default:
216 break;
217 }
218 }
219 return s;
220}

268CSSSelector::PseudoElementType CSSSelector::parsePseudoElementType(StringView name)
269{
270 if (name.isNull())
271 return PseudoElementUnknown;
272
273 auto type = parsePseudoElementString(name);
274 if (type == PseudoElementUnknown) {
275 if (name.startsWith("-webkit-"))
276 type = PseudoElementWebKitCustom;
277 }
278
279 if (type == PseudoElementHighlight && !RuntimeEnabledFeatures::sharedFeatures().highlightAPIEnabled())
280 return PseudoElementUnknown;
281
282 return type;
283}

285bool CSSSelector::operator==(const CSSSelector& other) const
286{
287 const CSSSelector* sel1 = this;
288 const CSSSelector* sel2 = &other;
289
290 while (sel1 && sel2) {
291 if (sel1->attribute() != sel2->attribute()
292 || sel1->relation() != sel2->relation()
293 || sel1->match() != sel2->match()
294 || sel1->value() != sel2->value()
295 || sel1->m_pseudoType != sel2->m_pseudoType
296 || sel1->argument() != sel2->argument()) {
297 return false;
298 }
299 if (sel1->match() == Tag) {
300 if (sel1->tagQName() != sel2->tagQName())
301 return false;
302 }
303 sel1 = sel1->tagHistory();
304 sel2 = sel2->tagHistory();
305 }
306
307 if (sel1 || sel2)
308 return false;
309
310 return true;
311}

313static void appendPseudoClassFunctionTail(StringBuilder& builder, const CSSSelector* selector)
314{
315 switch (selector->pseudoClassType()) {
316#if ENABLE(CSS_SELECTORS_LEVEL4)
317 case CSSSelector::PseudoClassDir:
318#endif
319 case CSSSelector::PseudoClassLang:
320 case CSSSelector::PseudoClassNthChild:
321 case CSSSelector::PseudoClassNthLastChild:
322 case CSSSelector::PseudoClassNthOfType:
323 case CSSSelector::PseudoClassNthLastOfType:
324#if ENABLE(CSS_SELECTORS_LEVEL4)
325 case CSSSelector::PseudoClassRole:
326#endif
327 builder.append(selector->argument());
328 builder.append(')');
329 break;
330 default:
331 break;
332 }

336static void appendLangArgumentList(StringBuilder& builder, const Vector<AtomString>& argumentList)
337{
338 for (unsigned i = 0, size = argumentList.size(); i < size; ++i)
339 builder.append('"', argumentList[i], '"', i != size - 1 ? ", " : "");
340}

342// http://dev.w3.org/csswg/css-syntax/#serializing-anb
343static void outputNthChildAnPlusB(const CSSSelector& selector, StringBuilder& builder)
344{
345 auto outputFirstTerm = [&builder] (int a) {
346 switch (a) {
347 case 1:
348 break;
349 case -1:
350 builder.append('-');
351 break;
352 default:
353 builder.append(a);
354 }
355 };
356
357 if (selector.argument() == nullAtom())
358 return;
359
360 int a = selector.nthA();
361 int b = selector.nthB();
362 if (a == 0 && b == 0)
363 builder.append('0');
364 else if (a == 0)
365 builder.append(b);
366 else if (b == 0) {
367 outputFirstTerm(a);
368 builder.append('n');
369 } else if (b < 0) {
370 outputFirstTerm(a);
371 builder.append('n', b);
372 } else {
373 outputFirstTerm(a);
374 builder.append("n+", b);
375 }
376}

378String CSSSelector::selectorText(const String& rightSide) const
379{
380 StringBuilder builder;
381
382 if (match() == CSSSelector::Tag && !m_tagIsForNamespaceRule) {
383 if (tagQName().prefix().isNull())
384 builder.append(tagQName().localName());
385 else
386 builder.append(tagQName().prefix().string(), '|', tagQName().localName());
387 }
388
389 const CSSSelector* cs = this;
390 while (true) {
391 if (cs->match() == CSSSelector::Id) {
392 builder.append('#');
393 serializeIdentifier(cs->serializingValue(), builder);
394 } else if (cs->match() == CSSSelector::Class) {
395 builder.append('.');
396 serializeIdentifier(cs->serializingValue(), builder);
397 } else if (cs->match() == CSSSelector::PseudoClass) {
398 switch (cs->pseudoClassType()) {
399#if ENABLE(FULLSCREEN_API)
400 case CSSSelector::PseudoClassAnimatingFullScreenTransition:
401 builder.append(":-webkit-animating-full-screen-transition");
402 break;
403#endif
404 case CSSSelector::PseudoClassAny: {
405 builder.append(":-webkit-any(");
406 cs->selectorList()->buildSelectorsText(builder);
407 builder.append(')');
408 break;
409 }
410 case CSSSelector::PseudoClassAnyLink:
411 builder.append(":any-link");
412 break;
413 case CSSSelector::PseudoClassAnyLinkDeprecated:
414 builder.append(":-webkit-any-link");
415 break;
416 case CSSSelector::PseudoClassAutofill:
417 builder.append(":autofill");
418 break;
419 case CSSSelector::PseudoClassAutofillStrongPassword:
420 builder.append(":-webkit-autofill-strong-password");
421 break;
422 case CSSSelector::PseudoClassAutofillStrongPasswordViewable:
423 builder.append(":-webkit-autofill-strong-password-viewable");
424 break;
425 case CSSSelector::PseudoClassDirectFocus:
426 builder.append(":-internal-direct-focus");
427 break;
428 case CSSSelector::PseudoClassDrag:
429 builder.append(":-webkit-drag");
430 break;
431 case CSSSelector::PseudoClassFullPageMedia:
432 builder.append(":-webkit-full-page-media");
433 break;
434#if ENABLE(FULLSCREEN_API)
435 case CSSSelector::PseudoClassFullScreen:

437 break;
438 case CSSSelector::PseudoClassFullScreenAncestor:
439 builder.append(":-webkit-full-screen-ancestor");
440 break;
441 case CSSSelector::PseudoClassFullScreenDocument:
442 builder.append(":-webkit-full-screen-document");
443 break;

445 builder.append(":-webkit-full-screen-controls-hidden");
446 break;
447#endif
448#if ENABLE(PICTURE_IN_PICTURE_API)
449 case CSSSelector::PseudoClassPictureInPicture:
450 builder.append(":picture-in-picture");
451 break;
452#endif
453 case CSSSelector::PseudoClassActive:
454 builder.append(":active");
455 break;
456 case CSSSelector::PseudoClassChecked:
457 builder.append(":checked");
458 break;
459 case CSSSelector::PseudoClassCornerPresent:
460 builder.append(":corner-present");
461 break;
462 case CSSSelector::PseudoClassDecrement:
463 builder.append(":decrement");
464 break;
465 case CSSSelector::PseudoClassDefault:
466 builder.append(":default");
467 break;
468#if ENABLE(CSS_SELECTORS_LEVEL4)
469 case CSSSelector::PseudoClassDir:
470 builder.append(":dir(");
471 appendPseudoClassFunctionTail(builder, cs);
472 break;
473#endif
474 case CSSSelector::PseudoClassDisabled:
475 builder.append(":disabled");
476 break;
477 case CSSSelector::PseudoClassDoubleButton:
478 builder.append(":double-button");
479 break;
480 case CSSSelector::PseudoClassEmpty:
481 builder.append(":empty");
482 break;
483 case CSSSelector::PseudoClassEnabled:
484 builder.append(":enabled");
485 break;
486 case CSSSelector::PseudoClassEnd:
487 builder.append(":end");
488 break;
489 case CSSSelector::PseudoClassFirstChild:
490 builder.append(":first-child");
491 break;
492 case CSSSelector::PseudoClassFirstOfType:
493 builder.append(":first-of-type");
494 break;
495 case CSSSelector::PseudoClassFocus:
496 builder.append(":focus");
497 break;
498 case CSSSelector::PseudoClassFocusVisible:
499 builder.append(":focus-visible");
500 break;
501 case CSSSelector::PseudoClassFocusWithin:
502 builder.append(":focus-within");

505 case CSSSelector::PseudoClassFuture:

509#if ENABLE(ATTACHMENT_ELEMENT)
510 case CSSSelector::PseudoClassHasAttachment:
511 builder.append(":has-attachment");
512 break;
513#endif
514 case CSSSelector::PseudoClassHorizontal:
515 builder.append(":horizontal");
516 break;
517 case CSSSelector::PseudoClassHover:

519 break;
520 case CSSSelector::PseudoClassInRange:

523 case CSSSelector::PseudoClassIncrement:

571 builder.append(" of ");
572 selectorList->buildSelectorsText(builder);
573 }
574 builder.append(')');
575 break;
576 case CSSSelector::PseudoClassNthLastOfType:
577 builder.append(":nth-last-of-type(");
578 appendPseudoClassFunctionTail(builder, cs);
579 break;
580 case CSSSelector::PseudoClassNthOfType:
581 builder.append(":nth-of-type(");
582 appendPseudoClassFunctionTail(builder, cs);
583 break;
584 case CSSSelector::PseudoClassOnlyChild:
585 builder.append(":only-child");
586 break;
587 case CSSSelector::PseudoClassOnlyOfType:
588 builder.append(":only-of-type");
589 break;
590 case CSSSelector::PseudoClassOptional:
591 builder.append(":optional");
592 break;
593 case CSSSelector::PseudoClassIs: {
594 builder.append(":is(");
595 cs->selectorList()->buildSelectorsText(builder);
596 builder.append(')');
597 break;
598 }
599 case CSSSelector::PseudoClassMatches: {
600 builder.append(":matches(");
601 cs->selectorList()->buildSelectorsText(builder);
602 builder.append(')');
603 break;
604 }
605 case CSSSelector::PseudoClassWhere: {
606 builder.append(":where(");
607 cs->selectorList()->buildSelectorsText(builder);
608 builder.append(')');
609 break;
610 }
611 case CSSSelector::PseudoClassPlaceholderShown:
612 builder.append(":placeholder-shown");
613 break;
614 case CSSSelector::PseudoClassOutOfRange:
615 builder.append(":out-of-range");
616 break;
617#if ENABLE(VIDEO)
618 case CSSSelector::PseudoClassPast:
619 builder.append(":past");
620 break;
621#endif
622 case CSSSelector::PseudoClassReadOnly:
623 builder.append(":read-only");
624 break;
625 case CSSSelector::PseudoClassReadWrite:
626 builder.append(":read-write");
627 break;
628 case CSSSelector::PseudoClassRequired:
629 builder.append(":required");
630 break;
631#if ENABLE(CSS_SELECTORS_LEVEL4)
632 case CSSSelector::PseudoClassRole:
633 builder.append(":role(");
634 appendPseudoClassFunctionTail(builder, cs);
635 break;
636#endif
637 case CSSSelector::PseudoClassRoot:
638 builder.append(":root");
639 break;
640 case CSSSelector::PseudoClassScope:
641 builder.append(":scope");
642 break;
643 case CSSSelector::PseudoClassSingleButton:
644 builder.append(":single-button");
645 break;
646 case CSSSelector::PseudoClassStart:
647 builder.append(":start");
648 break;
649 case CSSSelector::PseudoClassTarget:
650 builder.append(":target");
651 break;
652 case CSSSelector::PseudoClassValid:
653 builder.append(":valid");
654 break;
655 case CSSSelector::PseudoClassVertical:
656 builder.append(":vertical");
657 break;
658 case CSSSelector::PseudoClassVisited:
659 builder.append(":visited");
660 break;
661 case CSSSelector::PseudoClassWindowInactive:
662 builder.append(":window-inactive");
663 break;
664 case CSSSelector::PseudoClassHost:
665 builder.append(":host");
666 if (auto* selectorList = cs->selectorList()) {
667 builder.append('(');
668 selectorList->buildSelectorsText(builder);
669 builder.append(')');
670 }
671 break;
672 case CSSSelector::PseudoClassDefined:
673 builder.append(":defined");
674 break;
675 case CSSSelector::PseudoClassUnknown:
676 ASSERT_NOT_REACHED();
677 }
678 } else if (cs->match() == CSSSelector::PseudoElement) {
679 switch (cs->pseudoElementType()) {
680 case CSSSelector::PseudoElementSlotted:
681 builder.append("::slotted(");
682 cs->selectorList()->buildSelectorsText(builder);
683 builder.append(')');
684 break;
685 case CSSSelector::PseudoElementPart: {
686 builder.append("::part(");
687 bool isFirst = true;
688 for (auto& partName : *cs->argumentList()) {
689 if (!isFirst)
690 builder.append(' ');
691 isFirst = false;
692 builder.append(partName);
693 }
694 builder.append(')');
695 break;
696 }
697 case CSSSelector::PseudoElementWebKitCustomLegacyPrefixed:
698 if (cs->value() == "placeholder")
699 builder.append("::-webkit-input-placeholder");
700 if (cs->value() == "file-selector-button")
701 builder.append("::-webkit-file-upload-button");
702 break;
703#if ENABLE(VIDEO)
704 case CSSSelector::PseudoElementCue: {
705 if (auto* selectorList = cs->selectorList()) {
706 builder.append("::cue(");
707 selectorList->buildSelectorsText(builder);
708 builder.append(')');
709 } else
710 builder.append("::cue");
711 break;
712 }
713#endif
714 default:
715 builder.append("::", cs->serializingValue());
716 }
717 } else if (cs->isAttributeSelector()) {
718 builder.append('[');
719 if (auto& prefix = cs->attribute().prefix(); !prefix.isEmpty())
720 builder.append(prefix, '|');
721 builder.append(cs->attribute().localName());
722 switch (cs->match()) {
723 case CSSSelector::Exact:
724 builder.append('=');
725 break;
726 case CSSSelector::Set:
727 // set has no operator or value, just the attrName
728 builder.append(']');
729 break;
730 case CSSSelector::List:
731 builder.append("~=");
732 break;
733 case CSSSelector::Hyphen:
734 builder.append("|=");
735 break;
736 case CSSSelector::Begin:
737 builder.append("^=");
738 break;
739 case CSSSelector::End:
740 builder.append("$=");
741 break;
742 case CSSSelector::Contain:
743 builder.append("*=");
744 break;
745 default:
746 break;
747 }
748 if (cs->match() != CSSSelector::Set) {
749 serializeString(cs->serializingValue(), builder);
750 if (cs->attributeValueMatchingIsCaseInsensitive())
751 builder.append(" i]");
752 else
753 builder.append(']');
754 }
755 } else if (cs->match() == CSSSelector::PagePseudoClass) {
756 switch (cs->pagePseudoClassType()) {
757 case PagePseudoClassFirst:
758 builder.append(":first");
759 break;
760 case PagePseudoClassLeft:
761 builder.append(":left");
762 break;
763 case PagePseudoClassRight:
764 builder.append(":right");
765 break;
766 }
767 }
768
769 if (cs->relation() != CSSSelector::Subselector || !cs->tagHistory())
770 break;
771 cs = cs->tagHistory();
772 }

774 if (const CSSSelector* tagHistory = cs->tagHistory()) {
775 switch (cs->relation()) {
776 case CSSSelector::DescendantSpace:
777 return tagHistory->selectorText(" " + builder.toString() + rightSide);
778 case CSSSelector::Child:
779 return tagHistory->selectorText(" > " + builder.toString() + rightSide);
780 case CSSSelector::DirectAdjacent:
781 return tagHistory->selectorText(" + " + builder.toString() + rightSide);
782 case CSSSelector::IndirectAdjacent:
783 return tagHistory->selectorText(" ~ " + builder.toString() + rightSide);
784 case CSSSelector::Subselector:
785 ASSERT_NOT_REACHED();
786#if !ASSERT_ENABLED
787 FALLTHROUGH;
788#endif
789 case CSSSelector::ShadowDescendant:
790 builder.append(rightSide);
791 return tagHistory->selectorText(builder.toString());
792 }
793 }
794 builder.append(rightSide);
795 return builder.toString();
796}

861// a helper function for checking nth-arguments
862bool CSSSelector::RareData::matchNth(int count)
863{
864 if (!m_a)
865 return count == m_b;
866 else if (m_a > 0) {
867 if (count < m_b)
868 return false;
869 return (count - m_b) % m_a == 0;
870 } else {
871 if (count > m_b)
872 return false;
873 return (m_b - count) % (-m_a) == 0;
874 }
875}
