59static bool isAcceptableCSSStyleSheetParent(Node* parentNode)
60{
61 // Only these nodes can be parents of StyleSheets, and they need to call clearOwnerNode() when moved out of document.
62 return !parentNode
63 || parentNode->isDocumentNode()
64 || is<HTMLLinkElement>(*parentNode)
65 || is<HTMLStyleElement>(*parentNode)
66 || is<SVGStyleElement>(*parentNode)
67 || parentNode->nodeType() == Node::PROCESSING_INSTRUCTION_NODE;
68}

104CSSStyleSheet::~CSSStyleSheet()
105{
106 // For style rules outside the document, .parentStyleSheet can become null even if the style rule
107 // is still observable from JavaScript. This matches the behavior of .parentNode for nodes, but
108 // it's not ideal because it makes the CSSOM's behavior depend on the timing of garbage collection.
109 for (unsigned i = 0; i < m_childRuleCSSOMWrappers.size(); ++i) {
110 if (m_childRuleCSSOMWrappers[i])
111 m_childRuleCSSOMWrappers[i]->setParentStyleSheet(0);
112 }
113 if (m_mediaCSSOMWrapper)
114 m_mediaCSSOMWrapper->clearParentStyleSheet();
115
116 m_contents->unregisterClient(this);
117}

119CSSStyleSheet::WhetherContentsWereClonedForMutation CSSStyleSheet::willMutateRules()
120{
121 // If we are the only client it is safe to mutate.
122 if (m_contents->hasOneClient() && !m_contents->isInMemoryCache()) {
123 m_contents->setMutable();
124 return ContentsWereNotClonedForMutation;
125 }
126 // Only cacheable stylesheets should have multiple clients.
127 ASSERT(m_contents->isCacheable());
128
129 // Copy-on-write.
130 m_contents->unregisterClient(this);
131 m_contents = m_contents->copy();
132 m_contents->registerClient(this);
133
134 m_contents->setMutable();
135
136 // Any existing CSSOM wrappers need to be connected to the copied child rules.
137 reattachChildRuleCSSOMWrappers();
138
139 return ContentsWereClonedForMutation;
140}

149void CSSStyleSheet::didMutateRules(RuleMutationType mutationType, WhetherContentsWereClonedForMutation contentsWereClonedForMutation, StyleRuleKeyframes* insertedKeyframesRule)
150{
151 ASSERT(m_contents->isMutable());
152 ASSERT(m_contents->hasOneClient());
153
154 auto* scope = styleScope();
155 if (!scope)
156 return;
157
158 if (mutationType == RuleInsertion && !contentsWereClonedForMutation && !scope->activeStyleSheetsContains(this)) {
159 if (insertedKeyframesRule) {
160 if (auto* resolver = scope->resolverIfExists())
161 resolver->addKeyframeStyle(*insertedKeyframesRule);
162 return;
163 }
164 scope->didChangeActiveStyleSheetCandidates();
165 return;
166 }
167
168 scope->didChangeStyleSheetContents();
169
170 m_mutatedRules = true;
171}

173void CSSStyleSheet::didMutate()
174{

177 return;
178 scope->didChangeStyleSheetContents();

218CSSRule* CSSStyleSheet::item(unsigned index)
219{
220 unsigned ruleCount = length();
221 if (index >= ruleCount)
222 return nullptr;
223
224 ASSERT(m_childRuleCSSOMWrappers.isEmpty() || m_childRuleCSSOMWrappers.size() == ruleCount);
225 if (m_childRuleCSSOMWrappers.size() < ruleCount)
226 m_childRuleCSSOMWrappers.grow(ruleCount);
227
228 RefPtr<CSSRule>& cssRule = m_childRuleCSSOMWrappers[index];
229 if (!cssRule)
230 cssRule = m_contents->ruleAt(index)->createCSSOMWrapper(this);
231 return cssRule.get();
232}

234bool CSSStyleSheet::canAccessRules() const
235{
236 if (m_isOriginClean)
237 return m_isOriginClean.value();
238
239 URL baseURL = m_contents->baseURL();
240 if (baseURL.isEmpty())
241 return true;
242 Document* document = ownerDocument();
243 if (!document)
244 return true;
245 return document->securityOrigin().canRequest(baseURL);
246}

248ExceptionOr<unsigned> CSSStyleSheet::insertRule(const String& ruleString, unsigned index)
249{
250 ASSERT(m_childRuleCSSOMWrappers.isEmpty() || m_childRuleCSSOMWrappers.size() == m_contents->ruleCount());
251
252 if (index > length())
253 return Exception { IndexSizeError };
254 RefPtr<StyleRuleBase> rule = CSSParser::parseRule(m_contents.get().parserContext(), m_contents.ptr(), ruleString);
255
256 if (!rule)
257 return Exception { SyntaxError };
258
259 RuleMutationScope mutationScope(this, RuleInsertion, is<StyleRuleKeyframes>(*rule) ? downcast<StyleRuleKeyframes>(rule.get()) : nullptr);
260
261 bool success = m_contents.get().wrapperInsertRule(rule.releaseNonNull(), index);
262 if (!success)
263 return Exception { HierarchyRequestError };
264 if (!m_childRuleCSSOMWrappers.isEmpty())
265 m_childRuleCSSOMWrappers.insert(index, RefPtr<CSSRule>());
266
267 return index;
268}
269
270ExceptionOr<void> CSSStyleSheet::deleteRule(unsigned index)
271{
272 ASSERT(m_childRuleCSSOMWrappers.isEmpty() || m_childRuleCSSOMWrappers.size() == m_contents->ruleCount());
273
274 if (index >= length())
275 return Exception { IndexSizeError };
276 RuleMutationScope mutationScope(this);
277
278 m_contents->wrapperDeleteRule(index);
279
280 if (!m_childRuleCSSOMWrappers.isEmpty()) {
281 if (m_childRuleCSSOMWrappers[index])
282 m_childRuleCSSOMWrappers[index]->setParentStyleSheet(nullptr);
283 m_childRuleCSSOMWrappers.remove(index);
284 }
285
286 return { };
287}

289ExceptionOr<int> CSSStyleSheet::addRule(const String& selector, const String& style, std::optional<unsigned> index)
290{
291 auto text = makeString(selector, " { ", style, !style.isEmpty() ? " " : "", '}');
292 auto insertRuleResult = insertRule(text, index.value_or(length()));
293 if (insertRuleResult.hasException())
294 return insertRuleResult.releaseException();
295 // As per Microsoft documentation, always return -1.
296 return -1;
297}

299ExceptionOr<Ref<CSSRuleList>> CSSStyleSheet::cssRulesForBindings()
300{
301 auto cssRules = this->cssRules();
302 if (!cssRules)
303 return Exception { SecurityError, "Not allowed to access cross-origin stylesheet"_s };

307RefPtr<CSSRuleList> CSSStyleSheet::cssRules()
308{
309 if (!canAccessRules())
310 return nullptr;
311 if (!m_ruleListCSSOMWrapper)
312 m_ruleListCSSOMWrapper = makeUnique<StyleSheetCSSRuleList>(this);
313 return m_ruleListCSSOMWrapper.get();

346{
