39float CSSToLengthConversionData::zoom() const
40{
41 if (!m_zoom)
42 return m_style ? m_style->effectiveZoom() : 1;
43 return *m_zoom;
44}
45
46double CSSToLengthConversionData::viewportWidthFactor() const
47{
48 if (m_viewportDependencyDetectionStyle)
49 m_viewportDependencyDetectionStyle->setHasViewportUnits();
50
51 if (!m_renderView)
52 return 0;
53
54 return m_renderView->viewportSizeForCSSViewportUnits().width() / 100.0;
55}
56
57double CSSToLengthConversionData::viewportHeightFactor() const
58{
59 if (m_viewportDependencyDetectionStyle)
60 m_viewportDependencyDetectionStyle->setHasViewportUnits();
61
62 if (!m_renderView)
63 return 0;
64
65 return m_renderView->viewportSizeForCSSViewportUnits().height() / 100.0;
66}
67
68double CSSToLengthConversionData::viewportMinFactor() const
69{
70 if (m_viewportDependencyDetectionStyle)
71 m_viewportDependencyDetectionStyle->setHasViewportUnits();
72
73 if (!m_renderView)
74 return 0;
75
76 IntSize viewportSizeForCSSViewportUnits = m_renderView->viewportSizeForCSSViewportUnits();
77 return std::min(viewportSizeForCSSViewportUnits.width(), viewportSizeForCSSViewportUnits.height()) / 100.0;
78}
79
80double CSSToLengthConversionData::viewportMaxFactor() const
81{
82 if (m_viewportDependencyDetectionStyle)
83 m_viewportDependencyDetectionStyle->setHasViewportUnits();
84
85 if (!m_renderView)
86 return 0;
87
88 IntSize viewportSizeForCSSViewportUnits = m_renderView->viewportSizeForCSSViewportUnits();
89 return std::max(viewportSizeForCSSViewportUnits.width(), viewportSizeForCSSViewportUnits.height()) / 100.0;
90}
