31#pragma once
32
33#include "CSSPropertyNames.h"
34#include <optional>
35#include <wtf/Assertions.h>
36
37namespace WebCore {
38
39class RenderStyle;
40class RenderView;
41
42class CSSToLengthConversionData {
43public:
44 CSSToLengthConversionData(const RenderStyle* style, const RenderStyle* rootStyle, const RenderStyle* parentStyle, const RenderView* renderView, float zoom, std::optional<CSSPropertyID> propertyToCompute = std::nullopt, RenderStyle* viewportDependencyDetectionStyle = nullptr)
45 : m_style(style)
46 , m_rootStyle(rootStyle)
47 , m_parentStyle(parentStyle)
48 , m_viewportDependencyDetectionStyle(viewportDependencyDetectionStyle ? viewportDependencyDetectionStyle : const_cast<RenderStyle*>(style))
49 , m_renderView(renderView)
50 , m_zoom(zoom)
51 , m_propertyToCompute(propertyToCompute)
52 {
53 ASSERT(zoom > 0);
54 }
55
56 CSSToLengthConversionData(const RenderStyle* style, const RenderStyle* rootStyle, const RenderStyle* parentStyle, const RenderView* renderView, std::optional<CSSPropertyID> propertyToCompute = std::nullopt)
57 : m_style(style)
58 , m_rootStyle(rootStyle)
59 , m_parentStyle(parentStyle)
60 , m_viewportDependencyDetectionStyle(const_cast<RenderStyle*>(style))
61 , m_renderView(renderView)
62 , m_propertyToCompute(propertyToCompute)
63 {
64 }
65
66 CSSToLengthConversionData() = default;
67
68 const RenderStyle* style() const { return m_style; }
69 const RenderStyle* rootStyle() const { return m_rootStyle; }
70 const RenderStyle* parentStyle() const { return m_parentStyle; }
71 float zoom() const;
72 bool computingFontSize() const { return m_propertyToCompute == CSSPropertyFontSize; }
73 bool computingLineHeight() const { return m_propertyToCompute == CSSPropertyLineHeight; }
74 CSSPropertyID propertyToCompute() const { return m_propertyToCompute.value_or(CSSPropertyInvalid); }
75 const RenderView* renderView() const { return m_renderView; }
76
77 double viewportWidthFactor() const;
78 double viewportHeightFactor() const;
79 double viewportMinFactor() const;
80 double viewportMaxFactor() const;
81
82 CSSToLengthConversionData copyWithAdjustedZoom(float newZoom) const
83 {
84 return CSSToLengthConversionData(m_style, m_rootStyle, m_parentStyle, m_renderView, newZoom, m_propertyToCompute);
85 }
86
87 CSSToLengthConversionData copyWithAdjustedZoomAndPropertyToCompute(float zoom, std::optional<CSSPropertyID> propertyToCompute) const
88 {
89 return CSSToLengthConversionData(m_style, m_rootStyle, m_parentStyle, m_renderView, zoom, propertyToCompute);
90 }
91
92private:
93 const RenderStyle* m_style { nullptr };
94 const RenderStyle* m_rootStyle { nullptr };
95 const RenderStyle* m_parentStyle { nullptr };
96 RenderStyle* m_viewportDependencyDetectionStyle { nullptr };
97 const RenderView* m_renderView { nullptr };
98 std::optional<float> m_zoom;
99 std::optional<CSSPropertyID> m_propertyToCompute;
100};
101
102} // namespace WebCore
