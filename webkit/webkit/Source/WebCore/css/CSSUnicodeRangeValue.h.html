26#pragma once
27
28#include "CSSValue.h"
29
30namespace WebCore {
31
32class CSSUnicodeRangeValue final : public CSSValue {
33public:
34 static Ref<CSSUnicodeRangeValue> create(UChar32 from, UChar32 to)
35 {
36 return adoptRef(*new CSSUnicodeRangeValue(from, to));
37 }
38
39 UChar32 from() const { return m_from; }
40 UChar32 to() const { return m_to; }
41
42 String customCSSText() const;
43
44 bool equals(const CSSUnicodeRangeValue&) const;
45
46private:
47 CSSUnicodeRangeValue(UChar32 from, UChar32 to)
48 : CSSValue(UnicodeRangeClass)
49 , m_from(from)
50 , m_to(to)
51 {
52 }
53
54 UChar32 m_from;
55 UChar32 m_to;
56};
57
58} // namespace WebCore

60SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSUnicodeRangeValue, isUnicodeRangeValue())
