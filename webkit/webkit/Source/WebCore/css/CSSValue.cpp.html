91CSSValue::Type CSSValue::cssValueType() const
92{
93 if (isInheritedValue())
94 return CSS_INHERIT;
95 if (isPrimitiveValue())
96 return CSS_PRIMITIVE_VALUE;
97 if (isValueList())
98 return CSS_VALUE_LIST;
99 if (isInitialValue())
100 return CSS_INITIAL;
101 if (isUnsetValue())
102 return CSS_UNSET;
103 if (isRevertValue())
104 return CSS_REVERT;
105 return CSS_CUSTOM;
106}

108bool CSSValue::traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const
109{
110 if (is<CSSValueList>(*this))
111 return downcast<CSSValueList>(*this).traverseSubresources(handler);
112 if (is<CSSFontFaceSrcValue>(*this))
113 return downcast<CSSFontFaceSrcValue>(*this).traverseSubresources(handler);
114 if (is<CSSImageValue>(*this))
115 return downcast<CSSImageValue>(*this).traverseSubresources(handler);
116 if (is<CSSCrossfadeValue>(*this))
117 return downcast<CSSCrossfadeValue>(*this).traverseSubresources(handler);
118 if (is<CSSFilterImageValue>(*this))
119 return downcast<CSSFilterImageValue>(*this).traverseSubresources(handler);
120 if (is<CSSImageSetValue>(*this))
121 return downcast<CSSImageSetValue>(*this).traverseSubresources(handler);
122 return false;
123}

341ASCIILiteral CSSValue::separatorCSSText() const
342{
343 switch (m_valueSeparator) {
344 case SpaceSeparator:
345 return " "_s;
346 case CommaSeparator:
347 return ", "_s;
348 case SlashSeparator:
349 return " / "_s;
350 default:
351 ASSERT_NOT_REACHED();
352 }
353 return " "_s;
354}

494Ref<DeprecatedCSSOMValue> CSSValue::createDeprecatedCSSOMWrapper(CSSStyleDeclaration& styleDeclaration) const
495{
496 if (isImageValue() || isCursorImageValue())
497 return downcast<CSSImageValue>(this)->createDeprecatedCSSOMWrapper(styleDeclaration);
498 if (isPrimitiveValue())
499 return DeprecatedCSSOMPrimitiveValue::create(downcast<CSSPrimitiveValue>(*this), styleDeclaration);
500 if (isValueList())
501 return DeprecatedCSSOMValueList::create(downcast<CSSValueList>(*this), styleDeclaration);
502 return DeprecatedCSSOMComplexValue::create(*this, styleDeclaration);
503}

505bool CSSValue::treatAsInheritedValue(CSSPropertyID propertyID) const
506{
507 return classType() == InheritedClass || (classType() == UnsetClass && CSSProperty::isInheritedProperty(propertyID));
508}
509
510bool CSSValue::treatAsInitialValue(CSSPropertyID propertyID) const
511{
512 return classType() == InitialClass || (classType() == UnsetClass && !CSSProperty::isInheritedProperty(propertyID));
513}
