33String CSSValuePair::customCSSText() const
34{
35 String first = this->first()->cssText();
36 String second = this->second()->cssText();
37 if (m_encoding == IdenticalValueEncoding::Coalesce && first == second)
38 return first;
39 return makeString(first, separatorCSSText(), second);
40}
41
42bool CSSValuePair::equals(const CSSValuePair &other) const
43{
44 return m_valueSeparator == other.m_valueSeparator && m_first.get().equals(other.m_first) && m_second.get().equals(other.m_second);
45}
