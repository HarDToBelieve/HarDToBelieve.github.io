26#pragma once
27
28#include "CSSValue.h"
29
30namespace WebCore {
31
32/**
33 * Represents a pair of CSS values.
34 */
35class CSSValuePair : public CSSValue {
36public:
37 enum class IdenticalValueEncoding : uint8_t {
38 DoNotCoalesce,
39 Coalesce
40 };
41
42 static Ref<CSSValuePair> create(Ref<CSSValue>&& first, Ref<CSSValue>&& second, ValueSeparator separator)
43 {
44 return adoptRef(*new CSSValuePair(WTFMove(first), WTFMove(second), separator));
45 }
46 static Ref<CSSValuePair> create(Ref<CSSValue>&& first, Ref<CSSValue>&& second, ValueSeparator separator, IdenticalValueEncoding encoding)
47 {
48 return adoptRef(*new CSSValuePair(WTFMove(first), WTFMove(second), separator, encoding));
49 }
50
51 Ref<CSSValue> first() const { return m_first; }
52 Ref<CSSValue> second() const { return m_second; }
53
54 String customCSSText() const;
55 bool equals(const CSSValuePair& other) const;
56
57private:
58 explicit CSSValuePair(Ref<CSSValue>&& first, Ref<CSSValue>&& second, ValueSeparator separator)
59 : CSSValue(ValuePairClass)
60 , m_first(WTFMove(first))
61 , m_second(WTFMove(second))
62 {
63 m_valueSeparator = separator;
64 }
65
66 explicit CSSValuePair(Ref<CSSValue>&& first, Ref<CSSValue>&& second, ValueSeparator separator, IdenticalValueEncoding encoding)
67 : CSSValue(ValuePairClass)
68 , m_encoding(encoding)
69 , m_first(WTFMove(first))
70 , m_second(WTFMove(second))
71 {
72 m_valueSeparator = separator;
73 }
74
75 IdenticalValueEncoding m_encoding { IdenticalValueEncoding::Coalesce };
76 Ref<CSSValue> m_first;
77 Ref<CSSValue> m_second;
78};
79
80} // namespace WebCore
81
82SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSValuePair, isValuePair())
