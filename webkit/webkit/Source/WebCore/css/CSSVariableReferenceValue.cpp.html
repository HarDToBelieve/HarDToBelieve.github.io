66static bool resolveVariableFallback(CSSParserTokenRange range, Vector<CSSParserToken>& result, Style::BuilderState& builderState)
67{
68 if (range.atEnd())
69 return false;
70 ASSERT(range.peek().type() == CommaToken);
71 range.consumeIncludingWhitespace();
72 return resolveTokenRange(range, result, builderState);
73}
74
75static bool resolveVariableReference(CSSParserTokenRange range, CSSValueID functionId, Vector<CSSParserToken>& result, Style::BuilderState& builderState)
76{
77 ASSERT(functionId == CSSValueVar || functionId == CSSValueEnv);
78
79 auto& registeredProperties = builderState.document().getCSSRegisteredCustomPropertySet();
80 auto& style = builderState.style();
81
82 range.consumeWhitespace();
83 ASSERT(range.peek().type() == IdentToken);
84 String variableName = range.consumeIncludingWhitespace().value().toString();
85 ASSERT(range.atEnd() || (range.peek().type() == CommaToken));
86
87 // Apply this variable first, in case it is still unresolved
88 builderState.builder().applyCustomProperty(variableName);
89
90 // Apply fallback to detect cycles
91 Vector<CSSParserToken> fallbackResult;
92 bool fallbackReturn = resolveVariableFallback(CSSParserTokenRange(range), fallbackResult, builderState);
93
94
95 auto* property = functionId == CSSValueVar
96 ? style.getCustomProperty(variableName)
97 : builderState.document().constantProperties().values().get(variableName);
98 if (!property || property->isUnset()) {
99 auto* registered = registeredProperties.get(variableName);
100 if (registered && registered->initialValue())
101 property = registered->initialValue();
102 }
103
104 if (!property || property->isInvalid()) {
105 if (fallbackResult.size() > CSSVariableReferenceValue::maxSubstitutionTokens)
106 return false;
107
108 if (fallbackReturn)
109 result.appendVector(fallbackResult);
110 return fallbackReturn;
111 }
112
113 ASSERT(property->isResolved());
114 if (property->tokens().size() > CSSVariableReferenceValue::maxSubstitutionTokens)
115 return false;
116
117 result.appendVector(property->tokens());
118 return true;
119}
120
121static bool resolveTokenRange(CSSParserTokenRange range, Vector<CSSParserToken>& result, Style::BuilderState& builderState)
122{
123 bool success = true;
124 while (!range.atEnd()) {
125 auto functionId = range.peek().functionId();
126 if (functionId == CSSValueVar || functionId == CSSValueEnv)
127 success &= resolveVariableReference(range.consumeBlock(), functionId, result, builderState);
128 else
129 result.append(range.consume());
130 }
131 return success;
132}
133
134RefPtr<CSSVariableData> CSSVariableReferenceValue::resolveVariableReferences(Style::BuilderState& builderState) const
135{
136 Vector<CSSParserToken> resolvedTokens;
137 if (!resolveTokenRange(m_data->tokenRange(), resolvedTokens, builderState))
138 return nullptr;
139
140 return CSSVariableData::create(resolvedTokens);
141}
