44DOMCSSPaintWorklet* DOMCSSPaintWorklet::from(DOMCSSNamespace& css)
45{
46 auto* supplement = static_cast<DOMCSSPaintWorklet*>(Supplement<DOMCSSNamespace>::from(&css, supplementName()));
47 if (!supplement) {
48 auto newSupplement = makeUnique<DOMCSSPaintWorklet>(css);
49 supplement = newSupplement.get();
50 provideTo(&css, supplementName(), WTFMove(newSupplement));
51 }
52 return supplement;
53}

55const char* DOMCSSPaintWorklet::supplementName()
56{
57 return "DOMCSSPaintWorklet";
58}

60// FIXME: Get rid of this override and rely on the standard-compliant Worklet::addModule() instead.
61void PaintWorklet::addModule(const String& moduleURL, WorkletOptions&&, DOMPromiseDeferred<void>&& promise)
62{
63 auto* document = this->document();
64 if (!document) {
65 promise.reject(Exception { InvalidStateError, "This frame is detached"_s });
66 return;
67 }
68
69 // FIXME: We should download the source from the URL
70 // https://bugs.webkit.org/show_bug.cgi?id=191136
71 auto maybeContext = PaintWorkletGlobalScope::tryCreate(*document, ScriptSourceCode(moduleURL));
72 if (UNLIKELY(!maybeContext)) {
73 promise.reject(Exception { OutOfMemoryError });
74 return;
75 }
76 auto context = maybeContext.releaseNonNull();
77 context->evaluate();
78
79 Locker locker { context->paintDefinitionLock() };
80 for (auto& name : context->paintDefinitionMap().keys())
81 document->setPaintWorkletGlobalScopeForName(name, makeRef(context.get()));
82 promise.resolve();
83}
