38class Document;
39class Worklet;

42class PaintWorklet final : public Worklet {
43public:
44 static Ref<PaintWorklet> create(Document& document)
45 {
46 auto worklet = adoptRef(*new PaintWorklet(document));
47 worklet->suspendIfNeeded();
48 return worklet;
49 }
50
51 // Worklet.
52 void addModule(const String& moduleURL, WorkletOptions&&, DOMPromiseDeferred<void>&&) final;
53 Vector<Ref<WorkletGlobalScopeProxy>> createGlobalScopes() final;
54
55private:
56 explicit PaintWorklet(Document& document)
57 : Worklet(document)
58 { }
59};
