26#pragma once
27
28#include "DeprecatedCSSOMPrimitiveValue.h"
29#include "Rect.h"
30#include <wtf/RefPtr.h>
31
32namespace WebCore {
33
34class DeprecatedCSSOMRect final : public RefCounted<DeprecatedCSSOMRect> {
35public:
36 static Ref<DeprecatedCSSOMRect> create(const Rect& rect, CSSStyleDeclaration& owner)
37 {
38 return adoptRef(*new DeprecatedCSSOMRect(rect, owner));
39 }
40
41 DeprecatedCSSOMPrimitiveValue* top() const { return m_top.get(); }
42 DeprecatedCSSOMPrimitiveValue* right() const { return m_right.get(); }
43 DeprecatedCSSOMPrimitiveValue* bottom() const { return m_bottom.get(); }
44 DeprecatedCSSOMPrimitiveValue* left() const { return m_left.get(); }
45
46private:
47 DeprecatedCSSOMRect(const Rect& rect, CSSStyleDeclaration& owner)
48 {
49 if (rect.top())
50 m_top = rect.top()->createDeprecatedCSSOMPrimitiveWrapper(owner);
51 if (rect.right())
52 m_right = rect.right()->createDeprecatedCSSOMPrimitiveWrapper(owner);
53 if (rect.bottom())
54 m_bottom = rect.bottom()->createDeprecatedCSSOMPrimitiveWrapper(owner);
55 if (rect.left())
56 m_left = rect.left()->createDeprecatedCSSOMPrimitiveWrapper(owner);
57 }
58
59 RefPtr<DeprecatedCSSOMPrimitiveValue> m_top;
60 RefPtr<DeprecatedCSSOMPrimitiveValue> m_right;
61 RefPtr<DeprecatedCSSOMPrimitiveValue> m_bottom;
62 RefPtr<DeprecatedCSSOMPrimitiveValue> m_left;
63};
64
65} // namespace WebCore
