26#pragma once
27
28#include "CSSValueList.h"
29#include "DeprecatedCSSOMValue.h"
30
31namespace WebCore {
32
33class DeprecatedCSSOMValueList : public DeprecatedCSSOMValue {
34public:
35 static Ref<DeprecatedCSSOMValueList> create(const CSSValueList& value, CSSStyleDeclaration& owner)
36 {
37 return adoptRef(*new DeprecatedCSSOMValueList(value, owner));
38 }
39
40 String cssText() const;
41
42 size_t length() const { return m_values.size(); }
43 DeprecatedCSSOMValue* item(size_t index) { return index < m_values.size() ? m_values[index].ptr() : nullptr; }
44
45private:
46 DeprecatedCSSOMValueList(const CSSValueList& value, CSSStyleDeclaration& owner)
47 : DeprecatedCSSOMValue(DeprecatedValueListClass, owner)
48 {
49 m_valueSeparator = value.separator();
50 m_values.reserveInitialCapacity(value.length());
51 for (unsigned i = 0, size = value.length(); i < size; ++i)
52 m_values.uncheckedAppend(value.itemWithoutBoundsCheck(i)->createDeprecatedCSSOMWrapper(owner));
53 }
54
55 Vector<Ref<DeprecatedCSSOMValue>, 4> m_values;
56};
57
58} // namespace WebCore
59
60SPECIALIZE_TYPE_TRAITS_CSSOM_VALUE(DeprecatedCSSOMValueList, isValueList())
