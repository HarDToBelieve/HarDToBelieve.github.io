36/* MediaList is used to store 3 types of media related entities which mean the same:
37 * Media Queries, Media Types and Media Descriptors.
38 * Currently MediaList always tries to parse media queries and if parsing fails,
39 * tries to fallback to Media Descriptors if m_fallbackToDescriptor flag is set.
40 * Slight problem with syntax error handling:
41 * CSS 2.1 Spec (http://www.w3.org/TR/CSS21/media.html)
42 * specifies that failing media type parsing is a syntax error
43 * CSS 3 Media Queries Spec (http://www.w3.org/TR/css3-mediaqueries/)
44 * specifies that failing media query is a syntax error
45 * HTML 4.01 spec (http://www.w3.org/TR/REC-html40/present/styles.html#adef-media)
46 * specifies that Media Descriptors should be parsed with forward-compatible syntax
47 * DOM Level 2 Style Sheet spec (http://www.w3.org/TR/DOM-Level-2-Style/)
48 * talks about MediaList.mediaText and refers
49 * - to Media Descriptors of HTML 4.0 in context of StyleSheet
50 * - to Media Types of CSS 2.0 in context of CSSMediaRule and CSSImportRule
51 *
52 * These facts create situation where same (illegal) media specification may result in
53 * different parses depending on whether it is media attr of style element or part of
54 * css @media rule.
55 * <style media="screen and resolution > 40dpi"> ..</style> will be enabled on screen devices where as
56 * @media screen and resolution > 40dpi {..} will not.
57 * This gets more counter-intuitive in JavaScript:
58 * document.styleSheets[0].media.mediaText = "screen and resolution > 40dpi" will be ok and
59 * enabled, while
60 * document.styleSheets[0].cssRules[0].media.mediaText = "screen and resolution > 40dpi" will
61 * throw SyntaxError exception.
62 */

64Ref<MediaQuerySet> MediaQuerySet::create(const String& mediaString, MediaQueryParserContext context)
65{
66 if (mediaString.isEmpty())
67 return MediaQuerySet::create();
68
69 auto parsedMediaQuerySet = MediaQueryParser::parseMediaQuerySet(mediaString, context);
70 if (UNLIKELY(!parsedMediaQuerySet))
71 return MediaQuerySet::create();
72
73 return parsedMediaQuerySet.releaseNonNull();
74}

93bool MediaQuerySet::add(const String& queryString)
94{
95 // To "parse a media query" for a given string means to follow "the parse
96 // a media query list" steps and return "null" if more than one media query
97 // is returned, or else the returned media query.
98 auto result = create(queryString);
99
100 // Only continue if exactly one media query is found, as described above.
101 if (result->m_queries.size() != 1)
102 return false;
103
104 // If comparing with any of the media queries in the collection of media
105 // queries returns true terminate these steps.
106 for (size_t i = 0; i < m_queries.size(); ++i) {
107 if (m_queries[i] == result->m_queries[0])
108 return false;
109 }
110
111 m_queries.append(result->m_queries[0]);
112 return true;
113}
114
115bool MediaQuerySet::remove(const String& queryStringToRemove)
116{
117 // To "parse a media query" for a given string means to follow "the parse
118 // a media query list" steps and return "null" if more than one media query
119 // is returned, or else the returned media query.
120 auto result = create(queryStringToRemove);
121
122 // Only continue if exactly one media query is found, as described above.
123 if (result->m_queries.size() != 1)
124 return true;
125
126 // Remove any media query from the collection of media queries for which
127 // comparing with the media query returns true.
128 bool found = false;
129
130 // Using signed int here, since for the first value, --i will result in -1.
131 for (int i = 0; i < (int)m_queries.size(); ++i) {
132 if (m_queries[i] == result->m_queries[0]) {
133 m_queries.remove(i);
134 --i;
135 found = true;
136 }
137 }
138
139 return found;
140}

147String MediaQuerySet::mediaText() const
148{
149 StringBuilder text;
150 bool needComma = false;
151 for (auto& query : m_queries) {
152 if (needComma)
153 text.append(", ");
154 text.append(query.cssText());
155 needComma = true;
156 }
157 return text.toString();
158}

160void MediaQuerySet::shrinkToFit()
161{
162 m_queries.shrinkToFit();
163 for (auto& query : m_queries)
164 query.shrinkToFit();
165}

190String MediaList::item(unsigned index) const
191{
192 auto& queries = m_mediaQueries->queryVector();
193 if (index < queries.size())
194 return queries[index].cssText();
195 return String();
196}

198ExceptionOr<void> MediaList::deleteMedium(const String& medium)
199{
200 CSSStyleSheet::RuleMutationScope mutationScope(m_parentRule);
201
202 bool success = m_mediaQueries->remove(medium);
203 if (!success)
204 return Exception { NotFoundError };
205 if (m_parentStyleSheet)
206 m_parentStyleSheet->didMutate();
207 return { };
208}

210void MediaList::appendMedium(const String& medium)
211{
212 CSSStyleSheet::RuleMutationScope mutationScope(m_parentRule);
213
214 if (!m_mediaQueries->add(medium))
215 return;
216 if (m_parentStyleSheet)
217 m_parentStyleSheet->didMutate();
218}

228static void addResolutionWarningMessageToConsole(Document& document, const String& serializedExpression, const CSSPrimitiveValue& value)
229{
230 static NeverDestroyed<String> mediaQueryMessage(MAKE_STATIC_STRING_IMPL("Consider using 'dppx' units instead of '%replacementUnits%', as in CSS '%replacementUnits%' means dots-per-CSS-%lengthUnit%, not dots-per-physical-%lengthUnit%, so does not correspond to the actual '%replacementUnits%' of a screen. In media query expression: "));
231 static NeverDestroyed<String> mediaValueDPI(MAKE_STATIC_STRING_IMPL("dpi"));
232 static NeverDestroyed<String> mediaValueDPCM(MAKE_STATIC_STRING_IMPL("dpcm"));
233 static NeverDestroyed<String> lengthUnitInch(MAKE_STATIC_STRING_IMPL("inch"));
234 static NeverDestroyed<String> lengthUnitCentimeter(MAKE_STATIC_STRING_IMPL("centimeter"));
235
236 String message;
237 if (value.isDotsPerInch())
238 message = mediaQueryMessage.get().replace("%replacementUnits%", mediaValueDPI).replace("%lengthUnit%", lengthUnitInch);
239 else if (value.isDotsPerCentimeter())
240 message = mediaQueryMessage.get().replace("%replacementUnits%", mediaValueDPCM).replace("%lengthUnit%", lengthUnitCentimeter);
241 else
242 ASSERT_NOT_REACHED();
243
244 message.append(serializedExpression);
245
246 document.addConsoleMessage(MessageSource::CSS, MessageLevel::Debug, message);
247}

249void reportMediaQueryWarningIfNeeded(Document* document, const MediaQuerySet* mediaQuerySet)
250{
251 if (!mediaQuerySet || !document)
252 return;
253
254 for (auto& query : mediaQuerySet->queryVector()) {
255 if (!query.ignored() && !equalLettersIgnoringASCIICase(query.mediaType(), "print")) {
256 auto& expressions = query.expressions();
257 for (auto& expression : expressions) {
258 if (expression.mediaFeature() == MediaFeatureNames::resolution || expression.mediaFeature() == MediaFeatureNames::maxResolution || expression.mediaFeature() == MediaFeatureNames::minResolution) {
259 auto* value = expression.value();
260 if (is<CSSPrimitiveValue>(value)) {
261 auto& primitiveValue = downcast<CSSPrimitiveValue>(*value);
262 if (primitiveValue.isDotsPerInch() || primitiveValue.isDotsPerCentimeter())
263 addResolutionWarningMessageToConsole(*document, mediaQuerySet->mediaText(), primitiveValue);
264 }
265 }
266 }
267 }
268 }
