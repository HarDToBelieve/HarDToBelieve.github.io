29#pragma once
30
31#include "MediaQueryExpression.h"
32#include <wtf/Vector.h>
33
34namespace WebCore {
35
36class MediaQuery {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 enum Restrictor { Only, Not, None };
40
41 MediaQuery(Restrictor, const String& mediaType, Vector<MediaQueryExpression>&&);
42
43 Restrictor restrictor() const { return m_restrictor; }
44 const Vector<MediaQueryExpression>& expressions() const { return m_expressions; }
45 const String& mediaType() const { return m_mediaType; }
46 bool ignored() const { return m_ignored; }
47
48 const String& cssText() const;
49
50 bool operator==(const MediaQuery& other) const;
51
52 void shrinkToFit() { m_expressions.shrinkToFit(); }
53
54private:
55 String serialize() const;
56
57 String m_mediaType;
58 mutable String m_serializationCache;
59 Vector<MediaQueryExpression> m_expressions;
60 Restrictor m_restrictor;
61 bool m_ignored { false };
62};
63
64WTF::TextStream& operator<<(WTF::TextStream&, const MediaQuery&);
65
66} // namespace
