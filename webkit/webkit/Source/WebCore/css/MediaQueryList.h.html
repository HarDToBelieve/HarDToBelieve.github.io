30// MediaQueryList interface is specified at https://drafts.csswg.org/cssom-view/#the-mediaquerylist-interface
31// The objects of this class are returned by window.matchMedia. They may be used to
32// retrieve the current value of the given media query and to add/remove listeners that
33// will be called whenever the value of the query changes.
34
35class MediaQueryList final : public RefCounted<MediaQueryList>, public EventTargetWithInlineData, public ActiveDOMObject {
36 WTF_MAKE_ISO_ALLOCATED(MediaQueryList);
37public:
38 static Ref<MediaQueryList> create(Document&, MediaQueryMatcher&, Ref<MediaQuerySet>&&, bool matches);
39 ~MediaQueryList();
40
41 String media() const;
42 bool matches();
43
44 void addListener(RefPtr<EventListener>&&);
45 void removeListener(RefPtr<EventListener>&&);
46
47 void evaluate(MediaQueryEvaluator&, bool& notificationNeeded);
48
49 void detachFromMatcher();
50
51 using RefCounted::ref;
52 using RefCounted::deref;
53
54private:
55 MediaQueryList(Document&, MediaQueryMatcher&, Ref<MediaQuerySet>&&, bool matches);
56
57 void setMatches(bool);
58
59 EventTargetInterface eventTargetInterface() const final { return MediaQueryListEventTargetInterfaceType; }
60 ScriptExecutionContext* scriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
61 void refEventTarget() final { ref(); }
62 void derefEventTarget() final { deref(); }
63 void eventListenersDidChange() final;
64
65 // ActiveDOMObject.
66 const char* activeDOMObjectName() const final;
67 bool virtualHasPendingActivity() const final;
68
69 RefPtr<MediaQueryMatcher> m_matcher;
70 Ref<MediaQuerySet> m_media;
71 unsigned m_evaluationRound; // Indicates if the query has been evaluated after the last style selector change.
72 unsigned m_changeRound; // Used to know if the query has changed in the last style selector change.
73 bool m_matches;
74 bool m_hasChangeEventListener { false };
75};

77}
