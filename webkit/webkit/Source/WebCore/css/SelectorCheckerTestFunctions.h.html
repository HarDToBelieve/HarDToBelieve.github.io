101ALWAYS_INLINE bool isChecked(const Element& element)
102{
103 // Even though WinIE allows checked and indeterminate to co-exist, the CSS selector spec says that
104 // you can't be both checked and indeterminate. We will behave like WinIE behind the scenes and just
105 // obey the CSS spec here in the test for matching the pseudo.
106 if (is<HTMLInputElement>(element)) {
107 auto& inputElement = downcast<HTMLInputElement>(element);
108 return inputElement.shouldAppearChecked() && !inputElement.shouldAppearIndeterminate();
109 }
110 if (is<HTMLOptionElement>(element))
111 return const_cast<HTMLOptionElement&>(downcast<HTMLOptionElement>(element)).selected();
112
113 return false;
114}

161ALWAYS_INLINE bool containslanguageSubtagMatchingRange(StringView language, StringView range, unsigned languageLength, unsigned& position)
162{
163 unsigned languageSubtagsStartIndex = position;
164 unsigned languageSubtagsEndIndex = languageLength;
165 bool isAsteriskRange = range == "*";
166 do {
167 if (languageSubtagsStartIndex > 0)
168 languageSubtagsStartIndex += 1;
169
170 languageSubtagsEndIndex = std::min<unsigned>(language.find('-', languageSubtagsStartIndex), languageLength);
171
172 if (languageSubtagsStartIndex > languageSubtagsEndIndex)
173 return false;
174
175 StringView languageSubtag = language.substring(languageSubtagsStartIndex, languageSubtagsEndIndex - languageSubtagsStartIndex);
176 bool isEqual = equalIgnoringASCIICase(range, languageSubtag);
177 if (!isAsteriskRange) {
178 if ((!isEqual && !languageSubtagsStartIndex) || (languageSubtag.length() == 1 && languageSubtagsStartIndex > 0))
179 return false;
180 }
181 languageSubtagsStartIndex = languageSubtagsEndIndex;
182 if (isEqual || isAsteriskRange) {
183 position = languageSubtagsStartIndex;
184 return true;
185 }
186
187 } while (languageSubtagsStartIndex < languageLength);
188 return false;
189}

191ALWAYS_INLINE bool matchesLangPseudoClass(const Element& element, const Vector<AtomString>& argumentList)
192{
193 AtomString language;
194#if ENABLE(VIDEO)
195 if (is<WebVTTElement>(element))
196 language = downcast<WebVTTElement>(element).language();
197 else
198#endif
199 language = element.computeInheritedLanguage();
200
201 if (language.isEmpty())
202 return false;
203
204 // Implement basic and extended filterings of given language tags
205 // as specified in www.ietf.org/rfc/rfc4647.txt.
206 StringView languageStringView = language.string();
207 unsigned languageLength = language.length();
208 for (const AtomString& range : argumentList) {
209 if (range.isEmpty())
210 continue;
211
212 if (range == "*")
213 return true;
214
215 StringView rangeStringView = range.string();
216 if (equalIgnoringASCIICase(languageStringView, rangeStringView) && !languageStringView.contains('-'))
217 return true;
218
219 unsigned rangeLength = rangeStringView.length();
220 unsigned rangeSubtagsStartIndex = 0;
221 unsigned rangeSubtagsEndIndex = rangeLength;
222 unsigned lastMatchedLanguageSubtagIndex = 0;
223
224 bool matchedRange = true;
225 do {
226 if (rangeSubtagsStartIndex > 0)
227 rangeSubtagsStartIndex += 1;
228 if (rangeSubtagsStartIndex > languageLength)
229 return false;
230 rangeSubtagsEndIndex = std::min<unsigned>(rangeStringView.find('-', rangeSubtagsStartIndex), rangeLength);
231 StringView rangeSubtag = rangeStringView.substring(rangeSubtagsStartIndex, rangeSubtagsEndIndex - rangeSubtagsStartIndex);
232 if (!containslanguageSubtagMatchingRange(languageStringView, rangeSubtag, languageLength, lastMatchedLanguageSubtagIndex)) {
233 matchedRange = false;
234 break;
235 }
236 rangeSubtagsStartIndex = rangeSubtagsEndIndex;
237 } while (rangeSubtagsStartIndex < rangeLength);
238 if (matchedRange)
239 return true;
240 }
241 return false;
242}

269ALWAYS_INLINE bool scrollbarMatchesHoverPseudoClass(const SelectorChecker::CheckingContext& context)
270{
271 if (!context.scrollbarState)
272 return false;
273 auto scrollbarPart = context.scrollbarState->scrollbarPart;
274 auto hoveredPart = context.scrollbarState->hoveredPart;
275 if (scrollbarPart == ScrollbarBGPart)
276 return hoveredPart != NoPart;
277 if (scrollbarPart == TrackBGPart)
278 return hoveredPart == BackTrackPart || hoveredPart == ForwardTrackPart || hoveredPart == ThumbPart;
279 return scrollbarPart == hoveredPart;
280}

289 return pressedPart != NoPart;
290 if (scrollbarPart == TrackBGPart)
291 return pressedPart == BackTrackPart || pressedPart == ForwardTrackPart || pressedPart == ThumbPart;
292 return scrollbarPart == pressedPart;

337ALWAYS_INLINE bool scrollbarMatchesDoubleButtonPseudoClass(const SelectorChecker::CheckingContext& context)
338{
339 if (!context.scrollbarState)
340 return false;
341 auto scrollbarPart = context.scrollbarState->scrollbarPart;
342 auto buttonsPlacement = context.scrollbarState->buttonsPlacement;
343 if (scrollbarPart == BackButtonStartPart || scrollbarPart == ForwardButtonStartPart || scrollbarPart == BackTrackPart)
344 return buttonsPlacement == ScrollbarButtonsDoubleStart || buttonsPlacement == ScrollbarButtonsDoubleBoth;
345 if (scrollbarPart == BackButtonEndPart || scrollbarPart == ForwardButtonEndPart || scrollbarPart == ForwardTrackPart)
346 return buttonsPlacement == ScrollbarButtonsDoubleEnd || buttonsPlacement == ScrollbarButtonsDoubleBoth;
347 return false;
348}

381ALWAYS_INLINE bool matchesFullScreenPseudoClass(const Element& element)
382{
383 // While a Document is in the fullscreen state, and the document's current fullscreen
384 // element is an element in the document, the 'full-screen' pseudoclass applies to
385 // that element. Also, an <iframe>, <object> or <embed> element whose child browsing
386 // context's Document is in the fullscreen state has the 'full-screen' pseudoclass applied.
387 if (is<HTMLFrameElementBase>(element) && element.containsFullScreenElement())
388 return true;
389 if (!element.document().fullscreenManager().isFullscreen())
390 return false;
391 return &element == element.document().fullscreenManager().currentFullscreenElement();
392}

395{
396 if (&element != element.document().fullscreenManager().currentFullscreenElement())
397 return false;
398 return element.document().fullscreenManager().isAnimatingFullscreen();
399}

452// This needs to match a subset of elements matchesFocusPseudoClass match since direct focus is treated
453// as a part of focus pseudo class selectors in ElementRuleCollector::collectMatchingRules.
454ALWAYS_INLINE bool matchesDirectFocusPseudoClass(const Element& element)
455{
456 if (InspectorInstrumentation::forcePseudoState(element, CSSSelector::PseudoClassFocus))
457 return true;
458
459 return element.focused() && isFrameFocused(element);
