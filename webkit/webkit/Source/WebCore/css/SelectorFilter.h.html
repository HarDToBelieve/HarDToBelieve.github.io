31#include "Element.h"

36namespace WebCore {
37
38class CSSSelector;
39
40class SelectorFilter {
41public:
42 void pushParent(Element* parent);
43 void pushParentInitializingIfNeeded(Element& parent);
44 void popParent();
45 void popParentsUntil(Element* parent);
46 bool parentStackIsEmpty() const { return m_parentStack.isEmpty(); }
47 bool parentStackIsConsistent(const ContainerNode* parentNode) const;
48
49 using Hashes = std::array<unsigned, 4>;
50 bool fastRejectSelector(const Hashes&) const;
51 static Hashes collectHashes(const CSSSelector&);
52
53private:
54 void initializeParentStack(Element& parent);
55
56 struct ParentStackFrame {
57 ParentStackFrame() : element(0) { }
58 ParentStackFrame(Element* element) : element(element) { }
59 Element* element;
60 Vector<unsigned, 4> identifierHashes;
61 };
62 Vector<ParentStackFrame> m_parentStack;
63
64 // With 100 unique strings in the filter, 2^12 slot table has false positive rate of ~0.2%.
65 static const unsigned bloomFilterKeyBits = 12;
66 CountingBloomFilter<bloomFilterKeyBits> m_ancestorIdentifierFilter;
67};
68
69inline bool SelectorFilter::fastRejectSelector(const Hashes& hashes) const
70{
71 for (auto& hash : hashes) {
72 if (!hash)
73 return false;
74 if (!m_ancestorIdentifierFilter.mayContain(hash))
75 return true;
76 }
77 return false;
78}
