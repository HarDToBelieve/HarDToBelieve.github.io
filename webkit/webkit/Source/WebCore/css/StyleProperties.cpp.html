93ImmutableStyleProperties::ImmutableStyleProperties(const CSSProperty* properties, unsigned length, CSSParserMode cssParserMode)
94 : StyleProperties(cssParserMode, length)
95{
96 StylePropertyMetadata* metadataArray = const_cast<StylePropertyMetadata*>(this->metadataArray());
97 PackedPtr<CSSValue>* valueArray = bitwise_cast<PackedPtr<CSSValue>*>(this->valueArray());
98 for (unsigned i = 0; i < length; ++i) {
99 metadataArray[i] = properties[i].metadata();
100 auto* value = properties[i].value();
101 valueArray[i] = value;
102 value->ref();
103 }
104}

106ImmutableStyleProperties::~ImmutableStyleProperties()
107{
108 PackedPtr<CSSValue>* valueArray = bitwise_cast<PackedPtr<CSSValue>*>(this->valueArray());
109 for (unsigned i = 0; i < m_arraySize; ++i)
110 valueArray[i]->deref();
111}

113MutableStyleProperties::MutableStyleProperties(const StyleProperties& other)
114 : StyleProperties(other.cssParserMode(), MutablePropertiesType)
115{
116 ASSERT(other.type() != DeferredPropertiesType);
117 if (is<MutableStyleProperties>(other))
118 m_propertyVector = downcast<MutableStyleProperties>(other).m_propertyVector;
119 else {
120 const auto& immutableOther = downcast<ImmutableStyleProperties>(other);
121 unsigned propertyCount = immutableOther.propertyCount();
122 m_propertyVector.reserveInitialCapacity(propertyCount);
123 for (unsigned i = 0; i < propertyCount; ++i)
124 m_propertyVector.uncheckedAppend(immutableOther.propertyAt(i).toCSSProperty());
125 }

129{
130 if (auto value = getPropertyCSSValue(propertyID)) {
131 switch (propertyID) {
132 case CSSPropertyFillOpacity:
133 case CSSPropertyFloodOpacity:
134 case CSSPropertyOpacity:
135 case CSSPropertyStopOpacity:
136 case CSSPropertyStrokeOpacity:
137 // Opacity percentage values serialize as a fraction in the range 0-1, no "%".
138 if (is<CSSPrimitiveValue>(*value) && downcast<CSSPrimitiveValue>(*value).isPercentage())
139 return makeString(downcast<CSSPrimitiveValue>(*value).doubleValue() / 100);
140 FALLTHROUGH;
141 default:
142 return value->cssText();
143 }
144 }
145
146 {
147 auto shorthand = shorthandForProperty(propertyID);

309std::optional<Color> StyleProperties::propertyAsColor(CSSPropertyID property) const
310{
311 auto colorValue = getPropertyCSSValue(property);
312 if (!is<CSSPrimitiveValue>(colorValue))
313 return std::nullopt;
314
315 auto& primitiveColor = downcast<CSSPrimitiveValue>(*colorValue);
316 return primitiveColor.isRGBColor() ? primitiveColor.color() : CSSParser::parseColor(colorValue->cssText());
317}

333String StyleProperties::borderSpacingValue(const StylePropertyShorthand& shorthand) const
334{
335 auto horizontalValue = getPropertyCSSValue(shorthand.properties()[0]);
336 auto verticalValue = getPropertyCSSValue(shorthand.properties()[1]);
337
338 // While standard border-spacing property does not allow specifying border-spacing-vertical without
339 // specifying border-spacing-horizontal <http://www.w3.org/TR/CSS21/tables.html#separated-borders>,
340 // -webkit-border-spacing-vertical can be set without -webkit-border-spacing-horizontal.
341 if (!horizontalValue || !verticalValue)
342 return String();
343
344 String horizontalValueCSSText = horizontalValue->cssText();
345 String verticalValueCSSText = verticalValue->cssText();
346 if (horizontalValueCSSText == verticalValueCSSText)
347 return horizontalValueCSSText;
348 return horizontalValueCSSText + ' ' + verticalValueCSSText;
349}

351void StyleProperties::appendFontLonghandValueIfExplicit(CSSPropertyID propertyID, StringBuilder& result, String& commonValue) const
352{
353 int foundPropertyIndex = findPropertyIndex(propertyID);
354 if (foundPropertyIndex == -1)
355 return; // All longhands must have at least implicit values if "font" is specified.
356
357 if (propertyAt(foundPropertyIndex).isImplicit()) {
358 commonValue = String();
359 return;
360 }
361
362 char prefix = '\0';
363 switch (propertyID) {
364 case CSSPropertyFontStyle:
365 break; // No prefix.
366 case CSSPropertyFontFamily:
367 case CSSPropertyFontVariantAlternates:
368 case CSSPropertyFontVariantCaps:
369 case CSSPropertyFontVariantLigatures:
370 case CSSPropertyFontVariantNumeric:
371 case CSSPropertyFontVariantPosition:
372 case CSSPropertyFontVariantEastAsian:
373 case CSSPropertyFontWeight:
374 case CSSPropertyFontStretch:
375 prefix = ' ';
376 break;
377 case CSSPropertyLineHeight:
378 prefix = '/';
379 break;
380 default:
381 ASSERT_NOT_REACHED();
382 }
383
384 if (prefix && !result.isEmpty())
385 result.append(prefix);
386 String value = propertyAt(foundPropertyIndex).value()->cssText();
387 result.append(value);
388 if (!commonValue.isNull() && commonValue != value)
389 commonValue = String();
390}

392String StyleProperties::fontValue() const
393{
394 int fontSizePropertyIndex = findPropertyIndex(CSSPropertyFontSize);
395 int fontFamilyPropertyIndex = findPropertyIndex(CSSPropertyFontFamily);
396 if (fontSizePropertyIndex == -1 || fontFamilyPropertyIndex == -1)
397 return emptyString();
398
399 PropertyReference fontSizeProperty = propertyAt(fontSizePropertyIndex);
400 PropertyReference fontFamilyProperty = propertyAt(fontFamilyPropertyIndex);
401 if (fontSizeProperty.isImplicit() || fontFamilyProperty.isImplicit())
402 return emptyString();
403
404 String commonValue = fontSizeProperty.value()->cssText();
405 StringBuilder result;
406 appendFontLonghandValueIfExplicit(CSSPropertyFontStyle, result, commonValue);
407 appendFontLonghandValueIfExplicit(CSSPropertyFontVariantCaps, result, commonValue);
408 appendFontLonghandValueIfExplicit(CSSPropertyFontWeight, result, commonValue);
409 appendFontLonghandValueIfExplicit(CSSPropertyFontStretch, result, commonValue);
410 if (!result.isEmpty())
411 result.append(' ');
412 result.append(fontSizeProperty.value()->cssText());
413 appendFontLonghandValueIfExplicit(CSSPropertyLineHeight, result, commonValue);
414 if (!result.isEmpty())
415 result.append(' ');
416 result.append(fontFamilyProperty.value()->cssText());
417 if (isCSSWideValueKeyword(commonValue))
418 return commonValue;
419 return result.toString();
420}

422String StyleProperties::fontVariantValue() const
423{
424 String commonValue;
425 StringBuilder result;
426 appendFontLonghandValueIfExplicit(CSSPropertyFontVariantLigatures, result, commonValue);

437String StyleProperties::get2Values(const StylePropertyShorthand& shorthand) const
438{
439 // Assume the properties are in the usual order start, end.
440 int startValueIndex = findPropertyIndex(shorthand.properties()[0]);
441 int endValueIndex = findPropertyIndex(shorthand.properties()[1]);
442
443 if (startValueIndex == -1 || endValueIndex == -1)
444 return { };
445
446 auto start = propertyAt(startValueIndex);
447 auto end = propertyAt(endValueIndex);
448
449 // All 2 properties must be specified.
450 if (!start.value() || !end.value())
451 return { };
452
453 // Important flags must be the same
454 if (start.isImportant() != end.isImportant())
455 return { };
456
457 if (start.isInherited() && end.isInherited())
458 return getValueName(CSSValueInherit);
459
460 if (start.value()->isInitialValue() || end.value()->isInitialValue()) {
461 if (start.value()->isInitialValue() && end.value()->isInitialValue() && !start.isImplicit())
462 return getValueName(CSSValueInitial);
463 return { };
464 }
465
466 StringBuilder result;
467 result.append(start.value()->cssText());
468 if (!start.value()->equals(*end.value())) {
469 result.append(' ');
470 result.append(end.value()->cssText());
471 }
472 return result.toString();
473}

475String StyleProperties::get4Values(const StylePropertyShorthand& shorthand) const
476{
477 // Assume the properties are in the usual order top, right, bottom, left.
478 int topValueIndex = findPropertyIndex(shorthand.properties()[0]);
479 int rightValueIndex = findPropertyIndex(shorthand.properties()[1]);
480 int bottomValueIndex = findPropertyIndex(shorthand.properties()[2]);
481 int leftValueIndex = findPropertyIndex(shorthand.properties()[3]);
482
483 if (topValueIndex == -1 || rightValueIndex == -1 || bottomValueIndex == -1 || leftValueIndex == -1)
484 return String();
485
486 PropertyReference top = propertyAt(topValueIndex);
487 PropertyReference right = propertyAt(rightValueIndex);
488 PropertyReference bottom = propertyAt(bottomValueIndex);
489 PropertyReference left = propertyAt(leftValueIndex);
490
491 // All 4 properties must be specified.
492 if (!top.value() || !right.value() || !bottom.value() || !left.value())
493 return String();
494
495 // Important flags must be the same
496 if (top.isImportant() != right.isImportant() || right.isImportant() != bottom.isImportant() || bottom.isImportant() != left.isImportant())
497 return String();
498
499 if (top.isInherited() && right.isInherited() && bottom.isInherited() && left.isInherited())
500 return getValueName(CSSValueInherit);
501
502 if (top.value()->isInitialValue() || right.value()->isInitialValue() || bottom.value()->isInitialValue() || left.value()->isInitialValue()) {
503 if (top.value()->isInitialValue() && right.value()->isInitialValue() && bottom.value()->isInitialValue() && left.value()->isInitialValue() && !top.isImplicit()) {
504 // All components are "initial" and "top" is not implicit.
505 return getValueName(CSSValueInitial);
506 }
507 return String();
508 }
509
510 bool showLeft = !right.value()->equals(*left.value());
511 bool showBottom = !top.value()->equals(*bottom.value()) || showLeft;
512 bool showRight = !top.value()->equals(*right.value()) || showBottom;
513
514 StringBuilder result;
515 result.append(top.value()->cssText());
516 if (showRight) {
517 result.append(' ');
518 result.append(right.value()->cssText());
519 }
520 if (showBottom) {
521 result.append(' ');
522 result.append(bottom.value()->cssText());
523 }
524 if (showLeft) {
525 result.append(' ');
526 result.append(left.value()->cssText());
527 }
528 return result.toString();
529}

531String StyleProperties::getLayeredShorthandValue(const StylePropertyShorthand& shorthand) const
532{
533 StringBuilder result;
534
535 const unsigned size = shorthand.length();
536 Vector<RefPtr<CSSValue>> values(size);
537 size_t numLayers = 0;
538
539 for (unsigned i = 0; i < size; ++i) {
540 values[i] = getPropertyCSSValue(shorthand.properties()[i]);
541 if (!values[i]) {
542 // We don't have all longhand properties defined as required for the shorthand
543 // property and thus should not serialize to a shorthand value. See spec at
544 // https://www.w3.org/TR/cssom-1/#serialize-a-css-declaration-block
545 return String();
546 }
547 if (values[i]->isBaseValueList())
548 numLayers = std::max(downcast<CSSValueList>(*values[i]).length(), numLayers);
549 else
550 numLayers = std::max<size_t>(1U, numLayers);
551 }
552
553 String commonValue;
554
555 // Now stitch the properties together.
556 // Implicit initial values are flagged as such and can safely be omitted.
557 for (size_t i = 0; i < numLayers; i++) {
558 StringBuilder layerResult;
559 bool useRepeatXShorthand = false;
560 bool useRepeatYShorthand = false;
561 bool useSingleWordShorthand = false;
562 bool foundPositionYCSSProperty = false;
563 for (unsigned j = 0; j < size; j++) {
564 auto property = shorthand.properties()[j];
565
566 auto value = values[j];
567 if (value) {
568 if (value->isBaseValueList())
569 value = downcast<CSSValueList>(*value).item(i);
570 else {
571 // Color only belongs in the last layer.
572 if (property == CSSPropertyBackgroundColor) {
573 if (i != numLayers - 1)
574 value = nullptr;
575 } else if (i) // Other singletons only belong in the first layer.
576 value = nullptr;
577 }
578 }
579
580 // We need to report background-repeat as it was written in the CSS.
581 // If the property is implicit, then it was written with only one value. Here we figure out which value that was so we can report back correctly.
582 if (value && j < size - 1 && (property == CSSPropertyBackgroundRepeatX || property == CSSPropertyWebkitMaskRepeatX) && isPropertyImplicit(property)) {
583 // Make sure the value was not reset in the layer check just above.
584 auto nextProperty = shorthand.properties()[j + 1];
585 if (nextProperty == CSSPropertyBackgroundRepeatY || nextProperty == CSSPropertyWebkitMaskRepeatY) {
586 if (auto yValue = values[j + 1]) {
587 if (is<CSSValueList>(*yValue))
588 yValue = downcast<CSSValueList>(*yValue).itemWithoutBoundsCheck(i);
589 if (!is<CSSPrimitiveValue>(*value) || !is<CSSPrimitiveValue>(*yValue))
590 continue;
591
592 auto xId = downcast<CSSPrimitiveValue>(*value).valueID();
593 auto yId = downcast<CSSPrimitiveValue>(*yValue).valueID();
594 if (xId != yId) {
595 if (xId == CSSValueRepeat && yId == CSSValueNoRepeat) {
596 useRepeatXShorthand = true;
597 ++j;
598 } else if (xId == CSSValueNoRepeat && yId == CSSValueRepeat) {
599 useRepeatYShorthand = true;
600 continue;
601 }
602 } else {
603 useSingleWordShorthand = true;
604 ++j;
605 }
606 }
607 }
608 }
609
610 String valueText;
611 if (value && !value->isImplicitInitialValue()) {
612 if (!layerResult.isEmpty())
613 layerResult.append(' ');
614
615 if (property == CSSPropertyBackgroundSize || property == CSSPropertyWebkitMaskSize) {
616 if (!foundPositionYCSSProperty)
617 continue;
618 layerResult.append("/ ");
619 }
620
621 if (useRepeatXShorthand) {
622 useRepeatXShorthand = false;
623 layerResult.append(getValueName(CSSValueRepeatX));
624 } else if (useRepeatYShorthand) {
625 useRepeatYShorthand = false;
626 layerResult.append(getValueName(CSSValueRepeatY));
627 } else {
628 if (useSingleWordShorthand)
629 useSingleWordShorthand = false;
630 valueText = value->cssText();
631 layerResult.append(valueText);
632 }
633
634 if (property == CSSPropertyBackgroundPositionY || property == CSSPropertyWebkitMaskPositionY)
635 foundPositionYCSSProperty = true;
636 }
637
638 if (commonValue.isNull())
639 commonValue = valueText;
640 else if (commonValue != valueText)
641 commonValue = emptyString(); // Could use value here other than a CSS-wide value keyword or the null string.
642 }
643
644 if (!layerResult.isEmpty())
645 result.append(result.isEmpty() ? "" : ", ", layerResult.toString());
646 }
647
648 if (isCSSWideValueKeyword(commonValue))
649 return commonValue;
650
651 return result.isEmpty() ? String() : result.toString();
652}

660{

690{

883{

896 return CSSParser::parseValue(*this, propertyID, value, important, parserContext) == CSSParser::ParseResult::Changed;

943bool MutableStyleProperties::canUpdateInPlace(const CSSProperty& property, CSSProperty* toReplace) const
944{
945 // If the property is in a logical property group, we can't just update the value in-place,
946 // because afterwards there might be another property of the same group but different mapping logic.
947 // In that case the latter might override the former, so setProperty would have no effect.
948 CSSPropertyID id = property.id();
949 if (CSSProperty::isInLogicalPropertyGroup(id)) {
950 ASSERT(toReplace >= m_propertyVector.begin());
951 ASSERT(toReplace < m_propertyVector.end());
952 for (CSSProperty* it = toReplace + 1; it != m_propertyVector.end(); ++it) {
953 if (CSSProperty::areInSameLogicalPropertyGroupWithDifferentMappingLogic(id, it->id()))
954 return false;
955 }
956 }
957 return true;
958}

960bool MutableStyleProperties::setProperty(const CSSProperty& property, CSSProperty* slot)
961{
962 if (!removeShorthandProperty(property.id())) {
963 CSSProperty* toReplace = slot;
964 if (!slot) {
965 if (property.id() == CSSPropertyCustom) {
966 if (property.value())
967 toReplace = findCustomCSSPropertyWithName(downcast<CSSCustomPropertyValue>(*property.value()).name());
968 } else
969 toReplace = findCSSPropertyWithID(property.id());
970 }
971
972 if (toReplace) {
973 if (canUpdateInPlace(property, toReplace)) {
974 if (*toReplace == property)
975 return false;
976
977 *toReplace = property;
978 return true;
979 }
980 m_propertyVector.remove(toReplace - m_propertyVector.begin());
981 toReplace = nullptr;
982 }
983 }
984
985 m_propertyVector.append(property);
986 return true;
987}
