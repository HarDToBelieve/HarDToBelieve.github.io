43DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleRuleBase);
44class StyleRuleBase : public WTF::RefCountedBase {
45 WTF_MAKE_STRUCT_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(StyleRuleBase);
46public:
47 StyleRuleType type() const { return static_cast<StyleRuleType>(m_type); }
48
49 bool isCharsetRule() const { return type() == StyleRuleType::Charset; }
50 bool isCounterStyleRule() const { return type() == StyleRuleType::CounterStyle; }
51 bool isFontFaceRule() const { return type() == StyleRuleType::FontFace; }
52 bool isKeyframesRule() const { return type() == StyleRuleType::Keyframes; }
53 bool isKeyframeRule() const { return type() == StyleRuleType::Keyframe; }
54 bool isNamespaceRule() const { return type() == StyleRuleType::Namespace; }
55 bool isMediaRule() const { return type() == StyleRuleType::Media; }
56 bool isPageRule() const { return type() == StyleRuleType::Page; }
57 bool isStyleRule() const { return type() == StyleRuleType::Style; }
58 bool isSupportsRule() const { return type() == StyleRuleType::Supports; }
59 bool isImportRule() const { return type() == StyleRuleType::Import; }
60
61 Ref<StyleRuleBase> copy() const;
62
63 void deref() const;
64
65 // FIXME: There shouldn't be any need for the null parent version.
66 Ref<CSSRule> createCSSOMWrapper(CSSStyleSheet* parentSheet = nullptr) const;
67 Ref<CSSRule> createCSSOMWrapper(CSSRule* parentRule) const;
68
69protected:
70 explicit StyleRuleBase(StyleRuleType, bool hasDocumentSecurityOrigin = false);
71 StyleRuleBase(const StyleRuleBase&);
72 ~StyleRuleBase() = default;
73
74 bool hasDocumentSecurityOrigin() const { return m_hasDocumentSecurityOrigin; }
75
76private:
77 WEBCORE_EXPORT void destroy();
78
79 Ref<CSSRule> createCSSOMWrapper(CSSStyleSheet* parentSheet, CSSRule* parentRule) const;
80
81 unsigned m_type : 5; // StyleRuleType
82 // This is only needed to support getMatchedCSSRules.
83 unsigned m_hasDocumentSecurityOrigin : 1;
84};

285inline void StyleRule::wrapperAdoptSelectorList(CSSSelectorList&& selectors)
286{
287 m_selectorList = WTFMove(selectors);
288#if ENABLE(CSS_SELECTOR_JIT)
289 m_compiledSelectors = nullptr;
290#endif
291}

293#if ENABLE(CSS_SELECTOR_JIT)
294
295inline CompiledSelector& StyleRule::compiledSelectorForListIndex(unsigned index) const
296{
297 if (!m_compiledSelectors)
298 m_compiledSelectors = makeUniqueArray<CompiledSelector>(m_selectorList.listSize());
299 return m_compiledSelectors[index];
300}
301
302#endif

304inline const StyleProperties* StyleRule::propertiesWithoutDeferredParsing() const
305{
306 return m_properties->type() != DeferredPropertiesType ? &downcast<StyleProperties>(m_properties.get()) : nullptr;
307}
308
309inline const Vector<RefPtr<StyleRuleBase>>* StyleRuleGroup::childRulesWithoutDeferredParsing() const
310{
311 return !m_deferredRules ? &m_childRules : nullptr;
312}
