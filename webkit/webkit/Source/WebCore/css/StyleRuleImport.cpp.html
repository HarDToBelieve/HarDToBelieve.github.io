44StyleRuleImport::StyleRuleImport(const String& href, Ref<MediaQuerySet>&& media)
45 : StyleRuleBase(StyleRuleType::Import)
46 , m_styleSheetClient(this)
47 , m_strHref(href)
48 , m_mediaQueries(WTFMove(media))
49{
50 if (!m_mediaQueries)
51 m_mediaQueries = MediaQuerySet::create(String(), MediaQueryParserContext());

54StyleRuleImport::~StyleRuleImport()
55{
56 if (m_styleSheet)
57 m_styleSheet->clearOwnerRule();
58 if (m_cachedSheet)
59 m_cachedSheet->removeClient(m_styleSheetClient);
60}

62void StyleRuleImport::setCSSStyleSheet(const String& href, const URL& baseURL, const String& charset, const CachedCSSStyleSheet* cachedStyleSheet)
63{
64 if (m_styleSheet)
65 m_styleSheet->clearOwnerRule();
66
67 CSSParserContext context = m_parentStyleSheet ? m_parentStyleSheet->parserContext() : HTMLStandardMode;
68 context.charset = charset;
69 if (!baseURL.isNull())
70 context.baseURL = baseURL;
71
72 Document* document = m_parentStyleSheet ? m_parentStyleSheet->singleOwnerDocument() : nullptr;
73 m_styleSheet = StyleSheetContents::create(this, href, context);
74 if (m_parentStyleSheet->isContentOpaque() || !cachedStyleSheet->isCORSSameOrigin())
75 m_styleSheet->setAsOpaque();
76 m_styleSheet->parseAuthorStyleSheet(cachedStyleSheet, document ? &document->securityOrigin() : nullptr);
77
78 m_loading = false;
79
80 if (m_parentStyleSheet) {
81 m_parentStyleSheet->notifyLoadedSheet(cachedStyleSheet);
82 m_parentStyleSheet->checkLoaded();
83 }
84}

86bool StyleRuleImport::isLoading() const
87{
88 return m_loading || (m_styleSheet && m_styleSheet->isLoading());

91void StyleRuleImport::requestStyleSheet()
92{
93 if (!m_parentStyleSheet)
94 return;
95 auto* document = m_parentStyleSheet->singleOwnerDocument();
96 if (!document)
97 return;
98 auto* page = document->page();
99 if (!page)
100 return;
101
102 URL absURL;
103 if (!m_parentStyleSheet->baseURL().isNull())
104 // use parent styleheet's URL as the base URL
105 absURL = URL(m_parentStyleSheet->baseURL(), m_strHref);
106 else
107 absURL = document->completeURL(m_strHref);
108
109 // Check for a cycle in our import chain. If we encounter a stylesheet
110 // in our parent chain with the same URL, then just bail.
111 StyleSheetContents* rootSheet = m_parentStyleSheet;
112 for (StyleSheetContents* sheet = m_parentStyleSheet; sheet; sheet = sheet->parentStyleSheet()) {
113 if (equalIgnoringFragmentIdentifier(absURL, sheet->baseURL())
114 || equalIgnoringFragmentIdentifier(absURL, document->completeURL(sheet->originalURL())))
115 return;
116 rootSheet = sheet;
117 }
118
119 // FIXME: Skip Content Security Policy check when stylesheet is in a user agent shadow tree.
120 // See <https://bugs.webkit.org/show_bug.cgi?id=146663>.
121 CachedResourceRequest request(absURL, CachedResourceLoader::defaultCachedResourceOptions(), std::nullopt, String(m_parentStyleSheet->charset()));
122 request.setInitiator(cachedResourceRequestInitiators().css);
123 if (m_cachedSheet)
124 m_cachedSheet->removeClient(m_styleSheetClient);
125 if (m_parentStyleSheet->isUserStyleSheet()) {
126 ResourceLoaderOptions options {
127 SendCallbackPolicy::DoNotSendCallbacks,
128 ContentSniffingPolicy::SniffContent,
129 DataBufferingPolicy::BufferData,
130 StoredCredentialsPolicy::Use,
131 ClientCredentialPolicy::MayAskClientForCredentials,
132 FetchOptions::Credentials::Include,
133 SecurityCheckPolicy::SkipSecurityCheck,
134 FetchOptions::Mode::NoCors,
135 CertificateInfoPolicy::DoNotIncludeCertificateInfo,
136 ContentSecurityPolicyImposition::SkipPolicyCheck,
137 DefersLoadingPolicy::AllowDefersLoading,
138 CachingPolicy::AllowCaching
139 };
140 options.loadedFromOpaqueSource = m_parentStyleSheet->isContentOpaque() ? LoadedFromOpaqueSource::Yes : LoadedFromOpaqueSource::No;
141
142 request.setOptions(WTFMove(options));
143
144 m_cachedSheet = document->cachedResourceLoader().requestUserCSSStyleSheet(*page, WTFMove(request));
145 } else {
146 auto options = request.options();
147 options.loadedFromOpaqueSource = m_parentStyleSheet->isContentOpaque() ? LoadedFromOpaqueSource::Yes : LoadedFromOpaqueSource::No;
148 request.setOptions(WTFMove(options));
149 m_cachedSheet = document->cachedResourceLoader().requestCSSStyleSheet(WTFMove(request)).value_or(nullptr);
150 }
151 if (m_cachedSheet) {
152 // if the import rule is issued dynamically, the sheet may be
153 // removed from the pending sheet count, so let the doc know
154 // the sheet being imported is pending.
155 if (m_parentStyleSheet && m_parentStyleSheet->loadCompleted() && rootSheet == m_parentStyleSheet)
156 m_parentStyleSheet->startLoadingDynamicSheet();
157 m_loading = true;
158 m_cachedSheet->addClient(m_styleSheetClient);
159 } else if (m_parentStyleSheet) {
160 m_parentStyleSheet->setLoadErrorOccured();
161 m_parentStyleSheet->checkLoaded();
162 }
163}
