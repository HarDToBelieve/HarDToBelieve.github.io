52// Rough size estimate for the memory cache.
53unsigned StyleSheetContents::estimatedSizeInBytes() const
54{
55 // Note that this does not take into account size of the strings hanging from various objects.
56 // The assumption is that nearly all of of them are atoms that would exist anyway.
57 unsigned size = sizeof(*this);
58
59 // FIXME: This ignores the children of media and region rules.
60 // Most rules are StyleRules.
61 size += ruleCount() * StyleRule::averageSizeInBytes();
62
63 for (unsigned i = 0; i < m_importRules.size(); ++i) {
64 if (StyleSheetContents* sheet = m_importRules[i]->styleSheet())
65 size += sheet->estimatedSizeInBytes();
66 }
67 return size;
68}

109bool StyleSheetContents::isCacheable() const
110{
111 // FIXME: Support copying import rules.
112 if (!m_importRules.isEmpty())
113 return false;
114 // FIXME: Support cached stylesheets in import rules.
115 if (m_ownerRule)
116 return false;
117 // This would require dealing with multiple clients for load callbacks.
118 if (!m_loadCompleted)
119 return false;
120 if (m_didLoadErrorOccur)
121 return false;
122 // It is not the original sheet anymore.
123 if (m_isMutable)
124 return false;
125 // If the header is valid we are not going to need to check the SecurityOrigin.
126 // FIXME: Valid mime type avoids the check too.
127 if (!m_hasSyntacticallyValidCSSHeader)
128 return false;
129 return true;
130}

132void StyleSheetContents::parserAppendRule(Ref<StyleRuleBase>&& rule)
133{
134 ASSERT(!rule->isCharsetRule());
135
136 if (is<StyleRuleImport>(rule)) {
137 // Parser enforces that @import rules come before anything else except @charset.
138 ASSERT(m_childRules.isEmpty());
139 m_importRules.append(downcast<StyleRuleImport>(rule.ptr()));
140 m_importRules.last()->setParentStyleSheet(this);
141 m_importRules.last()->requestStyleSheet();
142 return;
143 }
144
145 if (is<StyleRuleNamespace>(rule)) {
146 // Parser enforces that @namespace rules come before all rules other than
147 // import/charset rules
148 ASSERT(m_childRules.isEmpty());
149 StyleRuleNamespace& namespaceRule = downcast<StyleRuleNamespace>(rule.get());
150 parserAddNamespace(namespaceRule.prefix(), namespaceRule.uri());
151 m_namespaceRules.append(downcast<StyleRuleNamespace>(rule.ptr()));
152 return;
153 }
154
155 if (is<StyleRuleMedia>(rule))
156 reportMediaQueryWarningIfNeeded(singleOwnerDocument(), &downcast<StyleRuleMedia>(rule.get()).mediaQueries());
157
158 // NOTE: The selector list has to fit into RuleData. <http://webkit.org/b/118369>
159 // If we're adding a rule with a huge number of selectors, split it up into multiple rules
160 if (is<StyleRule>(rule) && downcast<StyleRule>(rule.get()).selectorList().componentCount() > Style::RuleData::maximumSelectorComponentCount) {
161 m_childRules.appendVector(downcast<StyleRule>(rule.get()).splitIntoMultipleRulesWithMaximumSelectorComponentCount(Style::RuleData::maximumSelectorComponentCount));
162 return;
163 }
164
165 m_childRules.append(WTFMove(rule));
166}

168StyleRuleBase* StyleSheetContents::ruleAt(unsigned index) const
169{
170 ASSERT_WITH_SECURITY_IMPLICATION(index < ruleCount());
171
172 unsigned childVectorIndex = index;
173 if (childVectorIndex < m_importRules.size())
174 return m_importRules[childVectorIndex].get();
175
176 childVectorIndex -= m_importRules.size();
177
178 if (childVectorIndex < m_namespaceRules.size())
179 return m_namespaceRules[childVectorIndex].get();
180
181 childVectorIndex -= m_namespaceRules.size();
182
183 return m_childRules[childVectorIndex].get();
184}

186unsigned StyleSheetContents::ruleCount() const
187{
188 unsigned result = 0;
189 result += m_importRules.size();
190 result += m_namespaceRules.size();
191 result += m_childRules.size();
192 return result;
193}

219bool StyleSheetContents::wrapperInsertRule(Ref<StyleRuleBase>&& rule, unsigned index)
220{
221 ASSERT(m_isMutable);
222 ASSERT_WITH_SECURITY_IMPLICATION(index <= ruleCount());
223 // Parser::parseRule doesn't currently allow @charset so we don't need to deal with it.
224 ASSERT(!rule->isCharsetRule());
225
226 unsigned childVectorIndex = index;
227 if (childVectorIndex < m_importRules.size() || (childVectorIndex == m_importRules.size() && rule->isImportRule())) {
228 // Inserting non-import rule before @import is not allowed.
229 if (!is<StyleRuleImport>(rule))
230 return false;
231 m_importRules.insert(childVectorIndex, downcast<StyleRuleImport>(rule.ptr()));
232 m_importRules[childVectorIndex]->setParentStyleSheet(this);
233 m_importRules[childVectorIndex]->requestStyleSheet();
234 // FIXME: Stylesheet doesn't actually change meaningfully before the imported sheets are loaded.
235 return true;
236 }
237 // Inserting @import rule after a non-import rule is not allowed.
238 if (is<StyleRuleImport>(rule))
239 return false;
240 childVectorIndex -= m_importRules.size();
241
242
243 if (childVectorIndex < m_namespaceRules.size() || (childVectorIndex == m_namespaceRules.size() && rule->isNamespaceRule())) {
244 // Inserting non-namespace rules other than import rule before @namespace is
245 // not allowed.
246 if (!is<StyleRuleNamespace>(rule))
247 return false;
248 // Inserting @namespace rule when rules other than import/namespace/charset
249 // are present is not allowed.
250 if (!m_childRules.isEmpty())
251 return false;
252
253 StyleRuleNamespace& namespaceRule = downcast<StyleRuleNamespace>(rule.get());
254 m_namespaceRules.insert(index, downcast<StyleRuleNamespace>(rule.ptr()));
255
256 // For now to be compatible with IE and Firefox if a namespace rule with the same
257 // prefix is added, it overwrites previous ones.
258 // FIXME: The eventual correct behavior would be to ensure that the last value in
259 // the list wins.
260 parserAddNamespace(namespaceRule.prefix(), namespaceRule.uri());
261 return true;
262 }
263 if (is<StyleRuleNamespace>(rule))
264 return false;
265 childVectorIndex -= m_namespaceRules.size();
266
267 // If the number of selectors would overflow RuleData, we drop the operation.
268 if (is<StyleRule>(rule) && downcast<StyleRule>(rule.get()).selectorList().componentCount() > Style::RuleData::maximumSelectorComponentCount)
269 return false;
270
271 m_childRules.insert(childVectorIndex, WTFMove(rule));
272 return true;
273}

275void StyleSheetContents::wrapperDeleteRule(unsigned index)
276{
277 ASSERT(m_isMutable);
278 ASSERT_WITH_SECURITY_IMPLICATION(index < ruleCount());
279
280 unsigned childVectorIndex = index;
281 if (childVectorIndex < m_importRules.size()) {
282 m_importRules[childVectorIndex]->clearParentStyleSheet();
283 m_importRules.remove(childVectorIndex);
284 return;
285 }
286 childVectorIndex -= m_importRules.size();
287
288 if (childVectorIndex < m_namespaceRules.size()) {
289 if (!m_childRules.isEmpty())
290 return;
291 m_namespaceRules.remove(childVectorIndex);
292 return;
293 }
294 childVectorIndex -= m_namespaceRules.size();
295
296 m_childRules.remove(childVectorIndex);
297}

299void StyleSheetContents::parserAddNamespace(const AtomString& prefix, const AtomString& uri)
300{
301 ASSERT(!uri.isNull());
302 if (prefix.isNull()) {
303 m_defaultNamespace = uri;
304 return;
305 }
306 PrefixNamespaceURIMap::AddResult result = m_namespaces.add(prefix, uri);
307 if (result.isNewEntry)
308 return;
309 result.iterator->value = uri;
310}

312const AtomString& StyleSheetContents::namespaceURIFromPrefix(const AtomString& prefix)
313{
314 PrefixNamespaceURIMap::const_iterator it = m_namespaces.find(prefix);
315 if (it == m_namespaces.end())
316 return nullAtom();
317 return it->value;

320bool StyleSheetContents::parseAuthorStyleSheet(const CachedCSSStyleSheet* cachedStyleSheet, const SecurityOrigin* securityOrigin)
321{
322 bool isSameOriginRequest = securityOrigin && securityOrigin->canRequest(baseURL());
323 CachedCSSStyleSheet::MIMETypeCheckHint mimeTypeCheckHint = isStrictParserMode(m_parserContext.mode) || !isSameOriginRequest ? CachedCSSStyleSheet::MIMETypeCheckHint::Strict : CachedCSSStyleSheet::MIMETypeCheckHint::Lax;
324 bool hasValidMIMEType = true;
325 String sheetText = cachedStyleSheet->sheetText(mimeTypeCheckHint, &hasValidMIMEType);
326
327 if (!hasValidMIMEType) {
328 ASSERT(sheetText.isNull());
329 if (auto* document = singleOwnerDocument()) {
330 if (auto* page = document->page()) {
331 if (isStrictParserMode(m_parserContext.mode))
332 page->console().addMessage(MessageSource::Security, MessageLevel::Error, makeString("Did not parse stylesheet at '", cachedStyleSheet->url().stringCenterEllipsizedToLength(), "' because non CSS MIME types are not allowed in strict mode."));
333 else if (!cachedStyleSheet->mimeTypeAllowedByNosniff())
334 page->console().addMessage(MessageSource::Security, MessageLevel::Error, makeString("Did not parse stylesheet at '", cachedStyleSheet->url().stringCenterEllipsizedToLength(), "' because non CSS MIME types are not allowed when 'X-Content-Type-Options: nosniff' is given."));
335 else
336 page->console().addMessage(MessageSource::Security, MessageLevel::Error, makeString("Did not parse stylesheet at '", cachedStyleSheet->url().stringCenterEllipsizedToLength(), "' because non CSS MIME types are not allowed for cross-origin stylesheets."));
337 }
338 }
339 return false;
340 }
341
342 CSSParser(parserContext()).parseSheet(this, sheetText, CSSParser::RuleParsing::Deferred);
343 return true;
344}

362void StyleSheetContents::checkLoaded()
363{
364 if (isLoading())
365 return;
366
367 Ref<StyleSheetContents> protectedThis(*this);
368 StyleSheetContents* parentSheet = parentStyleSheet();
369 if (parentSheet) {
370 parentSheet->checkLoaded();
371 m_loadCompleted = true;
372 return;
373 }
374 RefPtr<Node> ownerNode = singleOwnerNode();
375 if (!ownerNode) {
376 m_loadCompleted = true;
377 return;
378 }
379 m_loadCompleted = ownerNode->sheetLoaded();
380 if (m_loadCompleted)
381 ownerNode->notifyLoadedSheetAndAllCriticalSubresources(m_didLoadErrorOccur);
382}

420static bool traverseRulesInVector(const Vector<RefPtr<StyleRuleBase>>& rules, const WTF::Function<bool (const StyleRuleBase&)>& handler)
421{
422 for (auto& rule : rules) {
423 if (handler(*rule))
424 return true;
425 switch (rule->type()) {
426 case StyleRuleType::Media: {
427 auto* childRules = downcast<StyleRuleMedia>(*rule).childRulesWithoutDeferredParsing();
428 if (childRules && traverseRulesInVector(*childRules, handler))
429 return true;
430 break;
431 }
432 case StyleRuleType::Import:
433 ASSERT_NOT_REACHED();
434 break;
435 case StyleRuleType::Style:
436 case StyleRuleType::FontFace:
437 case StyleRuleType::Page:
438 case StyleRuleType::Keyframes:
439 case StyleRuleType::Namespace:
440 case StyleRuleType::Unknown:
441 case StyleRuleType::Charset:
442 case StyleRuleType::CounterStyle:
443 case StyleRuleType::Keyframe:
444 case StyleRuleType::Supports:
445 break;
446 }
447 }
448 return false;
449}

451bool StyleSheetContents::traverseRules(const WTF::Function<bool (const StyleRuleBase&)>& handler) const
452{
453 for (auto& importRule : m_importRules) {
454 if (handler(*importRule))
455 return true;
456 auto* importedStyleSheet = importRule->styleSheet();
457 if (importedStyleSheet && importedStyleSheet->traverseRules(handler))
458 return true;
459 }
460 return traverseRulesInVector(m_childRules, handler);
461}

463bool StyleSheetContents::traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const
464{
465 return traverseRules([&] (const StyleRuleBase& rule) {
466 switch (rule.type()) {
467 case StyleRuleType::Style: {
468 auto* properties = downcast<StyleRule>(rule).propertiesWithoutDeferredParsing();
469 return properties && properties->traverseSubresources(handler);
470 }
471 case StyleRuleType::FontFace:
472 return downcast<StyleRuleFontFace>(rule).properties().traverseSubresources(handler);
473 case StyleRuleType::Import:
474 if (auto* cachedResource = downcast<StyleRuleImport>(rule).cachedCSSStyleSheet())
475 return handler(*cachedResource);
476 return false;
477 case StyleRuleType::CounterStyle:
478 return m_parserContext.counterStyleAtRuleImageSymbolsEnabled;
479 case StyleRuleType::Media:
480 case StyleRuleType::Page:
481 case StyleRuleType::Keyframes:
482 case StyleRuleType::Namespace:
483 case StyleRuleType::Unknown:
484 case StyleRuleType::Charset:
485 case StyleRuleType::Keyframe:
486 case StyleRuleType::Supports:
487 return false;
488 };
489 ASSERT_NOT_REACHED();
490 return false;
491 });
492}

494bool StyleSheetContents::subresourcesAllowReuse(CachePolicy cachePolicy, FrameLoader& loader) const
495{
496 bool hasFailedOrExpiredResources = traverseSubresources([cachePolicy, &loader](const CachedResource& resource) {
497 if (resource.loadFailedOrCanceled())
498 return true;
499 // We can't revalidate subresources individually so don't use reuse the parsed sheet if they need revalidation.
500 if (resource.makeRevalidationDecision(cachePolicy) != CachedResource::RevalidationDecision::No)
501 return true;
502
503#if ENABLE(CONTENT_EXTENSIONS)
504 // If a cached subresource is blocked or made HTTPS by a content blocker, we cannot reuse the cached stylesheet.
505 auto* page = loader.frame().page();
506 auto* documentLoader = loader.documentLoader();
507 if (page && documentLoader) {
508 const auto& request = resource.resourceRequest();
509 auto results = page->userContentProvider().processContentRuleListsForLoad(*page, request.url(), ContentExtensions::toResourceType(resource.type(), resource.resourceRequest().requester()), *documentLoader);
510 if (results.summary.blockedLoad || results.summary.madeHTTPS)
511 return true;
512 }
513#else
514 UNUSED_PARAM(loader);
515#endif
516
517 return false;
518 });
519 return !hasFailedOrExpiredResources;
520}
