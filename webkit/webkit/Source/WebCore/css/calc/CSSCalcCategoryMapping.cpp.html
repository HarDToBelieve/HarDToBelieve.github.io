34CalculationCategory calcUnitCategory(CSSUnitType type)
35{
36 switch (type) {
37 case CSSUnitType::CSS_NUMBER:
38 return CalculationCategory::Number;
39 case CSSUnitType::CSS_EMS:
40 case CSSUnitType::CSS_EXS:
41 case CSSUnitType::CSS_PX:
42 case CSSUnitType::CSS_CM:
43 case CSSUnitType::CSS_MM:
44 case CSSUnitType::CSS_IN:
45 case CSSUnitType::CSS_PT:
46 case CSSUnitType::CSS_PC:
47 case CSSUnitType::CSS_Q:
48 case CSSUnitType::CSS_LHS:
49 case CSSUnitType::CSS_RLHS:
50 case CSSUnitType::CSS_REMS:
51 case CSSUnitType::CSS_CHS:
52 case CSSUnitType::CSS_VW:
53 case CSSUnitType::CSS_VH:
54 case CSSUnitType::CSS_VMIN:
55 case CSSUnitType::CSS_VMAX:
56 return CalculationCategory::Length;
57 case CSSUnitType::CSS_PERCENTAGE:
58 return CalculationCategory::Percent;
59 case CSSUnitType::CSS_DEG:
60 case CSSUnitType::CSS_RAD:
61 case CSSUnitType::CSS_GRAD:
62 case CSSUnitType::CSS_TURN:
63 return CalculationCategory::Angle;
64 case CSSUnitType::CSS_MS:
65 case CSSUnitType::CSS_S:
66 return CalculationCategory::Time;
67 case CSSUnitType::CSS_HZ:
68 case CSSUnitType::CSS_KHZ:
69 return CalculationCategory::Frequency;
70 default:
71 return CalculationCategory::Other;
72 }
73}

116CSSUnitType canonicalUnitTypeForCalculationCategory(CalculationCategory category)
117{
118 switch (category) {
119 case CalculationCategory::Number: return CSSUnitType::CSS_NUMBER;
120 case CalculationCategory::Length: return CSSUnitType::CSS_PX;
121 case CalculationCategory::Percent: return CSSUnitType::CSS_PERCENTAGE;
122 case CalculationCategory::Angle: return CSSUnitType::CSS_DEG;
123 case CalculationCategory::Time: return CSSUnitType::CSS_MS;
124 case CalculationCategory::Frequency: return CSSUnitType::CSS_HZ;
125 case CalculationCategory::Other:
126 case CalculationCategory::PercentNumber:
127 case CalculationCategory::PercentLength:
128 ASSERT_NOT_REACHED();
129 break;
130 }
131 return CSSUnitType::CSS_UNKNOWN;
132}

134bool hasDoubleValue(CSSUnitType type)
135{
136 switch (type) {
137 case CSSUnitType::CSS_NUMBER:
138 case CSSUnitType::CSS_PERCENTAGE:
139 case CSSUnitType::CSS_EMS:
140 case CSSUnitType::CSS_EXS:
141 case CSSUnitType::CSS_CHS:
142 case CSSUnitType::CSS_REMS:
143 case CSSUnitType::CSS_PX:
144 case CSSUnitType::CSS_CM:
145 case CSSUnitType::CSS_MM:
146 case CSSUnitType::CSS_IN:
147 case CSSUnitType::CSS_PT:
148 case CSSUnitType::CSS_PC:
149 case CSSUnitType::CSS_DEG:
150 case CSSUnitType::CSS_RAD:
151 case CSSUnitType::CSS_GRAD:
152 case CSSUnitType::CSS_TURN:
153 case CSSUnitType::CSS_MS:
154 case CSSUnitType::CSS_S:
155 case CSSUnitType::CSS_HZ:
156 case CSSUnitType::CSS_KHZ:
157 case CSSUnitType::CSS_DIMENSION:
158 case CSSUnitType::CSS_VW:
159 case CSSUnitType::CSS_VH:
160 case CSSUnitType::CSS_VMIN:
161 case CSSUnitType::CSS_VMAX:
162 case CSSUnitType::CSS_DPPX:
163 case CSSUnitType::CSS_DPI:
164 case CSSUnitType::CSS_DPCM:
165 case CSSUnitType::CSS_FR:
166 case CSSUnitType::CSS_Q:
167 case CSSUnitType::CSS_LHS:
168 case CSSUnitType::CSS_RLHS:
169 return true;
170 case CSSUnitType::CSS_UNKNOWN:
171 case CSSUnitType::CSS_STRING:
172 case CSSUnitType::CSS_FONT_FAMILY:
173 case CSSUnitType::CSS_URI:
174 case CSSUnitType::CSS_IDENT:
175 case CSSUnitType::CustomIdent:
176 case CSSUnitType::CSS_ATTR:
177 case CSSUnitType::CSS_COUNTER:
178 case CSSUnitType::CSS_RECT:
179 case CSSUnitType::CSS_RGBCOLOR:
180 case CSSUnitType::CSS_PAIR:
181 case CSSUnitType::CSS_UNICODE_RANGE:
182 case CSSUnitType::CSS_COUNTER_NAME:
183 case CSSUnitType::CSS_SHAPE:
184 case CSSUnitType::CSS_QUAD:
185 case CSSUnitType::CSS_QUIRKY_EMS:
186 case CSSUnitType::CSS_CALC:
187 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_NUMBER:
188 case CSSUnitType::CSS_CALC_PERCENTAGE_WITH_LENGTH:
189 case CSSUnitType::CSS_PROPERTY_ID:
190 case CSSUnitType::CSS_VALUE_ID:
191 return false;
192 };
193 ASSERT_NOT_REACHED();
194 return false;
195}
