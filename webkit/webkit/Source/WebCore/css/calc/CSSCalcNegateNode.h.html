26#pragma once
27
28#include "CSSCalcExpressionNode.h"
29
30namespace WebCore {
31
32class CSSCalcNegateNode final : public CSSCalcExpressionNode {
33public:
34 static Ref<CSSCalcNegateNode> create(Ref<CSSCalcExpressionNode>&& child)
35 {
36 return adoptRef(*new CSSCalcNegateNode(WTFMove(child)));
37 }
38
39 const CSSCalcExpressionNode& child() const { return m_child.get(); }
40 CSSCalcExpressionNode& child() { return m_child.get(); }
41
42 void setChild(Ref<CSSCalcExpressionNode>&& child) { m_child = WTFMove(child); }
43
44private:
45 CSSCalcNegateNode(Ref<CSSCalcExpressionNode>&& child)
46 : CSSCalcExpressionNode(child->category())
47 , m_child(WTFMove(child))
48 {
49 }
50
51 std::unique_ptr<CalcExpressionNode> createCalcExpression(const CSSToLengthConversionData&) const final;
52
53 bool isZero() const final { return m_child->isZero(); }
54 double doubleValue(CSSUnitType unitType) const final { return -m_child->doubleValue(unitType); }
55 double computeLengthPx(const CSSToLengthConversionData& conversionData) const final { return -m_child->computeLengthPx(conversionData); }
56 Type type() const final { return Type::CssCalcNegate; }
57 CSSUnitType primitiveType() const final { return m_child->primitiveType(); }
58
59 void collectDirectComputationalDependencies(HashSet<CSSPropertyID>& properties) const final { m_child->collectDirectComputationalDependencies(properties); }
60 void collectDirectRootComputationalDependencies(HashSet<CSSPropertyID>& properties) const final { m_child->collectDirectRootComputationalDependencies(properties); }
61
62 void dump(TextStream&) const final;
63
64 Ref<CSSCalcExpressionNode> m_child;
65};
66
67}

69SPECIALIZE_TYPE_TRAITS_CSSCALCEXPRESSION_NODE(CSSCalcNegateNode, type() == WebCore::CSSCalcExpressionNode::Type::CssCalcNegate)
