43static const CalculationCategory addSubtractResult[static_cast<unsigned>(CalculationCategory::Angle)][static_cast<unsigned>(CalculationCategory::Angle)] = {
44// CalculationCategory::Number CalculationCategory::Length CalculationCategory::Percent CalculationCategory::PercentNumber CalculationCategory::PercentLength
45 { CalculationCategory::Number, CalculationCategory::Other, CalculationCategory::PercentNumber, CalculationCategory::PercentNumber, CalculationCategory::Other }, // CalculationCategory::Number
46 { CalculationCategory::Other, CalculationCategory::Length, CalculationCategory::PercentLength, CalculationCategory::Other, CalculationCategory::PercentLength }, // CalculationCategory::Length
47 { CalculationCategory::PercentNumber, CalculationCategory::PercentLength, CalculationCategory::Percent, CalculationCategory::PercentNumber, CalculationCategory::PercentLength }, // CalculationCategory::Percent
48 { CalculationCategory::PercentNumber, CalculationCategory::Other, CalculationCategory::PercentNumber, CalculationCategory::PercentNumber, CalculationCategory::Other }, // CalculationCategory::PercentNumber
49 { CalculationCategory::Other, CalculationCategory::PercentLength, CalculationCategory::PercentLength, CalculationCategory::Other, CalculationCategory::PercentLength }, // CalculationCategory::PercentLength
50};

52static CalculationCategory determineCategory(const CSSCalcExpressionNode& leftSide, const CSSCalcExpressionNode& rightSide, CalcOperator op)
53{
54 CalculationCategory leftCategory = leftSide.category();
55 CalculationCategory rightCategory = rightSide.category();
56 ASSERT(leftCategory < CalculationCategory::Other);
57 ASSERT(rightCategory < CalculationCategory::Other);
58
59 switch (op) {
60 case CalcOperator::Add:
61 case CalcOperator::Subtract:
62 if (leftCategory < CalculationCategory::Angle && rightCategory < CalculationCategory::Angle)
63 return addSubtractResult[static_cast<unsigned>(leftCategory)][static_cast<unsigned>(rightCategory)];
64 if (leftCategory == rightCategory)
65 return leftCategory;
66 return CalculationCategory::Other;
67 case CalcOperator::Multiply:
68 if (leftCategory != CalculationCategory::Number && rightCategory != CalculationCategory::Number)
69 return CalculationCategory::Other;
70 return leftCategory == CalculationCategory::Number ? rightCategory : leftCategory;
71 case CalcOperator::Divide:
72 if (rightCategory != CalculationCategory::Number || rightSide.isZero())
73 return CalculationCategory::Other;
74 return leftCategory;
75 case CalcOperator::Min:
76 case CalcOperator::Max:
77 case CalcOperator::Clamp:
78 ASSERT_NOT_REACHED();
79 return CalculationCategory::Other;
80 }
81
82 ASSERT_NOT_REACHED();
83 return CalculationCategory::Other;
84}

96static CalculationCategory determineCategory(const Vector<Ref<CSSCalcExpressionNode>>& nodes, CalcOperator op)
97{
98 if (nodes.isEmpty())
99 return CalculationCategory::Other;
100
101 auto currentCategory = nodes[0]->category();
102
103 for (unsigned i = 1; i < nodes.size(); ++i) {
104 const auto& node = nodes[i].get();
105
106 auto usedOperator = op;
107 if (node.type() == CSSCalcExpressionNode::Type::CssCalcInvert)
108 usedOperator = CalcOperator::Divide;
109
110 auto nextCategory = node.category();
111
112 switch (usedOperator) {
113 case CalcOperator::Add:
114 case CalcOperator::Subtract:
115 // <https://drafts.css-houdini.org/css-typed-om-1/#cssnumericvalue-add-two-types>
116 // At a + or - sub-expression, attempt to add the types of the left and right arguments.
117 // If this returns failure, the entire calculationâ€™s type is failure. Otherwise, the sub-expressionâ€™s type is the returned type.
118 if (currentCategory < CalculationCategory::Angle && nextCategory < CalculationCategory::Angle)
119 currentCategory = addSubtractResult[static_cast<unsigned>(currentCategory)][static_cast<unsigned>(nextCategory)];
120 else if (currentCategory != nextCategory)
121 return CalculationCategory::Other;
122 break;
123
124 case CalcOperator::Multiply:
125 // <https://drafts.css-houdini.org/css-typed-om-1/#cssnumericvalue-multiply-two-types>
126 // At a * sub-expression, multiply the types of the left and right arguments. The sub-expressionâ€™s type is the returned result.
127 if (currentCategory != CalculationCategory::Number && nextCategory != CalculationCategory::Number)
128 return CalculationCategory::Other;
129
130 currentCategory = currentCategory == CalculationCategory::Number ? nextCategory : currentCategory;
131 break;
132
133 case CalcOperator::Divide: {
134 auto invertCategory = categoryForInvert(nextCategory);
135
136 // At a / sub-expression, let left type be the result of finding the types of its left argument,
137 // and right type be the result of finding the types of its right argument and then inverting it.
138 // The sub-expressionâ€™s type is the result of multiplying the left type and right type.
139 if (invertCategory != CalculationCategory::Number || node.isZero())
140 return CalculationCategory::Other;
141 break;
142 }
143
144 case CalcOperator::Min:
145 case CalcOperator::Max:
146 case CalcOperator::Clamp:
147 // The type of a min(), max(), or clamp() expression is the result of adding the types of its comma-separated calculations
148 return CalculationCategory::Other;
149 }
150 }
151
152 return currentCategory;
153}

155static CalculationCategory resolvedTypeForMinOrMaxOrClamp(CalculationCategory category, CalculationCategory destinationCategory)
156{
157 switch (category) {
158 case CalculationCategory::Number:
159 case CalculationCategory::Length:
160 case CalculationCategory::PercentNumber:
161 case CalculationCategory::PercentLength:
162 case CalculationCategory::Angle:
163 case CalculationCategory::Time:
164 case CalculationCategory::Frequency:
165 case CalculationCategory::Other:
166 return category;
167
168 case CalculationCategory::Percent:
169 if (destinationCategory == CalculationCategory::Length)
170 return CalculationCategory::PercentLength;
171 if (destinationCategory == CalculationCategory::Number)
172 return CalculationCategory::PercentNumber;
173 return category;
174 }
175
176 return CalculationCategory::Other;
177}

179static bool isSamePair(CalculationCategory a, CalculationCategory b, CalculationCategory x, CalculationCategory y)
180{
181 return (a == x && b == y) || (a == y && b == x);

191static SortingCategory sortingCategoryForType(CSSUnitType unitType)
192{
193 static constexpr SortingCategory sortOrder[] = {
194 SortingCategory::Number, // CalculationCategory::Number,
195 SortingCategory::Dimension, // CalculationCategory::Length,
196 SortingCategory::Percent, // CalculationCategory::Percent,
197 SortingCategory::Number, // CalculationCategory::PercentNumber,
198 SortingCategory::Dimension, // CalculationCategory::PercentLength,
199 SortingCategory::Dimension, // CalculationCategory::Angle,
200 SortingCategory::Dimension, // CalculationCategory::Time,
201 SortingCategory::Dimension, // CalculationCategory::Frequency,
202 SortingCategory::Other, // UOther
203 };
204
205 COMPILE_ASSERT(ARRAY_SIZE(sortOrder) == static_cast<unsigned>(CalculationCategory::Other) + 1, sortOrder_size_should_match_UnitCategory);
206 return sortOrder[static_cast<unsigned>(calcUnitCategory(unitType))];
207}

225static CSSCalcPrimitiveValueNode::UnitConversion conversionToAddValuesWithTypes(CSSUnitType firstType, CSSUnitType secondType)
226{
227 if (firstType == CSSUnitType::CSS_UNKNOWN || secondType == CSSUnitType::CSS_UNKNOWN)
228 return CSSCalcPrimitiveValueNode::UnitConversion::Invalid;
229
230 auto firstCategory = calculationCategoryForCombination(firstType);
231
232 // Compatible types.
233 if (firstCategory != CalculationCategory::Other && firstCategory == calculationCategoryForCombination(secondType))
234 return CSSCalcPrimitiveValueNode::UnitConversion::Canonicalize;
235
236 // Matching types.
237 if (firstType == secondType && hasDoubleValue(firstType))
238 return CSSCalcPrimitiveValueNode::UnitConversion::Preserve;
239
240 return CSSCalcPrimitiveValueNode::UnitConversion::Invalid;
241}

262RefPtr<CSSCalcOperationNode> CSSCalcOperationNode::create(CalcOperator op, RefPtr<CSSCalcExpressionNode>&& leftSide, RefPtr<CSSCalcExpressionNode>&& rightSide)
263{
264 if (!leftSide || !rightSide)
265 return nullptr;
266
267 ASSERT(op == CalcOperator::Add || op == CalcOperator::Multiply);
268
269 ASSERT(leftSide->category() < CalculationCategory::Other);
270 ASSERT(rightSide->category() < CalculationCategory::Other);
271
272 auto newCategory = determineCategory(*leftSide, *rightSide, op);
273 if (newCategory == CalculationCategory::Other) {
274 LOG_WITH_STREAM(Calc, stream << "Failed to create CSSCalcOperationNode " << op << " node because unable to determine category from " << prettyPrintNode(*leftSide) << " and " << *rightSide);
275 return nullptr;
276 }
277
278 return adoptRef(new CSSCalcOperationNode(newCategory, op, leftSide.releaseNonNull(), rightSide.releaseNonNull()));
279}

310RefPtr<CSSCalcOperationNode> CSSCalcOperationNode::createMinOrMaxOrClamp(CalcOperator op, Vector<Ref<CSSCalcExpressionNode>>&& values, CalculationCategory destinationCategory)
311{
312 ASSERT(op == CalcOperator::Min || op == CalcOperator::Max || op == CalcOperator::Clamp);
313 ASSERT_IMPLIES(op == CalcOperator::Clamp, values.size() == 3);
314
315 std::optional<CalculationCategory> category = std::nullopt;
316 for (auto& value : values) {
317 auto valueCategory = resolvedTypeForMinOrMaxOrClamp(value->category(), destinationCategory);
318
319 ASSERT(valueCategory < CalculationCategory::Other);
320 if (!category) {
321 if (valueCategory == CalculationCategory::Other) {
322 LOG_WITH_STREAM(Calc, stream << "Failed to create CSSCalcOperationNode " << op << " node because unable to determine category from " << prettyPrintNodes(values));
323 return nullptr;
324 }
325 category = valueCategory;
326 }
327
328 if (category != valueCategory) {
329 if (isSamePair(category.value(), valueCategory, CalculationCategory::Length, CalculationCategory::PercentLength)) {
330 category = CalculationCategory::PercentLength;
331 continue;
332 }
333 if (isSamePair(category.value(), valueCategory, CalculationCategory::Number, CalculationCategory::PercentNumber)) {
334 category = CalculationCategory::PercentNumber;
335 continue;
336 }
337 return nullptr;
338 }
339 }
340
341 return adoptRef(new CSSCalcOperationNode(category.value(), op, WTFMove(values)));
342}

344void CSSCalcOperationNode::hoistChildrenWithOperator(CalcOperator op)
345{
346 ASSERT(op == CalcOperator::Add || op == CalcOperator::Multiply);
347
348 auto hasChildWithOperator = [&] (CalcOperator op) {
349 for (auto& child : m_children) {
350 if (is<CSSCalcOperationNode>(child.get()) && downcast<CSSCalcOperationNode>(child.get()).calcOperator() == op)
351 return true;
352 }
353 return false;
354 };
355
356 if (!hasChildWithOperator(op))
357 return;
358
359 Vector<Ref<CSSCalcExpressionNode>> newChildren;
360 for (auto& child : m_children) {
361 if (is<CSSCalcOperationNode>(child.get()) && downcast<CSSCalcOperationNode>(child.get()).calcOperator() == op) {
362 auto& children = downcast<CSSCalcOperationNode>(child.get()).children();
363 for (auto& childToMove : children)
364 newChildren.append(WTFMove(childToMove));
365 } else
366 newChildren.append(WTFMove(child));
367 }
368
369 newChildren.shrinkToFit();
370 m_children = WTFMove(newChildren);
371}

373bool CSSCalcOperationNode::canCombineAllChildren() const
374{
375 if (m_children.size() < 2)
376 return false;
377
378 if (!is<CSSCalcPrimitiveValueNode>(m_children[0]))
379 return false;
380
381 auto firstUnitType = m_children[0]->primitiveType();
382 auto firstCategory = calculationCategoryForCombination(m_children[0]->primitiveType());
383
384 for (unsigned i = 1; i < m_children.size(); ++i) {
385 auto& node = m_children[i];
386
387 if (!is<CSSCalcPrimitiveValueNode>(node))
388 return false;
389
390 auto nodeUnitType = node->primitiveType();
391 auto nodeCategory = calculationCategoryForCombination(nodeUnitType);
392
393 if (nodeCategory != firstCategory)
394 return false;
395
396 if (nodeCategory == CalculationCategory::Other && nodeUnitType != firstUnitType)
397 return false;
398
399 if (!hasDoubleValue(nodeUnitType))
400 return false;
401 }
402
403 return true;
404}

406void CSSCalcOperationNode::combineChildren()
407{
408 if (m_children.size() < 2)
409 return;
410
411 if (shouldSortChildren()) {
412 // <https://drafts.csswg.org/css-values-4/#sort-a-calculations-children>
413 std::stable_sort(m_children.begin(), m_children.end(), [](const auto& first, const auto& second) {
414 // Sort order: number, percentage, dimension, other.
415 SortingCategory firstCategory = sortingCategory(first.get());
416 SortingCategory secondCategory = sortingCategory(second.get());
417
418 if (firstCategory == SortingCategory::Dimension && secondCategory == SortingCategory::Dimension) {
419 // If nodes contains any dimensions, remove them from nodes, sort them by their units, and append them to ret.
420 auto firstUnitString = CSSPrimitiveValue::unitTypeString(first->primitiveType());
421 auto secondUnitString = CSSPrimitiveValue::unitTypeString(second->primitiveType());
422 return codePointCompareLessThan(firstUnitString, secondUnitString);
423 }
424
425 return static_cast<unsigned>(firstCategory) < static_cast<unsigned>(secondCategory);
426 });
427
428 LOG_WITH_STREAM(Calc, stream << "post-sort: " << *this);
429 }
430
431 if (calcOperator() == CalcOperator::Add) {
432 // For each set of rootâ€™s children that are numeric values with identical units,
433 // remove those children and replace them with a single numeric value containing
434 // the sum of the removed nodes, and with the same unit.
435 Vector<Ref<CSSCalcExpressionNode>> newChildren;
436 newChildren.reserveInitialCapacity(m_children.size());
437 newChildren.uncheckedAppend(m_children[0].copyRef());
438
439 CSSUnitType previousType = primitiveTypeForCombination(newChildren[0].get());
440
441 for (unsigned i = 1; i < m_children.size(); ++i) {
442 auto& currentNode = m_children[i];
443 CSSUnitType currentType = primitiveTypeForCombination(currentNode.get());
444
445 auto conversionType = conversionToAddValuesWithTypes(previousType, currentType);
446 if (conversionType != CSSCalcPrimitiveValueNode::UnitConversion::Invalid) {
447 downcast<CSSCalcPrimitiveValueNode>(newChildren.last().get()).add(downcast<CSSCalcPrimitiveValueNode>(currentNode.get()), conversionType);
448 continue;
449 }
450
451 previousType = primitiveTypeForCombination(currentNode);
452 newChildren.uncheckedAppend(currentNode.copyRef());
453 }
454
455 newChildren.shrinkToFit();
456 m_children = WTFMove(newChildren);
457 return;
458 }
459
460 if (calcOperator() == CalcOperator::Multiply) {
461 // If root has multiple children that are numbers (not percentages or dimensions),
462 // remove them and replace them with a single number containing the product of the removed nodes.
463 double multiplier = 1;
464
465 // Sorting will have put the number nodes first.
466 unsigned leadingNumberNodeCount = 0;
467 for (auto& node : m_children) {
468 auto nodeType = primitiveTypeForCombination(node.get());
469 if (nodeType != CSSUnitType::CSS_NUMBER)
470 break;
471
472 multiplier *= node->doubleValue(CSSUnitType::CSS_NUMBER);
473 ++leadingNumberNodeCount;
474 }
475
476 Vector<Ref<CSSCalcExpressionNode>> newChildren;
477 newChildren.reserveInitialCapacity(m_children.size());
478
479 // If root contains only two children, one of which is a number (not a percentage or dimension) and the other of
480 // which is a Sum whose children are all numeric values, multiply all of the Sumâ€™s children by the number, then
481 // return the Sum.
482 // The Sum's children simplification will have happened already.
483 bool didMultiply = false;
484 if (leadingNumberNodeCount && m_children.size() - leadingNumberNodeCount == 1) {
485 auto multiplicandCategory = calcUnitCategory(primitiveTypeForCombination(m_children.last().get()));
486 if (multiplicandCategory != CalculationCategory::Other) {
487 newChildren.uncheckedAppend(m_children.last().copyRef());
488 downcast<CSSCalcPrimitiveValueNode>(newChildren[0].get()).multiply(multiplier);
489 didMultiply = true;
490 } else if (is<CSSCalcOperationNode>(m_children.last().get()) && downcast<CSSCalcOperationNode>(m_children.last().get()).calcOperator() == CalcOperator::Add) {
491 // If we're multiplying with another operation that is an addition and all the added children
492 // are percentages or dimensions, we should multiply each child and make this expression an
493 // addition.
494 auto allChildrenArePrimitiveValues = [](const Vector<Ref<CSSCalcExpressionNode>>& children) -> bool
495 {
496 for (auto& child : children) {
497 if (!is<CSSCalcPrimitiveValueNode>(child.get()))
498 return false;
499 }
500 return true;
501 };
502
503 auto& children = downcast<CSSCalcOperationNode>(m_children.last().get()).children();
504 if (allChildrenArePrimitiveValues(children)) {
505 for (auto& child : children) {
506 newChildren.append(child.copyRef());
507 downcast<CSSCalcPrimitiveValueNode>(newChildren.last().get()).multiply(multiplier);
508 }
509 m_operator = CalcOperator::Add;
510 didMultiply = true;
511 }
512 }
513 }
514
515 if (!didMultiply) {
516 if (leadingNumberNodeCount) {
517 auto multiplierNode = CSSCalcPrimitiveValueNode::create(CSSPrimitiveValue::create(multiplier, CSSUnitType::CSS_NUMBER));
518 newChildren.uncheckedAppend(WTFMove(multiplierNode));
519 }
520
521 for (unsigned i = leadingNumberNodeCount; i < m_children.size(); ++i)
522 newChildren.uncheckedAppend(m_children[i].copyRef());
523 }
524
525 newChildren.shrinkToFit();
526 m_children = WTFMove(newChildren);
527 }
528
529 if (isMinOrMaxNode() && canCombineAllChildren()) {
530 auto combinedUnitType = m_children[0]->primitiveType();
531 auto category = calculationCategoryForCombination(combinedUnitType);
532 if (category != CalculationCategory::Other)
533 combinedUnitType = canonicalUnitTypeForCalculationCategory(category);
534
535 double resolvedValue = doubleValue(combinedUnitType);
536 auto newChild = CSSCalcPrimitiveValueNode::create(CSSPrimitiveValue::create(resolvedValue, combinedUnitType));
537
538 m_children.clear();
539 m_children.append(WTFMove(newChild));
540 }
541}

550Ref<CSSCalcExpressionNode> CSSCalcOperationNode::simplifyRecursive(Ref<CSSCalcExpressionNode>&& rootNode, int depth)
551{
552 if (is<CSSCalcOperationNode>(rootNode)) {
553 auto& operationNode = downcast<CSSCalcOperationNode>(rootNode.get());
554
555 auto& children = operationNode.children();
556 for (unsigned i = 0; i < children.size(); ++i) {
557 auto child = children[i].copyRef();
558 auto newNode = simplifyRecursive(WTFMove(child), depth + 1);
559 if (newNode.ptr() != children[i].ptr())
560 children[i] = WTFMove(newNode);
561 }
562 } else if (is<CSSCalcNegateNode>(rootNode)) {
563 auto& negateNode = downcast<CSSCalcNegateNode>(rootNode.get());
564 Ref<CSSCalcExpressionNode> child = negateNode.child();
565 auto newNode = simplifyRecursive(WTFMove(child), depth + 1);
566 if (newNode.ptr() != &negateNode.child())
567 negateNode.setChild(WTFMove(newNode));
568 } else if (is<CSSCalcInvertNode>(rootNode)) {
569 auto& invertNode = downcast<CSSCalcInvertNode>(rootNode.get());
570 Ref<CSSCalcExpressionNode> child = invertNode.child();
571 auto newNode = simplifyRecursive(WTFMove(child), depth + 1);
572 if (newNode.ptr() != &invertNode.child())
573 invertNode.setChild(WTFMove(newNode));
574 }
575
576 return simplifyNode(WTFMove(rootNode), depth);
577}

579Ref<CSSCalcExpressionNode> CSSCalcOperationNode::simplifyNode(Ref<CSSCalcExpressionNode>&& rootNode, int depth)
580{
581 if (is<CSSCalcPrimitiveValueNode>(rootNode)) {
582 // If root is a percentage that will be resolved against another value, and there is enough information
583 // available to resolve it, do so, and express the resulting numeric value in the appropriate canonical
584 // unit. Return the value.
585
586 // If root is a dimension that is not expressed in its canonical unit, and there is enough information
587 // available to convert it to the canonical unit, do so, and return the value.
588 auto& primitiveValueNode = downcast<CSSCalcPrimitiveValueNode>(rootNode.get());
589 primitiveValueNode.canonicalizeUnit();
590 return WTFMove(rootNode);
591 }
592
593 // If root is an operator node thatâ€™s not one of the calc-operator nodes, and all of its children are numeric values
594 // with enough information to computed the operation root represents, return the result of running rootâ€™s operation
595 // using its children, expressed in the resultâ€™s canonical unit.
596 if (is<CSSCalcOperationNode>(rootNode)) {
597 auto& calcOperationNode = downcast<CSSCalcOperationNode>(rootNode.get());
598 // Don't simplify at the root, otherwise we lose track of the operation for serialization.
599 if (calcOperationNode.children().size() == 1 && depth)
600 return WTFMove(calcOperationNode.children()[0]);
601
602 if (calcOperationNode.isCalcSumNode()) {
603 calcOperationNode.hoistChildrenWithOperator(CalcOperator::Add);
604 calcOperationNode.combineChildren();
605 }
606
607 if (calcOperationNode.isCalcProductNode()) {
608 calcOperationNode.hoistChildrenWithOperator(CalcOperator::Multiply);
609 calcOperationNode.combineChildren();
610 }
611
612 if (calcOperationNode.isMinOrMaxNode())
613 calcOperationNode.combineChildren();
614
615 // If only one child remains, return the child (except at the root).
616 auto shouldCombineParentWithOnlyChild = [](const CSSCalcOperationNode& parent, int depth)
617 {
618 if (parent.children().size() != 1)
619 return false;
620
621 // Always simplify below the root.
622 if (depth)
623 return true;
624
625 // At the root, preserve the root function by only merging nodes with the same function.
626 auto& child = parent.children().first();
627 if (!is<CSSCalcOperationNode>(child))
628 return false;
629
630 auto parentFunction = functionFromOperator(parent.calcOperator());
631 auto childFunction = functionFromOperator(downcast<CSSCalcOperationNode>(child.get()).calcOperator());
632 return childFunction == parentFunction;
633 };
634
635 if (shouldCombineParentWithOnlyChild(calcOperationNode, depth))
636 return WTFMove(calcOperationNode.children().first());
637
638 return WTFMove(rootNode);
639 }
640
641 if (is<CSSCalcNegateNode>(rootNode)) {
642 auto& childNode = downcast<CSSCalcNegateNode>(rootNode.get()).child();
643 // If rootâ€™s child is a numeric value, return an equivalent numeric value, but with the value negated (0 - value).
644 if (is<CSSCalcPrimitiveValueNode>(childNode) && downcast<CSSCalcPrimitiveValueNode>(childNode).isNumericValue()) {
645 downcast<CSSCalcPrimitiveValueNode>(childNode).negate();
646 return childNode;
647 }
648
649 // If rootâ€™s child is a Negate node, return the childâ€™s child.
650 if (is<CSSCalcNegateNode>(childNode))
651 return downcast<CSSCalcNegateNode>(childNode).child();
652
653 return WTFMove(rootNode);
654 }
655
656 if (is<CSSCalcInvertNode>(rootNode)) {
657 auto& childNode = downcast<CSSCalcInvertNode>(rootNode.get()).child();
658 // If rootâ€™s child is a number (not a percentage or dimension) return the reciprocal of the childâ€™s value.
659 if (is<CSSCalcPrimitiveValueNode>(childNode) && downcast<CSSCalcPrimitiveValueNode>(childNode).isNumericValue()) {
660 downcast<CSSCalcPrimitiveValueNode>(childNode).invert();
661 return childNode;
662 }
663
664 // If rootâ€™s child is an Invert node, return the childâ€™s child.
665 if (is<CSSCalcInvertNode>(childNode))
666 return downcast<CSSCalcInvertNode>(childNode).child();
667
668 return WTFMove(rootNode);
669 }
670
671 return WTFMove(rootNode);

674CSSUnitType CSSCalcOperationNode::primitiveType() const
675{
676 auto unitCategory = category();
677 switch (unitCategory) {
678 case CalculationCategory::Number:
679#if ASSERT_ENABLED
680 for (auto& child : m_children)
681 ASSERT(child->category() == CalculationCategory::Number);
682#endif
683 return CSSUnitType::CSS_NUMBER;
684
685 case CalculationCategory::Percent: {
686 if (m_children.isEmpty())
687 return CSSUnitType::CSS_UNKNOWN;
688
689 if (m_children.size() == 2) {
690 if (m_children[0]->category() == CalculationCategory::Number)
691 return m_children[1]->primitiveType();
692 if (m_children[1]->category() == CalculationCategory::Number)
693 return m_children[0]->primitiveType();
694 }
695 CSSUnitType firstType = m_children[0]->primitiveType();
696 for (auto& child : m_children) {
697 if (firstType != child->primitiveType())
698 return CSSUnitType::CSS_UNKNOWN;
699 }
700 return firstType;
701 }
702
703 case CalculationCategory::Length:
704 case CalculationCategory::Angle:
705 case CalculationCategory::Time:
706 case CalculationCategory::Frequency:
707 if (m_children.size() == 1)
708 return m_children.first()->primitiveType();
709 return canonicalUnitTypeForCalculationCategory(unitCategory);
710
711 case CalculationCategory::PercentLength:
712 case CalculationCategory::PercentNumber:
713 case CalculationCategory::Other:
714 return CSSUnitType::CSS_UNKNOWN;
715 }
716 ASSERT_NOT_REACHED();
717 return CSSUnitType::CSS_UNKNOWN;

721{

734 auto destinationCategory = CalculationCategory::Other;
735 if (category() == CalculationCategory::PercentLength)
736 destinationCategory = CalculationCategory::Length;
737 else if (category() == CalculationCategory::PercentNumber)
738 destinationCategory = CalculationCategory::Number;
739
740 return makeUnique<CalcExpressionOperation>(WTFMove(nodes), m_operator, destinationCategory);

774void CSSCalcOperationNode::buildCSSText(const CSSCalcExpressionNode& node, StringBuilder& builder)
775{
776 auto shouldOutputEnclosingCalc = [](const CSSCalcExpressionNode& rootNode) {
777 if (is<CSSCalcOperationNode>(rootNode)) {
778 auto& operationNode = downcast<CSSCalcOperationNode>(rootNode);
779 return operationNode.isCalcSumNode() || operationNode.isCalcProductNode();
780 }
781 return !is<CSSCalcPrimitiveValueNode>(rootNode);
782 };
783
784 bool outputCalc = shouldOutputEnclosingCalc(node);
785 if (outputCalc)
786 builder.append("calc(");
787
788 buildCSSTextRecursive(node, builder, GroupingParens::Omit);
789
790 if (outputCalc)
791 builder.append(')');
792}

795{

811// <https://drafts.csswg.org/css-values-4/#serialize-a-calculation-tree>
812void CSSCalcOperationNode::buildCSSTextRecursive(const CSSCalcExpressionNode& node, StringBuilder& builder, GroupingParens parens)
813{
814 // If root is a numeric value, or a non-math function, serialize root per the normal rules for it and return the result.
815 if (is<CSSCalcPrimitiveValueNode>(node)) {
816 auto& valueNode = downcast<CSSCalcPrimitiveValueNode>(node);
817 builder.append(valueNode.customCSSText());
818 return;
819 }
820
821 if (is<CSSCalcOperationNode>(node)) {
822 auto& operationNode = downcast<CSSCalcOperationNode>(node);
823
824 if (operationNode.isCalcSumNode()) {
825 // If root is a Sum node, let s be a string initially containing "(".
826 if (parens == GroupingParens::Include)
827 builder.append('(');
828
829 // Simplification already sorted children.
830 auto& children = operationNode.children();
831 ASSERT(children.size());
832 // Serialize rootâ€™s first child, and append it to s.
833 buildCSSTextRecursive(children.first(), builder);
834
835 // For each child of root beyond the first:
836 // If child is a Negate node, append " - " to s, then serialize the Negateâ€™s child and append the result to s.
837 // If child is a negative numeric value, append " - " to s, then serialize the negation of child as normal and append the result to s.
838 // Otherwise, append " + " to s, then serialize child and append the result to s.
839 for (unsigned i = 1; i < children.size(); ++i) {
840 auto& child = children[i];
841 if (is<CSSCalcNegateNode>(child)) {
842 builder.append(" - ");
843 buildCSSTextRecursive(downcast<CSSCalcNegateNode>(child.get()).child(), builder);
844 continue;
845 }
846
847 if (is<CSSCalcPrimitiveValueNode>(child)) {
848 auto& primitiveValueNode = downcast<CSSCalcPrimitiveValueNode>(child.get());
849 if (primitiveValueNode.isNegative()) {
850 builder.append(" - ");
851 // Serialize the negation of child.
852 auto unitType = primitiveValueNode.value().primitiveType();
853 builder.append(0 - primitiveValueNode.value().doubleValue(), CSSPrimitiveValue::unitTypeString(unitType));
854 continue;
855 }
856 }
857
858 builder.append(" + ");
859 buildCSSTextRecursive(child, builder);
860 }
861
862 if (parens == GroupingParens::Include)
863 builder.append(')');
864 return;
865 }
866
867 if (operationNode.isCalcProductNode()) {
868 // If root is a Product node, let s be a string initially containing "(".
869 if (parens == GroupingParens::Include)
870 builder.append('(');
871
872 // Simplification already sorted children.
873 auto& children = operationNode.children();
874 ASSERT(children.size());
875 // Serialize rootâ€™s first child, and append it to s.
876 buildCSSTextRecursive(children.first(), builder);
877
878 // For each child of root beyond the first:
879 // If child is an Invert node, append " / " to s, then serialize the Invertâ€™s child and append the result to s.
880 // Otherwise, append " * " to s, then serialize child and append the result to s.
881 for (unsigned i = 1; i < children.size(); ++i) {
882 auto& child = children[i];
883 if (is<CSSCalcInvertNode>(child)) {
884 builder.append(" / ");
885 buildCSSTextRecursive(downcast<CSSCalcInvertNode>(child.get()).child(), builder);
886 continue;
887 }
888
889 builder.append(" * ");
890 buildCSSTextRecursive(child, builder);
891 }
892
893 if (parens == GroupingParens::Include)
894 builder.append(')');
895 return;
896 }
897
898 // If root is anything but a Sum, Negate, Product, or Invert node, serialize a math function for the
899 // function corresponding to the node type, treating the nodeâ€™s children as the functionâ€™s
900 // comma-separated calculation arguments, and return the result.
901 builder.append(functionPrefixForOperator(operationNode.calcOperator()));
902
903 auto& children = operationNode.children();
904 ASSERT(children.size());
905 buildCSSTextRecursive(children.first(), builder, GroupingParens::Omit);
906
907 for (unsigned i = 1; i < children.size(); ++i) {
908 builder.append(", ");
909 buildCSSTextRecursive(children[i], builder, GroupingParens::Omit);
910 }
911
912 builder.append(')');
913 return;
914 }
915
916 if (is<CSSCalcNegateNode>(node)) {
917 auto& negateNode = downcast<CSSCalcNegateNode>(node);
918 // If root is a Negate node, let s be a string initially containing "(-1 * ".
919 builder.append("-1 *");
920 buildCSSTextRecursive(negateNode.child(), builder);
921 return;
922 }
923
924 if (is<CSSCalcInvertNode>(node)) {
925 auto& invertNode = downcast<CSSCalcInvertNode>(node);
926 // If root is an Invert node, let s be a string initially containing "(1 / ".
927 builder.append("1 / ");
928 buildCSSTextRecursive(invertNode.child(), builder);
929 return;
930 }
931}

960double CSSCalcOperationNode::evaluateOperator(CalcOperator op, const Vector<double>& children)
961{
962 switch (op) {
963 case CalcOperator::Add: {
964 double sum = 0;
965 for (auto& child : children)
966 sum += child;
967 return sum;
968 }
969 case CalcOperator::Subtract:
970 ASSERT(children.size() == 2);
971 return children[0] - children[1];
972 case CalcOperator::Multiply: {
973 double product = 1;
974 for (auto& child : children)
975 product *= child;
976 return product;
977 }
978 case CalcOperator::Divide:
979 ASSERT(children.size() == 1 || children.size() == 2);
980 if (children.size() == 1)
981 return std::numeric_limits<double>::quiet_NaN();
982 return children[0] / children[1];
983 case CalcOperator::Min: {
984 if (children.isEmpty())
985 return std::numeric_limits<double>::quiet_NaN();
986 double minimum = children[0];
987 for (auto child : children)
988 minimum = std::min(minimum, child);
989 return minimum;
990 }
991 case CalcOperator::Max: {
992 if (children.isEmpty())
993 return std::numeric_limits<double>::quiet_NaN();
994 double maximum = children[0];
995 for (auto child : children)
996 maximum = std::max(maximum, child);
997 return maximum;
998 }
999 case CalcOperator::Clamp: {
1000 if (children.size() != 3)
1001 return std::numeric_limits<double>::quiet_NaN();
1002 double min = children[0];
1003 double value = children[1];
1004 double max = children[2];
1005 return std::max(min, std::min(value, max));
1006 }
1007 }
1008 ASSERT_NOT_REACHED();
1009 return 0;
1010}
