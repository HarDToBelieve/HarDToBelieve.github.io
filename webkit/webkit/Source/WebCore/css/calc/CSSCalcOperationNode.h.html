27
28#include "CSSCalcExpressionNode.h"
29#include "CalcOperator.h"

31namespace WebCore {
32
33class CSSCalcOperationNode final : public CSSCalcExpressionNode {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 static RefPtr<CSSCalcOperationNode> create(CalcOperator, RefPtr<CSSCalcExpressionNode>&& leftSide, RefPtr<CSSCalcExpressionNode>&& rightSide);
37 static RefPtr<CSSCalcOperationNode> createSum(Vector<Ref<CSSCalcExpressionNode>>&& values);
38 static RefPtr<CSSCalcOperationNode> createProduct(Vector<Ref<CSSCalcExpressionNode>>&& values);
39 static RefPtr<CSSCalcOperationNode> createMinOrMaxOrClamp(CalcOperator, Vector<Ref<CSSCalcExpressionNode>>&& values, CalculationCategory destinationCategory);
40
41 static Ref<CSSCalcExpressionNode> simplify(Ref<CSSCalcExpressionNode>&&);
42
43 static void buildCSSText(const CSSCalcExpressionNode&, StringBuilder&);
44
45 CalcOperator calcOperator() const { return m_operator; }
46 bool isCalcSumNode() const { return m_operator == CalcOperator::Add; }
47 bool isCalcProductNode() const { return m_operator == CalcOperator::Multiply; }
48 bool isMinOrMaxNode() const { return m_operator == CalcOperator::Min || m_operator == CalcOperator::Max; }
49 bool shouldSortChildren() const { return isCalcSumNode() || isCalcProductNode(); }
50
51 void hoistChildrenWithOperator(CalcOperator);
52 void combineChildren();
53
54 bool canCombineAllChildren() const;
55
56 const Vector<Ref<CSSCalcExpressionNode>>& children() const { return m_children; }
57 Vector<Ref<CSSCalcExpressionNode>>& children() { return m_children; }
58
59private:
60 CSSCalcOperationNode(CalculationCategory category, CalcOperator op, Ref<CSSCalcExpressionNode>&& leftSide, Ref<CSSCalcExpressionNode>&& rightSide)
61 : CSSCalcExpressionNode(category)
62 , m_operator(op)
63 {
64 m_children.reserveInitialCapacity(2);
65 m_children.uncheckedAppend(WTFMove(leftSide));
66 m_children.uncheckedAppend(WTFMove(rightSide));
67 }
68
69 CSSCalcOperationNode(CalculationCategory category, CalcOperator op, Vector<Ref<CSSCalcExpressionNode>>&& children)
70 : CSSCalcExpressionNode(category)
71 , m_operator(op)
72 , m_children(WTFMove(children))
73 {
74 }
75
76 Type type() const final { return CssCalcOperation; }
77
78 bool isZero() const final
79 {
80 return !doubleValue(primitiveType());
81 }
82
83 bool equals(const CSSCalcExpressionNode&) const final;
84
85 std::unique_ptr<CalcExpressionNode> createCalcExpression(const CSSToLengthConversionData&) const final;
86
87 CSSUnitType primitiveType() const final;
88 double doubleValue(CSSUnitType) const final;
89 double computeLengthPx(const CSSToLengthConversionData&) const final;
90
91 void collectDirectComputationalDependencies(HashSet<CSSPropertyID>&) const final;
92 void collectDirectRootComputationalDependencies(HashSet<CSSPropertyID>&) const final;
93
94 void dump(TextStream&) const final;
95
96 static CSSCalcExpressionNode* getNumberSide(CSSCalcExpressionNode& leftSide, CSSCalcExpressionNode& rightSide)
97 {
98 if (leftSide.category() == CalculationCategory::Number)
99 return &leftSide;
100 if (rightSide.category() == CalculationCategory::Number)
101 return &rightSide;
102 return nullptr;
103 }
104
105 double evaluate(const Vector<double>& children) const
106 {
107 return evaluateOperator(m_operator, children);
108 }
109
110 static double evaluateOperator(CalcOperator, const Vector<double>&);
111 static Ref<CSSCalcExpressionNode> simplifyNode(Ref<CSSCalcExpressionNode>&&, int depth);
112 static Ref<CSSCalcExpressionNode> simplifyRecursive(Ref<CSSCalcExpressionNode>&&, int depth);
113
114 enum class GroupingParens {
115 Omit,
116 Include
117 };
118 static void buildCSSTextRecursive(const CSSCalcExpressionNode&, StringBuilder&, GroupingParens = GroupingParens::Include);
119
120 CalcOperator m_operator;
121 Vector<Ref<CSSCalcExpressionNode>> m_children;
122};
123
124}
