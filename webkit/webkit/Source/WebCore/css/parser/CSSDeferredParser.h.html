26#pragma once
27
28#include "CSSParserContext.h"
29#include <wtf/RefCounted.h>
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class CSSParserTokenRange;
35class ImmutableStyleProperties;
36class StyleRuleKeyframes;
37class StyleRuleBase;
38class StyleSheetContents;
39
40class CSSDeferredParser : public RefCounted<CSSDeferredParser> {
41public:
42 static Ref<CSSDeferredParser> create(const CSSParserContext& parserContext, const String& sheetText, StyleSheetContents& styleSheet)
43 {
44 return adoptRef(*new CSSDeferredParser(parserContext, sheetText, styleSheet));
45 }
46
47 CSSParserMode mode() const { return m_context.mode; }
48
49 const CSSParserContext& context() const { return m_context; }
50 StyleSheetContents* styleSheet() const;
51
52 Ref<ImmutableStyleProperties> parseDeclaration(const CSSParserTokenRange&);
53 void parseRuleList(const CSSParserTokenRange&, Vector<RefPtr<StyleRuleBase>>&);
54 void parseKeyframeList(const CSSParserTokenRange&, StyleRuleKeyframes&);
55
56 void adoptTokenizerEscapedStrings(Vector<String>&& strings) { m_escapedStrings = WTFMove(strings); }
57
58private:
59 CSSDeferredParser(const CSSParserContext&, const String&, StyleSheetContents&);
60
61 Vector<String> m_escapedStrings;
62 CSSParserContext m_context;
63
64 String m_sheetText;
65
66 WeakPtr<StyleSheetContents> m_styleSheet;
67};
68
69} // namespace WebCore
