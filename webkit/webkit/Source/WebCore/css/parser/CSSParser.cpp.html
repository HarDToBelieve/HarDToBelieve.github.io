89bool CSSParser::parseSupportsCondition(const String& condition)
90{
91 CSSParserImpl parser(m_context, condition);
92 if (!parser.tokenizer())
93 return false;
94 return CSSSupportsParser::supportsCondition(parser.tokenizer()->tokenRange(), parser, CSSSupportsParser::ForWindowCSS) == CSSSupportsParser::Supported;
95}

97Color CSSParser::parseColor(const String& string, bool strict)
98{
99 if (auto color = CSSParserFastPaths::parseSimpleColor(string, strict))
100 return *color;
101 // FIXME: Unclear why we want to ignore the boolean argument "strict" and always pass strictCSSParserContext here.
102 auto value = parseSingleValue(CSSPropertyColor, string, strictCSSParserContext());
103 if (!is<CSSPrimitiveValue>(value))
104 return { };
105 auto& primitiveValue = downcast<CSSPrimitiveValue>(*value);
106 if (!primitiveValue.isRGBColor())
107 return { };
108 return primitiveValue.color();
109}

129RefPtr<CSSValue> CSSParser::parseSingleValue(CSSPropertyID propertyID, const String& string, const CSSParserContext& context)
130{
131 if (string.isEmpty())
132 return nullptr;
133 if (auto value = CSSParserFastPaths::maybeParseValue(propertyID, string, context))
134 return value;
135 CSSTokenizer tokenizer(string);
136 return CSSPropertyParser::parseSingleValue(propertyID, tokenizer.tokenRange(), context);
137}

139CSSParser::ParseResult CSSParser::parseValue(MutableStyleProperties& declaration, CSSPropertyID propertyID, const String& string, bool important, const CSSParserContext& context)
140{
141 ASSERT(!string.isEmpty());
142 if (auto value = CSSParserFastPaths::maybeParseValue(propertyID, string, context))
143 return declaration.addParsedProperty(CSSProperty(propertyID, WTFMove(value), important)) ? CSSParser::ParseResult::Changed : CSSParser::ParseResult::Unchanged;
144 CSSParser parser(context);
145 return parser.parseValue(declaration, propertyID, string, important);
146}

178RefPtr<CSSValue> CSSParser::parseValueWithVariableReferences(CSSPropertyID propID, const CSSValue& value, Style::BuilderState& builderState)
179{
180 ASSERT((propID == CSSPropertyCustom && value.isCustomPropertyValue()) || (propID != CSSPropertyCustom && !value.isCustomPropertyValue()));
181 auto& style = builderState.style();
182 auto direction = style.direction();
183 auto writingMode = style.writingMode();
184
185 if (is<CSSPendingSubstitutionValue>(value)) {
186 // FIXME: Should have a resolvedShorthands cache to stop this from being done over and over for each longhand value.
187 auto& substitution = downcast<CSSPendingSubstitutionValue>(value);
188
189 auto shorthandID = substitution.shorthandPropertyId();
190 if (CSSProperty::isDirectionAwareProperty(shorthandID))
191 shorthandID = CSSProperty::resolveDirectionAwareProperty(shorthandID, direction, writingMode);
192
193 auto resolvedData = substitution.shorthandValue().resolveVariableReferences(builderState);
194 if (!resolvedData)
195 return nullptr;
196
197 ParsedPropertyVector parsedProperties;
198 if (!CSSPropertyParser::parseValue(shorthandID, false, resolvedData->tokens(), m_context, parsedProperties, StyleRuleType::Style))
199 return nullptr;
200
201 for (auto& property : parsedProperties) {
202 CSSPropertyID currentId = property.id();
203 if (CSSProperty::isDirectionAwareProperty(currentId))
204 currentId = CSSProperty::resolveDirectionAwareProperty(currentId, direction, writingMode);
205 if (currentId == propID)
206 return property.value();
207 }
208
209 return nullptr;
210 }
211
212 if (value.isVariableReferenceValue()) {
213 const CSSVariableReferenceValue& valueWithReferences = downcast<CSSVariableReferenceValue>(value);
214 auto resolvedData = valueWithReferences.resolveVariableReferences(builderState);
215 if (!resolvedData)
216 return nullptr;
217 return CSSPropertyParser::parseSingleValue(propID, resolvedData->tokens(), m_context);
218 }
219
220 const auto& customPropValue = downcast<CSSCustomPropertyValue>(value);
221 const auto& valueWithReferences = WTF::get<Ref<CSSVariableReferenceValue>>(customPropValue.value()).get();
222
223 auto& name = downcast<CSSCustomPropertyValue>(value).name();
224 auto* registered = builderState.document().getCSSRegisteredCustomPropertySet().get(name);
225 auto& syntax = registered ? registered->syntax : "*";
226 auto resolvedData = valueWithReferences.resolveVariableReferences(builderState);
227
228 if (!resolvedData)
229 return nullptr;
230
231 // FIXME handle REM cycles.
232 HashSet<CSSPropertyID> dependencies;
233 CSSPropertyParser::collectParsedCustomPropertyValueDependencies(syntax, false, dependencies, resolvedData->tokens(), m_context);
234
235 for (auto id : dependencies)
236 builderState.builder().applyProperty(id);
237
238 return CSSPropertyParser::parseTypedCustomPropertyValue(name, syntax, resolvedData->tokens(), builderState, m_context);
239}
