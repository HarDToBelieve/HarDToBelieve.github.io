34namespace WebCore {
35
36bool isValueAllowedInMode(unsigned short id, CSSParserMode mode)
37{
38 switch (id) {
39 case CSSValueInternalVariableValue:
40 return isUASheetBehavior(mode);
41 case CSSValueWebkitFocusRingColor:
42 return isUASheetBehavior(mode) || isQuirksModeBehavior(mode);
43 default:
44 return true;
45 }
46}
