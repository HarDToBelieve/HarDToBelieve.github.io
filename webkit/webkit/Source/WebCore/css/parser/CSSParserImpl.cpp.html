78 , m_styleSheet(styleSheet)

88CSSParser::ParseResult CSSParserImpl::parseValue(MutableStyleProperties* declaration, CSSPropertyID propertyID, const String& string, bool important, const CSSParserContext& context)
89{
90 CSSParserImpl parser(context, string);
91 auto ruleType = context.enclosingRuleType.value_or(StyleRuleType::Style);
92 parser.consumeDeclarationValue(parser.tokenizer()->tokenRange(), propertyID, important, ruleType);
93 if (parser.m_parsedProperties.isEmpty())
94 return CSSParser::ParseResult::Error;
95 return declaration->addParsedProperties(parser.m_parsedProperties) ? CSSParser::ParseResult::Changed : CSSParser::ParseResult::Unchanged;
96}

107static inline void filterProperties(bool important, const ParsedPropertyVector& input, ParsedPropertyVector& output, size_t& unusedEntries, std::bitset<numCSSProperties>& seenProperties, HashSet<AtomString>& seenCustomProperties)
108{
109 // Add properties in reverse order so that highest priority definitions are reached first. Duplicate definitions can then be ignored when found.
110 for (size_t i = input.size(); i--; ) {
111 const CSSProperty& property = input[i];
112 if (property.isImportant() != important)
113 continue;
114 const unsigned propertyIDIndex = property.id() - firstCSSProperty;
115
116 if (property.id() == CSSPropertyCustom) {
117 auto& name = downcast<CSSCustomPropertyValue>(*property.value()).name();
118 if (!seenCustomProperties.add(name).isNewEntry)
119 continue;
120 output[--unusedEntries] = property;
121 continue;
122 }
123
124 // FIXME-NEWPARSER: We won't support @apply yet.
125 /*else if (property.id() == CSSPropertyApplyAtRule) {
126 // FIXME: Do we need to do anything here?
127 } */
128
129
130 if (seenProperties.test(propertyIDIndex))
131 continue;
132 const unsigned relatedPropertyId = getRelatedPropertyId(property.id());
133 if (property.id() != CSSPropertyInvalid && relatedPropertyId != CSSPropertyInvalid) {
134 const unsigned relatedPropertyIDIndex = relatedPropertyId - firstCSSProperty;
135 seenProperties.set(relatedPropertyIDIndex);
136 seenProperties.set(propertyIDIndex);
137 } else
138 seenProperties.set(propertyIDIndex);
139
140 output[--unusedEntries] = property;
141 }
142}

150static Ref<ImmutableStyleProperties> createStyleProperties(ParsedPropertyVector& parsedProperties, CSSParserMode mode)
151{
152 std::bitset<numCSSProperties> seenProperties;
153 size_t unusedEntries = parsedProperties.size();
154 ParsedPropertyVector results(unusedEntries);
155 HashSet<AtomString> seenCustomProperties;
156
157 filterProperties(true, parsedProperties, results, unusedEntries, seenProperties, seenCustomProperties);
158 filterProperties(false, parsedProperties, results, unusedEntries, seenProperties, seenCustomProperties);
159
160 Ref<ImmutableStyleProperties> result = ImmutableStyleProperties::create(results.data() + unusedEntries, results.size() - unusedEntries, mode);
161 parsedProperties.clear();
162 return result;
163}

207bool CSSParserImpl::parseDeclarationList(MutableStyleProperties* declaration, const String& string, const CSSParserContext& context)
208{
209 CSSParserImpl parser(context, string);
210 auto ruleType = context.enclosingRuleType.value_or(StyleRuleType::Style);
211 parser.consumeDeclarationList(parser.tokenizer()->tokenRange(), ruleType);
212 if (parser.m_parsedProperties.isEmpty())
213 return false;
214
215 std::bitset<numCSSProperties> seenProperties;
216 size_t unusedEntries = parser.m_parsedProperties.size();
217 ParsedPropertyVector results(unusedEntries);
218 HashSet<AtomString> seenCustomProperties;
219 filterProperties(true, parser.m_parsedProperties, results, unusedEntries, seenProperties, seenCustomProperties);
220 filterProperties(false, parser.m_parsedProperties, results, unusedEntries, seenProperties, seenCustomProperties);
221 if (unusedEntries)
222 results.remove(0, unusedEntries);
223 return declaration->addParsedProperties(results);
224}

226RefPtr<StyleRuleBase> CSSParserImpl::parseRule(const String& string, const CSSParserContext& context, StyleSheetContents* styleSheet, AllowedRulesType allowedRules)
227{
228 CSSParserImpl parser(context, string, styleSheet);
229 CSSParserTokenRange range = parser.tokenizer()->tokenRange();
230 range.consumeWhitespace();
231 if (range.atEnd())
232 return nullptr; // Parse error, empty rule
233 RefPtr<StyleRuleBase> rule;
234 if (range.peek().type() == AtKeywordToken)
235 rule = parser.consumeAtRule(range, allowedRules);
236 else
237 rule = parser.consumeQualifiedRule(range, allowedRules);
238 if (!rule)
239 return nullptr; // Parse error, failed to consume rule
240 range.consumeWhitespace();
241 if (!rule || !range.atEnd())
242 return nullptr; // Parse error, trailing garbage
243 return rule;
244}

266CSSSelectorList CSSParserImpl::parsePageSelector(CSSParserTokenRange range, StyleSheetContents* styleSheet)
267{
268 // We only support a small subset of the css-page spec.
269 range.consumeWhitespace();
270 AtomString typeSelector;
271 if (range.peek().type() == IdentToken)
272 typeSelector = range.consume().value().toAtomString();
273
274 StringView pseudo;
275 if (range.peek().type() == ColonToken) {
276 range.consume();
277 if (range.peek().type() != IdentToken)
278 return CSSSelectorList();
279 pseudo = range.consume().value();
280 }
281
282 range.consumeWhitespace();
283 if (!range.atEnd())
284 return CSSSelectorList(); // Parse error; extra tokens in @page selector
285
286 std::unique_ptr<CSSParserSelector> selector;
287 if (!typeSelector.isNull() && pseudo.isNull())
288 selector = makeUnique<CSSParserSelector>(QualifiedName(nullAtom(), typeSelector, styleSheet->defaultNamespace()));
289 else {
290 selector = makeUnique<CSSParserSelector>();
291 if (!pseudo.isNull()) {
292 selector = std::unique_ptr<CSSParserSelector>(CSSParserSelector::parsePagePseudoSelector(pseudo));
293 if (!selector || selector->match() != CSSSelector::PagePseudoClass)
294 return CSSSelectorList();
295 }
296 if (!typeSelector.isNull())
297 selector->prependTagSelector(QualifiedName(nullAtom(), typeSelector, styleSheet->defaultNamespace()));
298 }
299
300 selector->setForPage();
301 return CSSSelectorList { Vector<std::unique_ptr<CSSParserSelector>>::from(WTFMove(selector)) };
302}

339static CSSParserImpl::AllowedRulesType computeNewAllowedRules(CSSParserImpl::AllowedRulesType allowedRules, StyleRuleBase* rule)
340{
341 if (!rule || allowedRules == CSSParserImpl::KeyframeRules || allowedRules == CSSParserImpl::CounterStyleRules || allowedRules == CSSParserImpl::NoRules)
342 return allowedRules;
343 ASSERT(allowedRules <= CSSParserImpl::RegularRules);
344 if (rule->isCharsetRule() || rule->isImportRule())
345 return CSSParserImpl::AllowImportRules;
346 if (rule->isNamespaceRule())
347 return CSSParserImpl::AllowNamespaceRules;
348 return CSSParserImpl::RegularRules;
349}

351template<typename T>
352bool CSSParserImpl::consumeRuleList(CSSParserTokenRange range, RuleListType ruleListType, const T callback)
353{
354 AllowedRulesType allowedRules = RegularRules;
355 switch (ruleListType) {
356 case TopLevelRuleList:
357 allowedRules = AllowCharsetRules;
358 break;
359 case RegularRuleList:
360 allowedRules = RegularRules;
361 break;
362 case KeyframesRuleList:
363 allowedRules = KeyframeRules;
364 break;
365 default:
366 ASSERT_NOT_REACHED();
367 }
368
369 bool seenRule = false;
370 bool firstRuleValid = false;
371 while (!range.atEnd()) {
372 RefPtr<StyleRuleBase> rule;
373 switch (range.peek().type()) {
374 case WhitespaceToken:
375 range.consumeWhitespace();
376 continue;
377 case AtKeywordToken:
378 rule = consumeAtRule(range, allowedRules);
379 break;
380 case CDOToken:
381 case CDCToken:
382 if (ruleListType == TopLevelRuleList) {
383 range.consume();
384 continue;
385 }
386 FALLTHROUGH;
387 default:
388 rule = consumeQualifiedRule(range, allowedRules);
389 break;
390 }
391 if (!seenRule) {
392 seenRule = true;
393 firstRuleValid = rule;
394 }
395 if (rule) {
396 allowedRules = computeNewAllowedRules(allowedRules, rule.get());
397 callback(rule);
398 }
399 }
400
401 return firstRuleValid;
402}

404RefPtr<StyleRuleBase> CSSParserImpl::consumeAtRule(CSSParserTokenRange& range, AllowedRulesType allowedRules)
405{
406 ASSERT(range.peek().type() == AtKeywordToken);
407 const StringView name = range.consumeIncludingWhitespace().value();
408 const CSSParserToken* preludeStart = &range.peek();
409 while (!range.atEnd() && range.peek().type() != LeftBraceToken && range.peek().type() != SemicolonToken)
410 range.consumeComponentValue();
411
412 CSSParserTokenRange prelude = range.makeSubRange(preludeStart, &range.peek());
413 CSSAtRuleID id = cssAtRuleID(name);
414
415 if (range.atEnd() || range.peek().type() == SemicolonToken) {
416 range.consume();
417 if (allowedRules == AllowCharsetRules && id == CSSAtRuleCharset)
418 return consumeCharsetRule(prelude);
419 if (allowedRules <= AllowImportRules && id == CSSAtRuleImport)
420 return consumeImportRule(prelude);
421 if (allowedRules <= AllowNamespaceRules && id == CSSAtRuleNamespace)
422 return consumeNamespaceRule(prelude);
423 // FIXME-NEWPARSER: Support "apply"
424 /*if (allowedRules == ApplyRules && id == CSSAtRuleApply) {
425 consumeApplyRule(prelude);
426 return nullptr; // consumeApplyRule just updates m_parsedProperties
427 }*/
428 return nullptr; // Parse error, unrecognised at-rule without block
429 }
430
431 CSSParserTokenRange block = range.consumeBlock();
432 if (allowedRules == KeyframeRules)
433 return nullptr; // Parse error, no at-rules supported inside @keyframes
434 if (allowedRules == NoRules || allowedRules == ApplyRules)
435 return nullptr; // Parse error, no at-rules with blocks supported inside declaration lists
436
437 ASSERT(allowedRules <= RegularRules);
438
439 switch (id) {
440 case CSSAtRuleMedia:
441 return consumeMediaRule(prelude, block);
442 case CSSAtRuleSupports:
443 return consumeSupportsRule(prelude, block);
444 case CSSAtRuleFontFace:
445 return consumeFontFaceRule(prelude, block);
446 case CSSAtRuleWebkitKeyframes:
447 return consumeKeyframesRule(true, prelude, block);
448 case CSSAtRuleKeyframes:
449 return consumeKeyframesRule(false, prelude, block);
450 case CSSAtRulePage:
451 return consumePageRule(prelude, block);
452 case CSSAtRuleCounterStyle:
453 return consumeCounterStyleRule(prelude, block);
454 default:
455 return nullptr; // Parse error, unrecognised at-rule with block
456 }
457}

459RefPtr<StyleRuleBase> CSSParserImpl::consumeQualifiedRule(CSSParserTokenRange& range, AllowedRulesType allowedRules)
460{
461 const CSSParserToken* preludeStart = &range.peek();
462 while (!range.atEnd() && range.peek().type() != LeftBraceToken)
463 range.consumeComponentValue();
464
465 if (range.atEnd())
466 return nullptr; // Parse error, EOF instead of qualified rule block
467
468 CSSParserTokenRange prelude = range.makeSubRange(preludeStart, &range.peek());
469 CSSParserTokenRange block = range.consumeBlockCheckingForEditability(m_styleSheet.get());
470
471 if (allowedRules <= RegularRules)
472 return consumeStyleRule(prelude, block);
473 if (allowedRules == KeyframeRules)
474 return consumeKeyframeStyleRule(prelude, block);
475
476 ASSERT_NOT_REACHED();
477 return nullptr;
478}

480// This may still consume tokens if it fails
481static AtomString consumeStringOrURI(CSSParserTokenRange& range)
482{
483 const CSSParserToken& token = range.peek();
484
485 if (token.type() == StringToken || token.type() == UrlToken)
486 return range.consumeIncludingWhitespace().value().toAtomString();
487
488 if (token.type() != FunctionToken || !equalIgnoringASCIICase(token.value(), "url"))
489 return AtomString();
490
491 CSSParserTokenRange contents = range.consumeBlock();
492 const CSSParserToken& uri = contents.consumeIncludingWhitespace();
493 if (uri.type() == BadStringToken || !contents.atEnd())
494 return AtomString();
495 return uri.value().toAtomString();
496}

506RefPtr<StyleRuleImport> CSSParserImpl::consumeImportRule(CSSParserTokenRange prelude)
507{
508 AtomString uri(consumeStringOrURI(prelude));
509 if (uri.isNull())
510 return nullptr; // Parse error, expected string or URI
511
512 if (m_observerWrapper) {
513 unsigned endOffset = m_observerWrapper->endOffset(prelude);
514 m_observerWrapper->observer().startRuleHeader(StyleRuleType::Import, m_observerWrapper->startOffset(prelude));
515 m_observerWrapper->observer().endRuleHeader(endOffset);
516 m_observerWrapper->observer().startRuleBody(endOffset);
517 m_observerWrapper->observer().endRuleBody(endOffset);
518 }
519
520 return StyleRuleImport::create(uri, MediaQueryParser::parseMediaQuerySet(prelude, MediaQueryParserContext(m_context)).releaseNonNull());
521}

523RefPtr<StyleRuleNamespace> CSSParserImpl::consumeNamespaceRule(CSSParserTokenRange prelude)
524{
525 AtomString namespacePrefix;
526 if (prelude.peek().type() == IdentToken)
527 namespacePrefix = prelude.consumeIncludingWhitespace().value().toAtomString();
528
529 AtomString uri(consumeStringOrURI(prelude));
530 if (uri.isNull() || !prelude.atEnd())
531 return nullptr; // Parse error, expected string or URI
532
533 return StyleRuleNamespace::create(namespacePrefix, uri);
534}

582 m_observerWrapper->observer().endRuleBody(m_observerWrapper->endOffset(block));

585}

604RefPtr<StyleRuleKeyframes> CSSParserImpl::consumeKeyframesRule(bool webkitPrefixed, CSSParserTokenRange prelude, CSSParserTokenRange block)
605{
606 CSSParserTokenRange rangeCopy = prelude; // For inspector callbacks
607 const CSSParserToken& nameToken = prelude.consumeIncludingWhitespace();
608 if (!prelude.atEnd())
609 return nullptr; // Parse error; expected single non-whitespace token in @keyframes header
610
611 String name;
612 if (nameToken.type() == IdentToken) {
613 name = nameToken.value().toString();
614 } else if (nameToken.type() == StringToken && webkitPrefixed)
615 name = nameToken.value().toString();
616 else
617 return nullptr; // Parse error; expected ident token in @keyframes header
618
619 if (m_deferredParser)
620 return StyleRuleKeyframes::create(name, makeUnique<DeferredStyleGroupRuleList>(block, *m_deferredParser));
621
622 if (m_observerWrapper) {
623 m_observerWrapper->observer().startRuleHeader(StyleRuleType::Keyframes, m_observerWrapper->startOffset(rangeCopy));
624 m_observerWrapper->observer().endRuleHeader(m_observerWrapper->endOffset(prelude));
625 m_observerWrapper->observer().startRuleBody(m_observerWrapper->previousTokenStartOffset(block));

706static void observeSelectors(CSSParserObserverWrapper& wrapper, CSSParserTokenRange selectors)
707{
708 // This is easier than hooking into the CSSSelectorParser
709 selectors.consumeWhitespace();
710 CSSParserTokenRange originalRange = selectors;
711 wrapper.observer().startRuleHeader(StyleRuleType::Style, wrapper.startOffset(originalRange));
712
713 while (!selectors.atEnd()) {
714 const CSSParserToken* selectorStart = &selectors.peek();
715 while (!selectors.atEnd() && selectors.peek().type() != CommaToken)
716 selectors.consumeComponentValue();
717 CSSParserTokenRange selector = selectors.makeSubRange(selectorStart, &selectors.peek());
718 selectors.consumeIncludingWhitespace();
719
720 wrapper.observer().observeSelector(wrapper.startOffset(selector), wrapper.endOffset(selector));
721 }
722
723 wrapper.observer().endRuleHeader(wrapper.endOffset(originalRange));
724}

726RefPtr<StyleRule> CSSParserImpl::consumeStyleRule(CSSParserTokenRange prelude, CSSParserTokenRange block)
727{
728 auto selectorList = parseCSSSelector(prelude, m_context, m_styleSheet.get());
729 if (!selectorList)
730 return nullptr; // Parse error, invalid selector list
731
732 if (m_observerWrapper)
733 observeSelectors(*m_observerWrapper, prelude);
734
735 if (m_deferredParser) {
736 // If a rule is empty (i.e., only whitespace), don't bother using
737 // deferred parsing. This allows the empty rule optimization in ElementRuleCollector
738 // to continue to work. Note we don't have to consider CommentTokens, since those
739 // are stripped out.
740 CSSParserTokenRange blockCopy = block;
741 blockCopy.consumeWhitespace();
742 if (!blockCopy.atEnd()) {
743 return StyleRule::create(createDeferredStyleProperties(block), m_context.hasDocumentSecurityOrigin, WTFMove(*selectorList));
744 }
745 }
746
747 consumeDeclarationList(block, StyleRuleType::Style);
748 return StyleRule::create(createStyleProperties(m_parsedProperties, m_context.mode), m_context.hasDocumentSecurityOrigin, WTFMove(*selectorList));
749}

751void CSSParserImpl::consumeDeclarationList(CSSParserTokenRange range, StyleRuleType ruleType)
752{
753 ASSERT(m_parsedProperties.isEmpty());
754
755 bool useObserver = m_observerWrapper && (ruleType == StyleRuleType::Style || ruleType == StyleRuleType::Keyframe || ruleType == StyleRuleType::CounterStyle);
756 if (useObserver) {
757 m_observerWrapper->observer().startRuleBody(m_observerWrapper->previousTokenStartOffset(range));
758 m_observerWrapper->skipCommentsBefore(range, true);
759 }
760
761 while (!range.atEnd()) {
762 switch (range.peek().type()) {
763 case WhitespaceToken:
764 case SemicolonToken:
765 range.consume();
766 break;
767 case IdentToken: {
768 const CSSParserToken* declarationStart = &range.peek();
769
770 if (useObserver)
771 m_observerWrapper->yieldCommentsBefore(range);
772
773 while (!range.atEnd() && range.peek().type() != SemicolonToken)
774 range.consumeComponentValue();
775
776 consumeDeclaration(range.makeSubRange(declarationStart, &range.peek()), ruleType);
777
778 if (useObserver)
779 m_observerWrapper->skipCommentsBefore(range, false);
780 break;
781 }
782 case AtKeywordToken: {
783 // FIXME-NEWPARSER: Support apply
784 AllowedRulesType allowedRules = /* ruleType == StyleRuleType::Style && RuntimeEnabledFeatures::cssApplyAtRulesEnabled() ? ApplyRules :*/ NoRules;
785 RefPtr<StyleRuleBase> rule = consumeAtRule(range, allowedRules);
786 ASSERT_UNUSED(rule, !rule);
787 break;
788 }
789 default: // Parse error, unexpected token in declaration list
790 while (!range.atEnd() && range.peek().type() != SemicolonToken)
791 range.consumeComponentValue();
792 break;
793 }
794 }
795
796 // Yield remaining comments
797 if (useObserver) {
798 m_observerWrapper->yieldCommentsBefore(range);
799 m_observerWrapper->observer().endRuleBody(m_observerWrapper->endOffset(range));
800 }
801}

803static void removeTrailingWhitespace(const CSSParserTokenRange& range, const CSSParserToken*& position)
804{
805 while (position != range.begin() && position[-1].type() == WhitespaceToken)
806 --position;
807}

809// https://drafts.csswg.org/css-syntax/#consume-declaration
810void CSSParserImpl::consumeDeclaration(CSSParserTokenRange range, StyleRuleType ruleType)
811{
812 CSSParserTokenRange rangeCopy = range; // For inspector callbacks
813
814 ASSERT(range.peek().type() == IdentToken);
815 auto& token = range.consumeIncludingWhitespace();
816 auto propertyID = token.parseAsCSSPropertyID();
817 if (range.consume().type() != ColonToken)
818 return; // Parse error
819 range.consumeWhitespace();
820
821 auto declarationValueEnd = range.end();
822 bool important = false;
823 if (!range.atEnd()) {
824 auto end = range.end();
825 removeTrailingWhitespace(range, end);
826 declarationValueEnd = end;
827 if (end[-1].type() == IdentToken && equalIgnoringASCIICase(end[-1].value(), "important")) {
828 --end;
829 removeTrailingWhitespace(range, end);
830 if (end[-1].type() == DelimiterToken && end[-1].delimiter() == '!') {
831 important = true;
832 --end;
833 removeTrailingWhitespace(range, end);
834 declarationValueEnd = end;
835 }
836 }
837 }
838
839 size_t propertiesCount = m_parsedProperties.size();
840
841 if (m_context.isPropertyRuntimeDisabled(propertyID) || isInternalCSSProperty(propertyID))
842 propertyID = CSSPropertyInvalid;
843
844 if (propertyID == CSSPropertyInvalid && CSSVariableParser::isValidVariableName(token)) {
845 AtomString variableName = token.value().toAtomString();
846 consumeCustomPropertyValue(range.makeSubRange(&range.peek(), declarationValueEnd), variableName, important);
847 }
848
849 if (important && (ruleType == StyleRuleType::FontFace || ruleType == StyleRuleType::Keyframe || ruleType == StyleRuleType::CounterStyle))
850 return;
851
852 if (propertyID != CSSPropertyInvalid)
853 consumeDeclarationValue(range.makeSubRange(&range.peek(), declarationValueEnd), propertyID, important, ruleType);
854
855 if (m_observerWrapper && (ruleType == StyleRuleType::Style || ruleType == StyleRuleType::Keyframe)) {
856 m_observerWrapper->observer().observeProperty(
857 m_observerWrapper->startOffset(rangeCopy), m_observerWrapper->endOffset(rangeCopy),
858 important, m_parsedProperties.size() != propertiesCount);
859 }
860}

866 else if (auto value = CSSVariableParser::parseDeclarationValue(variableName, range, m_context))

875Vector<double> CSSParserImpl::consumeKeyframeKeyList(CSSParserTokenRange range)
876{
877 Vector<double> result;
878 while (true) {
879 range.consumeWhitespace();
880 const CSSParserToken& token = range.consumeIncludingWhitespace();
881 if (token.type() == PercentageToken && token.numericValue() >= 0 && token.numericValue() <= 100)
882 result.append(token.numericValue() / 100);
883 else if (token.type() == IdentToken && equalIgnoringASCIICase(token.value(), "from"))
884 result.append(0);
885 else if (token.type() == IdentToken && equalIgnoringASCIICase(token.value(), "to"))
886 result.append(1);
887 else
888 return { }; // Parser error, invalid value in keyframe selector
889
890 if (range.atEnd()) {
891 result.shrinkToFit();
892 return result;
893 }
894
895 if (range.consume().type() != CommaToken)
896 return { }; // Parser error
897 }
