35namespace WebCore {
36
37unsigned CSSParserObserverWrapper::startOffset(const CSSParserTokenRange& range)
38{
39 return m_tokenOffsets[range.begin() - m_firstParserToken];
40}
41
42unsigned CSSParserObserverWrapper::previousTokenStartOffset(const CSSParserTokenRange& range)
43{
44 if (range.begin() == m_firstParserToken)
45 return 0;
46 return m_tokenOffsets[range.begin() - m_firstParserToken - 1];
47}
48
49unsigned CSSParserObserverWrapper::endOffset(const CSSParserTokenRange& range)
50{
51 return m_tokenOffsets[range.end() - m_firstParserToken];
52}
53
54void CSSParserObserverWrapper::skipCommentsBefore(const CSSParserTokenRange& range, bool leaveDirectlyBefore)
55{
56 unsigned startIndex = range.begin() - m_firstParserToken;
57 if (!leaveDirectlyBefore)
58 startIndex++;
59 while (m_commentIterator < m_commentOffsets.end() && m_commentIterator->tokensBefore < startIndex)
60 m_commentIterator++;
61}
62
63void CSSParserObserverWrapper::yieldCommentsBefore(const CSSParserTokenRange& range)
64{
65 unsigned startIndex = range.begin() - m_firstParserToken;
66 while (m_commentIterator < m_commentOffsets.end() && m_commentIterator->tokensBefore <= startIndex) {
67 m_observer.observeComment(m_commentIterator->startOffset, m_commentIterator->endOffset);
68 m_commentIterator++;
69 }
70}
