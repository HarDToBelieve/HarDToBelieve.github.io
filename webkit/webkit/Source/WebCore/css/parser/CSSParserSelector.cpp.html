35std::unique_ptr<CSSParserSelector> CSSParserSelector::parsePagePseudoSelector(StringView pseudoTypeString)
36{
37 CSSSelector::PagePseudoClassType pseudoType;
38 if (equalLettersIgnoringASCIICase(pseudoTypeString, "first"))
39 pseudoType = CSSSelector::PagePseudoClassFirst;
40 else if (equalLettersIgnoringASCIICase(pseudoTypeString, "left"))
41 pseudoType = CSSSelector::PagePseudoClassLeft;
42 else if (equalLettersIgnoringASCIICase(pseudoTypeString, "right"))
43 pseudoType = CSSSelector::PagePseudoClassRight;
44 else
45 return nullptr;
46
47 auto selector = makeUnique<CSSParserSelector>();
48 selector->m_selector->setMatch(CSSSelector::PagePseudoClass);
49 selector->m_selector->setPagePseudoType(pseudoType);
50 return selector;
51}

53std::unique_ptr<CSSParserSelector> CSSParserSelector::parsePseudoElementSelector(StringView pseudoTypeString)
54{
55 auto pseudoType = CSSSelector::parsePseudoElementType(pseudoTypeString);
56 if (pseudoType == CSSSelector::PseudoElementUnknown)
57 return nullptr;
58
59 auto selector = makeUnique<CSSParserSelector>();
60 selector->m_selector->setMatch(CSSSelector::PseudoElement);
61 selector->m_selector->setPseudoElementType(pseudoType);
62 AtomString name;
63 if (pseudoType != CSSSelector::PseudoElementWebKitCustomLegacyPrefixed)
64 name = pseudoTypeString.convertToASCIILowercase();
65 else {
66 if (equalLettersIgnoringASCIICase(pseudoTypeString, "-webkit-input-placeholder"))
67 name = AtomString("placeholder", AtomString::ConstructFromLiteral);
68 else if (equalLettersIgnoringASCIICase(pseudoTypeString, "-webkit-file-upload-button"))
69 name = AtomString("file-selector-button", AtomString::ConstructFromLiteral);
70 else {
71 ASSERT_NOT_REACHED();
72 name = pseudoTypeString.convertToASCIILowercase();
73 }
74 }
75 selector->m_selector->setValue(name);
76 return selector;
77}

79std::unique_ptr<CSSParserSelector> CSSParserSelector::parsePseudoClassSelector(StringView pseudoTypeString)
80{
81 auto pseudoType = parsePseudoClassAndCompatibilityElementString(pseudoTypeString);
82 if (pseudoType.pseudoClass != CSSSelector::PseudoClassUnknown) {
83 auto selector = makeUnique<CSSParserSelector>();
84 selector->m_selector->setMatch(CSSSelector::PseudoClass);
85 selector->m_selector->setPseudoClassType(pseudoType.pseudoClass);
86 return selector;
87 }
88 if (pseudoType.compatibilityPseudoElement != CSSSelector::PseudoElementUnknown) {
89 auto selector = makeUnique<CSSParserSelector>();
90 selector->m_selector->setMatch(CSSSelector::PseudoElement);
91 selector->m_selector->setPseudoElementType(pseudoType.compatibilityPseudoElement);
92 selector->m_selector->setValue(pseudoTypeString.convertToASCIILowercase());
93 return selector;
94 }
95 return nullptr;
96}

108CSSParserSelector::~CSSParserSelector()
109{
110 if (!m_tagHistory)
111 return;
112 Vector<std::unique_ptr<CSSParserSelector>, 16> toDelete;
113 std::unique_ptr<CSSParserSelector> selector = WTFMove(m_tagHistory);
114 while (true) {
115 std::unique_ptr<CSSParserSelector> next = WTFMove(selector->m_tagHistory);
116 toDelete.append(WTFMove(selector));
117 if (!next)
118 break;
119 selector = WTFMove(next);
120 }
121}

139static bool selectorListMatchesPseudoElement(const CSSSelectorList* selectorList)
140{
141 if (!selectorList)
142 return false;
143
144 for (const CSSSelector* subSelector = selectorList->first(); subSelector; subSelector = CSSSelectorList::next(subSelector)) {
145 for (const CSSSelector* selector = subSelector; selector; selector = selector->tagHistory()) {
146 if (selector->matchesPseudoElement())
147 return true;
148 if (const CSSSelectorList* subselectorList = selector->selectorList()) {
149 if (selectorListMatchesPseudoElement(subselectorList))
150 return true;
151 }
152 }
153 }
154 return false;
155}
156
157bool CSSParserSelector::matchesPseudoElement() const
158{
159 return m_selector->matchesPseudoElement() || selectorListMatchesPseudoElement(m_selector->selectorList());
160}

162void CSSParserSelector::insertTagHistory(CSSSelector::RelationType before, std::unique_ptr<CSSParserSelector> selector, CSSSelector::RelationType after)
163{
164 if (m_tagHistory)
165 selector->setTagHistory(WTFMove(m_tagHistory));
166 setRelation(before);
167 selector->setRelation(after);
168 m_tagHistory = WTFMove(selector);
169}

171void CSSParserSelector::appendTagHistory(CSSSelector::RelationType relation, std::unique_ptr<CSSParserSelector> selector)
172{
173 CSSParserSelector* end = this;
174 while (end->tagHistory())
175 end = end->tagHistory();
176
177 end->setRelation(relation);
178 end->setTagHistory(WTFMove(selector));
179}

181void CSSParserSelector::appendTagHistory(CSSParserSelectorCombinator relation, std::unique_ptr<CSSParserSelector> selector)
182{
183 CSSParserSelector* end = this;
184 while (end->tagHistory())
185 end = end->tagHistory();
186
187 CSSSelector::RelationType selectorRelation;
188 switch (relation) {
189 case CSSParserSelectorCombinator::Child:
190 selectorRelation = CSSSelector::Child;
191 break;
192 case CSSParserSelectorCombinator::DescendantSpace:
193 selectorRelation = CSSSelector::DescendantSpace;
194 break;
195 case CSSParserSelectorCombinator::DirectAdjacent:
196 selectorRelation = CSSSelector::DirectAdjacent;
197 break;
198 case CSSParserSelectorCombinator::IndirectAdjacent:
199 selectorRelation = CSSSelector::IndirectAdjacent;
200 break;
201 }
202 end->setRelation(selectorRelation);
203 end->setTagHistory(WTFMove(selector));
204}

206void CSSParserSelector::prependTagSelector(const QualifiedName& tagQName, bool tagIsForNamespaceRule)
207{
208 auto second = makeUnique<CSSParserSelector>();
209 second->m_selector = WTFMove(m_selector);
210 second->m_tagHistory = WTFMove(m_tagHistory);
211 m_tagHistory = WTFMove(second);
212
213 m_selector = makeUnique<CSSSelector>(tagQName, tagIsForNamespaceRule);
214 m_selector->setRelation(CSSSelector::Subselector);
215}
216
217std::unique_ptr<CSSParserSelector> CSSParserSelector::releaseTagHistory()
218{
219 setRelation(CSSSelector::Subselector);
220 return WTFMove(m_tagHistory);
221}
