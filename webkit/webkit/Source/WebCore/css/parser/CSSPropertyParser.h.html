43// Inputs: PropertyID, isImportant bool, CSSParserTokenRange.
44// Outputs: Vector of CSSProperties
45
46class CSSPropertyParser {
47 WTF_MAKE_NONCOPYABLE(CSSPropertyParser);
48public:
49 static bool parseValue(CSSPropertyID, bool important,
50 const CSSParserTokenRange&, const CSSParserContext&,
51 Vector<CSSProperty, 256>&, StyleRuleType);
52
53 // Parses a non-shorthand CSS property
54 static RefPtr<CSSValue> parseSingleValue(CSSPropertyID, const CSSParserTokenRange&, const CSSParserContext&);
55 static bool canParseTypedCustomPropertyValue(const String& syntax, const CSSParserTokenRange&, const CSSParserContext&);
56 static RefPtr<CSSCustomPropertyValue> parseTypedCustomPropertyValue(const String& name, const String& syntax, const CSSParserTokenRange&, const Style::BuilderState&, const CSSParserContext&);
57 static void collectParsedCustomPropertyValueDependencies(const String& syntax, bool isRoot, HashSet<CSSPropertyID>& dependencies, const CSSParserTokenRange&, const CSSParserContext&);
58
59 static RefPtr<CSSValue> parseCounterStyleDescriptor(CSSPropertyID, CSSParserTokenRange&, const CSSParserContext&);
60
61private:
62 CSSPropertyParser(const CSSParserTokenRange&, const CSSParserContext&, Vector<CSSProperty, 256>*, bool consumeWhitespace = true);
63
64 // FIXME: Rename once the CSSParserValue-based parseValue is removed
65 bool parseValueStart(CSSPropertyID, bool important);
66 bool consumeCSSWideKeyword(CSSPropertyID, bool important);
67 RefPtr<CSSValue> parseSingleValue(CSSPropertyID, CSSPropertyID = CSSPropertyInvalid);
68 bool canParseTypedCustomPropertyValue(const String& syntax);
69 RefPtr<CSSCustomPropertyValue> parseTypedCustomPropertyValue(const String& name, const String& syntax, const Style::BuilderState&);
70 void collectParsedCustomPropertyValueDependencies(const String& syntax, bool isRoot, HashSet<CSSPropertyID>& dependencies);
71
72 bool inQuirksMode() const { return m_context.mode == HTMLQuirksMode; }
73
74 bool parseViewportDescriptor(CSSPropertyID propId, bool important);
75 bool parseFontFaceDescriptor(CSSPropertyID);
76 bool parseCounterStyleDescriptor(CSSPropertyID, const CSSParserContext&);
77
78 void addProperty(CSSPropertyID, CSSPropertyID, Ref<CSSValue>&&, bool important, bool implicit = false);
79 void addExpandedPropertyForValue(CSSPropertyID propId, Ref<CSSValue>&&, bool);
80
81 bool consumeBorder(RefPtr<CSSValue>& width, RefPtr<CSSValue>& style, RefPtr<CSSValue>& color);
82
83 bool parseShorthand(CSSPropertyID, bool important);
84 bool consumeShorthandGreedily(const StylePropertyShorthand&, bool important);
85 bool consume2ValueShorthand(const StylePropertyShorthand&, bool important);
86 bool consume4ValueShorthand(const StylePropertyShorthand&, bool important);
87
88 // Legacy parsing allows <string>s for animation-name
89 bool consumeAnimationShorthand(const StylePropertyShorthand&, bool important);
90 bool consumeBackgroundShorthand(const StylePropertyShorthand&, bool important);
91 bool consumeOverflowShorthand(bool important);
92
93 bool consumeColumns(bool important);
94
95 bool consumeGridItemPositionShorthand(CSSPropertyID, bool important);
96 bool consumeGridTemplateRowsAndAreasAndColumns(CSSPropertyID, bool important);
97 bool consumeGridTemplateShorthand(CSSPropertyID, bool important);
98 bool consumeGridShorthand(bool important);
99 bool consumeGridAreaShorthand(bool important);
100
101 bool consumePlaceContentShorthand(bool important);
102 bool consumePlaceItemsShorthand(bool important);
103 bool consumePlaceSelfShorthand(bool important);
104
105 bool consumeFont(bool important);
106 bool consumeFontVariantShorthand(bool important);
107 bool consumeSystemFont(bool important);
108
109 bool consumeBorderSpacing(bool important);
110
111 // CSS3 Parsing Routines (for properties specific to CSS3)
112 bool consumeBorderImage(CSSPropertyID, bool important);
113
114 bool consumeFlex(bool important);
115
116 bool consumeLegacyBreakProperty(CSSPropertyID, bool important);
117
118 bool consumeTransformOrigin(bool important);
119 bool consumePerspectiveOrigin(bool important);
120
121 bool consumeOverscrollBehaviorShorthand(bool important);
122
123private:
124 // Inputs:
125 CSSParserTokenRange m_range;
126 const CSSParserContext& m_context;
127
128 // Outputs:
129 Vector<CSSProperty, 256>* m_parsedProperties;
130};
