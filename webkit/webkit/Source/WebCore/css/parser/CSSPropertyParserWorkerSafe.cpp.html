202static RefPtr<CSSFontFaceSrcValue> consumeFontFaceSrcURI(CSSParserTokenRange& range, const CSSParserContext& context)
203{
204 auto location = context.completeURL(CSSPropertyParserHelpers::consumeUrlAsStringView(range).toString()).resolvedURL.string();
205 if (location.isNull())
206 return nullptr;
207
208 String format;
209 if (range.peek().functionId() == CSSValueFormat) {
210 // https://drafts.csswg.org/css-fonts/#descdef-font-face-src
211 // FIXME: The format should be a comma-separated list; at this time we support only one.
212 // FIXME: We allow any identifier here and convert all to strings; specification calls for only certain identifiers.
213 auto args = CSSPropertyParserHelpers::consumeFunction(range);
214 auto& arg = args.consumeIncludingWhitespace();
215 if ((arg.type() != StringToken && arg.type() != IdentToken) || !args.atEnd())
216 return nullptr;
217 format = arg.value().toString();
218 }
219
220 // FIXME: Change CSSFontFaceSrcValue::create to take format so we don't need a separate setFormat call.
221 auto srcValue = CSSFontFaceSrcValue::create(location, context.isContentOpaque ? LoadedFromOpaqueSource::Yes : LoadedFromOpaqueSource::No);
222 srcValue->setFormat(format);
223 return srcValue;
224}

226static RefPtr<CSSValue> consumeFontFaceSrcLocal(CSSParserTokenRange& range)
227{
228 CSSParserTokenRange args = CSSPropertyParserHelpers::consumeFunction(range);
229 if (args.peek().type() == StringToken) {
230 const CSSParserToken& arg = args.consumeIncludingWhitespace();
231 if (!args.atEnd())
232 return nullptr;
233 return CSSFontFaceSrcValue::createLocal(arg.value().toString());
234 }
235 if (args.peek().type() == IdentToken) {
236 String familyName = CSSPropertyParserHelpers::concatenateFamilyName(args);
237 if (!args.atEnd())
238 return nullptr;
239 return CSSFontFaceSrcValue::createLocal(familyName);
240 }
241 return nullptr;
242}

244RefPtr<CSSValueList> consumeFontFaceSrc(CSSParserTokenRange& range, const CSSParserContext& context)
245{
246 RefPtr<CSSValueList> values = CSSValueList::createCommaSeparated();
247
248 do {
249 const CSSParserToken& token = range.peek();
250 RefPtr<CSSValue> parsedValue;
251 if (token.functionId() == CSSValueLocal)
252 parsedValue = consumeFontFaceSrcLocal(range);
253 else
254 parsedValue = consumeFontFaceSrcURI(range, context);
255 if (!parsedValue)
256 return nullptr;
257 values->append(parsedValue.releaseNonNull());
258 } while (CSSPropertyParserHelpers::consumeCommaIncludingWhitespace(range));
259 return values;
260}

282RefPtr<CSSFontStyleRangeValue> consumeFontStyleRange(CSSParserTokenRange& range, CSSParserMode cssParserMode, CSSValuePool& pool)
283{
284 auto keyword = consumeFontStyleKeywordValue(range, pool);
285 if (!keyword)
286 return nullptr;
287
288 if (keyword->valueID() != CSSValueOblique || range.atEnd())
289 return CSSFontStyleRangeValue::create(keyword.releaseNonNull());
290
291 // FIXME: This should probably not allow the unitless zero.
292 if (auto firstAngle = CSSPropertyParserHelpers::consumeAngleWorkerSafe(range, cssParserMode, pool, CSSPropertyParserHelpers::UnitlessQuirk::Forbid, CSSPropertyParserHelpers::UnitlessZeroQuirk::Allow)) {
293 auto firstAngleInDegrees = firstAngle->doubleValue(CSSUnitType::CSS_DEG);
294 if (!CSSPropertyParserHelpers::isFontStyleAngleInRange(firstAngleInDegrees))
295 return nullptr;
296 if (range.atEnd()) {
297 auto result = CSSValueList::createSpaceSeparated();
298 result->append(firstAngle.releaseNonNull());
299 return CSSFontStyleRangeValue::create(keyword.releaseNonNull(), WTFMove(result));
300 }
301
302 // FIXME: This should probably not allow the unitless zero.
303 auto secondAngle = CSSPropertyParserHelpers::consumeAngleWorkerSafe(range, cssParserMode, pool, CSSPropertyParserHelpers::UnitlessQuirk::Forbid, CSSPropertyParserHelpers::UnitlessZeroQuirk::Allow);
304 if (!secondAngle)
305 return nullptr;
306 auto secondAngleInDegrees = secondAngle->doubleValue(CSSUnitType::CSS_DEG);
307 if (!CSSPropertyParserHelpers::isFontStyleAngleInRange(secondAngleInDegrees) || firstAngleInDegrees > secondAngleInDegrees)
308 return nullptr;
309 auto result = CSSValueList::createSpaceSeparated();
310 result->append(firstAngle.releaseNonNull());
311 result->append(secondAngle.releaseNonNull());
312 return CSSFontStyleRangeValue::create(keyword.releaseNonNull(), WTFMove(result));
313 }
314
315 return nullptr;
316}

396static bool consumeOptionalDelimiter(CSSParserTokenRange& range, UChar value)
397{
398 if (!(range.peek().type() == DelimiterToken && range.peek().delimiter() == value))
399 return false;
400 range.consume();
401 return true;
402}

411static bool consumeAndAppendOptionalNumber(StringBuilder& builder, CSSParserTokenRange& range, CSSParserTokenType type = NumberToken)
412{
413 if (range.peek().type() != type)
414 return false;
415 auto originalText = range.consume().originalText();
416 if (originalText.isNull())
417 return false;
418 builder.append(originalText);
419 return true;
420}

422static bool consumeAndAppendOptionalDelimiter(StringBuilder& builder, CSSParserTokenRange& range, UChar value)
423{
424 if (!consumeOptionalDelimiter(range, value))
425 return false;
426 builder.append(value);
427 return true;
428}

430static void consumeAndAppendOptionalQuestionMarks(StringBuilder& builder, CSSParserTokenRange& range)
431{
432 while (consumeAndAppendOptionalDelimiter(builder, range, '?')) { }
433}

435static String consumeUnicodeRangeString(CSSParserTokenRange& range)
436{
437 if (!equalLettersIgnoringASCIICase(consumeIdentifier(range), "u"))
438 return { };
439 StringBuilder builder;
440 if (consumeAndAppendOptionalNumber(builder, range, DimensionToken))
441 consumeAndAppendOptionalQuestionMarks(builder, range);
442 else if (consumeAndAppendOptionalNumber(builder, range)) {
443 if (!(consumeAndAppendOptionalNumber(builder, range, DimensionToken) || consumeAndAppendOptionalNumber(builder, range)))
444 consumeAndAppendOptionalQuestionMarks(builder, range);
445 } else if (consumeOptionalDelimiter(range, '+')) {
446 builder.append('+');
447 if (auto identifier = consumeIdentifier(range); !identifier.isNull())
448 builder.append(identifier);
449 else if (!consumeAndAppendOptionalDelimiter(builder, range, '?'))
450 return { };
451 consumeAndAppendOptionalQuestionMarks(builder, range);
452 } else
453 return { };
454 return builder.toString();
455}

457struct UnicodeRange {
458 UChar32 start;
459 UChar32 end;
460};
461
462static std::optional<UnicodeRange> consumeUnicodeRange(CSSParserTokenRange& range)
463{
464 return readCharactersForParsing(consumeUnicodeRangeString(range), [&](auto buffer) -> std::optional<UnicodeRange> {
465 if (!skipExactly(buffer, '+'))
466 return std::nullopt;
467 UChar32 start = 0;
468 unsigned hexDigitCount = 0;
469 while (buffer.hasCharactersRemaining() && isASCIIHexDigit(*buffer)) {
470 if (++hexDigitCount > 6)
471 return std::nullopt;
472 start <<= 4;
473 start |= toASCIIHexValue(*buffer++);
474 }
475 auto end = start;
476 while (skipExactly(buffer, '?')) {
477 if (++hexDigitCount > 6)
478 return std::nullopt;
479 start <<= 4;
480 end <<= 4;
481 end |= 0xF;
482 }
483 if (!hexDigitCount)
484 return std::nullopt;
485 if (start == end && buffer.hasCharactersRemaining()) {
486 if (!skipExactly(buffer, '-'))
487 return std::nullopt;
488 end = 0;
489 hexDigitCount = 0;
490 while (buffer.hasCharactersRemaining() && isASCIIHexDigit(*buffer)) {
491 if (++hexDigitCount > 6)
492 return std::nullopt;
493 end <<= 4;
494 end |= toASCIIHexValue(*buffer++);
495 }
496 if (!hexDigitCount)
497 return std::nullopt;
498 }
499 if (buffer.hasCharactersRemaining())
500 return std::nullopt;
501 return { { start, end } };
502 });
503}

505RefPtr<CSSValueList> consumeFontFaceUnicodeRange(CSSParserTokenRange& range)
506{
507 auto values = CSSValueList::createCommaSeparated();
508 do {
509 auto unicodeRange = consumeUnicodeRange(range);
510 range.consumeWhitespace();
511 if (!unicodeRange || unicodeRange->end > UCHAR_MAX_VALUE || unicodeRange->start > unicodeRange->end)
512 return nullptr;
513 values->append(CSSUnicodeRangeValue::create(unicodeRange->start, unicodeRange->end));
514 } while (CSSPropertyParserHelpers::consumeCommaIncludingWhitespace(range));
515 return values;
516}

518static RefPtr<CSSFontFeatureValue> consumeFontFeatureTag(CSSParserTokenRange& range)
519{
520 // Feature tag name consists of 4-letter characters.
521 static const unsigned tagNameLength = 4;
522
523 const CSSParserToken& token = range.consumeIncludingWhitespace();
524 // Feature tag name comes first
525 if (token.type() != StringToken)
526 return nullptr;
527 if (token.value().length() != tagNameLength)
528 return nullptr;
529
530 FontTag tag;
531 for (unsigned i = 0; i < tag.size(); ++i) {
532 // Limits the range of characters to 0x20-0x7E, following the tag name rules defiend in the OpenType specification.
533 UChar character = token.value()[i];
534 if (character < 0x20 || character > 0x7E)
535 return nullptr;
536 tag[i] = toASCIILower(character);
537 }
538
539 int tagValue = 1;
540 if (!range.atEnd() && range.peek().type() != CommaToken) {
541 // Feature tag values could follow: <integer> | on | off
542 if (auto integer = CSSPropertyParserHelpers::consumeIntegerRaw(range, 0))
543 tagValue = *integer;
544 else if (range.peek().id() == CSSValueOn || range.peek().id() == CSSValueOff)
545 tagValue = range.consumeIncludingWhitespace().id() == CSSValueOn;
546 else
547 return nullptr;
548 }
549 return CSSFontFeatureValue::create(WTFMove(tag), tagValue);
550}

552RefPtr<CSSValue> consumeFontFeatureSettings(CSSParserTokenRange& range, CSSValuePool& pool)
553{
554 if (range.peek().id() == CSSValueNormal)
555 return CSSPropertyParserHelpers::consumeIdentWorkerSafe(range, pool);
556 RefPtr<CSSValueList> settings = CSSValueList::createCommaSeparated();
557 do {
558 RefPtr<CSSFontFeatureValue> fontFeatureValue = consumeFontFeatureTag(range);
559 if (!fontFeatureValue)
560 return nullptr;
561 settings->append(fontFeatureValue.releaseNonNull());
562 } while (CSSPropertyParserHelpers::consumeCommaIncludingWhitespace(range));
563 return settings;
564}
