45// https://drafts.csswg.org/css-syntax/#input-preprocessing
46static String preprocessString(String string)
47{
48 // We don't replace '\r' and '\f' with '\n' as the specification suggests, instead
49 // we treat them all the same in the isNewLine function below.
50 string.replace('\0', replacementCharacter);
51 return stringToUSVString(WTFMove(string));
52}

54std::unique_ptr<CSSTokenizer> CSSTokenizer::tryCreate(const String& string)
55{
56 bool success = true;
57 // We can't use makeUnique here because it does not have access to this private constructor.
58 auto tokenizer = std::unique_ptr<CSSTokenizer>(new CSSTokenizer(preprocessString(string), nullptr, &success));
59 if (UNLIKELY(!success))
60 return nullptr;
61 return tokenizer;
62}

64std::unique_ptr<CSSTokenizer> CSSTokenizer::tryCreate(const String& string, CSSParserObserverWrapper& wrapper)
65{
66 bool success = true;
67 // We can't use makeUnique here because it does not have access to this private constructor.
68 auto tokenizer = std::unique_ptr<CSSTokenizer>(new CSSTokenizer(preprocessString(string), &wrapper, &success));
69 if (UNLIKELY(!success))
70 return nullptr;
71 return tokenizer;

84CSSTokenizer::CSSTokenizer(String&& string, CSSParserObserverWrapper* wrapper, bool* constructionSuccessPtr)
85 : m_input(string)
86{
87 if (constructionSuccessPtr)
88 *constructionSuccessPtr = true;
89
90 if (string.isEmpty())
91 return;
92
93 // To avoid resizing we err on the side of reserving too much space.
94 // Most strings we tokenize have about 3.5 to 5 characters per token.
95 if (UNLIKELY(!m_tokens.tryReserveInitialCapacity(string.length() / 3))) {
96 // When constructionSuccessPtr is null, our policy is to crash on failure.
97 RELEASE_ASSERT(constructionSuccessPtr);
98 *constructionSuccessPtr = false;
99 return;
100 }
101
102 unsigned offset = 0;
103 while (true) {
104 CSSParserToken token = nextToken();
105 if (token.type() == EOFToken)
106 break;
107 if (token.type() == CommentToken) {
108 if (wrapper)
109 wrapper->addComment(offset, m_input.offset(), m_tokens.size());
110 } else {
111 if (UNLIKELY(!m_tokens.tryAppend(token))) {
112 // When constructionSuccessPtr is null, our policy is to crash on failure.
113 RELEASE_ASSERT(constructionSuccessPtr);
114 *constructionSuccessPtr = false;
115 return;
116 }
117 if (wrapper)
118 wrapper->addToken(offset);
119 }
120 offset = m_input.offset();
121 }
122
123 if (wrapper) {
124 wrapper->addToken(offset);
125 wrapper->finalizeConstruction(m_tokens.begin());
126 }
127}

139static bool isNewLine(UChar cc)
140{
141 // We check \r and \f here, since we have no preprocessing stage
142 return (cc == '\r' || cc == '\n' || cc == '\f');
143}

145// http://dev.w3.org/csswg/css-syntax/#check-if-two-code-points-are-a-valid-escape
146static bool twoCharsAreValidEscape(UChar first, UChar second)
147{
148 return first == '\\' && !isNewLine(second);
149}

151void CSSTokenizer::reconsume(UChar c)

156UChar CSSTokenizer::consume()
157{
158 UChar current = m_input.nextInputChar();
159 m_input.advance();
160 return current;

181CSSParserToken CSSTokenizer::blockEnd(CSSParserTokenType type, CSSParserTokenType startType)
182{
183 if (!m_blockStack.isEmpty() && m_blockStack.last() == startType) {
184 m_blockStack.removeLast();
185 return CSSParserToken(type, CSSParserToken::BlockEnd);

187 return CSSParserToken(type);

237CSSParserToken CSSTokenizer::lessThan(UChar cc)
238{
239 ASSERT_UNUSED(cc, cc == '<');
240 if (m_input.peek(0) == '!' && m_input.peek(1) == '-' && m_input.peek(2) == '-') {
241 m_input.advance(3);
242 return CSSParserToken(CDOToken);
243 }
244 return CSSParserToken(DelimiterToken, '<');

252CSSParserToken CSSTokenizer::hyphenMinus(UChar cc)
253{
254 if (nextCharsAreNumber(cc)) {
255 reconsume(cc);
256 return consumeNumericToken();
257 }
258 if (m_input.peek(0) == '-' && m_input.peek(1) == '>') {
259 m_input.advance(2);
260 return CSSParserToken(CDCToken);
261 }
262 if (nextCharsAreIdentifier(cc)) {
263 reconsume(cc);
264 return consumeIdentLikeToken();
265 }
266 return CSSParserToken(DelimiterToken, cc);
267}

290CSSParserToken CSSTokenizer::hash(UChar cc)
291{
292 UChar nextChar = m_input.peek(0);
293 if (isNameCodePoint(nextChar) || twoCharsAreValidEscape(nextChar, m_input.peek(1))) {
294 HashTokenType type = nextCharsAreIdentifier() ? HashTokenId : HashTokenUnrestricted;
295 return CSSParserToken(type, consumeName());
296 }
297
298 return CSSParserToken(DelimiterToken, cc);
299}

508CSSParserToken CSSTokenizer::nextToken()
509{
510 // Unlike the HTMLTokenizer, the CSS Syntax spec is written
511 // as a stateless, (fixed-size) look-ahead tokenizer.
512 // We could move to the stateful model and instead create
513 // states for all the "next 3 codepoints are X" cases.
514 // State-machine tokenizers are easier to write to handle
515 // incremental tokenization of partial sources.
516 // However, for now we follow the spec exactly.
517 UChar cc = consume();
518 CodePoint codePointFunc = 0;
519
520 if (isASCII(cc)) {
521 ASSERT_WITH_SECURITY_IMPLICATION(cc < codePointsNumber);
522 codePointFunc = codePoints[cc];
523 } else
524 codePointFunc = &CSSTokenizer::nameStart;
525
526 if (codePointFunc)
527 return ((this)->*(codePointFunc))(cc);
528 return CSSParserToken(DelimiterToken, cc);
529}

531// This method merges the following spec sections for efficiency
532// http://www.w3.org/TR/css3-syntax/#consume-a-number
533// http://www.w3.org/TR/css3-syntax/#convert-a-string-to-a-number
534CSSParserToken CSSTokenizer::consumeNumber()
535{
536 ASSERT(nextCharsAreNumber());
537
538 auto startOffset = m_input.offset();
539
540 NumericValueType type = IntegerValueType;
541 NumericSign sign = NoSign;
542 unsigned numberLength = 0;
543
544 UChar next = m_input.peek(0);
545 if (next == '+') {
546 ++numberLength;
547 sign = PlusSign;
548 } else if (next == '-') {
549 ++numberLength;
550 sign = MinusSign;
551 }
552
553 numberLength = m_input.skipWhilePredicate<isASCIIDigit>(numberLength);
554 next = m_input.peek(numberLength);
555 if (next == '.' && isASCIIDigit(m_input.peek(numberLength + 1))) {
556 type = NumberValueType;
557 numberLength = m_input.skipWhilePredicate<isASCIIDigit>(numberLength + 2);
558 next = m_input.peek(numberLength);
559 }
560
561 if (next == 'E' || next == 'e') {
562 next = m_input.peek(numberLength + 1);
563 if (isASCIIDigit(next)) {
564 type = NumberValueType;
565 numberLength = m_input.skipWhilePredicate<isASCIIDigit>(numberLength + 1);
566 } else if ((next == '+' || next == '-') && isASCIIDigit(m_input.peek(numberLength + 2))) {
567 type = NumberValueType;
568 numberLength = m_input.skipWhilePredicate<isASCIIDigit>(numberLength + 3);
569 }
570 }
571
572 double value = m_input.getDouble(0, numberLength);
573 m_input.advance(numberLength);
574
575 return CSSParserToken(value, type, sign, m_input.rangeAt(startOffset, m_input.offset() - startOffset));
576}

578// http://www.w3.org/TR/css3-syntax/#consume-a-numeric-token
579CSSParserToken CSSTokenizer::consumeNumericToken()
580{
581 CSSParserToken token = consumeNumber();
582 if (nextCharsAreIdentifier())
583 token.convertToDimensionWithUnit(consumeName());
584 else if (consumeIfNext('%'))
585 token.convertToPercentage();
586 return token;
587}

589// http://dev.w3.org/csswg/css-syntax/#consume-ident-like-token
590CSSParserToken CSSTokenizer::consumeIdentLikeToken()
591{
592 StringView name = consumeName();
593 if (consumeIfNext('(')) {
594 if (equalIgnoringASCIICase(name, "url")) {
595 // The spec is slightly different so as to avoid dropping whitespace
596 // tokens, but they wouldn't be used and this is easier.
597 m_input.advanceUntilNonWhitespace();
598 UChar next = m_input.peek(0);
599 if (next != '"' && next != '\'')
600 return consumeUrlToken();
601 }
602 return blockStart(LeftParenthesisToken, FunctionToken, name);
603 }
604 return CSSParserToken(IdentToken, name);
605}

607// http://dev.w3.org/csswg/css-syntax/#consume-a-string-token
608CSSParserToken CSSTokenizer::consumeStringTokenUntil(UChar endingCodePoint)
609{
610 // Strings without escapes get handled without allocations
611 for (unsigned size = 0; ; size++) {
612 UChar cc = m_input.peek(size);
613 if (cc == endingCodePoint) {
614 unsigned startOffset = m_input.offset();
615 m_input.advance(size + 1);
616 return CSSParserToken(StringToken, m_input.rangeAt(startOffset, size));
617 }
618 if (isNewLine(cc)) {
619 m_input.advance(size);
620 return CSSParserToken(BadStringToken);
621 }
622 if (cc == kEndOfFileMarker || cc == '\\')
623 break;
624 }
625
626 StringBuilder output;
627 while (true) {
628 UChar cc = consume();
629 if (cc == endingCodePoint || cc == kEndOfFileMarker)
630 return CSSParserToken(StringToken, registerString(output.toString()));
631 if (isNewLine(cc)) {
632 reconsume(cc);
633 return CSSParserToken(BadStringToken);
634 }
635 if (cc == '\\') {
636 if (m_input.nextInputChar() == kEndOfFileMarker)
637 continue;
638 if (isNewLine(m_input.peek(0)))
639 consumeSingleWhitespaceIfNext(); // This handles \r\n for us
640 else
641 output.appendCharacter(consumeEscape());
642 } else
643 output.append(cc);
644 }
645}

647// http://dev.w3.org/csswg/css-syntax/#non-printable-code-point
648static bool isNonPrintableCodePoint(UChar cc)
649{
650 return cc <= '\x8' || cc == '\xb' || (cc >= '\xe' && cc <= '\x1f') || cc == '\x7f';
651}
652
653// http://dev.w3.org/csswg/css-syntax/#consume-url-token
654CSSParserToken CSSTokenizer::consumeUrlToken()
655{
656 m_input.advanceUntilNonWhitespace();
657
658 // URL tokens without escapes get handled without allocations
659 for (unsigned size = 0; ; size++) {
660 UChar cc = m_input.peek(size);
661 if (cc == ')') {
662 unsigned startOffset = m_input.offset();
663 m_input.advance(size + 1);
664 return CSSParserToken(UrlToken, m_input.rangeAt(startOffset, size));
665 }
666 if (cc <= ' ' || cc == '\\' || cc == '"' || cc == '\'' || cc == '(' || cc == '\x7f')
667 break;
668 }
669
670 StringBuilder result;
671 while (true) {
672 UChar cc = consume();
673 if (cc == ')' || cc == kEndOfFileMarker)
674 return CSSParserToken(UrlToken, registerString(result.toString()));
675
676 if (isHTMLSpace(cc)) {
677 m_input.advanceUntilNonWhitespace();
678 if (consumeIfNext(')') || m_input.nextInputChar() == kEndOfFileMarker)
679 return CSSParserToken(UrlToken, registerString(result.toString()));
680 break;
681 }
682
683 if (cc == '"' || cc == '\'' || cc == '(' || isNonPrintableCodePoint(cc))
684 break;
685
686 if (cc == '\\') {
687 if (twoCharsAreValidEscape(cc, m_input.peek(0))) {
688 result.appendCharacter(consumeEscape());
689 continue;
690 }
691 break;
692 }
693
694 result.append(cc);
695 }
696
697 consumeBadUrlRemnants();
698 return CSSParserToken(BadUrlToken);
699}

701// http://dev.w3.org/csswg/css-syntax/#consume-the-remnants-of-a-bad-url
702void CSSTokenizer::consumeBadUrlRemnants()
703{
704 while (true) {
705 UChar cc = consume();
706 if (cc == ')' || cc == kEndOfFileMarker)
707 return;
708 if (twoCharsAreValidEscape(cc, m_input.peek(0)))
709 consumeEscape();
710 }
711}
712
713void CSSTokenizer::consumeSingleWhitespaceIfNext()
714{
715 // We check for \r\n and HTML spaces since we don't do preprocessing
716 UChar next = m_input.peek(0);
717 if (next == '\r' && m_input.peek(1) == '\n')
718 m_input.advance(2);
719 else if (isHTMLSpace(next))
720 m_input.advance();
721}
722
723void CSSTokenizer::consumeUntilCommentEndFound()
724{
725 UChar c = consume();
726 while (true) {
727 if (c == kEndOfFileMarker)
728 return;
729 if (c != '*') {
730 c = consume();
731 continue;
732 }
733 c = consume();
734 if (c == '/')
735 return;
736 }
737}
738
739bool CSSTokenizer::consumeIfNext(UChar character)
740{
741 // Since we're not doing replacement we can't tell the difference from
742 // a NUL in the middle and the kEndOfFileMarker, so character must not be
743 // NUL.
744 ASSERT(character);
745 if (m_input.peek(0) == character) {
746 m_input.advance();
747 return true;
748 }
749 return false;
750}

752// http://www.w3.org/TR/css3-syntax/#consume-a-name
753StringView CSSTokenizer::consumeName()
754{
755 // Names without escapes get handled without allocations
756 for (unsigned size = 0; ; ++size) {
757 UChar cc = m_input.peek(size);
758 if (isNameCodePoint(cc))
759 continue;
760 // peek will return NUL when we hit the end of the
761 // input. In that case we want to still use the rangeAt() fast path
762 // below.
763 if (cc == kEndOfFileMarker && m_input.offset() + size < m_input.length())
764 break;
765 if (cc == '\\')
766 break;
767 unsigned startOffset = m_input.offset();
768 m_input.advance(size);
769 return m_input.rangeAt(startOffset, size);
770 }
771
772 StringBuilder result;
773 while (true) {
774 UChar cc = consume();
775 if (isNameCodePoint(cc)) {
776 result.append(cc);
777 continue;
778 }
779 if (twoCharsAreValidEscape(cc, m_input.peek(0))) {
780 result.appendCharacter(consumeEscape());
781 continue;
782 }
783 reconsume(cc);
784 return registerString(result.toString());
785 }
786}

788// http://dev.w3.org/csswg/css-syntax/#consume-an-escaped-code-point
789UChar32 CSSTokenizer::consumeEscape()
790{
791 UChar cc = consume();
792 ASSERT(!isNewLine(cc));
793 if (isASCIIHexDigit(cc)) {
794 unsigned consumedHexDigits = 1;
795 StringBuilder hexChars;
796 hexChars.append(cc);
797 while (consumedHexDigits < 6 && isASCIIHexDigit(m_input.peek(0))) {
798 cc = consume();
799 hexChars.append(cc);
800 consumedHexDigits++;
801 };
802 consumeSingleWhitespaceIfNext();
803 auto codePoint = parseInteger<uint32_t>(hexChars, 16).value();
804 if (!codePoint || (0xD800 <= codePoint && codePoint <= 0xDFFF) || codePoint > 0x10FFFF)
805 return replacementCharacter;
806 return codePoint;
807 }
808
809 if (cc == kEndOfFileMarker)
810 return replacementCharacter;
811 return cc;
812}

820bool CSSTokenizer::nextCharsAreNumber(UChar first)
821{
822 UChar second = m_input.peek(0);
823 if (isASCIIDigit(first))
824 return true;
825 if (first == '+' || first == '-')
826 return ((isASCIIDigit(second)) || (second == '.' && isASCIIDigit(m_input.peek(1))));
827 if (first =='.')
828 return (isASCIIDigit(second));
829 return false;
830}

841bool CSSTokenizer::nextCharsAreIdentifier(UChar first)
842{
843 UChar second = m_input.peek(0);
844 if (isNameStartCodePoint(first) || twoCharsAreValidEscape(first, second))
845 return true;
846
847 if (first == '-')
848 return isNameStartCodePoint(second) || second == '-' || nextTwoCharsAreValidEscape();
849
850 return false;
851}
