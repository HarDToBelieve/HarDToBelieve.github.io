40bool CSSVariableParser::isValidVariableName(const CSSParserToken& token)
41{
42 if (token.type() != IdentToken)
43 return false;
44
45 StringView value = token.value();
46 return value.length() >= 2 && value[0] == '-' && value[1] == '-';
47}
48
49bool CSSVariableParser::isValidVariableName(const String& string)
50{
51 return string.length() >= 2 && string[0] == '-' && string[1] == '-';
52}

62static bool classifyBlock(CSSParserTokenRange range, bool& hasReferences, bool& hasAtApplyRule, const CSSParserContext& parserContext, bool isTopLevelBlock = true)
63{
64 while (!range.atEnd()) {
65 if (range.peek().getBlockType() == CSSParserToken::BlockStart) {
66 const CSSParserToken& token = range.peek();
67 CSSParserTokenRange block = range.consumeBlock();
68 if (token.functionId() == CSSValueVar) {
69 if (!isValidVariableReference(block, hasAtApplyRule, parserContext))
70 return false; // Bail if any references are invalid
71 hasReferences = true;
72 continue;
73 }
74 if (token.functionId() == CSSValueEnv && parserContext.constantPropertiesEnabled) {
75 if (!isValidConstantReference(block, hasAtApplyRule, parserContext))
76 return false; // Bail if any references are invalid
77 hasReferences = true;
78 continue;
79 }
80 if (!classifyBlock(block, hasReferences, hasAtApplyRule, parserContext, false))
81 return false;
82 continue;
83 }
84
85 ASSERT(range.peek().getBlockType() != CSSParserToken::BlockEnd);
86
87 const CSSParserToken& token = range.consume();
88 switch (token.type()) {
89 case AtKeywordToken: {
90 if (equalIgnoringASCIICase(token.value(), "apply")) {
91 range.consumeWhitespace();
92 const CSSParserToken& variableName = range.consumeIncludingWhitespace();
93 if (!CSSVariableParser::isValidVariableName(variableName)
94 || !(range.atEnd() || range.peek().type() == SemicolonToken || range.peek().type() == RightBraceToken))
95 return false;
96 hasAtApplyRule = true;
97 }
98 break;
99 }
100 case DelimiterToken: {
101 if (token.delimiter() == '!' && isTopLevelBlock)
102 return false;
103 break;
104 }
105 case RightParenthesisToken:
106 case RightBraceToken:
107 case RightBracketToken:
108 case BadStringToken:
109 case BadUrlToken:
110 return false;
111 case SemicolonToken:
112 if (isTopLevelBlock)
113 return false;
114 break;
115 default:
116 break;
117 }
118 }
119 return true;
120}

122bool isValidVariableReference(CSSParserTokenRange range, bool& hasAtApplyRule, const CSSParserContext& parserContext)
123{
124 range.consumeWhitespace();
125 if (!CSSVariableParser::isValidVariableName(range.consumeIncludingWhitespace()))
126 return false;
127 if (range.atEnd())
128 return true;
129
130 if (!CSSPropertyParserHelpers::consumeCommaIncludingWhitespace(range))
131 return false;
132 if (range.atEnd())
133 return true;
134
135 bool hasReferences = false;
136 return classifyBlock(range, hasReferences, hasAtApplyRule, parserContext);
137}
138
139bool isValidConstantReference(CSSParserTokenRange range, bool& hasAtApplyRule, const CSSParserContext& parserContext)
140{
141 range.consumeWhitespace();
142 if (!isValidConstantName(range.consumeIncludingWhitespace()))
143 return false;
144 if (range.atEnd())
145 return true;
146
147 if (!CSSPropertyParserHelpers::consumeCommaIncludingWhitespace(range))
148 return false;
149 if (range.atEnd())
150 return true;
151
152 bool hasReferences = false;
153 return classifyBlock(range, hasReferences, hasAtApplyRule, parserContext);
154}

156static CSSValueID classifyVariableRange(CSSParserTokenRange range, bool& hasReferences, bool& hasAtApplyRule, const CSSParserContext& parserContext)
157{
158 hasReferences = false;
159 hasAtApplyRule = false;
160
161 range.consumeWhitespace();
162 if (range.peek().type() == IdentToken) {
163 CSSValueID id = range.consumeIncludingWhitespace().id();
164 if (range.atEnd() && (id == CSSValueInherit || id == CSSValueInitial || id == CSSValueUnset || id == CSSValueRevert))
165 return id;
166 }
167
168 if (classifyBlock(range, hasReferences, hasAtApplyRule, parserContext))
169 return CSSValueInternalVariableValue;
170 return CSSValueInvalid;
171}

173bool CSSVariableParser::containsValidVariableReferences(CSSParserTokenRange range, const CSSParserContext& parserContext)
174{
175 bool hasReferences;
176 bool hasAtApplyRule;
177 CSSValueID type = classifyVariableRange(range, hasReferences, hasAtApplyRule, parserContext);
178 return type == CSSValueInternalVariableValue && hasReferences && !hasAtApplyRule;
179}

181RefPtr<CSSCustomPropertyValue> CSSVariableParser::parseDeclarationValue(const AtomString& variableName, CSSParserTokenRange range, const CSSParserContext& parserContext)
182{
183 if (range.atEnd())
184 return nullptr;
185
186 bool hasReferences;
187 bool hasAtApplyRule;
188 CSSValueID type = classifyVariableRange(range, hasReferences, hasAtApplyRule, parserContext);
189
190 if (type == CSSValueInvalid)
191 return nullptr;
192 if (type == CSSValueInternalVariableValue)
193 return CSSCustomPropertyValue::createUnresolved(variableName, CSSVariableReferenceValue::create(range));
194 return CSSCustomPropertyValue::createUnresolved(variableName, type);
195}
