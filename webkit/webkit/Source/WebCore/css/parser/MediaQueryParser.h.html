42class MediaQuerySet;
43struct CSSParserContext;
44
45class MediaQueryParser {
46 WTF_MAKE_NONCOPYABLE(MediaQueryParser);
47public:
48 static RefPtr<MediaQuerySet> parseMediaQuerySet(const String&, MediaQueryParserContext);
49 static RefPtr<MediaQuerySet> parseMediaQuerySet(CSSParserTokenRange, MediaQueryParserContext);
50 static RefPtr<MediaQuerySet> parseMediaCondition(CSSParserTokenRange, MediaQueryParserContext);
51
52private:
53 enum ParserType {
54 MediaQuerySetParser,
55 MediaConditionParser,
56 };
57
58 MediaQueryParser(ParserType, MediaQueryParserContext);
59 virtual ~MediaQueryParser();
60
61 RefPtr<MediaQuerySet> parseInternal(CSSParserTokenRange);
62
63 void processToken(const CSSParserToken&, CSSParserTokenRange&);
64
65 void readRestrictor(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
66 void readMediaNot(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
67 void readMediaType(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
68 void readAnd(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
69 void readFeatureStart(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
70 void readFeature(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
71 void readFeatureColon(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
72 void readFeatureValue(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
73 void readFeatureEnd(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
74 void skipUntilComma(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
75 void skipUntilBlockEnd(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
76 void done(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
77
78 using State = void (MediaQueryParser::*)(CSSParserTokenType, const CSSParserToken&, CSSParserTokenRange&);
79
80 void setStateAndRestrict(State, MediaQuery::Restrictor);
81 void handleBlocks(const CSSParserToken&);
82
83 void commitMediaQuery();
84
85 class MediaQueryData {
86 WTF_MAKE_NONCOPYABLE(MediaQueryData);
87 public:
88 explicit MediaQueryData(MediaQueryParserContext);
89 void clear();
90 void addExpression(CSSParserTokenRange&);
91 bool lastExpressionValid();
92 void removeLastExpression();
93 void setMediaType(const String& mediaType) { m_mediaType = mediaType; }
94
95 MediaQuery::Restrictor restrictor() const { return m_restrictor; }
96 Vector<MediaQueryExpression>& expressions() { return m_expressions; }
97 const std::optional<String>& mediaType() const { return m_mediaType; }
98
99 bool currentMediaQueryChanged() const
100 {
101 return (m_restrictor != MediaQuery::None || m_mediaType || !m_expressions.isEmpty());
102 }
103 MediaQuery::Restrictor restrictor() { return m_restrictor; }
104
105 void setRestrictor(MediaQuery::Restrictor restrictor) { m_restrictor = restrictor; }
106
107 void setMediaFeature(const String& str) { m_mediaFeature = str; }
108
109 void setMediaQueryParserContext(MediaQueryParserContext context) { m_context = context; }
110
111 private:
112 MediaQuery::Restrictor m_restrictor { MediaQuery::None };
113 std::optional<String> m_mediaType;
114 Vector<MediaQueryExpression> m_expressions;
115 String m_mediaFeature;
116 MediaQueryParserContext m_context;
117 };
118
119 State m_state;
120 ParserType m_parserType;
121 MediaQueryData m_mediaQueryData;
122 RefPtr<MediaQuerySet> m_querySet;
123 MediaQueryBlockWatcher m_blockWatcher;
124
125 const static State ReadRestrictor;
126 const static State ReadMediaNot;
127 const static State ReadMediaType;
128 const static State ReadAnd;
129 const static State ReadFeatureStart;
130 const static State ReadFeature;
131 const static State ReadFeatureColon;
132 const static State ReadFeatureValue;
133 const static State ReadFeatureEnd;
134 const static State SkipUntilComma;
135 const static State SkipUntilBlockEnd;
136 const static State Done;
137};
