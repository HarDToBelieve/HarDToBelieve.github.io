41enum class SelectorContext {
42 // Rule Collector needs a resolvingMode and can modify the tree as it matches.
43 RuleCollector,
44
45 // Query Selector does not modify the tree and never match :visited.
46 QuerySelector
47};
48
49void compileSelector(CompiledSelector&, const CSSSelector*, SelectorContext);
50
51#if CPU(ARM64E)
52extern "C" unsigned vmEntryToCSSJIT(uintptr_t, uintptr_t, uintptr_t, const void* codePtr);
53extern "C" void vmEntryToCSSJITAfter(void);
54#endif
55
56inline unsigned ruleCollectorSimpleSelectorChecker(CompiledSelector& compiledSelector, const Element* element, unsigned* value)
57{
58 ASSERT(compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker);
59#if CPU(ARM64E)
60 if (JSC::Options::useJITCage())
61 return vmEntryToCSSJIT(bitwise_cast<uintptr_t>(element), bitwise_cast<uintptr_t>(value), 0, compiledSelector.codeRef.code().executableAddress());
62#endif
63 using RuleCollectorSimpleSelectorChecker = unsigned(*)(const Element*, unsigned*);
64 return WTF::untagCFunctionPtr<RuleCollectorSimpleSelectorChecker, JSC::CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress())(element, value);
65}
66
67inline unsigned querySelectorSimpleSelectorChecker(CompiledSelector& compiledSelector, const Element* element)
68{
69 ASSERT(compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker);
70#if CPU(ARM64E)
71 if (JSC::Options::useJITCage())
72 return vmEntryToCSSJIT(bitwise_cast<uintptr_t>(element), 0, 0, compiledSelector.codeRef.code().executableAddress());
73#endif
74 using QuerySelectorSimpleSelectorChecker = unsigned(*)(const Element*);
75 return WTF::untagCFunctionPtr<QuerySelectorSimpleSelectorChecker, JSC::CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress())(element);
76}
77
78inline unsigned ruleCollectorSelectorCheckerWithCheckingContext(CompiledSelector& compiledSelector, const Element* element, SelectorChecker::CheckingContext* context, unsigned* value)
79{
80 ASSERT(compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext);
81#if CPU(ARM64E)
82 if (JSC::Options::useJITCage())
83 return vmEntryToCSSJIT(bitwise_cast<uintptr_t>(element), bitwise_cast<uintptr_t>(context), bitwise_cast<uintptr_t>(value), compiledSelector.codeRef.code().executableAddress());
84#endif
85 using RuleCollectorSelectorCheckerWithCheckingContext = unsigned(*)(const Element*, SelectorChecker::CheckingContext*, unsigned*);
86 return WTF::untagCFunctionPtr<RuleCollectorSelectorCheckerWithCheckingContext, JSC::CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress())(element, context, value);
87}
88
89inline unsigned querySelectorSelectorCheckerWithCheckingContext(CompiledSelector& compiledSelector, const Element* element, const SelectorChecker::CheckingContext* context)
90{
91 ASSERT(compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext);
92#if CPU(ARM64E)
93 if (JSC::Options::useJITCage())
94 return vmEntryToCSSJIT(bitwise_cast<uintptr_t>(element), bitwise_cast<uintptr_t>(context), 0, compiledSelector.codeRef.code().executableAddress());
95#endif
96 using QuerySelectorSelectorCheckerWithCheckingContext = unsigned(*)(const Element*, const SelectorChecker::CheckingContext*);
97 return WTF::untagCFunctionPtr<QuerySelectorSelectorCheckerWithCheckingContext, JSC::CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress())(element, context);
98}
