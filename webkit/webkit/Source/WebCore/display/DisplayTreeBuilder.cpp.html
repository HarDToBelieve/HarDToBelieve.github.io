50class PositioningContext {
51public:
52 PositioningContext(const Display::ContainerBox& rootDisplayBox)
53 : m_fixedPositionContainer({ rootDisplayBox, { } })
54 , m_absolutePositionContainer({ rootDisplayBox, { } })
55 , m_inFlowContainer ({ rootDisplayBox, { } })
56 {
57 }
58
59 PositioningContext contextForDescendants(const Layout::Box& layoutBox, const Layout::BoxGeometry geometry, const Display::ContainerBox& displayBox) const
60 {
61 auto currentOffset = containingBlockContextForLayoutBox(layoutBox).offsetFromRoot;
62
63 auto borderBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(geometry) };
64 currentOffset += toLayoutSize(borderBoxRect.location());
65
66 auto currentBoxes = ContainingBlockContext { displayBox, currentOffset };
67 return {
68 layoutBox.isContainingBlockForFixedPosition() ? currentBoxes : m_fixedPositionContainer,
69 layoutBox.isContainingBlockForOutOfFlowPosition() ? currentBoxes : m_absolutePositionContainer,
70 layoutBox.isContainingBlockForInFlow() ? currentBoxes : m_inFlowContainer
71 };
72 }
73
74 const ContainingBlockContext& containingBlockContextForLayoutBox(const Layout::Box& layoutBox) const
75 {
76 if (layoutBox.isFixedPositioned())
77 return m_fixedPositionContainer;
78
79 if (layoutBox.isOutOfFlowPositioned())
80 return m_absolutePositionContainer;
81
82 return m_inFlowContainer;
83 }
84
85 const ContainingBlockContext& inFlowContainingBlockContext() const { return m_inFlowContainer; }
86
87private:
88 PositioningContext(const ContainingBlockContext& fixedContainer, const ContainingBlockContext& absoluteContainer, const ContainingBlockContext& inFlowContainer)
89 : m_fixedPositionContainer(fixedContainer)
90 , m_absolutePositionContainer(absoluteContainer)
91 , m_inFlowContainer(inFlowContainer)
92 {
93 }
94
95 ContainingBlockContext m_fixedPositionContainer;
96 ContainingBlockContext m_absolutePositionContainer;
97 ContainingBlockContext m_inFlowContainer;
98};

103 PositioningContext positioningContext;

105 StackingItem& currentStackingContextItem;
106
107 AbsoluteFloatRect currentStackingItemPaintedContentExtent; // This stacking item's contents only.
108 AbsoluteFloatRect currentStackingItemPaintingExtent; // Including descendant items.
109};

120std::unique_ptr<Tree> TreeBuilder::build(const Layout::LayoutState& layoutState)
121{
122 ASSERT(layoutState.hasRoot());
123
124 auto& rootLayoutBox = layoutState.root();
125
126#if ENABLE(TREE_DEBUGGING)
127 LOG_WITH_STREAM(FormattingContextLayout, stream << "Building display tree for:\n" << layoutTreeAsText(rootLayoutBox, &layoutState));
128#endif
129
130 ASSERT(!m_tree);
131 m_tree = makeUnique<Tree>();
132
133 m_rootBackgroundPropgation = BoxFactory::determineRootBackgroundPropagation(rootLayoutBox);
134
135 auto geometry = layoutState.geometryForBox(rootLayoutBox);
136 auto rootDisplayBox = m_boxFactory.displayBoxForRootBox(rootLayoutBox, geometry, m_rootBackgroundPropgation);
137
138 auto insertionPosition = InsertionPosition { *rootDisplayBox };
139 auto rootStackingItem = makeUnique<StackingItem>(WTFMove(rootDisplayBox));
140
141 if (rootLayoutBox.firstChild()) {
142 auto& rootContainerBox = downcast<ContainerBox>(rootStackingItem->box());
143 m_stateStack->append({ { rootContainerBox }, *rootStackingItem, *rootStackingItem, { }, { } });
144 recursiveBuildDisplayTree(layoutState, *rootLayoutBox.firstChild(), insertionPosition);
145 }
146
147#if ENABLE(TREE_DEBUGGING)
148 LOG_WITH_STREAM(FormattingContextLayout, stream << "Display tree:\n" << displayTreeAsText(*rootStackingItem));
149#endif
150 m_tree->setRootStackingItem(WTFMove(rootStackingItem));
151 return WTFMove(m_tree);
152}

154void TreeBuilder::pushStateForBoxDescendants(const Layout::ContainerBox& layoutContainerBox, const Layout::BoxGeometry& layoutGeometry, const ContainerBox& displayBox, StackingItem* boxStackingItem)
155{
156 auto& positioningContext = m_stateStack->last().positioningContext;
157
158 auto currentStackingItem = [&]() -> StackingItem& {
159 if (boxStackingItem)
160 return *boxStackingItem;
161
162 return m_stateStack->last().currentStackingItem;
163 };
164
165 auto currentStackingContextItem = [&]() -> StackingItem& {
166 if (boxStackingItem && boxStackingItem->isStackingContext())
167 return *boxStackingItem;
168
169 return m_stateStack->last().currentStackingContextItem;
170 };
171
172 m_stateStack->append({
173 positioningContext.contextForDescendants(layoutContainerBox, layoutGeometry, displayBox),
174 currentStackingItem(),
175 currentStackingContextItem(),
176 { },
177 { }
178 });
179}

181void TreeBuilder::popState(const BoxModelBox& currentBox)
182{
183 ASSERT(m_stateStack->size() > 1);
184 auto& currentState = m_stateStack->last();
185 auto& previousState = m_stateStack->at(m_stateStack->size() - 2);
186
187 bool endingStackingItem = &currentState.currentStackingItem.box() == &currentBox;
188 bool endingStackingContextItem = &currentState.currentStackingContextItem.box() == &currentBox;
189
190 // We've accumulated into currentStackingItemPaintedContentExtent while building descendant boxes.
191 auto paintedContentExtent = currentState.currentStackingItemPaintedContentExtent;
192
193 // Often, but not always, containingBlockBox will be currentBox. If it has overflow clipping, clip the rect from descendants.
194 auto& containingBlockBox = currentState.inFlowContainingBlockBox();
195 if (containingBlockBox.style().hasClippedOverflow()) {
196 auto clipRect = containingBlockBox.absolutePaddingBoxRect();
197 paintedContentExtent.intersect(clipRect);
198 }
199
200 // Now union with this box's extent (which is outside the clip).
201 auto boxPaintingExtent = unionRect(paintedContentExtent, currentBox.absolutePaintingExtent());
202
203 if (endingStackingItem) {
204 currentState.currentStackingItem.setPaintedContentBounds(boxPaintingExtent);
205
206 // Now accumulate this item's extent (possibly clipped) into that of its parent item.
207 auto descendantItemsPaintingExtent = currentState.currentStackingItemPaintingExtent;
208
209 if (containingBlockBox.style().hasClippedOverflow()) {
210 auto clipRect = containingBlockBox.absolutePaddingBoxRect();
211 descendantItemsPaintingExtent.intersect(clipRect);
212 }
213
214 auto itemPaintingExtent = unionRect(descendantItemsPaintingExtent, boxPaintingExtent);
215 currentState.currentStackingItem.setPaintedBoundsIncludingDescendantItems(itemPaintingExtent);
216
217 previousState.currentStackingItemPaintingExtent.unite(itemPaintingExtent);
218 } else
219 previousState.currentStackingItemPaintedContentExtent.unite(boxPaintingExtent);
220
221 if (endingStackingContextItem)
222 currentState.currentStackingContextItem.sortLists();
223
224 m_stateStack->removeLast();
225}

227void TreeBuilder::didAppendNonContainerStackingItem(StackingItem& item)
228{
229 // This is s simplified version of pushStateForBoxDescendants/popState.
230 auto currentStackingContextItem = [&]() -> StackingItem& {
231 if (item.isStackingContext())
232 return item;
233
234 return m_stateStack->last().currentStackingContextItem;
235 };
236
237 m_stateStack->append({
238 m_stateStack->last().positioningContext,
239 item,
240 currentStackingContextItem(),
241 { },
242 { }
243 });
244
245 popState(item.box());
246}

248void TreeBuilder::accountForBoxPaintingExtent(const Box& box)

250 // FIXME: Need to account for transforms.
251 currentState().currentStackingItemPaintedContentExtent.unite(box.absolutePaintingExtent());

265void TreeBuilder::insert(std::unique_ptr<Box>&& box, InsertionPosition& insertionPosition) const
266{
267 box->setParent(&insertionPosition.container);
268 if (insertionPosition.currentChild) {
269 auto boxPtr = box.get();
270 insertionPosition.currentChild->setNextSibling(WTFMove(box));
271 insertionPosition.currentChild = boxPtr;
272 } else {
273 insertionPosition.currentChild = box.get();
274 insertionPosition.container.setFirstChild(WTFMove(box));
275 }
276}

278StackingItem* TreeBuilder::insertIntoTree(std::unique_ptr<Box>&& box, InsertionPosition& insertionPosition, WillTraverseDescendants willTraverseDescendants)
279{
280 if (box->participatesInZOrderSorting() && is<BoxModelBox>(*box)) {
281 auto boxModelBox = std::unique_ptr<BoxModelBox> { downcast<BoxModelBox>(box.release()) };
282 auto stackingItem = makeUnique<StackingItem>(WTFMove(boxModelBox));
283
284 // This painted bounds will get extended as we traverse box descendants.
285 stackingItem->setPaintedContentBounds(stackingItem->box().absolutePaintingExtent());
286
287 auto* stackingItemPtr = stackingItem.get();
288 currentState().currentStackingContextItem.addChildStackingItem(WTFMove(stackingItem));
289
290 if (willTraverseDescendants == WillTraverseDescendants::No)
291 didAppendNonContainerStackingItem(*stackingItemPtr);
292
293 return stackingItemPtr;
294 }
295
296 insert(WTFMove(box), insertionPosition);
297
298 if (willTraverseDescendants == WillTraverseDescendants::No)
299 accountForBoxPaintingExtent(*insertionPosition.currentChild);
300
301 return nullptr;
302}

304void TreeBuilder::buildInlineDisplayTree(const Layout::LayoutState& layoutState, const Layout::ContainerBox& inlineFormattingRoot, InsertionPosition& insertionPosition)
305{
306 auto& inlineFormattingState = layoutState.establishedInlineFormattingState(inlineFormattingRoot);
307
308 for (auto& run : inlineFormattingState.lineRuns()) {
309 if (run.text()) {
310 auto& lineGeometry = inlineFormattingState.lines().at(run.lineIndex());
311 auto textBox = m_boxFactory.displayBoxForTextRun(run, lineGeometry, positioningContext().inFlowContainingBlockContext());
312 insert(WTFMove(textBox), insertionPosition);
313 accountForBoxPaintingExtent(*insertionPosition.currentChild);
314 continue;
315 }
316
317 if (is<Layout::ContainerBox>(run.layoutBox())) {
318 recursiveBuildDisplayTree(layoutState, run.layoutBox(), insertionPosition);
319 continue;
320 }
321
322 // FIXME: Workaround for webkit.orgb/b/219335.
323 if (run.layoutBox().isLineBreakBox() && run.layoutBox().isOutOfFlowPositioned())
324 continue;
325
326 auto geometry = layoutState.geometryForBox(run.layoutBox());
327 auto displayBox = m_boxFactory.displayBoxForLayoutBox(run.layoutBox(), geometry, positioningContext().inFlowContainingBlockContext());
328 insertIntoTree(WTFMove(displayBox), insertionPosition, WillTraverseDescendants::No);
329 }
330}

332void TreeBuilder::recursiveBuildDisplayTree(const Layout::LayoutState& layoutState, const Layout::Box& layoutBox, InsertionPosition& insertionPosition)
333{
334 auto geometry = layoutState.geometryForBox(layoutBox);
335 std::unique_ptr<Box> displayBox;
336
337 auto& containingBlockContext = positioningContext().containingBlockContextForLayoutBox(layoutBox);
338 if (layoutBox.isBodyBox())
339 displayBox = m_boxFactory.displayBoxForBodyBox(layoutBox, geometry, containingBlockContext, m_rootBackgroundPropgation);
340 else
341 displayBox = m_boxFactory.displayBoxForLayoutBox(layoutBox, geometry, containingBlockContext);
342
343 Box& currentBox = *displayBox;
344
345 auto willTraverseDescendants = (is<Layout::ContainerBox>(layoutBox) && downcast<Layout::ContainerBox>(layoutBox).hasChild()) ? WillTraverseDescendants::Yes : WillTraverseDescendants::No;
346 auto* stackingItem = insertIntoTree(WTFMove(displayBox), insertionPosition, willTraverseDescendants);
347 if (willTraverseDescendants == WillTraverseDescendants::No)
348 return;
349
350 auto& layoutContainerBox = downcast<Layout::ContainerBox>(layoutBox);
351
352 ContainerBox& currentContainerBox = downcast<ContainerBox>(currentBox);
353 pushStateForBoxDescendants(layoutContainerBox, geometry, currentContainerBox, stackingItem);
354
355 auto insertionPositionForChildren = InsertionPosition { currentContainerBox };
356
357 enum class DescendantBoxInclusion { AllBoxes, OutOfFlowOnly };
358 auto boxInclusion = DescendantBoxInclusion::AllBoxes;
359
360 if (layoutContainerBox.establishesInlineFormattingContext()) {
361 buildInlineDisplayTree(layoutState, downcast<Layout::ContainerBox>(layoutContainerBox), insertionPositionForChildren);
362 boxInclusion = DescendantBoxInclusion::OutOfFlowOnly;
363 }
364
365 auto includeBox = [](DescendantBoxInclusion boxInclusion, const Layout::Box& box) {
366 switch (boxInclusion) {
367 case DescendantBoxInclusion::AllBoxes: return true;
368 case DescendantBoxInclusion::OutOfFlowOnly: return !box.isInFlow();
369 }
370 return false;
371 };
372
373 for (auto& child : Layout::childrenOfType<Layout::Box>(layoutContainerBox)) {
374 if (includeBox(boxInclusion, child) && layoutState.hasBoxGeometry(child))
375 recursiveBuildDisplayTree(layoutState, child, insertionPositionForChildren);
376 }
377
378 popState(currentContainerBox);
379}

383static void outputDisplayBox(TextStream& stream, const Box& displayBox, bool writeIndent = true)
384{
385 if (writeIndent)
386 stream.writeIndent();
387
388 stream << displayBox.debugDescription();
389 stream << "\n";
390}
391
392static void outputDisplayTree(TextStream& stream, const Box& displayBox, bool writeFirstItemIndent = true)
393{
394 outputDisplayBox(stream, displayBox, writeFirstItemIndent);
395
396 if (is<ContainerBox>(displayBox)) {
397 TextStream::IndentScope indent(stream);
398 for (auto child = downcast<ContainerBox>(displayBox).firstChild(); child; child = child->nextSibling())
399 outputDisplayTree(stream, *child);
400 }
401}
402
403String displayTreeAsText(const Box& box)
404{
405 TextStream stream(TextStream::LineMode::MultipleLine, TextStream::Formatting::SVGStyleRect);
406 outputDisplayTree(stream, box);
407 return stream.release();
408}

415
416static void outputStackingTree(TextStream& stream, const char* prefix, const StackingItem& stackingItem)
417{
418 stream.writeIndent();
419 stream << "[" << prefix << "] item (" << &stackingItem << ") " << stackingItem.paintedContentBounds() << " - ";
420
421 TextStream::IndentScope indent(stream);
422 outputDisplayTree(stream, stackingItem.box(), false);
423
424 {
425 TextStream::IndentScope indent(stream);
426 for (auto& childStackingItem : stackingItem.negativeZOrderList())
427 outputStackingTree(stream, "-", *childStackingItem);
428
429 for (auto& childStackingItem : stackingItem.positiveZOrderList())
430 outputStackingTree(stream, childStackingItem->isStackingContext() ? "+" : "p", *childStackingItem);
431 }
432}
433
434String displayTreeAsText(const StackingItem& stackingItem)
435{
436 TextStream stream(TextStream::LineMode::MultipleLine);
437 outputStackingTree(stream, "+", stackingItem);
438 return stream.release();
439}
