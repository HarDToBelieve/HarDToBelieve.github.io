38namespace Display {

40DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(Box);
41
42Box::Box(Tree& tree, AbsoluteFloatRect absoluteRect, Style&& displayStyle, OptionSet<TypeFlags> flags)
43 : m_tree(tree)
44 , m_absoluteBoxRect(absoluteRect)
45 , m_style(WTFMove(displayStyle))
46 , m_typeFlags(flags)
47{
48}
49
50Box::~Box() = default;
51
52void Box::setNextSibling(std::unique_ptr<Box>&& box)
53{
54 m_nextSibling = WTFMove(box);
55}
56
57bool Box::participatesInZOrderSorting() const
58{
59 return !isLineBreakBox() && style().participatesInZOrderSorting();
60}
61
62void Box::setNeedsDisplay(std::optional<AbsoluteFloatRect> subrect)
63{
64 m_tree.setBoxNeedsDisplay(*this, subrect);
65}
66
67const char* Box::boxName() const
68{
69 return "box";
70}
71
72String Box::debugDescription() const
73{
74 TextStream stream;
75 stream << boxName() << " " << absoluteBoxRect() << " (" << this << ")";
76 return stream.release();
77}
