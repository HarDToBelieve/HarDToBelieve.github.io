30#include "DisplayStyle.h"
31#include "FloatRect.h"
32#include <wtf/IsoMalloc.h>
33#include <wtf/OptionSet.h>

35namespace WebCore {
36namespace Display {
37
38class ContainerBox;
39class Tree;
40
41// FIXME: Make this a strong type.
42using AbsoluteFloatRect = FloatRect;
43
44DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(Box);
45class Box {
46 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(Box);
47public:
48 enum class TypeFlags : uint8_t {
49 BoxModelBox = 1 << 0,
50 ContainerBox = 1 << 1,
51 ImageBox = 1 << 2,
52 TextBox = 1 << 3,
53 LineBreakBox = 1 << 4, // FIXME: Workaround for webkit.org/b/219335
54 };
55
56 Box(Tree&, AbsoluteFloatRect, Style&&, OptionSet<TypeFlags> = { });
57 virtual ~Box();
58
59 const Style& style() const { return m_style; }
60
61 AbsoluteFloatRect absoluteBoxRect() const { return m_absoluteBoxRect; }
62
63 virtual AbsoluteFloatRect absolutePaintingExtent() const { return m_absoluteBoxRect; }
64
65 bool isBoxModelBox() const { return m_typeFlags.contains(TypeFlags::BoxModelBox); }
66 bool isContainerBox() const { return m_typeFlags.contains(TypeFlags::ContainerBox); }
67 bool isImageBox() const { return m_typeFlags.contains(TypeFlags::ImageBox); }
68 bool isReplacedBox() const { return m_typeFlags.contains(TypeFlags::ImageBox); /* and other types later. */ }
69 bool isTextBox() const { return m_typeFlags.contains(TypeFlags::TextBox); }
70 bool isLineBreakBox() const { return m_typeFlags.contains(TypeFlags::LineBreakBox); }
71
72 bool participatesInZOrderSorting() const;
73
74 ContainerBox* parent() const { return m_parent; }
75 void setParent(ContainerBox* parent) { m_parent = parent; }
76
77 const Box* nextSibling() const { return m_nextSibling.get(); }
78 void setNextSibling(std::unique_ptr<Box>&&);
79
80 void setNeedsDisplay(std::optional<AbsoluteFloatRect> subrect = std::nullopt);
81
82 virtual String debugDescription() const;
83
84private:
85 virtual const char* boxName() const;
86
87 const Tree& m_tree;
88 AbsoluteFloatRect m_absoluteBoxRect;
89 Style m_style;
90 ContainerBox* m_parent { nullptr };
91 std::unique_ptr<Box> m_nextSibling;
92 OptionSet<TypeFlags> m_typeFlags;
93};
94
95} // namespace Display
96} // namespace WebCore

101SPECIALIZE_TYPE_TRAITS_END()
