49bool BorderEdge::obscuresBackground() const
50{
51 if (!m_isPresent || m_isTransparent || !m_color.isOpaque() || m_style == BorderStyle::Hidden)
52 return false;
53
54 if (m_style == BorderStyle::Dotted || m_style == BorderStyle::Dashed || m_style == BorderStyle::Double)
55 return false;
56
57 return true;
58}
59
60bool BorderEdge::obscuresBackgroundEdge(float scale) const
61{
62 if (!m_isPresent || m_isTransparent || (m_width * scale) < floorToDevicePixel(2, scale) || !m_color.isOpaque() || m_style == BorderStyle::Hidden)
63 return false;
64
65 if (m_style == BorderStyle::Dotted || m_style == BorderStyle::Dashed)
66 return false;
67
68 if (m_style == BorderStyle::Double)
69 return m_width >= scale * floorToDevicePixel(5, scale); // The outer band needs to be >= 2px wide at unit scale.
70
71 return true;
72}

74RectEdges<BorderEdge> calculateBorderEdges(const RenderStyle& style, float pixelSnappingFactor, bool includeLogicalLeftEdge, bool includeLogicalRightEdge)
75{
76 bool horizontal = style.isHorizontalWritingMode();
77
78 auto constructBorderEdge = [&](float width, CSSPropertyID colorProperty, BorderStyle borderStyle, bool isTransparent, bool isPresent) ->BorderEdge {
79 float innerWidth = 0.f;
80 float outerWidth = 0.f;
81
82 if (borderStyle == BorderStyle::Double) {
83 if (borderStyle == BorderStyle::Double && width < floorToDevicePixel(3, pixelSnappingFactor))
84 borderStyle = BorderStyle::Solid;
85 else {
86 innerWidth = ceilToDevicePixel(width * 2 / 3, pixelSnappingFactor);
87 outerWidth = floorToDevicePixel(width / 3, pixelSnappingFactor);
88 }
89 }
90
91 float snappedWidth = floorToDevicePixel(width, pixelSnappingFactor);
92 return { snappedWidth, innerWidth, outerWidth, style.visitedDependentColorWithColorFilter(colorProperty), borderStyle, isTransparent, isPresent };
93 };
94
95 return {
96 constructBorderEdge(style.borderTopWidth(), CSSPropertyBorderTopColor, style.borderTopStyle(), style.borderTopIsTransparent(), !horizontal || includeLogicalRightEdge),
97 constructBorderEdge(style.borderRightWidth(), CSSPropertyBorderRightColor, style.borderRightStyle(), style.borderRightIsTransparent(), !horizontal || includeLogicalRightEdge),
98 constructBorderEdge(style.borderBottomWidth(), CSSPropertyBorderBottomColor, style.borderBottomStyle(), style.borderBottomIsTransparent(), horizontal || includeLogicalRightEdge),
99 constructBorderEdge(style.borderLeftWidth(), CSSPropertyBorderLeftColor, style.borderLeftStyle(), style.borderLeftIsTransparent(), !horizontal || includeLogicalLeftEdge)
100 };
101}

103std::pair<BoxSide, BoxSide> adjacentSidesForSide(BoxSide side)
104{
105 switch (side) {
106 case BoxSide::Top: return { BoxSide::Left, BoxSide::Right };
107 case BoxSide::Bottom: return { BoxSide::Left, BoxSide::Right };
108 case BoxSide::Left: return { BoxSide::Top, BoxSide::Bottom };
109 case BoxSide::Right: return { BoxSide::Top, BoxSide::Bottom };
110 }
111 return { BoxSide::Left, BoxSide::Right };
112}

116bool BoxDecorationData::hasBorder() const
117{
118 // FIXME: There's a tricky interaction between borders and border-image here: webkit.org/b/217900.
119 if (hasBorderImage())
120 return true;
121
122 for (auto side : allBoxSides) {
123 auto& edge = m_borderEdges[side];
124 if (edge.style() != BorderStyle::None && edge.width())
125 return true;
126 }
127 return false;
128}
129
130bool BoxDecorationData::borderObscuresBackground() const
131{
132 if (!hasBorder())
133 return false;
134
135 // Bail if we have any border-image for now. We could look at the image alpha to improve this.
136 // FIXME: Check border-image.
137
138 for (auto side : allBoxSides) {
139 if (!m_borderEdges.at(side).obscuresBackground())
140 return false;
141 }
142
143 return true;
144}
145
146bool BoxDecorationData::borderObscuresBackgroundEdge(const FloatSize& contextScale) const
147{
148 for (auto side : allBoxSides) {
149 auto& currEdge = m_borderEdges.at(side);
150 // FIXME: for vertical text
151 float axisScale = (side == BoxSide::Top || side == BoxSide::Bottom) ? contextScale.height() : contextScale.width();
152 if (!currEdge.obscuresBackgroundEdge(axisScale))
153 return false;
154 }
155
156 return true;
157}
