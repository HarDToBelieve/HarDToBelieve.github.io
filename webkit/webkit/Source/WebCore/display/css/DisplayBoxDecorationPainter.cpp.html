121// BorderStyle::Outset darkens the bottom and right (and maybe lightens the top and left)
122// BorderStyle::Inset darkens the top and left (and maybe lightens the bottom and right)
123bool BorderPainter::borderStyleHasUnmatchedColorsAtCorner(BorderStyle style, BoxSide side, BoxSide adjacentSide)
124{
125 // These styles match at the top/left and bottom/right.
126 if (style == BorderStyle::Inset || style == BorderStyle::Groove || style == BorderStyle::Ridge || style == BorderStyle::Outset) {
127 BoxSideSet topRightSides = { BoxSideFlag::Top, BoxSideFlag::Right };
128 BoxSideSet bottomLeftSides = { BoxSideFlag::Bottom, BoxSideFlag::Left };
129
130 BoxSideSet usedSides { edgeFlagForSide(side), edgeFlagForSide(adjacentSide) };
131 return usedSides == topRightSides || usedSides == bottomLeftSides;
132 }
133 return false;
134}

137{
138 auto& edge = m_edges.at(side);

167void BorderPainter::calculateBorderStyleColor(BorderStyle style, BoxSide side, Color& color)
168{
169 ASSERT(style == BorderStyle::Inset || style == BorderStyle::Outset);
170
171 // This values were derived empirically.
172 constexpr float baseDarkColorLuminance { 0.014443844f }; // Luminance of SRGBA<uint8_t> { 32, 32, 32 }
173 constexpr float baseLightColorLuminance { 0.83077f }; // Luminance of SRGBA<uint8_t> { 235, 235, 235 }
174
175 enum Operation { Darken, Lighten };
176
177 Operation operation = (side == BoxSide::Top || side == BoxSide::Left) == (style == BorderStyle::Inset) ? Darken : Lighten;
178
179 // Here we will darken the border decoration color when needed. This will yield a similar behavior as in FF.
180 if (operation == Darken) {
181 if (color.luminance() > baseDarkColorLuminance)
182 color = color.darkened();
183 } else {
184 if (color.luminance() < baseLightColorLuminance)
185 color = color.lightened();
186 }
187}

189// This assumes that we draw in order: top, bottom, left, right.
190bool BorderPainter::willBeOverdrawn(BoxSide side, BoxSide adjacentSide) const
191{
192 switch (side) {
193 case BoxSide::Top:
194 case BoxSide::Bottom: {
195 auto& edge = m_edges.at(side);
196 auto& adjacentEdge = m_edges.at(adjacentSide);
197
198 if (adjacentEdge.presentButInvisible())
199 return false;
200
201 if (!edgesShareColor(edge, adjacentEdge) && !adjacentEdge.color().isOpaque())
202 return false;
203
204 if (!borderStyleFillsBorderArea(adjacentEdge.style()))
205 return false;
206
207 return true;
208 }
209 case BoxSide::Left:
210 case BoxSide::Right:
211 // These draw last, so are never overdrawn.
212 return false;
213 }
214 return false;
215}

217bool BorderPainter::borderStylesRequireMitre(BoxSide side, BoxSide adjacentSide, BorderStyle style, BorderStyle adjacentStyle)
218{
219 if (style == BorderStyle::Double || adjacentStyle == BorderStyle::Double || adjacentStyle == BorderStyle::Groove || adjacentStyle == BorderStyle::Ridge)
220 return true;
221
222 if (borderStyleIsDottedOrDashed(style) != borderStyleIsDottedOrDashed(adjacentStyle))
223 return true;
224
225 if (style != adjacentStyle)
226 return true;
227
228 return borderStyleHasUnmatchedColorsAtCorner(style, side, adjacentSide);

231FloatRoundedRect BorderPainter::calculateAdjustedInnerBorder(const FloatRoundedRect& innerBorder, BoxSide side)
232{
233 // Expand the inner border as necessary to make it a rounded rect (i.e. radii contained within each edge).
234 // This function relies on the fact we only get radii not contained within each edge if one of the radii
235 // for an edge is zero, so we can shift the arc towards the zero radius corner.
236 auto newRadii = innerBorder.radii();
237 auto newRect = innerBorder.rect();
238
239 float overshoot;
240 float maxRadii;
241
242 switch (side) {
243 case BoxSide::Top:
244 overshoot = newRadii.topLeft().width() + newRadii.topRight().width() - newRect.width();
245 if (overshoot > 0) {
246 ASSERT(!(newRadii.topLeft().width() && newRadii.topRight().width()));
247 newRect.setWidth(newRect.width() + overshoot);
248 if (!newRadii.topLeft().width())
249 newRect.move(-overshoot, 0);
250 }
251 newRadii.setBottomLeft({ });
252 newRadii.setBottomRight({ });
253 maxRadii = std::max(newRadii.topLeft().height(), newRadii.topRight().height());
254 if (maxRadii > newRect.height())
255 newRect.setHeight(maxRadii);
256 break;
257
258 case BoxSide::Bottom:
259 overshoot = newRadii.bottomLeft().width() + newRadii.bottomRight().width() - newRect.width();
260 if (overshoot > 0) {
261 ASSERT(!(newRadii.bottomLeft().width() && newRadii.bottomRight().width()));
262 newRect.setWidth(newRect.width() + overshoot);
263 if (!newRadii.bottomLeft().width())
264 newRect.move(-overshoot, 0);
265 }
266 newRadii.setTopLeft({ });
267 newRadii.setTopRight({ });
268 maxRadii = std::max(newRadii.bottomLeft().height(), newRadii.bottomRight().height());
269 if (maxRadii > newRect.height()) {
270 newRect.move(0, newRect.height() - maxRadii);
271 newRect.setHeight(maxRadii);
272 }
273 break;
274
275 case BoxSide::Left:
276 overshoot = newRadii.topLeft().height() + newRadii.bottomLeft().height() - newRect.height();
277 if (overshoot > 0) {
278 ASSERT(!(newRadii.topLeft().height() && newRadii.bottomLeft().height()));
279 newRect.setHeight(newRect.height() + overshoot);
280 if (!newRadii.topLeft().height())
281 newRect.move(0, -overshoot);
282 }
283 newRadii.setTopRight({ });
284 newRadii.setBottomRight({ });
285 maxRadii = std::max(newRadii.topLeft().width(), newRadii.bottomLeft().width());
286 if (maxRadii > newRect.width())
287 newRect.setWidth(maxRadii);
288 break;
289
290 case BoxSide::Right:
291 overshoot = newRadii.topRight().height() + newRadii.bottomRight().height() - newRect.height();
292 if (overshoot > 0) {
293 ASSERT(!(newRadii.topRight().height() && newRadii.bottomRight().height()));
294 newRect.setHeight(newRect.height() + overshoot);
295 if (!newRadii.topRight().height())
296 newRect.move(0, -overshoot);
297 }
298 newRadii.setTopLeft({ });
299 newRadii.setBottomLeft({ });
300 maxRadii = std::max(newRadii.topRight().width(), newRadii.bottomRight().width());
301 if (maxRadii > newRect.width()) {
302 newRect.move(newRect.width() - maxRadii, 0);
303 newRect.setWidth(maxRadii);
304 }
305 break;
306 }
307
308 return FloatRoundedRect { newRect, newRadii };
309}

311bool BorderPainter::joinRequiresMitre(BoxSide side, BoxSide adjacentSide, bool allowOverdraw) const
312{
313 auto& edge = m_edges.at(side);
314 auto& adjacentEdge = m_edges.at(adjacentSide);
315
316 if ((edge.isTransparent() && adjacentEdge.isTransparent()) || !adjacentEdge.isPresent())
317 return false;
318
319 if (allowOverdraw && willBeOverdrawn(side, adjacentSide))
320 return false;
321
322 if (!edgesShareColor(edge, adjacentEdge))
323 return true;
324
325 if (borderStylesRequireMitre(side, adjacentSide, edge.style(), adjacentEdge.style()))
326 return true;
327
328 return false;
329}

331void BorderPainter::drawBoxSideFromPath(PaintingContext& paintingContext, const FloatRect& borderRect, const Path& borderPath, float thickness, float drawThickness, BoxSide side, Color color, BorderStyle borderStyle) const
332{
333 if (thickness <= 0)
334 return;
335
336 if (borderStyle == BorderStyle::Double && thickness < 3)
337 borderStyle = BorderStyle::Solid;
338
339 switch (borderStyle) {
340 case BorderStyle::None:
341 case BorderStyle::Hidden:
342 return;
343 case BorderStyle::Dotted:
344 case BorderStyle::Dashed: {
345 paintingContext.context.setStrokeColor(color);
346
347 // The stroke is doubled here because the provided path is the
348 // outside edge of the border so half the stroke is clipped off.
349 // The extra multiplier is so that the clipping mask can antialias
350 // the edges to prevent jaggies.
351 paintingContext.context.setStrokeThickness(drawThickness * 2 * 1.1f);
352 paintingContext.context.setStrokeStyle(borderStyle == BorderStyle::Dashed ? DashedStroke : DottedStroke);
353
354 // If the number of dashes that fit in the path is odd and non-integral then we
355 // will have an awkwardly-sized dash at the end of the path. To try to avoid that
356 // here, we simply make the whitespace dashes ever so slightly bigger.
357 // FIXME: This could be even better if we tried to manipulate the dash offset
358 // and possibly the gapLength to get the corners dash-symmetrical.
359 float dashLength = thickness * ((borderStyle == BorderStyle::Dashed) ? 3.0f : 1.0f);
360 float gapLength = dashLength;
361 float numberOfDashes = borderPath.length() / dashLength;
362 // Don't try to show dashes if we have less than 2 dashes + 2 gaps.
363 // FIXME: should do this test per side.
364 if (numberOfDashes >= 4) {
365 bool evenNumberOfFullDashes = !((int)numberOfDashes % 2);
366 bool integralNumberOfDashes = !(numberOfDashes - (int)numberOfDashes);
367 if (!evenNumberOfFullDashes && !integralNumberOfDashes) {
368 float numberOfGaps = numberOfDashes / 2;
369 gapLength += (dashLength / numberOfGaps);
370 }
371
372 DashArray lineDash;
373 lineDash.append(dashLength);
374 lineDash.append(gapLength);
375 paintingContext.context.setLineDash(lineDash, dashLength);
376 }
377
378 // FIXME: stroking the border path causes issues with tight corners:
379 // https://bugs.webkit.org/show_bug.cgi?id=58711
380 // Also, to get the best appearance we should stroke a path between the two borders.
381 paintingContext.context.strokePath(borderPath);
382 return;
383 }
384 case BorderStyle::Double: {
385 // Get the inner border rects for both the outer border line and the inner border line
386 // FIXME: Rect edges.
387 float outerBorderTopWidth = m_edges.top().outerWidth();
388 float innerBorderTopWidth = m_edges.top().innerWidth();
389
390 float outerBorderRightWidth = m_edges.right().outerWidth();
391 float innerBorderRightWidth = m_edges.right().innerWidth();
392
393 float outerBorderBottomWidth = m_edges.bottom().outerWidth();
394 float innerBorderBottomWidth = m_edges.bottom().innerWidth();
395
396 float outerBorderLeftWidth = m_edges.left().outerWidth();
397 float innerBorderLeftWidth = m_edges.left().innerWidth();
398
399 // Draw inner border line
400 {
401 GraphicsContextStateSaver stateSaver(paintingContext.context);
402 auto innerClip = roundedInsetBorderForRect(borderRect, m_borderRect.radii(), { innerBorderTopWidth, innerBorderRightWidth, innerBorderBottomWidth, innerBorderLeftWidth }, m_includeLeftEdge, m_includeRightEdge);
403
404 paintingContext.context.clipRoundedRect(innerClip);
405 drawBoxSideFromPath(paintingContext, borderRect, borderPath, thickness, drawThickness, side, color, BorderStyle::Solid);
406 }
407
408 // Draw outer border line
409 {
410 GraphicsContextStateSaver stateSaver(paintingContext.context);
411 auto outerRect = borderRect;
412 if (m_bleedAvoidance == BackgroundBleedAvoidance::UseTransparencyLayer) {
413 outerRect.inflate(1);
414 ++outerBorderTopWidth;
415 ++outerBorderBottomWidth;
416 ++outerBorderLeftWidth;
417 ++outerBorderRightWidth;
418 }
419
420 auto outerClip = roundedInsetBorderForRect(outerRect, m_borderRect.radii(), { outerBorderTopWidth, outerBorderRightWidth, outerBorderBottomWidth, outerBorderLeftWidth }, m_includeLeftEdge, m_includeRightEdge);
421 paintingContext.context.clipOutRoundedRect(outerClip);
422 drawBoxSideFromPath(paintingContext, borderRect, borderPath, thickness, drawThickness, side, color, BorderStyle::Solid);
423 }
424 return;
425 }
426 case BorderStyle::Ridge:
427 case BorderStyle::Groove:
428 {
429 BorderStyle s1;
430 BorderStyle s2;
431 if (borderStyle == BorderStyle::Groove) {
432 s1 = BorderStyle::Inset;
433 s2 = BorderStyle::Outset;
434 } else {
435 s1 = BorderStyle::Outset;
436 s2 = BorderStyle::Inset;
437 }
438
439 // Paint full border
440 drawBoxSideFromPath(paintingContext, borderRect, borderPath, thickness, drawThickness, side, color, s1);
441
442 // Paint inner only
443 GraphicsContextStateSaver stateSaver(paintingContext.context);
444 // FIXME: Should have pixel snapped these at display tree generation time.
445 // FIXME: RectEdges
446 float topWidth { m_edges.top().widthForPainting() / 2 };
447 float bottomWidth { m_edges.bottom().widthForPainting() / 2 };
448 float leftWidth { m_edges.left().widthForPainting() / 2 };
449 float rightWidth { m_edges.right().widthForPainting() / 2 };
450
451 auto clipRect = roundedInsetBorderForRect(borderRect, m_borderRect.radii(), { topWidth, rightWidth, bottomWidth, leftWidth }, m_includeLeftEdge, m_includeRightEdge);
452
453 paintingContext.context.clipRoundedRect(clipRect);
454 drawBoxSideFromPath(paintingContext, borderRect, borderPath, thickness, drawThickness, side, color, s2);
455 return;
456 }
457 case BorderStyle::Inset:
458 case BorderStyle::Outset:
459 calculateBorderStyleColor(borderStyle, side, color);
460 break;
461 default:
462 break;
463 }
464
465 paintingContext.context.setStrokeStyle(NoStroke);
466 paintingContext.context.setFillColor(color);
467 paintingContext.context.drawRect(borderRect);
468}

470void BorderPainter::clipBorderSidePolygon(PaintingContext& paintingContext, const FloatRoundedRect& outerBorder, const FloatRoundedRect& innerBorder, BoxSide side, bool firstEdgeMatches, bool secondEdgeMatches) const
471{
472 const FloatRect& outerRect = outerBorder.rect();
473 const FloatRect& innerRect = innerBorder.rect();
474
475 // For each side, create a quad that encompasses all parts of that side that may draw,
476 // including areas inside the innerBorder.
477 //
478 // 0----------------3
479 // 0 \ / 0
480 // |\ 1----------- 2 /|
481 // | 1 1 |
482 // | | | |
483 // | | | |
484 // | 2 2 |
485 // |/ 1------------2 \|
486 // 3 / \ 3
487 // 0----------------3
488 //
489 Vector<FloatPoint> quad;
490 quad.reserveInitialCapacity(4);
491 switch (side) {
492 case BoxSide::Top:
493 quad.uncheckedAppend(outerRect.minXMinYCorner());
494 quad.uncheckedAppend(innerRect.minXMinYCorner());
495 quad.uncheckedAppend(innerRect.maxXMinYCorner());
496 quad.uncheckedAppend(outerRect.maxXMinYCorner());
497
498 if (!innerBorder.radii().topLeft().isZero())
499 findIntersection(outerRect.minXMinYCorner(), innerRect.minXMinYCorner(), innerRect.minXMaxYCorner(), innerRect.maxXMinYCorner(), quad[1]);
500
501 if (!innerBorder.radii().topRight().isZero())
502 findIntersection(outerRect.maxXMinYCorner(), innerRect.maxXMinYCorner(), innerRect.minXMinYCorner(), innerRect.maxXMaxYCorner(), quad[2]);
503 break;
504
505 case BoxSide::Left:
506 quad.uncheckedAppend(outerRect.minXMinYCorner());
507 quad.uncheckedAppend(innerRect.minXMinYCorner());
508 quad.uncheckedAppend(innerRect.minXMaxYCorner());
509 quad.uncheckedAppend(outerRect.minXMaxYCorner());
510
511 if (!innerBorder.radii().topLeft().isZero())
512 findIntersection(outerRect.minXMinYCorner(), innerRect.minXMinYCorner(), innerRect.minXMaxYCorner(), innerRect.maxXMinYCorner(), quad[1]);
513
514 if (!innerBorder.radii().bottomLeft().isZero())
515 findIntersection(outerRect.minXMaxYCorner(), innerRect.minXMaxYCorner(), innerRect.minXMinYCorner(), innerRect.maxXMaxYCorner(), quad[2]);
516 break;
517
518 case BoxSide::Bottom:
519 quad.uncheckedAppend(outerRect.minXMaxYCorner());
520 quad.uncheckedAppend(innerRect.minXMaxYCorner());
521 quad.uncheckedAppend(innerRect.maxXMaxYCorner());
522 quad.uncheckedAppend(outerRect.maxXMaxYCorner());
523
524 if (!innerBorder.radii().bottomLeft().isZero())
525 findIntersection(outerRect.minXMaxYCorner(), innerRect.minXMaxYCorner(), innerRect.minXMinYCorner(), innerRect.maxXMaxYCorner(), quad[1]);
526
527 if (!innerBorder.radii().bottomRight().isZero())
528 findIntersection(outerRect.maxXMaxYCorner(), innerRect.maxXMaxYCorner(), innerRect.maxXMinYCorner(), innerRect.minXMaxYCorner(), quad[2]);
529 break;
530
531 case BoxSide::Right:
532 quad.uncheckedAppend(outerRect.maxXMinYCorner());
533 quad.uncheckedAppend(innerRect.maxXMinYCorner());
534 quad.uncheckedAppend(innerRect.maxXMaxYCorner());
535 quad.uncheckedAppend(outerRect.maxXMaxYCorner());
536
537 if (!innerBorder.radii().topRight().isZero())
538 findIntersection(outerRect.maxXMinYCorner(), innerRect.maxXMinYCorner(), innerRect.minXMinYCorner(), innerRect.maxXMaxYCorner(), quad[1]);
539
540 if (!innerBorder.radii().bottomRight().isZero())
541 findIntersection(outerRect.maxXMaxYCorner(), innerRect.maxXMaxYCorner(), innerRect.maxXMinYCorner(), innerRect.minXMaxYCorner(), quad[2]);
542 break;
543 }
544
545 // If the border matches both of its adjacent sides, don't anti-alias the clip, and
546 // if neither side matches, anti-alias the clip.
547 if (firstEdgeMatches == secondEdgeMatches) {
548 bool wasAntialiased = paintingContext.context.shouldAntialias();
549 paintingContext.context.setShouldAntialias(!firstEdgeMatches);
550 paintingContext.context.clipPath(Path::polygonPathFromPoints(quad), WindRule::NonZero);
551 paintingContext.context.setShouldAntialias(wasAntialiased);
552 return;
553 }
554
555 // Square off the end which shouldn't be affected by antialiasing, and clip.
556 Vector<FloatPoint> firstQuad = {
557 quad[0],
558 quad[1],
559 quad[2],
560 side == BoxSide::Top || side == BoxSide::Bottom ? FloatPoint(quad[3].x(), quad[2].y()) : FloatPoint(quad[2].x(), quad[3].y()),
561 quad[3]
562 };
563 bool wasAntialiased = paintingContext.context.shouldAntialias();
564 paintingContext.context.setShouldAntialias(!firstEdgeMatches);
565 paintingContext.context.clipPath(Path::polygonPathFromPoints(firstQuad), WindRule::NonZero);
566
567 Vector<FloatPoint> secondQuad = {
568 quad[0],
569 side == BoxSide::Top || side == BoxSide::Bottom ? FloatPoint(quad[0].x(), quad[1].y()) : FloatPoint(quad[1].x(), quad[0].y()),
570 quad[1],
571 quad[2],
572 quad[3]
573 };
574 // Antialiasing affects the second side.
575 paintingContext.context.setShouldAntialias(!secondEdgeMatches);
576 paintingContext.context.clipPath(Path::polygonPathFromPoints(secondQuad), WindRule::NonZero);
577
578 paintingContext.context.setShouldAntialias(wasAntialiased);
579}

581void BorderPainter::drawLineForBoxSide(PaintingContext& paintingContext, const FloatRect& rect, BoxSide side, Color color, BorderStyle borderStyle, float adjacentWidth1, float adjacentWidth2, bool antialias) const
582{
583 auto drawBorderRect = [&](const FloatRect& rect) {
584 if (rect.isEmpty())
585 return;
586 paintingContext.context.drawRect(rect);
587 };
588
589 auto drawLineFor = [&](const FloatRect& rect, BoxSide side, BorderStyle borderStyle, const FloatSize& adjacent) {
590 if (rect.isEmpty())
591 return;
592 drawLineForBoxSide(paintingContext, rect, side, color, borderStyle, adjacent.width(), adjacent.height(), antialias);
593 };
594
595 auto& edge = m_edges.at(side);
596
597 const float deviceScaleFactor = paintingContext.deviceScaleFactor;
598 float x1 = rect.x();
599 float x2 = rect.maxX();
600 float y1 = rect.y();
601 float y2 = rect.maxY();
602 float thickness;
603 float length;
604 if (side == BoxSide::Top || side == BoxSide::Bottom) {
605 thickness = y2 - y1;
606 length = x2 - x1;
607 } else {
608 thickness = x2 - x1;
609 length = y2 - y1;
610 }
611 // FIXME: We really would like this check to be an ASSERT as we don't want to draw empty borders. However
612 // nothing guarantees that the following recursive calls to drawLineForBoxSide will have non-null dimensions.
613 if (!thickness || !length)
614 return;
615
616 switch (borderStyle) {
617 case BorderStyle::None:
618 case BorderStyle::Hidden:
619 return;
620 case BorderStyle::Dotted:
621 case BorderStyle::Dashed: {
622 bool wasAntialiased = paintingContext.context.shouldAntialias();
623 StrokeStyle oldStrokeStyle = paintingContext.context.strokeStyle();
624 paintingContext.context.setShouldAntialias(antialias);
625 paintingContext.context.setStrokeColor(color);
626 paintingContext.context.setStrokeThickness(thickness);
627 paintingContext.context.setStrokeStyle(borderStyle == BorderStyle::Dashed ? DashedStroke : DottedStroke);
628 paintingContext.context.drawLine({ x1, y1 }, { x2, y2 });
629 paintingContext.context.setShouldAntialias(wasAntialiased);
630 paintingContext.context.setStrokeStyle(oldStrokeStyle);
631 break;
632 }
633 case BorderStyle::Double: {
634 float thirdOfThickness = ceilToDevicePixel(thickness / 3, deviceScaleFactor);
635 if (!adjacentWidth1 && !adjacentWidth2) {
636 StrokeStyle oldStrokeStyle = paintingContext.context.strokeStyle();
637 paintingContext.context.setStrokeStyle(NoStroke);
638 paintingContext.context.setFillColor(color);
639
640 bool wasAntialiased = paintingContext.context.shouldAntialias();
641 paintingContext.context.setShouldAntialias(antialias);
642
643 switch (side) {
644 case BoxSide::Top:
645 drawBorderRect({ x1, y1, length, edge.outerWidth() });
646 drawBorderRect({ x1, y2 - edge.innerWidth(), length, edge.innerWidth() });
647 break;
648 case BoxSide::Bottom:
649 drawBorderRect({ x1, y2 - edge.outerWidth(), length, edge.outerWidth() });
650 drawBorderRect({ x1, y1, length, edge.innerWidth() });
651 break;
652 case BoxSide::Left:
653 drawBorderRect({ x1, y1, edge.outerWidth(), length });
654 drawBorderRect({ x2 - edge.innerWidth(), y1, edge.innerWidth(), length });
655 break;
656 case BoxSide::Right:
657 drawBorderRect({ x2 - edge.outerWidth(), y1, edge.outerWidth(), length });
658 drawBorderRect({ x1, y1, edge.innerWidth(), length });
659 break;
660 }
661
662 paintingContext.context.setShouldAntialias(wasAntialiased);
663 paintingContext.context.setStrokeStyle(oldStrokeStyle);
664 } else {
665 float adjacent1BigThird = ceilToDevicePixel(adjacentWidth1 / 3, deviceScaleFactor);
666 float adjacent2BigThird = ceilToDevicePixel(adjacentWidth2 / 3, deviceScaleFactor);
667
668 float offset1 = floorToDevicePixel(fabs(adjacentWidth1) * 2 / 3, deviceScaleFactor);
669 float offset2 = floorToDevicePixel(fabs(adjacentWidth2) * 2 / 3, deviceScaleFactor);
670
671 float mitreOffset1 = adjacentWidth1 < 0 ? offset1 : 0;
672 float mitreOffset2 = adjacentWidth1 > 0 ? offset1 : 0;
673 float mitreOffset3 = adjacentWidth2 < 0 ? offset2 : 0;
674 float mitreOffset4 = adjacentWidth2 > 0 ? offset2 : 0;
675
676 FloatRect paintBorderRect;
677 switch (side) {
678 case BoxSide::Top:
679 paintBorderRect = snapRectToDevicePixels(LayoutRect(x1 + mitreOffset1, y1, (x2 - mitreOffset3) - (x1 + mitreOffset1), thirdOfThickness), deviceScaleFactor);
680 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
681
682 paintBorderRect = snapRectToDevicePixels(LayoutRect(x1 + mitreOffset2, y2 - thirdOfThickness, (x2 - mitreOffset4) - (x1 + mitreOffset2), thirdOfThickness), deviceScaleFactor);
683 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
684 break;
685 case BoxSide::Left:
686 paintBorderRect = snapRectToDevicePixels(LayoutRect(x1, y1 + mitreOffset1, thirdOfThickness, (y2 - mitreOffset3) - (y1 + mitreOffset1)), deviceScaleFactor);
687 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
688
689 paintBorderRect = snapRectToDevicePixels(LayoutRect(x2 - thirdOfThickness, y1 + mitreOffset2, thirdOfThickness, (y2 - mitreOffset4) - (y1 + mitreOffset2)), deviceScaleFactor);
690 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
691 break;
692 case BoxSide::Bottom:
693 paintBorderRect = snapRectToDevicePixels(LayoutRect(x1 + mitreOffset2, y1, (x2 - mitreOffset4) - (x1 + mitreOffset2), thirdOfThickness), deviceScaleFactor);
694 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
695
696 paintBorderRect = snapRectToDevicePixels(LayoutRect(x1 + mitreOffset1, y2 - thirdOfThickness, (x2 - mitreOffset3) - (x1 + mitreOffset1), thirdOfThickness), deviceScaleFactor);
697 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
698 break;
699 case BoxSide::Right:
700 paintBorderRect = snapRectToDevicePixels(LayoutRect(x1, y1 + mitreOffset2, thirdOfThickness, (y2 - mitreOffset4) - (y1 + mitreOffset2)), deviceScaleFactor);
701 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
702
703 paintBorderRect = snapRectToDevicePixels(LayoutRect(x2 - thirdOfThickness, y1 + mitreOffset1, thirdOfThickness, (y2 - mitreOffset3) - (y1 + mitreOffset1)), deviceScaleFactor);
704 drawLineFor(paintBorderRect, side, BorderStyle::Solid, FloatSize(adjacent1BigThird, adjacent2BigThird));
705 break;
706 }
707 }
708 break;
709 }
710 case BorderStyle::Ridge:
711 case BorderStyle::Groove: {
712 BorderStyle s1;
713 BorderStyle s2;
714 if (borderStyle == BorderStyle::Groove) {
715 s1 = BorderStyle::Inset;
716 s2 = BorderStyle::Outset;
717 } else {
718 s1 = BorderStyle::Outset;
719 s2 = BorderStyle::Inset;
720 }
721
722 float adjacent1BigHalf = ceilToDevicePixel(adjacentWidth1 / 2, deviceScaleFactor);
723 float adjacent2BigHalf = ceilToDevicePixel(adjacentWidth2 / 2, deviceScaleFactor);
724
725 float adjacent1SmallHalf = floorToDevicePixel(adjacentWidth1 / 2, deviceScaleFactor);
726 float adjacent2SmallHalf = floorToDevicePixel(adjacentWidth2 / 2, deviceScaleFactor);
727
728 float offset1 = 0;
729 float offset2 = 0;
730 float offset3 = 0;
731 float offset4 = 0;
732
733 if (((side == BoxSide::Top || side == BoxSide::Left) && adjacentWidth1 < 0) || ((side == BoxSide::Bottom || side == BoxSide::Right) && adjacentWidth1 > 0))
734 offset1 = floorToDevicePixel(adjacentWidth1 / 2, deviceScaleFactor);
735
736 if (((side == BoxSide::Top || side == BoxSide::Left) && adjacentWidth2 < 0) || ((side == BoxSide::Bottom || side == BoxSide::Right) && adjacentWidth2 > 0))
737 offset2 = ceilToDevicePixel(adjacentWidth2 / 2, deviceScaleFactor);
738
739 if (((side == BoxSide::Top || side == BoxSide::Left) && adjacentWidth1 > 0) || ((side == BoxSide::Bottom || side == BoxSide::Right) && adjacentWidth1 < 0))
740 offset3 = floorToDevicePixel(fabs(adjacentWidth1) / 2, deviceScaleFactor);
741
742 if (((side == BoxSide::Top || side == BoxSide::Left) && adjacentWidth2 > 0) || ((side == BoxSide::Bottom || side == BoxSide::Right) && adjacentWidth2 < 0))
743 offset4 = ceilToDevicePixel(adjacentWidth2 / 2, deviceScaleFactor);
744
745 float adjustedX = ceilToDevicePixel((x1 + x2) / 2, deviceScaleFactor);
746 float adjustedY = ceilToDevicePixel((y1 + y2) / 2, deviceScaleFactor);
747 // Quads can't use the default snapping rect functions.
748 x1 = roundToDevicePixel(x1, deviceScaleFactor);
749 x2 = roundToDevicePixel(x2, deviceScaleFactor);
750 y1 = roundToDevicePixel(y1, deviceScaleFactor);
751 y2 = roundToDevicePixel(y2, deviceScaleFactor);
752
753 switch (side) {
754 case BoxSide::Top:
755 drawLineFor(FloatRect(FloatPoint(x1 + offset1, y1), FloatPoint(x2 - offset2, adjustedY)), side, s1, FloatSize(adjacent1BigHalf, adjacent2BigHalf));
756 drawLineFor(FloatRect(FloatPoint(x1 + offset3, adjustedY), FloatPoint(x2 - offset4, y2)), side, s2, FloatSize(adjacent1SmallHalf, adjacent2SmallHalf));
757 break;
758 case BoxSide::Left:
759 drawLineFor(FloatRect(FloatPoint(x1, y1 + offset1), FloatPoint(adjustedX, y2 - offset2)), side, s1, FloatSize(adjacent1BigHalf, adjacent2BigHalf));
760 drawLineFor(FloatRect(FloatPoint(adjustedX, y1 + offset3), FloatPoint(x2, y2 - offset4)), side, s2, FloatSize(adjacent1SmallHalf, adjacent2SmallHalf));
761 break;
762 case BoxSide::Bottom:
763 drawLineFor(FloatRect(FloatPoint(x1 + offset1, y1), FloatPoint(x2 - offset2, adjustedY)), side, s2, FloatSize(adjacent1BigHalf, adjacent2BigHalf));
764 drawLineFor(FloatRect(FloatPoint(x1 + offset3, adjustedY), FloatPoint(x2 - offset4, y2)), side, s1, FloatSize(adjacent1SmallHalf, adjacent2SmallHalf));
765 break;
766 case BoxSide::Right:
767 drawLineFor(FloatRect(FloatPoint(x1, y1 + offset1), FloatPoint(adjustedX, y2 - offset2)), side, s2, FloatSize(adjacent1BigHalf, adjacent2BigHalf));
768 drawLineFor(FloatRect(FloatPoint(adjustedX, y1 + offset3), FloatPoint(x2, y2 - offset4)), side, s1, FloatSize(adjacent1SmallHalf, adjacent2SmallHalf));
769 break;
770 }
771 break;
772 }
773 case BorderStyle::Inset:
774 case BorderStyle::Outset:
775 calculateBorderStyleColor(borderStyle, side, color);
776 FALLTHROUGH;
777 case BorderStyle::Solid: {
778 StrokeStyle oldStrokeStyle = paintingContext.context.strokeStyle();
779 ASSERT(x2 >= x1);
780 ASSERT(y2 >= y1);
781 if (!adjacentWidth1 && !adjacentWidth2) {
782 paintingContext.context.setStrokeStyle(NoStroke);
783 paintingContext.context.setFillColor(color);
784 bool wasAntialiased = paintingContext.context.shouldAntialias();
785 paintingContext.context.setShouldAntialias(antialias);
786 drawBorderRect(snapRectToDevicePixels(LayoutRect(x1, y1, x2 - x1, y2 - y1), deviceScaleFactor));
787 paintingContext.context.setShouldAntialias(wasAntialiased);
788 paintingContext.context.setStrokeStyle(oldStrokeStyle);
789 return;
790 }
791
792 // FIXME: These roundings should be replaced by ASSERT(device pixel positioned) when all the callers have transitioned to device pixels.
793 x1 = roundToDevicePixel(x1, deviceScaleFactor);
794 y1 = roundToDevicePixel(y1, deviceScaleFactor);
795 x2 = roundToDevicePixel(x2, deviceScaleFactor);
796 y2 = roundToDevicePixel(y2, deviceScaleFactor);
797
798 Vector<FloatPoint> quad;
799 quad.reserveInitialCapacity(4);
800 switch (side) {
801 case BoxSide::Top:
802 quad.uncheckedAppend({ x1 + std::max<float>(-adjacentWidth1, 0), y1 });
803 quad.uncheckedAppend({ x1 + std::max<float>( adjacentWidth1, 0), y2 });
804 quad.uncheckedAppend({ x2 - std::max<float>( adjacentWidth2, 0), y2 });
805 quad.uncheckedAppend({ x2 - std::max<float>(-adjacentWidth2, 0), y1 });
806 break;
807 case BoxSide::Bottom:
808 quad.uncheckedAppend({ x1 + std::max<float>( adjacentWidth1, 0), y1 });
809 quad.uncheckedAppend({ x1 + std::max<float>(-adjacentWidth1, 0), y2 });
810 quad.uncheckedAppend({ x2 - std::max<float>(-adjacentWidth2, 0), y2 });
811 quad.uncheckedAppend({ x2 - std::max<float>( adjacentWidth2, 0), y1 });
812 break;
813 case BoxSide::Left:
814 quad.uncheckedAppend({ x1, y1 + std::max<float>(-adjacentWidth1, 0) });
815 quad.uncheckedAppend({ x1, y2 - std::max<float>(-adjacentWidth2, 0) });
816 quad.uncheckedAppend({ x2, y2 - std::max<float>( adjacentWidth2, 0) });
817 quad.uncheckedAppend({ x2, y1 + std::max<float>( adjacentWidth1, 0) });
818 break;
819 case BoxSide::Right:
820 quad.uncheckedAppend({ x1, y1 + std::max<float>( adjacentWidth1, 0) });
821 quad.uncheckedAppend({ x1, y2 - std::max<float>( adjacentWidth2, 0) });
822 quad.uncheckedAppend({ x2, y2 - std::max<float>(-adjacentWidth2, 0) });
823 quad.uncheckedAppend({ x2, y1 + std::max<float>(-adjacentWidth1, 0) });
824 break;
825 }
826
827 paintingContext.context.setStrokeStyle(NoStroke);
828 paintingContext.context.setFillColor(color);
829 bool wasAntialiased = paintingContext.context.shouldAntialias();
830 paintingContext.context.setShouldAntialias(antialias);
831 paintingContext.context.fillPath(Path::polygonPathFromPoints(quad));
832 paintingContext.context.setShouldAntialias(wasAntialiased);
833
834 paintingContext.context.setStrokeStyle(oldStrokeStyle);
835 break;
836 }
837 }
838}

840void BorderPainter::paintOneBorderSide(PaintingContext& paintingContext, const FloatRoundedRect& outerBorder, const FloatRoundedRect& innerBorder, const FloatRect& sideRect, BoxSide side, const Path* path, bool antialias, const Color* overrideColor) const
841{
842 auto& edgeToRender = m_edges.at(side);
843 ASSERT(edgeToRender.widthForPainting());
844
845 auto adjacentSides = adjacentSidesForSide(side);
846
847 auto& adjacentEdge1 = m_edges.at(adjacentSides.first);
848 auto& adjacentEdge2 = m_edges.at(adjacentSides.second);
849
850 bool mitreAdjacentSide1 = joinRequiresMitre(side, adjacentSides.first, !antialias);
851 bool mitreAdjacentSide2 = joinRequiresMitre(side, adjacentSides.second, !antialias);
852
853 bool adjacentSide1StylesMatch = colorsMatchAtCorner(side, adjacentSides.first);
854 bool adjacentSide2StylesMatch = colorsMatchAtCorner(side, adjacentSides.second);
855
856 const Color& colorToPaint = overrideColor ? *overrideColor : edgeToRender.color();
857
858 if (path) {
859 GraphicsContextStateSaver stateSaver(paintingContext.context);
860
861 clipBorderSidePolygon(paintingContext, outerBorder, innerBorder, side, adjacentSide1StylesMatch, adjacentSide2StylesMatch);
862
863 if (!innerBorder.isRenderable())
864 paintingContext.context.clipOutRoundedRect(FloatRoundedRect(calculateAdjustedInnerBorder(innerBorder, side)));
865
866 float thickness = std::max(std::max(edgeToRender.widthForPainting(), adjacentEdge1.widthForPainting()), adjacentEdge2.widthForPainting());
867 drawBoxSideFromPath(paintingContext, outerBorder.rect(), *path, edgeToRender.widthForPainting(), thickness, side, colorToPaint, edgeToRender.style());
868 return;
869 }
870
871 bool clipForStyle = styleRequiresClipPolygon(edgeToRender.style()) && (mitreAdjacentSide1 || mitreAdjacentSide2);
872 bool clipAdjacentSide1 = colorNeedsAntiAliasAtCorner(side, adjacentSides.first) && mitreAdjacentSide1;
873 bool clipAdjacentSide2 = colorNeedsAntiAliasAtCorner(side, adjacentSides.second) && mitreAdjacentSide2;
874 bool shouldClip = clipForStyle || clipAdjacentSide1 || clipAdjacentSide2;
875
876 GraphicsContextStateSaver clipStateSaver(paintingContext.context, shouldClip);
877 if (shouldClip) {
878 bool aliasAdjacentSide1 = clipAdjacentSide1 || (clipForStyle && mitreAdjacentSide1);
879 bool aliasAdjacentSide2 = clipAdjacentSide2 || (clipForStyle && mitreAdjacentSide2);
880 clipBorderSidePolygon(paintingContext, outerBorder, innerBorder, side, !aliasAdjacentSide1, !aliasAdjacentSide2);
881 // Since we clipped, no need to draw with a mitre.
882 mitreAdjacentSide1 = false;
883 mitreAdjacentSide2 = false;
884 }
885 drawLineForBoxSide(paintingContext, sideRect, side, colorToPaint, edgeToRender.style(), mitreAdjacentSide1 ? adjacentEdge1.widthForPainting() : 0, mitreAdjacentSide2 ? adjacentEdge2.widthForPainting() : 0, antialias);
886}

888void BorderPainter::paintBorderSides(PaintingContext& paintingContext, const FloatRoundedRect& outerBorder, const FloatRoundedRect& innerBorder, FloatSize innerBorderBleedAdjustment, BoxSideSet edgeSet, bool antialias, const Color* overrideColor) const
889{
890 bool renderRadii = outerBorder.isRounded();
891
892 Path roundedPath;
893 if (renderRadii)
894 roundedPath.addRoundedRect(outerBorder);
895
896 auto paintOneSide = [&](BoxSide side) {
897 auto& edge = m_edges.at(side);
898 if (!edge.shouldRender() || !edgeSet.contains(edgeFlagForSide(side)))
899 return;
900
901 auto sideRect = outerBorder.rect();
902 FloatSize firstRadius;
903 FloatSize secondRadius;
904
905 switch (side) {
906 case BoxSide::Top:
907 sideRect.setHeight(edge.widthForPainting() + innerBorderBleedAdjustment.height());
908 firstRadius = innerBorder.radii().topLeft();
909 secondRadius = innerBorder.radii().topRight();
910 break;
911 case BoxSide::Right:
912 sideRect.shiftXEdgeTo(sideRect.maxX() - edge.widthForPainting() - innerBorderBleedAdjustment.width());
913 firstRadius = innerBorder.radii().bottomRight();
914 secondRadius = innerBorder.radii().topRight();
915 break;
916 case BoxSide::Bottom:
917 sideRect.shiftYEdgeTo(sideRect.maxY() - edge.widthForPainting() - innerBorderBleedAdjustment.height());
918 firstRadius = innerBorder.radii().bottomLeft();
919 secondRadius = innerBorder.radii().bottomRight();
920 break;
921 case BoxSide::Left:
922 sideRect.setWidth(edge.widthForPainting() + innerBorderBleedAdjustment.width());
923 firstRadius = innerBorder.radii().bottomLeft();
924 secondRadius = innerBorder.radii().topLeft();
925 break;
926 }
927
928 bool usePath = renderRadii && (borderStyleHasInnerDetail(edge.style()) || borderWillArcInnerEdge(firstRadius, secondRadius));
929 paintOneBorderSide(paintingContext, outerBorder, innerBorder, sideRect, side, usePath ? &roundedPath : nullptr, antialias, overrideColor);
930 };
931
932 paintOneSide(BoxSide::Top);
933 paintOneSide(BoxSide::Bottom);
934 paintOneSide(BoxSide::Left);
935 paintOneSide(BoxSide::Right);
936}

938void BorderPainter::paintTranslucentBorderSides(PaintingContext& paintingContext, const FloatRoundedRect& outerBorder, const FloatRoundedRect& innerBorder, FloatSize innerBorderBleedAdjustment, BoxSideSet edgesToDraw, bool antialias) const
939{
940 // willBeOverdrawn assumes that we draw in order: top, bottom, left, right.
941 // This is different from BoxSide enum order.
942 static constexpr std::array<BoxSide, 4> paintOrderSides = { BoxSide::Top, BoxSide::Bottom, BoxSide::Left, BoxSide::Right };
943
944 while (edgesToDraw) {
945 // Find undrawn edges sharing a color.
946 Color commonColor;
947
948 BoxSideSet commonColorEdgeSet;
949 for (auto side : paintOrderSides) {
950 if (!edgesToDraw.contains(edgeFlagForSide(side)))
951 continue;
952
953 auto& edge = m_edges.at(side);
954 bool includeEdge;
955 if (commonColorEdgeSet.isEmpty()) {
956 commonColor = edge.color();
957 includeEdge = true;
958 } else
959 includeEdge = edge.color() == commonColor;
960
961 if (includeEdge)
962 commonColorEdgeSet.add(edgeFlagForSide(side));
963 }
964
965 bool useTransparencyLayer = includesAdjacentEdges(commonColorEdgeSet) && !commonColor.isOpaque();
966 {
967 auto transparencyScope = TransparencyLayerScope { paintingContext.context, commonColor.alphaAsFloat(), useTransparencyLayer };
968 if (useTransparencyLayer)
969 commonColor = commonColor.opaqueColor();
970
971 paintBorderSides(paintingContext, outerBorder, innerBorder, innerBorderBleedAdjustment, commonColorEdgeSet, antialias, &commonColor);
972 }
973
974 edgesToDraw.remove(commonColorEdgeSet);
975 }
976}

996void BorderPainter::paintBorders(PaintingContext& paintingContext) const
997{
998 bool haveAlphaColor = false;
999 bool haveAllSolidEdges = true;
1000 bool haveAllDoubleEdges = true;
1001 int numEdgesVisible = 4;
1002 bool allEdgesShareColor = true;
1003 std::optional<BoxSide> firstVisibleSide;
1004 BoxSideSet edgesToDraw;
1005
1006 for (auto side : allBoxSides) {
1007 auto& currEdge = m_edges.at(side);
1008
1009 if (currEdge.shouldRender())
1010 edgesToDraw.add(edgeFlagForSide(side));
1011
1012 if (currEdge.presentButInvisible()) {
1013 --numEdgesVisible;
1014 allEdgesShareColor = false;
1015 continue;
1016 }
1017
1018 if (!currEdge.widthForPainting()) {
1019 --numEdgesVisible;
1020 continue;
1021 }
1022
1023 if (!firstVisibleSide)
1024 firstVisibleSide = side;
1025 else if (currEdge.color() != m_edges.at(*firstVisibleSide).color())
1026 allEdgesShareColor = false;
1027
1028 if (!currEdge.color().isOpaque())
1029 haveAlphaColor = true;
1030
1031 if (currEdge.style() != BorderStyle::Solid)
1032 haveAllSolidEdges = false;
1033
1034 if (currEdge.style() != BorderStyle::Double)
1035 haveAllDoubleEdges = false;
1036 }
1037
1038 FloatRoundedRect innerBorderRect = roundedInsetBorderForRect(borderInnerRectAdjustedForBleedAvoidance(paintingContext), m_borderRect.radii(), borderWidths(m_edges), m_includeLeftEdge, m_includeRightEdge);
1039
1040 // FIXME: If no corner intersects the clip region, we can pretend outerBorder is rectangular to improve performance.
1041
1042 if ((haveAllSolidEdges || haveAllDoubleEdges) && allEdgesShareColor && innerBorderRect.isRenderable()) {
1043 // Fast path for drawing all solid edges and all unrounded double edges
1044 if (numEdgesVisible == 4 && (m_borderRect.isRounded() || haveAlphaColor) && (haveAllSolidEdges || (!m_borderRect.isRounded() && !innerBorderRect.isRounded()))) {
1045 Path path;
1046
1047 if (m_borderRect.isRounded() && m_bleedAvoidance != BackgroundBleedAvoidance::UseTransparencyLayer)
1048 path.addRoundedRect(m_borderRect);
1049 else
1050 path.addRect(m_borderRect.rect());
1051
1052 if (haveAllDoubleEdges) {
1053 auto innerThirdRect = m_borderRect.rect();
1054 auto outerThirdRect = m_borderRect.rect();
1055
1056 for (auto side : allBoxSides) {
1057 auto& edge = m_edges.at(side);
1058 float outerWidth = edge.outerWidth();
1059 float innerWidth = edge.innerWidth();
1060 switch (side) {
1061 case BoxSide::Top:
1062 innerThirdRect.shiftYEdgeTo(innerThirdRect.y() + innerWidth);
1063 outerThirdRect.shiftYEdgeTo(outerThirdRect.y() + outerWidth);
1064 break;
1065 case BoxSide::Right:
1066 innerThirdRect.setWidth(innerThirdRect.width() - innerWidth);
1067 outerThirdRect.setWidth(outerThirdRect.width() - outerWidth);
1068 break;
1069 case BoxSide::Bottom:
1070 innerThirdRect.setHeight(innerThirdRect.height() - innerWidth);
1071 outerThirdRect.setHeight(outerThirdRect.height() - outerWidth);
1072 break;
1073 case BoxSide::Left:
1074 innerThirdRect.shiftXEdgeTo(innerThirdRect.x() + innerWidth);
1075 outerThirdRect.shiftXEdgeTo(outerThirdRect.x() + outerWidth);
1076 break;
1077 }
1078 }
1079
1080 auto outerThirdRoundedRect = m_borderRect;
1081 outerThirdRoundedRect.setRect(outerThirdRect);
1082
1083 if (outerThirdRoundedRect.isRounded() && m_bleedAvoidance != BackgroundBleedAvoidance::UseTransparencyLayer)
1084 path.addRoundedRect(outerThirdRoundedRect);
1085 else
1086 path.addRect(outerThirdRoundedRect.rect());
1087
1088 auto innerThirdRoundedRect = innerBorderRect;
1089 innerThirdRoundedRect.setRect(innerThirdRect);
1090 if (innerThirdRoundedRect.isRounded())
1091 path.addRoundedRect(innerThirdRoundedRect);
1092 else
1093 path.addRect(innerThirdRoundedRect.rect());
1094 }
1095
1096 if (innerBorderRect.isRounded())
1097 path.addRoundedRect(innerBorderRect);
1098 else
1099 path.addRect(innerBorderRect.rect());
1100
1101 paintingContext.context.setFillRule(WindRule::EvenOdd);
1102 paintingContext.context.setFillColor(m_edges.top().color());
1103 paintingContext.context.fillPath(path);
1104 return;
1105 }
1106
1107 // Avoid creating transparent layers
1108 if (haveAllSolidEdges && numEdgesVisible != 4 && !m_borderRect.isRounded() && haveAlphaColor) {
1109 Path path;
1110

1258void BoxDecorationPainter::paintBoxShadow(PaintingContext& paintingContext, ShadowStyle shadowStyle) const
1259{
1260 if (!m_box.style().boxShadow())
1261 return;
1262
1263 auto borderRect = shadowStyle == ShadowStyle::Inset ? innerBorderRoundedRect() : borderRoundedRect();
1264 auto* boxDecorationData = m_box.boxDecorationData();
1265 bool hasBorderRadius = m_box.hasBorderRadius();
1266
1267 bool hasOpaqueBackground = m_box.style().backgroundColor().isOpaque();
1268
1269 auto paintNormalShadow = [&](const ShadowData& shadow) {
1270 // FIXME: Snapping here isn't ideal. It would be better to compute a rect which is border rect + offset + spread, and snap that at tree building time.
1271 auto shadowOffset = roundSizeToDevicePixels({ shadow.x(), shadow.y() }, paintingContext.deviceScaleFactor);
1272 float shadowPaintingExtent = ceilToDevicePixel(shadow.paintingExtent(), paintingContext.deviceScaleFactor);
1273 float shadowSpread = roundToDevicePixel(shadow.spread(), paintingContext.deviceScaleFactor);
1274 int shadowRadius = shadow.radius();
1275
1276 auto fillRect = borderRect;
1277 fillRect.inflate(shadowSpread);
1278 if (fillRect.isEmpty())
1279 return;
1280
1281 auto shadowRect = borderRect.rect();
1282 shadowRect.inflate(shadowPaintingExtent + shadowSpread);
1283 shadowRect.move(shadowOffset);
1284
1285 GraphicsContextStateSaver stateSaver(paintingContext.context);
1286 paintingContext.context.clip(shadowRect);
1287
1288 // Move the fill just outside the clip, adding at least 1 pixel of separation so that the fill does not
1289 // bleed in (due to antialiasing) if the context is transformed.
1290 float xOffset = shadowRect.width() + std::max<float>(0, shadowOffset.width()) + shadowPaintingExtent + 2 * shadowSpread + 1.0f;
1291 auto extraOffset = FloatSize { std::ceil(xOffset), 0 };
1292 shadowOffset -= extraOffset;
1293 fillRect.move(extraOffset);
1294
1295 auto rectToClipOut = borderRect;
1296 auto adjustedFillRect = fillRect;
1297
1298 auto shadowRectOrigin = fillRect.rect().location() + shadowOffset;
1299 auto adjustedShadowOffset = shadowRectOrigin - adjustedFillRect.rect().location();
1300
1301 paintingContext.context.setShadow(adjustedShadowOffset, shadowRadius, shadow.color(), shadow.isWebkitBoxShadow() ? ShadowRadiusMode::Legacy : ShadowRadiusMode::Default);
1302
1303 if (hasBorderRadius) {
1304 // If the box is opaque, it is unnecessary to clip it out. However, doing so saves time
1305 // when painting the shadow. On the other hand, it introduces subpixel gaps along the
1306 // corners. Those are avoided by insetting the clipping path by one pixel.
1307 if (hasOpaqueBackground)
1308 rectToClipOut.inflateWithRadii(-1.0f);
1309
1310 if (!rectToClipOut.isEmpty())
1311 paintingContext.context.clipOutRoundedRect(rectToClipOut);
1312
1313 auto influenceRect = FloatRoundedRect { shadowRect, borderRect.radii() };
1314 influenceRect.expandRadii(2 * shadowPaintingExtent + shadowSpread);
1315
1316 // FIXME: Optimize for clipped-out corners
1317 adjustedFillRect.expandRadii(shadowSpread);
1318 if (!adjustedFillRect.isRenderable())
1319 adjustedFillRect.adjustRadii();
1320 paintingContext.context.fillRoundedRect(adjustedFillRect, Color::black);
1321 } else {
1322 // If the box is opaque, it is unnecessary to clip it out. However, doing so saves time
1323 // when painting the shadow. On the other hand, it introduces subpixel gaps along the
1324 // edges if they are not pixel-aligned. Those are avoided by insetting the clipping path
1325 // by one pixel.
1326 if (hasOpaqueBackground) {
1327 // FIXME: The function to decide on the policy based on the transform should be a named function.
1328 // FIXME: It's not clear if this check is right. What about integral scale factors?
1329 AffineTransform transform = paintingContext.context.getCTM();
1330 if (transform.a() != 1 || (transform.d() != 1 && transform.d() != -1) || transform.b() || transform.c())
1331 rectToClipOut.inflate(-1.0f);
1332 }
1333
1334 if (!rectToClipOut.isEmpty())
1335 paintingContext.context.clipOut(rectToClipOut.rect());
1336
1337 paintingContext.context.fillRect(adjustedFillRect.rect(), Color::black);
1338 }
1339 };
1340
1341 auto paintInsetShadow = [&](const ShadowData& shadow) {
1342 auto shadowOffset = roundSizeToDevicePixels({ shadow.x(), shadow.y() }, paintingContext.deviceScaleFactor);
1343 float shadowPaintingExtent = ceilToDevicePixel(shadow.paintingExtent(), paintingContext.deviceScaleFactor);
1344 float shadowSpread = roundToDevicePixel(shadow.spread(), paintingContext.deviceScaleFactor);
1345 int shadowRadius = shadow.radius();
1346
1347 auto holeRect = borderRect.rect();
1348 holeRect.inflate(-shadowSpread);
1349
1350 if (!m_includeLeftEdge) {
1351 // FIXME: Need to take writing mode into account.
1352 holeRect.shiftXEdgeBy(-(std::max<float>(shadowOffset.width(), 0) + shadowPaintingExtent + shadowSpread));
1353 }
1354
1355 if (!m_includeRightEdge) {
1356 // FIXME: Need to take writing mode into account.
1357 holeRect.setWidth(holeRect.width() - std::min<float>(shadowOffset.width(), 0) + shadowPaintingExtent + shadowSpread);
1358 }
1359
1360 auto roundedHoleRect = FloatRoundedRect { holeRect, borderRect.radii() };
1361 if (shadowSpread && roundedHoleRect.isRounded()) {
1362 auto roundedRectCorrectingForSpread = [&]() {
1363 bool horizontal = true; // FIXME: Handle writing modes.
1364 auto borderWidth = borderWidths(boxDecorationData->borderEdges());
1365
1366 float leftWidth { (!horizontal || m_includeLeftEdge) ? borderWidth.left() + shadowSpread : 0 };
1367 float rightWidth { (!horizontal || m_includeRightEdge) ? borderWidth.right() + shadowSpread : 0 };
1368 float topWidth { (horizontal || m_includeLeftEdge) ? borderWidth.top() + shadowSpread : 0 };
1369 float bottomWidth { (horizontal || m_includeRightEdge) ? borderWidth.bottom() + shadowSpread : 0 };
1370
1371 return roundedInsetBorderForRect(m_borderRect.rect(), m_borderRect.radii(), { topWidth, rightWidth, bottomWidth, leftWidth }, m_includeLeftEdge, m_includeRightEdge);
1372 }();
1373 roundedHoleRect.setRadii(roundedRectCorrectingForSpread.radii());
1374 }
1375
1376 if (roundedHoleRect.isEmpty()) {
1377 if (hasBorderRadius)
1378 paintingContext.context.fillRoundedRect(borderRect, shadow.color());
1379 else
1380 paintingContext.context.fillRect(borderRect.rect(), shadow.color());
1381 return;
1382 }
1383
1384 auto areaCastingShadowInHole = [](const FloatRect& holeRect, float shadowExtent, float shadowSpread, FloatSize shadowOffset) {
1385 auto bounds(holeRect);
1386
1387 bounds.inflate(shadowExtent);
1388
1389 if (shadowSpread < 0)
1390 bounds.inflate(-shadowSpread);
1391
1392 auto offsetBounds = bounds;
1393 offsetBounds.move(-shadowOffset);
1394 return unionRect(bounds, offsetBounds);
1395 };
1396
1397 Color fillColor = shadow.color().opaqueColor();
1398 auto shadowCastingRect = areaCastingShadowInHole(borderRect.rect(), shadowPaintingExtent, shadowSpread, shadowOffset);
1399
1400 GraphicsContextStateSaver stateSaver(paintingContext.context);
1401 if (hasBorderRadius)
1402 paintingContext.context.clipRoundedRect(borderRect);
1403 else
1404 paintingContext.context.clip(borderRect.rect());
1405
1406 float xOffset = shadowCastingRect.width() + std::max<float>(0, shadowOffset.width()) + shadowPaintingExtent - 2 * shadowSpread + 1.0f;
1407 auto extraOffset = FloatSize { std::ceil(xOffset), 0 };
1408
1409 paintingContext.context.translate(extraOffset);
1410 shadowOffset -= extraOffset;
1411
1412 paintingContext.context.setShadow(shadowOffset, shadowRadius, shadow.color(), shadow.isWebkitBoxShadow() ? ShadowRadiusMode::Legacy : ShadowRadiusMode::Default);
1413 paintingContext.context.fillRectWithRoundedHole(shadowCastingRect, roundedHoleRect, fillColor);
1414 };
1415
1416
1417 for (auto* shadow = m_box.style().boxShadow(); shadow; shadow = shadow->next()) {
1418 if (shadow->style() != shadowStyle)
1419 continue;
1420
1421 LayoutSize shadowOffset(shadow->x(), shadow->y());
1422 if (shadowOffset.isZero() && !shadow->radius() && !shadow->spread())
1423 continue;
1424
1425 if (shadow->style() == ShadowStyle::Normal)
1426 paintNormalShadow(*shadow);
1427 else
1428 paintInsetShadow(*shadow);
1429 }
