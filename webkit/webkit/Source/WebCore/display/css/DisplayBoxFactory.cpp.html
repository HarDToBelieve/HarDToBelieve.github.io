59RootBackgroundPropagation BoxFactory::determineRootBackgroundPropagation(const Layout::ContainerBox& rootLayoutBox)
60{
61 auto* documentElementBox = documentElementBoxFromRootBox(rootLayoutBox);
62 auto* bodyBox = bodyBoxFromRootBox(rootLayoutBox);
63
64 if (documentElementBox && documentElementBox->style().hasBackground())
65 return RootBackgroundPropagation::None;
66
67 if (bodyBox && bodyBox->style().hasBackground())
68 return RootBackgroundPropagation::BodyToRoot;
69
70 return RootBackgroundPropagation::None;
71}

73std::unique_ptr<ContainerBox> BoxFactory::displayBoxForRootBox(const Layout::ContainerBox& rootLayoutBox, const Layout::BoxGeometry& geometry, RootBackgroundPropagation rootBackgroundPropagation) const
74{
75 ASSERT(is<Layout::InitialContainingBlock>(rootLayoutBox));
76
77 // FIXME: Need to do logical -> physical coordinate mapping here.
78 auto borderBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(geometry) };
79
80 auto* documentElementBox = documentElementBoxFromRootBox(rootLayoutBox);
81
82 const RenderStyle* styleForBackground = documentElementBox ? &documentElementBox->style() : nullptr;
83
84 if (rootBackgroundPropagation == RootBackgroundPropagation::BodyToRoot) {
85 if (auto* bodyBox = bodyBoxFromRootBox(rootLayoutBox))
86 styleForBackground = &bodyBox->style();
87 }
88
89 auto style = Style { rootLayoutBox.style(), styleForBackground };
90
91 auto rootBox = makeUnique<ContainerBox>(m_treeBuilder.tree(), snapRectToDevicePixels(borderBoxRect, m_pixelSnappingFactor), WTFMove(style));
92 // We pass rootBox as its own containingBlockBox here to allow it to be a reference everywhere else.
93 setupBoxModelBox(*rootBox, rootLayoutBox, geometry, { *rootBox, { 0, 0 } }, styleForBackground);
94 return rootBox;
95}

97std::unique_ptr<Box> BoxFactory::displayBoxForBodyBox(const Layout::Box& layoutBox, const Layout::BoxGeometry& geometry, const ContainingBlockContext& containingBlockContext, RootBackgroundPropagation rootBackgroundPropagation) const
98{
99 const RenderStyle* styleForBackground = &layoutBox.style();
100
101 if (rootBackgroundPropagation == RootBackgroundPropagation::BodyToRoot)
102 styleForBackground = nullptr;
103
104 auto style = Style { layoutBox.style(), styleForBackground };
105 return displayBoxForLayoutBox(layoutBox, geometry, containingBlockContext, styleForBackground, WTFMove(style));
106}

108std::unique_ptr<Box> BoxFactory::displayBoxForLayoutBox(const Layout::Box& layoutBox, const Layout::BoxGeometry& geometry, const ContainingBlockContext& containingBlockContext) const
109{
110 auto style = Style { layoutBox.style() };

114std::unique_ptr<Box> BoxFactory::displayBoxForLayoutBox(const Layout::Box& layoutBox, const Layout::BoxGeometry& geometry, const ContainingBlockContext& containingBlockContext, const RenderStyle* styleForBackground, Style&& style) const
115{
116 // FIXME: Need to map logical to physical rects.
117 auto borderBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(geometry) };
118 borderBoxRect.move(containingBlockContext.offsetFromRoot);
119 auto pixelSnappedBorderBoxRect = snapRectToDevicePixels(borderBoxRect, m_pixelSnappingFactor);
120
121 // FIXME: Handle isAnonymous()
122
123 if (is<Layout::ReplacedBox>(layoutBox)) {
124 // FIXME: Don't assume it's an image.
125 RefPtr<Image> image;
126 if (auto* cachedImage = downcast<Layout::ReplacedBox>(layoutBox).cachedImage())
127 image = cachedImage->image();
128
129 auto imageBox = makeUnique<ImageBox>(m_treeBuilder.tree(), pixelSnappedBorderBoxRect, WTFMove(style), WTFMove(image));
130 setupBoxModelBox(*imageBox, layoutBox, geometry, containingBlockContext, styleForBackground);
131 return imageBox;
132 }
133
134 if (is<Layout::ContainerBox>(layoutBox)) {
135 // FIXME: The decision to make a ContainerBox should be made based on whether this Display::Box will have children.
136 auto containerBox = makeUnique<ContainerBox>(m_treeBuilder.tree(), pixelSnappedBorderBoxRect, WTFMove(style));
137 setupBoxModelBox(*containerBox, layoutBox, geometry, containingBlockContext, styleForBackground);
138 return containerBox;
139 }
140
141 OptionSet<Box::TypeFlags> flags;
142 // FIXME: Workaround for webkit.org/b/219335.
143 if (layoutBox.isLineBreakBox())
144 flags.add(Box::TypeFlags::LineBreakBox);
145
146 return makeUnique<Box>(m_treeBuilder.tree(), snapRectToDevicePixels(borderBoxRect, m_pixelSnappingFactor), WTFMove(style), flags);
147}

149std::unique_ptr<Box> BoxFactory::displayBoxForTextRun(const Layout::LineRun& run, const Layout::InlineLineGeometry& lineGeometry, const ContainingBlockContext& containingBlockContext) const
150{
151 ASSERT(run.text());
152 auto lineRect = lineGeometry.lineBoxLogicalRect();
153 auto lineLayoutRect = LayoutRect { lineRect.left(), lineRect.top(), lineRect.width(), lineRect.height() };
154
155 auto runRect = LayoutRect { run.logicalLeft(), run.logicalTop(), run.logicalWidth(), run.logicalHeight() };
156 runRect.moveBy(lineLayoutRect.location());
157 runRect.move(containingBlockContext.offsetFromRoot);
158
159 auto style = Style { run.layoutBox().style() };
160 return makeUnique<TextBox>(m_treeBuilder.tree(), snapRectToDevicePixels(runRect, m_pixelSnappingFactor), WTFMove(style), run);
161}

163void BoxFactory::setupBoxGeometry(BoxModelBox& box, const Layout::Box&, const Layout::BoxGeometry& layoutGeometry, const ContainingBlockContext& containingBlockContext) const
164{
165 auto borderBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(layoutGeometry) };
166 borderBoxRect.move(containingBlockContext.offsetFromRoot);
167
168 auto paddingBoxRect = LayoutRect { layoutGeometry.paddingBox() };
169 paddingBoxRect.moveBy(borderBoxRect.location());
170 box.setAbsolutePaddingBoxRect(snapRectToDevicePixels(paddingBoxRect, m_pixelSnappingFactor));
171
172 auto contentBoxRect = LayoutRect { layoutGeometry.contentBox() };
173 contentBoxRect.moveBy(borderBoxRect.location());
174 box.setAbsoluteContentBoxRect(snapRectToDevicePixels(contentBoxRect, m_pixelSnappingFactor));
175
176 if (is<ReplacedBox>(box)) {
177 auto& replacedBox = downcast<ReplacedBox>(box);
178 // FIXME: Need to get the correct rect taking object-fit etc into account.
179 auto replacedContentRect = LayoutRect { layoutGeometry.contentBoxLeft(), layoutGeometry.contentBoxTop(), layoutGeometry.contentBoxWidth(), layoutGeometry.contentBoxHeight() };
180 replacedContentRect.moveBy(borderBoxRect.location());
181 auto pixelSnappedReplacedContentRect = snapRectToDevicePixels(replacedContentRect, m_pixelSnappingFactor);
182 replacedBox.setReplacedContentRect(pixelSnappedReplacedContentRect);
183 }
184}

186std::unique_ptr<BoxDecorationData> BoxFactory::constructBoxDecorationData(const Layout::Box& layoutBox, const Layout::BoxGeometry& layoutGeometry, const RenderStyle* styleForBackground, LayoutSize offsetFromRoot) const
187{
188 auto boxDecorationData = makeUnique<BoxDecorationData>();
189
190 if (styleForBackground) {
191 auto backgroundImageGeometry = calculateFillLayerImageGeometry(*styleForBackground, layoutGeometry, offsetFromRoot, m_pixelSnappingFactor);
192 boxDecorationData->setBackgroundImageGeometry(WTFMove(backgroundImageGeometry));
193 }
194
195 bool includeLogicalLeftEdge = true; // FIXME.
196 bool includeLogicalRightEdge = true; // FIXME.
197 auto borderEdges = calculateBorderEdges(layoutBox.style(), m_pixelSnappingFactor, includeLogicalLeftEdge, includeLogicalRightEdge);
198 boxDecorationData->setBorderEdges(WTFMove(borderEdges));
199
200 return boxDecorationData;
201}

203FloatPoint3D BoxFactory::computeTransformOrigin(const BoxModelBox& box, const Layout::BoxGeometry& layoutGeometry, const RenderStyle& renderStyle, LayoutSize offsetFromRoot) const
204{
205 auto transformBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(layoutGeometry) };
206
207 auto absoluteOrigin = LayoutPoint {
208 offsetFromRoot.width() + transformBoxRect.x() + valueForLength(renderStyle.transformOriginX(), transformBoxRect.width()),
209 offsetFromRoot.height() + transformBoxRect.y() + valueForLength(renderStyle.transformOriginY(), transformBoxRect.height())
210 };
211
212 auto snappedAbsoluteOrigin = roundPointToDevicePixels(absoluteOrigin, m_pixelSnappingFactor);
213 auto boxRelativeTransformOriginXY = snappedAbsoluteOrigin - box.absoluteBorderBoxRect().location();
214
215 return { boxRelativeTransformOriginXY.width(), boxRelativeTransformOriginXY.height(), renderStyle.transformOriginZ() };
216}

218TransformationMatrix BoxFactory::computeTransformationMatrix(const BoxModelBox& box, const Layout::BoxGeometry& layoutGeometry, const RenderStyle& renderStyle, LayoutSize offsetFromRoot) const
219{
220 auto boxRelativeTransformOrigin = computeTransformOrigin(box, layoutGeometry, renderStyle, offsetFromRoot);
221
222 // FIXME: Respect transform-box.
223 auto transformBoxRect = box.absoluteBorderBoxRect();
224
225 // FIXME: This is similar to RenderStyle::applyTransform(), but that fails to pixel snap transform origin.
226 auto transform = TransformationMatrix { };
227 transform.translate3d(boxRelativeTransformOrigin.x(), boxRelativeTransformOrigin.y(), boxRelativeTransformOrigin.z());
228
229 for (auto& operation : renderStyle.transform().operations())
230 operation->apply(transform, transformBoxRect.size());
231
232 transform.translate3d(-boxRelativeTransformOrigin.x(), -boxRelativeTransformOrigin.y(), -boxRelativeTransformOrigin.z());
233 return transform;
234
235}

237std::unique_ptr<BoxRareGeometry> BoxFactory::constructBoxRareGeometry(const BoxModelBox& box, const Layout::Box& layoutBox, const Layout::BoxGeometry& layoutGeometry, LayoutSize offsetFromRoot) const
238{
239 auto& renderStyle = layoutBox.style();
240
241 if (!box.style().hasTransform() && !renderStyle.hasBorderRadius())
242 return nullptr;
243
244 auto boxRareGeometry = makeUnique<BoxRareGeometry>();
245
246 if (renderStyle.hasBorderRadius()) {
247 bool includeLogicalLeftEdge = true; // FIXME.
248 bool includeLogicalRightEdge = true; // FIXME.
249
250 auto borderBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(layoutGeometry) };
251 auto borderRoundedRect = renderStyle.getRoundedBorderFor(borderBoxRect, includeLogicalLeftEdge, includeLogicalRightEdge);
252 auto snappedRoundedRect = borderRoundedRect.pixelSnappedRoundedRectForPainting(m_pixelSnappingFactor);
253
254 auto borderRadii = makeUnique<FloatRoundedRect::Radii>(snappedRoundedRect.radii());
255 boxRareGeometry->setBorderRadii(WTFMove(borderRadii));
256 }
257
258 auto transformationMatrix = computeTransformationMatrix(box, layoutGeometry, layoutBox.style(), offsetFromRoot);
259 boxRareGeometry->setTransform(WTFMove(transformationMatrix));
260
261 return boxRareGeometry;
262}

264void BoxFactory::setupBoxModelBox(BoxModelBox& box, const Layout::Box& layoutBox, const Layout::BoxGeometry& layoutGeometry, const ContainingBlockContext& containingBlockContext, const RenderStyle* styleForBackground) const
265{
266 setupBoxGeometry(box, layoutBox, layoutGeometry, containingBlockContext);
267
268 auto boxRareGeometry = constructBoxRareGeometry(box, layoutBox, layoutGeometry, containingBlockContext.offsetFromRoot);
269 box.setBoxRareGeometry(WTFMove(boxRareGeometry));
270
271 auto& renderStyle = layoutBox.style();
272 if (!(styleForBackground && styleForBackground->hasBackground()) && !renderStyle.hasBorder()) // FIXME: Misses border-radius.
273 return;
274
275 auto boxDecorationData = constructBoxDecorationData(layoutBox, layoutGeometry, styleForBackground, containingBlockContext.offsetFromRoot);
276 box.setBoxDecorationData(WTFMove(boxDecorationData));
277
278 if (box.participatesInZOrderSorting()) {
279 RefPtr<BoxClip> clip = containingBlockContext.box.clipForDescendants();
280 box.setAncestorClip(WTFMove(clip));
281 }
282}

284const Layout::ContainerBox* BoxFactory::documentElementBoxFromRootBox(const Layout::ContainerBox& rootLayoutBox)
285{
286 auto* documentBox = rootLayoutBox.firstChild();
287 if (!documentBox || !documentBox->isDocumentBox() || !is<Layout::ContainerBox>(documentBox))
288 return nullptr;
289
290 return downcast<Layout::ContainerBox>(documentBox);
291}
292
293const Layout::Box* BoxFactory::bodyBoxFromRootBox(const Layout::ContainerBox& rootLayoutBox)
294{
295 auto* documentBox = rootLayoutBox.firstChild();
296 if (!documentBox || !documentBox->isDocumentBox() || !is<Layout::ContainerBox>(documentBox))
297 return nullptr;
298
299 auto* bodyBox = downcast<Layout::ContainerBox>(documentBox)->firstChild();
300 if (!bodyBox || !bodyBox->isBodyBox())
301 return nullptr;
302
303 return bodyBox;
304}
