60bool BoxModelBox::hasBorderRadius() const
61{
62 return m_boxRareGeometry && m_boxRareGeometry->hasBorderRadius();
63}
64
65FloatRoundedRect BoxModelBox::borderRoundedRect() const
66{
67 auto borderRect = FloatRoundedRect { absoluteBorderBoxRect(), { } };
68 auto* borderRadii = m_boxRareGeometry ? m_boxRareGeometry->borderRadii() : nullptr;
69 if (borderRadii)
70 borderRect.setRadii(*borderRadii);
71
72 return borderRect;
73}
74
75FloatRoundedRect BoxModelBox::innerBorderRoundedRect() const
76{
77 if (!m_boxDecorationData)
78 return borderRoundedRect();
79
80 if (!hasBorderRadius())
81 return roundedInsetBorderForRect(absoluteBorderBoxRect(), { }, borderWidths(m_boxDecorationData->borderEdges()));
82
83 return roundedInsetBorderForRect(absoluteBorderBoxRect(), *m_boxRareGeometry->borderRadii(), borderWidths(m_boxDecorationData->borderEdges()));
84}

96RefPtr<BoxClip> BoxModelBox::clipForDescendants() const
97{
98 if (!style().hasClippedOverflow())
99 return m_ancestorClip;
100
101 auto clip = m_ancestorClip ? m_ancestorClip->copy() : BoxClip::create();
102
103 auto pushClip = [&](BoxClip& boxClip) {
104 if (hasBorderRadius()) {
105 auto roundedInnerBorder = innerBorderRoundedRect();
106 if (roundedInnerBorder.isRounded()) {
107 boxClip.pushRoundedClip(roundedInnerBorder);
108 return;
109 }
110
111 boxClip.pushClip(roundedInnerBorder.rect());
112 return;
113 }
114
115 boxClip.pushClip(absolutePaddingBoxRect());
116 };
117 pushClip(clip);
118
119 return clip;
120}
121
122AbsoluteFloatRect BoxModelBox::absolutePaintingExtent() const
123{
124 auto paintingExtent = absoluteBorderBoxRect();
125
126 if (auto* shadow = style().boxShadow())
127 shadow->adjustRectForShadow(paintingExtent, 0);
128
129 return paintingExtent;
130}
131
132const char* BoxModelBox::boxName() const
133{
134 return "box model box";
135}
136
137String BoxModelBox::debugDescription() const
138{
139 TextStream stream;
140 stream << boxName() << " " << absoluteBorderBoxRect() << " (" << this << ")";
141 if (m_ancestorClip)
142 stream << " ancestor clip " << m_ancestorClip->clipRect() << " affected by radius " << m_ancestorClip->affectedByBorderRadius();
143
144 return stream.release();
145}
