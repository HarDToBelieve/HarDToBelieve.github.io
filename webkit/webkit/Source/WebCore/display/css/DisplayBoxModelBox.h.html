33namespace WebCore {
34namespace Display {
35
36class BoxClip;
37class BoxDecorationData;
38class BoxRareGeometry;
39
40// A box in the sense of the CSS Box Model.
41// This box can draw backgrounds and borders.
42DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(BoxModelBox);
43class BoxModelBox : public Box {
44 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(BoxModelBox);
45 friend class BoxFactory;
46public:
47 BoxModelBox(Tree&, AbsoluteFloatRect borderBox, Style&&, OptionSet<TypeFlags> = { });
48 virtual ~BoxModelBox();
49
50 AbsoluteFloatRect absoluteBorderBoxRect() const { return absoluteBoxRect(); }
51 AbsoluteFloatRect absolutePaddingBoxRect() const { return m_paddingBoxRect; }
52 AbsoluteFloatRect absoluteContentBoxRect() const { return m_contentBoxRect; }
53
54 AbsoluteFloatRect absolutePaintingExtent() const override;
55
56 const BoxDecorationData* boxDecorationData() const { return m_boxDecorationData.get(); }
57 const BoxRareGeometry* rareGeometry() const { return m_boxRareGeometry.get(); }
58
59 const BoxClip* ancestorClip() const { return m_ancestorClip.get(); }
60 bool hasAncestorClip() const;
61
62 bool hasBorderRadius() const;
63 FloatRoundedRect borderRoundedRect() const;
64 FloatRoundedRect innerBorderRoundedRect() const;
65
66 String debugDescription() const override;
67
68private:
69 const char* boxName() const override;
70
71 void setAbsolutePaddingBoxRect(const AbsoluteFloatRect& box) { m_paddingBoxRect = box; }
72 void setAbsoluteContentBoxRect(const AbsoluteFloatRect& box) { m_contentBoxRect = box; }
73
74 void setBoxDecorationData(std::unique_ptr<BoxDecorationData>&&);
75 void setBoxRareGeometry(std::unique_ptr<BoxRareGeometry>&&);
76
77 void setAncestorClip(RefPtr<BoxClip>&&);
78 RefPtr<BoxClip> clipForDescendants() const;
79
80 AbsoluteFloatRect m_paddingBoxRect;
81 AbsoluteFloatRect m_contentBoxRect;
82
83 std::unique_ptr<BoxDecorationData> m_boxDecorationData;
84 std::unique_ptr<BoxRareGeometry> m_boxRareGeometry;
85 RefPtr<BoxClip> m_ancestorClip;
86};
87
88} // namespace Display
89} // namespace WebCore
