56void BoxPainter::paintBoxContent(const Box& box, PaintingContext& paintingContext)
57{
58 if (is<ImageBox>(box)) {
59 auto& imageBox = downcast<ImageBox>(box);
60
61 auto* image = imageBox.image();
62 auto imageRect = imageBox.replacedContentRect();
63
64 if (image)
65 paintingContext.context.drawImage(*image, imageRect);
66
67 return;
68 }
69
70 if (is<TextBox>(box)) {
71 auto& textBox = downcast<TextBox>(box);
72
73 auto& style = box.style();
74 auto textRect = box.absoluteBoxRect();
75
76 paintingContext.context.setStrokeColor(style.color());
77 paintingContext.context.setFillColor(style.color());
78
79 // FIXME: Add non-baseline align painting
80 auto baseline = textRect.y() + style.fontMetrics().ascent();
81 auto expansion = textBox.expansion();
82
83 auto textRun = TextRun { textBox.text().content().substring(textBox.text().start(), textBox.text().length()), textRect.x(), expansion.horizontalExpansion, expansion.behavior };
84 textRun.setTabSize(!style.collapseWhiteSpace(), style.tabSize());
85 paintingContext.context.drawText(style.fontCascade(), textRun, { textRect.x(), baseline });
86 return;
87 }
88}
89
90void BoxPainter::paintBox(const Box& box, PaintingContext& paintingContext, const IntRect& dirtyRect)
91{
92 auto absoluteRect = box.absoluteBoxRect();
93 // FIXME: Need to account for visual overflow.
94 if (!dirtyRect.intersects(enclosingIntRect(absoluteRect)))
95 return;
96
97 if (is<BoxModelBox>(box))
98 paintBoxDecorations(downcast<BoxModelBox>(box), paintingContext);
99
100 paintBoxContent(box, paintingContext);
101}
