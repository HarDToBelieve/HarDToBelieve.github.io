38// Storage for box geometry which is only used on boxes with rare style, like transforms or border-radius.
39DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(BoxRareGeometry);
40class BoxRareGeometry {
41 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(BoxRareGeometry);
42public:
43
44 const FloatRoundedRect::Radii* borderRadii() const { return m_borderRadii.get(); }
45 void setBorderRadii(std::unique_ptr<FloatRoundedRect::Radii>&& radii) { m_borderRadii = WTFMove(radii); }
46
47 bool hasBorderRadius() const;
48
49 // From transform, translate, rotate, scale properties.
50 const TransformationMatrix& transform() const { return m_transform; }
51 void setTransform(TransformationMatrix&& transform) { m_transform = WTFMove(transform); }
52
53private:
54 TransformationMatrix m_transform;
55 std::unique_ptr<FloatRoundedRect::Radii> m_borderRadii;
56};
