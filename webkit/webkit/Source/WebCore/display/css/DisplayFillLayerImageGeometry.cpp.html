41static inline LayoutUnit resolveWidthForRatio(LayoutUnit height, const LayoutSize& intrinsicRatio)
42{
43 return height * intrinsicRatio.width() / intrinsicRatio.height();
44}
45
46static inline LayoutUnit resolveHeightForRatio(LayoutUnit width, const LayoutSize& intrinsicRatio)
47{
48 return width * intrinsicRatio.height() / intrinsicRatio.width();
49}
50
51static inline LayoutSize resolveAgainstIntrinsicWidthOrHeightAndRatio(LayoutSize size, LayoutSize intrinsicRatio, LayoutUnit useWidth, LayoutUnit useHeight)
52{
53 if (intrinsicRatio.isEmpty()) {
54 if (useWidth)
55 return LayoutSize(useWidth, size.height());
56
57 return LayoutSize(size.width(), useHeight);
58 }
59
60 if (useWidth)
61 return LayoutSize(useWidth, resolveHeightForRatio(useWidth, intrinsicRatio));
62
63 return LayoutSize(resolveWidthForRatio(useHeight, intrinsicRatio), useHeight);
64}

66static inline LayoutSize resolveAgainstIntrinsicRatio(LayoutSize size, const LayoutSize& intrinsicRatio)
67{
68 // Two possible solutions: (size.width(), solutionHeight) or (solutionWidth, size.height())
69 // "... must be assumed to be the largest dimensions..." = easiest answer: the rect with the largest surface area.
70
71 LayoutUnit solutionWidth = resolveWidthForRatio(size.height(), intrinsicRatio);
72 LayoutUnit solutionHeight = resolveHeightForRatio(size.width(), intrinsicRatio);
73 if (solutionWidth <= size.width()) {
74 if (solutionHeight <= size.height()) {
75 // If both solutions fit, choose the one covering the larger area.
76 LayoutUnit areaOne = solutionWidth * size.height();
77 LayoutUnit areaTwo = size.width() * solutionHeight;
78 if (areaOne < areaTwo)
79 return LayoutSize(size.width(), solutionHeight);
80
81 return LayoutSize(solutionWidth, size.height());
82 }
83
84 // Only the first solution fits.
85 return LayoutSize(solutionWidth, size.height());
86 }
87
88 // Only the second solution fits, assert that.
89 ASSERT(solutionHeight <= size.height());
90 return LayoutSize(size.width(), solutionHeight);
91}

93static LayoutSize calculateImageIntrinsicDimensions(StyleImage* image, LayoutSize positioningAreaSize)
94{
95 // A generated image without a fixed size, will always return the container size as intrinsic size.
96 if (image->isGeneratedImage() && image->usesImageContainerSize())
97 return LayoutSize(positioningAreaSize.width(), positioningAreaSize.height());
98
99 // FIXME: Call computeIntrinsicDimensions().
100 auto imageSize = image->imageSize(nullptr, 1);
101 auto intrinsicRatio = imageSize;
102 Length intrinsicWidth = Length(intrinsicRatio.width(), LengthType::Fixed);
103 Length intrinsicHeight = Length(intrinsicRatio.height(), LengthType::Fixed);
104
105 ASSERT(!intrinsicWidth.isPercentOrCalculated());
106 ASSERT(!intrinsicHeight.isPercentOrCalculated());
107
108 LayoutSize resolvedSize(intrinsicWidth.value(), intrinsicHeight.value());
109 LayoutSize minimumSize(resolvedSize.width() > 0 ? 1 : 0, resolvedSize.height() > 0 ? 1 : 0);
110
111 // FIXME: Respect ScaleByEffectiveZoom.
112 resolvedSize.clampToMinimumSize(minimumSize);
113
114 if (!resolvedSize.isEmpty())
115 return resolvedSize;
116
117 // If the image has one of either an intrinsic width or an intrinsic height:
118 // * and an intrinsic aspect ratio, then the missing dimension is calculated from the given dimension and the ratio.
119 // * and no intrinsic aspect ratio, then the missing dimension is assumed to be the size of the rectangle that
120 // establishes the coordinate system for the 'background-position' property.
121 if (resolvedSize.width() > 0 || resolvedSize.height() > 0)
122 return resolveAgainstIntrinsicWidthOrHeightAndRatio(positioningAreaSize, LayoutSize(intrinsicRatio), resolvedSize.width(), resolvedSize.height());
123
124 // If the image has no intrinsic dimensions and has an intrinsic ratio the dimensions must be assumed to be the
125 // largest dimensions at that ratio such that neither dimension exceeds the dimensions of the rectangle that
126 // establishes the coordinate system for the 'background-position' property.
127 if (!intrinsicRatio.isEmpty())
128 return resolveAgainstIntrinsicRatio(positioningAreaSize, LayoutSize(intrinsicRatio));
129
130 // If the image has no intrinsic ratio either, then the dimensions must be assumed to be the rectangle that
131 // establishes the coordinate system for the 'background-position' property.
132 return positioningAreaSize;
133}

135static LayoutSize calculateFillTileSize(const FillLayer& fillLayer, LayoutSize positioningAreaSize, float pixelSnappingFactor)
136{
137 StyleImage* image = fillLayer.image();
138 FillSizeType type = fillLayer.size().type;
139 auto devicePixelSize = LayoutUnit { 1.0 / pixelSnappingFactor };
140
141 LayoutSize imageIntrinsicSize;
142 if (image) {
143 imageIntrinsicSize = calculateImageIntrinsicDimensions(image, positioningAreaSize);
144 imageIntrinsicSize.scale(1 / image->imageScaleFactor(), 1 / image->imageScaleFactor());
145 } else
146 imageIntrinsicSize = positioningAreaSize;
147
148 switch (type) {
149 case FillSizeType::Size: {
150 LayoutSize tileSize = positioningAreaSize;
151
152 Length layerWidth = fillLayer.size().size.width;
153 Length layerHeight = fillLayer.size().size.height;
154
155 if (layerWidth.isFixed())
156 tileSize.setWidth(layerWidth.value());
157 else if (layerWidth.isPercentOrCalculated()) {
158 auto resolvedWidth = valueForLength(layerWidth, positioningAreaSize.width());
159 // Non-zero resolved value should always produce some content.
160 tileSize.setWidth(!resolvedWidth ? resolvedWidth : std::max(devicePixelSize, resolvedWidth));
161 }
162
163 if (layerHeight.isFixed())
164 tileSize.setHeight(layerHeight.value());
165 else if (layerHeight.isPercentOrCalculated()) {
166 auto resolvedHeight = valueForLength(layerHeight, positioningAreaSize.height());
167 // Non-zero resolved value should always produce some content.
168 tileSize.setHeight(!resolvedHeight ? resolvedHeight : std::max(devicePixelSize, resolvedHeight));
169 }
170
171 // If one of the values is auto we have to use the appropriate
172 // scale to maintain our aspect ratio.
173 if (layerWidth.isAuto() && !layerHeight.isAuto()) {
174 if (imageIntrinsicSize.height())
175 tileSize.setWidth(imageIntrinsicSize.width() * tileSize.height() / imageIntrinsicSize.height());
176 } else if (!layerWidth.isAuto() && layerHeight.isAuto()) {
177 if (imageIntrinsicSize.width())
178 tileSize.setHeight(imageIntrinsicSize.height() * tileSize.width() / imageIntrinsicSize.width());
179 } else if (layerWidth.isAuto() && layerHeight.isAuto()) {
180 // If both width and height are auto, use the image's intrinsic size.
181 tileSize = imageIntrinsicSize;
182 }
183
184 tileSize.clampNegativeToZero();
185 return tileSize;
186 }
187 case FillSizeType::None: {
188 // If both values are ‘auto’ then the intrinsic width and/or height of the image should be used, if any.
189 if (!imageIntrinsicSize.isEmpty())
190 return imageIntrinsicSize;
191
192 // If the image has neither an intrinsic width nor an intrinsic height, its size is determined as for ‘contain’.
193 type = FillSizeType::Contain;
194 }
195 FALLTHROUGH;
196 case FillSizeType::Contain:
197 case FillSizeType::Cover: {
198 // Scale computation needs higher precision than what LayoutUnit can offer.
199 FloatSize localImageIntrinsicSize = imageIntrinsicSize;
200 FloatSize localPositioningAreaSize = positioningAreaSize;
201
202 float horizontalScaleFactor = localImageIntrinsicSize.width() ? (localPositioningAreaSize.width() / localImageIntrinsicSize.width()) : 1;
203 float verticalScaleFactor = localImageIntrinsicSize.height() ? (localPositioningAreaSize.height() / localImageIntrinsicSize.height()) : 1;
204 float scaleFactor = type == FillSizeType::Contain ? std::min(horizontalScaleFactor, verticalScaleFactor) : std::max(horizontalScaleFactor, verticalScaleFactor);
205
206 if (localImageIntrinsicSize.isEmpty())
207 return { };
208
209 return LayoutSize(localImageIntrinsicSize.scaled(scaleFactor).expandedTo({ devicePixelSize, devicePixelSize }));
210 }
211 }
212
213 ASSERT_NOT_REACHED();
214 return { };
215}

217static inline LayoutUnit getSpace(LayoutUnit areaSize, LayoutUnit tileSize)
218{
219 int numberOfTiles = areaSize / tileSize;
220 LayoutUnit space = -1;
221
222 if (numberOfTiles > 1)
223 space = (areaSize - numberOfTiles * tileSize) / (numberOfTiles - 1);
224
225 return space;
226}

228static LayoutUnit resolveEdgeRelativeLength(const Length& length, Edge edge, LayoutUnit availableSpace, const LayoutSize& areaSize, const LayoutSize& tileSize)
229{
230 LayoutUnit result = minimumValueForLength(length, availableSpace);
231
232 if (edge == Edge::Right)
233 return areaSize.width() - tileSize.width() - result;
234
235 if (edge == Edge::Bottom)
236 return areaSize.height() - tileSize.height() - result;
237
238 return result;
239}

241static FillLayerImageGeometry pixelSnappedFillLayerImageGeometry(LayoutRect& destinationRect, LayoutSize& tileSize, LayoutSize& phase, LayoutSize& space, FillAttachment attachment, float pixelSnappingFactor)
242{
243 return FillLayerImageGeometry {
244 snapRectToDevicePixels(destinationRect, pixelSnappingFactor),
245 snapRectToDevicePixels({ destinationRect.location(), tileSize }, pixelSnappingFactor).size(),
246 snapRectToDevicePixels({ destinationRect.location(), phase }, pixelSnappingFactor).size(),
247 snapRectToDevicePixels({ { }, space }, pixelSnappingFactor).size(),
248 attachment
249 };

252static FillLayerImageGeometry geometryForLayer(const FillLayer& fillLayer, LayoutRect borderBoxRect, const Layout::BoxGeometry& geometry, float pixelSnappingFactor)
253{
254 LayoutUnit left;
255 LayoutUnit top;
256 LayoutSize positioningAreaSize;
257
258 auto destinationRect = borderBoxRect;
259
260 switch (fillLayer.attachment()) {
261 case FillAttachment::ScrollBackground:
262 case FillAttachment::LocalBackground: {
263 LayoutUnit right;
264 LayoutUnit bottom;
265 if (fillLayer.origin() != FillBox::Border) {
266 left = geometry.borderLeft();
267 right = geometry.borderRight();
268 top = geometry.borderTop();
269 bottom = geometry.borderBottom();
270 if (fillLayer.origin() == FillBox::Content) {
271 left += geometry.paddingLeft().value_or(0);
272 right += geometry.paddingRight().value_or(0);
273 top += geometry.paddingTop().value_or(0);
274 bottom += geometry.paddingBottom().value_or(0);
275 }
276 }
277
278 // FIXME: Handle the root element sizing.
279 positioningAreaSize = borderBoxRect.size() - LayoutSize(left + right, top + bottom);
280 break;
281 }
282 case FillAttachment::FixedBackground: {
283 // FIXME: Handle fixed backgrounds.
284 positioningAreaSize = borderBoxRect.size();
285 break;
286 }
287 }
288
289 LayoutSize tileSize = calculateFillTileSize(fillLayer, positioningAreaSize, pixelSnappingFactor);
290
291 FillRepeat backgroundRepeatX = fillLayer.repeatX();
292 FillRepeat backgroundRepeatY = fillLayer.repeatY();
293 LayoutUnit availableWidth = positioningAreaSize.width() - tileSize.width();
294 LayoutUnit availableHeight = positioningAreaSize.height() - tileSize.height();
295
296 LayoutSize spaceSize;
297 LayoutSize phase;
298 LayoutSize noRepeat;
299 LayoutUnit computedXPosition = resolveEdgeRelativeLength(fillLayer.xPosition(), fillLayer.backgroundXOrigin(), availableWidth, positioningAreaSize, tileSize);
300 if (backgroundRepeatX == FillRepeat::Round && positioningAreaSize.width() > 0 && tileSize.width() > 0) {
301 int numTiles = std::max(1, roundToInt(positioningAreaSize.width() / tileSize.width()));
302 if (fillLayer.size().size.height.isAuto() && backgroundRepeatY != FillRepeat::Round)
303 tileSize.setHeight(tileSize.height() * positioningAreaSize.width() / (numTiles * tileSize.width()));
304
305 tileSize.setWidth(positioningAreaSize.width() / numTiles);
306 phase.setWidth(tileSize.width() ? tileSize.width() - fmodf((computedXPosition + left), tileSize.width()) : 0);
307 }
308
309 LayoutUnit computedYPosition = resolveEdgeRelativeLength(fillLayer.yPosition(), fillLayer.backgroundYOrigin(), availableHeight, positioningAreaSize, tileSize);
310 if (backgroundRepeatY == FillRepeat::Round && positioningAreaSize.height() > 0 && tileSize.height() > 0) {
311 int numTiles = std::max(1, roundToInt(positioningAreaSize.height() / tileSize.height()));
312 if (fillLayer.size().size.width.isAuto() && backgroundRepeatX != FillRepeat::Round)
313 tileSize.setWidth(tileSize.width() * positioningAreaSize.height() / (numTiles * tileSize.height()));
314
315 tileSize.setHeight(positioningAreaSize.height() / numTiles);
316 phase.setHeight(tileSize.height() ? tileSize.height() - fmodf((computedYPosition + top), tileSize.height()) : 0);
317 }
318
319 if (backgroundRepeatX == FillRepeat::Repeat) {
320 phase.setWidth(tileSize.width() ? tileSize.width() - fmodf(computedXPosition + left, tileSize.width()) : 0);
321 spaceSize.setWidth(0);
322 } else if (backgroundRepeatX == FillRepeat::Space && tileSize.width() > 0) {
323 LayoutUnit space = getSpace(positioningAreaSize.width(), tileSize.width());
324 if (space >= 0) {
325 LayoutUnit actualWidth = tileSize.width() + space;
326 computedXPosition = minimumValueForLength(Length(), availableWidth);
327 spaceSize.setWidth(space);
328 spaceSize.setHeight(0);
329 phase.setWidth(actualWidth ? actualWidth - fmodf((computedXPosition + left), actualWidth) : 0);
330 } else
331 backgroundRepeatX = FillRepeat::NoRepeat;
332 }
333
334 if (backgroundRepeatX == FillRepeat::NoRepeat) {
335 LayoutUnit xOffset = left + computedXPosition;
336 if (xOffset > 0)
337 destinationRect.move(xOffset, 0_lu);
338 xOffset = std::min<LayoutUnit>(xOffset, 0);
339 phase.setWidth(-xOffset);
340 destinationRect.setWidth(tileSize.width() + xOffset);
341 spaceSize.setWidth(0);
342 }
343
344 if (backgroundRepeatY == FillRepeat::Repeat) {
345 phase.setHeight(tileSize.height() ? tileSize.height() - fmodf(computedYPosition + top, tileSize.height()) : 0);
346 spaceSize.setHeight(0);
347 } else if (backgroundRepeatY == FillRepeat::Space && tileSize.height() > 0) {
348 LayoutUnit space = getSpace(positioningAreaSize.height(), tileSize.height());
349
350 if (space >= 0) {
351 LayoutUnit actualHeight = tileSize.height() + space;
352 computedYPosition = minimumValueForLength(Length(), availableHeight);
353 spaceSize.setHeight(space);
354 phase.setHeight(actualHeight ? actualHeight - fmodf((computedYPosition + top), actualHeight) : 0);
355 } else
356 backgroundRepeatY = FillRepeat::NoRepeat;
357 }
358 if (backgroundRepeatY == FillRepeat::NoRepeat) {
359 LayoutUnit yOffset = top + computedYPosition;
360 if (yOffset > 0)
361 destinationRect.move(0_lu, yOffset);
362 yOffset = std::min<LayoutUnit>(yOffset, 0);
363 phase.setHeight(-yOffset);
364 destinationRect.setHeight(tileSize.height() + yOffset);
365 spaceSize.setHeight(0);
366 }
367
368 if (fillLayer.attachment() == FillAttachment::FixedBackground) {
369 LayoutPoint attachmentPoint = borderBoxRect.location();
370 phase.expand(std::max<LayoutUnit>(attachmentPoint.x() - destinationRect.x(), 0), std::max<LayoutUnit>(attachmentPoint.y() - destinationRect.y(), 0));
371 }
372
373 destinationRect.intersect(borderBoxRect);
374 return pixelSnappedFillLayerImageGeometry(destinationRect, tileSize, phase, spaceSize, fillLayer.attachment(), pixelSnappingFactor);
375}

377Vector<FillLayerImageGeometry, 1> calculateFillLayerImageGeometry(const RenderStyle& renderStyle, const Layout::BoxGeometry& boxGeometry, LayoutSize offsetFromRoot, float pixelSnappingFactor)
378{
379 // FIXME: Need to map logical to physical rects.
380 auto borderBoxRect = LayoutRect { Layout::BoxGeometry::borderBoxRect(boxGeometry) };
381 borderBoxRect.move(offsetFromRoot);
382
383 Vector<FillLayerImageGeometry, 1> backgroundGeometry;
384
385 for (auto fillLayer = &renderStyle.backgroundLayers(); fillLayer; fillLayer = fillLayer->next())
386 backgroundGeometry.append(geometryForLayer(*fillLayer, borderBoxRect, boxGeometry, pixelSnappingFactor));
387
388 return backgroundGeometry;
389}
