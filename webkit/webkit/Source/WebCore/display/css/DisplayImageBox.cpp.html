39ImageBox::ImageBox(Tree& tree, AbsoluteFloatRect borderBox, Style&& displayStyle, RefPtr<Image>&& image)
40 : ReplacedBox(tree, borderBox, WTFMove(displayStyle), { TypeFlags::ImageBox })
41 , m_image(WTFMove(image))
42{
43}
44
45const char* ImageBox::boxName() const
46{
47 return "image box";
48}
49
50String ImageBox::debugDescription() const
51{
52 TextStream stream;
53 stream << boxName() << " " << absoluteBorderBoxRect() << " (" << this << ") replaced content rect: " << replacedContentRect() << " image: " << m_image.get();
54 return stream.release();
55}
