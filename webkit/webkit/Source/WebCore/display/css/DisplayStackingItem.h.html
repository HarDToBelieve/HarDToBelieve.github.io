35namespace WebCore {
36namespace Display {
37
38class BoxModelBox;
39
40// Container for a display box and its descendants, for boxes that participate in the CSS stacking
41// algorithm <https://www.w3.org/TR/CSS22/zindex.html>, i.e. positioned boxes, and those that create
42// CSS stacking context.
43
44class StackingItem {
45 WTF_MAKE_FAST_ALLOCATED;
46 friend class TreeBuilder;
47public:
48 using StackingItemList = Vector<std::unique_ptr<StackingItem>>;
49
50 explicit StackingItem(std::unique_ptr<BoxModelBox>&&);
51
52 void addChildStackingItem(std::unique_ptr<StackingItem>&&);
53 const BoxModelBox& box() const { return *m_box; }
54
55 bool isStackingContext() const;
56
57 // Bounds for the painted content of this stacking item (excluding descendant items), not taking clips from ancestors into account.
58 AbsoluteFloatRect paintedContentBounds() const { return m_paintedContentBounds; }
59 // Bounds for the painted content of this stacking item and descendant stacking items.
60 AbsoluteFloatRect paintedBoundsIncludingDescendantItems() const { return m_paintedBoundsIncludingDescendantItems; }
61
62 const StackingItemList& negativeZOrderList() const { return m_negativeZOrderList; }
63 const StackingItemList& positiveZOrderList() const { return m_positiveZOrderList; }
64
65private:
66 void sortLists();
67
68 void setPaintedContentBounds(const AbsoluteFloatRect& bounds) { m_paintedContentBounds = bounds; }
69 void setPaintedBoundsIncludingDescendantItems(const AbsoluteFloatRect& bounds) { m_paintedBoundsIncludingDescendantItems = bounds; }
70
71 std::unique_ptr<BoxModelBox> m_box;
72 StackingItemList m_negativeZOrderList;
73 StackingItemList m_positiveZOrderList;
74
75 AbsoluteFloatRect m_paintedContentBounds;
76 AbsoluteFloatRect m_paintedBoundsIncludingDescendantItems;
77};

79} // namespace Display
