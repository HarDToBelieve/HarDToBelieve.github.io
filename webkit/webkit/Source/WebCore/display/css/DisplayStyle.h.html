40namespace WebCore {

42class FillLayer;
43class RenderStyle;
44class ShadowData;

46namespace Display {
47
48// Style information needed to paint a Display::Box.
49// All colors should be resolved to their painted values [visitedDependentColorWithColorFilter()].
50// Should contain only absolute float values; no Lengths (which can contain calc values).
51// Should be sharable between boxes with different geometry but the same style.
52
53DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(Style);
54class Style {
55 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(Style);
56public:
57
58 enum class Flags : uint8_t {
59 Positioned = 1 << 0,
60 Floating = 1 << 1,
61 HasTransform = 1 << 2,
62 };
63
64 explicit Style(const RenderStyle&);
65 explicit Style(const RenderStyle&, const RenderStyle* styleForBackground);
66
67 const Color& color() const { return m_color; }
68
69 const Color& backgroundColor() const { return m_backgroundColor; }
70 bool hasBackground() const;
71 bool hasBackgroundImage() const;
72
73 const FillLayer* backgroundLayers() const { return m_backgroundLayers.get(); }
74 bool backgroundHasOpaqueTopLayer() const;
75
76 const ShadowData* boxShadow() const { return m_boxShadow.get(); }
77
78 std::optional<int> zIndex() const { return m_zIndex; }
79 bool isStackingContext() const { return m_zIndex.has_value(); }
80
81 bool isPositioned() const { return m_flags.contains(Flags::Positioned); }
82 bool isFloating() const { return m_flags.contains(Flags::Floating); }
83
84 // Just the transform property (not translate, rotate, scale).
85 bool hasTransform() const { return m_flags.contains(Flags::HasTransform); }
86
87 bool participatesInZOrderSorting() const { return isPositioned() || isStackingContext(); }
88
89 const FontCascade& fontCascade() const { return m_fontCascade; }
90 const FontMetrics& fontMetrics() const { return m_fontCascade.fontMetrics(); }
91
92 float opacity() const { return m_opacity; }
93
94 Overflow overflowX() const;
95 Overflow overflowY() const;
96 bool hasClippedOverflow() const { return m_overflowX != Overflow::Visible || m_overflowY != Overflow::Visible; }
97
98 WhiteSpace whiteSpace() const { return m_whiteSpace; }
99 bool autoWrap() const;
100 bool preserveNewline() const;
101 bool collapseWhiteSpace() const;
102
103 const TabSize& tabSize() const { return m_tabSize; }
104
105private:
106 void setupBackground(const RenderStyle&);
107
108 void setIsPositioned(bool value) { m_flags.set({ Flags::Positioned }, value); }
109 void setIsFloating(bool value) { m_flags.set({ Flags::Floating }, value); }
110 void setHasTransform(bool value) { m_flags.set({ Flags::HasTransform }, value); }
111
112 Color m_color;
113 Color m_backgroundColor;
114
115 RefPtr<FillLayer> m_backgroundLayers;
116 std::unique_ptr<ShadowData> m_boxShadow;
117
118 Overflow m_overflowX;
119 Overflow m_overflowY;
120
121 FontCascade m_fontCascade;
122 WhiteSpace m_whiteSpace;
123 TabSize m_tabSize;
124
125 float m_opacity;
126
127 std::optional<int> m_zIndex;
128 OptionSet<Flags> m_flags;
129};
130
131} // namespace Display
132} // namespace WebCore
