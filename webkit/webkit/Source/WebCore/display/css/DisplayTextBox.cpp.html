46const char* TextBox::boxName() const
47{
48 return "text box";
49}
50
51String TextBox::debugDescription() const
52{
53 TextStream stream;
54
55 stream << boxName() << " " << absoluteBoxRect() << " (" << this << ")";
56 auto textContent = text().content().substring(text().start(), text().length());
57 textContent.replaceWithLiteral('\\', "\\\\");
58 textContent.replaceWithLiteral('\n', "\\n");
59 const size_t maxPrintedLength = 80;
60 if (textContent.length() > maxPrintedLength) {
61 auto substring = textContent.substring(0, maxPrintedLength);
62 stream << " \"" << substring << "\"…";
63 } else
64 stream << " \"" << textContent << "\"";
65
66 return stream.release();
67}
