26#pragma once
27
28#include "AnimationEventBase.h"
29
30namespace WebCore {
31
32class AnimationEvent final : public AnimationEventBase {
33 WTF_MAKE_ISO_ALLOCATED(AnimationEvent);
34public:
35 static Ref<AnimationEvent> create(const AtomString& type, const String& animationName, double elapsedTime, const String& pseudoElement, std::optional<Seconds> timelineTime, WebAnimation* animation)
36 {
37 return adoptRef(*new AnimationEvent(type, animationName, elapsedTime, pseudoElement, timelineTime, animation));
38 }
39
40 struct Init : EventInit {
41 String animationName;
42 double elapsedTime { 0 };
43 String pseudoElement;
44 };
45
46 static Ref<AnimationEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
47 {
48 return adoptRef(*new AnimationEvent(type, initializer, isTrusted));
49 }
50
51 virtual ~AnimationEvent();
52
53 bool isAnimationEvent() const final { return true; }
54
55 const String& animationName() const;
56 double elapsedTime() const;
57 const String& pseudoElement() const;
58
59 EventInterface eventInterface() const override;
60
61private:
62 AnimationEvent(const AtomString& type, const String& animationName, double elapsedTime, const String& pseudoElement, std::optional<Seconds> timelineTime, WebAnimation*);
63 AnimationEvent(const AtomString&, const Init&, IsTrusted);
64
65 String m_animationName;
66 double m_elapsedTime;
67 String m_pseudoElement;
68};
69
70} // namespace WebCore
