26#pragma once
27
28#include "Event.h"
29
30namespace WebCore {
31
32class BeforeTextInsertedEvent final : public Event {
33 WTF_MAKE_ISO_ALLOCATED(BeforeTextInsertedEvent);
34public:
35 virtual ~BeforeTextInsertedEvent();
36
37 static Ref<BeforeTextInsertedEvent> create(const String& text)
38 {
39 return adoptRef(*new BeforeTextInsertedEvent(text));
40 }
41
42 EventInterface eventInterface() const override;
43
44 const String& text() const { return m_text; }
45 void setText(const String& s) { m_text = s; }
46
47private:
48 explicit BeforeTextInsertedEvent(const String&);
49 bool isBeforeTextInsertedEvent() const override { return true; }
50
51 String m_text;
52};
53
54} // namespace WebCore
