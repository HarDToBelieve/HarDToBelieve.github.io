36std::optional<BoundaryPoint> makeBoundaryPointBeforeNode(Node& node)
37{
38 auto parent = node.parentNode();
39 if (!parent)
40 return std::nullopt;
41 return BoundaryPoint { *parent, node.computeNodeIndex() };
42}
43
44std::optional<BoundaryPoint> makeBoundaryPointAfterNode(Node& node)
45{
46 auto parent = node.parentNode();
47 if (!parent)
48 return std::nullopt;
49 return BoundaryPoint { *parent, node.computeNodeIndex() + 1 };
50}

52static bool isOffsetBeforeChild(ContainerNode& container, unsigned offset, Node& child)
53{
54 if (!offset)
55 return true;
56 // If the container is not the parent, the child is part of a shadow tree, which we sort between offset 0 and offset 1.
57 if (child.parentNode() != &container)
58 return false;
59 unsigned currentOffset = 0;
60 for (auto currentChild = container.firstChild(); currentChild && currentChild != &child; currentChild = currentChild->nextSibling()) {
61 if (offset <= ++currentOffset)
62 return true;
63 }
64 return false;
65}

67// FIXME: Once we move to C++20, replace with the C++20 <=> operator.
68// FIXME: This could return std::strong_ordering if we had that, or the equivalent.
69static PartialOrdering order(unsigned a, unsigned b)
70{
71 if (a < b)
72 return PartialOrdering::less;
73 if (a > b)
74 return PartialOrdering::greater;
75 return PartialOrdering::equivalent;
76}

78template<TreeType treeType> PartialOrdering treeOrder(const BoundaryPoint& a, const BoundaryPoint& b)
79{
80 if (a.container.ptr() == b.container.ptr())
81 return order(a.offset, b.offset);
82
83 for (auto ancestor = b.container.ptr(); ancestor; ) {
84 auto nextAncestor = parent<treeType>(*ancestor);
85 if (nextAncestor == a.container.ptr())
86 return isOffsetBeforeChild(*nextAncestor, a.offset, *ancestor) ? PartialOrdering::less : PartialOrdering::greater;
87 ancestor = nextAncestor;
88 }
89
90 for (auto ancestor = a.container.ptr(); ancestor; ) {
91 auto nextAncestor = parent<treeType>(*ancestor);
92 if (nextAncestor == b.container.ptr())
93 return isOffsetBeforeChild(*nextAncestor, b.offset, *ancestor) ? PartialOrdering::greater : PartialOrdering::less;
94 ancestor = nextAncestor;
95 }
96
97 return treeOrder<treeType>(a.container, b.container);
98}

100PartialOrdering treeOrderForTesting(TreeType type, const BoundaryPoint& a, const BoundaryPoint& b)
101{
102 switch (type) {
103 case Tree:
104 return treeOrder<Tree>(a, b);
105 case ShadowIncludingTree:
106 return treeOrder<ShadowIncludingTree>(a, b);
107 case ComposedTree:
108 return treeOrder<ComposedTree>(a, b);
109 }
110 ASSERT_NOT_REACHED();
111 return PartialOrdering::unordered;
112}
