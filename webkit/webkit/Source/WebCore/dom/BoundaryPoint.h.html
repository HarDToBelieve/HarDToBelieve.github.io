26#pragma once
27
28#include "Node.h"
29
30namespace WebCore {
31
32struct BoundaryPoint {
33 Ref<Node> container;
34 unsigned offset { 0 };
35
36 BoundaryPoint(Ref<Node>&&, unsigned);
37
38 Document& document() const;
39};
40
41bool operator==(const BoundaryPoint&, const BoundaryPoint&);
42bool operator!=(const BoundaryPoint&, const BoundaryPoint&);
43
44template<TreeType = Tree> PartialOrdering treeOrder(const BoundaryPoint&, const BoundaryPoint&);
45
46WEBCORE_EXPORT std::optional<BoundaryPoint> makeBoundaryPointBeforeNode(Node&);
47WEBCORE_EXPORT std::optional<BoundaryPoint> makeBoundaryPointAfterNode(Node&);
48BoundaryPoint makeBoundaryPointBeforeNodeContents(Node&);
49BoundaryPoint makeBoundaryPointAfterNodeContents(Node&);
50
51WEBCORE_EXPORT PartialOrdering treeOrderForTesting(TreeType, const BoundaryPoint&, const BoundaryPoint&);
52
53inline BoundaryPoint::BoundaryPoint(Ref<Node>&& container, unsigned offset)
54 : container(WTFMove(container))
55 , offset(offset)
56{
57}
58
59inline Document& BoundaryPoint::document() const
60{
61 return container->document();
62}
63
64inline bool operator==(const BoundaryPoint& a, const BoundaryPoint& b)
65{
66 return a.container.ptr() == b.container.ptr() && a.offset == b.offset;
67}
68
69inline BoundaryPoint makeBoundaryPointBeforeNodeContents(Node& node)
70{
71 return { node, 0 };
72}
73
74inline BoundaryPoint makeBoundaryPointAfterNodeContents(Node& node)
75{
76 return { node, node.length() };
77}
