23#include "CDATASection.h"
24
25#include "Document.h"

28namespace WebCore {
29
30WTF_MAKE_ISO_ALLOCATED_IMPL(CDATASection);
31
32inline CDATASection::CDATASection(Document& document, const String& data)
33 : Text(document, data, CreateText)
34{
35}
36
37Ref<CDATASection> CDATASection::create(Document& document, const String& data)
38{
39 return adoptRef(*new CDATASection(document, data));
40}
41
42String CDATASection::nodeName() const
43{
44 return "#cdata-section"_s;
45}
46
47Node::NodeType CDATASection::nodeType() const
48{
49 return CDATA_SECTION_NODE;
50}
51
52Ref<Node> CDATASection::cloneNodeInternal(Document& targetDocument, CloningOperation)
53{
54 return create(targetDocument, data());
55}
56
57bool CDATASection::childTypeAllowed(NodeType) const
58{
59 return false;
60}
61
62Ref<Text> CDATASection::virtualCreate(const String& data)
63{
64 return create(document(), data);
65}
