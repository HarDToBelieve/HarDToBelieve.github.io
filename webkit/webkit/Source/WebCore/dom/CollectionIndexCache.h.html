32WEBCORE_EXPORT void reportExtraMemoryAllocatedForCollectionIndexCache(size_t);
33
34template <class Collection, class Iterator>
35class CollectionIndexCache {
36public:
37 CollectionIndexCache();
38
39 typedef typename std::iterator_traits<Iterator>::value_type NodeType;
40
41 unsigned nodeCount(const Collection&);
42 NodeType* nodeAt(const Collection&, unsigned index);
43
44 bool hasValidCache() const { return m_current || m_nodeCountValid || m_listValid; }
45 void invalidate();
46 size_t memoryCost()
47 {
48 // memoryCost() may be invoked concurrently from a GC thread, and we need to be careful
49 // about what data we access here and how. Accessing m_cachedList.capacity() is safe
50 // because it doesn't involve any pointer chasing.
51 return m_cachedList.capacity() * sizeof(NodeType*);
52 }
53
54private:
55 unsigned computeNodeCountUpdatingListCache(const Collection&);
56 NodeType* traverseBackwardTo(const Collection&, unsigned);
57 NodeType* traverseForwardTo(const Collection&, unsigned);
58
59 Iterator m_current { };
60 unsigned m_currentIndex { 0 };
61 unsigned m_nodeCount { 0 };
62 Vector<NodeType*> m_cachedList;
63 bool m_nodeCountValid : 1;
64 bool m_listValid : 1;
65};
66
67template<class Collection, class Iterator> CollectionIndexCache<Collection, Iterator>::CollectionIndexCache()
68 : m_nodeCountValid(false)
69 , m_listValid(false)
70{
71}
72
73template <class Collection, class Iterator>
74inline unsigned CollectionIndexCache<Collection, Iterator>::nodeCount(const Collection& collection)
75{
76 if (!m_nodeCountValid) {
77 if (!hasValidCache())
78 collection.willValidateIndexCache();
79 m_nodeCount = computeNodeCountUpdatingListCache(collection);
80 m_nodeCountValid = true;
81 }
82
83 return m_nodeCount;
84}
85
86template <class Collection, class Iterator>
87unsigned CollectionIndexCache<Collection, Iterator>::computeNodeCountUpdatingListCache(const Collection& collection)
88{
89 auto current = collection.collectionBegin();
90 if (!current)
91 return 0;
92
93 unsigned oldCapacity = m_cachedList.capacity();
94 while (current) {
95 m_cachedList.append(&*current);
96 unsigned traversed;
97 collection.collectionTraverseForward(current, 1, traversed);
98 ASSERT(traversed == (current ? 1 : 0));
99 }
100 m_listValid = true;
101
102 if (unsigned capacityDifference = m_cachedList.capacity() - oldCapacity)
103 reportExtraMemoryAllocatedForCollectionIndexCache(capacityDifference * sizeof(NodeType*));
104
105 return m_cachedList.size();
106}

108template <class Collection, class Iterator>
109inline typename CollectionIndexCache<Collection, Iterator>::NodeType* CollectionIndexCache<Collection, Iterator>::traverseBackwardTo(const Collection& collection, unsigned index)
110{
111 ASSERT(m_current);
112 ASSERT(index < m_currentIndex);
113
114 bool firstIsCloser = index < m_currentIndex - index;
115 if (firstIsCloser || !collection.collectionCanTraverseBackward()) {
116 m_current = collection.collectionBegin();
117 m_currentIndex = 0;
118 if (index)
119 collection.collectionTraverseForward(m_current, index, m_currentIndex);
120 ASSERT(m_current);
121 return &*m_current;
122 }
123
124 collection.collectionTraverseBackward(m_current, m_currentIndex - index);
125 m_currentIndex = index;
126
127 ASSERT(m_current);
128 return &*m_current;
129}

131template <class Collection, class Iterator>
132inline typename CollectionIndexCache<Collection, Iterator>::NodeType* CollectionIndexCache<Collection, Iterator>::traverseForwardTo(const Collection& collection, unsigned index)
133{
134 ASSERT(m_current);
135 ASSERT(index > m_currentIndex);
136 ASSERT(!m_nodeCountValid || index < m_nodeCount);
137
138 bool lastIsCloser = m_nodeCountValid && m_nodeCount - index < index - m_currentIndex;
139 if (lastIsCloser && collection.collectionCanTraverseBackward()) {
140 ASSERT(hasValidCache());
141 m_current = collection.collectionLast();
142 if (index < m_nodeCount - 1)
143 collection.collectionTraverseBackward(m_current, m_nodeCount - index - 1);
144 m_currentIndex = index;
145 ASSERT(m_current);
146 return &*m_current;
147 }
148
149 if (!hasValidCache())
150 collection.willValidateIndexCache();
151
152 unsigned traversedCount;
153 collection.collectionTraverseForward(m_current, index - m_currentIndex, traversedCount);
154 m_currentIndex = m_currentIndex + traversedCount;
155
156 if (!m_current) {
157 ASSERT(m_currentIndex < index);
158 // Failed to find the index but at least we now know the size.
159 m_nodeCount = m_currentIndex + 1;
160 m_nodeCountValid = true;
161 return nullptr;
162 }
163 ASSERT(hasValidCache());
164 return &*m_current;
165}

167template <class Collection, class Iterator>
168inline typename CollectionIndexCache<Collection, Iterator>::NodeType* CollectionIndexCache<Collection, Iterator>::nodeAt(const Collection& collection, unsigned index)
169{
170 if (m_nodeCountValid && index >= m_nodeCount)
171 return nullptr;
172
173 if (m_listValid)
174 return m_cachedList[index];
175
176 if (m_current) {
177 if (index > m_currentIndex)
178 return traverseForwardTo(collection, index);
179 if (index < m_currentIndex)
180 return traverseBackwardTo(collection, index);
181 return &*m_current;
182 }
183
184 bool lastIsCloser = m_nodeCountValid && m_nodeCount - index < index;
185 if (lastIsCloser && collection.collectionCanTraverseBackward()) {
186 ASSERT(hasValidCache());
187 m_current = collection.collectionLast();
188 if (index < m_nodeCount - 1)
189 collection.collectionTraverseBackward(m_current, m_nodeCount - index - 1);
190 m_currentIndex = index;
191 ASSERT(m_current);
192 return &*m_current;
193 }
194
195 if (!hasValidCache())
196 collection.willValidateIndexCache();
197
198 m_current = collection.collectionBegin();
199 m_currentIndex = 0;
200 bool startIsEnd = !m_current;
201 if (index && m_current) {
202 collection.collectionTraverseForward(m_current, index, m_currentIndex);
203 ASSERT(m_current || m_currentIndex < index);
204 }
205 if (!m_current) {
206 // Failed to find the index but at least we now know the size.
207 m_nodeCount = startIsEnd ? 0 : m_currentIndex + 1;
208 m_nodeCountValid = true;
209 return nullptr;
210 }
211 ASSERT(hasValidCache());
212 return &*m_current;
213}

215template <class Collection, class Iterator>
216void CollectionIndexCache<Collection, Iterator>::invalidate()
217{
218 m_current = { };
219 m_nodeCountValid = false;
220 m_listValid = false;
221 m_cachedList.shrink(0);
222}
