29#include "UIEvent.h"

31namespace WebCore {
32
33class CompositionEvent final : public UIEvent {
34 WTF_MAKE_ISO_ALLOCATED(CompositionEvent);
35public:
36 static Ref<CompositionEvent> create(const AtomString& type, RefPtr<WindowProxy>&& view, const String& data)
37 {
38 return adoptRef(*new CompositionEvent(type, WTFMove(view), data));
39 }
40
41 static Ref<CompositionEvent> createForBindings()
42 {
43 return adoptRef(*new CompositionEvent);
44 }
45
46 struct Init : UIEventInit {
47 String data;
48 };
49
50 static Ref<CompositionEvent> create(const AtomString& type, const Init& initializer)
51 {
52 return adoptRef(*new CompositionEvent(type, initializer));
53 }
54
55 virtual ~CompositionEvent();
56
57 void initCompositionEvent(const AtomString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&&, const String& data);
58
59 String data() const { return m_data; }
60
61 EventInterface eventInterface() const override;
62
63private:
64 CompositionEvent();
65 CompositionEvent(const AtomString& type, RefPtr<WindowProxy>&&, const String&);
66 CompositionEvent(const AtomString& type, const Init&);
67
68 bool isCompositionEvent() const override;
69
70 String m_data;
71};

73} // namespace WebCore
