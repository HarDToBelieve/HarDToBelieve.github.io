35namespace WebCore {
36
37class CSSCustomPropertyValue;
38class CSSVariableData;
39class Document;
40
41enum class ConstantProperty {
42 SafeAreaInsetTop,
43 SafeAreaInsetRight,
44 SafeAreaInsetBottom,
45 SafeAreaInsetLeft,
46 FullscreenInsetTop,
47 FullscreenInsetRight,
48 FullscreenInsetBottom,
49 FullscreenInsetLeft,
50 FullscreenAutoHideDuration,
51};
52
53class ConstantPropertyMap {
54 WTF_MAKE_FAST_ALLOCATED;
55public:
56 explicit ConstantPropertyMap(Document&);
57
58 typedef HashMap<AtomString, Ref<CSSCustomPropertyValue>> Values;
59 const Values& values() const;
60
61 void didChangeSafeAreaInsets();
62 void didChangeFullscreenInsets();
63 void setFullscreenAutoHideDuration(Seconds);
64
65private:
66 void buildValues();
67
68 const AtomString& nameForProperty(ConstantProperty) const;
69 void setValueForProperty(ConstantProperty, Ref<CSSVariableData>&&);
70
71 void updateConstantsForSafeAreaInsets();
72 void updateConstantsForFullscreen();
73
74 std::optional<Values> m_values;
75
76 Document& m_document;
77};
