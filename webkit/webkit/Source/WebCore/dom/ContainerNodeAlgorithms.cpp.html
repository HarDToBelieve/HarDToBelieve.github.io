42enum class TreeScopeChange { Changed, DidNotChange };
43
44static void notifyNodeInsertedIntoDocument(ContainerNode& parentOfInsertedTree, Node& node, TreeScopeChange treeScopeChange, NodeVector& postInsertionNotificationTargets)
45{
46 ASSERT(parentOfInsertedTree.isConnected());
47 ASSERT(!node.isConnected());
48 if (node.insertedIntoAncestor(Node::InsertionType { /* connectedToDocument */ true, treeScopeChange == TreeScopeChange::Changed }, parentOfInsertedTree) == Node::InsertedIntoAncestorResult::NeedsPostInsertionCallback)
49 postInsertionNotificationTargets.append(node);
50
51 if (!is<ContainerNode>(node))
52 return;
53
54 for (RefPtr<Node> child = downcast<ContainerNode>(node).firstChild(); child; child = child->nextSibling()) {
55 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(node.isConnected() && child->parentNode() == &node);
56 notifyNodeInsertedIntoDocument(parentOfInsertedTree, *child, treeScopeChange, postInsertionNotificationTargets);
57 }
58
59 if (!is<Element>(node))
60 return;
61
62 if (RefPtr<ShadowRoot> root = downcast<Element>(node).shadowRoot()) {
63 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(node.isConnected() && root->host() == &node);
64 notifyNodeInsertedIntoDocument(parentOfInsertedTree, *root, TreeScopeChange::DidNotChange, postInsertionNotificationTargets);
65 }
66}

68static void notifyNodeInsertedIntoTree(ContainerNode& parentOfInsertedTree, Node& node, TreeScopeChange treeScopeChange, NodeVector& postInsertionNotificationTargets)
69{
70 ASSERT(!parentOfInsertedTree.isConnected());
71 ASSERT(!node.isConnected());
72
73 if (node.insertedIntoAncestor(Node::InsertionType { /* connectedToDocument */ false, treeScopeChange == TreeScopeChange::Changed }, parentOfInsertedTree) == Node::InsertedIntoAncestorResult::NeedsPostInsertionCallback)
74 postInsertionNotificationTargets.append(node);
75
76 if (!is<ContainerNode>(node))
77 return;
78
79 for (RefPtr<Node> child = downcast<ContainerNode>(node).firstChild(); child; child = child->nextSibling())
80 notifyNodeInsertedIntoTree(parentOfInsertedTree, *child, treeScopeChange, postInsertionNotificationTargets);
81
82 if (!is<Element>(node))
83 return;
84
85 if (RefPtr<ShadowRoot> root = downcast<Element>(node).shadowRoot())
86 notifyNodeInsertedIntoTree(parentOfInsertedTree, *root, TreeScopeChange::DidNotChange, postInsertionNotificationTargets);
87}

89NodeVector notifyChildNodeInserted(ContainerNode& parentOfInsertedTree, Node& node)
90{
91 ASSERT(ScriptDisallowedScope::InMainThread::hasDisallowedScope());
92
93 InspectorInstrumentation::didInsertDOMNode(node.document(), node);
94
95 Ref<Document> protectDocument(node.document());
96 Ref<Node> protectNode(node);
97
98 NodeVector postInsertionNotificationTargets;
99
100 // Tree scope has changed if the container node into which "node" is inserted is in a document or a shadow root.
101 auto treeScopeChange = parentOfInsertedTree.isInTreeScope() ? TreeScopeChange::Changed : TreeScopeChange::DidNotChange;
102 if (parentOfInsertedTree.isConnected())
103 notifyNodeInsertedIntoDocument(parentOfInsertedTree, node, treeScopeChange, postInsertionNotificationTargets);
104 else
105 notifyNodeInsertedIntoTree(parentOfInsertedTree, node, treeScopeChange, postInsertionNotificationTargets);
106
107 return postInsertionNotificationTargets;
108}

110inline RemovedSubtreeObservability observabilityOfRemovedNode(Node& node)
111{
112 bool isRootOfRemovedTree = !node.parentNode();
113 return node.refCount() > 1 && !isRootOfRemovedTree ? RemovedSubtreeObservability::MaybeObservableByRefPtr : RemovedSubtreeObservability::NotObservable;
114}

116inline void updateObservability(RemovedSubtreeObservability& currentObservability, RemovedSubtreeObservability newStatus)
117{
118 if (newStatus == RemovedSubtreeObservability::MaybeObservableByRefPtr)
119 currentObservability = newStatus;
120}

122static RemovedSubtreeObservability notifyNodeRemovedFromDocument(ContainerNode& oldParentOfRemovedTree, TreeScopeChange treeScopeChange, Node& node)
123{
124 ASSERT(oldParentOfRemovedTree.isConnected());
125 ASSERT(node.isConnected());
126 node.removedFromAncestor(Node::RemovalType { /* disconnectedFromDocument */ true, treeScopeChange == TreeScopeChange::Changed }, oldParentOfRemovedTree);
127
128 auto observability = observabilityOfRemovedNode(node);
129 if (!is<ContainerNode>(node))
130 return observability;
131
132 for (RefPtr<Node> child = downcast<ContainerNode>(node).firstChild(); child; child = child->nextSibling()) {
133 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!node.isConnected() && child->parentNode() == &node);
134 updateObservability(observability, notifyNodeRemovedFromDocument(oldParentOfRemovedTree, treeScopeChange, *child.get()));
135 }
136
137 if (!is<Element>(node))
138 return observability;
139
140 if (RefPtr<ShadowRoot> root = downcast<Element>(node).shadowRoot()) {
141 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!node.isConnected() && root->host() == &node);
142 updateObservability(observability, notifyNodeRemovedFromDocument(oldParentOfRemovedTree, TreeScopeChange::DidNotChange, *root.get()));
143 }
144 return observability;
145}

147static RemovedSubtreeObservability notifyNodeRemovedFromTree(ContainerNode& oldParentOfRemovedTree, TreeScopeChange treeScopeChange, Node& node)
148{
149 ASSERT(!oldParentOfRemovedTree.isConnected());
150
151 node.removedFromAncestor(Node::RemovalType { /* disconnectedFromDocument */ false, treeScopeChange == TreeScopeChange::Changed }, oldParentOfRemovedTree);
152
153 auto observability = observabilityOfRemovedNode(node);
154 if (!is<ContainerNode>(node))
155 return observability;
156
157 for (RefPtr<Node> child = downcast<ContainerNode>(node).firstChild(); child; child = child->nextSibling())
158 updateObservability(observability, notifyNodeRemovedFromTree(oldParentOfRemovedTree, treeScopeChange, *child));
159
160 if (!is<Element>(node))
161 return observability;
162
163 if (RefPtr<ShadowRoot> root = downcast<Element>(node).shadowRoot())
164 updateObservability(observability, notifyNodeRemovedFromTree(oldParentOfRemovedTree, TreeScopeChange::DidNotChange, *root));
165
166 return observability;
167}

169RemovedSubtreeObservability notifyChildNodeRemoved(ContainerNode& oldParentOfRemovedTree, Node& child)
170{
171 // Assert that the caller of this function has an instance of ScriptDisallowedScope.
172 ASSERT(!isMainThread() || ScriptDisallowedScope::InMainThread::hasDisallowedScope());
173 ContainerChildRemovalScope removalScope(oldParentOfRemovedTree, child);
174
175 // Tree scope has changed if the container node from which "node" is removed is in a document or a shadow root.
176 auto treeScopeChange = oldParentOfRemovedTree.isInTreeScope() ? TreeScopeChange::Changed : TreeScopeChange::DidNotChange;
177 if (child.isConnected())
178 return notifyNodeRemovedFromDocument(oldParentOfRemovedTree, treeScopeChange, child);
179 return notifyNodeRemovedFromTree(oldParentOfRemovedTree, treeScopeChange, child);
180}

182void addChildNodesToDeletionQueue(Node*& head, Node*& tail, ContainerNode& container)
183{
184 // We have to tell all children that their parent has died.
185 RefPtr<Node> next = nullptr;
186 for (RefPtr<Node> node = container.firstChild(); node; node = next) {
187 ASSERT(!node->m_deletionHasBegun);
188
189 next = node->nextSibling();
190 node->setNextSibling(nullptr);
191 node->setParentNode(nullptr);
192 container.setFirstChild(next.get());
193 if (next)
194 next->setPreviousSibling(nullptr);
195
196 if (!node->refCount()) {
197#ifndef NDEBUG
198 node->m_deletionHasBegun = true;
199#endif
200 // Add the node to the list of nodes to be deleted.
201 // Reuse the nextSibling pointer for this purpose.
202 if (tail)
203 tail->setNextSibling(node.get());
204 else
205 head = node.get();
206
207 tail = node.get();
208 } else {
209 node->setTreeScopeRecursively(container.document());
210 if (node->isInTreeScope())
211 notifyChildNodeRemoved(container, *node);
212 ASSERT_WITH_SECURITY_IMPLICATION(!node->isInTreeScope());
213 }
214 }
215
216 container.setLastChild(nullptr);
217}

219void removeDetachedChildrenInContainer(ContainerNode& container)
220{
221 // List of nodes to be deleted.
222 Node* head = nullptr;
223 Node* tail = nullptr;
224
225 addChildNodesToDeletionQueue(head, tail, container);
226
227 Node* node;
228 Node* next;
229 while ((node = head)) {
230 ASSERT(node->m_deletionHasBegun);
231
232 next = node->nextSibling();
233 node->setNextSibling(nullptr);
234
235 head = next;
236 if (!next)
237 tail = nullptr;
238
239 if (is<ContainerNode>(*node))
240 addChildNodesToDeletionQueue(head, tail, downcast<ContainerNode>(*node));
241
242 delete node;
243 }
244}

246#ifndef NDEBUG
247static unsigned assertConnectedSubrameCountIsConsistent(ContainerNode& node)
248{
249 unsigned count = 0;
250
251 if (is<Element>(node)) {
252 if (is<HTMLFrameOwnerElement>(node) && downcast<HTMLFrameOwnerElement>(node).contentFrame())
253 ++count;
254
255 if (ShadowRoot* root = downcast<Element>(node).shadowRoot())
256 count += assertConnectedSubrameCountIsConsistent(*root);
257 }
258
259 for (auto& child : childrenOfType<Element>(node))
260 count += assertConnectedSubrameCountIsConsistent(child);
261
262 // If we undercount there's possibly a security bug since we'd leave frames
263 // in subtrees outside the document.
264 ASSERT(node.connectedSubframeCount() >= count);
265
266 // If we overcount it's safe, but not optimal because it means we'll traverse
267 // through the document in disconnectSubframes looking for frames that have
268 // already been disconnected.
269 ASSERT(node.connectedSubframeCount() == count);
270
271 return count;
272}
273#endif

275static void collectFrameOwners(Vector<Ref<HTMLFrameOwnerElement>>& frameOwners, ContainerNode& root)
276{
277 auto elementDescendants = descendantsOfType<Element>(root);
278 auto it = elementDescendants.begin();
279 auto end = elementDescendants.end();
280 while (it != end) {
281 Element& element = *it;
282 if (!element.connectedSubframeCount()) {
283 it.traverseNextSkippingChildren();
284 continue;
285 }
286
287 if (is<HTMLFrameOwnerElement>(element))
288 frameOwners.append(downcast<HTMLFrameOwnerElement>(element));
289
290 if (ShadowRoot* shadowRoot = element.shadowRoot())
291 collectFrameOwners(frameOwners, *shadowRoot);
292 ++it;
293 }
294}

296void disconnectSubframes(ContainerNode& root, SubframeDisconnectPolicy policy)
297{
298#ifndef NDEBUG
299 assertConnectedSubrameCountIsConsistent(root);
300#endif
301 ASSERT(root.connectedSubframeCount());
302
303 Vector<Ref<HTMLFrameOwnerElement>> frameOwners;
304
305 if (policy == RootAndDescendants) {
306 if (is<HTMLFrameOwnerElement>(root))
307 frameOwners.append(downcast<HTMLFrameOwnerElement>(root));
308 }
309
310 collectFrameOwners(frameOwners, root);
311
312 if (auto* shadowRoot = root.shadowRoot())
313 collectFrameOwners(frameOwners, *shadowRoot);
314
315 // Must disable frame loading in the subtree so an unload handler cannot
316 // insert more frames and create loaded frames in detached subtrees.
317 SubframeLoadingDisabler disabler(&root);
318
319 bool isFirst = true;
320 for (auto& owner : frameOwners) {
321 // Don't need to traverse up the tree for the first owner since no
322 // script could have moved it.
323 if (isFirst || root.containsIncludingShadowDOM(&owner.get()))
324 owner.get().disconnectContentFrame();
325 isFirst = false;
326 }
