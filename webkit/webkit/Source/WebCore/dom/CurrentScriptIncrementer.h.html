29#pragma once
30
31#include "Document.h"
32#include "ScriptElement.h"
33
34namespace WebCore {
35
36class CurrentScriptIncrementer {
37 WTF_MAKE_NONCOPYABLE(CurrentScriptIncrementer);
38public:
39 CurrentScriptIncrementer(Document& document, ScriptElement& scriptElement)
40 : m_document(document)
41 {
42 bool shouldPushNullForCurrentScript = scriptElement.element().isInShadowTree() || scriptElement.scriptType() != ScriptElement::ScriptType::Classic;
43 m_document.pushCurrentScript(shouldPushNullForCurrentScript ? nullptr : &scriptElement.element());
44 }
45
46 ~CurrentScriptIncrementer()
47 {
48 m_document.popCurrentScript();
49 }
50
51private:
52 Document& m_document;
53};
54
55} // namespace WebCore
