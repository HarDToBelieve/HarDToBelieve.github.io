46class CustomElementReactionQueueItem {
47public:
48 enum class Type {
49 ElementUpgrade,
50 Connected,
51 Disconnected,
52 Adopted,
53 AttributeChanged,
54 };
55
56 CustomElementReactionQueueItem(Type type)
57 : m_type(type)
58 { }
59
60 CustomElementReactionQueueItem(Document& oldDocument, Document& newDocument)
61 : m_type(Type::Adopted)
62 , m_oldDocument(&oldDocument)
63 , m_newDocument(&newDocument)
64 { }
65
66 CustomElementReactionQueueItem(const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
67 : m_type(Type::AttributeChanged)
68 , m_attributeName(attributeName)
69 , m_oldValue(oldValue)
70 , m_newValue(newValue)
71 { }
72
73 Type type() const { return m_type; }
74
75 void invoke(Element& element, JSCustomElementInterface& elementInterface)
76 {
77 switch (m_type) {
78 case Type::ElementUpgrade:
79 elementInterface.upgradeElement(element);
80 break;
81 case Type::Connected:
82 elementInterface.invokeConnectedCallback(element);
83 break;
84 case Type::Disconnected:
85 elementInterface.invokeDisconnectedCallback(element);
86 break;
87 case Type::Adopted:
88 elementInterface.invokeAdoptedCallback(element, *m_oldDocument, *m_newDocument);
89 break;
90 case Type::AttributeChanged:
91 ASSERT(m_attributeName);
92 elementInterface.invokeAttributeChangedCallback(element, m_attributeName.value(), m_oldValue, m_newValue);
93 break;
94 }
95 }
96
97private:
98 Type m_type;
99 RefPtr<Document> m_oldDocument;
100 RefPtr<Document> m_newDocument;
101 std::optional<QualifiedName> m_attributeName;
102 AtomString m_oldValue;
103 AtomString m_newValue;
104};

127void CustomElementReactionQueue::enqueueElementUpgrade(Element& element, bool alreadyScheduledToUpgrade)
128{
129 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
130 ASSERT(element.reactionQueue());
131 auto& queue = *element.reactionQueue();
132 if (alreadyScheduledToUpgrade)
133 ASSERT(queue.hasJustUpgradeReaction());
134 else
135 queue.m_items.append({CustomElementReactionQueueItem::Type::ElementUpgrade});
136 enqueueElementOnAppropriateElementQueue(element);
137}

139// https://html.spec.whatwg.org/multipage/custom-elements.html#concept-try-upgrade
140void CustomElementReactionQueue::tryToUpgradeElement(Element& element)
141{
142 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
143 ASSERT(element.isCustomElementUpgradeCandidate());
144 auto* window = element.document().domWindow();
145 if (!window)
146 return;
147
148 auto* registry = window->customElementRegistry();
149 if (!registry)
150 return;
151
152 auto* elementInterface = registry->findInterface(element);
153 if (!elementInterface)
154 return;
155
156 element.enqueueToUpgrade(*elementInterface);
157}

159void CustomElementReactionQueue::enqueueConnectedCallbackIfNeeded(Element& element)
160{
161 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
162 ASSERT(element.isDefinedCustomElement());
163 ASSERT(element.document().refCount() > 0);
164 ASSERT(element.reactionQueue());
165 auto& queue = *element.reactionQueue();
166 if (!queue.m_interface->hasConnectedCallback())
167 return;
168 queue.m_items.append({CustomElementReactionQueueItem::Type::Connected});
169 enqueueElementOnAppropriateElementQueue(element);
170}
171
172void CustomElementReactionQueue::enqueueDisconnectedCallbackIfNeeded(Element& element)
173{
174 ASSERT(element.isDefinedCustomElement());
175 if (element.document().refCount() <= 0)
176 return; // Don't enqueue disconnectedCallback if the entire document is getting destructed.
177 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
178 ASSERT(element.reactionQueue());
179 auto& queue = *element.reactionQueue();
180 if (!queue.m_interface->hasDisconnectedCallback())
181 return;
182 queue.m_items.append({CustomElementReactionQueueItem::Type::Disconnected});
183 enqueueElementOnAppropriateElementQueue(element);
184}

186void CustomElementReactionQueue::enqueueAdoptedCallbackIfNeeded(Element& element, Document& oldDocument, Document& newDocument)
187{
188 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
189 ASSERT(element.isDefinedCustomElement());
190 ASSERT(element.document().refCount() > 0);
191 ASSERT(element.reactionQueue());
192 auto& queue = *element.reactionQueue();
193 if (!queue.m_interface->hasAdoptedCallback())
194 return;
195 queue.m_items.append({oldDocument, newDocument});
196 enqueueElementOnAppropriateElementQueue(element);
197}

199void CustomElementReactionQueue::enqueueAttributeChangedCallbackIfNeeded(Element& element, const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
200{
201 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
202 ASSERT(element.isDefinedCustomElement());
203 ASSERT(element.document().refCount() > 0);
204 ASSERT(element.reactionQueue());
205 auto& queue = *element.reactionQueue();
206 if (!queue.m_interface->observesAttribute(attributeName.localName()))
207 return;
208 queue.m_items.append({attributeName, oldValue, newValue});
209 enqueueElementOnAppropriateElementQueue(element);
210}

212void CustomElementReactionQueue::enqueuePostUpgradeReactions(Element& element)
213{
214 ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
215 ASSERT(element.isCustomElementUpgradeCandidate());
216 if (!element.hasAttributes() && !element.isConnected())
217 return;
218
219 ASSERT(element.reactionQueue());
220 auto& queue = *element.reactionQueue();
221
222 if (element.hasAttributes()) {
223 for (auto& attribute : element.attributesIterator()) {
224 if (queue.m_interface->observesAttribute(attribute.localName()))
225 queue.m_items.append({attribute.name(), nullAtom(), attribute.value()});
226 }
227 }
228
229 if (element.isConnected() && queue.m_interface->hasConnectedCallback())
230 queue.m_items.append({CustomElementReactionQueueItem::Type::Connected});
231}

238void CustomElementReactionQueue::invokeAll(Element& element)
239{
240 while (!m_items.isEmpty()) {
241 Vector<CustomElementReactionQueueItem> items = WTFMove(m_items);
242 for (auto& item : items)
243 item.invoke(element, m_interface.get());
244 }
245}

247inline void CustomElementQueue::add(Element& element)
248{
249 ASSERT(!m_invoking);
250 // FIXME: Avoid inserting the same element multiple times.
251 m_elements.append(element);
252}
253
254inline void CustomElementQueue::invokeAll()
255{
256 RELEASE_ASSERT(!m_invoking);
257 SetForScope<bool> invoking(m_invoking, true);
258 unsigned originalSize = m_elements.size();
259 // It's possible for more elements to be enqueued if some IDL attributes were missing CEReactions.
260 // Invoke callbacks slightly later here instead of crashing / ignoring those cases.
261 for (unsigned i = 0; i < m_elements.size(); ++i) {
262 auto& element = m_elements[i].get();
263 auto* queue = element.reactionQueue();
264 ASSERT(queue);
265 queue->invokeAll(element);
266 }
267 ASSERT_UNUSED(originalSize, m_elements.size() == originalSize);
268 m_elements.clear();
269}

271inline void CustomElementQueue::processQueue(JSC::JSGlobalObject* state)
272{
273 if (!state) {
274 invokeAll();
275 return;
276 }
277
278 auto& vm = state->vm();
279 JSC::JSLockHolder lock(vm);
280
281 JSC::Exception* previousException = nullptr;
282 {
283 auto catchScope = DECLARE_CATCH_SCOPE(vm);
284 previousException = catchScope.exception();
285 if (previousException)
286 catchScope.clearException();
287 }
288
289 invokeAll();
290
291 if (previousException) {
292 auto throwScope = DECLARE_THROW_SCOPE(vm);
293 throwException(state, throwScope, previousException);
294 }
295}

297// https://html.spec.whatwg.org/multipage/custom-elements.html#enqueue-an-element-on-the-appropriate-element-queue
298void CustomElementReactionQueue::enqueueElementOnAppropriateElementQueue(Element& element)
299{
300 ASSERT(element.reactionQueue());
301 if (!CustomElementReactionStack::s_currentProcessingStack) {
302 element.document().windowEventLoop().backupElementQueue().add(element);
303 return;
304 }
305
306 auto*& queue = CustomElementReactionStack::s_currentProcessingStack->m_queue;
307 if (!queue) // We use a raw pointer to avoid genearing code to delete it in ~CustomElementReactionStack.
308 queue = new CustomElementQueue;
309 queue->add(element);
310}
