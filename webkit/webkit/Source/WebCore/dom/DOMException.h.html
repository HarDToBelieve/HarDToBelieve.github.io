29#pragma once
30
31#include "ExceptionCode.h"
32#include <wtf/text/WTFString.h>
33
34namespace WebCore {
35
36class Exception;
37
38class DOMException : public RefCounted<DOMException> {
39public:
40 static Ref<DOMException> create(ExceptionCode, const String& message = emptyString());
41 static Ref<DOMException> create(const Exception&);
42
43 // For DOM bindings.
44 static Ref<DOMException> create(const String& message, const String& name);
45
46 using LegacyCode = uint8_t;
47 LegacyCode legacyCode() const { return m_legacyCode; }
48
49 String name() const { return m_name; }
50 String message() const { return m_message; }
51
52 struct Description {
53 const ASCIILiteral name;
54 const ASCIILiteral message;
55 LegacyCode legacyCode;
56 };
57
58 WEBCORE_EXPORT static const Description& description(ExceptionCode);
59
60 static ASCIILiteral name(ExceptionCode ec) { return description(ec).name; }
61 static ASCIILiteral message(ExceptionCode ec) { return description(ec).message; }
62
63private:
64 DOMException(LegacyCode, const String& name, const String& message);
65
66 LegacyCode m_legacyCode;
67 String m_name;
68 String m_message;
69};
70
71} // namespace WebCore
