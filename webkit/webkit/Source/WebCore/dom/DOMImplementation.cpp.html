71ExceptionOr<Ref<DocumentType>> DOMImplementation::createDocumentType(const String& qualifiedName, const String& publicId, const String& systemId)
72{
73 auto parseResult = Document::parseQualifiedName(qualifiedName);
74 if (parseResult.hasException())
75 return parseResult.releaseException();
76 return DocumentType::create(m_document, qualifiedName, publicId, systemId);
77}

79static inline Ref<XMLDocument> createXMLDocument(const String& namespaceURI, const Settings& settings)
80{
81 if (namespaceURI == SVGNames::svgNamespaceURI)
82 return SVGDocument::create(nullptr, settings, URL());
83 if (namespaceURI == HTMLNames::xhtmlNamespaceURI)
84 return XMLDocument::createXHTML(nullptr, settings, URL());
85 return XMLDocument::create(nullptr, settings, URL());
86}
87
88ExceptionOr<Ref<XMLDocument>> DOMImplementation::createDocument(const String& namespaceURI, const String& qualifiedName, DocumentType* documentType)
89{
90 auto document = createXMLDocument(namespaceURI, m_document.settings());
91 document->setContextDocument(m_document.contextDocument());
92 document->setSecurityOriginPolicy(m_document.securityOriginPolicy());
93
94 RefPtr<Element> documentElement;
95 if (!qualifiedName.isEmpty()) {
96 ASSERT(!document->domWindow()); // If domWindow is not null, createElementNS could find CustomElementRegistry and arbitrary scripts.
97 auto result = document->createElementNS(namespaceURI, qualifiedName);
98 if (result.hasException())
99 return result.releaseException();
100 documentElement = result.releaseReturnValue();
101 }
102
103 if (documentType)
104 document->appendChild(*documentType);
105 if (documentElement)
106 document->appendChild(*documentElement);
107
108 return document;
109}

120Ref<HTMLDocument> DOMImplementation::createHTMLDocument(const String& title)
121{
122 auto document = HTMLDocument::create(nullptr, m_document.settings(), URL());
123 document->open();
124 document->write(nullptr, { "<!doctype html><html><head></head><body></body></html>"_s });
125 if (!title.isNull()) {
126 auto titleElement = HTMLTitleElement::create(titleTag, document);
127 titleElement->appendChild(document->createTextNode(title));
128 ASSERT(document->head());
129 document->head()->appendChild(titleElement);
130 }
131 document->setContextDocument(m_document.contextDocument());
132 document->setSecurityOriginPolicy(m_document.securityOriginPolicy());
133 return document;
134}

136Ref<Document> DOMImplementation::createDocument(const String& contentType, Frame* frame, const Settings& settings, const URL& url)
137{
138 // FIXME: Inelegant to have this here just because this is the home of DOM APIs for creating documents.
139 // This is internal, not a DOM API. Maybe we should put it in a new class called DocumentFactory,
140 // because of the analogy with HTMLElementFactory.
141
142 // Plug-ins cannot take over for HTML, XHTML, plain text, or non-PDF images.
143 if (equalLettersIgnoringASCIICase(contentType, "text/html"))
144 return HTMLDocument::create(frame, settings, url);
145 if (equalLettersIgnoringASCIICase(contentType, "application/xhtml+xml"))
146 return XMLDocument::createXHTML(frame, settings, url);
147 if (equalLettersIgnoringASCIICase(contentType, "text/plain"))
148 return TextDocument::create(frame, settings, url);
149 bool isImage = MIMETypeRegistry::isSupportedImageMIMEType(contentType);
150 if (frame && isImage && !MIMETypeRegistry::isPDFOrPostScriptMIMEType(contentType))
151 return ImageDocument::create(*frame, url);
152
153 // The "image documents for subframe PDFs" mode will override a PDF plug-in.
154 if (frame && !frame->isMainFrame() && MIMETypeRegistry::isPDFMIMEType(contentType) && frame->settings().useImageDocumentForSubframePDF())
155 return ImageDocument::create(*frame, url);
156
157#if ENABLE(VIDEO)
158 MediaEngineSupportParameters parameters;
159 parameters.type = ContentType { contentType };
160 parameters.url = url;
161 if (MediaPlayer::supportsType(parameters) != MediaPlayer::SupportsType::IsNotSupported)
162 return MediaDocument::create(frame, settings, url);
163#endif
164
165#if ENABLE(FTPDIR)
166 if (equalLettersIgnoringASCIICase(contentType, "application/x-ftp-directory"))
167 return FTPDirectoryDocument::create(frame, settings, url);
168#endif
169
170 if (frame && frame->loader().client().shouldAlwaysUsePluginDocument(contentType))
171 return PluginDocument::create(*frame, url);
172
173 // The following is the relatively costly lookup that requires initializing the plug-in database.
174 if (frame && frame->page()) {
175 auto allowedPluginTypes = frame->arePluginsEnabled()
176 ? PluginData::AllPlugins : PluginData::OnlyApplicationPlugins;
177 if (frame->page()->pluginData().supportsWebVisibleMimeType(contentType, allowedPluginTypes))
178 return PluginDocument::create(*frame, url);
179 }
180
181 // Items listed here, after the plug-in checks, can be overridden by plug-ins.
182 // For example, plug-ins can take over support for PDF or SVG.
183 if (frame && isImage)
184 return ImageDocument::create(*frame, url);
185 if (MIMETypeRegistry::isTextMIMEType(contentType))
186 return TextDocument::create(frame, settings, url);
187 if (equalLettersIgnoringASCIICase(contentType, "image/svg+xml"))
188 return SVGDocument::create(frame, settings, url);
189 if (MIMETypeRegistry::isXMLMIMEType(contentType)) {
190 auto document = XMLDocument::create(frame, settings, url);
191 document->overrideMIMEType(contentType);
192 return document;
193 }
194 return HTMLDocument::create(frame, settings, url);
195}
