31#pragma once
32
33#include "DOMPointInit.h"
34#include "ExceptionOr.h"
35#include "ScriptWrappable.h"
36#include <wtf/IsoMalloc.h>
37#include <wtf/RefCounted.h>
38
39namespace WebCore {
40
41struct DOMMatrixInit;
42class DOMPoint;
43
44class DOMPointReadOnly : public ScriptWrappable, public RefCounted<DOMPointReadOnly> {
45 WTF_MAKE_ISO_ALLOCATED_EXPORT(DOMPointReadOnly, WEBCORE_EXPORT);
46public:
47 static Ref<DOMPointReadOnly> create(double x, double y, double z, double w) { return adoptRef(*new DOMPointReadOnly(x, y, z, w)); }
48 static Ref<DOMPointReadOnly> create(const DOMPointInit& init) { return create(init.x, init.y, init.z, init.w); }
49 static Ref<DOMPointReadOnly> fromPoint(const DOMPointInit& init) { return create(init.x, init.y, init.z, init.w); }
50
51 double x() const { return m_x; }
52 double y() const { return m_y; }
53 double z() const { return m_z; }
54 double w() const { return m_w; }
55
56 ExceptionOr<Ref<DOMPoint>> matrixTransform(DOMMatrixInit&&) const;
57
58protected:
59 DOMPointReadOnly(double x, double y, double z, double w)
60 : m_x(x)
61 , m_y(y)
62 , m_z(z)
63 , m_w(w)
64 {
65 }
66
67 // Any of these can be NaN or Inf.
68 double m_x;
69 double m_y;
70 double m_z;
71 double m_w;
72};
73
74} // namespace WebCore
