28#include "DOMQuadInit.h"
29#include "ScriptWrappable.h"
30#include <wtf/Ref.h>
31#include <wtf/RefCounted.h>
32
33namespace WebCore {
34
35class DOMPoint;
36class DOMRect;
37struct DOMPointInit;
38struct DOMRectInit;
39
40class DOMQuad : public ScriptWrappable, public RefCounted<DOMQuad> {
41 WTF_MAKE_ISO_ALLOCATED(DOMQuad);
42public:
43 static Ref<DOMQuad> create(const DOMPointInit& p1, const DOMPointInit& p2, const DOMPointInit& p3, const DOMPointInit& p4) { return adoptRef(*new DOMQuad(p1, p2, p3, p4)); }
44 static Ref<DOMQuad> fromRect(const DOMRectInit& init) { return adoptRef(*new DOMQuad(init)); }
45 static Ref<DOMQuad> fromQuad(const DOMQuadInit& init) { return create(init.p1, init.p2, init.p3, init.p4); }
46
47 const DOMPoint& p1() const { return m_p1; }
48 const DOMPoint& p2() const { return m_p2; }
49 const DOMPoint& p3() const { return m_p3; }
50 const DOMPoint& p4() const { return m_p4; }
51
52 Ref<DOMRect> getBounds() const;
53
54private:
55 DOMQuad(const DOMPointInit&, const DOMPointInit&, const DOMPointInit&, const DOMPointInit&);
56 explicit DOMQuad(const DOMRectInit&);
57
58 Ref<DOMPoint> m_p1;
59 Ref<DOMPoint> m_p2;
60 Ref<DOMPoint> m_p3;
61 Ref<DOMPoint> m_p4;
62};

64}
