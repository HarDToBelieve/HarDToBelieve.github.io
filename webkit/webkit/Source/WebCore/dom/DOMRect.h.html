26#pragma once
27
28#include "DOMRectReadOnly.h"
29#include "FloatRect.h"
30#include "IntRect.h"
31
32namespace WebCore {
33
34class FloatQuad;
35
36class DOMRect : public DOMRectReadOnly {
37public:
38 static Ref<DOMRect> create() { return adoptRef(*new DOMRect()); }
39 static Ref<DOMRect> create(double x, double y, double width, double height) { return adoptRef(*new DOMRect(x, y, width, height)); }
40 static Ref<DOMRect> create(FloatRect rect) { return adoptRef(*new DOMRect(rect.x(), rect.y(), rect.width(), rect.height())); }
41 static Ref<DOMRect> create(IntRect rect) { return adoptRef(*new DOMRect(rect.x(), rect.y(), rect.width(), rect.height())); }
42 static Ref<DOMRect> fromRect(const DOMRectInit& init) { return create(init.x, init.y, init.width, init.height); }
43
44 void setX(double x) { m_x = x; }
45 void setY(double y) { m_y = y; }
46
47 void setWidth(double width) { m_width = width; }
48 void setHeight(double height) { m_height = height; }
49
50private:
51 DOMRect(double x, double y, double width, double height)
52 : DOMRectReadOnly(x, y, width, height)
53 {
54 }
55
56 DOMRect() = default;
57};
58static_assert(sizeof(DOMRect) == sizeof(DOMRectReadOnly), "");
