77String DataTransferItem::kind() const

79 return m_file ? "file"_s : "string"_s;
80}

82String DataTransferItem::type() const
83{
84 return isInDisabledMode() ? String() : m_type;
85}
86
87void DataTransferItem::getAsString(Document& document, RefPtr<StringCallback>&& callback) const
88{
89 if (!callback || !m_list || m_file)
90 return;
91
92 auto& dataTransfer = m_list->dataTransfer();
93 if (!dataTransfer.canReadData())
94 return;
95
96 // FIXME: Make this async.
97 callback->scheduleCallback(document, dataTransfer.getDataForItem(document, m_type));
98}
99
100RefPtr<File> DataTransferItem::getAsFile() const
101{
102 if (!m_list || !m_list->dataTransfer().canReadData())
103 return nullptr;
104 return m_file.copyRef();
105}
106
107RefPtr<FileSystemEntry> DataTransferItem::getAsEntry(ScriptExecutionContext& context) const
108{
109 auto file = getAsFile();
110 if (!file)
111 return nullptr;
112
113 return DOMFileSystem::createEntryForFile(context, *file);
114}
