55RefPtr<DataTransferItem> DataTransferItemList::item(unsigned index)
56{
57 auto& items = ensureItems();
58 if (items.size() <= index)
59 return nullptr;
60 return items[index].copyRef();

63static bool shouldExposeTypeInItemList(const String& type)
64{
65 return RuntimeEnabledFeatures::sharedFeatures().customPasteboardDataEnabled() || Pasteboard::isSafeTypeForDOMToReadAndWrite(type);
66}

68ExceptionOr<RefPtr<DataTransferItem>> DataTransferItemList::add(const String& data, const String& type)
69{
70 if (!m_dataTransfer.canWriteData())
71 return nullptr;
72
73 for (auto& item : ensureItems()) {
74 if (!item->isFile() && equalIgnoringASCIICase(item->type(), type))
75 return Exception { NotSupportedError };
76 }
77
78 String lowercasedType = type.convertToASCIILowercase();
79
80 if (!shouldExposeTypeInItemList(lowercasedType))
81 return nullptr;
82
83 m_dataTransfer.setDataFromItemList(lowercasedType, data);
84 ASSERT(m_items);
85 m_items->append(DataTransferItem::create(makeWeakPtr(*this), lowercasedType));
86 return m_items->last().ptr();
87}
88
89RefPtr<DataTransferItem> DataTransferItemList::add(Ref<File>&& file)
90{
91 if (!m_dataTransfer.canWriteData())
92 return nullptr;
93
94 ensureItems().append(DataTransferItem::create(makeWeakPtr(*this), file->type(), file.copyRef()));
95 m_dataTransfer.didAddFileToItemList();
96 return m_items->last().ptr();
97}

99ExceptionOr<void> DataTransferItemList::remove(unsigned index)
100{
101 if (!m_dataTransfer.canWriteData())
102 return Exception { InvalidStateError };
103
104 auto& items = ensureItems();
105 if (items.size() <= index)
106 return Exception { IndexSizeError }; // Matches Gecko. See https://github.com/whatwg/html/issues/2925
107
108 // FIXME: Remove the file from the pasteboard object once we add support for it.
109 Ref<DataTransferItem> removedItem = items[index].copyRef();
110 if (!removedItem->isFile())
111 m_dataTransfer.pasteboard().clear(removedItem->type());
112 removedItem->clearListAndPutIntoDisabledMode();
113 items.remove(index);
114 if (removedItem->isFile())
115 m_dataTransfer.updateFileList(scriptExecutionContext());
116
117 return { };
118}
119
120void DataTransferItemList::clear()
121{
122 m_dataTransfer.pasteboard().clear();
123 bool removedItemContainingFile = false;
124 if (m_items) {
125 for (auto& item : *m_items) {
126 removedItemContainingFile |= item->isFile();
127 item->clearListAndPutIntoDisabledMode();
128 }
129 m_items->clear();
130 }
131
132 if (removedItemContainingFile)
133 m_dataTransfer.updateFileList(scriptExecutionContext());
134}

136Vector<Ref<DataTransferItem>>& DataTransferItemList::ensureItems() const
137{
138 if (m_items)
139 return *m_items;
140
141 Vector<Ref<DataTransferItem>> items;
142 for (auto& type : m_dataTransfer.typesForItemList()) {
143 auto lowercasedType = type.convertToASCIILowercase();
144 if (shouldExposeTypeInItemList(lowercasedType))
145 items.append(DataTransferItem::create(makeWeakPtr(*const_cast<DataTransferItemList*>(this)), lowercasedType));
146 }
147
148 for (auto& file : m_dataTransfer.files(document()).files())
149 items.append(DataTransferItem::create(makeWeakPtr(*const_cast<DataTransferItemList*>(this)), file->type(), file.copyRef()));
150
151 m_items = WTFMove(items);
152
153 return *m_items;
154}

156static void removeStringItemOfLowercasedType(Vector<Ref<DataTransferItem>>& items, const String& lowercasedType)
157{
158 auto index = items.findMatching([lowercasedType](auto& item) {
159 return !item->isFile() && item->type() == lowercasedType;
160 });
161 if (index == notFound)
162 return;
163 items[index]->clearListAndPutIntoDisabledMode();
164 items.remove(index);
165}

167void DataTransferItemList::didClearStringData(const String& type)
168{
169 if (!m_items)
170 return;
171
172 auto& items = *m_items;
173 if (!type.isNull())
174 return removeStringItemOfLowercasedType(items, type.convertToASCIILowercase());
175
176 for (auto& item : items) {
177 if (!item->isFile())
178 item->clearListAndPutIntoDisabledMode();
179 }
180 items.removeAllMatching([](auto& item) {
181 return !item->isFile();
182 });
183}
184
185// https://html.spec.whatwg.org/multipage/dnd.html#dom-datatransfer-setdata
186void DataTransferItemList::didSetStringData(const String& type)
187{
188 if (!m_items)
189 return;
190
191 String lowercasedType = type.convertToASCIILowercase();
192 removeStringItemOfLowercasedType(*m_items, type.convertToASCIILowercase());
193
194 m_items->append(DataTransferItem::create(makeWeakPtr(*this), lowercasedType));
195}
