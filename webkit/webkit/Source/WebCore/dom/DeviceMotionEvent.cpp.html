51static std::optional<DeviceMotionEvent::Acceleration> convert(const DeviceMotionData::Acceleration* acceleration)
52{
53 if (!acceleration)
54 return std::nullopt;
55
56 return DeviceMotionEvent::Acceleration { acceleration->x(), acceleration->y(), acceleration->z() };
57}
58
59static std::optional<DeviceMotionEvent::RotationRate> convert(const DeviceMotionData::RotationRate* rotationRate)
60{
61 if (!rotationRate)
62 return std::nullopt;
63
64 return DeviceMotionEvent::RotationRate { rotationRate->alpha(), rotationRate->beta(), rotationRate->gamma() };
65}
66
67static RefPtr<DeviceMotionData::Acceleration> convert(std::optional<DeviceMotionEvent::Acceleration>&& acceleration)
68{
69 if (!acceleration)
70 return nullptr;
71
72 if (!acceleration->x && !acceleration->y && !acceleration->z)
73 return nullptr;
74
75 return DeviceMotionData::Acceleration::create(acceleration->x, acceleration->y, acceleration->z);
76}
77
78static RefPtr<DeviceMotionData::RotationRate> convert(std::optional<DeviceMotionEvent::RotationRate>&& rotationRate)
79{
80 if (!rotationRate)
81 return nullptr;
82
83 if (!rotationRate->alpha && !rotationRate->beta && !rotationRate->gamma)
84 return nullptr;
85
86 return DeviceMotionData::RotationRate::create(rotationRate->alpha, rotationRate->beta, rotationRate->gamma);
87}

109void DeviceMotionEvent::initDeviceMotionEvent(const AtomString& type, bool bubbles, bool cancelable, std::optional<DeviceMotionEvent::Acceleration>&& acceleration, std::optional<DeviceMotionEvent::Acceleration>&& accelerationIncludingGravity, std::optional<DeviceMotionEvent::RotationRate>&& rotationRate, std::optional<double> interval)
110{
111 if (isBeingDispatched())
112 return;
113
114 initEvent(type, bubbles, cancelable);
115 m_deviceMotionData = DeviceMotionData::create(convert(WTFMove(acceleration)), convert(WTFMove(accelerationIncludingGravity)), convert(WTFMove(rotationRate)), interval);
116}

130#if ENABLE(DEVICE_ORIENTATION)
131void DeviceMotionEvent::requestPermission(Document& document, PermissionPromise&& promise)
132{
133 auto* window = document.domWindow();
134 auto* page = document.page();
135 if (!window || !page)
136 return promise.reject(Exception { InvalidStateError, "No browsing context"_s });
137
138 String errorMessage;
139 if (!window->isAllowedToUseDeviceMotion(errorMessage)) {
140 document.addConsoleMessage(MessageSource::JS, MessageLevel::Warning, makeString("Call to requestPermission() failed, reason: ", errorMessage, "."));
141 return promise.resolve(PermissionState::Denied);
142 }
143
144 document.deviceOrientationAndMotionAccessController().shouldAllowAccess(document, [promise = WTFMove(promise)](auto permissionState) mutable {
145 if (permissionState == PermissionState::Prompt)
146 return promise.reject(Exception { NotAllowedError, "Requesting device motion access requires a user gesture to prompt"_s });
147 promise.resolve(permissionState);
148 });
149}
150#endif
