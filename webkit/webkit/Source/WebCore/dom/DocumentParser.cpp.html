32namespace WebCore {
33
34DocumentParser::DocumentParser(Document& document)
35 : m_state(ParserState::Parsing)
36 , m_documentWasLoadedAsPartOfNavigation(false)
37 , m_document(makeWeakPtr(document))
38{
39}
40
41DocumentParser::~DocumentParser()
42{
43 // Document is expected to call detach() before releasing its ref.
44 // This ASSERT is slightly awkward for parsers with a fragment case
45 // as there is no Document to release the ref.
46 ASSERT(!m_document);
47}
48
49void DocumentParser::startParsing()
50{
51 m_state = ParserState::Parsing;
52}
53
54void DocumentParser::prepareToStopParsing()
55{
56 ASSERT(m_state == ParserState::Parsing);
57 m_state = ParserState::Stopping;
58}
59
60void DocumentParser::stopParsing()
61{
62 m_state = ParserState::Stopped;
63}
64
65void DocumentParser::detach()
66{
67 m_state = ParserState::Detached;
68 m_document = nullptr;
69}

73}

77}
