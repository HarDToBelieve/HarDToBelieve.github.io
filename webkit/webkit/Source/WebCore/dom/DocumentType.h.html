28namespace WebCore {

30class NamedNodeMap;

32class DocumentType final : public Node {
33 WTF_MAKE_ISO_ALLOCATED(DocumentType);
34public:
35 static Ref<DocumentType> create(Document& document, const String& name, const String& publicId, const String& systemId)
36 {
37 return adoptRef(*new DocumentType(document, name, publicId, systemId));
38 }
39
40 const String& name() const { return m_name; }
41 const String& publicId() const { return m_publicId; }
42 const String& systemId() const { return m_systemId; }
43
44private:
45 DocumentType(Document&, const String& name, const String& publicId, const String& systemId);
46
47 String nodeName() const override;
48 NodeType nodeType() const override;
49 Ref<Node> cloneNodeInternal(Document&, CloningOperation) override;
50
51 String m_name;
52 String m_publicId;
53 String m_systemId;
54};
55
56} // namespace WebCore
