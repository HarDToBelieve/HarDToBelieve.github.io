26#pragma once
27
28#include "ScriptDisallowedScope.h"
29
30namespace WebCore {
31
32class ElementIteratorAssertions {
33public:
34 ElementIteratorAssertions(const Node* first = nullptr);
35 bool domTreeHasMutated() const;
36 void dropEventDispatchAssertion();
37 void clear();
38
39private:
40 const Document* m_document;
41 uint64_t m_initialDOMTreeVersion;
42 std::optional<ScriptDisallowedScope> m_eventDispatchAssertion;
43};
44
45// FIXME: No real point in doing these as inlines; they are for debugging and we usually turn off inlining in debug builds.
46
47inline ElementIteratorAssertions::ElementIteratorAssertions(const Node* first)
48 : m_document(first ? &first->document() : nullptr)
49 , m_initialDOMTreeVersion(first ? m_document->domTreeVersion() : 0)
50{
51 if (first)
52 m_eventDispatchAssertion = ScriptDisallowedScope();
53}
54
55inline bool ElementIteratorAssertions::domTreeHasMutated() const
56{
57 return m_document && m_document->domTreeVersion() != m_initialDOMTreeVersion;
58}
59
60inline void ElementIteratorAssertions::dropEventDispatchAssertion()
61{
62 m_eventDispatchAssertion = std::nullopt;
63}
64
65inline void ElementIteratorAssertions::clear()
66{
67 m_document = nullptr;
68 m_initialDOMTreeVersion = 0;
69 m_eventDispatchAssertion = std::nullopt;
70}
71
72} // namespace WebCore
