28#pragma once
29
30#include "HTMLFormElement.h"
31#include "TouchList.h"
32
33namespace WebCore {
34
35class EventContext {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 using EventInvokePhase = EventTarget::EventInvokePhase;
39
40 enum class Type : uint8_t {
41 Normal = 0,
42 MouseOrFocus,
43 Touch,
44 Window,
45 };
46
47 EventContext(Type, Node*, EventTarget* currentTarget, EventTarget* origin, int closedShadowDepth);
48 EventContext(Type, Node&, Node* currentTarget, EventTarget* origin, int closedShadowDepth);
49 ~EventContext();
50
51 Node* node() const { return m_node.get(); }
52 EventTarget* currentTarget() const { return m_currentTarget.get(); }
53 bool isCurrentTargetInShadowTree() const { return m_currentTargetIsInShadowTree; }
54 EventTarget* target() const { return m_target.get(); }
55 int closedShadowDepth() const { return m_closedShadowDepth; }
56
57 void handleLocalEvents(Event&, EventInvokePhase) const;
58
59 bool isMouseOrFocusEventContext() const { return m_type == Type::MouseOrFocus; }
60 bool isTouchEventContext() const { return m_type == Type::Touch; }
61 bool isWindowContext() const { return m_type == Type::Window; }
62
63 Node* relatedTarget() const { return m_relatedTarget.get(); }
64 void setRelatedTarget(Node*);
65
66#if ENABLE(TOUCH_EVENTS)
67 enum TouchListType { Touches, TargetTouches, ChangedTouches };
68 TouchList& touchList(TouchListType);
69#endif
70
71private:
72 inline EventContext(Type, Node* currentNode, RefPtr<EventTarget>&& currentTarget, EventTarget* origin, int closedShadowDepth, bool currentTargetIsInShadowTree = false);
73
74#if ENABLE(TOUCH_EVENTS)
75 void initializeTouchLists();
76#endif
77
78#if ASSERT_ENABLED
79 bool isUnreachableNode(EventTarget*) const;
80#endif
81
82 RefPtr<Node> m_node;
83 RefPtr<EventTarget> m_currentTarget;
84 RefPtr<EventTarget> m_target;
85 RefPtr<Node> m_relatedTarget;
86#if ENABLE(TOUCH_EVENTS)
87 RefPtr<TouchList> m_touches;
88 RefPtr<TouchList> m_targetTouches;
89 RefPtr<TouchList> m_changedTouches;
90#endif
91 int m_closedShadowDepth { 0 };
92 bool m_currentTargetIsInShadowTree { false };
93 bool m_contextNodeIsFormElement { false };
94 Type m_type { Type::Normal };
95};
96
97inline EventContext::EventContext(Type type, Node* node, RefPtr<EventTarget>&& currentTarget, EventTarget* origin, int closedShadowDepth, bool currentTargetIsInShadowTree)
98 : m_node { node }
99 , m_currentTarget { WTFMove(currentTarget) }
100 , m_target { origin }
101 , m_closedShadowDepth { closedShadowDepth }
102 , m_currentTargetIsInShadowTree { currentTargetIsInShadowTree }
103 , m_type { type }
104{
105 ASSERT(!isUnreachableNode(m_target.get()));
106#if ENABLE(TOUCH_EVENTS)
107 if (m_type == Type::Touch)
108 initializeTouchLists();
109#else
110 ASSERT(m_type != Type::Touch);
111#endif
112}
113
114inline EventContext::EventContext(Type type, Node* node, EventTarget* currentTarget, EventTarget* origin, int closedShadowDepth)
115 : EventContext(type, node, makeRefPtr(currentTarget), origin, closedShadowDepth)
116{
117 ASSERT(!is<Node>(currentTarget));
118}
119
120// This variant avoids calling EventTarget::ref() which is a virtual function call.
121inline EventContext::EventContext(Type type, Node& node, Node* currentTarget, EventTarget* origin, int closedShadowDepth)
122 : EventContext(type, &node, makeRefPtr(currentTarget), origin, closedShadowDepth, currentTarget && currentTarget->isInShadowTree())
123{
124 m_contextNodeIsFormElement = is<HTMLFormElement>(node);
125}
126
127inline void EventContext::setRelatedTarget(Node* relatedTarget)
128{
129 ASSERT(!isUnreachableNode(relatedTarget));
130 m_relatedTarget = relatedTarget;
131}
132
133#if ENABLE(TOUCH_EVENTS)
134
135inline TouchList& EventContext::touchList(TouchListType type)
136{
137 switch (type) {
138 case Touches:
139 return *m_touches;
140 case TargetTouches:
141 return *m_targetTouches;
142 case ChangedTouches:
143 return *m_changedTouches;
144 }
145 ASSERT_NOT_REACHED();
146 return *m_touches;
147}
148
149#endif
150
151} // namespace WebCore
