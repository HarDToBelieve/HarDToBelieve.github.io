50void EventDispatcher::dispatchScopedEvent(Node& node, Event& event)
51{
52 // Need to set the target here so the scoped event queue knows which node to dispatch to.
53 event.setTarget(EventPath::eventTargetRespectingTargetRules(node));
54 ScopedEventQueue::singleton().enqueueEvent(event);
55}

57static void callDefaultEventHandlersInBubblingOrder(Event& event, const EventPath& path)
58{
59 if (path.isEmpty())
60 return;
61
62 // Non-bubbling events call only one default event handler, the one for the target.
63 path.contextAt(0).node()->defaultEventHandler(event);
64 ASSERT(!event.defaultPrevented());
65
66 if (event.defaultHandled() || !event.bubbles())
67 return;
68
69 size_t size = path.size();
70 for (size_t i = 1; i < size; ++i) {
71 path.contextAt(i).node()->defaultEventHandler(event);
72 ASSERT(!event.defaultPrevented());
73 if (event.defaultHandled())
74 return;
75 }
76}

78static bool isInShadowTree(EventTarget* target)
79{
80 return is<Node>(target) && downcast<Node>(*target).isInShadowTree();
81}

83static void dispatchEventInDOM(Event& event, const EventPath& path)
84{
85 // Invoke capturing event listeners in the reverse order.
86 for (size_t i = path.size(); i > 0; --i) {
87 const EventContext& eventContext = path.contextAt(i - 1);
88 if (eventContext.currentTarget() == eventContext.target())
89 event.setEventPhase(Event::AT_TARGET);
90 else
91 event.setEventPhase(Event::CAPTURING_PHASE);
92 eventContext.handleLocalEvents(event, EventTarget::EventInvokePhase::Capturing);
93 if (event.propagationStopped())
94 return;
95 }
96
97 // Invoke bubbling event listeners.
98 size_t size = path.size();
99 for (size_t i = 0; i < size; ++i) {
100 const EventContext& eventContext = path.contextAt(i);
101 if (eventContext.currentTarget() == eventContext.target())
102 event.setEventPhase(Event::AT_TARGET);
103 else if (event.bubbles())
104 event.setEventPhase(Event::BUBBLING_PHASE);
105 else
106 continue;
107 eventContext.handleLocalEvents(event, EventTarget::EventInvokePhase::Bubbling);
108 if (event.propagationStopped())
109 return;
110 }
111}

113static bool shouldSuppressEventDispatchInDOM(Node& node, Event& event)
114{
115 if (!event.isTrusted())
116 return false;
117
118 auto frame = node.document().frame();
119 if (!frame)
120 return false;
121
122 if (!frame->mainFrame().loader().shouldSuppressTextInputFromEditing())
123 return false;
124
125 if (is<TextEvent>(event)) {
126 auto& textEvent = downcast<TextEvent>(event);
127 return textEvent.isKeyboard() || textEvent.isComposition();
128 }
129
130 return is<CompositionEvent>(event) || is<InputEvent>(event) || is<KeyboardEvent>(event);
131}

133static HTMLInputElement* findInputElementInEventPath(const EventPath& path)
134{
135 size_t size = path.size();
136 for (size_t i = 0; i < size; ++i) {
137 const EventContext& eventContext = path.contextAt(i);
138 if (is<HTMLInputElement>(eventContext.currentTarget()))
139 return downcast<HTMLInputElement>(eventContext.currentTarget());
140 }
141 return nullptr;
142}

144void EventDispatcher::dispatchEvent(Node& node, Event& event)
145{
146 ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::InMainThread::isEventDispatchAllowedInSubtree(node));
147
148 LOG_WITH_STREAM(Events, stream << "EventDispatcher::dispatchEvent " << event << " on node " << node);
149
150 auto protectedNode = makeRef(node);
151 auto protectedView = makeRefPtr(node.document().view());
152
153 EventPath eventPath { node, event };
154
155 std::optional<bool> shouldClearTargetsAfterDispatch;
156 for (size_t i = eventPath.size(); i > 0; --i) {
157 const EventContext& eventContext = eventPath.contextAt(i - 1);
158 // FIXME: We should also set shouldClearTargetsAfterDispatch to true if an EventTarget object in eventContext's touch target list
159 // is a node and its root is a shadow root.
160 if (eventContext.target()) {
161 shouldClearTargetsAfterDispatch = isInShadowTree(eventContext.target()) || isInShadowTree(eventContext.relatedTarget());
162 break;
163 }
164 }
165
166 ChildNodesLazySnapshot::takeChildNodesLazySnapshot();
167
168 event.resetBeforeDispatch();
169
170 event.setTarget(EventPath::eventTargetRespectingTargetRules(node));
171 if (!event.target())
172 return;
173
174 InputElementClickState clickHandlingState;
175
176 bool isActivationEvent = event.type() == eventNames().clickEvent;
177 RefPtr<HTMLInputElement> inputForLegacyPreActivationBehavior = is<HTMLInputElement>(node) ? &downcast<HTMLInputElement>(node) : nullptr;
178 if (!inputForLegacyPreActivationBehavior && isActivationEvent && event.bubbles())
179 inputForLegacyPreActivationBehavior = findInputElementInEventPath(eventPath);
180 if (inputForLegacyPreActivationBehavior)
181 inputForLegacyPreActivationBehavior->willDispatchEvent(event, clickHandlingState);
182
183 if (shouldSuppressEventDispatchInDOM(node, event))
184 event.stopPropagation();
185
186 if (!event.propagationStopped() && !eventPath.isEmpty()) {
187 event.setEventPath(eventPath);
188 dispatchEventInDOM(event, eventPath);
189 }
190
191 event.resetAfterDispatch();
192
193 if (clickHandlingState.stateful)
194 inputForLegacyPreActivationBehavior->didDispatchClickEvent(event, clickHandlingState);
195
196 // Call default event handlers. While the DOM does have a concept of preventing
197 // default handling, the detail of which handlers are called is an internal
198 // implementation detail and not part of the DOM.
199 if (!event.defaultPrevented() && !event.defaultHandled() && !event.isDefaultEventHandlerIgnored()) {
200 // FIXME: Not clear why we need to reset the target for the default event handlers.
201 // We should research this, and remove this code if possible.
202 auto* finalTarget = event.target();
203 event.setTarget(EventPath::eventTargetRespectingTargetRules(node));
204 callDefaultEventHandlersInBubblingOrder(event, eventPath);
205 event.setTarget(finalTarget);
206 }
207
208 if (shouldClearTargetsAfterDispatch.value_or(false)) {
209 event.setTarget(nullptr);
210 event.setRelatedTarget(nullptr);
211 // FIXME: We should also clear the event's touch target list.
212 }
213}

215template<typename T>
216static void dispatchEventWithType(const Vector<T*>& targets, Event& event)
217{
218 ASSERT(targets.size() >= 1);
219 ASSERT(*targets.begin());
220
221 EventPath eventPath { targets };
222 event.setTarget(*targets.begin());
223 event.setEventPath(eventPath);
224 event.resetBeforeDispatch();
225 dispatchEventInDOM(event, eventPath);
226 event.resetAfterDispatch();
227}
228
229void EventDispatcher::dispatchEvent(const Vector<EventTarget*>& targets, Event& event)
230{
231 dispatchEventWithType<EventTarget>(targets, event);
232}
