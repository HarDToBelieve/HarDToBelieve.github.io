33void EventLoop::queueTask(std::unique_ptr<EventLoopTask>&& task)
34{
35 ASSERT(task->taskSource() != TaskSource::Microtask);
36 ASSERT(task->group());
37 ASSERT(isContextThread());
38 scheduleToRunIfNeeded();
39 m_tasks.append(WTFMove(task));
40}
41
42void EventLoop::queueMicrotask(std::unique_ptr<EventLoopTask>&& microtask)
43{
44 ASSERT(microtask->taskSource() == TaskSource::Microtask);
45 microtaskQueue().append(WTFMove(microtask));
46 scheduleToRunIfNeeded(); // FIXME: Remove this once everything is integrated with the event loop.
47}

49void EventLoop::performMicrotaskCheckpoint()
50{
51 microtaskQueue().performMicrotaskCheckpoint();

54void EventLoop::resumeGroup(EventLoopTaskGroup& group)
55{
56 ASSERT(isContextThread());
57 if (!m_groupsWithSuspenedTasks.contains(group))
58 return;
59 scheduleToRunIfNeeded();
60}
61
62void EventLoop::registerGroup(EventLoopTaskGroup& group)
63{
64 ASSERT(isContextThread());
65 m_associatedGroups.add(group);
66}
67
68void EventLoop::unregisterGroup(EventLoopTaskGroup& group)
69{
70 ASSERT(isContextThread());
71 if (m_associatedGroups.remove(group))
72 stopAssociatedGroupsIfNecessary();
73}
74
75void EventLoop::stopAssociatedGroupsIfNecessary()
76{
77 ASSERT(isContextThread());
78 for (auto& group : m_associatedGroups) {
79 if (!group.isReadyToStop())
80 return;
81 }
82 auto associatedGroups = std::exchange(m_associatedGroups, { });
83 for (auto& group : associatedGroups)
84 group.stopAndDiscardAllTasks();
85}
86
87void EventLoop::stopGroup(EventLoopTaskGroup& group)
88{
89 ASSERT(isContextThread());
90 m_tasks.removeAllMatching([&group] (auto& task) {
91 return group.matchesTask(*task);
92 });
93}
94
95void EventLoop::scheduleToRunIfNeeded()
96{
97 if (m_isScheduledToRun)
98 return;
99 m_isScheduledToRun = true;
100 scheduleToRun();
101}

103void EventLoop::run()
104{
105 m_isScheduledToRun = false;
106 bool didPerformMicrotaskCheckpoint = false;
107
108 if (!m_tasks.isEmpty()) {
109 auto tasks = std::exchange(m_tasks, { });
110 m_groupsWithSuspenedTasks.clear();
111 Vector<std::unique_ptr<EventLoopTask>> remainingTasks;
112 for (auto& task : tasks) {
113 auto* group = task->group();
114 if (!group || group->isStoppedPermanently())
115 continue;
116
117 if (group->isSuspended()) {
118 m_groupsWithSuspenedTasks.add(group);
119 remainingTasks.append(WTFMove(task));
120 continue;
121 }
122
123 task->execute();
124 didPerformMicrotaskCheckpoint = true;
125 microtaskQueue().performMicrotaskCheckpoint();
126 }
127 for (auto& task : m_tasks)
128 remainingTasks.append(WTFMove(task));
129 m_tasks = WTFMove(remainingTasks);
130 }
131
132 // FIXME: Remove this once everything is integrated with the event loop.
133 if (!didPerformMicrotaskCheckpoint)
134 microtaskQueue().performMicrotaskCheckpoint();
135}

137void EventLoop::clearAllTasks()
138{
139 m_tasks.clear();
140 m_groupsWithSuspenedTasks.clear();
141}

143void EventLoopTaskGroup::queueTask(std::unique_ptr<EventLoopTask>&& task)
144{
145 if (m_state == State::Stopped || !m_eventLoop)
146 return;
147 ASSERT(task->group() == this);
148 m_eventLoop->queueTask(WTFMove(task));
149}

151class EventLoopFunctionDispatchTask : public EventLoopTask {
152public:
153 EventLoopFunctionDispatchTask(TaskSource source, EventLoopTaskGroup& group, EventLoop::TaskFunction&& function)
154 : EventLoopTask(source, group)
155 , m_function(WTFMove(function))
156 {
157 }
158
159 void execute() final { m_function(); }
160
161private:
162 EventLoop::TaskFunction m_function;
163};
164
165void EventLoopTaskGroup::queueTask(TaskSource source, EventLoop::TaskFunction&& function)
166{
167 return queueTask(makeUnique<EventLoopFunctionDispatchTask>(source, *this, WTFMove(function)));
168}
169
170void EventLoopTaskGroup::queueMicrotask(EventLoop::TaskFunction&& function)
171{
172 if (m_state == State::Stopped || !m_eventLoop)
173 return;
174 m_eventLoop->queueMicrotask(makeUnique<EventLoopFunctionDispatchTask>(TaskSource::Microtask, *this, WTFMove(function)));
175}
176
177void EventLoopTaskGroup::performMicrotaskCheckpoint()
178{
179 if (m_eventLoop)
180 m_eventLoop->performMicrotaskCheckpoint();
181}
182
183void EventLoopTaskGroup::runAtEndOfMicrotaskCheckpoint(EventLoop::TaskFunction&& function)
184{
185 if (m_state == State::Stopped || !m_eventLoop)
186 return;
187
188 microtaskQueue().addCheckpointTask(makeUnique<EventLoopFunctionDispatchTask>(TaskSource::IndexedDB, *this, WTFMove(function)));
189}
