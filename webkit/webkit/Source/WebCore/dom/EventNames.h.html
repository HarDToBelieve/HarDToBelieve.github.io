343struct EventNames {
344 WTF_MAKE_NONCOPYABLE(EventNames); WTF_MAKE_FAST_ALLOCATED;
345
346public:
347#define DOM_EVENT_NAMES_DECLARE(name) const AtomString name##Event;
348 DOM_EVENT_NAMES_FOR_EACH(DOM_EVENT_NAMES_DECLARE)
349#undef DOM_EVENT_NAMES_DECLARE
350
351 // FIXME: The friend declaration to makeUnique below does not work in windows port.
352 //
353 // template<class T, class... Args>
354 // friend typename std::_Unique_if<T>::_Single_object makeUnique(Args&&...);
355 //
356 // This create function should be deleted later and is only for keeping EventNames as private.
357 // makeUnique should be used instead.
358 //
359 template<class... Args>
360 static std::unique_ptr<EventNames> create(Args&&... args)
361 {
362 return std::unique_ptr<EventNames>(new EventNames(std::forward<Args>(args)...));
363 }
364
365 // FIXME: Inelegant to call these both event names and event types.
366 // We should choose one term and stick to it.
367 bool isWheelEventType(const AtomString& eventType) const;
368 bool isGestureEventType(const AtomString& eventType) const;
369 bool isTouchRelatedEventType(const AtomString& eventType, EventTarget&) const;
370 bool isTouchScrollBlockingEventType(const AtomString& eventType) const;
371#if ENABLE(GAMEPAD)
372 bool isGamepadEventType(const AtomString& eventType) const;
373#endif
374
375 std::array<std::reference_wrapper<const AtomString>, 13> touchRelatedEventNames() const;
376 std::array<std::reference_wrapper<const AtomString>, 16> extendedTouchRelatedEventNames() const;
377 std::array<std::reference_wrapper<const AtomString>, 3> gestureEventNames() const;
378
379private:
380 EventNames(); // Private to prevent accidental call to EventNames() instead of eventNames().
381 friend class ThreadGlobalData; // Allow ThreadGlobalData to create the per-thread EventNames object.
382
383 int dummy; // Needed to make initialization macro work.
384};
385
386const EventNames& eventNames();
387
388inline const EventNames& eventNames()
389{
390 return threadGlobalData().eventNames();
391}
392
393inline bool EventNames::isGestureEventType(const AtomString& eventType) const
394{
395 return eventType == gesturestartEvent || eventType == gesturechangeEvent || eventType == gestureendEvent;
396}
397
398inline bool EventNames::isTouchScrollBlockingEventType(const AtomString& eventType) const
399{
400 return eventType == touchstartEvent
401 || eventType == touchmoveEvent;
402}
403
404inline bool EventNames::isTouchRelatedEventType(const AtomString& eventType, EventTarget& target) const
405{
406#if ENABLE(TOUCH_EVENTS)
407 if (is<Node>(target) && downcast<Node>(target).document().quirks().shouldDispatchSimulatedMouseEvents(&target)) {
408 if (eventType == mousedownEvent || eventType == mousemoveEvent || eventType == mouseupEvent)
409 return true;
410 }
411#endif
412 UNUSED_PARAM(target);
413 return eventType == touchstartEvent
414 || eventType == touchmoveEvent
415 || eventType == touchendEvent
416 || eventType == touchcancelEvent
417 || eventType == touchforcechangeEvent
418 || eventType == pointeroverEvent
419 || eventType == pointerenterEvent
420 || eventType == pointerdownEvent
421 || eventType == pointermoveEvent
422 || eventType == pointerupEvent
423 || eventType == pointeroutEvent
424 || eventType == pointerleaveEvent
425 || eventType == pointercancelEvent;
426}
427
428inline bool EventNames::isWheelEventType(const AtomString& eventType) const
429{
430 return eventType == wheelEvent
431 || eventType == mousewheelEvent;
432}
433
434inline std::array<std::reference_wrapper<const AtomString>, 13> EventNames::touchRelatedEventNames() const
435{
436 return { { touchstartEvent, touchmoveEvent, touchendEvent, touchcancelEvent, touchforcechangeEvent, pointeroverEvent, pointerenterEvent, pointerdownEvent, pointermoveEvent, pointerupEvent, pointeroutEvent, pointerleaveEvent, pointercancelEvent } };
437}
438
439inline std::array<std::reference_wrapper<const AtomString>, 16> EventNames::extendedTouchRelatedEventNames() const
440{
441 return { { touchstartEvent, touchmoveEvent, touchendEvent, touchcancelEvent, touchforcechangeEvent, pointeroverEvent, pointerenterEvent, pointerdownEvent, pointermoveEvent, pointerupEvent, pointeroutEvent, pointerleaveEvent, pointercancelEvent, mousedownEvent, mousemoveEvent, mouseupEvent } };
442}
443
444inline std::array<std::reference_wrapper<const AtomString>, 3> EventNames::gestureEventNames() const
445{
446 return { { gesturestartEvent, gesturechangeEvent, gestureendEvent } };
447}

454 || eventType == gamepaddisconnectedEvent;
455}
