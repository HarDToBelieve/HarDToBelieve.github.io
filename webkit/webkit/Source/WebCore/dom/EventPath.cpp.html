38static inline bool shouldEventCrossShadowBoundary(Event& event, ShadowRoot& shadowRoot, EventTarget& target)
39{
40#if ENABLE(FULLSCREEN_API) && ENABLE(VIDEO)
41 // Video-only full screen is a mode where we use the shadow DOM as an implementation
42 // detail that should not be detectable by the web content.
43 if (is<Node>(target)) {
44 if (auto* element = downcast<Node>(target).document().fullscreenManager().currentFullscreenElement()) {
45 // FIXME: We assume that if the full screen element is a media element that it's
46 // the video-only full screen. Both here and elsewhere. But that is probably wrong.
47 if (element->isMediaElement() && shadowRoot.host() == element)
48 return false;
49 }
50 }
51#endif
52
53 bool targetIsInShadowRoot = is<Node>(target) && &downcast<Node>(target).treeScope().rootNode() == &shadowRoot;
54 return !targetIsInShadowRoot || event.composed();
55}

82EventPath::EventPath(Node& originalTarget, Event& event)
83{
84 buildPath(originalTarget, event);
85
86 if (auto* relatedTarget = event.relatedTarget(); is<Node>(relatedTarget) && !m_path.isEmpty())
87 setRelatedTarget(originalTarget, downcast<Node>(*relatedTarget));
88
89#if ENABLE(TOUCH_EVENTS)
90 if (is<TouchEvent>(event))
91 retargetTouchLists(downcast<TouchEvent>(event));

95void EventPath::buildPath(Node& originalTarget, Event& event)
96{
97 EventContext::Type contextType = [&]() {
98 if (is<MouseEvent>(event) || event.isFocusEvent())
99 return EventContext::Type::MouseOrFocus;
100#if ENABLE(TOUCH_EVENTS)
101 if (is<TouchEvent>(event))
102 return EventContext::Type::Touch;
103#endif
104 return EventContext::Type::Normal;
105 }();
106
107 Node* node = nodeOrHostIfPseudoElement(&originalTarget);
108 Node* target = node ? eventTargetRespectingTargetRules(*node) : nullptr;
109 int closedShadowDepth = 0;
110 // Depths are used to decided which nodes are excluded in event.composedPath when the tree is mutated during event dispatching.
111 // They could be negative for nodes outside the shadow tree of the target node.
112 while (node) {
113 while (node) {
114 m_path.append(EventContext { contextType, *node, eventTargetRespectingTargetRules(*node), target, closedShadowDepth });
115
116 if (is<ShadowRoot>(*node))
117 break;
118
119 ContainerNode* parent = node->parentNode();
120 if (UNLIKELY(!parent)) {
121 // https://dom.spec.whatwg.org/#interface-document
122 if (is<Document>(*node) && event.type() != eventNames().loadEvent) {
123 ASSERT(target);
124 if (target) {
125 if (auto* window = downcast<Document>(*node).domWindow())
126 m_path.append(EventContext { EventContext::Type::Window, node, window, target, closedShadowDepth });
127 }
128 }
129 return;
130 }
131
132 if (auto* shadowRootOfParent = parent->shadowRoot(); UNLIKELY(shadowRootOfParent)) {
133 if (auto* assignedSlot = shadowRootOfParent->findAssignedSlot(*node)) {
134 if (shadowRootOfParent->mode() != ShadowRootMode::Open)
135 closedShadowDepth++;
136 // node is assigned to a slot. Continue dispatching the event at this slot.
137 parent = assignedSlot;
138 }
139 }
140 node = parent;
141 }
142
143 bool exitingShadowTreeOfTarget = &target->treeScope() == &node->treeScope();
144 ShadowRoot& shadowRoot = downcast<ShadowRoot>(*node);
145 if (!shouldEventCrossShadowBoundary(event, shadowRoot, originalTarget))
146 return;
147 node = shadowRoot.host();
148 if (shadowRoot.mode() != ShadowRootMode::Open)
149 closedShadowDepth--;
150 if (exitingShadowTreeOfTarget)
151 target = eventTargetRespectingTargetRules(*node);
152 }
153}

155void EventPath::setRelatedTarget(Node& origin, Node& relatedNode)
156{
157 RelatedNodeRetargeter retargeter(relatedNode, *m_path[0].node());
158
159 bool originIsRelatedTarget = &origin == &relatedNode;
160 Node& rootNodeInOriginTreeScope = origin.treeScope().rootNode();
161 TreeScope* previousTreeScope = nullptr;
162 size_t originalEventPathSize = m_path.size();
163 for (unsigned contextIndex = 0; contextIndex < originalEventPathSize; contextIndex++) {
164 auto& context = m_path[contextIndex];
165 if (!context.isMouseOrFocusEventContext()) {
166 ASSERT(context.isWindowContext());
167 continue;
168 }
169
170 Node& currentTarget = *context.node();
171 TreeScope& currentTreeScope = currentTarget.treeScope();
172 if (UNLIKELY(previousTreeScope && &currentTreeScope != previousTreeScope))
173 retargeter.moveToNewTreeScope(previousTreeScope, currentTreeScope);
174
175 Node* currentRelatedNode = retargeter.currentNode(currentTarget);
176 if (UNLIKELY(!originIsRelatedTarget && context.target() == currentRelatedNode)) {
177 m_path.shrink(contextIndex);
178 break;
179 }
180
181 context.setRelatedTarget(currentRelatedNode);
182
183 if (UNLIKELY(originIsRelatedTarget && context.node() == &rootNodeInOriginTreeScope)) {
184 m_path.shrink(contextIndex + 1);
185 break;
186 }
187
188 previousTreeScope = &currentTreeScope;
189 }
190}

194void EventPath::retargetTouch(EventContext::TouchListType type, const Touch& touch)
195{
196 auto* eventTarget = touch.target();
197 if (!is<Node>(eventTarget))
198 return;
199
200 RelatedNodeRetargeter retargeter(downcast<Node>(*eventTarget), *m_path[0].node());
201 TreeScope* previousTreeScope = nullptr;
202 for (auto& context : m_path) {
203 Node& currentTarget = *context.node();
204 TreeScope& currentTreeScope = currentTarget.treeScope();
205 if (UNLIKELY(previousTreeScope && &currentTreeScope != previousTreeScope))
206 retargeter.moveToNewTreeScope(previousTreeScope, currentTreeScope);
207
208 if (context.isTouchEventContext()) {
209 Node* currentRelatedNode = retargeter.currentNode(currentTarget);
210 context.touchList(type).append(touch.cloneWithNewTarget(currentRelatedNode));
211 } else
212 ASSERT(context.isWindowContext());
213
214 previousTreeScope = &currentTreeScope;
215 }
216}

233// https://dom.spec.whatwg.org/#dom-event-composedpath
234// Any node whose depth computed in EventPath::buildPath is greater than the context object is excluded.
235// Because we can exit out of a closed shadow tree and re-enter another closed shadow tree via a slot,
236// we decrease the *allowed depth* whenever we moved to a "shallower" (closer-to-document) tree.
237Vector<EventTarget*> EventPath::computePathUnclosedToTarget(const EventTarget& target) const
238{
239 Vector<EventTarget*> path;
240 auto pathSize = m_path.size();
241 RELEASE_ASSERT(pathSize);
242 path.reserveInitialCapacity(pathSize);
243
244 auto currentTargetIndex = m_path.findMatching([&target] (auto& context) {
245 return context.currentTarget() == &target;
246 });
247 RELEASE_ASSERT(currentTargetIndex != notFound);
248 auto currentTargetDepth = m_path[currentTargetIndex].closedShadowDepth();
249
250 auto appendTargetWithLesserDepth = [&path] (const EventContext& currentContext, int& currentDepthAllowed) {
251 auto depth = currentContext.closedShadowDepth();
252 bool contextIsInsideInnerShadowTree = depth > currentDepthAllowed;
253 if (contextIsInsideInnerShadowTree)
254 return;
255 bool movedOutOfShadowTree = depth < currentDepthAllowed;
256 if (movedOutOfShadowTree)
257 currentDepthAllowed = depth;
258 path.uncheckedAppend(currentContext.currentTarget());
259 };
260
261 auto currentDepthAllowed = currentTargetDepth;
262 auto i = currentTargetIndex;
263 do {
264 appendTargetWithLesserDepth(m_path[i], currentDepthAllowed);
265 } while (i--);
266 path.reverse();
267
268 currentDepthAllowed = currentTargetDepth;
269 for (auto i = currentTargetIndex + 1; i < pathSize; ++i)
270 appendTargetWithLesserDepth(m_path[i], currentDepthAllowed);
271
272 return path;
273}

284static Node* moveOutOfAllShadowRoots(Node& startingNode)
285{
286 Node* node = &startingNode;
287 while (node->isInShadowTree())
288 node = downcast<ShadowRoot>(node->treeScope().rootNode()).host();
289 return node;
290}

292RelatedNodeRetargeter::RelatedNodeRetargeter(Node& relatedNode, Node& target)
293 : m_relatedNode(relatedNode)
294 , m_retargetedRelatedNode(&relatedNode)
295{
296 auto& targetTreeScope = target.treeScope();
297 TreeScope* currentTreeScope = &m_relatedNode.treeScope();
298 if (LIKELY(currentTreeScope == &targetTreeScope && target.isConnected() && m_relatedNode.isConnected()))
299 return;
300
301 if (&currentTreeScope->documentScope() != &targetTreeScope.documentScope()) {
302 m_hasDifferentTreeRoot = true;
303 m_retargetedRelatedNode = nullptr;
304 return;
305 }
306 if (relatedNode.isConnected() != target.isConnected()) {
307 m_hasDifferentTreeRoot = true;
308 m_retargetedRelatedNode = moveOutOfAllShadowRoots(relatedNode);
309 return;
310 }
311
312 collectTreeScopes();
313
314 // FIXME: We should collect this while constructing the event path.
315 Vector<TreeScope*, 8> targetTreeScopeAncestors;
316 for (TreeScope* currentTreeScope = &targetTreeScope; currentTreeScope; currentTreeScope = currentTreeScope->parentTreeScope())
317 targetTreeScopeAncestors.append(currentTreeScope);
318 ASSERT_WITH_SECURITY_IMPLICATION(!targetTreeScopeAncestors.isEmpty());
319
320 unsigned i = m_ancestorTreeScopes.size();
321 unsigned j = targetTreeScopeAncestors.size();
322 ASSERT_WITH_SECURITY_IMPLICATION(m_ancestorTreeScopes.last() == targetTreeScopeAncestors.last());
323 while (m_ancestorTreeScopes[i - 1] == targetTreeScopeAncestors[j - 1]) {
324 i--;
325 j--;
326 if (!i || !j)
327 break;
328 }
329
330 bool lowestCommonAncestorIsDocumentScope = i + 1 == m_ancestorTreeScopes.size();
331 if (lowestCommonAncestorIsDocumentScope && !relatedNode.isConnected() && !target.isConnected()) {
332 Node& relatedNodeAncestorInDocumentScope = i ? *downcast<ShadowRoot>(m_ancestorTreeScopes[i - 1]->rootNode()).shadowHost() : relatedNode;
333 Node& targetAncestorInDocumentScope = j ? *downcast<ShadowRoot>(targetTreeScopeAncestors[j - 1]->rootNode()).shadowHost() : target;
334 if (&targetAncestorInDocumentScope.rootNode() != &relatedNodeAncestorInDocumentScope.rootNode()) {
335 m_hasDifferentTreeRoot = true;
336 m_retargetedRelatedNode = moveOutOfAllShadowRoots(relatedNode);
337 return;
338 }
339 }
340
341 m_lowestCommonAncestorIndex = i;
342 m_retargetedRelatedNode = nodeInLowestCommonAncestor();
343}

351void RelatedNodeRetargeter::moveToNewTreeScope(TreeScope* previousTreeScope, TreeScope& newTreeScope)
352{
353 if (m_hasDifferentTreeRoot)
354 return;
355
356 auto& currentRelatedNodeScope = m_retargetedRelatedNode->treeScope();
357 if (previousTreeScope != &currentRelatedNodeScope) {
358 // currentRelatedNode is still outside our shadow tree. New tree scope may contain currentRelatedNode
359 // but there is no need to re-target it. Moving into a slot (thereby a deeper shadow tree) doesn't matter.
360 return;
361 }
362
363 bool enteredSlot = newTreeScope.parentTreeScope() == previousTreeScope;
364 if (enteredSlot) {
365 if (m_lowestCommonAncestorIndex) {
366 if (m_ancestorTreeScopes.isEmpty())
367 collectTreeScopes();
368 bool relatedNodeIsInSlot = m_ancestorTreeScopes[m_lowestCommonAncestorIndex - 1] == &newTreeScope;
369 if (relatedNodeIsInSlot) {
370 m_lowestCommonAncestorIndex--;
371 m_retargetedRelatedNode = nodeInLowestCommonAncestor();
372 ASSERT(&newTreeScope == &m_retargetedRelatedNode->treeScope());
373 }
374 } else
375 ASSERT(m_retargetedRelatedNode == &m_relatedNode);
376 } else {
377 ASSERT(previousTreeScope->parentTreeScope() == &newTreeScope);
378 m_lowestCommonAncestorIndex++;
379 ASSERT_WITH_SECURITY_IMPLICATION(m_ancestorTreeScopes.isEmpty() || m_lowestCommonAncestorIndex < m_ancestorTreeScopes.size());
380 m_retargetedRelatedNode = downcast<ShadowRoot>(currentRelatedNodeScope.rootNode()).host();
381 ASSERT(&newTreeScope == &m_retargetedRelatedNode->treeScope());
382 }
383}

385inline Node* RelatedNodeRetargeter::nodeInLowestCommonAncestor()
386{
387 if (!m_lowestCommonAncestorIndex)
388 return &m_relatedNode;
389 auto& rootNode = m_ancestorTreeScopes[m_lowestCommonAncestorIndex - 1]->rootNode();
390 return downcast<ShadowRoot>(rootNode).host();
391}
392
393void RelatedNodeRetargeter::collectTreeScopes()
394{
395 ASSERT(m_ancestorTreeScopes.isEmpty());
396 for (TreeScope* currentTreeScope = &m_relatedNode.treeScope(); currentTreeScope; currentTreeScope = currentTreeScope->parentTreeScope())
397 m_ancestorTreeScopes.append(currentTreeScope);
398 ASSERT_WITH_SECURITY_IMPLICATION(!m_ancestorTreeScopes.isEmpty());
399}
