61inline Node* EventPath::eventTargetRespectingTargetRules(Node& referenceNode)
62{
63 if (is<PseudoElement>(referenceNode))
64 return downcast<PseudoElement>(referenceNode).hostElement();
65
66 // Events sent to elements inside an SVG use element's shadow tree go to the use element.
67 if (is<SVGElement>(referenceNode)) {
68 if (auto useElement = downcast<SVGElement>(referenceNode).correspondingUseElement())
69 return useElement.get();
70 }
71
72 return &referenceNode;
73}
