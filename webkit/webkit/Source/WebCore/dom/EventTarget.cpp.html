81bool EventTarget::addEventListener(const AtomString& eventType, Ref<EventListener>&& listener, const AddEventListenerOptions& options)
82{
83#if ASSERT_ENABLED
84 listener->checkValidityForEventTarget(*this);
85#endif
86
87 if (options.signal && options.signal->aborted())
88 return false;
89
90 auto passive = options.passive;
91
92 if (!passive.has_value() && Quirks::shouldMakeEventListenerPassive(*this, eventType, listener.get()))
93 passive = true;
94
95 bool listenerCreatedFromScript = listener->type() == EventListener::JSEventListenerType && !listener->wasCreatedFromMarkup();
96
97 if (!ensureEventTargetData().eventListenerMap.add(eventType, listener.copyRef(), { options.capture, passive.value_or(false), options.once }))
98 return false;
99
100 if (options.signal) {
101 options.signal->addAlgorithm([weakThis = makeWeakPtr(*this), eventType, listener = makeWeakPtr(listener.get()), capture = options.capture] {
102 if (weakThis && listener)
103 weakThis->removeEventListener(eventType, *listener, capture);
104 });
105 }
106
107 if (listenerCreatedFromScript)
108 InspectorInstrumentation::didAddEventListener(*this, eventType, listener.get(), options.capture);
109
110 if (eventNames().isWheelEventType(eventType))
111 invalidateEventListenerRegions();
112
113 eventListenersDidChange();
114 return true;
115}

145bool EventTarget::removeEventListener(const AtomString& eventType, EventListener& listener, const EventListenerOptions& options)
146{
147 auto* data = eventTargetData();
148 if (!data)
149 return false;
150
151 InspectorInstrumentation::willRemoveEventListener(*this, eventType, listener, options.capture);
152
153 if (data->eventListenerMap.remove(eventType, listener, options.capture)) {
154 if (eventNames().isWheelEventType(eventType))
155 invalidateEventListenerRegions();
156
157 eventListenersDidChange();
158 return true;
159 }
160 return false;
161}

163bool EventTarget::setAttributeEventListener(const AtomString& eventType, RefPtr<EventListener>&& listener, DOMWrapperWorld& isolatedWorld)
164{
165 auto* existingListener = attributeEventListener(eventType, isolatedWorld);
166 if (!listener) {
167 if (existingListener)
168 removeEventListener(eventType, *existingListener, false);
169 return false;
170 }
171 if (existingListener) {
172 InspectorInstrumentation::willRemoveEventListener(*this, eventType, *existingListener, false);
173
174#if ASSERT_ENABLED
175 listener->checkValidityForEventTarget(*this);
176#endif
177
178 auto listenerPointer = listener.copyRef();
179 eventTargetData()->eventListenerMap.replace(eventType, *existingListener, listener.releaseNonNull(), { });
180
181 InspectorInstrumentation::didAddEventListener(*this, eventType, *listenerPointer, false);
182
183 return true;
184 }
185 return addEventListener(eventType, listener.releaseNonNull(), { });
186}

188EventListener* EventTarget::attributeEventListener(const AtomString& eventType, DOMWrapperWorld& isolatedWorld)
189{
190 for (auto& eventListener : eventListeners(eventType)) {
191 auto& listener = eventListener->callback();
192 if (!listener.isAttribute())
193 continue;
194
195 auto& listenerWorld = downcast<JSEventListener>(listener).isolatedWorld();
196 if (&listenerWorld == &isolatedWorld)
197 return &listener;
198 }
199
200 return nullptr;
201}

209ExceptionOr<bool> EventTarget::dispatchEventForBindings(Event& event)
210{
211 if (!event.isInitialized() || event.isBeingDispatched())
212 return Exception { InvalidStateError };
213
214 if (!scriptExecutionContext())
215 return false;
216
217 event.setUntrusted();
218
219 dispatchEvent(event);
220 return event.legacyReturnValue();
221}

223void EventTarget::dispatchEvent(Event& event)
224{
225 // FIXME: We should always use EventDispatcher.
226 ASSERT(event.isInitialized());
227 ASSERT(!event.isBeingDispatched());
228
229 event.setTarget(this);
230 event.setCurrentTarget(this);
231 event.setEventPhase(Event::AT_TARGET);
232 event.resetBeforeDispatch();
233 fireEventListeners(event, EventInvokePhase::Capturing);
234 fireEventListeners(event, EventInvokePhase::Bubbling);
235 event.resetAfterDispatch();
236}

242static const AtomString& legacyType(const Event& event)
243{
244 if (event.type() == eventNames().animationendEvent)
245 return eventNames().webkitAnimationEndEvent;
246
247 if (event.type() == eventNames().animationstartEvent)
248 return eventNames().webkitAnimationStartEvent;
249
250 if (event.type() == eventNames().animationiterationEvent)
251 return eventNames().webkitAnimationIterationEvent;
252
253 if (event.type() == eventNames().transitionendEvent)
254 return eventNames().webkitTransitionEndEvent;
255
256 // FIXME: This legacy name is not part of the specification (https://dom.spec.whatwg.org/#dispatching-events).
257 if (event.type() == eventNames().wheelEvent)
258 return eventNames().mousewheelEvent;
259
260 return nullAtom();
261}

263// https://dom.spec.whatwg.org/#concept-event-listener-invoke
264void EventTarget::fireEventListeners(Event& event, EventInvokePhase phase)
265{
266 ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::isEventAllowedInMainThread());
267 ASSERT(event.isInitialized());
268
269 auto* data = eventTargetData();
270 if (!data)
271 return;
272
273 SetForScope<bool> firingEventListenersScope(data->isFiringEventListeners, true);
274
275 if (auto* listenersVector = data->eventListenerMap.find(event.type())) {
276 innerInvokeEventListeners(event, *listenersVector, phase);
277 return;
278 }
279
280 // Only fall back to legacy types for trusted events.
281 if (!event.isTrusted())
282 return;
283
284 const AtomString& legacyTypeName = legacyType(event);
285 if (!legacyTypeName.isNull()) {
286 if (auto* legacyListenersVector = data->eventListenerMap.find(legacyTypeName)) {
287 AtomString typeName = event.type();
288 event.setType(legacyTypeName);
289 innerInvokeEventListeners(event, *legacyListenersVector, phase);
290 event.setType(typeName);
291 }
292 }
293}

295// Intentionally creates a copy of the listeners vector to avoid event listeners added after this point from being run.
296// Note that removal still has an effect due to the removed field in RegisteredEventListener.
297// https://dom.spec.whatwg.org/#concept-event-listener-inner-invoke
298void EventTarget::innerInvokeEventListeners(Event& event, EventListenerVector listeners, EventInvokePhase phase)
299{
300 Ref<EventTarget> protectedThis(*this);
301 ASSERT(!listeners.isEmpty());
302 ASSERT(scriptExecutionContext());
303
304 auto& context = *scriptExecutionContext();
305 bool contextIsDocument = is<Document>(context);
306 if (contextIsDocument)
307 InspectorInstrumentation::willDispatchEvent(downcast<Document>(context), event);
308
309 for (auto& registeredListener : listeners) {
310 if (UNLIKELY(registeredListener->wasRemoved()))
311 continue;
312
313 if (phase == EventInvokePhase::Capturing && !registeredListener->useCapture())
314 continue;
315 if (phase == EventInvokePhase::Bubbling && registeredListener->useCapture())
316 continue;
317
318 if (InspectorInstrumentation::isEventListenerDisabled(*this, event.type(), registeredListener->callback(), registeredListener->useCapture()))
319 continue;
320
321 // If stopImmediatePropagation has been called, we just break out immediately, without
322 // handling any more events on this target.
323 if (event.immediatePropagationStopped())
324 break;
325
326 // Make sure the JS wrapper and function stay alive until the end of this scope. Otherwise,
327 // event listeners with 'once' flag may get collected as soon as they get unregistered below,
328 // before we call the js function.
329 JSC::EnsureStillAliveScope wrapperProtector(registeredListener->callback().wrapper());
330 JSC::EnsureStillAliveScope jsFunctionProtector(registeredListener->callback().jsFunction());
331
332 // Do this before invocation to avoid reentrancy issues.
333 if (registeredListener->isOnce())
334 removeEventListener(event.type(), registeredListener->callback(), registeredListener->useCapture());
335
336 if (registeredListener->isPassive())
337 event.setInPassiveListener(true);
338
339#if ASSERT_ENABLED
340 registeredListener->callback().checkValidityForEventTarget(*this);
341#endif
342
343 InspectorInstrumentation::willHandleEvent(context, event, *registeredListener);
344 registeredListener->callback().handleEvent(context, event);
345 InspectorInstrumentation::didHandleEvent(context, event, *registeredListener);
346
347 if (registeredListener->isPassive())
348 event.setInPassiveListener(false);
349 }
350
351 if (contextIsDocument)
352 InspectorInstrumentation::didDispatchEvent(downcast<Document>(context), event);
353}

362const EventListenerVector& EventTarget::eventListeners(const AtomString& eventType)
363{
364 auto* data = eventTargetData();
365 auto* listenerVector = data ? data->eventListenerMap.find(eventType) : nullptr;

367 return listenerVector ? *listenerVector : emptyVector.get();

370void EventTarget::removeAllEventListeners()
371{
372 auto& threadData = threadGlobalData();
373 RELEASE_ASSERT(!threadData.isInRemoveAllEventListeners());
374
375 threadData.setIsInRemoveAllEventListeners(true);
376
377 auto* data = eventTargetData();
378 if (data && !data->eventListenerMap.isEmpty()) {
379 if (data->eventListenerMap.contains(eventNames().wheelEvent) || data->eventListenerMap.contains(eventNames().mousewheelEvent))
380 invalidateEventListenerRegions();
381
382 data->eventListenerMap.clear();
383 eventListenersDidChange();
384 }
385
386 threadData.setIsInRemoveAllEventListeners(false);
387}

389template<typename Visitor>
390void EventTarget::visitJSEventListeners(Visitor& visitor)
391{
392 EventTargetData* data = eventTargetDataConcurrently();
393 if (!data)
394 return;
395
396 Locker locker { data->eventListenerMap.lock() };
397 EventListenerIterator iterator(&data->eventListenerMap);
398 while (auto* listener = iterator.nextListener())
399 listener->visitJSFunction(visitor);
400}

405void EventTarget::invalidateEventListenerRegions()
406{
407 if (is<Element>(*this)) {
408 downcast<Element>(*this).invalidateEventListenerRegions();
409 return;
410 }
411
412 auto* document = [&]() -> Document* {
413 if (is<Document>(*this))
414 return &downcast<Document>(*this);
415 if (is<DOMWindow>(*this))
416 return downcast<DOMWindow>(*this).document();
417 return nullptr;
418 }();
419
420 if (document)
421 document->invalidateEventListenerRegions();
422}
