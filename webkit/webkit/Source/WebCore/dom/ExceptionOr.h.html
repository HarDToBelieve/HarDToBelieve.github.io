36template<typename T> class ExceptionOr {
37public:
38 using ReturnType = T;
39
40 ExceptionOr(Exception&&);
41 ExceptionOr(ReturnType&&);
42 template<typename OtherType> ExceptionOr(const OtherType&, typename std::enable_if<std::is_scalar<OtherType>::value && std::is_convertible<OtherType, ReturnType>::value>::type* = nullptr);
43
44 bool hasException() const;
45 const Exception& exception() const;
46 Exception releaseException();
47 const ReturnType& returnValue() const;
48 ReturnType releaseReturnValue();
49
50private:
51 Expected<ReturnType, Exception> m_value;
52#if ASSERT_ENABLED
53 bool m_wasReleased { false };
54#endif
55};

110template<typename ReturnType> inline bool ExceptionOr<ReturnType>::hasException() const
111{
112 return !m_value.has_value();

115template<typename ReturnType> inline const Exception& ExceptionOr<ReturnType>::exception() const
116{
117 ASSERT(!m_wasReleased);
118 return m_value.error();
119}

121template<typename ReturnType> inline Exception ExceptionOr<ReturnType>::releaseException()
122{
123 ASSERT(!std::exchange(m_wasReleased, true));
124 return WTFMove(m_value.error());
125}

196inline ExceptionOr<void> isolatedCopy(ExceptionOr<void>&& value)
197{
198 if (value.hasException())
199 return isolatedCopy(value.releaseException());
200 return { };
201}

203template <typename T> inline constexpr bool IsExceptionOr = WTF::IsTemplate<std::decay_t<T>, ExceptionOr>::value;

205template <typename T, bool isExceptionOr = IsExceptionOr<T>> struct TypeOrExceptionOrUnderlyingTypeImpl;
206template <typename T> struct TypeOrExceptionOrUnderlyingTypeImpl<T, true> { using Type = typename T::ReturnType; };
207template <typename T> struct TypeOrExceptionOrUnderlyingTypeImpl<T, false> { using Type = T; };
208template <typename T> using TypeOrExceptionOrUnderlyingType = typename TypeOrExceptionOrUnderlyingTypeImpl<T>::Type;

212namespace WTF {
213template<typename T> struct CrossThreadCopierBase<false, false, WebCore::ExceptionOr<T> > {
214 typedef WebCore::ExceptionOr<T> Type;
215 static Type copy(const Type& source)
216 {
217 if (source.hasException())
218 return crossThreadCopy(source.exception());
219 return crossThreadCopy(source.returnValue());
220 }
