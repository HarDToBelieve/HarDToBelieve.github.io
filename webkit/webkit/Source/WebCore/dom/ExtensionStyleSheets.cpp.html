60static Ref<CSSStyleSheet> createExtensionsStyleSheet(Document& document, URL url, const String& text, UserStyleLevel level)
61{
62 auto contents = StyleSheetContents::create(url.string(), CSSParserContext(document, url));
63 auto styleSheet = CSSStyleSheet::create(contents.get(), document, true);
64
65 contents->setIsUserStyleSheet(level == UserStyleUserLevel);
66 contents->parseString(text);
67
68 return styleSheet;
69}

71CSSStyleSheet* ExtensionStyleSheets::pageUserSheet()
72{
73 if (m_pageUserSheet)
74 return m_pageUserSheet.get();
75
76 Page* owningPage = m_document.page();
77 if (!owningPage)
78 return 0;
79
80 String userSheetText = owningPage->userStyleSheet();
81 if (userSheetText.isEmpty())
82 return 0;
83
84 m_pageUserSheet = createExtensionsStyleSheet(m_document, m_document.settings().userStyleSheetLocation(), userSheetText, UserStyleUserLevel);
85
86 return m_pageUserSheet.get();
87}

116void ExtensionStyleSheets::updateInjectedStyleSheetCache() const
117{
118 if (m_injectedStyleSheetCacheValid)
119 return;
120
121 m_injectedStyleSheetCacheValid = true;
122 m_injectedUserStyleSheets.clear();
123 m_injectedAuthorStyleSheets.clear();
124 m_injectedStyleSheetToSource.clear();
125
126 Page* owningPage = m_document.page();
127 if (!owningPage)
128 return;
129
130 auto addStyleSheet = [&](const UserStyleSheet& userStyleSheet) {
131 auto sheet = createExtensionsStyleSheet(const_cast<Document&>(m_document), userStyleSheet.url(), userStyleSheet.source(), userStyleSheet.level());
132
133 m_injectedStyleSheetToSource.set(sheet.copyRef(), userStyleSheet.source());
134
135 if (sheet->contents().isUserStyleSheet())
136 m_injectedUserStyleSheets.append(WTFMove(sheet));
137 else
138 m_injectedAuthorStyleSheets.append(WTFMove(sheet));
139 };
140
141 for (const auto& userStyleSheet : m_pageSpecificStyleSheets)
142 addStyleSheet(userStyleSheet);
143
144 owningPage->userContentProvider().forEachUserStyleSheet([&](const UserStyleSheet& userStyleSheet) {
145 if (userStyleSheet.pageID())
146 return;
147
148 if (userStyleSheet.injectedFrames() == UserContentInjectedFrames::InjectInTopFrameOnly && m_document.ownerElement())
149 return;
150
151 if (!UserContentURLPattern::matchesPatterns(m_document.url(), userStyleSheet.allowlist(), userStyleSheet.blocklist()))
152 return;
153
154 addStyleSheet(userStyleSheet);
155 });
156}

158void ExtensionStyleSheets::injectPageSpecificUserStyleSheet(const UserStyleSheet& userStyleSheet)
159{
160 m_pageSpecificStyleSheets.append(userStyleSheet);
161 invalidateInjectedStyleSheetCache();
162}

164void ExtensionStyleSheets::removePageSpecificUserStyleSheet(const UserStyleSheet& userStyleSheet)
165{
166 bool removedStyleSheet = m_pageSpecificStyleSheets.removeFirstMatching([&](const auto& styleSheet) {
167 return styleSheet.url() == userStyleSheet.url();
168 });
169
170 if (removedStyleSheet)
171 invalidateInjectedStyleSheetCache();
172}

174void ExtensionStyleSheets::invalidateInjectedStyleSheetCache()
175{
176 m_injectedStyleSheetCacheValid = false;
177 m_document.styleScope().didChangeStyleSheetEnvironment();
178}

180void ExtensionStyleSheets::addUserStyleSheet(Ref<StyleSheetContents>&& userSheet)
181{
182 ASSERT(userSheet.get().isUserStyleSheet());
183 m_userStyleSheets.append(CSSStyleSheet::create(WTFMove(userSheet), m_document));
184 m_document.styleScope().didChangeStyleSheetEnvironment();
185}

195void ExtensionStyleSheets::addDisplayNoneSelector(const String& identifier, const String& selector, uint32_t selectorID)
196{
197 auto result = m_contentExtensionSelectorSheets.add(identifier, nullptr);
198 if (result.isNewEntry) {
199 result.iterator->value = ContentExtensionStyleSheet::create(m_document);
200 m_userStyleSheets.append(&result.iterator->value->styleSheet());
201 }
202
203 if (result.iterator->value->addDisplayNoneSelector(selector, selectorID))
204 m_document.styleScope().didChangeStyleSheetEnvironment();
205}

207void ExtensionStyleSheets::maybeAddContentExtensionSheet(const String& identifier, StyleSheetContents& sheet)
208{
209 ASSERT(sheet.isUserStyleSheet());
210
211 if (m_contentExtensionSheets.contains(identifier))
212 return;
213
214 Ref<CSSStyleSheet> cssSheet = CSSStyleSheet::create(sheet, m_document);
215 m_contentExtensionSheets.set(identifier, &cssSheet.get());
216 m_userStyleSheets.append(adoptRef(cssSheet.leakRef()));
217 m_document.styleScope().didChangeStyleSheetEnvironment();
218
219}

227void ExtensionStyleSheets::detachFromDocument()
228{
229 if (m_pageUserSheet)
230 m_pageUserSheet->detachFromDocument();
231 for (auto& sheet : m_injectedUserStyleSheets)
232 sheet->detachFromDocument();
233 for (auto& sheet : m_injectedAuthorStyleSheets)
234 sheet->detachFromDocument();
235 for (auto& sheet : m_userStyleSheets)
236 sheet->detachFromDocument();
237 for (auto& sheet : m_authorStyleSheetsForTesting)
238 sheet->detachFromDocument();
239}
