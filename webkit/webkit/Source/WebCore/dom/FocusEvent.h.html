30#include "UIEvent.h"

32namespace WebCore {
33
34class Node;
35
36class FocusEvent final : public UIEvent {
37 WTF_MAKE_ISO_ALLOCATED(FocusEvent);
38public:
39 static Ref<FocusEvent> create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<WindowProxy>&& view, int detail, RefPtr<EventTarget>&& relatedTarget)
40 {
41 return adoptRef(*new FocusEvent(type, canBubble, cancelable, WTFMove(view), detail, WTFMove(relatedTarget)));
42 }
43
44 static Ref<FocusEvent> createForBindings()
45 {
46 return adoptRef(*new FocusEvent);
47 }
48
49 struct Init : UIEventInit {
50 RefPtr<EventTarget> relatedTarget;
51 };
52
53 static Ref<FocusEvent> create(const AtomString& type, const Init& initializer)
54 {
55 return adoptRef(*new FocusEvent(type, initializer));
56 }
57
58 EventTarget* relatedTarget() const final { return m_relatedTarget.get(); }
59
60private:
61 FocusEvent() = default;
62 FocusEvent(const AtomString& type, CanBubble, IsCancelable, RefPtr<WindowProxy>&&, int, RefPtr<EventTarget>&&);
63 FocusEvent(const AtomString& type, const Init&);
64
65 EventInterface eventInterface() const final;
66 bool isFocusEvent() const final;
67
68 void setRelatedTarget(EventTarget* relatedTarget) final { m_relatedTarget = relatedTarget; }
69
70 RefPtr<EventTarget> m_relatedTarget;
71};

73} // namespace WebCore
