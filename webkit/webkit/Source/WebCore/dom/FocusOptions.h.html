26#pragma once
27
28#include "FocusDirection.h"
29#include "SelectionRestorationMode.h"
30
31namespace WebCore {
32
33enum class FocusRemovalEventsMode : bool { Dispatch, DoNotDispatch };
34
35enum class FocusTrigger : bool { Other, Click };
36
37enum class FocusVisibility : bool { Invisible, Visible };
38
39struct FocusOptions {
40 SelectionRestorationMode selectionRestorationMode { SelectionRestorationMode::RestoreOrSelectAll };
41 FocusDirection direction { FocusDirection::None };
42 FocusRemovalEventsMode removalEventsMode { FocusRemovalEventsMode::Dispatch };
43 FocusTrigger trigger { FocusTrigger::Other };
44 FocusVisibility visibility { FocusVisibility::Invisible };
45 bool preventScroll { false };
46};
47
48} // namespace WebCore
