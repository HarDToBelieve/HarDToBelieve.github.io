60void FullscreenManager::requestFullscreenForElement(Element* element, FullscreenCheckType checkType)
61{
62 if (!element)
63 element = documentElement();
64
65 auto failedPreflights = [this, weakThis = makeWeakPtr(*this)](auto element) mutable {
66 m_fullscreenErrorEventTargetQueue.append(WTFMove(element));
67 m_document.eventLoop().queueTask(TaskSource::MediaElement, [weakThis = WTFMove(weakThis)]() mutable {
68 if (weakThis)
69 weakThis->dispatchFullscreenChangeEvents();
70 });
71 };
72
73 // 1. If any of the following conditions are true, terminate these steps and queue a task to fire
74 // an event named fullscreenerror with its bubbles attribute set to true on the context object's
75 // node document:
76
77 // This algorithm is not allowed to show a pop-up:
78 // An algorithm is allowed to show a pop-up if, in the task in which the algorithm is running, either:
79 // - an activation behavior is currently being processed whose click event was trusted, or
80 // - the event listener for a trusted click event is being handled.
81 if (!UserGestureIndicator::processingUserGesture()) {
82 ERROR_LOG(LOGIDENTIFIER, "!processingUserGesture; failing.");
83 failedPreflights(WTFMove(element));
84 return;
85 }
86
87 // We do not allow pressing the Escape key as a user gesture to enter fullscreen since this is the key
88 // to exit fullscreen.
89 if (UserGestureIndicator::currentUserGesture()->gestureType() == UserGestureType::EscapeKey) {
90 ERROR_LOG(LOGIDENTIFIER, "Current gesture is EscapeKey; failing.");
91 document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, "The Escape key may not be used as a user gesture to enter fullscreen"_s);
92 failedPreflights(WTFMove(element));
93 return;
94 }
95
96 // There is a previously-established user preference, security risk, or platform limitation.
97 if (!page() || !page()->settings().fullScreenEnabled()) {
98 ERROR_LOG(LOGIDENTIFIER, "!page() or fullscreen not enabled; failing.");
99 failedPreflights(WTFMove(element));
100 return;
101 }
102
103 bool hasKeyboardAccess = true;
104 if (!page()->chrome().client().supportsFullScreenForElement(*element, hasKeyboardAccess)) {
105 // The new full screen API does not accept a "flags" parameter, so fall back to disallowing
106 // keyboard input if the chrome client refuses to allow keyboard input.
107 hasKeyboardAccess = false;
108
109 if (!page()->chrome().client().supportsFullScreenForElement(*element, hasKeyboardAccess)) {
110 ERROR_LOG(LOGIDENTIFIER, "page does not support fullscreen for element; failing.");
111 failedPreflights(WTFMove(element));
112 return;
113 }
114 }
115
116 m_pendingFullscreenElement = element;
117
118 m_document.eventLoop().queueTask(TaskSource::MediaElement, [this, weakThis = makeWeakPtr(*this), element = makeRefPtr(element), checkType, hasKeyboardAccess, failedPreflights, identifier = LOGIDENTIFIER] () mutable {
119 if (!weakThis)
120 return;
121
122 // Don't allow fullscreen if it has been cancelled or a different fullscreen element
123 // has requested fullscreen.
124 if (m_pendingFullscreenElement != element) {
125 ERROR_LOG(identifier, "task - pending element mismatch; failing.");
126 failedPreflights(WTFMove(element));
127 return;
128 }
129
130 // Don't allow fullscreen if we're inside an exitFullscreen operation.
131 if (m_pendingExitFullscreen) {
132 ERROR_LOG(identifier, "task - pending exit fullscreen operation; failing.");
133 failedPreflights(WTFMove(element));
134 return;
135 }
136
137 // Don't allow fullscreen if document is hidden.
138 if (document().hidden()) {
139 ERROR_LOG(identifier, "task - document hidden; failing.");
140 failedPreflights(WTFMove(element));
141 return;
142 }
143
144 // The context object is not in a document.
145 if (!element->isConnected()) {
146 ERROR_LOG(identifier, "task - element not in document; failing.");
147 failedPreflights(WTFMove(element));
148 return;
149 }
150
151 // The context object's node document, or an ancestor browsing context's document does not have
152 // the fullscreen enabled flag set.
153 if (checkType == EnforceIFrameAllowFullscreenRequirement && !isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::Fullscreen, document())) {
154 ERROR_LOG(identifier, "task - ancestor document does not enable fullscreen; failing.");
155 failedPreflights(WTFMove(element));
156 return;
157 }
158
159 // The context object's node document fullscreen element stack is not empty and its top element
160 // is not an ancestor of the context object.
161 if (!m_fullscreenElementStack.isEmpty() && !m_fullscreenElementStack.last()->contains(element.get())) {
162 ERROR_LOG(identifier, "task - fullscreen stack not empty; failing.");
163 failedPreflights(WTFMove(element));
164 return;
165 }
166
167 // A descendant browsing context's document has a non-empty fullscreen element stack.
168 bool descendentHasNonEmptyStack = false;
169 for (Frame* descendant = frame() ? frame()->tree().traverseNext() : nullptr; descendant; descendant = descendant->tree().traverseNext()) {
170 if (descendant->document()->fullscreenManager().fullscreenElement()) {
171 descendentHasNonEmptyStack = true;
172 break;
173 }
174 }
175 if (descendentHasNonEmptyStack) {
176 ERROR_LOG(identifier, "task - descendent document has non-empty fullscreen stack; failing.");
177 failedPreflights(WTFMove(element));
178 return;
179 }
180
181 // 2. Let doc be element's node document. (i.e. "this")
182 Document* currentDoc = &document();
183
184 // 3. Let docs be all doc's ancestor browsing context's documents (if any) and doc.
185 Deque<Document*> docs;
186
187 do {
188 docs.prepend(currentDoc);
189 currentDoc = currentDoc->ownerElement() ? &currentDoc->ownerElement()->document() : nullptr;
190 } while (currentDoc);
191
192 // 4. For each document in docs, run these substeps:
193 Deque<Document*>::iterator current = docs.begin(), following = docs.begin();
194
195 do {
196 ++following;
197
198 // 1. Let following document be the document after document in docs, or null if there is no
199 // such document.
200 Document* currentDoc = *current;
201 Document* followingDoc = following != docs.end() ? *following : nullptr;
202
203 // 2. If following document is null, push context object on document's fullscreen element
204 // stack, and queue a task to fire an event named fullscreenchange with its bubbles attribute
205 // set to true on the document.
206 if (!followingDoc) {
207 currentDoc->fullscreenManager().pushFullscreenElementStack(*element);
208 addDocumentToFullscreenChangeEventQueue(*currentDoc);
209 continue;
210 }
211
212 // 3. Otherwise, if document's fullscreen element stack is either empty or its top element
213 // is not following document's browsing context container,
214 Element* topElement = currentDoc->fullscreenManager().fullscreenElement();
215 if (!topElement || topElement != followingDoc->ownerElement()) {
216 // ...push following document's browsing context container on document's fullscreen element
217 // stack, and queue a task to fire an event named fullscreenchange with its bubbles attribute
218 // set to true on document.
219 currentDoc->fullscreenManager().pushFullscreenElementStack(*followingDoc->ownerElement());
220 addDocumentToFullscreenChangeEventQueue(*currentDoc);
221 continue;
222 }
223
224 // 4. Otherwise, do nothing for this document. It stays the same.
225 } while (++current != docs.end());
226
227 // 5. Return, and run the remaining steps asynchronously.
228 // 6. Optionally, perform some animation.
229 m_areKeysEnabledInFullscreen = hasKeyboardAccess;
230 m_document.eventLoop().queueTask(TaskSource::MediaElement, [this, weakThis = WTFMove(weakThis), element = WTFMove(element), failedPreflights = WTFMove(failedPreflights), identifier] () mutable {
231 if (!weakThis)
232 return;
233
234 auto page = this->page();
235 if (!page || document().hidden() || m_pendingFullscreenElement != element || !element->isConnected()) {
236 ERROR_LOG(identifier, "task - page, document, or element mismatch; failing.");
237 failedPreflights(element);
238 return;
239 }
240 INFO_LOG(identifier, "task - success");
241 page->chrome().client().enterFullScreenForElement(*element.get());
242 });
243
244 // 7. Optionally, display a message indicating how the user can exit displaying the context object fullscreen.
245 });
246}

248void FullscreenManager::cancelFullscreen()
249{
250 // The Mozilla "cancelFullscreen()" API behaves like the W3C "fully exit fullscreen" behavior, which
251 // is defined as:
252 // "To fully exit fullscreen act as if the exitFullscreen() method was invoked on the top-level browsing
253 // context's document and subsequently empty that document's fullscreen element stack."
254 Document& topDocument = document().topDocument();
255 if (!topDocument.fullscreenManager().fullscreenElement()) {
256 // If there is a pending fullscreen element but no top document fullscreen element,
257 // there is a pending task in enterFullscreen(). Cause it to cancel and fire an error
258 // by clearing the pending fullscreen element.
259 m_pendingFullscreenElement = nullptr;
260 INFO_LOG(LOGIDENTIFIER, "Cancelling pending fullscreen request.");
261 return;
262 }
263
264 INFO_LOG(LOGIDENTIFIER);
265
266 // To achieve that aim, remove all the elements from the top document's stack except for the first before
267 // calling webkitExitFullscreen():
268 Vector<RefPtr<Element>> replacementFullscreenElementStack;
269 replacementFullscreenElementStack.append(topDocument.fullscreenManager().fullscreenElement());
270 topDocument.fullscreenManager().m_fullscreenElementStack.swap(replacementFullscreenElementStack);
271
272 topDocument.fullscreenManager().exitFullscreen();
273}

275void FullscreenManager::exitFullscreen()
276{
277 // The exitFullscreen() method must run these steps:
278
279 // 1. Let doc be the context object. (i.e. "this")
280 Document* currentDoc = &document();
281
282 // 2. If doc's fullscreen element stack is empty, terminate these steps.
283 if (m_fullscreenElementStack.isEmpty()) {
284 // If there is a pending fullscreen element but an empty fullscreen element stack,
285 // there is a pending task in requestFullscreenForElement(). Cause it to cancel and fire an error
286 // by clearing the pending fullscreen element.
287 INFO_LOG(LOGIDENTIFIER, "Cancelling pending fullscreen request.");
288 m_pendingFullscreenElement = nullptr;
289 return;
290 }
291
292 // 3. Let descendants be all the doc's descendant browsing context's documents with a non-empty fullscreen
293 // element stack (if any), ordered so that the child of the doc is last and the document furthest
294 // away from the doc is first.
295 Deque<RefPtr<Document>> descendants;
296 for (Frame* descendant = frame() ? frame()->tree().traverseNext() : nullptr; descendant; descendant = descendant->tree().traverseNext()) {
297 if (descendant->document()->fullscreenManager().fullscreenElement())
298 descendants.prepend(descendant->document());
299 }
300
301 // 4. For each descendant in descendants, empty descendant's fullscreen element stack, and queue a
302 // task to fire an event named fullscreenchange with its bubbles attribute set to true on descendant.
303 for (auto& document : descendants) {
304 document->fullscreenManager().clearFullscreenElementStack();
305 addDocumentToFullscreenChangeEventQueue(*document);
306 }
307
308 // 5. While doc is not null, run these substeps:
309 Element* newTop = nullptr;
310 while (currentDoc) {
311 // 1. Pop the top element of doc's fullscreen element stack.
312 currentDoc->fullscreenManager().popFullscreenElementStack();
313
314 // If doc's fullscreen element stack is non-empty and the element now at the top is either
315 // not in a document or its node document is not doc, repeat this substep.
316 newTop = currentDoc->fullscreenManager().fullscreenElement();
317 if (newTop && (!newTop->isConnected() || &newTop->document() != currentDoc))
318 continue;
319
320 // 2. Queue a task to fire an event named fullscreenchange with its bubbles attribute set to true
321 // on doc.
322 addDocumentToFullscreenChangeEventQueue(*currentDoc);
323
324 // 3. If doc's fullscreen element stack is empty and doc's browsing context has a browsing context
325 // container, set doc to that browsing context container's node document.
326 if (!newTop && currentDoc->ownerElement()) {
327 currentDoc = &currentDoc->ownerElement()->document();
328 continue;
329 }
330
331 // 4. Otherwise, set doc to null.
332 currentDoc = nullptr;
333 }
334
335 // 6. Return, and run the remaining steps asynchronously.
336 // 7. Optionally, perform some animation.
337 m_document.eventLoop().queueTask(TaskSource::MediaElement, [this, weakThis = makeWeakPtr(*this), newTop = makeRefPtr(newTop), fullscreenElement = m_fullscreenElement, identifier = LOGIDENTIFIER] {
338 if (!weakThis)
339 return;
340
341 auto* page = this->page();
342 if (!page) {
343 ERROR_LOG(identifier, "task - Document not in page; bailing.");
344 return;
345 }
346
347 // If there is a pending fullscreen element but no fullscreen element
348 // there is a pending task in requestFullscreenForElement(). Cause it to cancel and fire an error
349 // by clearing the pending fullscreen element.
350 if (!fullscreenElement && m_pendingFullscreenElement) {
351 INFO_LOG(identifier, "task - Cancelling pending fullscreen request.");
352 m_pendingFullscreenElement = nullptr;
353 return;
354 }
355
356 // Only exit out of full screen window mode if there are no remaining elements in the
357 // full screen stack.
358 if (!newTop) {
359 m_pendingExitFullscreen = true;
360 INFO_LOG(identifier, "task - Empty fullscreen stack; exiting.");
361 page->chrome().client().exitFullScreenForElement(fullscreenElement.get());
362 return;
363 }
364
365 // Otherwise, notify the chrome of the new full screen element.
366 INFO_LOG(identifier, "task - New top of fullscreen stack.");
367 page->chrome().client().enterFullScreenForElement(*newTop);
368 });
369}

388 fullscreenElement->parentElement()->invalidateStyleAndRenderersForSubtree();

391bool FullscreenManager::willEnterFullscreen(Element& element)
392{
393 if (!hasLivingRenderTree()) {
394 ERROR_LOG(LOGIDENTIFIER, "No livingRenderTree(); bailing");
395 return false;
396 }
397
398 if (backForwardCacheState() != Document::NotInBackForwardCache) {
399 ERROR_LOG(LOGIDENTIFIER, "Document in the BackForwardCache; bailing");
400 return false;
401 }
402
403 // Protect against being called after the document has been removed from the page.
404 if (!page()) {
405 ERROR_LOG(LOGIDENTIFIER, "Document no longer in page; bailing");
406 return false;
407 }
408
409 // If pending fullscreen element is unset or another element's was requested,
410 // issue a cancel fullscreen request to the client
411 if (m_pendingFullscreenElement != &element) {
412 INFO_LOG(LOGIDENTIFIER, "Pending element mismatch; issuing exit fullscreen request");
413 page()->chrome().client().exitFullScreenForElement(&element);
414 return true;
415 }
416
417 INFO_LOG(LOGIDENTIFIER);
418 ASSERT(page()->settings().fullScreenEnabled());
419
420 unwrapFullscreenRenderer(m_fullscreenRenderer.get(), m_fullscreenElement.get());
421
422 element.willBecomeFullscreenElement();
423
424 ASSERT(&element == m_pendingFullscreenElement);
425 m_pendingFullscreenElement = nullptr;
426 m_fullscreenElement = &element;
427
428 // Create a placeholder block for a the full-screen element, to keep the page from reflowing
429 // when the element is removed from the normal flow. Only do this for a RenderBox, as only
430 // a box will have a frameRect. The placeholder will be created in setFullscreenRenderer()
431 // during layout.
432 auto renderer = m_fullscreenElement->renderer();
433 bool shouldCreatePlaceholder = is<RenderBox>(renderer);
434 if (shouldCreatePlaceholder) {
435 m_savedPlaceholderFrameRect = downcast<RenderBox>(*renderer).frameRect();
436 m_savedPlaceholderRenderStyle = RenderStyle::clonePtr(renderer->style());
437 }
438
439 if (m_fullscreenElement != documentElement() && renderer)
440 RenderFullScreen::wrapExistingRenderer(*renderer, document());
441
442 m_fullscreenElement->setContainsFullScreenElementOnAncestorsCrossingFrameBoundaries(true);
443
444 document().resolveStyle(Document::ResolveStyleType::Rebuild);
445 dispatchFullscreenChangeEvents();
446
447 return true;
448}

451{
452 if (!m_fullscreenElement) {
453 ERROR_LOG(LOGIDENTIFIER, "No fullscreenElement; bailing");
454 return false;

458 ERROR_LOG(LOGIDENTIFIER, "No livingRenderTree(); bailing");

467

495bool FullscreenManager::didExitFullscreen()
496{
497 m_pendingExitFullscreen = false;
498 auto fullscreenElement = fullscreenOrPendingElement();
499 if (!fullscreenElement) {
500 ERROR_LOG(LOGIDENTIFIER, "No fullscreenOrPendingElement(); bailing");
501 return false;
502 }
503
504 if (!hasLivingRenderTree()) {
505 ERROR_LOG(LOGIDENTIFIER, "No livingRenderTree(); bailing");
506 return false;
507 }
508
509 if (backForwardCacheState() != Document::NotInBackForwardCache) {
510 ERROR_LOG(LOGIDENTIFIER, "Document in the BackForwardCache; bailing");
511 return false;
512 }
513 INFO_LOG(LOGIDENTIFIER);
514
515 fullscreenElement->setContainsFullScreenElementOnAncestorsCrossingFrameBoundaries(false);
516
517 if (m_fullscreenElement)
518 m_fullscreenElement->didStopBeingFullscreenElement();
519
520 m_areKeysEnabledInFullscreen = false;
521
522 unwrapFullscreenRenderer(m_fullscreenRenderer.get(), m_fullscreenElement.get());
523
524 m_fullscreenElement = nullptr;
525 m_pendingFullscreenElement = nullptr;
526 scheduleFullStyleRebuild();
527
528 // When webkitCancelFullscreen is called, we call webkitExitFullscreen on the topDocument(). That
529 // means that the events will be queued there. So if we have no events here, start the timer on
530 // the exiting document.
531 bool eventTargetQueuesEmpty = m_fullscreenChangeEventTargetQueue.isEmpty() && m_fullscreenErrorEventTargetQueue.isEmpty();
532 Document& exitingDocument = eventTargetQueuesEmpty ? topDocument() : document();
533
534 exitingDocument.fullscreenManager().dispatchFullscreenChangeEvents();
535 return true;
536}

538void FullscreenManager::setFullscreenRenderer(RenderTreeBuilder& builder, RenderFullScreen& renderer)
539{
540 if (&renderer == m_fullscreenRenderer)
541 return;
542
543 if (m_savedPlaceholderRenderStyle)
544 builder.createPlaceholderForFullScreen(renderer, WTFMove(m_savedPlaceholderRenderStyle), m_savedPlaceholderFrameRect);
545 else if (m_fullscreenRenderer && m_fullscreenRenderer->placeholder()) {
546 auto* placeholder = m_fullscreenRenderer->placeholder();
547 builder.createPlaceholderForFullScreen(renderer, RenderStyle::clonePtr(placeholder->style()), placeholder->frameRect());
548 }
549
550 if (m_fullscreenRenderer)
551 builder.destroy(*m_fullscreenRenderer);
552 ASSERT(!m_fullscreenRenderer);
553
554 m_fullscreenRenderer = makeWeakPtr(renderer);
555}

562void FullscreenManager::dispatchFullscreenChangeEvents()
563{
564 // Since we dispatch events in this function, it's possible that the
565 // document will be detached and GC'd. We protect it here to make sure we
566 // can finish the function successfully.
567 Ref<Document> protectedDocument(document());
568 Deque<RefPtr<Node>> changeQueue;
569 m_fullscreenChangeEventTargetQueue.swap(changeQueue);
570 Deque<RefPtr<Node>> errorQueue;
571 m_fullscreenErrorEventTargetQueue.swap(errorQueue);
572 dispatchFullscreenChangeOrErrorEvent(changeQueue, eventNames().webkitfullscreenchangeEvent, /* shouldNotifyMediaElement */ true);
573 dispatchFullscreenChangeOrErrorEvent(errorQueue, eventNames().webkitfullscreenerrorEvent, /* shouldNotifyMediaElement */ false);
574}
575
576void FullscreenManager::dispatchFullscreenChangeOrErrorEvent(Deque<RefPtr<Node>>& queue, const AtomString& eventName, bool shouldNotifyMediaElement)
577{
578 // Step 3 of https://fullscreen.spec.whatwg.org/#run-the-fullscreen-steps
579 while (!queue.isEmpty()) {
580 RefPtr<Node> node = queue.takeFirst();
581 if (!node)
582 node = documentElement();
583 // The dispatchEvent below may have blown away our documentElement.
584 if (!node)
585 continue;
586
587 // If the element was removed from our tree, also message the documentElement. Since we may
588 // have a document hierarchy, check that node isn't in another document.
589 if (!node->isConnected())
590 queue.append(documentElement());
591
592#if ENABLE(VIDEO)
593 if (shouldNotifyMediaElement && is<HTMLMediaElement>(*node))
594 downcast<HTMLMediaElement>(*node).enteredOrExitedFullscreen();
595#else
596 UNUSED_PARAM(shouldNotifyMediaElement);
597#endif
598 node->dispatchEvent(Event::create(eventName, Event::CanBubble::Yes, Event::IsCancelable::No, Event::IsComposed::Yes));
599 }

602void FullscreenManager::adjustFullscreenElementOnNodeRemoval(Node& node, Document::NodeRemoval nodeRemoval)
603{
604 auto fullscreenElement = fullscreenOrPendingElement();
605 if (!fullscreenElement)
606 return;
607
608 bool elementInSubtree = false;
609 if (nodeRemoval == Document::NodeRemoval::ChildrenOfNode)
610 elementInSubtree = fullscreenElement->isDescendantOf(node);
611 else
612 elementInSubtree = (fullscreenElement == &node) || fullscreenElement->isDescendantOf(node);
613
614 if (elementInSubtree) {
615 INFO_LOG(LOGIDENTIFIER, "Ancestor of fullscreen element removed; exiting fullscreen");
616 fullscreenElement->setContainsFullScreenElementOnAncestorsCrossingFrameBoundaries(false);
617 cancelFullscreen();
618 }

686void FullscreenManager::addDocumentToFullscreenChangeEventQueue(Document& document)
687{
688 Node* target = document.fullscreenManager().fullscreenElement();
689 if (!target)
