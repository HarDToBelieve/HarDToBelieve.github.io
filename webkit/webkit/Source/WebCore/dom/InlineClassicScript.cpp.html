34Ref<InlineClassicScript> InlineClassicScript::create(ScriptElement& scriptElement)
35{
36 auto& element = scriptElement.element();
37 return adoptRef(*new InlineClassicScript(
38 element.attributeWithoutSynchronization(HTMLNames::nonceAttr),
39 element.attributeWithoutSynchronization(HTMLNames::crossoriginAttr),
40 scriptElement.scriptCharset(),
41 element.localName(),
42 element.isInUserAgentShadowTree()));
43}
