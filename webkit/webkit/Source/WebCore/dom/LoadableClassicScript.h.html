37// A CachedResourceHandle alone does not prevent the underlying CachedResource
38// from purging its data buffer. This class holds a client until this class is
39// destroyed in order to guarantee that the data buffer will not be purged.
40class LoadableClassicScript final : public LoadableScript, private CachedResourceClient {
41public:
42 virtual ~LoadableClassicScript();
43
44 static Ref<LoadableClassicScript> create(const String& nonce, const String& integrity, ReferrerPolicy, const String& crossOriginMode, const String& charset, const AtomString& initiatorName, bool isInUserAgentShadowTree, bool isAsync);
45 bool isLoaded() const final;
46 std::optional<Error> error() const final;
47 bool wasCanceled() const final;
48
49 CachedScript& cachedScript() { return *m_cachedScript; }
50
51 bool isClassicScript() const final { return true; }
52 bool isModuleScript() const final { return false; }
53
54 void execute(ScriptElement&) final;
55
56 bool load(Document&, const URL&);
57
58private:
59 LoadableClassicScript(const String& nonce, const String& integrity, ReferrerPolicy policy, const String& crossOriginMode, const String& charset, const AtomString& initiatorName, bool isInUserAgentShadowTree, bool isAsync)
60 : LoadableScript(nonce, policy, crossOriginMode, charset, initiatorName, isInUserAgentShadowTree)
61 , m_integrity(integrity)
62 , m_isAsync(isAsync)
63 {
64 }
65
66 void notifyFinished(CachedResource&, const NetworkLoadMetrics&) final;
67
68 CachedResourceHandle<CachedScript> m_cachedScript { };
69 std::optional<Error> m_error { std::nullopt };
70 String m_integrity;
71 bool m_isAsync { false };
72};
73
74}

76SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::LoadableClassicScript)
77 static bool isType(const WebCore::LoadableScript& script) { return script.isClassicScript(); }
