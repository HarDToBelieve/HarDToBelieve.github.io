38namespace WebCore {
39
40class Blob;
41
42#if ENABLE(SERVICE_WORKER)
43using MessageEventSource = Variant<RefPtr<WindowProxy>, RefPtr<MessagePort>, RefPtr<ServiceWorker>>;
44#else
45using MessageEventSource = Variant<RefPtr<WindowProxy>, RefPtr<MessagePort>>;
46#endif
47
48class MessageEvent final : public Event {
49 WTF_MAKE_ISO_ALLOCATED(MessageEvent);
50public:
51 static Ref<MessageEvent> create(Vector<RefPtr<MessagePort>>&&, Ref<SerializedScriptValue>&&, const String& origin = { }, const String& lastEventId = { }, std::optional<MessageEventSource>&& source = std::nullopt);
52 static Ref<MessageEvent> create(const AtomString& type, Ref<SerializedScriptValue>&&, const String& origin, const String& lastEventId);
53 static Ref<MessageEvent> create(const String& data, const String& origin = { });
54 static Ref<MessageEvent> create(Ref<Blob>&& data, const String& origin);
55 static Ref<MessageEvent> create(Ref<ArrayBuffer>&& data, const String& origin = { });
56 static Ref<MessageEvent> createForBindings();
57
58 struct Init : EventInit {
59 JSC::JSValue data;
60 String origin;
61 String lastEventId;
62 std::optional<MessageEventSource> source;
63 Vector<RefPtr<MessagePort>> ports;
64 };
65 static Ref<MessageEvent> create(const AtomString& type, Init&&, IsTrusted = IsTrusted::No);
66
67 virtual ~MessageEvent();
68
69 void initMessageEvent(const AtomString& type, bool canBubble, bool cancelable, JSC::JSValue data, const String& origin, const String& lastEventId, std::optional<MessageEventSource>&&, Vector<RefPtr<MessagePort>>&&);
70
71 const String& origin() const { return m_origin; }
72 const String& lastEventId() const { return m_lastEventId; }
73 const std::optional<MessageEventSource>& source() const { return m_source; }
74 const Vector<RefPtr<MessagePort>>& ports() const { return m_ports; }
75
76 using DataType = Variant<JSValueInWrappedObject, Ref<SerializedScriptValue>, String, Ref<Blob>, Ref<ArrayBuffer>>;
77 const DataType& data() const { return m_data; }
78
79 JSValueInWrappedObject& cachedData() { return m_cachedData; }
80 JSValueInWrappedObject& cachedPorts() { return m_cachedPorts; }
81
82 size_t memoryCost() const;
83
84private:
85 MessageEvent();
86 MessageEvent(const AtomString& type, Init&&, IsTrusted);
87 MessageEvent(const AtomString& type, Ref<SerializedScriptValue>&& data, const String& origin, const String& lastEventId);
88 MessageEvent(DataType&&, const String& origin, const String& lastEventId = { }, std::optional<MessageEventSource>&& = std::nullopt, Vector<RefPtr<MessagePort>>&& = { });
89
90 EventInterface eventInterface() const final;
91
92 DataType m_data;
93 String m_origin;
94 String m_lastEventId;
95 std::optional<MessageEventSource> m_source;
96 Vector<RefPtr<MessagePort>> m_ports;
97
98 JSValueInWrappedObject m_cachedData;
99 JSValueInWrappedObject m_cachedPorts;
100};
