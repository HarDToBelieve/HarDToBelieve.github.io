47static Lock allMessagePortsLock;
48static HashMap<MessagePortIdentifier, MessagePort*>& allMessagePorts() WTF_REQUIRES_LOCK(allMessagePortsLock)
49{
50 static NeverDestroyed<HashMap<MessagePortIdentifier, MessagePort*>> map;
51 return map;

59void MessagePort::deref() const
60{
61 // This custom deref() function ensures that as long as the lock to allMessagePortsLock is taken, no MessagePort will be destroyed.
62 // This allows isExistingMessagePortLocallyReachable and notifyMessageAvailable to easily query the map and manipulate MessagePort instances.
63
64 if (!--m_refCount) {
65 Locker locker { allMessagePortsLock };
66
67 if (m_refCount)
68 return;
69
70 auto iterator = allMessagePorts().find(m_identifier);
71 if (iterator != allMessagePorts().end() && iterator->value == this)
72 allMessagePorts().remove(iterator);
73
74 delete this;
75 }
76}

78bool MessagePort::isExistingMessagePortLocallyReachable(const MessagePortIdentifier& identifier)
79{
80 Locker locker { allMessagePortsLock };
81 auto* port = allMessagePorts().get(identifier);
82 return port && port->isLocallyReachable();

98MessagePort::MessagePort(ScriptExecutionContext& scriptExecutionContext, const MessagePortIdentifier& local, const MessagePortIdentifier& remote)
99 : ActiveDOMObject(&scriptExecutionContext)
100 , m_identifier(local)
101 , m_remoteIdentifier(remote)
102{
103 LOG(MessagePorts, "Created MessagePort %s (%p) in process %" PRIu64, m_identifier.logString().utf8().data(), this, Process::identifier().toUInt64());
104
105 Locker locker { allMessagePortsLock };
106 allMessagePorts().set(m_identifier, this);
107
108 // Make sure the WeakPtrFactory gets initialized eagerly on the thread the MessagePort gets constructed on for thread-safety reasons.
109 initializeWeakPtrFactory();
110
111 m_scriptExecutionContext->createdMessagePort(*this);
112 suspendIfNeeded();
113
114 // Don't need to call processMessageWithMessagePortsSoon() here, because the port will not be opened until start() is invoked.
115}

117MessagePort::~MessagePort()
118{
119 LOG(MessagePorts, "Destroyed MessagePort %s (%p) in process %" PRIu64, m_identifier.logString().utf8().data(), this, Process::identifier().toUInt64());
120
121 ASSERT(allMessagePortsLock.isLocked());
122
123 if (m_entangled)
124 close();
125
126 if (m_scriptExecutionContext)
127 m_scriptExecutionContext->destroyedMessagePort(*this);
128}

135ExceptionOr<void> MessagePort::postMessage(JSC::JSGlobalObject& state, JSC::JSValue messageValue, PostMessageOptions&& options)
136{
137 LOG(MessagePorts, "Attempting to post message to port %s (to be received by port %s)", m_identifier.logString().utf8().data(), m_remoteIdentifier.logString().utf8().data());
138
139 registerLocalActivity();
140
141 Vector<RefPtr<MessagePort>> ports;
142 auto messageData = SerializedScriptValue::create(state, messageValue, WTFMove(options.transfer), ports);
143 if (messageData.hasException())
144 return messageData.releaseException();
145
146 if (!isEntangled())
147 return { };
148 ASSERT(m_scriptExecutionContext);
149
150 TransferredMessagePortArray transferredPorts;
151 // Make sure we aren't connected to any of the passed-in ports.
152 if (!ports.isEmpty()) {
153 for (auto& port : ports) {
154 if (port->identifier() == m_identifier || port->identifier() == m_remoteIdentifier)
155 return Exception { DataCloneError };
156 }
157
158 auto disentangleResult = MessagePort::disentanglePorts(WTFMove(ports));
159 if (disentangleResult.hasException())
160 return disentangleResult.releaseException();
161 transferredPorts = disentangleResult.releaseReturnValue();
162 }
163
164 MessageWithMessagePorts message { messageData.releaseReturnValue(), WTFMove(transferredPorts) };
165
166 LOG(MessagePorts, "Actually posting message to port %s (to be received by port %s)", m_identifier.logString().utf8().data(), m_remoteIdentifier.logString().utf8().data());
167
168 MessagePortChannelProvider::fromContext(*m_scriptExecutionContext).postMessageToRemote(WTFMove(message), m_remoteIdentifier);
169 return { };
170}

172TransferredMessagePort MessagePort::disentangle()
173{
174 ASSERT(m_entangled);
175 m_entangled = false;
176
177 registerLocalActivity();
178
179 MessagePortChannelProvider::fromContext(*m_scriptExecutionContext).messagePortDisentangled(m_identifier);
180
181 // We can't receive any messages or generate any events after this, so remove ourselves from the list of active ports.
182 m_scriptExecutionContext->destroyedMessagePort(*this);
183 m_scriptExecutionContext->willDestroyActiveDOMObject(*this);
184 m_scriptExecutionContext->willDestroyDestructionObserver(*this);
185
186 m_scriptExecutionContext = nullptr;
187
188 return { identifier(), remoteIdentifier() };
189}

191void MessagePort::registerLocalActivity()
192{
193 // Any time certain local operations happen, we dirty our own state to delay GC.
194 m_hasHadLocalActivitySinceLastCheck = true;
195 m_mightBeEligibleForGC = false;
196}

198// Invoked to notify us that there are messages available for this port.
199// This code may be called from another thread, and so should not call any non-threadsafe APIs (i.e. should not call into the entangled channel or access mutable variables).
200void MessagePort::messageAvailable()
201{
202 // This MessagePort object might be disentangled because the port is being transferred,
203 // in which case we'll notify it that messages are available once a new end point is created.
204 if (!m_scriptExecutionContext || m_scriptExecutionContext->activeDOMObjectsAreSuspended())
205 return;
206
207 m_scriptExecutionContext->processMessageWithMessagePortsSoon();
208}

210void MessagePort::start()
211{
212 // Do nothing if we've been cloned or closed.
213 if (!isEntangled())
214 return;
215
216 registerLocalActivity();
217
218 ASSERT(m_scriptExecutionContext);
219 if (m_started)
220 return;
221
222 m_started = true;
223 m_scriptExecutionContext->processMessageWithMessagePortsSoon();
224}

226void MessagePort::close()
227{
228 m_mightBeEligibleForGC = true;
229
230 if (m_closed)
231 return;
232 m_closed = true;
233
234 ensureOnMainThread([identifier = m_identifier] {
235 MessagePortChannelProvider::singleton().messagePortClosed(identifier);
236 });
237
238 removeAllEventListeners();
239}

241void MessagePort::contextDestroyed()

243 ASSERT(m_scriptExecutionContext);

245 close();
246 m_scriptExecutionContext = nullptr;

249void MessagePort::dispatchMessages()
250{
251 // Messages for contexts that are not fully active get dispatched too, but JSAbstractEventListener::handleEvent() doesn't call handlers for these.
252 // The HTML5 spec specifies that any messages sent to a document that is not fully active should be dropped, so this behavior is OK.
253 ASSERT(started());
254
255 if (!m_scriptExecutionContext || m_scriptExecutionContext->activeDOMObjectsAreSuspended() || !isEntangled())
256 return;
257
258 auto messagesTakenHandler = [this, weakThis = makeWeakPtr(this)](Vector<MessageWithMessagePorts>&& messages, Function<void()>&& completionCallback) mutable {
259 auto scopeExit = makeScopeExit(WTFMove(completionCallback));
260
261 if (!weakThis)
262 return;
263
264 LOG(MessagePorts, "MessagePort %s (%p) dispatching %zu messages", m_identifier.logString().utf8().data(), this, messages.size());
265
266 if (!m_scriptExecutionContext)
267 return;
268
269 if (!messages.isEmpty())
270 registerLocalActivity();
271
272 ASSERT(m_scriptExecutionContext->isContextThread());
273
274 bool contextIsWorker = is<WorkerGlobalScope>(*m_scriptExecutionContext);
275 for (auto& message : messages) {
276 // close() in Worker onmessage handler should prevent next message from dispatching.
277 if (contextIsWorker && downcast<WorkerGlobalScope>(*m_scriptExecutionContext).isClosing())
278 return;
279 auto ports = MessagePort::entanglePorts(*m_scriptExecutionContext, WTFMove(message.transferredPorts));
280 // Per specification, each MessagePort object has a task source called the port message queue.
281 queueTaskToDispatchEvent(*this, TaskSource::PostedMessageQueue, MessageEvent::create(WTFMove(ports), message.message.releaseNonNull()));
282 }
283 };
284
285 MessagePortChannelProvider::fromContext(*m_scriptExecutionContext).takeAllMessagesForPort(m_identifier, WTFMove(messagesTakenHandler));
286}

288void MessagePort::dispatchEvent(Event& event)
289{
290 if (m_closed)
291 return;
292
293 if (is<WorkerGlobalScope>(*m_scriptExecutionContext) && downcast<WorkerGlobalScope>(*m_scriptExecutionContext).isClosing())
294 return;
295
296 EventTarget::dispatchEvent(event);
297}

299void MessagePort::updateActivity(MessagePortChannelProvider::HasActivity hasActivity)
300{
301 bool hasHadLocalActivity = m_hasHadLocalActivitySinceLastCheck;
302 m_hasHadLocalActivitySinceLastCheck = false;
303
304 if (hasActivity == MessagePortChannelProvider::HasActivity::No && !hasHadLocalActivity)
305 m_isRemoteEligibleForGC = true;
306
307 if (hasActivity == MessagePortChannelProvider::HasActivity::Yes)
308 m_isRemoteEligibleForGC = false;
309
310 m_isAskingRemoteAboutGC = false;
311}

313bool MessagePort::virtualHasPendingActivity() const
314{
315 m_mightBeEligibleForGC = true;
316
317 // If the ScriptExecutionContext has been shut down on this object close()'ed, we can GC.
318 if (!m_scriptExecutionContext || m_closed)
319 return false;
320
321 // If this object has been idle since the remote port declared itself elgibile for GC, we can GC.
322 if (!m_hasHadLocalActivitySinceLastCheck && m_isRemoteEligibleForGC)
323 return false;
324
325 // If this MessagePort has no message event handler then the existence of remote activity cannot keep it alive.
326 if (!m_hasMessageEventListener)
327 return false;
328
329 // If we're not in the middle of asking the remote port about collectability, do so now.
330 if (!m_isAskingRemoteAboutGC) {
331 RefPtr<WorkerOrWorkletThread> workerOrWorkletThread;
332 if (is<WorkerOrWorkletGlobalScope>(*m_scriptExecutionContext))
333 workerOrWorkletThread = downcast<WorkerOrWorkletGlobalScope>(*m_scriptExecutionContext).workerOrWorkletThread();
334
335 callOnMainThread([remoteIdentifier = m_remoteIdentifier, weakThis = makeWeakPtr(const_cast<MessagePort*>(this)), workerOrWorkletThread = WTFMove(workerOrWorkletThread)]() mutable {
336 MessagePortChannelProvider::singleton().checkRemotePortForActivity(remoteIdentifier, [weakThis = WTFMove(weakThis), workerOrWorkletThread = WTFMove(workerOrWorkletThread)](auto hasActivity) mutable {
337 if (!workerOrWorkletThread) {
338 if (weakThis)
339 weakThis->updateActivity(hasActivity);
340 return;
341 }
342
343 workerOrWorkletThread->runLoop().postTaskForMode([weakThis = WTFMove(weakThis), hasActivity](auto&) mutable {
344 if (weakThis)
345 weakThis->updateActivity(hasActivity);
346 }, WorkerRunLoop::defaultMode());
347 });
348 });
349 m_isAskingRemoteAboutGC = true;
350 }
351
352 // Since we need an answer from the remote object, we have to pretend we have pending activity for now.
353 return true;
354}

368ExceptionOr<TransferredMessagePortArray> MessagePort::disentanglePorts(Vector<RefPtr<MessagePort>>&& ports)
369{
370 if (ports.isEmpty())
371 return TransferredMessagePortArray { };
372
373 // Walk the incoming array - if there are any duplicate ports, or null ports or cloned ports, throw an error (per section 8.3.3 of the HTML5 spec).
374 HashSet<MessagePort*> portSet;
375 for (auto& port : ports) {
376 if (!port || !port->m_entangled || !portSet.add(port.get()).isNewEntry)
377 return Exception { DataCloneError };
378 }
379
380 // Passed-in ports passed validity checks, so we can disentangle them.
381 TransferredMessagePortArray portArray;
382 portArray.reserveInitialCapacity(ports.size());
383 for (auto& port : ports)
384 portArray.uncheckedAppend(port->disentangle());
385
386 return portArray;
387}

389Vector<RefPtr<MessagePort>> MessagePort::entanglePorts(ScriptExecutionContext& context, TransferredMessagePortArray&& transferredPorts)
390{
391 LOG(MessagePorts, "Entangling %zu transferred ports to ScriptExecutionContext %s (%p)", transferredPorts.size(), context.url().string().utf8().data(), &context);
392
393 if (transferredPorts.isEmpty())
394 return { };
395
396 Vector<RefPtr<MessagePort>> ports;
397 ports.reserveInitialCapacity(transferredPorts.size());
398 for (auto& transferredPort : transferredPorts)
399 ports.uncheckedAppend(MessagePort::entangle(context, WTFMove(transferredPort)));
400 return ports;
401}

403Ref<MessagePort> MessagePort::entangle(ScriptExecutionContext& context, TransferredMessagePort&& transferredPort)
404{
405 auto port = MessagePort::create(context, transferredPort.first, transferredPort.second);
406 port->entangle();
407 return port;

410bool MessagePort::addEventListener(const AtomString& eventType, Ref<EventListener>&& listener, const AddEventListenerOptions& options)
411{
412 if (eventType == eventNames().messageEvent) {
413 if (listener->isAttribute())
414 start();
415 m_hasMessageEventListener = true;
416 registerLocalActivity();
417 }
418
419 return EventTargetWithInlineData::addEventListener(eventType, WTFMove(listener), options);
420}
421
422bool MessagePort::removeEventListener(const AtomString& eventType, EventListener& listener, const EventListenerOptions& options)
423{
424 auto result = EventTargetWithInlineData::removeEventListener(eventType, listener, options);
425
426 if (!hasEventListeners(eventNames().messageEvent))
427 m_hasMessageEventListener = false;
428
429 return result;
430}
