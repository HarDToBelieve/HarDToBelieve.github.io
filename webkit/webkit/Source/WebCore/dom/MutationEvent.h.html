24#pragma once
25
26#include "Event.h"
27#include "Node.h"
28
29namespace WebCore {
30
31class MutationEvent final : public Event {
32 WTF_MAKE_ISO_ALLOCATED(MutationEvent);
33public:
34 enum {
35 MODIFICATION = 1,
36 ADDITION = 2,
37 REMOVAL = 3
38 };
39
40 static Ref<MutationEvent> create(const AtomString& type, CanBubble canBubble, Node* relatedNode = nullptr, const String& prevValue = String(), const String& newValue = String())
41 {
42 return adoptRef(*new MutationEvent(type, canBubble, IsCancelable::No, relatedNode, prevValue, newValue));
43 }
44
45 static Ref<MutationEvent> createForBindings()
46 {
47 return adoptRef(*new MutationEvent);
48 }
49
50 WEBCORE_EXPORT void initMutationEvent(const AtomString& type, bool canBubble, bool cancelable, Node* relatedNode, const String& prevValue, const String& newValue, const String& attrName, unsigned short attrChange);
51
52 Node* relatedNode() const { return m_relatedNode.get(); }
53 String prevValue() const { return m_prevValue; }
54 String newValue() const { return m_newValue; }
55 String attrName() const { return m_attrName; }
56 unsigned short attrChange() const { return m_attrChange; }
57
58private:
59 MutationEvent() = default;
60 MutationEvent(const AtomString& type, CanBubble, IsCancelable, Node* relatedNode, const String& prevValue, const String& newValue);
61
62 EventInterface eventInterface() const final;
63
64 RefPtr<Node> m_relatedNode;
65 String m_prevValue;
66 String m_newValue;
67 String m_attrName;
68 unsigned short m_attrChange { 0 };
69};
70
71} // namespace WebCore
