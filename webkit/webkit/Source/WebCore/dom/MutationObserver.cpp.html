72bool MutationObserver::validateOptions(MutationObserverOptions options)
73{
74 return (options & (Attributes | CharacterData | ChildList))
75 && ((options & Attributes) || !(options & AttributeOldValue))
76 && ((options & Attributes) || !(options & AttributeFilter))
77 && ((options & CharacterData) || !(options & CharacterDataOldValue));
78}
79
80ExceptionOr<void> MutationObserver::observe(Node& node, const Init& init)
81{
82 MutationObserverOptions options = 0;
83
84 if (init.childList)
85 options |= ChildList;
86 if (init.subtree)
87 options |= Subtree;
88 if (init.attributeOldValue.value_or(false))
89 options |= AttributeOldValue;
90 if (init.characterDataOldValue.value_or(false))
91 options |= CharacterDataOldValue;
92
93 HashSet<AtomString> attributeFilter;
94 if (init.attributeFilter) {
95 for (auto& value : init.attributeFilter.value())
96 attributeFilter.add(value);
97 options |= AttributeFilter;
98 }
99
100 if (init.attributes ? init.attributes.value() : (options & (AttributeFilter | AttributeOldValue)))
101 options |= Attributes;
102
103 if (init.characterData ? init.characterData.value() : (options & CharacterDataOldValue))
104 options |= CharacterData;
105
106 if (!validateOptions(options))
107 return Exception { TypeError };
108
109 node.registerMutationObserver(*this, options, attributeFilter);
110
111 return { };
112}

119void MutationObserver::disconnect()
120{
121 m_pendingTargets.clear();
122 m_records.clear();
123 WeakHashSet registrations { m_registrations };
124 for (auto& registration : registrations) {
125 auto nodeRef = makeRefPtr(registration.node());
126 nodeRef->unregisterMutationObserver(registration);
127 }
128}

142void MutationObserver::enqueueMutationRecord(Ref<MutationRecord>&& mutation)
143{
144 ASSERT(isMainThread());
145 ASSERT(mutation->target());
146 auto document = makeRef(mutation->target()->document());
147
148 m_pendingTargets.add(*mutation->target());
149 m_records.append(WTFMove(mutation));
150
151 auto eventLoop = makeRef(document->windowEventLoop());
152 eventLoop->activeMutationObservers().add(this);
153 eventLoop->queueMutationObserverCompoundMicrotask();
154}

156void MutationObserver::enqueueSlotChangeEvent(HTMLSlotElement& slot)
157{
158 ASSERT(isMainThread());
159 auto eventLoop = makeRef(slot.document().windowEventLoop());
160 auto& list = eventLoop->signalSlotList();
161 ASSERT(list.findMatching([&slot](auto& entry) { return entry.ptr() == &slot; }) == notFound);
162 list.append(slot);
163
164 eventLoop->queueMutationObserverCompoundMicrotask();
165}

174bool MutationObserver::isReachableFromOpaqueRoots(JSC::AbstractSlotVisitor& visitor) const
175{
176 for (auto& registration : m_registrations) {
177 if (registration.isReachableFromOpaqueRoots(visitor))
178 return true;
179 }
180 return false;

183bool MutationObserver::canDeliver()
184{
185 return m_callback->canInvokeCallback();
186}
187
188void MutationObserver::deliver()
189{
190 ASSERT(canDeliver());
191
192 // Calling takeTransientRegistrations() can modify m_registrations, so it's necessary
193 // to make a copy of the transient registrations before operating on them.
194 Vector<MutationObserverRegistration*, 1> transientRegistrations;
195 Vector<std::unique_ptr<HashSet<GCReachableRef<Node>>>, 1> nodesToKeepAlive;
196 HashSet<GCReachableRef<Node>> pendingTargets;
197 pendingTargets.swap(m_pendingTargets);
198 for (auto& registration : m_registrations) {
199 if (registration.hasTransientRegistrations())
200 transientRegistrations.append(&registration);
201 }
202 for (auto& registration : transientRegistrations)
203 nodesToKeepAlive.append(registration->takeTransientRegistrations());
204
205 if (m_records.isEmpty()) {
206 ASSERT(m_pendingTargets.isEmpty());
207 return;
208 }
209
210 Vector<Ref<MutationRecord>> records;
211 records.swap(m_records);
212
213 // FIXME: Keep mutation observer callback as long as its observed nodes are alive. See https://webkit.org/b/179224.
214 if (m_callback->hasCallback()) {
215 auto* context = m_callback->scriptExecutionContext();
216 if (!context)
217 return;
218
219 InspectorInstrumentation::willFireObserverCallback(*context, "MutationObserver"_s);
220 m_callback->handleEvent(*this, records, *this);
221 InspectorInstrumentation::didFireObserverCallback(*context);
222 }
223}

225// https://dom.spec.whatwg.org/#notify-mutation-observers
226void MutationObserver::notifyMutationObservers(WindowEventLoop& eventLoop)
227{
228 if (!eventLoop.suspendedMutationObservers().isEmpty()) {
229 for (auto& observer : copyToVector(eventLoop.suspendedMutationObservers())) {
230 if (!observer->canDeliver())
231 continue;
232
233 eventLoop.suspendedMutationObservers().remove(observer);
234 eventLoop.activeMutationObservers().add(observer);
235 }
236 }
237
238 while (!eventLoop.activeMutationObservers().isEmpty() || !eventLoop.signalSlotList().isEmpty()) {
239 // 2. Let notify list be a copy of unit of related similar-origin browsing contexts' list of MutationObserver objects.
240 auto notifyList = copyToVector(eventLoop.activeMutationObservers());
241 eventLoop.activeMutationObservers().clear();
242 std::sort(notifyList.begin(), notifyList.end(), [](auto& lhs, auto& rhs) {
243 return lhs->m_priority < rhs->m_priority;
244 });
245
246 // 3. Let signalList be a copy of unit of related similar-origin browsing contexts' signal slot list.
247 // 4. Empty unit of related similar-origin browsing contexts' signal slot list.
248 Vector<GCReachableRef<HTMLSlotElement>> slotList;
249 if (!eventLoop.signalSlotList().isEmpty()) {
250 slotList.swap(eventLoop.signalSlotList());
251 for (auto& slot : slotList)
252 slot->didRemoveFromSignalSlotList();
253 }
254
255 // 5. For each MutationObserver object mo in notify list, execute a compound microtask subtask
256 for (auto& observer : notifyList) {
257 if (observer->canDeliver())
258 observer->deliver();
259 else
260 eventLoop.suspendedMutationObservers().add(observer);
261 }
262
263 // 6. For each slot slot in signalList, in order, fire an event named slotchange, with its bubbles attribute set to true, at slot.
264 for (auto& slot : slotList)
265 slot->dispatchSlotChangeEvent();
266 }
