31#pragma once
32
33#include "Document.h"
34#include "MutationObserver.h"
35#include <memory>
36#include <wtf/HashMap.h>
37
38namespace WebCore {
39
40class QualifiedName;
41
42class MutationObserverInterestGroup {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 MutationObserverInterestGroup(HashMap<Ref<MutationObserver>, MutationRecordDeliveryOptions>&&, MutationRecordDeliveryOptions);
46
47 static std::unique_ptr<MutationObserverInterestGroup> createForChildListMutation(Node& target)
48 {
49 if (!target.document().hasMutationObserversOfType(MutationObserver::ChildList))
50 return nullptr;
51
52 MutationRecordDeliveryOptions oldValueFlag = 0;
53 return createIfNeeded(target, MutationObserver::ChildList, oldValueFlag);
54 }
55
56 static std::unique_ptr<MutationObserverInterestGroup> createForCharacterDataMutation(Node& target)
57 {
58 if (!target.document().hasMutationObserversOfType(MutationObserver::CharacterData))
59 return nullptr;
60
61 return createIfNeeded(target, MutationObserver::CharacterData, MutationObserver::CharacterDataOldValue);
62 }
63
64 static std::unique_ptr<MutationObserverInterestGroup> createForAttributesMutation(Node& target, const QualifiedName& attributeName)
65 {
66 if (!target.document().hasMutationObserversOfType(MutationObserver::Attributes))
67 return nullptr;
68
69 return createIfNeeded(target, MutationObserver::Attributes, MutationObserver::AttributeOldValue, &attributeName);
70 }
71
72 bool isOldValueRequested() const;
73 void enqueueMutationRecord(Ref<MutationRecord>&&);
74
75private:
76 static std::unique_ptr<MutationObserverInterestGroup> createIfNeeded(Node& target, MutationObserver::MutationType, MutationRecordDeliveryOptions oldValueFlag, const QualifiedName* attributeName = nullptr);
77
78 bool hasOldValue(MutationRecordDeliveryOptions options) const { return options & m_oldValueFlag; }
79
80 HashMap<Ref<MutationObserver>, MutationRecordDeliveryOptions> m_observers;
81 MutationRecordDeliveryOptions m_oldValueFlag;
82};
83
84} // namespace WebCore
