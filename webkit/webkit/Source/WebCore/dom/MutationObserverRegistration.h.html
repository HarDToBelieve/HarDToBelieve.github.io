41class AbstractSlotVisitor;

44namespace WebCore {
45
46class QualifiedName;
47
48class MutationObserverRegistration : public CanMakeWeakPtr<MutationObserverRegistration> {
49 WTF_MAKE_FAST_ALLOCATED;
50public:
51 MutationObserverRegistration(MutationObserver&, Node&, MutationObserverOptions, const HashSet<AtomString>& attributeFilter);
52 ~MutationObserverRegistration();
53
54 void resetObservation(MutationObserverOptions, const HashSet<AtomString>& attributeFilter);
55 void observedSubtreeNodeWillDetach(Node&);
56 std::unique_ptr<HashSet<GCReachableRef<Node>>> takeTransientRegistrations();
57 bool hasTransientRegistrations() const { return m_transientRegistrationNodes && !m_transientRegistrationNodes->isEmpty(); }
58
59 bool shouldReceiveMutationFrom(Node&, MutationObserver::MutationType, const QualifiedName* attributeName) const;
60 bool isSubtree() const { return m_options & MutationObserver::Subtree; }
61
62 MutationObserver& observer() { return m_observer.get(); }
63 Node& node() { return m_node; }
64 MutationRecordDeliveryOptions deliveryOptions() const { return m_options & (MutationObserver::AttributeOldValue | MutationObserver::CharacterDataOldValue); }
65 MutationObserverOptions mutationTypes() const { return m_options & MutationObserver::AllMutationTypes; }
66
67 bool isReachableFromOpaqueRoots(JSC::AbstractSlotVisitor&) const;
68
69private:
70 Ref<MutationObserver> m_observer;
71 Node& m_node;
72 RefPtr<Node> m_nodeKeptAlive;
73 std::unique_ptr<HashSet<GCReachableRef<Node>>> m_transientRegistrationNodes;
74 MutationObserverOptions m_options;
75 HashSet<AtomString> m_attributeFilter;
76};
