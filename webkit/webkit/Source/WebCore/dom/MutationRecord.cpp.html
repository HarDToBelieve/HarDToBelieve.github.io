45static void visitNodeList(JSC::AbstractSlotVisitor& visitor, NodeList& nodeList)
46{
47 ASSERT(!nodeList.isLiveNodeList());
48 unsigned length = nodeList.length();
49 for (unsigned i = 0; i < length; ++i)
50 visitor.addOpaqueRoot(root(nodeList.item(i)));
51}

53class ChildListRecord final : public MutationRecord {
54public:
55 ChildListRecord(ContainerNode& target, Ref<NodeList>&& added, Ref<NodeList>&& removed, RefPtr<Node>&& previousSibling, RefPtr<Node>&& nextSibling)
56 : m_target(target)
57 , m_addedNodes(WTFMove(added))
58 , m_removedNodes(WTFMove(removed))
59 , m_previousSibling(WTFMove(previousSibling))
60 , m_nextSibling(WTFMove(nextSibling))
61 {
62 }
63
64private:
65 const AtomString& type() override;
66 Node* target() override { return m_target.ptr(); }
67 NodeList* addedNodes() override { return m_addedNodes.get(); }
68 NodeList* removedNodes() override { return m_removedNodes.get(); }
69 Node* previousSibling() override { return m_previousSibling.get(); }
70 Node* nextSibling() override { return m_nextSibling.get(); }
71
72 void visitNodesConcurrently(JSC::AbstractSlotVisitor& visitor) const final
73 {
74 visitor.addOpaqueRoot(root(m_target.ptr()));
75 if (m_addedNodes)
76 visitNodeList(visitor, *m_addedNodes);
77 if (m_removedNodes)
78 visitNodeList(visitor, *m_removedNodes);
79 }
80
81 Ref<ContainerNode> m_target;
82 RefPtr<NodeList> m_addedNodes;
83 RefPtr<NodeList> m_removedNodes;
84 RefPtr<Node> m_previousSibling;
85 RefPtr<Node> m_nextSibling;
86};

88class RecordWithEmptyNodeLists : public MutationRecord {
89public:
90 RecordWithEmptyNodeLists(Node& target, const String& oldValue)
91 : m_target(target)
92 , m_oldValue(oldValue)
93 {
94 }
95
96private:
97 Node* target() override { return m_target.ptr(); }
98 String oldValue() override { return m_oldValue; }
99 NodeList* addedNodes() override { return lazilyInitializeEmptyNodeList(m_addedNodes); }
100 NodeList* removedNodes() override { return lazilyInitializeEmptyNodeList(m_removedNodes); }
101
102 static NodeList* lazilyInitializeEmptyNodeList(RefPtr<NodeList>& nodeList)
103 {
104 if (!nodeList)
105 nodeList = StaticNodeList::create();
106 return nodeList.get();
107 }
108
109 void visitNodesConcurrently(JSC::AbstractSlotVisitor& visitor) const final
110 {
111 visitor.addOpaqueRoot(root(m_target.ptr()));
112 }
113
114 Ref<Node> m_target;
115 String m_oldValue;
116 RefPtr<NodeList> m_addedNodes;
117 RefPtr<NodeList> m_removedNodes;
118};

149class MutationRecordWithNullOldValue final : public MutationRecord {
150public:
151 MutationRecordWithNullOldValue(MutationRecord& record)
152 : m_record(record)
153 {
154 }
155
156private:
157 const AtomString& type() override { return m_record->type(); }
158 Node* target() override { return m_record->target(); }
159 NodeList* addedNodes() override { return m_record->addedNodes(); }
160 NodeList* removedNodes() override { return m_record->removedNodes(); }
161 Node* previousSibling() override { return m_record->previousSibling(); }
162 Node* nextSibling() override { return m_record->nextSibling(); }
163 const AtomString& attributeName() override { return m_record->attributeName(); }
164 const AtomString& attributeNamespace() override { return m_record->attributeNamespace(); }
165
166 String oldValue() override { return String(); }
167
168 void visitNodesConcurrently(JSC::AbstractSlotVisitor& visitor) const final
169 {
170 m_record->visitNodesConcurrently(visitor);
171 }
172
173 Ref<MutationRecord> m_record;
174};

196Ref<MutationRecord> MutationRecord::createChildList(ContainerNode& target, Ref<NodeList>&& added, Ref<NodeList>&& removed, RefPtr<Node>&& previousSibling, RefPtr<Node>&& nextSibling)
197{
198 return adoptRef(static_cast<MutationRecord&>(*new ChildListRecord(target, WTFMove(added), WTFMove(removed), WTFMove(previousSibling), WTFMove(nextSibling))));
199}

201Ref<MutationRecord> MutationRecord::createAttributes(Element& target, const QualifiedName& name, const AtomString& oldValue)
202{
203 return adoptRef(static_cast<MutationRecord&>(*new AttributesRecord(target, name, oldValue)));
204}
