37class PopStateEvent final : public Event {

39public:
40 virtual ~PopStateEvent();
41 static Ref<PopStateEvent> create(RefPtr<SerializedScriptValue>&&, History*);
42
43 struct Init : EventInit {
44 JSC::JSValue state;
45 };
46
47 static Ref<PopStateEvent> create(const AtomString&, const Init&, IsTrusted = IsTrusted::No);
48 static Ref<PopStateEvent> createForBindings();
49
50 const JSValueInWrappedObject& state() const { return m_state; }
51 SerializedScriptValue* serializedState() const { return m_serializedState.get(); }
52
53 RefPtr<SerializedScriptValue> trySerializeState(JSC::JSGlobalObject&);
54
55 History* history() const { return m_history.get(); }
56
57private:
58 PopStateEvent() = default;
59 PopStateEvent(const AtomString&, const Init&, IsTrusted);
60 PopStateEvent(RefPtr<SerializedScriptValue>&&, History*);
61
62 EventInterface eventInterface() const final;
63
64 JSValueInWrappedObject m_state;
65 RefPtr<SerializedScriptValue> m_serializedState;
66 bool m_triedToSerialize { false };
67 RefPtr<History> m_history;
68};
