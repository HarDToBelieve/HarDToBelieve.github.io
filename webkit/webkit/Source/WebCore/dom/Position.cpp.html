70static bool hasInlineRun(RenderObject& renderer)
71{
72 if (is<RenderBox>(renderer) && LayoutIntegration::runFor(downcast<RenderBox>(renderer)))
73 return true;
74 if (is<RenderText>(renderer) && LayoutIntegration::firstTextRunFor(downcast<RenderText>(renderer)))
75 return true;
76 if (is<RenderLineBreak>(renderer) && LayoutIntegration::runFor(downcast<RenderLineBreak>(renderer)))
77 return true;
78 return false;
79}

81static Node* nextRenderedEditable(Node* node)
82{
83 while ((node = nextLeafNode(node))) {
84 RenderObject* renderer = node->renderer();
85 if (!renderer || !node->hasEditableStyle())
86 continue;
87 if (hasInlineRun(*renderer))
88 return node;
89 }
90 return nullptr;
91}
92
93static Node* previousRenderedEditable(Node* node)
94{
95 while ((node = previousLeafNode(node))) {
96 RenderObject* renderer = node->renderer();
97 if (!renderer || !node->hasEditableStyle())
98 continue;
99 if (hasInlineRun(*renderer))
100 return node;
101 }
102 return nullptr;
103}

164Node* Position::containerNode() const
165{
166 if (!m_anchorNode)
167 return nullptr;
168
169 switch (anchorType()) {
170 case PositionIsBeforeChildren:
171 case PositionIsAfterChildren:
172 case PositionIsOffsetInAnchor:
173 return m_anchorNode.get();
174 case PositionIsBeforeAnchor:
175 case PositionIsAfterAnchor:
176 return m_anchorNode->parentNode();
177 }
178 ASSERT_NOT_REACHED();
179 return nullptr;

209int Position::computeOffsetInContainerNode() const
210{
211 if (!m_anchorNode)
212 return 0;
213
214 switch (anchorType()) {
215 case PositionIsBeforeChildren:
216 return 0;
217 case PositionIsAfterChildren:
218 return m_anchorNode->length();
219 case PositionIsOffsetInAnchor:
220 return m_offset;
221 case PositionIsBeforeAnchor:
222 return m_anchorNode->computeNodeIndex();
223 case PositionIsAfterAnchor:
224 return m_anchorNode->computeNodeIndex() + 1;
225 }
226 ASSERT_NOT_REACHED();
227 return 0;
228}

238// Neighbor-anchored positions are invalid DOM positions, so they need to be
239// fixed up before handing them off to the Range object.
240Position Position::parentAnchoredEquivalent() const
241{
242 if (!m_anchorNode)
243 return { };
244
245 // FIXME: This should only be necessary for legacy positions, but is also needed for positions before and after Tables
246 if (m_offset <= 0 && (m_anchorType != PositionIsAfterAnchor && m_anchorType != PositionIsAfterChildren)) {
247 if (m_anchorNode->parentNode() && (editingIgnoresContent(*m_anchorNode) || isRenderedTable(m_anchorNode.get())))
248 return positionInParentBeforeNode(m_anchorNode.get());
249 return Position(m_anchorNode.get(), 0, PositionIsOffsetInAnchor);
250 }
251
252 if (!m_anchorNode->isCharacterDataNode()
253 && (m_anchorType == PositionIsAfterAnchor || m_anchorType == PositionIsAfterChildren || static_cast<unsigned>(m_offset) == m_anchorNode->countChildNodes())
254 && (editingIgnoresContent(*m_anchorNode) || isRenderedTable(m_anchorNode.get()))
255 && containerNode()) {
256 return positionInParentAfterNode(m_anchorNode.get());
257 }
258
259 return { containerNode(), static_cast<unsigned>(computeOffsetInContainerNode()), PositionIsOffsetInAnchor };
260}

262RefPtr<Node> Position::firstNode() const
263{
264 auto container = makeRefPtr(containerNode());
265 if (!container)
266 return nullptr;
267 if (is<CharacterData>(*container))
268 return container;
269 if (auto* node = computeNodeAfterPosition())
270 return node;
271 if (!computeOffsetInContainerNode())
272 return container;
273 return NodeTraversal::nextSkippingChildren(*container);
274}

276Node* Position::computeNodeBeforePosition() const
277{
278 if (!m_anchorNode)
279 return nullptr;
280
281 switch (anchorType()) {
282 case PositionIsBeforeChildren:
283 return nullptr;
284 case PositionIsAfterChildren:
285 return m_anchorNode->lastChild();
286 case PositionIsOffsetInAnchor:
287 return m_offset ? m_anchorNode->traverseToChildAt(m_offset - 1) : nullptr;
288 case PositionIsBeforeAnchor:
289 return m_anchorNode->previousSibling();
290 case PositionIsAfterAnchor:
291 return m_anchorNode.get();
292 }
293 ASSERT_NOT_REACHED();
294 return nullptr;
295}

306 return nullptr;
307 case PositionIsOffsetInAnchor:

309 case PositionIsBeforeAnchor:
310 return m_anchorNode.get();
311 case PositionIsAfterAnchor:

337Position Position::previous(PositionMoveType moveType) const
338{
339 Node* node = deprecatedNode();
340 if (!node)
341 return *this;
342
343 // FIXME: Negative offsets shouldn't be allowed. We should catch this earlier.
344 ASSERT(deprecatedEditingOffset() >= 0);
345
346 unsigned offset = deprecatedEditingOffset();
347
348 if (anchorType() == PositionIsBeforeAnchor) {
349 node = containerNode();
350 if (!node)
351 return *this;
352
353 offset = computeOffsetInContainerNode();
354 }
355
356 if (offset > 0) {
357 if (Node* child = node->traverseToChildAt(offset - 1))
358 return lastPositionInOrAfterNode(child);
359
360 // There are two reasons child might be 0:
361 // 1) The node is node like a text node that is not an element, and therefore has no children.
362 // Going backward one character at a time is correct.
363 // 2) The old offset was a bogus offset like (<br>, 1), and there is no child.
364 // Going from 1 to 0 is correct.
365 switch (moveType) {
366 case CodePoint:
367 return makeDeprecatedLegacyPosition(node, offset - 1);
368 case Character:
369 return makeDeprecatedLegacyPosition(node, uncheckedPreviousOffset(node, offset));
370 case BackwardDeletion:
371 return makeDeprecatedLegacyPosition(node, uncheckedPreviousOffsetForBackwardDeletion(node, offset));
372 }
373 }
374
375 ContainerNode* parent = node->parentNode();
376 if (!parent)
377 return *this;
378
379 if (positionBeforeOrAfterNodeIsCandidate(*node))
380 return positionBeforeNode(node);
381
382 Node* previousSibling = node->previousSibling();
383 if (previousSibling && positionBeforeOrAfterNodeIsCandidate(*previousSibling))
384 return positionAfterNode(previousSibling);
385
386 return makeContainerOffsetPosition(parent, node->computeNodeIndex());
387}

389Position Position::next(PositionMoveType moveType) const
390{
391 ASSERT(moveType != BackwardDeletion);
392
393 Node* node = deprecatedNode();
394 if (!node)
395 return *this;
396
397 // FIXME: Negative offsets shouldn't be allowed. We should catch this earlier.
398 ASSERT(deprecatedEditingOffset() >= 0);
399
400 unsigned offset = deprecatedEditingOffset();
401
402 if (anchorType() == PositionIsAfterAnchor) {
403 node = containerNode();
404 if (!node)
405 return *this;
406
407 offset = computeOffsetInContainerNode();
408 }
409
410 Node* child = node->traverseToChildAt(offset);
411 if (child || (!node->hasChildNodes() && offset < static_cast<unsigned>(lastOffsetForEditing(*node)))) {
412 if (child)
413 return firstPositionInOrBeforeNode(child);
414
415 // There are two reasons child might be 0:
416 // 1) The node is node like a text node that is not an element, and therefore has no children.
417 // Going forward one character at a time is correct.
418 // 2) The new offset is a bogus offset like (<br>, 1), and there is no child.
419 // Going from 0 to 1 is correct.
420 return makeDeprecatedLegacyPosition(node, (moveType == Character) ? uncheckedNextOffset(node, offset) : offset + 1);
421 }
422
423 ContainerNode* parent = node->parentNode();
424 if (!parent)
425 return *this;
426
427 if (isRenderedTable(node) || editingIgnoresContent(*node))
428 return positionAfterNode(node);
429
430 Node* nextSibling = node->nextSibling();
431 if (nextSibling && positionBeforeOrAfterNodeIsCandidate(*nextSibling))
432 return positionBeforeNode(nextSibling);
433
434 return makeContainerOffsetPosition(parent, node->computeNodeIndex() + 1);
435}

452bool Position::atFirstEditingPositionForNode() const
453{
454 if (isNull())
455 return true;
456 // FIXME: Position before anchor shouldn't be considered as at the first editing position for node
457 // since that position resides outside of the node.
458 switch (m_anchorType) {
459 case PositionIsOffsetInAnchor:
460 return m_offset <= 0;
461 case PositionIsBeforeChildren:
462 case PositionIsBeforeAnchor:
463 return true;
464 case PositionIsAfterChildren:
465 case PositionIsAfterAnchor:
466 return !lastOffsetForEditing(*deprecatedNode());
467 }
468 ASSERT_NOT_REACHED();
469 return false;
470}

472bool Position::atLastEditingPositionForNode() const
473{
474 if (isNull())
475 return true;
476 // FIXME: Position after anchor shouldn't be considered as at the first editing position for node since that position resides outside of the node.
477 return m_anchorType == PositionIsAfterAnchor || m_anchorType == PositionIsAfterChildren || m_offset >= static_cast<unsigned>(lastOffsetForEditing(*deprecatedNode()));
478}

480// A position is considered at editing boundary if one of the following is true:
481// 1. It is the first position in the node and the next visually equivalent position
482// is non editable.
483// 2. It is the last position in the node and the previous visually equivalent position
484// is non editable.
485// 3. It is an editable position and both the next and previous visually equivalent
486// positions are both non editable.
487bool Position::atEditingBoundary() const
488{
489 Position nextPosition = downstream(CanCrossEditingBoundary);
490 if (atFirstEditingPositionForNode() && nextPosition.isNotNull() && !nextPosition.deprecatedNode()->hasEditableStyle())
491 return true;
492
493 Position prevPosition = upstream(CanCrossEditingBoundary);
494 if (atLastEditingPositionForNode() && prevPosition.isNotNull() && !prevPosition.deprecatedNode()->hasEditableStyle())
495 return true;
496
497 return nextPosition.isNotNull() && !nextPosition.deprecatedNode()->hasEditableStyle()
498 && prevPosition.isNotNull() && !prevPosition.deprecatedNode()->hasEditableStyle();
499}

501Node* Position::parentEditingBoundary() const
502{
503 if (!m_anchorNode)
504 return nullptr;
505
506 Node* documentElement = m_anchorNode->document().documentElement();
507 if (!documentElement)
508 return nullptr;
509
510 Node* boundary = m_anchorNode.get();
511 while (boundary != documentElement && boundary->nonShadowBoundaryParentNode() && m_anchorNode->hasEditableStyle() == boundary->parentNode()->hasEditableStyle())
512 boundary = boundary->nonShadowBoundaryParentNode();
513
514 return boundary;
515}

518bool Position::atStartOfTree() const
519{
520 if (isNull())
521 return true;
522
523 Node* container = containerNode();
524 if (container && container->parentNode())
525 return false;
526
527 switch (m_anchorType) {
528 case PositionIsOffsetInAnchor:
529 return m_offset <= 0;
530 case PositionIsBeforeAnchor:
531 return !m_anchorNode->previousSibling();
532 case PositionIsAfterAnchor:
533 return false;
534 case PositionIsBeforeChildren:
535 return true;
536 case PositionIsAfterChildren:
537 return !lastOffsetForEditing(*m_anchorNode);
538 }
539 ASSERT_NOT_REACHED();
540 return false;
541}

543bool Position::atEndOfTree() const
544{
545 if (isNull())
546 return true;
547
548 Node* container = containerNode();
549 if (container && container->parentNode())
550 return false;
551
552 switch (m_anchorType) {
553 case PositionIsOffsetInAnchor:
554 return m_offset >= static_cast<unsigned>(lastOffsetForEditing(*m_anchorNode));
555 case PositionIsBeforeAnchor:
556 return false;
557 case PositionIsAfterAnchor:
558 return !m_anchorNode->nextSibling();
559 case PositionIsBeforeChildren:
560 return !lastOffsetForEditing(*m_anchorNode);
561 case PositionIsAfterChildren:
562 return true;
563 }
564 ASSERT_NOT_REACHED();
565 return false;
566}

568// return first preceding DOM position rendered at a different location, or "this"
569Position Position::previousCharacterPosition(Affinity affinity) const
570{
571 if (isNull())
572 return { };
573
574 Node* fromRootEditableElement = deprecatedNode()->rootEditableElement();
575
576 bool atStartOfLine = isStartOfLine(VisiblePosition(*this, affinity));
577 bool rendered = isCandidate();
578
579 Position currentPosition = *this;
580 while (!currentPosition.atStartOfTree()) {
581 currentPosition = currentPosition.previous();
582
583 if (currentPosition.deprecatedNode()->rootEditableElement() != fromRootEditableElement)
584 return *this;
585
586 if (atStartOfLine || !rendered) {
587 if (currentPosition.isCandidate())
588 return currentPosition;
589 } else if (rendersInDifferentPosition(currentPosition))
590 return currentPosition;
591 }
592
593 return *this;
594}

596// return first following position rendered at a different location, or "this"
597Position Position::nextCharacterPosition(Affinity affinity) const
598{
599 if (isNull())
600 return { };
601
602 Node* fromRootEditableElement = deprecatedNode()->rootEditableElement();
603
604 bool atEndOfLine = isEndOfLine({ *this, affinity });
605 bool rendered = isCandidate();
606
607 Position currentPosition = *this;
608 while (!currentPosition.atEndOfTree()) {
609 currentPosition = currentPosition.next();
610
611 if (currentPosition.deprecatedNode()->rootEditableElement() != fromRootEditableElement)
612 return *this;
613
614 if (atEndOfLine || !rendered) {
615 if (currentPosition.isCandidate())
616 return currentPosition;
617 } else if (rendersInDifferentPosition(currentPosition))
618 return currentPosition;
619 }
620
621 return *this;
622}

624// Whether or not [node, 0] and [node, lastOffsetForEditing(node)] are their own VisiblePositions.
625// If true, adjacent candidates are visually distinct.
626// FIXME: Disregard nodes with renderers that have no height, as we do in isCandidate.
627// FIXME: Share code with isCandidate, if possible.
628static bool endsOfNodeAreVisuallyDistinctPositions(Node* node)
629{
630 if (!node || !node->renderer())
631 return false;
632
633 if (!node->renderer()->isInline())
634 return true;
635
636 // Don't include inline tables.
637 if (is<HTMLTableElement>(*node))
638 return false;
639
640 if (!node->renderer()->isReplaced() || !canHaveChildrenForEditing(*node) || !downcast<RenderBox>(*node->renderer()).height())
641 return false;
642
643 // There is a VisiblePosition inside an empty inline-block container.
644 if (!node->hasChildNodes())
645 return true;
646
647 return !Position::hasRenderedNonAnonymousDescendantsWithHeight(downcast<RenderElement>(*node->renderer()));
648}
649
650static Node* enclosingVisualBoundary(Node* node)
651{
652 while (node && !endsOfNodeAreVisuallyDistinctPositions(node))
653 node = node->parentNode();
654
655 return node;
656}

658// upstream() and downstream() want to return positions that are either in a
659// text node or at just before a non-text node. This method checks for that.
660static bool isStreamer(const PositionIterator& pos)
661{
662 if (!pos.node())
663 return true;
664
665 if (isAtomicNode(pos.node()))
666 return true;
667
668 return pos.atStartOfNode();
669}

671// This function and downstream() are used for moving back and forth between visually equivalent candidates.
672// For example, for the text node "foo bar" where whitespace is collapsible, there are two candidates
673// that map to the VisiblePosition between 'b' and the space. This function will return the left candidate
674// and downstream() will return the right one.
675// Also, upstream() will return [boundary, 0] for any of the positions from [boundary, 0] to the first candidate
676// in boundary, where endsOfNodeAreVisuallyDistinctPositions(boundary) is true.
677Position Position::upstream(EditingBoundaryCrossingRule rule) const
678{
679 Node* startNode = deprecatedNode();
680 if (!startNode)
681 return { };
682
683 // iterate backward from there, looking for a qualified position
684 Node* boundary = enclosingVisualBoundary(startNode);
685 // FIXME: PositionIterator should respect Before and After positions.
686 PositionIterator lastVisible = m_anchorType == PositionIsAfterAnchor ? makeDeprecatedLegacyPosition(m_anchorNode.get(), caretMaxOffset(*m_anchorNode)) : *this;
687 PositionIterator currentPosition = lastVisible;
688 bool startEditable = startNode->hasEditableStyle();
689 Node* lastNode = startNode;
690 bool boundaryCrossed = false;
691 for (; !currentPosition.atStart(); currentPosition.decrement()) {
692 auto& currentNode = *currentPosition.node();
693
694 // Don't check for an editability change if we haven't moved to a different node,
695 // to avoid the expense of computing hasEditableStyle().
696 if (&currentNode != lastNode) {
697 // Don't change editability.
698 bool currentEditable = currentNode.hasEditableStyle();
699 if (startEditable != currentEditable) {
700 if (rule == CannotCrossEditingBoundary)
701 break;
702 boundaryCrossed = true;
703 }
704 lastNode = &currentNode;
705 }
706
707 // If we've moved to a position that is visually distinct, return the last saved position. There
708 // is code below that terminates early if we're *about* to move to a visually distinct position.
709 if (endsOfNodeAreVisuallyDistinctPositions(&currentNode) && &currentNode != boundary)
710 return lastVisible;
711
712 // skip position in unrendered or invisible node
713 RenderObject* renderer = currentNode.renderer();
714 if (!renderer || renderer->style().visibility() != Visibility::Visible)
715 continue;
716
717 if (rule == CanCrossEditingBoundary && boundaryCrossed) {
718 lastVisible = currentPosition;
719 break;
720 }
721
722 // track last visible streamer position
723 if (isStreamer(currentPosition))
724 lastVisible = currentPosition;
725
726 // Don't move past a position that is visually distinct. We could rely on code above to terminate and
727 // return lastVisible on the next iteration, but we terminate early to avoid doing a computeNodeIndex() call.
728 if (endsOfNodeAreVisuallyDistinctPositions(&currentNode) && currentPosition.atStartOfNode())
729 return lastVisible;
730
731 // Return position after tables and nodes which have content that can be ignored.
732 if (editingIgnoresContent(currentNode) || isRenderedTable(&currentNode)) {
733 if (currentPosition.atEndOfNode())
734 return positionAfterNode(&currentNode);
735 continue;
736 }
737
738 // return current position if it is in rendered text
739 if (is<RenderText>(*renderer)) {
740 auto& textRenderer = downcast<RenderText>(*renderer);
741
742 auto firstTextRun = LayoutIntegration::firstTextRunInTextOrderFor(textRenderer);
743 if (!firstTextRun)
744 continue;
745
746 if (&currentNode != startNode) {
747 // This assertion fires in layout tests in the case-transform.html test because
748 // of a mix-up between offsets in the text in the DOM tree with text in the
749 // render tree which can have a different length due to case transformation.
750 // Until we resolve that, disable this so we can run the layout tests!
751 //ASSERT(currentOffset >= renderer->caretMaxOffset());
752 return makeDeprecatedLegacyPosition(&currentNode, renderer->caretMaxOffset());
753 }
754
755 unsigned textOffset = currentPosition.offsetInLeafNode();
756 for (auto run = firstTextRun; run;) {
757 if (textOffset > run->start() && textOffset <= run->end())
758 return currentPosition;
759
760 auto nextRun = run.nextTextRunInTextOrder();
761 if (textOffset == run->end() + 1 && nextRun && run.line() != nextRun.line())
762 return currentPosition;
763
764 run = nextRun;
765 }
766 }
767 }
768
769 return lastVisible;
770}

772// This function and upstream() are used for moving back and forth between visually equivalent candidates.
773// For example, for the text node "foo bar" where whitespace is collapsible, there are two candidates
774// that map to the VisiblePosition between 'b' and the space. This function will return the right candidate
775// and upstream() will return the left one.
776// Also, downstream() will return the last position in the last atomic node in boundary for all of the positions
777// in boundary after the last candidate, where endsOfNodeAreVisuallyDistinctPositions(boundary).
778// FIXME: This function should never be called when the line box tree is dirty. See https://bugs.webkit.org/show_bug.cgi?id=97264
779Position Position::downstream(EditingBoundaryCrossingRule rule) const
780{
781 Node* startNode = deprecatedNode();
782 if (!startNode)
783 return { };
784
785 // iterate forward from there, looking for a qualified position
786 Node* boundary = enclosingVisualBoundary(startNode);
787 // FIXME: PositionIterator should respect Before and After positions.
788 PositionIterator lastVisible = m_anchorType == PositionIsAfterAnchor ? makeDeprecatedLegacyPosition(m_anchorNode.get(), caretMaxOffset(*m_anchorNode)) : *this;
789 PositionIterator currentPosition = lastVisible;
790 bool startEditable = startNode->hasEditableStyle();
791 Node* lastNode = startNode;
792 bool boundaryCrossed = false;
793 for (; !currentPosition.atEnd(); currentPosition.increment()) {
794 auto& currentNode = *currentPosition.node();
795
796 // Don't check for an editability change if we haven't moved to a different node,
797 // to avoid the expense of computing hasEditableStyle().
798 if (&currentNode != lastNode) {
799 // Don't change editability.
800 bool currentEditable = currentNode.hasEditableStyle();
801 if (startEditable != currentEditable) {
802 if (rule == CannotCrossEditingBoundary)
803 break;
804 boundaryCrossed = true;
805 }
806
807 lastNode = &currentNode;
808 }
809
810 // stop before going above the body, up into the head
811 // return the last visible streamer position
812 if (is<HTMLBodyElement>(currentNode) && currentPosition.atEndOfNode())
813 break;
814
815 // Do not move to a visually distinct position.
816 if (endsOfNodeAreVisuallyDistinctPositions(&currentNode) && &currentNode != boundary)
817 return lastVisible;
818 // Do not move past a visually disinct position.
819 // Note: The first position after the last in a node whose ends are visually distinct
820 // positions will be [boundary->parentNode(), originalBlock->computeNodeIndex() + 1].
821 if (boundary && boundary->parentNode() == &currentNode)
822 return lastVisible;
823
824 // skip position in unrendered or invisible node
825 auto* renderer = currentNode.renderer();
826 if (!renderer || renderer->style().visibility() != Visibility::Visible)
827 continue;
828
829 if (rule == CanCrossEditingBoundary && boundaryCrossed) {
830 lastVisible = currentPosition;
831 break;
832 }
833
834 // track last visible streamer position
835 if (isStreamer(currentPosition))
836 lastVisible = currentPosition;
837
838 // Return position before tables and nodes which have content that can be ignored.
839 if (editingIgnoresContent(currentNode) || isRenderedTable(&currentNode)) {
840 if (currentPosition.atStartOfNode())
841 return positionBeforeNode(&currentNode);
842 continue;
843 }
844
845 // return current position if it is in rendered text
846 if (is<RenderText>(*renderer)) {
847 auto& textRenderer = downcast<RenderText>(*renderer);
848
849 auto firstTextRun = LayoutIntegration::firstTextRunInTextOrderFor(textRenderer);
850 if (!firstTextRun)
851 continue;
852
853 if (&currentNode != startNode) {
854 ASSERT(currentPosition.atStartOfNode());
855 return makeContainerOffsetPosition(&currentNode, textRenderer.caretMinOffset());
856 }
857
858 unsigned textOffset = currentPosition.offsetInLeafNode();
859 for (auto run = firstTextRun; run;) {
860 if (!run->length() && textOffset == run->start())
861 return currentPosition;
862
863 if (textOffset >= run->start() && textOffset < run->end())
864 return currentPosition;
865
866 auto nextRun = run.nextTextRunInTextOrder();
867 if (textOffset == run->end() && nextRun && run.line() != nextRun.line())
868 return currentPosition;
869
870 run = nextRun;
871 }
872 }
873 }
874
875 return lastVisible;
876}

878unsigned Position::positionCountBetweenPositions(const Position& a, const Position& b)
879{
880 if (a.isNull() || b.isNull())
881 return UINT_MAX;
882
883 Position endPos;
884 Position pos;
885 if (a > b) {
886 endPos = a;
887 pos = b;
888 } else if (a < b) {
889 endPos = b;
890 pos = a;
891 } else
892 return 0;
893
894 unsigned posCount = 0;
895 while (!pos.atEndOfTree() && pos != endPos) {
896 pos = pos.next();
897 ++posCount;
898 }
899 return posCount;
900}

970bool Position::isCandidate() const
971{
972 if (isNull())
973 return false;
974
975 auto* renderer = deprecatedNode()->renderer();
976 if (!renderer)
977 return false;
978
979 if (renderer->style().visibility() != Visibility::Visible)
980 return false;
981
982 if (renderer->isBR()) {
983 // FIXME: The condition should be m_anchorType == PositionIsBeforeAnchor, but for now we still need to support legacy positions.
984 return !m_offset && m_anchorType != PositionIsAfterAnchor && !nodeIsUserSelectNone(deprecatedNode()->parentNode());
985 }
986
987 if (is<RenderText>(*renderer))
988 return !nodeIsUserSelectNone(deprecatedNode()) && downcast<RenderText>(*renderer).containsCaretOffset(m_offset);
989
990 if (positionBeforeOrAfterNodeIsCandidate(*deprecatedNode())) {
991 return ((atFirstEditingPositionForNode() && m_anchorType == PositionIsBeforeAnchor)
992 || (atLastEditingPositionForNode() && m_anchorType == PositionIsAfterAnchor))
993 && !nodeIsUserSelectNone(deprecatedNode()->parentNode());
994 }
995
996 if (is<HTMLHtmlElement>(*m_anchorNode))
997 return false;
998
999 if (is<RenderBlockFlow>(*renderer) || is<RenderGrid>(*renderer) || is<RenderFlexibleBox>(*renderer)) {
1000 RenderBlock& block = downcast<RenderBlock>(*renderer);
1001 if (block.logicalHeight() || is<HTMLBodyElement>(*m_anchorNode) || m_anchorNode->isRootEditableElement()) {
1002 if (!Position::hasRenderedNonAnonymousDescendantsWithHeight(block))
1003 return atFirstEditingPositionForNode() && !Position::nodeIsUserSelectNone(deprecatedNode());
1004 return m_anchorNode->hasEditableStyle() && !Position::nodeIsUserSelectNone(deprecatedNode()) && atEditingBoundary();
1005 }
1006 return false;
1007 }
1008
1009 return m_anchorNode->hasEditableStyle() && !Position::nodeIsUserSelectNone(deprecatedNode()) && atEditingBoundary();
1010}

1029bool Position::rendersInDifferentPosition(const Position& position) const
1030{
1031 if (isNull() || position.isNull())
1032 return false;
1033
1034 auto* renderer = deprecatedNode()->renderer();
1035 if (!renderer)
1036 return false;
1037
1038 auto* positionRenderer = position.deprecatedNode()->renderer();
1039 if (!positionRenderer)
1040 return false;
1041
1042 if (renderer->style().visibility() != Visibility::Visible || positionRenderer->style().visibility() != Visibility::Visible)
1043 return false;
1044
1045 if (deprecatedNode() == position.deprecatedNode()) {
1046 if (is<HTMLBRElement>(*deprecatedNode()))
1047 return false;
1048
1049 if (m_offset == static_cast<unsigned>(position.deprecatedEditingOffset()))
1050 return false;
1051
1052 if (!is<Text>(*deprecatedNode()))
1053 return true;
1054 }
1055
1056 if (is<HTMLBRElement>(*deprecatedNode()) && position.isCandidate())
1057 return true;
1058
1059 if (is<HTMLBRElement>(*position.deprecatedNode()) && isCandidate())
1060 return true;
1061
1062 if (!inSameEnclosingBlockFlowElement(deprecatedNode(), position.deprecatedNode()))
1063 return true;
1064
1065 if (is<RenderText>(*renderer) && !downcast<RenderText>(*renderer).containsCaretOffset(m_offset))
1066 return false;
1067
1068 if (is<RenderText>(*positionRenderer) && !downcast<RenderText>(*positionRenderer).containsCaretOffset(position.m_offset))
1069 return false;
1070
1071 unsigned thisRenderedOffset = is<RenderText>(*renderer) ? downcast<RenderText>(*renderer).countRenderedCharacterOffsetsUntil(m_offset) : m_offset;
1072 unsigned positionRenderedOffset = is<RenderText>(*positionRenderer) ? downcast<RenderText>(*positionRenderer).countRenderedCharacterOffsetsUntil(position.m_offset) : position.m_offset;
1073
1074 if (renderer == positionRenderer && thisRenderedOffset == positionRenderedOffset)
1075 return false;
1076
1077 auto run1 = inlineRunAndOffset(Affinity::Downstream).run;
1078 auto run2 = position.inlineRunAndOffset(Affinity::Downstream).run;
1079
1080 LOG(Editing, "renderer: %p\n", renderer);
1081 LOG(Editing, "thisRenderedOffset: %d\n", thisRenderedOffset);
1082 LOG(Editing, "posRenderer: %p\n", positionRenderer);
1083 LOG(Editing, "posRenderedOffset: %d\n", positionRenderedOffset);
1084 LOG(Editing, "node min/max: %d:%d\n", caretMinOffset(*deprecatedNode()), caretMaxOffset(*deprecatedNode()));
1085 LOG(Editing, "pos node min/max: %d:%d\n", caretMinOffset(*position.deprecatedNode()), caretMaxOffset(*position.deprecatedNode()));
1086 LOG(Editing, "----------------------------------------------------------------------\n");
1087
1088 if (!run1 || !run2)
1089 return false;
1090
1091 if (run1.line() != run2.line())
1092 return true;
1093
1094 if (nextRenderedEditable(deprecatedNode()) == position.deprecatedNode()
1095 && thisRenderedOffset == static_cast<unsigned>(caretMaxOffset(*deprecatedNode())) && !positionRenderedOffset) {
1096 return false;
1097 }
1098
1099 if (previousRenderedEditable(deprecatedNode()) == position.deprecatedNode()
1100 && !thisRenderedOffset && positionRenderedOffset == static_cast<unsigned>(caretMaxOffset(*position.deprecatedNode()))) {
1101 return false;
1102 }
1103
1104 return true;
1105}

1197InlineRunAndOffset Position::inlineRunAndOffset(Affinity affinity, TextDirection primaryDirection) const
1198{
1199 auto caretOffset = static_cast<unsigned>(deprecatedEditingOffset());
1200
1201 auto node = deprecatedNode();
1202 if (!node)
1203 return { { }, caretOffset };
1204 auto renderer = node->renderer();
1205 if (!renderer)
1206 return { { }, caretOffset };
1207
1208 LayoutIntegration::RunIterator run;
1209
1210 if (renderer->isBR()) {
1211 auto& lineBreakRenderer = downcast<RenderLineBreak>(*renderer);
1212 if (!caretOffset)
1213 run = LayoutIntegration::runFor(lineBreakRenderer);
1214 } else if (is<RenderText>(*renderer)) {
1215 auto& textRenderer = downcast<RenderText>(*renderer);
1216
1217 auto textRun = LayoutIntegration::firstTextRunFor(textRenderer);
1218 LayoutIntegration::TextRunIterator candidate;
1219
1220 for (; textRun; ++textRun) {
1221 unsigned caretMinOffset = textRun->minimumCaretOffset();
1222 unsigned caretMaxOffset = textRun->maximumCaretOffset();
1223
1224 if (caretOffset < caretMinOffset || caretOffset > caretMaxOffset || (caretOffset == caretMaxOffset && textRun->isLineBreak()))
1225 continue;
1226
1227 if (caretOffset > caretMinOffset && caretOffset < caretMaxOffset)
1228 return { textRun, caretOffset };
1229
1230 if ((caretOffset == caretMaxOffset) ^ (affinity == Affinity::Downstream))
1231 break;
1232
1233 if ((caretOffset == caretMinOffset) ^ (affinity == Affinity::Upstream))
1234 break;
1235
1236 if (caretOffset == caretMaxOffset) {
1237 auto nextOnLine = textRun.nextOnLine();
1238 if (nextOnLine && nextOnLine->isLineBreak())
1239 break;
1240 }
1241
1242 candidate = textRun;
1243 }
1244
1245 if (candidate && !candidate.nextTextRun() && affinity == Affinity::Downstream) {
1246 textRun = searchAheadForBetterMatch(textRenderer);
1247 if (textRun)
1248 caretOffset = textRun->minimumCaretOffset();
1249 }
1250
1251 run = textRun ? textRun : candidate;
1252 } else {
1253 if (canHaveChildrenForEditing(*deprecatedNode()) && is<RenderBlockFlow>(*renderer) && hasRenderedNonAnonymousDescendantsWithHeight(downcast<RenderBlockFlow>(*renderer))) {
1254 // Try a visually equivalent position with possibly opposite editability. This helps in case |this| is in
1255 // an editable block but surrounded by non-editable positions. It acts to negate the logic at the beginning
1256 // of RenderObject::createVisiblePosition().
1257 Position equivalent = downstreamIgnoringEditingBoundaries(*this);
1258 if (equivalent == *this) {
1259 equivalent = upstreamIgnoringEditingBoundaries(*this);
1260 // FIXME: Can returning nullptr really be correct here?
1261 if (equivalent == *this || downstreamIgnoringEditingBoundaries(equivalent) == *this)
1262 return { { }, caretOffset };
1263 }
1264
1265 return equivalent.inlineRunAndOffset(Affinity::Upstream, primaryDirection);
1266 }
1267 if (is<RenderBox>(*renderer)) {
1268 run = LayoutIntegration::runFor(downcast<RenderBox>(*renderer));
1269 if (run && caretOffset > run->minimumCaretOffset() && caretOffset < run->maximumCaretOffset())
1270 return { run, caretOffset };
1271 }
1272 }
1273
1274 if (!run)
1275 return { { }, caretOffset };
1276
1277 unsigned char level = run->bidiLevel();
1278
1279 if (run->direction() == primaryDirection) {
1280 if (caretOffset == run->rightmostCaretOffset()) {
1281 auto nextRun = run.nextOnLine();
1282 if (!nextRun || nextRun->bidiLevel() >= level)
1283 return { run, caretOffset };
1284
1285 level = nextRun->bidiLevel();
1286
1287 auto previousRun = run.previousOnLine();
1288 for (; previousRun; previousRun.traversePreviousOnLine()) {
1289 if (previousRun->bidiLevel() <= level)
1290 break;
1291 }
1292
1293 if (previousRun && previousRun->bidiLevel() == level) // For example, abc FED 123 ^ CBA
1294 return { run, caretOffset };
1295
1296 // For example, abc 123 ^ CBA
1297 for (; nextRun; nextRun.traverseNextOnLine()) {
1298 if (nextRun->bidiLevel() < level)
1299 break;
1300 run = nextRun;
1301 }
1302 caretOffset = run->rightmostCaretOffset();
1303 } else {
1304 auto previousRun = run.previousOnLine();
1305 if (!previousRun || previousRun->bidiLevel() >= level)
1306 return { run, caretOffset };
1307
1308 level = previousRun->bidiLevel();
1309
1310 auto nextRun = run.nextOnLine();
1311 for (; nextRun; nextRun.traverseNextOnLine()) {
1312 if (nextRun->bidiLevel() <= level)
1313 break;
1314 }
1315
1316 if (nextRun && nextRun->bidiLevel() == level)
1317 return { run, caretOffset };
1318
1319 for (; previousRun; previousRun.traversePreviousOnLine()) {
1320 if (previousRun->bidiLevel() < level)
1321 break;
1322 run = previousRun;
1323 }
1324
1325 caretOffset = run->leftmostCaretOffset();
1326 }
1327 return { run, caretOffset };
1328 }
1329
1330 if (caretOffset == run->leftmostCaretOffset()) {
1331 auto previousRun = run.previousOnLineIgnoringLineBreak();
1332 if (!previousRun || previousRun->bidiLevel() < level) {
1333 // Left edge of a secondary run. Set to the right edge of the entire run.
1334 for (auto nextRun = run.nextOnLineIgnoringLineBreak(); nextRun; nextRun.traverseNextOnLineIgnoringLineBreak()) {
1335 if (nextRun->bidiLevel() < level)
1336 break;
1337 run = nextRun;
1338 }
1339 caretOffset = run->rightmostCaretOffset();
1340 } else if (previousRun->bidiLevel() > level) {
1341 // Right edge of a "tertiary" run. Set to the left edge of that run.
1342 for (auto tertiaryRun = run.previousOnLineIgnoringLineBreak(); tertiaryRun; tertiaryRun.traversePreviousOnLineIgnoringLineBreak()) {
1343 if (tertiaryRun->bidiLevel() <= level)
1344 break;
1345 run = tertiaryRun;
1346 }
1347 caretOffset = run->leftmostCaretOffset();
1348 }
1349 } else {
1350 auto nextRun = run.nextOnLineIgnoringLineBreak();
1351 if (!nextRun || nextRun->bidiLevel() < level) {
1352 // Right edge of a secondary run. Set to the left edge of the entire run.
1353 for (auto previousRun = run.previousOnLineIgnoringLineBreak(); previousRun; previousRun.traversePreviousOnLineIgnoringLineBreak()) {
1354 if (previousRun->bidiLevel() < level)
1355 break;
1356 run = previousRun;
1357 }
1358 caretOffset = run->leftmostCaretOffset();
1359 } else if (nextRun->bidiLevel() > level) {
1360 // Left edge of a "tertiary" run. Set to the right edge of that run.
1361 for (auto tertiaryRun = run.nextOnLineIgnoringLineBreak(); tertiaryRun; tertiaryRun.traverseNextOnLineIgnoringLineBreak()) {
1362 if (tertiaryRun->bidiLevel() <= level)
1363 break;
1364 run = tertiaryRun;
1365 }
1366 caretOffset = run->rightmostCaretOffset();
1367 }
1368 }
1369
1370 return { run, caretOffset };
1371}

1433bool Position::equals(const Position& other) const
1434{
1435 if (!m_anchorNode)
1436 return !m_anchorNode == !other.m_anchorNode;
1437 if (!other.m_anchorNode)
1438 return false;
1439
1440 switch (anchorType()) {
1441 case PositionIsBeforeChildren:
1442 ASSERT(!is<Text>(*m_anchorNode));
1443 switch (other.anchorType()) {
1444 case PositionIsBeforeChildren:
1445 ASSERT(!is<Text>(*other.m_anchorNode));
1446 return m_anchorNode == other.m_anchorNode;
1447 case PositionIsAfterChildren:
1448 ASSERT(!is<Text>(*other.m_anchorNode));
1449 return m_anchorNode == other.m_anchorNode && !m_anchorNode->hasChildNodes();
1450 case PositionIsOffsetInAnchor:
1451 return m_anchorNode == other.m_anchorNode && !other.m_offset;
1452 case PositionIsBeforeAnchor:
1453 return m_anchorNode->firstChild() == other.m_anchorNode;
1454 case PositionIsAfterAnchor:
1455 return false;
1456 }
1457 break;
1458 case PositionIsAfterChildren:
1459 ASSERT(!is<Text>(*m_anchorNode));
1460 switch (other.anchorType()) {
1461 case PositionIsBeforeChildren:
1462 ASSERT(!is<Text>(*other.m_anchorNode));
1463 return m_anchorNode == other.m_anchorNode && !m_anchorNode->hasChildNodes();
1464 case PositionIsAfterChildren:
1465 ASSERT(!is<Text>(*other.m_anchorNode));
1466 return m_anchorNode == other.m_anchorNode;
1467 case PositionIsOffsetInAnchor:
1468 return m_anchorNode == other.m_anchorNode && m_anchorNode->countChildNodes() == static_cast<unsigned>(m_offset);
1469 case PositionIsBeforeAnchor:
1470 return false;
1471 case PositionIsAfterAnchor:
1472 return m_anchorNode->lastChild() == other.m_anchorNode;
1473 }
1474 break;
1475 case PositionIsOffsetInAnchor:
1476 switch (other.anchorType()) {
1477 case PositionIsBeforeChildren:
1478 ASSERT(!is<Text>(*other.m_anchorNode));
1479 return m_anchorNode == other.m_anchorNode && !m_offset;
1480 case PositionIsAfterChildren:
1481 ASSERT(!is<Text>(*other.m_anchorNode));
1482 return m_anchorNode == other.m_anchorNode && m_offset == other.m_anchorNode->countChildNodes();
1483 case PositionIsOffsetInAnchor:
1484 return m_anchorNode == other.m_anchorNode && m_offset == other.m_offset;
1485 case PositionIsBeforeAnchor:
1486 return m_anchorNode->traverseToChildAt(m_offset) == other.m_anchorNode;
1487 case PositionIsAfterAnchor:
1488 return m_offset && m_anchorNode->traverseToChildAt(m_offset - 1) == other.m_anchorNode;
1489 }
1490 break;
1491 case PositionIsBeforeAnchor:
1492 switch (other.anchorType()) {
1493 case PositionIsBeforeChildren:
1494 ASSERT(!is<Text>(*other.m_anchorNode));
1495 return m_anchorNode == other.m_anchorNode->firstChild();
1496 case PositionIsAfterChildren:
1497 ASSERT(!is<Text>(*other.m_anchorNode));
1498 return false;
1499 case PositionIsOffsetInAnchor:
1500 return m_anchorNode == other.m_anchorNode->traverseToChildAt(other.m_offset);
1501 case PositionIsBeforeAnchor:
1502 return m_anchorNode == other.m_anchorNode;
1503 case PositionIsAfterAnchor:
1504 return m_anchorNode->previousSibling() == other.m_anchorNode;
1505 }
1506 break;
1507 case PositionIsAfterAnchor:
1508 switch (other.anchorType()) {
1509 case PositionIsBeforeChildren:
1510 ASSERT(!is<Text>(*other.m_anchorNode));
1511 return false;
1512 case PositionIsAfterChildren:
1513 ASSERT(!is<Text>(*other.m_anchorNode));
1514 return m_anchorNode == other.m_anchorNode->lastChild();
1515 case PositionIsOffsetInAnchor:
1516 return other.m_offset && m_anchorNode == other.m_anchorNode->traverseToChildAt(other.m_offset - 1);
1517 case PositionIsBeforeAnchor:
1518 return m_anchorNode->nextSibling() == other.m_anchorNode;
1519 case PositionIsAfterAnchor:
1520 return m_anchorNode == other.m_anchorNode;
1521 }
1522 break;
1523 }
1524
1525 ASSERT_NOT_REACHED();
1526 return false;
1527}

1612PartialOrdering documentOrder(const Position& a, const Position& b)
1613{
1614 if (a.isNull() || b.isNull())
1615 return a.isNull() && b.isNull() ? PartialOrdering::equivalent : PartialOrdering::unordered;
1616
1617 auto aContainer = a.containerNode();
1618 auto bContainer = b.containerNode();
1619
1620 if (!aContainer || !bContainer) {
1621 if (!commonInclusiveAncestor<ComposedTree>(*a.anchorNode(), *b.anchorNode()))
1622 return PartialOrdering::unordered;
1623 if (!aContainer && !bContainer && a.anchorType() == b.anchorType())
1624 return PartialOrdering::equivalent;
1625 if (bContainer)
1626 return a.anchorType() == Position::PositionIsBeforeAnchor ? PartialOrdering::less : PartialOrdering::greater;
1627 return b.anchorType() == Position::PositionIsBeforeAnchor ? PartialOrdering::greater : PartialOrdering::less;
1628 }
1629
1630 // FIXME: Avoid computing node offset for cases where we don't need to.
1631
1632 return treeOrder<ComposedTree>(*makeBoundaryPoint(a), *makeBoundaryPoint(b));
1633}
