26#pragma once
27
28#include "Event.h"
29#include "JSValueInWrappedObject.h"
30
31namespace WebCore {
32
33class DOMPromise;
34
35class PromiseRejectionEvent final : public Event {
36 WTF_MAKE_ISO_ALLOCATED(PromiseRejectionEvent);
37public:
38 struct Init : EventInit {
39 RefPtr<DOMPromise> promise;
40 JSC::JSValue reason;
41 };
42
43 static Ref<PromiseRejectionEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
44 {
45 return adoptRef(*new PromiseRejectionEvent(type, initializer, isTrusted));
46 }
47
48 virtual ~PromiseRejectionEvent();
49
50 DOMPromise& promise() const { return m_promise.get(); }
51 const JSValueInWrappedObject& reason() const { return m_reason; }
52
53 EventInterface eventInterface() const override { return PromiseRejectionEventInterfaceType; }
54
55private:
56 PromiseRejectionEvent(const AtomString&, const Init&, IsTrusted);
57
58 Ref<DOMPromise> m_promise;
59 JSValueInWrappedObject m_reason;
60};
61
62} // namespace WebCore
