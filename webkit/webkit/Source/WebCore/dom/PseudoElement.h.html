34class PseudoElement final : public Element {
35 WTF_MAKE_ISO_ALLOCATED(PseudoElement);
36public:
37 static Ref<PseudoElement> create(Element& host, PseudoId);
38 virtual ~PseudoElement();
39
40 Element* hostElement() const { return m_hostElement; }
41 void clearHostElement();
42
43 bool rendererIsNeeded(const RenderStyle&) override;
44
45 bool canStartSelection() const override { return false; }
46 bool canContainRangeEndPoint() const override { return false; }
47
48private:
49 PseudoElement(Element&, PseudoId);
50
51 PseudoId customPseudoId() const override { return m_pseudoId; }
52
53 Element* m_hostElement;
54 PseudoId m_pseudoId;
55};

59} // namespace WebCore
