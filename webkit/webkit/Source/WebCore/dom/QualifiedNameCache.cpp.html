28#include "QualifiedNameCache.h"

30namespace WebCore {
31
32struct QNameComponentsTranslator {
33 static unsigned hash(const QualifiedNameComponents& components)
34 {
35 return hashComponents(components);
36 }
37 static bool equal(QualifiedName::QualifiedNameImpl* name, const QualifiedNameComponents& c)
38 {
39 return c.m_prefix == name->m_prefix.impl() && c.m_localName == name->m_localName.impl() && c.m_namespace == name->m_namespace.impl();
40 }
41 static void translate(QualifiedName::QualifiedNameImpl*& location, const QualifiedNameComponents& components, unsigned)
42 {
43 location = &QualifiedName::QualifiedNameImpl::create(components.m_prefix, components.m_localName, components.m_namespace).leakRef();
44 }
45};
46
47Ref<QualifiedName::QualifiedNameImpl> QualifiedNameCache::getOrCreate(const QualifiedNameComponents& components)
48{
49 QNameSet::AddResult addResult = m_cache.add<QNameComponentsTranslator>(components);
50 return addResult.isNewEntry ? adoptRef(**addResult.iterator) : Ref<QualifiedName::QualifiedNameImpl> { **addResult.iterator };
51}
52
53void QualifiedNameCache::remove(QualifiedName::QualifiedNameImpl& impl)
54{
55 m_cache.remove(&impl);
56}
