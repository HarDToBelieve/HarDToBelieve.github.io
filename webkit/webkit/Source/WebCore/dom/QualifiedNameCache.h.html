45class QualifiedNameCache {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 QualifiedNameCache() = default;
49
50 Ref<QualifiedName::QualifiedNameImpl> getOrCreate(const QualifiedNameComponents&);
51 void remove(QualifiedName::QualifiedNameImpl&);
52
53private:
54 static const int staticQualifiedNamesCount = HTMLNames::HTMLTagsCount + HTMLNames::HTMLAttrsCount
55#if ENABLE(MATHML)
56 + MathMLNames::MathMLTagsCount + MathMLNames::MathMLAttrsCount
57#endif
58 + SVGNames::SVGTagsCount + SVGNames::SVGAttrsCount
59 + XLinkNames::XLinkAttrsCount
60 + XMLNSNames::XMLNSAttrsCount
61 + XMLNames::XMLAttrsCount;
62
63 struct QualifiedNameHashTraits : public HashTraits<QualifiedName::QualifiedNameImpl*> {
64 static const int minimumTableSize = WTF::HashTableCapacityForSize<staticQualifiedNamesCount>::value;
65 };
66
67 using QNameSet = HashSet<QualifiedName::QualifiedNameImpl*, QualifiedNameHash, QualifiedNameHashTraits>;
68 QNameSet m_cache;
69};
