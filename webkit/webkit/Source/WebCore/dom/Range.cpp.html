107{

117ExceptionOr<void> Range::setStart(Ref<Node>&& container, unsigned offset)
118{
119 auto childNode = checkNodeOffsetPair(container, offset);
120 if (childNode.hasException())
121 return childNode.releaseException();
122
123 m_start.set(WTFMove(container), offset, childNode.releaseReturnValue());
124 if (!is_lteq(treeOrder(makeBoundaryPoint(m_start), makeBoundaryPoint(m_end))))
125 m_end = m_start;
126 updateAssociatedSelection();
127 updateDocument();
128 return { };
129}
130
131ExceptionOr<void> Range::setEnd(Ref<Node>&& container, unsigned offset)
132{
133 auto childNode = checkNodeOffsetPair(container, offset);
134 if (childNode.hasException())
135 return childNode.releaseException();
136
137 m_end.set(WTFMove(container), offset, childNode.releaseReturnValue());
138 if (!is_lteq(treeOrder(makeBoundaryPoint(m_start), makeBoundaryPoint(m_end))))
139 m_start = m_end;
140 updateAssociatedSelection();
141 updateDocument();
142 return { };
143}

145void Range::collapse(bool toStart)

147 if (toStart)
148 m_end = m_start;
149 else
150 m_start = m_end;
151 updateAssociatedSelection();

154ExceptionOr<bool> Range::isPointInRange(Node& container, unsigned offset)
155{
156 if (auto checkResult = checkNodeOffsetPair(container, offset); checkResult.hasException()) {
157 // DOM specification requires this check be done first but since there are no side effects,
158 // we can do it in reverse order to avoid an extra root node check in the common case.
159 if (&container.rootNode() != &startContainer().rootNode())
160 return false;
161 return checkResult.releaseException();
162 }
163 return contains(makeSimpleRange(*this), { container, offset });
164}

166ExceptionOr<short> Range::comparePoint(Node& container, unsigned offset) const
167{
168 if (auto checkResult = checkNodeOffsetPair(container, offset); checkResult.hasException()) {
169 // DOM specification requires this check be done first but since there are no side effects,
170 // we can do it in reverse order to avoid an extra root node check in the common case.
171 if (&container.rootNode() != &startContainer().rootNode())
172 return Exception { WrongDocumentError };
173 return checkResult.releaseException();
174 }
175 auto ordering = treeOrder({ container, offset }, makeSimpleRange(*this));
176 if (is_lt(ordering))
177 return -1;
178 if (is_eq(ordering))
179 return 0;
180 if (is_gt(ordering))
181 return 1;
182 return Exception { WrongDocumentError };
183}

185ExceptionOr<Range::CompareResults> Range::compareNode(Node& node) const
186{
187 // FIXME: This deprecated function should be removed.
188 // We originally added it for interoperability with Firefox.
189 // Recent versions of Firefox have removed it.
190
191 // http://developer.mozilla.org/en/docs/DOM:range.compareNode
192 // This method returns 0, 1, 2, or 3 based on if the node is before, after,
193 // before and after(surrounds), or inside the range, respectively.
194
195 if (!node.isConnected() || &node.document() != m_ownerDocument.ptr()) {
196 // Match historical Firefox behavior.
197 return NODE_BEFORE;
198 }
199
200 auto nodeRange = makeRangeSelectingNode(node);
201 if (!nodeRange) {
202 // Match historical Firefox behavior.
203 return Exception { NotFoundError };
204 }
205
206 auto startOrdering = treeOrder(nodeRange->start, makeBoundaryPoint(m_start));
207 auto endOrdering = treeOrder(nodeRange->end, makeBoundaryPoint(m_end));
208 if (is_gteq(startOrdering) && is_lteq(endOrdering))
209 return NODE_INSIDE;
210 if (is_lteq(startOrdering) && is_gteq(endOrdering))
211 return NODE_BEFORE_AND_AFTER;
212 if (is_lteq(startOrdering))
213 return NODE_BEFORE;
214 if (is_gteq(endOrdering))
215 return NODE_AFTER;
216 return Exception { WrongDocumentError };
217}

219ExceptionOr<short> Range::compareBoundaryPoints(unsigned short how, const Range& sourceRange) const
220{
221 const RangeBoundaryPoint* thisPoint;
222 const RangeBoundaryPoint* otherPoint;
223 switch (how) {
224 case START_TO_START:
225 thisPoint = &m_start;
226 otherPoint = &sourceRange.m_start;
227 break;
228 case START_TO_END:
229 thisPoint = &m_end;
230 otherPoint = &sourceRange.m_start;
231 break;
232 case END_TO_END:
233 thisPoint = &m_end;
234 otherPoint = &sourceRange.m_end;
235 break;
236 case END_TO_START:
237 thisPoint = &m_start;
238 otherPoint = &sourceRange.m_end;
239 break;
240 default:
241 return Exception { NotSupportedError };
242 }
243 auto ordering = treeOrder(makeBoundaryPoint(*thisPoint), makeBoundaryPoint(*otherPoint));
244 if (is_lt(ordering))
245 return -1;
246 if (is_eq(ordering))
247 return 0;
248 if (is_gt(ordering))
249 return 1;
250 return Exception { WrongDocumentError };
251}

266static inline Node* highestAncestorUnderCommonRoot(Node* node, Node* commonRoot)
267{
268 if (node == commonRoot)
269 return 0;
270
271 ASSERT(commonRoot->contains(node));
272
273 while (node->parentNode() != commonRoot)
274 node = node->parentNode();
275
276 return node;
277}

279static inline Node* childOfCommonRootBeforeOffset(Node* container, unsigned offset, Node* commonRoot)
280{
281 ASSERT(container);
282 ASSERT(commonRoot);
283
284 if (!commonRoot->contains(container))
285 return 0;
286
287 if (container == commonRoot) {
288 container = container->firstChild();
289 for (unsigned i = 0; container && i < offset; i++)
290 container = container->nextSibling();
291 } else {
292 while (container->parentNode() != commonRoot)
293 container = container->parentNode();
294 }
295
296 return container;
297}

299ExceptionOr<RefPtr<DocumentFragment>> Range::processContents(ActionType action)
300{
301 RefPtr<DocumentFragment> fragment;
302 if (action == Extract || action == Clone)
303 fragment = DocumentFragment::create(m_ownerDocument);
304
305 if (collapsed())
306 return fragment;
307
308 RefPtr<Node> commonRoot = commonAncestorContainer();
309 ASSERT(commonRoot);
310
311 if (&startContainer() == &endContainer()) {
312 auto result = processContentsBetweenOffsets(action, fragment, &startContainer(), m_start.offset(), m_end.offset());
313 if (result.hasException())
314 return result.releaseException();
315 return fragment;
316 }
317
318 // Since mutation events can modify the range during the process, the boundary points need to be saved.
319 RangeBoundaryPoint originalStart(m_start);
320 RangeBoundaryPoint originalEnd(m_end);
321
322 // what is the highest node that partially selects the start / end of the range?
323 RefPtr<Node> partialStart = highestAncestorUnderCommonRoot(&originalStart.container(), commonRoot.get());
324 RefPtr<Node> partialEnd = highestAncestorUnderCommonRoot(&originalEnd.container(), commonRoot.get());
325
326 // Start and end containers are different.
327 // There are three possibilities here:
328 // 1. Start container == commonRoot (End container must be a descendant)
329 // 2. End container == commonRoot (Start container must be a descendant)
330 // 3. Neither is commonRoot, they are both descendants
331 //
332 // In case 3, we grab everything after the start (up until a direct child
333 // of commonRoot) into leftContents, and everything before the end (up until
334 // a direct child of commonRoot) into rightContents. Then we process all
335 // commonRoot children between leftContents and rightContents
336 //
337 // In case 1 or 2, we skip either processing of leftContents or rightContents,
338 // in which case the last lot of nodes either goes from the first or last
339 // child of commonRoot.
340 //
341 // These are deleted, cloned, or extracted (i.e. both) depending on action.
342
343 // Note that we are verifying that our common root hierarchy is still intact
344 // after any DOM mutation event, at various stages below. See webkit bug 60350.
345
346 RefPtr<Node> leftContents;
347 if (&originalStart.container() != commonRoot && commonRoot->contains(&originalStart.container())) {
348 auto firstResult = processContentsBetweenOffsets(action, nullptr, &originalStart.container(), originalStart.offset(), originalStart.container().length());
349 auto secondResult = processAncestorsAndTheirSiblings(action, &originalStart.container(), ProcessContentsForward, WTFMove(firstResult), commonRoot.get());
350 // FIXME: A bit peculiar that we silently ignore the exception here, but we do have at least some regression tests that rely on this behavior.
351 if (!secondResult.hasException())
352 leftContents = secondResult.releaseReturnValue();
353 }
354
355 RefPtr<Node> rightContents;
356 if (&endContainer() != commonRoot && commonRoot->contains(&originalEnd.container())) {
357 auto firstResult = processContentsBetweenOffsets(action, nullptr, &originalEnd.container(), 0, originalEnd.offset());
358 auto secondResult = processAncestorsAndTheirSiblings(action, &originalEnd.container(), ProcessContentsBackward, WTFMove(firstResult), commonRoot.get());
359 // FIXME: A bit peculiar that we silently ignore the exception here, but we do have at least some regression tests that rely on this behavior.
360 if (!secondResult.hasException())
361 rightContents = secondResult.releaseReturnValue();
362 }
363
364 // delete all children of commonRoot between the start and end container
365 RefPtr<Node> processStart = childOfCommonRootBeforeOffset(&originalStart.container(), originalStart.offset(), commonRoot.get());
366 if (processStart && &originalStart.container() != commonRoot) // processStart contains nodes before m_start.
367 processStart = processStart->nextSibling();
368 RefPtr<Node> processEnd = childOfCommonRootBeforeOffset(&originalEnd.container(), originalEnd.offset(), commonRoot.get());
369
370 // Collapse the range, making sure that the result is not within a node that was partially selected.
371 if (action == Extract || action == Delete) {
372 if (partialStart && commonRoot->contains(partialStart.get())) {
373 auto result = setStart(*partialStart->parentNode(), partialStart->computeNodeIndex() + 1);
374 if (result.hasException())
375 return result.releaseException();
376 } else if (partialEnd && commonRoot->contains(partialEnd.get())) {
377 auto result = setStart(*partialEnd->parentNode(), partialEnd->computeNodeIndex());
378 if (result.hasException())
379 return result.releaseException();
380 }
381 collapse(true);
382 }
383
384 // Now add leftContents, stuff in between, and rightContents to the fragment
385 // (or just delete the stuff in between)
386
387 if ((action == Extract || action == Clone) && leftContents) {
388 auto result = fragment->appendChild(*leftContents);
389 if (result.hasException())
390 return result.releaseException();
391 }
392
393 if (processStart) {
394 Vector<Ref<Node>> nodes;
395 for (Node* node = processStart.get(); node && node != processEnd; node = node->nextSibling())
396 nodes.append(*node);
397 auto result = processNodes(action, nodes, commonRoot.get(), fragment.get());
398 if (result.hasException())
399 return result.releaseException();
400 }
401
402 if ((action == Extract || action == Clone) && rightContents) {
403 auto result = fragment->appendChild(*rightContents);
404 if (result.hasException())
405 return result.releaseException();
406 }
407
408 return fragment;
409}

411static inline ExceptionOr<void> deleteCharacterData(CharacterData& data, unsigned startOffset, unsigned endOffset)
412{
413 if (data.length() - endOffset) {
414 auto result = data.deleteData(endOffset, data.length() - endOffset);
415 if (result.hasException())
416 return result.releaseException();
417 }
418 if (startOffset) {
419 auto result = data.deleteData(0, startOffset);
420 if (result.hasException())
421 return result.releaseException();
422 }
423 return { };
424}

426static ExceptionOr<RefPtr<Node>> processContentsBetweenOffsets(Range::ActionType action, RefPtr<DocumentFragment> fragment, RefPtr<Node> container, unsigned startOffset, unsigned endOffset)
427{
428 ASSERT(container);
429 ASSERT(startOffset <= endOffset);
430
431 RefPtr<Node> result;
432
433 switch (container->nodeType()) {
434 case Node::TEXT_NODE:
435 case Node::CDATA_SECTION_NODE:
436 case Node::COMMENT_NODE:
437 endOffset = std::min(endOffset, downcast<CharacterData>(*container).length());
438 startOffset = std::min(startOffset, endOffset);
439 if (action == Range::Extract || action == Range::Clone) {
440 Ref<CharacterData> characters = downcast<CharacterData>(container->cloneNode(true).get());
441 auto deleteResult = deleteCharacterData(characters, startOffset, endOffset);
442 if (deleteResult.hasException())
443 return deleteResult.releaseException();
444 if (fragment) {
445 result = fragment;
446 auto appendResult = result->appendChild(characters);
447 if (appendResult.hasException())
448 return appendResult.releaseException();
449 } else
450 result = WTFMove(characters);
451 }
452 if (action == Range::Extract || action == Range::Delete) {
453 auto deleteResult = downcast<CharacterData>(*container).deleteData(startOffset, endOffset - startOffset);
454 if (deleteResult.hasException())
455 return deleteResult.releaseException();
456 }
457 break;
458 case Node::PROCESSING_INSTRUCTION_NODE: {
459 auto& instruction = downcast<ProcessingInstruction>(*container);
460 endOffset = std::min(endOffset, downcast<ProcessingInstruction>(*container).data().length());
461 startOffset = std::min(startOffset, endOffset);
462 if (action == Range::Extract || action == Range::Clone) {
463 Ref<ProcessingInstruction> processingInstruction = downcast<ProcessingInstruction>(container->cloneNode(true).get());
464 processingInstruction->setData(processingInstruction->data().substring(startOffset, endOffset - startOffset));
465 if (fragment) {
466 result = fragment;
467 auto appendResult = result->appendChild(processingInstruction);
468 if (appendResult.hasException())
469 return appendResult.releaseException();
470 } else
471 result = WTFMove(processingInstruction);
472 }
473 if (action == Range::Extract || action == Range::Delete) {
474 String data { instruction.data() };
475 data.remove(startOffset, endOffset - startOffset);
476 instruction.setData(data);
477 }
478 break;
479 }
480 case Node::ELEMENT_NODE:
481 case Node::ATTRIBUTE_NODE:
482 case Node::DOCUMENT_NODE:
483 case Node::DOCUMENT_TYPE_NODE:
484 case Node::DOCUMENT_FRAGMENT_NODE:
485 // FIXME: Should we assert that some nodes never appear here?
486 if (action == Range::Extract || action == Range::Clone) {
487 if (fragment)
488 result = fragment;
489 else
490 result = container->cloneNode(false);
491 }
492 Vector<Ref<Node>> nodes;
493 Node* n = container->firstChild();
494 for (unsigned i = startOffset; n && i; i--)
495 n = n->nextSibling();
496 for (unsigned i = startOffset; n && i < endOffset; i++, n = n->nextSibling()) {
497 if (action != Range::Delete && n->isDocumentTypeNode()) {
498 return Exception { HierarchyRequestError };
499 }
500 nodes.append(*n);
501 }
502 auto processResult = processNodes(action, nodes, container.get(), result);
503 if (processResult.hasException())
504 return processResult.releaseException();
505 break;
506 }
507
508 return result;
509}

511static ExceptionOr<void> processNodes(Range::ActionType action, Vector<Ref<Node>>& nodes, Node* oldContainer, RefPtr<Node> newContainer)
512{
513 for (auto& node : nodes) {
514 switch (action) {
515 case Range::Delete: {
516 auto result = oldContainer->removeChild(node);
517 if (result.hasException())
518 return result.releaseException();
519 break;
520 }
521 case Range::Extract: {
522 auto result = newContainer->appendChild(node); // will remove node from its parent
523 if (result.hasException())
524 return result.releaseException();
525 break;
526 }
527 case Range::Clone: {
528 auto result = newContainer->appendChild(node->cloneNode(true));
529 if (result.hasException())
530 return result.releaseException();
531 break;
532 }
533 }
534 }
535 return { };
536}

538ExceptionOr<RefPtr<Node>> processAncestorsAndTheirSiblings(Range::ActionType action, Node* container, ContentsProcessDirection direction, ExceptionOr<RefPtr<Node>>&& passedClonedContainer, Node* commonRoot)
539{
540 if (passedClonedContainer.hasException())
541 return WTFMove(passedClonedContainer);
542
543 RefPtr<Node> clonedContainer = passedClonedContainer.releaseReturnValue();
544
545 Vector<Ref<ContainerNode>> ancestors;
546 for (ContainerNode* ancestor = container->parentNode(); ancestor && ancestor != commonRoot; ancestor = ancestor->parentNode())
547 ancestors.append(*ancestor);
548
549 RefPtr<Node> firstChildInAncestorToProcess = direction == ProcessContentsForward ? container->nextSibling() : container->previousSibling();
550 for (auto& ancestor : ancestors) {
551 if (action == Range::Extract || action == Range::Clone) {
552 auto clonedAncestor = ancestor->cloneNode(false); // Might have been removed already during mutation event.
553 if (clonedContainer) {
554 auto result = clonedAncestor->appendChild(*clonedContainer);
555 if (result.hasException())
556 return result.releaseException();
557 }
558 clonedContainer = WTFMove(clonedAncestor);
559 }
560
561 // Copy siblings of an ancestor of start/end containers
562 // FIXME: This assertion may fail if DOM is modified during mutation event
563 // FIXME: Share code with Range::processNodes
564 ASSERT(!firstChildInAncestorToProcess || firstChildInAncestorToProcess->parentNode() == ancestor.ptr());
565
566 Vector<Ref<Node>> nodes;
567 for (Node* child = firstChildInAncestorToProcess.get(); child;
568 child = (direction == ProcessContentsForward) ? child->nextSibling() : child->previousSibling())
569 nodes.append(*child);
570
571 for (auto& child : nodes) {
572 switch (action) {
573 case Range::Delete: {
574 auto result = ancestor->removeChild(child);
575 if (result.hasException())
576 return result.releaseException();
577 break;
578 }
579 case Range::Extract: // will remove child from ancestor
580 if (direction == ProcessContentsForward) {
581 auto result = clonedContainer->appendChild(child);
582 if (result.hasException())
583 return result.releaseException();
584 } else {
585 auto result = clonedContainer->insertBefore(child, clonedContainer->firstChild());
586 if (result.hasException())
587 return result.releaseException();
588 }
589 break;
590 case Range::Clone:
591 if (direction == ProcessContentsForward) {
592 auto result = clonedContainer->appendChild(child->cloneNode(true));
593 if (result.hasException())
594 return result.releaseException();
595 } else {
596 auto result = clonedContainer->insertBefore(child->cloneNode(true), clonedContainer->firstChild());
597 if (result.hasException())
598 return result.releaseException();
599 }
600 break;
601 }
602 }
603 firstChildInAncestorToProcess = direction == ProcessContentsForward ? ancestor->nextSibling() : ancestor->previousSibling();
604 }
605
606 return clonedContainer;
607}

625ExceptionOr<void> Range::insertNode(Ref<Node>&& node)
626{
627 auto startContainerNodeType = startContainer().nodeType();
628
629 if (startContainerNodeType == Node::COMMENT_NODE || startContainerNodeType == Node::PROCESSING_INSTRUCTION_NODE)
630 return Exception { HierarchyRequestError };
631 bool startIsText = startContainerNodeType == Node::TEXT_NODE;
632 if (startIsText && !startContainer().parentNode())
633 return Exception { HierarchyRequestError };
634 if (node.ptr() == &startContainer())
635 return Exception { HierarchyRequestError };
636
637 RefPtr<Node> referenceNode = startIsText ? &startContainer() : startContainer().traverseToChildAt(startOffset());
638 Node* parentNode = referenceNode ? referenceNode->parentNode() : &startContainer();
639 if (!is<ContainerNode>(parentNode))
640 return Exception { HierarchyRequestError };
641
642 Ref<ContainerNode> parent = downcast<ContainerNode>(*parentNode);
643
644 auto result = parent->ensurePreInsertionValidity(node, referenceNode.get());
645 if (result.hasException())
646 return result.releaseException();
647
648 EventQueueScope scope;
649 if (startIsText) {
650 auto result = downcast<Text>(startContainer()).splitText(startOffset());
651 if (result.hasException())
652 return result.releaseException();
653 referenceNode = result.releaseReturnValue();
654 }
655
656 if (referenceNode == node.ptr())
657 referenceNode = referenceNode->nextSibling();
658
659 auto removeResult = node->remove();
660 if (removeResult.hasException())
661 return removeResult.releaseException();
662
663 unsigned newOffset = referenceNode ? referenceNode->computeNodeIndex() : parent->countChildNodes();
664 if (is<DocumentFragment>(node))
665 newOffset += downcast<DocumentFragment>(node.get()).countChildNodes();
666 else
667 ++newOffset;
668
669 auto insertResult = parent->insertBefore(node, referenceNode.get());
670 if (insertResult.hasException())
671 return insertResult.releaseException();
672
673 if (collapsed())
674 return setEnd(WTFMove(parent), newOffset);
675
676 return { };
677}

679String Range::toString() const
680{
681 auto range = makeSimpleRange(*this);
682 StringBuilder builder;
683 for (auto& node : intersectingNodes(range)) {
684 if (is<Text>(node)) {
685 auto offsetRange = characterDataOffsetRange(range, node);
686 builder.appendSubstring(downcast<Text>(node).data(), offsetRange.start, offsetRange.end - offsetRange.start);
687 }
688 }
689 return builder.toString();
690}

692// https://w3c.github.io/DOM-Parsing/#widl-Range-createContextualFragment-DocumentFragment-DOMString-fragment
693ExceptionOr<Ref<DocumentFragment>> Range::createContextualFragment(const String& markup)
694{
695 Node& node = startContainer();
696 RefPtr<Element> element;
697 if (is<Document>(node) || is<DocumentFragment>(node))
698 element = nullptr;
699 else if (is<Element>(node))
700 element = &downcast<Element>(node);
701 else
702 element = node.parentElement();
703 if (!element || (element->document().isHTMLDocument() && is<HTMLHtmlElement>(*element)))
704 element = HTMLBodyElement::create(node.document());
705 return WebCore::createContextualFragment(*element, markup, AllowScriptingContentAndDoNotMarkAlreadyStarted);

708ExceptionOr<Node*> Range::checkNodeOffsetPair(Node& node, unsigned offset)
709{
710 switch (node.nodeType()) {
711 case Node::DOCUMENT_TYPE_NODE:
712 return Exception { InvalidNodeTypeError };
713 case Node::CDATA_SECTION_NODE:
714 case Node::COMMENT_NODE:
715 case Node::TEXT_NODE:
716 case Node::PROCESSING_INSTRUCTION_NODE:
717 if (offset > downcast<CharacterData>(node).length())
718 return Exception { IndexSizeError };
719 return nullptr;
720 case Node::ATTRIBUTE_NODE:
721 case Node::DOCUMENT_FRAGMENT_NODE:
722 case Node::DOCUMENT_NODE:
723 case Node::ELEMENT_NODE:
724 if (!offset)
725 return nullptr;
726 auto childBefore = node.traverseToChildAt(offset - 1);
727 if (!childBefore)
728 return Exception { IndexSizeError };
729 return childBefore;
730 }
731 ASSERT_NOT_REACHED();
732 return Exception { InvalidNodeTypeError };

790// https://dom.spec.whatwg.org/#dom-range-surroundcontents
791ExceptionOr<void> Range::surroundContents(Node& newParent)
792{
793 Ref<Node> protectedNewParent(newParent);
794
795 // Step 1: If a non-Text node is partially contained in the context object, then throw an InvalidStateError.
796 Node* startNonTextContainer = &startContainer();
797 if (startNonTextContainer->nodeType() == Node::TEXT_NODE)
798 startNonTextContainer = startNonTextContainer->parentNode();
799 Node* endNonTextContainer = &endContainer();
800 if (endNonTextContainer->nodeType() == Node::TEXT_NODE)
801 endNonTextContainer = endNonTextContainer->parentNode();
802 if (startNonTextContainer != endNonTextContainer)
803 return Exception { InvalidStateError };
804
805 // Step 2: If newParent is a Document, DocumentType, or DocumentFragment node, then throw an InvalidNodeTypeError.
806 switch (newParent.nodeType()) {
807 case Node::ATTRIBUTE_NODE:
808 case Node::DOCUMENT_FRAGMENT_NODE:
809 case Node::DOCUMENT_NODE:
810 case Node::DOCUMENT_TYPE_NODE:
811 return Exception { InvalidNodeTypeError };
812 case Node::CDATA_SECTION_NODE:
813 case Node::COMMENT_NODE:
814 case Node::ELEMENT_NODE:
815 case Node::PROCESSING_INSTRUCTION_NODE:
816 case Node::TEXT_NODE:
817 break;
818 }
819
820 // Step 3: Let fragment be the result of extracting context object.
821 auto fragment = extractContents();
822 if (fragment.hasException())
823 return fragment.releaseException();
824
825 // Step 4: If newParent has children, replace all with null within newParent.
826 if (newParent.hasChildNodes())
827 downcast<ContainerNode>(newParent).replaceAll(nullptr);
828
829 // Step 5: Insert newParent into context object.
830 auto insertResult = insertNode(newParent);
831 if (insertResult.hasException())
832 return insertResult.releaseException();
833
834 // Step 6: Append fragment to newParent.
835 auto appendResult = newParent.appendChild(fragment.releaseReturnValue());
836 if (appendResult.hasException())
837 return appendResult.releaseException();
838
839 // Step 7: Select newParent within context object.
840 return selectNode(newParent);
841}

970static inline void boundaryTextNodesSplit(RangeBoundaryPoint& boundary, Text& oldNode)
971{
972 auto* parent = oldNode.parentNode();
973 if (&boundary.container() == &oldNode) {
974 unsigned splitOffset = oldNode.length();
975 unsigned boundaryOffset = boundary.offset();
976 if (boundaryOffset > splitOffset) {
977 if (parent)
978 boundary.set(*oldNode.nextSibling(), boundaryOffset - splitOffset, 0);
979 else
980 boundary.setOffset(splitOffset);
981 }
982 return;
983 }
984 if (!parent)
985 return;
986 if (&boundary.container() == parent && boundary.childBefore() == &oldNode) {
987 auto* newChild = oldNode.nextSibling();
988 ASSERT(newChild);
989 boundary.setToAfterNode(*newChild);
990 }
991}

1002ExceptionOr<void> Range::expand(const String& unit)
1003{
1004 auto start = VisiblePosition { makeContainerOffsetPosition(&startContainer(), startOffset()) };
1005 auto end = VisiblePosition { makeContainerOffsetPosition(&endContainer(), endOffset()) };
1006 if (unit == "word") {
1007 start = startOfWord(start);
1008 end = endOfWord(end);
1009 } else if (unit == "sentence") {
1010 start = startOfSentence(start);
1011 end = endOfSentence(end);
1012 } else if (unit == "block") {
1013 start = startOfParagraph(start);
1014 end = endOfParagraph(end);
1015 } else if (unit == "document") {
1016 start = startOfDocument(start);
1017 end = endOfDocument(end);
1018 } else
1019 return { };
1020
1021 auto* startContainer = start.deepEquivalent().containerNode();
1022 if (!startContainer)
1023 return Exception { TypeError };
1024 auto result = setStart(*startContainer, start.deepEquivalent().computeOffsetInContainerNode());
1025 if (result.hasException())
1026 return result.releaseException();
1027 auto* endContainer = end.deepEquivalent().containerNode();
1028 if (!endContainer)
1029 return Exception { TypeError };
1030 return setEnd(*endContainer, end.deepEquivalent().computeOffsetInContainerNode());
1031}
