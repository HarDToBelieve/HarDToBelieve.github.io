28#include "BoundaryPoint.h"

31namespace WebCore {
32
33class RangeBoundaryPoint {
34public:
35 explicit RangeBoundaryPoint(Node& container);
36
37 Node& container() const;
38 unsigned offset() const;
39 Node* childBefore() const;
40
41 void set(Ref<Node>&& container, unsigned offset, Node* childBefore);
42 void setOffset(unsigned);
43
44 void setToBeforeNode(Node&);
45 void setToAfterNode(Node&);
46 void setToBeforeContents(Ref<Node>&&);
47 void setToAfterContents(Ref<Node>&&);
48
49 void childBeforeWillBeRemoved();
50 void invalidateOffset();
51
52private:
53 Ref<Node> m_container;
54 unsigned m_offset { 0 };
55 RefPtr<Node> m_childBefore;
56};
57
58BoundaryPoint makeBoundaryPoint(const RangeBoundaryPoint&);
59
60inline RangeBoundaryPoint::RangeBoundaryPoint(Node& container)
61 : m_container(container)
62{
63}
64
65inline Node& RangeBoundaryPoint::container() const
66{
67 return m_container;
68}
69
70inline Node* RangeBoundaryPoint::childBefore() const
71{
72 return m_childBefore.get();
73}
74
75inline unsigned RangeBoundaryPoint::offset() const
76{
77 return m_offset;
78}
79
80inline void RangeBoundaryPoint::set(Ref<Node>&& container, unsigned offset, Node* childBefore)
81{
82 ASSERT(childBefore == (offset ? container->traverseToChildAt(offset - 1) : nullptr));
83 m_container = WTFMove(container);
84 m_offset = offset;
85 m_childBefore = childBefore;
86}
87
88inline void RangeBoundaryPoint::setOffset(unsigned offset)
89{
90 ASSERT(m_container->isCharacterDataNode());
91 ASSERT(m_offset);
92 ASSERT(!m_childBefore);
93 m_offset = offset;
94}
95
96inline void RangeBoundaryPoint::setToBeforeNode(Node& child)
97{
98 ASSERT(child.parentNode());
99 m_container = *child.parentNode();
100 m_offset = child.computeNodeIndex();
101 m_childBefore = child.previousSibling();
102}
103
104inline void RangeBoundaryPoint::setToAfterNode(Node& child)
105{
106 ASSERT(child.parentNode());
107 m_container = *child.parentNode();
108 m_offset = child.computeNodeIndex() + 1;
109 m_childBefore = &child;
110}
111
112inline void RangeBoundaryPoint::setToBeforeContents(Ref<Node>&& container)
113{
114 m_container = WTFMove(container);
115 m_offset = 0;
116 m_childBefore = nullptr;
117}
118
119inline void RangeBoundaryPoint::setToAfterContents(Ref<Node>&& container)
120{
121 m_container = WTFMove(container);
122 m_offset = m_container->length();
123 m_childBefore = m_container->lastChild();
124}
125
126inline void RangeBoundaryPoint::childBeforeWillBeRemoved()
127{
128 ASSERT(m_offset);
129 ASSERT(m_childBefore);
130 --m_offset;
131 m_childBefore = m_childBefore->previousSibling();
132}
133
134inline void RangeBoundaryPoint::invalidateOffset()
135{
136 m_offset = m_childBefore ? m_childBefore->computeNodeIndex() + 1 : 0;
137}
138
139inline bool operator==(const RangeBoundaryPoint& a, const RangeBoundaryPoint& b)
140{
141 return &a.container() == &b.container() && a.offset() == b.offset();
142}
143
144inline BoundaryPoint makeBoundaryPoint(const RangeBoundaryPoint& point)
145{
146 return { point.container(), point.offset() };
147}
