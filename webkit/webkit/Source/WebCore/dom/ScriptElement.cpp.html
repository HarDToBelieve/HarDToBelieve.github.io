110static bool isLegacySupportedJavaScriptLanguage(const String& language)
111{
112 static constexpr ComparableLettersLiteral languageArray[] = {
113 "ecmascript",
114 "javascript",
115 "javascript1.0",
116 "javascript1.1",
117 "javascript1.2",
118 "javascript1.3",
119 "javascript1.4",
120 "javascript1.5",
121 "javascript1.6",
122 "javascript1.7",
123 "jscript",
124 "livescript",
125 };
126 static constexpr SortedArraySet languageSet { languageArray };
127 return languageSet.contains(language);
128}

135std::optional<ScriptElement::ScriptType> ScriptElement::determineScriptType(LegacyTypeSupport supportLegacyTypes) const
136{
137 // FIXME: isLegacySupportedJavaScriptLanguage() is not valid HTML5. It is used here to maintain backwards compatibility with existing layout tests. The specific violations are:
138 // - Allowing type=javascript. type= should only support MIME types, such as text/javascript.
139 // - Allowing a different set of languages for language= and type=. language= supports Javascript 1.1 and 1.4-1.6, but type= does not.
140 String type = typeAttributeValue();
141 String language = languageAttributeValue();
142 if (type.isEmpty()) {
143 if (language.isEmpty())
144 return ScriptType::Classic; // Assume text/javascript.
145 if (MIMETypeRegistry::isSupportedJavaScriptMIMEType("text/" + language))
146 return ScriptType::Classic;
147 if (isLegacySupportedJavaScriptLanguage(language))
148 return ScriptType::Classic;
149 return std::nullopt;
150 }
151 if (MIMETypeRegistry::isSupportedJavaScriptMIMEType(type.stripWhiteSpace()))
152 return ScriptType::Classic;
153 if (supportLegacyTypes == AllowLegacyTypeInTypeAttribute && isLegacySupportedJavaScriptLanguage(type))
154 return ScriptType::Classic;
155
156 // FIXME: XHTML spec defines "defer" attribute. But WebKit does not implement it for a long time.
157 // And module tag also uses defer attribute semantics. We disable script type="module" for non HTML document.
158 // Once "defer" is implemented, we can reconsider enabling modules in XHTML.
159 // https://bugs.webkit.org/show_bug.cgi?id=123387
160 if (!m_element.document().isHTMLDocument())
161 return std::nullopt;
162
163 // https://html.spec.whatwg.org/multipage/scripting.html#attr-script-type
164 // Setting the attribute to an ASCII case-insensitive match for the string "module" means that the script is a module script.
165 if (equalLettersIgnoringASCIICase(type, "module"))
166 return ScriptType::Module;
167 return std::nullopt;
168}

170// http://dev.w3.org/html5/spec/Overview.html#prepare-a-script
171bool ScriptElement::prepareScript(const TextPosition& scriptStartPosition, LegacyTypeSupport supportLegacyTypes)
172{
173 if (m_alreadyStarted)
174 return false;
175
176 bool wasParserInserted;
177 if (m_parserInserted) {
178 wasParserInserted = true;
179 m_parserInserted = false;
180 } else
181 wasParserInserted = false;
182
183 if (wasParserInserted && !hasAsyncAttribute())
184 m_forceAsync = true;
185
186 auto sourceText = scriptContent();
187 if (!hasSourceAttribute() && sourceText.isEmpty())
188 return false;
189
190 if (!m_element.isConnected())
191 return false;
192
193 ScriptType scriptType = ScriptType::Classic;
194 if (std::optional<ScriptType> result = determineScriptType(supportLegacyTypes))
195 scriptType = result.value();
196 else
197 return false;
198 m_isModuleScript = scriptType == ScriptType::Module;
199
200 if (wasParserInserted) {
201 m_parserInserted = true;
202 m_forceAsync = false;
203 }
204
205 m_alreadyStarted = true;
206
207 // FIXME: If script is parser inserted, verify it's still in the original document.
208 Document& document = m_element.document();
209
210 // FIXME: Eventually we'd like to evaluate scripts which are inserted into a
211 // viewless document but this'll do for now.
212 // See http://bugs.webkit.org/show_bug.cgi?id=5727
213 if (!document.frame())
214 return false;
215
216 if (scriptType == ScriptType::Classic && hasNoModuleAttribute())
217 return false;
218
219 if (!document.frame()->script().canExecuteScripts(AboutToExecuteScript))
220 return false;
221
222 if (scriptType == ScriptType::Classic && !isScriptForEventSupported())
223 return false;
224
225 // According to the spec, the module tag ignores the "charset" attribute as the same to the worker's
226 // importScript. But WebKit supports the "charset" for importScript intentionally. So to be consistent,
227 // even for the module tags, we handle the "charset" attribute.
228 if (!charsetAttributeValue().isEmpty())
229 m_characterEncoding = charsetAttributeValue();
230 else
231 m_characterEncoding = document.charset();
232
233 if (scriptType == ScriptType::Classic) {
234 if (hasSourceAttribute()) {
235 if (!requestClassicScript(sourceAttributeValue()))
236 return false;
237 }
238 } else {
239 ASSERT(scriptType == ScriptType::Module);
240 if (!requestModuleScript(scriptStartPosition))
241 return false;
242 }
243
244 // All the inlined module script is handled by requestModuleScript. It produces LoadableModuleScript and inlined module script
245 // is handled as the same to the external module script.
246
247 bool isClassicExternalScript = scriptType == ScriptType::Classic && hasSourceAttribute();
248 bool isParserInsertedDeferredScript = ((isClassicExternalScript && hasDeferAttribute()) || scriptType == ScriptType::Module)
249 && m_parserInserted && !hasAsyncAttribute();
250 if (isParserInsertedDeferredScript) {
251 m_willExecuteWhenDocumentFinishedParsing = true;
252 m_willBeParserExecuted = true;
253 } else if (isClassicExternalScript && m_parserInserted && !hasAsyncAttribute()) {
254 ASSERT(scriptType == ScriptType::Classic);
255 m_willBeParserExecuted = true;
256 } else if ((isClassicExternalScript || scriptType == ScriptType::Module) && !hasAsyncAttribute() && !m_forceAsync) {
257 m_willExecuteInOrder = true;
258 ASSERT(m_loadableScript);
259 document.scriptRunner().queueScriptForExecution(*this, *m_loadableScript, ScriptRunner::IN_ORDER_EXECUTION);
260 } else if (hasSourceAttribute() || scriptType == ScriptType::Module) {
261 ASSERT(m_loadableScript);
262 ASSERT(hasAsyncAttribute() || m_forceAsync);
263 document.scriptRunner().queueScriptForExecution(*this, *m_loadableScript, ScriptRunner::ASYNC_EXECUTION);
264 } else if (!hasSourceAttribute() && m_parserInserted && !document.haveStylesheetsLoaded()) {
265 ASSERT(scriptType == ScriptType::Classic);
266 m_willBeParserExecuted = true;
267 m_readyToBeParserExecuted = true;
268 } else {
269 ASSERT(scriptType == ScriptType::Classic);
270 TextPosition position = document.isInDocumentWrite() ? TextPosition() : scriptStartPosition;
271 executeClassicScript(ScriptSourceCode(sourceText, URL(document.url()), position, JSC::SourceProviderSourceType::Program, InlineClassicScript::create(*this)));
272 }
273
274 return true;
275}

277bool ScriptElement::requestClassicScript(const String& sourceURL)
278{
279 Ref<Document> originalDocument(m_element.document());
280 if (!m_element.dispatchBeforeLoadEvent(sourceURL))
281 return false;
282 bool didEventListenerDisconnectThisElement = !m_element.isConnected() || &m_element.document() != originalDocument.ptr();
283 if (didEventListenerDisconnectThisElement)
284 return false;
285
286 ASSERT(!m_loadableScript);
287 if (!stripLeadingAndTrailingHTMLSpaces(sourceURL).isEmpty()) {
288 auto script = LoadableClassicScript::create(
289 m_element.attributeWithoutSynchronization(HTMLNames::nonceAttr),
290 m_element.document().settings().subresourceIntegrityEnabled() ? m_element.attributeWithoutSynchronization(HTMLNames::integrityAttr).string() : emptyString(),
291 referrerPolicy(),
292 m_element.attributeWithoutSynchronization(HTMLNames::crossoriginAttr),
293 scriptCharset(),
294 m_element.localName(),
295 m_element.isInUserAgentShadowTree(),
296 hasAsyncAttribute());
297
298 auto scriptURL = m_element.document().completeURL(sourceURL);
299 m_element.document().willLoadScriptElement(scriptURL);
300
301 if (script->load(m_element.document(), scriptURL)) {
302 m_loadableScript = WTFMove(script);
303 m_isExternalScript = true;
304 }
305 }
306
307 if (m_loadableScript)
308 return true;
309
310 callOnMainThread([this, element = Ref<Element>(m_element)] {
311 dispatchErrorEvent();
312 });
313 return false;
314}

316bool ScriptElement::requestModuleScript(const TextPosition& scriptStartPosition)
317{
318 // https://html.spec.whatwg.org/multipage/urls-and-fetching.html#cors-settings-attributes
319 // Module is always CORS request. If attribute is not given, it should be same-origin credential.
320 String nonce = m_element.attributeWithoutSynchronization(HTMLNames::nonceAttr);
321 String crossOriginMode = m_element.attributeWithoutSynchronization(HTMLNames::crossoriginAttr);
322 if (crossOriginMode.isNull())
323 crossOriginMode = ScriptElementCachedScriptFetcher::defaultCrossOriginModeForModule;
324
325 if (hasSourceAttribute()) {
326 String sourceURL = sourceAttributeValue();
327 Ref<Document> originalDocument(m_element.document());
328 if (!m_element.dispatchBeforeLoadEvent(sourceURL))
329 return false;
330
331 bool didEventListenerDisconnectThisElement = !m_element.isConnected() || &m_element.document() != originalDocument.ptr();
332 if (didEventListenerDisconnectThisElement)
333 return false;
334
335 if (stripLeadingAndTrailingHTMLSpaces(sourceURL).isEmpty()) {
336 dispatchErrorEvent();
337 return false;
338 }
339
340 auto moduleScriptRootURL = m_element.document().completeURL(sourceURL);
341 if (!moduleScriptRootURL.isValid()) {
342 dispatchErrorEvent();
343 return false;
344 }
345
346 m_isExternalScript = true;
347 auto script = LoadableModuleScript::create(
348 nonce,
349 m_element.document().settings().subresourceIntegrityEnabled() ? m_element.attributeWithoutSynchronization(HTMLNames::integrityAttr).string() : emptyString(),
350 referrerPolicy(),
351 crossOriginMode,
352 scriptCharset(),
353 m_element.localName(),
354 m_element.isInUserAgentShadowTree());
355 m_loadableScript = WTFMove(script);
356 if (auto* frame = m_element.document().frame()) {
357 auto& script = downcast<LoadableModuleScript>(*m_loadableScript.get());
358 frame->script().loadModuleScript(script, moduleScriptRootURL.string(), makeRef(script.parameters()));
359 }
360 return true;
361 }
362
363 auto script = LoadableModuleScript::create(nonce, emptyString(), referrerPolicy(), crossOriginMode, scriptCharset(), m_element.localName(), m_element.isInUserAgentShadowTree());
364
365 TextPosition position = m_element.document().isInDocumentWrite() ? TextPosition() : scriptStartPosition;
366 ScriptSourceCode sourceCode(scriptContent(), URL(m_element.document().url()), position, JSC::SourceProviderSourceType::Module, script.copyRef());
367
368 ASSERT(m_element.document().contentSecurityPolicy());
369 const auto& contentSecurityPolicy = *m_element.document().contentSecurityPolicy();
370 bool hasKnownNonce = contentSecurityPolicy.allowScriptWithNonce(nonce, m_element.isInUserAgentShadowTree());
371 if (!contentSecurityPolicy.allowInlineScript(m_element.document().url().string(), m_startLineNumber, sourceCode.source(), hasKnownNonce))
372 return false;
373
374 m_loadableScript = WTFMove(script);
375 if (auto* frame = m_element.document().frame())
376 frame->script().loadModuleScript(downcast<LoadableModuleScript>(*m_loadableScript.get()), sourceCode);
377 return true;
378}

380void ScriptElement::executeClassicScript(const ScriptSourceCode& sourceCode)
381{
382 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::InMainThread::isScriptAllowed() || !isInWebProcess());
383 ASSERT(m_alreadyStarted);
384
385 if (sourceCode.isEmpty())
386 return;
387
388 if (!m_isExternalScript) {
389 ASSERT(m_element.document().contentSecurityPolicy());
390 const ContentSecurityPolicy& contentSecurityPolicy = *m_element.document().contentSecurityPolicy();
391 bool hasKnownNonce = contentSecurityPolicy.allowScriptWithNonce(m_element.attributeWithoutSynchronization(HTMLNames::nonceAttr), m_element.isInUserAgentShadowTree());
392 if (!contentSecurityPolicy.allowInlineScript(m_element.document().url().string(), m_startLineNumber, sourceCode.source(), hasKnownNonce))
393 return;
394 }
395
396 auto& document = m_element.document();
397 auto* frame = document.frame();
398 if (!frame)
399 return;
400
401 IgnoreDestructiveWriteCountIncrementer ignoreDestructiveWriteCountIncrementer(m_isExternalScript ? &document : nullptr);
402 CurrentScriptIncrementer currentScriptIncrementer(document, *this);
403
404 WTFBeginSignpost(this, "Execute Script Element", "executing classic script from URL: %{public}s async: %d defer: %d", m_isExternalScript ? sourceCode.url().string().utf8().data() : "inline", hasAsyncAttribute(), hasDeferAttribute());
405 frame->script().evaluateIgnoringException(sourceCode);
406 WTFEndSignpost(this, "Execute Script Element");
407}

409void ScriptElement::executeModuleScript(LoadableModuleScript& loadableModuleScript)
410{
411 // https://html.spec.whatwg.org/multipage/scripting.html#execute-the-script-block
412
413 ASSERT(!loadableModuleScript.error());
414
415 auto& document = m_element.document();
416 auto* frame = document.frame();
417 if (!frame)
418 return;
419
420 IgnoreDestructiveWriteCountIncrementer ignoreDestructiveWriteCountIncrementer(&document);
421 CurrentScriptIncrementer currentScriptIncrementer(document, *this);
422
423 WTFBeginSignpost(this, "Execute Script Element", "executing module script");
424 frame->script().linkAndEvaluateModuleScript(loadableModuleScript);
425 WTFEndSignpost(this, "Execute Script Element", "executing module script");
426}

428void ScriptElement::dispatchLoadEventRespectingUserGestureIndicator()
429{
430 if (MonotonicTime::now() - m_creationTime > maxUserGesturePropagationTime) {
431 dispatchLoadEvent();
432 return;
433 }
434
435 UserGestureIndicator indicator(m_userGestureToken);
436 dispatchLoadEvent();
437}

439void ScriptElement::executeScriptAndDispatchEvent(LoadableScript& loadableScript)
440{
441 if (std::optional<LoadableScript::Error> error = loadableScript.error()) {
442 if (std::optional<LoadableScript::ConsoleMessage> message = error->consoleMessage)
443 m_element.document().addConsoleMessage(message->source, message->level, message->message);
444 dispatchErrorEvent();
445 } else if (!loadableScript.wasCanceled()) {
446 ASSERT(!loadableScript.error());
447 loadableScript.execute(*this);
448 dispatchLoadEventRespectingUserGestureIndicator();
449 }
450}

452void ScriptElement::executePendingScript(PendingScript& pendingScript)
453{
454 if (auto* loadableScript = pendingScript.loadableScript())
455 executeScriptAndDispatchEvent(*loadableScript);
456 else {
457 ASSERT(!pendingScript.error());
458 ASSERT_WITH_MESSAGE(scriptType() == ScriptType::Classic, "Module script always have a loadableScript pointer.");
459 executeClassicScript(ScriptSourceCode(scriptContent(), URL(m_element.document().url()), pendingScript.startingPosition(), JSC::SourceProviderSourceType::Program, InlineClassicScript::create(*this)));
460 dispatchLoadEventRespectingUserGestureIndicator();
461 }

469bool ScriptElement::isScriptForEventSupported() const
470{
471 String eventAttribute = eventAttributeValue();
472 String forAttribute = forAttributeValue();
473 if (!eventAttribute.isNull() && !forAttribute.isNull()) {
474 forAttribute = stripLeadingAndTrailingHTMLSpaces(forAttribute);
475 if (!equalLettersIgnoringASCIICase(forAttribute, "window"))
476 return false;
477
478 eventAttribute = stripLeadingAndTrailingHTMLSpaces(eventAttribute);
479 if (!equalLettersIgnoringASCIICase(eventAttribute, "onload") && !equalLettersIgnoringASCIICase(eventAttribute, "onload()"))
480 return false;
481 }
482 return true;
483}
