27#pragma once
28
29#include "PendingScriptClient.h"
30#include "Timer.h"
31#include <wtf/HashSet.h>
32#include <wtf/Vector.h>
33
34namespace WebCore {
35
36class Document;
37class ScriptElement;
38class LoadableScript;
39
40class ScriptRunner : private PendingScriptClient {
41 WTF_MAKE_NONCOPYABLE(ScriptRunner); WTF_MAKE_FAST_ALLOCATED;
42public:
43 explicit ScriptRunner(Document&);
44 ~ScriptRunner();
45
46 enum ExecutionType { ASYNC_EXECUTION, IN_ORDER_EXECUTION };
47 void queueScriptForExecution(ScriptElement&, LoadableScript&, ExecutionType);
48 bool hasPendingScripts() const { return !m_scriptsToExecuteSoon.isEmpty() || !m_scriptsToExecuteInOrder.isEmpty() || !m_pendingAsyncScripts.isEmpty(); }
49 void suspend();
50 void resume();
51 void notifyScriptReady(ScriptElement*, ExecutionType);
52
53 void didBeginYieldingParser() { suspend(); }
54 void didEndYieldingParser() { resume(); }
55
56 void documentFinishedParsing();
57
58private:
59 void timerFired();
60
61 void notifyFinished(PendingScript&) override;
62
63 Document& m_document;
64 Vector<Ref<PendingScript>> m_scriptsToExecuteInOrder;
65 Vector<RefPtr<PendingScript>> m_scriptsToExecuteSoon; // http://www.whatwg.org/specs/web-apps/current-work/#set-of-scripts-that-will-execute-as-soon-as-possible
66 HashSet<Ref<PendingScript>> m_pendingAsyncScripts;
67 Timer m_timer;
68};

70}
