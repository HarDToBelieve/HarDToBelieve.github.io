87ShadowRoot::~ShadowRoot()
88{
89 if (isConnected())
90 document().didRemoveInDocumentShadowRoot(*this);
91
92 if (m_styleSheetList)
93 m_styleSheetList->detach();
94
95 // We cannot let ContainerNode destructor call willBeDeletedFrom()
96 // for this ShadowRoot instance because TreeScope destructor
97 // clears Node::m_treeScope thus ContainerNode is no longer able
98 // to access it Document reference after that.
99 willBeDeletedFrom(document());
100
101 ASSERT(!m_hasBegunDeletingDetachedChildren);
102 m_hasBegunDeletingDetachedChildren = true;
103
104 // We must remove all of our children first before the TreeScope destructor
105 // runs so we don't go through Node::setTreeScopeRecursively for each child with a
106 // destructed tree scope in each descendant.
107 removeDetachedChildren();
108}

125void ShadowRoot::childrenChanged(const ChildChange& childChange)
126{
127 DocumentFragment::childrenChanged(childChange);
128
129 if (!m_host || m_type == ShadowRootMode::UserAgent)
130 return; // Don't support first-child, nth-of-type, etc... in UA shadow roots as an optimization.
131
132 // FIXME: Avoid always invalidating style just for first-child, etc... as done in Element::childrenChanged.
133 switch (childChange.type) {
134 case ChildChange::Type::ElementInserted:
135 case ChildChange::Type::ElementRemoved:
136 m_host->invalidateStyleForSubtreeInternal();
137 break;
138 case ChildChange::Type::TextInserted:
139 case ChildChange::Type::TextRemoved:
140 case ChildChange::Type::TextChanged:
141 case ChildChange::Type::AllChildrenRemoved:
142 case ChildChange::Type::NonContentsChildRemoved:
143 case ChildChange::Type::NonContentsChildInserted:
144 case ChildChange::Type::AllChildrenReplaced:
145 break;
146 }
147}

156{

266static std::optional<std::pair<AtomString, AtomString>> parsePartMapping(StringView mappingString)
267{
268 const auto end = mappingString.length();
269
270 auto skipWhitespace = [&](auto position) {
271 while (position < end && isHTMLSpace(mappingString[position]))
272 ++position;
273 return position;
274 };
275
276 auto collectValue = [&](auto position) {
277 while (position < end && (!isHTMLSpace(mappingString[position]) && mappingString[position] != ':'))
278 ++position;
279 return position;
280 };
281
282 size_t begin = 0;
283 begin = skipWhitespace(begin);
284
285 auto firstPartEnd = collectValue(begin);
286 if (firstPartEnd == begin)
287 return { };
288
289 auto firstPart = mappingString.substring(begin, firstPartEnd - begin).toAtomString();
290
291 begin = skipWhitespace(firstPartEnd);
292 if (begin == end)
293 return std::make_pair(firstPart, firstPart);
294
295 if (mappingString[begin] != ':')
296 return { };
297
298 begin = skipWhitespace(begin + 1);
299
300 auto secondPartEnd = collectValue(begin);
301 if (secondPartEnd == begin)
302 return { };
303
304 auto secondPart = mappingString.substring(begin, secondPartEnd - begin).toAtomString();
305
306 begin = skipWhitespace(secondPartEnd);
307 if (begin != end)
308 return { };
309
310 return std::make_pair(firstPart, secondPart);
311}

313static ShadowRoot::PartMappings parsePartMappingsList(StringView mappingsListString)
314{
315 ShadowRoot::PartMappings mappings;
316
317 const auto end = mappingsListString.length();
318
319 size_t begin = 0;
320 while (begin < end) {
321 size_t mappingEnd = begin;
322 while (mappingEnd < end && mappingsListString[mappingEnd] != ',')
323 ++mappingEnd;
324
325 auto result = parsePartMapping(mappingsListString.substring(begin, mappingEnd - begin));
326 if (result)
327 mappings.add(result->first, Vector<AtomString, 1>()).iterator->value.append(result->second);
328
329 if (mappingEnd == end)
330 break;
331
332 begin = mappingEnd + 1;
333 }
334
335 return mappings;
336}

338const ShadowRoot::PartMappings& ShadowRoot::partMappings() const
339{
340 if (!m_partMappings) {
341 auto exportpartsValue = host()->attributeWithoutSynchronization(HTMLNames::exportpartsAttr);
342 m_partMappings = parsePartMappingsList(exportpartsValue);
343 }
344
345 return *m_partMappings;
346}

353Vector<ShadowRoot*> assignedShadowRootsIfSlotted(const Node& node)
354{
355 Vector<ShadowRoot*> result;
356 for (auto* slot = node.assignedSlot(); slot; slot = slot->assignedSlot()) {
357 ASSERT(slot->containingShadowRoot());
358 result.append(slot->containingShadowRoot());
359 }
360 return result;
361}
