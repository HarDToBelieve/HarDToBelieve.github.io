45class ShadowRoot final : public DocumentFragment, public TreeScope {
46 WTF_MAKE_ISO_ALLOCATED(ShadowRoot);
47public:
48
49 enum class DelegatesFocus : uint8_t { Yes, No };
50
51 static Ref<ShadowRoot> create(Document& document, ShadowRootMode type, DelegatesFocus delegatesFocus = DelegatesFocus::No)
52 {
53 return adoptRef(*new ShadowRoot(document, type, delegatesFocus));
54 }
55
56 static Ref<ShadowRoot> create(Document& document, std::unique_ptr<SlotAssignment>&& assignment)
57 {
58 return adoptRef(*new ShadowRoot(document, WTFMove(assignment)));
59 }
60
61 virtual ~ShadowRoot();
62
63 using TreeScope::rootNode;
64
65 WEBCORE_EXPORT Style::Scope& styleScope();

69 void setResetStyleInheritance(bool);
70
71 bool delegatesFocus() const { return m_delegatesFocus; }
72 bool containsFocusedElement() const { return m_containsFocusedElement; }
73 void setContainsFocusedElement(bool flag) { m_containsFocusedElement = flag; }

75 Element* host() const { return m_host.get(); }

78 String innerHTML() const;

83 ShadowRootMode mode() const { return m_type; }

131 bool m_resetStyleInheritance { false };
132 bool m_hasBegunDeletingDetachedChildren { false };
133 bool m_delegatesFocus { false };

135 ShadowRootMode m_type { ShadowRootMode::UserAgent };

140 std::unique_ptr<Style::Scope> m_styleScope;

142 mutable std::optional<PartMappings> m_partMappings;

144
145inline Element* ShadowRoot::activeElement() const
146{
147 return treeScope().focusedElementInScope();
148}
149
150inline ShadowRoot* Node::shadowRoot() const
151{
152 if (!is<Element>(*this))
153 return nullptr;
154 return downcast<Element>(*this).shadowRoot();
155}
156
157inline bool Node::isUserAgentShadowRoot() const
158{
159 return isShadowRoot() && downcast<ShadowRoot>(*this).mode() == ShadowRootMode::UserAgent;
160}
161
162inline ContainerNode* Node::parentOrShadowHostNode() const
163{
164 ASSERT(isMainThreadOrGCThread());
165 if (is<ShadowRoot>(*this))
166 return downcast<ShadowRoot>(*this).host();
167 return parentNode();
168}
169
170inline bool hasShadowRootParent(const Node& node)
171{
172 return node.parentNode() && node.parentNode()->isShadowRoot();
173}
