38static const AtomString& slotNameFromAttributeValue(const AtomString& value)
39{
40 return value == nullAtom() ? SlotAssignment::defaultSlotName() : value;
41}

43static const AtomString& slotNameFromSlotAttribute(const Node& child)
44{
45 if (is<Text>(child))
46 return SlotAssignment::defaultSlotName();
47
48 return slotNameFromAttributeValue(downcast<Element>(child).attributeWithoutSynchronization(slotAttr));
49}

51#if ASSERT_ENABLED
52static HTMLSlotElement* findSlotElement(ShadowRoot& shadowRoot, const AtomString& slotName)
53{
54 for (auto& slotElement : descendantsOfType<HTMLSlotElement>(shadowRoot)) {
55 if (slotNameFromAttributeValue(slotElement.attributeWithoutSynchronization(nameAttr)) == slotName)
56 return &slotElement;
57 }
58 return nullptr;

62static HTMLSlotElement* nextSlotElementSkippingSubtree(ContainerNode& startingNode, ContainerNode* skippedSubtree)
63{
64 Node* node = &startingNode;
65 do {
66 if (UNLIKELY(node == skippedSubtree))
67 node = NodeTraversal::nextSkippingChildren(*node);
68 else
69 node = NodeTraversal::next(*node);
70 } while (node && !is<HTMLSlotElement>(node));
71 return downcast<HTMLSlotElement>(node);
72}

78HTMLSlotElement* SlotAssignment::findAssignedSlot(const Node& node, ShadowRoot& shadowRoot)
79{
80 if (!is<Text>(node) && !is<Element>(node))
81 return nullptr;
82
83 auto* slot = m_slots.get(slotNameForHostChild(node));
84 if (!slot)
85 return nullptr;
86
87 return findFirstSlotElement(*slot, shadowRoot);
88}

90inline bool SlotAssignment::hasAssignedNodes(ShadowRoot& shadowRoot, Slot& slot)
91{
92 if (!m_slotAssignmentsIsValid)
93 assignSlots(shadowRoot);
94 return !slot.assignedNodes.isEmpty();

97void SlotAssignment::renameSlotElement(HTMLSlotElement& slotElement, const AtomString& oldName, const AtomString& newName, ShadowRoot& shadowRoot)
98{
99 ASSERT(m_slotElementsForConsistencyCheck.contains(&slotElement));
100
101 m_slotMutationVersion++;
102
103 removeSlotElementByName(oldName, slotElement, nullptr, shadowRoot);
104 addSlotElementByName(newName, slotElement, shadowRoot);
105}

107void SlotAssignment::addSlotElementByName(const AtomString& name, HTMLSlotElement& slotElement, ShadowRoot& shadowRoot)
108{
109#if ASSERT_ENABLED
110 ASSERT(!m_slotElementsForConsistencyCheck.contains(&slotElement));
111 m_slotElementsForConsistencyCheck.add(&slotElement);
112#endif
113
114 // FIXME: We should be able to do a targeted reconstruction.
115 shadowRoot.host()->invalidateStyleAndRenderersForSubtree();
116
117 auto& slotName = slotNameFromAttributeValue(name);
118 auto addResult = m_slots.ensure(slotName, [&] {
119 m_slotAssignmentsIsValid = false;
120 return makeUnique<Slot>();
121 });
122 auto& slot = *addResult.iterator->value;
123
124 if (!m_slotAssignmentsIsValid)
125 assignSlots(shadowRoot);
126
127 shadowRoot.host()->setHasShadowRootContainingSlots(true);
128 m_slotElementCount++;
129
130 bool needsSlotchangeEvent = shadowRoot.shouldFireSlotchangeEvent() && hasAssignedNodes(shadowRoot, slot);
131
132 slot.elementCount++;
133 if (slot.elementCount == 1) {
134 slot.element = makeWeakPtr(slotElement);
135 if (needsSlotchangeEvent)
136 slotElement.enqueueSlotChangeEvent();
137 return;
138 }
139
140 if (!needsSlotchangeEvent) {
141 ASSERT(slot.element || m_needsToResolveSlotElements);
142 slot.element = nullptr;
143 m_needsToResolveSlotElements = true;
144 return;
145 }
146
147 resolveSlotsAfterSlotMutation(shadowRoot, SlotMutationType::Insertion);
148}

150void SlotAssignment::removeSlotElementByName(const AtomString& name, HTMLSlotElement& slotElement, ContainerNode* oldParentOfRemovedTreeForRemoval, ShadowRoot& shadowRoot)
151{
152#if ASSERT_ENABLED
153 ASSERT(m_slotElementsForConsistencyCheck.contains(&slotElement));
154 m_slotElementsForConsistencyCheck.remove(&slotElement);
155#endif
156
157 ASSERT(m_slotElementCount > 0);
158 m_slotElementCount--;
159
160 if (auto host = makeRefPtr(shadowRoot.host())) {
161 // FIXME: We should be able to do a targeted reconstruction.
162 host->invalidateStyleAndRenderersForSubtree();
163 if (!m_slotElementCount)
164 host->setHasShadowRootContainingSlots(false);
165 }
166
167 auto* slot = m_slots.get(slotNameFromAttributeValue(name));
168 RELEASE_ASSERT(slot && slot->hasSlotElements());
169 bool needsSlotchangeEvent = shadowRoot.shouldFireSlotchangeEvent() && hasAssignedNodes(shadowRoot, *slot);
170
171 slot->elementCount--;
172 if (!slot->elementCount) {
173 slot->element = nullptr;
174 if (needsSlotchangeEvent && m_slotResolutionVersion != m_slotMutationVersion)
175 slotElement.enqueueSlotChangeEvent();
176 return;
177 }
178
179 if (!needsSlotchangeEvent) {
180 ASSERT(slot->element || m_needsToResolveSlotElements);
181 slot->element = nullptr;
182 m_needsToResolveSlotElements = true;
183 return;
184 }
185
186 bool elementWasRenamed = !oldParentOfRemovedTreeForRemoval;
187 if (elementWasRenamed && slot->element == &slotElement)
188 slotElement.enqueueSlotChangeEvent();
189
190 // A previous invocation to resolveSlotsAfterSlotMutation during this removal has updated this slot.
191 ASSERT(slot->element || (m_slotResolutionVersion == m_slotMutationVersion && !findSlotElement(shadowRoot, name)));
192 if (slot->element) {
193 resolveSlotsAfterSlotMutation(shadowRoot, elementWasRenamed ? SlotMutationType::Insertion : SlotMutationType::Removal,
194 m_willBeRemovingAllChildren ? oldParentOfRemovedTreeForRemoval : nullptr);
195 }
196
197 if (slot->oldElement == &slotElement) {
198 slotElement.enqueueSlotChangeEvent();
199 slot->oldElement = nullptr;
200 }
201}

203void SlotAssignment::resolveSlotsAfterSlotMutation(ShadowRoot& shadowRoot, SlotMutationType mutationType, ContainerNode* subtreeToSkip)
204{
205 if (m_slotResolutionVersion == m_slotMutationVersion)
206 return;
207 m_slotResolutionVersion = m_slotMutationVersion;
208
209 ASSERT(!subtreeToSkip || mutationType == SlotMutationType::Removal);
210 m_needsToResolveSlotElements = false;
211
212 for (auto& slot : m_slots.values())
213 slot->seenFirstElement = false;
214
215 unsigned slotCount = 0;
216 HTMLSlotElement* currentElement = nextSlotElementSkippingSubtree(shadowRoot, subtreeToSkip);
217 for (; currentElement; currentElement = nextSlotElementSkippingSubtree(*currentElement, subtreeToSkip)) {
218 auto& currentSlotName = slotNameFromAttributeValue(currentElement->attributeWithoutSynchronization(nameAttr));
219 auto* currentSlot = m_slots.get(currentSlotName);
220 if (!currentSlot) {
221 // A new slot may have been inserted with this node but appears later in the tree order.
222 // Such a slot would go through the fast path in addSlotElementByName,
223 // and any subsequently inserted slot of the same name would not result in any slotchange or invokation of this function.
224 ASSERT(mutationType == SlotMutationType::Insertion);
225 continue;
226 }
227 if (currentSlot->seenFirstElement) {
228 if (mutationType == SlotMutationType::Insertion && currentSlot->oldElement == currentElement) {
229 currentElement->enqueueSlotChangeEvent();
230 currentSlot->oldElement = nullptr;
231 }
232 continue;
233 }
234 currentSlot->seenFirstElement = true;
235 slotCount++;
236 ASSERT(currentSlot->element || !hasAssignedNodes(shadowRoot, *currentSlot));
237 if (currentSlot->element != currentElement) {
238 if (hasAssignedNodes(shadowRoot, *currentSlot)) {
239 currentSlot->oldElement = WTFMove(currentSlot->element);
240 currentElement->enqueueSlotChangeEvent();
241 }
242 currentSlot->element = makeWeakPtr(*currentElement);
243 }
244 }
245
246 if (slotCount == m_slots.size())
247 return;
248
249 if (mutationType == SlotMutationType::Insertion) {
250 // This code path is taken only when continue above for !currentSlot is taken.
251 // i.e. there is a new slot being inserted into the tree but we have yet to invoke addSlotElementByName on it.
252#if ASSERT_ENABLED
253 for (auto& entry : m_slots)
254 ASSERT(entry.value->seenFirstElement || !findSlotElement(shadowRoot, entry.key));
255#endif
256 return;
257 }
258
259 for (auto& slot : m_slots.values()) {
260 if (slot->seenFirstElement)
261 continue;
262 if (!slot->elementCount) {
263 // Taken the fast path for removal.
264 ASSERT(!slot->element);
265 continue;
266 }
267 // All slot elements have been removed for this slot.
268 slot->seenFirstElement = true;
269 ASSERT(slot->element);
270 if (hasAssignedNodes(shadowRoot, *slot))
271 slot->oldElement = WTFMove(slot->element);
272 slot->element = nullptr;
273 }
274}

276void SlotAssignment::slotFallbackDidChange(HTMLSlotElement& slotElement, ShadowRoot& shadowRoot)
277{
278 if (shadowRoot.mode() == ShadowRootMode::UserAgent)
279 return;
280
281 bool usesFallbackContent = !assignedNodesForSlot(slotElement, shadowRoot);
282 if (usesFallbackContent)
283 slotElement.enqueueSlotChangeEvent();
284}

286void SlotAssignment::resolveSlotsBeforeNodeInsertionOrRemoval(ShadowRoot& shadowRoot)
287{
288 ASSERT(shadowRoot.shouldFireSlotchangeEvent());
289 m_slotMutationVersion++;
290 m_willBeRemovingAllChildren = false;
291 if (m_needsToResolveSlotElements)
292 resolveAllSlotElements(shadowRoot);

295void SlotAssignment::willRemoveAllChildren(ShadowRoot& shadowRoot)

297 m_slotMutationVersion++;
298 m_willBeRemovingAllChildren = true;

300 resolveAllSlotElements(shadowRoot);

303void SlotAssignment::didChangeSlot(const AtomString& slotAttrValue, ShadowRoot& shadowRoot)
304{
305 auto& slotName = slotNameFromAttributeValue(slotAttrValue);
306 auto* slot = m_slots.get(slotName);
307 if (!slot)
308 return;
309
310 RenderTreeUpdater::tearDownRenderers(*shadowRoot.host());
311
312 slot->assignedNodes.clear();
313 m_slotAssignmentsIsValid = false;
314
315 auto slotElement = makeRefPtr(findFirstSlotElement(*slot, shadowRoot));
316 if (!slotElement)
317 return;
318
319 shadowRoot.host()->invalidateStyleAndRenderersForSubtree();
320
321 if (shadowRoot.shouldFireSlotchangeEvent())
322 slotElement->enqueueSlotChangeEvent();
323}

325void SlotAssignment::hostChildElementDidChange(const Element& childElement, ShadowRoot& shadowRoot)

327 didChangeSlot(childElement.attributeWithoutSynchronization(slotAttr), shadowRoot);

330const Vector<WeakPtr<Node>>* SlotAssignment::assignedNodesForSlot(const HTMLSlotElement& slotElement, ShadowRoot& shadowRoot)
331{
332 ASSERT(slotElement.containingShadowRoot() == &shadowRoot);
333 const AtomString& slotName = slotNameFromAttributeValue(slotElement.attributeWithoutSynchronization(nameAttr));
334 auto* slot = m_slots.get(slotName);
335
336 bool hasNotAddedSlotInInsertedIntoAncestorYet = shadowRoot.isConnected() && (!slotElement.isConnected() || slotElement.isInInsertedIntoAncestor());
337 if (hasNotAddedSlotInInsertedIntoAncestorYet)
338 return nullptr;
339 RELEASE_ASSERT(slot);
340
341 if (!m_slotAssignmentsIsValid)
342 assignSlots(shadowRoot);
343
344 if (slot->assignedNodes.isEmpty())
345 return nullptr;
346
347 RELEASE_ASSERT(slot->hasSlotElements());
348 if (slot->hasDuplicatedSlotElements() && findFirstSlotElement(*slot, shadowRoot) != &slotElement)
349 return nullptr;
350
351 return &slot->assignedNodes;
352}

354void SlotAssignment::willRemoveAssignedNode(const Node& node)
355{
356 if (!m_slotAssignmentsIsValid)
357 return;
358
359 if (!is<Text>(node) && !is<Element>(node))
360 return;
361
362 auto* slot = m_slots.get(slotNameForHostChild(node));
363 if (!slot || slot->assignedNodes.isEmpty())
364 return;
365
366 slot->assignedNodes.removeFirstMatching([&node](const auto& item) {
367 return item.get() == &node;
368 });
369}

376HTMLSlotElement* SlotAssignment::findFirstSlotElement(Slot& slot, ShadowRoot& shadowRoot)
377{
378 if (slot.shouldResolveSlotElement())
379 resolveAllSlotElements(shadowRoot);
380
381 ASSERT(!slot.element || m_slotElementsForConsistencyCheck.contains(slot.element.get()));
382 ASSERT(!!slot.element == !!slot.elementCount);
383
384 return slot.element.get();
385}
386
387void SlotAssignment::resolveAllSlotElements(ShadowRoot& shadowRoot)
388{
389 ASSERT(m_needsToResolveSlotElements);
390 m_needsToResolveSlotElements = false;
391
392 // FIXME: It's inefficient to reset all values. We should be able to void this in common case.
393 for (auto& entry : m_slots)
394 entry.value->seenFirstElement = false;
395
396 unsigned slotCount = m_slots.size();
397 for (auto& slotElement : descendantsOfType<HTMLSlotElement>(shadowRoot)) {
398 auto& slotName = slotNameFromAttributeValue(slotElement.attributeWithoutSynchronization(nameAttr));
399
400 auto* slot = m_slots.get(slotName);
401 RELEASE_ASSERT(slot); // slot must have been created when a slot was inserted.
402
403 if (slot->seenFirstElement)
404 continue;
405 slot->seenFirstElement = true;
406
407 slot->element = makeWeakPtr(slotElement);
408 slotCount--;
409 if (!slotCount)
410 break;
411 }
412}

414void SlotAssignment::assignSlots(ShadowRoot& shadowRoot)
415{
416 ASSERT(!m_slotAssignmentsIsValid);
417 m_slotAssignmentsIsValid = true;
418
419 for (auto& entry : m_slots)
420 entry.value->assignedNodes.shrink(0);
421
422 auto& host = *shadowRoot.host();
423 for (auto* child = host.firstChild(); child; child = child->nextSibling()) {
424 if (!is<Text>(*child) && !is<Element>(*child))
425 continue;
426 auto slotName = slotNameForHostChild(*child);
427 assignToSlot(*child, slotName);
428 }
429
430 for (auto& entry : m_slots)
431 entry.value->assignedNodes.shrinkToFit();
432}
433
434void SlotAssignment::assignToSlot(Node& child, const AtomString& slotName)
435{
436 ASSERT(!slotName.isNull());
437 if (slotName == defaultSlotName()) {
438 auto defaultSlotEntry = m_slots.find(defaultSlotName());
439 if (defaultSlotEntry != m_slots.end())
440 defaultSlotEntry->value->assignedNodes.append(makeWeakPtr(child));
441 return;
442 }
443
444 auto addResult = m_slots.ensure(slotName, [] {
445 return makeUnique<Slot>();
446 });
447 addResult.iterator->value->assignedNodes.append(makeWeakPtr(child));
448}
