39template<typename> class ExceptionOr;
40
41class StaticRange final : public AbstractRange, public SimpleRange {
42 WTF_MAKE_ISO_ALLOCATED(StaticRange);
43public:
44 struct Init {
45 RefPtr<Node> startContainer;
46 unsigned startOffset { 0 };
47 RefPtr<Node> endContainer;
48 unsigned endOffset { 0 };
49 };
50
51 static ExceptionOr<Ref<StaticRange>> create(Init&&);
52 WEBCORE_EXPORT static Ref<StaticRange> create(const SimpleRange&);
53 static Ref<StaticRange> create(SimpleRange&&);
54
55 Node& startContainer() const final { return SimpleRange::startContainer(); }
56 unsigned startOffset() const final { return SimpleRange::startOffset(); }
57 Node& endContainer() const final { return SimpleRange::endContainer(); }
58 unsigned endOffset() const final { return SimpleRange::endOffset(); }
59 bool collapsed() const final { return SimpleRange::collapsed(); }
60
61 void visitNodesConcurrently(JSC::AbstractSlotVisitor&) const;
62
63private:
64 StaticRange(SimpleRange&&);
65
66 bool isLiveRange() const final { return false; }
67};
68
69}

71SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::StaticRange)
72 static bool isType(const WebCore::AbstractRange& range) { return !range.isLiveRange(); }
