39void StringCallback::scheduleCallback(ScriptExecutionContext& context, const String& data)
40{
41 RefPtr<StringCallback> protectedThis(this);
42 context.postTask([protectedThis, data] (ScriptExecutionContext&) {
43 protectedThis->handleEvent(data);
44 });
45}
