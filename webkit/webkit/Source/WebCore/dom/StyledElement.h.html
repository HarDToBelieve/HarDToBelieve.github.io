34class Attribute;

40class StyledElement : public Element {
41 WTF_MAKE_ISO_ALLOCATED(StyledElement);
42public:
43 virtual ~StyledElement();
44
45 void invalidateStyleAttribute();
46
47 const StyleProperties* inlineStyle() const { return elementData() ? elementData()->m_inlineStyle.get() : nullptr; }
48
49 bool setInlineStyleProperty(CSSPropertyID, CSSValueID identifier, bool important = false);
50 bool setInlineStyleProperty(CSSPropertyID, CSSPropertyID identifier, bool important = false);
51 WEBCORE_EXPORT bool setInlineStyleProperty(CSSPropertyID, double value, CSSUnitType, bool important = false);
52 WEBCORE_EXPORT bool setInlineStyleProperty(CSSPropertyID, const String& value, bool important = false);
53 bool removeInlineStyleProperty(CSSPropertyID);
54 void removeAllInlineStyleProperties();
55
56 void synchronizeStyleAttributeInternal() const { const_cast<StyledElement*>(this)->synchronizeStyleAttributeInternalImpl(); }
57
58 WEBCORE_EXPORT CSSStyleDeclaration& cssomStyle();
59#if ENABLE(CSS_TYPED_OM)
60 StylePropertyMap& ensureAttributeStyleMap();
61#endif
62
63 // https://html.spec.whatwg.org/#presentational-hints
64 const StyleProperties* presentationalHintStyle() const;
65 virtual void collectPresentationalHintsForAttribute(const QualifiedName&, const AtomString&, MutableStyleProperties&) { }
66 virtual const StyleProperties* additionalPresentationalHintStyle() const { return nullptr; }
67 virtual void collectExtraStyleForPresentationalHints(MutableStyleProperties&) { }
68
69protected:
70 StyledElement(const QualifiedName& name, Document& document, ConstructionType type)
71 : Element(name, document, type)
72 {
73 }
74
75 void attributeChanged(const QualifiedName&, const AtomString& oldValue, const AtomString& newValue, AttributeModificationReason = ModifiedDirectly) override;
76
77 virtual bool hasPresentationalHintsForAttribute(const QualifiedName&) const { return false; }
78
79 void addPropertyToPresentationalHintStyle(MutableStyleProperties&, CSSPropertyID, CSSValueID identifier);
80 void addPropertyToPresentationalHintStyle(MutableStyleProperties&, CSSPropertyID, double value, CSSUnitType);
81 void addPropertyToPresentationalHintStyle(MutableStyleProperties&, CSSPropertyID, const String& value);
82
83 void addSubresourceAttributeURLs(ListHashSet<URL>&) const override;
84
85private:
86 void styleAttributeChanged(const AtomString& newStyleString, AttributeModificationReason);
87 void synchronizeStyleAttributeInternalImpl();
88
89 void inlineStyleChanged();
90 PropertySetCSSStyleDeclaration* inlineStyleCSSOMWrapper();
91 void setInlineStyleFromString(const AtomString&);
92 MutableStyleProperties& ensureMutableInlineStyle();
93
94 void rebuildPresentationalHintStyle();
95};
96
97inline const StyleProperties* StyledElement::presentationalHintStyle() const
98{
99 if (!elementData())
100 return nullptr;
101 if (elementData()->presentationalHintStyleIsDirty())
102 const_cast<StyledElement&>(*this).rebuildPresentationalHintStyle();
103 return elementData()->presentationalHintStyle();
104}
105
106} // namespace WebCore
