30ExceptionOr<Ref<TextDecoderStreamDecoder>> TextDecoderStreamDecoder::create(const String& label, bool fatal, bool ignoreBOM)
31{
32 auto textDecoder = TextDecoder::create(label, { fatal, ignoreBOM });
33 if (textDecoder.hasException())
34 return textDecoder.releaseException();
35
36 return adoptRef(*new TextDecoderStreamDecoder(textDecoder.releaseReturnValue()));
37}
38
39TextDecoderStreamDecoder::TextDecoderStreamDecoder(Ref<TextDecoder>&& textDecoder)
40 : m_textDecoder(WTFMove(textDecoder))
41{
42}
43
44ExceptionOr<String> TextDecoderStreamDecoder::decode(std::optional<BufferSource::VariantType> value)
45{
46 return m_textDecoder->decode(WTFMove(value), { true });
47}
48
49ExceptionOr<String> TextDecoderStreamDecoder::flush()
50{
51 return m_textDecoder->decode({ }, { false });
52}
