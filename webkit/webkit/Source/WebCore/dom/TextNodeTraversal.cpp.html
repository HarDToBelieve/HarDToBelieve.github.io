32namespace WebCore {
33namespace TextNodeTraversal {
34
35void appendContents(const ContainerNode& root, StringBuilder& result)
36{
37 for (Text* text = TextNodeTraversal::firstWithin(root); text; text = TextNodeTraversal::next(*text, &root))
38 result.append(text->data());
39}
40
41String contentsAsString(const ContainerNode& root)
42{
43 StringBuilder result;
44 appendContents(root, result);
45 return result.toString();
46}
47
48String contentsAsString(const Node& root)
49{
50 if (is<Text>(root))
51 return downcast<Text>(root).data();
52 if (is<ContainerNode>(root))
53 return contentsAsString(downcast<ContainerNode>(root));
54 return String();
55}
56
57String childTextContent(const ContainerNode& root)
58{
59 StringBuilder result;
60 for (Text* text = TextNodeTraversal::firstChild(root); text; text = TextNodeTraversal::nextSibling(*text))
61 result.append(text->data());
62 return result.toString();
63}
