26#pragma once
27
28#include "Document.h"
29
30namespace WebCore {
31
32class ThrowOnDynamicMarkupInsertionCountIncrementer {
33 WTF_MAKE_NONCOPYABLE(ThrowOnDynamicMarkupInsertionCountIncrementer);
34public:
35 explicit ThrowOnDynamicMarkupInsertionCountIncrementer(Document& document)
36 : m_document(document)
37 {
38 ++document.m_throwOnDynamicMarkupInsertionCount;
39 }
40
41 ~ThrowOnDynamicMarkupInsertionCountIncrementer()
42 {
43 ASSERT(m_document->m_throwOnDynamicMarkupInsertionCount);
44 --m_document->m_throwOnDynamicMarkupInsertionCount;
45 }
46
47private:
48 Ref<Document> m_document;
49};
50
51} // namespace WebCore
