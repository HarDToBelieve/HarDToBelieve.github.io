31namespace WebCore {
32
33class TransitionEvent final : public AnimationEventBase {
34 WTF_MAKE_ISO_ALLOCATED(TransitionEvent);
35public:
36 static Ref<TransitionEvent> create(const AtomString& type, const String& propertyName, double elapsedTime, const String& pseudoElement, std::optional<Seconds> timelineTime, WebAnimation* animation)
37 {
38 return adoptRef(*new TransitionEvent(type, propertyName, elapsedTime, pseudoElement, timelineTime, animation));
39 }
40
41 struct Init : EventInit {
42 String propertyName;
43 double elapsedTime { 0 };
44 String pseudoElement;
45 };
46
47 static Ref<TransitionEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
48 {
49 return adoptRef(*new TransitionEvent(type, initializer, isTrusted));
50 }
51
52 virtual ~TransitionEvent();
53
54 bool isTransitionEvent() const final { return true; }
55
56 const String& propertyName() const;
57 double elapsedTime() const;
58 const String& pseudoElement() const;
59
60 EventInterface eventInterface() const override;
61
62private:
63 TransitionEvent(const AtomString& type, const String& propertyName, double elapsedTime, const String& pseudoElement, std::optional<Seconds> timelineTime, WebAnimation*);
64 TransitionEvent(const AtomString& type, const Init& initializer, IsTrusted);
65
66 String m_propertyName;
67 double m_elapsedTime;
68 String m_pseudoElement;
69};

71} // namespace WebCore
