35NodeIteratorBase::NodeIteratorBase(Node& rootNode, unsigned whatToShow, RefPtr<NodeFilter>&& nodeFilter)
36 : m_root(rootNode)
37 , m_filter(WTFMove(nodeFilter))
38 , m_whatToShow(whatToShow)
39{
40}
41
42// https://dom.spec.whatwg.org/#concept-node-filter
43ExceptionOr<unsigned short> NodeIteratorBase::acceptNode(Node& node)
44{
45 if (m_isActive)
46 return Exception { InvalidStateError, "Recursive filters are not allowed"_s };
47
48 // The bit twiddling here is done to map DOM node types, which are given as integers from
49 // 1 through 14, to whatToShow bit masks.
50 if (!(((1 << (node.nodeType() - 1)) & m_whatToShow)))
51 return NodeFilter::FILTER_SKIP;
52
53 if (!m_filter)
54 return NodeFilter::FILTER_ACCEPT;
55
56 SetForScope<bool> isActive(m_isActive, true);
57 auto callbackResult = m_filter->acceptNode(node);
58 if (callbackResult.type() == CallbackResultType::ExceptionThrown)
59 return Exception { ExistingExceptionError };
60 return callbackResult.releaseReturnValue();
61}
