30auto UIEventWithKeyState::modifiersFromInitializer(const EventModifierInit& initializer) -> OptionSet<Modifier>
31{
32 OptionSet<Modifier> result;
33 if (initializer.ctrlKey)
34 result.add(Modifier::ControlKey);
35 if (initializer.altKey)
36 result.add(Modifier::AltKey);
37 if (initializer.shiftKey)
38 result.add(Modifier::ShiftKey);
39 if (initializer.metaKey)
40 result.add(Modifier::MetaKey);
41 if (initializer.modifierAltGraph)
42 result.add(Modifier::AltGraphKey);
43 if (initializer.modifierCapsLock)
44 result.add(Modifier::CapsLockKey);
45 return result;
46}
47
48bool UIEventWithKeyState::getModifierState(const String& keyIdentifier) const
49{
50 if (keyIdentifier == "Control")
51 return ctrlKey();
52 if (keyIdentifier == "Shift")
53 return shiftKey();
54 if (keyIdentifier == "Alt")
55 return altKey();
56 if (keyIdentifier == "Meta")
57 return metaKey();
58 if (keyIdentifier == "AltGraph")
59 return altGraphKey();
60 if (keyIdentifier == "CapsLock")
61 return capsLockKey();
62 // FIXME: The specification also has Fn, FnLock, Hyper, NumLock, Super, ScrollLock, Symbol, SymbolLock.
63 return false;
64}
65
66void UIEventWithKeyState::setModifierKeys(bool ctrlKey, bool altKey, bool shiftKey, bool metaKey, bool altGraphKey)
67{
68 OptionSet<Modifier> result;
69 if (ctrlKey)
70 result.add(Modifier::ControlKey);
71 if (altKey)
72 result.add(Modifier::AltKey);
73 if (shiftKey)
74 result.add(Modifier::ShiftKey);
75 if (metaKey)
76 result.add(Modifier::MetaKey);
77 if (altGraphKey)
78 result.add(Modifier::AltGraphKey);
79 m_modifiers = result;
80}
81
82UIEventWithKeyState* findEventWithKeyState(Event* event)
83{
84 for (Event* e = event; e; e = e->underlyingEvent())
85 if (e->isKeyboardEvent() || e->isMouseEvent())
86 return static_cast<UIEventWithKeyState*>(e);
87 return 0;
88}
