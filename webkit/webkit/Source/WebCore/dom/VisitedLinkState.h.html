36namespace WebCore {

38class Document;
39
40class VisitedLinkState {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 explicit VisitedLinkState(Document&);
44
45 void invalidateStyleForAllLinks();
46 void invalidateStyleForLink(SharedStringHash);
47 InsideLink determineLinkState(const Element&);
48
49private:
50 InsideLink determineLinkStateSlowCase(const Element&);
51
52 Document& m_document;
53 HashSet<SharedStringHash, SharedStringHashHash> m_linksCheckedForVisitedState;
54};
55
56inline InsideLink VisitedLinkState::determineLinkState(const Element& element)
57{
58 if (!element.isLink())
59 return InsideLink::NotInside;
60 return determineLinkStateSlowCase(element);
61}
