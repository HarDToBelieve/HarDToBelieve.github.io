36inline static unsigned determineDeltaMode(const PlatformWheelEvent& event)
37{
38 return event.granularity() == ScrollByPageWheelEvent ? WheelEvent::DOM_DELTA_PAGE : WheelEvent::DOM_DELTA_PIXEL;
39}
40
41static double wheelDeltaToDelta(int wheelDelta)
42{
43 // Avoid returning negative zero.
44 if (!wheelDelta)
45 return 0;
46 return -static_cast<double>(wheelDelta);
47}

51inline WheelEvent::WheelEvent(const AtomString& type, const Init& initializer)
52 : MouseEvent(type, initializer)
53 , m_wheelDelta(initializer.wheelDeltaX ? initializer.wheelDeltaX : clampTo<int>(-initializer.deltaX), initializer.wheelDeltaY ? initializer.wheelDeltaY : clampTo<int>(-initializer.deltaY))
54 , m_deltaX(initializer.deltaX ? initializer.deltaX : wheelDeltaToDelta(initializer.wheelDeltaX))
55 , m_deltaY(initializer.deltaY ? initializer.deltaY : wheelDeltaToDelta(initializer.wheelDeltaY))
56 , m_deltaZ(initializer.deltaZ)
57 , m_deltaMode(initializer.deltaMode)
58{
59}

61inline WheelEvent::WheelEvent(const PlatformWheelEvent& event, RefPtr<WindowProxy>&& view, IsCancelable isCancelable)
62 : MouseEvent(eventNames().wheelEvent, CanBubble::Yes, isCancelable, IsComposed::Yes, event.timestamp().approximateMonotonicTime(), WTFMove(view), 0,
63 event.globalPosition(), event.position() , { }, event.modifiers(), 0, 0, nullptr, 0, 0, IsSimulated::No, IsTrusted::Yes)
64 , m_wheelDelta(event.wheelTicksX() * TickMultiplier, event.wheelTicksY() * TickMultiplier)
65 , m_deltaX(-event.deltaX())
66 , m_deltaY(-event.deltaY())
67 , m_deltaMode(determineDeltaMode(event))
68 , m_underlyingPlatformEvent(event)
69{
70}

87void WheelEvent::initWebKitWheelEvent(int rawDeltaX, int rawDeltaY, RefPtr<WindowProxy>&& view, int screenX, int screenY, int pageX, int pageY, bool ctrlKey, bool altKey, bool shiftKey, bool metaKey)
88{
89 if (isBeingDispatched())
90 return;
91
92 initMouseEvent(eventNames().wheelEvent, true, true, WTFMove(view), 0, screenX, screenY, pageX, pageY, ctrlKey, altKey, shiftKey, metaKey, 0, nullptr);
93
94 // Normalize to 120 multiple for compatibility with IE.
95 m_wheelDelta = { rawDeltaX * TickMultiplier, rawDeltaY * TickMultiplier };
96 m_deltaX = wheelDeltaToDelta(rawDeltaX);
97 m_deltaY = wheelDeltaToDelta(rawDeltaY);
98
99 m_deltaMode = DOM_DELTA_PIXEL;
100
101 m_underlyingPlatformEvent = std::nullopt;
102}
