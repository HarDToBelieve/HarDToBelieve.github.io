41static HashMap<String, WindowEventLoop*>& windowEventLoopMap()
42{
43 RELEASE_ASSERT(isMainThread());
44 static NeverDestroyed<HashMap<String, WindowEventLoop*>> map;
45 return map.get();
46}

48static String agentClusterKeyOrNullIfUnique(const SecurityOrigin& origin)
49{
50 auto computeKey = [&] {
51 // https://html.spec.whatwg.org/multipage/webappapis.html#obtain-agent-cluster-key
52 if (origin.isUnique())
53 return origin.toString();
54 RegistrableDomain registrableDomain { origin.data() };
55 if (registrableDomain.isEmpty())
56 return origin.toString();
57 return makeString(origin.protocol(), "://", registrableDomain.string());
58 };
59 auto key = computeKey();
60 if (key.isEmpty() || key == "null"_s)
61 return { };
62 return key;
63}
64
65Ref<WindowEventLoop> WindowEventLoop::eventLoopForSecurityOrigin(const SecurityOrigin& origin)
66{
67 auto key = agentClusterKeyOrNullIfUnique(origin);
68 if (key.isNull())
69 return create({ });
70
71 auto addResult = windowEventLoopMap().add(key, nullptr);
72 if (UNLIKELY(addResult.isNewEntry)) {
73 auto newEventLoop = create(key);
74 addResult.iterator->value = newEventLoop.ptr();
75 return newEventLoop;
76 }
77 return *addResult.iterator->value;
78}

80inline Ref<WindowEventLoop> WindowEventLoop::create(const String& agentClusterKey)
81{
82 return adoptRef(*new WindowEventLoop(agentClusterKey));
83}

85inline WindowEventLoop::WindowEventLoop(const String& agentClusterKey)
86 : m_agentClusterKey(agentClusterKey)
87 , m_timer(*this, &WindowEventLoop::didReachTimeToRun)
88 , m_perpetualTaskGroupForSimilarOriginWindowAgents(*this)
89{
90}
91
92WindowEventLoop::~WindowEventLoop()
93{
94 if (m_agentClusterKey.isNull())
95 return;
96 auto didRemove = windowEventLoopMap().remove(m_agentClusterKey);
97 RELEASE_ASSERT(didRemove);
98}

117void WindowEventLoop::didReachTimeToRun()
118{
119 auto protectedThis = makeRef(*this); // Executing tasks may remove the last reference to this WindowEventLoop.
120 run();
121}

123void WindowEventLoop::queueMutationObserverCompoundMicrotask()
124{
125 if (m_mutationObserverCompoundMicrotaskQueuedFlag)
126 return;
127 m_mutationObserverCompoundMicrotaskQueuedFlag = true;
128 m_perpetualTaskGroupForSimilarOriginWindowAgents.queueMicrotask([this] {
129 // We can't make a Ref to WindowEventLoop in the lambda capture as that would result in a reference cycle & leak.
130 auto protectedThis = makeRef(*this);
131 m_mutationObserverCompoundMicrotaskQueuedFlag = false;
132
133 // FIXME: This check doesn't exist in the spec.
134 if (m_deliveringMutationRecords)
135 return;
136 m_deliveringMutationRecords = true;
137 MutationObserver::notifyMutationObservers(*this);
138 m_deliveringMutationRecords = false;
139 });
140}

142CustomElementQueue& WindowEventLoop::backupElementQueue()
143{
144 if (!m_processingBackupElementQueue) {
145 m_processingBackupElementQueue = true;
146 m_perpetualTaskGroupForSimilarOriginWindowAgents.queueMicrotask([this] {
147 // We can't make a Ref to WindowEventLoop in the lambda capture as that would result in a reference cycle & leak.
148 auto protectedThis = makeRef(*this);
149 m_processingBackupElementQueue = false;
150 ASSERT(m_customElementQueue);
151 CustomElementReactionQueue::processBackupQueue(*m_customElementQueue);
152 });
153 }
154 if (!m_customElementQueue)
155 m_customElementQueue = makeUnique<CustomElementQueue>();
156 return *m_customElementQueue;
157}

160{
