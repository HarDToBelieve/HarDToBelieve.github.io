36class CustomElementQueue;
37class Document;

42// https://html.spec.whatwg.org/multipage/webappapis.html#window-event-loop
43class WindowEventLoop final : public EventLoop {
44public:
45 static Ref<WindowEventLoop> eventLoopForSecurityOrigin(const SecurityOrigin&);
46
47 virtual ~WindowEventLoop();
48
49 void queueMutationObserverCompoundMicrotask();
50 Vector<GCReachableRef<HTMLSlotElement>>& signalSlotList() { return m_signalSlotList; }
51 HashSet<RefPtr<MutationObserver>>& activeMutationObservers() { return m_activeObservers; }
52 HashSet<RefPtr<MutationObserver>>& suspendedMutationObservers() { return m_suspendedObservers; }
53
54 CustomElementQueue& backupElementQueue();
55
56 WEBCORE_EXPORT static void breakToAllowRenderingUpdate();
57
58private:
59 static Ref<WindowEventLoop> create(const String&);
60 WindowEventLoop(const String&);
61
62 void scheduleToRun() final;
63 bool isContextThread() const final;
64 MicrotaskQueue& microtaskQueue() final;
65
66 void didReachTimeToRun();
67
68 String m_agentClusterKey;
69 Timer m_timer;
70 std::unique_ptr<MicrotaskQueue> m_microtaskQueue;
71
72 // Each task scheduled in event loop is associated with a document so that it can be suspened or stopped
73 // when the associated document is suspened or stopped. This task group is used to schedule a task
74 // which is not scheduled to a specific document, and should only be used when it's absolutely required.
75 EventLoopTaskGroup m_perpetualTaskGroupForSimilarOriginWindowAgents;
76
77 bool m_mutationObserverCompoundMicrotaskQueuedFlag { false };
78 bool m_deliveringMutationRecords { false }; // FIXME: This flag doesn't exist in the spec.
79 Vector<GCReachableRef<HTMLSlotElement>> m_signalSlotList; // https://dom.spec.whatwg.org/#signal-slot-list
80 HashSet<RefPtr<MutationObserver>> m_activeObservers;
81 HashSet<RefPtr<MutationObserver>> m_suspendedObservers;
82
83 std::unique_ptr<CustomElementQueue> m_customElementQueue;
84 bool m_processingBackupElementQueue { false };
85};
