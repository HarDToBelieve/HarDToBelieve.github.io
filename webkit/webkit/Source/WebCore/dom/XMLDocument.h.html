26#pragma once
27
28#include "Document.h"
29
30namespace WebCore {
31
32class XMLDocument : public Document {
33 WTF_MAKE_ISO_ALLOCATED(XMLDocument);
34public:
35 static Ref<XMLDocument> create(Frame* frame, const Settings& settings, const URL& url)
36 {
37 return adoptRef(*new XMLDocument(frame, settings, url));
38 }
39
40 static Ref<XMLDocument> createXHTML(Frame* frame, const Settings& settings, const URL& url)
41 {
42 return adoptRef(*new XMLDocument(frame, settings, url, XHTMLDocumentClass));
43 }
44
45protected:
46 XMLDocument(Frame* frame, const Settings& settings, const URL& url, unsigned documentClasses = DefaultDocumentClass)
47 : Document(frame, settings, url, XMLDocumentClass | documentClasses)
48 {
49 }
50};
51
52} // namespace WebCore
53
54SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::XMLDocument)
55 static bool isType(const WebCore::Document& document) { return document.isXMLDocument(); }
56 static bool isType(const WebCore::Node& node) { return is<WebCore::Document>(node) && isType(downcast<WebCore::Document>(node)); }
57SPECIALIZE_TYPE_TRAITS_END()
