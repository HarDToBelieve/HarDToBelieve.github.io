38 : m_scope(scope)

40}

42WorkerMessagePortChannelProvider::~WorkerMessagePortChannelProvider()
43{
44 while (!m_takeAllMessagesCallbacks.isEmpty()) {
45 auto first = m_takeAllMessagesCallbacks.begin();
46 first->value({ }, [] { });
47 m_takeAllMessagesCallbacks.remove(first);
48 }
49 while (!m_activityCallbacks.isEmpty()) {
50 auto first = m_activityCallbacks.begin();
51 first->value(HasActivity::No);
52 m_activityCallbacks.remove(first);
53 }
54}

89void WorkerMessagePortChannelProvider::takeAllMessagesForPort(const MessagePortIdentifier& identifier, CompletionHandler<void(Vector<MessageWithMessagePorts>&&, Function<void()>&&)>&& callback)
90{
91 uint64_t callbackIdentifier = ++m_lastCallbackIdentifier;
92 m_takeAllMessagesCallbacks.add(callbackIdentifier, WTFMove(callback));
93
94 callOnMainThread([this, workerThread = makeRefPtr(m_scope.workerOrWorkletThread()), callbackIdentifier, identifier]() mutable {
95 MessagePortChannelProvider::singleton().takeAllMessagesForPort(identifier, [this, workerThread = WTFMove(workerThread), callbackIdentifier](Vector<MessageWithMessagePorts>&& messages, Function<void()>&& completionHandler) {
96 workerThread->runLoop().postTaskForMode([this, callbackIdentifier, messages = WTFMove(messages), completionHandler = WTFMove(completionHandler)](auto&) mutable {
97 m_takeAllMessagesCallbacks.take(callbackIdentifier)(WTFMove(messages), [completionHandler = WTFMove(completionHandler)]() mutable {
98 callOnMainThread(WTFMove(completionHandler));
99 });
100 }, WorkerRunLoop::defaultMode());
101 });
102 });
103}
104
105void WorkerMessagePortChannelProvider::checkRemotePortForActivity(const MessagePortIdentifier& remoteTarget, CompletionHandler<void(HasActivity)>&& callback)
106{
107 uint64_t callbackIdentifier = ++m_lastCallbackIdentifier;
108 m_activityCallbacks.add(callbackIdentifier, WTFMove(callback));
109
110 callOnMainThread([this, workerThread = makeRefPtr(m_scope.workerOrWorkletThread()), callbackIdentifier, remoteTarget]() mutable {
111 MessagePortChannelProvider::singleton().checkRemotePortForActivity(remoteTarget, [this, workerThread = WTFMove(workerThread), callbackIdentifier](auto hasActivity) {
112 workerThread->runLoop().postTaskForMode([this, callbackIdentifier, hasActivity](auto&) mutable {
113 m_activityCallbacks.take(callbackIdentifier)(hasActivity);
114 }, WorkerRunLoop::defaultMode());
115 });
116 });
117}
