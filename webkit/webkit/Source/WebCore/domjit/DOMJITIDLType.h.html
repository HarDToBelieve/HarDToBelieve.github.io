28#include "IDLTypes.h"

30namespace WebCore { namespace DOMJIT {
31
32template<typename IDLType>
33struct IDLJSArgumentTypeSelect;
34
35template<> struct IDLJSArgumentTypeSelect<IDLBoolean> { typedef bool type; };
36template<> struct IDLJSArgumentTypeSelect<IDLByte> { typedef int32_t type; };
37template<> struct IDLJSArgumentTypeSelect<IDLOctet> { typedef int32_t type; };
38template<> struct IDLJSArgumentTypeSelect<IDLShort> { typedef int32_t type; };
39template<> struct IDLJSArgumentTypeSelect<IDLUnsignedShort> { typedef int32_t type; };
40template<> struct IDLJSArgumentTypeSelect<IDLLong> { typedef int32_t type; };
41template<> struct IDLJSArgumentTypeSelect<IDLDOMString> { typedef JSC::JSString* type; };
42template<> struct IDLJSArgumentTypeSelect<IDLAtomStringAdaptor<IDLDOMString>> { typedef JSC::JSString* type; };
43template<> struct IDLJSArgumentTypeSelect<IDLRequiresExistingAtomStringAdaptor<IDLDOMString>> { typedef JSC::JSString* type; };
44
45template<typename IDLType>
46using IDLJSArgumentType = typename IDLJSArgumentTypeSelect<IDLType>::type;
