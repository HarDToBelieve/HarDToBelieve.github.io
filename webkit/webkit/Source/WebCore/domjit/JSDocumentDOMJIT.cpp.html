51Ref<JSC::DOMJIT::CallDOMGetterSnippet> compileDocumentDocumentElementAttribute()
52{
53 Ref<JSC::DOMJIT::CallDOMGetterSnippet> snippet = JSC::DOMJIT::CallDOMGetterSnippet::create();
54 snippet->numGPScratchRegisters = 1;
55 snippet->setGenerator([=](CCallHelpers& jit, JSC::SnippetParams& params) {
56 JSValueRegs result = params[0].jsValueRegs();
57 GPRReg document = params[1].gpr();
58 GPRReg globalObject = params[2].gpr();
59 JSValue globalObjectValue = params[2].value();
60 GPRReg scratch = params.gpScratch(0);
61
62 jit.loadPtr(CCallHelpers::Address(document, JSDocument::offsetOfWrapped()), scratch);
63 DOMJIT::loadDocumentElement(jit, scratch, scratch);
64 auto nullCase = jit.branchTestPtr(CCallHelpers::Zero, scratch);
65 DOMJIT::toWrapper<Element>(jit, params, scratch, globalObject, result, DOMJIT::operationToJSElement, globalObjectValue);
66 auto done = jit.jump();
67
68 nullCase.link(&jit);
69 jit.moveValue(jsNull(), result);
70 done.link(&jit);
71
72 return CCallHelpers::JumpList();
73 });
74 snippet->effect = JSC::DOMJIT::Effect::forDef(DOMJIT::AbstractHeapRepository::Document_documentElement);
75 return snippet;
76}

78static void loadLocalName(CCallHelpers& jit, GPRReg htmlElement, GPRReg localNameImpl)
79{
80 jit.loadPtr(CCallHelpers::Address(htmlElement, Element::tagQNameMemoryOffset() + QualifiedName::implMemoryOffset()), localNameImpl);
81 jit.loadPtr(CCallHelpers::Address(localNameImpl, QualifiedName::QualifiedNameImpl::localNameMemoryOffset()), localNameImpl);
82}

84Ref<JSC::DOMJIT::CallDOMGetterSnippet> compileDocumentBodyAttribute()
85{
86 Ref<JSC::DOMJIT::CallDOMGetterSnippet> snippet = JSC::DOMJIT::CallDOMGetterSnippet::create();
87 snippet->numGPScratchRegisters = 2;
88 snippet->setGenerator([=](CCallHelpers& jit, JSC::SnippetParams& params) {
89 JSValueRegs result = params[0].jsValueRegs();
90 GPRReg document = params[1].gpr();
91 GPRReg globalObject = params[2].gpr();
92 JSValue globalObjectValue = params[2].value();
93 GPRReg scratch1 = params.gpScratch(0);
94 GPRReg scratch2 = params.gpScratch(1);
95
96 jit.loadPtr(CCallHelpers::Address(document, JSDocument::offsetOfWrapped()), scratch1);
97 DOMJIT::loadDocumentElement(jit, scratch1, scratch1);
98
99 CCallHelpers::JumpList nullCases;
100 CCallHelpers::JumpList successCases;
101 nullCases.append(jit.branchTestPtr(CCallHelpers::Zero, scratch1));
102 nullCases.append(DOMJIT::branchTestIsHTMLFlagOnNode(jit, CCallHelpers::Zero, scratch1));
103 // We ensured that the name of the given element is HTML qualified.
104 // It allows us to perform local name comparison!
105 loadLocalName(jit, scratch1, scratch2);
106 nullCases.append(jit.branchPtr(CCallHelpers::NotEqual, scratch2, CCallHelpers::TrustedImmPtr(HTMLNames::htmlTag->localName().impl())));
107
108 RELEASE_ASSERT(!CAST_OFFSET(Node*, ContainerNode*));
109 RELEASE_ASSERT(!CAST_OFFSET(Node*, Element*));
110 RELEASE_ASSERT(!CAST_OFFSET(Node*, HTMLElement*));
111
112 // Node* node = current.firstChild();
113 // while (node && !is<HTMLElement>(*node))
114 // node = node->nextSibling();
115 // return downcast<HTMLElement>(node);
116 jit.loadPtr(CCallHelpers::Address(scratch1, ContainerNode::firstChildMemoryOffset()), scratch1);
117
118 CCallHelpers::Label loopStart = jit.label();
119 nullCases.append(jit.branchTestPtr(CCallHelpers::Zero, scratch1));
120 auto notHTMLElementCase = DOMJIT::branchTestIsHTMLFlagOnNode(jit, CCallHelpers::Zero, scratch1);
121 // We ensured that the name of the given element is HTML qualified.
122 // It allows us to perform local name comparison!
123 loadLocalName(jit, scratch1, scratch2);
124 successCases.append(jit.branchPtr(CCallHelpers::Equal, scratch2, CCallHelpers::TrustedImmPtr(HTMLNames::bodyTag->localName().impl())));
125 successCases.append(jit.branchPtr(CCallHelpers::Equal, scratch2, CCallHelpers::TrustedImmPtr(HTMLNames::framesetTag->localName().impl())));
126
127 notHTMLElementCase.link(&jit);
128 jit.loadPtr(CCallHelpers::Address(scratch1, Node::nextSiblingMemoryOffset()), scratch1);
129 jit.jump().linkTo(loopStart, &jit);
130
131 successCases.link(&jit);
132 DOMJIT::toWrapper<HTMLElement>(jit, params, scratch1, globalObject, result, DOMJIT::operationToJSHTMLElement, globalObjectValue);
133 auto done = jit.jump();
134
135 nullCases.link(&jit);
136 jit.moveValue(jsNull(), result);
137 done.link(&jit);
138
139 return CCallHelpers::JumpList();
140 });
141 snippet->effect = JSC::DOMJIT::Effect::forDef(DOMJIT::AbstractHeapRepository::Document_body);
142 return snippet;
143}

148{
149 ASSERT(result);
150 ASSERT(globalObject);

153 JSC::JITOperationPrologueCallFrameTracer tracer(vm, callFrame);
154 return DOMJIT::toWrapperSlowImpl<Element>(globalObject, result);
