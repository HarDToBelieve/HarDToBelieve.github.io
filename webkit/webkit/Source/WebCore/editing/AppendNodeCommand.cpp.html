36namespace WebCore {
37
38AppendNodeCommand::AppendNodeCommand(Ref<ContainerNode>&& parent, Ref<Node>&& node, EditAction editingAction)
39 : SimpleEditCommand(parent->document(), editingAction)
40 , m_parent(WTFMove(parent))
41 , m_node(WTFMove(node))
42{
43 ASSERT(!m_node->parentNode());
44 ASSERT(m_parent->hasEditableStyle() || !m_parent->renderer());
45}
46
47void AppendNodeCommand::doApply()
48{
49 if (!m_parent->hasEditableStyle() && m_parent->renderer())
50 return;
51
52 m_parent->appendChild(m_node);
53}
54
55void AppendNodeCommand::doUnapply()
56{
57 if (!m_parent->hasEditableStyle())
58 return;
59
60 m_node->remove();
61}
62
63#ifndef NDEBUG
64void AppendNodeCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
65{
66 addNodeAndDescendants(m_parent.ptr(), nodes);
67 addNodeAndDescendants(m_node.ptr(), nodes);
68}
69#endif
