80static bool hasNoAttributeOrOnlyStyleAttribute(const StyledElement& element, ShouldStyleAttributeBeEmpty shouldStyleAttributeBeEmpty)
81{
82 if (!element.hasAttributes())
83 return true;
84
85 unsigned matchedAttributes = 0;
86 if (element.attributeWithoutSynchronization(classAttr) == styleSpanClassString())
87 matchedAttributes++;
88 if (element.hasAttribute(styleAttr) && (shouldStyleAttributeBeEmpty == AllowNonEmptyStyleAttribute
89 || !element.inlineStyle() || element.inlineStyle()->isEmpty()))
90 matchedAttributes++;
91
92 ASSERT(matchedAttributes <= element.attributeCount());
93 return matchedAttributes == element.attributeCount();
94}

203void ApplyStyleCommand::doApply()
204{
205 switch (m_propertyLevel) {
206 case PropertyDefault: {
207 // Apply the block-centric properties of the style.
208 auto blockStyle = m_style->extractAndRemoveBlockProperties();
209 if (!blockStyle->isEmpty())
210 applyBlockStyle(blockStyle);
211 // Apply any remaining styles to the inline elements.
212 if (!m_style->isEmpty() || m_styledInlineElement || m_isInlineElementToRemoveFunction) {
213 applyRelativeFontStyleChange(m_style.get());
214 applyInlineStyle(*m_style);
215 }
216 break;
217 }
218 case ForceBlockProperties:
219 // Force all properties to be applied as block styles.
220 applyBlockStyle(*m_style);
221 break;
222 }
223}

225void ApplyStyleCommand::applyBlockStyle(EditingStyle& style)
226{
227 // Update document layout once before removing styles so that we avoid the expense of
228 // updating before each and every call to check a computed style.
229 document().updateLayoutIgnorePendingStylesheets();
230
231 auto start = startPosition();
232 auto end = endPosition();
233 if (end < start)
234 std::swap(start, end);
235
236 VisiblePosition visibleStart(start);
237 VisiblePosition visibleEnd(end);
238
239 if (visibleStart.isNull() || visibleStart.isOrphan() || visibleEnd.isNull() || visibleEnd.isOrphan())
240 return;
241
242 // Save and restore the selection endpoints using their indices in the editable root, since
243 // addBlockStyleIfNeeded may moveParagraphs, which can remove these endpoints.
244 // Calculate start and end indices from the start of the tree that they're in.
245 auto scopeRoot = makeRefPtr(highestEditableRoot(visibleStart.deepEquivalent()));
246 if (!scopeRoot)
247 return;
248
249 auto scope = makeRangeSelectingNodeContents(*scopeRoot);
250 auto range = *makeSimpleRange(visibleStart, visibleEnd);
251 auto startIndex = characterCount({ scope.start, range.start }, TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions);
252 auto endIndex = characterCount({ scope.start, range.end }, TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions);
253
254 VisiblePosition paragraphStart(startOfParagraph(visibleStart));
255 VisiblePosition nextParagraphStart(endOfParagraph(paragraphStart).next());
256 if (visibleEnd != visibleStart && isStartOfParagraph(visibleEnd))
257 visibleEnd = visibleEnd.previous(CannotCrossEditingBoundary);
258 VisiblePosition beyondEnd(endOfParagraph(visibleEnd).next());
259 while (paragraphStart.isNotNull() && paragraphStart != beyondEnd) {
260 StyleChange styleChange(&style, paragraphStart.deepEquivalent());
261 if (styleChange.cssStyle() || m_removeOnly) {
262 RefPtr<Node> block = enclosingBlock(paragraphStart.deepEquivalent().deprecatedNode());
263 if (!m_removeOnly) {
264 RefPtr<Node> newBlock = moveParagraphContentsToNewBlockIfNecessary(paragraphStart.deepEquivalent());
265 if (newBlock)
266 block = newBlock;
267 }
268 ASSERT(!block || is<HTMLElement>(*block));
269 if (is<HTMLElement>(block)) {
270 removeCSSStyle(style, downcast<HTMLElement>(*block));
271 if (!m_removeOnly)
272 addBlockStyle(styleChange, downcast<HTMLElement>(*block));
273 }
274
275 if (nextParagraphStart.isOrphan())
276 nextParagraphStart = endOfParagraph(paragraphStart).next();
277 }
278
279 paragraphStart = nextParagraphStart;
280 nextParagraphStart = endOfParagraph(paragraphStart).next();
281 }
282
283 auto startPosition = makeDeprecatedLegacyPosition(resolveCharacterLocation(scope, startIndex, TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions));
284 auto endPosition = makeDeprecatedLegacyPosition(resolveCharacterLocation(scope, endIndex, TextIteratorBehavior::EmitsCharactersBetweenAllVisiblePositions));
285 updateStartEnd(startPosition, endPosition);
286}

295void ApplyStyleCommand::applyRelativeFontStyleChange(EditingStyle* style)
296{
297 static const float MinimumFontSize = 0.1f;
298
299 if (!style || !style->hasFontSizeDelta())
300 return;
301
302 auto start = startPosition();
303 auto end = endPosition();
304 if (end < start)
305 std::swap(start, end);
306
307 // Join up any adjacent text nodes.
308 if (is<Text>(start.deprecatedNode())) {
309 joinChildTextNodes(start.deprecatedNode()->parentNode(), start, end);
310 start = startPosition();
311 end = endPosition();
312 }
313
314 if (start.isNull() || end.isNull())
315 return;
316
317 if (is<Text>(*end.deprecatedNode()) && start.deprecatedNode()->parentNode() != end.deprecatedNode()->parentNode()) {
318 joinChildTextNodes(end.deprecatedNode()->parentNode(), start, end);
319 start = startPosition();
320 end = endPosition();
321 }
322
323 if (start.isNull() || end.isNull())
324 return;
325
326 // Split the start text nodes if needed to apply style.
327 if (isValidCaretPositionInTextNode(start)) {
328 splitTextAtStart(start, end);
329 start = startPosition();
330 end = endPosition();
331 }
332
333 if (start.isNull() || end.isNull())
334 return;
335
336 if (isValidCaretPositionInTextNode(end)) {
337 splitTextAtEnd(start, end);
338 start = startPosition();
339 end = endPosition();
340 }
341
342 if (start.isNull() || end.isNull())
343 return;
344
345 // Calculate loop end point.
346 // If the end node is before the start node (can only happen if the end node is
347 // an ancestor of the start node), we gather nodes up to the next sibling of the end node
348 RefPtr<Node> beyondEnd;
349 ASSERT(start.deprecatedNode());
350 ASSERT(end.deprecatedNode());
351 if (end.deprecatedNode()->contains(*start.deprecatedNode()))
352 beyondEnd = NodeTraversal::nextSkippingChildren(*end.deprecatedNode());
353 else
354 beyondEnd = NodeTraversal::next(*end.deprecatedNode());
355
356 start = start.upstream(); // Move upstream to ensure we do not add redundant spans.
357 auto startNode = makeRefPtr(start.deprecatedNode());
358
359 // Make sure we're not already at the end or the next NodeTraversal::next() will traverse past it.
360 if (startNode == beyondEnd)
361 return;
362
363 if (is<Text>(*startNode) && start.deprecatedEditingOffset() >= caretMaxOffset(*startNode)) {
364 // Move out of text node if range does not include its characters.
365 startNode = NodeTraversal::next(*startNode);
366 if (!startNode)
367 return;
368 }
369
370 // Store away font size before making any changes to the document.
371 // This ensures that changes to one node won't effect another.
372 HashMap<Ref<Node>, float> startingFontSizes;
373 for (auto node = startNode; node != beyondEnd; node = NodeTraversal::next(*node)) {
374 ASSERT(node);
375 startingFontSizes.set(*node, computedFontSize(node.get()));
376 }
377
378 // These spans were added by us. If empty after font size changes, they can be removed.
379 Vector<Ref<HTMLElement>> unstyledSpans;
380
381 RefPtr<Node> lastStyledNode;
382 bool reachedEnd = false;
383 for (auto node = startNode; node != beyondEnd && !reachedEnd; node = NodeTraversal::next(*node)) {
384 ASSERT(node);
385 RefPtr<HTMLElement> element;
386 if (is<HTMLElement>(*node)) {
387 // Only work on fully selected nodes.
388 if (!nodeFullySelected(downcast<HTMLElement>(*node), start, end))
389 continue;
390 element = &downcast<HTMLElement>(*node);
391 } else if (is<Text>(*node) && node->renderer() && node->parentNode() != lastStyledNode) {
392 // Last styled node was not parent node of this text node, but we wish to style this
393 // text node. To make this possible, add a style span to surround this text node.
394 auto span = createStyleSpanElement(document());
395 if (!surroundNodeRangeWithElement(*node, *node, span))
396 continue;
397 reachedEnd = node->isDescendantOf(beyondEnd.get());
398 element = WTFMove(span);
399 } else {
400 // Only handle HTML elements and text nodes.
401 continue;
402 }
403 lastStyledNode = node;
404
405 RefPtr<MutableStyleProperties> inlineStyle = copyStyleOrCreateEmpty(element->inlineStyle());
406 float currentFontSize = computedFontSize(node.get());
407 float desiredFontSize = std::max(MinimumFontSize, startingFontSizes.get(node.get()) + style->fontSizeDelta());
408 RefPtr<CSSValue> value = inlineStyle->getPropertyCSSValue(CSSPropertyFontSize);
409 if (value) {
410 element->removeInlineStyleProperty(CSSPropertyFontSize);
411 currentFontSize = computedFontSize(node.get());
412 }
413 if (currentFontSize != desiredFontSize) {
414 inlineStyle->setProperty(CSSPropertyFontSize, CSSValuePool::singleton().createValue(desiredFontSize, CSSUnitType::CSS_PX), false);
415 setNodeAttribute(*element, styleAttr, inlineStyle->asText());
416 }
417 if (inlineStyle->isEmpty()) {
418 removeNodeAttribute(*element, styleAttr);
419 if (isSpanWithoutAttributesOrUnstyledStyleSpan(*element))
420 unstyledSpans.append(element.releaseNonNull());
421 }
422 }
423
424 for (auto& unstyledSpan : unstyledSpans)
425 removeNodePreservingChildren(unstyledSpan);
426}

436void ApplyStyleCommand::cleanupUnstyledAppleStyleSpans(ContainerNode* dummySpanAncestor)
437{
438 if (!dummySpanAncestor)
439 return;
440
441 // Dummy spans are created when text node is split, so that style information
442 // can be propagated, which can result in more splitting. If a dummy span gets
443 // cloned/split, the new node is always a sibling of it. Therefore, we scan
444 // all the children of the dummy's parent
445
446 Vector<Ref<Element>> toRemove;
447 for (auto& child : childrenOfType<Element>(*dummySpanAncestor)) {
448 if (isSpanWithoutAttributesOrUnstyledStyleSpan(child))
449 toRemove.append(child);
450 }
451
452 for (auto& element : toRemove)
453 removeNodePreservingChildren(element.get());
454}

456RefPtr<HTMLElement> ApplyStyleCommand::splitAncestorsWithUnicodeBidi(Node* node, bool before, WritingDirection allowedDirection)
457{
458 // We are allowed to leave the highest ancestor with unicode-bidi unsplit if it is unicode-bidi: embed and direction: allowedDirection.
459 // In that case, we return the unsplit ancestor. Otherwise, we return 0.
460 auto block = makeRefPtr(enclosingBlock(node));
461 if (!block || block == node)
462 return nullptr;
463
464 RefPtr<Node> highestAncestorWithUnicodeBidi;
465 RefPtr<Node> nextHighestAncestorWithUnicodeBidi;
466 int highestAncestorUnicodeBidi = 0;
467 for (auto ancestor = makeRefPtr(node->parentNode()); ancestor != block; ancestor = ancestor->parentNode()) {
468 int unicodeBidi = toIdentifier(ComputedStyleExtractor(ancestor.get()).propertyValue(CSSPropertyUnicodeBidi));
469 if (unicodeBidi && unicodeBidi != CSSValueNormal) {
470 highestAncestorUnicodeBidi = unicodeBidi;
471 nextHighestAncestorWithUnicodeBidi = highestAncestorWithUnicodeBidi;
472 highestAncestorWithUnicodeBidi = ancestor;
473 }
474 }
475
476 if (!highestAncestorWithUnicodeBidi)
477 return nullptr;
478
479 RefPtr<HTMLElement> unsplitAncestor;
480
481 if (allowedDirection != WritingDirection::Natural && highestAncestorUnicodeBidi != CSSValueBidiOverride && is<HTMLElement>(*highestAncestorWithUnicodeBidi)) {
482 auto highestAncestorDirection = EditingStyle::create(highestAncestorWithUnicodeBidi.get(), EditingStyle::AllProperties)->textDirection();
483 if (highestAncestorDirection && *highestAncestorDirection == allowedDirection) {
484 if (!nextHighestAncestorWithUnicodeBidi)
485 return static_pointer_cast<HTMLElement>(WTFMove(highestAncestorWithUnicodeBidi));
486
487 unsplitAncestor = static_pointer_cast<HTMLElement>(highestAncestorWithUnicodeBidi);
488 highestAncestorWithUnicodeBidi = nextHighestAncestorWithUnicodeBidi;
489 }
490 }
491
492 // Split every ancestor through highest ancestor with embedding.
493 RefPtr<Node> currentNode = node;
494 while (currentNode) {
495 RefPtr<Element> parent = downcast<Element>(currentNode->parentNode());
496 if (before ? currentNode->previousSibling() : currentNode->nextSibling())
497 splitElement(*parent, before ? *currentNode : *currentNode->nextSibling());
498 if (parent == highestAncestorWithUnicodeBidi)
499 break;
500 currentNode = parent;
501 }
502
503 return unsplitAncestor;
504}

506void ApplyStyleCommand::removeEmbeddingUpToEnclosingBlock(Node* node, Node* unsplitAncestor)
507{
508 auto block = makeRefPtr(enclosingBlock(node));
509 if (!block || block == node)
510 return;
511
512 for (RefPtr<Node> ancestor = node->parentNode(), parent; ancestor != block && ancestor != unsplitAncestor; ancestor = parent) {
513 parent = ancestor->parentNode();
514 if (!is<StyledElement>(*ancestor))
515 continue;
516
517 StyledElement& element = downcast<StyledElement>(*ancestor);
518 int unicodeBidi = toIdentifier(ComputedStyleExtractor(&element).propertyValue(CSSPropertyUnicodeBidi));
519 if (!unicodeBidi || unicodeBidi == CSSValueNormal)
520 continue;
521
522 // FIXME: This code should really consider the mapped attribute 'dir', the inline style declaration,
523 // and all matching style rules in order to determine how to best set the unicode-bidi property to 'normal'.
524 // For now, it assumes that if the 'dir' attribute is present, then removing it will suffice, and
525 // otherwise it sets the property in the inline style declaration.
526 if (element.hasAttributeWithoutSynchronization(dirAttr)) {
527 // FIXME: If this is a BDO element, we should probably just remove it if it has no
528 // other attributes, like we (should) do with B and I elements.
529 removeNodeAttribute(element, dirAttr);
530 } else {
531 auto inlineStyle = copyStyleOrCreateEmpty(element.inlineStyle());
532 inlineStyle->setProperty(CSSPropertyUnicodeBidi, CSSValueNormal);
533 inlineStyle->removeProperty(CSSPropertyDirection);
534 setNodeAttribute(element, styleAttr, inlineStyle->asText());
535 if (isSpanWithoutAttributesOrUnstyledStyleSpan(element))
536 removeNodePreservingChildren(element);
537 }
538 }
539}

551void ApplyStyleCommand::applyInlineStyle(EditingStyle& style)
552{
553 RefPtr<ContainerNode> startDummySpanAncestor;
554 RefPtr<ContainerNode> endDummySpanAncestor;
555
556 // update document layout once before removing styles
557 // so that we avoid the expense of updating before each and every call
558 // to check a computed style
559 document().updateLayoutIgnorePendingStylesheets();
560
561 // adjust to the positions we want to use for applying style
562 auto start = startPosition();
563 auto end = endPosition();
564 if (end < start)
565 std::swap(start, end);
566
567 // split the start node and containing element if the selection starts inside of it
568 bool splitStart = isValidCaretPositionInTextNode(start);
569 if (splitStart) {
570 if (shouldSplitTextElement(start.deprecatedNode()->parentElement(), style))
571 splitTextElementAtStart(start, end);
572 else
573 splitTextAtStart(start, end);
574 start = startPosition();
575 end = endPosition();
576 startDummySpanAncestor = dummySpanAncestorForNode(start.deprecatedNode());
577 }
578
579 // split the end node and containing element if the selection ends inside of it
580 bool splitEnd = isValidCaretPositionInTextNode(end);
581 if (splitEnd) {
582 if (shouldSplitTextElement(end.deprecatedNode()->parentElement(), style))
583 splitTextElementAtEnd(start, end);
584 else
585 splitTextAtEnd(start, end);
586 start = startPosition();
587 end = endPosition();
588 endDummySpanAncestor = dummySpanAncestorForNode(end.deprecatedNode());
589 }
590
591 if (start.isNull() || end.isNull())
592 return;
593
594 // Remove style from the selection.
595 // Use the upstream position of the start for removing style.
596 // This will ensure we remove all traces of the relevant styles from the selection
597 // and prevent us from adding redundant ones, as described in:
598 // <rdar://problem/3724344> Bolding and unbolding creates extraneous tags
599 Position removeStart = start.upstream();
600 auto textDirection = style.textDirection();
601 RefPtr<EditingStyle> styleWithoutEmbedding;
602 RefPtr<EditingStyle> embeddingStyle;
603 if (textDirection) {
604 // Leave alone an ancestor that provides the desired single level embedding, if there is one.
605 auto startUnsplitAncestor = splitAncestorsWithUnicodeBidi(start.deprecatedNode(), true, *textDirection);
606 auto endUnsplitAncestor = splitAncestorsWithUnicodeBidi(end.deprecatedNode(), false, *textDirection);
607 removeEmbeddingUpToEnclosingBlock(start.deprecatedNode(), startUnsplitAncestor.get());
608 removeEmbeddingUpToEnclosingBlock(end.deprecatedNode(), endUnsplitAncestor.get());
609
610 // Avoid removing the dir attribute and the unicode-bidi and direction properties from the unsplit ancestors.
611 Position embeddingRemoveStart = removeStart;
612 if (startUnsplitAncestor && nodeFullySelected(*startUnsplitAncestor, removeStart, end))
613 embeddingRemoveStart = positionInParentAfterNode(startUnsplitAncestor.get());
614
615 Position embeddingRemoveEnd = end;
616 if (endUnsplitAncestor && nodeFullySelected(*endUnsplitAncestor, removeStart, end))
617 embeddingRemoveEnd = positionInParentBeforeNode(endUnsplitAncestor.get()).downstream();
618
619 if (embeddingRemoveEnd != removeStart || embeddingRemoveEnd != end) {
620 styleWithoutEmbedding = style.copy();
621 embeddingStyle = styleWithoutEmbedding->extractAndRemoveTextDirection();
622
623 if (embeddingRemoveStart <= embeddingRemoveEnd)
624 removeInlineStyle(*embeddingStyle, embeddingRemoveStart, embeddingRemoveEnd);
625 }
626 }
627
628 removeInlineStyle(styleWithoutEmbedding ? *styleWithoutEmbedding : style, removeStart, end);
629 start = startPosition();
630 end = endPosition();
631 if (start.isNull() || start.isOrphan() || end.isNull() || end.isOrphan())
632 return;
633
634 if (splitStart && mergeStartWithPreviousIfIdentical(start, end)) {
635 start = startPosition();
636 end = endPosition();
637 }
638
639 if (splitEnd) {
640 mergeEndWithNextIfIdentical(start, end);
641 start = startPosition();
642 end = endPosition();
643 }
644
645 if (start.isNull() || end.isNull())
646 return;
647
648 // update document layout once before running the rest of the function
649 // so that we avoid the expense of updating before each and every call
650 // to check a computed style
651 document().updateLayoutIgnorePendingStylesheets();
652
653 RefPtr<EditingStyle> styleToApply = &style;
654 if (textDirection) {
655 // Avoid applying the unicode-bidi and direction properties beneath ancestors that already have them.
656 auto embeddingStartNode = highestEmbeddingAncestor(start.deprecatedNode(), enclosingBlock(start.deprecatedNode()));
657 auto embeddingEndNode = highestEmbeddingAncestor(end.deprecatedNode(), enclosingBlock(end.deprecatedNode()));
658
659 if (embeddingStartNode || embeddingEndNode) {
660 Position embeddingApplyStart = embeddingStartNode ? positionInParentAfterNode(embeddingStartNode.get()) : start;
661 Position embeddingApplyEnd = embeddingEndNode ? positionInParentBeforeNode(embeddingEndNode.get()) : end;
662 ASSERT(embeddingApplyStart.isNotNull() && embeddingApplyEnd.isNotNull());
663
664 if (!embeddingStyle) {
665 styleWithoutEmbedding = style.copy();
666 embeddingStyle = styleWithoutEmbedding->extractAndRemoveTextDirection();
667 }
668 fixRangeAndApplyInlineStyle(*embeddingStyle, embeddingApplyStart, embeddingApplyEnd);
669
670 styleToApply = styleWithoutEmbedding;
671 }
672 }
673
674 fixRangeAndApplyInlineStyle(*styleToApply, start, end);
675
676 // Remove dummy style spans created by splitting text elements.
677 cleanupUnstyledAppleStyleSpans(startDummySpanAncestor.get());
678 if (endDummySpanAncestor != startDummySpanAncestor)
679 cleanupUnstyledAppleStyleSpans(endDummySpanAncestor.get());
680}

682void ApplyStyleCommand::fixRangeAndApplyInlineStyle(EditingStyle& style, const Position& start, const Position& end)
683{
684 auto startNode = makeRefPtr(start.deprecatedNode());
685
686 if (start.deprecatedEditingOffset() >= caretMaxOffset(*startNode)) {
687 startNode = NodeTraversal::next(*startNode);
688 if (!startNode || end < firstPositionInOrBeforeNode(startNode.get()))
689 return;
690 }
691
692 auto pastEndNode = makeRefPtr(end.deprecatedNode());
693 if (end.deprecatedEditingOffset() >= caretMaxOffset(*pastEndNode))
694 pastEndNode = NodeTraversal::nextSkippingChildren(*pastEndNode);
695
696 // FIXME: Callers should perform this operation on a Range that includes the br
697 // if they want style applied to the empty line.
698 // FIXME: Should this be using startNode instead of start.deprecatedNode()?
699 if (start == end && start.deprecatedNode()->hasTagName(brTag))
700 pastEndNode = NodeTraversal::next(*start.deprecatedNode());
701
702 // Start from the highest fully selected ancestor so that we can modify the fully selected node.
703 // e.g. When applying font-size: large on <font color="blue">hello</font>, we need to include the font element in our run
704 // to generate <font color="blue" size="4">hello</font> instead of <font color="blue"><font size="4">hello</font></font>
705 auto range = *makeSimpleRange(start, end);
706 auto editableRoot = makeRefPtr(startNode->rootEditableElement());
707 if (startNode != editableRoot) {
708 while (editableRoot && startNode->parentNode() != editableRoot && isNodeVisiblyContainedWithin(*startNode->parentNode(), range))
709 startNode = startNode->parentNode();
710 }
711
712 applyInlineStyleToNodeRange(style, *startNode, pastEndNode.get());
713}

716{

751void ApplyStyleCommand::applyInlineStyleToNodeRange(EditingStyle& style, Node& startNode, Node* pastEndNode)
752{
753 if (m_removeOnly)
754 return;
755
756 document().updateLayoutIgnorePendingStylesheets();
757
758 Vector<InlineRunToApplyStyle> runs;
759 RefPtr<Node> node = &startNode;
760 for (RefPtr<Node> next; node && node != pastEndNode; node = next) {
761 next = NodeTraversal::next(*node);
762
763 if (!node->renderer() || !node->hasEditableStyle())
764 continue;
765
766 if (!node->hasRichlyEditableStyle() && is<HTMLElement>(*node)) {
767 // This is a plaintext-only region. Only proceed if it's fully selected.
768 // pastEndNode is the node after the last fully selected node, so if it's inside node then
769 // node isn't fully selected.
770 if (pastEndNode && pastEndNode->isDescendantOf(*node))
771 break;
772 // Add to this element's inline style and skip over its contents.
773 HTMLElement& element = downcast<HTMLElement>(*node);
774 RefPtr<MutableStyleProperties> inlineStyle = copyStyleOrCreateEmpty(element.inlineStyle());
775 if (auto otherStyle = makeRefPtr(style.style()))
776 inlineStyle->mergeAndOverrideOnConflict(*otherStyle);
777 setNodeAttribute(element, styleAttr, inlineStyle->asText());
778 next = NodeTraversal::nextSkippingChildren(*node);
779 continue;
780 }
781
782 if (isBlock(node.get()))
783 continue;
784
785 if (node->hasChildNodes()) {
786 if (node->contains(pastEndNode) || containsNonEditableRegion(*node) || !node->parentNode()->hasEditableStyle())
787 continue;
788 if (editingIgnoresContent(*node)) {
789 next = NodeTraversal::nextSkippingChildren(*node);
790 continue;
791 }
792 }
793
794 auto runStart = node;
795 auto runEnd = node;
796 auto sibling = makeRefPtr(node->nextSibling());
797 while (sibling && sibling != pastEndNode && !sibling->contains(pastEndNode) && (!isBlock(sibling.get()) || sibling->hasTagName(brTag)) && !containsNonEditableRegion(*sibling)) {
798 runEnd = sibling;
799 sibling = runEnd->nextSibling();
800 }
801 next = NodeTraversal::nextSkippingChildren(*runEnd);
802
803 auto pastEndNode = makeRefPtr(NodeTraversal::nextSkippingChildren(*runEnd));
804 if (!shouldApplyInlineStyleToRun(style, runStart.get(), pastEndNode.get()))
805 continue;
806
807 runs.append(InlineRunToApplyStyle(runStart.get(), runEnd.get(), pastEndNode.get()));
808 }
809
810 for (auto& run : runs) {
811 removeConflictingInlineStyleFromRun(style, run.start, run.end, run.pastEndNode.get());
812 if (run.startAndEndAreStillInDocument())
813 run.positionForStyleComputation = positionToComputeInlineStyleChange(*run.start, run.dummyElement);
814 }
815
816 document().updateLayoutIgnorePendingStylesheets();
817
818 for (auto& run : runs)
819 run.change = StyleChange(&style, run.positionForStyleComputation);
820
821 for (auto& run : runs) {
822 if (run.dummyElement)
823 removeNode(*run.dummyElement);
824 if (run.startAndEndAreStillInDocument())
825 applyInlineStyleChange(run.start.releaseNonNull(), run.end.releaseNonNull(), run.change, AddStyledElement);
826 }
827}

880{

882 return false;
883
884 if (isStyledInlineElementToRemove(&element)) {
885 if (mode == RemoveNone)
886 return true;
887 if (extractedStyle)
888 extractedStyle->mergeInlineStyleOfElement(element, EditingStyle::OverrideValues);
889 removeNodePreservingChildren(element);
890 return true;
891 }
892
893 bool removed = false;
894 if (removeImplicitlyStyledElement(style, element, mode, extractedStyle))

900 // If the node was converted to a span, the span may still contain relevant
901 // styles which must be removed (e.g. <b style='font-weight: bold'>)
902 if (removeCSSStyle(style, element, mode, extractedStyle))
903 removed = true;
904
905 return removed;

EditingStyle::ExtractMatchingStyle : EditingStyle::DoNotExtractMatchingStyle))
934 return false;

940 removeNodePreservingChildren(element);

966{
967 if (!node)
968 return nullptr;
969

975 result = static_pointer_cast<HTMLElement>(ancestor);

977 // also stop at the unsplittable element to be consistent with other UAs

985void ApplyStyleCommand::applyInlineStyleToPushDown(Node& node, EditingStyle* style)
986{
987 node.document().updateStyleIfNeeded();
988
989 if (!style || style->isEmpty() || !node.renderer() || is<HTMLIFrameElement>(node))
990 return;
991
992 RefPtr<EditingStyle> newInlineStyle = style;
993 if (is<HTMLElement>(node) && downcast<HTMLElement>(node).inlineStyle()) {
994 newInlineStyle = style->copy();
995 newInlineStyle->mergeInlineStyleOfElement(downcast<HTMLElement>(node), EditingStyle::OverrideValues);
996 }
997
998 // Since addInlineStyleIfNeeded can't add styles to block-flow render objects, add style attribute instead.
999 // FIXME: applyInlineStyleToRange should be used here instead.
1000 if ((node.renderer()->isRenderBlockFlow() || node.hasChildNodes()) && is<HTMLElement>(node)) {
1001 setNodeAttribute(downcast<HTMLElement>(node), styleAttr, newInlineStyle->style()->asText());
1002 return;
1003 }
1004
1005 {
1006 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
1007
1008 if (node.renderer()->isText() && static_cast<RenderText*>(node.renderer())->isAllCollapsibleWhitespace())
1009 return;
1010 if (node.renderer()->isBR() && !node.renderer()->style().preserveNewline())
1011 return;
1012 }
1013
1014 // We can't wrap node with the styled element here because new styled element will never be removed if we did.
1015 // If we modified the child pointer in pushDownInlineStyleAroundNode to point to new style element
1016 // then we fall into an infinite loop where we keep removing and adding styled element wrapping node.
1017 addInlineStyleIfNeeded(newInlineStyle.get(), node, node, DoNotAddStyledElement);
1018}

1020void ApplyStyleCommand::pushDownInlineStyleAroundNode(EditingStyle& style, Node* targetNode)
1021{
1022 auto highestAncestor = highestAncestorWithConflictingInlineStyle(style, targetNode);
1023 if (!highestAncestor)
1024 return;
1025
1026 // The outer loop is traversing the tree vertically from highestAncestor to targetNode
1027 RefPtr<Node> current = highestAncestor;
1028 // Along the way, styled elements that contain targetNode are removed and accumulated into elementsToPushDown.
1029 // Each child of the removed element, exclusing ancestors of targetNode, is then wrapped by clones of elements in elementsToPushDown.
1030 Vector<Ref<Element>> elementsToPushDown;
1031 while (current && current != targetNode && current->contains(targetNode)) {
1032 auto currentChildren = collectChildNodes(*current);
1033
1034 RefPtr<StyledElement> styledElement;
1035 if (is<StyledElement>(*current) && isStyledInlineElementToRemove(downcast<Element>(current.get()))) {
1036 styledElement = downcast<StyledElement>(current.get());
1037 elementsToPushDown.append(*styledElement);
1038 }
1039
1040 auto styleToPushDown = EditingStyle::create();
1041 if (is<HTMLElement>(*current))
1042 removeInlineStyleFromElement(style, downcast<HTMLElement>(*current), RemoveIfNeeded, styleToPushDown.ptr());
1043
1044 // The inner loop will go through children on each level
1045 // FIXME: we should aggregate inline child elements together so that we don't wrap each child separately.
1046 for (Ref<Node>& childRef : currentChildren) {
1047 Node& child = childRef;
1048 if (!child.parentNode())
1049 continue;
1050 if (!child.contains(targetNode) && elementsToPushDown.size()) {
1051 for (auto& element : elementsToPushDown) {
1052 auto wrapper = element->cloneElementWithoutChildren(document());
1053 wrapper->removeAttribute(styleAttr);
1054 surroundNodeRangeWithElement(child, child, WTFMove(wrapper));
1055 }
1056 }
1057
1058 // Apply style to all nodes containing targetNode and their siblings but NOT to targetNode
1059 // But if we've removed styledElement then always apply the style.
1060 if (&child != targetNode || styledElement)
1061 applyInlineStyleToPushDown(child, styleToPushDown.ptr());
1062
1063 // We found the next node for the outer loop (contains targetNode)
1064 // When reached targetNode, stop the outer loop upon the completion of the current inner loop
1065 if (&child == targetNode || child.contains(targetNode))
1066 current = &child;
1067 }
1068 }
1069}

1071void ApplyStyleCommand::removeInlineStyle(EditingStyle& style, const Position& start, const Position& end)
1072{
1073 ASSERT(start.isNotNull());
1074 ASSERT(end.isNotNull());
1075 ASSERT(start.anchorNode()->isConnected());
1076 ASSERT(end.anchorNode()->isConnected());
1077 ASSERT(start <= end);
1078 // FIXME: We should assert that start/end are not in the middle of a text node.
1079
1080 Position pushDownStart = start.downstream();
1081 // If the pushDownStart is at the end of a text node, then this node is not fully selected.
1082 // Move it to the next deep quivalent position to avoid removing the style from this node.
1083 // e.g. if pushDownStart was at Position("hello", 5) in <b>hello<div>world</div></b>, we want Position("world", 0) instead.
1084 auto pushDownStartContainer = makeRefPtr(pushDownStart.containerNode());
1085 if (is<Text>(pushDownStartContainer) && static_cast<unsigned>(pushDownStart.computeOffsetInContainerNode()) == downcast<Text>(*pushDownStartContainer).length())
1086 pushDownStart = nextVisuallyDistinctCandidate(pushDownStart);
1087 // If pushDownEnd is at the start of a text node, then this node is not fully selected.
1088 // Move it to the previous deep equivalent position to avoid removing the style from this node.
1089 Position pushDownEnd = end.upstream();
1090 auto pushDownEndContainer = makeRefPtr(pushDownEnd.containerNode());
1091 if (is<Text>(pushDownEndContainer) && !pushDownEnd.computeOffsetInContainerNode())
1092 pushDownEnd = previousVisuallyDistinctCandidate(pushDownEnd);
1093
1094 pushDownInlineStyleAroundNode(style, pushDownStart.deprecatedNode());
1095 pushDownInlineStyleAroundNode(style, pushDownEnd.deprecatedNode());
1096
1097 // The s and e variables store the positions used to set the ending selection after style removal
1098 // takes place. This will help callers to recognize when either the start node or the end node
1099 // are removed from the document during the work of this function.
1100 // If pushDownInlineStyleAroundNode has pruned start.deprecatedNode() or end.deprecatedNode(),
1101 // use pushDownStart or pushDownEnd instead, which pushDownInlineStyleAroundNode won't prune.
1102 Position s = start.isNull() || start.isOrphan() ? pushDownStart : start;
1103 Position e = end.isNull() || end.isOrphan() ? pushDownEnd : end;
1104
1105 RefPtr<Node> node = start.deprecatedNode();
1106 while (node) {
1107 RefPtr<Node> next;
1108 if (editingIgnoresContent(*node)) {
1109 ASSERT(node == end.deprecatedNode() || !node->contains(end.deprecatedNode()));
1110 next = NodeTraversal::nextSkippingChildren(*node);
1111 } else
1112 next = NodeTraversal::next(*node);
1113
1114 if (is<HTMLElement>(*node) && nodeFullySelected(downcast<HTMLElement>(*node), start, end)) {
1115 Ref<HTMLElement> element = downcast<HTMLElement>(*node);
1116 RefPtr<Node> prev = NodeTraversal::previousPostOrder(element);
1117 RefPtr<Node> next = NodeTraversal::next(element);
1118 RefPtr<EditingStyle> styleToPushDown;
1119 RefPtr<Node> childNode;
1120 if (isStyledInlineElementToRemove(element.ptr())) {
1121 styleToPushDown = EditingStyle::create();
1122 childNode = element->firstChild();
1123 }
1124
1125 removeInlineStyleFromElement(style, element, RemoveIfNeeded, styleToPushDown.get());
1126 if (!element->isConnected()) {
1127 if (s.deprecatedNode() == element.ptr()) {
1128 // Since elem must have been fully selected, and it is at the start
1129 // of the selection, it is clear we can set the new s offset to 0.
1130 ASSERT(s.anchorType() == Position::PositionIsBeforeAnchor || s.offsetInContainerNode() <= 0);
1131 s = firstPositionInOrBeforeNode(next.get());
1132 }
1133 if (e.deprecatedNode() == element.ptr()) {
1134 // Since elem must have been fully selected, and it is at the end
1135 // of the selection, it is clear we can set the new e offset to
1136 // the max range offset of prev.
1137 ASSERT(s.anchorType() == Position::PositionIsAfterAnchor || !offsetIsBeforeLastNodeOffset(s.offsetInContainerNode(), s.containerNode()));
1138 e = lastPositionInOrAfterNode(prev.get());
1139 }
1140 }
1141
1142 if (styleToPushDown) {
1143 for (; childNode; childNode = childNode->nextSibling())
1144 applyInlineStyleToPushDown(*childNode, styleToPushDown.get());
1145 }
1146 }
1147 if (node == end.deprecatedNode())
1148 break;
1149 node = next.get();
1150 }
1151
1152 updateStartEnd(s, e);
1153}

1314bool ApplyStyleCommand::surroundNodeRangeWithElement(Node& startNode, Node& endNode, Ref<Element>&& elementToInsert)
1315{
1316 Ref<Node> protectedStartNode = startNode;
1317 Ref<Element> element = WTFMove(elementToInsert);
1318
1319 insertNodeBefore(element.copyRef(), startNode);
1320 if (!element->isContentRichlyEditable()) {
1321 removeNode(element);
1322 return false;
1323 }
1324
1325 RefPtr<Node> node = &startNode;
1326 while (node) {
1327 RefPtr<Node> next = node->nextSibling();
1328 if (isEditableNode(*node)) {
1329 removeNode(*node);
1330 appendNode(*node, element.copyRef());
1331 }
1332 if (node == &endNode)
1333 break;
1334 node = next;
1335 }
1336
1337 RefPtr<Node> nextSibling = element->nextSibling();
1338 RefPtr<Node> previousSibling = element->previousSibling();
1339
1340 if (nextSibling && nextSibling->hasEditableStyle() && areIdenticalElements(element, *nextSibling))
1341 mergeIdenticalElements(element, downcast<Element>(*nextSibling));
1342
1343 if (is<Element>(previousSibling) && previousSibling->hasEditableStyle()) {
1344 auto mergedElement = makeRefPtr(previousSibling->nextSibling());
1345 ASSERT(mergedElement);
1346 if (mergedElement->hasEditableStyle() && areIdenticalElements(*previousSibling, *mergedElement))
1347 mergeIdenticalElements(downcast<Element>(*previousSibling), downcast<Element>(*mergedElement));
1348 }
1349
1350 // FIXME: We should probably call updateStartEnd if the start or end was in the node
1351 // range so that the endingSelection() is canonicalized. See the comments at the end of
1352 // VisibleSelection::validate().
1353 return true;

1399void ApplyStyleCommand::applyInlineStyleChange(Node& passedStart, Node& passedEnd, StyleChange& styleChange, EAddStyledElement addStyledElement)
1400{
1401 RefPtr<Node> startNode = &passedStart;
1402 RefPtr<Node> endNode = &passedEnd;
1403 ASSERT(startNode->isConnected());
1404 ASSERT(endNode->isConnected());
1405
1406 // Find appropriate font and span elements top-down.
1407 RefPtr<HTMLFontElement> fontContainer;
1408 RefPtr<HTMLElement> styleContainer;
1409 while (startNode == endNode) {
1410 if (is<HTMLElement>(*startNode)) {
1411 auto& container = downcast<HTMLElement>(*startNode);
1412 if (is<HTMLFontElement>(container))
1413 fontContainer = &downcast<HTMLFontElement>(container);
1414 if (is<HTMLSpanElement>(container) || (!is<HTMLSpanElement>(styleContainer) && container.hasChildNodes()))
1415 styleContainer = &container;
1416 }
1417 auto* startNodeFirstChild = startNode->firstChild();
1418 if (!startNodeFirstChild)
1419 break;
1420 endNode = startNode->lastChild();
1421 startNode = startNodeFirstChild;
1422 }
1423
1424 // Font tags need to go outside of CSS so that CSS font sizes override leagcy font sizes.
1425 if (styleChange.applyFontColor() || styleChange.applyFontFace() || styleChange.applyFontSize()) {
1426 if (fontContainer) {
1427 if (styleChange.applyFontColor())
1428 setNodeAttribute(*fontContainer, colorAttr, styleChange.fontColor());
1429 if (styleChange.applyFontFace())
1430 setNodeAttribute(*fontContainer, faceAttr, styleChange.fontFace());
1431 if (styleChange.applyFontSize())
1432 setNodeAttribute(*fontContainer, sizeAttr, styleChange.fontSize());
1433 } else {
1434 auto fontElement = createFontElement(document());
1435 if (styleChange.applyFontColor())
1436 fontElement->setAttributeWithoutSynchronization(colorAttr, styleChange.fontColor());
1437 if (styleChange.applyFontFace())
1438 fontElement->setAttributeWithoutSynchronization(faceAttr, styleChange.fontFace());
1439 if (styleChange.applyFontSize())
1440 fontElement->setAttributeWithoutSynchronization(sizeAttr, styleChange.fontSize());
1441 surroundNodeRangeWithElement(*startNode, *endNode, WTFMove(fontElement));
1442 }
1443 }
1444
1445 if (auto styleToMerge = styleChange.cssStyle()) {
1446 if (styleContainer) {
1447 if (auto existingStyle = styleContainer->inlineStyle()) {
1448 auto inlineStyle = EditingStyle::create(existingStyle);
1449 inlineStyle->overrideWithStyle(*styleToMerge);
1450 setNodeAttribute(*styleContainer, styleAttr, inlineStyle->style()->asText());
1451 } else
1452 setNodeAttribute(*styleContainer, styleAttr, styleToMerge->asText());
1453 } else {
1454 auto styleElement = createStyleSpanElement(document());
1455 styleElement->setAttribute(styleAttr, styleToMerge->asText());
1456 surroundNodeRangeWithElement(*startNode, *endNode, WTFMove(styleElement));
1457 }
1458 }
1459
1460 if (styleChange.applyBold())
1461 surroundNodeRangeWithElement(*startNode, *endNode, createHTMLElement(document(), bTag));
1462
1463 if (styleChange.applyItalic())
1464 surroundNodeRangeWithElement(*startNode, *endNode, createHTMLElement(document(), iTag));
1465
1466 if (styleChange.applyUnderline())
1467 surroundNodeRangeWithElement(*startNode, *endNode, createHTMLElement(document(), uTag));
1468
1469 if (styleChange.applyLineThrough())
1470 surroundNodeRangeWithElement(*startNode, *endNode, createHTMLElement(document(), strikeTag));
1471
1472 if (styleChange.applySubscript())
1473 surroundNodeRangeWithElement(*startNode, *endNode, createHTMLElement(document(), subTag));
1474 else if (styleChange.applySuperscript())
1475 surroundNodeRangeWithElement(*startNode, *endNode, createHTMLElement(document(), supTag));
1476
1477 if (m_styledInlineElement && addStyledElement == AddStyledElement)
1478 surroundNodeRangeWithElement(*startNode, *endNode, m_styledInlineElement->cloneElementWithoutChildren(document()));
1479}

1492void ApplyStyleCommand::joinChildTextNodes(Node* node, const Position& start, const Position& end)
1493{

1500 Vector<Ref<Text>> textNodes;
1501 for (Text* textNode = TextNodeTraversal::firstChild(*node); textNode; textNode = TextNodeTraversal::nextSibling(*textNode))
1502 textNodes.append(*textNode);
