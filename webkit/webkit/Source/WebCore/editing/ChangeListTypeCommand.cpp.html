40static std::optional<std::pair<ChangeListTypeCommand::Type, Ref<HTMLElement>>> listConversionTypeForSelection(const VisibleSelection& selection)
41{
42 auto startNode = selection.start().containerNode();
43 auto endNode = selection.end().containerNode();
44 if (!startNode || !endNode)
45 return { };
46 auto commonAncestor = commonInclusiveAncestor<ComposedTree>(*startNode, *endNode);
47
48 RefPtr<HTMLElement> listToReplace;
49 if (is<HTMLUListElement>(commonAncestor) || is<HTMLOListElement>(commonAncestor))
50 listToReplace = downcast<HTMLElement>(commonAncestor);
51 else
52 listToReplace = enclosingList(commonAncestor);
53
54 if (is<HTMLUListElement>(listToReplace))
55 return {{ ChangeListTypeCommand::Type::ConvertToOrderedList, listToReplace.releaseNonNull() }};
56
57 if (is<HTMLOListElement>(listToReplace))
58 return {{ ChangeListTypeCommand::Type::ConvertToUnorderedList, listToReplace.releaseNonNull() }};
59
60 return std::nullopt;
61}
62
63std::optional<ChangeListTypeCommand::Type> ChangeListTypeCommand::listConversionType(Document& document)
64{
65 if (auto frame = makeRefPtr(document.frame())) {
66 if (auto typeAndElement = listConversionTypeForSelection(frame->selection().selection()))
67 return typeAndElement->first;
68 }
69 return std::nullopt;
70}
71
72Ref<HTMLElement> ChangeListTypeCommand::createNewList(const HTMLElement& listToReplace)
73{
74 RefPtr<HTMLElement> list;
75 if (m_type == Type::ConvertToOrderedList)
76 list = HTMLOListElement::create(document());
77 else
78 list = HTMLUListElement::create(document());
79 list->cloneDataFromElement(listToReplace);
80 return list.releaseNonNull();
81}
82
83void ChangeListTypeCommand::doApply()
84{
85 auto typeAndElement = listConversionTypeForSelection(endingSelection());
86 if (!typeAndElement || typeAndElement->first != m_type)
87 return;
88
89 auto listToReplace = WTFMove(typeAndElement->second);
90 auto newList = createNewList(listToReplace);
91 insertNodeBefore(newList.copyRef(), listToReplace);
92 moveRemainingSiblingsToNewParent(listToReplace->firstChild(), nullptr, newList);
93 removeNode(listToReplace);
94 setEndingSelection({ Position { newList.ptr(), Position::PositionIsAfterChildren }});
95}
