26#pragma once
27
28#include "CompositeEditCommand.h"
29#include "EditAction.h"
30#include <wtf/Ref.h>
31
32namespace WebCore {
33
34class Document;
35class HTMLElement;
36
37class ChangeListTypeCommand final : public CompositeEditCommand {
38public:
39 enum class Type : uint8_t { ConvertToOrderedList, ConvertToUnorderedList };
40 static std::optional<Type> listConversionType(Document&);
41 static Ref<ChangeListTypeCommand> create(Document& document, Type type)
42 {
43 return adoptRef(*new ChangeListTypeCommand(document, type));
44 }
45
46private:
47 ChangeListTypeCommand(Document& document, Type type)
48 : CompositeEditCommand(document)
49 , m_type(type)
50 {
51 }
52
53 EditAction editingAction() const final
54 {
55 if (m_type == Type::ConvertToOrderedList)
56 return EditAction::ConvertToOrderedList;
57
58 return EditAction::ConvertToUnorderedList;
59 }
60
61 bool preservesTypingStyle() const final { return true; }
62 void doApply() final;
63
64 Ref<HTMLElement> createNewList(const HTMLElement& listToReplace);
65
66 Type m_type;
67};
68
69} // namespace WebCore
