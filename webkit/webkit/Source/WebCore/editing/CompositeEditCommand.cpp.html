152static void postTextStateChangeNotification(AXObjectCache* cache, const VisiblePosition& position, const String& deletedText, const String& insertedText)
153{
154 ASSERT(cache);
155 auto node = makeRefPtr(highestEditableRoot(position.deepEquivalent(), HasEditableAXRole));
156 if (!node)
157 return;
158 if (insertedText.length() && deletedText.length())
159 cache->postTextReplacementNotification(node.get(), AXTextEditTypeDelete, insertedText, AXTextEditTypeInsert, deletedText, position);
160 else if (deletedText.length())
161 cache->postTextStateChangeNotification(node.get(), AXTextEditTypeInsert, deletedText, position);
162 else if (insertedText.length())
163 cache->postTextStateChangeNotification(node.get(), AXTextEditTypeDelete, insertedText, position);
164}

224void EditCommandComposition::unapply()
225{
226 ASSERT(m_document);
227 RefPtr<Frame> frame = m_document->frame();
228 if (!frame)
229 return;
230
231 if (!areRootEditabledElementsConnected())
232 return;
233
234 m_replacedText.captureTextForUnapply();
235
236 // Changes to the document may have been made since the last editing operation that require a layout, as in <rdar://problem/5658603>.
237 // Low level operations, like RemoveNodeCommand, don't require a layout because the high level operations that use them perform one
238 // if one is necessary (like for the creation of VisiblePositions).
239 m_document->updateLayoutIgnorePendingStylesheets();
240#if PLATFORM(IOS_FAMILY)
241 // FIXME: Where should iPhone code deal with the composition?
242 // Since editing commands don't save/restore the composition, undoing without fixing
243 // up the composition will leave a stale, invalid composition, as in <rdar://problem/6831637>.
244 // Desktop handles this in -[WebHTMLView _updateSelectionForInputManager], but the phone
245 // goes another route.
246 m_document->editor().cancelComposition();
247#endif
248
249 auto prohibitScrollingForScope = m_document->view() ? m_document->view()->prohibitScrollingWhenChangingContentSizeForScope() : nullptr;
250 if (!m_document->editor().willUnapplyEditing(*this))
251 return;
252
253 size_t size = m_commands.size();
254 for (size_t i = size; i; --i)
255 m_commands[i - 1]->doUnapply();
256
257 m_document->editor().unappliedEditing(*this);
258
259 if (AXObjectCache::accessibilityEnabled())
260 m_replacedText.postTextStateChangeNotificationForUnapply(m_document->existingAXObjectCache());
261
262 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(m_document->selection().isNone() || m_document->selection().isConnectedToDocument());
263}

265void EditCommandComposition::reapply()
266{
267 ASSERT(m_document);
268 RefPtr<Frame> frame = m_document->frame();
269 if (!frame)
270 return;
271
272 if (!areRootEditabledElementsConnected())
273 return;
274
275 m_replacedText.captureTextForReapply();
276
277 // Changes to the document may have been made since the last editing operation that require a layout, as in <rdar://problem/5658603>.
278 // Low level operations, like RemoveNodeCommand, don't require a layout because the high level operations that use them perform one
279 // if one is necessary (like for the creation of VisiblePositions).
280 m_document->updateLayoutIgnorePendingStylesheets();
281
282 auto prohibitScrollingForScope = m_document->view() ? m_document->view()->prohibitScrollingWhenChangingContentSizeForScope() : nullptr;
283 if (!m_document->editor().willReapplyEditing(*this))
284 return;
285
286 for (auto& command : m_commands)
287 command->doReapply();
288
289 m_document->editor().reappliedEditing(*this);
290
291 if (AXObjectCache::accessibilityEnabled())
292 m_replacedText.postTextStateChangeNotificationForReapply(m_document->existingAXObjectCache());
293
294 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(m_document->selection().isNone() || m_document->selection().isConnectedToDocument());
295}

349void CompositeEditCommand::apply()
350{
351 if (!endingSelection().isContentRichlyEditable()) {
352 switch (editingAction()) {
353 case EditAction::TypingDeleteSelection:
354 case EditAction::TypingDeleteBackward:
355 case EditAction::TypingDeleteForward:
356 case EditAction::TypingDeleteWordBackward:
357 case EditAction::TypingDeleteWordForward:
358 case EditAction::TypingDeleteLineBackward:
359 case EditAction::TypingDeleteLineForward:
360 case EditAction::TypingDeletePendingComposition:
361 case EditAction::TypingDeleteFinalComposition:
362 case EditAction::TypingInsertText:
363 case EditAction::TypingInsertLineBreak:
364 case EditAction::TypingInsertParagraph:
365 case EditAction::TypingInsertPendingComposition:
366 case EditAction::TypingInsertFinalComposition:
367 case EditAction::Paste:
368 case EditAction::DeleteByDrag:
369 case EditAction::SetInlineWritingDirection:
370 case EditAction::SetBlockWritingDirection:
371 case EditAction::Cut:
372 case EditAction::Unspecified:
373 case EditAction::Insert:
374 case EditAction::InsertReplacement:
375 case EditAction::InsertFromDrop:
376 case EditAction::Delete:
377 case EditAction::Dictation:
378 break;
379 default:
380 ASSERT_NOT_REACHED();
381 return;
382 }
383 }
384 ensureComposition();
385
386 // Changes to the document may have been made since the last editing operation that require a layout, as in <rdar://problem/5658603>.
387 // Low level operations, like RemoveNodeCommand, don't require a layout because the high level operations that use them perform one
388 // if one is necessary (like for the creation of VisiblePositions).
389 document().updateLayoutIgnorePendingStylesheets();
390
391 auto prohibitScrollingForScope = document().view() ? document().view()->prohibitScrollingWhenChangingContentSizeForScope() : nullptr;
392 if (!willApplyCommand())
393 return;
394
395 {
396 EventQueueScope eventQueueScope;
397 doApply();
398 }
399
400 didApplyCommand();
401 setShouldRetainAutocorrectionIndicator(false);
402}

562void CompositeEditCommand::insertNodeAfter(Ref<Node>&& insertChild, Node& refChild)
563{
564 auto parent = makeRefPtr(refChild.parentNode());
565 if (!parent)
566 return;
567
568 ASSERT(!parent->isShadowRoot());

577void CompositeEditCommand::insertNodeAt(Ref<Node>&& insertChild, const Position& editingPosition)
578{
579 ASSERT(isEditablePosition(editingPosition));
580 // For editing positions like [table, 0], insert before the table,
581 // likewise for replaced elements, brs, etc.
582 Position p = editingPosition.parentAnchoredEquivalent();
583 auto refChild = makeRefPtr(p.deprecatedNode());
584 int offset = p.deprecatedEditingOffset();
585
586 if (canHaveChildrenForEditing(*refChild)) {
587 auto child = makeRefPtr(refChild->firstChild());
588 for (int i = 0; child && i < offset; i++)
589 child = child->nextSibling();
590 if (child)
591 insertNodeBefore(WTFMove(insertChild), *child);
592 else
593 appendNode(WTFMove(insertChild), downcast<ContainerNode>(*refChild));
594 } else if (caretMinOffset(*refChild) >= offset)
595 insertNodeBefore(WTFMove(insertChild), *refChild);
596 else if (is<Text>(*refChild) && caretMaxOffset(*refChild) > offset) {
597 splitTextNode(downcast<Text>(*refChild), offset);
598
599 // Mutation events (bug 22634) from the text node insertion may have removed the refChild
600 if (!refChild->isConnected())
601 return;
602 insertNodeBefore(WTFMove(insertChild), *refChild);
603 } else
604 insertNodeAfter(WTFMove(insertChild), *refChild);
605}

930void CompositeEditCommand::rebalanceWhitespaceOnTextSubstring(Text& textNode, int startOffset, int endOffset)
931{
932 String text = textNode.data();
933 ASSERT(!text.isEmpty());
934
935 // Set upstream and downstream to define the extent of the whitespace surrounding text[offset].
936 unsigned upstream = std::max(0, startOffset);
937 while (upstream > 0 && deprecatedIsEditingWhitespace(text[upstream - 1]))
938 upstream--;
939
940 unsigned downstream = std::max(0, endOffset);
941 while (downstream < text.length() && deprecatedIsEditingWhitespace(text[downstream]))
942 downstream++;
943
944 int length = downstream - upstream;
945 if (!length)
946 return;
947
948 VisiblePosition visibleUpstreamPos(Position(&textNode, upstream));
949 VisiblePosition visibleDownstreamPos(Position(&textNode, downstream));
950
951 String string = text.substring(upstream, length);
952 // FIXME: Because of the problem mentioned at the top of this function, we must also use nbsps at the start/end of the string because
953 // this function doesn't get all surrounding whitespace, just the whitespace in the current text node.
954 String rebalancedString = stringWithRebalancedWhitespace(string, isStartOfParagraph(visibleUpstreamPos) || !upstream,
955 isEndOfParagraph(visibleDownstreamPos) || downstream == text.length());
956
957 if (string != rebalancedString)
958 replaceTextInNodePreservingMarkers(textNode, upstream, length, rebalancedString);
959}

1004void CompositeEditCommand::deleteInsignificantText(Text& textNode, unsigned start, unsigned end)
1005{
1006 if (start >= end)
1007 return;
1008
1009 document().updateLayout();
1010
1011 bool wholeTextNodeIsEmpty = false;
1012 String str;
1013 auto determineRemovalMode = [&] {
1014 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
1015 RenderText* textRenderer = textNode.renderer();
1016 if (!textRenderer)
1017 return;
1018
1019 auto run = LayoutIntegration::firstTextRunInTextOrderFor(*textRenderer);
1020 if (!run) {
1021 wholeTextNodeIsEmpty = true;
1022 return;
1023 }
1024
1025 unsigned length = textNode.length();
1026 if (start >= length || end > length)
1027 return;
1028
1029 unsigned removed = 0;
1030 LayoutIntegration::TextRunIterator previousRun;
1031
1032 // This loop structure works to process all gaps preceding a box,
1033 // and also will look at the gap after the last box.
1034 while (previousRun || run) {
1035 unsigned gapStart = previousRun ? previousRun->end() : 0;
1036 if (end < gapStart)
1037 break; // No more chance for any intersections
1038
1039 unsigned gapEnd = run ? run->start() : length;
1040 bool indicesIntersect = start <= gapEnd && end >= gapStart;
1041 int gapLen = gapEnd - gapStart;
1042 if (indicesIntersect && gapLen > 0) {
1043 gapStart = std::max(gapStart, start);
1044 gapEnd = std::min(gapEnd, end);
1045 if (str.isNull())
1046 str = textNode.data().substring(start, end - start);
1047 // remove text in the gap
1048 str.remove(gapStart - start - removed, gapLen);
1049 removed += gapLen;
1050 }
1051
1052 previousRun = run;
1053 if (run)
1054 run.traverseNextTextRunInTextOrder();
1055 }
1056 };
1057 determineRemovalMode();
1058
1059 if (wholeTextNodeIsEmpty) {
1060 removeNode(textNode);
1061 return;
1062 }
1063
1064 if (!str.isNull()) {
1065 // Replace the text between start and end with our pruned version.
1066 if (!str.isEmpty())
1067 replaceTextInNode(textNode, start, end - start, str);
1068 else {
1069 // Assert that we are not going to delete all of the text in the node.
1070 // If we were, that should have been done above with the call to
1071 // removeNode and return.
1072 ASSERT(start > 0 || end - start < textNode.length());
1073 deleteTextFromNode(textNode, start, end - start);
1074 }
1075 }
1076}

1178// If the paragraph is not entirely within it's own block, create one and move the paragraph into
1179// it, and return that block. Otherwise return 0.
1180RefPtr<Node> CompositeEditCommand::moveParagraphContentsToNewBlockIfNecessary(const Position& pos)
1181{
1182 if (pos.isNull())
1183 return nullptr;
1184
1185 document().updateLayoutIgnorePendingStylesheets();
1186
1187 // It's strange that this function is responsible for verifying that pos has not been invalidated
1188 // by an earlier call to this function. The caller, applyBlockStyle, should do this.
1189 VisiblePosition visiblePos(pos);
1190 VisiblePosition visibleParagraphStart(startOfParagraph(visiblePos));
1191 VisiblePosition visibleParagraphEnd = endOfParagraph(visiblePos);
1192 if (visibleParagraphStart.isNull() || visibleParagraphEnd.isNull())
1193 return nullptr;
1194
1195 VisiblePosition next = visibleParagraphEnd.next();
1196 VisiblePosition visibleEnd = next.isNotNull() ? next : visibleParagraphEnd;
1197
1198 Position upstreamStart = visibleParagraphStart.deepEquivalent().upstream();
1199 Position upstreamEnd = visibleEnd.deepEquivalent().upstream();
1200
1201 // If there are no VisiblePositions in the same block as pos then
1202 // upstreamStart will be outside the paragraph
1203 if (pos < upstreamStart)
1204 return nullptr;
1205
1206 // Perform some checks to see if we need to perform work in this function.
1207 if (isBlock(upstreamStart.deprecatedNode())) {
1208 // If the block is the root editable element, always move content to a new block,
1209 // since it is illegal to modify attributes on the root editable element for editing.
1210 if (upstreamStart.deprecatedNode() == editableRootForPosition(upstreamStart)) {
1211 // If the block is the root editable element and it contains no visible content, create a new
1212 // block but don't try and move content into it, since there's nothing for moveParagraphs to move.
1213 if (!Position::hasRenderedNonAnonymousDescendantsWithHeight(downcast<RenderElement>(*upstreamStart.deprecatedNode()->renderer())))
1214 return insertNewDefaultParagraphElementAt(upstreamStart);
1215 } else if (isBlock(upstreamEnd.deprecatedNode())) {
1216 if (!upstreamEnd.deprecatedNode()->isDescendantOf(upstreamStart.deprecatedNode())) {
1217 // If the paragraph end is a descendant of paragraph start, then we need to run
1218 // the rest of this function. If not, we can bail here.
1219 return nullptr;
1220 }
1221 } else if (enclosingBlock(upstreamEnd.deprecatedNode()) != upstreamStart.deprecatedNode()) {
1222 // The visibleEnd. If it is an ancestor of the paragraph start, then
1223 // we can bail as we have a full block to work with.
1224 if (upstreamStart.deprecatedNode()->isDescendantOf(enclosingBlock(upstreamEnd.deprecatedNode())))
1225 return nullptr;
1226 } else if (isEndOfEditableOrNonEditableContent(visibleEnd)) {
1227 // At the end of the editable region. We can bail here as well.
1228 return nullptr;
1229 }
1230 }
1231
1232 // If upstreamStart is not editable, then we can bail here.
1233 if (!isEditablePosition(upstreamStart))
1234 return nullptr;
1235 auto newBlock = insertNewDefaultParagraphElementAt(upstreamStart);
1236
1237 bool endWasBr = visibleParagraphEnd.deepEquivalent().deprecatedNode()->hasTagName(brTag);
1238
1239 moveParagraphs(visibleParagraphStart, visibleParagraphEnd, VisiblePosition(firstPositionInNode(newBlock.ptr())));
1240
1241 if (newBlock->lastChild() && newBlock->lastChild()->hasTagName(brTag) && !endWasBr)
1242 removeNode(*newBlock->lastChild());
1243
1244 return newBlock;
1245}

1258// Clone the paragraph between start and end under blockElement,
1259// preserving the hierarchy up to outerNode.
1260
1261void CompositeEditCommand::cloneParagraphUnderNewElement(const Position& start, const Position& end, Node* passedOuterNode, Element* blockElement)
1262{
1263 ASSERT(start <= end);
1264
1265 // First we clone the outerNode
1266 RefPtr<Node> lastNode;
1267 RefPtr<Node> outerNode = passedOuterNode;
1268
1269 if (outerNode->isRootEditableElement()) {
1270 lastNode = blockElement;
1271 } else {
1272 lastNode = outerNode->cloneNode(isRenderedTable(outerNode.get()));
1273 appendNode(*lastNode, *blockElement);
1274 }
1275
1276 if (start.deprecatedNode() != outerNode && lastNode->isElementNode() && start.anchorNode()->isDescendantOf(outerNode.get())) {
1277 Vector<RefPtr<Node>> ancestors;
1278
1279 // Insert each node from innerNode to outerNode (excluded) in a list.
1280 for (Node* n = start.deprecatedNode(); n && n != outerNode; n = n->parentNode())
1281 ancestors.append(n);
1282
1283 // Clone every node between start.deprecatedNode() and outerBlock.
1284
1285 for (size_t i = ancestors.size(); i != 0; --i) {
1286 auto item = std::exchange(ancestors[i - 1], nullptr);
1287 auto child = item->cloneNode(isRenderedTable(item.get()));
1288 appendNode(child.copyRef(), downcast<Element>(*lastNode));
1289 lastNode = WTFMove(child);
1290 }
1291 }
1292
1293 if (!start.deprecatedNode()->isConnected() || !end.deprecatedNode()->isConnected())
1294 return;
1295
1296 // Handle the case of paragraphs with more than one node,
1297 // cloning all the siblings until end.deprecatedNode() is reached.
1298
1299 if (start.deprecatedNode() != end.deprecatedNode() && !start.deprecatedNode()->isDescendantOf(end.deprecatedNode())) {
1300 // If end is not a descendant of outerNode we need to
1301 // find the first common ancestor to increase the scope
1302 // of our nextSibling traversal.
1303 while (!end.deprecatedNode()->isDescendantOf(outerNode.get()) && outerNode->parentNode()) {
1304 outerNode = outerNode->parentNode();
1305 }
1306
1307 RefPtr<Node> startNode = start.deprecatedNode();
1308 for (RefPtr<Node> node = NodeTraversal::nextSkippingChildren(*startNode, outerNode.get()); node; node = NodeTraversal::nextSkippingChildren(*node, outerNode.get())) {
1309 // Move lastNode up in the tree as much as node was moved up in the
1310 // tree by NodeTraversal::nextSkippingChildren, so that the relative depth between
1311 // node and the original start node is maintained in the clone.
1312 while (startNode->parentNode() && lastNode->parentNode() && startNode->parentNode() != node->parentNode()) {
1313 startNode = startNode->parentNode();
1314 lastNode = lastNode->parentNode();
1315 }
1316
1317 auto clonedNode = node->cloneNode(true);
1318 insertNodeAfter(clonedNode.copyRef(), *lastNode);
1319 lastNode = WTFMove(clonedNode);
1320 if (node == end.deprecatedNode() || end.deprecatedNode()->isDescendantOf(*node))
1321 break;
1322 }
1323 }
1324}

1678// Operations use this function to avoid inserting content into an anchor when at the start or the end of
1679// that anchor, as in NSTextView.
1680// FIXME: This is only an approximation of NSTextViews insertion behavior, which varies depending on how
1681// the caret was made.
1682Position CompositeEditCommand::positionAvoidingSpecialElementBoundary(const Position& original)
1683{
1684 if (original.isNull())
1685 return original;
1686
1687 VisiblePosition visiblePos(original);
1688 auto enclosingAnchor = makeRefPtr(enclosingAnchorElement(original));
1689 Position result = original;
1690
1691 if (!enclosingAnchor)
1692 return result;
1693
1694 // Don't avoid block level anchors, because that would insert content into the wrong paragraph.
1695 if (enclosingAnchor && !isBlock(enclosingAnchor.get())) {
1696 VisiblePosition firstInAnchor(firstPositionInNode(enclosingAnchor.get()));
1697 VisiblePosition lastInAnchor(lastPositionInNode(enclosingAnchor.get()));
1698 // If visually just after the anchor, insert *inside* the anchor unless it's the last
1699 // VisiblePosition in the document, to match NSTextView.
1700 if (visiblePos == lastInAnchor) {

1702 // also avoid structural elements like lists and blocks (5142012).

1704 pushAnchorElementDown(*enclosingAnchor);
1705 enclosingAnchor = enclosingAnchorElement(original);
1706 if (!enclosingAnchor)
1707 return original;
1708 }

1712 if (lineBreakExistsAtVisiblePosition(visiblePos) && downstream.deprecatedNode()->isDescendantOf(enclosingAnchor.get()))
1713 return original;

1740// to determine if the split is necessary. Returns the last split node.
1741RefPtr<Node> CompositeEditCommand::splitTreeToNode(Node& start, Node& end, bool shouldSplitAncestor)
1742{

1744
1745 RefPtr<Node> adjustedEnd = &end;
1746 if (shouldSplitAncestor && adjustedEnd->parentNode())
1747 adjustedEnd = adjustedEnd->parentNode();
1748
1749 ASSERT(adjustedEnd);
1750 RefPtr<Node> node;
1751 for (node = &start; node && node->parentNode() != adjustedEnd; node = node->parentNode()) {
1752 if (!node->parentNode() || !is<Element>(*node->parentNode()))
1753 break;

1756 VisiblePosition positionInNode = firstPositionInOrBeforeNode(node.get());
