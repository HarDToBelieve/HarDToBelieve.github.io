28#include "Color.h"

31namespace WebCore {
32
33enum class CompositionUnderlineColor : bool { GivenColor, TextColor };
34
35struct CompositionUnderline {
36
37 CompositionUnderline()
38 {
39 }
40
41 CompositionUnderline(unsigned startOffset, unsigned endOffset, CompositionUnderlineColor compositionUnderlineColor, const Color& color, bool thick)
42 : startOffset(startOffset)
43 , endOffset(endOffset)
44 , compositionUnderlineColor(compositionUnderlineColor)
45 , color(color)
46 , thick(thick)
47 {
48 }
49
50 unsigned startOffset { 0 };
51 unsigned endOffset { 0 };
52 CompositionUnderlineColor compositionUnderlineColor { CompositionUnderlineColor::TextColor };
53 Color color;
54 bool thick { false };
55};
56
57} // namespace WebCore
