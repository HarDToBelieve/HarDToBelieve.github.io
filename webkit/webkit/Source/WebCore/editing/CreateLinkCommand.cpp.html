33namespace WebCore {
34
35CreateLinkCommand::CreateLinkCommand(Document& document, const String& url)
36 : CompositeEditCommand(document)
37 , m_url(url)
38{
39}
40
41void CreateLinkCommand::doApply()
42{
43 if (endingSelection().isNone())
44 return;
45
46 auto anchorElement = HTMLAnchorElement::create(document());
47 anchorElement->setHref(m_url);
48
49 if (endingSelection().isRange())
50 applyStyledElement(WTFMove(anchorElement));
51 else {
52 insertNodeAt(anchorElement.copyRef(), endingSelection().start());
53 appendNode(Text::create(document(), m_url), anchorElement.copyRef());
54 setEndingSelection(VisibleSelection(positionInParentBeforeNode(anchorElement.ptr()), positionInParentAfterNode(anchorElement.ptr()), Affinity::Downstream, endingSelection().isDirectional()));
55 }
56}
