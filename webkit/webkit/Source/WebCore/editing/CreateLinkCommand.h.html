26#pragma once
27
28#include "CompositeEditCommand.h"
29
30namespace WebCore {
31
32class CreateLinkCommand : public CompositeEditCommand {
33public:
34 static Ref<CreateLinkCommand> create(Document& document, const String& linkURL)
35 {
36 return adoptRef(*new CreateLinkCommand(document, linkURL));
37 }
38
39 bool isCreateLinkCommand() const override { return true; }
40
41private:
42 CreateLinkCommand(Document&, const String& linkURL);
43
44 void doApply() override;
45 EditAction editingAction() const override { return EditAction::CreateLink; }
46
47 String inputEventData() const override { return m_url; }
48
49 String m_url;
50};
51
52} // namespace WebCore
