54static bool isTableRow(const Node* node)
55{
56 return node && node->hasTagName(trTag);
57}

59static bool isTableCellEmpty(Node* cell)
60{
61 ASSERT(isTableCell(cell));
62 return VisiblePosition(firstPositionInNode(cell)) == VisiblePosition(lastPositionInNode(cell));
63}

65static bool isTableRowEmpty(Node* row)
66{
67 if (!isTableRow(row))
68 return false;
69
70 for (auto child = makeRefPtr(row->firstChild()); child; child = child->nextSibling()) {
71 if (isTableCell(child.get()) && !isTableCellEmpty(child.get()))
72 return false;
73 }
74
75 return true;
76}

78static bool isSpecialHTMLElement(const Node& node)
79{
80 ScriptDisallowedScope scriptDisallowedScope;
81
82 if (!is<HTMLElement>(node))
83 return false;
84
85 if (downcast<HTMLElement>(node).isLink())
86 return true;
87
88 auto* renderer = downcast<HTMLElement>(node).renderer();
89 if (!renderer)
90 return false;
91
92 if (renderer->style().display() == DisplayType::Table || renderer->style().display() == DisplayType::InlineTable)
93 return true;
94
95 if (renderer->style().isFloating())
96 return true;
97
98 if (renderer->style().position() != PositionType::Static)
99 return true;
100
101 return false;
102}

104static RefPtr<HTMLElement> firstInSpecialElement(const Position& position)
105{
106 auto rootEditableElement = makeRefPtr(position.rootEditableElement());
107 for (auto node = makeRefPtr(position.deprecatedNode()); node && node->rootEditableElement() == rootEditableElement; node = node->parentNode()) {
108 if (!isSpecialHTMLElement(*node))
109 continue;
110 VisiblePosition visiblePosition = position;
111 VisiblePosition firstInElement = firstPositionInOrBeforeNode(node.get());
112 if ((isRenderedTable(node.get()) && visiblePosition == firstInElement.next()) || visiblePosition == firstInElement) {
113 RELEASE_ASSERT(is<HTMLElement>(node));
114 return static_pointer_cast<HTMLElement>(node);
115 }
116 }
117 return nullptr;
118}
119
120static RefPtr<HTMLElement> lastInSpecialElement(const Position& position)
121{
122 auto rootEditableElement = makeRefPtr(position.rootEditableElement());
123 for (auto node = makeRefPtr(position.deprecatedNode()); node && node->rootEditableElement() == rootEditableElement; node = node->parentNode()) {
124 if (!isSpecialHTMLElement(*node))
125 continue;
126 VisiblePosition visiblePosition = position;
127 VisiblePosition lastInElement = lastPositionInOrAfterNode(node.get());
128 if ((isRenderedTable(node.get()) && visiblePosition == lastInElement.previous()) || visiblePosition == lastInElement) {
129 RELEASE_ASSERT(is<HTMLElement>(node));
130 return static_pointer_cast<HTMLElement>(node);
131 }
132 }
133 return nullptr;
134}

136static std::pair<Position, RefPtr<HTMLElement>> positionBeforeContainingSpecialElement(const Position& position)
137{
138 auto element = firstInSpecialElement(position);
139 if (!element)
140 return { position, nullptr };
141 auto result = positionInParentBeforeNode(element.get());
142 if (result.isNull() || result.containerNode()->rootEditableElement() != position.containerNode()->rootEditableElement())
143 return { position, nullptr };
144 return { result, WTFMove(element) };
145}
146
147static std::pair<Position, RefPtr<HTMLElement>> positionAfterContainingSpecialElement(const Position& position)
148{
149 auto element = lastInSpecialElement(position);
150 if (!element)
151 return { position, nullptr };
152 auto result = positionInParentAfterNode(element.get());
153 if (result.isNull() || result.deprecatedNode()->rootEditableElement() != position.containerNode()->rootEditableElement())
154 return { position, nullptr };
155 return { result, WTFMove(element) };
156}

187void DeleteSelectionCommand::initializeStartEnd(Position& start, Position& end)
188{
189 start = m_selectionToDelete.start();
190 end = m_selectionToDelete.end();
191
192 // For HRs, we'll get a position at (HR,1) when hitting delete from the beginning of the previous line, or (HR,0) when forward deleting,
193 // but in these cases, we want to delete it, so manually expand the selection
194 if (start.deprecatedNode()->hasTagName(hrTag))
195 start = positionBeforeNode(start.deprecatedNode());
196 else if (end.deprecatedNode()->hasTagName(hrTag))
197 end = positionAfterNode(end.deprecatedNode());
198
199 // FIXME: This is only used so that moveParagraphs can avoid the bugs in special element expansion.
200 if (!m_expandForSpecialElements)
201 return;
202
203 while (1) {
204 auto [startBeforeSpecialElement, startSpecialContainer] = positionBeforeContainingSpecialElement(start);
205 auto [endAfterSpecialElement, endSpecialContainer] = positionAfterContainingSpecialElement(end);
206
207 if (!startSpecialContainer && !endSpecialContainer)
208 break;
209
210 m_mergeBlocksAfterDelete = false;
211
212 if (VisiblePosition(start) != m_selectionToDelete.visibleStart() || VisiblePosition(end) != m_selectionToDelete.visibleEnd())
213 break;
214
215 // If we're going to expand to include the startSpecialContainer, it must be fully selected.
216 if (startSpecialContainer && !endSpecialContainer && positionInParentAfterNode(startSpecialContainer.get()) >= end)
217 break;
218
219 // If we're going to expand to include the endSpecialContainer, it must be fully selected.
220 if (endSpecialContainer && !startSpecialContainer && start >= positionInParentBeforeNode(endSpecialContainer.get()))
221 break;
222
223 if (startSpecialContainer && startSpecialContainer->isDescendantOf(endSpecialContainer.get())) {
224 // Don't adjust the end yet, it is the end of a special element that contains the start
225 // special element (which may or may not be fully selected).
226 start = startBeforeSpecialElement;
227 } else if (endSpecialContainer && endSpecialContainer->isDescendantOf(startSpecialContainer.get())) {
228 // Don't adjust the start yet, it is the start of a special element that contains the end
229 // special element (which may or may not be fully selected).
230 end = endAfterSpecialElement;
231 } else {
232 start = startBeforeSpecialElement;
233 end = endAfterSpecialElement;
234 }
235 }
236}

257void DeleteSelectionCommand::smartDeleteParagraphSpacers()
258{
259 VisiblePosition visibleStart { m_upstreamStart };
260 VisiblePosition visibleEnd { m_downstreamEnd };
261 bool selectionEndsInParagraphSeperator = isEndOfParagraph(visibleEnd);
262 bool selectionEndIsEndOfContent = endOfEditableContent(visibleEnd) == visibleEnd;
263 bool startAndEndInSameUnsplittableElement = unsplittableElementForPosition(visibleStart.deepEquivalent()) == unsplittableElementForPosition(visibleEnd.deepEquivalent());
264 visibleStart = visibleStart.previous(CannotCrossEditingBoundary);
265 visibleEnd = visibleEnd.next(CannotCrossEditingBoundary);
266 bool previousPositionIsStartOfContent = startOfEditableContent(visibleStart) == visibleStart;
267 bool previousPositionIsBlankParagraph = isBlankParagraph(visibleStart);
268 bool endPositionIsBlankParagraph = isBlankParagraph(visibleEnd);
269 bool hasBlankParagraphAfterEndOrIsEndOfContent = !selectionEndIsEndOfContent && (endPositionIsBlankParagraph || selectionEndsInParagraphSeperator);
270 if (startAndEndInSameUnsplittableElement && previousPositionIsBlankParagraph && hasBlankParagraphAfterEndOrIsEndOfContent) {
271 m_needPlaceholder = false;
272 Position position;
273 if (endPositionIsBlankParagraph)
274 position = startOfNextParagraph(startOfNextParagraph(m_downstreamEnd)).deepEquivalent();
275 else
276 position = VisiblePosition(m_downstreamEnd).next().deepEquivalent();
277 m_upstreamEnd = position.upstream();
278 m_downstreamEnd = position.downstream();
279 m_trailingWhitespace = m_downstreamEnd.trailingWhitespacePosition(VisiblePosition::defaultAffinity);
280 setStartingSelectionOnSmartDelete(m_upstreamStart, m_downstreamEnd);
281 }
282 if (startAndEndInSameUnsplittableElement && selectionEndIsEndOfContent && previousPositionIsBlankParagraph && selectionEndsInParagraphSeperator) {
283 m_needPlaceholder = false;
284 VisiblePosition endOfParagraphBeforeStart;
285 if (previousPositionIsStartOfContent)
286 endOfParagraphBeforeStart = endOfParagraph(VisiblePosition { m_upstreamStart }.previous());
287 else
288 endOfParagraphBeforeStart = endOfParagraph(VisiblePosition { m_upstreamStart }.previous().previous());
289 auto position = endOfParagraphBeforeStart.deepEquivalent();
290 m_upstreamStart = position.upstream();
291 m_downstreamStart = position.downstream();
292 m_leadingWhitespace = m_upstreamStart.leadingWhitespacePosition(Affinity::Downstream);
293 setStartingSelectionOnSmartDelete(m_upstreamStart, m_upstreamEnd);
294 }
295}

297bool DeleteSelectionCommand::initializePositionData()
298{
299 Position start, end;
300 initializeStartEnd(start, end);
301
302 if (!isEditablePosition(start, ContentIsEditable))
303 start = firstEditablePositionAfterPositionInRoot(start, highestEditableRoot(start));
304 if (!isEditablePosition(end, ContentIsEditable))
305 end = lastEditablePositionBeforePositionInRoot(end, highestEditableRoot(start));
306
307 if (start.isNull() || end.isNull())
308 return false;
309
310 m_upstreamStart = start.upstream();
311 m_downstreamStart = start.downstream();
312 m_upstreamEnd = end.upstream();
313 m_downstreamEnd = end.downstream();
314
315 m_startRoot = editableRootForPosition(start);
316 m_endRoot = editableRootForPosition(end);
317
318 m_startTableRow = enclosingNodeOfType(start, &isTableRow);
319 m_endTableRow = enclosingNodeOfType(end, &isTableRow);
320
321 // Don't move content out of a table cell.
322 // If the cell is non-editable, enclosingNodeOfType won't return it by default, so
323 // tell that function that we don't care if it returns non-editable nodes.
324 auto startCell = makeRefPtr(enclosingNodeOfType(m_upstreamStart, &isTableCell, CanCrossEditingBoundary));
325 auto endCell = makeRefPtr(enclosingNodeOfType(m_downstreamEnd, &isTableCell, CanCrossEditingBoundary));
326 // FIXME: This isn't right. A borderless table with two rows and a single column would appear as two paragraphs.
327 if (endCell && endCell != startCell)
328 m_mergeBlocksAfterDelete = false;
329
330 // Usually the start and the end of the selection to delete are pulled together as a result of the deletion.
331 // Sometimes they aren't (like when no merge is requested), so we must choose one position to hold the caret
332 // and receive the placeholder after deletion.
333 VisiblePosition visibleEnd(m_downstreamEnd);
334 if (m_mergeBlocksAfterDelete && !isEndOfParagraph(visibleEnd))
335 m_endingPosition = m_downstreamEnd;
336 else
337 m_endingPosition = m_downstreamStart;
338
339 // We don't want to merge into a block if it will mean changing the quote level of content after deleting
340 // selections that contain a whole number paragraphs plus a line break, since it is unclear to most users
341 // that such a selection actually ends at the start of the next paragraph. This matches TextEdit behavior
342 // for indented paragraphs.
343 // Only apply this rule if the endingSelection is a range selection. If it is a caret, then other operations have created
344 // the selection we're deleting (like the process of creating a selection to delete during a backspace), and the user isn't in the situation described above.
345 if (numEnclosingMailBlockquotes(start) != numEnclosingMailBlockquotes(end)
346 && isStartOfParagraph(visibleEnd) && isStartOfParagraph(VisiblePosition(start))
347 && endingSelection().isRange()) {
348 m_mergeBlocksAfterDelete = false;
349 m_pruneStartBlockIfNecessary = true;
350 }
351
352 // Handle leading and trailing whitespace, as well as smart delete adjustments to the selection
353 m_leadingWhitespace = m_upstreamStart.leadingWhitespacePosition(m_selectionToDelete.affinity());
354 m_trailingWhitespace = m_downstreamEnd.trailingWhitespacePosition(VisiblePosition::defaultAffinity);
355
356 if (m_smartDelete) {
357
358 // skip smart delete if the selection to delete already starts or ends with whitespace
359 Position pos = VisiblePosition(m_upstreamStart, m_selectionToDelete.affinity()).deepEquivalent();
360 bool skipSmartDelete = pos.trailingWhitespacePosition(VisiblePosition::defaultAffinity, true).isNotNull();
361 if (!skipSmartDelete)
362 skipSmartDelete = m_downstreamEnd.leadingWhitespacePosition(VisiblePosition::defaultAffinity, true).isNotNull();
363
364 // extend selection upstream if there is whitespace there
365 bool hasLeadingWhitespaceBeforeAdjustment = m_upstreamStart.leadingWhitespacePosition(m_selectionToDelete.affinity(), true).isNotNull();
366 if (!skipSmartDelete && hasLeadingWhitespaceBeforeAdjustment) {
367 auto visiblePos = VisiblePosition(m_upstreamStart).previous();
368 pos = visiblePos.deepEquivalent();
369 // Expand out one character upstream for smart delete and recalculate
370 // positions based on this change.
371 m_upstreamStart = pos.upstream();
372 m_downstreamStart = pos.downstream();
373 m_leadingWhitespace = m_upstreamStart.leadingWhitespacePosition(visiblePos.affinity());
374
375 setStartingSelectionOnSmartDelete(m_upstreamStart, m_upstreamEnd);
376 }
377
378 // trailing whitespace is only considered for smart delete if there is no leading
379 // whitespace, as in the case where you double-click the first word of a paragraph.
380 if (!skipSmartDelete && !hasLeadingWhitespaceBeforeAdjustment && m_downstreamEnd.trailingWhitespacePosition(VisiblePosition::defaultAffinity, true).isNotNull()) {
381 // Expand out one character downstream for smart delete and recalculate
382 // positions based on this change.
383 pos = VisiblePosition(m_downstreamEnd).next().deepEquivalent();
384 m_upstreamEnd = pos.upstream();
385 m_downstreamEnd = pos.downstream();
386 m_trailingWhitespace = m_downstreamEnd.trailingWhitespacePosition(VisiblePosition::defaultAffinity);
387
388 setStartingSelectionOnSmartDelete(m_downstreamStart, m_downstreamEnd);
389 }
390
391 if (shouldSmartDeleteParagraphSpacers())
392 smartDeleteParagraphSpacers();
393 }
394
395 // We must pass call parentAnchoredEquivalent on the positions since some editing positions
396 // that appear inside their nodes aren't really inside them. [hr, 0] is one example.
397 // FIXME: parentAnchoredEquivalent should eventually be moved into enclosing element getters
398 // like the one below, since editing functions should obviously accept editing positions.
399 // FIXME: Passing false to enclosingNodeOfType tells it that it's OK to return a non-editable
400 // node. This was done to match existing behavior, but it seems wrong.
401 m_startBlock = enclosingNodeOfType(m_downstreamStart.parentAnchoredEquivalent(), &isBlock, CanCrossEditingBoundary);
402 m_endBlock = enclosingNodeOfType(m_upstreamEnd.parentAnchoredEquivalent(), &isBlock, CanCrossEditingBoundary);
403
404 return true;
405}

407void DeleteSelectionCommand::saveTypingStyleState()
408{
409 // A common case is deleting characters that are all from the same text node. In
410 // that case, the style at the start of the selection before deletion will be the
411 // same as the style at the start of the selection after deletion (since those
412 // two positions will be identical). Therefore there is no need to save the
413 // typing style at the start of the selection, nor is there a reason to
414 // compute the style at the start of the selection after deletion (see the
415 // early return in calculateTypingStyleAfterDelete).
416 // However, if typing style was previously set from another text node at the previous
417 // position (now deleted), we need to clear that style as well.
418 if (m_upstreamStart.deprecatedNode() == m_downstreamEnd.deprecatedNode() && m_upstreamStart.deprecatedNode()->isTextNode()) {
419 document().selection().clearTypingStyle();
420 return;
421 }
422
423 // Figure out the typing style in effect before the delete is done.
424 m_typingStyle = EditingStyle::create(m_selectionToDelete.start(), EditingStyle::EditingPropertiesInEffect);
425 m_typingStyle->removeStyleAddedByNode(enclosingAnchorElement(m_selectionToDelete.start()));
426
427 // If we're deleting into a Mail blockquote, save the style at end() instead of start()
428 // We'll use this later in computeTypingStyleAfterDelete if we end up outside of a Mail blockquote
429 if (enclosingNodeOfType(m_selectionToDelete.start(), isMailBlockquote))
430 m_deleteIntoBlockquoteStyle = EditingStyle::create(m_selectionToDelete.end());
431 else
432 m_deleteIntoBlockquoteStyle = nullptr;
433}

435bool DeleteSelectionCommand::handleSpecialCaseBRDelete()
436{
437 auto nodeAfterUpstreamStart = makeRefPtr(m_upstreamStart.computeNodeAfterPosition());
438 auto nodeAfterDownstreamStart = makeRefPtr(m_downstreamStart.computeNodeAfterPosition());
439 // Upstream end will appear before BR due to canonicalization
440 auto nodeAfterUpstreamEnd = makeRefPtr(m_upstreamEnd.computeNodeAfterPosition());
441
442 if (!nodeAfterUpstreamStart || !nodeAfterDownstreamStart)
443 return false;
444
445 // Check for special-case where the selection contains only a BR on a line by itself after another BR.
446 bool upstreamStartIsBR = nodeAfterUpstreamStart->hasTagName(brTag);
447 bool downstreamStartIsBR = nodeAfterDownstreamStart->hasTagName(brTag);
448 // We should consider that the BR is on a line by itself also when we have <br><br>. This test should be true only
449 // when the two elements are siblings and should be false in a case like <div><br></div><br>.
450 bool isBROnLineByItself = upstreamStartIsBR && downstreamStartIsBR && ((nodeAfterDownstreamStart == nodeAfterUpstreamEnd) || (nodeAfterUpstreamEnd && nodeAfterUpstreamEnd->hasTagName(brTag) && nodeAfterUpstreamStart->nextSibling() == nodeAfterUpstreamEnd));
451
452 if (isBROnLineByItself) {
453 removeNode(*nodeAfterDownstreamStart);
454 return true;
455 }
456
457 // FIXME: This code doesn't belong in here.
458 // We detect the case where the start is an empty line consisting of BR not wrapped in a block element.
459 if (upstreamStartIsBR && downstreamStartIsBR
460 && !(isStartOfBlock(positionBeforeNode(nodeAfterUpstreamStart.get())) && isEndOfBlock(positionAfterNode(nodeAfterDownstreamStart.get())))
461 && (!nodeAfterUpstreamEnd || nodeAfterUpstreamEnd->hasTagName(brTag) || nodeAfterUpstreamEnd->previousSibling() != nodeAfterUpstreamStart)) {
462 m_startsAtEmptyLine = true;
463 m_endingPosition = m_downstreamEnd;
464 }
465
466 return false;
467}

492void DeleteSelectionCommand::removeNodeUpdatingStates(Node& node, ShouldAssumeContentIsAlwaysEditable shouldAssumeContentIsAlwaysEditable)
493{
494 if (&node == m_startBlock && !isEndOfBlock(VisiblePosition(firstPositionInNode(m_startBlock.get())).previous()))
495 m_needPlaceholder = true;
496 else if (&node == m_endBlock && !isStartOfBlock(VisiblePosition(lastPositionInNode(m_startBlock.get())).next()))
497 m_needPlaceholder = true;
498
499 // FIXME: Update the endpoints of the range being deleted.
500 updatePositionForNodeRemoval(m_endingPosition, node);
501 updatePositionForNodeRemoval(m_leadingWhitespace, node);
502 updatePositionForNodeRemoval(m_trailingWhitespace, node);
503
504 CompositeEditCommand::removeNode(node, shouldAssumeContentIsAlwaysEditable);
505}

507static inline bool shouldRemoveContentOnly(const Node& node)
508{
509 return isTableStructureNode(&node) || node.isRootEditableElement();
510}
511
512void DeleteSelectionCommand::removeNode(Node& node, ShouldAssumeContentIsAlwaysEditable shouldAssumeContentIsAlwaysEditable)
513{
514 if (!node.parentNode())
515 return;
516
517 Ref<Node> protectedNode = node;
518 if (m_startRoot != m_endRoot && !(node.isDescendantOf(m_startRoot.get()) && node.isDescendantOf(m_endRoot.get()))) {
519 // If a node is not in both the start and end editable roots, remove it only if its inside an editable region.
520 if (!node.parentNode()->hasEditableStyle()) {
521 // Don't remove non-editable atomic nodes.
522 if (!node.firstChild())
523 return;
524 // Search this non-editable region for editable regions to empty.
525 auto child = makeRefPtr(node.firstChild());
526 while (child) {
527 auto nextChild = makeRefPtr(child->nextSibling());
528 removeNode(*child, shouldAssumeContentIsAlwaysEditable);
529 // Bail if nextChild is no longer node's child.
530 if (nextChild && nextChild->parentNode() != &node)
531 return;
532 child = WTFMove(nextChild);
533 }
534
535 // Don't remove editable regions that are inside non-editable ones, just clear them.
536 return;
537 }
538 }
539
540 if (shouldRemoveContentOnly(node)) {
541 // Do not remove an element of table structure; remove its contents.
542 // Likewise for the root editable element.
543 auto child = makeRefPtr(NodeTraversal::next(node, &node));
544 while (child) {
545 if (shouldRemoveContentOnly(*child)) {
546 child = NodeTraversal::next(*child, &node);
547 continue;
548 }
549 auto nextChild = makeRefPtr(NodeTraversal::nextSkippingChildren(*child, &node));
550 removeNodeUpdatingStates(*child, shouldAssumeContentIsAlwaysEditable);
551 child = WTFMove(nextChild);
552 }
553
554 ASSERT(is<Element>(node));
555 auto& element = downcast<Element>(node);
556 document().updateLayoutIgnorePendingStylesheets();
557 // Check if we need to insert a placeholder for descendant table cells.
558 auto descendant = makeRefPtr(ElementTraversal::next(element, &element));
559 while (descendant) {
560 auto nextDescendant = makeRefPtr(ElementTraversal::next(*descendant, &element));
561 insertBlockPlaceholderForTableCellIfNeeded(*descendant);
562 descendant = WTFMove(nextDescendant);
563 }
564 insertBlockPlaceholderForTableCellIfNeeded(element);
565 return;
566 }
567 removeNodeUpdatingStates(node, shouldAssumeContentIsAlwaysEditable);
568}

570static void updatePositionForTextRemoval(Node* node, int offset, int count, Position& position)
571{
572 if (position.anchorType() != Position::PositionIsOffsetInAnchor || position.containerNode() != node)
573 return;
574
575 if (position.offsetInContainerNode() > offset + count)
576 position.moveToOffset(position.offsetInContainerNode() - count);
577 else if (position.offsetInContainerNode() > offset)
578 position.moveToOffset(offset);
579}

581void DeleteSelectionCommand::deleteTextFromNode(Text& node, unsigned offset, unsigned count)
582{
583 // FIXME: Update the endpoints of the range being deleted.
584 updatePositionForTextRemoval(&node, offset, count, m_endingPosition);
585 updatePositionForTextRemoval(&node, offset, count, m_leadingWhitespace);
586 updatePositionForTextRemoval(&node, offset, count, m_trailingWhitespace);
587 updatePositionForTextRemoval(&node, offset, count, m_downstreamEnd);
588

614void DeleteSelectionCommand::handleGeneralDelete()
615{
616 if (m_upstreamStart.isNull())
617 return;
618
619 int startOffset = m_upstreamStart.deprecatedEditingOffset();
620 auto startNode = makeRefPtr(m_upstreamStart.deprecatedNode());
621
622 makeStylingElementsDirectChildrenOfEditableRootToPreventStyleLoss();
623
624 // Never remove the start block unless it's a table, in which case we won't merge content in.
625 if (startNode == m_startBlock && !startOffset && canHaveChildrenForEditing(*startNode) && !is<HTMLTableElement>(*startNode)) {
626 startOffset = 0;
627 startNode = NodeTraversal::next(*startNode);
628 if (!startNode)
629 return;
630 }
631
632 int startNodeCaretMaxOffset = caretMaxOffset(*startNode);
633 if (startOffset >= startNodeCaretMaxOffset && is<Text>(*startNode)) {
634 Text& text = downcast<Text>(*startNode);
635 if (text.length() > static_cast<unsigned>(startNodeCaretMaxOffset))
636 deleteTextFromNode(text, startNodeCaretMaxOffset, text.length() - startNodeCaretMaxOffset);
637 }
638
639 if (startOffset >= lastOffsetForEditing(*startNode)) {
640 startNode = NodeTraversal::nextSkippingChildren(*startNode);
641 startOffset = 0;
642 }
643
644 // Done adjusting the start. See if we're all done.
645 if (!startNode)
646 return;
647
648 if (startNode == m_downstreamEnd.deprecatedNode()) {
649 if (m_downstreamEnd.deprecatedEditingOffset() - startOffset > 0) {
650 if (is<Text>(*startNode)) {
651 // in a text node that needs to be trimmed
652 Text& text = downcast<Text>(*startNode);
653 deleteTextFromNode(text, startOffset, m_downstreamEnd.deprecatedEditingOffset() - startOffset);
654 } else {
655 removeChildrenInRange(*startNode, startOffset, m_downstreamEnd.deprecatedEditingOffset());
656 m_endingPosition = m_upstreamStart;
657 }
658 }
659
660 // The selection to delete is all in one node.
661 if (!startNode->renderer() || (!startOffset && m_downstreamEnd.atLastEditingPositionForNode()))
662 removeNode(*startNode);
663 }
664 else {
665 bool startNodeWasDescendantOfEndNode = m_upstreamStart.deprecatedNode()->isDescendantOf(m_downstreamEnd.deprecatedNode());
666 // The selection to delete spans more than one node.
667 auto node = startNode.copyRef();
668
669 if (startOffset > 0) {
670 if (is<Text>(*node)) {
671 // in a text node that needs to be trimmed
672 Text& text = downcast<Text>(*startNode);
673 deleteTextFromNode(text, startOffset, text.length() - startOffset);
674 node = NodeTraversal::next(*startNode);
675 } else {
676 node = startNode->traverseToChildAt(startOffset);
677 }
678 } else if (startNode == m_upstreamEnd.deprecatedNode() && is<Text>(*startNode)) {
679 Text& text = downcast<Text>(*startNode);
680 deleteTextFromNode(text, 0, m_upstreamEnd.deprecatedEditingOffset());
681 }
682
683 // handle deleting all nodes that are completely selected
684 while (node && node != m_downstreamEnd.deprecatedNode() && !m_downstreamEnd.isNull() && !m_downstreamEnd.isOrphan()) {
685 if (firstPositionInOrBeforeNode(node.get()) >= m_downstreamEnd) {
686 // NodeTraversal::nextSkippingChildren just blew past the end position, so stop deleting
687 node = nullptr;
688 } else if (!m_downstreamEnd.deprecatedNode()->isDescendantOf(*node)) {
689 RefPtr<Node> nextNode = NodeTraversal::nextSkippingChildren(*node);
690 // if we just removed a node from the end container, update end position so the
691 // check above will work
692 updatePositionForNodeRemoval(m_downstreamEnd, *node);
693 removeNode(*node);
694 node = nextNode.get();
695 } else {
696 auto lastDescendant = makeRefPtr(node->lastDescendant());
697 if (m_downstreamEnd.deprecatedNode() == lastDescendant && m_downstreamEnd.deprecatedEditingOffset() >= caretMaxOffset(*lastDescendant)) {
698 removeNode(*node);
699 node = nullptr;
700 } else
701 node = NodeTraversal::next(*node);
702 }
703 }
704
705 if (!m_downstreamEnd.isNull() && !m_downstreamEnd.isOrphan() && m_downstreamEnd.deprecatedNode() != startNode
706 && !m_upstreamStart.deprecatedNode()->isDescendantOf(m_downstreamEnd.deprecatedNode())
707 && m_downstreamEnd.deprecatedEditingOffset() >= caretMinOffset(*m_downstreamEnd.deprecatedNode())) {
708 if (m_downstreamEnd.atLastEditingPositionForNode() && !canHaveChildrenForEditing(*m_downstreamEnd.deprecatedNode())) {
709 // The node itself is fully selected, not just its contents. Delete it.
710 removeNode(*m_downstreamEnd.deprecatedNode());
711 } else {
712 if (is<Text>(*m_downstreamEnd.deprecatedNode())) {
713 // in a text node that needs to be trimmed
714 Text& text = downcast<Text>(*m_downstreamEnd.deprecatedNode());
715 if (m_downstreamEnd.deprecatedEditingOffset() > 0)
716 deleteTextFromNode(text, 0, m_downstreamEnd.deprecatedEditingOffset());
717 // Remove children of m_downstreamEnd.deprecatedNode() that come after m_upstreamStart.
718 // Don't try to remove children if m_upstreamStart was inside m_downstreamEnd.deprecatedNode()
719 // and m_upstreamStart has been removed from the document, because then we don't
720 // know how many children to remove.
721 // FIXME: Make m_upstreamStart a position we update as we remove content, then we can
722 // always know which children to remove.
723 } else if (!(startNodeWasDescendantOfEndNode && !m_upstreamStart.anchorNode()->isConnected())) {
724 unsigned offset = 0;
725 if (m_upstreamStart.deprecatedNode()->isDescendantOf(m_downstreamEnd.deprecatedNode())) {
726 auto n = makeRefPtr(m_upstreamStart.deprecatedNode());
727 while (n && n->parentNode() != m_downstreamEnd.deprecatedNode())
728 n = n->parentNode();
729 if (n)
730 offset = n->computeNodeIndex() + 1;
731 }
732 removeChildrenInRange(*m_downstreamEnd.deprecatedNode(), offset, m_downstreamEnd.deprecatedEditingOffset());
733 m_downstreamEnd = makeDeprecatedLegacyPosition(m_downstreamEnd.deprecatedNode(), offset);
734 }
735 }
736 }
737 }

756// If a selection starts in one block and ends in another, we have to merge to bring content before the
757// start together with content after the end.
758void DeleteSelectionCommand::mergeParagraphs()
759{
760 if (!m_mergeBlocksAfterDelete) {
761 if (m_pruneStartBlockIfNecessary) {
762 // We aren't going to merge into the start block, so remove it if it's empty.
763 prune(m_startBlock.get());
764 // Removing the start block during a deletion is usually an indication that we need
765 // a placeholder, but not in this case.
766 m_needPlaceholder = false;
767 }
768 return;
769 }
770
771 // It shouldn't have been asked to both try and merge content into the start block and prune it.
772 ASSERT(!m_pruneStartBlockIfNecessary);
773
774 // FIXME: Deletion should adjust selection endpoints as it removes nodes so that we never get into this state (4099839).
775 if (m_downstreamEnd.isNull() || m_upstreamStart.isNull() || m_downstreamEnd.isOrphan() || m_upstreamStart.isOrphan())
776 return;
777
778 if (m_upstreamStart >= m_downstreamEnd)
779 return;
780
781 VisiblePosition startOfParagraphToMove(m_downstreamEnd);
782 VisiblePosition mergeDestination(m_upstreamStart);
783
784 // m_downstreamEnd's block has been emptied out by deletion. There is no content inside of it to
785 // move, so just remove it.
786 auto endBlock = makeRefPtr(enclosingBlock(m_downstreamEnd.deprecatedNode()));
787 if (!endBlock)
788 return;
789
790 if (!endBlock->contains(startOfParagraphToMove.deepEquivalent().deprecatedNode()) || !startOfParagraphToMove.deepEquivalent().deprecatedNode()) {
791 removeNode(*endBlock);
792 return;
793 }
794
795 // We need to merge into m_upstreamStart's block, but it's been emptied out and collapsed by deletion.
796 if (!mergeDestination.deepEquivalent().deprecatedNode() || !mergeDestination.deepEquivalent().deprecatedNode()->isDescendantOf(enclosingBlock(m_upstreamStart.containerNode())) || m_startsAtEmptyLine) {
797 insertNodeAt(HTMLBRElement::create(document()), m_upstreamStart);
798 mergeDestination = VisiblePosition(m_upstreamStart);
799 }
800
801 if (mergeDestination == startOfParagraphToMove)
802 return;
803
804 VisiblePosition endOfParagraphToMove = endOfParagraph(startOfParagraphToMove, CanSkipOverEditingBoundary);
805
806 if (mergeDestination == endOfParagraphToMove)
807 return;
808
809 // The rule for merging into an empty block is: only do so if its farther to the right.
810 // FIXME: Consider RTL.
811 if (!m_startsAtEmptyLine && isStartOfParagraph(mergeDestination) && startOfParagraphToMove.absoluteCaretBounds().x() > mergeDestination.absoluteCaretBounds().x()) {
812 if (mergeDestination.deepEquivalent().downstream().deprecatedNode()->hasTagName(brTag)) {
813 auto nodeToRemove = makeRefPtr(mergeDestination.deepEquivalent().downstream().deprecatedNode());
814 removeNodeAndPruneAncestors(*nodeToRemove);
815 m_endingPosition = startOfParagraphToMove.deepEquivalent();
816 return;
817 }
818 }
819
820 // Block images, tables and horizontal rules cannot be made inline with content at mergeDestination. If there is
821 // any (!isStartOfParagraph(mergeDestination)), don't merge, just move the caret to just before the selection we deleted.
822 // See https://bugs.webkit.org/show_bug.cgi?id=25439
823 if (isRenderedAsNonInlineTableImageOrHR(startOfParagraphToMove.deepEquivalent().deprecatedNode()) && !isStartOfParagraph(mergeDestination)) {
824 m_endingPosition = m_upstreamStart;
825 return;
826 }
827
828 auto range = makeSimpleRange(startOfParagraphToMove, endOfParagraphToMove);
829 if (!range)
830 return;
831 auto rangeToBeReplaced = makeSimpleRange(mergeDestination);
832 if (!rangeToBeReplaced)
833 return;
834 if (!document().editor().client()->shouldMoveRangeAfterDelete(*range, *rangeToBeReplaced))
835 return;
836
837 // moveParagraphs will insert placeholders if it removes blocks that would require their use, don't let block
838 // removals that it does cause the insertion of *another* placeholder.
839 bool needPlaceholder = m_needPlaceholder;
840 bool paragraphToMergeIsEmpty = (startOfParagraphToMove == endOfParagraphToMove);
841 moveParagraph(startOfParagraphToMove, endOfParagraphToMove, mergeDestination, false, !paragraphToMergeIsEmpty);
842 m_needPlaceholder = needPlaceholder;
843 // The endingPosition was likely clobbered by the move, so recompute it (moveParagraph selects the moved paragraph).
844
845 // FIXME (Bug 211793): endingSelection() becomes disconnected in moveParagraph
846 if (auto* anchorNode = endingSelection().start().anchorNode(); anchorNode && anchorNode->isConnected())
847 m_endingPosition = endingSelection().start();
848}

850void DeleteSelectionCommand::removePreviouslySelectedEmptyTableRows()
851{
852 // DeleteSelectionCommand::removeNode does not remove rows but only empties them in preparation for this function.
853 // Instead, DeleteSelectionCommand::removeNodeUpdatingStates is used below, which calls a raw CompositeEditCommand::removeNode and adjusts selection.
854 if (m_endTableRow && m_endTableRow->isConnected() && m_endTableRow != m_startTableRow) {
855 auto row = makeRefPtr(m_endTableRow->previousSibling());
856 while (row && row != m_startTableRow) {
857 auto previousRow = makeRefPtr(row->previousSibling());
858 if (isTableRowEmpty(row.get()))
859 removeNodeUpdatingStates(*row, DoNotAssumeContentIsAlwaysEditable);
860 row = WTFMove(previousRow);
861 }
862 }
863
864 // Remove empty rows after the start row.
865 if (m_startTableRow && m_startTableRow->isConnected() && m_startTableRow != m_endTableRow) {
866 auto row = makeRefPtr(m_startTableRow->nextSibling());
867 while (row && row != m_endTableRow) {
868 auto nextRow = makeRefPtr(row->nextSibling());
869 if (isTableRowEmpty(row.get()))
870 removeNodeUpdatingStates(*row, DoNotAssumeContentIsAlwaysEditable);
871 row = WTFMove(nextRow);
872 }
873 }
874
875 if (m_endTableRow && m_endTableRow->isConnected() && m_endTableRow != m_startTableRow) {
876 if (isTableRowEmpty(m_endTableRow.get())) {
877 // Don't remove m_endTableRow if it's where we're putting the ending selection.
878 if (!m_endingPosition.deprecatedNode()->isDescendantOf(*m_endTableRow)) {
879 // FIXME: We probably shouldn't remove m_endTableRow unless it's fully selected, even if it is empty.
880 // We'll need to start adjusting the selection endpoints during deletion to know whether or not m_endTableRow
881 // was fully selected here.
882 removeNodeUpdatingStates(*m_endTableRow, DoNotAssumeContentIsAlwaysEditable);
883 }
884 }
885 }

888void DeleteSelectionCommand::calculateTypingStyleAfterDelete()
889{
890 if (!m_typingStyle)
891 return;
892
893 // Compute the difference between the style before the delete and the style now
894 // after the delete has been done. Set this style on the frame, so other editing
895 // commands being composed with this one will work, and also cache it on the command,
896 // so the Frame::appliedEditing can set it after the whole composite command
897 // has completed.
898
899 // If we deleted into a blockquote, but are now no longer in a blockquote, use the alternate typing style
900 if (m_deleteIntoBlockquoteStyle && !enclosingNodeOfType(m_endingPosition, isMailBlockquote, CanCrossEditingBoundary))
901 m_typingStyle = m_deleteIntoBlockquoteStyle;
902 m_deleteIntoBlockquoteStyle = nullptr;
903
904 m_typingStyle->prepareToApplyAt(m_endingPosition);
905 if (m_typingStyle->isEmpty())
906 m_typingStyle = nullptr;
907 // This is where we've deleted all traces of a style but not a whole paragraph (that's handled above).
908 // In this case if we start typing, the new characters should have the same style as the just deleted ones,
909 // but, if we change the selection, come back and start typing that style should be lost. Also see
910 // preserveTypingStyle() below.
911 document().selection().setTypingStyle(m_typingStyle.copyRef());
912}

968void DeleteSelectionCommand::doApply()
969{
970 // If selection has not been set to a custom selection when the command was created,
971 // use the current ending selection.
972 if (!m_hasSelectionToDelete)
973 m_selectionToDelete = endingSelection();
974
975 if (!m_selectionToDelete.isNonOrphanedRange())
976 return;
977
978 String originalString = originalStringForAutocorrectionAtBeginningOfSelection();
979
980 // If the deletion is occurring in a text field, and we're not deleting to replace the selection, then let the frame call across the bridge to notify the form delegate.
981 if (!m_replace) {
982 if (auto textControl = makeRefPtr(enclosingTextFormControl(m_selectionToDelete.start())); textControl && textControl->focused())
983 document().editor().textWillBeDeletedInTextField(textControl.get());
984 }
985
986 // save this to later make the selection with
987 auto affinity = m_selectionToDelete.affinity();
988
989 Position downstreamEnd = m_selectionToDelete.end().downstream();
990 m_needPlaceholder = isStartOfParagraph(m_selectionToDelete.visibleStart(), CanCrossEditingBoundary)
991 && isEndOfParagraph(m_selectionToDelete.visibleEnd(), CanCrossEditingBoundary)
992 && !lineBreakExistsAtVisiblePosition(m_selectionToDelete.visibleEnd());
993 if (m_needPlaceholder) {
994 // Don't need a placeholder when deleting a selection that starts just before a table
995 // and ends inside it (we do need placeholders to hold open empty cells, but that's
996 // handled elsewhere).
997 if (auto table = makeRefPtr(isLastPositionBeforeTable(m_selectionToDelete.visibleStart()))) {
998 if (m_selectionToDelete.end().deprecatedNode()->isDescendantOf(*table))
999 m_needPlaceholder = false;
1000 }
1001 }
1002
1003
1004 // set up our state
1005 if (!initializePositionData())
1006 return;
1007
1008 // Delete any text that may hinder our ability to fixup whitespace after the delete
1009 deleteInsignificantTextDownstream(m_trailingWhitespace);
1010
1011 saveTypingStyleState();
1012
1013 // deleting just a BR is handled specially, at least because we do not
1014 // want to replace it with a placeholder BR!
1015 if (handleSpecialCaseBRDelete()) {
1016 calculateTypingStyleAfterDelete();
1017 setEndingSelection(VisibleSelection(m_endingPosition, affinity, endingSelection().isDirectional()));
1018 clearTransientState();
1019 rebalanceWhitespace();
1020 return;
1021 }
1022
1023 handleGeneralDelete();
1024
1025 fixupWhitespace();
1026
1027 mergeParagraphs();
1028
1029 removePreviouslySelectedEmptyTableRows();
1030
1031 if (m_needPlaceholder) {
1032 if (m_sanitizeMarkup)
1033 removeRedundantBlocks();
1034
1035 // FIXME (Bug 212723): m_endingPosition becomes disconnected in moveParagraph()
1036 // because it is ancestor of the deleted element and gets pruned in removeNodeAndPruneAncestors().
1037 // Either removeNodeAndPruneAncestors() should not remove ending position or we should find
1038 // a different ending position.
1039 if (!m_endingPosition.containerNode() || !m_endingPosition.containerNode()->isConnected())
1040 return;
1041 insertNodeAt(HTMLBRElement::create(document()), m_endingPosition);
1042 }
1043
1044 bool shouldRebalaceWhiteSpace = true;
1045 if (!document().editor().behavior().shouldRebalanceWhiteSpacesInSecureField()) {
1046 if (auto endNode = makeRefPtr(m_endingPosition.deprecatedNode()); is<Text>(endNode)) {
1047 auto& textNode = downcast<Text>(*endNode);
1048 ScriptDisallowedScope scriptDisallowedScope;
1049 if (textNode.length() && textNode.renderer())
1050 shouldRebalaceWhiteSpace = textNode.renderer()->style().textSecurity() == TextSecurity::None;
1051 }
1052 }
1053 if (shouldRebalaceWhiteSpace)
1054 rebalanceWhitespaceAt(m_endingPosition);
1055
1056 calculateTypingStyleAfterDelete();
1057
1058 if (!originalString.isEmpty())
1059 document().editor().deletedAutocorrectionAtPosition(m_endingPosition, originalString);
1060
1061 setEndingSelection(VisibleSelection(m_endingPosition, affinity, endingSelection().isDirectional()));
1062 clearTransientState();
1063}
