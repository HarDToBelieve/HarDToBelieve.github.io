21#pragma once
22
23#include "EditingBehaviorType.h"
24
25namespace WebCore {
26
27class EditingBehavior {
28public:
29 explicit EditingBehavior(EditingBehaviorType type)
30 : m_type(type)
31 {
32 }
33
34 // Individual functions for each case where we have more than one style of editing behavior.
35 // Create a new function for any platform difference so we can control it here.
36
37 // When extending a selection beyond the top or bottom boundary of an editable area,
38 // maintain the horizontal position on Windows but extend it to the boundary of the editable
39 // content on Mac.
40 bool shouldMoveCaretToHorizontalBoundaryWhenPastTopOrBottom() const
41 {
42 return m_type != EditingBehaviorType::Windows;
43 }
44
45 // On Windows, selections should always be considered as directional, regardless if it is
46 // mouse-based or keyboard-based.
47 bool shouldConsiderSelectionAsDirectional() const { return m_type != EditingBehaviorType::Mac && m_type != EditingBehaviorType::iOS; }
48
49 // On Mac, when revealing a selection (for example as a result of a Find operation on the Browser),
50 // content should be scrolled such that the selection gets certer aligned.
51 bool shouldCenterAlignWhenSelectionIsRevealed() const { return m_type == EditingBehaviorType::Mac || m_type == EditingBehaviorType::iOS; }
52
53 // On Mac, style is considered present when present at the beginning of selection. On other platforms,
54 // style has to be present throughout the selection.
55 bool shouldToggleStyleBasedOnStartOfSelection() const { return m_type == EditingBehaviorType::Mac || m_type == EditingBehaviorType::iOS; }
56
57 // Standard Mac behavior when extending to a boundary is grow the selection rather than leaving the base
58 // in place and moving the extent. Matches NSTextView.
59 bool shouldAlwaysGrowSelectionWhenExtendingToBoundary() const { return m_type == EditingBehaviorType::Mac || m_type == EditingBehaviorType::iOS; }
60
61 // On Mac, when processing a contextual click, the object being clicked upon should be selected.
62 bool shouldSelectOnContextualMenuClick() const { return m_type == EditingBehaviorType::Mac; }
63
64 // On Linux, should be able to get and insert spelling suggestions without selecting the misspelled word.
65 bool shouldAllowSpellingSuggestionsWithoutSelection() const
66 {
67 return m_type == EditingBehaviorType::Unix;
68 }
69
70 // On Mac and Windows, pressing backspace (when it isn't handled otherwise) should navigate back.
71 bool shouldNavigateBackOnBackspace() const
72 {
73 return m_type != EditingBehaviorType::Unix;
74 }
75
76 // On Mac, selecting backwards by word/line from the middle of a word/line, and then going
77 // forward leaves the caret back in the middle with no selection, instead of directly selecting
78 // to the other end of the line/word (Unix/Windows behavior).
79 bool shouldExtendSelectionByWordOrLineAcrossCaret() const { return m_type != EditingBehaviorType::Mac && m_type != EditingBehaviorType::iOS; }
80
81 // Based on native behavior, when using ctrl(alt)+arrow to move caret by word, ctrl(alt)+left arrow moves caret to
82 // immediately before the word in all platforms, for example, the word break positions are: "|abc |def |hij |opq".
83 // But ctrl+right arrow moves caret to "abc |def |hij |opq" on Windows and "abc| def| hij| opq|" on Mac and Linux.
84 bool shouldSkipSpaceWhenMovingRight() const { return m_type == EditingBehaviorType::Windows; }
85
86 // On iOS the last entered character in a secure filed is shown momentarily, removing and adding back the
87 // space when deleting password cause space been showed insecurely.
88 bool shouldRebalanceWhiteSpacesInSecureField() const { return m_type != EditingBehaviorType::iOS; }
89
90 bool shouldSelectBasedOnDictionaryLookup() const { return m_type == EditingBehaviorType::Mac; }
91
92 // Linux and Windows always extend selections from the extent endpoint.
93 bool shouldAlwaysExtendSelectionFromExtentEndpoint() const { return m_type != EditingBehaviorType::Mac && m_type != EditingBehaviorType::iOS; }
94
95 // On iOS, we don't want to select all the text when focusing a field. Instead, match platform behavior by going to the end of the line.
96 bool shouldMoveSelectionToEndWhenFocusingTextInput() const { return m_type == EditingBehaviorType::iOS; }
97
98 // On iOS, when smart delete is on, it is always on, and should do not additional checks (i.e. TextGranularity::WordGranularity).
99 bool shouldAlwaysSmartDelete() const { return m_type == EditingBehaviorType::iOS; }
100
101 // On iOS, we should turn on smart insert and delete and newlines around paragraphs to match UIKit behaviour.
102 bool shouldSmartInsertDeleteParagraphs() const { return m_type == EditingBehaviorType::iOS; }
103
104private:
105 EditingBehaviorType m_type;
106};
107
108} // namespace WebCore
