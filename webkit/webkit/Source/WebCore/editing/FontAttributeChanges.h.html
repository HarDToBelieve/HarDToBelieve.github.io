34class EditingStyle;

37enum class EditAction : uint8_t;
38enum class VerticalAlignChange : uint8_t { Superscript, Baseline, Subscript };

40class FontChanges {
41public:
42 template<class Encoder> void encode(Encoder&) const;
43 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, FontChanges&);
44
45 void setFontName(const String& fontName) { m_fontName = fontName; }
46 void setFontFamily(const String& fontFamily) { m_fontFamily = fontFamily; }
47 void setFontSize(double fontSize) { m_fontSize = fontSize; }
48 void setFontSizeDelta(double fontSizeDelta) { m_fontSizeDelta = fontSizeDelta; }
49 void setBold(bool bold) { m_bold = bold; }
50 void setItalic(bool italic) { m_italic = italic; }
51
52 bool isEmpty() const
53 {
54 return !m_fontName && !m_fontFamily && !m_fontSize && !m_fontSizeDelta && !m_bold && !m_italic;
55 }
56
57 WEBCORE_EXPORT Ref<EditingStyle> createEditingStyle() const;
58 Ref<MutableStyleProperties> createStyleProperties() const;
59
60private:
61 const String& platformFontFamilyNameForCSS() const;
62
63 String m_fontName;
64 String m_fontFamily;
65 std::optional<double> m_fontSize;
66 std::optional<double> m_fontSizeDelta;
67 std::optional<bool> m_bold;
68 std::optional<bool> m_italic;
69};

71class FontAttributeChanges {
72public:
73 template<class Encoder> void encode(Encoder&) const;
74 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, FontAttributeChanges&);
75
76 void setVerticalAlign(VerticalAlignChange align) { m_verticalAlign = align; }
77 void setBackgroundColor(const Color& color) { m_backgroundColor = color; }
78 void setForegroundColor(const Color& color) { m_foregroundColor = color; }
79 void setShadow(const FontShadow& shadow) { m_shadow = shadow; }
80 void setStrikeThrough(bool strikeThrough) { m_strikeThrough = strikeThrough; }
81 void setUnderline(bool underline) { m_underline = underline; }
82 void setFontChanges(const FontChanges& fontChanges) { m_fontChanges = fontChanges; }
83
84 WEBCORE_EXPORT Ref<EditingStyle> createEditingStyle() const;
85 WEBCORE_EXPORT EditAction editAction() const;
86
87private:
88 std::optional<VerticalAlignChange> m_verticalAlign;
89 std::optional<Color> m_backgroundColor;
90 std::optional<Color> m_foregroundColor;
91 std::optional<FontShadow> m_shadow;
92 std::optional<bool> m_strikeThrough;
93 std::optional<bool> m_underline;
94 FontChanges m_fontChanges;
95};
96
97template<class Encoder>
98void FontChanges::encode(Encoder& encoder) const
99{
100 ASSERT(!m_fontSize || !m_fontSizeDelta);
101 encoder << m_fontName << m_fontFamily << m_fontSize << m_fontSizeDelta << m_bold << m_italic;
102}
103
104template<class Decoder>
105bool FontChanges::decode(Decoder& decoder, FontChanges& changes)
106{
107 if (!decoder.decode(changes.m_fontName))
108 return false;
109
110 if (!decoder.decode(changes.m_fontFamily))
111 return false;
112
113 if (!decoder.decode(changes.m_fontSize))
114 return false;
115
116 if (!decoder.decode(changes.m_fontSizeDelta))
117 return false;
118
119 if (!decoder.decode(changes.m_bold))
120 return false;
121
122 if (!decoder.decode(changes.m_italic))
123 return false;
124
125 ASSERT(!changes.m_fontSize || !changes.m_fontSizeDelta);
126 return true;
127}
128
129template<class Encoder>
130void FontAttributeChanges::encode(Encoder& encoder) const
131{
132 encoder << m_verticalAlign << m_backgroundColor << m_foregroundColor << m_shadow << m_strikeThrough << m_underline << m_fontChanges;
133}
134
135template<class Decoder>
136bool FontAttributeChanges::decode(Decoder& decoder, FontAttributeChanges& changes)
137{
138 if (!decoder.decode(changes.m_verticalAlign))
139 return false;
140
141 if (!decoder.decode(changes.m_backgroundColor))
142 return false;
143
144 if (!decoder.decode(changes.m_foregroundColor))
145 return false;
146
147 if (!decoder.decode(changes.m_shadow))
148 return false;
149
150 if (!decoder.decode(changes.m_strikeThrough))
151 return false;
152
153 if (!decoder.decode(changes.m_underline))
154 return false;
155
156 if (!decoder.decode(changes.m_fontChanges))
157 return false;
158
159 return true;
160}
