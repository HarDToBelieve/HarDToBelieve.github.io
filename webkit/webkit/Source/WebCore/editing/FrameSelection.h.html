359constexpr auto FrameSelection::defaultSetSelectionOptions(EUserTriggered userTriggered) -> OptionSet<SetSelectionOption>
360{
361 OptionSet<SetSelectionOption> options { CloseTyping, ClearTypingStyle };
362 if (userTriggered == UserTriggered)
363 options.add({ RevealSelection, FireSelectEvent, IsUserTriggered });
364 return options;
365}
