26#pragma once
27
28#include "ApplyBlockElementCommand.h"
29#include "EditAction.h"
30
31namespace WebCore {
32
33class IndentOutdentCommand : public ApplyBlockElementCommand {
34public:
35 enum EIndentType { Indent, Outdent };
36 static Ref<IndentOutdentCommand> create(Document& document, EIndentType type)
37 {
38 return adoptRef(*new IndentOutdentCommand(document, type));
39 }
40
41 bool preservesTypingStyle() const override { return true; }
42
43private:
44 IndentOutdentCommand(Document&, EIndentType);
45
46 EditAction editingAction() const override { return m_typeOfAction == Indent ? EditAction::Indent : EditAction::Outdent; }
47
48 void indentRegion(const VisiblePosition&, const VisiblePosition&);
49 void outdentRegion(const VisiblePosition&, const VisiblePosition&);
50 void outdentParagraph();
51 bool tryIndentingAsListItem(const Position&, const Position&);
52 void indentIntoBlockquote(const Position&, const Position&, RefPtr<Element>&);
53
54 void formatSelection(const VisiblePosition& startOfSelection, const VisiblePosition& endOfSelection) override;
55 void formatRange(const Position& start, const Position& end, const Position& endOfSelection, RefPtr<Element>& blockquoteForNextIndent) override;
56
57 EIndentType m_typeOfAction;
58};
59
60} // namespace WebCore
