42static Node* enclosingListChild(Node* node, Node* listNode)
43{
44 Node* listChild = enclosingListChild(node);
45 while (listChild && enclosingList(listChild) != listNode)
46 listChild = enclosingListChild(listChild->parentNode());
47 return listChild;
48}

50RefPtr<HTMLElement> InsertListCommand::insertList(Document& document, Type type)
51{
52 RefPtr<InsertListCommand> insertCommand = create(document, type);
53 insertCommand->apply();
54 return insertCommand->m_listElement;
55}

57HTMLElement* InsertListCommand::fixOrphanedListChild(Node& node)
58{
59 auto parentNode = makeRefPtr(node.parentNode());
60 if (parentNode && !parentNode->hasRichlyEditableStyle())
61 return nullptr;
62
63 auto listElement = HTMLUListElement::create(document());
64 insertNodeBefore(listElement.copyRef(), node);
65 if (!listElement->hasEditableStyle())
66 return nullptr;
67
68 removeNode(node);
69 appendNode(node, listElement.copyRef());
70 m_listElement = WTFMove(listElement);
71 return m_listElement.get();
72}

74Ref<HTMLElement> InsertListCommand::mergeWithNeighboringLists(HTMLElement& list)
75{
76 Ref<HTMLElement> protectedList = list;
77 Element* previousList = list.previousElementSibling();
78 if (canMergeLists(previousList, &list))
79 mergeIdenticalElements(*previousList, list);
80
81 Element* sibling = ElementTraversal::nextSibling(list);
82 if (!is<HTMLElement>(sibling))
83 return protectedList;
84
85 Ref<HTMLElement> nextList = downcast<HTMLElement>(*sibling);
86 if (canMergeLists(&list, nextList.ptr())) {
87 mergeIdenticalElements(list, nextList);
88 return nextList;
89 }
90 return protectedList;
91}

93bool InsertListCommand::selectionHasListOfType(const VisibleSelection& selection, const QualifiedName& listTag)
94{
95 VisiblePosition start = selection.visibleStart();
96
97 if (!enclosingList(start.deepEquivalent().deprecatedNode()))
98 return false;
99
100 VisiblePosition end = startOfParagraph(selection.visibleEnd());
101 while (start.isNotNull() && start != end) {
102 Element* listNode = enclosingList(start.deepEquivalent().deprecatedNode());
103 if (!listNode || !listNode->hasTagName(listTag))
104 return false;
105 start = startOfNextParagraph(start);
106 }
107
108 return true;
109}

117void InsertListCommand::doApply()
118{
119 VisiblePosition visibleEnd = endingSelection().visibleEnd();
120 VisiblePosition visibleStart = endingSelection().visibleStart();
121
122 if (visibleEnd.isNull() || visibleStart.isNull() || !endingSelection().isContentRichlyEditable())
123 return;
124
125 // When a selection ends at the start of a paragraph, we rarely paint
126 // the selection gap before that paragraph, because there often is no gap.
127 // In a case like this, it's not obvious to the user that the selection
128 // ends "inside" that paragraph, so it would be confusing if InsertUn{Ordered}List
129 // operated on that paragraph.
130 // FIXME: We paint the gap before some paragraphs that are indented with left
131 // margin/padding, but not others. We should make the gap painting more consistent and
132 // then use a left margin/padding rule here.
133 if (visibleEnd != visibleStart && isStartOfParagraph(visibleEnd, CanSkipOverEditingBoundary)) {
134 setEndingSelection(VisibleSelection(visibleStart, visibleEnd.previous(CannotCrossEditingBoundary), endingSelection().isDirectional()));
135 if (!endingSelection().rootEditableElement())
136 return;
137 }
138
139 auto& listTag = (m_type == Type::OrderedList) ? olTag : ulTag;
140 if (endingSelection().isRange()) {
141 VisibleSelection selection = selectionForParagraphIteration(endingSelection());
142 if (selection.isRange()) {
143 VisiblePosition startOfSelection = selection.visibleStart();
144 VisiblePosition endOfSelection = selection.visibleEnd();
145 VisiblePosition startOfLastParagraph = startOfParagraph(endOfSelection, CanSkipOverEditingBoundary);
146
147 if (startOfLastParagraph.isNotNull() && startOfParagraph(startOfSelection, CanSkipOverEditingBoundary) != startOfLastParagraph) {
148 bool forceCreateList = !selectionHasListOfType(selection, listTag);
149
150 auto currentSelection = *endingSelection().firstRange();
151 VisiblePosition startOfCurrentParagraph = startOfSelection;
152 while (startOfCurrentParagraph.isNotNull() && !inSameParagraph(startOfCurrentParagraph, startOfLastParagraph, CanCrossEditingBoundary)) {
153 // doApply() may operate on and remove the last paragraph of the selection from the document
154 // if it's in the same list item as startOfCurrentParagraph. Return early to avoid an
155 // infinite loop and because there is no more work to be done.
156 // FIXME(<rdar://problem/5983974>): The endingSelection() may be incorrect here. Compute
157 // the new location of endOfSelection and use it as the end of the new selection.
158 if (!startOfLastParagraph.deepEquivalent().anchorNode()->isConnected())
159 return;
160 setEndingSelection(startOfCurrentParagraph);
161
162 // Save and restore endOfSelection and startOfLastParagraph when necessary
163 // since moveParagraph and movePragraphWithClones can remove nodes.
164 // FIXME: This is an inefficient way to keep selection alive because indexForVisiblePosition walks from
165 // the beginning of the document to the endOfSelection everytime this code is executed.
166 // But not using index is hard because there are so many ways we can lose selection inside doApplyForSingleParagraph.
167 RefPtr<ContainerNode> scope;
168 int indexForEndOfSelection = indexForVisiblePosition(endOfSelection, scope);
169 doApplyForSingleParagraph(forceCreateList, listTag, currentSelection);
170 if (endOfSelection.isNull() || endOfSelection.isOrphan() || startOfLastParagraph.isNull() || startOfLastParagraph.isOrphan()) {
171 endOfSelection = visiblePositionForIndex(indexForEndOfSelection, scope.get());
172 // If endOfSelection is null, then some contents have been deleted from the document.
173 // This should never happen and if it did, exit early immediately because we've lost the loop invariant.
174 ASSERT(endOfSelection.isNotNull());
175 if (endOfSelection.isNull())
176 return;
177 startOfLastParagraph = startOfParagraph(endOfSelection, CanSkipOverEditingBoundary);
178 }
179
180 // Fetch the start of the selection after moving the first paragraph,
181 // because moving the paragraph will invalidate the original start.
182 // We'll use the new start to restore the original selection after
183 // we modified all selected paragraphs.
184 if (startOfCurrentParagraph == startOfSelection)
185 startOfSelection = endingSelection().visibleStart();
186
187 startOfCurrentParagraph = startOfNextParagraph(endingSelection().visibleStart());
188 }
189 setEndingSelection(endOfSelection);
190 doApplyForSingleParagraph(forceCreateList, listTag, currentSelection);
191 // Fetch the end of the selection, for the reason mentioned above.
192 endOfSelection = endingSelection().visibleEnd();
193 setEndingSelection(VisibleSelection(startOfSelection, endOfSelection, endingSelection().isDirectional()));
194 return;
195 }
196 }
197 }
198
199 auto range = endingSelection().firstRange();
200 doApplyForSingleParagraph(false, listTag, *range);
201}

208void InsertListCommand::doApplyForSingleParagraph(bool forceCreateList, const HTMLQualifiedName& listTag, SimpleRange& currentSelection)
209{
210 // FIXME: This will produce unexpected results for a selection that starts just before a
211 // table and ends inside the first cell, selectionForParagraphIteration should probably
212 // be renamed and deployed inside setEndingSelection().
213 Node* selectionNode = endingSelection().start().deprecatedNode();
214 Node* listChildNode = enclosingListChild(selectionNode);
215 bool switchListType = false;
216 if (listChildNode) {
217 // Remove the list child.
218 RefPtr<HTMLElement> listNode = enclosingList(listChildNode);
219 if (!listNode) {
220 RefPtr<HTMLElement> listElement = fixOrphanedListChild(*listChildNode);
221 if (!listElement || !listElement->isConnected())
222 return;
223
224 listNode = mergeWithNeighboringLists(*listElement);
225 }
226
227 if (!listNode->hasTagName(listTag)) {
228 // listChildNode will be removed from the list and a list of type m_type will be created.
229 switchListType = true;
230 }
231
232 // If the list is of the desired type, and we are not removing the list, then exit early.
233 if (!switchListType && forceCreateList)
234 return;
235
236 // If the entire list is selected, then convert the whole list.
237 if (switchListType && isNodeVisiblyContainedWithin(*listNode, currentSelection)) {
238 bool rangeStartIsInList = visiblePositionBeforeNode(*listNode) == makeDeprecatedLegacyPosition(currentSelection.start);
239 bool rangeEndIsInList = visiblePositionAfterNode(*listNode) == makeDeprecatedLegacyPosition(currentSelection.end);
240
241 RefPtr<HTMLElement> newList = createHTMLElement(document(), listTag);
242 insertNodeBefore(*newList, *listNode);
243 if (!newList->hasEditableStyle())
244 return;
245
246 auto* firstChildInList = enclosingListChild(VisiblePosition(firstPositionInNode(listNode.get())).deepEquivalent().deprecatedNode(), listNode.get());
247 Node* outerBlock = firstChildInList && isBlockFlowElement(*firstChildInList) ? firstChildInList : listNode.get();
248
249 moveParagraphWithClones(firstPositionInNode(listNode.get()), lastPositionInNode(listNode.get()), newList.get(), outerBlock);
250
251 // Manually remove listNode because moveParagraphWithClones sometimes leaves it behind in the document.
252 // See the bug 33668 and editing/execCommand/insert-list-orphaned-item-with-nested-lists.html.
253 // FIXME: This might be a bug in moveParagraphWithClones or deleteSelection.
254 if (listNode && listNode->isConnected())
255 removeNode(*listNode);
256
257 newList = mergeWithNeighboringLists(*newList);
258
259 // Restore the start and the end of current selection if they started inside listNode
260 // because moveParagraphWithClones could have removed them.
261 if (rangeStartIsInList && newList)
262 currentSelection.start = makeBoundaryPointBeforeNodeContents(*newList);
263 if (rangeEndIsInList && newList)
264 currentSelection.end = makeBoundaryPointAfterNodeContents(*newList);
265
266 setEndingSelection(VisiblePosition(firstPositionInNode(newList.get())));
267
268 return;
269 }
270
271 unlistifyParagraph(endingSelection().visibleStart(), listNode.get(), listChildNode);
272 }
273
274 if (!listChildNode || switchListType || forceCreateList)
275 m_listElement = listifyParagraph(endingSelection().visibleStart(), listTag);
276}

278void InsertListCommand::unlistifyParagraph(const VisiblePosition& originalStart, HTMLElement* listNode, Node* listChildNode)
279{
280 Node* nextListChild;
281 Node* previousListChild;
282 VisiblePosition start;
283 VisiblePosition end;
284
285 if (!listNode->parentNode()->hasEditableStyle())
286 return;
287
288 if (listChildNode->hasTagName(liTag)) {
289 start = firstPositionInNode(listChildNode);
290 end = lastPositionInNode(listChildNode);
291 nextListChild = listChildNode->nextSibling();
292 previousListChild = listChildNode->previousSibling();
293 } else {
294 // A paragraph is visually a list item minus a list marker. The paragraph will be moved.
295 start = startOfParagraph(originalStart, CanSkipOverEditingBoundary);
296 end = endOfParagraph(start, CanSkipOverEditingBoundary);
297 nextListChild = enclosingListChild(end.next().deepEquivalent().deprecatedNode(), listNode);
298 ASSERT(nextListChild != listChildNode);
299 previousListChild = enclosingListChild(start.previous().deepEquivalent().deprecatedNode(), listNode);
300 ASSERT(previousListChild != listChildNode);
301 }
302
303 if (start.isNull() || end.isNull())
304 return;
305
306 // When removing a list, we must always create a placeholder to act as a point of insertion
307 // for the list content being removed.
308 auto placeholder = HTMLBRElement::create(document());
309 RefPtr<Element> nodeToInsert = placeholder.copyRef();
310 // If the content of the list item will be moved into another list, put it in a list item
311 // so that we don't create an orphaned list child.
312 if (enclosingList(listNode)) {
313 nodeToInsert = HTMLLIElement::create(document());
314 appendNode(placeholder.copyRef(), *nodeToInsert);
315 }
316
317 if (nextListChild && previousListChild) {
318 // We want to pull listChildNode out of listNode, and place it before nextListChild
319 // and after previousListChild, so we split listNode and insert it between the two lists.
320 // But to split listNode, we must first split ancestors of listChildNode between it and listNode,
321 // if any exist.
322 // FIXME: We appear to split at nextListChild as opposed to listChildNode so that when we remove
323 // listChildNode below in moveParagraphs, previousListChild will be removed along with it if it is
324 // unrendered. But we ought to remove nextListChild too, if it is unrendered.
325 splitElement(*listNode, *splitTreeToNode(*nextListChild, *listNode));
326 insertNodeBefore(nodeToInsert.releaseNonNull(), *listNode);
327 } else if (nextListChild || listChildNode->parentNode() != listNode) {
328 // Just because listChildNode has no previousListChild doesn't mean there isn't any content
329 // in listNode that comes before listChildNode, as listChildNode could have ancestors
330 // between it and listNode. So, we split up to listNode before inserting the placeholder
331 // where we're about to move listChildNode to.
332 if (auto listChildNodeParentNode = makeRefPtr(listChildNode->parentNode()); listChildNodeParentNode && listChildNodeParentNode != listNode)
333 splitElement(*listNode, *splitTreeToNode(*listChildNode, *listNode).get());
334 insertNodeBefore(nodeToInsert.releaseNonNull(), *listNode);
335 } else
336 insertNodeAfter(nodeToInsert.releaseNonNull(), *listNode);
337
338 VisiblePosition insertionPoint = VisiblePosition(positionBeforeNode(placeholder.ptr()));
339 moveParagraphs(start, end, insertionPoint, true);
340}

342static Element* adjacentEnclosingList(const VisiblePosition& pos, const VisiblePosition& adjacentPos, const QualifiedName& listTag)
343{
344 Element* listNode = outermostEnclosingList(adjacentPos.deepEquivalent().deprecatedNode());
345
346 if (!listNode)
347 return 0;
348
349 Node* previousCell = enclosingTableCell(pos.deepEquivalent());
350 Node* currentCell = enclosingTableCell(adjacentPos.deepEquivalent());
351
352 if (!listNode->hasTagName(listTag)
353 || listNode->contains(pos.deepEquivalent().deprecatedNode())
354 || previousCell != currentCell
355 || enclosingList(listNode) != enclosingList(pos.deepEquivalent().deprecatedNode()))
356 return 0;
357
358 return listNode;
359}

361RefPtr<HTMLElement> InsertListCommand::listifyParagraph(const VisiblePosition& originalStart, const QualifiedName& listTag)
362{
363 VisiblePosition start = startOfParagraph(originalStart, CanSkipOverEditingBoundary);
364 VisiblePosition end = endOfParagraph(start, CanSkipOverEditingBoundary);
365
366 if (start.isNull() || end.isNull() || !start.deepEquivalent().containerNode()->hasEditableStyle() || !end.deepEquivalent().containerNode()->hasEditableStyle())
367 return 0;
368
369 // Check for adjoining lists.
370 auto listItemElement = HTMLLIElement::create(document());
371 auto placeholder = HTMLBRElement::create(document());
372 appendNode(placeholder.copyRef(), listItemElement.copyRef());
373
374 // Place list item into adjoining lists.
375 Element* previousList = adjacentEnclosingList(start.deepEquivalent(), start.previous(CannotCrossEditingBoundary), listTag);
376 Element* nextList = adjacentEnclosingList(start.deepEquivalent(), end.next(CannotCrossEditingBoundary), listTag);
377 RefPtr<HTMLElement> listElement;
378 if (previousList)
379 appendNode(WTFMove(listItemElement), *previousList);
380 else if (nextList)
381 insertNodeAt(WTFMove(listItemElement), positionBeforeNode(nextList));
382 else {
383 // Create the list.
384 listElement = createHTMLElement(document(), listTag);
385 appendNode(WTFMove(listItemElement), *listElement);
386
387 if (start == end && isBlock(start.deepEquivalent().deprecatedNode())) {
388 // Inserting the list into an empty paragraph that isn't held open
389 // by a br or a '\n', will invalidate start and end. Insert
390 // a placeholder and then recompute start and end.
391 auto blockPlaceholder = insertBlockPlaceholder(start.deepEquivalent());
392 start = positionBeforeNode(blockPlaceholder.get());
393 end = start;
394 }
395
396 // Insert the list at a position visually equivalent to start of the
397 // paragraph that is being moved into the list.
398 // Try to avoid inserting it somewhere where it will be surrounded by
399 // inline ancestors of start, since it is easier for editing to produce
400 // clean markup when inline elements are pushed down as far as possible.
401 Position insertionPos(start.deepEquivalent().upstream());
402 // Also avoid the containing list item.
403 Node* listChild = enclosingListChild(insertionPos.deprecatedNode());
404 if (listChild && listChild->hasTagName(liTag))
405 insertionPos = positionInParentBeforeNode(listChild);
406
407 if (!isEditablePosition(insertionPos))
408 return 0;
409
410 insertNodeAt(*listElement, insertionPos);
411
412 // We inserted the list at the start of the content we're about to move
413 // Update the start of content, so we don't try to move the list into itself. bug 19066
414 // Layout is necessary since start's node's inline renderers may have been destroyed by the insertion
415 // The end of the content may have changed after the insertion and layout so update it as well.
416 if (insertionPos == start.deepEquivalent()) {
417 listElement->document().updateLayoutIgnorePendingStylesheets();
418 start = startOfParagraph(originalStart, CanSkipOverEditingBoundary);
419 end = endOfParagraph(start, CanSkipOverEditingBoundary);
420 }
421 }
422
423 moveParagraph(start, end, positionBeforeNode(placeholder.ptr()), true);
424
425 if (listElement)
426 return mergeWithNeighboringLists(*listElement);
427
428 if (canMergeLists(previousList, nextList))
429 mergeIdenticalElements(*previousList, *nextList);
430
431 return listElement;
432}
