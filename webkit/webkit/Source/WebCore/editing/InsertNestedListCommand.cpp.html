37void InsertNestedListCommand::insertUnorderedList(Document& document)
38{
39 InsertNestedListCommand::create(document, Type::UnorderedList)->apply();
40}
41
42void InsertNestedListCommand::insertOrderedList(Document& document)
43{
44 InsertNestedListCommand::create(document, Type::OrderedList)->apply();
45}
46
47void InsertNestedListCommand::doApply()
48{
49 if (endingSelection().isNoneOrOrphaned() || !endingSelection().isContentRichlyEditable())
50 return;
51
52 if (auto enclosingItem = makeRefPtr(enclosingElementWithTag(endingSelection().visibleStart().deepEquivalent(), HTMLNames::liTag))) {
53 auto newListItem = HTMLLIElement::create(document());
54 insertNodeAfter(newListItem.copyRef(), *enclosingItem);
55 setEndingSelection({ Position { newListItem.ptr(), Position::PositionIsBeforeChildren }, Affinity::Downstream });
56
57 auto commandType = m_type == Type::OrderedList ? IncreaseSelectionListLevelCommand::Type::OrderedList : IncreaseSelectionListLevelCommand::Type::UnorderedList;
58 applyCommandToComposite(IncreaseSelectionListLevelCommand::create(document(), commandType));
59 return;
60 }
61
62 auto commandType = m_type == Type::OrderedList ? InsertListCommand::Type::OrderedList : InsertListCommand::Type::UnorderedList;
63 applyCommandToComposite(InsertListCommand::create(document(), commandType));
64}
