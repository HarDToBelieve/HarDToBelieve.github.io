26#pragma once
27
28#include "CompositeEditCommand.h"
29#include "EditAction.h"
30
31namespace WebCore {
32
33class InsertNestedListCommand final : public CompositeEditCommand {
34public:
35 static void insertUnorderedList(Document&);
36 static void insertOrderedList(Document&);
37
38private:
39 enum class Type : uint8_t { OrderedList, UnorderedList };
40
41 static Ref<InsertNestedListCommand> create(Document& document, Type type)
42 {
43 return adoptRef(*new InsertNestedListCommand(document, type));
44 }
45
46 InsertNestedListCommand(Document& document, Type type)
47 : CompositeEditCommand(document)
48 , m_type(type)
49 {
50 }
51
52 EditAction editingAction() const final
53 {
54 if (m_type == Type::OrderedList)
55 return EditAction::InsertOrderedList;
56
57 return EditAction::InsertUnorderedList;
58 }
59
60 bool preservesTypingStyle() const final { return true; }
61 void doApply() final;
62
63 Type m_type;
64};
65
66} // namespace WebCore
