45// When inserting a new line, we want to avoid nesting empty divs if we can. Otherwise, when
46// pasting, it's easy to have each new line be a div deeper than the previous. E.g., in the case
47// below, we want to insert at ^ instead of |.
48// <div>foo<div>bar</div>|</div>^
49static Element* highestVisuallyEquivalentDivBelowRoot(Element* startBlock)
50{
51 Element* curBlock = startBlock;
52 // We don't want to return a root node (if it happens to be a div, e.g., in a document fragment) because there are no
53 // siblings for us to append to.
54 while (!curBlock->nextSibling() && curBlock->parentElement()->hasTagName(divTag) && curBlock->parentElement()->parentElement()) {
55 if (curBlock->parentElement()->hasAttributes())
56 break;
57 curBlock = curBlock->parentElement();
58 }
59 return curBlock;
60}

74void InsertParagraphSeparatorCommand::calculateStyleBeforeInsertion(const Position& position)
75{
76 // It is only important to set a style to apply later if we're at the boundaries of
77 // a paragraph. Otherwise, content that is moved as part of the work of the command
78 // will lend their styles to the new paragraph without any extra work needed.
79 auto visiblePosition = VisiblePosition { position };
80 if (!isStartOfParagraph(visiblePosition) && !isEndOfParagraph(visiblePosition))
81 return;
82
83 m_style = EditingStyle::create(position, EditingStyle::EditingPropertiesInEffect);
84 m_style->mergeTypingStyle(*position.document());
85}

87void InsertParagraphSeparatorCommand::applyStyleAfterInsertion(Node* originalEnclosingBlock)
88{
89 // Not only do we break out of header tags, but we also do not preserve the typing style,
90 // in order to match other browsers.
91 if (originalEnclosingBlock->hasTagName(h1Tag) ||
92 originalEnclosingBlock->hasTagName(h2Tag) ||
93 originalEnclosingBlock->hasTagName(h3Tag) ||
94 originalEnclosingBlock->hasTagName(h4Tag) ||
95 originalEnclosingBlock->hasTagName(h5Tag))
96 return;
97
98 if (!m_style)
99 return;
100
101 m_style->prepareToApplyAt(endingSelection().start());
102 if (!m_style->isEmpty())
103 applyStyle(m_style.get());
104}

106bool InsertParagraphSeparatorCommand::shouldUseDefaultParagraphElement(Node* enclosingBlock) const
107{
108 if (m_mustUseDefaultParagraphElement)
109 return true;
110
111 // Assumes that if there was a range selection, it was already deleted.
112 if (!isEndOfBlock(endingSelection().visibleStart()))
113 return false;
114
115 return enclosingBlock->hasTagName(h1Tag) ||
116 enclosingBlock->hasTagName(h2Tag) ||
117 enclosingBlock->hasTagName(h3Tag) ||
118 enclosingBlock->hasTagName(h4Tag) ||
119 enclosingBlock->hasTagName(h5Tag);
120}

122void InsertParagraphSeparatorCommand::getAncestorsInsideBlock(const Node* insertionNode, Element* outerBlock, Vector<RefPtr<Element>>& ancestors)
123{
124 ancestors.clear();
125
126 // Build up list of ancestors elements between the insertion node and the outer block.
127 if (insertionNode != outerBlock) {
128 for (Element* n = insertionNode->parentElement(); n && n != outerBlock; n = n->parentElement())
129 ancestors.append(n);
130 }
131}

133Ref<Element> InsertParagraphSeparatorCommand::cloneHierarchyUnderNewBlock(const Vector<RefPtr<Element>>& ancestors, Ref<Element>&& blockToInsert)
134{
135 // Make clones of ancestors in between the start node and the start block.
136 RefPtr<Element> parent = WTFMove(blockToInsert);
137 for (size_t i = ancestors.size(); i != 0; --i) {
138 auto child = ancestors[i - 1]->cloneElementWithoutChildren(document());
139 // It should always be okay to remove id from the cloned elements, since the originals are not deleted.
140 child->removeAttribute(idAttr);
141 appendNode(child.copyRef(), parent.releaseNonNull());
142 parent = WTFMove(child);
143 }
144
145 return parent.releaseNonNull();
146}

148void InsertParagraphSeparatorCommand::doApply()
149{
150 if (endingSelection().isNoneOrOrphaned())
151 return;
152
153 auto insertionPosition = endingSelection().start();
154 auto affinity = endingSelection().affinity();
155
156 // Delete the current selection.
157 if (endingSelection().isRange()) {
158 calculateStyleBeforeInsertion(insertionPosition);
159 deleteSelection(false, true);
160 insertionPosition = endingSelection().start();
161 affinity = endingSelection().affinity();
162 }
163
164 // FIXME: The parentAnchoredEquivalent conversion needs to be moved into enclosingBlock.
165 RefPtr<Element> startBlock = enclosingBlock(insertionPosition.parentAnchoredEquivalent().containerNode());
166 Position canonicalPos = VisiblePosition(insertionPosition).deepEquivalent();
167 if (!startBlock
168 || !startBlock->nonShadowBoundaryParentNode()
169 || isRenderedTable(startBlock.get())
170 || isTableCell(startBlock.get())
171 || is<HTMLFormElement>(*startBlock)
172 // FIXME: If the node is hidden, we don't have a canonical position so we will do the wrong thing for tables and <hr>. https://bugs.webkit.org/show_bug.cgi?id=40342
173 || (!canonicalPos.isNull() && canonicalPos.deprecatedNode()->renderer() && canonicalPos.deprecatedNode()->renderer()->isTable())
174 || (!canonicalPos.isNull() && canonicalPos.deprecatedNode()->hasTagName(hrTag))) {
175 applyCommandToComposite(InsertLineBreakCommand::create(document()));
176 return;
177 }
178
179 // Use the leftmost candidate.
180 insertionPosition = insertionPosition.upstream();
181 if (!insertionPosition.isCandidate())
182 insertionPosition = insertionPosition.downstream();
183
184 // Adjust the insertion position after the delete
185 insertionPosition = positionAvoidingSpecialElementBoundary(insertionPosition);
186 VisiblePosition visiblePos(insertionPosition, affinity);
187 if (visiblePos.isNull())
188 return;
189
190 calculateStyleBeforeInsertion(insertionPosition);
191
192 //---------------------------------------------------------------------
193 // Handle special case of typing return on an empty list item
194 if (breakOutOfEmptyListItem())
195 return;
196
197 //---------------------------------------------------------------------
198 // Prepare for more general cases.
199
200 bool isFirstInBlock = isStartOfBlock(visiblePos);
201 bool isLastInBlock = isEndOfBlock(visiblePos);
202 bool nestNewBlock = false;
203
204 // FIXME: <http://webkit.org/b/211864> If insertionPosition is not editable, we should compute a position that is.
205 if (!isEditablePosition(insertionPosition))
206 return;
207
208 // Create block to be inserted.
209 RefPtr<Element> blockToInsert;
210 if (startBlock->isRootEditableElement()) {
211 blockToInsert = createDefaultParagraphElement(document());
212 nestNewBlock = true;
213 } else if (shouldUseDefaultParagraphElement(startBlock.get()))
214 blockToInsert = createDefaultParagraphElement(document());
215 else
216 blockToInsert = startBlock->cloneElementWithoutChildren(document());
217
218 //---------------------------------------------------------------------
219 // Handle case when position is in the last visible position in its block,
220 // including when the block is empty.
221 if (isLastInBlock) {
222 if (nestNewBlock) {
223 if (isFirstInBlock && !lineBreakExistsAtVisiblePosition(visiblePos)) {
224 // The block is empty. Create an empty block to
225 // represent the paragraph that we're leaving.
226 auto extraBlock = createDefaultParagraphElement(document());
227 appendNode(extraBlock.copyRef(), *startBlock);
228 if (!appendBlockPlaceholder(WTFMove(extraBlock)))
229 return;
230 }
231 appendNode(*blockToInsert, *startBlock);
232 } else {
233 // We can get here if we pasted a copied portion of a blockquote with a newline at the end and are trying to paste it
234 // into an unquoted area. We then don't want the newline within the blockquote or else it will also be quoted.
235 if (m_pasteBlockqutoeIntoUnquotedArea) {
236 if (Node* highestBlockquote = highestEnclosingNodeOfType(canonicalPos, &isMailBlockquote))
237 startBlock = downcast<Element>(highestBlockquote);
238 }
239
240 // Most of the time we want to stay at the nesting level of the startBlock (e.g., when nesting within lists). However,
241 // for div nodes, this can result in nested div tags that are hard to break out of.
242 Element* siblingNode = startBlock.get();
243 if (blockToInsert->hasTagName(divTag))
244 siblingNode = highestVisuallyEquivalentDivBelowRoot(startBlock.get());
245 insertNodeAfter(*blockToInsert, *siblingNode);
246 }
247
248 // Recreate the same structure in the new paragraph.
249
250 Vector<RefPtr<Element>> ancestors;
251 getAncestorsInsideBlock(positionOutsideTabSpan(insertionPosition).deprecatedNode(), startBlock.get(), ancestors);
252 auto parent = cloneHierarchyUnderNewBlock(ancestors, *blockToInsert);
253 auto* parentPtr = parent.ptr();
254
255 if (!appendBlockPlaceholder(WTFMove(parent)))
256 return;
257
258 setEndingSelection(VisibleSelection(firstPositionInNode(parentPtr), Affinity::Downstream, endingSelection().isDirectional()));
259 return;
260 }
261
262
263 //---------------------------------------------------------------------
264 // Handle case when position is in the first visible position in its block, and
265 // similar case where previous position is in another, presumeably nested, block.
266 if (isFirstInBlock || !inSameBlock(visiblePos, visiblePos.previous())) {
267 Node *refNode;
268
269 insertionPosition = positionOutsideTabSpan(insertionPosition);
270
271 if (isFirstInBlock && !nestNewBlock)
272 refNode = startBlock.get();
273 else if (isFirstInBlock && nestNewBlock) {
274 // startBlock should always have children, otherwise isLastInBlock would be true and it's handled above.
275 ASSERT(startBlock->firstChild());
276 refNode = startBlock->firstChild();
277 } else if (insertionPosition.containerNode() == startBlock && nestNewBlock) {
278 refNode = startBlock->traverseToChildAt(insertionPosition.computeOffsetInContainerNode());
279 ASSERT(refNode); // must be true or we'd be in the end of block case
280 } else
281 refNode = insertionPosition.deprecatedNode();
282
283 // find ending selection position easily before inserting the paragraph
284 insertionPosition = insertionPosition.downstream();
285
286 insertNodeBefore(*blockToInsert, *refNode);
287
288 // Recreate the same structure in the new paragraph.
289
290 Vector<RefPtr<Element>> ancestors;
291 getAncestorsInsideBlock(positionAvoidingSpecialElementBoundary(positionOutsideTabSpan(insertionPosition)).deprecatedNode(), startBlock.get(), ancestors);
292
293 auto parent = cloneHierarchyUnderNewBlock(ancestors, *blockToInsert);
294 if (!appendBlockPlaceholder(WTFMove(parent)))
295 return;
296
297 // In this case, we need to set the new ending selection.
298 setEndingSelection(VisibleSelection(insertionPosition, Affinity::Downstream, endingSelection().isDirectional()));
299 return;
300 }
301
302 //---------------------------------------------------------------------
303 // Handle the (more complicated) general case,
304
305 // All of the content in the current block after visiblePos is
306 // about to be wrapped in a new paragraph element. Add a br before
307 // it if visiblePos is at the start of a paragraph so that the
308 // content will move down a line.
309 if (isStartOfParagraph(visiblePos)) {
310 auto br = HTMLBRElement::create(document());
311 insertNodeAt(br.copyRef(), insertionPosition);
312 if (!br->parentNode())
313 return;
314 insertionPosition = positionInParentAfterNode(br.ptr());
315 // If the insertion point is a break element, there is nothing else
316 // we need to do.
317 if (auto* renderer = visiblePos.deepEquivalent().anchorNode()->renderer(); renderer && renderer->isBR()) {
318 setEndingSelection(VisibleSelection(insertionPosition, Affinity::Downstream, endingSelection().isDirectional()));
319 return;
320 }
321 }
322
323 // Move downstream. Typing style code will take care of carrying along the
324 // style of the upstream position.
325 insertionPosition = insertionPosition.downstream();
326
327 // At this point, the insertionPosition's node could be a container, and we want to make sure we include
328 // all of the correct nodes when building the ancestor list. So this needs to be the deepest representation of the position
329 // before we walk the DOM tree.
330 insertionPosition = positionOutsideTabSpan(VisiblePosition(insertionPosition).deepEquivalent());
331 if (insertionPosition.isNull())
332 return;
333
334 // If the returned position lies either at the end or at the start of an element that is ignored by editing
335 // we should move to its upstream or downstream position.
336 if (editingIgnoresContent(*insertionPosition.deprecatedNode())) {
337 if (insertionPosition.atLastEditingPositionForNode())
338 insertionPosition = insertionPosition.downstream();
339 else if (insertionPosition.atFirstEditingPositionForNode())
340 insertionPosition = insertionPosition.upstream();
341 }
342
343 // Make sure we do not cause a rendered space to become unrendered.
344 // FIXME: We need the affinity for pos, but pos.downstream() does not give it
345 Position leadingWhitespace = insertionPosition.leadingWhitespacePosition(VisiblePosition::defaultAffinity);
346 // FIXME: leadingWhitespacePosition is returning the position before preserved newlines for positions
347 // after the preserved newline, causing the newline to be turned into a nbsp.
348 if (is<Text>(leadingWhitespace.deprecatedNode())) {
349 Text& textNode = downcast<Text>(*leadingWhitespace.deprecatedNode());
350 ASSERT(!textNode.renderer() || textNode.renderer()->style().collapseWhiteSpace());
351 replaceTextInNodePreservingMarkers(textNode, leadingWhitespace.deprecatedEditingOffset(), 1, nonBreakingSpaceString());
352 }
353
354 // Split at pos if in the middle of a text node.
355 Position positionAfterSplit;
356 if (insertionPosition.anchorType() == Position::PositionIsOffsetInAnchor && is<Text>(*insertionPosition.containerNode())) {
357 Ref<Text> textNode = downcast<Text>(*insertionPosition.containerNode());
358 bool atEnd = static_cast<unsigned>(insertionPosition.offsetInContainerNode()) >= textNode->length();
359 if (insertionPosition.deprecatedEditingOffset() > 0 && !atEnd) {
360 splitTextNode(textNode, insertionPosition.offsetInContainerNode());
361 positionAfterSplit = firstPositionInNode(textNode.ptr());
362 if (!textNode->previousSibling())
363 return; // Bail out if mutation events detachd the split text node.
364 insertionPosition.moveToPosition(textNode->previousSibling(), insertionPosition.offsetInContainerNode());
365 visiblePos = VisiblePosition(insertionPosition);
366 }
367 }
368
369 // If we got detached due to mutation events, just bail out.
370 if (!startBlock->parentNode())
371 return;
372
373 // Put the added block in the tree.
374 if (nestNewBlock)
375 appendNode(*blockToInsert, *startBlock);
376 else
377 insertNodeAfter(*blockToInsert, *startBlock);
378
379 document().updateLayoutIgnorePendingStylesheets();
380
381 // If the paragraph separator was inserted at the end of a paragraph, an empty line must be
382 // created. All of the nodes, starting at visiblePos, are about to be added to the new paragraph
383 // element. If the first node to be inserted won't be one that will hold an empty line open, add a br.
384 if (isEndOfParagraph(visiblePos) && !lineBreakExistsAtVisiblePosition(visiblePos))
385 appendNode(HTMLBRElement::create(document()), *blockToInsert);
386
387 // Move the start node and the siblings of the start node.
388 if (VisiblePosition(insertionPosition) != VisiblePosition(positionBeforeNode(blockToInsert.get()))) {
389 Node* n;
390 if (insertionPosition.containerNode() == startBlock)
391 n = insertionPosition.computeNodeAfterPosition();
392 else {
393 Node* splitTo = insertionPosition.containerNode();
394 if (is<Text>(*splitTo) && insertionPosition.offsetInContainerNode() >= caretMaxOffset(*splitTo))
395 splitTo = NodeTraversal::next(*splitTo, startBlock.get());
396 ASSERT(splitTo);
397 splitTreeToNode(*splitTo, *startBlock);
398
399 for (n = startBlock->firstChild(); n; n = n->nextSibling()) {
400 if (VisiblePosition(insertionPosition) <= VisiblePosition(positionBeforeNode(n)))
401 break;
402 }
403 }
404
405 moveRemainingSiblingsToNewParent(n, blockToInsert.get(), *blockToInsert);
406 }
407
408 // Handle whitespace that occurs after the split
409 if (positionAfterSplit.isNotNull()) {
410 document().updateLayoutIgnorePendingStylesheets();
411 if (!positionAfterSplit.isRenderedCharacter()) {
412 // Clear out all whitespace and insert one non-breaking space
413 ASSERT(!positionAfterSplit.containerNode()->renderer() || positionAfterSplit.containerNode()->renderer()->style().collapseWhiteSpace());
414 deleteInsignificantTextDownstream(positionAfterSplit);
415 if (is<Text>(*positionAfterSplit.deprecatedNode()))
416 insertTextIntoNode(downcast<Text>(*positionAfterSplit.containerNode()), 0, nonBreakingSpaceString());
417 }
418 }
419
420 setEndingSelection(VisibleSelection(firstPositionInNode(blockToInsert.get()), Affinity::Downstream, endingSelection().isDirectional()));
421 applyStyleAfterInsertion(startBlock.get());
422}
