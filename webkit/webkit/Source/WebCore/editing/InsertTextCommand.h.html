28#include "CompositeEditCommand.h"

30namespace WebCore {
31
32class DocumentMarkerController;
33class Text;
34
35class TextInsertionMarkerSupplier : public RefCounted<TextInsertionMarkerSupplier> {
36public:
37 virtual ~TextInsertionMarkerSupplier() = default;
38 virtual void addMarkersToTextNode(Text&, unsigned offsetOfInsertion, const String& textInserted) = 0;
39protected:
40 TextInsertionMarkerSupplier() = default;
41};
42
43class InsertTextCommand : public CompositeEditCommand {
44public:
45 enum RebalanceType {
46 RebalanceLeadingAndTrailingWhitespaces,
47 RebalanceAllWhitespaces
48 };
49
50 static Ref<InsertTextCommand> create(Document& document, const String& text, bool selectInsertedText = false,
51 RebalanceType rebalanceType = RebalanceLeadingAndTrailingWhitespaces, EditAction editingAction = EditAction::Insert)
52 {
53 return adoptRef(*new InsertTextCommand(document, text, selectInsertedText, rebalanceType, editingAction));
54 }
55
56 static Ref<InsertTextCommand> createWithMarkerSupplier(Document& document, const String& text, Ref<TextInsertionMarkerSupplier>&& markerSupplier, EditAction editingAction = EditAction::Insert)
57 {
58 return adoptRef(*new InsertTextCommand(document, text, WTFMove(markerSupplier), editingAction));
59 }
60
61protected:
62 InsertTextCommand(Document&, const String& text, Ref<TextInsertionMarkerSupplier>&&, EditAction);
63 InsertTextCommand(Document&, const String& text, bool selectInsertedText, RebalanceType, EditAction);
64
65private:
66
67 void deleteCharacter();
68
69 void doApply() override;
70
71 bool isInsertTextCommand() const override { return true; }
72
73 Position positionInsideTextNode(const Position&);
74 Position insertTab(const Position&);
75
76 bool performTrivialReplace(const String&, bool selectInsertedText);
77 bool performOverwrite(const String&, bool selectInsertedText);
78 void setEndingSelectionWithoutValidation(const Position& startPosition, const Position& endPosition);
79
80 friend class TypingCommand;
81
82 String m_text;
83 bool m_selectInsertedText;
84 RebalanceType m_rebalanceType;
85 RefPtr<TextInsertionMarkerSupplier> m_markerSupplier;
86};
