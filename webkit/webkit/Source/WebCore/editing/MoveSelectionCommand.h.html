27
28#include "CompositeEditCommand.h"

30namespace WebCore {
31
32class DocumentFragment;
33
34class MoveSelectionCommand : public CompositeEditCommand {
35public:
36 static Ref<MoveSelectionCommand> create(Ref<DocumentFragment>&& fragment, const Position& position, bool smartInsert = false, bool smartDelete = false)
37 {
38 return adoptRef(*new MoveSelectionCommand(WTFMove(fragment), position, smartInsert, smartDelete));
39 }
40
41private:
42 MoveSelectionCommand(Ref<DocumentFragment>&&, const Position&, bool smartInsert, bool smartDelete);
43
44 void doApply() override;
45 EditAction editingAction() const override;
46 bool shouldDispatchInputEvents() const final { return false; }
47
48 Ref<DocumentFragment> m_fragment;
49 Position m_position;
50 bool m_smartInsert;
51 bool m_smartDelete;
52};

54} // namespace WebCore
