36RemoveNodeCommand::RemoveNodeCommand(Ref<Node>&& node, ShouldAssumeContentIsAlwaysEditable shouldAssumeContentIsAlwaysEditable, EditAction editingAction)
37 : SimpleEditCommand(node->document(), editingAction)
38 , m_node(WTFMove(node))
39 , m_shouldAssumeContentIsAlwaysEditable(shouldAssumeContentIsAlwaysEditable)
40{
41 ASSERT(m_node->parentNode());
42}
43
44void RemoveNodeCommand::doApply()
45{
46 ContainerNode* parent = m_node->parentNode();
47 if (!parent || (m_shouldAssumeContentIsAlwaysEditable == DoNotAssumeContentIsAlwaysEditable
48 && !isEditableNode(*parent) && parent->renderer()))
49 return;
50 ASSERT(isEditableNode(*parent) || !parent->renderer());
51
52 m_parent = parent;
53 m_refChild = m_node->nextSibling();
54
55 m_node->remove();
56}
57
58void RemoveNodeCommand::doUnapply()
59{
60 RefPtr<ContainerNode> parent = WTFMove(m_parent);
61 RefPtr<Node> refChild = WTFMove(m_refChild);
62 if (!parent || !parent->hasEditableStyle())
63 return;
64
65 parent->insertBefore(m_node, refChild.get());
66}
67
68#ifndef NDEBUG
69void RemoveNodeCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
70{
71 addNodeAndDescendants(m_parent.get(), nodes);
72 addNodeAndDescendants(m_refChild.get(), nodes);
73 addNodeAndDescendants(m_node.ptr(), nodes);
74}
75#endif
