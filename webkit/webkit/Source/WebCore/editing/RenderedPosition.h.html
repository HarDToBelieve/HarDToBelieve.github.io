31#pragma once
32
33#include "CaretRectComputation.h"
34#include "LayoutIntegrationLineIterator.h"
35#include "LayoutIntegrationRunIterator.h"
36#include "TextAffinity.h"

38namespace WebCore {
39
40class LayoutUnit;
41class Position;
42class RenderObject;
43class VisiblePosition;
44
45class RenderedPosition {
46public:
47 RenderedPosition();
48 explicit RenderedPosition(const VisiblePosition&);
49 explicit RenderedPosition(const Position&, Affinity);
50 bool isEquivalent(const RenderedPosition&) const;
51
52 bool isNull() const { return !m_renderer; }
53 LayoutIntegration::LineIterator line() const { return m_run ? m_run.line() : LayoutIntegration::LineIterator(); }
54
55 unsigned char bidiLevelOnLeft() const;
56 unsigned char bidiLevelOnRight() const;
57 RenderedPosition leftBoundaryOfBidiRun(unsigned char bidiLevelOfRun);
58 RenderedPosition rightBoundaryOfBidiRun(unsigned char bidiLevelOfRun);
59
60 enum ShouldMatchBidiLevel { MatchBidiLevel, IgnoreBidiLevel };
61 bool atLeftBoundaryOfBidiRun() const { return atLeftBoundaryOfBidiRun(IgnoreBidiLevel, 0); }
62 bool atRightBoundaryOfBidiRun() const { return atRightBoundaryOfBidiRun(IgnoreBidiLevel, 0); }
63 // The following two functions return true only if the current position is at the end of the bidi run
64 // of the specified bidi embedding level.
65 bool atLeftBoundaryOfBidiRun(unsigned char bidiLevelOfRun) const { return atLeftBoundaryOfBidiRun(MatchBidiLevel, bidiLevelOfRun); }
66 bool atRightBoundaryOfBidiRun(unsigned char bidiLevelOfRun) const { return atRightBoundaryOfBidiRun(MatchBidiLevel, bidiLevelOfRun); }
67
68 Position positionAtLeftBoundaryOfBiDiRun() const;
69 Position positionAtRightBoundaryOfBiDiRun() const;
70
71 IntRect absoluteRect(CaretRectMode = CaretRectMode::Normal) const;
72
73private:
74 bool operator==(const RenderedPosition&) const { return false; }
75 explicit RenderedPosition(const RenderObject*, LayoutIntegration::RunIterator, unsigned offset);
76
77 LayoutIntegration::RunIterator previousLeafOnLine() const;
78 LayoutIntegration::RunIterator nextLeafOnLine() const;
79 bool atLeftmostOffsetInBox() const { return m_run && m_offset == m_run->leftmostCaretOffset(); }
80 bool atRightmostOffsetInBox() const { return m_run && m_offset == m_run->rightmostCaretOffset(); }
81 bool atLeftBoundaryOfBidiRun(ShouldMatchBidiLevel, unsigned char bidiLevelOfRun) const;
82 bool atRightBoundaryOfBidiRun(ShouldMatchBidiLevel, unsigned char bidiLevelOfRun) const;
83
84 const RenderObject* m_renderer { nullptr };
85 LayoutIntegration::RunIterator m_run;
86 unsigned m_offset { 0 };
87
88 mutable std::optional<LayoutIntegration::RunIterator> m_previousLeafOnLine;
89 mutable std::optional<LayoutIntegration::RunIterator> m_nextLeafOnLine;
90};
91
92bool renderObjectContainsPosition(const RenderObject*, const Position&);
93
94} // namespace WebCore
