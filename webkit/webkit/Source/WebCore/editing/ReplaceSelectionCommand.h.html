37class ReplaceSelectionCommand : public CompositeEditCommand {
38public:
39 enum CommandOption {
40 SelectReplacement = 1 << 0,
41 SmartReplace = 1 << 1,
42 MatchStyle = 1 << 2,
43 PreventNesting = 1 << 3,
44 MovingParagraph = 1 << 4,
45 SanitizeFragment = 1 << 5,
46 IgnoreMailBlockquote = 1 << 6,
47 };
48
49 static Ref<ReplaceSelectionCommand> create(Document& document, RefPtr<DocumentFragment>&& fragment, OptionSet<CommandOption> options, EditAction editingAction = EditAction::Insert)
50 {
51 return adoptRef(*new ReplaceSelectionCommand(document, WTFMove(fragment), options, editingAction));
52 }

67 class InsertedNodes {
68 public:
69 void respondToNodeInsertion(Node*);
70 void willRemoveNodePreservingChildren(Node*);
71 void willRemovePossibleAncestorNode(Node*);
72 void willRemoveNode(Node*);
73 void didReplaceNode(Node*, Node* newNode);
74
75 bool isEmpty() { return !m_firstNodeInserted; }
76 Node* firstNodeInserted() const { return m_firstNodeInserted.get(); }
77 Node* lastLeafInserted() const
78 {
79 ASSERT(m_lastNodeInserted);
80 return m_lastNodeInserted->lastDescendant();
81 }
82 Node* pastLastLeaf() const
83 {
84 ASSERT(m_lastNodeInserted);
85 return NodeTraversal::next(*lastLeafInserted());
86 }
87
88 private:
89 RefPtr<Node> m_firstNodeInserted;
90 RefPtr<Node> m_lastNodeInserted;
91 };
