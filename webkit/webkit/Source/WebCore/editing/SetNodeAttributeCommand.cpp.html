33namespace WebCore {
34
35SetNodeAttributeCommand::SetNodeAttributeCommand(Ref<Element>&& element, const QualifiedName& attribute, const AtomString& value)
36 : SimpleEditCommand(element->document())
37 , m_element(WTFMove(element))
38 , m_attribute(attribute)
39 , m_value(value)
40{
41}
42
43void SetNodeAttributeCommand::doApply()
44{
45 m_oldValue = m_element->getAttribute(m_attribute);
46 m_element->setAttribute(m_attribute, m_value);
47}
48
49void SetNodeAttributeCommand::doUnapply()
50{
51 m_element->setAttribute(m_attribute, m_oldValue);
52 AtomStringImpl* nullString = nullptr;
53 m_oldValue = nullString;
54}
55
56#ifndef NDEBUG
57void SetNodeAttributeCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
58{
59 addNodeAndDescendants(m_element.ptr(), nodes);
60}
61#endif
