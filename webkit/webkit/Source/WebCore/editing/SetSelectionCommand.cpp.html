33namespace WebCore {
34
35SetSelectionCommand::SetSelectionCommand(const VisibleSelection& selection, OptionSet<FrameSelection::SetSelectionOption> options)
36 : SimpleEditCommand(selection.base().anchorNode()->document())
37 , m_options(options)
38 , m_selectionToSet(selection)
39{
40}
41
42void SetSelectionCommand::doApply()
43{
44 FrameSelection& selection = document().selection();
45
46 if (selection.shouldChangeSelection(m_selectionToSet) && !m_selectionToSet.isNoneOrOrphaned()) {
47 selection.setSelection(m_selectionToSet, m_options);
48 setEndingSelection(m_selectionToSet);
49 }
50}
51
52void SetSelectionCommand::doUnapply()
53{
54 FrameSelection& selection = document().selection();
55
56 if (selection.shouldChangeSelection(startingSelection()) && !startingSelection().isNoneOrOrphaned())
57 selection.setSelection(startingSelection(), m_options);
58}
