57RefPtr<SpellCheckRequest> SpellCheckRequest::create(OptionSet<TextCheckingType> options, TextCheckingProcessType type, const SimpleRange& checkingRange, const SimpleRange& automaticReplacementRange, const SimpleRange& paragraphRange)
58{
59 String text = plainText(checkingRange);
60 if (text.isEmpty())
61 return nullptr;
62 return adoptRef(*new SpellCheckRequest(checkingRange, automaticReplacementRange, paragraphRange, text, options, type));
63}

70void SpellCheckRequest::didSucceed(const Vector<TextCheckingResult>& results)
71{
72 if (!m_checker)
73 return;
74
75 Ref<SpellCheckRequest> protectedThis(*this);
76 m_checker->didCheckSucceed(m_requestData.identifier().value(), results);
77 m_checker = nullptr;
78}

80void SpellCheckRequest::didCancel()
81{
82 if (!m_checker)
83 return;
84
85 Ref<SpellCheckRequest> protectedThis(*this);
86 m_checker->didCheckCancel(m_requestData.identifier().value());
87 m_checker = nullptr;
88}

125void SpellChecker::timerFiredToProcessQueuedRequest()
126{
127 ASSERT(!m_requestQueue.isEmpty());
128 if (m_requestQueue.isEmpty())
129 return;
130
131 invokeRequest(m_requestQueue.takeFirst());
132}

139bool SpellChecker::canCheckAsynchronously(const SimpleRange& range) const
140{
141 return client() && isCheckable(range) && isAsynchronousEnabled();
142}
143
144bool SpellChecker::isCheckable(const SimpleRange& range) const
145{
146 bool foundRenderer = false;
147 for (auto& node : intersectingNodes(range)) {
148 if (node.renderer()) {
149 foundRenderer = true;
150 break;
151 }
152 }
153 if (!foundRenderer)
154 return false;
155 auto& node = range.start.container.get();
156 return !is<Element>(node) || downcast<Element>(node).isSpellCheckingEnabled();
157}

159void SpellChecker::requestCheckingFor(Ref<SpellCheckRequest>&& request)
160{
161 if (!canCheckAsynchronously(request->paragraphRange()))
162 return;
163
164 ASSERT(!request->data().identifier());
165 auto identifier = TextCheckingRequestIdentifier::generate();
166
167 m_lastRequestIdentifier = identifier;
168 request->setCheckerAndIdentifier(this, identifier);
169
170 if (m_timerToProcessQueuedRequest.isActive() || m_processingRequest) {
171 enqueueRequest(WTFMove(request));
172 return;
173 }
174
175 invokeRequest(WTFMove(request));
176}
177
178void SpellChecker::invokeRequest(Ref<SpellCheckRequest>&& request)
179{
180 ASSERT(!m_processingRequest);
181 if (!client())
182 return;
183 m_processingRequest = WTFMove(request);
184 client()->requestCheckingOfString(*m_processingRequest, m_document.selection().selection());
185}
186
187void SpellChecker::enqueueRequest(Ref<SpellCheckRequest>&& request)
188{
189 for (auto& queue : m_requestQueue) {
190 if (request->rootEditableElement() != queue->rootEditableElement())
191 continue;
192
193 queue = WTFMove(request);
194 return;
195 }
196
197 m_requestQueue.append(WTFMove(request));
198}

200void SpellChecker::didCheck(TextCheckingRequestIdentifier identifier, const Vector<TextCheckingResult>& results)
201{
202 ASSERT(m_processingRequest);
203 ASSERT(m_processingRequest->data().identifier() == identifier);
204 if (m_processingRequest->data().identifier() != identifier) {
205 m_requestQueue.clear();
206 return;
207 }
208
209 m_document.editor().markAndReplaceFor(*m_processingRequest, results);
210
211 if (m_lastProcessedIdentifier.toUInt64() < identifier.toUInt64())
212 m_lastProcessedIdentifier = identifier;
213
214 m_processingRequest = nullptr;
215 if (!m_requestQueue.isEmpty())
216 m_timerToProcessQueuedRequest.startOneShot(0_s);
217}
218
219void SpellChecker::didCheckSucceed(TextCheckingRequestIdentifier identifier, const Vector<TextCheckingResult>& results)
220{
221 TextCheckingRequestData requestData = m_processingRequest->data();
222 if (requestData.identifier() == identifier) {
223 OptionSet<DocumentMarker::MarkerType> markerTypes;
224 if (requestData.checkingTypes().contains(TextCheckingType::Spelling))
225 markerTypes.add(DocumentMarker::Spelling);
226 if (requestData.checkingTypes().contains(TextCheckingType::Grammar))
227 markerTypes.add(DocumentMarker::Grammar);
228 if (!markerTypes.isEmpty())
229 removeMarkers(m_processingRequest->checkingRange(), markerTypes);
230 }
231 didCheck(identifier, results);
232}
233
234void SpellChecker::didCheckCancel(TextCheckingRequestIdentifier identifier)
235{
236 didCheck(identifier, Vector<TextCheckingResult>());
237}
