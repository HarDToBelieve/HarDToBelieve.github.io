43#if USE(AUTOCORRECTION_PANEL)
44// On Mac OS X, we use this command to keep track of user undoing a correction for the first time.
45// This information is needed by spell checking service to update user specific data.
46class SpellingCorrectionRecordUndoCommand : public SimpleEditCommand {
47public:
48 static Ref<SpellingCorrectionRecordUndoCommand> create(Document& document, const String& corrected, const String& correction)
49 {
50 return adoptRef(*new SpellingCorrectionRecordUndoCommand(document, corrected, correction));
51 }
52private:
53 SpellingCorrectionRecordUndoCommand(Document& document, const String& corrected, const String& correction)
54 : SimpleEditCommand(document)
55 , m_corrected(corrected)
56 , m_correction(correction)
57 , m_hasBeenUndone(false)
58 {
59 }
60
61 void doApply() override
62 {
63 }
64
65 void doUnapply() override
66 {
67 if (!m_hasBeenUndone) {
68 document().editor().unappliedSpellCorrection(startingSelection(), m_corrected, m_correction);
69 m_hasBeenUndone = true;
70 }
71
72 }
73
74#ifndef NDEBUG
75 void getNodesInCommand(HashSet<Ref<Node>>&) override
76 {
77 }
78#endif
79
80 String m_corrected;
81 String m_correction;
82 bool m_hasBeenUndone;
83};
84#endif

86SpellingCorrectionCommand::SpellingCorrectionCommand(const SimpleRange& rangeToBeCorrected, const String& correction)
87 : CompositeEditCommand(rangeToBeCorrected.start.document(), EditAction::InsertReplacement)
88 , m_rangeToBeCorrected(rangeToBeCorrected)
89 , m_selectionToBeCorrected(m_rangeToBeCorrected)
90 , m_correction(correction)
91{
92}
93
94Ref<SpellingCorrectionCommand> SpellingCorrectionCommand::create(const SimpleRange& rangeToBeCorrected, const String& correction)
95{
96 return adoptRef(*new SpellingCorrectionCommand(rangeToBeCorrected, correction));
97}
98
99bool SpellingCorrectionCommand::willApplyCommand()
100{
101 m_correctionFragment = createFragmentFromText(m_rangeToBeCorrected, m_correction);
102 return CompositeEditCommand::willApplyCommand();
103}
104
105void SpellingCorrectionCommand::doApply()
106{
107 m_corrected = plainText(m_rangeToBeCorrected);
108 if (!m_corrected.length())
109 return;
110
111 if (!document().selection().shouldChangeSelection(m_selectionToBeCorrected))
112 return;
113
114 applyCommandToComposite(SetSelectionCommand::create(m_selectionToBeCorrected, FrameSelection::defaultSetSelectionOptions() | FrameSelection::SpellCorrectionTriggered));
115#if USE(AUTOCORRECTION_PANEL)
116 applyCommandToComposite(SpellingCorrectionRecordUndoCommand::create(document(), m_corrected, m_correction));
117#endif
118
119 applyCommandToComposite(ReplaceSelectionCommand::create(document(), WTFMove(m_correctionFragment), ReplaceSelectionCommand::MatchStyle, EditAction::Paste));
120}
121
122String SpellingCorrectionCommand::inputEventData() const
123{
124 if (isEditingTextAreaOrTextInput())
125 return m_correction;
126
127 return CompositeEditCommand::inputEventData();
128}
129
130Vector<RefPtr<StaticRange>> SpellingCorrectionCommand::targetRanges() const
131{
132 return { 1, StaticRange::create(m_rangeToBeCorrected) };
133}
134
135RefPtr<DataTransfer> SpellingCorrectionCommand::inputEventDataTransfer() const
136{
137 if (!isEditingTextAreaOrTextInput())
138 return DataTransfer::createForInputEvent(m_correction, serializeFragment(*m_correctionFragment, SerializedNodes::SubtreeIncludingNode));
139
140 return CompositeEditCommand::inputEventDataTransfer();
141}
142
143bool SpellingCorrectionCommand::shouldRetainAutocorrectionIndicator() const
144{
145 return true;
146}
