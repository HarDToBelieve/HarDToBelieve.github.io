36SplitElementCommand::SplitElementCommand(Ref<Element>&& element, Ref<Node>&& atChild)
37 : SimpleEditCommand(element->document())
38 , m_element2(WTFMove(element))
39 , m_atChild(WTFMove(atChild))
40{
41 ASSERT(m_atChild->parentNode() == m_element2.ptr());
42}
43
44void SplitElementCommand::executeApply()
45{
46 if (m_atChild->parentNode() != m_element2.ptr())
47 return;
48
49 Vector<Ref<Node>> children;
50 for (Node* node = m_element2->firstChild(); node != m_atChild.ptr(); node = node->nextSibling())
51 children.append(*node);
52
53 auto* parent = m_element2->parentNode();
54 if (!parent || !parent->hasEditableStyle())
55 return;
56 if (parent->insertBefore(*m_element1, m_element2.ptr()).hasException())
57 return;
58
59 // Delete id attribute from the second element because the same id cannot be used for more than one element
60 m_element2->removeAttribute(HTMLNames::idAttr);
61
62 for (auto& child : children)
63 m_element1->appendChild(child);
64}
65
66void SplitElementCommand::doApply()
67{
68 m_element1 = m_element2->cloneElementWithoutChildren(document());
69
70 executeApply();
71}
72
73void SplitElementCommand::doUnapply()
74{
75 if (!m_element1 || !m_element1->hasEditableStyle() || !m_element2->hasEditableStyle())
76 return;
77
78 Vector<Ref<Node>> children;
79 for (Node* node = m_element1->firstChild(); node; node = node->nextSibling())
80 children.append(*node);
81
82 RefPtr<Node> refChild = m_element2->firstChild();
83
84 for (auto& child : children)
85 m_element2->insertBefore(child, refChild.get());
86
87 // Recover the id attribute of the original element.
88 const AtomString& id = m_element1->getIdAttribute();
89 if (!id.isNull())
90 m_element2->setIdAttribute(id);
91
92 m_element1->remove();
93}
94
95void SplitElementCommand::doReapply()
96{
97 if (!m_element1)
98 return;
99
100 executeApply();
101}

103#ifndef NDEBUG
104void SplitElementCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
105{
106 addNodeAndDescendants(m_element1.get(), nodes);
107 addNodeAndDescendants(m_element2.ptr(), nodes);
108 addNodeAndDescendants(m_atChild.ptr(), nodes);
109}
