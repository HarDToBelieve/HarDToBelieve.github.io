37SplitTextNodeCommand::SplitTextNodeCommand(Ref<Text>&& text, int offset)
38 : SimpleEditCommand(text->document())
39 , m_text2(WTFMove(text))
40 , m_offset(offset)
41{
42 // NOTE: Various callers rely on the fact that the original node becomes
43 // the second node (i.e. the new node is inserted before the existing one).
44 // That is not a fundamental dependency (i.e. it could be re-coded), but
45 // rather is based on how this code happens to work.
46 ASSERT(m_text2->length() > 0);
47 ASSERT(m_offset > 0);
48 ASSERT(m_offset < m_text2->length());
49}
50
51void SplitTextNodeCommand::doApply()
52{
53 ContainerNode* parent = m_text2->parentNode();
54 if (!parent || !parent->hasEditableStyle())
55 return;
56
57 auto result = m_text2->substringData(0, m_offset);
58 if (result.hasException())
59 return;
60 auto prefixText = result.releaseReturnValue();
61 if (prefixText.isEmpty())
62 return;
63
64 m_text1 = Text::create(document(), WTFMove(prefixText));
65 ASSERT(m_text1);
66 document().markers().copyMarkers(m_text2, { 0, m_offset }, *m_text1);
67
68 insertText1AndTrimText2();
69}
70
71void SplitTextNodeCommand::doUnapply()
72{
73 if (!m_text1 || !m_text1->hasEditableStyle())
74 return;
75
76 ASSERT(&m_text1->document() == &document());
77
78 String prefixText = m_text1->data();
79
80 m_text2->insertData(0, prefixText);
81
82 document().markers().copyMarkers(*m_text1, { 0, prefixText.length() }, m_text2);
83 m_text1->remove();
84}
85
86void SplitTextNodeCommand::doReapply()
87{
88 if (!m_text1)
89 return;
90
91 ContainerNode* parent = m_text2->parentNode();
92 if (!parent || !parent->hasEditableStyle())
93 return;
94
95 insertText1AndTrimText2();
96}
97
98void SplitTextNodeCommand::insertText1AndTrimText2()
99{
100 if (m_text2->parentNode()->insertBefore(*m_text1, m_text2.ptr()).hasException())
101 return;
102 m_text2->deleteData(0, m_offset);
103}

107void SplitTextNodeCommand::getNodesInCommand(HashSet<Ref<Node>>& nodes)
108{
109 addNodeAndDescendants(m_text1.get(), nodes);
110 addNodeAndDescendants(m_text2.ptr(), nodes);
111}
