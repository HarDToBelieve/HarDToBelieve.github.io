52inline bool TextManipulationController::ExclusionRule::match(const Element& element) const
53{
54 return switchOn(rule, [&element] (ElementRule rule) {
55 return rule.localName == element.localName();
56 }, [&element] (AttributeRule rule) {
57 return equalIgnoringASCIICase(element.getAttribute(rule.name), rule.value);
58 }, [&element] (ClassRule rule) {
59 return element.hasClass() && element.classNames().contains(rule.className);
60 });
61}

63class ExclusionRuleMatcher {
64public:
65 using ExclusionRule = TextManipulationController::ExclusionRule;
66 using Type = TextManipulationController::ExclusionRule::Type;
67
68 ExclusionRuleMatcher(const Vector<ExclusionRule>& rules)
69 : m_rules(rules)
70 { }
71
72 bool isExcluded(Node* node)
73 {
74 if (!node)
75 return false;
76
77 RefPtr<Element> startingElement = is<Element>(*node) ? downcast<Element>(node) : node->parentElement();
78 if (!startingElement)
79 return false;
80
81 Type type = Type::Include;
82 RefPtr<Element> matchingElement;
83 for (auto& element : lineageOfType<Element>(*startingElement)) {
84 if (auto typeOrNullopt = typeForElement(element)) {
85 type = *typeOrNullopt;
86 matchingElement = &element;
87 break;
88 }
89 }
90
91 for (auto& element : lineageOfType<Element>(*startingElement)) {
92 m_cache.set(element, type);
93 if (&element == matchingElement)
94 break;
95 }
96
97 return type == Type::Exclude;
98 }
99
100 std::optional<Type> typeForElement(Element& element)
101 {
102 auto it = m_cache.find(element);
103 if (it != m_cache.end())
104 return it->value;
105
106 for (auto& rule : m_rules) {
107 if (rule.match(element))
108 return rule.type;
109 }
110
111 return std::nullopt;
112 }
113
114private:
115 const Vector<ExclusionRule>& m_rules;
116 HashMap<Ref<Element>, ExclusionRule::Type> m_cache;
117};

137static bool isInPrivateUseArea(UChar character)
138{
139 return 0xE000 <= character && character <= 0xF8FF;
140}
141
142static bool isTokenDelimiter(UChar character)
143{
144 return isHTMLLineBreak(character) || isInPrivateUseArea(character);
145}
146
147static bool isNotSpace(UChar character)
148{
149 if (character == noBreakSpace)
150 return false;
151
152 return isNotHTMLSpace(character);
153}

155class ParagraphContentIterator {
156public:
157 ParagraphContentIterator(const Position& start, const Position& end)
158 : m_iterator(*makeSimpleRange(start, end), TextIteratorBehavior::IgnoresStyleVisibility)
159 , m_node(start.firstNode())
160 , m_pastEndNode(end.firstNode())
161 {
162 if (shouldAdvanceIteratorPastCurrentNode())
163 advanceIteratorNodeAndUpdateText();
164 }
165
166 void advance()
167 {
168 m_text = std::nullopt;
169 advanceNode();
170
171 if (shouldAdvanceIteratorPastCurrentNode())
172 advanceIteratorNodeAndUpdateText();
173 }

175 struct CurrentContent {
176 RefPtr<Node> node;
177 Vector<String> text;
178 bool isTextContent { false };
179 bool isReplacedContent { false };
180 };
181
182 CurrentContent currentContent()
183 {
184 CurrentContent content = { m_node.copyRef(), m_text ? m_text.value() : Vector<String> { }, !!m_text };
185 if (content.node) {
186 if (auto* renderer = content.node->renderer()) {
187 if (renderer->isRenderReplaced()) {
188 content.isTextContent = false;
189 content.isReplacedContent = true;
190 }
191 }
192 }
193 return content;
194 }
195
196 bool atEnd() const { return !m_text && m_iterator.atEnd() && m_node == m_pastEndNode; }
197
198private:
199 bool shouldAdvanceIteratorPastCurrentNode() const
200 {
201 if (m_iterator.atEnd())
202 return false;
203
204 auto* iteratorNode = m_iterator.node();
205 return !iteratorNode || iteratorNode == m_node;
206 }
207
208 void advanceNode()
209 {
210 if (m_node == m_pastEndNode)
211 return;
212
213 m_node = NodeTraversal::next(*m_node);
214 if (!m_node)
215 m_node = m_pastEndNode;
216 }
217
218 void appendToText(Vector<String>& text, StringBuilder& stringBuilder)
219 {
220 if (!stringBuilder.isEmpty()) {
221 text.append(stringBuilder.toString());
222 stringBuilder.clear();
223 }
224 }
225
226 void advanceIteratorNodeAndUpdateText()
227 {
228 ASSERT(shouldAdvanceIteratorPastCurrentNode());
229
230 StringBuilder stringBuilder;
231 Vector<String> text;
232 while (shouldAdvanceIteratorPastCurrentNode()) {
233 auto iteratorText = m_iterator.text();
234 if (m_iterator.range().collapsed()) {
235 if (iteratorText == "\n") {
236 appendToText(text, stringBuilder);
237 text.append({ });
238 }
239 } else
240 stringBuilder.append(iteratorText);
241
242 m_iterator.advance();
243 }
244 appendToText(text, stringBuilder);
245 m_text = text;
246 }
247
248 TextIterator m_iterator;
249 RefPtr<Node> m_node;
250 RefPtr<Node> m_pastEndNode;
251 std::optional<Vector<String>> m_text;
252};

254static bool shouldExtractValueForTextManipulation(const HTMLInputElement& input)
255{
256 if (input.isSearchField() || equalIgnoringASCIICase(input.attributeWithoutSynchronization(HTMLNames::typeAttr), InputTypeNames::text()))
257 return !input.lastChangeWasUserEdit();
258
259 return input.isTextButton();
260}

262static bool isAttributeForTextManipulation(const QualifiedName& nameToCheck)
263{
264 using namespace HTMLNames;
265 static const QualifiedName* const attributeNames[] = {
266 &titleAttr.get(),
267 &altAttr.get(),
268 &placeholderAttr.get(),
269 &aria_labelAttr.get(),
270 &aria_placeholderAttr.get(),
271 &aria_roledescriptionAttr.get(),
272 &aria_valuetextAttr.get(),
273 };
274 for (auto& entry : attributeNames) {

276 return true;
277 }
278 return false;

291static std::optional<TextManipulationController::ManipulationTokenInfo> tokenInfo(Node* node)
292{
293 if (!node)
294 return std::nullopt;
295
296 TextManipulationController::ManipulationTokenInfo result;
297 result.documentURL = node->document().url();
298 if (auto element = is<Element>(node) ? makeRefPtr(downcast<Element>(*node)) : makeRefPtr(node->parentElement())) {
299 result.tagName = element->tagName();
300 if (element->hasAttributeWithoutSynchronization(HTMLNames::roleAttr))
301 result.roleAttribute = element->attributeWithoutSynchronization(HTMLNames::roleAttr);
302 if (auto frame = makeRefPtr(node->document().frame()); frame && frame->view() && element->renderer()) {
303 // FIXME: This doesn't account for overflow clip.
304 auto elementRect = element->renderer()->absoluteAnchorRect();
305 auto visibleContentRect = frame->view()->visibleContentRect();
306 result.isVisible = visibleContentRect.intersects(enclosingIntRect(elementRect));
307 }
308 }
309 return result;
310}

312static bool isEnclosingItemBoundaryElement(const Element& element)
313{
314 auto* renderer = element.renderer();
315 if (!renderer)
316 return false;
317
318 auto role = [](const Element& element) -> AccessibilityRole {
319 return AccessibilityObject::ariaRoleToWebCoreRole(element.attributeWithoutSynchronization(HTMLNames::roleAttr));
320 };
321
322 if (element.hasTagName(HTMLNames::buttonTag) || role(element) == AccessibilityRole::Button)
323 return true;
324
325 auto displayType = renderer->style().display();
326 if (element.hasTagName(HTMLNames::liTag) || element.hasTagName(HTMLNames::aTag)) {
327 if (displayType == DisplayType::Block || displayType == DisplayType::InlineBlock)
328 return true;
329
330 for (auto parent = makeRefPtr(element.parentElement()); parent; parent = parent->parentElement()) {
331 if (parent->hasTagName(HTMLNames::navTag) || role(*parent) == AccessibilityRole::LandmarkNavigation)
332 return true;
333 }
334 }
335
336 if (displayType == DisplayType::TableCell)
337 return true;
338
339 if (element.hasTagName(HTMLNames::spanTag) && displayType == DisplayType::InlineBlock)
340 return true;
341
342 if (displayType == DisplayType::Block && (element.hasTagName(HTMLNames::h1Tag) || element.hasTagName(HTMLNames::h2Tag) || element.hasTagName(HTMLNames::h3Tag)
343 || element.hasTagName(HTMLNames::h4Tag) || element.hasTagName(HTMLNames::h5Tag) || element.hasTagName(HTMLNames::h6Tag)))
344 return true;
345
346 return false;
347}

349TextManipulationController::ManipulationUnit TextManipulationController::createUnit(const Vector<String>& text, Node& textNode)
350{
351 ManipulationUnit unit = { textNode, { } };
352 for (auto& textEntry : text) {
353 if (!textEntry.isNull())
354 parse(unit, textEntry, textNode);
355 else {
356 if (unit.tokens.isEmpty())
357 unit.firstTokenContainsDelimiter = true;
358 unit.lastTokenContainsDelimiter = true;
359 }
360 }
361 return unit;
362}

364bool TextManipulationController::shouldExcludeNodeBasedOnStyle(const Node& node)
365{
366 auto* style = node.renderStyle();
367 if (!style)
368 return false;
369
370 auto& font = style->fontCascade().primaryFont();
371 auto familyName = font.platformData().familyName();
372 if (familyName.isEmpty())
373 return false;
374
375 auto iter = m_cachedFontFamilyExclusionResults.find(familyName);
376 if (iter != m_cachedFontFamilyExclusionResults.end())
377 return iter->value;
378
379 // FIXME: We should reconsider whether a node should be excluded if the primary font
380 // used to render the node changes, since this "icon font" heuristic may return a
381 // different result.
382 bool result = font.isProbablyOnlyUsedToRenderIcons();
383 m_cachedFontFamilyExclusionResults.set(familyName, result);
384 return result;
385}

387void TextManipulationController::parse(ManipulationUnit& unit, const String& text, Node& textNode)
388{
389 ExclusionRuleMatcher exclusionRuleMatcher(m_exclusionRules);
390 bool isNodeExcluded = exclusionRuleMatcher.isExcluded(&textNode) || shouldExcludeNodeBasedOnStyle(textNode);
391 size_t positionOfLastNonHTMLSpace = WTF::notFound;
392 size_t startPositionOfCurrentToken = 0;
393 size_t index = 0;
394 for (; index < text.length(); ++index) {
395 auto character = text[index];
396 if (isTokenDelimiter(character)) {
397 if (positionOfLastNonHTMLSpace != WTF::notFound && startPositionOfCurrentToken <= positionOfLastNonHTMLSpace) {
398 auto stringForToken = text.substring(startPositionOfCurrentToken, positionOfLastNonHTMLSpace + 1 - startPositionOfCurrentToken);
399 unit.tokens.append(ManipulationToken { m_tokenIdentifier.generate(), stringForToken, tokenInfo(&textNode), isNodeExcluded });
400 startPositionOfCurrentToken = positionOfLastNonHTMLSpace + 1;
401 }
402
403 while (index < text.length() && (isHTMLSpace(text[index]) || isInPrivateUseArea(text[index])))
404 ++index;
405
406 --index;
407
408 auto stringForToken = text.substring(startPositionOfCurrentToken, index + 1 - startPositionOfCurrentToken);
409 if (unit.tokens.isEmpty() && !unit.firstTokenContainsDelimiter)
410 unit.firstTokenContainsDelimiter = true;
411 unit.tokens.append(ManipulationToken { m_tokenIdentifier.generate(), stringForToken, tokenInfo(&textNode), true });
412 startPositionOfCurrentToken = index + 1;
413 unit.lastTokenContainsDelimiter = true;
414 } else if (isNotSpace(character)) {
415 if (!isNodeExcluded)
416 unit.areAllTokensExcluded = false;
417 positionOfLastNonHTMLSpace = index;
418 }
419 }
420
421 if (startPositionOfCurrentToken < text.length()) {
422 auto stringForToken = text.substring(startPositionOfCurrentToken, index + 1 - startPositionOfCurrentToken);
423 unit.tokens.append(ManipulationToken { m_tokenIdentifier.generate(), stringForToken, tokenInfo(&textNode), isNodeExcluded });
424 unit.lastTokenContainsDelimiter = false;
425 }
426}

428void TextManipulationController::addItemIfPossible(Vector<ManipulationUnit>&& units)
429{
430 if (units.isEmpty())
431 return;
432
433 size_t index = 0;
434 size_t end = units.size();
435 while (index < units.size() && units[index].areAllTokensExcluded)
436 ++index;
437
438 while (end > 0 && units[end - 1].areAllTokensExcluded)
439 --end;
440
441 if (index == end)
442 return;
443
444 ASSERT(end);
445 auto startPosition = firstPositionInOrBeforeNode(units[index].node.ptr());
446 auto endPosition = positionAfterNode(units[end - 1].node.ptr());
447 Vector<ManipulationToken> tokens;
448 for (; index < end; ++index)
449 tokens.appendVector(WTFMove(units[index].tokens));
450
451 addItem(ManipulationItemData { startPosition, endPosition, nullptr, nullQName(), WTFMove(tokens) });
452}

454void TextManipulationController::observeParagraphs(const Position& start, const Position& end)
455{
456 if (start.isNull() || end.isNull() || start.isOrphan() || end.isOrphan())
457 return;
458
459 auto document = makeRefPtr(start.document());
460 ASSERT(document);
461 // TextIterator's constructor may have updated the layout and executed arbitrary scripts.
462 if (document != start.document() || document != end.document())
463 return;
464
465 Vector<ManipulationUnit> unitsInCurrentParagraph;
466 Vector<Ref<Element>> enclosingItemBoundaryElements;
467 ParagraphContentIterator iterator { start, end };
468 for (; !iterator.atEnd(); iterator.advance()) {
469 auto content = iterator.currentContent();
470 auto* contentNode = content.node.get();
471 ASSERT(contentNode);
472
473 while (!enclosingItemBoundaryElements.isEmpty() && !enclosingItemBoundaryElements.last()->contains(contentNode)) {
474 addItemIfPossible(std::exchange(unitsInCurrentParagraph, { }));
475 enclosingItemBoundaryElements.removeLast();
476 }
477
478 if (m_manipulatedNodes.contains(*contentNode)) {
479 addItemIfPossible(std::exchange(unitsInCurrentParagraph, { }));
480 continue;
481 }
482
483 if (is<Element>(*contentNode)) {
484 auto& currentElement = downcast<Element>(*contentNode);
485 if (!content.isTextContent && canPerformTextManipulationByReplacingEntireTextContent(currentElement))
486 addItem(ManipulationItemData { Position(), Position(), makeWeakPtr(currentElement), nullQName(), { ManipulationToken { m_tokenIdentifier.generate(), currentElement.textContent(), tokenInfo(&currentElement) } } });
487
488 if (currentElement.hasAttributes()) {
489 for (auto& attribute : currentElement.attributesIterator()) {
490 if (isAttributeForTextManipulation(attribute.name()))
491 addItem(ManipulationItemData { Position(), Position(), makeWeakPtr(currentElement), attribute.name(), { ManipulationToken { m_tokenIdentifier.generate(), attribute.value(), tokenInfo(&currentElement) } } });
492 }
493 }
494
495 if (is<HTMLInputElement>(currentElement)) {
496 auto& input = downcast<HTMLInputElement>(currentElement);
497 if (shouldExtractValueForTextManipulation(input))
498 addItem(ManipulationItemData { { }, { }, makeWeakPtr(currentElement), HTMLNames::valueAttr, { ManipulationToken { m_tokenIdentifier.generate(), input.value(), tokenInfo(&currentElement) } } });
499 }
500
501 if (isEnclosingItemBoundaryElement(currentElement)) {
502 addItemIfPossible(std::exchange(unitsInCurrentParagraph, { }));
503 enclosingItemBoundaryElements.append(currentElement);
504 }
505 }
506
507 if (content.isReplacedContent) {
508 if (!unitsInCurrentParagraph.isEmpty())
509 unitsInCurrentParagraph.append(ManipulationUnit { *contentNode, { ManipulationToken { m_tokenIdentifier.generate(), "[]", tokenInfo(content.node.get()), true } } });
510 continue;
511 }
512
513 if (!content.isTextContent)
514 continue;
515
516 auto currentUnit = createUnit(content.text, *contentNode);
517 if (currentUnit.firstTokenContainsDelimiter)
518 addItemIfPossible(std::exchange(unitsInCurrentParagraph, { }));
519
520 if (unitsInCurrentParagraph.isEmpty() && currentUnit.areAllTokensExcluded)
521 continue;
522
523 bool currentUnitEndsWithDelimiter = currentUnit.lastTokenContainsDelimiter;
524 unitsInCurrentParagraph.append(WTFMove(currentUnit));
525
526 if (currentUnitEndsWithDelimiter)
527 addItemIfPossible(std::exchange(unitsInCurrentParagraph, { }));
528 }
529
530 addItemIfPossible(std::exchange(unitsInCurrentParagraph, { }));
531}

567void TextManipulationController::scheduleObservationUpdate()
568{
569 if (m_didScheduleObservationUpdate)
570 return;
571
572 if (!m_document)
573 return;
574
575 m_didScheduleObservationUpdate = true;
576
577 m_document->eventLoop().queueTask(TaskSource::InternalAsyncTask, [weakThis = makeWeakPtr(*this)] {
578 auto* controller = weakThis.get();
579 if (!controller)
580 return;
581
582 controller->m_didScheduleObservationUpdate = false;
583
584 HashSet<Ref<Node>> nodesToObserve;
585 for (auto& weakElement : controller->m_elementsWithNewRenderer)
586 nodesToObserve.add(weakElement);
587 controller->m_elementsWithNewRenderer.clear();
588
589 for (auto& text : controller->m_manipulatedTextsWithNewContent) {
590 if (!controller->m_manipulatedNodes.contains(text))
591 continue;
592 controller->m_manipulatedNodes.remove(text);
593 nodesToObserve.add(text);
594 }
595 controller->m_manipulatedTextsWithNewContent.clear();
596
597 for (auto& text : controller->m_textNodesWithNewRenderer)
598 nodesToObserve.add(text);
599 controller->m_textNodesWithNewRenderer.clear();
600
601 if (nodesToObserve.isEmpty())
602 return;
603
604 RefPtr<Node> commonAncestor;
605 for (auto& node : nodesToObserve) {
606 if (!node->isConnected())
607 continue;
608
609 if (auto host = makeRefPtr(node->shadowHost()); is<HTMLInputElement>(host.get()) && downcast<HTMLInputElement>(*host).lastChangeWasUserEdit())
610 continue;
611
612 if (!commonAncestor)
613 commonAncestor = is<ContainerNode>(node.get()) ? node.ptr() : node->parentNode();
614 else if (!node->isDescendantOf(commonAncestor.get()))
615 commonAncestor = commonInclusiveAncestor<ComposedTree>(*commonAncestor, node.get());
616 }
617
618 auto start = firstPositionInOrBeforeNode(commonAncestor.get());
619 auto end = lastPositionInOrAfterNode(commonAncestor.get());
620 controller->observeParagraphs(start, end);
621
622 if (controller->m_items.isEmpty() && commonAncestor) {
623 controller->m_manipulatedNodes.add(commonAncestor.get());
624 return;
625 }
626
627 controller->flushPendingItemsForCallback();
628 });
629}

631void TextManipulationController::addItem(ManipulationItemData&& itemData)
632{
633 const unsigned itemCallbackBatchingSize = 128;
634
635 ASSERT(m_document);
636 ASSERT(!itemData.tokens.isEmpty());
637 auto newID = m_itemIdentifier.generate();
638 m_pendingItemsForCallback.append(ManipulationItem {
639 newID,
640 itemData.tokens.map([](auto& token) { return token; })
641 });
642 m_items.add(newID, WTFMove(itemData));
643
644 if (m_pendingItemsForCallback.size() >= itemCallbackBatchingSize)
645 flushPendingItemsForCallback();
646}

648void TextManipulationController::flushPendingItemsForCallback()
649{
650 if (m_pendingItemsForCallback.isEmpty())
651 return;
652
653 m_callback(*m_document, m_pendingItemsForCallback);
654 m_pendingItemsForCallback.clear();
655}

657auto TextManipulationController::completeManipulation(const Vector<WebCore::TextManipulationController::ManipulationItem>& completionItems) -> Vector<ManipulationFailure>
658{
659 Vector<ManipulationFailure> failures;
660 HashSet<Ref<Node>> containersWithoutVisualOverflowBeforeReplacement;
661 for (unsigned i = 0; i < completionItems.size(); ++i) {
662 auto& itemToComplete = completionItems[i];
663 auto identifier = itemToComplete.identifier;
664 if (!identifier) {
665 failures.append(ManipulationFailure { identifier, i, ManipulationFailureType::InvalidItem });
666 continue;
667 }
668
669 auto itemDataIterator = m_items.find(identifier);
670 if (itemDataIterator == m_items.end()) {
671 failures.append(ManipulationFailure { identifier, i, ManipulationFailureType::InvalidItem });
672 continue;
673 }
674
675 ManipulationItemData itemData;
676 std::exchange(itemData, itemDataIterator->value);
677 m_items.remove(itemDataIterator);
678
679 auto failureOrNullopt = replace(itemData, itemToComplete.tokens, containersWithoutVisualOverflowBeforeReplacement);
680 if (failureOrNullopt)
681 failures.append(ManipulationFailure { identifier, i, *failureOrNullopt });
682 }
683
684 if (!containersWithoutVisualOverflowBeforeReplacement.isEmpty()) {
685 if (m_document)
686 m_document->updateLayoutIgnorePendingStylesheets();
687
688 for (auto& container : containersWithoutVisualOverflowBeforeReplacement) {
689 if (!is<StyledElement>(container))
690 continue;
691
692 auto& element = downcast<StyledElement>(container.get());
693 auto* box = element.renderBox();
694 if (!box || !box->hasVisualOverflow())
695 continue;
696
697 auto& style = box->style();
698 if (style.width().isFixed() && style.height().isFixed() && !style.hasOutOfFlowPosition() && !style.hasClip()) {
699 element.setInlineStyleProperty(CSSPropertyOverflowX, CSSValueHidden);
700 element.setInlineStyleProperty(CSSPropertyOverflowY, CSSValueAuto);
701 }
702 }
703 }
704
705 return failures;
706}

720Vector<Ref<Node>> TextManipulationController::getPath(Node* ancestor, Node* node)
721{
722 Vector<Ref<Node>> path;
723 RefPtr<ContainerNode> containerNode = is<ContainerNode>(*node) ? &downcast<ContainerNode>(*node) : node->parentNode();
724 for (; containerNode && containerNode != ancestor; containerNode = containerNode->parentNode())
725 path.append(*containerNode);
726 path.reverse();
727 return path;
728}

730void TextManipulationController::updateInsertions(Vector<NodeEntry>& lastTopDownPath, const Vector<Ref<Node>>& currentTopDownPath, Node* currentNode, HashSet<Ref<Node>>& insertedNodes, Vector<NodeInsertion>& insertions)
731{
732 size_t i = 0;
733 while (i < lastTopDownPath.size() && i < currentTopDownPath.size() && lastTopDownPath[i].first.ptr() == currentTopDownPath[i].ptr())
734 ++i;
735
736 if (i != lastTopDownPath.size() || i != currentTopDownPath.size()) {
737 if (i < lastTopDownPath.size())
738 lastTopDownPath.shrink(i);
739
740 for (;i < currentTopDownPath.size(); ++i) {
741 Ref<Node> node = currentTopDownPath[i];
742 if (!insertedNodes.add(node.copyRef()).isNewEntry) {
743 auto clonedNode = node->cloneNodeInternal(node->document(), Node::CloningOperation::OnlySelf);
744 if (auto* data = node->eventTargetData())
745 data->eventListenerMap.copyEventListenersNotCreatedFromMarkupToTarget(clonedNode.ptr());
746 node = WTFMove(clonedNode);
747 }
748 insertions.append(NodeInsertion { lastTopDownPath.size() ? lastTopDownPath.last().second.ptr() : nullptr, node.copyRef() });
749 lastTopDownPath.append({ currentTopDownPath[i].copyRef(), WTFMove(node) });
750 }
751 }
752
753 if (currentNode)
754 insertions.append(NodeInsertion { lastTopDownPath.size() ? lastTopDownPath.last().second.ptr() : nullptr, *currentNode });
755}

757auto TextManipulationController::replace(const ManipulationItemData& item, const Vector<ManipulationToken>& replacementTokens, HashSet<Ref<Node>>& containersWithoutVisualOverflowBeforeReplacement) -> std::optional<ManipulationFailureType>
758{
759 if (item.start.isOrphan() || item.end.isOrphan())
760 return ManipulationFailureType::ContentChanged;
761
762 if (item.start.isNull() || item.end.isNull()) {
763 RELEASE_ASSERT(item.tokens.size() == 1);
764 auto element = makeRefPtr(item.element.get());
765 if (!element)
766 return ManipulationFailureType::ContentChanged;
767 if (replacementTokens.size() > 1 && !canPerformTextManipulationByReplacingEntireTextContent(*element) && item.attributeName == nullQName())
768 return ManipulationFailureType::InvalidToken;
769 auto expectedTokenIdentifier = item.tokens[0].identifier;
770 StringBuilder newValue;
771 for (size_t i = 0; i < replacementTokens.size(); ++i) {
772 if (replacementTokens[i].identifier != expectedTokenIdentifier)
773 return ManipulationFailureType::InvalidToken;
774 if (i)
775 newValue.append(' ');
776 newValue.append(replacementTokens[i].content);
777 }
778 if (item.attributeName == nullQName())
779 element->setTextContent(newValue.toString());
780 else if (item.attributeName == HTMLNames::valueAttr && is<HTMLInputElement>(*element))
781 downcast<HTMLInputElement>(*element).setValue(newValue.toString());
782 else
783 element->setAttribute(item.attributeName, newValue.toString());
784 return std::nullopt;
785 }
786
787 size_t currentTokenIndex = 0;
788 HashMap<TokenIdentifier, TokenExchangeData> tokenExchangeMap;
789 RefPtr<Node> commonAncestor;
790 RefPtr<Node> firstContentNode;
791 RefPtr<Node> lastChildOfCommonAncestorInRange;
792 HashSet<Ref<Node>> nodesToRemove;
793 ParagraphContentIterator iterator { item.start, item.end };
794 for (; !iterator.atEnd(); iterator.advance()) {
795 auto content = iterator.currentContent();
796 ASSERT(content.node);
797
798 lastChildOfCommonAncestorInRange = content.node;
799 nodesToRemove.add(*content.node);
800
801 if (!content.isReplacedContent && !content.isTextContent)
802 continue;
803
804 Vector<ManipulationToken> tokensInCurrentNode;
805 if (content.isReplacedContent) {
806 if (currentTokenIndex >= item.tokens.size())
807 return ManipulationFailureType::ContentChanged;
808
809 tokensInCurrentNode.append(item.tokens[currentTokenIndex]);
810 } else
811 tokensInCurrentNode = createUnit(content.text, *content.node).tokens;
812
813 bool isNodeIncluded = WTF::anyOf(tokensInCurrentNode, [] (auto& token) {
814 return !token.isExcluded;
815 });
816 for (auto& token : tokensInCurrentNode) {
817 if (currentTokenIndex >= item.tokens.size())
818 return ManipulationFailureType::ContentChanged;
819
820 auto& currentToken = item.tokens[currentTokenIndex++];
821 bool isContentUnchanged = areEqualIgnoringLeadingAndTrailingWhitespaces(currentToken.content, token.content);
822 if (!content.isReplacedContent && !isContentUnchanged)
823 return ManipulationFailureType::ContentChanged;
824
825 tokenExchangeMap.set(currentToken.identifier, TokenExchangeData { content.node.copyRef(), currentToken.content, !isNodeIncluded });
826 }
827
828 if (!firstContentNode)
829 firstContentNode = content.node;
830
831 auto parentNode = content.node->parentNode();
832 if (!commonAncestor)
833 commonAncestor = parentNode;
834 else if (!parentNode->isDescendantOf(commonAncestor.get())) {
835 commonAncestor = commonInclusiveAncestor<ComposedTree>(*commonAncestor, *parentNode);
836 ASSERT(commonAncestor);
837 }
838 }
839
840 if (!firstContentNode)
841 return ManipulationFailureType::ContentChanged;
842
843 while (lastChildOfCommonAncestorInRange && lastChildOfCommonAncestorInRange->parentNode() != commonAncestor)
844 lastChildOfCommonAncestorInRange = lastChildOfCommonAncestorInRange->parentNode();
845
846 for (auto node = commonAncestor; node; node = node->parentNode())
847 nodesToRemove.remove(*node);
848
849 HashSet<Ref<Node>> reusedOriginalNodes;
850 Vector<NodeInsertion> insertions;
851 auto startTopDownPath = getPath(commonAncestor.get(), firstContentNode.get());
852 while (!startTopDownPath.isEmpty()) {
853 auto lastNode = startTopDownPath.last();
854 ASSERT(is<ContainerNode>(lastNode.get()));
855 if (!downcast<ContainerNode>(lastNode.get()).hasOneChild())
856 break;
857 nodesToRemove.add(startTopDownPath.takeLast());
858 }
859 auto lastTopDownPath = startTopDownPath.map([&](auto node) -> NodeEntry {
860 reusedOriginalNodes.add(node.copyRef());
861 return { node, node };
862 });
863
864 for (size_t index = 0; index < replacementTokens.size(); ++index) {
865 auto& replacementToken = replacementTokens[index];
866 auto it = tokenExchangeMap.find(replacementToken.identifier);
867 if (it == tokenExchangeMap.end())
868 return ManipulationFailureType::InvalidToken;
869
870 auto& exchangeData = it->value;
871 auto* originalNode = exchangeData.node.get();
872 ASSERT(originalNode);
873 auto replacementText = replacementToken.content;
874
875 RefPtr<Node> replacementNode;
876 if (exchangeData.isExcluded) {
877 if (exchangeData.isConsumed)
878 return ManipulationFailureType::ExclusionViolation;
879 exchangeData.isConsumed = true;
880
881 if (!replacementToken.content.isNull() && replacementToken.content != exchangeData.originalContent)
882 return ManipulationFailureType::ExclusionViolation;
883
884 replacementNode = originalNode;
885 for (RefPtr<Node> descendentNode = NodeTraversal::next(*originalNode, originalNode); descendentNode; descendentNode = NodeTraversal::next(*descendentNode, originalNode))
886 nodesToRemove.remove(*descendentNode);
887 } else
888 replacementNode = Text::create(commonAncestor->document(), replacementText);
889
890 auto topDownPath = getPath(commonAncestor.get(), originalNode);
891 updateInsertions(lastTopDownPath, topDownPath, replacementNode.get(), reusedOriginalNodes, insertions);
892 }
893
894 RefPtr<Node> node = item.end.firstNode();
895 if (node && lastChildOfCommonAncestorInRange->contains(node.get())) {
896 auto topDownPath = getPath(commonAncestor.get(), node->parentNode());
897 updateInsertions(lastTopDownPath, topDownPath, nullptr, reusedOriginalNodes, insertions);
898 }
899 while (lastChildOfCommonAncestorInRange->contains(node.get())) {
900 Ref<Node> parentNode = *node->parentNode();
901 while (!lastTopDownPath.isEmpty() && lastTopDownPath.last().first.ptr() != parentNode.ptr())
902 lastTopDownPath.removeLast();
903
904 insertions.append(NodeInsertion { lastTopDownPath.size() ? lastTopDownPath.last().second.ptr() : nullptr, *node, IsNodeManipulated::No });
905 lastTopDownPath.append({ *node, *node });
906 node = NodeTraversal::next(*node);
907 }
908
909 RefPtr<Node> insertionPointNode = lastChildOfCommonAncestorInRange->nextSibling();
910
911 for (auto& node : nodesToRemove)
912 node->remove();
913
914 for (auto& insertion : insertions) {
915 auto parentContainer = insertion.parentIfDifferentFromCommonAncestor;
916 if (!parentContainer) {
917 parentContainer = commonAncestor;
918 parentContainer->insertBefore(insertion.child, insertionPointNode.get());
919 } else
920 parentContainer->appendChild(insertion.child);
921
922 if (auto* box = parentContainer->renderBox()) {
923 if (!box->hasVisualOverflow())
924 containersWithoutVisualOverflowBeforeReplacement.add(*parentContainer);
925 }
926
927 if (insertion.isChildManipulated == IsNodeManipulated::Yes)
928 m_manipulatedNodes.add(insertion.child.get());
929 }
930
931 return std::nullopt;
932}

937 m_textNodesWithNewRenderer.remove(node);
