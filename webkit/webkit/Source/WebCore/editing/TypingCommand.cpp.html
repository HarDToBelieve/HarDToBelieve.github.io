56class TypingCommandLineOperation
57{
58public:
59 TypingCommandLineOperation(TypingCommand* typingCommand, bool selectInsertedText, const String& text)
60 : m_typingCommand(typingCommand)
61 , m_selectInsertedText(selectInsertedText)
62 , m_text(text)
63 { }
64
65 void operator()(size_t lineOffset, size_t lineLength, bool isLastLine) const
66 {
67 if (isLastLine) {
68 if (!lineOffset || lineLength > 0)
69 m_typingCommand->insertTextRunWithoutNewlines(m_text.substring(lineOffset, lineLength), m_selectInsertedText);
70 } else {
71 if (lineLength > 0)
72 m_typingCommand->insertTextRunWithoutNewlines(m_text.substring(lineOffset, lineLength), false);
73 m_typingCommand->insertParagraphSeparator();
74 }
75 }
76
77private:
78 TypingCommand* m_typingCommand;
79 bool m_selectInsertedText;
80 const String& m_text;
81};

83static inline EditAction editActionForTypingCommand(TypingCommand::ETypingCommand command, TextGranularity granularity, TypingCommand::TextCompositionType compositionType, bool isAutocompletion)
84{
85 if (compositionType == TypingCommand::TextCompositionPending) {
86 if (command == TypingCommand::InsertText)
87 return EditAction::TypingInsertPendingComposition;
88 if (command == TypingCommand::DeleteSelection)
89 return EditAction::TypingDeletePendingComposition;
90 ASSERT_NOT_REACHED();
91 }
92
93 if (compositionType == TypingCommand::TextCompositionFinal) {
94 if (command == TypingCommand::InsertText)
95 return EditAction::TypingInsertFinalComposition;
96 if (command == TypingCommand::DeleteSelection)
97 return EditAction::TypingDeleteFinalComposition;
98 ASSERT_NOT_REACHED();
99 }
100
101 switch (command) {
102 case TypingCommand::DeleteSelection:
103 return EditAction::TypingDeleteSelection;
104 case TypingCommand::DeleteKey: {
105 if (granularity == TextGranularity::WordGranularity)
106 return EditAction::TypingDeleteWordBackward;
107 if (granularity == TextGranularity::LineBoundary)
108 return EditAction::TypingDeleteLineBackward;
109 return EditAction::TypingDeleteBackward;
110 }
111 case TypingCommand::ForwardDeleteKey:
112 if (granularity == TextGranularity::WordGranularity)
113 return EditAction::TypingDeleteWordForward;
114 if (granularity == TextGranularity::LineBoundary)
115 return EditAction::TypingDeleteLineForward;
116 return EditAction::TypingDeleteForward;
117 case TypingCommand::InsertText:
118 return isAutocompletion ? EditAction::InsertReplacement : EditAction::TypingInsertText;
119 case TypingCommand::InsertLineBreak:
120 return EditAction::TypingInsertLineBreak;
121 case TypingCommand::InsertParagraphSeparator:
122 case TypingCommand::InsertParagraphSeparatorInQuotedContent:
123 return EditAction::TypingInsertParagraph;
124 default:
125 return EditAction::Unspecified;
126 }
127}

129static inline bool editActionIsDeleteByTyping(EditAction action)
130{
131 switch (action) {
132 case EditAction::TypingDeleteSelection:
133 case EditAction::TypingDeleteBackward:
134 case EditAction::TypingDeleteWordBackward:
135 case EditAction::TypingDeleteLineBackward:
136 case EditAction::TypingDeleteForward:
137 case EditAction::TypingDeleteWordForward:
138 case EditAction::TypingDeleteLineForward:
139 return true;
140 default:
141 return false;
142 }

165void TypingCommand::deleteSelection(Document& document, Options options, TextCompositionType compositionType)
166{
167 if (!document.selection().isRange())
168 return;
169
170 if (RefPtr<TypingCommand> lastTypingCommand = lastTypingCommandIfStillOpenForTyping(document)) {
171 lastTypingCommand->setIsAutocompletion(options & IsAutocompletion);
172 lastTypingCommand->setCompositionType(compositionType);
173 lastTypingCommand->setShouldPreventSpellChecking(options & PreventSpellChecking);
174 lastTypingCommand->deleteSelection(options & SmartDelete);
175 return;
176 }
177
178 TypingCommand::create(document, DeleteSelection, emptyString(), options, compositionType)->apply();
179}

181void TypingCommand::deleteKeyPressed(Document& document, Options options, TextGranularity granularity)
182{
183 if (granularity == TextGranularity::CharacterGranularity) {
184 if (RefPtr<TypingCommand> lastTypingCommand = lastTypingCommandIfStillOpenForTyping(document)) {
185 updateSelectionIfDifferentFromCurrentSelection(lastTypingCommand.get(), document);
186 lastTypingCommand->setIsAutocompletion(options & IsAutocompletion);
187 lastTypingCommand->setCompositionType(TextCompositionNone);
188 lastTypingCommand->setShouldPreventSpellChecking(options & PreventSpellChecking);
189 lastTypingCommand->deleteKeyPressed(granularity, options & AddsToKillRing);
190 return;
191 }
192 }
193
194 TypingCommand::create(document, DeleteKey, emptyString(), options, granularity)->apply();
195}

232// FIXME: We shouldn't need to take selectionForInsertion. It should be identical to FrameSelection's current selection.
233void TypingCommand::insertText(Document& document, const String& text, const VisibleSelection& selectionForInsertion, Options options, TextCompositionType compositionType)
234{
235 LOG(Editing, "TypingCommand::insertText (text %s)", text.utf8().data());
236
237 VisibleSelection currentSelection = document.selection().selection();
238
239 String newText = dispatchBeforeTextInsertedEvent(text, selectionForInsertion, compositionType == TextCompositionPending);
240
241 // Set the starting and ending selection appropriately if we are using a selection
242 // that is different from the current selection. In the future, we should change EditCommand
243 // to deal with custom selections in a general way that can be used by all of the commands.
244 if (RefPtr<TypingCommand> lastTypingCommand = lastTypingCommandIfStillOpenForTyping(document)) {
245 if (lastTypingCommand->endingSelection() != selectionForInsertion) {
246 lastTypingCommand->setStartingSelection(selectionForInsertion);
247 lastTypingCommand->setEndingSelection(selectionForInsertion);
248 }
249
250 lastTypingCommand->setIsAutocompletion(options & IsAutocompletion);
251 lastTypingCommand->setCompositionType(compositionType);
252 lastTypingCommand->setShouldRetainAutocorrectionIndicator(options & RetainAutocorrectionIndicator);
253 lastTypingCommand->setShouldPreventSpellChecking(options & PreventSpellChecking);
254 lastTypingCommand->insertTextAndNotifyAccessibility(newText, options & SelectInsertedText);
255 return;
256 }
257
258 auto cmd = TypingCommand::create(document, InsertText, newText, options, compositionType);
259 applyTextInsertionCommand(document.frame(), cmd.get(), selectionForInsertion, currentSelection);
260}

301{

351void TypingCommand::doApply()
352{
353 if (endingSelection().isNoneOrOrphaned())
354 return;
355
356 if (m_commandType == DeleteKey)
357 if (m_commands.isEmpty())
358 m_openedByBackwardDelete = true;
359
360 switch (m_commandType) {
361 case DeleteSelection:
362 deleteSelection(m_smartDelete);
363 return;
364 case DeleteKey:
365 deleteKeyPressed(m_granularity, m_shouldAddToKillRing);
366 return;
367 case ForwardDeleteKey:
368 forwardDeleteKeyPressed(m_granularity, m_shouldAddToKillRing);
369 return;
370 case InsertLineBreak:
371 insertLineBreakAndNotifyAccessibility();
372 return;
373 case InsertParagraphSeparator:
374 insertParagraphSeparatorAndNotifyAccessibility();
375 return;
376 case InsertParagraphSeparatorInQuotedContent:
377 insertParagraphSeparatorInQuotedContentAndNotifyAccessibility();
378 return;
379 case InsertText:
380 insertTextAndNotifyAccessibility(m_textToInsert, m_selectInsertedText);
381 return;
382 }
383
384 ASSERT_NOT_REACHED();
385}

427void TypingCommand::markMisspellingsAfterTyping(ETypingCommand commandType)
428{
429#if PLATFORM(MAC)
430 if (!document().editor().isContinuousSpellCheckingEnabled()
431 && !document().editor().isAutomaticQuoteSubstitutionEnabled()
432 && !document().editor().isAutomaticLinkDetectionEnabled()
433 && !document().editor().isAutomaticDashSubstitutionEnabled()
434 && !document().editor().isAutomaticTextReplacementEnabled())
435 return;
436 if (document().editor().isHandlingAcceptedCandidate())
437 return;
438#else
439 if (!document().editor().isContinuousSpellCheckingEnabled())
440 return;
441#endif
442 // Take a look at the selection that results after typing and determine whether we need to spellcheck.
443 // Since the word containing the current selection is never marked, this does a check to
444 // see if typing made a new word that is not in the current selection. Basically, you
445 // get this by being at the end of a word and typing a space.
446 VisiblePosition start(endingSelection().start(), endingSelection().affinity());
447 VisiblePosition previous = start.previous();
448 if (previous.isNotNull()) {
449#if !PLATFORM(IOS_FAMILY)
450 VisiblePosition p1 = startOfWord(previous, LeftWordIfOnBoundary);
451 VisiblePosition p2 = startOfWord(start, LeftWordIfOnBoundary);
452 if (p1 != p2) {
453 auto range = makeSimpleRange(p1, p2);
454 String strippedPreviousWord;
455 if (range && (commandType == TypingCommand::InsertText || commandType == TypingCommand::InsertLineBreak || commandType == TypingCommand::InsertParagraphSeparator || commandType == TypingCommand::InsertParagraphSeparatorInQuotedContent))
456 strippedPreviousWord = plainText(*range).stripWhiteSpace();
457 document().editor().markMisspellingsAfterTypingToWord(p1, endingSelection(), !strippedPreviousWord.isEmpty());
458 } else if (commandType == TypingCommand::InsertText)
459 document().editor().startAlternativeTextUITimer();
460#else
461 UNUSED_PARAM(commandType);
462 // If this bug gets fixed, this PLATFORM(IOS_FAMILY) code could be removed:
463 // <rdar://problem/7259611> Word boundary code on iPhone gives different results than desktop
464 EWordSide startWordSide = LeftWordIfOnBoundary;
465 UChar32 c = previous.characterAfter();
466 // FIXME: VisiblePosition::characterAfter() and characterBefore() do not emit newlines the same
467 // way as TextIterator, so we do an isEndOfParagraph check here.
468 if (isSpaceOrNewline(c) || c == noBreakSpace || isEndOfParagraph(previous)) {
469 startWordSide = RightWordIfOnBoundary;
470 }
471 VisiblePosition p1 = startOfWord(previous, startWordSide);
472 VisiblePosition p2 = startOfWord(start, startWordSide);
473 if (p1 != p2)
474 document().editor().markMisspellingsAfterTypingToWord(p1, endingSelection(), false);
475#endif // !PLATFORM(IOS_FAMILY)
476 }
477}

479bool TypingCommand::willAddTypingToOpenCommand(ETypingCommand commandType, TextGranularity granularity, const String& text, const std::optional<SimpleRange>& range)
480{
481 m_currentTextToInsert = text;
482 m_currentTypingEditAction = editActionForTypingCommand(commandType, granularity, m_compositionType, m_isAutocompletion);
483
484 if (!shouldDeferWillApplyCommandUntilAddingTypingCommand())
485 return true;
486
487 if (!range || isEditingTextAreaOrTextInput())
488 return document().editor().willApplyEditing(*this, CompositeEditCommand::targetRangesForBindings());
489
490 return document().editor().willApplyEditing(*this, { 1, StaticRange::create(*range) });
491}

493void TypingCommand::typingAddedToOpenCommand(ETypingCommand commandTypeForAddedTyping)
494{
495 RefPtr<Frame> protector(document().frame());
496
497 updatePreservesTypingStyle(commandTypeForAddedTyping);
498
499#if PLATFORM(COCOA)
500 document().editor().appliedEditing(*this);
501 // Since the spellchecking code may also perform corrections and other replacements, it should happen after the typing changes.
502 if (!m_shouldPreventSpellChecking)
503 markMisspellingsAfterTyping(commandTypeForAddedTyping);
504#else
505 // The old spellchecking code requires that checking be done first, to prevent issues like that in 6864072, where <doesn't> is marked as misspelled.
506 markMisspellingsAfterTyping(commandTypeForAddedTyping);
507 document().editor().appliedEditing(*this);
508#endif
509}

608bool TypingCommand::makeEditableRootEmpty()
609{
610 Element* root = endingSelection().rootEditableElement();
611 if (!root || !root->firstChild())
612 return false;
613
614 if (root->firstChild() == root->lastChild() && root->firstElementChild() && root->firstElementChild()->hasTagName(brTag)) {
615 // If there is a single child and it could be a placeholder, leave it alone.
616 if (root->renderer() && root->renderer()->isRenderBlockFlow())
617 return false;
618 }
619
620 while (Node* child = root->firstChild())
621 removeNode(*child);
622
623 addBlockPlaceholderIfNeeded(root);
624 setEndingSelection(VisibleSelection(firstPositionInNode(root), Affinity::Downstream, endingSelection().isDirectional()));
625
626 return true;
627}

629void TypingCommand::deleteKeyPressed(TextGranularity granularity, bool shouldAddToKillRing)
630{
631 RefPtr<Frame> protector(document().frame());
632
633 document().editor().updateMarkersForWordsAffectedByEditing(false);
634
635 VisibleSelection selectionToDelete;
636 VisibleSelection selectionAfterUndo;
637 bool expandForSpecialElements = false;
638
639 ASSERT(endingSelection().isCaretOrRange());
640
641 if (endingSelection().isRange()) {
642 selectionToDelete = endingSelection();
643 selectionAfterUndo = selectionToDelete;
644 expandForSpecialElements = true;
645 } else {
646 // After breaking out of an empty mail blockquote, we still want continue with the deletion
647 // so actual content will get deleted, and not just the quote style.
648 if (breakOutOfEmptyMailBlockquotedParagraph())
649 typingAddedToOpenCommand(DeleteKey);
650
651 m_smartDelete = false;
652
653 FrameSelection selection;
654 selection.setSelection(endingSelection());
655 selection.modify(FrameSelection::AlterationExtend, SelectionDirection::Backward, granularity);
656 if (shouldAddToKillRing && selection.isCaret() && granularity != TextGranularity::CharacterGranularity)
657 selection.modify(FrameSelection::AlterationExtend, SelectionDirection::Backward, TextGranularity::CharacterGranularity);
658
659 const VisiblePosition& visibleStart = endingSelection().visibleStart();
660 const VisiblePosition& previousPosition = visibleStart.previous(CannotCrossEditingBoundary);
661 Node* enclosingTableCell = enclosingNodeOfType(visibleStart.deepEquivalent(), &isTableCell);
662 const Node* enclosingTableCellForPreviousPosition = enclosingNodeOfType(previousPosition.deepEquivalent(), &isTableCell);
663 if (previousPosition.isNull() || enclosingTableCell != enclosingTableCellForPreviousPosition) {
664 // When the caret is at the start of the editable area in an empty list item, break out of the list item.
665 if (auto deleteListSelection = shouldBreakOutOfEmptyListItem()) {
666 if (willAddTypingToOpenCommand(DeleteKey, granularity, { }, *deleteListSelection.value().firstRange())) {
667 breakOutOfEmptyListItem();
668 typingAddedToOpenCommand(DeleteKey);
669 }
670 return;
671 }
672 }
673 if (previousPosition.isNull()) {
674 // When there are no visible positions in the editing root, delete its entire contents.
675 // FIXME: Dispatch a `beforeinput` event here and bail if preventDefault() was invoked.
676 if (visibleStart.next(CannotCrossEditingBoundary).isNull() && makeEditableRootEmpty()) {
677 typingAddedToOpenCommand(DeleteKey);
678 return;
679 }
680 }
681
682 // If we have a caret selection at the beginning of a cell, we have nothing to do.
683 if (enclosingTableCell && visibleStart == firstPositionInNode(enclosingTableCell))
684 return;
685
686 // If the caret is at the start of a paragraph after a table, move content into the last table cell.
687 if (isStartOfParagraph(visibleStart) && isFirstPositionAfterTable(visibleStart.previous(CannotCrossEditingBoundary))) {
688 // Unless the caret is just before a table. We don't want to move a table into the last table cell.
689 if (isLastPositionBeforeTable(visibleStart))
690 return;
691 // Extend the selection backward into the last cell, then deletion will handle the move.
692 selection.modify(FrameSelection::AlterationExtend, SelectionDirection::Backward, granularity);
693 // If the caret is just after a table, select the table and don't delete anything.
694 } else if (Node* table = isFirstPositionAfterTable(visibleStart)) {
695 setEndingSelection(VisibleSelection(positionBeforeNode(table), endingSelection().start(), Affinity::Downstream, endingSelection().isDirectional()));
696 typingAddedToOpenCommand(DeleteKey);
697 return;
698 }
699
700 selectionToDelete = selection.selection();
701
702 if (granularity == TextGranularity::CharacterGranularity && selectionToDelete.end().containerNode() == selectionToDelete.start().containerNode()
703 && selectionToDelete.end().computeOffsetInContainerNode() - selectionToDelete.start().computeOffsetInContainerNode() > 1) {
704 // If there are multiple Unicode code points to be deleted, adjust the range to match platform conventions.
705 selectionToDelete.setWithoutValidation(selectionToDelete.end(), selectionToDelete.end().previous(BackwardDeletion));
706 }
707
708 if (!startingSelection().isRange() || selectionToDelete.base() != startingSelection().start())
709 selectionAfterUndo = selectionToDelete;
710 else
711 // It's a little tricky to compute what the starting selection would have been in the original document.
712 // We can't let the VisibleSelection class's validation kick in or it'll adjust for us based on
713 // the current state of the document and we'll get the wrong result.
714 selectionAfterUndo.setWithoutValidation(startingSelection().end(), selectionToDelete.extent());
715 }
716
717 ASSERT(!selectionToDelete.isNone());
718 if (selectionToDelete.isNone()) {
719#if PLATFORM(IOS_FAMILY)
720 // Workaround for this bug:
721 // <rdar://problem/4653755> UIKit text widgets should use WebKit editing API to manipulate text
722 setEndingSelection(document().selection().selection());
723 closeTyping(document());
724#endif
725 return;
726 }
727
728 if (selectionToDelete.isCaret() || !document().selection().shouldDeleteSelection(selectionToDelete))
729 return;
730
731 if (!willAddTypingToOpenCommand(DeleteKey, granularity, { }, selectionToDelete.firstRange()))
732 return;
733
734 if (shouldAddToKillRing)
735 document().editor().addRangeToKillRing(*selectionToDelete.toNormalizedRange(), Editor::KillRingInsertionMode::PrependText);
736
737 // Post the accessibility notification before actually deleting the content while selectionToDelete is still valid
738 postTextStateChangeNotificationForDeletion(selectionToDelete);
739
740 // Make undo select everything that has been deleted, unless an undo will undo more than just this deletion.
741 // FIXME: This behaves like TextEdit except for the case where you open with text insertion and then delete
742 // more text than you insert. In that case all of the text that was around originally should be selected.
743 if (m_openedByBackwardDelete)
744 setStartingSelection(selectionAfterUndo);
745 CompositeEditCommand::deleteSelection(selectionToDelete, m_smartDelete, /* mergeBlocksAfterDelete*/ true, /* replace*/ false, expandForSpecialElements, /*sanitizeMarkup*/ true);
746 setSmartDelete(false);
747 typingAddedToOpenCommand(DeleteKey);
748}

750void TypingCommand::forwardDeleteKeyPressed(TextGranularity granularity, bool shouldAddToKillRing)
751{
752 RefPtr<Frame> protector(document().frame());
753
754 document().editor().updateMarkersForWordsAffectedByEditing(false);
755
756 VisibleSelection selectionToDelete;
757 VisibleSelection selectionAfterUndo;
758 bool expandForSpecialElements = false;
759
760 ASSERT(endingSelection().isCaretOrRange());
761
762 if (endingSelection().isRange()) {
763 selectionToDelete = endingSelection();
764 selectionAfterUndo = selectionToDelete;
765 expandForSpecialElements = true;
766 } else {
767 m_smartDelete = false;
768
769 // Handle delete at beginning-of-block case.
770 // Do nothing in the case that the caret is at the start of a
771 // root editable element or at the start of a document.
772 FrameSelection selection;
773 selection.setSelection(endingSelection());
774 selection.modify(FrameSelection::AlterationExtend, SelectionDirection::Forward, granularity);
775 if (selection.isNone())
776 return;
777 if (shouldAddToKillRing && selection.isCaret() && granularity != TextGranularity::CharacterGranularity)
778 selection.modify(FrameSelection::AlterationExtend, SelectionDirection::Forward, TextGranularity::CharacterGranularity);
779
780 Position downstreamEnd = endingSelection().end().downstream();
781 VisiblePosition visibleEnd = endingSelection().visibleEnd();
782 Node* enclosingTableCell = enclosingNodeOfType(visibleEnd.deepEquivalent(), &isTableCell);
783 if (enclosingTableCell && visibleEnd == lastPositionInNode(enclosingTableCell))
784 return;
785 if (visibleEnd == endOfParagraph(visibleEnd))
786 downstreamEnd = visibleEnd.next(CannotCrossEditingBoundary).deepEquivalent().downstream();
787 // When deleting tables: Select the table first, then perform the deletion
788 if (downstreamEnd.containerNode() && downstreamEnd.containerNode()->renderer() && downstreamEnd.containerNode()->renderer()->isTable()
789 && downstreamEnd.computeOffsetInContainerNode() <= caretMinOffset(*downstreamEnd.containerNode())) {
790 setEndingSelection(VisibleSelection(endingSelection().end(), positionAfterNode(downstreamEnd.containerNode()), Affinity::Downstream, endingSelection().isDirectional()));
791 typingAddedToOpenCommand(ForwardDeleteKey);
792 return;
793 }
794
795 // deleting to end of paragraph when at end of paragraph needs to merge the next paragraph (if any)
796 if (granularity == TextGranularity::ParagraphBoundary && selection.selection().isCaret() && isEndOfParagraph(selection.selection().visibleEnd()))
797 selection.modify(FrameSelection::AlterationExtend, SelectionDirection::Forward, TextGranularity::CharacterGranularity);
798
799 selectionToDelete = selection.selection();
800 if (!startingSelection().isRange() || selectionToDelete.base() != startingSelection().start())
801 selectionAfterUndo = selectionToDelete;
802 else {
803 // It's a little tricky to compute what the starting selection would have been in the original document.
804 // We can't let the VisibleSelection class's validation kick in or it'll adjust for us based on
805 // the current state of the document and we'll get the wrong result.
806 Position extent = startingSelection().end();
807 if (extent.containerNode() != selectionToDelete.end().containerNode())
808 extent = selectionToDelete.extent();
809 else {
810 int extraCharacters;
811 if (selectionToDelete.start().containerNode() == selectionToDelete.end().containerNode())
812 extraCharacters = selectionToDelete.end().computeOffsetInContainerNode() - selectionToDelete.start().computeOffsetInContainerNode();
813 else
814 extraCharacters = selectionToDelete.end().computeOffsetInContainerNode();
815 extent = Position(extent.containerNode(), extent.computeOffsetInContainerNode() + extraCharacters, Position::PositionIsOffsetInAnchor);
816 }
817 selectionAfterUndo.setWithoutValidation(startingSelection().start(), extent);
818 }
819 }
820
821 ASSERT(!selectionToDelete.isNone());
822 if (selectionToDelete.isNone()) {
823#if PLATFORM(IOS_FAMILY)
824 // Workaround for this bug:
825 // <rdar://problem/4653755> UIKit text widgets should use WebKit editing API to manipulate text
826 setEndingSelection(document().selection().selection());
827 closeTyping(document());
828#endif
829 return;
830 }
831
832 if (selectionToDelete.isCaret() || !document().selection().shouldDeleteSelection(selectionToDelete))
833 return;
834
835 if (!willAddTypingToOpenCommand(ForwardDeleteKey, granularity, { }, selectionToDelete.firstRange()))
836 return;
837
838 // Post the accessibility notification before actually deleting the content while selectionToDelete is still valid
839 postTextStateChangeNotificationForDeletion(selectionToDelete);
840
841 if (shouldAddToKillRing)
842 document().editor().addRangeToKillRing(*selectionToDelete.toNormalizedRange(), Editor::KillRingInsertionMode::AppendText);
843 // make undo select what was deleted
844 setStartingSelection(selectionAfterUndo);
845 CompositeEditCommand::deleteSelection(selectionToDelete, m_smartDelete, /* mergeBlocksAfterDelete*/ true, /* replace*/ false, expandForSpecialElements, /*sanitizeMarkup*/ true);
846 setSmartDelete(false);
847 typingAddedToOpenCommand(ForwardDeleteKey);
848}

852 if (!willAddTypingToOpenCommand(DeleteSelection, TextGranularity::CharacterGranularity))
