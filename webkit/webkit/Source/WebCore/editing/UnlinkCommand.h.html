28#include "CompositeEditCommand.h"

30namespace WebCore {
31
32class UnlinkCommand : public CompositeEditCommand {
33public:
34 static Ref<UnlinkCommand> create(Document& document)
35 {
36 return adoptRef(*new UnlinkCommand(document));
37 }
38
39private:
40 explicit UnlinkCommand(Document&);
41
42 void doApply() override;
43 EditAction editingAction() const override { return EditAction::Unlink; }
44};

46} // namespace WebCore
