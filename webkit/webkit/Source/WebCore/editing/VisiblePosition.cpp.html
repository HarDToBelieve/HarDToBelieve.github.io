56VisiblePosition::VisiblePosition(const Position& position, Affinity affinity)
57 : m_deepPosition { canonicalPosition(position) }
58{
59 if (affinity == Affinity::Upstream && !isNull()) {
60 auto upstreamCopy = *this;
61 upstreamCopy.m_affinity = Affinity::Upstream;
62 if (!inSameLine(*this, upstreamCopy))
63 m_affinity = Affinity::Upstream;
64 }
65}

67VisiblePosition VisiblePosition::next(EditingBoundaryCrossingRule rule, bool* reachedBoundary) const
68{
69 if (reachedBoundary)
70 *reachedBoundary = false;
71 // FIXME: Support CanSkipEditingBoundary
72 ASSERT(rule == CanCrossEditingBoundary || rule == CannotCrossEditingBoundary);
73 VisiblePosition next(nextVisuallyDistinctCandidate(m_deepPosition), m_affinity);
74
75 if (rule == CanCrossEditingBoundary)
76 return next;
77
78 return honorEditingBoundaryAtOrAfter(next, reachedBoundary);
79}

81VisiblePosition VisiblePosition::previous(EditingBoundaryCrossingRule rule, bool* reachedBoundary) const
82{
83 if (reachedBoundary)
84 *reachedBoundary = false;
85 // FIXME: Support CanSkipEditingBoundary
86 ASSERT(rule == CanCrossEditingBoundary || rule == CannotCrossEditingBoundary);
87 // find first previous DOM position that is visible
88 Position pos = previousVisuallyDistinctCandidate(m_deepPosition);
89
90 // return null visible position if there is no previous visible position
91 if (pos.atStartOfTree()) {
92 if (reachedBoundary)
93 *reachedBoundary = true;
94 return VisiblePosition();
95 }
96
97 VisiblePosition prev = pos;
98 ASSERT(prev != *this);
99
100#if ASSERT_ENABLED
101 // We should always be able to make the affinity downstream, because going previous from an
102 // upstream position can never yield another upstream position unless line wrap length is 0.
103 if (prev.isNotNull() && m_affinity == Affinity::Upstream) {
104 auto upstreamCopy = prev;
105 upstreamCopy.setAffinity(Affinity::Upstream);
106 ASSERT(inSameLine(upstreamCopy, prev));
107 }
108#endif
109
110 if (rule == CanCrossEditingBoundary)
111 return prev;
112
113 return honorEditingBoundaryAtOrBefore(prev, reachedBoundary);
114}

116Position VisiblePosition::leftVisuallyDistinctCandidate() const
117{
118 Position p = m_deepPosition;
119 if (p.isNull())
120 return Position();
121
122 Position downstreamStart = p.downstream();
123 TextDirection primaryDirection = p.primaryDirection();
124
125 while (true) {
126 auto [run, offset] = p.inlineRunAndOffset(m_affinity, primaryDirection);
127 if (!run)
128 return primaryDirection == TextDirection::LTR ? previousVisuallyDistinctCandidate(m_deepPosition) : nextVisuallyDistinctCandidate(m_deepPosition);
129
130 auto* renderer = &run->renderer();
131
132 while (true) {
133 if ((renderer->isReplaced() || renderer->isBR()) && offset == run->rightmostCaretOffset())
134 return run->isLeftToRightDirection() ? previousVisuallyDistinctCandidate(m_deepPosition) : nextVisuallyDistinctCandidate(m_deepPosition);
135
136 if (!renderer->node()) {
137 run.traversePreviousOnLine();
138 if (!run)
139 return primaryDirection == TextDirection::LTR ? previousVisuallyDistinctCandidate(m_deepPosition) : nextVisuallyDistinctCandidate(m_deepPosition);
140 renderer = &run->renderer();
141 offset = run->rightmostCaretOffset();
142 continue;
143 }
144
145 // Note that this may underflow the (unsigned) offset. This is fine and handled below.
146 offset = run->isLeftToRightDirection() ? renderer->previousOffset(offset) : renderer->nextOffset(offset);
147
148 auto caretMinOffset = run->minimumCaretOffset();
149 auto caretMaxOffset = run->maximumCaretOffset();
150
151 if (offset > caretMinOffset && offset < caretMaxOffset)
152 break;
153
154 if (offset != run->leftmostCaretOffset()) {
155 // Overshot to the left.
156 auto previousRun = run.previousOnLineIgnoringLineBreak();
157 if (!previousRun) {
158 Position positionOnLeft = primaryDirection == TextDirection::LTR ? previousVisuallyDistinctCandidate(m_deepPosition) : nextVisuallyDistinctCandidate(m_deepPosition);
159 auto runOnLeft = positionOnLeft.inlineRunAndOffset(m_affinity, primaryDirection).run;
160 if (runOnLeft && runOnLeft.line() == run.line())
161 return Position();
162 return positionOnLeft;
163 }
164
165 // Reposition at the other logical position corresponding to our edge's visual position and go for another round.
166 run = previousRun;
167 renderer = &run->renderer();
168 offset = previousRun->rightmostCaretOffset();
169 continue;
170 }
171
172 unsigned char level = run->bidiLevel();
173 auto previousRun = run.previousOnLine();
174
175 if (run->direction() == primaryDirection) {
176 if (!previousRun) {
177 auto logicalStart = (primaryDirection == TextDirection::LTR) ? run.line().logicalStartRunWithNode() : run.line().logicalEndRunWithNode();
178 if (logicalStart) {
179 run = logicalStart;
180 renderer = &run->renderer();
181 offset = primaryDirection == TextDirection::LTR ? run->minimumCaretOffset() : run->maximumCaretOffset();
182 }
183 break;
184 }
185 if (previousRun->bidiLevel() >= level)
186 break;
187
188 level = previousRun->bidiLevel();
189
190 auto nextRun = run;
191 do {
192 nextRun.traverseNextOnLine();
193 } while (nextRun && nextRun->bidiLevel() > level);
194
195 if (nextRun && nextRun->bidiLevel() == level)
196 break;
197
198 run = previousRun;
199 renderer = &run->renderer();
200 offset = run->rightmostCaretOffset();
201 if (run->direction() == primaryDirection)
202 break;
203 continue;
204 }
205
206 while (previousRun && !previousRun->renderer().node())
207 previousRun.traversePreviousOnLine();
208
209 if (previousRun) {
210 run = previousRun;
211 renderer = &run->renderer();
212 offset = run->rightmostCaretOffset();
213 if (run->bidiLevel() > level) {
214 do {
215 previousRun = previousRun.traversePreviousOnLine();
216 } while (previousRun && previousRun->bidiLevel() > level);
217
218 if (!previousRun || previousRun->bidiLevel() < level)
219 continue;
220 }
221 } else {
222 // Trailing edge of a secondary run. Set to the leading edge of the entire run.
223 while (true) {
224 while (auto nextRun = run.nextOnLine()) {
225 if (nextRun->bidiLevel() < level)
226 break;
227 run = nextRun;
228 }
229 if (run->bidiLevel() == level)
230 break;
231 level = run->bidiLevel();
232 while (auto previousRun = run.previousOnLine()) {
233 if (previousRun->bidiLevel() < level)
234 break;
235 run = previousRun;
236 }
237 if (run->bidiLevel() == level)
238 break;
239 level = run->bidiLevel();
240 }
241 renderer = &run->renderer();
242 offset = primaryDirection == TextDirection::LTR ? run->minimumCaretOffset() : run->maximumCaretOffset();
243 }
244 break;
245 }
246
247 p = makeDeprecatedLegacyPosition(renderer->node(), offset);
248
249 if ((p.isCandidate() && p.downstream() != downstreamStart) || p.atStartOfTree() || p.atEndOfTree())
250 return p;
251
252 ASSERT(p != m_deepPosition);
253 }
254}

256VisiblePosition VisiblePosition::left(bool stayInEditableContent, bool* reachedBoundary) const
257{
258 if (reachedBoundary)
259 *reachedBoundary = false;
260 Position pos = leftVisuallyDistinctCandidate();
261 // FIXME: Why can't we move left from the last position in a tree?
262 if (pos.atStartOfTree() || pos.atEndOfTree()) {
263 if (reachedBoundary)
264 *reachedBoundary = true;
265 return VisiblePosition();
266 }
267
268 VisiblePosition left = pos;
269 ASSERT(left != *this);
270
271 if (!stayInEditableContent)
272 return left;
273
274 // FIXME: This may need to do something different from "before".
275 return honorEditingBoundaryAtOrBefore(left, reachedBoundary);
276}

278Position VisiblePosition::rightVisuallyDistinctCandidate() const
279{
280 Position p = m_deepPosition;
281 if (p.isNull())
282 return Position();
283
284 Position downstreamStart = p.downstream();
285 TextDirection primaryDirection = p.primaryDirection();
286
287 while (true) {
288 auto [run, offset] = p.inlineRunAndOffset(m_affinity, primaryDirection);
289 if (!run)
290 return primaryDirection == TextDirection::LTR ? nextVisuallyDistinctCandidate(m_deepPosition) : previousVisuallyDistinctCandidate(m_deepPosition);
291
292 auto* renderer = &run->renderer();
293
294 while (true) {
295 if ((renderer->isReplaced() || renderer->isBR()) && offset == run->leftmostCaretOffset())
296 return run->isLeftToRightDirection() ? nextVisuallyDistinctCandidate(m_deepPosition) : previousVisuallyDistinctCandidate(m_deepPosition);
297
298 if (!renderer->node()) {
299 run.traverseNextOnLine();
300 if (!run)
301 return primaryDirection == TextDirection::LTR ? nextVisuallyDistinctCandidate(m_deepPosition) : previousVisuallyDistinctCandidate(m_deepPosition);
302 renderer = &run->renderer();
303 offset = run->leftmostCaretOffset();
304 continue;
305 }
306
307 // Note that this may underflow the (unsigned) offset. This is fine and handled below.
308 offset = run->isLeftToRightDirection() ? renderer->nextOffset(offset) : renderer->previousOffset(offset);
309
310 auto caretMinOffset = run->minimumCaretOffset();
311 auto caretMaxOffset = run->maximumCaretOffset();
312
313 if (offset > caretMinOffset && offset < caretMaxOffset)
314 break;
315
316 if (offset != run->rightmostCaretOffset()) {
317 // Overshot to the right.
318 auto nextRun = run.nextOnLineIgnoringLineBreak();
319 if (!nextRun) {
320 Position positionOnRight = primaryDirection == TextDirection::LTR ? nextVisuallyDistinctCandidate(m_deepPosition) : previousVisuallyDistinctCandidate(m_deepPosition);
321 auto runOnRight = positionOnRight.inlineRunAndOffset(m_affinity, primaryDirection).run;
322 if (runOnRight && runOnRight.line() == run.line())
323 return Position();
324 return positionOnRight;
325 }
326
327 // Reposition at the other logical position corresponding to our edge's visual position and go for another round.
328 run = nextRun;
329 renderer = &run->renderer();
330 offset = nextRun->leftmostCaretOffset();
331 continue;
332 }
333
334 unsigned char level = run->bidiLevel();
335 auto nextRun = run.nextOnLine();
336
337 if (run->direction() == primaryDirection) {
338 if (!nextRun) {
339 auto logicalEnd = primaryDirection == TextDirection::LTR ? run.line().logicalEndRunWithNode() : run.line().logicalStartRunWithNode();
340 if (logicalEnd) {
341 run = logicalEnd;
342 renderer = &run->renderer();
343 offset = primaryDirection == TextDirection::LTR ? run->maximumCaretOffset() : run->minimumCaretOffset();
344 }
345 break;
346 }
347
348 if (nextRun->bidiLevel() >= level)
349 break;
350
351 level = nextRun->bidiLevel();
352
353 auto previousRun = run;
354 do {
355 previousRun.traversePreviousOnLine();
356 } while (previousRun && previousRun->bidiLevel() > level);
357
358 if (previousRun && previousRun->bidiLevel() == level) // For example, abc FED 123 ^ CBA
359 break;
360
361 // For example, abc 123 ^ CBA or 123 ^ CBA abc
362 run = nextRun;
363 renderer = &run->renderer();
364 offset = run->leftmostCaretOffset();
365 if (run->direction() == primaryDirection)
366 break;
367 continue;
368 }
369
370 while (nextRun && !nextRun->renderer().node())
371 nextRun.traverseNextOnLine();
372
373 if (nextRun) {
374 run = nextRun;
375 renderer = &run->renderer();
376 offset = run->leftmostCaretOffset();
377
378 if (run->bidiLevel() > level) {
379 do {
380 nextRun.traverseNextOnLine();
381 } while (nextRun && nextRun->bidiLevel() > level);
382
383 if (!nextRun || nextRun->bidiLevel() < level)
384 continue;
385 }
386 } else {
387 // Trailing edge of a secondary run. Set to the leading edge of the entire run.
388 while (true) {
389 while (auto previousRun = run.previousOnLine()) {
390 if (previousRun->bidiLevel() < level)
391 break;
392 run = previousRun;
393 }
394 if (run->bidiLevel() == level)
395 break;
396 level = run->bidiLevel();
397 while (auto nextRun = run.nextOnLine()) {
398 if (nextRun->bidiLevel() < level)
399 break;
400 run = nextRun;
401 }
402 if (run->bidiLevel() == level)
403 break;
404 level = run->bidiLevel();
405 }
406 renderer = &run->renderer();
407 offset = primaryDirection == TextDirection::LTR ? run->maximumCaretOffset() : run->minimumCaretOffset();
408 }
409 break;
410 }
411
412 p = makeDeprecatedLegacyPosition(renderer->node(), offset);
413
414 if ((p.isCandidate() && p.downstream() != downstreamStart) || p.atStartOfTree() || p.atEndOfTree())
415 return p;
416
417 ASSERT(p != m_deepPosition);
418 }

443VisiblePosition VisiblePosition::honorEditingBoundaryAtOrBefore(const VisiblePosition& position, bool* reachedBoundary) const
444{
445 if (reachedBoundary)
446 *reachedBoundary = false;
447 if (position.isNull())
448 return position;
449
450 auto* highestRoot = highestEditableRoot(deepEquivalent());
451
452 // Return empty position if pos is not somewhere inside the editable region containing this position
453 if (highestRoot && !position.deepEquivalent().deprecatedNode()->isDescendantOf(*highestRoot)) {
454 if (reachedBoundary)
455 *reachedBoundary = true;
456 return VisiblePosition();
457 }
458
459 // Return position itself if the two are from the very same editable region, or both are non-editable
460 // FIXME: In the non-editable case, just because the new position is non-editable doesn't mean movement
461 // to it is allowed. VisibleSelection::adjustForEditableContent has this problem too.
462 if (highestEditableRoot(position.deepEquivalent()) == highestRoot) {
463 if (reachedBoundary)
464 *reachedBoundary = *this == position;
465 return position;
466 }
467
468 // Return empty position if this position is non-editable, but pos is editable
469 // FIXME: Move to the previous non-editable region.
470 if (!highestRoot) {
471 if (reachedBoundary)
472 *reachedBoundary = true;
473 return VisiblePosition();
474 }
475
476 // Return the last position before pos that is in the same editable region as this position
477 return lastEditablePositionBeforePositionInRoot(position.deepEquivalent(), highestRoot);
478}

480VisiblePosition VisiblePosition::honorEditingBoundaryAtOrAfter(const VisiblePosition& otherPosition, bool* reachedBoundary) const
481{
482 if (reachedBoundary)
483 *reachedBoundary = false;
484 if (otherPosition.isNull())
485 return otherPosition;
486
487 auto* highestRoot = highestEditableRoot(deepEquivalent());
488
489 // Return empty position if otherPosition is not somewhere inside the editable region containing this position
490 if (highestRoot && !otherPosition.deepEquivalent().deprecatedNode()->isDescendantOf(*highestRoot)) {
491 if (reachedBoundary)
492 *reachedBoundary = true;
493 return VisiblePosition();
494 }
495
496 // Return otherPosition itself if the two are from the very same editable region, or both are non-editable
497 // FIXME: In the non-editable case, just because the new position is non-editable doesn't mean movement
498 // to it is allowed. VisibleSelection::adjustForEditableContent has this problem too.
499 if (highestEditableRoot(otherPosition.deepEquivalent()) == highestRoot) {
500 if (reachedBoundary)
501 *reachedBoundary = *this == otherPosition;
502 return otherPosition;
503 }
504
505 // Return empty position if this position is non-editable, but otherPosition is editable
506 // FIXME: Move to the next non-editable region.
507 if (!highestRoot) {
508 if (reachedBoundary)
509 *reachedBoundary = true;
510 return VisiblePosition();
511 }
512
513 // Return the next position after pos that is in the same editable region as this position
514 return firstEditablePositionAfterPositionInRoot(otherPosition.deepEquivalent(), highestRoot);
515}

517static Position canonicalizeCandidate(const Position& candidate)
518{
519 if (candidate.isNull())
520 return Position();
521 ASSERT(candidate.isCandidate());
522 Position upstream = candidate.upstream();
523 if (upstream.isCandidate())
524 return upstream;
525 return candidate;
526}

528Position VisiblePosition::canonicalPosition(const Position& passedPosition)
529{
530 // The updateLayout call below can do so much that even the position passed
531 // in to us might get changed as a side effect. Specifically, there are code
532 // paths that pass selection endpoints, and updateLayout can change the selection.
533 Position position = passedPosition;
534
535 // FIXME (9535): Canonicalizing to the leftmost candidate means that if we're at a line wrap, we will
536 // ask renderers to paint downstream carets for other renderers.
537 // To fix this, we need to either a) add code to all paintCarets to pass the responsibility off to
538 // the appropriate renderer for VisiblePosition's like these, or b) canonicalize to the rightmost candidate
539 // unless the affinity is upstream.
540 if (position.isNull())
541 return Position();
542
543 ASSERT(position.document());
544 position.document()->updateLayoutIgnorePendingStylesheets();
545
546 Node* node = position.containerNode();
547
548 Position candidate = position.upstream();
549 if (candidate.isCandidate())
550 return candidate;
551 candidate = position.downstream();
552 if (candidate.isCandidate())
553 return candidate;
554
555 // When neither upstream or downstream gets us to a candidate (upstream/downstream won't leave
556 // blocks or enter new ones), we search forward and backward until we find one.
557 Position next = canonicalizeCandidate(nextCandidate(position));
558 Position prev = canonicalizeCandidate(previousCandidate(position));
559 Node* nextNode = next.deprecatedNode();
560 Node* prevNode = prev.deprecatedNode();
561
562 // The new position must be in the same editable element. Enforce that first.
563 // Unless the descent is from a non-editable html element to an editable body.
564 if (is<HTMLHtmlElement>(node) && !node->hasEditableStyle()) {
565 auto* body = node->document().bodyOrFrameset();
566 if (body && body->hasEditableStyle())
567 return next.isNotNull() ? next : prev;
568 }
569
570 Node* editingRoot = editableRootForPosition(position);
571
572 // If the html element is editable, descending into its body will look like a descent
573 // from non-editable to editable content since rootEditableElement() always stops at the body.
574 if ((editingRoot && editingRoot->hasTagName(htmlTag)) || (node && (node->isDocumentNode() || node->isShadowRoot())))
575 return next.isNotNull() ? next : prev;
576
577 bool prevIsInSameEditableElement = prevNode && editableRootForPosition(prev) == editingRoot;
578 bool nextIsInSameEditableElement = nextNode && editableRootForPosition(next) == editingRoot;
579 if (prevIsInSameEditableElement && !nextIsInSameEditableElement)
580 return prev;
581
582 if (nextIsInSameEditableElement && !prevIsInSameEditableElement)
583 return next;
584
585 if (!nextIsInSameEditableElement && !prevIsInSameEditableElement)
586 return Position();
587
588 // The new position should be in the same block flow element. Favor that.
589 Element* originalBlock = deprecatedEnclosingBlockFlowElement(node);
590 bool nextIsOutsideOriginalBlock = !nextNode->isDescendantOf(originalBlock) && nextNode != originalBlock;
591 bool prevIsOutsideOriginalBlock = !prevNode->isDescendantOf(originalBlock) && prevNode != originalBlock;
592 if (nextIsOutsideOriginalBlock && !prevIsOutsideOriginalBlock)
593 return prev;
594
595 return next;
596}

598UChar32 VisiblePosition::characterAfter() const
599{
600 // We canonicalize to the first of two equivalent candidates, but the second of the two candidates
601 // is the one that will be inside the text node containing the character after this visible position.
602 Position pos = m_deepPosition.downstream();
603 if (!pos.containerNode() || !pos.containerNode()->isTextNode())
604 return 0;
605 switch (pos.anchorType()) {
606 case Position::PositionIsAfterChildren:
607 case Position::PositionIsAfterAnchor:
608 case Position::PositionIsBeforeAnchor:
609 case Position::PositionIsBeforeChildren:
610 return 0;
611 case Position::PositionIsOffsetInAnchor:
612 break;
613 }
614 unsigned offset = static_cast<unsigned>(pos.offsetInContainerNode());
615 Text* textNode = pos.containerText();
616 unsigned length = textNode->length();
617 if (offset >= length)
618 return 0;
619
620 UChar32 ch;
621 U16_NEXT(textNode->data(), offset, length, ch);
622 return ch;
623}

635auto VisiblePosition::localCaretRect() const -> LocalCaretRect
636{
637 auto node = m_deepPosition.anchorNode();
638 if (!node)
639 return { };
640
641 auto runAndOffset = inlineRunAndOffset();
642 auto renderer = runAndOffset.run ? &runAndOffset.run->renderer() : node->renderer();
643 if (!renderer)
644 return { };
645
646 return { computeLocalCaretRect(*renderer, runAndOffset), const_cast<RenderObject*>(renderer) };
647}

656FloatRect VisiblePosition::absoluteSelectionBoundsForLine() const
657{
658 auto run = inlineRunAndOffset().run;
659 if (!run)
660 return { };
661
662 auto line = run.line();
663 auto localRect = FloatRect { FloatPoint { line->contentLogicalLeft(), line->selectionTop() }, FloatPoint { line->contentLogicalRight(), line->selectionBottom() } };
664 if (!line->isHorizontal())
665 localRect = localRect.transposedRect();
666 line->containingBlock().flipForWritingMode(localRect);
667 return line->containingBlock().localToAbsoluteQuad(localRect).boundingBox();
668}

670int VisiblePosition::lineDirectionPointForBlockDirectionNavigation() const
671{
672 auto localRect = localCaretRect();
673 if (localRect.rect.isEmpty() || !localRect.renderer)
674 return 0;
675
676 // This ignores transforms on purpose, for now. Vertical navigation is done
677 // without consulting transforms, so that 'up' in transformed text is 'up'
678 // relative to the text, not absolute 'up'.
679 auto caretPoint = localRect.renderer->localToAbsolute(localRect.rect.location());
680 RenderObject* containingBlock = localRect.renderer->containingBlock();
681 if (!containingBlock)
682 containingBlock = localRect.renderer; // Just use ourselves to determine the writing mode if we have no containing block.
683 return containingBlock->isHorizontalWritingMode() ? caretPoint.x() : caretPoint.y();
684}
