37const String& FontChanges::platformFontFamilyNameForCSS() const
38{
39 // The family name may not be specific enough to get us the font specified.
40 // In some cases, the only way to get exactly what we are looking for is to use
41 // the Postscript name. If we don't find a font with the same Postscript name,
42 // then we'll have to use the Postscript name to make the CSS specific enough.
43 // This logic was originally from WebHTMLView, and is now in WebCore so that it can
44 // be shared between WebKitLegacy and WebKit.
45 auto cfFontName = m_fontName.createCFString();
46 RetainPtr<CFStringRef> fontNameFromDescription;
47
48 FontDescription description;
49 description.setIsItalic(m_italic.value_or(false));
50 description.setWeight(FontSelectionValue { m_bold.value_or(false) ? 900 : 500 });
51 if (auto font = FontCache::singleton().fontForFamily(description, m_fontFamily))
52 fontNameFromDescription = adoptCF(CTFontCopyPostScriptName(font->getCTFont()));
53
54 if (fontNameFromDescription && CFStringCompare(cfFontName.get(), fontNameFromDescription.get(), 0) == kCFCompareEqualTo)
55 return m_fontFamily;
56
57 return m_fontName;
58}
