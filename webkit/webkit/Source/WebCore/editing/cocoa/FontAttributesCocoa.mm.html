39static NSString *cocoaTextListMarkerName(ListStyleType style, bool ordered)
40{
41 switch (style) {
42 case ListStyleType::Disc:
43 return NSTextListMarkerDisc;
44 case ListStyleType::Circle:
45 return NSTextListMarkerCircle;
46 case ListStyleType::Square:
47 return NSTextListMarkerSquare;
48 case ListStyleType::Decimal:
49 return NSTextListMarkerDecimal;
50 case ListStyleType::Octal:
51 return NSTextListMarkerOctal;
52 case ListStyleType::LowerRoman:
53 return NSTextListMarkerLowercaseRoman;
54 case ListStyleType::UpperRoman:
55 return NSTextListMarkerUppercaseRoman;
56 case ListStyleType::LowerAlpha:
57 return NSTextListMarkerLowercaseAlpha;
58 case ListStyleType::UpperAlpha:
59 return NSTextListMarkerUppercaseAlpha;
60 case ListStyleType::LowerLatin:
61 return NSTextListMarkerLowercaseLatin;
62 case ListStyleType::UpperLatin:
63 return NSTextListMarkerUppercaseLatin;
64 case ListStyleType::LowerHexadecimal:
65 return NSTextListMarkerLowercaseHexadecimal;
66 case ListStyleType::UpperHexadecimal:
67 return NSTextListMarkerUppercaseHexadecimal;
68 default:
69 // The remaining web-exposed list style types have no Cocoa equivalents.
70 // Fall back to default styles for ordered and unordered lists.
71 return ordered ? NSTextListMarkerDecimal : NSTextListMarkerDisc;
72 }
73}

75RetainPtr<NSTextList> TextList::createTextList() const
76{
77#if PLATFORM(MAC)
78 Class textListClass = NSTextList.class;
79#else
80 Class textListClass = PAL::getNSTextListClass();
81#endif
82 auto result = adoptNS([[textListClass alloc] initWithMarkerFormat:cocoaTextListMarkerName(style, ordered) options:0]);
83 [result setStartingItemNumber:startingItemNumber];
84 return result;
85}

87RetainPtr<NSDictionary> FontAttributes::createDictionary() const
88{
89 NSMutableDictionary *attributes = [NSMutableDictionary dictionary];
90 if (font)
91 attributes[NSFontAttributeName] = font.get();
92
93 if (foregroundColor.isValid())
94 attributes[NSForegroundColorAttributeName] = platformColor(foregroundColor);
95
96 if (backgroundColor.isValid())
97 attributes[NSBackgroundColorAttributeName] = platformColor(backgroundColor);
98
99 if (fontShadow.color.isValid() && (!fontShadow.offset.isZero() || fontShadow.blurRadius))
100 attributes[NSShadowAttributeName] = fontShadow.createShadow().get();
101
102 if (subscriptOrSuperscript == SubscriptOrSuperscript::Subscript)
103 attributes[NSSuperscriptAttributeName] = @(-1);
104 else if (subscriptOrSuperscript == SubscriptOrSuperscript::Superscript)
105 attributes[NSSuperscriptAttributeName] = @1;
106
107#if PLATFORM(MAC)
108 Class paragraphStyleClass = NSParagraphStyle.class;
109#else
110 Class paragraphStyleClass = PAL::getNSParagraphStyleClass();
111#endif
112 auto style = adoptNS([[paragraphStyleClass defaultParagraphStyle] mutableCopy]);
113
114 switch (horizontalAlignment) {
115 case HorizontalAlignment::Left:
116 [style setAlignment:NSTextAlignmentLeft];
117 break;
118 case HorizontalAlignment::Center:
119 [style setAlignment:NSTextAlignmentCenter];
120 break;
121 case HorizontalAlignment::Right:
122 [style setAlignment:NSTextAlignmentRight];
123 break;
124 case HorizontalAlignment::Justify:
125 [style setAlignment:NSTextAlignmentJustified];
126 break;
127 case HorizontalAlignment::Natural:
128 [style setAlignment:NSTextAlignmentNatural];
129 break;
130 }
131
132 if (!textLists.isEmpty()) {
133 [style setTextLists:createNSArray(textLists, [] (auto& textList) {
134 return textList.createTextList();
135 }).get()];
136 }
137
138 attributes[NSParagraphStyleAttributeName] = style.get();
139
140 if (hasUnderline)
141 attributes[NSUnderlineStyleAttributeName] = @(NSUnderlineStyleSingle);
142
143 if (hasStrikeThrough)
144 attributes[NSStrikethroughStyleAttributeName] = @(NSUnderlineStyleSingle);
145
146 return attributes;
147}
