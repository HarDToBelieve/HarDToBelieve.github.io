26#import "config.h"
27#import "WebArchiveResourceFromNSAttributedString.h"
28
29#import "ArchiveResource.h"
30#import "MIMETypeRegistry.h"
31
32using namespace WebCore;
33
34@implementation WebArchiveResourceFromNSAttributedString
35
36- (instancetype)initWithData:(NSData *)data URL:(NSURL *)URL MIMEType:(NSString *)MIMEType textEncodingName:(NSString *)textEncodingName frameName:(NSString *)frameName
37{
38 if (!(self = [super init]))
39 return nil;
40
41 if (!data || !URL || !MIMEType) {
42 [self release];
43 return nil;
44 }
45
46 if ([MIMEType isEqualToString:@"application/octet-stream"]) {
47 // FIXME: This is a workaround for <rdar://problem/36074429>, and can be removed once that is fixed.
48 auto mimeTypeFromURL = MIMETypeRegistry::mimeTypeForExtension(URL.pathExtension);
49 if (!mimeTypeFromURL.isEmpty())
50 MIMEType = mimeTypeFromURL;
51 }
52
53 resource = ArchiveResource::create(SharedBuffer::create(adoptNS([data copy]).get()), URL, MIMEType, textEncodingName, frameName, { });
54 if (!resource) {
55 [self release];
56 return nil;
57 }
58
59 return self;
60}
61
62- (NSString *)MIMEType
63{
64 return resource->mimeType();
65}
66
67- (NSURL *)URL
68{
69 return resource->url();
70}
71
72@end
