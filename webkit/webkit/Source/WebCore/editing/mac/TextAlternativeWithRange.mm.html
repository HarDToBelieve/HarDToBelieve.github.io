38TextAlternativeWithRange::TextAlternativeWithRange(NSTextAlternatives *anAlternatives, NSRange aRange)
39 : range { aRange }
40 , alternatives { anAlternatives }
41{
42}
43
44void collectDictationTextAlternatives(NSAttributedString *string, Vector<TextAlternativeWithRange>& alternatives) {
45 NSRange effectiveRange = NSMakeRange(0, 0);
46 NSUInteger length = [string length];
47 do {
48 NSDictionary *attributes = [string attributesAtIndex:effectiveRange.location effectiveRange:&effectiveRange];
49 if (!attributes)
50 break;
51 NSTextAlternatives *textAlternatives = [attributes objectForKey:NSTextAlternativesAttributeName];
52 if (textAlternatives)
53 alternatives.append(TextAlternativeWithRange(textAlternatives, effectiveRange));
54 effectiveRange.location = NSMaxRange(effectiveRange);
55 } while (effectiveRange.location < length);
56}
