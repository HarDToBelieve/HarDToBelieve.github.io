48namespace WebCore {
49
50class Blob;
51class BlobLoader;
52class DeferredPromise;
53class ReadableStream;
54class ScriptExecutionContext;
55class SharedBuffer;
56
57template<typename> class ExceptionOr;
58
59using BlobPartVariant = Variant<RefPtr<JSC::ArrayBufferView>, RefPtr<JSC::ArrayBuffer>, RefPtr<Blob>, String>;
60
61class Blob : public ScriptWrappable, public URLRegistrable, public RefCounted<Blob>, public ActiveDOMObject {
62 WTF_MAKE_ISO_ALLOCATED_EXPORT(Blob, WEBCORE_EXPORT);
63public:
64 static Ref<Blob> create(ScriptExecutionContext* context)
65 {
66 auto blob = adoptRef(*new Blob(context));
67 blob->suspendIfNeeded();
68 return blob;
69 }
70
71 static Ref<Blob> create(ScriptExecutionContext& context, Vector<BlobPartVariant>&& blobPartVariants, const BlobPropertyBag& propertyBag)
72 {
73 auto blob = adoptRef(*new Blob(context, WTFMove(blobPartVariants), propertyBag));
74 blob->suspendIfNeeded();
75 return blob;
76 }
77
78 static Ref<Blob> create(ScriptExecutionContext* context, Vector<uint8_t>&& data, const String& contentType)
79 {
80 auto blob = adoptRef(*new Blob(context, WTFMove(data), contentType));
81 blob->suspendIfNeeded();
82 return blob;
83 }
84
85 static Ref<Blob> deserialize(ScriptExecutionContext* context, const URL& srcURL, const String& type, long long size, const String& fileBackedPath)
86 {
87 ASSERT(Blob::isNormalizedContentType(type));
88 auto blob = adoptRef(*new Blob(deserializationContructor, context, srcURL, type, size, fileBackedPath));
89 blob->suspendIfNeeded();
90 return blob;
91 }
92
93 virtual ~Blob();
94
95 const URL& url() const { return m_internalURL; }
96 const String& type() const { return m_type; }
97
98 WEBCORE_EXPORT unsigned long long size() const;
99 virtual bool isFile() const { return false; }
100
101 // The checks described in the File API spec.
102 static bool isValidContentType(const String&);
103 // The normalization procedure described in the File API spec.
104 static String normalizedContentType(const String&);
105#if ASSERT_ENABLED
106 static bool isNormalizedContentType(const String&);
107 static bool isNormalizedContentType(const CString&);
108#endif
109
110 // URLRegistrable
111 URLRegistry& registry() const override;
112
113 Ref<Blob> slice(ScriptExecutionContext& context, long long start, long long end, const String& contentType) const
114 {
115 auto blob = adoptRef(*new Blob(&context, m_internalURL, start, end, contentType));
116 blob->suspendIfNeeded();
117 return blob;
118 }
119
120 void text(ScriptExecutionContext&, Ref<DeferredPromise>&&);
121 void arrayBuffer(ScriptExecutionContext&, Ref<DeferredPromise>&&);
122 ExceptionOr<Ref<ReadableStream>> stream(ScriptExecutionContext&);
123
124protected:
125 WEBCORE_EXPORT explicit Blob(ScriptExecutionContext*);
126 Blob(ScriptExecutionContext&, Vector<BlobPartVariant>&&, const BlobPropertyBag&);
127 Blob(ScriptExecutionContext*, Vector<uint8_t>&&, const String& contentType);
128
129 enum ReferencingExistingBlobConstructor { referencingExistingBlobConstructor };
130 Blob(ReferencingExistingBlobConstructor, ScriptExecutionContext*, const Blob&);
131
132 enum UninitializedContructor { uninitializedContructor };
133 Blob(UninitializedContructor, ScriptExecutionContext*, URL&&, String&& type);
134
135 enum DeserializationContructor { deserializationContructor };
136 Blob(DeserializationContructor, ScriptExecutionContext*, const URL& srcURL, const String& type, std::optional<unsigned long long> size, const String& fileBackedPath);
137
138 // For slicing.
139 Blob(ScriptExecutionContext*, const URL& srcURL, long long start, long long end, const String& contentType);
140
141private:
142 void loadBlob(ScriptExecutionContext&, FileReaderLoader::ReadType, CompletionHandler<void(BlobLoader&)>&&);
143
144 // ActiveDOMObject.
145 const char* activeDOMObjectName() const override;
146
147 // This is an internal URL referring to the blob data associated with this object. It serves
148 // as an identifier for this blob. The internal URL is never used to source the blob's content
149 // into an HTML or for FileRead'ing, public blob URLs must be used for those purposes.
150 URL m_internalURL;
151
152 String m_type;
153
154 mutable std::optional<unsigned long long> m_size;
155 HashSet<std::unique_ptr<BlobLoader>> m_blobLoaders;
156};
157
158} // namespace WebCore
