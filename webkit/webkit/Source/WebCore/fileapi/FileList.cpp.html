29#include <wtf/IsoMallocInlines.h>

31namespace WebCore {
32
33WTF_MAKE_ISO_ALLOCATED_IMPL(FileList);
34
35File* FileList::item(unsigned index) const
36{
37 if (index >= m_files.size())
38 return nullptr;
39 return m_files[index].ptr();
40}
41
42Vector<String> FileList::paths() const
43{
44 Vector<String> paths;
45 paths.reserveInitialCapacity(m_files.size());
46 for (auto& file : m_files)
47 paths.uncheckedAppend(file->path());
48 return paths;
49}
