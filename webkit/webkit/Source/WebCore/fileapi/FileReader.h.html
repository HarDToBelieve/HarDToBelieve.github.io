44namespace JSC {
45class ArrayBuffer;
46}

48namespace WebCore {
49
50class Blob;
51
52class FileReader final : public RefCounted<FileReader>, public ActiveDOMObject, public EventTargetWithInlineData, private FileReaderLoaderClient {
53 WTF_MAKE_ISO_ALLOCATED(FileReader);
54public:
55 static Ref<FileReader> create(ScriptExecutionContext&);
56
57 virtual ~FileReader();
58
59 enum ReadyState {
60 EMPTY = 0,
61 LOADING = 1,
62 DONE = 2
63 };
64
65 ExceptionOr<void> readAsArrayBuffer(Blob&);
66 ExceptionOr<void> readAsBinaryString(Blob&);
67 ExceptionOr<void> readAsText(Blob&, const String& encoding);
68 ExceptionOr<void> readAsDataURL(Blob&);
69 void abort();
70
71 void doAbort();
72
73 ReadyState readyState() const { return m_state; }
74 DOMException* error() { return m_error.get(); }
75 FileReaderLoader::ReadType readType() const { return m_readType; }
76 std::optional<Variant<String, RefPtr<JSC::ArrayBuffer>>> result() const;
77
78 using RefCounted::ref;
79 using RefCounted::deref;
80
81private:
82 explicit FileReader(ScriptExecutionContext&);
83
84 // ActiveDOMObject.
85 const char* activeDOMObjectName() const final;
86 void stop() final;
87 bool virtualHasPendingActivity() const final;
88
89 EventTargetInterface eventTargetInterface() const final { return FileReaderEventTargetInterfaceType; }
90 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
91 void refEventTarget() final { ref(); }
92 void derefEventTarget() final { deref(); }
93
94 void enqueueTask(Function<void()>&&);
95
96 void didStartLoading() final;
97 void didReceiveData() final;
98 void didFinishLoading() final;
99 void didFail(ExceptionCode errorCode) final;
100
101 ExceptionOr<void> readInternal(Blob&, FileReaderLoader::ReadType);
102 void fireErrorEvent(int httpStatusCode);
103 void fireEvent(const AtomString& type);
104
105 ReadyState m_state { EMPTY };
106 bool m_finishedLoading { false };
107 RefPtr<Blob> m_blob;
108 FileReaderLoader::ReadType m_readType { FileReaderLoader::ReadAsBinaryString };
109 String m_encoding;
110 std::unique_ptr<FileReaderLoader> m_loader;
111 RefPtr<DOMException> m_error;
112 MonotonicTime m_lastProgressNotificationTime { MonotonicTime::nan() };
113 HashMap<uint64_t, Function<void()>> m_pendingTasks;
114};
