42FileReaderSync::FileReaderSync()
43{
44}
45
46ExceptionOr<RefPtr<ArrayBuffer>> FileReaderSync::readAsArrayBuffer(ScriptExecutionContext& scriptExecutionContext, Blob& blob)
47{
48 FileReaderLoader loader(FileReaderLoader::ReadAsArrayBuffer, nullptr);
49 auto result = startLoading(scriptExecutionContext, loader, blob);
50 if (result.hasException())
51 return result.releaseException();
52 return loader.arrayBufferResult();
53}
54
55ExceptionOr<String> FileReaderSync::readAsBinaryString(ScriptExecutionContext& scriptExecutionContext, Blob& blob)
56{
57 FileReaderLoader loader(FileReaderLoader::ReadAsBinaryString, nullptr);
58 return startLoadingString(scriptExecutionContext, loader, blob);
59}
60
61ExceptionOr<String> FileReaderSync::readAsText(ScriptExecutionContext& scriptExecutionContext, Blob& blob, const String& encoding)
62{
63 FileReaderLoader loader(FileReaderLoader::ReadAsText, nullptr);
64 loader.setEncoding(encoding);
65 return startLoadingString(scriptExecutionContext, loader, blob);
66}
67
68ExceptionOr<String> FileReaderSync::readAsDataURL(ScriptExecutionContext& scriptExecutionContext, Blob& blob)
69{
70 FileReaderLoader loader(FileReaderLoader::ReadAsDataURL, nullptr);
71 loader.setDataType(blob.type());
72 return startLoadingString(scriptExecutionContext, loader, blob);
73}
74
75ExceptionOr<void> FileReaderSync::startLoading(ScriptExecutionContext& scriptExecutionContext, FileReaderLoader& loader, Blob& blob)
76{
77 loader.start(&scriptExecutionContext, blob);
78 auto error = loader.errorCode();
79 if (!error)
80 return { };
81
82 return Exception { error.value() };
83}
84
85ExceptionOr<String> FileReaderSync::startLoadingString(ScriptExecutionContext& scriptExecutionContext, FileReaderLoader& loader, Blob& blob)
86{
87 auto result = startLoading(scriptExecutionContext, loader, blob);
88 if (result.hasException())
89 return result.releaseException();
90 return loader.stringResult();
91}
