44class Blob;

48class WEBCORE_EXPORT NetworkSendQueue : public ContextDestructionObserver {
49public:
50 using WriteString = Function<void(const CString& utf8)>;
51 using WriteRawData = Function<void(const Span<const uint8_t>&)>;
52 enum class Continue { No, Yes };
53 using ProcessError = Function<Continue(ExceptionCode)>;
54 NetworkSendQueue(ScriptExecutionContext&, WriteString&&, WriteRawData&&, ProcessError&&);
55 ~NetworkSendQueue();
56
57 void enqueue(CString&& utf8);
58 void enqueue(const JSC::ArrayBuffer&, unsigned byteOffset, unsigned byteLength);
59 void enqueue(Blob&);
60
61 void clear();
62
63private:
64 void processMessages();
65
66 using Message = Variant<CString, Ref<SharedBuffer>, UniqueRef<BlobLoader>>;
67 Deque<Message> m_queue;
68
69 WriteString m_writeString;
70 WriteRawData m_writeRawData;
71 ProcessError m_processError;
72};
