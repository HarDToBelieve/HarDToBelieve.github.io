58CachedPage::CachedPage(Page& page)
59 : m_page(page)
60 , m_expirationTime(MonotonicTime::now() + page.settings().backForwardCacheExpirationInterval())
61 , m_cachedMainFrame(makeUnique<CachedFrame>(page.mainFrame()))
62#if ENABLE(RESOURCE_LOAD_STATISTICS)
63 , m_loadedSubresourceDomains(page.mainFrame().loader().client().loadedSubresourceDomains())
64#endif
65{
66#ifndef NDEBUG
67 cachedPageCounter.increment();
68#endif
69}
70
71CachedPage::~CachedPage()
72{
73#ifndef NDEBUG
74 cachedPageCounter.decrement();
75#endif
76
77 if (m_cachedMainFrame)
78 m_cachedMainFrame->destroy();
79}

81static void firePageShowAndPopStateEvents(Page& page)
82{
83 // Dispatching JavaScript events can cause frame destruction.
84 auto& mainFrame = page.mainFrame();
85 Vector<Ref<Frame>> childFrames;
86 for (auto* child = mainFrame.tree().traverseNextInPostOrder(CanWrap::Yes); child; child = child->tree().traverseNextInPostOrder(CanWrap::No))
87 childFrames.append(*child);
88
89 for (auto& child : childFrames) {
90 if (!child->tree().isDescendantOf(&mainFrame))
91 continue;
92 auto* document = child->document();
93 if (!document)
94 continue;
95
96 // This takes care of firing the visibilitychange event and making sure the document is reported as visible.
97 document->setVisibilityHiddenDueToDismissal(false);
98
99 document->dispatchPageshowEvent(PageshowEventPersisted);
100
101 auto* historyItem = child->loader().history().currentItem();
102 if (historyItem && historyItem->stateObject())
103 document->dispatchPopstateEvent(historyItem->stateObject());
104 }
105}

107class CachedPageRestorationScope {
108public:
109 CachedPageRestorationScope(Page& page)
110 : m_page(page)

112 m_page.setIsRestoringCachedPage(true);

114
115 ~CachedPageRestorationScope()
116 {
117 m_page.setIsRestoringCachedPage(false);

124void CachedPage::restore(Page& page)
125{
126 ASSERT(m_cachedMainFrame);
127 ASSERT(m_cachedMainFrame->view()->frame().isMainFrame());
128 ASSERT(!page.subframeCount());
129
130 CachedPageRestorationScope restorationScope(page);
131 m_cachedMainFrame->open();
132
133 // Restore the focus appearance for the focused element.
134 // FIXME: Right now we don't support pages w/ frames in the b/f cache. This may need to be tweaked when we add support for that.
135 Document* focusedDocument = page.focusController().focusedOrMainFrame().document();
136 if (Element* element = focusedDocument->focusedElement()) {
137#if PLATFORM(IOS_FAMILY)
138 // We don't want focused nodes changing scroll position when restoring from the cache
139 // as it can cause ugly jumps before we manage to restore the cached position.
140 page.mainFrame().selection().suppressScrolling();
141
142 bool hadProhibitsScrolling = false;
143 FrameView* frameView = page.mainFrame().view();
144 if (frameView) {
145 hadProhibitsScrolling = frameView->prohibitsScrolling();
146 frameView->setProhibitsScrolling(true);
147 }
148#endif
149 element->updateFocusAppearance(SelectionRestorationMode::RestoreOrSelectAll);
150#if PLATFORM(IOS_FAMILY)
151 if (frameView)
152 frameView->setProhibitsScrolling(hadProhibitsScrolling);
153 page.mainFrame().selection().restoreScrolling();
154#endif
155 }
156
157 if (m_needsDeviceOrPageScaleChanged)
158 page.mainFrame().deviceOrPageScaleFactorChanged();
159
160 page.setNeedsRecalcStyleInAllFrames();
161
162#if ENABLE(VIDEO)
163 if (m_needsCaptionPreferencesChanged)
164 page.captionPreferencesChanged();
165#endif
166
167 if (m_needsUpdateContentsSize) {
168 if (FrameView* frameView = page.mainFrame().view())
169 frameView->updateContentsSize();
170 }
171
172 firePageShowAndPopStateEvents(page);
173
174#if ENABLE(RESOURCE_LOAD_STATISTICS)
175 for (auto& domain : m_loadedSubresourceDomains)
176 page.mainFrame().loader().client().didLoadFromRegistrableDomain(WTFMove(domain));
177#endif
178
179 clear();
180}
181
182void CachedPage::clear()
183{
184 ASSERT(m_cachedMainFrame);
185 m_cachedMainFrame->clear();
186 m_cachedMainFrame = nullptr;
187#if ENABLE(VIDEO)
188 m_needsCaptionPreferencesChanged = false;
189#endif
190 m_needsDeviceOrPageScaleChanged = false;
191 m_needsUpdateContentsSize = false;
192#if ENABLE(RESOURCE_LOAD_STATISTICS)
193 m_loadedSubresourceDomains.clear();
194#endif
195}
196
197bool CachedPage::hasExpired() const
198{
199 return MonotonicTime::now() > m_expirationTime;
200}
