44int64_t HistoryItem::generateSequenceNumber()
45{
46 // Initialize to the current time to reduce the likelihood of generating
47 // identifiers that overlap with those from past/future browser sessions.
48 static long long next = static_cast<long long>(WallTime::now().secondsSinceEpoch().microseconds());
49 return ++next;
50}

88inline HistoryItem::HistoryItem(const HistoryItem& item)
89 : RefCounted<HistoryItem>()
90 , m_urlString(item.m_urlString)
91 , m_originalURLString(item.m_originalURLString)
92 , m_referrer(item.m_referrer)
93 , m_target(item.m_target)
94 , m_title(item.m_title)
95 , m_displayTitle(item.m_displayTitle)
96 , m_scrollPosition(item.m_scrollPosition)
97 , m_pageScaleFactor(item.m_pageScaleFactor)
98 , m_lastVisitWasFailure(item.m_lastVisitWasFailure)
99 , m_isTargetItem(item.m_isTargetItem)
100 , m_itemSequenceNumber(item.m_itemSequenceNumber)
101 , m_documentSequenceNumber(item.m_documentSequenceNumber)
102 , m_formContentType(item.m_formContentType)
103 , m_pruningReason(PruningReason::None)
104#if PLATFORM(IOS_FAMILY)
105 , m_obscuredInsets(item.m_obscuredInsets)
106 , m_scale(item.m_scale)
107 , m_scaleIsInitial(item.m_scaleIsInitial)
108#endif
109 , m_identifier(item.m_identifier)
110{
111 if (item.m_formData)
112 m_formData = item.m_formData->copy();
113
114 unsigned size = item.m_children.size();
115 m_children.reserveInitialCapacity(size);
116 for (unsigned i = 0; i < size; ++i)
117 m_children.uncheckedAppend(item.m_children[i]->copy());
118}

331void HistoryItem::addChildItem(Ref<HistoryItem>&& child)
332{
333 ASSERT(!childItemWithTarget(child->target()));
334 m_children.append(WTFMove(child));
335}
336
337void HistoryItem::setChildItem(Ref<HistoryItem>&& child)
338{
339 ASSERT(!child->isTargetItem());
340 unsigned size = m_children.size();
341 for (unsigned i = 0; i < size; ++i) {
342 if (m_children[i]->target() == child->target()) {
343 child->setIsTargetItem(m_children[i]->isTargetItem());
344 m_children[i] = WTFMove(child);
345 return;
346 }
347 }
348 m_children.append(WTFMove(child));
349}

351HistoryItem* HistoryItem::childItemWithTarget(const String& target)

386// We do same-document navigation if going to a different item and if either of the following is true:
387// - The other item corresponds to the same document (for history entries created via pushState or fragment changes).
388// - The other item corresponds to the same set of documents, including frames (for history entries created via regular navigation)
389bool HistoryItem::shouldDoSameDocumentNavigationTo(HistoryItem& otherItem) const
390{
391 // The following logic must be kept in sync with WebKit::WebBackForwardListItem::itemIsInSameDocument().
392 if (this == &otherItem)
393 return false;
394
395 if (stateObject() || otherItem.stateObject())
396 return documentSequenceNumber() == otherItem.documentSequenceNumber();
397
398 if ((url().hasFragmentIdentifier() || otherItem.url().hasFragmentIdentifier()) && equalIgnoringFragmentIdentifier(url(), otherItem.url()))
399 return documentSequenceNumber() == otherItem.documentSequenceNumber();
400
401 return hasSameDocumentTree(otherItem);
402}
403
404// Does a recursive check that this item and its descendants have the same
405// document sequence numbers as the other item.
406bool HistoryItem::hasSameDocumentTree(HistoryItem& otherItem) const
407{
408 if (documentSequenceNumber() != otherItem.documentSequenceNumber())
409 return false;
410
411 if (children().size() != otherItem.children().size())
412 return false;
413
414 for (size_t i = 0; i < children().size(); i++) {
415 auto& child = children()[i].get();
416 auto* otherChild = otherItem.childItemWithDocumentSequenceNumber(child.documentSequenceNumber());
417 if (!otherChild || !child.hasSameDocumentTree(*otherChild))
418 return false;
419 }
420
421 return true;
422}
423
424// Does a non-recursive check that this item and its immediate children have the
425// same frames as the other item.
426bool HistoryItem::hasSameFrames(HistoryItem& otherItem) const
427{
428 if (target() != otherItem.target())
429 return false;
430
431 if (children().size() != otherItem.children().size())
432 return false;
433
434 for (size_t i = 0; i < children().size(); i++) {
435 if (!otherItem.childItemWithTarget(children()[i]->target()))
436 return false;
437 }
438
439 return true;
440}

447void HistoryItem::setFormInfoFromRequest(const ResourceRequest& request)
448{
449 m_referrer = request.httpReferrer();
450
451 if (equalLettersIgnoringASCIICase(request.httpMethod(), "post")) {
452 // FIXME: Eventually we have to make this smart enough to handle the case where
453 // we have a stream for the body to handle the "data interspersed with files" feature.
454 m_formData = request.httpBody();
455 m_formContentType = request.httpContentType();
456 } else {
457 m_formData = nullptr;
458 m_formContentType = String();
459 }
460}

495int HistoryItem::showTreeWithIndent(unsigned indentLevel) const
496{
497 Vector<char> prefix;
498 for (unsigned i = 0; i < indentLevel; ++i)
499 prefix.append(" ", 2);
500 prefix.append("\0", 1);
501
502 fprintf(stderr, "%s+-%s (%p)\n", prefix.data(), m_urlString.utf8().data(), this);
503
504 int totalSubItems = 0;
505 for (unsigned i = 0; i < m_children.size(); ++i)
506 totalSubItems += m_children[i]->showTreeWithIndent(indentLevel + 1);
507 return totalSubItems + 1;
508}
