45

53template <typename CollectionClass>
54inline auto CollectionTraversal<CollectionTraversalType::Descendants>::begin(const CollectionClass& collection, ContainerNode& rootNode) -> Iterator
55{
56 auto it = descendantsOfType<Element>(rootNode).begin();
57 while (it && !collection.elementMatches(*it))
58 ++it;
59 // Drop iterator assertions because HTMLCollections / NodeList use a fine-grained invalidation scheme.
60 it.dropAssertions();
61 return it;
62}
63
64template <typename CollectionClass>
65inline auto CollectionTraversal<CollectionTraversalType::Descendants>::last(const CollectionClass& collection, ContainerNode& rootNode) -> Iterator
66{
67 Iterator it { rootNode, ElementTraversal::lastWithin(rootNode) };
68 while (it && !collection.elementMatches(*it))
69 --it;
70 // Drop iterator assertions because HTMLCollections / NodeList use a fine-grained invalidation scheme.
71 it.dropAssertions();
72 return it;
73}
74
75template <typename CollectionClass>
76inline void CollectionTraversal<CollectionTraversalType::Descendants>::traverseForward(const CollectionClass& collection, Iterator& current, unsigned count, unsigned& traversedCount)
77{
78 ASSERT(collection.elementMatches(*current));
79 for (traversedCount = 0; traversedCount < count; ++traversedCount) {
80 do {
81 ++current;
82 if (!current)
83 return;
84 } while (!collection.elementMatches(*current));
85 }
86}
87
88template <typename CollectionClass>
89inline void CollectionTraversal<CollectionTraversalType::Descendants>::traverseBackward(const CollectionClass& collection, Iterator& current, unsigned count)
90{
91 ASSERT(collection.elementMatches(*current));
92 for (; count; --count) {
93 do {
94 --current;
95 if (!current)
96 return;
97 } while (!collection.elementMatches(*current));
98 }
99}

183};
184
185template <typename CollectionClass>
186inline Element* CollectionTraversal<CollectionTraversalType::CustomForwardOnly>::begin(const CollectionClass& collection, ContainerNode&)
187{
188 return collection.customElementAfter(nullptr);
189}
190
191template <typename CollectionClass>
192inline Element* CollectionTraversal<CollectionTraversalType::CustomForwardOnly>::last(const CollectionClass&, ContainerNode&)
193{
194 ASSERT_NOT_REACHED();
195 return nullptr;
196}
197
198template <typename CollectionClass>
199inline void CollectionTraversal<CollectionTraversalType::CustomForwardOnly>::traverseForward(const CollectionClass& collection, Element*& current, unsigned count, unsigned& traversedCount)
200{
201 Element* element = current;
202 for (traversedCount = 0; traversedCount < count; ++traversedCount) {
203 element = collection.customElementAfter(element);
204 if (!element) {
205 current = nullptr;
206 return;
207 }
208 }
209 current = element;
210}

212template <typename CollectionClass>
213inline void CollectionTraversal<CollectionTraversalType::CustomForwardOnly>::traverseBackward(const CollectionClass&, Element*&, unsigned count)
214{
215 UNUSED_PARAM(count);
216 ASSERT_NOT_REACHED();
217}
