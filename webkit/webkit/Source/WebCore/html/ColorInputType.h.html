42class ColorInputType final : public BaseClickableWithKeyInputType, private ColorChooserClient {

66 void setValue(const String&, bool valueChanged, TextFieldEventBehavior) final;
67 void attributeChanged(const QualifiedName&) final;
68 void handleDOMActivateEvent(Event&) final;
69 void detach() final;
70 void elementDidBlur() final;
71 bool shouldRespectListAttribute() final;
72 bool typeMismatchFor(const String&) const final;
73 bool shouldResetOnDocumentActivation() final;
74 Color valueAsColor() const final;
75 void selectColor(StringView) final;

77 void endColorChooser();
