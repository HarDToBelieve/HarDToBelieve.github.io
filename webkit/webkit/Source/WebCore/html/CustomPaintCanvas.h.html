44class CanvasRenderingContext;
45class ImageBitmap;

48class CustomPaintCanvas final : public RefCounted<CustomPaintCanvas>, public CanvasBase, private ContextDestructionObserver {
49 WTF_MAKE_FAST_ALLOCATED;
50public:
51
52 static Ref<CustomPaintCanvas> create(ScriptExecutionContext&, unsigned width, unsigned height);
53 virtual ~CustomPaintCanvas();
54 bool isCustomPaintCanvas() const final { return true; }
55
56 ExceptionOr<RefPtr<PaintRenderingContext2D>> getContext();
57
58 CanvasRenderingContext* renderingContext() const final { return m_context.get(); }
59 GraphicsContext* drawingContext() const final;
60 GraphicsContext* existingDrawingContext() const final;
61
62 void didDraw(const std::optional<FloatRect>&) final { }
63
64 AffineTransform baseTransform() const final { ASSERT(m_destinationGraphicsContext && m_copiedBuffer); return m_copiedBuffer->baseTransform(); }
65 Image* copiedImage() const final;
66 void clearCopiedImage() const final;
67
68 void replayDisplayList(GraphicsContext*) const;
69
70 using RefCounted::ref;
71 using RefCounted::deref;
72
73private:
74 CustomPaintCanvas(ScriptExecutionContext&, unsigned width, unsigned height);
75
76 void refCanvasBase() final { ref(); }
77 void derefCanvasBase() final { deref(); }
78 ScriptExecutionContext* canvasBaseScriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
79
80 std::unique_ptr<CanvasRenderingContext> m_context;
81 mutable GraphicsContext* m_destinationGraphicsContext = nullptr;
82 mutable RefPtr<ImageBuffer> m_copiedBuffer;
83 mutable RefPtr<Image> m_copiedImage;
84};
