33namespace WebCore {
34
35class Blob;
36class ScriptExecutionContext;
37class URLRegistrable;
38class URLSearchParams;
39
40class DOMURL final : public RefCounted<DOMURL>, public URLDecomposition {
41public:
42 static ExceptionOr<Ref<DOMURL>> create(const String& url, const String& base);
43 static ExceptionOr<Ref<DOMURL>> create(const String& url, const DOMURL& base);
44 ~DOMURL();
45
46 const URL& href() const { return m_url; }
47 ExceptionOr<void> setHref(const String&);
48 void setQuery(const String&);
49
50 URLSearchParams& searchParams();
51
52 const String& toJSON() const { return m_url.string(); }
53
54 static String createObjectURL(ScriptExecutionContext&, Blob&);
55 static void revokeObjectURL(ScriptExecutionContext&, const String&);
56
57 static String createPublicURL(ScriptExecutionContext&, URLRegistrable&);
58
59private:
60 static ExceptionOr<Ref<DOMURL>> create(const String& url, const URL& base);
61 DOMURL(URL&& completeURL, const URL& baseURL);
62
63 URL fullURL() const final { return m_url; }
64 void setFullURL(const URL& fullURL) final { setHref(fullURL.string()); }
65
66 URL m_baseURL;
67 URL m_url;
68 RefPtr<URLSearchParams> m_searchParams;
69};
