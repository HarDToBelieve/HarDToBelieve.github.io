76StepRange DateTimeLocalInputType::createStepRange(AnyStepHandling anyStepHandling) const
77{
78 ASSERT(element());
79 const Decimal stepBase = parseToNumber(element()->attributeWithoutSynchronization(minAttr), 0);
80 const Decimal minimum = parseToNumber(element()->attributeWithoutSynchronization(minAttr), Decimal::fromDouble(DateComponents::minimumDateTime()));
81 const Decimal maximum = parseToNumber(element()->attributeWithoutSynchronization(maxAttr), Decimal::fromDouble(DateComponents::maximumDateTime()));
82 const Decimal step = StepRange::parseStep(anyStepHandling, dateTimeLocalStepDescription, element()->attributeWithoutSynchronization(stepAttr));
83 return StepRange(stepBase, RangeLimitations::Valid, minimum, maximum, step, dateTimeLocalStepDescription);
84}

101String DateTimeLocalInputType::formatDateTimeFieldsState(const DateTimeFieldsState& state) const
102{
103 if (!state.year || !state.month || !state.dayOfMonth || !state.hour || !state.minute || !state.meridiem)
104 return emptyString();
105
106 auto dateString = makeString(pad('0', 4, *state.year), '-', pad('0', 2, *state.month), '-', pad('0', 2, *state.dayOfMonth));
107 auto hourMinuteString = makeString(pad('0', 2, *state.hour23()), ':', pad('0', 2, *state.minute));
108
109 if (state.millisecond)
110 return makeString(dateString, 'T', hourMinuteString, ':', pad('0', 2, state.second ? *state.second : 0), '.', pad('0', 3, *state.millisecond));
111
112 if (state.second)
113 return makeString(dateString, 'T', hourMinuteString, ':', pad('0', 2, *state.second));
114
115 return makeString(dateString, 'T', hourMinuteString);
116}

118void DateTimeLocalInputType::setupLayoutParameters(DateTimeEditElement::LayoutParameters& layoutParameters, const DateComponents& date) const
119{
120 layoutParameters.shouldHaveMillisecondField = shouldHaveMillisecondField(date);
121
122 if (layoutParameters.shouldHaveMillisecondField || shouldHaveSecondField(date)) {
123 layoutParameters.dateTimeFormat = layoutParameters.locale.dateTimeFormatWithSeconds();
124 layoutParameters.fallbackDateTimeFormat = "yyyy-MM-dd'T'HH:mm:ss"_s;
125 } else {
126 layoutParameters.dateTimeFormat = layoutParameters.locale.dateTimeFormatWithoutSeconds();
127 layoutParameters.fallbackDateTimeFormat = "yyyy-MM-dd'T'HH:mm"_s;
128 }
129}
