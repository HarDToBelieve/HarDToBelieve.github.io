34namespace WebCore {
35
36class Document;
37class FileList;
38
39struct FileChooserFileInfo;
40
41class DirectoryFileListCreator : public ThreadSafeRefCounted<DirectoryFileListCreator> {
42public:
43 using CompletionHandler = Function<void(Ref<FileList>&&)>;
44
45 static Ref<DirectoryFileListCreator> create(CompletionHandler&& completionHandler)
46 {
47 return adoptRef(*new DirectoryFileListCreator(WTFMove(completionHandler)));
48 }
49
50 ~DirectoryFileListCreator();
51
52 void start(Document*, const Vector<FileChooserFileInfo>&);
53 void cancel();
54
55private:
56 explicit DirectoryFileListCreator(CompletionHandler&&);
57
58 RefPtr<WorkQueue> m_workQueue;
59 CompletionHandler m_completionHandler;
60};
