65 void checkBuffer(int len = 10)
66 {
67 if ((m_dest - m_buffer) > m_size - len) {
68 // Enlarge buffer
69 int newSize = std::max(m_size * 2, m_size + len);
70 int oldOffset = m_dest - m_buffer;
71 m_buffer = static_cast<UChar*>(fastRealloc(m_buffer, newSize * sizeof(UChar)));
72 m_dest = m_buffer + oldOffset;
73 m_size = newSize;
74 }
75 }

108void FTPDirectoryDocumentParser::appendEntry(const String& filename, const String& size, const String& date, bool isDirectory)
109{
110 auto& document = *this->document();
111
112 auto rowElement = m_tableElement->insertRow(-1).releaseReturnValue();
113 rowElement->setAttributeWithoutSynchronization(HTMLNames::classAttr, AtomString("ftpDirectoryEntryRow", AtomString::ConstructFromLiteral));
114
115 auto typeElement = HTMLTableCellElement::create(tdTag, document);
116 typeElement->appendChild(Text::create(document, String(&noBreakSpace, 1)));
117 if (isDirectory)
118 typeElement->setAttributeWithoutSynchronization(HTMLNames::classAttr, AtomString("ftpDirectoryIcon ftpDirectoryTypeDirectory", AtomString::ConstructFromLiteral));
119 else
120 typeElement->setAttributeWithoutSynchronization(HTMLNames::classAttr, AtomString("ftpDirectoryIcon ftpDirectoryTypeFile", AtomString::ConstructFromLiteral));
121 rowElement->appendChild(typeElement);
122
123 auto nameElement = createTDForFilename(filename);
124 nameElement->setAttributeWithoutSynchronization(HTMLNames::classAttr, AtomString("ftpDirectoryFileName", AtomString::ConstructFromLiteral));
125 rowElement->appendChild(nameElement);
126
127 auto dateElement = HTMLTableCellElement::create(tdTag, document);
128 dateElement->appendChild(Text::create(document, date));
129 dateElement->setAttributeWithoutSynchronization(HTMLNames::classAttr, AtomString("ftpDirectoryFileDate", AtomString::ConstructFromLiteral));
130 rowElement->appendChild(dateElement);
131
132 auto sizeElement = HTMLTableCellElement::create(tdTag, document);
133 sizeElement->appendChild(Text::create(document, size));
134 sizeElement->setAttributeWithoutSynchronization(HTMLNames::classAttr, AtomString("ftpDirectoryFileSize", AtomString::ConstructFromLiteral));
135 rowElement->appendChild(sizeElement);
136 document.setHasVisuallyNonEmptyCustomContent();
137}

139Ref<Element> FTPDirectoryDocumentParser::createTDForFilename(const String& filename)
140{
141 auto& document = *this->document();
142
143 String fullURL = document.baseURL().string();
144 if (fullURL.endsWith('/'))
145 fullURL = fullURL + filename;
146 else
147 fullURL = fullURL + '/' + filename;
148
149 auto anchorElement = HTMLAnchorElement::create(document);
150 anchorElement->setAttributeWithoutSynchronization(HTMLNames::hrefAttr, fullURL);
151 anchorElement->appendChild(Text::create(document, filename));
152
153 auto tdElement = HTMLTableCellElement::create(tdTag, document);
154 tdElement->appendChild(anchorElement);
155
156 return tdElement;
157}

159static String processFilesizeString(const String& size, bool isDirectory)
160{
161 if (isDirectory)
162 return "--"_s;
163
164 auto bytes = parseIntegerAllowingTrailingJunk<uint64_t>(size);
165 if (!bytes)
166 return unknownFileSizeText();
167
168 if (*bytes < 1000000)
169 return makeString(FormattedNumber::fixedWidth(*bytes / 1000.0, 2), " KB");
170 if (*bytes < 1000000000)
171 return makeString(FormattedNumber::fixedWidth(*bytes / 1000000.0, 2), " MB");
172 return makeString(FormattedNumber::fixedWidth(*bytes / 1000000000.0, 2), " GB");
173}

175static bool wasLastDayOfMonth(int year, int month, int day)
176{
177 static const int lastDays[] = { 31, 0, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };
178 if (month < 0 || month > 11)
179 return false;
180
181 if (month == 2) {
182 if (year % 4 == 0 && (year % 100 || year % 400 == 0)) {
183 if (day == 29)
184 return true;
185 return false;
186 }
187
188 if (day == 28)
189 return true;
190 return false;
191 }
192
193 return lastDays[month] == day;
194}

196static String processFileDateString(const FTPTime& fileTime)
197{
198 // FIXME: Need to localize this string?
199
200 String timeOfDay;
201
202 if (!(fileTime.tm_hour == 0 && fileTime.tm_min == 0 && fileTime.tm_sec == 0)) {
203 int hour = fileTime.tm_hour;
204 ASSERT(hour >= 0 && hour < 24);
205
206 if (hour < 12) {
207 if (hour == 0)
208 hour = 12;
209 timeOfDay = makeString(", ", hour, ':', pad('0', 2, fileTime.tm_min), " AM");
210 } else {
211 hour = hour - 12;
212 if (hour == 0)
213 hour = 12;
214 timeOfDay = makeString(", ", hour, ':', pad('0', 2, fileTime.tm_min), " PM");
215 }
216 }
217
218 // If it was today or yesterday, lets just do that - but we have to compare to the current time
219 GregorianDateTime now;
220 now.setToCurrentLocalTime();
221
222 if (fileTime.tm_year == now.year()) {
223 if (fileTime.tm_mon == now.month()) {
224 if (fileTime.tm_mday == now.monthDay())
225 return "Today" + timeOfDay;
226 if (fileTime.tm_mday == now.monthDay() - 1)
227 return "Yesterday" + timeOfDay;
228 }
229
230 if (now.monthDay() == 1 && (now.month() == fileTime.tm_mon + 1 || (now.month() == 0 && fileTime.tm_mon == 11)) &&
231 wasLastDayOfMonth(fileTime.tm_year, fileTime.tm_mon, fileTime.tm_mday))
232 return "Yesterday" + timeOfDay;
233 }
234
235 if (fileTime.tm_year == now.year() - 1 && fileTime.tm_mon == 12 && fileTime.tm_mday == 31 && now.month() == 1 && now.monthDay() == 1)
236 return "Yesterday" + timeOfDay;
237
238 static const char* months[] = { "Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec", "???" };
239
240 int month = fileTime.tm_mon;
241 if (month < 0 || month > 11)
242 month = 12;
243
244 String dateString;
245
246 if (fileTime.tm_year > -1)
247 dateString = makeString(months[month], ' ', fileTime.tm_mday, ", ", fileTime.tm_year);
248 else
249 dateString = makeString(months[month], ' ', fileTime.tm_mday, ", ", now.year());
250
251 return dateString + timeOfDay;
252}

254void FTPDirectoryDocumentParser::parseAndAppendOneLine(const String& inputLine)
255{
256 ListResult result;
257 CString latin1Input = inputLine.latin1();
258
259 FTPEntryType typeResult = parseOneFTPLine(latin1Input.data(), m_listState, result);
260
261 // FTPMiscEntry is a comment or usage statistic which we don't care about, and junk is invalid data - bail in these 2 cases
262 if (typeResult == FTPMiscEntry || typeResult == FTPJunkEntry)
263 return;
264
265 String filename(result.filename, result.filenameLength);
266 if (result.type == FTPDirectoryEntry) {
267 filename.append('/');
268
269 // We have no interest in linking to "current directory"
270 if (filename == "./")
271 return;
272 }
273
274 LOG(FTP, "Appending entry - %s, %s", filename.ascii().data(), result.fileSize.ascii().data());
275
276 appendEntry(filename, processFilesizeString(result.fileSize, result.type == FTPDirectoryEntry), processFileDateString(result.modifiedTime), result.type == FTPDirectoryEntry);
277}

287bool FTPDirectoryDocumentParser::loadDocumentTemplate()
288{
289 static SharedBuffer* templateDocumentData = createTemplateDocumentData(document()->settings()).leakRef();
290 // FIXME: Instead of storing the data, it would be more efficient if we could parse the template data into the
291 // template Document once, store that document, then "copy" it whenever we get an FTP directory listing.
292
293 if (!templateDocumentData) {
294 LOG_ERROR("Could not load templateData");
295 return false;
296 }
297
298 HTMLDocumentParser::insert(String(templateDocumentData->data(), templateDocumentData->size()));
299
300 auto& document = *this->document();
301
302 auto foundElement = makeRefPtr(document.getElementById(String("ftpDirectoryTable"_s)));
303 if (!foundElement)
304 LOG_ERROR("Unable to find element by id \"ftpDirectoryTable\" in the template document.");
305 else if (!is<HTMLTableElement>(foundElement))
306 LOG_ERROR("Element of id \"ftpDirectoryTable\" is not a table element");
307 else {
308 m_tableElement = downcast<HTMLTableElement>(foundElement.get());
309 return true;
310 }
311
312 m_tableElement = HTMLTableElement::create(document);
313 m_tableElement->setAttributeWithoutSynchronization(HTMLNames::idAttr, AtomString("ftpDirectoryTable", AtomString::ConstructFromLiteral));
314
315 // If we didn't find the table element, lets try to append our own to the body.
316 // If that fails for some reason, cram it on the end of the document as a last ditch effort.
317 if (auto body = makeRefPtr(document.bodyOrFrameset()))
318 body->appendChild(*m_tableElement);
319 else
320 document.appendChild(*m_tableElement);
321
322 return true;
323}

325void FTPDirectoryDocumentParser::createBasicDocument()
326{
327 LOG(FTP, "Creating a basic FTP document structure as no template was loaded");
328
329 auto& document = *this->document();
330
331 auto bodyElement = HTMLBodyElement::create(document);
332 document.appendChild(bodyElement);
333
334 m_tableElement = HTMLTableElement::create(document);
335 m_tableElement->setAttributeWithoutSynchronization(HTMLNames::idAttr, AtomString("ftpDirectoryTable", AtomString::ConstructFromLiteral));
336 m_tableElement->setAttribute(HTMLNames::styleAttr, AtomString("width:100%", AtomString::ConstructFromLiteral));
337
338 bodyElement->appendChild(*m_tableElement);
339
340 document.processViewport("width=device-width", ViewportArguments::ViewportMeta);
341}

343void FTPDirectoryDocumentParser::append(RefPtr<StringImpl>&& inputSource)
344{
345 // Make sure we have the table element to append to by loading the template set in the pref, or
346 // creating a very basic document with the appropriate table
347 if (!m_tableElement) {
348 if (!loadDocumentTemplate())
349 createBasicDocument();
350 ASSERT(m_tableElement);
351 }
352
353 bool foundNewLine = false;
354
355 m_dest = m_buffer;
356 SegmentedString string { String { WTFMove(inputSource) } };
357 while (!string.isEmpty()) {
358 UChar c = string.currentCharacter();
359
360 if (c == '\r') {
361 *m_dest++ = '\n';
362 foundNewLine = true;
363 // possibly skip an LF in the case of an CRLF sequence
364 m_skipLF = true;
365 } else if (c == '\n') {
366 if (!m_skipLF)
367 *m_dest++ = c;
368 else
369 m_skipLF = false;
370 } else {
371 *m_dest++ = c;
372 m_skipLF = false;
373 }
374
375 string.advance();
376
377 // Maybe enlarge the buffer
378 checkBuffer();
379 }
380
381 if (!foundNewLine) {
382 m_dest = m_buffer;
383 return;
384 }
385
386 UChar* start = m_buffer;
387 UChar* cursor = start;
388
389 while (cursor < m_dest) {
390 if (*cursor == '\n') {
391 m_carryOver.append(String(start, cursor - start));
392 LOG(FTP, "%s", m_carryOver.ascii().data());
393 parseAndAppendOneLine(m_carryOver);
394 m_carryOver = String();
395
396 start = ++cursor;
397 } else
398 cursor++;
399 }
400
401 // Copy the partial line we have left to the carryover buffer
402 if (cursor - start > 1)
403 m_carryOver.append(String(start, cursor - start - 1));
404}

406void FTPDirectoryDocumentParser::finish()
407{
408 // Possible the last line in the listing had no newline, so try to parse it now
409 if (!m_carryOver.isEmpty()) {
410 parseAndAppendOneLine(m_carryOver);
411 m_carryOver = String();
412 }
413
414 m_tableElement = nullptr;
415 fastFree(m_buffer);
416
417 HTMLDocumentParser::finish();
418}
