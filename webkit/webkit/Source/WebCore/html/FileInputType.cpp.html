139FormControlState FileInputType::saveFormControlState() const
140{
141 if (m_fileList->isEmpty())
142 return { };
143
144 auto length = Checked<size_t>(m_fileList->files().size()) * Checked<size_t>(2);
145
146 Vector<String> stateVector;
147 stateVector.reserveInitialCapacity(length);
148 for (auto& file : m_fileList->files()) {
149 stateVector.uncheckedAppend(file->path());
150 stateVector.uncheckedAppend(file->name());
151 }
152 return FormControlState { WTFMove(stateVector) };
153}

160bool FileInputType::appendFormData(DOMFormData& formData, bool multipart) const
161{
162 ASSERT(element());
163 auto fileList = makeRefPtr(element()->files());
164 ASSERT(fileList);
165
166 auto name = element()->name();
167
168 if (!multipart) {
169 // Send only the basenames.
170 // 4.10.16.4 and 4.10.16.6 sections in HTML5.
171
172 // Unlike the multipart case, we have no special handling for the empty
173 // fileList because Netscape doesn't support for non-multipart
174 // submission of file inputs, and Firefox doesn't add "name=" query
175 // parameter.
176 for (auto& file : fileList->files())
177 formData.append(name, file->name());
178 return true;
179 }
180
181 // If no filename at all is entered, return successful but empty.
182 // Null would be more logical, but Netscape posts an empty file. Argh.
183 if (fileList->isEmpty()) {
184 auto* document = element() ? &element()->document() : nullptr;
185 formData.append(name, File::create(document, emptyString()));
186 return true;
187 }
188
189
190 for (auto& file : fileList->files())
191 formData.append(name, file.get());
192 return true;
193}

208{
209 ASSERT(element());
210 auto& input = *element();
211
212 if (input.isDisabledFormControl())
213 return;
214
215 if (!UserGestureIndicator::processingUserGesture())
216 return;
217
218 if (auto* chrome = this->chrome()) {
219 applyFileChooserSettings();
220 chrome->runOpenPanel(*input.document().frame(), *m_fileChooser);
221 }
222
223 event.setDefaultHandled();
224}

242bool FileInputType::canSetValue(const String& value)
243{
244 // For security reasons, we don't allow setting the filename, but we do allow clearing it.
245 // The HTML5 spec (as of the 10/24/08 working draft) says that the value attribute isn't
246 // applicable to the file upload control at all, but for now we are keeping this behavior
247 // to avoid breaking existing websites that may be relying on this.
248 return value.isEmpty();
249}
250
251bool FileInputType::getTypeSpecificValue(String& value)
252{
253 if (m_fileList->isEmpty()) {
254 value = { };
255 return true;
256 }
257
258 // HTML5 tells us that we're supposed to use this goofy value for
259 // file input controls. Historically, browsers revealed the real
260 // file path, but that's a privacy problem. Code on the web
261 // decided to try to parse the value by looking for backslashes
262 // (because that's what Windows file paths use). To be compatible
263 // with that code, we make up a fake path for the file.
264 value = makeString("C:\\fakepath\\", m_fileList->file(0).name());
265 return true;
266}

312void FileInputType::requestIcon(const Vector<String>& paths)
313{
314 if (!paths.size()) {
315 iconLoaded(nullptr);
316 return;
317 }
318
319 auto* chrome = this->chrome();
320 if (!chrome) {
321 iconLoaded(nullptr);
322 return;
323 }
324
325 if (m_fileIconLoader)
326 m_fileIconLoader->invalidate();
327
328 FileIconLoaderClient& client = *this;
329 m_fileIconLoader = makeUnique<FileIconLoader>(client);
330
331 chrome->loadIconForFiles(paths, *m_fileIconLoader);
332}

334FileChooserSettings FileInputType::fileChooserSettings() const
335{
336 ASSERT(element());
337 auto& input = *element();
338
339 FileChooserSettings settings;
340 settings.allowsDirectories = allowsDirectories();
341 settings.allowsMultipleFiles = input.hasAttributeWithoutSynchronization(multipleAttr);
342 settings.acceptMIMETypes = input.acceptMIMETypes();
343 settings.acceptFileExtensions = input.acceptFileExtensions();
344 settings.selectedFiles = m_fileList->paths();
345#if ENABLE(MEDIA_CAPTURE)

372void FileInputType::setFiles(RefPtr<FileList>&& files, RequestIcon shouldRequestIcon)
373{
374 if (!files)
375 return;
376
377 ASSERT(element());
378 Ref<HTMLInputElement> protectedInputElement(*element());
379
380 unsigned length = files->length();
381
382 bool pathsChanged = false;
383 if (length != m_fileList->length())
384 pathsChanged = true;
385 else {
386 for (unsigned i = 0; i < length; ++i) {
387 if (files->file(i).path() != m_fileList->file(i).path()) {
388 pathsChanged = true;
389 break;
390 }
391 }
392 }
393
394 m_fileList = files.releaseNonNull();
395
396 protectedInputElement->setFormControlValueMatchesRenderer(true);
397 protectedInputElement->updateValidity();
398
399 if (shouldRequestIcon == RequestIcon::Yes) {
400 Vector<String> paths;
401 paths.reserveInitialCapacity(length);
402 for (auto& file : m_fileList->files())
403 paths.uncheckedAppend(file->path());
404 requestIcon(paths);
405 }
406
407 if (protectedInputElement->renderer())
408 protectedInputElement->renderer()->repaint();
409
410 if (pathsChanged) {
411 // This call may cause destruction of this instance.
412 // input instance is safe since it is ref-counted.
413 protectedInputElement->dispatchInputEvent();
414 protectedInputElement->dispatchChangeEvent();
415 }
416 protectedInputElement->setChangedSinceLastFormControlChangeEvent(false);
417}

419void FileInputType::filesChosen(const Vector<FileChooserFileInfo>& paths, const String& displayString, Icon* icon)
420{
421 if (!displayString.isEmpty())
422 m_displayString = displayString;
423
424 if (m_directoryFileListCreator)
425 m_directoryFileListCreator->cancel();
426
427 auto* document = element() ? &element()->document() : nullptr;
428 if (!allowsDirectories()) {
429 auto files = paths.map([document](auto& fileInfo) {
430 return File::create(document, fileInfo.path, fileInfo.replacementPath, fileInfo.displayName);
431 });
432 didCreateFileList(FileList::create(WTFMove(files)), icon);
433 return;
434 }
435
436 m_directoryFileListCreator = DirectoryFileListCreator::create([this, weakThis = makeWeakPtr(*this), icon = makeRefPtr(icon)](Ref<FileList>&& fileList) mutable {
437 ASSERT(isMainThread());
438 if (!weakThis)
439 return;
440 didCreateFileList(WTFMove(fileList), WTFMove(icon));
441 });
442 m_directoryFileListCreator->start(document, paths);
443}

445void FileInputType::filesChosen(const Vector<String>& paths, const Vector<String>& replacementPaths)
446{
447 ASSERT(element());
448 ASSERT(!paths.isEmpty());
449
450 size_t size = element()->hasAttributeWithoutSynchronization(multipleAttr) ? paths.size() : 1;
451
452 Vector<FileChooserFileInfo> files;
453 files.reserveInitialCapacity(size);
454
455 for (size_t i = 0; i < size; ++i)
456 files.uncheckedAppend({ paths[i], i < replacementPaths.size() ? replacementPaths[i] : nullString(), { } });
457
458 filesChosen(files);
459}

461void FileInputType::didCreateFileList(Ref<FileList>&& fileList, RefPtr<Icon>&& icon)
462{
463 auto protectedThis = makeRef(*this);
464
465 ASSERT(!allowsDirectories() || m_directoryFileListCreator);
466 m_directoryFileListCreator = nullptr;
467
468 setFiles(WTFMove(fileList), icon ? RequestIcon::Yes : RequestIcon::No);
469 if (icon && !m_fileList->isEmpty() && element())
470 iconLoaded(WTFMove(icon));
471}

489#if ENABLE(DRAG_SUPPORT)
490bool FileInputType::receiveDroppedFilesWithImageTranscoding(const Vector<String>& paths)
491{
492#if PLATFORM(MAC)
493 auto settings = fileChooserSettings();
494 auto allowedMIMETypes = MIMETypeRegistry::allowedMIMETypes(settings.acceptMIMETypes, settings.acceptFileExtensions);
495
496 auto transcodingPaths = findImagesForTranscoding(paths, allowedMIMETypes);
497 if (transcodingPaths.isEmpty())
498 return { };
499
500 auto transcodingMIMEType = MIMETypeRegistry::preferredImageMIMETypeForEncoding(allowedMIMETypes, { });
501 if (transcodingMIMEType.isNull())
502 return { };
503
504 auto transcodingUTI = WebCore::UTIFromMIMEType(transcodingMIMEType);
505 auto transcodingExtension = WebCore::MIMETypeRegistry::preferredExtensionForMIMEType(transcodingMIMEType);
506
507 auto callFilesChosen = [protectedThis = makeRef(*this), paths](const Vector<String>& replacementPaths) {
508 protectedThis->filesChosen(paths, replacementPaths);
509 };
510
511 sharedImageTranscodingQueue().dispatch([callFilesChosen = WTFMove(callFilesChosen), transcodingPaths = transcodingPaths.isolatedCopy(), transcodingUTI = transcodingUTI.isolatedCopy(), transcodingExtension = transcodingExtension.isolatedCopy()]() mutable {
512 ASSERT(!RunLoop::isMain());
513
514 auto replacementPaths = transcodeImages(transcodingPaths, transcodingUTI, transcodingExtension);
515 ASSERT(transcodingPaths.size() == replacementPaths.size());
516
517 RunLoop::main().dispatch([callFilesChosen = WTFMove(callFilesChosen), replacementPaths = replacementPaths.isolatedCopy()]() {
518 callFilesChosen(replacementPaths);
519 });
520 });
521
522 return true;
523#else
524 UNUSED_PARAM(paths);
525 return false;
526#endif
527}

529bool FileInputType::receiveDroppedFiles(const DragData& dragData)
530{
531 auto paths = dragData.asFilenames();
532 if (paths.isEmpty())
533 return false;
534
535 if (receiveDroppedFilesWithImageTranscoding(paths))
536 return true;
537
538 filesChosen(paths);
539 return true;
540}
541#endif // ENABLE(DRAG_SUPPORT)

548String FileInputType::defaultToolTip() const
549{
550 unsigned listSize = m_fileList->length();
551 if (!listSize) {
552 ASSERT(element());
553 if (element()->multiple())
554 return fileButtonNoFilesSelectedLabel();
555 return fileButtonNoFileSelectedLabel();
556 }
557
558 StringBuilder names;
559 for (unsigned i = 0; i < listSize; ++i) {
560 names.append(m_fileList->file(i).name());
561 if (i != listSize - 1)
562 names.append('\n');
563 }
564 return names.toString();
565}
