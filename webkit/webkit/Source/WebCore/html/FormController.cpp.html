37static inline HTMLFormElement* ownerFormForState(const HTMLFormControlElementWithState& control)
38{
39 // Assume controls with form attribute have no owners because we restore
40 // state during parsing and form owners of such controls might be
41 // indeterminate.
42 return control.hasAttributeWithoutSynchronization(formAttr) ? 0 : control.form();
43}

45// ----------------------------------------------------------------------------
46
47// Serilized form of FormControlState:
48// (',' means strings around it are separated in stateVector.)
49//
50// SerializedControlState ::= SkipState | RestoreState
51// SkipState ::= '0'
52// RestoreState ::= UnsignedNumber, ControlValue+
53// UnsignedNumber ::= [0-9]+
54// ControlValue ::= arbitrary string
55//
56// RestoreState has a sequence of ControlValues. The length of the
57// sequence is represented by UnsignedNumber.
58
59static inline void serializeFormControlStateTo(const FormControlState& formControlState, Vector<String>& stateVector)
60{
61 stateVector.append(String::number(formControlState.size()));
62 for (auto& value : formControlState)
63 stateVector.append(value.isNull() ? emptyString() : value);
64}
65
66static inline std::optional<FormControlState> deserializeFormControlState(const Vector<String>& stateVector, size_t& index)
67{
68 if (index >= stateVector.size())
69 return std::nullopt;
70 auto size = parseIntegerAllowingTrailingJunk<size_t>(stateVector[index++]).value_or(0);
71 if (index + size > stateVector.size())
72 return std::nullopt;
73 Vector<String> subvector;
74 subvector.reserveInitialCapacity(size);
75 for (size_t i = 0; i < size; ++i)
76 subvector.uncheckedAppend(stateVector[index++]);
77 return subvector;
78}

173class SavedFormState {
174 WTF_MAKE_NONCOPYABLE(SavedFormState);
175 WTF_MAKE_FAST_ALLOCATED;
176
177public:
178 SavedFormState() = default;
179 static std::unique_ptr<SavedFormState> deserialize(const Vector<String>&, size_t& index);
180 void serializeTo(Vector<String>&) const;
181 bool isEmpty() const { return m_stateForNewFormElements.isEmpty(); }

183 FormControlState takeControlState(const AtomString& name, const AtomString& type);
184
185 Vector<String> referencedFilePaths() const;
186
187private:
188 HashMap<FormElementKey, Deque<FormControlState>, FormElementKeyHash, FormElementKeyHashTraits> m_stateForNewFormElements;
189 size_t m_controlStateCount { 0 };
190};

192static bool isNotFormControlTypeCharacter(UChar ch)
193{
194 return !(ch == '-' || isASCIILower(ch));
195}
196
197std::unique_ptr<SavedFormState> SavedFormState::deserialize(const Vector<String>& stateVector, size_t& index)
198{
199 if (index >= stateVector.size())
200 return nullptr;
201 auto itemCount = parseIntegerAllowingTrailingJunk<size_t>(stateVector[index++]).value_or(0);
202 if (!itemCount)
203 return nullptr;
204 auto savedFormState = makeUnique<SavedFormState>();
205 while (itemCount--) {
206 if (index + 1 >= stateVector.size())
207 return nullptr;
208 String name = stateVector[index++];
209 String type = stateVector[index++];
210 auto state = deserializeFormControlState(stateVector, index);
211 if (type.isEmpty() || type.find(isNotFormControlTypeCharacter) != notFound || !state)
212 return nullptr;
213 savedFormState->appendControlState(name, type, state.value());
214 }
215 return savedFormState;
216}
217
218void SavedFormState::serializeTo(Vector<String>& stateVector) const
219{
220 stateVector.append(String::number(m_controlStateCount));
221 for (auto& element : m_stateForNewFormElements) {
222 const FormElementKey& key = element.key;
223 for (auto& controlState : element.value) {
224 stateVector.append(key.name());
225 stateVector.append(key.type());
226 serializeFormControlStateTo(controlState, stateVector);
227 }
228 }
229}
230
231void SavedFormState::appendControlState(const AtomString& name, const AtomString& type, const FormControlState& state)
232{
233 m_stateForNewFormElements.add(FormElementKey { name.impl(), type.impl() }, Deque<FormControlState> { }).iterator->value.append(state);
234 ++m_controlStateCount;
235}
236
237FormControlState SavedFormState::takeControlState(const AtomString& name, const AtomString& type)
238{
239 auto iterator = m_stateForNewFormElements.find(FormElementKey { name.impl(), type.impl() });
240 if (iterator == m_stateForNewFormElements.end())
241 return { };
242
243 auto state = iterator->value.takeFirst();
244 --m_controlStateCount;
245 if (iterator->value.isEmpty())
246 m_stateForNewFormElements.remove(iterator);
247 return state;
248}
249
250Vector<String> SavedFormState::referencedFilePaths() const
251{
252 Vector<String> toReturn;
253 for (auto& element : m_stateForNewFormElements) {
254 if (!equal(element.key.type(), "file", 4))
255 continue;
256 for (auto& state : element.value) {
257 for (auto& file : HTMLInputElement::filesFromFileInputFormControlState(state))
258 toReturn.append(file.path);
259 }
260 }
261 return toReturn;
262}

280static inline void recordFormStructure(const HTMLFormElement& form, StringBuilder& builder)
281{
282 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
283 // 2 is enough to distinguish forms in webkit.org/b/91209#c0
284 const size_t namedControlsToBeRecorded = 2;
285 auto& controls = form.unsafeAssociatedElements();
286 builder.append(" [");
287 for (size_t i = 0, namedControls = 0; i < controls.size() && namedControls < namedControlsToBeRecorded; ++i) {
288 auto* formAssociatedElement = controls[i]->asFormAssociatedElement();
289 if (!formAssociatedElement->isFormControlElementWithState())
290 continue;
291 RefPtr<HTMLFormControlElementWithState> control = static_cast<HTMLFormControlElementWithState*>(formAssociatedElement);
292 if (!ownerFormForState(*control))
293 continue;
294 AtomString name = control->name();
295 if (name.isEmpty())
296 continue;
297 namedControls++;
298 builder.append(name, ' ');
299 }
300 builder.append(']');
301}

303static inline String formSignature(const HTMLFormElement& form)
304{
305 URL actionURL = form.getURLAttribute(actionAttr);
306 // Remove the query part because it might contain volatile parameters such
307 // as a session key.
308 actionURL.setQuery(String());
309 StringBuilder builder;
310 if (!actionURL.isEmpty())
311 builder.append(actionURL.string());
312
313 recordFormStructure(form, builder);
314 return builder.toString();
315}
316
317AtomString FormKeyGenerator::formKey(const HTMLFormControlElementWithState& control)
318{
319 auto form = makeRefPtr(ownerFormForState(control));
320 if (!form) {
321 static MainThreadNeverDestroyed<const AtomString> formKeyForNoOwner("No owner", AtomString::ConstructFromLiteral);
322 return formKeyForNoOwner;
323 }
324
325 return m_formToKeyMap.ensure(*form, [this, &form] {
326 auto signature = formSignature(*form);
327 auto nextIndex = m_formSignatureToNextIndexMap.add(signature, 0).iterator->value++;
328 // FIXME: Would be nice to have makeAtomString to use to optimize the case where the string already exists.
329 return makeString(signature, " #", nextIndex);
330 }).iterator->value;
331}

333void FormKeyGenerator::willDeleteForm(HTMLFormElement* form)
334{
335 RELEASE_ASSERT(form);
336 m_formToKeyMap.remove(*form);
337}

345unsigned FormController::formElementsCharacterCount() const
346{

355static String formStateSignature()
356{
357 // In the legacy version of serialized state, the first item was a name
358 // attribute value of a form control. The following string literal should
359 // contain some characters which are rarely used for name attribute values.
360 static NeverDestroyed<String> signature(MAKE_STATIC_STRING_IMPL("\n\r?% WebKit serialized form state version 8 \n\r=&"));
361 return signature;
362}

364std::unique_ptr<FormController::SavedFormStateMap> FormController::createSavedFormStateMap(const FormElementListHashSet& controlList)
365{
366 FormKeyGenerator keyGenerator;
367 auto stateMap = makeUnique<SavedFormStateMap>();
368 for (auto& control : controlList) {
369 if (!control->shouldSaveAndRestoreFormControlState())
370 continue;
371 auto& formState = stateMap->add(keyGenerator.formKey(*control).impl(), nullptr).iterator->value;
372 if (!formState)
373 formState = makeUnique<SavedFormState>();
374 formState->appendControlState(control->name(), control->type(), control->saveFormControlState());
375 }
376 return stateMap;
377}
378
379Vector<String> FormController::formElementsState() const
380{
381 std::unique_ptr<SavedFormStateMap> stateMap = createSavedFormStateMap(m_formElementsWithState);
382 Vector<String> stateVector;
383 stateVector.reserveInitialCapacity(m_formElementsWithState.size() * 4);
384 stateVector.append(formStateSignature());
385 for (auto& state : *stateMap) {
386 stateVector.append(state.key.get());
387 state.value->serializeTo(stateVector);
388 }
389 bool hasOnlySignature = stateVector.size() == 1;
390 if (hasOnlySignature)
391 stateVector.clear();
392 return stateVector;
393}

399
400FormControlState FormController::takeStateForFormElement(const HTMLFormControlElementWithState& control)
401{
402 if (m_savedFormStateMap.isEmpty())
403 return FormControlState();
404 if (!m_formKeyGenerator)
405 m_formKeyGenerator = makeUnique<FormKeyGenerator>();
406 SavedFormStateMap::iterator it = m_savedFormStateMap.find(m_formKeyGenerator->formKey(control).impl());
407 if (it == m_savedFormStateMap.end())
408 return FormControlState();
409 FormControlState state = it->value->takeControlState(control.name(), control.type());
410 if (it->value->isEmpty())
411 m_savedFormStateMap.remove(it);
412 return state;
413}

415void FormController::formStatesFromStateVector(const Vector<String>& stateVector, SavedFormStateMap& map)
416{
417 map.clear();
418
419 size_t i = 0;
420 if (stateVector.size() < 1 || stateVector[i++] != formStateSignature())
421 return;
422
423 while (i + 1 < stateVector.size()) {
424 AtomString formKey = stateVector[i++];
425 auto state = SavedFormState::deserialize(stateVector, i);
426 if (!state) {
427 i = 0;
428 break;
429 }
430 map.add(formKey.impl(), WTFMove(state));
431 }
432 if (i != stateVector.size())
433 map.clear();
434}

442void FormController::restoreControlStateFor(HTMLFormControlElementWithState& control)
443{
444 // We don't save state of a control with shouldSaveAndRestoreFormControlState()
445 // == false. But we need to skip restoring process too because a control in
446 // another form might have the same pair of name and type and saved its state.
447 if (!control.shouldSaveAndRestoreFormControlState())
448 return;
449 if (ownerFormForState(control))
450 return;
451 auto state = takeStateForFormElement(control);
452 if (!state.isEmpty())
453 control.restoreFormControlState(state);
454}
455
456void FormController::restoreControlStateIn(HTMLFormElement& form)
457{
458 for (auto& element : form.copyAssociatedElementsVector()) {
459 if (!is<HTMLFormControlElementWithState>(element.get()))
460 continue;
461 auto& control = downcast<HTMLFormControlElementWithState>(element.get());
462 if (!control.shouldSaveAndRestoreFormControlState())
463 continue;
464 if (ownerFormForState(control) != &form)
465 continue;
466 auto state = takeStateForFormElement(control);
467 if (!state.isEmpty())
468 control.restoreFormControlState(state);
469 }
470}

477Vector<String> FormController::referencedFilePaths(const Vector<String>& stateVector)
478{
479 Vector<String> paths;
480 SavedFormStateMap map;
481 formStatesFromStateVector(stateVector, map);
482 for (auto& state : map.values())
483 paths.appendVector(state->referencedFilePaths());
484 return paths;
