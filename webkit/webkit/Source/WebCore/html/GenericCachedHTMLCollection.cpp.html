45template <CollectionTraversalType traversalType>
46bool GenericCachedHTMLCollection<traversalType>::elementMatches(Element& element) const
47{
48 switch (this->type()) {
49 case NodeChildren:
50 return true;
51 case DocImages:
52 return element.hasTagName(imgTag);
53 case DocScripts:
54 return element.hasTagName(scriptTag);
55 case DocForms:
56 return element.hasTagName(formTag);
57 case TableTBodies:
58 return element.hasTagName(tbodyTag);
59 case TRCells:
60 return element.hasTagName(tdTag) || element.hasTagName(thTag);
61 case TSectionRows:
62 return element.hasTagName(trTag);
63 case SelectedOptions:
64 return is<HTMLOptionElement>(element) && downcast<HTMLOptionElement>(element).selected();
65 case DataListOptions:
66 return is<HTMLOptionElement>(element);
67 case MapAreas:
68 return element.hasTagName(areaTag);
69 case DocApplets:
70 return is<HTMLObjectElement>(element) && downcast<HTMLObjectElement>(element).containsJavaApplet();
71 case DocEmbeds:
72 return element.hasTagName(embedTag);
73 case DocLinks:
74 return (element.hasTagName(aTag) || element.hasTagName(areaTag)) && element.hasAttributeWithoutSynchronization(hrefAttr);
75 case DocAnchors:
76 return element.hasTagName(aTag) && element.hasAttributeWithoutSynchronization(nameAttr);
77 case FieldSetElements:
78 return is<HTMLObjectElement>(element) || is<HTMLFormControlElement>(element);
79 case ByClass:
80 case ByTag:
81 case ByHTMLTag:
82 case AllDescendants:
83 case DocAll:
84 case DocumentAllNamedItems:
85 case DocumentNamedItems:
86 case FormControls:
87 case SelectOptions:
88 case TableRows:
89 case WindowNamedItems:
90 break;
91 }
92 // Remaining collection types have their own CachedHTMLCollection subclasses and are not using GenericCachedHTMLCollection.
93 ASSERT_NOT_REACHED();
94 return false;
95}
96template bool GenericCachedHTMLCollection<CollectionTraversalType::Descendants>::elementMatches(Element&) const;
97template bool GenericCachedHTMLCollection<CollectionTraversalType::ChildrenOnly>::elementMatches(Element&) const;
