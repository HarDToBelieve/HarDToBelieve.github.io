40static DialogEventSender& dialogCloseEventSender()
41{
42 static NeverDestroyed<DialogEventSender> sharedCloseEventSender(eventNames().closeEvent);
43 return sharedCloseEventSender;
44}

46HTMLDialogElement::HTMLDialogElement(const QualifiedName& tagName, Document& document)
47 : HTMLElement(tagName, document)
48{
49}

51HTMLDialogElement::~HTMLDialogElement()

53 dialogCloseEventSender().cancelEvent(*this);
54}

71void HTMLDialogElement::show()
72{
73 // If the element already has an open attribute, then return.
74 if (isOpen())
75 return;
76
77 setOpen(true);
78}
79
80ExceptionOr<void> HTMLDialogElement::showModal()
81{
82 // If subject already has an open attribute, then throw an "InvalidStateError" DOMException.
83 if (isOpen())
84 return Exception { InvalidStateError };
85
86 // If subject is not connected, then throw an "InvalidStateError" DOMException.
87 if (!isConnected())
88 return Exception { InvalidStateError };
89
90 setOpen(true);
91 m_isModal = true;
92
93 return { };
94}
95
96void HTMLDialogElement::close(const String& returnValue)
97{
98 if (!isOpen())
99 return;
100
101 setOpen(false);
102 if (!returnValue.isNull())
103 m_returnValue = returnValue;
104}
105
106void HTMLDialogElement::dispatchPendingEvent(DialogEventSender* eventSender)
107{
108 ASSERT_UNUSED(eventSender, eventSender == &dialogCloseEventSender());
109 dispatchEvent(Event::create(eventNames().closeEvent, Event::CanBubble::No, Event::IsCancelable::No));
110}
111
112void HTMLDialogElement::parseAttribute(const QualifiedName& name, const AtomString& value)
113{
114 if (name == HTMLNames::openAttr) {
115 bool oldValue = m_isOpen;
116 m_isOpen = !value.isNull();
117
118 // Emit close event
119 if (oldValue != m_isOpen && !m_isOpen) {
120 m_isModal = false;
121 dialogCloseEventSender().cancelEvent(*this);
122 dialogCloseEventSender().dispatchEventSoon(*this);
123 }
124 return;
125 }
126
127 HTMLElement::parseAttribute(name, value);
128}
129
130void HTMLDialogElement::setOpen(bool value)
131{
132 setAttributeWithoutSynchronization(HTMLNames::openAttr, value ? emptyAtom() : nullAtom());
133}
