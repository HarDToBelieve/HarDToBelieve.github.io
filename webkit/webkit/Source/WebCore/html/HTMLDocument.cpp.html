122// https://html.spec.whatwg.org/multipage/dom.html#dom-document-nameditem
123std::optional<Variant<RefPtr<WindowProxy>, RefPtr<Element>, RefPtr<HTMLCollection>>> HTMLDocument::namedItem(const AtomString& name)
124{
125 if (name.isNull() || !hasDocumentNamedItem(*name.impl()))
126 return std::nullopt;
127
128 if (UNLIKELY(documentNamedItemContainsMultipleElements(*name.impl()))) {
129 auto collection = documentNamedItems(name);
130 ASSERT(collection->length() > 1);
131 return Variant<RefPtr<WindowProxy>, RefPtr<Element>, RefPtr<HTMLCollection>> { RefPtr<HTMLCollection> { WTFMove(collection) } };
132 }
133
134 auto& element = *documentNamedItem(*name.impl());
135 if (UNLIKELY(is<HTMLIFrameElement>(element))) {
136 if (auto domWindow = makeRefPtr(downcast<HTMLIFrameElement>(element).contentWindow()))
137 return Variant<RefPtr<WindowProxy>, RefPtr<Element>, RefPtr<HTMLCollection>> { WTFMove(domWindow) };
138 }
139
140 return Variant<RefPtr<WindowProxy>, RefPtr<Element>, RefPtr<HTMLCollection>> { RefPtr<Element> { &element } };
141}

143Vector<AtomString> HTMLDocument::supportedPropertyNames() const
144{
145 if (Quirks::shouldOmitHTMLDocumentSupportedPropertyNames())
146 return { };
147
148 auto properties = m_documentNamedItem.keys();
149 // The specification says these should be sorted in document order but this would be expensive
150 // and other browser engines do not comply with this part of the specification. For now, just
151 // do an alphabetical sort to get consistent results.
152 std::sort(properties.begin(), properties.end(), WTF::codePointCompareLessThan);
153 return properties;
154}

177bool HTMLDocument::isCaseSensitiveAttribute(const QualifiedName& attributeName)
178{
179 static const auto caseInsensitiveAttributeSet = makeNeverDestroyed([] {
180 // This is the list of attributes in HTML 4.01 with values marked as "[CI]" or case-insensitive
181 // Mozilla treats all other values as case-sensitive, thus so do we.
182 static const QualifiedName* const names[] = {
183 &accept_charsetAttr.get(),
184 &acceptAttr.get(),
185 &alignAttr.get(),
186 &alinkAttr.get(),
187 &axisAttr.get(),
188 &bgcolorAttr.get(),
189 &charsetAttr.get(),
190 &checkedAttr.get(),
191 &clearAttr.get(),
192 &codetypeAttr.get(),
193 &colorAttr.get(),
194 &compactAttr.get(),
195 &declareAttr.get(),
196 &deferAttr.get(),
197 &dirAttr.get(),
198 &disabledAttr.get(),
199 &enctypeAttr.get(),
200 &faceAttr.get(),
201 &frameAttr.get(),
202 &hreflangAttr.get(),
203 &http_equivAttr.get(),
204 &langAttr.get(),
205 &languageAttr.get(),
206 &linkAttr.get(),
207 &mediaAttr.get(),
208 &methodAttr.get(),
209 &multipleAttr.get(),
210 &nohrefAttr.get(),
211 &noresizeAttr.get(),
212 &noshadeAttr.get(),
213 &nowrapAttr.get(),
214 &readonlyAttr.get(),
215 &relAttr.get(),
216 &revAttr.get(),
217 &rulesAttr.get(),
218 &scopeAttr.get(),
219 &scrollingAttr.get(),
220 &selectedAttr.get(),
221 &shapeAttr.get(),
222 &targetAttr.get(),
223 &textAttr.get(),
224 &typeAttr.get(),
225 &valignAttr.get(),
226 &valuetypeAttr.get(),
227 &vlinkAttr.get(),
228 };
229 MemoryCompactLookupOnlyRobinHoodHashSet<AtomString> set;
230 for (auto* name : names)
231 set.add(name->localName());
232 return set;
233 }());
234
235 bool isPossibleHTMLAttr = !attributeName.hasPrefix() && attributeName.namespaceURI().isNull();
236 return !isPossibleHTMLAttr || !caseInsensitiveAttributeSet.get().contains(attributeName.localName());
237}

239bool HTMLDocument::isFrameSet() const
240{
241 if (!documentElement())
242 return false;
243 return !!childrenOfType<HTMLFrameSetElement>(*documentElement()).first();
244}
