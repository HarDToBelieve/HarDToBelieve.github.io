196inline HTMLElement::HTMLElement(const QualifiedName& tagName, Document& document, ConstructionType type = CreateHTMLElement)
197 : StyledElement(tagName, document, type)
198{
199 ASSERT(tagName.localName().impl());
200}

202template<size_t tableSize> inline void HTMLElement::populateEventHandlerNameMap(EventHandlerNameMap& map, const QualifiedName* const (&table)[tableSize])
203{
204 populateEventHandlerNameMap(map, table, tableSize);
205}

207inline bool Node::hasTagName(const HTMLQualifiedName& name) const
208{
209 return is<HTMLElement>(*this) && downcast<HTMLElement>(*this).hasTagName(name);
210}
