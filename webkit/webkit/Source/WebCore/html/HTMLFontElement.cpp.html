55// http://www.whatwg.org/specs/web-apps/current-work/multipage/rendering.html#fonts-and-colors
56template <typename CharacterType>
57static bool parseFontSize(const CharacterType* characters, unsigned length, int& size)
58{
59
60 // Step 1
61 // Step 2
62 const CharacterType* position = characters;
63 const CharacterType* end = characters + length;
64
65 // Step 3
66 while (position < end) {
67 if (!isHTMLSpace(*position))
68 break;
69 ++position;
70 }
71
72 // Step 4
73 if (position == end)
74 return false;
75 ASSERT_WITH_SECURITY_IMPLICATION(position < end);
76
77 // Step 5
78 enum {
79 RelativePlus,
80 RelativeMinus,
81 Absolute
82 } mode;
83
84 switch (*position) {
85 case '+':
86 mode = RelativePlus;
87 ++position;
88 break;
89 case '-':
90 mode = RelativeMinus;
91 ++position;
92 break;
93 default:
94 mode = Absolute;
95 break;
96 }
97
98 // Step 6
99 StringBuilder digits;
100 digits.reserveCapacity(16);
101 while (position < end) {
102 if (!isASCIIDigit(*position))
103 break;
104 digits.append(*position++);
105 }
106
107 // Step 7
108 if (digits.isEmpty())
109 return false;
110
111 // Step 8
112 int value = parseInteger<int>(digits).value_or(0);
113
114 // Step 9
115 if (mode == RelativePlus)
116 value += 3;
117 else if (mode == RelativeMinus)
118 value = 3 - value;
119
120 // Step 10
121 if (value > 7)
122 value = 7;
123
124 // Step 11
125 if (value < 1)
126 value = 1;
127
128 size = value;
129 return true;
130}

132static bool parseFontSize(const String& input, int& size)
133{
134 if (input.isEmpty())
135 return false;
136
137 if (input.is8Bit())
138 return parseFontSize(input.characters8(), input.length(), size);
139
140 return parseFontSize(input.characters16(), input.length(), size);
141}

143bool HTMLFontElement::cssValueFromFontSizeNumber(const String& s, CSSValueID& size)
144{
145 int num = 0;
146 if (!parseFontSize(s, num))
147 return false;
148
149 switch (num) {
150 case 1:
151 // FIXME: The spec says that we're supposed to use CSSValueXxSmall here.
152 size = CSSValueXSmall;
153 break;
154 case 2:
155 size = CSSValueSmall;
156 break;
157 case 3:
158 size = CSSValueMedium;
159 break;
160 case 4:
161 size = CSSValueLarge;
162 break;
163 case 5:
164 size = CSSValueXLarge;
165 break;
166 case 6:
167 size = CSSValueXxLarge;
168 break;
169 case 7:
170 size = CSSValueWebkitXxxLarge;
171 break;
172 default:
173 ASSERT_NOT_REACHED();
174 }
175 return true;
176}

178bool HTMLFontElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
179{
180 if (name == sizeAttr || name == colorAttr || name == faceAttr)
181 return true;
182 return HTMLElement::hasPresentationalHintsForAttribute(name);
183}
184
185void HTMLFontElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
186{
187 if (name == sizeAttr) {
188 CSSValueID size = CSSValueInvalid;
189 if (cssValueFromFontSizeNumber(value, size))
190 addPropertyToPresentationalHintStyle(style, CSSPropertyFontSize, size);
191 } else if (name == colorAttr)
192 addHTMLColorToStyle(style, CSSPropertyColor, value);
193 else if (name == faceAttr) {
194 if (auto fontFaceValue = CSSValuePool::singleton().createFontFaceValue(value))
195 style.setProperty(CSSProperty(CSSPropertyFontFamily, WTFMove(fontFaceValue)));
196 } else
197 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
198}
