129bool HTMLFormControlElement::computeIsDisabledByFieldsetAncestor() const
130{
131 RefPtr<Element> previousAncestor;
132 for (RefPtr<Element> ancestor = parentElement(); ancestor; ancestor = ancestor->parentElement()) {
133 if (is<HTMLFieldSetElement>(*ancestor) && ancestor->hasAttributeWithoutSynchronization(disabledAttr)) {
134 HTMLFieldSetElement& fieldSetAncestor = downcast<HTMLFieldSetElement>(*ancestor);
135 bool isInFirstLegend = is<HTMLLegendElement>(previousAncestor) && previousAncestor == fieldSetAncestor.legend();
136 return !isInFirstLegend;
137 }
138 previousAncestor = ancestor;
139 }
140 return false;
141}

152void HTMLFormControlElement::parseAttribute(const QualifiedName& name, const AtomString& value)
153{
154 if (name == formAttr)
155 formAttributeChanged();
156 else if (name == disabledAttr) {
157 if (canBeActuallyDisabled()) {
158 bool oldDisabled = m_disabled;
159 m_disabled = !value.isNull();
160 if (oldDisabled != m_disabled)
161 disabledAttributeChanged();
162 }
163 } else if (name == readonlyAttr) {
164 bool wasReadOnly = m_isReadOnly;
165 m_isReadOnly = !value.isNull();
166 if (wasReadOnly != m_isReadOnly)
167 readOnlyStateChanged();
168 } else if (name == requiredAttr) {
169 bool wasRequired = m_isRequired;
170 m_isRequired = !value.isNull();
171 if (wasRequired != m_isRequired)
172 requiredStateChanged();
173 } else
174 HTMLElement::parseAttribute(name, value);
175}

204static bool shouldAutofocus(HTMLFormControlElement* element)
205{
206 if (!element->renderer())
207 return false;
208 if (!element->hasAttributeWithoutSynchronization(autofocusAttr))
209 return false;
210 if (!element->isConnected() || !element->document().renderView())
211 return false;
212 if (element->document().isSandboxed(SandboxAutomaticFeatures)) {
213 // FIXME: This message should be moved off the console once a solution to https://bugs.webkit.org/show_bug.cgi?id=103274 exists.
214 element->document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, "Blocked autofocusing on a form control because the form's frame is sandboxed and the 'allow-scripts' permission is not set."_s);
215 return false;
216 }
217
218 auto& document = element->document();
219 if (!document.frame()->isMainFrame() && !document.topDocument().securityOrigin().isSameOriginDomain(document.securityOrigin())) {
220 document.addConsoleMessage(MessageSource::Security, MessageLevel::Error, "Blocked autofocusing on a form control in a cross-origin subframe."_s);
221 return false;
222 }
223
224 if (element->hasAutofocused())
225 return false;
226
227 // FIXME: Should this set of hasTagName checks be replaced by a
228 // virtual member function?
229 if (is<HTMLInputElement>(*element))
230 return !downcast<HTMLInputElement>(*element).isInputTypeHidden();
231 if (element->hasTagName(selectTag))
232 return true;
233 if (element->hasTagName(keygenTag))
234 return true;
235 if (element->hasTagName(buttonTag))
236 return true;
237 if (is<HTMLTextAreaElement>(*element))
238 return true;
239
240 return false;
241}

243void HTMLFormControlElement::didAttachRenderers()
244{
245 // The call to updateFromElement() needs to go after the call through
246 // to the base class's attach() because that can sometimes do a close
247 // on the renderer.
248 if (renderer())
249 renderer()->updateFromElement();
250
251 if (shouldAutofocus(this)) {
252 setAutofocused();
253
254 RefPtr<HTMLFormControlElement> element = this;
255 auto frameView = makeRefPtr(document().view());
256 if (frameView && frameView->layoutContext().isInLayout()) {
257 frameView->queuePostLayoutCallback([element] {
258 element->focus({ SelectionRestorationMode::PlaceCaretAtStart });
259 });
260 } else {
261 Style::queuePostResolutionCallback([element] {
262 element->focus({ SelectionRestorationMode::PlaceCaretAtStart });
263 });
264 }
265 }
266}

292Node::InsertedIntoAncestorResult HTMLFormControlElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
293{
294 if (m_dataListAncestorState == NotInsideDataList)
295 m_dataListAncestorState = Unknown;
296
297 updateWillValidateAndValidity();
298 if (willValidate() && !isValidFormControlElement())
299 addInvalidElementToAncestorFromInsertionPoint(*this, &parentOfInsertedTree);
300 if (document().hasDisabledFieldsetElement())
301 setAncestorDisabled(computeIsDisabledByFieldsetAncestor());
302 HTMLElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
303 FormAssociatedElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
304 return InsertedIntoAncestorResult::NeedsPostInsertionCallback;
305}

312void HTMLFormControlElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
313{
314 bool wasMatchingInvalidPseudoClass = willValidate() && !isValidFormControlElement();
315
316 m_validationMessage = nullptr;
317 if (m_disabledByAncestorFieldset)
318 setAncestorDisabled(computeIsDisabledByFieldsetAncestor());
319
320 bool wasInsideDataList = false;
321 if (m_dataListAncestorState == InsideDataList) {
322 m_dataListAncestorState = Unknown;
323 wasInsideDataList = true;
324 }
325
326 HTMLElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
327 FormAssociatedElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
328
329 if (wasMatchingInvalidPseudoClass)
330 removeInvalidElementToAncestorFromInsertionPoint(*this, &oldParentOfRemovedTree);
331
332 if (wasInsideDataList)
333 updateWillValidateAndValidity();
334}

416bool HTMLFormControlElement::computeWillValidate() const
417{
418 if (m_dataListAncestorState == Unknown) {
419#if ENABLE(DATALIST_ELEMENT)
420 m_dataListAncestorState = (document().hasDataListElements() && ancestorsOfType<HTMLDataListElement>(*this).first()) ? InsideDataList : NotInsideDataList;

424 }
425 return m_dataListAncestorState == NotInsideDataList && !isDisabledOrReadOnly();

428bool HTMLFormControlElement::willValidate() const
429{
430 if (!m_willValidateInitialized || m_dataListAncestorState == Unknown) {
431 m_willValidateInitialized = true;
432 bool newWillValidate = computeWillValidate();
433 if (m_willValidate != newWillValidate)
434 m_willValidate = newWillValidate;
435 } else {
436 // If the following assertion fails, updateWillValidateAndValidity() is not
437 // called correctly when something which changes computeWillValidate() result
438 // is updated.
439 ASSERT(m_willValidate == computeWillValidate());
440 }
441 return m_willValidate;
442}
443
444void HTMLFormControlElement::updateWillValidateAndValidity()
445{
446 // We need to recalculate willValidate immediately because willValidate change can causes style change.
447 bool newWillValidate = computeWillValidate();
448 if (m_willValidateInitialized && m_willValidate == newWillValidate)
449 return;
450
451 bool wasValid = m_isValid;
452
453 m_willValidateInitialized = true;
454 m_willValidate = newWillValidate;
455
456 updateValidity();
457 invalidateStyleForSubtree();
458
459 if (!m_willValidate && !wasValid) {
460 removeInvalidElementToAncestorFromInsertionPoint(*this, parentNode());
461 if (RefPtr<HTMLFormElement> form = this->form())
462 form->removeInvalidAssociatedFormControlIfNeeded(*this);
463 }
464
465 if (!m_willValidate)
466 hideVisibleValidationMessage();
467}

469void HTMLFormControlElement::updateVisibleValidationMessage()

473 return;

478 m_validationMessage = makeUnique<ValidationMessage>(this);
479 m_validationMessage->updateValidationMessage(message);

488bool HTMLFormControlElement::checkValidity(Vector<RefPtr<HTMLFormControlElement>>* unhandledInvalidControls)
489{
490 if (!willValidate() || isValidFormControlElement())
491 return true;
492 // An event handler can deref this object.
493 Ref<HTMLFormControlElement> protectedThis(*this);
494 Ref<Document> originalDocument(document());
495 auto event = Event::create(eventNames().invalidEvent, Event::CanBubble::No, Event::IsCancelable::Yes);
496 dispatchEvent(event);
497 if (!event->defaultPrevented() && unhandledInvalidControls && isConnected() && originalDocument.ptr() == &document())
498 unhandledInvalidControls->append(this);
499 return false;
500}

512bool HTMLFormControlElement::reportValidity()
513{
514 Vector<RefPtr<HTMLFormControlElement>> elements;
515 if (checkValidity(&elements))
516 return true;
517
518 if (elements.isEmpty())
519 return false;
520
521 // Needs to update layout now because we'd like to call isFocusable(), which
522 // has !renderer()->needsLayout() assertion.
523 document().updateLayoutIgnorePendingStylesheets();
524
525 if (isConnected() && isFocusable()) {
526 focusAndShowValidationMessage();
527 return false;
528 }
529
530 if (document().frame()) {
531 auto message = makeString("An invalid form control with name='", name(), "' is not focusable.");
532 document().addConsoleMessage(MessageSource::Rendering, MessageLevel::Error, message);
533 }
534
535 return false;
536}

538void HTMLFormControlElement::focusAndShowValidationMessage()
539{
540 SetForScope<bool> isFocusingWithValidationMessageScope(m_isFocusingWithValidationMessage, true);
541
542 // Calling focus() will scroll the element into view.
543 focus();
544
545 // focus() will scroll the element into view and this scroll may happen asynchronously.
546 // Because scrolling the view hides the validation message, we need to show the validation
547 // message asynchronously as well.
548 callOnMainThread([this, protectedThis = makeRef(*this)] {
549 updateVisibleValidationMessage();
550 });
551}

578
579void HTMLFormControlElement::updateValidity()
580{
581 if (m_delayedUpdateValidityCount)
582 return;
583
584 bool willValidate = this->willValidate();
585 bool wasValid = m_isValid;
586
587 m_isValid = isValid();
588
589 if (willValidate && m_isValid != wasValid) {
590 // Update style for pseudo classes such as :valid :invalid.
591 invalidateStyleForSubtree();
592
593 if (!m_isValid) {
594 if (isConnected())
595 addInvalidElementToAncestorFromInsertionPoint(*this, parentNode());
596 if (HTMLFormElement* form = this->form())
597 form->registerInvalidAssociatedFormControl(*this);
598 } else {
599 if (isConnected())
600 removeInvalidElementToAncestorFromInsertionPoint(*this, parentNode());
601 if (HTMLFormElement* form = this->form())
602 form->removeInvalidAssociatedFormControlIfNeeded(*this);
603 }
604 }
605
606 // Updates only if this control already has a validtion message.
607 if (m_validationMessage && m_validationMessage->isVisible()) {
608 // Calls updateVisibleValidationMessage() even if m_isValid is not
609 // changed because a validation message can be chagned.
610 updateVisibleValidationMessage();
611 }
612}
