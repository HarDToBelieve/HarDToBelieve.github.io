33namespace WebCore {

35WTF_MAKE_ISO_ALLOCATED_IMPL(HTMLFormControlElementWithState);
36
37HTMLFormControlElementWithState::HTMLFormControlElementWithState(const QualifiedName& tagName, Document& document, HTMLFormElement* form)
38 : HTMLFormControlElement(tagName, document, form)
39{
40}
41
42HTMLFormControlElementWithState::~HTMLFormControlElementWithState() = default;
43
44Node::InsertedIntoAncestorResult HTMLFormControlElementWithState::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
45{
46 if (insertionType.connectedToDocument && !containingShadowRoot())
47 document().formController().registerFormElementWithState(*this);
48 return HTMLFormControlElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
49}
50
51void HTMLFormControlElementWithState::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
52{
53 if (removalType.disconnectedFromDocument && !containingShadowRoot() && !oldParentOfRemovedTree.containingShadowRoot())
54 document().formController().unregisterFormElementWithState(*this);
55 HTMLFormControlElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
56}
57
58bool HTMLFormControlElementWithState::shouldAutocomplete() const
59{
60 if (!form())
61 return true;
62 return form()->shouldAutocomplete();
63}
64
65bool HTMLFormControlElementWithState::shouldSaveAndRestoreFormControlState() const
66{
67 // We don't save/restore control state in a form with autocomplete=off.
68 return isConnected() && shouldAutocomplete();
69}
70
71FormControlState HTMLFormControlElementWithState::saveFormControlState() const
72{
73 return FormControlState();
74}
75
76void HTMLFormControlElementWithState::finishParsingChildren()
77{
78 HTMLFormControlElement::finishParsingChildren();
79 document().formController().restoreControlStateFor(*this);
80}
81
82bool HTMLFormControlElementWithState::isFormControlElementWithState() const
83{
84 return true;
85}
