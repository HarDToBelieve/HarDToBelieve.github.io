86HTMLFormElement::~HTMLFormElement()
87{
88 document().formController().willDeleteForm(*this);
89 if (!shouldAutocomplete())
90 document().unregisterForDocumentSuspensionCallbacks(*this);
91
92 m_defaultButton = nullptr;
93 for (auto& weakElement : m_associatedElements) {
94 auto element = makeRefPtr(weakElement.get());
95 ASSERT(element);
96 auto* associatedElement = element->asFormAssociatedElement();
97 ASSERT(associatedElement);
98 associatedElement->formWillBeDestroyed();
99 }
100 for (auto& imageElement : m_imageElements)
101 imageElement->m_form = nullptr;
102}

109bool HTMLFormElement::rendererIsNeeded(const RenderStyle& style)
110{
111 if (!m_wasDemoted)
112 return HTMLElement::rendererIsNeeded(style);
113
114 auto parent = parentNode();
115 auto parentRenderer = parent->renderer();
116
117 if (!parentRenderer)
118 return false;
119
120 // FIXME: Shouldn't we also check for table caption (see |formIsTablePart| below).
121 bool parentIsTableElementPart = (parentRenderer->isTable() && is<HTMLTableElement>(*parent))
122 || (parentRenderer->isTableRow() && parent->hasTagName(trTag))
123 || (parentRenderer->isTableSection() && parent->hasTagName(tbodyTag))
124 || (parentRenderer->isRenderTableCol() && parent->hasTagName(colTag))
125 || (parentRenderer->isTableCell() && parent->hasTagName(trTag));
126
127 if (!parentIsTableElementPart)
128 return true;
129
130 DisplayType display = style.display();
131 bool formIsTablePart = display == DisplayType::Table || display == DisplayType::InlineTable || display == DisplayType::TableRowGroup
132 || display == DisplayType::TableHeaderGroup || display == DisplayType::TableFooterGroup || display == DisplayType::TableRow
133 || display == DisplayType::TableColumnGroup || display == DisplayType::TableColumn || display == DisplayType::TableCell
134 || display == DisplayType::TableCaption;
135
136 return formIsTablePart;
137}

194void HTMLFormElement::submitImplicitly(Event& event, bool fromImplicitSubmissionTrigger)
195{
196 unsigned submissionTriggerCount = 0;
197 for (auto& formAssociatedElement : m_associatedElements) {
198 if (!is<HTMLFormControlElement>(*formAssociatedElement))
199 continue;
200 HTMLFormControlElement& formElement = downcast<HTMLFormControlElement>(*formAssociatedElement);
201 if (formElement.isSuccessfulSubmitButton()) {
202 if (formElement.renderer()) {
203 formElement.dispatchSimulatedClick(&event);
204 return;
205 }
206 } else if (formElement.canTriggerImplicitSubmission())
207 ++submissionTriggerCount;
208 }
209
210 if (!submissionTriggerCount)
211 return;
212
213 // Older iOS apps using WebViews expect the behavior of auto submitting multi-input forms.
214 if (fromImplicitSubmissionTrigger && (submissionTriggerCount == 1 || document().settings().allowMultiElementImplicitSubmission()))
215 submitIfPossible(&event);
216}

218bool HTMLFormElement::validateInteractively()
219{
220 for (auto& associatedElement : m_associatedElements) {
221 if (is<HTMLFormControlElement>(*associatedElement))
222 downcast<HTMLFormControlElement>(*associatedElement).hideVisibleValidationMessage();
223 }
224
225 Vector<RefPtr<HTMLFormControlElement>> unhandledInvalidControls;
226 if (!checkInvalidControlsAndCollectUnhandled(unhandledInvalidControls))
227 return true;
228 // Because the form has invalid controls, we abort the form submission and
229 // show a validation message on a focusable form control.
230
231 // Make sure layout is up-to-date in case we call isFocusable() (which
232 // has !renderer()->needsLayout() assertion).
233 ASSERT(!document().view() || !document().view()->needsLayout());
234
235 Ref<HTMLFormElement> protectedThis(*this);
236
237 // Focus on the first focusable control and show a validation message.
238 for (auto& control : unhandledInvalidControls) {
239 if (control->isConnected() && control->isFocusable()) {
240 control->focusAndShowValidationMessage();
241 break;
242 }
243 }
244
245 // Warn about all of unfocusable controls.
246 if (document().frame()) {
247 for (auto& control : unhandledInvalidControls) {
248 if (control->isConnected() && control->isFocusable())
249 continue;
250 auto message = makeString("An invalid form control with name='", control->name(), "' is not focusable.");
251 document().addConsoleMessage(MessageSource::Rendering, MessageLevel::Error, message);
252 }
253 }
254
255 return false;
256}

258void HTMLFormElement::submitIfPossible(Event* event, HTMLFormControlElement* submitter, FormSubmissionTrigger trigger)
259{
260 if (!isConnected())
261 return;
262
263 RefPtr<Frame> frame = document().frame();
264 if (m_isSubmittingOrPreparingForSubmission || !frame)
265 return;
266
267 m_isSubmittingOrPreparingForSubmission = true;
268 m_shouldSubmit = false;
269
270 bool shouldValidate = document().page() && document().page()->settings().interactiveFormValidationEnabled() && !noValidate();
271
272 if (shouldValidate) {
273 auto submitElement = makeRefPtr(submitter ? submitter : findSubmitter(event));
274 if (submitElement && submitElement->formNoValidate())
275 shouldValidate = false;
276 }
277
278 // Interactive validation must be done before dispatching the submit event.
279 if (shouldValidate && !validateInteractively()) {
280 m_isSubmittingOrPreparingForSubmission = false;
281 return;
282 }
283
284 auto targetFrame = frame->loader().findFrameForNavigation(effectiveTarget(event, submitter), &document());
285 if (!targetFrame)
286 targetFrame = frame.get();
287 auto formState = FormState::create(*this, textFieldValues(), document(), NotSubmittedByJavaScript);
288 targetFrame->loader().client().dispatchWillSendSubmitEvent(WTFMove(formState));
289
290 auto protectedThis = makeRef(*this);
291
292 auto submitEvent = Event::create(eventNames().submitEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes);
293 dispatchEvent(submitEvent);
294
295 // Event handling could have resulted in m_shouldSubmit becoming true as a side effect, too.
296 if (!submitEvent->defaultPrevented())
297 m_shouldSubmit = true;
298
299 m_isSubmittingOrPreparingForSubmission = false;
300
301 if (m_shouldSubmit)
302 submit(event, true, !submitter, trigger, submitter);
303}

315ExceptionOr<void> HTMLFormElement::requestSubmit(HTMLElement* submitter)
316{
317 // Update layout before processing form actions in case the style changes
318 // the form or button relationships.
319 document().updateLayoutIgnorePendingStylesheets();
320
321 RefPtr<HTMLFormControlElement> control;
322 if (submitter) {
323 // https://html.spec.whatwg.org/multipage/forms.html#dom-form-requestsubmit
324 if (!is<HTMLFormControlElement>(submitter))
325 return Exception { TypeError };
326 control = downcast<HTMLFormControlElement>(submitter);
327 if (!control->isSubmitButton())
328 return Exception { TypeError };
329 if (control->form() != this)
330 return Exception { NotFoundError };
331 }
332
333 submitIfPossible(nullptr, control.get(), SubmittedByJavaScript);
334 return { };
335}

337StringPairVector HTMLFormElement::textFieldValues() const
338{
339 StringPairVector result;
340 result.reserveInitialCapacity(m_associatedElements.size());
341 for (auto& weakElement : m_associatedElements) {
342 auto element = makeRefPtr(weakElement.get());
343 if (!is<HTMLInputElement>(element))
344 continue;
345 auto& input = downcast<HTMLInputElement>(*element);
346 if (!input.isTextField())
347 continue;
348 result.uncheckedAppend({ input.name().string(), input.value() });
349 }
350 return result;
351}

353RefPtr<HTMLFormControlElement> HTMLFormElement::findSubmitButton(HTMLFormControlElement* submitter, bool needButtonActivation)
354{
355 if (submitter)
356 return submitter;
357 if (!needButtonActivation)
358 return nullptr;
359 RefPtr<HTMLFormControlElement> firstSuccessfulSubmitButton;
360 for (auto& associatedElement : m_associatedElements) {
361 if (!is<HTMLFormControlElement>(*associatedElement))
362 continue;
363 auto& control = downcast<HTMLFormControlElement>(*associatedElement);
364 if (control.isActivatedSubmit())
365 return nullptr;
366 if (!firstSuccessfulSubmitButton && control.isSuccessfulSubmitButton())
367 firstSuccessfulSubmitButton = &control;
368 }
369 return firstSuccessfulSubmitButton;
370}

372void HTMLFormElement::submit(Event* event, bool activateSubmitButton, bool processingUserGesture, FormSubmissionTrigger trigger, HTMLFormControlElement* submitter)
373{
374 // The submitIfPossible function also does this check, but we need to do it here
375 // too, since there are some code paths that bypass that function.
376 if (!isConnected())
377 return;
378
379 RefPtr<FrameView> view = document().view();
380 RefPtr<Frame> frame = document().frame();
381 if (!view || !frame)
382 return;
383
384 if (m_isSubmittingOrPreparingForSubmission) {
385 m_shouldSubmit = true;
386 return;
387 }
388
389 m_isSubmittingOrPreparingForSubmission = true;
390 m_wasUserSubmitted = processingUserGesture;
391
392 auto firstSuccessfulSubmitButton = findSubmitButton(submitter, activateSubmitButton);
393 if (firstSuccessfulSubmitButton)
394 firstSuccessfulSubmitButton->setActivatedSubmit(true);
395
396 auto protectedThis = makeRef(*this); // Form submission can execute arbitary JavaScript.
397
398 auto shouldLockHistory = processingUserGesture ? LockHistory::No : LockHistory::Yes;
399 auto formSubmission = FormSubmission::create(*this, submitter, m_attributes, event, shouldLockHistory, trigger);
400 if (m_plannedFormSubmission)
401 m_plannedFormSubmission->cancel();
402
403 m_plannedFormSubmission = makeWeakPtr(formSubmission.get());
404
405 if (RuntimeEnabledFeatures::sharedFeatures().dialogElementEnabled() && formSubmission->method() == FormSubmission::Method::Dialog)
406 submitDialog(WTFMove(formSubmission));
407 else
408 frame->loader().submitForm(WTFMove(formSubmission));
409
410 if (firstSuccessfulSubmitButton)
411 firstSuccessfulSubmitButton->setActivatedSubmit(false);
412
413 m_shouldSubmit = false;
414 m_isSubmittingOrPreparingForSubmission = false;
415}

417// https://html.spec.whatwg.org/#submit-dialog
418void HTMLFormElement::submitDialog(Ref<FormSubmission>&& formSubmission)
419{
420 // Let subject be the nearest ancestor dialog element of form, if any.
421 RefPtr dialog = ancestorsOfType<HTMLDialogElement>(*this).first();
422
423 // If there isn't one, or if it does not have an open attribute, do nothing.
424 if (!dialog || !dialog->isOpen())
425 return;
426
427 // Then, close the dialog subject. If there is a result, let that be the return value.
428 dialog->close(formSubmission->returnValue());
429}

431void HTMLFormElement::reset()
432{
433 if (m_isInResetFunction)
434 return;
435
436 RefPtr<Frame> protectedFrame = document().frame();
437 if (!protectedFrame)
438 return;
439
440 Ref<HTMLFormElement> protectedThis(*this);
441
442 SetForScope<bool> isInResetFunctionRestorer(m_isInResetFunction, true);
443
444 auto event = Event::create(eventNames().resetEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes);
445 dispatchEvent(event);
446 if (!event->defaultPrevented())
447 resetAssociatedFormControlElements();
448}

450void HTMLFormElement::resetAssociatedFormControlElements()
451{
452 // Event handling can cause associated elements to be added or deleted while iterating
453 // over this collection. Protect these elements until we are done notifying them of
454 // the reset operation.
455 Vector<Ref<HTMLFormControlElement>> associatedFormControlElements;
456 associatedFormControlElements.reserveInitialCapacity(m_associatedElements.size());
457 for (auto& weakElement : m_associatedElements) {
458 auto* element = weakElement.get();
459 ASSERT(element);
460 if (is<HTMLFormControlElement>(element))
461 associatedFormControlElements.uncheckedAppend(downcast<HTMLFormControlElement>(*element));
462 }
463
464 for (auto& associatedFormControlElement : associatedFormControlElements)
465 associatedFormControlElement->reset();
466}

484void HTMLFormElement::parseAttribute(const QualifiedName& name, const AtomString& value)
485{
486 if (name == actionAttr) {
487 m_attributes.parseAction(value);
488
489 if (!m_attributes.action().isEmpty()) {
490 if (RefPtr<Frame> f = document().frame()) {
491 Frame& topFrame = f->tree().top();
492 MixedContentChecker::checkFormForMixedContent(topFrame, topFrame.document()->securityOrigin(), document().completeURL(m_attributes.action()));
493 }
494 }
495 } else if (name == targetAttr)
496 m_attributes.setTarget(value);
497 else if (name == methodAttr)
498 m_attributes.updateMethodType(value);
499 else if (name == enctypeAttr)
500 m_attributes.updateEncodingType(value);
501 else if (name == accept_charsetAttr)
502 m_attributes.setAcceptCharset(value);
503 else if (name == autocompleteAttr) {
504 if (!shouldAutocomplete())
505 document().registerForDocumentSuspensionCallbacks(*this);
506 else
507 document().unregisterForDocumentSuspensionCallbacks(*this);
508 } else
509 HTMLElement::parseAttribute(name, value);
510}

512unsigned HTMLFormElement::formElementIndexWithFormAttribute(Element* element, unsigned rangeStart, unsigned rangeEnd)
513{
514 if (m_associatedElements.isEmpty())
515 return 0;
516
517 ASSERT(rangeStart <= rangeEnd);
518
519 if (rangeStart == rangeEnd)
520 return rangeStart;
521
522 unsigned left = rangeStart;
523 unsigned right = rangeEnd - 1;
524 unsigned short position;
525
526 // Does binary search on m_associatedElements in order to find the index
527 // to be inserted.
528 while (left != right) {
529 unsigned middle = left + ((right - left) / 2);
530 ASSERT(middle < m_associatedElementsBeforeIndex || middle >= m_associatedElementsAfterIndex);
531 position = element->compareDocumentPosition(*m_associatedElements[middle]);
532 if (position & DOCUMENT_POSITION_FOLLOWING)
533 right = middle;
534 else
535 left = middle + 1;
536 }
537
538 ASSERT(left < m_associatedElementsBeforeIndex || left >= m_associatedElementsAfterIndex);
539 position = element->compareDocumentPosition(*m_associatedElements[left]);
540 if (position & DOCUMENT_POSITION_FOLLOWING)
541 return left;
542 return left + 1;
543}

545unsigned HTMLFormElement::formElementIndex(FormAssociatedElement* associatedElement)
546{
547 ASSERT(associatedElement);
548
549 HTMLElement& associatedHTMLElement = associatedElement->asHTMLElement();
550
551 // Treats separately the case where this element has the form attribute
552 // for performance consideration.
553 if (associatedHTMLElement.hasAttributeWithoutSynchronization(formAttr) && associatedHTMLElement.isConnected()) {
554 unsigned short position = compareDocumentPosition(associatedHTMLElement);
555 ASSERT(!(position & DOCUMENT_POSITION_DISCONNECTED));
556 if (position & DOCUMENT_POSITION_PRECEDING) {
557 ++m_associatedElementsBeforeIndex;
558 ++m_associatedElementsAfterIndex;
559 return HTMLFormElement::formElementIndexWithFormAttribute(&associatedHTMLElement, 0, m_associatedElementsBeforeIndex - 1);
560 }
561 if (position & DOCUMENT_POSITION_FOLLOWING && !(position & DOCUMENT_POSITION_CONTAINED_BY))
562 return HTMLFormElement::formElementIndexWithFormAttribute(&associatedHTMLElement, m_associatedElementsAfterIndex, m_associatedElements.size());
563 }
564
565 unsigned currentAssociatedElementsAfterIndex = m_associatedElementsAfterIndex;
566 ++m_associatedElementsAfterIndex;
567
568 if (!associatedHTMLElement.isDescendantOf(*this))
569 return currentAssociatedElementsAfterIndex;
570
571 auto descendants = descendantsOfType<HTMLElement>(*this);
572
573 // Check for the special case where this element is the very last thing in
574 // the form's tree of children; we don't want to walk the entire tree in that
575 // common case that occurs during parsing; instead we'll just return a value
576 // that says "add this form element to the end of the array".
577 if (!++descendants.beginAt(associatedHTMLElement))
578 return currentAssociatedElementsAfterIndex;
579
580 unsigned i = m_associatedElementsBeforeIndex;
581 for (auto& element : descendants) {
582 if (&element == &associatedHTMLElement)
583 return i;
584 if (!is<HTMLFormControlElement>(element) && !is<HTMLObjectElement>(element))
585 continue;
586 if (element.form() != this)
587 continue;
588 ++i;
589 }
590 return currentAssociatedElementsAfterIndex;
591}

593void HTMLFormElement::registerFormElement(FormAssociatedElement* e)
594{
595 m_associatedElements.insert(formElementIndex(e), makeWeakPtr(e->asHTMLElement()));
596
597 if (is<HTMLFormControlElement>(e)) {
598 HTMLFormControlElement& control = downcast<HTMLFormControlElement>(*e);
599 if (control.isSuccessfulSubmitButton()) {
600 if (!m_defaultButton)
601 control.invalidateStyleForSubtree();
602 else
603 resetDefaultButton();
604 }
605 }
606}

608void HTMLFormElement::removeFormElement(FormAssociatedElement* e)
609{
610 unsigned index = m_associatedElements.find(&e->asHTMLElement());
611 ASSERT(index < m_associatedElements.size());
612 if (index < m_associatedElementsBeforeIndex)
613 --m_associatedElementsBeforeIndex;
614 if (index < m_associatedElementsAfterIndex)
615 --m_associatedElementsAfterIndex;
616 removeFromPastNamesMap(e);
617 m_associatedElements.remove(index);
618
619 if (auto* nodeLists = this->nodeLists())
620 nodeLists->invalidateCaches();
621
622 if (e == m_defaultButton)
623 resetDefaultButton();
624}

715String HTMLFormElement::effectiveTarget(const Event* event, HTMLFormControlElement* overrideSubmitter) const
716{
717 if (auto submitter = makeRefPtr(overrideSubmitter ? overrideSubmitter : findSubmitter(event))) {
718 auto targetValue = submitter->attributeWithoutSynchronization(formtargetAttr);
719 if (!targetValue.isNull())
720 return targetValue;
721 }
722
723 auto targetValue = target();
724 if (!targetValue.isNull())
725 return targetValue;
726

744HTMLFormControlElement* HTMLFormElement::defaultButton() const
745{
746 if (m_defaultButton)
747 return m_defaultButton.get();
748 for (auto& associatedElement : m_associatedElements) {

750 continue;

760void HTMLFormElement::resetDefaultButton()
761{
762 if (!m_defaultButton) {
763 // Computing the default button is not cheap, we don't want to do it unless needed.
764 // If there was no default button set, the only style to invalidate is the element
765 // being added to the form. This is done explicitly in registerFormElement().
766 return;
767 }
768
769 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
770
771 auto oldDefault = WTFMove(m_defaultButton);
772 defaultButton();
773 if (m_defaultButton != oldDefault) {
774 if (oldDefault)
775 oldDefault->invalidateStyleForSubtree();
776 if (m_defaultButton)
777 m_defaultButton->invalidateStyleForSubtree();
778 }
779}

787bool HTMLFormElement::checkInvalidControlsAndCollectUnhandled(Vector<RefPtr<HTMLFormControlElement>>& unhandledInvalidControls)
788{
789 Ref<HTMLFormElement> protectedThis(*this);
790 // Copy m_associatedElements because event handlers called from
791 // HTMLFormControlElement::checkValidity() might change m_associatedElements.
792 auto elements = copyAssociatedElementsVector();
793 bool hasInvalidControls = false;
794 for (auto& element : elements) {
795 if (element->form() == this && is<HTMLFormControlElement>(element.get())) {
796 HTMLFormControlElement& control = downcast<HTMLFormControlElement>(element.get());
797 if (!control.checkValidity(&unhandledInvalidControls) && control.form() == this)
798 hasInvalidControls = true;
799 }
800 }
801 return hasInvalidControls;
802}

833{

878Vector<Ref<Element>> HTMLFormElement::namedElements(const AtomString& name)
879{
880 // http://www.whatwg.org/specs/web-apps/current-work/multipage/forms.html#dom-form-nameditem
881 Vector<Ref<Element>> namedItems = elements()->namedItems(name);
882
883 auto elementFromPast = elementFromPastNamesMap(name);
884 if (namedItems.size() == 1 && namedItems.first().ptr() != elementFromPast)
885 addToPastNamesMap(downcast<HTMLElement>(namedItems.first().get()).asFormNamedItem(), name);

887 namedItems.append(*elementFromPast);
888
