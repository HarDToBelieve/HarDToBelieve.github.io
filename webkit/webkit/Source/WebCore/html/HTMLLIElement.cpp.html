60bool HTMLLIElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
61{
62 if (name == typeAttr)
63 return true;
64 return HTMLElement::hasPresentationalHintsForAttribute(name);
65}

67void HTMLLIElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
68{
69 if (name == typeAttr) {
70 if (value == "a")
71 addPropertyToPresentationalHintStyle(style, CSSPropertyListStyleType, CSSValueLowerAlpha);
72 else if (value == "A")
73 addPropertyToPresentationalHintStyle(style, CSSPropertyListStyleType, CSSValueUpperAlpha);
74 else if (value == "i")
75 addPropertyToPresentationalHintStyle(style, CSSPropertyListStyleType, CSSValueLowerRoman);
76 else if (value == "I")
77 addPropertyToPresentationalHintStyle(style, CSSPropertyListStyleType, CSSValueUpperRoman);
78 else if (value == "1")
79 addPropertyToPresentationalHintStyle(style, CSSPropertyListStyleType, CSSValueDecimal);
80 else
81 addPropertyToPresentationalHintStyle(style, CSSPropertyListStyleType, value);
82 } else
83 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
84}

86void HTMLLIElement::parseAttribute(const QualifiedName& name, const AtomString& value)
87{
88 if (name == valueAttr) {
89 if (renderer() && renderer()->isListItem())
90 parseValue(value);
91 } else
92 HTMLElement::parseAttribute(name, value);
93}

95void HTMLLIElement::didAttachRenderers()
96{
97 if (!is<RenderListItem>(renderer()))
98 return;
99 auto& listItemRenderer = downcast<RenderListItem>(*renderer());
100
101 // Check if there is an enclosing list.
102 bool isInList = false;
103 for (auto& ancestor : ancestorsOfType<HTMLElement>(*this)) {
104 if (is<HTMLUListElement>(ancestor) || is<HTMLOListElement>(ancestor)) {
105 isInList = true;
106 break;
107 }
108 }
109
110 // If we are not in a list, tell the renderer so it can position us inside.
111 // We don't want to change our style to say "inside" since that would affect nested nodes.
112 if (!isInList)
113 listItemRenderer.setNotInList(true);
114
115 parseValue(attributeWithoutSynchronization(valueAttr));
116}
117
118inline void HTMLLIElement::parseValue(const AtomString& value)
119{
120 ASSERT(renderer());
121 std::optional<int> explicitValue;
122 if (auto parsedValue = parseHTMLInteger(value))
123 explicitValue = *parsedValue;
124 downcast<RenderListItem>(*renderer()).setExplicitValue(explicitValue);
125}
