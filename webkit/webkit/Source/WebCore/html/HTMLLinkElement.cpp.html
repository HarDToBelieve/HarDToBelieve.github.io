121void HTMLLinkElement::setDisabledState(bool disabled)
122{
123 DisabledState oldDisabledState = m_disabledState;
124 m_disabledState = disabled ? Disabled : EnabledViaScript;
125 if (oldDisabledState == m_disabledState)
126 return;
127
128 ASSERT(isConnected() || !styleSheetIsLoading());
129 if (!isConnected())
130 return;
131
132 // If we change the disabled state while the sheet is still loading, then we have to
133 // perform three checks:
134 if (styleSheetIsLoading()) {
135 // Check #1: The sheet becomes disabled while loading.
136 if (m_disabledState == Disabled)
137 removePendingSheet();
138
139 // Check #2: An alternate sheet becomes enabled while it is still loading.
140 if (m_relAttribute.isAlternate && m_disabledState == EnabledViaScript)
141 addPendingSheet(ActiveSheet);
142
143 // Check #3: A main sheet becomes enabled while it was still loading and
144 // after it was disabled via script. It takes really terrible code to make this
145 // happen (a double toggle for no reason essentially). This happens on
146 // virtualplastic.net, which manages to do about 12 enable/disables on only 3
147 // sheets. :)
148 if (!m_relAttribute.isAlternate && m_disabledState == EnabledViaScript && oldDisabledState == Disabled)
149 addPendingSheet(ActiveSheet);
150
151 // If the sheet is already loading just bail.
152 return;
153 }
154
155 // Load the sheet, since it's never been loaded before.
156 if (!m_sheet && m_disabledState == EnabledViaScript)
157 process();
158 else {
159 ASSERT(m_styleScope);
160 m_styleScope->didChangeActiveStyleSheetCandidates();
161 }
162}

164void HTMLLinkElement::parseAttribute(const QualifiedName& name, const AtomString& value)
165{
166 if (name == relAttr) {
167 m_relAttribute = LinkRelAttribute(document(), value);
168 if (m_relList)
169 m_relList->associatedAttributeValueChanged(value);
170 process();
171 return;
172 }
173 if (name == hrefAttr) {
174 bool wasLink = isLink();
175 setIsLink(!value.isNull() && !shouldProhibitLinks(this));
176 if (wasLink != isLink())
177 invalidateStyleForSubtree();
178 process();
179 return;
180 }
181 if (name == typeAttr) {
182 m_type = value;
183 process();
184 return;
185 }
186 if (name == sizesAttr) {
187 if (m_sizes)
188 m_sizes->associatedAttributeValueChanged(value);
189 process();
190 return;
191 }
192 if (name == mediaAttr) {
193 m_media = value.string().convertToASCIILowercase();
194 process();
195 if (m_sheet && !isDisabled())
196 m_styleScope->didChangeActiveStyleSheetCandidates();
197 return;
198 }
199 if (name == disabledAttr) {
200 setDisabledState(!value.isNull());
201 return;
202 }
203 if (name == titleAttr) {
204 if (m_sheet && !isInShadowTree())
205 m_sheet->setTitle(value);
206 return;
207 }
208 HTMLElement::parseAttribute(name, value);
209}

211bool HTMLLinkElement::shouldLoadLink()
212{
213 Ref<Document> originalDocument = document();
214 if (!dispatchBeforeLoadEvent(getNonEmptyURLAttribute(hrefAttr).string()))
215 return false;
216 // A beforeload handler might have removed us from the document or changed the document.
217 if (!isConnected() || &document() != originalDocument.ptr())
218 return false;
219 return true;
220}

237String HTMLLinkElement::as() const
238{
239 String as = attributeWithoutSynchronization(asAttr);
240 if (equalLettersIgnoringASCIICase(as, "fetch")
241 || equalLettersIgnoringASCIICase(as, "image")
242 || equalLettersIgnoringASCIICase(as, "script")
243 || equalLettersIgnoringASCIICase(as, "style")
244 || (document().settings().mediaPreloadingEnabled() && (equalLettersIgnoringASCIICase(as, "video") || equalLettersIgnoringASCIICase(as, "audio")))
245 || equalLettersIgnoringASCIICase(as, "track")
246 || equalLettersIgnoringASCIICase(as, "font"))
247 return as.convertToASCIILowercase();
248 return String();
249}

251void HTMLLinkElement::process()
252{
253 if (!isConnected()) {
254 ASSERT(!m_sheet);
255 return;
256 }
257
258 // Prevent recursive loading of link.
259 if (m_isHandlingBeforeLoad)
260 return;
261
262 URL url = getNonEmptyURLAttribute(hrefAttr);
263
264 LinkLoadParameters params {
265 m_relAttribute,
266 url,
267 attributeWithoutSynchronization(asAttr),
268 attributeWithoutSynchronization(mediaAttr),
269 attributeWithoutSynchronization(typeAttr),
270 attributeWithoutSynchronization(crossoriginAttr),
271 attributeWithoutSynchronization(imagesrcsetAttr),
272 attributeWithoutSynchronization(imagesizesAttr),
273 referrerPolicy(),
274 };
275
276 m_linkLoader.loadLink(params, document());
277
278 bool treatAsStyleSheet = false;
279 if (m_relAttribute.isStyleSheet) {
280 if (m_type.isNull())
281 treatAsStyleSheet = true;
282 else if (auto parsedContentType = ParsedContentType::create(m_type))
283 treatAsStyleSheet = equalLettersIgnoringASCIICase(parsedContentType->mimeType(), "text/css");
284 }
285 if (!treatAsStyleSheet)
286 treatAsStyleSheet = document().settings().treatsAnyTextCSSLinkAsStylesheet() && m_type.containsIgnoringASCIICase("text/css");
287
288 if (m_disabledState != Disabled && treatAsStyleSheet && document().frame() && url.isValid()) {
289 String charset = attributeWithoutSynchronization(charsetAttr);
290 if (!TextEncoding { charset }.isValid())
291 charset = document().charset();
292
293 if (m_cachedSheet) {
294 removePendingSheet();
295 m_cachedSheet->removeClient(*this);
296 m_cachedSheet = nullptr;
297 }
298
299 {
300 bool previous = m_isHandlingBeforeLoad;
301 m_isHandlingBeforeLoad = true;
302 auto scopeExit = makeScopeExit([&] { m_isHandlingBeforeLoad = previous; });
303 if (!shouldLoadLink())
304 return;
305 }
306
307 m_loading = true;
308
309 bool mediaQueryMatches = true;
310 if (!m_media.isEmpty()) {
311 std::optional<RenderStyle> documentStyle;
312 if (document().hasLivingRenderTree())
313 documentStyle = Style::resolveForDocument(document());
314 auto media = MediaQuerySet::create(m_media, MediaQueryParserContext(document()));
315 LOG(MediaQueries, "HTMLLinkElement::process evaluating queries");
316 mediaQueryMatches = MediaQueryEvaluator { document().frame()->view()->mediaType(), document(), documentStyle ? &*documentStyle : nullptr }.evaluate(media.get());
317 }
318
319 // Don't hold up render tree construction and script execution on stylesheets
320 // that are not needed for the rendering at the moment.
321 bool isActive = mediaQueryMatches && !isAlternate();
322 addPendingSheet(isActive ? ActiveSheet : InactiveSheet);
323
324 // Load stylesheets that are not needed for the rendering immediately with low priority.
325 std::optional<ResourceLoadPriority> priority;
326 if (!isActive)
327 priority = DefaultResourceLoadPriority::inactiveStyleSheet;
328
329 if (document().settings().subresourceIntegrityEnabled())
330 m_integrityMetadataForPendingSheetRequest = attributeWithoutSynchronization(HTMLNames::integrityAttr);
331
332 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
333 options.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
334 if (document().contentSecurityPolicy()->allowStyleWithNonce(attributeWithoutSynchronization(HTMLNames::nonceAttr)))
335 options.contentSecurityPolicyImposition = ContentSecurityPolicyImposition::SkipPolicyCheck;
336 options.integrity = m_integrityMetadataForPendingSheetRequest;
337 options.referrerPolicy = params.referrerPolicy;
338
339 auto request = createPotentialAccessControlRequest(WTFMove(url), WTFMove(options), document(), crossOrigin());
340 request.setPriority(WTFMove(priority));
341 request.setCharset(WTFMove(charset));
342 request.setInitiator(*this);
343
344 ASSERT_WITH_SECURITY_IMPLICATION(!m_cachedSheet);
345 m_cachedSheet = document().cachedResourceLoader().requestCSSStyleSheet(WTFMove(request)).value_or(nullptr);
346
347 if (m_cachedSheet)
348 m_cachedSheet->addClient(*this);
349 else {
350 // The request may have been denied if (for example) the stylesheet is local and the document is remote.
351 m_loading = false;
352 sheetLoaded();
353 notifyLoadedSheetAndAllCriticalSubresources(true);
354 }
355 } else if (m_sheet) {
356 // we no longer contain a stylesheet, e.g. perhaps rel or type was changed
357 clearSheet();
358 m_styleScope->didChangeActiveStyleSheetCandidates();
359 }
360}

387void HTMLLinkElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
388{
389 HTMLElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
390 if (!removalType.disconnectedFromDocument)
391 return;
392
393 m_linkLoader.cancelLoad();
394
395 bool wasLoading = styleSheetIsLoading();
396
397 if (m_sheet)
398 clearSheet();
399
400 if (wasLoading)
401 removePendingSheet();
402
403 if (m_styleScope) {
404 m_styleScope->removeStyleSheetCandidateNode(*this);
405 m_styleScope = nullptr;
406 }

415void HTMLLinkElement::initializeStyleSheet(Ref<StyleSheetContents>&& styleSheet, const CachedCSSStyleSheet& cachedStyleSheet, MediaQueryParserContext context)
416{
417 // FIXME: originClean should be turned to false except if fetch mode is CORS.
418 std::optional<bool> originClean;
419 if (cachedStyleSheet.options().mode == FetchOptions::Mode::Cors)
420 originClean = cachedStyleSheet.isCORSSameOrigin();
421
422 m_sheet = CSSStyleSheet::create(WTFMove(styleSheet), *this, originClean);
423 m_sheet->setMediaQueries(MediaQuerySet::create(m_media, context));
424 if (!isInShadowTree())
425 m_sheet->setTitle(title());
426
427 if (!m_sheet->canAccessRules())
428 m_sheet->contents().setAsOpaque();
429}

431void HTMLLinkElement::setCSSStyleSheet(const String& href, const URL& baseURL, const String& charset, const CachedCSSStyleSheet* cachedStyleSheet)
432{
433 if (!isConnected()) {
434 ASSERT(!m_sheet);
435 return;
436 }
437 auto frame = makeRefPtr(document().frame());
438 if (!frame)
439 return;
440
441 // Completing the sheet load may cause scripts to execute.
442 Ref<HTMLLinkElement> protectedThis(*this);
443
444 if (!cachedStyleSheet->errorOccurred() && !matchIntegrityMetadata(*cachedStyleSheet, m_integrityMetadataForPendingSheetRequest)) {
445 document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, makeString("Cannot load stylesheet ", integrityMismatchDescription(*cachedStyleSheet, m_integrityMetadataForPendingSheetRequest)));
446
447 m_loading = false;
448 sheetLoaded();
449 notifyLoadedSheetAndAllCriticalSubresources(true);
450 return;
451 }
452
453 CSSParserContext parserContext(document(), baseURL, charset);
454 auto cachePolicy = frame->loader().subresourceCachePolicy(baseURL);
455
456 if (auto restoredSheet = const_cast<CachedCSSStyleSheet*>(cachedStyleSheet)->restoreParsedStyleSheet(parserContext, cachePolicy, frame->loader())) {
457 ASSERT(restoredSheet->isCacheable());
458 ASSERT(!restoredSheet->isLoading());
459 initializeStyleSheet(restoredSheet.releaseNonNull(), *cachedStyleSheet, MediaQueryParserContext(document()));
460
461 m_loading = false;
462 sheetLoaded();
463 notifyLoadedSheetAndAllCriticalSubresources(false);
464 return;
465 }
466
467 auto styleSheet = StyleSheetContents::create(href, parserContext);
468 initializeStyleSheet(styleSheet.copyRef(), *cachedStyleSheet, MediaQueryParserContext(document()));
469
470 // FIXME: Set the visibility option based on m_sheet being clean or not.
471 // Best approach might be to set it on the style sheet content itself or its context parser otherwise.
472 if (!styleSheet.get().parseAuthorStyleSheet(cachedStyleSheet, &document().securityOrigin())) {
473 m_loading = false;
474 sheetLoaded();
475 notifyLoadedSheetAndAllCriticalSubresources(true);
476 return;
477 }
478
479 m_loading = false;
480 styleSheet.get().notifyLoadedSheet(cachedStyleSheet);
481 styleSheet.get().checkLoaded();
482
483 if (styleSheet.get().isCacheable())
484 const_cast<CachedCSSStyleSheet*>(cachedStyleSheet)->saveParsedStyleSheet(WTFMove(styleSheet));
485}

615void HTMLLinkElement::addSubresourceAttributeURLs(ListHashSet<URL>& urls) const
616{
617 HTMLElement::addSubresourceAttributeURLs(urls);
618
619 // Favicons are handled by a special case in LegacyWebArchive::create()
620 if (m_relAttribute.iconType)
621 return;
622
623 if (!m_relAttribute.isStyleSheet)
624 return;
625
626 // Append the URL of this link element.
627 addSubresourceURL(urls, href());
628
629 if (auto styleSheet = makeRefPtr(this->sheet())) {
630 styleSheet->contents().traverseSubresources([&] (auto& resource) {
631 urls.add(resource.url());
632 return false;
633 });
634 }
635}

637void HTMLLinkElement::addPendingSheet(PendingSheetType type)
638{
639 if (type <= m_pendingSheetType)
640 return;
641 m_pendingSheetType = type;
642
643 if (m_pendingSheetType == InactiveSheet)
644 return;
645 ASSERT(m_styleScope);
646 m_styleScope->addPendingSheet(*this);
647}
648
649void HTMLLinkElement::removePendingSheet()
650{
651 PendingSheetType type = m_pendingSheetType;
652 m_pendingSheetType = Unknown;
653
654 if (type == Unknown)
655 return;
656
657 ASSERT(m_styleScope);
658 if (type == InactiveSheet) {
659 // Document just needs to know about the sheet for exposure through document.styleSheets
660 m_styleScope->didChangeActiveStyleSheetCandidates();
661 return;
662 }
663
664 m_styleScope->removePendingSheet(*this);
665}
