64bool HTMLMetaElement::mediaAttributeMatches()
65{
66 auto& document = this->document();
67
68 if (!m_media)
69 m_media = MediaQuerySet::create(attributeWithoutSynchronization(mediaAttr).convertToASCIILowercase(), MediaQueryParserContext(document));
70
71 std::optional<RenderStyle> documentStyle;
72 if (document.hasLivingRenderTree())
73 documentStyle = Style::resolveForDocument(document);
74
75 String mediaType;
76 if (auto* frame = document.frame()) {
77 if (auto* frameView = frame->view())
78 mediaType = frameView->mediaType();
79 }
80
81 return MediaQueryEvaluator(mediaType, document, documentStyle ? &*documentStyle : nullptr).evaluate(*m_media);
82}

91void HTMLMetaElement::attributeChanged(const QualifiedName& name, const AtomString& oldValue, const AtomString& newValue, AttributeModificationReason reason)
92{
93 HTMLElement::attributeChanged(name, oldValue, newValue, reason);
94
95 if (!isConnected())
96 return;
97
98 if (name == nameAttr) {
99 if (equalLettersIgnoringASCIICase(oldValue, "theme-color") && !equalLettersIgnoringASCIICase(newValue, "theme-color"))
100 document().metaElementThemeColorChanged(*this);
101 return;
102 }
103}

105void HTMLMetaElement::parseAttribute(const QualifiedName& name, const AtomString& value)
106{
107 if (name == nameAttr) {
108 process();
109 return;
110 }
111
112 if (name == contentAttr) {
113 m_contentColor = std::nullopt;
114 process();
115 return;
116 }
117
118 if (name == http_equivAttr) {
119 process();
120 return;
121 }
122
123 if (name == mediaAttr) {
124 m_media = nullptr;
125 process();
126 return;
127 }
128
129 HTMLElement::parseAttribute(name, value);
130}

153void HTMLMetaElement::process()
154{
155 // Changing a meta tag while it's not in the tree shouldn't have any effect on the document.
156 if (!isConnected())
157 return;
158
159 const AtomString& contentValue = attributeWithoutSynchronization(contentAttr);
160 if (contentValue.isNull())
161 return;
162
163 if (equalLettersIgnoringASCIICase(name(), "viewport"))
164 document().processViewport(contentValue, ViewportArguments::ViewportMeta);
165 else if (document().settings().disabledAdaptationsMetaTagEnabled() && equalLettersIgnoringASCIICase(name(), "disabled-adaptations"))
166 document().processDisabledAdaptations(contentValue);
167#if ENABLE(DARK_MODE_CSS)
168 else if (equalLettersIgnoringASCIICase(name(), "color-scheme") || equalLettersIgnoringASCIICase(name(), "supported-color-schemes"))
169 document().processColorScheme(contentValue);
170#endif
171 else if (equalLettersIgnoringASCIICase(name(), "theme-color"))
172 document().metaElementThemeColorChanged(*this);
173#if PLATFORM(IOS_FAMILY)
174 else if (equalLettersIgnoringASCIICase(name(), "format-detection"))
175 document().processFormatDetection(contentValue);
176 else if (equalLettersIgnoringASCIICase(name(), "apple-mobile-web-app-orientations"))
177 document().processWebAppOrientations();
178#endif
179 else if (equalLettersIgnoringASCIICase(name(), "referrer"))
180 document().processReferrerPolicy(contentValue, ReferrerPolicySource::MetaTag);
181
182 const AtomString& httpEquivValue = attributeWithoutSynchronization(http_equivAttr);
183 if (!httpEquivValue.isNull())
184 document().processHttpEquiv(httpEquivValue, contentValue, isDescendantOf(document().head()));
185}
