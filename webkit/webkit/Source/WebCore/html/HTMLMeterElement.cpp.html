72void HTMLMeterElement::parseAttribute(const QualifiedName& name, const AtomString& value)
73{
74 if (name == valueAttr || name == minAttr || name == maxAttr || name == lowAttr || name == highAttr || name == optimumAttr)

76 else
77 LabelableElement::parseAttribute(name, value);
78}

144HTMLMeterElement::GaugeRegion HTMLMeterElement::gaugeRegion() const
145{
146 double lowValue = low();
147 double highValue = high();
148 double theValue = value();
149 double optimumValue = optimum();
150
151 if (optimumValue < lowValue) {
152 // The optimum range stays under low
153 if (theValue <= lowValue)
154 return GaugeRegionOptimum;
155 if (theValue <= highValue)
156 return GaugeRegionSuboptimal;
157 return GaugeRegionEvenLessGood;
158 }
159
160 if (highValue < optimumValue) {
161 // The optimum range stays over high
162 if (highValue <= theValue)
163 return GaugeRegionOptimum;
164 if (lowValue <= theValue)
165 return GaugeRegionSuboptimal;
166 return GaugeRegionEvenLessGood;
167 }
168
169 // The optimum range stays between high and low.
170 // According to the standard, <meter> never show GaugeRegionEvenLessGood in this case
171 // because the value is never less or greater than min or max.
172 if (lowValue <= theValue && theValue <= highValue)
173 return GaugeRegionOptimum;
174 return GaugeRegionSuboptimal;
175}

177double HTMLMeterElement::valueRatio() const
178{
179 double min = this->min();
180 double max = this->max();
181 double value = this->value();
182
183 if (max <= min)
184 return 0;
185 return (value - min) / (max - min);
186}

188static void setValueClass(HTMLElement& element, HTMLMeterElement::GaugeRegion gaugeRegion)
189{
190 switch (gaugeRegion) {
191 case HTMLMeterElement::GaugeRegionOptimum:
192 element.setAttribute(HTMLNames::classAttr, "optimum");
193 element.setPseudo("-webkit-meter-optimum-value");
194 return;
195 case HTMLMeterElement::GaugeRegionSuboptimal:
196 element.setAttribute(HTMLNames::classAttr, "suboptimum");
197 element.setPseudo("-webkit-meter-suboptimum-value");
198 return;
199 case HTMLMeterElement::GaugeRegionEvenLessGood:
200 element.setAttribute(HTMLNames::classAttr, "even-less-good");
201 element.setPseudo("-webkit-meter-even-less-good-value");
202 return;
203 default:
204 ASSERT_NOT_REACHED();
205 }
206}

208void HTMLMeterElement::didElementStateChange()
209{
210 m_value->setInlineStyleProperty(CSSPropertyWidth, valueRatio()*100, CSSUnitType::CSS_PERCENTAGE);
211 setValueClass(*m_value, gaugeRegion());
212
213 if (RenderMeter* render = renderMeter())
214 render->updateFromElement();
215}

217RenderMeter* HTMLMeterElement::renderMeter() const
218{
219 if (is<RenderMeter>(renderer()))
220 return downcast<RenderMeter>(renderer());
221 return nullptr;
222}

224void HTMLMeterElement::didAddUserAgentShadowRoot(ShadowRoot& root)
225{
226 ASSERT(!m_value);
227
228 static MainThreadNeverDestroyed<const String> shadowStyle(StringImpl::createWithoutCopying(meterElementShadowUserAgentStyleSheet, sizeof(meterElementShadowUserAgentStyleSheet)));
229
230 auto style = HTMLStyleElement::create(HTMLNames::styleTag, document(), false);
231 style->setTextContent(shadowStyle);
232 root.appendChild(style);
233
234 // Pseudos are set to allow author styling.
235 auto inner = HTMLDivElement::create(document());
236 inner->setIdAttribute("inner");
237 inner->setPseudo("-webkit-meter-inner-element");
238 root.appendChild(inner);
239
240 auto bar = HTMLDivElement::create(document());
241 bar->setIdAttribute("bar");
242 bar->setPseudo("-webkit-meter-bar");
243 inner->appendChild(bar);
244
245 m_value = HTMLDivElement::create(document());
246 m_value->setIdAttribute("value");
247 bar->appendChild(*m_value);
248
249 didElementStateChange();
250}
