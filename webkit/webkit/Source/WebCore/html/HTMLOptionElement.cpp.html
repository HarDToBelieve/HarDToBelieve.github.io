72ExceptionOr<Ref<HTMLOptionElement>> HTMLOptionElement::createForLegacyFactoryFunction(Document& document, const String& text, const String& value, bool defaultSelected, bool selected)
73{
74 auto element = create(document);
75
76 if (!text.isEmpty()) {
77 auto appendResult = element->appendChild(Text::create(document, text));
78 if (appendResult.hasException())
79 return appendResult.releaseException();
80 }
81
82 if (!value.isNull())
83 element->setValue(value);
84 if (defaultSelected)
85 element->setAttributeWithoutSynchronization(selectedAttr, emptyAtom());
86 element->setSelected(selected);
87
88 return element;
89}

91bool HTMLOptionElement::isFocusable() const
92{
93 if (!supportsFocus())
94 return false;
95 // Option elements do not have a renderer.
96 auto* style = const_cast<HTMLOptionElement&>(*this).computedStyle();
97 return style && style->display() != DisplayType::None;
98}

114void HTMLOptionElement::setText(const String &text)
115{
116 Ref<HTMLOptionElement> protectedThis(*this);
117
118 // Changing the text causes a recalc of a select's items, which will reset the selected
119 // index to the first item if the select is single selection with a menu list. We attempt to
120 // preserve the selected item.
121 RefPtr<HTMLSelectElement> select = ownerSelectElement();
122 bool selectIsMenuList = select && select->usesMenuList();
123 int oldSelectedIndex = selectIsMenuList ? select->selectedIndex() : -1;
124
125 // Handle the common special case where there's exactly 1 child node, and it's a text node.
126 RefPtr<Node> child = firstChild();
127 if (is<Text>(child) && !child->nextSibling())
128 downcast<Text>(*child).setData(text);
129 else {
130 removeChildren();
131 appendChild(Text::create(document(), text));
132 }
133
134 if (selectIsMenuList && select->selectedIndex() != oldSelectedIndex)
135 select->setSelectedIndex(oldSelectedIndex);
136}

148int HTMLOptionElement::index() const
149{
150 // It would be faster to cache the index, but harder to get it right in all cases.
151
152 RefPtr<HTMLSelectElement> selectElement = ownerSelectElement();
153 if (!selectElement)
154 return 0;
155
156 int optionIndex = 0;
157
158 for (auto& item : selectElement->listItems()) {
159 if (!is<HTMLOptionElement>(*item))
160 continue;
161 if (item == this)
162 return optionIndex;
163 ++optionIndex;
164 }
165
166 return 0;
167}

169void HTMLOptionElement::parseAttribute(const QualifiedName& name, const AtomString& value)
170{
171#if ENABLE(DATALIST_ELEMENT)
172 if (name == valueAttr) {
173 for (auto& dataList : ancestorsOfType<HTMLDataListElement>(*this))
174 dataList.optionElementChildrenChanged();
175 } else
176#endif
177 if (name == disabledAttr) {
178 bool oldDisabled = m_disabled;
179 m_disabled = !value.isNull();
180 if (oldDisabled != m_disabled) {
181 invalidateStyleForSubtree();
182 if (renderer() && renderer()->style().hasAppearance())
183 renderer()->theme().stateChanged(*renderer(), ControlStates::States::Enabled);
184 }
185 } else if (name == selectedAttr) {
186 invalidateStyleForSubtree();
187
188 // FIXME: This doesn't match what the HTML specification says.
189 // The specification implies that removing the selected attribute or
190 // changing the value of a selected attribute that is already present
191 // has no effect on whether the element is selected. Further, it seems
192 // that we need to do more than just set m_isSelected to select in that
193 // case; we'd need to do the other work from the setSelected function.
194 m_isSelected = !value.isNull();
195 } else
196 HTMLElement::parseAttribute(name, value);
197}

212bool HTMLOptionElement::selected() const
213{
214 if (RefPtr<HTMLSelectElement> select = ownerSelectElement())
215 select->updateListItemSelectedStates();
216 return m_isSelected;
217}
218
219void HTMLOptionElement::setSelected(bool selected)
220{
221 if (m_isSelected == selected)
222 return;
223
224 setSelectedState(selected);
225
226 if (RefPtr<HTMLSelectElement> select = ownerSelectElement())
227 select->optionSelectionStateChanged(*this, selected);
228}
229
230void HTMLOptionElement::setSelectedState(bool selected)
231{
232 if (m_isSelected == selected)
233 return;
234
235 m_isSelected = selected;
236 invalidateStyleForSubtree();
237}

305Node::InsertedIntoAncestorResult HTMLOptionElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
306{
307 if (RefPtr<HTMLSelectElement> select = ownerSelectElement()) {
308 select->setRecalcListItems();
309 select->updateValidity();
310 // Do not call selected() since calling updateListItemSelectedStates()
311 // at this time won't do the right thing. (Why, exactly?)
312 // FIXME: Might be better to call this unconditionally, always passing m_isSelected,
313 // rather than only calling it if we are selected.
314 if (m_isSelected)
315 select->optionSelectionStateChanged(*this, true);
316 select->scrollToSelection();
317 }
318
319 return HTMLElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
320}

322String HTMLOptionElement::collectOptionInnerText() const
323{
324 StringBuilder text;
325 for (RefPtr<Node> node = firstChild(); node; ) {
326 if (is<Text>(*node))
327 text.append(node->nodeValue());
328 // Text nodes inside script elements are not part of the option text.
329 if (is<Element>(*node) && isScriptElement(downcast<Element>(*node)))
330 node = NodeTraversal::nextSkippingChildren(*node, this);
331 else
332 node = NodeTraversal::next(*node, this);
333 }
334 return text.toString();
335}
