58 return adoptRef(*new HTMLPictureElement(tagName, document));
59}
60
61void HTMLPictureElement::sourcesChanged()
62{
63 for (auto& element : childrenOfType<HTMLImageElement>(*this))
64 element.selectImageSource(RelevantMutation::Yes);
65}
66
67void HTMLPictureElement::sourceDimensionAttributesChanged(const HTMLSourceElement& sourceElement)
68{
69 for (auto& element : childrenOfType<HTMLImageElement>(*this)) {
70 if (&sourceElement == element.sourceElement())
71 element.invalidateAttributeMapping();
72 }
73}

75#if USE(SYSTEM_PREVIEW)
76bool HTMLPictureElement::isSystemPreviewImage()
77{
78 if (!document().settings().systemPreviewEnabled())
79 return false;
80
81 auto* parent = parentElement();
82 if (!is<HTMLAnchorElement>(parent))
83 return false;
84 return downcast<HTMLAnchorElement>(parent)->isSystemPreviewLink();
85}
