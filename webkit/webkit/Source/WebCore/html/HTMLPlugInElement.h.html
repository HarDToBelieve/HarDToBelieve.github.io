36namespace WebCore {
37
38class PluginReplacement;
39class RenderWidget;
40class Widget;
41
42class HTMLPlugInElement : public HTMLFrameOwnerElement {
43 WTF_MAKE_ISO_ALLOCATED(HTMLPlugInElement);
44public:
45 virtual ~HTMLPlugInElement();
46
47 void resetInstance();
48
49 JSC::Bindings::Instance* bindingsInstance();
50
51 enum class PluginLoadingPolicy { DoNotLoad, Load };
52 WEBCORE_EXPORT Widget* pluginWidget(PluginLoadingPolicy = PluginLoadingPolicy::Load) const;
53
54 enum DisplayState {
55 Playing,
56 PreparingPluginReplacement,
57 DisplayingPluginReplacement,
58 };
59 DisplayState displayState() const { return m_displayState; }
60 void setDisplayState(DisplayState);
61
62 JSC::JSObject* scriptObjectForPluginReplacement();
63#if PLATFORM(COCOA)
64 JSValueInWrappedObject& pluginReplacementScriptObject() { return m_pluginReplacementScriptObject; }
65#endif
66
67 bool isCapturingMouseEvents() const { return m_isCapturingMouseEvents; }
68 void setIsCapturingMouseEvents(bool capturing) { m_isCapturingMouseEvents = capturing; }
69
70#if PLATFORM(IOS_FAMILY)
71 bool willRespondToMouseMoveEvents() final { return false; }
72#endif
73 bool willRespondToMouseClickEvents() final;
74
75 virtual bool isPlugInImageElement() const = 0;
76
77 bool isUserObservable() const;
78
79 WEBCORE_EXPORT bool isBelowSizeThreshold() const;
80
81 // Return whether or not the replacement content for blocked plugins is accessible to the user.
82 WEBCORE_EXPORT bool setReplacement(RenderEmbeddedObject::PluginUnavailabilityReason, const String& unavailabilityDescription);
83
84 WEBCORE_EXPORT bool isReplacementObscured();
85
86protected:
87 HTMLPlugInElement(const QualifiedName& tagName, Document&);
88
89 bool canContainRangeEndPoint() const override { return false; }
90 void willDetachRenderers() override;
91 bool hasPresentationalHintsForAttribute(const QualifiedName&) const override;
92 void collectPresentationalHintsForAttribute(const QualifiedName&, const AtomString&, MutableStyleProperties&) override;
93
94 virtual bool useFallbackContent() const { return false; }
95
96 void defaultEventHandler(Event&) final;
97
98 virtual bool requestObject(const String& url, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues);
99 RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) override;
100 void didAddUserAgentShadowRoot(ShadowRoot&) final;
101
102 // Subclasses should use guardedDispatchBeforeLoadEvent instead of calling dispatchBeforeLoadEvent directly.
103 bool guardedDispatchBeforeLoadEvent(const String& sourceURL);
104
105 // This will load the plugin if necessary.
106 virtual RenderWidget* renderWidgetLoadingPlugin() const;
107
108private:
109 void swapRendererTimerFired();
110 bool shouldOverridePlugin(const String& url, const String& mimeType);
111
112 bool dispatchBeforeLoadEvent(const String& sourceURL) = delete; // Generate a compile error if someone calls this by mistake.
113
114 bool supportsFocus() const final;
115
116 bool isKeyboardFocusable(KeyboardEvent*) const final;
117 bool isPluginElement() const final;
118 bool canLoadScriptURL(const URL&) const final;
119
120 RefPtr<JSC::Bindings::Instance> m_instance;
121 Timer m_swapRendererTimer;
122 RefPtr<PluginReplacement> m_pluginReplacement;
123#if PLATFORM(COCOA)
124 JSValueInWrappedObject m_pluginReplacementScriptObject;
125#endif
126 bool m_isCapturingMouseEvents { false };
127 bool m_inBeforeLoadEventHandler { false };
128 DisplayState m_displayState { Playing };
129};
130
131} // namespace WebCore
