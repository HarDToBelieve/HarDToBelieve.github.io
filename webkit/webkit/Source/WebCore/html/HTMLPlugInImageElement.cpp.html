73RenderEmbeddedObject* HTMLPlugInImageElement::renderEmbeddedObject() const
74{
75 // HTMLObjectElement and HTMLEmbedElement may return arbitrary renderers when using fallback content.
76 return is<RenderEmbeddedObject>(renderer()) ? downcast<RenderEmbeddedObject>(renderer()) : nullptr;

79bool HTMLPlugInImageElement::isImageType()
80{
81 if (m_serviceType.isEmpty() && protocolIs(m_url, "data"))
82 m_serviceType = mimeTypeFromDataURL(m_url);
83
84 if (auto frame = makeRefPtr(document().frame()))
85 return frame->loader().client().objectContentType(document().completeURL(m_url), m_serviceType) == ObjectContentType::Image;
86
87 return Image::supportsType(m_serviceType);
88}

95// Note that unlike HTMLFrameElementBase::canLoadURL this uses SecurityOrigin::canAccess.
96bool HTMLPlugInImageElement::canLoadURL(const URL& completeURL) const
97{
98 if (completeURL.protocolIsJavaScript()) {
99 RefPtr<Document> contentDocument = this->contentDocument();
100 if (contentDocument && !document().securityOrigin().isSameOriginDomain(contentDocument->securityOrigin()))
101 return false;
102 }
103
104 return !isProhibitedSelfReference(completeURL);
105}

107// We don't use m_url, or m_serviceType as they may not be the final values
108// that <object> uses depending on <param> values.
109bool HTMLPlugInImageElement::wouldLoadAsPlugIn(const String& relativeURL, const String& serviceType)
110{
111 ASSERT(document().frame());
112 URL completedURL;
113 if (!relativeURL.isEmpty())
114 completedURL = document().completeURL(relativeURL);
115 return document().frame()->loader().client().objectContentType(completedURL, serviceType) == ObjectContentType::PlugIn;
116}

118RenderPtr<RenderElement> HTMLPlugInImageElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition& insertionPosition)
119{
120 ASSERT(document().backForwardCacheState() == Document::NotInBackForwardCache);
121
122 if (displayState() >= PreparingPluginReplacement)
123 return HTMLPlugInElement::createElementRenderer(WTFMove(style), insertionPosition);
124
125 // Once a plug-in element creates its renderer, it needs to be told when the document goes
126 // inactive or reactivates so it can clear the renderer before going into the back/forward cache.
127 if (!m_needsDocumentActivationCallbacks) {
128 m_needsDocumentActivationCallbacks = true;
129 document().registerForDocumentSuspensionCallbacks(*this);
130 }
131
132 if (useFallbackContent())
133 return RenderElement::createFor(*this, WTFMove(style));
134
135 if (isImageType())
136 return createRenderer<RenderImage>(*this, WTFMove(style));
137
138 return HTMLPlugInElement::createElementRenderer(WTFMove(style), insertionPosition);
139}

146void HTMLPlugInImageElement::willRecalcStyle(Style::Change change)
147{
148 // Make sure style recalcs scheduled by a child shadow tree don't trigger reconstruction and cause flicker.
149 if (change == Style::Change::None && styleValidity() == Style::Validity::Valid)
150 return;
151
152 // FIXME: There shoudn't be need to force render tree reconstruction here.
153 // It is only done because loading and load event dispatching is tied to render tree construction.
154 if (!useFallbackContent() && needsWidgetUpdate() && renderer() && !isImageType())
155 invalidateStyleAndRenderersForSubtree();
156}

162 HTMLPlugInElement::didRecalcStyle(styleChange);

165void HTMLPlugInImageElement::didAttachRenderers()
166{
167 m_needsWidgetUpdate = true;
168 scheduleUpdateForAfterStyleResolution();
169
170 // Update the RenderImageResource of the associated RenderImage.
171 if (m_imageLoader && is<RenderImage>(renderer())) {
172 auto& renderImageResource = downcast<RenderImage>(*renderer()).imageResource();
173 if (!renderImageResource.cachedImage())
174 renderImageResource.setCachedImage(m_imageLoader->image());
175 }
176
177 HTMLPlugInElement::didAttachRenderers();
178}

180void HTMLPlugInImageElement::willDetachRenderers()
181{
182 auto widget = makeRefPtr(pluginWidget(PluginLoadingPolicy::DoNotLoad));
183 if (is<PluginViewBase>(widget))
184 downcast<PluginViewBase>(*widget).willDetachRenderer();
185
186 HTMLPlugInElement::willDetachRenderers();
187}

189void HTMLPlugInImageElement::scheduleUpdateForAfterStyleResolution()
190{
191 if (m_hasUpdateScheduledForAfterStyleResolution)
192 return;
193
194 document().incrementLoadEventDelayCount();
195
196 m_hasUpdateScheduledForAfterStyleResolution = true;
197
198 Style::queuePostResolutionCallback([protectedThis = makeRef(*this)] {
199 protectedThis->updateAfterStyleResolution();
200 });
201}
202
203void HTMLPlugInImageElement::updateAfterStyleResolution()
204{
205 m_hasUpdateScheduledForAfterStyleResolution = false;
206
207 // Do this after style resolution, since the image or widget load might complete synchronously
208 // and cause us to re-enter otherwise. Also, we can't really answer the question "do I have a renderer"
209 // accurately until after style resolution.
210
211 if (renderer() && !useFallbackContent()) {
212 if (isImageType()) {
213 if (!m_imageLoader)
214 m_imageLoader = makeUnique<HTMLImageLoader>(*this);
215 if (m_needsImageReload)
216 m_imageLoader->updateFromElementIgnoringPreviousError();
217 else
218 m_imageLoader->updateFromElement();
219 } else {
220 if (needsWidgetUpdate() && renderEmbeddedObject() && !renderEmbeddedObject()->isPluginUnavailable())
221 updateWidget(CreatePlugins::No);
222 }
223 }
224
225 // Either we reloaded the image just now, or we had some reason not to.
226 // Either way, clear the flag now, since we don't need to remember to try again.
227 m_needsImageReload = false;
228
229 document().decrementLoadEventDelayCount();
230}

232void HTMLPlugInImageElement::didMoveToNewDocument(Document& oldDocument, Document& newDocument)
233{
234 ASSERT_WITH_SECURITY_IMPLICATION(&document() == &newDocument);
235 if (m_needsDocumentActivationCallbacks) {
236 oldDocument.unregisterForDocumentSuspensionCallbacks(*this);
237 newDocument.registerForDocumentSuspensionCallbacks(*this);
238 }
239
240 if (m_imageLoader)
241 m_imageLoader->elementDidMoveToNewDocument(oldDocument);
242
243 if (m_hasUpdateScheduledForAfterStyleResolution) {
244 oldDocument.decrementLoadEventDelayCount();
245 newDocument.incrementLoadEventDelayCount();
246 }
247
248 HTMLPlugInElement::didMoveToNewDocument(oldDocument, newDocument);
249}

267bool HTMLPlugInImageElement::shouldBypassCSPForPDFPlugin(const String& contentType) const
268{
269#if ENABLE(PDFKIT_PLUGIN)
270 // We only consider bypassing this CSP check if plugins are disabled. In that case we know that
271 // any plugin used is a browser implementation detail. It is not safe to skip this check
272 // if plugins are enabled in case an external plugin is used to load PDF content.
273 // FIXME: Check for alternative PDF plugins here so we can bypass this CSP check for PDFPlugin even when plugins are enabled.
274 if (document().frame()->arePluginsEnabled())
275 return false;
276
277 return document().frame()->loader().client().shouldUsePDFPlugin(contentType, document().url().path());
278#else
279 UNUSED_PARAM(contentType);
280 return false;
281#endif
282}

284bool HTMLPlugInImageElement::canLoadPlugInContent(const String& relativeURL, const String& mimeType) const
285{
286 // Elements in user agent show tree should load whatever the embedding document policy is.
287 if (isInUserAgentShadowTree())
288 return true;
289
290 URL completedURL;
291 if (!relativeURL.isEmpty())
292 completedURL = document().completeURL(relativeURL);
293
294 ASSERT(document().contentSecurityPolicy());
295 const ContentSecurityPolicy& contentSecurityPolicy = *document().contentSecurityPolicy();
296
297 contentSecurityPolicy.upgradeInsecureRequestIfNeeded(completedURL, ContentSecurityPolicy::InsecureRequestType::Load);
298
299 if (!shouldBypassCSPForPDFPlugin(mimeType) && !contentSecurityPolicy.allowObjectFromSource(completedURL))
300 return false;
301
302 auto& declaredMimeType = document().isPluginDocument() && document().ownerElement() ?
303 document().ownerElement()->attributeWithoutSynchronization(HTMLNames::typeAttr) : attributeWithoutSynchronization(HTMLNames::typeAttr);
304 return contentSecurityPolicy.allowPluginType(mimeType, declaredMimeType, completedURL);
305}

307bool HTMLPlugInImageElement::requestObject(const String& relativeURL, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues)
308{
309 ASSERT(document().frame());
310
311 if (relativeURL.isEmpty() && mimeType.isEmpty())
312 return false;
313
314 if (!canLoadPlugInContent(relativeURL, mimeType)) {
315 renderEmbeddedObject()->setPluginUnavailabilityReason(RenderEmbeddedObject::PluginBlockedByContentSecurityPolicy);
316 return false;
317 }
318
319 if (HTMLPlugInElement::requestObject(relativeURL, mimeType, paramNames, paramValues))
320 return true;
321
322 return document().frame()->loader().subframeLoader().requestObject(*this, relativeURL, getNameAttribute(), mimeType, paramNames, paramValues);
323}

325void HTMLPlugInImageElement::updateImageLoaderWithNewURLSoon()
326{
327 if (m_needsImageReload)
328 return;
329
330 m_needsImageReload = true;
331 if (inRenderedDocument())
332 scheduleUpdateForAfterStyleResolution();
333 invalidateStyle();
334}
