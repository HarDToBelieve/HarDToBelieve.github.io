38inline HTMLPreElement::HTMLPreElement(const QualifiedName& tagName, Document& document)

43Ref<HTMLPreElement> HTMLPreElement::create(const QualifiedName& tagName, Document& document)
44{

46}
47
48bool HTMLPreElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
49{
50 if (name == wrapAttr)
51 return true;
52 return HTMLElement::hasPresentationalHintsForAttribute(name);
53}
54
55void HTMLPreElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
56{
57 if (name == wrapAttr)
58 style.setProperty(CSSPropertyWhiteSpace, CSSValuePreWrap);
59 else
60 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
61}
