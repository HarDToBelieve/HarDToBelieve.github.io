65void HTMLScriptElement::parseAttribute(const QualifiedName& name, const AtomString& value)
66{
67 if (name == srcAttr)
68 handleSourceAttribute(value);
69 else if (name == asyncAttr)
70 handleAsyncAttribute();
71 else
72 HTMLElement::parseAttribute(name, value);
73}

175void HTMLScriptElement::dispatchLoadEvent()
176{
177 ASSERT(!haveFiredLoadEvent());
178 setHaveFiredLoadEvent(true);
179
180 dispatchEvent(Event::create(eventNames().loadEvent, Event::CanBubble::No, Event::IsCancelable::No));
181}

198ReferrerPolicy HTMLScriptElement::referrerPolicy() const
199{
200 if (document().settings().referrerPolicyAttributeEnabled())
201 return parseReferrerPolicy(attributeWithoutSynchronization(referrerpolicyAttr), ReferrerPolicySource::ReferrerPolicyAttribute).value_or(ReferrerPolicy::EmptyString);
202 return ReferrerPolicy::EmptyString;
203}
