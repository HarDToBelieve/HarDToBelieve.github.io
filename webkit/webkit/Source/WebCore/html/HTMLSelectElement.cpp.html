114void HTMLSelectElement::optionSelectedByUser(int optionIndex, bool fireOnChangeNow, bool allowMultipleSelection)
115{
116 // User interaction such as mousedown events can cause list box select elements to send change events.
117 // This produces that same behavior for changes triggered by other code running on behalf of the user.
118 if (!usesMenuList()) {
119 updateSelectedState(optionToListIndex(optionIndex), allowMultipleSelection, false);
120 updateValidity();
121 if (auto* renderer = this->renderer())
122 renderer->updateFromElement();
123 if (fireOnChangeNow)
124 listBoxOnChange();
125 return;
126 }
127
128 // Bail out if this index is already the selected one, to avoid running unnecessary JavaScript that can mess up
129 // autofill when there is no actual change (see https://bugs.webkit.org/show_bug.cgi?id=35256 and <rdar://7467917>).
130 // The selectOption function does not behave this way, possibly because other callers need a change event even
131 // in cases where the selected option is not change.
132 if (optionIndex == selectedIndex())
133 return;
134
135 selectOption(optionIndex, DeselectOtherOptions | (fireOnChangeNow ? DispatchChangeEvent : 0) | UserDriven);
136}

138bool HTMLSelectElement::hasPlaceholderLabelOption() const
139{
140 // The select element has no placeholder label option if it has an attribute "multiple" specified or a display size of non-1.
141 //
142 // The condition "size() > 1" is not compliant with the HTML5 spec as of Dec 3, 2010. "size() != 1" is correct.
143 // Using "size() > 1" here because size() may be 0 in WebKit.
144 // See the discussion at https://bugs.webkit.org/show_bug.cgi?id=43887
145 //
146 // "0 size()" happens when an attribute "size" is absent or an invalid size attribute is specified.
147 // In this case, the display size should be assumed as the default.
148 // The default display size is 1 for non-multiple select elements, and 4 for multiple select elements.
149 //
150 // Finally, if size() == 0 and non-multiple, the display size can be assumed as 1.
151 if (multiple() || size() > 1)
152 return false;
153
154 int listIndex = optionToListIndex(0);
155 ASSERT(listIndex >= 0);
156 if (listIndex < 0)
157 return false;
158 HTMLOptionElement& option = downcast<HTMLOptionElement>(*listItems()[listIndex]);
159 return !listIndex && option.value().isEmpty();
160}

173bool HTMLSelectElement::valueMissing() const
174{
175 if (!willValidate())
176 return false;
177
178 if (!isRequired())
179 return false;
180
181 int firstSelectionIndex = selectedIndex();
182
183 // If a non-placeholer label option is selected (firstSelectionIndex > 0), it's not value-missing.
184 return firstSelectionIndex < 0 || (!firstSelectionIndex && hasPlaceholderLabelOption());
185}

251String HTMLSelectElement::value() const
252{
253 for (auto* item : listItems()) {
254 if (is<HTMLOptionElement>(*item)) {
255 HTMLOptionElement& option = downcast<HTMLOptionElement>(*item);
256 if (option.selected())
257 return option.value();
258 }

263void HTMLSelectElement::setValue(const String& value)
264{
265 // Find the option with value() matching the given parameter and make it the current selection.
266 unsigned optionIndex = 0;
267 for (auto* item : listItems()) {
268 if (is<HTMLOptionElement>(*item)) {
269 if (downcast<HTMLOptionElement>(*item).value() == value) {
270 setSelectedIndex(optionIndex);
271 return;
272 }
273 ++optionIndex;
274 }
275 }
276
277 setSelectedIndex(-1);
278}

291void HTMLSelectElement::parseAttribute(const QualifiedName& name, const AtomString& value)
292{
293 if (name == sizeAttr) {
294 unsigned oldSize = m_size;
295 unsigned size = limitToOnlyHTMLNonNegative(value);
296
297 // Ensure that we've determined selectedness of the items at least once prior to changing the size.
298 if (oldSize != size)
299 updateListItemSelectedStates();
300
301 m_size = size;
302 updateValidity();
303 if (m_size != oldSize) {
304 invalidateStyleAndRenderersForSubtree();
305 setRecalcListItems();
306 updateValidity();
307 }
308 } else if (name == multipleAttr)
309 parseMultipleAttribute(value);
310 else
311 HTMLFormControlElementWithState::parseAttribute(name, value);
312}

426ExceptionOr<void> HTMLSelectElement::setItem(unsigned index, HTMLOptionElement* option)
427{
428 if (!option) {
429 remove(index);
430 return { };
431 }
432
433 if (index > maxSelectItems - 1)
434 index = maxSelectItems - 1;
435
436 int diff = index - length();
437
438 RefPtr<HTMLOptionElement> before;
439 // Out of array bounds? First insert empty dummies.
440 if (diff > 0) {
441 auto result = setLength(index);
442 if (result.hasException())
443 return result;
444 // Replace an existing entry?
445 } else if (diff < 0) {
446 before = item(index + 1);
447 remove(index);
448 }
449
450 // Finally add the new element.
451 auto result = add(option, HTMLElementOrInt { before.get() });
452 if (result.hasException())
453 return result;
454
455 if (diff >= 0 && option->selected())
456 optionSelectionStateChanged(*option, true);
457
458 return { };
459}
460
461ExceptionOr<void> HTMLSelectElement::setLength(unsigned newLength)
462{
463 if (newLength > length() && newLength > maxSelectItems) {
464 document().addConsoleMessage(MessageSource::Other, MessageLevel::Warning, makeString("Blocked attempt to expand the option list to ", newLength, " items. The maximum number of items allowed is ", maxSelectItems, '.'));
465 return { };
466 }
467
468 int diff = length() - newLength;
469
470 if (diff < 0) { // Add dummy elements.
471 do {
472 auto result = add(HTMLOptionElement::create(document()).ptr(), std::nullopt);
473 if (result.hasException())
474 return result;
475 } while (++diff);
476 } else {
477 auto& items = listItems();
478
479 // Removing children fires mutation events, which might mutate the DOM further, so we first copy out a list
480 // of elements that we intend to remove then attempt to remove them one at a time.
481 Vector<Ref<HTMLOptionElement>> itemsToRemove;
482 size_t optionIndex = 0;
483 for (auto& item : items) {
484 if (is<HTMLOptionElement>(*item) && optionIndex++ >= newLength) {
485 ASSERT(item->parentNode());
486 itemsToRemove.append(downcast<HTMLOptionElement>(*item));
487 }
488 }
489
490 // FIXME: Clients can detect what order we remove the options in; is it good to remove them in ascending order?
491 // FIXME: This ignores exceptions. A previous version passed through the exception only for the last item removed.
492 // What exception behavior do we want?
493 for (auto& item : itemsToRemove)
494 item->remove();
495 }
496 return { };
497}

513// Returns the 1st valid item |skip| items from |listIndex| in direction |direction| if there is one.
514// Otherwise, it returns the valid item closest to that boundary which is past |listIndex| if there is one.
515// Otherwise, it returns |listIndex|.
516// Valid means that it is enabled and an option element.
517int HTMLSelectElement::nextValidIndex(int listIndex, SkipDirection direction, int skip) const
518{
519 ASSERT(direction == -1 || direction == 1);
520 auto& listItems = this->listItems();
521 int lastGoodIndex = listIndex;
522 int size = listItems.size();
523 for (listIndex += direction; listIndex >= 0 && listIndex < size; listIndex += direction) {
524 --skip;
525 if (!listItems[listIndex]->isDisabledFormControl() && is<HTMLOptionElement>(*listItems[listIndex])) {
526 lastGoodIndex = listIndex;
527 if (skip <= 0)
528 break;
529 }
530 }
531 return lastGoodIndex;
532}

560// Returns the index of the next valid item one page away from |startIndex| in direction |direction|.
561int HTMLSelectElement::nextSelectableListIndexPageAway(int startIndex, SkipDirection direction) const
562{
563 auto& items = listItems();
564
565 // Can't use m_size because renderer forces a minimum size.
566 int pageSize = 0;
567 auto* renderer = this->renderer();
568 if (is<RenderListBox>(*renderer))
569 pageSize = downcast<RenderListBox>(*renderer).size() - 1; // -1 so we still show context.
570
571 // One page away, but not outside valid bounds.
572 // If there is a valid option item one page away, the index is chosen.
573 // If there is no exact one page away valid option, returns startIndex or the most far index.
574 int edgeIndex = direction == SkipForwards ? 0 : items.size() - 1;
575 int skipAmount = pageSize + (direction == SkipForwards ? startIndex : edgeIndex - startIndex);
576 return nextValidIndex(edgeIndex, direction, skipAmount);
577}

579void HTMLSelectElement::selectAll()
580{
581 ASSERT(!usesMenuList());
582 if (!renderer() || !m_multiple)
583 return;
584
585 // Save the selection so it can be compared to the new selectAll selection
586 // when dispatching change events.
587 saveLastSelection();
588
589 m_activeSelectionState = true;
590 setActiveSelectionAnchorIndex(nextSelectableListIndex(-1));
591 setActiveSelectionEndIndex(previousSelectableListIndex(-1));
592 if (m_activeSelectionAnchorIndex < 0)
593 return;
594
595 updateListBoxSelection(false);
596 listBoxOnChange();
597 updateValidity();
598}

600void HTMLSelectElement::saveLastSelection()
601{
602 if (usesMenuList()) {
603 m_lastOnChangeIndex = selectedIndex();
604 return;
605 }
606
607 m_lastOnChangeSelection.clear();
608 for (auto& element : listItems())
609 m_lastOnChangeSelection.append(is<HTMLOptionElement>(*element) && downcast<HTMLOptionElement>(*element).selected());
610}

629void HTMLSelectElement::updateListBoxSelection(bool deselectOtherOptions)
630{
631 ASSERT(renderer());
632
633#if !PLATFORM(IOS_FAMILY)
634 ASSERT(renderer()->isListBox() || m_multiple);
635#else
636 ASSERT(renderer()->isMenuList() || m_multiple);
637#endif
638
639 ASSERT(!listItems().size() || m_activeSelectionAnchorIndex >= 0);
640
641 unsigned start = std::min(m_activeSelectionAnchorIndex, m_activeSelectionEndIndex);
642 unsigned end = std::max(m_activeSelectionAnchorIndex, m_activeSelectionEndIndex);
643
644 auto& items = listItems();
645 for (unsigned i = 0; i < items.size(); ++i) {
646 auto& element = *items[i];
647 if (!is<HTMLOptionElement>(element) || downcast<HTMLOptionElement>(element).isDisabledFormControl())
648 continue;
649
650 if (i >= start && i <= end)
651 downcast<HTMLOptionElement>(element).setSelectedState(m_activeSelectionState);
652 else if (deselectOtherOptions || i >= m_cachedStateForActiveSelection.size())
653 downcast<HTMLOptionElement>(element).setSelectedState(false);
654 else
655 downcast<HTMLOptionElement>(element).setSelectedState(m_cachedStateForActiveSelection[i]);
656 }
657
658 invalidateSelectedItems();
659 scrollToSelection();
660 updateValidity();
661}
662
663void HTMLSelectElement::listBoxOnChange()
664{
665 ASSERT(!usesMenuList() || m_multiple);
666
667 auto& items = listItems();
668
669 // If the cached selection list is empty, or the size has changed, then fire
670 // dispatchFormControlChangeEvent, and return early.
671 if (m_lastOnChangeSelection.isEmpty() || m_lastOnChangeSelection.size() != items.size()) {
672 dispatchFormControlChangeEvent();
673 return;
674 }
675
676 // Update m_lastOnChangeSelection and fire dispatchFormControlChangeEvent.
677 bool fireOnChange = false;
678 for (unsigned i = 0; i < items.size(); ++i) {
679 auto& element = *items[i];
680 bool selected = is<HTMLOptionElement>(element) && downcast<HTMLOptionElement>(element).selected();
681 if (selected != m_lastOnChangeSelection[i])
682 fireOnChange = true;
683 m_lastOnChangeSelection[i] = selected;
684 }
685
686 if (fireOnChange) {
687 dispatchInputEvent();
688 dispatchFormControlChangeEvent();
689 }
690}

692void HTMLSelectElement::dispatchChangeEventForMenuList()
693{
694 ASSERT(usesMenuList());
695
696 int selected = selectedIndex();
697 if (m_lastOnChangeIndex != selected && m_isProcessingUserDrivenChange) {
698 m_lastOnChangeIndex = selected;
699 m_isProcessingUserDrivenChange = false;
700 dispatchInputEvent();
701 dispatchFormControlChangeEvent();
702 }

756void HTMLSelectElement::setRecalcListItems()
757{
758 m_shouldRecalcListItems = true;
759 // Manual selection anchor is reset when manipulating the select programmatically.
760 m_activeSelectionAnchorIndex = -1;
761 setOptionsChangedOnRenderer();
762 invalidateStyleForSubtree();
763 if (!isConnected()) {
764 if (HTMLCollection* collection = cachedHTMLCollection(SelectOptions))
765 collection->invalidateCache();

767 if (!isConnected())
768 invalidateSelectedItems();
769 if (auto* cache = document().existingAXObjectCache())
770 cache->childrenChanged(this);

773void HTMLSelectElement::recalcListItems(bool updateSelectedStates) const
774{
775 m_listItems.clear();
776
777 m_shouldRecalcListItems = false;
778
779 RefPtr<HTMLOptionElement> foundSelected;
780 RefPtr<HTMLOptionElement> firstOption;
781 for (RefPtr<Element> currentElement = ElementTraversal::firstWithin(*this); currentElement; ) {
782 if (!is<HTMLElement>(*currentElement)) {
783 currentElement = ElementTraversal::nextSkippingChildren(*currentElement, this);
784 continue;
785 }
786 HTMLElement& current = downcast<HTMLElement>(*currentElement);
787
788 // Only consider optgroup elements that are direct children of the select element.
789 if (is<HTMLOptGroupElement>(current) && current.parentNode() == this) {
790 m_listItems.append(&current);
791 if (RefPtr<Element> nextElement = ElementTraversal::firstWithin(current)) {
792 currentElement = nextElement;
793 continue;
794 }
795 }
796
797 if (is<HTMLOptionElement>(current)) {
798 m_listItems.append(&current);
799
800 if (updateSelectedStates && !m_multiple) {
801 HTMLOptionElement& option = downcast<HTMLOptionElement>(current);
802 if (!firstOption)
803 firstOption = &option;
804 if (option.selected()) {
805 if (foundSelected)
806 foundSelected->setSelectedState(false);
807 foundSelected = &option;
808 } else if (m_size <= 1 && !foundSelected && !option.isDisabledFormControl()) {
809 foundSelected = &option;
810 foundSelected->setSelectedState(true);
811 }
812 }
813 }
814
815 if (current.hasTagName(hrTag))
816 m_listItems.append(&current);
817
818 // In conforming HTML code, only <optgroup> and <option> will be found
819 // within a <select>. We call NodeTraversal::nextSkippingChildren so that we only step
820 // into those tags that we choose to. For web-compat, we should cope
821 // with the case where odd tags like a <div> have been added but we
822 // handle this because such tags have already been removed from the
823 // <select>'s subtree at this point.
824 currentElement = ElementTraversal::nextSkippingChildren(*currentElement, this);
825 }
826
827 if (!foundSelected && m_size <= 1 && firstOption && !firstOption->selected())
828 firstOption->setSelectedState(true);
829}

863void HTMLSelectElement::selectOption(int optionIndex, SelectOptionFlags flags)
864{
865 bool shouldDeselect = !m_multiple || (flags & DeselectOtherOptions);
866
867 auto& items = listItems();
868 int listIndex = optionToListIndex(optionIndex);
869
870 RefPtr<HTMLElement> element;
871 if (listIndex >= 0)
872 element = items[listIndex];
873
874 if (shouldDeselect)
875 deselectItemsWithoutValidation(element.get());
876
877 if (is<HTMLOptionElement>(element)) {
878 if (m_activeSelectionAnchorIndex < 0 || shouldDeselect)
879 setActiveSelectionAnchorIndex(listIndex);
880 if (m_activeSelectionEndIndex < 0 || shouldDeselect)
881 setActiveSelectionEndIndex(listIndex);
882 downcast<HTMLOptionElement>(*element).setSelectedState(true);
883 }
884
885 invalidateSelectedItems();
886 updateValidity();
887
888 // For the menu list case, this is what makes the selected element appear.
889 if (auto* renderer = this->renderer())
890 renderer->updateFromElement();
891
892 scrollToSelection();
893
894 if (usesMenuList()) {
895 m_isProcessingUserDrivenChange = flags & UserDriven;
896 if (flags & DispatchChangeEvent)
897 dispatchChangeEventForMenuList();
898 if (auto* renderer = this->renderer()) {
899 if (is<RenderMenuList>(*renderer))
900 downcast<RenderMenuList>(*renderer).didSetSelectedIndex(listIndex);
901 else
902 downcast<RenderListBox>(*renderer).selectionChanged();
903 }
904 }
905}

907int HTMLSelectElement::optionToListIndex(int optionIndex) const
908{
909 auto& items = listItems();
910 int listSize = static_cast<int>(items.size());
911 if (optionIndex < 0 || optionIndex >= listSize)
912 return -1;
913
914 int optionIndex2 = -1;
915 for (int listIndex = 0; listIndex < listSize; ++listIndex) {
916 if (is<HTMLOptionElement>(*items[listIndex])) {
917 ++optionIndex2;
918 if (optionIndex2 == optionIndex)
919 return listIndex;
920 }
921 }
922
923 return -1;
924}
925
926int HTMLSelectElement::listToOptionIndex(int listIndex) const
927{
928 auto& items = listItems();
929 if (listIndex < 0 || listIndex >= static_cast<int>(items.size()) || !is<HTMLOptionElement>(*items[listIndex]))
930 return -1;
931
932 // Actual index of option not counting OPTGROUP entries that may be in list.
933 int optionIndex = 0;
934 for (int i = 0; i < listIndex; ++i) {
935 if (is<HTMLOptionElement>(*items[i]))
936 ++optionIndex;
937 }
938
939 return optionIndex;
940}

988size_t HTMLSelectElement::searchOptionsForValue(const String& value, size_t listIndexStart, size_t listIndexEnd) const
989{
990 auto& items = listItems();
991 size_t loopEndIndex = std::min(items.size(), listIndexEnd);
992 for (size_t i = listIndexStart; i < loopEndIndex; ++i) {
993 if (!is<HTMLOptionElement>(*items[i]))
994 continue;
995 if (downcast<HTMLOptionElement>(*items[i]).value() == value)
996 return i;
997 }
998 return notFound;

1001void HTMLSelectElement::restoreFormControlState(const FormControlState& state)
1002{
1003 recalcListItems();
1004
1005 auto& items = listItems();
1006 size_t itemsSize = items.size();
1007 if (!itemsSize)
1008 return;
1009
1010 for (auto& element : items) {
1011 if (!is<HTMLOptionElement>(*element))
1012 continue;
1013 downcast<HTMLOptionElement>(*element).setSelectedState(false);
1014 }
1015
1016 if (!multiple()) {
1017 size_t foundIndex = searchOptionsForValue(state[0], 0, itemsSize);
1018 if (foundIndex != notFound)
1019 downcast<HTMLOptionElement>(*items[foundIndex]).setSelectedState(true);
1020 } else {
1021 size_t startIndex = 0;
1022 for (auto& value : state) {
1023 size_t foundIndex = searchOptionsForValue(value, startIndex, itemsSize);
1024 if (foundIndex == notFound)
1025 foundIndex = searchOptionsForValue(value, 0, startIndex);
1026 if (foundIndex == notFound)
1027 continue;
1028 downcast<HTMLOptionElement>(*items[foundIndex]).setSelectedState(true);
1029 startIndex = foundIndex + 1;
1030 }
1031 }
1032
1033 invalidateSelectedItems();
1034 setOptionsChangedOnRenderer();
1035 updateValidity();
1036}

1067void HTMLSelectElement::reset()
1068{
1069 RefPtr<HTMLOptionElement> firstOption;
1070 RefPtr<HTMLOptionElement> selectedOption;
1071
1072 for (auto& element : listItems()) {
1073 if (!is<HTMLOptionElement>(*element))
1074 continue;
1075
1076 HTMLOptionElement& option = downcast<HTMLOptionElement>(*element);
1077 if (option.hasAttributeWithoutSynchronization(selectedAttr)) {
1078 if (selectedOption && !m_multiple)
1079 selectedOption->setSelectedState(false);
1080 option.setSelectedState(true);
1081 selectedOption = &option;
1082 } else
1083 option.setSelectedState(false);
1084
1085 if (!firstOption)
1086 firstOption = &option;
1087 }
1088
1089 if (!selectedOption && firstOption && !m_multiple && m_size <= 1)
1090 firstOption->setSelectedState(true);
1091
1092 invalidateSelectedItems();
1093 setOptionsChangedOnRenderer();
1094 invalidateStyleForSubtree();
1095 updateValidity();
1096}

1132void HTMLSelectElement::menuListDefaultEventHandler(Event& event)
1133{
1134 ASSERT(renderer());
1135 ASSERT(renderer()->isMenuList());
1136
1137 if (event.type() == eventNames().keydownEvent) {
1138 if (!is<KeyboardEvent>(event))
1139 return;
1140
1141 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
1142 if (platformHandleKeydownEvent(&keyboardEvent))
1143 return;
1144
1145 // When using spatial navigation, we want to be able to navigate away
1146 // from the select element when the user hits any of the arrow keys,
1147 // instead of changing the selection.
1148 if (isSpatialNavigationEnabled(document().frame())) {
1149 if (!m_activeSelectionState)
1150 return;
1151 }
1152
1153 const String& keyIdentifier = keyboardEvent.keyIdentifier();
1154 bool handled = true;
1155 auto& listItems = this->listItems();
1156 int listIndex = optionToListIndex(selectedIndex());
1157
1158 // When using caret browsing, we want to be able to move the focus
1159 // out of the select element when user hits a left or right arrow key.
1160 if (document().settings().caretBrowsingEnabled()) {
1161 if (keyIdentifier == "Left" || keyIdentifier == "Right")
1162 return;
1163 }
1164
1165 if (keyIdentifier == "Down" || keyIdentifier == "Right")
1166 listIndex = nextValidIndex(listIndex, SkipForwards, 1);
1167 else if (keyIdentifier == "Up" || keyIdentifier == "Left")
1168 listIndex = nextValidIndex(listIndex, SkipBackwards, 1);
1169 else if (keyIdentifier == "PageDown")
1170 listIndex = nextValidIndex(listIndex, SkipForwards, 3);
1171 else if (keyIdentifier == "PageUp")
1172 listIndex = nextValidIndex(listIndex, SkipBackwards, 3);
1173 else if (keyIdentifier == "Home")
1174 listIndex = nextValidIndex(-1, SkipForwards, 1);
1175 else if (keyIdentifier == "End")
1176 listIndex = nextValidIndex(listItems.size(), SkipBackwards, 1);
1177 else
1178 handled = false;
1179
1180 if (handled && static_cast<size_t>(listIndex) < listItems.size())
1181 selectOption(listToOptionIndex(listIndex), DeselectOtherOptions | DispatchChangeEvent | UserDriven);
1182
1183 if (handled)
1184 keyboardEvent.setDefaultHandled();
1185 }
1186
1187 // Use key press event here since sending simulated mouse events
1188 // on key down blocks the proper sending of the key press event.
1189 if (event.type() == eventNames().keypressEvent) {
1190 if (!is<KeyboardEvent>(event))
1191 return;
1192
1193 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
1194 int keyCode = keyboardEvent.keyCode();
1195 bool handled = false;
1196
1197 if (keyCode == ' ' && isSpatialNavigationEnabled(document().frame())) {
1198 // Use space to toggle arrow key handling for selection change or spatial navigation.
1199 m_activeSelectionState = !m_activeSelectionState;
1200 keyboardEvent.setDefaultHandled();
1201 return;
1202 }
1203
1204 if (RenderTheme::singleton().popsMenuBySpaceOrReturn()) {
1205 if (keyCode == ' ' || keyCode == '\r') {
1206 focus();
1207 document().updateStyleIfNeeded();
1208
1209 // Calling focus() may remove the renderer or change the renderer type.
1210 auto* renderer = this->renderer();
1211 if (!is<RenderMenuList>(renderer))
1212 return;
1213
1214 // Save the selection so it can be compared to the new selection
1215 // when dispatching change events during selectOption, which
1216 // gets called from RenderMenuList::valueChanged, which gets called
1217 // after the user makes a selection from the menu.
1218 saveLastSelection();
1219 downcast<RenderMenuList>(*renderer).showPopup();
1220 handled = true;
1221 }
1222 } else if (RenderTheme::singleton().popsMenuByArrowKeys()) {
1223 if (keyCode == ' ') {
1224 focus();
1225 document().updateStyleIfNeeded();
1226
1227 // Calling focus() may remove the renderer or change the renderer type.
1228 auto* renderer = this->renderer();
1229 if (!is<RenderMenuList>(renderer))
1230 return;
1231
1232 // Save the selection so it can be compared to the new selection
1233 // when dispatching change events during selectOption, which
1234 // gets called from RenderMenuList::valueChanged, which gets called
1235 // after the user makes a selection from the menu.
1236 saveLastSelection();
1237 downcast<RenderMenuList>(*renderer).showPopup();
1238 handled = true;
1239 } else if (keyCode == '\r') {
1240 if (form())
1241 form()->submitImplicitly(keyboardEvent, false);
1242 dispatchChangeEventForMenuList();
1243 handled = true;
1244 }
1245 }
1246
1247 if (handled)
1248 keyboardEvent.setDefaultHandled();
1249 }
1250
1251 if (event.type() == eventNames().mousedownEvent && is<MouseEvent>(event) && downcast<MouseEvent>(event).button() == LeftButton) {
1252 focus();
1253#if !PLATFORM(IOS_FAMILY)
1254 document().updateStyleIfNeeded();
1255
1256 auto* renderer = this->renderer();
1257 if (is<RenderMenuList>(renderer)) {
1258 auto& menuList = downcast<RenderMenuList>(*renderer);
1259 ASSERT(!menuList.popupIsVisible());
1260 // Save the selection so it can be compared to the new
1261 // selection when we call onChange during selectOption,
1262 // which gets called from RenderMenuList::valueChanged,
1263 // which gets called after the user makes a selection from
1264 // the menu.
1265 saveLastSelection();
1266 menuList.showPopup();
1267 }
1268#endif
1269 event.setDefaultHandled();
1270 }
1271
1272#if !PLATFORM(IOS_FAMILY)
1273 if (event.type() == eventNames().blurEvent && !focused()) {
1274 auto& menuList = downcast<RenderMenuList>(*renderer());
1275 if (menuList.popupIsVisible())
1276 menuList.hidePopup();
1277 }
1278#endif
1279}

1281void HTMLSelectElement::updateSelectedState(int listIndex, bool multi, bool shift)
1282{
1283 auto& items = listItems();
1284 int listSize = static_cast<int>(items.size());
1285 if (listIndex < 0 || listIndex >= listSize)
1286 return;
1287
1288 // Save the selection so it can be compared to the new selection when
1289 // dispatching change events during mouseup, or after autoscroll finishes.
1290 saveLastSelection();
1291
1292 m_activeSelectionState = true;
1293
1294 bool shiftSelect = m_multiple && shift;
1295 bool multiSelect = m_multiple && multi && !shift;
1296
1297 auto& clickedElement = *items[listIndex];
1298 if (is<HTMLOptionElement>(clickedElement)) {
1299 // Keep track of whether an active selection (like during drag
1300 // selection), should select or deselect.
1301 if (downcast<HTMLOptionElement>(clickedElement).selected() && multiSelect)
1302 m_activeSelectionState = false;
1303 if (!m_activeSelectionState)
1304 downcast<HTMLOptionElement>(clickedElement).setSelectedState(false);
1305 }
1306
1307 // If we're not in any special multiple selection mode, then deselect all
1308 // other items, excluding the clicked option. If no option was clicked, then
1309 // this will deselect all items in the list.
1310 if (!shiftSelect && !multiSelect)
1311 deselectItemsWithoutValidation(&clickedElement);
1312
1313 // If the anchor hasn't been set, and we're doing a single selection or a
1314 // shift selection, then initialize the anchor to the first selected index.
1315 if (m_activeSelectionAnchorIndex < 0 && !multiSelect)
1316 setActiveSelectionAnchorIndex(selectedIndex());
1317
1318 // Set the selection state of the clicked option.
1319 if (is<HTMLOptionElement>(clickedElement) && !downcast<HTMLOptionElement>(clickedElement).isDisabledFormControl())
1320 downcast<HTMLOptionElement>(clickedElement).setSelectedState(true);
1321
1322 // If there was no selectedIndex() for the previous initialization, or If
1323 // we're doing a single selection, or a multiple selection (using cmd or
1324 // ctrl), then initialize the anchor index to the listIndex that just got
1325 // clicked.
1326 if (m_activeSelectionAnchorIndex < 0 || !shiftSelect)
1327 setActiveSelectionAnchorIndex(listIndex);
1328
1329 invalidateSelectedItems();
1330 setActiveSelectionEndIndex(listIndex);
1331 updateListBoxSelection(!multiSelect);
1332}

1334void HTMLSelectElement::listBoxDefaultEventHandler(Event& event)
1335{
1336 auto& listItems = this->listItems();
1337
1338 if (event.type() == eventNames().mousedownEvent && is<MouseEvent>(event) && downcast<MouseEvent>(event).button() == LeftButton) {
1339 focus();
1340 document().updateStyleIfNeeded();
1341
1342 // Calling focus() may remove or change our renderer, in which case we don't want to handle the event further.
1343 auto* renderer = this->renderer();
1344 if (!is<RenderListBox>(renderer))
1345 return;
1346 auto& renderListBox = downcast<RenderListBox>(*renderer);
1347
1348 // Convert to coords relative to the list box if needed.
1349 MouseEvent& mouseEvent = downcast<MouseEvent>(event);
1350 IntPoint localOffset = roundedIntPoint(renderListBox.absoluteToLocal(mouseEvent.absoluteLocation(), UseTransforms));
1351 int listIndex = renderListBox.listIndexAtOffset(toIntSize(localOffset));
1352 if (listIndex >= 0) {
1353 if (!isDisabledFormControl()) {
1354#if PLATFORM(COCOA)
1355 updateSelectedState(listIndex, mouseEvent.metaKey(), mouseEvent.shiftKey());
1356#else
1357 updateSelectedState(listIndex, mouseEvent.ctrlKey(), mouseEvent.shiftKey());
1358#endif
1359 }
1360 if (RefPtr<Frame> frame = document().frame())
1361 frame->eventHandler().setMouseDownMayStartAutoscroll();
1362
1363 mouseEvent.setDefaultHandled();
1364 }
1365 } else if (event.type() == eventNames().mousemoveEvent && is<MouseEvent>(event) && !downcast<RenderListBox>(*renderer()).canBeScrolledAndHasScrollableArea()) {
1366 MouseEvent& mouseEvent = downcast<MouseEvent>(event);
1367 if (mouseEvent.button() != LeftButton || !mouseEvent.buttonDown())
1368 return;
1369
1370 auto& renderListBox = downcast<RenderListBox>(*renderer());
1371 IntPoint localOffset = roundedIntPoint(renderListBox.absoluteToLocal(mouseEvent.absoluteLocation(), UseTransforms));
1372 int listIndex = renderListBox.listIndexAtOffset(toIntSize(localOffset));
1373 if (listIndex >= 0) {
1374 if (!isDisabledFormControl()) {
1375 if (m_multiple) {
1376 // Only extend selection if there is something selected.
1377 if (m_activeSelectionAnchorIndex < 0)
1378 return;
1379
1380 setActiveSelectionEndIndex(listIndex);
1381 updateListBoxSelection(false);
1382 } else {
1383 setActiveSelectionAnchorIndex(listIndex);
1384 setActiveSelectionEndIndex(listIndex);
1385 updateListBoxSelection(true);
1386 }
1387 }
1388 mouseEvent.setDefaultHandled();
1389 }
1390 } else if (event.type() == eventNames().mouseupEvent && is<MouseEvent>(event) && downcast<MouseEvent>(event).button() == LeftButton && document().frame()->eventHandler().autoscrollRenderer() != renderer()) {
1391 // This click or drag event was not over any of the options.
1392 if (m_lastOnChangeSelection.isEmpty())
1393 return;
1394 // This makes sure we fire dispatchFormControlChangeEvent for a single
1395 // click. For drag selection, onChange will fire when the autoscroll
1396 // timer stops.
1397 listBoxOnChange();
1398 } else if (event.type() == eventNames().keydownEvent) {
1399 if (!is<KeyboardEvent>(event))
1400 return;
1401
1402 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
1403 const String& keyIdentifier = keyboardEvent.keyIdentifier();
1404
1405 bool handled = false;
1406 int endIndex = 0;
1407 if (m_activeSelectionEndIndex < 0) {
1408 // Initialize the end index
1409 if (keyIdentifier == "Down" || keyIdentifier == "PageDown") {
1410 int startIndex = lastSelectedListIndex();
1411 handled = true;
1412 if (keyIdentifier == "Down")
1413 endIndex = nextSelectableListIndex(startIndex);
1414 else
1415 endIndex = nextSelectableListIndexPageAway(startIndex, SkipForwards);
1416 } else if (keyIdentifier == "Up" || keyIdentifier == "PageUp") {
1417 int startIndex = optionToListIndex(selectedIndex());
1418 handled = true;
1419 if (keyIdentifier == "Up")
1420 endIndex = previousSelectableListIndex(startIndex);
1421 else
1422 endIndex = nextSelectableListIndexPageAway(startIndex, SkipBackwards);
1423 }
1424 } else {
1425 // Set the end index based on the current end index.
1426 if (keyIdentifier == "Down") {
1427 endIndex = nextSelectableListIndex(m_activeSelectionEndIndex);
1428 handled = true;
1429 } else if (keyIdentifier == "Up") {
1430 endIndex = previousSelectableListIndex(m_activeSelectionEndIndex);
1431 handled = true;
1432 } else if (keyIdentifier == "PageDown") {
1433 endIndex = nextSelectableListIndexPageAway(m_activeSelectionEndIndex, SkipForwards);
1434 handled = true;
1435 } else if (keyIdentifier == "PageUp") {
1436 endIndex = nextSelectableListIndexPageAway(m_activeSelectionEndIndex, SkipBackwards);
1437 handled = true;
1438 }
1439 }
1440 if (keyIdentifier == "Home") {
1441 endIndex = firstSelectableListIndex();
1442 handled = true;
1443 } else if (keyIdentifier == "End") {
1444 endIndex = lastSelectableListIndex();
1445 handled = true;
1446 }
1447
1448 if (isSpatialNavigationEnabled(document().frame()))
1449 // Check if the selection moves to the boundary.
1450 if (keyIdentifier == "Left" || keyIdentifier == "Right" || ((keyIdentifier == "Down" || keyIdentifier == "Up") && endIndex == m_activeSelectionEndIndex))
1451 return;
1452
1453 if (endIndex >= 0 && handled) {
1454 // Save the selection so it can be compared to the new selection
1455 // when dispatching change events immediately after making the new
1456 // selection.
1457 saveLastSelection();
1458
1459 ASSERT_UNUSED(listItems, !listItems.size() || static_cast<size_t>(endIndex) < listItems.size());
1460 setActiveSelectionEndIndex(endIndex);
1461
1462#if PLATFORM(COCOA)
1463 m_allowsNonContiguousSelection = m_multiple && isSpatialNavigationEnabled(document().frame());
1464#else
1465 m_allowsNonContiguousSelection = m_multiple && (isSpatialNavigationEnabled(document().frame()) || keyboardEvent.ctrlKey());
1466#endif
1467 bool selectNewItem = keyboardEvent.shiftKey() || !m_allowsNonContiguousSelection;
1468
1469 if (selectNewItem)
1470 m_activeSelectionState = true;
1471 // If the anchor is unitialized, or if we're going to deselect all
1472 // other options, then set the anchor index equal to the end index.
1473 bool deselectOthers = !m_multiple || (!keyboardEvent.shiftKey() && selectNewItem);
1474 if (m_activeSelectionAnchorIndex < 0 || deselectOthers) {
1475 if (deselectOthers)
1476 deselectItemsWithoutValidation();
1477 setActiveSelectionAnchorIndex(m_activeSelectionEndIndex);
1478 }
1479
1480 downcast<RenderListBox>(*renderer()).scrollToRevealElementAtListIndex(endIndex);
1481 if (selectNewItem) {
1482 updateListBoxSelection(deselectOthers);
1483 listBoxOnChange();
1484 } else
1485 scrollToSelection();
1486
1487 keyboardEvent.setDefaultHandled();
1488 }
1489 } else if (event.type() == eventNames().keypressEvent) {
1490 if (!is<KeyboardEvent>(event))
1491 return;
1492 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
1493 int keyCode = keyboardEvent.keyCode();
1494
1495 if (keyCode == '\r') {
1496 if (form())
1497 form()->submitImplicitly(keyboardEvent, false);
1498 keyboardEvent.setDefaultHandled();
1499 } else if (m_multiple && keyCode == ' ' && m_allowsNonContiguousSelection) {
1500 // Use space to toggle selection change.
1501 m_activeSelectionState = !m_activeSelectionState;
1502 ASSERT(m_activeSelectionEndIndex >= 0);
1503 ASSERT(m_activeSelectionEndIndex < static_cast<int>(listItems.size()));
1504 ASSERT(is<HTMLOptionElement>(*listItems[m_activeSelectionEndIndex]));
1505 updateSelectedState(m_activeSelectionEndIndex, true /*multi*/, false /*shift*/);
1506 listBoxOnChange();
1507 keyboardEvent.setDefaultHandled();
1508 }
1509 }
1510}
