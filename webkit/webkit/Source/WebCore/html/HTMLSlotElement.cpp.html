55HTMLSlotElement::InsertedIntoAncestorResult HTMLSlotElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
56{
57 SetForScope isInInsertedIntoAncestor { m_isInInsertedIntoAncestor, true };
58
59 auto insertionResult = HTMLElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
60 ASSERT_UNUSED(insertionResult, insertionResult == InsertedIntoAncestorResult::Done);
61
62 if (insertionType.treeScopeChanged && isInShadowTree()) {
63 if (auto* shadowRoot = containingShadowRoot())
64 shadowRoot->addSlotElementByName(attributeWithoutSynchronization(nameAttr), *this);
65 }
66
67 return InsertedIntoAncestorResult::Done;
68}

70void HTMLSlotElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
71{
72 if (removalType.treeScopeChanged && oldParentOfRemovedTree.isInShadowTree()) {
73 auto* oldShadowRoot = oldParentOfRemovedTree.containingShadowRoot();
74 ASSERT(oldShadowRoot);
75 oldShadowRoot->removeSlotElementByName(attributeWithoutSynchronization(nameAttr), *this, oldParentOfRemovedTree);
76 }
77
78 HTMLElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
79}

101const Vector<WeakPtr<Node>>* HTMLSlotElement::assignedNodes() const
102{
103 auto shadowRoot = makeRefPtr(containingShadowRoot());
104 if (!shadowRoot)
105 return nullptr;
106
107 return shadowRoot->assignedNodesForSlot(*this);
108}
109
110static void flattenAssignedNodes(Vector<Ref<Node>>& nodes, const HTMLSlotElement& slot)
111{
112 if (!slot.containingShadowRoot())
113 return;
114
115 auto* assignedNodes = slot.assignedNodes();
116 if (!assignedNodes) {
117 for (RefPtr<Node> child = slot.firstChild(); child; child = child->nextSibling()) {
118 if (is<HTMLSlotElement>(*child))
119 flattenAssignedNodes(nodes, downcast<HTMLSlotElement>(*child));
120 else if (is<Text>(*child) || is<Element>(*child))
121 nodes.append(*child);
122 }
123 return;
124 }
125 for (auto& nodeWeakPtr : *assignedNodes) {
126 auto* node = nodeWeakPtr.get();
127 if (UNLIKELY(!node)) {
128 ASSERT_NOT_REACHED();
129 continue;
130 }
131 if (is<HTMLSlotElement>(*node) && downcast<HTMLSlotElement>(*node).containingShadowRoot())
132 flattenAssignedNodes(nodes, downcast<HTMLSlotElement>(*node));
133 else
134 nodes.append(*node);
135 }
136}

138Vector<Ref<Node>> HTMLSlotElement::assignedNodes(const AssignedNodesOptions& options) const
139{
140 if (options.flatten) {
141 if (!isInShadowTree())
142 return { };
143 Vector<Ref<Node>> nodes;
144 flattenAssignedNodes(nodes, *this);
145 return nodes;
146 }
147
148 if (auto* nodes = assignedNodes(); nodes) {
149 return WTF::compactMap(*nodes, [](auto& nodeWeakPtr) -> RefPtr<Node> {
150 return nodeWeakPtr.get();
151 });
152 }
153
154 return { };
155}
156
157Vector<Ref<Element>> HTMLSlotElement::assignedElements(const AssignedNodesOptions& options) const
158{
159 return WTF::compactMap(assignedNodes(options), [](auto&& node) -> RefPtr<Element> {
160 if (!is<Element>(node))
161 return nullptr;
162 return static_reference_cast<Element>(WTFMove(node));
163 });
164}

166void HTMLSlotElement::enqueueSlotChangeEvent()
167{
168 // https://dom.spec.whatwg.org/#signal-a-slot-change
169 if (m_inSignalSlotList)
170 return;
171 m_inSignalSlotList = true;
172 MutationObserver::enqueueSlotChangeEvent(*this);
173}
174
175void HTMLSlotElement::dispatchSlotChangeEvent()
176{
177 m_inSignalSlotList = false;
178
179 Ref<Event> event = Event::create(eventNames().slotchangeEvent, Event::CanBubble::Yes, Event::IsCancelable::No);
180 event->setTarget(this);
181 dispatchEvent(event);
182}
