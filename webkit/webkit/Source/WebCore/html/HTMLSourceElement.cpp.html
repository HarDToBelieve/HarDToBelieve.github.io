74Node::InsertedIntoAncestorResult HTMLSourceElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
75{
76 HTMLElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
77 RefPtr<Element> parent = parentElement();
78 if (parent == &parentOfInsertedTree) {
79#if ENABLE(VIDEO)
80 if (is<HTMLMediaElement>(*parent))
81 downcast<HTMLMediaElement>(*parent).sourceWasAdded(*this);
82 else
83#endif
84#if ENABLE(MODEL_ELEMENT)
85 if (is<HTMLModelElement>(*parent))
86 downcast<HTMLModelElement>(*parent).sourcesChanged();
87 else
88#endif
89 if (is<HTMLPictureElement>(*parent)) {
90 // The new source element only is a relevant mutation if it precedes any img element.
91 m_shouldCallSourcesChanged = true;
92 for (const Node* node = previousSibling(); node; node = node->previousSibling()) {
93 if (is<HTMLImageElement>(*node))
94 m_shouldCallSourcesChanged = false;
95 }
96 if (m_shouldCallSourcesChanged)
97 downcast<HTMLPictureElement>(*parent).sourcesChanged();
98 }
99 }
100 return InsertedIntoAncestorResult::Done;
101}

103void HTMLSourceElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
104{
105 HTMLElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
106 if (!parentNode() && is<Element>(oldParentOfRemovedTree)) {
107#if ENABLE(VIDEO)
108 if (is<HTMLMediaElement>(oldParentOfRemovedTree))
109 downcast<HTMLMediaElement>(oldParentOfRemovedTree).sourceWasRemoved(*this);
110 else
111#endif
112#if ENABLE(MODEL_ELEMENT)
113 if (is<HTMLModelElement>(oldParentOfRemovedTree))
114 downcast<HTMLModelElement>(oldParentOfRemovedTree).sourcesChanged();
115 else
116#endif
117 if (m_shouldCallSourcesChanged) {
118 downcast<HTMLPictureElement>(oldParentOfRemovedTree).sourcesChanged();
119 m_shouldCallSourcesChanged = false;
120 }
121 }
122}

124void HTMLSourceElement::scheduleErrorEvent()
125{
126 LOG(Media, "HTMLSourceElement::scheduleErrorEvent - %p", this);
127 if (m_errorEventTimer.isActive())
128 return;
129
130 m_errorEventTimer.startOneShot(0_s);
131}
132
133void HTMLSourceElement::cancelPendingErrorEvent()
134{
135 LOG(Media, "HTMLSourceElement::cancelPendingErrorEvent - %p", this);
136 m_errorEventTimer.stop();
137}

139void HTMLSourceElement::errorEventTimerFired()
140{
141 LOG(Media, "HTMLSourceElement::errorEventTimerFired - %p", this);

143}

155void HTMLSourceElement::suspend(ReasonForSuspension reason)
156{
157 // FIXME: Shouldn't this also stop the timer for PageWillBeSuspended?
158 if (reason == ReasonForSuspension::BackForwardCache) {
159 m_shouldRescheduleErrorEventOnResume = m_errorEventTimer.isActive();
160 m_errorEventTimer.stop();
161 }
162}
163
164void HTMLSourceElement::resume()
165{
166 if (m_shouldRescheduleErrorEventOnResume) {
167 m_errorEventTimer.startOneShot(0_s);
168 m_shouldRescheduleErrorEventOnResume = false;
169 }
170}
171
172void HTMLSourceElement::stop()
173{
174 cancelPendingErrorEvent();
175}

177void HTMLSourceElement::parseAttribute(const QualifiedName& name, const AtomString& value)
178{
179 HTMLElement::parseAttribute(name, value);
180 if (name == srcsetAttr || name == sizesAttr || name == mediaAttr || name == typeAttr) {
181 if (name == mediaAttr)
182 m_cachedParsedMediaAttribute = std::nullopt;
183 auto parent = makeRefPtr(parentNode());
184 if (m_shouldCallSourcesChanged)
185 downcast<HTMLPictureElement>(*parent).sourcesChanged();
186 }
187#if ENABLE(MODEL_ELEMENT)
188 if (name == srcAttr || name == typeAttr) {
189 RefPtr<Element> parent = parentElement();
190 if (is<HTMLModelElement>(parent))
191 downcast<HTMLModelElement>(*parent).sourcesChanged();
192 }
193#endif
194}

196const MediaQuerySet* HTMLSourceElement::parsedMediaAttribute(Document& document) const
197{
198 if (!m_cachedParsedMediaAttribute) {
199 RefPtr<const MediaQuerySet> parsedAttribute;
200 auto& value = attributeWithoutSynchronization(mediaAttr);
201 if (!value.isNull())
202 parsedAttribute = MediaQuerySet::create(value, MediaQueryParserContext(document));
203 m_cachedParsedMediaAttribute = WTFMove(parsedAttribute);
204 }
205 return m_cachedParsedMediaAttribute.value().get();
206}

208void HTMLSourceElement::attributeChanged(const QualifiedName& name, const AtomString& oldValue, const AtomString& newValue, AttributeModificationReason reason)
209{
210 if (name == widthAttr || name == heightAttr) {
211 if (RefPtr parent = parentNode(); is<HTMLPictureElement>(parent))
212 downcast<HTMLPictureElement>(*parent).sourceDimensionAttributesChanged(*this);
213 }
214 HTMLElement::attributeChanged(name, oldValue, newValue, reason);
215}
