53Ref<HTMLSummaryElement> HTMLSummaryElement::create(const QualifiedName& tagName, Document& document)
54{
55 Ref<HTMLSummaryElement> summary = adoptRef(*new HTMLSummaryElement(tagName, document));
56 summary->addShadowRoot(ShadowRoot::create(document, makeUnique<SummarySlotElement>()));
57 return summary;
58}

66RenderPtr<RenderElement> HTMLSummaryElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
67{
68 // <summary> elements with display:list-item should not be rendered as list items because they'd end up with two markers before the text (one from summary element and the other as a list item).
69 return RenderElement::createFor(*this, WTFMove(style), { RenderElement::ConstructBlockLevelRendererFor::ListItem, RenderElement::ConstructBlockLevelRendererFor::Inline, RenderElement::ConstructBlockLevelRendererFor::TableOrTablePart });
70}

72void HTMLSummaryElement::didAddUserAgentShadowRoot(ShadowRoot& root)
73{
74 root.appendChild(DetailsMarkerControl::create(document()));
75 root.appendChild(HTMLSlotElement::create(slotTag, document()));
76}

78RefPtr<HTMLDetailsElement> HTMLSummaryElement::detailsElement() const
79{
80 auto* parent = parentElement();
81 if (parent && is<HTMLDetailsElement>(*parent))
82 return downcast<HTMLDetailsElement>(parent);
83 // Fallback summary element is in the shadow tree.
84 auto* host = shadowHost();
85 if (host && is<HTMLDetailsElement>(*host))
86 return downcast<HTMLDetailsElement>(host);
87 return nullptr;
88}
89
90bool HTMLSummaryElement::isActiveSummary() const
91{
92 RefPtr<HTMLDetailsElement> details = detailsElement();
93 if (!details)
94 return false;
95 return details->isActiveSummary(*this);
96}

98static bool isClickableControl(EventTarget* target)
99{
100 if (!is<Element>(target))
101 return false;
102 auto& element = downcast<Element>(*target);
103 return is<HTMLFormControlElement>(element) || is<HTMLFormControlElement>(element.shadowHost());
104}

116void HTMLSummaryElement::defaultEventHandler(Event& event)
117{
118 if (isActiveSummary() && renderer()) {
119 if (event.type() == eventNames().DOMActivateEvent && !isClickableControl(event.target())) {
120 if (RefPtr<HTMLDetailsElement> details = detailsElement())
121 details->toggleOpen();
122 event.setDefaultHandled();
123 return;
124 }
125
126 if (is<KeyboardEvent>(event)) {
127 KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
128 if (keyboardEvent.type() == eventNames().keydownEvent && keyboardEvent.keyIdentifier() == "U+0020") {
129 setActive(true, true);
130 // No setDefaultHandled() - IE dispatches a keypress in this case.
131 return;
132 }
133 if (keyboardEvent.type() == eventNames().keypressEvent) {
134 switch (keyboardEvent.charCode()) {
135 case '\r':
136 dispatchSimulatedClick(&event);
137 keyboardEvent.setDefaultHandled();
138 return;
139 case ' ':
140 // Prevent scrolling down the page.
141 keyboardEvent.setDefaultHandled();
142 return;
143 }
144 }
145 if (keyboardEvent.type() == eventNames().keyupEvent && keyboardEvent.keyIdentifier() == "U+0020") {
146 if (active())
147 dispatchSimulatedClick(&event);
148 keyboardEvent.setDefaultHandled();
149 return;
150 }
151 }
152 }
153
154 HTMLElement::defaultEventHandler(event);
155}
156
157bool HTMLSummaryElement::willRespondToMouseClickEvents()
158{
159 if (isActiveSummary() && renderer())
160 return true;
161
162 return HTMLElement::willRespondToMouseClickEvents();
163}
