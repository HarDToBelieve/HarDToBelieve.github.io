45bool HTMLTablePartElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
46{
47 if (name == bgcolorAttr || name == backgroundAttr || name == valignAttr || name == heightAttr)
48 return true;
49 return HTMLElement::hasPresentationalHintsForAttribute(name);
50}
51
52void HTMLTablePartElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
53{
54 if (name == bgcolorAttr)
55 addHTMLColorToStyle(style, CSSPropertyBackgroundColor, value);
56 else if (name == backgroundAttr) {
57 String url = stripLeadingAndTrailingHTMLSpaces(value);
58 if (!url.isEmpty())
59 style.setProperty(CSSProperty(CSSPropertyBackgroundImage, CSSImageValue::create(document().completeURL(url), LoadedFromOpaqueSource::No)));
60 } else if (name == valignAttr) {
61 if (equalLettersIgnoringASCIICase(value, "top"))
62 addPropertyToPresentationalHintStyle(style, CSSPropertyVerticalAlign, CSSValueTop);
63 else if (equalLettersIgnoringASCIICase(value, "middle"))
64 addPropertyToPresentationalHintStyle(style, CSSPropertyVerticalAlign, CSSValueMiddle);
65 else if (equalLettersIgnoringASCIICase(value, "bottom"))
66 addPropertyToPresentationalHintStyle(style, CSSPropertyVerticalAlign, CSSValueBottom);
67 else if (equalLettersIgnoringASCIICase(value, "baseline"))
68 addPropertyToPresentationalHintStyle(style, CSSPropertyVerticalAlign, CSSValueBaseline);
69 else
70 addPropertyToPresentationalHintStyle(style, CSSPropertyVerticalAlign, value);
71 } else if (name == alignAttr) {
72 if (equalLettersIgnoringASCIICase(value, "middle") || equalLettersIgnoringASCIICase(value, "center"))
73 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, CSSValueWebkitCenter);
74 else if (equalLettersIgnoringASCIICase(value, "absmiddle"))
75 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, CSSValueCenter);
76 else if (equalLettersIgnoringASCIICase(value, "left"))
77 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, CSSValueWebkitLeft);
78 else if (equalLettersIgnoringASCIICase(value, "right"))
79 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, CSSValueWebkitRight);
80 else
81 addPropertyToPresentationalHintStyle(style, CSSPropertyTextAlign, value);
82 } else if (name == heightAttr) {
83 if (!value.isEmpty())
84 addHTMLLengthToStyle(style, CSSPropertyHeight, value);
85 } else
86 HTMLElement::collectPresentationalHintsForAttribute(name, value, style);
87}
