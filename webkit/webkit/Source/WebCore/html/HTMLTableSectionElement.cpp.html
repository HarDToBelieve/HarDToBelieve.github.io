61ExceptionOr<Ref<HTMLElement>> HTMLTableSectionElement::insertRow(int index)
62{
63 if (index < -1)
64 return Exception { IndexSizeError };
65 auto children = rows();
66 int numRows = children->length();
67 if (index > numRows)
68 return Exception { IndexSizeError };
69 auto row = HTMLTableRowElement::create(trTag, document());
70 ExceptionOr<void> result;
71 if (numRows == index || index == -1)
72 result = appendChild(row);
73 else
74 result = insertBefore(row, index < 1 ? firstChild() : children->item(index));
75 if (result.hasException())
76 return result.releaseException();
77 return Ref<HTMLElement> { WTFMove(row) };
78}
79
80ExceptionOr<void> HTMLTableSectionElement::deleteRow(int index)
81{
82 auto children = rows();
83 int numRows = children->length();
84 if (index == -1) {
85 if (!numRows)
86 return { };
87 index = numRows - 1;
88 }
89 if (index < 0 || index >= numRows)
90 return Exception { IndexSizeError };
91 return removeChild(*children->item(index));
92}
93
94int HTMLTableSectionElement::numRows() const
95{
96 auto rows = childrenOfType<HTMLTableRowElement>(*this);
97 return std::distance(rows.begin(), { });
98}

100Ref<HTMLCollection> HTMLTableSectionElement::rows()
101{
102 return ensureRareData().ensureNodeLists().addCachedCollection<GenericCachedHTMLCollection<CollectionTypeTraits<TSectionRows>::traversalType>>(*this, TSectionRows);
103}
