131String HTMLTextFormControlElement::strippedPlaceholder() const
132{
133 // According to the HTML5 specification, we need to remove CR and LF from
134 // the attribute value.
135 const AtomString& attributeValue = attributeWithoutSynchronization(placeholderAttr);
136 if (!attributeValue.contains(newlineCharacter) && !attributeValue.contains(carriageReturn))
137 return attributeValue;
138
139 StringBuilder stripped;
140 unsigned length = attributeValue.length();
141 stripped.reserveCapacity(length);
142 for (unsigned i = 0; i < length; ++i) {
143 UChar character = attributeValue[i];
144 if (character == newlineCharacter || character == carriageReturn)
145 continue;
146 stripped.append(character);
147 }
148 return stripped.toString();
149}

224ExceptionOr<void> HTMLTextFormControlElement::setRangeText(const String& replacement, unsigned start, unsigned end, const String& selectionMode)
225{
226 if (start > end)
227 return Exception { IndexSizeError };
228
229 String text = innerTextValue();
230 unsigned textLength = text.length();
231 unsigned replacementLength = replacement.length();
232 unsigned newSelectionStart = selectionStart();
233 unsigned newSelectionEnd = selectionEnd();
234
235 start = std::min(start, textLength);
236 end = std::min(end, textLength);
237
238 if (start < end)
239 text.replace(start, end - start, replacement);
240 else
241 text.insert(replacement, start);
242
243 setInnerTextValue(text);
244
245 // FIXME: This shouldn't need synchronous style update, or renderer at all.
246 if (!renderer())
247 document().updateStyleIfNeeded();
248
249 if (!renderer())
250 return { };
251
252 subtreeHasChanged();
253
254 if (equalLettersIgnoringASCIICase(selectionMode, "select")) {
255 newSelectionStart = start;
256 newSelectionEnd = start + replacementLength;
257 } else if (equalLettersIgnoringASCIICase(selectionMode, "start"))
258 newSelectionStart = newSelectionEnd = start;
259 else if (equalLettersIgnoringASCIICase(selectionMode, "end"))
260 newSelectionStart = newSelectionEnd = start + replacementLength;
261 else {
262 // Default is "preserve".
263 long delta = replacementLength - (end - start);
264
265 if (newSelectionStart > end)
266 newSelectionStart += delta;
267 else if (newSelectionStart > start)
268 newSelectionStart = start;
269
270 if (newSelectionEnd > end)
271 newSelectionEnd += delta;
272 else if (newSelectionEnd > start)
273 newSelectionEnd = start + replacementLength;
274 }
275
276 setSelectionRange(newSelectionStart, newSelectionEnd, SelectionHasNoDirection);
277
278 return { };
279}

292void HTMLTextFormControlElement::setSelectionRange(int start, int end, TextFieldSelectionDirection direction, SelectionRevealMode revealMode, const AXTextStateChangeIntent& intent)
293{
294 if (!isTextField())
295 return;
296
297 end = std::max(end, 0);
298 start = std::min(std::max(start, 0), end);
299
300 auto innerText = innerTextElement();
301 bool hasFocus = document().focusedElement() == this;
302 if (!hasFocus && innerText) {
303 if (!isConnected()) {
304 cacheSelection(start, end, direction);
305 return;
306 }
307
308 // FIXME: Removing this synchronous layout requires fixing setSelectionWithoutUpdatingAppearance not needing up-to-date style.
309 document().updateLayoutIgnorePendingStylesheets();
310
311 if (!isTextField())
312 return;
313
314 // Double-check our connected state after the layout update.
315 if (!isConnected()) {
316 cacheSelection(start, end, direction);
317 return;
318 }
319
320 // Double-check the state of innerTextElement after the layout.
321 innerText = innerTextElement();
322 auto* rendererTextControl = renderer();
323
324 if (innerText && rendererTextControl) {
325 if (rendererTextControl->style().visibility() == Visibility::Hidden || !innerText->renderBox() || !innerText->renderBox()->height()) {
326 cacheSelection(start, end, direction);
327 return;
328 }
329 }
330 }
331
332 Position startPosition = positionForIndex(innerText.get(), start);
333 Position endPosition;
334 if (start == end)
335 endPosition = startPosition;
336 else {
337 if (direction == SelectionHasBackwardDirection) {
338 endPosition = startPosition;
339 startPosition = positionForIndex(innerText.get(), end);
340 } else
341 endPosition = positionForIndex(innerText.get(), end);
342 }
343
344 if (RefPtr<Frame> frame = document().frame())
345 frame->selection().moveWithoutValidationTo(startPosition, endPosition, direction != SelectionHasNoDirection, !hasFocus, revealMode, intent);
346}

404static const AtomString& directionString(TextFieldSelectionDirection direction)
405{
406 static MainThreadNeverDestroyed<const AtomString> none("none", AtomString::ConstructFromLiteral);
407 static MainThreadNeverDestroyed<const AtomString> forward("forward", AtomString::ConstructFromLiteral);
408 static MainThreadNeverDestroyed<const AtomString> backward("backward", AtomString::ConstructFromLiteral);
409
410 switch (direction) {
411 case SelectionHasNoDirection:
412 return none;
413 case SelectionHasForwardDirection:
414 return forward;
415 case SelectionHasBackwardDirection:
416 return backward;
417 }
418
419 ASSERT_NOT_REACHED();
420 return none;
421}

455std::optional<SimpleRange> HTMLTextFormControlElement::selection() const
456{
457 if (!renderer() || !isTextField() || !hasCachedSelection())
458 return std::nullopt;
459
460 unsigned start = m_cachedSelectionStart;
461 unsigned end = m_cachedSelectionEnd;
462
463 ASSERT(start <= end);
464 auto innerText = innerTextElement();
465 if (!innerText)
466 return std::nullopt;
467
468 if (!innerText->firstChild())
469 return SimpleRange { { *innerText, 0 }, { *innerText, 0 } };
470
471 unsigned offset = 0;
472 Node* startNode = nullptr;
473 Node* endNode = nullptr;
474 for (RefPtr<Node> node = innerText->firstChild(); node; node = NodeTraversal::next(*node, innerText.get())) {
475 ASSERT(!node->firstChild());
476 ASSERT(node->isTextNode() || node->hasTagName(brTag));
477
478 unsigned length = is<Text>(*node) ? downcast<Text>(*node).length() : 1;
479
480 if (offset <= start && start <= offset + length)
481 setContainerAndOffsetForRange(node.get(), start - offset, startNode, start);
482
483 if (offset <= end && end <= offset + length) {
484 setContainerAndOffsetForRange(node.get(), end - offset, endNode, end);
485 break;
486 }
487
488 offset += length;
489 }
490
491 if (!startNode || !endNode)
492 return std::nullopt;
493
494 return SimpleRange { { *startNode, start }, { *endNode, end } };
495}

554static void stripTrailingNewline(StringBuilder& result)
555{
556 // Remove one trailing newline; there's always one that's collapsed out by rendering.
557 size_t size = result.length();
558 if (size && result[size - 1] == newlineCharacter)
559 result.shrink(size - 1);
560}

562static String innerTextValueFrom(TextControlInnerTextElement& innerText)
563{
564 StringBuilder result;
565 for (RefPtr<Node> node = innerText.firstChild(); node; node = NodeTraversal::next(*node, &innerText)) {
566 if (is<HTMLBRElement>(*node))
567 result.append(newlineCharacter);
568 else if (is<Text>(*node))
569 result.append(downcast<Text>(*node).data());
570 }
571 stripTrailingNewline(result);
572 return result.toString();

575void HTMLTextFormControlElement::setInnerTextValue(const String& value)
576{
577 LayoutDisallowedScope layoutDisallowedScope(LayoutDisallowedScope::Reason::PerformanceOptimization);
578 auto innerText = innerTextElement();
579 if (!innerText)
580 return;
581
582 ASSERT(isTextField());
583 String previousValue = innerTextValueFrom(*innerText);
584 bool textIsChanged = value != previousValue;
585 if (textIsChanged || !innerText->hasChildNodes()) {
586#if ENABLE(ACCESSIBILITY) && !PLATFORM(COCOA)
587 if (textIsChanged && renderer()) {
588 if (AXObjectCache* cache = document().existingAXObjectCache())
589 cache->postNotification(this, AXObjectCache::AXValueChanged, PostTarget::ObservableParent);
590 }
591#endif
592
593 {
594 // Events dispatched on the inner text element cannot execute arbitrary author scripts.
595 ScriptDisallowedScope::EventAllowedScope allowedScope(*userAgentShadowRoot());
596
597 innerText->setInnerText(value);
598
599 if (value.endsWith('\n') || value.endsWith('\r'))
600 innerText->appendChild(HTMLBRElement::create(document()));
601 }
602
603#if ENABLE(ACCESSIBILITY) && PLATFORM(COCOA)
604 if (textIsChanged && renderer()) {
605 if (AXObjectCache* cache = document().existingAXObjectCache())
606 cache->deferTextReplacementNotificationForTextControl(*this, previousValue);
607 }
608#endif
609 }
610
611 setFormControlValueMatchesRenderer(true);
612}

620static Position positionForIndex(TextControlInnerTextElement* innerText, unsigned index)
621{
622 unsigned remainingCharactersToMoveForward = index;
623 RefPtr<Node> lastBrOrText = innerText;
624 for (RefPtr<Node> node = innerText; node; node = NodeTraversal::next(*node, innerText)) {
625 if (node->hasTagName(brTag)) {
626 if (!remainingCharactersToMoveForward)
627 return positionBeforeNode(node.get());
628 remainingCharactersToMoveForward--;
629 lastBrOrText = node;
630 } else if (is<Text>(*node)) {
631 Text& text = downcast<Text>(*node);
632 if (remainingCharactersToMoveForward < text.length())
633 return Position(&text, remainingCharactersToMoveForward);
634 remainingCharactersToMoveForward -= text.length();
635 lastBrOrText = node;
636 }
637 }
638 return lastPositionInOrAfterNode(lastBrOrText.get());
639}

641unsigned HTMLTextFormControlElement::indexForPosition(const Position& passedPosition) const
642{
643 auto innerText = innerTextElement();
644 if (!innerText || !innerText->contains(passedPosition.anchorNode()) || passedPosition.isNull())
645 return 0;
646
647 if (positionBeforeNode(innerText.get()) == passedPosition)
648 return 0;
649
650 unsigned index = 0;
651 RefPtr<Node> startNode = passedPosition.computeNodeBeforePosition();
652 if (!startNode)
653 startNode = passedPosition.containerNode();
654 ASSERT(startNode);
655 ASSERT(innerText->contains(startNode.get()));
656
657 for (RefPtr<Node> node = startNode; node; node = NodeTraversal::previous(*node, innerText.get())) {
658 if (is<Text>(*node)) {
659 unsigned length = downcast<Text>(*node).length();
660 if (node == passedPosition.containerNode())
661 index += std::min<unsigned>(length, passedPosition.offsetInContainerNode());
662 else
663 index += length;
664 } else if (is<HTMLBRElement>(*node))
665 ++index;
666 }
667
668 unsigned length = innerTextValue().length();
669 index = std::min(index, length); // FIXME: We shouldn't have to call innerTextValue() just to ignore the last LF. See finishText.
670#if 0
671 // FIXME: This assertion code was never built, has bit rotted, and needs to be fixed before it can be enabled:
672 // https://bugs.webkit.org/show_bug.cgi?id=205706.
673#if ASSERT_ENABLED
674 VisiblePosition visiblePosition = passedPosition;
675 unsigned indexComputedByVisiblePosition = 0;
676 if (visiblePosition.isNotNull())
677 indexComputedByVisiblePosition = WebCore::indexForVisiblePosition(innerText, visiblePosition, false /* forSelectionPreservation */);
678 ASSERT(index == indexComputedByVisiblePosition);
679#endif
680#endif
681 return index;
682}

684static void getNextSoftBreak(LegacyRootInlineBox*& line, Node*& breakNode, unsigned& breakOffset)
685{
686 LegacyRootInlineBox* next;
687 for (; line; line = next) {
688 next = line->nextRootBox();
689 if (next && !line->endsWithBreak()) {
690 ASSERT(line->lineBreakObj());
691 breakNode = line->lineBreakObj()->node();
692 breakOffset = line->lineBreakPos();
693 line = next;
694 return;
695 }
696 }
697 breakNode = 0;
698 breakOffset = 0;
699}

701String HTMLTextFormControlElement::valueWithHardLineBreaks() const
702{
703 // FIXME: It's not acceptable to ignore the HardWrap setting when there is no renderer.
704 // While we have no evidence this has ever been a practical problem, it would be best to fix it some day.
705 if (!isTextField())
706 return value();
707
708 auto innerText = innerTextElement();
709 if (!innerText)
710 return value();
711
712 RenderTextControlInnerBlock* renderer = innerText->renderer();
713 if (!renderer)
714 return value();
715
716 Node* breakNode;
717 unsigned breakOffset;
718 LegacyRootInlineBox* line = renderer->firstRootBox();
719 if (!line)
720 return value();
721
722 getNextSoftBreak(line, breakNode, breakOffset);
723
724 StringBuilder result;
725 for (RefPtr<Node> node = innerText->firstChild(); node; node = NodeTraversal::next(*node, innerText.get())) {
726 if (is<HTMLBRElement>(*node))
727 result.append(newlineCharacter);
728 else if (is<Text>(*node)) {
729 String data = downcast<Text>(*node).data();
730 unsigned length = data.length();
731 unsigned position = 0;
732 while (breakNode == node && breakOffset <= length) {
733 if (breakOffset > position) {
734 result.appendSubstring(data, position, breakOffset - position);
735 position = breakOffset;
736 result.append(newlineCharacter);
737 }
738 getNextSoftBreak(line, breakNode, breakOffset);
739 }
740 result.appendSubstring(data, position, length - position);
741 }
742 while (breakNode == node)
743 getNextSoftBreak(line, breakNode, breakOffset);
744 }
745 stripTrailingNewline(result);
746 return result.toString();
747}

749HTMLTextFormControlElement* enclosingTextFormControl(const Position& position)
750{
751 ASSERT(position.isNull() || position.anchorType() == Position::PositionIsOffsetInAnchor
752 || position.containerNode() || !position.anchorNode()->shadowHost()
753 || hasShadowRootParent(*position.anchorNode()));
754
755 RefPtr<Node> container = position.containerNode();
756 if (!container)
757 return nullptr;
758 RefPtr<Element> ancestor = container->shadowHost();
759 return ancestor && ancestor->isTextField() ? downcast<HTMLTextFormControlElement>(ancestor.get()) : nullptr;
760}

762String HTMLTextFormControlElement::directionForFormData() const
763{
764 auto direction = [this] {
765 for (auto& element : lineageOfType<HTMLElement>(*this)) {
766 auto& value = element.attributeWithoutSynchronization(dirAttr);
767 if (equalLettersIgnoringASCIICase(value, "rtl"))
768 return TextDirection::RTL;
769 if (equalLettersIgnoringASCIICase(value, "ltr"))
770 return TextDirection::LTR;
771 if (equalLettersIgnoringASCIICase(value, "auto")) {
772 bool isAuto;
773 return element.directionalityIfhasDirAutoAttribute(isAuto);
774 }
775 }
776 return TextDirection::LTR;
777 }();
778
779 return direction == TextDirection::LTR ? "ltr"_s : "rtl"_s;
780}

798void HTMLTextFormControlElement::adjustInnerTextStyle(const RenderStyle& parentStyle, RenderStyle& textBlockStyle) const
799{
800 // The inner block, if present, always has its direction set to LTR,
801 // so we need to inherit the direction and unicode-bidi style from the element.
802 textBlockStyle.setDirection(parentStyle.direction());
803 textBlockStyle.setUnicodeBidi(parentStyle.unicodeBidi());
804
805 if (auto innerText = innerTextElement()) {
806 if (const StyleProperties* properties = innerText->presentationalHintStyle()) {
807 RefPtr<CSSValue> value = properties->getPropertyCSSValue(CSSPropertyWebkitUserModify);
808 if (is<CSSPrimitiveValue>(value))
809 textBlockStyle.setUserModify(downcast<CSSPrimitiveValue>(*value));
810 }
811 }
812
813 if (isDisabledFormControl())
814 textBlockStyle.setColor(RenderTheme::singleton().disabledTextColor(textBlockStyle.visitedDependentColorWithColorFilter(CSSPropertyColor), parentStyle.visitedDependentColorWithColorFilter(CSSPropertyBackgroundColor)));
815#if PLATFORM(IOS_FAMILY)
816 if (textBlockStyle.textSecurity() != TextSecurity::None && !textBlockStyle.isLeftToRightDirection()) {
817 // Preserve the alignment but force the direction to LTR so that the last-typed, unmasked character
818 // (which cannot have RTL directionality) will appear to the right of the masked characters. See <rdar://problem/7024375>.
819
820 switch (textBlockStyle.textAlign()) {
821 case TextAlignMode::Start:
822 case TextAlignMode::Justify:
823 textBlockStyle.setTextAlign(TextAlignMode::Right);
824 break;
825 case TextAlignMode::End:
826 textBlockStyle.setTextAlign(TextAlignMode::Left);
827 break;
828 case TextAlignMode::Left:
829 case TextAlignMode::Right:
830 case TextAlignMode::Center:
831 case TextAlignMode::WebKitLeft:
832 case TextAlignMode::WebKitRight:
833 case TextAlignMode::WebKitCenter:
834 break;
835 }
836
837 textBlockStyle.setDirection(TextDirection::LTR);
838 }
839#endif
840}
