39Ref<HTMLWBRElement> HTMLWBRElement::create(const QualifiedName& tagName, Document& document)
40{
41 return adoptRef(*new HTMLWBRElement(tagName, document));
42}
43
44HTMLWBRElement::HTMLWBRElement(const QualifiedName& tagName, Document& document)
45 : HTMLElement(tagName, document)
46{
47 ASSERT(hasTagName(wbrTag));
48}

50RenderPtr<RenderElement> HTMLWBRElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
51{
52 return createRenderer<RenderLineBreak>(*this, WTFMove(style));
53}
