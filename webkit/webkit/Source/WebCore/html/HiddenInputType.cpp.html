46const AtomString& HiddenInputType::formControlType() const
47{
48 return InputTypeNames::hidden();
49}

51FormControlState HiddenInputType::saveFormControlState() const
52{
53 // valueAttributeWasUpdatedAfterParsing() never be true for form controls create by createElement() or cloneNode().
54 // It's OK for now because we restore values only to form controls created by parsing.
55 ASSERT(element());
56 return element()->valueAttributeWasUpdatedAfterParsing() ? FormControlState { { element()->value() } } : FormControlState { };
57}
58
59void HiddenInputType::restoreFormControlState(const FormControlState& state)
60{
61 ASSERT(element());
62 element()->setAttributeWithoutSynchronization(valueAttr, state[0]);
63}

92bool HiddenInputType::appendFormData(DOMFormData& formData, bool isMultipartForm) const
93{
94 ASSERT(element());
95 auto name = element()->name();
96
97 if (equalIgnoringASCIICase(name, "_charset_")) {
98 formData.append(name, String { formData.encoding().name() });
99 return true;
100 }
101 return InputType::appendFormData(formData, isMultipartForm);
102}
