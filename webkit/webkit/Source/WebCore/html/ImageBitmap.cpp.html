81{
82 if (scriptExecutionContext.isDocument()) {
83 auto& document = downcast<Document>(scriptExecutionContext);

85 auto hostWindow = document.view()->root()->hostWindow();
86 return ImageBuffer::create(size, renderingMode, ShouldUseDisplayList::No, RenderingPurpose::Canvas, resolutionScale, DestinationColorSpace::SRGB(), PixelFormat::BGRA8, hostWindow);

111void ImageBitmap::createPromise(ScriptExecutionContext& scriptExecutionContext, ImageBitmap::Source&& source, ImageBitmapOptions&& options, int sx, int sy, int sw, int sh, ImageBitmap::Promise&& promise)
112{
113 // 1. If either the sw or sh arguments are specified but zero, return a promise
114 // rejected with an "RangeError" DOMException and abort these steps.
115 if (!sw || !sh) {
116 promise.reject(RangeError, "Cannot create ImageBitmap with a width or height of 0");
117 return;
118 }
119
120 auto left = sw >= 0 ? sx : sx + sw;
121 auto top = sh >= 0 ? sy : sy + sh;
122 auto width = std::abs(sw);
123 auto height = std::abs(sh);
124
125 WTF::switchOn(source,
126 [&] (auto& specificSource) {
127 createPromise(scriptExecutionContext, specificSource, WTFMove(options), IntRect { left, top, width, height }, WTFMove(promise));
128 }
129 );
130}

132static bool taintsOrigin(CachedImage& cachedImage)
133{
134 auto* image = cachedImage.image();
135 if (!image)
136 return false;
137
138 if (image->sourceURL().protocolIsData())
139 return false;
140
141 if (!image->hasSingleSecurityOrigin())
142 return true;
143
144 if (!cachedImage.isCORSSameOrigin())
145 return true;
146
147 return false;
148}
149
150#if ENABLE(VIDEO)
151static bool taintsOrigin(SecurityOrigin* origin, HTMLVideoElement& video)
152{
153 if (!video.hasSingleSecurityOrigin())
154 return true;
155
156 if (video.player()->didPassCORSAccessCheck())
157 return false;
158
159 auto url = video.currentSrc();
160 if (url.protocolIsData())
161 return false;
162
163 return !origin->canRequest(url);
164}

167// https://html.spec.whatwg.org/multipage/imagebitmap-and-animations.html#cropped-to-the-source-rectangle-with-formatting
168static ExceptionOr<IntRect> croppedSourceRectangleWithFormatting(IntSize inputSize, ImageBitmapOptions& options, std::optional<IntRect> rect)
169{
170 // 2. If either or both of resizeWidth and resizeHeight members of options are less
171 // than or equal to 0, then return a promise rejected with "InvalidStateError"
172 // DOMException and abort these steps.
173 if ((options.resizeWidth && options.resizeWidth.value() <= 0) || (options.resizeHeight && options.resizeHeight.value() <= 0))
174 return Exception { InvalidStateError, "Invalid resize dimensions" };
175
176 // 3. If sx, sy, sw and sh are specified, let sourceRectangle be a rectangle whose
177 // corners are the four points (sx, sy), (sx+sw, sy),(sx+sw, sy+sh), (sx,sy+sh).
178 // Otherwise let sourceRectangle be a rectangle whose corners are the four points
179 // (0,0), (width of input, 0), (width of input, height of input), (0, height of
180 // input).
181 auto sourceRectangle = rect.value_or(IntRect { 0, 0, inputSize.width(), inputSize.height() });
182
183 // 4. Clip sourceRectangle to the dimensions of input.
184 sourceRectangle.intersect(IntRect { 0, 0, inputSize.width(), inputSize.height() });
185
186 return { WTFMove(sourceRectangle) };
187}
188
189static IntSize outputSizeForSourceRectangle(IntRect sourceRectangle, ImageBitmapOptions& options)
190{
191 // 5. Let outputWidth be determined as follows:
192 auto outputWidth = [&] () -> int {
193 if (options.resizeWidth)
194 return options.resizeWidth.value();
195 if (options.resizeHeight)
196 return ceil(sourceRectangle.width() * static_cast<double>(options.resizeHeight.value()) / sourceRectangle.height());
197 return sourceRectangle.width();
198 }();
199
200 // 6. Let outputHeight be determined as follows:
201 auto outputHeight = [&] () -> int {
202 if (options.resizeHeight)
203 return options.resizeHeight.value();
204 if (options.resizeWidth)
205 return ceil(sourceRectangle.height() * static_cast<double>(options.resizeWidth.value()) / sourceRectangle.width());
206 return sourceRectangle.height();
207 }();
208
209 return { outputWidth, outputHeight };
210}

212static InterpolationQuality interpolationQualityForResizeQuality(ImageBitmapOptions::ResizeQuality resizeQuality)
213{
214 switch (resizeQuality) {
215 case ImageBitmapOptions::ResizeQuality::Pixelated:
216 return InterpolationQuality::DoNotInterpolate;
217 case ImageBitmapOptions::ResizeQuality::Low:
218 return InterpolationQuality::Default; // Low is the default.
219 case ImageBitmapOptions::ResizeQuality::Medium:
220 return InterpolationQuality::Medium;
221 case ImageBitmapOptions::ResizeQuality::High:
222 return InterpolationQuality::High;
223 }
224 ASSERT_NOT_REACHED();
225 return InterpolationQuality::Default;
226}

228static ImageOrientation imageOrientationForOrientation(ImageBitmapOptions::Orientation orientation)
229{
230 if (orientation == ImageBitmapOptions::Orientation::FlipY)
231 return ImageOrientation(ImageOrientation::OriginBottomLeft);
232 return ImageOrientation();

235static AlphaPremultiplication alphaPremultiplicationForPremultiplyAlpha(ImageBitmapOptions::PremultiplyAlpha premultiplyAlpha)
236{
237 // The default is to premultiply - this is the least surprising behavior.
238 if (premultiplyAlpha == ImageBitmapOptions::PremultiplyAlpha::None)
239 return AlphaPremultiplication::Unpremultiplied;
240 return AlphaPremultiplication::Premultiplied;

243void ImageBitmap::resolveWithBlankImageBuffer(ScriptExecutionContext& scriptExecutionContext, bool originClean, Promise&& promise)
244{
245 // Source rectangle likely doesn't intersect the source image.
246 // Behavior isn't well specified, but WPT tests expect no Promise rejection (and of course no crashes).
247 // Resolve Promise with a blank 1x1 ImageBitmap.
248 auto bitmapData = createImageBuffer(scriptExecutionContext, FloatSize(1, 1), bufferRenderingMode);
249
250 // 9. If the origin of image's image is not the same origin as the origin specified by the
251 // entry settings object, then set the origin-clean flag of the ImageBitmap object's
252 // bitmap to false.
253 OptionSet<SerializationState> serializationState;
254 if (originClean)
255 serializationState.add(SerializationState::OriginClean);
256
257 // 7. Create a new ImageBitmap object.
258 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData), serializationState));
259
260 // 10. Return a new promise, but continue running these steps in parallel.
261 // 11. Resolve the promise with the new ImageBitmap object as the value.
262 promise.resolve(WTFMove(imageBitmap));
263}
264
265// FIXME: More steps from https://html.spec.whatwg.org/multipage/imagebitmap-and-animations.html#cropped-to-the-source-rectangle-with-formatting
266
267// 7. Place input on an infinite transparent black grid plane, positioned so that its
268// top left corner is at the origin of the plane, with the x-coordinate increasing
269// to the right, and the y-coordinate increasing down, and with each pixel in the
270// input image data occupying a cell on the plane's grid.
271
272// 8. Let output be the rectangle on the plane denoted by sourceRectangle.
273
274// 9. Scale output to the size specified by outputWidth and outputHeight. The user
275// agent should use the value of the resizeQuality option to guide the choice of
276// scaling algorithm.
277
278// 10. If the value of the imageOrientation member of options is "flipY", output must
279// be flipped vertically, disregarding any image orientation metadata of the source
280// (such as EXIF metadata), if any.
281
282// 11. If image is an img element or a Blob object, let val be the value of the
283// colorSpaceConversion member of options, and then run these substeps:
284//
285// 1. If val is "default", the color space conversion behavior is implementation-specific,
286// and should be chosen according to the color space that the implementation uses for
287// drawing images onto the canvas.
288//
289// 2. If val is "none", output must be decoded without performing any color space
290// conversions. This means that the image decoding algorithm must ignore color profile
291// metadata embedded in the source data as well as the display device color profile.
292
293// 12. Let val be the value of premultiplyAlpha member of options, and then run these substeps:
294//
295// 1. If val is "default", the alpha premultiplication behavior is implementation-specific,
296// and should be chosen according to implementation deems optimal for drawing images
297// onto the canvas.
298//
299// 2. If val is "premultiply", the output that is not premultiplied by alpha must have its
300// color components multiplied by alpha and that is premultiplied by alpha must be left
301// untouched.
302//
303// 3. If val is "none", the output that is not premultiplied by alpha must be left untouched
304// and that is premultiplied by alpha must have its color components divided by alpha.
305
306// 13. Return output.
307
308void ImageBitmap::createPromise(ScriptExecutionContext& scriptExecutionContext, RefPtr<HTMLImageElement>& imageElement, ImageBitmapOptions&& options, std::optional<IntRect> rect, ImageBitmap::Promise&& promise)
309{
310 // 2. If image is not completely available, then return a promise rejected with
311 // an "InvalidStateError" DOMException and abort these steps.
312
313 auto* cachedImage = imageElement->cachedImage();
314 if (!cachedImage || !imageElement->complete()) {
315 promise.reject(InvalidStateError, "Cannot create ImageBitmap that is not completely available");
316 return;
317 }
318
319 // 3. If image's media data has no intrinsic dimensions (e.g. it's a vector graphic
320 // with no specified content size), and both or either of the resizeWidth and
321 // resizeHeight options are not specified, then return a promise rejected with
322 // an "InvalidStateError" DOMException and abort these steps.
323
324 auto imageSize = cachedImage->imageSizeForRenderer(imageElement->renderer(), 1.0f);
325 if ((!imageSize.width() || !imageSize.height()) && (!options.resizeWidth || !options.resizeHeight)) {
326 promise.reject(InvalidStateError, "Cannot create ImageBitmap from a source with no intrinsic size without providing resize dimensions");
327 return;
328 }
329
330 // 4. If image's media data has no intrinsic dimensions (e.g. it's a vector graphics
331 // with no specified content size), it should be rendered to a bitmap of the size
332 // specified by the resizeWidth and the resizeHeight options.
333
334 if (!imageSize.width() && !imageSize.height()) {
335 imageSize.setWidth(options.resizeWidth.value());
336 imageSize.setHeight(options.resizeHeight.value());
337 }
338
339 // 5. If the sw and sh arguments are not specified and image's media data has both or
340 // either of its intrinsic width and intrinsic height values equal to 0, then return
341 // a promise rejected with an "InvalidStateError" DOMException and abort these steps.
342 // 6. If the sh argument is not specified and image's media data has an intrinsic height
343 // of 0, then return a promise rejected with an "InvalidStateError" DOMException and
344 // abort these steps.
345
346 // FIXME: It's unclear how these steps can happen, since step 4 required setting a
347 // width and height for the image.
348
349 if (!rect && (!imageSize.width() || !imageSize.height())) {
350 promise.reject(InvalidStateError, "Cannot create ImageBitmap from a source with no intrinsic size without providing dimensions");
351 return;
352 }
353
354 // 8. Let the ImageBitmap object's bitmap data be a copy of image's media data, cropped to
355 // the source rectangle with formatting. If this is an animated image, the ImageBitmap
356 // object's bitmap data must only be taken from the default image of the animation (the
357 // one that the format defines is to be used when animation is not supported or is disabled),
358 // or, if there is no such image, the first frame of the animation.
359
360 auto sourceRectangle = croppedSourceRectangleWithFormatting(roundedIntSize(imageSize), options, WTFMove(rect));
361 if (sourceRectangle.hasException()) {
362 promise.reject(sourceRectangle.releaseException());
363 return;
364 }
365
366 auto outputSize = outputSizeForSourceRectangle(sourceRectangle.returnValue(), options);
367 auto bitmapData = createImageBuffer(scriptExecutionContext, outputSize, bufferRenderingMode);
368 auto imageForRender = cachedImage->imageForRenderer(imageElement->renderer());
369 if (!imageForRender) {
370 promise.reject(InvalidStateError, "Cannot create ImageBitmap from image that can't be rendered");
371 return;
372 }
373
374 if (!bitmapData) {
375 resolveWithBlankImageBuffer(scriptExecutionContext, !taintsOrigin(*cachedImage), WTFMove(promise));
376 return;
377 }
378
379 FloatRect destRect(FloatPoint(), outputSize);
380 bitmapData->context().drawImage(*imageForRender, destRect, sourceRectangle.releaseReturnValue(), { interpolationQualityForResizeQuality(options.resizeQuality), imageOrientationForOrientation(options.imageOrientation) });
381
382 // 9. If the origin of image's image is not the same origin as the origin specified by the
383 // entry settings object, then set the origin-clean flag of the ImageBitmap object's
384 // bitmap to false.
385 OptionSet<SerializationState> serializationState;
386 if (!taintsOrigin(*cachedImage))
387 serializationState.add(SerializationState::OriginClean);
388
389 if (alphaPremultiplicationForPremultiplyAlpha(options.premultiplyAlpha) == AlphaPremultiplication::Premultiplied)
390 serializationState.add(SerializationState::PremultiplyAlpha);
391
392 // 7. Create a new ImageBitmap object.
393 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData), serializationState));
394
395 // 10. Return a new promise, but continue running these steps in parallel.
396 // 11. Resolve the promise with the new ImageBitmap object as the value.
397
398 promise.resolve(WTFMove(imageBitmap));
399}

413void ImageBitmap::createPromise(ScriptExecutionContext& scriptExecutionContext, CanvasBase& canvas, ImageBitmapOptions&& options, std::optional<IntRect> rect, ImageBitmap::Promise&& promise)
414{
415 // 2. If the canvas element's bitmap has either a horizontal dimension or a vertical
416 // dimension equal to zero, then return a promise rejected with an "InvalidStateError"
417 // DOMException and abort these steps.
418 auto size = canvas.size();
419 if (!size.width() || !size.height()) {
420 promise.reject(InvalidStateError, "Cannot create ImageBitmap from a canvas that has zero width or height");
421 return;
422 }
423
424 // 4. Let the ImageBitmap object's bitmap data be a copy of the canvas element's bitmap
425 // data, cropped to the source rectangle with formatting.
426
427 auto sourceRectangle = croppedSourceRectangleWithFormatting(size, options, WTFMove(rect));
428 if (sourceRectangle.hasException()) {
429 promise.reject(sourceRectangle.releaseException());
430 return;
431 }
432
433 auto outputSize = outputSizeForSourceRectangle(sourceRectangle.returnValue(), options);
434 auto bitmapData = createImageBuffer(scriptExecutionContext, outputSize, bufferRenderingMode);
435
436 auto imageForRender = canvas.copiedImage();
437 if (!imageForRender) {
438 promise.reject(InvalidStateError, "Cannot create ImageBitmap from canvas that can't be rendered");
439 return;
440 }
441
442 if (!bitmapData) {
443 resolveWithBlankImageBuffer(scriptExecutionContext, canvas.originClean(), WTFMove(promise));
444 return;
445 }
446
447 FloatRect destRect(FloatPoint(), outputSize);
448 bitmapData->context().drawImage(*imageForRender, destRect, sourceRectangle.releaseReturnValue(), { interpolationQualityForResizeQuality(options.resizeQuality), imageOrientationForOrientation(options.imageOrientation) });
449
450 // 5. Set the origin-clean flag of the ImageBitmap object's bitmap to the same value as
451 // the origin-clean flag of the canvas element's bitmap.
452 OptionSet<SerializationState> serializationState;
453 if (canvas.originClean())
454 serializationState.add(SerializationState::OriginClean);
455
456 if (alphaPremultiplicationForPremultiplyAlpha(options.premultiplyAlpha) == AlphaPremultiplication::Premultiplied)
457 serializationState.add(SerializationState::PremultiplyAlpha);
458
459 // 3. Create a new ImageBitmap object.
460 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData), serializationState));
461
462 // 6. Return a new promise, but continue running these steps in parallel.
463 // 7. Resolve the promise with the new ImageBitmap object as the value.
464
465 promise.resolve(WTFMove(imageBitmap));
466}

468#if ENABLE(VIDEO)
469void ImageBitmap::createPromise(ScriptExecutionContext& scriptExecutionContext, RefPtr<HTMLVideoElement>& video, ImageBitmapOptions&& options, std::optional<IntRect> rect, ImageBitmap::Promise&& promise)
470{
471 // https://html.spec.whatwg.org/multipage/#dom-createimagebitmap
472 // WHATWG HTML 2102913b313078cd8eeac7e81e6a8756cbd3e773
473 // Steps 3-7.
474 // (Step 3 is handled in croppedSourceRectangleWithFormatting.)
475
476 // 4. Check the usability of the image argument. If this throws an exception
477 // or returns bad, then return p rejected with an "InvalidStateError"
478 // DOMException.
479 if (video->readyState() == HTMLMediaElement::HAVE_NOTHING || video->readyState() == HTMLMediaElement::HAVE_METADATA) {
480 promise.reject(InvalidStateError, "Cannot create ImageBitmap before the HTMLVideoElement has data");
481 return;
482 }
483
484 // 6.1. If image's networkState attribute is NETWORK_EMPTY, then return p
485 // rejected with an "InvalidStateError" DOMException.
486 if (video->networkState() == HTMLMediaElement::NETWORK_EMPTY) {
487 promise.reject(InvalidStateError, "Cannot create ImageBitmap before the HTMLVideoElement has data");
488 return;
489 }
490
491 // 6.2. Set imageBitmap's bitmap data to a copy of the frame at the current
492 // playback position, at the media resource's intrinsic width and
493 // intrinsic height (i.e., after any aspect-ratio correction has been
494 // applied), cropped to the source rectangle with formatting.
495 auto size = video->player() ? roundedIntSize(video->player()->naturalSize()) : IntSize();
496 auto maybeSourceRectangle = croppedSourceRectangleWithFormatting(size, options, WTFMove(rect));
497 if (maybeSourceRectangle.hasException()) {
498 promise.reject(maybeSourceRectangle.releaseException());
499 return;
500 }
501 auto sourceRectangle = maybeSourceRectangle.releaseReturnValue();
502
503 auto outputSize = outputSizeForSourceRectangle(sourceRectangle, options);
504
505 // FIXME: Add support for color spaces / pixel formats to ImageBitmap.
506 auto bitmapData = video->createBufferForPainting(outputSize, bufferRenderingMode, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
507 if (!bitmapData) {
508 resolveWithBlankImageBuffer(scriptExecutionContext, !taintsOrigin(scriptExecutionContext.securityOrigin(), *video), WTFMove(promise));
509 return;
510 }
511
512 {
513 GraphicsContext& c = bitmapData->context();
514 GraphicsContextStateSaver stateSaver(c);
515 c.clip(FloatRect(FloatPoint(), outputSize));
516 auto scaleX = float(outputSize.width()) / float(sourceRectangle.width());
517 auto scaleY = float(outputSize.height()) / float(sourceRectangle.height());
518 if (options.imageOrientation == ImageBitmapOptions::Orientation::FlipY) {
519 c.scale(FloatSize(scaleX, -scaleY));
520 c.translate(IntPoint(-sourceRectangle.location().x(), sourceRectangle.location().y() - outputSize.height()));
521 } else {
522 c.scale(FloatSize(scaleX, scaleY));
523 c.translate(-sourceRectangle.location());
524 }
525 video->paintCurrentFrameInContext(c, FloatRect(FloatPoint(), size));
526 }
527
528 // 6.3. If the origin of image's video is not same origin with entry
529 // settings object's origin, then set the origin-clean flag of
530 // image's bitmap to false.
531 OptionSet<SerializationState> serializationState;
532 if (!taintsOrigin(scriptExecutionContext.securityOrigin(), *video))
533 serializationState.add(SerializationState::OriginClean);
534
535 if (alphaPremultiplicationForPremultiplyAlpha(options.premultiplyAlpha) == AlphaPremultiplication::Premultiplied)
536 serializationState.add(SerializationState::PremultiplyAlpha);
537
538 // 5. Let imageBitmap be a new ImageBitmap object.
539 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData), serializationState));
540
541 // 6.4.1. Resolve p with imageBitmap.
542 promise.resolve(WTFMove(imageBitmap));
543}
544#endif

553void ImageBitmap::createPromise(ScriptExecutionContext& scriptExecutionContext, RefPtr<ImageBitmap>& existingImageBitmap, ImageBitmapOptions&& options, std::optional<IntRect> rect, ImageBitmap::Promise&& promise)
554{
555 // 2. If image's [[Detached]] internal slot value is true, return a promise
556 // rejected with an "InvalidStateError" DOMException and abort these steps.
557 if (existingImageBitmap->isDetached() || !existingImageBitmap->buffer()) {
558 promise.reject(InvalidStateError, "Cannot create ImageBitmap from a detached ImageBitmap");
559 return;
560 }
561
562 // 4. Let the ImageBitmap object's bitmap data be a copy of the image argument's
563 // bitmap data, cropped to the source rectangle with formatting.
564 auto sourceRectangle = croppedSourceRectangleWithFormatting(existingImageBitmap->buffer()->logicalSize(), options, WTFMove(rect));
565 if (sourceRectangle.hasException()) {
566 promise.reject(sourceRectangle.releaseException());
567 return;
568 }
569
570 auto outputSize = outputSizeForSourceRectangle(sourceRectangle.returnValue(), options);
571 auto bitmapData = createImageBuffer(scriptExecutionContext, outputSize, bufferRenderingMode);
572
573 if (!bitmapData) {
574 resolveWithBlankImageBuffer(scriptExecutionContext, existingImageBitmap->originClean(), WTFMove(promise));
575 return;
576 }
577
578 auto imageForRender = existingImageBitmap->buffer()->copyImage();
579
580 FloatRect destRect(FloatPoint(), outputSize);
581 bitmapData->context().drawImage(*imageForRender, destRect, sourceRectangle.releaseReturnValue(), { interpolationQualityForResizeQuality(options.resizeQuality), imageOrientationForOrientation(options.imageOrientation) });
582
583 // 5. Set the origin-clean flag of the ImageBitmap object's bitmap to the same
584 // value as the origin-clean flag of the bitmap of the image argument.
585 OptionSet<SerializationState> serializationState;
586 if (existingImageBitmap->originClean())
587 serializationState.add(SerializationState::OriginClean);
588
589 if (alphaPremultiplicationForPremultiplyAlpha(options.premultiplyAlpha) == AlphaPremultiplication::Premultiplied) {
590 serializationState.add(SerializationState::PremultiplyAlpha);
591
592 // At least in the Core Graphics backend, when creating an ImageBitmap from
593 // an ImageBitmap, the alpha channel of bitmapData isn't premultiplied even
594 // though the alpha mode of the internal surface claims it is. Instruct
595 // users of this ImageBitmap to ignore the internal surface's alpha mode.
596 serializationState.add(SerializationState::ForciblyPremultiplyAlpha);
597 }
598
599 // 3. Create a new ImageBitmap object.
600 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData), serializationState));
601
602 // 6. Return a new promise, but continue running these steps in parallel.
603 // 7. Resolve the promise with the new ImageBitmap object as the value.
604 promise.resolve(WTFMove(imageBitmap));
605}

731void ImageBitmap::createFromBuffer(ScriptExecutionContext& scriptExecutionContext, Ref<ArrayBuffer>&& arrayBuffer, String mimeType, long long expectedContentLength, const URL& sourceURL, ImageBitmapOptions&& options, std::optional<IntRect> rect, Promise&& promise)
732{
733 if (!arrayBuffer->byteLength()) {
734 promise.reject(InvalidStateError, "Cannot create an ImageBitmap from an empty buffer");
735 return;
736 }
737
738 auto sharedBuffer = SharedBuffer::create(static_cast<const char*>(arrayBuffer->data()), arrayBuffer->byteLength());
739 auto observer = ImageBitmapImageObserver::create(mimeType, expectedContentLength, sourceURL);
740 auto image = BitmapImage::create(observer.ptr());
741 auto result = image->setData(sharedBuffer.copyRef(), true);
742 if (result != EncodedDataStatus::Complete) {
743 promise.reject(InvalidStateError, "Cannot decode the data in the argument to createImageBitmap");
744 return;
745 }
746
747 auto sourceRectangle = croppedSourceRectangleWithFormatting(roundedIntSize(image->size()), options, rect);
748 if (sourceRectangle.hasException()) {
749 promise.reject(sourceRectangle.releaseException());
750 return;
751 }
752
753 auto outputSize = outputSizeForSourceRectangle(sourceRectangle.returnValue(), options);
754 auto bitmapData = createImageBuffer(scriptExecutionContext, outputSize, bufferRenderingMode);
755 if (!bitmapData) {
756 promise.reject(InvalidStateError, "Cannot create an image buffer from the argument to createImageBitmap");
757 return;
758 }
759
760 FloatRect destRect(FloatPoint(), outputSize);
761 bitmapData->context().drawImage(image, destRect, sourceRectangle.releaseReturnValue(), { interpolationQualityForResizeQuality(options.resizeQuality), imageOrientationForOrientation(options.imageOrientation) });
762
763 OptionSet<SerializationState> serializationState = SerializationState::OriginClean;
764 if (alphaPremultiplicationForPremultiplyAlpha(options.premultiplyAlpha) == AlphaPremultiplication::Premultiplied)
765 serializationState.add(SerializationState::PremultiplyAlpha);
766
767 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData), serializationState));
768
769 promise.resolve(WTFMove(imageBitmap));
770}

778void ImageBitmap::createPromise(ScriptExecutionContext& scriptExecutionContext, RefPtr<ImageData>& imageData, ImageBitmapOptions&& options, std::optional<IntRect> rect, ImageBitmap::Promise&& promise)
779{
780 // 6.1. Let buffer be image's data attribute value's [[ViewedArrayBuffer]]
781 // internal slot.
782 // 6.2. If IsDetachedBuffer(buffer) is true, then return p rejected with an
783 // "InvalidStateError" DOMException.
784 if (imageData->data().isDetached()) {
785 promise.reject(InvalidStateError, "ImageData's viewed buffer has been detached");
786 return;
787 }
788
789 // 6.3. Set imageBitmap's bitmap data to image's image data, cropped to the
790 // source rectangle with formatting.
791 auto sourceRectangle = croppedSourceRectangleWithFormatting(imageData->size(), options, WTFMove(rect));
792 if (sourceRectangle.hasException()) {
793 promise.reject(sourceRectangle.releaseException());
794 return;
795 }
796
797 auto outputSize = outputSizeForSourceRectangle(sourceRectangle.returnValue(), options);
798 auto bitmapData = createImageBuffer(scriptExecutionContext, outputSize, bufferRenderingMode);
799
800 if (!bitmapData) {
801 resolveWithBlankImageBuffer(scriptExecutionContext, true, WTFMove(promise));
802 return;
803 }
804
805 // If no cropping, resizing, flipping, etc. are needed, then simply use the
806 // resulting ImageBuffer directly.
807 auto alphaPremultiplication = alphaPremultiplicationForPremultiplyAlpha(options.premultiplyAlpha);
808 if (sourceRectangle.returnValue().location().isZero() && sourceRectangle.returnValue().size() == imageData->size() && sourceRectangle.returnValue().size() == outputSize && options.imageOrientation == ImageBitmapOptions::Orientation::None) {
809 bitmapData->putPixelBuffer(imageData->pixelBuffer(), sourceRectangle.releaseReturnValue(), { }, alphaPremultiplication);
810
811 auto imageBitmap = create(ImageBitmapBacking(WTFMove(bitmapData)));
812 // The result is implicitly origin-clean, and alpha premultiplication has already been handled.
813 promise.resolve(WTFMove(imageBitmap));
814 return;
815 }
816
817 // 6.3. Set imageBitmap's bitmap data to image's image data, cropped to the
818 // source rectangle with formatting.
819 auto tempBitmapData = createImageBuffer(scriptExecutionContext, imageData->size(), bufferRenderingMode);
820 if (!tempBitmapData) {
821 resolveWithBlankImageBuffer(scriptExecutionContext, true, WTFMove(promise));
822 return;
823 }
824 tempBitmapData->putPixelBuffer(imageData->pixelBuffer(), IntRect(0, 0, imageData->width(), imageData->height()), { }, alphaPremultiplication);
825 FloatRect destRect(FloatPoint(), outputSize);
826 bitmapData->context().drawImageBuffer(*tempBitmapData, destRect, sourceRectangle.releaseReturnValue(), { interpolationQualityForResizeQuality(options.resizeQuality), imageOrientationForOrientation(options.imageOrientation) });
827
828 // 6.4.1. Resolve p with ImageBitmap.
829 auto imageBitmap = create({ WTFMove(bitmapData) });
830 // The result is implicitly origin-clean, and alpha premultiplication has already been handled.
831 promise.resolve(WTFMove(imageBitmap));
832}
