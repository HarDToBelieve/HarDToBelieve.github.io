29namespace WebCore {
30
31ImageBitmapBacking::ImageBitmapBacking(RefPtr<ImageBuffer>&& bitmapData, OptionSet<SerializationState> serializationState)
32 : m_bitmapData(WTFMove(bitmapData))
33 , m_serializationState(serializationState)
34{
35 ASSERT(m_bitmapData);
36}
37
38ImageBuffer* ImageBitmapBacking::buffer() const
39{
40 return m_bitmapData.get();
41}
42
43RefPtr<ImageBuffer> ImageBitmapBacking::takeImageBuffer()
44{
45 return WTFMove(m_bitmapData);
46}
47
48unsigned ImageBitmapBacking::width() const
49{
50 // FIXME: Is this the right width?
51 return m_bitmapData ? m_bitmapData->logicalSize().width() : 0;
52}
53
54unsigned ImageBitmapBacking::height() const
55{
56 // FIXME: Is this the right height?
57 return m_bitmapData ? m_bitmapData->logicalSize().height() : 0;
58}
