57bool ImageInputType::appendFormData(DOMFormData& formData, bool) const
58{
59 ASSERT(element());
60 if (!element()->isActivatedSubmit())
61 return false;
62
63 auto& name = element()->name();
64 if (name.isEmpty()) {
65 formData.append("x"_s, String::number(m_clickLocation.x()));
66 formData.append("y"_s, String::number(m_clickLocation.y()));
67 return true;
68 }
69
70 formData.append(makeString(name, ".x"), String::number(m_clickLocation.x()));
71 formData.append(makeString(name, ".y"), String::number(m_clickLocation.y()));
72
73 auto value = element()->value();
74 if (!value.isEmpty())
75 formData.append(name, value);
76
77 return true;
78}

80void ImageInputType::handleDOMActivateEvent(Event& event)
81{
82 ASSERT(element());
83 Ref<HTMLInputElement> protectedElement(*element());
84 if (protectedElement->isDisabledFormControl() || !protectedElement->form())
85 return;
86
87 Ref<HTMLFormElement> protectedForm(*protectedElement->form());
88
89 protectedElement->setActivatedSubmit(true);
90
91 m_clickLocation = IntPoint();
92 if (event.underlyingEvent()) {
93 Event& underlyingEvent = *event.underlyingEvent();
94 if (is<MouseEvent>(underlyingEvent)) {
95 MouseEvent& mouseEvent = downcast<MouseEvent>(underlyingEvent);
96 if (!mouseEvent.isSimulated())
97 m_clickLocation = IntPoint(mouseEvent.offsetX(), mouseEvent.offsetY());
98 }
99 }
100
101 // Update layout before processing form actions in case the style changes
102 // the Form or button relationships.
103 protectedElement->document().updateLayoutIgnorePendingStylesheets();
104
105 if (auto currentForm = protectedElement->form())
106 currentForm->submitIfPossible(&event); // Event handlers can run.
107
108 protectedElement->setActivatedSubmit(false);
109 event.setDefaultHandled();
110}

118void ImageInputType::attributeChanged(const QualifiedName& name)
119{
120 if (name == altAttr) {
121 if (auto* element = this->element()) {
122 auto* renderer = element->renderer();
123 if (is<RenderImage>(renderer))
124 downcast<RenderImage>(*renderer).updateAltText();
125 }
126 } else if (name == srcAttr) {
127 if (auto* element = this->element()) {
128 if (element->renderer())
129 element->ensureImageLoader().updateFromElementIgnoringPreviousError();
130 }
131 }
132 BaseButtonInputType::attributeChanged(name);
133}

135void ImageInputType::attach()
136{
137 BaseButtonInputType::attach();
138
139 ASSERT(element());
140 HTMLImageLoader& imageLoader = element()->ensureImageLoader();
141 imageLoader.updateFromElement();
142
143 auto* renderer = downcast<RenderImage>(element()->renderer());
144 if (!renderer)
145 return;
146
147 if (imageLoader.hasPendingBeforeLoadEvent())
148 return;
149
150 auto& imageResource = renderer->imageResource();
151 imageResource.setCachedImage(imageLoader.image());
152
153 // If we have no image at all because we have no src attribute, set
154 // image height and width for the alt text instead.
155 if (!imageLoader.image() && !imageResource.cachedImage())
156 renderer->setImageSizeForAltText();
157}

174unsigned ImageInputType::height() const
175{
176 ASSERT(element());
177 Ref<HTMLInputElement> element(*this->element());
178
179 element->document().updateLayout();
180
181 if (auto* renderer = element->renderer())
182 return adjustForAbsoluteZoom(downcast<RenderBox>(*renderer).contentHeight(), *renderer);
183
184 // Check the attribute first for an explicit pixel value.
185 if (auto optionalHeight = parseHTMLNonNegativeInteger(element->attributeWithoutSynchronization(heightAttr)))
186 return optionalHeight.value();
187
188 // If the image is available, use its height.
189 auto* imageLoader = element->imageLoader();
190 if (imageLoader && imageLoader->image())
191 return imageLoader->image()->imageSizeForRenderer(element->renderer(), 1).height().toUnsigned();
192
193 return 0;
194}
