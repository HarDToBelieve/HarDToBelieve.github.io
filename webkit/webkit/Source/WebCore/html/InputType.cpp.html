98static InputTypeFactoryMap createInputTypeFactoryMap()
99{
100 static const struct InputTypes {
101 InputTypeConditionalFunction conditionalFunction;
102 InputTypeNameFunction nameFunction;
103 InputTypeFactoryFunction factoryFunction;
104 } inputTypes[] = {
105 { nullptr, &InputTypeNames::button, &createInputType<ButtonInputType> },
106 { nullptr, &InputTypeNames::checkbox, &createInputType<CheckboxInputType> },
107#if ENABLE(INPUT_TYPE_COLOR)
108 { &Settings::inputTypeColorEnabled, &InputTypeNames::color, &createInputType<ColorInputType> },
109#endif
110#if ENABLE(INPUT_TYPE_DATE)
111 { &Settings::inputTypeDateEnabled, &InputTypeNames::date, &createInputType<DateInputType> },
112#endif
113#if ENABLE(INPUT_TYPE_DATETIMELOCAL)
114 { &Settings::inputTypeDateTimeLocalEnabled, &InputTypeNames::datetimelocal, &createInputType<DateTimeLocalInputType> },
115#endif
116 { nullptr, &InputTypeNames::email, &createInputType<EmailInputType> },
117 { nullptr, &InputTypeNames::file, &createInputType<FileInputType> },
118 { nullptr, &InputTypeNames::hidden, &createInputType<HiddenInputType> },
119 { nullptr, &InputTypeNames::image, &createInputType<ImageInputType> },
120#if ENABLE(INPUT_TYPE_MONTH)
121 { &Settings::inputTypeMonthEnabled, &InputTypeNames::month, &createInputType<MonthInputType> },
122#endif
123 { nullptr, &InputTypeNames::number, &createInputType<NumberInputType> },
124 { nullptr, &InputTypeNames::password, &createInputType<PasswordInputType> },
125 { nullptr, &InputTypeNames::radio, &createInputType<RadioInputType> },
126 { nullptr, &InputTypeNames::range, &createInputType<RangeInputType> },
127 { nullptr, &InputTypeNames::reset, &createInputType<ResetInputType> },
128 { nullptr, &InputTypeNames::search, &createInputType<SearchInputType> },
129 { nullptr, &InputTypeNames::submit, &createInputType<SubmitInputType> },
130 { nullptr, &InputTypeNames::telephone, &createInputType<TelephoneInputType> },
131#if ENABLE(INPUT_TYPE_TIME)
132 { &Settings::inputTypeTimeEnabled, &InputTypeNames::time, &createInputType<TimeInputType> },
133#endif
134 { nullptr, &InputTypeNames::url, &createInputType<URLInputType> },
135#if ENABLE(INPUT_TYPE_WEEK)
136 { &Settings::inputTypeWeekEnabled, &InputTypeNames::week, &createInputType<WeekInputType> },
137#endif
138 // No need to register "text" because it is the default type.
139 };
140
141 InputTypeFactoryMap map;
142 for (auto& inputType : inputTypes)
143 map.add(inputType.nameFunction(), std::make_pair(inputType.conditionalFunction, inputType.factoryFunction));
144 return map;
145}

147Ref<InputType> InputType::create(HTMLInputElement& element, const AtomString& typeName)
148{
149 if (!typeName.isEmpty()) {
150 static const auto factoryMap = makeNeverDestroyed(createInputTypeFactoryMap());
151 auto&& [conditional, factory] = factoryMap.get().get(typeName);
152 if (factory && (!conditional || std::invoke(conditional, element.document().settings())))
153 return factory(element);
154 }
155 return adoptRef(*new TextInputType(element));
156}

261bool InputType::rangeUnderflow(const String& value) const
262{
263 if (!isSteppable())
264 return false;
265
266 const Decimal numericValue = parseToNumberOrNaN(value);
267 if (!numericValue.isFinite())
268 return false;
269
270 return numericValue < createStepRange(AnyStepHandling::Reject).minimum();
271}

388bool InputType::isInRange(const String& value) const
389{
390 if (!isSteppable())
391 return false;
392
393 StepRange stepRange(createStepRange(AnyStepHandling::Reject));
394 if (!stepRange.hasRangeLimitations())
395 return false;
396
397 const Decimal numericValue = parseToNumberOrNaN(value);
398 if (!numericValue.isFinite())
399 return true;
400
401 return numericValue >= stepRange.minimum() && numericValue <= stepRange.maximum();
402}
403
404bool InputType::isOutOfRange(const String& value) const
405{
406 if (!isSteppable() || value.isEmpty())
407 return false;
408
409 StepRange stepRange(createStepRange(AnyStepHandling::Reject));
410 if (!stepRange.hasRangeLimitations())
411 return false;
412
413 const Decimal numericValue = parseToNumberOrNaN(value);
414 if (!numericValue.isFinite())
415 return true;
416
417 return numericValue < stepRange.minimum() || numericValue > stepRange.maximum();
418}

448String InputType::validationMessage() const
449{
450 ASSERT(element());
451 String value = element()->value();
452
453 // The order of the following checks is meaningful. e.g. We'd like to show the
454 // badInput message even if the control has other validation errors.
455 if (hasBadInput())
456 return badInputText();
457
458 if (valueMissing(value))
459 return valueMissingText();
460
461 if (typeMismatch())
462 return typeMismatchText();
463
464 if (patternMismatch(value))
465 return validationMessagePatternMismatchText();
466
467 if (element()->tooShort())
468 return validationMessageTooShortText(numGraphemeClusters(value), element()->minLength());
469
470 if (element()->tooLong())
471 return validationMessageTooLongText(numGraphemeClusters(value), element()->effectiveMaxLength());
472
473 if (!isSteppable())
474 return emptyString();
475
476 const Decimal numericValue = parseToNumberOrNaN(value);
477 if (!numericValue.isFinite())
478 return emptyString();
479
480 StepRange stepRange(createStepRange(AnyStepHandling::Reject));
481
482 if (numericValue < stepRange.minimum())
483 return validationMessageRangeUnderflowText(serialize(stepRange.minimum()));
484
485 if (numericValue > stepRange.maximum())
486 return validationMessageRangeOverflowText(serialize(stepRange.maximum()));
487
488 if (stepRange.stepMismatch(numericValue)) {
489 const String stepString = stepRange.hasStep() ? serializeForNumberType(stepRange.step() / stepRange.stepScaleFactor()) : emptyString();
490 return validationMessageStepMismatchText(serialize(stepRange.stepBase()), stepString);
491 }
492
493 return emptyString();
494}

712void InputType::setValue(const String& sanitizedValue, bool valueChanged, TextFieldEventBehavior eventBehavior)
713{
714 ASSERT(element());
715 element()->setValueInternal(sanitizedValue, eventBehavior);
716 if (!valueChanged)
717 return;
718 element()->invalidateStyleForSubtree();
719
720 switch (eventBehavior) {
721 case DispatchChangeEvent:
722 element()->dispatchFormControlChangeEvent();
723 break;
724 case DispatchInputAndChangeEvent:
725 element()->dispatchFormControlInputEvent();
726 if (auto element = this->element())
727 element->dispatchFormControlChangeEvent();
728 break;
729 case DispatchNoEvent:
730 break;
731 }
732}

905ExceptionOr<void> InputType::applyStep(int count, AnyStepHandling anyStepHandling, TextFieldEventBehavior eventBehavior)
906{
907 StepRange stepRange(createStepRange(anyStepHandling));
908 if (!stepRange.hasStep())
909 return Exception { InvalidStateError };
910
911 ASSERT(element());
912 const Decimal current = parseToNumberOrNaN(element()->value());
913 if (!current.isFinite())
914 return Exception { InvalidStateError };
915 Decimal newValue = current + stepRange.step() * count;
916 if (!newValue.isFinite())
917 return Exception { InvalidStateError };
918
919 const Decimal acceptableErrorValue = stepRange.acceptableError();
920 if (newValue - stepRange.minimum() < -acceptableErrorValue)
921 return Exception { InvalidStateError };
922 if (newValue < stepRange.minimum())
923 newValue = stepRange.minimum();
924
925 if (!equalLettersIgnoringASCIICase(element()->attributeWithoutSynchronization(stepAttr), "any"))
926 newValue = stepRange.alignValueForStep(current, newValue);
927
928 if (newValue - stepRange.maximum() > acceptableErrorValue)
929 return Exception { InvalidStateError };
930 if (newValue > stepRange.maximum())
931 newValue = stepRange.maximum();
932
933 auto protectedThis = makeRef(*this);
934 auto result = setValueAsDecimal(newValue, eventBehavior);
935 if (result.hasException() || !element())
936 return result;
937
938 if (AXObjectCache* cache = element()->document().existingAXObjectCache())
939 cache->postNotification(element(), AXObjectCache::AXValueChanged);
940
941 return result;
942}

964void InputType::stepUpFromRenderer(int n)
965{
966 // The differences from stepUp()/stepDown():
967 //
968 // Difference 1: the current value
969 // If the current value is not a number, including empty, the current value is assumed as 0.
970 // * If 0 is in-range, and matches to step value
971 // - The value should be the +step if n > 0
972 // - The value should be the -step if n < 0
973 // If -step or +step is out of range, new value should be 0.
974 // * If 0 is smaller than the minimum value
975 // - The value should be the minimum value for any n
976 // * If 0 is larger than the maximum value
977 // - The value should be the maximum value for any n
978 // * If 0 is in-range, but not matched to step value
979 // - The value should be the larger matched value nearest to 0 if n > 0
980 // e.g. <input type=number min=-100 step=3> -> 2
981 // - The value should be the smaller matched value nearest to 0 if n < 0
982 // e.g. <input type=number min=-100 step=3> -> -1
983 // As for date/datetime-local/month/time/week types, the current value is assumed as "the current local date/time".
984 // As for datetime type, the current value is assumed as "the current date/time in UTC".
985 // If the current value is smaller than the minimum value:
986 // - The value should be the minimum value if n > 0
987 // - Nothing should happen if n < 0
988 // If the current value is larger than the maximum value:
989 // - The value should be the maximum value if n < 0
990 // - Nothing should happen if n > 0
991 //
992 // Difference 2: clamping steps
993 // If the current value is not matched to step value:
994 // - The value should be the larger matched value nearest to 0 if n > 0
995 // e.g. <input type=number value=3 min=-100 step=3> -> 5
996 // - The value should be the smaller matched value nearest to 0 if n < 0
997 // e.g. <input type=number value=3 min=-100 step=3> -> 2
998 //
999 // n is assumed as -n if step < 0.
1000
1001 ASSERT(isSteppable());
1002 if (!isSteppable())
1003 return;
1004 ASSERT(n);
1005 if (!n)
1006 return;
1007
1008 StepRange stepRange(createStepRange(AnyStepHandling::Default));
1009
1010 // FIXME: Not any changes after stepping, even if it is an invalid value, may be better.
1011 // (e.g. Stepping-up for <input type="number" value="foo" step="any" /> => "foo")
1012 if (!stepRange.hasStep())
1013 return;
1014
1015 EventQueueScope scope;
1016 const Decimal step = stepRange.step();
1017
1018 int sign;
1019 if (step > 0)
1020 sign = n;
1021 else if (step < 0)
1022 sign = -n;
1023 else
1024 sign = 0;
1025
1026 ASSERT(element());
1027 String currentStringValue = element()->value();
1028 Decimal current = parseToNumberOrNaN(currentStringValue);
1029 if (!current.isFinite()) {
1030 current = defaultValueForStepUp();
1031 const Decimal nextDiff = step * n;
1032 if (current < stepRange.minimum() - nextDiff)
1033 current = stepRange.minimum() - nextDiff;
1034 if (current > stepRange.maximum() - nextDiff)
1035 current = stepRange.maximum() - nextDiff;
1036 setValueAsDecimal(current, DispatchNoEvent);
1037 }
1038 if ((sign > 0 && current < stepRange.minimum()) || (sign < 0 && current > stepRange.maximum()))
1039 setValueAsDecimal(sign > 0 ? stepRange.minimum() : stepRange.maximum(), DispatchInputAndChangeEvent);
1040 else {
1041 if (stepMismatch(element()->value())) {
1042 ASSERT(!step.isZero());
1043 const Decimal base = stepRange.stepBase();
1044 Decimal newValue;
1045 if (sign < 0)
1046 newValue = base + ((current - base) / step).floor() * step;
1047 else if (sign > 0)
1048 newValue = base + ((current - base) / step).ceiling() * step;
1049 else
1050 newValue = current;
1051
1052 if (newValue < stepRange.minimum())
1053 newValue = stepRange.minimum();
1054 if (newValue > stepRange.maximum())
1055 newValue = stepRange.maximum();
1056
1057 setValueAsDecimal(newValue, n == 1 || n == -1 ? DispatchInputAndChangeEvent : DispatchNoEvent);
1058 if (n > 1)
1059 applyStep(n - 1, AnyStepHandling::Default, DispatchInputAndChangeEvent);
1060 else if (n < -1)
1061 applyStep(n + 1, AnyStepHandling::Default, DispatchInputAndChangeEvent);
1062 } else
1063 applyStep(n, AnyStepHandling::Default, DispatchInputAndChangeEvent);
1064 }
1065}
