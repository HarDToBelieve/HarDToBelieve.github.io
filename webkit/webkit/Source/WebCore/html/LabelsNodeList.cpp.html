37WTF_MAKE_ISO_ALLOCATED_IMPL(LabelsNodeList);
38
39LabelsNodeList::LabelsNodeList(LabelableElement& element)
40 : CachedLiveNodeList(element, InvalidateOnForTypeAttrChange)
41{
42}
43
44Ref<LabelsNodeList> LabelsNodeList::create(LabelableElement& element, const AtomString&)
45{
46 return adoptRef(*new LabelsNodeList(element));
47}
48
49LabelsNodeList::~LabelsNodeList()
50{
51 ownerNode().nodeLists()->removeCacheWithAtomName(*this, starAtom());
52}
53
54bool LabelsNodeList::elementMatches(Element& testNode) const
55{
56 return is<HTMLLabelElement>(testNode) && downcast<HTMLLabelElement>(testNode).control() == &ownerNode();
57}
