38class LazyFrameLoadIntersectionObserverCallback final : public IntersectionObserverCallback {
39public:
40 static Ref<LazyFrameLoadIntersectionObserverCallback> create(Document& document)
41 {
42 return adoptRef(*new LazyFrameLoadIntersectionObserverCallback(document));
43 }
44
45private:
46 CallbackResult<void> handleEvent(IntersectionObserver&, const Vector<Ref<IntersectionObserverEntry>>& entries, IntersectionObserver&) final
47 {
48 ASSERT(!entries.isEmpty());
49
50 for (auto& entry : entries) {
51 if (!entry->isIntersecting())
52 continue;
53 auto* element = entry->target();
54 if (is<HTMLIFrameElement>(element)) {
55 downcast<HTMLIFrameElement>(*element).lazyLoadFrameObserver().unobserve();
56 downcast<HTMLIFrameElement>(*element).loadDeferredFrame();
57 }
58 }
59 return { };
60 }
61
62 LazyFrameLoadIntersectionObserverCallback(Document& document)
63 : IntersectionObserverCallback(&document)
64 {
65 }
66};
67
68LazyLoadFrameObserver::LazyLoadFrameObserver(HTMLIFrameElement& element)
69 : m_element(element)
70{
71}
72
73void LazyLoadFrameObserver::observe(const AtomString& frameURL, const ReferrerPolicy& referrerPolicy)
74{
75 auto& frameObserver = m_element.lazyLoadFrameObserver();
76 auto* intersectionObserver = frameObserver.intersectionObserver(m_element.document());
77 if (!intersectionObserver)
78 return;
79 m_frameURL = frameURL;
80 m_referrerPolicy = referrerPolicy;
81 intersectionObserver->observe(m_element);
82}
83
84void LazyLoadFrameObserver::unobserve()
85{
86 auto& frameObserver = m_element.lazyLoadFrameObserver();
87 ASSERT(frameObserver.isObserved(m_element));
88 frameObserver.m_observer->unobserve(m_element);
89}
90
91IntersectionObserver* LazyLoadFrameObserver::intersectionObserver(Document& document)
92{
93 if (!m_observer) {
94 auto callback = LazyFrameLoadIntersectionObserverCallback::create(document);
95 IntersectionObserver::Init options { std::nullopt, emptyString(), { } };
96 auto observer = IntersectionObserver::create(document, WTFMove(callback), WTFMove(options));
97 if (observer.hasException())
98 return nullptr;
99 m_observer = observer.releaseReturnValue();
100 }
101 return m_observer.get();
102}
103
104bool LazyLoadFrameObserver::isObserved(Element& element) const
105{
106 return m_observer && m_observer->observationTargets().contains(&element);
107}
