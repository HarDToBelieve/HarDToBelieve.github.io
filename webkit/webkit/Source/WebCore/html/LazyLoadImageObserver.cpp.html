36namespace WebCore {
37
38class LazyImageLoadIntersectionObserverCallback final : public IntersectionObserverCallback {
39public:
40 static Ref<LazyImageLoadIntersectionObserverCallback> create(Document& document)
41 {
42 return adoptRef(*new LazyImageLoadIntersectionObserverCallback(document));
43 }
44
45private:
46 CallbackResult<void> handleEvent(IntersectionObserver&, const Vector<Ref<IntersectionObserverEntry>>& entries, IntersectionObserver&) final
47 {
48 ASSERT(!entries.isEmpty());
49
50 for (auto& entry : entries) {
51 if (!entry->isIntersecting())
52 continue;
53 auto* element = entry->target();
54 if (is<HTMLImageElement>(element)) {
55 downcast<HTMLImageElement>(*element).loadDeferredImage();
56 element->document().lazyLoadImageObserver().unobserve(*element, element->document());
57 }
58 }
59 return { };
60 }
61
62 LazyImageLoadIntersectionObserverCallback(Document& document)
63 : IntersectionObserverCallback(&document)
64 {
65 }
66};
67
68void LazyLoadImageObserver::observe(Element& element)
69{
70 auto& observer = element.document().lazyLoadImageObserver();
71 auto* intersectionObserver = observer.intersectionObserver(element.document());
72 if (!intersectionObserver)
73 return;
74 intersectionObserver->observe(element);
75}
76
77void LazyLoadImageObserver::unobserve(Element& element, Document& document)
78{
79 if (auto& observer = document.lazyLoadImageObserver().m_observer)
80 observer->unobserve(element);
81}
82
83IntersectionObserver* LazyLoadImageObserver::intersectionObserver(Document& document)
84{
85 if (!m_observer) {
86 auto callback = LazyImageLoadIntersectionObserverCallback::create(document);
87 IntersectionObserver::Init options { std::nullopt, emptyString(), { } };
88 auto observer = IntersectionObserver::create(document, WTFMove(callback), WTFMove(options));
89 if (observer.hasException())
90 return nullptr;
91 m_observer = observer.returnValue().ptr();
92 }
93 return m_observer.get();
94}
95
96bool LazyLoadImageObserver::isObserved(Element& element) const
97{
98 return m_observer && m_observer->observationTargets().contains(&element);
99}
