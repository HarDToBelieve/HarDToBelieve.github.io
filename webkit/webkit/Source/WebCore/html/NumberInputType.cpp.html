57struct RealNumberRenderSize {
58 unsigned sizeBeforeDecimalPoint;
59 unsigned sizeAfteDecimalPoint;
60
61 RealNumberRenderSize max(const RealNumberRenderSize& other) const
62 {
63 return {
64 std::max(sizeBeforeDecimalPoint, other.sizeBeforeDecimalPoint),
65 std::max(sizeAfteDecimalPoint, other.sizeAfteDecimalPoint)
66 };
67 }
68};
69
70static RealNumberRenderSize calculateRenderSize(const Decimal& value)
71{
72 ASSERT(value.isFinite());
73 const unsigned sizeOfDigits = String::number(value.value().coefficient()).length();
74 const unsigned sizeOfSign = value.isNegative() ? 1 : 0;
75 const int exponent = value.exponent();
76 if (exponent >= 0)
77 return { sizeOfSign + sizeOfDigits, 0 };
78
79 const int sizeBeforeDecimalPoint = exponent + sizeOfDigits;
80 if (sizeBeforeDecimalPoint > 0) {
81 // In case of "123.456"
82 return { sizeOfSign + sizeBeforeDecimalPoint, sizeOfDigits - sizeBeforeDecimalPoint };
83 }
84
85 // In case of "0.00012345"
86 const unsigned sizeOfZero = 1;
87 const unsigned numberOfZeroAfterDecimalPoint = -sizeBeforeDecimalPoint;
88 return { sizeOfSign + sizeOfZero , numberOfZeroAfterDecimalPoint + sizeOfDigits };
89}

110ExceptionOr<void> NumberInputType::setValueAsDouble(double newValue, TextFieldEventBehavior eventBehavior) const
111{
112 // FIXME: We should use numeric_limits<double>::max for number input type.
113 const double floatMax = std::numeric_limits<float>::max();
114 if (newValue < -floatMax || newValue > floatMax)
115 return Exception { InvalidStateError };
116 ASSERT(element());
117 element()->setValue(serializeForNumberType(newValue), eventBehavior);
118 return { };
119}

144StepRange NumberInputType::createStepRange(AnyStepHandling anyStepHandling) const
145{
146 static NeverDestroyed<const StepRange::StepDescription> stepDescription(numberDefaultStep, numberDefaultStepBase, numberStepScaleFactor);
147
148 ASSERT(element());
149 const Decimal stepBase = parseToDecimalForNumberType(element()->attributeWithoutSynchronization(minAttr), numberDefaultStepBase);
150 // FIXME: We should use numeric_limits<double>::max for number input type.
151 const Decimal floatMax = Decimal::fromDouble(std::numeric_limits<float>::max());
152 const Element& element = *this->element();
153
154 RangeLimitations rangeLimitations = RangeLimitations::Invalid;
155 auto extractBound = [&] (const QualifiedName& attributeName, const Decimal& defaultValue) -> Decimal {
156 const AtomString& attributeValue = element.attributeWithoutSynchronization(attributeName);
157 Decimal valueFromAttribute = parseToNumberOrNaN(attributeValue);
158 if (valueFromAttribute.isFinite()) {
159 rangeLimitations = RangeLimitations::Valid;
160 return valueFromAttribute;
161 }
162 return defaultValue;
163 };
164 Decimal minimum = extractBound(minAttr, -floatMax);
165 Decimal maximum = extractBound(maxAttr, floatMax);
166
167 const Decimal step = StepRange::parseStep(anyStepHandling, stepDescription, element.attributeWithoutSynchronization(stepAttr));
168 return StepRange(stepBase, rangeLimitations, minimum, maximum, step, stepDescription);
169}

171bool NumberInputType::sizeShouldIncludeDecoration(int defaultSize, int& preferredSize) const
172{
173 preferredSize = defaultSize;
174
175 ASSERT(element());
176 auto& stepString = element()->attributeWithoutSynchronization(stepAttr);
177 if (equalLettersIgnoringASCIICase(stepString, "any"))
178 return false;
179
180 const Decimal minimum = parseToDecimalForNumberType(element()->attributeWithoutSynchronization(minAttr));
181 if (!minimum.isFinite())
182 return false;
183
184 const Decimal maximum = parseToDecimalForNumberType(element()->attributeWithoutSynchronization(maxAttr));
185 if (!maximum.isFinite())
186 return false;
187
188 const Decimal step = parseToDecimalForNumberType(stepString, 1);
189 ASSERT(step.isFinite());
190
191 RealNumberRenderSize size = calculateRenderSize(minimum).max(calculateRenderSize(maximum).max(calculateRenderSize(step)));
192
193 preferredSize = size.sizeBeforeDecimalPoint + size.sizeAfteDecimalPoint + (size.sizeAfteDecimalPoint ? 1 : 0);
194
195 return true;
196}

198float NumberInputType::decorationWidth() const
199{
200 ASSERT(element());
201
202 float width = 0;
203 RefPtr<HTMLElement> spinButton = element()->innerSpinButtonElement();
204 if (RenderBox* spinRenderer = spinButton ? spinButton->renderBox() : 0) {
205 width += spinRenderer->borderAndPaddingLogicalWidth();
206 // Since the width of spinRenderer is not calculated yet, spinRenderer->logicalWidth() returns 0.
207 // So computedStyle()->logicalWidth() is used instead.
208 width += spinButton->computedStyle()->logicalWidth().value();
209 }
210 return width;
211}

233static bool isE(UChar ch)
234{
235 return ch == 'e' || ch == 'E';

238String NumberInputType::localizeValue(const String& proposedValue) const
239{
240 if (proposedValue.isEmpty())
241 return proposedValue;
242 // We don't localize scientific notations.
243 if (proposedValue.find(isE) != notFound)
244 return proposedValue;
245 ASSERT(element());
246 return element()->locale().convertToLocalizedNumber(proposedValue);
247}

266String NumberInputType::sanitizeValue(const String& proposedValue) const
267{
268 if (proposedValue.isEmpty())
269 return proposedValue;
270 return std::isfinite(parseToDoubleForNumberType(proposedValue)) ? proposedValue : emptyString();
271}

273bool NumberInputType::hasBadInput() const
274{
275 ASSERT(element());
276 String standardValue = convertFromVisibleValue(element()->innerTextValue());
277 return !standardValue.isEmpty() && !std::isfinite(parseToDoubleForNumberType(standardValue));

290void NumberInputType::attributeChanged(const QualifiedName& name)
291{
292 ASSERT(element());
293 if (name == maxAttr || name == minAttr) {
294 if (auto* element = this->element()) {
295 element->invalidateStyleForSubtree();
296 if (auto* renderer = element->renderer())
297 renderer->setNeedsLayoutAndPrefWidthsRecalc();
298 }
299 } else if (name == stepAttr) {
300 if (auto* element = this->element()) {
301 if (auto* renderer = element->renderer())
302 renderer->setNeedsLayoutAndPrefWidthsRecalc();
303 }
304 }
305 TextFieldInputType::attributeChanged(name);
306}
