93Ref<OffscreenCanvas> OffscreenCanvas::create(ScriptExecutionContext& scriptExecutionContext, std::unique_ptr<DetachedOffscreenCanvas>&& detachedCanvas)
94{
95 Ref<OffscreenCanvas> clone = adoptRef(*new OffscreenCanvas(scriptExecutionContext, detachedCanvas->size().width(), detachedCanvas->size().height()));
96 clone->setImageBuffer(detachedCanvas->takeImageBuffer());
97 if (!detachedCanvas->originClean())
98 clone->setOriginTainted();
99
100 callOnMainThread([detachedCanvas = WTFMove(detachedCanvas), placeholderData = makeRef(*clone->m_placeholderData)] () mutable {
101 placeholderData->canvas = detachedCanvas->takePlaceholderCanvas();
102 if (placeholderData->canvas) {
103 auto& placeholderContext = downcast<PlaceholderRenderingContext>(*placeholderData->canvas->renderingContext());
104 auto& imageBufferPipe = placeholderContext.imageBufferPipe();
105 if (imageBufferPipe)
106 placeholderData->bufferPipeSource = imageBufferPipe->source();
107 }
108 });
109
110 return clone;
111}

179static bool shouldEnableWebGL(bool webGLEnabled, bool acceleratedCompositingEnabled)
180{
181 if (!webGLEnabled)
182 return false;
183
184 if (!requiresAcceleratedCompositingForWebGL())
185 return true;
186

190void OffscreenCanvas::createContextWebGL(RenderingContextType contextType, WebGLContextAttributes&& attrs)
191{
192 ASSERT(!m_context);
193
194 auto scriptExecutionContext = this->scriptExecutionContext();
195 if (scriptExecutionContext->isWorkerGlobalScope()) {
196 WorkerGlobalScope& workerGlobalScope = downcast<WorkerGlobalScope>(*scriptExecutionContext);
197 if (!shouldEnableWebGL(workerGlobalScope.settingsValues().webGLEnabled, workerGlobalScope.settingsValues().acceleratedCompositingEnabled))
198 return;
199 } else if (scriptExecutionContext->isDocument()) {
200 auto& settings = downcast<Document>(*scriptExecutionContext).settings();
201 if (!shouldEnableWebGL(settings.webGLEnabled(), settings.acceleratedCompositingEnabled()))
202 return;
203 } else
204 return;
205 GraphicsContextGLWebGLVersion webGLVersion = GraphicsContextGLWebGLVersion::WebGL1;
206#if ENABLE(WEBGL2)
207 webGLVersion = (contextType == RenderingContextType::Webgl) ? GraphicsContextGLWebGLVersion::WebGL1 : GraphicsContextGLWebGLVersion::WebGL2;
208#else
209 UNUSED_PARAM(contextType);
210#endif
211 m_context = WebGLRenderingContextBase::create(*this, attrs, webGLVersion);
212}
213
214#endif // ENABLE(WEBGL)

216ExceptionOr<std::optional<OffscreenRenderingContext>> OffscreenCanvas::getContext(JSC::JSGlobalObject& state, RenderingContextType contextType, Vector<JSC::Strong<JSC::Unknown>>&& arguments)
217{
218 if (m_detached)
219 return Exception { InvalidStateError };
220
221 if (contextType == RenderingContextType::_2d) {
222 if (m_context) {
223 if (!is<OffscreenCanvasRenderingContext2D>(*m_context))
224 return { { std::nullopt } };
225 return { { RefPtr<OffscreenCanvasRenderingContext2D> { &downcast<OffscreenCanvasRenderingContext2D>(*m_context) } } };
226 }
227
228 auto scope = DECLARE_THROW_SCOPE(state.vm());
229 auto settings = convert<IDLDictionary<CanvasRenderingContext2DSettings>>(state, !arguments.isEmpty() ? arguments[0].get() : JSC::jsUndefined());
230 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
231
232 m_context = makeUnique<OffscreenCanvasRenderingContext2D>(*this, WTFMove(settings));
233 if (!m_context)
234 return { { std::nullopt } };
235
236 return { { RefPtr<OffscreenCanvasRenderingContext2D> { &downcast<OffscreenCanvasRenderingContext2D>(*m_context) } } };
237 }
238#if ENABLE(WEBGL)
239 else {
240 if (m_context) {
241 if (is<WebGLRenderingContext>(*m_context))
242 return { { RefPtr<WebGLRenderingContext> { &downcast<WebGLRenderingContext>(*m_context) } } };
243#if ENABLE(WEBGL2)
244 if (is<WebGL2RenderingContext>(*m_context))
245 return { { RefPtr<WebGL2RenderingContext> { &downcast<WebGL2RenderingContext>(*m_context) } } };
246#endif
247 return { { std::nullopt } };
248 }
249
250 auto scope = DECLARE_THROW_SCOPE(state.vm());
251 auto attributes = convert<IDLDictionary<WebGLContextAttributes>>(state, !arguments.isEmpty() ? arguments[0].get() : JSC::jsUndefined());
252 RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
253
254 createContextWebGL(contextType, WTFMove(attributes));
255 if (!m_context)
256 return { { std::nullopt } };
257
258#if ENABLE(WEBGL2)
259 if (is<WebGL2RenderingContext>(*m_context))
260 return { { RefPtr<WebGL2RenderingContext> { &downcast<WebGL2RenderingContext>(*m_context) } } };
261#endif
262 return { { RefPtr<WebGLRenderingContext> { &downcast<WebGLRenderingContext>(*m_context) } } };
263 }
264#endif
265
266 return Exception { TypeError };
267}

269ExceptionOr<RefPtr<ImageBitmap>> OffscreenCanvas::transferToImageBitmap()
270{
271 if (m_detached || !m_context)
272 return Exception { InvalidStateError };
273
274 if (is<OffscreenCanvasRenderingContext2D>(*m_context)) {
275 if (!width() || !height())
276 return { RefPtr<ImageBitmap> { nullptr } };
277
278 if (!m_hasCreatedImageBuffer) {
279 auto buffer = ImageBitmap::createImageBuffer(*canvasBaseScriptExecutionContext(), size(), RenderingMode::Unaccelerated);
280 return { ImageBitmap::create(ImageBitmapBacking(WTFMove(buffer))) };
281 }
282
283 // As the canvas context state is stored in GraphicsContext, which is owned
284 // by buffer(), to avoid resetting the context state, we have to make a copy and
285 // clear the original buffer rather than returning the original buffer.
286 auto bufferCopy = buffer()->copyRectToBuffer(FloatRect(FloatPoint(), buffer()->logicalSize()), buffer()->colorSpace(), *drawingContext());
287 downcast<OffscreenCanvasRenderingContext2D>(*m_context).clearCanvas();
288 clearCopiedImage();
289
290 return { ImageBitmap::create(ImageBitmapBacking(WTFMove(bufferCopy), originClean() ? SerializationState::OriginClean : SerializationState())) };
291 }
292
293#if ENABLE(WEBGL)
294 if (is<WebGLRenderingContext>(*m_context)) {
295 auto webGLContext = &downcast<WebGLRenderingContext>(*m_context);
296
297 // FIXME: We're supposed to create an ImageBitmap using the backing
298 // store from this canvas (or its context), but for now we'll just
299 // create a new bitmap and paint into it.
300
301 auto imageBitmap = ImageBitmap::create(*canvasBaseScriptExecutionContext(), size());
302 if (!imageBitmap->buffer())
303 return { RefPtr<ImageBitmap> { nullptr } };
304
305 auto* gc3d = webGLContext->graphicsContextGL();
306 gc3d->paintRenderingResultsToCanvas(*imageBitmap->buffer());
307
308 // FIXME: The transfer algorithm requires that the canvas effectively
309 // creates a new backing store. Since we're not doing that yet, we
310 // need to erase what's there.
311
312 GCGLfloat clearColor[4] { };
313 gc3d->getFloatv(GraphicsContextGL::COLOR_CLEAR_VALUE, clearColor);
314 gc3d->clearColor(0, 0, 0, 0);
315 gc3d->clear(GraphicsContextGL::COLOR_BUFFER_BIT | GraphicsContextGL::DEPTH_BUFFER_BIT | GraphicsContextGL::STENCIL_BUFFER_BIT);
316 gc3d->clearColor(clearColor[0], clearColor[1], clearColor[2], clearColor[3]);
317
318 return { WTFMove(imageBitmap) };
319 }
320#endif
321
322 return Exception { NotSupportedError };
323}

340void OffscreenCanvas::convertToBlob(ImageEncodeOptions&& options, Ref<DeferredPromise>&& promise)
341{
342 if (!originClean()) {
343 promise->reject(SecurityError);
344 return;
345 }
346 if (size().isEmpty()) {
347 promise->reject(IndexSizeError);
348 return;
349 }
350 if (m_detached || !buffer()) {
351 promise->reject(InvalidStateError);
352 return;
353 }
354
355 makeRenderingResultsAvailable();
356
357 auto encodingMIMEType = toEncodingMimeType(options.type);
358 auto quality = qualityFromDouble(options.quality);
359
360 Vector<uint8_t> blobData = buffer()->toData(encodingMIMEType, quality);
361 if (blobData.isEmpty()) {
362 promise->reject(EncodingError);
363 return;
364 }
365
366 Ref<Blob> blob = Blob::create(canvasBaseScriptExecutionContext(), WTFMove(blobData), encodingMIMEType);
367 promise->resolveWithNewlyCreated<IDLInterface<Blob>>(WTFMove(blob));
368}

377Image* OffscreenCanvas::copiedImage() const
378{
379 if (m_detached)
380 return nullptr;
381
382 if (!m_copiedImage && buffer()) {
383 if (m_context)
384 m_context->paintRenderingResultsToCanvas();
385 m_copiedImage = buffer()->copyImage(CopyBackingStore, PreserveResolution::Yes);
386 }
387 return m_copiedImage.get();
388}

409std::unique_ptr<DetachedOffscreenCanvas> OffscreenCanvas::detach()
410{
411 if (!canDetach())
412 return nullptr;
413
414 m_detached = true;
415
416 auto detached = makeUnique<DetachedOffscreenCanvas>(takeImageBuffer(), size(), originClean());
417 detached->m_placeholderCanvas = std::exchange(m_placeholderData->canvas, nullptr);
418
419 return detached;
420}

422void OffscreenCanvas::setPlaceholderCanvas(HTMLCanvasElement& canvas)
423{
424 ASSERT(!m_context);
425 ASSERT(isMainThread());
426 m_placeholderData->canvas = makeWeakPtr(canvas);
427 auto& placeholderContext = downcast<PlaceholderRenderingContext>(*canvas.renderingContext());
428 auto& imageBufferPipe = placeholderContext.imageBufferPipe();
429 if (imageBufferPipe)
430 m_placeholderData->bufferPipeSource = imageBufferPipe->source();
431}

433void OffscreenCanvas::pushBufferToPlaceholder()
434{
435 callOnMainThread([placeholderData = makeRef(*m_placeholderData)] () mutable {
436 Locker locker { placeholderData->bufferLock };
437 if (placeholderData->canvas && placeholderData->pendingCommitBuffer)
438 placeholderData->canvas->setImageBufferAndMarkDirty(WTFMove(placeholderData->pendingCommitBuffer));
439 placeholderData->pendingCommitBuffer = nullptr;
440 });
441}
442
443void OffscreenCanvas::commitToPlaceholderCanvas()
444{
445 auto* imageBuffer = buffer();
446 if (!imageBuffer)
447 return;
448
449 // FIXME: Transfer texture over if we're using accelerated compositing
450 if (m_context && (m_context->isWebGL() || m_context->isAccelerated()))
451 m_context->paintRenderingResultsToCanvas();
452
453 if (m_placeholderData->bufferPipeSource) {
454 auto bufferCopy = imageBuffer->copyRectToBuffer(FloatRect(FloatPoint(), imageBuffer->logicalSize()), DestinationColorSpace::SRGB(), imageBuffer->context());
455 if (bufferCopy)
456 m_placeholderData->bufferPipeSource->handle(WTFMove(bufferCopy));
457 }
458
459 Locker locker { m_placeholderData->bufferLock };
460 bool shouldPushBuffer = !m_placeholderData->pendingCommitBuffer;
461 m_placeholderData->pendingCommitBuffer = imageBuffer->copyRectToBuffer(FloatRect(FloatPoint(), imageBuffer->logicalSize()), DestinationColorSpace::SRGB(), imageBuffer->context());
462 if (m_placeholderData->pendingCommitBuffer && shouldPushBuffer)
463 pushBufferToPlaceholder();
464}

466void OffscreenCanvas::scheduleCommitToPlaceholderCanvas()
467{
468 if (!m_hasScheduledCommit) {
469 auto& scriptContext = *scriptExecutionContext();
470 m_hasScheduledCommit = true;
471 scriptContext.postTask([protectedThis = makeRef(*this), this] (ScriptExecutionContext&) {
472 m_hasScheduledCommit = false;
473 commitToPlaceholderCanvas();
474 });
475 }
476}

488void OffscreenCanvas::createImageBuffer() const
489{
490 m_hasCreatedImageBuffer = true;
491
492 if (!width() || !height())
493 return;
494
495 setImageBuffer(ImageBitmap::createImageBuffer(*canvasBaseScriptExecutionContext(), size(), RenderingMode::Unaccelerated));

498RefPtr<ImageBuffer> OffscreenCanvas::takeImageBuffer() const
499{
500 ASSERT(m_detached);
501
502 if (size().isEmpty())
503 return nullptr;
504
505 clearCopiedImage();
506 return setImageBuffer(nullptr);
507}

509void OffscreenCanvas::reset()
510{
511 resetGraphicsContextState();
512 if (is<OffscreenCanvasRenderingContext2D>(m_context.get()))
513 downcast<OffscreenCanvasRenderingContext2D>(*m_context).reset();
514
515 m_hasCreatedImageBuffer = false;
516 setImageBuffer(nullptr);
517 clearCopiedImage();
518
519 notifyObserversCanvasResized();
520 scheduleCommitToPlaceholderCanvas();
521}
