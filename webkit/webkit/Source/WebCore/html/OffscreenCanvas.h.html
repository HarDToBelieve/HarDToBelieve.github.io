73class DetachedOffscreenCanvas {
74 WTF_MAKE_NONCOPYABLE(DetachedOffscreenCanvas);
75 WTF_MAKE_FAST_ALLOCATED;
76 friend class OffscreenCanvas;
77
78public:
79 DetachedOffscreenCanvas(RefPtr<ImageBuffer>&&, const IntSize&, bool originClean);
80
81 RefPtr<ImageBuffer> takeImageBuffer();
82 const IntSize& size() const { return m_size; }
83 bool originClean() const { return m_originClean; }
84 size_t memoryCost() const
85 {
86 auto* buffer = m_buffer.get();
87 if (buffer)
88 return buffer->memoryCost();
89 return 0;
90 }
91 WeakPtr<HTMLCanvasElement> takePlaceholderCanvas();
92
93private:
94 RefPtr<ImageBuffer> m_buffer;
95 IntSize m_size;
96 bool m_originClean;
97 WeakPtr<HTMLCanvasElement> m_placeholderCanvas;
98};
