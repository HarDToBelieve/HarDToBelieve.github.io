40namespace WebCore {
41
42const AtomString& PasswordInputType::formControlType() const
43{
44 return InputTypeNames::password();
45}
46
47bool PasswordInputType::shouldSaveAndRestoreFormControlState() const
48{
49 return false;
50}
51
52FormControlState PasswordInputType::saveFormControlState() const
53{
54 // Should never save/restore password fields.
55 ASSERT_NOT_REACHED();
56 return FormControlState();
57}
58
59void PasswordInputType::restoreFormControlState(const FormControlState&)
60{
61 // Should never save/restore password fields.
62 ASSERT_NOT_REACHED();
63}
64
65bool PasswordInputType::shouldUseInputMethod() const
66{
67#if PLATFORM(GTK) || PLATFORM(WPE)
68 // Input methods are enabled for the password field in GTK and WPE ports
69 // because the input methods are notified that the active editable element
70 // is a password field.
71 return true;
72#else
73 // Input methods are disabled for the password field because otherwise
74 // anyone can access the underlying password and display it in clear text.
75 return false;
76#endif
77}
78
79bool PasswordInputType::shouldResetOnDocumentActivation()
80{
81 return true;
82}
83
84bool PasswordInputType::shouldRespectListAttribute()
85{
86 return false;
87}
