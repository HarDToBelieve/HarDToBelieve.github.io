25#pragma once
26
27#include "HTMLDocument.h"
28
29namespace WebCore {
30
31class HTMLPlugInElement;
32class Widget;
33
34class PluginDocument final : public HTMLDocument {
35 WTF_MAKE_ISO_ALLOCATED(PluginDocument);
36public:
37 static Ref<PluginDocument> create(Frame& frame, const URL& url)
38 {
39 return adoptRef(*new PluginDocument(frame, url));
40 }
41
42 WEBCORE_EXPORT Widget* pluginWidget();
43 HTMLPlugInElement* pluginElement() { return m_pluginElement.get(); }
44
45 void setPluginElement(HTMLPlugInElement&);
46 void detachFromPluginElement();
47
48 void cancelManualPluginLoad();
49
50 bool shouldLoadPluginManually() const { return m_shouldLoadPluginManually; }
51
52private:
53 PluginDocument(Frame&, const URL&);
54
55 Ref<DocumentParser> createParser() final;
56
57 RefPtr<HTMLPlugInElement> m_pluginElement;
58 bool m_shouldLoadPluginManually { true };
59};
60
61} // namespace WebCore
62
63SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::PluginDocument)
64 static bool isType(const WebCore::Document& document) { return document.isPluginDocument(); }
65 static bool isType(const WebCore::Node& node) { return is<WebCore::Document>(node) && isType(downcast<WebCore::Document>(node)); }
66SPECIALIZE_TYPE_TRAITS_END()
