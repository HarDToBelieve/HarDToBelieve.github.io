59static RefPtr<HTMLInputElement> nonEmptyRadioButton(Element& element)
60{
61 if (!is<HTMLInputElement>(element))
62 return nullptr;
63
64 auto& inputElement = downcast<HTMLInputElement>(element);
65 if (!inputElement.isRadioButton() || inputElement.value().isEmpty())
66 return nullptr;
67 return &inputElement;
68}
69
70String RadioNodeList::value() const
71{
72 auto length = this->length();
73 for (unsigned i = 0; i < length; ++i) {
74 if (auto button = nonEmptyRadioButton(*item(i))) {
75 if (button->checked())
76 return button->value();
77 }
78 }
79 return String();
80}
81
82void RadioNodeList::setValue(const String& value)
83{
84 auto length = this->length();
85 for (unsigned i = 0; i < length; ++i) {
86 if (auto button = nonEmptyRadioButton(*item(i))) {
87 if (button->value() == value) {
88 button->setChecked(true);
89 return;
90 }
91 }
92 }
93}

95bool RadioNodeList::elementMatches(Element& element) const
96{
97 if (!is<HTMLObjectElement>(element) && !is<HTMLFormControlElement>(element))
98 return false;
99
100 if (is<HTMLInputElement>(element) && downcast<HTMLInputElement>(element).isImageButton())
101 return false;
102
103 if (is<HTMLFormElement>(ownerNode())) {
104 RefPtr<HTMLFormElement> form;
105 if (is<HTMLObjectElement>(element))
106 form = downcast<HTMLObjectElement>(element).form();
107 else
108 form = downcast<HTMLFormControlElement>(element).form();
109 if (!form || form != &ownerNode())
110 return false;
111 }
112
113 return element.getIdAttribute() == m_name || element.getNameAttribute() == m_name;
114}
