38RubyElement::RubyElement(const QualifiedName& tagName, Document& document)

54RenderPtr<RenderElement> RubyElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition& insertionPosition)
55{
56 if (style.display() == DisplayType::Inline)
57 return createRenderer<RenderRubyAsInline>(*this, WTFMove(style));
58 if (style.display() == DisplayType::Block || style.display() == DisplayType::InlineBlock)
59 return createRenderer<RenderRubyAsBlock>(*this, WTFMove(style));
60 return HTMLElement::createElementRenderer(WTFMove(style), insertionPosition);
61}
