37TypeAhead::TypeAhead(TypeAheadDataSource* dataSource)
38 : m_dataSource(dataSource)
39 , m_repeatingChar(0)
40{
41}

43static const Seconds typeAheadTimeout { 1_s };
44
45static String stripLeadingWhiteSpace(const String& string)
46{
47 unsigned length = string.length();
48 unsigned i;
49 for (i = 0; i < length; ++i) {
50 if (string[i] != noBreakSpace && !isSpaceOrNewline(string[i]))
51 break;
52 }
53 return string.substring(i, length - i);
54}

56int TypeAhead::handleEvent(KeyboardEvent* event, MatchModeFlags matchMode)
57{
58 if (event->timeStamp() < m_lastTypeTime)
59 return -1;
60
61 int optionCount = m_dataSource->optionCount();
62 Seconds delta = event->timeStamp() - m_lastTypeTime;
63 m_lastTypeTime = event->timeStamp();
64
65 UChar c = event->charCode();
66
67 if (delta > typeAheadTimeout)
68 m_buffer.clear();
69 m_buffer.append(c);
70
71 if (optionCount < 1)
72 return -1;
73
74 int searchStartOffset = 1;
75 String prefix;
76 if (matchMode & CycleFirstChar && c == m_repeatingChar) {
77 // The user is likely trying to cycle through all the items starting
78 // with this character, so just search on the character.
79 prefix = String(&c, 1);
80 m_repeatingChar = c;
81 } else if (matchMode & MatchPrefix) {
82 prefix = m_buffer.toString();
83 if (m_buffer.length() > 1) {
84 m_repeatingChar = 0;
85 searchStartOffset = 0;
86 } else
87 m_repeatingChar = c;
88 }
89
90 if (!prefix.isEmpty()) {
91 int selected = m_dataSource->indexOfSelectedOption();
92 int index = (selected < 0 ? 0 : selected) + searchStartOffset;
93 index %= optionCount;
94
95 String prefixWithCaseFolded(prefix.foldCase());
96 for (int i = 0; i < optionCount; ++i, index = (index + 1) % optionCount) {
97 // Fold the option string and check if its prefix is equal to the folded prefix.
98 String text = m_dataSource->optionAtIndex(index);
99 if (stripLeadingWhiteSpace(text).foldCase().startsWith(prefixWithCaseFolded))
100 return index;
101 }
102 }
103
104 if (matchMode & MatchIndex) {
105 int index = parseIntegerAllowingTrailingJunk<int>(m_buffer).value_or(0);
106 if (index > 0 && index <= optionCount)
107 return index - 1;
108 }
109 return -1;
110}
