28#include <wtf/MonotonicTime.h>
29#include <wtf/text/StringBuilder.h>

32namespace WebCore {
33
34class KeyboardEvent;
35
36class TypeAheadDataSource {
37public:
38 virtual ~TypeAheadDataSource() = default;
39
40 virtual int indexOfSelectedOption() const = 0;
41 virtual int optionCount() const = 0;
42 virtual String optionAtIndex(int index) const = 0;
43};
44
45class TypeAhead {
46public:
47 TypeAhead(TypeAheadDataSource*);
48
49 enum ModeFlag {
50 MatchPrefix = 1 << 0,
51 CycleFirstChar = 1 << 1,
52 MatchIndex = 1 << 2,
53 };
54 typedef unsigned MatchModeFlags;
55
56 // Returns the index for the matching option.
57 int handleEvent(KeyboardEvent*, MatchModeFlags);
58
59private:
60 TypeAheadDataSource* m_dataSource;
61 MonotonicTime m_lastTypeTime;
62 UChar m_repeatingChar;
63 StringBuilder m_buffer;
64};
