87static unsigned countASCIIDigits(StringView string)
88{
89 unsigned length = string.length();
90 for (unsigned count = 0; count < length; ++count) {
91 if (!isASCIIDigit(string[count]))
92 return count;
93 }
94 return length;
95}

97void URLDecomposition::setHost(StringView value)
98{
99 auto fullURL = this->fullURL();
100 if (value.isEmpty() && !fullURL.protocolIs("file"))
101 return;
102
103 size_t separator = value.reverseFind(':');
104 if (!separator)
105 return;
106
107 if (fullURL.cannotBeABaseURL() || !fullURL.canSetHostOrPort())
108 return;
109
110 // No port if no colon or rightmost colon is within the IPv6 section.
111 size_t ipv6Separator = value.reverseFind(']');
112 if (separator == notFound || (ipv6Separator != notFound && ipv6Separator > separator))
113 fullURL.setHost(value);
114 else {
115 // Multiple colons are acceptable only in case of IPv6.
116 if (value.find(':') != separator && ipv6Separator == notFound)
117 return;
118 unsigned portLength = countASCIIDigits(value.substring(separator + 1));
119 if (!portLength) {
120 fullURL.setHost(value.substring(0, separator));
121 } else {
122 auto portNumber = parseInteger<uint16_t>(value.substring(separator + 1, portLength));
123 if (portNumber && WTF::isDefaultPortForProtocol(*portNumber, fullURL.protocol()))
124 fullURL.setHostAndPort(value.substring(0, separator));
125 else
126 fullURL.setHostAndPort(value.substring(0, separator + 1 + portLength));
127 }
128 }
129 if (fullURL.isValid())
130 setFullURL(fullURL);
131}

138static StringView removeAllLeadingSolidusCharacters(StringView string)
139{
140 unsigned i;
141 unsigned length = string.length();
142 for (i = 0; i < length; ++i) {
143 if (string[i] != '/')
144 break;
145 }
146 return string.substring(i);
147}

149void URLDecomposition::setHostname(StringView value)
150{
151 auto fullURL = this->fullURL();
152 auto host = removeAllLeadingSolidusCharacters(value);
153 if (host.isEmpty() && !fullURL.protocolIs("file"))
154 return;
155 if (fullURL.cannotBeABaseURL() || !fullURL.canSetHostOrPort())
156 return;
157 fullURL.setHost(host);
158 if (fullURL.isValid())
159 setFullURL(fullURL);
160}

170// Outer optional is whether we could parse at all. Inner optional is "no port specified".
171static std::optional<std::optional<uint16_t>> parsePort(StringView string, StringView protocol)
172{
173 auto digitsOnly = string.left(countASCIIDigits(string));
174 if (digitsOnly.isEmpty())
175 return std::optional<uint16_t> { std::nullopt };
176 auto port = parseInteger<uint16_t>(digitsOnly);
177 if (!port)
178 return std::nullopt;
179 if (WTF::isDefaultPortForProtocol(*port, protocol))
180 return std::optional<uint16_t> { std::nullopt };
181 return { { *port } };
182}

184void URLDecomposition::setPort(StringView value)
185{
186 auto fullURL = this->fullURL();
187 if (fullURL.host().isEmpty() || fullURL.cannotBeABaseURL() || fullURL.protocolIs("file") || !fullURL.canSetHostOrPort())
188 return;
189 auto port = parsePort(value, fullURL.protocol());
190 if (!port)
191 return;
192 fullURL.setPort(*port);
193 setFullURL(fullURL);
194}

216void URLDecomposition::setSearch(const String& value)
217{
218 auto fullURL = this->fullURL();
219 if (value.isEmpty()) {
220 // If the given value is the empty string, set url's query to null.
221 fullURL.setQuery({ });
222 } else {
223 String newSearch = value;
224 // Make sure that '#' in the query does not leak to the hash.
225 fullURL.setQuery(newSearch.replaceWithLiteral('#', "%23"));
226 }
227 setFullURL(fullURL);
228}

230String URLDecomposition::hash() const
231{
232 // FIXME: Why convert this string to an atom here instead of just a string? Intentionally to save memory? An error?
233 auto fullURL = this->fullURL();
234 return fullURL.fragmentIdentifier().isEmpty() ? emptyAtom() : fullURL.fragmentIdentifierWithLeadingNumberSign().toAtomString();
235}
236
237void URLDecomposition::setHash(StringView value)
238{
239 auto fullURL = this->fullURL();
240 if (value.isEmpty())
241 fullURL.removeFragmentIdentifier();
242 else
243 fullURL.setFragmentIdentifier(value.startsWith('#') ? value.substring(1) : value);
244 setFullURL(fullURL);
245}
