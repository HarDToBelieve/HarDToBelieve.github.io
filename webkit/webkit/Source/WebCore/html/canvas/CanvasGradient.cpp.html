71ExceptionOr<void> CanvasGradient::addColorStop(double value, const String& colorString)
72{
73 if (!(value >= 0 && value <= 1))
74 return Exception { IndexSizeError };
75
76 // Treat currentColor as black, as required by the standard.
77 Color color = isCurrentColorString(colorString) ? Color::black : parseColor(colorString, m_canvas);
78 if (!color.isValid())
79 return Exception { SyntaxError };
80
81 m_gradient->addColorStop({ static_cast<float>(value), WTFMove(color) });
82 return { };
83}
