28
29#include "CanvasGradient.h"
30#include "CanvasPattern.h"

35namespace WebCore {
36
37class Document;
38class GraphicsContext;
39class CanvasBase;
40
41class CanvasStyle {
42public:
43 CanvasStyle();
44 CanvasStyle(Color);
45 CanvasStyle(const SRGBA<float>&);
46 CanvasStyle(CanvasGradient&);
47 CanvasStyle(CanvasPattern&);
48
49 static CanvasStyle createFromString(const String& color, CanvasBase&);
50 static CanvasStyle createFromStringWithOverrideAlpha(const String& color, float alpha, CanvasBase&);
51
52 bool isValid() const { return !WTF::holds_alternative<Invalid>(m_style); }
53 bool isCurrentColor() const { return WTF::holds_alternative<CurrentColor>(m_style); }
54 std::optional<float> overrideAlpha() const { return WTF::get<CurrentColor>(m_style).overrideAlpha; }
55
56 String color() const;
57 RefPtr<CanvasGradient> canvasGradient() const;
58 RefPtr<CanvasPattern> canvasPattern() const;
59
60 void applyFillColor(GraphicsContext&) const;
61 void applyStrokeColor(GraphicsContext&) const;
62
63 bool isEquivalentColor(const CanvasStyle&) const;
64 bool isEquivalent(const SRGBA<float>&) const;
65
66private:
67 struct Invalid { };
68
69 struct CurrentColor {
70 std::optional<float> overrideAlpha;
71 };
72
73 CanvasStyle(CurrentColor);
74
75 Variant<Invalid, Color, RefPtr<CanvasGradient>, RefPtr<CanvasPattern>, CurrentColor> m_style;
76};
77
78bool isCurrentColorString(const String& colorString);
79
80Color currentColor(CanvasBase&);
81Color parseColor(const String& colorString, CanvasBase&);
82Color parseColorOrCurrentColor(const String& colorString, CanvasBase&);
83
84inline CanvasStyle::CanvasStyle()
85 : m_style(Invalid { })
86{
87}
88
89inline RefPtr<CanvasGradient> CanvasStyle::canvasGradient() const
90{
91 if (!WTF::holds_alternative<RefPtr<CanvasGradient>>(m_style))
92 return nullptr;
93 return WTF::get<RefPtr<CanvasGradient>>(m_style);
94}
95
96inline RefPtr<CanvasPattern> CanvasStyle::canvasPattern() const
97{
98 if (!WTF::holds_alternative<RefPtr<CanvasPattern>>(m_style))
99 return nullptr;
100 return WTF::get<RefPtr<CanvasPattern>>(m_style);
101}
102
103inline String CanvasStyle::color() const
104{
105 return serializationForHTML(WTF::get<Color>(m_style));
106}
