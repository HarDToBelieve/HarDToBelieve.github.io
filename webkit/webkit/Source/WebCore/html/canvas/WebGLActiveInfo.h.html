28#if ENABLE(WEBGL)

30#include "GraphicsContextGL.h"
31#include <wtf/RefCounted.h>
32#include <wtf/text/WTFString.h>

34namespace WebCore {
35
36class WebGLActiveInfo : public RefCounted<WebGLActiveInfo> {
37public:
38 static Ref<WebGLActiveInfo> create(const String& name, GCGLenum type, GCGLint size)
39 {
40 return adoptRef(*new WebGLActiveInfo(name, type, size));
41 }
42 String name() const { return m_name; }
43 GCGLenum type() const { return m_type; }
44 GCGLint size() const { return m_size; }
45
46private:
47 WebGLActiveInfo(const String& name, GCGLenum type, GCGLint size)
48 : m_name(name)
49 , m_type(type)
50 , m_size(size)
51 {
52 ASSERT(name.length());
53 ASSERT(type);
54 ASSERT(size);
55 }
56 String m_name;
57 GCGLenum m_type;
58 GCGLint m_size;
59};

61} // namespace WebCore
