35namespace WebCore {
36
37WTF_MAKE_ISO_ALLOCATED_IMPL(WebGLCompressedTextureATC);
38
39WebGLCompressedTextureATC::WebGLCompressedTextureATC(WebGLRenderingContextBase& context)
40 : WebGLExtension(context)
41{
42 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_AMD_compressed_ATC_texture");
43
44 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_ATC_RGB_AMD);
45 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_ATC_RGBA_EXPLICIT_ALPHA_AMD);
46 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_ATC_RGBA_INTERPOLATED_ALPHA_AMD);
47}
48
49WebGLCompressedTextureATC::~WebGLCompressedTextureATC() = default;
50
51WebGLExtension::ExtensionName WebGLCompressedTextureATC::getName() const
52{
53 return WebGLCompressedTextureATCName;
54}
55
56bool WebGLCompressedTextureATC::supported(WebGLRenderingContextBase& context)
57{
58 return context.graphicsContextGL()->getExtensions().supports("GL_AMD_compressed_ATC_texture");
59}
