35namespace WebCore {
36
37WTF_MAKE_ISO_ALLOCATED_IMPL(WebGLCompressedTextureETC);
38
39WebGLCompressedTextureETC::WebGLCompressedTextureETC(WebGLRenderingContextBase& context)
40 : WebGLExtension(context)
41{
42 context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_compressed_texture_etc");
43
44 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_R11_EAC);
45 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_SIGNED_R11_EAC);
46 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RG11_EAC);
47 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_SIGNED_RG11_EAC);
48 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGB8_ETC2);
49 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_SRGB8_ETC2);
50 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGB8_PUNCHTHROUGH_ALPHA1_ETC2);
51 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_SRGB8_PUNCHTHROUGH_ALPHA1_ETC2);
52 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGBA8_ETC2_EAC);
53 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_SRGB8_ALPHA8_ETC2_EAC);
54}
55
56WebGLCompressedTextureETC::~WebGLCompressedTextureETC() = default;
57
58WebGLExtension::ExtensionName WebGLCompressedTextureETC::getName() const
59{
60 return WebGLCompressedTextureETCName;
61}
62
63bool WebGLCompressedTextureETC::supported(WebGLRenderingContextBase& context)
64{
65 return context.graphicsContextGL()->getExtensions().supports("GL_ANGLE_compressed_texture_etc");
66}
