40WebGLCompressedTextureS3TC::WebGLCompressedTextureS3TC(WebGLRenderingContextBase& context)
41 : WebGLExtension(context)
42{
43 auto& extensions = context.graphicsContextGL()->getExtensions();
44 extensions.ensureEnabled("GL_EXT_texture_compression_dxt1");
45 extensions.ensureEnabled("GL_ANGLE_texture_compression_dxt3");
46 extensions.ensureEnabled("GL_ANGLE_texture_compression_dxt5");
47
48 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGB_S3TC_DXT1_EXT);
49 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGBA_S3TC_DXT1_EXT);
50 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGBA_S3TC_DXT3_EXT);
51 context.addCompressedTextureFormat(ExtensionsGL::COMPRESSED_RGBA_S3TC_DXT5_EXT);
52}
53
54WebGLCompressedTextureS3TC::~WebGLCompressedTextureS3TC() = default;
55
56WebGLExtension::ExtensionName WebGLCompressedTextureS3TC::getName() const
57{
58 return WebGLCompressedTextureS3TCName;
59}
60
61bool WebGLCompressedTextureS3TC::supported(WebGLRenderingContextBase& context)
62{
63 auto& extensions = context.graphicsContextGL()->getExtensions();
64#if USE(ANGLE)
65 return extensions.supports("GL_EXT_texture_compression_dxt1")
66 && extensions.supports("GL_ANGLE_texture_compression_dxt3")
67 && extensions.supports("GL_ANGLE_texture_compression_dxt5");
68#else
69 return extensions.supports("GL_EXT_texture_compression_s3tc")
70 || extensions.supports("GL_EXT_texture_compression_dxt1");
71#endif
72}
