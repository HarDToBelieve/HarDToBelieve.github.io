275#if !USE(ANGLE)
276 bool isAttachmentComplete(WebGLFramebuffer::WebGLAttachment* attachedObject, GCGLenum attachment, const char** reason)
277 {
278 ASSERT(attachedObject && attachedObject->isValid());
279 ASSERT(reason);
280 GCGLenum format = attachedObject->getFormat();
281 unsigned need = GraphicsContextGLOpenGL::getClearBitsByAttachmentType(attachment);
282 unsigned have = GraphicsContextGLOpenGL::getClearBitsByFormat(format);
283
284 if ((need & have) != need) {
285 *reason = "attachment type is not correct for attachment";
286 return false;
287 }
288 if (!attachedObject->getWidth() || !attachedObject->getHeight()) {
289 *reason = "attachment has a 0 dimension";
290 return false;
291 }
292 if ((attachment == GraphicsContextGL::DEPTH_ATTACHMENT || attachment == GraphicsContextGL::STENCIL_ATTACHMENT)
293 && format == GraphicsContextGL::DEPTH_STENCIL) {
294 *reason = "attachment DEPTH_STENCIL not allowed on DEPTH or STENCIL attachment";
295 return false;
296 }
297 return true;
298 }
299#endif

338void WebGLFramebuffer::setAttachmentForBoundFramebuffer(GCGLenum target, GCGLenum attachment, GCGLenum texTarget, WebGLTexture* texture, GCGLint level, GCGLint layer)
339{
340 ASSERT(object());
341 ASSERT(isBound(target));
342 setAttachmentInternal(attachment, texTarget, texture, level, layer);
343 if (context()->isWebGL2()) {
344 GCGLuint textureID = objectOrZero(texture);
345 // texTarget can be 0 if detaching using framebufferTextureLayer.
346 ASSERT(texTarget || !textureID);
347 switch (texTarget) {
348 case 0:
349 case GraphicsContextGL::TEXTURE_3D:
350 case GraphicsContextGL::TEXTURE_2D_ARRAY:
351 context()->graphicsContextGL()->framebufferTextureLayer(target, attachment, textureID, level, layer);
352 break;
353 default:
354 ASSERT(!layer);
355 context()->graphicsContextGL()->framebufferTexture2D(target, attachment, texTarget, textureID, level);
356 break;
357 }
358 } else {
359 ASSERT(!layer);
360#if USE(ANGLE)
361 context()->graphicsContextGL()->framebufferTexture2D(target, attachment, texTarget, objectOrZero(texture), level);
362#else
363 GCGLuint textureID = objectOrZero(texture);
364 if (attachment == GraphicsContextGL::DEPTH_STENCIL_ATTACHMENT) {
365 context()->graphicsContextGL()->framebufferTexture2D(target, GraphicsContextGL::DEPTH_ATTACHMENT, texTarget, textureID, level);
366 context()->graphicsContextGL()->framebufferTexture2D(target, GraphicsContextGL::STENCIL_ATTACHMENT, texTarget, textureID, level);
367 } else
368 context()->graphicsContextGL()->framebufferTexture2D(target, attachment, texTarget, textureID, level);
369#endif
370 }
371}

407void WebGLFramebuffer::removeAttachmentFromBoundFramebuffer(const AbstractLocker& locker, GCGLenum target, GCGLenum attachment)
408{
409 if (!context()) {
410 // Context has been deleted - should not be calling this.
411 return;
412 }
413
414#if ASSERT_ENABLED
415 ASSERT(isBound(target));
416#else
417 UNUSED_PARAM(target);
418#endif
419 if (!object())
420 return;
421
422 RefPtr<WebGLAttachment> attachmentObject = getAttachment(attachment);
423 if (attachmentObject) {
424 attachmentObject->onDetached(locker, context()->graphicsContextGL());
425 m_attachments.remove(attachment);
426 drawBuffersIfNecessary(false);
427#if !USE(ANGLE)
428 switch (attachment) {
429 case GraphicsContextGL::DEPTH_STENCIL_ATTACHMENT:
430 attach(target, GraphicsContextGL::DEPTH_ATTACHMENT, GraphicsContextGL::DEPTH_ATTACHMENT);
431 attach(target, GraphicsContextGL::STENCIL_ATTACHMENT, GraphicsContextGL::STENCIL_ATTACHMENT);
432 break;
433 case GraphicsContextGL::DEPTH_ATTACHMENT:
434 attach(target, GraphicsContextGL::DEPTH_STENCIL_ATTACHMENT, GraphicsContextGL::DEPTH_ATTACHMENT);
435 break;
436 case GraphicsContextGL::STENCIL_ATTACHMENT:
437 attach(target, GraphicsContextGL::DEPTH_STENCIL_ATTACHMENT, GraphicsContextGL::STENCIL_ATTACHMENT);
438 break;
439 }
440#endif
441 }
442}

444void WebGLFramebuffer::removeAttachmentFromBoundFramebuffer(const AbstractLocker& locker, GCGLenum target, WebGLSharedObject* attachment)
445{
446 ASSERT(isBound(target));
447 if (!object())
448 return;
449 if (!attachment)
450 return;
451
452 bool checkMore = true;
453 do {
454 checkMore = false;
455 for (auto& entry : m_attachments) {
456 RefPtr<WebGLAttachment> attachmentObject = entry.value.get();
457 if (attachmentObject->isSharedObject(attachment)) {
458 GCGLenum attachmentType = entry.key;
459 attachmentObject->unattach(context()->graphicsContextGL(), target, attachmentType);
460 removeAttachmentFromBoundFramebuffer(locker, target, attachmentType);
461 checkMore = true;
462 break;
463 }
464 }
465 } while (checkMore);
466}

501GCGLenum WebGLFramebuffer::checkStatus(const char** reason) const
502{
503#if ENABLE(WEBXR)
504 // https://immersive-web.github.io/webxr/#opaque-framebuffer
505 if (m_opaque) {
506 if (m_opaqueActive)
507 return GL_FRAMEBUFFER_COMPLETE;
508 *reason = "An opaque framebuffer is considered incomplete outside of a requestAnimationFrame";
509 return GL_FRAMEBUFFER_UNSUPPORTED;
510 }
511#endif
512
513 unsigned int count = 0;
514 GCGLsizei width = 0, height = 0;
515 bool haveDepth = false;
516 bool haveStencil = false;
517 bool haveDepthStencil = false;
518 for (auto& entry : m_attachments) {
519 RefPtr<WebGLAttachment> attachment = entry.value.get();
520 if (!isAttachmentComplete(attachment.get(), entry.key, reason))
521 return GraphicsContextGL::FRAMEBUFFER_INCOMPLETE_ATTACHMENT;
522 if (!attachment->isValid()) {
523 *reason = "attachment is not valid";
524 return GraphicsContextGL::FRAMEBUFFER_UNSUPPORTED;
525 }
526 GCGLenum attachmentFormat = attachment->getFormat();
527
528 // Attaching an SRGB_EXT format attachment to a framebuffer is invalid.
529 if (attachmentFormat == ExtensionsGL::SRGB_EXT)
530 attachmentFormat = 0;
531
532 if (!attachmentFormat) {
533 *reason = "attachment is an unsupported format";
534 return GraphicsContextGL::FRAMEBUFFER_INCOMPLETE_ATTACHMENT;
535 }
536 switch (entry.key) {
537 case GraphicsContextGL::DEPTH_ATTACHMENT:
538 haveDepth = true;
539 break;
540 case GraphicsContextGL::STENCIL_ATTACHMENT:
541 haveStencil = true;
542 break;
543 case GraphicsContextGL::DEPTH_STENCIL_ATTACHMENT:
544 haveDepthStencil = true;
545 break;
546 }
547 if (!count) {
548 width = attachment->getWidth();
549 height = attachment->getHeight();
550 } else {
551 if (width != attachment->getWidth() || height != attachment->getHeight()) {
552 *reason = "attachments do not have the same dimensions";
553 return GraphicsContextGL::FRAMEBUFFER_INCOMPLETE_DIMENSIONS;
554 }
555 }
556 ++count;
557 }
558 if (!count) {
559 *reason = "no attachments";
560 return GraphicsContextGL::FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT;
561 }
562 if (!width || !height) {
563 *reason = "framebuffer has a 0 dimension";
564 return GraphicsContextGL::FRAMEBUFFER_INCOMPLETE_ATTACHMENT;
565 }
566 // WebGL specific: no conflicting DEPTH/STENCIL/DEPTH_STENCIL attachments.
567 if ((haveDepthStencil && (haveDepth || haveStencil)) || (haveDepth && haveStencil)) {
568 *reason = "conflicting DEPTH/STENCIL/DEPTH_STENCIL attachments";
569 return GraphicsContextGL::FRAMEBUFFER_UNSUPPORTED;
570 }
571 return GraphicsContextGL::FRAMEBUFFER_COMPLETE;
572}

575{
576 if (checkStatus(reason) != GraphicsContextGL::FRAMEBUFFER_COMPLETE)
577 return false;
578 return initializeAttachments(context3d, reason);
579}

598#if !USE(ANGLE)
599bool WebGLFramebuffer::initializeAttachments(GraphicsContextGL* g3d, const char** reason)
600{
601 if (!context()) {
602 // Context has been deleted - should not be calling this.
603 return false;
604 }
605 Locker locker { objectGraphLockForContext() };
606
607 ASSERT(object());
608 GCGLbitfield mask = 0;
609
610 for (auto& entry : m_attachments) {
611 GCGLenum attachmentType = entry.key;
612 RefPtr<WebGLAttachment> attachment = entry.value.get();
613 if (!attachment->isInitialized())
614 mask |= GraphicsContextGLOpenGL::getClearBitsByAttachmentType(attachmentType);
615 }
616 if (!mask)
617 return true;
618
619 // We only clear un-initialized renderbuffers when they are ready to be
620 // read, i.e., when the framebuffer is complete.
621 if (g3d->checkFramebufferStatus(GraphicsContextGL::FRAMEBUFFER) != GraphicsContextGL::FRAMEBUFFER_COMPLETE) {
622 *reason = "framebuffer not complete";
623 return false;
624 }
625
626 bool initColor = mask & GraphicsContextGL::COLOR_BUFFER_BIT;
627 bool initDepth = mask & GraphicsContextGL::DEPTH_BUFFER_BIT;
628 bool initStencil = mask & GraphicsContextGL::STENCIL_BUFFER_BIT;
629
630 GCGLfloat colorClearValue[] = {0, 0, 0, 0}, depthClearValue = 0;
631 GCGLint stencilClearValue = 0;
632 GCGLboolean colorMask[] = {0, 0, 0, 0}, depthMask = 0;
633 GCGLuint stencilMask = 0xffffffff;
634 GCGLboolean isScissorEnabled = 0;
635 GCGLboolean isDitherEnabled = 0;
636 if (initColor) {
637 g3d->getFloatv(GraphicsContextGL::COLOR_CLEAR_VALUE, colorClearValue);
638 g3d->getBooleanv(GraphicsContextGL::COLOR_WRITEMASK, colorMask);
639 g3d->clearColor(0, 0, 0, 0);
640 g3d->colorMask(true, true, true, true);
641 }
642 if (initDepth) {
643 depthClearValue = g3d->getFloat(GraphicsContextGL::DEPTH_CLEAR_VALUE);
644 depthMask = g3d->getBoolean(GraphicsContextGL::DEPTH_WRITEMASK);
645 g3d->clearDepth(1.0f);
646 g3d->depthMask(true);
647 }
648 if (initStencil) {
649 stencilClearValue = g3d->getInteger(GraphicsContextGL::STENCIL_CLEAR_VALUE);
650 stencilMask = g3d->getInteger(GraphicsContextGL::STENCIL_WRITEMASK);
651 g3d->clearStencil(0);
652 g3d->stencilMask(0xffffffff);
653 }
654 isScissorEnabled = g3d->isEnabled(GraphicsContextGL::SCISSOR_TEST);
655 g3d->disable(GraphicsContextGL::SCISSOR_TEST);
656 isDitherEnabled = g3d->isEnabled(GraphicsContextGL::DITHER);
657 g3d->disable(GraphicsContextGL::DITHER);
658
659 g3d->clear(mask);
660
661 if (initColor) {
662 g3d->clearColor(colorClearValue[0], colorClearValue[1], colorClearValue[2], colorClearValue[3]);
663 g3d->colorMask(colorMask[0], colorMask[1], colorMask[2], colorMask[3]);
664 }
665 if (initDepth) {
666 g3d->clearDepth(depthClearValue);
667 g3d->depthMask(depthMask);
668 }
669 if (initStencil) {
670 g3d->clearStencil(stencilClearValue);
671 g3d->stencilMask(stencilMask);
672 }
673 if (isScissorEnabled)
674 g3d->enable(GraphicsContextGL::SCISSOR_TEST);
675 else
676 g3d->disable(GraphicsContextGL::SCISSOR_TEST);
677 if (isDitherEnabled)
678 g3d->enable(GraphicsContextGL::DITHER);
679 else
680 g3d->disable(GraphicsContextGL::DITHER);
681
682 for (AttachmentMap::iterator it = m_attachments.begin(); it != m_attachments.end(); ++it) {
683 GCGLenum attachmentType = it->key;
684 auto attachment = it->value;
685 GCGLbitfield bits = GraphicsContextGLOpenGL::getClearBitsByAttachmentType(attachmentType);
686 if (bits & mask)
687 attachment->setInitialized();
688 }
689 return true;
690}
691#endif

693bool WebGLFramebuffer::isBound(GCGLenum target) const
694{

698void WebGLFramebuffer::drawBuffers(const Vector<GCGLenum>& bufs)
699{
700 m_drawBuffers = bufs;
701 m_filteredDrawBuffers.resize(m_drawBuffers.size());
702 for (auto& buffer : m_filteredDrawBuffers)
703 buffer = GraphicsContextGL::NONE;
704 drawBuffersIfNecessary(true);
705}

707void WebGLFramebuffer::drawBuffersIfNecessary(bool force)
708{
709 if (context()->isWebGL2() || context()->m_webglDrawBuffers) {
710 bool reset = force;
711 // This filtering works around graphics driver bugs on macOS.
712 for (size_t i = 0; i < m_drawBuffers.size(); ++i) {
713 if (m_drawBuffers[i] != GraphicsContextGL::NONE && getAttachment(m_drawBuffers[i])) {
714 if (m_filteredDrawBuffers[i] != m_drawBuffers[i]) {
715 m_filteredDrawBuffers[i] = m_drawBuffers[i];
716 reset = true;
717 }
718 } else {
719 if (m_filteredDrawBuffers[i] != GraphicsContextGL::NONE) {
720 m_filteredDrawBuffers[i] = GraphicsContextGL::NONE;
721 reset = true;
722 }
723 }
724 }
725 if (reset) {
726 if (context()->isWebGL2())
727 context()->graphicsContextGL()->drawBuffers(m_filteredDrawBuffers);
728 else
729 context()->graphicsContextGL()->getExtensions().drawBuffersEXT(m_filteredDrawBuffers);
730 }
731 }
732}

734GCGLenum WebGLFramebuffer::getDrawBuffer(GCGLenum drawBuffer)
735{
736 int index = static_cast<int>(drawBuffer - ExtensionsGL::DRAW_BUFFER0_EXT);
737 ASSERT(index >= 0);
738 if (index < static_cast<int>(m_drawBuffers.size()))
739 return m_drawBuffers[index];
740 if (drawBuffer == ExtensionsGL::DRAW_BUFFER0_EXT)
741 return GraphicsContextGL::COLOR_ATTACHMENT0;
742 return GraphicsContextGL::NONE;
743}
