40void WebGLObject::setObject(PlatformGLObject object)
41{
42 ASSERT(!m_object);
43 ASSERT(!m_deleted);
44 m_object = object;
45}
46
47void WebGLObject::runDestructor()
48{
49 auto& lock = objectGraphLockForContext();
50 if (lock.isHeld()) {
51 // Destruction of WebGLObjects can happen in chains triggered from GC.
52 // The lock must be held only once, at the beginning of the chain.
53 auto locker = AbstractLocker(NoLockingNecessary);
54 deleteObject(locker, nullptr);
55 } else {
56 Locker locker { lock };
57 deleteObject(locker, nullptr);
58 }
59}
60
61void WebGLObject::deleteObject(const AbstractLocker& locker, GraphicsContextGL* context3d)
62{
63 m_deleted = true;
64 if (!m_object)
65 return;
66
67 if (!hasGroupOrContext())
68 return;
69
70 if (!m_attachmentCount) {
71 if (!context3d)
72 context3d = getAGraphicsContextGL();
73
74 if (context3d)
75 deleteObjectImpl(locker, context3d, m_object);
76 }
77
78 if (!m_attachmentCount)
79 m_object = 0;
80}
81
82void WebGLObject::detach()
83{
84 m_attachmentCount = 0; // Make sure OpenGL resource is deleted.
85}
86
87void WebGLObject::onDetached(const AbstractLocker& locker, GraphicsContextGL* context3d)
88{
89 if (m_attachmentCount)
90 --m_attachmentCount;
91 if (m_deleted)
92 deleteObject(locker, context3d);
93}
