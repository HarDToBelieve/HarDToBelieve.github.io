66WebGLProgram::WebGLProgram(WebGLRenderingContextBase& ctx)
67 : WebGLSharedObject(ctx)
68 , ContextDestructionObserver(ctx.scriptExecutionContext())
69{
70 ASSERT(scriptExecutionContext());
71
72 {
73 Locker locker { instancesLock() };
74 instances().add(this, &ctx);
75 }
76
77 setObject(ctx.graphicsContextGL()->createProgram());
78}
79
80WebGLProgram::~WebGLProgram()
81{
82 InspectorInstrumentation::willDestroyWebGLProgram(*this);
83
84 {
85 Locker locker { instancesLock() };
86 ASSERT(instances().contains(this));
87 instances().remove(this);
88 }
89
90 if (!hasGroupOrContext())
91 return;
92
93 runDestructor();
94}

103void WebGLProgram::deleteObjectImpl(const AbstractLocker& locker, GraphicsContextGL* context3d, PlatformGLObject obj)
104{
105 context3d->deleteProgram(obj);
106 if (m_vertexShader) {
107 m_vertexShader->onDetached(locker, context3d);
108 m_vertexShader = nullptr;
109 }
110 if (m_fragmentShader) {
111 m_fragmentShader->onDetached(locker, context3d);
112 m_fragmentShader = nullptr;
113 }
114}

116unsigned WebGLProgram::numActiveAttribLocations()
117{

122GCGLint WebGLProgram::getActiveAttribLocation(GCGLuint index)
123{
124 cacheInfoIfNeeded();
125 if (index >= numActiveAttribLocations())
126 return -1;
127 return m_activeAttribLocations[index];
128}

130bool WebGLProgram::isUsingVertexAttrib0()
131{
132 cacheInfoIfNeeded();
133 for (unsigned ii = 0; ii < numActiveAttribLocations(); ++ii) {
134 if (!getActiveAttribLocation(ii))
135 return true;
136 }
137 return false;
138}

170bool WebGLProgram::attachShader(const AbstractLocker&, WebGLShader* shader)
171{
172 if (!shader || !shader->object())
173 return false;
174 switch (shader->getType()) {
175 case GraphicsContextGL::VERTEX_SHADER:
176 if (m_vertexShader)
177 return false;
178 m_vertexShader = shader;
179 return true;
180 case GraphicsContextGL::FRAGMENT_SHADER:
181 if (m_fragmentShader)
182 return false;
183 m_fragmentShader = shader;
184 return true;
185 default:
186 return false;
187 }
188}
189
190bool WebGLProgram::detachShader(const AbstractLocker&, WebGLShader* shader)
191{
192 if (!shader || !shader->object())
193 return false;
194 switch (shader->getType()) {
195 case GraphicsContextGL::VERTEX_SHADER:
196 if (m_vertexShader != shader)
197 return false;
198 m_vertexShader = nullptr;
199 return true;
200 case GraphicsContextGL::FRAGMENT_SHADER:
201 if (m_fragmentShader != shader)
202 return false;
203 m_fragmentShader = nullptr;
204 return true;
205 default:
206 return false;
207 }
208}

216void WebGLProgram::cacheActiveAttribLocations(GraphicsContextGL* context3d)
217{
218 m_activeAttribLocations.clear();
219
220 GCGLint numAttribs = context3d->getProgrami(object(), GraphicsContextGL::ACTIVE_ATTRIBUTES);
221 m_activeAttribLocations.resize(static_cast<size_t>(numAttribs));
222 for (int i = 0; i < numAttribs; ++i) {
223 GraphicsContextGL::ActiveInfo info;
224#if USE(ANGLE)
225 context3d->getActiveAttrib(object(), i, info);
226#else
227 static_cast<GraphicsContextGLOpenGL*>(context3d)->getActiveAttribImpl(object(), i, info);
228#endif
229 m_activeAttribLocations[i] = context3d->getAttribLocation(object(), info.name);
230 }
231}
232
233void WebGLProgram::cacheInfoIfNeeded()
234{
235 if (m_infoValid)
236 return;
237
238 if (!object())
239 return;
240
241 GraphicsContextGL* context = getAGraphicsContextGL();
242 if (!context)
243 return;
244 GCGLint linkStatus = context->getProgrami(object(), GraphicsContextGL::LINK_STATUS);
245 m_linkStatus = linkStatus;
246 if (m_linkStatus) {
247 cacheActiveAttribLocations(context);
248 m_requiredTransformFeedbackBufferCount = m_requiredTransformFeedbackBufferCountAfterNextLink;
249 }
250 m_infoValid = true;
251}
