296 template <class TypedArray, class DataType>
297 class TypedList {
298 public:
299 using VariantType = Variant<RefPtr<TypedArray>, Vector<DataType>>;
300
301 TypedList(VariantType&& variant)
302 : m_variant(WTFMove(variant))
303 {
304 }
305
306 const DataType* data() const
307 {
308 return WTF::switchOn(m_variant,
309 [] (const RefPtr<TypedArray>& typedArray) -> const DataType* { return typedArray->data(); },
310 [] (const Vector<DataType>& vector) -> const DataType* { return vector.data(); }
311 );
312 }
313
314 GCGLsizei length() const
315 {
316 return WTF::switchOn(m_variant,
317 [] (const RefPtr<TypedArray>& typedArray) -> GCGLsizei { return typedArray->length(); },
318 [] (const Vector<DataType>& vector) -> GCGLsizei { return vector.size(); }
319 );
320 }
321
322 private:
323 VariantType m_variant;
324 };

794 template <typename T> bool validateTexImageSubRectangle(const char* functionName,
795 TexImageFunctionID functionID,
796 T* image,
797 const IntRect& subRect,
798 GCGLsizei depth,
799 GCGLint unpackImageHeight,
800 bool* selectingSubRectangle)
801 {
802 ASSERT(functionName);
803 ASSERT(selectingSubRectangle);
804 if (!image) {
805 // Probably indicates a failure to allocate the image.
806 synthesizeGLError(GraphicsContextGL::OUT_OF_MEMORY, functionName, "out of memory");
807 return false;
808 }
809
810 int imageWidth = static_cast<int>(image->width());
811 int imageHeight = static_cast<int>(image->height());
812 *selectingSubRectangle = !(!subRect.x() && !subRect.y() && subRect.width() == imageWidth && subRect.height() == imageHeight);
813 // If the source image rect selects anything except the entire
814 // contents of the image, assert that we're running WebGL 2.0,
815 // since this should never happen for WebGL 1.0 (even though
816 // the code could support it). If the image is null, that will
817 // be signaled as an error later.
818 ASSERT(!*selectingSubRectangle || isWebGL2());
819
820 if (!subRect.isValid() || subRect.x() < 0 || subRect.y() < 0
821 || subRect.maxX() > imageWidth || subRect.maxY() > imageHeight
822 || subRect.width() < 0 || subRect.height() < 0) {
823 synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, functionName,
824 "source sub-rectangle specified via pixel unpack parameters is invalid");
825 return false;
826 }
827
828 if (functionID == TexImageFunctionID::TexImage3D || functionID == TexImageFunctionID::TexSubImage3D) {
829 ASSERT(unpackImageHeight >= 0);
830
831 if (depth < 1) {
832 synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, functionName,
833 "Can't define a 3D texture with depth < 1");
834 return false;
835 }
836
837 // According to the WebGL 2.0 spec, specifying depth > 1 means
838 // to select multiple rectangles stacked vertically.
839 Checked<GCGLint, RecordOverflow> maxYAccessed;
840 if (unpackImageHeight)
841 maxYAccessed = unpackImageHeight;
842 else
843 maxYAccessed = subRect.height();
844 maxYAccessed *= depth - 1;
845 maxYAccessed += subRect.height();
846 maxYAccessed += subRect.y();
847
848 if (maxYAccessed.hasOverflowed()) {
849 synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, functionName,
850 "Out-of-range parameters passed for 3D texture upload");
851 return false;
852 }
853
854 if (maxYAccessed > imageHeight) {
855 synthesizeGLError(GraphicsContextGL::INVALID_OPERATION, functionName,
856 "Not enough data supplied to upload to a 3D texture with depth > 1");
857 return false;
858 }
859 } else {
860 ASSERT(depth >= 1);
861 ASSERT(!unpackImageHeight);
862 }
863 return true;
864 }

1152template<typename T>
1153inline unsigned WebGLRenderingContextBase::getMaxIndex(const RefPtr<JSC::ArrayBuffer> elementArrayBuffer, GCGLintptr uoffset, GCGLsizei n)
1154{
1155 unsigned maxIndex = 0;
1156 T restartIndex = 0;
1157
1158#if ENABLE(WEBGL2)
1159 // WebGL 2 spec enforces that GL_PRIMITIVE_RESTART_FIXED_INDEX is always enabled, so ignore the restart index.
1160 if (isWebGL2())
1161 restartIndex = std::numeric_limits<T>::max();
1162#endif
1163
1164 // Make uoffset an element offset.
1165 uoffset /= sizeof(T);
1166 const T* p = static_cast<const T*>(elementArrayBuffer->data()) + uoffset;
1167 while (n-- > 0) {
1168 if (*p != restartIndex && *p > maxIndex)
1169 maxIndex = *p;
1170 ++p;
1171 }
1172
1173 return maxIndex;
1174}
