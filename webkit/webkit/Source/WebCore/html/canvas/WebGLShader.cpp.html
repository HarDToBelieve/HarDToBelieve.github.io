39Ref<WebGLShader> WebGLShader::create(WebGLRenderingContextBase& ctx, GCGLenum type)
40{
41 return adoptRef(*new WebGLShader(ctx, type));
42}
43
44WebGLShader::WebGLShader(WebGLRenderingContextBase& ctx, GCGLenum type)
45 : WebGLSharedObject(ctx)
46 , m_type(type)
47 , m_source(emptyString())
48 , m_isValid(false)
49{
50 setObject(ctx.graphicsContextGL()->createShader(type));
51}
52
53WebGLShader::~WebGLShader()
54{
55 if (!hasGroupOrContext())
56 return;
57
58 runDestructor();
59}
60
61void WebGLShader::deleteObjectImpl(const AbstractLocker&, GraphicsContextGL* context3d, PlatformGLObject object)
62{
63 context3d->deleteShader(object);
64}
