32namespace WTF {
33class AbstractLocker;
34class Lock;
35}
36
37namespace WebCore {
38
39class GraphicsContextGL;
40class WebGLContextGroup;
41class WebGLRenderingContextBase;
42
43// WebGLSharedObject the base class for objects that can be shared by multiple WebGLRenderingContexts.
44class WebGLSharedObject : public WebGLObject {
45public:
46 virtual ~WebGLSharedObject();
47
48 WebGLContextGroup* contextGroup() const { return m_contextGroup; }
49
50 virtual bool isRenderbuffer() const { return false; }
51 virtual bool isTexture() const { return false; }
52
53 bool validate(const WebGLContextGroup* contextGroup, const WebGLRenderingContextBase&) const override
54 {
55 return contextGroup == m_contextGroup;
56 }
57
58 void detachContextGroup(const WTF::AbstractLocker&);
59
60 void detachContextGroupWithoutDeletingObject();
61
62 Lock& objectGraphLockForContext() override;
63
64protected:
65 WebGLSharedObject(WebGLRenderingContextBase&);
66
67 bool hasGroupOrContext() const override;
68
69 GraphicsContextGL* getAGraphicsContextGL() const override;
70
71private:
72 WebGLContextGroup* m_contextGroup;
73};

75} // namespace WebCore
