71void WebGLTexture::setTarget(GCGLenum target, GCGLint maxLevel)
72{
73 if (!object())
74 return;
75 // Target is finalized the first time bindTexture() is called.
76 if (m_target)
77 return;
78#if USE(ANGLE)
79 UNUSED_PARAM(maxLevel);
80 m_target = target;
81#else
82 switch (target) {
83 case GraphicsContextGL::TEXTURE_2D:
84 m_target = target;
85 m_info.resize(1);
86 m_info[0].resize(maxLevel);
87 break;
88 case GraphicsContextGL::TEXTURE_CUBE_MAP:
89 m_target = target;
90 m_info.resize(6);
91 for (int ii = 0; ii < 6; ++ii)
92 m_info[ii].resize(maxLevel);
93 break;
94 }
95#endif // USE(ANGLE)
96}

103GCGLint WebGLTexture::computeLevelCount(GCGLsizei width, GCGLsizei height)
104{
105 // return 1 + log2Floor(std::max(width, height));
106 GCGLsizei n = std::max(width, height);
107 if (n <= 0)
108 return 0;
109 GCGLint log = 0;
110 GCGLsizei value = n;
111 for (int ii = 4; ii >= 0; --ii) {
112 int shift = (1 << ii);
113 GCGLsizei x = (value >> shift);
114 if (x) {
115 value = x;
116 log += shift;
117 }
118 }
119 ASSERT(value == 1);
120 return log + 1;
121}

126void WebGLTexture::setParameteri(GCGLenum pname, GCGLint param)
127{
128 if (!object() || !m_target)
129 return;
130 switch (pname) {
131 case GraphicsContextGL::TEXTURE_MIN_FILTER:
132 switch (param) {
133 case GraphicsContextGL::NEAREST:
134 case GraphicsContextGL::LINEAR:
135 case GraphicsContextGL::NEAREST_MIPMAP_NEAREST:
136 case GraphicsContextGL::LINEAR_MIPMAP_NEAREST:
137 case GraphicsContextGL::NEAREST_MIPMAP_LINEAR:
138 case GraphicsContextGL::LINEAR_MIPMAP_LINEAR:
139 m_minFilter = param;
140 break;
141 }
142 break;
143 case GraphicsContextGL::TEXTURE_MAG_FILTER:
144 switch (param) {
145 case GraphicsContextGL::NEAREST:
146 case GraphicsContextGL::LINEAR:
147 m_magFilter = param;
148 break;
149 }
150 break;
151 case GraphicsContextGL::TEXTURE_WRAP_S:
152 switch (param) {
153 case GraphicsContextGL::CLAMP_TO_EDGE:
154 case GraphicsContextGL::MIRRORED_REPEAT:
155 case GraphicsContextGL::REPEAT:
156 m_wrapS = param;
157 break;
158 }
159 break;
160 case GraphicsContextGL::TEXTURE_WRAP_T:
161 switch (param) {
162 case GraphicsContextGL::CLAMP_TO_EDGE:
163 case GraphicsContextGL::MIRRORED_REPEAT:
164 case GraphicsContextGL::REPEAT:
165 m_wrapT = param;
166 break;
167 }
168 break;
169 default:
170 return;
171 }
172 update();
173}

183void WebGLTexture::setLevelInfo(GCGLenum target, GCGLint level, GCGLenum internalFormat, GCGLsizei width, GCGLsizei height, GCGLenum type)
184{
185 if (!object() || !m_target)
186 return;
187 // We assume level, internalFormat, width, height, and type have all been
188 // validated already.
189 int index = mapTargetToIndex(target);
190 if (index < 0)
191 return;
192 m_info[index][level].setInfo(internalFormat, width, height, type);
193 update();
194}

196void WebGLTexture::generateMipmapLevelInfo()
197{
198 if (!object() || !m_target)
199 return;
200 if (!canGenerateMipmaps())
201 return;
202 if (!m_isComplete) {
203 for (size_t ii = 0; ii < m_info.size(); ++ii) {
204 const LevelInfo& info0 = m_info[ii][0];
205 GCGLsizei width = info0.width;
206 GCGLsizei height = info0.height;
207 GCGLint levelCount = computeLevelCount(width, height);
208 for (GCGLint level = 1; level < levelCount; ++level) {
209 width = std::max(1, width >> 1);
210 height = std::max(1, height >> 1);
211 LevelInfo& info = m_info[ii][level];
212 info.setInfo(info0.internalFormat, width, height, info0.type);
213 }
214 }
215 m_isComplete = true;
216 }
217 m_needToUseBlackTexture = false;
218}

270bool WebGLTexture::isNPOT(GCGLsizei width, GCGLsizei height)
271{
272 ASSERT(width >= 0 && height >= 0);
273 if (!width || !height)
274 return false;
275 if ((width & (width - 1)) || (height & (height - 1)))
276 return true;
277 return false;
278}

287bool WebGLTexture::needToUseBlackTexture(TextureExtensionFlag extensions) const
288{
289 if (!object())
290 return false;
291 if (m_needToUseBlackTexture)
292 return true;
293 if (m_magFilter == GraphicsContextGL::NEAREST && (m_minFilter == GraphicsContextGL::NEAREST || m_minFilter == GraphicsContextGL::NEAREST_MIPMAP_NEAREST))
294 return false;
295 if (m_isForWebGL1 && m_isHalfFloatType && !(extensions & TextureExtensionHalfFloatLinearEnabled))
296 return true;
297 if (m_isFloatType && !(extensions & TextureExtensionFloatLinearEnabled))
298 return true;
299 return false;
300}

302bool WebGLTexture::canGenerateMipmaps()
303{
304 if (isNPOT())
305 return false;
306 const LevelInfo& first = m_info[0][0];
307 for (size_t ii = 0; ii < m_info.size(); ++ii) {
308 const LevelInfo& info = m_info[ii][0];
309 if (!info.valid
310 || info.width != first.width || info.height != first.height
311 || info.internalFormat != first.internalFormat || (m_isForWebGL1 && info.type != first.type))
312 return false;
313 }
314 return true;
315}

330int WebGLTexture::mapTargetToIndex(GCGLenum target) const
331{
332 if (m_target == GraphicsContextGL::TEXTURE_2D) {
333 if (target == GraphicsContextGL::TEXTURE_2D)
334 return 0;
335 } else if (m_target == GraphicsContextGL::TEXTURE_CUBE_MAP) {
336 switch (target) {
337 case GraphicsContextGL::TEXTURE_CUBE_MAP_POSITIVE_X:
338 return 0;
339 case GraphicsContextGL::TEXTURE_CUBE_MAP_NEGATIVE_X:
340 return 1;
341 case GraphicsContextGL::TEXTURE_CUBE_MAP_POSITIVE_Y:
342 return 2;
343 case GraphicsContextGL::TEXTURE_CUBE_MAP_NEGATIVE_Y:
344 return 3;
345 case GraphicsContextGL::TEXTURE_CUBE_MAP_POSITIVE_Z:
346 return 4;
347 case GraphicsContextGL::TEXTURE_CUBE_MAP_NEGATIVE_Z:
348 return 5;
349 }
350 }
351 return -1;
352}

354static bool internalFormatIsFloatType(GCGLenum internalFormat)
355{
356 switch (internalFormat) {
357 case GraphicsContextGL::R32F:
358 case GraphicsContextGL::RG32F:
359 case GraphicsContextGL::RGB32F:
360 case GraphicsContextGL::RGBA32F:
361 case GraphicsContextGL::DEPTH_COMPONENT32F:
362 case GraphicsContextGL::DEPTH32F_STENCIL8:
363 return true;
364 default:
365 return false;
366 }
367}

371 switch (internalFormat) {
372 case GraphicsContextGL::R16F:
373 case GraphicsContextGL::RG16F:
374 case GraphicsContextGL::R11F_G11F_B10F:
375 case GraphicsContextGL::RGB9_E5:

378 return true;

384void WebGLTexture::update()
385{
386 m_isNPOT = false;
387 for (size_t ii = 0; ii < m_info.size(); ++ii) {
388 if (isNPOT(m_info[ii][0].width, m_info[ii][0].height)) {
389 m_isNPOT = true;
390 break;
391 }
392 }
393 m_isComplete = true;
394 const LevelInfo& first = m_info[0][0];
395 GCGLint levelCount = computeLevelCount(first.width, first.height);
396 if (levelCount < 1)
397 m_isComplete = false;
398 else {
399 for (size_t ii = 0; ii < m_info.size() && m_isComplete; ++ii) {
400 const LevelInfo& info0 = m_info[ii][0];
401 if (!info0.valid
402 || info0.width != first.width || info0.height != first.height
403 || info0.internalFormat != first.internalFormat || (m_isForWebGL1 && info0.type != first.type)) {
404 m_isComplete = false;
405 break;
406 }
407 GCGLsizei width = info0.width;
408 GCGLsizei height = info0.height;
409 for (GCGLint level = 1; level < levelCount; ++level) {
410 width = std::max(1, width >> 1);
411 height = std::max(1, height >> 1);
412 const LevelInfo& info = m_info[ii][level];
413 if (!info.valid
414 || info.width != width || info.height != height
415 || info.internalFormat != info0.internalFormat || (m_isForWebGL1 && info.type != info0.type)) {
416 m_isComplete = false;
417 break;
418 }
419
420 }
421 }
422 }
423
424 m_isFloatType = false;
425 if (m_isForWebGL1) {
426 if (m_isComplete) {
427 if (m_isForWebGL1)
428 m_isFloatType = m_info[0][0].type == GraphicsContextGL::FLOAT;
429 else
430 m_isFloatType = internalFormatIsFloatType(m_info[0][0].internalFormat);
431 } else {
432 for (size_t ii = 0; ii < m_info.size(); ++ii) {
433 if ((m_isForWebGL1 && m_info[ii][0].type == GraphicsContextGL::FLOAT)
434 || (!m_isForWebGL1 && internalFormatIsFloatType(m_info[ii][0].internalFormat))) {
435 m_isFloatType = true;
436 break;
437 }
438 }
439 }
440 }
441
442 m_isHalfFloatType = false;
443 if (m_isForWebGL1) {
444 if (m_isComplete) {
445 if (m_isForWebGL1)
446 m_isHalfFloatType = internalFormatIsHalfFloatType(m_info[0][0].internalFormat);
447 else
448 m_isHalfFloatType = m_info[0][0].type == GraphicsContextGL::HALF_FLOAT_OES;
449 } else {
450 for (size_t ii = 0; ii < m_info.size(); ++ii) {
451 if ((m_isForWebGL1 && m_info[ii][0].type == GraphicsContextGL::HALF_FLOAT_OES)
452 || (!m_isForWebGL1 && internalFormatIsHalfFloatType(m_info[ii][0].internalFormat))) {
453 m_isHalfFloatType = true;
454 break;
455 }
456 }
457 }
458 }
459
460 m_needToUseBlackTexture = false;
461 // NPOT
462 if (m_isNPOT && ((m_minFilter != GraphicsContextGL::NEAREST && m_minFilter != GraphicsContextGL::LINEAR)
463 || m_wrapS != GraphicsContextGL::CLAMP_TO_EDGE || m_wrapT != GraphicsContextGL::CLAMP_TO_EDGE))
464 m_needToUseBlackTexture = true;
465 // Completeness
466 if (!m_isComplete && m_minFilter != GraphicsContextGL::NEAREST && m_minFilter != GraphicsContextGL::LINEAR)
467 m_needToUseBlackTexture = true;
468}

470const WebGLTexture::LevelInfo* WebGLTexture::getLevelInfo(GCGLenum target, GCGLint level) const
471{
472 if (!object() || !m_target)
473 return 0;
474 int targetIndex = mapTargetToIndex(target);
475 if (targetIndex < 0 || targetIndex >= static_cast<int>(m_info.size()))
476 return 0;
477 if (level < 0 || level >= static_cast<GCGLint>(m_info[targetIndex].size()))
478 return 0;
479 return &(m_info[targetIndex][level]);
480}
