33namespace WebCore {
34
35class WebGLTexture final : public WebGLSharedObject {
36public:
37
38 enum TextureExtensionFlag {
39 TextureExtensionsDisabled = 0,
40 TextureExtensionFloatLinearEnabled = 1 << 0,
41 TextureExtensionHalfFloatLinearEnabled = 2 << 0
42 };
43
44 virtual ~WebGLTexture();
45
46 static Ref<WebGLTexture> create(WebGLRenderingContextBase&);
47
48 void setTarget(GCGLenum target, GCGLint maxLevel);
49#if !USE(ANGLE)
50 void setParameteri(GCGLenum pname, GCGLint param);
51 void setParameterf(GCGLenum pname, GCGLfloat param);
52#endif
53
54 GCGLenum getTarget() const { return m_target; }
55
56#if !USE(ANGLE)
57 int getMinFilter() const { return m_minFilter; }
58
59 void setLevelInfo(GCGLenum target, GCGLint level, GCGLenum internalFormat, GCGLsizei width, GCGLsizei height, GCGLenum type);
60
61 bool canGenerateMipmaps();
62 // Generate all level information.
63 void generateMipmapLevelInfo();
64
65 GCGLenum getInternalFormat(GCGLenum target, GCGLint level) const;
66 GCGLenum getType(GCGLenum target, GCGLint level) const;
67 GCGLsizei getWidth(GCGLenum target, GCGLint level) const;
68 GCGLsizei getHeight(GCGLenum target, GCGLint level) const;
69 bool isValid(GCGLenum target, GCGLint level) const;
70 void markInvalid(GCGLenum target, GCGLint level);
71
72 // Whether width/height is NotPowerOfTwo.
73 static bool isNPOT(GCGLsizei, GCGLsizei);
74
75 bool isNPOT() const;
76 // Determine if texture sampling should always return [0, 0, 0, 1] (OpenGL ES 2.0 Sec 3.8.2).
77 bool needToUseBlackTexture(TextureExtensionFlag) const;
78
79 bool immutable() const { return m_immutable; }
80 void setImmutable() { m_immutable = true; }
81
82 bool isCompressed() const;
83 void setCompressed();
84#endif
85
86 bool hasEverBeenBound() const { return object() && m_target; }
87
88 static GCGLint computeLevelCount(GCGLsizei width, GCGLsizei height);
89
90private:
91 WebGLTexture(WebGLRenderingContextBase&);
92
93 void deleteObjectImpl(const WTF::AbstractLocker&, GraphicsContextGL*, PlatformGLObject) override;
94
95 bool isTexture() const override { return true; }
96
97#if !USE(ANGLE)
98 class LevelInfo {
99 public:
100 LevelInfo()
101 : valid(false)
102 , internalFormat(0)
103 , width(0)
104 , height(0)
105 , type(0)
106 {
107 }
108
109 void setInfo(GCGLenum internalFmt, GCGLsizei w, GCGLsizei h, GCGLenum tp)
110 {
111 valid = true;
112 internalFormat = internalFmt;
113 width = w;
114 height = h;
115 type = tp;
116 }
117
118 bool valid;
119 GCGLenum internalFormat;
120 GCGLsizei width;
121 GCGLsizei height;
122 GCGLenum type;
123 };
124
125 void update();
126#endif // !USE(ANGLE)
127
128 int mapTargetToIndex(GCGLenum) const;
129
130#if !USE(ANGLE)
131 const LevelInfo* getLevelInfo(GCGLenum target, GCGLint level) const;
132#endif // !USE(ANGLE)
133
134 GCGLenum m_target;
135
136#if !USE(ANGLE)
137 GCGLenum m_minFilter;
138 GCGLenum m_magFilter;
139 GCGLenum m_wrapS;
140 GCGLenum m_wrapT;
141
142 Vector<Vector<LevelInfo>> m_info;
143
144 bool m_isNPOT;
145 bool m_isComplete;
146 bool m_needToUseBlackTexture;
147 bool m_isCompressed;
148 bool m_isFloatType;
149 bool m_isHalfFloatType;
150 bool m_isForWebGL1;
151 bool m_immutable { false };
152#endif
153};
