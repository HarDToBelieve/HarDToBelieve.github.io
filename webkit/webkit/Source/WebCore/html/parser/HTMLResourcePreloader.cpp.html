40URL PreloadRequest::completeURL(Document& document)
41{
42 return document.completeURL(m_resourceURL, m_baseURL.isEmpty() ? document.baseURL() : m_baseURL);
43}
44
45CachedResourceRequest PreloadRequest::resourceRequest(Document& document)
46{
47 ASSERT(isMainThread());
48
49 bool skipContentSecurityPolicyCheck = false;
50 if (m_resourceType == CachedResource::Type::Script)
51 skipContentSecurityPolicyCheck = document.contentSecurityPolicy()->allowScriptWithNonce(m_nonceAttribute);
52 else if (m_resourceType == CachedResource::Type::CSSStyleSheet)
53 skipContentSecurityPolicyCheck = document.contentSecurityPolicy()->allowStyleWithNonce(m_nonceAttribute);
54
55 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
56 if (skipContentSecurityPolicyCheck)
57 options.contentSecurityPolicyImposition = ContentSecurityPolicyImposition::SkipPolicyCheck;
58
59 String crossOriginMode = m_crossOriginMode;
60 if (m_moduleScript == ModuleScript::Yes) {
61 if (crossOriginMode.isNull())
62 crossOriginMode = ScriptElementCachedScriptFetcher::defaultCrossOriginModeForModule;
63 }
64 if (m_resourceType == CachedResource::Type::Script || m_resourceType == CachedResource::Type::ImageResource)
65 options.referrerPolicy = m_referrerPolicy;
66 auto request = createPotentialAccessControlRequest(completeURL(document), WTFMove(options), document, crossOriginMode);
67 request.setInitiator(m_initiator);
68
69 if (m_scriptIsAsync && m_resourceType == CachedResource::Type::Script && m_moduleScript == ModuleScript::No)
70 request.setPriority(DefaultResourceLoadPriority::asyncScript);
71
72 return request;
73}
74
75void HTMLResourcePreloader::preload(PreloadRequestStream requests)
76{
77 for (auto& request : requests)
78 preload(WTFMove(request));
79}
80
81void HTMLResourcePreloader::preload(std::unique_ptr<PreloadRequest> preload)
82{
83 ASSERT(m_document.frame());
84 ASSERT(m_document.renderView());
85 if (!preload->media().isEmpty() && !MediaQueryEvaluator::mediaAttributeMatches(m_document, preload->media()))
86 return;
87
88 m_document.cachedResourceLoader().preload(preload->resourceType(), preload->resourceRequest(m_document));
89}
