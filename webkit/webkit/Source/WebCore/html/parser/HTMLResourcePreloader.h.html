31namespace WebCore {
32
33class PreloadRequest {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 enum class ModuleScript {
37 Yes,
38 No,
39 };
40 PreloadRequest(const String& initiator, const String& resourceURL, const URL& baseURL, CachedResource::Type resourceType, const String& mediaAttribute, ModuleScript moduleScript, const ReferrerPolicy& referrerPolicy)
41 : m_initiator(initiator)
42 , m_resourceURL(resourceURL)
43 , m_baseURL(baseURL.isolatedCopy())
44 , m_resourceType(resourceType)
45 , m_mediaAttribute(mediaAttribute)
46 , m_moduleScript(moduleScript)
47 , m_referrerPolicy(referrerPolicy)
48 {
49 }
50
51 CachedResourceRequest resourceRequest(Document&);
52
53 const String& charset() const { return m_charset; }
54 const String& media() const { return m_mediaAttribute; }
55 void setCharset(const String& charset) { m_charset = charset.isolatedCopy(); }
56 void setCrossOriginMode(const String& mode) { m_crossOriginMode = mode; }
57 void setNonce(const String& nonce) { m_nonceAttribute = nonce; }
58 void setScriptIsAsync(bool value) { m_scriptIsAsync = value; }
59 CachedResource::Type resourceType() const { return m_resourceType; }
60
61private:
62 URL completeURL(Document&);
63
64 String m_initiator;
65 String m_resourceURL;
66 URL m_baseURL;
67 String m_charset;
68 CachedResource::Type m_resourceType;
69 String m_mediaAttribute;
70 String m_crossOriginMode;
71 String m_nonceAttribute;
72 bool m_scriptIsAsync { false };
73 ModuleScript m_moduleScript;
74 ReferrerPolicy m_referrerPolicy;
75};
76
77typedef Vector<std::unique_ptr<PreloadRequest>> PreloadRequestStream;
78
79class HTMLResourcePreloader : public CanMakeWeakPtr<HTMLResourcePreloader> {
80 WTF_MAKE_NONCOPYABLE(HTMLResourcePreloader); WTF_MAKE_FAST_ALLOCATED;
81public:
82 explicit HTMLResourcePreloader(Document& document)
83 : m_document(document)
84 {
85 }
86
87 void preload(PreloadRequestStream);
88 void preload(std::unique_ptr<PreloadRequest>);
89
90private:
91 Document& m_document;
92};
