228inline HTMLTreeBuilder::~HTMLTreeBuilder()
229{
230#if ASSERT_ENABLED
231 ASSERT(!m_destroyed);
232 ASSERT(!m_destructionProhibited);
233 m_destroyed = true;
234#endif
235}
236
237inline void HTMLTreeBuilder::setShouldSkipLeadingNewline(bool shouldSkip)
238{
239 ASSERT(!m_destroyed);
240 m_shouldSkipLeadingNewline = shouldSkip;
241}
242
243inline bool HTMLTreeBuilder::isParsingFragment() const
244{
245 ASSERT(!m_destroyed);
246 return !!m_fragmentContext.fragment();
247}
248
249inline bool HTMLTreeBuilder::hasParserBlockingScriptWork() const
250{
251 ASSERT(!m_destroyed);
252 ASSERT(!(m_scriptToProcess && m_customElementToConstruct));
253 return m_scriptToProcess || m_customElementToConstruct;
254}
255
256inline DocumentFragment* HTMLTreeBuilder::FragmentParsingContext::fragment() const
257{
258 return m_fragment;
259}
