39template<typename CharacterType> inline bool isNotASCIISpace(CharacterType c)
40{
41 return !isASCIISpace(c);
42}

44template<typename CharacterType, typename DelimiterType> bool skipExactly(const CharacterType*& position, const CharacterType* end, DelimiterType delimiter)
45{
46 if (position < end && *position == delimiter) {
47 ++position;
48 return true;
49 }
50 return false;
51}

53template<typename CharacterType, typename DelimiterType> bool skipExactly(StringParsingBuffer<CharacterType>& buffer, DelimiterType delimiter)
54{
55 if (buffer.hasCharactersRemaining() && *buffer == delimiter) {
56 ++buffer;
57 return true;
58 }
59 return false;
60}

171template<typename CharacterType, unsigned lowercaseLettersArraySize> bool skipExactlyIgnoringASCIICase(const CharacterType*& position, const CharacterType* end, const char (&lowercaseLetters)[lowercaseLettersArraySize])
172{
173 constexpr auto lowercaseLettersLength = lowercaseLettersArraySize - 1;
174
175 if (position + lowercaseLettersLength > end)
176 return false;
177 if (!WTF::equalLettersIgnoringASCIICase(position, lowercaseLettersLength, lowercaseLetters))
178 return false;
179 position += lowercaseLettersLength;
180 return true;
181}
182
183template<typename CharacterType, unsigned lowercaseLettersArraySize> bool skipExactlyIgnoringASCIICase(StringParsingBuffer<CharacterType>& buffer, const char (&lowercaseLetters)[lowercaseLettersArraySize])
184{
185 constexpr auto lowercaseLettersLength = lowercaseLettersArraySize - 1;
186
187 if (buffer.lengthRemaining() < lowercaseLettersLength)
188 return false;
189 if (!WTF::equalLettersIgnoringASCIICase(buffer.position(), lowercaseLettersLength, lowercaseLetters))
190 return false;
191 buffer += lowercaseLettersLength;
192 return true;
193}

195template<typename CharacterType, unsigned characterCount> constexpr bool skipCharactersExactly(const CharacterType*& ptr, const CharacterType* end, const CharacterType(&str)[characterCount])
196{
197 if (end - ptr < characterCount)
198 return false;
199 if (memcmp(str, ptr, sizeof(CharacterType) * characterCount))
200 return false;
201 ptr += characterCount;
202 return true;
203}
204
205template<typename CharacterType, unsigned characterCount> constexpr bool skipCharactersExactly(StringParsingBuffer<CharacterType>& buffer, const CharacterType(&str)[characterCount])
206{
207 if (buffer.lengthRemaining() < characterCount)
208 return false;
209 if (memcmp(str, buffer.position(), sizeof(CharacterType) * characterCount))
210 return false;
211 buffer += characterCount;
212 return true;
213}
