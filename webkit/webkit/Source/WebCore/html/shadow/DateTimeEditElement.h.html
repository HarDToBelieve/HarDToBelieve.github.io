34#include <wtf/WeakPtr.h>

36namespace WebCore {
37
38class Locale;
39
40class DateTimeEditElement final : public HTMLDivElement, public DateTimeFieldElement::FieldOwner {
41 WTF_MAKE_ISO_ALLOCATED(DateTimeEditElement);
42public:
43 class EditControlOwner : public CanMakeWeakPtr<EditControlOwner> {
44 public:
45 virtual ~EditControlOwner();
46 virtual void didBlurFromControl() = 0;
47 virtual void didChangeValueFromControl() = 0;
48 virtual String formatDateTimeFieldsState(const DateTimeFieldsState&) const = 0;
49 virtual bool isEditControlOwnerDisabled() const = 0;
50 virtual bool isEditControlOwnerReadOnly() const = 0;
51 virtual AtomString localeIdentifier() const = 0;
52 };
53
54 struct LayoutParameters {
55 String dateTimeFormat;
56 String fallbackDateTimeFormat;
57 Locale& locale;
58 bool shouldHaveMillisecondField { false };
59
60 LayoutParameters(Locale& locale)
61 : locale(locale)
62 {
63 }
64 };
65
66 static Ref<DateTimeEditElement> create(Document&, EditControlOwner&);
67
68 virtual ~DateTimeEditElement();
69 void addField(Ref<DateTimeFieldElement>);
70 Element& fieldsWrapperElement() const;
71 void focusByOwner();
72 void resetFields();
73 void setEmptyValue(const LayoutParameters&);
74 void setValueAsDate(const LayoutParameters&, const DateComponents&);
75 String value() const;
76
77private:
78 // Datetime can be represented by at most 8 fields:
79 // 1. year
80 // 2. month
81 // 3. day-of-month
82 // 4. hour
83 // 5. minute
84 // 6. second
85 // 7. millisecond
86 // 8. AM/PM
87 static constexpr int maximumNumberOfFields = 8;
88
89 DateTimeEditElement(Document&, EditControlOwner&);
90
91 size_t fieldIndexOf(const DateTimeFieldElement&) const;
92 DateTimeFieldElement* focusedFieldElement() const;
93 void layout(const LayoutParameters&);
94 DateTimeFieldsState valueAsDateTimeFieldsState() const;
95
96 bool focusOnNextFocusableField(size_t startIndex);
97
98 // DateTimeFieldElement::FieldOwner functions:
99 void didBlurFromField(Event&) final;
100 void fieldValueChanged() final;
101 bool focusOnNextField(const DateTimeFieldElement&) final;
102 bool focusOnPreviousField(const DateTimeFieldElement&) final;
103 bool isFieldOwnerDisabled() const final;
104 bool isFieldOwnerReadOnly() const final;
105 AtomString localeIdentifier() const final;
106 const GregorianDateTime& placeholderDate() const final;
107
108 Vector<Ref<DateTimeFieldElement>, maximumNumberOfFields> m_fields;
109 WeakPtr<EditControlOwner> m_editControlOwner;
110 GregorianDateTime m_placeholderDate;
111};

113} // namespace WebCore
