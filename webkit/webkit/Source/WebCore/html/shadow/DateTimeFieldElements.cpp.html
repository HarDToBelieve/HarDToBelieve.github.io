76 return element;

78
79void DateTimeHourFieldElement::populateDateTimeFieldsState(DateTimeFieldsState& state)
80{
81 if (!hasValue())
82 return;
83
84 int value = valueAsInteger();
85
86 switch (maximum()) {
87 case 11:
88 state.hour = value ?: 12;
89 return;
90 case 12:
91 state.hour = value;
92 return;
93 case 23:
94 state.hour = (value % 12) ?: 12;
95 state.meridiem = value >= 12 ? DateTimeFieldsState::Meridiem::PM : DateTimeFieldsState::Meridiem::AM;
96 return;
97 case 24:
98 if (value == 24) {
99 state.hour = 12;
100 state.meridiem = DateTimeFieldsState::Meridiem::AM;
101 return;
102 }
103 state.hour = (value % 12) ?: 12;
104 state.meridiem = value >= 12 ? DateTimeFieldsState::Meridiem::PM : DateTimeFieldsState::Meridiem::AM;
105 return;
106 }
107}
108
109void DateTimeHourFieldElement::setValueAsDate(const DateComponents& date)
110{
111 int hour = date.hour();
112
113 switch (maximum()) {
114 case 11:
115 setValueAsInteger(hour % 12);
116 return;
117 case 12:
118 setValueAsInteger((hour % 12) ?: 12);
119 return;
120 case 23:
121 setValueAsInteger(hour);
122 return;
123 case 24:
124 setValueAsInteger(hour + 1);
125 return;
126 }
127}
