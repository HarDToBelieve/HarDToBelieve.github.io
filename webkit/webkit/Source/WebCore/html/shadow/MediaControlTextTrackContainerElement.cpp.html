93void MediaControlTextTrackContainerElement::updateDisplay()
94{
95 if (m_mediaElement && !m_mediaElement->closedCaptionsVisible())
96 removeChildren();
97
98 // 1. If the media element is an audio element, or is another playback
99 // mechanism with no rendering area, abort these steps. There is nothing to
100 // render.
101 if (!m_mediaElement || !m_mediaElement->isVideo() || m_videoDisplaySize.size().isEmpty())
102 return;
103
104 // 2. Let video be the media element or other playback mechanism.
105 HTMLVideoElement& video = downcast<HTMLVideoElement>(*m_mediaElement);
106
107 // 3. Let output be an empty list of absolutely positioned CSS block boxes.
108
109 // 4. If the user agent is exposing a user interface for video, add to
110 // output one or more completely transparent positioned CSS block boxes that
111 // cover the same region as the user interface.
112
113 // 5. If the last time these rules were run, the user agent was not exposing
114 // a user interface for video, but now it is, let reset be true. Otherwise,
115 // let reset be false.
116
117 // There is nothing to be done explicitly for 4th and 5th steps, as
118 // everything is handled through CSS. The caption box is on top of the
119 // controls box, in a container set with the -webkit-box display property.
120
121 // 6. Let tracks be the subset of video's list of text tracks that have as
122 // their rules for updating the text track rendering these rules for
123 // updating the display of WebVTT text tracks, and whose text track mode is
124 // showing or showing by default.
125 // 7. Let cues be an empty list of text track cues.
126 // 8. For each track track in tracks, append to cues all the cues from
127 // track's list of cues that have their text track cue active flag set.
128 CueList activeCues = video.currentlyActiveCues();
129
130 // 9. If reset is false, then, for each text track cue cue in cues: if cue's
131 // text track cue display state has a set of CSS boxes, then add those boxes
132 // to output, and remove cue from cues.
133
134 // There is nothing explicitly to be done here, as all the caching occurs
135 // within the TextTrackCue instance itself. If parameters of the cue change,
136 // the display tree is cleared.
137
138 // If the number of CSS boxes in the output is less than the number of cues
139 // we wish to render (e.g., we are adding another cue in a set of roll-up
140 // cues), remove all the existing CSS boxes representing the cues and re-add
141 // them so that the new cue is at the bottom.
142 // FIXME: Calling countChildNodes() here is inefficient. We don't need to
143 // traverse all children just to check if there are less children than cues.
144 if (countChildNodes() < activeCues.size())
145 removeChildren();
146
147 activeCues.removeAllMatching([] (CueInterval& cueInterval) {
148 RefPtr<TextTrackCue> cue = cueInterval.data();
149 return !cue->track()
150 || !cue->track()->isRendered()
151 || cue->track()->mode() == TextTrack::Mode::Disabled
152 || !cue->isActive()
153 || !cue->isRenderable();
154 });
155
156 // Sort the active cues for the appropriate display order. For example, for roll-up
157 // or paint-on captions, we need to add the cues in reverse chronological order,
158 // so that the newest captions appear at the bottom.
159 std::sort(activeCues.begin(), activeCues.end(), &compareCueIntervalForDisplay);
160
161 if (m_mediaElement->closedCaptionsVisible()) {
162 // 10. For each text track cue in cues that has not yet had
163 // corresponding CSS boxes added to output, in text track cue order, run the
164 // following substeps:
165 for (auto& interval : activeCues) {
166 auto cue = interval.data();
167 cue->setFontSize(m_fontSize, m_videoDisplaySize.size(), m_fontSizeIsImportant);
168 if (is<VTTCue>(*cue))
169 processActiveVTTCue(downcast<VTTCue>(*cue));
170 else {
171 auto displayBox = cue->getDisplayTree(m_videoDisplaySize.size(), m_fontSize);
172 if (displayBox->hasChildNodes() && !contains(displayBox.get()))
173 appendChild(*displayBox);
174 }
175 }
176 }
177
178 // 11. Return output.
179 if (hasChildNodes())
180 show();
181 else
182 hide();
183
184 updateTextTrackRepresentationIfNeeded();
185 updateTextTrackStyle();
186}

188void MediaControlTextTrackContainerElement::updateTextTrackRepresentationImageIfNeeded()
189{
190 if (!m_needsToGenerateTextTrackRepresentation)
191 return;
192
193 m_needsToGenerateTextTrackRepresentation = false;

203void MediaControlTextTrackContainerElement::processActiveVTTCue(VTTCue& cue)
204{
205 DEBUG_LOG(LOGIDENTIFIER, "adding and positioning cue: \"", cue.text(), "\", start=", cue.startTime(), ", end=", cue.endTime());
206 Ref<TextTrackCueBox> displayBox = *cue.getDisplayTree(m_videoDisplaySize.size(), m_fontSize);
207
208 if (auto region = cue.track()->regions()->getRegionById(cue.regionId())) {
209 // Let region be the WebVTT region whose region identifier
210 // matches the text track cue region identifier of cue.
211 Ref<HTMLDivElement> regionNode = region->getDisplayTree();
212
213 if (!contains(regionNode.ptr()))
214 appendChild(region->getDisplayTree());
215
216 region->appendTextTrackCueBox(WTFMove(displayBox));
217 } else {
218 // If cue has an empty text track cue region identifier or there is no
219 // WebVTT region whose region identifier is identical to cue's text
220 // track cue region identifier, run the following substeps:
221 if (displayBox->hasChildNodes() && !contains(displayBox.ptr())) {
222 // Note: the display tree of a cue is removed when the active flag of the cue is unset.
223 appendChild(displayBox);
224 }
225 }
226}

228void MediaControlTextTrackContainerElement::updateActiveCuesFontSize()
229{
230 if (!document().page())
231 return;
232
233 if (!m_mediaElement)
234 return;
235
236 float smallestDimension = std::min(m_videoDisplaySize.size().height(), m_videoDisplaySize.size().width());
237 float fontScale = document().page()->group().captionPreferences().captionFontSizeScaleAndImportance(m_fontSizeIsImportant);
238 m_fontSize = lroundf(smallestDimension * fontScale);
239
240 for (auto& activeCue : m_mediaElement->currentlyActiveCues()) {
241 RefPtr<TextTrackCue> cue = activeCue.data();
242 if (cue->isRenderable())
243 cue->setFontSize(m_fontSize, m_videoDisplaySize.size(), m_fontSizeIsImportant);
244 }
245}

247void MediaControlTextTrackContainerElement::updateTextStrokeStyle()
248{
249 if (!document().page())
250 return;
251
252 if (!m_mediaElement)
253 return;
254
255 String language;
256
257 // FIXME: Since it is possible to have more than one text track enabled, the following code may not find the correct language.
258 // The default UI only allows a user to enable one track at a time, so it should be OK for now, but we should consider doing
259 // this differently, see <https://bugs.webkit.org/show_bug.cgi?id=169875>.
260 if (auto* tracks = m_mediaElement->textTracks()) {
261 for (unsigned i = 0; i < tracks->length(); ++i) {
262 auto track = tracks->item(i);
263 if (track && track->mode() == TextTrack::Mode::Showing) {
264 language = track->validBCP47Language();
265 break;
266 }
267 }
268 }
269
270 float strokeWidth;
271 bool important;
272
273 // FIXME: find a way to set this property in the stylesheet like the other user style preferences, see <https://bugs.webkit.org/show_bug.cgi?id=169874>.
274 if (document().page()->group().captionPreferences().captionStrokeWidthForFont(m_fontSize, language, strokeWidth, important))
275 setInlineStyleProperty(CSSPropertyStrokeWidth, strokeWidth, CSSUnitType::CSS_PX, important);
276}

278void MediaControlTextTrackContainerElement::updateTextTrackRepresentationIfNeeded()
279{
280 if (!m_mediaElement)
281 return;
282
283 auto requiresTextTrackRepresentation = m_mediaElement->requiresTextTrackRepresentation();
284 if (!hasChildNodes() || !requiresTextTrackRepresentation) {
285 if (m_textTrackRepresentation) {
286 if (!requiresTextTrackRepresentation)
287 clearTextTrackRepresentation();
288 else
289 m_textTrackRepresentation->setHidden(true);
290 }
291 return;
292 }
293
294 if (!m_textTrackRepresentation) {
295 ALWAYS_LOG(LOGIDENTIFIER);
296
297 m_textTrackRepresentation = TextTrackRepresentation::create(*this);
298 if (document().page())
299 m_textTrackRepresentation->setContentScale(document().page()->deviceScaleFactor());
300 m_mediaElement->setTextTrackRepresentation(m_textTrackRepresentation.get());
301 }
302
303 m_needsToGenerateTextTrackRepresentation = true;
304}

318void MediaControlTextTrackContainerElement::updateTextTrackStyle()
319{
320 if (m_textTrackRepresentation) {
321 setInlineStyleProperty(CSSPropertyPosition, CSSValueAbsolute);
322 setInlineStyleProperty(CSSPropertyWidth, m_videoDisplaySize.size().width(), CSSUnitType::CSS_PX);
323 setInlineStyleProperty(CSSPropertyHeight, m_videoDisplaySize.size().height(), CSSUnitType::CSS_PX);
324 setInlineStyleProperty(CSSPropertyLeft, 0, CSSUnitType::CSS_PX);
325 setInlineStyleProperty(CSSPropertyTop, 0, CSSUnitType::CSS_PX);

330 removeInlineStyleProperty(CSSPropertyWidth);
331 removeInlineStyleProperty(CSSPropertyHeight);
332 removeInlineStyleProperty(CSSPropertyLeft);
333 removeInlineStyleProperty(CSSPropertyTop);
334}

348bool MediaControlTextTrackContainerElement::updateVideoDisplaySize()
349{
350 if (!document().page())
351 return false;
352
353 if (!m_mediaElement)
354 return false;
355
356 IntRect videoBox;
357 if (m_textTrackRepresentation) {
358 videoBox = m_textTrackRepresentation->bounds();
359 float deviceScaleFactor = document().page()->deviceScaleFactor();
360 videoBox.setWidth(videoBox.width() * deviceScaleFactor);
361 videoBox.setHeight(videoBox.height() * deviceScaleFactor);
362 } else {
363 if (!is<RenderVideo>(m_mediaElement->renderer()))
364 return false;
365 videoBox = downcast<RenderVideo>(*m_mediaElement->renderer()).videoBox();
366 }
367
368 if (m_videoDisplaySize == videoBox)
369 return false;
370
371 m_videoDisplaySize = videoBox;
372 return true;
373}

375void MediaControlTextTrackContainerElement::updateSizes(ForceUpdate force)
376{
377 if (!updateVideoDisplaySize() && force != ForceUpdate::Yes)
378 return;
379
380 if (!document().page() || !m_mediaElement)
381 return;
382
383 m_mediaElement->syncTextTrackBounds();
384
385 updateActiveCuesFontSize();
386 updateTextStrokeStyle();
387 for (auto& activeCue : m_mediaElement->currentlyActiveCues())
388 activeCue.data()->recalculateStyles();
389
390 document().eventLoop().queueTask(TaskSource::MediaElement, [weakThis = makeWeakPtr(*this)] () {
391 if (weakThis)
392 weakThis->updateDisplay();
393 });
394}

396RefPtr<Image> MediaControlTextTrackContainerElement::createTextTrackRepresentationImage()
397{
398 if (!hasChildNodes())
399 return nullptr;
400
401 RefPtr<Frame> frame = document().frame();
402 if (!frame)
403 return nullptr;
404
405 document().updateLayout();
406
407 auto* renderer = this->renderer();
408 if (!renderer)
409 return nullptr;
410
411 if (!renderer->hasLayer())
412 return nullptr;
413
414 RenderLayer* layer = downcast<RenderLayerModelObject>(*renderer).layer();
415
416 float deviceScaleFactor = 1;
417 if (Page* page = document().page())
418 deviceScaleFactor = page->deviceScaleFactor();
419
420 IntRect paintingRect = IntRect(IntPoint(), layer->size());
421
422 // FIXME (149422): This buffer should not be unconditionally unaccelerated.
423 auto buffer = ImageBuffer::create(paintingRect.size(), RenderingMode::Unaccelerated, deviceScaleFactor, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
424 if (!buffer)
425 return nullptr;
426
427 auto paintFlags = RenderLayer::paintLayerPaintingCompositingAllPhasesFlags();
428 paintFlags.add(RenderLayer::PaintLayerFlag::TemporaryClipRects);
429 layer->paint(buffer->context(), paintingRect, LayoutSize(), { PaintBehavior::FlattenCompositingLayers, PaintBehavior::Snapshotting }, nullptr, paintFlags);
430
431 return ImageBuffer::sinkIntoImage(WTFMove(buffer));
432}

450bool MediaControlTextTrackContainerElement::isShowing() const
451{

453
454 // Following the code from show() and hide() above, we only have
455 // to check for the presense of inline display.
