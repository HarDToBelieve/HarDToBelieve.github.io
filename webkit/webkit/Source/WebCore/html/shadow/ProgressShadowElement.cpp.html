45ProgressShadowElement::ProgressShadowElement(Document& document)
46 : HTMLDivElement(HTMLNames::divTag, document)
47{
48}
49
50HTMLProgressElement* ProgressShadowElement::progressElement() const
51{
52 return downcast<HTMLProgressElement>(shadowHost());
53}
54
55bool ProgressShadowElement::rendererIsNeeded(const RenderStyle& style)
56{
57 RenderObject* progressRenderer = progressElement()->renderer();
58 return progressRenderer && !progressRenderer->style().hasAppearance() && HTMLDivElement::rendererIsNeeded(style);
59}
60
61ProgressInnerElement::ProgressInnerElement(Document& document)
62 : ProgressShadowElement(document)
63{
64}
65
66RenderPtr<RenderElement> ProgressInnerElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
67{
68 return createRenderer<RenderProgress>(*this, WTFMove(style));
69}
70
71bool ProgressInnerElement::rendererIsNeeded(const RenderStyle& style)
72{
73 RenderObject* progressRenderer = progressElement()->renderer();
74 return progressRenderer && !progressRenderer->style().hasAppearance() && HTMLDivElement::rendererIsNeeded(style);
75}

92Ref<ProgressInnerElement> ProgressInnerElement::create(Document& document)
93{
94 static MainThreadNeverDestroyed<const AtomString> webkitProgressInnerElementName("-webkit-progress-inner-element", AtomString::ConstructFromLiteral);
95 Ref<ProgressInnerElement> result = adoptRef(*new ProgressInnerElement(document));
96 result->setPseudo(webkitProgressInnerElementName);
97 return result;
98}
