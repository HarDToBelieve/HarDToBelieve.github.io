27#pragma once
28
29#include "HTMLDivElement.h"
30#include "PopupOpeningObserver.h"
31#include "Timer.h"
32
33namespace WebCore {
34
35class SpinButtonElement final : public HTMLDivElement, public PopupOpeningObserver {
36 WTF_MAKE_ISO_ALLOCATED(SpinButtonElement);
37public:
38 enum UpDownState {
39 Indeterminate, // Hovered, but the event is not handled.
40 Down,
41 Up,
42 };
43
44 class SpinButtonOwner {
45 public:
46 virtual ~SpinButtonOwner() = default;
47 virtual void focusAndSelectSpinButtonOwner() = 0;
48 virtual bool shouldSpinButtonRespondToMouseEvents() = 0;
49 virtual bool shouldSpinButtonRespondToWheelEvents() = 0;
50 virtual void spinButtonStepDown() = 0;
51 virtual void spinButtonStepUp() = 0;
52 };
53
54 // The owner of SpinButtonElement must call removeSpinButtonOwner
55 // because SpinButtonElement can be outlive SpinButtonOwner
56 // implementation, e.g. during event handling.
57 static Ref<SpinButtonElement> create(Document&, SpinButtonOwner&);
58 UpDownState upDownState() const { return m_upDownState; }
59 void releaseCapture();
60 void removeSpinButtonOwner() { m_spinButtonOwner = 0; }
61
62 void step(int amount);
63
64 bool willRespondToMouseMoveEvents() override;
65 bool willRespondToMouseClickEvents() override;
66
67 void forwardEvent(Event&);
68
69private:
70 SpinButtonElement(Document&, SpinButtonOwner&);
71
72 void willDetachRenderers() override;
73 bool isSpinButtonElement() const override { return true; }
74 bool isDisabledFormControl() const override { return shadowHost() && shadowHost()->isDisabledFormControl(); }
75 bool matchesReadWritePseudoClass() const override;
76 void defaultEventHandler(Event&) override;
77 void willOpenPopup() override;
78 void doStepAction(int);
79 void startRepeatingTimer();
80 void stopRepeatingTimer();
81 void repeatingTimerFired();
82 void setHovered(bool, Style::InvalidationScope) override;
83 bool shouldRespondToMouseEvents();
84 bool isMouseFocusable() const override { return false; }
85
86 SpinButtonOwner* m_spinButtonOwner;
87 bool m_capturing;
88 UpDownState m_upDownState;
89 UpDownState m_pressStartingState;
90 Timer m_repeatingTimer;
91};
92
93} // namespace WebCore
94
95SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::SpinButtonElement)
96 static bool isType(const WebCore::Element& element) { return element.isSpinButtonElement(); }
97 static bool isType(const WebCore::Node& node) { return is<WebCore::Element>(node) && isType(downcast<WebCore::Element>(node)); }
98SPECIALIZE_TYPE_TRAITS_END()
