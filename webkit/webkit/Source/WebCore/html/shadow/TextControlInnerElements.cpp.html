106std::optional<Style::ElementStyle> TextControlInnerElement::resolveCustomStyle(const RenderStyle&, const RenderStyle* shadowHostStyle)
107{
108 auto newStyle = RenderStyle::createPtr();
109 newStyle->inheritFrom(*shadowHostStyle);
110 newStyle->setFlexGrow(1);
111 newStyle->setMinWidth(Length { 0, LengthType::Fixed }); // Needed for correct shrinking.
112 newStyle->setDisplay(DisplayType::Block);
113 newStyle->setDirection(TextDirection::LTR);
114 // We don't want the shadow DOM to be editable, so we set this block to read-only in case the input itself is editable.
115 newStyle->setUserModify(UserModify::ReadOnly);
116
117 if (isStrongPasswordTextField(shadowHost())) {
118 newStyle->setFlexShrink(0);
119 newStyle->setTextOverflow(TextOverflow::Clip);
120 newStyle->setOverflowX(Overflow::Hidden);
121 newStyle->setOverflowY(Overflow::Hidden);
122
123 // Set "flex-basis: 1em". Note that CSSPrimitiveValue::computeLengthInt() only needs the element's
124 // style to calculate em lengths. Since the element might not be in a document, just pass nullptr
125 // for the root element style, the parent element style, and the render view.
126 auto emSize = CSSPrimitiveValue::create(1, CSSUnitType::CSS_EMS);
127 int pixels = emSize->computeLength<int>(CSSToLengthConversionData { newStyle.get(), nullptr, nullptr, nullptr, 1.0, std::nullopt });
128 newStyle->setFlexBasis(Length { pixels, LengthType::Fixed });
129 }
130
131 return { WTFMove(newStyle) };
132}

150void TextControlInnerTextElement::updateInnerTextElementEditabilityImpl(bool isEditable, bool initialization)
151{
152 static MainThreadNeverDestroyed<const AtomString> plainTextOnlyName("plaintext-only", AtomString::ConstructFromLiteral);
153 static MainThreadNeverDestroyed<const AtomString> falseName("false", AtomString::ConstructFromLiteral);
154 const auto& value = isEditable ? plainTextOnlyName.get() : falseName.get();
155 if (initialization) {
156 Vector<Attribute> attributes { Attribute(contenteditableAttr, value) };
157 parserSetAttributes(attributes);
158 } else
159 setAttributeWithoutSynchronization(contenteditableAttr, value);
160}

162void TextControlInnerTextElement::defaultEventHandler(Event& event)
163{
164 // FIXME: In the future, we should add a way to have default event listeners.
165 // Then we would add one to the text field's inner div, and we wouldn't need this subclass.
166 // Or possibly we could just use a normal event listener.
167 if (event.isBeforeTextInsertedEvent()) {
168 // A TextControlInnerTextElement can have no host if its been detached,
169 // but kept alive by an EditCommand. In this case, an undo/redo can
170 // cause events to be sent to the TextControlInnerTextElement. To
171 // prevent an infinite loop, we must check for this case before sending
172 // the event up the chain.
173 if (auto host = makeRefPtr(shadowHost()))
174 host->defaultEventHandler(event);
175 }
176 if (!event.defaultHandled())
177 HTMLDivElement::defaultEventHandler(event);
178}

212std::optional<Style::ElementStyle> TextControlPlaceholderElement::resolveCustomStyle(const RenderStyle& parentStyle, const RenderStyle* shadowHostStyle)
213{
214 auto style = resolveStyle(&parentStyle);
215
216 auto& controlElement = downcast<HTMLTextFormControlElement>(*containingShadowRoot()->host());
217 style.renderStyle->setDisplay(controlElement.isPlaceholderVisible() ? DisplayType::Block : DisplayType::None);
218
219 if (is<HTMLInputElement>(controlElement)) {
220 auto& inputElement = downcast<HTMLInputElement>(controlElement);
221 style.renderStyle->setTextOverflow(inputElement.shouldTruncateText(*shadowHostStyle) ? TextOverflow::Ellipsis : TextOverflow::Clip);
222 }
223 return style;
224}

242std::optional<Style::ElementStyle> SearchFieldResultsButtonElement::resolveCustomStyle(const RenderStyle& parentStyle, const RenderStyle* shadowHostStyle)
243{
244 auto input = makeRefPtr(downcast<HTMLInputElement>(shadowHost()));
245 if (input && input->maxResults() >= 0)
246 return std::nullopt;
247
248 if (shadowHostStyle && shadowHostStyle->appearance() != SearchFieldPart) {
249 SetForScope<bool> canAdjustStyleForAppearance(m_canAdjustStyleForAppearance, false);
250 return resolveStyle(&parentStyle);
251 }
252
253 return std::nullopt;

256void SearchFieldResultsButtonElement::defaultEventHandler(Event& event)
257{
258 // On mousedown, bring up a menu, if needed
259 auto input = makeRefPtr(downcast<HTMLInputElement>(shadowHost()));
260 if (input && event.type() == eventNames().mousedownEvent && is<MouseEvent>(event) && downcast<MouseEvent>(event).button() == LeftButton) {
261 input->focus();
262 input->select();
263#if !PLATFORM(IOS_FAMILY)
264 document().updateStyleIfNeeded();
265
266 if (auto* renderer = input->renderer()) {
267 auto& searchFieldRenderer = downcast<RenderSearchField>(*renderer);
268 if (searchFieldRenderer.popupIsVisible())
269 searchFieldRenderer.hidePopup();
270 else if (input->maxResults() > 0)
271 searchFieldRenderer.showPopup();
272 }
273#endif
274 event.setDefaultHandled();
275 }
276
277 if (!event.defaultHandled())
278 HTMLDivElement::defaultEventHandler(event);
279}

317void SearchFieldCancelButtonElement::defaultEventHandler(Event& event)
318{
319 RefPtr<HTMLInputElement> input(downcast<HTMLInputElement>(shadowHost()));
320 if (!input || input->isDisabledOrReadOnly()) {
321 if (!event.defaultHandled())
322 HTMLDivElement::defaultEventHandler(event);
323 return;
324 }
325
326 if (event.type() == eventNames().mousedownEvent && is<MouseEvent>(event) && downcast<MouseEvent>(event).button() == LeftButton) {
327 input->focus();
328 input->select();
329 event.setDefaultHandled();
330 }
331
332 if (event.type() == eventNames().clickEvent) {
333 input->setValueForUser(emptyString());
334 input->onSearch();
335 event.setDefaultHandled();
336 }
337
338 if (!event.defaultHandled())
339 HTMLDivElement::defaultEventHandler(event);
340}
