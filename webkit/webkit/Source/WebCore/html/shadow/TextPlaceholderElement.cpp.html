36
37Ref<TextPlaceholderElement> TextPlaceholderElement::create(Document& document, const LayoutSize& size)
38{
39 return adoptRef(*new TextPlaceholderElement { document, size });
40}
41
42TextPlaceholderElement::TextPlaceholderElement(Document& document, const LayoutSize& size)
43 : HTMLDivElement { HTMLNames::divTag, document }
44{
45 // FIXME: Move to User Agent stylesheet. See <https://webkit.org/b/208745>.
46 setInlineStyleProperty(CSSPropertyDisplay, size.width() ? CSSValueInlineBlock : CSSValueBlock, false);
47 setInlineStyleProperty(CSSPropertyVerticalAlign, CSSValueTop, false);
48 setInlineStyleProperty(CSSPropertyVisibility, CSSValueHidden, true);
49 setInlineStyleProperty(CSSPropertyWidth, size.width(), CSSUnitType::CSS_PX);
50 setInlineStyleProperty(CSSPropertyHeight, size.height(), CSSUnitType::CSS_PX);
51}
52
53auto TextPlaceholderElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree) -> InsertedIntoAncestorResult
54{
55 if (insertionType.treeScopeChanged) {
56 if (auto shadowHost = makeRefPtr(parentOfInsertedTree.shadowHost()); is<HTMLTextFormControlElement>(shadowHost))
57 downcast<HTMLTextFormControlElement>(*shadowHost).setCanShowPlaceholder(false);
58 }
59 return HTMLDivElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
60}
61
62void TextPlaceholderElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
63{
64 if (removalType.treeScopeChanged) {
65 if (auto shadowHost = makeRefPtr(oldParentOfRemovedTree.shadowHost()); is<HTMLTextFormControlElement>(shadowHost))
66 downcast<HTMLTextFormControlElement>(*shadowHost).setCanShowPlaceholder(true);
67 }
68 HTMLDivElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
69}
