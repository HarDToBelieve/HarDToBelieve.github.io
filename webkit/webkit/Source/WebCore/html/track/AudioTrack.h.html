52 static const AtomString& alternativeKeyword();

54 static const AtomString& mainKeyword();

56 static const AtomString& translationKeyword();
57 static const AtomString& commentaryKeyword();
58
59 bool enabled() const final { return m_enabled; }
60 void setEnabled(const bool);
61
62 void clearClient() final { m_client = nullptr; }
63 AudioTrackClient* client() const { return m_client; }
64
65 size_t inbandTrackIndex() const;
66
67 const AudioTrackPrivate& privateTrack() const { return m_private; }
68 void setPrivate(AudioTrackPrivate&);

70 void setMediaElement(WeakPtr<HTMLMediaElement>) override;
71#if !RELEASE_LOG_DISABLED
72 void setLogger(const Logger&, const void*) final;
73#endif

75private:

78 bool isValidKind(const AtomString&) const final;

80 // AudioTrackPrivateClient

83 // TrackPrivateBaseClient

86 void languageChanged(const AtomString&) final;
87 void willRemove() final;

89 void updateKindFromPrivate();

91#if !RELEASE_LOG_DISABLED
