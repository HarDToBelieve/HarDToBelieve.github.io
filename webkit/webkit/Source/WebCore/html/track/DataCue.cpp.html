102RefPtr<ArrayBuffer> DataCue::data() const
103{
104 if (m_platformValue)
105 return m_platformValue->data();
106
107 if (!m_data)
108 return nullptr;
109
110 return ArrayBuffer::create(*m_data);
111}

113void DataCue::setData(ArrayBuffer& data)
114{
115 m_platformValue = nullptr;
116 m_value.clear();
117 m_data = ArrayBuffer::create(data);
118}

120bool DataCue::cueContentsMatch(const TextTrackCue& cue) const
121{
122 const DataCue* dataCue = downcast<DataCue>(&cue);
123 RefPtr<ArrayBuffer> otherData = dataCue->data();
124 if ((otherData && !m_data) || (!otherData && m_data))
125 return false;
126 if (m_data && m_data->byteLength() != otherData->byteLength())
127 return false;
128 if (m_data && m_data->data() && memcmp(m_data->data(), otherData->data(), m_data->byteLength()))
129 return false;
130
131 auto otherPlatformValue = dataCue->platformValue();
132 if ((otherPlatformValue && !m_platformValue) || (!otherPlatformValue && m_platformValue))
133 return false;
134 if (m_platformValue && !m_platformValue->isEqual(*otherPlatformValue))
135 return false;
136
137 JSC::JSValue thisValue = valueOrNull();
138 JSC::JSValue otherValue = dataCue->valueOrNull();
139 if ((otherValue && !thisValue) || (!otherValue && thisValue))
140 return false;
141 if (!JSC::JSValue::strictEqual(nullptr, thisValue, otherValue))
142 return false;
143
144 return true;
145}

147JSC::JSValue DataCue::value(JSC::JSGlobalObject& state) const
148{
149 if (m_platformValue)
150 return m_platformValue->deserialize(&state);
151
152 if (m_value)
153 return m_value.get();
154
155 return JSC::jsNull();
156}
157
158void DataCue::setValue(JSC::JSGlobalObject& state, JSC::JSValue value)
159{
160 // FIXME: this should use a SerializedScriptValue.
161 m_value.set(state.vm(), value);
162 m_platformValue = nullptr;
163 m_data = nullptr;
164}
165
166JSValue DataCue::valueOrNull() const
167{
168 if (m_value)
169 return m_value.get();
170
171 return jsNull();
172}
