38class JSValue;

41namespace WebCore {
42
43class ScriptExecutionContext;
44
45class DataCue final : public TextTrackCue {
46 WTF_MAKE_ISO_ALLOCATED(DataCue);
47public:
48 static Ref<DataCue> create(Document&, double start, double end, ArrayBuffer& data);
49 static Ref<DataCue> create(Document&, double start, double end, JSC::JSValue, const String& type);
50 static Ref<DataCue> create(Document&, const MediaTime& start, const MediaTime& end, const void* data, unsigned length);
51 static Ref<DataCue> create(Document&, const MediaTime& start, const MediaTime& end, Ref<SerializedPlatformDataCue>&&, const String& type);
52
53 virtual ~DataCue();
54
55 RefPtr<JSC::ArrayBuffer> data() const;
56 void setData(JSC::ArrayBuffer&);
57
58 const SerializedPlatformDataCue* platformValue() const { return m_platformValue.get(); }
59
60 JSC::JSValue value(JSC::JSGlobalObject&) const;
61 void setValue(JSC::JSGlobalObject&, JSC::JSValue);
62
63 String type() const { return m_type; }
64 void setType(const String& type) { m_type = type; }
65
66private:
67 DataCue(Document&, const MediaTime& start, const MediaTime& end, ArrayBuffer&, const String&);
68 DataCue(Document&, const MediaTime& start, const MediaTime& end, const void*, unsigned);
69 DataCue(Document&, const MediaTime& start, const MediaTime& end, Ref<SerializedPlatformDataCue>&&, const String&);
70 DataCue(Document&, const MediaTime& start, const MediaTime& end, JSC::JSValue, const String&);
71
72 JSC::JSValue valueOrNull() const;
73 CueType cueType() const final { return Data; }
74 bool cueContentsMatch(const TextTrackCue&) const final;
75 void toJSON(JSON::Object&) const final;
76
77 RefPtr<ArrayBuffer> m_data;
78 String m_type;
79 RefPtr<SerializedPlatformDataCue> m_platformValue;
80 // FIXME: The following use of JSC::Strong is incorrect and can lead to storage leaks
81 // due to reference cycles; we should use JSValueInWrappedObject instead.
82 // https://bugs.webkit.org/show_bug.cgi?id=201173
83 JSC::Strong<JSC::Unknown> m_value;
84};
85
