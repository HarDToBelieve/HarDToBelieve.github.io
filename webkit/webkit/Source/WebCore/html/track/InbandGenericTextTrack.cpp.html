80void InbandGenericTextTrack::updateCueFromCueData(TextTrackCueGeneric& cue, InbandGenericCue& inbandCue)
81{
82 cue.willChange();
83
84 cue.setStartTime(inbandCue.startTime());
85 MediaTime endTime = inbandCue.endTime();
86 if (endTime.isPositiveInfinite() && mediaElement())
87 endTime = mediaElement()->durationMediaTime();
88 cue.setEndTime(endTime);
89 cue.setText(inbandCue.content());
90 cue.setId(inbandCue.id());
91 cue.setBaseFontSizeRelativeToVideoHeight(inbandCue.baseFontSize());
92 cue.setFontSizeMultiplier(inbandCue.relativeFontSize());
93 cue.setFontName(inbandCue.fontName());
94
95 if (inbandCue.position() > 0)
96 cue.setPosition(std::round(inbandCue.position()));
97 if (inbandCue.line() > 0)
98 cue.setLine(std::round(inbandCue.line()));
99 if (inbandCue.size() > 0)
100 cue.setSize(std::round(inbandCue.size()));
101 if (inbandCue.backgroundColor().isValid())
102 cue.setBackgroundColor(inbandCue.backgroundColor());
103 if (inbandCue.foregroundColor().isValid())
104 cue.setForegroundColor(inbandCue.foregroundColor());
105 if (inbandCue.highlightColor().isValid())
106 cue.setHighlightColor(inbandCue.highlightColor());
107
108 if (inbandCue.align() == GenericCueData::Alignment::Start)
109 cue.setAlign("start"_s);
110 else if (inbandCue.align() == GenericCueData::Alignment::Middle)
111 cue.setAlign("middle"_s);
112 else if (inbandCue.align() == GenericCueData::Alignment::End)
113 cue.setAlign("end"_s);
114 cue.setSnapToLines(false);
115
116 cue.didChange();
117}

119void InbandGenericTextTrack::addGenericCue(InbandGenericCue& inbandCue)
120{
121 if (m_cueMap.find(inbandCue.uniqueId()))
122 return;
123
124 auto cue = TextTrackCueGeneric::create(document(), inbandCue.startTime(), inbandCue.endTime(), inbandCue.content());
125 updateCueFromCueData(cue.get(), inbandCue);
126 if (hasCue(cue, TextTrackCue::IgnoreDuration)) {
127 INFO_LOG(LOGIDENTIFIER, "ignoring already added cue: ", cue.get());
128 return;
129 }
130
131 INFO_LOG(LOGIDENTIFIER, "added cue: ", cue.get());
132
133 if (inbandCue.status() != GenericCueData::Status::Complete)
134 m_cueMap.add(inbandCue.uniqueId(), cue);
135
136 addCue(WTFMove(cue));
137}
138
139void InbandGenericTextTrack::updateGenericCue(InbandGenericCue& inbandCue)
140{
141 auto cue = makeRefPtr(m_cueMap.find(inbandCue.uniqueId()));
142 if (!cue)
143 return;
144
145 updateCueFromCueData(*cue, inbandCue);
146
147 if (inbandCue.status() == GenericCueData::Status::Complete)
148 m_cueMap.remove(inbandCue.uniqueId());
149}
150
151void InbandGenericTextTrack::removeGenericCue(InbandGenericCue& inbandCue)
152{
153 auto cue = makeRefPtr(m_cueMap.find(inbandCue.uniqueId()));
154 if (cue) {
155 INFO_LOG(LOGIDENTIFIER, *cue);
156 removeCue(*cue);
157 } else
158 INFO_LOG(LOGIDENTIFIER, "UNABLE to find cue: ", inbandCue);
159
160}
161
162ExceptionOr<void> InbandGenericTextTrack::removeCue(TextTrackCue& cue)
163{
164 auto result = TextTrack::removeCue(cue);
165 if (!result.hasException())
166 m_cueMap.remove(cue);
167 return result;
168}

187void InbandGenericTextTrack::newCuesParsed()
188{
189 for (auto& cueData : parser().takeCues()) {
190 auto cue = VTTCue::create(document(), cueData);
191 if (hasCue(cue, TextTrackCue::IgnoreDuration)) {
192 INFO_LOG(LOGIDENTIFIER, "ignoring already added cue: ", cue.get());
193 return;
194 }
195 INFO_LOG(LOGIDENTIFIER, cue.get());
196 addCue(WTFMove(cue));
197 }
198}
