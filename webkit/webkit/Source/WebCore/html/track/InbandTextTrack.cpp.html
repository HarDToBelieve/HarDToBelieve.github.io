42Ref<InbandTextTrack> InbandTextTrack::create(Document& document, TextTrackClient& client, InbandTextTrackPrivate& trackPrivate)
43{
44 switch (trackPrivate.cueFormat()) {
45 case InbandTextTrackPrivate::CueFormat::Data:
46 return InbandDataTextTrack::create(document, client, trackPrivate);
47 case InbandTextTrackPrivate::CueFormat::Generic:
48 return InbandGenericTextTrack::create(document, client, trackPrivate);
49 case InbandTextTrackPrivate::CueFormat::WebVTT:
50 return InbandWebVTTTextTrack::create(document, client, trackPrivate);
51 }
52 ASSERT_NOT_REACHED();
53 auto textTrack = InbandDataTextTrack::create(document, client, trackPrivate);
54 textTrack->suspendIfNeeded();
55 return textTrack;
56}

58InbandTextTrack::InbandTextTrack(Document& document, TextTrackClient& client, InbandTextTrackPrivate& trackPrivate)
59 : TextTrack(&document, &client, emptyAtom(), trackPrivate.id(), trackPrivate.label(), trackPrivate.language(), InBand)
60 , m_private(trackPrivate)
61{
62 m_private->setClient(this);
63 updateKindFromPrivate();
64}
65
66InbandTextTrack::~InbandTextTrack()
67{
68 m_private->setClient(nullptr);
69}

71void InbandTextTrack::setPrivate(InbandTextTrackPrivate& trackPrivate)
72{
73 if (m_private.ptr() == &trackPrivate)
74 return;
75
76 m_private->setClient(nullptr);
77 m_private = trackPrivate;
78 m_private->setClient(this);
79
80 setModeInternal(mode());
81 updateKindFromPrivate();
82 setId(m_private->id());
83}
84
85void InbandTextTrack::setMode(Mode mode)
86{
87 TextTrack::setMode(mode);
88 setModeInternal(mode);
89}

91static inline InbandTextTrackPrivate::Mode toPrivate(TextTrack::Mode mode)
92{
93 switch (mode) {
94 case TextTrack::Mode::Disabled:
95 return InbandTextTrackPrivate::Mode::Disabled;
96 case TextTrack::Mode::Hidden:
97 return InbandTextTrackPrivate::Mode::Hidden;
98 case TextTrack::Mode::Showing:
99 return InbandTextTrackPrivate::Mode::Showing;
100 }
101 ASSERT_NOT_REACHED();
102 return InbandTextTrackPrivate::Mode::Disabled;
103}
104
105void InbandTextTrack::setModeInternal(Mode mode)
106{
107 m_private->setMode(toPrivate(mode));
108}

168void InbandTextTrack::updateKindFromPrivate()
169{
170 switch (m_private->kind()) {
171 case InbandTextTrackPrivate::Kind::Subtitles:
172 setKind(Kind::Subtitles);
173 return;
174 case InbandTextTrackPrivate::Kind::Captions:
175 setKind(Kind::Captions);
176 return;
177 case InbandTextTrackPrivate::Kind::Descriptions:
178 setKind(Kind::Descriptions);
179 return;
180 case InbandTextTrackPrivate::Kind::Chapters:
181 setKind(Kind::Chapters);
182 return;
183 case InbandTextTrackPrivate::Kind::Metadata:
184 setKind(Kind::Metadata);
185 return;
186 case InbandTextTrackPrivate::Kind::Forced:
187 setKind(Kind::Forced);
188 return;
189 case InbandTextTrackPrivate::Kind::None:
190 break;
191 }
192 ASSERT_NOT_REACHED();
193}
