101TextTrack::TextTrack(ScriptExecutionContext* context, TextTrackClient* client, const AtomString& kind, const AtomString& id, const AtomString& label, const AtomString& language, TextTrackType type)
102 : TrackBase(TrackBase::TextTrack, id, label, language)
103 , ActiveDOMObject(context)
104 , m_client(client)
105 , m_trackType(type)
106{
107 if (kind == captionsKeyword())
108 m_kind = Kind::Captions;
109 else if (kind == chaptersKeyword())
110 m_kind = Kind::Chapters;
111 else if (kind == descriptionsKeyword())
112 m_kind = Kind::Descriptions;
113 else if (kind == forcedKeyword())
114 m_kind = Kind::Forced;
115 else if (kind == metadataKeyword())
116 m_kind = Kind::Metadata;
117}

189void TextTrack::setKind(Kind newKind)
190{
191 auto oldKind = m_kind;
192
193 // 10.1 kind, on setting:
194 // 1. If the value being assigned to this attribute does not match one of the text track kinds,
195 // then abort these steps.
196
197 // 2. Update this attribute to the new value.
198 m_kind = newKind;
199
200#if ENABLE(MEDIA_SOURCE)
201 // 3. If the sourceBuffer attribute on this track is not null, then queue a task to fire a simple
202 // event named change at sourceBuffer.textTracks.
203 if (m_sourceBuffer)
204 m_sourceBuffer->textTracks().scheduleChangeEvent();
205
206 // 4. Queue a task to fire a simple event named change at the TextTrackList object referenced by
207 // the textTracks attribute on the HTMLMediaElement.
208 if (mediaElement())
209 mediaElement()->ensureTextTracks().scheduleChangeEvent();
210#endif
211
212 if (oldKind != m_kind) {
213 ALWAYS_LOG(LOGIDENTIFIER, m_kind);
214 if (m_client)
215 m_client->textTrackKindChanged(*this);
216 }
217}

219void TextTrack::setKindKeywordIgnoringASCIICase(StringView keyword)
220{
221 if (keyword.isNull()) {
222 // The missing value default is the subtitles state.
223 setKind(Kind::Subtitles);
224 return;
225 }
226 if (equalLettersIgnoringASCIICase(keyword, "captions"))
227 setKind(Kind::Captions);
228 else if (equalLettersIgnoringASCIICase(keyword, "chapters"))
229 setKind(Kind::Chapters);
230 else if (equalLettersIgnoringASCIICase(keyword, "descriptions"))
231 setKind(Kind::Descriptions);
232 else if (equalLettersIgnoringASCIICase(keyword, "forced"))
233 setKind(Kind::Forced);
234 else if (equalLettersIgnoringASCIICase(keyword, "metadata"))
235 setKind(Kind::Metadata);
236 else if (equalLettersIgnoringASCIICase(keyword, "subtitles"))
237 setKind(Kind::Subtitles);
238 else {
239 // The invalid value default is the metadata state.
240 setKind(Kind::Metadata);
241 }
242}

244void TextTrack::setMode(Mode mode)
245{
246 // On setting, if the new value isn't equal to what the attribute would currently
247 // return, the new value must be processed as follows ...
248 if (m_mode == mode)
249 return;
250
251 ALWAYS_LOG(LOGIDENTIFIER, mode);
252
253 // If mode changes to disabled, remove this track's cues from the client
254 // because they will no longer be accessible from the cues() function.
255 if (mode == Mode::Disabled && m_client && m_cues)
256 m_client->textTrackRemoveCues(*this, *m_cues);
257
258 if (mode != Mode::Showing && m_cues) {
259 for (size_t i = 0; i < m_cues->length(); ++i)
260 m_cues->item(i)->removeDisplayTree();
261 }
262
263 m_mode = mode;
264
265 if (m_client)
266 m_client->textTrackModeChanged(*this);
267}

310ExceptionOr<void> TextTrack::addCue(Ref<TextTrackCue>&& cue)
311{
312 // 4.7.10.12.6 Text tracks exposing in-band metadata
313 // The UA will use DataCue to expose only text track cue objects that belong to a text track that has a text
314 // track kind of metadata.
315 // If a DataCue is added to a TextTrack via the addCue() method but the text track does not have its text
316 // track kind set to metadata, throw a InvalidNodeTypeError exception and don't add the cue to the TextTrackList
317 // of the TextTrack.
318 if (is<DataCue>(cue.get()) && m_kind != Kind::Metadata)
319 return Exception { InvalidNodeTypeError };
320
321 INFO_LOG(LOGIDENTIFIER, cue.get());
322
323 // TODO(93143): Add spec-compliant behavior for negative time values.
324 if (!cue->startMediaTime().isValid() || !cue->endMediaTime().isValid() || cue->startMediaTime() < MediaTime::zeroTime() || cue->endMediaTime() < MediaTime::zeroTime())
325 return { };
326
327 // 4.8.10.12.5 Text track API
328
329 // The addCue(cue) method of TextTrack objects, when invoked, must run the following steps:
330
331 auto cueTrack = makeRefPtr(cue->track());
332 if (cueTrack == this)
333 return { };
334
335 // 1. If the given cue is in a text track list of cues, then remove cue from that text track
336 // list of cues.
337 if (cueTrack)
338 cueTrack->removeCue(cue.get());
339
340 // 2. Add cue to the method's TextTrack object's text track's text track list of cues.
341 cue->setTrack(this);
342 ensureTextTrackCueList().add(cue.copyRef());
343
344 if (m_client)
345 m_client->textTrackAddCue(*this, cue);
346
347 return { };
348}

350ExceptionOr<void> TextTrack::removeCue(TextTrackCue& cue)
351{
352 // 4.8.10.12.5 Text track API
353
354 // The removeCue(cue) method of TextTrack objects, when invoked, must run the following steps:
355
356 // 1. If the given cue is not currently listed in the method's TextTrack
357 // object's text track's text track list of cues, then throw a NotFoundError exception.
358 if (cue.track() != this)
359 return Exception { NotFoundError };
360 if (!m_cues)
361 return Exception { InvalidStateError };
362
363 INFO_LOG(LOGIDENTIFIER, cue);
364
365 // 2. Remove cue from the method's TextTrack object's text track's text track list of cues.
366 m_cues->remove(cue);
367 cue.setIsActive(false);
368 cue.setTrack(nullptr);
369 if (m_client)
370 m_client->textTrackRemoveCue(*this, cue);
371
372 return { };
373}

375void TextTrack::removeCuesNotInTimeRanges(PlatformTimeRanges& buffered)
376{
377 ASSERT(shouldPurgeCuesFromUnbufferedRanges());
378
379 if (!m_cues)
380 return;
381
382 Vector<Ref<TextTrackCue>> toPurge;
383 for (size_t i = 0; i < m_cues->length(); ++i) {
384 auto cue = m_cues->item(i);
385 ASSERT(cue->track() == this);
386
387 PlatformTimeRanges activeCueRange { cue->startMediaTime(), cue->endMediaTime() };
388 activeCueRange.intersectWith(buffered);
389 if (!activeCueRange.length())
390 toPurge.append(*cue);
391 }
392
393 if (!toPurge.size())
394 return;
395
396 INFO_LOG(LOGIDENTIFIER, "purging ", toPurge.size());
397
398 for (auto& cue : toPurge)
399 removeCue(cue);
400}

423void TextTrack::addRegion(Ref<VTTRegion>&& region)
424{
425 auto& regionList = ensureVTTRegionList();
426
427 // 1. If the given region is in a text track list of regions, then remove
428 // region from that text track list of regions.
429 auto regionTrack = makeRefPtr(region->track());
430 if (regionTrack && regionTrack != this)
431 regionTrack->removeRegion(region.get());
432
433 // 2. If the method's TextTrack object's text track list of regions contains
434 // a region with the same identifier as region replace the values of that
435 // region's width, height, anchor point, viewport anchor point and scroll
436 // attributes with those of region.
437 auto existingRegion = makeRefPtr(regionList.getRegionById(region->id()));
438 if (existingRegion) {
439 existingRegion->updateParametersFromRegion(region);
440 return;
441 }
442
443 // Otherwise: add region to the method's TextTrack object's text track list of regions.
444 region->setTrack(this);
445 regionList.add(WTFMove(region));
446}

448ExceptionOr<void> TextTrack::removeRegion(VTTRegion& region)
449{
450 // 1. If the given region is not currently listed in the method's TextTrack
451 // object's text track list of regions, then throw a NotFoundError exception.
452 if (region.track() != this)
453 return Exception { NotFoundError };
454
455 ASSERT(m_regions);
456 m_regions->remove(region);
457 region.setTrack(nullptr);
458 return { };

522bool TextTrack::hasCue(TextTrackCue& cue, TextTrackCue::CueMatchRules match)
523{
524 if (cue.startMediaTime() < MediaTime::zeroTime() || cue.endMediaTime() < MediaTime::zeroTime())
525 return false;
526
527 if (!m_cues || !m_cues->length())
528 return false;
529
530 size_t searchStart = 0;
531 size_t searchEnd = m_cues->length();
532
533 while (1) {
534 ASSERT(searchStart <= m_cues->length());
535 ASSERT(searchEnd <= m_cues->length());
536
537 RefPtr<TextTrackCue> existingCue;
538
539 // Cues in the TextTrackCueList are maintained in start time order.
540 if (searchStart == searchEnd) {
541 if (!searchStart)
542 return false;
543
544 // If there is more than one cue with the same start time, back up to first one so we
545 // consider all of them.
546 while (searchStart >= 2 && cue.hasEquivalentStartTime(*m_cues->item(searchStart - 2)))
547 --searchStart;
548
549 bool firstCompare = true;
550 while (1) {
551 if (!firstCompare)
552 ++searchStart;
553 firstCompare = false;
554 if (searchStart > m_cues->length())
555 return false;
556
557 existingCue = m_cues->item(searchStart - 1);
558 if (!existingCue)
559 return false;
560
561 if (cue.startMediaTime() > (existingCue->startMediaTime() + startTimeVariance()))
562 return false;
563
564 if (existingCue->isEqual(cue, match))
565 return true;
566 }
567 }
568
569 size_t index = (searchStart + searchEnd) / 2;
570 existingCue = m_cues->item(index);
571 if ((cue.startMediaTime() + startTimeVariance()) < existingCue->startMediaTime() || (match != TextTrackCue::IgnoreDuration && cue.hasEquivalentStartTime(*existingCue) && cue.endMediaTime() > existingCue->endMediaTime()))
572 searchEnd = index;
573 else
574 searchStart = index + 1;
575 }
576
577 ASSERT_NOT_REACHED();
578 return false;
579}

581bool TextTrack::isMainProgramContent() const
582{
583 // "Main program" content is intrinsic to the presentation of the media file, regardless of locale. Content such as
584 // directors commentary is not "main program" because it is not essential for the presentation. HTML5 doesn't have
585 // a way to express this in a machine-reable form, it is typically done with the track label, so we assume that caption
586 // tracks are main content and all other track types are not.
587 return m_kind == Kind::Captions;
588}
