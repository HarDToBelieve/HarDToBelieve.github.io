65class TextTrackCue : public RefCounted<TextTrackCue>, public EventTargetWithInlineData, public ActiveDOMObject {

67public:
68 static const AtomString& cueShadowPseudoId();

72 static ExceptionOr<Ref<TextTrackCue>> create(Document&, double start, double end, DocumentFragment&);
73
74 TextTrack* track() const;
75 void setTrack(TextTrack*);
76
77 const String& id() const { return m_id; }
78 void setId(const String&);
79
80 double startTime() const { return startMediaTime().toDouble(); }
81 void setStartTime(double);
82
83 double endTime() const { return endMediaTime().toDouble(); }
84 void setEndTime(double);
85
86 bool pauseOnExit() const { return m_pauseOnExit; }
87 void setPauseOnExit(bool);
88
89 MediaTime startMediaTime() const { return m_startTime; }
90 void setStartTime(const MediaTime&);
91
92 MediaTime endMediaTime() const { return m_endTime; }
93 void setEndTime(const MediaTime&);
94
95 bool isActive() const;
96 virtual void setIsActive(bool);
97
98 virtual bool isOrderedBefore(const TextTrackCue*) const;
99 virtual bool isPositionedAbove(const TextTrackCue* cue) const { return isOrderedBefore(cue); }
100
101 bool hasEquivalentStartTime(const TextTrackCue&) const;
102
103 enum CueType { Generic, Data, ConvertedToWebVTT, WebVTT };
104 virtual CueType cueType() const { return CueType::Generic; }
105 virtual bool isRenderable() const;
106
107 enum CueMatchRules { MatchAllFields, IgnoreDuration };
108 bool isEqual(const TextTrackCue&, CueMatchRules) const;
109
110 void willChange();
111 virtual void didChange();
112
113 virtual RefPtr<TextTrackCueBox> getDisplayTree(const IntSize& videoSize, int fontSize);
114 virtual void removeDisplayTree();
115
116 virtual RefPtr<DocumentFragment> getCueAsHTML();
117
118 String toJSONString() const;
119
120 using RefCounted::ref;
121 using RefCounted::deref;
122
123 virtual void recalculateStyles() { m_displayTreeNeedsUpdate = true; }
124 virtual void setFontSize(int fontSize, const IntSize& videoSize, bool important);
125 virtual void updateDisplayTree(const MediaTime&) { }
126
127 unsigned cueIndex() const;
128
129protected:
130 TextTrackCue(Document&, const MediaTime& start, const MediaTime& end);
131
132 Document& ownerDocument() { return m_document; }
133
134 virtual bool cueContentsMatch(const TextTrackCue&) const;
135 virtual void toJSON(JSON::Object&) const;

137private:

140 // EventTarget
141 void refEventTarget() final { ref(); }

151 void rebuildDisplayTree();
