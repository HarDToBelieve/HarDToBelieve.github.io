74void TextTrackCueGenericBoxElement::applyCSSProperties(const IntSize& videoSize)
75{
76 RefPtr<TextTrackCueGeneric> cue = static_cast<TextTrackCueGeneric*>(getCue());
77 if (!cue)
78 return;
79
80 setInlineStyleProperty(CSSPropertyPosition, CSSValueAbsolute);
81 setInlineStyleProperty(CSSPropertyUnicodeBidi, CSSValuePlaintext);
82
83 Ref<HTMLSpanElement> cueElement = cue->element();
84
85 double textPosition = cue->calculateComputedTextPosition();
86 double linePosition = cue->calculateComputedLinePosition();
87
88 CSSValueID alignment = cue->getCSSAlignment();
89 float size = static_cast<float>(cue->getCSSSize());
90 if (cue->useDefaultPosition()) {
91 setInlineStyleProperty(CSSPropertyBottom, 0, CSSUnitType::CSS_PX);
92 setInlineStyleProperty(CSSPropertyMarginBottom, 1.0, CSSUnitType::CSS_PERCENTAGE);
93 } else {
94 setInlineStyleProperty(CSSPropertyLeft, static_cast<float>(textPosition), CSSUnitType::CSS_PERCENTAGE);
95 setInlineStyleProperty(CSSPropertyTop, static_cast<float>(linePosition), CSSUnitType::CSS_PERCENTAGE);
96
97 double authorFontSize = videoSize.height() * cue->baseFontSizeRelativeToVideoHeight() / 100.0;
98 if (!authorFontSize)
99 authorFontSize = DEFAULTCAPTIONFONTSIZE;
100
101 if (cue->fontSizeMultiplier())
102 authorFontSize *= cue->fontSizeMultiplier() / 100;
103
104 double multiplier = fontSizeFromCaptionUserPrefs() / authorFontSize;
105 double newCueSize = std::min(size * multiplier, 100.0);
106 if (cue->getWritingDirection() == VTTCue::Horizontal) {
107 setInlineStyleProperty(CSSPropertyWidth, newCueSize, CSSUnitType::CSS_PERCENTAGE);
108 if ((alignment == CSSValueMiddle || alignment == CSSValueCenter) && multiplier != 1.0)
109 setInlineStyleProperty(CSSPropertyLeft, static_cast<double>(textPosition - (newCueSize - cue->getCSSSize()) / 2), CSSUnitType::CSS_PERCENTAGE);
110 } else {
111 setInlineStyleProperty(CSSPropertyHeight, newCueSize, CSSUnitType::CSS_PERCENTAGE);
112 if ((alignment == CSSValueMiddle || alignment == CSSValueCenter) && multiplier != 1.0)
113 setInlineStyleProperty(CSSPropertyTop, static_cast<double>(linePosition - (newCueSize - cue->getCSSSize()) / 2), CSSUnitType::CSS_PERCENTAGE);
114 }
115 }
116
117 double maxSize = 100.0;
118
119 if (alignment == CSSValueEnd || alignment == CSSValueRight)
120 maxSize = textPosition;
121 else if (alignment == CSSValueStart || alignment == CSSValueLeft)
122 maxSize = 100.0 - textPosition;
123
124 if (cue->getWritingDirection() == VTTCue::Horizontal) {
125 setInlineStyleProperty(CSSPropertyMinWidth, "min-content");
126 setInlineStyleProperty(CSSPropertyMaxWidth, maxSize, CSSUnitType::CSS_PERCENTAGE);
127 } else {
128 setInlineStyleProperty(CSSPropertyMinHeight, "min-content");
129 setInlineStyleProperty(CSSPropertyMaxHeight, maxSize, CSSUnitType::CSS_PERCENTAGE);
130 }
131
132 if (cue->foregroundColor().isValid())
133 cueElement->setInlineStyleProperty(CSSPropertyColor, serializationForHTML(cue->foregroundColor()));
134 if (cue->highlightColor().isValid())
135 cueElement->setInlineStyleProperty(CSSPropertyBackgroundColor, serializationForHTML(cue->highlightColor()));
136
137 if (cue->getWritingDirection() == VTTCue::Horizontal)
138 setInlineStyleProperty(CSSPropertyHeight, CSSValueAuto);
139 else
140 setInlineStyleProperty(CSSPropertyWidth, CSSValueAuto);
141
142 if (cue->baseFontSizeRelativeToVideoHeight())
143 cue->setFontSize(cue->baseFontSizeRelativeToVideoHeight(), videoSize, false);
144
145 if (cue->getAlignment() == VTTCue::Center)
146 setInlineStyleProperty(CSSPropertyTextAlign, CSSValueCenter);
147 else if (cue->getAlignment() == VTTCue::End)
148 setInlineStyleProperty(CSSPropertyTextAlign, CSSValueEnd);
149 else
150 setInlineStyleProperty(CSSPropertyTextAlign, CSSValueStart);
151
152 if (cue->backgroundColor().isValid())
153 setInlineStyleProperty(CSSPropertyBackgroundColor, serializationForHTML(cue->backgroundColor()));
154 setInlineStyleProperty(CSSPropertyWritingMode, cue->getCSSWritingMode(), false);
155 setInlineStyleProperty(CSSPropertyWhiteSpace, CSSValuePreWrap);
156
157 // Make sure shadow or stroke is not clipped.
158 setInlineStyleProperty(CSSPropertyOverflow, CSSValueVisible);
159 cueElement->setInlineStyleProperty(CSSPropertyOverflow, CSSValueVisible);
160}

195void TextTrackCueGeneric::setFontSize(int fontSize, const IntSize& videoSize, bool important)
196{
197 if (!hasDisplayTree() || !fontSize)
198 return;
199
200 if (important || !baseFontSizeRelativeToVideoHeight()) {
201 VTTCue::setFontSize(fontSize, videoSize, important);
202 return;
203 }
204
205 double size = videoSize.height() * baseFontSizeRelativeToVideoHeight() / 100;
206 if (fontSizeMultiplier())
207 size *= fontSizeMultiplier() / 100;
208 displayTreeInternal().setInlineStyleProperty(CSSPropertyFontSize, lround(size), CSSUnitType::CSS_PX);
209}

211bool TextTrackCueGeneric::cueContentsMatch(const TextTrackCue& otherTextTrackCue) const
212{
213 auto& other = downcast<TextTrackCueGeneric>(otherTextTrackCue);
214 return VTTCue::cueContentsMatch(other)
215 && m_baseFontSizeRelativeToVideoHeight == other.m_baseFontSizeRelativeToVideoHeight
216 && m_fontSizeMultiplier == other.m_fontSizeMultiplier
217 && m_fontName == other.m_fontName
218 && m_foregroundColor == other.m_foregroundColor
219 && m_backgroundColor == other.m_backgroundColor;
220}

222bool TextTrackCueGeneric::isOrderedBefore(const TextTrackCue* that) const
223{
224 if (VTTCue::isOrderedBefore(that))
225 return true;
226
227 if (is<TextTrackCueGeneric>(*that) && startTime() == that->startTime() && endTime() == that->endTime()) {
228 // Further order generic cues by their calculated line value.
229 auto thisPosition = getPositionCoordinates();
230 auto thatPosition = downcast<TextTrackCueGeneric>(*that).getPositionCoordinates();
231 return thisPosition.second > thatPosition.second || (thisPosition.second == thatPosition.second && thisPosition.first < thatPosition.first);
232 }
233
234 return false;
235}
236
237bool TextTrackCueGeneric::isPositionedAbove(const TextTrackCue* that) const
238{
239 if (is<TextTrackCueGeneric>(*that)) {
240 if (startTime() == that->startTime() && endTime() == that->endTime()) {
241 // Further order generic cues by their calculated line value.
242 auto thisPosition = getPositionCoordinates();
243 auto thatPosition = downcast<TextTrackCueGeneric>(*that).getPositionCoordinates();
244 return thisPosition.second > thatPosition.second || (thisPosition.second == thatPosition.second && thisPosition.first < thatPosition.first);
245 }
246 return startTime() > that->startTime();
247 }
248
249 return VTTCue::isOrderedBefore(that);
250}

252void TextTrackCueGeneric::toJSON(JSON::Object& object) const
253{
254 if (m_foregroundColor.isValid())
255 object.setString("foregroundColor"_s, serializationForHTML(m_foregroundColor));
256 if (m_backgroundColor.isValid())
257 object.setString("backgroundColor"_s, serializationForHTML(m_backgroundColor));
258 if (m_highlightColor.isValid())
259 object.setString("highlightColor"_s, serializationForHTML(m_highlightColor));
260 if (m_baseFontSizeRelativeToVideoHeight)
261 object.setDouble("relativeFontSize"_s, m_baseFontSizeRelativeToVideoHeight);
262 if (m_fontSizeMultiplier)
263 object.setDouble("fontSizeMultiplier"_s, m_fontSizeMultiplier);
264 if (!m_fontName.isEmpty())
265 object.setString("font"_s, m_fontName);
266}
