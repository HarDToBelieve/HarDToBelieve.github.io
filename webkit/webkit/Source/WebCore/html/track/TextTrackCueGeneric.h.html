33namespace WebCore {
34
35// A "generic" cue is a non-WebVTT cue, so it is not positioned/sized with the WebVTT logic.
36class TextTrackCueGeneric final : public VTTCue {
37 WTF_MAKE_ISO_ALLOCATED_EXPORT(TextTrackCueGeneric, WEBCORE_EXPORT);
38public:
39 WEBCORE_EXPORT static Ref<TextTrackCueGeneric> create(ScriptExecutionContext&, const MediaTime& start, const MediaTime& end, const String& content);
40
41 ExceptionOr<void> setLine(const LineAndPositionSetting&) final;
42 ExceptionOr<void> setPosition(const LineAndPositionSetting&) final;
43
44 bool useDefaultPosition() const { return m_useDefaultPosition; }
45
46 double baseFontSizeRelativeToVideoHeight() const { return m_baseFontSizeRelativeToVideoHeight; }
47 void setBaseFontSizeRelativeToVideoHeight(double size) { m_baseFontSizeRelativeToVideoHeight = size; }
48
49 double fontSizeMultiplier() const { return m_fontSizeMultiplier; }
50 void setFontSizeMultiplier(double size) { m_fontSizeMultiplier = size; }
51
52 const String& fontName() const { return m_fontName; }
53 void setFontName(const String& name) { m_fontName = name; }
54
55 const Color& foregroundColor() const { return m_foregroundColor; }
56 void setForegroundColor(const Color& color) { m_foregroundColor = color; }
57
58 const Color& backgroundColor() const { return m_backgroundColor; }
59 void setBackgroundColor(const Color& color) { m_backgroundColor = color; }
60
61 const Color& highlightColor() const { return m_highlightColor; }
62 void setHighlightColor(const Color& color) { m_highlightColor = color; }
63
64 void setFontSize(int, const IntSize&, bool important) final;
65
66private:
67 TextTrackCueGeneric(Document&, const MediaTime& start, const MediaTime& end, const String&);
68
69 bool isOrderedBefore(const TextTrackCue*) const final;
70 bool isPositionedAbove(const TextTrackCue*) const final;
71
72 Ref<VTTCueBox> createDisplayTree() final;
73
74 bool cueContentsMatch(const TextTrackCue&) const final;
75
76 CueType cueType() const final { return ConvertedToWebVTT; }
77
78 void toJSON(JSON::Object&) const final;
79
80 Color m_foregroundColor;
81 Color m_backgroundColor;
82 Color m_highlightColor;
83 double m_baseFontSizeRelativeToVideoHeight { 0 };
84 double m_fontSizeMultiplier { 0 };
85 String m_fontName;
86 bool m_useDefaultPosition { true };
87};
88
89} // namespace WebCore

99SPECIALIZE_TYPE_TRAITS_END()
