36class TextTrackList final : public TrackListBase {

39 static Ref<TextTrackList> create(WeakPtr<HTMLMediaElement> element, ScriptExecutionContext* context)
40 {
41 auto list = adoptRef(*new TextTrackList(element, context));
42 list->suspendIfNeeded();
43 return list;
44 }
45 virtual ~TextTrackList();
46
47 void clearElement() override;
48
49 unsigned length() const override;
50 int getTrackIndex(TextTrack&);

52 bool contains(TrackBase&) const override;
53
54 TextTrack* item(unsigned index) const;
55 TextTrack* getTrackById(const AtomString&);
56 TextTrack* lastItem() const { return item(length() - 1); }
57
58 void append(Ref<TextTrack>&&);
59 void remove(TrackBase&, bool scheduleEvent = true) override;
60
61 // EventTarget
62 EventTargetInterface eventTargetInterface() const override;
63
64private:
65 TextTrackList(WeakPtr<HTMLMediaElement>, ScriptExecutionContext*);
66 const char* activeDOMObjectName() const final;
67
68 void invalidateTrackIndexesAfterTrack(TextTrack&);
69
70 Vector<RefPtr<TrackBase>> m_addTrackTracks;
71 Vector<RefPtr<TrackBase>> m_elementTracks;
72};
