83// See: https://tools.ietf.org/html/bcp47#section-2.1
84static bool isValidBCP47LanguageTag(const String& languageTag)
85{
86 auto const length = languageTag.length();
87
88 // Max length picked as double the longest example tag in spec which is 49 characters:
89 // https://tools.ietf.org/html/bcp47#section-4.4.2
90 if (length < 2 || length > 100)
91 return false;
92
93 UChar firstChar = languageTag[0];
94
95 if (!isASCIIAlpha(firstChar))
96 return false;
97
98 UChar secondChar = languageTag[1];
99
100 if (length == 2)
101 return isASCIIAlpha(secondChar);
102
103 bool grandFatheredIrregularOrPrivateUse = (firstChar == 'i' || firstChar == 'x') && secondChar == '-';
104 unsigned nextCharIndexToCheck;
105
106 if (!grandFatheredIrregularOrPrivateUse) {
107 if (!isASCIIAlpha(secondChar))
108 return false;
109
110 if (length == 3)
111 return isASCIIAlpha(languageTag[2]);
112
113 if (isASCIIAlpha(languageTag[2])) {
114 if (languageTag[3] == '-')
115 nextCharIndexToCheck = 4;
116 else
117 return false;
118 } else if (languageTag[2] == '-')
119 nextCharIndexToCheck = 3;
120 else
121 return false;
122 } else
123 nextCharIndexToCheck = 2;
124
125 for (; nextCharIndexToCheck < length; ++nextCharIndexToCheck) {
126 UChar c = languageTag[nextCharIndexToCheck];
127 if (isASCIIAlphanumeric(c) || c == '-')
128 continue;
129 return false;
130 }
131 return true;
132}

134void TrackBase::setLanguage(const AtomString& language)
135{
136 m_language = language;
137 if (language.isEmpty() || isValidBCP47LanguageTag(language)) {
138 m_validBCP47Language = language;
139 return;
140 }
141
142 m_validBCP47Language = emptyAtom();
143
144 auto element = this->element();
145 if (!element)
146 return;
147
148 String message;
149 if (language.contains((UChar)'\0'))
150 message = "The language contains a null character and is not a valid BCP 47 language tag."_s;
151 else
152 message = makeString("The language '", language, "' is not a valid BCP 47 language tag.");
153
154 element->document().addConsoleMessage(MessageSource::Rendering, MessageLevel::Warning, message);
155}
