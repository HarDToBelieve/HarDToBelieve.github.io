37static inline std::optional<TrackEvent::TrackEventTrack> convertToTrackEventTrack(Ref<TrackBase>&& track)
38{
39 switch (track->type()) {
40 case TrackBase::BaseTrack:
41 return std::nullopt;
42 case TrackBase::TextTrack:
43 return TrackEvent::TrackEventTrack { RefPtr<TextTrack>(&downcast<TextTrack>(track.get())) };
44 case TrackBase::AudioTrack:
45 return TrackEvent::TrackEventTrack { RefPtr<AudioTrack>(&downcast<AudioTrack>(track.get())) };
46 case TrackBase::VideoTrack:
47 return TrackEvent::TrackEventTrack { RefPtr<VideoTrack>(&downcast<VideoTrack>(track.get())) };
48 }
49
50 ASSERT_NOT_REACHED();
51 return std::nullopt;
52}
53
54TrackEvent::TrackEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Ref<TrackBase>&& track)
55 : Event(type, canBubble, cancelable)
56 , m_track(convertToTrackEventTrack(WTFMove(track)))
57{
58}
59
60TrackEvent::TrackEvent(const AtomString& type, Init&& initializer, IsTrusted isTrusted)
61 : Event(type, initializer, isTrusted)
62 , m_track(WTFMove(initializer.track))
63{
64}

66TrackEvent::~TrackEvent() = default;
