33namespace WebCore {
34
35class VTTRegionList : public RefCounted<VTTRegionList> {
36public:
37 static Ref<VTTRegionList> create();
38
39 unsigned length() const;
40 VTTRegion* item(unsigned index) const;
41 VTTRegion* getRegionById(const String&) const;
42
43 void add(Ref<VTTRegion>&&);
44 void remove(VTTRegion&);
45
46private:
47 VTTRegionList() = default;
48
49 void clear();
50
51 Vector<Ref<VTTRegion>> m_vector;
52};
53
54inline Ref<VTTRegionList> VTTRegionList::create()
55{
56 return adoptRef(*new VTTRegionList);
57}
58
59inline unsigned VTTRegionList::length() const
60{
61 return m_vector.size();
62}
63
64inline void VTTRegionList::clear()
65{
66 m_vector.clear();
67}
