37VTTScanner::VTTScanner(const String& line)
38 : m_is8Bit(line.is8Bit())
39{
40 if (m_is8Bit) {
41 m_data.characters8 = line.characters8();
42 m_end.characters8 = m_data.characters8 + line.length();
43 } else {
44 m_data.characters16 = line.characters16();
45 m_end.characters16 = m_data.characters16 + line.length();
46 }
47}

49bool VTTScanner::scan(char c)
50{
51 if (!match(c))
52 return false;
53 advance();
54 return true;
55}
56
57bool VTTScanner::scan(const LChar* characters, size_t charactersCount)
58{
59 unsigned matchLength = m_is8Bit ? m_end.characters8 - m_data.characters8 : m_end.characters16 - m_data.characters16;
60 if (matchLength < charactersCount)
61 return false;
62 bool matched;
63 if (m_is8Bit)
64 matched = WTF::equal(m_data.characters8, characters, charactersCount);
65 else
66 matched = WTF::equal(m_data.characters16, characters, charactersCount);
67 if (matched)
68 advance(charactersCount);
69 return matched;
70}

72bool VTTScanner::scanRun(const Run& run, const String& toMatch)
73{
74 ASSERT(run.start() == position());
75 ASSERT(run.start() <= end());
76 ASSERT(run.end() >= run.start());
77 ASSERT(run.end() <= end());
78 size_t matchLength = run.length();
79 if (toMatch.length() > matchLength)
80 return false;
81 bool matched;
82 if (m_is8Bit)
83 matched = WTF::equal(toMatch.impl(), m_data.characters8, matchLength);
84 else
85 matched = WTF::equal(toMatch.impl(), m_data.characters16, matchLength);
86 if (matched)
87 seekTo(run.end());
88 return matched;
89}

91void VTTScanner::skipRun(const Run& run)
92{
93 ASSERT(run.start() <= end());
94 ASSERT(run.end() >= run.start());
95 ASSERT(run.end() <= end());
96 seekTo(run.end());
97}

99String VTTScanner::extractString(const Run& run)
100{
101 ASSERT(run.start() == position());
102 ASSERT(run.start() <= end());
103 ASSERT(run.end() >= run.start());
104 ASSERT(run.end() <= end());
105 String s;
106 if (m_is8Bit)
107 s = String(m_data.characters8, run.length());
108 else
109 s = String(m_data.characters16, run.length());
110 seekTo(run.end());
111 return s;
112}

114String VTTScanner::restOfInputAsString()
115{
116 Run rest(position(), end(), m_is8Bit);
117 return extractString(rest);

120unsigned VTTScanner::scanDigits(int& number)
121{
122 Run runOfDigits = collectWhile<isASCIIDigit>();
123 if (runOfDigits.isEmpty()) {
124 number = 0;
125 return 0;
126 }
127
128 StringView string;
129 unsigned numDigits = runOfDigits.length();
130 if (m_is8Bit)
131 string = { m_data.characters8, numDigits };
132 else
133 string = { m_data.characters16, numDigits };
134
135 // Since these are ASCII digits, the only failure mode is overflow, so use the maximum int value.
136 number = parseInteger<int>(string).value_or(std::numeric_limits<int>::max());
137
138 // Consume the digits.
139 seekTo(runOfDigits.end());
140 return numDigits;
141}

143bool VTTScanner::scanFloat(float& number, bool* isNegative)
144{
145 bool negative = scan('-');
146 Run integerRun = collectWhile<isASCIIDigit>();
147
148 seekTo(integerRun.end());
149 Run decimalRun(position(), position(), m_is8Bit);
150 if (scan('.')) {
151 decimalRun = collectWhile<isASCIIDigit>();
152 seekTo(decimalRun.end());
153 }
154
155 // At least one digit required.
156 if (integerRun.isEmpty() && decimalRun.isEmpty()) {
157 // Restore to starting position.
158 seekTo(integerRun.start());
159 return false;
160 }
161
162 size_t lengthOfFloat = Run(integerRun.start(), position(), m_is8Bit).length();
163 bool validNumber;
164 if (m_is8Bit)
165 number = charactersToFloat(integerRun.start(), lengthOfFloat, &validNumber);
166 else
167 number = charactersToFloat(reinterpret_cast<const UChar*>(integerRun.start()), lengthOfFloat, &validNumber);
168
169 if (!validNumber)
170 number = std::numeric_limits<float>::max();
171 else if (negative)
172 number = -number;
173
174 if (isNegative)
175 *isNegative = negative;
176
177 return true;
178}
