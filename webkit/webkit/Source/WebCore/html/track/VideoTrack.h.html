45class VideoTrack final : public MediaTrackBase, private VideoTrackPrivateClient {
46public:
47 static Ref<VideoTrack> create(VideoTrackClient& client, VideoTrackPrivate& trackPrivate)
48 {
49 return adoptRef(*new VideoTrack(client, trackPrivate));
50 }
51 virtual ~VideoTrack();
52
53 static const AtomString& alternativeKeyword();
54 static const AtomString& captionsKeyword();
55 static const AtomString& mainKeyword();
56 static const AtomString& signKeyword();
57 static const AtomString& subtitlesKeyword();
58 static const AtomString& commentaryKeyword();
59
60 bool selected() const { return m_selected; }
61 virtual void setSelected(const bool);
62
63 void clearClient() final { m_client = nullptr; }
64 VideoTrackClient* client() const { return m_client; }
65
66 size_t inbandTrackIndex();
67
68#if ENABLE(MEDIA_SOURCE)
69 void setKind(const AtomString&) final;
70 void setLanguage(const AtomString&) final;
71#endif
72
73 const MediaDescription& description() const;
74
75 void setPrivate(VideoTrackPrivate&);
76 void setMediaElement(WeakPtr<HTMLMediaElement>) override;
77#if !RELEASE_LOG_DISABLED
78 void setLogger(const Logger&, const void*) final;
79#endif
80
81private:
82 VideoTrack(VideoTrackClient&, VideoTrackPrivate&);
83
84 bool isValidKind(const AtomString&) const final;
85
86 // VideoTrackPrivateClient
87 void selectedChanged(bool) final;
88
89 // TrackPrivateBaseClient
90 void idChanged(const AtomString&) final;
91 void labelChanged(const AtomString&) final;
92 void languageChanged(const AtomString&) final;
93 void willRemove() final;
94
95 bool enabled() const final { return selected(); }
96
97 void updateKindFromPrivate();
98
99#if !RELEASE_LOG_DISABLED
100 const char* logClassName() const final { return "VideoTrack"; }
101#endif
102
103 VideoTrackClient* m_client { nullptr };
104 Ref<VideoTrackPrivate> m_private;
105 bool m_selected { false };
106};
