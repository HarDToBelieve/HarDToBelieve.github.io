44void VideoTrackList::append(Ref<VideoTrack>&& track)
45{
46 // Insert tracks in the media file order.
47 size_t index = track->inbandTrackIndex();
48 size_t insertionIndex;
49 for (insertionIndex = 0; insertionIndex < m_inbandTracks.size(); ++insertionIndex) {
50 auto& otherTrack = downcast<VideoTrack>(*m_inbandTracks[insertionIndex]);
51 if (otherTrack.inbandTrackIndex() > index)
52 break;
53 }
54 m_inbandTracks.insert(insertionIndex, track.ptr());
55
56 ASSERT(!track->mediaElement() || track->mediaElement() == mediaElement());
57 track->setMediaElement(mediaElement());
58
59 scheduleAddTrackEvent(WTFMove(track));
60}
61
62VideoTrack* VideoTrackList::item(unsigned index) const
63{
64 if (index < m_inbandTracks.size())
65 return downcast<VideoTrack>(m_inbandTracks[index].get());
66 return nullptr;
67}
68
69VideoTrack* VideoTrackList::getTrackById(const AtomString& id) const
70{
71 for (auto& inbandTracks : m_inbandTracks) {
72 auto& track = downcast<VideoTrack>(*inbandTracks);
73 if (track.id() == id)
74 return &track;
75 }
76 return nullptr;
77}

79int VideoTrackList::selectedIndex() const
80{
81 // 4.8.10.10.1 AudioTrackList and VideoTrackList objects
82 // The VideoTrackList.selectedIndex attribute must return the index of the
83 // currently selected track, if any. If the VideoTrackList object does not
84 // currently represent any tracks, or if none of the tracks are selected,
85 // it must instead return âˆ’1.
86 for (unsigned i = 0; i < length(); ++i) {
87 if (downcast<VideoTrack>(*m_inbandTracks[i]).selected())
88 return i;
89 }
90 return -1;
91}
