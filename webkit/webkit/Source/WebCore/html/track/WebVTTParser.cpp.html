59bool WebVTTParser::parseFloatPercentageValue(VTTScanner& valueScanner, float& percentage)
60{
61 float number;
62 if (!valueScanner.scanFloat(number))
63 return false;
64 // '%' must be present and at the end of the setting value.
65 if (!valueScanner.scan('%'))
66 return false;
67
68 if (number < 0 || number > 100)
69 return false;
70
71 percentage = number;
72 return true;
73}

75bool WebVTTParser::parseFloatPercentageValuePair(VTTScanner& valueScanner, char delimiter, FloatPoint& valuePair)
76{
77 float firstCoord;
78 if (!parseFloatPercentageValue(valueScanner, firstCoord))
79 return false;
80
81 if (!valueScanner.scan(delimiter))
82 return false;
83
84 float secondCoord;
85 if (!parseFloatPercentageValue(valueScanner, secondCoord))
86 return false;
87
88 valuePair = FloatPoint(firstCoord, secondCoord);
89 return true;
90}

128void WebVTTParser::parseCueData(const ISOWebVTTCue& data)
129{
130 auto cue = WebVTTCueData::create();
131
132 MediaTime startTime = data.presentationTime();
133 cue->setStartTime(startTime);
134 cue->setEndTime(startTime + data.duration());
135
136 cue->setContent(data.cueText());
137 cue->setId(data.id());
138 cue->setSettings(data.settings());
139

141 if (WebVTTParser::collectTimeStamp(data.originalStartTime(), originalStartTime))
142 cue->setOriginalStartTime(originalStartTime);
143
144 m_cuelist.append(WTFMove(cue));
145 m_client.newCuesParsed();
146}

156void WebVTTParser::parse()
157{
158 // WebVTT parser algorithm. (5.1 WebVTT file parsing.)
159 // Steps 1 - 3 - Initial setup.
160 while (auto line = m_lineReader.nextLine()) {
161 switch (m_state) {
162 case Initial:
163 // Steps 4 - 9 - Check for a valid WebVTT signature.
164 if (!hasRequiredFileIdentifier(*line)) {
165 m_client.fileFailedToParse();
166 return;
167 }
168
169 m_state = Header;
170 break;
171
172 case Header:
173 // Steps 11 - 14 - Collect WebVTT block
174 m_state = collectWebVTTBlock(*line);
175 break;
176
177 case Region:
178 m_state = collectRegionSettings(*line);
179 break;
180
181 case Style:
182 m_state = collectStyleSheet(*line);
183 break;
184
185 case Id:
186 // Steps 17 - 20 - Allow any number of line terminators, then initialize new cue values.
187 if (line->isEmpty())
188 break;
189
190 // Step 21 - Cue creation (start a new cue).
191 resetCueValues();
192
193 // Steps 22 - 25 - Check if this line contains an optional identifier or timing data.
194 m_state = collectCueId(*line);
195 break;
196
197 case TimingsAndSettings:
198 // Steps 26 - 27 - Discard current cue if the line is empty.
199 if (line->isEmpty()) {
200 m_state = Id;
201 break;
202 }
203
204 // Steps 28 - 29 - Collect cue timings and settings.
205 m_state = collectTimingsAndSettings(*line);
206 break;
207
208 case CueText:
209 // Steps 31 - 41 - Collect the cue text, create a cue, and add it to the output.
210 m_state = collectCueText(*line);
211 break;
212
213 case BadCue:
214 // Steps 42 - 48 - Discard lines until an empty line or a potential timing line is seen.
215 m_state = ignoreBadCue(*line);
216 break;
217
218 case Finished:
219 ASSERT_NOT_REACHED();
220 break;
221 }
222 }
223}

225void WebVTTParser::fileFinished()
226{
227 ASSERT(m_state != Finished);
228 constexpr uint8_t endLines[] = { '\n', '\n' };
229 parseBytes(endLines, 2);
230 m_state = Finished;
231}

241bool WebVTTParser::hasRequiredFileIdentifier(const String& line)
242{
243 // A WebVTT file identifier consists of an optional BOM character,
244 // the string "WEBVTT" followed by an optional space or tab character,
245 // and any number of characters that are not line terminators ...
246 if (!line.startsWith(fileIdentifier))
247 return false;
248 if (line.length() > fileIdentifierLength && !isHTMLSpace(line[fileIdentifierLength]))
249 return false;
250 return true;
251}

263WebVTTParser::ParseState WebVTTParser::collectWebVTTBlock(const String& line)
264{
265 // collect a WebVTT block parsing. (WebVTT parser algorithm step 14)
266
267 if (checkAndCreateRegion(line))
268 return Region;
269
270 if (checkStyleSheet(line))
271 return Style;
272
273 // Handle cue block.
274 ParseState state = checkAndRecoverCue(line);
275 if (state != Header) {
276 if (!m_regionList.isEmpty())
277 m_client.newRegionsParsed();
278 if (!m_styleSheets.isEmpty())
279 m_client.newStyleSheetsParsed();
280 if (!m_previousLine.isEmpty() && !m_previousLine.contains("-->"))
281 m_currentId = m_previousLine;
282
283 return state;
284 }
285
286 // store previous line for cue id.
287 // length is more than 1 line clear m_previousLine and ignore line.
288 if (m_previousLine.isEmpty())
289 m_previousLine = line;
290 else
291 m_previousLine = emptyString();
292
293 return state;
294}

302 return state;
303 }
304 return Header;

317bool WebVTTParser::checkAndCreateRegion(const String& line)
318{
319 if (m_previousLine.contains("-->"))
320 return false;
321 // line starts with the substring "REGION" and remaining characters
322 // zero or more U+0020 SPACE characters or U+0009 CHARACTER TABULATION
323 // (tab) characters expected other than these charecters it is invalid.
324 if (line.startsWith("REGION") && line.substring(regionIdentifierLength).isAllSpecialCharacters<isASpace>()) {
325 m_currentRegion = VTTRegion::create(m_document);
326 return true;
327 }
328 return false;
329}

331bool WebVTTParser::checkAndStoreRegion(const String& line)
332{
333 if (!line.isEmpty() && !line.contains("-->"))
334 return false;
335
336 if (!m_currentRegion->id().isEmpty()) {
337 m_regionList.removeFirstMatching([this] (auto& region) {
338 return region->id() == m_currentRegion->id();
339 });
340 m_regionList.append(m_currentRegion.releaseNonNull());
341 }
342 m_currentRegion = nullptr;
343 return true;

346bool WebVTTParser::checkStyleSheet(const String& line)
347{
348 if (m_previousLine.contains("-->"))
349 return false;
350 // line starts with the substring "STYLE" and remaining characters
351 // zero or more U+0020 SPACE characters or U+0009 CHARACTER TABULATION
352 // (tab) characters expected other than these charecters it is invalid.
353 if (line.startsWith("STYLE") && line.substring(styleIdentifierLength).isAllSpecialCharacters<isASpace>())
354 return true;

359bool WebVTTParser::checkAndStoreStyleSheet(const String& line)
360{
361 if (!line.isEmpty() && !line.contains("-->"))
362 return false;
363
364 auto styleSheetText = WTFMove(m_currentSourceStyleSheet);
365
366 // WebVTTMode disallows non-data URLs.
367 auto contents = StyleSheetContents::create(CSSParserContext(WebVTTMode));
368 if (!contents->parseString(styleSheetText))
369 return true;
370
371 auto& namespaceRules = contents->namespaceRules();
372 if (namespaceRules.size())
373 return true;
374
375 auto& importRules = contents->importRules();
376 if (importRules.size())
377 return true;
378
379 auto& childRules = contents->childRules();
380 if (!childRules.size())
381 return true;
382
383 StringBuilder sanitizedStyleSheetBuilder;
384
385 for (const auto& rule : childRules) {
386 if (!rule->isStyleRule())
387 return true;
388 const auto& styleRule = downcast<StyleRule>(*rule);
389
390 const auto& selectorList = styleRule.selectorList();
391 if (selectorList.listSize() != 1)
392 return true;
393 auto selector = selectorList.selectorAt(0);
394 auto selectorText = selector->selectorText();
395
396 bool isCue = selectorText == "::cue" || selectorText.startsWith("::cue(");
397 if (!isCue)
398 return true;
399
400 if (styleRule.properties().isEmpty())
401 continue;
402
403 sanitizedStyleSheetBuilder.append(selectorText, " { ", styleRule.properties().asText(), " }\n");
404 }
405
406 // It would be more stylish to parse the stylesheet only once instead of serializing a sanitized version.
407 if (!sanitizedStyleSheetBuilder.isEmpty())
408 m_styleSheets.append(sanitizedStyleSheetBuilder.toString());
409
410 return true;
411}

421WebVTTParser::ParseState WebVTTParser::collectTimingsAndSettings(const String& line)
422{
423 if (line.isEmpty())
424 return BadCue;
425
426 VTTScanner input(line);
427
428 // Collect WebVTT cue timings and settings. (5.3 WebVTT cue timings and settings parsing.)
429 // Steps 1 - 3 - Let input be the string being parsed and position be a pointer into input
430 input.skipWhile<isHTMLSpace<UChar>>();
431
432 // Steps 4 - 5 - Collect a WebVTT timestamp. If that fails, then abort and return failure. Otherwise, let cue's text track cue start time be the collected time.
433 if (!collectTimeStamp(input, m_currentStartTime))
434 return BadCue;
435
436 input.skipWhile<isHTMLSpace<UChar>>();
437
438 // Steps 6 - 9 - If the next three characters are not "-->", abort and return failure.
439 if (!input.scan("-->"))
440 return BadCue;
441
442 input.skipWhile<isHTMLSpace<UChar>>();
443
444 // Steps 10 - 11 - Collect a WebVTT timestamp. If that fails, then abort and return failure. Otherwise, let cue's text track cue end time be the collected time.
445 if (!collectTimeStamp(input, m_currentEndTime))
446 return BadCue;
447
448 input.skipWhile<isHTMLSpace<UChar>>();
449
450 // Step 12 - Parse the WebVTT settings for the cue (conducted in TextTrackCue).
451 m_currentSettings = input.restOfInputAsString();
452 return CueText;
453}

455WebVTTParser::ParseState WebVTTParser::collectCueText(const String& line)
456{
457 // Step 34.
458 if (line.isEmpty()) {
459 createNewCue();
460 return Id;
461 }
462 // Step 35.
463 if (line.contains("-->")) {
464 // Step 39-40.
465 createNewCue();
466
467 // Step 41 - New iteration of the cue loop.
468 return recoverCue(line);
469 }
470 if (!m_currentContent.isEmpty())
471 m_currentContent.append('\n');
472 m_currentContent.append(line);
473
474 return CueText;
475}

477WebVTTParser::ParseState WebVTTParser::recoverCue(const String& line)
478{
479 // Step 17 and 21.
480 resetCueValues();
481
482 // Step 22.
483 return collectTimingsAndSettings(line);
484}

486WebVTTParser::ParseState WebVTTParser::ignoreBadCue(const String& line)

488 if (line.isEmpty())
489 return Id;

491 return recoverCue(line);
492 return BadCue;

495// A helper class for the construction of a "cue fragment" from the cue text.
496class WebVTTTreeBuilder {
497public:
498 WebVTTTreeBuilder(Document& document)
499 : m_document(document) { }
500
501 Ref<DocumentFragment> buildFromString(const String& cueText);
502
503private:
504 void constructTreeFromToken(Document&);
505
506 WebVTTToken m_token;
507 RefPtr<ContainerNode> m_currentNode;
508 Vector<AtomString> m_languageStack;
509 Document& m_document;
510};
511
512Ref<DocumentFragment> WebVTTTreeBuilder::buildFromString(const String& cueText)
513{
514 // Cue text processing based on
515 // 5.4 WebVTT cue text parsing rules, and
516 // 5.5 WebVTT cue text DOM construction rules.
517 auto fragment = DocumentFragment::create(m_document);
518
519 if (cueText.isEmpty()) {
520 fragment->parserAppendChild(Text::create(m_document, emptyString()));
521 return fragment;
522 }
523
524 m_currentNode = fragment.ptr();
525
526 WebVTTTokenizer tokenizer(cueText);
527 m_languageStack.clear();
528
529 while (tokenizer.nextToken(m_token))
530 constructTreeFromToken(m_document);
531
532 return fragment;
533}

536{
537 WebVTTTreeBuilder treeBuilder(document);
538 return treeBuilder.buildFromString(cueText);
539}

572bool WebVTTParser::collectTimeStamp(VTTScanner& input, MediaTime& timeStamp)
573{
574 // Collect a WebVTT timestamp (5.3 WebVTT cue timings and settings parsing.)
575 // Steps 1 - 4 - Initial checks, let most significant units be minutes.
576 enum Mode { minutes, hours };
577 Mode mode = minutes;
578
579 // Steps 5 - 7 - Collect a sequence of characters that are 0-9.
580 // If not 2 characters or value is greater than 59, interpret as hours.
581 int value1;
582 unsigned value1Digits = input.scanDigits(value1);
583 if (!value1Digits)
584 return false;
585 if (value1Digits != 2 || value1 > 59)
586 mode = hours;
587
588 // Steps 8 - 11 - Collect the next sequence of 0-9 after ':' (must be 2 chars).
589 int value2;
590 if (!input.scan(':') || input.scanDigits(value2) != 2)
591 return false;
592
593 // Step 12 - Detect whether this timestamp includes hours.
594 int value3;
595 if (mode == hours || input.match(':')) {
596 if (!input.scan(':') || input.scanDigits(value3) != 2)
597 return false;
598 } else {
599 value3 = value2;
600 value2 = value1;
601 value1 = 0;
602 }
603
604 // Steps 13 - 17 - Collect next sequence of 0-9 after '.' (must be 3 chars).
605 int value4;
606 if (!input.scan('.') || input.scanDigits(value4) != 3)
607 return false;
608 if (value2 > 59 || value3 > 59)
609 return false;
610
611 // Steps 18 - 19 - Calculate result.
612 timeStamp = MediaTime::createWithDouble((value1 * secondsPerHour) + (value2 * secondsPerMinute) + value3 + (value4 * secondsPerMillisecond));
613 return true;
614}

616static WebVTTNodeType tokenToNodeType(WebVTTToken& token)
617{
618 switch (token.name().length()) {
619 case 1:
620 if (token.name()[0] == 'c')
621 return WebVTTNodeTypeClass;
622 if (token.name()[0] == 'v')
623 return WebVTTNodeTypeVoice;
624 if (token.name()[0] == 'b')
625 return WebVTTNodeTypeBold;
626 if (token.name()[0] == 'i')
627 return WebVTTNodeTypeItalic;
628 if (token.name()[0] == 'u')
629 return WebVTTNodeTypeUnderline;
630 break;
631 case 2:
632 if (token.name()[0] == 'r' && token.name()[1] == 't')
633 return WebVTTNodeTypeRubyText;
634 break;
635 case 4:
636 if (token.name()[0] == 'r' && token.name()[1] == 'u' && token.name()[2] == 'b' && token.name()[3] == 'y')
637 return WebVTTNodeTypeRuby;
638 if (token.name()[0] == 'l' && token.name()[1] == 'a' && token.name()[2] == 'n' && token.name()[3] == 'g')
639 return WebVTTNodeTypeLanguage;
640 break;
641 }
642 return WebVTTNodeTypeNone;
643}

645void WebVTTTreeBuilder::constructTreeFromToken(Document& document)
646{
647 // http://dev.w3.org/html5/webvtt/#webvtt-cue-text-dom-construction-rules
648
649 switch (m_token.type()) {
650 case WebVTTTokenTypes::Character: {
651 m_currentNode->parserAppendChild(Text::create(document, m_token.characters()));
652 break;
653 }
654 case WebVTTTokenTypes::StartTag: {
655 WebVTTNodeType nodeType = tokenToNodeType(m_token);
656 if (nodeType == WebVTTNodeTypeNone)
657 break;
658
659 WebVTTNodeType currentType = is<WebVTTElement>(*m_currentNode) ? downcast<WebVTTElement>(*m_currentNode).webVTTNodeType() : WebVTTNodeTypeNone;
660 // <rt> is only allowed if the current node is <ruby>.
661 if (nodeType == WebVTTNodeTypeRubyText && currentType != WebVTTNodeTypeRuby)
662 break;
663
664 auto child = WebVTTElement::create(nodeType, document);
665 if (!m_token.classes().isEmpty())
666 child->setAttributeWithoutSynchronization(classAttr, m_token.classes());
667
668 if (nodeType == WebVTTNodeTypeVoice)
669 child->setAttributeWithoutSynchronization(WebVTTElement::voiceAttributeName(), m_token.annotation());
670 else if (nodeType == WebVTTNodeTypeLanguage) {
671 m_languageStack.append(m_token.annotation());
672 child->setAttributeWithoutSynchronization(WebVTTElement::langAttributeName(), m_languageStack.last());
673 }
674 if (!m_languageStack.isEmpty())
675 child->setLanguage(m_languageStack.last());
676 m_currentNode->parserAppendChild(child);
677 m_currentNode = WTFMove(child);
678 break;
679 }
680 case WebVTTTokenTypes::EndTag: {
681 WebVTTNodeType nodeType = tokenToNodeType(m_token);
682 if (nodeType == WebVTTNodeTypeNone)
683 break;
684
685 // The only non-VTTElement would be the DocumentFragment root. (Text
686 // nodes and PIs will never appear as m_currentNode.)
687 if (!is<WebVTTElement>(*m_currentNode))
688 break;
689
690 WebVTTNodeType currentType = downcast<WebVTTElement>(*m_currentNode).webVTTNodeType();
691 bool matchesCurrent = nodeType == currentType;
692 if (!matchesCurrent) {
693 // </ruby> auto-closes <rt>
694 if (currentType == WebVTTNodeTypeRubyText && nodeType == WebVTTNodeTypeRuby) {
695 if (m_currentNode->parentNode())
696 m_currentNode = m_currentNode->parentNode();
697 } else
698 break;
699 }
700 if (nodeType == WebVTTNodeTypeLanguage)
701 m_languageStack.removeLast();
702 if (m_currentNode->parentNode())
703 m_currentNode = m_currentNode->parentNode();
704 break;
705 }
706 case WebVTTTokenTypes::TimestampTag: {
707 String charactersString = m_token.characters();
708 MediaTime parsedTimeStamp;
709 if (WebVTTParser::collectTimeStamp(charactersString, parsedTimeStamp))
710 m_currentNode->parserAppendChild(ProcessingInstruction::create(document, "timestamp", charactersString));
711 break;
712 }
713 default:
714 break;
715 }
716}
