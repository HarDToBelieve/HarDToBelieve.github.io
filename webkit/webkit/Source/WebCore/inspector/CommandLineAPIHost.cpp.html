77void CommandLineAPIHost::inspect(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue object, JSC::JSValue hints)
78{
79 if (!m_instrumentingAgents)
80 return;
81
82 auto* inspectorAgent = m_instrumentingAgents->persistentInspectorAgent();
83 if (!inspectorAgent)
84 return;
85
86 auto objectValue = Inspector::toInspectorValue(&lexicalGlobalObject, object);
87 if (!objectValue)
88 return;
89
90 auto hintsValue = Inspector::toInspectorValue(&lexicalGlobalObject, hints);
91 if (!hintsValue)
92 return;
93
94 auto hintsObject = hintsValue->asObject();
95 if (!hintsObject)
96 return;
97
98 auto remoteObject = Protocol::BindingTraits<Protocol::Runtime::RemoteObject>::runtimeCast(objectValue.releaseNonNull());
99 inspectorAgent->inspect(WTFMove(remoteObject), hintsObject.releaseNonNull());
100}

102CommandLineAPIHost::EventListenersRecord CommandLineAPIHost::getEventListeners(JSGlobalObject& lexicalGlobalObject, EventTarget& target)
103{
104 auto* scriptExecutionContext = target.scriptExecutionContext();
105 if (!scriptExecutionContext)
106 return { };
107
108 EventListenersRecord result;
109
110 VM& vm = lexicalGlobalObject.vm();
111
112 for (auto& eventType : target.eventTypes()) {
113 Vector<CommandLineAPIHost::ListenerEntry> entries;
114
115 for (auto& eventListener : target.eventListeners(eventType)) {
116 if (!is<JSEventListener>(eventListener->callback()))
117 continue;
118
119 auto& jsListener = downcast<JSEventListener>(eventListener->callback());
120
121 // Hide listeners from other contexts.
122 if (&jsListener.isolatedWorld() != &currentWorld(lexicalGlobalObject))
123 continue;
124
125 auto* function = jsListener.ensureJSFunction(*scriptExecutionContext);
126 if (!function)
127 continue;
128
129 entries.append({ Strong<JSObject>(vm, function), eventListener->useCapture(), eventListener->isPassive(), eventListener->isOnce() });
130 }
131
132 if (!entries.isEmpty())
133 result.append({ eventType, WTFMove(entries) });
134 }
135
136 return result;
137}

166JSC::JSValue CommandLineAPIHost::inspectedObject(JSC::JSGlobalObject& lexicalGlobalObject)
167{
168 if (!m_inspectedObject)
169 return jsUndefined();
170
171 JSC::JSLockHolder lock(&lexicalGlobalObject);
172 auto scriptValue = m_inspectedObject->get(lexicalGlobalObject);
173 return scriptValue ? scriptValue : jsUndefined();
174}

190JSValue CommandLineAPIHost::wrapper(JSGlobalObject* exec, JSDOMGlobalObject* globalObject)
191{
192 JSValue value = m_wrappers.getWrapper(globalObject);
193 if (value)
194 return value;
195
196 JSObject* prototype = JSCommandLineAPIHost::createPrototype(exec->vm(), *globalObject);
197 Structure* structure = JSCommandLineAPIHost::createStructure(exec->vm(), globalObject, prototype);
198 JSCommandLineAPIHost* commandLineAPIHost = JSCommandLineAPIHost::create(structure, globalObject, makeRef(*this));
199 m_wrappers.addWrapper(globalObject, commandLineAPIHost);
200
201 return commandLineAPIHost;
202}
