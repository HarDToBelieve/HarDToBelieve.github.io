45#define ERROR_IF_NO_ACTIVE_AUDIT() \
46 if (!m_auditAgent.hasActiveAudit()) \
47 return Exception { NotAllowedError, "Cannot be called outside of a Web Inspector Audit"_s };

50 : m_auditAgent(auditAgent)

52}

54static AXCoreObject* accessiblityObjectForNode(Node& node)
55{
56 if (!AXObjectCache::accessibilityEnabled())
57 AXObjectCache::enableAccessibility();
58
59 if (AXObjectCache* axObjectCache = node.document().axObjectCache())
60 return axObjectCache->getOrCreate(&node);
61
62 return nullptr;
63}

65ExceptionOr<Vector<Ref<Node>>> InspectorAuditAccessibilityObject::getElementsByComputedRole(Document& document, const String& role, Node* container)
66{
67 ERROR_IF_NO_ACTIVE_AUDIT();
68
69 Vector<Ref<Node>> nodes;
70
71 for (Element& element : descendantsOfType<Element>(is<ContainerNode>(container) ? downcast<ContainerNode>(*container) : document)) {
72 if (AXCoreObject* axObject = accessiblityObjectForNode(element)) {
73 if (axObject->computedRoleString() == role)
74 nodes.append(element);
75 }
76 }
77
78 return nodes;
79}

81ExceptionOr<RefPtr<Node>> InspectorAuditAccessibilityObject::getActiveDescendant(Node& node)
82{
83 ERROR_IF_NO_ACTIVE_AUDIT();
84
85 if (AXCoreObject* axObject = accessiblityObjectForNode(node)) {
86 if (AXCoreObject* activeDescendant = axObject->activeDescendant())

88 }

93static void addChildren(AXCoreObject& parentObject, Vector<RefPtr<Node>>& childNodes)
94{
95 for (const auto& childObject : parentObject.children()) {
96 if (Node* childNode = childObject->node())
97 childNodes.append(childNode);
98 else
99 addChildren(*childObject, childNodes);
100 }
101}

103ExceptionOr<std::optional<Vector<RefPtr<Node>>>> InspectorAuditAccessibilityObject::getChildNodes(Node& node)
104{
105 ERROR_IF_NO_ACTIVE_AUDIT();
106
107 std::optional<Vector<RefPtr<Node>>> result;
108
109 if (AXCoreObject* axObject = accessiblityObjectForNode(node)) {
110 Vector<RefPtr<Node>> childNodes;
111 addChildren(*axObject, childNodes);
112 result = WTFMove(childNodes);
113 }
114
115 return result;
116}

118ExceptionOr<std::optional<InspectorAuditAccessibilityObject::ComputedProperties>> InspectorAuditAccessibilityObject::getComputedProperties(Node& node)
119{
120 ERROR_IF_NO_ACTIVE_AUDIT();
121
122 std::optional<InspectorAuditAccessibilityObject::ComputedProperties> result;
123
124 if (AXCoreObject* axObject = accessiblityObjectForNode(node)) {
125 ComputedProperties computedProperties;
126
127 AXCoreObject* current = axObject;
128 while (current && (!computedProperties.busy || !computedProperties.busy.value())) {
129 computedProperties.busy = current->isBusy();
130 current = current->parentObject();
131 }
132
133 if (axObject->supportsChecked()) {
134 AccessibilityButtonState checkValue = axObject->checkboxOrRadioValue();
135 if (checkValue == AccessibilityButtonState::On)
136 computedProperties.checked = "true"_s;
137 else if (checkValue == AccessibilityButtonState::Mixed)
138 computedProperties.checked = "mixed"_s;
139 else if (axObject->isChecked())
140 computedProperties.checked = "true"_s;
141 else
142 computedProperties.checked = "false"_s;
143 }
144
145 switch (axObject->currentState()) {
146 case AccessibilityCurrentState::False:
147 computedProperties.currentState = "false"_s;
148 break;
149 case AccessibilityCurrentState::True:
150 computedProperties.currentState = "true"_s;
151 break;
152 case AccessibilityCurrentState::Page:
153 computedProperties.currentState = "page"_s;
154 break;
155 case AccessibilityCurrentState::Step:
156 computedProperties.currentState = "step"_s;
157 break;
158 case AccessibilityCurrentState::Location:
159 computedProperties.currentState = "location"_s;
160 break;
161 case AccessibilityCurrentState::Date:
162 computedProperties.currentState = "date"_s;
163 break;
164 case AccessibilityCurrentState::Time:
165 computedProperties.currentState = "time"_s;
166 break;
167 }
168
169 computedProperties.disabled = !axObject->isEnabled();
170
171 if (axObject->supportsExpanded())
172 computedProperties.expanded = axObject->isExpanded();
173
174 if (is<Element>(node) && axObject->canSetFocusAttribute())
175 computedProperties.focused = axObject->isFocused();
176
177 computedProperties.headingLevel = axObject->headingLevel();
178 computedProperties.hidden = axObject->isAXHidden() || axObject->isDOMHidden();
179 computedProperties.hierarchicalLevel = axObject->hierarchicalLevel();
180 computedProperties.ignored = axObject->accessibilityIsIgnored();
181 computedProperties.ignoredByDefault = axObject->accessibilityIsIgnoredByDefault();
182
183 String invalidValue = axObject->invalidStatus();
184 if (invalidValue == "false")
185 computedProperties.invalidStatus = "false"_s;
186 else if (invalidValue == "grammar")
187 computedProperties.invalidStatus = "grammar"_s;
188 else if (invalidValue == "spelling")
189 computedProperties.invalidStatus = "spelling"_s;
190 else
191 computedProperties.invalidStatus = "true"_s;
192
193 computedProperties.isPopUpButton = axObject->isPopUpButton() || axObject->hasPopup();
194 computedProperties.label = axObject->computedLabel();
195
196 if (axObject->supportsLiveRegion()) {
197 computedProperties.liveRegionAtomic = axObject->liveRegionAtomic();
198
199 String ariaRelevantAttrValue = axObject->liveRegionRelevant();
200 if (!ariaRelevantAttrValue.isEmpty()) {
201 Vector<String> liveRegionRelevant;
202 String ariaRelevantAdditions = "additions";
203 String ariaRelevantRemovals = "removals";
204 String ariaRelevantText = "text";
205
206 const auto& values = SpaceSplitString(ariaRelevantAttrValue, true);
207 if (values.contains("all")) {
208 liveRegionRelevant.append(ariaRelevantAdditions);
209 liveRegionRelevant.append(ariaRelevantRemovals);
210 liveRegionRelevant.append(ariaRelevantText);
211 } else {
212 if (values.contains(ariaRelevantAdditions))
213 liveRegionRelevant.append(ariaRelevantAdditions);
214 if (values.contains(ariaRelevantRemovals))
215 liveRegionRelevant.append(ariaRelevantRemovals);
216 if (values.contains(ariaRelevantText))
217 liveRegionRelevant.append(ariaRelevantText);
218 }
219 computedProperties.liveRegionRelevant = liveRegionRelevant;
220 }
221
222 computedProperties.liveRegionStatus = axObject->liveRegionStatus();
223 }
224
225 computedProperties.pressed = axObject->pressedIsPresent() && axObject->isPressed();
226
227 if (axObject->isTextControl())
228 computedProperties.readonly = !axObject->canSetValueAttribute();
229
230 if (axObject->supportsRequiredAttribute())
231 computedProperties.required = axObject->isRequired();
232
233 computedProperties.role = axObject->computedRoleString();
234 computedProperties.selected = axObject->isSelected();
235
236 result = computedProperties;
237 }
238
239 return result;
240}

242ExceptionOr<std::optional<Vector<RefPtr<Node>>>> InspectorAuditAccessibilityObject::getControlledNodes(Node& node)
243{
244 ERROR_IF_NO_ACTIVE_AUDIT();
245
246 std::optional<Vector<RefPtr<Node>>> result;
247
248 if (AXCoreObject* axObject = accessiblityObjectForNode(node)) {
249 Vector<RefPtr<Node>> controlledNodes;
250
251 Vector<Element*> controlledElements;
252 axObject->elementsFromAttribute(controlledElements, HTMLNames::aria_controlsAttr);
253 for (Element* controlledElement : controlledElements) {
254 if (controlledElement)
255 controlledNodes.append(controlledElement);
256 }
257
258 result = WTFMove(controlledNodes);
259 }
260
261 return result;
262}

314
315 result = WTFMove(ownedNodes);

334ExceptionOr<std::optional<Vector<RefPtr<Node>>>> InspectorAuditAccessibilityObject::getSelectedChildNodes(Node& node)
335{
336 ERROR_IF_NO_ACTIVE_AUDIT();
337
338 std::optional<Vector<RefPtr<Node>>> result;
339
340 if (AXCoreObject* axObject = accessiblityObjectForNode(node)) {
341 Vector<RefPtr<Node>> selectedChildNodes;
342
343 AXCoreObject::AccessibilityChildrenVector selectedChildren;
344 axObject->selectedChildren(selectedChildren);
345 for (auto& selectedChildObject : selectedChildren) {
346 if (Node* selectedChildNode = selectedChildObject->node())
347 selectedChildNodes.append(selectedChildNode);
348 }
349
350 result = WTFMove(selectedChildNodes);
351 }
352
353 return result;
