44#define ERROR_IF_NO_ACTIVE_AUDIT() \
45 if (!m_auditAgent.hasActiveAudit()) \
46 return Exception { NotAllowedError, "Cannot be called outside of a Web Inspector Audit"_s };

48InspectorAuditResourcesObject::InspectorAuditResourcesObject(InspectorAuditAgent& auditAgent)
49 : m_auditAgent(auditAgent)
50{
51}
52
53InspectorAuditResourcesObject::~InspectorAuditResourcesObject()
54{
55 for (auto* cachedResource : m_resources.values())
56 cachedResource->removeClient(clientForResource(*cachedResource));
57}
58
59ExceptionOr<Vector<InspectorAuditResourcesObject::Resource>> InspectorAuditResourcesObject::getResources(Document& document)
60{
61 ERROR_IF_NO_ACTIVE_AUDIT();
62
63 Vector<Resource> resources;
64
65 auto* frame = document.frame();
66 if (!frame)
67 return Exception { NotAllowedError, "Cannot be called with a detached document"_s };
68
69 for (auto* cachedResource : InspectorPageAgent::cachedResourcesForFrame(frame)) {
70 Resource resource;
71 resource.url = cachedResource->url().string();
72 resource.mimeType = cachedResource->mimeType();
73
74 bool exists = false;
75 for (const auto& entry : m_resources) {
76 if (entry.value == cachedResource) {
77 resource.id = entry.key;
78 exists = true;
79 break;
80 }
81 }
82 if (!exists) {
83 cachedResource->addClient(clientForResource(*cachedResource));
84
85 resource.id = String::number(m_resources.size() + 1);
86 m_resources.add(resource.id, cachedResource);
87 }
88
89 resources.append(WTFMove(resource));
90 }
91
92 return resources;
93}

95ExceptionOr<InspectorAuditResourcesObject::ResourceContent> InspectorAuditResourcesObject::getResourceContent(Document& document, const String& id)
96{
97 ERROR_IF_NO_ACTIVE_AUDIT();
98
99 auto* frame = document.frame();
100 if (!frame)
101 return Exception { NotAllowedError, "Cannot be called with a detached document"_s };
102
103 auto* cachedResource = m_resources.get(id);
104 if (!cachedResource)
105 return Exception { NotFoundError, makeString("Unknown identifier "_s, id) };
106
107 Protocol::ErrorString errorString;
108 ResourceContent resourceContent;
109 InspectorPageAgent::resourceContent(errorString, frame, cachedResource->url(), &resourceContent.data, &resourceContent.base64Encoded);
110 if (!errorString.isEmpty())
111 return Exception { NotFoundError, errorString };
112
113 return resourceContent;
114}

116CachedResourceClient& InspectorAuditResourcesObject::clientForResource(const CachedResource& cachedResource)
117{
118 if (is<CachedCSSStyleSheet>(cachedResource))
119 return m_cachedStyleSheetClient;
120
121 if (is<CachedFont>(cachedResource))
122 return m_cachedFontClient;
123
124 if (is<CachedImage>(cachedResource))
125 return m_cachedImageClient;
126
127 if (is<CachedRawResource>(cachedResource))
128 return m_cachedRawResourceClient;
129
130 if (is<CachedSVGDocument>(cachedResource))
131 return m_cachedSVGDocumentClient;
132
133 return m_cachedResourceClient;
134}
