26#pragma once
27
28#include <wtf/CompletionHandler.h>
29#include <wtf/Expected.h>
30#include <wtf/JSONValues.h>
31#include <wtf/RefCounted.h>
32#include <wtf/WeakPtr.h>
33#include <wtf/text/WTFString.h>
34
35namespace JSC {
36class JSValue;
37}
38
39namespace WebCore {
40
41class DOMPromise;
42class JSDOMGlobalObject;
43class Page;
44struct ExceptionDetails;
45
46class InspectorFrontendAPIDispatcher final
47 : public RefCounted<InspectorFrontendAPIDispatcher>
48 , public CanMakeWeakPtr<InspectorFrontendAPIDispatcher> {
49public:
50 enum class EvaluationError { ExecutionSuspended, ContextDestroyed, InternalError };
51 using ValueOrException = Expected<JSC::JSValue, ExceptionDetails>;
52 using EvaluationResult = Expected<ValueOrException, EvaluationError>;
53 using EvaluationResultHandler = CompletionHandler<void(EvaluationResult)>;
54
55 enum class UnsuspendSoon { Yes, No };
56
57 WEBCORE_EXPORT ~InspectorFrontendAPIDispatcher();
58
59 static Ref<InspectorFrontendAPIDispatcher> create(Page& page)
60 {
61 return adoptRef(*new InspectorFrontendAPIDispatcher(page));
62 }
63
64 WEBCORE_EXPORT void reset();
65 WEBCORE_EXPORT void frontendLoaded();
66
67 // If it's not currently safe to evaluate JavaScript on the frontend page, the
68 // dispatcher will become suspended and dispatch any queued evaluations when unsuspended.
69 WEBCORE_EXPORT void suspend(UnsuspendSoon = UnsuspendSoon::No);
70 WEBCORE_EXPORT void unsuspend();
71 bool isSuspended() const { return m_suspended; }
72
73 EvaluationResult dispatchCommandWithResultSync(const String& command, Vector<Ref<JSON::Value>>&& arguments = { });
74 WEBCORE_EXPORT void dispatchCommandWithResultAsync(const String& command, Vector<Ref<JSON::Value>>&& arguments = { }, EvaluationResultHandler&& handler = { });
75
76 // Used to forward messages from the backend connection to the frontend.
77 WEBCORE_EXPORT void dispatchMessageAsync(const String& message);
78
79 WEBCORE_EXPORT void evaluateExpressionForTesting(const String&);
80
81 // Convenience method to obtain a JSDOMGlobalObject for the frontend page.
82 // This is used to convert between C++ values and frontend JSC::JSValue objects.
83 WEBCORE_EXPORT JSDOMGlobalObject* frontendGlobalObject();
84private:
85 WEBCORE_EXPORT InspectorFrontendAPIDispatcher(Page&);
86
87 void evaluateOrQueueExpression(const String&, EvaluationResultHandler&& handler = { });
88 void evaluateQueuedExpressions();
89 void invalidateQueuedExpressions();
90 void invalidatePendingResponses();
91 ValueOrException evaluateExpression(const String&);
92
93 WeakPtr<Page> m_frontendPage;
94 Vector<std::pair<String, EvaluationResultHandler>> m_queuedEvaluations;
95 HashMap<Ref<DOMPromise>, EvaluationResultHandler> m_pendingResponses;
96 bool m_frontendLoaded { false };
97 bool m_suspended { false };
98};
99
100} // namespace WebCore
