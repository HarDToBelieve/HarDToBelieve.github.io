33#include "ExceptionOr.h"

36namespace WebCore {
37
38class InspectorHistory final {
39 WTF_MAKE_NONCOPYABLE(InspectorHistory);
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 class Action {
43 WTF_MAKE_FAST_ALLOCATED;
44 public:
45 virtual ~Action() = default;
46
47 virtual String mergeId() { return emptyString(); }
48 virtual void merge(std::unique_ptr<Action>) { };
49
50 virtual ExceptionOr<void> perform() = 0;
51
52 virtual ExceptionOr<void> undo() = 0;
53 virtual ExceptionOr<void> redo() = 0;
54
55 virtual bool isUndoableStateMark() { return false; }
56
57 private:
58 String m_name;
59 };
60
61 InspectorHistory() = default;
62
63 ExceptionOr<void> perform(std::unique_ptr<Action>);
64 void markUndoableState();
65
66 ExceptionOr<void> undo();
67 ExceptionOr<void> redo();
68 void reset();
69
70private:
71 Vector<std::unique_ptr<Action>> m_history;
72 size_t m_afterLastActionIndex { 0 };
73};
