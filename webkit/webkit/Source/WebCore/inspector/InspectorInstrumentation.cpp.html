604void InspectorInstrumentation::didReceiveResourceResponseImpl(InstrumentingAgents& instrumentingAgents, unsigned long identifier, DocumentLoader* loader, const ResourceResponse& response, ResourceLoader* resourceLoader)
605{
606 if (LIKELY(!instrumentingAgents.inspectorEnvironment().developerExtrasEnabled()))
607 return;
608

698void InspectorInstrumentation::didCommitLoadImpl(InstrumentingAgents& instrumentingAgents, Frame& frame, DocumentLoader* loader)
699{
700 if (LIKELY(!instrumentingAgents.inspectorEnvironment().developerExtrasEnabled()))
701 return;
702
703 if (!frame.page())
704 return;
705
706 if (!loader)
707 return;
708
709 ASSERT(loader->frame() == &frame);
710
711 if (frame.isMainFrame()) {
712 if (auto* consoleAgent = instrumentingAgents.webConsoleAgent())
713 consoleAgent->reset();
714
715 if (auto* networkAgent = instrumentingAgents.enabledNetworkAgent())
716 networkAgent->mainFrameNavigated(*loader);
717
718 if (auto* cssAgent = instrumentingAgents.enabledCSSAgent())
719 cssAgent->reset();
720
721 if (auto* databaseAgent = instrumentingAgents.enabledDatabaseAgent())
722 databaseAgent->didCommitLoad();
723
724 if (auto* domAgent = instrumentingAgents.persistentDOMAgent())
725 domAgent->setDocument(frame.document());
726
727 if (auto* layerTreeAgent = instrumentingAgents.enabledLayerTreeAgent())
728 layerTreeAgent->reset();
729
730 if (auto* pageDebuggerAgent = instrumentingAgents.enabledPageDebuggerAgent())
731 pageDebuggerAgent->mainFrameNavigated();
732
733 if (auto* domDebuggerAgent = instrumentingAgents.enabledDOMDebuggerAgent())
734 domDebuggerAgent->mainFrameNavigated();
735
736 if (auto* enabledPageHeapAgent = instrumentingAgents.enabledPageHeapAgent())
737 enabledPageHeapAgent->mainFrameNavigated();
738 }
739
740 if (auto* pageAgent = instrumentingAgents.enabledPageAgent())
741 pageAgent->frameNavigated(frame);
742
743 if (auto* pageRuntimeAgent = instrumentingAgents.enabledPageRuntimeAgent())
744 pageRuntimeAgent->frameNavigated(frame);
745
746 if (auto* canvasAgent = instrumentingAgents.enabledCanvasAgent())
747 canvasAgent->frameNavigated(frame);
748
749 if (auto* animationAgent = instrumentingAgents.enabledAnimationAgent())
750 animationAgent->frameNavigated(frame);
751
752 if (auto* domAgent = instrumentingAgents.persistentDOMAgent())
753 domAgent->didCommitLoad(frame.document());
754
755 if (frame.isMainFrame()) {
756 if (auto* timelineAgent = instrumentingAgents.trackingTimelineAgent())
757 timelineAgent->mainFrameNavigated();
758 }
759}
