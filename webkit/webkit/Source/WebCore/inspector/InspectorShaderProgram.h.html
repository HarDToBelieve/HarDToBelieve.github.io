35namespace WebCore {
36
37class InspectorCanvas;
38
39#if ENABLE(WEBGL)
40class WebGLProgram;
41class WebGLRenderingContextBase;
42#endif
43
44#if ENABLE(WEBGPU)
45class WebGPUPipeline;
46#endif
47
48class InspectorShaderProgram final : public RefCounted<InspectorShaderProgram> {
49public:
50#if ENABLE(WEBGL)
51 static Ref<InspectorShaderProgram> create(WebGLProgram&, InspectorCanvas&);
52#endif
53#if ENABLE(WEBGPU)
54 static Ref<InspectorShaderProgram> create(WebGPUPipeline&, InspectorCanvas&);
55#endif
56
57 const String& identifier() const { return m_identifier; }
58 InspectorCanvas& canvas() const { return m_canvas; }
59
60#if ENABLE(WEBGL)
61 WebGLProgram* program() const;
62#endif
63#if ENABLE(WEBGPU)
64 WebGPUPipeline* pipeline() const;
65#endif
66
67 String requestShaderSource(Inspector::Protocol::Canvas::ShaderType);
68 bool updateShader(Inspector::Protocol::Canvas::ShaderType, const String& source);
69
70 bool disabled() const { return m_disabled; }
71 void setDisabled(bool disabled) { m_disabled = disabled; }
72
73 bool highlighted() const { return m_highlighted; }
74 void setHighlighted(bool value) { m_highlighted = value; }
75
76 Ref<Inspector::Protocol::Canvas::ShaderProgram> buildObjectForShaderProgram();
77
78private:
79#if ENABLE(WEBGL)
80 InspectorShaderProgram(WebGLProgram&, InspectorCanvas&);
81#endif
82#if ENABLE(WEBGPU)
83 InspectorShaderProgram(WebGPUPipeline&, InspectorCanvas&);
84#endif
85
86 String m_identifier;
87 InspectorCanvas& m_canvas;
88
89 Variant<
90#if ENABLE(WEBGL)
91 std::reference_wrapper<WebGLProgram>,
92#endif
93#if ENABLE(WEBGPU)
94 std::reference_wrapper<WebGPUPipeline>,
95#endif
96 WTF::Monostate
97 > m_program;
98
99 bool m_disabled { false };
100 bool m_highlighted { false };
101};
102
103} // namespace WebCore
