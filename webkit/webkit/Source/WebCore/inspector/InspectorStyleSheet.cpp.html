102static void flattenSourceData(RuleSourceDataList& dataList, RuleSourceDataList& target)
103{
104 for (auto& data : dataList) {
105 if (data->type == WebCore::StyleRuleType::Style)
106 target.append(data.copyRef());
107 else if (data->type == WebCore::StyleRuleType::Media)
108 flattenSourceData(data->childRules, target);
109 else if (data->type == WebCore::StyleRuleType::Supports)
110 flattenSourceData(data->childRules, target);
111 }
112}
113
114void ParsedStyleSheet::setSourceData(std::unique_ptr<RuleSourceDataList> sourceData)
115{
116 if (!sourceData) {
117 m_sourceData.reset();
118 return;
119 }
120
121 m_sourceData = makeUnique<RuleSourceDataList>();
122
123 // FIXME: This is a temporary solution to retain the original flat sourceData structure
124 // containing only style rules, even though CSSParser now provides the full rule source data tree.
125 // Normally, we should just assign m_sourceData = sourceData;
126 flattenSourceData(*sourceData, *m_sourceData);
127}
128
129WebCore::CSSRuleSourceData* ParsedStyleSheet::ruleSourceDataAt(unsigned index) const
130{
131 if (!hasSourceData() || index >= m_sourceData->size())
132 return nullptr;
133
134 return m_sourceData->at(index).ptr();
135}

178void StyleSheetHandler::startRuleHeader(StyleRuleType type, unsigned offset)
179{
180 // Pop off data for a previous invalid rule.
181 if (m_currentRuleData)
182 m_currentRuleDataStack.removeLast();
183
184 auto data = CSSRuleSourceData::create(type);
185 data->ruleHeaderRange.start = offset;
186 m_currentRuleData = data.copyRef();
187 m_currentRuleDataStack.append(WTFMove(data));
188}
189
190template <typename CharacterType> inline void StyleSheetHandler::setRuleHeaderEnd(const CharacterType* dataStart, unsigned listEndOffset)
191{
192 while (listEndOffset > 1) {
193 if (isHTMLSpace<CharacterType>(*(dataStart + listEndOffset - 1)))
194 --listEndOffset;
195 else
196 break;
197 }
198
199 m_currentRuleDataStack.last()->ruleHeaderRange.end = listEndOffset;
200 if (!m_currentRuleDataStack.last()->selectorRanges.isEmpty())
201 m_currentRuleDataStack.last()->selectorRanges.last().end = listEndOffset;
202}
203
204void StyleSheetHandler::endRuleHeader(unsigned offset)
205{
206 ASSERT(!m_currentRuleDataStack.isEmpty());
207
208 if (m_parsedText.is8Bit())
209 setRuleHeaderEnd<LChar>(m_parsedText.characters8(), offset);
210 else
211 setRuleHeaderEnd<UChar>(m_parsedText.characters16(), offset);
212}

214void StyleSheetHandler::observeSelector(unsigned startOffset, unsigned endOffset)

216 ASSERT(m_currentRuleDataStack.size());
217 m_currentRuleDataStack.last()->selectorRanges.append(SourceRange(startOffset, endOffset));
218}

220void StyleSheetHandler::startRuleBody(unsigned offset)
221{
222 m_currentRuleData = nullptr;
223 ASSERT(!m_currentRuleDataStack.isEmpty());
224
225 // Skip the rule body opening brace.
226 if (m_parsedText[offset] == '{')
227 ++offset;
228
229 m_currentRuleDataStack.last()->ruleBodyRange.start = offset;
230}
231
232void StyleSheetHandler::endRuleBody(unsigned offset)
233{
234 ASSERT(!m_currentRuleDataStack.isEmpty());
235 m_currentRuleDataStack.last()->ruleBodyRange.end = offset;
236 auto rule = popRuleData();
237 fixUnparsedPropertyRanges(rule.ptr());
238 if (m_currentRuleDataStack.isEmpty())
239 m_ruleSourceDataResult->append(WTFMove(rule));
240 else
241 m_currentRuleDataStack.last()->childRules.append(WTFMove(rule));
242}

244Ref<CSSRuleSourceData> StyleSheetHandler::popRuleData()
245{
246 ASSERT(!m_currentRuleDataStack.isEmpty());
247 m_currentRuleData = nullptr;
248 auto data = WTFMove(m_currentRuleDataStack.last());
249 m_currentRuleDataStack.removeLast();
250 return data;
251}

253template <typename CharacterType>
254static inline void fixUnparsedProperties(const CharacterType* characters, CSSRuleSourceData* ruleData)
255{
256 Vector<CSSPropertySourceData>& propertyData = ruleData->styleSourceData->propertyData;
257 unsigned size = propertyData.size();
258 if (!size)
259 return;
260
261 unsigned styleStart = ruleData->ruleBodyRange.start;
262
263 CSSPropertySourceData* nextData = &(propertyData.at(0));
264 for (unsigned i = 0; i < size; ++i) {
265 CSSPropertySourceData* currentData = nextData;
266 nextData = i < size - 1 ? &(propertyData.at(i + 1)) : nullptr;
267
268 if (currentData->parsedOk)
269 continue;
270 if (currentData->range.end > 0 && characters[styleStart + currentData->range.end - 1] == ';')
271 continue;
272
273 unsigned propertyEnd;
274 if (!nextData)
275 propertyEnd = ruleData->ruleBodyRange.end - 1;
276 else
277 propertyEnd = styleStart + nextData->range.start - 1;
278
279 while (isHTMLSpace<CharacterType>(characters[propertyEnd]))
280 --propertyEnd;
281
282 // propertyEnd points at the last property text character.
283 unsigned newRangeEnd = (propertyEnd - styleStart) + 1;
284 if (currentData->range.end != newRangeEnd) {
285 currentData->range.end = newRangeEnd;
286 unsigned valueStart = styleStart + currentData->range.start + currentData->name.length();
287 while (valueStart < propertyEnd && characters[valueStart] != ':')
288 ++valueStart;
289
290 // Shift past the ':'.
291 if (valueStart < propertyEnd)
292 ++valueStart;
293
294 while (valueStart < propertyEnd && isHTMLSpace<CharacterType>(characters[valueStart]))
295 ++valueStart;
296
297 // Need to exclude the trailing ';' from the property value.
298 currentData->value = String(characters + valueStart, propertyEnd - valueStart + (characters[propertyEnd] == ';' ? 0 : 1));
299 }
300 }
301}

303void StyleSheetHandler::fixUnparsedPropertyRanges(CSSRuleSourceData* ruleData)
304{
305 if (!ruleData->styleSourceData)
306 return;
307
308 if (m_parsedText.is8Bit()) {
309 fixUnparsedProperties<LChar>(m_parsedText.characters8(), ruleData);
310 return;
311 }
312
313 fixUnparsedProperties<UChar>(m_parsedText.characters16(), ruleData);
314}

316void StyleSheetHandler::observeProperty(unsigned startOffset, unsigned endOffset, bool isImportant, bool isParsed)
317{
318 if (m_currentRuleDataStack.isEmpty() || !m_currentRuleDataStack.last()->styleSourceData)
319 return;
320
321 ASSERT(endOffset <= m_parsedText.length());
322
323 // Include semicolon in the property text.
324 if (endOffset < m_parsedText.length() && m_parsedText[endOffset] == ';')
325 ++endOffset;
326
327 ASSERT(startOffset < endOffset);
328 String propertyString = m_parsedText.substring(startOffset, endOffset - startOffset).stripWhiteSpace();
329 if (propertyString.endsWith(';'))
330 propertyString = propertyString.left(propertyString.length() - 1);
331 size_t colonIndex = propertyString.find(':');
332 ASSERT(colonIndex != notFound);
333
334 String name = propertyString.left(colonIndex).stripWhiteSpace();
335 String value = propertyString.substring(colonIndex + 1, propertyString.length()).stripWhiteSpace();
336
337 // FIXME-NEWPARSER: The property range is relative to the declaration start offset, but no
338 // good reason for it, and it complicates fixUnparsedProperties.
339 SourceRange& topRuleBodyRange = m_currentRuleDataStack.last()->ruleBodyRange;
340 m_currentRuleDataStack.last()->styleSourceData->propertyData.append(CSSPropertySourceData(name, value, isImportant, false, isParsed, SourceRange(startOffset - topRuleBodyRange.start, endOffset - topRuleBodyRange.start)));
341}

343void StyleSheetHandler::observeComment(unsigned startOffset, unsigned endOffset)
344{
345 ASSERT(endOffset <= m_parsedText.length());
346
347 if (m_currentRuleDataStack.isEmpty() || !m_currentRuleDataStack.last()->ruleHeaderRange.end || !m_currentRuleDataStack.last()->styleSourceData)
348 return;
349
350 // The lexer is not inside a property AND it is scanning a declaration-aware
351 // rule body.
352 String commentText = m_parsedText.substring(startOffset, endOffset - startOffset);
353
354 ASSERT(commentText.startsWith("/*"));
355 commentText = commentText.substring(2);
356
357 // Require well-formed comments.
358 if (!commentText.endsWith("*/"))
359 return;
360 commentText = commentText.substring(0, commentText.length() - 2).stripWhiteSpace();
361 if (commentText.isEmpty())
362 return;
363
364 // FIXME: Use the actual rule type rather than STYLE_RULE?
365 RuleSourceDataList sourceData;
366
367 StyleSheetHandler handler(commentText, m_document, &sourceData);
368 CSSParser::parseDeclarationForInspector(parserContextForDocument(m_document), commentText, handler);
369 Vector<CSSPropertySourceData>& commentPropertyData = sourceData.first()->styleSourceData->propertyData;
370 if (commentPropertyData.size() != 1)
371 return;
372 CSSPropertySourceData& propertyData = commentPropertyData.at(0);
373 bool parsedOk = propertyData.parsedOk || propertyData.name.startsWith("-moz-") || propertyData.name.startsWith("-o-") || propertyData.name.startsWith("-webkit-") || propertyData.name.startsWith("-ms-");
374 if (!parsedOk || propertyData.range.length() != commentText.length())
375 return;
376
377 // FIXME-NEWPARSER: The property range is relative to the declaration start offset, but no
378 // good reason for it, and it complicates fixUnparsedProperties.
379 SourceRange& topRuleBodyRange = m_currentRuleDataStack.last()->ruleBodyRange;
380 m_currentRuleDataStack.last()->styleSourceData->propertyData.append(CSSPropertySourceData(propertyData.name, propertyData.value, false, true, true, SourceRange(startOffset - topRuleBodyRange.start, endOffset - topRuleBodyRange.start)));
381}

383static RefPtr<Protocol::CSS::SourceRange> buildSourceRangeObject(const SourceRange& range, const Vector<size_t>& lineEndings, int* endingLine = nullptr)
384{
385 if (lineEndings.isEmpty())
386 return nullptr;
387
388 TextPosition start = ContentSearchUtilities::textPositionFromOffset(range.start, lineEndings);
389 TextPosition end = ContentSearchUtilities::textPositionFromOffset(range.end, lineEndings);
390
391 if (endingLine)
392 *endingLine = end.m_line.zeroBasedInt();
393
394 return Protocol::CSS::SourceRange::create()
395 .setStartLine(start.m_line.zeroBasedInt())
396 .setStartColumn(start.m_column.zeroBasedInt())
397 .setEndLine(end.m_line.zeroBasedInt())
398 .setEndColumn(end.m_column.zeroBasedInt())
399 .release();
400}

402static RefPtr<CSSRuleList> asCSSRuleList(CSSStyleSheet* styleSheet)
403{
404 if (!styleSheet)
405 return nullptr;
406
407 auto list = StaticCSSRuleList::create();
408 Vector<RefPtr<CSSRule>>& listRules = list->rules();
409 for (unsigned i = 0, size = styleSheet->length(); i < size; ++i)
410 listRules.append(styleSheet->item(i));
411 return list;
412}

414static RefPtr<CSSRuleList> asCSSRuleList(CSSRule* rule)
415{
416 if (!rule)
417 return nullptr;
418
419 if (is<CSSMediaRule>(*rule))
420 return &downcast<CSSMediaRule>(*rule).cssRules();
421
422 if (is<CSSKeyframesRule>(*rule))
423 return &downcast<CSSKeyframesRule>(*rule).cssRules();
424
425 if (is<CSSSupportsRule>(*rule))
426 return &downcast<CSSSupportsRule>(*rule).cssRules();
427
428 return nullptr;
429}

431static Ref<JSON::ArrayOf<Protocol::CSS::Grouping>> buildArrayForGroupings(CSSRule& rule)
432{
433 auto groupingsPayload = JSON::ArrayOf<Protocol::CSS::Grouping>::create();
434
435 auto* parentRule = &rule;
436 while (parentRule) {
437 RefPtr<Protocol::CSS::Grouping> ruleGroupingPayload;
438
439 if (is<CSSMediaRule>(parentRule)) {
440 auto* media = downcast<CSSMediaRule>(parentRule)->media();
441 if (media && media->length() && media->mediaText() != "all") {
442 ruleGroupingPayload = Protocol::CSS::Grouping::create()
443 .setText(media->mediaText())
444 .setType(Protocol::CSS::Grouping::Type::MediaRule)
445 .release();
446 }
447 } else if (is<CSSImportRule>(parentRule)) {
448 auto& media = downcast<CSSImportRule>(parentRule)->media();
449 if (media.length() && media.mediaText() != "all") {
450 ruleGroupingPayload = Protocol::CSS::Grouping::create()
451 .setText(media.mediaText())
452 .setType(Protocol::CSS::Grouping::Type::MediaImportRule)
453 .release();
454 }
455 } else if (is<CSSSupportsRule>(parentRule)) {
456 ruleGroupingPayload = Protocol::CSS::Grouping::create()
457 .setText(downcast<CSSSupportsRule>(parentRule)->conditionText())
458 .setType(Protocol::CSS::Grouping::Type::SupportsRule)
459 .release();
460 }
461
462 if (ruleGroupingPayload) {
463 if (auto* parentStyleSheet = parentRule->parentStyleSheet()) {
464 String sourceURL = parentStyleSheet->contents().baseURL().string();
465 if (sourceURL.isEmpty()) {
466 if (auto* ownerDocument = parentStyleSheet->ownerDocument())
467 sourceURL = InspectorDOMAgent::documentURLString(ownerDocument);
468 }
469 if (!sourceURL.isEmpty())
470 ruleGroupingPayload->setSourceURL(sourceURL);
471 }
472
473 groupingsPayload->addItem(ruleGroupingPayload.releaseNonNull());
474 }
475
476 if (parentRule->parentRule()) {
477 parentRule = parentRule->parentRule();
478 continue;
479 }
480
481 auto* styleSheet = parentRule->parentStyleSheet();
482 while (styleSheet) {
483 auto* media = styleSheet->media();
484 if (media && media->length() && media->mediaText() != "all") {
485 auto sheetGroupingPayload = Protocol::CSS::Grouping::create()
486 .setText(media->mediaText())
487 .setType(is<HTMLStyleElement>(styleSheet->ownerNode()) ? Protocol::CSS::Grouping::Type::MediaStyleNode: Protocol::CSS::Grouping::Type::MediaLinkNode)
488 .release();
489
490 String sourceURL;
491 if (auto* ownerDocument = styleSheet->ownerDocument())
492 sourceURL = ownerDocument->url().string();
493 else if (!styleSheet->contents().baseURL().isEmpty())
494 sourceURL = styleSheet->contents().baseURL().string();
495 if (!sourceURL.isEmpty())
496 sheetGroupingPayload->setSourceURL(sourceURL);
497
498 groupingsPayload->addItem(WTFMove(sheetGroupingPayload));
499 }
500
501 parentRule = styleSheet->ownerRule();
502 if (parentRule)
503 break;
504
505 styleSheet = styleSheet->parentStyleSheet();
506 }
507 }
508
509 return groupingsPayload;
510}

527{
528 auto result = styleWithProperties();

572static String lowercasePropertyName(const String& name)

574 // Custom properties are case-sensitive.
575 if (name.startsWith("--"))

577 return name.convertToASCIILowercase();

580Vector<InspectorStyleProperty> InspectorStyle::collectProperties(bool includeAll) const
581{
582 Vector<InspectorStyleProperty> result;
583 HashSet<String> sourcePropertyNames;
584
585 auto sourceData = extractSourceData();
586 auto* sourcePropertyData = sourceData ? &sourceData->styleSourceData->propertyData : nullptr;
587 if (sourcePropertyData) {
588 auto styleDeclarationOrException = text();
589 ASSERT(!styleDeclarationOrException.hasException());
590 String styleDeclaration = styleDeclarationOrException.hasException() ? emptyString() : styleDeclarationOrException.releaseReturnValue();
591 for (auto& sourceData : *sourcePropertyData) {
592 InspectorStyleProperty p(sourceData, true, sourceData.disabled);
593 p.setRawTextFromStyleDeclaration(styleDeclaration);
594 result.append(p);
595 sourcePropertyNames.add(lowercasePropertyName(sourceData.name));
596 }
597 }
598
599 for (int i = 0, size = m_style->length(); i < size; ++i) {
600 String name = m_style->item(i);
601 if (sourcePropertyNames.add(lowercasePropertyName(name)))
602 result.append(InspectorStyleProperty(CSSPropertySourceData(name, m_style->getPropertyValue(name), !m_style->getPropertyPriority(name).isEmpty(), false, true, SourceRange()), false, false));
603 }
604
605 if (includeAll) {
606 for (auto i = firstCSSProperty; i < lastCSSProperty; ++i) {
607 auto id = convertToCSSPropertyID(i);
608 // FIXME: Should take account for flags in settings().
609 if (isInternalCSSProperty(id) || !isEnabledCSSProperty(id))
610 continue;
611
612 auto name = getPropertyNameString(id);
613 if (!sourcePropertyNames.add(lowercasePropertyName(name)))
614 continue;
615
616 auto value = m_style->getPropertyValue(name);
617 if (value.isEmpty())
618 continue;
619
620 result.append(InspectorStyleProperty(CSSPropertySourceData(name, value, !m_style->getPropertyPriority(name).isEmpty(), false, true, SourceRange()), false, false));
621 }
622 }
623
624 return result;
625}

627Ref<Protocol::CSS::CSSStyle> InspectorStyle::styleWithProperties() const
628{
629 auto properties = collectProperties(false);
630
631 auto propertiesObject = JSON::ArrayOf<Protocol::CSS::CSSProperty>::create();
632 auto shorthandEntries = ArrayOf<Protocol::CSS::ShorthandEntry>::create();
633 HashMap<String, RefPtr<Protocol::CSS::CSSProperty>> propertyNameToPreviousActiveProperty;
634 HashSet<String> foundShorthands;
635 String previousPriority;
636 String previousStatus;
637 Vector<size_t> lineEndings = m_parentStyleSheet ? m_parentStyleSheet->lineEndings() : Vector<size_t> { };
638 auto sourceData = extractSourceData();
639 unsigned ruleBodyRangeStart = sourceData ? sourceData->ruleBodyRange.start : 0;
640
641 for (Vector<InspectorStyleProperty>::iterator it = properties.begin(), itEnd = properties.end(); it != itEnd; ++it) {
642 const CSSPropertySourceData& propertyEntry = it->sourceData;
643 const String& name = propertyEntry.name;
644
645 auto status = it->disabled ? Protocol::CSS::CSSPropertyStatus::Disabled : Protocol::CSS::CSSPropertyStatus::Active;
646
647 auto property = Protocol::CSS::CSSProperty::create()
648 .setName(lowercasePropertyName(name))
649 .setValue(propertyEntry.value)
650 .release();
651
652 propertiesObject->addItem(property.copyRef());
653
654 CSSPropertyID propertyId = cssPropertyID(name);
655
656 // Default "parsedOk" == true.
657 if (!propertyEntry.parsedOk || isInternalCSSProperty(propertyId))
658 property->setParsedOk(false);
659 if (it->hasRawText())
660 property->setText(it->rawText);
661
662 // Default "priority" == "".
663 if (propertyEntry.important)
664 property->setPriority("important");
665
666 if (it->hasSource) {
667 // The property range is relative to the style body start.
668 // Should be converted into an absolute range (relative to the stylesheet start)
669 // for the proper conversion into line:column.
670 SourceRange absolutePropertyRange = propertyEntry.range;
671 absolutePropertyRange.start += ruleBodyRangeStart;
672 absolutePropertyRange.end += ruleBodyRangeStart;
673 if (auto range = buildSourceRangeObject(absolutePropertyRange, lineEndings))
674 property->setRange(range.releaseNonNull());
675 }
676
677 if (!it->disabled) {
678 if (it->hasSource) {
679 ASSERT(sourceData);
680 property->setImplicit(false);
681
682 // Parsed property overrides any property with the same name. Non-parsed property overrides
683 // previous non-parsed property with the same name (if any).
684 bool shouldInactivate = false;
685
686 // Canonicalize property names to treat non-prefixed and vendor-prefixed property names the same (opacity vs. -webkit-opacity).
687 String canonicalPropertyName = propertyId ? getPropertyNameString(propertyId) : name;
688 HashMap<String, RefPtr<Protocol::CSS::CSSProperty>>::iterator activeIt = propertyNameToPreviousActiveProperty.find(canonicalPropertyName);
689 if (activeIt != propertyNameToPreviousActiveProperty.end()) {
690 if (propertyEntry.parsedOk) {
691 auto newPriority = activeIt->value->getString(Protocol::CSS::CSSProperty::priorityKey);
692 if (!!newPriority)
693 previousPriority = newPriority;
694
695 auto newStatus = activeIt->value->getString(Protocol::CSS::CSSProperty::statusKey);
696 if (!!newStatus) {
697 previousStatus = newStatus;
698 if (previousStatus != Protocol::Helpers::getEnumConstantValue(Protocol::CSS::CSSPropertyStatus::Inactive)) {
699 if (propertyEntry.important || !newPriority) // Priority not set == "not important".
700 shouldInactivate = true;
701 else if (status == Protocol::CSS::CSSPropertyStatus::Active) {
702 // Inactivate a non-important property following the same-named important property.
703 status = Protocol::CSS::CSSPropertyStatus::Inactive;
704 }
705 }
706 }
707 } else {
708 auto previousParsedOk = activeIt->value->getBoolean(Protocol::CSS::CSSProperty::parsedOkKey);
709 if (previousParsedOk && !previousParsedOk)
710 shouldInactivate = true;
711 }
712 } else
713 propertyNameToPreviousActiveProperty.set(canonicalPropertyName, property.copyRef());
714
715 if (shouldInactivate) {
716 activeIt->value->setStatus(Protocol::CSS::CSSPropertyStatus::Inactive);
717 propertyNameToPreviousActiveProperty.set(canonicalPropertyName, property.copyRef());
718 }
719 } else {
720 bool implicit = m_style->isPropertyImplicit(name);
721 // Default "implicit" == false.
722 if (implicit)
723 property->setImplicit(true);
724 status = Protocol::CSS::CSSPropertyStatus::Style;
725
726 String shorthand = m_style->getPropertyShorthand(name);
727 if (!shorthand.isEmpty()) {
728 if (!foundShorthands.contains(shorthand)) {
729 foundShorthands.add(shorthand);
730 auto entry = Protocol::CSS::ShorthandEntry::create()
731 .setName(shorthand)
732 .setValue(shorthandValue(shorthand))
733 .release();
734 shorthandEntries->addItem(WTFMove(entry));
735 }
736 }
737 }
738 }
739
740 // Default "status" == "style".
741 if (status != Protocol::CSS::CSSPropertyStatus::Style)
742 property->setStatus(status);
743 }
744
745 return Protocol::CSS::CSSStyle::create()
746 .setCssProperties(WTFMove(propertiesObject))
747 .setShorthandEntries(WTFMove(shorthandEntries))
748 .release();
749}

763String InspectorStyle::shorthandValue(const String& shorthandProperty) const
764{
765 String value = m_style->getPropertyValue(shorthandProperty);
766 if (!value.isEmpty())
767 return value;
768 StringBuilder builder;
769 for (unsigned i = 0; i < m_style->length(); ++i) {
770 String individualProperty = m_style->item(i);
771 if (m_style->getPropertyShorthand(individualProperty) != shorthandProperty)
772 continue;
773 if (m_style->isPropertyImplicit(individualProperty))
774 continue;
775 String individualValue = m_style->getPropertyValue(individualProperty);
776 if (individualValue == "initial")
777 continue;
778 if (!builder.isEmpty())
779 builder.append(' ');
780 builder.append(individualValue);
781 }
782 return builder.toString();
783}

791 if (m_style->getPropertyShorthand(individualProperty) != shorthandProperty)

891
892ExceptionOr<void> InspectorStyleSheet::setRuleSelector(const InspectorCSSId& id, const String& selector)
893{
894 if (!m_pageStyleSheet)
895 return Exception { NotSupportedError };
896
897 // If the selector is invalid, do not proceed any further.
898 if (!isValidSelectorListString(selector, m_pageStyleSheet->ownerDocument()))
899 return Exception { SyntaxError };
900
901 CSSStyleRule* rule = ruleForId(id);
902 if (!rule)
903 return Exception { NotFoundError };
904
905 CSSStyleSheet* styleSheet = rule->parentStyleSheet();
906 if (!styleSheet || !ensureParsedDataReady())
907 return Exception { NotFoundError };
908
909 // If the stylesheet is already mutated at this point, that must mean that our data has been modified
910 // elsewhere. This should never happen as ensureParsedDataReady would return false in that case.
911 ASSERT(!styleSheetMutated());
912
913 rule->setSelectorText(selector);
914 auto sourceData = ruleSourceDataFor(&rule->style());
915 if (!sourceData)
916 return Exception { NotFoundError };
917
918 String sheetText = m_parsedStyleSheet->text();
919 sheetText.replace(sourceData->ruleHeaderRange.start, sourceData->ruleHeaderRange.length(), selector);
920 m_parsedStyleSheet->setText(sheetText);
921 m_pageStyleSheet->clearHadRulesMutation();
922 fireStyleSheetChanged();
923 return { };
924}
925
926ExceptionOr<CSSStyleRule*> InspectorStyleSheet::addRule(const String& selector)
927{
928 if (!m_pageStyleSheet)
929 return Exception { NotSupportedError };
930
931 if (!isValidSelectorListString(selector, m_pageStyleSheet->ownerDocument()))
932 return Exception { SyntaxError };
933
934 auto text = this->text();
935 if (text.hasException())
936 return text.releaseException();
937
938 auto addRuleResult = m_pageStyleSheet->addRule(selector, emptyString(), std::nullopt);
939 if (addRuleResult.hasException())
940 return addRuleResult.releaseException();
941
942 StringBuilder styleSheetText;
943 styleSheetText.append(text.releaseReturnValue());
944
945 if (!styleSheetText.isEmpty())
946 styleSheetText.append('\n');
947
948 styleSheetText.append(selector, " {}");
949
950 // Using setText() as this operation changes the stylesheet rule set.
951 setText(styleSheetText.toString());
952
953 // Inspector Style Sheets are always treated as though their parsed data is ready.
954 if (m_origin == Protocol::CSS::StyleSheetOrigin::Inspector)
955 fireStyleSheetChanged();
956 else
957 reparseStyleSheet(styleSheetText.toString());
958
959 ASSERT(m_pageStyleSheet->length());
960 unsigned lastRuleIndex = m_pageStyleSheet->length() - 1;
961 CSSRule* rule = m_pageStyleSheet->item(lastRuleIndex);
962 ASSERT(rule);
963
964 CSSStyleRule* styleRule = InspectorCSSAgent::asCSSStyleRule(*rule);
965 if (!styleRule) {
966 // What we just added has to be a CSSStyleRule - we cannot handle other types of rules yet.
967 // If it is not a style rule, pretend we never touched the stylesheet.
968 m_pageStyleSheet->deleteRule(lastRuleIndex);
969 return Exception { SyntaxError };
970 }
971
972 return styleRule;
973}
974
975ExceptionOr<void> InspectorStyleSheet::deleteRule(const InspectorCSSId& id)
976{
977 if (!m_pageStyleSheet)
978 return Exception { NotSupportedError };
979
980 RefPtr<CSSStyleRule> rule = ruleForId(id);
981 if (!rule)
982 return Exception { NotFoundError };
983 CSSStyleSheet* styleSheet = rule->parentStyleSheet();
984 if (!styleSheet || !ensureParsedDataReady())
985 return Exception { NotFoundError };
986
987 auto sourceData = ruleSourceDataFor(&rule->style());
988 if (!sourceData)
989 return Exception { NotFoundError };
990
991 auto deleteRuleResult = styleSheet->deleteRule(id.ordinal());
992 if (deleteRuleResult.hasException())
993 return deleteRuleResult.releaseException();
994
995 // |rule| MAY NOT be addressed after this!
996
997 String sheetText = m_parsedStyleSheet->text();
998 sheetText.remove(sourceData->ruleHeaderRange.start, sourceData->ruleBodyRange.end - sourceData->ruleHeaderRange.start + 1);
999 setText(sheetText);
1000 fireStyleSheetChanged();
1001 return { };
1002}

1055static Ref<Protocol::CSS::CSSSelector> buildObjectForSelectorHelper(const String& selectorText, const CSSSelector& selector)
1056{
1057 auto inspectorSelector = Protocol::CSS::CSSSelector::create()
1058 .setText(selectorText)
1059 .release();
1060
1061 auto specificity = selector.computeSpecificity();
1062
1063 auto tuple = JSON::ArrayOf<int>::create();
1064 tuple->addItem(static_cast<int>((specificity & CSSSelector::idMask) >> 16));
1065 tuple->addItem(static_cast<int>((specificity & CSSSelector::classMask) >> 8));
1066 tuple->addItem(static_cast<int>(specificity & CSSSelector::elementMask));
1067 inspectorSelector->setSpecificity(WTFMove(tuple));
1068
1069 return inspectorSelector;
1070}
1071
1072static Ref<JSON::ArrayOf<Protocol::CSS::CSSSelector>> selectorsFromSource(const CSSRuleSourceData* sourceData, const String& sheetText, const CSSSelectorList& selectorList)
1073{
1074 static NeverDestroyed<JSC::Yarr::RegularExpression> comment("/\\*[^]*?\\*/", JSC::Yarr::TextCaseSensitive, JSC::Yarr::MultilineEnabled);
1075
1076 auto result = JSON::ArrayOf<Protocol::CSS::CSSSelector>::create();
1077 const CSSSelector* selector = selectorList.first();
1078 for (auto& range : sourceData->selectorRanges) {
1079 // If we don't have a selector, that means the SourceData for this CSSStyleSheet
1080 // no longer matches up with the actual rules in the CSSStyleSheet.
1081 ASSERT(selector);
1082 if (!selector)
1083 break;
1084
1085 String selectorText = sheetText.substring(range.start, range.length());
1086
1087 // We don't want to see any comments in the selector components, only the meaningful parts.
1088 replace(selectorText, comment, String());
1089 result->addItem(buildObjectForSelectorHelper(selectorText.stripWhiteSpace(), *selector));
1090
1091 selector = CSSSelectorList::next(selector);
1092 }
1093 return result;
1094}

1101Ref<Protocol::CSS::SelectorList> InspectorStyleSheet::buildObjectForSelectorList(CSSStyleRule* rule, int& endingLine)
1102{
1103 RefPtr<CSSRuleSourceData> sourceData;
1104 if (ensureParsedDataReady())
1105 sourceData = ruleSourceDataFor(&rule->style());
1106 RefPtr<JSON::ArrayOf<Protocol::CSS::CSSSelector>> selectors;
1107
1108 // This intentionally does not rely on the source data to avoid catching the trailing comments (before the declaration starting '{').
1109 String selectorText = rule->selectorText();
1110
1111 if (sourceData)
1112 selectors = selectorsFromSource(sourceData.get(), m_parsedStyleSheet->text(), rule->styleRule().selectorList());
1113 else {
1114 selectors = JSON::ArrayOf<Protocol::CSS::CSSSelector>::create();
1115 const CSSSelectorList& selectorList = rule->styleRule().selectorList();
1116 for (const CSSSelector* selector = selectorList.first(); selector; selector = CSSSelectorList::next(selector))
1117 selectors->addItem(buildObjectForSelector(selector));
1118 }
1119 auto result = Protocol::CSS::SelectorList::create()
1120 .setSelectors(selectors.releaseNonNull())
1121 .setText(selectorText)
1122 .release();
1123 if (sourceData) {
1124 if (auto range = buildSourceRangeObject(sourceData->ruleHeaderRange, lineEndings(), &endingLine))
1125 result->setRange(range.releaseNonNull());
1126 }
1127 return result;
1128}

1442void InspectorStyleSheet::collectFlatRules(RefPtr<CSSRuleList>&& ruleList, CSSStyleRuleVector* result)
1443{

1445 return;
