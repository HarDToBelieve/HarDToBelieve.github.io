48class InspectorCSSId {
49public:
50 InspectorCSSId() = default;
51
52 explicit InspectorCSSId(const JSON::Object& value)
53 {
54 m_styleSheetId = value.getString("styleSheetId"_s);
55 if (!m_styleSheetId)
56 return;
57
58 if (auto ordinal = value.getInteger("ordinal"_s))
59 m_ordinal = *ordinal;
60 else
61 m_styleSheetId = String();
62 }
63
64 InspectorCSSId(const String& styleSheetId, unsigned ordinal)
65 : m_styleSheetId(styleSheetId)
66 , m_ordinal(ordinal)
67 {
68 }
69
70 bool isEmpty() const { return m_styleSheetId.isEmpty(); }
71
72 const String& styleSheetId() const { return m_styleSheetId; }
73 unsigned ordinal() const { return m_ordinal; }
74
75 // ID type is either Inspector::Protocol::CSS::CSSStyleId or Inspector::Protocol::CSS::CSSRuleId.
76 template<typename ID>
77 RefPtr<ID> asProtocolValue() const
78 {
79 if (isEmpty())
80 return nullptr;
81
82 return ID::create()
83 .setStyleSheetId(m_styleSheetId)
84 .setOrdinal(m_ordinal)
85 .release();
86 }
87
88private:
89 String m_styleSheetId;
90 unsigned m_ordinal = {0};
91};

93struct InspectorStyleProperty {
94 InspectorStyleProperty()
95 : hasSource(false)
96 , disabled(false)
97 {
98 }
99
100 InspectorStyleProperty(CSSPropertySourceData sourceData, bool hasSource, bool disabled)
101 : sourceData(sourceData)
102 , hasSource(hasSource)
103 , disabled(disabled)
104 {
105 }
106
107 void setRawTextFromStyleDeclaration(const String& styleDeclaration)
108 {
109 unsigned start = sourceData.range.start;
110 unsigned end = sourceData.range.end;
111 ASSERT_WITH_SECURITY_IMPLICATION(start < end);
112 ASSERT(end <= styleDeclaration.length());
113 rawText = styleDeclaration.substring(start, end - start);
114 }
115
116 bool hasRawText() const { return !rawText.isEmpty(); }
117
118 CSSPropertySourceData sourceData;
119 bool hasSource;
120 bool disabled;
121 String rawText;
122};
