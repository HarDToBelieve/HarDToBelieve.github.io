61unsigned NetworkResourcesData::ResourceData::removeContent()
62{
63 unsigned result = 0;
64 if (hasData()) {
65 ASSERT(!hasContent());
66 result = m_dataBuffer->size();
67 m_dataBuffer = nullptr;
68 }
69
70 if (hasContent()) {
71 ASSERT(!hasData());
72 result = m_content.sizeInBytes();
73 m_content = String();
74 }
75 return result;
76}
77
78unsigned NetworkResourcesData::ResourceData::evictContent()
79{
80 m_isContentEvicted = true;
81 return removeContent();
82}

89void NetworkResourcesData::ResourceData::appendData(const uint8_t* data, size_t dataLength)
90{
91 ASSERT(!hasContent());
92 if (!m_dataBuffer)
93 m_dataBuffer = SharedBuffer::create(data, dataLength);
94 else
95 m_dataBuffer->append(data, dataLength);
96}
97
98unsigned NetworkResourcesData::ResourceData::decodeDataToContent()
99{
100 ASSERT(!hasContent());
101
102 size_t dataLength = m_dataBuffer->size();
103
104 if (m_decoder) {
105 m_base64Encoded = false;
106 m_content = m_decoder->decodeAndFlush(m_dataBuffer->data(), dataLength);
107 } else {
108 m_base64Encoded = true;
109 m_content = base64EncodeToString(m_dataBuffer->data(), dataLength);
110 }
111
112 m_dataBuffer = nullptr;
113
114 return m_content.sizeInBytes() - dataLength;
115}

146void NetworkResourcesData::responseReceived(const String& requestId, const String& frameId, const ResourceResponse& response, InspectorPageAgent::ResourceType type, bool forceBufferData)
147{
148 ResourceData* resourceData = resourceDataForRequestId(requestId);
149 if (!resourceData)
150 return;
151
152 resourceData->setFrameId(frameId);
153 resourceData->setURL(response.url().string());
154 resourceData->setHTTPStatusCode(response.httpStatusCode());
155 resourceData->setHTTPStatusText(response.httpStatusText());
156 resourceData->setType(type);
157 resourceData->setForceBufferData(forceBufferData);
158 resourceData->setMIMEType(response.mimeType());
159 resourceData->setResponseTimestamp(WallTime::now());
160
161 if (InspectorNetworkAgent::shouldTreatAsText(response.mimeType()))
162 resourceData->setDecoder(InspectorNetworkAgent::createTextDecoder(response.mimeType(), response.textEncodingName()));
163
164 if (auto& certificateInfo = response.certificateInfo())
165 resourceData->setCertificateInfo(certificateInfo);
166}

184void NetworkResourcesData::setResourceContent(const String& requestId, const String& content, bool base64Encoded)
185{
186 if (content.isNull())
187 return;
188
189 ResourceData* resourceData = resourceDataForRequestId(requestId);
190 if (!resourceData)
191 return;
192
193 size_t dataLength = content.sizeInBytes();
194 if (dataLength > m_maximumSingleResourceContentSize)
195 return;
196 if (resourceData->isContentEvicted())
197 return;
198
199 if (ensureFreeSpace(dataLength) && !resourceData->isContentEvicted()) {
200 // We can not be sure that we didn't try to save this request data while it was loading, so remove it, if any.
201 if (resourceData->hasContent() || resourceData->hasData())
202 m_contentSize -= resourceData->removeContent();
203 m_requestIdsDeque.append(requestId);
204 resourceData->setContent(content, base64Encoded);
205 m_contentSize += dataLength;
206 }
207}

209static bool shouldBufferResourceData(const NetworkResourcesData::ResourceData& resourceData)
210{
211 if (resourceData.forceBufferData())
212 return true;
213
214 if (resourceData.decoder())
215 return true;
216
217 // Buffer data for Web Inspector when the rest of the system would not normally buffer.
218 if (resourceData.cachedResource() && resourceData.cachedResource()->dataBufferingPolicy() == DataBufferingPolicy::DoNotBufferData)
219 return true;
220
221 return false;
222}

224NetworkResourcesData::ResourceData const* NetworkResourcesData::maybeAddResourceData(const String& requestId, const uint8_t* data, size_t dataLength)
225{
226 ResourceData* resourceData = resourceDataForRequestId(requestId);
227 if (!resourceData)
228 return nullptr;
229
230 if (!shouldBufferResourceData(*resourceData))
231 return resourceData;
232
233 if (resourceData->dataLength() + dataLength > m_maximumSingleResourceContentSize)
234 m_contentSize -= resourceData->evictContent();
235 if (resourceData->isContentEvicted())
236 return resourceData;
237
238 if (ensureFreeSpace(dataLength) && !resourceData->isContentEvicted()) {
239 m_requestIdsDeque.append(requestId);
240 resourceData->appendData(data, dataLength);
241 m_contentSize += dataLength;
242 }
243
244 return resourceData;
245}

247void NetworkResourcesData::maybeDecodeDataToContent(const String& requestId)
248{
249 ResourceData* resourceData = resourceDataForRequestId(requestId);
250 if (!resourceData)
251 return;
252
253 if (!resourceData->hasData())
254 return;
255
256 m_contentSize += resourceData->decodeDataToContent();
257 size_t dataLength = resourceData->content().sizeInBytes();
258 if (dataLength > m_maximumSingleResourceContentSize)
259 m_contentSize -= resourceData->evictContent();
260}

284NetworkResourcesData::ResourceData const* NetworkResourcesData::dataForURL(const String& url)
285{
286 if (url.isNull())
287 return nullptr;
288
289 NetworkResourcesData::ResourceData* mostRecentResourceData = nullptr;
290
291 for (auto* resourceData : resources()) {
292 // responseTimestamp is checked so that we only grab the most recent response for the URL, instead of potentionally getting a more stale response.
293 if (resourceData->url() == url && resourceData->httpStatusCode() != 304 && (!mostRecentResourceData || (resourceData->responseTimestamp() > mostRecentResourceData->responseTimestamp())))
294 mostRecentResourceData = resourceData;
295 }
296
297 return mostRecentResourceData;
298}

300Vector<String> NetworkResourcesData::removeCachedResource(CachedResource* cachedResource)
301{
302 Vector<String> result;
303 for (auto& entry : m_requestIdToResourceDataMap) {
304 ResourceData* resourceData = entry.value.get();
305 if (resourceData->cachedResource() == cachedResource) {
306 resourceData->setCachedResource(nullptr);
307 result.append(entry.key);
308 }
309 }
310
311 return result;
312}

314void NetworkResourcesData::clear(std::optional<String> preservedLoaderId)
315{
316 m_requestIdsDeque.clear();
317 m_contentSize = 0;
318
319 if (!preservedLoaderId)
320 m_requestIdToResourceDataMap.clear();
321 else {
322 m_requestIdToResourceDataMap.removeIf([loaderId = *preservedLoaderId] (auto& entry) {
323 return entry.value->loaderId() != loaderId;
324 });
325 }
326}

340void NetworkResourcesData::ensureNoDataForRequestId(const String& requestId)
341{
342 auto result = m_requestIdToResourceDataMap.take(requestId);
343 if (!result)
344 return;
345
346 ResourceData* resourceData = result.get();
347 if (resourceData->hasContent() || resourceData->hasData())
348 m_contentSize -= resourceData->evictContent();
349}
350
351bool NetworkResourcesData::ensureFreeSpace(size_t size)
352{
353 if (size > m_maximumResourcesContentSize)
354 return false;
355
356 while (size > m_maximumResourcesContentSize - m_contentSize) {
357 String requestId = m_requestIdsDeque.takeFirst();
358 ResourceData* resourceData = resourceDataForRequestId(requestId);
359 if (resourceData)
360 m_contentSize -= resourceData->evictContent();
361 }
362 return true;
363}
