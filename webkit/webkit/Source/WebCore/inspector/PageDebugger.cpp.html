56PageDebugger::PageDebugger(Page& page)
57 : Debugger(WebCore::commonVM())
58 , m_page(page)

60}

62void PageDebugger::attachDebugger()
63{
64 JSC::Debugger::attachDebugger();
65
66 m_page.setDebugger(this);
67}

69void PageDebugger::detachDebugger(bool isBeingDestroyed)
70{
71 JSC::Debugger::detachDebugger(isBeingDestroyed);
72
73 m_page.setDebugger(nullptr);
74 if (!isBeingDestroyed)
75 recompileAllJSFunctions();
76}

98void PageDebugger::runEventLoopWhilePaused()
99{
100 JSC::Debugger::runEventLoopWhilePaused();
101
102#if PLATFORM(IOS_FAMILY)
103 // On iOS, running an EventLoop causes us to run a nested WebRunLoop.
104 // Since the WebThread is autoreleased at the end of run loop iterations
105 // we need to gracefully handle releasing and reacquiring the lock.
106 if (WebThreadIsEnabled()) {
107 ASSERT(WebThreadIsLockedOrDisabled());
108 JSC::JSLock::DropAllLocks dropAllLocks(vm());
109 WebRunLoopEnableNested();
110
111 runEventLoopWhilePausedInternal();
112
113 WebRunLoopDisableNested();
114 ASSERT(WebThreadIsLockedOrDisabled());
115 return;
116 }
117#endif
118
119 runEventLoopWhilePausedInternal();
120}
121
122void PageDebugger::runEventLoopWhilePausedInternal()
123{
124 TimerBase::fireTimersInNestedEventLoop();
125
126 m_page.incrementNestedRunLoopCount();
127
128 while (!doneProcessingDebuggerEvents()) {
129 if (!platformShouldContinueRunningEventLoopWhilePaused())
130 break;
131 }
132
133 m_page.decrementNestedRunLoopCount();
134}

148void PageDebugger::setJavaScriptPaused(const PageGroup& pageGroup, bool paused)
149{
150 for (auto& page : pageGroup.pages()) {
151 for (auto* frame = &page.mainFrame(); frame; frame = frame->tree().traverseNext())
152 setJavaScriptPaused(*frame, paused);
153
154 if (auto* frontendClient = page.inspectorController().inspectorFrontendClient()) {
155 if (paused)
156 frontendClient->pagePaused();
157 else
158 frontendClient->pageUnpaused();
159 }
160 }
161}
162
163void PageDebugger::setJavaScriptPaused(Frame& frame, bool paused)
164{
165 if (!frame.script().canExecuteScripts(NotAboutToExecuteScript))
166 return;
167
168 frame.script().setPaused(paused);
169
170 ASSERT(frame.document());
171 auto& document = *frame.document();
172 if (paused) {
173 document.suspendScriptedAnimationControllerCallbacks();
174 document.suspendActiveDOMObjects(ReasonForSuspension::JavaScriptDebuggerPaused);
175 } else {
176 document.resumeActiveDOMObjects(ReasonForSuspension::JavaScriptDebuggerPaused);
177 document.resumeScriptedAnimationControllerCallbacks();
178 }
179
180 if (auto* view = frame.view()) {
181 for (auto& child : view->children()) {
182 if (!is<PluginViewBase>(child))
183 continue;
184 downcast<PluginViewBase>(child.get()).setJavaScriptPaused(paused);
185 }
186 }
187}
