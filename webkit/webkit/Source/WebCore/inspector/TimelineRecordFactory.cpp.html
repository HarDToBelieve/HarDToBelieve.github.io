46Ref<JSON::Object> TimelineRecordFactory::createGenericRecord(double startTime, int maxCallStackDepth)
47{
48 Ref<JSON::Object> record = JSON::Object::create();
49 record->setDouble("startTime"_s, startTime);
50
51 if (maxCallStackDepth) {
52 Ref<ScriptCallStack> stackTrace = createScriptCallStack(JSExecState::currentState(), maxCallStackDepth);
53 if (stackTrace->size())
54 record->setValue("stackTrace"_s, stackTrace->buildInspectorArray());
55 }
56 return record;
57}

143static Ref<JSON::Array> createQuad(const FloatQuad& quad)
144{
145 Ref<JSON::Array> array = JSON::Array::create();
146 array->pushDouble(quad.p1().x());
147 array->pushDouble(quad.p1().y());
148 array->pushDouble(quad.p2().x());
149 array->pushDouble(quad.p2().y());
150 array->pushDouble(quad.p3().x());
151 array->pushDouble(quad.p3().y());
152 array->pushDouble(quad.p4().x());
153 array->pushDouble(quad.p4().y());
154 return array;
155}

157Ref<JSON::Object> TimelineRecordFactory::createPaintData(const FloatQuad& quad)
158{
159 Ref<JSON::Object> data = JSON::Object::create();

162}
