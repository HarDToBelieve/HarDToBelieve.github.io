37WebInjectedScriptManager::WebInjectedScriptManager(InspectorEnvironment& environment, Ref<InjectedScriptHost>&& host)
38 : InjectedScriptManager(environment, WTFMove(host))
39{
40}

42void WebInjectedScriptManager::connect()
43{
44 InjectedScriptManager::connect();
45
46 m_commandLineAPIHost = CommandLineAPIHost::create();
47}
48
49void WebInjectedScriptManager::disconnect()
50{
51 InjectedScriptManager::disconnect();
52
53 if (m_commandLineAPIHost) {
54 m_commandLineAPIHost->disconnect();
55 m_commandLineAPIHost = nullptr;
56 }
57}

59void WebInjectedScriptManager::discardInjectedScripts()
60{
61 InjectedScriptManager::discardInjectedScripts();
62
63 if (m_commandLineAPIHost)
64 m_commandLineAPIHost->clearAllWrappers();
65}

72void WebInjectedScriptManager::discardInjectedScriptsFor(DOMWindow* window)
73{
74 if (m_scriptStateToId.isEmpty())
75 return;
76
77 Vector<long> idsToRemove;
78 for (const auto& it : m_idToInjectedScript) {
79 JSC::JSGlobalObject* lexicalGlobalObject = it.value.globalObject();
80 if (window != domWindowFromExecState(lexicalGlobalObject))
81 continue;
82 m_scriptStateToId.remove(lexicalGlobalObject);
83 idsToRemove.append(it.key);
84 }
85
86 for (auto& id : idsToRemove)
87 m_idToInjectedScript.remove(id);
88
89 // Now remove script states that have id but no injected script.
90 Vector<JSC::JSGlobalObject*> scriptStatesToRemove;
91 for (const auto& it : m_scriptStateToId) {
92 JSC::JSGlobalObject* lexicalGlobalObject = it.key;
93 if (window == domWindowFromExecState(lexicalGlobalObject))
94 scriptStatesToRemove.append(lexicalGlobalObject);
95 }
96
97 for (auto& lexicalGlobalObject : scriptStatesToRemove)
98 m_scriptStateToId.remove(lexicalGlobalObject);
99}
