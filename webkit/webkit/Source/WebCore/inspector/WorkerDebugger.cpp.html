47WorkerDebugger::WorkerDebugger(WorkerOrWorkletGlobalScope& context)
48 : Debugger(context.script()->vm())
49 , m_globalScope(context)
50{
51}
52
53void WorkerDebugger::attachDebugger()
54{
55 JSC::Debugger::attachDebugger();
56
57 m_globalScope.script()->attachDebugger(this);
58}
59
60void WorkerDebugger::detachDebugger(bool isBeingDestroyed)
61{
62 JSC::Debugger::detachDebugger(isBeingDestroyed);
63
64 if (m_globalScope.script())
65 m_globalScope.script()->detachDebugger(this);
66 if (!isBeingDestroyed)
67 recompileAllJSFunctions();
68}
69
70void WorkerDebugger::recompileAllJSFunctions()
71{
72 JSC::JSLockHolder lock(vm());
73 JSC::Debugger::recompileAllJSFunctions();
74}
75
76void WorkerDebugger::runEventLoopWhilePaused()
77{
78 JSC::Debugger::runEventLoopWhilePaused();
79
80 TimerBase::fireTimersInNestedEventLoop();
81
82 MessageQueueWaitResult result;
83 do {
84 result = m_globalScope.workerOrWorkletThread()->runLoop().runInDebuggerMode(m_globalScope);
85 } while (result != MessageQueueTerminated && !doneProcessingDebuggerEvents());
86}
87
88void WorkerDebugger::reportException(JSC::JSGlobalObject* exec, JSC::Exception* exception) const
89{
90 JSC::Debugger::reportException(exec, exception);
91
92 WebCore::reportException(exec, exception);
93}
