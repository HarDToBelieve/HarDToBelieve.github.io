28#include "WorkerDebuggerProxy.h"

31#include <JavaScriptCore/InspectorFrontendChannel.h>

33namespace WebCore {
34
35class WorkerToPageFrontendChannel final : public Inspector::FrontendChannel {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 explicit WorkerToPageFrontendChannel(WorkerOrWorkletGlobalScope& globalScope)
39 : m_globalScope(globalScope)
40 {
41 }
42 ~WorkerToPageFrontendChannel() override = default;
43
44private:
45 ConnectionType connectionType() const override { return ConnectionType::Local; }
46
47 void sendMessageToFrontend(const String& message) override
48 {
49 if (auto* workerDebuggerProxy = m_globalScope.workerOrWorkletThread()->workerDebuggerProxy())
50 workerDebuggerProxy->postMessageToDebugger(message);
51 }
52
53 WorkerOrWorkletGlobalScope& m_globalScope;
54};
