62Protocol::ErrorStringOr<void> InspectorApplicationCacheAgent::enable()
63{
64 if (m_instrumentingAgents.enabledApplicationCacheAgent() == this)
65 return makeUnexpected("ApplicationCache domain already enabled"_s);
66
67 m_instrumentingAgents.setEnabledApplicationCacheAgent(this);
68
69 // We need to pass initial navigator.onOnline.
70 networkStateChanged();
71
72 return { };
73}
74
75Protocol::ErrorStringOr<void> InspectorApplicationCacheAgent::disable()
76{
77 if (m_instrumentingAgents.enabledApplicationCacheAgent() != this)
78 return makeUnexpected("ApplicationCache domain already disabled"_s);
79
80 m_instrumentingAgents.setEnabledApplicationCacheAgent(nullptr);
81
82 return { };
83}

85void InspectorApplicationCacheAgent::updateApplicationCacheStatus(Frame* frame)
86{
87 auto* pageAgent = m_instrumentingAgents.enabledPageAgent();
88 if (!pageAgent)
89 return;
90
91 if (!frame)
92 return;
93
94 auto* documentLoader = frame->loader().documentLoader();
95 if (!documentLoader)
96 return;
97
98 auto& host = documentLoader->applicationCacheHost();
99 int status = host.status();
100 auto manifestURL = host.applicationCacheInfo().manifest.string();
101
102 m_frontendDispatcher->applicationCacheStatusUpdated(pageAgent->frameId(frame), manifestURL, status);
103}

108}

110Expected<Ref<JSON::ArrayOf<Protocol::ApplicationCache::FrameWithManifest>>, Protocol::ErrorString> InspectorApplicationCacheAgent::getFramesWithManifests()
111{
112 auto* pageAgent = m_instrumentingAgents.enabledPageAgent();
113 if (!pageAgent)
114 return makeUnexpected("Page domain must be enabled"_s);
115
116 auto result = JSON::ArrayOf<Protocol::ApplicationCache::FrameWithManifest>::create();
117 for (Frame* frame = &m_inspectedPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {
118 auto* documentLoader = frame->loader().documentLoader();
119 if (!documentLoader)
120 continue;
121
122 auto& host = documentLoader->applicationCacheHost();
123 String manifestURL = host.applicationCacheInfo().manifest.string();
124 if (!manifestURL.isEmpty()) {
125 result->addItem(Protocol::ApplicationCache::FrameWithManifest::create()
126 .setFrameId(pageAgent->frameId(frame))
127 .setManifestURL(manifestURL)
128 .setStatus(static_cast<int>(host.status()))
129 .release());
130 }
131 }
132 return result;
133}

135DocumentLoader* InspectorApplicationCacheAgent::assertFrameWithDocumentLoader(Protocol::ErrorString& errorString, const Inspector::Protocol::Network::FrameId& frameId)
136{
137 auto* pageAgent = m_instrumentingAgents.enabledPageAgent();
138 if (!pageAgent) {
139 errorString = "Page domain must be enabled"_s;
140 return nullptr;
141 }
142
143 auto* frame = pageAgent->assertFrame(errorString, frameId);
144 if (!frame)
145 return nullptr;
146
147 return InspectorPageAgent::assertDocumentLoader(errorString, frame);
148}

150Expected<String, Protocol::ErrorString> InspectorApplicationCacheAgent::getManifestForFrame(const Inspector::Protocol::Network::FrameId& frameId)
151{
152 Protocol::ErrorString errorString;
153
154 DocumentLoader* documentLoader = assertFrameWithDocumentLoader(errorString, frameId);
155 if (!documentLoader)
156 return makeUnexpected(errorString);
157
158 return documentLoader->applicationCacheHost().applicationCacheInfo().manifest.string();
159}

161Expected<Ref<Protocol::ApplicationCache::ApplicationCache>, Protocol::ErrorString> InspectorApplicationCacheAgent::getApplicationCacheForFrame(const Inspector::Protocol::Network::FrameId& frameId)
162{
163 Protocol::ErrorString errorString;
164
165 auto* documentLoader = assertFrameWithDocumentLoader(errorString, frameId);
166 if (!documentLoader)
167 return makeUnexpected(errorString);
168
169 auto& host = documentLoader->applicationCacheHost();
170 return buildObjectForApplicationCache(host.resourceList(), host.applicationCacheInfo());
171}

173Ref<Protocol::ApplicationCache::ApplicationCache> InspectorApplicationCacheAgent::buildObjectForApplicationCache(const Vector<ApplicationCacheHost::ResourceInfo>& applicationCacheResources, const ApplicationCacheHost::CacheInfo& applicationCacheInfo)
174{
175 return Protocol::ApplicationCache::ApplicationCache::create()
176 .setManifestURL(applicationCacheInfo.manifest.string())
177 .setSize(applicationCacheInfo.size)
178 .setCreationTime(applicationCacheInfo.creationTime)
179 .setUpdateTime(applicationCacheInfo.updateTime)
180 .setResources(buildArrayForApplicationCacheResources(applicationCacheResources))
181 .release();
182}
183
184Ref<JSON::ArrayOf<Protocol::ApplicationCache::ApplicationCacheResource>> InspectorApplicationCacheAgent::buildArrayForApplicationCacheResources(const Vector<ApplicationCacheHost::ResourceInfo>& applicationCacheResources)
185{
186 auto result = JSON::ArrayOf<Protocol::ApplicationCache::ApplicationCacheResource>::create();
187 for (auto& info : applicationCacheResources)
188 result->addItem(buildObjectForApplicationCacheResource(info));
189 return result;
190}
191
192Ref<Protocol::ApplicationCache::ApplicationCacheResource> InspectorApplicationCacheAgent::buildObjectForApplicationCacheResource(const ApplicationCacheHost::ResourceInfo& resourceInfo)
193{
194 auto types = makeString(resourceInfo.isMaster ? "Master " : "",
195 resourceInfo.isManifest ? "Manifest " : "",
196 resourceInfo.isFallback ? "Fallback " : "",
197 resourceInfo.isForeign ? "Foreign " : "",
198 resourceInfo.isExplicit ? "Explicit " : "");
199 return Protocol::ApplicationCache::ApplicationCacheResource::create()
200 .setUrl(resourceInfo.resource.string())
201 .setSize(static_cast<int>(resourceInfo.size))
202 .setType(types)
203 .release();
204}
