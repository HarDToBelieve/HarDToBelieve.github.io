159 class PendingInterceptRequest {
160 WTF_MAKE_NONCOPYABLE(PendingInterceptRequest);
161 WTF_MAKE_FAST_ALLOCATED;
162 public:
163 PendingInterceptRequest(RefPtr<ResourceLoader> loader, Function<void(const ResourceRequest&)>&& callback)
164 : m_loader(loader)
165 , m_completionCallback(WTFMove(callback))
166 { }
167
168 void continueWithOriginalRequest()
169 {
170 if (!m_loader->reachedTerminalState())
171 m_completionCallback(m_loader->request());
172 }
173
174 void continueWithRequest(const ResourceRequest& request)
175 {
176 m_completionCallback(request);
177 }
178
179 PendingInterceptRequest() = default;
180 RefPtr<ResourceLoader> m_loader;
181 Function<void(const ResourceRequest&)> m_completionCallback;
182 };

184 class PendingInterceptResponse {
185 WTF_MAKE_NONCOPYABLE(PendingInterceptResponse);
186 WTF_MAKE_FAST_ALLOCATED;
187 public:
188 PendingInterceptResponse(const ResourceResponse& originalResponse, CompletionHandler<void(const ResourceResponse&, RefPtr<SharedBuffer>)>&& completionHandler)
189 : m_originalResponse(originalResponse)
190 , m_completionHandler(WTFMove(completionHandler))
191 { }
192
193 ~PendingInterceptResponse()
194 {
195 ASSERT(m_responded);
196 }
197
198 ResourceResponse originalResponse() { return m_originalResponse; }
199
200 void respondWithOriginalResponse()
201 {
202 respond(m_originalResponse, nullptr);
203 }
204
205 void respond(const ResourceResponse& response, RefPtr<SharedBuffer> data)
206 {
207 ASSERT(!m_responded);
208 if (m_responded)
209 return;
210
211 m_responded = true;
212
213 m_completionHandler(response, data);
214 }
215
216 private:
217 ResourceResponse m_originalResponse;
218 CompletionHandler<void(const ResourceResponse&, RefPtr<SharedBuffer>)> m_completionHandler;
219 bool m_responded { false };
220 };

231 struct Intercept {
232 String url;
233 bool caseSensitive { true };
234 bool isRegex { false };
235 Inspector::Protocol::Network::NetworkStage networkStage { Inspector::Protocol::Network::NetworkStage::Response };
236
237 inline bool operator==(const Intercept& other) const
238 {
239 return url == other.url
240 && caseSensitive == other.caseSensitive
241 && isRegex == other.isRegex
242 && networkStage == other.networkStage;
243 }
244 };
