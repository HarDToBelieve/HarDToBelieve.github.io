88static bool decodeBuffer(const uint8_t* buffer, unsigned size, const String& textEncodingName, String* result)
89{
90 if (buffer) {
91 TextEncoding encoding(textEncodingName);
92 if (!encoding.isValid())
93 encoding = WindowsLatin1Encoding();
94 *result = encoding.decode(buffer, size);
95 return true;
96 }
97 return false;
98}

113bool InspectorPageAgent::dataContent(const uint8_t* data, unsigned size, const String& textEncodingName, bool withBase64Encode, String* result)
114{
115 if (withBase64Encode) {
116 *result = base64EncodeToString(data, size);

119
120 return decodeBuffer(data, size, textEncodingName, result);

123Vector<CachedResource*> InspectorPageAgent::cachedResourcesForFrame(Frame* frame)
124{
125 Vector<CachedResource*> result;
126
127 for (auto& cachedResourceHandle : frame->document()->cachedResourceLoader().allCachedResources().values()) {
128 auto* cachedResource = cachedResourceHandle.get();
129 if (cachedResource->resourceRequest().hiddenFromInspector())
130 continue;
131
132 switch (cachedResource->type()) {
133 case CachedResource::Type::ImageResource:
134 // Skip images that were not auto loaded (images disabled in the user agent).
135 case CachedResource::Type::SVGFontResource:
136 case CachedResource::Type::FontResource:
137 // Skip fonts that were referenced in CSS but never used/downloaded.
138 if (cachedResource->stillNeedsLoad())
139 continue;
140 break;
141 default:
142 // All other CachedResource types download immediately.
143 break;
144 }
145
146 result.append(cachedResource);
147 }
148
149 return result;
150}

152void InspectorPageAgent::resourceContent(Protocol::ErrorString& errorString, Frame* frame, const URL& url, String* result, bool* base64Encoded)
153{
154 DocumentLoader* loader = assertDocumentLoader(errorString, frame);
155 if (!loader)
156 return;
157
158 RefPtr<SharedBuffer> buffer;
159 bool success = false;
160 if (equalIgnoringFragmentIdentifier(url, loader->url())) {
161 *base64Encoded = false;
162 success = mainResourceContent(frame, *base64Encoded, result);
163 }
164
165 if (!success) {
166 if (auto* resource = cachedResource(frame, url))
167 success = InspectorNetworkAgent::cachedResourceContent(*resource, result, base64Encoded);
168 }
169
170 if (!success)
171 errorString = "Missing resource for given url"_s;
172}

174String InspectorPageAgent::sourceMapURLForResource(CachedResource* cachedResource)
175{
176 if (!cachedResource)
177 return String();
178
179 // Scripts are handled in a separate path.
180 if (cachedResource->type() != CachedResource::Type::CSSStyleSheet)
181 return String();
182
183 String sourceMapHeader = cachedResource->response().httpHeaderField(HTTPHeaderName::SourceMap);
184 if (!sourceMapHeader.isEmpty())
185 return sourceMapHeader;
186
187 sourceMapHeader = cachedResource->response().httpHeaderField(HTTPHeaderName::XSourceMap);
188 if (!sourceMapHeader.isEmpty())
189 return sourceMapHeader;
190
191 String content;
192 bool base64Encoded;
193 if (InspectorNetworkAgent::cachedResourceContent(*cachedResource, &content, &base64Encoded) && !base64Encoded)
194 return ContentSearchUtilities::findStylesheetSourceMapURL(content);
195
196 return String();
197}

199CachedResource* InspectorPageAgent::cachedResource(Frame* frame, const URL& url)
200{
201 if (url.isNull())
202 return nullptr;
203
204 CachedResource* cachedResource = frame->document()->cachedResourceLoader().cachedResource(MemoryCache::removeFragmentIdentifierIfNeeded(url));
205 if (!cachedResource) {
206 ResourceRequest request(url);
207 request.setDomainForCachePartition(frame->document()->domainForCachePartition());
208 cachedResource = MemoryCache::singleton().resourceForRequest(request, frame->page()->sessionID());
209 }
210
211 return cachedResource;
212}

280InspectorPageAgent::ResourceType InspectorPageAgent::inspectorResourceType(const CachedResource& cachedResource)
281{
282 if (cachedResource.type() == CachedResource::Type::RawResource) {
283 switch (cachedResource.resourceRequest().requester()) {
284 case ResourceRequest::Requester::Fetch:
285 return InspectorPageAgent::FetchResource;
286 case ResourceRequest::Requester::Main:
287 return InspectorPageAgent::DocumentResource;
288 default:
289 return InspectorPageAgent::XHRResource;
290 }
291 }
292
293 return inspectorResourceType(cachedResource.type());

340Protocol::ErrorStringOr<void> InspectorPageAgent::enable()
341{
342 if (m_instrumentingAgents.enabledPageAgent() == this)
343 return makeUnexpected("Page domain already enabled"_s);
344
345 m_instrumentingAgents.setEnabledPageAgent(this);
346
347 auto& stopwatch = m_environment.executionStopwatch();
348 stopwatch.reset();
349 stopwatch.start();
350
351#if ENABLE(DARK_MODE_CSS) || HAVE(OS_DARK_MODE_SUPPORT)
352 defaultAppearanceDidChange(m_inspectedPage.defaultUseDarkAppearance());
353#endif
354
355 return { };
356}

409Protocol::ErrorStringOr<void> InspectorPageAgent::navigate(const String& url)
410{
411 UserGestureIndicator indicator { ProcessingUserGesture };
412 Frame& frame = m_inspectedPage.mainFrame();
413
414 ResourceRequest resourceRequest { frame.document()->completeURL(url) };
415 FrameLoadRequest frameLoadRequest { *frame.document(), frame.document()->securityOrigin(), WTFMove(resourceRequest), "_self"_s, InitiatedByMainFrame::Unknown };
416 frameLoadRequest.disableNavigationToInvalidURL();
417 frame.loader().changeLocation(WTFMove(frameLoadRequest));
418

506static Ref<Protocol::Page::Cookie> buildObjectForCookie(const Cookie& cookie)
507{
508 return Protocol::Page::Cookie::create()
509 .setName(cookie.name)
510 .setValue(cookie.value)

543Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::Page::Cookie>>> InspectorPageAgent::getCookies()
544{
545 ListHashSet<Cookie> allRawCookies;
546
547 for (Frame* frame = &m_inspectedPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {
548 Document* document = frame->document();
549 if (!document || !document->page())
550 continue;
551
552 for (auto& url : allResourcesURLsForFrame(frame)) {
553 Vector<Cookie> rawCookiesForURLInDocument;
554 if (!document->page()->cookieJar().getRawCookies(*document, url, rawCookiesForURLInDocument))
555 continue;
556
557 for (auto& rawCookieForURLInDocument : rawCookiesForURLInDocument)
558 allRawCookies.add(rawCookieForURLInDocument);
559 }
560 }
561
562 return buildArrayForCookies(allRawCookies);
563}

565static std::optional<Cookie> parseCookieObject(Protocol::ErrorString& errorString, Ref<JSON::Object>&& cookieObject)
566{
567 Cookie cookie;
568
569 cookie.name = cookieObject->getString(Protocol::Page::Cookie::nameKey);
570 if (!cookie.name) {
571 errorString = "Invalid value for key name in given cookie";
572 return std::nullopt;
573 }
574
575 cookie.value = cookieObject->getString(Protocol::Page::Cookie::valueKey);
576 if (!cookie.value) {
577 errorString = "Invalid value for key value in given cookie";
578 return std::nullopt;
579 }
580
581 cookie.domain = cookieObject->getString(Protocol::Page::Cookie::domainKey);
582 if (!cookie.domain) {
583 errorString = "Invalid value for key domain in given cookie";
584 return std::nullopt;
585 }
586
587 cookie.path = cookieObject->getString(Protocol::Page::Cookie::pathKey);
588 if (!cookie.path) {
589 errorString = "Invalid value for key path in given cookie";
590 return std::nullopt;
591 }
592
593 auto httpOnly = cookieObject->getBoolean(Protocol::Page::Cookie::httpOnlyKey);
594 if (!httpOnly) {
595 errorString = "Invalid value for key httpOnly in given cookie";
596 return std::nullopt;
597 }
598
599 cookie.httpOnly = *httpOnly;
600
601 auto secure = cookieObject->getBoolean(Protocol::Page::Cookie::secureKey);
602 if (!secure) {
603 errorString = "Invalid value for key secure in given cookie";
604 return std::nullopt;
605 }
606
607 cookie.secure = *secure;
608
609 auto session = cookieObject->getBoolean(Protocol::Page::Cookie::sessionKey);
610 cookie.expires = cookieObject->getDouble(Protocol::Page::Cookie::expiresKey);
611 if (!session && !cookie.expires) {
612 errorString = "Invalid value for key expires in given cookie";
613 return std::nullopt;
614 }
615
616 cookie.session = *session;
617
618 auto sameSiteString = cookieObject->getString(Protocol::Page::Cookie::sameSiteKey);
619 if (!sameSiteString) {
620 errorString = "Invalid value for key sameSite in given cookie";
621 return std::nullopt;
622 }
623
624 auto sameSite = Protocol::Helpers::parseEnumValueFromString<Protocol::Page::CookieSameSitePolicy>(sameSiteString);
625 if (!sameSite) {
626 errorString = "Invalid value for key sameSite in given cookie";
627 return std::nullopt;
628 }
629
630 switch (sameSite.value()) {
631 case Protocol::Page::CookieSameSitePolicy::None:
632 cookie.sameSite = Cookie::SameSitePolicy::None;
633 break;
634
635 case Protocol::Page::CookieSameSitePolicy::Lax:
636 cookie.sameSite = Cookie::SameSitePolicy::Lax;
637 break;
638
639 case Protocol::Page::CookieSameSitePolicy::Strict:
640 cookie.sameSite = Cookie::SameSitePolicy::Strict;
641 break;
642 }
643
644 return cookie;
645}

647Protocol::ErrorStringOr<void> InspectorPageAgent::setCookie(Ref<JSON::Object>&& cookieObject)
648{
649 Protocol::ErrorString errorString;
650
651 auto cookie = parseCookieObject(errorString, WTFMove(cookieObject));
652 if (!cookie)
653 return makeUnexpected(errorString);
654
655 for (auto* frame = &m_inspectedPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {
656 if (auto* document = frame->document()) {
657 if (auto* page = document->page())
658 page->cookieJar().setRawCookie(*document, cookie.value());
659 }
660 }
661
662 return { };
663}

706Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::GenericTypes::SearchMatch>>> InspectorPageAgent::searchInResource(const Protocol::Network::FrameId& frameId, const String& url, const String& query, std::optional<bool>&& caseSensitive, std::optional<bool>&& isRegex, const Protocol::Network::RequestId& requestId)
707{
708 Protocol::ErrorString errorString;
709
710 if (!!requestId) {
711 if (auto* networkAgent = m_instrumentingAgents.enabledNetworkAgent()) {
712 RefPtr<JSON::ArrayOf<Protocol::GenericTypes::SearchMatch>> result;
713 networkAgent->searchInRequest(errorString, requestId, query, caseSensitive && *caseSensitive, isRegex && *isRegex, result);
714 if (!result)
715 return makeUnexpected(errorString);
716 return result.releaseNonNull();
717 }
718 }
719
720 Frame* frame = assertFrame(errorString, frameId);
721 if (!frame)
722 return makeUnexpected(errorString);
723
724 DocumentLoader* loader = assertDocumentLoader(errorString, frame);
725 if (!loader)
726 return makeUnexpected(errorString);
727
728 URL kurl({ }, url);
729
730 String content;
731 bool success = false;
732 if (equalIgnoringFragmentIdentifier(kurl, loader->url()))
733 success = mainResourceContent(frame, false, &content);
734
735 if (!success) {
736 if (auto* resource = cachedResource(frame, kurl)) {
737 if (auto textContent = InspectorNetworkAgent::textContentForCachedResource(*resource)) {
738 content = *textContent;
739 success = true;
740 }
741 }
742 }
743
744 if (!success)
745 return JSON::ArrayOf<Protocol::GenericTypes::SearchMatch>::create();
746
747 return ContentSearchUtilities::searchInTextByLines(content, query, caseSensitive && *caseSensitive, isRegex && *isRegex);
748}

759Protocol::ErrorStringOr<Ref<JSON::ArrayOf<Protocol::Page::SearchResult>>> InspectorPageAgent::searchInResources(const String& text, std::optional<bool>&& caseSensitive, std::optional<bool>&& isRegex)
760{
761 auto result = JSON::ArrayOf<Protocol::Page::SearchResult>::create();
762
763 auto searchStringType = (isRegex && *isRegex) ? ContentSearchUtilities::SearchStringType::Regex : ContentSearchUtilities::SearchStringType::ContainsString;
764 auto regex = ContentSearchUtilities::createRegularExpressionForSearchString(text, caseSensitive && *caseSensitive, searchStringType);
765
766 for (Frame* frame = &m_inspectedPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {
767 for (auto* cachedResource : cachedResourcesForFrame(frame)) {
768 if (auto textContent = InspectorNetworkAgent::textContentForCachedResource(*cachedResource)) {
769 int matchesCount = ContentSearchUtilities::countRegularExpressionMatches(regex, *textContent);
770 if (matchesCount)
771 result->addItem(buildObjectForSearchResult(frameId(frame), cachedResource->url().string(), matchesCount));
772 }
773 }
774 }
775
776 if (auto* networkAgent = m_instrumentingAgents.enabledNetworkAgent())
777 networkAgent->searchOtherRequests(regex, result);
778
779 return result;
780}

953 .setUrl(frame->document()->url().string())
954 .setMimeType(frame->loader().documentLoader()->responseMIMEType())
955 .setSecurityOrigin(frame->document()->securityOrigin().toRawString())

969Ref<Protocol::Page::FrameResourceTree> InspectorPageAgent::buildObjectForFrameTree(Frame* frame)
970{
971 ASSERT_ARG(frame, frame);
972
973 auto frameObject = buildObjectForFrame(frame);
974 auto subresources = JSON::ArrayOf<Protocol::Page::FrameResource>::create();
975 auto result = Protocol::Page::FrameResourceTree::create()
976 .setFrame(WTFMove(frameObject))
977 .setResources(subresources.copyRef())
978 .release();
979
980 for (auto* cachedResource : cachedResourcesForFrame(frame)) {
981 auto resourceObject = Protocol::Page::FrameResource::create()
982 .setUrl(cachedResource->url().string())
983 .setType(cachedResourceTypeJSON(*cachedResource))
984 .setMimeType(cachedResource->response().mimeType())
985 .release();
986 if (cachedResource->wasCanceled())
987 resourceObject->setCanceled(true);
988 else if (cachedResource->status() == CachedResource::LoadError || cachedResource->status() == CachedResource::DecodeError)
989 resourceObject->setFailed(true);
990 String sourceMappingURL = InspectorPageAgent::sourceMapURLForResource(cachedResource);
991 if (!sourceMappingURL.isEmpty())
992 resourceObject->setSourceMapURL(sourceMappingURL);
993 String targetId = cachedResource->resourceRequest().initiatorIdentifier();
994 if (!targetId.isEmpty())
995 resourceObject->setTargetId(targetId);
996 subresources->addItem(WTFMove(resourceObject));
997 }
998
999 RefPtr<JSON::ArrayOf<Protocol::Page::FrameResourceTree>> childrenArray;
1000 for (Frame* child = frame->tree().firstChild(); child; child = child->tree().nextSibling()) {
1001 if (!childrenArray) {
1002 childrenArray = JSON::ArrayOf<Protocol::Page::FrameResourceTree>::create();
1003 result->setChildFrames(*childrenArray);
1004 }
1005 childrenArray->addItem(buildObjectForFrameTree(child));
1006 }
1007 return result;
1008}
