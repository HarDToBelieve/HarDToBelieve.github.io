75static CFRunLoopRef currentRunLoop()
76{
77#if PLATFORM(IOS_FAMILY)
78 // A race condition during WebView deallocation can lead to a crash if the layer sync run loop
79 // observer is added to the main run loop <rdar://problem/9798550>. However, for responsiveness,
80 // we still allow this, see <rdar://problem/7403328>. Since the race condition and subsequent
81 // crash are especially troublesome for iBooks, we never allow the observer to be added to the
82 // main run loop in iBooks.
83 if (CocoaApplication::isIBooks())
84 return WebThreadRunLoop();
85#endif
86 return CFRunLoopGetCurrent();
87}

159Protocol::ErrorStringOr<void> InspectorTimelineAgent::setInstruments(Ref<JSON::Array>&& instruments)
160{
161 Vector<Protocol::Timeline::Instrument> newInstruments;
162 newInstruments.reserveCapacity(instruments->length());
163
164 for (const auto& instrumentValue : instruments.get()) {
165 auto instrumentString = instrumentValue->asString();
166 if (!instrumentString)
167 return makeUnexpected("Unexpected non-string value in given instruments"_s);
168
169 auto instrument = Protocol::Helpers::parseEnumValueFromString<Protocol::Timeline::Instrument>(instrumentString);
170 if (!instrument)
171 return makeUnexpected(makeString("Unknown instrument: ", instrumentString));
172
173 newInstruments.uncheckedAppend(*instrument);
174 }
175
176 m_instruments.swap(newInstruments);
177
178 return { };
179}

181void InspectorTimelineAgent::internalStart(std::optional<int>&& maxCallStackDepth)
182{
183 if (m_tracking)
184 return;
185
186 if (maxCallStackDepth && *maxCallStackDepth > 0)
187 m_maxCallStackDepth = *maxCallStackDepth;
188 else
189 m_maxCallStackDepth = 5;
190
191 m_instrumentingAgents.setTrackingTimelineAgent(this);
192
193 m_environment.debugger().addObserver(*this);
194
195 m_tracking = true;
196
197 // FIXME: Abstract away platform-specific code once https://bugs.webkit.org/show_bug.cgi?id=142748 is fixed.
198
199#if PLATFORM(COCOA)
200 m_frameStartObserver = makeUnique<RunLoopObserver>(static_cast<CFIndex>(RunLoopObserver::WellKnownRunLoopOrders::InspectorFrameBegin), [this]() {
201 if (!m_tracking || m_environment.debugger().isPaused())
202 return;
203
204 if (!m_runLoopNestingLevel)
205 pushCurrentRecord(JSON::Object::create(), TimelineRecordType::RenderingFrame, false, nullptr);
206 m_runLoopNestingLevel++;
207 });
208
209 m_frameStopObserver = makeUnique<RunLoopObserver>(static_cast<CFIndex>(RunLoopObserver::WellKnownRunLoopOrders::InspectorFrameEnd), [this]() {
210 if (!m_tracking || m_environment.debugger().isPaused())
211 return;
212
213 ASSERT(m_runLoopNestingLevel > 0);
214 m_runLoopNestingLevel--;
215 if (m_runLoopNestingLevel)
216 return;
217
218 if (m_startedComposite)
219 didComposite();
220
221 didCompleteCurrentRecord(TimelineRecordType::RenderingFrame);
222 });
223
224 m_frameStartObserver->schedule(currentRunLoop(), kCFRunLoopEntry | kCFRunLoopAfterWaiting);
225 m_frameStopObserver->schedule(currentRunLoop(), kCFRunLoopExit | kCFRunLoopBeforeWaiting);
226
227 // Create a runloop record and increment the runloop nesting level, to capture the current turn of the main runloop
228 // (which is the outer runloop if recording started while paused in the debugger).
229 pushCurrentRecord(JSON::Object::create(), TimelineRecordType::RenderingFrame, false, nullptr);
230
231 m_runLoopNestingLevel = 1;
232#elif USE(GLIB_EVENT_LOOP)
233 m_runLoopObserver = makeUnique<RunLoop::Observer>([this](RunLoop::Event event, const String& name) {
234 if (!m_tracking || m_environment.debugger().isPaused())
235 return;
236
237 switch (event) {
238 case RunLoop::Event::WillDispatch:
239 pushCurrentRecord(TimelineRecordFactory::createRenderingFrameData(name), TimelineRecordType::RenderingFrame, false, nullptr);
240 break;
241 case RunLoop::Event::DidDispatch:
242 didCompleteCurrentRecord(TimelineRecordType::RenderingFrame);
243 break;
244 }
245 });
246 RunLoop::current().observe(*m_runLoopObserver);
247#endif
248
249 m_frontendDispatcher->recordingStarted(timestamp());
250
251 if (auto* client = m_inspectedPage.inspectorController().inspectorClient())
252 client->timelineRecordingChanged(true);
253}
254
255void InspectorTimelineAgent::internalStop()
256{
257 if (!m_tracking)
258 return;
259
260 m_instrumentingAgents.setTrackingTimelineAgent(nullptr);
261
262 m_environment.debugger().removeObserver(*this, true);
263
264#if PLATFORM(COCOA)
265 m_frameStartObserver = nullptr;
266 m_frameStopObserver = nullptr;
267 m_runLoopNestingLevel = 0;
268#elif USE(GLIB_EVENT_LOOP)
269 m_runLoopObserver = nullptr;
270#endif
271
272 // Complete all pending records to prevent discarding events that are currently in progress.
273 while (!m_recordStack.isEmpty())
274 didCompleteCurrentRecord(m_recordStack.last().type);
275
276 m_recordStack.clear();
277
278 m_tracking = false;
279 m_startedComposite = false;
280 m_autoCapturePhase = AutoCapturePhase::None;
281
282 m_frontendDispatcher->recordingStopped(timestamp());
283
284 if (auto* client = m_inspectedPage.inspectorController().inspectorClient())
285 client->timelineRecordingChanged(false);
286}

293void InspectorTimelineAgent::startFromConsole(JSC::JSGlobalObject* exec, const String& title)
294{
295 // Allow duplicate unnamed profiles. Disallow duplicate named profiles.
296 if (!title.isEmpty()) {
297 for (const TimelineRecordEntry& record : m_pendingConsoleProfileRecords) {
298 auto recordTitle = record.data->getString("title"_s);
299 if (recordTitle == title) {
300 if (auto* consoleAgent = m_instrumentingAgents.webConsoleAgent()) {
301 // FIXME: Send an enum to the frontend for localization?
302 String warning = title.isEmpty() ? "Unnamed Profile already exists"_s : makeString("Profile \"", ScriptArguments::truncateStringForConsoleMessage(title), "\" already exists");
303 consoleAgent->addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::ConsoleAPI, MessageType::Profile, MessageLevel::Warning, warning));
304 }
305 return;
306 }
307 }
308 }
309
310 if (!m_tracking && m_pendingConsoleProfileRecords.isEmpty())
311 startProgrammaticCapture();
312
313 m_pendingConsoleProfileRecords.append(createRecordEntry(TimelineRecordFactory::createConsoleProfileData(title), TimelineRecordType::ConsoleProfile, true, frameFromExecState(exec)));
314}
315
316void InspectorTimelineAgent::stopFromConsole(JSC::JSGlobalObject*, const String& title)
317{
318 // Stop profiles in reverse order. If the title is empty, then stop the last profile.
319 // Otherwise, match the title of the profile to stop.
320 for (int i = m_pendingConsoleProfileRecords.size() - 1; i >= 0; --i) {
321 const TimelineRecordEntry& record = m_pendingConsoleProfileRecords[i];
322
323 auto recordTitle = record.data->getString("title"_s);
324 if (title.isEmpty() || recordTitle == title) {
325 didCompleteRecordEntry(record);
326 m_pendingConsoleProfileRecords.remove(i);
327
328 if (!m_trackingFromFrontend && m_pendingConsoleProfileRecords.isEmpty())
329 stopProgrammaticCapture();
330
331 return;
332 }
333 }
334
335 if (auto* consoleAgent = m_instrumentingAgents.webConsoleAgent()) {
336 // FIXME: Send an enum to the frontend for localization?
337 String warning = title.isEmpty() ? "No profiles exist"_s : makeString("Profile \"", ScriptArguments::truncateStringForConsoleMessage(title), "\" does not exist");
338 consoleAgent->addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::ConsoleAPI, MessageType::ProfileEnd, MessageLevel::Warning, warning));
339 }
340}

379void InspectorTimelineAgent::didLayout(RenderObject& root)
380{
381 if (m_recordStack.isEmpty())
382 return;
383 TimelineRecordEntry& entry = m_recordStack.last();
384 ASSERT(entry.type == TimelineRecordType::Layout);
385 Vector<FloatQuad> quads;
386 root.absoluteQuads(quads);

390 ASSERT_NOT_REACHED();
391 didCompleteCurrentRecord(TimelineRecordType::Layout);

485void InspectorTimelineAgent::mainFrameStartedLoading()
486{
487 if (m_tracking)
488 return;
489
490 if (!m_autoCaptureEnabled)
491 return;
492
493 if (m_instruments.isEmpty())
494 return;
495
496 m_autoCapturePhase = AutoCapturePhase::BeforeLoad;
497
498 // Pre-emptively disable breakpoints. The frontend must re-enable them.
499 if (auto* webDebuggerAgent = m_instrumentingAgents.enabledWebDebuggerAgent())
500 webDebuggerAgent->setBreakpointsActive(false);
501
502 // Inform the frontend we started an auto capture. The frontend must stop capture.
503 m_frontendDispatcher->autoCaptureStarted();
504
505 toggleInstruments(InstrumentState::Start);
506}
507
508void InspectorTimelineAgent::mainFrameNavigated()
509{
510 if (m_autoCapturePhase == AutoCapturePhase::BeforeLoad) {
511 m_autoCapturePhase = AutoCapturePhase::FirstNavigation;
512 toggleInstruments(InstrumentState::Start);
513 m_autoCapturePhase = AutoCapturePhase::AfterFirstNavigation;
514 }
515}
516
517void InspectorTimelineAgent::startProgrammaticCapture()
518{
519 ASSERT(!m_tracking);
520
521 // Disable breakpoints during programmatic capture.
522 if (auto* webDebuggerAgent = m_instrumentingAgents.enabledWebDebuggerAgent()) {
523 m_programmaticCaptureRestoreBreakpointActiveValue = webDebuggerAgent->breakpointsActive();
524 if (m_programmaticCaptureRestoreBreakpointActiveValue)
525 webDebuggerAgent->setBreakpointsActive(false);
526 } else
527 m_programmaticCaptureRestoreBreakpointActiveValue = false;
528
529 toggleScriptProfilerInstrument(InstrumentState::Start); // Ensure JavaScript samping data.
530 toggleTimelineInstrument(InstrumentState::Start); // Ensure Console Profile event records.
531 toggleInstruments(InstrumentState::Start); // Any other instruments the frontend wants us to record.
532}
533
534void InspectorTimelineAgent::stopProgrammaticCapture()
535{
536 ASSERT(m_tracking);
537 ASSERT(!m_trackingFromFrontend);
538
539 toggleInstruments(InstrumentState::Stop);
540 toggleTimelineInstrument(InstrumentState::Stop);
541 toggleScriptProfilerInstrument(InstrumentState::Stop);
542
543 // Re-enable breakpoints if they were enabled.
544 if (m_programmaticCaptureRestoreBreakpointActiveValue) {
545 if (auto* webDebuggerAgent = m_instrumentingAgents.enabledWebDebuggerAgent())
546 webDebuggerAgent->setBreakpointsActive(true);
547 }
548}

550void InspectorTimelineAgent::toggleInstruments(InstrumentState state)
551{
552 for (auto instrumentType : m_instruments) {
553 switch (instrumentType) {
554 case Protocol::Timeline::Instrument::ScriptProfiler: {
555 toggleScriptProfilerInstrument(state);
556 break;
557 }
558 case Protocol::Timeline::Instrument::Heap: {
559 toggleHeapInstrument(state);
560 break;
561 }
562 case Protocol::Timeline::Instrument::CPU: {
563 toggleCPUInstrument(state);
564 break;
565 }
566 case Protocol::Timeline::Instrument::Memory: {
567 toggleMemoryInstrument(state);
568 break;
569 }
570 case Protocol::Timeline::Instrument::Timeline:
571 toggleTimelineInstrument(state);
572 break;
573 case Protocol::Timeline::Instrument::Animation:
574 toggleAnimationInstrument(state);
575 break;
576 }
577 }
578}

682static Protocol::Timeline::EventType toProtocol(TimelineRecordType type)
683{
684 switch (type) {
685 case TimelineRecordType::EventDispatch:
686 return Protocol::Timeline::EventType::EventDispatch;
687 case TimelineRecordType::ScheduleStyleRecalculation:
688 return Protocol::Timeline::EventType::ScheduleStyleRecalculation;
689 case TimelineRecordType::RecalculateStyles:
690 return Protocol::Timeline::EventType::RecalculateStyles;
691 case TimelineRecordType::InvalidateLayout:
692 return Protocol::Timeline::EventType::InvalidateLayout;
693 case TimelineRecordType::Layout:
694 return Protocol::Timeline::EventType::Layout;
695 case TimelineRecordType::Paint:
696 return Protocol::Timeline::EventType::Paint;
697 case TimelineRecordType::Composite:
698 return Protocol::Timeline::EventType::Composite;
699 case TimelineRecordType::RenderingFrame:
700 return Protocol::Timeline::EventType::RenderingFrame;
701
702 case TimelineRecordType::TimerInstall:
703 return Protocol::Timeline::EventType::TimerInstall;
704 case TimelineRecordType::TimerRemove:
705 return Protocol::Timeline::EventType::TimerRemove;
706 case TimelineRecordType::TimerFire:
707 return Protocol::Timeline::EventType::TimerFire;
708
709 case TimelineRecordType::EvaluateScript:
710 return Protocol::Timeline::EventType::EvaluateScript;
711
712 case TimelineRecordType::TimeStamp:
713 return Protocol::Timeline::EventType::TimeStamp;
714 case TimelineRecordType::Time:
715 return Protocol::Timeline::EventType::Time;
716 case TimelineRecordType::TimeEnd:
717 return Protocol::Timeline::EventType::TimeEnd;
718
719 case TimelineRecordType::FunctionCall:
720 return Protocol::Timeline::EventType::FunctionCall;
721 case TimelineRecordType::ProbeSample:
722 return Protocol::Timeline::EventType::ProbeSample;
723 case TimelineRecordType::ConsoleProfile:
724 return Protocol::Timeline::EventType::ConsoleProfile;
725
726 case TimelineRecordType::RequestAnimationFrame:
727 return Protocol::Timeline::EventType::RequestAnimationFrame;
728 case TimelineRecordType::CancelAnimationFrame:
729 return Protocol::Timeline::EventType::CancelAnimationFrame;
730 case TimelineRecordType::FireAnimationFrame:
731 return Protocol::Timeline::EventType::FireAnimationFrame;
732
733 case TimelineRecordType::ObserverCallback:
734 return Protocol::Timeline::EventType::ObserverCallback;
735 }
736
737 return Protocol::Timeline::EventType::TimeStamp;
738}

740void InspectorTimelineAgent::addRecordToTimeline(Ref<JSON::Object>&& record, TimelineRecordType type)
741{
742 record->setString(Protocol::Timeline::TimelineEvent::typeKey, Protocol::Helpers::getEnumConstantValue(toProtocol(type)));
743
744 if (m_recordStack.isEmpty()) {
745 // FIXME: runtimeCast is a hack. We do it because we can't build TimelineEvent directly now.
746 auto recordObject = Protocol::BindingTraits<Protocol::Timeline::TimelineEvent>::runtimeCast(WTFMove(record));
747 sendEvent(WTFMove(recordObject));
748 } else {
749 const TimelineRecordEntry& parent = m_recordStack.last();
750 // Nested paint records are an implementation detail and add no information not already contained in the parent.
751 if (type == TimelineRecordType::Paint && parent.type == type)
752 return;
753
754 parent.children->pushObject(WTFMove(record));
755 }
756}

770void InspectorTimelineAgent::didCompleteRecordEntry(const TimelineRecordEntry& entry)
771{
772 entry.record->setObject(Protocol::Timeline::TimelineEvent::dataKey, entry.data.copyRef());
773 if (entry.children)
774 entry.record->setArray(Protocol::Timeline::TimelineEvent::childrenKey, *entry.children);
775 entry.record->setDouble("endTime"_s, timestamp());
776 addRecordToTimeline(entry.record.copyRef(), entry.type);
777}

779void InspectorTimelineAgent::didCompleteCurrentRecord(TimelineRecordType type)
780{
781 // An empty stack could merely mean that the timeline agent was turned on in the middle of
782 // an event. Don't treat as an error.
783 if (!m_recordStack.isEmpty()) {
784 TimelineRecordEntry entry = m_recordStack.last();
785 m_recordStack.removeLast();
786 ASSERT_UNUSED(type, entry.type == type);
787
788 // Don't send RenderingFrame records that have no children to reduce noise.
789 if (entry.type == TimelineRecordType::RenderingFrame && !entry.children->length())
790 return;
791
792 didCompleteRecordEntry(entry);
793 }
794}
