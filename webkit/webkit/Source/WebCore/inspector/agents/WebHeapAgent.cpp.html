34namespace WebCore {

36using namespace Inspector;
37
38struct GarbageCollectionData {
39 Protocol::Heap::GarbageCollection::Type type;
40 Seconds startTime;
41 Seconds endTime;
42};
43
44class SendGarbageCollectionEventsTask final {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 SendGarbageCollectionEventsTask(WebHeapAgent&);
48 void addGarbageCollection(GarbageCollectionData&&);
49 void reset();
50private:
51 void timerFired();
52
53 WebHeapAgent& m_agent;
54 Lock m_collectionsLock;
55 Vector<GarbageCollectionData> m_collections WTF_GUARDED_BY_LOCK(m_collectionsLock);
56 RunLoop::Timer<SendGarbageCollectionEventsTask> m_timer;
57};
58
59SendGarbageCollectionEventsTask::SendGarbageCollectionEventsTask(WebHeapAgent& agent)
60 : m_agent(agent)
61 , m_timer(RunLoop::main(), this, &SendGarbageCollectionEventsTask::timerFired)
62{
63}
64
65void SendGarbageCollectionEventsTask::addGarbageCollection(GarbageCollectionData&& collection)
66{
67 {
68 Locker locker { m_collectionsLock };
69 m_collections.append(WTFMove(collection));
70 }
71
72 if (!m_timer.isActive())
73 m_timer.startOneShot(0_s);
74}
75
76void SendGarbageCollectionEventsTask::reset()
77{
78 {
79 Locker locker { m_collectionsLock };
80 m_collections.clear();
81 }
82
83 m_timer.stop();
84}
85
86void SendGarbageCollectionEventsTask::timerFired()
87{
88 Vector<GarbageCollectionData> collectionsToSend;
89
90 {
91 Locker locker { m_collectionsLock };
92 m_collections.swap(collectionsToSend);
93 }
94
95 m_agent.dispatchGarbageCollectionEventsAfterDelay(WTFMove(collectionsToSend));
96}

98WebHeapAgent::WebHeapAgent(WebAgentContext& context)
99 : InspectorHeapAgent(context)
100 , m_instrumentingAgents(context.instrumentingAgents)
101 , m_sendGarbageCollectionEventsTask(makeUnique<SendGarbageCollectionEventsTask>(*this))
102{
103}
104
105WebHeapAgent::~WebHeapAgent() = default;
106
107Protocol::ErrorStringOr<void> WebHeapAgent::enable()
108{
109 auto result = InspectorHeapAgent::enable();
110
111 if (auto* consoleAgent = m_instrumentingAgents.webConsoleAgent())
112 consoleAgent->setHeapAgent(this);
113
114 return result;
115}
116
117Protocol::ErrorStringOr<void> WebHeapAgent::disable()
118{
119 m_sendGarbageCollectionEventsTask->reset();
120
121 if (auto* consoleAgent = m_instrumentingAgents.webConsoleAgent())
122 consoleAgent->setHeapAgent(nullptr);
123
124 return InspectorHeapAgent::disable();
125}
126
127void WebHeapAgent::dispatchGarbageCollectedEvent(Protocol::Heap::GarbageCollection::Type type, Seconds startTime, Seconds endTime)
128{
129 // Dispatch the event asynchronously because this method may be
130 // called between collection and sweeping and we don't want to
131 // create unexpected JavaScript allocations that the Sweeper does
132 // not expect to encounter. JavaScript allocations could happen
133 // with WebKitLegacy's in process inspector which shares the same
134 // VM as the inspected page.
135
136 GarbageCollectionData data = {type, startTime, endTime};
137 m_sendGarbageCollectionEventsTask->addGarbageCollection(WTFMove(data));
138}
139
140void WebHeapAgent::dispatchGarbageCollectionEventsAfterDelay(Vector<GarbageCollectionData>&& collections)
141{
142 for (auto& collection : collections)
143 InspectorHeapAgent::dispatchGarbageCollectedEvent(collection.type, collection.startTime, collection.endTime);
144}
