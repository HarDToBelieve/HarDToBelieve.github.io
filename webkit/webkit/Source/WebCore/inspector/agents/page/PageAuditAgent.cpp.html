59InjectedScript PageAuditAgent::injectedScriptForEval(std::optional<Protocol::Runtime::ExecutionContextId>&& executionContextId)
60{
61 if (executionContextId)
62 return injectedScriptManager().injectedScriptForId(*executionContextId);
63
64 JSC::JSGlobalObject* scriptState = mainWorldExecState(&m_inspectedPage.mainFrame());
65 return injectedScriptManager().injectedScriptFor(scriptState);
66}
67
68InjectedScript PageAuditAgent::injectedScriptForEval(Protocol::ErrorString& errorString, std::optional<Protocol::Runtime::ExecutionContextId>&& executionContextId)
69{
70 InjectedScript injectedScript = injectedScriptForEval(WTFMove(executionContextId));
71 if (injectedScript.hasNoValue()) {
72 if (executionContextId)
73 errorString = "Missing injected script for given executionContextId"_s;
74 else
75 errorString = "Internal error: main world execution context not found"_s;
76 }
77 return injectedScript;
78}

80void PageAuditAgent::populateAuditObject(JSC::JSGlobalObject* lexicalGlobalObject, JSC::Strong<JSC::JSObject>& auditObject)
81{
82 InspectorAuditAgent::populateAuditObject(lexicalGlobalObject, auditObject);
83
84 ASSERT(lexicalGlobalObject);
85 if (!lexicalGlobalObject)
86 return;
87
88 if (auto* globalObject = JSC::jsCast<JSDOMGlobalObject*>(lexicalGlobalObject)) {
89 JSC::VM& vm = globalObject->vm();
90 JSC::JSLockHolder lock(vm);
91
92 if (JSC::JSValue jsInspectorAuditAccessibilityObject = toJSNewlyCreated(lexicalGlobalObject, globalObject, InspectorAuditAccessibilityObject::create(*this)))
93 auditObject->putDirect(vm, JSC::Identifier::fromString(vm, "Accessibility"), jsInspectorAuditAccessibilityObject);
94
95 if (JSC::JSValue jsInspectorAuditDOMObject = toJSNewlyCreated(lexicalGlobalObject, globalObject, InspectorAuditDOMObject::create(*this)))
96 auditObject->putDirect(vm, JSC::Identifier::fromString(vm, "DOM"), jsInspectorAuditDOMObject);
97
98 if (JSC::JSValue jsInspectorAuditResourcesObject = toJSNewlyCreated(lexicalGlobalObject, globalObject, InspectorAuditResourcesObject::create(*this)))
99 auditObject->putDirect(vm, JSC::Identifier::fromString(vm, "Resources"), jsInspectorAuditResourcesObject);
100 }
101}
