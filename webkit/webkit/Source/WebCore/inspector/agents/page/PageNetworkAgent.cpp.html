50Protocol::Network::LoaderId PageNetworkAgent::loaderIdentifier(DocumentLoader* loader)
51{
52 if (loader) {
53 if (auto* pageAgent = m_instrumentingAgents.enabledPageAgent())

59Protocol::Network::FrameId PageNetworkAgent::frameIdentifier(DocumentLoader* loader)
60{

68Vector<WebSocket*> PageNetworkAgent::activeWebSockets()
69{
70 Vector<WebSocket*> webSockets;
71
72 for (auto* webSocket : WebSocket::allActiveWebSockets()) {
73 auto channel = webSocket->channel();
74 if (!channel)
75 continue;
76
77 if (!channel->hasCreatedHandshake())
78 continue;
79
80 if (!is<Document>(webSocket->scriptExecutionContext()))
81 continue;
82
83 // FIXME: <https://webkit.org/b/168475> Web Inspector: Correctly display iframe's WebSockets
84 auto* document = downcast<Document>(webSocket->scriptExecutionContext());
85 if (document->page() != &m_inspectedPage)
86 continue;
87
88 webSockets.append(webSocket);
89 }
90
91 return webSockets;
92}

99ScriptExecutionContext* PageNetworkAgent::scriptExecutionContext(Protocol::ErrorString& errorString, const Protocol::Network::FrameId& frameId)
100{
101 auto* pageAgent = m_instrumentingAgents.enabledPageAgent();
102 if (!pageAgent) {
103 errorString = "Page domain must be enabled"_s;
104 return nullptr;
105 }
106
107 auto* frame = pageAgent->assertFrame(errorString, frameId);
108 if (!frame)
109 return nullptr;
110
111 auto* document = frame->document();
112 if (!document) {
113 errorString = "Missing frame of docuemnt for given frameId"_s;
114 return nullptr;
115 }
116
117 return document;
118}
