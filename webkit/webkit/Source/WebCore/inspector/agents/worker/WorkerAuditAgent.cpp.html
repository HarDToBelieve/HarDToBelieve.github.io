42WorkerAuditAgent::WorkerAuditAgent(WorkerAgentContext& context)
43 : InspectorAuditAgent(context)
44 , m_globalScope(context.globalScope)
45{
46 ASSERT(context.globalScope.isContextThread());
47}
48
49WorkerAuditAgent::~WorkerAuditAgent() = default;
50
51InjectedScript WorkerAuditAgent::injectedScriptForEval(Protocol::ErrorString& errorString, std::optional<Protocol::Runtime::ExecutionContextId>&& executionContextId)
52{
53 if (executionContextId) {
54 errorString = "executionContextId is not supported for workers as there is only one execution context"_s;
55 return InjectedScript();
56 }
57
58 return injectedScriptManager().injectedScriptFor(globalObject(m_globalScope));
59}
