51BoxGeometry& FormattingState::boxGeometry(const Box& layoutBox)
52{
53 // Should never need to mutate a display box outside of the formatting context.
54 ASSERT(&layoutState().establishedFormattingState(layoutBox.formattingContextRoot()) == this);
55 // Anonymous text wrappers do not need to compute box geometry. They initiate inline runs.
56 ASSERT(!layoutBox.isInlineTextBox());
57 return layoutState().ensureGeometryForBox(layoutBox);
58}
