35namespace WebCore {
36namespace Layout {
37
38class Box;
39class FloatingState;
40enum class StyleDiff;
41
42class FormattingState {
43 WTF_MAKE_NONCOPYABLE(FormattingState);
44 WTF_MAKE_ISO_ALLOCATED(FormattingState);
45public:
46 FloatingState& floatingState() const { return m_floatingState; }
47
48 void markNeedsLayout(const Box&, StyleDiff);
49 bool needsLayout(const Box&);
50
51 void setIntrinsicWidthConstraintsForBox(const Box&, IntrinsicWidthConstraints);
52 std::optional<IntrinsicWidthConstraints> intrinsicWidthConstraintsForBox(const Box&) const;
53 void clearIntrinsicWidthConstraints(const Box&);
54
55 void setIntrinsicWidthConstraints(IntrinsicWidthConstraints intrinsicWidthConstraints) { m_intrinsicWidthConstraints = intrinsicWidthConstraints; }
56 std::optional<IntrinsicWidthConstraints> intrinsicWidthConstraints() const { return m_intrinsicWidthConstraints; }
57
58 bool isBlockFormattingState() const { return m_type == Type::Block; }
59 bool isInlineFormattingState() const { return m_type == Type::Inline; }
60 bool isTableFormattingState() const { return m_type == Type::Table; }
61 bool isFlexFormattingState() const { return m_type == Type::Flex; }
62
63 LayoutState& layoutState() const { return m_layoutState; }
64
65 // FIXME: We need to find a way to limit access to mutatable geometry.
66 BoxGeometry& boxGeometry(const Box& layoutBox);
67 // Since we layout the out-of-flow boxes at the end of the formatting context layout, it's okay to store them in the formatting state -as opposed to the containing block level.
68 using OutOfFlowBoxList = Vector<WeakPtr<const Box>>;
69 void addOutOfFlowBox(const Box& outOfFlowBox) { m_outOfFlowBoxes.append(makeWeakPtr(outOfFlowBox)); }
70 void removeOutOfFlowBox(const Box&);
71 const OutOfFlowBoxList& outOfFlowBoxes() const { return m_outOfFlowBoxes; }
72
73protected:
74 enum class Type { Block, Inline, Table, Flex };
75 FormattingState(Ref<FloatingState>&&, Type, LayoutState&);
76 ~FormattingState();
77
78private:
79 LayoutState& m_layoutState;
80 Ref<FloatingState> m_floatingState;
81 HashMap<const Box*, IntrinsicWidthConstraints> m_intrinsicWidthConstraintsForBoxes;
82 std::optional<IntrinsicWidthConstraints> m_intrinsicWidthConstraints;
83 // FIXME: This needs WeakListHashSet
84 OutOfFlowBoxList m_outOfFlowBoxes;
85 Type m_type;
86};
87
88inline void FormattingState::setIntrinsicWidthConstraintsForBox(const Box& layoutBox, IntrinsicWidthConstraints intrinsicWidthConstraints)
89{
90 ASSERT(!m_intrinsicWidthConstraintsForBoxes.contains(&layoutBox));
91 ASSERT(&m_layoutState.formattingStateForBox(layoutBox) == this);
92 m_intrinsicWidthConstraintsForBoxes.set(&layoutBox, intrinsicWidthConstraints);
93}
94
95inline void FormattingState::clearIntrinsicWidthConstraints(const Box& layoutBox)
96{
97 m_intrinsicWidthConstraints = { };
98 m_intrinsicWidthConstraintsForBoxes.remove(&layoutBox);
99}
100
101inline std::optional<IntrinsicWidthConstraints> FormattingState::intrinsicWidthConstraintsForBox(const Box& layoutBox) const
102{
103 ASSERT(&m_layoutState.formattingStateForBox(layoutBox) == this);
104 auto iterator = m_intrinsicWidthConstraintsForBoxes.find(&layoutBox);
105 if (iterator == m_intrinsicWidthConstraintsForBoxes.end())
106 return { };
107 return iterator->value;
108}
109
110}
111}
