62void LayoutContext::layout(const LayoutSize& rootContentBoxSize, InvalidationState& invalidationState)
63{
64 // Set the geometry on the root.
65 // Note that we never layout the root box. It has to have an already computed geometry (in case of ICB, it's the view geometry).
66 // ICB establishes the initial BFC, but it does not live in a formatting context and while a non-ICB root(subtree layout) has to have a formatting context,
67 // we could not lay it out even if we wanted to since it's outside of this LayoutContext.
68 auto& boxGeometry = layoutState().geometryForRootBox();
69 boxGeometry.setHorizontalMargin({ });
70 boxGeometry.setVerticalMargin({ });
71 boxGeometry.setBorder({ });
72 boxGeometry.setPadding({ });
73 boxGeometry.setLogicalTopLeft({ });
74 boxGeometry.setContentBoxHeight(rootContentBoxSize.height());
75 boxGeometry.setContentBoxWidth(rootContentBoxSize.width());
76
77 layoutWithPreparedRootGeometry(invalidationState);
78}
79
80void LayoutContext::layoutWithPreparedRootGeometry(InvalidationState& invalidationState)
81{
82 PhaseScope scope(Phase::Type::Layout);
83
84 auto& formattingContextRootsForLayout = invalidationState.formattingContextRoots();
85 // When invalidation is empty, we assume constraint mutation and start running layout on the context root. Layout logic should be able to figure out the damage.
86 if (formattingContextRootsForLayout.computesEmpty())
87 return layoutFormattingContextSubtree(m_layoutState.root(), invalidationState);
88
89 for (auto& formattingContextRoot : formattingContextRootsForLayout)
90 layoutFormattingContextSubtree(formattingContextRoot, invalidationState);
91}
92
93void LayoutContext::layoutFormattingContextSubtree(const ContainerBox& formattingContextRoot, InvalidationState& invalidationState)
94{
95 RELEASE_ASSERT(formattingContextRoot.establishesFormattingContext());
96 if (!formattingContextRoot.hasChild())
97 return;
98
99 auto formattingContext = createFormattingContext(formattingContextRoot, layoutState());
100 auto& boxGeometry = layoutState().geometryForBox(formattingContextRoot);
101
102 if (formattingContextRoot.hasInFlowOrFloatingChild()) {
103 auto constraintsForInFlowContent = ConstraintsForInFlowContent { { boxGeometry.contentBoxLeft(), boxGeometry.contentBoxWidth() }, boxGeometry.contentBoxTop() };
104 formattingContext->layoutInFlowContent(invalidationState, constraintsForInFlowContent);
105 }
106
107 // FIXME: layoutFormattingContextSubtree() does not perform layout on the root, rather it lays out the root's content.
108 // It constructs an FC for descendant boxes and runs layout on them. The formattingContextRoot is laid out in the FC in which it lives (parent formatting context).
109 // It also means that the formattingContextRoot has to have a valid/clean geometry at this point.
110 {
111 auto constraints = ConstraintsForOutOfFlowContent { { boxGeometry.paddingBoxLeft(), boxGeometry.paddingBoxWidth() },
112 { boxGeometry.paddingBoxTop(), boxGeometry.paddingBoxHeight() }, boxGeometry.contentBoxWidth() };
113 formattingContext->layoutOutOfFlowContent(invalidationState, constraints);
114 }
115}

117std::unique_ptr<FormattingContext> LayoutContext::createFormattingContext(const ContainerBox& formattingContextRoot, LayoutState& layoutState)
118{
119 ASSERT(formattingContextRoot.establishesFormattingContext());
120 if (formattingContextRoot.establishesInlineFormattingContext()) {
121 auto& inlineFormattingState = layoutState.ensureInlineFormattingState(formattingContextRoot);
122 return makeUnique<InlineFormattingContext>(formattingContextRoot, inlineFormattingState);
123 }
124
125 if (formattingContextRoot.establishesBlockFormattingContext()) {
126 ASSERT(!formattingContextRoot.establishesInlineFormattingContext());
127 auto& blockFormattingState = layoutState.ensureBlockFormattingState(formattingContextRoot);
128 if (formattingContextRoot.isTableWrapperBox())
129 return makeUnique<TableWrapperBlockFormattingContext>(formattingContextRoot, blockFormattingState);
130 return makeUnique<BlockFormattingContext>(formattingContextRoot, blockFormattingState);
131 }
132
133 if (formattingContextRoot.establishesFlexFormattingContext()) {
134 auto& flexFormattingState = layoutState.ensureFlexFormattingState(formattingContextRoot);
135 return makeUnique<FlexFormattingContext>(formattingContextRoot, flexFormattingState);
136 }
137
138 if (formattingContextRoot.establishesTableFormattingContext()) {
139 auto& tableFormattingState = layoutState.ensureTableFormattingState(formattingContextRoot);
140 return makeUnique<TableFormattingContext>(formattingContextRoot, tableFormattingState);
141 }
142
143 CRASH();
144}
