48LayoutState::LayoutState(const Document& document, const ContainerBox& rootContainer)
49 : m_rootContainer(makeWeakPtr(rootContainer))
50{
51 // It makes absolutely no sense to construct a dedicated layout state for a non-formatting context root (layout would be a no-op).
52 ASSERT(root().establishesFormattingContext());
53
54 auto quirksMode = [&] {
55 if (document.inLimitedQuirksMode())
56 return LayoutState::QuirksMode::Limited;
57 if (document.inQuirksMode())
58 return LayoutState::QuirksMode::Yes;
59 return LayoutState::QuirksMode::No;
60 };
61 setQuirksMode(quirksMode());
62}

71BoxGeometry& LayoutState::ensureGeometryForBoxSlow(const Box& layoutBox)
72{
73 if (layoutBox.canCacheForLayoutState(*this)) {
74 ASSERT(!layoutBox.cachedGeometryForLayoutState(*this));
75 auto newBox = makeUnique<BoxGeometry>();
76 auto& newBoxPtr = *newBox;
77 layoutBox.setCachedGeometryForLayoutState(*this, WTFMove(newBox));
78 return newBoxPtr;
79 }
80
81 return *m_layoutBoxToBoxGeometry.ensure(&layoutBox, [] {
82 return makeUnique<BoxGeometry>();
83 }).iterator->value;
84}

91bool LayoutState::hasFormattingState(const ContainerBox& formattingContextRoot) const
92{
93 ASSERT(formattingContextRoot.establishesFormattingContext());
94 return m_blockFormattingStates.contains(&formattingContextRoot)
95 || m_inlineFormattingStates.contains(&formattingContextRoot)
96 || m_tableFormattingStates.contains(&formattingContextRoot)
97 || m_flexFormattingStates.contains(&formattingContextRoot);
98}

100FormattingState& LayoutState::establishedFormattingState(const ContainerBox& formattingContextRoot) const
101{
102 if (RuntimeEnabledFeatures::sharedFeatures().layoutFormattingContextIntegrationEnabled()) {
103 ASSERT(&formattingContextRoot == m_rootContainer.get());
104 return *m_rootInlineFormattingStateForIntegration;
105 }
106
107 if (auto* formattingState = m_inlineFormattingStates.get(&formattingContextRoot))
108 return *formattingState;
109
110 if (auto* formattingState = m_blockFormattingStates.get(&formattingContextRoot))
111 return *formattingState;
112
113 if (auto* formattingState = m_tableFormattingStates.get(&formattingContextRoot))
114 return *formattingState;
115
116 if (auto* formattingState = m_flexFormattingStates.get(&formattingContextRoot))
117 return *formattingState;
118
119 CRASH();
120}

152FormattingState& LayoutState::ensureFormattingState(const ContainerBox& formattingContextRoot)
153{
154 if (formattingContextRoot.establishesInlineFormattingContext())
155 return ensureInlineFormattingState(formattingContextRoot);
156
157 if (formattingContextRoot.establishesBlockFormattingContext())
158 return ensureBlockFormattingState(formattingContextRoot);
159
160 if (formattingContextRoot.establishesTableFormattingContext())
161 return ensureTableFormattingState(formattingContextRoot);
162
163 if (formattingContextRoot.establishesFlexFormattingContext())
164 return ensureFlexFormattingState(formattingContextRoot);
165
166 CRASH();
167}

169InlineFormattingState& LayoutState::ensureInlineFormattingState(const ContainerBox& formattingContextRoot)
170{
171 ASSERT(formattingContextRoot.establishesInlineFormattingContext());
172
173 auto create = [&] {
174 // If the block container box that initiates this inline formatting context also establishes a block context, the floats outside of the formatting root
175 // should not interfere with the content inside.
176 // <div style="float: left"></div><div style="overflow: hidden"> <- is a non-intrusive float, because overflow: hidden triggers new block formatting context.</div>
177 if (formattingContextRoot.establishesBlockFormattingContext())
178 return makeUnique<InlineFormattingState>(FloatingState::create(*this, formattingContextRoot), *this);
179
180 // Otherwise, the formatting context inherits the floats from the parent formatting context.
181 // Find the formatting state in which this formatting root lives, not the one it creates and use its floating state.
182 auto& parentFormattingState = ensureFormattingState(formattingContextRoot.formattingContextRoot());
183 auto& parentFloatingState = parentFormattingState.floatingState();
184 return makeUnique<InlineFormattingState>(parentFloatingState, *this);
185 };
186
187 if (RuntimeEnabledFeatures::sharedFeatures().layoutFormattingContextIntegrationEnabled()) {
188 if (!m_rootInlineFormattingStateForIntegration) {
189 ASSERT(&formattingContextRoot == m_rootContainer.get());
190 m_rootInlineFormattingStateForIntegration = create();
191 }
192 return *m_rootInlineFormattingStateForIntegration;
193 }
194
195 return *m_inlineFormattingStates.ensure(&formattingContextRoot, create).iterator->value;
196}

198BlockFormattingState& LayoutState::ensureBlockFormattingState(const ContainerBox& formattingContextRoot)
199{
200 ASSERT(formattingContextRoot.establishesBlockFormattingContext());
201
202 auto create = [&] {
203 return makeUnique<BlockFormattingState>(FloatingState::create(*this, formattingContextRoot), *this);
204 };
205
206 return *m_blockFormattingStates.ensure(&formattingContextRoot, create).iterator->value;
207}

209TableFormattingState& LayoutState::ensureTableFormattingState(const ContainerBox& formattingContextRoot)
210{
211 ASSERT(formattingContextRoot.establishesTableFormattingContext());
212
213 auto create = [&] {
214 // Table formatting context always establishes a new floating state -and it stays empty.
215 return makeUnique<TableFormattingState>(FloatingState::create(*this, formattingContextRoot), *this, formattingContextRoot);
216 };
217
218 return *m_tableFormattingStates.ensure(&formattingContextRoot, create).iterator->value;
219}
