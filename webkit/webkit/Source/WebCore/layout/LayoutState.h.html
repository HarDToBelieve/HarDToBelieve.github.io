124inline bool LayoutState::hasBoxGeometry(const Box& layoutBox) const
125{
126 if (layoutBox.cachedGeometryForLayoutState(*this))
127 return true;
128 return m_layoutBoxToBoxGeometry.contains(&layoutBox);
129}
130
131inline BoxGeometry& LayoutState::ensureGeometryForBox(const Box& layoutBox)
132{
133 if (auto* boxGeometry = layoutBox.cachedGeometryForLayoutState(*this))
134 return *boxGeometry;
135 return ensureGeometryForBoxSlow(layoutBox);
136}
137
138inline const BoxGeometry& LayoutState::geometryForBox(const Box& layoutBox) const
139{
140 if (auto* boxGeometry = layoutBox.cachedGeometryForLayoutState(*this))
141 return *boxGeometry;
142 ASSERT(m_layoutBoxToBoxGeometry.contains(&layoutBox));
143 return *m_layoutBoxToBoxGeometry.get(&layoutBox);
144}

146#ifndef NDEBUG
147inline void LayoutState::registerFormattingContext(const FormattingContext& formattingContext)
148{
149 // Multiple formatting contexts of the same root within a layout frame indicates defective layout logic.
150 ASSERT(!m_formattingContextList.contains(&formattingContext));
151 m_formattingContextList.add(&formattingContext);
152}
153#endif

155// These Layout::Box function are here to allow inlining.
156inline bool Box::canCacheForLayoutState(const LayoutState& layoutState) const
157{
158 return !m_cachedLayoutState || m_cachedLayoutState.get() == &layoutState;
159}
160
161inline BoxGeometry* Box::cachedGeometryForLayoutState(const LayoutState& layoutState) const
162{
163 if (m_cachedLayoutState.get() != &layoutState)
164 return nullptr;
165 return m_cachedGeometryForLayoutState.get();
166}
