55struct Position {
56 operator LayoutUnit() const { return value; }
57 LayoutUnit value;
58};
59
60inline bool operator<(const Position& a, const Position& b)
61{
62 return a.value < b.value;
63}
64
65inline bool operator==(const Position& a, const Position& b)
66{
67 return a.value == b.value;
68}

70struct Point {
71 // FIXME: Use Position<Horizontal>, Position<Vertical> to avoid top/left vs. x/y confusion.
72 LayoutUnit x; // left
73 LayoutUnit y; // top
74
75 Point() = default;
76 Point(LayoutUnit, LayoutUnit);
77 Point(LayoutPoint);
78 static Point max() { return { LayoutUnit::max(), LayoutUnit::max() }; }
79
80 void move(LayoutSize);
81 void moveBy(LayoutPoint);
82 operator LayoutPoint() const { return { x, y }; }
83};
84
85// FIXME: Wrap these into structs.
86using PointInContextRoot = Point;
87using PositionInContextRoot = Position;
88
89inline Point::Point(LayoutPoint point)
90 : x(point.x())
91 , y(point.y())
92{
93}
94
95inline Point::Point(LayoutUnit x, LayoutUnit y)
96 : x(x)
97 , y(y)
98{
99}
100
101inline void Point::move(LayoutSize offset)
102{
103 x += offset.width();
104 y += offset.height();
105}
106
107inline void Point::moveBy(LayoutPoint offset)
108{
109 x += offset.x();
110 y += offset.y();
111}

113// Margin, border, padding
114struct HorizontalEdges {
115 LayoutUnit left;
116 LayoutUnit right;
117};
118
119struct VerticalEdges {
120 LayoutUnit top;
121 LayoutUnit bottom;
122};
123
124struct Edges {
125 HorizontalEdges horizontal;
126 VerticalEdges vertical;
127
128 LayoutUnit width() const { return horizontal.left + horizontal.right; }
129 LayoutUnit height() const { return vertical.top + vertical.bottom; }
130};
131
132inline Edges operator/(const Edges& edge, size_t value)
133{
134 return { { edge.horizontal.left / value, edge.horizontal.right / value }, { edge.vertical.top / value, edge.vertical.bottom / value } };
135}

198struct SlotPosition {
199 SlotPosition() = default;
200 SlotPosition(size_t column, size_t row);
201
202 size_t column { 0 };
203 size_t row { 0 };
204};
205
206inline SlotPosition::SlotPosition(size_t column, size_t row)
207 : column(column)
208 , row(row)
209{
210}
211
212inline bool operator==(const SlotPosition& a, const SlotPosition& b)
213{
214 return a.column == b.column && a.row == b.row;
215}

225namespace WTF {
226struct SlotPositionHash {
227 static unsigned hash(const WebCore::Layout::SlotPosition& slotPosition) { return pairIntHash(slotPosition.column, slotPosition.row); }
228 static bool equal(const WebCore::Layout::SlotPosition& a, const WebCore::Layout::SlotPosition& b) { return a == b; }
229 static const bool safeToCompareToEmptyOrDeleted = true;
230};
231template<> struct HashTraits<WebCore::Layout::SlotPosition> : GenericHashTraits<WebCore::Layout::SlotPosition> {
232 static WebCore::Layout::SlotPosition emptyValue() { return WebCore::Layout::SlotPosition(0, std::numeric_limits<size_t>::max()); }
233
234 static void constructDeletedValue(WebCore::Layout::SlotPosition& slot) { slot.column = std::numeric_limits<size_t>::max(); }
235 static bool isDeletedValue(const WebCore::Layout::SlotPosition& slot) { return slot.column == std::numeric_limits<size_t>::max(); }
236};
237template<> struct DefaultHash<WebCore::Layout::SlotPosition> : SlotPositionHash { };
