36struct ComputedVerticalMargin {
37 std::optional<LayoutUnit> before;
38 std::optional<LayoutUnit> after;

41struct UsedVerticalMargin {
42 struct NonCollapsedValues {
43 LayoutUnit before;
44 LayoutUnit after;
45 };
46 NonCollapsedValues nonCollapsedValues;
47
48 struct CollapsedValues {
49 std::optional<LayoutUnit> before;
50 std::optional<LayoutUnit> after;
51 bool isCollapsedThrough { false };
52 };
53 CollapsedValues collapsedValues;
54
55 // FIXME: This structure might need to change to indicate that the cached value is not necessarily the same as the box's computed margin value.
56 // This only matters in case of collapse through margins when they collapse into another sibling box.
57 // <div style="margin: 1px"></div><div style="margin: 10px"></div> <- the second div's before/after marings collapse through and the same time they collapse into
58 // the first div. When the parent computes its before margin, it should see the second div's collapsed through margin as the value to collapse width (adjoining margin value).
59 // So while the first div's before margin is not 10px, the cached value is 10px so that when we compute the parent's margin we just need to check the first
60 // inflow child's cached margin values.
61 struct PositiveAndNegativePair {
62 struct Values {
63 bool isNonZero() const { return positive.value_or(0) || negative.value_or(0); }
64
65 std::optional<LayoutUnit> positive;
66 std::optional<LayoutUnit> negative;
67 bool isQuirk { false };
68 };
69 Values before;
70 Values after;
71 };
72 PositiveAndNegativePair positiveAndNegativeValues;
73};
74
75static inline LayoutUnit marginBefore(const UsedVerticalMargin& usedVerticalMargin)
76{
77 return usedVerticalMargin.collapsedValues.before.value_or(usedVerticalMargin.nonCollapsedValues.before);
78}
79
80static inline LayoutUnit marginAfter(const UsedVerticalMargin& usedVerticalMargin)
81{
82 if (usedVerticalMargin.collapsedValues.isCollapsedThrough)
83 return 0_lu;
84 return usedVerticalMargin.collapsedValues.after.value_or(usedVerticalMargin.nonCollapsedValues.after);
85}

97struct PrecomputedMarginBefore {
98 LayoutUnit usedValue() const { return collapsedValue.value_or(nonCollapsedValue); }
99 LayoutUnit nonCollapsedValue;
100 std::optional<LayoutUnit> collapsedValue;

102};
