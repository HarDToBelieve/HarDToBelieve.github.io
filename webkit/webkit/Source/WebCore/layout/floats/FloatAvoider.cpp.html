40// Floating boxes intersect their margin box with the other floats in the context,
41// while other float avoiders (e.g. non-floating formatting context roots) intersect their border box.
42FloatAvoider::FloatAvoider(LayoutPoint absoluteTopLeft, LayoutUnit borderBoxWidth, const Edges& margin, HorizontalEdges containingBlockAbsoluteContentBox, bool isFloatingPositioned, bool isLeftAligned)
43 : m_absoluteTopLeft(absoluteTopLeft)
44 , m_borderBoxWidth(borderBoxWidth)
45 , m_margin(margin)
46 , m_containingBlockAbsoluteContentBox(containingBlockAbsoluteContentBox)
47 , m_isFloatingPositioned(isFloatingPositioned)
48 , m_isLeftAligned(isLeftAligned)
49{
50 m_absoluteTopLeft.setX(initialHorizontalPosition());
51}
52
53void FloatAvoider::setHorizontalPosition(LayoutUnit horizontalPosition)
54{
55 if (isLeftAligned() && isFloatingBox())
56 horizontalPosition += marginStart();
57 if (!isLeftAligned()) {
58 horizontalPosition -= borderBoxWidth();
59 if (isFloatingBox())
60 horizontalPosition -= marginEnd();
61 }
62
63 auto constrainedByContainingBlock = [&] {
64 // Horizontal position is constrained by the containing block's content box.
65 // Compute the horizontal position for the new floating by taking both the contining block and the current left/right floats into account.
66 if (isLeftAligned())
67 return std::max(m_containingBlockAbsoluteContentBox.left + marginStart(), horizontalPosition);
68 // Make sure it does not overflow the containing block on the right.
69 return std::min(horizontalPosition, m_containingBlockAbsoluteContentBox.right - marginBoxWidth() + marginStart());
70 }();
71 m_absoluteTopLeft.setX(constrainedByContainingBlock);
72}
73
74void FloatAvoider::setVerticalPosition(LayoutUnit verticalPosition)
75{
76 if (isFloatingBox())
77 verticalPosition += marginBefore();
78 m_absoluteTopLeft.setY(verticalPosition);
79}
80
81LayoutUnit FloatAvoider::initialHorizontalPosition() const
82{
83 if (isLeftAligned())
84 return { m_containingBlockAbsoluteContentBox.left + marginStart() };
85 return { m_containingBlockAbsoluteContentBox.right - marginEnd() - borderBoxWidth() };
86}
87
88bool FloatAvoider::overflowsContainingBlock() const
89{
90 auto left = m_absoluteTopLeft.x() - marginStart();
91 if (m_containingBlockAbsoluteContentBox.left > left)
92 return true;
93
94 auto right = left + marginBoxWidth();
95 return m_containingBlockAbsoluteContentBox.right < right;
96}
