36namespace WebCore {
37
38namespace Layout {
39
40class Box;
41class BoxGeometry;
42class FloatingContext;
43class LayoutState;
44class Rect;
45
46// FloatingState holds the floating boxes per formatting context.
47class FloatingState : public RefCounted<FloatingState> {
48 WTF_MAKE_ISO_ALLOCATED(FloatingState);
49public:
50 static Ref<FloatingState> create(LayoutState& layoutState, const ContainerBox& formattingContextRoot) { return adoptRef(*new FloatingState(layoutState, formattingContextRoot)); }
51
52 const ContainerBox& root() const { return *m_formattingContextRoot; }
53
54 class FloatItem {
55 public:
56 FloatItem(const Box&, BoxGeometry absoluteBoxGeometry);
57
58 // FIXME: This c'tor is only used by the render tree integation codepath.
59 enum class Position { Left, Right };
60 FloatItem(Position, BoxGeometry absoluteBoxGeometry);
61
62 bool isLeftPositioned() const { return m_position == Position::Left; }
63 bool isInFormattingContextOf(const ContainerBox& formattingContextRoot) const { return m_layoutBox->isInFormattingContextOf(formattingContextRoot); }
64
65 Rect rectWithMargin() const { return BoxGeometry::marginBoxRect(m_absoluteBoxGeometry); }
66 BoxGeometry::HorizontalMargin horizontalMargin() const { return m_absoluteBoxGeometry.horizontalMargin(); }
67 PositionInContextRoot bottom() const { return { BoxGeometry::borderBoxRect(m_absoluteBoxGeometry).bottom() }; }
68
69#if ASSERT_ENABLED
70 const Box* floatBox() const { return m_layoutBox.get(); }
71#endif
72 private:
73 WeakPtr<const Box> m_layoutBox;
74 Position m_position;
75 BoxGeometry m_absoluteBoxGeometry;
76 };
77 using FloatList = Vector<FloatItem>;
78 const FloatList& floats() const { return m_floats; }
79 const FloatItem* last() const { return floats().isEmpty() ? nullptr : &m_floats.last(); }
80
81 void append(FloatItem);
82 void clear() { m_floats.clear(); }
83
84private:
85 friend class FloatingContext;
86 FloatingState(LayoutState&, const ContainerBox& formattingContextRoot);
87 LayoutState& layoutState() const { return m_layoutState; }
88
89 LayoutState& m_layoutState;
90 WeakPtr<const ContainerBox> m_formattingContextRoot;
91 FloatList m_floats;
92};

94}
