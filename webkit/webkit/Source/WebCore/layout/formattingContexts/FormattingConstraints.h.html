48struct ConstraintsForInFlowContent {
49 ConstraintsForInFlowContent(HorizontalConstraints, LayoutUnit logicalTop);
50
51 HorizontalConstraints horizontal() const { return m_horizontal; }
52 LayoutUnit logicalTop() const { return m_logicalTop; }
53
54 enum BaseTypeFlag : uint8_t {
55 GenericContent = 1 << 0,
56 TableContent = 1 << 1
57 };
58 bool isConstraintsForTableContent() const { return baseTypeFlags().contains(TableContent); }
59
60protected:
61 ConstraintsForInFlowContent(HorizontalConstraints, LayoutUnit logicalTop, OptionSet<BaseTypeFlag>);
62
63private:
64 OptionSet<BaseTypeFlag> baseTypeFlags() const { return OptionSet<BaseTypeFlag>::fromRaw(m_baseTypeFlags); }
65
66 unsigned m_baseTypeFlags : 2; // OptionSet<BaseTypeFlag>
67 HorizontalConstraints m_horizontal;
68 LayoutUnit m_logicalTop;
69};

90struct IntrinsicWidthConstraints {
91 void expand(LayoutUnit horizontalValue);
92 IntrinsicWidthConstraints& operator+=(const IntrinsicWidthConstraints&);
93 IntrinsicWidthConstraints& operator+=(LayoutUnit);
94 IntrinsicWidthConstraints& operator-=(const IntrinsicWidthConstraints&);
95 IntrinsicWidthConstraints& operator-=(LayoutUnit);
96
97 LayoutUnit minimum;
98 LayoutUnit maximum;
99};
100
101inline void IntrinsicWidthConstraints::expand(LayoutUnit horizontalValue)
102{
103 minimum += horizontalValue;
104 maximum += horizontalValue;
105}
106
107inline IntrinsicWidthConstraints& IntrinsicWidthConstraints::operator+=(const IntrinsicWidthConstraints& other)
108{
109 minimum += other.minimum;
110 maximum += other.maximum;
111 return *this;
112}
113
114inline IntrinsicWidthConstraints& IntrinsicWidthConstraints::operator+=(LayoutUnit value)
115{
116 expand(value);
117 return *this;
118}
119
120inline IntrinsicWidthConstraints& IntrinsicWidthConstraints::operator-=(const IntrinsicWidthConstraints& other)
121{
122 minimum -= other.minimum;
123 maximum -= other.maximum;
124 return *this;
125}
126
127inline IntrinsicWidthConstraints& IntrinsicWidthConstraints::operator-=(LayoutUnit value)
128{
129 expand(-value);
130 return *this;
131}
