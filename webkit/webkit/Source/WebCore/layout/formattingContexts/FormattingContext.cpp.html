74void FormattingContext::computeOutOfFlowHorizontalGeometry(const Box& layoutBox, const ConstraintsForOutOfFlowContent& constraints)
75{
76 ASSERT(layoutBox.isOutOfFlowPositioned());
77 auto compute = [&](std::optional<LayoutUnit> usedWidth) {
78 return formattingGeometry().outOfFlowHorizontalGeometry(layoutBox, constraints.horizontal, constraints.vertical, { usedWidth, { } });
79 };
80
81 auto containingBlockWidth = constraints.horizontal.logicalWidth;
82 auto horizontalGeometry = compute({ });
83 if (auto maxWidth = formattingGeometry().computedMaxWidth(layoutBox, containingBlockWidth)) {
84 auto maxHorizontalGeometry = compute(maxWidth);
85 if (horizontalGeometry.contentWidthAndMargin.contentWidth > maxHorizontalGeometry.contentWidthAndMargin.contentWidth)
86 horizontalGeometry = maxHorizontalGeometry;
87 }
88
89 if (auto minWidth = formattingGeometry().computedMinWidth(layoutBox, containingBlockWidth)) {
90 auto minHorizontalGeometry = compute(minWidth);
91 if (horizontalGeometry.contentWidthAndMargin.contentWidth < minHorizontalGeometry.contentWidthAndMargin.contentWidth)
92 horizontalGeometry = minHorizontalGeometry;
93 }
94
95 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
96 boxGeometry.setLogicalLeft(horizontalGeometry.left + horizontalGeometry.contentWidthAndMargin.usedMargin.start);
97 boxGeometry.setContentBoxWidth(horizontalGeometry.contentWidthAndMargin.contentWidth);
98 auto& usedHorizontalMargin = horizontalGeometry.contentWidthAndMargin.usedMargin;
99 boxGeometry.setHorizontalMargin({ usedHorizontalMargin.start, usedHorizontalMargin.end });
100}

102void FormattingContext::computeOutOfFlowVerticalGeometry(const Box& layoutBox, const ConstraintsForOutOfFlowContent& constraints)
103{
104 ASSERT(layoutBox.isOutOfFlowPositioned());
105 auto compute = [&](std::optional<LayoutUnit> usedHeight) {
106 return formattingGeometry().outOfFlowVerticalGeometry(layoutBox, constraints.horizontal, constraints.vertical, { usedHeight });
107 };
108
109 auto containingBlockHeight = constraints.vertical.logicalHeight;
110 auto verticalGeometry = compute({ });
111 if (auto maxHeight = formattingGeometry().computedMaxHeight(layoutBox, containingBlockHeight)) {
112 auto maxVerticalGeometry = compute(maxHeight);
113 if (verticalGeometry.contentHeightAndMargin.contentHeight > maxVerticalGeometry.contentHeightAndMargin.contentHeight)
114 verticalGeometry = maxVerticalGeometry;
115 }
116
117 if (auto minHeight = formattingGeometry().computedMinHeight(layoutBox, containingBlockHeight)) {
118 auto minVerticalGeometry = compute(minHeight);
119 if (verticalGeometry.contentHeightAndMargin.contentHeight < minVerticalGeometry.contentHeightAndMargin.contentHeight)
120 verticalGeometry = minVerticalGeometry;
121 }
122
123 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
124 auto nonCollapsedVerticalMargin = verticalGeometry.contentHeightAndMargin.nonCollapsedMargin;
125 boxGeometry.setLogicalTop(verticalGeometry.top + nonCollapsedVerticalMargin.before);
126 boxGeometry.setContentBoxHeight(verticalGeometry.contentHeightAndMargin.contentHeight);
127 // Margins of absolutely positioned boxes do not collapse.
128 boxGeometry.setVerticalMargin({ nonCollapsedVerticalMargin.before, nonCollapsedVerticalMargin.after });
129}

135 boxGeometry.setPadding(formattingGeometry().computedPadding(layoutBox, horizontalConstraint.logicalWidth));

138void FormattingContext::layoutOutOfFlowContent(InvalidationState& invalidationState, const ConstraintsForOutOfFlowContent& constraints)
139{
140 LOG_WITH_STREAM(FormattingContextLayout, stream << "Start: layout out-of-flow content -> context: " << &layoutState() << " root: " << &root());
141
142 collectOutOfFlowDescendantsIfNeeded();
143
144 auto constraintsForLayoutBox = [&] (const auto& outOfFlowBox) {
145 auto& containingBlock = outOfFlowBox.containingBlock();
146 return &containingBlock == &root() ? constraints : formattingGeometry().constraintsForOutOfFlowContent(containingBlock);
147 };
148
149 for (auto& outOfFlowBox : formattingState().outOfFlowBoxes()) {
150 ASSERT(outOfFlowBox->establishesFormattingContext());
151 if (!invalidationState.needsLayout(*outOfFlowBox))
152 continue;
153
154 auto containingBlockConstraints = constraintsForLayoutBox(*outOfFlowBox);
155 auto horizontalConstraintsForBorderAndPadding = HorizontalConstraints { containingBlockConstraints.horizontal.logicalLeft, containingBlockConstraints.borderAndPaddingConstraints };
156 computeBorderAndPadding(*outOfFlowBox, horizontalConstraintsForBorderAndPadding);
157
158 computeOutOfFlowHorizontalGeometry(*outOfFlowBox, containingBlockConstraints);
159 auto outOfFlowBoxHasContent = is<ContainerBox>(*outOfFlowBox) && downcast<ContainerBox>(*outOfFlowBox).hasChild();
160 if (outOfFlowBoxHasContent) {
161 auto& containerBox = downcast<ContainerBox>(*outOfFlowBox);
162 auto formattingContext = LayoutContext::createFormattingContext(containerBox, layoutState());
163 if (containerBox.hasInFlowOrFloatingChild())
164 formattingContext->layoutInFlowContent(invalidationState, formattingGeometry().constraintsForInFlowContent(containerBox));
165 computeOutOfFlowVerticalGeometry(containerBox, containingBlockConstraints);
166 formattingContext->layoutOutOfFlowContent(invalidationState, formattingGeometry().constraintsForOutOfFlowContent(containerBox));
167 } else
168 computeOutOfFlowVerticalGeometry(*outOfFlowBox, containingBlockConstraints);
169 }
170 LOG_WITH_STREAM(FormattingContextLayout, stream << "End: layout out-of-flow content -> context: " << &layoutState() << " root: " << &root());
171}

173const BoxGeometry& FormattingContext::geometryForBox(const Box& layoutBox, std::optional<EscapeReason> escapeReason) const
174{
175 UNUSED_PARAM(escapeReason);
176#if ASSERT_ENABLED
177 auto isOkToAccessBoxGeometry = [&] {
178 if (!is<InitialContainingBlock>(layoutBox) && &layoutBox.formattingContextRoot() == &root()) {
179 // This is the non-escape case of accessing a box's geometry information within the same formatting context.
180 return true;
181 }
182
183 if (!escapeReason) {
184 // Any geometry access outside of the formatting context without a valid reason is considered an escape.
185 return false;
186 }
187
188 if (*escapeReason == EscapeReason::DocumentBoxStretchesToViewportQuirk) {
189 ASSERT(layoutState().inQuirksMode());
190 return is<InitialContainingBlock>(layoutBox);
191 }
192
193 if (*escapeReason == EscapeReason::BodyStretchesToViewportQuirk) {
194 ASSERT(layoutState().inQuirksMode());
195 return is<InitialContainingBlock>(layoutBox) || layoutBox.isDocumentBox();
196
197 }
198
199 if (*escapeReason == EscapeReason::TableQuirkNeedsGeometryFromEstablishedFormattingContext) {
200 // This is the case when the table formatting root collects geometry information from the cell's
201 // formatting context to be able to determine width/height. see shouldIgnoreChildContentVerticalMargin
202 ASSERT(root().establishesTableFormattingContext());
203 return &layoutBox.formattingContextRoot().formattingContextRoot() == &root();
204 }
205
206 if (*escapeReason == EscapeReason::OutOfFlowBoxNeedsInFlowGeometry) {
207 // When computing the static position of an out-of-flow box, we need to gather sibling/parent geometry information
208 // as if the out-of-flow box was a simple inflow box.
209 // Now since the out-of-flow and the sibling/parent boxes could very well be in different containing block subtrees
210 // the formatting context they live in could also be very different.
211 return true;
212 }
213
214 if (*escapeReason == EscapeReason::FloatBoxIsAlwaysRelativeToFloatStateRoot) {
215 // Float box top/left values are mapped relative to the FloatState's root. Inline formatting contexts(A) inherit floats from parent
216 // block formatting contexts(B). Floats in these inline formatting contexts(A) need to be mapped to the parent, block formatting context(B).
217 auto& formattingContextRoot = layoutBox.formattingContextRoot();
218 return &formattingContextRoot == &root() || &formattingContextRoot == &root().formattingContextRoot();
219 }
220
221 if (*escapeReason == EscapeReason::FindFixedHeightAncestorQuirk) {
222 ASSERT(layoutState().inQuirksMode());
223 // Find the first containing block with fixed height quirk. See Quirks::heightValueOfNearestContainingBlockWithFixedHeight.
224 // This is only to check if the targetFormattingRoot is an ancestor formatting root.
225 if (is<InitialContainingBlock>(layoutBox))
226 return true;
227 auto& targetFormattingRoot = layoutBox.formattingContextRoot();
228 auto* ancestorFormattingContextRoot = &root().formattingContextRoot();
229 while (true) {
230 if (&targetFormattingRoot == ancestorFormattingContextRoot)
231 return true;
232 ancestorFormattingContextRoot = &ancestorFormattingContextRoot->formattingContextRoot();
233 if (is<InitialContainingBlock>(*ancestorFormattingContextRoot))
234 return true;
235 }
236 return false;
237 }
238
239 if (*escapeReason == EscapeReason::TableNeedsAccessToTableWrapper) {
240 // Tables are wrapped in a 2 level formatting context structure. A <table> element initiates a block formatting context for its principal table box
241 // where the caption and the table content live. It also initiates a table wrapper box which establishes the table formatting context.
242 // In many cases the TFC needs access to the parent (generated) BFC.
243 return &layoutBox == &root().formattingContextRoot();
244 }
245
246 ASSERT_NOT_REACHED();
247 return false;
248 };
249#endif
250 ASSERT(isOkToAccessBoxGeometry());
251 ASSERT(layoutState().hasBoxGeometry(layoutBox));
252 return layoutState().geometryForBox(layoutBox);
253}

255void FormattingContext::collectOutOfFlowDescendantsIfNeeded()
256{
257 if (!formattingState().outOfFlowBoxes().isEmpty())
258 return;
259 auto& root = this->root();
260 if (!root.hasChild())
261 return;
262 if (!root.isPositioned() && !is<InitialContainingBlock>(root))
263 return;
264 // Collect the out-of-flow descendants at the formatting root level (as opposed to at the containing block level, though they might be the same).
265 // FIXME: Turn this into a register-self as boxes are being inserted.
266 for (auto& descendant : descendantsOfType<Box>(root)) {
267 if (!descendant.isOutOfFlowPositioned())
268 continue;
269 if (&descendant.formattingContextRoot() != &root)
270 continue;
271 formattingState().addOutOfFlowBox(descendant);
272 }
273}

275#ifndef NDEBUG
276void FormattingContext::validateGeometryConstraintsAfterLayout() const
277{
278 auto& formattingContextRoot = root();
279 // FIXME: add a descendantsOfType<> flavor that stops at nested formatting contexts
280 for (auto& layoutBox : descendantsOfType<Box>(formattingContextRoot)) {
281 if (&layoutBox.formattingContextRoot() != &formattingContextRoot)
282 continue;
283 auto& containingBlockGeometry = geometryForBox(layoutBox.containingBlock());
284 auto& boxGeometry = geometryForBox(layoutBox);
285
286 // 10.3.3 Block-level, non-replaced elements in normal flow
287 // 10.3.7 Absolutely positioned, non-replaced elements
288 if ((layoutBox.isBlockLevelBox() || layoutBox.isOutOfFlowPositioned()) && !layoutBox.isReplacedBox()) {
289 // margin-left + border-left-width + padding-left + width + padding-right + border-right-width + margin-right = width of containing block
290 auto containingBlockWidth = containingBlockGeometry.contentBoxWidth();
291 ASSERT(boxGeometry.horizontalMarginBorderAndPadding() + boxGeometry.contentBoxWidth() == containingBlockWidth);
292 }
293
294 // 10.6.4 Absolutely positioned, non-replaced elements
295 if (layoutBox.isOutOfFlowPositioned() && !layoutBox.isReplacedBox()) {
296 // top + margin-top + border-top-width + padding-top + height + padding-bottom + border-bottom-width + margin-bottom + bottom = height of containing block
297 auto containingBlockHeight = containingBlockGeometry.contentBoxHeight();
298 ASSERT(BoxGeometry::borderBoxTop(boxGeometry) + boxGeometry.marginBefore() + boxGeometry.borderTop() + boxGeometry.paddingTop().value_or(0) + boxGeometry.contentBoxHeight()
299 + boxGeometry.paddingBottom().value_or(0) + boxGeometry.borderBottom() + boxGeometry.marginAfter() == containingBlockHeight);
300 }
301 }
302}
