56class FormattingContext {
57 WTF_MAKE_ISO_ALLOCATED(FormattingContext);
58public:
59 virtual ~FormattingContext();
60
61 virtual void layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent&) = 0;
62 void layoutOutOfFlowContent(InvalidationState&, const ConstraintsForOutOfFlowContent&);
63 virtual IntrinsicWidthConstraints computedIntrinsicWidthConstraints() = 0;
64 virtual LayoutUnit usedContentHeight() const = 0;
65
66 const ContainerBox& root() const { return *m_root; }
67 LayoutState& layoutState() const;
68 const FormattingState& formattingState() const { return m_formattingState; }
69 virtual const FormattingGeometry& formattingGeometry() const = 0;
70 virtual const FormattingQuirks& formattingQuirks() const = 0;
71
72 enum class EscapeReason {
73 TableQuirkNeedsGeometryFromEstablishedFormattingContext,
74 OutOfFlowBoxNeedsInFlowGeometry,
75 FloatBoxIsAlwaysRelativeToFloatStateRoot,
76 FindFixedHeightAncestorQuirk,
77 DocumentBoxStretchesToViewportQuirk,
78 BodyStretchesToViewportQuirk,
79 TableNeedsAccessToTableWrapper
80 };
81 const BoxGeometry& geometryForBox(const Box&, std::optional<EscapeReason> = std::nullopt) const;
82
83 bool isBlockFormattingContext() const { return root().establishesBlockFormattingContext(); }
84 bool isInlineFormattingContext() const { return root().establishesInlineFormattingContext(); }
85 bool isTableFormattingContext() const { return root().establishesTableFormattingContext(); }
86 bool isTableWrapperBlockFormattingContext() const { return isBlockFormattingContext() && root().isTableWrapperBox(); }
87 bool isFlexFormattingContext() const { return root().establishesFlexFormattingContext(); }
88
89protected:
90 FormattingContext(const ContainerBox& formattingContextRoot, FormattingState&);
91
92 FormattingState& formattingState() { return m_formattingState; }
93 void computeBorderAndPadding(const Box&, const HorizontalConstraints&);
94
95#ifndef NDEBUG
96 virtual void validateGeometryConstraintsAfterLayout() const;
97#endif
98
99 using LayoutQueue = Vector<const Box*>;
100private:
101 void collectOutOfFlowDescendantsIfNeeded();
102 void computeOutOfFlowVerticalGeometry(const Box&, const ConstraintsForOutOfFlowContent&);
103 void computeOutOfFlowHorizontalGeometry(const Box&, const ConstraintsForOutOfFlowContent&);
104
105 WeakPtr<const ContainerBox> m_root;
106 FormattingState& m_formattingState;
107};
