44LayoutUnit FormattingQuirks::heightValueOfNearestContainingBlockWithFixedHeight(const Box& layoutBox) const
45{
46 ASSERT(layoutState().inQuirksMode());
47 // In quirks mode, we go and travers the containing block chain to find a block level box with fixed height value, even if it means leaving
48 // the current formatting context. FIXME: surely we need to do some tricks here when block direction support is added.
49 auto& formattingContext = this->formattingContext();
50 auto* containingBlock = &layoutBox.containingBlock();
51 LayoutUnit bodyAndDocumentVerticalMarginPaddingAndBorder;
52 while (containingBlock) {
53 auto containingBlockHeight = containingBlock->style().logicalHeight();
54 if (containingBlockHeight.isFixed())
55 return LayoutUnit(containingBlockHeight.value() - bodyAndDocumentVerticalMarginPaddingAndBorder);
56
57 // If the only fixed value box we find is the ICB, then ignore the body and the document (vertical) margin, padding and border. So much quirkiness.
58 // -and it's totally insane because now we freely travel across formatting context boundaries and computed margins are nonexistent.
59 if (containingBlock->isBodyBox() || containingBlock->isDocumentBox()) {
60 auto& formattingGeometry = formattingContext.formattingGeometry();
61 auto horizontalConstraints = formattingGeometry.constraintsForInFlowContent(containingBlock->containingBlock(), FormattingContext::EscapeReason::FindFixedHeightAncestorQuirk).horizontal();
62 auto verticalMargin = formattingGeometry.computedVerticalMargin(*containingBlock, horizontalConstraints);
63
64 auto& boxGeometry = formattingContext.geometryForBox(*containingBlock, FormattingContext::EscapeReason::FindFixedHeightAncestorQuirk);
65 auto verticalPadding = boxGeometry.paddingTop().value_or(0) + boxGeometry.paddingBottom().value_or(0);
66 auto verticalBorder = boxGeometry.borderTop() + boxGeometry.borderBottom();
67 bodyAndDocumentVerticalMarginPaddingAndBorder += verticalMargin.before.value_or(0) + verticalMargin.after.value_or(0) + verticalPadding + verticalBorder;
68 }
69
70 if (is<InitialContainingBlock>(*containingBlock))
71 break;
72 containingBlock = &containingBlock->containingBlock();
73 }
74 // Initial containing block has to have a height.
75 return formattingContext.geometryForBox(layoutBox.initialContainingBlock(), FormattingContext::EscapeReason::FindFixedHeightAncestorQuirk).contentBox().height() - bodyAndDocumentVerticalMarginPaddingAndBorder;
76}

78}
