61void BlockFormattingContext::layoutInFlowContent(InvalidationState& invalidationState, const ConstraintsForInFlowContent& constraints)
62{
63 // 9.4.1 Block formatting contexts
64 // In a block formatting context, boxes are laid out one after the other, vertically, beginning at the top of a containing block.
65 // The vertical distance between two sibling boxes is determined by the 'margin' properties.
66 // Vertical margins between adjacent block-level boxes in a block formatting context collapse.
67 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Start] -> block formatting context -> formatting root(" << &root() << ")");
68 auto& formattingRoot = root();
69 ASSERT(formattingRoot.hasInFlowOrFloatingChild());
70 auto& floatingState = formattingState().floatingState();
71 auto floatingContext = FloatingContext { *this, floatingState };
72
73 LayoutQueue layoutQueue;
74 enum class LayoutDirection { Child, Sibling };
75 auto appendNextToLayoutQueue = [&] (const auto& layoutBox, auto direction) {
76 if (direction == LayoutDirection::Child) {
77 if (!is<ContainerBox>(layoutBox))
78 return false;
79 for (auto* child = downcast<ContainerBox>(layoutBox).firstInFlowOrFloatingChild(); child; child = child->nextInFlowOrFloatingSibling()) {
80 if (!invalidationState.needsLayout(*child))
81 continue;
82 layoutQueue.append(child);
83 return true;
84 }
85 return false;
86 }
87
88 if (direction == LayoutDirection::Sibling) {
89 for (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling(); nextSibling; nextSibling = nextSibling->nextInFlowOrFloatingSibling()) {
90 if (!invalidationState.needsLayout(*nextSibling))
91 continue;
92 layoutQueue.append(nextSibling);
93 return true;
94 }
95 return false;
96 }
97 ASSERT_NOT_REACHED();
98 return false;
99 };
100
101 auto constraintsForLayoutBox = [&] (const auto& layoutBox) {
102 auto& containingBlock = layoutBox.containingBlock();
103 return &containingBlock == &formattingRoot ? constraints : formattingGeometry().constraintsForInFlowContent(containingBlock);
104 };
105
106 // This is a post-order tree traversal layout.
107 // The root container layout is done in the formatting context it lives in, not that one it creates, so let's start with the first child.
108 appendNextToLayoutQueue(formattingRoot, LayoutDirection::Child);
109 // 1. Go all the way down to the leaf node
110 // 2. Compute static position and width as we traverse down
111 // 3. As we climb back on the tree, compute height and finialize position
112 // (Any subtrees with new formatting contexts need to layout synchronously)
113 while (!layoutQueue.isEmpty()) {
114 // Traverse down on the descendants and compute width/static position until we find a leaf node.
115 while (true) {
116 auto& layoutBox = *layoutQueue.last();
117 auto containingBlockConstraints = constraintsForLayoutBox(layoutBox);
118
119 computeBorderAndPadding(layoutBox, containingBlockConstraints.horizontal());
120 computeStaticVerticalPosition(layoutBox, containingBlockConstraints.logicalTop());
121 computeWidthAndMargin(floatingContext, layoutBox, { constraints, containingBlockConstraints });
122 computeStaticHorizontalPosition(layoutBox, containingBlockConstraints.horizontal());
123 computePositionToAvoidFloats(floatingContext, layoutBox, { constraints, containingBlockConstraints });
124
125 if (layoutBox.establishesFormattingContext()) {
126 if (is<ContainerBox>(layoutBox) && downcast<ContainerBox>(layoutBox).hasInFlowOrFloatingChild()) {
127 auto& containerBox = downcast<ContainerBox>(layoutBox);
128 if (containerBox.establishesInlineFormattingContext()) {
129 // IFCs inherit floats from parent FCs. We need final vertical position to find intruding floats.
130 precomputeVerticalPositionForBoxAndAncestors(containerBox, { constraints, containingBlockConstraints });
131 }
132 // Layout the inflow descendants of this formatting context root.
133 auto formattingContext = LayoutContext::createFormattingContext(containerBox, layoutState());
134 if (containerBox.isTableWrapperBox())
135 downcast<TableWrapperBlockFormattingContext>(*formattingContext).setHorizontalConstraintsIgnoringFloats(containingBlockConstraints.horizontal());
136 formattingContext->layoutInFlowContent(invalidationState, formattingGeometry().constraintsForInFlowContent(containerBox));
137 }
138 break;
139 }
140 if (!appendNextToLayoutQueue(layoutBox, LayoutDirection::Child))
141 break;
142 }
143
144 // Climb back on the ancestors and compute height/final position.
145 while (!layoutQueue.isEmpty()) {
146 auto& layoutBox = *layoutQueue.takeLast();
147 auto containingBlockConstraints = constraintsForLayoutBox(layoutBox);
148
149 // All inflow descendants (if there are any) are laid out by now. Let's compute the box's height and vertical margin.
150 computeHeightAndMargin(layoutBox, containingBlockConstraints);
151 if (layoutBox.isFloatingPositioned())
152 floatingState.append(floatingContext.toFloatItem(layoutBox));
153 else {
154 // Adjust the vertical position now that we've got final margin values for non-float avoider boxes.
155 // Float avoiders have pre-computed vertical positions when floats are present.
156 if (!layoutBox.isFloatAvoider() || floatingContext.isEmpty()) {
157 auto& formattingState = this->formattingState();
158 auto& boxGeometry = formattingState.boxGeometry(layoutBox);
159 boxGeometry.setLogicalTop(verticalPositionWithMargin(layoutBox, formattingState.usedVerticalMargin(layoutBox), containingBlockConstraints.logicalTop()));
160 }
161 }
162 auto establishesFormattingContext = layoutBox.establishesFormattingContext();
163 if (establishesFormattingContext) {
164 // Now that we computed the box's height, we can layout the out-of-flow descendants.
165 if (is<ContainerBox>(layoutBox) && downcast<ContainerBox>(layoutBox).hasChild()) {
166 auto& containerBox = downcast<ContainerBox>(layoutBox);
167 LayoutContext::createFormattingContext(containerBox, layoutState())->layoutOutOfFlowContent(invalidationState, formattingGeometry().constraintsForOutOfFlowContent(containerBox));
168 }
169 }
170 if (!establishesFormattingContext && is<ContainerBox>(layoutBox))
171 placeInFlowPositionedChildren(downcast<ContainerBox>(layoutBox), containingBlockConstraints.horizontal());
172
173 if (appendNextToLayoutQueue(layoutBox, LayoutDirection::Sibling))
174 break;
175 }
176 }
177 // Place the inflow positioned children.
178 placeInFlowPositionedChildren(formattingRoot, constraints.horizontal());
179 LOG_WITH_STREAM(FormattingContextLayout, stream << "[End] -> block formatting context -> formatting root(" << &root() << ")");
180}

182LayoutUnit BlockFormattingContext::usedContentHeight() const
183{
184 // 10.6.7 'Auto' heights for block formatting context roots
185
186 // If it has block-level children, the height is the distance between the top margin-edge of the topmost block-level
187 // child box and the bottom margin-edge of the bottommost block-level child box.
188
189 // In addition, if the element has any floating descendants whose bottom margin edge is below the element's bottom content edge,
190 // then the height is increased to include those edges. Only floats that participate in this block formatting context are taken
191 // into account, e.g., floats inside absolutely positioned descendants or other floats are not.
192 auto top = std::optional<LayoutUnit> { };
193 auto bottom = std::optional<LayoutUnit> { };
194 if (root().hasInFlowChild()) {
195 top = BoxGeometry::marginBoxRect(geometryForBox(*root().firstInFlowChild())).top();
196 bottom = BoxGeometry::marginBoxRect(geometryForBox(*root().lastInFlowChild())).bottom();
197 }
198
199 auto floatingContext = FloatingContext { *this, formattingState().floatingState() };
200 if (auto floatTop = floatingContext.top()) {
201 top = std::min(*floatTop, top.value_or(*floatTop));
202 auto floatBottom = *floatingContext.bottom();
203 bottom = std::max(floatBottom, bottom.value_or(floatBottom));
204 }
205 return *bottom - *top;
206}

208std::optional<LayoutUnit> BlockFormattingContext::usedAvailableWidthForFloatAvoider(const FloatingContext& floatingContext, const Box& layoutBox, const ConstraintsPair& constraintsPair)
209{
210 // Normally the available width for an in-flow block level box is the width of the containing block's content box.
211 // However (and can't find it anywhere in the spec) non-floating positioned float avoider block level boxes are constrained by existing floats.
212 ASSERT(layoutBox.isFloatAvoider());
213 if (floatingContext.isEmpty())
214 return { };
215 // Float clear pushes the block level box either below the floats, or just one side below but the other side could overlap.
216 // What it means is that the used available width always matches the containing block's constraint.
217 if (layoutBox.hasFloatClear())
218 return { };
219
220 ASSERT(layoutBox.establishesFormattingContext());
221 // Vertical static position is not computed yet for this formatting context root, so let's just pre-compute it for now.
222 precomputeVerticalPositionForBoxAndAncestors(layoutBox, constraintsPair);
223
224 auto logicalTopInFormattingContextRootCoordinate = [&] (auto& floatAvoider) {
225 auto top = BoxGeometry::borderBoxTop(geometryForBox(floatAvoider));
226 for (auto& ancestor : containingBlockChainWithinFormattingContext(floatAvoider))
227 top += BoxGeometry::borderBoxTop(geometryForBox(ancestor));
228 return top;
229 };
230
231 auto floatConstraintsInContainingBlockCoordinate = [&] (auto floatConstraints) {
232 if (!floatConstraints.left && !floatConstraints.right)
233 return FloatingContext::Constraints { };
234 auto offset = LayoutSize { };
235 for (auto& ancestor : containingBlockChainWithinFormattingContext(layoutBox))
236 offset += toLayoutSize(BoxGeometry::borderBoxTopLeft(geometryForBox(ancestor)));
237 if (floatConstraints.left)
238 floatConstraints.left = PointInContextRoot { *floatConstraints.left - offset };
239 if (floatConstraints.right)
240 floatConstraints.right = PointInContextRoot { *floatConstraints.right - offset };
241 return floatConstraints;
242 };
243
244 // FIXME: Check if the non-yet-computed height affects this computation - and whether we have to resolve it at a later point.
245 auto logicalTop = logicalTopInFormattingContextRootCoordinate(layoutBox);
246 auto constraints = floatConstraintsInContainingBlockCoordinate(floatingContext.constraints(logicalTop, logicalTop));
247 if (!constraints.left && !constraints.right)
248 return { };
249 // Shrink the available space if the floats are actually intruding at this vertical position.
250 auto availableWidth = constraintsPair.containingBlock.horizontal().logicalWidth;
251 if (constraints.left)
252 availableWidth -= constraints.left->x;
253 if (constraints.right)
254 availableWidth -= std::max(0_lu, constraintsPair.containingBlock.horizontal().logicalRight() - constraints.right->x);
255 return availableWidth;
256}

258void BlockFormattingContext::placeInFlowPositionedChildren(const ContainerBox& containerBox, const HorizontalConstraints& horizontalConstraints)
259{
260 LOG_WITH_STREAM(FormattingContextLayout, stream << "Start: move in-flow positioned children -> parent: " << &containerBox);
261 for (auto& childBox : childrenOfType<Box>(containerBox)) {
262 if (!childBox.isInFlowPositioned())
263 continue;

266 }
267 LOG_WITH_STREAM(FormattingContextLayout, stream << "End: move in-flow positioned children -> parent: " << &containerBox);
268}

280void BlockFormattingContext::precomputeVerticalPositionForBoxAndAncestors(const Box& layoutBox, const ConstraintsPair& constraintsPair)
281{
282 // In order to figure out whether a box should avoid a float, we need to know the final positions of both (ignore relative positioning for now).
283 // In block formatting context the final position for a normal flow box includes
284 // 1. the static position and
285 // 2. the corresponding (non)collapsed margins.
286 // Now the vertical margins are computed when all the descendants are finalized, because the margin values might be depending on the height of the box
287 // (and the height might be based on the content).
288 // So when we get to the point where we intersect the box with the float to decide if the box needs to move, we don't yet have the final vertical position.
289 //
290 // The idea here is that as long as we don't cross the block formatting context boundary, we should be able to pre-compute the final top position.
291 // FIXME: we currently don't account for the "clear" property when computing the final position for an ancestor.
292 auto& formattingGeometry = this->formattingGeometry();
293 for (auto* ancestor = &layoutBox; ancestor && ancestor != &root(); ancestor = &ancestor->containingBlock()) {
294 auto constraintsForAncestor = [&] {
295 auto& containingBlock = ancestor->containingBlock();
296 return &containingBlock == &root() ? constraintsPair.formattingContextRoot : formattingGeometry.constraintsForInFlowContent(containingBlock);
297 }();
298
299 auto computedVerticalMargin = formattingGeometry.computedVerticalMargin(*ancestor, constraintsForAncestor.horizontal());
300 auto usedNonCollapsedMargin = UsedVerticalMargin::NonCollapsedValues { computedVerticalMargin.before.value_or(0), computedVerticalMargin.after.value_or(0) };
301 auto precomputedMarginBefore = marginCollapse().precomputedMarginBefore(*ancestor, usedNonCollapsedMargin, formattingGeometry);
302
303 auto& boxGeometry = formattingState().boxGeometry(*ancestor);
304 auto nonCollapsedValues = UsedVerticalMargin::NonCollapsedValues { precomputedMarginBefore.nonCollapsedValue, { } };
305 auto collapsedValues = UsedVerticalMargin::CollapsedValues { precomputedMarginBefore.collapsedValue, { }, false };
306 auto verticalMargin = UsedVerticalMargin { nonCollapsedValues, collapsedValues, { precomputedMarginBefore.positiveAndNegativeMarginBefore, { } } };
307
308 formattingState().setUsedVerticalMargin(*ancestor, verticalMargin);
309 boxGeometry.setVerticalMargin({ marginBefore(verticalMargin), marginAfter(verticalMargin) });
310 boxGeometry.setLogicalTop(verticalPositionWithMargin(*ancestor, verticalMargin, constraintsForAncestor.logicalTop()));
311#if ASSERT_ENABLED
312 setPrecomputedMarginBefore(*ancestor, precomputedMarginBefore);
313 boxGeometry.setHasPrecomputedMarginBefore();
314#endif
315 }
316}

318void BlockFormattingContext::computePositionToAvoidFloats(const FloatingContext& floatingContext, const Box& layoutBox, const ConstraintsPair& constraintsPair)
319{
320 if (!layoutBox.isFloatAvoider())
321 return;
322 // In order to position a float avoider we need to know its vertical position relative to its formatting context root (and not just its containing block),
323 // because all the already-placed floats (floats that we are trying to avoid here) in this BFC might belong
324 // to a different set of containing blocks (but they all descendants of the BFC root).
325 // However according to the BFC rules, at this point of the layout flow we don't yet have computed vertical positions for the ancestors.
326 if (layoutBox.isFloatingPositioned()) {
327 precomputeVerticalPositionForBoxAndAncestors(layoutBox, constraintsPair);
328 formattingState().boxGeometry(layoutBox).setLogicalTopLeft(floatingContext.positionForFloat(layoutBox, constraintsPair.containingBlock.horizontal()));
329 return;
330 }
331 // Non-float positioned float avoiders (formatting context roots and clear boxes) should be fine unless there are floats in this context.
332 if (floatingContext.isEmpty())
333 return;
334 precomputeVerticalPositionForBoxAndAncestors(layoutBox, constraintsPair);
335 if (layoutBox.hasFloatClear())
336 return computeVerticalPositionForFloatClear(floatingContext, layoutBox);
337
338 ASSERT(layoutBox.establishesFormattingContext());
339 formattingState().boxGeometry(layoutBox).setLogicalTopLeft(floatingContext.positionForNonFloatingFloatAvoider(layoutBox, constraintsPair.containingBlock.horizontal()));
340}

342void BlockFormattingContext::computeVerticalPositionForFloatClear(const FloatingContext& floatingContext, const Box& layoutBox)
343{
344 ASSERT(layoutBox.hasFloatClear());
345 if (floatingContext.isEmpty())
346 return;
347 auto verticalPositionAndClearance = floatingContext.verticalPositionWithClearance(layoutBox);
348 if (!verticalPositionAndClearance)
349 return;
350
351 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
352 ASSERT(verticalPositionAndClearance->position >= BoxGeometry::borderBoxTop(boxGeometry));
353 boxGeometry.setLogicalTop(verticalPositionAndClearance->position);
354 if (verticalPositionAndClearance->clearance)
355 formattingState().setHasClearance(layoutBox);
356 // FIXME: Reset the margin values on the ancestors/previous siblings now that the float avoider with clearance does not margin collapse anymore.
357}

359void BlockFormattingContext::computeWidthAndMargin(const FloatingContext& floatingContext, const Box& layoutBox, const ConstraintsPair& constraintsPair)
360{
361 auto availableWidthFloatAvoider = std::optional<LayoutUnit> { };
362 if (layoutBox.isFloatAvoider()) {
363 // Float avoiders' available width might be shrunk by existing floats in the context.
364 availableWidthFloatAvoider = usedAvailableWidthForFloatAvoider(floatingContext, layoutBox, constraintsPair);
365 }
366 auto contentWidthAndMargin = formattingGeometry().computedContentWidthAndMargin(layoutBox, constraintsPair.containingBlock.horizontal(), availableWidthFloatAvoider);
367 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
368 boxGeometry.setContentBoxWidth(contentWidthAndMargin.contentWidth);
369 boxGeometry.setHorizontalMargin({ contentWidthAndMargin.usedMargin.start, contentWidthAndMargin.usedMargin.end });
370}

372void BlockFormattingContext::computeHeightAndMargin(const Box& layoutBox, const ConstraintsForInFlowContent& constraints)
373{
374 auto compute = [&](std::optional<LayoutUnit> usedHeight) -> ContentHeightAndMargin {
375 if (layoutBox.isInFlow())
376 return formattingGeometry().inFlowContentHeightAndMargin(layoutBox, constraints.horizontal(), { usedHeight });
377
378 if (layoutBox.isFloatingPositioned())
379 return formattingGeometry().floatingContentHeightAndMargin(layoutBox, constraints.horizontal(), { usedHeight });
380
381 ASSERT_NOT_REACHED();
382 return { };
383 };
384
385 auto contentHeightAndMargin = compute({ });
386 if (auto maxHeight = formattingGeometry().computedMaxHeight(layoutBox)) {
387 if (contentHeightAndMargin.contentHeight > *maxHeight) {
388 auto maxHeightAndMargin = compute(maxHeight);
389 // Used height should remain the same.
390 ASSERT((layoutState().inQuirksMode() && (layoutBox.isBodyBox() || layoutBox.isDocumentBox())) || maxHeightAndMargin.contentHeight == *maxHeight);
391 contentHeightAndMargin = { *maxHeight, maxHeightAndMargin.nonCollapsedMargin };
392 }
393 }
394
395 if (auto minHeight = formattingGeometry().computedMinHeight(layoutBox)) {
396 if (contentHeightAndMargin.contentHeight < *minHeight) {
397 auto minHeightAndMargin = compute(minHeight);
398 // Used height should remain the same.
399 ASSERT((layoutState().inQuirksMode() && (layoutBox.isBodyBox() || layoutBox.isDocumentBox())) || minHeightAndMargin.contentHeight == *minHeight);
400 contentHeightAndMargin = { *minHeight, minHeightAndMargin.nonCollapsedMargin };
401 }
402 }
403
404 // 1. Compute collapsed margins.
405 // 2. Adjust vertical position using the collapsed values.
406 // 3. Adjust previous in-flow sibling margin after using this margin.
407 auto marginCollapse = this->marginCollapse();
408 auto verticalMargin = marginCollapse.collapsedVerticalValues(layoutBox, contentHeightAndMargin.nonCollapsedMargin);
409 // Cache the computed positive and negative margin value pair.
410 formattingState().setUsedVerticalMargin(layoutBox, verticalMargin);
411
412#if ASSERT_ENABLED
413 if (hasPrecomputedMarginBefore(layoutBox) && precomputedMarginBefore(layoutBox).usedValue() != marginBefore(verticalMargin)) {
414 // When the pre-computed margin turns out to be incorrect, we need to re-layout this subtree with the correct margin values.
415 // <div style="float: left"></div>
416 // <div>
417 // <div style="margin-bottom: 200px"></div>
418 // </div>
419 // The float box triggers margin before computation on the ancestor chain to be able to intersect with other floats in the same floating context.
420 // However in some cases the parent margin-top collapses with some next siblings (nephews) and there's no way to be able to properly
421 // account for that without laying out every node in the FC (in the example, the margin-bottom pushes down the float).
422 ASSERT_NOT_IMPLEMENTED_YET();
423 }
424#endif
425 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
426 boxGeometry.setContentBoxHeight(contentHeightAndMargin.contentHeight);
427 boxGeometry.setVerticalMargin({ marginBefore(verticalMargin), marginAfter(verticalMargin) });
428 // Adjust the previous sibling's margin bottom now that this box's vertical margin is computed.
429 updateMarginAfterForPreviousSibling(layoutBox);
430}

432IntrinsicWidthConstraints BlockFormattingContext::computedIntrinsicWidthConstraints()
433{
434 auto& formattingState = this->formattingState();
435 ASSERT(!formattingState.intrinsicWidthConstraints());
436 // Visit the in-flow descendants and compute their min/max intrinsic width if needed.
437 // 1. Go all the way down to the leaf node
438 // 2. Check if actually need to visit all the boxes as we traverse down (already computed, container's min/max does not depend on descendants etc)
439 // 3. As we climb back on the tree, compute min/max intrinsic width
440 // (Any subtrees with new formatting contexts need to layout synchronously)
441 Vector<const Box*> queue;
442 if (root().hasInFlowOrFloatingChild())
443 queue.append(root().firstInFlowOrFloatingChild());
444
445 IntrinsicWidthConstraints constraints;
446 auto maximumHorizontalStackingWidth = LayoutUnit { };
447 auto currentHorizontalStackingWidth = LayoutUnit { };
448 while (!queue.isEmpty()) {
449 while (true) {
450 // Check if we have to deal with descendant content.
451 auto& layoutBox = *queue.last();
452 // Float avoiders are all establish a new formatting context. No need to look inside them.
453 if (layoutBox.isFloatAvoider() && !layoutBox.hasFloatClear())
454 break;
455 // Non-floating block level boxes reset the current horizontal float stacking.
456 // SPEC: This is a bit odd as floating positioning is a formatting context level concept:
457 // e.g.
458 // <div style="float: left; width: 10px;"></div>
459 // <div></div>
460 // <div style="float: left; width: 40px;"></div>
461 // ...will produce a max width of 40px which makes the floats vertically stacked.
462 // Vertically stacked floats makes me think we haven't managed to provide the maximum preferred width for the content.
463 maximumHorizontalStackingWidth = std::max(currentHorizontalStackingWidth, maximumHorizontalStackingWidth);
464 currentHorizontalStackingWidth = { };
465 // Already has computed intrinsic constraints.
466 if (formattingState.intrinsicWidthConstraintsForBox(layoutBox))
467 break;
468 // Box with fixed width defines their descendant content intrinsic width.
469 if (layoutBox.style().width().isFixed())
470 break;
471 // Non-float avoider formatting context roots are opaque to intrinsic width computation.
472 if (layoutBox.establishesFormattingContext())
473 break;
474 // No relevant child content.
475 if (!is<ContainerBox>(layoutBox) || !downcast<ContainerBox>(layoutBox).hasInFlowOrFloatingChild())
476 break;
477 queue.append(downcast<ContainerBox>(layoutBox).firstInFlowOrFloatingChild());
478 }
479 // Compute min/max intrinsic width bottom up if needed.
480 while (!queue.isEmpty()) {
481 auto& layoutBox = *queue.takeLast();
482 auto desdendantConstraints = formattingState.intrinsicWidthConstraintsForBox(layoutBox);
483 if (!desdendantConstraints) {
484 desdendantConstraints = formattingGeometry().intrinsicWidthConstraints(layoutBox);
485 formattingState.setIntrinsicWidthConstraintsForBox(layoutBox, *desdendantConstraints);
486 }
487 constraints.minimum = std::max(constraints.minimum, desdendantConstraints->minimum);
488 auto willStackHorizontally = layoutBox.isFloatAvoider() && !layoutBox.hasFloatClear();
489 if (willStackHorizontally)
490 currentHorizontalStackingWidth += desdendantConstraints->maximum;
491 else
492 constraints.maximum = std::max(constraints.maximum, desdendantConstraints->maximum);
493 // Move over to the next sibling or take the next box in the queue.
494 if (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling()) {
495 queue.append(nextSibling);
496 break;
497 }
498 }
499 }
500 maximumHorizontalStackingWidth = std::max(currentHorizontalStackingWidth, maximumHorizontalStackingWidth);
501 constraints.maximum = std::max(constraints.maximum, maximumHorizontalStackingWidth);
502 formattingState.setIntrinsicWidthConstraints(constraints);
503 return constraints;
504}

506LayoutUnit BlockFormattingContext::verticalPositionWithMargin(const Box& layoutBox, const UsedVerticalMargin& verticalMargin, LayoutUnit containingBlockContentBoxTop) const
507{
508 ASSERT(!layoutBox.isOutOfFlowPositioned());
509 // Now that we've computed the final margin before, let's shift the box's vertical position if needed.
510 // 1. Check if the box has clearance. If so, we've already precomputed/finalized the top value and vertical margin does not impact it anymore.
511 // 2. Check if the margin before collapses with the previous box's margin after. if not -> return previous box's bottom including margin after + marginBefore
512 // 3. Check if the previous box's margins collapse through. If not -> return previous box' bottom excluding margin after + marginBefore (they are supposed to be equal)
513 // 4. Go to previous box and start from step #1 until we hit the parent box.
514 auto& boxGeometry = geometryForBox(layoutBox);
515 if (formattingState().hasClearance(layoutBox))
516 return BoxGeometry::borderBoxTop(boxGeometry);
517
518 auto* currentLayoutBox = &layoutBox;
519 while (currentLayoutBox) {
520 if (!currentLayoutBox->previousInFlowSibling())
521 break;
522 auto& previousInFlowSibling = *currentLayoutBox->previousInFlowSibling();
523 if (!marginCollapse().marginBeforeCollapsesWithPreviousSiblingMarginAfter(*currentLayoutBox)) {
524 auto& previousBoxGeometry = geometryForBox(previousInFlowSibling);
525 return BoxGeometry::marginBoxRect(previousBoxGeometry).bottom() + marginBefore(verticalMargin);
526 }
527
528 if (!marginCollapse().marginsCollapseThrough(previousInFlowSibling)) {
529 auto& previousBoxGeometry = geometryForBox(previousInFlowSibling);
530 return BoxGeometry::borderBoxRect(previousBoxGeometry).bottom() + marginBefore(verticalMargin);
531 }
532 currentLayoutBox = &previousInFlowSibling;
533 }
534
535 // Adjust vertical position depending whether this box directly or indirectly adjoins with its parent.
536 auto directlyAdjoinsParent = !layoutBox.previousInFlowSibling();
537 if (directlyAdjoinsParent) {
538 // If the top and bottom margins of a box are adjoining, then it is possible for margins to collapse through it.
539 // In this case, the position of the element depends on its relationship with the other elements whose margins are being collapsed.
540 if (verticalMargin.collapsedValues.isCollapsedThrough) {
541 // If the element's margins are collapsed with its parent's top margin, the top border edge of the box is defined to be the same as the parent's.
542 if (marginCollapse().marginBeforeCollapsesWithParentMarginBefore(layoutBox))
543 return containingBlockContentBoxTop;
544 // Otherwise, either the element's parent is not taking part in the margin collapsing, or only the parent's bottom margin is involved.
545 // The position of the element's top border edge is the same as it would have been if the element had a non-zero bottom border.
546 auto beforeMarginWithBottomBorder = marginCollapse().marginBeforeIgnoringCollapsingThrough(layoutBox, verticalMargin.nonCollapsedValues);
547 return containingBlockContentBoxTop + beforeMarginWithBottomBorder;
548 }
549 // Non-collapsed through box vertical position depending whether the margin collapses.
550 if (marginCollapse().marginBeforeCollapsesWithParentMarginBefore(layoutBox))
551 return containingBlockContentBoxTop;
552
553 return containingBlockContentBoxTop + marginBefore(verticalMargin);
554 }
555 // At this point this box indirectly (via collapsed through previous in-flow siblings) adjoins the parent. Let's check if it margin collapses with the parent.
556 auto& containingBlock = layoutBox.containingBlock();
557 ASSERT(containingBlock.firstInFlowChild());
558 ASSERT(containingBlock.firstInFlowChild() != &layoutBox);
559 if (marginCollapse().marginBeforeCollapsesWithParentMarginBefore(*containingBlock.firstInFlowChild()))
560 return containingBlockContentBoxTop;
561
562 return containingBlockContentBoxTop + marginBefore(verticalMargin);
563}

565void BlockFormattingContext::updateMarginAfterForPreviousSibling(const Box& layoutBox)
566{
567 auto marginCollapse = this->marginCollapse();
568 auto& formattingState = this->formattingState();
569 // 1. Get the margin before value from the next in-flow sibling. This is the same as this box's margin after value now since they are collapsed.
570 // 2. Update the collapsed margin after value as well as the positive/negative cache.
571 // 3. Check if the box's margins collapse through.
572 // 4. If so, update the positive/negative cache.
573 // 5. In case of collapsed through margins check if the before margin collapes with the previous inflow sibling's after margin.
574 // 6. If so, jump to #2.
575 // 7. No need to propagate to parent because its margin is not computed yet (pre-computed at most).
576 auto* currentBox = &layoutBox;
577 while (marginCollapse.marginBeforeCollapsesWithPreviousSiblingMarginAfter(*currentBox)) {
578 auto& previousSibling = *currentBox->previousInFlowSibling();
579 auto previousSiblingVerticalMargin = formattingState.usedVerticalMargin(previousSibling);
580
581 auto collapsedVerticalMarginBefore = previousSiblingVerticalMargin.collapsedValues.before;
582 auto collapsedVerticalMarginAfter = geometryForBox(*currentBox).marginBefore();
583
584 auto marginsCollapseThrough = marginCollapse.marginsCollapseThrough(previousSibling);
585 if (marginsCollapseThrough)
586 collapsedVerticalMarginBefore = collapsedVerticalMarginAfter;
587
588 // Update positive/negative cache.
589 auto previousSiblingPositiveNegativeMargin = formattingState.usedVerticalMargin(previousSibling).positiveAndNegativeValues;
590 auto positiveNegativeMarginBefore = formattingState.usedVerticalMargin(*currentBox).positiveAndNegativeValues.before;
591
592 auto adjustedPreviousSiblingVerticalMargin = previousSiblingVerticalMargin;
593 adjustedPreviousSiblingVerticalMargin.positiveAndNegativeValues.after = marginCollapse.computedPositiveAndNegativeMargin(positiveNegativeMarginBefore, previousSiblingPositiveNegativeMargin.after);
594 if (marginsCollapseThrough) {
595 adjustedPreviousSiblingVerticalMargin.positiveAndNegativeValues.before = marginCollapse.computedPositiveAndNegativeMargin(previousSiblingPositiveNegativeMargin.before, adjustedPreviousSiblingVerticalMargin.positiveAndNegativeValues.after);
596 adjustedPreviousSiblingVerticalMargin.positiveAndNegativeValues.after = adjustedPreviousSiblingVerticalMargin.positiveAndNegativeValues.before;
597 }
598 formattingState.setUsedVerticalMargin(previousSibling, adjustedPreviousSiblingVerticalMargin);
599
600 if (!marginsCollapseThrough)
601 break;
602
603 currentBox = &previousSibling;
604 }
605}
