39// This class implements positioning and sizing for boxes participating in a block formatting context.
40class BlockFormattingGeometry : public FormattingGeometry {
41public:
42 BlockFormattingGeometry(const BlockFormattingContext&);
43
44 ContentHeightAndMargin inFlowContentHeightAndMargin(const Box&, const HorizontalConstraints&, const OverriddenVerticalValues&) const;
45 ContentWidthAndMargin inFlowContentWidthAndMargin(const Box&, const HorizontalConstraints&, const OverriddenHorizontalValues&) const;
46
47 LayoutUnit staticVerticalPosition(const Box&, LayoutUnit containingBlockContentBoxTop) const;
48 LayoutUnit staticHorizontalPosition(const Box&, const HorizontalConstraints&) const;
49
50 IntrinsicWidthConstraints intrinsicWidthConstraints(const Box&) const;
51
52 ContentWidthAndMargin computedContentWidthAndMargin(const Box&, const HorizontalConstraints&, std::optional<LayoutUnit> availableWidthFloatAvoider) const;
53
54private:
55 ContentHeightAndMargin inFlowNonReplacedContentHeightAndMargin(const Box&, const HorizontalConstraints&, const OverriddenVerticalValues&) const;
56 ContentWidthAndMargin inFlowNonReplacedContentWidthAndMargin(const Box&, const HorizontalConstraints&, const OverriddenHorizontalValues&) const;
57 ContentWidthAndMargin inFlowReplacedContentWidthAndMargin(const ReplacedBox&, const HorizontalConstraints&, const OverriddenHorizontalValues&) const;
58
59 const BlockFormattingContext& formattingContext() const { return downcast<BlockFormattingContext>(FormattingGeometry::formattingContext()); }
60};
61
