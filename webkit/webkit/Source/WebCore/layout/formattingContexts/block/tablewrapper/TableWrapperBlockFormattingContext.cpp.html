55void TableWrapperBlockFormattingContext::layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent& constraints)
56{
57 // The table generates a principal block container box called the table wrapper box that contains the table box itself and any caption boxes
58 // (in document order). The table box is a block-level box that contains the table's internal table boxes.
59 // The caption boxes are principal block-level boxes that retain their own content, padding, margin, and border areas, and are rendered
60 // as normal block boxes inside the table wrapper box. Whether the caption boxes are placed before or after the table box is decided by
61 // the 'caption-side' property, as described below.
62 for (auto& child : childrenOfType<ContainerBox>(root())) {
63 if (child.isTableBox())
64 layoutTableBox(child, constraints);
65 else if (child.isTableCaption())
66 ASSERT_NOT_IMPLEMENTED_YET();
67 else
68 ASSERT_NOT_REACHED();
69 }
70}

72void TableWrapperBlockFormattingContext::layoutTableBox(const ContainerBox& tableBox, const ConstraintsForInFlowContent& constraints)
73{
74 layoutState().ensureTableFormattingState(tableBox);
75
76 computeBorderAndPaddingForTableBox(tableBox, constraints.horizontal());
77 computeStaticVerticalPosition(tableBox, constraints.logicalTop());
78 computeWidthAndMarginForTableBox(tableBox, constraints.horizontal());
79 computeStaticHorizontalPosition(tableBox, constraints.horizontal());
80
81 if (tableBox.hasChild()) {
82 auto& formattingGeometry = this->formattingGeometry();
83 auto invalidationState = InvalidationState { };
84 auto constraints = ConstraintsForTableContent { formattingGeometry.constraintsForInFlowContent(tableBox), formattingGeometry.computedHeight(tableBox) };
85 LayoutContext::createFormattingContext(tableBox, layoutState())->layoutInFlowContent(invalidationState, constraints);
86 }
87
88 computeHeightAndMarginForTableBox(tableBox, constraints);
89}

91void TableWrapperBlockFormattingContext::computeBorderAndPaddingForTableBox(const ContainerBox& tableBox, const HorizontalConstraints& horizontalConstraints)
92{
93 ASSERT(tableBox.isTableBox());
94 if (!tableBox.hasChild() || tableBox.style().borderCollapse() == BorderCollapse::Separate) {
95 BlockFormattingContext::computeBorderAndPadding(tableBox, horizontalConstraints);
96 return;
97 }
98 // UAs must compute an initial left and right border width for the table by examining
99 // the first and last cells in the first row of the table.
100 // The left border width of the table is half of the first cell's collapsed left border,
101 // and the right border width of the table is half of the last cell's collapsed right border.
102 // The top border width of the table is computed by examining all cells who collapse their top
103 // borders with the top border of the table. The top border width of the table is equal to half of the
104 // maximum collapsed top border. The bottom border width is computed by examining all cells whose bottom borders collapse
105 // with the bottom of the table. The bottom border width is equal to half of the maximum collapsed bottom border.
106 auto& grid = layoutState().establishedTableFormattingState(tableBox).tableGrid();
107 auto tableBorder = formattingGeometry().computedBorder(tableBox);
108
109 auto& firstColumnFirstRowBox = grid.slot({ 0 , 0 })->cell().box();
110 auto leftBorder = std::max(tableBorder.horizontal.left, formattingGeometry().computedBorder(firstColumnFirstRowBox).horizontal.left);
111
112 auto& lastColumnFirstRow = grid.slot({ grid.columns().size() - 1, 0 })->cell().box();
113 auto rightBorder = std::max(tableBorder.horizontal.right, formattingGeometry().computedBorder(lastColumnFirstRow).horizontal.right);
114
115 auto topBorder = tableBorder.vertical.top;
116 auto bottomBorder = tableBorder.vertical.bottom;
117 auto lastRowIndex = grid.rows().size() - 1;
118 for (size_t columnIndex = 0; columnIndex < grid.columns().size(); ++columnIndex) {
119 auto& boxInFirstRox = grid.slot({ columnIndex, 0 })->cell().box();
120 auto& boxInLastRow = grid.slot({ columnIndex, lastRowIndex })->cell().box();
121
122 topBorder = std::max(topBorder, formattingGeometry().computedBorder(boxInFirstRox).vertical.top);
123 bottomBorder = std::max(bottomBorder, formattingGeometry().computedBorder(boxInLastRow).vertical.bottom);
124 }
125
126 topBorder = std::max(topBorder, formattingGeometry().computedBorder(*tableBox.firstChild()).vertical.top);
127 for (auto& section : childrenOfType<ContainerBox>(tableBox)) {
128 auto horiztonalBorder = formattingGeometry().computedBorder(section).horizontal;
129 leftBorder = std::max(leftBorder, horiztonalBorder.left);
130 rightBorder = std::max(rightBorder, horiztonalBorder.right);
131 }
132 bottomBorder = std::max(bottomBorder, formattingGeometry().computedBorder(*tableBox.lastChild()).vertical.bottom);
133
134 auto& rows = grid.rows().list();
135 topBorder = std::max(topBorder, formattingGeometry().computedBorder(rows.first().box()).vertical.top);
136 for (auto& row : rows) {
137 auto horiztonalBorder = formattingGeometry().computedBorder(row.box()).horizontal;
138 leftBorder = std::max(leftBorder, horiztonalBorder.left);
139 rightBorder = std::max(rightBorder, horiztonalBorder.right);
140 }
141 bottomBorder = std::max(bottomBorder, formattingGeometry().computedBorder(rows.last().box()).vertical.bottom);
142
143 auto collapsedBorder = Edges { { leftBorder, rightBorder }, { topBorder, bottomBorder } };
144 grid.setCollapsedBorder(collapsedBorder);
145
146 auto& boxGeometry = formattingState().boxGeometry(tableBox);
147 boxGeometry.setBorder(collapsedBorder / 2);
148 boxGeometry.setPadding(formattingGeometry().computedPadding(tableBox, horizontalConstraints.logicalWidth));
149}

151void TableWrapperBlockFormattingContext::computeWidthAndMarginForTableBox(const ContainerBox& tableBox, const HorizontalConstraints& horizontalConstraints)
152{
153 ASSERT(tableBox.isTableBox());
154 // This is a special table "fit-content size" behavior handling. Not in the spec though.
155 // Table returns its final width as min/max. Use this final width value to computed horizontal margins etc.
156 auto& formattingStateForTableBox = layoutState().establishedTableFormattingState(tableBox);
157 auto intrinsicWidthConstraints = IntrinsicWidthConstraints { };
158 if (auto precomputedIntrinsicWidthConstraints = formattingStateForTableBox.intrinsicWidthConstraints())
159 intrinsicWidthConstraints = *precomputedIntrinsicWidthConstraints;
160 else {
161 if (tableBox.hasChild())
162 intrinsicWidthConstraints = LayoutContext::createFormattingContext(tableBox, layoutState())->computedIntrinsicWidthConstraints();
163 formattingStateForTableBox.setIntrinsicWidthConstraints(intrinsicWidthConstraints);
164 }
165
166 auto availableHorizontalSpace = horizontalConstraints.logicalWidth;
167 // 1. The table generates a principal block container box called the table wrapper box that contains the table box itself and any caption boxes.
168 // 2. The table wrapper box establishes a block formatting context, and the table box establishes a table formatting context.
169 // 3. The computed values of properties 'position', 'float', 'margin-*', 'top', 'right', 'bottom', and 'left' on
170 // the table element are used on the table wrapper box and not the table box; all other values of non-inheritable
171 // properties are used on the table box and not the table wrapper box.
172 // 4. In a block formatting context, each box's left outer edge touches the left edge of the containing block.
173 // This is true even in the presence of floats, unless the box establishes a new block formatting context (in which case the
174 // box itself may become narrower due to the floats)
175 //
176 // Now consider the following case:
177 // <div style="display: block; width: 500px;">
178 // <div style="float: left; width: 100px;"></div>
179 // <div style="display: table; width: 10%;"></div>
180 // </div>
181 // </div>
182 // 1. We create a table wrapper box to wrap the "display: table" block level box (#1).
183 // 2. The table wrapper box's width property is set to auto (#3).
184 // 3. Since it establishes a new block formatting context, the available horizontal space gets shrunk by the float (#4)
185 // 4. The table wrapper box's used width computes to 500px - 100px -> 400px;
186 //
187 // Now we are inside the BFC established by the table wrapper box and try to resolve the table's width -> %10.
188 // According to the normal BFC rules, it should compute to 10% of the containing block's logical width: 400px -> 40px.
189 // However in practice it computes to 50px (10% of 500px).
190 // Similar setup with non-table content would resolve the inner block level box's width to 40px;
191 // This needs clarification in the spec.
192 auto horizontalConstraintForResolvingWidth = m_horizontalConstraintsIgnoringFloats.logicalWidth;
193 auto& formattingGeometry = this->formattingGeometry();
194 auto computedWidth = formattingGeometry.computedWidth(tableBox, horizontalConstraintForResolvingWidth);
195 auto computedMaxWidth = formattingGeometry.computedMaxWidth(tableBox, horizontalConstraintForResolvingWidth);
196 auto computedMinWidth = formattingGeometry.computedMinWidth(tableBox, horizontalConstraintForResolvingWidth);
197 // Use the generic shrink-to-fit-width logic as the initial width for the table.
198 auto usedWidth = std::min(std::max(intrinsicWidthConstraints.minimum, availableHorizontalSpace), intrinsicWidthConstraints.maximum);
199 if (computedWidth || computedMinWidth || computedMaxWidth) {
200 if (computedWidth) {
201 // Normalize the computed width value first.
202 if (computedMaxWidth && *computedWidth > *computedMaxWidth)
203 computedWidth = computedMaxWidth;
204 if (computedMinWidth && *computedWidth < *computedMinWidth)
205 computedWidth = computedMinWidth;
206 usedWidth = *computedWidth < intrinsicWidthConstraints.minimum ? intrinsicWidthConstraints.minimum : *computedWidth;
207 }
208
209 if (computedMaxWidth && *computedMaxWidth < usedWidth)
210 usedWidth = intrinsicWidthConstraints.minimum;
211 if (computedMinWidth && *computedMinWidth > usedWidth)
212 usedWidth = *computedMinWidth;
213 }
214
215 auto contentWidthAndMargin = formattingGeometry.inFlowContentWidthAndMargin(tableBox, horizontalConstraints, OverriddenHorizontalValues { usedWidth, { } });
216
217 auto& boxGeometry = formattingState().boxGeometry(tableBox);
218 boxGeometry.setContentBoxWidth(contentWidthAndMargin.contentWidth);
219 boxGeometry.setHorizontalMargin({ contentWidthAndMargin.usedMargin.start, contentWidthAndMargin.usedMargin.end });
220}

222void TableWrapperBlockFormattingContext::computeHeightAndMarginForTableBox(const ContainerBox& tableBox, const ConstraintsForInFlowContent& constraints)
223{
224 ASSERT(tableBox.isTableBox());
225 // Table is a special BFC content. Its height is mainly driven by the content. Computed height, min-height and max-height are all
226 // already been taken into account during the TFC layout.
227 auto overriddenTableHeight = [&]() -> std::optional<LayoutUnit> {
228 if (layoutState().inQuirksMode())
229 return formattingQuirks().overriddenTableHeight(tableBox);
230 if (tableBox.hasInFlowOrFloatingChild())
231 return formattingGeometry().contentHeightForFormattingContextRoot(tableBox);
232 return { };
233 }();
234
235 auto heightAndMargin = formattingGeometry().inFlowContentHeightAndMargin(tableBox, constraints.horizontal(), { overriddenTableHeight });
236 auto verticalMargin = marginCollapse().collapsedVerticalValues(tableBox, heightAndMargin.nonCollapsedMargin);
237 // Cache the computed positive and negative margin value pair.
238 formattingState().setUsedVerticalMargin(tableBox, verticalMargin);
239
240 auto& boxGeometry = formattingState().boxGeometry(tableBox);
241 boxGeometry.setLogicalTop(verticalPositionWithMargin(tableBox, verticalMargin, constraints.logicalTop()));
242 boxGeometry.setContentBoxHeight(heightAndMargin.contentHeight);
243 boxGeometry.setVerticalMargin({ marginBefore(verticalMargin), marginAfter(verticalMargin) });
244 // Adjust the previous sibling's margin bottom now that this box's vertical margin is computed.
245 updateMarginAfterForPreviousSibling(tableBox);
246}
