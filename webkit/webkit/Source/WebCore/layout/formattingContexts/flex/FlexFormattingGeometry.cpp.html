42IntrinsicWidthConstraints FlexFormattingGeometry::intrinsicWidthConstraints(const ContainerBox& flexItem) const
43{
44 auto fixedMarginBorderAndPadding = [&](auto& layoutBox) {
45 auto& style = layoutBox.style();
46 return fixedValue(style.marginStart()).value_or(0)
47 + LayoutUnit { style.borderLeftWidth() }
48 + fixedValue(style.paddingLeft()).value_or(0)
49 + fixedValue(style.paddingRight()).value_or(0)
50 + LayoutUnit { style.borderRightWidth() }
51 + fixedValue(style.marginEnd()).value_or(0);
52 };
53
54 auto computedIntrinsicWidthConstraints = [&]() -> IntrinsicWidthConstraints {
55 auto logicalWidth = flexItem.style().logicalWidth();
56 // Minimum/maximum width can't be depending on the containing block's width.
57 auto needsResolvedContainingBlockWidth = logicalWidth.isCalculated() || logicalWidth.isPercent() || logicalWidth.isRelative();
58 if (needsResolvedContainingBlockWidth)
59 return { };
60
61 if (auto width = fixedValue(logicalWidth))
62 return { *width, *width };
63
64 ASSERT(flexItem.establishesFormattingContext());
65 auto& layoutState = this->layoutState();
66 auto intrinsicWidthConstraints = LayoutContext::createFormattingContext(flexItem, const_cast<LayoutState&>(layoutState))->computedIntrinsicWidthConstraints();
67 if (logicalWidth.isMinContent())
68 return { intrinsicWidthConstraints.minimum, intrinsicWidthConstraints.minimum };
69 if (logicalWidth.isMaxContent())
70 return { intrinsicWidthConstraints.maximum, intrinsicWidthConstraints.maximum };
71 return intrinsicWidthConstraints;
72 };
73 auto intrinsicWidthConstraints = constrainByMinMaxWidth(flexItem, computedIntrinsicWidthConstraints());
74 intrinsicWidthConstraints.expand(fixedMarginBorderAndPadding(flexItem));
75 return intrinsicWidthConstraints;
76}
