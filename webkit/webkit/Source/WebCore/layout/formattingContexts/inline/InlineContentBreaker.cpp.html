43static inline bool hasTrailingTextContent(const InlineContentBreaker::ContinuousContent& continuousContent)
44{
45 for (auto& run : WTF::makeReversedRange(continuousContent.runs())) {
46 auto& inlineItem = run.inlineItem;
47 if (inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd())
48 continue;
49 return inlineItem.isText();
50 }
51 return false;
52}

55static inline bool hasLeadingTextContent(const InlineContentBreaker::ContinuousContent& continuousContent)
56{
57 for (auto& run : continuousContent.runs()) {
58 auto& inlineItem = run.inlineItem;
59 if (inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd())
60 continue;
61 return inlineItem.isText();
62 }
63 return false;
64}

66static inline bool hasTextRun(const InlineContentBreaker::ContinuousContent& continuousContent)
67{
68 // <span>text</span> is considered a text run even with the [inline box start][inline box end] inline items.
69 // Based on standards commit boundary rules it would be enough to check the first inline item, but due to the table quirk, we can have
70 // image and text next to each other inside a continuous set of runs (see InlineFormattingContext::Quirks::hasSoftWrapOpportunityAtImage).
71 for (auto& run : continuousContent.runs()) {
72 if (run.inlineItem.isText())
73 return true;
74 }
75 return false;
76}

78static inline bool isVisuallyEmptyWhitespaceContent(const InlineContentBreaker::ContinuousContent& continuousContent)
79{
80 // [<span></span> ] [<span> </span>] [ <span style="padding: 0px;"></span>] are all considered visually empty whitespace content.
81 // [<span style="border: 1px solid red"></span> ] while this is whitespace content only, it is not considered visually empty.
82 // Due to commit boundary rules, we just need to check the first non-typeless inline item (can't have both [img] and [text])
83 for (auto& run : continuousContent.runs()) {
84 auto& inlineItem = run.inlineItem;
85 // FIXME: check for padding border etc.
86 if (inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd())
87 continue;
88 return inlineItem.isText() && downcast<InlineTextItem>(inlineItem).isWhitespace();
89 }
90 return false;
91}

93static inline bool isNonContentRunsOnly(const InlineContentBreaker::ContinuousContent& continuousContent)
94{
95 // <span></span> <- non content runs.
96 for (auto& run : continuousContent.runs()) {
97 auto& inlineItem = run.inlineItem;
98 if (inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd())
99 continue;
100 return false;
101 }
102 return true;
103}

105static inline std::optional<size_t> firstTextRunIndex(const InlineContentBreaker::ContinuousContent& continuousContent)
106{
107 auto& runs = continuousContent.runs();
108 for (size_t index = 0; index < runs.size(); ++index) {
109 if (runs[index].inlineItem.isText())
110 return index;
111 }
112 return { };

115bool InlineContentBreaker::isWrappingAllowed(const InlineItem& inlineItem)
116{
117 auto& styleToUse = inlineItem.isBox() ? inlineItem.layoutBox().parent().style() : inlineItem.layoutBox().style();
118 // Do not try to wrap overflown 'pre' and 'no-wrap' content to next line.
119 return styleToUse.whiteSpace() != WhiteSpace::Pre && styleToUse.whiteSpace() != WhiteSpace::NoWrap;
120}

122bool InlineContentBreaker::shouldKeepEndOfLineWhitespace(const ContinuousContent& continuousContent) const
123{
124 // Grab the style and check for white-space property to decide whether we should let this whitespace content overflow the current line.
125 // Note that the "keep" in this context means we let the whitespace content sit on the current line.
126 // It might very well get collapsed when we close the line (normal/nowrap/pre-line).
127 // See https://www.w3.org/TR/css-text-3/#white-space-property
128 auto whitespace = continuousContent.runs()[*firstTextRunIndex(continuousContent)].inlineItem.style().whiteSpace();
129 return whitespace == WhiteSpace::Normal || whitespace == WhiteSpace::NoWrap || whitespace == WhiteSpace::PreWrap || whitespace == WhiteSpace::PreLine;
130}

132InlineContentBreaker::Result InlineContentBreaker::processInlineContent(const ContinuousContent& candidateContent, const LineStatus& lineStatus)
133{
134 ASSERT(!std::isnan(lineStatus.availableWidth));
135 auto processCandidateContent = [&] {
136 if (candidateContent.logicalWidth() <= lineStatus.availableWidth)
137 return Result { Result::Action::Keep };
138#if USE_FLOAT_AS_INLINE_LAYOUT_UNIT
139 // Preferred width computation sums up floats while line breaker subtracts them. This can lead to epsilon-scale differences.
140 if (WTF::areEssentiallyEqual(candidateContent.logicalWidth(), lineStatus.availableWidth))
141 return Result { Result::Action::Keep };
142#endif
143 return processOverflowingContent(candidateContent, lineStatus);
144 };
145
146 auto result = processCandidateContent();
147 if (result.action == Result::Action::Wrap && lineStatus.trailingSoftHyphenWidth && hasLeadingTextContent(candidateContent)) {
148 // A trailing soft hyphen with a wrapped text content turns into a visible hyphen.
149 // Let's check if there's enough space for the hyphen character.
150 auto hyphenOverflows = *lineStatus.trailingSoftHyphenWidth > lineStatus.availableWidth;
151 auto action = hyphenOverflows ? Result::Action::RevertToLastNonOverflowingWrapOpportunity : Result::Action::WrapWithHyphen;
152 result = { action, IsEndOfLine::Yes };
153 }
154 return result;
155}

157struct OverflowingTextContent {
158 size_t runIndex { 0 }; // Overflowing run index. There's always an overflowing run.
159 struct BreakingPosition {
160 size_t runIndex { 0 };
161 struct TrailingContent {
162 // Trailing content is either the run's left side (when we break the run somewhere in the middle) or the previous run.
163 // Sometimes the breaking position is at the very beginning of the first run, so there's no trailing run at all.
164 bool overflows { false };
165 std::optional<InlineContentBreaker::PartialRun> partialRun { };
166 };
167 std::optional<TrailingContent> trailingContent { };
168 };
169 std::optional<BreakingPosition> breakingPosition { }; // Where we actually break this overflowing content.
170};

172InlineContentBreaker::Result InlineContentBreaker::processOverflowingContent(const ContinuousContent& overflowContent, const LineStatus& lineStatus) const
173{
174 auto continuousContent = ContinuousContent { overflowContent };
175 ASSERT(!continuousContent.runs().isEmpty());
176
177 ASSERT(continuousContent.logicalWidth() > lineStatus.availableWidth);
178 if (continuousContent.hasTrailingCollapsibleContent()) {
179 ASSERT(hasTrailingTextContent(overflowContent));
180 // First check if the content fits without the trailing collapsible part.
181 if (continuousContent.nonCollapsibleLogicalWidth() <= lineStatus.availableWidth)
182 return { Result::Action::Keep, IsEndOfLine::No };
183 // Now check if we can trim the line too.
184 if (lineStatus.hasFullyCollapsibleTrailingRun && continuousContent.isFullyCollapsible()) {
185 // If this new content is fully collapsible, it should surely fit.
186 return { Result::Action::Keep, IsEndOfLine::No };
187 }
188 } else if (lineStatus.collapsibleWidth && isNonContentRunsOnly(continuousContent)) {
189 // Let's see if the non-content runs fit when the line has trailing collapsible content.
190 // "text content <span style="padding: 1px"></span>" <- the <span></span> runs could fit after collapsing the trailing whitespace.
191 if (continuousContent.logicalWidth() <= lineStatus.availableWidth + lineStatus.collapsibleWidth)
192 return { Result::Action::Keep };
193 }
194 if (isVisuallyEmptyWhitespaceContent(continuousContent) && shouldKeepEndOfLineWhitespace(continuousContent)) {
195 // This overflowing content apparently falls into the remove/hang end-of-line-spaces category.
196 // see https://www.w3.org/TR/css-text-3/#white-space-property matrix
197 return { Result::Action::Keep };
198 }
199
200 size_t overflowingRunIndex = 0;
201 if (hasTextRun(continuousContent)) {
202 auto tryBreakingContentWithText = [&]() -> std::optional<Result> {
203 // 1. This text content is not breakable.
204 // 2. This breakable text content does not fit at all. Not even the first glyph. This is a very special case.
205 // 3. We can break the content but it still overflows.
206 // 4. Managed to break the content before the overflow point.
207 auto overflowingContent = processOverflowingContentWithText(continuousContent, lineStatus);
208 overflowingRunIndex = overflowingContent.runIndex;
209 if (!overflowingContent.breakingPosition)
210 return { };
211 auto trailingContent = overflowingContent.breakingPosition->trailingContent;
212 if (!trailingContent) {
213 // We tried to break the content but the available space can't even accommodate the first glyph.
214 // 1. Wrap the content over to the next line when we've got content on the line already.
215 // 2. Keep the first glyph on the empty line (or keep the whole run if it has only one glyph/completely empty).
216 if (lineStatus.hasContent)
217 return Result { Result::Action::Wrap, IsEndOfLine::Yes };
218 auto leadingTextRunIndex = *firstTextRunIndex(continuousContent);
219 auto& inlineTextItem = downcast<InlineTextItem>(continuousContent.runs()[leadingTextRunIndex].inlineItem);
220 if (inlineTextItem.length() <= 1)
221 return Result { Result::Action::Keep, IsEndOfLine::Yes };
222 auto firstCharacterWidth = TextUtil::width(inlineTextItem, inlineTextItem.start(), inlineTextItem.start() + 1, lineStatus.contentLogicalRight);
223 auto firstCharacterRun = PartialRun { 1, firstCharacterWidth };
224 return Result { Result::Action::Break, IsEndOfLine::Yes, Result::PartialTrailingContent { leadingTextRunIndex, firstCharacterRun } };
225 }
226 if (trailingContent->overflows && lineStatus.hasContent) {
227 // We managed to break a run with overflow but the line already has content. Let's wrap it to the next line.
228 return Result { Result::Action::Wrap, IsEndOfLine::Yes };
229 }
230 // Either we managed to break with no overflow or the line is empty.
231 auto trailingPartialContent = Result::PartialTrailingContent { overflowingContent.breakingPosition->runIndex, trailingContent->partialRun };
232 return Result { Result::Action::Break, IsEndOfLine::Yes, trailingPartialContent };
233 };
234 if (auto result = tryBreakingContentWithText())
235 return *result;
236 } else if (continuousContent.runs().size() > 1) {
237 // FIXME: Add support for various content.
238 auto& runs = continuousContent.runs();
239 for (size_t i = 0; i < runs.size(); ++i) {
240 if (runs[i].inlineItem.isBox()) {
241 overflowingRunIndex = i;
242 break;
243 }
244 }
245 }
246
247 // If we are not allowed to break this overflowing content, we still need to decide whether keep it or wrap it to the next line.
248 if (!lineStatus.hasContent)
249 return { Result::Action::Keep, IsEndOfLine::No };
250 // Now either wrap this content over to the next line or revert back to an earlier wrapping opportunity, or not wrap at all.
251 auto shouldWrapUnbreakableContentToNextLine = [&] {
252 // The individual runs in this continuous content don't break, let's check if we are allowed to wrap this content to next line (e.g. pre would prevent us from wrapping).
253 // Parent style drives the wrapping behavior here.
254 // e.g. <div style="white-space: nowrap">some text<div style="display: inline-block; white-space: pre-wrap"></div></div>.
255 // While the inline-block has pre-wrap which allows wrapping, the content lives in a nowrap context.
256 return isWrappingAllowed(continuousContent.runs()[overflowingRunIndex].inlineItem);
257 };
258 if (shouldWrapUnbreakableContentToNextLine())
259 return { Result::Action::Wrap, IsEndOfLine::Yes };
260 if (lineStatus.hasWrapOpportunityAtPreviousPosition)
261 return { Result::Action::RevertToLastWrapOpportunity, IsEndOfLine::Yes };
262 return { Result::Action::Keep, IsEndOfLine::No };
263}

265OverflowingTextContent InlineContentBreaker::processOverflowingContentWithText(const ContinuousContent& continuousContent, const LineStatus& lineStatus) const
266{
267 auto& runs = continuousContent.runs();
268 ASSERT(!runs.isEmpty());
269
270 auto isBreakableRun = [] (auto& run) {
271 ASSERT(run.inlineItem.isText() || run.inlineItem.isInlineBoxStart() || run.inlineItem.isInlineBoxEnd() || run.inlineItem.layoutBox().isImage());
272 if (!run.inlineItem.isText()) {
273 // Can't break horizontal spacing -> e.g. <span style="padding-right: 100px;">textcontent</span>, if the [inline box end] is the overflown inline item
274 // we need to check if there's another inline item beyond the [inline box end] to split.
275 return false;
276 }
277 // Check if this text run needs to stay on the current line.
278 return isWrappingAllowed(run.inlineItem);
279 };
280
281 auto findTrailingRunIndex = [&] (auto breakableRunIndex) -> std::optional<size_t> {
282 // When the breaking position is at the beginning of the run, the trailing run is the previous one.
283 if (!breakableRunIndex)
284 return { };
285 // Try not break content at inline box boundary
286 // e.g. <span>fits</span><span>overflows</span>
287 // when the text "overflows" completely overflows, let's break the content right before the '<span>'.
288 // FIXME: Add support for subsequent empty inline boxes e.g.
289 auto trailingCandidateIndex = breakableRunIndex - 1;
290 auto isAtInlineBox = runs[trailingCandidateIndex].inlineItem.isInlineBoxStart();
291 return !isAtInlineBox ? trailingCandidateIndex : trailingCandidateIndex ? std::make_optional(trailingCandidateIndex - 1) : std::nullopt;
292 };
293
294 // Check where the overflow occurs and use the corresponding style to figure out the breaking behaviour.
295 // <span style="word-break: normal">first</span><span style="word-break: break-all">second</span><span style="word-break: normal">third</span>
296
297 // First find the overflowing run.
298 auto accumulatedContentWidth = InlineLayoutUnit { };
299 auto overflowingRunIndex = runs.size();
300 for (size_t index = 0; index < runs.size(); ++index) {
301 auto runLogicalWidth = runs[index].logicalWidth;
302 if (accumulatedContentWidth + runLogicalWidth > lineStatus.availableWidth) {
303 overflowingRunIndex = index;
304 break;
305 }
306 accumulatedContentWidth += runLogicalWidth;
307 }
308 // We have to have an overflowing run.
309 RELEASE_ASSERT(overflowingRunIndex < runs.size());
310
311 auto tryBreakingOverflowingRun = [&]() -> std::optional<OverflowingTextContent::BreakingPosition> {
312 auto overflowingRun = runs[overflowingRunIndex];
313 if (!isBreakableRun(overflowingRun))
314 return { };
315 if (auto partialRun = tryBreakingTextRun(overflowingRun, lineStatus.contentLogicalRight + accumulatedContentWidth, std::max(0.0f, lineStatus.availableWidth - accumulatedContentWidth), lineStatus.hasWrapOpportunityAtPreviousPosition)) {
316 if (partialRun->length)
317 return OverflowingTextContent::BreakingPosition { overflowingRunIndex, OverflowingTextContent::BreakingPosition::TrailingContent { false, partialRun } };
318 // When the breaking position is at the beginning of the run, the trailing run is the previous one.
319 if (auto trailingRunIndex = findTrailingRunIndex(overflowingRunIndex))
320 return OverflowingTextContent::BreakingPosition { *trailingRunIndex, OverflowingTextContent::BreakingPosition::TrailingContent { } };
321 // Sometimes we can't accommodate even the very first character.
322 // Note that this is different from when there's no breakable run in this set.
323 return OverflowingTextContent::BreakingPosition { };
324 }
325 return { };
326 };
327 // Check if we actually break this run.
328 if (auto breakingPosition = tryBreakingOverflowingRun())
329 return { overflowingRunIndex, breakingPosition };
330
331 auto tryBreakingPreviousNonOverflowingRuns = [&]() -> std::optional<OverflowingTextContent::BreakingPosition> {
332 auto previousContentWidth = accumulatedContentWidth;
333 for (auto index = overflowingRunIndex; index--;) {
334 auto& run = runs[index];
335 previousContentWidth -= run.logicalWidth;
336 if (!isBreakableRun(run))
337 continue;
338 ASSERT(run.inlineItem.isText());
339 if (auto partialRun = tryBreakingTextRun(run, lineStatus.contentLogicalRight + previousContentWidth, { }, lineStatus.hasWrapOpportunityAtPreviousPosition)) {
340 // We know this run fits, so if breaking is allowed on the run, it should return a non-empty left-side
341 // since it's either at hyphen position or the entire run is returned.
342 ASSERT(partialRun->length);
343 auto runIsFullyAccommodated = partialRun->length == downcast<InlineTextItem>(run.inlineItem).length();
344 return OverflowingTextContent::BreakingPosition { index, OverflowingTextContent::BreakingPosition::TrailingContent { false, runIsFullyAccommodated ? std::nullopt : partialRun } };
345 }
346 }
347 return { };
348 };
349 // We did not manage to break the run that actually overflows the line.
350 // Let's try to find a previous breaking position starting from the overflowing run. It surely fits.
351 if (auto breakingPosition = tryBreakingPreviousNonOverflowingRuns())
352 return { overflowingRunIndex, breakingPosition };
353
354 auto tryBreakingNextOverflowingRuns = [&]() -> std::optional<OverflowingTextContent::BreakingPosition> {
355 auto nextContentWidth = accumulatedContentWidth + runs[overflowingRunIndex].logicalWidth;
356 for (auto index = overflowingRunIndex + 1; index < runs.size(); ++index) {
357 auto& run = runs[index];
358 if (isBreakableRun(run)) {
359 ASSERT(run.inlineItem.isText());
360 // We know that this run does not fit the available space. If we can break it at any position, let's just use the start of the run.
361 if (wordBreakBehavior(run.inlineItem.style(), lineStatus.hasWrapOpportunityAtPreviousPosition) == WordBreakRule::AtArbitraryPosition) {
362 // We must be on an inline box boundary. Let's go back to the run in front of the inline box start run.
363 // e.g. <span>unbreakable_and_overflow<span style="word-break: break-all">breakable</span>
364 // We are at "breakable", <span> is at index - 1 and the trailing run is at index - 2.
365 ASSERT(runs[index - 1].inlineItem.isInlineBoxStart());
366 auto trailingRunIndex = findTrailingRunIndex(index);
367 if (!trailingRunIndex) {
368 // This continuous content did not fit from the get-go. No trailing run.
369 return OverflowingTextContent::BreakingPosition { };
370 }
371 // At worst we are back to the overflowing run, like in the example above.
372 ASSERT(*trailingRunIndex >= overflowingRunIndex);
373 return OverflowingTextContent::BreakingPosition { *trailingRunIndex, OverflowingTextContent::BreakingPosition::TrailingContent { true } };
374 }
375 if (auto partialRun = tryBreakingTextRun(run, lineStatus.contentLogicalRight + nextContentWidth, { }, lineStatus.hasWrapOpportunityAtPreviousPosition)) {
376 ASSERT(partialRun->length);
377 // We managed to break this text run mid content. It has to be a hyphen break.
378 return OverflowingTextContent::BreakingPosition { index, OverflowingTextContent::BreakingPosition::TrailingContent { true, partialRun } };
379 }
380 }
381 nextContentWidth += run.logicalWidth;
382 }
383 return { };
384 };
385 // At this point we know that there's no breakable run all the way to the overflowing run.
386 // Now we need to check if any run after the overflowing content can break.
387 // e.g. <span>this_content_overflows_but_not_breakable<span><span style="word-break: break-all">but_this_is_breakable</span>
388 if (auto breakingPosition = tryBreakingNextOverflowingRuns())
389 return { overflowingRunIndex, breakingPosition };
390
391 // Give up, there's no breakable run in here.
392 return { overflowingRunIndex };
393}

395OptionSet<InlineContentBreaker::WordBreakRule> InlineContentBreaker::wordBreakBehavior(const RenderStyle& style, bool hasWrapOpportunityAtPreviousPosition) const
396{
397 // Disregard any prohibition against line breaks mandated by the word-break property.
398 // The different wrapping opportunities must not be prioritized. Hyphenation is not applied.
399 if (style.lineBreak() == LineBreak::Anywhere)
400 return { WordBreakRule::AtArbitraryPosition };
401 // Breaking is allowed within “words”.
402 if (style.wordBreak() == WordBreak::BreakAll)
403 return { WordBreakRule::AtArbitraryPosition };
404 // Breaking is forbidden within “words”.
405 if (style.wordBreak() == WordBreak::KeepAll)
406 return { };
407
408 auto breakRules = OptionSet<WordBreakRule> { };
409 auto hyphenationIsAllowed = !n_hyphenationIsDisabled && style.hyphens() == Hyphens::Auto && canHyphenate(style.computedLocale());
410 if (hyphenationIsAllowed)
411 breakRules.add({ WordBreakRule::AtHyphenationOpportunities });
412 // For compatibility with legacy content, the word-break property also supports a deprecated break-word keyword.
413 // When specified, this has the same effect as word-break: normal and overflow-wrap: anywhere, regardless of the actual value of the overflow-wrap property.
414 if (style.wordBreak() == WordBreak::BreakWord && !hasWrapOpportunityAtPreviousPosition) {
415 breakRules.add({ WordBreakRule::AtArbitraryPosition });
416 return breakRules;
417 }
418 // OverflowWrap::Break: An otherwise unbreakable sequence of characters may be broken at an arbitrary point if there are no otherwise-acceptable break points in the line.
419 if (style.overflowWrap() == OverflowWrap::Break && !hasWrapOpportunityAtPreviousPosition) {
420 breakRules.add({ WordBreakRule::AtArbitraryPosition });
421 return breakRules;
422 }
423 return breakRules;
424}

426std::optional<InlineContentBreaker::PartialRun> InlineContentBreaker::tryBreakingTextRun(const ContinuousContent::Run& overflowingRun, InlineLayoutUnit logicalLeft, std::optional<InlineLayoutUnit> availableWidth, bool hasWrapOpportunityAtPreviousPosition) const
427{
428 ASSERT(overflowingRun.inlineItem.isText());
429 auto& inlineTextItem = downcast<InlineTextItem>(overflowingRun.inlineItem);
430 auto& style = inlineTextItem.style();
431 auto availableSpaceIsInfinite = !availableWidth.has_value();
432
433 auto breakRules = wordBreakBehavior(style, hasWrapOpportunityAtPreviousPosition);
434 if (breakRules.isEmpty())
435 return { };
436
437 if (breakRules.contains(WordBreakRule::AtHyphenationOpportunities)) {
438 auto tryBreakingAtHyphenationOpportunity = [&]() -> std::optional<PartialRun> {
439 // Find the hyphen position as follows:
440 // 1. Split the text by taking the hyphen width into account
441 // 2. Find the last hyphen position before the split position
442 if (!availableSpaceIsInfinite && !*availableWidth) {
443 // We won't be able to find hyphen location when there's no available space.
444 return { };
445 }
446 auto runLength = inlineTextItem.length();
447 unsigned limitBefore = style.hyphenationLimitBefore() == RenderStyle::initialHyphenationLimitBefore() ? 0 : style.hyphenationLimitBefore();
448 unsigned limitAfter = style.hyphenationLimitAfter() == RenderStyle::initialHyphenationLimitAfter() ? 0 : style.hyphenationLimitAfter();
449 // Check if this run can accommodate the before/after limits at all before start measuring text.
450 if (limitBefore >= runLength || limitAfter >= runLength || limitBefore + limitAfter > runLength)
451 return { };
452
453 unsigned leftSideLength = runLength;
454 auto& fontCascade = style.fontCascade();
455 auto hyphenWidth = InlineLayoutUnit { fontCascade.width(TextRun { StringView { style.hyphenString() } }) };
456 if (!availableSpaceIsInfinite) {
457 auto availableWidthExcludingHyphen = *availableWidth - hyphenWidth;
458 if (availableWidthExcludingHyphen <= 0 || !enoughWidthForHyphenation(availableWidthExcludingHyphen, fontCascade.pixelSize()))
459 return { };
460 leftSideLength = TextUtil::split(inlineTextItem, overflowingRun.logicalWidth, availableWidthExcludingHyphen, logicalLeft).length;
461 }
462 if (leftSideLength < limitBefore)
463 return { };
464 // Adjust before index to accommodate the limit-after value (it's the last potential hyphen location in this run).
465 auto hyphenBefore = std::min(leftSideLength, runLength - limitAfter) + 1;
466 unsigned hyphenLocation = lastHyphenLocation(StringView(inlineTextItem.inlineTextBox().content()).substring(inlineTextItem.start(), inlineTextItem.length()), hyphenBefore, style.computedLocale());
467 if (!hyphenLocation || hyphenLocation < limitBefore)
468 return { };
469 // hyphenLocation is relative to the start of this InlineItemText.
470 ASSERT(inlineTextItem.start() + hyphenLocation < inlineTextItem.end());
471 auto trailingPartialRunWidthWithHyphen = TextUtil::width(inlineTextItem, inlineTextItem.start(), inlineTextItem.start() + hyphenLocation, logicalLeft);
472 return PartialRun { hyphenLocation, trailingPartialRunWidthWithHyphen, hyphenWidth };
473 };
474 if (auto partialRun = tryBreakingAtHyphenationOpportunity())
475 return partialRun;
476 }
477
478 if (breakRules.contains(WordBreakRule::AtArbitraryPosition)) {
479 auto tryBreakingAtArbitraryPosition = [&]() -> PartialRun {
480 if (!inlineTextItem.length()) {
481 // Empty text runs may be breakable based on style, but in practice we can't really split them any further.
482 return { };
483 }
484 if (availableSpaceIsInfinite) {
485 // When the run can be split at arbitrary position let's just return the entire run when it is intended to fit on the line.
486 ASSERT(inlineTextItem.length());
487 auto trailingPartialRunWidth = TextUtil::width(inlineTextItem, logicalLeft);
488 return { inlineTextItem.length(), trailingPartialRunWidth };
489 }
490 if (!*availableWidth) {
491 // Fast path for cases when there's no room at all. The content is breakable but we don't have space for it.
492 return { };
493 }
494 auto splitData = TextUtil::split(inlineTextItem, overflowingRun.logicalWidth, *availableWidth, logicalLeft);
495 return { splitData.length, splitData.logicalWidth };
496 };
497 // With arbitrary breaking there's always a valid breaking position (even if it is before the first position).
498 return tryBreakingAtArbitraryPosition();
499 }
500 return { };
501}

503void InlineContentBreaker::ContinuousContent::append(const InlineItem& inlineItem, InlineLayoutUnit logicalWidth, std::optional<InlineLayoutUnit> collapsibleWidth)
504{
505 m_runs.append({ inlineItem, logicalWidth });
506 m_logicalWidth = clampTo<InlineLayoutUnit>(m_logicalWidth + logicalWidth);
507 if (!collapsibleWidth) {
508 m_collapsibleLogicalWidth = { };
509 return;
510 }
511 if (*collapsibleWidth == logicalWidth) {
512 // Fully collapsible run.
513 m_collapsibleLogicalWidth += logicalWidth;
514 ASSERT(m_collapsibleLogicalWidth <= m_logicalWidth);
515 return;
516 }
517 // Partially collapsible run.
518 m_collapsibleLogicalWidth = *collapsibleWidth;
519 ASSERT(m_collapsibleLogicalWidth <= m_logicalWidth);
520}
