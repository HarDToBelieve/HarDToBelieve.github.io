64static inline const Box* nextInlineLevelBoxToLayout(const Box& layoutBox, const ContainerBox& stayWithin)
65{
66 // Atomic inline-level boxes and floats are opaque boxes meaning that they are
67 // responsible for their own content (do not need to descend into their subtrees).
68 // Only inline boxes may have relevant descendant content.
69 if (layoutBox.isInlineBox()) {
70 if (is<ContainerBox>(layoutBox) && downcast<ContainerBox>(layoutBox).hasInFlowOrFloatingChild()) {
71 // Anonymous inline text boxes/line breaks can't have descendant content by definition.
72 ASSERT(!layoutBox.isInlineTextBox() && !layoutBox.isLineBreakBox());
73 return downcast<ContainerBox>(layoutBox).firstInFlowOrFloatingChild();
74 }
75 }
76
77 for (auto* nextInPreOrder = &layoutBox; nextInPreOrder && nextInPreOrder != &stayWithin; nextInPreOrder = &nextInPreOrder->parent()) {
78 if (auto* nextSibling = nextInPreOrder->nextInFlowOrFloatingSibling())
79 return nextSibling;
80 }
81 return nullptr;
82}

84void InlineFormattingContext::layoutInFlowContent(InvalidationState& invalidationState, const ConstraintsForInFlowContent& constraints)
85{
86 LOG_WITH_STREAM(FormattingContextLayout, stream << "[Start] -> inline formatting context -> formatting root(" << &root() << ")");
87 ASSERT(root().hasInFlowOrFloatingChild());
88
89 invalidateFormattingState(invalidationState);
90 auto* layoutBox = root().firstInFlowOrFloatingChild();
91 // 1. Visit each inline box and partially compute their geometry (margins, padding and borders).
92 // 2. Collect the inline items (flatten the the layout tree) and place them on lines in bidirectional order.
93 while (layoutBox) {
94 ASSERT(layoutBox->isInlineLevelBox() || layoutBox->isFloatingPositioned());
95
96 if (layoutBox->isAtomicInlineLevelBox() || layoutBox->isFloatingPositioned()) {
97 // Inline-blocks, inline-tables and replaced elements (img, video) can be sized but not yet positioned.
98 if (is<ContainerBox>(layoutBox) && layoutBox->establishesFormattingContext()) {
99 ASSERT(layoutBox->isInlineBlockBox() || layoutBox->isInlineTableBox() || layoutBox->isFloatingPositioned());
100 auto& formattingRoot = downcast<ContainerBox>(*layoutBox);
101 computeBorderAndPadding(formattingRoot, constraints.horizontal());
102 computeWidthAndMargin(formattingRoot, constraints.horizontal());
103
104 if (formattingRoot.hasChild()) {
105 auto formattingContext = LayoutContext::createFormattingContext(formattingRoot, layoutState());
106 if (formattingRoot.hasInFlowOrFloatingChild())
107 formattingContext->layoutInFlowContent(invalidationState, formattingGeometry().constraintsForInFlowContent(formattingRoot));
108 computeHeightAndMargin(formattingRoot, constraints.horizontal());
109 formattingContext->layoutOutOfFlowContent(invalidationState, formattingGeometry().constraintsForOutOfFlowContent(formattingRoot));
110 } else
111 computeHeightAndMargin(formattingRoot, constraints.horizontal());
112 } else {
113 // Replaced and other type of leaf atomic inline boxes.
114 computeBorderAndPadding(*layoutBox, constraints.horizontal());
115 computeWidthAndMargin(*layoutBox, constraints.horizontal());
116 computeHeightAndMargin(*layoutBox, constraints.horizontal());
117 }
118 } else if (layoutBox->isLineBreakBox()) {
119 auto& boxGeometry = formattingState().boxGeometry(*layoutBox);
120 boxGeometry.setHorizontalMargin({ });
121 boxGeometry.setBorder({ });
122 boxGeometry.setPadding({ });
123 boxGeometry.setContentBoxWidth({ });
124 boxGeometry.setVerticalMargin({ });
125 } else if (layoutBox->isInlineBox()) {
126 // Text wrapper boxes (anonymous inline level boxes) don't have box geometries (they only generate runs).
127 if (!layoutBox->isInlineTextBox()) {
128 // Inline boxes (<span>) can't get sized/positioned yet. At this point we can only compute their margins, borders and padding.
129 computeBorderAndPadding(*layoutBox, constraints.horizontal());
130 computeHorizontalMargin(*layoutBox, constraints.horizontal());
131 formattingState().boxGeometry(*layoutBox).setVerticalMargin({ });
132 }
133 } else
134 ASSERT_NOT_REACHED();
135
136 layoutBox = nextInlineLevelBoxToLayout(*layoutBox, root());
137 }
138
139 collectInlineContentIfNeeded();
140
141 auto& inlineItems = formattingState().inlineItems();
142 lineLayout(inlineItems, { 0, inlineItems.size() }, constraints);
143 LOG_WITH_STREAM(FormattingContextLayout, stream << "[End] -> inline formatting context -> formatting root(" << &root() << ")");
144}

154LayoutUnit InlineFormattingContext::usedContentHeight() const
155{
156 // 10.6.7 'Auto' heights for block formatting context roots
157
158 // If it only has inline-level children, the height is the distance between the top of the topmost line box and the bottom of the bottommost line box.
159
160 // In addition, if the element has any floating descendants whose bottom margin edge is below the element's bottom content edge,
161 // then the height is increased to include those edges. Only floats that participate in this block formatting context are taken
162 // into account, e.g., floats inside absolutely positioned descendants or other floats are not.
163 auto& lines = formattingState().lines();
164 // Even empty content generates a line.
165 ASSERT(!lines.isEmpty());
166 auto top = LayoutUnit { lines.first().lineBoxLogicalRect().top() };
167 auto bottom = LayoutUnit { lines.last().lineBoxLogicalRect().bottom() + formattingState().clearGapAfterLastLine() };
168
169 auto floatingContext = FloatingContext { *this, formattingState().floatingState() };
170 if (auto floatBottom = floatingContext.bottom()) {
171 bottom = std::max(*floatBottom, bottom);
172 top = std::min(*floatingContext.top(), top);
173 }
174 return bottom - top;
175}

177void InlineFormattingContext::lineLayout(InlineItems& inlineItems, LineBuilder::InlineItemRange needsLayoutRange, const ConstraintsForInFlowContent& constraints)
178{
179 auto& formattingState = this->formattingState();
180 formattingState.lineRuns().reserveInitialCapacity(formattingState.inlineItems().size());
181 InlineLayoutUnit lineLogicalTop = constraints.logicalTop();
182 struct PreviousLine {
183 LineBuilder::InlineItemRange range;
184 size_t overflowContentLength { 0 };
185 std::optional<InlineLayoutUnit> overflowLogicalWidth;
186 };
187 std::optional<PreviousLine> previousLine;
188 auto& floatingState = formattingState.floatingState();
189 auto floatingContext = FloatingContext { *this, floatingState };
190 auto isFirstLine = formattingState.lines().isEmpty();
191
192 auto lineBuilder = LineBuilder { *this, floatingState, constraints.horizontal(), inlineItems };
193 while (!needsLayoutRange.isEmpty()) {
194 // Turn previous line's overflow content length into the next line's leading content partial length.
195 // "sp[<-line break->]lit_content" -> overflow length: 11 -> leading partial content length: 11.
196 auto partialLeadingContentLength = previousLine ? previousLine->overflowContentLength : 0;
197 auto leadingLogicalWidth = previousLine ? previousLine->overflowLogicalWidth : std::nullopt;
198 auto initialLineHeight = [&]() -> InlineLayoutUnit {
199 if (layoutState().inStandardsMode())
200 return root().style().computedLineHeight();
201 return formattingQuirks().initialLineHeight();
202 }();
203 auto initialLineConstraints = InlineRect { lineLogicalTop, constraints.horizontal().logicalLeft, constraints.horizontal().logicalWidth, initialLineHeight };
204 auto lineContent = lineBuilder.layoutInlineContent(needsLayoutRange, partialLeadingContentLength, leadingLogicalWidth, initialLineConstraints, isFirstLine);
205 auto lineLogicalRect = computeGeometryForLineContent(lineContent, constraints.horizontal());
206
207 auto lineContentRange = lineContent.inlineItemRange;
208 if (!lineContentRange.isEmpty()) {
209 ASSERT(needsLayoutRange.start < lineContentRange.end);
210 isFirstLine = false;
211 lineLogicalTop = formattingGeometry().logicalTopForNextLine(lineContent, lineLogicalRect.bottom(), floatingContext);
212 if (lineContent.isLastLineWithInlineContent) {
213 // The final content height of this inline formatting context should include the cleared floats as well.
214 formattingState.setClearGapAfterLastLine(lineLogicalTop - lineLogicalRect.bottom());
215 }
216 // When the trailing content is partial, we need to reuse the last InlineTextItem.
217 auto lastInlineItemNeedsPartialLayout = lineContent.partialTrailingContentLength;
218 if (lastInlineItemNeedsPartialLayout) {
219 auto lineLayoutHasAdvanced = !previousLine
220 || lineContentRange.end > previousLine->range.end
221 || (previousLine->overflowContentLength && previousLine->overflowContentLength > lineContent.partialTrailingContentLength);
222 if (!lineLayoutHasAdvanced) {
223 ASSERT_NOT_REACHED();
224 // Move over to the next run if we are stuck on this partial content (when the overflow content length remains the same).
225 // We certainly lose some content, but we would be busy looping otherwise.
226 lastInlineItemNeedsPartialLayout = false;
227 }
228 }
229 needsLayoutRange.start = lastInlineItemNeedsPartialLayout ? lineContentRange.end - 1 : lineContentRange.end;
230 previousLine = PreviousLine { lineContentRange, lineContent.partialTrailingContentLength, lineContent.overflowLogicalWidth };
231 continue;
232 }
233 // Floats prevented us placing any content on the line.
234 ASSERT(lineContent.runs.isEmpty());
235 ASSERT(lineContent.hasIntrusiveFloat);
236 // Move the next line below the intrusive float(s).
237 auto logicalTopCandidateForNextLine = [&] {
238 auto lineBottomWithNoInlineContent = LayoutUnit { std::max(lineLogicalRect.bottom(), initialLineConstraints.bottom()) };
239 auto floatConstraints = floatingContext.constraints(toLayoutUnit(lineLogicalTop), lineBottomWithNoInlineContent);
240 ASSERT(floatConstraints.left || floatConstraints.right);
241 if (floatConstraints.left && floatConstraints.right) {
242 // In case of left and right constraints, we need to pick the one that's closer to the current line.
243 return std::min(floatConstraints.left->y, floatConstraints.right->y);
244 }
245 if (floatConstraints.left)
246 return floatConstraints.left->y;
247 if (floatConstraints.right)
248 return floatConstraints.right->y;
249 ASSERT_NOT_REACHED();
250 return lineBottomWithNoInlineContent;
251 };
252 lineLogicalTop = logicalTopCandidateForNextLine();
253 }
254}

256IntrinsicWidthConstraints InlineFormattingContext::computedIntrinsicWidthConstraints()
257{
258 auto& layoutState = this->layoutState();
259 ASSERT(!formattingState().intrinsicWidthConstraints());
260
261 if (!root().hasInFlowOrFloatingChild()) {
262 auto constraints = formattingGeometry().constrainByMinMaxWidth(root(), { });
263 formattingState().setIntrinsicWidthConstraints(constraints);
264 return constraints;
265 }
266
267 Vector<const Box*> formattingContextRootList;
268 auto horizontalConstraints = HorizontalConstraints { 0_lu, 0_lu };
269 auto* layoutBox = root().firstInFlowOrFloatingChild();
270 // In order to compute the max/min widths, we need to compute margins, borders and padding for certain inline boxes first.
271 while (layoutBox) {
272 if (layoutBox->isInlineTextBox()) {
273 layoutBox = nextInlineLevelBoxToLayout(*layoutBox, root());
274 continue;
275 }
276 if (layoutBox->isReplacedBox()) {
277 computeBorderAndPadding(*layoutBox, horizontalConstraints);
278 computeWidthAndMargin(*layoutBox, horizontalConstraints);
279 } else if (layoutBox->isFloatingPositioned() || layoutBox->isAtomicInlineLevelBox()) {
280 ASSERT(layoutBox->establishesFormattingContext());
281 formattingContextRootList.append(layoutBox);
282
283 computeBorderAndPadding(*layoutBox, horizontalConstraints);
284 computeHorizontalMargin(*layoutBox, horizontalConstraints);
285 computeIntrinsicWidthForFormattingRoot(*layoutBox);
286 } else if (layoutBox->isInlineBox()) {
287 computeBorderAndPadding(*layoutBox, horizontalConstraints);
288 computeHorizontalMargin(*layoutBox, horizontalConstraints);
289 } else
290 ASSERT_NOT_REACHED();
291 layoutBox = nextInlineLevelBoxToLayout(*layoutBox, root());
292 }
293
294 collectInlineContentIfNeeded();
295
296 auto maximumLineWidth = [&](auto availableWidth) {
297 // Switch to the min/max formatting root width values before formatting the lines.
298 for (auto* formattingRoot : formattingContextRootList) {
299 auto intrinsicWidths = layoutState.formattingStateForBox(*formattingRoot).intrinsicWidthConstraintsForBox(*formattingRoot);
300 auto& boxGeometry = formattingState().boxGeometry(*formattingRoot);
301 auto contentWidth = (availableWidth ? intrinsicWidths->maximum : intrinsicWidths->minimum) - boxGeometry.horizontalMarginBorderAndPadding();
302 boxGeometry.setContentBoxWidth(contentWidth);
303 }
304 return computedIntrinsicWidthForConstraint(availableWidth);
305 };
306
307 auto minimumContentWidth = ceiledLayoutUnit(maximumLineWidth(0));
308 auto maximumContentWidth = ceiledLayoutUnit(maximumLineWidth(maxInlineLayoutUnit()));
309 auto constraints = formattingGeometry().constrainByMinMaxWidth(root(), { minimumContentWidth, maximumContentWidth });
310 formattingState().setIntrinsicWidthConstraints(constraints);
311 return constraints;
312}

314InlineLayoutUnit InlineFormattingContext::computedIntrinsicWidthForConstraint(InlineLayoutUnit availableWidth) const
315{
316 auto& inlineItems = formattingState().inlineItems();
317 auto lineBuilder = LineBuilder { *this, inlineItems };
318 auto layoutRange = LineBuilder::InlineItemRange { 0 , inlineItems.size() };
319 auto maximumLineWidth = InlineLayoutUnit { };
320 auto maximumFloatWidth = LayoutUnit { };
321 while (!layoutRange.isEmpty()) {
322 auto intrinsicContent = lineBuilder.computedIntrinsicWidth(layoutRange, availableWidth);
323 layoutRange.start = intrinsicContent.inlineItemRange.end;
324 maximumLineWidth = std::max(maximumLineWidth, intrinsicContent.logicalWidth);
325 // FIXME: Add support for clear.
326 for (auto* floatBox : intrinsicContent.floats)
327 maximumFloatWidth += geometryForBox(*floatBox).marginBoxWidth();
328 }
329 return maximumLineWidth + maximumFloatWidth;
330}

332void InlineFormattingContext::computeIntrinsicWidthForFormattingRoot(const Box& formattingRoot)
333{
334 ASSERT(formattingRoot.establishesFormattingContext());
335 auto constraints = IntrinsicWidthConstraints { };
336 if (auto fixedWidth = formattingGeometry().fixedValue(formattingRoot.style().logicalWidth()))
337 constraints = { *fixedWidth, *fixedWidth };
338 else {
339 auto hasInflowOrFloatingContent = is<ContainerBox>(formattingRoot) && downcast<ContainerBox>(formattingRoot).hasInFlowOrFloatingChild();
340 // The intrinsic sizes of the size containment box are determined as if the element had no content.
341 auto shouldIgnoreChildContent = formattingRoot.isSizeContainmentBox();
342 if (hasInflowOrFloatingContent && !shouldIgnoreChildContent)
343 constraints = LayoutContext::createFormattingContext(downcast<ContainerBox>(formattingRoot), layoutState())->computedIntrinsicWidthConstraints();
344 }
345 constraints = formattingGeometry().constrainByMinMaxWidth(formattingRoot, constraints);
346 constraints.expand(geometryForBox(formattingRoot).horizontalMarginBorderAndPadding());
347 formattingState().setIntrinsicWidthConstraintsForBox(formattingRoot, constraints);
348}

356void InlineFormattingContext::computeWidthAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints)
357{
358 auto compute = [&](std::optional<LayoutUnit> usedWidth) {
359 if (layoutBox.isFloatingPositioned())
360 return formattingGeometry().floatingContentWidthAndMargin(layoutBox, horizontalConstraints, { usedWidth, { } });
361 if (layoutBox.isInlineBlockBox())
362 return formattingGeometry().inlineBlockContentWidthAndMargin(layoutBox, horizontalConstraints, { usedWidth, { } });
363 if (layoutBox.isReplacedBox())
364 return formattingGeometry().inlineReplacedContentWidthAndMargin(downcast<ReplacedBox>(layoutBox), horizontalConstraints, { }, { usedWidth, { } });
365 ASSERT_NOT_REACHED();
366 return ContentWidthAndMargin { };
367 };
368
369 auto contentWidthAndMargin = compute({ });
370
371 auto availableWidth = horizontalConstraints.logicalWidth;
372 if (auto maxWidth = formattingGeometry().computedMaxWidth(layoutBox, availableWidth)) {
373 auto maxWidthAndMargin = compute(maxWidth);
374 if (contentWidthAndMargin.contentWidth > maxWidthAndMargin.contentWidth)
375 contentWidthAndMargin = maxWidthAndMargin;
376 }
377
378 auto minWidth = formattingGeometry().computedMinWidth(layoutBox, availableWidth).value_or(0);
379 auto minWidthAndMargin = compute(minWidth);
380 if (contentWidthAndMargin.contentWidth < minWidthAndMargin.contentWidth)
381 contentWidthAndMargin = minWidthAndMargin;
382
383 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
384 boxGeometry.setContentBoxWidth(contentWidthAndMargin.contentWidth);
385 boxGeometry.setHorizontalMargin({ contentWidthAndMargin.usedMargin.start, contentWidthAndMargin.usedMargin.end });
386}

388void InlineFormattingContext::computeHeightAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints)
389{
390 auto compute = [&](std::optional<LayoutUnit> usedHeight) {
391 if (layoutBox.isFloatingPositioned())
392 return formattingGeometry().floatingContentHeightAndMargin(layoutBox, horizontalConstraints, { usedHeight });
393 if (layoutBox.isInlineBlockBox())
394 return formattingGeometry().inlineBlockContentHeightAndMargin(layoutBox, horizontalConstraints, { usedHeight });
395 if (layoutBox.isReplacedBox())
396 return formattingGeometry().inlineReplacedContentHeightAndMargin(downcast<ReplacedBox>(layoutBox), horizontalConstraints, { }, { usedHeight });
397 ASSERT_NOT_REACHED();
398 return ContentHeightAndMargin { };
399 };
400
401 auto contentHeightAndMargin = compute({ });
402 if (auto maxHeight = formattingGeometry().computedMaxHeight(layoutBox)) {
403 auto maxHeightAndMargin = compute(maxHeight);
404 if (contentHeightAndMargin.contentHeight > maxHeightAndMargin.contentHeight)
405 contentHeightAndMargin = maxHeightAndMargin;
406 }
407
408 if (auto minHeight = formattingGeometry().computedMinHeight(layoutBox)) {
409 auto minHeightAndMargin = compute(minHeight);
410 if (contentHeightAndMargin.contentHeight < minHeightAndMargin.contentHeight)
411 contentHeightAndMargin = minHeightAndMargin;
412 }
413 auto& boxGeometry = formattingState().boxGeometry(layoutBox);
414 boxGeometry.setContentBoxHeight(contentHeightAndMargin.contentHeight);
415 boxGeometry.setVerticalMargin({ contentHeightAndMargin.nonCollapsedMargin.before, contentHeightAndMargin.nonCollapsedMargin.after });
416}

418void InlineFormattingContext::collectInlineContentIfNeeded()
419{
420 auto& formattingState = this->formattingState();
421 if (!formattingState.inlineItems().isEmpty())
422 return;
423 // Traverse the tree and create inline items out of inline boxes and leaf nodes. This essentially turns the tree inline structure into a flat one.
424 // <span>text<span></span><img></span> -> [InlineBoxStart][InlineLevelBox][InlineBoxStart][InlineBoxEnd][InlineLevelBox][InlineBoxEnd]
425 ASSERT(root().hasInFlowOrFloatingChild());
426 LayoutQueue layoutQueue;
427 layoutQueue.append(root().firstInFlowOrFloatingChild());
428 while (!layoutQueue.isEmpty()) {
429 while (true) {
430 auto& layoutBox = *layoutQueue.last();
431 auto isBoxWithInlineContent = layoutBox.isInlineBox() && !layoutBox.isInlineTextBox() && !layoutBox.isLineBreakBox();
432 if (!isBoxWithInlineContent)
433 break;
434 // This is the start of an inline box (e.g. <span>).
435 formattingState.addInlineItem({ layoutBox, InlineItem::Type::InlineBoxStart });
436 auto& inlineBoxWithInlineContent = downcast<ContainerBox>(layoutBox);
437 if (!inlineBoxWithInlineContent.hasInFlowOrFloatingChild())
438 break;
439 layoutQueue.append(inlineBoxWithInlineContent.firstInFlowOrFloatingChild());
440 }
441
442 while (!layoutQueue.isEmpty()) {
443 auto& layoutBox = *layoutQueue.takeLast();
444 if (is<LineBreakBox>(layoutBox)) {
445 auto& lineBreakBox = downcast<LineBreakBox>(layoutBox);
446 formattingState.addInlineItem({ layoutBox, lineBreakBox.isOptional() ? InlineItem::Type::WordBreakOpportunity : InlineItem::Type::HardLineBreak });
447 } else if (layoutBox.isFloatingPositioned())
448 formattingState.addInlineItem({ layoutBox, InlineItem::Type::Float });
449 else if (layoutBox.isAtomicInlineLevelBox())
450 formattingState.addInlineItem({ layoutBox, InlineItem::Type::Box });
451 else if (layoutBox.isInlineTextBox()) {
452 InlineTextItem::createAndAppendTextItems(formattingState.inlineItems(), downcast<InlineTextBox>(layoutBox));
453 } else if (layoutBox.isInlineBox())
454 formattingState.addInlineItem({ layoutBox, InlineItem::Type::InlineBoxEnd });
455 else
456 ASSERT_NOT_REACHED();
457
458 if (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling()) {
459 layoutQueue.append(nextSibling);
460 break;
461 }
462 }
463 }
464}

466InlineRect InlineFormattingContext::computeGeometryForLineContent(const LineBuilder::LineContent& lineContent, const HorizontalConstraints& horizontalConstraints)
467{
468 auto& formattingState = this->formattingState();
469 auto& formattingGeometry = this->formattingGeometry();
470
471 formattingState.addLineBox(formattingGeometry.lineBoxForLineContent(lineContent));
472 const auto& lineBox = formattingState.lineBoxes().last();
473 auto lineIndex = formattingState.lines().size();
474 auto& lineBoxLogicalRect = lineBox.logicalRect();
475 if (!lineBox.hasContent()) {
476 // Fast path for lines with no content e.g. <div><span></span><span></span></div> or <span><div></div></span> where we construct empty pre and post blocks.
477 ASSERT(!lineBox.contentLogicalWidth() && !lineBoxLogicalRect.height());
478 auto updateInlineBoxesGeometryIfApplicable = [&] {
479 if (!lineBox.hasInlineBox())
480 return;
481 Vector<const Box*> layoutBoxList;
482 // Collect the empty inline boxes that showed up first on this line.
483 // Note that an inline box end on an empty line does not make the inline box taller.
484 // (e.g. <div>text<span><br></span></div>) <- the <span> inline box is as tall as the line even though the </span> is after a <br> so technically is on the following (empty)line.
485 for (auto& lineRun : lineContent.runs) {
486 if (lineRun.isInlineBoxStart())
487 layoutBoxList.append(&lineRun.layoutBox());
488 }
489 for (auto* layoutBox : layoutBoxList) {
490 auto& boxGeometry = formattingState.boxGeometry(*layoutBox);
491 auto inlineBoxLogicalHeight = LayoutUnit::fromFloatCeil(lineBox.logicalBorderBoxForInlineBox(*layoutBox, boxGeometry).height());
492 boxGeometry.setContentBoxHeight(inlineBoxLogicalHeight);
493 boxGeometry.setContentBoxWidth({ });
494 boxGeometry.setLogicalTopLeft(toLayoutPoint(lineBoxLogicalRect.topLeft()));
495 }
496 };
497 updateInlineBoxesGeometryIfApplicable();
498 formattingState.addLine({ lineBoxLogicalRect, { { }, { } }, { }, { }, { } });
499 return lineBoxLogicalRect;
500 }
501
502 auto rootInlineBoxLogicalRect = lineBox.logicalRectForRootInlineBox();
503 auto enclosingTopAndBottom = InlineLineGeometry::EnclosingTopAndBottom { rootInlineBoxLogicalRect.top(), rootInlineBoxLogicalRect.bottom() };
504 HashSet<const Box*> inlineBoxStartSet;
505 HashSet<const Box*> inlineBoxEndSet;
506
507 auto constructLineRunsAndUpdateBoxGeometry = [&] {
508 // Create the inline runs on the current line. This is mostly text and atomic inline runs.
509 for (auto& lineRun : lineContent.runs) {
510 // FIXME: We should not need to construct a line run for <br>.
511 auto& layoutBox = lineRun.layoutBox();
512 if (lineRun.isText()) {
513 formattingState.addLineRun({ lineIndex, layoutBox, lineBox.logicalRectForTextRun(lineRun), lineRun.expansion(), lineRun.textContent() });
514 continue;
515 }
516 if (lineRun.isLineBreak()) {
517 if (layoutBox.isLineBreakBox()) {
518 // Only hard linebreaks have associated layout boxes.
519 auto lineBreakBoxRect = lineBox.logicalRectForLineBreakBox(layoutBox);
520 formattingState.addLineRun({ lineIndex, layoutBox, lineBreakBoxRect, lineRun.expansion(), { } });
521
522 auto& boxGeometry = formattingState.boxGeometry(layoutBox);
523 lineBreakBoxRect.moveBy(lineBoxLogicalRect.topLeft());
524 boxGeometry.setLogicalTopLeft(toLayoutPoint(lineBreakBoxRect.topLeft()));
525 boxGeometry.setContentBoxHeight(toLayoutUnit(lineBreakBoxRect.height()));
526 } else
527 formattingState.addLineRun({ lineIndex, layoutBox, lineBox.logicalRectForTextRun(lineRun), lineRun.expansion(), lineRun.textContent() });
528 continue;
529 }
530 if (lineRun.isBox()) {
531 ASSERT(layoutBox.isAtomicInlineLevelBox());
532 auto& boxGeometry = formattingState.boxGeometry(layoutBox);
533 auto logicalBorderBox = lineBox.logicalBorderBoxForAtomicInlineLevelBox(layoutBox, boxGeometry);
534 formattingState.addLineRun({ lineIndex, layoutBox, logicalBorderBox, lineRun.expansion(), { } });
535
536 auto borderBoxLogicalTopLeft = logicalBorderBox.topLeft();
537 // Note that inline boxes are relative to the line and their top position can be negative.
538 borderBoxLogicalTopLeft.moveBy(lineBoxLogicalRect.topLeft());
539 if (layoutBox.isInFlowPositioned())
540 borderBoxLogicalTopLeft += formattingGeometry.inFlowPositionedPositionOffset(layoutBox, horizontalConstraints);
541 // Atomic inline boxes are all set. Their margin/border/content box geometries are already computed. We just have to position them here.
542 boxGeometry.setLogicalTopLeft(toLayoutPoint(borderBoxLogicalTopLeft));
543
544 auto borderBoxTop = borderBoxLogicalTopLeft.y();
545 auto borderBoxBottom = borderBoxTop + boxGeometry.borderBoxHeight();
546 enclosingTopAndBottom.top = std::min(enclosingTopAndBottom.top, borderBoxTop);
547 enclosingTopAndBottom.bottom = std::max(enclosingTopAndBottom.bottom, borderBoxBottom);
548 continue;
549 }
550 if (lineRun.isInlineBoxStart()) {
551 auto& boxGeometry = formattingState.boxGeometry(layoutBox);
552 auto inlineBoxLogicalRect = lineBox.logicalBorderBoxForInlineBox(layoutBox, boxGeometry);
553 formattingState.addLineRun({ lineIndex, layoutBox, inlineBoxLogicalRect, lineRun.expansion(), { } });
554 inlineBoxStartSet.add(&layoutBox);
555 enclosingTopAndBottom.top = std::min(enclosingTopAndBottom.top, inlineBoxLogicalRect.top());
556 continue;
557 }
558 if (lineRun.isInlineBoxEnd()) {
559 inlineBoxEndSet.add(&layoutBox);
560 if (!inlineBoxStartSet.contains(&layoutBox)) {
561 // An inline box can span multiple lines. Use the [inline box end] signal to include it in the enclosing geometry
562 // only when it starts at a previous line.
563 auto inlineBoxLogicalRect = lineBox.logicalBorderBoxForInlineBox(layoutBox, formattingState.boxGeometry(layoutBox));
564 enclosingTopAndBottom.bottom = std::max(enclosingTopAndBottom.bottom, inlineBoxLogicalRect.bottom());
565 }
566 continue;
567 }
568 ASSERT(lineRun.isWordBreakOpportunity());
569 }
570 };
571 constructLineRunsAndUpdateBoxGeometry();
572
573 auto updateBoxGeometryForInlineBoxes = [&] {
574 // FIXME: We may want to keep around an inline box only set.
575 if (!lineBox.hasInlineBox())
576 return;
577 // Grab the inline boxes (even those that don't have associated layout boxes on the current line due to line wrapping)
578 // and update their geometries.
579 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes()) {
580 if (!inlineLevelBox.isInlineBox())
581 continue;
582 auto& layoutBox = inlineLevelBox.layoutBox();
583 auto& boxGeometry = formattingState.boxGeometry(layoutBox);
584 // Inline boxes may or may not be wrapped and have runs on multiple lines (e.g. <span>first line<br>second line<br>third line</span>)
585 auto inlineBoxBorderBox = lineBox.logicalBorderBoxForInlineBox(layoutBox, boxGeometry);
586 auto inlineBoxSize = LayoutSize { LayoutUnit::fromFloatCeil(inlineBoxBorderBox.width()), LayoutUnit::fromFloatCeil(inlineBoxBorderBox.height()) };
587 auto logicalRect = Rect { LayoutPoint { inlineBoxBorderBox.topLeft() }, inlineBoxSize };
588 logicalRect.moveBy(LayoutPoint { lineBoxLogicalRect.topLeft() });
589 if (inlineBoxStartSet.contains(&layoutBox)) {
590 // This inline box showed up first on this line.
591 boxGeometry.setLogicalTopLeft(logicalRect.topLeft());
592 auto contentBoxHeight = logicalRect.height() - (boxGeometry.verticalBorder() + boxGeometry.verticalPadding().value_or(0_lu));
593 boxGeometry.setContentBoxHeight(contentBoxHeight);
594 auto contentBoxWidth = logicalRect.width() - (boxGeometry.horizontalBorder() + boxGeometry.horizontalPadding().value_or(0_lu));
595 boxGeometry.setContentBoxWidth(contentBoxWidth);
596 continue;
597 }
598 // Middle or end of the inline box. Let's stretch the box as needed.
599 auto enclosingBorderBoxRect = BoxGeometry::borderBoxRect(boxGeometry);
600 enclosingBorderBoxRect.expandToContain(logicalRect);
601 boxGeometry.setLogicalLeft(enclosingBorderBoxRect.left());
602
603 boxGeometry.setContentBoxHeight(enclosingBorderBoxRect.height() - (boxGeometry.verticalBorder() + boxGeometry.verticalPadding().value_or(0_lu)));
604 boxGeometry.setContentBoxWidth(enclosingBorderBoxRect.width() - (boxGeometry.horizontalBorder() + boxGeometry.horizontalPadding().value_or(0_lu)));
605 }
606 };
607 updateBoxGeometryForInlineBoxes();
608
609 auto constructLineGeometry = [&] {
610 formattingState.addLine({ lineBoxLogicalRect, enclosingTopAndBottom, lineBox.alignmentBaseline(), rootInlineBoxLogicalRect.left(), lineContent.contentLogicalWidth });
611 };
612 constructLineGeometry();
613
614 return lineBoxLogicalRect;
615}
