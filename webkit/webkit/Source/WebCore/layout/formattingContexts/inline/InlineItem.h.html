30#include "LayoutBox.h"

33namespace WebCore {
34namespace Layout {
35
36class InlineItem {
37public:
38 enum class Type : uint8_t { Text, HardLineBreak, SoftLineBreak, WordBreakOpportunity, Box, Float, InlineBoxStart, InlineBoxEnd };
39 InlineItem(const Box& layoutBox, Type);
40
41 Type type() const { return m_type; }
42 const Box& layoutBox() const { return *m_layoutBox; }
43 const RenderStyle& style() const { return layoutBox().style(); }
44
45 bool isText() const { return type() == Type::Text; }
46 bool isBox() const { return type() == Type::Box; }
47 bool isFloat() const { return type() == Type::Float; }
48 bool isLineBreak() const { return isSoftLineBreak() || isHardLineBreak(); }
49 bool isWordBreakOpportunity() const { return type() == Type::WordBreakOpportunity; }
50 bool isSoftLineBreak() const { return type() == Type::SoftLineBreak; }
51 bool isHardLineBreak() const { return type() == Type::HardLineBreak; }
52 bool isInlineBoxStart() const { return type() == Type::InlineBoxStart; }
53 bool isInlineBoxEnd() const { return type() == Type::InlineBoxEnd; }
54
55private:
56 const Box* m_layoutBox { nullptr };
57 Type m_type { };
58
59protected:
60 // For InlineTextItem
61 enum class TextItemType : uint8_t { Undefined, Whitespace, NonWhitespace };
62 TextItemType m_textItemType { TextItemType::Undefined };
63 bool m_hasWidth { false };
64 bool m_hasTrailingSoftHyphen { false };
65 bool m_isWordSeparator { false };
66 InlineLayoutUnit m_width { };
67 unsigned m_length { 0 };
68
69 // For InlineTextItem and InlineSoftLineBreakItem
70 unsigned m_startOrPosition { 0 };
71};
72
73inline InlineItem::InlineItem(const Box& layoutBox, Type type)
74 : m_layoutBox(&layoutBox)
75 , m_type(type)
76{
77}

82SPECIALIZE_TYPE_TRAITS_END()
83
84}
