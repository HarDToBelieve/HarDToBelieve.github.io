186inline Line::Run::TrailingWhitespace Line::Run::trailingWhitespaceType(const InlineTextItem& inlineTextItem) const
187{
188 if (!inlineTextItem.isWhitespace())
189 return TrailingWhitespace::None;
190 if (InlineTextItem::shouldPreserveSpacesAndTabs(inlineTextItem))
191 return TrailingWhitespace::NotCollapsible;
192 if (inlineTextItem.length() == 1)
193 return TrailingWhitespace::Collapsible;
194 return TrailingWhitespace::Collapsed;
195}

197inline void Line::Run::setNeedsHyphen(InlineLayoutUnit hyphenLogicalWidth)
198{
199 ASSERT(m_textContent);
200 m_textContent->setNeedsHyphen();
201 m_logicalWidth += hyphenLogicalWidth;
202}
