43static inline bool endsWithSoftWrapOpportunity(const InlineTextItem& currentTextItem, const InlineTextItem& nextInlineTextItem)
44{
45 ASSERT(!nextInlineTextItem.isWhitespace());
46 // We are at the position after a whitespace.
47 if (currentTextItem.isWhitespace())
48 return true;
49 // When both these non-whitespace runs belong to the same layout box, it's guaranteed that
50 // they are split at a soft breaking opportunity. See InlineTextItem::moveToNextBreakablePosition.
51 if (&currentTextItem.inlineTextBox() == &nextInlineTextItem.inlineTextBox())
52 return true;
53 // Now we need to collect at least 3 adjacent characters to be able to make a decision whether the previous text item ends with breaking opportunity.
54 // [ex-][ample] <- second to last[x] last[-] current[a]
55 // We need at least 1 character in the current inline text item and 2 more from previous inline items.
56 auto previousContent = currentTextItem.inlineTextBox().content();
57 auto lineBreakIterator = LazyLineBreakIterator { nextInlineTextItem.inlineTextBox().content() };
58 auto previousContentLength = previousContent.length();
59 // FIXME: We should look into the entire uncommitted content for more text context.
60 UChar lastCharacter = previousContentLength ? previousContent[previousContentLength - 1] : 0;
61 if (lastCharacter == softHyphen && currentTextItem.style().hyphens() == Hyphens::None)
62 return false;
63 UChar secondToLastCharacter = previousContentLength > 1 ? previousContent[previousContentLength - 2] : 0;
64 lineBreakIterator.setPriorContext(lastCharacter, secondToLastCharacter);
65 // Now check if we can break right at the inline item boundary.
66 // With the [ex-ample], findNextBreakablePosition should return the startPosition (0).
67 // FIXME: Check if there's a more correct way of finding breaking opportunities.
68 return !TextUtil::findNextBreakablePosition(lineBreakIterator, 0, nextInlineTextItem.style());
69}

71static inline bool isAtSoftWrapOpportunity(const InlineFormattingContext& inlineFormattingContext, const InlineItem& current, const InlineItem& next)
72{
73 // "is at" simple means that there's a soft wrap opportunity right after the [current].
74 // [text][ ][text][inline box start]... (<div>text content<span>..</div>)
75 // soft wrap indexes: 0 and 1 definitely, 2 depends on the content after the [inline box start].
76
77 // https://drafts.csswg.org/css-text-3/#line-break-details
78 // Figure out if the new incoming content puts the uncommitted content on a soft wrap opportunity.
79 // e.g. [inline box start][prior_continuous_content][inline box end] (<span>prior_continuous_content</span>)
80 // An incoming <img> box would enable us to commit the "<span>prior_continuous_content</span>" content
81 // but an incoming text content would not necessarily.
82 ASSERT(current.isText() || current.isBox() || current.isFloat());
83 ASSERT(next.isText() || next.isBox() || next.isFloat());
84 if (current.isText() && next.isText()) {
85 auto& currentInlineTextItem = downcast<InlineTextItem>(current);
86 auto& nextInlineTextItem = downcast<InlineTextItem>(next);
87 if (currentInlineTextItem.isWhitespace()) {
88 // [ ][text] : after [whitespace] position is a soft wrap opportunity.
89 return true;
90 }
91 if (nextInlineTextItem.isWhitespace()) {
92 // [text][ ] (<span>text</span> )
93 // white-space: break-spaces: line breaking opportunity exists after every preserved white space character, but not before.
94 return nextInlineTextItem.style().whiteSpace() != WhiteSpace::BreakSpaces;
95 }
96 if (current.style().lineBreak() == LineBreak::Anywhere || next.style().lineBreak() == LineBreak::Anywhere) {
97 // There is a soft wrap opportunity around every typographic character unit, including around any punctuation character
98 // or preserved white spaces, or in the middle of words.
99 return true;
100 }
101 // Both current and next items are non-whitespace text.
102 // [text][text] : is a continuous content.
103 // [text-][text] : after [hyphen] position is a soft wrap opportunity.
104 return endsWithSoftWrapOpportunity(currentInlineTextItem, nextInlineTextItem);
105 }
106 if (current.isFloat() || next.isFloat()) {
107 // While floats are not part of the inline content and they are not supposed to introduce soft wrap opportunities,
108 // e.g. [text][float box][float box][text][float box][text] is essentially just [text][text][text]
109 // figuring out whether a float (or set of floats) should stay on the line or not (and handle potentially out of order inline items)
110 // brings in unnecessary complexity.
111 return true;
112 }
113 if (current.isBox() || next.isBox()) {
114 auto isImageContent = current.layoutBox().isImage() || next.layoutBox().isImage();
115 if (isImageContent && inlineFormattingContext.layoutState().inQuirksMode())
116 return inlineFormattingContext.formattingQuirks().hasSoftWrapOpportunityAtImage();
117 // [text][inline box start][inline box end][inline box] (text<span></span><img>) : there's a soft wrap opportunity between the [text] and [img].
118 // The line breaking behavior of a replaced element or other atomic inline is equivalent to an ideographic character.
119 return true;
120 }
121 ASSERT_NOT_REACHED();
122 return true;
123}

172inline void LineCandidate::InlineContent::appendInlineItem(const InlineItem& inlineItem, InlineLayoutUnit logicalWidth)
173{
174 ASSERT(inlineItem.isText() || inlineItem.isBox() || inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd());
175 auto collapsibleWidth = [&]() -> std::optional<InlineLayoutUnit> {
176 if (!inlineItem.isText())
177 return { };
178 auto& inlineTextItem = downcast<InlineTextItem>(inlineItem);
179 if (inlineTextItem.isWhitespace() && !InlineTextItem::shouldPreserveSpacesAndTabs(inlineTextItem)) {
180 // Fully collapsible trailing content.
181 return logicalWidth;
182 }
183 // Check for partially collapsible content.
184 if (m_ignoreTrailingLetterSpacing)
185 return { };
186 auto letterSpacing = inlineItem.style().letterSpacing();
187 if (letterSpacing <= 0)
188 return { };
189 ASSERT(logicalWidth > letterSpacing);
190 return letterSpacing;
191 };
192 m_continuousContent.append(inlineItem, logicalWidth, collapsibleWidth());
193 m_hasInlineLevelBox = m_hasInlineLevelBox || inlineItem.isBox() || inlineItem.isInlineBoxStart();
194}

210InlineLayoutUnit LineBuilder::inlineItemWidth(const InlineItem& inlineItem, InlineLayoutUnit contentLogicalLeft) const
211{
212 if (is<InlineTextItem>(inlineItem)) {
213 auto& inlineTextItem = downcast<InlineTextItem>(inlineItem);
214 if (auto contentWidth = inlineTextItem.width())
215 return *contentWidth;
216 if (!inlineTextItem.isWhitespace() || InlineTextItem::shouldPreserveSpacesAndTabs(inlineTextItem))
217 return TextUtil::width(inlineTextItem, contentLogicalLeft);
218 return TextUtil::width(inlineTextItem, inlineTextItem.start(), inlineTextItem.start() + 1, contentLogicalLeft);
219 }
220
221 if (inlineItem.isLineBreak() || inlineItem.isWordBreakOpportunity())
222 return { };
223
224 auto& layoutBox = inlineItem.layoutBox();
225 auto& boxGeometry = m_inlineFormattingContext.geometryForBox(layoutBox);
226
227 if (layoutBox.isFloatingPositioned())
228 return boxGeometry.marginBoxWidth();
229
230 if (layoutBox.isReplacedBox())
231 return boxGeometry.marginBoxWidth();
232
233 if (inlineItem.isInlineBoxStart())
234 return boxGeometry.marginStart() + boxGeometry.borderLeft() + boxGeometry.paddingLeft().value_or(0);
235
236 if (inlineItem.isInlineBoxEnd())
237 return boxGeometry.marginEnd() + boxGeometry.borderRight() + boxGeometry.paddingRight().value_or(0);
238
239 // Non-replaced inline box (e.g. inline-block)
240 return boxGeometry.marginBoxWidth();
241}

260LineBuilder::LineContent LineBuilder::layoutInlineContent(const InlineItemRange& needsLayoutRange, size_t partialLeadingContentLength, std::optional<InlineLayoutUnit> overflowLogicalWidth, const InlineRect& initialLineLogicalRect, bool isFirstLine)
261{
262 initialize(initialConstraintsForLine(initialLineLogicalRect, isFirstLine));
263
264 auto committedContent = placeInlineContent(needsLayoutRange, partialLeadingContentLength, overflowLogicalWidth);
265 auto committedRange = close(needsLayoutRange, committedContent);
266
267 auto isLastLine = isLastLineWithInlineContent(committedRange, needsLayoutRange.end, committedContent.partialTrailingContentLength);
268 return LineContent { committedRange, committedContent.partialTrailingContentLength, committedContent.overflowLogicalWidth, m_floats, m_contentIsConstrainedByFloat
269 , m_lineLogicalRect.topLeft()
270 , m_lineLogicalRect.width()
271 , m_line.contentLogicalWidth()
272 , isLastLine
273 , m_line.nonSpanningInlineLevelBoxCount()
274 , m_line.runs()};
275}

296LineBuilder::CommittedContent LineBuilder::placeInlineContent(const InlineItemRange& needsLayoutRange, size_t partialLeadingContentLength, std::optional<InlineLayoutUnit> leadingLogicalWidth)
297{
298 auto lineCandidate = LineCandidate { layoutState().shouldIgnoreTrailingLetterSpacing() };
299 auto inlineContentBreaker = InlineContentBreaker { };
300
301 auto currentItemIndex = needsLayoutRange.start;
302 size_t committedInlineItemCount = 0;
303 while (currentItemIndex < needsLayoutRange.end) {
304 // 1. Collect the set of runs that we can commit to the line as one entity e.g. <span>text_and_span_start_span_end</span>.
305 // 2. Apply floats and shrink the available horizontal space e.g. <span>intru_<div style="float: left"></div>sive_float</span>.
306 // 3. Check if the content fits the line and commit the content accordingly (full, partial or not commit at all).
307 // 4. Return if we are at the end of the line either by not being able to fit more content or because of an explicit line break.
308 candidateContentForLine(lineCandidate, currentItemIndex, needsLayoutRange, partialLeadingContentLength, std::exchange(leadingLogicalWidth, std::nullopt), m_line.contentLogicalRight());
309 // Now check if we can put this content on the current line.
310 auto result = Result { };
311 if (lineCandidate.floatItem) {
312 ASSERT(lineCandidate.inlineContent.isEmpty());
313 handleFloatContent(*lineCandidate.floatItem);
314 // Floats never terminate the line.
315 } else
316 result = handleInlineContent(inlineContentBreaker, needsLayoutRange, lineCandidate);
317 auto isEndOfLine = result.isEndOfLine == InlineContentBreaker::IsEndOfLine::Yes;
318 if (!result.committedCount.isRevert) {
319 committedInlineItemCount += result.committedCount.value;
320 auto& inlineContent = lineCandidate.inlineContent;
321 auto inlineContentIsFullyCommitted = inlineContent.continuousContent().runs().size() == result.committedCount.value && !result.partialTrailingContentLength;
322 if (inlineContentIsFullyCommitted) {
323 if (auto* wordBreakOpportunity = inlineContent.trailingWordBreakOpportunity()) {
324 // <wbr> needs to be on the line as an empty run so that we can construct an inline box and compute basic geometry.
325 ++committedInlineItemCount;
326 m_line.append(*wordBreakOpportunity, { });
327 }
328 if (inlineContent.trailingLineBreak()) {
329 // Fully committed (or empty) content followed by a line break means "end of line".
330 // FIXME: This will put the line break box at the end of the line while in case of some inline boxes, the line break
331 // could very well be at an earlier position. This has no visual implications at this point though (only geometry correctness on the line break box).
332 // e.g. <span style="border-right: 10px solid green">text<br></span> where the <br>'s horizontal position is before the right border and not after.
333 m_line.append(*inlineContent.trailingLineBreak(), { });
334 ++committedInlineItemCount;
335 isEndOfLine = true;
336 }
337 }
338 } else
339 committedInlineItemCount = result.committedCount.value;
340
341 if (isEndOfLine) {
342 // We can't place any more items on the current line.
343 return { committedInlineItemCount, result.partialTrailingContentLength, result.overflowLogicalWidth };
344 }
345 currentItemIndex = needsLayoutRange.start + committedInlineItemCount + m_floats.size();
346 partialLeadingContentLength = { };
347 }
348 // Looks like we've run out of runs.
349 return { committedInlineItemCount, { } };
350}

352LineBuilder::InlineItemRange LineBuilder::close(const InlineItemRange& needsLayoutRange, const CommittedContent& committedContent)
353{
354 ASSERT(committedContent.inlineItemCount || !m_floats.isEmpty() || m_contentIsConstrainedByFloat);
355 auto numberOfCommittedItems = committedContent.inlineItemCount + m_floats.size();
356 auto trailingInlineItemIndex = needsLayoutRange.start + numberOfCommittedItems - 1;
357 auto lineRange = InlineItemRange { needsLayoutRange.start, trailingInlineItemIndex + 1 };
358 ASSERT(lineRange.end <= needsLayoutRange.end);
359 if (!committedContent.inlineItemCount) {
360 // Line is empty, we only managed to place float boxes.
361 return lineRange;
362 }
363 auto availableWidth = m_lineLogicalRect.width() - m_line.contentLogicalRight();
364 m_line.removeCollapsibleContent(availableWidth);
365 auto horizontalAlignment = root().style().textAlign();
366 auto runsExpandHorizontally = horizontalAlignment == TextAlignMode::Justify && !isLastLineWithInlineContent(lineRange, needsLayoutRange.end, committedContent.partialTrailingContentLength);
367 if (runsExpandHorizontally)
368 m_line.applyRunExpansion(m_lineLogicalRect.width() - m_line.contentLogicalRight());
369 auto lineEndsWithHyphen = false;
370 if (!m_line.runs().isEmpty()) {
371 auto& lastTextContent = m_line.runs().last().textContent();
372 lineEndsWithHyphen = lastTextContent && lastTextContent->needsHyphen();
373 }
374 m_successiveHyphenatedLineCount = lineEndsWithHyphen ? m_successiveHyphenatedLineCount + 1 : 0;
375 return lineRange;
376}

378std::optional<HorizontalConstraints> LineBuilder::floatConstraints(const InlineRect& lineLogicalRect) const
379{
380 auto* floatingState = this->floatingState();
381 if (!floatingState || floatingState->floats().isEmpty())
382 return { };
383
384 // Check for intruding floats and adjust logical left/available width for this line accordingly.
385 auto floatingContext = FloatingContext { formattingContext(), *floatingState };
386 auto constraints = floatingContext.constraints(toLayoutUnit(lineLogicalRect.top()), toLayoutUnit(lineLogicalRect.bottom()));
387 // Check if these values actually constrain the line.
388 if (constraints.left && constraints.left->x <= lineLogicalRect.left())
389 constraints.left = { };
390
391 if (constraints.right && constraints.right->x >= lineLogicalRect.right())
392 constraints.right = { };
393
394 if (!constraints.left && !constraints.right)
395 return { };
396
397 auto lineLogicalLeft = lineLogicalRect.left();
398 auto lineLogicalRight = lineLogicalRect.right();
399 if (constraints.left && constraints.right) {
400 lineLogicalRight = constraints.right->x;
401 lineLogicalLeft = constraints.left->x;
402 } else if (constraints.left) {
403 ASSERT(constraints.left->x >= lineLogicalLeft);
404 lineLogicalLeft = constraints.left->x;
405 } else if (constraints.right) {
406 // Right float boxes may overflow the containing block on the left.
407 lineLogicalRight = std::max<InlineLayoutUnit>(lineLogicalLeft, constraints.right->x);
408 }
409 return HorizontalConstraints { toLayoutUnit(lineLogicalLeft), toLayoutUnit(lineLogicalRight - lineLogicalLeft) };
410}

412LineBuilder::UsedConstraints LineBuilder::initialConstraintsForLine(const InlineRect& initialLineLogicalRect, bool isFirstLine) const
413{
414 auto lineLogicalLeft = initialLineLogicalRect.left();
415 auto lineLogicalRight = initialLineLogicalRect.right();
416 auto lineIsConstrainedByFloat = false;
417
418 if (auto lineConstraints = floatConstraints(initialLineLogicalRect)) {
419 lineLogicalLeft = lineConstraints->logicalLeft;
420 lineLogicalRight = lineConstraints->logicalRight();
421 lineIsConstrainedByFloat = true;
422 }
423
424 auto computedTextIndent = [&]() -> InlineLayoutUnit {
425 // text-indent property specifies the indentation applied to lines of inline content in a block.
426 // The indent is treated as a margin applied to the start edge of the line box.
427 // Unless otherwise specified, only lines that are the first formatted line of an element are affected.
428 // For example, the first line of an anonymous block box is only affected if it is the first child of its parent element.
429 // FIXME: Add support for each-line.
430 // [Integration] root()->parent() would normally produce a valid layout box.
431 auto& root = this->root();
432 auto isFormattingContextRootCandidateToTextIndent = !root.isAnonymous();
433 if (root.isAnonymous()) {
434 // Unless otherwise specified by the each-line and/or hanging keywords, only lines that are the first formatted line
435 // of an element are affected.
436 // For example, the first line of an anonymous block box is only affected if it is the first child of its parent element.
437 auto isIntegratedRootBoxFirstChild = layoutState().isIntegratedRootBoxFirstChild();
438 if (isIntegratedRootBoxFirstChild == LayoutState::IsIntegratedRootBoxFirstChild::NotApplicable)
439 isFormattingContextRootCandidateToTextIndent = root.parent().firstInFlowChild() == &root;
440 else
441 isFormattingContextRootCandidateToTextIndent = isIntegratedRootBoxFirstChild == LayoutState::IsIntegratedRootBoxFirstChild::Yes;
442 }
443 if (!isFormattingContextRootCandidateToTextIndent)
444 return { };
445 auto invertLineRange = root.style().textIndentType() == TextIndentType::Hanging;
446 // text-indent: hanging inverts which lines are affected.
447 // inverted line range -> all the lines except the first one.
448 // !inverted line range -> first line gets the indent.
449 auto shouldIndent = invertLineRange != isFirstLine;
450 if (!shouldIndent)
451 return { };
452
453 auto textIndent = root.style().textIndent();
454 if (textIndent == RenderStyle::initialTextIndent())
455 return { };
456 return { minimumValueForLength(textIndent, initialLineLogicalRect.width()) };
457 };
458 lineLogicalLeft += computedTextIndent();
459 return UsedConstraints { { initialLineLogicalRect.top(), lineLogicalLeft, lineLogicalRight - lineLogicalLeft, initialLineLogicalRect.height() }, lineIsConstrainedByFloat };
460}

462void LineBuilder::candidateContentForLine(LineCandidate& lineCandidate, size_t currentInlineItemIndex, const InlineItemRange& layoutRange, size_t partialLeadingContentLength, std::optional<InlineLayoutUnit> leadingLogicalWidth, InlineLayoutUnit currentLogicalRight)
463{
464 ASSERT(currentInlineItemIndex < layoutRange.end);
465 lineCandidate.reset();
466 // 1. Simply add any overflow content from the previous line to the candidate content. It's always a text content.
467 // 2. Find the next soft wrap position or explicit line break.
468 // 3. Collect floats between the inline content.
469 auto softWrapOpportunityIndex = nextWrapOpportunity(currentInlineItemIndex, layoutRange);
470 // softWrapOpportunityIndex == layoutRange.end means we don't have any wrap opportunity in this content.
471 ASSERT(softWrapOpportunityIndex <= layoutRange.end);
472
473 if (partialLeadingContentLength) {
474 // Handle leading partial content first (overflowing text from the previous line).
475 // Construct a partial leading inline item.
476 m_partialLeadingTextItem = downcast<InlineTextItem>(m_inlineItems[currentInlineItemIndex]).right(partialLeadingContentLength);
477 auto itemWidth = leadingLogicalWidth ? *std::exchange(leadingLogicalWidth, std::nullopt) : inlineItemWidth(*m_partialLeadingTextItem, currentLogicalRight);
478 lineCandidate.inlineContent.appendInlineItem(*m_partialLeadingTextItem, itemWidth);
479 currentLogicalRight += itemWidth;
480 ++currentInlineItemIndex;
481 }
482
483 for (auto index = currentInlineItemIndex; index < softWrapOpportunityIndex; ++index) {
484 auto& inlineItem = m_inlineItems[index];
485 if (inlineItem.isFloat()) {
486 lineCandidate.floatItem = &inlineItem;
487 // This is a soft wrap opportunity, must be the only item in the list.
488 ASSERT(currentInlineItemIndex + 1 == softWrapOpportunityIndex);
489 continue;
490 }
491 if (inlineItem.isText() || inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd() || inlineItem.isBox()) {
492 ASSERT(!leadingLogicalWidth || inlineItem.isText());
493 auto logicalWidth = leadingLogicalWidth ? *std::exchange(leadingLogicalWidth, std::nullopt) : inlineItemWidth(inlineItem, currentLogicalRight);
494 lineCandidate.inlineContent.appendInlineItem(inlineItem, logicalWidth);
495 currentLogicalRight += logicalWidth;
496 continue;
497 }
498 if (inlineItem.isLineBreak() || inlineItem.isWordBreakOpportunity()) {
499 // Since both <br> and <wbr> are explicit word break opportunities they have to be trailing items in this candidate run list unless they are embedded in inline boxes.
500 // e.g. <span><wbr></span>
501#if ASSERT_ENABLED
502 for (auto i = index + 1; i < softWrapOpportunityIndex; ++i)
503 ASSERT(m_inlineItems[i].isInlineBoxEnd());
504#endif
505 inlineItem.isLineBreak() ? lineCandidate.inlineContent.appendTrailingLineBreak(inlineItem) : lineCandidate.inlineContent.appendtrailingWordBreakOpportunity(inlineItem);
506 continue;
507 }
508 ASSERT_NOT_REACHED();
509 }
510 auto inlineContentEndsInSoftWrapOpportunity = [&] {
511 if (!softWrapOpportunityIndex || softWrapOpportunityIndex == layoutRange.end) {
512 // This candidate inline content ends because the entire content ends and not because there's a soft wrap opportunity.
513 return false;
514 }
515 if (m_inlineItems[softWrapOpportunityIndex - 1].isFloat()) {
516 // While we stop at floats, they are not considered real soft wrap opportunities.
517 return false;
518 }
519 // See https://www.w3.org/TR/css-text-3/#line-break-details
520 auto& trailingInlineItem = m_inlineItems[softWrapOpportunityIndex - 1];
521 if (trailingInlineItem.isBox() || trailingInlineItem.isLineBreak() || trailingInlineItem.isWordBreakOpportunity() || trailingInlineItem.isInlineBoxEnd()) {
522 // For Web-compatibility there is a soft wrap opportunity before and after each replaced element or other atomic inline.
523 return true;
524 }
525 if (trailingInlineItem.isText()) {
526 auto& inlineTextItem = downcast<InlineTextItem>(trailingInlineItem);
527 if (inlineTextItem.isWhitespace())
528 return true;
529 // Now in case of non-whitespace trailing content, we need to check if the actual soft wrap opportunity belongs to the next set.
530 // e.g. "this_is_the_trailing_run<span> <-but_this_space_here_is_the_soft_wrap_opportunity"
531 // When there's an inline box start(<span>)/end(</span>) between the trailing and the (next)leading run, while we break before the inline box start (<span>)
532 // the actual soft wrap position is after the inline box start (<span>) but in terms of line breaking continuity the inline box start (<span>) and the whitespace run belong together.
533 RELEASE_ASSERT(layoutRange.end <= m_inlineItems.size());
534 for (auto index = softWrapOpportunityIndex; index < layoutRange.end; ++index) {
535 if (m_inlineItems[index].isInlineBoxStart() || m_inlineItems[index].isInlineBoxEnd())
536 continue;
537 // FIXME: Check if [non-whitespace][inline-box][no-whitespace] content has rules about it.
538 // For now let's say the soft wrap position belongs to the next set of runs when [non-whitespace][inline-box][whitespace], [non-whitespace][inline-box][box] etc.
539 return m_inlineItems[index].isText() && !downcast<InlineTextItem>(m_inlineItems[index]).isWhitespace();
540 }
541 return true;
542 }
543 ASSERT_NOT_REACHED();
544 return true;
545 };
546 lineCandidate.inlineContent.setHasTrailingSoftWrapOpportunity(inlineContentEndsInSoftWrapOpportunity());
547}

549size_t LineBuilder::nextWrapOpportunity(size_t startIndex, const LineBuilder::InlineItemRange& layoutRange) const
550{
551 // 1. Find the start candidate by skipping leading non-content items e.g "<span><span>start". Opportunity is after "<span><span>".
552 // 2. Find the end candidate by skipping non-content items inbetween e.g. "<span><span>start</span>end". Opportunity is after "</span>".
553 // 3. Check if there's a soft wrap opportunity between the 2 candidate inline items and repeat.
554 // 4. Any force line break/explicit wrap content inbetween is considered as wrap opportunity.
555
556 // [ex-][inline box start][inline box end][float][ample] (ex-<span></span><div style="float:left"></div>ample). Wrap index is at [ex-].
557 // [ex][inline box start][amp-][inline box start][le] (ex<span>amp-<span>ample). Wrap index is at [amp-].
558 // [ex-][inline box start][line break][ample] (ex-<span><br>ample). Wrap index is after [br].
559 auto previousInlineItemIndex = std::optional<size_t> { };
560 for (auto index = startIndex; index < layoutRange.end; ++index) {
561 auto& inlineItem = m_inlineItems[index];
562 if (inlineItem.isLineBreak() || inlineItem.isWordBreakOpportunity()) {
563 // We always stop at explicit wrapping opportunities e.g. <br>. However the wrap position may be at later position.
564 // e.g. <span><span><br></span></span> <- wrap position is after the second </span>
565 // but in case of <span><br><span></span></span> <- wrap position is right after <br>.
566 for (++index; index < layoutRange.end && m_inlineItems[index].isInlineBoxEnd(); ++index) { }
567 return index;
568 }
569 if (inlineItem.isInlineBoxStart() || inlineItem.isInlineBoxEnd()) {
570 // Need to see what comes next to decide.
571 continue;
572 }
573 ASSERT(inlineItem.isText() || inlineItem.isBox() || inlineItem.isFloat());
574 if (!previousInlineItemIndex) {
575 previousInlineItemIndex = index;
576 continue;
577 }
578 // At this point previous and current items are not necessarily adjacent items e.g "previous<span>current</span>"
579 auto& previousItem = m_inlineItems[*previousInlineItemIndex];
580 auto& currentItem = m_inlineItems[index];
581 if (isAtSoftWrapOpportunity(m_inlineFormattingContext, previousItem, currentItem)) {
582 if (*previousInlineItemIndex + 1 == index && (!previousItem.isText() || !currentItem.isText())) {
583 // We only know the exact soft wrap opportunity index when the previous and current items are next to each other.
584 return index;
585 }
586 // There's a soft wrap opportunity between 'previousInlineItemIndex' and 'index'.
587 // Now forward-find from the start position to see where we can actually wrap.
588 // [ex-][ample] vs. [ex-][inline box start][inline box end][ample]
589 // where [ex-] is startContent and [ample] is the nextContent.
590 for (auto candidateIndex = *previousInlineItemIndex + 1; candidateIndex < index; ++candidateIndex) {
591 if (m_inlineItems[candidateIndex].isInlineBoxStart()) {
592 // inline content and [inline box start] and [inline box end] form unbreakable content.
593 // ex-<span></span>ample : wrap opportunity is after "ex-".
594 // ex-</span></span>ample : wrap opportunity is after "ex-</span></span>".
595 // ex-</span><span>ample</span> : wrap opportunity is after "ex-</span>".
596 // ex-<span><span>ample</span></span> : wrap opportunity is after "ex-".
597 return candidateIndex;
598 }
599 }
600 return index;
601 }
602 previousInlineItemIndex = index;
603 }
604 return layoutRange.end;
605}

607void LineBuilder::handleFloatContent(const InlineItem& floatItem)
608{
609 auto& floatBox = floatItem.layoutBox();
610 m_floats.append(&floatBox);
611
612 auto* floatingState = this->floatingState();
613 if (!floatingState)
614 return;
615
616 ASSERT(formattingState());
617 auto& boxGeometry = formattingState()->boxGeometry(floatBox);
618 // Set static position first.
619 boxGeometry.setLogicalTopLeft(LayoutPoint { m_lineLogicalRect.topLeft() });
620 // Float it.
621 ASSERT(m_rootHorizontalConstraints);
622 auto floatingContext = FloatingContext { formattingContext(), *floatingState };
623 auto floatingPosition = floatingContext.positionForFloat(floatBox, *m_rootHorizontalConstraints);
624 boxGeometry.setLogicalTopLeft(floatingPosition);
625 floatingState->append(floatingContext.toFloatItem(floatBox));
626 // Check if this float shrinks the line (they don't get positioned higher than the line).
627 if (floatingPosition.y() > m_lineLogicalRect.bottom())
628 return;
629
630 m_contentIsConstrainedByFloat = true;
631 auto floatBoxWidth = inlineItemWidth(floatItem, { });
632 if (floatBox.isLeftFloatingPositioned())
633 m_lineLogicalRect.setLeft(m_lineLogicalRect.left() + floatBoxWidth);
634 m_lineLogicalRect.expandHorizontally(-floatBoxWidth);
635}

637LineBuilder::Result LineBuilder::handleInlineContent(InlineContentBreaker& inlineContentBreaker, const InlineItemRange& layoutRange, const LineCandidate& lineCandidate)
638{
639 auto& inlineContent = lineCandidate.inlineContent;
640 if (inlineContent.continuousContent().runs().isEmpty()) {
641 ASSERT(inlineContent.trailingLineBreak() || inlineContent.trailingWordBreakOpportunity());
642 return { inlineContent.trailingLineBreak() ? InlineContentBreaker::IsEndOfLine::Yes : InlineContentBreaker::IsEndOfLine::No };
643 }
644 auto shouldDisableHyphenation = [&] {
645 auto& style = root().style();
646 unsigned limitLines = style.hyphenationLimitLines() == RenderStyle::initialHyphenationLimitLines() ? std::numeric_limits<unsigned>::max() : style.hyphenationLimitLines();
647 return m_successiveHyphenatedLineCount >= limitLines;
648 };
649 if (shouldDisableHyphenation())
650 inlineContentBreaker.setHyphenationDisabled();
651
652 auto& continuousInlineContent = lineCandidate.inlineContent.continuousContent();
653 auto lineLogicalRectForCandidateContent = [&] {
654 // Check if the candidate content would stretch the line and whether additional floats are getting in the way.
655 if (!floatingState() || !inlineContent.hasInlineLevelBox())
656 return m_lineLogicalRect;
657 auto maximumLineLogicalHeight = m_lineLogicalRect.height();
658 for (auto& run : continuousInlineContent.runs()) {
659 // FIXME: Add support for inline boxes too.
660 if (!run.inlineItem.isBox())
661 continue;
662 maximumLineLogicalHeight = std::max(maximumLineLogicalHeight, InlineLayoutUnit { formattingContext().geometryForBox(run.inlineItem.layoutBox()).marginBoxHeight() });
663 }
664 if (maximumLineLogicalHeight == m_lineLogicalRect.height())
665 return m_lineLogicalRect;
666 auto adjustedLineLogicalRect = InlineRect { m_lineLogicalRect.top(), m_lineLogicalRect.left(), m_lineLogicalRect.width(), maximumLineLogicalHeight };
667 if (auto horizontalConstraints = floatConstraints(adjustedLineLogicalRect)) {
668 adjustedLineLogicalRect.setLeft(horizontalConstraints->logicalLeft);
669 adjustedLineLogicalRect.setWidth(horizontalConstraints->logicalWidth);
670 }
671 return adjustedLineLogicalRect;
672 }();
673 auto availableWidth = [&] {
674 auto availableWidthForContent = lineLogicalRectForCandidateContent.width() - m_line.contentLogicalRight();
675 return std::isnan(availableWidthForContent) ? maxInlineLayoutUnit() : availableWidthForContent;
676 }();
677 // While the floats are not considered to be on the line, they make the line contentful for line breaking.
678 auto lineHasContent = !m_line.runs().isEmpty() || m_contentIsConstrainedByFloat;
679 auto lineStatus = InlineContentBreaker::LineStatus { m_line.contentLogicalRight(), availableWidth, m_line.trimmableTrailingWidth(), m_line.trailingSoftHyphenWidth(), m_line.isTrailingRunFullyTrimmable(), lineHasContent, !m_wrapOpportunityList.isEmpty() };
680 auto result = inlineContentBreaker.processInlineContent(continuousInlineContent, lineStatus);
681 auto& candidateRuns = continuousInlineContent.runs();
682 if (result.action == InlineContentBreaker::Result::Action::Keep) {
683 // This continuous content can be fully placed on the current line.
684 m_lineLogicalRect = lineLogicalRectForCandidateContent;
685 for (auto& run : candidateRuns)
686 m_line.append(run.inlineItem, run.logicalWidth);
687 if (lineCandidate.inlineContent.hasTrailingSoftWrapOpportunity()) {
688 // Check if we are allowed to wrap at this position.
689 auto& trailingItem = candidateRuns.last().inlineItem;
690 // FIXME: There must be a way to decide if the trailing run actually ended up on the line.
691 // Let's just deal with collapsed leading whitespace for now.
692 if (!m_line.runs().isEmpty() && InlineContentBreaker::isWrappingAllowed(trailingItem))
693 m_wrapOpportunityList.append(&trailingItem);
694 }
695 return { result.isEndOfLine, { candidateRuns.size(), false } };
696 }
697
698 auto eligibleOverflowWidthAsLeading = [&] () -> std::optional<InlineLayoutUnit> {
699 // FIXME: Add support for other types of continuous content.
700 ASSERT(result.action == InlineContentBreaker::Result::Action::Wrap || result.action == InlineContentBreaker::Result::Action::Break);
701 if (candidateRuns.size() != 1 || !candidateRuns.first().inlineItem.isText())
702 return { };
703 auto& inlineTextItem = downcast<InlineTextItem>(candidateRuns.first().inlineItem);
704 if (inlineTextItem.isWhitespace())
705 return { };
706 if (result.action == InlineContentBreaker::Result::Action::Wrap)
707 return candidateRuns.first().logicalWidth;
708 if (result.action == InlineContentBreaker::Result::Action::Break && result.partialTrailingContent->partialRun)
709 return candidateRuns.first().logicalWidth - result.partialTrailingContent->partialRun->logicalWidth;
710 return { };
711 };
712
713 if (result.action == InlineContentBreaker::Result::Action::Wrap) {
714 ASSERT(result.isEndOfLine == InlineContentBreaker::IsEndOfLine::Yes);
715 // This continuous content can't be placed on the current line. Nothing to commit at this time.
716 return { InlineContentBreaker::IsEndOfLine::Yes, { }, { }, eligibleOverflowWidthAsLeading() };
717 }
718 if (result.action == InlineContentBreaker::Result::Action::WrapWithHyphen) {
719 ASSERT(result.isEndOfLine == InlineContentBreaker::IsEndOfLine::Yes);
720 // This continuous content can't be placed on the current line, nothing to commit.
721 // However we need to make sure that the current line gains a trailing hyphen.
722 ASSERT(m_line.trailingSoftHyphenWidth());
723 m_line.addTrailingHyphen(*m_line.trailingSoftHyphenWidth());
724 return { InlineContentBreaker::IsEndOfLine::Yes };
725 }
726 if (result.action == InlineContentBreaker::Result::Action::RevertToLastWrapOpportunity) {
727 ASSERT(result.isEndOfLine == InlineContentBreaker::IsEndOfLine::Yes);
728 // Not only this content can't be placed on the current line, but we even need to revert the line back to an earlier position.
729 ASSERT(!m_wrapOpportunityList.isEmpty());
730 return { InlineContentBreaker::IsEndOfLine::Yes, { rebuildLine(layoutRange, *m_wrapOpportunityList.last()), true } };
731 }
732 if (result.action == InlineContentBreaker::Result::Action::RevertToLastNonOverflowingWrapOpportunity) {
733 ASSERT(result.isEndOfLine == InlineContentBreaker::IsEndOfLine::Yes);
734 ASSERT(!m_wrapOpportunityList.isEmpty());
735 return { InlineContentBreaker::IsEndOfLine::Yes, { rebuildLineForTrailingSoftHyphen(layoutRange), true } };
736 }
737 if (result.action == InlineContentBreaker::Result::Action::Break) {
738 ASSERT(result.isEndOfLine == InlineContentBreaker::IsEndOfLine::Yes);
739 // Commit the combination of full and partial content on the current line.
740 ASSERT(result.partialTrailingContent);
741 commitPartialContent(candidateRuns, *result.partialTrailingContent);
742 // When breaking multiple runs <span style="word-break: break-all">text</span><span>content</span>, we might end up breaking them at run boundary.
743 // It simply means we don't really have a partial run. Partial content yes, but not partial run.
744 auto trailingRunIndex = result.partialTrailingContent->trailingRunIndex;
745 auto committedInlineItemCount = trailingRunIndex + 1;
746 if (!result.partialTrailingContent->partialRun)
747 return { InlineContentBreaker::IsEndOfLine::Yes, { committedInlineItemCount, false } };
748
749 auto partialRun = *result.partialTrailingContent->partialRun;
750 auto& trailingInlineTextItem = downcast<InlineTextItem>(candidateRuns[trailingRunIndex].inlineItem);
751 ASSERT(partialRun.length < trailingInlineTextItem.length());
752 auto overflowLength = trailingInlineTextItem.length() - partialRun.length;
753 return { InlineContentBreaker::IsEndOfLine::Yes, { committedInlineItemCount, false }, overflowLength, eligibleOverflowWidthAsLeading() };
754 }
755 ASSERT_NOT_REACHED();
756 return { InlineContentBreaker::IsEndOfLine::No };
757}

759void LineBuilder::commitPartialContent(const InlineContentBreaker::ContinuousContent::RunList& runs, const InlineContentBreaker::Result::PartialTrailingContent& partialTrailingContent)
760{
761 for (size_t index = 0; index < runs.size(); ++index) {
762 auto& run = runs[index];
763 if (partialTrailingContent.trailingRunIndex == index) {
764 // Create and commit partial trailing item.
765 if (auto partialRun = partialTrailingContent.partialRun) {
766 ASSERT(run.inlineItem.isText());
767 auto& trailingInlineTextItem = downcast<InlineTextItem>(runs[partialTrailingContent.trailingRunIndex].inlineItem);
768 auto partialTrailingTextItem = trailingInlineTextItem.left(partialRun->length);
769 m_line.append(partialTrailingTextItem, partialRun->logicalWidth);
770 if (auto hyphenWidth = partialRun->hyphenWidth)
771 m_line.addTrailingHyphen(*hyphenWidth);
772 return;
773 }
774 // The partial run is the last content to commit.
775 m_line.append(run.inlineItem, run.logicalWidth);
776 return;
777 }
778 m_line.append(run.inlineItem, run.logicalWidth);
779 }

782size_t LineBuilder::rebuildLine(const InlineItemRange& layoutRange, const InlineItem& lastInlineItemToAdd)
783{
784 ASSERT(!m_wrapOpportunityList.isEmpty());
785 // We might already have added floats. They shrink the available horizontal space for the line.
786 // Let's just reuse what the line has at this point.
787 m_line.initialize();
788 auto currentItemIndex = layoutRange.start;
789 if (m_partialLeadingTextItem) {
790 m_line.append(*m_partialLeadingTextItem, inlineItemWidth(*m_partialLeadingTextItem, { }));
791 if (&m_partialLeadingTextItem.value() == &lastInlineItemToAdd)
792 return 1;
793 ++currentItemIndex;
794 }
795 for (; currentItemIndex < layoutRange.end; ++currentItemIndex) {
796 auto& inlineItem = m_inlineItems[currentItemIndex];
797 m_line.append(inlineItem, inlineItemWidth(inlineItem, m_line.contentLogicalRight()));
798 if (&inlineItem == &lastInlineItemToAdd)
799 return currentItemIndex - layoutRange.start + 1;
800 }
801 return layoutRange.size();
802}

804size_t LineBuilder::rebuildLineForTrailingSoftHyphen(const InlineItemRange& layoutRange)
805{
806 ASSERT(!m_wrapOpportunityList.isEmpty());
807 // Revert all the way back to a wrap opportunity when either a soft hyphen fits or no hyphen is required.
808 for (auto i = m_wrapOpportunityList.size(); i-- > 1;) {
809 auto& softWrapOpportunityItem = *m_wrapOpportunityList[i];
810 // FIXME: If this turns out to be a perf issue, we could also traverse the wrap list and keep adding the items
811 // while watching the available width very closely.
812 auto committedCount = rebuildLine(layoutRange, softWrapOpportunityItem);
813 auto availableWidth = m_lineLogicalRect.width() - m_line.contentLogicalRight();
814 auto trailingSoftHyphenWidth = m_line.trailingSoftHyphenWidth();
815 // Check if the trailing hyphen now fits the line (or we don't need hyhen anymore).
816 if (!trailingSoftHyphenWidth || trailingSoftHyphenWidth <= availableWidth) {
817 if (trailingSoftHyphenWidth)
818 m_line.addTrailingHyphen(*trailingSoftHyphenWidth);
819 return committedCount;
820 }
821 }
822 // Have at least some content on the line.
823 auto committedCount = rebuildLine(layoutRange, *m_wrapOpportunityList.first());
824 if (auto trailingSoftHyphenWidth = m_line.trailingSoftHyphenWidth())
825 m_line.addTrailingHyphen(*trailingSoftHyphenWidth);
826 return committedCount;
827}

829bool LineBuilder::isLastLineWithInlineContent(const InlineItemRange& lineRange, size_t lastInlineItemIndex, bool hasPartialTrailingContent) const
830{
831 if (hasPartialTrailingContent)
832 return false;
833 if (lineRange.end == lastInlineItemIndex)
834 return true;
835 // Omit floats to see if this is the last line with inline content.
836 for (auto i = lastInlineItemIndex; i--;) {
837 if (!m_inlineItems[i].isFloat())
838 return i == lineRange.end - 1;
839 }
840 // There has to be at least one non-float item.
841 ASSERT_NOT_REACHED();
842 return false;
