37struct LineRun {
38 WTF_MAKE_STRUCT_FAST_ALLOCATED;
39 struct Text {
40 WTF_MAKE_STRUCT_FAST_ALLOCATED;
41 public:
42 Text(size_t position, size_t length, const String&);
43
44 size_t start() const { return m_start; }
45 size_t end() const { return start() + length(); }
46 size_t length() const { return m_length; }
47 String content() const { return m_contentString; }
48
49 bool needsHyphen() const { return m_needsHyphen; }
50 void setNeedsHyphen() { m_needsHyphen = true; }
51
52 void expand(size_t delta) { m_length += delta; }
53 void shrink(size_t delta) { m_length -= delta; }
54
55 private:
56 size_t m_start { 0 };
57 size_t m_length { 0 };
58 bool m_needsHyphen { false };
59 String m_contentString;
60 };

62 struct Expansion;
63 LineRun(size_t lineIndex, const Box&, const InlineRect&, Expansion, std::optional<Text> = std::nullopt);
64
65 const InlineRect& logicalRect() const { return m_logicalRect; }
66
67 InlineLayoutUnit logicalTop() const { return logicalRect().top(); }
68 InlineLayoutUnit logicalBottom() const { return logicalRect().bottom(); }
69 InlineLayoutUnit logicalLeft() const { return logicalRect().left(); }
70 InlineLayoutUnit logicalRight() const { return logicalRect().right(); }
71
72 InlineLayoutUnit logicalWidth() const { return logicalRect().width(); }
73 InlineLayoutUnit logicalHeight() const { return logicalRect().height(); }
74
75 void moveVertically(InlineLayoutUnit offset) { m_logicalRect.moveVertically(offset); }
76 std::optional<Text>& text() { return m_text; }
77 const std::optional<Text>& text() const { return m_text; }
78
79 struct Expansion {
80 ExpansionBehavior behavior { DefaultExpansion };
81 InlineLayoutUnit horizontalExpansion { 0 };
82 };
83 Expansion expansion() const { return m_expansion; }
84
85 const Box& layoutBox() const { return *m_layoutBox; }
86 size_t lineIndex() const { return m_lineIndex; }
87
88private:
89 const size_t m_lineIndex;
90 WeakPtr<const Layout::Box> m_layoutBox;
91 InlineRect m_logicalRect;
92 Expansion m_expansion;
93 std::optional<Text> m_text;
94};
