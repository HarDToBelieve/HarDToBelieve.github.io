35class InlineRect {
36public:
37 InlineRect() = default;
38 InlineRect(InlineLayoutUnit top, InlineLayoutUnit left, InlineLayoutUnit width, InlineLayoutUnit height);
39 InlineRect(const InlineLayoutPoint& topLeft, InlineLayoutUnit width, InlineLayoutUnit height);
40 InlineRect(const InlineLayoutPoint& topLeft, const InlineLayoutSize&);
41
42 InlineLayoutUnit top() const;
43 InlineLayoutUnit left() const;
44 InlineLayoutPoint topLeft() const;
45
46 InlineLayoutUnit bottom() const;
47 InlineLayoutUnit right() const;
48
49 InlineLayoutUnit width() const;
50 InlineLayoutUnit height() const;
51 InlineLayoutSize size() const;
52
53 void setTop(InlineLayoutUnit);
54 void setBottom(InlineLayoutUnit);
55 void setLeft(InlineLayoutUnit);
56 void setTopLeft(const InlineLayoutPoint&);
57 void setWidth(InlineLayoutUnit);
58 void setHeight(InlineLayoutUnit);
59
60 void moveHorizontally(InlineLayoutUnit);
61 void moveVertically(InlineLayoutUnit);
62 void moveBy(InlineLayoutPoint);
63
64 void expand(std::optional<InlineLayoutUnit>, std::optional<InlineLayoutUnit>);
65 void expandToContain(const InlineRect&);
66 void expandHorizontally(InlineLayoutUnit delta) { expand(delta, { }); }
67 void expandVertically(InlineLayoutUnit delta) { expand({ }, delta); }
68 void expandVerticallyToContain(const InlineRect&);
69 void inflate(InlineLayoutUnit);
70
71 operator InlineLayoutRect() const;
72
73private:
74#if ASSERT_ENABLED
75 void invalidateTop() { m_hasValidTop = false; }
76 void invalidateLeft() { m_hasValidLeft = false; }
77 void invalidateWidth() { m_hasValidWidth = false; }
78 void invalidateHeight() { m_hasValidHeight = false; }
79 void invalidatePosition();
80
81 bool hasValidPosition() const { return m_hasValidTop && m_hasValidLeft; }
82 bool hasValidSize() const { return m_hasValidWidth && m_hasValidHeight; }
83 bool hasValidGeometry() const { return hasValidPosition() && hasValidSize(); }
84
85 void setHasValidPosition();
86 void setHasValidSize();
87
88 bool m_hasValidTop { false };
89 bool m_hasValidLeft { false };
90 bool m_hasValidWidth { false };
91 bool m_hasValidHeight { false };
92#endif // ASSERT_ENABLED
93 InlineLayoutRect m_rect;
94};
95
96inline InlineRect::InlineRect(InlineLayoutUnit top, InlineLayoutUnit left, InlineLayoutUnit width, InlineLayoutUnit height)
97 : m_rect(left, top, width, height)
98{
99#if ASSERT_ENABLED
100 m_hasValidTop = true;
101 m_hasValidLeft = true;
102 m_hasValidWidth = true;
103 m_hasValidHeight = true;
104#endif
105}
106
107inline InlineRect::InlineRect(const InlineLayoutPoint& topLeft, InlineLayoutUnit width, InlineLayoutUnit height)
108 : InlineRect(topLeft.y(), topLeft.x(), width, height)
109{
110}
111
112inline InlineRect::InlineRect(const InlineLayoutPoint& topLeft, const InlineLayoutSize& size)
113 : InlineRect(topLeft.y(), topLeft.x(), size.width(), size.height())
114{
115}
116
117#if ASSERT_ENABLED
118inline void InlineRect::invalidatePosition()
119{
120 invalidateTop();
121 invalidateLeft();
122}
123
124inline void InlineRect::setHasValidPosition()
125{
126 m_hasValidTop = true;
127 m_hasValidLeft = true;
128}
129
130inline void InlineRect::setHasValidSize()
131{
132 m_hasValidWidth = true;
133 m_hasValidHeight = true;
134}
135#endif // ASSERT_ENABLED
136
137inline InlineLayoutUnit InlineRect::top() const
138{
139 ASSERT(m_hasValidTop);
140 return m_rect.y();
141}
142
143inline InlineLayoutUnit InlineRect::left() const
144{
145 ASSERT(m_hasValidLeft);
146 return m_rect.x();
147}
148
149inline InlineLayoutUnit InlineRect::bottom() const
150{
151 ASSERT(m_hasValidTop && m_hasValidHeight);
152 return m_rect.maxY();
153}
154
155inline InlineLayoutUnit InlineRect::right() const
156{
157 ASSERT(m_hasValidLeft && m_hasValidWidth);
158 return m_rect.maxX();
159}
160
161inline InlineLayoutPoint InlineRect::topLeft() const
162{
163 ASSERT(hasValidPosition());
164 return m_rect.minXMinYCorner();
165}
166
167inline InlineLayoutSize InlineRect::size() const
168{
169 ASSERT(hasValidSize());
170 return m_rect.size();
171}
172
173inline InlineLayoutUnit InlineRect::width() const
174{
175 ASSERT(m_hasValidWidth);
176 return m_rect.width();
177}
178
179inline InlineLayoutUnit InlineRect::height() const
180{
181 ASSERT(m_hasValidHeight);
182 return m_rect.height();
183}
184
185inline void InlineRect::setTopLeft(const InlineLayoutPoint& topLeft)
186{
187#if ASSERT_ENABLED
188 setHasValidPosition();
189#endif
190 m_rect.setLocation(topLeft);
191}
192
193inline void InlineRect::setTop(InlineLayoutUnit top)
194{
195#if ASSERT_ENABLED
196 m_hasValidTop = true;
197#endif
198 m_rect.setY(top);
199}
200
201inline void InlineRect::setBottom(InlineLayoutUnit bottom)
202{
203#if ASSERT_ENABLED
204 m_hasValidTop = true;
205 m_hasValidHeight = true;
206#endif
207 m_rect.shiftMaxYEdgeTo(bottom);
208}
209
210inline void InlineRect::setLeft(InlineLayoutUnit left)
211{
212#if ASSERT_ENABLED
213 m_hasValidLeft = true;
214#endif
215 m_rect.setX(left);
216}
217
218inline void InlineRect::setWidth(InlineLayoutUnit width)
219{
220#if ASSERT_ENABLED
221 m_hasValidWidth = true;
222#endif
223 m_rect.setWidth(width);
224}
225
226inline void InlineRect::setHeight(InlineLayoutUnit height)
227{
228#if ASSERT_ENABLED
229 m_hasValidHeight = true;
230#endif
231 m_rect.setHeight(height);
232}
233
234inline void InlineRect::moveHorizontally(InlineLayoutUnit offset)
235{
236 ASSERT(m_hasValidLeft);
237 m_rect.move(InlineLayoutSize { offset, 0 });
238}
239
240inline void InlineRect::moveVertically(InlineLayoutUnit offset)
241{
242 ASSERT(m_hasValidTop);
243 m_rect.move(InlineLayoutSize { 0, offset });
244}
245
246inline void InlineRect::moveBy(InlineLayoutPoint offset)
247{
248 ASSERT(m_hasValidTop);
249 ASSERT(m_hasValidLeft);
250 m_rect.moveBy(offset);
251}
252
253inline void InlineRect::expand(std::optional<InlineLayoutUnit> width, std::optional<InlineLayoutUnit> height)
254{
255 ASSERT(!width || m_hasValidWidth);
256 ASSERT(!height || m_hasValidHeight);
257 m_rect.expand(width.value_or(0), height.value_or(0));
258}
259
260inline void InlineRect::expandToContain(const InlineRect& other)
261{
262#if ASSERT_ENABLED
263 m_hasValidTop = true;
264 m_hasValidLeft = true;
265 m_hasValidWidth = true;
266 m_hasValidHeight = true;
267#endif
268 m_rect.uniteEvenIfEmpty(other);
269}
270
271inline void InlineRect::expandVerticallyToContain(const InlineRect& other)
272{
273 auto containTop = std::min(top(), other.top());
274 auto containBottom = std::max(bottom(), other.bottom());
275 setTop(containTop);
276 setBottom(containBottom);
277}
278
279inline void InlineRect::inflate(InlineLayoutUnit inflate)
280{
281 ASSERT(hasValidGeometry());
282 m_rect.inflate(inflate);
283}
284
285inline InlineRect::operator InlineLayoutRect() const
286{
287 ASSERT(hasValidGeometry());
288 return m_rect;
289}
