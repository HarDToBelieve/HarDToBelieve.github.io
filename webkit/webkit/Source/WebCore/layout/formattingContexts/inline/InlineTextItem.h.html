33namespace WebCore {
34namespace Layout {
35
36using InlineItems = Vector<InlineItem>;
37
38class InlineTextItem : public InlineItem {
39public:
40 static void createAndAppendTextItems(InlineItems&, const InlineTextBox&);
41
42 unsigned start() const { return m_startOrPosition; }
43 unsigned end() const { return start() + length(); }
44 unsigned length() const { return m_length; }
45
46 bool isWhitespace() const { return m_textItemType == TextItemType::Whitespace; }
47 bool isWordSeparator() const { return m_isWordSeparator; }
48 bool hasTrailingSoftHyphen() const { return m_hasTrailingSoftHyphen; }
49 std::optional<InlineLayoutUnit> width() const { return m_hasWidth ? std::make_optional(m_width) : std::optional<InlineLayoutUnit> { }; }
50 bool isEmptyContent() const;
51
52 const InlineTextBox& inlineTextBox() const { return downcast<InlineTextBox>(layoutBox()); }
53
54 InlineTextItem left(unsigned length) const;
55 InlineTextItem right(unsigned length) const;
56
57 static bool shouldPreserveSpacesAndTabs(const InlineTextItem&);
58
59private:
60 using InlineItem::TextItemType;
61
62 InlineTextItem(const InlineTextBox&, unsigned start, unsigned length, bool hasTrailingSoftHyphen, bool isWordSeparator, std::optional<InlineLayoutUnit> width, TextItemType);
63 explicit InlineTextItem(const InlineTextBox&);
64
65 static InlineTextItem createWhitespaceItem(const InlineTextBox&, unsigned start, unsigned length, bool isWordSeparator, std::optional<InlineLayoutUnit> width);
66 static InlineTextItem createNonWhitespaceItem(const InlineTextBox&, unsigned start, unsigned length, bool hasTrailingSoftHyphen, std::optional<InlineLayoutUnit> width);
67 static InlineTextItem createEmptyItem(const InlineTextBox&);
68};
69
70inline InlineTextItem InlineTextItem::createWhitespaceItem(const InlineTextBox& inlineTextBox, unsigned start, unsigned length, bool isWordSeparator, std::optional<InlineLayoutUnit> width)
71{
72 return { inlineTextBox, start, length, false, isWordSeparator, width, TextItemType::Whitespace };
73}
74
75inline InlineTextItem InlineTextItem::createNonWhitespaceItem(const InlineTextBox& inlineTextBox, unsigned start, unsigned length, bool hasTrailingSoftHyphen, std::optional<InlineLayoutUnit> width)
76{
77 // FIXME: Use the following list of non-whitespace characters to set the "isWordSeparator" bit: noBreakSpace, ethiopicWordspace, aegeanWordSeparatorLine aegeanWordSeparatorDot ugariticWordDivider.
78 return { inlineTextBox, start, length, hasTrailingSoftHyphen, false, width, TextItemType::NonWhitespace };
79}
80
81inline InlineTextItem InlineTextItem::createEmptyItem(const InlineTextBox& inlineTextBox)
82{
83 return InlineTextItem { inlineTextBox };
84}
85
86inline InlineTextItem::InlineTextItem(const InlineTextBox& inlineTextBox, unsigned start, unsigned length, bool hasTrailingSoftHyphen, bool isWordSeparator, std::optional<InlineLayoutUnit> width, TextItemType textItemType)
87 : InlineItem(inlineTextBox, Type::Text)
88{
89 m_startOrPosition = start;
90 m_length = length;
91 m_hasWidth = !!width;
92 m_hasTrailingSoftHyphen = hasTrailingSoftHyphen;
93 m_isWordSeparator = isWordSeparator;
94 m_width = width.value_or(0);
95 m_textItemType = textItemType;
96}
97
98inline InlineTextItem::InlineTextItem(const InlineTextBox& inlineTextBox)
99 : InlineItem(inlineTextBox, Type::Text)
100{
101}
102
103inline InlineTextItem InlineTextItem::left(unsigned length) const
104{
105 RELEASE_ASSERT(length <= this->length());
106 ASSERT(m_textItemType != TextItemType::Undefined);
107 ASSERT(length);
108 return { inlineTextBox(), start(), length, false, isWordSeparator(), std::nullopt, m_textItemType };
109}
110
111inline InlineTextItem InlineTextItem::right(unsigned length) const
112{
113 RELEASE_ASSERT(length <= this->length());
114 ASSERT(m_textItemType != TextItemType::Undefined);
115 ASSERT(length);
116 return { inlineTextBox(), end() - length, length, hasTrailingSoftHyphen(), isWordSeparator(), std::nullopt, m_textItemType };
117}
118
119}
120}
