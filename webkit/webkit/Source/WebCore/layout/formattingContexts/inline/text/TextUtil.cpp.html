42{
43 return TextUtil::width(inlineTextItem, inlineTextItem.start(), inlineTextItem.end(), contentLogicalLeft);

46InlineLayoutUnit TextUtil::width(const InlineTextItem& inlineTextItem, unsigned from, unsigned to, InlineLayoutUnit contentLogicalLeft)
47{
48 RELEASE_ASSERT(from >= inlineTextItem.start());
49 RELEASE_ASSERT(to <= inlineTextItem.end());
50 if (inlineTextItem.isWhitespace() && !InlineTextItem::shouldPreserveSpacesAndTabs(inlineTextItem)) {
51 auto spaceWidth = inlineTextItem.style().fontCascade().spaceWidth();
52 return std::isnan(spaceWidth) ? 0.0f : std::isinf(spaceWidth) ? maxInlineLayoutUnit() : spaceWidth;
53 }
54 return TextUtil::width(inlineTextItem.inlineTextBox(), from, to, contentLogicalLeft);
55}
56
57InlineLayoutUnit TextUtil::width(const InlineTextBox& inlineTextBox, unsigned from, unsigned to, InlineLayoutUnit contentLogicalLeft)
58{
59 if (from == to)
60 return 0;
61
62 auto& style = inlineTextBox.style();
63 auto& font = style.fontCascade();
64 auto text = inlineTextBox.content();
65 ASSERT(to <= text.length());
66 auto hasKerningOrLigatures = font.enableKerning() || font.requiresShaping();
67 auto measureWithEndSpace = hasKerningOrLigatures && to < text.length() && text[to] == ' ';
68 if (measureWithEndSpace)
69 ++to;
70 float width = 0;
71 if (inlineTextBox.canUseSimplifiedContentMeasuring())
72 width = font.widthForSimpleText(StringView(text).substring(from, to - from));
73 else {
74 auto tabWidth = style.collapseWhiteSpace() ? TabSize(0) : style.tabSize();
75 WebCore::TextRun run(StringView(text).substring(from, to - from), contentLogicalLeft);
76 if (tabWidth)
77 run.setTabSize(true, tabWidth);
78 width = font.width(run);
79 }
80
81 if (measureWithEndSpace)
82 width -= (font.spaceWidth() + font.wordSpacing());
83
84 return std::isnan(width) ? 0.0f : std::isinf(width) ? maxInlineLayoutUnit() : width;
85}

87TextUtil::SplitData TextUtil::split(const InlineTextItem& inlineTextItem, InlineLayoutUnit textWidth, InlineLayoutUnit availableWidth, InlineLayoutUnit contentLogicalLeft)
88{
89 ASSERT(availableWidth >= 0);
90 auto startPosition = inlineTextItem.start();
91 auto length = inlineTextItem.length();
92 ASSERT(length);
93
94 auto left = startPosition;
95 // Pathological case of (extremely)long string and narrow lines.
96 // Adjust the range so that we can pick a reasonable midpoint.
97 InlineLayoutUnit averageCharacterWidth = textWidth / length;
98 unsigned offset = toLayoutUnit(2 * availableWidth / averageCharacterWidth).toUnsigned();
99 auto right = std::min<unsigned>(left + offset, (startPosition + length - 1));
100 // Preserve the left width for the final split position so that we don't need to remeasure the left side again.
101 InlineLayoutUnit leftSideWidth = 0;
102 while (left < right) {
103 auto middle = (left + right) / 2;
104 auto width = TextUtil::width(inlineTextItem, startPosition, middle + 1, contentLogicalLeft);
105 if (width < availableWidth) {
106 left = middle + 1;
107 leftSideWidth = width;
108 } else if (width > availableWidth)
109 right = middle;
110 else {
111 right = middle + 1;
112 leftSideWidth = width;
113 break;
114 }
115 }
116 return { startPosition, right - startPosition, leftSideWidth };
117}

119unsigned TextUtil::findNextBreakablePosition(LazyLineBreakIterator& lineBreakIterator, unsigned startPosition, const RenderStyle& style)
120{
121 auto keepAllWordsForCJK = style.wordBreak() == WordBreak::KeepAll;
122 auto breakNBSP = style.autoWrap() && style.nbspMode() == NBSPMode::Space;
123
124 if (keepAllWordsForCJK) {
125 if (breakNBSP)
126 return nextBreakablePositionKeepingAllWords(lineBreakIterator, startPosition);
127 return nextBreakablePositionKeepingAllWordsIgnoringNBSP(lineBreakIterator, startPosition);
128 }
129
130 if (lineBreakIterator.mode() == LineBreakIteratorMode::Default) {
131 if (breakNBSP)
132 return WebCore::nextBreakablePosition(lineBreakIterator, startPosition);
133 return nextBreakablePositionIgnoringNBSP(lineBreakIterator, startPosition);
134 }
135
136 if (breakNBSP)
137 return nextBreakablePositionWithoutShortcut(lineBreakIterator, startPosition);
138 return nextBreakablePositionIgnoringNBSPWithoutShortcut(lineBreakIterator, startPosition);
139}

141bool TextUtil::shouldPreserveSpacesAndTabs(const Box& layoutBox)
142{
143 // https://www.w3.org/TR/css-text-3/#white-space-property
144 auto whitespace = layoutBox.style().whiteSpace();
145 return whitespace == WhiteSpace::Pre || whitespace == WhiteSpace::PreWrap || whitespace == WhiteSpace::BreakSpaces;
146}
147
148bool TextUtil::shouldPreserveNewline(const Box& layoutBox)
149{
150 auto whitespace = layoutBox.style().whiteSpace();
151 // https://www.w3.org/TR/css-text-3/#white-space-property
152 return whitespace == WhiteSpace::Pre || whitespace == WhiteSpace::PreWrap || whitespace == WhiteSpace::BreakSpaces || whitespace == WhiteSpace::PreLine;
153}

155LineBreakIteratorMode TextUtil::lineBreakIteratorMode(LineBreak lineBreak)
156{
157 switch (lineBreak) {
158 case LineBreak::Auto:
159 case LineBreak::AfterWhiteSpace:
160 case LineBreak::Anywhere:
161 return LineBreakIteratorMode::Default;
162 case LineBreak::Loose:
163 return LineBreakIteratorMode::Loose;
164 case LineBreak::Normal:
165 return LineBreakIteratorMode::Normal;
166 case LineBreak::Strict:
167 return LineBreakIteratorMode::Strict;
168 }
169 ASSERT_NOT_REACHED();
170 return LineBreakIteratorMode::Default;
171}
