57void TableFormattingContext::layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent& constraints)
58{
59 auto availableHorizontalSpace = constraints.horizontal().logicalWidth;
60 auto availableVerticalSpace = downcast<ConstraintsForTableContent>(constraints).availableVerticalSpaceForContent();
61 // 1. Compute width and height for the grid.
62 computeAndDistributeExtraSpace(availableHorizontalSpace, availableVerticalSpace);
63 // 2. Finalize cells.
64 setUsedGeometryForCells(availableHorizontalSpace, availableVerticalSpace);
65 // 3. Finalize rows.
66 setUsedGeometryForRows(availableHorizontalSpace);
67 // 4. Finalize sections.
68 setUsedGeometryForSections(constraints);
69}

71LayoutUnit TableFormattingContext::usedContentHeight() const
72{
73 // Table has to have some section content, at least one <tbody>.
74 auto top = BoxGeometry::marginBoxRect(geometryForBox(*root().firstInFlowChild())).top();
75 auto bottom = BoxGeometry::marginBoxRect(geometryForBox(*root().lastInFlowChild())).bottom();
76 return bottom - top;
77}

79void TableFormattingContext::setUsedGeometryForCells(LayoutUnit availableHorizontalSpace, std::optional<LayoutUnit> availableVerticalSpace)
80{
81 auto& grid = formattingState().tableGrid();
82 auto& columnList = grid.columns().list();
83 auto& rowList = grid.rows().list();
84 auto& formattingGeometry = this->formattingGeometry();
85 // Final table cell layout. At this point all percentage values can be resolved.
86 auto sectionOffset = LayoutUnit { };
87 auto* currentSection = &rowList.first().box().parent();
88 for (auto& cell : grid.cells()) {
89 auto& cellBox = cell->box();
90 auto& cellBoxGeometry = formattingState().boxGeometry(cellBox);
91 auto& section = rowList[cell->startRow()].box().parent();
92 if (&section != currentSection) {
93 currentSection = &section;
94 // While the grid is a continuous flow of rows, in the display tree they are relative to their sections.
95 sectionOffset = rowList[cell->startRow()].logicalTop();
96 }
97 // Internal table elements do not have margins.
98 cellBoxGeometry.setHorizontalMargin({ });
99 cellBoxGeometry.setVerticalMargin({ });
100
101 cellBoxGeometry.setBorder(formattingGeometry.computedCellBorder(*cell));
102 cellBoxGeometry.setPadding(formattingGeometry.computedPadding(cellBox, availableHorizontalSpace));
103 cellBoxGeometry.setLogicalTop(rowList[cell->startRow()].logicalTop() - sectionOffset);
104 cellBoxGeometry.setLogicalLeft(columnList[cell->startColumn()].usedLogicalLeft());
105 cellBoxGeometry.setContentBoxWidth(formattingGeometry.horizontalSpaceForCellContent(*cell));
106
107 if (cellBox.hasInFlowOrFloatingChild()) {
108 auto invalidationState = InvalidationState { };
109 // FIXME: This should probably be part of the invalidation state to indicate when we re-layout the cell multiple times as part of the multi-pass table algorithm.
110 auto& floatingStateForCellContent = layoutState().ensureBlockFormattingState(cellBox).floatingState();
111 floatingStateForCellContent.clear();
112 LayoutContext::createFormattingContext(cellBox, layoutState())->layoutInFlowContent(invalidationState, formattingGeometry.constraintsForInFlowContent(cellBox));
113 }
114 cellBoxGeometry.setContentBoxHeight(formattingGeometry.verticalSpaceForCellContent(*cell, availableVerticalSpace));
115
116 auto computeIntrinsicVerticalPaddingForCell = [&] {
117 auto cellLogicalHeight = rowList[cell->startRow()].logicalHeight();
118 for (size_t rowIndex = cell->startRow() + 1; rowIndex < cell->endRow(); ++rowIndex)
119 cellLogicalHeight += rowList[rowIndex].logicalHeight();
120 cellLogicalHeight += (cell->rowSpan() - 1) * grid.verticalSpacing();
121 // Intrinsic padding is the extra padding for the cell box when it is shorter than the row. Cell boxes have to
122 // fill the available vertical space
123 // e.g <td height=100px></td><td height=1px></td>
124 // the second <td> ends up being 100px tall too with the extra intrinsic padding.
125
126 // FIXME: Find out if it is ok to use the regular padding here to align the content box inside a tall cell or we need to
127 // use some kind of intrinsic padding similar to RenderTableCell.
128 auto paddingTop = cellBoxGeometry.paddingTop().value_or(LayoutUnit { });
129 auto paddingBottom = cellBoxGeometry.paddingBottom().value_or(LayoutUnit { });
130 auto intrinsicPaddingTop = LayoutUnit { };
131 auto intrinsicPaddingBottom = LayoutUnit { };
132
133 switch (cellBox.style().verticalAlign()) {
134 case VerticalAlign::Middle: {
135 auto intrinsicVerticalPadding = std::max(0_lu, cellLogicalHeight - cellBoxGeometry.verticalMarginBorderAndPadding() - cellBoxGeometry.contentBoxHeight());
136 intrinsicPaddingTop = intrinsicVerticalPadding / 2;
137 intrinsicPaddingBottom = intrinsicVerticalPadding / 2;
138 break;
139 }
140 case VerticalAlign::Baseline: {
141 auto rowBaseline = LayoutUnit { rowList[cell->startRow()].baseline() };
142 auto cellBaseline = LayoutUnit { cell->baseline() };
143 intrinsicPaddingTop = std::max(0_lu, rowBaseline - cellBaseline - cellBoxGeometry.borderTop());
144 intrinsicPaddingBottom = std::max(0_lu, cellLogicalHeight - cellBoxGeometry.verticalMarginBorderAndPadding() - intrinsicPaddingTop - cellBoxGeometry.contentBoxHeight());
145 break;
146 }
147 default:
148 ASSERT_NOT_IMPLEMENTED_YET();
149 break;
150 }
151 if (intrinsicPaddingTop && cellBox.hasInFlowOrFloatingChild()) {
152 auto adjustCellContentWithInstrinsicPaddingBefore = [&] {
153 // Child boxes (and runs) are always in the coordinate system of the containing block's border box.
154 // The content box (where the child content lives) is inside the padding box, which is inside the border box.
155 // In order to compute the child box top/left position, we need to know both the padding and the border offsets.
156 // Normally by the time we start positioning the child content, we already have computed borders and padding for the containing block.
157 // This is different with table cells where the final padding offset depends on the content height as we use
158 // the padding box to vertically align the table cell content.
159 auto& formattingState = layoutState().establishedFormattingState(cellBox);
160 for (auto* child = cellBox.firstInFlowOrFloatingChild(); child; child = child->nextInFlowOrFloatingSibling()) {
161 if (child->isInlineTextBox())
162 continue;
163 formattingState.boxGeometry(*child).moveVertically(intrinsicPaddingTop);
164 }
165 if (cellBox.establishesInlineFormattingContext()) {
166 auto& inlineFormattingStatee = layoutState().establishedInlineFormattingState(cellBox);
167 for (auto& run : inlineFormattingStatee.lineRuns())
168 run.moveVertically(intrinsicPaddingTop);
169 for (auto& line : inlineFormattingStatee.lines())
170 line.moveVertically(intrinsicPaddingTop);
171 }
172 };
173 adjustCellContentWithInstrinsicPaddingBefore();
174 }
175 cellBoxGeometry.setVerticalPadding({ paddingTop + intrinsicPaddingTop, paddingBottom + intrinsicPaddingBottom });
176 };
177 computeIntrinsicVerticalPaddingForCell();
178 }

181void TableFormattingContext::setUsedGeometryForRows(LayoutUnit availableHorizontalSpace)
182{
183 auto& grid = formattingState().tableGrid();
184 auto& rows = grid.rows().list();
185
186 auto rowLogicalTop = grid.verticalSpacing();
187 const ContainerBox* previousRow = nullptr;
188 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
189 auto& row = rows[rowIndex];
190 auto& rowBox = row.box();
191 auto& rowBoxGeometry = formattingState().boxGeometry(rowBox);
192
193 rowBoxGeometry.setPadding(formattingGeometry().computedPadding(rowBox, availableHorizontalSpace));
194 // Internal table elements do not have margins.
195 rowBoxGeometry.setHorizontalMargin({ });
196 rowBoxGeometry.setVerticalMargin({ });
197
198 auto computedRowBorder = [&] {
199 auto border = formattingGeometry().computedBorder(rowBox);
200 if (!grid.collapsedBorder())
201 return border;
202 // Border collapsing delegates borders to table/cells.
203 border.horizontal = { };
204 if (!rowIndex)
205 border.vertical.top = { };
206 if (rowIndex == rows.size() - 1)
207 border.vertical.bottom = { };
208 return border;
209 }();
210 if (computedRowBorder.height() > row.logicalHeight()) {
211 // FIXME: This is an odd quirk when the row border overflows the row.
212 // We don't paint row borders so it does not matter too much, but if we don't
213 // set this fake border value, than we either end up with a negative content box
214 // or with a wide frame box.
215 // If it happens to cause issues in the display tree, we could also consider
216 // a special frame box override, where padding box + border != frame box.
217 computedRowBorder.vertical.top = { };
218 computedRowBorder.vertical.bottom = { };
219 }
220 rowBoxGeometry.setContentBoxHeight(row.logicalHeight() - computedRowBorder.height());
221
222 auto rowLogicalWidth = grid.columns().logicalWidth() + 2 * grid.horizontalSpacing();
223 if (computedRowBorder.width() > rowLogicalWidth) {
224 // See comment above.
225 computedRowBorder.horizontal.left = { };
226 computedRowBorder.horizontal.right = { };
227 }
228 rowBoxGeometry.setContentBoxWidth(rowLogicalWidth - computedRowBorder.width());
229 rowBoxGeometry.setBorder(computedRowBorder);
230
231 if (previousRow && &previousRow->parent() != &rowBox.parent()) {
232 // This row is in a different section.
233 rowLogicalTop = { };
234 }
235 rowBoxGeometry.setLogicalTop(rowLogicalTop);
236 rowBoxGeometry.setLogicalLeft({ });
237
238 rowLogicalTop += row.logicalHeight() + grid.verticalSpacing();
239 previousRow = &rowBox;
240 }
241
242 auto& columns = grid.columns();
243 Vector<InlineLayoutUnit> rowBaselines(rows.size(), 0);
244 // Now that cells are laid out, let's compute the row baselines.
245 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
246 for (size_t columnIndex = 0; columnIndex < columns.size(); ++columnIndex) {
247 auto& slot = *grid.slot({ columnIndex, rowIndex });
248 if (slot.isRowSpanned())
249 continue;
250 if (slot.hasRowSpan())
251 continue;
252 auto& cell = slot.cell();
253 rowBaselines[rowIndex] = std::max(rowBaselines[rowIndex], cell.baseline());
254 }
255 }
256 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex)
257 rows[rowIndex].setBaseline(rowBaselines[rowIndex]);
258}

260void TableFormattingContext::setUsedGeometryForSections(const ConstraintsForInFlowContent& constraints)
261{
262 auto& grid = formattingState().tableGrid();
263 auto& tableBox = root();
264 auto sectionWidth = grid.columns().logicalWidth() + 2 * grid.horizontalSpacing();
265 auto logicalTop = constraints.logicalTop();
266 auto verticalSpacing = grid.verticalSpacing();
267 auto paddingBefore = std::optional<LayoutUnit> { verticalSpacing };
268 auto paddingAfter = verticalSpacing;
269 for (auto& sectionBox : childrenOfType<ContainerBox>(tableBox)) {
270 auto& sectionBoxGeometry = formattingState().boxGeometry(sectionBox);
271 // Section borders are either collapsed or ignored.
272 sectionBoxGeometry.setBorder({ });
273 // Use fake vertical padding to space out the sections.
274 sectionBoxGeometry.setPadding(Edges { { }, { paddingBefore.value_or(0_lu), paddingAfter } });
275 paddingBefore = std::nullopt;
276 // Internal table elements do not have margins.
277 sectionBoxGeometry.setHorizontalMargin({ });
278 sectionBoxGeometry.setVerticalMargin({ });
279
280 sectionBoxGeometry.setContentBoxWidth(sectionWidth);
281 auto sectionContentHeight = LayoutUnit { };
282 size_t rowCount = 0;
283 for (auto& rowBox : childrenOfType<ContainerBox>(sectionBox)) {
284 sectionContentHeight += geometryForBox(rowBox).borderBoxHeight();
285 ++rowCount;
286 }
287 sectionContentHeight += verticalSpacing * (rowCount - 1);
288 sectionBoxGeometry.setContentBoxHeight(sectionContentHeight);
289 sectionBoxGeometry.setLogicalLeft(constraints.horizontal().logicalLeft);
290 sectionBoxGeometry.setLogicalTop(logicalTop);
291
292 logicalTop += sectionBoxGeometry.borderBoxHeight();
293 }
294}

296IntrinsicWidthConstraints TableFormattingContext::computedIntrinsicWidthConstraints()
297{
298 ASSERT(!root().isSizeContainmentBox());
299 // Tables have a slightly different concept of shrink to fit. It's really only different with non-auto "width" values, where
300 // a generic shrink-to fit block level box like a float box would be just sized to the computed value of "width", tables
301 // can actually be stretched way over.
302 auto& grid = formattingState().tableGrid();
303 if (auto computedWidthConstraints = grid.widthConstraints())
304 return *computedWidthConstraints;
305
306 // Compute the minimum/maximum width of each column.
307 auto computedWidthConstraints = computedPreferredWidthForColumns();
308 grid.setWidthConstraints(computedWidthConstraints);
309 return computedWidthConstraints;
310}
311
312IntrinsicWidthConstraints TableFormattingContext::computedPreferredWidthForColumns()
313{
314 auto& formattingState = this->formattingState();
315 auto& grid = formattingState.tableGrid();
316 ASSERT(!grid.widthConstraints());
317
318 // Column preferred width computation as follows:
319 // 1. Collect fixed column widths set by <colgroup>'s and <col>s
320 // 2. Collect each cells' width constraints and adjust fixed width column values.
321 // 3. Find the min/max width for each columns using the cell constraints and the <col> fixed widths but ignore column spans.
322 // 4. Distribute column spanning cells min/max widths.
323 // 5. Add them all up and return the computed min/max widths.
324 // 2. Collect the fixed width <col>s.
325 auto& columnList = grid.columns().list();
326 auto& formattingGeometry = this->formattingGeometry();
327 auto collectColsFixedWidth = [&] {
328 for (auto& column : columnList) {
329 auto fixedWidth = [&] () -> std::optional<LayoutUnit> {
330 auto* columnBox = column.box();
331 if (!columnBox) {
332 // Anonymous columns don't have associated layout boxes and can't have fixed col size.
333 return { };
334 }
335 if (auto width = columnBox->columnWidth())
336 return width;
337 return formattingGeometry.computedColumnWidth(*columnBox);
338 }();
339 if (fixedWidth)
340 column.setComputedLogicalWidth({ *fixedWidth, LengthType::Fixed });
341 }
342 };
343 collectColsFixedWidth();
344
345 auto hasColumnWithPercentWidth = false;
346 auto hasColumnWithFixedWidth = false;
347 Vector<std::optional<LayoutUnit>> maximumFixedColumnWidths(columnList.size());
348 Vector<std::optional<float>> maximumPercentColumnWidths(columnList.size());
349
350 auto collectCellsIntrinsicWidthConstraints = [&] {
351 for (auto& cell : grid.cells()) {
352 auto& cellBox = cell->box();
353 ASSERT(cellBox.establishesBlockFormattingContext());
354
355 auto intrinsicWidth = formattingState.intrinsicWidthConstraintsForBox(cellBox);
356 if (!intrinsicWidth) {
357 intrinsicWidth = formattingGeometry.intrinsicWidthConstraintsForCellContent(*cell);
358 formattingState.setIntrinsicWidthConstraintsForBox(cellBox, *intrinsicWidth);
359 }
360 auto cellPosition = cell->position();
361 auto& cellStyle = cellBox.style();
362 // Expand it with border and padding.
363 auto horizontalBorderAndPaddingWidth = formattingGeometry.computedCellBorder(*cell).width()
364 + formattingGeometry.fixedValue(cellStyle.paddingLeft()).value_or(0)
365 + formattingGeometry.fixedValue(cellStyle.paddingRight()).value_or(0);
366 intrinsicWidth->expand(horizontalBorderAndPaddingWidth);
367 // Spanner cells put their intrinsic widths on the initial slots.
368 grid.slot(cellPosition)->setWidthConstraints(*intrinsicWidth);
369
370 auto cellLogicalWidth = cellStyle.logicalWidth();
371 auto columnIndex = cellPosition.column;
372 switch (cellLogicalWidth.type()) {
373 case LengthType::Fixed: {
374 auto fixedWidth = LayoutUnit { cellLogicalWidth.value() } + horizontalBorderAndPaddingWidth;
375 maximumFixedColumnWidths[columnIndex] = std::max(maximumFixedColumnWidths[columnIndex].value_or(0_lu), fixedWidth);
376 hasColumnWithFixedWidth = true;
377 break;
378 }
379 case LengthType::Percent: {
380 maximumPercentColumnWidths[columnIndex] = std::max(maximumPercentColumnWidths[columnIndex].value_or(0.f), cellLogicalWidth.percent());
381 hasColumnWithPercentWidth = true;
382 break;
383 }
384 case LengthType::Relative:
385 ASSERT_NOT_IMPLEMENTED_YET();
386 break;
387 default:
388 break;
389 }
390 }
391 };
392 collectCellsIntrinsicWidthConstraints();
393
394 Vector<IntrinsicWidthConstraints> columnIntrinsicWidths(columnList.size());
395 Vector<SlotPosition> spanningCellPositionList;
396 size_t numberOfActualColumns = 0;
397 auto computeColumnsIntrinsicWidthConstraints = [&] {
398 // 3. Collect he min/max width for each column but ignore column spans for now.
399 for (size_t columnIndex = 0; columnIndex < columnList.size(); ++columnIndex) {
400 auto columnHasNonSpannedCell = false;
401 for (size_t rowIndex = 0; rowIndex < grid.rows().size(); ++rowIndex) {
402 auto& slot = *grid.slot({ columnIndex, rowIndex });
403 if (slot.isColumnSpanned())
404 continue;
405 columnHasNonSpannedCell = true;
406 if (slot.hasColumnSpan()) {
407 spanningCellPositionList.append({ columnIndex, rowIndex });
408 continue;
409 }
410 auto widthConstraints = slot.widthConstraints();
411 if (auto fixedColumnWidth = maximumFixedColumnWidths[columnIndex])
412 widthConstraints.maximum = std::max(*fixedColumnWidth, widthConstraints.minimum);
413
414 columnIntrinsicWidths[columnIndex].minimum = std::max(widthConstraints.minimum, columnIntrinsicWidths[columnIndex].minimum);
415 columnIntrinsicWidths[columnIndex].maximum = std::max(widthConstraints.maximum, columnIntrinsicWidths[columnIndex].maximum);
416 }
417 if (columnHasNonSpannedCell)
418 ++numberOfActualColumns;
419 }
420 };
421 computeColumnsIntrinsicWidthConstraints();
422
423 auto resolveSpanningCells = [&] {
424 // 4. Distribute the spanning min/max widths.
425 for (auto spanningCellPosition : spanningCellPositionList) {
426 auto& slot = *grid.slot(spanningCellPosition);
427 auto& cell = slot.cell();
428 ASSERT(slot.hasColumnSpan());
429 auto widthConstraintsToDistribute = slot.widthConstraints();
430 for (size_t columnSpanIndex = cell.startColumn(); columnSpanIndex < cell.endColumn(); ++columnSpanIndex)
431 widthConstraintsToDistribute -= columnIntrinsicWidths[columnSpanIndex];
432 // <table style="border-spacing: 50px"><tr><td colspan=2>long long text</td></tr><tr><td>lo</td><td>xt</td><tr></table>
433 // [long long text]
434 // [lo] [xt]
435 // While it looks like the spanning cell has to distribute all its spanning width, the border-spacing takes most of the space and
436 // no distribution is needed at all.
437 widthConstraintsToDistribute -= (cell.columnSpan() - 1) * grid.horizontalSpacing();
438 // FIXME: Check if fixed width columns should be skipped here.
439 widthConstraintsToDistribute.minimum = std::max(LayoutUnit { }, widthConstraintsToDistribute.minimum / cell.columnSpan());
440 widthConstraintsToDistribute.maximum = std::max(LayoutUnit { }, widthConstraintsToDistribute.maximum / cell.columnSpan());
441 if (widthConstraintsToDistribute.minimum || widthConstraintsToDistribute.maximum) {
442 for (size_t columnSpanIndex = cell.startColumn(); columnSpanIndex < cell.endColumn(); ++columnSpanIndex)
443 columnIntrinsicWidths[columnSpanIndex] += widthConstraintsToDistribute;
444 }
445 }
446 };
447 resolveSpanningCells();
448
449 // 5. The table min/max widths is just the accumulated column constraints with the percent adjustment.
450 auto tableWidthConstraints = IntrinsicWidthConstraints { };
451 for (auto& columnIntrinsicWidth : columnIntrinsicWidths)
452 tableWidthConstraints += columnIntrinsicWidth;
453
454 auto adjustColumnsWithPercentAndFixedWidthValues = [&] {
455 // 6. Adjust the table max width with the percent column values if applicable.
456 if (!hasColumnWithFixedWidth && !hasColumnWithPercentWidth)
457 return;
458
459 if (hasColumnWithFixedWidth && !hasColumnWithPercentWidth) {
460 for (size_t columnIndex = 0; columnIndex < columnList.size(); ++columnIndex) {
461 if (auto fixedWidth = maximumFixedColumnWidths[columnIndex])
462 columnList[columnIndex].setComputedLogicalWidth({ *fixedWidth, LengthType::Fixed });
463 }
464 return;
465 }
466
467 auto remainingPercent = 100.0f;
468 auto percentMaximumWidth = LayoutUnit { };
469 auto nonPercentColumnsWidth = LayoutUnit { };
470 // Resolve the percent values as follows
471 // - the percent value is resolved against the column maximum width (fixed or content based) as if the max value represented the percentage value
472 // e.g 50% with the maximum width of 100px produces a resolved width of 200px for the column.
473 // - find the largest resolved value across the columns and used that as the maximum width for the percent based columns.
474 // - Compute the non-percent based columns width by using the remaining percent value (e.g 50% and 10% columns would leave 40% for the rest of the columns)
475 for (size_t columnIndex = 0; columnIndex < columnList.size(); ++columnIndex) {
476 auto nonPercentColumnWidth = columnIntrinsicWidths[columnIndex].maximum;
477 if (auto fixedWidth = maximumFixedColumnWidths[columnIndex]) {
478 columnList[columnIndex].setComputedLogicalWidth({ *fixedWidth, LengthType::Fixed });
479 nonPercentColumnWidth = std::max(nonPercentColumnWidth, *fixedWidth);
480 }
481 if (!maximumPercentColumnWidths[columnIndex]) {
482 nonPercentColumnsWidth += nonPercentColumnWidth;
483 continue;
484 }
485 auto percent = std::min(*maximumPercentColumnWidths[columnIndex], remainingPercent);
486 columnList[columnIndex].setComputedLogicalWidth({ percent, LengthType::Percent });
487 percentMaximumWidth = std::max(percentMaximumWidth, LayoutUnit { nonPercentColumnWidth * 100.0f / percent });
488 remainingPercent -= percent;
489 }
490 ASSERT(remainingPercent >= 0.f);
491 auto adjustedMaximumWidth = percentMaximumWidth;
492 if (remainingPercent)
493 adjustedMaximumWidth = std::max(adjustedMaximumWidth, LayoutUnit { nonPercentColumnsWidth * 100.0f / remainingPercent });
494 else {
495 // When the table has percent width column(s) and they add up to (or over) 100%, the maximum width is computed to
496 // only constrained by the available horizontal width.
497 // This is a very odd transition of going from 99.9% to 100%, where 99.9% computes normally (see above)
498 // but as soon as we hit the 100% mark, the table suddenly stretches all the way to the horizontal available space.
499 // It may very well be an ancient bug we need to support (it maps to the epsilon value in AutoTableLayout::computeIntrinsicLogicalWidths which is to avoid division by zero).
500 adjustedMaximumWidth = LayoutUnit::max();
501 }
502 tableWidthConstraints.maximum = std::max(tableWidthConstraints.maximum, adjustedMaximumWidth);
503 };
504 adjustColumnsWithPercentAndFixedWidthValues();
505
506 // Expand the preferred width with leading and trailing cell spacing (note that column spanners count as one cell).
507 tableWidthConstraints += (numberOfActualColumns + 1) * grid.horizontalSpacing();
508 return tableWidthConstraints;
509}

511void TableFormattingContext::computeAndDistributeExtraSpace(LayoutUnit availableHorizontalSpace, std::optional<LayoutUnit> availableVerticalSpace)
512{
513 // Compute and balance the column and row spaces.
514 auto& grid = formattingState().tableGrid();
515 auto& columns = grid.columns().list();
516 auto tableLayout = this->tableLayout();
517
518 // Columns first.
519 auto distributedHorizontalSpaces = tableLayout.distributedHorizontalSpace(availableHorizontalSpace);
520 ASSERT(distributedHorizontalSpaces.size() == columns.size());
521 auto columnLogicalLeft = grid.horizontalSpacing();
522 for (size_t columnIndex = 0; columnIndex < columns.size(); ++columnIndex) {
523 auto& column = columns[columnIndex];
524 column.setUsedLogicalLeft(columnLogicalLeft);
525 column.setUsedLogicalWidth(distributedHorizontalSpaces[columnIndex]);
526 columnLogicalLeft += distributedHorizontalSpaces[columnIndex] + grid.horizontalSpacing();
527 }
528
529 auto& formattingGeometry = this->formattingGeometry();
530 // Rows second.
531 auto& rows = grid.rows().list();
532 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
533 for (size_t columnIndex = 0; columnIndex < columns.size(); ++columnIndex) {
534 auto& slot = *grid.slot({ columnIndex, rowIndex });
535 if (slot.isRowSpanned())
536 continue;
537 auto layoutCellContent = [&](auto& cell) {
538 auto& cellBox = cell.box();
539 auto& cellBoxGeometry = formattingState().boxGeometry(cellBox);
540 cellBoxGeometry.setBorder(formattingGeometry.computedCellBorder(cell));
541 cellBoxGeometry.setPadding(formattingGeometry.computedPadding(cellBox, availableHorizontalSpace));
542 cellBoxGeometry.setContentBoxWidth(formattingGeometry.horizontalSpaceForCellContent(cell));
543
544 if (cellBox.hasInFlowOrFloatingChild()) {
545 auto invalidationState = InvalidationState { };
546 LayoutContext::createFormattingContext(cellBox, layoutState())->layoutInFlowContent(invalidationState, formattingGeometry.constraintsForInFlowContent(cellBox));
547 }
548 cellBoxGeometry.setContentBoxHeight(formattingGeometry.verticalSpaceForCellContent(cell, availableVerticalSpace));
549 };
550 layoutCellContent(slot.cell());
551 if (slot.hasRowSpan())
552 continue;
553 // The minimum height of a row (without spanning-related height distribution) is defined as the height of an hypothetical
554 // linebox containing the cells originating in the row.
555 auto& cell = slot.cell();
556 cell.setBaseline(formattingGeometry.usedBaselineForCell(cell.box()));
557 }
558 }
559
560 auto distributedVerticalSpaces = tableLayout.distributedVerticalSpace(availableVerticalSpace);
561 ASSERT(distributedVerticalSpaces.size() == rows.size());
562 auto rowLogicalTop = grid.verticalSpacing();
563 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
564 auto& row = rows[rowIndex];
565 row.setLogicalHeight(distributedVerticalSpaces[rowIndex]);
566 row.setLogicalTop(rowLogicalTop);
567 rowLogicalTop += distributedVerticalSpaces[rowIndex] + grid.verticalSpacing();
568 }
569}
570
571}
