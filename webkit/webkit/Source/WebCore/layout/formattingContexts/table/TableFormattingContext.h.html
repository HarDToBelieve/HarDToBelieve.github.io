38namespace WebCore {
39namespace Layout {
40
41class InvalidationState;
42
43// This class implements the layout logic for table formatting contexts.
44// https://www.w3.org/TR/CSS22/tables.html
45class TableFormattingContext final : public FormattingContext {
46 WTF_MAKE_ISO_ALLOCATED(TableFormattingContext);
47public:
48 TableFormattingContext(const ContainerBox& formattingContextRoot, TableFormattingState&);
49 void layoutInFlowContent(InvalidationState&, const ConstraintsForInFlowContent&) override;
50 LayoutUnit usedContentHeight() const override;
51
52 const TableFormattingGeometry& formattingGeometry() const final { return m_tableFormattingGeometry; }
53 const TableFormattingQuirks& formattingQuirks() const final { return m_tableFormattingQuirks; }
54 const TableFormattingState& formattingState() const { return downcast<TableFormattingState>(FormattingContext::formattingState()); }
55
56private:
57 class TableLayout {
58 public:
59 TableLayout(const TableFormattingContext&, const TableGrid&);
60
61 using DistributedSpaces = Vector<LayoutUnit>;
62 DistributedSpaces distributedHorizontalSpace(LayoutUnit availableHorizontalSpace);
63 DistributedSpaces distributedVerticalSpace(std::optional<LayoutUnit> availableVerticalSpace);
64
65 private:
66 const TableFormattingContext& formattingContext() const { return m_formattingContext; }
67
68 const TableFormattingContext& m_formattingContext;
69 const TableGrid& m_grid;
70 };
71
72 TableFormattingContext::TableLayout tableLayout() const { return TableLayout(*this, formattingState().tableGrid()); }
73
74 IntrinsicWidthConstraints computedIntrinsicWidthConstraints() override;
75 void setUsedGeometryForCells(LayoutUnit availableHorizontalSpace, std::optional<LayoutUnit> availableVerticalSpace);
76 void setUsedGeometryForRows(LayoutUnit availableHorizontalSpace);
77 void setUsedGeometryForSections(const ConstraintsForInFlowContent&);
78
79 IntrinsicWidthConstraints computedPreferredWidthForColumns();
80 void computeAndDistributeExtraSpace(LayoutUnit availableHorizontalSpace, std::optional<LayoutUnit> availableVerticalSpace);
81
82 TableFormattingState& formattingState() { return downcast<TableFormattingState>(FormattingContext::formattingState()); }
83
84 const TableFormattingGeometry m_tableFormattingGeometry;
85 const TableFormattingQuirks m_tableFormattingQuirks;
86};
87
88}
89}
