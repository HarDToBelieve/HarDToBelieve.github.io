86void TableGrid::appendCell(const ContainerBox& cellBox)
87{
88 auto rowSpan = cellBox.rowSpan();
89 auto columnSpan = cellBox.columnSpan();
90 auto isInNewRow = !cellBox.previousSibling();
91 auto initialSlotPosition = SlotPosition { };
92
93 if (!m_cells.isEmpty()) {
94 auto& lastCell = m_cells.last();
95 auto lastSlotPosition = lastCell->position();
96 // First table cell in this row?
97 if (isInNewRow)
98 initialSlotPosition = SlotPosition { 0, lastSlotPosition.row + 1 };
99 else
100 initialSlotPosition = SlotPosition { lastSlotPosition.column + 1, lastSlotPosition.row };
101
102 // Pick the next available slot by avoiding row and column spanners.
103 while (true) {
104 if (!m_slotMap.contains(initialSlotPosition))
105 break;
106 ++initialSlotPosition.column;
107 }
108 }
109 auto cell = makeUnique<Cell>(cellBox, initialSlotPosition, CellSpan { columnSpan, rowSpan });
110 // Row and column spanners create additional slots.
111 for (size_t row = 0; row < rowSpan; ++row) {
112 for (auto column = cell->startColumn(); column < cell->endColumn(); ++column) {
113 auto position = SlotPosition { column, cell->startRow() + row };
114 ASSERT(!m_slotMap.contains(position));
115 // This slot is spanned by a cell at the initial slow position.
116 auto isColumnSpanned = column != cell->startColumn();
117 auto isRowSpanned = !!row;
118 m_slotMap.add(position, makeUnique<Slot>(*cell, isColumnSpanned, isRowSpanned));
119 }
120 }
121 // Initialize columns/rows if needed.
122 auto missingNumberOfColumns = std::max<int>(0, initialSlotPosition.column + columnSpan - m_columns.size());
123 for (auto column = 0; column < missingNumberOfColumns; ++column)
124 m_columns.addAnonymousColumn();
125
126 if (isInNewRow)
127 m_rows.addRow(cellBox.parent());
128
129 m_cells.add(WTFMove(cell));
130}
