45struct ColumnSpan {
46 static size_t hasSpan(const TableGrid::Slot& slot) { return slot.hasColumnSpan(); }
47 static size_t isSpanned(const TableGrid::Slot& slot) { return slot.isColumnSpanned(); }
48
49 static size_t spanCount(const TableGrid::Cell& cell) { return cell.columnSpan(); }
50 static size_t startSpan(const TableGrid::Cell& cell) { return cell.startColumn(); }
51 static size_t endSpan(const TableGrid::Cell& cell) { return cell.endColumn(); }
52
53 static size_t index(size_t columnIndex, size_t /*rowIndex*/) { return columnIndex; }
54 static size_t size(const TableGrid& grid) { return grid.columns().size(); }
55
56 static LayoutUnit spacing(const TableGrid& grid) { return grid.horizontalSpacing(); }
57};

73struct GridSpace {
74 bool isEmpty() const { return !preferredSize; }
75
76 enum class Type {
77 Percent,
78 Fixed,
79 Relative,
80 Auto
81 };
82 Type type { Type::Auto };
83 // Preferred width/height for column/row we start the distribution with (usually the minimum content width).
84 float preferredSize { 0 };
85 // The base space value for the distribution computation e.g [ 9 ] [ 1 ] values for 2 columns in the table means that if
86 // we've go 100px flexible space to distribute, 90px and 10px go to the columns respectively.
87 float flexBase { 0 };
88 float minimumSize { 0 };
89};
90
91inline static GridSpace& operator-(GridSpace& a, const GridSpace& b)
92{
93 a.preferredSize = std::max(0.0f, a.preferredSize - b.preferredSize);
94 a.flexBase = std::max(0.0f, a.flexBase - b.flexBase);
95 return a;
96}
97
98inline static GridSpace& operator+=(GridSpace& a, const GridSpace& b)
99{
100 a.preferredSize += b.preferredSize;
101 a.flexBase += b.flexBase;
102 return a;
103}
104
105inline static GridSpace& operator-=(GridSpace& a, const GridSpace& b)
106{
107 return a - b;
108}
109
110inline static GridSpace& operator/(GridSpace& a, unsigned value)
111{
112 a.preferredSize /= value;
113 a.flexBase /= value;
114 return a;
115}

117template <typename SpanType>
118static Vector<LayoutUnit> distributeAvailableSpace(const TableGrid& grid, LayoutUnit availableSpace, const WTF::Function<GridSpace(const TableGrid::Slot&, size_t)>& slotSpace)
119{
120 auto& columns = grid.columns();
121 auto& rows = grid.rows();
122 Vector<std::optional<GridSpace>> resolvedItems(SpanType::size(grid));
123 auto collectNonSpanningCells = [&] {
124 // 1. Collect the non-spanning spaces first. They are used for the final distribution as well as for distributing the spanning space.
125 for (size_t columnIndex = 0; columnIndex < columns.size(); ++columnIndex) {
126 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
127 auto& slot = *grid.slot({ columnIndex, rowIndex });
128 if (SpanType::hasSpan(slot) || SpanType::isSpanned(slot))
129 continue;
130 auto index = SpanType::index(columnIndex, rowIndex);
131 auto currentSpace = slotSpace(slot, index);
132 if (!resolvedItems[index]) {
133 resolvedItems[index] = currentSpace;
134 continue;
135 }
136 auto& resolvedItem = resolvedItems[index];
137 // FIXME: Add support for mixed column/row types e.g. first row first column is fixed, while second row first column is percent.
138 ASSERT(resolvedItem->type == currentSpace.type);
139 resolvedItem->preferredSize = std::max(resolvedItem->preferredSize, currentSpace.preferredSize);
140 resolvedItem->flexBase = std::max(resolvedItem->flexBase, currentSpace.flexBase);
141 }
142 }
143 };
144 collectNonSpanningCells();
145
146 auto collectAndDistributeSpanningCells = [&] {
147 // 2. Collect the spanning cells.
148 struct SpanningCell {
149 SlotPosition position;
150 GridSpace unresolvedSpace;
151 };
152 Vector<SpanningCell> spanningCells;
153 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
154 for (size_t columnIndex = 0; columnIndex < columns.size(); ++columnIndex) {
155 auto& slot = *grid.slot({ columnIndex, rowIndex });
156 if (SpanType::hasSpan(slot))
157 spanningCells.append({ { columnIndex, rowIndex }, slotSpace(slot, SpanType::index(columnIndex, rowIndex)) });
158 }
159 }
160 // We need these spanning cells in the order of the number of columns/rows they span so that
161 // we can resolve overlapping spans starting with the shorter ones e.g.
162 // <td colspan=4>#a</td><td>#b</td>
163 // <td colspan=2>#c</td><td colspan=3>#d</td>
164 std::sort(spanningCells.begin(), spanningCells.end(), [&] (auto& a, auto& b) {
165 return SpanType::spanCount(grid.slot(a.position)->cell()) < SpanType::spanCount(grid.slot(b.position)->cell());
166 });
167
168 // 3. Distribute the spanning cells' mimimum space across the columns/rows using the non-spanning spaces.
169 // e.g. [ 1 ][ 5 ][ 1 ]
170 // [ 9 ][ 1 ]
171 // The initial widths are: [ 2 ][ 7 ][ 1 ]
172 for (auto spanningCell : spanningCells) {
173 auto& cell = grid.slot(spanningCell.position)->cell();
174 auto unresolvedSpanningSpace = spanningCell.unresolvedSpace;
175 if (!resolvedItems[SpanType::startSpan(cell)] || !resolvedItems[SpanType::endSpan(cell) - 1]) {
176 // <td colspan=4>#a</td><td>#b</td>
177 // <td colspan=2>#c</td><td colspan=3>#d</td>
178 // Unresolved columns are: 1 2 3 4
179 // 1. Take colspan=2 (shortest span) and resolve column 1 and 2
180 // 2. Take colspan=3 and resolve column 3 and 4 (5 is resolved because it already has a non-spanning cell).
181 // 3. colspan=4 needs no resolving because all the spanned columns (1 2 3 4) have already been resolved.
182 auto unresolvedColumnCount = cell.columnSpan();
183 for (auto spanIndex = SpanType::startSpan(cell); spanIndex < SpanType::endSpan(cell); ++spanIndex) {
184 if (!resolvedItems[spanIndex])
185 continue;
186 ASSERT(unresolvedColumnCount);
187 --unresolvedColumnCount;
188 unresolvedSpanningSpace -= *resolvedItems[spanIndex];
189 }
190 ASSERT(unresolvedColumnCount);
191 auto equalSpaceForSpannedColumns = unresolvedSpanningSpace / unresolvedColumnCount;
192 for (auto spanIndex = SpanType::startSpan(cell); spanIndex < SpanType::endSpan(cell); ++spanIndex) {
193 if (resolvedItems[spanIndex])
194 continue;
195 resolvedItems[spanIndex] = equalSpaceForSpannedColumns;
196 }
197 } else {
198 // 1. Collect the non-spanning resolved spaces.
199 // 2. Distribute the extra space among the spanned columns/rows based on the resolved space values.
200 // e.g. spanning width: [ 9 ]. Resolved widths for the spanned columns: [ 1 ] [ 2 ]
201 // New resolved widths: [ 3 ] [ 6 ].
202 auto resolvedSpanningSpace = GridSpace { };
203 for (auto spanIndex = SpanType::startSpan(cell); spanIndex < SpanType::endSpan(cell); ++spanIndex)
204 resolvedSpanningSpace += *resolvedItems[spanIndex];
205 if (resolvedSpanningSpace.preferredSize >= unresolvedSpanningSpace.preferredSize) {
206 // The spanning cell fits the spanned columns/rows just fine. Nothing to distribute.
207 continue;
208 }
209 auto spacing = SpanType::spacing(grid) * (SpanType::spanCount(cell) - 1);
210 auto spaceToDistribute = unresolvedSpanningSpace - GridSpace { { }, spacing, spacing } - resolvedSpanningSpace;
211 if (!spaceToDistribute.isEmpty()) {
212 auto columnsFlexBase = spaceToDistribute.flexBase / resolvedSpanningSpace.flexBase;
213 for (auto spanIndex = SpanType::startSpan(cell); spanIndex < SpanType::endSpan(cell); ++spanIndex)
214 *resolvedItems[spanIndex] += GridSpace { resolvedItems[spanIndex]->type, resolvedItems[spanIndex]->preferredSize * columnsFlexBase, resolvedItems[spanIndex]->flexBase * columnsFlexBase };
215 }
216 }
217 }
218 };
219 collectAndDistributeSpanningCells();
220
221 Vector<LayoutUnit> distributedSpaces(resolvedItems.size());
222 auto spaceToDistribute = 0.f;
223 auto computeSpaceToDistribute = [&] {
224 // 4. Distribute the extra space using the final resolved sizes.
225 spaceToDistribute = availableSpace - (resolvedItems.size() + 1) * SpanType::spacing(grid);
226 auto adjustabledSpace = GridSpace { };
227 for (auto& resolvedItem : resolvedItems) {
228 ASSERT(resolvedItem);
229 adjustabledSpace += *resolvedItem;
230 }
231 spaceToDistribute -= adjustabledSpace.preferredSize;
232 };
233 computeSpaceToDistribute();
234
235 // Let's start with the preferred size for each column.
236 for (size_t columnIndex = 0; columnIndex < resolvedItems.size(); ++columnIndex)
237 distributedSpaces[columnIndex] = LayoutUnit { resolvedItems[columnIndex]->preferredSize };
238
239 auto distributeSpace = [&] {
240 if (!spaceToDistribute)
241 return;
242 // Setup the priority lists. We use these when expanding/shrinking slots.
243 Vector<size_t> autoColumnIndexes;
244 Vector<size_t> relativeColumnIndexes;
245 Vector<size_t> fixedColumnIndexes;
246 Vector<size_t> percentColumnIndexes;
247
248 for (size_t columnIndex = 0; columnIndex < resolvedItems.size(); ++columnIndex) {
249 switch (resolvedItems[columnIndex]->type) {
250 case GridSpace::Type::Percent:
251 percentColumnIndexes.append(columnIndex);
252 break;
253 case GridSpace::Type::Fixed:
254 fixedColumnIndexes.append(columnIndex);
255 break;
256 case GridSpace::Type::Relative:
257 relativeColumnIndexes.append(columnIndex);
258 break;
259 case GridSpace::Type::Auto:
260 autoColumnIndexes.append(columnIndex);
261 break;
262 default:
263 ASSERT_NOT_REACHED();
264 break;
265 }
266 }
267
268 if (spaceToDistribute > 0) {
269 // Each column can get some extra space.
270 auto hasSpaceToDistribute = [&] {
271 ASSERT(spaceToDistribute > -LayoutUnit::epsilon());
272 return spaceToDistribute > LayoutUnit::epsilon();
273 };
274
275 auto expandSpace = [&](const auto& columnIndexes) {
276 auto adjustabledSpace = GridSpace { };
277 for (auto& columnIndex : columnIndexes)
278 adjustabledSpace += *resolvedItems[columnIndex];
279
280 auto columnsFlexBase = adjustabledSpace.flexBase ? spaceToDistribute / adjustabledSpace.flexBase : 0.f;
281 for (auto& columnIndex : columnIndexes) {
282 auto extraSpace = columnsFlexBase * resolvedItems[columnIndex]->flexBase;
283 distributedSpaces[columnIndex] += LayoutUnit { extraSpace };
284 spaceToDistribute -= extraSpace;
285 if (!hasSpaceToDistribute())
286 return;
287 }
288 };
289 // We distribute the extra space among columns in the priority order as follows:
290 expandSpace(fixedColumnIndexes);
291 if (hasSpaceToDistribute())
292 expandSpace(percentColumnIndexes);
293 if (hasSpaceToDistribute())
294 expandSpace(relativeColumnIndexes);
295 if (hasSpaceToDistribute())
296 expandSpace(autoColumnIndexes);
297 ASSERT(!hasSpaceToDistribute());
298 return;
299 }
300 // Can't accommodate the preferred width. Let's use the priority list to shrink columns.
301 auto spaceNeeded = -spaceToDistribute;
302
303 auto needsMoreSpace = [&] {
304 ASSERT(spaceNeeded > -LayoutUnit::epsilon());
305 return spaceNeeded > LayoutUnit::epsilon();
306 };
307
308 auto shrinkSpace = [&](const auto& columnIndexes) {
309 auto adjustabledSpace = GridSpace { };
310 for (auto& columnIndex : columnIndexes)
311 adjustabledSpace += *resolvedItems[columnIndex];
312
313 auto columnsFlexBase = adjustabledSpace.flexBase ? spaceNeeded / adjustabledSpace.flexBase : 0.f;
314 for (auto& columnIndex : columnIndexes) {
315 auto& resolvedItem = *resolvedItems[columnIndex];
316 auto spaceToRemove = std::min(resolvedItem.preferredSize - resolvedItem.minimumSize, columnsFlexBase * resolvedItem.flexBase);
317 spaceToRemove = std::min(spaceToRemove, spaceNeeded);
318 distributedSpaces[columnIndex] -= spaceToRemove;
319 spaceNeeded -= spaceToRemove;
320 if (!needsMoreSpace())
321 return;
322 }
323 };
324 shrinkSpace(autoColumnIndexes);
325 if (needsMoreSpace())
326 shrinkSpace(relativeColumnIndexes);
327 if (needsMoreSpace())
328 shrinkSpace(fixedColumnIndexes);
329 if (needsMoreSpace())
330 shrinkSpace(percentColumnIndexes);
331 ASSERT(!needsMoreSpace());
332 };
333 distributeSpace();
334
335 return distributedSpaces;
336}

338TableFormattingContext::TableLayout::DistributedSpaces TableFormattingContext::TableLayout::distributedHorizontalSpace(LayoutUnit availableHorizontalSpace)
339{
340 auto hasEnoughAvailableSpaceForMaximumWidth = availableHorizontalSpace >= m_grid.widthConstraints()->maximum;
341 return distributeAvailableSpace<ColumnSpan>(m_grid, availableHorizontalSpace, [&] (const TableGrid::Slot& slot, size_t columnIndex) {
342 auto& column = m_grid.columns().list()[columnIndex];
343 auto columnWidth = std::optional<float> { };
344 auto type = GridSpace::Type::Auto;
345
346 auto& computedLogicalWidth = column.computedLogicalWidth();
347 switch (computedLogicalWidth.type()) {
348 case LengthType::Fixed:
349 columnWidth = computedLogicalWidth.value();
350 type = GridSpace::Type::Fixed;
351 break;
352 case LengthType::Percent:
353 columnWidth = computedLogicalWidth.value() * availableHorizontalSpace / 100.0f;
354 type = GridSpace::Type::Percent;
355 break;
356 case LengthType::Relative:
357 ASSERT_NOT_IMPLEMENTED_YET();
358 break;
359 default:
360 break;
361 }
362
363 float minimumContentWidth = slot.widthConstraints().minimum;
364 float maximumContentWidth = slot.widthConstraints().maximum;
365 auto preferredWidth = std::max(minimumContentWidth, columnWidth.value_or(maximumContentWidth));
366 if (!hasEnoughAvailableSpaceForMaximumWidth) {
367 // maximum width > available horizontal space
368 // preferred width: it's always at least as wide as the minimum content width and if column fixed width is set then the greater of the two.
369 // base value for adjustable space distribution for a cell: preferred width - minimum content width
370 // total adjustable horizontal space: available horizontal space - preferred width(s)
371 // column extra width: total adjustable space / base value for distribution for all columns * base value for distribution for this column
372 // column final width: preferred width + extra width
373 // e.g
374 // <table style="width: 120px">
375 // <td style="width: 80px;">some content</td>
376 // <td style="width: 20px;">some content></td>
377 // </table>
378 // maximum width (160px) > available horizontal space (120)
379 // "some content" -> minimum content width is 50px
380 // preferred widths: 80px and 50px
381 // base values for distribution: 30px and 0px
382 // total adjustable space: 120px - 80px - 50px = -10px <- Note the negative available space with over-constrained values.
383 // columns extra widths: -10px / (30px + 0px) * 30px = -10px and -10px / (30px + 0px) * 0px = 0px
384 // columns final widths: 70px and 50px (first column is narrower than its preferred width and the second is as wide as the minimum content width)
385 return GridSpace { type, preferredWidth, preferredWidth - minimumContentWidth, minimumContentWidth };
386 }
387
388 // maximum width <= available horizontal space
389 // preferred width: it's usually the maximum content width but at least as wide as the minimum content width (or the column fixed width)
390 // base value for adjustable space distribution for a cell: preferred width (again usually the maximum content width)
391 // total adjustable horizontal space: available horizontal space - preferred width(s)
392 // column extra width: total adjustable space / base value for distribution for all columns * base value for distribution for this column
393 // column final width: preferred width + extra width
394 // e.g
395 // <table style="width: 2600px">
396 // <td style="width: 40px;"><div style="width: 200px;"></div></td>
397 // <td style="width: 300px;"><div style="width: 100px;"></div></td>
398 // </table>
399 // maximum width (500px) < available horizontal space (2600px)
400 // preferred widths: 200px and 300px
401 // base values for distribution: 200px and 300px
402 // total adjustable space: 2600px - 500px = 2100px
403 // columns extra widths: 2100px / 500px * 200px = 840px and 2100px / 500px * 300px = 1260px
404 // columns final widths: 1040px and 1560px
405 return GridSpace { type, preferredWidth, preferredWidth, minimumContentWidth };
406 });
407}

409TableFormattingContext::TableLayout::DistributedSpaces TableFormattingContext::TableLayout::distributedVerticalSpace(std::optional<LayoutUnit> availableVerticalSpace)
410{
411 auto& rows = m_grid.rows();
412 auto& columns = m_grid.columns();
413
414 Vector<LayoutUnit> rowHeight(rows.size());
415 auto tableUsedHeight = LayoutUnit { };
416 // 2. Collect initial, baseline aligned row heights.
417 for (size_t rowIndex = 0; rowIndex < rows.size(); ++rowIndex) {
418 auto maximumColumnAscent = InlineLayoutUnit { };
419 auto maximumColumnDescent = InlineLayoutUnit { };
420 // Initial minimum height is the computed height if available <tr style="height: 100px"><td></td></tr>
421 rowHeight[rowIndex] = formattingContext().formattingGeometry().computedHeight(rows.list()[rowIndex].box(), availableVerticalSpace).value_or(0_lu);
422 for (size_t columnIndex = 0; columnIndex < columns.size(); ++columnIndex) {
423 auto& slot = *m_grid.slot({ columnIndex, rowIndex });
424 if (slot.isRowSpanned())
425 continue;
426 if (slot.hasRowSpan())
427 continue;
428 // The minimum height of a row (without spanning-related height distribution) is defined as the height of an hypothetical
429 // linebox containing the cells originating in the row.
430 auto& cell = slot.cell();
431 auto& cellBox = cell.box();
432 auto height = formattingContext().geometryForBox(cellBox).borderBoxHeight();
433 if (cellBox.style().verticalAlign() == VerticalAlign::Baseline) {
434 maximumColumnAscent = std::max(maximumColumnAscent, cell.baseline());
435 maximumColumnDescent = std::max(maximumColumnDescent, height - cell.baseline());
436 rowHeight[rowIndex] = std::max(rowHeight[rowIndex], LayoutUnit { maximumColumnAscent + maximumColumnDescent });
437 } else
438 rowHeight[rowIndex] = std::max(rowHeight[rowIndex], height);
439 }
440 tableUsedHeight += rowHeight[rowIndex];
441 }
442 // FIXME: Collect spanning row maximum heights.
443
444 tableUsedHeight += (rows.size() + 1) * m_grid.verticalSpacing();
445 auto availableSpace = std::max(availableVerticalSpace.value_or(0_lu), tableUsedHeight);
446 // Distribute extra space if the table is supposed to be taller than the sum of the row heights.
447 return distributeAvailableSpace<RowSpan>(m_grid, availableSpace, [&] (const TableGrid::Slot& slot, size_t rowIndex) {
448 if (slot.hasRowSpan()) {
449 auto borderBoxHeight = formattingContext().geometryForBox(slot.cell().box()).borderBoxHeight();
450 return GridSpace { { }, borderBoxHeight, borderBoxHeight, borderBoxHeight };
451 }
452 auto& rows = m_grid.rows();
453 auto computedRowHeight = formattingContext().formattingGeometry().computedHeight(rows.list()[rowIndex].box(), { });
454 auto height = std::max<float>(rowHeight[rowIndex], computedRowHeight.value_or(0_lu));
455 return GridSpace { { }, height, height, height };
456 });
457}
