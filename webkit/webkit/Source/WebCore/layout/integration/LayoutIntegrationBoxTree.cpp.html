44static constexpr size_t smallTreeThreshold = 8;
45
46static RenderStyle rootBoxStyle(const RenderStyle& style)
47{
48 auto clonedStyle = RenderStyle::clone(style);
49 clonedStyle.setEffectiveDisplay(DisplayType::Block);
50 return clonedStyle;
51}
52
53BoxTree::BoxTree(RenderBlockFlow& flow)
54 : m_flow(flow)
55 , m_root(rootBoxStyle(flow.style()))
56{
57 if (flow.isAnonymous())
58 m_root.setIsAnonymous();
59
60 buildTree();
61}
62
63void BoxTree::buildTree()
64{
65 auto createChildBox = [&](RenderObject& childRenderer) -> std::unique_ptr<Layout::Box> {
66 if (is<RenderText>(childRenderer)) {
67 auto& textRenderer = downcast<RenderText>(childRenderer);
68 auto style = RenderStyle::createAnonymousStyleWithDisplay(textRenderer.style(), DisplayType::Inline);
69 return makeUnique<Layout::InlineTextBox>(textRenderer.text(), textRenderer.canUseSimplifiedTextMeasuring(), WTFMove(style));
70 }
71
72 auto style = RenderStyle::clone(childRenderer.style());
73 if (childRenderer.isLineBreak()) {
74 style.setDisplay(DisplayType::Inline);
75 style.setFloating(Float::None);
76 style.setPosition(PositionType::Static);
77 return makeUnique<Layout::LineBreakBox>(downcast<RenderLineBreak>(childRenderer).isWBR(), WTFMove(style));
78 }
79
80 if (is<RenderReplaced>(childRenderer))
81 return makeUnique<Layout::ReplacedBox>(Layout::Box::ElementAttributes { is<RenderImage>(childRenderer) ? Layout::Box::ElementType::Image : Layout::Box::ElementType::GenericElement }, WTFMove(style));
82
83 if (is<RenderBlock>(childRenderer))
84 return makeUnique<Layout::ReplacedBox>(Layout::Box::ElementAttributes { Layout::Box::ElementType::GenericElement }, WTFMove(style));
85
86 if (is<RenderInline>(childRenderer)) {
87 if (childRenderer.parent()->isAnonymousBlock()) {
88 // This looks like continuation renderer.
89 auto& renderInline = downcast<RenderInline>(childRenderer);
90 auto shouldNotRetainBorderPaddingAndMarginStart = renderInline.isContinuation();
91 auto shouldNotRetainBorderPaddingAndMarginEnd = !renderInline.isContinuation() && renderInline.inlineContinuation();
92 if (shouldNotRetainBorderPaddingAndMarginStart) {
93 style.setMarginStart(RenderStyle::initialMargin());
94 style.resetBorderLeft();
95 style.setPaddingLeft(RenderStyle::initialPadding());
96 }
97 if (shouldNotRetainBorderPaddingAndMarginEnd) {
98 style.setMarginEnd(RenderStyle::initialMargin());
99 style.resetBorderRight();
100 style.setPaddingRight(RenderStyle::initialPadding());
101 }
102 }
103 return makeUnique<Layout::ContainerBox>(Layout::Box::ElementAttributes { Layout::Box::ElementType::GenericElement }, WTFMove(style));
104 }
105
106 ASSERT_NOT_REACHED();
107 return nullptr;
108 };
109
110 for (auto walker = InlineWalker(m_flow); !walker.atEnd(); walker.advance()) {
111 if (walker.atEndOfInline())
112 continue;
113 auto& childRenderer = *walker.current();
114 auto childBox = createChildBox(childRenderer);
115 appendChild(WTFMove(childBox), childRenderer);
116 }
117}

119void BoxTree::appendChild(std::unique_ptr<Layout::Box> childBox, RenderObject& childRenderer)
120{
121 auto& parentBox = downcast<Layout::ContainerBox>(layoutBoxForRenderer(*childRenderer.parent()));
122 parentBox.appendChild(*childBox);
123
124 m_boxes.append({ WTFMove(childBox), &childRenderer });
125
126 if (m_boxes.size() > smallTreeThreshold) {
127 if (m_rendererToBoxMap.isEmpty()) {
128 for (auto& entry : m_boxes)
129 m_rendererToBoxMap.add(entry.renderer, entry.box.get());
130 } else
131 m_rendererToBoxMap.add(&childRenderer, m_boxes.last().box.get());
132 }
133}

135void BoxTree::updateStyle(const RenderBoxModelObject& renderer)
136{
137 auto& layoutBox = layoutBoxForRenderer(renderer);
138 auto& style = renderer.style();
139 if (&layoutBox == &rootLayoutBox())
140 layoutBox.updateStyle(rootBoxStyle(style));
141 else
142 layoutBox.updateStyle(style);
143
144 if (is<Layout::ContainerBox>(layoutBox)) {
145 for (auto* child = downcast<Layout::ContainerBox>(layoutBox).firstChild(); child; child = child->nextSibling()) {
146 if (child->isInlineTextBox())
147 child->updateStyle(RenderStyle::createAnonymousStyleWithDisplay(style, DisplayType::Inline));
148 }
149 }
150}

152Layout::Box& BoxTree::layoutBoxForRenderer(const RenderObject& renderer)
153{
154 if (&renderer == &m_flow)
155 return m_root;
156
157 if (m_boxes.size() <= smallTreeThreshold) {
158 auto index = m_boxes.findMatching([&](auto& entry) {
159 return entry.renderer == &renderer;
160 });
161 RELEASE_ASSERT(index != notFound);
162 return *m_boxes[index].box;
163 }
164
165 return *m_rendererToBoxMap.get(&renderer);
166}
167
168const Layout::Box& BoxTree::layoutBoxForRenderer(const RenderObject& renderer) const
169{
170 return const_cast<BoxTree&>(*this).layoutBoxForRenderer(renderer);
171}
172
173RenderObject& BoxTree::rendererForLayoutBox(const Layout::Box& box)
174{
175 if (&box == &m_root)
176 return m_flow;
177
178 if (m_boxes.size() <= smallTreeThreshold) {
179 auto index = m_boxes.findMatching([&](auto& entry) {
180 return entry.box.get() == &box;
181 });
182 RELEASE_ASSERT(index != notFound);
183 return *m_boxes[index].renderer;
184 }
185
186 if (m_boxToRendererMap.isEmpty()) {
187 for (auto& entry : m_boxes)
188 m_boxToRendererMap.add(entry.box.get(), entry.renderer);
189 }
190 return *m_boxToRendererMap.get(&box);
191}
192
193const RenderObject& BoxTree::rendererForLayoutBox(const Layout::Box& box) const
194{
195 return const_cast<BoxTree&>(*this).rendererForLayoutBox(box);
196}
