290static void printTextForSubtree(const RenderObject& renderer, unsigned& charactersLeft, TextStream& stream)
291{
292 if (!charactersLeft)
293 return;
294 if (is<RenderText>(renderer)) {
295 String text = downcast<RenderText>(renderer).text();
296 text = text.stripWhiteSpace();
297 unsigned len = std::min(charactersLeft, text.length());
298 stream << text.left(len);
299 charactersLeft -= len;
300 return;
301 }
302 if (!is<RenderElement>(renderer))
303 return;
304 for (const auto* child = downcast<RenderElement>(renderer).firstChild(); child; child = child->nextSibling())
305 printTextForSubtree(*child, charactersLeft, stream);
306}

308static unsigned textLengthForSubtree(const RenderObject& renderer)
309{
310 if (is<RenderText>(renderer))
311 return downcast<RenderText>(renderer).text().length();
312 if (!is<RenderElement>(renderer))
313 return 0;
314 unsigned textLength = 0;
315 for (const auto* child = downcast<RenderElement>(renderer).firstChild(); child; child = child->nextSibling())
316 textLength += textLengthForSubtree(*child);
317 return textLength;
318}

320static void collectNonEmptyLeafRenderBlockFlows(const RenderObject& renderer, HashSet<const RenderBlockFlow*>& leafRenderers)
321{
322 if (is<RenderText>(renderer)) {
323 if (!downcast<RenderText>(renderer).text().length())
324 return;
325 // Find RenderBlockFlow ancestor.
326 for (const auto* current = renderer.parent(); current; current = current->parent()) {
327 if (!is<RenderBlockFlow>(current))
328 continue;
329 leafRenderers.add(downcast<RenderBlockFlow>(current));
330 break;
331 }
332 return;
333 }
334 if (!is<RenderElement>(renderer))
335 return;
336 for (const auto* child = downcast<RenderElement>(renderer).firstChild(); child; child = child->nextSibling())
337 collectNonEmptyLeafRenderBlockFlows(*child, leafRenderers);
338}

340static void collectNonEmptyLeafRenderBlockFlowsForCurrentPage(HashSet<const RenderBlockFlow*>& leafRenderers)
341{
342 for (const auto* document : Document::allDocuments()) {
343 if (!document->renderView() || document->backForwardCacheState() != Document::NotInBackForwardCache)
344 continue;
345 if (!document->isHTMLDocument() && !document->isXHTMLDocument())
346 continue;
347 collectNonEmptyLeafRenderBlockFlows(*document->renderView(), leafRenderers);
348 }
349}

351static void printModernLineLayoutBlockList(void)
352{
353 HashSet<const RenderBlockFlow*> leafRenderers;
354 collectNonEmptyLeafRenderBlockFlowsForCurrentPage(leafRenderers);
355 if (!leafRenderers.size()) {
356 WTFLogAlways("No text found in this document\n");
357 return;
358 }
359 TextStream stream;
360 stream << "---------------------------------------------------\n";
361 for (const auto* flow : leafRenderers) {
362 auto reasons = canUseForLineLayoutWithReason(*flow, IncludeReasons::All);
363 if (reasons.isEmpty())
364 continue;
365 unsigned printedLength = 30;
366 stream << "\"";
367 printTextForSubtree(*flow, printedLength, stream);
368 for (;printedLength > 0; --printedLength)
369 stream << " ";
370 stream << "\"(" << textLengthForSubtree(*flow) << "):";
371 printReasons(reasons, stream);
372 stream << "\n";
373 }
374 stream << "---------------------------------------------------\n";
375 WTFLogAlways("%s", stream.release().utf8().data());
376}

378static void printModernLineLayoutCoverage(void)
379{
380 HashSet<const RenderBlockFlow*> leafRenderers;
381 collectNonEmptyLeafRenderBlockFlowsForCurrentPage(leafRenderers);
382 if (!leafRenderers.size()) {
383 WTFLogAlways("No text found in this document\n");
384 return;
385 }
386 TextStream stream;
387 HashMap<AvoidanceReason, unsigned, DefaultHash<uint64_t>, WTF::UnsignedWithZeroKeyHashTraits<uint64_t>> flowStatistics;
388 unsigned textLength = 0;
389 unsigned unsupportedTextLength = 0;
390 unsigned numberOfUnsupportedLeafBlocks = 0;
391 unsigned supportedButForcedToLineLayoutTextLength = 0;
392 unsigned numberOfSupportedButForcedToLineLayoutLeafBlocks = 0;
393 for (const auto* flow : leafRenderers) {
394 auto flowLength = textLengthForSubtree(*flow);
395 textLength += flowLength;
396 auto reasons = canUseForLineLayoutWithReason(*flow, IncludeReasons::All);
397 if (reasons.isEmpty()) {
398 if (flow->lineLayoutPath() == RenderBlockFlow::ForceLineBoxesPath) {
399 supportedButForcedToLineLayoutTextLength += flowLength;
400 ++numberOfSupportedButForcedToLineLayoutLeafBlocks;
401 }
402 continue;
403 }
404 ++numberOfUnsupportedLeafBlocks;
405 unsupportedTextLength += flowLength;
406 for (auto reason : reasons) {
407 auto result = flowStatistics.add(static_cast<uint64_t>(reason), flowLength);
408 if (!result.isNewEntry)
409 result.iterator->value += flowLength;
410 }
411 }
412 stream << "---------------------------------------------------\n";
413 if (supportedButForcedToLineLayoutTextLength) {
414 stream << "Modern line layout potential coverage: " << (float)(textLength - unsupportedTextLength) / (float)textLength * 100 << "%\n\n";
415 stream << "Modern line layout actual coverage: " << (float)(textLength - unsupportedTextLength - supportedButForcedToLineLayoutTextLength) / (float)textLength * 100 << "%\nForced line layout blocks: " << numberOfSupportedButForcedToLineLayoutLeafBlocks << " content length: " << supportedButForcedToLineLayoutTextLength << "(" << (float)supportedButForcedToLineLayoutTextLength / (float)textLength * 100 << "%)";
416 } else
417 stream << "Modern line layout coverage: " << (float)(textLength - unsupportedTextLength) / (float)textLength * 100 << "%";
418 stream << "\n\n";
419 stream << "Number of blocks: total(" << leafRenderers.size() << ") legacy(" << numberOfUnsupportedLeafBlocks << ")\nContent length: total(" <<
420 textLength << ") legacy(" << unsupportedTextLength << ")\n";
421 for (const auto& reasonEntry : flowStatistics) {
422 printReason(static_cast<AvoidanceReason>(reasonEntry.key), stream);
423 stream << ": " << (float)reasonEntry.value / (float)textLength * 100 << "%\n";
424 }
425 stream << "---------------------------------------------------\n";
426 WTFLogAlways("%s", stream.release().utf8().data());
427}

432template<> OptionSet<AvoidanceReason> canUseForCharacter(UChar character, bool textIsJustified, IncludeReasons includeReasons)
433{
434 OptionSet<AvoidanceReason> reasons;
435 if (textIsJustified) {
436 if (character == noBreakSpace)
437 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasJustifiedNonBreakingSpace, reasons, includeReasons);
438 // Include characters up to Latin Extended-B and some punctuation range when text is justified.
439 bool isLatinIncludingExtendedB = character <= 0x01FF;
440 bool isPunctuationRange = character >= 0x2010 && character <= 0x2027;
441 if (!(isLatinIncludingExtendedB || isPunctuationRange))
442 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasJustifiedNonLatinText, reasons, includeReasons);
443 }
444
445 if (U16_IS_SURROGATE(character))
446 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextHasSurrogatePair, reasons, includeReasons);
447
448 UCharDirection direction = u_charDirection(character);
449 if (direction == U_RIGHT_TO_LEFT || direction == U_RIGHT_TO_LEFT_ARABIC
450 || direction == U_RIGHT_TO_LEFT_EMBEDDING || direction == U_RIGHT_TO_LEFT_OVERRIDE
451 || direction == U_LEFT_TO_RIGHT_EMBEDDING || direction == U_LEFT_TO_RIGHT_OVERRIDE
452 || direction == U_POP_DIRECTIONAL_FORMAT || direction == U_BOUNDARY_NEUTRAL)
453 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextHasDirectionCharacter, reasons, includeReasons);
454
455 return reasons;
456}

458template<> OptionSet<AvoidanceReason> canUseForCharacter(LChar character, bool textIsJustified, IncludeReasons)
459{
460 if (textIsJustified && character == noBreakSpace)
461 return { AvoidanceReason::FlowHasJustifiedNonBreakingSpace };
462 return { };
463}

465template <typename CharacterType>
466static OptionSet<AvoidanceReason> canUseForText(const CharacterType* text, unsigned length, const FontCascade& fontCascade, std::optional<float> lineHeightConstraint,
467 bool textIsJustified, IncludeReasons includeReasons)
468{
469 OptionSet<AvoidanceReason> reasons;
470 auto& primaryFont = fontCascade.primaryFont();
471 auto& fontMetrics = primaryFont.fontMetrics();
472 auto availableSpaceForGlyphAscent = fontMetrics.ascent();
473 auto availableSpaceForGlyphDescent = fontMetrics.descent();
474 if (lineHeightConstraint) {
475 auto lineHeightPadding = *lineHeightConstraint - fontMetrics.height();
476 availableSpaceForGlyphAscent += lineHeightPadding / 2;
477 availableSpaceForGlyphDescent += lineHeightPadding / 2;
478 }
479
480 for (unsigned i = 0; i < length; ++i) {
481 auto character = text[i];
482 auto characterReasons = canUseForCharacter(character, textIsJustified, includeReasons);
483 if (characterReasons)
484 ADD_REASONS_AND_RETURN_IF_NEEDED(characterReasons, reasons, includeReasons);
485
486 auto glyphData = fontCascade.glyphDataForCharacter(character, false);
487 if (!glyphData.isValid() || glyphData.font != &primaryFont)
488 SET_REASON_AND_RETURN_IF_NEEDED(FlowPrimaryFontIsInsufficient, reasons, includeReasons);
489
490 if (lineHeightConstraint) {
491 auto bounds = primaryFont.boundsForGlyph(glyphData.glyph);
492 if (ceilf(-bounds.y()) > availableSpaceForGlyphAscent || ceilf(bounds.maxY()) > availableSpaceForGlyphDescent)
493 SET_REASON_AND_RETURN_IF_NEEDED(FlowFontHasOverflowGlyph, reasons, includeReasons);
494 }
495 }
496 return reasons;
497}

499static OptionSet<AvoidanceReason> canUseForText(StringView text, const FontCascade& fontCascade, std::optional<float> lineHeightConstraint, bool textIsJustified, IncludeReasons includeReasons)
500{
501 if (text.is8Bit())
502 return canUseForText(text.characters8(), text.length(), fontCascade, lineHeightConstraint, textIsJustified, includeReasons);
503 return canUseForText(text.characters16(), text.length(), fontCascade, lineHeightConstraint, textIsJustified, includeReasons);
504}

506static OptionSet<AvoidanceReason> canUseForFontAndText(const RenderBoxModelObject& container, IncludeReasons includeReasons)
507{
508 OptionSet<AvoidanceReason> reasons;
509 // We assume that all lines have metrics based purely on the primary font.
510 const auto& style = container.style();
511 auto& fontCascade = style.fontCascade();
512 if (fontCascade.primaryFont().isInterstitial())
513 SET_REASON_AND_RETURN_IF_NEEDED(FlowIsMissingPrimaryFont, reasons, includeReasons);
514 std::optional<float> lineHeightConstraint;
515 if (style.lineBoxContain().contains(LineBoxContain::Glyphs))
516 lineHeightConstraint = container.lineHeight(false, HorizontalLine, PositionOfInteriorLineBoxes).toFloat();
517 bool flowIsJustified = style.textAlign() == TextAlignMode::Justify;
518 for (const auto& textRenderer : childrenOfType<RenderText>(container)) {
519 // FIXME: Do not return until after checking all children.
520 if (textRenderer.isCombineText())
521 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextIsCombineText, reasons, includeReasons);
522 if (textRenderer.isCounter())
523 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextIsRenderCounter, reasons, includeReasons);
524 if (textRenderer.isQuote())
525 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextIsRenderQuote, reasons, includeReasons);
526 if (textRenderer.isTextFragment())
527 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextIsTextFragment, reasons, includeReasons);
528 if (textRenderer.isSVGInlineText())
529 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextIsSVGInlineText, reasons, includeReasons);
530 if (!textRenderer.canUseSimpleFontCodePath()) {
531 // No need to check the code path at this point. We already know it can't be simple.
532 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasComplexFontCodePath, reasons, includeReasons);
533 } else {
534 WebCore::TextRun run(String(textRenderer.text()));
535 run.setCharacterScanForCodePath(false);
536 if (style.fontCascade().codePath(run) != FontCascade::CodePath::Simple)
537 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasComplexFontCodePath, reasons, includeReasons);
538 }
539
540 auto textReasons = canUseForText(textRenderer.stringView(), fontCascade, lineHeightConstraint, flowIsJustified, includeReasons);
541 if (textReasons)
542 ADD_REASONS_AND_RETURN_IF_NEEDED(textReasons, reasons, includeReasons);
543 }
544 return reasons;
545}

547static OptionSet<AvoidanceReason> canUseForStyle(const RenderStyle& style, IncludeReasons includeReasons)
548{
549 OptionSet<AvoidanceReason> reasons;
550 if ((style.overflowX() != Overflow::Visible && style.overflowX() != Overflow::Hidden)
551 || (style.overflowY() != Overflow::Visible && style.overflowY() != Overflow::Hidden))
552 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasOverflowNotVisible, reasons, includeReasons);
553 if (style.textOverflow() == TextOverflow::Ellipsis)
554 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextOverflow, reasons, includeReasons);
555 if (!style.textDecorationsInEffect().isEmpty() && (style.textUnderlinePosition() != TextUnderlinePosition::Auto || !style.textUnderlineOffset().isAuto() || !style.textDecorationThickness().isAuto()))
556 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasUnsupportedUnderlineDecoration, reasons, includeReasons);
557 if (!style.isLeftToRightDirection())
558 SET_REASON_AND_RETURN_IF_NEEDED(FlowIsNotLTR, reasons, includeReasons);
559 if (!(style.lineBoxContain().contains(LineBoxContain::Block)))
560 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasLineBoxContainProperty, reasons, includeReasons);
561 if (style.writingMode() != WritingMode::TopToBottom)
562 SET_REASON_AND_RETURN_IF_NEEDED(FlowIsNotTopToBottom, reasons, includeReasons);
563 if (style.unicodeBidi() != UBNormal)
564 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasNonNormalUnicodeBiDi, reasons, includeReasons);
565 if (style.rtlOrdering() != Order::Logical)
566 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasRTLOrdering, reasons, includeReasons);
567 if (style.lineAlign() != LineAlign::None)
568 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasLineAlignEdges, reasons, includeReasons);
569 if (style.lineSnap() != LineSnap::None)
570 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasLineSnap, reasons, includeReasons);
571 if (style.textEmphasisFill() != TextEmphasisFill::Filled || style.textEmphasisMark() != TextEmphasisMark::None)
572 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextEmphasisFillOrMark, reasons, includeReasons);
573 if (style.textShadow())
574 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextShadow, reasons, includeReasons);
575 if (style.hasPseudoStyle(PseudoId::FirstLine))
576 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasPseudoFirstLine, reasons, includeReasons);
577 if (style.hasPseudoStyle(PseudoId::FirstLetter))
578 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasPseudoFirstLetter, reasons, includeReasons);
579 if (style.hasTextCombine())
580 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextCombine, reasons, includeReasons);
581 if (style.backgroundClip() == FillBox::Text)
582 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextFillBox, reasons, includeReasons);
583 if (style.borderFit() == BorderFit::Lines)
584 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasBorderFitLines, reasons, includeReasons);
585 if (style.lineBreak() != LineBreak::Auto)
586 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasNonAutoLineBreak, reasons, includeReasons);
587 if (style.nbspMode() != NBSPMode::Normal)
588 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasWebKitNBSPMode, reasons, includeReasons);
589 // Special handling of text-security:disc is not yet implemented in the simple line layout code path.
590 // See RenderBlock::updateSecurityDiscCharacters.
591 if (style.textSecurity() != TextSecurity::None)
592 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextSecurity, reasons, includeReasons);
593 if (style.hyphens() == Hyphens::Auto) {
594 auto textReasons = canUseForText(style.hyphenString(), style.fontCascade(), std::nullopt, false, includeReasons);
595 if (textReasons)
596 ADD_REASONS_AND_RETURN_IF_NEEDED(textReasons, reasons, includeReasons);
597 }
598 return reasons;
599}

601static OptionSet<AvoidanceReason> canUseForChild(const RenderBlockFlow& flow, const RenderObject& child, IncludeReasons includeReasons)
602{
603 OptionSet<AvoidanceReason> reasons;
604 if (child.selectionState() != RenderObject::HighlightState::None)
605 SET_REASON_AND_RETURN_IF_NEEDED(FlowChildIsSelected, reasons, includeReasons);
606 if (is<RenderCounter>(child))
607 SET_REASON_AND_RETURN_IF_NEEDED(FlowTextIsRenderCounter, reasons, includeReasons);
608 if (is<RenderText>(child)) {
609 const auto& renderText = downcast<RenderText>(child);
610 if (renderText.textNode() && !renderText.document().markers().markersFor(*renderText.textNode()).isEmpty())
611 SET_REASON_AND_RETURN_IF_NEEDED(FlowIncludesDocumentMarkers, reasons, includeReasons);
612 return reasons;
613 }
614
615 if (flow.containsFloats()) {
616 // Non-text content may stretch the line and we don't yet have support for dynamic float avoiding (as the line grows).
617 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasUnsupportedFloat, reasons, includeReasons);
618 }
619
620 if (is<RenderLineBreak>(child))
621 return reasons;
622
623 if (child.isFieldset()) {
624 // Fieldsets don't follow the standard CSS box model. They require special handling.
625 SET_REASON_AND_RETURN_IF_NEEDED(UnsupportedFieldset, reasons, includeReasons)
626 }
627
628#if ALLOW_IMAGES || ALLOW_ALL_REPLACED || ALLOW_INLINE_BLOCK
629 auto isSupportedStyle = [] (const auto& style) {
630 if (style.verticalAlign() == VerticalAlign::Sub || style.verticalAlign() == VerticalAlign::Super)
631 return false;
632 if (style.width().isPercent() || style.height().isPercent())
633 return false;
634 if (style.minWidth().isPercent() || style.maxWidth().isPercent())
635 return false;
636 if (style.minHeight().isPercent() || style.maxHeight().isPercent())
637 return false;
638 if (style.boxShadow())
639 return false;
640 return true;
641 };
642#endif
643#if ALLOW_IMAGES || ALLOW_ALL_REPLACED
644 if (is<RenderReplaced>(child)) {
645 auto& replaced = downcast<RenderReplaced>(child);
646 if (replaced.isFloating() || replaced.isPositioned())
647 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxIsFloatingOrPositioned, reasons, includeReasons)
648
649 if (replaced.isSVGRoot())
650 SET_REASON_AND_RETURN_IF_NEEDED(ContentIsSVG, reasons, includeReasons);
651
652 if (!isSupportedStyle(replaced.style()))
653 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxHasUnsupportedStyle, reasons, includeReasons);
654
655 if (is<RenderImage>(replaced)) {
656 auto& image = downcast<RenderImage>(replaced);
657 if (image.imageMap())
658 SET_REASON_AND_RETURN_IF_NEEDED(UnsupportedImageMap, reasons, includeReasons);
659 return reasons;
660 }
661#if !ALLOW_ALL_REPLACED
662 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasNonSupportedChild, reasons, includeReasons);
663#endif
664 return reasons;
665 }
666#endif
667
668#if ALLOW_INLINE_BLOCK
669 if (is<RenderBlockFlow>(child)) {
670 auto& block = downcast<RenderBlockFlow>(child);
671 if (!block.isReplaced() || !block.isInline())
672 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasNonSupportedChild, reasons, includeReasons)
673 if (block.isFloating() || block.isPositioned())
674 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxIsFloatingOrPositioned, reasons, includeReasons)
675 if (block.isRubyRun())
676 SET_REASON_AND_RETURN_IF_NEEDED(ContentIsRuby, reasons, includeReasons);
677
678 auto& style = block.style();
679 if (!isSupportedStyle(style))
680 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxHasUnsupportedStyle, reasons, includeReasons)
681 if (style.display() != DisplayType::InlineBlock)
682 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxHasUnsupportedStyle, reasons, includeReasons)
683
684 return reasons;
685 }
686#endif
687
688#if ALLOW_INLINES
689 if (is<RenderInline>(child)) {
690 auto& renderInline = downcast<RenderInline>(child);
691 if (renderInline.isSVGInline())
692 SET_REASON_AND_RETURN_IF_NEEDED(ContentIsSVG, reasons, includeReasons);
693 if (renderInline.isRubyInline() || renderInline.isQuote())
694 SET_REASON_AND_RETURN_IF_NEEDED(ContentIsRuby, reasons, includeReasons);
695 if (renderInline.requiresLayer())
696 SET_REASON_AND_RETURN_IF_NEEDED(InlineBoxNeedsLayer, reasons, includeReasons)
697 if (flow.fragmentedFlowState() != RenderObject::NotInsideFragmentedFlow)
698 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasNonSupportedChild, reasons, includeReasons);
699
700 auto& style = renderInline.style();
701 if (!isSupportedStyle(style))
702 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxHasUnsupportedStyle, reasons, includeReasons)
703 if (style.hasBorder() || style.borderImage().hasImage())
704 SET_REASON_AND_RETURN_IF_NEEDED(InlineBoxHasBorderOrBorderImage, reasons, includeReasons);
705 if (style.hasBackground())
706 SET_REASON_AND_RETURN_IF_NEEDED(InlineBoxHasBackground, reasons, includeReasons);
707 if (style.hasOutline())
708 SET_REASON_AND_RETURN_IF_NEEDED(ContentHasOutline, reasons, includeReasons);
709 if (renderInline.marginLeft() < 0 || renderInline.marginRight() < 0 || renderInline.marginTop() < 0 || renderInline.marginBottom() < 0
710 || renderInline.paddingLeft() < 0 || renderInline.paddingRight() < 0 || renderInline.paddingTop() < 0 || renderInline.paddingBottom() < 0)
711 SET_REASON_AND_RETURN_IF_NEEDED(InlineBoxHasMarginOrPadding, reasons, includeReasons);
712 if (renderInline.isInFlowPositioned())
713 SET_REASON_AND_RETURN_IF_NEEDED(ChildBoxIsFloatingOrPositioned, reasons, includeReasons);
714 if (renderInline.containingBlock()->style().lineBoxContain() != RenderStyle::initialLineBoxContain())
715 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasLineBoxContainProperty, reasons, includeReasons);
716 auto fontAndTextReasons = canUseForFontAndText(downcast<RenderInline>(child), includeReasons);
717 if (fontAndTextReasons)
718 ADD_REASONS_AND_RETURN_IF_NEEDED(fontAndTextReasons, reasons, includeReasons);
719 auto styleReasons = canUseForStyle(style, includeReasons);
720 if (styleReasons)
721 ADD_REASONS_AND_RETURN_IF_NEEDED(styleReasons, reasons, includeReasons);
722
723 return reasons;
724 }
725#endif
726
727 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasNonSupportedChild, reasons, includeReasons);
728 return reasons;
729}

731OptionSet<AvoidanceReason> canUseForLineLayoutWithReason(const RenderBlockFlow& flow, IncludeReasons includeReasons)
732{
733#ifndef NDEBUG
734 static std::once_flag onceFlag;
735 std::call_once(onceFlag, [] {
736 PAL::registerNotifyCallback("com.apple.WebKit.showModernLineLayoutCoverage", WTF::Function<void()> { printModernLineLayoutCoverage });
737 PAL::registerNotifyCallback("com.apple.WebKit.showModernLineLayoutReasons", WTF::Function<void()> { printModernLineLayoutBlockList });
738 });
739#endif
740 OptionSet<AvoidanceReason> reasons;
741 // FIXME: For tests that disable SLL and expect to get CLL.
742 if (!flow.settings().simpleLineLayoutEnabled())
743 SET_REASON_AND_RETURN_IF_NEEDED(FeatureIsDisabled, reasons, includeReasons);
744 auto establishesInlineFormattingContext = [&] {
745 if (flow.isRenderView()) {
746 // RenderView initiates a block formatting context.
747 return false;
748 }
749 ASSERT(flow.parent());
750 // FIXME: This should really get the first _inflow_ child.
751 auto* firstChild = flow.firstChild();
752 if (!firstChild) {
753 // Empty block containers do not initiate inline formatting context.
754 return false;
755 }
756 return firstChild->isInline() || firstChild->isInlineBlockOrInlineTable();
757 };
758 if (!establishesInlineFormattingContext())
759 SET_REASON_AND_RETURN_IF_NEEDED(FlowDoesNotEstablishInlineFormattingContext, reasons, includeReasons);
760 if (flow.fragmentedFlowState() != RenderObject::NotInsideFragmentedFlow) {
761 auto* fragmentedFlow = flow.enclosingFragmentedFlow();
762 if (!is<RenderMultiColumnFlow>(fragmentedFlow))
763 SET_REASON_AND_RETURN_IF_NEEDED(FlowIsInsideANonMultiColumnThread, reasons, includeReasons);
764 auto& columnThread = downcast<RenderMultiColumnFlow>(*fragmentedFlow);
765 if (columnThread.parent() != &flow.view())
766 SET_REASON_AND_RETURN_IF_NEEDED(MultiColumnFlowIsNotTopLevel, reasons, includeReasons);
767 if (columnThread.hasColumnSpanner())
768 SET_REASON_AND_RETURN_IF_NEEDED(MultiColumnFlowHasColumnSpanner, reasons, includeReasons);
769 auto& style = flow.style();
770 if (style.verticalAlign() != VerticalAlign::Baseline)
771 SET_REASON_AND_RETURN_IF_NEEDED(MultiColumnFlowVerticalAlign, reasons, includeReasons);
772 if (style.isFloating())
773 SET_REASON_AND_RETURN_IF_NEEDED(MultiColumnFlowIsFloating, reasons, includeReasons);
774 }
775 if (!flow.isHorizontalWritingMode())
776 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasHorizonalWritingMode, reasons, includeReasons);
777 if (flow.hasOutline())
778 SET_REASON_AND_RETURN_IF_NEEDED(ContentHasOutline, reasons, includeReasons);
779 if (flow.isRubyText() || flow.isRubyBase())
780 SET_REASON_AND_RETURN_IF_NEEDED(ContentIsRuby, reasons, includeReasons);
781 if (!flow.style().hangingPunctuation().isEmpty())
782 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasHangingPunctuation, reasons, includeReasons);
783
784 // Printing does pagination without a flow thread.
785 if (flow.document().paginated())
786 SET_REASON_AND_RETURN_IF_NEEDED(FlowIsPaginated, reasons, includeReasons);
787 if (flow.document().highlightRegisterIfExists() && !flow.document().highlightRegisterIfExists()->map().isEmpty())
788 SET_REASON_AND_RETURN_IF_NEEDED(FlowIncludesHighlights, reasons, includeReasons);
789#if ENABLE(APP_HIGHLIGHTS)
790 if (flow.document().appHighlightRegisterIfExists() && !flow.document().appHighlightRegisterIfExists()->map().isEmpty())
791 SET_REASON_AND_RETURN_IF_NEEDED(FlowIncludesHighlights, reasons, includeReasons);
792#endif
793 if (flow.firstLineBlock())
794 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasPseudoFirstLine, reasons, includeReasons);
795 if (flow.isAnonymousBlock() && flow.parent()->style().textOverflow() == TextOverflow::Ellipsis)
796 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasTextOverflow, reasons, includeReasons);
797 if (flow.parent()->isDeprecatedFlexibleBox())
798 SET_REASON_AND_RETURN_IF_NEEDED(FlowIsDepricatedFlexBox, reasons, includeReasons);
799 // FIXME: Placeholders do something strange.
800 if (is<RenderTextControl>(*flow.parent()) && downcast<RenderTextControl>(*flow.parent()).textFormControlElement().placeholderElement())
801 SET_REASON_AND_RETURN_IF_NEEDED(FlowParentIsPlaceholderElement, reasons, includeReasons);
802 // FIXME: Implementation of wrap=hard looks into lineboxes.
803 if (flow.parent()->isTextArea() && flow.parent()->element()->hasAttributeWithoutSynchronization(HTMLNames::wrapAttr))
804 SET_REASON_AND_RETURN_IF_NEEDED(FlowParentIsTextAreaWithWrapping, reasons, includeReasons);
805 // This currently covers <blockflow>#text</blockflow>, <blockflow>#text<br></blockflow> and mutiple (sibling) RenderText cases.
806 // The <blockflow><inline>#text</inline></blockflow> case is also popular and should be relatively easy to cover.
807 for (auto walker = InlineWalker(const_cast<RenderBlockFlow&>(flow)); !walker.atEnd(); walker.advance()) {
808 if (auto childReasons = canUseForChild(flow, *walker.current(), includeReasons))
809 ADD_REASONS_AND_RETURN_IF_NEEDED(childReasons, reasons, includeReasons);
810 }
811 auto styleReasons = canUseForStyle(flow.style(), includeReasons);
812 if (styleReasons)
813 ADD_REASONS_AND_RETURN_IF_NEEDED(styleReasons, reasons, includeReasons);
814 // We can't use the code path if any lines would need to be shifted below floats. This is because we don't keep per-line y coordinates.
815 if (flow.containsFloats()) {
816 for (auto& floatingObject : *flow.floatingObjectSet()) {
817 ASSERT(floatingObject);
818 // if a float has a shape, we cannot tell if content will need to be shifted until after we lay it out,
819 // since the amount of space is not uniform for the height of the float.
820 if (floatingObject->renderer().shapeOutsideInfo())
821 SET_REASON_AND_RETURN_IF_NEEDED(FlowHasUnsupportedFloat, reasons, includeReasons);
822 }
823 }
824 auto fontAndTextReasons = canUseForFontAndText(flow, includeReasons);
825 if (fontAndTextReasons)
826 ADD_REASONS_AND_RETURN_IF_NEEDED(fontAndTextReasons, reasons, includeReasons);
827 return reasons;
828}

830bool canUseForLineLayout(const RenderBlockFlow& flow)
831{
832 return canUseForLineLayoutWithReason(flow, IncludeReasons::First).isEmpty();
833}

835bool canUseForLineLayoutAfterStyleChange(const RenderBlockFlow& blockContainer, StyleDifference diff)
836{
837 switch (diff) {
838 case StyleDifference::Equal:
839 case StyleDifference::RecompositeLayer:
840 return true;
841 case StyleDifference::Repaint:
842 case StyleDifference::RepaintIfTextOrBorderOrOutline:
843 case StyleDifference::RepaintLayer:
844 // FIXME: We could do a more focused style check by matching RendererStyle::changeRequiresRepaint&co.
845 return canUseForStyle(blockContainer.style(), IncludeReasons::First).isEmpty();
846 case StyleDifference::LayoutPositionedMovementOnly:
847 return true;
848 case StyleDifference::SimplifiedLayout:
849 case StyleDifference::SimplifiedLayoutAndPositionedMovement:
850 return canUseForStyle(blockContainer.style(), IncludeReasons::First).isEmpty();
851 case StyleDifference::Layout:
852 case StyleDifference::NewStyle:
853 return canUseForLineLayout(blockContainer);
854 }
855 ASSERT_NOT_REACHED();
856 return canUseForLineLayout(blockContainer);
857}
