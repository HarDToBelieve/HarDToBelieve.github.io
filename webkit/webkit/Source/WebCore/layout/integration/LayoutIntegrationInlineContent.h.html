47class LineLayout;
48
49struct InlineContent : public RefCounted<InlineContent> {
50 static Ref<InlineContent> create(const LineLayout& lineLayout) { return adoptRef(*new InlineContent(lineLayout)); }
51 ~InlineContent();
52
53 using Runs = Vector<Run>;
54 using Lines = Vector<Line>;
55
56 Runs runs;
57 Lines lines;
58 Vector<NonRootInlineBox> nonRootInlineBoxes;
59
60 float clearGapAfterLastLine { 0 };
61
62 const Line& lineForRun(const Run& run) const { return lines[run.lineIndex()]; }
63 WTF::IteratorRange<const Run*> runsForRect(const LayoutRect&) const;
64 void shrinkToFit();
65
66 const LineLayout& lineLayout() const;
67 const RenderObject& rendererForLayoutBox(const Layout::Box&) const;
68 const RenderBlockFlow& containingBlock() const;
69
70private:
71 InlineContent(const LineLayout&);
72
73 WeakPtr<const LineLayout> m_lineLayout;
74};
75
76inline void InlineContent::shrinkToFit()
77{
78 runs.shrinkToFit();
79 lines.shrinkToFit();
80}
