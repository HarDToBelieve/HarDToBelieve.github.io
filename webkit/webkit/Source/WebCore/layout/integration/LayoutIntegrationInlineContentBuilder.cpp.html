60inline static float lineOverflowWidth(const RenderBlockFlow& flow, InlineLayoutUnit lineBoxLogicalWidth, InlineLayoutUnit lineContentLogicalWidth)
61{
62 // FIXME: It's the copy of the lets-adjust-overflow-for-the-caret behavior from LegacyLineLayout::addOverflowFromInlineChildren.
63 auto endPadding = flow.hasOverflowClip() ? flow.paddingEnd() : 0_lu;
64 if (!endPadding)
65 endPadding = flow.endPaddingWidthForCaret();
66 if (flow.hasOverflowClip() && !endPadding && flow.element() && flow.element()->isRootEditableElement())
67 endPadding = 1;
68 lineContentLogicalWidth += endPadding;
69 return std::max(lineBoxLogicalWidth, lineContentLogicalWidth);
70}

73class Iterator {
74public:
75 Iterator() = default;
76 Iterator(const Layout::InlineLineRuns* runList, size_t currentRunIndex);
77
78 void increment();
79 unsigned offset() const { return m_offset; }
80 UCharDirection direction() const;
81
82 bool operator==(const Iterator& other) const { return offset() == other.offset(); }
83 bool operator!=(const Iterator& other) const { return offset() != other.offset(); };
84 bool atEnd() const { return !m_runList || m_runIndex == m_runList->size(); };
85
86private:
87 const Layout::LineRun& currentRun() const { return m_runList->at(m_runIndex); }
88
89 const Layout::InlineLineRuns* m_runList { nullptr };
90 size_t m_offset { 0 };
91 size_t m_runIndex { 0 };
92 size_t m_runOffset { 0 };
93};
94
95Iterator::Iterator(const Layout::InlineLineRuns* runList, size_t runIndex)
96 : m_runList(runList)
97 , m_runIndex(runIndex)
98{
99}
100
101UCharDirection Iterator::direction() const
102{
103 ASSERT(m_runList);
104 ASSERT(!atEnd());
105 auto& textContent = currentRun().text();
106 if (!textContent)
107 return U_OTHER_NEUTRAL;
108 return u_charDirection(textContent->content()[textContent->start() + m_runOffset]);
109}
110
111void Iterator::increment()
112{
113 ASSERT(m_runList);
114 ASSERT(!atEnd());
115 ++m_offset;
116 auto& currentRun = this->currentRun();
117 if (auto& textContent = currentRun.text()) {
118 if (++m_runOffset < textContent->length())
119 return;
120 }
121 ++m_runIndex;
122 m_runOffset = 0;
123}

145BidiRun::BidiRun(unsigned start, unsigned end, BidiContext* context, UCharDirection direction)
146 : m_start(start)
147 , m_end(end)
148 , m_level(context->level())
149{
150 ASSERT(context);
151 if (direction == U_OTHER_NEUTRAL)
152 direction = context->dir();
153 if (m_level % 2)
154 m_level = (direction == U_LEFT_TO_RIGHT || direction == U_ARABIC_NUMBER || direction == U_EUROPEAN_NUMBER) ? m_level + 1 : m_level;
155 else
156 m_level = (direction == U_RIGHT_TO_LEFT) ? m_level + 1 : (direction == U_ARABIC_NUMBER || direction == U_EUROPEAN_NUMBER) ? m_level + 2 : m_level;
157}

167void InlineContentBuilder::build(const Layout::InlineFormattingContext& inlineFormattingContext, InlineContent& inlineContent) const
168{
169 auto& inlineFormattingState = inlineFormattingContext.formattingState();
170 auto lineLevelVisualAdjustmentsForRuns = computeLineLevelVisualAdjustmentsForRuns(inlineFormattingState);
171 createDisplayLineRuns(inlineFormattingState, inlineContent, lineLevelVisualAdjustmentsForRuns);
172 createDisplayNonRootInlineBoxes(inlineFormattingContext, inlineContent);
173 createDisplayLines(inlineFormattingState, inlineContent, lineLevelVisualAdjustmentsForRuns);
174}

176InlineContentBuilder::LineLevelVisualAdjustmentsForRunsList InlineContentBuilder::computeLineLevelVisualAdjustmentsForRuns(const Layout::InlineFormattingState& inlineFormattingState) const
177{
178 auto& lines = inlineFormattingState.lines();
179 auto& rootStyle = m_layoutState.root().style();
180 auto shouldCheckHorizontalOverflowForContentReplacement = rootStyle.overflowX() == Overflow::Hidden && rootStyle.textOverflow() != TextOverflow::Clip;
181
182 LineLevelVisualAdjustmentsForRunsList lineLevelVisualAdjustmentsForRuns(lines.size());
183 for (size_t lineIndex = 0; lineIndex < lines.size(); ++lineIndex) {
184 auto lineNeedsLegacyIntegralVerticalPosition = [&] {
185 // Legacy inline tree integral rounds the vertical position for certain content (see LegacyInlineFlowBox::placeBoxesInBlockDirection and ::addToLine).
186 auto& rootInlineBox = inlineFormattingState.lineBoxes()[lineIndex].rootInlineBox();
187 auto& nonRootInlineLevelBoxList = inlineFormattingState.lineBoxes()[lineIndex].nonRootInlineLevelBoxes();
188 if (nonRootInlineLevelBoxList.isEmpty()) {
189 // This is text content only with root inline box.
190 return true;
191 }
192 for (auto& inlineLevelBox : nonRootInlineLevelBoxList) {
193 // See shouldClearDescendantsHaveSameLineHeightAndBaseline in LegacyInlineFlowBox::addToLine.
194 auto contentPreventsIntegralSnapping = inlineLevelBox.isAtomicInlineLevelBox() || (inlineLevelBox.isLineBreakBox() && !m_layoutState.inStandardsMode());
195 if (contentPreventsIntegralSnapping)
196 return false;
197
198 auto& rootInlineBoxStyle = rootInlineBox.style();
199 auto& inlineLevelBoxStyle = inlineLevelBox.style();
200 auto stylePreventsIntegralSnapping = rootInlineBoxStyle.lineHeight() != inlineLevelBoxStyle.lineHeight() || inlineLevelBoxStyle.verticalAlign() != VerticalAlign::Baseline;
201 if (stylePreventsIntegralSnapping)
202 return false;
203
204 auto& rootInlineBoxFontMetrics = rootInlineBoxStyle.fontCascade().fontMetrics();
205 auto& inlineLevelBoxFontMetrics = inlineLevelBoxStyle.fontCascade().fontMetrics();
206 auto fontPreventsIntegralSnapping = !rootInlineBoxFontMetrics.hasIdenticalAscentDescentAndLineGap(inlineLevelBoxFontMetrics);
207 if (fontPreventsIntegralSnapping)
208 return false;
209 }
210 return true;
211 };
212 lineLevelVisualAdjustmentsForRuns[lineIndex].needsIntegralPosition = lineNeedsLegacyIntegralVerticalPosition();
213 if (shouldCheckHorizontalOverflowForContentReplacement) {
214 auto& line = lines[lineIndex];
215 auto lineBoxLogicalWidth = line.lineBoxLogicalRect().width();
216 auto overflowWidth = lineOverflowWidth(m_blockFlow, lineBoxLogicalWidth, line.contentLogicalWidth());
217 lineLevelVisualAdjustmentsForRuns[lineIndex].needsTrailingContentReplacement = overflowWidth > lineBoxLogicalWidth;
218 }
219 }
220 return lineLevelVisualAdjustmentsForRuns;
221}

223void InlineContentBuilder::createDisplayLineRuns(const Layout::InlineFormattingState& inlineFormattingState, InlineContent& inlineContent, const LineLevelVisualAdjustmentsForRunsList& lineLevelVisualAdjustmentsForRuns) const
224{
225 auto& runList = inlineFormattingState.lineRuns();
226 if (runList.isEmpty())
227 return;
228 auto& lines = inlineFormattingState.lines();
229
230#if PROCESS_BIDI_CONTENT
231 BidiResolver<Iterator, BidiRun> bidiResolver;
232 // FIXME: Add support for override.
233 bidiResolver.setStatus(BidiStatus(m_layoutState.root().style().direction(), false));
234 // FIXME: Grab the nested isolates from the previous line.
235 bidiResolver.setPosition(Iterator(&runList, 0), 0);
236 bidiResolver.createBidiRunsForLine(Iterator(&runList, runList.size()));
237#endif
238
239 Vector<bool> hasAdjustedTrailingLineList(lines.size(), false);
240
241 auto createDisplayBoxRun = [&](auto& lineRun) {
242 auto& layoutBox = lineRun.layoutBox();
243 auto lineIndex = lineRun.lineIndex();
244 auto& lineBoxLogicalRect = lines[lineIndex].lineBoxLogicalRect();
245 // Inline boxes are relative to the line box while final runs need to be relative to the parent box
246 // FIXME: Shouldn't we just leave them be relative to the line box?
247 auto runRect = FloatRect { lineRun.logicalRect() };
248 auto& geometry = m_layoutState.geometryForBox(layoutBox);
249 runRect.moveBy({ lineBoxLogicalRect.left(), lineBoxLogicalRect.top() });
250 runRect.setSize({ geometry.borderBoxWidth(), geometry.borderBoxHeight() });
251 if (lineLevelVisualAdjustmentsForRuns[lineIndex].needsIntegralPosition)
252 runRect.setY(roundToInt(runRect.y()));
253 // FIXME: Add support for non-text ink overflow.
254 // FIXME: Add support for cases when the run is after ellipsis.
255 inlineContent.runs.append({ lineIndex, layoutBox, runRect, runRect, { }, { } });
256 };
257
258 auto createDisplayTextRunForRange = [&](auto& lineRun, auto startOffset, auto endOffset) {
259 RELEASE_ASSERT(startOffset < endOffset);
260 auto& layoutBox = lineRun.layoutBox();
261 auto lineIndex = lineRun.lineIndex();
262 auto& lineBoxLogicalRect = lines[lineIndex].lineBoxLogicalRect();
263 auto runRect = FloatRect { lineRun.logicalRect() };
264 runRect.moveBy({ lineBoxLogicalRect.left(), lineBoxLogicalRect.top() });
265 if (lineLevelVisualAdjustmentsForRuns[lineIndex].needsIntegralPosition)
266 runRect.setY(roundToInt(runRect.y()));
267
268 auto& style = layoutBox.style();
269 auto text = lineRun.text();
270 auto adjustedContentToRender = [&] {
271 auto originalContent = text->content().substring(text->start(), text->length());
272 if (text->needsHyphen())
273 return makeString(originalContent, style.hyphenString());
274 if (lineLevelVisualAdjustmentsForRuns[lineIndex].needsTrailingContentReplacement) {
275 // Currently it's ellipsis replacement only, but adding support for "text-overflow: string" should be relatively simple.
276 if (hasAdjustedTrailingLineList[lineIndex]) {
277 // This line already has adjusted trailing. Any runs after the ellipsis should render blank.
278 return emptyString();
279 }
280 auto runLogicalRect = lineRun.logicalRect();
281 auto ellipsisWidth = style.fontCascade().width(WebCore::TextRun { &horizontalEllipsis });
282 if (runLogicalRect.right() + ellipsisWidth > lineBoxLogicalRect.right()) {
283 // The next run with ellipsis would surely overflow. So let's just add it to this run even if
284 // it makes the run wider than it originally was.
285 hasAdjustedTrailingLineList[lineIndex] = true;
286 float resultWidth = 0;
287 auto maxWidth = lineBoxLogicalRect.width() - runLogicalRect.left();
288 return StringTruncator::rightTruncate(originalContent, maxWidth, style.fontCascade(), resultWidth, true);
289 }
290 }
291 return String();
292 };
293
294 auto computedInkOverflow = [&] (auto runRect) {
295 auto inkOverflow = runRect;
296 auto initialContaingBlockSize = m_layoutState.viewportSize();
297 auto strokeOverflow = std::ceil(style.computedStrokeWidth(ceiledIntSize(initialContaingBlockSize)));
298 inkOverflow.inflate(strokeOverflow);
299 auto letterSpacing = style.fontCascade().letterSpacing();
300 if (letterSpacing < 0) {
301 // Last letter's negative spacing shrinks logical rect. Push it to ink overflow.
302 inkOverflow.expand(-letterSpacing, { });
303 }
304 return inkOverflow;
305 };
306 RELEASE_ASSERT(startOffset >= text->start() && startOffset < text->end());
307 RELEASE_ASSERT(endOffset > text->start() && endOffset <= text->end());
308 auto textContent = Run::TextContent { startOffset, endOffset - startOffset, text->content(), adjustedContentToRender(), text->needsHyphen() };
309 auto expansion = Run::Expansion { lineRun.expansion().behavior, lineRun.expansion().horizontalExpansion };
310 auto displayRun = Run { lineIndex, layoutBox, runRect, computedInkOverflow(runRect), expansion, textContent };
311 inlineContent.runs.append(displayRun);
312 };
313
314 inlineContent.runs.reserveInitialCapacity(inlineFormattingState.lineRuns().size());
315 for (auto& lineRun : inlineFormattingState.lineRuns()) {
316 if (auto& text = lineRun.text())
317 createDisplayTextRunForRange(lineRun, text->start(), text->end());
318 else
319 createDisplayBoxRun(lineRun);
320 }
321}

323void InlineContentBuilder::createDisplayLines(const Layout::InlineFormattingState& inlineFormattingState, InlineContent& inlineContent, const LineLevelVisualAdjustmentsForRunsList& lineLevelVisualAdjustmentsForRuns) const
324{
325 auto& lines = inlineFormattingState.lines();
326 auto& runs = inlineContent.runs;
327 auto& nonRootInlineBoxes = inlineContent.nonRootInlineBoxes;
328 size_t runIndex = 0;
329 size_t inlineBoxIndex = 0;
330 inlineContent.lines.reserveInitialCapacity(lines.size());
331 for (size_t lineIndex = 0; lineIndex < lines.size(); ++lineIndex) {
332 auto& line = lines[lineIndex];
333 auto& lineBoxLogicalRect = line.lineBoxLogicalRect();
334 // FIXME: This is where the logical to physical translate should happen.
335 auto scrollableOverflowRect = FloatRect { lineBoxLogicalRect.left(), lineBoxLogicalRect.top(), lineOverflowWidth(m_blockFlow, lineBoxLogicalRect.width(), line.contentLogicalWidth()), lineBoxLogicalRect.height() };
336
337 auto firstRunIndex = runIndex;
338 auto lineInkOverflowRect = scrollableOverflowRect;
339 // Collect overflow from runs.
340 for (; runIndex < runs.size() && runs[runIndex].lineIndex() == lineIndex; ++runIndex) {
341 auto& run = runs[runIndex];
342 lineInkOverflowRect.unite(run.inkOverflow());
343
344 auto& layoutBox = run.layoutBox();
345 if (!layoutBox.isReplacedBox())
346 continue;
347
348 // Similar to LegacyInlineFlowBox::addReplacedChildOverflow.
349 auto& box = downcast<RenderBox>(m_boxTree.rendererForLayoutBox(layoutBox));
350 if (!box.hasSelfPaintingLayer()) {
351 auto childInkOverflow = box.logicalVisualOverflowRectForPropagation(&box.parent()->style());
352 childInkOverflow.move(run.rect().x(), run.rect().y());
353 lineInkOverflowRect.unite(childInkOverflow);
354 }
355 auto childScrollableOverflow = box.logicalLayoutOverflowRectForPropagation(&box.parent()->style());
356 childScrollableOverflow.move(run.rect().x(), run.rect().y());
357 scrollableOverflowRect.unite(childScrollableOverflow);
358 }
359 // Collect scrollable overflow from inline boxes. All other inline level boxes (e.g atomic inline level boxes) stretch the line.
360 while (inlineBoxIndex < nonRootInlineBoxes.size() && nonRootInlineBoxes[inlineBoxIndex].lineIndex() == lineIndex) {
361 auto& inlineBox = nonRootInlineBoxes[inlineBoxIndex++];
362 if (inlineBox.canContributeToLineOverflow())
363 scrollableOverflowRect.unite(inlineBox.rect());
364 }
365
366 auto adjustedLineBoxRect = FloatRect { lineBoxLogicalRect };
367 // Final enclosing top and bottom values are in the same coordinate space as the line itself.
368 auto enclosingTopAndBottom = line.enclosingTopAndBottom() + lineBoxLogicalRect.top();
369 if (lineLevelVisualAdjustmentsForRuns[lineIndex].needsIntegralPosition) {
370 adjustedLineBoxRect.setY(roundToInt(adjustedLineBoxRect.y()));
371 enclosingTopAndBottom.top = roundToInt(enclosingTopAndBottom.top);
372 enclosingTopAndBottom.bottom = roundToInt(enclosingTopAndBottom.bottom);
373 }
374 auto runCount = runIndex - firstRunIndex;
375 inlineContent.lines.append({ firstRunIndex, runCount, adjustedLineBoxRect, enclosingTopAndBottom.top, enclosingTopAndBottom.bottom, scrollableOverflowRect, lineInkOverflowRect, line.baseline(), line.contentLogicalLeft(), line.contentLogicalWidth() });
376 }
377}

379void InlineContentBuilder::createDisplayNonRootInlineBoxes(const Layout::InlineFormattingContext& inlineFormattingContext, InlineContent& inlineContent) const
380{
381 auto& inlineFormattingState = inlineFormattingContext.formattingState();
382 auto& inlineFormattingGeometry = downcast<Layout::InlineFormattingGeometry>(inlineFormattingContext.formattingGeometry());
383 for (size_t lineIndex = 0; lineIndex < inlineFormattingState.lineBoxes().size(); ++lineIndex) {
384 auto& lineBox = inlineFormattingState.lineBoxes()[lineIndex];
385 if (!lineBox.hasInlineBox())
386 continue;
387
388 auto& lineBoxLogicalRect = lineBox.logicalRect();
389 for (auto& inlineLevelBox : lineBox.nonRootInlineLevelBoxes()) {
390 if (!inlineLevelBox.isInlineBox())
391 continue;
392 auto& layoutBox = inlineLevelBox.layoutBox();
393 auto& boxGeometry = m_layoutState.geometryForBox(layoutBox);
394 auto inlineBoxRect = lineBox.logicalBorderBoxForInlineBox(layoutBox, boxGeometry);
395 inlineBoxRect.moveBy(lineBoxLogicalRect.topLeft());
396
397 inlineContent.nonRootInlineBoxes.append({ lineIndex, layoutBox, inlineBoxRect, inlineFormattingGeometry.inlineLevelBoxAffectsLineBox(inlineLevelBox, lineBox) });
398 }
399 }
