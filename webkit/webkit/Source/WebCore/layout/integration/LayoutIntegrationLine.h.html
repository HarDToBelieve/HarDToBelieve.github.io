33namespace WebCore {
34namespace LayoutIntegration {

36class Line {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 Line(size_t firstRunIndex, size_t runCount, const FloatRect& lineBoxRect, float enclosingContentTop, float enclosingContentBottom, const FloatRect& scrollableOverflow, const FloatRect& inkOverflow, float baseline, float contentLeft, float contentWidth)
40 : m_firstRunIndex(firstRunIndex)
41 , m_runCount(runCount)
42 , m_lineBoxRect(lineBoxRect)
43 , m_enclosingContentTop(enclosingContentTop)
44 , m_enclosingContentBottom(enclosingContentBottom)
45 , m_scrollableOverflow(scrollableOverflow)
46 , m_inkOverflow(inkOverflow)
47 , m_baseline(baseline)
48 , m_contentLeft(contentLeft)
49 , m_contentWidth(contentWidth)
50 {
51 }
52
53 size_t firstRunIndex() const { return m_firstRunIndex; }
54 size_t runCount() const { return m_runCount; }
55
56 float lineBoxTop() const { return m_lineBoxRect.y(); }
57 float lineBoxBottom() const { return m_lineBoxRect.maxY(); }
58 float lineBoxLeft() const { return m_lineBoxRect.x(); }
59 float lineBoxRight() const { return m_lineBoxRect.maxX(); }
60
61 float enclosingContentTop() const { return m_enclosingContentTop; }
62 float enclosingContentBottom() const { return m_enclosingContentBottom; }
63
64 const FloatRect& scrollableOverflow() const { return m_scrollableOverflow; }
65 const FloatRect& inkOverflow() const { return m_inkOverflow; }
66
67 float baseline() const { return m_baseline; }
68
69 float contentLeft() const { return m_contentLeft; }
70 float contentWidth() const { return m_contentWidth; }
71
72private:
73 size_t m_firstRunIndex { 0 };
74 size_t m_runCount { 0 };
75 // This is line box geometry (see https://www.w3.org/TR/css-inline-3/#line-box).
76 FloatRect m_lineBoxRect;
77 // Enclosing top and bottom includes all inline level boxes (border box) vertically.
78 // While the line box usually enclose them as well, its vertical geometry is based on
79 // the layout bounds of the inline level boxes which may be different when line-height is present.
80 float m_enclosingContentTop { 0 };
81 float m_enclosingContentBottom { 0 };
82 FloatRect m_scrollableOverflow;
83 FloatRect m_inkOverflow;
84 float m_baseline { 0 };
85 float m_contentLeft { 0 };
86 float m_contentWidth { 0 };
87};
88
89class NonRootInlineBox {
90public:
91 NonRootInlineBox(size_t lineIndex, const Layout::Box& layoutBox, const FloatRect& rect, bool canContributeToLineOverflow)
92 : m_lineIndex(lineIndex)
93 , m_layoutBox(makeWeakPtr(layoutBox))
94 , m_rect(rect)
95 , m_canContributeToLineOverflow(canContributeToLineOverflow)
96 {
97 }
98
99 const Layout::Box& layoutBox() const { return *m_layoutBox; }
100 const RenderStyle& style() const { return m_layoutBox->style(); }
101
102 size_t lineIndex() const { return m_lineIndex; }
103
104 FloatRect rect() const { return m_rect; }
105
106 bool canContributeToLineOverflow() const { return m_canContributeToLineOverflow; }
107
108private:
109 const size_t m_lineIndex;
110 WeakPtr<const Layout::Box> m_layoutBox;
111 FloatRect m_rect;
112 bool m_canContributeToLineOverflow { false };
113};
