107RunIterator LineIterator::logicalStartRunWithNode() const
108{
109 for (auto run = logicalStartRun(); run; run.traverseNextOnLineInLogicalOrder()) {
110 if (run->renderer().node())
111 return run;
112 }
113 return { };
114}

116RunIterator LineIterator::logicalEndRunWithNode() const
117{

120 return run;
121 }
122 return { };

145RunIterator LineIterator::closestRunForPoint(const IntPoint& pointInContents, bool editableOnly)
146{
147 if (atEnd())
148 return { };
149 return closestRunForLogicalLeftPosition(m_line.isHorizontal() ? pointInContents.x() : pointInContents.y(), editableOnly);
150}
151
152RunIterator LineIterator::closestRunForLogicalLeftPosition(int leftPosition, bool editableOnly)
153{
154 auto isEditable = [&](auto run)
155 {
156 return run && run->renderer().node() && run->renderer().node()->hasEditableStyle();
157 };
158
159 auto firstRun = this->firstRun();
160 auto lastRun = this->lastRun();
161
162 if (firstRun != lastRun) {
163 if (firstRun->isLineBreak())
164 firstRun = firstRun.nextOnLineIgnoringLineBreak();
165 else if (lastRun->isLineBreak())
166 lastRun = lastRun.previousOnLineIgnoringLineBreak();
167 }
168
169 if (firstRun == lastRun && (!editableOnly || isEditable(firstRun)))
170 return firstRun;
171
172 if (firstRun && leftPosition <= firstRun->logicalLeft() && !firstRun->renderer().isListMarker() && (!editableOnly || isEditable(firstRun)))
173 return firstRun;
174
175 if (lastRun && leftPosition >= lastRun->logicalRight() && !lastRun->renderer().isListMarker() && (!editableOnly || isEditable(lastRun)))
176 return lastRun;
177
178 auto closestRun = lastRun;
179 for (auto run = firstRun; run; run = run.traverseNextOnLineIgnoringLineBreak()) {
180 if (!run->renderer().isListMarker() && (!editableOnly || isEditable(run))) {
181 if (leftPosition < run->logicalRight())
182 return run;
183 closestRun = run;
184 }
185 }
186
187 return closestRun;
188}

190int PathLine::blockDirectionPointInLine() const
191{
192 return !containingBlock().style().isFlippedBlocksWritingMode() ? std::max(top(), selectionTop()) : std::min(bottom(), selectionBottom());
