37struct Strut {
38 unsigned lineBreak;
39 float offset;
40};
41
42struct PaginatedLine {
43 LayoutUnit top;
44 LayoutUnit height;
45};
46using PaginatedLines = Vector<PaginatedLine, 20>;
47
48static PaginatedLine computeLineTopAndBottomWithOverflow(const RenderBlockFlow&, InlineContent::Lines& lines, unsigned lineIndex, Vector<Strut>& struts)
49{
50 LayoutUnit offset = 0;
51 for (auto& strut : struts) {
52 if (strut.lineBreak > lineIndex)
53 break;
54 offset += strut.offset;
55 }
56 auto overflowRect = LayoutRect(lines[lineIndex].inkOverflow());
57 return { overflowRect.y() + offset, overflowRect.height() };
58}
59
60static unsigned computeLineBreakIndex(unsigned breakCandidate, unsigned lineCount, int orphansNeeded, int widowsNeeded,
61 const Vector<Strut>& struts)
62{
63 // First line does not fit the current page.
64 if (!breakCandidate)
65 return breakCandidate;
66
67 int widowsOnTheNextPage = lineCount - breakCandidate;
68 if (widowsNeeded <= widowsOnTheNextPage)
69 return breakCandidate;
70 // Only break after the first line with widows.
71 auto lineBreak = std::max<int>(lineCount - widowsNeeded, 1);
72 if (orphansNeeded > lineBreak)
73 return breakCandidate;
74 // Break on current page only.
75 if (struts.isEmpty())
76 return lineBreak;
77 ASSERT(struts.last().lineBreak + 1 < lineCount);
78 return std::max<unsigned>(struts.last().lineBreak + 1, lineBreak);
79}

81static LayoutUnit computeOffsetAfterLineBreak(LayoutUnit lineBreakPosition, bool isFirstLine, bool atTheTopOfColumnOrPage, const RenderBlockFlow& flow)
82{
83 // No offset for top of the page lines unless widows pushed the line break.
84 LayoutUnit offset = isFirstLine ? flow.borderAndPaddingBefore() : 0_lu;
85 if (atTheTopOfColumnOrPage)
86 return offset;
87 return offset + flow.pageRemainingLogicalHeightForOffset(lineBreakPosition, RenderBlockFlow::ExcludePageBoundary);
88}

90static void setPageBreakForLine(unsigned lineBreakIndex, PaginatedLines& lines, RenderBlockFlow& flow, Vector<Strut>& struts,
91 bool atTheTopOfColumnOrPage, bool lineDoesNotFit)
92{
93 auto line = lines.at(lineBreakIndex);
94 auto remainingLogicalHeight = flow.pageRemainingLogicalHeightForOffset(line.top, RenderBlockFlow::ExcludePageBoundary);
95 auto& style = flow.style();
96 auto firstLineDoesNotFit = !lineBreakIndex && line.height < flow.pageLogicalHeightForOffset(line.top);
97 auto moveOrphanToNextColumn = lineDoesNotFit && !style.hasAutoOrphans() && style.orphans() > (short)lineBreakIndex;
98 if (firstLineDoesNotFit || moveOrphanToNextColumn) {
99 auto firstLine = lines.first();
100 auto firstLineUpperOverhang = std::max(LayoutUnit(-firstLine.top), 0_lu);
101 flow.setPaginationStrut(line.top + remainingLogicalHeight + firstLineUpperOverhang);
102 return;
103 }
104 if (atTheTopOfColumnOrPage)
105 flow.setPageBreak(line.top, line.height);
106 else
107 flow.setPageBreak(line.top, line.height - remainingLogicalHeight);
108 struts.append({ lineBreakIndex, computeOffsetAfterLineBreak(lines[lineBreakIndex].top, !lineBreakIndex, atTheTopOfColumnOrPage, flow) });
109}
110
111static void updateMinimumPageHeight(RenderBlockFlow& flow, const InlineContent& inlineContent, unsigned lineCount)
112{
113 auto& style = flow.style();
114 auto widows = style.hasAutoWidows() ? 1 : std::max<int>(style.widows(), 1);
115 auto orphans = style.hasAutoOrphans() ? 1 : std::max<int>(style.orphans(), 1);
116 auto minimumLineCount = std::min<unsigned>(std::max(widows, orphans), lineCount);
117 flow.updateMinimumPageHeight(0, LayoutUnit(inlineContent.lines[minimumLineCount - 1].lineBoxBottom()));
118}

120static Ref<InlineContent> makeAdjustedContent(const InlineContent& inlineContent, Vector<float> adjustments)
121{
122 auto moveVertically = [](FloatRect rect, float offset) {
123 rect.move(FloatSize(0, offset));
124 return rect;
125 };
126
127 auto adjustedLine = [&](const Line& line, float offset)
128 {
129 return Line {
130 line.firstRunIndex(),
131 line.runCount(),
132 moveVertically({ FloatPoint { line.lineBoxLeft(), line.lineBoxTop() }, FloatPoint { line.lineBoxRight(), line.lineBoxBottom() } }, offset),
133 line.enclosingContentTop() + offset,
134 line.enclosingContentBottom() + offset,
135 moveVertically(line.scrollableOverflow(), offset),
136 moveVertically(line.inkOverflow(), offset),
137 line.baseline(),
138 line.contentLeft(),
139 line.contentWidth()
140 };
141 };
142
143 auto adjustedRun = [&](const Run& run, float offset)
144 {
145 return Run {
146 run.lineIndex(),
147 run.layoutBox(),
148 moveVertically(run.rect(), offset),
149 moveVertically(run.inkOverflow(), offset),
150 run.expansion(),
151 run.textContent()
152 };
153 };
154
155 auto adjustedContent = InlineContent::create(inlineContent.lineLayout());
156
157 for (size_t lineIndex = 0; lineIndex < inlineContent.lines.size(); ++lineIndex)
158 adjustedContent->lines.append(adjustedLine(inlineContent.lines[lineIndex], adjustments[lineIndex]));
159
160 for (auto& run : inlineContent.runs)
161 adjustedContent->runs.append(adjustedRun(run, adjustments[run.lineIndex()]));
162
163 return adjustedContent;
164}

166Ref<InlineContent> adjustLinePositionsForPagination(InlineContent& inlineContent, RenderBlockFlow& flow)
167{
168 Vector<Strut> struts;
169 auto lineCount = inlineContent.lines.size();
170 updateMinimumPageHeight(flow, inlineContent, lineCount);
171 // First pass with no pagination offset?
172 if (!flow.pageLogicalHeightForOffset(0))
173 return inlineContent;
174
175 auto widows = flow.style().hasAutoWidows() ? 1 : std::max<int>(flow.style().widows(), 1);
176 auto orphans = flow.style().hasAutoOrphans() ? 1 : std::max<int>(flow.style().orphans(), 1);
177 PaginatedLines lines;
178 for (unsigned lineIndex = 0; lineIndex < lineCount; ++lineIndex) {
179 auto line = computeLineTopAndBottomWithOverflow(flow, inlineContent.lines, lineIndex, struts);
180 lines.append(line);
181 auto remainingHeight = flow.pageRemainingLogicalHeightForOffset(line.top, RenderBlockFlow::ExcludePageBoundary);
182 auto atTheTopOfColumnOrPage = flow.pageLogicalHeightForOffset(line.top) == remainingHeight;
183 auto lineDoesNotFit = line.height > remainingHeight;
184 if (lineDoesNotFit || (atTheTopOfColumnOrPage && lineIndex)) {
185 auto lineBreakIndex = computeLineBreakIndex(lineIndex, lineCount, orphans, widows, struts);
186 // Are we still at the top of the column/page?
187 atTheTopOfColumnOrPage = atTheTopOfColumnOrPage ? lineIndex == lineBreakIndex : false;
188 setPageBreakForLine(lineBreakIndex, lines, flow, struts, atTheTopOfColumnOrPage, lineDoesNotFit);
189 // Recompute line positions that we already visited but widow break pushed them to a new page.
190 for (auto i = lineBreakIndex; i < lines.size(); ++i)
191 lines.at(i) = computeLineTopAndBottomWithOverflow(flow, inlineContent.lines, i, struts);
192 }
193 }
194
195 if (struts.isEmpty())
196 return inlineContent;
197
198 auto adjustments = Vector<float>(lineCount, 0);
199 for (auto& strut : struts) {
200 for (auto lineIndex = strut.lineBreak; lineIndex < lineCount; ++lineIndex)
201 adjustments[lineIndex] += strut.offset;
202 }
203
204 return makeAdjustedContent(inlineContent, adjustments);
205}
