41struct Run {
42 WTF_MAKE_STRUCT_FAST_ALLOCATED;
43 struct TextContent {
44 WTF_MAKE_STRUCT_FAST_ALLOCATED;
45 public:
46 TextContent(size_t position, size_t length, const String&, String adjustedContentToRender, bool hasHyphen);
47
48 size_t start() const { return m_start; }
49 size_t end() const { return start() + length(); }
50 size_t length() const { return m_length; }
51 StringView originalContent() const { return StringView(m_originalContent).substring(m_start, m_length); }
52 StringView renderedContent() const { return m_adjustedContentToRender.isNull() ? originalContent() : m_adjustedContentToRender; }
53 bool hasHyphen() const { return m_hasHyphen; }
54
55 private:
56 size_t m_start { 0 };
57 size_t m_length { 0 };
58 bool m_hasHyphen { false };
59 String m_originalContent;
60 String m_adjustedContentToRender;
61 };
62
63 struct Expansion;
64 Run(size_t lineIndex, const Layout::Box&, const FloatRect&, const FloatRect& inkOverflow, Expansion, std::optional<TextContent> = std::nullopt);
65
66 const FloatRect& rect() const { return m_rect; }
67 const FloatRect& inkOverflow() const { return m_inkOverflow; }
68
69 std::optional<TextContent>& textContent() { return m_textContent; }
70 const std::optional<TextContent>& textContent() const { return m_textContent; }
71 // FIXME: This information should be preserved at Run construction time.
72 bool isLineBreak() const { return layoutBox().isLineBreakBox() || (textContent() && textContent()->originalContent() == "\n" && style().preserveNewline()); }
73
74 struct Expansion {
75 ExpansionBehavior behavior { DefaultExpansion };
76 InlineLayoutUnit horizontalExpansion { 0 };
77 };
78 Expansion expansion() const { return m_expansion; }
79
80 CachedImage* image() const { return m_cachedImage; }
81
82 bool hasUnderlyingLayout() const { return !!m_layoutBox; }
83
84 const Layout::Box& layoutBox() const { return *m_layoutBox; }
85 const RenderStyle& style() const { return m_layoutBox->style(); }
86
87 size_t lineIndex() const { return m_lineIndex; }
88
89private:
90 // FIXME: Find out the Display::Run <-> paint style setup.
91 const size_t m_lineIndex;
92 WeakPtr<const Layout::Box> m_layoutBox;
93 CachedImage* m_cachedImage { nullptr };
94 FloatRect m_rect;
95 FloatRect m_inkOverflow;
96 Expansion m_expansion;
97 std::optional<TextContent> m_textContent;
98};
99
100inline Run::Run(size_t lineIndex, const Layout::Box& layoutBox, const FloatRect& rect, const FloatRect& inkOverflow, Expansion expansion, std::optional<TextContent> textContent)
101 : m_lineIndex(lineIndex)
102 , m_layoutBox(makeWeakPtr(layoutBox))
103 , m_rect(rect)
104 , m_inkOverflow(inkOverflow)
105 , m_expansion(expansion)
106 , m_textContent(textContent)
107{
108}

110inline Run::TextContent::TextContent(size_t start, size_t length, const String& originalContent, String adjustedContentToRender, bool hasHyphen)
111 : m_start(start)
112 , m_length(length)
113 , m_hasHyphen(hasHyphen)
114 , m_originalContent(originalContent)
115 , m_adjustedContentToRender(adjustedContentToRender)
116{
117}

119}
