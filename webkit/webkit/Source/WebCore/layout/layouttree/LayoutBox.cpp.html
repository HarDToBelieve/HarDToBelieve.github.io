65bool Box::establishesFormattingContext() const
66{
67 // We need the final tree structure to tell whether a box establishes a certain formatting context.
68 ASSERT(!Phase::isInTreeBuilding());
69 return establishesBlockFormattingContext()
70 || establishesInlineFormattingContext()
71 || establishesTableFormattingContext()
72 || establishesFlexFormattingContext()
73 || establishesIndependentFormattingContext();
74}
75
76bool Box::establishesBlockFormattingContext() const
77{
78 // ICB always creates a new (inital) block formatting context.
79 if (is<InitialContainingBlock>(*this))
80 return true;
81
82 if (isTableWrapperBox())
83 return true;
84
85 // A block box that establishes an independent formatting context establishes a new block formatting context for its contents.
86 if (isBlockBox() && establishesIndependentFormattingContext())
87 return true;
88
89 // 9.4.1 Block formatting contexts
90 // Floats, absolutely positioned elements, block containers (such as inline-blocks, table-cells, and table-captions)
91 // that are not block boxes, and block boxes with 'overflow' other than 'visible' (except when that value has been propagated to the viewport)
92 // establish new block formatting contexts for their contents.
93 if (isFloatingPositioned()) {
94 // Not all floating or out-of-positioned block level boxes establish BFC.
95 // See [9.7 Relationships between 'display', 'position', and 'float'] for details.
96 return style().display() == DisplayType::Block;
97 }
98
99 if (isBlockContainer() && !isBlockBox())
100 return true;
101
102 if (isBlockBox() && !isOverflowVisible())
103 return true;
104
105 return false;
106}

108bool Box::establishesInlineFormattingContext() const
109{
110 // 9.4.2 Inline formatting contexts
111 // An inline formatting context is established by a block container box that contains no block-level boxes.
112 if (!isBlockContainer())
113 return false;
114
115 if (!isContainerBox())
116 return false;
117
118 // FIXME ???
119 if (!downcast<ContainerBox>(*this).firstInFlowChild())
120 return false;
121
122 // It's enough to check the first in-flow child since we can't have both block and inline level sibling boxes.
123 return downcast<ContainerBox>(*this).firstInFlowChild()->isInlineLevelBox();
124}

196const ContainerBox& Box::containingBlock() const
197{
198 // Finding the containing block by traversing the tree during tree construction could provide incorrect result.
199 ASSERT(!Phase::isInTreeBuilding());
200 // If we ever end up here with the ICB, we must be doing something not-so-great.
201 RELEASE_ASSERT(!is<InitialContainingBlock>(*this));
202 // The containing block in which the root element lives is a rectangle called the initial containing block.
203 // For other elements, if the element's position is 'relative' or 'static', the containing block is formed by the
204 // content edge of the nearest block container ancestor box or which establishes a formatting context.
205 // If the element has 'position: fixed', the containing block is established by the viewport
206 // If the element has 'position: absolute', the containing block is established by the nearest ancestor with a
207 // 'position' of 'absolute', 'relative' or 'fixed'.
208 if (!isPositioned() || isInFlowPositioned()) {
209 auto* ancestor = &parent();
210 for (; !is<InitialContainingBlock>(*ancestor); ancestor = &ancestor->parent()) {
211 if (ancestor->isContainingBlockForInFlow())
212 return *ancestor;
213 }
214 return *ancestor;
215 }
216
217 if (isFixedPositioned()) {
218 auto* ancestor = &parent();
219 for (; !is<InitialContainingBlock>(*ancestor); ancestor = &ancestor->parent()) {
220 if (ancestor->isContainingBlockForFixedPosition())
221 return *ancestor;
222 }
223 return *ancestor;
224 }
225
226 if (isOutOfFlowPositioned()) {
227 auto* ancestor = &parent();
228 for (; !is<InitialContainingBlock>(*ancestor); ancestor = &ancestor->parent()) {
229 if (ancestor->isContainingBlockForOutOfFlowPosition())
230 return *ancestor;
231 }
232 return *ancestor;
233 }
234
235 ASSERT_NOT_REACHED();
236 return initialContainingBlock();
237}

239const ContainerBox& Box::formattingContextRoot() const
240{
241 // Finding the context root by traversing the tree during tree construction could provide incorrect result.
242 ASSERT(!Phase::isInTreeBuilding());
243 // We should never need to ask this question on the ICB.
244 ASSERT(!is<InitialContainingBlock>(*this));
245 // A box lives in the same formatting context as its containing block unless the containing block establishes a formatting context.
246 // However relatively positioned (inflow) inline container lives in the formatting context where its parent lives unless
247 // the parent establishes a formatting context.
248 //
249 // <div id=outer style="position: absolute"><div id=inner><span style="position: relative">content</span></div></div>
250 // While the relatively positioned inline container (span) is placed relative to its containing block "outer", it lives in the inline
251 // formatting context established by "inner".
252 auto& ancestor = isInlineLevelBox() && isInFlowPositioned() ? parent() : containingBlock();
253 if (ancestor.establishesFormattingContext())
254 return ancestor;
255 return ancestor.formattingContextRoot();
256}

258const InitialContainingBlock& Box::initialContainingBlock() const
259{
260 if (is<InitialContainingBlock>(*this))
261 return downcast<InitialContainingBlock>(*this);
262
263 auto* ancestor = &parent();
264 for (; !is<InitialContainingBlock>(*ancestor); ancestor = &ancestor->parent()) { }
265 return downcast<InitialContainingBlock>(*ancestor);
266}

268bool Box::isInFormattingContextOf(const ContainerBox& formattingContextRoot) const
269{
270 ASSERT(formattingContextRoot.establishesFormattingContext());
271 ASSERT(!is<InitialContainingBlock>(*this));
272 auto* ancestor = &containingBlock();
273 while (ancestor) {
274 if (ancestor == &formattingContextRoot)
275 return true;
276 if (is<InitialContainingBlock>(*ancestor))
277 return false;
278 ancestor = &ancestor->containingBlock();
279 }
280 ASSERT_NOT_REACHED();
281 return false;
282}

295{
296 // Block level elements generate block level boxes.
297 auto display = m_style.display();

302 || display == DisplayType::Grid
303 || display == DisplayType::FlowRoot;

312bool Box::isInlineLevelBox() const
313{
314 // Inline level elements generate inline level boxes.
315 auto display = m_style.display();
316 return display == DisplayType::Inline
317 || display == DisplayType::InlineBox
318 || display == DisplayType::InlineFlex
319 || display == DisplayType::InlineGrid
320 || isInlineBlockBox()
321 || isInlineTableBox();

324bool Box::isInlineBox() const
325{
326 // An inline box is one that is both inline-level and whose contents participate in its containing inline formatting context.
327 // A non-replaced element with a 'display' value of 'inline' generates an inline box.
328 return m_style.display() == DisplayType::Inline && !isReplacedBox();
329}

331bool Box::isAtomicInlineLevelBox() const
332{
333 // Inline-level boxes that are not inline boxes (such as replaced inline-level elements, inline-block elements, and inline-table elements)
334 // are called atomic inline-level boxes because they participate in their inline formatting context as a single opaque box.
335 return isInlineLevelBox() && !isInlineBox();
336}

355bool Box::isLayoutContainmentBox() const
356{
357 auto supportsLayoutContainment = [&] {
358 // If the element does not generate a principal box (as is the case with display values of contents or none),
359 // or its principal box is an internal table box other than table-cell, or an internal ruby box, or a non-atomic inline-level box,
360 // layout containment has no effect.
361 if (isInternalTableBox())
362 return isTableCell();
363 if (isInternalRubyBox())
364 return false;
365 if (isInlineLevelBox())
366 return isAtomicInlineLevelBox();
367 return true;
368 };
369 return m_style.contain().contains(Containment::Layout) && supportsLayoutContainment();
370}
371
372bool Box::isSizeContainmentBox() const
373{
374 auto supportsSizeContainment = [&] {
375 // If the element does not generate a principal box (as is the case with display: contents or display: none),
376 // or its inner display type is table, or its principal box is an internal table box, or an internal ruby box,
377 // or a non-atomic inline-level box, size containment has no effect.
378 if (isInternalTableBox() || isTableBox())
379 return false;
380 if (isInternalRubyBox())
381 return false;
382 if (isInlineLevelBox())
383 return isAtomicInlineLevelBox();
384 return true;
385 };
386 return m_style.contain().contains(Containment::Size) && supportsSizeContainment();
387}

429{

439bool Box::isOverflowVisible() const
440{
441 auto isOverflowVisible = m_style.overflowX() == Overflow::Visible || m_style.overflowY() == Overflow::Visible;
442 // UAs must apply the 'overflow' property set on the root element to the viewport. When the root element is an HTML "HTML" element
443 // or an XHTML "html" element, and that element has an HTML "BODY" element or an XHTML "body" element as a child,
444 // user agents must instead apply the 'overflow' property from the first such child element to the viewport,
445 // if the value on the root element is 'visible'. The 'visible' value when used for the viewport must be interpreted as 'auto'.
446 // The element from which the value is propagated must have a used value for 'overflow' of 'visible'.
447 if (isBodyBox()) {
448 auto& documentBox = containingBlock();
449 if (!documentBox.isDocumentBox())
450 return isOverflowVisible;
451 if (!documentBox.isOverflowVisible())
452 return isOverflowVisible;
453 return true;
454 }
455 if (is<InitialContainingBlock>(*this)) {
456 auto* documentBox = downcast<ContainerBox>(*this).firstChild();
457 if (!documentBox || !documentBox->isDocumentBox() || !is<ContainerBox>(documentBox))
458 return isOverflowVisible;
459 auto* bodyBox = downcast<ContainerBox>(documentBox)->firstChild();
460 if (!bodyBox || !bodyBox->isBodyBox())
461 return isOverflowVisible;
462 auto& bodyBoxStyle = bodyBox->style();
463 return bodyBoxStyle.overflowX() == Overflow::Visible || bodyBoxStyle.overflowY() == Overflow::Visible;
464 }
465 return isOverflowVisible;
466}

468bool Box::isPaddingApplicable() const
469{
470 if (isAnonymous())
471 return false;
472
473 if (isTableBox() && style().borderCollapse() == BorderCollapse::Collapse) {
474 // When the table collapses its borders with inner table elements, there's no room for padding.
475 return false;
476 }
477
478 // 8.4 Padding properties:
479 // Applies to: all elements except table-row-group, table-header-group, table-footer-group, table-row, table-column-group and table-column
480 return !isTableHeader()
481 && !isTableBody()
482 && !isTableFooter()
483 && !isTableRow()
484 && !isTableColumnGroup()
485 && !isTableColumn();
486}
