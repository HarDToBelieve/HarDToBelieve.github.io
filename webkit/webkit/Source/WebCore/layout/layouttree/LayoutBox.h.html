45class Box : public CanMakeWeakPtr<Box> {
46 WTF_MAKE_ISO_ALLOCATED(Box);
47public:
48 enum class ElementType {
49 Document,
50 Body,
51 TableWrapperBox, // The table generates a principal block container box called the table wrapper box that contains the table box and any caption boxes.
52 TableBox, // The table box is a block-level box that contains the table's internal table boxes.
53 Image,
54 IFrame,
55 GenericElement
56 };
57
58 struct ElementAttributes {
59 ElementType elementType;
60 };
61
62 enum BaseTypeFlag : uint8_t {
63 BoxFlag = 1 << 0,
64 LegacyInlineTextBoxFlag = 1 << 1,
65 LineBreakBoxFlag = 1 << 2,
66 ReplacedBoxFlag = 1 << 3,
67 InitialContainingBlockFlag = 1 << 4,
68 ContainerBoxFlag = 1 << 5
69 };
70
71 virtual ~Box();
72
73 bool establishesFormattingContext() const;
74 bool establishesBlockFormattingContext() const;
75 bool establishesInlineFormattingContext() const;
76 bool establishesTableFormattingContext() const;
77 bool establishesFlexFormattingContext() const;
78 bool establishesIndependentFormattingContext() const;
79
80 bool isInFlow() const { return !isFloatingOrOutOfFlowPositioned(); }
81 bool isPositioned() const { return isInFlowPositioned() || isOutOfFlowPositioned(); }
82 bool isInFlowPositioned() const { return isRelativelyPositioned() || isStickyPositioned(); }
83 bool isOutOfFlowPositioned() const { return isAbsolutelyPositioned(); }
84 bool isRelativelyPositioned() const;
85 bool isStickyPositioned() const;
86 bool isAbsolutelyPositioned() const;
87 bool isFixedPositioned() const;
88 bool isFloatingPositioned() const;
89 bool isLeftFloatingPositioned() const;
90 bool isRightFloatingPositioned() const;
91 bool hasFloatClear() const;
92 bool isFloatAvoider() const;
93
94 bool isFloatingOrOutOfFlowPositioned() const { return isFloatingPositioned() || isOutOfFlowPositioned(); }
95
96 const ContainerBox& containingBlock() const;
97 const ContainerBox& formattingContextRoot() const;
98 const InitialContainingBlock& initialContainingBlock() const;
99
100 bool isContainingBlockForInFlow() const;
101 bool isContainingBlockForFixedPosition() const;
102 bool isContainingBlockForOutOfFlowPosition() const;
103
104 bool isInFormattingContextOf(const ContainerBox&) const;
105
106 bool isAnonymous() const { return m_isAnonymous; }
107
108 // Block level elements generate block level boxes.
109 bool isBlockLevelBox() const;
110 // A block-level box that is also a block container.
111 bool isBlockBox() const;
112 // A block-level box is also a block container box unless it is a table box or the principal box of a replaced element.
113 bool isBlockContainer() const;
114 bool isInlineLevelBox() const;
115 bool isInlineBox() const;
116 bool isAtomicInlineLevelBox() const;
117 bool isInlineBlockBox() const;
118 bool isInlineTableBox() const;
119 bool isInitialContainingBlock() const { return baseTypeFlags().contains(InitialContainingBlockFlag); }
120 bool isLayoutContainmentBox() const;
121 bool isSizeContainmentBox() const;
122
123 bool isDocumentBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::Document; }
124 bool isBodyBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::Body; }
125 bool isTableWrapperBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::TableWrapperBox; }
126 bool isTableBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::TableBox; }
127 bool isTableCaption() const { return style().display() == DisplayType::TableCaption; }
128 bool isTableHeader() const { return style().display() == DisplayType::TableHeaderGroup; }
129 bool isTableBody() const { return style().display() == DisplayType::TableRowGroup; }
130 bool isTableFooter() const { return style().display() == DisplayType::TableFooterGroup; }
131 bool isTableRow() const { return style().display() == DisplayType::TableRow; }
132 bool isTableColumnGroup() const { return style().display() == DisplayType::TableColumnGroup; }
133 bool isTableColumn() const { return style().display() == DisplayType::TableColumn; }
134 bool isTableCell() const { return style().display() == DisplayType::TableCell; }
135 bool isInternalTableBox() const;
136 bool isFlexBox() const { return style().display() == DisplayType::Flex; }
137 bool isFlexItem() const;
138 bool isIFrame() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::IFrame; }
139 bool isImage() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::Image; }
140 bool isInternalRubyBox() const { return false; }

142 const ContainerBox& parent() const { return *m_parent; }
143 const Box* nextSibling() const { return m_nextSibling; }
144 const Box* nextInFlowSibling() const;

146 const Box* previousSibling() const { return m_previousSibling; }
147 const Box* previousInFlowSibling() const;

155 bool isInlineTextBox() const { return baseTypeFlags().contains(LegacyInlineTextBoxFlag); }
156 bool isLineBreakBox() const { return baseTypeFlags().contains(LineBreakBoxFlag); }
157 bool isReplacedBox() const { return baseTypeFlags().contains(ReplacedBoxFlag); }

159 bool isPaddingApplicable() const;
160 bool isOverflowVisible() const;

167 size_t rowSpan() const;
168
169 void setColumnSpan(size_t);
170 size_t columnSpan() const;
171
172 void setColumnWidth(LayoutUnit);
173 std::optional<LayoutUnit> columnWidth() const;
174
175 void setParent(ContainerBox& parent) { m_parent = &parent; }
176 void setNextSibling(Box& nextSibling) { m_nextSibling = &nextSibling; }
177 void setPreviousSibling(Box& previousSibling) { m_previousSibling = &previousSibling; }

179 void setIsAnonymous() { m_isAnonymous = true; }

182 BoxGeometry* cachedGeometryForLayoutState(const LayoutState&) const;
183 void setCachedGeometryForLayoutState(LayoutState&, std::unique_ptr<BoxGeometry>) const;

185protected:
186 Box(std::optional<ElementAttributes>, RenderStyle&&, OptionSet<BaseTypeFlag>);

195 std::optional<LayoutUnit> columnWidth;
196 };

214 Box* m_previousSibling { nullptr };
215 Box* m_nextSibling { nullptr };
216

221 unsigned m_baseTypeFlags : 6; // OptionSet<BaseTypeFlag>
222 bool m_hasRareData : 1;
223 bool m_isAnonymous : 1;
224};
225
226inline bool Box::isContainingBlockForInFlow() const
227{
228 return isBlockContainer() || establishesFormattingContext();
229}
230
231inline bool Box::isContainingBlockForFixedPosition() const
232{
233 return isInitialContainingBlock() || isLayoutContainmentBox() || style().hasTransform();
234}
235
236inline bool Box::isContainingBlockForOutOfFlowPosition() const
237{
238 return isInitialContainingBlock() || isPositioned() || isLayoutContainmentBox() || style().hasTransform();
239}
