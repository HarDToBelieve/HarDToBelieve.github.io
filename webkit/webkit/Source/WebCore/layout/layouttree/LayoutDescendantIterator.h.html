32namespace WebCore {
33namespace Layout {
34
35template <typename T>
36class LayoutDescendantIterator : public LayoutIterator<T> {
37public:
38 LayoutDescendantIterator(const ContainerBox& root);
39 LayoutDescendantIterator(const ContainerBox& root, const T* current);
40 LayoutDescendantIterator& operator++();
41};
42
43template <typename T>
44class LayoutDescendantIteratorAdapter {
45public:
46 LayoutDescendantIteratorAdapter(const ContainerBox& root);
47 LayoutDescendantIterator<T> begin();
48 LayoutDescendantIterator<T> end();
49 LayoutDescendantIterator<T> at(const T&);
50
51private:
52 const ContainerBox& m_root;
53};
54
55template <typename T> LayoutDescendantIteratorAdapter<T> descendantsOfType(const Box&);
56
57// LayoutDescendantIterator
58
59template <typename T>
60inline LayoutDescendantIterator<T>::LayoutDescendantIterator(const ContainerBox& root)
61 : LayoutIterator<T>(&root)
62{
63}
64
65template <typename T>
66inline LayoutDescendantIterator<T>::LayoutDescendantIterator(const ContainerBox& root, const T* current)
67 : LayoutIterator<T>(&root, current)
68{
69}
70
71template <typename T>
72inline LayoutDescendantIterator<T>& LayoutDescendantIterator<T>::operator++()
73{
74 return static_cast<LayoutDescendantIterator<T>&>(LayoutIterator<T>::traverseNext());
75}
76
77// LayoutDescendantIteratorAdapter
78
79template <typename T>
80inline LayoutDescendantIteratorAdapter<T>::LayoutDescendantIteratorAdapter(const ContainerBox& root)
81 : m_root(root)
82{
83}
84
85template <typename T>
86inline LayoutDescendantIterator<T> LayoutDescendantIteratorAdapter<T>::begin()
87{
88 return LayoutDescendantIterator<T>(m_root, Traversal::firstWithin<T>(m_root));
89}
90
91template <typename T>
92inline LayoutDescendantIterator<T> LayoutDescendantIteratorAdapter<T>::end()
93{
94 return LayoutDescendantIterator<T>(m_root);
95}
96
97template <typename T>
98inline LayoutDescendantIterator<T> LayoutDescendantIteratorAdapter<T>::at(const T& current)
99{
100 return LayoutDescendantIterator<T>(m_root, &current);
101}
102
103// Standalone functions
104
105template <typename T>
106inline LayoutDescendantIteratorAdapter<T> descendantsOfType(const ContainerBox& root)
107{
108 return LayoutDescendantIteratorAdapter<T>(root);
109}

111}
