35template <typename T>
36class LayoutIterator {
37public:
38 LayoutIterator(const ContainerBox* root);
39 LayoutIterator(const ContainerBox* root, const T* current);
40
41 const T& operator*() const;
42 const T* operator->() const;
43
44 bool operator==(const LayoutIterator& other) const;
45 bool operator!=(const LayoutIterator& other) const;
46
47 LayoutIterator& traverseNext();
48 LayoutIterator& traverseNextSibling();
49
50private:
51 const ContainerBox* m_root;
52 const T* m_current;
53};

55// Similar to WTF::is<>() but without the static_assert() making sure the check is necessary.
56template <typename T, typename U>
57inline bool isLayoutBoxOfType(const U& layoutBox) { return TypeCastTraits<const T, const U>::isOfType(layoutBox); }
58
59namespace LayoutBoxTraversal {
60
61template <typename U>
62inline const Box* firstChild(U& object)
63{
64 return object.firstChild();
65}
66
67inline const Box* firstChild(const Box& box)
68{
69 if (is<ContainerBox>(box))
70 return downcast<ContainerBox>(box).firstChild();
71 return nullptr;
72}
73
74inline const Box* nextAncestorSibling(const Box& current, const ContainerBox& stayWithin)
75{
76 for (auto* ancestor = &current.parent(); !is<InitialContainingBlock>(*ancestor); ancestor = &ancestor->parent()) {
77 if (ancestor == &stayWithin)
78 return nullptr;
79 if (auto* sibling = ancestor->nextSibling())
80 return sibling;
81 }
82 return nullptr;
83}
84
85template <typename U>
86inline const Box* next(const U& current, const ContainerBox& stayWithin)
87{
88 if (auto* child = firstChild(current))
89 return child;
90
91 if (&current == &stayWithin)
92 return nullptr;
93
94 if (auto* sibling = current.nextSibling())
95 return sibling;
96
97 return nextAncestorSibling(current, stayWithin);
98}
99
100}
101// Traversal helpers
102namespace Traversal {
103
104template <typename T, typename U>
105inline const T* firstChild(U& current)
106{
107 auto* object = LayoutBoxTraversal::firstChild(current);
108 while (object && !isLayoutBoxOfType<T>(*object))
109 object = object->nextSibling();
110 return static_cast<const T*>(object);
111}
112
113template <typename T>
114inline const T* nextSibling(const T& current)
115{
116 auto* object = current.nextSibling();
117 while (object && !isLayoutBoxOfType<T>(*object))
118 object = object->nextSibling();
119 return static_cast<const T*>(object);
120}
121
122template <typename T, typename U>
123inline const T* firstWithin(const U& stayWithin)
124{
125 auto* descendant = LayoutBoxTraversal::firstChild(stayWithin);
126 while (descendant && !isLayoutBoxOfType<T>(*descendant))
127 descendant = LayoutBoxTraversal::next(*descendant, stayWithin);
128 return static_cast<const T*>(descendant);
129}
130
131template <typename T, typename U>
132inline const T* next(const U& current, const ContainerBox& stayWithin)
133{
134 auto* descendant = LayoutBoxTraversal::next(current, stayWithin);
135 while (descendant && !isLayoutBoxOfType<T>(*descendant))
136 descendant = LayoutBoxTraversal::next(*descendant, stayWithin);
137 return static_cast<const T*>(descendant);
138}

140}

144template <typename T>
145inline LayoutIterator<T>::LayoutIterator(const ContainerBox* root)
146 : m_root(root)
147 , m_current(nullptr)
148{
149}
150
151template <typename T>
152inline LayoutIterator<T>::LayoutIterator(const ContainerBox* root, const T* current)
153 : m_root(root)
154 , m_current(current)
155{
156}
157
158template <typename T>
159inline LayoutIterator<T>& LayoutIterator<T>::traverseNextSibling()
160{
161 ASSERT(m_current);
162 m_current = Traversal::nextSibling<T>(*m_current);
163 return *this;
164}
165
166template <typename T>
167inline LayoutIterator<T>& LayoutIterator<T>::traverseNext()
168{
169 ASSERT(m_current);
170 m_current = Traversal::next<T>(*m_current, *m_root);
171 return *this;
172}
173
174template <typename T>
175inline const T& LayoutIterator<T>::operator*() const
176{
177 ASSERT(m_current);
178 return *m_current;
179}
180
181template <typename T>
182inline const T* LayoutIterator<T>::operator->() const
183{
184 ASSERT(m_current);
185 return m_current;
186}
187
188template <typename T>
189inline bool LayoutIterator<T>::operator==(const LayoutIterator& other) const
190{
191 ASSERT(m_root == other.m_root);
192 return m_current == other.m_current;
193}
194
195template <typename T>
196inline bool LayoutIterator<T>::operator!=(const LayoutIterator& other) const
197{
198 return !(*this == other);
199}
