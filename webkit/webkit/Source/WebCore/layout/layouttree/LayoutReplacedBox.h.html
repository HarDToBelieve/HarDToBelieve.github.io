37namespace WebCore {
38namespace Layout {
39
40class ReplacedBox : public Box {
41 WTF_MAKE_ISO_ALLOCATED(ReplacedBox);
42public:
43 ReplacedBox(std::optional<ElementAttributes>, RenderStyle&&);
44 virtual ~ReplacedBox() = default;
45
46 void setCachedImage(CachedImage& cachedImage) { m_cachedImage = &cachedImage; }
47 CachedImage* cachedImage() const { return m_cachedImage; }
48
49 // FIXME: Temporary until after intrinsic size change is tracked internally.
50 void setIntrinsicSize(LayoutSize size) { m_intrinsicSize = size; }
51 void setIntrinsicRatio(LayoutUnit ratio) { m_intrinsicRatio = ratio; };
52
53 bool hasIntrinsicWidth() const;
54 bool hasIntrinsicHeight() const;
55 bool hasIntrinsicRatio() const;
56 LayoutUnit intrinsicWidth() const;
57 LayoutUnit intrinsicHeight() const;
58 LayoutUnit intrinsicRatio() const;
59
60 void setBaseline(LayoutUnit baseline) { m_baseline = baseline; }
61 std::optional<LayoutUnit> baseline() const { return m_baseline; }
62
63private:
64 bool hasAspectRatio() const;
65
66 std::optional<LayoutSize> m_intrinsicSize;
67 std::optional<LayoutUnit> m_intrinsicRatio;
68 std::optional<LayoutUnit> m_baseline;
69 CachedImage* m_cachedImage { nullptr };
70};

72}
