55Vector<ContentFilter::Type>& ContentFilter::types()
56{
57 static NeverDestroyed<Vector<ContentFilter::Type>> types {
58 Vector<ContentFilter::Type>::from(
59#if HAVE(PARENTAL_CONTROLS)
60 type<ParentalControlsContentFilter>(),
61#endif
62 type<NetworkExtensionContentFilter>()
63 )
64 };
65 return types;
66}

68std::unique_ptr<ContentFilter> ContentFilter::create(ContentFilterClient& client)
69{
70 Container filters;
71 for (auto& type : types()) {
72 auto filter = type.create();
73 filters.append(WTFMove(filter));
74 }
75
76 if (filters.isEmpty())
77 return nullptr;
78
79 return makeUnique<ContentFilter>(WTFMove(filters), client);
80}

95bool ContentFilter::continueAfterWillSendRequest(ResourceRequest& request, const ResourceResponse& redirectResponse)
96{
97 Ref<ContentFilterClient> protectedClient { m_client };
98
99 LOG(ContentFiltering, "ContentFilter received request for <%s> with redirect response from <%s>.\n", request.url().string().ascii().data(), redirectResponse.url().string().ascii().data());
100#if !LOG_DISABLED
101 ResourceRequest originalRequest { request };
102#endif
103 ASSERT(m_state == State::Stopped || m_state == State::Filtering);
104 forEachContentFilterUntilBlocked([&request, &redirectResponse](PlatformContentFilter& contentFilter) {
105 contentFilter.willSendRequest(request, redirectResponse);
106 });
107 if (m_state == State::Blocked)
108 request = ResourceRequest();
109#if !LOG_DISABLED
110 if (request != originalRequest)
111 LOG(ContentFiltering, "ContentFilter changed request url to <%s>.\n", originalRequest.url().string().ascii().data());
112#endif
113 return !request.isNull();
114}

116void ContentFilter::startFilteringMainResource(CachedRawResource& resource)
117{
118 if (m_state != State::Stopped)
119 return;
120
121 LOG(ContentFiltering, "ContentFilter will start filtering main resource at <%s>.\n", resource.url().string().ascii().data());
122 m_state = State::Filtering;
123 ASSERT(!m_mainResource);
124 m_mainResource = &resource;
125}
126
127void ContentFilter::stopFilteringMainResource()
128{
129 if (m_state != State::Blocked)
130 m_state = State::Stopped;
131 m_mainResource = nullptr;
132}

134bool ContentFilter::continueAfterResponseReceived(const ResourceResponse& response)
135{
136 Ref<ContentFilterClient> protectedClient { m_client };
137
138 if (m_state == State::Filtering) {
139 LOG(ContentFiltering, "ContentFilter received response from <%s>.\n", response.url().string().ascii().data());
140 forEachContentFilterUntilBlocked([&response](PlatformContentFilter& contentFilter) {
141 contentFilter.responseReceived(response);
142 });
143 }
144
145 return m_state != State::Blocked;
146}
147
148bool ContentFilter::continueAfterDataReceived(const uint8_t* data, int length)
149{
150 Ref<ContentFilterClient> protectedClient { m_client };
151
152 if (m_state == State::Filtering) {
153 LOG(ContentFiltering, "ContentFilter received %d bytes of data from <%s>.\n", length, m_mainResource->url().string().ascii().data());
154 forEachContentFilterUntilBlocked([data, length](PlatformContentFilter& contentFilter) {
155 contentFilter.addData(data, length);
156 });
157
158 if (m_state == State::Allowed)
159 deliverResourceData(*m_mainResource);
160 return false;
161 }
162
163 return m_state != State::Blocked;
164}
165
166bool ContentFilter::continueAfterNotifyFinished(CachedResource& resource)
167{
168 ASSERT_UNUSED(resource, &resource == m_mainResource);
169 Ref<ContentFilterClient> protectedClient { m_client };
170
171 if (m_mainResource->errorOccurred())
172 return true;
173
174 if (m_state == State::Filtering) {
175 LOG(ContentFiltering, "ContentFilter will finish filtering main resource at <%s>.\n", m_mainResource->url().string().ascii().data());
176 forEachContentFilterUntilBlocked([](PlatformContentFilter& contentFilter) {
177 contentFilter.finishedAddingData();
178 });
179
180 if (m_state != State::Blocked) {
181 m_state = State::Allowed;
182 deliverResourceData(*m_mainResource);
183 }
184
185 if (m_state == State::Stopped)
186 return false;
187 }
188
189 return m_state != State::Blocked;
190}

192template <typename Function>
193inline void ContentFilter::forEachContentFilterUntilBlocked(Function&& function)
194{
195 bool allFiltersAllowedLoad { true };
196 for (auto& contentFilter : m_contentFilters) {
197 if (!contentFilter->needsMoreData()) {
198 ASSERT(!contentFilter->didBlockData());
199 continue;
200 }
201
202 function(contentFilter.get());
203
204 if (contentFilter->didBlockData()) {
205 ASSERT(!m_blockingContentFilter);
206 m_blockingContentFilter = &contentFilter;
207 didDecide(State::Blocked);
208 return;
209 } else if (contentFilter->needsMoreData())
210 allFiltersAllowedLoad = false;
211 }
212
213 if (allFiltersAllowedLoad)
214 didDecide(State::Allowed);
215}
216
217void ContentFilter::didDecide(State state)
218{
219 ASSERT(m_state != State::Allowed);
220 ASSERT(m_state != State::Blocked);
221 ASSERT(state == State::Allowed || state == State::Blocked);
222 LOG(ContentFiltering, "ContentFilter decided load should be %s for main resource at <%s>.\n", state == State::Allowed ? "allowed" : "blocked", m_mainResource ? m_mainResource->url().string().ascii().data() : "");
223 m_state = state;
224 if (m_state != State::Blocked)
225 return;
226
227 m_blockedError = m_client.contentFilterDidBlock(m_blockingContentFilter->unblockHandler(), m_blockingContentFilter->unblockRequestDeniedScript());
228 m_client.cancelMainResourceLoadForContentFilter(m_blockedError);
229}

231void ContentFilter::deliverResourceData(CachedResource& resource)
232{
233 ASSERT(m_state == State::Allowed);
234 ASSERT(resource.dataBufferingPolicy() == DataBufferingPolicy::BufferData);
235 if (auto* resourceBuffer = resource.resourceBuffer())
236 m_client.dataReceivedThroughContentFilter(resourceBuffer->data(), resourceBuffer->size());
237}

239static const URL& blockedPageURL()
240{
241 static const auto blockedPageURL = makeNeverDestroyed([] () -> URL {
242 auto webCoreBundle = CFBundleGetBundleWithIdentifier(CFSTR("com.apple.WebCore"));
243 return adoptCF(CFBundleCopyResourceURL(webCoreBundle, CFSTR("ContentFilterBlockedPage"), CFSTR("html"), nullptr)).get();
244 }());
245 return blockedPageURL;
246}

248bool ContentFilter::continueAfterSubstituteDataRequest(const DocumentLoader& activeLoader, const SubstituteData& substituteData)
249{
250 if (auto contentFilter = activeLoader.contentFilter()) {
251 if (contentFilter->m_state == State::Blocked && !contentFilter->m_isLoadingBlockedPage)
252 return contentFilter->m_blockedError.failingURL() != substituteData.failingURL();
253 }
254
255 if (activeLoader.request().url() == blockedPageURL()) {
256 ASSERT(activeLoader.substituteData().isValid());
257 return activeLoader.substituteData().failingURL() != substituteData.failingURL();
258 }
259
260 return true;
261}
262
263bool ContentFilter::willHandleProvisionalLoadFailure(const ResourceError& error) const
264{
265 if (m_state != State::Blocked)
266 return false;
267
268 if (m_blockedError.errorCode() != error.errorCode() || m_blockedError.domain() != error.domain())
269 return false;
270
271 ASSERT(m_blockedError.failingURL() == error.failingURL());
272 return true;
273}
274
275void ContentFilter::handleProvisionalLoadFailure(const ResourceError& error)
276{
277 ASSERT(willHandleProvisionalLoadFailure(error));
278
279 RefPtr<SharedBuffer> replacementData { m_blockingContentFilter->replacementData() };
280 ResourceResponse response { URL(), "text/html"_s, static_cast<long long>(replacementData->size()), "UTF-8"_s };
281 SubstituteData substituteData { WTFMove(replacementData), error.failingURL(), response, SubstituteData::SessionHistoryVisibility::Hidden };
282 SetForScope<bool> loadingBlockedPage { m_isLoadingBlockedPage, true };
283 m_client.handleProvisionalLoadFailureFromContentFilter(blockedPageURL(), substituteData);
284}
