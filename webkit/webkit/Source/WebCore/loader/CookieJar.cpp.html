44static ShouldRelaxThirdPartyCookieBlocking shouldRelaxThirdPartyCookieBlocking(const Document& document)
45{
46 if (auto* page = document.page())
47 return page->shouldRelaxThirdPartyCookieBlocking();
48 return ShouldRelaxThirdPartyCookieBlocking::No;
49}

56IncludeSecureCookies CookieJar::shouldIncludeSecureCookies(const Document& document, const URL& url)
57{
58 return (url.protocolIs("https") && !document.foundMixedContent().contains(SecurityContext::MixedContentType::Active)) ? IncludeSecureCookies::Yes : IncludeSecureCookies::No;
59}

61SameSiteInfo CookieJar::sameSiteInfo(const Document& document, IsForDOMCookieAccess isAccessForDOM)
62{
63 if (auto* loader = document.loader())
64 return SameSiteInfo::create(loader->request(), isAccessForDOM);
65 return { };
66}

75String CookieJar::cookies(Document& document, const URL& url) const
76{
77 TraceScope scope(FetchCookiesStart, FetchCookiesEnd);
78
79 auto includeSecureCookies = shouldIncludeSecureCookies(document, url);
80
81 std::optional<FrameIdentifier> frameID;
82 std::optional<PageIdentifier> pageID;
83 if (auto* frame = document.frame()) {
84 frameID = frame->loader().frameID();
85 pageID = frame->loader().pageID();
86 }
87
88 std::pair<String, bool> result;
89 if (auto* session = m_storageSessionProvider->storageSession())
90 result = session->cookiesForDOM(document.firstPartyForCookies(), sameSiteInfo(document, IsForDOMCookieAccess::Yes), url, frameID, pageID, includeSecureCookies, ShouldAskITP::Yes, shouldRelaxThirdPartyCookieBlocking(document));
91 else
92 ASSERT_NOT_REACHED();
93
94 if (result.second)
95 document.setSecureCookiesAccessed();
96
97 return result.first;
98}

100CookieRequestHeaderFieldProxy CookieJar::cookieRequestHeaderFieldProxy(const Document& document, const URL& url)
101{
102 TraceScope scope(FetchCookiesStart, FetchCookiesEnd);
103
104 std::optional<FrameIdentifier> frameID;
105 std::optional<PageIdentifier> pageID;
106 if (auto* frame = document.frame()) {
107 frameID = frame->loader().frameID();
108 pageID = frame->loader().pageID();
109 }
110
111 return { document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, shouldIncludeSecureCookies(document, url) };
112}

129bool CookieJar::cookiesEnabled(const Document&) const
130{
131 if (auto* session = m_storageSessionProvider->storageSession())
132 return session->cookieAcceptPolicy() != HTTPCookieAcceptPolicy::Never;
133
134 ASSERT_NOT_REACHED();
135 return false;
136}

138std::pair<String, SecureCookiesAccessed> CookieJar::cookieRequestHeaderFieldValue(const URL& firstParty, const SameSiteInfo& sameSiteInfo, const URL& url, std::optional<FrameIdentifier> frameID, std::optional<PageIdentifier> pageID, IncludeSecureCookies includeSecureCookies) const
139{
140 if (auto* session = m_storageSessionProvider->storageSession()) {
141 std::pair<String, bool> result = session->cookieRequestHeaderFieldValue(firstParty, sameSiteInfo, url, frameID, pageID, includeSecureCookies, ShouldAskITP::Yes, ShouldRelaxThirdPartyCookieBlocking::No);
142 return { result.first, result.second ? SecureCookiesAccessed::Yes : SecureCookiesAccessed::No };
143 }
144
145 ASSERT_NOT_REACHED();
146 return { };
147}

149String CookieJar::cookieRequestHeaderFieldValue(Document& document, const URL& url) const
150{
151 std::optional<FrameIdentifier> frameID;
152 std::optional<PageIdentifier> pageID;
153 if (auto* frame = document.frame()) {
154 frameID = frame->loader().frameID();
155 pageID = frame->loader().pageID();
156 }
157
158 auto result = cookieRequestHeaderFieldValue(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, shouldIncludeSecureCookies(document, url));
159 if (result.second == SecureCookiesAccessed::Yes)
160 document.setSecureCookiesAccessed();
161 return result.first;
162}

164bool CookieJar::getRawCookies(const Document& document, const URL& url, Vector<Cookie>& cookies) const
165{
166 std::optional<FrameIdentifier> frameID;
167 std::optional<PageIdentifier> pageID;
168 if (auto* frame = document.frame()) {
169 frameID = frame->loader().frameID();
170 pageID = frame->loader().pageID();
171 }
172
173 if (auto* session = m_storageSessionProvider->storageSession())
174 return session->getRawCookies(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, ShouldAskITP::Yes, shouldRelaxThirdPartyCookieBlocking(document), cookies);
175
176 ASSERT_NOT_REACHED();
177 return false;
178}
