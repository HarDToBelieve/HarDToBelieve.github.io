61void CrossOriginPreflightChecker::validatePreflightResponse(DocumentThreadableLoader& loader, ResourceRequest&& request, unsigned long identifier, const ResourceResponse& response)
62{
63 auto* frame = loader.document().frame();
64 if (!frame) {
65 ASSERT_NOT_REACHED();
66 return;
67 }
68
69 auto* page = loader.document().page();
70 if (!page) {
71 ASSERT_NOT_REACHED();
72 return;
73 }
74
75 auto result = WebCore::validatePreflightResponse(page->sessionID(), request, response, loader.options().storedCredentialsPolicy, loader.securityOrigin(), &CrossOriginAccessControlCheckDisabler::singleton());
76 if (!result) {
77 loader.document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, result.error());
78 loader.preflightFailure(identifier, ResourceError(errorDomainWebKitInternal, 0, request.url(), result.error(), ResourceError::Type::AccessControl));
79 return;
80 }
81
82 // FIXME: <https://webkit.org/b/164889> Web Inspector: Show Preflight Request information in inspector
83 // This is only showing success preflight requests and responses but we should show network events
84 // for preflight failures and distinguish them better from non-preflight requests.
85 NetworkLoadMetrics emptyMetrics;
86 InspectorInstrumentation::didReceiveResourceResponse(*frame, identifier, frame->loader().documentLoader(), response, nullptr);
87 InspectorInstrumentation::didFinishLoading(frame, frame->loader().documentLoader(), identifier, emptyMetrics, nullptr);
88
89 loader.preflightSuccess(WTFMove(request));
90}

92void CrossOriginPreflightChecker::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
93{
94 ASSERT_UNUSED(resource, &resource == m_resource);
95 if (m_resource->loadFailedOrCanceled()) {
96 ResourceError preflightError = m_resource->resourceError();
97 // If the preflight was cancelled by underlying code, it probably means the request was blocked due to some access control policy.
98 // FIXME:: According fetch, we should just pass the error to the layer above. But this may impact some clients like XHR or EventSource.
99 if (preflightError.isNull() || preflightError.isCancellation() || preflightError.isGeneral())
100 preflightError.setType(ResourceError::Type::AccessControl);
101
102 if (!preflightError.isTimeout())
103 m_loader.document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, "CORS-preflight request was blocked"_s);
104 m_loader.preflightFailure(m_resource->identifier(), preflightError);
105 return;
106 }
107 validatePreflightResponse(m_loader, WTFMove(m_request), m_resource->identifier(), m_resource->response());
108}

110void CrossOriginPreflightChecker::redirectReceived(CachedResource& resource, ResourceRequest&&, const ResourceResponse& response, CompletionHandler<void(ResourceRequest&&)>&& completionHandler)
111{
112 ASSERT_UNUSED(resource, &resource == m_resource);

117void CrossOriginPreflightChecker::startPreflight()
118{
119 ResourceLoaderOptions options;
120 options.referrerPolicy = m_loader.options().referrerPolicy;
121 options.contentSecurityPolicyImposition = ContentSecurityPolicyImposition::SkipPolicyCheck;
122 options.serviceWorkersMode = ServiceWorkersMode::None;
123 options.initiatorContext = m_loader.options().initiatorContext;
124
125 CachedResourceRequest preflightRequest(createAccessControlPreflightRequest(m_request, m_loader.securityOrigin(), m_loader.referrer()), options);
126 preflightRequest.setInitiator(m_loader.options().initiator);
127
128 ASSERT(!m_resource);
129 m_resource = m_loader.document().cachedResourceLoader().requestRawResource(WTFMove(preflightRequest)).value_or(nullptr);
130 if (m_resource)
131 m_resource->addClient(*this);
132}

134void CrossOriginPreflightChecker::doPreflight(DocumentThreadableLoader& loader, ResourceRequest&& request)
135{
136 if (!loader.document().frame())
137 return;
138
139 ResourceRequest preflightRequest = createAccessControlPreflightRequest(request, loader.securityOrigin(), loader.referrer());
140 ResourceError error;
141 ResourceResponse response;
142 RefPtr<SharedBuffer> data;
143
144 unsigned identifier = loader.document().frame()->loader().loadResourceSynchronously(preflightRequest, ClientCredentialPolicy::CannotAskClientForCredentials, FetchOptions { }, { }, error, response, data);
145
146 if (!error.isNull()) {
147 // If the preflight was cancelled by underlying code, it probably means the request was blocked due to some access control policy.
148 // FIXME:: According fetch, we should just pass the error to the layer above. But this may impact some clients like XHR or EventSource.
149 if (error.isCancellation() || error.isGeneral())
150 error.setType(ResourceError::Type::AccessControl);
151
152 if (!error.isTimeout())
153 loader.document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, "CORS-preflight request was blocked"_s);
154
155 loader.preflightFailure(identifier, error);
156 return;
157 }
158
159 // FIXME: Ideally, we should ask platformLoadResourceSynchronously to set ResourceResponse isRedirected and use it here.
160 bool isRedirect = preflightRequest.url().strippedForUseAsReferrer() != response.url().strippedForUseAsReferrer();
161 if (isRedirect || !response.isSuccessful()) {
162 auto errorMessage = "Preflight response is not successful"_s;
163 loader.document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, errorMessage);
164
165 loader.preflightFailure(identifier, ResourceError { errorDomainWebKitInternal, 0, request.url(), errorMessage, ResourceError::Type::AccessControl });
166 return;
167 }
168
169 validatePreflightResponse(loader, WTFMove(request), identifier, response);
170}
