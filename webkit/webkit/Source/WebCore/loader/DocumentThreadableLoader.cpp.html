105bool DocumentThreadableLoader::shouldSetHTTPHeadersToKeep() const
106{
107 if (m_options.mode == FetchOptions::Mode::Cors && shouldPerformSecurityChecks())
108 return true;
109
110#if ENABLE(SERVICE_WORKER)
111 if (m_options.serviceWorkersMode == ServiceWorkersMode::All && m_async)
112 return m_options.serviceWorkerRegistrationIdentifier || m_document.activeServiceWorker();
113#endif
114
115 return false;
116}

118DocumentThreadableLoader::DocumentThreadableLoader(Document& document, ThreadableLoaderClient& client, BlockingBehavior blockingBehavior, ResourceRequest&& request, const ThreadableLoaderOptions& options, RefPtr<SecurityOrigin>&& origin, std::unique_ptr<ContentSecurityPolicy>&& contentSecurityPolicy, String&& referrer, ShouldLogError shouldLogError)
119 : m_client(&client)
120 , m_document(document)
121 , m_options(options)
122 , m_origin(WTFMove(origin))
123 , m_referrer(WTFMove(referrer))
124 , m_sameOriginRequest(securityOrigin().canRequest(request.url()))
125 , m_simpleRequest(true)
126 , m_async(blockingBehavior == LoadAsynchronously)
127 , m_delayCallbacksForIntegrityCheck(!m_options.integrity.isEmpty())
128 , m_contentSecurityPolicy(WTFMove(contentSecurityPolicy))
129 , m_shouldLogError(shouldLogError)
130{
131 relaxAdoptionRequirement();
132
133 // Setting a referrer header is only supported in the async code path.
134 ASSERT(m_async || m_referrer.isEmpty());
135
136 if (document.settings().disallowSyncXHRDuringPageDismissalEnabled() && !m_async && (!document.page() || !document.page()->areSynchronousLoadsAllowed())) {
137 document.didRejectSyncXHRDuringPageDismissal();
138 logErrorAndFail(ResourceError(errorDomainWebKitInternal, 0, request.url(), "Synchronous loads are not allowed at this time"));
139 return;
140 }
141
142 // Referrer and Origin headers should be set after the preflight if any.
143 ASSERT(!request.hasHTTPReferrer() && !request.hasHTTPOrigin());
144
145 ASSERT_WITH_SECURITY_IMPLICATION(isAllowedByContentSecurityPolicy(request.url(), ContentSecurityPolicy::RedirectResponseReceived::No));
146
147 m_options.storedCredentialsPolicy = (m_options.credentials == FetchOptions::Credentials::Include || (m_options.credentials == FetchOptions::Credentials::SameOrigin && m_sameOriginRequest)) ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse;
148
149 ASSERT(!request.httpHeaderFields().contains(HTTPHeaderName::Origin));
150
151 // Copy headers if we need to replay the request after a redirection.
152 if (m_options.mode == FetchOptions::Mode::Cors)
153 m_originalHeaders = request.httpHeaderFields();
154
155 if (shouldSetHTTPHeadersToKeep())
156 m_options.httpHeadersToKeep = httpHeadersToKeepFromCleaning(request.httpHeaderFields());
157
158 bool shouldDisableCORS = document.isRunningUserScripts() && LegacySchemeRegistry::isUserExtensionScheme(request.url().protocol().toStringWithoutCopying());
159 if (auto* page = document.page())
160 shouldDisableCORS |= page->shouldDisableCorsForRequestTo(request.url());
161
162 if (shouldDisableCORS) {
163 m_options.mode = FetchOptions::Mode::NoCors;
164 m_options.filteringPolicy = ResponseFilteringPolicy::Disable;
165 m_responsesCanBeOpaque = false;
166 }
167
168 m_options.cspResponseHeaders = m_options.contentSecurityPolicyEnforcement != ContentSecurityPolicyEnforcement::DoNotEnforce ? this->contentSecurityPolicy().responseHeaders() : ContentSecurityPolicyResponseHeaders { };
169
170 // As per step 11 of https://fetch.spec.whatwg.org/#main-fetch, data scheme (if same-origin data-URL flag is set) and about scheme are considered same-origin.
171 if (request.url().protocolIsData())
172 m_sameOriginRequest = options.sameOriginDataURLFlag == SameOriginDataURLFlag::Set;
173
174 if (m_sameOriginRequest || m_options.mode == FetchOptions::Mode::NoCors || m_options.mode == FetchOptions::Mode::Navigate) {
175 loadRequest(WTFMove(request), SecurityCheckPolicy::DoSecurityCheck);
176 return;
177 }
178
179 if (m_options.mode == FetchOptions::Mode::SameOrigin) {
180 logErrorAndFail(ResourceError(errorDomainWebKitInternal, 0, request.url(), "Cross origin requests are not allowed when using same-origin fetch mode."));
181 return;
182 }
183
184 makeCrossOriginAccessRequest(WTFMove(request));
185}

187bool DocumentThreadableLoader::checkURLSchemeAsCORSEnabled(const URL& url)
188{
189 // Cross-origin requests are only allowed for HTTP and registered schemes. We would catch this when checking response headers later, but there is no reason to send a request that's guaranteed to be denied.
190 if (!LegacySchemeRegistry::shouldTreatURLSchemeAsCORSEnabled(url.protocol().toStringWithoutCopying())) {
191 logErrorAndFail(ResourceError(errorDomainWebKitInternal, 0, url, "Cross origin requests are only supported for HTTP.", ResourceError::Type::AccessControl));
192 return false;
193 }
194 return true;
195}

197void DocumentThreadableLoader::makeCrossOriginAccessRequest(ResourceRequest&& request)
198{
199 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
200
201#if PLATFORM(IOS_FAMILY)
202 bool needsPreflightQuirk = IOSApplication::isMoviStarPlus() && applicationSDKVersion() < DYLD_IOS_VERSION_12_0 && (m_options.preflightPolicy == PreflightPolicy::Consider || m_options.preflightPolicy == PreflightPolicy::Force);
203#else
204 bool needsPreflightQuirk = false;
205#endif
206
207 if ((m_options.preflightPolicy == PreflightPolicy::Consider && isSimpleCrossOriginAccessRequest(request.httpMethod(), request.httpHeaderFields())) || m_options.preflightPolicy == PreflightPolicy::Prevent || (shouldPerformSecurityChecks() && !needsPreflightQuirk)) {
208 if (checkURLSchemeAsCORSEnabled(request.url()))
209 makeSimpleCrossOriginAccessRequest(WTFMove(request));
210 } else {
211#if ENABLE(SERVICE_WORKER)
212 if (m_options.serviceWorkersMode == ServiceWorkersMode::All && m_async) {
213 if (m_options.serviceWorkerRegistrationIdentifier || document().activeServiceWorker()) {
214 ASSERT(!m_bypassingPreflightForServiceWorkerRequest);
215 m_bypassingPreflightForServiceWorkerRequest = WTFMove(request);
216 m_options.serviceWorkersMode = ServiceWorkersMode::Only;
217 loadRequest(ResourceRequest { m_bypassingPreflightForServiceWorkerRequest.value() }, SecurityCheckPolicy::SkipSecurityCheck);
218 return;
219 }
220 }
221#endif
222 if (!needsPreflightQuirk && !checkURLSchemeAsCORSEnabled(request.url()))
223 return;
224
225 m_simpleRequest = false;
226 if (auto* page = document().page(); page && CrossOriginPreflightResultCache::singleton().canSkipPreflight(page->sessionID(), securityOrigin().toString(), request.url(), m_options.storedCredentialsPolicy, request.httpMethod(), request.httpHeaderFields()))
227 preflightSuccess(WTFMove(request));
228 else
229 makeCrossOriginAccessRequestWithPreflight(WTFMove(request));
230 }
231}
232
233void DocumentThreadableLoader::makeSimpleCrossOriginAccessRequest(ResourceRequest&& request)
234{
235 ASSERT(m_options.preflightPolicy != PreflightPolicy::Force || shouldPerformSecurityChecks());
236 ASSERT(m_options.preflightPolicy == PreflightPolicy::Prevent || isSimpleCrossOriginAccessRequest(request.httpMethod(), request.httpHeaderFields()) || shouldPerformSecurityChecks());
237
238 updateRequestForAccessControl(request, securityOrigin(), m_options.storedCredentialsPolicy);
239 loadRequest(WTFMove(request), SecurityCheckPolicy::DoSecurityCheck);
240}

242void DocumentThreadableLoader::makeCrossOriginAccessRequestWithPreflight(ResourceRequest&& request)
243{
244 if (m_async) {
245 m_preflightChecker.emplace(*this, WTFMove(request));

258void DocumentThreadableLoader::cancel()
259{
260 Ref<DocumentThreadableLoader> protectedThis(*this);
261
262 // Cancel can re-enter and m_resource might be null here as a result.
263 if (m_client && m_resource) {
264 // FIXME: This error is sent to the client in didFail(), so it should not be an internal one. Use FrameLoaderClient::cancelledError() instead.
265 ResourceError error(errorDomainWebKitInternal, 0, m_resource->url(), "Load cancelled", ResourceError::Type::Cancellation);
266 m_client->didFail(error);
267 }
268 clearResource();
269 m_client = nullptr;
270}

293void DocumentThreadableLoader::clearResource()
294{
295 // Script can cancel and restart a request reentrantly within removeClient(),
296 // which could lead to calling CachedResource::removeClient() multiple times for
297 // this DocumentThreadableLoader. Save off a copy of m_resource and clear it to
298 // prevent the reentrancy.
299 if (CachedResourceHandle<CachedRawResource> resource = m_resource) {
300 m_resource = nullptr;
301 resource->removeClient(*this);
302 }
303 if (m_preflightChecker)
304 m_preflightChecker = std::nullopt;
305}

307void DocumentThreadableLoader::redirectReceived(CachedResource& resource, ResourceRequest&& request, const ResourceResponse& redirectResponse, CompletionHandler<void(ResourceRequest&&)>&& completionHandler)
308{
309 ASSERT(m_client);
310 ASSERT_UNUSED(resource, &resource == m_resource);
311
312 Ref<DocumentThreadableLoader> protectedThis(*this);
313 --m_options.maxRedirectCount;
314
315 // FIXME: We restrict this check to Fetch API for the moment, as this might disrupt WorkerScriptLoader.
316 // Reassess this check based on https://github.com/whatwg/fetch/issues/393 discussions.
317 // We should also disable that check in navigation mode.
318 if (!request.url().protocolIsInHTTPFamily() && m_options.initiator == cachedResourceRequestInitiators().fetch) {
319 reportRedirectionWithBadScheme(request.url());
320 clearResource();
321 return completionHandler(WTFMove(request));
322 }
323
324 if (platformStrategies()->loaderStrategy()->havePerformedSecurityChecks(redirectResponse)) {
325 completionHandler(WTFMove(request));
326 return;
327 }
328
329 if (!isAllowedByContentSecurityPolicy(request.url(), redirectResponse.isNull() ? ContentSecurityPolicy::RedirectResponseReceived::No : ContentSecurityPolicy::RedirectResponseReceived::Yes)) {
330 reportContentSecurityPolicyError(redirectResponse.url());
331 clearResource();
332 return completionHandler(WTFMove(request));
333 }
334
335 // Allow same origin requests to continue after allowing clients to audit the redirect.
336 if (isAllowedRedirect(request.url()))
337 return completionHandler(WTFMove(request));
338
339 // Force any subsequent request to use these checks.
340 m_sameOriginRequest = false;
341
342 ASSERT(m_resource);
343 ASSERT(m_originalHeaders);
344
345 // Use a unique for subsequent loads if needed.
346 // https://fetch.spec.whatwg.org/#concept-http-redirect-fetch (Step 10).
347 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
348 if (!securityOrigin().canRequest(redirectResponse.url()) && !protocolHostAndPortAreEqual(redirectResponse.url(), request.url()))
349 m_origin = SecurityOrigin::createUnique();
350
351 // Except in case where preflight is needed, loading should be able to continue on its own.
352 // But we also handle credentials here if it is restricted to SameOrigin.
353 if (m_options.credentials != FetchOptions::Credentials::SameOrigin && m_simpleRequest && isSimpleCrossOriginAccessRequest(request.httpMethod(), *m_originalHeaders))
354 return completionHandler(WTFMove(request));
355
356 if (m_options.credentials == FetchOptions::Credentials::SameOrigin)
357 m_options.storedCredentialsPolicy = StoredCredentialsPolicy::DoNotUse;
358
359 clearResource();
360
361 m_referrer = request.httpReferrer();
362 if (m_referrer.isNull())
363 m_options.referrerPolicy = ReferrerPolicy::NoReferrer;
364
365 // Let's fetch the request with the original headers (equivalent to request cloning specified by fetch algorithm).
366 // Do not copy the Authorization header if removed by the network layer.
367 if (!request.httpHeaderFields().contains(HTTPHeaderName::Authorization))
368 m_originalHeaders->remove(HTTPHeaderName::Authorization);
369 request.setHTTPHeaderFields(*m_originalHeaders);
370
371#if ENABLE(SERVICE_WORKER)
372 if (redirectResponse.source() != ResourceResponse::Source::ServiceWorker && redirectResponse.source() != ResourceResponse::Source::MemoryCache)
373 m_options.serviceWorkersMode = ServiceWorkersMode::None;
374#endif
375 makeCrossOriginAccessRequest(ResourceRequest(request));
376 completionHandler(WTFMove(request));
377}

400void DocumentThreadableLoader::didReceiveResponse(unsigned long identifier, const ResourceResponse& response)
401{
402 ASSERT(m_client);
403 ASSERT(response.type() != ResourceResponse::Type::Error);
404
405 InspectorInstrumentation::didReceiveThreadableLoaderResponse(*this, identifier);
406
407 if (m_delayCallbacksForIntegrityCheck)
408 return;
409
410 if (options().filteringPolicy == ResponseFilteringPolicy::Disable) {
411 m_client->didReceiveResponse(identifier, response);
412 return;
413 }
414
415 if (response.type() == ResourceResponse::Type::Default) {
416 m_client->didReceiveResponse(identifier, ResourceResponse::filter(response, m_options.credentials == FetchOptions::Credentials::Include ? ResourceResponse::PerformExposeAllHeadersCheck::No : ResourceResponse::PerformExposeAllHeadersCheck::Yes));
417 if (response.tainting() == ResourceResponse::Tainting::Opaque) {
418 clearResource();
419 if (m_client)
420 m_client->didFinishLoading(identifier);
421 }
422 return;
423 }
424 ASSERT(response.type() == ResourceResponse::Type::Opaqueredirect || response.source() == ResourceResponse::Source::ServiceWorker || response.source() == ResourceResponse::Source::MemoryCache);
425 m_client->didReceiveResponse(identifier, response);
426}

469
470void DocumentThreadableLoader::didFinishLoading(unsigned long identifier)
471{
472 ASSERT(m_client);
473
474 if (m_delayCallbacksForIntegrityCheck) {
475 if (!matchIntegrityMetadata(*m_resource, m_options.integrity)) {
476 reportIntegrityMetadataError(*m_resource, m_options.integrity);
477 return;
478 }
479
480 auto response = m_resource->response();
481
482 if (options().filteringPolicy == ResponseFilteringPolicy::Disable) {
483 m_client->didReceiveResponse(identifier, response);
484 if (auto* buffer = m_resource->resourceBuffer()) {
485 buffer->forEachSegment([&](auto& segment) {
486 m_client->didReceiveData(segment.data(), segment.size());
487 });
488 }
489 } else {
490 ASSERT(response.type() == ResourceResponse::Type::Default);
491
492 m_client->didReceiveResponse(identifier, ResourceResponse::filter(response, m_options.credentials == FetchOptions::Credentials::Include ? ResourceResponse::PerformExposeAllHeadersCheck::No : ResourceResponse::PerformExposeAllHeadersCheck::Yes));
493 if (auto* buffer = m_resource->resourceBuffer()) {
494 buffer->forEachSegment([&](auto& segment) {
495 m_client->didReceiveData(segment.data(), segment.size());
496 });
497 }
498 }
499 }
500
501 m_client->didFinishLoading(identifier);
502}
503
504void DocumentThreadableLoader::didFail(unsigned long, const ResourceError& error)
505{
506 ASSERT(m_client);
507#if ENABLE(SERVICE_WORKER)
508 if (m_bypassingPreflightForServiceWorkerRequest && error.isCancellation()) {
509 clearResource();
510
511 m_options.serviceWorkersMode = ServiceWorkersMode::None;
512 makeCrossOriginAccessRequestWithPreflight(WTFMove(m_bypassingPreflightForServiceWorkerRequest.value()));
513 ASSERT(m_bypassingPreflightForServiceWorkerRequest->isNull());
514 m_bypassingPreflightForServiceWorkerRequest = std::nullopt;
515 return;
516 }
517#endif
518
519 if (m_shouldLogError == ShouldLogError::Yes)
520 logError(m_document, error, m_options.initiator);
521
522 m_client->didFail(error);
523}

525void DocumentThreadableLoader::preflightSuccess(ResourceRequest&& request)
526{
527 ResourceRequest actualRequest(WTFMove(request));
528 updateRequestForAccessControl(actualRequest, securityOrigin(), m_options.storedCredentialsPolicy);

536void DocumentThreadableLoader::preflightFailure(unsigned long identifier, const ResourceError& error)
537{

548void DocumentThreadableLoader::loadRequest(ResourceRequest&& request, SecurityCheckPolicy securityCheck)
549{
550 Ref<DocumentThreadableLoader> protectedThis(*this);
551
552 // Any credential should have been removed from the cross-site requests.
553 const URL& requestURL = request.url();
554 m_options.securityCheck = securityCheck;
555 ASSERT(m_sameOriginRequest || !requestURL.hasCredentials());
556
557 if (!m_referrer.isNull())
558 request.setHTTPReferrer(m_referrer);
559
560 if (m_async) {
561 ResourceLoaderOptions options = m_options;
562 options.clientCredentialPolicy = m_sameOriginRequest ? ClientCredentialPolicy::MayAskClientForCredentials : ClientCredentialPolicy::CannotAskClientForCredentials;
563 options.contentSecurityPolicyImposition = ContentSecurityPolicyImposition::SkipPolicyCheck;
564
565 // If there is integrity metadata to validate, we must buffer.
566 if (!m_options.integrity.isEmpty())
567 options.dataBufferingPolicy = DataBufferingPolicy::BufferData;
568
569 request.setAllowCookies(m_options.storedCredentialsPolicy == StoredCredentialsPolicy::Use);
570 CachedResourceRequest newRequest(WTFMove(request), options);
571 newRequest.setInitiator(m_options.initiator);
572 newRequest.setOrigin(securityOrigin());
573
574 ASSERT(!m_resource);
575 if (m_resource) {
576 CachedResourceHandle<CachedRawResource> resource = std::exchange(m_resource, nullptr);
577 resource->removeClient(*this);
578 }
579
580 auto cachedResource = m_document.cachedResourceLoader().requestRawResource(WTFMove(newRequest));
581 m_resource = cachedResource.value_or(nullptr);
582 if (m_resource)
583 m_resource->addClient(*this);
584 else
585 logErrorAndFail(cachedResource.error());
586 return;
587 }
588
589 // If credentials mode is 'Omit', we should disable cookie sending.
590 ASSERT(m_options.credentials != FetchOptions::Credentials::Omit);
591
592 ResourceLoadTiming loadTiming;
593 loadTiming.markStartTime();
594
595 // FIXME: ThreadableLoaderOptions.sniffContent is not supported for synchronous requests.
596 RefPtr<SharedBuffer> data;
597 ResourceError error;
598 ResourceResponse response;
599 unsigned long identifier = std::numeric_limits<unsigned long>::max();
600 if (auto* frame = m_document.frame()) {
601 if (!MixedContentChecker::canRunInsecureContent(*frame, m_document.securityOrigin(), requestURL))
602 return;
603 auto& frameLoader = frame->loader();
604 identifier = frameLoader.loadResourceSynchronously(request, m_options.clientCredentialPolicy, m_options, *m_originalHeaders, error, response, data);
605 }
606
607 loadTiming.markEndTime();
608
609 if (!error.isNull() && response.httpStatusCode() <= 0) {
610 if (requestURL.isLocalFile()) {
611 // We don't want XMLHttpRequest to raise an exception for file:// resources, see <rdar://problem/4962298>.
612 // FIXME: XMLHttpRequest quirks should be in XMLHttpRequest code, not in DocumentThreadableLoader.cpp.
613 didReceiveResponse(identifier, response);
614 didFinishLoading(identifier);
615 return;
616 }
617 logErrorAndFail(error);
618 return;
619 }
620
621 if (response.containsInvalidHTTPHeaders()) {
622 didFail(identifier, ResourceError(errorDomainWebKitInternal, 0, request.url(), "Response contained invalid HTTP headers", ResourceError::Type::General));
623 return;
624 }
625
626 if (!shouldPerformSecurityChecks()) {
627 // FIXME: FrameLoader::loadSynchronously() does not tell us whether a redirect happened or not, so we guess by comparing the
628 // request and response URLs. This isn't a perfect test though, since a server can serve a redirect to the same URL that was
629 // requested. Also comparing the request and response URLs as strings will fail if the requestURL still has its credentials.
630 bool didRedirect = requestURL != response.url();
631 if (didRedirect) {
632 if (!isAllowedByContentSecurityPolicy(response.url(), ContentSecurityPolicy::RedirectResponseReceived::Yes)) {
633 reportContentSecurityPolicyError(requestURL);
634 return;
635 }
636 if (!isAllowedRedirect(response.url())) {
637 reportCrossOriginResourceSharingError(requestURL);
638 return;
639 }
640 }
641
642 if (!m_sameOriginRequest) {
643 if (m_options.mode == FetchOptions::Mode::NoCors)
644 response.setTainting(ResourceResponse::Tainting::Opaque);
645 else {
646 ASSERT(m_options.mode == FetchOptions::Mode::Cors);
647 response.setTainting(ResourceResponse::Tainting::Cors);
648 auto accessControlCheckResult = passesAccessControlCheck(response, m_options.storedCredentialsPolicy, securityOrigin(), &CrossOriginAccessControlCheckDisabler::singleton());
649 if (!accessControlCheckResult) {
650 logErrorAndFail(ResourceError(errorDomainWebKitInternal, 0, response.url(), accessControlCheckResult.error(), ResourceError::Type::AccessControl));
651 return;
652 }
653 }
654 }
655 }
656 didReceiveResponse(identifier, response);
657
658 if (data) {
659 data->forEachSegment([&](auto& segment) {
660 didReceiveData(identifier, segment.data(), segment.size());
661 });
662 }
663
664 const auto* timing = response.deprecatedNetworkLoadMetricsOrNull();
665 auto resourceTiming = ResourceTiming::fromSynchronousLoad(requestURL, m_options.initiator, loadTiming, timing ? *timing : NetworkLoadMetrics { }, response, securityOrigin());
666 if (options().initiatorContext == InitiatorContext::Worker)
667 finishedTimingForWorkerLoad(resourceTiming);
668 else {
669 if (auto* window = document().domWindow())
670 window->performance().addResourceTiming(WTFMove(resourceTiming));
671 }
672
673 didFinishLoading(identifier);
674}

676bool DocumentThreadableLoader::isAllowedByContentSecurityPolicy(const URL& url, ContentSecurityPolicy::RedirectResponseReceived redirectResponseReceived)
677{
678 switch (m_options.contentSecurityPolicyEnforcement) {
679 case ContentSecurityPolicyEnforcement::DoNotEnforce:
680 return true;
681 case ContentSecurityPolicyEnforcement::EnforceChildSrcDirective:
682 return contentSecurityPolicy().allowChildContextFromSource(url, redirectResponseReceived);
683 case ContentSecurityPolicyEnforcement::EnforceConnectSrcDirective:
684 return contentSecurityPolicy().allowConnectToSource(url, redirectResponseReceived);
685 case ContentSecurityPolicyEnforcement::EnforceScriptSrcDirective:
686 return contentSecurityPolicy().allowScriptFromSource(url, redirectResponseReceived);
687 }
688 ASSERT_NOT_REACHED();
689 return false;
690}

692bool DocumentThreadableLoader::isAllowedRedirect(const URL& url)
693{
694 if (m_options.mode == FetchOptions::Mode::NoCors)
695 return true;
696
697 return m_sameOriginRequest && securityOrigin().canRequest(url);
698}
