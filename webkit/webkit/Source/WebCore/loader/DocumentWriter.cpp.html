56static inline bool canReferToParentFrameEncoding(const Frame* frame, const Frame* parentFrame)
57{
58 if (is<XMLDocument>(frame->document()))
59 return false;
60 return parentFrame && parentFrame->document()->securityOrigin().isSameOriginDomain(frame->document()->securityOrigin());
61}

63// This is only called by ScriptController::executeIfJavaScriptURL
64// and always contains the result of evaluating a javascript: url.
65// This is the <iframe src="javascript:'html'"> case.
66void DocumentWriter::replaceDocumentWithResultOfExecutingJavascriptURL(const String& source, Document* ownerDocument)
67{
68 m_frame->loader().stopAllLoaders();
69
70 // If we are in the midst of changing the frame's document, don't execute script
71 // that modifies the document further:
72 if (m_frame->documentIsBeingReplaced())
73 return;
74
75 begin(m_frame->document()->url(), true, ownerDocument);
76
77 // begin() might fire an unload event, which will result in a situation where no new document has been attached,
78 // and the old document has been detached. Therefore, bail out if no document is attached.
79 if (!m_frame->document())
80 return;
81
82 if (!source.isNull()) {
83 if (!m_hasReceivedSomeData) {
84 m_hasReceivedSomeData = true;
85 m_frame->document()->setCompatibilityMode(DocumentCompatibilityMode::NoQuirksMode);
86 }
87
88 // FIXME: This should call DocumentParser::appendBytes instead of append
89 // to support RawDataDocumentParsers.
90 if (DocumentParser* parser = m_frame->document()->parser())
91 parser->append(source.impl());
92 }
93
94 end();
95}

110Ref<Document> DocumentWriter::createDocument(const URL& url)
111{
112 if (!m_frame->loader().stateMachine().isDisplayingInitialEmptyDocument() && m_frame->loader().client().shouldAlwaysUsePluginDocument(m_mimeType))
113 return PluginDocument::create(*m_frame, url);
114#if PLATFORM(IOS_FAMILY)
115 if (MIMETypeRegistry::isPDFMIMEType(m_mimeType) && (m_frame->isMainFrame() || !m_frame->settings().useImageDocumentForSubframePDF()))
116 return SinkDocument::create(*m_frame, url);
117#endif
118 if (!m_frame->loader().client().hasHTMLView())
119 return Document::createNonRenderedPlaceholder(*m_frame, url);
120 return DOMImplementation::createDocument(m_mimeType, m_frame.get(), m_frame->settings(), url);
121}

123bool DocumentWriter::begin(const URL& urlReference, bool dispatch, Document* ownerDocument)
124{
125 // We grab a local copy of the URL because it's easy for callers to supply
126 // a URL that will be deallocated during the execution of this function.
127 // For example, see <https://bugs.webkit.org/show_bug.cgi?id=66360>.
128 URL url = urlReference;
129
130 // Create a new document before clearing the frame, because it may need to
131 // inherit an aliased security context.
132 Ref<Document> document = createDocument(url);
133
134 // If the new document is for a Plugin but we're supposed to be sandboxed from Plugins,
135 // then replace the document with one whose parser will ignore the incoming data (bug 39323)
136 if (document->isPluginDocument() && document->isSandboxed(SandboxPlugins))
137 document = SinkDocument::create(*m_frame, url);
138
139 // FIXME: Do we need to consult the content security policy here about blocked plug-ins?
140
141 bool shouldReuseDefaultView = m_frame->loader().stateMachine().isDisplayingInitialEmptyDocument()
142 && m_frame->document()->isSecureTransitionTo(url)
143 && (!m_frame->window() || !m_frame->window()->wasWrappedWithoutInitializedSecurityOrigin());
144
145 // Temporarily extend the lifetime of the existing document so that FrameLoader::clear() doesn't destroy it as
146 // we need to retain its ongoing set of upgraded requests in new navigation contexts per <http://www.w3.org/TR/upgrade-insecure-requests/>
147 // and we may also need to inherit its Content Security Policy below.
148 RefPtr<Document> existingDocument = m_frame->document();
149
150 Function<void()> handleDOMWindowCreation = [this, document, shouldReuseDefaultView] {
151 if (shouldReuseDefaultView)
152 document->takeDOMWindowFrom(*m_frame->document());
153 else
154 document->createDOMWindow();
155 };
156
157 m_frame->loader().clear(document.ptr(), !shouldReuseDefaultView, !shouldReuseDefaultView, true, WTFMove(handleDOMWindowCreation));
158 clear();
159
160 // m_frame->loader().clear() might fire unload event which could remove the view of the document.
161 // Bail out if document has no view.
162 if (!document->view())
163 return false;
164
165 if (!shouldReuseDefaultView)
166 m_frame->script().updatePlatformScriptObjects();
167
168 m_frame->loader().setOutgoingReferrer(url);
169 m_frame->setDocument(document.copyRef());
170
171 if (m_decoder)
172 document->setDecoder(m_decoder.get());
173 if (ownerDocument) {
174 // |document| is the result of evaluating a JavaScript URL.
175 document->setCookieURL(ownerDocument->cookieURL());
176 document->setSecurityOriginPolicy(ownerDocument->securityOriginPolicy());
177 document->setStrictMixedContentMode(ownerDocument->isStrictMixedContentMode());
178
179 document->setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { url }, document));
180 document->contentSecurityPolicy()->copyStateFrom(ownerDocument->contentSecurityPolicy());
181 document->contentSecurityPolicy()->setInsecureNavigationRequestsToUpgrade(ownerDocument->contentSecurityPolicy()->takeNavigationRequestsToUpgrade());
182 } else if (existingDocument) {
183 if (url.protocolIsData() || url.protocolIsBlob()) {
184 document->setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { url }, document));
185 document->contentSecurityPolicy()->copyStateFrom(existingDocument->contentSecurityPolicy());
186
187 // Fix up 'self' for blob: and data:, which is inherited from its embedding document or opener.
188 auto* parentFrame = m_frame->tree().parent();
189 if (auto* ownerFrame = parentFrame ? parentFrame : m_frame->loader().opener())
190 document->contentSecurityPolicy()->updateSourceSelf(ownerFrame->document()->securityOrigin());
191 }
192 document->contentSecurityPolicy()->setInsecureNavigationRequestsToUpgrade(existingDocument->contentSecurityPolicy()->takeNavigationRequestsToUpgrade());
193 }
194
195 auto protectedFrame = makeRef(*m_frame);
196
197 m_frame->loader().didBeginDocument(dispatch);
198
199 document->implicitOpen();
200
201 // We grab a reference to the parser so that we'll always send data to the
202 // original parser, even if the document acquires a new parser (e.g., via
203 // document.open).
204 m_parser = document->parser();
205
206 if (m_frame->view() && m_frame->loader().client().hasHTMLView())
207 m_frame->view()->setContentsSize(IntSize());
208
209 m_state = State::Started;
210 return true;
211}

213TextResourceDecoder& DocumentWriter::decoder()
214{
215 if (!m_decoder) {
216 m_decoder = TextResourceDecoder::create(m_mimeType,
217 m_frame->settings().defaultTextEncodingName(),
218 m_frame->settings().usesEncodingDetector());
219 Frame* parentFrame = m_frame->tree().parent();
220 // Set the hint encoding to the parent frame encoding only if
221 // the parent and the current frames share the security origin.
222 // We impose this condition because somebody can make a child frame
223 // containing a carefully crafted html/javascript in one encoding
224 // that can be mistaken for hintEncoding (or related encoding) by
225 // an auto detector. When interpreted in the latter, it could be
226 // an attack vector.
227 // FIXME: This might be too cautious for non-7bit-encodings and
228 // we may consider relaxing this later after testing.
229 if (canReferToParentFrameEncoding(m_frame.get(), parentFrame))
230 m_decoder->setHintEncoding(parentFrame->document()->decoder());
231 if (m_encoding.isEmpty()) {
232 if (canReferToParentFrameEncoding(m_frame.get(), parentFrame))
233 m_decoder->setEncoding(parentFrame->document()->textEncoding(), TextResourceDecoder::EncodingFromParentFrame);
234 } else {
235 m_decoder->setEncoding(m_encoding,
236 m_encodingWasChosenByUser ? TextResourceDecoder::UserChosenEncoding : TextResourceDecoder::EncodingFromHTTPHeader);
237 }
238 m_frame->document()->setDecoder(m_decoder.get());
239 }
240 return *m_decoder;
241}

243void DocumentWriter::reportDataReceived()
244{
245 ASSERT(m_decoder);
246 if (m_hasReceivedSomeData)
247 return;
248 m_hasReceivedSomeData = true;
249 if (m_decoder->encoding().usesVisualOrdering())
250 m_frame->document()->setVisuallyOrdered();
251 m_frame->document()->resolveStyle(Document::ResolveStyleType::Rebuild);
252}

254void DocumentWriter::addData(const uint8_t* bytes, size_t length)
255{
256 // FIXME: Change these to ASSERT once https://bugs.webkit.org/show_bug.cgi?id=80427 has been resolved.
257 RELEASE_ASSERT(m_state != State::NotStarted);
258 if (m_state == State::Finished) {
259 ASSERT_NOT_REACHED();
260 return;
261 }
262 ASSERT(m_parser);
263 m_parser->appendBytes(*this, bytes, length);
264}

266void DocumentWriter::insertDataSynchronously(const String& markup)

268 ASSERT(m_state != State::NotStarted);

271 m_parser->insert(markup);

274void DocumentWriter::end()
275{
276 ASSERT(m_frame->page());
277 ASSERT(m_frame->document());
278
279 // The parser is guaranteed to be released after this point. begin() would
280 // have to be called again before we can start writing more data.
281 m_state = State::Finished;
282
283 // http://bugs.webkit.org/show_bug.cgi?id=10854
284 // The frame's last ref may be removed and it can be deleted by checkCompleted(),
285 // so we'll add a protective refcount
286 Ref<Frame> protect(*m_frame);
287
288 if (!m_parser)
289 return;
290 // FIXME: m_parser->finish() should imply m_parser->flush().
291 m_parser->flush(*this);
292 if (!m_parser)
293 return;
294 m_parser->finish();
295 m_parser = nullptr;
296}
