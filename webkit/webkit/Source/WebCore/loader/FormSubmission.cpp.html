58static int64_t generateFormDataIdentifier()
59{
60 // Initialize to the current time to reduce the likelihood of generating
61 // identifiers that overlap with those from past/future browser sessions.
62 static int64_t nextIdentifier = static_cast<int64_t>(WallTime::now().secondsSinceEpoch().microseconds());
63 return ++nextIdentifier;
64}

66static void appendMailtoPostFormDataToURL(URL& url, const FormData& data, const String& encodingType)
67{
68 String body = data.flattenToString();
69
70 if (equalLettersIgnoringASCIICase(encodingType, "text/plain")) {
71 // Convention seems to be to decode, and s/&/\r\n/. Also, spaces are encoded as %20.
72 body = decodeURLEscapeSequences(makeString(body.replaceWithLiteral('&', "\r\n").replace('+', ' '), "\r\n"));
73 }
74
75 Vector<char> bodyData;
76 bodyData.append("body=", 5);
77 FormDataBuilder::encodeStringAsFormData(bodyData, body.utf8());
78 body = String(bodyData.data(), bodyData.size()).replaceWithLiteral('+', "%20");
79
80 auto query = url.query();
81 if (query.isEmpty())
82 url.setQuery(body);
83 else
84 url.setQuery(makeString(query, '&', body));
85}

100String FormSubmission::Attributes::parseEncodingType(const String& type)
101{
102 if (equalLettersIgnoringASCIICase(type, "multipart/form-data"))
103 return "multipart/form-data"_s;
104 if (equalLettersIgnoringASCIICase(type, "text/plain"))
105 return "text/plain"_s;
106 return "application/x-www-form-urlencoded"_s;
107}

109void FormSubmission::Attributes::updateEncodingType(const String& type)
110{
111 m_encodingType = parseEncodingType(type);
112 m_isMultiPartForm = (m_encodingType == "multipart/form-data");

115FormSubmission::Method FormSubmission::Attributes::parseMethodType(const String& type)
116{
117 if (RuntimeEnabledFeatures::sharedFeatures().dialogElementEnabled() && equalLettersIgnoringASCIICase(type, "dialog"))

155static TextEncoding encodingFromAcceptCharset(const String& acceptCharset, Document& document)
156{
157 String normalizedAcceptCharset = acceptCharset;
158 normalizedAcceptCharset.replace(',', ' ');
159
160 for (auto& charset : normalizedAcceptCharset.split(' ')) {
161 TextEncoding encoding(charset);
162 if (encoding.isValid())
163 return encoding;
164 }
165
166 return document.textEncoding();
167}

169Ref<FormSubmission> FormSubmission::create(HTMLFormElement& form, HTMLFormControlElement* overrideSubmitter, const Attributes& attributes, Event* event, LockHistory lockHistory, FormSubmissionTrigger trigger)
170{
171 auto copiedAttributes = attributes;
172
173 auto submitter = makeRefPtr(overrideSubmitter ? overrideSubmitter : form.findSubmitter(event));
174 if (submitter) {
175 AtomString attributeValue;
176 if (!(attributeValue = submitter->attributeWithoutSynchronization(formactionAttr)).isNull())
177 copiedAttributes.parseAction(attributeValue);
178 if (!(attributeValue = submitter->attributeWithoutSynchronization(formenctypeAttr)).isNull())
179 copiedAttributes.updateEncodingType(attributeValue);
180 if (!(attributeValue = submitter->attributeWithoutSynchronization(formmethodAttr)).isNull())
181 copiedAttributes.updateMethodType(attributeValue);
182 if (!(attributeValue = submitter->attributeWithoutSynchronization(formtargetAttr)).isNull())
183 copiedAttributes.setTarget(attributeValue);
184 }
185
186 auto& document = form.document();
187 auto encodingType = copiedAttributes.encodingType();
188 auto actionURL = document.completeURL(copiedAttributes.action().isEmpty() ? document.url().string() : copiedAttributes.action());
189
190 if (RuntimeEnabledFeatures::sharedFeatures().dialogElementEnabled() && copiedAttributes.method() == Method::Dialog) {
191 String returnValue = submitter ? submitter->resultForDialogSubmit() : emptyString();
192 return adoptRef(*new FormSubmission(copiedAttributes.method(), returnValue, actionURL, form.effectiveTarget(event, submitter.get()), encodingType, lockHistory, event));
193 }
194
195 ASSERT(copiedAttributes.method() == Method::Post || copiedAttributes.method() == Method::Get);
196
197 bool isMailtoForm = actionURL.protocolIs("mailto");
198 bool isMultiPartForm = false;
199
200 document.contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(actionURL, ContentSecurityPolicy::InsecureRequestType::FormSubmission);
201
202 if (copiedAttributes.method() == Method::Post) {
203 isMultiPartForm = copiedAttributes.isMultiPartForm();
204 if (isMultiPartForm && isMailtoForm) {
205 encodingType = "application/x-www-form-urlencoded";
206 isMultiPartForm = false;
207 }
208 }
209
210 auto dataEncoding = isMailtoForm ? UTF8Encoding() : encodingFromAcceptCharset(copiedAttributes.acceptCharset(), document);
211 auto domFormData = DOMFormData::create(dataEncoding.encodingForFormSubmissionOrURLParsing());
212 StringPairVector formValues;
213
214 bool containsPasswordData = false;
215 for (auto& control : form.copyAssociatedElementsVector()) {
216 auto& element = control->asHTMLElement();
217 if (!element.isDisabledFormControl())
218 control->appendFormData(domFormData, isMultiPartForm);
219 if (is<HTMLInputElement>(element)) {
220 auto& input = downcast<HTMLInputElement>(element);
221 if (input.isTextField()) {
222 formValues.append({ input.name(), input.value() });
223 input.addSearchResult();
224 }
225 if (input.isPasswordField() && !input.value().isEmpty())
226 containsPasswordData = true;
227 }
228 }
229
230 RefPtr<FormData> formData;
231 String boundary;
232
233 if (isMultiPartForm) {
234 formData = FormData::createMultiPart(domFormData);
235 boundary = formData->boundary().data();
236 } else {
237 formData = FormData::create(domFormData, attributes.method() == Method::Get ? FormData::FormURLEncoded : FormData::parseEncodingType(encodingType));
238 if (copiedAttributes.method() == Method::Post && isMailtoForm) {
239 // Convert the form data into a string that we put into the URL.
240 appendMailtoPostFormDataToURL(actionURL, *formData, encodingType);
241 formData = FormData::create();
242 }
243 }
244
245 formData->setIdentifier(generateFormDataIdentifier());
246 formData->setContainsPasswordData(containsPasswordData);
247
248 auto formState = FormState::create(form, WTFMove(formValues), document, trigger);
249
250 return adoptRef(*new FormSubmission(copiedAttributes.method(), actionURL, form.effectiveTarget(event, submitter.get()), encodingType, WTFMove(formState), formData.releaseNonNull(), boundary, lockHistory, event));
251}

253URL FormSubmission::requestURL() const
254{
255 ASSERT(m_method == Method::Post || m_method == Method::Get);
256
257 if (m_method == Method::Post)
258 return m_action;
259
260 URL requestURL(m_action);
261 requestURL.setQuery(m_formData->flattenToString());
262 return requestURL;
263}

265void FormSubmission::populateFrameLoadRequest(FrameLoadRequest& frameRequest)
266{
267 ASSERT(m_method == Method::Post || m_method == Method::Get);
268
269 if (!m_target.isEmpty())
270 frameRequest.setFrameName(m_target);
271
272 if (!m_referrer.isEmpty())
273 frameRequest.resourceRequest().setHTTPReferrer(m_referrer);
274
275 if (m_method == Method::Post) {
276 frameRequest.resourceRequest().setHTTPMethod("POST");
277 frameRequest.resourceRequest().setHTTPBody(m_formData.copyRef());
278
279 // construct some user headers if necessary
280 if (m_boundary.isEmpty())
281 frameRequest.resourceRequest().setHTTPContentType(m_contentType);
282 else
283 frameRequest.resourceRequest().setHTTPContentType(m_contentType + "; boundary=" + m_boundary);
284 }
285
286 frameRequest.resourceRequest().setURL(requestURL());
287 FrameLoader::addHTTPOriginIfNeeded(frameRequest.resourceRequest(), m_origin);
288}
