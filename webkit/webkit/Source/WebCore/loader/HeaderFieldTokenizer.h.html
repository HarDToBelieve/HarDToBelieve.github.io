27#pragma once
28
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33// Parses header fields into tokens, quoted strings and separators.
34// Commonly used by ParsedContent* classes.
35class HeaderFieldTokenizer {
36
37public:
38 explicit HeaderFieldTokenizer(const String&);
39
40 // Try to parse a separator character, a token or either a token or a quoted
41 // string from the |header_field| input. Return |true| on success. Return
42 // |false| if the separator character, the token or the quoted string is
43 // missing or invalid.
44 bool consume(UChar);
45 String consumeToken();
46 String consumeTokenOrQuotedString();
47
48 // Consume all characters before (but excluding) any of the characters from
49 // the Vector parameter are found.
50 // Because we potentially have to iterate through the entire Vector for each
51 // character of the base string, the Vector should be small (< 3 members).
52 void consumeBeforeAnyCharMatch(const Vector<UChar>&);
53
54 bool isConsumed() const { return m_index >= m_input.length(); }
55
56private:
57 String consumeQuotedString();
58 void skipSpaces();
59
60 unsigned m_index = 0;
61 const String m_input;
62};
63
64} // namespace WebCore
