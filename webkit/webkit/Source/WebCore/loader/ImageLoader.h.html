24
25#include "CachedImageClient.h"
26#include "CachedResourceHandle.h"
27#include "Element.h"
28#include "Timer.h"
29#include <wtf/Vector.h>

32namespace WebCore {
33
34class DeferredPromise;
35class Document;
36class ImageLoader;
37class Page;
38class RenderImageResource;
39
40template<typename T> class EventSender;
41using ImageEventSender = EventSender<ImageLoader>;
42
43enum class RelevantMutation : bool { Yes, No };
44
45class ImageLoader : public CachedImageClient, public CanMakeWeakPtr<ImageLoader> {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 virtual ~ImageLoader();
49
50 // This function should be called when the element is attached to a document; starts
51 // loading if a load hasn't already been started.
52 void updateFromElement(RelevantMutation = RelevantMutation::No);
53
54 // This function should be called whenever the 'src' attribute is set, even if its value
55 // doesn't change; starts new load unconditionally (matches Firefox and Opera behavior).
56 void updateFromElementIgnoringPreviousError(RelevantMutation = RelevantMutation::No);
57
58 void elementDidMoveToNewDocument(Document&);
59
60 Element& element() { return m_element; }
61 const Element& element() const { return m_element; }
62
63 bool imageComplete() const { return m_imageComplete; }
64
65 CachedImage* image() const { return m_image.get(); }
66 void clearImage(); // Cancels pending beforeload and load events, and doesn't dispatch new ones.
67
68 size_t pendingDecodePromisesCountForTesting() const { return m_decodingPromises.size(); }
69 void decode(Ref<DeferredPromise>&&);
70
71 void setLoadManually(bool loadManually) { m_loadManually = loadManually; }
72
73 bool hasPendingBeforeLoadEvent() const { return m_hasPendingBeforeLoadEvent; }
74 bool hasPendingActivity() const { return m_hasPendingLoadEvent || m_hasPendingErrorEvent; }
75
76 void dispatchPendingEvent(ImageEventSender*);
77
78 static void dispatchPendingBeforeLoadEvents(Page*);
79 static void dispatchPendingLoadEvents(Page*);
80 static void dispatchPendingErrorEvents(Page*);
81
82 void loadDeferredImage();
83
84 bool isDeferred() const { return m_lazyImageLoadState == LazyImageLoadState::Deferred || m_lazyImageLoadState == LazyImageLoadState::LoadImmediately; }
85
86 Document& document() { return m_element.document(); }
87
88protected:
89 explicit ImageLoader(Element&);
90 void notifyFinished(CachedResource&, const NetworkLoadMetrics&) override;
91
92private:
93 void resetLazyImageLoading(Document&);
94 enum class LazyImageLoadState : uint8_t { None, Deferred, LoadImmediately, FullImage };
95
96 virtual void dispatchLoadEvent() = 0;
97 virtual String sourceURI(const AtomString&) const = 0;
98
99 void updatedHasPendingEvent();
100
101 void dispatchPendingBeforeLoadEvent();
102 void dispatchPendingLoadEvent();
103 void dispatchPendingErrorEvent();
104
105 RenderImageResource* renderImageResource();
106 void updateRenderer();
107
108 void clearImageWithoutConsideringPendingLoadEvent();
109 void clearFailedLoadURL();
110
111 bool hasPendingDecodePromises() const { return !m_decodingPromises.isEmpty(); }
112 void resolveDecodePromises();
113 void rejectDecodePromises(const char* message);
114 void decode();
115
116 void timerFired();
117
118 VisibleInViewportState imageVisibleInViewport(const Document&) const override;
119
120 Element& m_element;
121 CachedResourceHandle<CachedImage> m_image;
122 Timer m_derefElementTimer;
123 RefPtr<Element> m_protectedElement;
124 AtomString m_failedLoadURL;
125 Vector<RefPtr<DeferredPromise>> m_decodingPromises;
126 bool m_hasPendingBeforeLoadEvent : 1;
127 bool m_hasPendingLoadEvent : 1;
128 bool m_hasPendingErrorEvent : 1;
129 bool m_imageComplete : 1;
130 bool m_loadManually : 1;
131 bool m_elementIsProtected : 1;
132 LazyImageLoadState m_lazyImageLoadState { LazyImageLoadState::None };
133};
