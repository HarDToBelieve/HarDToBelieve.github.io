34LinkPreloadResourceClient::LinkPreloadResourceClient(LinkLoader& loader, CachedResource& resource)
35{
36 m_loader = makeWeakPtr(loader);
37 m_resource = &resource;
38}
39
40void LinkPreloadResourceClient::triggerEvents(const CachedResource& resource)
41{
42 if (m_loader)
43 m_loader->triggerEvents(resource);
44}
