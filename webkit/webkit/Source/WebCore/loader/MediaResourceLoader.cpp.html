73void MediaResourceLoader::sendH2Ping(const URL& url, CompletionHandler<void(Expected<Seconds, ResourceError>&&)>&& completionHandler)
74{
75 if (!m_document || !m_document->frame())
76 return completionHandler(makeUnexpected(internalError(url)));
77
78 m_document->frame()->loader().client().sendH2Ping(url, WTFMove(completionHandler));
79}

82RefPtr<PlatformMediaResource> MediaResourceLoader::requestResource(ResourceRequest&& request, LoadOptions options)
83{
84 if (!m_document)
85 return nullptr;
86
87 DataBufferingPolicy bufferingPolicy = options & LoadOption::BufferData ? DataBufferingPolicy::BufferData : DataBufferingPolicy::DoNotBufferData;
88 auto cachingPolicy = options & LoadOption::DisallowCaching ? CachingPolicy::DisallowCaching : CachingPolicy::AllowCaching;
89
90 request.setRequester(ResourceRequest::Requester::Media);
91
92 if (m_element)
93 request.setInspectorInitiatorNodeIdentifier(InspectorInstrumentation::identifierForNode(*m_element));
94
95#if PLATFORM(MAC)
96 // FIXME: Workaround for <rdar://problem/26071607>. We are not able to do CORS checking on 304 responses because they are usually missing the headers we need.
97 if (!m_crossOriginMode.isNull())
98 request.makeUnconditional();
99#endif
100
101 ContentSecurityPolicyImposition contentSecurityPolicyImposition = m_element && m_element->isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
102 ResourceLoaderOptions loaderOptions {
103 SendCallbackPolicy::SendCallbacks,
104 ContentSniffingPolicy::DoNotSniffContent,
105 bufferingPolicy,
106 StoredCredentialsPolicy::Use,
107 ClientCredentialPolicy::MayAskClientForCredentials,
108 FetchOptions::Credentials::Include,
109 SecurityCheckPolicy::DoSecurityCheck,
110 FetchOptions::Mode::NoCors,
111 CertificateInfoPolicy::DoNotIncludeCertificateInfo,
112 contentSecurityPolicyImposition,
113 DefersLoadingPolicy::AllowDefersLoading,
114 cachingPolicy };
115 loaderOptions.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
116 loaderOptions.destination = m_destination;
117 auto cachedRequest = createPotentialAccessControlRequest(WTFMove(request), WTFMove(loaderOptions), *m_document, m_crossOriginMode);
118 if (m_element)
119 cachedRequest.setInitiator(*m_element);
120
121 auto resource = m_document->cachedResourceLoader().requestMedia(WTFMove(cachedRequest)).value_or(nullptr);
122 if (!resource)
123 return nullptr;
124
125 Ref<MediaResource> mediaResource = MediaResource::create(*this, resource);
126 m_resources.add(mediaResource.ptr());
127
128 return mediaResource;
129}

137void MediaResourceLoader::addResponseForTesting(const ResourceResponse& response)
138{
139 const auto maximumResponsesForTesting = 5;
140 if (!shouldRecordResponsesForTesting || m_responsesForTesting.size() > maximumResponsesForTesting)
141 return;
142 m_responsesForTesting.append(response);

173void MediaResource::responseReceived(CachedResource& resource, const ResourceResponse& response, CompletionHandler<void()>&& completionHandler)
174{
175 ASSERT_UNUSED(resource, &resource == m_resource);
176 CompletionHandlerCallingScope completionHandlerCaller(WTFMove(completionHandler));
177
178 if (!m_loader->document())
179 return;
180
181 RefPtr<MediaResource> protectedThis(this);
182 if (m_resource->resourceError().isAccessControl()) {
183 static NeverDestroyed<const String> consoleMessage("Cross-origin media resource load denied by Cross-Origin Resource Sharing policy.");
184 m_loader->document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, consoleMessage.get());
185 m_didPassAccessControlCheck = false;
186 if (m_client)
187 m_client->accessControlCheckFailed(*this, ResourceError(errorDomainWebKitInternal, 0, response.url(), consoleMessage.get()));
188 stop();
189 return;
190 }
191
192 m_didPassAccessControlCheck = m_resource->options().mode == FetchOptions::Mode::Cors;
193 if (m_client)
194 m_client->responseReceived(*this, response, [this, protectedThis = makeRef(*this), completionHandler = completionHandlerCaller.release()] (auto shouldContinue) mutable {
195 if (completionHandler)
196 completionHandler();
197 if (shouldContinue == ShouldContinuePolicyCheck::No)
198 stop();
199 });
200
201 m_loader->addResponseForTesting(response);
202}

204bool MediaResource::shouldCacheResponse(CachedResource& resource, const ResourceResponse& response)
205{
206 ASSERT_UNUSED(resource, &resource == m_resource);

243void MediaResource::notifyFinished(CachedResource& resource, const NetworkLoadMetrics& metrics)
244{
245 ASSERT_UNUSED(resource, &resource == m_resource);
246
247 RefPtr<MediaResource> protectedThis(this);
248 if (m_client) {
249 if (m_resource->loadFailedOrCanceled())
250 m_client->loadFailed(*this, m_resource->resourceError());
251 else
252 m_client->loadFinished(*this, metrics);
253 }
254 stop();
255}
