48class MediaResourceLoader final : public PlatformMediaResourceLoader, public CanMakeWeakPtr<MediaResourceLoader>, public ContextDestructionObserver {
49public:
50 WEBCORE_EXPORT MediaResourceLoader(Document&, Element&, const String& crossOriginMode, FetchOptions::Destination);
51 WEBCORE_EXPORT virtual ~MediaResourceLoader();
52
53 RefPtr<PlatformMediaResource> requestResource(ResourceRequest&&, LoadOptions) final;
54 void sendH2Ping(const URL&, CompletionHandler<void(Expected<Seconds, ResourceError>&&)>&&) final;
55 void removeResource(MediaResource&);
56
57 Document* document() { return m_document.get(); }
58 const String& crossOriginMode() const { return m_crossOriginMode; }
59
60 WEBCORE_EXPORT static void recordResponsesForTesting();
61 Vector<ResourceResponse> responsesForTesting() const { return m_responsesForTesting; }
62 void addResponseForTesting(const ResourceResponse&);
63
64private:
65 void contextDestroyed() override;
66
67 WeakPtr<Document> m_document;
68 WeakPtr<Element> m_element;
69 String m_crossOriginMode;
70 HashSet<MediaResource*> m_resources;
71 Vector<ResourceResponse> m_responsesForTesting;
72 FetchOptions::Destination m_destination;
73};

75class MediaResource : public PlatformMediaResource, CachedRawResourceClient {
76public:
77 static Ref<MediaResource> create(MediaResourceLoader&, CachedResourceHandle<CachedRawResource>);
78 virtual ~MediaResource();
79
80 // PlatformMediaResource
81 void stop() override;
82 bool didPassAccessControlCheck() const override { return m_didPassAccessControlCheck; }
83
84 // CachedRawResourceClient
85 void responseReceived(CachedResource&, const ResourceResponse&, CompletionHandler<void()>&&) override;
86 void redirectReceived(CachedResource&, ResourceRequest&&, const ResourceResponse&, CompletionHandler<void(ResourceRequest&&)>&&) override;
87 bool shouldCacheResponse(CachedResource&, const ResourceResponse&) override;
88 void dataSent(CachedResource&, unsigned long long, unsigned long long) override;
89 void dataReceived(CachedResource&, const uint8_t*, int) override;
90 void notifyFinished(CachedResource&, const NetworkLoadMetrics&) override;
91
92private:
93 MediaResource(MediaResourceLoader&, CachedResourceHandle<CachedRawResource>);
94 Ref<MediaResourceLoader> m_loader;
95 bool m_didPassAccessControlCheck { false };
96 CachedResourceHandle<CachedRawResource> m_resource;
97};
