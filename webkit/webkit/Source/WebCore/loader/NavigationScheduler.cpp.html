122 : ScheduledNavigation(delay, lockHistory, lockBackForwardList, duringLoad, isLocationChange, initiatingDocument.shouldOpenExternalURLsPolicyToPropagate())
123 , m_initiatingDocument { makeRef(initiatingDocument) }

130 void didStartTimer(Frame& frame, Timer& timer) override
131 {
132 if (m_haveToldClient)
133 return;
134 m_haveToldClient = true;
135
136 UserGestureIndicator gestureIndicator(userGestureToForward());
137 Ref<Frame> protectedFrame(frame);
138 frame.loader().clientRedirected(URL(m_url), delay(), WallTime::now() + timer.nextFireInterval(), lockBackForwardList());

150 // gesture state will sometimes be set and sometimes not within
151 // dispatchDidCancelClientRedirect().
152 frame.loader().clientRedirectCancelledOrFinished(newLoadInProgress);
153 }
154

156 SecurityOrigin* securityOrigin() const { return m_securityOrigin.get(); }
157 const URL& url() const { return m_url; }
158 String referrer() const { return m_referrer; }

160private:
161 Ref<Document> m_initiatingDocument;
162 RefPtr<SecurityOrigin> m_securityOrigin;
163 URL m_url;

168class ScheduledRedirect : public ScheduledURLNavigation {
169public:
170 ScheduledRedirect(Document& initiatingDocument, double delay, SecurityOrigin* securityOrigin, const URL& url, LockHistory lockHistory, LockBackForwardList lockBackForwardList)
171 : ScheduledURLNavigation(initiatingDocument, delay, securityOrigin, url, String(), lockHistory, lockBackForwardList, false, false)
172 {
173 clearUserGesture();
174 }
175
176 bool shouldStartTimer(Frame& frame) override
177 {
178 return frame.loader().allAncestorsAreComplete();
179 }
180
181 void fire(Frame& frame) override
182 {
183 UserGestureIndicator gestureIndicator { userGestureToForward() };
184
185 bool refresh = equalIgnoringFragmentIdentifier(frame.document()->url(), url());
186 ResourceRequest resourceRequest { url(), referrer(), refresh ? ResourceRequestCachePolicy::ReloadIgnoringCacheData : ResourceRequestCachePolicy::UseProtocolCachePolicy };
187 if (initiatedByMainFrame() == InitiatedByMainFrame::Yes)
188 resourceRequest.setRequester(ResourceRequest::Requester::Main);
189 FrameLoadRequest frameLoadRequest { initiatingDocument(), *securityOrigin(), WTFMove(resourceRequest), "_self", initiatedByMainFrame() };
190 frameLoadRequest.setLockHistory(lockHistory());
191 frameLoadRequest.setLockBackForwardList(lockBackForwardList());
192 frameLoadRequest.disableNavigationToInvalidURL();
193 frameLoadRequest.setShouldOpenExternalURLsPolicy(shouldOpenExternalURLs());
194
195 frame.loader().changeLocation(WTFMove(frameLoadRequest));
196 }

254class ScheduledHistoryNavigation : public ScheduledNavigation {
255public:
256 explicit ScheduledHistoryNavigation(int historySteps)
257 : ScheduledNavigation(0, LockHistory::No, LockBackForwardList::No, false, true)
258 , m_historySteps(historySteps)
259 {
260 }
261
262 void fire(Frame& frame) override
263 {
264 UserGestureIndicator gestureIndicator(userGestureToForward());
265
266 if (!m_historySteps) {
267 // Special case for go(0) from a frame -> reload only the frame
268 // To follow Firefox and IE's behavior, history reload can only navigate the self frame.
269 frame.loader().changeLocation(frame.document()->url(), "_self", 0, ReferrerPolicy::EmptyString, shouldOpenExternalURLs());
270 return;
271 }
272
273 // go(i!=0) from a frame navigates into the history of the frame only,
274 // in both IE and NS (but not in Mozilla). We can't easily do that.
275 frame.page()->backForward().goBackOrForward(m_historySteps);
276 }
277
278private:
279 int m_historySteps;
280};

282class ScheduledFormSubmission : public ScheduledNavigation {
283public:
284 ScheduledFormSubmission(Ref<FormSubmission>&& submission, LockBackForwardList lockBackForwardList, bool duringLoad)
285 : ScheduledNavigation(0, submission->lockHistory(), lockBackForwardList, duringLoad, true, submission->state().sourceDocument().shouldOpenExternalURLsPolicyToPropagate())
286 , m_submission(WTFMove(submission))
287 {
288 }
289
290 void fire(Frame& frame) override
291 {
292 if (m_submission->wasCancelled())
293 return;
294
295 UserGestureIndicator gestureIndicator(userGestureToForward());
296
297 // The submitForm function will find a target frame before using the redirection timer.
298 // Now that the timer has fired, we need to repeat the security check which normally is done when
299 // selecting a target, in case conditions have changed. Other code paths avoid this by targeting
300 // without leaving a time window. If we fail the check just silently drop the form submission.
301 auto& requestingDocument = m_submission->state().sourceDocument();
302 if (!requestingDocument.canNavigate(&frame))
303 return;
304 FrameLoadRequest frameLoadRequest { requestingDocument, requestingDocument.securityOrigin(), { }, { }, initiatedByMainFrame() };
305 frameLoadRequest.setLockHistory(lockHistory());
306 frameLoadRequest.setLockBackForwardList(lockBackForwardList());
307 frameLoadRequest.setReferrerPolicy(ReferrerPolicy::EmptyString);
308 frameLoadRequest.setShouldOpenExternalURLsPolicy(shouldOpenExternalURLs());
309 m_submission->populateFrameLoadRequest(frameLoadRequest);
310 frame.loader().loadFrameRequest(WTFMove(frameLoadRequest), m_submission->event(), m_submission->takeState());
311 }
312
313 void didStartTimer(Frame& frame, Timer& timer) override
314 {
315 if (m_haveToldClient)
316 return;
317 m_haveToldClient = true;
318
319 UserGestureIndicator gestureIndicator(userGestureToForward());
320 frame.loader().clientRedirected(m_submission->requestURL(), delay(), WallTime::now() + timer.nextFireInterval(), lockBackForwardList());
321 }
322
323 void didStopTimer(Frame& frame, NewLoadInProgress newLoadInProgress) override
324 {
325 if (!m_haveToldClient)
326 return;
327
328 // Do not set a UserGestureIndicator because
329 // clientRedirectCancelledOrFinished() is also called from many places
330 // inside FrameLoader, where the gesture state is not set and is in
331 // fact unavailable. We need to be consistent with them, otherwise the
332 // gesture state will sometimes be set and sometimes not within
333 // dispatchDidCancelClientRedirect().
334 frame.loader().clientRedirectCancelledOrFinished(newLoadInProgress);
335 }
336
337private:
338 Ref<FormSubmission> m_submission;
339 bool m_haveToldClient { false };

410void NavigationScheduler::scheduleRedirect(Document& initiatingDocument, double delay, const URL& url)
411{
412 if (!shouldScheduleNavigation(url))
413 return;
414 if (delay < 0 || delay > INT_MAX / 1000)
415 return;
416 if (url.isEmpty())
417 return;
418
419 // We want a new back/forward list item if the refresh timeout is > 1 second.
420 if (!m_redirect || delay <= m_redirect->delay()) {
421 auto lockBackForwardList = delay <= 1 ? LockBackForwardList::Yes : LockBackForwardList::No;
422 schedule(makeUnique<ScheduledRedirect>(initiatingDocument, delay, &m_frame.document()->securityOrigin(), url, LockHistory::Yes, lockBackForwardList));
423 }
424}

426LockBackForwardList NavigationScheduler::mustLockBackForwardList(Frame& targetFrame)
427{
428 // Non-user navigation before the page has finished firing onload should not create a new back/forward item.
429 // See https://webkit.org/b/42861 for the original motivation for this.
430 if (!UserGestureIndicator::processingUserGesture() && targetFrame.loader().documentLoader() && !targetFrame.loader().documentLoader()->wasOnloadDispatched())
431 return LockBackForwardList::Yes;
432
433 // Navigation of a subframe during loading of an ancestor frame does not create a new back/forward item.
434 // The definition of "during load" is any time before all handlers for the load event have been run.
435 // See https://bugs.webkit.org/show_bug.cgi?id=14957 for the original motivation for this.
436 for (Frame* ancestor = targetFrame.tree().parent(); ancestor; ancestor = ancestor->tree().parent()) {
437 Document* document = ancestor->document();
438 if (!ancestor->loader().isComplete() || (document && document->processingLoadEvent()))
439 return LockBackForwardList::Yes;
440 }
441 return LockBackForwardList::No;
442}

444void NavigationScheduler::scheduleLocationChange(Document& initiatingDocument, SecurityOrigin& securityOrigin, const URL& url, const String& referrer, LockHistory lockHistory, LockBackForwardList lockBackForwardList, CompletionHandler<void()>&& completionHandler)
445{
446 if (!shouldScheduleNavigation(url))
447 return completionHandler();
448
449 if (lockBackForwardList == LockBackForwardList::No)
450 lockBackForwardList = mustLockBackForwardList(m_frame);
451
452 FrameLoader& loader = m_frame.loader();
453
454 // If the URL we're going to navigate to is the same as the current one, except for the
455 // fragment part, we don't need to schedule the location change.
456 if (url.hasFragmentIdentifier() && equalIgnoringFragmentIdentifier(m_frame.document()->url(), url)) {
457 ResourceRequest resourceRequest { m_frame.document()->completeURL(url.string()), referrer, ResourceRequestCachePolicy::UseProtocolCachePolicy };
458 auto* frame = lexicalFrameFromCommonVM();
459 auto initiatedByMainFrame = frame && frame->isMainFrame() ? InitiatedByMainFrame::Yes : InitiatedByMainFrame::Unknown;
460
461 FrameLoadRequest frameLoadRequest { initiatingDocument, securityOrigin, WTFMove(resourceRequest), "_self"_s, initiatedByMainFrame };
462 frameLoadRequest.setLockHistory(lockHistory);
463 frameLoadRequest.setLockBackForwardList(lockBackForwardList);
464 frameLoadRequest.disableNavigationToInvalidURL();
465 frameLoadRequest.setShouldOpenExternalURLsPolicy(initiatingDocument.shouldOpenExternalURLsPolicyToPropagate());
466 loader.changeLocation(WTFMove(frameLoadRequest));
467 return completionHandler();
468 }
469
470 // Handle a location change of a page with no document as a special case.
471 // This may happen when a frame changes the location of another frame.
472 bool duringLoad = !loader.stateMachine().committedFirstRealDocumentLoad();
473
474 schedule(makeUnique<ScheduledLocationChange>(initiatingDocument, &securityOrigin, url, referrer, lockHistory, lockBackForwardList, duringLoad, WTFMove(completionHandler)));
475}

477void NavigationScheduler::scheduleFormSubmission(Ref<FormSubmission>&& submission)
478{
479 ASSERT(m_frame.page());
480
481 // FIXME: Do we need special handling for form submissions where the URL is the same
482 // as the current one except for the fragment part? See scheduleLocationChange above.
483
484 // Handle a location change of a page with no document as a special case.
485 // This may happen when a frame changes the location of another frame.
486 bool duringLoad = !m_frame.loader().stateMachine().committedFirstRealDocumentLoad();
487
488 // If this is a child frame and the form submission was triggered by a script, lock the back/forward list
489 // to match IE and Opera.
490 // See https://bugs.webkit.org/show_bug.cgi?id=32383 for the original motivation for this.
491 LockBackForwardList lockBackForwardList = mustLockBackForwardList(m_frame);
492 if (lockBackForwardList == LockBackForwardList::No
493 && (submission->state().formSubmissionTrigger() == SubmittedByJavaScript && m_frame.tree().parent() && !UserGestureIndicator::processingUserGesture())) {
494 lockBackForwardList = LockBackForwardList::Yes;
495 }
496
497 schedule(makeUnique<ScheduledFormSubmission>(WTFMove(submission), lockBackForwardList, duringLoad));
498}

511void NavigationScheduler::scheduleHistoryNavigation(int steps)
512{
513 LOG(History, "NavigationScheduler %p scheduleHistoryNavigation(%d) - shouldSchedule %d", this, steps, shouldScheduleNavigation());
514 if (!shouldScheduleNavigation())
515 return;
516
517 // Invalid history navigations (such as history.forward() during a new load) have the side effect of cancelling any scheduled
518 // redirects. We also avoid the possibility of cancelling the current load by avoiding the scheduled redirection altogether.
519 BackForwardController& backForward = m_frame.page()->backForward();
520 if ((steps > 0 && static_cast<unsigned>(steps) > backForward.forwardCount())
521 || (steps < 0 && static_cast<unsigned>(-steps) > backForward.backCount())) {
522 cancel();
523 return;
524 }
525
526 // In all other cases, schedule the history traversal to occur asynchronously.
527 schedule(makeUnique<ScheduledHistoryNavigation>(steps));
528}

536void NavigationScheduler::timerFired()
537{
538 if (!m_frame.page())
539 return;
540 if (m_frame.page()->defersLoading()) {
541 InspectorInstrumentation::frameClearedScheduledNavigation(m_frame);
542 return;
543 }
544
545 Ref<Frame> protect(m_frame);
546
547 std::unique_ptr<ScheduledNavigation> redirect = std::exchange(m_redirect, nullptr);
548 LOG(History, "NavigationScheduler %p timerFired - firing redirect %p", this, redirect.get());
549
550 redirect->fire(m_frame);
551 InspectorInstrumentation::frameClearedScheduledNavigation(m_frame);
552}
553
554void NavigationScheduler::schedule(std::unique_ptr<ScheduledNavigation> redirect)
555{
556 ASSERT(m_frame.page());
557
558 Ref<Frame> protect(m_frame);
559
560 // If a redirect was scheduled during a load, then stop the current load.
561 // Otherwise when the current load transitions from a provisional to a
562 // committed state, pending redirects may be cancelled.
563 if (redirect->wasDuringLoad()) {
564 if (DocumentLoader* provisionalDocumentLoader = m_frame.loader().provisionalDocumentLoader())
565 provisionalDocumentLoader->stopLoading();
566 m_frame.loader().stopLoading(UnloadEventPolicy::UnloadAndPageHide);
567 }
568
569 cancel();
570 m_redirect = WTFMove(redirect);
571
572 if (!m_frame.loader().isComplete() && m_redirect->isLocationChange())
573 m_frame.loader().completed();
574
575 if (!m_frame.page())
576 return;
577
578 startTimer();
579}
580
581void NavigationScheduler::startTimer()
582{
583 if (!m_redirect)
584 return;
585
586 ASSERT(m_frame.page());
587 if (m_timer.isActive())
588 return;
589 if (!m_redirect->shouldStartTimer(m_frame))
590 return;
591
592 Seconds delay = 1_s * m_redirect->delay();
593 m_timer.startOneShot(delay);
594 InspectorInstrumentation::frameScheduledNavigation(m_frame, delay);
595 m_redirect->didStartTimer(m_frame, m_timer); // m_redirect may be null on return (e.g. the client canceled the load)
596}
597
598void NavigationScheduler::cancel(NewLoadInProgress newLoadInProgress)
599{
600 LOG(History, "NavigationScheduler %p cancel(newLoadInProgress=%d)", this, newLoadInProgress == NewLoadInProgress::Yes);
601
602 if (m_timer.isActive())
603 InspectorInstrumentation::frameClearedScheduledNavigation(m_frame);
604 m_timer.stop();
605
606 if (auto redirect = std::exchange(m_redirect, nullptr))
607 redirect->didStopTimer(m_frame, newLoadInProgress);
608}
