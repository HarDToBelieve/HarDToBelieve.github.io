66// Returns true if we should block the load.
67static bool processContentRuleListsForLoad(const Frame& frame, ResourceRequest& request, OptionSet<ContentExtensions::ResourceType> resourceType)
68{
69 auto* documentLoader = frame.loader().documentLoader();
70 if (!documentLoader)
71 return false;
72 auto* page = frame.page();
73 if (!page)
74 return false;
75 auto results = page->userContentProvider().processContentRuleListsForLoad(*page, request.url(), resourceType, *documentLoader);
76 bool result = results.summary.blockedLoad;
77 ContentExtensions::applyResultsToRequest(WTFMove(results), page, request);
78 return result;
79}

83void PingLoader::loadImage(Frame& frame, const URL& url)
84{
85 ASSERT(frame.document());
86 auto& document = *frame.document();
87
88 if (!document.securityOrigin().canDisplay(url)) {
89 FrameLoader::reportLocalLoadFailed(&frame, url.string());
90 return;
91 }
92
93 if (!portAllowed(url)) {
94 FrameLoader::reportBlockedLoadFailed(frame, url);
95 return;
96 }
97
98 ResourceRequest request(url);
99#if ENABLE(CONTENT_EXTENSIONS)
100 if (processContentRuleListsForLoad(frame, request, ContentExtensions::ResourceType::Image))
101 return;
102#endif
103
104 document.contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(request, ContentSecurityPolicy::InsecureRequestType::Load);
105
106 request.setHTTPHeaderField(HTTPHeaderName::CacheControl, HTTPHeaderValues::maxAge0());
107
108 HTTPHeaderMap originalRequestHeader = request.httpHeaderFields();
109
110 String referrer = SecurityPolicy::generateReferrerHeader(document.referrerPolicy(), request.url(), frame.loader().outgoingReferrer());
111 if (!referrer.isEmpty())
112 request.setHTTPReferrer(referrer);
113 frame.loader().updateRequestAndAddExtraFields(request, IsMainResource::No);
114
115 startPingLoad(frame, request, WTFMove(originalRequestHeader), ShouldFollowRedirects::Yes, ContentSecurityPolicyImposition::DoPolicyCheck, ReferrerPolicy::EmptyString);
116}

118// http://www.whatwg.org/specs/web-apps/current-work/multipage/links.html#hyperlink-auditing
119void PingLoader::sendPing(Frame& frame, const URL& pingURL, const URL& destinationURL)
120{
121 ASSERT(frame.document());
122
123 if (!pingURL.protocolIsInHTTPFamily())
124 return;
125
126 ResourceRequest request(pingURL);
127 request.setRequester(ResourceRequest::Requester::Ping);
128
129#if ENABLE(CONTENT_EXTENSIONS)
130 if (processContentRuleListsForLoad(frame, request, ContentExtensions::ResourceType::Ping))
131 return;
132#endif
133
134 auto& document = *frame.document();
135 document.contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(request, ContentSecurityPolicy::InsecureRequestType::Load);
136
137 request.setHTTPMethod("POST");
138 request.setHTTPContentType("text/ping");
139 request.setHTTPBody(FormData::create("PING"));
140 request.setHTTPHeaderField(HTTPHeaderName::CacheControl, HTTPHeaderValues::maxAge0());
141
142 HTTPHeaderMap originalRequestHeader = request.httpHeaderFields();
143
144 auto& sourceOrigin = document.securityOrigin();
145 FrameLoader::addHTTPOriginIfNeeded(request, SecurityPolicy::generateOriginHeader(document.referrerPolicy(), request.url(), sourceOrigin));
146
147 frame.loader().updateRequestAndAddExtraFields(request, IsMainResource::No);
148 request.setHTTPHeaderField(HTTPHeaderName::PingTo, destinationURL.string());
149 if (!SecurityPolicy::shouldHideReferrer(pingURL, frame.loader().outgoingReferrer()))
150 request.setHTTPHeaderField(HTTPHeaderName::PingFrom, document.url().string());
151
152 startPingLoad(frame, request, WTFMove(originalRequestHeader), ShouldFollowRedirects::Yes, ContentSecurityPolicyImposition::DoPolicyCheck, ReferrerPolicy::NoReferrer);
153}

155void PingLoader::sendViolationReport(Frame& frame, const URL& reportURL, Ref<FormData>&& report, ViolationReportType reportType)
156{
157 ASSERT(frame.document());
158
159 ResourceRequest request(reportURL);
160#if ENABLE(CONTENT_EXTENSIONS)
161 if (processContentRuleListsForLoad(frame, request, ContentExtensions::ResourceType::Other))
162 return;
163#endif
164
165 auto& document = *frame.document();
166 document.contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(request, ContentSecurityPolicy::InsecureRequestType::Load);
167
168 request.setHTTPMethod("POST"_s);
169 request.setHTTPBody(WTFMove(report));
170 switch (reportType) {
171 case ViolationReportType::ContentSecurityPolicy:
172 request.setHTTPContentType("application/csp-report"_s);
173 break;
174 case ViolationReportType::XSSAuditor:
175 request.setHTTPContentType("application/json"_s);
176 break;
177 }
178
179 bool removeCookies = true;
180 if (document.securityOrigin().isSameSchemeHostPort(SecurityOrigin::create(reportURL).get()))
181 removeCookies = false;
182 if (removeCookies)
183 request.setAllowCookies(false);
184
185 HTTPHeaderMap originalRequestHeader = request.httpHeaderFields();
186
187 frame.loader().updateRequestAndAddExtraFields(request, IsMainResource::No);
188
189 String referrer = SecurityPolicy::generateReferrerHeader(document.referrerPolicy(), reportURL, frame.loader().outgoingReferrer());
190 if (!referrer.isEmpty())
191 request.setHTTPReferrer(referrer);
192
193 startPingLoad(frame, request, WTFMove(originalRequestHeader), ShouldFollowRedirects::No, ContentSecurityPolicyImposition::SkipPolicyCheck, ReferrerPolicy::EmptyString);
194}

196void PingLoader::startPingLoad(Frame& frame, ResourceRequest& request, HTTPHeaderMap&& originalRequestHeaders, ShouldFollowRedirects shouldFollowRedirects, ContentSecurityPolicyImposition policyCheck, ReferrerPolicy referrerPolicy)
197{
198 unsigned long identifier = frame.page()->progress().createUniqueIdentifier();
199 // FIXME: Why activeDocumentLoader? I would have expected documentLoader().
200 // It seems like the PingLoader should be associated with the current
201 // Document in the Frame, but the activeDocumentLoader will be associated
202 // with the provisional DocumentLoader if there is a provisional
203 // DocumentLoader.
204 bool shouldUseCredentialStorage = frame.loader().client().shouldUseCredentialStorage(frame.loader().activeDocumentLoader(), identifier);
205 ResourceLoaderOptions options;
206 options.credentials = shouldUseCredentialStorage ? FetchOptions::Credentials::Include : FetchOptions::Credentials::Omit;
207 options.redirect = shouldFollowRedirects == ShouldFollowRedirects::Yes ? FetchOptions::Redirect::Follow : FetchOptions::Redirect::Error;
208 options.keepAlive = true;
209 options.contentSecurityPolicyImposition = policyCheck;
210 options.referrerPolicy = referrerPolicy;
211 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks;
212 options.cache = FetchOptions::Cache::NoCache;
213
214 // FIXME: Deprecate the ping load code path.
215 if (platformStrategies()->loaderStrategy()->usePingLoad()) {
216 InspectorInstrumentation::willSendRequestOfType(&frame, identifier, frame.loader().activeDocumentLoader(), request, InspectorInstrumentation::LoadType::Ping);
217
218 platformStrategies()->loaderStrategy()->startPingLoad(frame, request, WTFMove(originalRequestHeaders), options, policyCheck, [protectedFrame = makeRef(frame), identifier] (const ResourceError& error, const ResourceResponse& response) {
219 if (!response.isNull())
220 InspectorInstrumentation::didReceiveResourceResponse(protectedFrame, identifier, protectedFrame->loader().activeDocumentLoader(), response, nullptr);
221 if (!error.isNull()) {
222 InspectorInstrumentation::didFailLoading(protectedFrame.ptr(), protectedFrame->loader().activeDocumentLoader(), identifier, error);
223 return;
224 }
225 InspectorInstrumentation::didFinishLoading(protectedFrame.ptr(), protectedFrame->loader().activeDocumentLoader(), identifier, { }, nullptr);
226 });
227 return;
228 }
229
230 CachedResourceRequest cachedResourceRequest { ResourceRequest { request }, options };
231 frame.document()->cachedResourceLoader().requestPingResource(WTFMove(cachedResourceRequest));
232}
