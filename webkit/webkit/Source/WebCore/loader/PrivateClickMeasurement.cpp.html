40static const char privateClickMeasurementTriggerAttributionPath[] = "/.well-known/private-click-measurement/trigger-attribution/";
41static const char privateClickMeasurementTokenSignaturePath[] = "/.well-known/private-click-measurement/sign-unlinkable-token/";
42static const char privateClickMeasurementTokenPublicKeyPath[] = "/.well-known/private-click-measurement/get-token-public-key/";
43static const char privateClickMeasurementReportAttributionPath[] = "/.well-known/private-click-measurement/report-attribution/";
44const size_t privateClickMeasurementAttributionTriggerDataPathSegmentSize = 2;
45const size_t privateClickMeasurementPriorityPathSegmentSize = 2;

47const Seconds PrivateClickMeasurement::maxAge()
48{
49 return 24_h * 7;

52bool PrivateClickMeasurement::isValid() const
53{
54 return m_attributionTriggerData
55 && m_attributionTriggerData.value().isValid()
56 && m_sourceID.isValid()
57 && !m_sourceSite.registrableDomain.isEmpty()
58 && !m_destinationSite.registrableDomain.isEmpty()
59 && (m_timesToSend.sourceEarliestTimeToSend || m_timesToSend.destinationEarliestTimeToSend);
60}

62Expected<PrivateClickMeasurement::AttributionTriggerData, String> PrivateClickMeasurement::parseAttributionRequest(const URL& redirectURL)
63{
64 auto path = StringView(redirectURL.string()).substring(redirectURL.pathStart(), redirectURL.pathEnd() - redirectURL.pathStart());
65 if (path.isEmpty() || !path.startsWith(privateClickMeasurementTriggerAttributionPath))
66 return makeUnexpected(nullString());
67
68 if (!redirectURL.protocolIs("https") || redirectURL.hasCredentials() || redirectURL.hasQuery() || redirectURL.hasFragmentIdentifier())
69 return makeUnexpected("[Private Click Measurement] Conversion was not accepted because the URL's protocol is not HTTPS or the URL contains one or more of username, password, query string, and fragment."_s);
70
71
72 auto prefixLength = sizeof(privateClickMeasurementTriggerAttributionPath) - 1;
73 if (path.length() == prefixLength + privateClickMeasurementAttributionTriggerDataPathSegmentSize) {
74 auto attributionTriggerDataUInt64 = parseInteger<uint64_t>(path.substring(prefixLength, privateClickMeasurementAttributionTriggerDataPathSegmentSize));
75 if (!attributionTriggerDataUInt64 || *attributionTriggerDataUInt64 > AttributionTriggerData::MaxEntropy)
76 return makeUnexpected(makeString("[Private Click Measurement] Conversion was not accepted because the conversion data could not be parsed or was higher than the allowed maximum of "_s, AttributionTriggerData::MaxEntropy, "."_s));
77
78 return AttributionTriggerData { static_cast<uint32_t>(*attributionTriggerDataUInt64), Priority { 0 } };
79 }
80
81 if (path.length() == prefixLength + privateClickMeasurementAttributionTriggerDataPathSegmentSize + 1 + privateClickMeasurementPriorityPathSegmentSize) {
82 auto attributionTriggerDataUInt64 = parseInteger<uint64_t>(path.substring(prefixLength, privateClickMeasurementAttributionTriggerDataPathSegmentSize));
83 if (!attributionTriggerDataUInt64 || *attributionTriggerDataUInt64 > AttributionTriggerData::MaxEntropy)
84 return makeUnexpected(makeString("[Private Click Measurement] Conversion was not accepted because the conversion data could not be parsed or was higher than the allowed maximum of "_s, AttributionTriggerData::MaxEntropy, "."_s));
85
86 auto attributionPriorityUInt64 = parseInteger<uint64_t>(path.substring(prefixLength + privateClickMeasurementAttributionTriggerDataPathSegmentSize + 1, privateClickMeasurementPriorityPathSegmentSize));
87 if (!attributionPriorityUInt64 || *attributionPriorityUInt64 > Priority::MaxEntropy)
88 return makeUnexpected(makeString("[Private Click Measurement] Conversion was not accepted because the priority could not be parsed or was higher than the allowed maximum of "_s, Priority::MaxEntropy, "."_s));
89
90 return AttributionTriggerData { static_cast<uint32_t>(*attributionTriggerDataUInt64), Priority { static_cast<uint32_t>(*attributionPriorityUInt64) } };
91 }
92
93 return makeUnexpected("[Private Click Measurement] Conversion was not accepted because the URL path contained unrecognized parts."_s);
94}

96bool PrivateClickMeasurement::hasPreviouslyBeenReported()
97{
98 return !m_timesToSend.sourceEarliestTimeToSend || !m_timesToSend.destinationEarliestTimeToSend;

101static Seconds randomlyBetweenTwentyFourAndFortyEightHours()
102{
103 return 24_h + Seconds(randomNumber() * (24_h).value());
104}
105
106PrivateClickMeasurement::AttributionSecondsUntilSendData PrivateClickMeasurement::attributeAndGetEarliestTimeToSend(AttributionTriggerData&& attributionTriggerData)
107{
108 if (!attributionTriggerData.isValid() || (m_attributionTriggerData && m_attributionTriggerData->priority >= attributionTriggerData.priority))
109 return { };
110
111 m_attributionTriggerData = WTFMove(attributionTriggerData);
112 // 24-48 hour delay before sending. This helps privacy since the conversion and the attribution
113 // requests are detached and the time of the attribution does not reveal the time of the conversion.
114 auto sourceSecondsUntilSend = randomlyBetweenTwentyFourAndFortyEightHours();
115 auto destinationSecondsUntilSend = randomlyBetweenTwentyFourAndFortyEightHours();
116 m_timesToSend = { WallTime::now() + sourceSecondsUntilSend, WallTime::now() + destinationSecondsUntilSend };
117
118 return AttributionSecondsUntilSendData { sourceSecondsUntilSend, destinationSecondsUntilSend };
119}

121bool PrivateClickMeasurement::hasHigherPriorityThan(const PrivateClickMeasurement& other) const
122{
123 if (!other.m_attributionTriggerData)
124 return true;
125
126 if (!m_attributionTriggerData)
127 return false;
128
129 return m_attributionTriggerData->priority > other.m_attributionTriggerData->priority;
130}

132static URL makeValidURL(const RegistrableDomain& domain, const char* path)
133{
134 URL validURL { { }, makeString("https://", domain.string(), path) };
135 return validURL.isValid() ? validURL : URL { };
136}

138static URL attributionReportURL(const RegistrableDomain& domain)
139{
140 return makeValidURL(domain, privateClickMeasurementReportAttributionPath);

159Ref<JSON::Object> PrivateClickMeasurement::attributionReportJSON() const
160{
161 auto reportDetails = JSON::Object::create();
162 if (!m_attributionTriggerData || !isValid())
163 return reportDetails;
164
165 reportDetails->setString("source_engagement_type"_s, "click"_s);
166 reportDetails->setString("source_site"_s, m_sourceSite.registrableDomain.string());
167 reportDetails->setInteger("source_id"_s, m_sourceID.id);
168 reportDetails->setString("attributed_on_site"_s, m_destinationSite.registrableDomain.string());
169 reportDetails->setInteger("trigger_data"_s, m_attributionTriggerData->data);
170 reportDetails->setInteger("version"_s, 2);
171
172 // This token has been kept secret this far and cannot be linked to the unlinkable token.
173 if (m_sourceSecretToken) {
174 reportDetails->setString("source_secret_token"_s, m_sourceSecretToken->tokenBase64URL);
175 reportDetails->setString("source_secret_token_signature"_s, m_sourceSecretToken->signatureBase64URL);
176 }
177
178 return reportDetails;
179}

183static constexpr uint32_t EphemeralSourceNonceRequiredNumberOfBytes = 16;
184
185bool PrivateClickMeasurement::EphemeralSourceNonce::isValid() const
186{
187 // FIXME: Investigate if we can do with a simple length check instead of decoding.
188 // https://bugs.webkit.org/show_bug.cgi?id=221945
189 auto digest = base64URLDecode(nonce);
190 if (!digest)
191 return false;
192 return digest->size() == EphemeralSourceNonceRequiredNumberOfBytes;
193}

215Ref<JSON::Object> PrivateClickMeasurement::tokenSignatureJSON() const
216{
217 auto reportDetails = JSON::Object::create();
218 if (!m_ephemeralSourceNonce || !m_ephemeralSourceNonce->isValid())
219 return reportDetails;
220
221 if (m_sourceUnlinkableToken.valueBase64URL.isEmpty())
222 return reportDetails;
223
224 reportDetails->setString("source_engagement_type"_s, "click"_s);
225 reportDetails->setString("source_nonce"_s, m_ephemeralSourceNonce->nonce);
226 // This token can not be linked to the secret token.
227 reportDetails->setString("source_unlinkable_token"_s, m_sourceUnlinkableToken.valueBase64URL);
228 reportDetails->setInteger("version"_s, 2);
229 return reportDetails;
230}
