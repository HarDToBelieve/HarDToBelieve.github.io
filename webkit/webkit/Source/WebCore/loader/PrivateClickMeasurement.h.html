188 struct AttributionSecondsUntilSendData {
189 std::optional<Seconds> sourceSeconds;
190 std::optional<Seconds> destinationSeconds;
191
192 bool hasValidSecondsUntilSendValues()
193 {
194 return sourceSeconds && destinationSeconds;
195 }
196
197 std::optional<Seconds> minSecondsUntilSend()
198 {
199 if (!sourceSeconds && !destinationSeconds)
200 return std::nullopt;
201
202 if (sourceSeconds && destinationSeconds)
203 return std::min(sourceSeconds, destinationSeconds);
204
205 return sourceSeconds ? sourceSeconds : destinationSeconds;
206 }
207
208 template<class Encoder>
209 void encode(Encoder& encoder) const
210 {
211 encoder << sourceSeconds << destinationSeconds;
212 }
213
214 template<class Decoder>
215 static std::optional<AttributionSecondsUntilSendData> decode(Decoder& decoder)
216 {
217 std::optional<std::optional<Seconds>> sourceSeconds;
218 decoder >> sourceSeconds;
219 if (!sourceSeconds)
220 return std::nullopt;
221
222 std::optional<std::optional<Seconds>> destinationSeconds;
223 decoder >> destinationSeconds;
224 if (!destinationSeconds)
225 return std::nullopt;
226
227 return AttributionSecondsUntilSendData { WTFMove(*sourceSeconds), WTFMove(*destinationSeconds) };
228 }

231 struct AttributionTimeToSendData {
232 std::optional<WallTime> sourceEarliestTimeToSend;
233 std::optional<WallTime> destinationEarliestTimeToSend;
234
235 std::optional<WallTime> earliestTimeToSend()
236 {
237 if (!sourceEarliestTimeToSend && !destinationEarliestTimeToSend)
238 return std::nullopt;
239
240 if (sourceEarliestTimeToSend && destinationEarliestTimeToSend)
241 return std::min(sourceEarliestTimeToSend, destinationEarliestTimeToSend);
242
243 return sourceEarliestTimeToSend ? sourceEarliestTimeToSend : destinationEarliestTimeToSend;
244 }
245
246 std::optional<WallTime> latestTimeToSend()
247 {
248 if (!sourceEarliestTimeToSend && !destinationEarliestTimeToSend)
249 return std::nullopt;
250
251 if (sourceEarliestTimeToSend && destinationEarliestTimeToSend)
252 return std::max(sourceEarliestTimeToSend, destinationEarliestTimeToSend);
253
254 return sourceEarliestTimeToSend ? sourceEarliestTimeToSend : destinationEarliestTimeToSend;
255 }
256
257 std::optional<AttributionReportEndpoint> attributionReportEndpoint()
258 {
259 if (sourceEarliestTimeToSend && destinationEarliestTimeToSend) {
260 if (*sourceEarliestTimeToSend < *destinationEarliestTimeToSend)
261 return AttributionReportEndpoint::Source;
262
263 return AttributionReportEndpoint::Destination;
264 }
265
266 if (sourceEarliestTimeToSend)
267 return AttributionReportEndpoint::Source;
268
269 if (destinationEarliestTimeToSend)
270 return AttributionReportEndpoint::Destination;
271
272 return std::nullopt;
273 }
274
275 template<class Encoder>
276 void encode(Encoder& encoder) const
277 {
278 encoder << sourceEarliestTimeToSend << destinationEarliestTimeToSend;
279 }
280
281 template<class Decoder>
282 static std::optional<AttributionTimeToSendData> decode(Decoder& decoder)
283 {
284 std::optional<std::optional<WallTime>> sourceEarliestTimeToSend;
285 decoder >> sourceEarliestTimeToSend;
286 if (!sourceEarliestTimeToSend)
287 return std::nullopt;
288
289 std::optional<std::optional<WallTime>> destinationEarliestTimeToSend;
290 decoder >> destinationEarliestTimeToSend;
291 if (!destinationEarliestTimeToSend)
292 return std::nullopt;
293
294 return AttributionTimeToSendData { WTFMove(*sourceEarliestTimeToSend), WTFMove(*destinationEarliestTimeToSend) };
295 }

409template<class Decoder>
410std::optional<PrivateClickMeasurement> PrivateClickMeasurement::decode(Decoder& decoder)
411{
412 std::optional<uint32_t> sourceID;
413 decoder >> sourceID;
414 if (!sourceID)
415 return std::nullopt;
416
417 std::optional<RegistrableDomain> sourceRegistrableDomain;
418 decoder >> sourceRegistrableDomain;
419 if (!sourceRegistrableDomain)
420 return std::nullopt;
421
422 std::optional<RegistrableDomain> destinationRegistrableDomain;
423 decoder >> destinationRegistrableDomain;
424 if (!destinationRegistrableDomain)
425 return std::nullopt;
426
427 std::optional<String> sourceDescription;
428 decoder >> sourceDescription;
429 if (!sourceDescription)
430 return std::nullopt;
431
432 std::optional<String> purchaser;
433 decoder >> purchaser;
434 if (!purchaser)
435 return std::nullopt;
436
437 std::optional<WallTime> timeOfAdClick;
438 decoder >> timeOfAdClick;
439 if (!timeOfAdClick)
440 return std::nullopt;
441
442 std::optional<std::optional<EphemeralSourceNonce>> ephemeralSourceNonce;
443 decoder >> ephemeralSourceNonce;
444 if (!ephemeralSourceNonce)
445 return std::nullopt;
446
447 std::optional<std::optional<AttributionTriggerData>> attributionTriggerData;
448 decoder >> attributionTriggerData;
449 if (!attributionTriggerData)
450 return std::nullopt;
451
452 std::optional<AttributionTimeToSendData> timesToSend;
453 decoder >> timesToSend;
454 if (!timesToSend)
455 return std::nullopt;
456
457 PrivateClickMeasurement attribution {
458 SourceID { WTFMove(*sourceID) },
459 SourceSite { WTFMove(*sourceRegistrableDomain) },
460 AttributionDestinationSite { WTFMove(*destinationRegistrableDomain) },
461 WTFMove(*sourceDescription),
462 WTFMove(*purchaser),
463 WTFMove(*timeOfAdClick)
464 };
465 attribution.m_ephemeralSourceNonce = WTFMove(*ephemeralSourceNonce);
466 attribution.m_attributionTriggerData = WTFMove(*attributionTriggerData);
467 attribution.m_timesToSend = WTFMove(*timesToSend);
468
469 return attribution;
470}

495template<class Decoder>
496std::optional<PrivateClickMeasurement::AttributionTriggerData> PrivateClickMeasurement::AttributionTriggerData::decode(Decoder& decoder)
497{
498 std::optional<uint32_t> data;
499 decoder >> data;
500 if (!data)
501 return std::nullopt;
502
503 std::optional<PriorityValue> priority;
504 decoder >> priority;
505 if (!priority)
506 return std::nullopt;
507
508 std::optional<WasSent> wasSent;
509 decoder >> wasSent;
510 if (!wasSent)
511 return std::nullopt;
512
513 return AttributionTriggerData { WTFMove(*data), Priority { *priority }, *wasSent };
514}
