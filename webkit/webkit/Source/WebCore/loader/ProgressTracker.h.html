26#pragma once
27
28#include "Timer.h"
29#include <wtf/Forward.h>
30#include <wtf/HashMap.h>
31#include <wtf/Noncopyable.h>
32#include <wtf/RefPtr.h>
33#include <wtf/UniqueRef.h>
34
35namespace WebCore {
36
37class Frame;
38class ResourceResponse;
39class ProgressTrackerClient;
40struct ProgressItem;
41
42class ProgressTracker {
43 WTF_MAKE_NONCOPYABLE(ProgressTracker);
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 explicit ProgressTracker(UniqueRef<ProgressTrackerClient>&&);
47 ~ProgressTracker();
48
49 ProgressTrackerClient& client() { return m_client.get(); }
50
51 static unsigned long createUniqueIdentifier();
52
53 WEBCORE_EXPORT double estimatedProgress() const;
54
55 void progressStarted(Frame&);
56 void progressCompleted(Frame&);
57
58 void incrementProgress(unsigned long identifier, const ResourceResponse&);
59 void incrementProgress(unsigned long identifier, unsigned bytesReceived);
60 void completeProgress(unsigned long identifier);
61
62 long long totalPageAndResourceBytesToLoad() const { return m_totalPageAndResourceBytesToLoad; }
63 long long totalBytesReceived() const { return m_totalBytesReceived; }
64
65 bool isMainLoadProgressing() const;
66
67private:
68 void reset();
69 void finalProgressComplete();
70
71 void progressHeartbeatTimerFired();
72
73 static unsigned long s_uniqueIdentifier;
74
75 UniqueRef<ProgressTrackerClient> m_client;
76 RefPtr<Frame> m_originatingProgressFrame;
77 HashMap<unsigned long, std::unique_ptr<ProgressItem>> m_progressItems;
78 Timer m_progressHeartbeatTimer;
79
80 long long m_totalPageAndResourceBytesToLoad { 0 };
81 long long m_totalBytesReceived { 0 };
82 long long m_totalBytesReceivedBeforePreviousHeartbeat { 0 };
83
84 double m_lastNotifiedProgressValue { 0 };
85 double m_progressValue { 0 };
86
87 MonotonicTime m_mainLoadCompletionTime;
88 MonotonicTime m_lastNotifiedProgressTime;
89
90 int m_numProgressTrackedFrames { 0 };
91 unsigned m_heartbeatsWithNoProgress { 0 };
92
93 bool m_finalProgressChangedSent { false };
94 bool m_isMainLoad { false };
95};
96
97} // namespace WebCore
