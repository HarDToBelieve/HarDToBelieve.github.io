37template<typename CharacterType> static std::optional<ResourceCryptographicDigest::Algorithm> parseHashAlgorithmAdvancingPosition(StringParsingBuffer<CharacterType>& buffer)
38{
39 // FIXME: This would be much cleaner with a lookup table of pairs of label / algorithm enum values, but I can't
40 // figure out how to keep the labels compiletime strings for skipExactlyIgnoringASCIICase.
41
42 if (skipExactlyIgnoringASCIICase(buffer, "sha256"))
43 return ResourceCryptographicDigest::Algorithm::SHA256;
44 if (skipExactlyIgnoringASCIICase(buffer, "sha384"))
45 return ResourceCryptographicDigest::Algorithm::SHA384;
46 if (skipExactlyIgnoringASCIICase(buffer, "sha512"))
47 return ResourceCryptographicDigest::Algorithm::SHA512;
48
49 return std::nullopt;
50}

52template<typename CharacterType> static std::optional<ResourceCryptographicDigest> parseCryptographicDigestImpl(StringParsingBuffer<CharacterType>& buffer)
53{
54 if (buffer.atEnd())
55 return std::nullopt;
56
57 auto algorithm = parseHashAlgorithmAdvancingPosition(buffer);
58 if (!algorithm)
59 return std::nullopt;
60
61 if (!skipExactly(buffer, '-'))
62 return std::nullopt;
63
64 auto beginHashValue = buffer.position();
65 skipWhile<isBase64OrBase64URLCharacter>(buffer);
66 skipExactly(buffer, '=');
67 skipExactly(buffer, '=');
68
69 if (buffer.position() == beginHashValue)
70 return std::nullopt;
71
72 StringView hashValue(beginHashValue, buffer.position() - beginHashValue);
73
74 if (auto digest = base64Decode(hashValue, Base64DecodeOptions::ValidatePadding))
75 return ResourceCryptographicDigest { *algorithm, WTFMove(*digest) };
76
77 if (auto digest = base64URLDecode(hashValue))
78 return ResourceCryptographicDigest { *algorithm, WTFMove(*digest) };
79
80 return std::nullopt;
81}

93template<typename CharacterType> static std::optional<EncodedResourceCryptographicDigest> parseEncodedCryptographicDigestImpl(StringParsingBuffer<CharacterType>& buffer)
94{
95 if (buffer.atEnd())
96 return std::nullopt;
97
98 auto algorithm = parseHashAlgorithmAdvancingPosition(buffer);
99 if (!algorithm)
100 return std::nullopt;
101
102 if (!skipExactly(buffer, '-'))
103 return std::nullopt;
104
105 auto beginHashValue = buffer.position();
106 skipWhile<isBase64OrBase64URLCharacter>(buffer);
107 skipExactly(buffer, '=');
108 skipExactly(buffer, '=');
109
110 if (buffer.position() == beginHashValue)
111 return std::nullopt;
112
113 return EncodedResourceCryptographicDigest { *algorithm, String(beginHashValue, buffer.position() - beginHashValue) };
114}

126std::optional<ResourceCryptographicDigest> decodeEncodedResourceCryptographicDigest(const EncodedResourceCryptographicDigest& encodedDigest)
127{
128 if (auto digest = base64Decode(encodedDigest.digest, Base64DecodeOptions::ValidatePadding))
129 return ResourceCryptographicDigest { encodedDigest.algorithm, WTFMove(*digest) };
130
131 if (auto digest = base64URLDecode(encodedDigest.digest))
132 return ResourceCryptographicDigest { encodedDigest.algorithm, WTFMove(*digest) };
133
134 return std::nullopt;
135}

137static PAL::CryptoDigest::Algorithm toCryptoDigestAlgorithm(ResourceCryptographicDigest::Algorithm algorithm)
138{
139 switch (algorithm) {
140 case ResourceCryptographicDigest::Algorithm::SHA256:
141 return PAL::CryptoDigest::Algorithm::SHA_256;
142 case ResourceCryptographicDigest::Algorithm::SHA384:
143 return PAL::CryptoDigest::Algorithm::SHA_384;
144 case ResourceCryptographicDigest::Algorithm::SHA512:
145 return PAL::CryptoDigest::Algorithm::SHA_512;
146 }
147 ASSERT_NOT_REACHED();
148 return PAL::CryptoDigest::Algorithm::SHA_512;
149}

151ResourceCryptographicDigest cryptographicDigestForBytes(ResourceCryptographicDigest::Algorithm algorithm, const void* bytes, size_t length)
152{
153 auto cryptoDigest = PAL::CryptoDigest::create(toCryptoDigestAlgorithm(algorithm));
154 cryptoDigest->addBytes(bytes, length);
155 return { algorithm, cryptoDigest->computeHash() };
156}
157
158ResourceCryptographicDigest cryptographicDigestForSharedBuffer(ResourceCryptographicDigest::Algorithm algorithm, const SharedBuffer* buffer)
159{
160 auto cryptoDigest = PAL::CryptoDigest::create(toCryptoDigestAlgorithm(algorithm));
161 if (buffer) {
162 buffer->forEachSegment([&](auto& segment) {
163 cryptoDigest->addBytes(segment.data(), segment.size());
164 });
165 }
166 return { algorithm, cryptoDigest->computeHash() };
167}
