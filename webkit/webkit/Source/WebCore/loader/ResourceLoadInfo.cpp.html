38OptionSet<ResourceType> toResourceType(CachedResource::Type type, ResourceRequestBase::Requester requester)
39{
40 switch (type) {
41 case CachedResource::Type::LinkPrefetch:
42 case CachedResource::Type::MainResource:
43 return { ResourceType::Document };
44 case CachedResource::Type::SVGDocumentResource:
45 return { ResourceType::SVGDocument };
46 case CachedResource::Type::ImageResource:
47 return { ResourceType::Image };
48 case CachedResource::Type::CSSStyleSheet:
49#if ENABLE(XSLT)
50 case CachedResource::Type::XSLStyleSheet:
51#endif
52 return { ResourceType::StyleSheet };
53
54 case CachedResource::Type::Script:
55 return { ResourceType::Script };
56
57 case CachedResource::Type::FontResource:
58 case CachedResource::Type::SVGFontResource:
59 return { ResourceType::Font };
60
61 case CachedResource::Type::MediaResource:
62 return { ResourceType::Media };
63
64 case CachedResource::Type::RawResource:
65 if (requester == ResourceRequestBase::Requester::XHR
66 || requester == ResourceRequestBase::Requester::Fetch)
67 return { ResourceType::Fetch };
68 FALLTHROUGH;
69 case CachedResource::Type::Beacon:
70 case CachedResource::Type::Ping:
71 case CachedResource::Type::Icon:
72#if ENABLE(MODEL_ELEMENT)
73 case CachedResource::Type::ModelResource:
74#endif
75#if ENABLE(APPLICATION_MANIFEST)
76 case CachedResource::Type::ApplicationManifest:
77#endif
78 return { ResourceType::Other };
79
80 case CachedResource::Type::TextTrackResource:
81 return { ResourceType::Media };
82
83 };
84 ASSERT_NOT_REACHED();
85 return { };
86}

88std::optional<OptionSet<ResourceType>> readResourceType(StringView name)
89{
90 if (name == "document")
91 return { ResourceType::Document };
92 if (name == "image")
93 return { ResourceType::Image };
94 if (name == "style-sheet")
95 return { ResourceType::StyleSheet };
96 if (name == "script")
97 return { ResourceType::Script };
98 if (name == "font")
99 return { ResourceType::Font };
100 if (name == "raw")
101 return {{ ResourceType::Fetch, ResourceType::WebSocket, ResourceType::Other, ResourceType::Ping }};
102 if (name == "websocket")
103 return { ResourceType::WebSocket };
104 if (name == "fetch")
105 return { ResourceType::Fetch };
106 if (name == "other")
107 return {{ ResourceType::Other, ResourceType::Ping }};
108 if (name == "svg-document")
109 return { ResourceType::SVGDocument };
110 if (name == "media")
111 return { ResourceType::Media };
112 if (name == "popup")
113 return { ResourceType::Popup };
114 if (name == "ping")
115 return { ResourceType::Ping };
116 return std::nullopt;
117}
118
119std::optional<OptionSet<LoadType>> readLoadType(StringView name)
120{
121 if (name == "first-party")
122 return { LoadType::FirstParty };
123 if (name == "third-party")
124 return { LoadType::ThirdParty };
125 return std::nullopt;
126}
127
128std::optional<OptionSet<LoadContext>> readLoadContext(StringView name)
129{
130 if (name == "top-frame")
131 return { LoadContext::TopFrame };
132 if (name == "child-frame")
133 return { LoadContext::ChildFrame };
134 return std::nullopt;
135}
136
137bool ResourceLoadInfo::isThirdParty() const
138{
139 return !RegistrableDomain(mainDocumentURL).matches(resourceURL);
140}
141
142ResourceFlags ResourceLoadInfo::getResourceFlags() const
143{
144 ResourceFlags flags = 0;
145 ASSERT(!type.isEmpty());
146 flags |= type.toRaw();
147 flags |= isThirdParty() ? static_cast<ResourceFlags>(LoadType::ThirdParty) : static_cast<ResourceFlags>(LoadType::FirstParty);
148 flags |= mainFrameContext ? static_cast<ResourceFlags>(LoadContext::TopFrame) : static_cast<ResourceFlags>(LoadContext::ChildFrame);
149 return flags;
150}
