64void ResourceLoadNotifier::willSendRequest(ResourceLoader* loader, ResourceRequest& clientRequest, const ResourceResponse& redirectResponse)
65{
66 m_frame.loader().applyUserAgentIfNeeded(clientRequest);
67
68 dispatchWillSendRequest(loader->documentLoader(), loader->identifier(), clientRequest, redirectResponse, loader->cachedResource());
69}
70
71void ResourceLoadNotifier::didReceiveResponse(ResourceLoader* loader, const ResourceResponse& r)
72{
73 loader->documentLoader()->addResponse(r);
74
75 if (Page* page = m_frame.page())
76 page->progress().incrementProgress(loader->identifier(), r);
77
78 dispatchDidReceiveResponse(loader->documentLoader(), loader->identifier(), r, loader);
79}

97void ResourceLoadNotifier::didFailToLoad(ResourceLoader* loader, const ResourceError& error)
98{
99 if (Page* page = m_frame.page())
100 page->progress().completeProgress(loader->identifier());
101
102 // Notifying the FrameLoaderClient may cause the frame to be destroyed.
103 Ref<Frame> protect(m_frame);
104 if (!error.isNull())
105 m_frame.loader().client().dispatchDidFailLoading(loader->documentLoader(), loader->identifier(), error);
106
107 InspectorInstrumentation::didFailLoading(&m_frame, loader->documentLoader(), loader->identifier(), error);
108}

110void ResourceLoadNotifier::assignIdentifierToInitialRequest(unsigned long identifier, DocumentLoader* loader, const ResourceRequest& request)
111{
112 bool pageIsProvisionallyLoading = false;
113 if (auto* frameLoader = loader ? loader->frameLoader() : nullptr)
114 pageIsProvisionallyLoading = frameLoader->provisionalDocumentLoader() == loader;
115
116 if (pageIsProvisionallyLoading)
117 m_initialRequestIdentifier = identifier;
118
119 m_frame.loader().client().assignIdentifierToInitialRequest(identifier, loader, request);
120}

122void ResourceLoadNotifier::dispatchWillSendRequest(DocumentLoader* loader, unsigned long identifier, ResourceRequest& request, const ResourceResponse& redirectResponse, const CachedResource* cachedResource)
123{
124#if USE(QUICK_LOOK)
125 // Always allow QuickLook-generated URLs based on the protocol scheme.
126 if (!request.isNull() && isQuickLookPreviewURL(request.url()))
127 return;
128#endif
129
130 String oldRequestURL = request.url().string();
131
132 ASSERT(m_frame.loader().documentLoader());
133 if (m_frame.loader().documentLoader())
134 m_frame.loader().documentLoader()->didTellClientAboutLoad(request.url().string());
135
136 if (auto* page = m_frame.page()) {
137 if (!page->loadsSubresources()) {
138 if (!m_frame.isMainFrame() || (m_initialRequestIdentifier && *m_initialRequestIdentifier != identifier))
139 request = { };
140 } else if (!page->allowsLoadFromURL(request.url()))
141 request = { };
142 }
143
144 // Notifying the FrameLoaderClient may cause the frame to be destroyed.
145 Ref<Frame> protect(m_frame);
146 m_frame.loader().client().dispatchWillSendRequest(loader, identifier, request, redirectResponse);
147
148 // If the URL changed, then we want to put that new URL in the "did tell client" set too.
149 if (!request.isNull() && oldRequestURL != request.url().string() && m_frame.loader().documentLoader())
150 m_frame.loader().documentLoader()->didTellClientAboutLoad(request.url().string());
151
152 InspectorInstrumentation::willSendRequest(&m_frame, identifier, loader, request, redirectResponse, cachedResource);
153}

191void ResourceLoadNotifier::sendRemainingDelegateMessages(DocumentLoader* loader, unsigned long identifier, const ResourceRequest& request, const ResourceResponse& response, const uint8_t* data, int dataLength, int encodedDataLength, const ResourceError& error)
192{
193 // If the request is null, willSendRequest cancelled the load. We should
194 // only dispatch didFailLoading in this case.
195 if (request.isNull()) {
196 ASSERT(error.isCancellation());
197 dispatchDidFailLoading(loader, identifier, error);
198 return;
199 }
200
201 if (!response.isNull())
202 dispatchDidReceiveResponse(loader, identifier, response);
203
204 if (dataLength > 0)
205 dispatchDidReceiveData(loader, identifier, data, dataLength, encodedDataLength);
206
207 if (error.isNull()) {
208 NetworkLoadMetrics emptyMetrics;
209 dispatchDidFinishLoading(loader, identifier, emptyMetrics, nullptr);
210 } else
211 dispatchDidFailLoading(loader, identifier, error);
212}
