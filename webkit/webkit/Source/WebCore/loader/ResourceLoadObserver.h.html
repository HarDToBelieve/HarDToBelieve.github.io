26#pragma once
27
28#include "ResourceLoadStatistics.h"
29#include <wtf/CompletionHandler.h>
30#include <wtf/Forward.h>
31
32namespace WebCore {
33
34class Document;
35class Frame;
36class ResourceRequest;
37class ResourceResponse;
38
39class ResourceLoadObserver {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 using TopFrameDomain = WebCore::RegistrableDomain;
43 using SubResourceDomain = WebCore::RegistrableDomain;
44
45 // https://fetch.spec.whatwg.org/#request-destination-script-like
46 enum class FetchDestinationIsScriptLike : bool { Yes, No };
47
48 WEBCORE_EXPORT static ResourceLoadObserver& shared();
49 WEBCORE_EXPORT static ResourceLoadObserver* sharedIfExists();
50 WEBCORE_EXPORT static void setShared(ResourceLoadObserver&);
51
52 virtual ~ResourceLoadObserver() { }
53
54 virtual void logSubresourceLoading(const Frame*, const ResourceRequest& /* newRequest */, const ResourceResponse& /* redirectResponse */, FetchDestinationIsScriptLike) { }
55 virtual void logWebSocketLoading(const URL& /* targetURL */, const URL& /* mainFrameURL */) { }
56 virtual void logUserInteractionWithReducedTimeResolution(const Document&) { }
57 virtual void logFontLoad(const Document&, const String& /* familyName */, bool /* loadStatus */) { }
58 virtual void logCanvasRead(const Document&) { }
59 virtual void logCanvasWriteOrMeasure(const Document&, const String& /* textWritten */) { }
60 virtual void logNavigatorAPIAccessed(const Document&, const ResourceLoadStatistics::NavigatorAPI) { }
61 virtual void logScreenAPIAccessed(const Document&, const ResourceLoadStatistics::ScreenAPI) { }
62 virtual void logSubresourceLoadingForTesting(const RegistrableDomain& /* firstPartyDomain */, const RegistrableDomain& /* thirdPartyDomain */, bool /* shouldScheduleNotification */) { }
63
64 virtual String statisticsForURL(const URL&) { return { }; }
65 virtual void updateCentralStatisticsStore(CompletionHandler<void()>&& completionHandler) { completionHandler(); }
66 virtual void clearState() { }
67
68 virtual bool hasStatistics() const { return false; }
69
70 virtual void setDomainsWithUserInteraction(HashSet<RegistrableDomain>&&) { }
71 virtual void setDomainsWithCrossPageStorageAccess(HashMap<TopFrameDomain, SubResourceDomain>&&, CompletionHandler<void()>&& completionHandler) { completionHandler(); }
72 virtual bool hasCrossPageStorageAccess(const SubResourceDomain&, const TopFrameDomain&) const { return false; }
73 virtual bool hasHadUserInteraction(const RegistrableDomain&) const { return false; }
74};
75
76} // namespace WebCore
