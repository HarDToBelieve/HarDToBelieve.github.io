53ResourceTiming::ResourceTiming(const URL& url, const String& initiator, const ResourceLoadTiming& loadTiming, const ResourceResponse& response, const SecurityOrigin&)

61ResourceTiming::ResourceTiming(const URL& url, const String& initiator, const ResourceLoadTiming& timing, const NetworkLoadMetrics& networkLoadMetrics, const ResourceResponse& response, const SecurityOrigin&)

70void ResourceTiming::initServerTiming(const ResourceResponse& response)
71{
72 if (RuntimeEnabledFeatures::sharedFeatures().serverTimingEnabled() && !m_networkLoadMetrics.failsTAOCheck)
73 m_serverTiming = ServerTimingParser::parseServerTiming(response.httpHeaderField(HTTPHeaderName::ServerTiming));
74}
75
76Vector<Ref<PerformanceServerTiming>> ResourceTiming::populateServerTiming() const
77{
78 return WTF::map(m_serverTiming, [] (auto& entry) {
79 return PerformanceServerTiming::create(String(entry.name), entry.duration, String(entry.description));
80 });
81}
82
83ResourceTiming ResourceTiming::isolatedCopy() const
84{
85 return ResourceTiming(
86 m_url.isolatedCopy(),
87 m_initiator.isolatedCopy(),
88 m_resourceLoadTiming.isolatedCopy(),
89 m_networkLoadMetrics.isolatedCopy(),
90 crossThreadCopy(m_serverTiming)
91 );
92}
