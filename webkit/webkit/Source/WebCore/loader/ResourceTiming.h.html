34
35class CachedResource;
36class PerformanceServerTiming;
37class ResourceResponse;
38class ResourceLoadTiming;
39class SecurityOrigin;
40
41class ResourceTiming {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 static ResourceTiming fromMemoryCache(const URL&, const String& initiator, const ResourceLoadTiming&, const ResourceResponse&, const SecurityOrigin&);
45 static ResourceTiming fromLoad(CachedResource&, const URL&, const String& initiator, const ResourceLoadTiming&, const NetworkLoadMetrics&, const SecurityOrigin&);
46 static ResourceTiming fromSynchronousLoad(const URL&, const String& initiator, const ResourceLoadTiming&, const NetworkLoadMetrics&, const ResourceResponse&, const SecurityOrigin&);
47
48 const URL& url() const { return m_url; }
49 const String& initiator() const { return m_initiator; }
50 const ResourceLoadTiming& resourceLoadTiming() const { return m_resourceLoadTiming; }
51 const NetworkLoadMetrics& networkLoadMetrics() const { return m_networkLoadMetrics; }
52 Vector<Ref<PerformanceServerTiming>> populateServerTiming() const;
53 ResourceTiming isolatedCopy() const;
54
55 void overrideInitiatorName(const String& name) { m_initiator = name; }
56
57private:
58 ResourceTiming(const URL&, const String& initiator, const ResourceLoadTiming&, const NetworkLoadMetrics&, const ResourceResponse&, const SecurityOrigin&);
59 ResourceTiming(const URL&, const String& initiator, const ResourceLoadTiming&, const ResourceResponse&, const SecurityOrigin&);
60 ResourceTiming(URL&& url, String&& initiator, const ResourceLoadTiming& resourceLoadTiming, NetworkLoadMetrics&& networkLoadMetrics, Vector<ServerTiming>&& serverTiming)
61 : m_url(WTFMove(url))
62 , m_initiator(WTFMove(initiator))
63 , m_resourceLoadTiming(resourceLoadTiming)
64 , m_networkLoadMetrics(WTFMove(networkLoadMetrics))
65 , m_serverTiming(WTFMove(serverTiming))
66 {
67 }
68 void initServerTiming(const ResourceResponse&);
69
70 URL m_url;
71 String m_initiator;
72 ResourceLoadTiming m_resourceLoadTiming;
73 NetworkLoadMetrics m_networkLoadMetrics;
74 Vector<ServerTiming> m_serverTiming;
75};
