41bool ResourceTimingInformation::shouldAddResourceTiming(CachedResource& resource)
42{
43 // FIXME: We can be less restrictive here.
44 // <https://github.com/w3c/resource-timing/issues/100>
45 if (!resource.resourceRequest().url().protocolIsInHTTPFamily())
46 return false;
47 if (resource.errorOccurred())
48 return false;
49 if (resource.wasCanceled())
50 return false;
51
52 if (resource.options().loadedFromOpaqueSource == LoadedFromOpaqueSource::Yes)
53 return false;
54
55 return true;
56}
57
58void ResourceTimingInformation::addResourceTiming(CachedResource& resource, Document& document, ResourceTiming&& resourceTiming)
59{
60 if (!ResourceTimingInformation::shouldAddResourceTiming(resource))
61 return;
62
63 auto iterator = m_initiatorMap.find(&resource);
64 if (iterator == m_initiatorMap.end())
65 return;
66
67 InitiatorInfo& info = iterator->value;
68 if (info.added == Added)
69 return;
70
71 Document* initiatorDocument = &document;
72 if (resource.type() == CachedResource::Type::MainResource && document.frame() && document.frame()->loader().shouldReportResourceTimingToParentFrame())
73 initiatorDocument = document.parentDocument();
74 if (!initiatorDocument)
75 return;
76
77 auto* initiatorWindow = initiatorDocument->domWindow();
78 if (!initiatorWindow)
79 return;
80
81 resourceTiming.overrideInitiatorName(info.name);
82
83 initiatorWindow->performance().addResourceTiming(WTFMove(resourceTiming));
84
85 info.added = Added;
86}
87
88void ResourceTimingInformation::removeResourceTiming(CachedResource& resource)
89{
90 m_initiatorMap.remove(&resource);
91}
92
93void ResourceTimingInformation::storeResourceTimingInitiatorInformation(const CachedResourceHandle<CachedResource>& resource, const AtomString& initiatorName, Frame* frame)
94{
95 ASSERT(resource.get());
96
97 if (resource->type() == CachedResource::Type::MainResource) {
98 // <iframe>s should report the initial navigation requested by the parent document, but not subsequent navigations.
99 ASSERT(frame);
100 if (frame->ownerElement()) {
101 InitiatorInfo info = { frame->ownerElement()->localName(), NotYetAdded };
102 m_initiatorMap.add(resource.get(), info);
103 }
104 } else {
105 InitiatorInfo info = { initiatorName, NotYetAdded };
106 m_initiatorMap.add(resource.get(), info);
107 }
108}
