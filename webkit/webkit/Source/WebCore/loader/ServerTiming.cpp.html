26#include "config.h"
27#include "ServerTiming.h"

29namespace WebCore {
30
31void ServerTiming::setParameter(const String& name, const String& value)
32{
33 if (equalIgnoringASCIICase(name, "dur")) {
34 if (!durationSet) {
35 duration = value.toDouble();
36 durationSet = true;
37 }
38 return;
39 }
40 if (equalIgnoringASCIICase(name, "desc")) {
41 if (!descriptionSet) {
42 description = value;
43 descriptionSet = true;
44 }
45 return;
46 }
47}
48
49ServerTiming ServerTiming::isolatedCopy() const
50{
51 return ServerTiming(name.isolatedCopy(), duration, description.isolatedCopy(), durationSet, descriptionSet);
52}
