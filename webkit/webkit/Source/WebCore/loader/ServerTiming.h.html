30namespace WebCore {
31
32struct ServerTiming {
33 String name;
34 double duration = 0;
35 String description;
36
37 bool durationSet = false;
38 bool descriptionSet = false;
39
40 ServerTiming(String&& name);
41 ServerTiming(String&& name, double duration, String&& description);
42 ServerTiming(String&& name, double duration, String&& description, bool durationSet, bool descriptionSet);
43
44 void setParameter(const String&, const String&);
45 ServerTiming isolatedCopy() const;
46};
47
48inline ServerTiming::ServerTiming(String&& name)
49 : name(WTFMove(name))
50{
51}
52
53inline ServerTiming::ServerTiming(String&& name, double duration, String&& description)
54 : name(WTFMove(name))
55 , duration(duration)
56 , description(WTFMove(description))
57{
58}
59
60inline ServerTiming::ServerTiming(String&& name, double duration, String&& description, bool durationSet, bool descriptionSet)
61 : name(WTFMove(name))
62 , duration(duration)
63 , description(WTFMove(description))
64 , durationSet(durationSet)
65 , descriptionSet(descriptionSet)
66{
67}

69} // namespace WebCore
