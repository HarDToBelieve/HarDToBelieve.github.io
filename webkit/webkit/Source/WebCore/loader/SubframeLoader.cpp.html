74bool FrameLoader::SubframeLoader::requestFrame(HTMLFrameOwnerElement& ownerElement, const String& urlString, const AtomString& frameName, LockHistory lockHistory, LockBackForwardList lockBackForwardList)
75{
76 // Support for <frame src="javascript:string">
77 URL scriptURL;
78 URL url;
79 if (WTF::protocolIsJavaScript(urlString)) {
80 scriptURL = completeURL(urlString); // completeURL() encodes the URL.
81 url = aboutBlankURL();
82 } else
83 url = completeURL(urlString);
84
85 if (shouldConvertInvalidURLsToBlank() && !url.isValid())
86 url = aboutBlankURL();
87
88 // If we will schedule a JavaScript URL load, we need to delay the firing of the load event at least until we've run the JavaScript in the URL.
89 CompletionHandlerCallingScope stopDelayingLoadEvent;
90 if (!scriptURL.isEmpty()) {
91 ownerElement.document().incrementLoadEventDelayCount();
92 stopDelayingLoadEvent = CompletionHandlerCallingScope([ownerDocument = makeRef(ownerElement.document())] {
93 ownerDocument->decrementLoadEventDelayCount();
94 });
95 }
96
97 Frame* frame = loadOrRedirectSubframe(ownerElement, url, frameName, lockHistory, lockBackForwardList);
98 if (!frame)
99 return false;
100
101 if (!scriptURL.isEmpty() && ownerElement.canLoadScriptURL(scriptURL)) {
102 // FIXME: Some sites rely on the javascript:'' loading synchronously, which is why we have this special case.
103 // Blink has the same workaround (https://bugs.chromium.org/p/chromium/issues/detail?id=923585).
104 if (urlString == "javascript:''" || urlString == "javascript:\"\"")
105 frame->script().executeJavaScriptURL(scriptURL);
106 else
107 frame->navigationScheduler().scheduleLocationChange(ownerElement.document(), ownerElement.document().securityOrigin(), scriptURL, m_frame.loader().outgoingReferrer(), lockHistory, lockBackForwardList, stopDelayingLoadEvent.release());
108 }
109
110 return true;
111}

113bool FrameLoader::SubframeLoader::resourceWillUsePlugin(const String& url, const String& mimeType)
114{
115 URL completedURL;
116 if (!url.isEmpty())
117 completedURL = completeURL(url);
118
119 bool useFallback;
120 return shouldUsePlugin(completedURL, mimeType, false, useFallback);
121}

123bool FrameLoader::SubframeLoader::pluginIsLoadable(const URL& url, const String& mimeType)
124{
125 auto* document = m_frame.document();
126
127 if (MIMETypeRegistry::isJavaAppletMIMEType(mimeType)) {
128 if (!m_frame.settings().isJavaEnabled())
129 return false;
130 if (document && document->securityOrigin().isLocal() && !m_frame.settings().isJavaEnabledForLocalFiles())
131 return false;
132 }
133
134 if (document) {
135 if (document->isSandboxed(SandboxPlugins))
136 return false;
137
138 if (!document->securityOrigin().canDisplay(url)) {
139 FrameLoader::reportLocalLoadFailed(&m_frame, url.string());
140 return false;
141 }
142
143 if (!portAllowed(url)) {
144 FrameLoader::reportBlockedLoadFailed(m_frame, url);
145 return false;
146 }
147
148 if (!MixedContentChecker::canRunInsecureContent(m_frame, document->securityOrigin(), url))
149 return false;
150 }
151
152 return true;
153}

155static String findPluginMIMETypeFromURL(Page& page, const URL& url)
156{
157 auto lastPathComponent = url.lastPathComponent();
158 size_t dotIndex = lastPathComponent.reverseFind('.');
159 if (dotIndex == notFound)
160 return { };
161
162 auto extensionFromURL = lastPathComponent.substring(dotIndex + 1);
163
164 for (auto& type : page.pluginData().webVisibleMimeTypes()) {
165 for (auto& extension : type.extensions) {
166 if (equalIgnoringASCIICase(extensionFromURL, extension))
167 return type.type;
168 }
169 }
170
171 return { };
172}

174bool FrameLoader::SubframeLoader::requestPlugin(HTMLPlugInImageElement& ownerElement, const URL& url, const String& explicitMIMEType, const Vector<String>& paramNames, const Vector<String>& paramValues, bool useFallback)
175{
176 String mimeType = explicitMIMEType;
177 if (mimeType.isEmpty()) {
178 if (auto page = ownerElement.document().page())
179 mimeType = findPluginMIMETypeFromURL(*page, url);
180 }
181
182 // Application plug-ins are plug-ins implemented by the user agent, for example Qt plug-ins,
183 // as opposed to third-party code such as Flash. The user agent decides whether or not they are
184 // permitted, rather than WebKit.
185 if (!(m_frame.arePluginsEnabled() || MIMETypeRegistry::isApplicationPluginMIMEType(mimeType)))
186 return false;
187
188 if (!pluginIsLoadable(url, explicitMIMEType))
189 return false;
190
191 ASSERT(ownerElement.hasTagName(objectTag) || ownerElement.hasTagName(embedTag));
192 return loadPlugin(ownerElement, url, explicitMIMEType, paramNames, paramValues, useFallback);
193}
194
195static void logPluginRequest(Page* page, const String& mimeType, const URL& url)
196{
197 if (!page)
198 return;
199
200 String newMIMEType = mimeType;
201 if (!newMIMEType) {
202 // Try to figure out the MIME type from the URL extension.
203 newMIMEType = findPluginMIMETypeFromURL(*page, url);
204 if (!newMIMEType)
205 return;
206 }
207
208 String pluginFile = page->pluginData().pluginFileForWebVisibleMimeType(newMIMEType);
209 String description = !pluginFile ? newMIMEType : pluginFile;
210 page->sawPlugin(description);
211}

213bool FrameLoader::SubframeLoader::requestObject(HTMLPlugInImageElement& ownerElement, const String& url, const AtomString& frameName, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues)
214{
215 if (url.isEmpty() && mimeType.isEmpty())
216 return false;
217
218 auto& document = ownerElement.document();
219
220 URL completedURL;
221 if (!url.isEmpty())
222 completedURL = completeURL(url);
223
224 document.contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(completedURL, ContentSecurityPolicy::InsecureRequestType::Load);
225
226 bool hasFallbackContent = is<HTMLObjectElement>(ownerElement) && downcast<HTMLObjectElement>(ownerElement).hasFallbackContent();
227
228 bool useFallback;
229 if (shouldUsePlugin(completedURL, mimeType, hasFallbackContent, useFallback)) {
230 bool success = requestPlugin(ownerElement, completedURL, mimeType, paramNames, paramValues, useFallback);
231 logPluginRequest(document.page(), mimeType, completedURL);
232 return success;
233 }
234
235 // If the plug-in element already contains a subframe, loadOrRedirectSubframe will re-use it. Otherwise,
236 // it will create a new frame and set it as the RenderWidget's Widget, causing what was previously
237 // in the widget to be torn down.
238 return loadOrRedirectSubframe(ownerElement, completedURL, frameName, LockHistory::Yes, LockBackForwardList::Yes);
239}

241Frame* FrameLoader::SubframeLoader::loadOrRedirectSubframe(HTMLFrameOwnerElement& ownerElement, const URL& requestURL, const AtomString& frameName, LockHistory lockHistory, LockBackForwardList lockBackForwardList)
242{
243 auto& initiatingDocument = ownerElement.document();
244
245 URL upgradedRequestURL = requestURL;
246 initiatingDocument.contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(upgradedRequestURL, ContentSecurityPolicy::InsecureRequestType::Load);
247
248 RefPtr<Frame> frame = makeRefPtr(ownerElement.contentFrame());
249 if (frame)
250 frame->navigationScheduler().scheduleLocationChange(initiatingDocument, initiatingDocument.securityOrigin(), upgradedRequestURL, m_frame.loader().outgoingReferrer(), lockHistory, lockBackForwardList);
251 else
252 frame = loadSubframe(ownerElement, upgradedRequestURL, frameName, m_frame.loader().outgoingReferrer());
253
254 if (!frame)
255 return nullptr;
256
257 ASSERT(ownerElement.contentFrame() == frame || !ownerElement.contentFrame());
258 return ownerElement.contentFrame();
259}

261RefPtr<Frame> FrameLoader::SubframeLoader::loadSubframe(HTMLFrameOwnerElement& ownerElement, const URL& url, const String& name, const String& referrer)
262{
263 Ref<Frame> protect(m_frame);
264 auto document = makeRef(ownerElement.document());
265
266 if (!document->securityOrigin().canDisplay(url)) {
267 FrameLoader::reportLocalLoadFailed(&m_frame, url.string());
268 return nullptr;
269 }
270
271 if (!portAllowed(url)) {
272 FrameLoader::reportBlockedLoadFailed(m_frame, url);
273 return nullptr;
274 }
275
276 if (!SubframeLoadingDisabler::canLoadFrame(ownerElement))
277 return nullptr;
278
279 if (!m_frame.page() || m_frame.page()->subframeCount() >= Page::maxNumberOfFrames)
280 return nullptr;
281
282 // Prevent initial empty document load from triggering load events.
283 document->incrementLoadEventDelayCount();
284
285 auto frame = m_frame.loader().client().createFrame(name, ownerElement);
286 if (!frame) {
287 m_frame.loader().checkCallImplicitClose();
288 document->decrementLoadEventDelayCount();
289 return nullptr;
290 }
291 ReferrerPolicy policy = ownerElement.referrerPolicy();
292 if (policy == ReferrerPolicy::EmptyString)
293 policy = document->referrerPolicy();
294 String referrerToUse = SecurityPolicy::generateReferrerHeader(policy, url, referrer);
295
296 m_frame.loader().loadURLIntoChildFrame(url, referrerToUse, frame.get());
297
298 document->decrementLoadEventDelayCount();
299
300 // The frame's onload handler may have removed it from the document.
301 if (!frame || !frame->tree().parent()) {
302 m_frame.loader().checkCallImplicitClose();
303 return nullptr;
304 }
305
306 // All new frames will have m_isComplete set to true at this point due to synchronously loading
307 // an empty document in FrameLoader::init(). But many frames will now be starting an
308 // asynchronous load of url, so we set m_isComplete to false and then check if the load is
309 // actually completed below. (Note that we set m_isComplete to false even for synchronous
310 // loads, so that checkCompleted() below won't bail early.)
311 // FIXME: Can we remove this entirely? m_isComplete normally gets set to false when a load is committed.
312 frame->loader().started();
313
314 auto* renderer = ownerElement.renderer();
315 auto* view = frame->view();
316 if (is<RenderWidget>(renderer) && view)
317 downcast<RenderWidget>(*renderer).setWidget(view);
318
319 m_frame.loader().checkCallImplicitClose();
320
321 // Some loads are performed synchronously (e.g., about:blank and loads
322 // cancelled by returning a null ResourceRequest from requestFromDelegate).
323 // In these cases, the synchronous load would have finished
324 // before we could connect the signals, so make sure to send the
325 // completed() signal for the child by hand and mark the load as being
326 // complete.
327 // FIXME: In this case the Frame will have finished loading before
328 // it's being added to the child list. It would be a good idea to
329 // create the child first, then invoke the loader separately.
330 if (frame->loader().state() == FrameState::Complete && !frame->loader().policyDocumentLoader())
331 frame->loader().checkCompleted();
332
333 if (!frame->tree().parent())
334 return nullptr;
335
336 return frame;
337}

339bool FrameLoader::SubframeLoader::shouldUsePlugin(const URL& url, const String& mimeType, bool hasFallback, bool& useFallback)
340{
341 if (m_frame.loader().client().shouldAlwaysUsePluginDocument(mimeType)) {
342 useFallback = false;
343 return true;
344 }
345
346 ObjectContentType objectType = m_frame.loader().client().objectContentType(url, mimeType);
347 // If an object's content can't be handled and it has no fallback, let
348 // it be handled as a plugin to show the broken plugin icon.
349 useFallback = objectType == ObjectContentType::None && hasFallback;
350
351 return objectType == ObjectContentType::None || objectType == ObjectContentType::PlugIn;
352}

354bool FrameLoader::SubframeLoader::loadPlugin(HTMLPlugInImageElement& pluginElement, const URL& url, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues, bool useFallback)
355{
356 if (useFallback)
357 return false;
358
359 auto& document = pluginElement.document();
360 auto* renderer = pluginElement.renderEmbeddedObject();
361
362 // FIXME: This code should not depend on renderer!
363 if (!renderer)
364 return false;
365
366 IntSize contentSize = roundedIntSize(LayoutSize(renderer->contentWidth(), renderer->contentHeight()));
367 bool loadManually = is<PluginDocument>(document) && !m_containsPlugins && downcast<PluginDocument>(document).shouldLoadPluginManually();
368
369#if PLATFORM(IOS_FAMILY)
370 // On iOS, we only tell the plugin to be in full page mode if the containing plugin document is the top level document.
371 if (document.ownerElement())
372 loadManually = false;
373#endif
374
375 auto weakRenderer = makeWeakPtr(*renderer);
376
377 auto widget = m_frame.loader().client().createPlugin(contentSize, pluginElement, url, paramNames, paramValues, mimeType, loadManually);
378
379 // The call to createPlugin *may* cause this renderer to disappear from underneath.
380 if (!weakRenderer)
381 return false;
382
383 if (!widget) {
384 if (!renderer->isPluginUnavailable())
385 renderer->setPluginUnavailabilityReason(RenderEmbeddedObject::PluginMissing);
386 return false;
387 }
388
389 renderer->setWidget(WTFMove(widget));
390 m_containsPlugins = true;
391 return true;
392}
