46// This is a slight misnomer. It handles the higher level logic of loading both subframes and plugins.
47class FrameLoader::SubframeLoader {
48 WTF_MAKE_NONCOPYABLE(SubframeLoader); WTF_MAKE_FAST_ALLOCATED;
49public:
50 explicit SubframeLoader(Frame&);
51
52 void clear();
53
54 bool requestFrame(HTMLFrameOwnerElement&, const String& url, const AtomString& frameName, LockHistory = LockHistory::Yes, LockBackForwardList = LockBackForwardList::Yes);
55 bool requestObject(HTMLPlugInImageElement&, const String& url, const AtomString& frameName,
56 const String& serviceType, const Vector<String>& paramNames, const Vector<String>& paramValues);
57
58 bool containsPlugins() const { return m_containsPlugins; }
59
60 bool resourceWillUsePlugin(const String& url, const String& mimeType);
61
62private:
63 bool requestPlugin(HTMLPlugInImageElement&, const URL&, const String& serviceType, const Vector<String>& paramNames, const Vector<String>& paramValues, bool useFallback);
64 Frame* loadOrRedirectSubframe(HTMLFrameOwnerElement&, const URL&, const AtomString& frameName, LockHistory, LockBackForwardList);
65 RefPtr<Frame> loadSubframe(HTMLFrameOwnerElement&, const URL&, const String& name, const String& referrer);
66 bool loadPlugin(HTMLPlugInImageElement&, const URL&, const String& mimeType, const Vector<String>& paramNames, const Vector<String>& paramValues, bool useFallback);
67
68 bool shouldUsePlugin(const URL&, const String& mimeType, bool hasFallback, bool& useFallback);
69 bool pluginIsLoadable(const URL&, const String& mimeType);
70
71 URL completeURL(const String&) const;
72
73 bool shouldConvertInvalidURLsToBlank() const;
74
75 bool m_containsPlugins { false };
76 Frame& m_frame;
77};
