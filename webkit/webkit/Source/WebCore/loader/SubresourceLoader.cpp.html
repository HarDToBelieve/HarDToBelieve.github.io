98SubresourceLoader::SubresourceLoader(Frame& frame, CachedResource& resource, const ResourceLoaderOptions& options)
99 : ResourceLoader(frame, options)
100 , m_resource(&resource)
101 , m_state(Uninitialized)
102 , m_requestCountTracker(std::in_place, frame.document()->cachedResourceLoader(), resource)
103{
104#ifndef NDEBUG
105 subresourceLoaderCounter.increment();
106#endif
107#if ENABLE(CONTENT_EXTENSIONS)
108 m_resourceType = ContentExtensions::toResourceType(resource.type(), resource.resourceRequest().requester());

111}

122void SubresourceLoader::create(Frame& frame, CachedResource& resource, ResourceRequest&& request, const ResourceLoaderOptions& options, CompletionHandler<void(RefPtr<SubresourceLoader>&&)>&& completionHandler)
123{
124 auto subloader(adoptRef(*new SubresourceLoader(frame, resource, options)));
125#if PLATFORM(IOS_FAMILY)
126 if (!IOSApplication::isWebProcess()) {
127 // On iOS, do not invoke synchronous resource load delegates while resource load scheduling
128 // is disabled to avoid re-entering style selection from a different thread (see <rdar://problem/9121719>).
129 // FIXME: This should be fixed for all ports in <https://bugs.webkit.org/show_bug.cgi?id=56647>.
130 subloader->m_iOSOriginalRequest = request;
131 return completionHandler(WTFMove(subloader));
132 }
133#endif
134 subloader->init(WTFMove(request), [subloader, completionHandler = WTFMove(completionHandler)] (bool initialized) mutable {
135 if (!initialized)
136 return completionHandler(nullptr);
137 completionHandler(WTFMove(subloader));
138 });
139}

144 // FIXME: this should probably be removed.

146 init(ResourceRequest(m_iOSOriginalRequest), [this, protectedThis = makeRef(*this)] (bool success) {
147 if (!success)
148 return;
149 m_iOSOriginalRequest = ResourceRequest();
150 start();
151 });
152}

163void SubresourceLoader::init(ResourceRequest&& request, CompletionHandler<void(bool)>&& completionHandler)
164{
165 ResourceLoader::init(WTFMove(request), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (bool initialized) mutable {
166 if (!initialized)
167 return completionHandler(false);
168 if (!m_documentLoader) {
169 ASSERT_NOT_REACHED();
170 SUBRESOURCELOADER_RELEASE_LOG_ERROR("init: resource load canceled because document loader is null");
171 return completionHandler(false);
172 }
173 ASSERT(!reachedTerminalState());
174 m_state = Initialized;
175 m_documentLoader->addSubresourceLoader(this);
176 m_origin = m_resource->origin();
177 completionHandler(true);
178 });
179}

186void SubresourceLoader::willSendRequestInternal(ResourceRequest&& newRequest, const ResourceResponse& redirectResponse, CompletionHandler<void(ResourceRequest&&)>&& completionHandler)
187{
188 // Store the previous URL because the call to ResourceLoader::willSendRequest will modify it.
189 URL previousURL = request().url();
190 Ref<SubresourceLoader> protectedThis(*this);
191
192 if (!newRequest.url().isValid()) {
193 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because new request is invalid");
194 cancel(cannotShowURLError());
195 return completionHandler(WTFMove(newRequest));
196 }
197
198 if (newRequest.requester() != ResourceRequestBase::Requester::Main) {
199 tracePoint(SubresourceLoadWillStart);
200 ResourceLoadObserver::shared().logSubresourceLoading(m_frame.get(), newRequest, redirectResponse,
201 (isScriptLikeDestination(options().destination) ? ResourceLoadObserver::FetchDestinationIsScriptLike::Yes : ResourceLoadObserver::FetchDestinationIsScriptLike::No));
202 }
203
204 auto continueWillSendRequest = [this, protectedThis = makeRef(*this), redirectResponse] (CompletionHandler<void(ResourceRequest&&)>&& completionHandler, ResourceRequest&& newRequest) mutable {
205 if (newRequest.isNull() || reachedTerminalState()) {
206 if (newRequest.isNull())
207 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because new request is NULL (1)");
208 else
209 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because reached terminal state");
210 return completionHandler(WTFMove(newRequest));
211 }
212
213 ResourceLoader::willSendRequestInternal(WTFMove(newRequest), redirectResponse, [this, protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler), redirectResponse] (ResourceRequest&& request) mutable {
214 if (reachedTerminalState()) {
215 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: reached terminal state; calling completion handler");
216 return completionHandler(WTFMove(request));
217 }
218
219 if (request.isNull()) {
220 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because request is NULL (2)");
221 cancel();
222 return completionHandler(WTFMove(request));
223 }
224
225 if (m_resource->type() == CachedResource::Type::MainResource && !redirectResponse.isNull())
226 m_documentLoader->willContinueMainResourceLoadAfterRedirect(request);
227
228 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load finished; calling completion handler");
229 completionHandler(WTFMove(request));
230 });
231 };
232
233 ASSERT(!newRequest.isNull());
234 if (!redirectResponse.isNull()) {
235 if (options().redirect != FetchOptions::Redirect::Follow) {
236 if (options().redirect == FetchOptions::Redirect::Error) {
237 ResourceError error { errorDomainWebKitInternal, 0, request().url(), makeString("Not allowed to follow a redirection while loading ", request().url().string()), ResourceError::Type::AccessControl };
238
239 if (m_frame && m_frame->document())
240 m_frame->document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, error.localizedDescription());
241
242 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because not allowed to follow a redirect");
243
244 cancel(error);
245 return completionHandler(WTFMove(newRequest));
246 }
247
248 ResourceResponse opaqueRedirectedResponse = redirectResponse;
249 opaqueRedirectedResponse.setType(ResourceResponse::Type::Opaqueredirect);
250 opaqueRedirectedResponse.setTainting(ResourceResponse::Tainting::Opaqueredirect);
251 m_resource->responseReceived(opaqueRedirectedResponse);
252 if (reachedTerminalState()) {
253 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: reached terminal state");
254 return completionHandler(WTFMove(newRequest));
255 }
256
257 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load completed");
258
259 NetworkLoadMetrics emptyMetrics;
260 didFinishLoading(emptyMetrics);
261 return completionHandler(WTFMove(newRequest));
262 } else if (m_redirectCount++ >= options().maxRedirectCount) {
263 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because too many redirects");
264 cancel(ResourceError(String(), 0, request().url(), "Too many redirections"_s, ResourceError::Type::General));
265 return completionHandler(WTFMove(newRequest));
266 }
267
268 // CachedResources are keyed off their original request URL.
269 // Requesting the same original URL a second time can redirect to a unique second resource.
270 // Therefore, if a redirect to a different destination URL occurs, we should no longer consider this a revalidation of the first resource.
271 // Doing so would have us reusing the resource from the first request if the second request's revalidation succeeds.
272 if (newRequest.isConditional() && m_resource->resourceToRevalidate() && newRequest.url() != m_resource->resourceToRevalidate()->response().url()) {
273 newRequest.makeUnconditional();
274 MemoryCache::singleton().revalidationFailed(*m_resource);
275 if (m_frame && m_frame->page())
276 m_frame->page()->diagnosticLoggingClient().logDiagnosticMessageWithResult(DiagnosticLoggingKeys::cachedResourceRevalidationKey(), emptyString(), DiagnosticLoggingResultFail, ShouldSample::Yes);
277 }
278
279 if (!m_documentLoader->cachedResourceLoader().updateRequestAfterRedirection(m_resource->type(), newRequest, options())) {
280 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because CachedResourceLoader::updateRequestAfterRedirection (really CachedResourceLoader::canRequestAfterRedirection) said no");
281 cancel();
282 return completionHandler(WTFMove(newRequest));
283 }
284
285 if (!portAllowed(newRequest.url())) {
286 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load (redirect) canceled because it attempted to use a blocked port");
287 if (m_frame)
288 FrameLoader::reportBlockedLoadFailed(*m_frame, newRequest.url());
289 cancel(frameLoader()->blockedError(newRequest));
290 return completionHandler(WTFMove(newRequest));
291 }
292
293 auto accessControlCheckResult = checkRedirectionCrossOriginAccessControl(request(), redirectResponse, newRequest);
294 if (!accessControlCheckResult) {
295 auto errorMessage = makeString("Cross-origin redirection to ", newRequest.url().string(), " denied by Cross-Origin Resource Sharing policy: ", accessControlCheckResult.error());
296 if (m_frame && m_frame->document())
297 m_frame->document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, errorMessage);
298 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because crosss-origin redirection denied by CORS policy");
299 cancel(ResourceError(String(), 0, request().url(), errorMessage, ResourceError::Type::AccessControl));
300 return completionHandler(WTFMove(newRequest));
301 }
302
303 if (m_resource->isImage() && m_documentLoader->cachedResourceLoader().shouldDeferImageLoad(newRequest.url())) {
304 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource load canceled because it's an image that should be defered");
305 cancel();
306 return completionHandler(WTFMove(newRequest));
307 }
308 m_resource->redirectReceived(WTFMove(newRequest), redirectResponse, [this, protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler), continueWillSendRequest = WTFMove(continueWillSendRequest)] (ResourceRequest&& request) mutable {
309 SUBRESOURCELOADER_RELEASE_LOG("willSendRequestInternal: resource done notifying clients");
310 continueWillSendRequest(WTFMove(completionHandler), WTFMove(request));
311 });
312 return;
313 }
314
315 continueWillSendRequest(WTFMove(completionHandler), WTFMove(newRequest));
316}

349static bool isLocationURLFailure(const ResourceResponse& response)
350{
351 auto locationString = response.httpHeaderField(HTTPHeaderName::Location);
352 return !locationString.isNull() && locationString.isEmpty();
353}
354
355void SubresourceLoader::didReceiveResponse(const ResourceResponse& response, CompletionHandler<void()>&& policyCompletionHandler)
356{
357 ASSERT(!response.isNull());
358 ASSERT(m_state == Initialized);
359
360 CompletionHandlerCallingScope completionHandlerCaller(WTFMove(policyCompletionHandler));
361
362 if (response.containsInvalidHTTPHeaders()) {
363 didFail(ResourceError(errorDomainWebKitInternal, 0, request().url(), "Response contained invalid HTTP headers", ResourceError::Type::General));
364 return;
365 }
366
367#if USE(QUICK_LOOK)
368 if (shouldCreatePreviewLoaderForResponse(response)) {
369 m_previewLoader = makeUnique<LegacyPreviewLoader>(*this, response);
370 if (m_previewLoader->didReceiveResponse(response))
371 return;
372 }
373#endif
374#if ENABLE(SERVICE_WORKER)
375 // Implementing step 10 of https://fetch.spec.whatwg.org/#main-fetch for service worker responses.
376 if (response.source() == ResourceResponse::Source::ServiceWorker && response.url() != request().url()) {
377 auto& loader = m_documentLoader->cachedResourceLoader();
378 if (!loader.allowedByContentSecurityPolicy(m_resource->type(), response.url(), options(), ContentSecurityPolicy::RedirectResponseReceived::Yes)) {
379 SUBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: canceling load because not allowed by content policy");
380 cancel(ResourceError({ }, 0, response.url(), { }, ResourceError::Type::General));
381 return;
382 }
383 }
384#endif
385
386 if (auto error = validateRangeRequestedFlag(request(), response)) {
387 SUBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: canceling load because receiving a range requested response for a non-range request");
388 cancel(WTFMove(*error));
389 return;
390 }
391
392 // We want redirect responses to be processed through willSendRequestInternal. Exceptions are
393 // redirection with no Location headers and fetch in manual redirect mode. Or in rare circumstances,
394 // cases of too many redirects from CFNetwork (<rdar://problem/30610988>).
395#if !PLATFORM(COCOA)
396 ASSERT(response.httpStatusCode() < 300 || response.httpStatusCode() >= 400 || response.httpStatusCode() == 304 || !response.httpHeaderField(HTTPHeaderName::Location) || response.type() == ResourceResponse::Type::Opaqueredirect);
397#endif
398
399 // Reference the object in this method since the additional processing can do
400 // anything including removing the last reference to this object; one example of this is 3266216.
401 Ref<SubresourceLoader> protectedThis(*this);
402
403 if (shouldIncludeCertificateInfo())
404 response.includeCertificateInfo();
405
406 if (!m_resource) {
407 ASSERT_NOT_REACHED();
408 RELEASE_LOG_FAULT(Loading, "Resource was unexpectedly null in SubresourceLoader::didReceiveResponse");
409 }
410
411 if (m_resource && m_resource->resourceToRevalidate()) {
412 if (response.httpStatusCode() == 304) {
413 // 304 Not modified / Use local copy
414 // Existing resource is ok, just use it updating the expiration time.
415 ResourceResponse revalidationResponse = response;
416 revalidationResponse.setSource(ResourceResponse::Source::MemoryCacheAfterValidation);
417 m_resource->setResponse(revalidationResponse);
418 MemoryCache::singleton().revalidationSucceeded(*m_resource, revalidationResponse);
419 if (m_frame && m_frame->page())
420 m_frame->page()->diagnosticLoggingClient().logDiagnosticMessageWithResult(DiagnosticLoggingKeys::cachedResourceRevalidationKey(), emptyString(), DiagnosticLoggingResultPass, ShouldSample::Yes);
421 if (!reachedTerminalState())
422 ResourceLoader::didReceiveResponse(revalidationResponse, [completionHandlerCaller = WTFMove(completionHandlerCaller)] { });
423 return;
424 }
425 // Did not get 304 response, continue as a regular resource load.
426 MemoryCache::singleton().revalidationFailed(*m_resource);
427 if (m_frame && m_frame->page())
428 m_frame->page()->diagnosticLoggingClient().logDiagnosticMessageWithResult(DiagnosticLoggingKeys::cachedResourceRevalidationKey(), emptyString(), DiagnosticLoggingResultFail, ShouldSample::Yes);
429 }
430
431 auto accessControlCheckResult = checkResponseCrossOriginAccessControl(response);
432 if (!accessControlCheckResult) {
433 if (m_frame && m_frame->document())
434 m_frame->document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, accessControlCheckResult.error());
435 SUBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: canceling load because of cross origin access control");
436 cancel(ResourceError(String(), 0, request().url(), accessControlCheckResult.error(), ResourceError::Type::AccessControl));
437 return;
438 }
439
440 if (response.isRedirection()) {
441 if (options().redirect == FetchOptions::Redirect::Follow && isLocationURLFailure(response)) {
442 // Implementing https://fetch.spec.whatwg.org/#concept-http-redirect-fetch step 3
443 cancel();
444 return;
445 }
446 if (options().redirect == FetchOptions::Redirect::Manual) {
447 ResourceResponse opaqueRedirectedResponse = response;
448 opaqueRedirectedResponse.setType(ResourceResponse::Type::Opaqueredirect);
449 opaqueRedirectedResponse.setTainting(ResourceResponse::Tainting::Opaqueredirect);
450 if (m_resource)
451 m_resource->responseReceived(opaqueRedirectedResponse);
452 if (!reachedTerminalState())
453 ResourceLoader::didReceiveResponse(opaqueRedirectedResponse, [completionHandlerCaller = WTFMove(completionHandlerCaller)] { });
454 return;
455 }
456 }
457
458 if (m_resource)
459 m_resource->responseReceived(response);
460 if (reachedTerminalState())
461 return;
462
463 bool isResponseMultipart = response.isMultipart();
464 if (options().mode != FetchOptions::Mode::Navigate)
465 LinkLoader::loadLinksFromHeader(response.httpHeaderField(HTTPHeaderName::Link), m_documentLoader->url(), *m_frame->document(), LinkLoader::MediaAttributeCheck::SkipMediaAttributeCheck);
466 ResourceLoader::didReceiveResponse(response, [this, protectedThis = WTFMove(protectedThis), isResponseMultipart, completionHandlerCaller = WTFMove(completionHandlerCaller)]() mutable {
467 if (reachedTerminalState())
468 return;
469
470 // FIXME: Main resources have a different set of rules for multipart than images do.
471 // Hopefully we can merge those 2 paths.
472 if (isResponseMultipart && m_resource && m_resource->type() != CachedResource::Type::MainResource) {
473 m_loadingMultipartContent = true;
474
475 // We don't count multiParts in a CachedResourceLoader's request count
476 m_requestCountTracker = std::nullopt;
477 if (!m_resource->isImage()) {
478 SUBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: canceling load because something about a multi-part non-image");
479 cancel();
480 return;
481 }
482 }
483
484 auto* buffer = resourceData();
485 if (m_loadingMultipartContent && buffer && buffer->size()) {
486 // The resource data will change as the next part is loaded, so we need to make a copy.
487 if (m_resource)
488 m_resource->finishLoading(buffer->copy().ptr(), { });
489 clearResourceData();
490 // Since a subresource loader does not load multipart sections progressively, data was delivered to the loader all at once.
491 // After the first multipart section is complete, signal to delegates that this load is "finished"
492 NetworkLoadMetrics emptyMetrics;
493 m_documentLoader->subresourceLoaderFinishedLoadingOnePart(this);
494 didFinishLoadingOnePart(emptyMetrics);
495 }
496
497 checkForHTTPStatusCodeError();
498
499 if (m_inAsyncResponsePolicyCheck)
500 m_policyForResponseCompletionHandler = completionHandlerCaller.release();
501 });
502}

504void SubresourceLoader::didReceiveResponsePolicy()
505{
506 ASSERT(m_inAsyncResponsePolicyCheck);
507 m_inAsyncResponsePolicyCheck = false;
508 if (auto completionHandler = WTFMove(m_policyForResponseCompletionHandler))
509 completionHandler();
510}

536void SubresourceLoader::didReceiveDataOrBuffer(const uint8_t* data, int length, RefPtr<SharedBuffer>&& buffer, long long encodedDataLength, DataPayloadType dataPayloadType)
537{
538 ASSERT(m_resource);
539
540 if (m_resource->response().httpStatusCode() >= 400 && !m_resource->shouldIgnoreHTTPStatusCodeErrors())
541 return;
542 ASSERT(!m_resource->resourceToRevalidate());
543 ASSERT(!m_resource->errorOccurred());
544 ASSERT(m_state == Initialized);
545 // Reference the object in this method since the additional processing can do
546 // anything including removing the last reference to this object; one example of this is 3266216.
547 Ref<SubresourceLoader> protectedThis(*this);
548
549 ResourceLoader::didReceiveDataOrBuffer(data, length, buffer.copyRef(), encodedDataLength, dataPayloadType);
550
551 if (!m_loadingMultipartContent) {
552 if (auto* resourceData = this->resourceData())
553 m_resource->updateBuffer(*resourceData);
554 else
555 m_resource->updateData(buffer ? buffer->data() : data, buffer ? buffer->size() : length);
556 }
557}
558
559bool SubresourceLoader::checkForHTTPStatusCodeError()
560{
561 if (m_resource->response().httpStatusCode() < 400 || m_resource->shouldIgnoreHTTPStatusCodeErrors())
562 return false;
563
564 m_state = Finishing;
565 m_resource->error(CachedResource::LoadError);
566 cancel();
567 return true;
568}

570static void logResourceLoaded(Frame* frame, CachedResource::Type type)
571{
572 if (!frame || !frame->page())
573 return;
574
575 String resourceType;
576 switch (type) {
577 case CachedResource::Type::MainResource:
578 resourceType = DiagnosticLoggingKeys::mainResourceKey();
579 break;
580 case CachedResource::Type::ImageResource:
581 resourceType = DiagnosticLoggingKeys::imageKey();
582 break;
583#if ENABLE(XSLT)
584 case CachedResource::Type::XSLStyleSheet:
585#endif
586 case CachedResource::Type::CSSStyleSheet:
587 resourceType = DiagnosticLoggingKeys::styleSheetKey();
588 break;
589 case CachedResource::Type::Script:
590 resourceType = DiagnosticLoggingKeys::scriptKey();
591 break;
592 case CachedResource::Type::FontResource:
593 case CachedResource::Type::SVGFontResource:
594 resourceType = DiagnosticLoggingKeys::fontKey();
595 break;
596 case CachedResource::Type::Beacon:
597 case CachedResource::Type::Ping:
598 case CachedResource::Type::MediaResource:
599#if ENABLE(MODEL_ELEMENT)
600 case CachedResource::Type::ModelResource:
601#endif
602 case CachedResource::Type::Icon:
603 case CachedResource::Type::RawResource:
604 resourceType = DiagnosticLoggingKeys::rawKey();
605 break;
606 case CachedResource::Type::SVGDocumentResource:

608 break;
609#if ENABLE(APPLICATION_MANIFEST)

611 resourceType = DiagnosticLoggingKeys::applicationManifestKey();
612 break;

623Expected<void, String> SubresourceLoader::checkResponseCrossOriginAccessControl(const ResourceResponse& response)
624{
625 if (!m_resource->isCrossOrigin() || options().mode != FetchOptions::Mode::Cors)
626 return { };
627
628#if ENABLE(SERVICE_WORKER)
629 if (response.source() == ResourceResponse::Source::ServiceWorker) {
630 if (response.tainting() == ResourceResponse::Tainting::Opaque) {
631 // FIXME: This should have an error message.
632 return makeUnexpected(String());
633 }
634 return { };
635 }
636#endif
637
638 ASSERT(m_origin);
639
640 return passesAccessControlCheck(response, options().credentials == FetchOptions::Credentials::Include ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse, *m_origin, &CrossOriginAccessControlCheckDisabler::singleton());
641}

643Expected<void, String> SubresourceLoader::checkRedirectionCrossOriginAccessControl(const ResourceRequest& previousRequest, const ResourceResponse& redirectResponse, ResourceRequest& newRequest)
644{
645 bool crossOriginFlag = m_resource->isCrossOrigin();
646 bool isNextRequestCrossOrigin = m_origin && !m_origin->canRequest(newRequest.url());
647
648 if (isNextRequestCrossOrigin)
649 m_resource->setCrossOrigin();
650
651 ASSERT(options().mode != FetchOptions::Mode::SameOrigin || !m_resource->isCrossOrigin());
652
653 // Implementing https://fetch.spec.whatwg.org/#concept-http-redirect-fetch step 7 & 8.
654 if (options().mode == FetchOptions::Mode::Cors) {
655 if (m_resource->isCrossOrigin()) {
656 auto locationString = redirectResponse.httpHeaderField(HTTPHeaderName::Location);
657 String errorMessage = validateCrossOriginRedirectionURL(URL(redirectResponse.url(), locationString));
658 if (!errorMessage.isNull())
659 return makeUnexpected(WTFMove(errorMessage));
660 }
661
662 ASSERT(m_origin);
663 if (crossOriginFlag) {
664 auto accessControlCheckResult = passesAccessControlCheck(redirectResponse, options().storedCredentialsPolicy, *m_origin, &CrossOriginAccessControlCheckDisabler::singleton());
665 if (!accessControlCheckResult)
666 return accessControlCheckResult;
667 }
668 }
669
670 bool redirectingToNewOrigin = false;
671 if (m_resource->isCrossOrigin()) {
672 if (!crossOriginFlag && isNextRequestCrossOrigin)
673 redirectingToNewOrigin = true;
674 else
675 redirectingToNewOrigin = !protocolHostAndPortAreEqual(previousRequest.url(), newRequest.url());
676 }
677
678 // Implementing https://fetch.spec.whatwg.org/#concept-http-redirect-fetch step 10.
679 if (crossOriginFlag && redirectingToNewOrigin)
680 m_origin = SecurityOrigin::createUnique();
681
682 newRequest.redirectAsGETIfNeeded(previousRequest, redirectResponse);
683
684 // Implementing https://fetch.spec.whatwg.org/#concept-http-redirect-fetch step 14.
685 updateReferrerPolicy(redirectResponse.httpHeaderField(HTTPHeaderName::ReferrerPolicy));
686
687 if (options().mode == FetchOptions::Mode::Cors && redirectingToNewOrigin) {
688 cleanHTTPRequestHeadersForAccessControl(newRequest, options().httpHeadersToKeep);
689 updateRequestForAccessControl(newRequest, *m_origin, options().storedCredentialsPolicy);
690 }
691
692 updateRequestReferrer(newRequest, referrerPolicy(), previousRequest.httpReferrer());
693
694 FrameLoader::addHTTPOriginIfNeeded(newRequest, m_origin ? m_origin->toString() : String());
695
696 return { };
697}

707void SubresourceLoader::didFinishLoading(const NetworkLoadMetrics& networkLoadMetrics)
708{
709 SUBRESOURCELOADER_RELEASE_LOG("didFinishLoading:");
710
711#if USE(QUICK_LOOK)
712 if (auto previewLoader = m_previewLoader.get()) {
713 if (previewLoader->didFinishLoading())
714 return;
715 }
716#endif
717
718 if (m_state != Initialized)
719 return;
720 ASSERT(!reachedTerminalState());
721 ASSERT(!m_resource->resourceToRevalidate());
722 // FIXME (129394): We should cancel the load when a decode error occurs instead of continuing the load to completion.
723 ASSERT(!m_resource->errorOccurred() || m_resource->status() == CachedResource::DecodeError || !m_resource->isLoading());
724 LOG(ResourceLoading, "Received '%s'.", m_resource->url().string().latin1().data());
725 logResourceLoaded(m_frame.get(), m_resource->type());
726
727 Ref<SubresourceLoader> protectedThis(*this);
728 CachedResourceHandle<CachedResource> protectResource(m_resource);
729
730 m_loadTiming.markEndTime();
731
732 if (networkLoadMetrics.isComplete())
733 reportResourceTiming(networkLoadMetrics);
734 else {
735 // This is the legacy path for platforms (and ResourceHandle paths) that do not provide
736 // complete load metrics in didFinishLoad. In those cases, fall back to the possibility
737 // that they populated partial load timing information on the ResourceResponse.
738 const auto* timing = m_resource->response().deprecatedNetworkLoadMetricsOrNull();
739 reportResourceTiming(timing ? *timing : NetworkLoadMetrics { });
740 }
741
742 if (m_resource->type() != CachedResource::Type::MainResource)
743 tracePoint(SubresourceLoadDidEnd);
744
745 m_state = Finishing;
746 m_resource->finishLoading(resourceData(), networkLoadMetrics);
747
748 if (wasCancelled()) {
749 SUBRESOURCELOADER_RELEASE_LOG("didFinishLoading: was canceled");
750 return;
751 }
752
753 m_resource->finish();
754 ASSERT(!reachedTerminalState());
755 didFinishLoadingOnePart(networkLoadMetrics);
756 notifyDone(LoadCompletionType::Finish);
757
758 if (reachedTerminalState()) {
759 SUBRESOURCELOADER_RELEASE_LOG("didFinishLoading: reached terminal state");
760 return;
761 }
762 SUBRESOURCELOADER_RELEASE_LOG("didFinishLoading: Did not reach terminal state");
763 releaseResources();
764}

766void SubresourceLoader::didFail(const ResourceError& error)
767{
768 SUBRESOURCELOADER_RELEASE_LOG("didFail: (type=%d, code=%d)", static_cast<int>(error.type()), error.errorCode());
769
770#if USE(QUICK_LOOK)
771 if (auto previewLoader = m_previewLoader.get())
772 previewLoader->didFail();
773#endif
774
775 if (m_state != Initialized)
776 return;
777
778 ASSERT(!reachedTerminalState());
779 LOG(ResourceLoading, "Failed to load '%s'.\n", m_resource->url().string().latin1().data());
780
781 if (m_frame->document() && error.isAccessControl() && m_resource->type() != CachedResource::Type::Ping)
782 m_frame->document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, error.localizedDescription());
783
784 Ref<SubresourceLoader> protectedThis(*this);
785 CachedResourceHandle<CachedResource> protectResource(m_resource);
786 m_state = Finishing;
787
788 if (m_resource->type() != CachedResource::Type::MainResource)
789 tracePoint(SubresourceLoadDidEnd);
790
791 if (m_resource->resourceToRevalidate())
792 MemoryCache::singleton().revalidationFailed(*m_resource);
793 m_resource->setResourceError(error);
794 if (!m_resource->isPreloaded())
795 MemoryCache::singleton().remove(*m_resource);
796 m_resource->error(CachedResource::LoadError);
797 cleanupForError(error);
798 notifyDone(LoadCompletionType::Cancel);
799 if (reachedTerminalState())
800 return;
801 releaseResources();
802}

806 SUBRESOURCELOADER_RELEASE_LOG("willCancel: (type=%d, code=%d)", static_cast<int>(error.type()), error.errorCode());

808#if PLATFORM(IOS_FAMILY)

810 // CachedResourceLoader stores resources in the memory cache immediately,

848void SubresourceLoader::notifyDone(LoadCompletionType type)
849{
850 if (reachedTerminalState())
851 return;
852
853 m_requestCountTracker = std::nullopt;
854 bool shouldPerformPostLoadActions = true;
855#if PLATFORM(IOS_FAMILY)
856 if (m_state == CancelledWhileInitializing)
857 shouldPerformPostLoadActions = false;
858#endif
859 if (m_documentLoader)

885void SubresourceLoader::reportResourceTiming(const NetworkLoadMetrics& networkLoadMetrics)
886{
887 if (!ResourceTimingInformation::shouldAddResourceTiming(*m_resource))
888 return;
889
890 Document* document = m_documentLoader->cachedResourceLoader().document();
891 if (!document)
892 return;
893
894 SecurityOrigin& origin = m_origin ? *m_origin : document->securityOrigin();
895 auto resourceTiming = ResourceTiming::fromLoad(*m_resource, m_resource->resourceRequest().url(), m_resource->initiatorName(), m_loadTiming, networkLoadMetrics, origin);
896
897 // Worker resources loaded here are all CachedRawResources loaded through WorkerThreadableLoader.
898 // Pass the ResourceTiming information on so that WorkerThreadableLoader may add them to the
899 // Worker's Performance object.
900 if (options().initiatorContext == InitiatorContext::Worker) {
901 ASSERT(m_origin);
902 ASSERT(is<CachedRawResource>(m_resource));
903 downcast<CachedRawResource>(*m_resource).finishedTimingForWorkerLoad(WTFMove(resourceTiming));
904 return;
905 }
906
907 ASSERT(options().initiatorContext == InitiatorContext::Document);
908 m_documentLoader->cachedResourceLoader().resourceTimingInformation().addResourceTiming(*m_resource, *document, WTFMove(resourceTiming));
909}
