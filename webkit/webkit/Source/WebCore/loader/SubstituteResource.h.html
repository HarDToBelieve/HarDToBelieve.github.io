26#pragma once
27
28#include "ResourceLoader.h"
29#include "ResourceResponse.h"
30#include "SharedBuffer.h"
31
32namespace WebCore {
33
34class SubstituteResource : public RefCounted<SubstituteResource> {
35public:
36 virtual ~SubstituteResource() = default;
37
38 const URL& url() const { return m_url; }
39 const ResourceResponse& response() const { return m_response; }
40 SharedBuffer& data() const { return static_reference_cast<SharedBuffer>(m_data); }
41
42 virtual void deliver(ResourceLoader& loader) { loader.deliverResponseAndData(m_response, m_data->copy()); }
43
44protected:
45 SubstituteResource(URL&& url, ResourceResponse&& response, Ref<SharedBuffer>&& data)
46 : m_url(WTFMove(url))
47 , m_response(WTFMove(response))
48 , m_data(WTFMove(data))
49 {
50 }
51
52private:
53 URL m_url;
54 ResourceResponse m_response;
55 Ref<SharedBuffer> m_data;
56};
57
58} // namespace WebCore
