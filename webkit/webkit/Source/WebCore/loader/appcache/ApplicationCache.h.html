26#pragma once
27
28#include <wtf/HashMap.h>
29#include <wtf/WeakPtr.h>
30#include <wtf/text/StringHash.h>
31
32namespace WebCore {
33
34class ApplicationCacheGroup;
35class ApplicationCacheResource;
36class ResourceRequest;
37
38using FallbackURLVector = Vector<std::pair<URL, URL>>;
39
40class ApplicationCache : public RefCounted<ApplicationCache> {
41public:
42 static Ref<ApplicationCache> create() { return adoptRef(*new ApplicationCache); }
43
44 ~ApplicationCache();
45
46 void addResource(Ref<ApplicationCacheResource>&&);
47
48 void setManifestResource(Ref<ApplicationCacheResource>&&);
49 ApplicationCacheResource* manifestResource() const;
50
51 void setGroup(ApplicationCacheGroup*);
52 ApplicationCacheGroup* group() const;
53
54 bool isComplete();
55
56 ApplicationCacheResource* resourceForRequest(const ResourceRequest&);
57 ApplicationCacheResource* resourceForURL(const String& url);
58
59 void setAllowsAllNetworkRequests(bool value) { m_allowAllNetworkRequests = value; }
60 bool allowsAllNetworkRequests() const { return m_allowAllNetworkRequests; }
61 void setOnlineAllowlist(const Vector<URL>&);
62 const Vector<URL>& onlineAllowlist() const { return m_onlineAllowlist; }
63 bool isURLInOnlineAllowlist(const URL&); // There is an entry in online allowlist that has the same origin as the resource's URL and that is a prefix match for the resource's URL.
64
65 void setFallbackURLs(const FallbackURLVector&);
66 const FallbackURLVector& fallbackURLs() const { return m_fallbackURLs; }
67 bool urlMatchesFallbackNamespace(const URL&, URL* fallbackURL = nullptr);
68
69#ifndef NDEBUG
70 void dump();
71#endif
72
73 using ResourceMap = HashMap<String, RefPtr<ApplicationCacheResource>>;
74 const ResourceMap& resources() const { return m_resources; }
75
76 void setStorageID(unsigned storageID) { m_storageID = storageID; }
77 unsigned storageID() const { return m_storageID; }
78 void clearStorageID();
79
80 static bool requestIsHTTPOrHTTPSGet(const ResourceRequest&);
81
82 int64_t estimatedSizeInStorage() const { return m_estimatedSizeInStorage; }
83
84private:
85 ApplicationCache();
86
87 WeakPtr<ApplicationCacheGroup> m_group;
88 ResourceMap m_resources;
89 WeakPtr<ApplicationCacheResource> m_manifest;
90
91 bool m_allowAllNetworkRequests { false };
92 Vector<URL> m_onlineAllowlist;
93 FallbackURLVector m_fallbackURLs;
94
95 // The total size of the resources belonging to this Application Cache instance.
96 // This is an estimation of the size this Application Cache occupies in the database file.
97 int64_t m_estimatedSizeInStorage { 0 };
98
99 unsigned m_storageID { 0 };
100};
101
102} // namespace WebCore
