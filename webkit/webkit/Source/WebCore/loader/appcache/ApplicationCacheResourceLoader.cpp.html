33RefPtr<ApplicationCacheResourceLoader> ApplicationCacheResourceLoader::create(unsigned type, CachedResourceLoader& loader, ResourceRequest&& request, CompletionHandler<void(ResourceOrError&&)>&& callback)
34{
35 ResourceLoaderOptions options;
36 options.storedCredentialsPolicy = StoredCredentialsPolicy::Use;
37 options.credentials = FetchOptions::Credentials::Include;
38 options.applicationCacheMode = ApplicationCacheMode::Bypass;
39 options.certificateInfoPolicy = CertificateInfoPolicy::IncludeCertificateInfo;
40 CachedResourceRequest cachedResourceRequest { WTFMove(request), options };
41 auto resource = loader.requestRawResource(WTFMove(cachedResourceRequest));
42 if (!resource.has_value()) {
43 callback(makeUnexpected(Error::CannotCreateResource));
44 return nullptr;
45 }
46 return adoptRef(*new ApplicationCacheResourceLoader { type, WTFMove(resource.value()), WTFMove(callback) });
47}

49ApplicationCacheResourceLoader::ApplicationCacheResourceLoader(unsigned type, CachedResourceHandle<CachedRawResource>&& resource, CompletionHandler<void(ResourceOrError&&)>&& callback)
50 : m_type(type)
51 , m_resource(WTFMove(resource))
52 , m_callback(WTFMove(callback))
53{
54 m_resource->addClient(*this);
55}
56
57ApplicationCacheResourceLoader::~ApplicationCacheResourceLoader()
58{
59 if (auto callback = WTFMove(m_callback))
60 callback(makeUnexpected(Error::Abort));
61
62 if (m_resource)
63 m_resource->removeClient(*this);
64}
65
66void ApplicationCacheResourceLoader::cancel(Error error)
67{
68 auto protectedThis = makeRef(*this);
69
70 if (auto callback = WTFMove(m_callback))
71 callback(makeUnexpected(error));
72
73 if (m_resource) {
74 m_resource->removeClient(*this);
75 m_resource = nullptr;
76 }
77}

79void ApplicationCacheResourceLoader::responseReceived(CachedResource& resource, const ResourceResponse& response, CompletionHandler<void()>&& completionHandler)
80{
81 ASSERT_UNUSED(resource, &resource == m_resource);
82 CompletionHandlerCallingScope completionHandlerCaller(WTFMove(completionHandler));
83
84 if (response.httpStatusCode() == 404 || response.httpStatusCode() == 410) {
85 cancel(Error::NotFound);
86 return;
87 }
88
89 if (response.httpStatusCode() == 304) {
90 notifyFinished(*m_resource, { });
91 return;
92 }
93
94 if (response.httpStatusCode() / 100 != 2) {
95 cancel(Error::NotOK);
96 return;
97 }
98
99 m_applicationCacheResource = ApplicationCacheResource::create(m_resource->url(), response, m_type);
100}
101
102void ApplicationCacheResourceLoader::dataReceived(CachedResource&, const uint8_t* data, int length)
103{
104 m_applicationCacheResource->data().append(data, length);
105}
106
107void ApplicationCacheResourceLoader::redirectReceived(CachedResource&, ResourceRequest&& newRequest, const ResourceResponse&, CompletionHandler<void(ResourceRequest&&)>&& callback)
108{
109 m_hasRedirection = true;
110 bool isRedirectionDisallowed = (m_type & ApplicationCacheResource::Type::Manifest) || (m_type & ApplicationCacheResource::Explicit) || (m_type & ApplicationCacheResource::Fallback);
111
112 if (isRedirectionDisallowed) {
113 cancel(Error::RedirectForbidden);
114 callback({ });
115 return;
116 }
117 callback(WTFMove(newRequest));
118}
119
120void ApplicationCacheResourceLoader::notifyFinished(CachedResource& resource, const NetworkLoadMetrics&)
121{
122 auto protectedThis = makeRef(*this);
123
124 ASSERT_UNUSED(resource, &resource == m_resource);
125
126 if (m_resource->errorOccurred()) {
127 cancel(Error::NetworkError);
128 return;
129 }
130 if (auto callback = WTFMove(m_callback))
131 callback(WTFMove(m_applicationCacheResource));
132
133 CachedResourceHandle<CachedRawResource> resourceHandle;
134 std::swap(resourceHandle, m_resource);
135 if (resourceHandle)
136 resourceHandle->removeClient(*this);
137}
