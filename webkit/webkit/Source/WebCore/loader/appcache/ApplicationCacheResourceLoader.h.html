26#pragma once
27
28#include "ApplicationCacheResource.h"
29#include "CachedRawResource.h"
30#include "CachedRawResourceClient.h"
31#include "CachedResourceHandle.h"
32#include <wtf/CompletionHandler.h>
33#include <wtf/RefCounted.h>
34
35namespace WebCore {
36
37class CachedResourceLoader;
38class ResourceRequest;
39
40class ApplicationCacheResourceLoader final : public RefCounted<ApplicationCacheResourceLoader>, private CachedRawResourceClient {
41public:
42 enum class Error { Abort, NetworkError, CannotCreateResource, NotFound, NotOK, RedirectForbidden };
43 using ResourceOrError = Expected<RefPtr<ApplicationCacheResource>, Error>;
44
45 static RefPtr<ApplicationCacheResourceLoader> create(unsigned, CachedResourceLoader&, ResourceRequest&&, CompletionHandler<void(ResourceOrError&&)>&&);
46 ~ApplicationCacheResourceLoader();
47
48 void cancel(Error = Error::Abort);
49
50 const CachedResource* resource() const { return m_resource.get(); }
51 bool hasRedirection() const { return m_hasRedirection; }
52 unsigned type() const { return m_type; }
53
54private:
55 explicit ApplicationCacheResourceLoader(unsigned, CachedResourceHandle<CachedRawResource>&&, CompletionHandler<void(ResourceOrError&&)>&&);
56
57 // CachedRawResourceClient
58 void responseReceived(CachedResource&, const ResourceResponse&, CompletionHandler<void()>&&) final;
59 void dataReceived(CachedResource&, const uint8_t* data, int dataLength) final;
60 void redirectReceived(CachedResource&, ResourceRequest&&, const ResourceResponse&, CompletionHandler<void(ResourceRequest&&)>&&) final;
61 void notifyFinished(CachedResource&, const NetworkLoadMetrics&) final;
62
63 unsigned m_type;
64 CachedResourceHandle<CachedRawResource> m_resource;
65 RefPtr<ApplicationCacheResource> m_applicationCacheResource;
66 CompletionHandler<void(ResourceOrError&&)> m_callback;
67 bool m_hasRedirection { false };
68};

70} // namespace WebCore
