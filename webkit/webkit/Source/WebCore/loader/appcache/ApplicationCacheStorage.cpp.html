48template <class T>
49class StorageIDJournal {
50public:
51 ~StorageIDJournal()
52 {
53 for (auto& record : m_records)
54 record.restore();
55 }
56
57 void add(T* resource, unsigned storageID)
58 {
59 m_records.append(Record(resource, storageID));
60 }
61
62 void commit()
63 {
64 m_records.clear();
65 }
66
67private:
68 class Record {
69 public:
70 Record() : m_resource(nullptr), m_storageID(0) { }
71 Record(T* resource, unsigned storageID) : m_resource(resource), m_storageID(storageID) { }
72
73 void restore()
74 {
75 m_resource->setStorageID(m_storageID);
76 }
77
78 private:
79 T* m_resource;
80 unsigned m_storageID;
81 };
82
83 Vector<Record> m_records;
84};

86static unsigned urlHostHash(const URL& url)
87{
88 StringView host = url.host();
89 if (host.is8Bit())
90 return AlreadyHashed::avoidDeletedValue(StringHasher::computeHashAndMaskTop8Bits(host.characters8(), host.length()));
91 return AlreadyHashed::avoidDeletedValue(StringHasher::computeHashAndMaskTop8Bits(host.characters16(), host.length()));
92}

94ApplicationCacheGroup* ApplicationCacheStorage::loadCacheGroup(const URL& manifestURL)
95{
96 SQLiteTransactionInProgressAutoCounter transactionCounter;
97
98 openDatabase(false);
99 if (!m_database.isOpen())
100 return nullptr;
101
102 auto statement = m_database.prepareStatement("SELECT id, manifestURL, newestCache FROM CacheGroups WHERE newestCache IS NOT NULL AND manifestURL=?"_s);
103 if (!statement)
104 return nullptr;
105
106 statement->bindText(1, manifestURL.string());
107
108 int result = statement->step();
109 if (result == SQLITE_DONE)
110 return nullptr;
111
112 if (result != SQLITE_ROW) {
113 LOG_ERROR("Could not load cache group, error \"%s\"", m_database.lastErrorMsg());
114 return nullptr;
115 }
116
117 unsigned newestCacheStorageID = static_cast<unsigned>(statement->columnInt64(2));
118
119 auto cache = loadCache(newestCacheStorageID);
120 if (!cache)
121 return nullptr;
122
123 auto& group = *new ApplicationCacheGroup(*this, manifestURL);
124 group.setStorageID(static_cast<unsigned>(statement->columnInt64(0)));
125 group.setNewestCache(cache.releaseNonNull());
126 return &group;
127}

129ApplicationCacheGroup* ApplicationCacheStorage::findOrCreateCacheGroup(const URL& manifestURL)
130{
131 ASSERT(!manifestURL.hasFragmentIdentifier());
132
133 auto result = m_cachesInMemory.add(manifestURL.string(), nullptr);
134 if (!result.isNewEntry) {
135 ASSERT(result.iterator->value);
136 return result.iterator->value;
137 }
138
139 // Look up the group in the database
140 auto* group = loadCacheGroup(manifestURL);
141
142 // If the group was not found we need to create it
143 if (!group) {
144 group = new ApplicationCacheGroup(*this, manifestURL);
145 m_cacheHostSet.add(urlHostHash(manifestURL));
146 }
147
148 result.iterator->value = group;
149 return group;
150}

157void ApplicationCacheStorage::loadManifestHostHashes()
158{
159 static bool hasLoadedHashes = false;
160
161 if (hasLoadedHashes)
162 return;
163
164 // We set this flag to true before the database has been opened
165 // to avoid trying to open the database over and over if it doesn't exist.
166 hasLoadedHashes = true;
167
168 SQLiteTransactionInProgressAutoCounter transactionCounter;
169
170 openDatabase(false);
171 if (!m_database.isOpen())
172 return;
173
174 // Fetch the host hashes.
175 auto statement = m_database.prepareStatement("SELECT manifestHostHash FROM CacheGroups"_s);
176 if (!statement)
177 return;
178
179 while (statement->step() == SQLITE_ROW)
180 m_cacheHostSet.add(static_cast<unsigned>(statement->columnInt64(0)));
181}

183ApplicationCacheGroup* ApplicationCacheStorage::cacheGroupForURL(const URL& url)
184{
185 ASSERT(!url.hasFragmentIdentifier());
186
187 loadManifestHostHashes();
188
189 // Hash the host name and see if there's a manifest with the same host.
190 if (!m_cacheHostSet.contains(urlHostHash(url)))
191 return nullptr;
192
193 // Check if a cache already exists in memory.
194 for (const auto& group : m_cachesInMemory.values()) {
195 ASSERT(!group->isObsolete());
196
197 if (!protocolHostAndPortAreEqual(url, group->manifestURL()))
198 continue;
199
200 if (ApplicationCache* cache = group->newestCache()) {
201 ApplicationCacheResource* resource = cache->resourceForURL(url.string());
202 if (!resource)
203 continue;
204 if (resource->type() & ApplicationCacheResource::Foreign)
205 continue;
206 return group;
207 }
208 }
209
210 if (!m_database.isOpen())
211 return nullptr;
212
213 SQLiteTransactionInProgressAutoCounter transactionCounter;
214
215 // Check the database. Look for all cache groups with a newest cache.
216 auto statement = m_database.prepareStatement("SELECT id, manifestURL, newestCache FROM CacheGroups WHERE newestCache IS NOT NULL"_s);
217 if (!statement)
218 return nullptr;
219
220 int result;
221 while ((result = statement->step()) == SQLITE_ROW) {
222 URL manifestURL = URL({ }, statement->columnText(1));
223
224 if (m_cachesInMemory.contains(manifestURL.string()))
225 continue;
226
227 if (!protocolHostAndPortAreEqual(url, manifestURL))
228 continue;
229
230 // We found a cache group that matches. Now check if the newest cache has a resource with
231 // a matching URL.
232 unsigned newestCacheID = static_cast<unsigned>(statement->columnInt64(2));
233 auto cache = loadCache(newestCacheID);
234 if (!cache)
235 continue;
236
237 auto* resource = cache->resourceForURL(url.string());
238 if (!resource)
239 continue;
240 if (resource->type() & ApplicationCacheResource::Foreign)
241 continue;
242
243 auto& group = *new ApplicationCacheGroup(*this, manifestURL);
244 group.setStorageID(static_cast<unsigned>(statement->columnInt64(0)));
245 group.setNewestCache(cache.releaseNonNull());
246 m_cachesInMemory.set(group.manifestURL().string(), &group);
247
248 return &group;
249 }
250
251 if (result != SQLITE_DONE)
252 LOG_ERROR("Could not load cache group, error \"%s\"", m_database.lastErrorMsg());
253
254 return nullptr;
255}

257ApplicationCacheGroup* ApplicationCacheStorage::fallbackCacheGroupForURL(const URL& url)
258{
259 SQLiteTransactionInProgressAutoCounter transactionCounter;
260
261 ASSERT(!url.hasFragmentIdentifier());
262
263 // Check if an appropriate cache already exists in memory.
264 for (auto* group : m_cachesInMemory.values()) {
265 ASSERT(!group->isObsolete());
266
267 if (ApplicationCache* cache = group->newestCache()) {
268 URL fallbackURL;
269 if (cache->isURLInOnlineAllowlist(url))
270 continue;
271 if (!cache->urlMatchesFallbackNamespace(url, &fallbackURL))
272 continue;
273 if (cache->resourceForURL(fallbackURL.string())->type() & ApplicationCacheResource::Foreign)
274 continue;
275 return group;
276 }
277 }
278
279 if (!m_database.isOpen())
280 return nullptr;
281
282 // Check the database. Look for all cache groups with a newest cache.
283 auto statement = m_database.prepareStatement("SELECT id, manifestURL, newestCache FROM CacheGroups WHERE newestCache IS NOT NULL"_s);
284 if (!statement)
285 return nullptr;
286
287 int result;
288 while ((result = statement->step()) == SQLITE_ROW) {
289 URL manifestURL = URL({ }, statement->columnText(1));
290
291 if (m_cachesInMemory.contains(manifestURL.string()))
292 continue;
293
294 // Fallback namespaces always have the same origin as manifest URL, so we can avoid loading caches that cannot match.
295 if (!protocolHostAndPortAreEqual(url, manifestURL))
296 continue;
297
298 // We found a cache group that matches. Now check if the newest cache has a resource with
299 // a matching fallback namespace.
300 unsigned newestCacheID = static_cast<unsigned>(statement->columnInt64(2));
301 auto cache = loadCache(newestCacheID);
302
303 URL fallbackURL;
304 if (cache->isURLInOnlineAllowlist(url))
305 continue;
306 if (!cache->urlMatchesFallbackNamespace(url, &fallbackURL))
307 continue;
308 if (cache->resourceForURL(fallbackURL.string())->type() & ApplicationCacheResource::Foreign)
309 continue;
310
311 auto& group = *new ApplicationCacheGroup(*this, manifestURL);
312 group.setStorageID(static_cast<unsigned>(statement->columnInt64(0)));
313 group.setNewestCache(cache.releaseNonNull());
314
315 m_cachesInMemory.set(group.manifestURL().string(), &group);
316
317 return &group;
318 }
319
320 if (result != SQLITE_DONE)
321 LOG_ERROR("Could not load cache group, error \"%s\"", m_database.lastErrorMsg());
322
323 return nullptr;
324}

326void ApplicationCacheStorage::cacheGroupDestroyed(ApplicationCacheGroup& group)
327{
328 if (group.isObsolete()) {
329 ASSERT(!group.storageID());
330 ASSERT(m_cachesInMemory.get(group.manifestURL().string()) != &group);
331 return;
332 }
333
334 ASSERT(m_cachesInMemory.get(group.manifestURL().string()) == &group);
335
336 m_cachesInMemory.remove(group.manifestURL().string());
337
338 // If the cache group is half-created, we don't want it in the saved set (as it is not stored in database).
339 if (!group.storageID())
340 m_cacheHostSet.remove(urlHostHash(group.manifestURL()));
341}
342
343void ApplicationCacheStorage::cacheGroupMadeObsolete(ApplicationCacheGroup& group)
344{
345 ASSERT(m_cachesInMemory.get(group.manifestURL().string()) == &group);
346 ASSERT(m_cacheHostSet.contains(urlHostHash(group.manifestURL())));
347
348 if (auto* newestCache = group.newestCache())
349 remove(newestCache);
350
351 m_cachesInMemory.remove(group.manifestURL().string());
352 m_cacheHostSet.remove(urlHostHash(group.manifestURL()));
353}

370int64_t ApplicationCacheStorage::spaceNeeded(int64_t cacheToSave)
371{
372 int64_t spaceNeeded = 0;
373 auto fileSize = FileSystem::fileSize(m_cacheFile);
374 if (!fileSize)
375 return 0;
376
377 int64_t currentSize = *fileSize + flatFileAreaSize();
378
379 // Determine the amount of free space we have available.
380 int64_t totalAvailableSize = 0;
381 if (m_maximumSize < currentSize) {
382 // The max size is smaller than the actual size of the app cache file.
383 // This can happen if the client previously imposed a larger max size
384 // value and the app cache file has already grown beyond the current
385 // max size value.
386 // The amount of free space is just the amount of free space inside
387 // the database file. Note that this is always 0 if SQLite is compiled
388 // with AUTO_VACUUM = 1.
389 totalAvailableSize = m_database.freeSpaceSize();
390 } else {
391 // The max size is the same or larger than the current size.
392 // The amount of free space available is the amount of free space
393 // inside the database file plus the amount we can grow until we hit
394 // the max size.
395 totalAvailableSize = (m_maximumSize - currentSize) + m_database.freeSpaceSize();
396 }
397
398 // The space needed to be freed in order to accommodate the failed cache is
399 // the size of the failed cache minus any already available free space.
400 spaceNeeded = cacheToSave - totalAvailableSize;
401 // The space needed value must be positive (or else the total already
402 // available free space would be larger than the size of the failed cache and
403 // saving of the cache should have never failed).
404 ASSERT(spaceNeeded);
405 return spaceNeeded;
406}

413bool ApplicationCacheStorage::calculateQuotaForOrigin(const SecurityOrigin& origin, int64_t& quota)
414{
415 SQLiteTransactionInProgressAutoCounter transactionCounter;
416
417 // If an Origin record doesn't exist, then the COUNT will be 0 and quota will be 0.
418 // Using the count to determine if a record existed or not is a safe way to determine
419 // if a quota of 0 is real, from the record, or from null.
420 auto statement = m_database.prepareStatement("SELECT COUNT(quota), quota FROM Origins WHERE origin=?"_s);
421 if (!statement)
422 return false;
423
424 statement->bindText(1, origin.data().databaseIdentifier());
425 int result = statement->step();
426
427 // Return the quota, or if it was null the default.
428 if (result == SQLITE_ROW) {
429 bool wasNoRecord = !statement->columnInt64(0);
430 quota = wasNoRecord ? m_defaultOriginQuota : statement->columnInt64(1);
431 return true;
432 }
433
434 LOG_ERROR("Could not get the quota of an origin, error \"%s\"", m_database.lastErrorMsg());
435 return false;
436}

438bool ApplicationCacheStorage::calculateUsageForOrigin(const SecurityOriginData& origin, int64_t& usage)
439{
440 SQLiteTransactionInProgressAutoCounter transactionCounter;
441
442 // If an Origins record doesn't exist, then the SUM will be null,
443 // which will become 0, as expected, when converting to a number.
444 auto statement = m_database.prepareStatement("SELECT SUM(Caches.size)"
445 " FROM CacheGroups"
446 " INNER JOIN Origins ON CacheGroups.origin = Origins.origin"
447 " INNER JOIN Caches ON CacheGroups.id = Caches.cacheGroup"
448 " WHERE Origins.origin=?"_s);
449 if (!statement)
450 return false;
451
452 statement->bindText(1, origin.databaseIdentifier());
453 int result = statement->step();
454
455 if (result == SQLITE_ROW) {
456 usage = statement->columnInt64(0);
457 return true;
458 }
459
460 LOG_ERROR("Could not get the quota of an origin, error \"%s\"", m_database.lastErrorMsg());
461 return false;
462}

464bool ApplicationCacheStorage::calculateRemainingSizeForOriginExcludingCache(const SecurityOrigin& origin, ApplicationCache* cache, int64_t& remainingSize)
465{
466 SQLiteTransactionInProgressAutoCounter transactionCounter;
467
468 openDatabase(false);
469 if (!m_database.isOpen())
470 return false;
471
472 // Remaining size = total origin quota - size of all caches with origin excluding the provided cache.
473 // Keep track of the number of caches so we can tell if the result was a calculation or not.
474 int64_t excludingCacheIdentifier = cache ? cache->storageID() : 0;
475 auto query = [excludingCacheIdentifier]() {
476 if (excludingCacheIdentifier) {
477 return "SELECT COUNT(Caches.size), Origins.quota - SUM(Caches.size)"
478 " FROM CacheGroups"
479 " INNER JOIN Origins ON CacheGroups.origin = Origins.origin"
480 " INNER JOIN Caches ON CacheGroups.id = Caches.cacheGroup"
481 " WHERE Origins.origin=?"
482 " AND Caches.id!=?"_s;
483 }
484 return "SELECT COUNT(Caches.size), Origins.quota - SUM(Caches.size)"
485 " FROM CacheGroups"
486 " INNER JOIN Origins ON CacheGroups.origin = Origins.origin"
487 " INNER JOIN Caches ON CacheGroups.id = Caches.cacheGroup"
488 " WHERE Origins.origin=?"_s;
489 }();
490
491 auto statement = m_database.prepareStatement(query);
492 if (!statement)
493 return false;
494
495 statement->bindText(1, origin.data().databaseIdentifier());
496 if (excludingCacheIdentifier != 0)
497 statement->bindInt64(2, excludingCacheIdentifier);
498 int result = statement->step();
499
500 // If the count was 0 that then we have to query the origin table directly
501 // for its quota. Otherwise we can use the calculated value.
502 if (result == SQLITE_ROW) {
503 int64_t numberOfCaches = statement->columnInt64(0);
504 if (numberOfCaches == 0)
505 calculateQuotaForOrigin(origin, remainingSize);
506 else
507 remainingSize = statement->columnInt64(1);
508 return true;
509 }
510
511 LOG_ERROR("Could not get the remaining size of an origin's quota, error \"%s\"", m_database.lastErrorMsg());
512 return false;
513}

515bool ApplicationCacheStorage::storeUpdatedQuotaForOrigin(const SecurityOrigin* origin, int64_t quota)
516{
517 SQLiteTransactionInProgressAutoCounter transactionCounter;
518
519 openDatabase(true);
520 if (!m_database.isOpen())
521 return false;
522
523 if (!ensureOriginRecord(origin))
524 return false;
525
526 auto updateStatement = m_database.prepareStatement("UPDATE Origins SET quota=? WHERE origin=?"_s);
527 if (!updateStatement)
528 return false;
529
530 updateStatement->bindInt64(1, quota);
531 updateStatement->bindText(2, origin->data().databaseIdentifier());
532
533 return executeStatement(updateStatement.value());

548// Update the schemaVersion when the schema of any the Application Cache
549// SQLite tables changes. This allows the database to be rebuilt when
550// a new, incompatible change has been introduced to the database schema.
551static const int schemaVersion = 7;
552
553void ApplicationCacheStorage::verifySchemaVersion()
554{
555 ASSERT(SQLiteDatabaseTracker::hasTransactionInProgress());
556
557 auto versionStatement = m_database.prepareStatement("PRAGMA user_version"_s);
558 int version = versionStatement ? versionStatement->columnInt(0) : 0;
559 if (version == schemaVersion)
560 return;
561
562 // Version will be 0 if we just created an empty file. Trying to delete tables would cause errors, because they don't exist yet.
563 if (version)
564 deleteTables();
565
566 // Update user version.
567 SQLiteTransaction setDatabaseVersion(m_database);
568 setDatabaseVersion.begin();
569
570 auto statement = m_database.prepareStatementSlow(makeString("PRAGMA user_version=%", schemaVersion));
571 if (!statement)
572 return;
573
574 executeStatement(statement.value());
575 setDatabaseVersion.commit();
576}

578void ApplicationCacheStorage::openDatabase(bool createIfDoesNotExist)
579{
580 SQLiteTransactionInProgressAutoCounter transactionCounter;
581
582 if (m_database.isOpen())
583 return;
584
585 // The cache directory should never be null, but if it for some weird reason is we bail out.
586 if (m_cacheDirectory.isNull())
587 return;
588
589 m_cacheFile = FileSystem::pathByAppendingComponent(m_cacheDirectory, "ApplicationCache.db");
590 if (!createIfDoesNotExist && !FileSystem::fileExists(m_cacheFile))
591 return;
592
593 FileSystem::makeAllDirectories(m_cacheDirectory);
594 m_database.open(m_cacheFile);
595
596 if (!m_database.isOpen())
597 return;
598
599 verifySchemaVersion();
600
601 // Create tables
602 executeSQLCommand("CREATE TABLE IF NOT EXISTS CacheGroups (id INTEGER PRIMARY KEY AUTOINCREMENT, "
603 "manifestHostHash INTEGER NOT NULL ON CONFLICT FAIL, manifestURL TEXT UNIQUE ON CONFLICT FAIL, newestCache INTEGER, origin TEXT)"_s);
604 executeSQLCommand("CREATE TABLE IF NOT EXISTS Caches (id INTEGER PRIMARY KEY AUTOINCREMENT, cacheGroup INTEGER, size INTEGER)"_s);
605 executeSQLCommand("CREATE TABLE IF NOT EXISTS CacheWhitelistURLs (url TEXT NOT NULL ON CONFLICT FAIL, cache INTEGER NOT NULL ON CONFLICT FAIL)"_s);
606 executeSQLCommand("CREATE TABLE IF NOT EXISTS CacheAllowsAllNetworkRequests (wildcard INTEGER NOT NULL ON CONFLICT FAIL, cache INTEGER NOT NULL ON CONFLICT FAIL)"_s);
607 executeSQLCommand("CREATE TABLE IF NOT EXISTS FallbackURLs (namespace TEXT NOT NULL ON CONFLICT FAIL, fallbackURL TEXT NOT NULL ON CONFLICT FAIL, "
608 "cache INTEGER NOT NULL ON CONFLICT FAIL)"_s);
609 executeSQLCommand("CREATE TABLE IF NOT EXISTS CacheEntries (cache INTEGER NOT NULL ON CONFLICT FAIL, type INTEGER, resource INTEGER NOT NULL)"_s);
610 executeSQLCommand("CREATE TABLE IF NOT EXISTS CacheResources (id INTEGER PRIMARY KEY AUTOINCREMENT, url TEXT NOT NULL ON CONFLICT FAIL, "
611 "statusCode INTEGER NOT NULL, responseURL TEXT NOT NULL, mimeType TEXT, textEncodingName TEXT, headers TEXT, data INTEGER NOT NULL ON CONFLICT FAIL)"_s);
612 executeSQLCommand("CREATE TABLE IF NOT EXISTS CacheResourceData (id INTEGER PRIMARY KEY AUTOINCREMENT, data BLOB, path TEXT)"_s);
613 executeSQLCommand("CREATE TABLE IF NOT EXISTS DeletedCacheResources (id INTEGER PRIMARY KEY AUTOINCREMENT, path TEXT)"_s);
614 executeSQLCommand("CREATE TABLE IF NOT EXISTS Origins (origin TEXT UNIQUE ON CONFLICT IGNORE, quota INTEGER NOT NULL ON CONFLICT FAIL)"_s);
615
616 // When a cache is deleted, all its entries and its allowlist should be deleted.
617 executeSQLCommand("CREATE TRIGGER IF NOT EXISTS CacheDeleted AFTER DELETE ON Caches"
618 " FOR EACH ROW BEGIN"
619 " DELETE FROM CacheEntries WHERE cache = OLD.id;"
620 " DELETE FROM CacheWhitelistURLs WHERE cache = OLD.id;"
621 " DELETE FROM CacheAllowsAllNetworkRequests WHERE cache = OLD.id;"
622 " DELETE FROM FallbackURLs WHERE cache = OLD.id;"
623 " END"_s);
624
625 // When a cache entry is deleted, its resource should also be deleted.
626 executeSQLCommand("CREATE TRIGGER IF NOT EXISTS CacheEntryDeleted AFTER DELETE ON CacheEntries"
627 " FOR EACH ROW BEGIN"
628 " DELETE FROM CacheResources WHERE id = OLD.resource;"
629 " END"_s);
630
631 // When a cache resource is deleted, its data blob should also be deleted.
632 executeSQLCommand("CREATE TRIGGER IF NOT EXISTS CacheResourceDeleted AFTER DELETE ON CacheResources"
633 " FOR EACH ROW BEGIN"
634 " DELETE FROM CacheResourceData WHERE id = OLD.data;"
635 " END"_s);
636
637 // When a cache resource is deleted, if it contains a non-empty path, that path should
638 // be added to the DeletedCacheResources table so the flat file at that path can
639 // be deleted at a later time.
640 executeSQLCommand("CREATE TRIGGER IF NOT EXISTS CacheResourceDataDeleted AFTER DELETE ON CacheResourceData"
641 " FOR EACH ROW"
642 " WHEN OLD.path NOT NULL BEGIN"
643 " INSERT INTO DeletedCacheResources (path) values (OLD.path);"
644 " END"_s);
645}

657bool ApplicationCacheStorage::store(ApplicationCacheGroup* group, GroupStorageIDJournal* journal)
658{
659 ASSERT(SQLiteDatabaseTracker::hasTransactionInProgress());
660 ASSERT(group->storageID() == 0);
661 ASSERT(journal);
662
663 // For some reason, an app cache may be partially written to disk. In particular, there may be
664 // a cache group with an identical manifest URL and associated cache entries. We want to remove
665 // this cache group and its associated cache entries so that we can create it again (below) as
666 // a way to repair it.
667 deleteCacheGroupRecord(group->manifestURL().string());
668
669 auto statement = m_database.prepareStatement("INSERT INTO CacheGroups (manifestHostHash, manifestURL, origin) VALUES (?, ?, ?)"_s);
670 if (!statement)
671 return false;
672
673 statement->bindInt64(1, urlHostHash(group->manifestURL()));
674 statement->bindText(2, group->manifestURL().string());
675 statement->bindText(3, group->origin().data().databaseIdentifier());
676
677 if (!executeStatement(statement.value()))
678 return false;
679
680 unsigned groupStorageID = static_cast<unsigned>(m_database.lastInsertRowID());
681
682 if (!ensureOriginRecord(&group->origin()))
683 return false;
684
685 group->setStorageID(groupStorageID);
686 journal->add(group, 0);
687 return true;
688}

690bool ApplicationCacheStorage::store(ApplicationCache* cache, ResourceStorageIDJournal* storageIDJournal)
691{
692 ASSERT(SQLiteDatabaseTracker::hasTransactionInProgress());
693 ASSERT(cache->storageID() == 0);
694 ASSERT(cache->group()->storageID() != 0);
695 ASSERT(storageIDJournal);
696
697 auto statement = m_database.prepareStatement("INSERT INTO Caches (cacheGroup, size) VALUES (?, ?)"_s);
698 if (!statement)
699 return false;
700
701 statement->bindInt64(1, cache->group()->storageID());
702 statement->bindInt64(2, cache->estimatedSizeInStorage());
703
704 if (!executeStatement(statement.value()))
705 return false;
706
707 unsigned cacheStorageID = static_cast<unsigned>(m_database.lastInsertRowID());
708
709 // Store all resources
710 for (auto& resource : cache->resources().values()) {
711 unsigned oldStorageID = resource->storageID();
712 if (!store(resource.get(), cacheStorageID))
713 return false;
714
715 // Storing the resource succeeded. Log its old storageID in case
716 // it needs to be restored later.
717 storageIDJournal->add(resource.get(), oldStorageID);
718 }
719
720 // Store the online allowlist
721 const Vector<URL>& onlineAllowlist = cache->onlineAllowlist();
722 {
723 for (auto& allowlistURL : onlineAllowlist) {
724 auto statement = m_database.prepareStatement("INSERT INTO CacheWhitelistURLs (url, cache) VALUES (?, ?)"_s);
725 if (!statement)
726 return false;
727
728 statement->bindText(1, allowlistURL.string());
729 statement->bindInt64(2, cacheStorageID);
730
731 if (!executeStatement(statement.value()))
732 return false;
733 }
734 }
735
736 // Store online allowlist wildcard flag.
737 {
738 auto statement = m_database.prepareStatement("INSERT INTO CacheAllowsAllNetworkRequests (wildcard, cache) VALUES (?, ?)"_s);
739 if (!statement)
740 return false;
741
742 statement->bindInt64(1, cache->allowsAllNetworkRequests());
743 statement->bindInt64(2, cacheStorageID);
744
745 if (!executeStatement(statement.value()))
746 return false;
747 }
748
749 // Store fallback URLs.
750 const FallbackURLVector& fallbackURLs = cache->fallbackURLs();
751 {
752 for (auto& fallbackURL : fallbackURLs) {
753 auto statement = m_database.prepareStatement("INSERT INTO FallbackURLs (namespace, fallbackURL, cache) VALUES (?, ?, ?)"_s);
754 if (!statement)
755 return false;
756
757 statement->bindText(1, fallbackURL.first.string());
758 statement->bindText(2, fallbackURL.second.string());
759 statement->bindInt64(3, cacheStorageID);
760
761 if (!executeStatement(statement.value()))
762 return false;
763 }
764 }
765
766 cache->setStorageID(cacheStorageID);
767 return true;
768}

770bool ApplicationCacheStorage::store(ApplicationCacheResource* resource, unsigned cacheStorageID)
771{
772 ASSERT(SQLiteDatabaseTracker::hasTransactionInProgress());
773 ASSERT(cacheStorageID);
774 ASSERT(!resource->storageID());
775
776 openDatabase(true);
777
778 // openDatabase(true) could still fail, for example when cacheStorage is full or no longer available.
779 if (!m_database.isOpen())
780 return false;
781
782 // First, insert the data
783 auto dataStatement = m_database.prepareStatement("INSERT INTO CacheResourceData (data, path) VALUES (?, ?)"_s);
784 if (!dataStatement)
785 return false;
786
787
788 String fullPath;
789 if (!resource->path().isEmpty())
790 dataStatement->bindText(2, FileSystem::pathFileName(resource->path()));
791 else if (shouldStoreResourceAsFlatFile(resource)) {
792 // First, check to see if creating the flat file would violate the maximum total quota. We don't need
793 // to check the per-origin quota here, as it was already checked in storeNewestCache().
794 if (m_database.totalSize() + flatFileAreaSize() + static_cast<int64_t>(resource->data().size()) > m_maximumSize) {
795 m_isMaximumSizeReached = true;
796 return false;
797 }
798
799 String flatFileDirectory = FileSystem::pathByAppendingComponent(m_cacheDirectory, m_flatFileSubdirectoryName);
800 FileSystem::makeAllDirectories(flatFileDirectory);
801
802 String extension;
803
804 String fileName = resource->response().suggestedFilename();
805 size_t dotIndex = fileName.reverseFind('.');
806 if (dotIndex != notFound && dotIndex < (fileName.length() - 1))
807 extension = fileName.substring(dotIndex);
808
809 String path;
810 if (!writeDataToUniqueFileInDirectory(resource->data(), flatFileDirectory, path, extension))
811 return false;
812
813 fullPath = FileSystem::pathByAppendingComponent(flatFileDirectory, path);
814 resource->setPath(fullPath);
815 dataStatement->bindText(2, path);
816 } else {
817 if (resource->data().size())
818 dataStatement->bindBlob(1, resource->data());
819 }
820
821 if (!dataStatement->executeCommand()) {
822 // Clean up the file which we may have written to:
823 if (!fullPath.isEmpty())
824 FileSystem::deleteFile(fullPath);
825
826 return false;
827 }
828
829 unsigned dataId = static_cast<unsigned>(m_database.lastInsertRowID());
830
831 // Then, insert the resource
832
833 // Serialize the headers
834 StringBuilder stringBuilder;
835
836 for (const auto& header : resource->response().httpHeaderFields()) {
837 stringBuilder.append(header.key);
838 stringBuilder.append(':');
839 stringBuilder.append(header.value);
840 stringBuilder.append('\n');
841 }
842
843 String headers = stringBuilder.toString();
844
845 auto resourceStatement = m_database.prepareStatement("INSERT INTO CacheResources (url, statusCode, responseURL, headers, data, mimeType, textEncodingName) VALUES (?, ?, ?, ?, ?, ?, ?)"_s);
846 if (!resourceStatement)
847 return false;
848
849 // The same ApplicationCacheResource are used in ApplicationCacheResource::size()
850 // to calculate the approximate size of an ApplicationCacheResource object. If
851 // you change the code below, please also change ApplicationCacheResource::size().
852 resourceStatement->bindText(1, resource->url().string());
853 resourceStatement->bindInt64(2, resource->response().httpStatusCode());
854 resourceStatement->bindText(3, resource->response().url().string());
855 resourceStatement->bindText(4, headers);
856 resourceStatement->bindInt64(5, dataId);
857 resourceStatement->bindText(6, resource->response().mimeType());
858 resourceStatement->bindText(7, resource->response().textEncodingName());
859
860 if (!executeStatement(resourceStatement.value()))
861 return false;
862
863 unsigned resourceId = static_cast<unsigned>(m_database.lastInsertRowID());
864
865 // Finally, insert the cache entry
866 auto entryStatement = m_database.prepareStatement("INSERT INTO CacheEntries (cache, type, resource) VALUES (?, ?, ?)"_s);
867 if (!entryStatement)
868 return false;
869
870 entryStatement->bindInt64(1, cacheStorageID);
871 entryStatement->bindInt64(2, resource->type());
872 entryStatement->bindInt64(3, resourceId);
873
874 if (!executeStatement(entryStatement.value()))
875 return false;
876
877 // Did we successfully write the resource data to a file? If so,
878 // release the resource's data and free up a potentially large amount
879 // of memory:
880 if (!fullPath.isEmpty())
881 resource->data().clear();
882
883 resource->setStorageID(resourceId);
884 return true;
885}

957bool ApplicationCacheStorage::checkOriginQuota(ApplicationCacheGroup* group, ApplicationCache* oldCache, ApplicationCache* newCache, int64_t& totalSpaceNeeded)
958{
959 // Check if the oldCache with the newCache would reach the per-origin quota.
960 int64_t remainingSpaceInOrigin;
961 auto& origin = group->origin();
962 if (calculateRemainingSizeForOriginExcludingCache(origin, oldCache, remainingSpaceInOrigin)) {
963 if (remainingSpaceInOrigin < newCache->estimatedSizeInStorage()) {
964 int64_t quota;
965 if (calculateQuotaForOrigin(origin, quota)) {
966 totalSpaceNeeded = quota - remainingSpaceInOrigin + newCache->estimatedSizeInStorage();
967 return false;
968 }
969
970 ASSERT_NOT_REACHED();
971 totalSpaceNeeded = 0;
972 return false;
973 }
974 }
975
976 return true;

979bool ApplicationCacheStorage::storeNewestCache(ApplicationCacheGroup& group, ApplicationCache* oldCache, FailureReason& failureReason)
980{
981 openDatabase(true);
982
983 if (!m_database.isOpen())
984 return false;
985
986 m_isMaximumSizeReached = false;
987 m_database.setMaximumSize(m_maximumSize - flatFileAreaSize());
988
989 SQLiteTransaction storeCacheTransaction(m_database);
990
991 storeCacheTransaction.begin();
992
993 // Check if this would reach the per-origin quota.
994 int64_t totalSpaceNeededIgnored;
995 if (!checkOriginQuota(&group, oldCache, group.newestCache(), totalSpaceNeededIgnored)) {
996 failureReason = OriginQuotaReached;
997 return false;
998 }
999
1000 GroupStorageIDJournal groupStorageIDJournal;
1001 if (!group.storageID()) {
1002 // Store the group
1003 if (!store(&group, &groupStorageIDJournal)) {
1004 checkForMaxSizeReached();
1005 failureReason = isMaximumSizeReached() ? TotalQuotaReached : DiskOrOperationFailure;
1006 return false;
1007 }
1008 }
1009
1010 ASSERT(group.newestCache());
1011 ASSERT(!group.isObsolete());
1012 ASSERT(!group.newestCache()->storageID());
1013
1014 // Log the storageID changes to the in-memory resource objects. The journal
1015 // object will roll them back automatically in case a database operation
1016 // fails and this method returns early.
1017 ResourceStorageIDJournal resourceStorageIDJournal;
1018
1019 // Store the newest cache
1020 if (!store(group.newestCache(), &resourceStorageIDJournal)) {
1021 checkForMaxSizeReached();
1022 failureReason = isMaximumSizeReached() ? TotalQuotaReached : DiskOrOperationFailure;
1023 return false;
1024 }
1025
1026 // Update the newest cache in the group.
1027
1028 auto statement = m_database.prepareStatement("UPDATE CacheGroups SET newestCache=? WHERE id=?"_s);
1029 if (!statement) {
1030 failureReason = DiskOrOperationFailure;
1031 return false;
1032 }
1033
1034 statement->bindInt64(1, group.newestCache()->storageID());
1035 statement->bindInt64(2, group.storageID());
1036
1037 if (!executeStatement(statement.value())) {
1038 failureReason = DiskOrOperationFailure;
1039 return false;
1040 }
1041
1042 groupStorageIDJournal.commit();
1043 resourceStorageIDJournal.commit();
1044 storeCacheTransaction.commit();
1045 return true;
1046}

1055template<typename CharacterType>
1056static inline void parseHeader(const CharacterType* header, unsigned headerLength, ResourceResponse& response)
1057{
1058 ASSERT(find(header, headerLength, ':') != notFound);
1059 unsigned colonPosition = find(header, headerLength, ':');
1060
1061 // Save memory by putting the header names into atom strings so each is stored only once,
1062 // even though the setHTTPHeaderField function does not require an atom string.
1063 AtomString headerName { header, colonPosition };
1064 String headerValue { header + colonPosition + 1, headerLength - colonPosition - 1 };
1065
1066 response.setHTTPHeaderField(headerName, headerValue);
1067}
1068
1069static inline void parseHeaders(const String& headers, ResourceResponse& response)
1070{
1071 unsigned startPos = 0;
1072 size_t endPos;
1073 while ((endPos = headers.find('\n', startPos)) != notFound) {
1074 ASSERT(startPos != endPos);
1075
1076 if (headers.is8Bit())
1077 parseHeader(headers.characters8() + startPos, endPos - startPos, response);
1078 else
1079 parseHeader(headers.characters16() + startPos, endPos - startPos, response);
1080
1081 startPos = endPos + 1;
1082 }
1083
1084 if (startPos != headers.length()) {
1085 if (headers.is8Bit())
1086 parseHeader(headers.characters8(), headers.length(), response);
1087 else
1088 parseHeader(headers.characters16(), headers.length(), response);
1089 }
1090}

1092RefPtr<ApplicationCache> ApplicationCacheStorage::loadCache(unsigned storageID)
1093{
1094 ASSERT(SQLiteDatabaseTracker::hasTransactionInProgress());
1095 auto cacheStatement = m_database.prepareStatement(
1096 "SELECT url, statusCode, type, mimeType, textEncodingName, headers, CacheResourceData.data, CacheResourceData.path FROM CacheEntries INNER JOIN CacheResources ON CacheEntries.resource=CacheResources.id "
1097 "INNER JOIN CacheResourceData ON CacheResourceData.id=CacheResources.data WHERE CacheEntries.cache=?"_s);
1098 if (!cacheStatement) {
1099 LOG_ERROR("Could not prepare cache statement, error \"%s\"", m_database.lastErrorMsg());
1100 return nullptr;
1101 }
1102
1103 cacheStatement->bindInt64(1, storageID);
1104
1105 auto cache = ApplicationCache::create();
1106
1107 String flatFileDirectory = FileSystem::pathByAppendingComponent(m_cacheDirectory, m_flatFileSubdirectoryName);
1108
1109 int result;
1110 while ((result = cacheStatement->step()) == SQLITE_ROW) {
1111 URL url({ }, cacheStatement->columnText(0));
1112
1113 int httpStatusCode = cacheStatement->columnInt(1);
1114
1115 unsigned type = static_cast<unsigned>(cacheStatement->columnInt64(2));
1116 auto data = SharedBuffer::create(cacheStatement->columnBlob(6));
1117
1118 String path = cacheStatement->columnText(7);
1119 long long size = 0;
1120 if (path.isEmpty())
1121 size = data->size();
1122 else {
1123 path = FileSystem::pathByAppendingComponent(flatFileDirectory, path);
1124 size = FileSystem::fileSize(path).value_or(0);
1125 }
1126
1127 String mimeType = cacheStatement->columnText(3);
1128 String textEncodingName = cacheStatement->columnText(4);
1129
1130 ResourceResponse response(url, mimeType, size, textEncodingName);
1131 response.setHTTPStatusCode(httpStatusCode);
1132
1133 String headers = cacheStatement->columnText(5);
1134 parseHeaders(headers, response);
1135
1136 auto resource = ApplicationCacheResource::create(url, response, type, WTFMove(data), path);
1137
1138 if (type & ApplicationCacheResource::Manifest)
1139 cache->setManifestResource(WTFMove(resource));
1140 else
1141 cache->addResource(WTFMove(resource));
1142 }
1143
1144 if (result != SQLITE_DONE)
1145 LOG_ERROR("Could not load cache resources, error \"%s\"", m_database.lastErrorMsg());
1146
1147 if (!cache->manifestResource()) {
1148 LOG_ERROR("Could not load application cache because there was no manifest resource");
1149 return nullptr;
1150 }
1151
1152 // Load the online allowlist
1153 auto allowlistStatement = m_database.prepareStatement("SELECT url FROM CacheWhitelistURLs WHERE cache=?"_s);
1154 if (!allowlistStatement)
1155 return nullptr;
1156 allowlistStatement->bindInt64(1, storageID);
1157
1158 Vector<URL> allowlist;
1159 while ((result = allowlistStatement->step()) == SQLITE_ROW)
1160 allowlist.append(URL({ }, allowlistStatement->columnText(0)));
1161
1162 if (result != SQLITE_DONE)
1163 LOG_ERROR("Could not load cache online allowlist, error \"%s\"", m_database.lastErrorMsg());
1164
1165 cache->setOnlineAllowlist(allowlist);
1166
1167 // Load online allowlist wildcard flag.
1168 auto allowlistWildcardStatement = m_database.prepareStatement("SELECT wildcard FROM CacheAllowsAllNetworkRequests WHERE cache=?"_s);
1169 if (!allowlistWildcardStatement)
1170 return nullptr;
1171 allowlistWildcardStatement->bindInt64(1, storageID);
1172
1173 result = allowlistWildcardStatement->step();
1174 if (result != SQLITE_ROW)
1175 LOG_ERROR("Could not load cache online allowlist wildcard flag, error \"%s\"", m_database.lastErrorMsg());
1176
1177 cache->setAllowsAllNetworkRequests(allowlistWildcardStatement->columnInt64(0));
1178
1179 if (allowlistWildcardStatement->step() != SQLITE_DONE)
1180 LOG_ERROR("Too many rows for online allowlist wildcard flag");
1181
1182 // Load fallback URLs.
1183 auto fallbackStatement = m_database.prepareStatement("SELECT namespace, fallbackURL FROM FallbackURLs WHERE cache=?"_s);
1184 if (!fallbackStatement)
1185 return nullptr;
1186 fallbackStatement->bindInt64(1, storageID);
1187
1188 FallbackURLVector fallbackURLs;
1189 while ((result = fallbackStatement->step()) == SQLITE_ROW)
1190 fallbackURLs.append(std::make_pair(URL({ }, fallbackStatement->columnText(0)), URL({ }, fallbackStatement->columnText(1))));
1191
1192 if (result != SQLITE_DONE)
1193 LOG_ERROR("Could not load fallback URLs, error \"%s\"", m_database.lastErrorMsg());
1194
1195 cache->setFallbackURLs(fallbackURLs);
1196
1197 cache->setStorageID(storageID);
1198
1199 return cache;
1200}

1276bool ApplicationCacheStorage::writeDataToUniqueFileInDirectory(SharedBuffer& data, const String& directory, String& path, const String& fileExtension)
1277{
1278 String fullPath;
1279
1280 do {
1281 path = FileSystem::encodeForFileName(createCanonicalUUIDString()) + fileExtension;
1282 // Guard against the above function being called on a platform which does not implement
1283 // createCanonicalUUIDString().
1284 ASSERT(!path.isEmpty());
1285 if (path.isEmpty())
1286 return false;
1287
1288 fullPath = FileSystem::pathByAppendingComponent(directory, path);
1289 } while (FileSystem::parentPath(fullPath) != directory || FileSystem::fileExists(fullPath));
1290

1292 if (!handle)
1293 return false;
1294
1295 int64_t writtenBytes = 0;
1296 data.forEachSegment([&](auto& segment) {
1297 writtenBytes += FileSystem::writeToFile(handle, segment.data(), segment.size());
1298 });
1299 FileSystem::closeFile(handle);
1300
1301 if (writtenBytes != static_cast<int64_t>(data.size())) {
1302 FileSystem::deleteFile(fullPath);
1303 return false;
1304 }
1305
1306 return true;
1307}

1401
1402void ApplicationCacheStorage::checkForDeletedResources()
1403{
1404 openDatabase(false);
1405 if (!m_database.isOpen())
1406 return;
1407
1408 // Select only the paths in DeletedCacheResources that do not also appear in CacheResourceData:
1409 auto selectPaths = m_database.prepareStatement("SELECT DeletedCacheResources.path "
1410 "FROM DeletedCacheResources "
1411 "LEFT JOIN CacheResourceData "
1412 "ON DeletedCacheResources.path = CacheResourceData.path "
1413 "WHERE (SELECT DeletedCacheResources.path == CacheResourceData.path) IS NULL"_s);
1414
1415 if (!selectPaths)
1416 return;
1417
1418 if (selectPaths->step() != SQLITE_ROW)
1419 return;
1420
1421 do {
1422 String path = selectPaths->columnText(0);
1423 if (path.isEmpty())
1424 continue;
1425
1426 String flatFileDirectory = FileSystem::pathByAppendingComponent(m_cacheDirectory, m_flatFileSubdirectoryName);
1427 String fullPath = FileSystem::pathByAppendingComponent(flatFileDirectory, path);
1428
1429 // Don't exit the flatFileDirectory! This should only happen if the "path" entry contains a directory
1430 // component, but protect against it regardless.
1431 if (FileSystem::parentPath(fullPath) != flatFileDirectory)
1432 continue;
1433
