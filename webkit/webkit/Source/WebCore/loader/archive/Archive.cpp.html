32namespace WebCore {
33
34Archive::~Archive() = default;
35
36void Archive::clearAllSubframeArchives()
37{
38 HashSet<Archive*> clearedArchives;
39 clearedArchives.add(this);
40 clearAllSubframeArchives(clearedArchives);
41}
42
43void Archive::clearAllSubframeArchives(HashSet<Archive*>& clearedArchives)
44{
45 ASSERT(clearedArchives.contains(this));
46 for (auto& archive : m_subframeArchives) {
47 if (clearedArchives.add(archive.ptr()))
48 archive->clearAllSubframeArchives(clearedArchives);
49 }
50 m_subframeArchives.clear();
51}
