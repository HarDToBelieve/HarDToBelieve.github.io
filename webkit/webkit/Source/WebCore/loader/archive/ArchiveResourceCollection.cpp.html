36void ArchiveResourceCollection::addAllResources(Archive& archive)
37{
38 for (auto& subresource : archive.subresources())
39 m_subresources.set(subresource->url().string(), subresource.ptr());
40
41 for (auto& subframeArchive : archive.subframeArchives()) {
42 ASSERT(subframeArchive->mainResource());
43 auto frameName = subframeArchive->mainResource()->frameName();
44 if (frameName.isNull()) {
45 // In the MHTML case, frames don't have a name, so we use the URL instead.
46 frameName = subframeArchive->mainResource()->url().string();
47 }
48 m_subframes.set(frameName, subframeArchive.ptr());
49 }
50}
51
52// FIXME: Adding a resource directly to a DocumentLoader/ArchiveResourceCollection seems like bad design, but is API some apps rely on.
53// Can we change the design in a manner that will let us deprecate that API without reducing functionality of those apps?
54void ArchiveResourceCollection::addResource(Ref<ArchiveResource>&& resource)
55{
56 auto& url = resource->url();
57 m_subresources.set(url.string(), WTFMove(resource));
58}
59
60ArchiveResource* ArchiveResourceCollection::archiveResourceForURL(const URL& url)
61{
62 if (auto* resource = m_subresources.get(url.string()))
63 return resource;
64 if (!url.protocolIs("https"))
65 return nullptr;
66 URL httpURL = url;
67 httpURL.setProtocol("http");
68 return m_subresources.get(httpURL.string());
69}
70
71RefPtr<Archive> ArchiveResourceCollection::popSubframeArchive(const String& frameName, const URL& url)
72{
73 if (auto archive = m_subframes.take(frameName))
74 return archive;
75 return m_subframes.take(url.string());
76}
