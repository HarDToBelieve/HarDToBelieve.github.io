34static NSString * const LegacyWebArchiveResourceResponseKey = @"WebResourceResponse";
35
36// FIXME: If is is possible to parse in a serialized NSURLResponse manually, without using
37// NSKeyedUnarchiver, manipulating plists directly, we would prefer to do that instead.
38ResourceResponse LegacyWebArchive::createResourceResponseFromMacArchivedData(CFDataRef responseData)
39{
40 ASSERT(responseData);
41 if (!responseData)
42 return ResourceResponse();
43
44 NSURLResponse *response = nil;
45 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:(__bridge NSData *)responseData error:nullptr]);
46 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
47 @try {
48 response = [unarchiver decodeObjectOfClass:NSURLResponse.class forKey:LegacyWebArchiveResourceResponseKey];
49 [unarchiver finishDecoding];
50 } @catch (NSException *exception) {
51 LOG_ERROR("Failed to decode NS(HTTP)URLResponse: %@", exception);
52 response = nil;
53 }
54
55 return ResourceResponse(response);
56}
57
58RetainPtr<CFDataRef> LegacyWebArchive::createPropertyListRepresentation(const ResourceResponse& response)
59{
60 NSURLResponse *nsResponse = response.nsURLResponse();
61 ASSERT(nsResponse);
62 if (!nsResponse)
63 return nullptr;
64
65 auto archiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
66 [archiver encodeObject:nsResponse forKey:LegacyWebArchiveResourceResponseKey];
67 return (__bridge CFDataRef)archiver.get().encodedData;
68}
