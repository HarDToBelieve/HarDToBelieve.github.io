42NSString *suggestedFilenameWithMIMEType(NSURL *url, const String& mimeType)
43{
44 // Get the filename from the URL. Try the lastPathComponent first.
45 NSString *lastPathComponent = [[url path] lastPathComponent];
46 NSString *filename = filenameByFixingIllegalCharacters(lastPathComponent);
47 NSString *extension = nil;
48
49 if ([filename length] == 0 || [lastPathComponent isEqualToString:@"/"]) {
50 // lastPathComponent is no good, try the host.
51 auto host = URL(url).host().createNSString();
52 filename = filenameByFixingIllegalCharacters(host.get());
53 if ([filename length] == 0) {
54 // Can't make a filename using this URL, use "unknown".
55 filename = copyImageUnknownFileLabel();
56 }
57 } else {
58 // Save the extension for later correction. Only correct the extension of the lastPathComponent.
59 // For example, if the filename ends up being the host, we wouldn't want to correct ".com" in "www.apple.com".
60 extension = [filename pathExtension];
61 }
62
63 // Do not correct filenames that are reported with a mime type of tar, and
64 // have a filename which has .tar in it or ends in .tgz
65 if ((mimeType == "application/tar" || mimeType == "application/x-tar")
66 && (String(filename).containsIgnoringASCIICase(".tar")
67 || String(filename).endsWithIgnoringASCIICase(".tgz"))) {
68 return filename;
69 }
70
71 // I don't think we need to worry about this for the image case
72 // If the type is known, check the extension and correct it if necessary.
73 if (mimeType != "application/octet-stream" && mimeType != "text/plain") {
74 Vector<String> extensions = MIMETypeRegistry::extensionsForMIMEType(mimeType);
75
76 if (extensions.isEmpty() || !extensions.contains(String(extension))) {
77 // The extension doesn't match the MIME type. Correct this.
78 NSString *correctExtension = MIMETypeRegistry::preferredExtensionForMIMEType(mimeType);
79 if ([correctExtension length] != 0) {
80 // Append the correct extension.
81 filename = [filename stringByAppendingPathExtension:correctExtension];
82 }
83 }
84 }
85
86 return filename;
87}

89NSString *filenameByFixingIllegalCharacters(NSString *string)
90{
91 auto filename = adoptNS([string mutableCopy]);
92
93 // Strip null characters.
94 unichar nullChar = 0;
95 [filename replaceOccurrencesOfString:[NSString stringWithCharacters:&nullChar length:0] withString:@"" options:0 range:NSMakeRange(0, [filename length])];
96
97 // Replace "/" with "-".
98 [filename replaceOccurrencesOfString:@"/" withString:@"-" options:0 range:NSMakeRange(0, [filename length])];
99
100 // Replace ":" with "-".
101 [filename replaceOccurrencesOfString:@":" withString:@"-" options:0 range:NSMakeRange(0, [filename length])];
102
103 // Strip leading dots.
104 while ([filename hasPrefix:@"."])
105 [filename deleteCharactersInRange:NSMakeRange(0, 1)];
106
107 return filename.autorelease();
108}
