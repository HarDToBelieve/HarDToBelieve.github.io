63unsigned MathMLElement::colSpan() const
64{
65 if (!hasTagName(mtdTag))
66 return 1u;
67 auto& colSpanValue = attributeWithoutSynchronization(columnspanAttr);
68 return std::max(1u, limitToOnlyHTMLNonNegative(colSpanValue, 1u));
69}
70
71unsigned MathMLElement::rowSpan() const
72{
73 if (!hasTagName(mtdTag))
74 return 1u;
75 auto& rowSpanValue = attributeWithoutSynchronization(rowspanAttr);
76 static const unsigned maxRowspan = 8190; // This constant comes from HTMLTableCellElement.
77 return std::max(1u, std::min(limitToOnlyHTMLNonNegative(rowSpanValue, 1u), maxRowspan));
78}

80void MathMLElement::parseAttribute(const QualifiedName& name, const AtomString& value)
81{
82 if (name == hrefAttr) {
83 bool wasLink = isLink();
84 setIsLink(!value.isNull() && !shouldProhibitLinks(this));
85 if (wasLink != isLink())
86 invalidateStyleForSubtree();
87 } else if (name == rowspanAttr) {
88 if (is<RenderTableCell>(renderer()) && hasTagName(mtdTag))
89 downcast<RenderTableCell>(*renderer()).colSpanOrRowSpanChanged();
90 } else if (name == columnspanAttr) {
91 if (is<RenderTableCell>(renderer()) && hasTagName(mtdTag))
92 downcast<RenderTableCell>(renderer())->colSpanOrRowSpanChanged();
93 } else if (name == HTMLNames::tabindexAttr) {
94 if (value.isEmpty())
95 setTabIndexExplicitly(std::nullopt);
96 else if (auto optionalTabIndex = parseHTMLInteger(value))
97 setTabIndexExplicitly(optionalTabIndex.value());
98 } else {
99 auto& eventName = HTMLElement::eventNameForEventHandlerAttribute(name);
100 if (!eventName.isNull()) {
101 setAttributeEventListener(eventName, name, value);
102 return;
103 }
104
105 StyledElement::parseAttribute(name, value);
106 }
107}

109bool MathMLElement::hasPresentationalHintsForAttribute(const QualifiedName& name) const
110{
111 if (name == backgroundAttr || name == colorAttr || name == dirAttr || name == fontfamilyAttr || name == fontsizeAttr || name == fontstyleAttr || name == fontweightAttr || name == mathbackgroundAttr || name == mathcolorAttr || name == mathsizeAttr || name == displaystyleAttr)
112 return true;
113 return StyledElement::hasPresentationalHintsForAttribute(name);
114}

116static String convertMathSizeIfNeeded(const AtomString& value)
117{
118 if (value == "small")
119 return "0.75em";
120 if (value == "normal")
121 return "1em";
122 if (value == "big")
123 return "1.5em";
124
125 // FIXME: mathsize accepts any MathML length, including named spaces (see parseMathMLLength).
126 // FIXME: Might be better to use double than float.
127 // FIXME: Might be better to use "shortest" numeric formatting instead of fixed width.
128 bool ok = false;
129 float unitlessValue = value.toFloat(&ok);
130 if (!ok)
131 return value;
132 return makeString(FormattedNumber::fixedWidth(unitlessValue * 100, 3), '%');
133}

135void MathMLElement::collectPresentationalHintsForAttribute(const QualifiedName& name, const AtomString& value, MutableStyleProperties& style)
136{
137 if (name == mathbackgroundAttr)
138 addPropertyToPresentationalHintStyle(style, CSSPropertyBackgroundColor, value);
139 else if (name == mathsizeAttr)
140 addPropertyToPresentationalHintStyle(style, CSSPropertyFontSize, convertMathSizeIfNeeded(value));
141 else if (name == mathcolorAttr)
142 addPropertyToPresentationalHintStyle(style, CSSPropertyColor, value);
143 else if (name == dirAttr) {
144 if (document().settings().coreMathMLEnabled() || hasTagName(mathTag) || hasTagName(mrowTag) || hasTagName(mstyleTag) || isMathMLToken())
145 addPropertyToPresentationalHintStyle(style, CSSPropertyDirection, value);
146 } else if (name == displaystyleAttr) {
147 if (equalLettersIgnoringASCIICase(value, "false"))
148 addPropertyToPresentationalHintStyle(style, CSSPropertyMathStyle, CSSValueCompact);
149 else if (equalLettersIgnoringASCIICase(value, "true"))
150 addPropertyToPresentationalHintStyle(style, CSSPropertyMathStyle, CSSValueNormal);
151 } else {
152 if (document().settings().coreMathMLEnabled()) {
153 StyledElement::collectPresentationalHintsForAttribute(name, value, style);
154 return;
155 }
156 // FIXME: The following are deprecated attributes that should lose if there is a conflict with a non-deprecated attribute.
157 if (name == fontsizeAttr)
158 addPropertyToPresentationalHintStyle(style, CSSPropertyFontSize, value);
159 else if (name == backgroundAttr)
160 addPropertyToPresentationalHintStyle(style, CSSPropertyBackgroundColor, value);
161 else if (name == colorAttr)
162 addPropertyToPresentationalHintStyle(style, CSSPropertyColor, value);
163 else if (name == fontstyleAttr)
164 addPropertyToPresentationalHintStyle(style, CSSPropertyFontStyle, value);
165 else if (name == fontweightAttr)
166 addPropertyToPresentationalHintStyle(style, CSSPropertyFontWeight, value);
167 else if (name == fontfamilyAttr)
168 addPropertyToPresentationalHintStyle(style, CSSPropertyFontFamily, value);
169 else {
170 ASSERT(!hasPresentationalHintsForAttribute(name));
171 StyledElement::collectPresentationalHintsForAttribute(name, value, style);
172 }
173 }

187void MathMLElement::defaultEventHandler(Event& event)
188{
189 if (isLink()) {
190 if (focused() && isEnterKeyKeydownEvent(event)) {
191 event.setDefaultHandled();
192 dispatchSimulatedClick(&event);
193 return;
194 }
195 if (MouseEvent::canTriggerActivationBehavior(event)) {
196 auto& href = attributeWithoutSynchronization(hrefAttr);
197 const auto& url = stripLeadingAndTrailingHTMLSpaces(href);
198 event.setDefaultHandled();
199 if (auto* frame = document().frame())
200 frame->loader().changeLocation(document().completeURL(url), "_self", &event, ReferrerPolicy::EmptyString, document().shouldOpenExternalURLsPolicyToPropagate());
201 return;
202 }
203 }
204
205 StyledElement::defaultEventHandler(event);
206}

216bool MathMLElement::isKeyboardFocusable(KeyboardEvent* event) const
217{
218 if (isFocusable() && StyledElement::supportsFocus())
219 return StyledElement::isKeyboardFocusable(event);
220
221 if (isLink())
222 return document().frame()->eventHandler().tabsToLinks(event);
223
224 return StyledElement::isKeyboardFocusable(event);
225}

227bool MathMLElement::isMouseFocusable() const
228{
229 // Links are focusable by default, but only allow links with tabindex or contenteditable to be mouse focusable.
230 // https://bugs.webkit.org/show_bug.cgi?id=26856
231 if (isLink())
232 return StyledElement::supportsFocus();
233
234 return StyledElement::isMouseFocusable();
235}
236
237bool MathMLElement::isURLAttribute(const Attribute& attribute) const
238{
239 return attribute.name().localName() == hrefAttr || StyledElement::isURLAttribute(attribute);
240}
241
242bool MathMLElement::supportsFocus() const
243{
244 if (hasEditableStyle())
245 return StyledElement::supportsFocus();
246 // If not a link we should still be able to focus the element if it has tabIndex.
247 return isLink() || StyledElement::supportsFocus();
248}
