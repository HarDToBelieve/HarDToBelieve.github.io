59void MathMLMathElement::parseAttribute(const QualifiedName& name, const AtomString& value)
60{
61 bool mathVariantAttribute = name == mathvariantAttr;
62 if (mathVariantAttribute)
63 m_mathVariant = std::nullopt;
64 if ((mathVariantAttribute) && renderer())
65 MathMLStyle::resolveMathMLStyleTree(renderer());
66
67 MathMLElement::parseAttribute(name, value);
68}

70void MathMLMathElement::didAttachRenderers()
71{
72 MathMLRowElement::didAttachRenderers();
73
74 MathMLStyle::resolveMathMLStyleTree(renderer());
75}
