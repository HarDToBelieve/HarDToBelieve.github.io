39using namespace MathMLNames;

41static RootType rootTypeOf(const QualifiedName& tagName)
42{
43 if (tagName.matches(msqrtTag))
44 return RootType::SquareRoot;
45 ASSERT(tagName.matches(mrootTag));
46 return RootType::RootWithIndex;
47}
48
49inline MathMLRootElement::MathMLRootElement(const QualifiedName& tagName, Document& document)
50 : MathMLRowElement(tagName, document)
51 , m_rootType(rootTypeOf(tagName))
52{
53}
54
55Ref<MathMLRootElement> MathMLRootElement::create(const QualifiedName& tagName, Document& document)
56{
57 return adoptRef(*new MathMLRootElement(tagName, document));
58}
59
60RenderPtr<RenderElement> MathMLRootElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
61{
62 ASSERT(hasTagName(msqrtTag) || hasTagName(mrootTag));
63 return createRenderer<RenderMathMLRoot>(*this, WTFMove(style));
64}
