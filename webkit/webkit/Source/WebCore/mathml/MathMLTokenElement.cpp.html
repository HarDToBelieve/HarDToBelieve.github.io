44MathMLTokenElement::MathMLTokenElement(const QualifiedName& tagName, Document& document)

50Ref<MathMLTokenElement> MathMLTokenElement::create(const QualifiedName& tagName, Document& document)
51{
52 return adoptRef(*new MathMLTokenElement(tagName, document));

55void MathMLTokenElement::didAttachRenderers()
56{
57 MathMLPresentationElement::didAttachRenderers();
58 auto* mathmlRenderer = renderer();
59 if (is<RenderMathMLToken>(mathmlRenderer))
60 downcast<RenderMathMLToken>(*mathmlRenderer).updateTokenContent();
61}
62
63void MathMLTokenElement::childrenChanged(const ChildChange& change)
64{
65 MathMLPresentationElement::childrenChanged(change);
66 auto* mathmlRenderer = renderer();
67 if (is<RenderMathMLToken>(mathmlRenderer))
68 downcast<RenderMathMLToken>(*mathmlRenderer).updateTokenContent();
69}

71RenderPtr<RenderElement> MathMLTokenElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
72{
73 ASSERT(hasTagName(MathMLNames::miTag) || hasTagName(MathMLNames::mnTag) || hasTagName(MathMLNames::msTag) || hasTagName(MathMLNames::mtextTag));
74
75 return createRenderer<RenderMathMLToken>(*this, WTFMove(style));
76}
77
78bool MathMLTokenElement::childShouldCreateRenderer(const Node& child) const
79{
80 // The HTML specification defines <mi>, <mo>, <mn>, <ms> and <mtext> as insertion points.
81 return isPhrasingContent(child) && StyledElement::childShouldCreateRenderer(child);
82}
83
84std::optional<UChar32> MathMLTokenElement::convertToSingleCodePoint(StringView string)
85{
86 auto codePoints = stripLeadingAndTrailingHTTPSpaces(string).codePoints();
87 auto iterator = codePoints.begin();
88 if (iterator == codePoints.end())
89 return std::nullopt;
90 std::optional<UChar32> character = *iterator;
91 ++iterator;
92 return iterator == codePoints.end() ? character : std::nullopt;
93}
