26#include "config.h"
27#include "ActivityState.h"
28
29#include <wtf/text/TextStream.h>
30
31namespace WebCore {
32
33TextStream& operator<<(TextStream& ts, OptionSet<ActivityState::Flag> flags)
34{
35 bool didAppend = false;
36
37 auto appendIf = [&](auto flag, auto message) {
38 if (!flags.contains(flag))
39 return;
40 if (didAppend)
41 ts << ", ";
42 ts << message;
43 didAppend = true;
44 };
45
46 appendIf(ActivityState::WindowIsActive, "active window");
47 appendIf(ActivityState::IsFocused, "focused");
48 appendIf(ActivityState::IsVisible, "visible");
49 appendIf(ActivityState::IsVisibleOrOccluded, "visible or occluded");
50 appendIf(ActivityState::IsInWindow, "in-window");
51 appendIf(ActivityState::IsVisuallyIdle, "visually idle");
52 appendIf(ActivityState::IsAudible, "audible");
53 appendIf(ActivityState::IsLoading, "loading");
54 appendIf(ActivityState::IsCapturingMedia, "capturing media");
55
56 return ts;
57}
58
59} // namespace WebCore
