43// Delay time in second for start autoscroll if pointer is in border edge of scrollable element.
44static const Seconds autoscrollDelay { 200_ms };

74void AutoscrollController::startAutoscrollForSelection(RenderObject* renderer)
75{
76 // We don't want to trigger the autoscroll or the panScroll if it's already active
77 if (m_autoscrollTimer.isActive())
78 return;
79 RenderBox* scrollable = RenderBox::findAutoscrollable(renderer);
80 if (!scrollable)
81 return;
82 m_autoscrollType = AutoscrollForSelection;
83 m_autoscrollRenderer = scrollable;
84 startAutoscrollTimer();
85}

87void AutoscrollController::stopAutoscrollTimer(bool rendererIsBeingDestroyed)
88{
89 RenderBox* scrollable = m_autoscrollRenderer;
90 m_autoscrollTimer.stop();
91 m_autoscrollRenderer = nullptr;
92
93 if (!scrollable)
94 return;
95
96 Frame& frame = scrollable->frame();
97 if (autoscrollInProgress() && frame.eventHandler().mouseDownWasInSubframe()) {
98 if (auto subframe = frame.eventHandler().subframeForTargetNode(frame.eventHandler().mousePressNode()))
99 subframe->eventHandler().stopAutoscrollTimer(rendererIsBeingDestroyed);
100 return;
101 }
102
103 if (!rendererIsBeingDestroyed)
104 scrollable->stopAutoscroll();
105#if ENABLE(PAN_SCROLLING)
106 if (panScrollInProgress()) {
107 FrameView& frameView = scrollable->view().frameView();
108 frameView.removePanScrollIcon();
109 frameView.setCursor(pointerCursor());
110 }
111#endif
112
113 m_autoscrollType = NoAutoscroll;
114
115#if ENABLE(PAN_SCROLLING)
116 // If we're not in the top frame we notify it that we are not doing a panScroll any more.
117 if (!frame.isMainFrame())
118 frame.mainFrame().eventHandler().didPanScrollStop();
119#endif
120}

122void AutoscrollController::updateAutoscrollRenderer()
123{
124 if (!m_autoscrollRenderer)
125 return;
126
127 RenderObject* renderer = m_autoscrollRenderer;
128
129#if ENABLE(PAN_SCROLLING)
130 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::AllowChildFrameContent };
131 HitTestResult hitTest = m_autoscrollRenderer->frame().eventHandler().hitTestResultAtPoint(m_panScrollStartPos, hitType);
132
133 if (Node* nodeAtPoint = hitTest.innerNode())
134 renderer = nodeAtPoint->renderer();
135#endif
136
137 while (renderer && !(is<RenderBox>(*renderer) && downcast<RenderBox>(*renderer).canAutoscroll()))
138 renderer = renderer->parent();
139 m_autoscrollRenderer = is<RenderBox>(renderer) ? downcast<RenderBox>(renderer) : nullptr;
140}

142void AutoscrollController::updateDragAndDrop(Node* dropTargetNode, const IntPoint& eventPosition, WallTime eventTime)
143{
144 if (!dropTargetNode) {
145 stopAutoscrollTimer();
146 return;
147 }
148
149 RenderBox* scrollable = RenderBox::findAutoscrollable(dropTargetNode->renderer());
150 if (!scrollable) {
151 stopAutoscrollTimer();
152 return;
153 }
154
155 Frame& frame = scrollable->frame();
156
157 Page* page = frame.page();
158 if (!page || !page->settings().autoscrollForDragAndDropEnabled()) {
159 stopAutoscrollTimer();
160 return;
161 }
162
163 IntSize offset = scrollable->calculateAutoscrollDirection(eventPosition);
164 if (offset.isZero()) {
165 stopAutoscrollTimer();
166 return;
167 }
168
169 m_dragAndDropAutoscrollReferencePosition = eventPosition + offset;
170
171 if (m_autoscrollType == NoAutoscroll) {
172 m_autoscrollType = AutoscrollForDragAndDrop;
173 m_autoscrollRenderer = scrollable;
174 m_dragAndDropAutoscrollStartTime = eventTime;
175 startAutoscrollTimer();
176 } else if (m_autoscrollRenderer != scrollable) {
177 m_dragAndDropAutoscrollStartTime = eventTime;
178 m_autoscrollRenderer = scrollable;
179 }
180}

193void AutoscrollController::handleMouseReleaseEvent(const PlatformMouseEvent& mouseEvent)
194{
195 switch (m_autoscrollType) {
196 case AutoscrollForPan:
197 if (mouseEvent.button() == MiddleButton)
198 m_autoscrollType = AutoscrollForPanCanStop;
199 break;
200 case AutoscrollForPanCanStop:
201 stopAutoscrollTimer();
202 break;
203 default:
204 break;
205 }

213void AutoscrollController::startPanScrolling(RenderBox* scrollable, const IntPoint& lastKnownMousePosition)
214{
215 // We don't want to trigger the autoscroll or the panScroll if it's already active
216 if (m_autoscrollTimer.isActive())
217 return;
218
219 m_autoscrollType = AutoscrollForPan;
220 m_autoscrollRenderer = scrollable;
221 m_panScrollStartPos = lastKnownMousePosition;
222
223 if (FrameView* view = scrollable->frame().view())
224 view->addPanScrollIcon(lastKnownMousePosition);
225 scrollable->frame().eventHandler().didPanScrollStart();
226 startAutoscrollTimer();
227}
228#else
229bool AutoscrollController::panScrollInProgress() const
230{
231 return false;
232}
233#endif

235void AutoscrollController::autoscrollTimerFired()
236{
237 if (!m_autoscrollRenderer) {
238 stopAutoscrollTimer();
239 return;
240 }
241
242 Frame& frame = m_autoscrollRenderer->frame();
243 switch (m_autoscrollType) {
244 case AutoscrollForDragAndDrop:
245 if (WallTime::now() - m_dragAndDropAutoscrollStartTime > autoscrollDelay)
246 m_autoscrollRenderer->autoscroll(m_dragAndDropAutoscrollReferencePosition);
247 break;
248 case AutoscrollForSelection: {
249 if (!frame.eventHandler().shouldUpdateAutoscroll()) {
250 stopAutoscrollTimer();
251 return;
252 }
253#if ENABLE(DRAG_SUPPORT)
254 frame.eventHandler().updateSelectionForMouseDrag();
255#endif
256 m_autoscrollRenderer->autoscroll(frame.eventHandler().targetPositionInWindowForSelectionAutoscroll());
257 break;
258 }
259 case NoAutoscroll:
260 break;
261#if ENABLE(PAN_SCROLLING)
262 case AutoscrollForPanCanStop:
263 case AutoscrollForPan:
264 // we verify that the main frame hasn't received the order to stop the panScroll
265 if (Frame* mainFrame = getMainFrame(&frame)) {
266 if (!mainFrame->eventHandler().panScrollInProgress()) {
267 stopAutoscrollTimer();
268 return;
269 }
270 }
271 if (FrameView* view = frame.view())
272 updatePanScrollState(view, frame.eventHandler().lastKnownMousePosition());
273 m_autoscrollRenderer->panScroll(m_panScrollStartPos);
274 break;
275#endif
276 }
277}

279void AutoscrollController::startAutoscrollTimer()
280{
281 m_autoscrollTimer.startRepeating(autoscrollInterval);
282}

284#if ENABLE(PAN_SCROLLING)
285void AutoscrollController::updatePanScrollState(FrameView* view, const IntPoint& lastKnownMousePosition)
286{
287 // At the original click location we draw a 4 arrowed icon. Over this icon there won't be any scroll
288 // So we don't want to change the cursor over this area
289 bool east = m_panScrollStartPos.x() < (lastKnownMousePosition.x() - ScrollView::noPanScrollRadius);
290 bool west = m_panScrollStartPos.x() > (lastKnownMousePosition.x() + ScrollView::noPanScrollRadius);
291 bool north = m_panScrollStartPos.y() > (lastKnownMousePosition.y() + ScrollView::noPanScrollRadius);
292 bool south = m_panScrollStartPos.y() < (lastKnownMousePosition.y() - ScrollView::noPanScrollRadius);
293
294 if (m_autoscrollType == AutoscrollForPan && (east || west || north || south))
295 m_autoscrollType = AutoscrollForPanCanStop;
296
297 if (north) {
298 if (east)
299 view->setCursor(northEastPanningCursor());
300 else if (west)
301 view->setCursor(northWestPanningCursor());
302 else
303 view->setCursor(northPanningCursor());
304 } else if (south) {
305 if (east)
306 view->setCursor(southEastPanningCursor());
307 else if (west)
308 view->setCursor(southWestPanningCursor());
309 else
310 view->setCursor(southPanningCursor());
311 } else if (east)
312 view->setCursor(eastPanningCursor());
313 else if (west)
314 view->setCursor(westPanningCursor());
315 else
316 view->setCursor(middlePanningCursor());
317}
318#endif
