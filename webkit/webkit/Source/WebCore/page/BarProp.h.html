36namespace WebCore {
37
38class BarProp final : public ScriptWrappable, public RefCounted<BarProp>, public DOMWindowProperty {
39 WTF_MAKE_ISO_ALLOCATED(BarProp);
40public:
41 enum Type { Locationbar, Menubar, Personalbar, Scrollbars, Statusbar, Toolbar };
42
43 static Ref<BarProp> create(DOMWindow& window, Type type) { return adoptRef(*new BarProp(window, type)); }
44
45 Type type() const;
46 bool visible() const;
47
48private:
49 BarProp(DOMWindow&, Type);
50 Type m_type;
51};
