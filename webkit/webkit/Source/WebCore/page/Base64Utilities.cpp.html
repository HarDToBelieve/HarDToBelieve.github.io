33ExceptionOr<String> Base64Utilities::btoa(const String& stringToEncode)
34{
35 if (stringToEncode.isNull())
36 return String();
37
38 if (!stringToEncode.isAllLatin1())
39 return Exception { InvalidCharacterError };
40
41 return base64EncodeToString(stringToEncode.latin1());
42}
43
44ExceptionOr<String> Base64Utilities::atob(const String& encodedString)
45{
46 if (encodedString.isNull())
47 return String();
48
49 auto decodedData = base64Decode(encodedString, { Base64DecodeOptions::ValidatePadding, Base64DecodeOptions::IgnoreSpacesAndNewLines, Base64DecodeOptions::DiscardVerticalTab });
50 if (!decodedData)
51 return Exception { InvalidCharacterError };
52
53 return String(decodedData->data(), decodedData->size());
54}
