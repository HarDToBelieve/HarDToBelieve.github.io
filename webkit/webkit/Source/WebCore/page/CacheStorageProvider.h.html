31namespace WebCore {
32
33class CacheStorageProvider : public RefCounted<CacheStorageProvider> {
34public:
35 class DummyCacheStorageConnection final : public WebCore::CacheStorageConnection {
36 public:
37 static Ref<DummyCacheStorageConnection> create() { return adoptRef(*new DummyCacheStorageConnection()); }
38
39 private:
40 DummyCacheStorageConnection()
41 {
42 }
43
44 void open(const ClientOrigin&, const String&, DOMCacheEngine::CacheIdentifierCallback&&) final { }
45 void remove(uint64_t, DOMCacheEngine::CacheIdentifierCallback&&) final { }
46 void retrieveCaches(const ClientOrigin&, uint64_t, DOMCacheEngine::CacheInfosCallback&&) final { }
47 void retrieveRecords(uint64_t, const RetrieveRecordsOptions&, DOMCacheEngine::RecordsCallback&&) final { }
48 void batchDeleteOperation(uint64_t, const ResourceRequest&, CacheQueryOptions&&, DOMCacheEngine::RecordIdentifiersCallback&&) final { }
49 void batchPutOperation(uint64_t, Vector<DOMCacheEngine::Record>&&, DOMCacheEngine::RecordIdentifiersCallback&&) final { }
50 void reference(uint64_t) final { }
51 void dereference(uint64_t) final { }
52 };
53
54 static Ref<CacheStorageProvider> create() { return adoptRef(*new CacheStorageProvider); }
55 virtual Ref<CacheStorageConnection> createCacheStorageConnection() { return DummyCacheStorageConnection::create(); }
56 virtual ~CacheStorageProvider() { };
57
58protected:
59 CacheStorageProvider() = default;
60};
