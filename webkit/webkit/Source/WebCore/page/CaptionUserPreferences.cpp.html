77void CaptionUserPreferences::notify()
78{

200static String trackDisplayName(TextTrack* track)
201{
202 if (track == &TextTrack::captionMenuOffItem())
203 return textTrackOffMenuItemText();
204 if (track == &TextTrack::captionMenuAutomaticItem())
205 return textTrackAutomaticMenuItemText();
206
207 if (track->label().isEmpty() && track->validBCP47Language().isEmpty())
208 return trackNoLabelText();
209 if (!track->label().isEmpty())
210 return track->label();
211 return track->validBCP47Language();
212}

229Vector<RefPtr<TextTrack>> CaptionUserPreferences::sortedTrackListForMenu(TextTrackList* trackList, HashSet<TextTrack::Kind> kinds)
230{
231 ASSERT(trackList);
232
233 Vector<RefPtr<TextTrack>> tracksForMenu;
234
235 for (unsigned i = 0, length = trackList->length(); i < length; ++i) {
236 TextTrack* track = trackList->item(i);
237 if (kinds.contains(track->kind()))
238 tracksForMenu.append(track);
239 }
240
241 Collator collator;
242
243 std::sort(tracksForMenu.begin(), tracksForMenu.end(), [&] (auto& a, auto& b) {
244 return collator.collate(trackDisplayName(a.get()), trackDisplayName(b.get())) < 0;
245 });
246
247 if (kinds.contains(TextTrack::Kind::Subtitles) || kinds.contains(TextTrack::Kind::Captions) || kinds.contains(TextTrack::Kind::Descriptions)) {
248 tracksForMenu.insert(0, &TextTrack::captionMenuOffItem());
249 tracksForMenu.insert(1, &TextTrack::captionMenuAutomaticItem());
250 }
251
252 return tracksForMenu;
253}

294int CaptionUserPreferences::textTrackSelectionScore(TextTrack* track, HTMLMediaElement* mediaElement) const
295{
296 CaptionDisplayMode displayMode = captionDisplayMode();
297 if (displayMode == Manual)
298 return 0;
299
300 bool legacyOverride = mediaElement->webkitClosedCaptionsVisible();
301 if (displayMode == AlwaysOn && (!userPrefersSubtitles() && !userPrefersCaptions() && !legacyOverride))
302 return 0;
303 if (track->kind() != TextTrack::Kind::Captions && track->kind() != TextTrack::Kind::Subtitles && track->kind() != TextTrack::Kind::Forced)
304 return 0;
305 if (!track->isMainProgramContent())
306 return 0;
307
308 bool trackHasOnlyForcedSubtitles = track->containsOnlyForcedSubtitles();
309 if (!legacyOverride && ((trackHasOnlyForcedSubtitles && displayMode != ForcedOnly) || (!trackHasOnlyForcedSubtitles && displayMode == ForcedOnly)))
310 return 0;
311
312 Vector<String> userPreferredCaptionLanguages = preferredLanguages();
313
314 if ((displayMode == Automatic && !legacyOverride) || trackHasOnlyForcedSubtitles) {
315
316 if (!mediaElement || !mediaElement->player())
317 return 0;
318
319 String textTrackLanguage = track->validBCP47Language();
320 if (textTrackLanguage.isEmpty())
321 return 0;
322
323 Vector<String> languageList;
324 languageList.reserveCapacity(1);
325
326 String audioTrackLanguage;
327 if (testingMode())
328 audioTrackLanguage = primaryAudioTrackLanguageOverride();
329 else
330 audioTrackLanguage = mediaElement->player()->languageOfPrimaryAudioTrack();
331
332 if (audioTrackLanguage.isEmpty())
333 return 0;
334
335 bool exactMatch;
336 if (trackHasOnlyForcedSubtitles) {
337 languageList.append(audioTrackLanguage);
338 size_t offset = indexOfBestMatchingLanguageInList(textTrackLanguage, languageList, exactMatch);
339
340 // Only consider a forced-only track if it IS in the same language as the primary audio track.
341 if (offset)
342 return 0;
343 } else {
344 languageList.append(defaultLanguage(ShouldMinimizeLanguages::No));
345
346 // Only enable a text track if the current audio track is NOT in the user's preferred language ...
347 size_t offset = indexOfBestMatchingLanguageInList(audioTrackLanguage, languageList, exactMatch);
348 if (!offset)
349 return 0;
350
351 // and the text track matches the user's preferred language.
352 offset = indexOfBestMatchingLanguageInList(textTrackLanguage, languageList, exactMatch);
353 if (offset)
354 return 0;
355 }
356
357 userPreferredCaptionLanguages = languageList;
358 }
359
360 int trackScore = 0;
361
362 if (userPrefersCaptions()) {
363 // When the user prefers accessibility tracks, rank is SDH, then CC, then subtitles.
364 if (track->kind() == TextTrack::Kind::Subtitles)
365 trackScore = 1;
366 else if (track->isClosedCaptions())
367 trackScore = 2;
368 else
369 trackScore = 3;
370 } else {
371 // When the user prefers translation tracks, rank is subtitles, then SDH, then CC tracks.
372 if (track->kind() == TextTrack::Kind::Subtitles)
373 trackScore = 3;
374 else if (!track->isClosedCaptions())
375 trackScore = 2;
376 else
377 trackScore = 1;
378 }
379
380 return trackScore + textTrackLanguageSelectionScore(track, userPreferredCaptionLanguages);
381}

383int CaptionUserPreferences::textTrackLanguageSelectionScore(TextTrack* track, const Vector<String>& preferredLanguages) const
384{
385 if (track->validBCP47Language().isEmpty())
386 return 0;
387
388 bool exactMatch;
389 size_t languageMatchIndex = indexOfBestMatchingLanguageInList(track->validBCP47Language(), preferredLanguages, exactMatch);
390 if (languageMatchIndex >= preferredLanguages.size())
391 return 0;
392
393 // Matching a track language is more important than matching track type, so this multiplier must be
394 // greater than the maximum value returned by textTrackSelectionScore.
395 int bonus = exactMatch ? 1 : 0;
396 return (preferredLanguages.size() + bonus - languageMatchIndex) * 10;
397}

399void CaptionUserPreferences::setCaptionsStyleSheetOverride(const String& override)
400{
401 if (override == m_captionsStyleSheetOverride)
402 return;
403
404 m_captionsStyleSheetOverride = override;
405 updateCaptionStyleSheetOverride();
406 if (!m_timer.isActive())
407 m_timer.startOneShot(0_s);

410void CaptionUserPreferences::updateCaptionStyleSheetOverride()
411{
412 String captionsOverrideStyleSheet = captionsStyleSheetOverride();
413 for (auto& page : m_pageGroup.pages())
414 page.setCaptionUserPreferencesStyleSheet(captionsOverrideStyleSheet);
415}
