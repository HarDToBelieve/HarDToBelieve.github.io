128CaptionUserPreferencesMediaAF::CaptionUserPreferencesMediaAF(PageGroup& group)
129 : CaptionUserPreferences(group)
130#if HAVE(MEDIA_ACCESSIBILITY_FRAMEWORK)
131 , m_updateStyleSheetTimer(*this, &CaptionUserPreferencesMediaAF::updateTimerFired)
132 , m_listeningForPreferenceChanges(false)

134{

136 if (!initialized) {

140 if (!MediaToolboxLibrary())

150 setUserPrefersCaptions(false);
151 setUserPrefersSubtitles(false);
152 setUserPrefersTextDescriptions(false);
153 endBlockingNotifications();
154 }

181void CaptionUserPreferencesMediaAF::setCaptionDisplayMode(CaptionUserPreferences::CaptionDisplayMode mode)
182{
183 if (testingMode() || !MediaAccessibilityLibrary()) {
184 CaptionUserPreferences::setCaptionDisplayMode(mode);
185 return;
186 }
187
188 if (captionDisplayMode() == Manual)
189 return;
190
191 MACaptionAppearanceDisplayType displayType = kMACaptionAppearanceDisplayTypeForcedOnly;
192 switch (mode) {
193 case Automatic:
194 displayType = kMACaptionAppearanceDisplayTypeAutomatic;
195 break;
196 case ForcedOnly:
197 displayType = kMACaptionAppearanceDisplayTypeForcedOnly;
198 break;
199 case AlwaysOn:
200 displayType = kMACaptionAppearanceDisplayTypeAlwaysOn;
201 break;
202 default:
203 ASSERT_NOT_REACHED();
204 break;
205 }
206
207 MACaptionAppearanceSetDisplayType(kMACaptionAppearanceDomainUser, displayType);
208}

210CaptionUserPreferences::CaptionDisplayMode CaptionUserPreferencesMediaAF::platformCaptionDisplayMode()
211{
212 MACaptionAppearanceDisplayType displayType = MACaptionAppearanceGetDisplayType(kMACaptionAppearanceDomainUser);
213 switch (displayType) {
214 case kMACaptionAppearanceDisplayTypeForcedOnly:
215 return ForcedOnly;
216
217 case kMACaptionAppearanceDisplayTypeAutomatic:
218 return Automatic;
219
220 case kMACaptionAppearanceDisplayTypeAlwaysOn:
221 return AlwaysOn;
222 }
223
224 ASSERT_NOT_REACHED();
225 return ForcedOnly;
226}

258void CaptionUserPreferencesMediaAF::setInterestedInCaptionPreferenceChanges()
259{
260 if (m_listeningForPreferenceChanges)
261 return;
262
263 if (!MediaAccessibilityLibrary())
264 return;
265
266 if (!kMAXCaptionAppearanceSettingsChangedNotification && !canLoad_MediaAccessibility_kMAAudibleMediaSettingsChangedNotification())
267 return;
268
269 m_listeningForPreferenceChanges = true;
270 m_registeringForNotification = true;
271
272 if (kMAXCaptionAppearanceSettingsChangedNotification)
273 CFNotificationCenterAddObserver(CFNotificationCenterGetLocalCenter(), this, userCaptionPreferencesChangedNotificationCallback, kMAXCaptionAppearanceSettingsChangedNotification, 0, CFNotificationSuspensionBehaviorCoalesce);
274 if (canLoad_MediaAccessibility_kMAAudibleMediaSettingsChangedNotification())
275 CFNotificationCenterAddObserver(CFNotificationCenterGetLocalCenter(), this, userCaptionPreferencesChangedNotificationCallback, kMAAudibleMediaSettingsChangedNotification, 0, CFNotificationSuspensionBehaviorCoalesce);
276 m_registeringForNotification = false;
277
278 // Generating and registering the caption stylesheet can be expensive and this method is called indirectly when the parser creates an audio or
279 // video element, so do it after a brief pause.
280 m_updateStyleSheetTimer.startOneShot(0_s);
281}

285 if (m_registeringForNotification)
286 return;
287
288 if (m_listeningForPreferenceChanges)
289 updateCaptionStyleSheetOverride();
290
291 CaptionUserPreferences::captionPreferencesChanged();
292}

294String CaptionUserPreferencesMediaAF::captionsWindowCSS() const
295{
296 MACaptionAppearanceBehavior behavior;
297 RetainPtr<CGColorRef> color = adoptCF(MACaptionAppearanceCopyWindowColor(kMACaptionAppearanceDomainUser, &behavior));
298
299 Color windowColor(color.get());
300 if (!windowColor.isValid())
301 windowColor = Color::transparentBlack;
302
303 bool important = behavior == kMACaptionAppearanceBehaviorUseValue;
304 CGFloat opacity = MACaptionAppearanceGetWindowOpacity(kMACaptionAppearanceDomainUser, &behavior);
305 if (!important)
306 important = behavior == kMACaptionAppearanceBehaviorUseValue;
307 String windowStyle = colorPropertyCSS(CSSPropertyBackgroundColor, windowColor.colorWithAlpha(opacity), important);
308
309 if (!opacity)
310 return windowStyle;
311
312 return makeString(windowStyle, getPropertyNameString(CSSPropertyPadding), ": .4em !important;");
313}

315String CaptionUserPreferencesMediaAF::captionsBackgroundCSS() const
316{
317 // This default value must be the same as the one specified in mediaControls.css for -webkit-media-text-track-past-nodes
318 // and webkit-media-text-track-future-nodes.
319 constexpr auto defaultBackgroundColor = Color::black.colorWithAlphaByte(204);
320
321 MACaptionAppearanceBehavior behavior;
322
323 RetainPtr<CGColorRef> color = adoptCF(MACaptionAppearanceCopyBackgroundColor(kMACaptionAppearanceDomainUser, &behavior));
324 Color backgroundColor(color.get());
325 if (!backgroundColor.isValid())
326 backgroundColor = defaultBackgroundColor;
327
328 bool important = behavior == kMACaptionAppearanceBehaviorUseValue;
329 CGFloat opacity = MACaptionAppearanceGetBackgroundOpacity(kMACaptionAppearanceDomainUser, &behavior);
330 if (!important)
331 important = behavior == kMACaptionAppearanceBehaviorUseValue;
332 return colorPropertyCSS(CSSPropertyBackgroundColor, backgroundColor.colorWithAlpha(opacity), important);
333}

384bool CaptionUserPreferencesMediaAF::captionStrokeWidthForFont(float fontSize, const String& language, float& strokeWidth, bool& important) const
385{
386 if (!canLoad_MediaAccessibility_MACaptionAppearanceCopyFontDescriptorWithStrokeForStyle())
387 return false;
388
389 MACaptionAppearanceBehavior behavior;
390 auto trackLanguage = language.createCFString();
391 CGFloat strokeWidthPt;
392
393 auto fontDescriptor = adoptCF(MACaptionAppearanceCopyFontDescriptorWithStrokeForStyle(kMACaptionAppearanceDomainUser, &behavior, kMACaptionAppearanceFontStyleDefault, trackLanguage.get(), fontSize, &strokeWidthPt));
394
395 if (!fontDescriptor)
396 return false;
397
398 // Since only half of the stroke is visible because the stroke is drawn before the fill, we double the stroke width here.
399 strokeWidth = strokeWidthPt * 2;
400 important = behavior == kMACaptionAppearanceBehaviorUseValue;
401 return true;
402}

404String CaptionUserPreferencesMediaAF::captionsTextEdgeCSS() const
405{
406 static NeverDestroyed<const String> edgeStyleRaised(MAKE_STATIC_STRING_IMPL(" -.1em -.1em .16em "));
407 static NeverDestroyed<const String> edgeStyleDepressed(MAKE_STATIC_STRING_IMPL(" .1em .1em .16em "));
408 static NeverDestroyed<const String> edgeStyleDropShadow(MAKE_STATIC_STRING_IMPL(" 0 .1em .16em "));
409
410 MACaptionAppearanceBehavior behavior;
411 MACaptionAppearanceTextEdgeStyle textEdgeStyle = MACaptionAppearanceGetTextEdgeStyle(kMACaptionAppearanceDomainUser, &behavior);
412
413 if (textEdgeStyle == kMACaptionAppearanceTextEdgeStyleUndefined || textEdgeStyle == kMACaptionAppearanceTextEdgeStyleNone)
414 return emptyString();
415
416 StringBuilder builder;
417 bool important = behavior == kMACaptionAppearanceBehaviorUseValue;
418 if (textEdgeStyle == kMACaptionAppearanceTextEdgeStyleRaised)
419 appendCSS(builder, CSSPropertyTextShadow, makeString(edgeStyleRaised.get(), " black"), important);
420 else if (textEdgeStyle == kMACaptionAppearanceTextEdgeStyleDepressed)
421 appendCSS(builder, CSSPropertyTextShadow, makeString(edgeStyleDepressed.get(), " black"), important);
422 else if (textEdgeStyle == kMACaptionAppearanceTextEdgeStyleDropShadow)
423 appendCSS(builder, CSSPropertyTextShadow, makeString(edgeStyleDropShadow.get(), " black"), important);
424
425 if (textEdgeStyle == kMACaptionAppearanceTextEdgeStyleDropShadow || textEdgeStyle == kMACaptionAppearanceTextEdgeStyleUniform) {
426 appendCSS(builder, CSSPropertyStrokeColor, "black", important);
427 appendCSS(builder, CSSPropertyPaintOrder, getValueName(CSSValueStroke), important);
428 appendCSS(builder, CSSPropertyStrokeLinejoin, getValueName(CSSValueRound), important);
429 appendCSS(builder, CSSPropertyStrokeLinecap, getValueName(CSSValueRound), important);
430 }
431
432 return builder.toString();
433}

435String CaptionUserPreferencesMediaAF::captionsDefaultFontCSS() const
436{
437 MACaptionAppearanceBehavior behavior;
438
439 auto font = adoptCF(MACaptionAppearanceCopyFontDescriptorForStyle(kMACaptionAppearanceDomainUser, &behavior, kMACaptionAppearanceFontStyleDefault));
440 if (!font)
441 return emptyString();
442
443 auto name = adoptCF(static_cast<CFStringRef>(CTFontDescriptorCopyAttribute(font.get(), kCTFontNameAttribute)));
444 if (!name)
445 return emptyString();
446
447 StringBuilder builder;
448 builder.append("font-family: \"", name.get(), '"');
449 if (auto cascadeList = adoptCF(static_cast<CFArrayRef>(CTFontDescriptorCopyAttribute(font.get(), kCTFontCascadeListAttribute)))) {
450 for (CFIndex i = 0; i < CFArrayGetCount(cascadeList.get()); i++) {
451 auto fontCascade = static_cast<CTFontDescriptorRef>(CFArrayGetValueAtIndex(cascadeList.get(), i));
452 if (!fontCascade)
453 continue;
454 auto fontCascadeName = adoptCF(static_cast<CFStringRef>(CTFontDescriptorCopyAttribute(fontCascade, kCTFontNameAttribute)));
455 if (!fontCascadeName)
456 continue;
457 builder.append(", \"", fontCascadeName.get(), '"');
458 }
459 }
460 builder.append(behavior == kMACaptionAppearanceBehaviorUseValue ? " !important;" : ";");
461 return builder.toString();
462}

464float CaptionUserPreferencesMediaAF::captionFontSizeScaleAndImportance(bool& important) const
465{
466 if (testingMode() || !MediaAccessibilityLibrary())
467 return CaptionUserPreferences::captionFontSizeScaleAndImportance(important);
468
469 MACaptionAppearanceBehavior behavior;
470 CGFloat characterScale = CaptionUserPreferences::captionFontSizeScaleAndImportance(important);
471 CGFloat scaleAdjustment = MACaptionAppearanceGetRelativeCharacterSize(kMACaptionAppearanceDomainUser, &behavior);
472
473 if (!scaleAdjustment)
474 return characterScale;
475
476 important = behavior == kMACaptionAppearanceBehaviorUseValue;
477#if defined(__LP64__) && __LP64__
478 return narrowPrecisionToFloat(scaleAdjustment * characterScale);
479#else
480 return scaleAdjustment * characterScale;
481#endif
482}

497Vector<String> CaptionUserPreferencesMediaAF::preferredLanguages() const
498{
499 auto preferredLanguages = CaptionUserPreferences::preferredLanguages();
500 if (testingMode() || !MediaAccessibilityLibrary())
501 return preferredLanguages;
502
503 if (cachedPreferredLanguages().has_value())
504 return cachedPreferredLanguages().value();
505
506 auto captionLanguages = platformPreferredLanguages();
507 if (!captionLanguages.size())
508 return preferredLanguages;
509
510 Vector<String> captionAndPreferredLanguages;
511 captionAndPreferredLanguages.reserveInitialCapacity(captionLanguages.size() + preferredLanguages.size());
512 captionAndPreferredLanguages.appendVector(WTFMove(captionLanguages));
513 captionAndPreferredLanguages.appendVector(WTFMove(preferredLanguages));
514 return captionAndPreferredLanguages;
515}

517Vector<String> CaptionUserPreferencesMediaAF::platformPreferredLanguages()
518{
519 auto captionLanguages = adoptCF(MACaptionAppearanceCopySelectedLanguages(kMACaptionAppearanceDomainUser));
520 CFIndex captionLanguagesCount = captionLanguages ? CFArrayGetCount(captionLanguages.get()) : 0;
521
522 Vector<String> preferredLanguages;
523 preferredLanguages.reserveInitialCapacity(captionLanguagesCount);
524 for (CFIndex i = 0; i < captionLanguagesCount; i++)
525 preferredLanguages.uncheckedAppend(static_cast<CFStringRef>(CFArrayGetValueAtIndex(captionLanguages.get(), i)));
526
527 return preferredLanguages;

563String CaptionUserPreferencesMediaAF::captionsStyleSheetOverride() const
564{
565 if (testingMode())
566 return CaptionUserPreferences::captionsStyleSheetOverride();
567
568 StringBuilder captionsOverrideStyleSheet;
569
570#if HAVE(MEDIA_ACCESSIBILITY_FRAMEWORK)
571 if (!MediaAccessibilityLibrary())
572 return CaptionUserPreferences::captionsStyleSheetOverride();
573
574 String captionsColor = captionsTextColorCSS();
575 String edgeStyle = captionsTextEdgeCSS();
576 String fontName = captionsDefaultFontCSS();
577 String background = captionsBackgroundCSS();
578 if (!background.isEmpty() || !captionsColor.isEmpty() || !edgeStyle.isEmpty() || !fontName.isEmpty())
579 captionsOverrideStyleSheet.append(" ::", TextTrackCue::cueShadowPseudoId(), '{', background, captionsColor, edgeStyle, fontName, '}');
580
581 String windowColor = captionsWindowCSS();
582 String windowCornerRadius = windowRoundedCornerRadiusCSS();
583 if (!windowColor.isEmpty() || !windowCornerRadius.isEmpty())
584 captionsOverrideStyleSheet.append(" ::", TextTrackCue::cueBackdropShadowPseudoId(), '{', windowColor, windowCornerRadius, '}');
585#endif // HAVE(MEDIA_ACCESSIBILITY_FRAMEWORK)
586
587 LOG(Media, "CaptionUserPreferencesMediaAF::captionsStyleSheetOverrideSetting style to:\n%s", captionsOverrideStyleSheet.toString().utf8().data());
588
589 return captionsOverrideStyleSheet.toString();
590}

592static String languageIdentifier(const String& languageCode)
593{
594 if (languageCode.isEmpty())
595 return languageCode;
596
597 String lowercaseLanguageCode = languageCode.convertToASCIILowercase();
598
599 // Need 2U here to disambiguate String::operator[] from operator(NSString*, int)[] in a production build.
600 if (lowercaseLanguageCode.length() >= 3 && (lowercaseLanguageCode[2U] == '_' || lowercaseLanguageCode[2U] == '-'))
601 lowercaseLanguageCode.truncate(2);
602
603 return lowercaseLanguageCode;
604}

606static String addTextTrackKindDisplayNameIfNeeded(const TextTrack& track, const String& text)
607{
608 String result;
609
610 if (track.isClosedCaptions()) {
611 if (!text.contains(textTrackKindClosedCaptionsDisplayName()))
612 result = addTextTrackKindClosedCaptionsSuffix(text);
613 } else {
614 switch (track.kind()) {
615 case TextTrack::Kind::Subtitles:
616 // Subtitle text tracks are only shown in a dedicated "Subtitle" grouping, meaning it
617 // would look odd to add it as a suffix (e.g. "Subtitles > English Subtitles").
618 break;
619
620 case TextTrack::Kind::Captions:
621 if (!text.contains(textTrackKindCaptionsDisplayName()))
622 result = addTextTrackKindCaptionsSuffix(text);
623 break;
624
625 case TextTrack::Kind::Descriptions:
626 if (!text.contains(textTrackKindDescriptionsDisplayName()))
627 result = addTextTrackKindDescriptionsSuffix(text);
628 break;
629
630 case TextTrack::Kind::Chapters:
631 if (!text.contains(textTrackKindChaptersDisplayName()))
632 result = addTextTrackKindChaptersSuffix(text);
633 break;
634
635 case TextTrack::Kind::Metadata:
636 if (!text.contains(textTrackKindMetadataDisplayName()))
637 result = addTextTrackKindMetadataSuffix(text);
638 break;
639
640 case TextTrack::Kind::Forced:
641 // Handled below.
642 break;
643 }
644 }
645
646 if (result.isEmpty())
647 result = text;
648
649 if (track.isSDH() && !text.contains(textTrackKindSDHDisplayName()))
650 result = addTextTrackKindSDHSuffix(result);
651
652 if (track.isEasyToRead() && !text.contains(textTrackKindEasyReaderDisplayName()))
653 result = addTextTrackKindEasyReaderSuffix(result);
654
655 if ((track.containsOnlyForcedSubtitles() || track.kind() == TextTrack::Kind::Forced) && !text.contains(textTrackKindForcedDisplayName()))
656 result = addTextTrackKindForcedSuffix(result);
657
658 return result;
659}

661static String addAudioTrackKindDisplayNameIfNeeded(const AudioTrack& track, const String& text)
662{
663 // Audio tracks are only shown in a dedicated "Languages" grouping, meaning it would look odd to
664 // add it as a suffix (e.g. "Languages > English Language").
665
666 if ((track.kind() == AudioTrack::descriptionKeyword() || track.kind() == AudioTrack::mainDescKeyword()) && !text.contains(audioTrackKindDescriptionsDisplayName()))
667 return addAudioTrackKindDescriptionsSuffix(text);
668
669 if (track.kind() == AudioTrack::commentaryKeyword() && !text.contains(audioTrackKindCommentaryDisplayName()))
670 return addAudioTrackKindCommentarySuffix(text);
671
672 return text;
673}

676{

693static String trackDisplayName(const TrackBase& track)
694{
695 if (&track == &TextTrack::captionMenuOffItem())
696 return textTrackOffMenuItemText();
697 if (&track == &TextTrack::captionMenuAutomaticItem())
698 return textTrackAutomaticMenuItemText();
699
700 String result;
701
702 String label = track.label();
703 String trackLanguageIdentifier = track.validBCP47Language();
704
705 auto preferredLanguages = userPreferredLanguages(ShouldMinimizeLanguages::No);
706 auto defaultLanguage = !preferredLanguages.isEmpty() ? preferredLanguages[0] : emptyString(); // This matches `WTF::defaultLanguage`.
707 auto currentLocale = adoptCF(CFLocaleCreate(kCFAllocatorDefault, defaultLanguage.createCFString().get()));
708 auto localeIdentifier = adoptCF(CFLocaleCreateCanonicalLocaleIdentifierFromString(kCFAllocatorDefault, trackLanguageIdentifier.createCFString().get()));
709 String languageDisplayName = adoptCF(CFLocaleCopyDisplayNameForPropertyValue(currentLocale.get(), kCFLocaleLanguageCode, localeIdentifier.get())).get();
710
711 bool exactMatch;
712 bool matchesDefaultLanguage = !indexOfBestMatchingLanguageInList(trackLanguageIdentifier, { defaultLanguage }, exactMatch);
713
714 // Only add the language if the track isn't for the default language or the track's label doesn't already contain the language.
715 if (!label.isEmpty() && (matchesDefaultLanguage || (!languageDisplayName.isEmpty() && label.contains(languageDisplayName))))
716 result = addTrackKindDisplayNameIfNeeded(track, label);
717 else {
718 String languageAndLocale = adoptCF(CFLocaleCopyDisplayNameForPropertyValue(currentLocale.get(), kCFLocaleIdentifier, trackLanguageIdentifier.createCFString().get())).get();
719 if (!languageAndLocale.isEmpty())
720 result = languageAndLocale;
721 else if (!languageDisplayName.isEmpty())
722 result = languageDisplayName;
723 else
724 result = localeIdentifier.get();
725
726 result = addTrackKindDisplayNameIfNeeded(track, result);
727
728 if (!label.isEmpty() && !result.contains(label))
729 result = addTrackLabelAsSuffix(result, label);
730 }
731
732 if (result.isEmpty())
733 return trackNoLabelText();
734
735 return result;
736}

748static bool textTrackCompare(const RefPtr<TextTrack>& a, const RefPtr<TextTrack>& b)
749{
750 auto preferredLanguages = userPreferredLanguages(ShouldMinimizeLanguages::No);
751
752 bool aExactMatch;
753 auto aUserLanguageIndex = indexOfBestMatchingLanguageInList(a->validBCP47Language(), preferredLanguages, aExactMatch);
754
755 bool bExactMatch;
756 auto bUserLanguageIndex = indexOfBestMatchingLanguageInList(b->validBCP47Language(), preferredLanguages, bExactMatch);
757
758 if (aUserLanguageIndex != bUserLanguageIndex)
759 return aUserLanguageIndex < bUserLanguageIndex;
760 if (aExactMatch != bExactMatch)
761 return aExactMatch;
762
763 String aLanguageDisplayName = displayNameForLanguageLocale(languageIdentifier(a->validBCP47Language()));
764 String bLanguageDisplayName = displayNameForLanguageLocale(languageIdentifier(b->validBCP47Language()));
765
766 Collator collator;
767
768 // Tracks not in the user's preferred language sort first by language ...
769 if (auto languageDisplayNameComparison = collator.collate(aLanguageDisplayName, bLanguageDisplayName))
770 return languageDisplayNameComparison < 0;
771
772 // ... but when tracks have the same language, main program content sorts next highest ...
773 bool aIsMainContent = a->isMainProgramContent();
774 bool bIsMainContent = b->isMainProgramContent();
775 if (aIsMainContent != bIsMainContent)
776 return aIsMainContent;
777
778 // ... and main program tracks sort higher than CC tracks ...
779 bool aIsCC = a->isClosedCaptions();
780 bool bIsCC = b->isClosedCaptions();
781 if (aIsCC != bIsCC)
782 return aIsCC;
783
784 // ... and tracks of the same type and language sort by the menu item text.
785 if (auto trackDisplayComparison = collator.collate(trackDisplayName(*a), trackDisplayName(*b)))
786 return trackDisplayComparison < 0;
787
788 // ... and if the menu item text is the same, compare the unique IDs
789 return a->uniqueId() < b->uniqueId();
790}

792Vector<RefPtr<AudioTrack>> CaptionUserPreferencesMediaAF::sortedTrackListForMenu(AudioTrackList* trackList)
793{
794 ASSERT(trackList);
795
796 Vector<RefPtr<AudioTrack>> tracksForMenu;
797
798 for (unsigned i = 0, length = trackList->length(); i < length; ++i) {
799 AudioTrack* track = trackList->item(i);
800 String language = displayNameForLanguageLocale(track->validBCP47Language());
801 tracksForMenu.append(track);
802 }
803
804 Collator collator;
805
806 std::sort(tracksForMenu.begin(), tracksForMenu.end(), [&] (auto& a, auto& b) {
807 if (auto trackDisplayComparison = collator.collate(trackDisplayName(*a), trackDisplayName(*b)))
808 return trackDisplayComparison < 0;
809
810 return a->uniqueId() < b->uniqueId();
811 });
812
813 return tracksForMenu;
814}

816Vector<RefPtr<TextTrack>> CaptionUserPreferencesMediaAF::sortedTrackListForMenu(TextTrackList* trackList, HashSet<TextTrack::Kind> kinds)
817{
818 ASSERT(trackList);
819
820 Vector<RefPtr<TextTrack>> tracksForMenu;
821 HashSet<String> languagesIncluded;
822 CaptionDisplayMode displayMode = captionDisplayMode();
823 bool prefersAccessibilityTracks = userPrefersCaptions();
824 bool filterTrackList = shouldFilterTrackMenu();
825 bool requestingCaptionsOrDescriptionsOrSubtitles = kinds.contains(TextTrack::Kind::Subtitles) || kinds.contains(TextTrack::Kind::Captions) || kinds.contains(TextTrack::Kind::Descriptions);
826
827 for (unsigned i = 0, length = trackList->length(); i < length; ++i) {
828 TextTrack* track = trackList->item(i);
829 if (!kinds.contains(track->kind()))
830 continue;
831
832 String language = displayNameForLanguageLocale(track->validBCP47Language());
833
834 if (displayMode == Manual) {
835 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - adding '%s' track with language '%s' because selection mode is 'manual'", track->kindKeyword().string().utf8().data(), language.utf8().data());
836 tracksForMenu.append(track);
837 continue;
838 }
839
840 if (requestingCaptionsOrDescriptionsOrSubtitles) {
841 if (track->containsOnlyForcedSubtitles()) {
842 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - skipping '%s' track with language '%s' because it contains only forced subtitles", track->kindKeyword().string().utf8().data(), language.utf8().data());
843 continue;
844 }
845
846 if (track->isEasyToRead()) {
847 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - adding '%s' track with language '%s' because it is 'easy to read'", track->kindKeyword().string().utf8().data(), language.utf8().data());
848 if (!language.isEmpty())
849 languagesIncluded.add(language);
850 tracksForMenu.append(track);
851 continue;
852 }
853
854 if (track->mode() == TextTrack::Mode::Showing) {
855 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - adding '%s' track with language '%s' because it is already visible", track->kindKeyword().string().utf8().data(), language.utf8().data());
856 if (!language.isEmpty())
857 languagesIncluded.add(language);
858 tracksForMenu.append(track);
859 continue;
860 }
861
862 if (!language.isEmpty() && track->isMainProgramContent()) {
863 bool isAccessibilityTrack = track->kind() == TextTrack::Kind::Captions;
864 if (prefersAccessibilityTracks) {
865 // In the first pass, include only caption tracks if the user prefers accessibility tracks.
866 if (!isAccessibilityTrack && filterTrackList) {
867 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - skipping '%s' track with language '%s' because it is NOT an accessibility track", track->kindKeyword().string().utf8().data(), language.utf8().data());
868 continue;
869 }
870 } else {
871 // In the first pass, only include the first non-CC or SDH track with each language if the user prefers translation tracks.
872 if (isAccessibilityTrack && filterTrackList) {
873 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - skipping '%s' track with language '%s' because it is an accessibility track", track->kindKeyword().string().utf8().data(), language.utf8().data());
874 continue;
875 }
876 if (languagesIncluded.contains(language) && filterTrackList) {
877 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - skipping '%s' track with language '%s' because it is not the first with this language", track->kindKeyword().string().utf8().data(), language.utf8().data());
878 continue;
879 }
880 }
881 }
882
883 if (!language.isEmpty())
884 languagesIncluded.add(language);
885 }
886
887 tracksForMenu.append(track);
888
889 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - adding '%s' track with language '%s', is%s main program content", track->kindKeyword().string().utf8().data(), language.utf8().data(), track->isMainProgramContent() ? "" : " NOT");
890 }
891
892 if (requestingCaptionsOrDescriptionsOrSubtitles) {
893 // Now that we have filtered for the user's accessibility/translation preference, add all tracks with a unique language without regard to track type.
894 for (unsigned i = 0, length = trackList->length(); i < length; ++i) {
895 TextTrack* track = trackList->item(i);
896 String language = displayNameForLanguageLocale(track->language());
897
898 if (tracksForMenu.contains(track))
899 continue;
900
901 if (!kinds.contains(track->kind()))
902 continue;
903
904 // All candidates with no languge were added the first time through.
905 if (language.isEmpty())
906 continue;
907
908 if (track->containsOnlyForcedSubtitles())
909 continue;
910
911 if (!languagesIncluded.contains(language) && track->isMainProgramContent()) {
912 languagesIncluded.add(language);
913 tracksForMenu.append(track);
914 LOG(Media, "CaptionUserPreferencesMediaAF::sortedTrackListForMenu - adding '%s' track with language '%s' because it is the only track with this language", track->kindKeyword().string().utf8().data(), language.utf8().data());
915 }
916 }
917 }
918
919 if (tracksForMenu.isEmpty())
920 return tracksForMenu;
921
922 std::sort(tracksForMenu.begin(), tracksForMenu.end(), textTrackCompare);
923
924 if (requestingCaptionsOrDescriptionsOrSubtitles) {
925 tracksForMenu.insert(0, &TextTrack::captionMenuOffItem());
926 tracksForMenu.insert(1, &TextTrack::captionMenuAutomaticItem());
927 }
928
929 return tracksForMenu;
930}
