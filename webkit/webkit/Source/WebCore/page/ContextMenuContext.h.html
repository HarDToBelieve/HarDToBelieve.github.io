27#pragma once
28
29#if ENABLE(CONTEXT_MENUS)
30
31#include "HitTestResult.h"
32#include "Image.h"
33
34namespace WebCore {
35
36class ContextMenuContext {
37public:
38 enum class Type : uint8_t {
39 ContextMenu,
40#if ENABLE(SERVICE_CONTROLS)
41 ServicesMenu,
42#endif // ENABLE(SERVICE_CONTROLS)
43#if ENABLE(MEDIA_CONTROLS_CONTEXT_MENUS)
44 MediaControls,
45#endif // ENABLE(MEDIA_CONTROLS_CONTEXT_MENUS)
46 };
47
48 ContextMenuContext();
49 ContextMenuContext(Type, const HitTestResult&);
50
51 Type type() const { return m_type; }
52
53 const HitTestResult& hitTestResult() const { return m_hitTestResult; }
54
55 void setSelectedText(const String& selectedText) { m_selectedText = selectedText; }
56 const String& selectedText() const { return m_selectedText; }
57
58#if ENABLE(SERVICE_CONTROLS)
59 void setControlledImage(Image* controlledImage) { m_controlledImage = controlledImage; }
60 Image* controlledImage() const { return m_controlledImage.get(); }
61#endif
62
63private:
64 Type m_type { Type::ContextMenu };
65 HitTestResult m_hitTestResult;
66 String m_selectedText;
67
68#if ENABLE(SERVICE_CONTROLS)
69 RefPtr<Image> m_controlledImage;
70#endif
71};
72
73} // namespace WebCore
74
75namespace WTF {
76
77template<> struct EnumTraits<WebCore::ContextMenuContext::Type> {
78 using values = EnumValues<
79 WebCore::ContextMenuContext::Type,
80 WebCore::ContextMenuContext::Type::ContextMenu
81#if ENABLE(SERVICE_CONTROLS)
82 , WebCore::ContextMenuContext::Type::ServicesMenu
83#endif // ENABLE(SERVICE_CONTROLS)
84#if ENABLE(MEDIA_CONTROLS_CONTEXT_MENUS)
85 , WebCore::ContextMenuContext::Type::MediaControls
86#endif // ENABLE(MEDIA_CONTROLS_CONTEXT_MENUS)
87 >;
88};
89
90} // namespace WTF

92#endif // ENABLE(CONTEXT_MENUS)
