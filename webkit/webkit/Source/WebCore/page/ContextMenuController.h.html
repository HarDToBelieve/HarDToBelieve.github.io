27
28#if ENABLE(CONTEXT_MENUS)
29
30#include "ContextMenuContext.h"
31#include "ContextMenuItem.h"
32#include "HitTestRequest.h"
33#include <wtf/OptionSet.h>
34
35namespace WebCore {
36
37class ContextMenuClient;
38class ContextMenuProvider;
39class Event;
40class HitTestResult;
41class Page;
42
43class ContextMenuController {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 ContextMenuController(Page&, ContextMenuClient&);
47 ~ContextMenuController();
48
49 Page& page() { return m_page; }
50 ContextMenuClient& client() { return m_client; }
51
52 ContextMenu* contextMenu() const { return m_contextMenu.get(); }
53 WEBCORE_EXPORT void clearContextMenu();
54
55 void handleContextMenuEvent(Event&);
56 void showContextMenu(Event&, ContextMenuProvider&);
57
58 void populate();
59 WEBCORE_EXPORT void contextMenuItemSelected(ContextMenuAction, const String& title);
60 void addInspectElementItem();
61
62 WEBCORE_EXPORT void checkOrEnableIfNeeded(ContextMenuItem&) const;
63
64 void setContextMenuContext(const ContextMenuContext& context) { m_context = context; }
65 const ContextMenuContext& context() const { return m_context; }
66 const HitTestResult& hitTestResult() const { return m_context.hitTestResult(); }
67
68#if USE(ACCESSIBILITY_CONTEXT_MENUS)
69 void showContextMenuAt(Frame&, const IntPoint& clickPoint);
70#endif
71
72private:
73 std::unique_ptr<ContextMenu> maybeCreateContextMenu(Event&, OptionSet<HitTestRequest::Type> hitType, ContextMenuContext::Type);
74 void showContextMenu(Event&);
75
76 void appendItem(ContextMenuItem&, ContextMenu* parentMenu);
77
78 void createAndAppendFontSubMenu(ContextMenuItem&);
79 void createAndAppendSpellingAndGrammarSubMenu(ContextMenuItem&);
80 void createAndAppendSpellingSubMenu(ContextMenuItem&);
81 void createAndAppendSpeechSubMenu(ContextMenuItem&);
82 void createAndAppendWritingDirectionSubMenu(ContextMenuItem&);
83 void createAndAppendTextDirectionSubMenu(ContextMenuItem&);
84 void createAndAppendSubstitutionsSubMenu(ContextMenuItem&);
85 void createAndAppendTransformationsSubMenu(ContextMenuItem&);
86#if PLATFORM(GTK)
87 void createAndAppendUnicodeSubMenu(ContextMenuItem&);
88#endif
89
90 Page& m_page;
91 ContextMenuClient& m_client;
92 std::unique_ptr<ContextMenu> m_contextMenu;
93 RefPtr<ContextMenuProvider> m_menuProvider;
94 ContextMenuContext m_context;
95 bool m_isHandlingContextMenuEvent { false };
96};
97
98} // namespace WebCore
99
100#endif // ENABLE(CONTEXT_MENUS)
