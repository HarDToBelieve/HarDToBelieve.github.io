51static const Seconds minIntervalForNonUserObservableChangeTimers { 1_s }; // Empirically determined to maximize battery life.
52static const int maxTimerNestingLevel = 5;
53
54class DOMTimerFireState {
55public:
56 DOMTimerFireState(ScriptExecutionContext& context, int nestingLevel)
57 : m_context(context)
58 , m_contextIsDocument(is<Document>(m_context))
59 // For worker threads, don't update the current DOMTimerFireState.
60 // Setting this from workers would not be thread-safe, and its not relevant to current uses.
61 , m_initialDOMTreeVersion(m_contextIsDocument ? downcast<Document>(m_context).domTreeVersion() : 0)
62 , m_previous(m_contextIsDocument ? std::exchange(current, this) : nullptr)
63 {
64 m_context.setTimerNestingLevel(nestingLevel);
65 }
66
67 ~DOMTimerFireState()
68 {
69 if (m_contextIsDocument)
70 current = m_previous;
71 m_context.setTimerNestingLevel(0);
72 }
73
74 Document* contextDocument() const { return m_contextIsDocument ? &downcast<Document>(m_context) : nullptr; }
75
76 void setScriptMadeUserObservableChanges() { m_scriptMadeUserObservableChanges = true; }
77 void setScriptMadeNonUserObservableChanges() { m_scriptMadeNonUserObservableChanges = true; }
78
79 bool scriptMadeNonUserObservableChanges() const { return m_scriptMadeNonUserObservableChanges; }
80 bool scriptMadeUserObservableChanges() const
81 {
82 if (m_scriptMadeUserObservableChanges)
83 return true;
84
85 Document* document = contextDocument();
86 // To be conservative, we also consider any DOM Tree change to be user observable.
87 return document && document->domTreeVersion() != m_initialDOMTreeVersion;
88 }
89
90 static DOMTimerFireState* current;
91
92private:
93 ScriptExecutionContext& m_context;
94 bool m_contextIsDocument;
95 bool m_scriptMadeNonUserObservableChanges { false };
96 bool m_scriptMadeUserObservableChanges { false };
97 uint64_t m_initialDOMTreeVersion;
98 DOMTimerFireState* m_previous;
99};

103struct NestedTimersMap {
104 typedef HashMap<int, Ref<DOMTimer>>::const_iterator const_iterator;
105
106 static NestedTimersMap* instanceForContext(ScriptExecutionContext& context)
107 {
108 // For worker threads, we don't use NestedTimersMap as doing so would not
109 // be thread safe.
110 if (is<Document>(context))
111 return &instance();
112 return nullptr;
113 }
114
115 void startTracking()
116 {
117 // Make sure we start with an empty HashMap. In theory, it is possible the HashMap is not
118 // empty if a timer fires during the execution of another timer (may happen with the
119 // in-process Web Inspector).
120 nestedTimers.clear();
121 isTrackingNestedTimers = true;
122 }
123
124 void stopTracking()
125 {
126 isTrackingNestedTimers = false;
127 nestedTimers.clear();
128 }
129
130 void add(int timeoutId, Ref<DOMTimer>&& timer)
131 {
132 if (isTrackingNestedTimers)
133 nestedTimers.add(timeoutId, WTFMove(timer));
134 }
135
136 void remove(int timeoutId)
137 {
138 if (isTrackingNestedTimers)
139 nestedTimers.remove(timeoutId);
140 }
141
142 const_iterator begin() const { return nestedTimers.begin(); }
143 const_iterator end() const { return nestedTimers.end(); }
144
145private:
146 static NestedTimersMap& instance()
147 {
148 static NeverDestroyed<NestedTimersMap> map;
149 return map;
150 }
151
152 static bool isTrackingNestedTimers;
153 HashMap<int /* timeoutId */, Ref<DOMTimer>> nestedTimers;
154};

175int DOMTimer::install(ScriptExecutionContext& context, std::unique_ptr<ScheduledAction> action, Seconds timeout, bool singleShot)
176{
177 Ref<DOMTimer> timer = adoptRef(*new DOMTimer(context, WTFMove(action), timeout, singleShot));
178 timer->suspendIfNeeded();
179
180 // Keep asking for the next id until we're given one that we don't already have.
181 do {
182 timer->m_timeoutId = context.circularSequentialID();
183 } while (!context.addTimeout(timer->m_timeoutId, timer.get()));
184
185 InspectorInstrumentation::didInstallTimer(context, timer->m_timeoutId, timeout, singleShot);
186
187 // Keep track of nested timer installs.
188 if (NestedTimersMap* nestedTimers = NestedTimersMap::instanceForContext(context))
189 nestedTimers->add(timer->m_timeoutId, timer.get());
190#if PLATFORM(IOS_FAMILY)
191 if (is<Document>(context)) {
192 auto& document = downcast<Document>(context);
193 document.contentChangeObserver().didInstallDOMTimer(timer.get(), timeout, singleShot);
194 if (DeferDOMTimersForScope::isDeferring())
195 document.domTimerHoldingTank().add(timer.get());
196 }
197#endif
198 return timer->m_timeoutId;
199}

201void DOMTimer::removeById(ScriptExecutionContext& context, int timeoutId)
202{
203 // timeout IDs have to be positive, and 0 and -1 are unsafe to
204 // even look up since they are the empty and deleted value
205 // respectively
206 if (timeoutId <= 0)
207 return;
208
209#if PLATFORM(IOS_FAMILY)
210 if (is<Document>(context)) {
211 auto& document = downcast<Document>(context);
212 if (auto* timer = document.findTimeout(timeoutId)) {
213 document.contentChangeObserver().didRemoveDOMTimer(*timer);
214 if (auto* holdingTank = document.domTimerHoldingTankIfExists())
215 holdingTank->remove(*timer);
216 }
217 }
218#endif
219
220 if (NestedTimersMap* nestedTimers = NestedTimersMap::instanceForContext(context))
221 nestedTimers->remove(timeoutId);
222
223 InspectorInstrumentation::didRemoveTimer(context, timeoutId);
224 context.removeTimeout(timeoutId);
225}

235void DOMTimer::updateThrottlingStateIfNecessary(const DOMTimerFireState& fireState)
236{
237 Document* contextDocument = fireState.contextDocument();
238 // We don't throttle timers in worker threads.
239 if (!contextDocument)
240 return;
241
242 if (UNLIKELY(!isDOMTimersThrottlingEnabled(*contextDocument))) {
243 if (m_throttleState == ShouldThrottle) {
244 // Unthrottle the timer in case it was throttled before the setting was updated.
245 LOG(DOMTimers, "%p - Unthrottling DOM timer because throttling was disabled via settings.", this);
246 m_throttleState = ShouldNotThrottle;
247 updateTimerIntervalIfNecessary();
248 }
249 return;
250 }
251
252 if (fireState.scriptMadeUserObservableChanges()) {
253 if (m_throttleState != ShouldNotThrottle) {
254 m_throttleState = ShouldNotThrottle;
255 updateTimerIntervalIfNecessary();
256 }
257 } else if (fireState.scriptMadeNonUserObservableChanges()) {
258 if (m_throttleState != ShouldThrottle) {
259 m_throttleState = ShouldThrottle;
260 updateTimerIntervalIfNecessary();
261 }
262 }

265void DOMTimer::scriptDidInteractWithPlugin(HTMLPlugInElement& pluginElement)
266{
267 if (!DOMTimerFireState::current)
268 return;
269
270 if (pluginElement.isUserObservable())
271 DOMTimerFireState::current->setScriptMadeUserObservableChanges();
272 else
273 DOMTimerFireState::current->setScriptMadeNonUserObservableChanges();
274}

276void DOMTimer::fired()
277{
278 // Retain this - if the timer is cancelled while this function is on the stack (implicitly and always
279 // for one-shot timers, or if removeById is called on itself from within an interval timer fire) then
280 // wait unit the end of this function to delete DOMTimer.
281 Ref<DOMTimer> protectedThis(*this);
282
283 bool oneShot = !repeatInterval();
284
285 ASSERT(scriptExecutionContext());
286 ScriptExecutionContext& context = *scriptExecutionContext();
287
288#if PLATFORM(IOS_FAMILY)
289 if (is<Document>(context)) {
290 auto& document = downcast<Document>(context);
291 if (auto* holdingTank = document.domTimerHoldingTankIfExists(); holdingTank && holdingTank->contains(*this)) {
292 if (oneShot)
293 startOneShot(0_s);
294 return;
295 }
296 }
297#endif
298
299 DOMTimerFireState fireState(context, std::min(m_nestingLevel + 1, maxTimerNestingLevel));
300
301 if (m_userGestureTokenToForward && m_userGestureTokenToForward->hasExpired(UserGestureToken::maximumIntervalForUserGestureForwarding))
302 m_userGestureTokenToForward = nullptr;
303
304 ASSERT(!isSuspended());
305 ASSERT(!context.activeDOMObjectsAreSuspended());
306 UserGestureIndicator gestureIndicator(m_userGestureTokenToForward);
307 // Only the first execution of a multi-shot timer should get an affirmative user gesture indicator.
308 m_userGestureTokenToForward = nullptr;
309
310 InspectorInstrumentation::willFireTimer(context, m_timeoutId, oneShot);
311
312 // Simple case for non-one-shot timers.
313 if (isActive()) {
314 if (m_nestingLevel < maxTimerNestingLevel) {
315 m_nestingLevel++;
316 updateTimerIntervalIfNecessary();
317 }
318
319 m_action->execute(context);
320
321 InspectorInstrumentation::didFireTimer(context, m_timeoutId, oneShot);
322
323 updateThrottlingStateIfNecessary(fireState);
324 return;
325 }
326
327 context.removeTimeout(m_timeoutId);
328
329 // Keep track nested timer installs.
330 NestedTimersMap* nestedTimers = NestedTimersMap::instanceForContext(context);
331 if (nestedTimers)
332 nestedTimers->startTracking();
333
334#if PLATFORM(IOS_FAMILY)
335 ContentChangeObserver::DOMTimerScope observingScope(is<Document>(context) ? &downcast<Document>(context) : nullptr, *this);
336#endif
337 m_action->execute(context);
338
339 InspectorInstrumentation::didFireTimer(context, m_timeoutId, oneShot);
340
341 // Check if we should throttle nested single-shot timers.
342 if (nestedTimers) {
343 for (auto& idAndTimer : *nestedTimers) {
344 auto& timer = idAndTimer.value;
345 if (timer->isActive() && !timer->repeatInterval())
346 timer->updateThrottlingStateIfNecessary(fireState);
347 }
348 nestedTimers->stopTracking();
349 }
350}

360void DOMTimer::updateTimerIntervalIfNecessary()
361{
362 ASSERT(m_nestingLevel <= maxTimerNestingLevel);
363
364 auto previousInterval = m_currentTimerInterval;
365 m_currentTimerInterval = intervalClampedToMinimum();
366 if (previousInterval == m_currentTimerInterval)
367 return;
368
369 if (repeatInterval()) {
370 ASSERT(repeatInterval() == previousInterval);
371 LOG(DOMTimers, "%p - Updating DOMTimer's repeat interval from %.2f ms to %.2f ms due to throttling.", this, previousInterval.milliseconds(), m_currentTimerInterval.milliseconds());
372 augmentRepeatInterval(m_currentTimerInterval - previousInterval);
373 } else {
374 LOG(DOMTimers, "%p - Updating DOMTimer's fire interval from %.2f ms to %.2f ms due to throttling.", this, previousInterval.milliseconds(), m_currentTimerInterval.milliseconds());
375 augmentFireInterval(m_currentTimerInterval - previousInterval);
376 }
377}

379Seconds DOMTimer::intervalClampedToMinimum() const
380{
381 ASSERT(scriptExecutionContext());
382 ASSERT(m_nestingLevel <= maxTimerNestingLevel);
383
384 Seconds interval = std::max(1_ms, m_originalInterval);
385
386 // Only apply throttling to repeating timers.
387 if (m_nestingLevel < maxTimerNestingLevel)
388 return interval;
389
390 // Apply two throttles - the global (per Page) minimum, and also a per-timer throttle.
391 interval = std::max(interval, scriptExecutionContext()->minimumDOMTimerInterval());
392 if (m_throttleState == ShouldThrottle)
393 interval = std::max(interval, minIntervalForNonUserObservableChangeTimers);
394 return interval;
395}

397std::optional<MonotonicTime> DOMTimer::alignedFireTime(MonotonicTime fireTime) const
398{
399 Seconds alignmentInterval = scriptExecutionContext()->domTimerAlignmentInterval(m_nestingLevel >= maxTimerNestingLevel);
400 if (!alignmentInterval)
401 return std::nullopt;
402
403 static const double randomizedProportion = randomNumber();
404
405 // Force alignment to randomizedAlignment fraction of the way between alignemntIntervals, e.g.
406 // if alignmentInterval is 10_ms and randomizedAlignment is 0.3 this will align to 3, 13, 23, ...
407 Seconds randomizedOffset = alignmentInterval * randomizedProportion;
408 MonotonicTime adjustedFireTime = fireTime - randomizedOffset;
409 return adjustedFireTime - (adjustedFireTime % alignmentInterval) + alignmentInterval + randomizedOffset;
410}
