31namespace WebCore {
32
33class DeprecatedGlobalSettings {
34public:
35 DeprecatedGlobalSettings() = delete;
36
37#if PLATFORM(WIN)
38 static void setShouldUseHighResolutionTimers(bool);
39 static bool shouldUseHighResolutionTimers() { return gShouldUseHighResolutionTimers; }
40#endif
41
42#if USE(AVFOUNDATION)
43 WEBCORE_EXPORT static void setAVFoundationEnabled(bool flag);
44 static bool isAVFoundationEnabled() { return gAVFoundationEnabled; }
45 WEBCORE_EXPORT static void setAVFoundationNSURLSessionEnabled(bool flag);
46 static bool isAVFoundationNSURLSessionEnabled() { return gAVFoundationNSURLSessionEnabled; }
47#endif
48
49#if USE(GSTREAMER)
50 WEBCORE_EXPORT static void setGStreamerEnabled(bool flag);
51 static bool isGStreamerEnabled() { return gGStreamerEnabled; }
52#endif
53
54 WEBCORE_EXPORT static void setMockScrollbarsEnabled(bool flag);
55 WEBCORE_EXPORT static bool mockScrollbarsEnabled();
56
57 WEBCORE_EXPORT static void setUsesOverlayScrollbars(bool flag);
58 static bool usesOverlayScrollbars();
59
60 WEBCORE_EXPORT static void setUsesMockScrollAnimator(bool);
61 static bool usesMockScrollAnimator();
62
63 static bool lowPowerVideoAudioBufferSizeEnabled() { return gLowPowerVideoAudioBufferSizeEnabled; }
64 WEBCORE_EXPORT static void setLowPowerVideoAudioBufferSizeEnabled(bool);
65
66 static bool resourceLoadStatisticsEnabled() { return gResourceLoadStatisticsEnabledEnabled; }
67 WEBCORE_EXPORT static void setResourceLoadStatisticsEnabled(bool);
68
69#if PLATFORM(IOS_FAMILY)
70 WEBCORE_EXPORT static void setAudioSessionCategoryOverride(unsigned);
71 static unsigned audioSessionCategoryOverride();
72
73 WEBCORE_EXPORT static void setNetworkDataUsageTrackingEnabled(bool);
74 static bool networkDataUsageTrackingEnabled();
75
76 WEBCORE_EXPORT static void setNetworkInterfaceName(const String&);
77 static const String& networkInterfaceName();
78
79 static void setDisableScreenSizeOverride(bool flag) { gDisableScreenSizeOverride = flag; }
80 static bool disableScreenSizeOverride() { return gDisableScreenSizeOverride; }
81
82 static void setShouldOptOutOfNetworkStateObservation(bool flag) { gShouldOptOutOfNetworkStateObservation = flag; }
83 static bool shouldOptOutOfNetworkStateObservation() { return gShouldOptOutOfNetworkStateObservation; }
84#endif
85
86#if USE(AUDIO_SESSION)
87 static void setShouldManageAudioSessionCategory(bool flag) { gManageAudioSession = flag; }
88 static bool shouldManageAudioSessionCategory() { return gManageAudioSession; }
89#endif
90
91 WEBCORE_EXPORT static void setAllowsAnySSLCertificate(bool);
92 static bool allowsAnySSLCertificate();
93
94private:
95#if USE(AVFOUNDATION)
96 WEBCORE_EXPORT static bool gAVFoundationEnabled;
97 WEBCORE_EXPORT static bool gAVFoundationNSURLSessionEnabled;
98#endif
99
100#if USE(GSTREAMER)
101 WEBCORE_EXPORT static bool gGStreamerEnabled;
102#endif
103
104 static bool gMockScrollbarsEnabled;
105 static bool gUsesOverlayScrollbars;
106 static bool gMockScrollAnimatorEnabled;
107
108#if PLATFORM(WIN)
109 static bool gShouldUseHighResolutionTimers;
110#endif
111#if PLATFORM(IOS_FAMILY)
112 static bool gNetworkDataUsageTrackingEnabled;
113 WEBCORE_EXPORT static bool gShouldOptOutOfNetworkStateObservation;
114 WEBCORE_EXPORT static bool gDisableScreenSizeOverride;
115#endif
116 WEBCORE_EXPORT static bool gManageAudioSession;
117
118 static bool gLowPowerVideoAudioBufferSizeEnabled;
119 WEBCORE_EXPORT static bool gResourceLoadStatisticsEnabledEnabled;
120 static bool gAllowsAnySSLCertificate;
121};
