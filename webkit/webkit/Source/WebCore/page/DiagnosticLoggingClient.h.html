28#include "DiagnosticLoggingDomain.h"
29#include "DiagnosticLoggingResultType.h"
30#include <wtf/FastMalloc.h>
31#include <wtf/HashMap.h>
32#include <wtf/RandomNumber.h>
33#include <wtf/Variant.h>
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38enum class ShouldSample : bool { No, Yes };
39
40class DiagnosticLoggingClient {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 virtual void logDiagnosticMessage(const String& message, const String& description, ShouldSample) = 0;
44 virtual void logDiagnosticMessageWithResult(const String& message, const String& description, DiagnosticLoggingResultType, ShouldSample) = 0;
45 virtual void logDiagnosticMessageWithValue(const String& message, const String& description, double value, unsigned significantFigures, ShouldSample) = 0;
46 virtual void logDiagnosticMessageWithEnhancedPrivacy(const String& message, const String& description, ShouldSample) = 0;
47
48 using ValuePayload = Variant<String, uint64_t, int64_t, bool, double>;
49 using ValueDictionary = HashMap<String, ValuePayload>;
50
51 virtual void logDiagnosticMessageWithValueDictionary(const String& message, const String& description, const ValueDictionary&, ShouldSample) = 0;
52 virtual void logDiagnosticMessageWithDomain(const String& message, DiagnosticLoggingDomain) = 0;
53
54 static bool shouldLogAfterSampling(ShouldSample);
55
56 virtual ~DiagnosticLoggingClient() = default;
57};
58
59inline bool DiagnosticLoggingClient::shouldLogAfterSampling(ShouldSample shouldSample)
60{
61 if (shouldSample == ShouldSample::No)
62 return true;
63
64 static const double selectionProbability = 0.05;
65 return randomNumber() <= selectionProbability;
66}
67
68} // namespace WebCore
