788String DiagnosticLoggingKeys::memoryUsageToDiagnosticLoggingKey(uint64_t memoryUsage)
789{
790 if (memoryUsage < 32 * MB)
791 return "below32"_s;
792 if (memoryUsage < 64 * MB)
793 return "32to64"_s;
794 if (memoryUsage < 128 * MB)
795 return "64to128"_s;
796 if (memoryUsage < 256 * MB)
797 return "128to256"_s;
798 if (memoryUsage < 512 * MB)
799 return "256to512"_s;
800 if (memoryUsage < 1024 * MB)
801 return "512to1024"_s;
802 if (memoryUsage < 2048 * MB)
803 return "1024to2048"_s;
804 if (memoryUsage < 4096llu * MB)
805 return "2048to4096"_s;
806 if (memoryUsage < 8192llu * MB)
807 return "4096to8192"_s;
808 if (memoryUsage < 16384llu * MB)
809 return "8192to16384"_s;
810 if (memoryUsage < 32768llu * MB)
811 return "16384to32768"_s;
812 return "over32768"_s;
813}

815String DiagnosticLoggingKeys::foregroundCPUUsageToDiagnosticLoggingKey(double cpuUsage)
816{
817 if (cpuUsage < 10)
818 return "below10"_s;
819 if (cpuUsage < 20)
820 return "10to20"_s;
821 if (cpuUsage < 40)
822 return "20to40"_s;
823 if (cpuUsage < 60)
824 return "40to60"_s;
825 if (cpuUsage < 80)
826 return "60to80"_s;
827 return "over80"_s;
828}
829
830String DiagnosticLoggingKeys::backgroundCPUUsageToDiagnosticLoggingKey(double cpuUsage)
831{
832 if (cpuUsage < 1)
833 return "below1"_s;
834 if (cpuUsage < 5)
835 return "1to5"_s;
836 if (cpuUsage < 10)
837 return "5to10"_s;
838 if (cpuUsage < 30)
839 return "10to30"_s;
840 if (cpuUsage < 50)
841 return "30to50"_s;
842 if (cpuUsage < 70)
843 return "50to70"_s;
844 return "over70"_s;
845}
