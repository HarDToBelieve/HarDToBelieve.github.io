45class DragClient {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 virtual bool useLegacyDragClient() { return true; }
49
50 virtual void willPerformDragDestinationAction(DragDestinationAction, const DragData&) = 0;
51 virtual void willPerformDragSourceAction(DragSourceAction, const IntPoint&, DataTransfer&) = 0;
52 virtual void didConcludeEditDrag() { }
53 virtual OptionSet<DragSourceAction> dragSourceActionMaskForPoint(const IntPoint& rootViewPoint) = 0;
54
55 virtual void startDrag(DragItem, DataTransfer&, Frame&) = 0;
56 virtual void dragEnded() { }
57
58 virtual void beginDrag(DragItem, Frame&, const IntPoint&, const IntPoint&, DataTransfer&, DragSourceAction) { }
59
60#if PLATFORM(COCOA)
61 // Mac-specific helper function to allow access to web archives and NSPasteboard extras in WebKit.

64#endif
65
66 virtual ~DragClient() = default;
67};
