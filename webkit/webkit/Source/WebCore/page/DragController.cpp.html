105bool isDraggableLink(const Element& element)
106{
107 if (is<HTMLAnchorElement>(element)) {
108 auto& anchorElement = downcast<HTMLAnchorElement>(element);
109 if (!anchorElement.isLiveLink())
110 return false;
111#if ENABLE(DATA_DETECTION)
112 return !DataDetection::isDataDetectorURL(anchorElement.href());
113#else
114 return true;
115#endif
116 }
117 if (is<SVGAElement>(element))
118 return element.isLink();
119 return false;
120}

146static RefPtr<DocumentFragment> documentFragmentFromDragData(const DragData& dragData, Frame& frame, const SimpleRange& context, bool allowPlainText, bool& chosePlainText)
147{
148 chosePlainText = false;
149
150 if (dragData.containsCompatibleContent()) {
151 if (auto fragment = frame.editor().webContentFromPasteboard(*Pasteboard::create(dragData), context, allowPlainText, chosePlainText))
152 return fragment;
153
154 if (dragData.containsURL(DragData::DoNotConvertFilenames)) {
155 String title;
156 String url = dragData.asURL(DragData::DoNotConvertFilenames, &title);
157 if (!url.isEmpty()) {
158 auto& document = context.start.document();
159 auto anchor = HTMLAnchorElement::create(document);
160 anchor->setHref(url);
161 if (title.isEmpty()) {
162 // Try the plain text first because the URL might be normalized or escaped.
163 if (dragData.containsPlainText())
164 title = dragData.asPlainText();
165 if (title.isEmpty())
166 title = url;
167 }
168 anchor->appendChild(document.createTextNode(title));
169 auto fragment = document.createDocumentFragment();
170 fragment->appendChild(anchor);
171 return fragment;
172 }
173 }
174 }
175 if (allowPlainText && dragData.containsPlainText()) {
176 chosePlainText = true;
177 return createFragmentFromText(context, dragData.asPlainText()).ptr();
178 }
179
180 return nullptr;
181}

240bool DragController::performDragOperation(const DragData& dragData)
241{
242 if (!m_droppedImagePlaceholders.isEmpty() && m_droppedImagePlaceholderRange && tryToUpdateDroppedImagePlaceholders(dragData)) {
243 m_droppedImagePlaceholders.clear();
244 m_droppedImagePlaceholderRange = std::nullopt;
245 m_documentUnderMouse = nullptr;
246 clearDragCaret();
247 return true;
248 }
249
250 removeAllDroppedImagePlaceholders();
251
252 SetForScope<bool> isPerformingDrop(m_isPerformingDrop, true);
253 IgnoreSelectionChangeForScope ignoreSelectionChanges { m_page.focusController().focusedOrMainFrame() };
254
255 m_documentUnderMouse = m_page.mainFrame().documentAtPoint(dragData.clientPosition());
256
257 ShouldOpenExternalURLsPolicy shouldOpenExternalURLsPolicy = ShouldOpenExternalURLsPolicy::ShouldNotAllow;
258 if (m_documentUnderMouse)
259 shouldOpenExternalURLsPolicy = m_documentUnderMouse->shouldOpenExternalURLsPolicyToPropagate();
260
261 if ((m_dragDestinationActionMask.contains(DragDestinationAction::DHTML)) && dragIsHandledByDocument(m_dragHandlingMethod)) {
262 client().willPerformDragDestinationAction(DragDestinationAction::DHTML, dragData);
263 Ref<Frame> mainFrame(m_page.mainFrame());
264 bool preventedDefault = false;
265 if (mainFrame->view())
266 preventedDefault = mainFrame->eventHandler().performDragAndDrop(createMouseEvent(dragData), Pasteboard::create(dragData), dragData.draggingSourceOperationMask(), dragData.containsFiles());
267 if (preventedDefault) {
268 clearDragCaret();
269 m_documentUnderMouse = nullptr;
270 return true;
271 }
272 }
273
274 if ((m_dragDestinationActionMask.contains(DragDestinationAction::Edit)) && concludeEditDrag(dragData)) {
275 client().didConcludeEditDrag();
276 m_documentUnderMouse = nullptr;
277 clearDragCaret();
278 return true;
279 }
280
281 m_documentUnderMouse = nullptr;
282 clearDragCaret();
283
284 if (!operationForLoad(dragData))
285 return false;
286
287 auto urlString = dragData.asURL();
288 if (urlString.isEmpty())
289 return false;
290
291 client().willPerformDragDestinationAction(DragDestinationAction::Load, dragData);
292 FrameLoadRequest frameLoadRequest { m_page.mainFrame(), ResourceRequest { urlString } };
293 frameLoadRequest.setShouldOpenExternalURLsPolicy(shouldOpenExternalURLsPolicy);
294 frameLoadRequest.setIsRequestFromClientOrUserInput();
295 m_page.mainFrame().loader().load(WTFMove(frameLoadRequest));
296 return true;
297}

310std::optional<DragOperation> DragController::dragEnteredOrUpdated(const DragData& dragData)
311{
312 mouseMovedIntoDocument(m_page.mainFrame().documentAtPoint(dragData.clientPosition()));
313
314 m_dragDestinationActionMask = dragData.dragDestinationActionMask();
315 if (m_dragDestinationActionMask.isEmpty()) {
316 clearDragCaret(); // FIXME: Why not call mouseMovedIntoDocument(nullptr)?
317 return std::nullopt;
318 }
319
320 std::optional<DragOperation> dragOperation;
321 m_dragHandlingMethod = tryDocumentDrag(dragData, m_dragDestinationActionMask, dragOperation);
322 if (m_dragHandlingMethod == DragHandlingMethod::None && (m_dragDestinationActionMask.contains(DragDestinationAction::Load))) {
323 dragOperation = operationForLoad(dragData);
324 if (dragOperation)
325 m_dragHandlingMethod = DragHandlingMethod::PageLoad;
326 } else if (m_dragHandlingMethod == DragHandlingMethod::SetColor)
327 dragOperation = DragOperation::Copy;
328
329 updateSupportedTypeIdentifiersForDragHandlingMethod(m_dragHandlingMethod, dragData);
330 return dragOperation;
331}

333static HTMLInputElement* asFileInput(Node& node)
334{
335 if (!is<HTMLInputElement>(node))
336 return nullptr;
337
338 auto* inputElement = &downcast<HTMLInputElement>(node);
339
340 // If this is a button inside of the a file input, move up to the file input.
341 if (inputElement->isTextButton() && is<ShadowRoot>(inputElement->treeScope().rootNode())) {
342 auto& host = *downcast<ShadowRoot>(inputElement->treeScope().rootNode()).host();
343 inputElement = is<HTMLInputElement>(host) ? &downcast<HTMLInputElement>(host) : nullptr;
344 }
345
346 return inputElement && inputElement->isFileUpload() ? inputElement : nullptr;
347}

367// This can return null if an empty document is loaded.
368static Element* elementUnderMouse(Document* documentUnderMouse, const IntPoint& p)
369{
370 Frame* frame = documentUnderMouse->frame();
371 float zoomFactor = frame ? frame->pageZoomFactor() : 1;
372 LayoutPoint point(p.x() * zoomFactor, p.y() * zoomFactor);
373
374 HitTestResult result(point);
375 documentUnderMouse->hitTest(HitTestRequest(), result);
376
377 auto* node = result.innerNode();
378 if (!node)
379 return nullptr;
380 // FIXME: Use parentElementInComposedTree here.
381 auto* element = is<Element>(*node) ? &downcast<Element>(*node) : node->parentElement();
382 auto* host = element->shadowHost();
383 return host ? host : element;
384}

394DragHandlingMethod DragController::tryDocumentDrag(const DragData& dragData, OptionSet<DragDestinationAction> destinationActionMask, std::optional<DragOperation>& dragOperation)
395{
396 if (!m_documentUnderMouse)
397 return DragHandlingMethod::None;
398
399 if (m_dragInitiator && !m_documentUnderMouse->securityOrigin().canReceiveDragData(m_dragInitiator->securityOrigin()))
400 return DragHandlingMethod::None;
401
402 bool isHandlingDrag = false;
403 if (destinationActionMask.contains(DragDestinationAction::DHTML)) {
404 isHandlingDrag = tryDHTMLDrag(dragData, dragOperation);
405 // Do not continue if m_documentUnderMouse has been reset by tryDHTMLDrag.
406 // tryDHTMLDrag fires dragenter event. The event listener that listens
407 // to this event may create a nested message loop (open a modal dialog),
408 // which could process dragleave event and reset m_documentUnderMouse in
409 // dragExited.
410 if (!m_documentUnderMouse)
411 return DragHandlingMethod::None;
412 }
413
414 // It's unclear why this check is after tryDHTMLDrag.
415 // We send drag events in tryDHTMLDrag and that may be the reason.
416 RefPtr<FrameView> frameView = m_documentUnderMouse->view();
417 if (!frameView)
418 return DragHandlingMethod::None;
419
420 if (isHandlingDrag) {
421 clearDragCaret();
422 m_numberOfItemsToBeAccepted = dragData.numberOfFiles();
423 return DragHandlingMethod::NonDefault;
424 }
425
426 if (destinationActionMask.contains(DragDestinationAction::Edit) && canProcessDrag(dragData)) {
427 if (dragData.containsColor()) {
428 dragOperation = DragOperation::Generic;
429 return DragHandlingMethod::SetColor;
430 }
431
432 IntPoint point = frameView->windowToContents(dragData.clientPosition());
433 Element* element = elementUnderMouse(m_documentUnderMouse.get(), point);
434 if (!element)
435 return DragHandlingMethod::None;
436
437 HTMLInputElement* elementAsFileInput = asFileInput(*element);
438 if (m_fileInputElementUnderMouse != elementAsFileInput) {
439 if (m_fileInputElementUnderMouse)
440 m_fileInputElementUnderMouse->setCanReceiveDroppedFiles(false);
441 m_fileInputElementUnderMouse = elementAsFileInput;
442 }
443
444 if (!m_fileInputElementUnderMouse)
445 m_page.dragCaretController().setCaretPosition(m_documentUnderMouse->frame()->visiblePositionForPoint(point));
446 else
447 clearDragCaret();
448
449 Frame* innerFrame = element->document().frame();
450 dragOperation = dragIsMove(innerFrame->selection(), dragData) ? DragOperation::Move : DragOperation::Copy;
451
452 unsigned numberOfFiles = dragData.numberOfFiles();
453 if (m_fileInputElementUnderMouse) {
454 if (m_fileInputElementUnderMouse->isDisabledFormControl())
455 m_numberOfItemsToBeAccepted = 0;
456 else if (m_fileInputElementUnderMouse->multiple())
457 m_numberOfItemsToBeAccepted = numberOfFiles;
458 else if (numberOfFiles > 1)
459 m_numberOfItemsToBeAccepted = 0;
460 else
461 m_numberOfItemsToBeAccepted = 1;
462
463 if (!m_numberOfItemsToBeAccepted)
464 dragOperation = std::nullopt;
465 m_fileInputElementUnderMouse->setCanReceiveDroppedFiles(m_numberOfItemsToBeAccepted);
466 } else {
467 // We are not over a file input element. The dragged item(s) will loaded into the view,
468 // dropped as text paths on other input elements, or handled by script on the page.
469 m_numberOfItemsToBeAccepted = numberOfFiles;
470 }
471
472 if (m_fileInputElementUnderMouse)
473 return DragHandlingMethod::UploadFile;
474
475 if (m_page.dragCaretController().isContentRichlyEditable())
476 return DragHandlingMethod::EditRichText;
477
478 return DragHandlingMethod::EditPlainText;
479 }
480
481 // We are not over an editable region. Make sure we're clearing any prior drag cursor.
482 clearDragCaret();
483 if (m_fileInputElementUnderMouse)
484 m_fileInputElementUnderMouse->setCanReceiveDroppedFiles(false);
485 m_fileInputElementUnderMouse = nullptr;
486 return DragHandlingMethod::None;
487}

495std::optional<DragOperation> DragController::operationForLoad(const DragData& dragData)
496{
497 Document* document = m_page.mainFrame().documentAtPoint(dragData.clientPosition());
498
499 bool pluginDocumentAcceptsDrags = false;
500
501 if (is<PluginDocument>(document)) {
502 const Widget* widget = downcast<PluginDocument>(*document).pluginWidget();
503 const PluginViewBase* pluginView = is<PluginViewBase>(widget) ? downcast<PluginViewBase>(widget) : nullptr;
504
505 if (pluginView)
506 pluginDocumentAcceptsDrags = pluginView->shouldAllowNavigationFromDrags();
507 }
508
509 if (document && (m_didInitiateDrag || (is<PluginDocument>(*document) && !pluginDocumentAcceptsDrags) || document->hasEditableStyle()))
510 return std::nullopt;
511 return dragOperation(dragData);
512}

536bool DragController::concludeEditDrag(const DragData& dragData)
537{
538 RefPtr<HTMLInputElement> fileInput = m_fileInputElementUnderMouse;
539 if (m_fileInputElementUnderMouse) {
540 m_fileInputElementUnderMouse->setCanReceiveDroppedFiles(false);
541 m_fileInputElementUnderMouse = nullptr;
542 }
543
544 if (!m_documentUnderMouse)
545 return false;
546
547 IntPoint point = m_documentUnderMouse->view()->windowToContents(dragData.clientPosition());
548 Element* element = elementUnderMouse(m_documentUnderMouse.get(), point);
549 if (!element)
550 return false;
551 RefPtr<Frame> innerFrame = element->document().frame();
552 ASSERT(innerFrame);
553
554 if (m_page.dragCaretController().hasCaret() && !dispatchTextInputEventFor(innerFrame.get(), dragData))
555 return true;
556
557 if (dragData.containsColor()) {
558 Color color = dragData.asColor();
559 if (!color.isValid())
560 return false;
561#if ENABLE(INPUT_TYPE_COLOR)
562 if (isEnabledColorInput(*element)) {
563 auto& input = downcast<HTMLInputElement>(*element);
564 input.setValue(serializationForHTML(color), DispatchInputAndChangeEvent);
565 return true;
566 }
567#endif
568 auto innerRange = innerFrame->selection().selection().toNormalizedRange();
569 if (!innerRange)
570 return false;
571 auto style = MutableStyleProperties::create();
572 style->setProperty(CSSPropertyColor, serializationForHTML(color), false);
573 if (!innerFrame->editor().shouldApplyStyle(style, *innerRange))
574 return false;
575 client().willPerformDragDestinationAction(DragDestinationAction::Edit, dragData);
576 innerFrame->editor().applyStyle(style.ptr(), EditAction::SetColor);
577 return true;
578 }
579
580 if (dragData.containsFiles() && fileInput) {
581 // fileInput should be the element we hit tested for, unless it was made
582 // display:none in a drop event handler.
583 ASSERT(fileInput == element || !fileInput->renderer());
584 if (fileInput->isDisabledFormControl())
585 return false;
586
587 return fileInput->receiveDroppedFiles(dragData);
588 }
589
590 if (!m_page.dragController().canProcessDrag(dragData))
591 return false;
592
593 VisibleSelection dragCaret = m_page.dragCaretController().caretPosition();
594 auto range = dragCaret.toNormalizedRange();
595 RefPtr<Element> rootEditableElement = innerFrame->selection().selection().rootEditableElement();
596
597 // For range to be null a WebKit client must have done something bad while
598 // manually controlling drag behaviour
599 if (!range)
600 return false;
601
602 ResourceCacheValidationSuppressor validationSuppressor(range->start.document().cachedResourceLoader());
603 auto& editor = innerFrame->editor();
604 bool isMove = dragIsMove(innerFrame->selection(), dragData);
605 if (isMove || dragCaret.isContentRichlyEditable()) {
606 bool chosePlainText = false;
607 auto fragment = documentFragmentFromDragData(dragData, *innerFrame, *range, true, chosePlainText);
608 if (!fragment || !editor.shouldInsertFragment(*fragment, range, EditorInsertAction::Dropped))
609 return false;
610
611 client().willPerformDragDestinationAction(DragDestinationAction::Edit, dragData);
612
613 if (editor.client() && editor.client()->performTwoStepDrop(*fragment, *range, isMove))
614 return true;
615
616 if (isMove) {
617 // NSTextView behavior is to always smart delete on moving a selection,
618 // but only to smart insert if the selection granularity is word granularity.
619 bool smartDelete = editor.smartInsertDeleteEnabled();
620 bool smartInsert = smartDelete && innerFrame->selection().granularity() == TextGranularity::WordGranularity && dragData.canSmartReplace();
621 MoveSelectionCommand::create(fragment.releaseNonNull(), dragCaret.base(), smartInsert, smartDelete)->apply();
622 } else {
623 if (setSelectionToDragCaret(innerFrame.get(), dragCaret, point)) {
624 OptionSet<ReplaceSelectionCommand::CommandOption> options { ReplaceSelectionCommand::SelectReplacement, ReplaceSelectionCommand::PreventNesting };
625 if (dragData.canSmartReplace())
626 options.add(ReplaceSelectionCommand::SmartReplace);
627 if (chosePlainText)
628 options.add(ReplaceSelectionCommand::MatchStyle);
629 ReplaceSelectionCommand::create(*m_documentUnderMouse, fragment.releaseNonNull(), options, EditAction::InsertFromDrop)->apply();
630 }
631 }
632 } else {
633 String text = dragData.asPlainText();
634 if (text.isEmpty() || !editor.shouldInsertText(text, *range, EditorInsertAction::Dropped))
635 return false;
636
637 client().willPerformDragDestinationAction(DragDestinationAction::Edit, dragData);
638 auto fragment = createFragmentFromText(*range, text);
639 if (editor.client() && editor.client()->performTwoStepDrop(fragment.get(), *range, isMove))
640 return true;
641
642 if (setSelectionToDragCaret(innerFrame.get(), dragCaret, point))
643 ReplaceSelectionCommand::create(*m_documentUnderMouse, WTFMove(fragment), { ReplaceSelectionCommand::SelectReplacement, ReplaceSelectionCommand::MatchStyle, ReplaceSelectionCommand::PreventNesting }, EditAction::InsertFromDrop)->apply();
644 }
645
646 if (rootEditableElement) {
647 if (Frame* frame = rootEditableElement->document().frame())
648 frame->eventHandler().updateDragStateAfterEditDragIfNeeded(*rootEditableElement);
649 }
650
651 return true;
652}

654bool DragController::canProcessDrag(const DragData& dragData)
655{
656 IntPoint point = m_page.mainFrame().view()->windowToContents(dragData.clientPosition());
657 HitTestResult result = HitTestResult(point);
658 if (!m_page.mainFrame().contentRenderer())
659 return false;
660
661 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::AllowChildFrameContent };
662 result = m_page.mainFrame().eventHandler().hitTestResultAtPoint(point, hitType);
663
664 auto* dragNode = result.innerNonSharedNode();
665 if (!dragNode)
666 return false;
667
668 DragData::DraggingPurpose dragPurpose = DragData::DraggingPurpose::ForEditing;
669 if (asFileInput(*dragNode))
670 dragPurpose = DragData::DraggingPurpose::ForFileUpload;
671#if ENABLE(INPUT_TYPE_COLOR)
672 else if (isEnabledColorInput(*dragNode) || isInShadowTreeOfEnabledColorInput(*dragNode))
673 dragPurpose = DragData::DraggingPurpose::ForColorControl;
674#endif
675
676 if (!dragData.containsCompatibleContent(dragPurpose))
677 return false;
678
679 if (dragPurpose == DragData::DraggingPurpose::ForFileUpload)
680 return true;
681
682#if ENABLE(INPUT_TYPE_COLOR)
683 if (dragPurpose == DragData::DraggingPurpose::ForColorControl)
684 return true;
685#endif
686
687 if (!dragNode->hasEditableStyle())
688 return false;
689
690 if (m_didInitiateDrag && m_documentUnderMouse == m_dragInitiator && result.isSelected())
691 return false;
692
693 return true;
694}

696static std::optional<DragOperation> defaultOperationForDrag(OptionSet<DragOperation> sourceOperationMask)
697{
698 // This is designed to match IE's operation fallback for the case where
699 // the page calls preventDefault() in a drag event but doesn't set dropEffect.
700 if (sourceOperationMask.containsAll({ DragOperation::Copy, DragOperation::Link, DragOperation::Generic, DragOperation::Private, DragOperation::Move, DragOperation::Delete }))
701 return DragOperation::Copy;
702 if (sourceOperationMask.isEmpty())
703 return std::nullopt;
704 if (sourceOperationMask.contains(DragOperation::Move))
705 return DragOperation::Move;
706 if (sourceOperationMask.contains(DragOperation::Generic))
707 return DragController::platformGenericDragOperation();
708 if (sourceOperationMask.contains(DragOperation::Copy))
709 return DragOperation::Copy;
710 if (sourceOperationMask.contains(DragOperation::Link))
711 return DragOperation::Link;
712
713 // FIXME: Does IE really return "generic" even if no operations were allowed by the source?
714 return DragOperation::Generic;
715}

717bool DragController::tryDHTMLDrag(const DragData& dragData, std::optional<DragOperation>& operation)
718{
719 ASSERT(m_documentUnderMouse);
720 Ref<Frame> mainFrame(m_page.mainFrame());
721 RefPtr<FrameView> viewProtector = mainFrame->view();
722 if (!viewProtector)
723 return false;
724
725 auto sourceOperationMask = dragData.draggingSourceOperationMask();
726 auto targetResponse = mainFrame->eventHandler().updateDragAndDrop(createMouseEvent(dragData), [&dragData]() {
727 return Pasteboard::create(dragData);
728 }, sourceOperationMask, dragData.containsFiles());
729 if (!targetResponse.accept)
730 return false;
731
732 if (!targetResponse.operationMask)
733 operation = defaultOperationForDrag(sourceOperationMask);
734 else if (!sourceOperationMask.containsAny(targetResponse.operationMask.value())) // The element picked an operation which is not supported by the source.
735 operation = std::nullopt;
736 else
737 operation = defaultOperationForDrag(targetResponse.operationMask.value());
738
739 return true;
740}

770Element* DragController::draggableElement(const Frame* sourceFrame, Element* startElement, const IntPoint& dragOrigin, DragState& state) const
771{
772 state.type = sourceFrame->selection().contains(dragOrigin) ? DragSourceAction::Selection : OptionSet<DragSourceAction>({ });
773 if (!startElement)
774 return nullptr;
775
776#if ENABLE(ATTACHMENT_ELEMENT)
777 if (auto attachment = enclosingAttachmentElement(*startElement)) {
778 auto& selection = sourceFrame->selection().selection();
779 bool isSingleAttachmentSelection = selection.start() == Position(attachment.get(), Position::PositionIsBeforeAnchor) && selection.end() == Position(attachment.get(), Position::PositionIsAfterAnchor);
780 auto selectedRange = selection.firstRange();
781 if (isSingleAttachmentSelection || !selectedRange || !contains<ComposedTree>(*selectedRange, *attachment)) {
782 state.type = DragSourceAction::Attachment;
783 return attachment.get();
784 }
785 }
786#endif
787
788 auto selectionDragElement = state.type.contains(DragSourceAction::Selection) && m_dragSourceAction.contains(DragSourceAction::Selection) ? startElement : nullptr;
789 if (HTMLElement::isImageOverlayText(startElement))
790 return selectionDragElement;
791
792 for (auto* element = startElement; element; element = element->parentOrShadowHostElement()) {
793 auto* renderer = element->renderer();
794 if (!renderer)
795 continue;
796
797 UserDrag dragMode = renderer->style().userDrag();
798 if (m_dragSourceAction.contains(DragSourceAction::DHTML) && dragMode == UserDrag::Element) {
799 state.type.add(DragSourceAction::DHTML);
800 return element;
801 }
802 if (dragMode == UserDrag::Auto) {
803 if ((m_dragSourceAction.contains(DragSourceAction::Image))
804 && is<HTMLImageElement>(*element)
805 && imageElementIsDraggable(downcast<HTMLImageElement>(*element), *sourceFrame)) {
806 state.type.add(DragSourceAction::Image);
807 return element;
808 }
809 if (m_dragSourceAction.contains(DragSourceAction::Link) && isDraggableLink(*element)) {
810 state.type.add(DragSourceAction::Link);
811 return element;
812 }
813#if ENABLE(ATTACHMENT_ELEMENT)
814 if (m_dragSourceAction.contains(DragSourceAction::Attachment)
815 && is<HTMLAttachmentElement>(*element)
816 && downcast<HTMLAttachmentElement>(*element).file()) {
817 state.type.add(DragSourceAction::Attachment);
818 return element;
819 }
820#endif
821#if ENABLE(INPUT_TYPE_COLOR)
822 if (m_dragSourceAction.contains(DragSourceAction::Color) && isEnabledColorInput(*element)) {
823 state.type.add(DragSourceAction::Color);
824 return element;
825 }
826#endif
827 }
828 }
829
830 // We either have nothing to drag or we have a selection and we're not over a draggable element.
831 return selectionDragElement;
832}

940std::optional<HitTestResult> DragController::hitTestResultForDragStart(Frame& source, Element& element, const IntPoint& location) const
941{
942 if (!source.view() || !source.contentRenderer())
943 return std::nullopt;

945 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::AllowChildFrameContent };

953 return std::nullopt;

956 return { hitTestResult };

959bool DragController::startDrag(Frame& src, const DragState& state, OptionSet<DragOperation> sourceOperationMask, const PlatformMouseEvent& dragEvent, const IntPoint& dragOrigin, HasNonDefaultPasteboardData hasData)
960{
961 if (!state.source)
962 return false;
963
964 Ref<Frame> protector(src);
965 auto hitTestResult = hitTestResultForDragStart(src, *state.source, dragOrigin);
966 if (!hitTestResult)
967 return false;
968
969 auto linkURL = hitTestResult->absoluteLinkURL();
970 auto imageURL = hitTestResult->absoluteImageURL();
971
972 IntPoint mouseDraggedPoint = src.view()->windowToContents(dragEvent.position());
973
974 m_draggingImageURL = URL();
975 m_sourceDragOperationMask = sourceOperationMask;
976
977 DragImage dragImage;
978 IntPoint dragLoc(0, 0);
979 IntPoint dragImageOffset(0, 0);
980
981 ASSERT(state.dataTransfer);
982
983 DataTransfer& dataTransfer = *state.dataTransfer;
984 if (state.type == DragSourceAction::DHTML) {
985 dragImage = DragImage { dataTransfer.createDragImage(dragImageOffset) };
986 // We allow DHTML/JS to set the drag image, even if its a link, image or text we're dragging.
987 // This is in the spirit of the IE API, which allows overriding of pasteboard data and DragOp.
988 if (dragImage) {
989 dragLoc = dragLocForDHTMLDrag(mouseDraggedPoint, dragOrigin, dragImageOffset, !linkURL.isEmpty());
990 m_dragOffset = dragImageOffset;
991 }
992 }
993
994 if (state.type == DragSourceAction::Selection || !imageURL.isEmpty() || !linkURL.isEmpty()) {
995 // Selection, image, and link drags receive a default set of allowed drag operations that
996 // follows from:
997 // http://trac.webkit.org/browser/trunk/WebKit/mac/WebView/WebHTMLView.mm?rev=48526#L3430
998 m_sourceDragOperationMask.add({ DragOperation::Generic, DragOperation::Copy });
999 }
1000
1001 ASSERT(state.source);
1002 Element& element = *state.source;
1003
1004 bool mustUseLegacyDragClient = hasData == HasNonDefaultPasteboardData::Yes || client().useLegacyDragClient();
1005
1006 IntRect dragImageBounds;
1007 Image* image = getImage(element);
1008 if (state.type == DragSourceAction::Selection) {
1009 PasteboardWriterData pasteboardWriterData;
1010
1011 if (hasData == HasNonDefaultPasteboardData::No) {
1012 if (src.selection().selection().isNone()) {
1013 // The page may have cleared out the selection in the dragstart handler, in which case we should bail
1014 // out of the drag, since there is no content to write to the pasteboard.
1015 return false;
1016 }
1017
1018 // FIXME: This entire block is almost identical to the code in Editor::copy, and the code should be shared.
1019 auto selectionRange = src.selection().selection().toNormalizedRange();
1020 ASSERT(selectionRange);
1021
1022 src.editor().willWriteSelectionToPasteboard(*selectionRange);
1023 auto selection = src.selection().selection();
1024 bool shouldDragAsPlainText = enclosingTextFormControl(selection.start());
1025 if (auto range = selection.range(); range && HTMLElement::isInsideImageOverlay(*range))
1026 shouldDragAsPlainText = true;
1027
1028 if (shouldDragAsPlainText) {
1029 if (mustUseLegacyDragClient)
1030 dataTransfer.pasteboard().writePlainText(src.editor().selectedTextForDataTransfer(), Pasteboard::CannotSmartReplace);
1031 else {
1032 PasteboardWriterData::PlainText plainText;
1033 plainText.canSmartCopyOrDelete = false;
1034 plainText.text = src.editor().selectedTextForDataTransfer();
1035 pasteboardWriterData.setPlainText(WTFMove(plainText));
1036 }
1037 } else {
1038 if (mustUseLegacyDragClient) {
1039#if !PLATFORM(WIN)
1040 src.editor().writeSelectionToPasteboard(dataTransfer.pasteboard());
1041#else
1042 // FIXME: Convert Windows to match the other platforms and delete this.
1043 dataTransfer.pasteboard().writeSelection(*selectionRange, src.editor().canSmartCopyOrDelete(), src, IncludeImageAltTextForDataTransfer);
1044#endif
1045 } else {
1046#if PLATFORM(COCOA)
1047 src.editor().writeSelection(pasteboardWriterData);
1048#endif
1049 }
1050 }
1051
1052 src.editor().didWriteSelectionToPasteboard();
1053 }
1054 client().willPerformDragSourceAction(DragSourceAction::Selection, dragOrigin, dataTransfer);
1055 if (!dragImage) {
1056 TextIndicatorData textIndicator;
1057 dragImage = DragImage { dissolveDragImageToFraction(createDragImageForSelection(src, textIndicator), DragImageAlpha) };
1058 if (textIndicator.contentImage)
1059 dragImage.setIndicatorData(textIndicator);
1060 dragLoc = dragLocForSelectionDrag(src);
1061 m_dragOffset = IntPoint(dragOrigin.x() - dragLoc.x(), dragOrigin.y() - dragLoc.y());
1062 }
1063
1064 if (!dragImage)
1065 return false;
1066
1067 if (mustUseLegacyDragClient) {
1068 doSystemDrag(WTFMove(dragImage), dragLoc, dragOrigin, src, state, { });
1069 return true;
1070 }
1071
1072 DragItem dragItem;
1073 dragItem.imageAnchorPoint = dragImageAnchorPointForSelectionDrag(src, mouseDraggedPoint);
1074 dragItem.image = WTFMove(dragImage);
1075 dragItem.data = WTFMove(pasteboardWriterData);
1076
1077 beginDrag(WTFMove(dragItem), src, dragOrigin, mouseDraggedPoint, dataTransfer, DragSourceAction::Selection);
1078
1079 return true;
1080 }
1081
1082 if (!src.document()->securityOrigin().canDisplay(linkURL)) {
1083 src.document()->addConsoleMessage(MessageSource::Security, MessageLevel::Error, "Not allowed to drag local resource: " + linkURL.stringCenterEllipsizedToLength());
1084 return false;
1085 }
1086
1087 if (!imageURL.isEmpty() && image && !image->isNull() && m_dragSourceAction.contains(DragSourceAction::Image)) {
1088 // We shouldn't be starting a drag for an image that can't provide an extension.
1089 // This is an early detection for problems encountered later upon drop.
1090 ASSERT(!image->filenameExtension().isEmpty());
1091
1092#if ENABLE(ATTACHMENT_ELEMENT)
1093 auto attachmentInfo = src.editor().promisedAttachmentInfo(element);
1094#else
1095 PromisedAttachmentInfo attachmentInfo;
1096#endif
1097
1098 if (hasData == HasNonDefaultPasteboardData::No) {
1099 m_draggingImageURL = imageURL;
1100 if (element.isContentRichlyEditable())
1101 selectElement(element);
1102 if (!attachmentInfo)
1103 declareAndWriteDragImage(dataTransfer, element, !linkURL.isEmpty() ? linkURL : imageURL, hitTestResult->altDisplayString());
1104 }
1105
1106 client().willPerformDragSourceAction(DragSourceAction::Image, dragOrigin, dataTransfer);
1107
1108 if (!dragImage)
1109 doImageDrag(element, dragOrigin, hitTestResult->imageRect(), src, m_dragOffset, state, WTFMove(attachmentInfo));
1110 else {
1111 // DHTML defined drag image
1112 doSystemDrag(WTFMove(dragImage), dragLoc, dragOrigin, src, state, WTFMove(attachmentInfo));
1113 }
1114
1115 return true;
1116 }
1117
1118 if (!linkURL.isEmpty() && m_dragSourceAction.contains(DragSourceAction::Link)) {
1119 PasteboardWriterData pasteboardWriterData;
1120
1121 String textContentWithSimplifiedWhiteSpace = hitTestResult->textContent().simplifyWhiteSpace();
1122
1123 if (hasData == HasNonDefaultPasteboardData::No) {
1124 // Simplify whitespace so the title put on the dataTransfer resembles what the user sees
1125 // on the web page. This includes replacing newlines with spaces.
1126 if (mustUseLegacyDragClient)
1127 src.editor().copyURL(linkURL, textContentWithSimplifiedWhiteSpace, dataTransfer.pasteboard());
1128 else
1129 pasteboardWriterData.setURLData(src.editor().pasteboardWriterURL(linkURL, textContentWithSimplifiedWhiteSpace));
1130 } else if (dataTransfer.pasteboard().canWriteTrustworthyWebURLsPboardType()) {
1131 // Make sure the pasteboard also contains trustworthy link data
1132 // but don't overwrite more general pasteboard types.
1133 PasteboardURL pasteboardURL;
1134 pasteboardURL.url = linkURL;
1135 pasteboardURL.title = hitTestResult->textContent();
1136 dataTransfer.pasteboard().writeTrustworthyWebURLsPboardType(pasteboardURL);
1137 }
1138
1139 const VisibleSelection& sourceSelection = src.selection().selection();
1140 if (sourceSelection.isCaret() && sourceSelection.isContentEditable()) {
1141 // a user can initiate a drag on a link without having any text
1142 // selected. In this case, we should expand the selection to
1143 // the enclosing anchor element
1144 Position pos = sourceSelection.base();
1145 Node* node = enclosingAnchorElement(pos);
1146 if (node)
1147 src.selection().setSelection(VisibleSelection::selectionFromContentsOfNode(node));
1148 }
1149
1150 client().willPerformDragSourceAction(DragSourceAction::Link, dragOrigin, dataTransfer);
1151 if (!dragImage) {
1152 TextIndicatorData textIndicator;
1153 dragImage = DragImage { createDragImageForLink(element, linkURL, textContentWithSimplifiedWhiteSpace, textIndicator, src.settings().fontRenderingMode(), m_page.deviceScaleFactor()) };
1154 if (dragImage) {
1155 m_dragOffset = dragOffsetForLinkDragImage(dragImage.get());
1156 dragLoc = IntPoint(dragOrigin.x() + m_dragOffset.x(), dragOrigin.y() + m_dragOffset.y());
1157 dragImage = DragImage { platformAdjustDragImageForDeviceScaleFactor(dragImage.get(), m_page.deviceScaleFactor()) };
1158 if (textIndicator.contentImage)
1159 dragImage.setIndicatorData(textIndicator);
1160 }
1161 }
1162
1163 if (mustUseLegacyDragClient) {
1164 doSystemDrag(WTFMove(dragImage), dragLoc, dragOrigin, src, state, { });
1165 return true;
1166 }
1167
1168 DragItem dragItem;
1169 dragItem.imageAnchorPoint = dragImage ? anchorPointForLinkDragImage(dragImage.get()) : FloatPoint();
1170 dragItem.image = WTFMove(dragImage);
1171 dragItem.data = WTFMove(pasteboardWriterData);
1172
1173 beginDrag(WTFMove(dragItem), src, dragOrigin, mouseDraggedPoint, dataTransfer, DragSourceAction::Selection);
1174
1175 return true;
1176 }
1177
1178#if ENABLE(ATTACHMENT_ELEMENT)
1179 if (is<HTMLAttachmentElement>(element) && m_dragSourceAction.contains(DragSourceAction::Attachment)) {
1180 auto& attachment = downcast<HTMLAttachmentElement>(element);
1181 auto* attachmentRenderer = attachment.renderer();
1182
1183 src.editor().setIgnoreSelectionChanges(true);
1184 auto previousSelection = src.selection().selection();
1185 selectElement(element);
1186
1187 PromisedAttachmentInfo promisedAttachment;
1188 if (hasData == HasNonDefaultPasteboardData::No) {
1189 auto& editor = src.editor();
1190 promisedAttachment = editor.promisedAttachmentInfo(attachment);
1191#if PLATFORM(COCOA)
1192 if (!promisedAttachment && editor.client()) {
1193 // Otherwise, if no file URL is specified, call out to the injected bundle to populate the pasteboard with data.
1194 editor.willWriteSelectionToPasteboard(src.selection().selection().toNormalizedRange());
1195 editor.writeSelectionToPasteboard(dataTransfer.pasteboard());
1196 editor.didWriteSelectionToPasteboard();
1197 }
1198#endif
1199 }
1200
1201 client().willPerformDragSourceAction(DragSourceAction::Attachment, dragOrigin, dataTransfer);
1202
1203 if (!dragImage) {
1204 TextIndicatorData textIndicator;
1205 if (attachmentRenderer)
1206 attachmentRenderer->setShouldDrawBorder(false);
1207 dragImage = DragImage { dissolveDragImageToFraction(createDragImageForSelection(src, textIndicator), DragImageAlpha) };
1208 if (attachmentRenderer)
1209 attachmentRenderer->setShouldDrawBorder(true);
1210 if (textIndicator.contentImage)
1211 dragImage.setIndicatorData(textIndicator);
1212 dragLoc = dragLocForSelectionDrag(src);
1213 m_dragOffset = IntPoint(dragOrigin.x() - dragLoc.x(), dragOrigin.y() - dragLoc.y());
1214 }
1215 doSystemDrag(WTFMove(dragImage), dragLoc, dragOrigin, src, state, WTFMove(promisedAttachment));
1216 if (!element.isContentRichlyEditable())
1217 src.selection().setSelection(previousSelection);
1218 src.editor().setIgnoreSelectionChanges(false);
1219 return true;
1220 }
1221#endif
1222
1223#if ENABLE(INPUT_TYPE_COLOR)
1224 bool isColorControl = is<HTMLInputElement>(state.source) && downcast<HTMLInputElement>(*state.source).isColorControl();
1225 if (isColorControl && m_dragSourceAction.contains(DragSourceAction::Color)) {
1226 auto& input = downcast<HTMLInputElement>(*state.source);
1227 auto color = input.valueAsColor();
1228
1229 Path visiblePath;
1230 dragImage = DragImage { createDragImageForColor(color, input.boundsInRootViewSpace(), input.document().page()->pageScaleFactor(), visiblePath) };
1231 dragImage.setVisiblePath(visiblePath);
1232 dataTransfer.pasteboard().write(color);
1233 dragImageOffset = IntPoint { dragImageSize(dragImage.get()) };
1234 dragLoc = dragLocForDHTMLDrag(mouseDraggedPoint, dragOrigin, dragImageOffset, false);
1235
1236 client().willPerformDragSourceAction(DragSourceAction::Color, dragOrigin, dataTransfer);
1237 doSystemDrag(WTFMove(dragImage), dragLoc, dragOrigin, src, state, { });
1238 return true;
1239 }
1240#endif
1241
1242 if (state.type == DragSourceAction::DHTML && dragImage) {
1243 ASSERT(m_dragSourceAction.contains(DragSourceAction::DHTML));
1244 client().willPerformDragSourceAction(DragSourceAction::DHTML, dragOrigin, dataTransfer);
1245 doSystemDrag(WTFMove(dragImage), dragLoc, dragOrigin, src, state, { });
1246 return true;
1247 }
1248
1249 return false;
1250}

1252void DragController::doImageDrag(Element& element, const IntPoint& dragOrigin, const IntRect& layoutRect, Frame& frame, IntPoint& dragImageOffset, const DragState& state, PromisedAttachmentInfo&& attachmentInfo)
1253{
1254 IntPoint mouseDownPoint = dragOrigin;
1255 DragImage dragImage;
1256 IntPoint scaledOrigin;
1257
1258 if (!element.renderer())
1259 return;
1260
1261 ImageOrientation orientation = element.renderer()->imageOrientation();
1262
1263 Image* image = getImage(element);
1264 if (image && !layoutRect.isEmpty() && shouldUseCachedImageForDragImage(*image) && (dragImage = DragImage { createDragImageFromImage(image, orientation) })) {
1265 dragImage = DragImage { fitDragImageToMaxSize(dragImage.get(), layoutRect.size(), maxDragImageSize()) };
1266 IntSize fittedSize = dragImageSize(dragImage.get());
1267
1268 dragImage = DragImage { platformAdjustDragImageForDeviceScaleFactor(dragImage.get(), m_page.deviceScaleFactor()) };
1269 dragImage = DragImage { dissolveDragImageToFraction(dragImage.get(), DragImageAlpha) };
1270
1271 // Properly orient the drag image and orient it differently if it's smaller than the original.
1272 float scale = fittedSize.width() / (float)layoutRect.width();
1273 float dx = scale * (layoutRect.x() - mouseDownPoint.x());
1274 float originY = layoutRect.y();
1275#if PLATFORM(COCOA)
1276 // Compensate for accursed flipped coordinates in Cocoa.

1324void DragController::doSystemDrag(DragImage image, const IntPoint& dragLoc, const IntPoint& eventPos, Frame& frame, const DragState& state, PromisedAttachmentInfo&& promisedAttachmentInfo)
1325{
1326 m_didInitiateDrag = true;
1327 m_dragInitiator = frame.document();
1328 // Protect this frame and view, as a load may occur mid drag and attempt to unload this frame
1329 Ref<Frame> frameProtector(m_page.mainFrame());
1330 RefPtr<FrameView> viewProtector = frameProtector->view();
1331
1332 DragItem item;
1333 item.image = WTFMove(image);
1334 ASSERT(state.type.hasExactlyOneBitSet());
1335 item.sourceAction = state.type.toSingleValue();
1336 item.promisedAttachmentInfo = WTFMove(promisedAttachmentInfo);
1337
1338 auto eventPositionInRootViewCoordinates = frame.view()->contentsToRootView(eventPos);
1339 auto dragLocationInRootViewCoordinates = frame.view()->contentsToRootView(dragLoc);
1340 item.eventPositionInContentCoordinates = viewProtector->rootViewToContents(eventPositionInRootViewCoordinates);
1341 item.dragLocationInContentCoordinates = viewProtector->rootViewToContents(dragLocationInRootViewCoordinates);
1342 item.dragLocationInWindowCoordinates = viewProtector->contentsToWindow(item.dragLocationInContentCoordinates);
1343 if (auto element = state.source) {
1344 auto dataTransferImageElement = state.dataTransfer->dragImageElement();
1345 if (state.type == DragSourceAction::DHTML) {

1347 IntSize dragPreviewSize;

1349 dragPreviewSize = dataTransferImageElement->boundsInRootViewSpace().size();
1350 else {
1351 dragPreviewSize = dragImageSize(item.image.get());
1352 if (auto* page = frame.page())
1353 dragPreviewSize.scale(1 / page->deviceScaleFactor());
1354 }
1355 item.dragPreviewFrameInRootViewCoordinates = { dragLocationInRootViewCoordinates, WTFMove(dragPreviewSize) };
1356 } else {
1357 // We can position the preview using the bounds of the drag source element.
1358 item.dragPreviewFrameInRootViewCoordinates = element->boundsInRootViewSpace();
1359 }
1360
1361 if (auto link = containingLinkElement(*element)) {
1362 auto titleAttribute = link->attributeWithoutSynchronization(HTMLNames::titleAttr);
1363 item.title = titleAttribute.isEmpty() ? link->innerText() : titleAttribute.string();
1364 item.url = frame.document()->completeURL(stripLeadingAndTrailingHTMLSpaces(link->getAttribute(HTMLNames::hrefAttr)));

1422void DragController::insertDroppedImagePlaceholdersAtCaret(const Vector<IntSize>& imageSizes)
1423{
1424 auto& caretController = m_page.dragCaretController();
1425 if (!caretController.isContentRichlyEditable())
1426 return;
1427
1428 auto dropCaret = caretController.caretPosition();
1429 if (dropCaret.isNull())
1430 return;
1431
1432 auto document = makeRefPtr(dropCaret.deepEquivalent().document());
1433 if (!document)
1434 return;
1435
1436 auto frame = makeRefPtr(document->frame());
1437 if (!frame)
1438 return;
1439
1440 IgnoreSelectionChangeForScope selectionChange { *frame };
1441
1442 auto fragment = DocumentFragment::create(*document);
1443 for (auto& size : imageSizes) {
1444 ASSERT(!size.isEmpty());
1445 auto image = HTMLImageElement::create(*document);
1446 image->setAttributeWithoutSynchronization(HTMLNames::widthAttr, AtomString::number(size.width()));
1447 image->setAttributeWithoutSynchronization(HTMLNames::heightAttr, AtomString::number(size.height()));
1448 image->setInlineStyleProperty(CSSPropertyMaxWidth, 100, CSSUnitType::CSS_PERCENTAGE);
1449 image->setInlineStyleProperty(CSSPropertyBackgroundColor, serializationForCSS(Color::black.colorWithAlphaByte(13)));
1450 image->setIsDroppedImagePlaceholder();
1451 fragment->appendChild(WTFMove(image));
1452 }
1453
1454 frame->selection().setSelection(dropCaret);
1455
1456 auto command = ReplaceSelectionCommand::create(*document, WTFMove(fragment), { ReplaceSelectionCommand::PreventNesting, ReplaceSelectionCommand::SmartReplace }, EditAction::InsertFromDrop);
1457 command->apply();
1458
1459 auto insertedContentRange = command->insertedContentRange();
1460 if (!insertedContentRange) {
1461 ASSERT_NOT_REACHED();
1462 return;
1463 }
1464
1465 auto container = commonInclusiveAncestor<ComposedTree>(*insertedContentRange);
1466 if (!is<ContainerNode>(container)) {
1467 ASSERT_NOT_REACHED();
1468 return;
1469 }
1470
1471 Vector<Ref<HTMLImageElement>> placeholders;
1472 for (auto& placeholder : descendantsOfType<HTMLImageElement>(downcast<ContainerNode>(*container))) {
1473 if (intersects<ComposedTree>(*insertedContentRange, placeholder))
1474 placeholders.append(placeholder);
1475 }
1476
1477 if (placeholders.size() != imageSizes.size()) {
1478 ASSERT_NOT_REACHED();
1479 return;
1480 }
1481
1482 for (size_t i = 0; i < placeholders.size(); ++i) {
1483 auto& placeholder = placeholders[i];
1484 auto imageSize = imageSizes[i];
1485 double clientWidth = placeholder->clientWidth();
1486 double heightRespectingAspectRatio = (imageSize.height() * clientWidth) / imageSize.width();

1488 }

1490 document->updateLayout();

1492 m_droppedImagePlaceholders = WTFMove(placeholders);
1493 m_droppedImagePlaceholderRange = WTFMove(insertedContentRange);
1494
1495 frame->selection().clear();
1496 caretController.setCaretPosition(makeDeprecatedLegacyPosition(m_droppedImagePlaceholderRange->start));
1497}
