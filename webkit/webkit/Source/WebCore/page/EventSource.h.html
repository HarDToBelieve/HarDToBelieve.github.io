43namespace WebCore {
44
45class MessageEvent;
46class TextResourceDecoder;
47class ThreadableLoader;
48
49class EventSource final : public RefCounted<EventSource>, public EventTargetWithInlineData, private ThreadableLoaderClient, public ActiveDOMObject {
50 WTF_MAKE_ISO_ALLOCATED(EventSource);
51public:
52 struct Init {
53 bool withCredentials;
54 };
55 static ExceptionOr<Ref<EventSource>> create(ScriptExecutionContext&, const String& url, const Init&);
56 virtual ~EventSource();
57
58 const String& url() const;
59 bool withCredentials() const;
60
61 using State = short;
62 static const State CONNECTING = 0;
63 static const State OPEN = 1;
64 static const State CLOSED = 2;
65
66 State readyState() const;
67
68 void close();
69
70 using RefCounted::ref;
71 using RefCounted::deref;
72
73private:
74 EventSource(ScriptExecutionContext&, const URL&, const Init&);
75
76 EventTargetInterface eventTargetInterface() const final { return EventSourceEventTargetInterfaceType; }
77 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
78
79 void refEventTarget() final { ref(); }
80 void derefEventTarget() final { deref(); }
81
82 void dispatchErrorEvent();
83 void doExplicitLoadCancellation();
84
85 // ThreadableLoaderClient
86 void didReceiveResponse(unsigned long, const ResourceResponse&) final;
87 void didReceiveData(const uint8_t*, int) final;
88 void didFinishLoading(unsigned long) final;
89 void didFail(const ResourceError&) final;
90
91 // ActiveDOMObject
92 void stop() final;
93 const char* activeDOMObjectName() const final;
94 void suspend(ReasonForSuspension) final;
95 void resume() final;
96 bool virtualHasPendingActivity() const final;
97
98 void connect();
99 void networkRequestEnded();
100 void scheduleInitialConnect();
101 void scheduleReconnect();
102 void abortConnectionAttempt();
103 void parseEventStream();
104 void parseEventStreamLine(unsigned position, std::optional<unsigned> fieldLength, unsigned lineLength);
105 void dispatchMessageEvent();
106
107 bool responseIsValid(const ResourceResponse&) const;
108
109 static const uint64_t defaultReconnectDelay;
110
111 URL m_url;
112 bool m_withCredentials;
113 State m_state { CONNECTING };
114
115 Ref<TextResourceDecoder> m_decoder;
116 RefPtr<ThreadableLoader> m_loader;
117 SuspendableTimer m_connectTimer;
118 Vector<UChar> m_receiveBuffer;
119 bool m_discardTrailingNewline { false };
120 bool m_requestInFlight { false };
121 bool m_isSuspendedForBackForwardCache { false };
122 bool m_isDoingExplicitCancellation { false };
123 bool m_shouldReconnectOnResume { false };
124
125 AtomString m_eventName;
126 Vector<UChar> m_data;
127 String m_currentlyParsedEventId;
128 String m_lastEventId;
129 uint64_t m_reconnectDelay { defaultReconnectDelay };
130 String m_eventStreamOrigin;
131};
132
133inline const String& EventSource::url() const
134{
135 return m_url.string();
136}
137
138inline bool EventSource::withCredentials() const
139{
140 return m_withCredentials;
141}
142
143inline EventSource::State EventSource::readyState() const
144{
145 return m_state;
146}
