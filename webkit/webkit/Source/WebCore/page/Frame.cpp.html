153Frame::Frame(Page& page, HTMLFrameOwnerElement* ownerElement, UniqueRef<FrameLoaderClient>&& frameLoaderClient)
154 : m_mainFrame(ownerElement ? page.mainFrame() : *this)
155 , m_page(makeWeakPtr(page))
156 , m_settings(&page.settings())
157 , m_treeNode(*this, parentFromOwnerElement(ownerElement))
158 , m_loader(makeUniqueRef<FrameLoader>(*this, WTFMove(frameLoaderClient)))
159 , m_navigationScheduler(makeUniqueRef<NavigationScheduler>(*this))
160 , m_ownerElement(makeWeakPtr(ownerElement))
161 , m_script(makeUniqueRef<ScriptController>(*this))
162 , m_pageZoomFactor(parentPageZoomFactor(this))
163 , m_textZoomFactor(parentTextZoomFactor(this))
164 , m_eventHandler(makeUniqueRef<EventHandler>(*this))
165{
166 ProcessWarming::initializeNames();
167
168 if (ownerElement) {
169 m_mainFrame.selfOnlyRef();
170 ownerElement->setContentFrame(*this);
171 }
172
173#ifndef NDEBUG
174 frameCounter.increment();
175#endif
176
177 // Pause future ActiveDOMObjects if this frame is being created while the page is in a paused state.
178 Frame* parent = parentFromOwnerElement(ownerElement);
179 if (parent && parent->activeDOMObjectsAndAnimationsSuspended())
180 suspendActiveDOMObjectsAndAnimations();
181}

194Frame::~Frame()
195{
196 setView(nullptr);
197 navigationScheduler().cancel();
198
199 if (!loader().isComplete())
200 loader().closeURL();
201
202 loader().clear(document(), false);
203 script().updatePlatformScriptObjects();
204
205 // FIXME: We should not be doing all this work inside the destructor
206
207#ifndef NDEBUG
208 frameCounter.decrement();
209#endif
210
211 disconnectOwnerElement();
212
213 while (auto* destructionObserver = m_destructionObservers.takeAny())
214 destructionObserver->frameDestroyed();
215
216 if (!isMainFrame())
217 m_mainFrame.selfOnlyDeref();
218}

240void Frame::setView(RefPtr<FrameView>&& view)
241{
242 // We the custom scroll bars as early as possible to prevent m_doc->detach()
243 // from messing with the view such that its scroll bars won't be torn down.
244 // FIXME: We should revisit this.
245 if (m_view)
246 m_view->prepareForDetach();
247
248 // Prepare for destruction now, so any unload event handlers get run and the DOMWindow is
249 // notified. If we wait until the view is destroyed, then things won't be hooked up enough for
250 // these calls to work.
251 if (!view && m_doc && m_doc->backForwardCacheState() != Document::InBackForwardCache)
252 m_doc->willBeRemovedFromFrame();
253
254 if (m_view)
255 m_view->layoutContext().unscheduleLayout();
256
257 m_eventHandler->clear();
258
259 RELEASE_ASSERT(!m_doc || !m_doc->hasLivingRenderTree());
260
261 m_view = WTFMove(view);
262
263 // Only one form submission is allowed per view of a part.
264 // Since this part may be getting reused as a result of being
265 // pulled from the back/forward cache, reset this flag.
266 loader().resetMultipleFormSubmissionProtection();
267}

269void Frame::setDocument(RefPtr<Document>&& newDocument)
270{
271 ASSERT(!newDocument || newDocument->frame() == this);
272
273 if (m_documentIsBeingReplaced)
274 return;
275
276 m_documentIsBeingReplaced = true;
277
278 if (isMainFrame()) {
279 if (m_page)
280 m_page->didChangeMainDocument();
281 m_loader->client().dispatchDidChangeMainDocument();
282
283 // We want to generate the same unique names whenever a page is loaded to avoid making layout tests
284 // flaky and for things like form state restoration to work. To achieve this, we reset our frame
285 // identifier generator every time the page is navigated.
286 tree().resetFrameIdentifiers();
287 }
288
289#if ENABLE(ATTACHMENT_ELEMENT)
290 if (m_doc) {
291 for (auto& attachment : m_doc->attachmentElementsByIdentifier().values())
292 editor().didRemoveAttachmentElement(attachment);
293 }
294#endif
295
296 if (m_doc && m_doc->backForwardCacheState() != Document::InBackForwardCache)
297 m_doc->willBeRemovedFromFrame();
298
299 m_doc = newDocument.copyRef();
300 ASSERT(!m_doc || m_doc->domWindow());
301 ASSERT(!m_doc || m_doc->domWindow()->frame() == this);
302
303 // Don't use m_doc because it can be overwritten and we want to guarantee
304 // that the document is not destroyed during this function call.
305 if (newDocument)
306 newDocument->didBecomeCurrentDocumentInFrame();
307
308#if ENABLE(ATTACHMENT_ELEMENT)
309 if (m_doc) {
310 for (auto& attachment : m_doc->attachmentElementsByIdentifier().values())
311 editor().didInsertAttachmentElement(attachment);
312 }
313#endif
314
315 if (m_page && m_doc && isMainFrame() && !loader().stateMachine().isDisplayingInitialEmptyDocument())
316 m_page->mainFrameDidChangeToNonInitialEmptyDocument();
317
318 InspectorInstrumentation::frameDocumentUpdated(*this);
319
320 m_documentIsBeingReplaced = false;
321}

323void Frame::invalidateContentEventRegionsIfNeeded(InvalidateContentEventRegionsReason reason)
324{
325 if (!m_page || !m_doc || !m_doc->renderView())
326 return;
327
328 bool needsUpdateForWheelEventHandlers = false;
329 bool needsUpdateForTouchActionElements = false;
330 bool needsUpdateForEditableElements = false;
331#if ENABLE(WHEEL_EVENT_REGIONS)
332 needsUpdateForWheelEventHandlers = m_doc->hasWheelEventHandlers() || reason == InvalidateContentEventRegionsReason::EventHandlerChange;
333#else
334 UNUSED_PARAM(reason);
335#endif
336#if ENABLE(TOUCH_ACTION_REGIONS)
337 // Document::mayHaveElementsWithNonAutoTouchAction never changes from true to false currently.
338 needsUpdateForTouchActionElements = m_doc->mayHaveElementsWithNonAutoTouchAction();
339#endif
340#if ENABLE(EDITABLE_REGION)
341 // Document::mayHaveEditableElements never changes from true to false currently.
342 needsUpdateForEditableElements = m_doc->mayHaveEditableElements() && m_page->shouldBuildEditableRegion();
343#endif
344 if (!needsUpdateForTouchActionElements && !needsUpdateForEditableElements && !needsUpdateForWheelEventHandlers)
345 return;
346
347 if (!m_doc->renderView()->compositor().viewNeedsToInvalidateEventRegionOfEnclosingCompositingLayerForRepaint())
348 return;
349
350 if (m_ownerElement)
351 m_ownerElement->document().invalidateEventRegionsForFrame(*m_ownerElement);
352}

376static JSC::Yarr::RegularExpression createRegExpForLabels(const Vector<String>& labels)
377{
378 // REVIEW- version of this call in FrameMac.mm caches based on the NSArray ptrs being
379 // the same across calls. We can't do that.
380
381 static NeverDestroyed<JSC::Yarr::RegularExpression> wordRegExp("\\w");
382 StringBuilder pattern;
383 pattern.append('(');
384 for (unsigned i = 0, numLabels = labels.size(); i < numLabels; i++) {
385 auto& label = labels[i];
386
387 bool startsWithWordCharacter = false;
388 bool endsWithWordCharacter = false;
389 if (label.length()) {
390 startsWithWordCharacter = wordRegExp.get().match(label.substring(0, 1)) >= 0;
391 endsWithWordCharacter = wordRegExp.get().match(label.substring(label.length() - 1, 1)) >= 0;
392 }
393
394 // Search for word boundaries only if label starts/ends with "word characters".
395 // If we always searched for word boundaries, this wouldn't work for languages such as Japanese.
396 pattern.append(i ? "|" : "", startsWithWordCharacter ? "\\b" : "", label, endsWithWordCharacter ? "\\b" : "");
397 }
398 pattern.append(')');
399 return JSC::Yarr::RegularExpression(pattern.toString(), JSC::Yarr::TextCaseInsensitive);
400}

402String Frame::searchForLabelsAboveCell(const JSC::Yarr::RegularExpression& regExp, HTMLTableCellElement* cell, size_t* resultDistanceFromStartOfCell)
403{
404 HTMLTableCellElement* aboveCell = cell->cellAbove();
405 if (aboveCell) {
406 // search within the above cell we found for a match
407 size_t lengthSearched = 0;
408 for (Text* textNode = TextNodeTraversal::firstWithin(*aboveCell); textNode; textNode = TextNodeTraversal::next(*textNode, aboveCell)) {
409 if (!textNode->renderer() || textNode->renderer()->style().visibility() != Visibility::Visible)
410 continue;
411 // For each text chunk, run the regexp
412 String nodeString = textNode->data();
413 int pos = regExp.searchRev(nodeString);
414 if (pos >= 0) {
415 if (resultDistanceFromStartOfCell)
416 *resultDistanceFromStartOfCell = lengthSearched;
417 return nodeString.substring(pos, regExp.matchedLength());
418 }
419 lengthSearched += nodeString.length();
420 }
421 }
422
423 // Any reason in practice to search all cells in that are above cell?
424 if (resultDistanceFromStartOfCell)
425 *resultDistanceFromStartOfCell = notFound;
426 return String();
427}

429// FIXME: This should take an Element&.
430String Frame::searchForLabelsBeforeElement(const Vector<String>& labels, Element* element, size_t* resultDistance, bool* resultIsInCellAbove)
431{
432 ASSERT(element);
433 JSC::Yarr::RegularExpression regExp = createRegExpForLabels(labels);
434 // We stop searching after we've seen this many chars
435 const unsigned int charsSearchedThreshold = 500;
436 // This is the absolute max we search. We allow a little more slop than
437 // charsSearchedThreshold, to make it more likely that we'll search whole nodes.
438 const unsigned int maxCharsSearched = 600;
439 // If the starting element is within a table, the cell that contains it
440 HTMLTableCellElement* startingTableCell = nullptr;
441 bool searchedCellAbove = false;
442
443 if (resultDistance)
444 *resultDistance = notFound;
445 if (resultIsInCellAbove)
446 *resultIsInCellAbove = false;
447
448 // walk backwards in the node tree, until another element, or form, or end of tree
449 int unsigned lengthSearched = 0;
450 Node* n;
451 for (n = NodeTraversal::previous(*element); n && lengthSearched < charsSearchedThreshold; n = NodeTraversal::previous(*n)) {
452 // We hit another form element or the start of the form - bail out
453 if (is<HTMLFormElement>(*n) || is<HTMLFormControlElement>(*n))
454 break;
455
456 if (n->hasTagName(tdTag) && !startingTableCell)
457 startingTableCell = downcast<HTMLTableCellElement>(n);
458 else if (is<HTMLTableRowElement>(*n) && startingTableCell) {
459 String result = searchForLabelsAboveCell(regExp, startingTableCell, resultDistance);
460 if (!result.isEmpty()) {
461 if (resultIsInCellAbove)
462 *resultIsInCellAbove = true;
463 return result;
464 }
465 searchedCellAbove = true;
466 } else if (n->isTextNode() && n->renderer() && n->renderer()->style().visibility() == Visibility::Visible) {
467 // For each text chunk, run the regexp
468 String nodeString = n->nodeValue();
469 // add 100 for slop, to make it more likely that we'll search whole nodes
470 if (lengthSearched + nodeString.length() > maxCharsSearched)
471 nodeString = nodeString.right(charsSearchedThreshold - lengthSearched);
472 int pos = regExp.searchRev(nodeString);
473 if (pos >= 0) {
474 if (resultDistance)
475 *resultDistance = lengthSearched;
476 return nodeString.substring(pos, regExp.matchedLength());
477 }
478 lengthSearched += nodeString.length();
479 }
480 }
481
482 // If we started in a cell, but bailed because we found the start of the form or the
483 // previous element, we still might need to search the row above us for a label.
484 if (startingTableCell && !searchedCellAbove) {
485 String result = searchForLabelsAboveCell(regExp, startingTableCell, resultDistance);
486 if (!result.isEmpty()) {
487 if (resultIsInCellAbove)
488 *resultIsInCellAbove = true;
489 return result;
490 }
491 }
492 return String();
493}

495static String matchLabelsAgainstString(const Vector<String>& labels, const String& stringToMatch)
496{
497 if (stringToMatch.isEmpty())
498 return String();
499
500 String mutableStringToMatch = stringToMatch;
501
502 // Make numbers and _'s in field names behave like word boundaries, e.g., "address2"
503 replace(mutableStringToMatch, JSC::Yarr::RegularExpression("\\d"), " ");
504 mutableStringToMatch.replace('_', ' ');
505
506 JSC::Yarr::RegularExpression regExp = createRegExpForLabels(labels);
507 // Use the largest match we can find in the whole string
508 int pos;
509 int length;
510 int bestPos = -1;
511 int bestLength = -1;
512 int start = 0;
513 do {
514 pos = regExp.match(mutableStringToMatch, start);
515 if (pos != -1) {
516 length = regExp.matchedLength();
517 if (length >= bestLength) {
518 bestPos = pos;
519 bestLength = length;
520 }
521 start = pos + 1;
522 }
523 } while (pos != -1);
524
525 if (bestPos != -1)
526 return mutableStringToMatch.substring(bestPos, bestLength);
527 return String();
528}

530String Frame::matchLabelsAgainstElement(const Vector<String>& labels, Element* element)
531{
532 // Match against the name element, then against the id element if no match is found for the name element.
533 // See 7538330 for one popular site that benefits from the id element check.
534 // FIXME: This code is mirrored in FrameMac.mm. It would be nice to make the Mac code call the platform-agnostic
535 // code, which would require converting the NSArray of NSStrings to a Vector of Strings somewhere along the way.
536 String resultFromNameAttribute = matchLabelsAgainstString(labels, element->getNameAttribute());
537 if (!resultFromNameAttribute.isEmpty())
538 return resultFromNameAttribute;
539
540 return matchLabelsAgainstString(labels, element->attributeWithoutSynchronization(idAttr));
541}

556bool Frame::requestDOMPasteAccess()
557{
558 if (m_settings->javaScriptCanAccessClipboard() && m_settings->domPasteAllowed())
559 return true;
560
561 if (!m_doc)
562 return false;
563
564 if (editor().isPastingFromMenuOrKeyBinding())
565 return true;
566
567 if (!m_settings->domPasteAccessRequestsEnabled())
568 return false;
569
570 auto gestureToken = UserGestureIndicator::currentUserGesture();
571 if (!gestureToken || !gestureToken->processingUserGesture())
572 return false;
573
574 switch (gestureToken->domPasteAccessPolicy()) {
575 case DOMPasteAccessPolicy::Granted:
576 return true;
577 case DOMPasteAccessPolicy::Denied:
578 return false;
579 case DOMPasteAccessPolicy::NotRequestedYet: {
580 auto* client = editor().client();
581 if (!client)
582 return false;
583
584 auto response = client->requestDOMPasteAccess(m_doc->originIdentifierForPasteboard());
585 gestureToken->didRequestDOMPasteAccess(response);
586 switch (response) {
587 case DOMPasteAccessResponse::GrantedForCommand:
588 case DOMPasteAccessResponse::GrantedForGesture:
589 return true;
590 case DOMPasteAccessResponse::DeniedForGesture:
591 return false;
592 }
593 }
594 }
595
596 ASSERT_NOT_REACHED();
597 return false;
598}

600void Frame::setPrinting(bool printing, const FloatSize& pageSize, const FloatSize& originalPageSize, float maximumShrinkRatio, AdjustViewSizeOrNot shouldAdjustViewSize)
601{
602 if (!view())
603 return;
604 // In setting printing, we should not validate resources already cached for the document.
605 // See https://bugs.webkit.org/show_bug.cgi?id=43704
606 ResourceCacheValidationSuppressor validationSuppressor(m_doc->cachedResourceLoader());
607
608 m_doc->setPrinting(printing);
609 view()->adjustMediaTypeForPrinting(printing);
610
611 // FIXME: Consider invoking Page::updateRendering or an equivalent.
612 m_doc->styleScope().didChangeStyleSheetEnvironment();
613 m_doc->evaluateMediaQueriesAndReportChanges();
614 if (!view())
615 return;
616
617 auto& frameView = *view();
618 if (shouldUsePrintingLayout())
619 frameView.forceLayoutForPagination(pageSize, originalPageSize, maximumShrinkRatio, shouldAdjustViewSize);
620 else {
621 frameView.forceLayout();
622 if (shouldAdjustViewSize == AdjustViewSize)
623 frameView.adjustViewSize();
624 }
625
626 // Subframes of the one we're printing don't lay out to the page size.
627 for (RefPtr<Frame> child = tree().firstChild(); child; child = child->tree().nextSibling())
628 child->setPrinting(printing, FloatSize(), FloatSize(), 0, shouldAdjustViewSize);
629}

631bool Frame::shouldUsePrintingLayout() const
632{
633 // Only top frame being printed should be fit to page size.
634 // Subframes should be constrained by parents only.
635 return m_doc->printing() && (!tree().parent() || !tree().parent()->m_doc->printing());
636}

638FloatSize Frame::resizePageRectsKeepingRatio(const FloatSize& originalSize, const FloatSize& expectedSize)
639{
640 FloatSize resultSize;
641 if (!contentRenderer())
642 return FloatSize();
643
644 if (contentRenderer()->style().isHorizontalWritingMode()) {
645 ASSERT(fabs(originalSize.width()) > std::numeric_limits<float>::epsilon());
646 float ratio = originalSize.height() / originalSize.width();
647 resultSize.setWidth(floorf(expectedSize.width()));
648 resultSize.setHeight(floorf(resultSize.width() * ratio));
649 } else {
650 ASSERT(fabs(originalSize.height()) > std::numeric_limits<float>::epsilon());
651 float ratio = originalSize.width() / originalSize.height();
652 resultSize.setHeight(floorf(expectedSize.height()));
653 resultSize.setWidth(floorf(resultSize.height() * ratio));
654 }
655 return resultSize;
656}

658void Frame::injectUserScripts(UserScriptInjectionTime injectionTime)
659{
660 if (!m_page)
661 return;
662
663 if (loader().stateMachine().creatingInitialEmptyDocument() && !settings().shouldInjectUserScriptsInInitialEmptyDocument())
664 return;
665
666 bool pageWasNotified = m_page->hasBeenNotifiedToInjectUserScripts();
667 m_page->userContentProvider().forEachUserScript([this, protectedThis = makeRef(*this), injectionTime, pageWasNotified] (DOMWrapperWorld& world, const UserScript& script) {
668 if (script.injectionTime() == injectionTime) {
669 if (script.waitForNotificationBeforeInjecting() == WaitForNotificationBeforeInjecting::Yes && !pageWasNotified)
670 addUserScriptAwaitingNotification(world, script);
671 else
672 injectUserScriptImmediately(world, script);
673 }
674 });
675}

677void Frame::injectUserScriptImmediately(DOMWrapperWorld& world, const UserScript& script)
678{
679#if ENABLE(APP_BOUND_DOMAINS)
680 if (loader().client().shouldEnableInAppBrowserPrivacyProtections()) {
681 if (auto* document = this->document())
682 document->addConsoleMessage(MessageSource::Security, MessageLevel::Warning, "Ignoring user script injection for non-app bound domain."_s);
683 FRAME_RELEASE_LOG_ERROR(Loading, "injectUserScriptImmediately: Ignoring user script injection for non app-bound domain");
684 return;
685 }
686 loader().client().notifyPageOfAppBoundBehavior();
687#endif
688
689 auto* document = this->document();
690 if (!document)
691 return;
692 if (script.injectedFrames() == UserContentInjectedFrames::InjectInTopFrameOnly && !isMainFrame())
693 return;
694 if (!UserContentURLPattern::matchesPatterns(document->url(), script.allowlist(), script.blocklist()))
695 return;
696 if (!m_script->shouldAllowUserAgentScripts(*document))
697 return;
698
699 document->setAsRunningUserScripts();
700 loader().client().willInjectUserScript(world);
701 m_script->evaluateInWorldIgnoringException(ScriptSourceCode(script.source(), URL(script.url())), world);
702}

770void Frame::willDetachPage()
771{
772 if (Frame* parent = tree().parent())
773 parent->loader().checkLoadComplete();
774
775 for (auto& observer : m_destructionObservers)
776 observer->willDetachPage();
777
778 // FIXME: It's unclear as to why this is called more than once, but it is,
779 // so page() could be NULL.
780 if (page() && page()->focusController().focusedFrame() == this)
781 page()->focusController().setFocusedFrame(nullptr);
782
783 if (page() && page()->scrollingCoordinator() && m_view)
784 page()->scrollingCoordinator()->willDestroyScrollableArea(*m_view);
785
786 script().clearScriptObjects();
787 script().updatePlatformScriptObjects();
788
789 // We promise that the Frame is always connected to a Page while the render tree is live.
790 //
791 // The render tree can be torn down in a few different ways, but the two important ones are:
792 //
793 // - When calling Frame::setView() with a null FrameView*. This is always done before calling
794 // Frame::willDetachPage (this function.) Hence the assertion below.
795 //
796 // - When adding a document to the back/forward cache, the tree is torn down before instantiating
797 // the CachedPage+CachedFrame object tree.
798 ASSERT(!document() || !document()->renderView());
799}

848std::optional<SimpleRange> Frame::rangeForPoint(const IntPoint& framePoint)
849{
850 auto position = visiblePositionForPoint(framePoint);
851
852 auto containerText = position.deepEquivalent().containerText();
853 if (!containerText || !containerText->renderer() || containerText->renderer()->style().userSelect() == UserSelect::None)
854 return std::nullopt;
855
856 if (auto previousCharacterRange = makeSimpleRange(position.previous(), position)) {
857 if (editor().firstRectForRange(*previousCharacterRange).contains(framePoint))
858 return *previousCharacterRange;
859 }
860
861 if (auto nextCharacterRange = makeSimpleRange(position, position.next())) {
862 if (editor().firstRectForRange(*nextCharacterRange).contains(framePoint))
863 return *nextCharacterRange;
864 }
865
866 return std::nullopt;
867}

869void Frame::createView(const IntSize& viewportSize, const std::optional<Color>& backgroundColor,
870 const IntSize& fixedLayoutSize, const IntRect& fixedVisibleContentRect,
871 bool useFixedLayout, ScrollbarMode horizontalScrollbarMode, bool horizontalLock,
872 ScrollbarMode verticalScrollbarMode, bool verticalLock)
873{
874 ASSERT(m_page);
875
876 bool isMainFrame = this->isMainFrame();
877
878 if (isMainFrame && view())
879 view()->setParentVisible(false);
880
881 setView(nullptr);
882
883 RefPtr<FrameView> frameView;
884 if (isMainFrame) {
885 frameView = FrameView::create(*this, viewportSize);
886 frameView->setFixedLayoutSize(fixedLayoutSize);
887#if USE(COORDINATED_GRAPHICS)
888 frameView->setFixedVisibleContentRect(fixedVisibleContentRect);
889#else
890 UNUSED_PARAM(fixedVisibleContentRect);
891#endif
892 frameView->setUseFixedLayout(useFixedLayout);
893 } else
894 frameView = FrameView::create(*this);
895
896 frameView->setScrollbarModes(horizontalScrollbarMode, verticalScrollbarMode, horizontalLock, verticalLock);
897
898 setView(frameView.copyRef());
899
900 frameView->updateBackgroundRecursively(backgroundColor);
901
902 if (isMainFrame)
903 frameView->setParentVisible(true);
904
905 if (ownerRenderer())
906 ownerRenderer()->setWidget(frameView);
907
908 if (HTMLFrameOwnerElement* owner = ownerElement())
909 view()->setCanHaveScrollbars(owner->scrollingMode() != ScrollbarAlwaysOff);
910}

939void Frame::setPageAndTextZoomFactors(float pageZoomFactor, float textZoomFactor)
940{
941 if (m_pageZoomFactor == pageZoomFactor && m_textZoomFactor == textZoomFactor)
942 return;
943
944 Page* page = this->page();
945 if (!page)
946 return;
947
948 Document* document = this->document();
949 if (!document)
950 return;
951
952 editor().dismissCorrectionPanelAsIgnored();
953
954 // Respect SVGs zoomAndPan="disabled" property in standalone SVG documents.
955 // FIXME: How to handle compound documents + zoomAndPan="disabled"? Needs SVG WG clarification.
956 if (is<SVGDocument>(*document) && !downcast<SVGDocument>(*document).zoomAndPanEnabled())
957 return;
958
959 std::optional<ScrollPosition> scrollPositionAfterZoomed;
960 if (m_pageZoomFactor != pageZoomFactor) {
961 // Compute the scroll position with scale after zooming to stay the same position in the content.
962 if (FrameView* view = this->view()) {
963 scrollPositionAfterZoomed = view->scrollPosition();
964 scrollPositionAfterZoomed->scale(pageZoomFactor / m_pageZoomFactor);
965 }
966 }
967 m_pageZoomFactor = pageZoomFactor;
968 m_textZoomFactor = textZoomFactor;
969
970 document->resolveStyle(Document::ResolveStyleType::Rebuild);
971
972 for (RefPtr<Frame> child = tree().firstChild(); child; child = child->tree().nextSibling())
973 child->setPageAndTextZoomFactors(m_pageZoomFactor, m_textZoomFactor);
974
975 if (FrameView* view = this->view()) {
976 if (document->renderView() && document->renderView()->needsLayout() && view->didFirstLayout())
977 view->layoutContext().layout();
978
979 // Scrolling to the calculated position must be done after the layout.
980 if (scrollPositionAfterZoomed)
981 view->setScrollPosition(scrollPositionAfterZoomed.value());
982 }
983}

999void Frame::suspendActiveDOMObjectsAndAnimations()
1000{
1001 bool wasSuspended = activeDOMObjectsAndAnimationsSuspended();
1002
1003 m_activeDOMObjectsAndAnimationsSuspendedCount++;
1004
1005 if (wasSuspended)
1006 return;
1007
1008 // FIXME: Suspend/resume calls will not match if the frame is navigated, and gets a new document.

1014void Frame::resumeActiveDOMObjectsAndAnimations()
1015{
1016 if (!activeDOMObjectsAndAnimationsSuspended())
1017 return;
1018
1019 m_activeDOMObjectsAndAnimationsSuspendedCount--;
1020
1021 if (activeDOMObjectsAndAnimationsSuspended())
1022 return;
1023
1024 if (!m_doc)
1025 return;
1026
1027 // FIXME: Suspend/resume calls will not match if the frame is navigated, and gets a new document.
1028 m_doc->resumeScheduledTasks(ReasonForSuspension::PageWillBeSuspended);
1029
1030 // Frame::clearTimers() suspended animations and pending relayouts.
1031
1032 if (auto* timelines = m_doc->timelinesController())
1033 timelines->resumeAnimations();
1034 if (m_view)
1035 m_view->layoutContext().scheduleLayout();
1036}
