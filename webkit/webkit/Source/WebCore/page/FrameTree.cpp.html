51void FrameTree::setName(const AtomString& name)
52{
53 m_name = name;
54 if (!parent()) {
55 m_uniqueName = name;
56 return;
57 }
58 m_uniqueName = nullAtom(); // Remove our old frame name so it's not considered in uniqueChildName.
59 m_uniqueName = parent()->tree().uniqueChildName(name);
60}

73void FrameTree::appendChild(Frame& child)
74{
75 ASSERT(child.page() == m_thisFrame.page());
76 child.tree().m_parent = makeWeakPtr(m_thisFrame);
77 WeakPtr<Frame> oldLast = m_lastChild;
78 m_lastChild = makeWeakPtr(child);
79
80 if (oldLast) {
81 child.tree().m_previousSibling = oldLast;
82 oldLast->tree().m_nextSibling = &child;
83 } else
84 m_firstChild = &child;
85
86 m_scopedChildCount = invalidCount;
87
88 ASSERT(!m_lastChild->tree().m_nextSibling);
89}
90
91void FrameTree::removeChild(Frame& child)
92{
93 WeakPtr<Frame>& newLocationForPrevious = m_lastChild == &child ? m_lastChild : child.tree().m_nextSibling->tree().m_previousSibling;
94 RefPtr<Frame>& newLocationForNext = m_firstChild == &child ? m_firstChild : child.tree().m_previousSibling->tree().m_nextSibling;
95
96 child.tree().m_parent = nullptr;
97 newLocationForPrevious = std::exchange(child.tree().m_previousSibling, nullptr);
98 newLocationForNext = WTFMove(child.tree().m_nextSibling);
99
100 m_scopedChildCount = invalidCount;
101}

103AtomString FrameTree::uniqueChildName(const AtomString& requestedName) const
104{
105 // If the requested name (the frame's "name" attribute) is unique, just use that.
106 if (!requestedName.isEmpty() && !child(requestedName) && !equalIgnoringASCIICase(requestedName, "_blank"))
107 return requestedName;
108
109 // The "name" attribute was not unique or absent. Generate a name based on a counter on the main frame that gets reset
110 // on navigation. The name uses HTML comment syntax to avoid collisions with author names.
111 return generateUniqueName();
112}
113
114AtomString FrameTree::generateUniqueName() const
115{
116 auto& top = this->top();
117 if (&top.tree() != this)
118 return top.tree().generateUniqueName();
119
120 return makeString("<!--frame", ++m_frameIDGenerator, "-->");
121}

123static bool inScope(Frame& frame, TreeScope& scope)
124{
125 Document* document = frame.document();
126 if (!document)
127 return false;
128 HTMLFrameOwnerElement* owner = document->ownerElement();
129 if (!owner)
130 return false;
131 return &owner->treeScope() == &scope;
132}

134inline Frame* FrameTree::scopedChild(unsigned index, TreeScope* scope) const
135{
136 if (!scope)
137 return nullptr;
138
139 unsigned scopedIndex = 0;
140 for (Frame* result = firstChild(); result; result = result->tree().nextSibling()) {
141 if (inScope(*result, *scope)) {
142 if (scopedIndex == index)
143 return result;
144 scopedIndex++;
145 }
146 }
147
148 return nullptr;
149}
150
151inline Frame* FrameTree::scopedChild(const AtomString& name, TreeScope* scope) const
152{
153 if (!scope)
154 return nullptr;
155
156 for (Frame* child = firstChild(); child; child = child->tree().nextSibling()) {
157 if (child->tree().uniqueName() == name && inScope(*child, *scope))
158 return child;
159 }
160 return nullptr;
161}

163inline unsigned FrameTree::scopedChildCount(TreeScope* scope) const
164{
165 if (!scope)
166 return 0;
167
168 unsigned scopedCount = 0;
169 for (Frame* result = firstChild(); result; result = result->tree().nextSibling()) {
170 if (inScope(*result, *scope))
171 scopedCount++;
172 }
173
174 return scopedCount;
175}

226// FrameTree::find() only returns frames in pages that are related to the active
227// page by an opener <-> openee relationship.
228static bool isFrameFamiliarWith(Frame& frameA, Frame& frameB)
229{
230 if (frameA.page() == frameB.page())
231 return true;
232
233 auto* frameAOpener = frameA.mainFrame().loader().opener();
234 auto* frameBOpener = frameB.mainFrame().loader().opener();
235 return (frameAOpener && frameAOpener->page() == frameB.page()) || (frameBOpener && frameBOpener->page() == frameA.page()) || (frameAOpener && frameBOpener && frameAOpener->page() == frameBOpener->page());
236}

238Frame* FrameTree::find(const AtomString& name, Frame& activeFrame) const
239{
240 // FIXME: _current is not part of the HTML specification.
241 if (equalIgnoringASCIICase(name, "_self") || name == "_current" || name.isEmpty())
242 return &m_thisFrame;
243
244 if (equalIgnoringASCIICase(name, "_top"))
245 return &top();
246
247 if (equalIgnoringASCIICase(name, "_parent"))
248 return parent() ? parent() : &m_thisFrame;
249
250 // Since "_blank" should never be any frame's name, the following is only an optimization.
251 if (equalIgnoringASCIICase(name, "_blank"))
252 return nullptr;
253
254 // Search subtree starting with this frame first.
255 for (Frame* frame = &m_thisFrame; frame; frame = frame->tree().traverseNext(&m_thisFrame)) {
256 if (frame->tree().uniqueName() == name)
257 return frame;
258 }
259
260 // Then the rest of the tree.
261 for (Frame* frame = &m_thisFrame.mainFrame(); frame; frame = frame->tree().traverseNext()) {
262 if (frame->tree().uniqueName() == name)
263 return frame;
264 }
265
266 // Search the entire tree of each of the other pages in this namespace.
267 // FIXME: Is random order OK?
268 Page* page = m_thisFrame.page();
269 if (!page)
270 return nullptr;
271
272 for (auto& otherPage : page->group().pages()) {
273 if (&otherPage == page)
274 continue;
275 for (auto* frame = &otherPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {
276 if (frame->tree().uniqueName() == name && isFrameFamiliarWith(activeFrame, *frame))
277 return frame;
278 }
279 }
280
281 return nullptr;
282}

284bool FrameTree::isDescendantOf(const Frame* ancestor) const
285{
286 if (!ancestor)
287 return false;
288
289 if (m_thisFrame.page() != ancestor->page())
290 return false;
291
292 for (Frame* frame = &m_thisFrame; frame; frame = frame->tree().parent())
293 if (frame == ancestor)
294 return true;
295 return false;
296}

298Frame* FrameTree::traverseNext(const Frame* stayWithin) const
299{
300 Frame* child = firstChild();
301 if (child) {
302 ASSERT(!stayWithin || child->tree().isDescendantOf(stayWithin));
303 return child;
304 }
305
306 if (&m_thisFrame == stayWithin)
307 return nullptr;
308
309 Frame* sibling = nextSibling();
310 if (sibling) {
311 ASSERT(!stayWithin || sibling->tree().isDescendantOf(stayWithin));
312 return sibling;
313 }
314
315 Frame* frame = &m_thisFrame;
316 while (!sibling && (!stayWithin || frame->tree().parent() != stayWithin)) {
317 frame = frame->tree().parent();
318 if (!frame)
319 return nullptr;
320 sibling = frame->tree().nextSibling();
321 }
322
323 if (frame) {
324 ASSERT(!stayWithin || !sibling || sibling->tree().isDescendantOf(stayWithin));
325 return sibling;
326 }
327
328 return nullptr;
329}

331Frame* FrameTree::firstRenderedChild() const
332{
333 Frame* child = firstChild();
334 if (!child)
335 return nullptr;
336
337 if (child->ownerRenderer())
338 return child;
339
340 while ((child = child->tree().nextSibling())) {
341 if (child->ownerRenderer())
342 return child;
343 }
344
345 return nullptr;
346}

348Frame* FrameTree::nextRenderedSibling() const
349{
350 Frame* sibling = &m_thisFrame;
351
352 while ((sibling = sibling->tree().nextSibling())) {
353 if (sibling->ownerRenderer())
354 return sibling;
355 }
356
357 return nullptr;
358}

360Frame* FrameTree::traverseNextRendered(const Frame* stayWithin) const
361{
362 Frame* child = firstRenderedChild();
363 if (child) {
364 ASSERT(!stayWithin || child->tree().isDescendantOf(stayWithin));
365 return child;
366 }
367
368 if (&m_thisFrame == stayWithin)
369 return nullptr;
370
371 Frame* sibling = nextRenderedSibling();
372 if (sibling) {
373 ASSERT(!stayWithin || sibling->tree().isDescendantOf(stayWithin));
374 return sibling;
375 }
376
377 Frame* frame = &m_thisFrame;
378 while (!sibling && (!stayWithin || frame->tree().parent() != stayWithin)) {
379 frame = frame->tree().parent();
380 if (!frame)
381 return nullptr;
382 sibling = frame->tree().nextRenderedSibling();
383 }
384
385 if (frame) {
386 ASSERT(!stayWithin || !sibling || sibling->tree().isDescendantOf(stayWithin));
387 return sibling;
388 }
389
390 return nullptr;
391}

393Frame* FrameTree::traverseNext(CanWrap canWrap, DidWrap* didWrap) const
394{
395 if (Frame* result = traverseNext())
396 return result;
397
398 if (canWrap == CanWrap::Yes) {
399 if (didWrap)
400 *didWrap = DidWrap::Yes;
401 return &m_thisFrame.mainFrame();
402 }
403
404 return nullptr;
405}

407Frame* FrameTree::traversePrevious(CanWrap canWrap, DidWrap* didWrap) const
408{
409 // FIXME: besides the wrap feature, this is just the traversePreviousNode algorithm
410
411 if (Frame* prevSibling = previousSibling())
412 return prevSibling->tree().deepLastChild();
413 if (Frame* parentFrame = parent())
414 return parentFrame;
415
416 // no siblings, no parent, self==top
417 if (canWrap == CanWrap::Yes) {
418 if (didWrap)
419 *didWrap = DidWrap::Yes;
420 return deepLastChild();
421 }
422
423 // top view is always the last one in this ordering, so prev is nil without wrap
424 return nullptr;
425}
426
427Frame* FrameTree::traverseNextInPostOrder(CanWrap canWrap) const
428{
429 if (m_nextSibling)
430 return m_nextSibling->tree().deepFirstChild();
431 if (m_parent)
432 return m_parent.get();
433 if (canWrap == CanWrap::Yes)
434 return deepFirstChild();
435 return nullptr;
436}

438Frame* FrameTree::deepFirstChild() const
439{
440 Frame* result = &m_thisFrame;
441 while (auto* next = result->tree().firstChild())
442 result = next;
443 return result;
444}
445
446Frame* FrameTree::deepLastChild() const
447{
448 Frame* result = &m_thisFrame;
449 for (Frame* last = lastChild(); last; last = last->tree().lastChild())
450 result = last;
451
452 return result;
453}
454
455Frame& FrameTree::top() const
456{
457 Frame* frame = &m_thisFrame;
458 for (Frame* parent = &m_thisFrame; parent; parent = parent->tree().parent())
459 frame = parent;
460 return *frame;
461}

467static void printIndent(int indent)
468{
469 for (int i = 0; i < indent; ++i)
470 printf(" ");
471}
472
473static void printFrames(const WebCore::Frame& frame, const WebCore::Frame* targetFrame, int indent)
474{
475 if (&frame == targetFrame) {
476 printf("--> ");
477 printIndent(indent - 1);
478 } else
479 printIndent(indent);
480
481 WebCore::FrameView* view = frame.view();
482 printf("Frame %p %dx%d\n", &frame, view ? view->width() : 0, view ? view->height() : 0);
483 printIndent(indent);
484 printf(" ownerElement=%p\n", frame.ownerElement());
485 printIndent(indent);
486 printf(" frameView=%p (needs layout %d)\n", view, view ? view->needsLayout() : false);
487 printIndent(indent);
488 printf(" renderView=%p\n", view ? view->renderView() : nullptr);
489 printIndent(indent);
490 printf(" ownerRenderer=%p\n", frame.ownerRenderer());
491 printIndent(indent);
492 printf(" document=%p (needs style recalc %d)\n", frame.document(), frame.document() ? frame.document()->childNeedsStyleRecalc() : false);
493 printIndent(indent);
494 printf(" uri=%s\n", frame.document()->documentURI().utf8().data());
495
496 for (WebCore::Frame* child = frame.tree().firstChild(); child; child = child->tree().nextSibling())
497 printFrames(*child, targetFrame, indent + 1);
498}

500void showFrameTree(const WebCore::Frame* frame)
501{
502 if (!frame) {
503 printf("Null input frame\n");
504 return;
505 }
506
507 printFrames(frame->tree().top(), frame, 0);
508}
