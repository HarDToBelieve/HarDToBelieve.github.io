58#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
59void FrameViewLayoutContext::layoutUsingFormattingContext()
60{
61 if (!RuntimeEnabledFeatures::sharedFeatures().layoutFormattingContextEnabled())
62 return;
63 // FrameView::setContentsSize temporary disables layout.
64 if (m_disableSetNeedsLayoutCount)
65 return;
66
67 auto& renderView = *this->renderView();
68 m_layoutTree = Layout::TreeBuilder::buildLayoutTree(renderView);
69 m_layoutState = makeUnique<Layout::LayoutState>(*document(), m_layoutTree->root());
70 // FIXME: This is not the real invalidation yet.
71 auto invalidationState = Layout::InvalidationState { };
72 auto layoutContext = Layout::LayoutContext { *m_layoutState };
73 layoutContext.layout(view().layoutSize(), invalidationState);
74
75 // Clean up the render tree state when we don't run RenderView::layout.
76 if (renderView.needsLayout()) {
77 auto contentSize = Layout::BoxGeometry::marginBoxRect(m_layoutState->geometryForBox(*m_layoutState->root().firstChild())).size();
78 renderView.setSize(contentSize);
79 renderView.repaintViewRectangle({ 0, 0, contentSize.width(), contentSize.height() });
80
81 for (auto& descendant : descendantsOfType<RenderObject>(renderView))
82 descendant.clearNeedsLayout();
83 renderView.clearNeedsLayout();
84 }
85#ifndef NDEBUG
86 Layout::LayoutContext::verifyAndOutputMismatchingLayoutTree(*m_layoutState, renderView);
87#endif
88}

91static bool isObjectAncestorContainerOf(RenderElement& ancestor, RenderElement& descendant)
92{
93 for (auto* renderer = &descendant; renderer; renderer = renderer->container()) {
94 if (renderer == &ancestor)
95 return true;
96 }
97 return false;
98}

100#ifndef NDEBUG
101class RenderTreeNeedsLayoutChecker {
102public :
103 RenderTreeNeedsLayoutChecker(const RenderElement& layoutRoot)
104 : m_layoutRoot(layoutRoot)
105 {
106 }
107
108 ~RenderTreeNeedsLayoutChecker()
109 {
110 auto reportNeedsLayoutError = [] (const RenderObject& renderer) {
111 WTFReportError(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, "post-layout: dirty renderer(s)");
112 renderer.showRenderTreeForThis();
113 ASSERT_NOT_REACHED();
114 };
115
116 if (m_layoutRoot.needsLayout()) {
117 reportNeedsLayoutError(m_layoutRoot);
118 return;
119 }
120
121 for (auto* descendant = m_layoutRoot.firstChild(); descendant; descendant = descendant->nextInPreOrder(&m_layoutRoot)) {
122 if (!descendant->needsLayout())
123 continue;
124
125 reportNeedsLayoutError(*descendant);
126 return;
127 }
128 }
129
130private:
131 const RenderElement& m_layoutRoot;
132};

169void FrameViewLayoutContext::layout()
170{
171 LOG_WITH_STREAM(Layout, stream << "FrameView " << &view() << " FrameViewLayoutContext::layout() with size " << view().layoutSize());
172
173 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(!frame().document()->inRenderTreeUpdate());
174 ASSERT(LayoutDisallowedScope::isLayoutAllowed());
175 ASSERT(!view().isPainting());
176 ASSERT(frame().view() == &view());
177 ASSERT(frame().document());
178 ASSERT(frame().document()->backForwardCacheState() == Document::NotInBackForwardCache
179 || frame().document()->backForwardCacheState() == Document::AboutToEnterBackForwardCache);
180 if (!canPerformLayout()) {
181 LOG(Layout, " is not allowed, bailing");
182 return;
183 }
184
185 Ref<FrameView> protectView(view());
186 LayoutScope layoutScope(*this);
187 TraceScope tracingScope(LayoutStart, LayoutEnd);
188 InspectorInstrumentation::willLayout(view().frame());
189 WeakPtr<RenderElement> layoutRoot;
190
191 m_layoutTimer.stop();
192 m_setNeedsLayoutWasDeferred = false;
193
194#if !LOG_DISABLED
195 if (m_firstLayout && !frame().ownerElement())
196 LOG(Layout, "FrameView %p elapsed time before first layout: %.3fs", this, document()->timeSinceDocumentCreation().value());
197#endif
198#if PLATFORM(IOS_FAMILY)
199 if (view().updateFixedPositionLayoutRect() && subtreeLayoutRoot())
200 convertSubtreeLayoutToFullLayout();
201#endif
202 if (handleLayoutWithFrameFlatteningIfNeeded())
203 return;
204
205 {
206 SetForScope<LayoutPhase> layoutPhase(m_layoutPhase, LayoutPhase::InPreLayout);
207
208 // If this is a new top-level layout and there are any remaining tasks from the previous layout, finish them now.
209 if (!isLayoutNested() && m_asynchronousTasksTimer.isActive() && !view().isInChildFrameWithFrameFlattening())
210 runAsynchronousTasks();
211
212 updateStyleForLayout();
213 if (view().hasOneRef())
214 return;
215
216 view().autoSizeIfEnabled();
217 if (!renderView())
218 return;
219
220 layoutRoot = makeWeakPtr(subtreeLayoutRoot() ? subtreeLayoutRoot() : renderView());
221 m_needsFullRepaint = is<RenderView>(layoutRoot.get()) && (m_firstLayout || renderView()->printing());
222 view().willDoLayout(layoutRoot);
223 m_firstLayout = false;
224 }
225 {
226 SetForScope<LayoutPhase> layoutPhase(m_layoutPhase, LayoutPhase::InRenderTreeLayout);
227 ScriptDisallowedScope::InMainThread scriptDisallowedScope;
228 SubtreeLayoutStateMaintainer subtreeLayoutStateMaintainer(subtreeLayoutRoot());
229 RenderView::RepaintRegionAccumulator repaintRegionAccumulator(renderView());
230#ifndef NDEBUG
231 RenderTreeNeedsLayoutChecker checker(*layoutRoot);
232#endif
233 layoutRoot->layout();
234#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
235 layoutUsingFormattingContext();
236#endif
237 ++m_layoutCount;
238#if ENABLE(TEXT_AUTOSIZING)
239 applyTextSizingIfNeeded(*layoutRoot.get());
240#endif
241 clearSubtreeLayoutRoot();
242 }
243 {
244 SetForScope<LayoutPhase> layoutPhase(m_layoutPhase, LayoutPhase::InViewSizeAdjust);
245 if (is<RenderView>(layoutRoot.get()) && !renderView()->printing()) {
246 // This is to protect m_needsFullRepaint's value when layout() is getting re-entered through adjustViewSize().
247 SetForScope<bool> needsFullRepaint(m_needsFullRepaint);
248 view().adjustViewSize();
249 // FIXME: Firing media query callbacks synchronously on nested frames could produced a detached FrameView here by
250 // navigating away from the current document (see webkit.org/b/173329).
251 if (view().hasOneRef())
252 return;
253 }
254 }
255 {
256 SetForScope<LayoutPhase> layoutPhase(m_layoutPhase, LayoutPhase::InPostLayout);
257 if (m_needsFullRepaint)
258 renderView()->repaintRootContents();
259 ASSERT(!layoutRoot->needsLayout());
260 view().didLayout(layoutRoot);
261 runOrScheduleAsynchronousTasks();
262 }
263 InspectorInstrumentation::didLayout(view().frame(), *layoutRoot);
264 DebugPageOverlays::didLayout(view().frame());
265}

267void FrameViewLayoutContext::runOrScheduleAsynchronousTasks()
268{
269 if (m_asynchronousTasksTimer.isActive())
270 return;
271
272 if (view().isInChildFrameWithFrameFlattening()) {
273 // While flattening frames, we defer post layout tasks to avoid getting stuck in a cycle,
274 // except updateWidgetPositions() which is required to kick off subframe layout in certain cases.
275 if (!m_inAsynchronousTasks)
276 view().updateWidgetPositions();
277 m_asynchronousTasksTimer.startOneShot(0_s);
278 return;
279 }
280
281 // If we are already in performPostLayoutTasks(), defer post layout tasks until after we return
282 // to avoid re-entrancy.
283 if (m_inAsynchronousTasks) {
284 m_asynchronousTasksTimer.startOneShot(0_s);
285 return;
286 }
287
288 runAsynchronousTasks();
289 if (needsLayout()) {
290 // If runAsynchronousTasks() made us layout again, let's defer the tasks until after we return.
291 m_asynchronousTasksTimer.startOneShot(0_s);
292 layout();
293 }
294}
295
296void FrameViewLayoutContext::runAsynchronousTasks()
297{
298 m_asynchronousTasksTimer.stop();
299 if (m_inAsynchronousTasks)
300 return;
301 SetForScope<bool> inAsynchronousTasks(m_inAsynchronousTasks, true);
302 view().performPostLayoutTasks();
303}
304
305void FrameViewLayoutContext::flushAsynchronousTasks()
306{
307 if (!m_asynchronousTasksTimer.isActive())
308 return;
309 runAsynchronousTasks();
310}

312void FrameViewLayoutContext::reset()
313{
314 m_layoutPhase = LayoutPhase::OutsideLayout;
315 clearSubtreeLayoutRoot();
316 m_layoutCount = 0;
317 m_layoutSchedulingIsEnabled = true;
318 m_layoutTimer.stop();
319 m_firstLayout = true;
320 m_asynchronousTasksTimer.stop();
321 m_needsFullRepaint = true;
322}

324bool FrameViewLayoutContext::needsLayout() const
325{
326 // This can return true in cases where the document does not have a body yet.
327 // Document::shouldScheduleLayout takes care of preventing us from scheduling
328 // layout in that case.
329 auto* renderView = this->renderView();
330 return isLayoutPending()
331 || (renderView && renderView->needsLayout())
332 || subtreeLayoutRoot()
333 || (m_disableSetNeedsLayoutCount && m_setNeedsLayoutWasDeferred);
334}

362void FrameViewLayoutContext::scheduleLayout()
363{
364 // FIXME: We should assert the page is not in the back/forward cache, but that is causing
365 // too many false assertions. See <rdar://problem/7218118>.
366 ASSERT(frame().view() == &view());
367
368 if (subtreeLayoutRoot())
369 convertSubtreeLayoutToFullLayout();
370 if (!isLayoutSchedulingEnabled())
371 return;
372 if (!needsLayout())
373 return;
374 if (!frame().document()->shouldScheduleLayout())
375 return;
376 InspectorInstrumentation::didInvalidateLayout(frame());
377 // When frame flattening is enabled, the contents of the frame could affect the layout of the parent frames.
378 // Also invalidate parent frame starting from the owner element of this frame.
379 if (frame().ownerRenderer() && view().isInChildFrameWithFrameFlattening())
380 frame().ownerRenderer()->setNeedsLayout(MarkContainingBlockChain);
381
382 if (m_layoutTimer.isActive())
383 return;
384
385#if !LOG_DISABLED
386 if (!frame().document()->ownerElement())
387 LOG(Layout, "FrameView %p layout timer scheduled at %.3fs", this, frame().document()->timeSinceDocumentCreation().value());
388#endif
389
390 m_layoutTimer.startOneShot(0_s);
391}

394{
395 if (m_asynchronousTasksTimer.isActive())
396 m_asynchronousTasksTimer.stop();

398 if (!m_layoutTimer.isActive())

402 if (!frame().document()->ownerElement())

404#endif

409void FrameViewLayoutContext::scheduleSubtreeLayout(RenderElement& layoutRoot)
410{
411 ASSERT(renderView());
412 auto& renderView = *this->renderView();
413
414 // Try to catch unnecessary work during render tree teardown.
415 ASSERT(!renderView.renderTreeBeingDestroyed());
416 ASSERT(frame().view() == &view());
417
418 if (renderView.needsLayout() && !subtreeLayoutRoot()) {
419 layoutRoot.markContainingBlocksForLayout(ScheduleRelayout::No);
420 return;
421 }
422
423 if (!isLayoutPending() && isLayoutSchedulingEnabled()) {
424 ASSERT(!layoutRoot.container() || is<RenderView>(layoutRoot.container()) || !layoutRoot.container()->needsLayout());
425 setSubtreeLayoutRoot(layoutRoot);
426 InspectorInstrumentation::didInvalidateLayout(frame());
427 m_layoutTimer.startOneShot(0_s);
428 return;
429 }
430
431 auto* subtreeLayoutRoot = this->subtreeLayoutRoot();
432 if (subtreeLayoutRoot == &layoutRoot)
433 return;
434
435 if (!subtreeLayoutRoot) {
436 // We already have a pending (full) layout. Just mark the subtree for layout.
437 layoutRoot.markContainingBlocksForLayout(ScheduleRelayout::No);
438 InspectorInstrumentation::didInvalidateLayout(frame());
439 return;
440 }
441
442 if (isObjectAncestorContainerOf(*subtreeLayoutRoot, layoutRoot)) {
443 // Keep the current root.
444 layoutRoot.markContainingBlocksForLayout(ScheduleRelayout::No, subtreeLayoutRoot);
445 ASSERT(!subtreeLayoutRoot->container() || is<RenderView>(subtreeLayoutRoot->container()) || !subtreeLayoutRoot->container()->needsLayout());
446 return;
447 }
448
449 if (isObjectAncestorContainerOf(layoutRoot, *subtreeLayoutRoot)) {
450 // Re-root at newRelayoutRoot.
451 subtreeLayoutRoot->markContainingBlocksForLayout(ScheduleRelayout::No, &layoutRoot);
452 setSubtreeLayoutRoot(layoutRoot);
453 ASSERT(!layoutRoot.container() || is<RenderView>(layoutRoot.container()) || !layoutRoot.container()->needsLayout());
454 InspectorInstrumentation::didInvalidateLayout(frame());
455 return;
456 }
457 // Two disjoint subtrees need layout. Mark both of them and issue a full layout instead.
458 convertSubtreeLayoutToFullLayout();
459 layoutRoot.markContainingBlocksForLayout(ScheduleRelayout::No);
460 InspectorInstrumentation::didInvalidateLayout(frame());
461}

489bool FrameViewLayoutContext::canPerformLayout() const
490{
491 if (isInRenderTreeLayout())
492 return false;
493
494 if (layoutDisallowed())
495 return false;
496
497 if (view().isPainting())
498 return false;
499
500 if (!subtreeLayoutRoot() && !frame().document()->renderView())
501 return false;
502
503 return true;
504}

506#if ENABLE(TEXT_AUTOSIZING)
507void FrameViewLayoutContext::applyTextSizingIfNeeded(RenderElement& layoutRoot)
508{
509 auto& settings = layoutRoot.settings();
510 bool idempotentMode = settings.textAutosizingUsesIdempotentMode();
511 if (!settings.textAutosizingEnabled() || idempotentMode || renderView()->printing())
512 return;
513 auto minimumZoomFontSize = settings.minimumZoomFontSize();
514 if (!idempotentMode && !minimumZoomFontSize)
515 return;
516 auto textAutosizingWidth = layoutRoot.page().textAutosizingWidth();
517 if (auto overrideWidth = settings.textAutosizingWindowSizeOverrideWidth())
518 textAutosizingWidth = overrideWidth;
519 if (!idempotentMode && !textAutosizingWidth)
520 return;
521 layoutRoot.adjustComputedFontSizesOnBlocks(minimumZoomFontSize, textAutosizingWidth);
522 if (!layoutRoot.needsLayout())
523 return;
524 LOG(TextAutosizing, "Text Autosizing: minimumZoomFontSize=%.2f textAutosizingWidth=%.2f", minimumZoomFontSize, textAutosizingWidth);
525 layoutRoot.layout();
526}

529void FrameViewLayoutContext::updateStyleForLayout()
530{
531 Document& document = *frame().document();
532
533 // FIXME: This shouldn't be necessary, but see rdar://problem/36670246.
534 if (!document.styleScope().resolverIfExists())
535 document.styleScope().didChangeStyleSheetEnvironment();
536
537 // Viewport-dependent media queries may cause us to need completely different style information.
538 document.styleScope().evaluateMediaQueriesForViewportChange();
539
540 document.updateElementsAffectedByMediaQueries();
541 // If there is any pagination to apply, it will affect the RenderView's style, so we should
542 // take care of that now.
543 view().applyPaginationToViewport();
544 // Always ensure our style info is up-to-date. This can happen in situations where
545 // the layout beats any sort of style recalc update that needs to occur.
546 document.updateStyleIfNeeded();
547}

549bool FrameViewLayoutContext::handleLayoutWithFrameFlatteningIfNeeded()
550{
551 if (!view().isInChildFrameWithFrameFlattening())
552 return false;
553
554 startLayoutAtMainFrameViewIfNeeded();
555 auto* layoutRoot = subtreeLayoutRoot() ? subtreeLayoutRoot() : frame().document()->renderView();
556 return !layoutRoot || !layoutRoot->needsLayout();
557}
558
559void FrameViewLayoutContext::startLayoutAtMainFrameViewIfNeeded()
560{
561 // When we start a layout at the child level as opposed to the topmost frame view and this child
562 // frame requires flattening, we need to re-initiate the layout at the topmost view. Layout
563 // will hit this view eventually.
564 auto* parentView = view().parentFrameView();
565 if (!parentView)
566 return;
567
568 // In the middle of parent layout, no need to restart from topmost.
569 if (parentView->layoutContext().isInLayout())
570 return;
571
572 // Parent tree is clean. Starting layout from it would have no effect.
573 if (!parentView->needsLayout())
574 return;
575
576 while (parentView->parentFrameView())
577 parentView = parentView->parentFrameView();
578
579 LOG(Layout, " frame flattening, starting from root");
580 parentView->layoutContext().layout();
581}

628bool FrameViewLayoutContext::pushLayoutState(RenderBox& renderer, const LayoutSize& offset, LayoutUnit pageHeight, bool pageHeightChanged)
629{
630 // We push LayoutState even if layoutState is disabled because it stores layoutDelta too.
631 auto* layoutState = this->layoutState();
632 if (!layoutState || !needsFullRepaint() || layoutState->isPaginated() || renderer.enclosingFragmentedFlow()
633 || layoutState->lineGrid() || (renderer.style().lineGrid() != RenderStyle::initialLineGrid() && renderer.isRenderBlockFlow())) {
634 m_layoutStateStack.append(makeUnique<RenderLayoutState>(m_layoutStateStack, renderer, offset, pageHeight, pageHeightChanged));
635 return true;
636 }
637 return false;
638}
