34class Document;
35class Frame;
36class FrameView;
37class LayoutScope;
38class LayoutSize;
39class RenderBlockFlow;
40class RenderBox;
41class RenderObject;
42class RenderElement;
43class RenderLayoutState;
44class RenderView;
45#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
46namespace Layout {
47class LayoutState;
48class LayoutTree;
49}

51
52class FrameViewLayoutContext {
53public:
54 FrameViewLayoutContext(FrameView&);
55 ~FrameViewLayoutContext();
56
57 void layout();
58 bool needsLayout() const;
59
60 // We rely on the side-effects of layout, like compositing updates, to update state in various subsystems
61 // whose dependencies are poorly defined. This call triggers such updates.
62 void setNeedsLayoutAfterViewConfigurationChange();
63
64 void scheduleLayout();
65 void scheduleSubtreeLayout(RenderElement& layoutRoot);
66 void unscheduleLayout();
67
68 void startDisallowingLayout() { ++m_layoutDisallowedCount; }
69 void endDisallowingLayout() { ASSERT(m_layoutDisallowedCount > 0); --m_layoutDisallowedCount; }
70
71 void disableSetNeedsLayout();
72 void enableSetNeedsLayout();
73
74 enum class LayoutPhase : uint8_t {
75 OutsideLayout,
76 InPreLayout,
77 InRenderTreeLayout,
78 InViewSizeAdjust,
79 InPostLayout
80 };
81 LayoutPhase layoutPhase() const { return m_layoutPhase; }
82 bool isLayoutNested() const { return m_layoutNestedState == LayoutNestedState::Nested; }
83 bool isLayoutPending() const { return m_layoutTimer.isActive(); }
84 bool isInLayout() const { return layoutPhase() != LayoutPhase::OutsideLayout; }
85 bool isInRenderTreeLayout() const { return layoutPhase() == LayoutPhase::InRenderTreeLayout; }
86 bool inPaintableState() const { return layoutPhase() != LayoutPhase::InRenderTreeLayout && layoutPhase() != LayoutPhase::InViewSizeAdjust && (layoutPhase() != LayoutPhase::InPostLayout || inAsynchronousTasks()); }
87
88 unsigned layoutCount() const { return m_layoutCount; }
89
90 RenderElement* subtreeLayoutRoot() const;
91 void clearSubtreeLayoutRoot() { m_subtreeLayoutRoot.clear(); }
92 void convertSubtreeLayoutToFullLayout();
93
94 void reset();
95 void resetFirstLayoutFlag() { m_firstLayout = true; }
96 bool didFirstLayout() const { return !m_firstLayout; }
97
98 void setNeedsFullRepaint() { m_needsFullRepaint = true; }
99 bool needsFullRepaint() const { return m_needsFullRepaint; }
100
101 void flushAsynchronousTasks();
102
103 RenderLayoutState* layoutState() const PURE_FUNCTION;
104 // Returns true if layoutState should be used for its cached offset and clip.
105 bool isPaintOffsetCacheEnabled() const { return !m_paintOffsetCacheDisableCount && layoutState(); }
106#ifndef NDEBUG
107 void checkLayoutState();
108#endif
109 // layoutDelta is used transiently during layout to store how far an object has moved from its
110 // last layout location, in order to repaint correctly.
111 // If we're doing a full repaint m_layoutState will be 0, but in that case layoutDelta doesn't matter.
112 LayoutSize layoutDelta() const;
113 void addLayoutDelta(const LayoutSize& delta);
114#if ASSERT_ENABLED
115 bool layoutDeltaMatches(const LayoutSize& delta);
116#endif
117 using LayoutStateStack = Vector<std::unique_ptr<RenderLayoutState>>;
118
119#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
120 const Layout::LayoutState* layoutFormattingState() const { return m_layoutState.get(); }
121#endif
122
123private:
124 friend class LayoutScope;
125 friend class LayoutStateMaintainer;
126 friend class LayoutStateDisabler;
127 friend class SubtreeLayoutStateMaintainer;
128 friend class PaginatedLayoutStateMaintainer;
129
130 bool canPerformLayout() const;
131 bool layoutDisallowed() const { return m_layoutDisallowedCount; }
132 bool isLayoutSchedulingEnabled() const { return m_layoutSchedulingIsEnabled; }
133
134 void layoutTimerFired();
135 void runAsynchronousTasks();
136 void runOrScheduleAsynchronousTasks();
137 bool inAsynchronousTasks() const { return m_inAsynchronousTasks; }
138
139 void setSubtreeLayoutRoot(RenderElement&);
140
141#if ENABLE(TEXT_AUTOSIZING)
142 void applyTextSizingIfNeeded(RenderElement& layoutRoot);
143#endif
144 void updateStyleForLayout();
145
146 bool handleLayoutWithFrameFlatteningIfNeeded();
147 void startLayoutAtMainFrameViewIfNeeded();
148
149 // These functions may only be accessed by LayoutStateMaintainer.
150 // Subtree push/pop
151 void pushLayoutState(RenderElement&);
152 bool pushLayoutStateForPaginationIfNeeded(RenderBlockFlow&);
153 bool pushLayoutState(RenderBox& renderer, const LayoutSize& offset, LayoutUnit pageHeight = 0_lu, bool pageHeightChanged = false);
154 void popLayoutState();
155
156 // Suspends the LayoutState optimization. Used under transforms that cannot be represented by
157 // LayoutState (common in SVG) and when manipulating the render tree during layout in ways
158 // that can trigger repaint of a non-child (e.g. when a list item moves its list marker around).
159 // Note that even when disabled, LayoutState is still used to store layoutDelta.
160 // These functions may only be accessed by LayoutStateMaintainer or LayoutStateDisabler.
161 void disablePaintOffsetCache() { m_paintOffsetCacheDisableCount++; }
162 void enablePaintOffsetCache() { ASSERT(m_paintOffsetCacheDisableCount > 0); m_paintOffsetCacheDisableCount--; }
163#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
164 void layoutUsingFormattingContext();
165#endif
166
167 Frame& frame() const;
168 FrameView& view() const;
169 RenderView* renderView() const;
170 Document* document() const;
171
172 FrameView& m_frameView;
173 Timer m_layoutTimer;
174 Timer m_asynchronousTasksTimer;
175 WeakPtr<RenderElement> m_subtreeLayoutRoot;
176
177 bool m_layoutSchedulingIsEnabled { true };
178 bool m_firstLayout { true };
179 bool m_needsFullRepaint { true };
180 bool m_inAsynchronousTasks { false };
181 bool m_setNeedsLayoutWasDeferred { false };
182 LayoutPhase m_layoutPhase { LayoutPhase::OutsideLayout };
183 enum class LayoutNestedState : uint8_t { NotInLayout, NotNested, Nested };
184 LayoutNestedState m_layoutNestedState { LayoutNestedState::NotInLayout };
185 unsigned m_layoutCount { 0 };
186 unsigned m_disableSetNeedsLayoutCount { 0 };
187 int m_layoutDisallowedCount { 0 };
188 unsigned m_paintOffsetCacheDisableCount { 0 };
189 LayoutStateStack m_layoutStateStack;
190#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
191 std::unique_ptr<Layout::LayoutState> m_layoutState;
192 std::unique_ptr<Layout::LayoutTree> m_layoutTree;
193#endif
194};
