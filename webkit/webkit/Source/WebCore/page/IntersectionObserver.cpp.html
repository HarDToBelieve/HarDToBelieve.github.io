44static ExceptionOr<LengthBox> parseRootMargin(String& rootMargin)
45{
46 CSSTokenizer tokenizer(rootMargin);
47 auto tokenRange = tokenizer.tokenRange();
48 Vector<Length, 4> margins;
49 while (!tokenRange.atEnd()) {
50 if (margins.size() == 4)
51 return Exception { SyntaxError, "Failed to construct 'IntersectionObserver': Extra text found at the end of rootMargin." };
52 RefPtr<CSSPrimitiveValue> parsedValue = CSSPropertyParserHelpers::consumeLengthOrPercent(tokenRange, HTMLStandardMode, ValueRange::All);
53 if (!parsedValue || parsedValue->isCalculated())
54 return Exception { SyntaxError, "Failed to construct 'IntersectionObserver': rootMargin must be specified in pixels or percent." };
55 if (parsedValue->isPercentage())
56 margins.append(Length(parsedValue->doubleValue(), LengthType::Percent));
57 else if (parsedValue->isPx())
58 margins.append(Length(parsedValue->intValue(), LengthType::Fixed));
59 else
60 return Exception { SyntaxError, "Failed to construct 'IntersectionObserver': rootMargin must be specified in pixels or percent." };
61 }
62 switch (margins.size()) {
63 case 0:
64 for (unsigned i = 0; i < 4; ++i)
65 margins.append(Length(0, LengthType::Fixed));
66 break;
67 case 1:
68 for (unsigned i = 0; i < 3; ++i)
69 margins.append(margins[0]);
70 break;
71 case 2:
72 margins.append(margins[0]);
73 margins.append(margins[1]);
74 break;
75 case 3:
76 margins.append(margins[1]);
77 break;
78 case 4:
79 break;
80 default:
81 ASSERT_NOT_REACHED();
82 }
83
84 return LengthBox(WTFMove(margins[0]), WTFMove(margins[1]), WTFMove(margins[2]), WTFMove(margins[3]));
85}

87ExceptionOr<Ref<IntersectionObserver>> IntersectionObserver::create(Document& document, Ref<IntersectionObserverCallback>&& callback, IntersectionObserver::Init&& init)
88{
89 ContainerNode* root = nullptr;
90 if (init.root) {
91 WTF::switchOn(*init.root, [&root] (RefPtr<Element> element) {
92 root = element.get();
93 }, [&root] (RefPtr<Document> document) {
94 root = document.get();
95 });
96 }
97
98 auto rootMarginOrException = parseRootMargin(init.rootMargin);
99 if (rootMarginOrException.hasException())
100 return rootMarginOrException.releaseException();
101
102 Vector<double> thresholds;
103 WTF::switchOn(init.threshold, [&thresholds] (double initThreshold) {
104 thresholds.reserveInitialCapacity(1);
105 thresholds.uncheckedAppend(initThreshold);
106 }, [&thresholds] (Vector<double>& initThresholds) {
107 thresholds = WTFMove(initThresholds);
108 });
109
110 for (auto threshold : thresholds) {
111 if (!(threshold >= 0 && threshold <= 1))
112 return Exception { RangeError, "Failed to construct 'IntersectionObserver': all thresholds must lie in the range [0.0, 1.0]." };
113 }
114
115 return adoptRef(*new IntersectionObserver(document, WTFMove(callback), root, rootMarginOrException.releaseReturnValue(), WTFMove(thresholds)));
116}

118IntersectionObserver::IntersectionObserver(Document& document, Ref<IntersectionObserverCallback>&& callback, ContainerNode* root, LengthBox&& parsedRootMargin, Vector<double>&& thresholds)
119 : ActiveDOMObject(callback->scriptExecutionContext())
120 , m_root(root)
121 , m_rootMargin(WTFMove(parsedRootMargin))
122 , m_thresholds(WTFMove(thresholds))
123 , m_callback(WTFMove(callback))
124{
125 if (is<Document>(m_root)) {
126 auto& observerData = downcast<Document>(*m_root).ensureIntersectionObserverData();
127 observerData.observers.append(makeWeakPtr(this));
128 } else if (m_root) {
129 ASSERT(is<Element>(m_root));
130 auto& observerData = downcast<Element>(*m_root).ensureIntersectionObserverData();
131 observerData.observers.append(makeWeakPtr(this));
132 } else if (auto* frame = document.frame())
133 m_implicitRootDocument = makeWeakPtr(frame->mainFrame().document());
134
135 std::sort(m_thresholds.begin(), m_thresholds.end());
136 suspendIfNeeded();
137}

139IntersectionObserver::~IntersectionObserver()
140{
141 if (is<Document>(m_root)) {
142 downcast<Document>(*m_root).intersectionObserverDataIfExists()->observers.removeFirst(this);
143 } else if (m_root) {
144 ASSERT(is<Element>(m_root));
145 downcast<Element>(*m_root).intersectionObserverDataIfExists()->observers.removeFirst(this);
146 }
147 disconnect();
148}

150String IntersectionObserver::rootMargin() const
151{
152 StringBuilder stringBuilder;
153 for (auto side : allBoxSides) {
154 auto& length = m_rootMargin.at(side);
155 stringBuilder.append(length.intValue(), length.isPercent() ? "%" : "px", side != BoxSide::Left ? " " : "");
156 }
157 return stringBuilder.toString();
158}

160void IntersectionObserver::observe(Element& target)
161{
162 if (!trackingDocument() || !m_callback || m_observationTargets.contains(&target))
163 return;
164
165 target.ensureIntersectionObserverData().registrations.append({ makeWeakPtr(this), std::nullopt });
166 bool hadObservationTargets = hasObservationTargets();
167 m_observationTargets.append(&target);
168 auto* document = trackingDocument();
169 if (!hadObservationTargets)
170 document->addIntersectionObserver(*this);
171 document->scheduleInitialIntersectionObservationUpdate();
172}
173
174void IntersectionObserver::unobserve(Element& target)
175{
176 if (!removeTargetRegistration(target))
177 return;
178
179 bool removed = m_observationTargets.removeFirst(&target);
180 ASSERT_UNUSED(removed, removed);
181
182 if (!hasObservationTargets()) {
183 if (auto* document = trackingDocument())
184 document->removeIntersectionObserver(*this);
185 }
186}
187
188void IntersectionObserver::disconnect()
189{
190 if (!hasObservationTargets())
191 return;
192
193 removeAllTargets();
194 if (auto* document = trackingDocument())
195 document->removeIntersectionObserver(*this);
196}

198auto IntersectionObserver::takeRecords() -> TakenRecords
199{

203void IntersectionObserver::targetDestroyed(Element& target)
204{
205 m_observationTargets.removeFirst(&target);
206 if (!hasObservationTargets()) {
207 if (auto* document = trackingDocument())
208 document->removeIntersectionObserver(*this);
209 }
210}

212bool IntersectionObserver::removeTargetRegistration(Element& target)
213{
214 auto* observerData = target.intersectionObserverDataIfExists();
215 if (!observerData)
216 return false;
217
218 auto& registrations = observerData->registrations;
219 return registrations.removeFirstMatching([this](auto& registration) {
220 return registration.observer.get() == this;
221 });
222}
223
224void IntersectionObserver::removeAllTargets()
225{
226 for (auto* target : m_observationTargets) {
227 bool removed = removeTargetRegistration(*target);
228 ASSERT_UNUSED(removed, removed);
229 }
230 m_observationTargets.clear();
231}

233void IntersectionObserver::rootDestroyed()
234{
235 ASSERT(m_root);
236 disconnect();
237 m_root = nullptr;

240std::optional<ReducedResolutionSeconds> IntersectionObserver::nowTimestamp() const
241{
242 if (!m_callback)
243 return std::nullopt;
244
245 auto* context = m_callback->scriptExecutionContext();
246 if (!context)
247 return std::nullopt;
248
249 ASSERT(context->isDocument());
250 auto& document = downcast<Document>(*context);
251 if (auto* window = document.domWindow())
252 return window->frozenNowTimestamp();
253
254 return std::nullopt;
255}

257void IntersectionObserver::appendQueuedEntry(Ref<IntersectionObserverEntry>&& entry)
258{
259 ASSERT(entry->target());
260 m_pendingTargets.append(*entry->target());
261 m_queuedEntries.append(WTFMove(entry));
262}
263
264void IntersectionObserver::notify()
265{
266 if (m_queuedEntries.isEmpty()) {
267 ASSERT(m_pendingTargets.isEmpty());
268 return;
269 }
270
271 auto* context = m_callback->scriptExecutionContext();
272 if (!context)
273 return;
274
275 auto takenRecords = takeRecords();
276
277 InspectorInstrumentation::willFireObserverCallback(*context, "IntersectionObserver"_s);
278 m_callback->handleEvent(*this, WTFMove(takenRecords.records), *this);
279 InspectorInstrumentation::didFireObserverCallback(*context);
280}

282bool IntersectionObserver::virtualHasPendingActivity() const
283{
284 return (hasObservationTargets() && trackingDocument()) || !m_queuedEntries.isEmpty();
