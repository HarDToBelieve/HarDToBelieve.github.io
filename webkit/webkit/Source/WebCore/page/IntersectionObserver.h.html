40namespace WebCore {
41
42class Document;
43class Element;
44class ContainerNode;
45
46struct IntersectionObserverRegistration {
47 WeakPtr<IntersectionObserver> observer;
48 std::optional<size_t> previousThresholdIndex;
49};
50
51struct IntersectionObserverData {
52 WTF_MAKE_STRUCT_FAST_ALLOCATED;
53
54 // IntersectionObservers for which the node that owns this IntersectionObserverData is the root.
55 // An IntersectionObserver is only owned by a JavaScript wrapper. ActiveDOMObject::virtualHasPendingActivity
56 // is overridden to keep this wrapper alive while the observer has ongoing observations.
57 Vector<WeakPtr<IntersectionObserver>> observers;
58
59 // IntersectionObserverRegistrations for which the node that owns this IntersectionObserverData is the target.
60 Vector<IntersectionObserverRegistration> registrations;
61};
62
63class IntersectionObserver : public RefCounted<IntersectionObserver>, public ActiveDOMObject, public CanMakeWeakPtr<IntersectionObserver> {
64public:
65 struct Init {
66 std::optional<Variant<RefPtr<Element>, RefPtr<Document>>> root;
67 String rootMargin;
68 Variant<double, Vector<double>> threshold;
69 };
70
71 static ExceptionOr<Ref<IntersectionObserver>> create(Document&, Ref<IntersectionObserverCallback>&&, Init&&);
72
73 ~IntersectionObserver();
74
75 Document* trackingDocument() const { return m_root ? &m_root->document() : m_implicitRootDocument.get(); }
76
77 ContainerNode* root() const { return m_root; }
78 String rootMargin() const;
79 const LengthBox& rootMarginBox() const { return m_rootMargin; }
80 const Vector<double>& thresholds() const { return m_thresholds; }
81 const Vector<Element*> observationTargets() const { return m_observationTargets; }
82
83 void observe(Element&);
84 void unobserve(Element&);
85 void disconnect();
86
87 struct TakenRecords {
88 Vector<Ref<IntersectionObserverEntry>> records;
89 Vector<GCReachableRef<Element>> pendingTargets;
90 };
91 TakenRecords takeRecords();
92
93 void targetDestroyed(Element&);
94 bool hasObservationTargets() const { return m_observationTargets.size(); }
95 void rootDestroyed();
96
97 std::optional<ReducedResolutionSeconds> nowTimestamp() const;
98
99 void appendQueuedEntry(Ref<IntersectionObserverEntry>&&);
100 void notify();
101
102 IntersectionObserverCallback* callbackConcurrently() { return m_callback.get(); }
103
104private:
105 IntersectionObserver(Document&, Ref<IntersectionObserverCallback>&&, ContainerNode* root, LengthBox&& parsedRootMargin, Vector<double>&& thresholds);
106
107 // ActiveDOMObject.
108 bool virtualHasPendingActivity() const override;
109 const char* activeDOMObjectName() const override;
110 void stop() override;
111
112 bool removeTargetRegistration(Element&);
113 void removeAllTargets();
114
115 WeakPtr<Document> m_implicitRootDocument;
116 ContainerNode* m_root;
117 LengthBox m_rootMargin;
118 Vector<double> m_thresholds;
119 RefPtr<IntersectionObserverCallback> m_callback;
120 Vector<Element*> m_observationTargets;
121 Vector<GCReachableRef<Element>> m_pendingTargets;
122 Vector<Ref<IntersectionObserverEntry>> m_queuedEntries;
123};
