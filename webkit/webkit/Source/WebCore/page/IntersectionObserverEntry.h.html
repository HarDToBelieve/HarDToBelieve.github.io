27
28#if ENABLE(INTERSECTION_OBSERVER)
29
30#include "DOMRectReadOnly.h"
31#include "Element.h"
32#include <wtf/RefCounted.h>
33#include <wtf/Vector.h>
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38class Element;
39
40class IntersectionObserverEntry : public RefCounted<IntersectionObserverEntry> {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43
44 struct Init {
45 double time;
46 std::optional<DOMRectInit> rootBounds;
47 DOMRectInit boundingClientRect;
48 DOMRectInit intersectionRect;
49 double intersectionRatio;
50 RefPtr<Element> target;
51 bool isIntersecting;
52 };
53
54 static Ref<IntersectionObserverEntry> create(const Init& init)
55 {
56 return WTF::adoptRef(*new IntersectionObserverEntry(init));
57 }
58
59 double time() const { return m_time; }
60 DOMRectReadOnly* rootBounds() const { return m_rootBounds.get(); }
61 DOMRectReadOnly* boundingClientRect() const { return m_boundingClientRect.get(); }
62 DOMRectReadOnly* intersectionRect() const { return m_intersectionRect.get(); }
63 Element* target() const { return m_target.get(); }
64
65 bool isIntersecting() const { return m_isIntersecting; }
66 double intersectionRatio() const { return m_intersectionRatio; }
67
68private:
69 IntersectionObserverEntry(const Init&);
70
71 double m_time { 0 };
72 RefPtr<DOMRectReadOnly> m_rootBounds;
73 RefPtr<DOMRectReadOnly> m_boundingClientRect;
74 RefPtr<DOMRectReadOnly> m_intersectionRect;
75 double m_intersectionRatio { 0 };
76 RefPtr<Element> m_target;
77 bool m_isIntersecting { false };
78};
79
80
81} // namespace WebCore
82
83#endif // ENABLE(INTERSECTION_OBSERVER)
