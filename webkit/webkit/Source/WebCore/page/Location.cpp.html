51inline const URL& Location::url() const
52{
53 if (!frame())
54 return aboutBlankURL();
55
56 const URL& url = frame()->document()->url();
57 if (!url.isValid())
58 return aboutBlankURL(); // Use "about:blank" while the page is still loading (before we have a frame).
59
60 return url;
61}

63String Location::href() const
64{

66 urlWithoutCredentials.removeCredentials();
67 return urlWithoutCredentials.string();

75String Location::host() const
76{
77 // Note: this is the IE spec. The NS spec swaps the two, it says
78 // "The hostname property is the concatenation of the host and port properties, separated by a colon."
79 return url().hostAndPort();

109Ref<DOMStringList> Location::ancestorOrigins() const
110{
111 auto origins = DOMStringList::create();
112 auto* frame = this->frame();
113 if (!frame)
114 return origins;
115 for (auto* ancestor = frame->tree().parent(); ancestor; ancestor = ancestor->tree().parent())
116 origins->append(ancestor->document()->securityOrigin().toString());
117 return origins;
118}

193ExceptionOr<void> Location::setHash(DOMWindow& activeWindow, DOMWindow& firstWindow, const String& hash)
194{
195 auto* frame = this->frame();
196 if (!frame)
197 return { };
198 ASSERT(frame->document());
199 auto url = frame->document()->url();
200 auto oldFragmentIdentifier = url.fragmentIdentifier();
201 auto newFragmentIdentifier = hash;
202 if (hash[0] == '#')
203 newFragmentIdentifier = hash.substring(1);
204 url.setFragmentIdentifier(newFragmentIdentifier);
205 // Note that by parsing the URL and *then* comparing fragments, we are
206 // comparing fragments post-canonicalization, and so this handles the
207 // cases where fragment identifiers are ignored or invalid.
208 if (equalIgnoringNullity(oldFragmentIdentifier, url.fragmentIdentifier()))
209 return { };
210 return setLocation(activeWindow, firstWindow, url.string());
211}

220ExceptionOr<void> Location::replace(DOMWindow& activeWindow, DOMWindow& firstWindow, const String& urlString)
221{
222 auto* frame = this->frame();
223 if (!frame)
224 return { };
225 ASSERT(frame->document());
226 ASSERT(frame->document()->domWindow());
227
228 Frame* firstFrame = firstWindow.frame();
229 if (!firstFrame || !firstFrame->document())
230 return { };
231
232 URL completedURL = firstFrame->document()->completeURL(urlString);
233 if (!completedURL.isValid())
234 return Exception { SyntaxError };
235
236 // We call DOMWindow::setLocation directly here because replace() always operates on the current frame.
237 frame->document()->domWindow()->setLocation(activeWindow, completedURL, LockHistoryAndBackForwardList);
238 return { };
239}

241void Location::reload(DOMWindow& activeWindow)
242{
243 auto* frame = this->frame();
244 if (!frame)
245 return;
246
247 ASSERT(activeWindow.document());
248 ASSERT(frame->document());
249 ASSERT(frame->document()->domWindow());
250
251 auto& activeDocument = *activeWindow.document();
252 auto& targetDocument = *frame->document();
253
254 // FIXME: It's not clear this cross-origin security check is valuable.
255 // We allow one page to change the location of another. Why block attempts to reload?
256 // Other location operations simply block use of JavaScript URLs cross origin.
257 if (!activeDocument.securityOrigin().isSameOriginDomain(targetDocument.securityOrigin())) {
258 auto& targetWindow = *targetDocument.domWindow();
259 targetWindow.printErrorMessage(targetWindow.crossDomainAccessErrorMessage(activeWindow, IncludeTargetOrigin::Yes));
260 return;
261 }
262
263 if (targetDocument.url().protocolIsJavaScript())
264 return;
265
266 frame->navigationScheduler().scheduleRefresh(activeDocument);
267}

269ExceptionOr<void> Location::setLocation(DOMWindow& activeWindow, DOMWindow& firstWindow, const String& urlString)
270{
271 auto* frame = this->frame();
272 ASSERT(frame);
273
274 Frame* firstFrame = firstWindow.frame();
275 if (!firstFrame || !firstFrame->document())
276 return { };
277
278 URL completedURL = firstFrame->document()->completeURL(urlString);
279 // FIXME: The specification says to throw a SyntaxError if the URL is not valid.
280 if (completedURL.isNull())
281 return { };
282
283 if (!activeWindow.document()->canNavigate(frame, completedURL))
284 return Exception { SecurityError };
285
286 ASSERT(frame->document());
287 ASSERT(frame->document()->domWindow());
288 frame->document()->domWindow()->setLocation(activeWindow, completedURL);
289 return { };
290}
