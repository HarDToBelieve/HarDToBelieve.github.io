36namespace WebCore {
37
38class LoggedInStatus {
39 WTF_MAKE_ISO_ALLOCATED_EXPORT(LoggedInStatus, WEBCORE_EXPORT);
40public:
41 static constexpr uint32_t UsernameMaxLength = 64;
42 static constexpr Seconds TimeToLiveShort { 24_h * 7 };
43 static constexpr Seconds TimeToLiveLong { 24_h * 90 };
44
45 enum class CredentialTokenType : bool { LegacyCookie, HTTPStateToken };
46 enum class AuthenticationType : uint8_t { WebAuthn, PasswordManager, Unmanaged };
47
48 WEBCORE_EXPORT static ExceptionOr<UniqueRef<LoggedInStatus>> create(const RegistrableDomain&, const String& username, CredentialTokenType, AuthenticationType);
49 WEBCORE_EXPORT static ExceptionOr<UniqueRef<LoggedInStatus>> create(const RegistrableDomain&, const String& username, CredentialTokenType, AuthenticationType, Seconds timeToLive);
50
51 WEBCORE_EXPORT void setTimeToLive(Seconds);
52 WEBCORE_EXPORT bool hasExpired() const;
53 WEBCORE_EXPORT WallTime expiry() const;
54
55 const RegistrableDomain& registrableDomain() const { return m_domain; }
56 const String& username() const { return m_username; }
57 CredentialTokenType credentialTokenType() const { return m_tokenType; }
58 AuthenticationType authenticationType() const { return m_authType; }
59 WallTime loggedInTime() const { return m_loggedInTime; }
60
61private:
62 LoggedInStatus(const RegistrableDomain&, const String& username, CredentialTokenType, AuthenticationType, Seconds timeToLive);
63
64 RegistrableDomain m_domain;
65 String m_username;
66 CredentialTokenType m_tokenType;
67 AuthenticationType m_authType;
68 WallTime m_loggedInTime;
69 Seconds m_timeToLive;
70};
