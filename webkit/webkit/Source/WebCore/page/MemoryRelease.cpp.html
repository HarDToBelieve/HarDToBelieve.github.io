66static void releaseNoncriticalMemory(MaintainMemoryCache maintainMemoryCache)
67{
68 RenderTheme::singleton().purgeCaches();
69
70 FontCache::singleton().purgeInactiveFontData();
71 FontCache::singleton().clearWidthCaches();
72
73 TextPainter::clearGlyphDisplayLists();
74
75 for (auto* document : Document::allDocuments()) {
76 document->clearSelectorQueryCache();
77
78#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
79 if (auto* renderView = document->renderView())
80 LayoutIntegration::LineLayout::releaseCaches(*renderView);
81#endif
82 }
83
84 if (maintainMemoryCache == MaintainMemoryCache::No)
85 MemoryCache::singleton().pruneDeadResourcesToSize(0);
86
87 InlineStyleSheetOwner::clearCache();
88}

90static void releaseCriticalMemory(Synchronous synchronous, MaintainBackForwardCache maintainBackForwardCache, MaintainMemoryCache maintainMemoryCache)
91{
92 // Right now, the only reason we call release critical memory while not under memory pressure is if the process is about to be suspended.
93 if (maintainBackForwardCache == MaintainBackForwardCache::No) {
94 PruningReason pruningReason = MemoryPressureHandler::singleton().isUnderMemoryPressure() ? PruningReason::MemoryPressure : PruningReason::ProcessSuspended;
95 BackForwardCache::singleton().pruneToSizeNow(0, pruningReason);
96 }
97
98 if (maintainMemoryCache == MaintainMemoryCache::No) {
99 auto shouldDestroyDecodedDataForAllLiveResources = true;
100 MemoryCache::singleton().pruneLiveResourcesToSize(0, shouldDestroyDecodedDataForAllLiveResources);
101 }
102
103 CSSValuePool::singleton().drain();
104
105 Page::forEachPage([](auto& page) {
106 page.cookieJar().clearCache();
107 });
108
109 for (auto& document : copyToVectorOf<RefPtr<Document>>(Document::allDocuments())) {
110 document->styleScope().releaseMemory();
111 document->fontSelector().emptyCaches();
112 document->cachedResourceLoader().garbageCollectDocumentResources();
113 }
114
115 GCController::singleton().deleteAllCode(JSC::DeleteAllCodeIfNotCollecting);
116
117#if ENABLE(VIDEO)
118 for (auto* mediaElement : HTMLMediaElement::allMediaElements()) {
119 if (mediaElement->paused())
120 mediaElement->purgeBufferedDataIfPossible();
121 }
122#endif
123
124 if (synchronous == Synchronous::Yes) {
125 GCController::singleton().garbageCollectNow();
126 } else {
127#if PLATFORM(IOS_FAMILY)
128 GCController::singleton().garbageCollectNowIfNotDoneRecently();
129#else
130 GCController::singleton().garbageCollectSoon();
131#endif
132 }
133
134 WorkerGlobalScope::releaseMemoryInWorkers(synchronous);
135}

137void releaseMemory(Critical critical, Synchronous synchronous, MaintainBackForwardCache maintainBackForwardCache, MaintainMemoryCache maintainMemoryCache)
138{
139 TraceScope scope(MemoryPressureHandlerStart, MemoryPressureHandlerEnd, static_cast<uint64_t>(critical), static_cast<uint64_t>(synchronous));
140
141 if (critical == Critical::Yes) {
142 // Return unused pages back to the OS now as this will likely give us a little memory to work with.
143 WTF::releaseFastMallocFreeMemory();
144 releaseCriticalMemory(synchronous, maintainBackForwardCache, maintainMemoryCache);
145 }
146
147 releaseNoncriticalMemory(maintainMemoryCache);
148
149 platformReleaseMemory(critical);
150
151 if (synchronous == Synchronous::Yes) {
152 // FastMalloc has lock-free thread specific caches that can only be cleared from the thread itself.
153 WorkerOrWorkletThread::releaseFastMallocFreeMemoryInAllThreads();
154#if ENABLE(SCROLLING_THREAD)
155 ScrollingThread::dispatch(WTF::releaseFastMallocFreeMemory);
156#endif
157 WTF::releaseFastMallocFreeMemory();
158 }
159
160#if ENABLE(RESOURCE_USAGE)
161 Page::forEachPage([&](Page& page) {
162 InspectorInstrumentation::didHandleMemoryPressure(page, critical);
163 });
164#endif
165}

167void releaseGraphicsMemory(Critical critical, Synchronous synchronous)
168{
169 TraceScope scope(MemoryPressureHandlerStart, MemoryPressureHandlerEnd, static_cast<uint64_t>(critical), static_cast<uint64_t>(synchronous));
170
171 platformReleaseGraphicsMemory(critical);
172
173 WTF::releaseFastMallocFreeMemory();
174}

188void logMemoryStatisticsAtTimeOfDeath()
189{
190#if !RELEASE_LOG_DISABLED
191#if PLATFORM(COCOA)
192 auto pageSize = vmPageSize();
193 auto pages = pagesPerVMTag();
194
195 RELEASE_LOG(MemoryPressure, "Dirty memory per VM tag at time of death:");
196 for (unsigned i = 0; i < 256; ++i) {
197 size_t dirty = pages[i].dirty * pageSize;
198 if (!dirty)
199 continue;
200 String tagName = displayNameForVMTag(i);
201 if (!tagName)
202 tagName = makeString("Tag ", i);
203 RELEASE_LOG(MemoryPressure, "%16s: %lu MB", tagName.latin1().data(), dirty / MB);
204 }
205#endif
206
207 auto& vm = commonVM();
208 JSC::JSLockHolder locker(vm);
209 RELEASE_LOG(MemoryPressure, "Memory usage statistics at time of death:");
210 RELEASE_LOG(MemoryPressure, "GC heap size: %zu", vm.heap.size());
211 RELEASE_LOG(MemoryPressure, "GC heap extra memory size: %zu", vm.heap.extraMemorySize());
212#if ENABLE(RESOURCE_USAGE)
213 RELEASE_LOG(MemoryPressure, "GC heap external memory: %zu", vm.heap.externalMemorySize());
214#endif
215 RELEASE_LOG(MemoryPressure, "Global object count: %zu", vm.heap.globalObjectCount());
216
217 RELEASE_LOG(MemoryPressure, "Page count: %u", pageCount());
218 RELEASE_LOG(MemoryPressure, "Document count: %u", Document::allDocuments().size());
219 RELEASE_LOG(MemoryPressure, "Live JavaScript objects:");
220 auto typeCounts = vm.heap.objectTypeCounts();
221 for (auto& it : *typeCounts)
222 RELEASE_LOG(MemoryPressure, " %s: %d", it.key, it.value);
223#endif
224}
