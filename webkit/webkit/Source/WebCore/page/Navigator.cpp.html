77const String& Navigator::userAgent() const
78{
79 auto* frame = this->frame();
80 if (!frame || !frame->page())
81 return m_userAgent;
82 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
83 ResourceLoadObserver::shared().logNavigatorAPIAccessed(*frame->document(), ResourceLoadStatistics::NavigatorAPI::UserAgent);
84 if (m_userAgent.isNull())
85 m_userAgent = frame->loader().userAgent(frame->document()->url());
86 return m_userAgent;

113static std::optional<URL> shareableURLForShareData(ScriptExecutionContext& context, const ShareData& data)
114{
115 if (data.url.isNull())
116 return std::nullopt;
117
118 auto url = context.completeURL(data.url);
119 if (!url.isValid())
120 return std::nullopt;
121 if (!url.protocolIsInHTTPFamily() && !url.protocolIsData())
122 return std::nullopt;
123
124 return url;
125}

127bool Navigator::canShare(Document& document, const ShareData& data)
128{
129 auto* frame = this->frame();
130 if (!frame || !frame->page())
131 return false;
132
133 bool hasShareableTitleOrText = !data.title.isNull() || !data.text.isNull();
134 bool hasShareableURL = !!shareableURLForShareData(document, data);
135#if ENABLE(FILE_SHARE)
136 bool hasShareableFiles = document.settings().webShareFileAPIEnabled() && !data.files.isEmpty();
137#else
138 bool hasShareableFiles = false;
139#endif
140
141 return hasShareableTitleOrText || hasShareableURL || hasShareableFiles;
142}

144void Navigator::share(Document& document, const ShareData& data, Ref<DeferredPromise>&& promise)
145{
146 if (!canShare(document, data)) {
147 promise->reject(TypeError);
148 return;
149 }
150
151 auto* window = this->window();
152 // Note that the specification does not indicate we should consume user activation. We are intentionally stricter here.
153 if (!window || !window->consumeTransientActivation() || m_hasPendingShare) {
154 promise->reject(NotAllowedError);
155 return;
156 }
157
158 std::optional<URL> url = shareableURLForShareData(document, data);
159 ShareDataWithParsedURL shareData = {
160 data,
161 url,
162 { },
163 };
164#if ENABLE(FILE_SHARE)
165 if (document.settings().webShareFileAPIEnabled() && !data.files.isEmpty()) {
166 if (m_loader)
167 m_loader->cancel();
168
169 m_loader = ShareDataReader::create([this, promise = WTFMove(promise)] (ExceptionOr<ShareDataWithParsedURL&> readData) mutable {
170 showShareData(readData, WTFMove(promise));
171 });
172 m_loader->start(&document, WTFMove(shareData));
173 return;
174 }
175#endif
176 this->showShareData(shareData, WTFMove(promise));
177}

179void Navigator::showShareData(ExceptionOr<ShareDataWithParsedURL&> readData, Ref<DeferredPromise>&& promise)
180{
181 if (readData.hasException()) {
182 promise->reject(readData.releaseException());
183 return;
184 }
185
186 auto* frame = this->frame();
187 if (!frame || !frame->page())
188 return;
189
190 if (frame->page()->isControlledByAutomation()) {
191 promise->resolve();
192 return;
193 }
194
195 m_hasPendingShare = true;
196 auto shareData = readData.returnValue();
197
198 frame->page()->chrome().showShareSheet(shareData, [promise = WTFMove(promise), this] (bool completed) {
199 m_hasPendingShare = false;
200 if (completed) {
201 promise->resolve();
202 return;
203 }
204 promise->reject(Exception { AbortError, "Abort due to cancellation of share."_s });
205 });
206}

208void Navigator::initializePluginAndMimeTypeArrays()
209{
210 if (m_plugins)
211 return;
212
213 auto* frame = this->frame();
214 if (!frame || !frame->page()) {
215 m_plugins = DOMPluginArray::create(*this);
216 m_mimeTypes = DOMMimeTypeArray::create(*this);
217 return;
218 }
219
220 auto [publiclyVisiblePlugins, additionalWebVisiblePlugins] = frame->page()->pluginData().publiclyVisiblePluginsAndAdditionalWebVisiblePlugins();
221
222 Vector<Ref<DOMPlugin>> publiclyVisibleDOMPlugins;
223 Vector<Ref<DOMPlugin>> additionalWebVisibleDOMPlugins;
224 Vector<Ref<DOMMimeType>> webVisibleDOMMimeTypes;
225
226 publiclyVisibleDOMPlugins.reserveInitialCapacity(publiclyVisiblePlugins.size());
227 for (auto& plugin : publiclyVisiblePlugins) {
228 auto wrapper = DOMPlugin::create(*this, plugin);
229 webVisibleDOMMimeTypes.appendVector(wrapper->mimeTypes());
230 publiclyVisibleDOMPlugins.uncheckedAppend(WTFMove(wrapper));
231 }
232
233 additionalWebVisibleDOMPlugins.reserveInitialCapacity(additionalWebVisiblePlugins.size());
234 for (auto& plugin : additionalWebVisiblePlugins) {
235 auto wrapper = DOMPlugin::create(*this, plugin);
236 webVisibleDOMMimeTypes.appendVector(wrapper->mimeTypes());
237 additionalWebVisibleDOMPlugins.uncheckedAppend(WTFMove(wrapper));
238 }
239
240 std::sort(publiclyVisibleDOMPlugins.begin(), publiclyVisibleDOMPlugins.end(), [](const Ref<DOMPlugin>& a, const Ref<DOMPlugin>& b) {
241 if (auto nameComparison = codePointCompare(a->info().name, b->info().name))
242 return nameComparison < 0;
243 return codePointCompareLessThan(a->info().bundleIdentifier, b->info().bundleIdentifier);
244 });
245
246 std::sort(webVisibleDOMMimeTypes.begin(), webVisibleDOMMimeTypes.end(), [](const Ref<DOMMimeType>& a, const Ref<DOMMimeType>& b) {
247 if (auto typeComparison = codePointCompare(a->type(), b->type()))
248 return typeComparison < 0;
249 return codePointCompareLessThan(a->enabledPlugin()->info().bundleIdentifier, b->enabledPlugin()->info().bundleIdentifier);
250 });
251
252 // NOTE: It is not necessary to sort additionalWebVisibleDOMPlugins, as they are only accessible via
253 // named property look up, so their order is not exposed.
254
255 m_plugins = DOMPluginArray::create(*this, WTFMove(publiclyVisibleDOMPlugins), WTFMove(additionalWebVisibleDOMPlugins));
256 m_mimeTypes = DOMMimeTypeArray::create(*this, WTFMove(webVisibleDOMMimeTypes));
257}

279bool Navigator::cookieEnabled() const
280{
281 auto* frame = this->frame();
282 if (!frame)
283 return false;
284
285 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
286 ResourceLoadObserver::shared().logNavigatorAPIAccessed(*frame->document(), ResourceLoadStatistics::NavigatorAPI::CookieEnabled);
287
288 auto* page = frame->page();
289 if (!page)
290 return false;
291
292 if (!page->settings().cookieEnabled())
293 return false;
294
295 auto* document = frame->document();
296 if (!document)
297 return false;
298
299 return page->cookieJar().cookiesEnabled(*document);
300}

302bool Navigator::javaEnabled() const
303{
304 auto* frame = this->frame();
305 if (!frame)
306 return false;
307
308 if (RuntimeEnabledFeatures::sharedFeatures().webAPIStatisticsEnabled())
309 ResourceLoadObserver::shared().logNavigatorAPIAccessed(*frame->document(), ResourceLoadStatistics::NavigatorAPI::JavaEnabled);
310
311 if (!frame->settings().isJavaEnabled())
312 return false;
313 if (frame->document()->securityOrigin().isLocal() && !frame->settings().isJavaEnabledForLocalFiles())
314 return false;
315
316 return true;
317}
