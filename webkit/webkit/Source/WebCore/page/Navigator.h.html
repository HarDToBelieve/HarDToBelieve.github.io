37class Navigator final : public NavigatorBase, public ScriptWrappable, public DOMWindowProperty, public Supplementable<Navigator> {
38 WTF_MAKE_ISO_ALLOCATED(Navigator);
39public:
40 static Ref<Navigator> create(ScriptExecutionContext* context, DOMWindow& window) { return adoptRef(*new Navigator(context, window)); }
41 virtual ~Navigator();
42
43 String appVersion() const;
44 DOMPluginArray& plugins();
45 DOMMimeTypeArray& mimeTypes();
46 bool cookieEnabled() const;
47 bool javaEnabled() const;
48 const String& userAgent() const final;
49 String platform() const final;
50 void userAgentChanged();
51 bool onLine() const final;
52 bool canShare(Document&, const ShareData&);
53 void share(Document&, const ShareData&, Ref<DeferredPromise>&&);
54
55#if PLATFORM(IOS_FAMILY)
56 bool standalone() const;
57#endif
58
59 void getStorageUpdates();
60
61#if ENABLE(IOS_TOUCH_EVENTS) && !PLATFORM(MACCATALYST)
62 int maxTouchPoints() const { return 5; }
63#else
64 int maxTouchPoints() const { return 0; }
65#endif
66
67private:
68 void showShareData(ExceptionOr<ShareDataWithParsedURL&>, Ref<DeferredPromise>&&);
69 explicit Navigator(ScriptExecutionContext*, DOMWindow&);
70
71 void initializePluginAndMimeTypeArrays();
72
73 mutable RefPtr<ShareDataReader> m_loader;
74 mutable bool m_hasPendingShare { false };
75 mutable RefPtr<DOMPluginArray> m_plugins;
76 mutable RefPtr<DOMMimeTypeArray> m_mimeTypes;
77 mutable String m_userAgent;
78 mutable String m_platform;
79};
