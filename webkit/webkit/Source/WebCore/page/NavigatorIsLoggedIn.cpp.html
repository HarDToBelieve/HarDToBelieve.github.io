34NavigatorIsLoggedIn* NavigatorIsLoggedIn::from(Navigator& navigator)
35{
36 auto* supplement = static_cast<NavigatorIsLoggedIn*>(Supplement<Navigator>::from(&navigator, supplementName()));
37 if (!supplement) {
38 auto newSupplement = makeUnique<NavigatorIsLoggedIn>(navigator);
39 supplement = newSupplement.get();
40 provideTo(&navigator, supplementName(), WTFMove(newSupplement));
41 }
42 return supplement;
43}
44
45const char* NavigatorIsLoggedIn::supplementName()
46{
47 return "NavigatorIsLoggedIn";
48}
49
50void NavigatorIsLoggedIn::setLoggedIn(Navigator& navigator, Ref<DeferredPromise>&& promise)
51{
52 NavigatorIsLoggedIn::from(navigator)->setLoggedIn(WTFMove(promise));
53}

60void NavigatorIsLoggedIn::isLoggedIn(Navigator& navigator, Ref<DeferredPromise>&& promise)

64
65void NavigatorIsLoggedIn::setLoggedIn(Ref<DeferredPromise>&& promise)
66{
67 if (m_navigator.cookieEnabled())
68 promise->resolve();
69 else
70 promise->reject();
71}
72
73void NavigatorIsLoggedIn::setLoggedOut(Ref<DeferredPromise>&& promise)
74{
75 promise->resolve();
76}
77
78void NavigatorIsLoggedIn::isLoggedIn(Ref<DeferredPromise>&& promise)
79{
80 if (m_navigator.cookieEnabled())
81 promise->resolve<IDLBoolean>(true);
82 else
83 promise->resolve<IDLBoolean>(false);
84}
