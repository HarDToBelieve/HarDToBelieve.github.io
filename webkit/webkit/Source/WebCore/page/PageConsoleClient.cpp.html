120void PageConsoleClient::addMessage(std::unique_ptr<Inspector::ConsoleMessage>&& consoleMessage)
121{
122 if (!m_page.usesEphemeralSession()) {
123 String message;
124 if (consoleMessage->type() == MessageType::Image) {
125 ASSERT(consoleMessage->arguments());
126 consoleMessage->arguments()->getFirstArgumentAsString(message);
127 } else
128 message = consoleMessage->message();
129 m_page.chrome().client().addMessageToConsole(consoleMessage->source(), consoleMessage->level(), message, consoleMessage->line(), consoleMessage->column(), consoleMessage->url());
130
131 if (UNLIKELY(m_page.settings().logsPageMessagesToSystemConsoleEnabled() || shouldPrintExceptions())) {
132 if (consoleMessage->type() == MessageType::Image) {
133 ASSERT(consoleMessage->arguments());
134 ConsoleClient::printConsoleMessageWithArguments(consoleMessage->source(), consoleMessage->type(), consoleMessage->level(), consoleMessage->arguments()->globalObject(), *consoleMessage->arguments());
135 } else
136 ConsoleClient::printConsoleMessage(consoleMessage->source(), consoleMessage->type(), consoleMessage->level(), consoleMessage->toString(), consoleMessage->url(), consoleMessage->line(), consoleMessage->column());
137 }
138 }
139
140 InspectorInstrumentation::addMessageToConsole(m_page, WTFMove(consoleMessage));
141}

143void PageConsoleClient::addMessage(MessageSource source, MessageLevel level, const String& message, unsigned long requestIdentifier, Document* document)
144{
145 String url;
146 unsigned line = 0;
147 unsigned column = 0;
148 if (document)
149 document->getParserLocation(url, line, column);
150
151 addMessage(source, level, message, url, line, column, 0, JSExecState::currentState(), requestIdentifier);
152}

156 addMessage(source, level, message, String(), 0, 0, WTFMove(callStack), 0);

159void PageConsoleClient::addMessage(MessageSource source, MessageLevel level, const String& messageText, const String& suggestedURL, unsigned suggestedLineNumber, unsigned suggestedColumnNumber, RefPtr<ScriptCallStack>&& callStack, JSC::JSGlobalObject* lexicalGlobalObject, unsigned long requestIdentifier)
160{
161 if (muteCount && source != MessageSource::ConsoleAPI)
162 return;
163
164 std::unique_ptr<Inspector::ConsoleMessage> message;
165
166 if (callStack)
167 message = makeUnique<Inspector::ConsoleMessage>(source, MessageType::Log, level, messageText, callStack.releaseNonNull(), requestIdentifier);
168 else
169 message = makeUnique<Inspector::ConsoleMessage>(source, MessageType::Log, level, messageText, suggestedURL, suggestedLineNumber, suggestedColumnNumber, lexicalGlobalObject, requestIdentifier);
170
171 addMessage(WTFMove(message));
172}

175void PageConsoleClient::messageWithTypeAndLevel(MessageType type, MessageLevel level, JSC::JSGlobalObject* lexicalGlobalObject, Ref<Inspector::ScriptArguments>&& arguments)
176{
177 String messageText;
178 bool gotMessage = arguments->getFirstArgumentAsString(messageText);
179
180 auto message = makeUnique<Inspector::ConsoleMessage>(MessageSource::ConsoleAPI, type, level, messageText, arguments.copyRef(), lexicalGlobalObject);
181
182 String url = message->url();
183 unsigned lineNumber = message->line();
184 unsigned columnNumber = message->column();
185
186 InspectorInstrumentation::addMessageToConsole(m_page, WTFMove(message));
187
188 if (m_page.usesEphemeralSession())
189 return;
190
191 if (gotMessage)
192 m_page.chrome().client().addMessageToConsole(MessageSource::ConsoleAPI, level, messageText, lineNumber, columnNumber, url);
193
194 if (m_page.settings().logsPageMessagesToSystemConsoleEnabled() || PageConsoleClient::shouldPrintExceptions())
195 ConsoleClient::printConsoleMessageWithArguments(MessageSource::ConsoleAPI, type, level, lexicalGlobalObject, WTFMove(arguments));
196}

245static JSC::JSObject* objectArgumentAt(ScriptArguments& arguments, unsigned index)

248}

250static CanvasRenderingContext* canvasRenderingContext(JSC::VM& vm, JSC::JSValue target)
251{
252 if (auto* canvas = JSHTMLCanvasElement::toWrapped(vm, target))
253 return canvas->renderingContext();
254#if ENABLE(OFFSCREEN_CANVAS)
255 if (auto* canvas = JSOffscreenCanvas::toWrapped(vm, target))
256 return canvas->renderingContext();
257#endif
258 if (auto* context = JSCanvasRenderingContext2D::toWrapped(vm, target))
259 return context;
260 if (auto* context = JSImageBitmapRenderingContext::toWrapped(vm, target))
261 return context;
262#if ENABLE(WEBGL)
263 if (auto* context = JSWebGLRenderingContext::toWrapped(vm, target))
264 return context;
265#endif
266#if ENABLE(WEBGL2)
267 if (auto* context = JSWebGL2RenderingContext::toWrapped(vm, target))
268 return context;
269#endif
270 return nullptr;
271}

273void PageConsoleClient::record(JSC::JSGlobalObject* lexicalGlobalObject, Ref<ScriptArguments>&& arguments)
274{
275 if (LIKELY(!InspectorInstrumentation::hasFrontends()))
276 return;
277
278 if (auto* target = objectArgumentAt(arguments, 0)) {
279 if (auto* context = canvasRenderingContext(lexicalGlobalObject->vm(), target))
280 InspectorInstrumentation::consoleStartRecordingCanvas(*context, *lexicalGlobalObject, objectArgumentAt(arguments, 1));
281 }
282}
283
284void PageConsoleClient::recordEnd(JSC::JSGlobalObject* lexicalGlobalObject, Ref<ScriptArguments>&& arguments)
285{
286 if (LIKELY(!InspectorInstrumentation::hasFrontends()))
287 return;
288
289 if (auto* target = objectArgumentAt(arguments, 0)) {
290 if (auto* context = canvasRenderingContext(lexicalGlobalObject->vm(), target))
291 InspectorInstrumentation::consoleStopRecordingCanvas(*context);
292 }
293}

295static std::optional<String> snapshotCanvas(HTMLCanvasElement& canvasElement, CanvasRenderingContext& canvasRenderingContext)
296{
297#if ENABLE(WEBGL)
298 if (is<WebGLRenderingContextBase>(canvasRenderingContext))
299 downcast<WebGLRenderingContextBase>(canvasRenderingContext).setPreventBufferClearForInspector(true);
300#else
301 UNUSED_PARAM(canvasRenderingContext);
302#endif
303
304 auto result = canvasElement.toDataURL("image/png"_s);
305
306#if ENABLE(WEBGL)
307 if (is<WebGLRenderingContextBase>(canvasRenderingContext))
308 downcast<WebGLRenderingContextBase>(canvasRenderingContext).setPreventBufferClearForInspector(false);
309#endif
310
311 if (!result.hasException())
312 return result.releaseReturnValue().string;
313
314 return std::nullopt;
315}

317void PageConsoleClient::screenshot(JSC::JSGlobalObject* lexicalGlobalObject, Ref<ScriptArguments>&& arguments)
318{
319 JSC::VM& vm = lexicalGlobalObject->vm();
320 String dataURL;
321 JSC::JSValue target;
322
323 if (arguments->argumentCount()) {
324 auto possibleTarget = arguments->argumentAt(0);
325
326 if (auto* node = JSNode::toWrapped(vm, possibleTarget)) {
327 target = possibleTarget;
328 if (UNLIKELY(InspectorInstrumentation::hasFrontends())) {
329 RefPtr<ImageBuffer> snapshot;
330
331 // Only try to do something special for subclasses of Node if they're detached from the DOM tree.
332 if (!node->document().contains(node)) {
333 auto snapshotImageElement = [&snapshot] (HTMLImageElement& imageElement) {
334 if (auto* cachedImage = imageElement.cachedImage()) {
335 auto* image = cachedImage->image();
336 if (image && image != &Image::nullImage()) {
337 snapshot = ImageBuffer::create(image->size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
338 snapshot->context().drawImage(*image, FloatPoint(0, 0));
339 }
340 }
341 };
342
343 if (is<HTMLImageElement>(node))
344 snapshotImageElement(downcast<HTMLImageElement>(*node));
345 else if (is<HTMLPictureElement>(node)) {
346 if (auto* firstImage = childrenOfType<HTMLImageElement>(downcast<HTMLPictureElement>(*node)).first())
347 snapshotImageElement(*firstImage);
348 }
349#if ENABLE(VIDEO)
350 else if (is<HTMLVideoElement>(node)) {
351 auto& videoElement = downcast<HTMLVideoElement>(*node);
352 unsigned videoWidth = videoElement.videoWidth();
353 unsigned videoHeight = videoElement.videoHeight();
354 snapshot = ImageBuffer::create(FloatSize(videoWidth, videoHeight), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
355 videoElement.paintCurrentFrameInContext(snapshot->context(), FloatRect(0, 0, videoWidth, videoHeight));
356 }
357#endif
358 else if (is<HTMLCanvasElement>(node)) {
359 auto& canvasElement = downcast<HTMLCanvasElement>(*node);
360 if (auto* canvasRenderingContext = canvasElement.renderingContext()) {
361 if (auto result = snapshotCanvas(canvasElement, *canvasRenderingContext))
362 dataURL = result.value();
363 }
364 }
365 }
366
367 if (dataURL.isEmpty()) {
368 if (!snapshot)
369 snapshot = WebCore::snapshotNode(m_page.mainFrame(), *node, { { }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() });
370
371 if (snapshot)
372 dataURL = snapshot->toDataURL("image/png"_s, std::nullopt, PreserveResolution::Yes);
373 }
374 }
375 } else if (auto* imageData = JSImageData::toWrapped(vm, possibleTarget)) {
376 target = possibleTarget;
377 if (UNLIKELY(InspectorInstrumentation::hasFrontends())) {
378 auto sourceSize = imageData->size();
379 if (auto imageBuffer = ImageBuffer::create(sourceSize, RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8)) {
380 IntRect sourceRect(IntPoint(), sourceSize);
381 imageBuffer->putPixelBuffer(imageData->pixelBuffer(), sourceRect);
382 dataURL = imageBuffer->toDataURL("image/png"_s, std::nullopt, PreserveResolution::Yes);
383 }
384 }
385 } else if (auto* imageBitmap = JSImageBitmap::toWrapped(vm, possibleTarget)) {
386 target = possibleTarget;
387 if (UNLIKELY(InspectorInstrumentation::hasFrontends())) {
388 if (auto* imageBuffer = imageBitmap->buffer())
389 dataURL = imageBuffer->toDataURL("image/png"_s, std::nullopt, PreserveResolution::Yes);
390 }
391 } else if (auto* context = canvasRenderingContext(vm, possibleTarget)) {
392 auto& canvas = context->canvasBase();
393 if (is<HTMLCanvasElement>(canvas)) {
394 target = possibleTarget;
395 if (UNLIKELY(InspectorInstrumentation::hasFrontends())) {
396 if (auto result = snapshotCanvas(downcast<HTMLCanvasElement>(canvas), *context))
397 dataURL = result.value();
398 }
399 }
400
401 // FIXME: <https://webkit.org/b/180833> Web Inspector: support OffscreenCanvas for Canvas related operations
402 } else {
403 String base64;
404 if (possibleTarget.getString(lexicalGlobalObject, base64) && base64.startsWithIgnoringASCIICase("data:"_s) && base64.length() > 5) {
405 target = possibleTarget;
406 dataURL = base64;
407 }
408 }
409 }
410
411 if (UNLIKELY(InspectorInstrumentation::hasFrontends())) {
412 if (!target) {
413 // If no target is provided, capture an image of the viewport.
414 IntRect imageRect(IntPoint::zero(), m_page.mainFrame().view()->sizeForVisibleContent());
415 if (auto snapshot = WebCore::snapshotFrameRect(m_page.mainFrame(), imageRect, { { SnapshotFlags::InViewCoordinates }, PixelFormat::BGRA8, DestinationColorSpace::SRGB() }))
416 dataURL = snapshot->toDataURL("image/png"_s, std::nullopt, PreserveResolution::Yes);
417 }
418
419 if (dataURL.isEmpty()) {
420 addMessage(makeUnique<Inspector::ConsoleMessage>(MessageSource::ConsoleAPI, MessageType::Image, MessageLevel::Error, "Could not capture screenshot"_s, WTFMove(arguments)));
421 return;
422 }
423 }
424
425 Vector<JSC::Strong<JSC::Unknown>> adjustedArguments;
426 adjustedArguments.append({ vm, target ? target : JSC::jsNontrivialString(vm, "Viewport"_s) });
427 for (size_t i = (!target ? 0 : 1); i < arguments->argumentCount(); ++i)
428 adjustedArguments.append({ vm, arguments->argumentAt(i) });
429 arguments = ScriptArguments::create(lexicalGlobalObject, WTFMove(adjustedArguments));
430 addMessage(makeUnique<Inspector::ConsoleMessage>(MessageSource::ConsoleAPI, MessageType::Image, MessageLevel::Log, dataURL, WTFMove(arguments)));
431}
