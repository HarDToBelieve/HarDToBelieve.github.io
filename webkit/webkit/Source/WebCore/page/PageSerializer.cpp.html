68static bool isCharsetSpecifyingNode(const Node& node)
69{
70 if (!is<HTMLElement>(node))
71 return false;
72
73 const HTMLElement& element = downcast<HTMLElement>(node);
74 if (!element.hasTagName(HTMLNames::metaTag))
75 return false;
76 HTMLMetaCharsetParser::AttributeList attributes;
77 if (element.hasAttributes()) {
78 for (const Attribute& attribute : element.attributesIterator()) {
79 // FIXME: We should deal appropriately with the attribute if they have a namespace.
80 attributes.append(std::make_pair(attribute.name().toString(), attribute.value().string()));
81 }
82 }
83 return HTMLMetaCharsetParser::encodingFromMetaAttributes(attributes).isValid();
84}
85
86static bool shouldIgnoreElement(const Element& element)
87{
88 return element.hasTagName(HTMLNames::scriptTag) || element.hasTagName(HTMLNames::noscriptTag) || isCharsetSpecifyingNode(element);
89}

91static const QualifiedName& frameOwnerURLAttributeName(const HTMLFrameOwnerElement& frameOwner)
92{
93 // FIXME: We should support all frame owners including applets.
94 return is<HTMLObjectElement>(frameOwner) ? HTMLNames::dataAttr : HTMLNames::srcAttr;
95}

111PageSerializer::SerializerMarkupAccumulator::SerializerMarkupAccumulator(PageSerializer& serializer, Document& document, Vector<Node*>* nodes)
112 : MarkupAccumulator(nodes, ResolveURLs::Yes)
113 , m_serializer(serializer)
114 , m_document(document)
115{
116 // MarkupAccumulator does not serialize the <?xml ... line, so we add it explicitly to ensure the right encoding is specified.
117 if (m_document.isXMLDocument() || m_document.xmlStandalone())
118 append("<?xml version=\"", m_document.xmlVersion(), "\" encoding=\"", m_document.charset(), "\"?>");
119}

121void PageSerializer::SerializerMarkupAccumulator::appendText(StringBuilder& out, const Text& text)
122{
123 Element* parent = text.parentElement();
124 if (parent && !shouldIgnoreElement(*parent))
125 MarkupAccumulator::appendText(out, text);

128void PageSerializer::SerializerMarkupAccumulator::appendStartTag(StringBuilder& out, const Element& element, Namespaces* namespaces)
129{
130 if (!shouldIgnoreElement(element))
131 MarkupAccumulator::appendStartTag(out, element, namespaces);
132
133 if (element.hasTagName(HTMLNames::headTag))
134 out.append("<meta charset=\"", m_document.charset(), "\">");
135
136 // FIXME: For object (plugins) tags and video tag we could replace them by an image of their current contents.
137}

139void PageSerializer::SerializerMarkupAccumulator::appendCustomAttributes(StringBuilder& out, const Element& element, Namespaces* namespaces)
140{
141 if (!is<HTMLFrameOwnerElement>(element))
142 return;
143
144 const HTMLFrameOwnerElement& frameOwner = downcast<HTMLFrameOwnerElement>(element);
145 Frame* frame = frameOwner.contentFrame();
146 if (!frame)
147 return;
148
149 URL url = frame->document()->url();
150 if (url.isValid() && !url.protocolIsAbout())
151 return;
152
153 // We need to give a fake location to blank frames so they can be referenced by the serialized frame.
154 url = m_serializer.urlForBlankFrame(frame);
155 appendAttribute(out, element, Attribute(frameOwnerURLAttributeName(frameOwner), url.string()), namespaces);
156}

158void PageSerializer::SerializerMarkupAccumulator::appendEndTag(StringBuilder& out, const Element& element)
159{
160 if (!shouldIgnoreElement(element))
161 MarkupAccumulator::appendEndTag(out, element);
162}

174void PageSerializer::serializeFrame(Frame* frame)
175{
176 Document* document = frame->document();
177 URL url = document->url();
178 if (!url.isValid() || url.protocolIsAbout()) {
179 // For blank frames we generate a fake URL so they can be referenced by their containing frame.
180 url = urlForBlankFrame(frame);
181 }
182
183 if (m_resourceURLs.contains(url)) {
184 // FIXME: We could have 2 frame with the same URL but which were dynamically changed and have now
185 // different content. So we should serialize both and somehow rename the frame src in the containing
186 // frame. Arg!
187 return;
188 }
189
190 Vector<Node*> nodes;
191 SerializerMarkupAccumulator accumulator(*this, *document, &nodes);
192 TextEncoding textEncoding(document->charset());
193 CString data;
194 if (!textEncoding.isValid()) {
195 // FIXME: iframes used as images trigger this. We should deal with them correctly.
196 return;
197 }
198 String text = accumulator.serializeNodes(*document->documentElement(), SerializedNodes::SubtreeIncludingNode);
199 m_resources.append({ url, document->suggestedMIMEType(), SharedBuffer::create(textEncoding.encode(text, UnencodableHandling::Entities)) });
200 m_resourceURLs.add(url);
201
202 for (auto& node : nodes) {
203 if (!is<Element>(*node))
204 continue;
205
206 Element& element = downcast<Element>(*node);
207 // We have to process in-line style as it might contain some resources (typically background images).
208 if (is<StyledElement>(element))
209 retrieveResourcesForProperties(downcast<StyledElement>(element).inlineStyle(), document);
210
211 if (is<HTMLImageElement>(element)) {
212 HTMLImageElement& imageElement = downcast<HTMLImageElement>(element);
213 URL url = document->completeURL(imageElement.attributeWithoutSynchronization(HTMLNames::srcAttr));
214 CachedImage* cachedImage = imageElement.cachedImage();
215 addImageToResources(cachedImage, imageElement.renderer(), url);
216 } else if (is<HTMLLinkElement>(element)) {
217 HTMLLinkElement& linkElement = downcast<HTMLLinkElement>(element);
218 if (CSSStyleSheet* sheet = linkElement.sheet()) {
219 URL url = document->completeURL(linkElement.attributeWithoutSynchronization(HTMLNames::hrefAttr));
220 serializeCSSStyleSheet(sheet, url);
221 ASSERT(m_resourceURLs.contains(url));
222 }
223 } else if (is<HTMLStyleElement>(element)) {
224 if (CSSStyleSheet* sheet = downcast<HTMLStyleElement>(element).sheet())
225 serializeCSSStyleSheet(sheet, URL());
226 }
227 }
228
229 for (Frame* childFrame = frame->tree().firstChild(); childFrame; childFrame = childFrame->tree().nextSibling())
230 serializeFrame(childFrame);
231}

233void PageSerializer::serializeCSSStyleSheet(CSSStyleSheet* styleSheet, const URL& url)
234{
235 StringBuilder cssText;
236 for (unsigned i = 0; i < styleSheet->length(); ++i) {
237 CSSRule* rule = styleSheet->item(i);
238 String itemText = rule->cssText();
239 if (!itemText.isEmpty()) {
240 cssText.append(itemText);
241 if (i < styleSheet->length() - 1)
242 cssText.append("\n\n");
243 }
244 Document* document = styleSheet->ownerDocument();
245 // Some rules have resources associated with them that we need to retrieve.
246 if (is<CSSImportRule>(*rule)) {
247 CSSImportRule& importRule = downcast<CSSImportRule>(*rule);
248 URL importURL = document->completeURL(importRule.href());
249 if (m_resourceURLs.contains(importURL))
250 continue;
251 serializeCSSStyleSheet(importRule.styleSheet(), importURL);
252 } else if (is<CSSFontFaceRule>(*rule)) {
253 // FIXME: Add support for font face rule. It is not clear to me at this point if the actual otf/eot file can
254 // be retrieved from the CSSFontFaceRule object.
255 } else if (is<CSSStyleRule>(*rule))
256 retrieveResourcesForRule(downcast<CSSStyleRule>(*rule).styleRule(), document);
257 }
258
259 if (url.isValid() && !m_resourceURLs.contains(url)) {
260 // FIXME: We should check whether a charset has been specified and if none was found add one.
261 TextEncoding textEncoding(styleSheet->contents().charset());
262 ASSERT(textEncoding.isValid());
263 m_resources.append({ url, "text/css"_s, SharedBuffer::create(textEncoding.encode(cssText.toString(), UnencodableHandling::Entities)) });
264 m_resourceURLs.add(url);
265 }
266}

268void PageSerializer::addImageToResources(CachedImage* image, RenderElement* imageRenderer, const URL& url)
269{
270 if (!url.isValid() || m_resourceURLs.contains(url))
271 return;
272
273 if (!image || image->image() == &Image::nullImage())
274 return;
275
276 RefPtr<SharedBuffer> data = imageRenderer ? image->imageForRenderer(imageRenderer)->data() : 0;
277 if (!data)
278 data = image->image()->data();
279
280 if (!data) {
281 LOG_ERROR("No data for image %s", url.string().utf8().data());
282 return;
283 }
284
285 m_resources.append({ url, image->response().mimeType(), WTFMove(data) });
286 m_resourceURLs.add(url);
287}

294void PageSerializer::retrieveResourcesForProperties(const StyleProperties* styleDeclaration, Document* document)
295{
296 if (!styleDeclaration)
297 return;
298
299 // The background-image and list-style-image (for ul or ol) are the CSS properties
300 // that make use of images. We iterate to make sure we include any other
301 // image properties there might be.
302 unsigned propertyCount = styleDeclaration->propertyCount();
303 for (unsigned i = 0; i < propertyCount; ++i) {
304 RefPtr<CSSValue> cssValue = styleDeclaration->propertyAt(i).value();
305 if (!is<CSSImageValue>(*cssValue))
306 continue;
307
308 auto* image = downcast<CSSImageValue>(*cssValue).cachedImage();
309 if (!image)
310 continue;
311
312 addImageToResources(image, nullptr, document->completeURL(image->url().string()));
313 }

316URL PageSerializer::urlForBlankFrame(Frame* frame)
317{
318 auto iterator = m_blankFrameURLs.find(frame);
319 if (iterator != m_blankFrameURLs.end())
320 return iterator->value;
321 URL fakeURL { { }, makeString("wyciwyg://frame/", m_blankFrameCounter++) };
322 m_blankFrameURLs.add(frame, fakeURL);
323 return fakeURL;
324}
