38DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(PerformanceEntry);
39
40PerformanceEntry::PerformanceEntry(const String& name, double startTime, double finishTime)
41 : m_name(name)
42 , m_startTime(startTime)
43 , m_duration(finishTime - startTime)
44{
45}
46
47PerformanceEntry::~PerformanceEntry() = default;
48
49std::optional<PerformanceEntry::Type> PerformanceEntry::parseEntryTypeString(const String& entryType)
50{
51 if (entryType == "navigation")
52 return std::optional<Type>(Type::Navigation);
53
54 if (entryType == "mark")
55 return std::optional<Type>(Type::Mark);
56 if (entryType == "measure")
57 return std::optional<Type>(Type::Measure);
58
59 if (entryType == "resource")
60 return std::optional<Type>(Type::Resource);
61
62 if (RuntimeEnabledFeatures::sharedFeatures().paintTimingEnabled()) {
63 if (entryType == "paint")
64 return std::optional<Type>(Type::Paint);
65 }
66
67 return std::nullopt;
68}
