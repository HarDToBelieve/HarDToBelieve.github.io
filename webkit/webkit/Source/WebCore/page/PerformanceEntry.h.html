32#pragma once
33
34#include "Performance.h"
35#include <wtf/RefCounted.h>
36#include <wtf/text/WTFString.h>
37
38namespace WebCore {
39
40DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(PerformanceEntry);
41class PerformanceEntry : public RefCounted<PerformanceEntry> {
42 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(PerformanceEntry);
43public:
44 virtual ~PerformanceEntry();
45
46 const String& name() const { return m_name; }
47 virtual double startTime() const { return m_startTime; }
48 virtual double duration() const { return m_duration; }
49
50 enum class Type : uint8_t {
51 Navigation = 1 << 0,
52 Mark = 1 << 1,
53 Measure = 1 << 2,
54 Resource = 1 << 3,
55 Paint = 1 << 4
56 };
57
58 virtual Type performanceEntryType() const = 0;
59 virtual ASCIILiteral entryType() const = 0;
60
61 static std::optional<Type> parseEntryTypeString(const String& entryType);
62
63 static bool startTimeCompareLessThan(const RefPtr<PerformanceEntry>& a, const RefPtr<PerformanceEntry>& b)
64 {
65 return a->startTime() < b->startTime();
66 }
67
68protected:
69 PerformanceEntry(const String& name, double startTime, double finishTime);
70
71private:
72 const String m_name;
73 const double m_startTime;
74 const double m_duration;
75};

77} // namespace WebCore
