41static double performanceNow(ScriptExecutionContext& scriptExecutionContext)
42{
43 // FIXME: We should consider moving the Performance object to be owned by the
44 // the ScriptExecutionContext to avoid this.
45
46 if (is<Document>(scriptExecutionContext)) {
47 if (auto window = downcast<Document>(scriptExecutionContext).domWindow())
48 return window->performance().now();
49 return 0;
50 }
51
52 if (is<WorkerGlobalScope>(scriptExecutionContext))
53 return downcast<WorkerGlobalScope>(scriptExecutionContext).performance().now();
54
55 return 0;
56}

58ExceptionOr<Ref<PerformanceMark>> PerformanceMark::create(JSC::JSGlobalObject& globalObject, ScriptExecutionContext& scriptExecutionContext, const String& name, std::optional<PerformanceMarkOptions>&& markOptions)
59{
60 if (is<Document>(scriptExecutionContext) && PerformanceUserTiming::isRestrictedMarkName(name))
61 return Exception { SyntaxError };
62
63 double startTime;
64 JSC::JSValue detail;
65 if (markOptions) {
66 if (markOptions->startTime) {
67 if (*markOptions->startTime < 0)
68 return Exception { TypeError };
69 startTime = *markOptions->startTime;
70 } else
71 startTime = performanceNow(scriptExecutionContext);
72
73 if (markOptions->detail.isUndefined())
74 detail = JSC::jsNull();
75 else
76 detail = markOptions->detail;
77 } else {
78 startTime = performanceNow(scriptExecutionContext);
79 detail = JSC::jsNull();
80 }
81
82 Vector<RefPtr<MessagePort>> ignoredMessagePorts;
83 auto serializedDetail = SerializedScriptValue::create(globalObject, detail, { }, ignoredMessagePorts);
84 if (serializedDetail.hasException())
85 return serializedDetail.releaseException();
86
87 return adoptRef(*new PerformanceMark(name, startTime, serializedDetail.releaseReturnValue()));
88}
