39PerformanceObserver::PerformanceObserver(ScriptExecutionContext& scriptExecutionContext, Ref<PerformanceObserverCallback>&& callback)
40 : m_callback(WTFMove(callback))
41{
42 if (is<Document>(scriptExecutionContext)) {
43 auto& document = downcast<Document>(scriptExecutionContext);
44 if (DOMWindow* window = document.domWindow())
45 m_performance = &window->performance();
46 } else if (is<WorkerGlobalScope>(scriptExecutionContext)) {
47 auto& workerGlobalScope = downcast<WorkerGlobalScope>(scriptExecutionContext);
48 m_performance = &workerGlobalScope.performance();
49 } else
50 ASSERT_NOT_REACHED();
51}

53void PerformanceObserver::disassociate()
54{
55 m_performance = nullptr;
56 m_registered = false;
57}

59ExceptionOr<void> PerformanceObserver::observe(Init&& init)
60{
61 if (!m_performance)
62 return Exception { TypeError };
63
64 bool isBuffered = false;
65 OptionSet<PerformanceEntry::Type> filter;
66 if (init.entryTypes) {
67 if (init.type)
68 return Exception { TypeError, "either entryTypes or type must be provided"_s };
69 if (m_registered && m_isTypeObserver)
70 return Exception { InvalidModificationError, "observer type can't be changed once registered"_s };
71 for (auto& entryType : *init.entryTypes) {
72 if (auto type = PerformanceEntry::parseEntryTypeString(entryType))
73 filter.add(*type);
74 }
75 if (filter.isEmpty())
76 return { };
77 m_typeFilter = filter;
78 } else {
79 if (!init.type)
80 return Exception { TypeError, "no type or entryTypes were provided"_s };
81 if (m_registered && !m_isTypeObserver)
82 return Exception { InvalidModificationError, "observer type can't be changed once registered"_s };
83 m_isTypeObserver = true;
84 if (auto type = PerformanceEntry::parseEntryTypeString(*init.type))
85 filter.add(*type);
86 else
87 return { };
88 if (init.buffered) {
89 isBuffered = true;
90 auto oldSize = m_entriesToDeliver.size();
91 m_performance->appendBufferedEntriesByType(*init.type, m_entriesToDeliver, *this);
92 auto begin = m_entriesToDeliver.begin();
93 auto oldEnd = begin + oldSize;
94 auto end = m_entriesToDeliver.end();
95 std::stable_sort(oldEnd, end, PerformanceEntry::startTimeCompareLessThan);
96 std::inplace_merge(begin, oldEnd, end, PerformanceEntry::startTimeCompareLessThan);
97 }
98 m_typeFilter.add(filter);
99 }
100
101 if (!m_registered) {
102 m_performance->registerPerformanceObserver(*this);
103 m_registered = true;
104 }
105 if (isBuffered)
106 deliver();
107
108 return { };
109}

111Vector<RefPtr<PerformanceEntry>> PerformanceObserver::takeRecords()
112{
113 return std::exchange(m_entriesToDeliver, { });
114}
115
116void PerformanceObserver::disconnect()
117{
118 if (m_performance)
119 m_performance->unregisterPerformanceObserver(*this);
120
121 m_registered = false;
122 m_entriesToDeliver.clear();
123 m_typeFilter = { };
124}
125
126void PerformanceObserver::queueEntry(PerformanceEntry& entry)
127{
128 m_entriesToDeliver.append(&entry);
129}
130
131void PerformanceObserver::deliver()
132{
133 if (m_entriesToDeliver.isEmpty())
134 return;
135
136 auto* context = m_callback->scriptExecutionContext();
137 if (!context)
138 return;
139
140 Vector<RefPtr<PerformanceEntry>> entries = std::exchange(m_entriesToDeliver, { });
141 auto list = PerformanceObserverEntryList::create(WTFMove(entries));
142
143 InspectorInstrumentation::willFireObserverCallback(*context, "PerformanceObserver"_s);
144 m_callback->handleEvent(*this, list, *this);
145 InspectorInstrumentation::didFireObserverCallback(*context);
146}
147
148Vector<String> PerformanceObserver::supportedEntryTypes(ScriptExecutionContext& context)
149{
150 Vector<String> entryTypes = {
151 "mark"_s,
152 "measure"_s,
153 };
154
155 if (context.settingsValues().performanceNavigationTimingAPIEnabled)
156 entryTypes.append("navigation"_s);
157
158 if (is<Document>(context) && downcast<Document>(context).supportsPaintTiming())
159 entryTypes.append("paint"_s);
160
161 entryTypes.append("resource"_s);
162
163 return entryTypes;
164}
