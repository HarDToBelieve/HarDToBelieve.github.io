26#pragma once
27
28#include "ExceptionOr.h"
29#include "PerformanceEntry.h"
30#include "PerformanceObserverCallback.h"
31#include <wtf/OptionSet.h>
32#include <wtf/RefCounted.h>
33#include <wtf/Vector.h>
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38class Performance;
39class ScriptExecutionContext;
40
41class PerformanceObserver : public RefCounted<PerformanceObserver> {
42public:
43 struct Init {
44 std::optional<Vector<String>> entryTypes;
45 std::optional<String> type;
46 bool buffered;
47 };
48
49 static Ref<PerformanceObserver> create(ScriptExecutionContext& context, Ref<PerformanceObserverCallback>&& callback)
50 {
51 return adoptRef(*new PerformanceObserver(context, WTFMove(callback)));
52 }
53
54 static Vector<String> supportedEntryTypes(ScriptExecutionContext&);
55
56 void disassociate();
57
58 ExceptionOr<void> observe(Init&&);
59 void disconnect();
60 Vector<RefPtr<PerformanceEntry>> takeRecords();
61
62 OptionSet<PerformanceEntry::Type> typeFilter() const { return m_typeFilter; }
63
64 bool hasNavigationTiming() const { return m_hasNavigationTiming; }
65 void addedNavigationTiming() { m_hasNavigationTiming = true; }
66
67 void queueEntry(PerformanceEntry&);
68 void deliver();
69
70 bool isRegistered() const { return m_registered; }
71 PerformanceObserverCallback& callback() { return m_callback.get(); }
72
73private:
74 PerformanceObserver(ScriptExecutionContext&, Ref<PerformanceObserverCallback>&&);
75
76 RefPtr<Performance> m_performance;
77 Vector<RefPtr<PerformanceEntry>> m_entriesToDeliver;
78 Ref<PerformanceObserverCallback> m_callback;
79 OptionSet<PerformanceEntry::Type> m_typeFilter;
80 bool m_registered { false };
81 bool m_isTypeObserver { false };
82 bool m_hasNavigationTiming { false };
83};
84
85} // namespace WebCore
