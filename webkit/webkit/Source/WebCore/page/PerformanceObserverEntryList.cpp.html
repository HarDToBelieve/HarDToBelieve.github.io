33Ref<PerformanceObserverEntryList> PerformanceObserverEntryList::create(Vector<RefPtr<PerformanceEntry>>&& entries)
34{
35 return adoptRef(*new PerformanceObserverEntryList(WTFMove(entries)));
36}
37
38PerformanceObserverEntryList::PerformanceObserverEntryList(Vector<RefPtr<PerformanceEntry>>&& entries)
39 : m_entries(WTFMove(entries))
40{
41 ASSERT(!m_entries.isEmpty());
42
43 std::stable_sort(m_entries.begin(), m_entries.end(), PerformanceEntry::startTimeCompareLessThan);
44}
45
46Vector<RefPtr<PerformanceEntry>> PerformanceObserverEntryList::getEntriesByType(const String& entryType) const
47{
48 return getEntriesByName(String(), entryType);
49}
50
51Vector<RefPtr<PerformanceEntry>> PerformanceObserverEntryList::getEntriesByName(const String& name, const String& entryType) const
52{
53 Vector<RefPtr<PerformanceEntry>> entries;
54
55 // PerformanceObservers can only be registered for valid types.
56 // So if the incoming entryType is an unknown type, there will be no matches.
57 std::optional<PerformanceEntry::Type> type;
58 if (!entryType.isNull()) {
59 type = PerformanceEntry::parseEntryTypeString(entryType);
60 if (!type)
61 return entries;
62 }
63
64 for (auto& entry : m_entries) {
65 if (!name.isNull() && entry->name() != name)
66 continue;
67 if (type && entry->performanceEntryType() != *type)
68 continue;
69 entries.append(entry.copyRef());
70 }
71
72 return entries;
73}
