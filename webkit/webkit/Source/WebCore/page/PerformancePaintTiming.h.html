26#pragma once
27
28#include "PerformanceEntry.h"
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class PerformancePaintTiming final : public PerformanceEntry {
34public:
35 static Ref<PerformancePaintTiming> createFirstContentfulPaint(DOMHighResTimeStamp timeStamp)
36 {
37 return adoptRef(*new PerformancePaintTiming("first-contentful-paint"_s, timeStamp));
38 }
39
40private:
41 PerformancePaintTiming(const String& name, DOMHighResTimeStamp timeStamp)
42 : PerformanceEntry(name, timeStamp, timeStamp)
43 {
44 }
45
46 ASCIILiteral entryType() const final { return "paint"_s; }
47 Type performanceEntryType() const final { return Type::Paint; }
48
49 ~PerformancePaintTiming() = default;
50};
51
52} // namespace WebCore
