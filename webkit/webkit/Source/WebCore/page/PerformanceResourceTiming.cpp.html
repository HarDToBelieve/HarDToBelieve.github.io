46static double networkLoadTimeToDOMHighResTimeStamp(MonotonicTime timeOrigin, MonotonicTime timeStamp)
47{
48 if (!timeStamp)
49 return 0.0;
50 ASSERT(timeOrigin);
51 return Performance::reduceTimeResolution(timeStamp - timeOrigin).milliseconds();
52}
53
54static double fetchStart(MonotonicTime timeOrigin, const ResourceTiming& resourceTiming)
55{
56 if (auto fetchStart = resourceTiming.networkLoadMetrics().fetchStart)
57 return networkLoadTimeToDOMHighResTimeStamp(timeOrigin, fetchStart);
58
59 // fetchStart is a required property.
60 auto startTime = resourceTiming.resourceLoadTiming().startTime();
61 ASSERT(startTime);
62 return networkLoadTimeToDOMHighResTimeStamp(timeOrigin, startTime);
63}
64
65static double entryStartTime(MonotonicTime timeOrigin, const ResourceTiming& resourceTiming)
66{
67 if (resourceTiming.networkLoadMetrics().failsTAOCheck
68 || !resourceTiming.networkLoadMetrics().redirectCount)
69 return fetchStart(timeOrigin, resourceTiming);
70
71 if (resourceTiming.networkLoadMetrics().redirectStart)
72 return networkLoadTimeToDOMHighResTimeStamp(timeOrigin, resourceTiming.networkLoadMetrics().redirectStart);
73
74 return networkLoadTimeToDOMHighResTimeStamp(timeOrigin, resourceTiming.resourceLoadTiming().startTime());
75}
76
77static double entryEndTime(MonotonicTime timeOrigin, const ResourceTiming& resourceTiming)
78{
79 if (resourceTiming.networkLoadMetrics().responseEnd)
80 return networkLoadTimeToDOMHighResTimeStamp(timeOrigin, resourceTiming.networkLoadMetrics().responseEnd);
81
82 return networkLoadTimeToDOMHighResTimeStamp(timeOrigin, resourceTiming.resourceLoadTiming().endTime());
83}

111double PerformanceResourceTiming::redirectStart() const
112{
113 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
114 return 0.0;
115
116 if (!m_resourceTiming.networkLoadMetrics().redirectCount)
117 return 0.0;
118
119 return networkLoadTimeToDOMHighResTimeStamp(m_timeOrigin, m_resourceTiming.networkLoadMetrics().redirectStart);
120}
121
122double PerformanceResourceTiming::redirectEnd() const
123{
124 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
125 return 0.0;
126
127 if (!m_resourceTiming.networkLoadMetrics().redirectCount)
128 return 0.0;
129
130 // These two times are so close to each other that we don't record two timestamps.
131 // See https://www.w3.org/TR/resource-timing-2/#attribute-descriptions
132 return fetchStart();
133}

184double PerformanceResourceTiming::secureConnectionStart() const
185{
186 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
187 return 0.0;
188
189 if (m_resourceTiming.networkLoadMetrics().secureConnectionStart == reusedTLSConnectionSentinel)
190 return fetchStart();

198double PerformanceResourceTiming::requestStart() const
199{
200 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
201 return 0.0;
202
203 // requestStart is 0 when a network request is not made.
204 if (!m_resourceTiming.networkLoadMetrics().requestStart)
205 return connectEnd();
206
207 return networkLoadTimeToDOMHighResTimeStamp(m_timeOrigin, m_resourceTiming.networkLoadMetrics().requestStart);
208}

210double PerformanceResourceTiming::responseStart() const
211{
212 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
213 return 0.0;
214
215 // responseStart is 0 when a network request is not made.
216 if (!m_resourceTiming.networkLoadMetrics().responseStart)
217 return requestStart();
218
219 return networkLoadTimeToDOMHighResTimeStamp(m_timeOrigin, m_resourceTiming.networkLoadMetrics().responseStart);
220}
221
222double PerformanceResourceTiming::responseEnd() const
223{
224 // responseEnd is a required property.
225 ASSERT(m_resourceTiming.networkLoadMetrics().isComplete() || m_resourceTiming.resourceLoadTiming().endTime());
226
227 if (m_resourceTiming.networkLoadMetrics().isComplete()) {
228 if (m_resourceTiming.networkLoadMetrics().responseEnd)
229 return networkLoadTimeToDOMHighResTimeStamp(m_timeOrigin, m_resourceTiming.networkLoadMetrics().responseEnd);
230
231 // responseEnd is 0 when a network request is not made.
232 // This should mean all other properties are empty.
233 ASSERT(!m_resourceTiming.networkLoadMetrics().responseStart);
234 ASSERT(!m_resourceTiming.networkLoadMetrics().requestStart);
235 ASSERT(!m_resourceTiming.networkLoadMetrics().requestStart);
236 ASSERT(!m_resourceTiming.networkLoadMetrics().secureConnectionStart);
237 ASSERT(!m_resourceTiming.networkLoadMetrics().connectEnd);
238 ASSERT(!m_resourceTiming.networkLoadMetrics().connectStart);
239 ASSERT(!m_resourceTiming.networkLoadMetrics().domainLookupEnd);
240 ASSERT(!m_resourceTiming.networkLoadMetrics().domainLookupStart);
241 }
242
243 return networkLoadTimeToDOMHighResTimeStamp(m_timeOrigin, m_resourceTiming.resourceLoadTiming().endTime());
244}

246uint64_t PerformanceResourceTiming::transferSize() const
247{
248 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
249 return 0;
250
251 auto encodedBodySize = m_resourceTiming.networkLoadMetrics().responseBodyBytesReceived;
252 if (encodedBodySize == std::numeric_limits<uint64_t>::max())
253 return 0;
254
255 // https://w3c.github.io/resource-timing/#dom-performanceresourcetiming-transfersize
256 return encodedBodySize + 300;
257}

259uint64_t PerformanceResourceTiming::encodedBodySize() const
260{
261 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
262 return 0;
263
264 auto encodedBodySize = m_resourceTiming.networkLoadMetrics().responseBodyBytesReceived;
265 if (encodedBodySize == std::numeric_limits<uint64_t>::max())
266 return 0;
267
268 return encodedBodySize;
269}

271uint64_t PerformanceResourceTiming::decodedBodySize() const
272{
273 if (m_resourceTiming.networkLoadMetrics().failsTAOCheck)
274 return 0;
275
276 auto decodedBodySize = m_resourceTiming.networkLoadMetrics().responseBodyDecodedSize;
277 if (decodedBodySize == std::numeric_limits<uint64_t>::max())
278 return 0;
279
280 return decodedBodySize;
281}
