44 : m_page(page)

46}

48void PointerLockController::requestPointerLock(Element* target)
49{
50 if (!target || !target->isConnected() || m_documentOfRemovedElementWhileWaitingForUnlock) {
51 enqueueEvent(eventNames().pointerlockerrorEvent, target);
52 return;
53 }
54
55 if (m_documentAllowedToRelockWithoutUserGesture != &target->document() && !UserGestureIndicator::processingUserGesture()) {
56 enqueueEvent(eventNames().pointerlockerrorEvent, target);
57 return;
58 }
59
60 if (target->document().isSandboxed(SandboxPointerLock)) {
61 // FIXME: This message should be moved off the console once a solution to https://bugs.webkit.org/show_bug.cgi?id=103274 exists.
62 target->document().addConsoleMessage(MessageSource::Security, MessageLevel::Error, "Blocked pointer lock on an element because the element's frame is sandboxed and the 'allow-pointer-lock' permission is not set."_s);
63 enqueueEvent(eventNames().pointerlockerrorEvent, target);
64 return;
65 }
66
67 if (m_element) {
68 if (&m_element->document() != &target->document()) {
69 enqueueEvent(eventNames().pointerlockerrorEvent, target);
70 return;
71 }
72 m_element = target;
73 enqueueEvent(eventNames().pointerlockchangeEvent, target);
74 m_page.pointerCaptureController().pointerLockWasApplied();
75 } else {
76 m_lockPending = true;
77 m_element = target;
78 if (!m_page.chrome().client().requestPointerLock()) {
79 clearElement();
80 enqueueEvent(eventNames().pointerlockerrorEvent, target);
81 }
82 }
83}
84
85void PointerLockController::requestPointerUnlock()
86{
87 if (!m_element)
88 return;
89
90 m_unlockPending = true;
91 m_page.chrome().client().requestPointerUnlock();
92}
93
94void PointerLockController::requestPointerUnlockAndForceCursorVisible()
95{
96 m_documentAllowedToRelockWithoutUserGesture = nullptr;
97
98 if (!m_element)
99 return;
100
101 m_unlockPending = true;
102 m_page.chrome().client().requestPointerUnlock();
103 m_forceCursorVisibleUponUnlock = true;
104}
105
106void PointerLockController::elementWasRemoved(Element& element)
107{
108 if (m_element == &element) {
109 m_documentOfRemovedElementWhileWaitingForUnlock = makeWeakPtr(m_element->document());
110 // Set element null immediately to block any future interaction with it
111 // including mouse events received before the unlock completes.
112 requestPointerUnlock();
113 clearElement();
114 }
115}
116
117void PointerLockController::documentDetached(Document& document)
118{
119 if (m_documentAllowedToRelockWithoutUserGesture == &document)
120 m_documentAllowedToRelockWithoutUserGesture = nullptr;
121
122 if (m_element && &m_element->document() == &document) {
123 m_documentOfRemovedElementWhileWaitingForUnlock = makeWeakPtr(m_element->document());
124 requestPointerUnlock();
125 clearElement();
126 }
127}

129bool PointerLockController::isLocked() const
130{
131 return m_element && !m_lockPending;

134bool PointerLockController::lockPending() const

144void PointerLockController::didAcquirePointerLock()
145{
146 if (!m_lockPending)
147 return;
148
149 ASSERT(m_element);
150
151 enqueueEvent(eventNames().pointerlockchangeEvent, m_element.get());
152 m_lockPending = false;
153 m_forceCursorVisibleUponUnlock = false;
154 m_documentAllowedToRelockWithoutUserGesture = makeWeakPtr(m_element->document());
155}
156
157void PointerLockController::didNotAcquirePointerLock()
158{
159 enqueueEvent(eventNames().pointerlockerrorEvent, m_element.get());
160 clearElement();
161 m_unlockPending = false;
162}
163
164void PointerLockController::didLosePointerLock()
165{
166 if (!m_unlockPending)
167 m_documentAllowedToRelockWithoutUserGesture = nullptr;
168
169 enqueueEvent(eventNames().pointerlockchangeEvent, m_element ? &m_element->document() : m_documentOfRemovedElementWhileWaitingForUnlock.get());
170 clearElement();
171 m_unlockPending = false;
172 m_documentOfRemovedElementWhileWaitingForUnlock = nullptr;
173 if (m_forceCursorVisibleUponUnlock) {
174 m_forceCursorVisibleUponUnlock = false;
175 m_page.chrome().client().setCursorHiddenUntilMouseMoves(false);
176 }
177}

179void PointerLockController::dispatchLockedMouseEvent(const PlatformMouseEvent& event, const AtomString& eventType)
180{
181 if (!m_element || !m_element->document().frame())
182 return;
183
184 Ref protectedElement { *m_element };
185 protectedElement->dispatchMouseEvent(event, eventType, event.clickCount());
186
187 // Create click events
188 if (eventType == eventNames().mouseupEvent)
189 protectedElement->dispatchMouseEvent(event, eventNames().clickEvent, event.clickCount());
190}
191
192void PointerLockController::dispatchLockedWheelEvent(const PlatformWheelEvent& event)
193{
194 if (!m_element || !m_element->document().frame())
195 return;
196
197 OptionSet<EventHandling> defaultHandling;
198 m_element->dispatchWheelEvent(event, defaultHandling);
199}
200
201void PointerLockController::clearElement()
202{
203 m_lockPending = false;
204 m_element = nullptr;
205}
206
207void PointerLockController::enqueueEvent(const AtomString& type, Element* element)
208{
209 if (element)
210 enqueueEvent(type, &element->document());
211}
212
213void PointerLockController::enqueueEvent(const AtomString& type, Document* document)
214{
215 // FIXME: Spec doesn't specify which task source use.
216 if (auto protectedDocument = makeRefPtr(document))
217 protectedDocument->queueTaskToDispatchEvent(TaskSource::UserInteraction, Event::create(type, Event::CanBubble::Yes, Event::IsCancelable::No));
218}
