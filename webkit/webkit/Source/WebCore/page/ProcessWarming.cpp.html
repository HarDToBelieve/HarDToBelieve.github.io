52void ProcessWarming::initializeNames()
53{
54 AtomString::init();
55 HTMLNames::init();
56 QualifiedName::init();
57 MediaFeatureNames::init();
58 SVGNames::init();
59 XLinkNames::init();
60 MathMLNames::init();
61 XMLNSNames::init();
62 XMLNames::init();
63 WebKitFontFamilyNames::init();
64}
65
66void ProcessWarming::prewarmGlobally()
67{
68 initializeNames();
69
70 // Prewarms user agent stylesheet.
71 Style::UserAgentStyle::initDefaultStyleSheet();
72
73 // Prewarms JS VM.
74 commonVM();
75
76 // Prewarm font cache
77 FontCache::singleton().prewarmGlobally();
78
79#if ENABLE(TELEPHONE_NUMBER_DETECTION)
80 TelephoneNumberDetector::prewarm();
81#endif
82
83#if ENABLE(GPU_DRIVER_PREWARMING)
84 GPUDevice::prewarm();
85#endif
86}
87
88WebCore::PrewarmInformation ProcessWarming::collectPrewarmInformation()
89{
90 return { FontCache::singleton().collectPrewarmInformation() };
91}
92
93void ProcessWarming::prewarmWithInformation(const PrewarmInformation& prewarmInfo)
94{
95 FontCache::singleton().prewarm(prewarmInfo.fontCache);
96}
