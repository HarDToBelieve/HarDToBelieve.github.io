364bool Quirks::shouldDispatchSimulatedMouseEvents(EventTarget* target) const
365{
366 if (RuntimeEnabledFeatures::sharedFeatures().mouseEventsSimulationEnabled())
367 return true;
368
369 if (!needsQuirks())
370 return false;
371
372 auto doShouldDispatchChecks = [this] () -> ShouldDispatchSimulatedMouseEvents {
373 auto* loader = m_document->loader();
374 if (!loader || loader->simulatedMouseEventsDispatchPolicy() != SimulatedMouseEventsDispatchPolicy::Allow)
375 return ShouldDispatchSimulatedMouseEvents::No;
376
377 if (isAmazon())
378 return ShouldDispatchSimulatedMouseEvents::Yes;
379 if (isGoogleMaps())
380 return ShouldDispatchSimulatedMouseEvents::Yes;
381
382 auto& url = m_document->topDocument().url();
383 auto host = url.host().convertToASCIILowercase();
384
385 if (host == "wix.com" || host.endsWith(".wix.com")) {
386 // Disable simulated mouse dispatching for template selection.
387 return url.path().startsWithIgnoringASCIICase("/website/templates/") ? ShouldDispatchSimulatedMouseEvents::No : ShouldDispatchSimulatedMouseEvents::Yes;
388 }
389
390 if ((host == "desmos.com" || host.endsWith(".desmos.com")) && url.path().startsWithIgnoringASCIICase("/calculator/"))
391 return ShouldDispatchSimulatedMouseEvents::Yes;
392 if (host == "figma.com" || host.endsWith(".figma.com"))
393 return ShouldDispatchSimulatedMouseEvents::Yes;
394 if (host == "trello.com" || host.endsWith(".trello.com"))
395 return ShouldDispatchSimulatedMouseEvents::Yes;
396 if (host == "airtable.com" || host.endsWith(".airtable.com"))
397 return ShouldDispatchSimulatedMouseEvents::Yes;
398 if (host == "msn.com" || host.endsWith(".msn.com"))
399 return ShouldDispatchSimulatedMouseEvents::Yes;
400 if (host == "flipkart.com" || host.endsWith(".flipkart.com"))
401 return ShouldDispatchSimulatedMouseEvents::Yes;
402 if (host == "iqiyi.com" || host.endsWith(".iqiyi.com"))
403 return ShouldDispatchSimulatedMouseEvents::Yes;
404 if (host == "trailers.apple.com")
405 return ShouldDispatchSimulatedMouseEvents::Yes;
406 if (host == "soundcloud.com")
407 return ShouldDispatchSimulatedMouseEvents::Yes;
408 if (host == "naver.com")
409 return ShouldDispatchSimulatedMouseEvents::Yes;
410 if (host == "nba.com" || host.endsWith(".nba.com"))
411 return ShouldDispatchSimulatedMouseEvents::Yes;
412 if (host.endsWith(".naver.com")) {
413 // Disable the quirk for tv.naver.com subdomain to be able to simulate hover on videos.
414 if (host == "tv.naver.com")
415 return ShouldDispatchSimulatedMouseEvents::No;
416 // Disable the quirk for mail.naver.com subdomain to be able to tap on mail subjects.
417 if (host == "mail.naver.com")
418 return ShouldDispatchSimulatedMouseEvents::No;
419 // Disable the quirk on the mobile site.
420 // FIXME: Maybe this quirk should be disabled for "m." subdomains on all sites? These are generally mobile sites that don't need mouse events.
421 if (host == "m.naver.com")
422 return ShouldDispatchSimulatedMouseEvents::No;
423 return ShouldDispatchSimulatedMouseEvents::Yes;
424 }
425 if (host == "mybinder.org" || host.endsWith(".mybinder.org"))
426 return ShouldDispatchSimulatedMouseEvents::DependingOnTargetFor_mybinder_org;
427 return ShouldDispatchSimulatedMouseEvents::No;
428 };
429
430 if (m_shouldDispatchSimulatedMouseEventsQuirk == ShouldDispatchSimulatedMouseEvents::Unknown)
431 m_shouldDispatchSimulatedMouseEventsQuirk = doShouldDispatchChecks();
432
433 switch (m_shouldDispatchSimulatedMouseEventsQuirk) {
434 case ShouldDispatchSimulatedMouseEvents::Unknown:
435 ASSERT_NOT_REACHED();
436 return false;
437
438 case ShouldDispatchSimulatedMouseEvents::No:
439 return false;
440
441 case ShouldDispatchSimulatedMouseEvents::DependingOnTargetFor_mybinder_org:
442 if (is<Node>(target)) {
443 for (auto* node = downcast<Node>(target); node; node = node->parentNode()) {
444 if (is<Element>(node) && downcast<Element>(*node).classList().contains("lm-DockPanel-tabBar"))
445 return true;
446 }
447 }
448 return false;
449
450 case ShouldDispatchSimulatedMouseEvents::Yes:
451 return true;
452 }
453
454 ASSERT_NOT_REACHED();
455 return false;
456}

458bool Quirks::shouldDispatchedSimulatedMouseEventsAssumeDefaultPrevented(EventTarget* target) const
459{
460 if (!needsQuirks() || !shouldDispatchSimulatedMouseEvents(target))
461 return false;
462
463 if (isAmazon() && is<Element>(target)) {
464 // When panning on an Amazon product image, we're either touching on the #magnifierLens element
465 // or its previous sibling.
466 auto& element = downcast<Element>(*target);
467 if (element.getIdAttribute() == "magnifierLens")
468 return true;
469 if (auto* sibling = element.nextElementSibling())
470 return sibling->getIdAttribute() == "magnifierLens";
471 }
472
473 if (equalLettersIgnoringASCIICase(m_document->topDocument().url().host(), "soundcloud.com") && is<Element>(target))
474 return downcast<Element>(*target).classList().contains("sceneLayer");
475
476 return false;
477}

479std::optional<Event::IsCancelable> Quirks::simulatedMouseEventTypeForTarget(EventTarget* target) const
480{
481 if (!shouldDispatchSimulatedMouseEvents(target))
482 return { };
483
484 // On Google Maps, we want to limit simulated mouse events to dragging the little man that allows entering into Street View.
485 if (isGoogleMaps()) {
486 if (is<Element>(target) && downcast<Element>(target)->getAttribute("class") == "widget-expand-button-pegman-icon")
487 return Event::IsCancelable::Yes;
488 return { };
489 }
490
491 auto host = m_document->topDocument().url().host();
492 if (equalLettersIgnoringASCIICase(host, "desmos.com") || host.endsWithIgnoringASCIICase(".desmos.com"))
493 return Event::IsCancelable::No;
494
495 if (equalLettersIgnoringASCIICase(host, "airtable.com") || host.endsWithIgnoringASCIICase(".airtable.com")) {
496 // We want to limit simulated mouse events to elements under <div id="paneContainer"> to allow for column re-ordering and multiple cell selection.
497 if (is<Node>(target)) {
498 auto* node = downcast<Node>(target);
499 if (auto* paneContainer = node->treeScope().getElementById(AtomString("paneContainer"))) {
500 if (paneContainer->contains(node))
501 return Event::IsCancelable::Yes;
502 }
503 }
504 return { };
505 }
506
507 return Event::IsCancelable::Yes;
508}

619// FIXME(<rdar://problem/50394969>): Remove after desmos.com adopts inputmode="none".
620bool Quirks::needsInputModeNoneImplicitly(const HTMLElement& element) const
621{
622#if PLATFORM(IOS_FAMILY)
623 if (!needsQuirks())
624 return false;
625
626 if (element.hasTagName(HTMLNames::inputTag)) {
627 if (!equalLettersIgnoringASCIICase(m_document->url().host(), "calendar.google.com"))
628 return false;
629 static NeverDestroyed<QualifiedName> dataInitialValueAttr(nullAtom(), "data-initial-value", nullAtom());
630 static NeverDestroyed<QualifiedName> dataPreviousValueAttr(nullAtom(), "data-previous-value", nullAtom());
631
632 return equalLettersIgnoringASCIICase(element.attributeWithoutSynchronization(HTMLNames::autocompleteAttr), "off")
633 && element.hasAttributeWithoutSynchronization(dataInitialValueAttr)
634 && element.hasAttributeWithoutSynchronization(dataPreviousValueAttr);
635 }
636
637 if (!element.hasTagName(HTMLNames::textareaTag))
638 return false;
639
640 auto& url = m_document->url();
641 auto host = url.host();
642 if (!host.endsWithIgnoringASCIICase(".desmos.com"))
643 return false;
644
645 return element.parentElement() && element.parentElement()->classNames().contains("dcg-mq-textarea");
646#else
647 UNUSED_PARAM(element);
648 return false;
649#endif
650}

833bool Quirks::shouldBypassBackForwardCache() const
834{
835 if (!needsQuirks())
836 return false;
837
838 auto topURL = m_document->topDocument().url();
839 auto host = topURL.host();
840
841 // Vimeo.com used to bypass the back/forward cache by serving "Cache-Control: no-store" over HTTPS.
842 // We started caching such content in r250437 but the vimeo.com content unfortunately is not currently compatible
843 // because it changes the opacity of its body to 0 when navigating away and fails to restore the original opacity
844 // when coming back from the back/forward cache (e.g. in 'pageshow' event handler). See <rdar://problem/56996057>.
845 if (topURL.protocolIs("https") && equalLettersIgnoringASCIICase(host, "vimeo.com")) {
846 if (auto* documentLoader = m_document->frame() ? m_document->frame()->loader().documentLoader() : nullptr)
847 return documentLoader->response().cacheControlContainsNoStore();
848 }
849
850 // Google Docs used to bypass the back/forward cache by serving "Cache-Control: no-store" over HTTPS.
851 // We started caching such content in r250437 but the Google Docs index page unfortunately is not currently compatible
852 // because it puts an overlay (with class "docs-homescreen-freeze-el-full") over the page when navigating away and fails
853 // to remove it when coming back from the back/forward cache (e.g. in 'pageshow' event handler). See <rdar://problem/57670064>.
854 // Note that this does not check for docs.google.com host because of hosted G Suite apps.
855 static MainThreadNeverDestroyed<const AtomString> googleDocsOverlayDivClass("docs-homescreen-freeze-el-full", AtomString::ConstructFromLiteral);
856 auto* firstChildInBody = m_document->body() ? m_document->body()->firstChild() : nullptr;
857 if (is<HTMLDivElement>(firstChildInBody)) {
858 auto& div = downcast<HTMLDivElement>(*firstChildInBody);
859 if (div.hasClass() && div.classNames().contains(googleDocsOverlayDivClass))
860 return true;
861 }
862
863 return false;
864}

879bool Quirks::shouldMakeEventListenerPassive(const EventTarget& eventTarget, const AtomString& eventType, const EventListener& eventListener)
880{
881 auto eventTargetIsRoot = [](const EventTarget& eventTarget) {
882 if (is<DOMWindow>(eventTarget))
883 return true;
884
885 if (is<Node>(eventTarget)) {
886 auto& node = downcast<Node>(eventTarget);
887 return is<Document>(node) || node.document().documentElement() == &node || node.document().body() == &node;
888 }
889 return false;
890 };
891
892 auto documentFromEventTarget = [](const EventTarget& eventTarget) -> Document* {
893 return downcast<Document>(eventTarget.scriptExecutionContext());
894 };
895
896 if (eventNames().isTouchScrollBlockingEventType(eventType)) {
897 if (eventTargetIsRoot(eventTarget)) {
898 if (auto* document = documentFromEventTarget(eventTarget))
899 return document->settings().passiveTouchListenersAsDefaultOnDocument();
900 }
901 return false;
902 }
903
904 if (eventNames().isWheelEventType(eventType)) {
905 if (eventTargetIsRoot(eventTarget)) {
906 if (auto* document = documentFromEventTarget(eventTarget))
907 return document->settings().passiveWheelListenersAsDefaultOnDocument();
908 }
909 return false;
910 }
911
912 if (eventType == eventNames().mousewheelEvent) {
913 if (!is<JSEventListener>(eventListener))
914 return false;
915
916 // For SmoothScroll.js
917 // Matches Blink intervention in https://chromium.googlesource.com/chromium/src/+/b6b13c9cfe64d52a4168d9d8d1ad9bb8f0b46a2a%5E%21/
918 if (is<DOMWindow>(eventTarget)) {
919 auto* document = downcast<DOMWindow>(eventTarget).document();
920 if (!document || !document->quirks().needsQuirks())
921 return false;
922
923 auto& jsEventListener = downcast<JSEventListener>(eventListener);
924 if (jsEventListener.functionName() == "ssc_wheel")
925 return true;
926 }
927
928 return false;
929 }
930
931 return false;
932}
