38RemoteDOMWindow::RemoteDOMWindow(Ref<RemoteFrame>&& frame, GlobalWindowIdentifier&& identifier)
39 : AbstractDOMWindow(WTFMove(identifier))
40 , m_frame(WTFMove(frame))
41{
42 m_frame->setWindow(this);
43}
44
45RemoteDOMWindow::~RemoteDOMWindow()
46{
47 if (m_frame)

49}
