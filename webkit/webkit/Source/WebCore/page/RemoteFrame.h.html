26#pragma once
27
28#include "AbstractFrame.h"
29#include "GlobalFrameIdentifier.h"
30#include <wtf/Ref.h>
31#include <wtf/RefPtr.h>
32#include <wtf/TypeCasts.h>
33
34namespace WebCore {
35
36class RemoteDOMWindow;
37
38class RemoteFrame final : public AbstractFrame {
39public:
40 static Ref<RemoteFrame> create(GlobalFrameIdentifier&& frameIdentifier)
41 {
42 return adoptRef(* new RemoteFrame(WTFMove(frameIdentifier)));
43 }
44 ~RemoteFrame();
45
46 const GlobalFrameIdentifier& identifier() const { return m_identifier; }
47
48 void setWindow(RemoteDOMWindow* window) { m_window = window; }
49 RemoteDOMWindow* window() const { return m_window; }
50
51 void setOpener(AbstractFrame* opener) { m_opener = opener; }
52 AbstractFrame* opener() const { return m_opener.get(); }
53
54private:
55 WEBCORE_EXPORT explicit RemoteFrame(GlobalFrameIdentifier&&);
56
57 bool isRemoteFrame() const final { return true; }
58 bool isLocalFrame() const final { return false; }
59
60 AbstractDOMWindow* virtualWindow() const final;
61
62 GlobalFrameIdentifier m_identifier;
63 RemoteDOMWindow* m_window { nullptr };
64
65 RefPtr<AbstractFrame> m_opener;
66};
67
68} // namespace WebCore
69
70SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::RemoteFrame)
71 static bool isType(const WebCore::AbstractFrame& frame) { return frame.isRemoteFrame(); }
72SPECIALIZE_TYPE_TRAITS_END()
