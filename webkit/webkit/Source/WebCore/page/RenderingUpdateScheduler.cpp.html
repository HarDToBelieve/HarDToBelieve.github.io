39RenderingUpdateScheduler::RenderingUpdateScheduler(Page& page)
40 : m_page(page)
41{
42 windowScreenDidChange(page.chrome().displayID());
43}
44
45bool RenderingUpdateScheduler::scheduleAnimation()
46{
47 if (m_useTimer)
48 return false;
49
50 return DisplayRefreshMonitorManager::sharedManager().scheduleAnimation(*this);
51}
52
53void RenderingUpdateScheduler::adjustRenderingUpdateFrequency()
54{
55 auto renderingUpdateFramesPerSecond = m_page.preferredRenderingUpdateFramesPerSecond();
56 if (renderingUpdateFramesPerSecond) {
57 setPreferredFramesPerSecond(renderingUpdateFramesPerSecond.value());
58 m_useTimer = false;
59 } else
60 m_useTimer = true;
61
62 if (isScheduled()) {
63 clearScheduled();
64 scheduleRenderingUpdate();
65 }
66}
67
68void RenderingUpdateScheduler::scheduleRenderingUpdate()
69{
70 LOG_WITH_STREAM(EventLoop, stream << "RenderingUpdateScheduler for page " << &m_page << " scheduleTimedRenderingUpdate() - already scheduled " << isScheduled() << " page visible " << m_page.isVisible());
71
72 if (isScheduled())
73 return;
74
75 // Optimize the case when an invisible page wants just to schedule layer flush.
76 if (!m_page.isVisible()) {
77 triggerRenderingUpdate();
78 return;
79 }
80
81 tracePoint(ScheduleRenderingUpdate);
82
83 if (!scheduleAnimation()) {
84 LOG_WITH_STREAM(DisplayLink, stream << "RenderingUpdateScheduler::scheduleRenderingUpdate for interval " << m_page.preferredRenderingUpdateInterval() << " falling back to timer");
85 startTimer(m_page.preferredRenderingUpdateInterval());
86 }
87}
88
89bool RenderingUpdateScheduler::isScheduled() const
90{
91 ASSERT_IMPLIES(m_refreshTimer.get(), m_scheduled);
92 return m_scheduled;
93}
94
95void RenderingUpdateScheduler::startTimer(Seconds delay)
96{
97 LOG_WITH_STREAM(EventLoop, stream << "RenderingUpdateScheduler for page " << &m_page << " startTimer(" << delay << ")");
98
99 ASSERT(!isScheduled());
100 m_refreshTimer = makeUnique<Timer>(*this, &RenderingUpdateScheduler::displayRefreshFired);
101 m_refreshTimer->startOneShot(delay);
102 m_scheduled = true;
103}
104
105void RenderingUpdateScheduler::clearScheduled()
106{
107 m_scheduled = false;
108 m_refreshTimer = nullptr;
109}
110
111DisplayRefreshMonitorFactory* RenderingUpdateScheduler::displayRefreshMonitorFactory() const
112{
113 return m_page.chrome().client().displayRefreshMonitorFactory();
114}
115
116void RenderingUpdateScheduler::windowScreenDidChange(PlatformDisplayID displayID)
117{
118 adjustRenderingUpdateFrequency();
119 DisplayRefreshMonitorManager::sharedManager().windowScreenDidChange(displayID, *this);
120}
121
122void RenderingUpdateScheduler::displayRefreshFired()
123{
124 LOG_WITH_STREAM(EventLoop, stream << "RenderingUpdateScheduler for page " << &m_page << " displayRefreshFired()");
125
126 tracePoint(TriggerRenderingUpdate);
127
128 clearScheduled();
129 triggerRenderingUpdate();
130}
131
132void RenderingUpdateScheduler::triggerRenderingUpdateForTesting()
133{
134 triggerRenderingUpdate();
135}
136
137void RenderingUpdateScheduler::triggerRenderingUpdate()
138{
139 m_page.chrome().client().triggerRenderingUpdate();
140}
