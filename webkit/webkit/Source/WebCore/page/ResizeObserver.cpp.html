37Ref<ResizeObserver> ResizeObserver::create(Document& document, Ref<ResizeObserverCallback>&& callback)
38{
39 return adoptRef(*new ResizeObserver(document, WTFMove(callback)));
40}

42ResizeObserver::ResizeObserver(Document& document, Ref<ResizeObserverCallback>&& callback)
43 : ActiveDOMObject(callback->scriptExecutionContext())
44 , m_document(makeWeakPtr(document))
45 , m_callback(WTFMove(callback))
46{
47 suspendIfNeeded();
48}
49
50ResizeObserver::~ResizeObserver()
51{
52 disconnect();
53 if (m_document)
54 m_document->removeResizeObserver(*this);
55}

57void ResizeObserver::observe(Element& target)
58{
59 if (!m_callback)
60 return;
61
62 auto position = m_observations.findMatching([&](auto& observation) {
63 return observation->target() == &target;
64 });
65
66 if (position != notFound)
67 return;
68
69 auto& observerData = target.ensureResizeObserverData();
70 observerData.observers.append(makeWeakPtr(this));
71
72 m_observations.append(ResizeObservation::create(&target));
73 m_pendingTargets.append(target);
74
75 if (m_document) {
76 m_document->addResizeObserver(*this);
77 m_document->scheduleRenderingUpdate(RenderingUpdateStep::ResizeObservations);
78 }
79}
80
81void ResizeObserver::unobserve(Element& target)
82{
83 if (!removeTarget(target))
84 return;
85
86 removeObservation(target);
87}
88
89void ResizeObserver::disconnect()
90{
91 removeAllTargets();
92}

94void ResizeObserver::targetDestroyed(Element& target)
95{
96 removeObservation(target);

99size_t ResizeObserver::gatherObservations(size_t deeperThan)
100{
101 m_hasSkippedObservations = false;
102 size_t minObservedDepth = maxElementDepth();
103 for (const auto& observation : m_observations) {
104 LayoutSize currentSize;
105 if (observation->elementSizeChanged(currentSize)) {
106 size_t depth = observation->targetElementDepth();
107 if (depth > deeperThan) {
108 observation->updateObservationSize(currentSize);
109 m_activeObservations.append(observation.get());
110 minObservedDepth = std::min(depth, minObservedDepth);
111 } else
112 m_hasSkippedObservations = true;
113 }
114 }
115 return minObservedDepth;
116}
117
118void ResizeObserver::deliverObservations()
119{
120 Vector<Ref<ResizeObserverEntry>> entries;
121 for (const auto& observation : m_activeObservations) {
122 ASSERT(observation->target());
123 entries.append(ResizeObserverEntry::create(observation->target(), observation->computeContentRect()));
124 }
125 m_activeObservations.clear();
126
127 auto* context = m_callback->scriptExecutionContext();
128 if (!context)
129 return;
130
131 InspectorInstrumentation::willFireObserverCallback(*context, "ResizeObserver"_s);
132 m_callback->handleEvent(*this, entries, *this);
133 InspectorInstrumentation::didFireObserverCallback(*context);
134}

136bool ResizeObserver::removeTarget(Element& target)
137{
138 auto* observerData = target.resizeObserverData();
139 if (!observerData)
140 return false;
141
142 auto& observers = observerData->observers;
143 return observers.removeFirst(this);
144}
145
146void ResizeObserver::removeAllTargets()
147{
148 for (auto& observation : m_observations) {
149 bool removed = removeTarget(*observation->target());
150 ASSERT_UNUSED(removed, removed);
151 }
152 m_pendingTargets.clear();
153 m_activeObservations.clear();
154 m_observations.clear();
155}
156
157bool ResizeObserver::removeObservation(const Element& target)
158{
159 m_pendingTargets.removeFirstMatching([&target](auto& pendingTarget) {
160 return pendingTarget.ptr() == &target;
161 });
162
163 m_activeObservations.removeFirstMatching([&target](auto& observation) {
164 return observation->target() == &target;
165 });
166
167 return m_observations.removeFirstMatching([&target](auto& observation) {
168 return observation->target() == &target;
169 });
170}
171
172bool ResizeObserver::virtualHasPendingActivity() const
173{
174 return (hasObservations() && m_document) || !m_activeObservations.isEmpty();
175}
176
177const char* ResizeObserver::activeDOMObjectName() const
178{
179 return "ResizeObserver";
180}
181
182void ResizeObserver::stop()
183{
184 disconnect();
185 m_callback = nullptr;
186}
