26#pragma once
27
28#if ENABLE(RESIZE_OBSERVER)
29
30#include "DOMRectReadOnly.h"
31#include "Element.h"
32#include "FloatRect.h"
33#include <wtf/RefCounted.h>
34
35namespace WebCore {
36
37class Element;
38
39class ResizeObserverEntry : public RefCounted<ResizeObserverEntry> {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 static Ref<ResizeObserverEntry> create(Element* target, const FloatRect& contentRect)
43 {
44 return adoptRef(*new ResizeObserverEntry(target, contentRect));
45 }
46
47 Element* target() const { return m_target.get(); }
48 DOMRectReadOnly* contentRect() const { return m_contentRect.get(); }
49
50private:
51 ResizeObserverEntry(Element* target, const FloatRect& contentRect)
52 : m_target(target)
53 , m_contentRect(DOMRectReadOnly::create(contentRect.x(), contentRect.y(), contentRect.width(), contentRect.height()))
54 {
55 }
56
57 RefPtr<Element> m_target;
58 RefPtr<DOMRectReadOnly> m_contentRect;
59};
60
61} // namespace WebCore
62#endif // ENABLE(RESIZE_OBSERVER)
