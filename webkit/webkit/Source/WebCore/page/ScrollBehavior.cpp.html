36bool useSmoothScrolling(ScrollBehavior behavior, Element* associatedElement, SmoothScrollFeatureEnablement overrideFeatureEnablement)
37{
38 if (!associatedElement)
39 return false;
40
41 // FIXME: Should we use document()->scrollingElement()?
42 // See https://bugs.webkit.org/show_bug.cgi?id=205059
43 if (associatedElement == associatedElement->document().scrollingElement())
44 associatedElement = associatedElement->document().documentElement();
45
46 if (!associatedElement->renderer() || (overrideFeatureEnablement == SmoothScrollFeatureEnablement::Default && !associatedElement->document().settings().CSSOMViewSmoothScrollingEnabled()))
47 return false;
48
49 // https://drafts.csswg.org/cssom-view/#scrolling
50 switch (behavior) {
51 case ScrollBehavior::Auto:
52 return associatedElement->renderer()->style().useSmoothScrolling();
53 case ScrollBehavior::Instant:
54 return false;
55 case ScrollBehavior::Smooth:
56 return true;
57 }
58 ASSERT_NOT_REACHED();
59 return false;
60}
