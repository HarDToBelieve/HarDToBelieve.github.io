29#pragma once
30
31#include "ScrollOptions.h"
32#include <cmath>
33
34namespace WebCore {
35
36struct ScrollToOptions : ScrollOptions {
37 std::optional<double> left;
38 std::optional<double> top;
39
40 ScrollToOptions() = default;
41 ScrollToOptions(double x, double y)
42 : left(x)
43 , top(y)
44 { }
45};
46
47inline double normalizeNonFiniteValueOrFallBackTo(std::optional<double> value, double fallbackValue)
48{
49 // Normalize non-finite values (https://drafts.csswg.org/cssom-view/#normalize-non-finite-values).
50 return value ? (std::isfinite(*value) ? *value : 0) : fallbackValue;
51}
52
53// FIXME(https://webkit.org/b/88339): Consider using FloatPoint or DoublePoint for fallback and return values.
54inline ScrollToOptions normalizeNonFiniteCoordinatesOrFallBackTo(const ScrollToOptions& value, double x, double y)
55{
56 ScrollToOptions options = value;
57 options.left = normalizeNonFiniteValueOrFallBackTo(value.left, x);
58 options.top = normalizeNonFiniteValueOrFallBackTo(value.top, y);
59 return options;
60}
