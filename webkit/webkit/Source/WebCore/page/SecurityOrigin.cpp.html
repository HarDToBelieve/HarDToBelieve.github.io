55static bool schemeRequiresHost(const URL& url)
56{
57 // We expect URLs with these schemes to have authority components. If the
58 // URL lacks an authority component, we get concerned and mark the origin
59 // as unique.
60 return url.protocolIsInHTTPFamily() || url.protocolIs("ftp");
61}

63bool SecurityOrigin::shouldIgnoreHost(const URL& url)
64{
65 return url.protocolIsData() || url.protocolIsAbout() || url.protocolIsJavaScript() || url.protocolIs("file");
66}

68bool SecurityOrigin::shouldUseInnerURL(const URL& url)
69{
70 // FIXME: Blob URLs don't have inner URLs. Their form is "blob:<inner-origin>/<UUID>", so treating the part after "blob:" as a URL is incorrect.
71 if (url.protocolIsBlob())
72 return true;
73 UNUSED_PARAM(url);
74 return false;

78// that all the URL schemes we currently support that use inner URLs for their

82 // FIXME: Update this callsite to use the innerURL member function when

94static bool shouldTreatAsUniqueOrigin(const URL& url)
95{
96 if (!url.isValid())
97 return true;
98
99 // FIXME: Do we need to unwrap the URL further?
100 URL innerURL = SecurityOrigin::shouldUseInnerURL(url) ? SecurityOrigin::extractInnerURL(url) : url;
101
102 // FIXME: Check whether innerURL is valid.
103
104 // For edge case URLs that were probably misparsed, make sure that the origin is unique.
105 // This is an additional safety net against bugs in URL parsing, and for network back-ends that parse URLs differently,
106 // and could misinterpret another component for hostname.
107 if (schemeRequiresHost(innerURL) && innerURL.host().isEmpty())
108 return true;
109
110 if (LegacySchemeRegistry::shouldTreatURLSchemeAsNoAccess(innerURL.protocol().toStringWithoutCopying()))
111 return true;
112
113 // https://url.spec.whatwg.org/#origin with some additions
114 if (url.hasSpecialScheme()
115#if PLATFORM(COCOA)
116 || !linkedOnOrAfter(SDKVersion::FirstWithNullOriginForNonSpecialSchemedURLs)
117 || url.protocolIs("applewebdata")
118 || url.protocolIs("x-apple-ql-id")
119 || url.protocolIs("x-apple-ql-id2")
120 || url.protocolIs("x-apple-ql-magic")
121#endif
122#if PLATFORM(GTK) || PLATFORM(WPE)
123 || url.protocolIs("resource")
124#endif
125 || url.protocolIs("blob"))
126 return false;
127
128 return !LegacySchemeRegistry::schemeIsHandledBySchemeHandler(url.protocol());
129}

131static bool isLoopbackIPAddress(StringView host)
132{
133 // The IPv6 loopback address is 0:0:0:0:0:0:0:1, which compresses to ::1.
134 if (host == "[::1]")
135 return true;
136
137 // Check to see if it's a valid IPv4 address that has the form 127.*.*.*.
138 if (!host.startsWith("127."))
139 return false;
140 size_t dotsFound = 0;
141 for (size_t i = 0; i < host.length(); ++i) {
142 if (host[i] == '.') {
143 dotsFound++;
144 continue;
145 }
146 if (!isASCIIDigit(host[i]))
147 return false;
148 }
149 return dotsFound == 3;
150}

152// https://w3c.github.io/webappsec-secure-contexts/#is-origin-trustworthy (Editor's Draft, 17 November 2016)
153static bool shouldTreatAsPotentiallyTrustworthy(const String& protocol, StringView host)
154{
155 if (LegacySchemeRegistry::shouldTreatURLSchemeAsSecure(protocol))
156 return true;
157
158 if (SecurityOrigin::isLocalHostOrLoopbackIPAddress(host))
159 return true;
160
161 if (LegacySchemeRegistry::shouldTreatURLSchemeAsLocal(protocol))
162 return true;
163
164 if (LegacySchemeRegistry::schemeIsHandledBySchemeHandler(protocol))
165 return true;
166
167 return false;
168}

170bool shouldTreatAsPotentiallyTrustworthy(const URL& url)
171{
172 return shouldTreatAsPotentiallyTrustworthy(url.protocol().toStringWithoutCopying(), url.host());
173}

175SecurityOrigin::SecurityOrigin(const URL& url)
176 : m_data(SecurityOriginData::fromURL(url))
177 , m_isLocal(LegacySchemeRegistry::shouldTreatURLSchemeAsLocal(m_data.protocol))
178{
179 // document.domain starts as m_data.host, but can be set by the DOM.
180 m_domain = m_data.host;
181
182 if (m_data.port && WTF::isDefaultPortForProtocol(m_data.port.value(), m_data.protocol))
183 m_data.port = std::nullopt;
184
185 // By default, only local SecurityOrigins can load local resources.
186 m_canLoadLocalResources = isLocal();
187
188 if (m_canLoadLocalResources)
189 m_filePath = url.fileSystemPath(); // In case enforceFilePathSeparation() is called.
190}

216Ref<SecurityOrigin> SecurityOrigin::create(const URL& url)
217{
218 if (RefPtr<SecurityOrigin> cachedOrigin = getCachedOrigin(url))
219 return cachedOrigin.releaseNonNull();
220
221 if (shouldTreatAsUniqueOrigin(url))
222 return adoptRef(*new SecurityOrigin);
223
224 if (shouldUseInnerURL(url))
225 return adoptRef(*new SecurityOrigin(extractInnerURL(url)));
226
227 return adoptRef(*new SecurityOrigin(url));
228}

256bool SecurityOrigin::isSecure(const URL& url)
257{
258 // Invalid URLs are secure, as are URLs which have a secure protocol.
259 if (!url.isValid() || LegacySchemeRegistry::shouldTreatURLSchemeAsSecure(url.protocol().toStringWithoutCopying()))
260 return true;
261
262 // URLs that wrap inner URLs are secure if those inner URLs are secure.
263 if (shouldUseInnerURL(url))
264 return LegacySchemeRegistry::shouldTreatURLSchemeAsSecure(extractInnerURL(url).protocol().toStringWithoutCopying()) || BlobURL::isSecureBlobURL(url);
265
266 return false;
267}

269bool SecurityOrigin::isSameOriginDomain(const SecurityOrigin& other) const
270{
271 if (m_universalAccess)
272 return true;
273
274 if (this == &other)
275 return true;
276
277 if (isUnique() || other.isUnique())
278 return false;
279
280 // Here are two cases where we should permit access:
281 //
282 // 1) Neither document has set document.domain. In this case, we insist
283 // that the scheme, host, and port of the URLs match.
284 //
285 // 2) Both documents have set document.domain. In this case, we insist
286 // that the documents have set document.domain to the same value and
287 // that the scheme of the URLs match.
288 //
289 // This matches the behavior of Firefox 2 and Internet Explorer 6.
290 //
291 // Internet Explorer 7 and Opera 9 are more strict in that they require
292 // the port numbers to match when both pages have document.domain set.
293 //
294 // FIXME: Evaluate whether we can tighten this policy to require matched
295 // port numbers.
296 //
297 // Opera 9 allows access when only one page has set document.domain, but
298 // this is a security vulnerability.
299
300 bool canAccess = false;
301 if (m_data.protocol == other.m_data.protocol) {
302 if (!m_domainWasSetInDOM && !other.m_domainWasSetInDOM) {
303 if (m_data.host == other.m_data.host && m_data.port == other.m_data.port)
304 canAccess = true;
305 } else if (m_domainWasSetInDOM && other.m_domainWasSetInDOM) {
306 if (m_domain == other.m_domain)
307 canAccess = true;
308 }
309 }
310
311 if (canAccess && isLocal())
312 canAccess = passesFileCheck(other);
313
314 return canAccess;
315}

324bool SecurityOrigin::canRequest(const URL& url) const
325{
326 if (m_universalAccess)
327 return true;
328
329 if (getCachedOrigin(url) == this)
330 return true;
331
332 if (isUnique())
333 return false;
334
335 Ref<SecurityOrigin> targetOrigin(SecurityOrigin::create(url));
336
337 if (targetOrigin->isUnique())
338 return false;
339
340 // We call isSameSchemeHostPort here instead of canAccess because we want
341 // to ignore document.domain effects.
342 if (isSameSchemeHostPort(targetOrigin.get()))
343 return true;
344
345 if (SecurityPolicy::isAccessAllowed(*this, targetOrigin.get(), url))
346 return true;
347
348 return false;
349}

359// This is a hack to allow keep navigation to http/https feeds working. To remove this
360// we need to introduce new API akin to registerURLSchemeAsLocal, that registers a
361// protocols navigation policy.
362// feed(|s|search): is considered a 'nesting' scheme by embedders that support it, so it can be
363// local or remote depending on what is nested. Currently we just check if we are nesting
364// http or https, otherwise we ignore the nesting for the purpose of a security check. We need
365// a facility for registering nesting schemes, and some generalized logic for them.
366// This function should be removed as an outcome of https://bugs.webkit.org/show_bug.cgi?id=69196
367static bool isFeedWithNestedProtocolInHTTPFamily(const URL& url)
368{
369 const String& string = url.string();
370 if (!startsWithLettersIgnoringASCIICase(string, "feed"))
371 return false;
372 return startsWithLettersIgnoringASCIICase(string, "feed://")
373 || startsWithLettersIgnoringASCIICase(string, "feed:http:")
374 || startsWithLettersIgnoringASCIICase(string, "feed:https:")
375 || startsWithLettersIgnoringASCIICase(string, "feeds:http:")
376 || startsWithLettersIgnoringASCIICase(string, "feeds:https:")
377 || startsWithLettersIgnoringASCIICase(string, "feedsearch:http:")
378 || startsWithLettersIgnoringASCIICase(string, "feedsearch:https:");
379}

381bool SecurityOrigin::canDisplay(const URL& url) const
382{
383 ASSERT(!isInNetworkProcess());
384 if (m_universalAccess)
385 return true;
386
387 if (url.pathEnd() > maximumURLSize)
388 return false;
389
390#if !PLATFORM(IOS_FAMILY) && !ENABLE(BUBBLEWRAP_SANDBOX)
391 if (m_data.protocol == "file" && url.isLocalFile() && !FileSystem::filesHaveSameVolume(m_filePath, url.fileSystemPath()))
392 return false;
393#endif
394
395 if (isFeedWithNestedProtocolInHTTPFamily(url))
396 return true;
397
398 String protocol = url.protocol().toString();
399
400 if (LegacySchemeRegistry::canDisplayOnlyIfCanRequest(protocol))
401 return canRequest(url);
402
403 if (LegacySchemeRegistry::shouldTreatURLSchemeAsDisplayIsolated(protocol))
404 return equalIgnoringASCIICase(m_data.protocol, protocol) || SecurityPolicy::isAccessAllowed(*this, url);
405
406 if (!SecurityPolicy::restrictAccessToLocal())
407 return true;
408
409 if (url.isLocalFile() && url.fileSystemPath() == m_filePath)
410 return true;
411
412 if (LegacySchemeRegistry::shouldTreatURLSchemeAsLocal(protocol))
413 return canLoadLocalResources() || SecurityPolicy::isAccessAllowed(*this, url);
414
415 return true;
416}

418bool SecurityOrigin::canAccessStorage(const SecurityOrigin* topOrigin, ShouldAllowFromThirdParty shouldAllowFromThirdParty) const
419{
420 if (isUnique())
421 return false;
422
423 if (isLocal() && !needsStorageAccessFromFileURLsQuirk() && !m_universalAccess && shouldAllowFromThirdParty != AlwaysAllowFromThirdParty)
424 return false;
425
426 if (m_storageBlockingPolicy == StorageBlockingPolicy::BlockAll)
427 return false;
428
429 // FIXME: This check should be replaced with an ASSERT once we can guarantee that topOrigin is not null.
430 if (!topOrigin)
431 return true;
432
433 if (topOrigin->m_storageBlockingPolicy == StorageBlockingPolicy::BlockAll)
434 return false;
435
436 if (shouldAllowFromThirdParty == AlwaysAllowFromThirdParty)
437 return true;
438
439 if (m_universalAccess)
440 return true;
441
442 if ((m_storageBlockingPolicy == StorageBlockingPolicy::BlockThirdParty || topOrigin->m_storageBlockingPolicy == StorageBlockingPolicy::BlockThirdParty) && !topOrigin->isSameOriginAs(*this))
443 return false;
444
445 return true;
446}

468bool SecurityOrigin::isMatchingRegistrableDomainSuffix(const String& domainSuffix, bool treatIPAddressAsDomain) const
469{
470 if (domainSuffix.isEmpty())
471 return false;
472
473 auto ipAddressSetting = treatIPAddressAsDomain ? OriginAccessEntry::TreatIPAddressAsDomain : OriginAccessEntry::TreatIPAddressAsIPAddress;
474 OriginAccessEntry accessEntry { protocol(), domainSuffix, OriginAccessEntry::AllowSubdomains, ipAddressSetting };
475 if (!accessEntry.matchesOrigin(*this))
476 return false;
477
478 // Always return true if it is an exact match.
479 if (domainSuffix.length() == host().length())
480 return true;
481
482#if ENABLE(PUBLIC_SUFFIX_LIST)
483 return !isPublicSuffix(domainSuffix);
484#else
485 return true;
486#endif
487}

549static inline bool areOriginsMatching(const SecurityOrigin& origin1, const SecurityOrigin& origin2)
550{
551 ASSERT(&origin1 != &origin2);
552
553 if (origin1.isUnique() || origin2.isUnique())
554 return origin1.isUnique() == origin2.isUnique();
555
556 if (origin1.protocol() != origin2.protocol())
557 return false;
558
559 if (origin1.protocol() == "file")
560 return origin1.enforcesFilePathSeparation() == origin2.enforcesFilePathSeparation();
561
562 if (origin1.host() != origin2.host())
563 return false;
564
565 return origin1.port() == origin2.port();
566}

568// This function mimics the result of string comparison of serialized origins.
569bool serializedOriginsMatch(const SecurityOrigin& origin1, const SecurityOrigin& origin2)
570{
571 if (&origin1 == &origin2)
572 return true;
573
574 ASSERT(!areOriginsMatching(origin1, origin2) || (origin1.toString() == origin2.toString()));
575 return areOriginsMatching(origin1, origin2);
576}
577
578bool serializedOriginsMatch(const SecurityOrigin* origin1, const SecurityOrigin* origin2)
579{
580 if (!origin1 || !origin2)
581 return origin1 == origin2;
582
583 return serializedOriginsMatch(*origin1, *origin2);
584}

600bool SecurityOrigin::equal(const SecurityOrigin* other) const
601{
602 if (other == this)
603 return true;
604
605 if (!isSameSchemeHostPort(*other))
606 return false;
607
608 if (m_domainWasSetInDOM != other->m_domainWasSetInDOM)
609 return false;
610
611 if (m_domainWasSetInDOM && m_domain != other->m_domain)
612 return false;
613
614 return true;
615}

617bool SecurityOrigin::isSameSchemeHostPort(const SecurityOrigin& other) const
618{
619 if (m_data != other.m_data)
620 return false;
621
622 if (isLocal() && !passesFileCheck(other))
623 return false;
624
625 return true;

628bool SecurityOrigin::isLocalHostOrLoopbackIPAddress(StringView host)
629{
630 if (isLoopbackIPAddress(host))
631 return true;
632
633 // FIXME: Ensure that localhost resolves to the loopback address.
634 if (equalLettersIgnoringASCIICase(host, "localhost") || host.endsWithIgnoringASCIICase(".localhost"))
635 return true;
636
637 return false;
638}
