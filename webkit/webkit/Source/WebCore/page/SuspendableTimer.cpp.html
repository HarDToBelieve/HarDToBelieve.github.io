32namespace WebCore {
33
34SuspendableTimerBase::SuspendableTimerBase(ScriptExecutionContext* context)
35 : ActiveDOMObject(context)
36{
37}
38
39SuspendableTimerBase::~SuspendableTimerBase() = default;
40
41bool SuspendableTimerBase::virtualHasPendingActivity() const
42{
43 return isActive();
44}
45
46void SuspendableTimerBase::stop()
47{
48 if (!m_suspended)
49 TimerBase::stop();
50
51 // Make it less likely that SuspendableTimer is accidentally revived and fires after being stopped.
52 // It is not allowed for an ActiveDOMObject to become active again after stop().
53 m_suspended = true;
54 m_savedIsActive = false;
55
56 didStop();
57}
58
59void SuspendableTimerBase::suspend(ReasonForSuspension)
60{
61 ASSERT(!m_suspended);
62 m_suspended = true;
63
64 m_savedIsActive = TimerBase::isActive();
65 if (m_savedIsActive) {
66 m_savedNextFireInterval = TimerBase::nextUnalignedFireInterval();
67 m_savedRepeatInterval = TimerBase::repeatInterval();
68 TimerBase::stop();
69 }
70}
71
72void SuspendableTimerBase::resume()
73{
74 ASSERT(m_suspended);
75 m_suspended = false;
76
77 if (m_savedIsActive)
78 start(m_savedNextFireInterval, m_savedRepeatInterval);
79}
80
81void SuspendableTimerBase::didStop()
82{
83}
84
85void SuspendableTimerBase::cancel()
86{
87 if (!m_suspended)
88 TimerBase::stop();
89 else
90 m_suspended = false;
91}
92
93void SuspendableTimerBase::startRepeating(Seconds repeatInterval)
94{
95 if (!m_suspended)
96 TimerBase::startRepeating(repeatInterval);
97 else {
98 m_savedIsActive = true;
99 m_savedNextFireInterval = repeatInterval;
100 m_savedRepeatInterval = repeatInterval;
101 }
102}
103
104void SuspendableTimerBase::startOneShot(Seconds interval)
105{
106 if (!m_suspended)
107 TimerBase::startOneShot(interval);
108 else {
109 m_savedIsActive = true;
110 m_savedNextFireInterval = interval;
111 m_savedRepeatInterval = 0_s;
112 }
113}
114
115Seconds SuspendableTimerBase::repeatInterval() const
116{
117 if (!m_suspended)
118 return TimerBase::repeatInterval();
119 if (m_savedIsActive)
120 return m_savedRepeatInterval;
121 return 0_s;
122}
123
124void SuspendableTimerBase::augmentFireInterval(Seconds delta)
125{
126 if (!m_suspended)
127 TimerBase::augmentFireInterval(delta);
128 else if (m_savedIsActive) {
129 m_savedNextFireInterval += delta;
130 } else {
131 m_savedIsActive = true;
132 m_savedNextFireInterval = delta;
133 m_savedRepeatInterval = 0_s;
134 }
135}
136
137void SuspendableTimerBase::augmentRepeatInterval(Seconds delta)
138{
139 if (!m_suspended)
140 TimerBase::augmentRepeatInterval(delta);
141 else if (m_savedIsActive) {
142 m_savedNextFireInterval += delta;
143 m_savedRepeatInterval += delta;
144 } else {
145 m_savedIsActive = true;
146 m_savedNextFireInterval = delta;
147 m_savedRepeatInterval = delta;
148 }
149}
150
151const char* SuspendableTimer::activeDOMObjectName() const
152{
153 return "SuspendableTimer";
154}
