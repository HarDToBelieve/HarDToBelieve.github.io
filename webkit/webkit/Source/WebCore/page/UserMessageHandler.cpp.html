36namespace WebCore {

38UserMessageHandler::UserMessageHandler(Frame& frame, UserMessageHandlerDescriptor& descriptor)
39 : FrameDestructionObserver(&frame)
40 , m_descriptor(&descriptor)
41{
42}
43
44UserMessageHandler::~UserMessageHandler() = default;
45
46ExceptionOr<void> UserMessageHandler::postMessage(RefPtr<SerializedScriptValue>&& value, Ref<DeferredPromise>&& promise)
47{
48 // Check to see if the descriptor has been removed. This can happen if the host application has
49 // removed the named message handler at the WebKit2 API level.
50 if (!m_descriptor) {
51 promise->reject(Exception { InvalidAccessError });
52 return Exception { InvalidAccessError };
53 }
54
55 m_descriptor->didPostMessage(*this, value.get(), [promise = WTFMove(promise)](SerializedScriptValue* result, const String& errorMessage) {
56 auto* globalObject = promise->globalObject();
57 if (!globalObject)
58 return;
59
60 if (!errorMessage.isNull()) {
61 JSC::JSLockHolder lock(globalObject);
62 promise->reject<IDLAny>(JSC::createError(globalObject, errorMessage));
63 return;
64 }
65
66 ASSERT(result);
67 promise->resolve<IDLAny>(result->deserialize(*globalObject, globalObject));
68 });
69 return { };
70}
