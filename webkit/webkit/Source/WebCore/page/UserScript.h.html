26#pragma once
27
28#include <wtf/URL.h>
29#include "UserContentTypes.h"
30#include "UserScriptTypes.h"
31#include <wtf/Vector.h>
32
33namespace WebCore {
34
35class UserScript {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 ~UserScript() = default;
39 UserScript(const UserScript&) = default;
40 UserScript(UserScript&&) = default;
41 UserScript& operator=(const UserScript&) = default;
42 UserScript& operator=(UserScript&&) = default;
43
44 UserScript(String&& source, URL&& url, Vector<String>&& allowlist, Vector<String>&& blocklist, UserScriptInjectionTime injectionTime, UserContentInjectedFrames injectedFrames, WaitForNotificationBeforeInjecting waitForNotification)
45 : m_source(WTFMove(source))
46 , m_url(WTFMove(url))
47 , m_allowlist(WTFMove(allowlist))
48 , m_blocklist(WTFMove(blocklist))
49 , m_injectionTime(injectionTime)
50 , m_injectedFrames(injectedFrames)
51 , m_waitForNotificationBeforeInjecting(waitForNotification)
52 {
53 }
54
55 const String& source() const { return m_source; }
56 const URL& url() const { return m_url; }
57 const Vector<String>& allowlist() const { return m_allowlist; }
58 const Vector<String>& blocklist() const { return m_blocklist; }
59 UserScriptInjectionTime injectionTime() const { return m_injectionTime; }
60 UserContentInjectedFrames injectedFrames() const { return m_injectedFrames; }
61 WaitForNotificationBeforeInjecting waitForNotificationBeforeInjecting() const { return m_waitForNotificationBeforeInjecting; }
62
63 template<class Encoder> void encode(Encoder&) const;
64 template<class Decoder> static std::optional<UserScript> decode(Decoder&);
65
66private:
67 String m_source;
68 URL m_url;
69 Vector<String> m_allowlist;
70 Vector<String> m_blocklist;
71 UserScriptInjectionTime m_injectionTime { UserScriptInjectionTime::DocumentStart };
72 UserContentInjectedFrames m_injectedFrames { UserContentInjectedFrames::InjectInAllFrames };
73 WaitForNotificationBeforeInjecting m_waitForNotificationBeforeInjecting { WaitForNotificationBeforeInjecting::No };
74};
75
76template<class Encoder>
77void UserScript::encode(Encoder& encoder) const
78{
79 encoder << m_source;
80 encoder << m_url;
81 encoder << m_allowlist;
82 encoder << m_blocklist;
83 encoder << m_injectionTime;
84 encoder << m_injectedFrames;
85 encoder << m_waitForNotificationBeforeInjecting;
86}
87
88template<class Decoder>
89std::optional<UserScript> UserScript::decode(Decoder& decoder)
90{
91 std::optional<String> source;
92 decoder >> source;
93 if (!source)
94 return std::nullopt;
95
96 std::optional<URL> url;
97 decoder >> url;
98 if (!url)
99 return std::nullopt;
100
101 std::optional<Vector<String>> allowlist;
102 decoder >> allowlist;
103 if (!allowlist)
104 return std::nullopt;
105
106 std::optional<Vector<String>> blocklist;
107 decoder >> blocklist;
108 if (!blocklist)
109 return std::nullopt;
110
111 std::optional<UserScriptInjectionTime> injectionTime;
112 decoder >> injectionTime;
113 if (!injectionTime)
114 return std::nullopt;
115
116 std::optional<UserContentInjectedFrames> injectedFrames;
117 decoder >> injectedFrames;
118 if (!injectedFrames)
119 return std::nullopt;
120
121 std::optional<WaitForNotificationBeforeInjecting> waitForNotification;
122 decoder >> waitForNotification;
123 if (!waitForNotification)
124 return std::nullopt;
125
126 return {{
127 WTFMove(*source),
128 WTFMove(*url),
129 WTFMove(*allowlist),
130 WTFMove(*blocklist),
131 WTFMove(*injectionTime),
132 WTFMove(*injectedFrames),
133 WTFMove(*waitForNotification)
134 }};
135}
