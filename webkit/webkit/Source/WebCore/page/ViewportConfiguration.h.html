41class ViewportConfiguration {
42 WTF_MAKE_NONCOPYABLE(ViewportConfiguration); WTF_MAKE_FAST_ALLOCATED;
43public:
44 // FIXME: unify with ViewportArguments.
45 struct Parameters {
46 double width { 0 };
47 double height { 0 };
48 double initialScale { 0 };
49 double initialScaleIgnoringLayoutScaleFactor { 0 };
50 double minimumScale { 0 };
51 double maximumScale { 0 };
52 bool allowsUserScaling { false };
53 bool allowsShrinkToFit { false };
54 bool avoidsUnsafeArea { true };
55
56 bool widthIsSet { false };
57 bool heightIsSet { false };
58 bool initialScaleIsSet { false };
59
60 bool operator==(const Parameters& other) const
61 {
62 return width == other.width && height == other.height
63 && initialScale == other.initialScale && initialScaleIgnoringLayoutScaleFactor == other.initialScaleIgnoringLayoutScaleFactor && minimumScale == other.minimumScale && maximumScale == other.maximumScale
64 && allowsUserScaling == other.allowsUserScaling && allowsShrinkToFit == other.allowsShrinkToFit && avoidsUnsafeArea == other.avoidsUnsafeArea
65 && widthIsSet == other.widthIsSet && heightIsSet == other.heightIsSet && initialScaleIsSet == other.initialScaleIsSet;
66 }
67 };

93 constexpr double minimumEffectiveDeviceWidth() const
94 {
95 if (shouldIgnoreMinimumEffectiveDeviceWidth())
96 return 0;
97 return m_canIgnoreScalingConstraints ? m_minimumEffectiveDeviceWidthWhenIgnoringScalingConstraints : m_minimumEffectiveDeviceWidth;
98 }
99
100 constexpr bool isKnownToLayOutWiderThanViewport() const { return m_isKnownToLayOutWiderThanViewport; }
101 WEBCORE_EXPORT bool setIsKnownToLayOutWiderThanViewport(bool value);
102
103 constexpr bool shouldIgnoreMinimumEffectiveDeviceWidth() const
104 {
105 if (shouldShrinkToFitMinimumEffectiveDeviceWidthWhenIgnoringScalingConstraints())
106 return false;
107
108 if (m_canIgnoreScalingConstraints)
109 return true;
110
111 if (m_viewportArguments == ViewportArguments())
112 return false;
113
114 if ((m_viewportArguments.zoom == 1. || m_viewportArguments.width == ViewportArguments::ValueDeviceWidth) && !m_isKnownToLayOutWiderThanViewport)
115 return true;
116
117 return false;
118 }
119
120 constexpr bool shouldShrinkToFitMinimumEffectiveDeviceWidthWhenIgnoringScalingConstraints() const
121 {
122 return m_canIgnoreScalingConstraints && m_minimumEffectiveDeviceWidthWhenIgnoringScalingConstraints;
123 }

168 constexpr bool layoutSizeIsExplicitlyScaled() const
169 {
170 return m_layoutSizeScaleFactor != 1;
171 }
172
173 constexpr double forceAlwaysUserScalableMaximumScale() const
174 {
175 const double forceAlwaysUserScalableMaximumScaleIgnoringLayoutScaleFactor = 5;
176 return forceAlwaysUserScalableMaximumScaleIgnoringLayoutScaleFactor * effectiveLayoutSizeScaleFactor();
177 }
178
179 constexpr double forceAlwaysUserScalableMinimumScale() const
180 {
181 const double forceAlwaysUserScalableMinimumScaleIgnoringLayoutScaleFactor = 1;
182 return forceAlwaysUserScalableMinimumScaleIgnoringLayoutScaleFactor * effectiveLayoutSizeScaleFactor();
183 }
184
185 constexpr double effectiveLayoutSizeScaleFactor() const
186 {
187 if (!m_viewLayoutSize.width() || !minimumEffectiveDeviceWidth())
188 return m_layoutSizeScaleFactor;
189 return m_layoutSizeScaleFactor * m_viewLayoutSize.width() / std::max<double>(minimumEffectiveDeviceWidth(), m_viewLayoutSize.width());
190 }

203 double m_minimumEffectiveDeviceWidth { 0 };
204 double m_minimumEffectiveDeviceWidthWhenIgnoringScalingConstraints { 0 };
205 bool m_canIgnoreScalingConstraints;
206 bool m_forceAlwaysUserScalable;
207 bool m_isKnownToLayOutWiderThanViewport { false };
208};
