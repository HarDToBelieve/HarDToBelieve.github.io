33#define WEBKIT_NAMESPACE_RELEASE_LOG_ERROR(channel, fmt, ...) RELEASE_LOG_ERROR(channel, "%p - WebKitNamespace::" fmt, this, ##__VA_ARGS__)
34
35#if ENABLE(USER_MESSAGE_HANDLERS)
36
37#include "DOMWindow.h"
38#include "UserMessageHandlersNamespace.h"
39
40namespace WebCore {
41
42WebKitNamespace::WebKitNamespace(DOMWindow& window, UserContentProvider& userContentProvider)
43 : DOMWindowProperty(&window)
44 , m_messageHandlerNamespace(UserMessageHandlersNamespace::create(*window.frame(), userContentProvider))
45{
46 ASSERT(window.frame());
47}
48
49WebKitNamespace::~WebKitNamespace() = default;
50
51UserMessageHandlersNamespace* WebKitNamespace::messageHandlers()
52{
53#if ENABLE(APP_BOUND_DOMAINS)
54 if (frame()) {
55 if (frame()->loader().client().shouldEnableInAppBrowserPrivacyProtections()) {
56 WEBKIT_NAMESPACE_RELEASE_LOG_ERROR(Loading, "Ignoring messageHandlers() request for non app-bound domain");
57 return nullptr;
58 }
59 frame()->loader().client().notifyPageOfAppBoundBehavior();
60 }
61#endif
62
63 return &m_messageHandlerNamespace.get();
64}
65
66} // namespace WebCore
67
68#endif // ENABLE(USER_MESSAGE_HANDLERS)
