45WheelEventTestMonitor::WheelEventTestMonitor(Page& page)
46 : m_page(page)
47{
48}

50void WheelEventTestMonitor::clearAllTestDeferrals()
51{
52 Locker locker { m_lock };
53
54 ASSERT(isMainThread());
55 m_deferCompletionReasons.clear();
56 m_completionCallback = nullptr;
57 m_everHadDeferral = false;
58 m_receivedWheelEndOrCancel = false;
59 m_receivedMomentumEnd = false;
60 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::clearAllTestDeferrals: cleared all test state.");
61}
62
63void WheelEventTestMonitor::setTestCallbackAndStartMonitoring(bool expectWheelEndOrCancel, bool expectMomentumEnd, WTF::Function<void()>&& functionCallback)
64{
65 Locker locker { m_lock };
66
67 ASSERT(isMainThread());
68 m_completionCallback = WTFMove(functionCallback);
69#if ENABLE(KINETIC_SCROLLING)
70 m_expectWheelEndOrCancel = expectWheelEndOrCancel;
71 m_expectMomentumEnd = expectMomentumEnd;
72#else
73 UNUSED_PARAM(expectWheelEndOrCancel);
74 UNUSED_PARAM(expectMomentumEnd);
75#endif
76
77 m_page.scheduleRenderingUpdate(RenderingUpdateStep::WheelEventMonitorCallbacks);
78
79 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::setTestCallbackAndStartMonitoring - expect end/cancel " << expectWheelEndOrCancel << ", expect momentum end " << expectMomentumEnd);
80}
81
82void WheelEventTestMonitor::deferForReason(ScrollableAreaIdentifier identifier, DeferReason reason)
83{
84 Locker locker { m_lock };
85
86 m_deferCompletionReasons.ensure(identifier, [] {
87 return OptionSet<DeferReason>();
88 }).iterator->value.add(reason);
89
90 m_everHadDeferral = true;
91
92 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " (=) WheelEventTestMonitor::deferForReason: id=" << identifier << ", reason=" << reason);
93}
94
95void WheelEventTestMonitor::removeDeferralForReason(ScrollableAreaIdentifier identifier, DeferReason reason)
96{
97 Locker locker { m_lock };
98
99 auto it = m_deferCompletionReasons.find(identifier);
100 if (it == m_deferCompletionReasons.end()) {
101 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " (=) WheelEventTestMonitor::removeDeferralForReason: failed to find defer for id=" << identifier << ", reason=" << reason);
102 return;
103 }
104
105 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " (=) WheelEventTestMonitor::removeDeferralForReason: id=" << identifier << ", reason=" << reason);
106 it->value.remove(reason);
107
108 if (it->value.isEmpty())
109 m_deferCompletionReasons.remove(it);
110
111 scheduleCallbackCheck();
112}

114void WheelEventTestMonitor::receivedWheelEvent(const PlatformWheelEvent& event)
115{
116#if ENABLE(KINETIC_SCROLLING)
117 Locker locker { m_lock };
118
119 if (event.phase() == PlatformWheelEventPhase::Ended || event.phase() == PlatformWheelEventPhase::Cancelled)
120 m_receivedWheelEndOrCancel = true;
121
122 if (event.momentumPhase() == PlatformWheelEventPhase::Ended)
123 m_receivedMomentumEnd = true;
124#endif
125}
126
127void WheelEventTestMonitor::scheduleCallbackCheck()
128{
129 ensureOnMainThread([weakPage = makeWeakPtr(m_page)] {
130 if (weakPage)
131 weakPage->scheduleRenderingUpdate(RenderingUpdateStep::WheelEventMonitorCallbacks);
132 });
133}
134
135void WheelEventTestMonitor::checkShouldFireCallbacks()
136{
137 ASSERT(isMainThread());
138 {
139 Locker locker { m_lock };
140
141 if (!m_deferCompletionReasons.isEmpty()) {
142 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::checkShouldFireCallbacks - scrolling still active, reasons " << m_deferCompletionReasons);
143 return;
144 }
145
146 if (!m_everHadDeferral) {
147 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::checkShouldFireCallbacks - have not yet seen any deferral reasons");
148 return;
149 }
150
151 if (m_expectWheelEndOrCancel && !m_receivedWheelEndOrCancel) {
152 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::checkShouldFireCallbacks - have not seen end of of wheel phase");
153 return;
154 }
155
156 if (m_expectMomentumEnd && !m_receivedMomentumEnd) {
157 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::checkShouldFireCallbacks - have not seen end of of momentum phase");
158 return;
159 }
160 }
161
162 if (auto functionCallback = WTFMove(m_completionCallback)) {
163 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::checkShouldFireCallbacks: scrolling is idle, FIRING TEST");
164 functionCallback();
165 } else
166 LOG_WITH_STREAM(WheelEventTestMonitor, stream << " WheelEventTestMonitor::checkShouldFireCallbacks - no callback");
167}

169TextStream& operator<<(TextStream& ts, WheelEventTestMonitor::DeferReason reason)
170{
171 switch (reason) {
172 case WheelEventTestMonitor::HandlingWheelEvent: ts << "handling wheel event"; break;
173 case WheelEventTestMonitor::HandlingWheelEventOnMainThread: ts << "handling wheel event on main thread"; break;
174 case WheelEventTestMonitor::PostMainThreadWheelEventHandling: ts << "post-main thread event handling"; break;
175 case WheelEventTestMonitor::RubberbandInProgress: ts << "rubberbanding"; break;
176 case WheelEventTestMonitor::ScrollSnapInProgress: ts << "scroll-snapping"; break;
177 case WheelEventTestMonitor::ScrollingThreadSyncNeeded: ts << "scrolling thread sync needed"; break;
178 case WheelEventTestMonitor::ContentScrollInProgress: ts << "content scrolling"; break;
179 case WheelEventTestMonitor::RequestedScrollPosition: ts << "requested scroll position"; break;
180 }
181 return ts;
182}

184TextStream& operator<<(TextStream& ts, const WheelEventTestMonitor::ScrollableAreaReasonMap& reasonMap)
185{
186 for (const auto& regionReasonsPair : reasonMap)
187 ts << " scroll region: " << regionReasonsPair.key << " reasons: " << regionReasonsPair.value;
188
189 return ts;
190}
