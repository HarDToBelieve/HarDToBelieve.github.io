44// https://html.spec.whatwg.org/#feature-separator
45static bool isSeparator(UChar character, FeatureMode mode)
46{
47 if (mode == FeatureMode::Viewport)
48 return character == ' ' || character == '\t' || character == '\n' || character == '\r' || character == '=' || character == ',';
49
50 return isASCIISpace(character) || character == '=' || character == ',';
51}

53WindowFeatures parseWindowFeatures(StringView featuresString)
54{
55 // The IE rule is: all features except for channelmode and fullscreen default to YES, but
56 // if the user specifies a feature string, all features default to NO. (There is no public
57 // standard that applies to this method.)
58 //
59 // <http://msdn.microsoft.com/workshop/author/dhtml/reference/methods/open_0.asp>
60 // We always allow a window to be resized, which is consistent with Firefox.
61
62 WindowFeatures features;
63
64 if (featuresString.isEmpty())
65 return features;
66
67 features.menuBarVisible = false;
68 features.statusBarVisible = false;
69 features.toolBarVisible = false;
70 features.locationBarVisible = false;
71 features.scrollbarsVisible = false;
72 features.noopener = false;
73
74 processFeaturesString(featuresString, FeatureMode::Window, [&features](StringView key, StringView value) {
75 setWindowFeature(features, key, value);
76 });
77
78 return features;
79}

81// Window: https://html.spec.whatwg.org/#concept-window-open-features-tokenize
82// Viewport: https://developer.apple.com/library/content/documentation/AppleApplications/Reference/SafariHTMLRef/Articles/MetaTags.html#//apple_ref/doc/uid/TP40008193-SW6
83// FIXME: We should considering aligning Viewport feature parsing with Window features parsing.
84void processFeaturesString(StringView features, FeatureMode mode, const WTF::Function<void(StringView type, StringView value)>& callback)
85{
86 unsigned length = features.length();
87 for (unsigned i = 0; i < length; ) {
88 // Skip to first non-separator.
89 while (i < length && isSeparator(features[i], mode))
90 ++i;
91 unsigned keyBegin = i;
92
93 // Skip to first separator.
94 while (i < length && !isSeparator(features[i], mode))
95 i++;
96 unsigned keyEnd = i;
97
98 // Skip to first '=', but don't skip past a ',' or a non-separator.
99 while (i < length && features[i] != '=' && features[i] != ',' && (mode == FeatureMode::Viewport || isSeparator(features[i], mode)))
100 ++i;
101
102 // Skip to first non-separator, but don't skip past a ','.
103 if (mode == FeatureMode::Viewport || (i < length && isSeparator(features[i], mode))) {
104 while (i < length && isSeparator(features[i], mode) && features[i] != ',')
105 ++i;
106 unsigned valueBegin = i;
107
108 // Skip to first separator.
109 while (i < length && !isSeparator(features[i], mode))
110 ++i;
111 unsigned valueEnd = i;
112 callback(features.substring(keyBegin, keyEnd - keyBegin), features.substring(valueBegin, valueEnd - valueBegin));
113 } else
114 callback(features.substring(keyBegin, keyEnd - keyBegin), StringView());
115 }
116}

118OptionSet<DisabledAdaptations> parseDisabledAdaptations(const String& disabledAdaptationsString)
119{
120 OptionSet<DisabledAdaptations> disabledAdaptations;
121 for (auto& name : disabledAdaptationsString.split(',')) {
122 auto normalizedName = name.stripWhiteSpace().convertToASCIILowercase();
123 if (normalizedName == watchAdaptationName())
124 disabledAdaptations.add(DisabledAdaptations::Watch);
125 }
126 return disabledAdaptations;
127}

129static void setWindowFeature(WindowFeatures& features, StringView key, StringView value)
130{
131 // Listing a key with no value is shorthand for key=yes
132 int numericValue;
133 if (value.isEmpty() || equalLettersIgnoringASCIICase(value, "yes"))
134 numericValue = 1;
135 else
136 numericValue = parseIntegerAllowingTrailingJunk<int>(value).value_or(0);
137
138 // We treat key of "resizable" here as an additional feature rather than setting resizeable to true.
139 // This is consistent with Firefox, but could also be handled at another level.
140
141 if (equalLettersIgnoringASCIICase(key, "left") || equalLettersIgnoringASCIICase(key, "screenx"))
142 features.x = numericValue;
143 else if (equalLettersIgnoringASCIICase(key, "top") || equalLettersIgnoringASCIICase(key, "screeny"))
144 features.y = numericValue;
145 else if (equalLettersIgnoringASCIICase(key, "width") || equalLettersIgnoringASCIICase(key, "innerwidth"))
146 features.width = numericValue;
147 else if (equalLettersIgnoringASCIICase(key, "height") || equalLettersIgnoringASCIICase(key, "innerheight"))
148 features.height = numericValue;
149 else if (equalLettersIgnoringASCIICase(key, "menubar"))
150 features.menuBarVisible = numericValue;
151 else if (equalLettersIgnoringASCIICase(key, "toolbar"))
152 features.toolBarVisible = numericValue;
153 else if (equalLettersIgnoringASCIICase(key, "location"))
154 features.locationBarVisible = numericValue;
155 else if (equalLettersIgnoringASCIICase(key, "status"))
156 features.statusBarVisible = numericValue;
157 else if (equalLettersIgnoringASCIICase(key, "fullscreen"))
158 features.fullscreen = numericValue;
159 else if (equalLettersIgnoringASCIICase(key, "scrollbars"))
160 features.scrollbarsVisible = numericValue;
161 else if (equalLettersIgnoringASCIICase(key, "noopener"))
162 features.noopener = numericValue;
163 else if (equalLettersIgnoringASCIICase(key, "noreferrer"))
164 features.noreferrer = numericValue;
165 else if (numericValue == 1)
166 features.additionalFeatures.append(key.toString());
167}

169WindowFeatures parseDialogFeatures(const String& dialogFeaturesString, const FloatRect& screenAvailableRect)
170{
171 auto featuresMap = parseDialogFeaturesMap(dialogFeaturesString);
172
173 // The following features from Microsoft's documentation are not implemented:
174 // - default font settings
175 // - width, height, left, and top specified in units other than "px"
176 // - edge (sunken or raised, default is raised)
177 // - dialogHide: trusted && boolFeature(features, "dialoghide"), makes dialog hide when you print
178 // - help: boolFeature(features, "help", true), makes help icon appear in dialog (what does it do on Windows?)
179 // - unadorned: trusted && boolFeature(features, "unadorned");
180
181 WindowFeatures features;
182
183 features.menuBarVisible = false;
184 features.toolBarVisible = false;
185 features.locationBarVisible = false;
186 features.dialog = true;
187
188 float width = floatFeature(featuresMap, "dialogwidth", 100, screenAvailableRect.width()).value_or(620); // default here came from frame size of dialog in MacIE
189 float height = floatFeature(featuresMap, "dialogheight", 100, screenAvailableRect.height()).value_or(450); // default here came from frame size of dialog in MacIE
190
191 features.width = width;
192 features.height = height;
193
194 features.x = floatFeature(featuresMap, "dialogleft", screenAvailableRect.x(), screenAvailableRect.maxX() - width);
195 features.y = floatFeature(featuresMap, "dialogtop", screenAvailableRect.y(), screenAvailableRect.maxY() - height);
196
197 if (boolFeature(featuresMap, "center").value_or(true)) {
198 if (!features.x)
199 features.x = screenAvailableRect.x() + (screenAvailableRect.width() - width) / 2;
200 if (!features.y)
201 features.y = screenAvailableRect.y() + (screenAvailableRect.height() - height) / 2;
202 }
203
204 features.resizable = boolFeature(featuresMap, "resizable").value_or(false);
205 features.scrollbarsVisible = boolFeature(featuresMap, "scroll").value_or(true);
206 features.statusBarVisible = boolFeature(featuresMap, "status").value_or(false);
207
208 return features;
209}

211static std::optional<bool> boolFeature(const DialogFeaturesMap& features, const char* key)
212{
213 auto it = features.find(key);
214 if (it == features.end())
215 return std::nullopt;
216
217 auto& value = it->value;
218 return value.isNull()
219 || value == "1"
220 || equalLettersIgnoringASCIICase(value, "yes")
221 || equalLettersIgnoringASCIICase(value, "on");
222}

224static std::optional<float> floatFeature(const DialogFeaturesMap& features, const char* key, float min, float max)
225{
226 auto it = features.find(key);
227 if (it == features.end())
228 return std::nullopt;
229
230 // FIXME: The toDouble function does not offer a way to tell "0q" from string with no digits in it: Both
231 // return the number 0 and false for ok. But "0q" should yield the minimum rather than the default.
232 bool ok;
233 double parsedNumber = it->value.toDouble(&ok);
234 if ((!parsedNumber && !ok) || std::isnan(parsedNumber))
235 return std::nullopt;
236 if (parsedNumber < min || max <= min)
237 return min;
238 if (parsedNumber > max)
239 return max;
240
241 // FIXME: Seems strange to cast a double to int and then convert back to a float. Why is this a good idea?
242 return static_cast<int>(parsedNumber);
243}

245static DialogFeaturesMap parseDialogFeaturesMap(const String& string)
246{
247 // FIXME: Not clear why we take such a different approach to parsing dialog features
248 // as opposed to window features (using a map, different parsing quirks).
249
250 DialogFeaturesMap features;
251
252 for (auto& featureString : string.split(';')) {
253 size_t separatorPosition = featureString.find('=');
254 size_t colonPosition = featureString.find(':');
255 if (separatorPosition != notFound && colonPosition != notFound)
256 continue; // ignore strings that have both = and :
257 if (separatorPosition == notFound)
258 separatorPosition = colonPosition;
259
260 String key = featureString.left(separatorPosition).stripWhiteSpace();
261
262 // Null string for value indicates key without value.
263 String value;
264 if (separatorPosition != notFound) {
265 value = featureString.substring(separatorPosition + 1).stripWhiteSpace();
266 value = value.left(value.find(' '));
267 }
268
269 features.set(key, value);
270 }
271
272 return features;
273}
