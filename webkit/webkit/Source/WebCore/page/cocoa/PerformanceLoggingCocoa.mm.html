34std::optional<uint64_t> PerformanceLogging::physicalFootprint()
35{
36 task_vm_info_data_t vmInfo;
37 mach_msg_type_number_t count = TASK_VM_INFO_COUNT;
38 kern_return_t result = task_info(mach_task_self(), TASK_VM_INFO, (task_info_t) &vmInfo, &count);
39 if (result != KERN_SUCCESS)
40 return std::nullopt;
41 return vmInfo.phys_footprint;
42}
43
44void PerformanceLogging::getPlatformMemoryUsageStatistics(HashMap<const char*, size_t>& stats)
45{
46 task_vm_info_data_t vmInfo;
47 mach_msg_type_number_t count = TASK_VM_INFO_COUNT;
48 kern_return_t err = task_info(mach_task_self(), TASK_VM_INFO, (task_info_t) &vmInfo, &count);
49 if (err != KERN_SUCCESS)
50 return;
51 stats.add("internal", static_cast<size_t>(vmInfo.internal));
52 stats.add("compressed", static_cast<size_t>(vmInfo.compressed));
53 stats.add("phys_footprint", static_cast<size_t>(vmInfo.phys_footprint));
54 stats.add("resident_size", static_cast<size_t>(vmInfo.resident_size));
55 stats.add("virtual_size", static_cast<size_t>(vmInfo.virtual_size));
56}
