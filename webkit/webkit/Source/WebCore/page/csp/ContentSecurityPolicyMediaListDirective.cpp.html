38template<typename CharacterType> static bool isMediaTypeCharacter(CharacterType c)
39{
40 return !isASCIISpace(c) && c != '/';
41}
42
43ContentSecurityPolicyMediaListDirective::ContentSecurityPolicyMediaListDirective(const ContentSecurityPolicyDirectiveList& directiveList, const String& name, const String& value)
44 : ContentSecurityPolicyDirective(directiveList, name, value)
45{
46 parse(value);
47}
48
49bool ContentSecurityPolicyMediaListDirective::allows(const String& type) const
50{
51 return m_pluginTypes.contains(type);
52}
53
54void ContentSecurityPolicyMediaListDirective::parse(const String& value)
55{
56 // 'plugin-types ____;' OR 'plugin-types;'
57 if (value.isEmpty()) {
58 directiveList().policy().reportInvalidPluginTypes(value);
59 return;
60 }
61
62 readCharactersForParsing(value, [&](auto buffer) {
63 while (buffer.hasCharactersRemaining()) {
64 // _____ OR _____mime1/mime1
65 // ^ ^
66 skipWhile<isASCIISpace>(buffer);
67 if (buffer.atEnd())
68 return;
69
70 // mime1/mime1 mime2/mime2
71 // ^
72 auto begin = buffer.position();
73 if (!skipExactly<isMediaTypeCharacter>(buffer)) {
74 skipWhile<isNotASCIISpace>(buffer);
75 directiveList().policy().reportInvalidPluginTypes(String(begin, buffer.position() - begin));
76 continue;
77 }
78 skipWhile<isMediaTypeCharacter>(buffer);
79
80 // mime1/mime1 mime2/mime2
81 // ^
82 if (!skipExactly(buffer, '/')) {
83 skipWhile<isNotASCIISpace>(buffer);
84 directiveList().policy().reportInvalidPluginTypes(String(begin, buffer.position() - begin));
85 continue;
86 }
87
88 // mime1/mime1 mime2/mime2
89 // ^
90 if (!skipExactly<isMediaTypeCharacter>(buffer)) {
91 skipWhile<isNotASCIISpace>(buffer);
92 directiveList().policy().reportInvalidPluginTypes(String(begin, buffer.position() - begin));
93 continue;
94 }
95 skipWhile<isMediaTypeCharacter>(buffer);
96
97 // mime1/mime1 mime2/mime2 OR mime1/mime1 OR mime1/mime1/error
98 // ^ ^ ^
99 if (buffer.hasCharactersRemaining() && isNotASCIISpace(*buffer)) {
100 skipWhile<isNotASCIISpace>(buffer);
101 directiveList().policy().reportInvalidPluginTypes(String(begin, buffer.position() - begin));
102 continue;
103 }
104 m_pluginTypes.add(String(begin, buffer.position() - begin));
105
106 ASSERT(buffer.atEnd() || isASCIISpace(*buffer));
107 }
108 });
109}
