32namespace WebCore {
33
34ContentSecurityPolicyResponseHeaders::ContentSecurityPolicyResponseHeaders(const ResourceResponse& response)
35{
36 String policyValue = response.httpHeaderField(HTTPHeaderName::ContentSecurityPolicy);
37 if (!policyValue.isEmpty())
38 m_headers.append({ policyValue, ContentSecurityPolicyHeaderType::Enforce });
39
40 policyValue = response.httpHeaderField(HTTPHeaderName::ContentSecurityPolicyReportOnly);
41 if (!policyValue.isEmpty())
42 m_headers.append({ policyValue, ContentSecurityPolicyHeaderType::Report });
43
44 m_httpStatusCode = response.httpStatusCode();
45}
46
47ContentSecurityPolicyResponseHeaders ContentSecurityPolicyResponseHeaders::isolatedCopy() const
48{
49 ContentSecurityPolicyResponseHeaders isolatedCopy;
50 isolatedCopy.m_headers.reserveInitialCapacity(m_headers.size());
51 for (auto& header : m_headers)
52 isolatedCopy.m_headers.uncheckedAppend({ header.first.isolatedCopy(), header.second });
53 isolatedCopy.m_httpStatusCode = m_httpStatusCode;
54 return isolatedCopy;
55}
