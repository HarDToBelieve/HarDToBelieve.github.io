43static bool isCSPDirectiveName(const String& name)
44{
45 return equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::baseURI)
46 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::connectSrc)
47 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::defaultSrc)
48 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::fontSrc)
49 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::formAction)
50 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::frameSrc)
51 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::imgSrc)
52 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::mediaSrc)
53 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::objectSrc)
54 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::pluginTypes)
55 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::reportURI)
56 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::sandbox)
57 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::scriptSrc)
58 || equalIgnoringASCIICase(name, ContentSecurityPolicyDirectiveNames::styleSrc);
59}

61template<typename CharacterType> static bool isSourceCharacter(CharacterType c)
62{
63 return !isASCIISpace(c);
64}
65
66template<typename CharacterType> static bool isHostCharacter(CharacterType c)
67{
68 return isASCIIAlphanumeric(c) || c == '-';
69}
70
71template<typename CharacterType> static bool isPathComponentCharacter(CharacterType c)
72{
73 return c != '?' && c != '#';
74}
75
76template<typename CharacterType> static bool isSchemeContinuationCharacter(CharacterType c)
77{
78 return isASCIIAlphanumeric(c) || c == '+' || c == '-' || c == '.';
79}
80
81template<typename CharacterType> static bool isNotColonOrSlash(CharacterType c)
82{
83 return c != ':' && c != '/';
84}

86template<typename CharacterType> static bool isSourceListNone(StringParsingBuffer<CharacterType> buffer)
87{
88 skipWhile<isASCIISpace>(buffer);
89
90 if (!skipExactlyIgnoringASCIICase(buffer, "'none'"))
91 return false;
92
93 skipWhile<isASCIISpace>(buffer);
94
95 return buffer.atEnd();
96}

116bool ContentSecurityPolicySourceList::isProtocolAllowedByStar(const URL& url) const
117{
118 if (m_policy.allowContentSecurityPolicySourceStarToMatchAnyProtocol())
119 return true;
120
121 // Although not allowed by the Content Security Policy Level 3 spec., we allow a data URL to match
122 // "img-src *" and either a data URL or blob URL to match "media-src *" for web compatibility.
123 bool isAllowed = url.protocolIsInHTTPFamily() || url.protocolIs("ws") || url.protocolIs("wss") || m_policy.protocolMatchesSelf(url);
124 if (equalIgnoringASCIICase(m_directiveName, ContentSecurityPolicyDirectiveNames::imgSrc))
125 isAllowed |= url.protocolIsData();
126 else if (equalIgnoringASCIICase(m_directiveName, ContentSecurityPolicyDirectiveNames::mediaSrc))
127 isAllowed |= url.protocolIsData() || url.protocolIsBlob();
128 return isAllowed;
129}

131bool ContentSecurityPolicySourceList::matches(const URL& url, bool didReceiveRedirectResponse) const
132{
133 if (m_allowStar && isProtocolAllowedByStar(url))
134 return true;
135
136 if (m_allowSelf && m_policy.urlMatchesSelf(url, equalIgnoringASCIICase(m_directiveName, ContentSecurityPolicyDirectiveNames::frameSrc)
137))
138 return true;
139
140 for (auto& entry : m_list) {
141 if (entry.matches(url, didReceiveRedirectResponse))
142 return true;
143 }
144
145 return false;
146}

158// source-list = *WSP [ source *( 1*WSP source ) *WSP ]
159// / *WSP "'none'" *WSP
160//
161template<typename CharacterType> void ContentSecurityPolicySourceList::parse(StringParsingBuffer<CharacterType> buffer)
162{
163 while (buffer.hasCharactersRemaining()) {
164 skipWhile<isASCIISpace>(buffer);
165 if (buffer.atEnd())
166 return;
167
168 auto beginSource = buffer.position();
169 skipWhile<isSourceCharacter>(buffer);
170
171 auto sourceBuffer = StringParsingBuffer { beginSource, buffer.position() };
172
173 if (parseNonceSource(sourceBuffer))
174 continue;
175
176 if (parseHashSource(sourceBuffer))
177 continue;
178
179 if (auto source = parseSource(sourceBuffer)) {
180 // Wildcard hosts and keyword sources ('self', 'unsafe-inline',
181 // etc.) aren't stored in m_list, but as attributes on the source
182 // list itself.
183 if (source->scheme.isEmpty() && source->host.value.isEmpty())
184 continue;
185 if (isCSPDirectiveName(source->host.value))
186 m_policy.reportDirectiveAsSourceExpression(m_directiveName, source->host.value);
187 m_list.append(ContentSecurityPolicySource(m_policy, source->scheme, source->host.value, source->port.value, source->path, source->host.hasWildcard, source->port.hasWildcard));
188 } else
189 m_policy.reportInvalidSourceExpression(m_directiveName, String(beginSource, buffer.position() - beginSource));
190
191 ASSERT(buffer.atEnd() || isASCIISpace(*buffer));
192 }
193
194 m_list.shrinkToFit();
195}

197// source = scheme ":"
198// / ( [ scheme "://" ] host [ port ] [ path ] )
199// / "'self'"
200//
201template<typename CharacterType> std::optional<ContentSecurityPolicySourceList::Source> ContentSecurityPolicySourceList::parseSource(StringParsingBuffer<CharacterType> buffer)
202{
203 if (buffer.atEnd())
204 return std::nullopt;
205
206 if (skipExactlyIgnoringASCIICase(buffer, "'none'"))
207 return std::nullopt;
208
209 Source source;
210
211 if (buffer.lengthRemaining() == 1 && *buffer == '*') {
212 m_allowStar = true;
213 return source;
214 }
215
216 if (skipExactlyIgnoringASCIICase(buffer, "'self'")) {
217 m_allowSelf = true;
218 return source;
219 }
220
221 if (skipExactlyIgnoringASCIICase(buffer, "'unsafe-inline'")) {
222 m_allowInline = true;
223 return source;
224 }
225
226 if (skipExactlyIgnoringASCIICase(buffer, "'unsafe-eval'")) {
227 m_allowEval = true;
228 return source;
229 }
230
231 auto begin = buffer.position();
232 auto beginHost = begin;
233 auto beginPath = buffer.end();
234 const CharacterType* beginPort = nullptr;
235
236 skipWhile<isNotColonOrSlash>(buffer);
237
238 if (buffer.atEnd()) {
239 // host
240 // ^
241 auto host = parseHost(StringParsingBuffer { beginHost, buffer.position() });
242 if (!host)
243 return std::nullopt;
244
245 source.host = WTFMove(*host);
246 return source;
247 }
248
249 if (buffer.hasCharactersRemaining() && *buffer == '/') {
250 // host/path || host/ || /
251 // ^ ^ ^
252 auto host = parseHost(StringParsingBuffer { beginHost, buffer.position() });
253 if (!host)
254 return std::nullopt;
255
256 auto path = parsePath(buffer);
257 if (!path)
258 return std::nullopt;
259
260 source.host = WTFMove(*host);
261 source.path = WTFMove(*path);
262 return source;
263 }
264
265 if (buffer.hasCharactersRemaining() && *buffer == ':') {
266 if (buffer.lengthRemaining() == 1) {
267 // scheme:
268 // ^
269 auto scheme = parseScheme(StringParsingBuffer { begin, buffer.position() });
270 if (!scheme)
271 return std::nullopt;
272
273 source.scheme = WTFMove(*scheme);
274 return source;
275 }
276
277 if (buffer[1] == '/') {
278 // scheme://host || scheme://
279 // ^ ^
280 auto scheme = parseScheme(StringParsingBuffer { begin, buffer.position() });
281 if (!scheme
282 || !skipExactly(buffer, ':')
283 || !skipExactly(buffer, '/')
284 || !skipExactly(buffer, '/'))
285 return std::nullopt;
286 if (buffer.atEnd())
287 return std::nullopt;
288
289 source.scheme = WTFMove(*scheme);
290
291 beginHost = buffer.position();
292 skipWhile<isNotColonOrSlash>(buffer);
293 }
294
295 if (buffer.hasCharactersRemaining() && *buffer == ':') {
296 // host:port || scheme://host:port
297 // ^ ^
298 beginPort = buffer.position();
299 skipUntil(buffer, '/');
300 }
301 }
302
303 if (buffer.hasCharactersRemaining() && *buffer == '/') {
304 // scheme://host/path || scheme://host:port/path
305 // ^ ^
306 if (buffer.position() == beginHost)
307 return std::nullopt;
308
309 beginPath = buffer.position();
310 }
311
312 auto host = parseHost(StringParsingBuffer { beginHost, beginPort ? beginPort : beginPath });
313 if (!host)
314 return std::nullopt;
315
316 if (beginPort) {
317 auto port = parsePort(StringParsingBuffer { beginPort, beginPath });
318 if (!port)
319 return std::nullopt;
320
321 source.port = WTFMove(*port);
322 }
323
324 if (beginPath != buffer.end()) {
325 auto path = parsePath(StringParsingBuffer { beginPath, buffer.end() });
326 if (!path)
327 return std::nullopt;
328
329 source.path = WTFMove(*path);
330 }
331
332 source.host = WTFMove(*host);
333 return source;
334}

336// ; <scheme> production from RFC 3986
337// scheme = ALPHA *( ALPHA / DIGIT / "+" / "-" / "." )
338//
339template<typename CharacterType> std::optional<String> ContentSecurityPolicySourceList::parseScheme(StringParsingBuffer<CharacterType> buffer)
340{
341 ASSERT(buffer.position() <= buffer.end());
342
343 if (buffer.atEnd())
344 return std::nullopt;
345
346 auto begin = buffer.position();
347
348 if (!skipExactly<isASCIIAlpha>(buffer))
349 return std::nullopt;
350
351 skipWhile<isSchemeContinuationCharacter>(buffer);
352
353 if (!buffer.atEnd())
354 return std::nullopt;
355
356 return String(begin, buffer.position() - begin);
357}

359// host = [ "*." ] 1*host-char *( "." 1*host-char )
360// / "*"
361// host-char = ALPHA / DIGIT / "-"
362//
363template<typename CharacterType> std::optional<ContentSecurityPolicySourceList::Host> ContentSecurityPolicySourceList::parseHost(StringParsingBuffer<CharacterType> buffer)
364{
365 ASSERT(buffer.position() <= buffer.end());
366
367 if (buffer.atEnd())
368 return std::nullopt;
369
370 Host host;
371
372 if (skipExactly(buffer, '*')) {
373 host.hasWildcard = true;
374
375 if (buffer.atEnd())
376 return host;
377
378 if (!skipExactly(buffer, '.'))
379 return std::nullopt;
380 }
381
382 auto hostBegin = buffer.position();
383
384 while (buffer.hasCharactersRemaining()) {
385 if (!skipExactly<isHostCharacter>(buffer))
386 return std::nullopt;
387
388 skipWhile<isHostCharacter>(buffer);
389
390 if (buffer.hasCharactersRemaining() && !skipExactly(buffer, '.'))
391 return std::nullopt;
392 }
393
394 ASSERT(buffer.atEnd());
395 host.value = String(hostBegin, buffer.position() - hostBegin);
396 return host;
397}

399template<typename CharacterType> std::optional<String> ContentSecurityPolicySourceList::parsePath(StringParsingBuffer<CharacterType> buffer)
400{
401 ASSERT(buffer.position() <= buffer.end());
402
403 auto begin = buffer.position();
404 skipWhile<isPathComponentCharacter>(buffer);
405 // path/to/file.js?query=string || path/to/file.js#anchor
406 // ^ ^
407 if (buffer.hasCharactersRemaining())
408 m_policy.reportInvalidPathCharacter(m_directiveName, String(begin, buffer.end() - begin), *buffer);
409
410 ASSERT(buffer.position() <= buffer.end());
411 ASSERT(buffer.atEnd() || (*buffer == '#' || *buffer == '?'));
412
413 return decodeURLEscapeSequences(StringView(begin, buffer.position() - begin));
414}

416// port = ":" ( 1*DIGIT / "*" )
417//
418template<typename CharacterType> std::optional<ContentSecurityPolicySourceList::Port> ContentSecurityPolicySourceList::parsePort(StringParsingBuffer<CharacterType> buffer)
419{
420 ASSERT(buffer.position() <= buffer.end());
421
422 if (!skipExactly(buffer, ':'))
423 ASSERT_NOT_REACHED();
424
425 if (buffer.atEnd())
426 return std::nullopt;
427
428 if (buffer.lengthRemaining() == 1 && *buffer == '*') {
429 Port port;
430 port.hasWildcard = true;
431 return port;
432 }
433
434 auto begin = buffer.position();
435 skipWhile<isASCIIDigit>(buffer);
436
437 if (!buffer.atEnd())
438 return std::nullopt;
439
440 unsigned length = buffer.position() - begin;
441 auto portInteger = parseInteger<uint16_t>({ begin, length }).value_or(0);
442 if (!portInteger)
443 return std::nullopt;
444
445 Port port;
446 port.value = portInteger;
447 return port;
448}

450// Match Blink's behavior of allowing an equal sign to appear anywhere in the value of the nonce
451// even though this does not match the behavior of Content Security Policy Level 3 spec.,
452// <https://w3c.github.io/webappsec-csp/> (29 February 2016).
453template<typename CharacterType> static bool isNonceCharacter(CharacterType c)
454{
455 return isBase64OrBase64URLCharacter(c) || c == '=';
456}
457
458// nonce-source = "'nonce-" nonce-value "'"
459// nonce-value = base64-value
460template<typename CharacterType> bool ContentSecurityPolicySourceList::parseNonceSource(StringParsingBuffer<CharacterType> buffer)
461{
462 if (!skipExactlyIgnoringASCIICase(buffer, "'nonce-"))
463 return false;
464
465 auto beginNonceValue = buffer.position();
466 skipWhile<isNonceCharacter>(buffer);
467 if (buffer.atEnd() || buffer.position() == beginNonceValue || *buffer != '\'')
468 return false;
469 m_nonces.add(String(beginNonceValue, buffer.position() - beginNonceValue));
470 return true;
471}
472
473// hash-source = "'" hash-algorithm "-" base64-value "'"
474// hash-algorithm = "sha256" / "sha384" / "sha512"
475// base64-value = 1*( ALPHA / DIGIT / "+" / "/" / "-" / "_" )*2( "=" )
476template<typename CharacterType> bool ContentSecurityPolicySourceList::parseHashSource(StringParsingBuffer<CharacterType> buffer)
477{
478 if (buffer.atEnd())
479 return false;
480
481 if (!skipExactly(buffer, '\''))
482 return false;
483
484 auto digest = parseCryptographicDigest(buffer);
485 if (!digest)
486 return false;
487
488 if (buffer.atEnd() || *buffer != '\'')
489 return false;
490
491 if (digest->value.size() > ContentSecurityPolicyHash::maximumDigestLength)
492 return false;
493
494 m_hashAlgorithmsUsed.add(digest->algorithm);
495 m_hashes.add(WTFMove(*digest));
496 return true;
497}
