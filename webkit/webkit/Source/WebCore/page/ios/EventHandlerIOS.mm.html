102bool EventHandler::wheelEvent(WebEvent *event)
103{
104 Page* page = m_frame.page();
105 if (!page)
106 return false;
107
108 CurrentEventScope scope(event);
109
110 auto wheelEvent = PlatformEventFactory::createPlatformWheelEvent(event);
111 OptionSet<WheelEventProcessingSteps> processingSteps = { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch };
112
113 if (wheelEvent.isGestureStart())
114 m_wheelScrollGestureState = std::nullopt;
115 else if (wheelEvent.phase() == PlatformWheelEventPhase::Changed || wheelEvent.momentumPhase() == PlatformWheelEventPhase::Changed) {
116 if (m_wheelScrollGestureState && *m_wheelScrollGestureState == WheelScrollGestureState::NonBlocking)
117 processingSteps = { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch };
118 }
119
120 bool eventWasHandled = handleWheelEvent(wheelEvent, processingSteps);
121 event.wasHandled = eventWasHandled;
122 return eventWasHandled;
123}

142bool EventHandler::tabsToAllFormControls(KeyboardEvent* event) const
143{
144 Page* page = m_frame.page();
145 if (!page)
146 return false;
147
148 KeyboardUIMode keyboardUIMode = page->chrome().client().keyboardUIMode();
149 bool handlingOptionTab = event && isKeyboardOptionTab(*event);
150
151 // If tab-to-links is off, option-tab always highlights all controls.
152 if ((keyboardUIMode & KeyboardAccessTabsToLinks) == 0 && handlingOptionTab)
153 return true;
154
155 // If system preferences say to include all controls, we always include all controls.
156 if (keyboardUIMode & KeyboardAccessFull)
157 return true;
158
159 // Otherwise tab-to-links includes all controls, unless the sense is flipped via option-tab.
160 if (keyboardUIMode & KeyboardAccessTabsToLinks)
161 return !handlingOptionTab;
162
163 return handlingOptionTab;
164}

201bool EventHandler::passWidgetMouseDownEventToWidget(const MouseEventWithHitTestResults& event)
202{
203 // Figure out which view to send the event to.
204 auto* target = event.targetNode() ? event.targetNode()->renderer() : nullptr;
205 if (!is<RenderWidget>(target))
206 return false;
207
208 // Double-click events don't exist in Cocoa. Since passWidgetMouseDownEventToWidget() will
209 // just pass currentEvent down to the widget, we don't want to call it for events that
210 // don't correspond to Cocoa events. The mousedown/ups will have already been passed on as
211 // part of the pressed/released handling.
212 return passMouseDownEventToWidget(downcast<RenderWidget>(*target).widget());
213}

215bool EventHandler::passWidgetMouseDownEventToWidget(RenderWidget* renderWidget)
216{
217 return passMouseDownEventToWidget(renderWidget->widget());
218}

220static bool lastEventIsMouseUp()
221{
222 // Many AppKit widgets run their own event loops and consume events while the mouse is down.
223 // When they finish, currentEvent is the mouseUp that they exited on. We need to update
224 // the WebCore state with this mouseUp, which we never saw. This method lets us detect
225 // that state. Handling this was critical when we used AppKit widgets for form elements.
226 // It's not clear in what cases this is helpful now -- it's possible it can be removed.
227
228 BEGIN_BLOCK_OBJC_EXCEPTIONS
229 WebEvent *currentEventAfterHandlingMouseDown = [WAKWindow currentEvent];
230 return currentEventAfterHandlingMouseDown
231 && EventHandler::currentEvent() != currentEventAfterHandlingMouseDown
232 && currentEventAfterHandlingMouseDown.type == WebEventMouseUp
233 && currentEventAfterHandlingMouseDown.timestamp >= EventHandler::currentEvent().timestamp;
234 END_BLOCK_OBJC_EXCEPTIONS
235
236 return false;
237}

239bool EventHandler::passMouseDownEventToWidget(Widget* pWidget)
240{
241 // FIXME: This function always returns true. It should be changed either to return
242 // false in some cases or the return value should be removed.
243
244 RefPtr<Widget> widget = pWidget;
245
246 if (!widget) {
247 LOG_ERROR("hit a RenderWidget without a corresponding Widget, means a frame is half-constructed");
248 return true;
249 }
250
251 // In WebKit2 we will never have a native widget. Just return early and let the regular event handler machinery take care of
252 // dispatching the event.
253 if (!widget->platformWidget())
254 return false;
255
256 BEGIN_BLOCK_OBJC_EXCEPTIONS
257
258 NSView *nodeView = widget->platformWidget();
259 ASSERT(nodeView);
260 ASSERT([nodeView superview]);
261 NSView *view = [nodeView hitTest:[[nodeView superview] convertPoint:currentEvent().locationInWindow fromView:nil]];
262 if (!view) {
263 // We probably hit the border of a RenderWidget
264 return true;
265 }
266
267 Page* page = m_frame.page();
268 if (!page)
269 return true;
270
271 if (page->chrome().client().firstResponder() != view) {
272 // Normally [NSWindow sendEvent:] handles setting the first responder.
273 // But in our case, the event was sent to the view representing the entire web page.
274 if ([view acceptsFirstResponder] && [view needsPanelToBecomeKey])
275 page->chrome().client().makeFirstResponder(view);
276 }
277
278 // We need to "defer loading" while tracking the mouse, because tearing down the
279 // page while an AppKit control is tracking the mouse can cause a crash.
280
281 // FIXME: In theory, WebCore now tolerates tear-down while tracking the
282 // mouse. We should confirm that, and then remove the deferrsLoading
283 // hack entirely.
284
285 bool wasDeferringLoading = page->defersLoading();
286 if (!wasDeferringLoading)
287 page->setDefersLoading(true);
288
289 ASSERT(!m_sendingEventToSubview);
290 m_sendingEventToSubview = true;
291
292 {
293 WidgetHierarchyUpdatesSuspensionScope suspendWidgetHierarchyUpdates;
294 [view mouseDown:currentEvent()];
295 }
296
297 m_sendingEventToSubview = false;
298
299 if (!wasDeferringLoading)
300 page->setDefersLoading(false);
301
302 // Remember which view we sent the event to, so we can direct the release event properly.
303 m_mouseDownView = view;
304 m_mouseDownWasInSubframe = false;
305
306 // Many AppKit widgets run their own event loops and consume events while the mouse is down.
307 // When they finish, currentEvent is the mouseUp that they exited on. We need to update
308 // the EventHandler state with this mouseUp, which we never saw.
309 // If this event isn't a mouseUp, we assume that the mouseUp will be coming later. There
310 // is a hole here if the widget consumes both the mouseUp and subsequent events.
311 if (lastEventIsMouseUp())
312 m_mousePressed = false;
313
314 END_BLOCK_OBJC_EXCEPTIONS
315
316 return true;
317}

319// Note that this does the same kind of check as [target isDescendantOf:superview].
320// There are two differences: This is a lot slower because it has to walk the whole
321// tree, and this works in cases where the target has already been deallocated.
322static bool findViewInSubviews(NSView *superview, NSView *target)
323{
324 BEGIN_BLOCK_OBJC_EXCEPTIONS
325 NSEnumerator *e = [[superview subviews] objectEnumerator];
326 NSView *subview;
327 while ((subview = [e nextObject])) {
328 if (subview == target || findViewInSubviews(subview, target)) {
329 return true;
330 }
331 }
332 END_BLOCK_OBJC_EXCEPTIONS
333
334 return false;
335}

337NSView *EventHandler::mouseDownViewIfStillGood()
338{
339 // Since we have no way of tracking the lifetime of m_mouseDownView, we have to assume that
340 // it could be deallocated already. We search for it in our subview tree; if we don't find
341 // it, we set it to nil.
342 NSView *mouseDownView = m_mouseDownView;
343 if (!mouseDownView) {
344 return nil;
345 }
346 FrameView* topFrameView = m_frame.view();
347 NSView *topView = topFrameView ? topFrameView->platformWidget() : nil;
348 if (!topView || !findViewInSubviews(topView, mouseDownView)) {
349 m_mouseDownView = nil;
350 return nil;
351 }
352 return mouseDownView;
353}

360bool EventHandler::eventLoopHandleMouseUp(const MouseEventWithHitTestResults&)
361{
362 NSView *view = mouseDownViewIfStillGood();
363 if (!view)
364 return false;
365
366 if (!m_mouseDownWasInSubframe) {
367 ASSERT(!m_sendingEventToSubview);
368 m_sendingEventToSubview = true;
369 BEGIN_BLOCK_OBJC_EXCEPTIONS
370 [view mouseUp:currentEvent()];
371 END_BLOCK_OBJC_EXCEPTIONS
372 m_sendingEventToSubview = false;
373 }
374
375 return true;
376}

378bool EventHandler::passSubframeEventToSubframe(MouseEventWithHitTestResults& event, Frame& subframe, HitTestResult* hitTestResult)
379{
380 BEGIN_BLOCK_OBJC_EXCEPTIONS
381
382 WebEventType currentEventType = currentEvent().type;
383 switch (currentEventType) {
384 case WebEventMouseMoved: {
385 // Since we're passing in currentNSEvent() here, we can call
386 // handleMouseMoveEvent() directly, since the save/restore of
387 // currentNSEvent() that mouseMoved() does would have no effect.
388 ASSERT(!m_sendingEventToSubview);
389 m_sendingEventToSubview = true;
390 subframe.eventHandler().handleMouseMoveEvent(currentPlatformMouseEvent(), hitTestResult);
391 m_sendingEventToSubview = false;
392 return true;
393 }
394 case WebEventMouseDown: {
395 auto* node = event.targetNode();
396 if (!node)
397 return false;
398 auto* renderer = node->renderer();
399 if (!is<RenderWidget>(renderer))
400 return false;
401 auto* widget = downcast<RenderWidget>(*renderer).widget();
402 if (!widget || !widget->isFrameView())
403 return false;
404 if (!passWidgetMouseDownEventToWidget(downcast<RenderWidget>(renderer)))
405 return false;
406 m_mouseDownWasInSubframe = true;
407 return true;
408 }
409 case WebEventMouseUp: {
410 if (!m_mouseDownWasInSubframe)
411 return false;
412 ASSERT(!m_sendingEventToSubview);
413 m_sendingEventToSubview = true;
414 subframe.eventHandler().handleMouseReleaseEvent(currentPlatformMouseEvent());
415 m_sendingEventToSubview = false;
416 return true;
417 }
418 case WebEventKeyDown:
419 case WebEventKeyUp:
420 case WebEventScrollWheel:
421 case WebEventTouchBegin:
422 case WebEventTouchCancel:
423 case WebEventTouchChange:
424 case WebEventTouchEnd:
425 return false;
426 }
427 END_BLOCK_OBJC_EXCEPTIONS
428
429 return false;
430}

432bool EventHandler::passWheelEventToWidget(const PlatformWheelEvent&, Widget& widget, OptionSet<WheelEventProcessingSteps>)
433{
434 BEGIN_BLOCK_OBJC_EXCEPTIONS
435
436 NSView* nodeView = widget.platformWidget();
437 if (!nodeView) {
438 // WK2 code path. No wheel events on iOS anyway.
439 return false;
440 }
441
442 if (currentEvent().type != WebEventScrollWheel || m_sendingEventToSubview)
443 return false;
444
445 ASSERT(nodeView);
446 ASSERT([nodeView superview]);
447 NSView *view = [nodeView hitTest:[[nodeView superview] convertPoint:currentEvent().locationInWindow fromView:nil]];
448 if (!view) {
449 // We probably hit the border of a RenderWidget
450 return false;
451 }
452
453 ASSERT(!m_sendingEventToSubview);
454 m_sendingEventToSubview = true;
455 [view scrollWheel:currentEvent()];
456 m_sendingEventToSubview = false;
457 return true;
458
459 END_BLOCK_OBJC_EXCEPTIONS
460 return false;
461}

500void EventHandler::mouseMoved(WebEvent *event)
501{
502 // Reject a mouse moved if the button is down - screws up tracking during autoscroll
503 // These happen because WebKit sometimes has to fake up moved events.
504 if (!m_frame.document() || !m_frame.view() || m_mousePressed || m_sendingEventToSubview)
505 return;
506
507 BEGIN_BLOCK_OBJC_EXCEPTIONS
508 auto& document = *m_frame.document();
509 // Ensure we start mouse move event dispatching on a clear tree.
510 document.updateStyleIfNeeded();
511 CurrentEventScope scope(event);
512 {
513 ContentChangeObserver::MouseMovedScope observingScope(document);
514 event.wasHandled = mouseMoved(currentPlatformMouseEvent());
515 // Run style recalc to be able to capture content changes as the result of the mouse move event.
516 document.updateStyleIfNeeded();
517 callOnMainThread([protectedFrame = makeRef(m_frame)] {
518 // This is called by WebKitLegacy only.
519 if (auto* document = protectedFrame->document())
520 document->contentChangeObserver().willNotProceedWithFixedObservationTimeWindow();
521 });
522 }
523
524 END_BLOCK_OBJC_EXCEPTIONS
525}

615static IntPoint adjustAutoscrollDestinationForInsetEdges(IntPoint autoscrollPoint, std::optional<IntPoint> initialAutoscrollPoint, FloatRect unobscuredRootViewRect)
616{
617 IntPoint resultPoint = autoscrollPoint;
618
619 const float edgeInset = 75;
620 const float maximumScrollingSpeed = 20;
621 const float insetDistanceThreshold = edgeInset / 2;
622
623 // FIXME: Ideally we would only inset on edges that touch the edge of the screen,
624 // like macOS, but we don't have enough information in WebCore to do that currently.
625 FloatRect insetUnobscuredRootViewRect = unobscuredRootViewRect;
626
627 if (initialAutoscrollPoint) {
628 IntSize autoscrollDelta = autoscrollPoint - *initialAutoscrollPoint;
629
630 // Inset edges in the direction of the autoscroll.
631 // Do not apply insets until you drag in the direction of the edge at least `insetDistanceThreshold`,
632 // to make it possible to select text that abuts the edge of `unobscuredRootViewRect` without causing
633 // unwanted autoscrolling.
634 if (autoscrollDelta.width() < insetDistanceThreshold)
635 insetUnobscuredRootViewRect.shiftXEdgeTo(insetUnobscuredRootViewRect.x() + std::min<float>(edgeInset, -autoscrollDelta.width() - insetDistanceThreshold));
636 else if (autoscrollDelta.width() > insetDistanceThreshold)
637 insetUnobscuredRootViewRect.shiftMaxXEdgeTo(insetUnobscuredRootViewRect.maxX() - std::min<float>(edgeInset, autoscrollDelta.width() - insetDistanceThreshold));
638
639 if (autoscrollDelta.height() < insetDistanceThreshold)
640 insetUnobscuredRootViewRect.shiftYEdgeTo(insetUnobscuredRootViewRect.y() + std::min<float>(edgeInset, -autoscrollDelta.height() - insetDistanceThreshold));
641 else if (autoscrollDelta.height() > insetDistanceThreshold)
642 insetUnobscuredRootViewRect.shiftMaxYEdgeTo(insetUnobscuredRootViewRect.maxY() - std::min<float>(edgeInset, autoscrollDelta.height() - insetDistanceThreshold));
643 }
644
645 // If the current autoscroll point is beyond the edge of the view (respecting insets), shift it outside
646 // of the view, so that autoscrolling will occur. The distance we move outside of the view is scaled from
647 // `edgeInset` to `maximumScrollingSpeed` so that the inset's contribution to the speed is independent of its size.
648 if (autoscrollPoint.x() < insetUnobscuredRootViewRect.x()) {
649 float distanceFromEdge = autoscrollPoint.x() - insetUnobscuredRootViewRect.x();
650 if (distanceFromEdge < 0)
651 resultPoint.setX(unobscuredRootViewRect.x() + ((distanceFromEdge / edgeInset) * maximumScrollingSpeed));
652 } else if (autoscrollPoint.x() >= insetUnobscuredRootViewRect.maxX()) {
653 float distanceFromEdge = autoscrollPoint.x() - insetUnobscuredRootViewRect.maxX();
654 if (distanceFromEdge > 0)
655 resultPoint.setX(unobscuredRootViewRect.maxX() + ((distanceFromEdge / edgeInset) * maximumScrollingSpeed));
656 }
657
658 if (autoscrollPoint.y() < insetUnobscuredRootViewRect.y()) {
659 float distanceFromEdge = autoscrollPoint.y() - insetUnobscuredRootViewRect.y();
660 if (distanceFromEdge < 0)
661 resultPoint.setY(unobscuredRootViewRect.y() + ((distanceFromEdge / edgeInset) * maximumScrollingSpeed));
662 } else if (autoscrollPoint.y() >= insetUnobscuredRootViewRect.maxY()) {
663 float distanceFromEdge = autoscrollPoint.y() - insetUnobscuredRootViewRect.maxY();
664 if (distanceFromEdge > 0)
665 resultPoint.setY(unobscuredRootViewRect.maxY() + ((distanceFromEdge / edgeInset) * maximumScrollingSpeed));
666 }
667
668 return resultPoint;
669}

671IntPoint EventHandler::targetPositionInWindowForSelectionAutoscroll() const
672{
673 Ref<Frame> protectedFrame(m_frame);
674
675 if (!m_frame.view())
676 return { };
677 auto& frameView = *m_frame.view();
678
679 // All work is done in "unscrolled" root view coordinates (as if delegatesScrolling were off),
680 // so that when the autoscrolling timer fires, it uses the new scroll position, so that it
681 // can keep scrolling without the client pushing a new contents-space target position via startSelectionAutoscroll.
682 auto scrollPosition = toIntSize(frameView.documentScrollPositionRelativeToViewOrigin());
683
684 FloatRect unobscuredContentRectInUnscrolledRootViewCoordinates = frameView.contentsToRootView(frameView.unobscuredContentRect());

702bool EventHandler::tryToBeginDragAtPoint(const IntPoint& clientPosition, const IntPoint&)
703{
704 Ref<Frame> protectedFrame(m_frame);
705
706 auto* document = m_frame.document();
707 if (!document)
708 return false;
709
710 SetForScope<bool> shouldAllowMouseDownToStartDrag { m_shouldAllowMouseDownToStartDrag, true };
711
712 document->updateLayoutIgnorePendingStylesheets();
713
714 FloatPoint adjustedClientPositionAsFloatPoint(clientPosition);
715 protectedFrame->nodeRespondingToClickEvents(clientPosition, adjustedClientPositionAsFloatPoint);
716 IntPoint adjustedClientPosition = roundedIntPoint(adjustedClientPositionAsFloatPoint);
717 IntPoint adjustedGlobalPosition = protectedFrame->view()->windowToContents(adjustedClientPosition);
718
719 PlatformMouseEvent syntheticMousePressEvent(adjustedClientPosition, adjustedGlobalPosition, LeftButton, PlatformEvent::MousePressed, 1, false, false, false, false, WallTime::now(), 0, NoTap);
720 PlatformMouseEvent syntheticMouseMoveEvent(adjustedClientPosition, adjustedGlobalPosition, LeftButton, PlatformEvent::MouseMoved, 0, false, false, false, false, WallTime::now(), 0, NoTap);
721
722 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::Active, HitTestRequest::Type::DisallowUserAgentShadowContent };
723 auto documentPoint = protectedFrame->view() ? protectedFrame->view()->windowToContents(syntheticMouseMoveEvent.position()) : syntheticMouseMoveEvent.position();
724 auto hitTestedMouseEvent = document->prepareMouseEvent(hitType, documentPoint, syntheticMouseMoveEvent);
725
726 auto subframe = subframeForHitTestResult(hitTestedMouseEvent);
727 if (subframe && subframe->eventHandler().tryToBeginDragAtPoint(adjustedClientPosition, adjustedGlobalPosition))
728 return true;
729
730 if (!eventMayStartDrag(syntheticMousePressEvent))
731 return false;
732
733 handleMousePressEvent(syntheticMousePressEvent);
734 bool handledDrag = m_mouseDownMayStartDrag && handleMouseDraggedEvent(hitTestedMouseEvent, DontCheckDragHysteresis);
735 // Reset this bit to prevent autoscrolling from updating the selection with the last mouse location.
736 m_mouseDownMayStartSelect = false;
737 // Reset this bit to ensure that WebChromeClientIOS::observedContentChange() is called by EventHandler::mousePressed()
738 // when we would process the next tap after a drag interaction.
739 m_mousePressed = false;
740 return handledDrag;
741}
