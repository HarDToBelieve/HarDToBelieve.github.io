34void Chrome::focusNSView(NSView* view)
35{
36 BEGIN_BLOCK_OBJC_EXCEPTIONS
37
38 // Handle the WK2 case where there is no view passed in.
39 if (!view) {
40 client().makeFirstResponder();
41 return;
42 }
43
44 NSResponder *firstResponder = client().firstResponder();
45 if (firstResponder == view)
46 return;
47
48 if (![view window] || ![view superview] || ![view acceptsFirstResponder])
49 return;
50
51 client().makeFirstResponder(view);
52
53 // Setting focus can actually cause a style change which might
54 // remove the view from its superview while it's being made
55 // first responder. This confuses AppKit so we must restore
56 // the old first responder.
57 if (![view superview])
58 client().makeFirstResponder(firstResponder);
59
60 END_BLOCK_OBJC_EXCEPTIONS
61}
