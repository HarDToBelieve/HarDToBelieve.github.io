48void Page::platformInitialize()
49{
50#if PLATFORM(IOS_FAMILY)
51 addSchedulePair(SchedulePair::create(WebThreadNSRunLoop(), kCFRunLoopCommonModes));
52#else
53 addSchedulePair(SchedulePair::create([[NSRunLoop currentRunLoop] getCFRunLoop], kCFRunLoopCommonModes));
54#endif
55
56 static std::once_flag onceFlag;
57 std::call_once(onceFlag, [] {
58#if ENABLE(TREE_DEBUGGING)
59 PAL::registerNotifyCallback("com.apple.WebKit.showRenderTree", printRenderTreeForLiveDocuments);
60 PAL::registerNotifyCallback("com.apple.WebKit.showLayerTree", printLayerTreeForLiveDocuments);
61 PAL::registerNotifyCallback("com.apple.WebKit.showGraphicsLayerTree", printGraphicsLayerTreeForLiveDocuments);
62#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
63 PAL::registerNotifyCallback("com.apple.WebKit.showLayoutTree", Layout::printLayoutTreeForLiveDocuments);
64#endif
65#endif // ENABLE(TREE_DEBUGGING)
66
67 PAL::registerNotifyCallback("com.apple.WebKit.showAllDocuments", [] {
68 unsigned numPages = 0;
69 Page::forEachPage([&numPages](Page&) {
70 ++numPages;
71 });
72
73 WTFLogAlways("%u live pages:", numPages);
74
75 Page::forEachPage([](Page& page) {
76 const auto* mainFrameDocument = page.mainFrame().document();
77 WTFLogAlways("Page %p with main document %p %s", &page, mainFrameDocument, mainFrameDocument ? mainFrameDocument->url().string().utf8().data() : "");
78 });
79
80 WTFLogAlways("%u live documents:", Document::allDocuments().size());
81 for (const auto* document : Document::allDocuments()) {
82 const char* documentType = is<SVGDocument>(document) ? "SVGDocument" : "Document";
83 WTFLogAlways("%s %p %llu (refCount %d, referencingNodeCount %d) %s", documentType, document, document->identifier().toUInt64(), document->refCount(), document->referencingNodeCount(), document->url().string().utf8().data());
84 }
85 });
86 });
87}

89void Page::addSchedulePair(Ref<SchedulePair>&& pair)
90{
91 if (!m_scheduledRunLoopPairs)
92 m_scheduledRunLoopPairs = makeUnique<SchedulePairHashSet>();
93 m_scheduledRunLoopPairs->add(pair.ptr());
94
95 for (Frame* frame = &m_mainFrame.get(); frame; frame = frame->tree().traverseNext()) {
96 if (DocumentLoader* documentLoader = frame->loader().documentLoader())
97 documentLoader->schedule(pair);
98 if (DocumentLoader* documentLoader = frame->loader().provisionalDocumentLoader())
99 documentLoader->schedule(pair);
100 }
101
102 // FIXME: make SharedTimerMac use these SchedulePairs.
103}
104
105void Page::removeSchedulePair(Ref<SchedulePair>&& pair)
106{
107 ASSERT(m_scheduledRunLoopPairs);
108 if (!m_scheduledRunLoopPairs)
109 return;
110
111 m_scheduledRunLoopPairs->remove(pair.ptr());
112
113 for (Frame* frame = &m_mainFrame.get(); frame; frame = frame->tree().traverseNext()) {
114 if (DocumentLoader* documentLoader = frame->loader().documentLoader())
115 documentLoader->unschedule(pair);
116 if (DocumentLoader* documentLoader = frame->loader().provisionalDocumentLoader())
117 documentLoader->unschedule(pair);
118 }
119}
