80static const uint8_t AlignmentNone = 0;
81static const uint8_t AlignmentLeft = 1 << 0;
82static const uint8_t AlignmentRight = 1 << 1;
83
84static void expandForGap(Vector<LayoutRect>& rects, uint8_t* alignments, const GapRects& gap)
85{
86 if (!gap.left().isEmpty()) {
87 LayoutUnit leftEdge = gap.left().x();
88 for (unsigned i = 0; i < rects.size(); ++i) {
89 if (alignments[i] & AlignmentLeft)
90 rects[i].shiftXEdgeTo(leftEdge);
91 }
92 }
93
94 if (!gap.right().isEmpty()) {
95 LayoutUnit rightEdge = gap.right().maxX();
96 for (unsigned i = 0; i < rects.size(); ++i) {
97 if (alignments[i] & AlignmentRight)
98 rects[i].shiftMaxXEdgeTo(rightEdge);
99 }
100 }
101}

103static inline void stitchRects(Vector<LayoutRect>& rects)
104{
105 if (rects.size() <= 1)
106 return;
107
108 Vector<LayoutRect> newRects;
109
110 // FIXME: Need to support vertical layout.
111 // First stitch together all the rects on the first line of the selection.
112 size_t indexFromStart = 0;
113 LayoutUnit firstTop = rects[indexFromStart].y();
114 LayoutRect& currentRect = rects[indexFromStart];
115 while (indexFromStart < rects.size() && rects[indexFromStart].y() == firstTop)
116 currentRect.unite(rects[indexFromStart++]);
117
118 newRects.append(currentRect);
119 if (indexFromStart == rects.size()) {
120 // All the rects are on one line. There is nothing else to do.
121 rects.swap(newRects);
122 return;
123 }
124
125 // Next stitch together all the rects on the last line of the selection.
126 size_t indexFromEnd = rects.size() - 1;
127 LayoutUnit lastTop = rects[indexFromEnd].y();
128 LayoutRect lastRect = rects[indexFromEnd];
129 while (indexFromEnd >= indexFromStart && rects[indexFromEnd].y() == lastTop)
130 lastRect.unite(rects[indexFromEnd--]);
131
132 // indexFromStart is the index of the first rectangle on the second line.
133 // indexFromEnd is the index of the last rectangle on the second to the last line.
134 // if they are equal, there is one additional rectangle for the line in the middle.
135 if (indexFromEnd == indexFromStart)
136 newRects.append(rects[indexFromStart]);
137
138 if (indexFromEnd <= indexFromStart) {
139 // There are no more rects to stitch. Just append the last line.
140 newRects.append(lastRect);
141 rects.swap(newRects);
142 return;
143 }
144
145 // Stitch together all the rects after the first line until the second to the last included.
146 currentRect = rects[indexFromStart];
147 while (indexFromStart != indexFromEnd)
148 currentRect.unite(rects[++indexFromStart]);
149
150 newRects.append(currentRect);
151 newRects.append(lastRect);
152
153 rects.swap(newRects);
154}

156static void compactRectsWithGapRects(Vector<LayoutRect>& rects, const Vector<GapRects>& gapRects)
157{
158 stitchRects(rects);
159
160 // FIXME: The following alignments are correct for LTR text.
161 // We should also account for RTL.
162 uint8_t alignments[3];
163 if (rects.size() == 1) {
164 alignments[0] = AlignmentLeft | AlignmentRight;
165 alignments[1] = AlignmentNone;
166 alignments[2] = AlignmentNone;
167 } else if (rects.size() == 2) {
168 alignments[0] = AlignmentRight;
169 alignments[1] = AlignmentLeft;
170 alignments[2] = AlignmentNone;
171 } else {
172 alignments[0] = AlignmentRight;
173 alignments[1] = AlignmentLeft | AlignmentRight;
174 alignments[2] = AlignmentLeft;
175 }
176
177 // Account for each GapRects by extending the edge of certain LayoutRects to meet the gap.
178 for (auto& gap : gapRects)
179 expandForGap(rects, alignments, gap);
180
181 // If we have 3 rects we might need one final GapRects to align the edges.
182 if (rects.size() == 3) {
183 LayoutRect left;
184 LayoutRect right;
185 for (unsigned i = 0; i < 3; ++i) {
186 if (alignments[i] & AlignmentLeft) {
187 if (left.isEmpty())
188 left = rects[i];
189 else if (rects[i].x() < left.x())
190 left = rects[i];
191 }
192 if (alignments[i] & AlignmentRight) {
193 if (right.isEmpty())
194 right = rects[i];
195 else if ((rects[i].x() + rects[i].width()) > (right.x() + right.width()))
196 right = rects[i];
197 }
198 }
199
200 if (!left.isEmpty() || !right.isEmpty()) {
201 GapRects gap;
202 gap.uniteLeft(left);
203 gap.uniteRight(right);
204 expandForGap(rects, alignments, gap);
205 }
206 }

209void ServicesOverlayController::selectionRectsDidChange(const Vector<LayoutRect>& rects, const Vector<GapRects>& gapRects, bool isTextOnly)
210{
211 m_currentSelectionRects = rects;
212 m_isTextOnly = isTextOnly;
213
214 m_lastSelectionChangeTime = MonotonicTime::now();
215
216 compactRectsWithGapRects(m_currentSelectionRects, gapRects);
217
218 // DataDetectors needs these reversed in order to place the arrow in the right location.
219 m_currentSelectionRects.reverse();
220
221 LOG(Services, "ServicesOverlayController - Selection rects changed - Now have %lu\n", rects.size());
222 invalidateHighlightsOfType(DataDetectorHighlight::Type::Selection);
223}

240void ServicesOverlayController::buildPotentialHighlightsIfNeeded()
241{
242 if (m_dirtyHighlightTypes.isEmpty())
243 return;
244
245 if (m_dirtyHighlightTypes.contains(DataDetectorHighlight::Type::TelephoneNumber))
246 buildPhoneNumberHighlights();
247
248 if (m_dirtyHighlightTypes.contains(DataDetectorHighlight::Type::Selection))
249 buildSelectionHighlight();
250
251 m_dirtyHighlightTypes = { };
252
253 if (m_potentialHighlights.isEmpty()) {
254 if (m_servicesOverlay)
255 m_page.pageOverlayController().uninstallPageOverlay(*m_servicesOverlay, PageOverlay::FadeMode::DoNotFade);
256 return;
257 }
258
259 if (telephoneNumberRangesForFocusedFrame().isEmpty() && !hasRelevantSelectionServices())
260 return;
261
262 createOverlayIfNeeded();
263
264 bool mouseIsOverButton;
265 determineActiveHighlight(mouseIsOverButton);
266}

268bool ServicesOverlayController::mouseIsOverHighlight(DataDetectorHighlight& highlight, bool& mouseIsOverButton) const
269{
270 if (!PAL::isDataDetectorsFrameworkAvailable())
271 return false;
272
273 Boolean onButton;
274 bool hovered = PAL::softLink_DataDetectors_DDHighlightPointIsOnHighlight(highlight.highlight(), (CGPoint)m_mousePosition, &onButton);
275 mouseIsOverButton = onButton;
276 return hovered;

279Seconds ServicesOverlayController::remainingTimeUntilHighlightShouldBeShown(DataDetectorHighlight* highlight) const
280{
281 if (!highlight)
282 return 0_s;
283
284 Seconds minimumTimeUntilHighlightShouldBeShown = 200_ms;
285 if (m_page.focusController().focusedOrMainFrame().selection().selection().isContentEditable())
286 minimumTimeUntilHighlightShouldBeShown = 1_s;
287
288 bool mousePressed = mainFrame().eventHandler().mousePressed();
289
290 // Highlight hysteresis is only for selection services, because telephone number highlights are already much more stable
291 // by virtue of being expanded to include the entire telephone number. However, we will still avoid highlighting
292 // telephone numbers while the mouse is down.
293 if (highlight->type() == DataDetectorHighlight::Type::TelephoneNumber)
294 return mousePressed ? minimumTimeUntilHighlightShouldBeShown : 0_s;
295
296 MonotonicTime now = MonotonicTime::now();
297 Seconds timeSinceLastSelectionChange = now - m_lastSelectionChangeTime;
298 Seconds timeSinceHighlightBecameActive = now - m_nextActiveHighlightChangeTime;
299 Seconds timeSinceLastMouseUp = mousePressed ? 0_s : now - m_lastMouseUpTime;
300
301 return minimumTimeUntilHighlightShouldBeShown - std::min(std::min(timeSinceLastSelectionChange, timeSinceHighlightBecameActive), timeSinceLastMouseUp);
302}

336void ServicesOverlayController::buildPhoneNumberHighlights()
337{
338 Vector<SimpleRange> phoneNumberRanges;
339 for (Frame* frame = &mainFrame(); frame; frame = frame->tree().traverseNext())
340 phoneNumberRanges.appendVector(frame->editor().detectedTelephoneNumberRanges());
341
342 if (phoneNumberRanges.isEmpty()) {
343 removeAllPotentialHighlightsOfType(DataDetectorHighlight::Type::TelephoneNumber);
344 return;
345 }
346
347 if (!PAL::isDataDetectorsFrameworkAvailable())
348 return;
349
350 HashSet<RefPtr<DataDetectorHighlight>> newPotentialHighlights;
351
352 FrameView& mainFrameView = *mainFrame().view();
353
354 for (auto& range : phoneNumberRanges) {
355 // FIXME: This makes a big rect if the range extends from the end of one line to the start of the next. Handle that case better?
356 auto rect = enclosingIntRect(unitedBoundingBoxes(RenderObject::absoluteTextQuads(range)));
357
358 // Convert to the main document's coordinate space.
359 // FIXME: It's a little crazy to call contentsToWindow and then windowToContents in order to get the right coordinate space.
360 // We should consider adding conversion functions to ScrollView for contentsToDocument(). Right now, contentsToRootView() is
361 // not equivalent to what we need when you have a topContentInset or a header banner.
362 auto* viewForRange = range.start.document().view();
363 if (!viewForRange)
364 continue;
365
366 rect.setLocation(mainFrameView.windowToContents(viewForRange->contentsToWindow(rect.location())));
367
368 CGRect cgRect = rect;
369#if HAVE(DD_HIGHLIGHT_CREATE_WITH_SCALE)
370 auto ddHighlight = adoptCF(PAL::softLink_DataDetectors_DDHighlightCreateWithRectsInVisibleRectWithStyleScaleAndDirection(nullptr, &cgRect, 1, mainFrameView.visibleContentRect(), DDHighlightStyleBubbleStandard | DDHighlightStyleStandardIconArrow, YES, NSWritingDirectionNatural, NO, YES, 0));
371#else
372 auto ddHighlight = adoptCF(PAL::softLink_DataDetectors_DDHighlightCreateWithRectsInVisibleRectWithStyleAndDirection(nullptr, &cgRect, 1, mainFrameView.visibleContentRect(), DDHighlightStyleBubbleStandard | DDHighlightStyleStandardIconArrow, YES, NSWritingDirectionNatural, NO, YES));
373#endif
374 auto highlight = DataDetectorHighlight::createForTelephoneNumber(m_page, *this, WTFMove(ddHighlight), WTFMove(range));
375 m_highlights.add(highlight.get());
376 newPotentialHighlights.add(WTFMove(highlight));
377 }
378
379 replaceHighlightsOfTypePreservingEquivalentHighlights(newPotentialHighlights, DataDetectorHighlight::Type::TelephoneNumber);
380}

382void ServicesOverlayController::buildSelectionHighlight()
383{
384 if (m_currentSelectionRects.isEmpty()) {
385 removeAllPotentialHighlightsOfType(DataDetectorHighlight::Type::Selection);
386 return;
387 }
388
389 if (!PAL::isDataDetectorsFrameworkAvailable())
390 return;
391
392 HashSet<RefPtr<DataDetectorHighlight>> newPotentialHighlights;
393
394 Vector<CGRect> cgRects;
395 cgRects.reserveCapacity(m_currentSelectionRects.size());
396
397 if (auto selectionRange = m_page.selection().firstRange()) {
398 FrameView* mainFrameView = mainFrame().view();
399 if (!mainFrameView)
400 return;
401
402 auto viewForRange = makeRefPtr(selectionRange->start.document().view());
403 if (!viewForRange)
404 return;
405
406 for (auto& rect : m_currentSelectionRects) {
407 IntRect currentRect = snappedIntRect(rect);
408 currentRect.setLocation(mainFrameView->windowToContents(viewForRange->contentsToWindow(currentRect.location())));
409 cgRects.append(currentRect);
410 }
411
412 if (!cgRects.isEmpty()) {
413 CGRect visibleRect = mainFrameView->visibleContentRect();
414#if HAVE(DD_HIGHLIGHT_CREATE_WITH_SCALE)
415 auto ddHighlight = adoptCF(PAL::softLink_DataDetectors_DDHighlightCreateWithRectsInVisibleRectWithStyleScaleAndDirection(nullptr, cgRects.begin(), cgRects.size(), visibleRect, DDHighlightStyleBubbleNone | DDHighlightStyleStandardIconArrow | DDHighlightStyleButtonShowAlways, YES, NSWritingDirectionNatural, NO, YES, 0));
416#else
417 auto ddHighlight = adoptCF(PAL::softLink_DataDetectors_DDHighlightCreateWithRectsInVisibleRectWithStyleAndDirection(nullptr, cgRects.begin(), cgRects.size(), visibleRect, DDHighlightStyleBubbleNone | DDHighlightStyleStandardIconArrow | DDHighlightStyleButtonShowAlways, YES, NSWritingDirectionNatural, NO, YES));
418#endif
419 auto highlight = DataDetectorHighlight::createForSelection(m_page, *this, WTFMove(ddHighlight), WTFMove(*selectionRange));
420 m_highlights.add(highlight.get());
421 newPotentialHighlights.add(WTFMove(highlight));
422 }
423 }
424
425 replaceHighlightsOfTypePreservingEquivalentHighlights(newPotentialHighlights, DataDetectorHighlight::Type::Selection);
426}

428void ServicesOverlayController::replaceHighlightsOfTypePreservingEquivalentHighlights(HashSet<RefPtr<DataDetectorHighlight>>& newPotentialHighlights, DataDetectorHighlight::Type type)
429{
430 // If any old Highlights are equivalent (by Range) to a new Highlight, reuse the old
431 // one so that any metadata is retained.
432 HashSet<RefPtr<DataDetectorHighlight>> reusedPotentialHighlights;
433
434 for (auto& oldHighlight : m_potentialHighlights) {
435 if (oldHighlight->type() != type)
436 continue;
437
438 for (auto& newHighlight : newPotentialHighlights) {
439 if (areEquivalent(oldHighlight.get(), newHighlight.get())) {
440 oldHighlight->setHighlight(newHighlight->highlight());
441
442 reusedPotentialHighlights.add(oldHighlight);
443 newPotentialHighlights.remove(newHighlight);
444
445 break;
446 }
447 }
448 }
449
450 removeAllPotentialHighlightsOfType(type);
451
452 m_potentialHighlights.add(newPotentialHighlights.begin(), newPotentialHighlights.end());
453 m_potentialHighlights.add(reusedPotentialHighlights.begin(), reusedPotentialHighlights.end());
454}

479DataDetectorHighlight* ServicesOverlayController::findTelephoneNumberHighlightContainingSelectionHighlight(DataDetectorHighlight& selectionHighlight)
480{
481 if (selectionHighlight.type() != DataDetectorHighlight::Type::Selection)
482 return nullptr;
483
484 auto selectionRange = m_page.selection().toNormalizedRange();
485 if (!selectionRange)
486 return nullptr;
487
488 for (auto& highlight : m_potentialHighlights) {
489 if (highlight->type() == DataDetectorHighlight::Type::TelephoneNumber && contains<ComposedTree>(highlight->range(), *selectionRange))
490 return highlight.get();
491 }
492
493 return nullptr;
494}

496void ServicesOverlayController::determineActiveHighlight(bool& mouseIsOverActiveHighlightButton)
497{
498 buildPotentialHighlightsIfNeeded();
499
500 mouseIsOverActiveHighlightButton = false;
501
502 RefPtr<DataDetectorHighlight> newActiveHighlight;
503
504 for (auto& highlight : m_potentialHighlights) {
505 if (highlight->type() == DataDetectorHighlight::Type::Selection) {
506 // If we've already found a new active highlight, and it's
507 // a telephone number highlight, prefer that over this selection highlight.
508 if (newActiveHighlight && newActiveHighlight->type() == DataDetectorHighlight::Type::TelephoneNumber)
509 continue;
510
511 // If this highlight has no compatible services, it can't be active.
512 if (!hasRelevantSelectionServices())
513 continue;
514 }
515
516 // If this highlight isn't hovered, it can't be active.
517 bool mouseIsOverButton;
518 if (!mouseIsOverHighlight(*highlight, mouseIsOverButton))
519 continue;
520
521 newActiveHighlight = highlight;
522 mouseIsOverActiveHighlightButton = mouseIsOverButton;
523 }
524
525 // If our new active highlight is a selection highlight that is completely contained
526 // by one of the phone number highlights, we'll make the phone number highlight active even if it's not hovered.
527 if (newActiveHighlight && newActiveHighlight->type() == DataDetectorHighlight::Type::Selection) {
528 if (DataDetectorHighlight* containedTelephoneNumberHighlight = findTelephoneNumberHighlightContainingSelectionHighlight(*newActiveHighlight)) {
529 newActiveHighlight = containedTelephoneNumberHighlight;
530
531 // We will always initially choose the telephone number highlight over the selection highlight if the
532 // mouse is over the telephone number highlight's button, so we know that it's not hovered if we got here.
533 mouseIsOverActiveHighlightButton = false;
534 }
535 }
536
537 if (!areEquivalent(m_activeHighlight.get(), newActiveHighlight.get())) {
538 // When transitioning to a new highlight, we might end up in determineActiveHighlight multiple times
539 // before the new highlight actually becomes active. Keep track of the last next-but-not-yet-active
540 // highlight, and only reset the active highlight hysteresis when that changes.
541 if (m_nextActiveHighlight != newActiveHighlight) {
542 m_nextActiveHighlight = newActiveHighlight;
543 m_nextActiveHighlightChangeTime = MonotonicTime::now();
544 }
545
546 m_currentMouseDownOnButtonHighlight = nullptr;
547
548 if (m_activeHighlight) {
549 m_activeHighlight->fadeOut();
550 m_activeHighlight = nullptr;
551 }
552
553 auto remainingTimeUntilHighlightShouldBeShown = this->remainingTimeUntilHighlightShouldBeShown(newActiveHighlight.get());
554 if (remainingTimeUntilHighlightShouldBeShown > 0_s) {
555 m_determineActiveHighlightTimer.startOneShot(remainingTimeUntilHighlightShouldBeShown);
556 return;
557 }
558
559 m_activeHighlight = WTFMove(m_nextActiveHighlight);
560
561 if (m_activeHighlight) {
562 Ref<GraphicsLayer> highlightLayer = m_activeHighlight->layer();
563 m_servicesOverlay->layer().addChild(WTFMove(highlightLayer));
564 m_activeHighlight->fadeIn();
565 }
566 }

569bool ServicesOverlayController::mouseEvent(PageOverlay&, const PlatformMouseEvent& event)
570{
571 m_mousePosition = mainFrame().view()->windowToContents(event.position());
572
573 bool mouseIsOverActiveHighlightButton = false;
574 determineActiveHighlight(mouseIsOverActiveHighlightButton);
575
576 // Cancel the potential click if any button other than the left button changes state, and ignore the event.
577 if (event.button() != MouseButton::LeftButton) {
578 m_currentMouseDownOnButtonHighlight = nullptr;
579 return false;
580 }
581
582 // If the mouse lifted while still over the highlight button that it went down on, then that is a click.
583 if (event.type() == PlatformEvent::MouseReleased) {
584 auto mouseDownHighlight = m_currentMouseDownOnButtonHighlight.copyRef();
585 m_currentMouseDownOnButtonHighlight = nullptr;
586
587 m_lastMouseUpTime = MonotonicTime::now();
588
589 if (mouseIsOverActiveHighlightButton && mouseDownHighlight) {
590 handleClick(m_mousePosition, *mouseDownHighlight);
591 return true;
592 }
593
594 return false;
595 }
596
597 // If the mouse moved outside of the button tracking a potential click, stop tracking the click.
598 if (event.type() == PlatformEvent::MouseMoved) {
599 if (m_currentMouseDownOnButtonHighlight && mouseIsOverActiveHighlightButton)
600 return true;
601
602 m_currentMouseDownOnButtonHighlight = nullptr;
603 return false;
604 }
605
606 // If the mouse went down over the active highlight's button, track this as a potential click.
607 if (event.type() == PlatformEvent::MousePressed) {
608 if (m_activeHighlight && mouseIsOverActiveHighlightButton) {
609 m_currentMouseDownOnButtonHighlight = m_activeHighlight;
610 return true;
611 }
612
613 return false;
614 }
615
616 return false;
617}

632void ServicesOverlayController::handleClick(const IntPoint& clickPoint, DataDetectorHighlight& highlight)
633{
634 FrameView* frameView = mainFrame().view();
635 if (!frameView)
636 return;
637
638 IntPoint windowPoint = frameView->contentsToWindow(clickPoint);
639
640 if (highlight.type() == DataDetectorHighlight::Type::Selection) {
641 auto telephoneNumberRanges = telephoneNumberRangesForFocusedFrame();
642 Vector<String> selectedTelephoneNumbers;
643 selectedTelephoneNumbers.reserveCapacity(telephoneNumberRanges.size());
644 for (auto& range : telephoneNumberRanges)
645 selectedTelephoneNumbers.append(plainText(range));
646
647 m_page.chrome().client().handleSelectionServiceClick(m_page.focusController().focusedOrMainFrame().selection(), selectedTelephoneNumbers, windowPoint);
648 } else if (highlight.type() == DataDetectorHighlight::Type::TelephoneNumber)
649 m_page.chrome().client().handleTelephoneNumberClick(plainText(highlight.range()), windowPoint);
650}
