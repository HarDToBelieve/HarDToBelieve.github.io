45TextIndicatorWindow::TextIndicatorWindow(NSView *targetView)

56void TextIndicatorWindow::setAnimationProgress(float progress)
57{
58 if (!m_textIndicator)
59 return;
60
61 [m_textIndicatorLayer setAnimationProgress:progress];
62}

64void TextIndicatorWindow::clearTextIndicator(WebCore::TextIndicatorDismissalAnimation animation)
65{
66 RefPtr<TextIndicator> textIndicator = WTFMove(m_textIndicator);
67
68 if ([m_textIndicatorLayer isFadingOut])
69 return;
70
71 if (textIndicator && [m_textIndicatorLayer indicatorWantsManualAnimation:*textIndicator] && [m_textIndicatorLayer hasCompletedAnimation] && animation == WebCore::TextIndicatorDismissalAnimation::FadeOut) {
72 startFadeOut();
73 return;
74 }
75
76 closeWindow();
77}

79void TextIndicatorWindow::setTextIndicator(Ref<TextIndicator> textIndicator, CGRect textBoundingRectInScreenCoordinates, TextIndicatorLifetime lifetime)
80{
81 if (m_textIndicator == textIndicator.ptr())
82 return;
83
84 closeWindow();
85
86 m_textIndicator = textIndicator.ptr();
87
88 CGFloat horizontalMargin = dropShadowBlurRadius * 2 + TextIndicator::defaultHorizontalMargin;
89 CGFloat verticalMargin = dropShadowBlurRadius * 2 + TextIndicator::defaultVerticalMargin;
90
91 if ([m_textIndicatorLayer indicatorWantsBounce:*m_textIndicator]) {
92 horizontalMargin = std::max(horizontalMargin, textBoundingRectInScreenCoordinates.size.width * (WebCore::midBounceScale - 1) + horizontalMargin);
93 verticalMargin = std::max(verticalMargin, textBoundingRectInScreenCoordinates.size.height * (WebCore::midBounceScale - 1) + verticalMargin);
94 }
95
96 horizontalMargin = CGCeiling(horizontalMargin);
97 verticalMargin = CGCeiling(verticalMargin);
98
99 CGRect contentRect = CGRectInset(textBoundingRectInScreenCoordinates, -horizontalMargin, -verticalMargin);
100 NSRect windowContentRect = [NSWindow contentRectForFrameRect:NSRectFromCGRect(contentRect) styleMask:NSWindowStyleMaskBorderless];
101 NSRect integralWindowContentRect = NSIntegralRect(windowContentRect);
102 NSPoint fractionalTextOffset = NSMakePoint(windowContentRect.origin.x - integralWindowContentRect.origin.x, windowContentRect.origin.y - integralWindowContentRect.origin.y);
103 m_textIndicatorWindow = adoptNS([[NSWindow alloc] initWithContentRect:integralWindowContentRect styleMask:NSWindowStyleMaskBorderless backing:NSBackingStoreBuffered defer:NO]);
104 [m_textIndicatorWindow setBackgroundColor:[NSColor clearColor]];
105 [m_textIndicatorWindow setOpaque:NO];
106 [m_textIndicatorWindow setIgnoresMouseEvents:YES];
107
108 NSRect frame = NSMakeRect(0, 0, [m_textIndicatorWindow frame].size.width, [m_textIndicatorWindow frame].size.height);
109 m_textIndicatorLayer = adoptNS([[WebTextIndicatorLayer alloc] initWithFrame:frame
110 textIndicator:*m_textIndicator margin:NSMakeSize(horizontalMargin, verticalMargin) offset:fractionalTextOffset]);
111 m_textIndicatorView = adoptNS([[NSView alloc] initWithFrame:frame]);
112 [m_textIndicatorView setLayer:m_textIndicatorLayer.get()];
113 [m_textIndicatorView setWantsLayer:YES];
114 [m_textIndicatorWindow setContentView:m_textIndicatorView.get()];
115
116 [[m_targetView window] addChildWindow:m_textIndicatorWindow.get() ordered:NSWindowAbove];
117 [m_textIndicatorWindow setReleasedWhenClosed:NO];
118
119 if (m_textIndicator->presentationTransition() != TextIndicatorPresentationTransition::None)
120 [m_textIndicatorLayer present];
121
122 if (lifetime == TextIndicatorLifetime::Temporary)
123 m_temporaryTextIndicatorTimer.startOneShot(WebCore::timeBeforeFadeStarts);
124}

126void TextIndicatorWindow::closeWindow()
127{
128 if (!m_textIndicatorWindow)
129 return;
130
131 if ([m_textIndicatorLayer isFadingOut])
132 return;
133
134 m_temporaryTextIndicatorTimer.stop();
135
136 [[m_textIndicatorWindow parentWindow] removeChildWindow:m_textIndicatorWindow.get()];
137 [m_textIndicatorWindow close];
138 m_textIndicatorWindow = nullptr;
139}
140
141void TextIndicatorWindow::startFadeOut()
142{
143 [m_textIndicatorLayer setFadingOut:YES];
144 RetainPtr<NSWindow> indicatorWindow = m_textIndicatorWindow;
145 [m_textIndicatorLayer hideWithCompletionHandler:[indicatorWindow] {
146 [[indicatorWindow parentWindow] removeChildWindow:indicatorWindow.get()];
147 [indicatorWindow close];
148 }];
149}
