37class TextStream;

40namespace WebCore {
41
42class Element;
43class Frame;
44class PlatformWheelEvent;
45class ScrollableArea;
46
47class ScrollLatchingController {
48 WTF_MAKE_FAST_ALLOCATED;
49public:
50 ScrollLatchingController();
51 ~ScrollLatchingController();
52
53 void clear();
54
55 void receivedWheelEvent(const PlatformWheelEvent&);
56 FloatSize cumulativeEventDelta() const { return m_cumulativeEventDelta; }
57
58 // Frame containing latched scroller (may be the frame or some sub-scroller).
59 Frame* latchedFrame() const;
60
61 // Returns true if no frame is latched, or latching is in the given frame (in which case latchedScroller will be non-null).
62 bool latchingAllowsScrollingInFrame(const Frame&, WeakPtr<ScrollableArea>& latchedScroller) const;
63
64 void updateAndFetchLatchingStateForFrame(Frame&, const PlatformWheelEvent&, RefPtr<Element>& latchedElement, WeakPtr<ScrollableArea>&, bool& isOverWidget);
65
66 void removeLatchingStateForTarget(const Element&);
67 void removeLatchingStateForFrame(const Frame&);
68
69 void dump(WTF::TextStream&) const;
70
71private:
72 struct FrameState {
73 WeakPtr<Element> wheelEventElement;
74 WeakPtr<ScrollableArea> scrollableArea;
75 Frame* frame { nullptr };
76 bool isOverWidget { false };
77 };
78
79 void clearOrScheduleClearIfNeeded(const PlatformWheelEvent&);
80 void clearTimerFired();
81
82 bool hasStateForFrame(const Frame&) const;
83 FrameState* stateForFrame(const Frame&);
84 const FrameState* stateForFrame(const Frame&) const;
85
86 bool shouldLatchToScrollableArea(const Frame&, ScrollableArea*, FloatSize) const;
87
88 FloatSize m_cumulativeEventDelta;
89 Vector<FrameState> m_frameStateStack;
90 Timer m_clearLatchingStateTimer;
91};
92
93WTF::TextStream& operator<<(WTF::TextStream&, const ScrollLatchingController&);

95} // namespace WebCore
