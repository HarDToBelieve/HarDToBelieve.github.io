47ScrollingStateFrameScrollingNode::ScrollingStateFrameScrollingNode(const ScrollingStateFrameScrollingNode& stateNode, ScrollingStateTree& adoptiveTree)
48 : ScrollingStateScrollingNode(stateNode, adoptiveTree)
49 , m_eventTrackingRegions(stateNode.eventTrackingRegions())
50 , m_layoutViewport(stateNode.layoutViewport())
51 , m_minLayoutViewportOrigin(stateNode.minLayoutViewportOrigin())
52 , m_maxLayoutViewportOrigin(stateNode.maxLayoutViewportOrigin())
53 , m_overrideVisualViewportSize(stateNode.overrideVisualViewportSize())
54 , m_frameScaleFactor(stateNode.frameScaleFactor())
55 , m_topContentInset(stateNode.topContentInset())
56 , m_headerHeight(stateNode.headerHeight())
57 , m_footerHeight(stateNode.footerHeight())
58 , m_behaviorForFixed(stateNode.scrollBehaviorForFixedElements())
59 , m_fixedElementsLayoutRelativeToFrame(stateNode.fixedElementsLayoutRelativeToFrame())
60 , m_visualViewportIsSmallerThanLayoutViewport(stateNode.visualViewportIsSmallerThanLayoutViewport())
61 , m_asyncFrameOrOverflowScrollingEnabled(stateNode.asyncFrameOrOverflowScrollingEnabled())
62 , m_wheelEventGesturesBecomeNonBlocking(stateNode.wheelEventGesturesBecomeNonBlocking())
63 , m_scrollingPerformanceTestingEnabled(stateNode.scrollingPerformanceTestingEnabled())
64{
65 if (hasChangedProperty(Property::RootContentsLayer))
66 setRootContentsLayer(stateNode.rootContentsLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
67
68 if (hasChangedProperty(Property::CounterScrollingLayer))
69 setCounterScrollingLayer(stateNode.counterScrollingLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
70
71 if (hasChangedProperty(Property::InsetClipLayer))
72 setInsetClipLayer(stateNode.insetClipLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
73
74 if (hasChangedProperty(Property::ContentShadowLayer))
75 setContentShadowLayer(stateNode.contentShadowLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
76
77 if (hasChangedProperty(Property::HeaderLayer))
78 setHeaderLayer(stateNode.headerLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
79
80 if (hasChangedProperty(Property::FooterLayer))
81 setFooterLayer(stateNode.footerLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
82}

312void ScrollingStateFrameScrollingNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
313{
314 ts << "Frame scrolling node";
315
316 ScrollingStateScrollingNode::dumpProperties(ts, behavior);
317
318 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeLayerIDs) {
319 ts.dumpProperty("root contents layer ID", m_rootContentsLayer.layerID());
320 if (m_counterScrollingLayer.layerID())
321 ts.dumpProperty("counter scrolling layer ID", m_counterScrollingLayer.layerID());
322 if (m_insetClipLayer.layerID())
323 ts.dumpProperty("inset clip layer ID", m_insetClipLayer.layerID());
324 if (m_contentShadowLayer.layerID())
325 ts.dumpProperty("content shadow layer ID", m_contentShadowLayer.layerID());
326 if (m_headerLayer.layerID())
327 ts.dumpProperty("header layer ID", m_headerLayer.layerID());
328 if (m_footerLayer.layerID())
329 ts.dumpProperty("footer layer ID", m_footerLayer.layerID());
330 }
331
332 if (m_frameScaleFactor != 1)
333 ts.dumpProperty("frame scale factor", m_frameScaleFactor);
334 if (m_topContentInset)
335 ts.dumpProperty("top content inset", m_topContentInset);
336 if (m_headerHeight)
337 ts.dumpProperty("header height", m_headerHeight);
338 if (m_footerHeight)
339 ts.dumpProperty("footer height", m_footerHeight);
340
341 ts.dumpProperty("layout viewport", m_layoutViewport);
342 ts.dumpProperty("min layout viewport origin", m_minLayoutViewportOrigin);
343 ts.dumpProperty("max layout viewport origin", m_maxLayoutViewportOrigin);
344
345 if (m_overrideVisualViewportSize)
346 ts.dumpProperty("override visual viewport size", m_overrideVisualViewportSize.value());
347
348 if (!m_eventTrackingRegions.asynchronousDispatchRegion.isEmpty()) {
349 TextStream::GroupScope scope(ts);
350 ts << "asynchronous event dispatch region";
351 for (auto rect : m_eventTrackingRegions.asynchronousDispatchRegion.rects()) {
352 ts << "\n";
353 ts << indent << rect;
354 }
355 }
356
357 auto& synchronousDispatchRegionMap = m_eventTrackingRegions.eventSpecificSynchronousDispatchRegions;
358 if (!synchronousDispatchRegionMap.isEmpty()) {
359 auto eventRegionNames = copyToVector(synchronousDispatchRegionMap.keys());
360 std::sort(eventRegionNames.begin(), eventRegionNames.end(), WTF::codePointCompareLessThan);
361 for (const auto& name : eventRegionNames) {
362 const auto& region = synchronousDispatchRegionMap.get(name);
363 TextStream::GroupScope scope(ts);
364 ts << "synchronous event dispatch region for event " << name;
365 for (auto rect : region.rects()) {
366 ts << "\n";
367 ts << indent << rect;
368 }
369 }
370 }
371
372 ts.dumpProperty("behavior for fixed", m_behaviorForFixed);
373
374 if (m_visualViewportIsSmallerThanLayoutViewport)
375 ts.dumpProperty("visual viewport smaller than layout viewport", m_visualViewportIsSmallerThanLayoutViewport);
376
377 if (m_fixedElementsLayoutRelativeToFrame)
378 ts.dumpProperty("fixed elements lay out relative to frame", m_fixedElementsLayoutRelativeToFrame);
379}
