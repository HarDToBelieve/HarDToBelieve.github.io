42ScrollingStateScrollingNode::ScrollingStateScrollingNode(const ScrollingStateScrollingNode& stateNode, ScrollingStateTree& adoptiveTree)
43 : ScrollingStateNode(stateNode, adoptiveTree)
44 , m_scrollableAreaSize(stateNode.scrollableAreaSize())
45 , m_totalContentsSize(stateNode.totalContentsSize())
46 , m_reachableContentsSize(stateNode.reachableContentsSize())
47 , m_scrollPosition(stateNode.scrollPosition())
48 , m_scrollOrigin(stateNode.scrollOrigin())
49 , m_snapOffsetsInfo(stateNode.m_snapOffsetsInfo)
50#if PLATFORM(MAC)
51 , m_verticalScrollerImp(stateNode.verticalScrollerImp())
52 , m_horizontalScrollerImp(stateNode.horizontalScrollerImp())
53#endif
54 , m_scrollableAreaParameters(stateNode.scrollableAreaParameters())
55 , m_requestedScrollData(stateNode.requestedScrollData())
56#if ENABLE(SCROLLING_THREAD)
57 , m_synchronousScrollingReasons(stateNode.synchronousScrollingReasons())
58#endif
59 , m_isMonitoringWheelEvents(stateNode.isMonitoringWheelEvents())
60{
61 scrollingStateTree().scrollingNodeAdded();
62
63 if (hasChangedProperty(Property::ScrollContainerLayer))
64 setScrollContainerLayer(stateNode.scrollContainerLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
65
66 if (hasChangedProperty(Property::ScrolledContentsLayer))
67 setScrolledContentsLayer(stateNode.scrolledContentsLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
68
69 if (hasChangedProperty(Property::VerticalScrollbarLayer))
70 setVerticalScrollbarLayer(stateNode.verticalScrollbarLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
71
72 if (hasChangedProperty(Property::HorizontalScrollbarLayer))
73 setHorizontalScrollbarLayer(stateNode.horizontalScrollbarLayer().toRepresentation(adoptiveTree.preferredLayerRepresentation()));
74}

76ScrollingStateScrollingNode::~ScrollingStateScrollingNode()

78 scrollingStateTree().scrollingNodeRemoved();
79}

81OptionSet<ScrollingStateNode::Property> ScrollingStateScrollingNode::applicableProperties() const
82{
83 // Note that this list does not include Property::RequestedScrollPosition, which is imperative, not stateful.
84 constexpr OptionSet<Property> nodeProperties = {
85 Property::ScrollableAreaSize,
86 Property::TotalContentsSize,
87 Property::ReachableContentsSize,
88 Property::ScrollPosition,
89 Property::ScrollOrigin,
90 Property::ScrollableAreaParams,
91#if ENABLE(SCROLLING_THREAD)
92 Property::ReasonsForSynchronousScrolling,
93#endif
94 Property::SnapOffsetsInfo,
95 Property::CurrentHorizontalSnapOffsetIndex,
96 Property::CurrentVerticalSnapOffsetIndex,
97 Property::IsMonitoringWheelEvents,
98 Property::ScrollContainerLayer,
99 Property::ScrolledContentsLayer,
100 Property::HorizontalScrollbarLayer,
101 Property::VerticalScrollbarLayer,
102 Property::PainterForScrollbar
103 };
104
105 auto properties = ScrollingStateNode::applicableProperties();
106 properties.add(nodeProperties);
107 return properties;
108}

260void ScrollingStateScrollingNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
261{
262 ScrollingStateNode::dumpProperties(ts, behavior);
263
264 if (m_scrollPosition != FloatPoint()) {
265 TextStream::GroupScope scope(ts);
266 ts << "scroll position "
267 << TextStream::FormatNumberRespectingIntegers(m_scrollPosition.x()) << " "
268 << TextStream::FormatNumberRespectingIntegers(m_scrollPosition.y());
269 }
270
271 if (!m_scrollableAreaSize.isEmpty()) {
272 TextStream::GroupScope scope(ts);
273 ts << "scrollable area size "
274 << TextStream::FormatNumberRespectingIntegers(m_scrollableAreaSize.width()) << " "
275 << TextStream::FormatNumberRespectingIntegers(m_scrollableAreaSize.height());
276 }
277
278 if (!m_totalContentsSize.isEmpty()) {
279 TextStream::GroupScope scope(ts);
280 ts << "contents size "
281 << TextStream::FormatNumberRespectingIntegers(m_totalContentsSize.width()) << " "
282 << TextStream::FormatNumberRespectingIntegers(m_totalContentsSize.height());
283 }
284
285 if (m_reachableContentsSize != m_totalContentsSize)
286 ts.dumpProperty("reachable contents size", m_reachableContentsSize);
287
288 if (!m_requestedScrollData.scrollPosition.isZero()) {
289 TextStream::GroupScope scope(ts);
290 ts << "requested scroll position "
291 << TextStream::FormatNumberRespectingIntegers(m_requestedScrollData.scrollPosition.x()) << " "
292 << TextStream::FormatNumberRespectingIntegers(m_requestedScrollData.scrollPosition.y());
293 }
294 if (m_requestedScrollData.scrollType == ScrollType::Programmatic)
295 ts.dumpProperty("requested scroll position represents programmatic scroll", true);
296
297 if (m_requestedScrollData.clamping == ScrollClamping::Unclamped)
298 ts.dumpProperty("requested scroll position clamping", m_requestedScrollData.clamping);
299
300 if (m_scrollOrigin != IntPoint())
301 ts.dumpProperty("scroll origin", m_scrollOrigin);
302
303 if (m_snapOffsetsInfo.horizontalSnapOffsets.size())
304 ts.dumpProperty("horizontal snap offsets", m_snapOffsetsInfo.horizontalSnapOffsets);
305
306 if (m_snapOffsetsInfo.verticalSnapOffsets.size())
307 ts.dumpProperty("vertical snap offsets", m_snapOffsetsInfo.verticalSnapOffsets);
308
309 if (m_currentHorizontalSnapPointIndex)
310 ts.dumpProperty("current horizontal snap point index", m_currentHorizontalSnapPointIndex);
311
312 if (m_currentVerticalSnapPointIndex)
313 ts.dumpProperty("current vertical snap point index", m_currentVerticalSnapPointIndex);
314
315 ts.dumpProperty("scrollable area parameters", m_scrollableAreaParameters);
316
317#if ENABLE(SCROLLING_THREAD)
318 if (!m_synchronousScrollingReasons.isEmpty())
319 ts.dumpProperty("Scrolling on main thread because:", ScrollingCoordinator::synchronousScrollingReasonsAsText(m_synchronousScrollingReasons));
320#endif
321
322 if (m_isMonitoringWheelEvents)
323 ts.dumpProperty("expects wheel event test trigger", m_isMonitoringWheelEvents);
324
325 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeLayerIDs) {
326 if (m_scrollContainerLayer.layerID())
327 ts.dumpProperty("scroll container layer", m_scrollContainerLayer.layerID());
328 if (m_scrolledContentsLayer.layerID())
329 ts.dumpProperty("scrolled contents layer", m_scrolledContentsLayer.layerID());
330 }
331}
