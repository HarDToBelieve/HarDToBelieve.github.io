75void ScrollingStateStickyNode::updateConstraints(const StickyPositionViewportConstraints& constraints)
76{
77 if (m_constraints == constraints)
78 return;
79
80 LOG_WITH_STREAM(Scrolling, stream << "ScrollingStateStickyNode " << scrollingNodeID() << " updateConstraints with constraining rect " << constraints.constrainingRectAtLastLayout() << " sticky offset " << constraints.stickyOffsetAtLastLayout() << " layer pos at last layout " << constraints.layerPositionAtLastLayout());
81
82 m_constraints = constraints;
83 setPropertyChanged(Property::ViewportConstraints);
84}

86FloatPoint ScrollingStateStickyNode::computeLayerPosition(const LayoutRect& viewportRect) const
87{
88 // This logic follows ScrollingTreeStickyNode::computeLayerPosition().
89 auto computeLayerPositionForScrollingNode = [&](ScrollingStateNode& scrollingStateNode) {
90 FloatRect constrainingRect;
91 if (is<ScrollingStateFrameScrollingNode>(scrollingStateNode))
92 constrainingRect = viewportRect;
93 else {
94 auto& overflowScrollingNode = downcast<ScrollingStateOverflowScrollingNode>(scrollingStateNode);
95 constrainingRect = FloatRect(overflowScrollingNode.scrollPosition(), m_constraints.constrainingRectAtLastLayout().size());
96 }
97 return m_constraints.layerPositionForConstrainingRect(constrainingRect);
98 };
99
100 for (auto* ancestor = parent(); ancestor; ancestor = ancestor->parent()) {
101 if (is<ScrollingStateOverflowScrollProxyNode>(*ancestor)) {
102 auto& overflowProxyNode = downcast<ScrollingStateOverflowScrollProxyNode>(*ancestor);
103 auto overflowNode = scrollingStateTree().stateNodeForID(overflowProxyNode.overflowScrollingNode());
104 if (!overflowNode)
105 break;
106
107 return computeLayerPositionForScrollingNode(*overflowNode);
108 }
109
110 if (is<ScrollingStateScrollingNode>(*ancestor))
111 return computeLayerPositionForScrollingNode(*ancestor);
112
113 if (is<ScrollingStateFixedNode>(*ancestor) || is<ScrollingStateStickyNode>(*ancestor)) {
114 // FIXME: Do we need scrolling tree nodes at all for nested cases?
115 return m_constraints.layerPositionAtLastLayout();
116 }
117 }
118 ASSERT_NOT_REACHED();
119 return m_constraints.layerPositionAtLastLayout();
120}

122void ScrollingStateStickyNode::reconcileLayerPositionForViewportRect(const LayoutRect& viewportRect, ScrollingLayerPositionAction action)
123{
124 FloatPoint position = computeLayerPosition(viewportRect);
125 if (layer().representsGraphicsLayer()) {
126 auto* graphicsLayer = static_cast<GraphicsLayer*>(layer());
127
128 LOG_WITH_STREAM(Compositing, stream << "ScrollingStateStickyNode " << scrollingNodeID() << " reconcileLayerPositionForViewportRect " << action << " position of layer " << graphicsLayer->primaryLayerID() << " to " << position << " sticky offset " << m_constraints.stickyOffsetAtLastLayout());
129
130 switch (action) {
131 case ScrollingLayerPositionAction::Set:
132 graphicsLayer->setPosition(position);
133 break;
134
135 case ScrollingLayerPositionAction::SetApproximate:
136 graphicsLayer->setApproximatePosition(position);
137 break;
138
139 case ScrollingLayerPositionAction::Sync:
140 graphicsLayer->syncPosition(position);
141 break;
142 }
143 }
144}

146void ScrollingStateStickyNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
147{
148 ts << "Sticky node";
149 ScrollingStateNode::dumpProperties(ts, behavior);
150
151 if (m_constraints.anchorEdges()) {
152 TextStream::GroupScope scope(ts);
153 ts << "anchor edges: ";
154 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeLeft))
155 ts << "AnchorEdgeLeft ";
156 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeRight))
157 ts << "AnchorEdgeRight ";
158 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeTop))
159 ts << "AnchorEdgeTop ";
160 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeBottom))
161 ts << "AnchorEdgeBottom";
162 }
163
164 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeLeft))
165 ts.dumpProperty("left offset", m_constraints.leftOffset());
166 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeRight))
167 ts.dumpProperty("right offset", m_constraints.rightOffset());
168 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeTop))
169 ts.dumpProperty("top offset", m_constraints.topOffset());
170 if (m_constraints.hasAnchorEdge(ViewportConstraints::AnchorEdgeBottom))
171 ts.dumpProperty("bottom offset", m_constraints.bottomOffset());
172
173 ts.dumpProperty("containing block rect", m_constraints.containingBlockRect());
174
175 ts.dumpProperty("sticky box rect", m_constraints.stickyBoxRect());
176
177 ts.dumpProperty("constraining rect", m_constraints.constrainingRectAtLastLayout());
178
179 ts.dumpProperty("sticky offset at last layout", m_constraints.stickyOffsetAtLastLayout());
180
181 ts.dumpProperty("layer position at last layout", m_constraints.layerPositionAtLastLayout());
182}
