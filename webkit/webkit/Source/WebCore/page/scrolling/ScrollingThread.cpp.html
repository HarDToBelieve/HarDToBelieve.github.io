36namespace WebCore {
37
38ScrollingThread& ScrollingThread::singleton()
39{
40 static LazyNeverDestroyed<ScrollingThread> scrollingThread;
41 static std::once_flag onceFlag;
42 std::call_once(onceFlag, [] {
43 scrollingThread.construct();
44 });
45
46 return scrollingThread;
47}
48
49ScrollingThread::ScrollingThread()
50{
51 BinarySemaphore semaphore;
52 m_thread = Thread::create("WebCore: Scrolling", [this, &semaphore] {
53 WTF::Thread::setCurrentThreadIsUserInteractive();
54 m_runLoop = &RunLoop::current();
55 semaphore.signal();
56 m_runLoop->run();
57 });
58
59 semaphore.wait();
60}
61
62bool ScrollingThread::isCurrentThread()
63{
64 return ScrollingThread::singleton().m_thread == &Thread::current();
65}
66
67void ScrollingThread::dispatch(Function<void ()>&& function)
68{
69 ScrollingThread::singleton().runLoop().dispatch(WTFMove(function));
70}
71
72void ScrollingThread::dispatchBarrier(Function<void ()>&& function)
73{
74 dispatch([function = WTFMove(function)]() mutable {
75 callOnMainThread(WTFMove(function));
76 });
77}
