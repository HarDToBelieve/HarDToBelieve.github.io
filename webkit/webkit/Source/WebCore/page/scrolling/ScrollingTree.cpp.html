49struct CommitTreeState {
50 // unvisitedNodes starts with all nodes in the map; we remove nodes as we visit them. At the end, it's the unvisited nodes.
51 // We can't use orphanNodes for this, because orphanNodes won't contain descendants of removed nodes.
52 HashSet<ScrollingNodeID> unvisitedNodes;
53 // Nodes with non-empty synchronousScrollingReasons.
54 HashSet<ScrollingNodeID> synchronousScrollingNodes;
55 // orphanNodes keeps child nodes alive while we rebuild child lists.
56 OrphanScrollingNodeMap orphanNodes;
57};

66OptionSet<WheelEventProcessingSteps> ScrollingTree::computeWheelProcessingSteps(const PlatformWheelEvent& wheelEvent)
67{
68 if (!m_rootNode)
69 return { WheelEventProcessingSteps::ScrollingThread };
70
71 FloatPoint position = wheelEvent.position();
72 position.move(m_rootNode->viewToContentsOffset(m_treeState.mainFrameScrollPosition));
73
74 if (!m_treeState.eventTrackingRegions.isEmpty()) {
75 const EventNames& names = eventNames();
76 IntPoint roundedPosition = roundedIntPoint(position);
77
78 // Event regions are affected by page scale, so no need to map through scale.
79 bool isSynchronousDispatchRegion = m_treeState.eventTrackingRegions.trackingTypeForPoint(names.wheelEvent, roundedPosition) == TrackingType::Synchronous
80 || m_treeState.eventTrackingRegions.trackingTypeForPoint(names.mousewheelEvent, roundedPosition) == TrackingType::Synchronous;
81 LOG_WITH_STREAM(Scrolling, stream << "\nScrollingTree::determineWheelEventProcessing: wheelEvent " << wheelEvent << " mapped to content point " << position << ", in non-fast region " << isSynchronousDispatchRegion);
82
83 if (isSynchronousDispatchRegion)
84 return { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch };
85 }
86
87#if ENABLE(WHEEL_EVENT_REGIONS)
88 auto eventListenerTypes = eventListenerRegionTypesForPoint(position);
89 if (eventListenerTypes.contains(EventListenerRegionType::NonPassiveWheel)) {
90 if (m_treeState.gestureState.value_or(WheelScrollGestureState::Blocking) == WheelScrollGestureState::NonBlocking)
91 return { WheelEventProcessingSteps::ScrollingThread, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch };
92
93 return { WheelEventProcessingSteps::MainThreadForScrolling, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch };
94 }
95
96 if (eventListenerTypes.contains(EventListenerRegionType::Wheel))
97 return { WheelEventProcessingSteps::ScrollingThread, WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch };
98#endif
99 return { WheelEventProcessingSteps::ScrollingThread };
100}

102OptionSet<WheelEventProcessingSteps> ScrollingTree::determineWheelEventProcessing(const PlatformWheelEvent& wheelEvent)
103{
104 // This method is invoked by the event handling thread
105 Locker locker { m_treeStateLock };
106
107 auto latchedNodeAndSteps = m_latchingController.latchingDataForEvent(wheelEvent, m_allowLatching);
108 if (latchedNodeAndSteps) {
109 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollingTree::determineWheelEventProcessing " << wheelEvent << " have latched node " << latchedNodeAndSteps->scrollingNodeID << " steps " << latchedNodeAndSteps->processingSteps);
110 return latchedNodeAndSteps->processingSteps;
111 }
112 if (wheelEvent.isGestureStart() || wheelEvent.isNonGestureEvent())
113 m_treeState.gestureState = std::nullopt;
114
115 auto processingSteps = computeWheelProcessingSteps(wheelEvent);
116
117 m_latchingController.receivedWheelEvent(wheelEvent, processingSteps, m_allowLatching);
118
119 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTree::determineWheelEventProcessing: processingSteps " << processingSteps);
120
121 return processingSteps;
122}

124WheelEventHandlingResult ScrollingTree::handleWheelEvent(const PlatformWheelEvent& wheelEvent, OptionSet<WheelEventProcessingSteps> processingSteps)
125{
126 LOG_WITH_STREAM(Scrolling, stream << "\nScrollingTree " << this << " handleWheelEvent " << wheelEvent);
127
128 Locker locker { m_treeLock };
129
130 if (isMonitoringWheelEvents())
131 receivedWheelEvent(wheelEvent);
132
133 m_latchingController.receivedWheelEvent(wheelEvent, processingSteps, m_allowLatching);
134
135 auto result = [&] {
136 if (!m_rootNode)
137 return WheelEventHandlingResult::unhandled();
138
139 if (!asyncFrameOrOverflowScrollingEnabled()) {
140 auto result = m_rootNode->handleWheelEvent(wheelEvent);
141 if (result.wasHandled)
142 m_gestureState.nodeDidHandleEvent(m_rootNode->scrollingNodeID(), wheelEvent);
143 return result;
144 }
145
146 if (m_gestureState.handleGestureCancel(wheelEvent)) {
147 clearNodesWithUserScrollInProgress();
148 return WheelEventHandlingResult::handled();
149 }
150
151 m_gestureState.receivedWheelEvent(wheelEvent);
152
153 if (auto latchedNodeAndSteps = m_latchingController.latchingDataForEvent(wheelEvent, m_allowLatching)) {
154 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollingTree::handleWheelEvent: has latched node " << latchedNodeAndSteps->scrollingNodeID);
155 auto* node = nodeForID(latchedNodeAndSteps->scrollingNodeID);
156 if (is<ScrollingTreeScrollingNode>(node)) {
157 auto result = downcast<ScrollingTreeScrollingNode>(*node).handleWheelEvent(wheelEvent);
158 if (result.wasHandled) {
159 m_latchingController.nodeDidHandleEvent(latchedNodeAndSteps->scrollingNodeID, processingSteps, wheelEvent, m_allowLatching);
160 m_gestureState.nodeDidHandleEvent(latchedNodeAndSteps->scrollingNodeID, wheelEvent);
161 }
162 return result;
163 }
164 }
165
166 FloatPoint position = wheelEvent.position();
167 {
168 Locker locker { m_treeStateLock };
169 position.move(m_rootNode->viewToContentsOffset(m_treeState.mainFrameScrollPosition));
170 }
171 auto node = scrollingNodeForPoint(position);
172
173 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTree::handleWheelEvent found node " << (node ? node->scrollingNodeID() : 0) << " for point " << position);
174
175 return handleWheelEventWithNode(wheelEvent, processingSteps, node.get());
176 }();
177
178 static constexpr OptionSet<WheelEventProcessingSteps> mainThreadSteps = { WheelEventProcessingSteps::MainThreadForNonBlockingDOMEventDispatch, WheelEventProcessingSteps::MainThreadForBlockingDOMEventDispatch };
179 result.steps.add(processingSteps & mainThreadSteps);
180 return result;
181}

183WheelEventHandlingResult ScrollingTree::handleWheelEventWithNode(const PlatformWheelEvent& wheelEvent, OptionSet<WheelEventProcessingSteps> processingSteps, ScrollingTreeNode* node, EventTargeting eventTargeting)
184{
185 while (node) {
186 if (is<ScrollingTreeScrollingNode>(*node)) {
187 auto& scrollingNode = downcast<ScrollingTreeScrollingNode>(*node);
188 auto result = scrollingNode.handleWheelEvent(wheelEvent, eventTargeting);
189
190 if (result.wasHandled) {
191 m_latchingController.nodeDidHandleEvent(scrollingNode.scrollingNodeID(), processingSteps, wheelEvent, m_allowLatching);
192 m_gestureState.nodeDidHandleEvent(scrollingNode.scrollingNodeID(), wheelEvent);
193 return result;
194 }
195
196 if (result.needsMainThreadProcessing() || eventTargeting != EventTargeting::Propagate)
197 return result;
198 }
199
200 if (is<ScrollingTreeOverflowScrollProxyNode>(*node)) {
201 if (auto relatedNode = nodeForID(downcast<ScrollingTreeOverflowScrollProxyNode>(*node).overflowScrollingNodeID())) {
202 node = relatedNode;
203 continue;
204 }
205 }
206
207 node = node->parent();
208 }
209 return WheelEventHandlingResult::unhandled();
210}

225void ScrollingTree::traverseScrollingTree(VisitorFunction&& visitorFunction)
226{
227 Locker locker { m_treeLock };
228 if (!m_rootNode)
229 return;
230
231 auto function = WTFMove(visitorFunction);
232 traverseScrollingTreeRecursive(*m_rootNode, function);
233}

235void ScrollingTree::traverseScrollingTreeRecursive(ScrollingTreeNode& node, const VisitorFunction& visitorFunction)
236{
237 bool scrolledSinceLastCommit = false;
238 std::optional<FloatPoint> scrollPosition;
239 if (is<ScrollingTreeScrollingNode>(node)) {
240 auto& scrollingNode = downcast<ScrollingTreeScrollingNode>(node);
241 scrollPosition = scrollingNode.currentScrollPosition();
242 scrolledSinceLastCommit = scrollingNode.scrolledSinceLastCommit();
243 }
244
245 std::optional<FloatPoint> layoutViewportOrigin;
246 if (is<ScrollingTreeFrameScrollingNode>(node))
247 layoutViewportOrigin = downcast<ScrollingTreeFrameScrollingNode>(node).layoutViewport().location();
248
249 visitorFunction(node.scrollingNodeID(), node.nodeType(), scrollPosition, layoutViewportOrigin, scrolledSinceLastCommit);
250
251 for (auto& child : node.children())
252 traverseScrollingTreeRecursive(child.get(), visitorFunction);
253}

265void ScrollingTree::commitTreeState(std::unique_ptr<ScrollingStateTree>&& scrollingStateTree)
266{
267 SetForScope<bool> inCommitTreeState(m_inCommitTreeState, true);
268 Locker locker { m_treeLock };
269
270 bool rootStateNodeChanged = scrollingStateTree->hasNewRootStateNode();
271
272 LOG(ScrollingTree, "\nScrollingTree %p commitTreeState", this);
273
274 auto* rootNode = scrollingStateTree->rootStateNode();
275 if (rootNode
276 && (rootStateNodeChanged
277 || rootNode->hasChangedProperty(ScrollingStateNode::Property::EventTrackingRegion)
278 || rootNode->hasChangedProperty(ScrollingStateNode::Property::ScrolledContentsLayer)
279 || rootNode->hasChangedProperty(ScrollingStateNode::Property::AsyncFrameOrOverflowScrollingEnabled)
280 || rootNode->hasChangedProperty(ScrollingStateNode::Property::WheelEventGesturesBecomeNonBlocking)
281 || rootNode->hasChangedProperty(ScrollingStateNode::Property::ScrollingPerformanceTestingEnabled)
282 || rootNode->hasChangedProperty(ScrollingStateNode::Property::IsMonitoringWheelEvents))) {
283 Locker locker { m_treeStateLock };
284
285 if (rootStateNodeChanged || rootNode->hasChangedProperty(ScrollingStateNode::Property::ScrolledContentsLayer))
286 m_treeState.mainFrameScrollPosition = { };
287
288 if (rootStateNodeChanged || rootNode->hasChangedProperty(ScrollingStateNode::Property::EventTrackingRegion))
289 m_treeState.eventTrackingRegions = scrollingStateTree->rootStateNode()->eventTrackingRegions();
290
291 if (rootStateNodeChanged || rootNode->hasChangedProperty(ScrollingStateNode::Property::AsyncFrameOrOverflowScrollingEnabled))
292 m_asyncFrameOrOverflowScrollingEnabled = scrollingStateTree->rootStateNode()->asyncFrameOrOverflowScrollingEnabled();
293
294 if (rootStateNodeChanged || rootNode->hasChangedProperty(ScrollingStateNode::Property::WheelEventGesturesBecomeNonBlocking))
295 m_wheelEventGesturesBecomeNonBlocking = scrollingStateTree->rootStateNode()->wheelEventGesturesBecomeNonBlocking();
296
297 if (rootStateNodeChanged || rootNode->hasChangedProperty(ScrollingStateNode::Property::ScrollingPerformanceTestingEnabled))
298 m_scrollingPerformanceTestingEnabled = scrollingStateTree->rootStateNode()->scrollingPerformanceTestingEnabled();
299
300 if (rootStateNodeChanged || rootNode->hasChangedProperty(ScrollingStateNode::Property::IsMonitoringWheelEvents))
301 m_isMonitoringWheelEvents = scrollingStateTree->rootStateNode()->isMonitoringWheelEvents();
302 }
303
304 m_overflowRelatedNodesMap.clear();
305 m_activeOverflowScrollProxyNodes.clear();
306 m_activePositionedNodes.clear();
307
308 CommitTreeState commitState;
309 for (auto nodeID : m_nodeMap.keys())
310 commitState.unvisitedNodes.add(nodeID);
311
312 updateTreeFromStateNodeRecursive(rootNode, commitState);
313 propagateSynchronousScrollingReasons(commitState.synchronousScrollingNodes);
314
315 for (auto nodeID : commitState.unvisitedNodes) {
316 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTree::commitTreeState - removing unvisited node " << nodeID);
317
318 m_latchingController.nodeWasRemoved(nodeID);
319
320 if (auto node = m_nodeMap.take(nodeID))
321 node->willBeDestroyed();
322 }
323
324 LOG_WITH_STREAM(ScrollingTree, stream << "committed ScrollingTree" << scrollingTreeAsText(ScrollingStateTreeAsTextBehaviorDebug));
325}

327void ScrollingTree::updateTreeFromStateNodeRecursive(const ScrollingStateNode* stateNode, CommitTreeState& state)
328{
329 if (!stateNode) {
330 removeAllNodes();
331 m_rootNode = nullptr;
332 return;
333 }
334
335 ScrollingNodeID nodeID = stateNode->scrollingNodeID();
336 ScrollingNodeID parentNodeID = stateNode->parentNodeID();
337
338 auto it = m_nodeMap.find(nodeID);
339
340 RefPtr<ScrollingTreeNode> node;
341 if (it != m_nodeMap.end()) {
342 node = it->value;
343 state.unvisitedNodes.remove(nodeID);
344 } else {
345 node = createScrollingTreeNode(stateNode->nodeType(), nodeID);
346 if (!parentNodeID) {
347 // This is the root node. Clear the node map.
348 ASSERT(stateNode->isFrameScrollingNode());
349 m_rootNode = downcast<ScrollingTreeFrameScrollingNode>(node.get());
350 removeAllNodes();
351 }
352 m_nodeMap.set(nodeID, node.get());
353 }
354
355 if (parentNodeID) {
356 auto parentIt = m_nodeMap.find(parentNodeID);
357 ASSERT_WITH_SECURITY_IMPLICATION(parentIt != m_nodeMap.end());
358 if (parentIt != m_nodeMap.end()) {
359 auto* parent = parentIt->value.get();
360
361 auto* oldParent = node->parent();
362 if (oldParent)
363 oldParent->removeChild(*node);
364
365 if (oldParent != parent)
366 node->setParent(parent);
367
368 parent->appendChild(*node);
369 } else {
370 // FIXME: Use WeakPtr in m_nodeMap.
371 m_nodeMap.remove(nodeID);
372 }
373 }
374
375 node->commitStateBeforeChildren(*stateNode);
376
377 // Move all children into the orphanNodes map. Live ones will get added back as we recurse over children.
378 for (auto& childScrollingNode : node->children()) {
379 childScrollingNode->setParent(nullptr);
380 state.orphanNodes.add(childScrollingNode->scrollingNodeID(), childScrollingNode.ptr());
381 }
382 node->removeAllChildren();
383
384 // Now update the children if we have any.
385 if (auto children = stateNode->children()) {
386 for (auto& child : *children)
387 updateTreeFromStateNodeRecursive(child.get(), state);
388 }
389
390 node->commitStateAfterChildren(*stateNode);
391 node->didCompleteCommitForNode();
392
393#if ENABLE(SCROLLING_THREAD)
394 if (is<ScrollingTreeScrollingNode>(*node) && !downcast<ScrollingTreeScrollingNode>(*node).synchronousScrollingReasons().isEmpty())
395 state.synchronousScrollingNodes.add(nodeID);
396#endif
397}

450void ScrollingTree::notifyRelatedNodesAfterScrollPositionChange(ScrollingTreeScrollingNode& changedNode)
451{
452 Vector<ScrollingNodeID> additionalUpdateRoots;
453
454 if (is<ScrollingTreeOverflowScrollingNode>(changedNode))
455 additionalUpdateRoots = overflowRelatedNodes().get(changedNode.scrollingNodeID());
456
457 notifyRelatedNodesRecursive(changedNode);
458
459 for (auto positionedNodeID : additionalUpdateRoots) {
460 auto* positionedNode = nodeForID(positionedNodeID);
461 if (positionedNode)
462 notifyRelatedNodesRecursive(*positionedNode);
463 }
464}
465
466void ScrollingTree::notifyRelatedNodesRecursive(ScrollingTreeNode& node)
467{
468 node.applyLayerPositions();
469
470 for (auto& child : node.children()) {
471 // Never need to cross frame boundaries, since scroll layer adjustments are isolated to each document.
472 if (is<ScrollingTreeFrameScrollingNode>(child))
473 continue;
474
475 notifyRelatedNodesRecursive(child.get());
476 }
477}

646bool ScrollingTree::willWheelEventStartSwipeGesture(const PlatformWheelEvent& wheelEvent)
647{
648 if (wheelEvent.phase() != PlatformWheelEventPhase::Began)
649 return false;
650
651 Locker locker { m_swipeStateLock };
652
653 if (wheelEvent.deltaX() > 0 && m_swipeState.mainFramePinnedState.left() && !m_swipeState.canRubberBand.left())
654 return true;
655 if (wheelEvent.deltaX() < 0 && m_swipeState.mainFramePinnedState.right() && !m_swipeState.canRubberBand.right())
656 return true;
657 if (wheelEvent.deltaY() > 0 && m_swipeState.mainFramePinnedState.top() && !m_swipeState.canRubberBand.top())
658 return true;
659 if (wheelEvent.deltaY() < 0 && m_swipeState.mainFramePinnedState.bottom() && !m_swipeState.canRubberBand.bottom())
660 return true;
661
662 return false;
663}

709String ScrollingTree::scrollingTreeAsText(ScrollingStateTreeAsTextBehavior behavior)
710{
711 TextStream ts(TextStream::LineMode::MultipleLine);
712
713 {
714 TextStream::GroupScope scope(ts);
715 ts << "scrolling tree";
716
717 Locker locker { m_treeStateLock };
718
719 if (auto latchedNodeID = m_latchingController.latchedNodeID())
720 ts.dumpProperty("latched node", latchedNodeID.value());
721
722 if (!m_treeState.mainFrameScrollPosition.isZero())
723 ts.dumpProperty("main frame scroll position", m_treeState.mainFrameScrollPosition);
724
725 if (m_rootNode) {
726 TextStream::GroupScope scope(ts);
727 m_rootNode->dump(ts, behavior | ScrollingStateTreeAsTextBehaviorIncludeLayerPositions);
728 }
729
730 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs) {
731 if (!m_overflowRelatedNodesMap.isEmpty()) {
732 TextStream::GroupScope scope(ts);
733 ts << "overflow related nodes";
734 {
735 TextStream::IndentScope indentScope(ts);
736 for (auto& it : m_overflowRelatedNodesMap)
737 ts << "\n" << indent << it.key << " -> " << it.value;
738 }
739 }
740
741#if ENABLE(SCROLLING_THREAD)
742 if (!m_activeOverflowScrollProxyNodes.isEmpty()) {
743 TextStream::GroupScope scope(ts);
744 ts << "overflow scroll proxy nodes";
745 {
746 TextStream::IndentScope indentScope(ts);
747 for (auto& node : m_activeOverflowScrollProxyNodes)
748 ts << "\n" << indent << node->scrollingNodeID();
749 }
750 }
751
752 if (!m_activePositionedNodes.isEmpty()) {
753 TextStream::GroupScope scope(ts);
754 ts << "active positioned nodes";
755 {
756 TextStream::IndentScope indentScope(ts);
757 for (const auto& node : m_activePositionedNodes)
758 ts << "\n" << indent << node->scrollingNodeID();
759 }
760 }
761#endif // ENABLE(SCROLLING_THREAD)
762 }
763 }
764 return ts.release();
765}
