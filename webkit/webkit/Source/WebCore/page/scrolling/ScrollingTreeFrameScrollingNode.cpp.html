40ScrollingTreeFrameScrollingNode::ScrollingTreeFrameScrollingNode(ScrollingTree& scrollingTree, ScrollingNodeType nodeType, ScrollingNodeID nodeID)
41 : ScrollingTreeScrollingNode(scrollingTree, nodeType, nodeID)
42{
43 ASSERT(isFrameScrollingNode());
44}

48void ScrollingTreeFrameScrollingNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
49{
50 ScrollingTreeScrollingNode::commitStateBeforeChildren(stateNode);
51
52 const ScrollingStateFrameScrollingNode& state = downcast<ScrollingStateFrameScrollingNode>(stateNode);
53
54 if (state.hasChangedProperty(ScrollingStateNode::Property::FrameScaleFactor))
55 m_frameScaleFactor = state.frameScaleFactor();
56
57 if (state.hasChangedProperty(ScrollingStateNode::Property::HeaderHeight))
58 m_headerHeight = state.headerHeight();
59
60 if (state.hasChangedProperty(ScrollingStateNode::Property::FooterHeight))
61 m_footerHeight = state.footerHeight();
62
63 if (state.hasChangedProperty(ScrollingStateNode::Property::BehaviorForFixedElements))
64 m_behaviorForFixed = state.scrollBehaviorForFixedElements();
65
66 if (state.hasChangedProperty(ScrollingStateNode::Property::TopContentInset))
67 m_topContentInset = state.topContentInset();
68
69 if (state.hasChangedProperty(ScrollingStateNode::Property::VisualViewportIsSmallerThanLayoutViewport))
70 m_visualViewportIsSmallerThanLayoutViewport = state.visualViewportIsSmallerThanLayoutViewport();
71
72 if (state.hasChangedProperty(ScrollingStateNode::Property::FixedElementsLayoutRelativeToFrame))
73 m_fixedElementsLayoutRelativeToFrame = state.fixedElementsLayoutRelativeToFrame();
74
75 if (state.hasChangedProperty(ScrollingStateNode::Property::LayoutViewport)) {
76 m_layoutViewport = state.layoutViewport();
77 updateViewportForCurrentScrollPosition({ });
78 }
79
80 if (state.hasChangedProperty(ScrollingStateNode::Property::MinLayoutViewportOrigin))
81 m_minLayoutViewportOrigin = state.minLayoutViewportOrigin();
82
83 if (state.hasChangedProperty(ScrollingStateNode::Property::MaxLayoutViewportOrigin))
84 m_maxLayoutViewportOrigin = state.maxLayoutViewportOrigin();
85
86 if (state.hasChangedProperty(ScrollingStateNode::Property::OverrideVisualViewportSize))
87 m_overrideVisualViewportSize = state.overrideVisualViewportSize();
88}

90bool ScrollingTreeFrameScrollingNode::scrollPositionAndLayoutViewportMatch(const FloatPoint& position, std::optional<FloatRect> overrideLayoutViewport)
91{
92 return position == currentScrollPosition() && (!overrideLayoutViewport || overrideLayoutViewport.value() == m_layoutViewport);
93}
94
95FloatRect ScrollingTreeFrameScrollingNode::layoutViewportForScrollPosition(const FloatPoint& visibleContentOrigin, float scale, ScrollBehaviorForFixedElements fixedBehavior) const
96{
97 FloatSize visualViewportSize = m_overrideVisualViewportSize.value_or(scrollableAreaSize());
98 FloatRect visibleContentRect(visibleContentOrigin, visualViewportSize);
99 LayoutRect visualViewport(FrameView::visibleDocumentRect(visibleContentRect, headerHeight(), footerHeight(), totalContentsSize(), scale));
100 LayoutRect layoutViewport(m_layoutViewport);
101
102 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeFrameScrollingNode " << scrollingNodeID() << " layoutViewportForScrollPosition: " << "(visibleContentOrigin " << visibleContentOrigin << ", visualViewportSize " << visualViewportSize << ") fixed behavior " << m_behaviorForFixed);
103 LOG_WITH_STREAM(Scrolling, stream << " layoutViewport: " << layoutViewport);
104 LOG_WITH_STREAM(Scrolling, stream << " visualViewport: " << visualViewport);
105 LOG_WITH_STREAM(Scrolling, stream << " scroll positions: min: " << minLayoutViewportOrigin() << " max: "<< maxLayoutViewportOrigin());
106
107 LayoutPoint newLocation = FrameView::computeLayoutViewportOrigin(LayoutRect(visualViewport), LayoutPoint(minLayoutViewportOrigin()), LayoutPoint(maxLayoutViewportOrigin()), layoutViewport, fixedBehavior);
108
109 if (layoutViewport.location() != newLocation) {
110 layoutViewport.setLocation(newLocation);
111 LOG_WITH_STREAM(Scrolling, stream << " new layoutViewport " << layoutViewport);
112 }
113
114 return layoutViewport;
115}

117void ScrollingTreeFrameScrollingNode::updateViewportForCurrentScrollPosition(std::optional<FloatRect> overrideLayoutViewport)
118{
119 if (overrideLayoutViewport)
120 setLayoutViewport(overrideLayoutViewport.value());
121 else
122 setLayoutViewport(layoutViewportForScrollPosition(currentScrollPosition(), frameScaleFactor()));
123}
124
125FloatRect ScrollingTreeFrameScrollingNode::layoutViewportRespectingRubberBanding() const
126{
127 return layoutViewportForScrollPosition(currentScrollPosition(), frameScaleFactor(), StickToViewportBounds);
128}
129
130FloatSize ScrollingTreeFrameScrollingNode::viewToContentsOffset(const FloatPoint& scrollPosition) const
131{
132 return toFloatSize(scrollPosition) - FloatSize(0, headerHeight() + topContentInset());
133}

135void ScrollingTreeFrameScrollingNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
136{
137 ts << "frame scrolling node";
138 ScrollingTreeScrollingNode::dumpProperties(ts, behavior);
139
140 ts.dumpProperty("layout viewport", m_layoutViewport);
141 ts.dumpProperty("min layoutViewport origin", m_minLayoutViewportOrigin);
142 ts.dumpProperty("max layoutViewport origin", m_maxLayoutViewportOrigin);
143
144 if (m_overrideVisualViewportSize)
145 ts.dumpProperty("override visual viewport size", m_overrideVisualViewportSize.value());
146
147 if (m_frameScaleFactor != 1)
148 ts.dumpProperty("frame scale factor", m_frameScaleFactor);
149 if (m_topContentInset)
150 ts.dumpProperty("top content inset", m_topContentInset);
151
152 if (m_headerHeight)
153 ts.dumpProperty("header height", m_headerHeight);
154 if (m_footerHeight)
155 ts.dumpProperty("footer height", m_footerHeight);
156
157 ts.dumpProperty("behavior for fixed", m_behaviorForFixed);
158 if (m_fixedElementsLayoutRelativeToFrame)
159 ts.dumpProperty("fixed elements lay out relative to frame", m_fixedElementsLayoutRelativeToFrame);
160 if (m_visualViewportIsSmallerThanLayoutViewport)
161 ts.dumpProperty("visual viewport is smaller than layout viewport", m_visualViewportIsSmallerThanLayoutViewport);
162}
