37ScrollingTreeGestureState::ScrollingTreeGestureState(ScrollingTree& scrollingTree)
38 : m_scrollingTree(scrollingTree)
39{
40}
41
42void ScrollingTreeGestureState::receivedWheelEvent(const PlatformWheelEvent& event)
43{
44 if (event.isGestureStart()) {
45 clearAllNodes();
46 return;
47 }
48}
49
50bool ScrollingTreeGestureState::handleGestureCancel(const PlatformWheelEvent& event)
51{
52 if (event.isGestureCancel()) {
53 if (m_mayBeginNodeID)
54 m_scrollingTree.handleWheelEventPhase(*m_mayBeginNodeID, PlatformWheelEventPhase::Cancelled);
55 return true;
56 }
57
58 return false;
59}
60
61void ScrollingTreeGestureState::nodeDidHandleEvent(ScrollingNodeID nodeID, const PlatformWheelEvent& event)
62{
63 switch (event.phase()) {
64 case PlatformWheelEventPhase::MayBegin:
65 m_mayBeginNodeID = nodeID;
66 m_scrollingTree.handleWheelEventPhase(nodeID, event.phase());
67 break;
68 case PlatformWheelEventPhase::Cancelled:
69 // handleGestureCancel() should have been called first.
70 ASSERT_NOT_REACHED();
71 break;
72 case PlatformWheelEventPhase::Began:
73 m_activeNodeID = nodeID;
74 m_scrollingTree.handleWheelEventPhase(nodeID, event.phase());
75 break;
76 case PlatformWheelEventPhase::Ended:
77 if (m_activeNodeID)
78 m_scrollingTree.handleWheelEventPhase(*m_activeNodeID, event.phase());
79 break;
80 case PlatformWheelEventPhase::Changed:
81 case PlatformWheelEventPhase::Stationary:
82 case PlatformWheelEventPhase::None:
83 break;
84 }
85
86 switch (event.momentumPhase()) {
87 case PlatformWheelEventPhase::MayBegin:
88 case PlatformWheelEventPhase::Cancelled:
89 ASSERT_NOT_REACHED();
90 break;
91 case PlatformWheelEventPhase::Began:
92 m_activeNodeID = nodeID;
93 m_scrollingTree.handleWheelEventPhase(nodeID, event.momentumPhase());
94 break;
95 case PlatformWheelEventPhase::Ended:
96 if (m_activeNodeID)
97 m_scrollingTree.handleWheelEventPhase(*m_activeNodeID, event.momentumPhase());
98 break;
99 case PlatformWheelEventPhase::Changed:
100 case PlatformWheelEventPhase::Stationary:
101 case PlatformWheelEventPhase::None:
102 break;
103 }
104}
105
106void ScrollingTreeGestureState::clearAllNodes()
107{
108 m_mayBeginNodeID = 0;
109 m_activeNodeID = 0;
110}
