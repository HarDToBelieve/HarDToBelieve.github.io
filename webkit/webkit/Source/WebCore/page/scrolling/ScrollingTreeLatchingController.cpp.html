39// See also ScrollLatchingController.cpp
40static const Seconds resetLatchedStateTimeout { 100_ms };
41
42ScrollingTreeLatchingController::ScrollingTreeLatchingController() = default;
43
44void ScrollingTreeLatchingController::receivedWheelEvent(const PlatformWheelEvent& wheelEvent, OptionSet<WheelEventProcessingSteps> processingSteps, bool allowLatching)
45{
46 if (!allowLatching)
47 return;
48
49 Locker locker { m_latchedNodeLock };
50 if (wheelEvent.isGestureStart() && !latchedNodeIsRelevant()) {
51 if (m_latchedNodeAndSteps) {
52 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollingTreeLatchingController " << this << " receivedWheelEvent - " << (MonotonicTime::now() - m_lastLatchedNodeInterationTime).milliseconds() << "ms since last event, clearing latched node");
53 m_latchedNodeAndSteps.reset();
54 }
55 m_processingStepsForCurrentGesture = processingSteps;
56 }
57}
58
59std::optional<ScrollingTreeLatchingController::ScrollingNodeAndProcessingSteps> ScrollingTreeLatchingController::latchingDataForEvent(const PlatformWheelEvent& wheelEvent, bool allowLatching) const
60{
61 if (!allowLatching)
62 return std::nullopt;
63
64 Locker locker { m_latchedNodeLock };
65
66 // If we have a latched node, use it.
67 if (wheelEvent.useLatchedEventElement() && m_latchedNodeAndSteps && latchedNodeIsRelevant()) {
68 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollingTreeLatchingController " << this << " latchedNodeForEvent: returning " << m_latchedNodeAndSteps->scrollingNodeID);
69 return m_latchedNodeAndSteps;
70 }
71
72 return std::nullopt;
73}

75std::optional<ScrollingNodeID> ScrollingTreeLatchingController::latchedNodeID() const
76{
77 Locker locker { m_latchedNodeLock };
78 if (m_latchedNodeAndSteps)
79 return m_latchedNodeAndSteps->scrollingNodeID;
80
81 return std::nullopt;
82}

90void ScrollingTreeLatchingController::nodeDidHandleEvent(ScrollingNodeID scrollingNodeID, OptionSet<WheelEventProcessingSteps> processingSteps, const PlatformWheelEvent& wheelEvent, bool allowLatching)
91{
92 if (!allowLatching)
93 return;
94
95 Locker locker { m_latchedNodeLock };
96
97 if (wheelEvent.useLatchedEventElement() && m_latchedNodeAndSteps && m_latchedNodeAndSteps->scrollingNodeID == scrollingNodeID) {
98 if (wheelEvent.isEndOfMomentumScroll())
99 m_lastLatchedNodeInterationTime = { };
100 else
101 m_lastLatchedNodeInterationTime = MonotonicTime::now();
102 return;
103 }
104
105 auto shouldLatch = [&]() {
106 if (wheelEvent.delta().isZero())
107 return false;
108
109 if (wheelEvent.isGestureStart())
110 return true;
111
112 if (!wheelEvent.isGestureContinuation())
113 return false;
114
115 if (m_processingStepsForCurrentGesture.value_or(OptionSet<WheelEventProcessingSteps> { }).contains(WheelEventProcessingSteps::MainThreadForScrolling) && processingSteps.contains(WheelEventProcessingSteps::ScrollingThread))
116 return true;
117
118 return false;
119 };
120
121 if (!shouldLatch())
122 return;
123
124 m_processingStepsForCurrentGesture = processingSteps;
125
126 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollingTreeLatchingController " << this << " nodeDidHandleEvent: latching to " << scrollingNodeID);
127 m_latchedNodeAndSteps = ScrollingNodeAndProcessingSteps { scrollingNodeID, processingSteps };
128 m_lastLatchedNodeInterationTime = MonotonicTime::now();
129}
130
131void ScrollingTreeLatchingController::nodeWasRemoved(ScrollingNodeID nodeID)
132{
133 Locker locker { m_latchedNodeLock };
134 if (m_latchedNodeAndSteps && m_latchedNodeAndSteps->scrollingNodeID == nodeID)
135 m_latchedNodeAndSteps.reset();
136}
137
138void ScrollingTreeLatchingController::clearLatchedNode()
139{
140 Locker locker { m_latchedNodeLock };
141 LOG_WITH_STREAM(ScrollLatching, stream << "ScrollingTreeLatchingController " << this << " clearLatchedNode");
142 m_latchedNodeAndSteps.reset();
143}
144
145bool ScrollingTreeLatchingController::latchedNodeIsRelevant() const
146{
147 auto secondsSinceLastInteraction = MonotonicTime::now() - m_lastLatchedNodeInterationTime;
148 return secondsSinceLastInteraction < resetLatchedStateTimeout;
149}
