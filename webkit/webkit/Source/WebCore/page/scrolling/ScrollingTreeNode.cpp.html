38ScrollingTreeNode::ScrollingTreeNode(ScrollingTree& scrollingTree, ScrollingNodeType nodeType, ScrollingNodeID nodeID)
39 : m_scrollingTree(scrollingTree)
40 , m_nodeType(nodeType)
41 , m_nodeID(nodeID)
42{
43}

45ScrollingTreeNode::~ScrollingTreeNode() = default;
46
47void ScrollingTreeNode::appendChild(Ref<ScrollingTreeNode>&& childNode)
48{
49 RELEASE_ASSERT(m_scrollingTree.inCommitTreeState());
50
51 childNode->setParent(this);
52
53 m_children.append(WTFMove(childNode));
54}
55
56void ScrollingTreeNode::removeChild(ScrollingTreeNode& node)
57{
58 RELEASE_ASSERT(m_scrollingTree.inCommitTreeState());
59
60 size_t index = m_children.findMatching([&](auto& child) {
61 return &node == child.ptr();
62 });
63
64 // The index will be notFound if the node to remove is a deeper-than-1-level descendant or
65 // if node is the root state node.
66 if (index != notFound) {
67 m_children.remove(index);
68 return;
69 }
70
71 for (auto& child : m_children)
72 child->removeChild(node);
73}
74
75void ScrollingTreeNode::removeAllChildren()
76{
77 RELEASE_ASSERT(m_scrollingTree.inCommitTreeState());
78
79 m_children.clear();
80}
81
82bool ScrollingTreeNode::isRootNode() const
83{
84 return m_scrollingTree.rootNode() == this;
85}

87void ScrollingTreeNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
88{
89 if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs)
90 ts.dumpProperty("nodeID", scrollingNodeID());
91}
92
93ScrollingTreeFrameScrollingNode* ScrollingTreeNode::enclosingFrameNodeIncludingSelf()
94{
95 auto* node = this;
96 while (node && !node->isFrameScrollingNode())
97 node = node->parent();
98
99 return downcast<ScrollingTreeFrameScrollingNode>(node);
100}
101
102ScrollingTreeScrollingNode* ScrollingTreeNode::enclosingScrollingNodeIncludingSelf()
103{
104 auto* node = this;
105 while (node && !node->isScrollingNode())
106 node = node->parent();
107
108 return downcast<ScrollingTreeScrollingNode>(node);
109}
110
111void ScrollingTreeNode::dump(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
112{
113 dumpProperties(ts, behavior);
114
115 for (auto& child : m_children) {
116 TextStream::GroupScope scope(ts);
117 child->dump(ts, behavior);
118 }
119}
