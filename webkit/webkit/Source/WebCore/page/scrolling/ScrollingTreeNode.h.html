37namespace WebCore {
38
39class ScrollingStateFixedNode;
40class ScrollingTree;
41class ScrollingTreeFrameScrollingNode;
42class ScrollingTreeScrollingNode;
43
44class ScrollingTreeNode : public ThreadSafeRefCounted<ScrollingTreeNode> {
45 WTF_MAKE_FAST_ALLOCATED;
46 friend class ScrollingTree;
47public:
48 virtual ~ScrollingTreeNode();
49
50 ScrollingNodeType nodeType() const { return m_nodeType; }
51 ScrollingNodeID scrollingNodeID() const { return m_nodeID; }
52
53 bool isFixedNode() const { return nodeType() == ScrollingNodeType::Fixed; }
54 bool isStickyNode() const { return nodeType() == ScrollingNodeType::Sticky; }
55 bool isPositionedNode() const { return nodeType() == ScrollingNodeType::Positioned; }
56 bool isScrollingNode() const { return isFrameScrollingNode() || isOverflowScrollingNode(); }
57 bool isFrameScrollingNode() const { return nodeType() == ScrollingNodeType::MainFrame || nodeType() == ScrollingNodeType::Subframe; }
58 bool isFrameHostingNode() const { return nodeType() == ScrollingNodeType::FrameHosting; }
59 bool isOverflowScrollingNode() const { return nodeType() == ScrollingNodeType::Overflow; }
60 bool isOverflowScrollProxyNode() const { return nodeType() == ScrollingNodeType::OverflowProxy; }
61
62 virtual void commitStateBeforeChildren(const ScrollingStateNode&) = 0;
63 virtual void commitStateAfterChildren(const ScrollingStateNode&) { }
64 virtual void didCompleteCommitForNode() { }
65
66 virtual void willBeDestroyed() { }
67
68 ScrollingTreeNode* parent() const { return m_parent; }
69 void setParent(ScrollingTreeNode* parent) { m_parent = parent; }
70
71 WEBCORE_EXPORT bool isRootNode() const;
72
73 const Vector<Ref<ScrollingTreeNode>>& children() const { return m_children; }
74
75 void appendChild(Ref<ScrollingTreeNode>&&);
76 void removeChild(ScrollingTreeNode&);
77 void removeAllChildren();
78
79 WEBCORE_EXPORT ScrollingTreeFrameScrollingNode* enclosingFrameNodeIncludingSelf();
80 WEBCORE_EXPORT ScrollingTreeScrollingNode* enclosingScrollingNodeIncludingSelf();
81
82 WEBCORE_EXPORT void dump(WTF::TextStream&, ScrollingStateTreeAsTextBehavior) const;
83
84protected:
85 ScrollingTreeNode(ScrollingTree&, ScrollingNodeType, ScrollingNodeID);
86 ScrollingTree& scrollingTree() const { return m_scrollingTree; }
87
88 virtual void applyLayerPositions() = 0;
89
90 WEBCORE_EXPORT virtual void dumpProperties(WTF::TextStream&, ScrollingStateTreeAsTextBehavior) const;
91
92 Vector<Ref<ScrollingTreeNode>> m_children;
93
94private:
95 ScrollingTree& m_scrollingTree;
96
97 const ScrollingNodeType m_nodeType;
98 const ScrollingNodeID m_nodeID;
99
100 ScrollingTreeNode* m_parent { nullptr };
101};
102
103} // namespace WebCore
