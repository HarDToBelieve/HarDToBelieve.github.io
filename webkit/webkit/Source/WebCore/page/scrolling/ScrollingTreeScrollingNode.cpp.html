49void ScrollingTreeScrollingNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
50{
51 const ScrollingStateScrollingNode& state = downcast<ScrollingStateScrollingNode>(stateNode);
52
53 if (state.hasChangedProperty(ScrollingStateNode::Property::ScrollableAreaSize))
54 m_scrollableAreaSize = state.scrollableAreaSize();
55
56 if (state.hasChangedProperty(ScrollingStateNode::Property::TotalContentsSize)) {
57 if (scrollingTree().isRubberBandInProgressForNode(scrollingNodeID()))
58 m_totalContentsSizeForRubberBand = m_totalContentsSize;
59 else
60 m_totalContentsSizeForRubberBand = state.totalContentsSize();
61
62 m_totalContentsSize = state.totalContentsSize();
63 }
64
65 if (state.hasChangedProperty(ScrollingStateNode::Property::ReachableContentsSize))
66 m_reachableContentsSize = state.reachableContentsSize();
67
68 if (state.hasChangedProperty(ScrollingStateNode::Property::ScrollPosition)) {
69 m_lastCommittedScrollPosition = state.scrollPosition();
70 if (m_isFirstCommit && !state.hasChangedProperty(ScrollingStateNode::Property::RequestedScrollPosition))
71 m_currentScrollPosition = m_lastCommittedScrollPosition;
72 }
73
74 if (state.hasChangedProperty(ScrollingStateNode::Property::ScrollOrigin))
75 m_scrollOrigin = state.scrollOrigin();
76
77 if (state.hasChangedProperty(ScrollingStateNode::Property::SnapOffsetsInfo))
78 m_snapOffsetsInfo = state.snapOffsetsInfo();
79
80 if (state.hasChangedProperty(ScrollingStateNode::Property::CurrentHorizontalSnapOffsetIndex))
81 m_currentHorizontalSnapPointIndex = state.currentHorizontalSnapPointIndex();
82
83 if (state.hasChangedProperty(ScrollingStateNode::Property::CurrentVerticalSnapOffsetIndex))
84 m_currentVerticalSnapPointIndex = state.currentVerticalSnapPointIndex();
85
86 if (state.hasChangedProperty(ScrollingStateNode::Property::ScrollableAreaParams))
87 m_scrollableAreaParameters = state.scrollableAreaParameters();
88
89#if ENABLE(SCROLLING_THREAD)
90 if (state.hasChangedProperty(ScrollingStateNode::Property::ReasonsForSynchronousScrolling))
91 m_synchronousScrollingReasons = state.synchronousScrollingReasons();
92#endif
93
94 if (state.hasChangedProperty(ScrollingStateNode::Property::ScrollContainerLayer))
95 m_scrollContainerLayer = state.scrollContainerLayer();
96
97 if (state.hasChangedProperty(ScrollingStateNode::Property::ScrolledContentsLayer))
98 m_scrolledContentsLayer = state.scrolledContentsLayer();
99}

101void ScrollingTreeScrollingNode::commitStateAfterChildren(const ScrollingStateNode& stateNode)
102{
103 const ScrollingStateScrollingNode& scrollingStateNode = downcast<ScrollingStateScrollingNode>(stateNode);
104 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::RequestedScrollPosition)) {
105 const auto& requestedScrollData = scrollingStateNode.requestedScrollData();
106 scrollingTree().scrollingTreeNodeRequestsScroll(scrollingNodeID(), requestedScrollData.scrollPosition, requestedScrollData.scrollType, requestedScrollData.clamping);
107 }
108
109 // This synthetic bit is added back in ScrollingTree::propagateSynchronousScrollingReasons().
110#if ENABLE(SCROLLING_THREAD)
111 m_synchronousScrollingReasons.remove(SynchronousScrollingReason::DescendantScrollersHaveSynchronousScrolling);
112#endif
113 m_isFirstCommit = false;
114}

126bool ScrollingTreeScrollingNode::canHandleWheelEvent(const PlatformWheelEvent& wheelEvent, EventTargeting eventTargeting) const
127{
128 if (!canHaveScrollbars())
129 return false;
130
131 // MayBegin is used to flash scrollbars; if this node is scrollable, it can handle it.
132 if (wheelEvent.phase() == PlatformWheelEventPhase::MayBegin)
133 return true;
134
135 // We always rubber-band the latched node, or the root node.
136 // The stateless wheel event doesn't trigger rubber-band.
137 if (isLatchedNode() || eventTargeting == EventTargeting::NodeOnly || (isRootNode() && !wheelEvent.isNonGestureEvent()))
138 return true;
139
140 return eventCanScrollContents(wheelEvent);
141}

153FloatPoint ScrollingTreeScrollingNode::minimumScrollPosition() const
154{
155 auto minimumScrollOffset = FloatPoint { };
156 return ScrollableArea::scrollPositionFromOffset(minimumScrollOffset, toFloatSize(scrollOrigin()));
157}

159FloatPoint ScrollingTreeScrollingNode::maximumScrollPosition() const
160{
161 FloatPoint contentSizePoint(totalContentsSize());
162 auto maximumScrollOffset = FloatPoint(contentSizePoint - scrollableAreaSize()).expandedTo(FloatPoint());
163 return ScrollableArea::scrollPositionFromOffset(maximumScrollOffset, toFloatSize(scrollOrigin()));
164}

166bool ScrollingTreeScrollingNode::eventCanScrollContents(const PlatformWheelEvent& wheelEvent) const
167{
168#if PLATFORM(GTK)
169 // In case of GTK platform the end of momentum scroll events
170 // always have a delta of 0. More information in the documentation
171 // of the API gdk_event_is_scroll_stop_event.
172 if (wheelEvent.isEndOfNonMomentumScroll())
173 return true;
174#endif
175
176 if (wheelEvent.delta().isZero())
177 return false;
178
179 auto wheelDelta = wheelEvent.delta();
180
181 if (!m_scrollableAreaParameters.allowsHorizontalScrolling)
182 wheelDelta.setWidth(0);
183
184 if (!m_scrollableAreaParameters.allowsVerticalScrolling)
185 wheelDelta.setHeight(0);
186
187 auto oldScrollPosition = currentScrollPosition();
188 auto newScrollPosition = (oldScrollPosition - wheelDelta).constrainedBetween(minimumScrollPosition(), maximumScrollPosition());
189 return newScrollPosition != oldScrollPosition;
190}

192RectEdges<bool> ScrollingTreeScrollingNode::edgePinnedState() const
193{
194 auto scrollPosition = currentScrollPosition();
195 auto minScrollPosition = minimumScrollPosition();
196 auto maxScrollPosition = maximumScrollPosition();
197
198 bool horizontallyUnscrollable = !allowsHorizontalScrolling();
199 bool verticallyUnscrollable = !allowsVerticalScrolling();
200
201 // Top, right, bottom, left.
202 return {
203 verticallyUnscrollable || scrollPosition.y() <= minScrollPosition.y(),
204 horizontallyUnscrollable || scrollPosition.x() >= maxScrollPosition.x(),
205 verticallyUnscrollable || scrollPosition.y() >= maxScrollPosition.y(),
206 horizontallyUnscrollable || scrollPosition.x() <= minScrollPosition.x()
207 };
208}

243void ScrollingTreeScrollingNode::scrollTo(const FloatPoint& position, ScrollType scrollType, ScrollClamping clamp)
244{
245 if (position == m_currentScrollPosition)
246 return;
247
248 scrollingTree().setIsHandlingProgrammaticScroll(scrollType == ScrollType::Programmatic);
249
250 if (scrollType == ScrollType::Programmatic)
251 willDoProgrammaticScroll(position);
252
253 m_currentScrollPosition = adjustedScrollPosition(position, clamp);
254
255 LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeScrollingNode " << scrollingNodeID() << " scrollTo " << position << " adjusted to "
256 << m_currentScrollPosition << " (" << scrollType << ", " << clamp << ") (delta from last committed position " << (m_lastCommittedScrollPosition - m_currentScrollPosition) << ")"
257 << " rubberbanding " << scrollingTree().isRubberBandInProgressForNode(scrollingNodeID()));
258
259 updateViewportForCurrentScrollPosition();
260 currentScrollPositionChanged(scrollType);
261
262 scrollingTree().setIsHandlingProgrammaticScroll(false);
263}

282void ScrollingTreeScrollingNode::wasScrolledByDelegatedScrolling(const FloatPoint& position, std::optional<FloatRect> overrideLayoutViewport, ScrollingLayerPositionAction scrollingLayerPositionAction)
283{
284 bool scrollPositionChanged = !scrollPositionAndLayoutViewportMatch(position, overrideLayoutViewport);
285 if (!scrollPositionChanged && scrollingLayerPositionAction != ScrollingLayerPositionAction::Set)
286 return;
287
288 m_currentScrollPosition = adjustedScrollPosition(position, ScrollClamping::Unclamped);
289 updateViewportForCurrentScrollPosition(overrideLayoutViewport);
290
291 repositionRelatedLayers();
292
293 scrollingTree().notifyRelatedNodesAfterScrollPositionChange(*this);
294 scrollingTree().scrollingTreeNodeDidScroll(*this, scrollingLayerPositionAction);
295 scrollingTree().setNeedsApplyLayerPositionsAfterCommit();
296}

298void ScrollingTreeScrollingNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
299{
300 ScrollingTreeNode::dumpProperties(ts, behavior);
301 ts.dumpProperty("scrollable area size", m_scrollableAreaSize);
302 ts.dumpProperty("total content size", m_totalContentsSize);
303 if (m_totalContentsSizeForRubberBand != m_totalContentsSize)
304 ts.dumpProperty("total content size for rubber band", m_totalContentsSizeForRubberBand);
305 if (m_reachableContentsSize != m_totalContentsSize)
306 ts.dumpProperty("reachable content size", m_reachableContentsSize);
307 ts.dumpProperty("last committed scroll position", m_lastCommittedScrollPosition);
308
309 if (m_scrollOrigin != IntPoint())
310 ts.dumpProperty("scroll origin", m_scrollOrigin);
311
312 if (m_snapOffsetsInfo.horizontalSnapOffsets.size())
313 ts.dumpProperty("horizontal snap offsets", m_snapOffsetsInfo.horizontalSnapOffsets);
314
315 if (m_snapOffsetsInfo.verticalSnapOffsets.size())
316 ts.dumpProperty("vertical snap offsets", m_snapOffsetsInfo.verticalSnapOffsets);
317
318 if (m_currentHorizontalSnapPointIndex)
319 ts.dumpProperty("current horizontal snap point index", m_currentHorizontalSnapPointIndex);
320
321 if (m_currentVerticalSnapPointIndex)
322 ts.dumpProperty("current vertical snap point index", m_currentVerticalSnapPointIndex);
323
324 ts.dumpProperty("scrollable area parameters", m_scrollableAreaParameters);
325
326#if ENABLE(SCROLLING_THREAD)
327 if (!m_synchronousScrollingReasons.isEmpty())
328 ts.dumpProperty("synchronous scrolling reasons", ScrollingCoordinator::synchronousScrollingReasonsAsText(m_synchronousScrollingReasons));
329#endif
330}
