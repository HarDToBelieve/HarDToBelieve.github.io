26#pragma once
27
28#if ENABLE(ASYNC_SCROLLING)
29
30#include "IntRect.h"
31#include "ScrollSnapOffsetsInfo.h"
32#include "ScrollableArea.h"
33#include "ScrollingTree.h"
34#include "ScrollingTreeNode.h"
35
36namespace WebCore {
37
38class ScrollingTree;
39class ScrollingStateScrollingNode;
40struct WheelEventHandlingResult;
41
42class WEBCORE_EXPORT ScrollingTreeScrollingNode : public ScrollingTreeNode {
43 friend class ScrollingTreeScrollingNodeDelegate;
44#if PLATFORM(MAC)
45 friend class ScrollingTreeScrollingNodeDelegateMac;
46#endif
47 friend class ScrollingTree;
48
49public:
50 virtual ~ScrollingTreeScrollingNode();
51
52 void commitStateBeforeChildren(const ScrollingStateNode&) override;
53 void commitStateAfterChildren(const ScrollingStateNode&) override;
54 void didCompleteCommitForNode() final;
55
56 virtual bool canHandleWheelEvent(const PlatformWheelEvent&, EventTargeting) const;
57 virtual WheelEventHandlingResult handleWheelEvent(const PlatformWheelEvent&, EventTargeting = EventTargeting::Propagate);
58
59 FloatPoint currentScrollPosition() const { return m_currentScrollPosition; }
60 FloatPoint currentScrollOffset() const { return ScrollableArea::scrollOffsetFromPosition(m_currentScrollPosition, toFloatSize(m_scrollOrigin)); }
61 FloatPoint lastCommittedScrollPosition() const { return m_lastCommittedScrollPosition; }
62 FloatSize scrollDeltaSinceLastCommit() const { return m_currentScrollPosition - m_lastCommittedScrollPosition; }
63
64 const IntPoint& scrollOrigin() const { return m_scrollOrigin; }
65
66 RectEdges<bool> edgePinnedState() const;
67
68 bool isUserScrollProgress() const;
69 void setUserScrollInProgress(bool);
70
71 bool isScrollSnapInProgress() const;
72 void setScrollSnapInProgress(bool);
73
74 // These are imperative; they adjust the scrolling layers.
75 void scrollTo(const FloatPoint&, ScrollType = ScrollType::User, ScrollClamping = ScrollClamping::Clamped);
76 void scrollBy(const FloatSize&, ScrollClamping = ScrollClamping::Clamped);
77
78 virtual void stopScrollAnimations() { };
79
80 void wasScrolledByDelegatedScrolling(const FloatPoint& position, std::optional<FloatRect> overrideLayoutViewport = { }, ScrollingLayerPositionAction = ScrollingLayerPositionAction::Sync);
81
82#if ENABLE(SCROLLING_THREAD)
83 OptionSet<SynchronousScrollingReason> synchronousScrollingReasons() const { return m_synchronousScrollingReasons; }
84 void addSynchronousScrollingReason(SynchronousScrollingReason reason) { m_synchronousScrollingReasons.add(reason); }
85 bool hasSynchronousScrollingReasons() const { return !m_synchronousScrollingReasons.isEmpty(); }
86#endif
87
88 const FloatSize& scrollableAreaSize() const { return m_scrollableAreaSize; }
89 const FloatSize& totalContentsSize() const { return m_totalContentsSize; }
90
91 bool horizontalScrollbarHiddenByStyle() const { return m_scrollableAreaParameters.horizontalScrollbarHiddenByStyle; }
92 bool verticalScrollbarHiddenByStyle() const { return m_scrollableAreaParameters.verticalScrollbarHiddenByStyle; }
93 bool canHaveHorizontalScrollbar() const { return m_scrollableAreaParameters.horizontalScrollbarMode != ScrollbarAlwaysOff; }
94 bool canHaveVerticalScrollbar() const { return m_scrollableAreaParameters.verticalScrollbarMode != ScrollbarAlwaysOff; }
95 bool canHaveScrollbars() const { return m_scrollableAreaParameters.horizontalScrollbarMode != ScrollbarAlwaysOff || m_scrollableAreaParameters.verticalScrollbarMode != ScrollbarAlwaysOff; }
96
97 const FloatScrollSnapOffsetsInfo& snapOffsetsInfo() const;
98 std::optional<unsigned> currentHorizontalSnapPointIndex() const;
99 std::optional<unsigned> currentVerticalSnapPointIndex() const;
100 void setCurrentHorizontalSnapPointIndex(std::optional<unsigned>);
101 void setCurrentVerticalSnapPointIndex(std::optional<unsigned>);
102
103 bool useDarkAppearanceForScrollbars() const { return m_scrollableAreaParameters.useDarkAppearanceForScrollbars; }
104
105 bool eventCanScrollContents(const PlatformWheelEvent&) const;
106
107 bool scrolledSinceLastCommit() const { return m_scrolledSinceLastCommit; }
108
109 const LayerRepresentation& scrollContainerLayer() const { return m_scrollContainerLayer; }
110 const LayerRepresentation& scrolledContentsLayer() const { return m_scrolledContentsLayer; }
111
112protected:
113 ScrollingTreeScrollingNode(ScrollingTree&, ScrollingNodeType, ScrollingNodeID);
114
115 virtual FloatPoint minimumScrollPosition() const;
116 virtual FloatPoint maximumScrollPosition() const;
117
118 FloatPoint clampScrollPosition(const FloatPoint&) const;
119
120 virtual void willDoProgrammaticScroll(const FloatPoint&) { }
121
122 virtual FloatPoint adjustedScrollPosition(const FloatPoint&, ScrollClamping = ScrollClamping::Clamped) const;
123
124 virtual void currentScrollPositionChanged(ScrollType, ScrollingLayerPositionAction = ScrollingLayerPositionAction::Sync);
125 virtual void updateViewportForCurrentScrollPosition(std::optional<FloatRect> = { }) { }
126 virtual bool scrollPositionAndLayoutViewportMatch(const FloatPoint& position, std::optional<FloatRect> overrideLayoutViewport);
127
128 virtual void repositionScrollingLayers() { }
129 virtual void repositionRelatedLayers() { }
130
131 void applyLayerPositions() override;
132
133 const FloatSize& reachableContentsSize() const { return m_reachableContentsSize; }
134
135 bool isLatchedNode() const;
136
137 // If the totalContentsSize changes in the middle of a rubber-band, we still want to use the old totalContentsSize for the sake of
138 // computing the stretchAmount(). Using the old value will keep the animation smooth. When there is no rubber-band in progress at
139 // all, m_totalContentsSizeForRubberBand should be equivalent to m_totalContentsSize.
140 const FloatSize& totalContentsSizeForRubberBand() const { return m_totalContentsSizeForRubberBand; }
141 void setTotalContentsSizeForRubberBand(const FloatSize& totalContentsSizeForRubberBand) { m_totalContentsSizeForRubberBand = totalContentsSizeForRubberBand; }
142
143 ScrollElasticity horizontalScrollElasticity() const { return m_scrollableAreaParameters.horizontalScrollElasticity; }
144 ScrollElasticity verticalScrollElasticity() const { return m_scrollableAreaParameters.verticalScrollElasticity; }
145
146 bool allowsHorizontalScrolling() const { return m_scrollableAreaParameters.allowsHorizontalScrolling; }
147 bool allowsVerticalScrolling() const { return m_scrollableAreaParameters.allowsVerticalScrolling; }
148
149 void dumpProperties(WTF::TextStream&, ScrollingStateTreeAsTextBehavior) const override;
150
151private:
152 FloatSize m_scrollableAreaSize;
153 FloatSize m_totalContentsSize;
154 FloatSize m_totalContentsSizeForRubberBand;
155 FloatSize m_reachableContentsSize;
156 FloatPoint m_lastCommittedScrollPosition;
157 FloatPoint m_currentScrollPosition;
158 IntPoint m_scrollOrigin;
159 FloatScrollSnapOffsetsInfo m_snapOffsetsInfo;
160 std::optional<unsigned> m_currentHorizontalSnapPointIndex;
161 std::optional<unsigned> m_currentVerticalSnapPointIndex;
162 ScrollableAreaParameters m_scrollableAreaParameters;
163#if ENABLE(SCROLLING_THREAD)
164 OptionSet<SynchronousScrollingReason> m_synchronousScrollingReasons;
165#endif
166 bool m_isFirstCommit { true };
167 bool m_scrolledSinceLastCommit { false };
168
169 LayerRepresentation m_scrollContainerLayer;
170 LayerRepresentation m_scrolledContentsLayer;
171};
172
173} // namespace WebCore

177#endif // ENABLE(ASYNC_SCROLLING)
