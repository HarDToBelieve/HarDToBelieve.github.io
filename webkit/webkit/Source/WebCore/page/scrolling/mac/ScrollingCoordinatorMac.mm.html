65void ScrollingCoordinatorMac::pageDestroyed()
66{
67 AsyncScrollingCoordinator::pageDestroyed();
68
69 // Invalidating the scrolling tree will break the reference cycle between the ScrollingCoordinator and ScrollingTree objects.
70 RefPtr<ThreadedScrollingTree> scrollingTree = static_pointer_cast<ThreadedScrollingTree>(releaseScrollingTree());
71 ScrollingThread::dispatch([scrollingTree] {
72 scrollingTree->invalidate();
73 });
74}

76bool ScrollingCoordinatorMac::handleWheelEventForScrolling(const PlatformWheelEvent& wheelEvent, ScrollingNodeID targetNodeID, std::optional<WheelScrollGestureState> gestureState)
77{
78 ASSERT(isMainThread());
79 ASSERT(m_page);
80
81 if (scrollingTree()->willWheelEventStartSwipeGesture(wheelEvent))
82 return false;
83
84 LOG_WITH_STREAM(Scrolling, stream << "ScrollingCoordinatorMac::handleWheelEventForScrolling " << wheelEvent << " - sending event to scrolling thread, node " << targetNodeID << " gestureState " << gestureState);
85
86 auto deferrer = WheelEventTestMonitorCompletionDeferrer { m_page->wheelEventTestMonitor().get(), reinterpret_cast<WheelEventTestMonitor::ScrollableAreaIdentifier>(targetNodeID), WheelEventTestMonitor::PostMainThreadWheelEventHandling };
87
88 RefPtr<ThreadedScrollingTree> threadedScrollingTree = downcast<ThreadedScrollingTree>(scrollingTree());
89 ScrollingThread::dispatch([threadedScrollingTree, wheelEvent, targetNodeID, gestureState, deferrer = WTFMove(deferrer)] {
90 threadedScrollingTree->handleWheelEventAfterMainThread(wheelEvent, targetNodeID, gestureState);
91 });
92 return true;
93}

103void ScrollingCoordinatorMac::scheduleTreeStateCommit()
104{
105 scheduleRenderingUpdate();
106}
107
108void ScrollingCoordinatorMac::commitTreeStateIfNeeded()
109{
110 willCommitTree();
111
112 LOG_WITH_STREAM(Scrolling, stream << "ScrollingCoordinatorMac::commitTreeState, has changes " << scrollingStateTree()->hasChangedProperties());
113
114 if (!scrollingStateTree()->hasChangedProperties())
115 return;
116
117 LOG_WITH_STREAM(ScrollingTree, stream << scrollingStateTreeAsText(ScrollingStateTreeAsTextBehaviorDebug));
118
119 auto stateTree = scrollingStateTree()->commit(LayerRepresentation::PlatformLayerRepresentation);
120 scrollingTree()->commitTreeState(WTFMove(stateTree));
121
122 updateTiledScrollingIndicator();
123}
124
125void ScrollingCoordinatorMac::willStartRenderingUpdate()
126{
127 RefPtr<ThreadedScrollingTree> threadedScrollingTree = downcast<ThreadedScrollingTree>(scrollingTree());
128 threadedScrollingTree->willStartRenderingUpdate();
129 commitTreeStateIfNeeded();
130 synchronizeStateFromScrollingTree();
131}

133void ScrollingCoordinatorMac::didCompleteRenderingUpdate()

136}

138void ScrollingCoordinatorMac::updateTiledScrollingIndicator()
139{
140 FrameView* frameView = m_page->mainFrame().view();
141 if (!frameView)
142 return;
143
144 TiledBacking* tiledBacking = frameView->tiledBacking();
145 if (!tiledBacking)
146 return;
147
148 ScrollingModeIndication indicatorMode;
149 if (shouldUpdateScrollLayerPositionSynchronously(*frameView))
150 indicatorMode = SynchronousScrollingBecauseOfStyleIndication;
151 else
152 indicatorMode = AsyncScrollingIndication;
153
154 tiledBacking->setScrollingModeIndication(indicatorMode);
155}
156
157void ScrollingCoordinatorMac::startMonitoringWheelEvents(bool clearLatchingState)
158{
159 if (clearLatchingState)
160 scrollingTree()->clearLatchedNode();
161 auto monitor = m_page->wheelEventTestMonitor();
162 scrollingTree()->setWheelEventTestMonitor(WTFMove(monitor));
163}
164
165void ScrollingCoordinatorMac::stopMonitoringWheelEvents()
166{
167 scrollingTree()->setWheelEventTestMonitor(nullptr);
168}
