58void ScrollingTreeScrollingNodeDelegateMac::updateFromStateNode(const ScrollingStateScrollingNode& scrollingStateNode)
59{
60 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::PainterForScrollbar)) {
61 releaseReferencesToScrollerImpsOnTheMainThread();
62 m_verticalScrollerImp = scrollingStateNode.verticalScrollerImp();
63 m_horizontalScrollerImp = scrollingStateNode.horizontalScrollerImp();
64 }
65
66 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::SnapOffsetsInfo))
67 m_scrollController.setSnapOffsetsInfo(scrollingStateNode.snapOffsetsInfo().convertUnits<LayoutScrollSnapOffsetsInfo>());
68
69 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::CurrentHorizontalSnapOffsetIndex))
70 m_scrollController.setActiveScrollSnapIndexForAxis(ScrollEventAxis::Horizontal, scrollingStateNode.currentHorizontalSnapPointIndex());
71
72 if (scrollingStateNode.hasChangedProperty(ScrollingStateNode::Property::CurrentVerticalSnapOffsetIndex))
73 m_scrollController.setActiveScrollSnapIndexForAxis(ScrollEventAxis::Vertical, scrollingStateNode.currentVerticalSnapPointIndex());
74}

86bool ScrollingTreeScrollingNodeDelegateMac::handleWheelEvent(const PlatformWheelEvent& wheelEvent)
87{
88 bool wasInMomentumPhase = m_inMomentumPhase;
89
90 if (wheelEvent.momentumPhase() == PlatformWheelEventPhase::Began)
91 m_inMomentumPhase = true;
92 else if (wheelEvent.momentumPhase() == PlatformWheelEventPhase::Ended || wheelEvent.momentumPhase() == PlatformWheelEventPhase::Cancelled)
93 m_inMomentumPhase = false;
94
95 if (wasInMomentumPhase != m_inMomentumPhase) {
96 [m_verticalScrollerImp setUsePresentationValue:m_inMomentumPhase];
97 [m_horizontalScrollerImp setUsePresentationValue:m_inMomentumPhase];
98 }
99
100 auto deferrer = WheelEventTestMonitorCompletionDeferrer { scrollingTree().wheelEventTestMonitor(), reinterpret_cast<WheelEventTestMonitor::ScrollableAreaIdentifier>(scrollingNode().scrollingNodeID()), WheelEventTestMonitor::HandlingWheelEvent };
101
102 bool wasInUserScroll = m_scrollController.isUserScrollInProgress();
103 m_scrollController.updateGestureInProgressState(wheelEvent);
104 bool isInUserScroll = m_scrollController.isUserScrollInProgress();
105 if (isInUserScroll != wasInUserScroll)
106 scrollingNode().setUserScrollInProgress(isInUserScroll);
107
108 // PlatformWheelEventPhase::MayBegin fires when two fingers touch the trackpad, and is used to flash overlay scrollbars.
109 // We know we're scrollable at this point, so handle the event.
110 if (wheelEvent.phase() == PlatformWheelEventPhase::MayBegin)
111 return true;
112
113 return m_scrollController.handleWheelEvent(wheelEvent);
114}

124bool ScrollingTreeScrollingNodeDelegateMac::scrollPositionIsNotRubberbandingEdge(const FloatPoint& targetPosition) const
125{
126#if ENABLE(RUBBER_BANDING)
127 if (!m_scrollController.isRubberBandInProgress())
128 return false;
129
130 auto rubberbandingEdges = m_scrollController.rubberBandingEdges();
131
132 auto minimumScrollPosition = this->minimumScrollPosition();
133 auto maximumScrollPosition = this->maximumScrollPosition();
134
135 for (auto side : allBoxSides) {
136 if (!rubberbandingEdges[side])
137 continue;
138
139 switch (side) {
140 case BoxSide::Top:
141 if (targetPosition.y() != minimumScrollPosition.y())
142 return true;
143 break;
144 case BoxSide::Right:
145 if (targetPosition.x() != maximumScrollPosition.x())
146 return true;
147 break;
148 case BoxSide::Bottom:
149 if (targetPosition.y() != maximumScrollPosition.y())
150 return true;
151 break;
152 case BoxSide::Left:
153 if (targetPosition.x() != minimumScrollPosition.x())
154 return true;
155 break;
156 }
157 }
158#else
159 UNUSED_PARAM(targetPosition);
160#endif
161 return false;
162}

179bool ScrollingTreeScrollingNodeDelegateMac::isPinnedForScrollDeltaOnAxis(float scrollDelta, ScrollEventAxis axis, float scrollLimit) const
180{
181 auto scrollPosition = currentScrollPosition();
182 switch (axis) {
183 case ScrollEventAxis::Vertical:
184 if (!allowsVerticalScrolling())
185 return true;
186
187 if (scrollDelta < 0) {
188 auto topOffset = scrollPosition.y() - minimumScrollPosition().y();
189 return topOffset <= scrollLimit;
190 }
191
192 if (scrollDelta > 0) {
193 auto bottomOffset = maximumScrollPosition().y() - scrollPosition.y();
194 return bottomOffset <= scrollLimit;
195 }
196 break;
197 case ScrollEventAxis::Horizontal:
198 if (!allowsHorizontalScrolling())
199 return true;
200
201 if (scrollDelta < 0) {
202 auto leftOffset = scrollPosition.x() - minimumScrollPosition().x();
203 return leftOffset <= scrollLimit;
204 }
205
206 if (scrollDelta > 0) {
207 auto rightOffset = maximumScrollPosition().x() - scrollPosition.x();
208 return rightOffset <= scrollLimit;
209 }
210 break;
211 }
212
213 return false;
214}

246bool ScrollingTreeScrollingNodeDelegateMac::allowsHorizontalStretching(const PlatformWheelEvent& wheelEvent) const
247{
248 switch (horizontalScrollElasticity()) {
249 case ScrollElasticityAutomatic: {
250 bool scrollbarsAllowStretching = allowsHorizontalScrolling() || !allowsVerticalScrolling();
251 bool eventPreventsStretching = wheelEvent.isGestureStart() && isPinnedForScrollDeltaOnAxis(-wheelEvent.deltaX(), ScrollEventAxis::Horizontal);
252 return scrollbarsAllowStretching && !eventPreventsStretching;
253 }
254 case ScrollElasticityNone:
255 return false;
256 case ScrollElasticityAllowed: {
257 auto scrollDirection = ScrollController::directionFromEvent(wheelEvent, ScrollEventAxis::Horizontal);
258 if (scrollDirection)
259 return shouldRubberBandInDirection(scrollDirection.value());
260 return true;
261 }
262 }
263
264 ASSERT_NOT_REACHED();
265 return false;
266}

268bool ScrollingTreeScrollingNodeDelegateMac::allowsVerticalStretching(const PlatformWheelEvent& wheelEvent) const
269{
270 switch (verticalScrollElasticity()) {
271 case ScrollElasticityAutomatic: {
272 bool scrollbarsAllowStretching = allowsVerticalScrolling() || !allowsHorizontalScrolling();
273 bool eventPreventsStretching = wheelEvent.isGestureStart() && isPinnedForScrollDeltaOnAxis(-wheelEvent.deltaY(), ScrollEventAxis::Vertical);
274 return scrollbarsAllowStretching && !eventPreventsStretching;
275 }
276 case ScrollElasticityNone:
277 return false;
278 case ScrollElasticityAllowed: {
279 auto scrollDirection = ScrollController::directionFromEvent(wheelEvent, ScrollEventAxis::Vertical);
280 if (scrollDirection)
281 return shouldRubberBandInDirection(scrollDirection.value());
282 return true;
283 }
284 }
285
286 ASSERT_NOT_REACHED();
287 return false;
288}

290IntSize ScrollingTreeScrollingNodeDelegateMac::stretchAmount() const
291{
292 IntSize stretch;
293 auto scrollPosition = currentScrollPosition();
294
295 if (scrollPosition.y() < minimumScrollPosition().y())
296 stretch.setHeight(scrollPosition.y() - minimumScrollPosition().y());
297 else if (scrollPosition.y() > maximumScrollPosition().y())
298 stretch.setHeight(scrollPosition.y() - maximumScrollPosition().y());
299
300 if (scrollPosition.x() < minimumScrollPosition().x())
301 stretch.setWidth(scrollPosition.x() - minimumScrollPosition().x());
302 else if (scrollPosition.x() > maximumScrollPosition().x())
303 stretch.setWidth(scrollPosition.x() - maximumScrollPosition().x());
304
305 return stretch;
306}

308bool ScrollingTreeScrollingNodeDelegateMac::isPinnedForScrollDelta(const FloatSize& delta) const
309{
310 // This "offset < 1" logic was added in r107488. Unclear if it's needed.
311 constexpr float scrollOffsetLimit = 1.0f - std::numeric_limits<float>::epsilon();
312
313 if (fabsf(delta.height()) >= fabsf(delta.width()))
314 return isPinnedForScrollDeltaOnAxis(delta.height(), ScrollEventAxis::Vertical, scrollOffsetLimit);
315
316 if (delta.width())
317 return isPinnedForScrollDeltaOnAxis(delta.width(), ScrollEventAxis::Horizontal, scrollOffsetLimit);
318
319 return false;
320}

337bool ScrollingTreeScrollingNodeDelegateMac::shouldRubberBandInDirection(ScrollDirection direction) const
338{
339 if (scrollingNode().isRootNode())
340 return scrollingTree().mainFrameCanRubberBandInDirection(direction);
341
342 switch (direction) {
343 case ScrollDirection::ScrollUp:
344 case ScrollDirection::ScrollDown:
345 return allowsVerticalScrolling();
346 case ScrollDirection::ScrollLeft:
347 case ScrollDirection::ScrollRight:
348 return allowsHorizontalScrolling();
349 }
350 return true;
351}

386void ScrollingTreeScrollingNodeDelegateMac::immediateScrollOnAxis(ScrollEventAxis axis, float delta)
387{
388 const FloatPoint& scrollPosition = currentScrollPosition();
389 FloatPoint change;
390 if (axis == ScrollEventAxis::Horizontal)
391 change = FloatPoint(scrollPosition.x() + delta, scrollPosition.y());
392 else
393 change = FloatPoint(scrollPosition.x(), scrollPosition.y() + delta);
394
395 immediateScrollBy(change - scrollPosition);
396}

444void ScrollingTreeScrollingNodeDelegateMac::updateScrollbarPainters()
445{
446 if (m_inMomentumPhase && (m_verticalScrollerImp || m_horizontalScrollerImp)) {
447 BEGIN_BLOCK_OBJC_EXCEPTIONS
448 auto scrollOffset = scrollingNode().currentScrollOffset();
449
450 [CATransaction lock];
451
452 if ([m_verticalScrollerImp shouldUsePresentationValue]) {
453 float presentationValue;
454 float overhangAmount;
455 ScrollableArea::computeScrollbarValueAndOverhang(scrollOffset.y(), totalContentsSize().height(), scrollableAreaSize().height(), presentationValue, overhangAmount);
456 [m_verticalScrollerImp setPresentationValue:presentationValue];
457 }
458
459 if ([m_horizontalScrollerImp shouldUsePresentationValue]) {
460 float presentationValue;
461 float overhangAmount;
462 ScrollableArea::computeScrollbarValueAndOverhang(scrollOffset.x(), totalContentsSize().width(), scrollableAreaSize().width(), presentationValue, overhangAmount);
463 [m_horizontalScrollerImp setPresentationValue:presentationValue];
464 }
465
466 [CATransaction unlock];
467 END_BLOCK_OBJC_EXCEPTIONS
468 }
469}
