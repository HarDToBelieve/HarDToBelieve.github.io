27#ifndef ContentType_h
28#define ContentType_h
29
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34class ContentType {
35public:
36 WEBCORE_EXPORT explicit ContentType(String&& type);
37 WEBCORE_EXPORT explicit ContentType(const String& type);
38 ContentType() = default;
39
40 WEBCORE_EXPORT static const String& codecsParameter();
41 static const String& profilesParameter();
42
43 WEBCORE_EXPORT String parameter(const String& parameterName) const;
44 WEBCORE_EXPORT String containerType() const;
45 Vector<String> codecs() const;
46 Vector<String> profiles() const;
47 const String& raw() const { return m_type; }
48 bool isEmpty() const { return m_type.isEmpty(); }
49
50 WEBCORE_EXPORT String toJSONString() const;
51
52 template<class Encoder>
53 void encode(Encoder& encoder) const
54 {
55 encoder << m_type;
56 }
57
58 template <class Decoder>
59 static std::optional<ContentType> decode(Decoder& decoder)
60 {
61 std::optional<String> type;
62 decoder >> type;
63 if (!type)
64 return std::nullopt;
65
66 return { ContentType(WTFMove(*type)) };
67 }
68
69private:
70 String m_type;
71};
72
73} // namespace WebCore
74
75namespace WTF {
76template<typename Type> struct LogArgument;
77
78template <>
79struct LogArgument<WebCore::ContentType> {
80 static String toString(const WebCore::ContentType& type)
81 {
82 return type.toJSONString();
83 }
84};
85
86} // namespace WTF
87
88#endif // ContentType_h
