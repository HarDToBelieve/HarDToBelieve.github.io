38class ContextMenuController;

40class ContextMenu {
41 WTF_MAKE_NONCOPYABLE(ContextMenu); WTF_MAKE_FAST_ALLOCATED;
42public:
43 ContextMenu();
44
45 void setItems(const Vector<ContextMenuItem>& items) { m_items = items; }
46 const Vector<ContextMenuItem>& items() const { return m_items; }
47
48 void appendItem(const ContextMenuItem& item) { m_items.append(item); }
49
50private:
51 Vector<ContextMenuItem> m_items;
52};
