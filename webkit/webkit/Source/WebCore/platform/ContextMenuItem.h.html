164class ContextMenuItem {
165 WTF_MAKE_FAST_ALLOCATED;
166public:
167 WEBCORE_EXPORT ContextMenuItem(ContextMenuItemType, ContextMenuAction, const String&, ContextMenu* subMenu = 0);
168 WEBCORE_EXPORT ContextMenuItem(ContextMenuItemType, ContextMenuAction, const String&, bool enabled, bool checked, unsigned indentationLevel = 0);
169
170 WEBCORE_EXPORT ~ContextMenuItem();
171
172 void setType(ContextMenuItemType);
173 WEBCORE_EXPORT ContextMenuItemType type() const;
174
175 void setAction(ContextMenuAction);
176 WEBCORE_EXPORT ContextMenuAction action() const;
177
178 void setChecked(bool = true);
179 WEBCORE_EXPORT bool checked() const;
180
181 void setEnabled(bool = true);
182 WEBCORE_EXPORT bool enabled() const;
183
184 void setIndentationLevel(unsigned);
185 WEBCORE_EXPORT unsigned indentationLevel() const;
186
187 void setSubMenu(ContextMenu*);
188
189 WEBCORE_EXPORT ContextMenuItem(ContextMenuAction, const String&, bool enabled, bool checked, const Vector<ContextMenuItem>& subMenuItems, unsigned indentationLevel = 0);
190 ContextMenuItem();
191
192 bool isNull() const;
193
194 void setTitle(const String& title) { m_title = title; }
195 const String& title() const { return m_title; }
196
197 const Vector<ContextMenuItem>& subMenuItems() const { return m_subMenuItems; }
198private:
199 ContextMenuItemType m_type;
200 ContextMenuAction m_action;
201 String m_title;
202 bool m_enabled;
203 bool m_checked;
204 unsigned m_indentationLevel;
205 Vector<ContextMenuItem> m_subMenuItems;
206};
