44// HTML defines minimum week of year is one.
45static constexpr int minimumWeekNumber = 1;
46
47// HTML defines maximum week of year is 53.
48static constexpr int maximumWeekNumber = 53;
49
50static constexpr int maximumMonthInMaximumYear = 8; // This is September, since months are 0 based.
51static constexpr int maximumDayInMaximumMonth = 13;
52static constexpr int maximumWeekInMaximumYear = 37; // The week of 275760-09-13
53
54static constexpr int daysInMonth[12] = { 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31 };

56static bool isLeapYear(int year)
57{
58 if (year % 4)
59 return false;
60 if (!(year % 400))
61 return true;
62 if (!(year % 100))
63 return false;
64 return true;
65}

67// 'month' is 0-based.
68static int maxDayOfMonth(int year, int month)
69{
70 if (month != 1) // February?
71 return daysInMonth[month];
72 return isLeapYear(year) ? 29 : 28;
73}
74
75// 'month' is 0-based.
76static int dayOfWeek(int year, int month, int day)
77{
78 int shiftedMonth = month + 2;
79 // 2:January, 3:Feburuary, 4:March, ...
80
81 // Zeller's congruence
82 if (shiftedMonth <= 3) {
83 shiftedMonth += 12;
84 year--;
85 }
86 // 4:March, ..., 14:January, 15:February
87
88 int highYear = year / 100;
89 int lowYear = year % 100;
90 // We add 6 to make the result Sunday-origin.
91 int result = (day + 13 * shiftedMonth / 5 + lowYear + lowYear / 4 + highYear / 4 + 5 * highYear + 6) % 7;
92 return result;
93}

95int DateComponents::maxWeekNumberInYear() const
96{
97 int day = dayOfWeek(m_year, 0, 1); // January 1.
98 return day == Thursday || (day == Wednesday && isLeapYear(m_year)) ? maximumWeekNumber : maximumWeekNumber - 1;
99}

101template<typename CharacterType> static unsigned countDigits(StringParsingBuffer<CharacterType> buffer)
102{
103 auto begin = buffer.position();
104 skipWhile<isASCIIDigit>(buffer);
105 return buffer.position() - begin;
106}

108// Differences from parseInteger<int>: Takes StringParsingBuffer. Does not allow leading or trailing spaces. Does not allow leading "+".
109template<typename CharacterType> static std::optional<int> parseInt(StringParsingBuffer<CharacterType>& buffer, unsigned maximumNumberOfDigitsToParse)
110{
111 if (maximumNumberOfDigitsToParse > buffer.lengthRemaining() || !maximumNumberOfDigitsToParse)
112 return std::nullopt;
113
114 // We don't need to handle negative numbers for ISO 8601.
115
116 int value = 0;
117 unsigned digitsRemaining = 0;
118 while (digitsRemaining < maximumNumberOfDigitsToParse) {
119 if (!isASCIIDigit(*buffer))
120 return std::nullopt;
121 int digit = *buffer - '0';
122 if (value > (INT_MAX - digit) / 10) // Check for overflow.
123 return std::nullopt;
124 value = value * 10 + digit;
125 ++digitsRemaining;
126 ++buffer;
127 }
128 return value;
129}

131template<typename CharacterType> static std::optional<int> parseIntWithinLimits(StringParsingBuffer<CharacterType>& buffer, unsigned maximumNumberOfDigitsToParse, int minimumValue, int maximumValue)
132{
133 auto value = parseInt(buffer, maximumNumberOfDigitsToParse);
134 if (!(value && *value >= minimumValue && *value <= maximumValue))
135 return std::nullopt;
136 return value;
137}

139template<typename CharacterType> bool DateComponents::parseYear(StringParsingBuffer<CharacterType>& buffer)
140{
141 unsigned digitsLength = countDigits(buffer);
142 // Needs at least 4 digits according to the standard.
143 if (digitsLength < 4)
144 return false;
145
146 auto year = parseIntWithinLimits(buffer, digitsLength, minimumYear(), maximumYear());
147 if (!year)
148 return false;
149
150 m_year = *year;
151 return true;
152}

154static bool withinHTMLDateLimits(int year, int month)
155{
156 if (year < DateComponents::minimumYear())
157 return false;
158 if (year < DateComponents::maximumYear())
159 return true;
160 return month <= maximumMonthInMaximumYear;
161}
162
163static bool withinHTMLDateLimits(int year, int month, int monthDay)
164{
165 if (year < DateComponents::minimumYear())
166 return false;
167 if (year < DateComponents::maximumYear())
168 return true;
169 if (month < maximumMonthInMaximumYear)
170 return true;
171 return monthDay <= maximumDayInMaximumMonth;
172}
173
174static bool withinHTMLDateLimits(int year, int month, int monthDay, int hour, int minute, int second, int millisecond)
175{
176 if (year < DateComponents::minimumYear())
177 return false;
178 if (year < DateComponents::maximumYear())
179 return true;
180 if (month < maximumMonthInMaximumYear)
181 return true;
182 if (monthDay < maximumDayInMaximumMonth)
183 return true;
184 if (monthDay > maximumDayInMaximumMonth)
185 return false;
186 // (year, month, monthDay) = (maximumYear, maximumMonthInMaximumYear, maximumDayInMaximumMonth)
187 return !hour && !minute && !second && !millisecond;
188}

238bool DateComponents::addDay(int dayDiff)
239{
240 ASSERT(m_monthDay);
241
242 int day = m_monthDay + dayDiff;
243 if (day > maxDayOfMonth(m_year, m_month)) {
244 day = m_monthDay;
245 int year = m_year;
246 int month = m_month;
247 int maxDay = maxDayOfMonth(year, month);
248 for (; dayDiff > 0; --dayDiff) {
249 ++day;
250 if (day > maxDay) {
251 day = 1;
252 ++month;
253 if (month >= 12) { // month is 0-origin.
254 month = 0;
255 ++year;
256 }
257 maxDay = maxDayOfMonth(year, month);
258 }
259 }
260 if (!withinHTMLDateLimits(year, month, day))
261 return false;
262 m_year = year;
263 m_month = month;
264 } else if (day < 1) {
265 int month = m_month;
266 int year = m_year;
267 day = m_monthDay;
268 for (; dayDiff < 0; ++dayDiff) {
269 --day;
270 if (day < 1) {
271 --month;
272 if (month < 0) {
273 month = 11;
274 --year;
275 }
276 day = maxDayOfMonth(year, month);
277 }
278 }
279 if (!withinHTMLDateLimits(year, month, day))
280 return false;
281 m_year = year;
282 m_month = month;
283 } else {
284 if (!withinHTMLDateLimits(m_year, m_month, day))
285 return false;
286 }
287 m_monthDay = day;
288 return true;
289}

291bool DateComponents::addMinute(int minute)
292{
293 // This function is used to adjust timezone offset. So m_year, m_month,
294 // m_monthDay have values between the lower and higher limits.
295 ASSERT(withinHTMLDateLimits(m_year, m_month, m_monthDay));
296
297 int carry;
298 // minute can be negative or greater than 59.
299 minute += m_minute;
300 if (minute > 59) {
301 carry = minute / 60;
302 minute = minute % 60;
303 } else if (minute < 0) {
304 carry = (59 - minute) / 60;
305 minute += carry * 60;
306 carry = -carry;
307 ASSERT(minute >= 0 && minute <= 59);
308 } else {
309 if (!withinHTMLDateLimits(m_year, m_month, m_monthDay, m_hour, minute, m_second, m_millisecond))
310 return false;
311 m_minute = minute;
312 return true;
313 }
314
315 int hour = m_hour + carry;
316 if (hour > 23) {
317 carry = hour / 24;
318 hour = hour % 24;
319 } else if (hour < 0) {
320 carry = (23 - hour) / 24;
321 hour += carry * 24;
322 carry = -carry;
323 ASSERT(hour >= 0 && hour <= 23);
324 } else {
325 if (!withinHTMLDateLimits(m_year, m_month, m_monthDay, hour, minute, m_second, m_millisecond))
326 return false;
327 m_minute = minute;
328 m_hour = hour;
329 return true;
330 }
331 if (!addDay(carry))
332 return false;
333 if (!withinHTMLDateLimits(m_year, m_month, m_monthDay, hour, minute, m_second, m_millisecond))
334 return false;
335 m_minute = minute;
336 m_hour = hour;
337 return true;
338}

340// Parses a timezone part, and adjust year, month, monthDay, hour, minute, second, millisecond.
341template<typename CharacterType> bool DateComponents::parseTimeZone(StringParsingBuffer<CharacterType>& buffer)
342{
343 if (skipExactly(buffer, 'Z'))
344 return true;
345
346 bool minus;
347 if (skipExactly(buffer, '+'))
348 minus = false;
349 else if (skipExactly(buffer, '-'))
350 minus = true;
351 else
352 return false;
353
354 auto hour = parseIntWithinLimits(buffer, 2, 0, 23);
355 if (!hour)
356 return false;
357
358 if (!skipExactly(buffer, ':'))
359 return false;
360
361 auto minute = parseIntWithinLimits(buffer, 2, 0, 59);
362 if (!minute)
363 return false;
364
365 if (minus) {
366 *hour = -*hour;
367 *minute = -*minute;
368 }
369
370 // Subtract the timezone offset.
371 if (!addMinute(-(*hour * 60 + *minute)))
372 return false;
373 return true;
374}

376template<typename CharacterType> bool DateComponents::parseMonth(StringParsingBuffer<CharacterType>& buffer)
377{
378 if (!parseYear(buffer))
379 return false;
380
381 if (!skipExactly(buffer, '-'))
382 return false;
383
384 auto month = parseIntWithinLimits(buffer, 2, 1, 12);
385 if (!month)
386 return false;
387 --*month;
388
389 if (!withinHTMLDateLimits(m_year, *month))
390 return false;
391
392 m_month = *month;
393 m_type = DateComponentsType::Month;
394 return true;
395}

397template<typename CharacterType> bool DateComponents::parseDate(StringParsingBuffer<CharacterType>& buffer)
398{
399 if (!parseMonth(buffer))
400 return false;
401
402 if (!skipExactly(buffer, '-'))
403 return false;
404
405 auto day = parseIntWithinLimits(buffer, 2, 1, maxDayOfMonth(m_year, m_month));
406 if (!day)
407 return false;
408
409 if (!withinHTMLDateLimits(m_year, m_month, *day))
410 return false;
411
412 m_monthDay = *day;
413 m_type = DateComponentsType::Date;
414 return true;
415}

417template<typename CharacterType> bool DateComponents::parseWeek(StringParsingBuffer<CharacterType>& buffer)
418{
419 if (!parseYear(buffer))
420 return false;
421
422 if (!skipExactly(buffer, '-'))
423 return false;
424 if (!skipExactly(buffer, 'W'))
425 return false;
426
427 auto week = parseIntWithinLimits(buffer, 2, minimumWeekNumber, maxWeekNumberInYear());
428 if (!week)
429 return false;
430
431 if (m_year == maximumYear() && *week > maximumWeekInMaximumYear)
432 return false;
433
434 m_week = *week;
435 m_type = DateComponentsType::Week;
436 return true;
437}

439template<typename CharacterType> bool DateComponents::parseTime(StringParsingBuffer<CharacterType>& buffer)
440{
441 auto hour = parseIntWithinLimits(buffer, 2, 0, 23);
442 if (!hour)
443 return false;
444
445 if (!skipExactly(buffer, ':'))
446 return false;
447
448 auto minute = parseIntWithinLimits(buffer, 2, 0, 59);
449 if (!minute)
450 return false;
451
452 // Optional second part.
453 // Do not return with false because the part is optional.
454 std::optional<int> second;
455 std::optional<int> millisecond;
456
457 auto temporaryBuffer = buffer;
458 if (skipExactly(temporaryBuffer, ':')) {
459 second = parseIntWithinLimits(temporaryBuffer, 2, 0, 59);
460 if (second) {
461 // Only after successfully parsing and validating 'second', do we increment the real buffer.
462 buffer += temporaryBuffer.position() - buffer.position();
463
464 // Optional fractional second part.
465 if (skipExactly(temporaryBuffer, '.')) {
466 unsigned digitsLength = countDigits(temporaryBuffer);
467 if (digitsLength > 0) {
468 if (digitsLength == 1) {
469 millisecond = parseInt(temporaryBuffer, 1);
470 *millisecond *= 100;
471 } else if (digitsLength == 2) {
472 millisecond = parseInt(temporaryBuffer, 2);
473 *millisecond *= 10;
474 } else {
475 // Regardless of the number of digits, we only ever parse at most 3. All other
476 // digits after that are ignored, but the buffer is incremented as if they were
477 // all parsed.
478 millisecond = parseInt(temporaryBuffer, 3);
479 }
480
481 // Due to the countDigits above, the parseInt calls should all be successful.
482 ASSERT(millisecond);
483
484 // Only after successfully parsing and validating 'millisecond', do we increment the real buffer.
485 buffer += 1 + digitsLength;
486 }
487 }
488 }
489 }
490
491 m_hour = *hour;
492 m_minute = *minute;
493 m_second = second.value_or(0);
494 m_millisecond = millisecond.value_or(0);
495 m_type = DateComponentsType::Time;
496 return true;
497}

499template<typename CharacterType> bool DateComponents::parseDateTimeLocal(StringParsingBuffer<CharacterType>& buffer)

501 if (!parseDate(buffer))

517static inline double positiveFmod(double value, double divider)
518{
519 double remainder = fmod(value, divider);
520 return remainder < 0 ? remainder + divider : remainder;
521}

573void DateComponents::setMillisecondsSinceMidnightInternal(double msInDay)
574{
575 ASSERT(msInDay >= 0 && msInDay < msPerDay);
576 m_millisecond = static_cast<int>(fmod(msInDay, msPerSecond));
577 double value = floor(msInDay / msPerSecond);
578 m_second = static_cast<int>(fmod(value, secondsPerMinute));
579 value = floor(value / secondsPerMinute);
580 m_minute = static_cast<int>(fmod(value, minutesPerHour));
581 m_hour = static_cast<int>(value / minutesPerHour);
582}

584bool DateComponents::setMillisecondsSinceEpochForDateInternal(double ms)
585{
586 m_year = msToYear(ms);
587 int yearDay = dayInYear(ms, m_year);
588 m_month = monthFromDayInYear(yearDay, isLeapYear(m_year));
589 m_monthDay = dayInMonthFromDayInYear(yearDay, isLeapYear(m_year));
590 return true;
591}

644bool DateComponents::setMonthsSinceEpoch(double months)
645{
646 if (!std::isfinite(months))
647 return false;
648 months = round(months);
649 double doubleMonth = positiveFmod(months, 12);
650 double doubleYear = 1970 + (months - doubleMonth) / 12;
651 if (doubleYear < minimumYear() || maximumYear() < doubleYear)
652 return false;
653 int year = static_cast<int>(doubleYear);
654 int month = static_cast<int>(doubleMonth);
655 if (!withinHTMLDateLimits(year, month))
656 return false;
657 m_year = year;
658 m_month = month;
659 m_type = DateComponentsType::Month;
660 return true;
661}

663// Offset from January 1st to Monday of the ISO 8601's first week.
664// ex. If January 1st is Friday, such Monday is 3 days later. Returns 3.
665static int offsetTo1stWeekStart(int year)
666{
667 int offsetTo1stWeekStart = 1 - dayOfWeek(year, 0, 1);
668 if (offsetTo1stWeekStart <= -4)
669 offsetTo1stWeekStart += 7;
670 return offsetTo1stWeekStart;
671}
672
673bool DateComponents::setMillisecondsSinceEpochForWeek(double ms)
674{
675 m_type = DateComponentsType::Invalid;
676 if (!std::isfinite(ms))
677 return false;
678 ms = round(ms);
679
680 m_year = msToYear(ms);
681 if (m_year < minimumYear() || m_year > maximumYear())
682 return false;
683
684 int yearDay = dayInYear(ms, m_year);
685 int offset = offsetTo1stWeekStart(m_year);
686 if (yearDay < offset) {
687 // The day belongs to the last week of the previous year.
688 m_year--;
689 if (m_year <= minimumYear())
690 return false;
691 m_week = maxWeekNumberInYear();
692 } else {
693 m_week = ((yearDay - offset) / 7) + 1;
694 if (m_week > maxWeekNumberInYear()) {
695 m_year++;
696 m_week = 1;
697 }
698 if (m_year > maximumYear() || (m_year == maximumYear() && m_week > maximumWeekInMaximumYear))
699 return false;
700 }
701 m_type = DateComponentsType::Week;
702 return true;
703}

705double DateComponents::millisecondsSinceEpochForTime() const
706{
707 ASSERT(m_type == DateComponentsType::Time || m_type == DateComponentsType::DateTimeLocal);
708 return ((m_hour * minutesPerHour + m_minute) * secondsPerMinute + m_second) * msPerSecond + m_millisecond;
709}
710
711double DateComponents::millisecondsSinceEpoch() const
712{
713 switch (m_type) {
714 case DateComponentsType::Date:
715 return dateToDaysFrom1970(m_year, m_month, m_monthDay) * msPerDay;
716 case DateComponentsType::DateTimeLocal:
717 return dateToDaysFrom1970(m_year, m_month, m_monthDay) * msPerDay + millisecondsSinceEpochForTime();
718 case DateComponentsType::Month:
719 return dateToDaysFrom1970(m_year, m_month, 1) * msPerDay;
720 case DateComponentsType::Time:
721 return millisecondsSinceEpochForTime();
722 case DateComponentsType::Week:
723 return (dateToDaysFrom1970(m_year, 0, 1) + offsetTo1stWeekStart(m_year) + (m_week - 1) * 7) * msPerDay;
724 case DateComponentsType::Invalid:
725 break;
726 }
727 ASSERT_NOT_REACHED();
728 return invalidMilliseconds();
729}

731double DateComponents::monthsSinceEpoch() const
732{
733 ASSERT(m_type == DateComponentsType::Month);
734 return (m_year - 1970) * 12 + m_month;
735}

737String DateComponents::toStringForTime(SecondFormat format) const
738{
739 ASSERT(m_type == DateComponentsType::DateTimeLocal || m_type == DateComponentsType::Time);
740 SecondFormat effectiveFormat = format;
741 if (m_millisecond)
742 effectiveFormat = SecondFormat::Millisecond;
743 else if (format == SecondFormat::None && m_second)
744 effectiveFormat = SecondFormat::Second;
745
746 switch (effectiveFormat) {
747 default:
748 ASSERT_NOT_REACHED();
749#if !ASSERT_ENABLED
750 FALLTHROUGH; // To None.
751#endif
752 case SecondFormat::None:
753 return makeString(pad('0', 2, m_hour), ':', pad('0', 2, m_minute));
754 case SecondFormat::Second:
755 return makeString(pad('0', 2, m_hour), ':', pad('0', 2, m_minute), ':', pad('0', 2, m_second));
756 case SecondFormat::Millisecond:
757 return makeString(pad('0', 2, m_hour), ':', pad('0', 2, m_minute), ':', pad('0', 2, m_second), '.', pad('0', 3, m_millisecond));
758 }
759}

761String DateComponents::toString(SecondFormat format) const
762{
763 switch (m_type) {
764 case DateComponentsType::Date:
765 return makeString(pad('0', 4, m_year), '-', pad('0', 2, m_month + 1), '-', pad('0', 2, m_monthDay));
766 case DateComponentsType::DateTimeLocal:
767 return makeString(pad('0', 4, m_year), '-', pad('0', 2, m_month + 1), '-', pad('0', 2, m_monthDay), 'T', toStringForTime(format));
768 case DateComponentsType::Month:
769 return makeString(pad('0', 4, m_year), '-', pad('0', 2, m_month + 1));
770 case DateComponentsType::Time:
771 return toStringForTime(format);
772 case DateComponentsType::Week:
773 return makeString(pad('0', 4, m_year), "-W", pad('0', 2, m_week));
774 case DateComponentsType::Invalid:
775 break;
776 }
777 ASSERT_NOT_REACHED();
778 return "(Invalid DateComponents)"_str;
779}
