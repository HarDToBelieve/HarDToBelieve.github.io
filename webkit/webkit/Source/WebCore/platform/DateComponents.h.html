37namespace WebCore {

39enum class SecondFormat : uint8_t {
40 None, // Suppress the second part and the millisecond part if they are 0.
41 Second, // Always show the second part, and suppress the millisecond part if it is 0.
42 Millisecond // Always show the second part and the millisecond part.
43};
44
45enum class DateComponentsType : uint8_t {
46 Invalid,
47 Date,
48 DateTimeLocal,
49 Month,
50 Time,
51 Week,
52};
53
54// A DateComponents instance represents one of the following date and time combinations:
55// * Month type: year-month
56// * Date type: year-month-day
57// * Week type: year-week
58// * Time type: hour-minute-second-millisecond
59// * DateTime or DateTimeLocal type: year-month-day hour-minute-second-millisecond
60class DateComponents {
61public:
62 DateComponents()
63 : m_millisecond(0)
64 , m_second(0)
65 , m_minute(0)
66 , m_hour(0)
67 , m_monthDay(0)
68 , m_month(0)
69 , m_year(0)
70 , m_week(0)
71 , m_type(DateComponentsType::Invalid)
72 {
73 }
74
75 int millisecond() const { return m_millisecond; }
76 int second() const { return m_second; }
77 int minute() const { return m_minute; }
78 int hour() const { return m_hour; }
79 int monthDay() const { return m_monthDay; }
80 int month() const { return m_month; }
81 int fullYear() const { return m_year; }
82 int week() const { return m_week; }
83 DateComponentsType type() const { return m_type; }
84
85 // Returns an ISO 8601 representation for this instance.
86 // The format argument is valid for DateTimeLocal and Time types.
87 String toString(SecondFormat = SecondFormat::None) const;
88
89 // Sets year and month.
90 static std::optional<DateComponents> fromParsingMonth(StringView);
91 // Sets year, month and monthDay.
92 static std::optional<DateComponents> fromParsingDate(StringView);
93 // Sets year and week.
94 static std::optional<DateComponents> fromParsingWeek(StringView);
95 // Sets hour, minute, second and millisecond.
96 static std::optional<DateComponents> fromParsingTime(StringView);
97 // Sets year, month, monthDay, hour, minute, second and millisecond.
98 static std::optional<DateComponents> fromParsingDateTimeLocal(StringView);
99
100
101 // For Date type. Updates m_year, m_month and m_monthDay.
102 static std::optional<DateComponents> fromMillisecondsSinceEpochForDate(double ms);
103 // For DateTimeLocal type. Updates m_year, m_month, m_monthDay, m_hour, m_minute, m_second and m_millisecond.
104 static std::optional<DateComponents> fromMillisecondsSinceEpochForDateTimeLocal(double ms);
105 // For Month type. Updates m_year and m_month.
106 static std::optional<DateComponents> fromMillisecondsSinceEpochForMonth(double ms);
107 // For Week type. Updates m_year and m_week.
108 static std::optional<DateComponents> fromMillisecondsSinceEpochForWeek(double ms);
109
110 // For Time type. Updates m_hour, m_minute, m_second and m_millisecond.
111 static std::optional<DateComponents> fromMillisecondsSinceMidnight(double ms);
112
113 // Another initializer for Month type. Updates m_year and m_month.
114 static std::optional<DateComponents> fromMonthsSinceEpoch(double months);
115
116 // Returns the number of milliseconds from 1970-01-01 00:00:00 UTC.
117 // For a DateComponents initialized with parseDateTimeLocal(),
118 // millisecondsSinceEpoch() returns a value for UTC timezone.
119 double millisecondsSinceEpoch() const;
120
121 // Returns the number of months from 1970-01.
122 // Do not call this for types other than Month.
123 double monthsSinceEpoch() const;
124
125 static inline double invalidMilliseconds() { return std::numeric_limits<double>::quiet_NaN(); }
126
127 // Minimum and maxmimum limits for setMillisecondsSince*(),
128 // setMonthsSinceEpoch(), millisecondsSinceEpoch(), and monthsSinceEpoch().
129 static constexpr inline double minimumDate() { return -62135596800000.0; } // 0001-01-01T00:00Z
130 static constexpr inline double minimumDateTime() { return -62135596800000.0; } // ditto.
131 static constexpr inline double minimumMonth() { return (1 - 1970) * 12.0 + 1 - 1; } // 0001-01
132 static constexpr inline double minimumTime() { return 0; } // 00:00:00.000
133 static constexpr inline double minimumWeek() { return -62135596800000.0; } // 0001-01-01, the first Monday of 0001.
134 static constexpr inline double maximumDate() { return 8640000000000000.0; } // 275760-09-13T00:00Z
135 static constexpr inline double maximumDateTime() { return 8640000000000000.0; } // ditto.
136 static constexpr inline double maximumMonth() { return (275760 - 1970) * 12.0 + 9 - 1; } // 275760-09
137 static constexpr inline double maximumTime() { return 86399999; } // 23:59:59.999
138 static constexpr inline double maximumWeek() { return 8639999568000000.0; } // 275760-09-08, the Monday of the week including 275760-09-13.
139
140 // HTML uses ISO-8601 format with year >= 1. Gregorian calendar started in
141 // 1582. However, we need to support 0001-01-01 in Gregorian calendar rule.
142 static constexpr inline int minimumYear() { return 1; }
143
144 // Date in ECMAScript can't represent dates later than 275760-09-13T00:00Z.
145 // So, we have the same upper limit in HTML5 date/time types.
146 static constexpr inline int maximumYear() { return 275760; }
147
148private:

152 template<typename CharacterType> bool parseDate(StringParsingBuffer<CharacterType>&);
153 template<typename CharacterType> bool parseWeek(StringParsingBuffer<CharacterType>&);
154 template<typename CharacterType> bool parseTime(StringParsingBuffer<CharacterType>&);
155 template<typename CharacterType> bool parseDateTimeLocal(StringParsingBuffer<CharacterType>&);

157 // The following setMillisecondsSinceEpochFor*() functions take
158 // the number of milliseconds since 1970-01-01 00:00:00.000 UTC as
159 // the argument, and update all fields for the corresponding
160 // DateComponents type. The functions return true if it succeeds, and
161 // false if they fail.
162 bool setMillisecondsSinceEpochForDate(double);
163 bool setMillisecondsSinceEpochForDateTimeLocal(double);
164 bool setMillisecondsSinceEpochForMonth(double);
165 bool setMillisecondsSinceEpochForWeek(double);
166 bool setMillisecondsSinceMidnight(double);
167 bool setMonthsSinceEpoch(double);
168 bool setMillisecondsSinceEpochForDateInternal(double);
169 void setMillisecondsSinceMidnightInternal(double);

171 // Returns the maximum week number in this DateComponents's year.
172 // The result is either of 52 and 53.
173 int maxWeekNumberInYear() const;
174
175 bool addDay(int);
176 bool addMinute(int);
177

179 double millisecondsSinceEpochForTime() const;

198 int m_hour;
199 int m_monthDay; // 1 - 31

201 int m_year; // 1582 -
202 int m_week; // 1 - 53
203
204 DateComponentsType m_type;
205};
