30
31#include <wtf/Assertions.h>
32#include <wtf/FileSystem.h>
33
34namespace WebCore {
35
36class WEBCORE_EXPORT FileHandle final {
37public:
38 FileHandle() = default;
39 FileHandle(const String& path, FileSystem::FileOpenMode);
40 FileHandle(const String& path, FileSystem::FileOpenMode, OptionSet<FileSystem::FileLockMode>);
41 FileHandle(const FileHandle& other) = delete;
42 FileHandle(FileHandle&& other);
43
44 ~FileHandle();
45
46 FileHandle& operator=(const FileHandle& other) = delete;
47 FileHandle& operator=(FileHandle&& other);
48
49 explicit operator bool() const;
50
51 bool open(const String& path, FileSystem::FileOpenMode);
52 bool open();
53 int read(void* data, int length);
54 int write(const void* data, int length);
55 bool printf(const char* format, ...) WTF_ATTRIBUTE_PRINTF(2, 3);
56 void close();
57
58private:
59 String m_path;
60 FileSystem::FileOpenMode m_mode { FileSystem::FileOpenMode::Read };
61 FileSystem::PlatformFileHandle m_fileHandle { FileSystem::invalidPlatformFileHandle };
62 OptionSet<FileSystem::FileLockMode> m_lockMode;
63 bool m_shouldLock { false };
64};

66} // namespace WebCore
