24#include "Length.h"
25#include "RectEdges.h"
26#include "WritingMode.h"

28namespace WebCore {
29
30class LengthBox : public RectEdges<Length> {
31public:
32 LengthBox()
33 : LengthBox(LengthType::Auto)
34 {
35 }
36
37 explicit LengthBox(LengthType type)
38 : RectEdges(Length(type), Length(type), Length(type), Length(type))
39 {
40 }
41
42 explicit LengthBox(int v)
43 : RectEdges(Length(v, LengthType::Fixed), Length(v, LengthType::Fixed), Length(v, LengthType::Fixed), Length(v, LengthType::Fixed))
44 {
45 }
46
47 LengthBox(int top, int right, int bottom, int left)
48 : RectEdges(Length(top, LengthType::Fixed), Length(right, LengthType::Fixed), Length(bottom, LengthType::Fixed), Length(left, LengthType::Fixed))
49 {
50 }
51
52 LengthBox(Length&& top, Length&& right, Length&& bottom, Length&& left)
53 : RectEdges { WTFMove(top), WTFMove(right), WTFMove(bottom), WTFMove(left) }
54 {
55 }
56
57 LengthBox(const LengthBox&) = default;
58
59 bool isZero() const
60 {
61 return top().isZero() && right().isZero() && bottom().isZero() && left().isZero();
62 }
63};
64
65using LayoutBoxExtent = RectEdges<LayoutUnit>;
66using FloatBoxExtent = RectEdges<float>;
67
68WTF::TextStream& operator<<(WTF::TextStream&, const LengthBox&);
69WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, const FloatBoxExtent&);

71} // namespace WebCore
