26#pragma once
27
28#include "Length.h"
29
30namespace WebCore {
31
32struct BlendingContext;
33
34struct LengthPoint {
35public:
36 LengthPoint()
37 {
38 }
39
40 LengthPoint(Length x, Length y)
41 : m_x(WTFMove(x))
42 , m_y(WTFMove(y))
43 {
44 }
45
46 bool operator==(const LengthPoint& o) const
47 {
48 return m_x == o.m_x && m_y == o.m_y;
49 }
50
51 bool operator!=(const LengthPoint& o) const
52 {
53 return !(*this == o);
54 }
55
56 void setX(Length x) { m_x = WTFMove(x); }
57 const Length& x() const { return m_x; }
58
59 void setY(Length y) { m_y = WTFMove(y); }
60 const Length& y() const { return m_y; }
61
62private:
63 // FIXME: it would be nice to pack the two Lengths together better somehow (to avoid padding between them).
64 Length m_x;
65 Length m_y;
66};
67
68inline LengthPoint blend(const LengthPoint& from, const LengthPoint& to, const BlendingContext& context)
69{
70 return LengthPoint(blend(from.x(), to.x(), context), blend(from.y(), to.y(), context));
71}

73WTF::TextStream& operator<<(WTF::TextStream&, const LengthPoint&);
74
75} // namespace WebCore
