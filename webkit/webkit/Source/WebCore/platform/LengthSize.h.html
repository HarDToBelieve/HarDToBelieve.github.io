20#pragma once
21
22#include "Length.h"
23
24namespace WebCore {
25
26struct BlendingContext;
27
28struct LengthSize {
29 Length width;
30 Length height;
31
32 bool isEmpty() const { return width.isZero() || height.isZero(); }
33 bool isZero() const { return width.isZero() && height.isZero(); }
34};
35
36ALWAYS_INLINE bool operator==(const LengthSize& a, const LengthSize& b)
37{
38 return a.width == b.width && a.height == b.height;
39}
40
41inline LengthSize blend(const LengthSize& from, const LengthSize& to, const BlendingContext& context)
42{
43 return { blend(from.width, to.width, context), blend(from.height, to.height, context) };
44}
45
46WTF::TextStream& operator<<(WTF::TextStream&, const LengthSize&);
47
48} // namespace WebCore
