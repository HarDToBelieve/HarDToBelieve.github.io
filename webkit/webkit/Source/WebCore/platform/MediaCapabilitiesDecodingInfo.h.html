26#pragma once
27
28#include "MediaCapabilitiesInfo.h"
29#include "MediaDecodingConfiguration.h"
30
31namespace WebCore {
32
33struct MediaCapabilitiesDecodingInfo : MediaCapabilitiesInfo {
34 // FIXME(C++17): remove the following constructors once all compilers support extended
35 // aggregate initialization:
36 // <http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2015/p0017r1.html>
37 MediaCapabilitiesDecodingInfo() = default;
38 MediaCapabilitiesDecodingInfo(MediaDecodingConfiguration&& supportedConfiguration)
39 : MediaCapabilitiesDecodingInfo({ }, WTFMove(supportedConfiguration))
40 {
41 }
42 MediaCapabilitiesDecodingInfo(MediaCapabilitiesInfo&& info, MediaDecodingConfiguration&& supportedConfiguration)
43 : MediaCapabilitiesInfo(WTFMove(info))
44 , supportedConfiguration(WTFMove(supportedConfiguration))
45 {
46 }
47
48 MediaDecodingConfiguration supportedConfiguration;
49
50 template<class Encoder> void encode(Encoder&) const;
51 template<class Decoder> static std::optional<MediaCapabilitiesDecodingInfo> decode(Decoder&);
52};
53
54template<class Encoder>
55void MediaCapabilitiesDecodingInfo::encode(Encoder& encoder) const
56{
57 MediaCapabilitiesInfo::encode(encoder);
58 encoder << supportedConfiguration;
59}
60
61template<class Decoder>
62std::optional<MediaCapabilitiesDecodingInfo> MediaCapabilitiesDecodingInfo::decode(Decoder& decoder)
63{
64 auto info = MediaCapabilitiesInfo::decode(decoder);
65 if (!info)
66 return std::nullopt;
67
68 std::optional<MediaDecodingConfiguration> supportedConfiguration;
69 decoder >> supportedConfiguration;
70 if (!supportedConfiguration)
71 return std::nullopt;
72
73 return MediaCapabilitiesDecodingInfo(
74 WTFMove(*info),
75 WTFMove(*supportedConfiguration)
76 );
77}

79} // namespace WebCore
