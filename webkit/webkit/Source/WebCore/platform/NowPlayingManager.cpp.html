44void NowPlayingManager::addClient(Client& client)
45{
46 m_client = makeWeakPtr(client);
47 ensureRemoteCommandListenerCreated();
48}

50void NowPlayingManager::removeClient(Client& client)
51{
52 if (m_client.get() != &client)
53 return;
54
55 m_remoteCommandListener = nullptr;
56 m_client.clear();
57 m_nowPlayingInfo = { };
58
59 clearNowPlayingInfo();
60}

75bool NowPlayingManager::setNowPlayingInfo(const NowPlayingInfo& nowPlayingInfo)
76{
77 if (m_nowPlayingInfo && *m_nowPlayingInfo == nowPlayingInfo)
78 return false;
79 m_nowPlayingInfo = nowPlayingInfo;
80 // We do not want to send the artwork's image over each time nowPlayingInfo gets updated.
81 // So if present we store it once locally. On the receiving end, a null imageData indicates to use the cached image.
82 if (!nowPlayingInfo.artwork)
83 m_nowPlayingInfoArtwork = { };
84 else if (!m_nowPlayingInfoArtwork || nowPlayingInfo.artwork->src != m_nowPlayingInfoArtwork->src)
85 m_nowPlayingInfoArtwork = ArtworkCache { nowPlayingInfo.artwork->src, nowPlayingInfo.artwork->imageData };
86 else
87 m_nowPlayingInfo->artwork->imageData = nullptr;
88
89 setNowPlayingInfoPrivate(*m_nowPlayingInfo);
90 m_setAsNowPlayingApplication = true;
91 return true;
92}
93
94void NowPlayingManager::setNowPlayingInfoPrivate(const NowPlayingInfo& nowPlayingInfo)
95{
96 setSupportsSeeking(nowPlayingInfo.supportsSeeking);
97#if PLATFORM(COCOA)
98 if (nowPlayingInfo.artwork && !nowPlayingInfo.artwork->imageData) {
99 ASSERT(m_nowPlayingInfoArtwork, "cached value must have been initialized");
100 NowPlayingInfo nowPlayingInfoRebuilt = nowPlayingInfo;
101 nowPlayingInfoRebuilt.artwork->imageData = m_nowPlayingInfoArtwork->imageData;
102 MediaSessionManagerCocoa::setNowPlayingInfo(!m_setAsNowPlayingApplication, nowPlayingInfoRebuilt);
103 return;
104 }
105 MediaSessionManagerCocoa::setNowPlayingInfo(!m_setAsNowPlayingApplication, nowPlayingInfo);
106#else
107 (void)nowPlayingInfo;
108#endif
109}
