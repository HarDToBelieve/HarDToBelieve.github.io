33PasteboardWriterData::PasteboardWriterData() = default;

39bool PasteboardWriterData::isEmpty() const
40{
41 return !m_plainText && !m_url && !m_webContent;
42}
43
44void PasteboardWriterData::setPlainText(PlainText plainText)
45{
46 ASSERT(!m_plainText);
47
48 m_plainText = WTFMove(plainText);
49}
50
51void PasteboardWriterData::setURLData(URLData url)
52{
53 ASSERT(!m_url);
54
55 m_url = WTFMove(url);
56}
57
58void PasteboardWriterData::setWebContent(WebContent webContent)
59{
60 ASSERT(!m_webContent);
61
62 m_webContent = webContent;
63}
