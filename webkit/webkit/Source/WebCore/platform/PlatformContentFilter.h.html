26#ifndef PlatformContentFilter_h
27#define PlatformContentFilter_h
28
29#include <wtf/Ref.h>
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34class ContentFilterUnblockHandler;
35class ResourceRequest;
36class ResourceResponse;
37class SharedBuffer;
38
39class PlatformContentFilter {
40 WTF_MAKE_FAST_ALLOCATED;
41 WTF_MAKE_NONCOPYABLE(PlatformContentFilter);
42
43public:
44 enum class State {
45 Stopped,
46 Filtering,
47 Allowed,
48 Blocked,
49 };
50
51 bool needsMoreData() const { return m_state == State::Filtering; }
52 bool didBlockData() const { return m_state == State::Blocked; }
53
54 virtual ~PlatformContentFilter() = default;
55 virtual void willSendRequest(ResourceRequest&, const ResourceResponse&) = 0;
56 virtual void responseReceived(const ResourceResponse&) = 0;
57 virtual void addData(const uint8_t* data, int length) = 0;
58 virtual void finishedAddingData() = 0;
59 virtual Ref<SharedBuffer> replacementData() const = 0;
60#if ENABLE(CONTENT_FILTERING)
61 virtual ContentFilterUnblockHandler unblockHandler() const = 0;
62#endif
63 virtual String unblockRequestDeniedScript() const { return emptyString(); }
64
65protected:
66 PlatformContentFilter() = default;
67 State m_state { State::Filtering };
68};
69
70} // namespace WebCore
71
72#endif // PlatformContentFilter_h
