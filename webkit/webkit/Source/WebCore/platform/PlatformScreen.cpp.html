33namespace WebCore {
34
35static ScreenProperties& screenProperties()
36{
37 static NeverDestroyed<ScreenProperties> screenProperties;
38 return screenProperties;
39}
40
41const ScreenProperties& getScreenProperties()
42{
43 return screenProperties();
44}
45
46PlatformDisplayID primaryScreenDisplayID()
47{
48 return screenProperties().primaryDisplayID;
49}
50
51void setScreenProperties(const ScreenProperties& properties)
52{
53 screenProperties() = properties;
54}
55
56const ScreenData* screenData(PlatformDisplayID screenDisplayID)
57{
58 if (screenProperties().screenDataMap.isEmpty())
59 return nullptr;
60
61 // Return property of the first screen if the screen is not found in the map.
62 if (auto displayID = screenDisplayID ? screenDisplayID : primaryScreenDisplayID()) {
63 auto properties = screenProperties().screenDataMap.find(displayID);
64 if (properties != screenProperties().screenDataMap.end())
65 return &properties->value;
66 }
67
68 // Last resort: use the first item in the screen list.
69 return &screenProperties().screenDataMap.begin()->value;
70}
