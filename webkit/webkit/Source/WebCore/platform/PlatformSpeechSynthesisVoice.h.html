34class PlatformSpeechSynthesisVoice : public RefCounted<PlatformSpeechSynthesisVoice> {
35public:
36 WEBCORE_EXPORT static Ref<PlatformSpeechSynthesisVoice> create(const String& voiceURI, const String& name, const String& lang, bool localService, bool isDefault);
37 static Ref<PlatformSpeechSynthesisVoice> create();
38
39 const String& voiceURI() const { return m_voiceURI; }
40 void setVoiceURI(const String& voiceURI) { m_voiceURI = voiceURI; }
41
42 const String& name() const { return m_name; }
43 void setName(const String& name) { m_name = name; }
44
45 const String& lang() const { return m_lang; }
46 void setLang(const String& lang) { m_lang = lang; }
47
48 bool localService() const { return m_localService; }
49 void setLocalService(bool localService) { m_localService = localService; }
50
51 bool isDefault() const { return m_default; }
52 void setIsDefault(bool isDefault) { m_default = isDefault; }
53
54private:
55 PlatformSpeechSynthesisVoice(const String& voiceURI, const String& name, const String& lang, bool localService, bool isDefault);
56 PlatformSpeechSynthesisVoice() = default;
57
58 String m_voiceURI;
59 String m_name;
60 String m_lang;
61 bool m_localService { false };
62 bool m_default { false };
63};
