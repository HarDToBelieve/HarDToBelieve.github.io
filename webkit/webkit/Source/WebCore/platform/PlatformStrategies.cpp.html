26#include "config.h"
27#include "PlatformStrategies.h"
28
29namespace WebCore {
30
31static PlatformStrategies* s_platformStrategies;
32
33bool hasPlatformStrategies()
34{
35 return !!s_platformStrategies;
36}
37
38PlatformStrategies* platformStrategies()
39{
40 ASSERT(s_platformStrategies);
41
42 return s_platformStrategies;
43}
44
45void setPlatformStrategies(PlatformStrategies* platformStrategies)
46{
47 if (!s_platformStrategies) {
48 s_platformStrategies = platformStrategies;
49 return;
50 }
51
52 // FIXME: This happens when mixing different platform strategies, and we should probably
53 // throw an exception here in release builds.
54 ASSERT(platformStrategies == s_platformStrategies);
55}
56
57} // namespace WebCore
