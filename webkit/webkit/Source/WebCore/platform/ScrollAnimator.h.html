54class ScrollAnimator : private ScrollControllerClient {
55 WTF_MAKE_FAST_ALLOCATED;
56public:
57 static std::unique_ptr<ScrollAnimator> create(ScrollableArea&);
58
59 explicit ScrollAnimator(ScrollableArea&);
60 virtual ~ScrollAnimator();
61
62 enum ScrollBehavior {
63 DoDirectionalSnapping = 1 << 0,
64 NeverAnimate = 1 << 1,
65 };
66
67 // Computes a scroll destination for the given parameters. Returns false if
68 // already at the destination. Otherwise, starts scrolling towards the
69 // destination and returns true. Scrolling may be immediate or animated.
70 // The base class implementation always scrolls immediately, never animates.
71 virtual bool scroll(ScrollbarOrientation, ScrollGranularity, float step, float multiplier, OptionSet<ScrollBehavior>);
72
73 bool scrollToOffsetWithoutAnimation(const FloatPoint&, ScrollClamping = ScrollClamping::Clamped);
74 virtual bool scrollToPositionWithoutAnimation(const FloatPoint& position, ScrollClamping = ScrollClamping::Clamped);
75
76 bool scrollToOffsetWithAnimation(const FloatPoint&);
77 virtual bool scrollToPositionWithAnimation(const FloatPoint&);
78
79 ScrollableArea& scrollableArea() const { return m_scrollableArea; }
80
81 virtual bool handleWheelEvent(const PlatformWheelEvent&);
82
83#if ENABLE(TOUCH_EVENTS)
84 virtual bool handleTouchEvent(const PlatformTouchEvent&);
85#endif
86
87#if PLATFORM(COCOA)
88 virtual void handleWheelEventPhase(PlatformWheelEventPhase) { }
89#endif
90
91 void setCurrentPosition(const FloatPoint&);
92 const FloatPoint& currentPosition() const { return m_currentPosition; }
93
94 virtual void cancelAnimations();
95
96 virtual void contentAreaWillPaint() const { }
97 virtual void mouseEnteredContentArea() { }
98 virtual void mouseExitedContentArea() { }
99 virtual void mouseMovedInContentArea() { }
100 virtual void mouseEnteredScrollbar(Scrollbar*) const { }
101 virtual void mouseExitedScrollbar(Scrollbar*) const { }
102 virtual void mouseIsDownInScrollbar(Scrollbar*, bool) const { }
103 virtual void willStartLiveResize() { }
104 virtual void contentsResized() const;
105 virtual void willEndLiveResize();
106 virtual void contentAreaDidShow() { }
107 virtual void contentAreaDidHide() { }
108
109 virtual void lockOverlayScrollbarStateToHidden(bool) { }
110 virtual bool scrollbarsCanBeActive() const { return true; }
111
112 virtual void didAddVerticalScrollbar(Scrollbar*);
113 virtual void willRemoveVerticalScrollbar(Scrollbar*) { }
114 virtual void didAddHorizontalScrollbar(Scrollbar*);
115 virtual void willRemoveHorizontalScrollbar(Scrollbar*) { }
116
117 virtual void invalidateScrollbarPartLayers(Scrollbar*) { }
118
119 virtual void verticalScrollbarLayerDidChange() { }
120 virtual void horizontalScrollbarLayerDidChange() { }
121
122 virtual bool shouldScrollbarParticipateInHitTesting(Scrollbar*) { return true; }
123
124 virtual void notifyContentAreaScrolled(const FloatSize& delta) { UNUSED_PARAM(delta); }
125
126 virtual bool isUserScrollInProgress() const { return false; }
127 virtual bool isRubberBandInProgress() const { return false; }
128 virtual bool isScrollSnapInProgress() const { return false; }
129
130 virtual String horizontalScrollbarStateForTesting() const { return emptyString(); }
131 virtual String verticalScrollbarStateForTesting() const { return emptyString(); }
132
133 void setWheelEventTestMonitor(RefPtr<WheelEventTestMonitor>&& testMonitor) { m_wheelEventTestMonitor = testMonitor; }
134
135 FloatPoint adjustScrollOffsetForSnappingIfNeeded(const FloatPoint& offset, ScrollSnapPointSelectionMethod);
136 float adjustScrollOffsetForSnappingIfNeeded(ScrollEventAxis, float newOffset, ScrollSnapPointSelectionMethod);
137
138 std::unique_ptr<ScrollControllerTimer> createTimer(Function<void()>&&) final;
139
140 void startAnimationCallback(ScrollController&) final;
141 void stopAnimationCallback(ScrollController&) final;
142
143 void scrollControllerAnimationTimerFired();
144
145 virtual bool processWheelEventForScrollSnap(const PlatformWheelEvent&) { return false; }
146 void updateScrollSnapState();
147 bool activeScrollSnapIndexDidChange() const;
148 std::optional<unsigned> activeScrollSnapIndexForAxis(ScrollEventAxis) const;
149 void setActiveScrollSnapIndexForAxis(ScrollEventAxis, std::optional<unsigned> index);
150 void setSnapOffsetsInfo(const LayoutScrollSnapOffsetsInfo&);
151 const LayoutScrollSnapOffsetsInfo* snapOffsetsInfo() const;
152 void resnapAfterLayout();
153
154 // ScrollControllerClient.
155 FloatPoint scrollOffset() const override;
156 void immediateScrollOnAxis(ScrollEventAxis, float delta) override;
157 float pageScaleFactor() const override;
158 LayoutSize scrollExtent() const override;
159 FloatSize viewportSize() const override;
160#if PLATFORM(MAC)
161 void deferWheelEventTestCompletionForReason(WheelEventTestMonitor::ScrollableAreaIdentifier, WheelEventTestMonitor::DeferReason) const override;
162 void removeWheelEventTestCompletionDeferralForReason(WheelEventTestMonitor::ScrollableAreaIdentifier, WheelEventTestMonitor::DeferReason) const override;
163#endif
164
165protected:
166 virtual void notifyPositionChanged(const FloatSize& delta);
167 void updateActiveScrollSnapIndexForOffset();
168
169 FloatPoint offsetFromPosition(const FloatPoint& position);
170 FloatPoint positionFromOffset(const FloatPoint& offset);
171 FloatSize deltaFromStep(ScrollbarOrientation, float step, float multiplier);
172
173 ScrollableArea& m_scrollableArea;
174 RefPtr<WheelEventTestMonitor> m_wheelEventTestMonitor;
175 ScrollController m_scrollController;
176 Timer m_scrollControllerAnimationTimer;
177 FloatPoint m_currentPosition;
178
179 std::unique_ptr<ScrollAnimation> m_scrollAnimation;
180};
