44void ScrollController::animationCallback(MonotonicTime currentTime)
45{
46 LOG_WITH_STREAM(Scrolling, stream << "ScrollController " << this << " animationCallback: isAnimatingRubberBand " << m_isAnimatingRubberBand << " isAnimatingScrollSnap " << m_isAnimatingScrollSnap);
47
48 updateScrollSnapAnimatingState(currentTime);
49 updateRubberBandAnimatingState(currentTime);
50}

52void ScrollController::startOrStopAnimationCallbacks()
53{
54 bool needsCallbacks = m_isAnimatingRubberBand || m_isAnimatingScrollSnap;
55 if (needsCallbacks == m_isRunningAnimatingCallback)
56 return;
57
58 if (needsCallbacks) {
59 m_client.startAnimationCallback(*this);
60 m_isRunningAnimatingCallback = true;
61 return;
62 }
63
64 m_client.stopAnimationCallback(*this);
65 m_isRunningAnimatingCallback = false;
66}

68void ScrollController::setIsAnimatingRubberBand(bool isAnimatingRubberBand)
69{

86bool ScrollController::usesScrollSnap() const
87{
88 return !!m_scrollSnapState;
89}
90
91void ScrollController::setSnapOffsetsInfo(const LayoutScrollSnapOffsetsInfo& snapOffsetInfo)
92{
93 if (snapOffsetInfo.isEmpty()) {
94 m_scrollSnapState = nullptr;
95 return;
96 }
97
98 bool shouldComputeCurrentSnapIndices = !m_scrollSnapState;
99 if (!m_scrollSnapState)
100 m_scrollSnapState = makeUnique<ScrollSnapAnimatorState>();
101
102 m_scrollSnapState->setSnapOffsetInfo(snapOffsetInfo);
103
104 if (shouldComputeCurrentSnapIndices)
105 updateActiveScrollSnapIndexForClientOffset();
106
107 LOG_WITH_STREAM(ScrollSnap, stream << "ScrollController " << this << " setSnapOffsetsInfo new state: " << ValueOrNull(m_scrollSnapState.get()));
108}

130void ScrollController::setNearestScrollSnapIndexForAxisAndOffset(ScrollEventAxis axis, int offset)
131{
132 if (!usesScrollSnap())
133 return;
134
135 float scaleFactor = m_client.pageScaleFactor();
136 ScrollSnapAnimatorState& snapState = *m_scrollSnapState;
137
138 auto snapOffsets = snapState.snapOffsetsForAxis(axis);
139 LayoutSize viewportSize(m_client.viewportSize().width(), m_client.viewportSize().height());
140 std::optional<unsigned> activeIndex;
141 if (snapOffsets.size())
142 activeIndex = snapState.snapOffsetInfo().closestSnapOffset(axis, viewportSize, LayoutUnit(offset / scaleFactor), 0).second;
143
144 if (activeIndex == activeScrollSnapIndexForAxis(axis))
145 return;
146
147 m_activeScrollSnapIndexDidChange = true;
148 setActiveScrollSnapIndexForAxis(axis, activeIndex);
149}

151float ScrollController::adjustScrollDestination(ScrollEventAxis axis, float destinationOffset, float velocity, std::optional<float> originalOffset)
152{
153 if (!usesScrollSnap())
154 return destinationOffset;
155
156 ScrollSnapAnimatorState& snapState = *m_scrollSnapState;
157 auto snapOffsets = snapState.snapOffsetsForAxis(axis);
158 if (!snapOffsets.size())
159 return destinationOffset;
160
161 std::optional<LayoutUnit> originalOffsetInLayoutUnits;
162 if (originalOffset)
163 originalOffsetInLayoutUnits = LayoutUnit(*originalOffset / m_client.pageScaleFactor());
164 LayoutSize viewportSize(m_client.viewportSize().width(), m_client.viewportSize().height());
165 LayoutUnit offset = snapState.snapOffsetInfo().closestSnapOffset(axis, viewportSize, LayoutUnit(destinationOffset / m_client.pageScaleFactor()), velocity, originalOffsetInLayoutUnits).first;
166 return offset * m_client.pageScaleFactor();
167}

170void ScrollController::updateActiveScrollSnapIndexForClientOffset()
171{
172 if (!usesScrollSnap())
173 return;
174
175 ScrollOffset offset = roundedIntPoint(m_client.scrollOffset());
176 setNearestScrollSnapIndexForAxisAndOffset(ScrollEventAxis::Horizontal, offset.x());
177 setNearestScrollSnapIndexForAxisAndOffset(ScrollEventAxis::Vertical, offset.y());
178}
179
180void ScrollController::resnapAfterLayout()
181{
182 if (!usesScrollSnap())
183 return;
184
185 // If we are already snapped in a particular axis, maintain that. Otherwise, snap to the nearest eligible snap point.
186 ScrollOffset offset = roundedIntPoint(m_client.scrollOffset());
187 ScrollSnapAnimatorState& snapState = *m_scrollSnapState;
188
189 auto activeHorizontalIndex = m_scrollSnapState->activeSnapIndexForAxis(ScrollEventAxis::Horizontal);
190 if (!activeHorizontalIndex || *activeHorizontalIndex >= snapState.snapOffsetsForAxis(ScrollEventAxis::Horizontal).size())
191 setNearestScrollSnapIndexForAxisAndOffset(ScrollEventAxis::Horizontal, offset.x());
192
193 auto activeVerticalIndex = m_scrollSnapState->activeSnapIndexForAxis(ScrollEventAxis::Vertical);
194 if (!activeVerticalIndex || *activeVerticalIndex >= snapState.snapOffsetsForAxis(ScrollEventAxis::Vertical).size())
195 setNearestScrollSnapIndexForAxisAndOffset(ScrollEventAxis::Vertical, offset.y());
196
197}
