38class TextStream;

41namespace WebCore {
42
43enum class ScrollSnapState {
44 Snapping,
45 Gliding,
46 DestinationReached,
47 UserInteraction
48};
49
50class ScrollSnapAnimatorState {
51 WTF_MAKE_FAST_ALLOCATED;
52public:
53 const Vector<SnapOffset<LayoutUnit>>& snapOffsetsForAxis(ScrollEventAxis axis) const
54 {
55 return axis == ScrollEventAxis::Horizontal ? m_snapOffsetsInfo.horizontalSnapOffsets : m_snapOffsetsInfo.verticalSnapOffsets;
56 }
57
58 const LayoutScrollSnapOffsetsInfo& snapOffsetInfo() const { return m_snapOffsetsInfo; }
59 void setSnapOffsetInfo(const LayoutScrollSnapOffsetsInfo& newInfo) { m_snapOffsetsInfo = newInfo; }
60
61 ScrollSnapState currentState() const { return m_currentState; }
62
63 std::optional<unsigned> activeSnapIndexForAxis(ScrollEventAxis axis) const
64 {
65 return axis == ScrollEventAxis::Horizontal ? m_activeSnapIndexX : m_activeSnapIndexY;
66 }
67
68 void setActiveSnapIndexForAxis(ScrollEventAxis axis, std::optional<unsigned> index)
69 {
70 if (axis == ScrollEventAxis::Horizontal)
71 m_activeSnapIndexX = index;
72 else
73 m_activeSnapIndexY = index;
74 }
75
76 FloatPoint currentAnimatedScrollOffset(MonotonicTime, bool& isAnimationComplete) const;
77
78 // State transition helpers.
79 void transitionToSnapAnimationState(const FloatSize& contentSize, const FloatSize& viewportSize, float pageScale, const FloatPoint& initialOffset);
80 void transitionToGlideAnimationState(const FloatSize& contentSize, const FloatSize& viewportSize, float pageScale, const FloatPoint& initialOffset, const FloatSize& initialVelocity, const FloatSize& initialDelta);
81 void transitionToUserInteractionState();
82 void transitionToDestinationReachedState();
83
84private:
85 std::pair<float, std::optional<unsigned>> targetOffsetForStartOffset(ScrollEventAxis, const FloatSize& viewportSize, float maxScrollOffset, float startOffset, float predictedOffset, float pageScale, float initialDelta) const;
86 void teardownAnimationForState(ScrollSnapState);
87 void setupAnimationForState(ScrollSnapState, const FloatSize& contentSize, const FloatSize& viewportSize, float pageScale, const FloatPoint& initialOffset, const FloatSize& initialVelocity, const FloatSize& initialDelta);
88
89 ScrollSnapState m_currentState { ScrollSnapState::UserInteraction };
90
91 LayoutScrollSnapOffsetsInfo m_snapOffsetsInfo;
92
93 std::optional<unsigned> m_activeSnapIndexX;
94 std::optional<unsigned> m_activeSnapIndexY;
95
96 MonotonicTime m_startTime;
97 std::unique_ptr<ScrollingMomentumCalculator> m_momentumCalculator;
98};
99
100WTF::TextStream& operator<<(WTF::TextStream&, const ScrollSnapAnimatorState&);
101
102} // namespace WebCore
