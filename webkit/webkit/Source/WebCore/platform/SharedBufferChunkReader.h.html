39
40class SharedBuffer;
41
42class SharedBufferChunkReader {
43public:
44 SharedBufferChunkReader(SharedBuffer*, const Vector<char>& separator);
45 SharedBufferChunkReader(SharedBuffer*, const char* separator);
46
47 void setSeparator(const Vector<char>&);
48 void setSeparator(const char*);
49
50 // Returns false when the end of the buffer was reached.
51 bool nextChunk(Vector<uint8_t>& data, bool includeSeparator = false);
52
53 // Returns a null string when the end of the buffer has been reached.
54 String nextChunkAsUTF8StringWithLatin1Fallback(bool includeSeparator = false);
55
56 // Reads size bytes at the current location in the buffer, without changing the buffer position.
57 // Returns the number of bytes read. That number might be less than the specified size if the end of the buffer was reached.
58 size_t peek(Vector<uint8_t>&, size_t);
59
60private:
61 SharedBuffer* m_buffer;
62 size_t m_bufferPosition { 0 };
63 const uint8_t* m_segment { nullptr };
64 size_t m_segmentLength { 0 };
65 size_t m_segmentIndex { 0 };
66 bool m_reachedEndOfFile;
67 Vector<char> m_separator { false };
68 size_t m_separatorIndex { 0 };
69};
