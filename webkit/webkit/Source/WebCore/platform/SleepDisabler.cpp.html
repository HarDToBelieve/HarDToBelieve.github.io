29#include "SleepDisablerClient.h"

31namespace WebCore {
32
33SleepDisabler::SleepDisabler(const char* reason, PAL::SleepDisabler::Type type)
34 : m_type(type)
35{
36 if (sleepDisablerClient()) {
37 m_identifier = SleepDisablerIdentifier::generate();
38 sleepDisablerClient()->didCreateSleepDisabler(m_identifier, reason, type == PAL::SleepDisabler::Type::Display);
39 return;
40 }
41
42 m_platformSleepDisabler = PAL::SleepDisabler::create(reason, type);
43}
44
45SleepDisabler::~SleepDisabler()
46{
47 if (sleepDisablerClient())
48 sleepDisablerClient()->didDestroySleepDisabler(m_identifier);
49}
