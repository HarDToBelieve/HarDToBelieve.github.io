121void StaticPasteboard::write(const PasteboardImage& image)
122{
123 // FIXME: This should ideally remember the image data, so that when this StaticPasteboard
124 // is committed to the native pasteboard, we'll preserve the image as well. For now, stick
125 // with our existing behavior, which prevents image data from being copied in the case where
126 // any non-default data was written by the page.
127 m_fileContentState = Pasteboard::FileContentState::InMemoryImage;
128
129#if PLATFORM(MAC)
130 if (!image.dataInHTMLFormat.isEmpty())
131 writeMarkup(image.dataInHTMLFormat);
132#else
133 UNUSED_PARAM(image);
134#endif
135
136#if !PLATFORM(WIN)
137 if (Pasteboard::canExposeURLToDOMWhenPasteboardContainsFiles(image.url.url.string()))
138 write(image.url);
139#endif
140}
141
142void StaticPasteboard::write(const PasteboardWebContent& content)
143{
144 String markup;
145 String text;
146
147#if PLATFORM(COCOA)
148 markup = content.dataInHTMLFormat;
149 text = content.dataInStringFormat;
150#elif PLATFORM(GTK) || USE(LIBWPE)
151 markup = content.markup;
152 text = content.text;
153#endif
154
155 if (!markup.isEmpty())
156 writeMarkup(markup);
157
158 if (!text.isEmpty())
159 writePlainText(text, SmartReplaceOption::CannotSmartReplace);
160}
