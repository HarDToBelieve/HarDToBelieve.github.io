50LengthSize Theme::minimumControlSize(ControlPart part, const FontCascade& fontCascade, const LengthSize& zoomedSize, const LengthSize& nonShrinkableZoomedSize, float zoom) const
51{
52 auto minSize = minimumControlSize(part, fontCascade, zoomedSize, zoom);
53 if (part == ControlPart::RadioPart) {
54 if (zoomedSize.width.isIntrinsicOrAuto())
55 minSize.width = nonShrinkableZoomedSize.width;
56 if (zoomedSize.height.isIntrinsicOrAuto())
57 minSize.height = nonShrinkableZoomedSize.height;
58 }
59 return minSize;
60}

118void Theme::drawNamedImage(const String& name, GraphicsContext& context, const FloatRect& rect) const
119{
120 // We only handle one icon at the moment.
121 if (name != "wireless-playback")
122 return;
123
124 GraphicsContextStateSaver stateSaver(context);
125 context.setFillColor(Color::black);
126
127 // Draw a generic Wireless Playback icon.
128
129 context.scale(rect.size() / 100);
130 context.translate(8, 1);
131
132 Path outline;
133 outline.moveTo(FloatPoint(59, 58.7));
134 outline.addBezierCurveTo(FloatPoint(58.1, 58.7), FloatPoint(57.2, 58.4), FloatPoint(56.4, 57.7));
135 outline.addLineTo(FloatPoint(42, 45.5));
136 outline.addLineTo(FloatPoint(27.6, 57.8));
137 outline.addBezierCurveTo(FloatPoint(25.9, 59.2), FloatPoint(23.4, 59), FloatPoint(22, 57.3));
138 outline.addBezierCurveTo(FloatPoint(20.6, 55.6), FloatPoint(20.8, 53.1), FloatPoint(22.5, 51.7));
139 outline.addLineTo(FloatPoint(39.5, 37.3));
140 outline.addBezierCurveTo(FloatPoint(41, 36), FloatPoint(43.2, 36), FloatPoint(44.7, 37.3));
141 outline.addLineTo(FloatPoint(61.7, 51.7));
142 outline.addBezierCurveTo(FloatPoint(63.4, 53.1), FloatPoint(63.6, 55.7), FloatPoint(62.2, 57.3));
143 outline.addBezierCurveTo(FloatPoint(61.3, 58.2), FloatPoint(60.1, 58.7), FloatPoint(59, 58.7));
144 outline.addLineTo(FloatPoint(59, 58.7));
145 outline.closeSubpath();
146
147 outline.moveTo(FloatPoint(42, 98));
148 outline.addBezierCurveTo(FloatPoint(39.8, 98), FloatPoint(38, 96.3), FloatPoint(38, 94.2));
149 outline.addLineTo(FloatPoint(38, 43.6));
150 outline.addBezierCurveTo(FloatPoint(38, 41.5), FloatPoint(39.8, 39.8), FloatPoint(42, 39.8));
151 outline.addBezierCurveTo(FloatPoint(44.2, 39.8), FloatPoint(46, 41.5), FloatPoint(46, 43.6));
152 outline.addLineTo(FloatPoint(46, 94.2));
153 outline.addBezierCurveTo(FloatPoint(46, 96.3), FloatPoint(44.2, 98), FloatPoint(42, 98));
154 outline.addLineTo(FloatPoint(42, 98));
155 outline.closeSubpath();
156
157 outline.moveTo(FloatPoint(83.6, 41.6));
158 outline.addBezierCurveTo(FloatPoint(83.6, 18.6), FloatPoint(65, 0), FloatPoint(42, 0));
159 outline.addBezierCurveTo(FloatPoint(19, 0), FloatPoint(0.4, 18.6), FloatPoint(0.4, 41.6));
160 outline.addBezierCurveTo(FloatPoint(0.4, 62.2), FloatPoint(15, 79.2), FloatPoint(35, 82.6));
161 outline.addLineTo(FloatPoint(35, 74.5));
162 outline.addBezierCurveTo(FloatPoint(20, 71.2), FloatPoint(8.4, 57.7), FloatPoint(8.4, 41.6));
163 outline.addBezierCurveTo(FloatPoint(8.4, 23.1), FloatPoint(23.5, 8), FloatPoint(42, 8));
164 outline.addBezierCurveTo(FloatPoint(60.5, 8), FloatPoint(75.5, 23.1), FloatPoint(75.5, 41.6));
165 outline.addBezierCurveTo(FloatPoint(75.6, 57.7), FloatPoint(64, 71.2), FloatPoint(49, 74.5));
166 outline.addLineTo(FloatPoint(49, 82.6));
167 outline.addBezierCurveTo(FloatPoint(69, 79.3), FloatPoint(83.6, 62.2), FloatPoint(83.6, 41.6));
168 outline.addLineTo(FloatPoint(83.6, 41.6));
169 outline.closeSubpath();
170
171 context.fillPath(outline);
172}
