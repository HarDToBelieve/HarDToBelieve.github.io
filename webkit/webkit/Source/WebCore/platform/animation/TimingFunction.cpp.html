38TextStream& operator<<(TextStream& ts, const TimingFunction& timingFunction)
39{
40 switch (timingFunction.type()) {
41 case TimingFunction::LinearFunction:
42 ts << "linear";
43 break;
44 case TimingFunction::CubicBezierFunction: {
45 auto& function = downcast<CubicBezierTimingFunction>(timingFunction);
46 ts << "cubic-bezier(" << function.x1() << ", " << function.y1() << ", " << function.x2() << ", " << function.y2() << ")";
47 break;
48 }
49 case TimingFunction::StepsFunction: {
50 auto& function = downcast<StepsTimingFunction>(timingFunction);
51 ts << "steps(" << function.numberOfSteps();
52 if (auto stepPosition = function.stepPosition()) {
53 ts << ", ";
54 switch (stepPosition.value()) {
55 case StepsTimingFunction::StepPosition::JumpStart:
56 ts << "jump-start";
57 break;
58
59 case StepsTimingFunction::StepPosition::JumpEnd:
60 ts << "jump-end";
61 break;
62
63 case StepsTimingFunction::StepPosition::JumpNone:
64 ts << "jump-none";
65 break;
66
67 case StepsTimingFunction::StepPosition::JumpBoth:
68 ts << "jump-both";
69 break;
70
71 case StepsTimingFunction::StepPosition::Start:
72 ts << "start";
73 break;
74
75 case StepsTimingFunction::StepPosition::End:
76 ts << "end";
77 break;
78 }
79 }
80 ts << ")";
81 break;
82 }
83 case TimingFunction::SpringFunction: {
84 auto& function = downcast<SpringTimingFunction>(timingFunction);
85 ts << "spring(" << function.mass() << " " << function.stiffness() << " " << function.damping() << " " << function.initialVelocity() << ")";
86 break;
87 }
88 }
89 return ts;
90}

92double TimingFunction::transformTime(double inputTime, double duration, bool before) const
93{
94 switch (m_type) {
95 case TimingFunction::CubicBezierFunction: {
96 auto& function = downcast<CubicBezierTimingFunction>(*this);
97 if (function.isLinear())
98 return inputTime;
99 // The epsilon value we pass to UnitBezier::solve given that the animation is going to run over |dur| seconds. The longer the
100 // animation, the more precision we need in the timing function result to avoid ugly discontinuities.
101 auto epsilon = 1.0 / (1000.0 * duration);
102 return UnitBezier(function.x1(), function.y1(), function.x2(), function.y2()).solve(inputTime, epsilon);
103 }
104 case TimingFunction::StepsFunction: {
105 // https://drafts.csswg.org/css-easing-1/#step-timing-functions
106 auto& function = downcast<StepsTimingFunction>(*this);
107 auto steps = function.numberOfSteps();
108 auto stepPosition = function.stepPosition();
109 // 1. Calculate the current step as floor(input progress value × steps).
110 auto currentStep = std::floor(inputTime * steps);
111 // 2. If the step position property is start, increment current step by one.
112 if (stepPosition == StepsTimingFunction::StepPosition::JumpStart || stepPosition == StepsTimingFunction::StepPosition::Start || stepPosition == StepsTimingFunction::StepPosition::JumpBoth)
113 ++currentStep;
114 // 3. If both of the following conditions are true:
115 // - the before flag is set, and
116 // - input progress value × steps mod 1 equals zero (that is, if input progress value × steps is integral), then
117 // decrement current step by one.
118 if (before && !fmod(inputTime * steps, 1))
119 currentStep--;
120 // 4. If input progress value ≥ 0 and current step < 0, let current step be zero.
121 if (inputTime >= 0 && currentStep < 0)
122 currentStep = 0;
123 // 5. Calculate jumps based on the step position.
124 if (stepPosition == StepsTimingFunction::StepPosition::JumpNone)
125 --steps;
126 else if (stepPosition == StepsTimingFunction::StepPosition::JumpBoth)
127 ++steps;
128 // 6. If input progress value ≤ 1 and current step > jumps, let current step be jumps.
129 if (inputTime <= 1 && currentStep > steps)
130 currentStep = steps;
131 // 7. The output progress value is current step / jumps.
132 return currentStep / steps;
133 }
134 case TimingFunction::SpringFunction: {
135 auto& function = downcast<SpringTimingFunction>(*this);
136 return SpringSolver(function.mass(), function.stiffness(), function.damping(), function.initialVelocity()).solve(inputTime * duration);
137 }
138 case TimingFunction::LinearFunction:
139 return inputTime;
140 }
141
142 ASSERT_NOT_REACHED();
143 return 0;
144}

146ExceptionOr<RefPtr<TimingFunction>> TimingFunction::createFromCSSText(const String& cssText)
147{
148 auto properties = MutableStyleProperties::create();
149 properties->parseDeclaration(makeString("animation-timing-function:", cssText), CSSParserContext(HTMLStandardMode));
150 if (auto value = properties->getPropertyCSSValue(CSSPropertyAnimationTimingFunction)) {
151 if (auto function = createFromCSSValue(*value))
152 return function;
153 }
154 return Exception { TypeError };
155}

157RefPtr<TimingFunction> TimingFunction::createFromCSSValue(const CSSValue& value)
158{
159 if (is<CSSPrimitiveValue>(value)) {
160 switch (downcast<CSSPrimitiveValue>(value).valueID()) {
161 case CSSValueLinear:
162 return LinearTimingFunction::create();
163 case CSSValueEase:
164 return CubicBezierTimingFunction::create();
165 case CSSValueEaseIn:
166 return CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseIn);
167 case CSSValueEaseOut:
168 return CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseOut);
169 case CSSValueEaseInOut:
170 return CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseInOut);
171 case CSSValueStepStart:
172 return StepsTimingFunction::create(1, StepsTimingFunction::StepPosition::Start);
173 case CSSValueStepEnd:
174 return StepsTimingFunction::create(1, StepsTimingFunction::StepPosition::End);
175 default:
176 return nullptr;
177 }
178 }
179
180 if (is<CSSCubicBezierTimingFunctionValue>(value)) {
181 auto& cubicTimingFunction = downcast<CSSCubicBezierTimingFunctionValue>(value);
182 return CubicBezierTimingFunction::create(cubicTimingFunction.x1(), cubicTimingFunction.y1(), cubicTimingFunction.x2(), cubicTimingFunction.y2());
183 }
184 if (is<CSSStepsTimingFunctionValue>(value)) {
185 auto& stepsTimingFunction = downcast<CSSStepsTimingFunctionValue>(value);
186 return StepsTimingFunction::create(stepsTimingFunction.numberOfSteps(), stepsTimingFunction.stepPosition());
187 }
188 if (is<CSSSpringTimingFunctionValue>(value)) {
189 auto& springTimingFunction = downcast<CSSSpringTimingFunctionValue>(value);
190 return SpringTimingFunction::create(springTimingFunction.mass(), springTimingFunction.stiffness(), springTimingFunction.damping(), springTimingFunction.initialVelocity());
191 }
192
193 return nullptr;
194}

196String TimingFunction::cssText() const
197{
198 if (m_type == TimingFunction::CubicBezierFunction) {
199 auto& function = downcast<CubicBezierTimingFunction>(*this);
200 if (function.x1() == 0.25 && function.y1() == 0.1 && function.x2() == 0.25 && function.y2() == 1.0)
201 return "ease";
202 if (function.x1() == 0.42 && !function.y1() && function.x2() == 1.0 && function.y2() == 1.0)
203 return "ease-in";
204 if (!function.x1() && !function.y1() && function.x2() == 0.58 && function.y2() == 1.0)
205 return "ease-out";
206 if (function.x1() == 0.42 && !function.y1() && function.x2() == 0.58 && function.y2() == 1.0)
207 return "ease-in-out";
208 return makeString("cubic-bezier(", function.x1(), ", ", function.y1(), ", ", function.x2(), ", ", function.y2(), ')');
209 }
210
211 if (m_type == TimingFunction::StepsFunction) {
212 auto& function = downcast<StepsTimingFunction>(*this);
213 if (function.stepPosition() == StepsTimingFunction::StepPosition::JumpEnd || function.stepPosition() == StepsTimingFunction::StepPosition::End)
214 return makeString("steps(", function.numberOfSteps(), ')');
215 }
216
217 TextStream stream;
218 stream << *this;
219 return stream.release();
220}
