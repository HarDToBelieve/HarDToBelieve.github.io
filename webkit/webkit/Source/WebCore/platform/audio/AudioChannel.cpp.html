41void AudioChannel::scale(float scale)
42{
43 if (isSilent())
44 return;
45
46 VectorMath::multiplyByScalar(data(), scale, mutableData(), length());
47}

49void AudioChannel::copyFrom(const AudioChannel* sourceChannel)
50{
51 bool isSafe = (sourceChannel && sourceChannel->length() >= length());
52 ASSERT(isSafe);
53
54 if (!isSafe || sourceChannel->isSilent()) {
55 zero();
56 return;
57 }
58 memcpy(mutableData(), sourceChannel->data(), sizeof(float) * length());
59}
60
61void AudioChannel::copyFromRange(const AudioChannel* sourceChannel, unsigned startFrame, unsigned endFrame)
62{
63 // Check that range is safe for reading from sourceChannel.
64 bool isRangeSafe = sourceChannel && startFrame < endFrame && endFrame <= sourceChannel->length();
65 ASSERT(isRangeSafe);
66 if (!isRangeSafe)
67 return;
68
69 if (sourceChannel->isSilent() && isSilent())
70 return;
71
72 // Check that this channel has enough space.
73 size_t rangeLength = endFrame - startFrame;
74 bool isRangeLengthSafe = rangeLength <= length();
75 ASSERT(isRangeLengthSafe);
76 if (!isRangeLengthSafe)
77 return;
78
79 const float* source = sourceChannel->data();
80 float* destination = mutableData();
81
82 if (sourceChannel->isSilent()) {
83 if (rangeLength == length())
84 zero();
85 else
86 memset(destination, 0, sizeof(float) * rangeLength);
87 } else
88 memcpy(destination, source + startFrame, sizeof(float) * rangeLength);
89}

91void AudioChannel::sumFrom(const AudioChannel* sourceChannel)
92{
93 bool isSafe = sourceChannel && sourceChannel->length() >= length();
94 ASSERT(isSafe);
95 if (!isSafe)
96 return;
97
98 if (sourceChannel->isSilent())
99 return;
100
101 if (isSilent())
102 copyFrom(sourceChannel);
103 else
104 VectorMath::add(data(), sourceChannel->data(), mutableData(), length());
105}

107float AudioChannel::maxAbsValue() const
108{
109 if (isSilent())
110 return 0;
111
112 return VectorMath::maximumMagnitude(data(), length());
113}
