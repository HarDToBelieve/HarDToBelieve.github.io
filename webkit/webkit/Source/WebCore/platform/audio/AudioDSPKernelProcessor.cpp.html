41// setNumberOfChannels() may later be called if the object is not yet in an "initialized" state.
42AudioDSPKernelProcessor::AudioDSPKernelProcessor(float sampleRate, unsigned numberOfChannels)
43 : AudioProcessor(sampleRate, numberOfChannels)
44{
45}
46
47void AudioDSPKernelProcessor::initialize()
48{
49 if (isInitialized())
50 return;
51
52 ASSERT(!m_kernels.size());
53 {
54 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
55 // explicitly allow the following allocation(s).
56 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
57
58 // Create processing kernels, one per channel.
59 for (unsigned i = 0; i < numberOfChannels(); ++i)
60 m_kernels.append(createKernel());
61 }
62 m_initialized = true;
63 m_hasJustReset = true;
64}
65
66void AudioDSPKernelProcessor::uninitialize()
67{
68 if (!isInitialized())
69 return;
70
71 m_kernels.clear();
72
73 m_initialized = false;
74}
75
76void AudioDSPKernelProcessor::process(const AudioBus* source, AudioBus* destination, size_t framesToProcess)
77{
78 ASSERT(source && destination);
79 if (!source || !destination)
80 return;
81
82 if (!isInitialized()) {
83 destination->zero();
84 return;
85 }
86
87 bool channelCountMatches = source->numberOfChannels() == destination->numberOfChannels() && source->numberOfChannels() == m_kernels.size();
88 ASSERT(channelCountMatches);
89 if (!channelCountMatches)
90 return;
91
92 for (unsigned i = 0; i < m_kernels.size(); ++i)
93 m_kernels[i]->process(source->channel(i)->data(), destination->channel(i)->mutableData(), framesToProcess);
94}
95
96void AudioDSPKernelProcessor::processOnlyAudioParams(size_t framesToProcess)
97{
98 if (!isInitialized())
99 return;
100
101 for (unsigned i = 0; i < m_kernels.size(); ++i)
102 m_kernels[i]->processOnlyAudioParams(framesToProcess);
103}
104
105// Resets filter state
106void AudioDSPKernelProcessor::reset()
107{
108 if (!isInitialized())
109 return;
110
111 // Forces snap to parameter values - first time.
112 // Any processing depending on this value must set it to false at the appropriate time.
113 m_hasJustReset = true;
114
115 for (unsigned i = 0; i < m_kernels.size(); ++i)
116 m_kernels[i]->reset();
117}
118
119void AudioDSPKernelProcessor::setNumberOfChannels(unsigned numberOfChannels)
120{
121 if (numberOfChannels == m_numberOfChannels)
122 return;
123
124 ASSERT(!isInitialized());
125 if (!isInitialized())
126 m_numberOfChannels = numberOfChannels;
127}
128
129double AudioDSPKernelProcessor::tailTime() const
130{
131 // It is expected that all the kernels have the same tailTime.
132 return !m_kernels.isEmpty() ? m_kernels.first()->tailTime() : 0;
133}
134
135double AudioDSPKernelProcessor::latencyTime() const
136{
137 // It is expected that all the kernels have the same latencyTime.
138 return !m_kernels.isEmpty() ? m_kernels.first()->latencyTime() : 0;
139}
140
141bool AudioDSPKernelProcessor::requiresTailProcessing() const
142{
143 // Always return true even if the tail time and latency might both be zero.
144 return true;
145}
