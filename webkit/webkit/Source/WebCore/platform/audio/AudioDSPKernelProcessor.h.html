31#ifndef AudioDSPKernelProcessor_h
32#define AudioDSPKernelProcessor_h
33
34#include "AudioBus.h"
35#include "AudioProcessor.h"
36#include <wtf/Vector.h>
37
38namespace WebCore {
39
40class AudioBus;
41class AudioDSPKernel;
42class AudioProcessor;
43
44// AudioDSPKernelProcessor processes one input -> one output (N channels each)
45// It uses one AudioDSPKernel object per channel to do the processing, thus there is no cross-channel processing.
46// Despite this limitation it turns out to be a very common and useful type of processor.
47
48class AudioDSPKernelProcessor : public AudioProcessor {
49public:
50 // numberOfChannels may be later changed if object is not yet in an "initialized" state
51 AudioDSPKernelProcessor(float sampleRate, unsigned numberOfChannels);
52
53 // Subclasses create the appropriate type of processing kernel here.
54 // We'll call this to create a kernel for each channel.
55 virtual std::unique_ptr<AudioDSPKernel> createKernel() = 0;
56
57 // AudioProcessor methods
58 void initialize() override;
59 void uninitialize() override;
60 void process(const AudioBus* source, AudioBus* destination, size_t framesToProcess) override;
61 void processOnlyAudioParams(size_t framesToProcess) override;
62 void reset() override;
63 void setNumberOfChannels(unsigned) override;
64 unsigned numberOfChannels() const override { return m_numberOfChannels; }
65
66 double tailTime() const override;
67 double latencyTime() const override;
68 bool requiresTailProcessing() const override;
69
70protected:
71 Vector<std::unique_ptr<AudioDSPKernel>> m_kernels;
72 bool m_hasJustReset { true };
73};
74
75} // namespace WebCore
76
77#endif // AudioDSPKernelProcessor_h
