26#ifndef AudioHardwareListener_h
27#define AudioHardwareListener_h
28
29#include <wtf/Ref.h>
30#include <wtf/RefCounted.h>
31
32namespace WebCore {
33
34enum class AudioHardwareActivityType {
35 Unknown,
36 IsActive,
37 IsInactive
38};
39
40class AudioHardwareListener : public RefCounted<AudioHardwareListener> {
41public:
42 class Client {
43 public:
44 virtual ~Client() = default;
45 virtual void audioHardwareDidBecomeActive() = 0;
46 virtual void audioHardwareDidBecomeInactive() = 0;
47 virtual void audioOutputDeviceChanged() = 0;
48 };
49
50 using CreationFunction = Function<Ref<AudioHardwareListener>(AudioHardwareListener::Client&)>;
51 WEBCORE_EXPORT static void setCreationFunction(CreationFunction&&);
52 WEBCORE_EXPORT static void resetCreationFunction();
53
54 WEBCORE_EXPORT static Ref<AudioHardwareListener> create(Client&);
55 virtual ~AudioHardwareListener() = default;
56
57 AudioHardwareActivityType hardwareActivity() const { return m_activity; }
58
59 struct BufferSizeRange {
60 size_t minimum { 0 };
61 size_t maximum { 0 };
62 operator bool() const { return minimum && maximum; }
63 size_t nearest(size_t value) const { return std::min(std::max(value, minimum), maximum); }
64 };
65 BufferSizeRange supportedBufferSizes() const { return m_supportedBufferSizes; }
66
67protected:
68 WEBCORE_EXPORT AudioHardwareListener(Client&);
69
70 void setHardwareActivity(AudioHardwareActivityType activity) { m_activity = activity; }
71 void setSupportedBufferSizes(BufferSizeRange sizes) { m_supportedBufferSizes = sizes; }
72
73 Client& m_client;
74 AudioHardwareActivityType m_activity { AudioHardwareActivityType::Unknown };
75 BufferSizeRange m_supportedBufferSizes;
76};
77
78}
79
80#endif // AudioHardwareListener_h
