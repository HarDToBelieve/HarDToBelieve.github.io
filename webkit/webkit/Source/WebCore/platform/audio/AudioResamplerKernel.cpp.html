37AudioResamplerKernel::AudioResamplerKernel(AudioResampler* resampler)
38 : m_resampler(resampler)
39 // The buffer size must be large enough to hold up to two extra sample frames for the linear interpolation.
40 , m_sourceBuffer(2 + static_cast<int>(AudioUtilities::renderQuantumSize * AudioResampler::MaxRate))
41{
42 m_lastValues[0] = 0.0f;
43 m_lastValues[1] = 0.0f;
44}
45
46float* AudioResamplerKernel::getSourcePointer(size_t framesToProcess, size_t* numberOfSourceFramesNeededP)
47{
48 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
49
50 // Calculate the next "virtual" index. After process() is called, m_virtualReadIndex will equal this value.
51 double nextFractionalIndex = m_virtualReadIndex + framesToProcess * rate();
52
53 // Because we're linearly interpolating between the previous and next sample we need to round up so we include the next sample.
54 int endIndex = static_cast<int>(nextFractionalIndex + 1.0); // round up to next integer index
55
56 // Determine how many input frames we'll need.
57 // We need to fill the buffer up to and including endIndex (so add 1) but we've already buffered m_fillIndex frames from last time.
58 size_t framesNeeded = 1 + endIndex - m_fillIndex;
59 if (numberOfSourceFramesNeededP)
60 *numberOfSourceFramesNeededP = framesNeeded;
61
62 // Do bounds checking for the source buffer.
63 bool isGood = m_fillIndex < m_sourceBuffer.size() && m_fillIndex + framesNeeded <= m_sourceBuffer.size();
64 ASSERT(isGood);
65 if (!isGood)
66 return 0;
67
68 return m_sourceBuffer.data() + m_fillIndex;
69}

71void AudioResamplerKernel::process(float* destination, size_t framesToProcess)
72{
73 ASSERT(framesToProcess <= AudioUtilities::renderQuantumSize);
74
75 float* source = m_sourceBuffer.data();
76
77 double rate = this->rate();
78 rate = std::max(0.0, rate);
79 rate = std::min(AudioResampler::MaxRate, rate);
80
81 // Start out with the previous saved values (if any).
82 if (m_fillIndex > 0) {
83 source[0] = m_lastValues[0];
84 source[1] = m_lastValues[1];
85 }
86
87 // Make a local copy.
88 double virtualReadIndex = m_virtualReadIndex;
89
90 // Sanity check source buffer access.
91 ASSERT(framesToProcess > 0);
92 ASSERT(virtualReadIndex >= 0 && 1 + static_cast<unsigned>(virtualReadIndex + (framesToProcess - 1) * rate) < m_sourceBuffer.size());
93
94 // Do the linear interpolation.
95 int n = framesToProcess;
96 while (n--) {
97 unsigned readIndex = static_cast<unsigned>(virtualReadIndex);
98 double interpolationFactor = virtualReadIndex - readIndex;
99
100 double sample1 = source[readIndex];
101 double sample2 = source[readIndex + 1];
102
103 double sample = (1.0 - interpolationFactor) * sample1 + interpolationFactor * sample2;
104
105 *destination++ = static_cast<float>(sample);
106
107 virtualReadIndex += rate;
108 }
109
110 // Save the last two sample-frames which will later be used at the beginning of the source buffer the next time around.
111 int readIndex = static_cast<int>(virtualReadIndex);
112 m_lastValues[0] = source[readIndex];
113 m_lastValues[1] = source[readIndex + 1];
114 m_fillIndex = 2;
115
116 // Wrap the virtual read index back to the start of the buffer.
117 virtualReadIndex -= readIndex;
118
119 // Put local copy back into member variable.
120 m_virtualReadIndex = virtualReadIndex;
121}

123void AudioResamplerKernel::reset()
124{
125 m_virtualReadIndex = 0.0;
126 m_fillIndex = 0;
127 m_lastValues[0] = 0.0f;
128 m_lastValues[1] = 0.0f;
129}
130
131double AudioResamplerKernel::rate() const
132{
133 return m_resampler->rate();
134}
