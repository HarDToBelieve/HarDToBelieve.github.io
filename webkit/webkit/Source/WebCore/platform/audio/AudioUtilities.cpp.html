36double discreteTimeConstantForSampleRate(double timeConstant, double sampleRate)
37{
38 return 1 - exp(-1 / (sampleRate * timeConstant));
39}

41size_t timeToSampleFrame(double time, double sampleRate, SampleFrameRounding rounding)
42{
43 // To compute the desired frame number, we pretend we're actually running the
44 // context at a much higher sample rate (by a factor of |oversampleFactor|).
45 // Round this to get the nearest frame number at the higher rate. Then
46 // convert back to the original rate to get a new frame number that may not be
47 // an integer. Then use the specified |rounding_mode| to round this to the
48 // integer frame number that we need.
49 //
50 // Doing this partially solves the issue where Fs * (k / Fs) != k when doing
51 // floating point arithmtic for integer k and Fs is the sample rate. By
52 // oversampling and rounding, we'll get k back most of the time.
53 //
54 // The oversampling factor MUST be a power of two so as not to introduce
55 // additional round-off in computing the oversample frame number.
56 constexpr double oversampleFactor = 1024;
57 double frame = std::round(time * sampleRate * oversampleFactor) / oversampleFactor;
58
59 switch (rounding) {
60 case SampleFrameRounding::Nearest:
61 frame = std::round(frame);
62 break;
63 case SampleFrameRounding::Down:
64 frame = std::floor(frame);
65 break;
66 case SampleFrameRounding::Up:
67 frame = std::ceil(frame);
68 break;
69 }
70
71 // Just return the largest possible size_t value if necessary.
72 if (frame >= std::numeric_limits<size_t>::max())
73 return std::numeric_limits<size_t>::max();
74
75 return static_cast<size_t>(frame);
76}
