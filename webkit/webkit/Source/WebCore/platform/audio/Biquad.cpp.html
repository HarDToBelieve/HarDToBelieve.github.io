53Biquad::Biquad()
54{
55#if USE(ACCELERATE)
56 // Allocate two samples more for filter history
57 m_inputBuffer.resize(kBufferSize + 2);
58 m_outputBuffer.resize(kBufferSize + 2);
59#endif
60
61 // Allocate enough space for the a-rate filter coefficients to handle a
62 // rendering quantum of 128 frames.
63 m_b0.resize(AudioUtilities::renderQuantumSize);
64 m_b1.resize(AudioUtilities::renderQuantumSize);
65 m_b2.resize(AudioUtilities::renderQuantumSize);
66 m_a1.resize(AudioUtilities::renderQuantumSize);
67 m_a2.resize(AudioUtilities::renderQuantumSize);
68
69 // Initialize as pass-thru (straight-wire, no filter effect)
70 setNormalizedCoefficients(0, 1, 0, 0, 1, 0, 0);
71
72 reset(); // clear filter memory
73}

77void Biquad::process(const float* sourceP, float* destP, size_t framesToProcess)
78{
79 if (hasSampleAccurateValues()) {
80 size_t n = framesToProcess;
81
82 // Create local copies of member variables
83 double x1 = m_x1;
84 double x2 = m_x2;
85 double y1 = m_y1;
86 double y2 = m_y2;
87
88 auto* b0 = m_b0.data();
89 auto* b1 = m_b1.data();
90 auto* b2 = m_b2.data();
91 auto* a1 = m_a1.data();
92 auto* a2 = m_a2.data();
93
94 for (size_t k = 0; k < n; ++k) {
95 // FIXME: this can be optimized by pipelining the multiply adds...
96 float x = *sourceP++;
97 float y = b0[k] * x + b1[k] * x1 + b2[k] * x2 - a1[k] * y1 - a2[k] * y2;
98
99 *destP++ = y;
100
101 // Update state variables
102 x2 = x1;
103 x1 = x;
104 y2 = y1;
105 y1 = y;
106 }
107
108 // Local variables back to member. Flush denormals here so we
109 // don't slow down the inner loop above.
110 m_x1 = DenormalDisabler::flushDenormalFloatToZero(x1);
111 m_x2 = DenormalDisabler::flushDenormalFloatToZero(x2);
112 m_y1 = DenormalDisabler::flushDenormalFloatToZero(y1);
113 m_y2 = DenormalDisabler::flushDenormalFloatToZero(y2);
114
115 // There is an assumption here that once we have sample accurate values we
116 // can never go back to not having sample accurate values. This is
117 // currently true in the way AudioParamTimline is implemented: once an
118 // event is inserted, sample accurate processing is always enabled.
119 //
120 // If so, then we never have to update the state variables for the MACOSX
121 // path. The structure of the state variable in these cases aren't well
122 // documented so it's not clear how to update them anyway.
123 return;
124 }
125
126#if USE(ACCELERATE)
127 auto* inputP = m_inputBuffer.data();
128 auto* outputP = m_outputBuffer.data();
129
130 // Set up filter state. This is needed in case we're switching from
131 // filtering with variable coefficients (i.e., with automations) to
132 // fixed coefficients (without automations).
133 inputP[0] = m_x2;
134 inputP[1] = m_x1;
135 outputP[0] = m_y2;
136 outputP[1] = m_y1;
137
138 // Use vecLib if available
139 processFast(sourceP, destP, framesToProcess);
140
141 ASSERT(framesToProcess >= 2);
142
143 // Copy the last inputs and outputs to the filter memory variables.
144 // This is needed because the next rendering quantum might be an
145 // automation which needs the history to continue correctly. Because
146 // sourceP and destP can be the same block of memory, we can't read from
147 // sourceP to get the last inputs. Fortunately, processFast has put the
148 // last inputs in input[0] and input[1].
149 m_x1 = inputP[1];
150 m_x2 = inputP[0];
151 m_y1 = destP[framesToProcess - 1];
152 m_y2 = destP[framesToProcess - 2];
153#else
154 size_t n = framesToProcess;
155
156 // Create local copies of member variables
157 double x1 = m_x1;
158 double x2 = m_x2;
159 double y1 = m_y1;
160 double y2 = m_y2;
161
162 double b0 = m_b0[0];
163 double b1 = m_b1[0];
164 double b2 = m_b2[0];
165 double a1 = m_a1[0];
166 double a2 = m_a2[0];
167
168 while (n--) {
169 // FIXME: this can be optimized by pipelining the multiply adds...
170 float x = *sourceP++;
171 float y = b0 * x + b1 * x1 + b2 * x2 - a1 * y1 - a2 * y2;
172
173 *destP++ = y;
174
175 // Update state variables
176 x2 = x1;
177 x1 = x;
178 y2 = y1;
179 y1 = y;
180 }
181
182 // Local variables back to member. Flush denormals here so we
183 // don't slow down the inner loop above.
184 m_x1 = DenormalDisabler::flushDenormalFloatToZero(x1);
185 m_x2 = DenormalDisabler::flushDenormalFloatToZero(x2);
186 m_y1 = DenormalDisabler::flushDenormalFloatToZero(y1);
187 m_y2 = DenormalDisabler::flushDenormalFloatToZero(y2);
188#endif
189}

195void Biquad::processFast(const float* sourceP, float* destP, size_t framesToProcess)
196{
197 double filterCoefficients[5];
198 filterCoefficients[0] = m_b0[0];
199 filterCoefficients[1] = m_b1[0];
200 filterCoefficients[2] = m_b2[0];
201 filterCoefficients[3] = m_a1[0];
202 filterCoefficients[4] = m_a2[0];
203
204 double* inputP = m_inputBuffer.data();
205 double* outputP = m_outputBuffer.data();
206
207 double* input2P = inputP + 2;
208 double* output2P = outputP + 2;
209
210 // Break up processing into smaller slices (kBufferSize) if necessary.
211
212 size_t n = framesToProcess;
213
214 while (n > 0) {
215 int framesThisTime = n < kBufferSize ? n : kBufferSize;
216
217 // Copy input to input buffer
218 for (int i = 0; i < framesThisTime; ++i)
219 input2P[i] = *sourceP++;
220
221 processSliceFast(inputP, outputP, filterCoefficients, framesThisTime);
222
223 // Copy output buffer to output (converts float -> double).
224 for (int i = 0; i < framesThisTime; ++i)
225 *destP++ = static_cast<float>(output2P[i]);
226
227 n -= framesThisTime;
228 }
229}

231void Biquad::processSliceFast(double* sourceP, double* destP, double* coefficientsP, size_t framesToProcess)
232{
233 // Use double-precision for filter stability
234 vDSP_deq22D(sourceP, 1, coefficientsP, destP, 1, framesToProcess);
235
236 // Save history. Note that sourceP and destP reference m_inputBuffer and m_outputBuffer respectively.
237 // These buffers are allocated (in the constructor) with space for two extra samples so it's OK to access
238 // array values two beyond framesToProcess.
239 sourceP[0] = sourceP[framesToProcess - 2 + 2];
240 sourceP[1] = sourceP[framesToProcess - 1 + 2];
241 destP[0] = destP[framesToProcess - 2 + 2];
242 destP[1] = destP[framesToProcess - 1 + 2];
243}

265void Biquad::setLowpassParams(size_t index, double cutoff, double resonance)
266{
267 // Limit cutoff to 0 to 1.
268 cutoff = std::max(0.0, std::min(cutoff, 1.0));
269
270 if (cutoff == 1) {
271 // When cutoff is 1, the z-transform is 1.
272 setNormalizedCoefficients(index, 1, 0, 0, 1, 0, 0);
273 } else if (cutoff > 0) {
274 // Compute biquad coefficients for lowpass filter
275 resonance = pow(10.0, 0.05 * resonance);
276
277 double theta = piDouble * cutoff;
278 double alpha = sin(theta) / (2 * resonance);
279 double cosw = cos(theta);
280 double beta = (1 - cosw) / 2;
281
282 double b0 = beta;
283 double b1 = 2 * beta;
284 double b2 = beta;
285
286 double a0 = 1 + alpha;
287 double a1 = -2 * cosw;
288 double a2 = 1 - alpha;
289
290 setNormalizedCoefficients(index, b0, b1, b2, a0, a1, a2);
291 } else {
292 // When cutoff is zero, nothing gets through the filter, so set
293 // coefficients up correctly.
294 setNormalizedCoefficients(index, 0, 0, 0, 1, 0, 0);
295 }
296}

298void Biquad::setHighpassParams(size_t index, double cutoff, double resonance)
299{
300 // Limit cutoff to 0 to 1.
301 cutoff = std::max(0.0, std::min(cutoff, 1.0));
302
303 if (cutoff == 1) {
304 // The z-transform is 0.
305 setNormalizedCoefficients(index, 0, 0, 0, 1, 0, 0);
306 } else if (cutoff > 0) {
307 // Compute biquad coefficients for highpass filter
308 resonance = pow(10.0, 0.05 * resonance);
309
310 double theta = piDouble * cutoff;
311 double alpha = sin(theta) / (2 * resonance);
312 double cosw = cos(theta);
313 double beta = (1 + cosw) / 2;
314
315 double b0 = beta;
316 double b1 = -(2 * beta);
317 double b2 = beta;
318
319 double a0 = 1 + alpha;
320 double a1 = -2 * cosw;
321 double a2 = 1 - alpha;
322
323 setNormalizedCoefficients(index, b0, b1, b2, a0, a1, a2);
324 } else {
325 // When cutoff is zero, we need to be careful because the above
326 // gives a quadratic divided by the same quadratic, with poles
327 // and zeros on the unit circle in the same place. When cutoff
328 // is zero, the z-transform is 1.
329 setNormalizedCoefficients(index, 1, 0, 0, 1, 0, 0);
330 }
331}

333void Biquad::setNormalizedCoefficients(size_t index, double b0, double b1, double b2, double a0, double a1, double a2)
334{
335 double a0Inverse = 1 / a0;
336
337 m_b0[index] = b0 * a0Inverse;
338 m_b1[index] = b1 * a0Inverse;
339 m_b2[index] = b2 * a0Inverse;
340 m_a1[index] = a1 * a0Inverse;
341 m_a2[index] = a2 * a0Inverse;
342}

344void Biquad::setLowShelfParams(size_t index, double frequency, double dbGain)
345{
346 // Clip frequencies to between 0 and 1, inclusive.
347 frequency = std::max(0.0, std::min(frequency, 1.0));
348
349 double A = pow(10.0, dbGain / 40);
350
351 if (frequency == 1) {
352 // The z-transform is a constant gain.
353 setNormalizedCoefficients(index, A * A, 0, 0, 1, 0, 0);
354 } else if (frequency > 0) {
355 double w0 = piDouble * frequency;
356 double S = 1; // filter slope (1 is max value)
357 double alpha = 0.5 * sin(w0) * sqrt((A + 1 / A) * (1 / S - 1) + 2);
358 double k = cos(w0);
359 double k2 = 2 * sqrt(A) * alpha;
360 double aPlusOne = A + 1;
361 double aMinusOne = A - 1;
362
363 double b0 = A * (aPlusOne - aMinusOne * k + k2);
364 double b1 = 2 * A * (aMinusOne - aPlusOne * k);
365 double b2 = A * (aPlusOne - aMinusOne * k - k2);
366 double a0 = aPlusOne + aMinusOne * k + k2;
367 double a1 = -2 * (aMinusOne + aPlusOne * k);
368 double a2 = aPlusOne + aMinusOne * k - k2;
369
370 setNormalizedCoefficients(index, b0, b1, b2, a0, a1, a2);
371 } else {
372 // When frequency is 0, the z-transform is 1.
373 setNormalizedCoefficients(index, 1, 0, 0, 1, 0, 0);
374 }
375}

377void Biquad::setHighShelfParams(size_t index, double frequency, double dbGain)
378{
379 // Clip frequencies to between 0 and 1, inclusive.
380 frequency = std::max(0.0, std::min(frequency, 1.0));
381
382 double A = pow(10.0, dbGain / 40);
383
384 if (frequency == 1) {
385 // The z-transform is 1.
386 setNormalizedCoefficients(index, 1, 0, 0, 1, 0, 0);
387 } else if (frequency > 0) {
388 double w0 = piDouble * frequency;
389 double S = 1; // filter slope (1 is max value)
390 double alpha = 0.5 * sin(w0) * sqrt((A + 1 / A) * (1 / S - 1) + 2);
391 double k = cos(w0);
392 double k2 = 2 * sqrt(A) * alpha;
393 double aPlusOne = A + 1;
394 double aMinusOne = A - 1;
395
396 double b0 = A * (aPlusOne + aMinusOne * k + k2);
397 double b1 = -2 * A * (aMinusOne + aPlusOne * k);
398 double b2 = A * (aPlusOne + aMinusOne * k - k2);
399 double a0 = aPlusOne - aMinusOne * k + k2;
400 double a1 = 2 * (aMinusOne - aPlusOne * k);
401 double a2 = aPlusOne - aMinusOne * k - k2;
402
403 setNormalizedCoefficients(index, b0, b1, b2, a0, a1, a2);
404 } else {
405 // When frequency = 0, the filter is just a gain, A^2.
406 setNormalizedCoefficients(index, A * A, 0, 0, 1, 0, 0);
407 }

410void Biquad::setPeakingParams(size_t index, double frequency, double Q, double dbGain)
411{
412 // Clip frequencies to between 0 and 1, inclusive.
413 frequency = std::max(0.0, std::min(frequency, 1.0));
414
415 // Don't let Q go negative, which causes an unstable filter.
416 Q = std::max(0.0, Q);
417
418 double A = pow(10.0, dbGain / 40);
419
420 if (frequency > 0 && frequency < 1) {
421 if (Q > 0) {
422 double w0 = piDouble * frequency;
423 double alpha = sin(w0) / (2 * Q);
424 double k = cos(w0);
425
426 double b0 = 1 + alpha * A;
427 double b1 = -2 * k;
428 double b2 = 1 - alpha * A;
429 double a0 = 1 + alpha / A;
430 double a1 = -2 * k;
431 double a2 = 1 - alpha / A;
432
433 setNormalizedCoefficients(index, b0, b1, b2, a0, a1, a2);
434 } else {
435 // When Q = 0, the above formulas have problems. If we look at
436 // the z-transform, we can see that the limit as Q->0 is A^2, so
437 // set the filter that way.
438 setNormalizedCoefficients(index, A * A, 0, 0, 1, 0, 0);
439 }
440 } else {
441 // When frequency is 0 or 1, the z-transform is 1.
442 setNormalizedCoefficients(index, 1, 0, 0, 1, 0, 0);
443 }
444}

514void Biquad::setBandpassParams(size_t index, double frequency, double Q)
515{
516 // No negative frequencies allowed.
517 frequency = std::max(0.0, frequency);
518
519 // Don't let Q go negative, which causes an unstable filter.
520 Q = std::max(0.0, Q);
521
522 if (frequency > 0 && frequency < 1) {
523 double w0 = piDouble * frequency;
524 if (Q > 0) {
525 double alpha = sin(w0) / (2 * Q);
526 double k = cos(w0);
527
528 double b0 = alpha;
529 double b1 = 0;
530 double b2 = -alpha;
531 double a0 = 1 + alpha;
532 double a1 = -2 * k;
533 double a2 = 1 - alpha;
534
535 setNormalizedCoefficients(index, b0, b1, b2, a0, a1, a2);
536 } else {
537 // When Q = 0, the above formulas have problems. If we look at
538 // the z-transform, we can see that the limit as Q->0 is 1, so
539 // set the filter that way.
540 setNormalizedCoefficients(index, 1, 0, 0, 1, 0, 0);
541 }
542 } else {
543 // When the cutoff is zero, the z-transform approaches 0, if Q
544 // > 0. When both Q and cutoff are zero, the z-transform is
545 // pretty much undefined. What should we do in this case?
546 // For now, just make the filter 0. When the cutoff is 1, the
547 // z-transform also approaches 0.
548 setNormalizedCoefficients(index, 0, 0, 0, 1, 0, 0);
549 }

552void Biquad::getFrequencyResponse(unsigned nFrequencies, const float* frequency, float* magResponse, float* phaseResponse)
553{
554 // Evaluate the Z-transform of the filter at given normalized
555 // frequency from 0 to 1. (1 corresponds to the Nyquist
556 // frequency.)
557 //
558 // The z-transform of the filter is
559 //
560 // H(z) = (b0 + b1*z^(-1) + b2*z^(-2))/(1 + a1*z^(-1) + a2*z^(-2))
561 //
562 // Evaluate as
563 //
564 // b0 + (b1 + b2*z1)*z1
565 // --------------------
566 // 1 + (a1 + a2*z1)*z1
567 //
568 // with z1 = 1/z and z = exp(j*pi*frequency). Hence z1 = exp(-j*pi*frequency)
569
570 // Make local copies of the coefficients as a micro-optimization.
571 double b0 = m_b0[0];
572 double b1 = m_b1[0];
573 double b2 = m_b2[0];
574 double a1 = m_a1[0];
575 double a2 = m_a2[0];
576
577 for (unsigned k = 0; k < nFrequencies; ++k) {
578 if (frequency[k] < 0 || frequency[k] > 1) {
579 // Out-of-bounds frequencies should return NaN.
580 magResponse[k] = std::nanf("");
581 phaseResponse[k] = std::nanf("");
582 } else {
583 double omega = -piDouble * frequency[k];
584 std::complex<double> z = std::complex<double>(cos(omega), sin(omega));
585 std::complex<double> numerator = b0 + (b1 + b2 * z) * z;
586 std::complex<double> denominator = std::complex<double>(1, 0) + (a1 + a2 * z) * z;
587 std::complex<double> response = numerator / denominator;
588 magResponse[k] = static_cast<float>(abs(response));
589 phaseResponse[k] = static_cast<float>(atan2(imag(response), real(response)));
590 }
591 }
592}

594static double repeatedRootResponse(double n, double c1, double c2, double r, double logEPS)
595{
596 // The response is h(n) = r^(n-2)*[c1*(n+1)*r^2+c2]. We're looking
597 // for n such that |h(n)| = eps. Equivalently, we want a root
598 // of the equation log(|h(n)|) - log(eps) = 0 or
599 //
600 // (n-2)*log(r) + log(|c1*(n+1)*r^2+c2|) - log(eps)
601 //
602 // This helps with finding a nuemrical solution because this
603 // approximately linearizes the response for large n.
604
605 return (n - 2) * std::log(r) + std::log(std::fabs(c1 * (n + 1) * r * r + c2)) - logEPS;
606}
607
608
609// Regula Falsi root finder, Illinois variant
610// (https://en.wikipedia.org/wiki/False_position_method#The_Illinois_algorithm).
611//
612// This finds a root of the repeated root response where the root is
613// assumed to lie between |low| and |high|. The response is given by
614// |c1|, |c2|, and |r| as determined by |RepeatedRootResponse|.
615// |logEPS| is the log the the maximum allowed amplitude in the
616// response.
617static double rootFinder(double low, double high, double logEPS, double c1, double c2, double r)
618{
619 // Desired accuray of the root (in frames). This doesn't need to be
620 // super-accurate, so half frame is good enough, and should be less
621 // than 1 because the algorithm may prematurely terminate.
622 constexpr double accuracyThreshold = 0.5;
623 // Max number of iterations to do. If we haven't converged by now,
624 // just return whatever we've found.
625 constexpr int maxIterations = 10;
626
627 int side = 0;
628 double root = 0;
629 double fLow = repeatedRootResponse(low, c1, c2, r, logEPS);
630 double fHigh = repeatedRootResponse(high, c1, c2, r, logEPS);
631
632 // The function values must be finite and have opposite signs!
633 ASSERT(std::isfinite(fLow));
634 ASSERT(std::isfinite(fHigh));
635 ASSERT(fLow * fHigh <= 0);
636
637 int iteration;
638 for (iteration = 0; iteration < maxIterations; ++iteration) {
639 root = (fLow * high - fHigh * low) / (fLow - fHigh);
640 if (fabs(high - low) < accuracyThreshold * std::fabs(high + low))
641 break;
642 double fr = repeatedRootResponse(root, c1, c2, r, logEPS);
643
644 ASSERT(std::isfinite(fr));
645
646 if (fr * fHigh > 0) {
647 // fr and fHigh have same sign. Copy root to fHigh
648 high = root;
649 fHigh = fr;
650 side = -1;
651 } else if (fLow * fr > 0) {
652 // fr and fLow have same sign. Copy root to fLow
653 low = root;
654 fLow = fr;
655 if (side == 1)
656 fHigh /= 2;
657 side = 1;
658 } else {
659 // fLow * fr looks like zero, so assume we've converged.
660 break;
661 }
662 }
663
664 // Want to know if the max number of iterations is ever exceeded so
665 // we can understand why that happened.
666 ASSERT(iteration < maxIterations);
667
668 return root;
669}
670
671double Biquad::tailFrame(size_t coefIndex, double maxFrame)
672{
673 // The Biquad filter is given by
674 //
675 // H(z) = (b0 + b1/z + b2/z^2)/(1 + a1/z + a2/z^2).
676 //
677 // To compute the tail time, compute the impulse response, h(n), of
678 // H(z), which we can do analytically. From this impulse response,
679 // find the value n0 where |h(n)| <= eps for n >= n0.
680 //
681 // Assume first that the two poles of H(z) are not repeated, say r1
682 // and r2. Then, we can compute a partial fraction expansion of
683 // H(z):
684 //
685 // H(z) = (b0+b1/z+b2/z^2)/[(1-r1/z)*(1-r2/z)]
686 // = b0 + C2/(z-r2) - C1/(z-r1)
687 //
688 // where
689 // C2 = (b0*r2^2+b1*r2+b2)/(r2-r1)
690 // C1 = (b0*r1^2+b1*r1+b2)/(r2-r1)
691 //
692 // Expand H(z) then this in powers of 1/z gives:
693 //
694 // H(z) = b0 -(C2/r2+C1/r1) + sum(C2*r2^(i-1)/z^i + C1*r1^(i-1)/z^i)
695 //
696 // Thus, for n > 1 (we don't care about small n),
697 //
698 // h(n) = C2*r2^(n-1) + C1*r1^(n-1)
699 //
700 // We need to find n0 such that |h(n)| < eps for n > n0.
701 //
702 // Case 1: r1 and r2 are real and distinct, with |r1|>=|r2|.
703 //
704 // Then
705 //
706 // h(n) = C1*r1^(n-1)*(1 + C2/C1*(r2/r1)^(n-1))
707 //
708 // so
709 //
710 // |h(n)| = |C1|*|r1|^(n-1)*|1+C2/C1*(r2/r1)^(n-1)|
711 // <= |C1|*|r1|^(n-1)*[1 + |C2/C1|*|r2/r1|^(n-1)]
712 // <= |C1|*|r1|^(n-1)*[1 + |C2/C1|]
713 //
714 // so
715 //
716 // |h(n)| = |C1|*|r1|^(n-1)*|1+C2/C1*(r2/r1)^(n-1)|
717 // <= |C1|*|r1|^(n-1)*[1 + |C2/C1|*|r2/r1|^(n-1)]
718 // <= |C1|*|r1|^(n-1)*[1 + |C2/C1|]
719 //
720 // by using the triangle inequality and the fact that |r2|<=|r1|.
721 // And we want |h(n)|<=eps which is true if
722 //
723 // |C1|*|r1|^(n-1)*[1 + |C2/C1|] <= eps
724 //
725 // or
726 //
727 // n >= 1 + log(eps/C)/log(|r1|)
728 //
729 // where C = |C1|*[1+|C2/C1|] = |C1| + |C2|.
730 //
731 // Case 2: r1 and r2 are complex
732 //
733 // Thne we can write r1=r*exp(i*p) and r2=r*exp(-i*p). So,
734 //
735 // |h(n)| = |C2*r^(n-1)*exp(-i*p*(n-1)) + C1*r^(n-1)*exp(i*p*(n-1))|
736 // = |C1|*r^(n-1)*|1 + C2/C1*exp(-i*p*(n-1))/exp(i*n*(n-1))|
737 // <= |C1|*r^(n-1)*[1 + |C2/C1|]
738 //
739 // Again, this is easily solved to give
740 //
741 // n >= 1 + log(eps/C)/log(r)
742 //
743 // where C = |C1|*[1+|C2/C1|] = |C1| + |C2|.
744 //
745 // Case 3: Repeated roots, r1=r2=r.
746 //
747 // In this case,
748 //
749 // H(z) = (b0+b1/z+b2/z^2)/[(1-r/z)^2
750 //
751 // Expanding this in powers of 1/z gives:
752 //
753 // H(z) = C1*sum((i+1)*r^i/z^i) - C2 * sum(r^(i-2)/z^i) + b2/r^2
754 // = b2/r^2 + sum([C1*(i+1)*r^i + C2*r^(i-2)]/z^i)
755 // where
756 // C1 = (b0*r^2+b1*r+b2)/r^2
757 // C2 = b1*r+2*b2
758 //
759 // Thus, the impulse response is
760 //
761 // h(n) = C1*(n+1)*r^n + C2*r^(n-2)
762 // = r^(n-2)*[C1*(n+1)*r^2+C2]
763 //
764 // So
765 //
766 // |h(n)| = |r|^(n-2)*|C1*(n+1)*r^2+C2|
767 //
768 // To find n such that |h(n)| < eps, we need a numerical method in
769 // general, so there's no real reason to simplify this or use other
770 // approximations. Just solve |h(n)|=eps directly.
771 //
772 // Thus, for an set of filter coefficients, we can compute the tail
773 // time.
774 //
775
776 // If the maximum amplitude of the impulse response is less than
777 // this, we assume that we've reached the tail of the response.
778 // Currently, this means that the impulse is less than 1 bit of a
779 // 16-bit PCM value.
780 constexpr double maxTailAmplitude = 1 / 32768.0;
781
782 // Find the roots of 1+a1/z+a2/z^2 = 0. Or equivalently,
783 // z^2+a1*z+a2 = 0. From the quadratic formula the roots are
784 // (-a1+/-sqrt(a1^2-4*a2))/2.
785
786 double a1 = m_a1[coefIndex];
787 double a2 = m_a2[coefIndex];
788 double b0 = m_b0[coefIndex];
789 double b1 = m_b1[coefIndex];
790 double b2 = m_b2[coefIndex];
791
792 double discrim = a1 * a1 - 4 * a2;
793 if (discrim > 0) {
794 // Compute the real roots so that r1 has the largest magnitude.
795 double rplus = (-a1 + std::sqrt(discrim)) / 2;
796 double rminus = (-a1 - std::sqrt(discrim)) / 2;
797 double r1 = std::fabs(rplus) >= std::fabs(rminus) ? rplus : rminus;
798 // Use the fact that a2 = r1*r2
799 double r2 = a2 / r1;
800
801 double c1 = (b0 * r1 * r1 + b1 * r1 + b2) / (r2 - r1);
802 double c2 = (b0 * r2 * r2 + b1 * r2 + b2) / (r2 - r1);
803
804 ASSERT(std::isfinite(r1));
805 ASSERT(std::isfinite(r2));
806 ASSERT(std::isfinite(c1));
807 ASSERT(std::isfinite(c2));
808
809 // It's possible for maxTailAmplitude to be greater than c1 + c2.
810 // This may produce a negative tail frame. Just clamp the tail
811 // frame to 0.
812 double tailFrame = clampTo(1 + std::log(maxTailAmplitude / (std::fabs(c1) + std::fabs(c2))) / std::log(std::fabs(r1)), 0);
813
814 ASSERT(std::isfinite(tailFrame));
815 return tailFrame;
816 }
817
818 if (discrim < 0) {
819 // Two complex roots.
820 // One root is -a1/2 + i*sqrt(-discrim)/2.
821 double x = -a1 / 2;
822 double y = std::sqrt(-discrim) / 2;
823 std::complex<double> r1(x, y);
824 std::complex<double> r2(x, -y);
825 double r = std::hypot(x, y);
826
827 ASSERT(std::isfinite(r));
828
829 double tailFrame = 0;
830 // It's possible for r to be 1. (LPF with Q very large can cause this.)
831 if (r == 1)
832 tailFrame = maxFrame;
833 else {
834 double c1 = std::abs((b0 * r1 * r1 + b1 * r1 + b2) / (r2 - r1));
835 double c2 = std::abs((b0 * r2 * r2 + b1 * r2 + b2) / (r2 - r1));
836
837 ASSERT(std::isfinite(c1));
838 ASSERT(std::isfinite(c2));
839 tailFrame = 1 + std::log(maxTailAmplitude / (c1 + c2)) / std::log(r);
840 if (!c1 && !c2) {
841 // If c1 = c2 = 0, then H(z) = b0. Hence, there's no tail
842 // because this is just a wire from input to output.
843 tailFrame = 0;
844 } else {
845 // Otherwise, check that the tail has finite length. Not
846 // strictly necessary, but we want to know if this ever
847 // happens.
848 ASSERT(std::isfinite(tailFrame));
849 }
850 }
851 return tailFrame;
852 }
853
854 // Repeated roots. This should be pretty rare because all the
855 // coefficients need to be just the right values to get a
856 // discriminant of exactly zero.
857 double r = -a1 / 2;
858
859 double tailFrame = 0;
860 if (!r) {
861 // Double pole at 0. This just delays the signal by 2 frames,
862 // so set the tail frame to 2.
863 tailFrame = 2;
864 } else {
865 double c1 = (b0 * r * r + b1 * r + b2) / (r * r);
866 double c2 = b1 * r + 2 * b2;
867
868 ASSERT(std::isfinite(c1));
869 ASSERT(std::isfinite(c2));
870
871 // It can happen that c1=c2=0. This basically means that H(z) =
872 // constant, which is the limiting case for several of the
873 // biquad filters.
874 if (!c1 && !c2)
875 tailFrame = 0;
876 else {
877 // The function c*(n+1)*r^n is not monotonic, but it's easy to
878 // find the max point since the derivative is
879 // c*r^n*(1+(n+1)*log(r)). This has a root at
880 // -(1+log(r))/log(r). so we can start our search from that
881 // point to maxFrames.
882
883 double low = std::clamp(-(1 + std::log(r)) / std::log(r), 1.0, static_cast<double>(maxFrame - 1));
884 double high = maxFrame;
885
886 ASSERT(std::isfinite(low));
887 ASSERT(std::isfinite(high));
888
889 tailFrame = rootFinder(low, high, std::log(maxTailAmplitude), c1, c2, r);
890 }
891 }
892
893 return tailFrame;
894}
