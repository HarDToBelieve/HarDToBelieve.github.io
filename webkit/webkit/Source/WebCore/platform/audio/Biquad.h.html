29#ifndef Biquad_h
30#define Biquad_h
31
32#include "AudioArray.h"
33#include <complex>
34#include <sys/types.h>
35
36namespace WebCore {
37
38// A basic biquad (two-zero / two-pole digital filter)
39//
40// It can be configured to a number of common and very useful filters:
41// lowpass, highpass, shelving, parameteric, notch, allpass, ...
42
43class Biquad final {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 Biquad();
47 ~Biquad();
48
49 void process(const float* sourceP, float* destP, size_t framesToProcess);
50
51 bool hasSampleAccurateValues() const { return m_hasSampleAccurateValues; }
52 void setHasSampleAccurateValues(bool hasSampleAccurateValues) { m_hasSampleAccurateValues = hasSampleAccurateValues; }
53
54 // frequency is 0 - 1 normalized, resonance and dbGain are in decibels.
55 // Q is a unitless quality factor.
56 void setLowpassParams(size_t index, double frequency, double resonance);
57 void setHighpassParams(size_t index, double frequency, double resonance);
58 void setBandpassParams(size_t index, double frequency, double Q);
59 void setLowShelfParams(size_t index, double frequency, double dbGain);
60 void setHighShelfParams(size_t index, double frequency, double dbGain);
61 void setPeakingParams(size_t index, double frequency, double Q, double dbGain);
62 void setAllpassParams(size_t index, double frequency, double Q);
63 void setNotchParams(size_t index, double frequency, double Q);
64
65 // Resets filter state
66 void reset();
67
68 // Filter response at a set of n frequencies. The magnitude and
69 // phase response are returned in magResponse and phaseResponse.
70 // The phase response is in radians.
71 void getFrequencyResponse(unsigned nFrequencies, const float* frequency, float* magResponse, float* phaseResponse);
72
73 // Compute tail frame based on the filter coefficents at index
74 // |coefIndex|. The tail frame is the frame number where the
75 // impulse response of the filter falls below a threshold value.
76 // The maximum allowed frame value is given by |maxFrame|. This
77 // limits how much work is done in computing the frame number.
78 double tailFrame(size_t coefIndex, double maxFrame);
79
80private:
81 void setNormalizedCoefficients(size_t index, double b0, double b1, double b2, double a0, double a1, double a2);
82
83 // Filter coefficients. The filter is defined as
84 //
85 // y[n] + m_a1*y[n-1] + m_a2*y[n-2] = m_b0*x[n] + m_b1*x[n-1] + m_b2*x[n-2].
86 AudioDoubleArray m_b0;
87 AudioDoubleArray m_b1;
88 AudioDoubleArray m_b2;
89 AudioDoubleArray m_a1;
90 AudioDoubleArray m_a2;
91
92#if USE(ACCELERATE)
93 void processFast(const float* sourceP, float* destP, size_t framesToProcess);
94 void processSliceFast(double* sourceP, double* destP, double* coefficientsP, size_t framesToProcess);
95 AudioDoubleArray m_inputBuffer;
96 AudioDoubleArray m_outputBuffer;
97#endif
98
99 // Filter memory
100 double m_x1; // input delayed by 1 sample
101 double m_x2; // input delayed by 2 samples
102 double m_y1; // output delayed by 1 sample
103 double m_y2; // output delayed by 2 samples
104
105 bool m_hasSampleAccurateValues { false };
106};
107
108} // namespace WebCore
109
110#endif // Biquad_h
