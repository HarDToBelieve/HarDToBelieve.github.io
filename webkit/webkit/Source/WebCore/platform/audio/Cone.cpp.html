40double ConeEffect::gain(FloatPoint3D sourcePosition, FloatPoint3D sourceOrientation, FloatPoint3D listenerPosition)
41{
42 if (sourceOrientation.isZero() || ((m_innerAngle == 360.0) && (m_outerAngle == 360.0)))
43 return 1.0; // no cone specified - unity gain
44
45 // Normalized source-listener vector
46 FloatPoint3D sourceToListener = listenerPosition - sourcePosition;
47 sourceToListener.normalize();
48
49 FloatPoint3D normalizedSourceOrientation = sourceOrientation;
50 normalizedSourceOrientation.normalize();
51
52 // Angle between the source orientation vector and the source-listener vector
53 double dotProduct = sourceToListener.dot(normalizedSourceOrientation);
54 double angle = 180.0 * acos(dotProduct) / piDouble;
55 double absAngle = fabs(angle);
56
57 // Divide by 2.0 here since API is entire angle (not half-angle)
58 double absInnerAngle = fabs(m_innerAngle) / 2.0;
59 double absOuterAngle = fabs(m_outerAngle) / 2.0;
60 double gain = 1.0;
61
62 if (absAngle <= absInnerAngle)
63 // No attenuation
64 gain = 1.0;
65 else if (absAngle >= absOuterAngle)
66 // Max attenuation
67 gain = m_outerGain;
68 else {
69 // Between inner and outer cones
70 // inner -> outer, x goes from 0 -> 1
71 double x = (absAngle - absInnerAngle) / (absOuterAngle - absInnerAngle);
72 gain = (1.0 - x) + m_outerGain * x;
73 }
74
75 return gain;
76}
