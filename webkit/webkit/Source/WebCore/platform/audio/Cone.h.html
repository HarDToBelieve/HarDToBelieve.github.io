29#ifndef Cone_h
30#define Cone_h
31
32#include "FloatPoint3D.h"
33
34namespace WebCore {
35
36// Cone gain is defined according to the OpenAL specification
37
38class ConeEffect final {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 ConeEffect();
42
43 // Returns scalar gain for the given source/listener positions/orientations
44 double gain(FloatPoint3D sourcePosition, FloatPoint3D sourceOrientation, FloatPoint3D listenerPosition);
45
46 // Angles in degrees
47 void setInnerAngle(double innerAngle) { m_innerAngle = innerAngle; }
48 double innerAngle() const { return m_innerAngle; }
49
50 void setOuterAngle(double outerAngle) { m_outerAngle = outerAngle; }
51 double outerAngle() const { return m_outerAngle; }
52
53 void setOuterGain(double outerGain) { m_outerGain = outerGain; }
54 double outerGain() const { return m_outerGain; }
55
56protected:
57 double m_innerAngle { 360 };
58 double m_outerAngle { 360 };
59 double m_outerGain { 0 };
60};
61
62} // namespace WebCore
63
64#endif // Cone_h
