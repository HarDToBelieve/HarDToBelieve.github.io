49void DirectConvolver::process(AudioFloatArray* convolutionKernel, const float* sourceP, float* destP, size_t framesToProcess)
50{
51 ASSERT(framesToProcess == m_inputBlockSize);
52 if (framesToProcess != m_inputBlockSize)
53 return;
54
55 // Only support kernelSize <= m_inputBlockSize
56 size_t kernelSize = convolutionKernel->size();
57 ASSERT(kernelSize <= m_inputBlockSize);
58 if (kernelSize > m_inputBlockSize)
59 return;
60
61 float* kernelP = convolutionKernel->data();
62
63 // Sanity check
64 bool isCopyGood = kernelP && sourceP && destP && m_buffer.data();
65 ASSERT(isCopyGood);
66 if (!isCopyGood)
67 return;
68
69 float* inputP = m_buffer.data() + m_inputBlockSize;

71 // Copy samples to 2nd half of input buffer.
