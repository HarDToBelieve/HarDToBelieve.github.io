38namespace WebCore {

40DistanceEffect::DistanceEffect() = default;
41
42double DistanceEffect::gain(double distance)
43{
44 // don't go beyond maximum distance
45 distance = std::min(distance, m_maxDistance);
46
47 // if clamped, don't get closer than reference distance
48 if (m_isClamped)
49 distance = std::max(distance, m_refDistance);
50
51 switch (m_model) {
52 case DistanceModelType::Linear:
53 return linearGain(distance);
54 case DistanceModelType::Inverse:
55 return inverseGain(distance);
56 case DistanceModelType::Exponential:
57 return exponentialGain(distance);
58 }
59 ASSERT_NOT_REACHED();
60 return 0.0;
61}
62
63double DistanceEffect::linearGain(double distance)
64{
65 auto clampedRolloffFactor = std::clamp(m_rolloffFactor, 0.0, 1.0);
66 // We want a gain that decreases linearly from m_refDistance to
67 // m_maxDistance. The gain is 1 at m_refDistance.
68 return (1.0 - clampedRolloffFactor * (distance - m_refDistance) / (m_maxDistance - m_refDistance));
69}
70
71double DistanceEffect::inverseGain(double distance)
72{
73 return m_refDistance / (m_refDistance + m_rolloffFactor * (distance - m_refDistance));
74}
75
76double DistanceEffect::exponentialGain(double distance)
77{
78 return pow(distance / m_refDistance, -m_rolloffFactor);
79}
