44DynamicsCompressor::DynamicsCompressor(float sampleRate, unsigned numberOfChannels)
45 : m_numberOfChannels(numberOfChannels)
46 , m_sampleRate(sampleRate)
47 , m_compressor(sampleRate, numberOfChannels)
48{
49 setNumberOfChannels(numberOfChannels);
50 initializeParameters();
51}

53void DynamicsCompressor::setParameterValue(unsigned parameterID, float value)
54{
55 ASSERT(parameterID < ParamLast);
56 if (parameterID < ParamLast)
57 m_parameters[parameterID] = value;
58}

60void DynamicsCompressor::initializeParameters()
61{
62 // Initializes compressor to default values.
63
64 m_parameters[ParamThreshold] = -24; // dB
65 m_parameters[ParamKnee] = 30; // dB
66 m_parameters[ParamRatio] = 12; // unit-less
67 m_parameters[ParamAttack] = 0.003f; // seconds
68 m_parameters[ParamRelease] = 0.250f; // seconds
69 m_parameters[ParamPreDelay] = 0.006f; // seconds
70
71 // Release zone values 0 -> 1.
72 m_parameters[ParamReleaseZone1] = 0.09f;
73 m_parameters[ParamReleaseZone2] = 0.16f;
74 m_parameters[ParamReleaseZone3] = 0.42f;
75 m_parameters[ParamReleaseZone4] = 0.98f;
76
77 m_parameters[ParamPostGain] = 0; // dB
78 m_parameters[ParamReduction] = 0; // dB
79
80 // Linear crossfade (0 -> 1).
81 m_parameters[ParamEffectBlend] = 1;
82}
83
84float DynamicsCompressor::parameterValue(unsigned parameterID)
85{
86 ASSERT(parameterID < ParamLast);
87 return m_parameters[parameterID];
88}

90void DynamicsCompressor::process(const AudioBus* sourceBus, AudioBus* destinationBus, unsigned framesToProcess)
91{
92 // Though numberOfChannels is retrived from destinationBus, we still name it numberOfChannels instead of numberOfDestinationChannels.
93 // It's because we internally match sourceChannels's size to destinationBus by channel up/down mix. Thus we need numberOfChannels
94 // to do the loop work for both m_sourceChannels and m_destinationChannels.
95
96 unsigned numberOfChannels = destinationBus->numberOfChannels();
97 unsigned numberOfSourceChannels = sourceBus->numberOfChannels();
98
99 ASSERT(numberOfChannels == m_numberOfChannels && numberOfSourceChannels);
100
101 if (numberOfChannels != m_numberOfChannels || !numberOfSourceChannels) {
102 destinationBus->zero();
103 return;
104 }
105
106 switch (numberOfChannels) {
107 case 2: // stereo
108 m_sourceChannels[0] = sourceBus->channel(0)->data();
109
110 if (numberOfSourceChannels > 1)
111 m_sourceChannels[1] = sourceBus->channel(1)->data();
112 else
113 // Simply duplicate mono channel input data to right channel for stereo processing.
114 m_sourceChannels[1] = m_sourceChannels[0];
115
116 break;
117 default:
118 // FIXME : support other number of channels.
119 ASSERT_NOT_REACHED();
120 destinationBus->zero();
121 return;
122 }
123
124 for (unsigned i = 0; i < numberOfChannels; ++i)
125 m_destinationChannels[i] = destinationBus->channel(i)->mutableData();
126
127 float dbThreshold = parameterValue(ParamThreshold);
128 float dbKnee = parameterValue(ParamKnee);
129 float ratio = parameterValue(ParamRatio);
130 float attackTime = parameterValue(ParamAttack);
131 float releaseTime = parameterValue(ParamRelease);
132 float preDelayTime = parameterValue(ParamPreDelay);
133
134 // This is effectively a master volume on the compressed signal (pre-blending).
135 float dbPostGain = parameterValue(ParamPostGain);
136
137 // Linear blending value from dry to completely processed (0 -> 1)
138 // 0 means the signal is completely unprocessed.
139 // 1 mixes in only the compressed signal.
140 float effectBlend = parameterValue(ParamEffectBlend);
141
142 float releaseZone1 = parameterValue(ParamReleaseZone1);
143 float releaseZone2 = parameterValue(ParamReleaseZone2);
144 float releaseZone3 = parameterValue(ParamReleaseZone3);
145 float releaseZone4 = parameterValue(ParamReleaseZone4);
146
147 // Apply compression to the source signal.
148 m_compressor.process(m_sourceChannels.get(),
149 m_destinationChannels.get(),
150 numberOfChannels,
151 framesToProcess,
152
153 dbThreshold,
154 dbKnee,
155 ratio,
156 attackTime,
157 releaseTime,
158 preDelayTime,
159 dbPostGain,
160 effectBlend,
161
162 releaseZone1,
163 releaseZone2,
164 releaseZone3,
165 releaseZone4
166 );
167
168 // Update the compression amount.
169 setParameterValue(ParamReduction, m_compressor.meteringGain());
170}
171
172void DynamicsCompressor::reset()
173{
174 m_compressor.reset();
175}

177void DynamicsCompressor::setNumberOfChannels(unsigned numberOfChannels)
178{
179 m_sourceChannels = makeUniqueArray<const float*>(numberOfChannels);
180 m_destinationChannels = makeUniqueArray<float*>(numberOfChannels);
181
182 m_compressor.setNumberOfChannels(numberOfChannels);
183 m_numberOfChannels = numberOfChannels;
184}
