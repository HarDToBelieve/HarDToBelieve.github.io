29#ifndef DynamicsCompressor_h
30#define DynamicsCompressor_h
31
32#include "DynamicsCompressorKernel.h"
33#include "ZeroPole.h"
34#include <memory>
35#include <wtf/UniqueArray.h>
36
37namespace WebCore {
38
39class AudioBus;
40
41// DynamicsCompressor implements a flexible audio dynamics compression effect such as
42// is commonly used in musical production and game audio. It lowers the volume
43// of the loudest parts of the signal and raises the volume of the softest parts,
44// making the sound richer, fuller, and more controlled.
45
46class DynamicsCompressor final {
47 WTF_MAKE_FAST_ALLOCATED;
48 WTF_MAKE_NONCOPYABLE(DynamicsCompressor);
49public:
50 enum {
51 ParamThreshold,
52 ParamKnee,
53 ParamRatio,
54 ParamAttack,
55 ParamRelease,
56 ParamPreDelay,
57 ParamReleaseZone1,
58 ParamReleaseZone2,
59 ParamReleaseZone3,
60 ParamReleaseZone4,
61 ParamPostGain,
62 ParamEffectBlend,
63 ParamReduction,
64 ParamLast
65 };
66
67 DynamicsCompressor(float sampleRate, unsigned numberOfChannels);
68
69 void process(const AudioBus* sourceBus, AudioBus* destinationBus, unsigned framesToProcess);
70 void reset();
71 void setNumberOfChannels(unsigned);
72
73 void setParameterValue(unsigned parameterID, float value);
74 float parameterValue(unsigned parameterID);
75
76 float sampleRate() const { return m_sampleRate; }
77 float nyquist() const { return m_sampleRate / 2; }
78
79 double tailTime() const { return m_compressor.tailTime(); }
80 double latencyTime() const { return m_compressor.latencyFrames() / static_cast<double>(sampleRate()); }
81 bool requiresTailProcessing() const
82 {
83 // Always return true even if the tail time and latency might both be zero.
84 return true;
85 }
86
87protected:
88 unsigned m_numberOfChannels;
89
90 // m_parameters holds the tweakable compressor parameters.
91 float m_parameters[ParamLast];
92 void initializeParameters();
93
94 float m_sampleRate;
95
96 UniqueArray<const float*> m_sourceChannels;
97 UniqueArray<float*> m_destinationChannels;
98
99 // The core compressor.
100 DynamicsCompressorKernel m_compressor;
101};
102
103} // namespace WebCore
104
105#endif // DynamicsCompressor_h
