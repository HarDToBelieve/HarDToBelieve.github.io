29#ifndef DynamicsCompressorKernel_h
30#define DynamicsCompressorKernel_h
31
32#include "AudioArray.h"
33#include <memory>
34#include <wtf/Vector.h>
35
36namespace WebCore {
37
38class DynamicsCompressorKernel final {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 explicit DynamicsCompressorKernel(float sampleRate, unsigned numberOfChannels);
42
43 void setNumberOfChannels(unsigned);
44
45 // Performs stereo-linked compression.
46 void process(const float* sourceChannels[],
47 float* destinationChannels[],
48 unsigned numberOfChannels,
49 unsigned framesToProcess,
50
51 float dbThreshold,
52 float dbKnee,
53 float ratio,
54 float attackTime,
55 float releaseTime,
56 float preDelayTime,
57 float dbPostGain,
58 float effectBlend,
59
60 float releaseZone1,
61 float releaseZone2,
62 float releaseZone3,
63 float releaseZone4
64 );
65
66 void reset();
67
68 unsigned latencyFrames() const { return m_lastPreDelayFrames; }
69
70 float sampleRate() const { return m_sampleRate; }
71
72 float meteringGain() const { return m_meteringGain; }
73
74 double tailTime() const;
75
76protected:
77 static constexpr float uninitializedValue = -1;
78 float m_sampleRate;
79
80 float m_detectorAverage;
81 float m_compressorGain;
82
83 // Metering
84 float m_meteringReleaseK;
85 float m_meteringGain;
86
87 // Lookahead section.
88 enum { MaxPreDelayFrames = 1024 };
89 enum { MaxPreDelayFramesMask = MaxPreDelayFrames - 1 };
90 enum { DefaultPreDelayFrames = 256 }; // setPreDelayTime() will override this initial value
91 unsigned m_lastPreDelayFrames { DefaultPreDelayFrames };
92 void setPreDelayTime(float);
93
94 Vector<std::unique_ptr<AudioFloatArray>> m_preDelayBuffers;
95 int m_preDelayReadIndex { 0 };
96 int m_preDelayWriteIndex { DefaultPreDelayFrames };
97
98 float m_maxAttackCompressionDiffDb;
99
100 // Static compression curve.
101 float kneeCurve(float x, float k);
102 float saturate(float x, float k);
103 float slopeAt(float x, float k);
104 float kAtSlope(float desiredSlope);
105
106 float updateStaticCurveParameters(float dbThreshold, float dbKnee, float ratio);
107
108 // Amount of input change in dB required for 1 dB of output change.
109 // This applies to the portion of the curve above m_kneeThresholdDb (see below).
110 float m_ratio { uninitializedValue };
111 float m_slope { uninitializedValue }; // Inverse ratio.
112
113 // The input to output change below the threshold is linear 1:1.
114 float m_linearThreshold { uninitializedValue };
115 float m_dbThreshold { uninitializedValue };
116
117 // m_dbKnee is the number of dB above the threshold before we enter the "ratio" portion of the curve.
118 // m_kneeThresholdDb = m_dbThreshold + m_dbKnee
119 // The portion between m_dbThreshold and m_kneeThresholdDb is the "soft knee" portion of the curve
120 // which transitions smoothly from the linear portion to the ratio portion.
121 float m_dbKnee { uninitializedValue };
122 float m_kneeThreshold { uninitializedValue };
123 float m_kneeThresholdDb { uninitializedValue };
124 float m_ykneeThresholdDb { uninitializedValue };
125
126 // Internal parameter for the knee portion of the curve.
127 float m_K { uninitializedValue };
128};
129
130} // namespace WebCore
131
132#endif // DynamicsCompressorKernel_h
