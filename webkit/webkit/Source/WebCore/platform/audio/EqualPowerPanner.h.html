25#ifndef EqualPowerPanner_h
26#define EqualPowerPanner_h
27
28#include "Panner.h"
29
30namespace WebCore {
31
32// Common type of stereo panner as found in normal audio mixing equipment.
33
34class EqualPowerPanner final : public Panner {
35public:
36 explicit EqualPowerPanner(float sampleRate);
37
38 void pan(double azimuth, double elevation, const AudioBus* inputBus, AudioBus* outputBuf, size_t framesToProcess) final;
39 void panWithSampleAccurateValues(double* azimuth, double* elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess) final;
40
41 void reset() override { m_isFirstRender = true; }
42
43 double tailTime() const override { return 0; }
44 double latencyTime() const override { return 0; }
45 bool requiresTailProcessing() const final { return false; }
46
47private:
48 void calculateDesiredGain(double& desiredGainL, double& desiredGainR, double azimuth, unsigned numberOfChannels);
49
50 // For smoothing / de-zippering
51 bool m_isFirstRender { true };
52 double m_smoothingConstant;
53
54 double m_gainL { 0 };
55 double m_gainR { 0 };
56};
57
58} // namespace WebCore
59
60#endif // EqualPowerPanner_h
