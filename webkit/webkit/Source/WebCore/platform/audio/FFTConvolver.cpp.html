39FFTConvolver::FFTConvolver(size_t fftSize)
40 : m_frame(fftSize)
41 , m_inputBuffer(fftSize) // 2nd half of buffer is always zeroed
42 , m_outputBuffer(fftSize)
43 , m_lastOverlapBuffer(fftSize / 2)
44{
45}
46
47void FFTConvolver::process(FFTFrame* fftKernel, const float* sourceP, float* destP, size_t framesToProcess)
48{
49 size_t halfSize = fftSize() / 2;
50
51 // framesToProcess must be an exact multiple of halfSize,
52 // or halfSize is a multiple of framesToProcess when halfSize > framesToProcess.
53 bool isGood = !(halfSize % framesToProcess && framesToProcess % halfSize);
54 ASSERT(isGood);
55 if (!isGood)
56 return;
57
58 size_t numberOfDivisions = halfSize <= framesToProcess ? (framesToProcess / halfSize) : 1;
59 size_t divisionSize = numberOfDivisions == 1 ? framesToProcess : halfSize;
60
61 for (size_t i = 0; i < numberOfDivisions; ++i, sourceP += divisionSize, destP += divisionSize) {
62 // Copy samples to input buffer (note contraint above!)
63 float* inputP = m_inputBuffer.data();
64
65 // Sanity check
66 bool isCopyGood1 = sourceP && inputP && m_readWriteIndex + divisionSize <= m_inputBuffer.size();
67 ASSERT(isCopyGood1);
68 if (!isCopyGood1)
69 return;
70
71 memcpy(inputP + m_readWriteIndex, sourceP, sizeof(float) * divisionSize);
72
73 // Copy samples from output buffer
74 float* outputP = m_outputBuffer.data();
75
76 // Sanity check
77 bool isCopyGood2 = destP && outputP && m_readWriteIndex + divisionSize <= m_outputBuffer.size();
78 ASSERT(isCopyGood2);
79 if (!isCopyGood2)
80 return;
81
82 memcpy(destP, outputP + m_readWriteIndex, sizeof(float) * divisionSize);
83 m_readWriteIndex += divisionSize;
84
85 // Check if it's time to perform the next FFT
86 if (m_readWriteIndex == halfSize) {
87 // The input buffer is now filled (get frequency-domain version)
88 m_frame.doFFT(m_inputBuffer.data());
89 m_frame.multiply(*fftKernel);
90 m_frame.doInverseFFT(m_outputBuffer.data());
91
92 // Overlap-add 1st half from previous time
93 VectorMath::add(m_outputBuffer.data(), m_lastOverlapBuffer.data(), m_outputBuffer.data(), halfSize);
94
95 // Finally, save 2nd half of result
96 bool isCopyGood3 = m_outputBuffer.size() == 2 * halfSize && m_lastOverlapBuffer.size() == halfSize;
97 ASSERT(isCopyGood3);
98 if (!isCopyGood3)
99 return;
100
101 memcpy(m_lastOverlapBuffer.data(), m_outputBuffer.data() + halfSize, sizeof(float) * halfSize);
102
103 // Reset index back to start for next time
104 m_readWriteIndex = 0;
105 }
106 }
107}
108
109void FFTConvolver::reset()
110{
111 m_lastOverlapBuffer.zero();
112 m_readWriteIndex = 0;
113}
