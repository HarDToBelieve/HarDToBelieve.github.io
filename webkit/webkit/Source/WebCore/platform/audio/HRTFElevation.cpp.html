47// Total number of components of an HRTF database.
48constexpr size_t TotalNumberOfResponses = 240;
49
50// Number of frames in an individual impulse response.
51constexpr size_t ResponseFrameSize = 256;
52
53// Sample-rate of the spatialization impulse responses as stored in the resource file.
54// The impulse responses may be resampled to a different sample-rate (depending on the audio hardware) when they are loaded.
55constexpr float ResponseSampleRate = 44100;

61#ifdef USE_CONCATENATED_IMPULSE_RESPONSES
62// Lazily load a concatenated HRTF database for given subject and store it in a
63// local hash table to ensure quick efficient future retrievals.
64static AudioBus* getConcatenatedImpulseResponsesForSubject(const String& subjectName)
65{
66 typedef HashMap<String, AudioBus*> AudioBusMap;
67 static NeverDestroyed<AudioBusMap> audioBusMap;
68
69 AudioBus* bus;
70 AudioBusMap::iterator iterator = audioBusMap.get().find(subjectName);
71 if (iterator == audioBusMap.get().end()) {
72 auto concatenatedImpulseResponses = AudioBus::loadPlatformResource(subjectName.utf8().data(), ResponseSampleRate);
73 ASSERT(concatenatedImpulseResponses);
74 if (!concatenatedImpulseResponses)
75 return 0;
76
77 bus = concatenatedImpulseResponses.leakRef();
78 audioBusMap.get().set(subjectName, bus);
79 } else
80 bus = iterator->value;
81
82 size_t responseLength = bus->length();
83 size_t expectedLength = static_cast<size_t>(TotalNumberOfResponses * ResponseFrameSize);
84
85 // Check number of channels and length. For now these are fixed and known.
86 bool isBusGood = responseLength == expectedLength && bus->numberOfChannels() == 2;
87 ASSERT(isBusGood);
88 if (!isBusGood)
89 return 0;
90
91 return bus;
92}
93#endif

95// Takes advantage of the symmetry and creates a composite version of the two measured versions. For example, we have both azimuth 30 and -30 degrees
96// where the roles of left and right ears are reversed with respect to each other.
97bool HRTFElevation::calculateSymmetricKernelsForAzimuthElevation(int azimuth, int elevation, float sampleRate, const String& subjectName,
98 RefPtr<HRTFKernel>& kernelL, RefPtr<HRTFKernel>& kernelR)
99{
100 RefPtr<HRTFKernel> kernelL1;
101 RefPtr<HRTFKernel> kernelR1;
102 bool success = calculateKernelsForAzimuthElevation(azimuth, elevation, sampleRate, subjectName, kernelL1, kernelR1);
103 if (!success)
104 return false;
105
106 // And symmetric version
107 int symmetricAzimuth = !azimuth ? 0 : 360 - azimuth;
108
109 RefPtr<HRTFKernel> kernelL2;
110 RefPtr<HRTFKernel> kernelR2;
111 success = calculateKernelsForAzimuthElevation(symmetricAzimuth, elevation, sampleRate, subjectName, kernelL2, kernelR2);
112 if (!success)
113 return false;
114
115 // Notice L/R reversal in symmetric version.
116 kernelL = HRTFKernel::createInterpolatedKernel(kernelL1.get(), kernelR2.get(), 0.5f);
117 kernelR = HRTFKernel::createInterpolatedKernel(kernelR1.get(), kernelL2.get(), 0.5f);
118
119 return true;
120}

122bool HRTFElevation::calculateKernelsForAzimuthElevation(int azimuth, int elevation, float sampleRate, const String& subjectName,
123 RefPtr<HRTFKernel>& kernelL, RefPtr<HRTFKernel>& kernelR)
124{
125 // Valid values for azimuth are 0 -> 345 in 15 degree increments.
126 // Valid values for elevation are -45 -> +90 in 15 degree increments.
127
128 bool isAzimuthGood = azimuth >= 0 && azimuth <= 345 && (azimuth / 15) * 15 == azimuth;
129 ASSERT(isAzimuthGood);
130 if (!isAzimuthGood)
131 return false;
132
133 bool isElevationGood = elevation >= -45 && elevation <= 90 && (elevation / 15) * 15 == elevation;
134 ASSERT(isElevationGood);
135 if (!isElevationGood)
136 return false;
137
138 // Construct the resource name from the subject name, azimuth, and elevation, for example:
139 // "IRC_Composite_C_R0195_T015_P000"
140 // Note: the passed in subjectName is not a string passed in via JavaScript or the web.
141 // It's passed in as an internal ASCII identifier and is an implementation detail.
142 int positiveElevation = elevation < 0 ? elevation + 360 : elevation;
143
144#ifdef USE_CONCATENATED_IMPULSE_RESPONSES
145 AudioBus* bus(getConcatenatedImpulseResponsesForSubject(subjectName));
146
147 if (!bus)
148 return false;
149
150 int elevationIndex = positiveElevation / AzimuthSpacing;
151 if (positiveElevation > 90)
152 elevationIndex -= AzimuthSpacing;
153
154 // The concatenated impulse response is a bus containing all
155 // the elevations per azimuth, for all azimuths by increasing
156 // order. So for a given azimuth and elevation we need to compute
157 // the index of the wanted audio frames in the concatenated table.
158 unsigned index = ((azimuth / AzimuthSpacing) * HRTFDatabase::NumberOfRawElevations) + elevationIndex;
159 bool isIndexGood = index < TotalNumberOfResponses;
160 ASSERT(isIndexGood);
161 if (!isIndexGood)
162 return false;
163
164 // Extract the individual impulse response from the concatenated
165 // responses and potentially sample-rate convert it to the desired
166 // (hardware) sample-rate.
167 unsigned startFrame = index * ResponseFrameSize;
168 unsigned stopFrame = startFrame + ResponseFrameSize;
169 auto preSampleRateConvertedResponse = AudioBus::createBufferFromRange(bus, startFrame, stopFrame);
170 auto response = AudioBus::createBySampleRateConverting(preSampleRateConvertedResponse.get(), false, sampleRate);
171 AudioChannel* leftEarImpulseResponse = response->channel(AudioBus::ChannelLeft);
172 AudioChannel* rightEarImpulseResponse = response->channel(AudioBus::ChannelRight);
173#else
174 auto resourceName = makeString("IRC_", subjectName, "_C_R0195_T", pad('0', 3, azimuth), "_P", pad('0', 3, positiveElevation)).utf8();
175
176 RefPtr<AudioBus> impulseResponse(AudioBus::loadPlatformResource(resourceName.data(), sampleRate));
177
178 ASSERT(impulseResponse.get());
179 if (!impulseResponse.get())
180 return false;
181
182 size_t responseLength = impulseResponse->length();
183 size_t expectedLength = static_cast<size_t>(256 * (sampleRate / 44100.0));
184
185 // Check number of channels and length. For now these are fixed and known.
186 bool isBusGood = responseLength == expectedLength && impulseResponse->numberOfChannels() == 2;
187 ASSERT(isBusGood);
188 if (!isBusGood)
189 return false;
190
191 AudioChannel* leftEarImpulseResponse = impulseResponse->channelByType(AudioBus::ChannelLeft);
192 AudioChannel* rightEarImpulseResponse = impulseResponse->channelByType(AudioBus::ChannelRight);
193#endif
194
195 // Note that depending on the fftSize returned by the panner, we may be truncating the impulse response we just loaded in.
196 const size_t fftSize = HRTFPanner::fftSizeForSampleRate(sampleRate);
197 kernelL = HRTFKernel::create(leftEarImpulseResponse, fftSize, sampleRate);
198 kernelR = HRTFKernel::create(rightEarImpulseResponse, fftSize, sampleRate);
199
200 return true;
201}

203// The range of elevations for the IRCAM impulse responses varies depending on azimuth, but the minimum elevation appears to always be -45.
204//
205// Here's how it goes:
206static const int maxElevations[] = {

208 //
209 90, // 0
210 45, // 15
211 60, // 30
212 45, // 45
213 75, // 60
214 45, // 75
215 60, // 90

233};

235std::unique_ptr<HRTFElevation> HRTFElevation::createForSubject(const String& subjectName, int elevation, float sampleRate)
236{
237 bool isElevationGood = elevation >= -45 && elevation <= 90 && (elevation / 15) * 15 == elevation;
238 ASSERT(isElevationGood);
239 if (!isElevationGood)
240 return nullptr;
241
242 auto kernelListL = makeUnique<HRTFKernelList>(NumberOfTotalAzimuths);
243 auto kernelListR = makeUnique<HRTFKernelList>(NumberOfTotalAzimuths);
244
245 // Load convolution kernels from HRTF files.
246 int interpolatedIndex = 0;
247 for (unsigned rawIndex = 0; rawIndex < NumberOfRawAzimuths; ++rawIndex) {
248 // Don't let elevation exceed maximum for this azimuth.
249 int maxElevation = maxElevations[rawIndex];
250 int actualElevation = std::min(elevation, maxElevation);
251
252 bool success = calculateKernelsForAzimuthElevation(rawIndex * AzimuthSpacing, actualElevation, sampleRate, subjectName, kernelListL->at(interpolatedIndex), kernelListR->at(interpolatedIndex));
253 if (!success)
254 return nullptr;
255
256 interpolatedIndex += InterpolationFactor;
257 }
258
259 // Now go back and interpolate intermediate azimuth values.
260 for (unsigned i = 0; i < NumberOfTotalAzimuths; i += InterpolationFactor) {
261 int j = (i + InterpolationFactor) % NumberOfTotalAzimuths;
262
263 // Create the interpolated convolution kernels and delays.
264 for (unsigned jj = 1; jj < InterpolationFactor; ++jj) {
265 float x = float(jj) / float(InterpolationFactor); // interpolate from 0 -> 1
266
267 (*kernelListL)[i + jj] = HRTFKernel::createInterpolatedKernel(kernelListL->at(i).get(), kernelListL->at(j).get(), x);
268 (*kernelListR)[i + jj] = HRTFKernel::createInterpolatedKernel(kernelListR->at(i).get(), kernelListR->at(j).get(), x);
269 }
270 }
271
272 return makeUnique<HRTFElevation>(WTFMove(kernelListL), WTFMove(kernelListR), elevation, sampleRate);
273}

275std::unique_ptr<HRTFElevation> HRTFElevation::createByInterpolatingSlices(HRTFElevation* hrtfElevation1, HRTFElevation* hrtfElevation2, float x, float sampleRate)
276{
277 ASSERT(hrtfElevation1 && hrtfElevation2);
278 if (!hrtfElevation1 || !hrtfElevation2)
279 return nullptr;
280
281 ASSERT(x >= 0.0 && x < 1.0);
282
283 auto kernelListL = makeUnique<HRTFKernelList>(NumberOfTotalAzimuths);
284 auto kernelListR = makeUnique<HRTFKernelList>(NumberOfTotalAzimuths);
285
286 HRTFKernelList* kernelListL1 = hrtfElevation1->kernelListL();
287 HRTFKernelList* kernelListR1 = hrtfElevation1->kernelListR();
288 HRTFKernelList* kernelListL2 = hrtfElevation2->kernelListL();
289 HRTFKernelList* kernelListR2 = hrtfElevation2->kernelListR();
290
291 // Interpolate kernels of corresponding azimuths of the two elevations.
292 for (unsigned i = 0; i < NumberOfTotalAzimuths; ++i) {
293 (*kernelListL)[i] = HRTFKernel::createInterpolatedKernel(kernelListL1->at(i).get(), kernelListL2->at(i).get(), x);
294 (*kernelListR)[i] = HRTFKernel::createInterpolatedKernel(kernelListR1->at(i).get(), kernelListR2->at(i).get(), x);
295 }
296
297 // Interpolate elevation angle.
298 double angle = (1.0 - x) * hrtfElevation1->elevationAngle() + x * hrtfElevation2->elevationAngle();
299
300 return makeUnique<HRTFElevation>(WTFMove(kernelListL), WTFMove(kernelListR), static_cast<int>(angle), sampleRate);
301}

303void HRTFElevation::getKernelsFromAzimuth(double azimuthBlend, unsigned azimuthIndex, HRTFKernel* &kernelL, HRTFKernel* &kernelR, double& frameDelayL, double& frameDelayR)
304{
305 bool checkAzimuthBlend = azimuthBlend >= 0.0 && azimuthBlend < 1.0;
306 ASSERT(checkAzimuthBlend);
307 if (!checkAzimuthBlend)
308 azimuthBlend = 0.0;
309
310 unsigned numKernels = m_kernelListL->size();
311
312 bool isIndexGood = azimuthIndex < numKernels;
313 ASSERT(isIndexGood);
314 if (!isIndexGood) {
315 kernelL = 0;
316 kernelR = 0;
317 return;
318 }
319
320 // Return the left and right kernels.
321 kernelL = m_kernelListL->at(azimuthIndex).get();
322 kernelR = m_kernelListR->at(azimuthIndex).get();
323
324 frameDelayL = m_kernelListL->at(azimuthIndex)->frameDelay();
325 frameDelayR = m_kernelListR->at(azimuthIndex)->frameDelay();
326
327 int azimuthIndex2 = (azimuthIndex + 1) % numKernels;
328 double frameDelay2L = m_kernelListL->at(azimuthIndex2)->frameDelay();
329 double frameDelay2R = m_kernelListR->at(azimuthIndex2)->frameDelay();
330
331 // Linearly interpolate delays.
332 frameDelayL = (1.0 - azimuthBlend) * frameDelayL + azimuthBlend * frameDelay2L;
333 frameDelayR = (1.0 - azimuthBlend) * frameDelayR + azimuthBlend * frameDelay2R;
334}
