43// Takes the input AudioChannel as an input impulse response and calculates the average group delay.
44// This represents the initial delay before the most energetic part of the impulse response.
45// The sample-frame delay is removed from the impulseP impulse response, and this value is returned.
46// the length of the passed in AudioChannel must be a power of 2.
47static float extractAverageGroupDelay(AudioChannel* channel, size_t analysisFFTSize)
48{
49 ASSERT(channel);
50
51 float* impulseP = channel->mutableData();
52
53 bool isSizeGood = channel->length() >= analysisFFTSize;
54 ASSERT(isSizeGood);
55 if (!isSizeGood)
56 return 0;
57
58 // Check for power-of-2.
59 ASSERT(1UL << static_cast<unsigned>(log2(analysisFFTSize)) == analysisFFTSize);
60
61 FFTFrame estimationFrame(analysisFFTSize);
62 estimationFrame.doFFT(impulseP);
63
64 float frameDelay = narrowPrecisionToFloat(estimationFrame.extractAverageGroupDelay());
65 estimationFrame.doInverseFFT(impulseP);
66
67 return frameDelay;
68}

70HRTFKernel::HRTFKernel(AudioChannel* channel, size_t fftSize, float sampleRate)
71 : m_sampleRate(sampleRate)
72{
73 ASSERT(channel);
74
75 // Determine the leading delay (average group delay) for the response.
76 m_frameDelay = extractAverageGroupDelay(channel, fftSize / 2);
77
78 float* impulseResponse = channel->mutableData();
79 size_t responseLength = channel->length();
80
81 // We need to truncate to fit into 1/2 the FFT size (with zero padding) in order to do proper convolution.
82 size_t truncatedResponseLength = std::min(responseLength, fftSize / 2); // truncate if necessary to max impulse response length allowed by FFT
83
84 // Quick fade-out (apply window) at truncation point
85 unsigned numberOfFadeOutFrames = static_cast<unsigned>(sampleRate / 4410); // 10 sample-frames @44.1KHz sample-rate
86 ASSERT(numberOfFadeOutFrames < truncatedResponseLength);
87 if (numberOfFadeOutFrames < truncatedResponseLength) {
88 for (unsigned i = truncatedResponseLength - numberOfFadeOutFrames; i < truncatedResponseLength; ++i) {
89 float x = 1.0f - static_cast<float>(i - (truncatedResponseLength - numberOfFadeOutFrames)) / numberOfFadeOutFrames;
90 impulseResponse[i] *= x;
91 }
92 }
93
94 m_fftFrame = makeUnique<FFTFrame>(fftSize);
95 m_fftFrame->doPaddedFFT(impulseResponse, truncatedResponseLength);
96}
97
98size_t HRTFKernel::fftSize() const
99{
100 return m_fftFrame->fftSize();
101}

103std::unique_ptr<AudioChannel> HRTFKernel::createImpulseResponse()
104{
105 auto channel = makeUnique<AudioChannel>(fftSize());
106 FFTFrame fftFrame(*m_fftFrame);
107
108 // Add leading delay back in.
109 fftFrame.addConstantGroupDelay(m_frameDelay);
110 fftFrame.doInverseFFT(channel->mutableData());
111
112 return channel;
113}

115// Interpolates two kernels with x: 0 -> 1 and returns the result.
116RefPtr<HRTFKernel> HRTFKernel::createInterpolatedKernel(HRTFKernel* kernel1, HRTFKernel* kernel2, float x)
117{
118 ASSERT(kernel1 && kernel2);
119 if (!kernel1 || !kernel2)
120 return nullptr;
121
122 ASSERT(x >= 0.0 && x < 1.0);
123 x = std::min(1.0f, std::max(0.0f, x));
124
125 float sampleRate1 = kernel1->sampleRate();
126 float sampleRate2 = kernel2->sampleRate();
127 ASSERT(sampleRate1 == sampleRate2);
128 if (sampleRate1 != sampleRate2)
129 return nullptr;
130
131 float frameDelay = (1 - x) * kernel1->frameDelay() + x * kernel2->frameDelay();
132
133 std::unique_ptr<FFTFrame> interpolatedFrame = FFTFrame::createInterpolatedFrame(*kernel1->fftFrame(), *kernel2->fftFrame(), x);
134 return HRTFKernel::create(WTFMove(interpolatedFrame), frameDelay, sampleRate1);
135}
