41// The value of 2 milliseconds is larger than the largest delay which exists in any HRTFKernel from the default HRTFDatabase (0.0136 seconds).
42// We ASSERT the delay values used in process() with this value.
43constexpr double MaxDelayTimeSeconds = 0.002;

45HRTFPanner::HRTFPanner(float sampleRate, HRTFDatabaseLoader* databaseLoader)
46 : Panner(PanningModelType::HRTF)
47 , m_databaseLoader(databaseLoader)
48 , m_sampleRate(sampleRate)
49 , m_convolverL1(fftSizeForSampleRate(sampleRate))
50 , m_convolverR1(fftSizeForSampleRate(sampleRate))
51 , m_convolverL2(fftSizeForSampleRate(sampleRate))

60 ASSERT(databaseLoader);

65size_t HRTFPanner::fftSizeForSampleRate(float sampleRate)
66{
67 // The HRTF impulse responses (loaded as audio resources) are 512
68 // sample-frames @44.1KHz. Currently, we truncate the impulse responses to
69 // half this size, but an FFT-size of twice impulse response size is needed
70 // (for convolution). So for sample rates around 44.1KHz an FFT size of 512
71 // is good. For different sample rates, the truncated response is resampled.
72 // The resampled length is used to compute the FFT size by choosing a power
73 // of two that is greater than or equal the resampled length. This power of
74 // two is doubled to get the actual FFT size.
75
76 const int truncatedImpulseLength = 256;
77 double sampleRateRatio = sampleRate / 44100;
78 double resampledLength = truncatedImpulseLength * sampleRateRatio;
79
80 // This is the size used for analysis frames in the HRTF kernel. The
81 // convolvers used by the kernel are twice this size.
82 int analysisFFTSize = 1 << static_cast<unsigned>(log2(resampledLength));
83
84 // Don't let the analysis size be smaller than the supported size
85 analysisFFTSize = std::max(analysisFFTSize, FFTFrame::minFFTSize());
86
87 int convolverFFTSize = 2 * analysisFFTSize;
88
89 // Make sure this size of convolver is supported.
90 ASSERT(convolverFFTSize <= FFTFrame::maxFFTSize());
91
92 return convolverFFTSize;
93
94}

106int HRTFPanner::calculateDesiredAzimuthIndexAndBlend(double azimuth, double& azimuthBlend)
107{
108 // Convert the azimuth angle from the range -180 -> +180 into the range 0 -> 360.
109 // The azimuth index may then be calculated from this positive value.
110 if (azimuth < 0)
111 azimuth += 360.0;
112
113 HRTFDatabase* database = m_databaseLoader->database();
114 ASSERT(database);
115
116 int numberOfAzimuths = database->numberOfAzimuths();
117 const double angleBetweenAzimuths = 360.0 / numberOfAzimuths;
118
119 // Calculate the azimuth index and the blend (0 -> 1) for interpolation.
120 double desiredAzimuthIndexFloat = azimuth / angleBetweenAzimuths;
121 int desiredAzimuthIndex = static_cast<int>(desiredAzimuthIndexFloat);
122 azimuthBlend = desiredAzimuthIndexFloat - static_cast<double>(desiredAzimuthIndex);
123
124 // We don't immediately start using this azimuth index, but instead approach this index from the last index we rendered at.
125 // This minimizes the clicks and graininess for moving sources which occur otherwise.
126 desiredAzimuthIndex = std::max(0, desiredAzimuthIndex);
127 desiredAzimuthIndex = std::min(numberOfAzimuths - 1, desiredAzimuthIndex);
128 return desiredAzimuthIndex;
129}

131void HRTFPanner::pan(double desiredAzimuth, double elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess)
132{
133 unsigned numInputChannels = inputBus ? inputBus->numberOfChannels() : 0;
134
135 bool isInputGood = inputBus && numInputChannels >= 1 && numInputChannels <= 2;
136 ASSERT(isInputGood);
137
138 bool isOutputGood = outputBus && outputBus->numberOfChannels() == 2 && framesToProcess <= outputBus->length();
139 ASSERT(isOutputGood);
140
141 if (!isInputGood || !isOutputGood) {
142 if (outputBus)
143 outputBus->zero();
144 return;
145 }
146
147 // This code only runs as long as the context is alive and after database has been loaded.
148 HRTFDatabase* database = m_databaseLoader->database();
149 ASSERT(database);
150 if (!database) {
151 outputBus->zero();
152 return;
153 }
154
155 // IRCAM HRTF azimuths values from the loaded database is reversed from the panner's notion of azimuth.
156 double azimuth = -desiredAzimuth;
157
158 bool isAzimuthGood = azimuth >= -180.0 && azimuth <= 180.0;
159 ASSERT(isAzimuthGood);
160 if (!isAzimuthGood) {
161 outputBus->zero();
162 return;
163 }
164
165 // Normally, we'll just be dealing with mono sources.
166 // If we have a stereo input, implement stereo panning with left source processed by left HRTF, and right source by right HRTF.
167 const AudioChannel* inputChannelL = inputBus->channelByType(AudioBus::ChannelLeft);
168 const AudioChannel* inputChannelR = numInputChannels > 1 ? inputBus->channelByType(AudioBus::ChannelRight) : 0;
169
170 // Get source and destination pointers.
171 const float* sourceL = inputChannelL->data();
172 const float* sourceR = numInputChannels > 1 ? inputChannelR->data() : sourceL;
173 float* destinationL = outputBus->channelByType(AudioBus::ChannelLeft)->mutableData();
174 float* destinationR = outputBus->channelByType(AudioBus::ChannelRight)->mutableData();
175
176 double azimuthBlend;
177 int desiredAzimuthIndex = calculateDesiredAzimuthIndexAndBlend(azimuth, azimuthBlend);
178
179 // Initially snap azimuth and elevation values to first values encountered.
180 if (!m_azimuthIndex1) {
181 m_azimuthIndex1 = desiredAzimuthIndex;
182 m_elevation1 = elevation;
183 }
184 if (!m_azimuthIndex2) {
185 m_azimuthIndex2 = desiredAzimuthIndex;
186 m_elevation2 = elevation;
187 }
188
189 // Cross-fade / transition over a period of around 45 milliseconds.
190 // This is an empirical value tuned to be a reasonable trade-off between
191 // smoothness and speed.
192 const double fadeFrames = sampleRate() <= 48000 ? 2048 : 4096;
193
194 // Check for azimuth and elevation changes, initiating a cross-fade if needed.
195 if (!m_crossfadeX && m_crossfadeSelection == CrossfadeSelection1) {
196 if (desiredAzimuthIndex != *m_azimuthIndex1 || elevation != m_elevation1) {
197 // Cross-fade from 1 -> 2
198 m_crossfadeIncr = 1 / fadeFrames;
199 m_azimuthIndex2 = desiredAzimuthIndex;
200 m_elevation2 = elevation;
201 }
202 }
203 if (m_crossfadeX == 1 && m_crossfadeSelection == CrossfadeSelection2) {
204 if (desiredAzimuthIndex != *m_azimuthIndex2 || elevation != m_elevation2) {
205 // Cross-fade from 2 -> 1
206 m_crossfadeIncr = -1 / fadeFrames;
207 m_azimuthIndex1 = desiredAzimuthIndex;
208 m_elevation1 = elevation;
209 }
210 }
211
212 // This algorithm currently requires that we process in power-of-two size chunks at least AudioUtilities::renderQuantumSize.
213 ASSERT(1UL << static_cast<int>(log2(framesToProcess)) == framesToProcess);
214 ASSERT(framesToProcess >= AudioUtilities::renderQuantumSize);
215
216 const unsigned framesPerSegment = AudioUtilities::renderQuantumSize;
217 const unsigned numberOfSegments = framesToProcess / framesPerSegment;
218
219 for (unsigned segment = 0; segment < numberOfSegments; ++segment) {
220 // Get the HRTFKernels and interpolated delays.
221 HRTFKernel* kernelL1;
222 HRTFKernel* kernelR1;
223 HRTFKernel* kernelL2;
224 HRTFKernel* kernelR2;
225 double frameDelayL1;
226 double frameDelayR1;
227 double frameDelayL2;
228 double frameDelayR2;
229 database->getKernelsFromAzimuthElevation(azimuthBlend, *m_azimuthIndex1, m_elevation1, kernelL1, kernelR1, frameDelayL1, frameDelayR1);
230 database->getKernelsFromAzimuthElevation(azimuthBlend, *m_azimuthIndex2, m_elevation2, kernelL2, kernelR2, frameDelayL2, frameDelayR2);
231
232 bool areKernelsGood = kernelL1 && kernelR1 && kernelL2 && kernelR2;
233 ASSERT(areKernelsGood);
234 if (!areKernelsGood) {
235 outputBus->zero();
236 return;
237 }
238
239 ASSERT(frameDelayL1 / sampleRate() < MaxDelayTimeSeconds && frameDelayR1 / sampleRate() < MaxDelayTimeSeconds);
240 ASSERT(frameDelayL2 / sampleRate() < MaxDelayTimeSeconds && frameDelayR2 / sampleRate() < MaxDelayTimeSeconds);
241
242 // Crossfade inter-aural delays based on transitions.
243 double frameDelayL = (1 - m_crossfadeX) * frameDelayL1 + m_crossfadeX * frameDelayL2;
244 double frameDelayR = (1 - m_crossfadeX) * frameDelayR1 + m_crossfadeX * frameDelayR2;
245
246 // Calculate the source and destination pointers for the current segment.
247 unsigned offset = segment * framesPerSegment;
248 const float* segmentSourceL = sourceL + offset;
249 const float* segmentSourceR = sourceR + offset;
250 float* segmentDestinationL = destinationL + offset;
251 float* segmentDestinationR = destinationR + offset;
252
253 // First run through delay lines for inter-aural time difference.
254 m_delayLineL.setDelayFrames(frameDelayL);
255 m_delayLineR.setDelayFrames(frameDelayR);
256 m_delayLineL.process(segmentSourceL, segmentDestinationL, framesPerSegment);
257 m_delayLineR.process(segmentSourceR, segmentDestinationR, framesPerSegment);
258
259 bool needsCrossfading = m_crossfadeIncr;
260
261 // Have the convolvers render directly to the final destination if we're not cross-fading.
262 float* convolutionDestinationL1 = needsCrossfading ? m_tempL1.data() : segmentDestinationL;
263 float* convolutionDestinationR1 = needsCrossfading ? m_tempR1.data() : segmentDestinationR;
264 float* convolutionDestinationL2 = needsCrossfading ? m_tempL2.data() : segmentDestinationL;
265 float* convolutionDestinationR2 = needsCrossfading ? m_tempR2.data() : segmentDestinationR;
266
267 // Now do the convolutions.
268 // Note that we avoid doing convolutions on both sets of convolvers if we're not currently cross-fading.
269
270 if (m_crossfadeSelection == CrossfadeSelection1 || needsCrossfading) {
271 m_convolverL1.process(kernelL1->fftFrame(), segmentDestinationL, convolutionDestinationL1, framesPerSegment);
272 m_convolverR1.process(kernelR1->fftFrame(), segmentDestinationR, convolutionDestinationR1, framesPerSegment);
273 }
274
275 if (m_crossfadeSelection == CrossfadeSelection2 || needsCrossfading) {
276 m_convolverL2.process(kernelL2->fftFrame(), segmentDestinationL, convolutionDestinationL2, framesPerSegment);
277 m_convolverR2.process(kernelR2->fftFrame(), segmentDestinationR, convolutionDestinationR2, framesPerSegment);
278 }
279
280 if (needsCrossfading) {
281 // Apply linear cross-fade.
282 float x = m_crossfadeX;
283 float incr = m_crossfadeIncr;
284 for (unsigned i = 0; i < framesPerSegment; ++i) {
285 segmentDestinationL[i] = (1 - x) * convolutionDestinationL1[i] + x * convolutionDestinationL2[i];
286 segmentDestinationR[i] = (1 - x) * convolutionDestinationR1[i] + x * convolutionDestinationR2[i];
287 x += incr;
288 }
289 // Update cross-fade value from local.
290 m_crossfadeX = x;
291
292 if (m_crossfadeIncr > 0 && fabs(m_crossfadeX - 1) < m_crossfadeIncr) {
293 // We've fully made the crossfade transition from 1 -> 2.
294 m_crossfadeSelection = CrossfadeSelection2;
295 m_crossfadeX = 1;
296 m_crossfadeIncr = 0;
297 } else if (m_crossfadeIncr < 0 && fabs(m_crossfadeX) < -m_crossfadeIncr) {
298 // We've fully made the crossfade transition from 2 -> 1.
299 m_crossfadeSelection = CrossfadeSelection1;
300 m_crossfadeX = 0;
301 m_crossfadeIncr = 0;
302 }
303 }
304 }
305}

307void HRTFPanner::panWithSampleAccurateValues(double* azimuth, double* elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess)
308{
309 // Sample-accurate (a-rate) HRTF panner is not implemented, just k-rate. Just
310 // grab the current azimuth/elevation and use that.
311 //
312 // We are assuming that the inherent smoothing in the HRTF processing is good
313 // enough, and we don't want to increase the complexity of the HRTF panner by
314 // 15-20 times. (We need to compute one output sample for each possibly
315 // different impulse response. That N^2. Previously, we used an FFT to do
316 // them all at once for a complexity of N/log2(N). Hence, N/log2(N) times
317 // more complex.)
318 pan(azimuth[0], elevation[0], inputBus, outputBus, framesToProcess);
319}

321double HRTFPanner::tailTime() const
322{
323 // Because HRTFPanner is implemented with a DelayKernel and a FFTConvolver, the tailTime of the HRTFPanner
324 // is the sum of the tailTime of the DelayKernel and the tailTime of the FFTConvolver, which is MaxDelayTimeSeconds
325 // and fftSize() / 2, respectively.
326 return MaxDelayTimeSeconds + (fftSize() / 2) / static_cast<double>(sampleRate());
327}
328
329double HRTFPanner::latencyTime() const
330{
331 // The latency of a FFTConvolver is also fftSize() / 2, and is in addition to its tailTime of the
332 // same value.
333 return (fftSize() / 2) / static_cast<double>(sampleRate());
334}
335
336bool HRTFPanner::requiresTailProcessing() const
337{
338 // Always return true since the tail and latency are never zero.
339 return true;
340}
