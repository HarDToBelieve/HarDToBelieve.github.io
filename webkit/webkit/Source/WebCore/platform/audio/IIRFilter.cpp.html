40// The length of the memory buffers for the IIR filter. This MUST be a power of
41// two and must be greater than the possible length of the filter coefficients.
42constexpr int bufferLength = 32;
43static_assert(bufferLength >= IIRFilter::maxOrder + 1, "Internal IIR buffer length must be greater than maximum IIR Filter order.");
44
45static std::complex<double> evaluatePolynomial(const double* coefficients, std::complex<double> z, int order)
46{
47 // Use Horner's method to evaluate the polynomial P(z) = sum(coef[k]*z^k, k, 0, order);
48 std::complex<double> result = 0;
49 for (int k = order; k >= 0; --k)
50 result = result * z + std::complex<double>(coefficients[k]);
51 return result;
52}

54IIRFilter::IIRFilter(const Vector<double>& feedforward, const Vector<double>& feedback)
55 : m_feedforward(feedforward)
56 , m_feedback(feedback)
57{
58 m_xBuffer.fill(0, bufferLength);
59 m_yBuffer.fill(0, bufferLength);
60}

62void IIRFilter::reset()
63{
64 m_xBuffer.fill(0);
65 m_yBuffer.fill(0);
66 m_bufferIndex = 0;
67}

69void IIRFilter::process(const float* source, float* destination, size_t framesToProcess)
70{
71 // Compute
72 //
73 // y[n] = sum(b[k] * x[n - k], k = 0, M) - sum(a[k] * y[n - k], k = 1, N)
74 //
75 // where b[k] are the feedforward coefficients and a[k] are the feedback
76 // coefficients of the filter.
77 // This is a Direct Form I implementation of an IIR Filter. Should we
78 // consider doing a different implementation such as Transposed Direct Form
79 // II?
80 const double* feedback = m_feedback.data();
81 const double* feedforward = m_feedforward.data();
82
83 ASSERT(feedback);
84 ASSERT(feedforward);
85
86 // Sanity check to see if the feedback coefficients have been scaled appropriately. It must be EXACTLY 1!
87 ASSERT(feedback[0] == 1);
88
89 int feedbackLength = m_feedback.size();
90 int feedforwardLength = m_feedforward.size();
91 int minLength = std::min(feedbackLength, feedforwardLength);
92
93 double* xBuffer = m_xBuffer.data();
94 double* yBuffer = m_yBuffer.data();
95
96 for (size_t n = 0; n < framesToProcess; ++n) {
97 // To help minimize roundoff, we compute using double's, even though the
98 // filter coefficients only have single precision values.
99 double yn = feedforward[0] * source[n];
100
101 // Run both the feedforward and feedback terms together, when possible.
102 for (int k = 1; k < minLength; ++k) {
103 int m = (m_bufferIndex - k) & (bufferLength - 1);
104 yn += feedforward[k] * xBuffer[m];
105 yn -= feedback[k] * yBuffer[m];
106 }
107
108 // Handle any remaining feedforward or feedback terms.
109 for (int k = minLength; k < feedforwardLength; ++k)
110 yn += feedforward[k] * xBuffer[(m_bufferIndex - k) & (bufferLength - 1)];
111 for (int k = minLength; k < feedbackLength; ++k)
112 yn -= feedback[k] * yBuffer[(m_bufferIndex - k) & (bufferLength - 1)];
113
114 // Save the current input and output values in the memory buffers for the
115 // next output.
116 m_xBuffer[m_bufferIndex] = source[n];
117 m_yBuffer[m_bufferIndex] = yn;
118
119 m_bufferIndex = (m_bufferIndex + 1) & (bufferLength - 1);
120
121 destination[n] = yn;
122 }
123}

125void IIRFilter::getFrequencyResponse(unsigned length, const float* frequency, float* magResponse, float* phaseResponse)
126{
127 // Evaluate the z-transform of the filter at the given normalized frequencies
128 // from 0 to 1. (One corresponds to the Nyquist frequency.)
129 //
130 // The z-tranform of the filter is
131 //
132 // H(z) = sum(b[k]*z^(-k), k, 0, M) / sum(a[k]*z^(-k), k, 0, N);
133 //
134 // The desired frequency response is H(exp(j*omega)), where omega is in [0,
135 // 1).
136 //
137 // Let P(x) = sum(c[k]*x^k, k, 0, P) be a polynomial of order P. Then each of
138 // the sums in H(z) is equivalent to evaluating a polynomial at the point
139 // 1/z.
140
141 for (unsigned k = 0; k < length; ++k) {
142 if (frequency[k] < 0 || frequency[k] > 1) {
143 // Out-of-bounds frequencies should return NaN.
144 magResponse[k] = std::nanf("");
145 phaseResponse[k] = std::nanf("");
146 } else {
147 // zRecip = 1/z = exp(-j*frequency)
148 double omega = -piDouble * frequency[k];
149 auto zRecip = std::complex<double>(cos(omega), sin(omega));
150
151 auto numerator = evaluatePolynomial(m_feedforward.data(), zRecip, m_feedforward.size() - 1);
152 auto denominator = evaluatePolynomial(m_feedback.data(), zRecip, m_feedback.size() - 1);
153 auto response = numerator / denominator;
154 magResponse[k] = static_cast<float>(abs(response));
155 phaseResponse[k] = static_cast<float>(atan2(imag(response), real(response)));
156 }
157 }
158}

160double IIRFilter::tailTime(double sampleRate, bool isFilterStable)
161{
162 // The maximum tail time. This is somewhat arbitrary, but we're assuming that
163 // no one is going to expect the IIRFilter to produce an output after this
164 // much time after the inputs have stopped.
165 constexpr double maxTailTime = 10;
166
167 // If the maximum amplitude of the impulse response is less than this, we
168 // assume that we've reached the tail of the response. Currently, this means
169 // that the impulse is less than 1 bit of a 16-bit PCM value.
170 constexpr float maxTailAmplitude = 1 / 32768.0;
171
172 // If filter is not stable, just return max tail. Since the filter is not
173 // stable, the impulse response won't converge to zero, so we don't need to
174 // find the impulse response to find the actual tail time.
175 if (!isFilterStable || !sampleRate)
176 return maxTailTime;
177
178 // How to compute the tail time? We're going to filter an impulse
179 // for |maxTailTime| seconds, in blocks of kRenderQuantumFrames at
180 // a time. The maximum magnitude of this block is saved. After all
181 // of the samples have been computed, find the last block with a
182 // maximum magnitude greater than |kMaxTaileAmplitude|. That block
183 // index + 1 will be the tail time. We don't need to be
184 // super-accurate in computing the tail time since we process on
185 // blocks, block accuracy is good enough, and the value just needs
186 // to be larger than the "real" tail time, so we don't prematurely
187 // zero out the output of the node.
188
189 // Number of render quanta needed to reach the max tail time.
190 int numberOfBlocks = std::ceil(sampleRate * maxTailTime / AudioUtilities::renderQuantumSize);
191 RELEASE_ASSERT(numberOfBlocks);
192
193 // Input and output buffers for filtering.
194 AudioFloatArray input(AudioUtilities::renderQuantumSize);
195 AudioFloatArray output(AudioUtilities::renderQuantumSize);
196
197 // Array to hold the max magnitudes
198 AudioFloatArray magnitudes(numberOfBlocks);
199
200 // Create the impulse input signal.
201 input[0] = 1;
202
203 // Process the first block and get the max magnitude of the output.
204 process(input.data(), output.data(), AudioUtilities::renderQuantumSize);
205 magnitudes[0] = VectorMath::maximumMagnitude(output.data(), AudioUtilities::renderQuantumSize);
206
207 // Process the rest of the signal, getting the max magnitude of the
208 // output for each block.
209 input[0] = 0;
210
211 for (int k = 1; k < numberOfBlocks; ++k) {
212 process(input.data(), output.data(), AudioUtilities::renderQuantumSize);
213 magnitudes[k] = VectorMath::maximumMagnitude(output.data(), AudioUtilities::renderQuantumSize);
214 }
215
216 // Done computing the impulse response; reset the state so the actual node
217 // starts in the expected initial state.
218 reset();
219
220 // Find the last block with amplitude greater than the threshold.
221 int index = numberOfBlocks - 1;
222 for (int k = index; k >= 0; --k) {
223 if (magnitudes[k] > maxTailAmplitude) {
224 index = k;
225 break;
226 }
227 }
228
229 // The magnitude first become lower than the threshold at the next block.
230 // Compute the corresponding time value value; that's the tail time.
231 return (index + 1) * AudioUtilities::renderQuantumSize / sampleRate;
232}
