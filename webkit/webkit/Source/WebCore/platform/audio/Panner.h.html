29#ifndef Panner_h
30#define Panner_h
31
32#include <memory>
33
34namespace WebCore {
35
36class AudioBus;
37class HRTFDatabaseLoader;
38
39enum class PanningModelType {
40 Equalpower,
41 HRTF
42};
43
44// Abstract base class for panning a mono or stereo source.
45
46class Panner {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 static std::unique_ptr<Panner> create(PanningModelType, float sampleRate, HRTFDatabaseLoader*);
50
51 virtual ~Panner() { };
52
53 PanningModelType panningModel() const { return m_panningModel; }
54
55 virtual void pan(double azimuth, double elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess) = 0;
56 virtual void panWithSampleAccurateValues(double* azimuth, double* elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess) = 0;
57 virtual void reset() = 0;
58
59 virtual double tailTime() const = 0;
60 virtual double latencyTime() const = 0;
61 virtual bool requiresTailProcessing() const = 0;
62
63protected:
64 Panner(PanningModelType model)
65 : m_panningModel(model)
66 {
67 }
68
69 PanningModelType m_panningModel;
70};
71
72} // namespace WebCore
73
74#endif // Panner_h
