146void PlatformMediaSession::setActive(bool active)
147{
148 if (m_active == active)
149 return;

153 PlatformMediaSessionManager::sharedManager().addSession(*this);

155 PlatformMediaSessionManager::sharedManager().removeSession(*this);

158void PlatformMediaSession::setState(State state)
159{
160 if (state == m_state)
161 return;
162
163 ALWAYS_LOG(LOGIDENTIFIER, state);
164 m_state = state;
165 if (m_state == State::Playing)
166 m_hasPlayedSinceLastInterruption = true;
167 PlatformMediaSessionManager::sharedManager().sessionStateChanged(*this);
168}

170void PlatformMediaSession::beginInterruption(InterruptionType type)
171{
172 ALWAYS_LOG(LOGIDENTIFIER, "state = ", m_state, ", interruption type = ", type, ", interruption count = ", m_interruptionCount);
173
174 // When interruptions are overridden, m_interruptionType doesn't get set.
175 // Give nested interruptions a chance when the previous interruptions were overridden.
176 if (++m_interruptionCount > 1 && m_interruptionType != NoInterruption)
177 return;
178
179 if (client().shouldOverrideBackgroundPlaybackRestriction(type)) {
180 ALWAYS_LOG(LOGIDENTIFIER, "returning early because client says to override interruption");
181 return;
182 }
183
184 m_stateToRestore = state();
185 m_notifyingClient = true;
186 setState(Interrupted);
187 m_interruptionType = type;
188 client().suspendPlayback();
189 m_notifyingClient = false;
190}
191
192void PlatformMediaSession::endInterruption(EndInterruptionFlags flags)
193{
194 ALWAYS_LOG(LOGIDENTIFIER, "flags = ", (int)flags, ", stateToRestore = ", m_stateToRestore, ", interruption count = ", m_interruptionCount);
195
196 if (!m_interruptionCount) {
197 ALWAYS_LOG(LOGIDENTIFIER, "!! ignoring spurious interruption end !!");
198 return;
199 }
200
201 if (--m_interruptionCount)
202 return;
203
204 if (m_interruptionType == NoInterruption)
205 return;
206
207 State stateToRestore = m_stateToRestore;
208 m_stateToRestore = Idle;
209 m_interruptionType = NoInterruption;
210 setState(stateToRestore);
211
212 if (stateToRestore == Autoplaying)
213 client().resumeAutoplaying();
214
215 bool shouldResume = flags & MayResumePlaying && stateToRestore == Playing;
216 client().mayResumePlayback(shouldResume);
217}

219void PlatformMediaSession::clientWillBeginAutoplaying()
220{
221 if (m_notifyingClient)
222 return;
223
224 ALWAYS_LOG(LOGIDENTIFIER, "state = ", m_state);
225 if (state() == Interrupted) {
226 m_stateToRestore = Autoplaying;
227 ALWAYS_LOG(LOGIDENTIFIER, " setting stateToRestore to \"Autoplaying\"");
228 return;
229 }
230
231 setState(Autoplaying);
232}
233
234bool PlatformMediaSession::clientWillBeginPlayback()
235{
236 if (m_notifyingClient)
237 return true;
238
239 ALWAYS_LOG(LOGIDENTIFIER, "state = ", m_state);
240
241 if (!PlatformMediaSessionManager::sharedManager().sessionWillBeginPlayback(*this)) {
242 if (state() == Interrupted)
243 m_stateToRestore = Playing;
244 return false;
245 }
246
247 setState(Playing);
248 return true;
249}
250
251bool PlatformMediaSession::processClientWillPausePlayback(DelayCallingUpdateNowPlaying shouldDelayCallingUpdateNowPlaying)
252{
253 if (m_notifyingClient)
254 return true;
255
256 ALWAYS_LOG(LOGIDENTIFIER, "state = ", m_state);
257 if (state() == Interrupted) {
258 m_stateToRestore = Paused;
259 ALWAYS_LOG(LOGIDENTIFIER, " setting stateToRestore to \"Paused\"");
260 return false;
261 }
262
263 setState(Paused);
264 PlatformMediaSessionManager::sharedManager().sessionWillEndPlayback(*this, shouldDelayCallingUpdateNowPlaying);
265 return true;
266}

330void PlatformMediaSession::isPlayingToWirelessPlaybackTargetChanged(bool isWireless)
331{
332 if (isWireless == m_isPlayingToWirelessPlaybackTarget)
333 return;
334
335 m_isPlayingToWirelessPlaybackTarget = isWireless;
336
337 // Save and restore the interruption count so it doesn't get out of sync if beginInterruption is called because
338 // if we in the background.
339 int interruptionCount = m_interruptionCount;
340 PlatformMediaSessionManager::sharedManager().sessionIsPlayingToWirelessPlaybackTargetChanged(*this);
341 m_interruptionCount = interruptionCount;
342}

373static inline bool isPlayingAudio(PlatformMediaSession::MediaType mediaType)
374{
375#if ENABLE(VIDEO)
376 return mediaType == MediaElementSession::MediaType::VideoAudio || mediaType == MediaElementSession::MediaType::Audio;
377#else
378 UNUSED_PARAM(mediaType);
379 return false;
380#endif
381}

383bool PlatformMediaSession::canPlayConcurrently(const PlatformMediaSession& otherSession) const
384{
385 auto mediaType = this->mediaType();
386 auto otherMediaType = otherSession.mediaType();
387 if (otherMediaType != mediaType && (!isPlayingAudio(mediaType) || !isPlayingAudio(otherMediaType)))
388 return true;
389
390 auto groupID = client().mediaSessionGroupIdentifier();
391 auto otherGroupID = otherSession.client().mediaSessionGroupIdentifier();
392 if (!groupID || !otherGroupID || groupID != otherGroupID)
393 return false;
394
395 return m_client.hasMediaStreamSource() || otherSession.m_client.hasMediaStreamSource();
396}
