34PushPullFIFO::PushPullFIFO(unsigned numberOfChannels, size_t fifoLength)
35 : m_fifoLength(fifoLength)
36{
37 ASSERT(m_fifoLength <= maxFIFOLength);
38 m_fifoBus = AudioBus::create(numberOfChannels, m_fifoLength);
39}

41PushPullFIFO::~PushPullFIFO() = default;

43// Push the data from |inputBus| to FIFO. The size of push is determined by
44// the length of |inputBus|.
45void PushPullFIFO::push(const AudioBus* inputBus)
46{
47 ASSERT(inputBus);
48 ASSERT(inputBus->length() <= m_fifoLength);
49 ASSERT(m_indexWrite < m_fifoLength);
50
51 const size_t inputBusLength = inputBus->length();
52 const size_t remainder = m_fifoLength - m_indexWrite;
53
54 for (unsigned i = 0; i < m_fifoBus->numberOfChannels(); ++i) {
55 float* fifoBusChannel = m_fifoBus->channel(i)->mutableData();
56 const float* inputBusChannel = inputBus->channel(i)->data();
57 if (remainder >= inputBusLength) {
58 // The remainder is big enough for the input data.
59 memcpy(fifoBusChannel + m_indexWrite, inputBusChannel, inputBusLength * sizeof(*fifoBusChannel));
60 } else {
61 // The input data overflows the remainder size. Wrap around the index.
62 memcpy(fifoBusChannel + m_indexWrite, inputBusChannel, remainder * sizeof(*fifoBusChannel));
63 memcpy(fifoBusChannel, inputBusChannel + remainder, (inputBusLength - remainder) * sizeof(*fifoBusChannel));
64 }
65 }
66
67 // Update the write index; wrap it around if necessary.
68 m_indexWrite = (m_indexWrite + inputBusLength) % m_fifoLength;
69
70 // In case of overflow, move the |indexRead| to the updated |indexWrite| to
71 // avoid reading overwritten frames by the next pull.
72 if (inputBusLength > m_fifoLength - m_framesAvailable)
73 m_indexRead = m_indexWrite;
74
75 // Update the number of frames available in FIFO.
76 m_framesAvailable = std::min(m_framesAvailable + inputBusLength, m_fifoLength);
77 ASSERT(((m_indexRead + m_framesAvailable) % m_fifoLength) == m_indexWrite);
78}

80// Pull the data out of FIFO to |outputBus|. If remaining frame in the FIFO
81// is less than the frames to pull, provides remaining frame plus the silence.
82size_t PushPullFIFO::pull(AudioBus* outputBus, size_t framesRequested)
83{
84 ASSERT(outputBus);
85 ASSERT(framesRequested <= outputBus->length());
86 ASSERT(framesRequested <= m_fifoLength);
87 ASSERT(m_indexRead < m_fifoLength);
88
89 const size_t remainder = m_fifoLength - m_indexRead;
90 const size_t framesToFill = std::min(m_framesAvailable, framesRequested);
91
92 for (unsigned i = 0; i < m_fifoBus->numberOfChannels(); ++i) {
93 const float* fifoBusChannel = m_fifoBus->channel(i)->data();
94 float* outputBusChannel = outputBus->channel(i)->mutableData();
95
96 // Fill up the output bus with the available frames first.
97 if (remainder >= framesToFill) {
98 // The remainder is big enough for the frames to pull.
99 memcpy(outputBusChannel, fifoBusChannel + m_indexRead, framesToFill * sizeof(*fifoBusChannel));
100 } else {
101 // The frames to pull is bigger than the remainder size.
102 // Wrap around the index.
103 memcpy(outputBusChannel, fifoBusChannel + m_indexRead, remainder * sizeof(*fifoBusChannel));
104 memcpy(outputBusChannel + remainder, fifoBusChannel, (framesToFill - remainder) * sizeof(*fifoBusChannel));
105 }
106
107 // The frames available was not enough to fulfill the requested frames. Fill
108 // the rest of the channel with silence.
109 if (framesRequested > framesToFill)
110 memset(outputBusChannel + framesToFill, 0, (framesRequested - framesToFill) * sizeof(*outputBusChannel));
111 }
112
113 // Update the read index; wrap it around if necessary.
114 m_indexRead = (m_indexRead + framesToFill) % m_fifoLength;
115
116 // In case of underflow, move the |indexWrite| to the updated |indexRead|.
117 if (framesRequested > framesToFill)
118 m_indexWrite = m_indexRead;
119
120 // Update the number of frames in FIFO.
121 m_framesAvailable -= framesToFill;
122 ASSERT(((m_indexRead + m_framesAvailable) % m_fifoLength) == m_indexWrite);
123
124 // |framesRequested > m_framesAvailable| means the frames in FIFO is not
125 // enough to fulfill the requested frames from the audio device.
126 return framesRequested > m_framesAvailable ? framesRequested - m_framesAvailable : 0;
127}

129unsigned PushPullFIFO::numberOfChannels() const
130{
131 return m_fifoBus->numberOfChannels();
132}
