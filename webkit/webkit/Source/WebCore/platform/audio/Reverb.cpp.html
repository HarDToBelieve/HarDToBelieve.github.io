44// Empirical gain calibration tested across many impulse responses to ensure perceived volume is same as dry (unprocessed) signal
45constexpr float GainCalibration = -58;
46constexpr float GainCalibrationSampleRate = 44100;
47
48// A minimum power value to when normalizing a silent (or very quiet) impulse response
49constexpr float MinPower = 0.000125f;
50
51static float calculateNormalizationScale(AudioBus* response)
52{
53 // Normalize by RMS power
54 size_t numberOfChannels = response->numberOfChannels();
55 size_t length = response->length();
56
57 float power = 0;
58
59 for (size_t i = 0; i < numberOfChannels; ++i)
60 power += VectorMath::sumOfSquares(response->channel(i)->data(), length);
61
62 power = sqrt(power / (numberOfChannels * length));
63
64 // Protect against accidental overload
65 if (std::isinf(power) || std::isnan(power) || power < MinPower)
66 power = MinPower;
67
68 float scale = 1 / power;
69
70 scale *= powf(10, GainCalibration * 0.05f); // calibrate to make perceived volume same as unprocessed
71
72 // Scale depends on sample-rate.
73 if (response->sampleRate())
74 scale *= GainCalibrationSampleRate / response->sampleRate();
75
76 // True-stereo compensation
77 if (response->numberOfChannels() == 4)
78 scale *= 0.5f;
79
80 return scale;
81}

83Reverb::Reverb(AudioBus* impulseResponse, size_t renderSliceSize, size_t maxFFTSize, bool useBackgroundThreads, bool normalize)
84{
85 float scale = 1;
86
87 if (normalize)
88 scale = calculateNormalizationScale(impulseResponse);
89
90 initialize(impulseResponse, renderSliceSize, maxFFTSize, useBackgroundThreads, scale);
91}

93void Reverb::initialize(AudioBus* impulseResponseBuffer, size_t renderSliceSize, size_t maxFFTSize, bool useBackgroundThreads, float scale)
94{
95 m_impulseResponseLength = impulseResponseBuffer->length();
96
97 // The reverb can handle a mono impulse response and still do stereo processing
98 m_numberOfResponseChannels = impulseResponseBuffer->numberOfChannels();
99 unsigned convolverCount = std::max(m_numberOfResponseChannels, 2u);
100 m_convolvers.reserveCapacity(convolverCount);
101
102 int convolverRenderPhase = 0;
103 for (unsigned i = 0; i < convolverCount; ++i) {
104 auto* channel = impulseResponseBuffer->channel(std::min(i, m_numberOfResponseChannels - 1));
105
106 m_convolvers.append(makeUnique<ReverbConvolver>(channel, renderSliceSize, maxFFTSize, convolverRenderPhase, useBackgroundThreads, scale));
107
108 convolverRenderPhase += renderSliceSize;
109 }
110
111 // For "True" stereo processing we allocate a temporary buffer to avoid repeatedly allocating it in the process() method.
112 // It can be bad to allocate memory in a real-time thread.
113 if (m_numberOfResponseChannels == 4)
114 m_tempBuffer = AudioBus::create(2, MaxFrameSize);
115}

117void Reverb::process(const AudioBus* sourceBus, AudioBus* destinationBus, size_t framesToProcess)
118{
119 // Do a fairly comprehensive sanity check.
120 // If these conditions are satisfied, all of the source and destination
121 // pointers will be valid for the various matrixing cases.
122 ASSERT(sourceBus);
123 ASSERT(destinationBus);
124 ASSERT(sourceBus->numberOfChannels() > 0u);
125 ASSERT(destinationBus->numberOfChannels() > 0u);
126 ASSERT(framesToProcess <= MaxFrameSize);
127 ASSERT(framesToProcess <= sourceBus->length());
128 ASSERT(framesToProcess <= destinationBus->length());
129
130 // For now only handle mono or stereo output
131 if (destinationBus->numberOfChannels() > 2) {
132 destinationBus->zero();
133 return;
134 }
135
136 AudioChannel* destinationChannelL = destinationBus->channel(0);
137 const AudioChannel* sourceChannelL = sourceBus->channel(0);
138
139 // Handle input -> output matrixing...
140 size_t numInputChannels = sourceBus->numberOfChannels();
141 size_t numOutputChannels = destinationBus->numberOfChannels();
142 size_t numberOfResponseChannels = m_numberOfResponseChannels;
143
144 ASSERT(numInputChannels <= 2ul);
145 ASSERT(numOutputChannels <= 2ul);
146 ASSERT(numberOfResponseChannels == 1 || numberOfResponseChannels == 2 || numberOfResponseChannels == 4);
147
148 // These are the possible combinations of number inputs, response
149 // channels and outputs channels that need to be supported:
150 //
151 // numInputChannels: 1 or 2
152 // numberOfResponseChannels: 1, 2, or 4
153 // numOutputChannels: 1 or 2
154 //
155 // Not all possible combinations are valid. numOutputChannels is
156 // one only if both numInputChannels and numberOfResponseChannels are 1.
157 // Otherwise numOutputChannels MUST be 2.
158 //
159 // The valid combinations are
160 //
161 // Case in -> resp -> out
162 // 1 1 -> 1 -> 1
163 // 2 1 -> 2 -> 2
164 // 3 1 -> 4 -> 2
165 // 4 2 -> 1 -> 2
166 // 5 2 -> 2 -> 2
167 // 6 2 -> 4 -> 2
168
169 if (numInputChannels == 2 && (numberOfResponseChannels == 1 || numberOfResponseChannels == 2) && numOutputChannels == 2) {
170 // Case 4 and 5: 2 -> 2 -> 2 or 2 -> 1 -> 2.
171 //
172 // These can be handled in the same way because in the latter
173 // case, two connvolvers are still created with the second being a
174 // copy of the first.
175 const AudioChannel* sourceChannelR = sourceBus->channel(1);
176 AudioChannel* destinationChannelR = destinationBus->channel(1);
177 m_convolvers[0]->process(sourceChannelL, destinationChannelL, framesToProcess);
178 m_convolvers[1]->process(sourceChannelR, destinationChannelR, framesToProcess);
179 } else if (numInputChannels == 1 && numOutputChannels == 2 && numberOfResponseChannels == 2) {
180 // Case 2: 1 -> 2 -> 2
181 for (int i = 0; i < 2; ++i) {
182 AudioChannel* destinationChannel = destinationBus->channel(i);
183 m_convolvers[i]->process(sourceChannelL, destinationChannel, framesToProcess);
184 }
185 } else if (numInputChannels == 1 && numberOfResponseChannels == 1) {
186 // Case 1: 1 -> 1 -> 1
187 ASSERT(numOutputChannels == 1ul);
188 m_convolvers[0]->process(sourceChannelL, destinationChannelL, framesToProcess);
189 } else if (numInputChannels == 2 && numberOfResponseChannels == 4 && numOutputChannels == 2) {
190 // Case 6: 2 -> 4 -> 2 ("True" stereo)
191 const AudioChannel* sourceChannelR = sourceBus->channel(1);
192 AudioChannel* destinationChannelR = destinationBus->channel(1);
193
194 AudioChannel* tempChannelL = m_tempBuffer->channel(0);
195 AudioChannel* tempChannelR = m_tempBuffer->channel(1);
196
197 // Process left virtual source
198 m_convolvers[0]->process(sourceChannelL, destinationChannelL, framesToProcess);
199 m_convolvers[1]->process(sourceChannelL, destinationChannelR, framesToProcess);
200
201 // Process right virtual source
202 m_convolvers[2]->process(sourceChannelR, tempChannelL, framesToProcess);
203 m_convolvers[3]->process(sourceChannelR, tempChannelR, framesToProcess);
204
205 destinationBus->sumFrom(*m_tempBuffer);
206 } else if (numInputChannels == 1 && numberOfResponseChannels == 4 && numOutputChannels == 2) {
207 // Case 3: 1 -> 4 -> 2 (Processing mono with "True" stereo impulse
208 // response) This is an inefficient use of a four-channel impulse
209 // response, but we should handle the case.
210 AudioChannel* destinationChannelR = destinationBus->channel(1);
211
212 AudioChannel* tempChannelL = m_tempBuffer->channel(0);
213 AudioChannel* tempChannelR = m_tempBuffer->channel(1);
214
215 // Process left virtual source
216 m_convolvers[0]->process(sourceChannelL, destinationChannelL, framesToProcess);
217 m_convolvers[1]->process(sourceChannelL, destinationChannelR, framesToProcess);
218
219 // Process right virtual source
220 m_convolvers[2]->process(sourceChannelL, tempChannelL, framesToProcess);
221 m_convolvers[3]->process(sourceChannelL, tempChannelR, framesToProcess);
222
223 destinationBus->sumFrom(*m_tempBuffer);
224 } else {
225 ASSERT_NOT_REACHED();
226 destinationBus->zero();
227 }
228}

230void Reverb::reset()
231{
232 for (size_t i = 0; i < m_convolvers.size(); ++i)
233 m_convolvers[i]->reset();
234}
235
236size_t Reverb::latencyFrames() const
237{
238 return !m_convolvers.isEmpty() ? m_convolvers.first()->latencyFrames() : 0;
239}
