41constexpr int InputBufferSize = 8 * 16384;
42
43// We only process the leading portion of the impulse response in the real-time thread. We don't exceed this length.
44// It turns out then, that the background thread has about 278msec of scheduling slop.
45// Empirically, this has been found to be a good compromise between giving enough time for scheduling slop,
46// while still minimizing the amount of processing done in the primary (high-priority) thread.
47// This was found to be a good value on Mac OS X, and may work well on other platforms as well, assuming
48// the very rough scheduling latencies are similar on these time-scales. Of course, this code may need to be
49// tuned for individual platforms if this assumption is found to be incorrect.
50constexpr size_t RealtimeFrameLimit = 8192 + 4096; // ~278msec @ 44.1KHz
51
52constexpr size_t MinFFTSize = 128;
53constexpr size_t MaxRealtimeFFTSize = 2048;
54
55ReverbConvolver::ReverbConvolver(AudioChannel* impulseResponse, size_t renderSliceSize, size_t maxFFTSize, size_t convolverRenderPhase, bool useBackgroundThreads, float scale)
56 : m_impulseResponseLength(impulseResponse->length())
57 , m_accumulationBuffer(impulseResponse->length() + renderSliceSize)
58 , m_inputBuffer(InputBufferSize)
59 , m_minFFTSize(MinFFTSize) // First stage will have this size - successive stages will double in size each time
60 , m_maxFFTSize(maxFFTSize) // until we hit m_maxFFTSize
61 , m_useBackgroundThreads(useBackgroundThreads)
62{
63 // If we are using background threads then don't exceed this FFT size for the
64 // stages which run in the real-time thread. This avoids having only one or two
65 // large stages (size 16384 or so) at the end which take a lot of time every several
66 // processing slices. This way we amortize the cost over more processing slices.
67 m_maxRealtimeFFTSize = MaxRealtimeFFTSize;
68
69 // For the moment, a good way to know if we have real-time constraint is to check if we're using background threads.
70 // Otherwise, assume we're being run from a command-line tool.
71 bool hasRealtimeConstraint = useBackgroundThreads;
72
73 const float* response = impulseResponse->data();
74 size_t totalResponseLength = impulseResponse->length();
75
76 // The total latency is zero because the direct-convolution is used in the leading portion.
77 size_t reverbTotalLatency = 0;
78
79 size_t stageOffset = 0;
80 int i = 0;
81 size_t fftSize = m_minFFTSize;
82 while (stageOffset < totalResponseLength) {
83 size_t stageSize = fftSize / 2;
84
85 // For the last stage, it's possible that stageOffset is such that we're straddling the end
86 // of the impulse response buffer (if we use stageSize), so reduce the last stage's length...
87 if (stageSize + stageOffset > totalResponseLength)
88 stageSize = totalResponseLength - stageOffset;
89
90 // This "staggers" the time when each FFT happens so they don't all happen at the same time
91 int renderPhase = convolverRenderPhase + i * renderSliceSize;
92
93 bool useDirectConvolver = !stageOffset;
94
95 auto stage = makeUnique<ReverbConvolverStage>(response, totalResponseLength, reverbTotalLatency, stageOffset, stageSize, fftSize, renderPhase, renderSliceSize, &m_accumulationBuffer, scale, useDirectConvolver);
96
97 bool isBackgroundStage = false;
98
99 if (this->useBackgroundThreads() && stageOffset > RealtimeFrameLimit) {
100 m_backgroundStages.append(WTFMove(stage));
101 isBackgroundStage = true;
102 } else
103 m_stages.append(WTFMove(stage));
104
105 stageOffset += stageSize;
106 ++i;
107
108 if (!useDirectConvolver) {
109 // Figure out next FFT size
110 fftSize *= 2;
111 }
112
113 if (hasRealtimeConstraint && !isBackgroundStage && fftSize > m_maxRealtimeFFTSize)
114 fftSize = m_maxRealtimeFFTSize;
115 if (fftSize > m_maxFFTSize)
116 fftSize = m_maxFFTSize;
117 }
118
119 // Start up background thread
120 // FIXME: would be better to up the thread priority here. It doesn't need to be real-time, but higher than the default...
121 if (this->useBackgroundThreads() && m_backgroundStages.size() > 0) {
122 m_backgroundThread = Thread::create("convolution background thread", [this] {
123 backgroundThreadEntry();
124 }, ThreadType::Audio);
125 }
126}
127
128ReverbConvolver::~ReverbConvolver()
129{
130 // Wait for background thread to stop
131 if (useBackgroundThreads() && m_backgroundThread) {
132 m_wantsToExit = true;
133
134 // Wake up thread so it can return
135 {
136 Locker locker { m_backgroundThreadLock };
137 m_moreInputBuffered = true;
138 m_backgroundThreadConditionVariable.notifyOne();
139 }
140
141 m_backgroundThread->waitForCompletion();
142 }
143}
144
145void ReverbConvolver::backgroundThreadEntry()
146{
147 while (!m_wantsToExit) {
148 // Wait for realtime thread to give us more input
149 m_moreInputBuffered = false;
150 {
151 Locker locker { m_backgroundThreadLock };
152
153 m_backgroundThreadConditionVariable.wait(m_backgroundThreadLock, [this] { return m_moreInputBuffered || m_wantsToExit; });
154 }
155
156 // Process all of the stages until their read indices reach the input buffer's write index
157 int writeIndex = m_inputBuffer.writeIndex();
158
159 // Even though it doesn't seem like every stage needs to maintain its own version of readIndex
160 // we do this in case we want to run in more than one background thread.
161 int readIndex;
162
163 while ((readIndex = m_backgroundStages[0]->inputReadIndex()) != writeIndex) { // FIXME: do better to detect buffer overrun...
164 // The ReverbConvolverStages need to process in amounts which evenly divide half the FFT size
165 const int SliceSize = MinFFTSize / 2;
166
167 // Accumulate contributions from each stage
168 for (size_t i = 0; i < m_backgroundStages.size(); ++i)
169 m_backgroundStages[i]->processInBackground(this, SliceSize);
170 }
171 }
172}

174void ReverbConvolver::process(const AudioChannel* sourceChannel, AudioChannel* destinationChannel, size_t framesToProcess)
175{
176 bool isSafe = sourceChannel && destinationChannel && sourceChannel->length() >= framesToProcess && destinationChannel->length() >= framesToProcess;
177 ASSERT(isSafe);
178 if (!isSafe)
179 return;
180
181 const float* source = sourceChannel->data();
182 float* destination = destinationChannel->mutableData();
183 bool isDataSafe = source && destination;
184 ASSERT(isDataSafe);
185 if (!isDataSafe)
186 return;
187
188 // Feed input buffer (read by all threads)
189 m_inputBuffer.write(source, framesToProcess);
190
191 // Accumulate contributions from each stage
192 for (size_t i = 0; i < m_stages.size(); ++i)
193 m_stages[i]->process(source, framesToProcess);
194
195 // Finally read from accumulation buffer
196 m_accumulationBuffer.readAndClear(destination, framesToProcess);
197
198 // Now that we've buffered more input, wake up our background thread.
199
200 // We use use std::unique_lock with std::try_lock here because this is run on the real-time
201 // thread where it is a disaster for the lock to be contended (causes audio glitching). It's OK if we fail to
202 // signal from time to time, since we'll get to it the next time we're called. We're called repeatedly
203 // and frequently (around every 3ms). The background thread is processing well into the future and has a considerable amount of
204 // leeway here...
205 if (!m_backgroundThreadLock.tryLock())
206 return;
207 Locker locker { AdoptLock, m_backgroundThreadLock };
208
209 m_moreInputBuffered = true;
210 m_backgroundThreadConditionVariable.notifyOne();
211}
212
213void ReverbConvolver::reset()
214{
215 for (size_t i = 0; i < m_stages.size(); ++i)
216 m_stages[i]->reset();
217
218 for (size_t i = 0; i < m_backgroundStages.size(); ++i)
219 m_backgroundStages[i]->reset();
220
221 m_accumulationBuffer.reset();
222 m_inputBuffer.reset();
223}
