45ReverbConvolverStage::ReverbConvolverStage(const float* impulseResponse, size_t, size_t reverbTotalLatency, size_t stageOffset, size_t stageLength,
46 size_t fftSize, size_t renderPhase, size_t renderSliceSize, ReverbAccumulationBuffer* accumulationBuffer, float scale, bool directMode)
47 : m_accumulationBuffer(accumulationBuffer)
48 , m_directMode(directMode)
49{
50 ASSERT(impulseResponse);
51 ASSERT(accumulationBuffer);
52
53 if (!m_directMode) {
54 m_fftKernel = makeUnique<FFTFrame>(fftSize);
55 m_fftKernel->doPaddedFFT(impulseResponse + stageOffset, stageLength);
56 // Account for the normalization (if any) of the convolver. By linearity,
57 // we can scale the FFT by the factor instead of the input. We do it this
58 // way so we don't need to create a temporary for the scaled result before
59 // computing the FFT.
60 if (scale != 1)
61 m_fftKernel->scaleFFT(scale);
62 m_fftConvolver = makeUnique<FFTConvolver>(fftSize);
63 } else {
64 ASSERT(!stageOffset);
65 ASSERT(stageLength <= fftSize / 2);
66
67 m_directKernel = makeUnique<AudioFloatArray>(fftSize / 2);
68 m_directKernel->copyToRange(impulseResponse, 0, stageLength);
69 // Account for the normalization (if any) of the convolver node.
70 if (scale != 1)
71 VectorMath::multiplyByScalar(m_directKernel->data(), scale, m_directKernel->data(), stageLength);
72 m_directConvolver = makeUnique<DirectConvolver>(renderSliceSize);
73 }
74 m_temporaryBuffer.resize(renderSliceSize);
75
76 // The convolution stage at offset stageOffset needs to have a corresponding delay to cancel out the offset.
77 size_t totalDelay = stageOffset + reverbTotalLatency;
78
79 // But, the FFT convolution itself incurs fftSize / 2 latency, so subtract this out...
80 size_t halfSize = fftSize / 2;
81 if (!m_directMode) {
82 ASSERT(totalDelay >= halfSize);
83 if (totalDelay >= halfSize)
84 totalDelay -= halfSize;
85 }
86
87 // We divide up the total delay, into pre and post delay sections so that we can schedule at exactly the moment when the FFT will happen.
88 // This is coordinated with the other stages, so they don't all do their FFTs at the same time...
89 int maxPreDelayLength = std::min(halfSize, totalDelay);
90 m_preDelayLength = totalDelay > 0 ? renderPhase % maxPreDelayLength : 0;
91 if (m_preDelayLength > totalDelay)
92 m_preDelayLength = 0;
93
94 m_postDelayLength = totalDelay - m_preDelayLength;
95 m_preReadWriteIndex = 0;
96 m_framesProcessed = 0; // total frames processed so far
97
98 size_t delayBufferSize = m_preDelayLength < fftSize ? fftSize : m_preDelayLength;
99 delayBufferSize = delayBufferSize < renderSliceSize ? renderSliceSize : delayBufferSize;
100 m_preDelayBuffer.resize(delayBufferSize);
101}

105void ReverbConvolverStage::processInBackground(ReverbConvolver* convolver, size_t framesToProcess)
106{
107 ReverbInputBuffer* inputBuffer = convolver->inputBuffer();
108 float* source = inputBuffer->directReadFrom(&m_inputReadIndex, framesToProcess);
109 process(source, framesToProcess);
110}
111
112void ReverbConvolverStage::process(const float* source, size_t framesToProcess)
113{
114 ASSERT(source);
115 if (!source)
116 return;
117
118 // Deal with pre-delay stream : note special handling of zero delay.
119
120 const float* preDelayedSource;
121 float* preDelayedDestination;
122 float* temporaryBuffer;
123 bool isTemporaryBufferSafe = false;
124 if (m_preDelayLength > 0) {
125 // Handles both the read case (call to process() ) and the write case (memcpy() )
126 bool isPreDelaySafe = m_preReadWriteIndex + framesToProcess <= m_preDelayBuffer.size();
127 ASSERT(isPreDelaySafe);
128 if (!isPreDelaySafe)
129 return;
130
131 isTemporaryBufferSafe = framesToProcess <= m_temporaryBuffer.size();
132
133 preDelayedDestination = m_preDelayBuffer.data() + m_preReadWriteIndex;
134 preDelayedSource = preDelayedDestination;
135 temporaryBuffer = m_temporaryBuffer.data();
136 } else {
137 // Zero delay
138 preDelayedDestination = 0;
139 preDelayedSource = source;
140 temporaryBuffer = m_preDelayBuffer.data();
141
142 isTemporaryBufferSafe = framesToProcess <= m_preDelayBuffer.size();
143 }
144
145 ASSERT(isTemporaryBufferSafe);
146 if (!isTemporaryBufferSafe)
147 return;
148
149 if (m_framesProcessed < m_preDelayLength) {
150 // For the first m_preDelayLength frames don't process the convolver, instead simply buffer in the pre-delay.
151 // But while buffering the pre-delay, we still need to update our index.
152 m_accumulationBuffer->updateReadIndex(&m_accumulationReadIndex, framesToProcess);
153 } else {
154 // Now, run the convolution (into the delay buffer).
155 // An expensive FFT will happen every fftSize / 2 frames.
156 // We process in-place here...
157 if (!m_directMode)
158 m_fftConvolver->process(m_fftKernel.get(), preDelayedSource, temporaryBuffer, framesToProcess);
159 else
160 m_directConvolver->process(m_directKernel.get(), preDelayedSource, temporaryBuffer, framesToProcess);
161
162 // Now accumulate into reverb's accumulation buffer.
163 m_accumulationBuffer->accumulate(temporaryBuffer, framesToProcess, &m_accumulationReadIndex, m_postDelayLength);
164 }
165
166 // Finally copy input to pre-delay.
167 if (m_preDelayLength > 0) {
168 memcpy(preDelayedDestination, source, sizeof(float) * framesToProcess);
169 m_preReadWriteIndex += framesToProcess;
170
171 ASSERT(m_preReadWriteIndex <= m_preDelayLength);
172 if (m_preReadWriteIndex >= m_preDelayLength)
173 m_preReadWriteIndex = 0;
174 }
175
176 m_framesProcessed += framesToProcess;
177}
178
179void ReverbConvolverStage::reset()
180{
181 if (!m_directMode)
182 m_fftConvolver->reset();
183 else
184 m_directConvolver->reset();
185 m_preDelayBuffer.zero();
186 m_accumulationReadIndex = 0;
187 m_inputReadIndex = 0;
188 m_framesProcessed = 0;
189}
