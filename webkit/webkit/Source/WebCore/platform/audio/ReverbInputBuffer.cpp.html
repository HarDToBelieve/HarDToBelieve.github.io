37ReverbInputBuffer::ReverbInputBuffer(size_t length)
38 : m_buffer(length)
39{
40}
41
42void ReverbInputBuffer::write(const float* sourceP, size_t numberOfFrames)
43{
44 size_t bufferLength = m_buffer.size();
45 bool isCopySafe = m_writeIndex + numberOfFrames <= bufferLength;
46 ASSERT(isCopySafe);
47 if (!isCopySafe)
48 return;
49
50 memcpy(m_buffer.data() + m_writeIndex, sourceP, sizeof(float) * numberOfFrames);
51
52 m_writeIndex += numberOfFrames;
53 ASSERT(m_writeIndex <= bufferLength);
54
55 if (m_writeIndex >= bufferLength)
56 m_writeIndex = 0;
57}
58
59float* ReverbInputBuffer::directReadFrom(int* readIndex, size_t numberOfFrames)
60{
61 size_t bufferLength = m_buffer.size();
62 bool isPointerGood = readIndex && *readIndex >= 0 && *readIndex + numberOfFrames <= bufferLength;
63 ASSERT(isPointerGood);
64 if (!isPointerGood) {
65 // Should never happen in practice but return pointer to start of buffer (avoid crash)
66 if (readIndex)
67 *readIndex = 0;
68 return m_buffer.data();
69 }
70
71 float* sourceP = m_buffer.data();
72 float* p = sourceP + *readIndex;
73
74 // Update readIndex
75 *readIndex = (*readIndex + numberOfFrames) % bufferLength;
76
77 return p;
78}
79
80void ReverbInputBuffer::reset()
81{
82 m_buffer.zero();
83 m_writeIndex = 0;
84}
