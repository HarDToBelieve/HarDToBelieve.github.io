47// Initial input buffer layout, dividing into regions r0 to r4 (note: r0, r3
48// and r4 will move after the first load):
49//
50// |----------------|----------------------------------------------------------------|----------------|
51//
52// m_requestFrames
53// <-------------------------------------------------------------------------------->
54// r0 (during first load)
55//
56// kernelSize / 2 kernelSize / 2 kernelSize / 2 kernelSize / 2
57// <---------------> <---------------> <---------------> <--------------->
58// r1 r2 r3 r4
59//
60// m_blockSize == r4 - r2
61// <--------------------------------------->
62//
63// m_requestFrames
64// <------------------ ... ----------------->
65// r0 (during second load)
66//
67// On the second request r0 slides to the right by kernelSize / 2 and r3, r4
68// and m_blockSize are reinitialized via step (3) in the algorithm below.
69//
70// These new regions remain constant until a Flush() occurs. While complicated,
71// this allows us to reduce jitter by always requesting the same amount from the
72// provided callback.
73
74// The Algorithm:
75//
76// 1) Allocate input_buffer of size: m_requestFrames + kernelSize; this ensures
77// there's enough room to read m_requestFrames from the callback into region
78// r0 (which will move between the first and subsequent passes).
79//
80// 2) Let r1, r2 each represent half the kernel centered around r0:
81//
82// r0 = m_inputBuffer + kernelSize / 2
83// r1 = m_inputBuffer
84// r2 = r0
85//
86// r0 is always m_requestFrames in size. r1, r2 are kernelSize / 2 in
87// size. r1 must be zero initialized to avoid convolution with garbage (see
88// step (5) for why).
89//
90// 3) Let r3, r4 each represent half the kernel right aligned with the end of
91// r0 and choose m_blockSize as the distance in frames between r4 and r2:
92//
93// r3 = r0 + m_requestFrames - kernelSize
94// r4 = r0 + m_requestFrames - kernelSize / 2
95// m_blockSize = r4 - r2 = m_requestFrames - kernelSize / 2
96//
97// 4) Consume m_requestFrames frames into r0.
98//
99// 5) Position kernel centered at start of r2 and generate output frames until
100// the kernel is centered at the start of r4 or we've finished generating
101// all the output frames.
102//
103// 6) Wrap left over data from the r3 to r1 and r4 to r2.
104//
105// 7) If we're on the second load, in order to avoid overwriting the frames we
106// just wrapped from r4 we need to slide r0 to the right by the size of
107// r4, which is kernelSize / 2:
108//
109// r0 = r0 + kernelSize / 2 = m_inputBuffer + kernelSize
110//
111// r3, r4, and m_blockSize then need to be reinitialized, so goto (3).
112//
113// 8) Else, if we're not on the second load, goto (4).
114//
115// note: we're glossing over how the sub-sample handling works with m_virtualSourceIndex, etc.
116
117namespace WebCore {
118
119constexpr unsigned kernelSize { 32 };
120constexpr unsigned numberOfKernelOffsets { 32 };
121constexpr unsigned kernelStorageSize { kernelSize * (numberOfKernelOffsets + 1) };
122
123static size_t calculateChunkSize(unsigned blockSize, double scaleFactor)
124{
125 return blockSize / scaleFactor;
126}

128SincResampler::SincResampler(double scaleFactor, unsigned requestFrames, Function<void(float* buffer, size_t framesToProcess)>&& provideInput)
129 : m_scaleFactor(scaleFactor)
130 , m_kernelStorage(kernelStorageSize)
131 , m_requestFrames(requestFrames)
132 , m_provideInput(WTFMove(provideInput))
133 , m_inputBuffer(m_requestFrames + kernelSize) // See input buffer layout above.
134 , m_r1(m_inputBuffer.data())
135 , m_r2(m_inputBuffer.data() + kernelSize / 2)
136{
137 ASSERT(m_provideInput);
138 ASSERT(m_requestFrames > 0);
139 updateRegions(false);
140 ASSERT(m_blockSize > kernelSize);
141 initializeKernel();
142}

144void SincResampler::updateRegions(bool isSecondLoad)
145{
146 // Setup various region pointers in the buffer (see diagram above). If we're
147 // on the second load we need to slide m_r0 to the right by kernelSize / 2.
148 m_r0 = m_inputBuffer.data() + (isSecondLoad ? kernelSize : kernelSize / 2);
149 m_r3 = m_r0 + m_requestFrames - kernelSize;
150 m_r4 = m_r0 + m_requestFrames - kernelSize / 2;
151 m_blockSize = m_r4 - m_r2;
152 m_chunkSize = calculateChunkSize(m_blockSize, m_scaleFactor);
153
154 // m_r1 at the beginning of the buffer.
155 ASSERT(m_r1 == m_inputBuffer.data());
156 // m_r1 left of m_r2, m_r4 left of m_r3 and size correct.
157 ASSERT((m_r2 - m_r1) == (m_r4 - m_r3));
158 // m_r2 left of r3.
159 ASSERT(m_r2 <= m_r3);
160}

162void SincResampler::initializeKernel()
163{
164 // Blackman window parameters.
165 double alpha = 0.16;
166 double a0 = 0.5 * (1.0 - alpha);
167 double a1 = 0.5;
168 double a2 = 0.5 * alpha;
169
170 // sincScaleFactor is basically the normalized cutoff frequency of the low-pass filter.
171 double sincScaleFactor = m_scaleFactor > 1.0 ? 1.0 / m_scaleFactor : 1.0;
172
173 // The sinc function is an idealized brick-wall filter, but since we're windowing it the
174 // transition from pass to stop does not happen right away. So we should adjust the
175 // lowpass filter cutoff slightly downward to avoid some aliasing at the very high-end.
176 // FIXME: this value is empirical and to be more exact should vary depending on kernelSize.
177 sincScaleFactor *= 0.9;
178
179 int n = kernelSize;
180 int halfSize = n / 2;
181
182 // Generates a set of windowed sinc() kernels.
183 // We generate a range of sub-sample offsets from 0.0 to 1.0.
184 for (unsigned offsetIndex = 0; offsetIndex <= numberOfKernelOffsets; ++offsetIndex) {
185 double subsampleOffset = static_cast<double>(offsetIndex) / numberOfKernelOffsets;
186
187 for (int i = 0; i < n; ++i) {
188 // Compute the sinc() with offset.
189 double s = sincScaleFactor * piDouble * (i - halfSize - subsampleOffset);
190 double sinc = !s ? 1.0 : sin(s) / s;
191 sinc *= sincScaleFactor;
192
193 // Compute Blackman window, matching the offset of the sinc().
194 double x = (i - subsampleOffset) / n;
195 double window = a0 - a1 * cos(2.0 * piDouble * x) + a2 * cos(4.0 * piDouble * x);
196
197 // Window the sinc() function and store at the correct offset.
198 m_kernelStorage[i + offsetIndex * kernelSize] = sinc * window;
199 }
200 }
201}

203void SincResampler::processBuffer(const float* source, float* destination, unsigned numberOfSourceFrames, double scaleFactor)
204{
205 SincResampler resampler(scaleFactor, AudioUtilities::renderQuantumSize, [source, numberOfSourceFrames](float* buffer, size_t framesToProcess) mutable {
206 // Clamp to number of frames available and zero-pad.
207 size_t framesToCopy = std::min<size_t>(numberOfSourceFrames, framesToProcess);
208 memcpy(buffer, source, sizeof(float) * framesToCopy);
209
210 // Zero-pad if necessary.
211 if (framesToCopy < framesToProcess)
212 memset(buffer + framesToCopy, 0, sizeof(float) * (framesToProcess - framesToCopy));
213
214 numberOfSourceFrames -= framesToCopy;
215 source += framesToCopy;
216 });
217
218 unsigned numberOfDestinationFrames = static_cast<unsigned>(numberOfSourceFrames / scaleFactor);
219 unsigned remaining = numberOfDestinationFrames;
220
221 while (remaining) {
222 unsigned framesThisTime = std::min<unsigned>(remaining, AudioUtilities::renderQuantumSize);
223 resampler.process(destination, framesThisTime);
224
225 destination += framesThisTime;
226 remaining -= framesThisTime;
227 }
228}
229
230void SincResampler::process(float* destination, size_t framesToProcess)
231{
232 unsigned numberOfDestinationFrames = framesToProcess;
233
234 // Step (1)
235 // Prime the input buffer at the start of the input stream.
236 if (!m_isBufferPrimed) {
237 m_provideInput(m_r0, m_requestFrames);
238 m_isBufferPrimed = true;
239 }
240
241 // Step (2)
242
243 while (numberOfDestinationFrames) {
244 while (m_virtualSourceIndex < m_blockSize) {
245 // m_virtualSourceIndex lies in between two kernel offsets so figure out what they are.
246 int sourceIndexI = static_cast<int>(m_virtualSourceIndex);
247 double subsampleRemainder = m_virtualSourceIndex - sourceIndexI;
248
249 double virtualOffsetIndex = subsampleRemainder * numberOfKernelOffsets;
250 int offsetIndex = static_cast<int>(virtualOffsetIndex);
251
252 float* k1 = m_kernelStorage.data() + offsetIndex * kernelSize;
253 float* k2 = k1 + kernelSize;
254
255 // Ensure |k1|, |k2| are 16-byte aligned for SIMD usage. Should always be true so long as kernelSize is a multiple of 16.
256 ASSERT(!(reinterpret_cast<uintptr_t>(k1) & 0x0F));
257 ASSERT(!(reinterpret_cast<uintptr_t>(k2) & 0x0F));
258
259 // Initialize input pointer based on quantized m_virtualSourceIndex.
260 float* inputP = m_r1 + sourceIndexI;
261
262 // Figure out how much to weight each kernel's "convolution".
263 double kernelInterpolationFactor = virtualOffsetIndex - offsetIndex;
264
265 *destination++ = convolve(inputP, k1, k2, kernelInterpolationFactor);
266
267 // Advance the virtual index.
268 m_virtualSourceIndex += m_scaleFactor;
269
270 --numberOfDestinationFrames;
271 if (!numberOfDestinationFrames)
272 return;
273 }
274
275 // Wrap back around to the start.
276 ASSERT(m_virtualSourceIndex >= m_blockSize);
277 m_virtualSourceIndex -= m_blockSize;
278
279 // Step (3) Copy r3 to r1.
280 // This wraps the last input frames back to the start of the buffer.
281 memcpy(m_r1, m_r3, sizeof(float) * kernelSize);
282
283 // Step (4) -- Reinitialize regions if necessary.
284 if (m_r0 == m_r2)
285 updateRegions(true);
286
287 // Step (5)
288 // Refresh the buffer with more input.
289 m_provideInput(m_r0, m_requestFrames);
290 }
291}

293float SincResampler::convolve(const float* inputP, const float* k1, const float* k2, float kernelInterpolationFactor)
294{
295#if USE(ACCELERATE)
296 float sum1;
297 float sum2;
298 vDSP_dotpr(inputP, 1, k1, 1, &sum1, kernelSize);
299 vDSP_dotpr(inputP, 1, k2, 1, &sum2, kernelSize);
300
301 // Linearly interpolate the two "convolutions".
302 return (1.0f - kernelInterpolationFactor) * sum1 + kernelInterpolationFactor * sum2;
303#elif CPU(X86_SSE2)
304 __m128 m_input;
305 __m128 m_sums1 = _mm_setzero_ps();
306 __m128 m_sums2 = _mm_setzero_ps();
307
308 // Based on |inputP| alignment, we need to use loadu or load.
309 if (reinterpret_cast<uintptr_t>(inputP) & 0x0F) {
310 for (unsigned i = 0; i < kernelSize; i += 4) {
311 m_input = _mm_loadu_ps(inputP + i);
312 m_sums1 = _mm_add_ps(m_sums1, _mm_mul_ps(m_input, _mm_load_ps(k1 + i)));
313 m_sums2 = _mm_add_ps(m_sums2, _mm_mul_ps(m_input, _mm_load_ps(k2 + i)));
314 }
315 } else {
316 for (unsigned i = 0; i < kernelSize; i += 4) {
317 m_input = _mm_load_ps(inputP + i);
318 m_sums1 = _mm_add_ps(m_sums1, _mm_mul_ps(m_input, _mm_load_ps(k1 + i)));
319 m_sums2 = _mm_add_ps(m_sums2, _mm_mul_ps(m_input, _mm_load_ps(k2 + i)));
320 }
321 }
322
323 // Linearly interpolate the two "convolutions".
324 m_sums1 = _mm_mul_ps(m_sums1, _mm_set_ps1(1.0f - kernelInterpolationFactor));
325 m_sums2 = _mm_mul_ps(m_sums2, _mm_set_ps1(kernelInterpolationFactor));
326 m_sums1 = _mm_add_ps(m_sums1, m_sums2);
327
328 // Sum components together.
329 float result;
330 m_sums2 = _mm_add_ps(_mm_movehl_ps(m_sums1, m_sums1), m_sums1);
331 _mm_store_ss(&result, _mm_add_ss(m_sums2, _mm_shuffle_ps(m_sums2, m_sums2, 1)));
332
333 return result;
334#elif HAVE(ARM_NEON_INTRINSICS)
335 float32x4_t m_input;
336 float32x4_t m_sums1 = vmovq_n_f32(0);
337 float32x4_t m_sums2 = vmovq_n_f32(0);
338
339 const float* upper = inputP + kernelSize;
340 for (; inputP < upper; ) {
341 m_input = vld1q_f32(inputP);
342 inputP += 4;
343 m_sums1 = vmlaq_f32(m_sums1, m_input, vld1q_f32(k1));
344 k1 += 4;
345 m_sums2 = vmlaq_f32(m_sums2, m_input, vld1q_f32(k2));
346 k2 += 4;
347 }
348
349 // Linearly interpolate the two "convolutions".
350 m_sums1 = vmlaq_f32(vmulq_f32(m_sums1, vmovq_n_f32(1.0 - kernelInterpolationFactor)), m_sums2, vmovq_n_f32(kernelInterpolationFactor));
351
352 // Sum components together.
353 float32x2_t m_half = vadd_f32(vget_high_f32(m_sums1), vget_low_f32(m_sums1));
354 return vget_lane_f32(vpadd_f32(m_half, m_half), 0);
355#else
356 float sum1 = 0;
357 float sum2 = 0;
358
359 // Generate a single output sample.
360 int n = kernelSize;
361 while (n--) {
362 sum1 += *inputP * *k1++;
363 sum2 += *inputP++ * *k2++;
364 }
365
366 // Linearly interpolate the two "convolutions".
367 return (1.0f - kernelInterpolationFactor) * sum1 + kernelInterpolationFactor * sum2;
368#endif
369}
