40void panWithSampleAccurateValues(const AudioBus* inputBus, AudioBus* outputBus, const float* panValues, size_t framesToProcess)
41{
42 bool isInputSafe = inputBus && (inputBus->numberOfChannels() == 1 || inputBus->numberOfChannels() == 2) && framesToProcess <= inputBus->length();
43 ASSERT(isInputSafe);
44 if (!isInputSafe)
45 return;
46
47 unsigned numberOfInputChannels = inputBus->numberOfChannels();
48
49 bool isOutputSafe = outputBus && outputBus->numberOfChannels() == 2 && framesToProcess <= outputBus->length();
50 ASSERT(isOutputSafe);
51 if (!isOutputSafe)
52 return;
53
54 const float* sourceL = inputBus->channel(0)->data();
55 const float* sourceR = numberOfInputChannels > 1 ? inputBus->channel(1)->data() : sourceL;
56 float* destinationL = outputBus->channelByType(AudioBus::ChannelLeft)->mutableData();
57 float* destinationR = outputBus->channelByType(AudioBus::ChannelRight)->mutableData();
58
59 if (!sourceL || !sourceR || !destinationL || !destinationR)
60 return;
61
62 double gainL;
63 double gainR;
64 double panRadian;
65
66 int n = framesToProcess;
67
68 // Handles mono source case first, then stereo source case.
69 if (numberOfInputChannels == 1) {
70 while (n--) {
71 float inputL = *sourceL++;
72 double pan = clampTo(*panValues++, -1.0, 1.0);
73 // Pan from left to right [-1; 1] will be normalized as [0; 1].
74 panRadian = (pan * 0.5 + 0.5) * piOverTwoDouble;
75 gainL = cos(panRadian);
76 gainR = sin(panRadian);
77 *destinationL++ = static_cast<float>(inputL * gainL);
78 *destinationR++ = static_cast<float>(inputL * gainR);
79 }
80 } else {
81 while (n--) {
82 float inputL = *sourceL++;
83 float inputR = *sourceR++;
84 double pan = clampTo(*panValues++, -1.0, 1.0);
85 // Normalize [-1; 0] to [0; 1]. Do nothing when [0; 1].
86 panRadian = (pan <= 0 ? pan + 1 : pan) * piOverTwoDouble;
87 gainL = cos(panRadian);
88 gainR = sin(panRadian);
89 if (pan <= 0) {
90 *destinationL++ = static_cast<float>(inputL + inputR * gainL);
91 *destinationR++ = static_cast<float>(inputR * gainR);
92 } else {
93 *destinationL++ = static_cast<float>(inputL * gainL);
94 *destinationR++ = static_cast<float>(inputR + inputL * gainR);
95 }
96 }
97 }
98}

100void panToTargetValue(const AudioBus* inputBus, AudioBus* outputBus, float panValue, size_t framesToProcess)
101{
102 bool isInputSafe = inputBus && (inputBus->numberOfChannels() == 1 || inputBus->numberOfChannels() == 2) && framesToProcess <= inputBus->length();
103 ASSERT(isInputSafe);
104 if (!isInputSafe)
105 return;
106
107 unsigned numberOfInputChannels = inputBus->numberOfChannels();
108
109 bool isOutputSafe = outputBus && outputBus->numberOfChannels() == 2 && framesToProcess <= outputBus->length();
110 ASSERT(isOutputSafe);
111 if (!isOutputSafe)
112 return;
113
114 const float* sourceL = inputBus->channel(0)->data();
115 const float* sourceR = numberOfInputChannels > 1 ? inputBus->channel(1)->data() : sourceL;
116 float* destinationL = outputBus->channelByType(AudioBus::ChannelLeft)->mutableData();
117 float* destinationR = outputBus->channelByType(AudioBus::ChannelRight)->mutableData();
118
119 if (!sourceL || !sourceR || !destinationL || !destinationR)
120 return;
121
122 float targetPan = clampTo(panValue, -1.0, 1.0);
123
124 if (numberOfInputChannels == 1) {
125 double panRadian = (targetPan * 0.5 + 0.5) * piOverTwoDouble;
126
127 double gainL = cos(panRadian);
128 double gainR = sin(panRadian);
129
130 VectorMath::multiplyByScalar(sourceL, gainL, destinationL, framesToProcess);
131 VectorMath::multiplyByScalar(sourceL, gainR, destinationR, framesToProcess);
132 } else {
133 double panRadian = (targetPan <= 0 ? targetPan + 1 : targetPan) * piOverTwoDouble;
134
135 double gainL = cos(panRadian);
136 double gainR = sin(panRadian);
137
138 if (targetPan <= 0) {
139 VectorMath::multiplyByScalarThenAddToVector(sourceR, gainL, sourceL, destinationL, framesToProcess);
140 VectorMath::multiplyByScalar(sourceR, gainR, destinationR, framesToProcess);
141 } else {
142 VectorMath::multiplyByScalar(sourceL, gainL, destinationL, framesToProcess);
143 VectorMath::multiplyByScalarThenAddToVector(sourceL, gainR, sourceR, destinationR, framesToProcess);
144 }
145 }
