29#ifndef UpSampler_h
30#define UpSampler_h
31
32#include "AudioArray.h"
33#include "DirectConvolver.h"
34
35namespace WebCore {
36
37// UpSampler up-samples the source stream by a factor of 2x.
38
39class UpSampler final {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 explicit UpSampler(size_t inputBlockSize);
43
44 // The destination buffer |destP| is of size sourceFramesToProcess * 2.
45 void process(const float* sourceP, float* destP, size_t sourceFramesToProcess);
46
47 void reset();
48
49 // Latency based on the source sample-rate.
50 size_t latencyFrames() const;
51
52private:
53 enum { DefaultKernelSize = 128 };
54
55 size_t m_inputBlockSize;
56
57 // Computes ideal band-limited filter coefficients to sample in between each source sample-frame.
58 // This filter will be used to compute the odd sample-frames of the output.
59 void initializeKernel();
60 AudioFloatArray m_kernel;
61
62 // Computes the odd sample-frames of the output.
63 DirectConvolver m_convolver;
64
65 AudioFloatArray m_tempBuffer;
66
67 // Delay line for generating the even sample-frames of the output.
68 // The source samples are delayed exactly to match the linear phase delay of the FIR filter (convolution)
69 // used to generate the odd sample-frames of the output.
70 AudioFloatArray m_inputBuffer;
71};
72
73} // namespace WebCore
74
75#endif // UpSampler_h
