152void multiplyByScalarThenAddToOutput(const float* inputVector, float scalar, float* outputVector, size_t numberOfElementsToProcess)
153{
154 size_t n = numberOfElementsToProcess;
155
156#if CPU(X86_SSE2)
157 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
158 while (!is16ByteAligned(inputVector) && n) {
159 *outputVector += scalar * *inputVector;
160 inputVector++;
161 outputVector++;
162 n--;
163 }
164
165 // Now the inputVector is aligned, use SSE.
166 size_t tailFrames = n % 4;
167 const float* endP = outputVector + n - tailFrames;
168
169 __m128 pSource;
170 __m128 dest;
171 __m128 temp;
172 __m128 mScale = _mm_set_ps1(scalar);
173
174 bool destAligned = is16ByteAligned(outputVector);
175
176#define SSE2_MULT_ADD(loadInstr, storeInstr) \
177 while (outputVector < endP) \
178 { \
179 pSource = _mm_load_ps(inputVector); \
180 temp = _mm_mul_ps(pSource, mScale); \
181 dest = _mm_##loadInstr##_ps(outputVector); \
182 dest = _mm_add_ps(dest, temp); \
183 _mm_##storeInstr##_ps(outputVector, dest); \
184 inputVector += 4; \
185 outputVector += 4; \
186 }
187
188 if (destAligned)
189 SSE2_MULT_ADD(load, store)
190 else
191 SSE2_MULT_ADD(loadu, storeu)
192
193 n = tailFrames;
194#elif HAVE(ARM_NEON_INTRINSICS)
195 size_t tailFrames = n % 4;
196 const float* endP = outputVector + n - tailFrames;
197
198 float32x4_t k = vdupq_n_f32(scalar);
199 while (outputVector < endP) {
200 float32x4_t source = vld1q_f32(inputVector);
201 float32x4_t dest = vld1q_f32(outputVector);
202
203 dest = vmlaq_f32(dest, source, k);
204 vst1q_f32(outputVector, dest);
205
206 inputVector += 4;
207 outputVector += 4;
208 }
209 n = tailFrames;
210#endif
211 while (n--) {
212 *outputVector += *inputVector * scalar;
213 ++inputVector;
214 ++outputVector;
215 }
216}

218void multiplyByScalar(const float* inputVector, float scalar, float* outputVector, size_t numberOfElementsToProcess)
219{
220 size_t n = numberOfElementsToProcess;
221
222#if CPU(X86_SSE2)
223 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
224 while (!is16ByteAligned(inputVector) && n) {
225 *outputVector = scalar * *inputVector;
226 inputVector++;
227 outputVector++;
228 n--;
229 }
230
231 // Now the inputVector address is aligned and start to apply SSE.
232 size_t group = n / 4;
233 __m128 mScale = _mm_set_ps1(scalar);
234 __m128* pSource;
235 __m128* pDest;
236 __m128 dest;
237
238
239 if (!is16ByteAligned(outputVector)) {
240 while (group--) {
241 pSource = reinterpret_cast<__m128*>(const_cast<float*>(inputVector));
242 dest = _mm_mul_ps(*pSource, mScale);
243 _mm_storeu_ps(outputVector, dest);
244
245 inputVector += 4;
246 outputVector += 4;
247 }
248 } else {
249 while (group--) {
250 pSource = reinterpret_cast<__m128*>(const_cast<float*>(inputVector));
251 pDest = reinterpret_cast<__m128*>(outputVector);
252 *pDest = _mm_mul_ps(*pSource, mScale);
253
254 inputVector += 4;
255 outputVector += 4;
256 }
257 }
258
259 // Non-SSE handling for remaining frames which is less than 4.
260 n %= 4;
261#elif HAVE(ARM_NEON_INTRINSICS)
262 size_t tailFrames = n % 4;
263 const float* endP = outputVector + n - tailFrames;
264
265 while (outputVector < endP) {
266 float32x4_t source = vld1q_f32(inputVector);
267 vst1q_f32(outputVector, vmulq_n_f32(source, scalar));
268
269 inputVector += 4;
270 outputVector += 4;
271 }
272 n = tailFrames;
273#endif
274 while (n--) {
275 *outputVector = scalar * *inputVector;
276 ++inputVector;
277 ++outputVector;
278 }
279}

281void addScalar(const float* inputVector, float scalar, float* outputVector, size_t numberOfElementsToProcess)
282{
283 size_t n = numberOfElementsToProcess;
284
285#if CPU(X86_SSE2)
286 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
287 while (!is16ByteAligned(inputVector) && n) {
288 *outputVector = *inputVector + scalar;
289 inputVector++;
290 outputVector++;
291 n--;
292 }
293
294 // Now the inputVector address is aligned and start to apply SSE.
295 size_t group = n / 4;
296 __m128 mScalar = _mm_set_ps1(scalar);
297 __m128* pSource;
298 __m128* pDest;
299 __m128 dest;
300
301 bool destAligned = is16ByteAligned(outputVector);
302 if (destAligned) { // all aligned
303 while (group--) {
304 pSource = reinterpret_cast<__m128*>(const_cast<float*>(inputVector));
305 pDest = reinterpret_cast<__m128*>(outputVector);
306 *pDest = _mm_add_ps(*pSource, mScalar);
307
308 inputVector += 4;
309 outputVector += 4;
310 }
311 } else {
312 while (group--) {
313 pSource = reinterpret_cast<__m128*>(const_cast<float*>(inputVector));
314 dest = _mm_add_ps(*pSource, mScalar);
315 _mm_storeu_ps(outputVector, dest);
316
317 inputVector += 4;
318 outputVector += 4;
319 }
320 }
321
322 // Non-SSE handling for remaining frames which is less than 4.
323 n %= 4;
324#elif HAVE(ARM_NEON_INTRINSICS)
325 size_t tailFrames = n % 4;
326 const float* endP = outputVector + n - tailFrames;
327 float32x4_t scalarVector = vdupq_n_f32(scalar);
328
329 while (outputVector < endP) {
330 float32x4_t source = vld1q_f32(inputVector);
331 vst1q_f32(outputVector, vaddq_f32(source, scalarVector));
332
333 inputVector += 4;
334 outputVector += 4;
335 }
336 n = tailFrames;
337#endif
338 while (n--) {
339 *outputVector = *inputVector + scalar;
340 ++inputVector;
341 ++outputVector;
342 }
343}

345void add(const float* inputVector1, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess)
346{
347 size_t n = numberOfElementsToProcess;
348
349#if CPU(X86_SSE2)
350 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
351 while (!is16ByteAligned(inputVector1) && n) {
352 *outputVector = *inputVector1 + *inputVector2;
353 inputVector1++;
354 inputVector2++;
355 outputVector++;
356 n--;
357 }
358
359 // Now the inputVector1 address is aligned and start to apply SSE.
360 size_t group = n / 4;
361 __m128* pSource1;
362 __m128* pSource2;
363 __m128* pDest;
364 __m128 source2;
365 __m128 dest;
366
367 bool source2Aligned = is16ByteAligned(inputVector2);
368 bool destAligned = is16ByteAligned(outputVector);
369
370 if (source2Aligned && destAligned) { // all aligned
371 while (group--) {
372 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
373 pSource2 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector2));
374 pDest = reinterpret_cast<__m128*>(outputVector);
375 *pDest = _mm_add_ps(*pSource1, *pSource2);
376
377 inputVector1 += 4;
378 inputVector2 += 4;
379 outputVector += 4;
380 }
381
382 } else if (source2Aligned && !destAligned) { // source2 aligned but dest not aligned
383 while (group--) {
384 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
385 pSource2 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector2));
386 dest = _mm_add_ps(*pSource1, *pSource2);
387 _mm_storeu_ps(outputVector, dest);
388
389 inputVector1 += 4;
390 inputVector2 += 4;
391 outputVector += 4;
392 }
393
394 } else if (!source2Aligned && destAligned) { // source2 not aligned but dest aligned
395 while (group--) {
396 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
397 source2 = _mm_loadu_ps(inputVector2);
398 pDest = reinterpret_cast<__m128*>(outputVector);
399 *pDest = _mm_add_ps(*pSource1, source2);
400
401 inputVector1 += 4;
402 inputVector2 += 4;
403 outputVector += 4;
404 }
405 } else if (!source2Aligned && !destAligned) { // both source2 and dest not aligned
406 while (group--) {
407 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
408 source2 = _mm_loadu_ps(inputVector2);
409 dest = _mm_add_ps(*pSource1, source2);
410 _mm_storeu_ps(outputVector, dest);
411
412 inputVector1 += 4;
413 inputVector2 += 4;
414 outputVector += 4;
415 }
416 }
417
418 // Non-SSE handling for remaining frames which is less than 4.
419 n %= 4;
420#elif HAVE(ARM_NEON_INTRINSICS)
421 size_t tailFrames = n % 4;
422 const float* endP = outputVector + n - tailFrames;
423
424 while (outputVector < endP) {
425 float32x4_t source1 = vld1q_f32(inputVector1);
426 float32x4_t source2 = vld1q_f32(inputVector2);
427 vst1q_f32(outputVector, vaddq_f32(source1, source2));
428
429 inputVector1 += 4;
430 inputVector2 += 4;
431 outputVector += 4;
432 }
433 n = tailFrames;
434#endif
435 while (n--) {
436 *outputVector = *inputVector1 + *inputVector2;
437 ++inputVector1;
438 ++inputVector2;
439 ++outputVector;
440 }
441}

443void substract(const float* inputVector1, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess)
444{
445 size_t n = numberOfElementsToProcess;
446
447#if CPU(X86_SSE2)
448 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
449 while (!is16ByteAligned(inputVector1) && n) {
450 *outputVector = *inputVector1 - *inputVector2;
451 inputVector1++;
452 inputVector2++;
453 outputVector++;
454 n--;
455 }
456
457 // Now the inputVector1 address is aligned and start to apply SSE.
458 size_t group = n / 4;
459 __m128* pSource1;
460 __m128* pSource2;
461 __m128* pDest;
462 __m128 source2;
463 __m128 dest;
464
465 bool source2Aligned = is16ByteAligned(inputVector2);
466 bool destAligned = is16ByteAligned(outputVector);
467
468 if (source2Aligned && destAligned) { // all aligned
469 while (group--) {
470 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
471 pSource2 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector2));
472 pDest = reinterpret_cast<__m128*>(outputVector);
473 *pDest = _mm_sub_ps(*pSource1, *pSource2);
474
475 inputVector1 += 4;
476 inputVector2 += 4;
477 outputVector += 4;
478 }
479 } else if (source2Aligned && !destAligned) { // source2 aligned but dest not aligned
480 while (group--) {
481 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
482 pSource2 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector2));
483 dest = _mm_sub_ps(*pSource1, *pSource2);
484 _mm_storeu_ps(outputVector, dest);
485
486 inputVector1 += 4;
487 inputVector2 += 4;
488 outputVector += 4;
489 }
490 } else if (!source2Aligned && destAligned) { // source2 not aligned but dest aligned
491 while (group--) {
492 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
493 source2 = _mm_loadu_ps(inputVector2);
494 pDest = reinterpret_cast<__m128*>(outputVector);
495 *pDest = _mm_sub_ps(*pSource1, source2);
496
497 inputVector1 += 4;
498 inputVector2 += 4;
499 outputVector += 4;
500 }
501 } else if (!source2Aligned && !destAligned) { // both source2 and dest not aligned
502 while (group--) {
503 pSource1 = reinterpret_cast<__m128*>(const_cast<float*>(inputVector1));
504 source2 = _mm_loadu_ps(inputVector2);
505 dest = _mm_sub_ps(*pSource1, source2);
506 _mm_storeu_ps(outputVector, dest);
507
508 inputVector1 += 4;
509 inputVector2 += 4;
510 outputVector += 4;
511 }
512 }
513
514 // Non-SSE handling for remaining frames which is less than 4.
515 n %= 4;
516#elif HAVE(ARM_NEON_INTRINSICS)
517 size_t tailFrames = n % 4;
518 const float* endP = outputVector + n - tailFrames;
519
520 while (outputVector < endP) {
521 float32x4_t source1 = vld1q_f32(inputVector1);
522 float32x4_t source2 = vld1q_f32(inputVector2);
523 vst1q_f32(outputVector, vsubq_f32(source1, source2));
524
525 inputVector1 += 4;
526 inputVector2 += 4;
527 outputVector += 4;
528 }
529 n = tailFrames;
530#endif
531 while (n--) {
532 *outputVector = *inputVector1 - *inputVector2;
533 ++inputVector1;
534 ++inputVector2;
535 ++outputVector;
536 }
537}

539void interpolate(const float* inputVector1, float* inputVector2, float interpolationFactor, float* outputVector, size_t numberOfElementsToProcess)
540{
541 if (inputVector1 != outputVector)
542 memcpy(outputVector, inputVector1, numberOfElementsToProcess * sizeof(float));
543
544 // inputVector2[k] = inputVector2[k] - inputVector1[k]
545 substract(inputVector2, inputVector1, inputVector2, numberOfElementsToProcess);
546
547 // outputVector[k] = outputVector[k] + interpolationFactor * inputVector2[k]
548 // = inputVector1[k] + interpolationFactor * (inputVector2[k] - inputVector1[k]);
549 multiplyByScalarThenAddToOutput(inputVector2, interpolationFactor, outputVector, numberOfElementsToProcess);
550}

552void multiply(const float* inputVector1, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess)
553{
554 size_t n = numberOfElementsToProcess;
555
556#if CPU(X86_SSE2)
557 // If the inputVector1 address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
558 while (!is16ByteAligned(inputVector1) && n) {
559 *outputVector = *inputVector1 * *inputVector2;
560 inputVector1++;
561 inputVector2++;
562 outputVector++;
563 n--;
564 }
565
566 // Now the inputVector1 address aligned and start to apply SSE.
567 size_t tailFrames = n % 4;
568 const float* endP = outputVector + n - tailFrames;
569 __m128 pSource1;
570 __m128 pSource2;
571 __m128 dest;
572
573 bool source2Aligned = is16ByteAligned(inputVector2);
574 bool destAligned = is16ByteAligned(outputVector);
575
576#define SSE2_MULT(loadInstr, storeInstr) \
577 while (outputVector < endP) \
578 { \
579 pSource1 = _mm_load_ps(inputVector1); \
580 pSource2 = _mm_##loadInstr##_ps(inputVector2); \
581 dest = _mm_mul_ps(pSource1, pSource2); \
582 _mm_##storeInstr##_ps(outputVector, dest); \
583 inputVector1 += 4; \
584 inputVector2 += 4; \
585 outputVector += 4; \
586 }
587
588 if (source2Aligned && destAligned) // Both aligned.
589 SSE2_MULT(load, store)
590 else if (source2Aligned && !destAligned) // Source2 is aligned but dest not.
591 SSE2_MULT(load, storeu)
592 else if (!source2Aligned && destAligned) // Dest is aligned but source2 not.
593 SSE2_MULT(loadu, store)
594 else // Neither aligned.
595 SSE2_MULT(loadu, storeu)
596
597 n = tailFrames;
598#elif HAVE(ARM_NEON_INTRINSICS)
599 size_t tailFrames = n % 4;
600 const float* endP = outputVector + n - tailFrames;
601
602 while (outputVector < endP) {
603 float32x4_t source1 = vld1q_f32(inputVector1);
604 float32x4_t source2 = vld1q_f32(inputVector2);
605 vst1q_f32(outputVector, vmulq_f32(source1, source2));
606
607 inputVector1 += 4;
608 inputVector2 += 4;
609 outputVector += 4;
610 }
611 n = tailFrames;
612#endif
613 while (n--) {
614 *outputVector = *inputVector1 * *inputVector2;
615 ++inputVector1;
616 ++inputVector2;
617 ++outputVector;
618 }
619}

621void multiplyComplex(const float* realVector1, const float* imagVector1, const float* realVector2, const float* imag2P, float* realOutputVector, float* imagDestP, size_t numberOfElementsToProcess)
622{
623 unsigned i = 0;
624#if CPU(X86_SSE2)
625 // Only use the SSE optimization in the very common case that all addresses are 16-byte aligned.
626 // Otherwise, fall through to the scalar code below.
627 if (is16ByteAligned(realVector1) && is16ByteAligned(imagVector1) && is16ByteAligned(realVector2) && is16ByteAligned(imag2P) && is16ByteAligned(realOutputVector) && is16ByteAligned(imagDestP)) {
628 unsigned endSize = numberOfElementsToProcess - numberOfElementsToProcess % 4;
629 while (i < endSize) {
630 __m128 real1 = _mm_load_ps(realVector1 + i);
631 __m128 real2 = _mm_load_ps(realVector2 + i);
632 __m128 imag1 = _mm_load_ps(imagVector1 + i);
633 __m128 imag2 = _mm_load_ps(imag2P + i);
634 __m128 real = _mm_mul_ps(real1, real2);
635 real = _mm_sub_ps(real, _mm_mul_ps(imag1, imag2));
636 __m128 imag = _mm_mul_ps(real1, imag2);
637 imag = _mm_add_ps(imag, _mm_mul_ps(imag1, real2));
638 _mm_store_ps(realOutputVector + i, real);
639 _mm_store_ps(imagDestP + i, imag);
640 i += 4;
641 }
642 }
643#elif HAVE(ARM_NEON_INTRINSICS)
644 unsigned endSize = numberOfElementsToProcess - numberOfElementsToProcess % 4;
645 while (i < endSize) {
646 float32x4_t real1 = vld1q_f32(realVector1 + i);
647 float32x4_t real2 = vld1q_f32(realVector2 + i);
648 float32x4_t imag1 = vld1q_f32(imagVector1 + i);
649 float32x4_t imag2 = vld1q_f32(imag2P + i);
650
651 float32x4_t realResult = vmlsq_f32(vmulq_f32(real1, real2), imag1, imag2);
652 float32x4_t imagResult = vmlaq_f32(vmulq_f32(real1, imag2), imag1, real2);
653
654 vst1q_f32(realOutputVector + i, realResult);
655 vst1q_f32(imagDestP + i, imagResult);
656
657 i += 4;
658 }
659#endif
660 for (; i < numberOfElementsToProcess; ++i) {
661 // Read and compute result before storing them, in case the
662 // destination is the same as one of the sources.
663 realOutputVector[i] = realVector1[i] * realVector2[i] - imagVector1[i] * imag2P[i];
664 imagDestP[i] = realVector1[i] * imag2P[i] + imagVector1[i] * realVector2[i];
665 }
666}

668float sumOfSquares(const float* inputVector, size_t numberOfElementsToProcess)
669{
670 size_t n = numberOfElementsToProcess;
671 float sum = 0;
672
673#if CPU(X86_SSE2)
674 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
675 while (!is16ByteAligned(inputVector) && n) {
676 float sample = *inputVector;
677 sum += sample * sample;
678 inputVector++;
679 n--;
680 }
681
682 // Now the inputVector is aligned, use SSE.
683 size_t tailFrames = n % 4;
684 const float* endP = inputVector + n - tailFrames;
685 __m128 source;
686 __m128 mSum = _mm_setzero_ps();
687
688 while (inputVector < endP) {
689 source = _mm_load_ps(inputVector);
690 source = _mm_mul_ps(source, source);
691 mSum = _mm_add_ps(mSum, source);
692 inputVector += 4;
693 }
694
695 // Summarize the SSE results.
696 const float* groupSumP = reinterpret_cast<float*>(&mSum);
697 sum += groupSumP[0] + groupSumP[1] + groupSumP[2] + groupSumP[3];
698
699 n = tailFrames;
700#elif HAVE(ARM_NEON_INTRINSICS)
701 size_t tailFrames = n % 4;
702 const float* endP = inputVector + n - tailFrames;
703
704 float32x4_t fourSum = vdupq_n_f32(0);
705 while (inputVector < endP) {
706 float32x4_t source = vld1q_f32(inputVector);
707 fourSum = vmlaq_f32(fourSum, source, source);
708 inputVector += 4;
709 }
710 float32x2_t twoSum = vadd_f32(vget_low_f32(fourSum), vget_high_f32(fourSum));
711
712 float groupSum[2];
713 vst1_f32(groupSum, twoSum);
714 sum += groupSum[0] + groupSum[1];
715
716 n = tailFrames;
717#endif
718
719 while (n--) {
720 float sample = *inputVector;
721 sum += sample * sample;
722 ++inputVector;
723 }
724
725 return sum;
726}

728float maximumMagnitude(const float* inputVector, size_t numberOfElementsToProcess)
729{
730 size_t n = numberOfElementsToProcess;
731 float max = 0;
732
733#if CPU(X86_SSE2)
734 // If the inputVector address is not 16-byte aligned, the first several frames (at most three) should be processed separately.
735 while (!is16ByteAligned(inputVector) && n) {
736 max = std::max(max, std::abs(*inputVector));
737 inputVector++;
738 n--;
739 }
740
741 // Now the inputVector is aligned, use SSE.
742 size_t tailFrames = n % 4;
743 const float* endP = inputVector + n - tailFrames;
744 __m128 source;
745 __m128 mMax = _mm_setzero_ps();
746 int mask = 0x7FFFFFFF;
747 __m128 mMask = _mm_set1_ps(*reinterpret_cast<float*>(&mask));
748
749 while (inputVector < endP) {
750 source = _mm_load_ps(inputVector);
751 // Calculate the absolute value by anding source with mask, the sign bit is set to 0.
752 source = _mm_and_ps(source, mMask);
753 mMax = _mm_max_ps(mMax, source);
754 inputVector += 4;
755 }
756
757 // Get max from the SSE results.
758 const float* groupMaxP = reinterpret_cast<float*>(&mMax);
759 max = std::max(max, groupMaxP[0]);
760 max = std::max(max, groupMaxP[1]);
761 max = std::max(max, groupMaxP[2]);
762 max = std::max(max, groupMaxP[3]);
763
764 n = tailFrames;
765#elif HAVE(ARM_NEON_INTRINSICS)
766 size_t tailFrames = n % 4;
767 const float* endP = inputVector + n - tailFrames;
768
769 float32x4_t fourMax = vdupq_n_f32(0);
770 while (inputVector < endP) {
771 float32x4_t source = vld1q_f32(inputVector);
772 fourMax = vmaxq_f32(fourMax, vabsq_f32(source));
773 inputVector += 4;
774 }
775 float32x2_t twoMax = vmax_f32(vget_low_f32(fourMax), vget_high_f32(fourMax));
776
777 float groupMax[2];
778 vst1_f32(groupMax, twoMax);
779 max = std::max(groupMax[0], groupMax[1]);
780
781 n = tailFrames;
782#endif
783
784 while (n--) {
785 max = std::max(max, std::abs(*inputVector));
786 ++inputVector;
787 }
788
789 return max;
790}

792void clamp(const float* inputVector, float minimum, float maximum, float* outputVector, size_t numberOfElementsToProcess)
793{
794 size_t n = numberOfElementsToProcess;
795
796 // FIXME: Optimize for SSE2.
797#if HAVE(ARM_NEON_INTRINSICS)
798 size_t tailFrames = n % 4;
799 const float* endP = outputVector + n - tailFrames;
800
801 float32x4_t low = vdupq_n_f32(minimum);
802 float32x4_t high = vdupq_n_f32(maximum);
803 while (outputVector < endP) {
804 float32x4_t source = vld1q_f32(inputVector);
805 vst1q_f32(outputVector, vmaxq_f32(vminq_f32(source, high), low));
806 inputVector += 4;
807 outputVector += 4;
808 }
809 n = tailFrames;
810#endif
811 while (n--) {
812 *outputVector = std::clamp(*inputVector, minimum, maximum);
813 ++inputVector;
814 ++outputVector;
815 }
816}
