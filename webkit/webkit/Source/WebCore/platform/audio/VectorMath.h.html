26#pragma once
27
28// Defines the interface for several vector math functions whose implementation will ideally be optimized.
29
30namespace WebCore {
31
32namespace VectorMath {
33
34// Multiples inputVector by scalar then adds the result to outputVector (simplified vsma).
35// for (n = 0; n < numberOfElementsToProcess; ++n)
36// outputVector[n] += inputVector[n] * scalar;
37void multiplyByScalarThenAddToOutput(const float* inputVector, float scalar, float* outputVector, size_t numberOfElementsToProcess);
38
39// Adds a vector inputVector2 to the product of a scalar value and a single-precision vector inputVector1 (vsma).
40// for (n = 0; n < numberOfElementsToProcess; ++n)
41// outputVector[n] = inputVector1[n] * scalar + inputVector2[n];
42void multiplyByScalarThenAddToVector(const float* inputVector1, float scalar, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess);
43
44// Multiplies the sum of two vectors by a scalar value (vasm).
45void addVectorsThenMultiplyByScalar(const float* inputVector1, const float* inputVector2, float scalar, float* outputVector, size_t numberOfElementsToProcess);
46
47void multiplyByScalar(const float* inputVector, float scalar, float* outputVector, size_t numberOfElementsToProcess);
48void addScalar(const float* inputVector, float scalar, float* outputVector, size_t numberOfElementsToProcess);
49void add(const float* inputVector1, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess);
50void substract(const float* inputVector1, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess);
51
52// Finds the maximum magnitude of a float vector.
53float maximumMagnitude(const float* inputVector, size_t numberOfElementsToProcess);
54
55// Sums the squares of a float vector's elements (svesq).
56float sumOfSquares(const float* inputVector, size_t numberOfElementsToProcess);
57
58// For an element-by-element multiply of two float vectors.
59void multiply(const float* inputVector1, const float* inputVector2, float* outputVector, size_t numberOfElementsToProcess);
60
61// Multiplies two complex vectors (zvmul).
62void multiplyComplex(const float* realVector1, const float* imagVector1, const float* realVector2, const float* imagVector2, float* realOutputVector, float* imagOutputVector, size_t numberOfElementsToProcess);
63
64// Copies elements while clipping values to the threshold inputs.
65void clamp(const float* inputVector, float mininum, float maximum, float* outputVector, size_t numberOfElementsToProcess);
66
67void linearToDecibels(const float* inputVector, float* outputVector, size_t numberOfElementsToProcess);
68
69// Calculates the linear interpolation between the supplied single-precision vectors
70// for (n = 0; n < numberOfElementsToProcess; ++n)
71// outputVector[n] = inputVector1[n] + interpolationFactor * (inputVector2[n] - inputVector1[n]);
72// NOTE: Internal implementation may modify inputVector2.
73void interpolate(const float* inputVector1, float* inputVector2, float interpolationFactor, float* outputVector, size_t numberOfElementsToProcess);
74
75} // namespace VectorMath
76
77} // namespace WebCore
