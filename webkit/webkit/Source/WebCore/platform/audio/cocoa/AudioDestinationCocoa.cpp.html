44Ref<AudioDestination> AudioDestination::create(AudioIOCallback& callback, const String&, unsigned numberOfInputChannels, unsigned numberOfOutputChannels, float sampleRate)
45{
46 // FIXME: make use of inputDeviceId as appropriate.
47
48 // FIXME: Add support for local/live audio input.
49 if (numberOfInputChannels)
50 WTFLogAlways("AudioDestination::create(%u, %u, %f) - unhandled input channels", numberOfInputChannels, numberOfOutputChannels, sampleRate);
51
52 if (numberOfOutputChannels > AudioSession::sharedSession().maximumNumberOfOutputChannels())
53 WTFLogAlways("AudioDestination::create(%u, %u, %f) - unhandled output channels", numberOfInputChannels, numberOfOutputChannels, sampleRate);
54
55 if (AudioDestinationCocoa::createOverride)
56 return AudioDestinationCocoa::createOverride(callback, sampleRate);
57
58 auto destination = adoptRef(*new AudioDestinationCocoa(callback, numberOfOutputChannels, sampleRate));
59 return destination;
60}

72AudioDestinationCocoa::AudioDestinationCocoa(AudioIOCallback& callback, unsigned numberOfOutputChannels, float sampleRate, bool configureAudioOutputUnit)
73 : AudioDestination(callback)
74 , m_audioOutputUnitAdaptor(*this)
75 , m_outputBus(AudioBus::create(numberOfOutputChannels, AudioUtilities::renderQuantumSize, false).releaseNonNull())
76 , m_renderBus(AudioBus::create(numberOfOutputChannels, AudioUtilities::renderQuantumSize).releaseNonNull())
77 , m_fifo(makeUniqueRef<PushPullFIFO>(numberOfOutputChannels, fifoSize))
78 , m_contextSampleRate(sampleRate)
79{
80 if (configureAudioOutputUnit)
81 m_audioOutputUnitAdaptor.configure(hardwareSampleRate(), numberOfOutputChannels);
82
83 auto hardwareSampleRate = this->hardwareSampleRate();
84 if (sampleRate != hardwareSampleRate) {
85 double scaleFactor = static_cast<double>(sampleRate) / hardwareSampleRate;
86 m_resampler = makeUnique<MultiChannelResampler>(scaleFactor, numberOfOutputChannels, AudioUtilities::renderQuantumSize, [this](AudioBus* bus, size_t framesToProcess) {
87 ASSERT_UNUSED(framesToProcess, framesToProcess == AudioUtilities::renderQuantumSize);
88 callRenderCallback(nullptr, bus, AudioUtilities::renderQuantumSize, m_outputTimestamp);
89 });
90 }
91}

105void AudioDestinationCocoa::start(Function<void(Function<void()>&&)>&& dispatchToRenderThread, CompletionHandler<void(bool)>&& completionHandler)
106{
107 ASSERT(isMainThread());
108 LOG(Media, "AudioDestinationCocoa::start");
109 {
110 Locker locker { m_dispatchToRenderThreadLock };
111 m_dispatchToRenderThread = WTFMove(dispatchToRenderThread);
112 }
113 startRendering(WTFMove(completionHandler));
114}

116void AudioDestinationCocoa::startRendering(CompletionHandler<void(bool)>&& completionHandler)
117{
118 ASSERT(isMainThread());
119 auto success = m_audioOutputUnitAdaptor.start() == noErr;
120 if (success)
121 setIsPlaying(true);
122
123 callOnMainThread([completionHandler = WTFMove(completionHandler), success]() mutable {
124 completionHandler(success);
125 });
126}

167void AudioDestinationCocoa::getAudioStreamBasicDescription(AudioStreamBasicDescription& streamFormat)
168{
169 const int bytesPerFloat = sizeof(Float32);
170 const int bitsPerByte = 8;
171 streamFormat.mSampleRate = hardwareSampleRate();
172 streamFormat.mFormatID = kAudioFormatLinearPCM;
173 streamFormat.mFormatFlags = kAudioFormatFlagsNativeFloatPacked | kAudioFormatFlagIsNonInterleaved;
174 streamFormat.mBytesPerPacket = bytesPerFloat;
175 streamFormat.mFramesPerPacket = 1;
176 streamFormat.mBytesPerFrame = bytesPerFloat;
177 streamFormat.mChannelsPerFrame = numberOfOutputChannels();
178 streamFormat.mBitsPerChannel = bitsPerByte * bytesPerFloat;
179}

181static void assignAudioBuffersToBus(AudioBuffer* buffers, AudioBus& bus, UInt32 numberOfBuffers, UInt32 numberOfFrames, UInt32 frameOffset, UInt32 framesThisTime)
182{
183 for (UInt32 i = 0; i < numberOfBuffers; ++i) {
184 UInt32 bytesPerFrame = buffers[i].mDataByteSize / numberOfFrames;
185 UInt32 byteOffset = frameOffset * bytesPerFrame;
186 auto* memory = reinterpret_cast<float*>(reinterpret_cast<char*>(buffers[i].mData) + byteOffset);
187 bus.setChannelMemory(i, memory, framesThisTime);
188 }
189}

196// Pulls on our provider to get rendered audio stream.
197OSStatus AudioDestinationCocoa::render(double sampleTime, uint64_t hostTime, UInt32 numberOfFrames, AudioBufferList* ioData)
198{
199 ASSERT(!isMainThread());
200
201 if (!hasEnoughFrames(numberOfFrames))
202 return noErr;
203
204 m_outputTimestamp = {
205 Seconds { sampleTime / sampleRate() },
206 MonotonicTime::fromMachAbsoluteTime(hostTime)
207 };
208
209 auto* buffers = ioData->mBuffers;
210 auto numberOfBuffers = ioData->mNumberBuffers;
211
212 // Associate the destination data array with the output bus then fill the FIFO.
213 assignAudioBuffersToBus(buffers, m_outputBus.get(), numberOfBuffers, numberOfFrames, 0, numberOfFrames);
214 size_t framesToRender;
215
216 {
217 Locker locker { m_fifoLock };
218 framesToRender = m_fifo->pull(m_outputBus.ptr(), numberOfFrames);
219 }
220
221 // When there is a AudioWorklet, we do rendering on the AudioWorkletThread.
222 if (!m_dispatchToRenderThreadLock.tryLock())
223 return -1;
224
225 Locker locker { AdoptLock, m_dispatchToRenderThreadLock };
226 if (!m_dispatchToRenderThread)
227 renderOnRenderingTheadIfPlaying(framesToRender);
228 else {
229 m_dispatchToRenderThread([protectedThis = makeRef(*this), framesToRender]() mutable {
230 protectedThis->renderOnRenderingTheadIfPlaying(framesToRender);
231 });
232 }
233
234 return noErr;
235}
236
237void AudioDestinationCocoa::renderOnRenderingTheadIfPlaying(size_t framesToRender)
238{
239 if (m_isPlaying)
240 renderOnRenderingThead(framesToRender);
241}
242
243// This runs on the AudioWorkletThread when AudioWorklet is enabled, on the audio device's rendering thread otherwise.
244void AudioDestinationCocoa::renderOnRenderingThead(size_t framesToRender)
245{
246 for (size_t pushedFrames = 0; pushedFrames < framesToRender; pushedFrames += AudioUtilities::renderQuantumSize) {
247 if (m_resampler)
248 m_resampler->process(m_renderBus.ptr(), AudioUtilities::renderQuantumSize);
249 else
250 callRenderCallback(nullptr, m_renderBus.ptr(), AudioUtilities::renderQuantumSize, m_outputTimestamp);
251
252 Locker locker { m_fifoLock };
253 m_fifo->push(m_renderBus.ptr());
254 }
255}
