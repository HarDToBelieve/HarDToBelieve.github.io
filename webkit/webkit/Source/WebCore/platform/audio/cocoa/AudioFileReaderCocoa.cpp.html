48static WARN_UNUSED_RETURN AudioBufferList* tryCreateAudioBufferList(size_t numberOfBuffers)
49{
50 CheckedSize bufferListSize = sizeof(AudioBufferList) - sizeof(AudioBuffer);
51 bufferListSize += numberOfBuffers * sizeof(AudioBuffer);
52 if (bufferListSize.hasOverflowed())
53 return nullptr;
54
55 auto allocated = tryFastCalloc(1, bufferListSize);
56 AudioBufferList* bufferList;
57 if (!allocated.getValue(bufferList))
58 return nullptr;
59
60 bufferList->mNumberBuffers = numberOfBuffers;
61 return bufferList;
62}
63
64static inline void destroyAudioBufferList(AudioBufferList* bufferList)
65{
66 fastFree(bufferList);
67}
68
69static bool validateAudioBufferList(AudioBufferList* bufferList)
70{
71 if (!bufferList)
72 return false;
73
74 std::optional<unsigned> expectedDataSize;
75 const AudioBuffer* buffer = bufferList->mBuffers;
76 const AudioBuffer* bufferEnd = buffer + bufferList->mNumberBuffers;
77 for ( ; buffer < bufferEnd; ++buffer) {
78 if (!buffer->mData)
79 return false;
80
81 unsigned dataSize = buffer->mDataByteSize;
82 if (!expectedDataSize)
83 expectedDataSize = dataSize;
84 else if (*expectedDataSize != dataSize)
85 return false;
86 }
87 return true;
88}

90AudioFileReader::AudioFileReader(const void* data, size_t dataSize)
91 : m_data(data)
92 , m_dataSize(dataSize)
93{
94 if (PAL::AudioFileOpenWithCallbacks(this, readProc, 0, getSizeProc, 0, 0, &m_audioFileID) != noErr)
95 return;
96
97 if (PAL::ExtAudioFileWrapAudioFileID(m_audioFileID, false, &m_extAudioFileRef) != noErr)
98 m_extAudioFileRef = 0;
99}
100
101AudioFileReader::~AudioFileReader()
102{
103 if (m_extAudioFileRef)
104 PAL::ExtAudioFileDispose(m_extAudioFileRef);
105
106 m_extAudioFileRef = 0;
107
108 if (m_audioFileID)
109 PAL::AudioFileClose(m_audioFileID);
110
111 m_audioFileID = 0;
112}

114OSStatus AudioFileReader::readProc(void* clientData, SInt64 position, UInt32 requestCount, void* buffer, UInt32* actualCount)
115{
116 auto* audioFileReader = static_cast<AudioFileReader*>(clientData);
117
118 auto dataSize = audioFileReader->dataSize();
119 auto* data = audioFileReader->data();
120 size_t bytesToRead = 0;
121
122 if (static_cast<UInt64>(position) < dataSize) {
123 size_t bytesAvailable = dataSize - static_cast<size_t>(position);
124 bytesToRead = requestCount <= bytesAvailable ? requestCount : bytesAvailable;
125 memcpy(buffer, static_cast<const uint8_t*>(data) + position, bytesToRead);
126 }
127
128 if (actualCount)
129 *actualCount = bytesToRead;
130
131 return noErr;
132}

134SInt64 AudioFileReader::getSizeProc(void* clientData)
135{
136 return static_cast<AudioFileReader*>(clientData)->dataSize();
137}

139RefPtr<AudioBus> AudioFileReader::createBus(float sampleRate, bool mixToMono)
140{
141 if (!m_extAudioFileRef)
142 return nullptr;
143
144 // Get file's data format
145 UInt32 size = sizeof(m_fileDataFormat);
146 if (PAL::ExtAudioFileGetProperty(m_extAudioFileRef, kExtAudioFileProperty_FileDataFormat, &size, &m_fileDataFormat) != noErr)
147 return nullptr;
148
149 size_t numberOfChannels = m_fileDataFormat.mChannelsPerFrame;
150
151 // Number of frames
152 SInt64 numberOfFrames64 = 0;
153 size = sizeof(numberOfFrames64);
154 if (PAL::ExtAudioFileGetProperty(m_extAudioFileRef, kExtAudioFileProperty_FileLengthFrames, &size, &numberOfFrames64) != noErr || numberOfFrames64 <= 0)
155 return nullptr;
156
157 double fileSampleRate = m_fileDataFormat.mSampleRate;
158
159 // Make client format same number of channels as file format, but tweak a few things.
160 // Client format will be linear PCM (canonical), and potentially change sample-rate.
161 m_clientDataFormat = m_fileDataFormat;
162
163 const int bytesPerFloat = sizeof(Float32);
164 const int bitsPerByte = 8;
165 m_clientDataFormat.mFormatID = kAudioFormatLinearPCM;
166 m_clientDataFormat.mFormatFlags = kAudioFormatFlagsNativeFloatPacked | kAudioFormatFlagIsNonInterleaved;
167 m_clientDataFormat.mBytesPerPacket = bytesPerFloat;
168 m_clientDataFormat.mFramesPerPacket = 1;
169 m_clientDataFormat.mBytesPerFrame = bytesPerFloat;
170 m_clientDataFormat.mChannelsPerFrame = numberOfChannels;
171 m_clientDataFormat.mBitsPerChannel = bitsPerByte * bytesPerFloat;
172
173 if (sampleRate)
174 m_clientDataFormat.mSampleRate = sampleRate;
175
176 if (PAL::ExtAudioFileSetProperty(m_extAudioFileRef, kExtAudioFileProperty_ClientDataFormat, sizeof(AudioStreamBasicDescription), &m_clientDataFormat) != noErr)
177 return nullptr;
178
179 // Change numberOfFrames64 to destination sample-rate
180 numberOfFrames64 = numberOfFrames64 * (m_clientDataFormat.mSampleRate / fileSampleRate);
181 size_t numberOfFrames = static_cast<size_t>(numberOfFrames64);
182
183 size_t busChannelCount = mixToMono ? 1 : numberOfChannels;
184
185 // Create AudioBus where we'll put the PCM audio data
186 auto audioBus = AudioBus::create(busChannelCount, numberOfFrames);
187 audioBus->setSampleRate(narrowPrecisionToFloat(m_clientDataFormat.mSampleRate)); // save for later
188
189 // Only allocated in the mixToMono case; deallocated on destruction.
190 AudioFloatArray leftChannel;
191 AudioFloatArray rightChannel;
192
193 AudioBufferList* bufferList = tryCreateAudioBufferList(numberOfChannels);
194 if (!bufferList) {
195 RELEASE_LOG_FAULT(WebAudio, "tryCreateAudioBufferList(%ld) returned null", numberOfChannels);
196 return nullptr;
197 }
198 const size_t bufferSize = numberOfFrames * sizeof(float);
199
200 RELEASE_ASSERT(bufferList->mNumberBuffers == numberOfChannels);
201 if (mixToMono && numberOfChannels == 2) {
202 leftChannel.resize(numberOfFrames);
203 rightChannel.resize(numberOfFrames);
204
205 bufferList->mBuffers[0].mNumberChannels = 1;
206 bufferList->mBuffers[0].mDataByteSize = bufferSize;
207 bufferList->mBuffers[0].mData = leftChannel.data();
208
209 bufferList->mBuffers[1].mNumberChannels = 1;
210 bufferList->mBuffers[1].mDataByteSize = bufferSize;
211 bufferList->mBuffers[1].mData = rightChannel.data();
212 } else {
213 RELEASE_ASSERT(!mixToMono || numberOfChannels == 1);
214
215 // For True-stereo (numberOfChannels == 4)
216 for (size_t i = 0; i < numberOfChannels; ++i) {
217 audioBus->channel(i)->zero();
218 bufferList->mBuffers[i].mNumberChannels = 1;
219 bufferList->mBuffers[i].mDataByteSize = bufferSize;
220 bufferList->mBuffers[i].mData = audioBus->channel(i)->mutableData();
221 ASSERT(bufferList->mBuffers[i].mData);
222 }
223 }
224
225 if (!validateAudioBufferList(bufferList)) {
226 RELEASE_LOG_FAULT(WebAudio, "Generated buffer in AudioFileReader::createBus() did not pass validation");
227 ASSERT_NOT_REACHED();
228 destroyAudioBufferList(bufferList);
229 return nullptr;
230 }
231
232 // Read from the file (or in-memory version)
233 UInt32 framesToRead = numberOfFrames;
234 if (PAL::ExtAudioFileRead(m_extAudioFileRef, &framesToRead, bufferList) != noErr) {
235 destroyAudioBufferList(bufferList);
236 return nullptr;
237 }
238
239 if (mixToMono && numberOfChannels == 2) {
240 // Mix stereo down to mono
241 float* destL = audioBus->channel(0)->mutableData();
242 for (size_t i = 0; i < numberOfFrames; ++i)
243 destL[i] = 0.5f * (leftChannel[i] + rightChannel[i]);
244 }
245
246 // Cleanup
247 destroyAudioBufferList(bufferList);
248
249 return audioBus;
250}

252RefPtr<AudioBus> createBusFromInMemoryAudioFile(const void* data, size_t dataSize, bool mixToMono, float sampleRate)
253{
254 AudioFileReader reader(data, dataSize);
255 return reader.createBus(sampleRate, mixToMono);
256}
