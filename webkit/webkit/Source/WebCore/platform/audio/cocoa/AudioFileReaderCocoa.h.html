34#include <CoreAudio/CoreAudioTypes.h>

37using ExtAudioFileRef = struct OpaqueExtAudioFile*;
38using AudioFileID = struct OpaqueAudioFileID*;
39
40namespace WebCore {
41
42class AudioBus;
43
44// Wrapper class for AudioFile and ExtAudioFile CoreAudio APIs for reading files and in-memory versions of them...
45
46class AudioFileReader {
47public:
48 explicit AudioFileReader(const void* data, size_t dataSize);
49 ~AudioFileReader();
50
51 RefPtr<AudioBus> createBus(float sampleRate, bool mixToMono); // Returns nullptr on error
52
53 const void* data() const { return m_data; }
54 size_t dataSize() const { return m_dataSize; }
55
56private:
57 static OSStatus readProc(void* clientData, SInt64 position, UInt32 requestCount, void* buffer, UInt32* actualCount);
58 static SInt64 getSizeProc(void* clientData);
59
60 const void* m_data = { nullptr };
61 size_t m_dataSize = { 0 };
62
63 AudioFileID m_audioFileID = { nullptr };
64 ExtAudioFileRef m_extAudioFileRef = { nullptr };
65
66 AudioStreamBasicDescription m_fileDataFormat;
67 AudioStreamBasicDescription m_clientDataFormat;
68};
69
70}

72#endif // ENABLE(WEB_AUDIO)
