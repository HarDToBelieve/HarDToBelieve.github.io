33namespace WebCore {
34
35AudioOutputUnitAdaptor::AudioOutputUnitAdaptor(AudioUnitRenderer& renderer)
36 : m_outputUnit(0)
37 , m_audioUnitRenderer(renderer)
38{
39}
40
41AudioOutputUnitAdaptor::~AudioOutputUnitAdaptor()
42{
43 if (m_outputUnit)
44 PAL::AudioComponentInstanceDispose(m_outputUnit);
45}
46
47OSStatus AudioOutputUnitAdaptor::start()
48{
49 auto result = PAL::AudioOutputUnitStart(m_outputUnit);
50 if (result != noErr)
51 WTFLogAlways("ERROR: AudioOutputUnitStart() call failed with error code: %ld", static_cast<long>(result));
52 return result;
53}
54
55OSStatus AudioOutputUnitAdaptor::stop()
56{
57 return PAL::AudioOutputUnitStop(m_outputUnit);
58}
59
60// DefaultOutputUnit callback
61OSStatus AudioOutputUnitAdaptor::inputProc(void* userData, AudioUnitRenderActionFlags*, const AudioTimeStamp* timeStamp, UInt32 /*busNumber*/, UInt32 numberOfFrames, AudioBufferList* ioData)
62{
63 auto* adaptor = static_cast<AudioOutputUnitAdaptor*>(userData);
64 double sampleTime = 0.;
65 uint64_t hostTime = 0;
66 if (timeStamp) {
67 sampleTime = timeStamp->mSampleTime;
68 hostTime = timeStamp->mHostTime;
69 }
70
71 return adaptor->m_audioUnitRenderer.render(sampleTime, hostTime, numberOfFrames, ioData);
72}
