36class WEBCORE_EXPORT CAAudioStreamDescription final : public AudioStreamDescription {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 CAAudioStreamDescription();
40 CAAudioStreamDescription(const AudioStreamBasicDescription&);
41 CAAudioStreamDescription(double, uint32_t, PCMFormat, bool);
42 ~CAAudioStreamDescription();
43
44 const PlatformDescription& platformDescription() const final;
45
46 PCMFormat format() const final;
47
48 double sampleRate() const final;
49 bool isPCM() const final;
50 bool isInterleaved() const final;
51 bool isSignedInteger() const final;
52 bool isFloat() const final;
53 bool isNativeEndian() const final;
54
55 uint32_t numberOfInterleavedChannels() const final;
56 uint32_t numberOfChannelStreams() const final;
57 uint32_t numberOfChannels() const final;
58 uint32_t sampleWordSize() const final;
59 uint32_t bytesPerFrame() const;
60 uint32_t bytesPerPacket() const;
61 uint32_t formatFlags() const;
62
63 bool operator==(const AudioStreamBasicDescription& other) const;
64 bool operator!=(const AudioStreamBasicDescription& other) const;
65 bool operator==(const AudioStreamDescription& other) const;
66 bool operator!=(const AudioStreamDescription& other) const;
67
68 const AudioStreamBasicDescription& streamDescription() const;
69 AudioStreamBasicDescription& streamDescription();
70
71 template<class Encoder> void encode(Encoder&) const;
72 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, CAAudioStreamDescription&);
73
74private:
75 AudioStreamBasicDescription m_streamDescription;
76 mutable PlatformDescription m_platformDescription;
77 mutable PCMFormat m_format { None };
78};
79
80template<class Encoder>
81void CAAudioStreamDescription::encode(Encoder& encoder) const
82{
83 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(&m_streamDescription), sizeof(m_streamDescription), 1);
84}
85
86template<class Decoder>
87bool CAAudioStreamDescription::decode(Decoder& decoder, CAAudioStreamDescription& description)
88{
89 return decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(&description.m_streamDescription), sizeof(description.m_streamDescription), 1);
90}
91
92inline CAAudioStreamDescription toCAAudioStreamDescription(const AudioStreamDescription& description)
93{
94 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
95 return CAAudioStreamDescription(*WTF::get<const AudioStreamBasicDescription*>(description.platformDescription().description));
96}
