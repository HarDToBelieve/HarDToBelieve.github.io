63static CheckedSize computeCapacityBytes(const CAAudioStreamDescription& format, size_t frameCount)
64{
65 CheckedSize capacityBytes = format.bytesPerFrame();
66 capacityBytes *= frameCount;
67 return capacityBytes;
68}
69
70static CheckedSize computeSizeForBuffers(const CAAudioStreamDescription& format, size_t frameCount)
71{
72 auto sizeForBuffers = computeCapacityBytes(format, frameCount);
73 sizeForBuffers *= format.numberOfChannelStreams();
74 return sizeForBuffers;
75}

77UniqueRef<CARingBuffer> CARingBuffer::adoptStorage(UniqueRef<CARingBufferStorage>&& storage, const CAAudioStreamDescription& format, size_t frameCount)
78{
79 // Validate the parameters as they may be coming from an untrusted process.
80 auto expectedStorageSize = computeSizeForBuffers(format, frameCount);
81 if (expectedStorageSize.hasOverflowed()) {
82 RELEASE_LOG_FAULT(Media, "CARingBuffer::adoptStorage: Overflowed when trying to compute the storage size");
83 return makeUniqueRef<CARingBuffer>();
84 }
85 if (storage->size() < expectedStorageSize) {
86 RELEASE_LOG_FAULT(Media, "CARingBuffer::adoptStorage: Storage size is insufficient for format and frameCount");
87 return makeUniqueRef<CARingBuffer>();
88 }
89
90 auto ringBuffer = makeUniqueRef<CARingBuffer>(WTFMove(storage));
91 ringBuffer->initializeAfterAllocation(format, frameCount);
92 return ringBuffer;
93}

95void CARingBuffer::initializeAfterAllocation(const CAAudioStreamDescription& format, size_t frameCount)
96{
97 m_description = format;
98 m_channelCount = format.numberOfChannelStreams();
99 m_bytesPerFrame = format.bytesPerFrame();
100 m_frameCount = frameCount;
101 m_capacityBytes = computeCapacityBytes(format, frameCount);
102
103 m_pointers.resize(m_channelCount);
104 Byte* channelData = static_cast<Byte*>(m_buffers->data());
105
106 for (auto& pointer : m_pointers) {
107 pointer = channelData;
108 channelData += m_capacityBytes;
109 }
110
111 flush();
112}

114bool CARingBuffer::allocate(const CAAudioStreamDescription& format, size_t frameCount)
115{
116 deallocate();
117 frameCount = WTF::roundUpToPowerOfTwo(frameCount);
118
119 auto sizeForBuffers = computeSizeForBuffers(format, frameCount);
120 if (sizeForBuffers.hasOverflowed()) {
121 RELEASE_LOG_FAULT(Media, "CARingBuffer::allocate: Overflowed when trying to compute the storage size");
122 return false;
123 }
124
125 if (UNLIKELY(!m_buffers->allocate(sizeForBuffers, format, frameCount))) {
126 RELEASE_LOG_FAULT(Media, "CARingBuffer::allocate: Failed to allocate buffer of the requested size: %lu", sizeForBuffers.value());
127 return false;
128 }
129
130 initializeAfterAllocation(format, frameCount);
131 return true;
132}

143static void ZeroRange(Vector<Byte*>& pointers, size_t offset, size_t nbytes)
144{
145 for (auto& pointer : pointers)
146 memset(pointer + offset, 0, nbytes);
147}
148
149static void StoreABL(Vector<Byte*>& pointers, size_t destOffset, const AudioBufferList* list, size_t srcOffset, size_t nbytes)
150{
151 ASSERT(list->mNumberBuffers == pointers.size());
152 const AudioBuffer* src = list->mBuffers;
153 for (auto& pointer : pointers) {
154 if (srcOffset > src->mDataByteSize)
155 continue;
156 memcpy(pointer + destOffset, static_cast<Byte*>(src->mData) + srcOffset, std::min<size_t>(nbytes, src->mDataByteSize - srcOffset));
157 ++src;
158 }
159}

161static void FetchABL(AudioBufferList* list, size_t destOffset, Vector<Byte*>& pointers, size_t srcOffset, size_t nbytes, AudioStreamDescription::PCMFormat format, CARingBuffer::FetchMode mode)
162{
163 ASSERT(list->mNumberBuffers == pointers.size());
164 AudioBuffer* dest = list->mBuffers;
165 for (auto& pointer : pointers) {
166 if (destOffset > dest->mDataByteSize)
167 continue;
168
169 nbytes = std::min<size_t>(nbytes, dest->mDataByteSize - destOffset);
170 if (mode == CARingBuffer::Copy)
171 memcpy(static_cast<Byte*>(dest->mData) + destOffset, pointer + srcOffset, nbytes);
172 else {
173 switch (format) {
174 case AudioStreamDescription::Int16: {
175 int16_t* destination = static_cast<int16_t*>(dest->mData);
176 int16_t* source = reinterpret_cast<int16_t*>(pointer + srcOffset);
177 for (size_t i = 0; i < nbytes / sizeof(int16_t); i++)
178 destination[i] += source[i];
179 break;
180 }
181 case AudioStreamDescription::Int32: {
182 int32_t* destination = static_cast<int32_t*>(dest->mData);
183 vDSP_vaddi(destination, 1, reinterpret_cast<int32_t*>(pointer + srcOffset), 1, destination, 1, nbytes / sizeof(int32_t));
184 break;
185 }
186 case AudioStreamDescription::Float32: {
187 float* destination = static_cast<float*>(dest->mData);
188 vDSP_vadd(destination, 1, reinterpret_cast<float*>(pointer + srcOffset), 1, destination, 1, nbytes / sizeof(float));
189 break;
190 }
191 case AudioStreamDescription::Float64: {
192 double* destination = static_cast<double*>(dest->mData);
193 vDSP_vaddD(destination, 1, reinterpret_cast<double*>(pointer + srcOffset), 1, destination, 1, nbytes / sizeof(double));
194 break;
195 }
196 case AudioStreamDescription::None:
197 ASSERT_NOT_REACHED();
198 break;
199 }
200 }
201 ++dest;
202 }
203}

205inline void ZeroABL(AudioBufferList* list, size_t destOffset, size_t nbytes)
206{
207 int nBuffers = list->mNumberBuffers;
208 AudioBuffer* dest = list->mBuffers;
209 while (--nBuffers >= 0) {
210 if (destOffset > dest->mDataByteSize)
211 continue;
212 memset(static_cast<Byte*>(dest->mData) + destOffset, 0, std::min<size_t>(nbytes, dest->mDataByteSize - destOffset));
213 ++dest;
214 }

231void CARingBufferStorageVector::flush()
232{
233 Locker locker { m_currentFrameBoundsLock };
234 for (auto& timeBounds : m_timeBoundsQueue) {
235 timeBounds.m_startFrame = 0;
236 timeBounds.m_endFrame = 0;
237 timeBounds.m_updateCounter = 0;
238 }
239 m_timeBoundsQueuePtr = 0;
240}

242CARingBuffer::Error CARingBuffer::store(const AudioBufferList* list, size_t framesToWrite, uint64_t startFrame)
243{
244 if (!framesToWrite)
245 return Ok;
246
247 if (framesToWrite > m_frameCount)
248 return TooMuch;
249
250 uint64_t endFrame = startFrame + framesToWrite;
251
252 if (startFrame < currentEndFrame()) {
253 // Throw everything out when going backwards.
254 setCurrentFrameBounds(startFrame, startFrame);
255 } else if (endFrame - currentStartFrame() <= m_frameCount) {
256 // The buffer has not yet wrapped and will not need to.
257 // No-op.
258 } else {
259 // Advance the start time past the region we are about to overwrite
260 // starting one buffer of time behind where we're writing.
261 uint64_t newStartFrame = endFrame - m_frameCount;
262 uint64_t newEndFrame = std::max(newStartFrame, currentEndFrame());
263 setCurrentFrameBounds(newStartFrame, newEndFrame);
264 }
265
266 // Write the new frames.
267 size_t offset0;
268 size_t offset1;
269 uint64_t curEnd = currentEndFrame();
270
271 if (startFrame > curEnd) {
272 // We are skipping some samples, so zero the range we are skipping.
273 offset0 = frameOffset(curEnd);
274 offset1 = frameOffset(startFrame);
275 if (offset0 < offset1)
276 ZeroRange(m_pointers, offset0, offset1 - offset0);
277 else {
278 ZeroRange(m_pointers, offset0, m_capacityBytes - offset0);
279 ZeroRange(m_pointers, 0, offset1);
280 }
281 offset0 = offset1;
282 } else
283 offset0 = frameOffset(startFrame);
284
285 offset1 = frameOffset(endFrame);
286 if (offset0 < offset1)
287 StoreABL(m_pointers, offset0, list, 0, offset1 - offset0);
288 else {
289 size_t nbytes = m_capacityBytes - offset0;
290 StoreABL(m_pointers, offset0, list, 0, nbytes);
291 StoreABL(m_pointers, 0, list, nbytes, offset1);
292 }
293
294 // Now update the end time.
295 setCurrentFrameBounds(currentStartFrame(), endFrame);
296
297 return Ok;
298}

305void CARingBufferStorageVector::setCurrentFrameBounds(uint64_t startTime, uint64_t endTime)
306{
307 Locker locker { m_currentFrameBoundsLock };
308 uint32_t nextPtr = m_timeBoundsQueuePtr.load() + 1;
309 uint32_t index = nextPtr & kGeneralRingTimeBoundsQueueMask;
310
311 m_timeBoundsQueue[index].m_startFrame = startTime;
312 m_timeBoundsQueue[index].m_endFrame = endTime;
313 m_timeBoundsQueue[index].m_updateCounter = nextPtr;
314 m_timeBoundsQueuePtr++;
315}

338void CARingBuffer::clipTimeBounds(uint64_t& startRead, uint64_t& endRead)
339{
340 uint64_t startTime;
341 uint64_t endTime;
342
343 getCurrentFrameBoundsWithoutUpdate(startTime, endTime);
344
345 if (startRead > endTime || endRead < startTime) {
346 endRead = startRead;
347 return;
348 }
349
350 startRead = std::max(startRead, startTime);
351 endRead = std::min(endRead, endTime);
352 endRead = std::max(endRead, startRead);
353}

382bool CARingBuffer::fetchIfHasEnoughData(AudioBufferList* list, size_t frameCount, uint64_t startFrame, FetchMode mode)
383{
384 // When the RingBuffer is backed by shared memory, getCurrentFrameBounds() makes sure we pull frame bounds from shared memory before fetching.
385 uint64_t start, end;
386 getCurrentFrameBounds(start, end);
387 if (startFrame < start || startFrame + frameCount > end)
388 return false;
389
390 fetchInternal(list, frameCount, startFrame, mode);
391 return true;
392}

398 fetchInternal(list, frameCount, startRead, mode);

401void CARingBuffer::fetchInternal(AudioBufferList* list, size_t nFrames, uint64_t startRead, FetchMode mode)
402{
403 if (!nFrames)
404 return;
405
406 startRead = std::max<uint64_t>(0, startRead);
407
408 uint64_t endRead = startRead + nFrames;
409
410 uint64_t startRead0 = startRead;
411 uint64_t endRead0 = endRead;
412
413 clipTimeBounds(startRead, endRead);
414
415 if (startRead == endRead) {
416 ZeroABL(list, 0, nFrames * m_bytesPerFrame);
417 return;
418 }
419
420 size_t byteSize = static_cast<size_t>((endRead - startRead) * m_bytesPerFrame);
421
422 size_t destStartByteOffset = static_cast<size_t>(std::max<uint64_t>(0, (startRead - startRead0) * m_bytesPerFrame));
423
424 if (destStartByteOffset > 0)
425 ZeroABL(list, 0, std::min<size_t>(nFrames * m_bytesPerFrame, destStartByteOffset));
426
427 size_t destEndSize = static_cast<size_t>(std::max<uint64_t>(0, endRead0 - endRead));
428 if (destEndSize > 0)
429 ZeroABL(list, destStartByteOffset + byteSize, destEndSize * m_bytesPerFrame);
430
431 size_t offset0 = frameOffset(startRead);
432 size_t offset1 = frameOffset(endRead);
433 size_t nbytes;
434
435 if (offset0 < offset1) {
436 nbytes = offset1 - offset0;
437 FetchABL(list, destStartByteOffset, m_pointers, offset0, nbytes, m_description.format(), mode);
438 } else {
439 nbytes = m_capacityBytes - offset0;
440 FetchABL(list, destStartByteOffset, m_pointers, offset0, nbytes, m_description.format(), mode);
441 if (offset1)
442 FetchABL(list, destStartByteOffset + nbytes, m_pointers, 0, offset1, m_description.format(), mode);
443 nbytes += offset1;
444 }
445
446 int channelCount = list->mNumberBuffers;
447 AudioBuffer* dest = list->mBuffers;
448 while (--channelCount >= 0) {
449 dest->mDataByteSize = nbytes;
450 dest++;
451 }
452}
