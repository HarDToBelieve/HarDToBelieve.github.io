82void MediaSessionManagerCocoa::updateSessionState()
83{
84 int videoCount = 0;
85 int videoAudioCount = 0;
86 int audioCount = 0;
87 int webAudioCount = 0;
88 int captureCount = countActiveAudioCaptureSources();
89 bool hasAudibleAudioOrVideoMediaType = false;
90 forEachSession([&] (auto& session) mutable {
91 auto type = session.mediaType();
92 switch (type) {
93 case PlatformMediaSession::MediaType::None:
94 break;
95 case PlatformMediaSession::MediaType::Video:
96 ++videoCount;
97 break;
98 case PlatformMediaSession::MediaType::VideoAudio:
99 ++videoAudioCount;
100 break;
101 case PlatformMediaSession::MediaType::Audio:
102 ++audioCount;
103 break;
104 case PlatformMediaSession::MediaType::WebAudio:
105 if (session.canProduceAudio())
106 ++webAudioCount;
107 break;
108 }
109
110 if (!hasAudibleAudioOrVideoMediaType) {
111 if ((type == PlatformMediaSession::MediaType::VideoAudio || type == PlatformMediaSession::MediaType::Audio) && session.canProduceAudio() && session.hasPlayedSinceLastInterruption())
112 hasAudibleAudioOrVideoMediaType = true;
113 if (session.isPlayingToWirelessPlaybackTarget())
114 hasAudibleAudioOrVideoMediaType = true;
115 }
116 });
117
118 ALWAYS_LOG(LOGIDENTIFIER, "types: "
119 "AudioCapture(", captureCount, "), "
120 "Video(", videoCount, "), "
121 "Audio(", audioCount, "), "
122 "VideoAudio(", videoAudioCount, "), "
123 "WebAudio(", webAudioCount, ")");
124
125 size_t bufferSize = m_defaultBufferSize;
126 if (webAudioCount)
127 bufferSize = AudioUtilities::renderQuantumSize;
128 else if (captureCount) {
129 // In case of audio capture, we want to grab 20 ms chunks to limit the latency so that it is not noticeable by users
130 // while having a large enough buffer so that the audio rendering remains stable, hence a computation based on sample rate.
131 bufferSize = AudioSession::sharedSession().sampleRate() / 50;
132 } else if (m_supportedAudioHardwareBufferSizes && DeprecatedGlobalSettings::lowPowerVideoAudioBufferSizeEnabled())
133 bufferSize = m_supportedAudioHardwareBufferSizes.nearest(kLowPowerVideoBufferSize);
134
135 AudioSession::sharedSession().setPreferredBufferSize(bufferSize);
136
137 if (!DeprecatedGlobalSettings::shouldManageAudioSessionCategory())
138 return;
139
140 RouteSharingPolicy policy = RouteSharingPolicy::Default;
141 auto category = AudioSession::CategoryType::None;
142 if (captureCount)
143 category = AudioSession::CategoryType::PlayAndRecord;
144 else if (hasAudibleAudioOrVideoMediaType) {
145 category = AudioSession::CategoryType::MediaPlayback;
146 policy = RouteSharingPolicy::LongFormAudio;
147 } else if (webAudioCount)
148 category = AudioSession::CategoryType::AmbientSound;
149
150 ALWAYS_LOG(LOGIDENTIFIER, "setting category = ", category, ", policy = ", policy);
151 AudioSession::sharedSession().setCategory(category, policy);
152}

154void MediaSessionManagerCocoa::beginInterruption(PlatformMediaSession::InterruptionType type)
155{
156 if (type == PlatformMediaSession::InterruptionType::SystemInterruption) {
157 forEachSession([] (auto& session) {

159 });
160 }
161
162 PlatformMediaSessionManager::beginInterruption(type);
163}

170void MediaSessionManagerCocoa::scheduleSessionStatusUpdate()
171{
172 callOnMainThread([this] () mutable {
173 m_nowPlayingManager->setSupportsSeeking(computeSupportsSeeking());
174 updateNowPlayingInfo();
175
176 forEachSession([] (auto& session) {
177 session.updateMediaUsageIfChanged();
178 });
179 });
180}

182bool MediaSessionManagerCocoa::sessionWillBeginPlayback(PlatformMediaSession& session)
183{
184 if (!PlatformMediaSessionManager::sessionWillBeginPlayback(session))
185 return false;
186
187 scheduleSessionStatusUpdate();
188 return true;
189}

196void MediaSessionManagerCocoa::addSession(PlatformMediaSession& session)
197{
198 m_nowPlayingManager->addClient(*this);
199
200 if (!m_audioHardwareListener) {
201 m_audioHardwareListener = AudioHardwareListener::create(*this);
202 m_supportedAudioHardwareBufferSizes = m_audioHardwareListener->supportedBufferSizes();
203 }
204
205 PlatformMediaSessionManager::addSession(session);
206}
207
208void MediaSessionManagerCocoa::removeSession(PlatformMediaSession& session)
209{
210 PlatformMediaSessionManager::removeSession(session);
211
212 if (hasNoSession()) {
213 m_nowPlayingManager->removeClient(*this);
214 m_audioHardwareListener = nullptr;
215 }
216
217 scheduleSessionStatusUpdate();
218}

227void MediaSessionManagerCocoa::sessionWillEndPlayback(PlatformMediaSession& session, DelayCallingUpdateNowPlaying delayCallingUpdateNowPlaying)
228{
229 PlatformMediaSessionManager::sessionWillEndPlayback(session, delayCallingUpdateNowPlaying);
230
231 callOnMainThread([weakSession = makeWeakPtr(session)] {
232 if (weakSession)
233 weakSession->updateMediaUsageIfChanged();
234 });
235
236 if (delayCallingUpdateNowPlaying == DelayCallingUpdateNowPlaying::No)
237 updateNowPlayingInfo();
238 else {
239 callOnMainThread([this] {
240 updateNowPlayingInfo();
241 });
242 }
243}

268void MediaSessionManagerCocoa::clearNowPlayingInfo()
269{
270 if (!isMediaRemoteFrameworkAvailable())
271 return;
272
273 if (canLoad_MediaRemote_MRMediaRemoteSetNowPlayingVisibility())
274 MRMediaRemoteSetNowPlayingVisibility(MRMediaRemoteGetLocalOrigin(), MRNowPlayingClientVisibilityNeverVisible);
275
276 MRMediaRemoteSetCanBeNowPlayingApplication(false);
277 MRMediaRemoteSetNowPlayingInfo(nullptr);
278 MRMediaRemoteSetNowPlayingApplicationPlaybackStateForOrigin(MRMediaRemoteGetLocalOrigin(), kMRPlaybackStateStopped, dispatch_get_main_queue(), ^(MRMediaRemoteError error) {
279#if LOG_DISABLED
280 UNUSED_PARAM(error);
281#else
282 if (error)
283 WTFLogAlways("MRMediaRemoteSetNowPlayingApplicationPlaybackStateForOrigin(stopped) failed with error %d", error);
284#endif
285 });
286}

288void MediaSessionManagerCocoa::setNowPlayingInfo(bool setAsNowPlayingApplication, const NowPlayingInfo& nowPlayingInfo)
289{
290 if (!isMediaRemoteFrameworkAvailable())
291 return;
292
293 if (setAsNowPlayingApplication)
294 MRMediaRemoteSetCanBeNowPlayingApplication(true);
295
296 auto info = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 4, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
297
298 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoArtist, nowPlayingInfo.artist.createCFString().get());
299 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoAlbum, nowPlayingInfo.album.createCFString().get());
300 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoTitle, nowPlayingInfo.title.createCFString().get());
301
302 if (std::isfinite(nowPlayingInfo.duration) && nowPlayingInfo.duration != MediaPlayer::invalidTime()) {
303 auto cfDuration = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberDoubleType, &nowPlayingInfo.duration));
304 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoDuration, cfDuration.get());
305 }
306
307 double rate = nowPlayingInfo.isPlaying ? 1 : 0;
308 auto cfRate = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberDoubleType, &rate));
309 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoPlaybackRate, cfRate.get());
310
311 // FIXME: This is a workaround Control Center not updating the artwork when refreshed.
312 // We force the identifier to be reloaded to the new artwork if available.
313 auto lastUpdatedNowPlayingInfoUniqueIdentifier = nowPlayingInfo.artwork ? nowPlayingInfo.artwork->src.hash() : nowPlayingInfo.uniqueIdentifier.toUInt64();
314 auto cfIdentifier = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberLongLongType, &lastUpdatedNowPlayingInfoUniqueIdentifier));
315 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoUniqueIdentifier, cfIdentifier.get());
316
317 if (std::isfinite(nowPlayingInfo.currentTime) && nowPlayingInfo.currentTime != MediaPlayer::invalidTime() && nowPlayingInfo.supportsSeeking) {
318 auto cfCurrentTime = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberDoubleType, &nowPlayingInfo.currentTime));
319 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoElapsedTime, cfCurrentTime.get());
320 }
321 if (nowPlayingInfo.artwork && nowPlayingInfo.artwork->imageData) {
322 auto nsArtwork = nowPlayingInfo.artwork->imageData->createNSData();
323 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoArtworkData, nsArtwork.get());
324 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoArtworkMIMEType, nowPlayingInfo.artwork->mimeType.createCFString().get());
325 CFDictionarySetValue(info.get(), kMRMediaRemoteNowPlayingInfoArtworkIdentifier, String::number(nowPlayingInfo.artwork->src.hash()).createCFString().get());
326 }
327
328 if (canLoad_MediaRemote_MRMediaRemoteSetParentApplication() && !nowPlayingInfo.sourceApplicationIdentifier.isEmpty())
329 MRMediaRemoteSetParentApplication(MRMediaRemoteGetLocalOrigin(), nowPlayingInfo.sourceApplicationIdentifier.createCFString().get());
330
331 MRPlaybackState playbackState = (nowPlayingInfo.isPlaying) ? kMRPlaybackStatePlaying : kMRPlaybackStatePaused;
332 MRMediaRemoteSetNowPlayingApplicationPlaybackStateForOrigin(MRMediaRemoteGetLocalOrigin(), playbackState, dispatch_get_main_queue(), ^(MRMediaRemoteError error) {
333#if LOG_DISABLED
334 UNUSED_PARAM(error);
335#else
336 if (error)
337 WTFLogAlways("MRMediaRemoteSetNowPlayingApplicationPlaybackStateForOrigin(playing) failed with error %d", error);
338#endif
339 });
340 MRMediaRemoteSetNowPlayingInfoWithMergePolicy(info.get(), MRMediaRemoteMergePolicyReplace);
341
342 if (canLoad_MediaRemote_MRMediaRemoteSetNowPlayingVisibility()) {
343 MRNowPlayingClientVisibility visibility = nowPlayingInfo.allowsNowPlayingControlsVisibility ? MRNowPlayingClientVisibilityAlwaysVisible : MRNowPlayingClientVisibilityNeverVisible;
344 MRMediaRemoteSetNowPlayingVisibility(MRMediaRemoteGetLocalOrigin(), visibility);
345 }
346}

357void MediaSessionManagerCocoa::updateNowPlayingInfo()
358{
359 if (!isMediaRemoteFrameworkAvailable())
360 return;
361
362 BEGIN_BLOCK_OBJC_EXCEPTIONS
363
364 std::optional<NowPlayingInfo> nowPlayingInfo;
365 if (auto* session = nowPlayingEligibleSession())
366 nowPlayingInfo = session->nowPlayingInfo();
367
368 if (!nowPlayingInfo) {
369
370 if (m_registeredAsNowPlayingApplication) {
371 ALWAYS_LOG(LOGIDENTIFIER, "clearing now playing info");
372 m_nowPlayingManager->clearNowPlayingInfo();
373 }
374
375 m_registeredAsNowPlayingApplication = false;
376 m_nowPlayingActive = false;
377 m_lastUpdatedNowPlayingTitle = emptyString();
378 m_lastUpdatedNowPlayingDuration = NAN;
379 m_lastUpdatedNowPlayingElapsedTime = NAN;
380 m_lastUpdatedNowPlayingInfoUniqueIdentifier = { };
381
382 return;
383 }
384
385 m_haveEverRegisteredAsNowPlayingApplication = true;
386
387 if (m_nowPlayingManager->setNowPlayingInfo(*nowPlayingInfo))
388 ALWAYS_LOG(LOGIDENTIFIER, "title = \"", nowPlayingInfo->title, "\", isPlaying = ", nowPlayingInfo->isPlaying, ", duration = ", nowPlayingInfo->duration, ", now = ", nowPlayingInfo->currentTime, ", id = ", nowPlayingInfo->uniqueIdentifier.toUInt64(), ", registered = ", m_registeredAsNowPlayingApplication, ", src = \"", nowPlayingInfo->artwork ? nowPlayingInfo->artwork->src : String(), "\"");
389
390 if (!m_registeredAsNowPlayingApplication) {
391 m_registeredAsNowPlayingApplication = true;
392 providePresentingApplicationPIDIfNecessary();
393 }
394
395 if (!nowPlayingInfo->title.isEmpty())
396 m_lastUpdatedNowPlayingTitle = nowPlayingInfo->title;
397
398 double duration = nowPlayingInfo->duration;
399 if (std::isfinite(duration) && duration != MediaPlayer::invalidTime())
400 m_lastUpdatedNowPlayingDuration = duration;
401
402 m_lastUpdatedNowPlayingInfoUniqueIdentifier = nowPlayingInfo->uniqueIdentifier;
403
404 double currentTime = nowPlayingInfo->currentTime;
405 if (std::isfinite(currentTime) && currentTime != MediaPlayer::invalidTime() && nowPlayingInfo->supportsSeeking)
406 m_lastUpdatedNowPlayingElapsedTime = currentTime;
407
408 m_nowPlayingActive = nowPlayingInfo->allowsNowPlayingControlsVisibility;
409
410 END_BLOCK_OBJC_EXCEPTIONS
411}

413void MediaSessionManagerCocoa::audioOutputDeviceChanged()
414{
415 ASSERT(m_audioHardwareListener);
416 m_supportedAudioHardwareBufferSizes = m_audioHardwareListener->supportedBufferSizes();
417 m_defaultBufferSize = AudioSession::sharedSession().preferredBufferSize();
418 AudioSession::sharedSession().audioOutputDeviceChanged();
419 updateSessionState();
420}
