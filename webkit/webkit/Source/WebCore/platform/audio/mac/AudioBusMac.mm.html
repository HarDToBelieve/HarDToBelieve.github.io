41RefPtr<AudioBus> AudioBus::loadPlatformResource(const char* name, float sampleRate)
42{
43 @autoreleasepool {
44 NSBundle *bundle = [NSBundle bundleForClass:[WebCoreAudioBundleClass class]];
45 NSURL *audioFileURL = [bundle URLForResource:[NSString stringWithUTF8String:name] withExtension:@"wav" subdirectory:@"audio"];
46 if (NSData *audioData = [NSData dataWithContentsOfURL:audioFileURL options:NSDataReadingMappedIfSafe error:nil])
47 return createBusFromInMemoryAudioFile([audioData bytes], [audioData length], false, sampleRate);
48 }
49
50 ASSERT_NOT_REACHED();
51 return nullptr;
52}
