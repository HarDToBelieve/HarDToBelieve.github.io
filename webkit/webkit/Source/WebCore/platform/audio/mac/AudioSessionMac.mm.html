43static AudioDeviceID defaultDevice()
44{
45 AudioDeviceID deviceID = kAudioDeviceUnknown;
46 UInt32 infoSize = sizeof(deviceID);
47
48 AudioObjectPropertyAddress defaultOutputDeviceAddress = {
49 kAudioHardwarePropertyDefaultOutputDevice,
50 kAudioObjectPropertyScopeGlobal,
51#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
52 kAudioObjectPropertyElementMain
53#else
54 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
55 kAudioObjectPropertyElementMaster
56 ALLOW_DEPRECATED_DECLARATIONS_END
57#endif
58 };
59 OSStatus result = AudioObjectGetPropertyData(kAudioObjectSystemObject, &defaultOutputDeviceAddress, 0, 0, &infoSize, (void*)&deviceID);
60 if (result)
61 return 0; // error
62 return deviceID;
63}

68static float defaultDeviceTransportIsBluetooth()

70 if (isPlayingToBluetoothOverride)

74 kAudioDevicePropertyTransportType,

86 if (AudioObjectGetPropertyData(defaultDevice(), &audioDeviceTransportTypeProperty, 0, 0, &transportSize, &transportType))
87 return false;
88
89 return transportType == kAudioDeviceTransportTypeBluetooth || transportType == kAudioDeviceTransportTypeBluetoothLE;

113OSStatus AudioSessionMac::handleSampleRateChange(AudioObjectID device, UInt32, const AudioObjectPropertyAddress* sampleRateAddress, void* inClientData)
114{
115 ASSERT(inClientData);
116 if (!inClientData)
117 return noErr;
118
119 auto* session = static_cast<AudioSessionMac*>(inClientData);
120
121 Float64 nominalSampleRate;
122 UInt32 nominalSampleRateSize = sizeof(Float64);
123 OSStatus result = AudioObjectGetPropertyData(device, sampleRateAddress, 0, 0, &nominalSampleRateSize, (void*)&nominalSampleRate);
124 if (result)
125 return result;
126
127 session->m_sampleRate = narrowPrecisionToFloat(nominalSampleRate);
128 return noErr;
129}

188void AudioSessionMac::setCategory(CategoryType category, RouteSharingPolicy)
189{
190#if ENABLE(ROUTING_ARBITRATION)
191 bool playingToBluetooth = defaultDeviceTransportIsBluetooth();
192 if (category == m_category && m_playingToBluetooth && *m_playingToBluetooth == playingToBluetooth)
193 return;
194
195 m_category = category;
196
197 if (m_setupArbitrationOngoing) {
198 RELEASE_LOG_ERROR(Media, "AudioSessionMac::setCategory() - a beginArbitrationWithCategory is still ongoing");
199 return;
200 }
201
202 if (!m_routingArbitrationClient)
203 return;
204
205 if (m_inRoutingArbitration) {
206 m_inRoutingArbitration = false;
207 m_routingArbitrationClient->leaveRoutingAbritration();
208 }
209
210 if (category == CategoryType::AmbientSound || category == CategoryType::SoloAmbientSound || category == CategoryType::AudioProcessing || category == CategoryType::None)
211 return;
212
213 using RoutingArbitrationError = AudioSessionRoutingArbitrationClient::RoutingArbitrationError;
214 using DefaultRouteChanged = AudioSessionRoutingArbitrationClient::DefaultRouteChanged;
215
216 m_playingToBluetooth = playingToBluetooth;
217 m_setupArbitrationOngoing = true;
218 m_routingArbitrationClient->beginRoutingArbitrationWithCategory(m_category, [this] (RoutingArbitrationError error, DefaultRouteChanged defaultRouteChanged) {
219 m_setupArbitrationOngoing = false;
220 if (error != RoutingArbitrationError::None) {
221 RELEASE_LOG_ERROR(Media, "AudioSessionMac::setCategory() - beginArbitrationWithCategory:%s failed with error %s", convertEnumerationToString(m_category).ascii().data(), convertEnumerationToString(error).ascii().data());
222 return;
223 }
224
225 m_inRoutingArbitration = true;
226
227 // FIXME: Do we need to reset sample rate and buffer size for the new default device?
228 if (defaultRouteChanged == DefaultRouteChanged::Yes)
229 LOG(Media, "AudioSessionMac::setCategory() - defaultRouteChanged!");
230 });
231#else
232 m_category = category;
233#endif
234}

245float AudioSessionMac::sampleRate() const
246{
247 if (!m_sampleRate) {
248 addSampleRateObserverIfNeeded();
249
250 Float64 nominalSampleRate;
251 UInt32 nominalSampleRateSize = sizeof(Float64);
252
253 AudioObjectPropertyAddress nominalSampleRateAddress = {
254 kAudioDevicePropertyNominalSampleRate,
255 kAudioObjectPropertyScopeGlobal,
256#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
257 kAudioObjectPropertyElementMain
258#else
259 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
260 kAudioObjectPropertyElementMaster
261 ALLOW_DEPRECATED_DECLARATIONS_END
262#endif
263 };
264 OSStatus result = AudioObjectGetPropertyData(defaultDevice(), &nominalSampleRateAddress, 0, 0, &nominalSampleRateSize, (void*)&nominalSampleRate);
265 if (result != noErr) {
266 RELEASE_LOG_ERROR(Media, "AudioSessionMac::sampleRate() - AudioObjectGetPropertyData() failed with error %d", result);
267 return 44100;
268 }
269
270 m_sampleRate = narrowPrecisionToFloat(nominalSampleRate);
271 if (!*m_sampleRate) {
272 RELEASE_LOG_ERROR(Media, "AudioSessionMac::sampleRate() - AudioObjectGetPropertyData() return an invalid sample rate");
273 m_sampleRate = 44100;
274 }
275 }
276 return *m_sampleRate;
277}

316size_t AudioSessionMac::maximumNumberOfOutputChannels() const
317{
318 AudioObjectPropertyAddress sizeAddress = {
319 kAudioDevicePropertyStreamConfiguration,
320 kAudioObjectPropertyScopeOutput,
321#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
322 kAudioObjectPropertyElementMain
323#else
324 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
325 kAudioObjectPropertyElementMaster
326 ALLOW_DEPRECATED_DECLARATIONS_END
327#endif
328 };
329
330 UInt32 size = 0;
331 OSStatus result = AudioObjectGetPropertyDataSize(defaultDevice(), &sizeAddress, 0, 0, &size);
332 if (result || !size)
333 return 0;
334
335 auto listMemory = makeUniqueArray<uint8_t>(size);
336 auto* audioBufferList = reinterpret_cast<AudioBufferList*>(listMemory.get());
337
338 result = AudioObjectGetPropertyData(defaultDevice(), &sizeAddress, 0, 0, &size, audioBufferList);
339 if (result)
340 return 0;
341
342 size_t channels = 0;
343 for (UInt32 i = 0; i < audioBufferList->mNumberBuffers; ++i)
344 channels += audioBufferList->mBuffers[i].mNumberChannels;
345 return channels;
346}

369void AudioSessionMac::setPreferredBufferSize(size_t bufferSize)
370{
371 if (m_bufferSize == bufferSize)
372 return;
373
374 AudioValueRange bufferSizeRange = {0, 0};
375 UInt32 bufferSizeRangeSize = sizeof(AudioValueRange);
376 AudioObjectPropertyAddress bufferSizeRangeAddress = {
377 kAudioDevicePropertyBufferFrameSizeRange,
378 kAudioObjectPropertyScopeGlobal,
379#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
380 kAudioObjectPropertyElementMain
381#else
382 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
383 kAudioObjectPropertyElementMaster
384 ALLOW_DEPRECATED_DECLARATIONS_END
385#endif
386 };
387 OSStatus result = AudioObjectGetPropertyData(defaultDevice(), &bufferSizeRangeAddress, 0, 0, &bufferSizeRangeSize, &bufferSizeRange);
388 if (result)
389 return;
390
391 size_t minBufferSize = static_cast<size_t>(bufferSizeRange.mMinimum);
392 size_t maxBufferSize = static_cast<size_t>(bufferSizeRange.mMaximum);
393 UInt32 bufferSizeOut = std::min(maxBufferSize, std::max(minBufferSize, bufferSize));
394
395 AudioObjectPropertyAddress preferredBufferSizeAddress = {
396 kAudioDevicePropertyBufferFrameSize,
397 kAudioObjectPropertyScopeGlobal,
398#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
399 kAudioObjectPropertyElementMain
400#else
401 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
402 kAudioObjectPropertyElementMaster
403 ALLOW_DEPRECATED_DECLARATIONS_END
404#endif
405 };
406
407 result = AudioObjectSetPropertyData(defaultDevice(), &preferredBufferSizeAddress, 0, 0, sizeof(bufferSizeOut), (void*)&bufferSizeOut);
408
409 if (!result)
410 m_bufferSize = bufferSizeOut;
411
412#if !LOG_DISABLED
413 if (result)
414 LOG(Media, "AudioSessionMac::setPreferredBufferSize(%zu) - failed with error %d", bufferSize, static_cast<int>(result));
415 else
416 LOG(Media, "AudioSessionMac::setPreferredBufferSize(%zu)", bufferSize);
417#endif
418}

420bool AudioSessionMac::isMuted() const
421{
422 UInt32 mute = 0;
423 UInt32 muteSize = sizeof(mute);
424 AudioObjectPropertyAddress muteAddress = {
425 kAudioDevicePropertyMute,
426 kAudioDevicePropertyScopeOutput,
427#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
428 kAudioObjectPropertyElementMain
429#else
430 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
431 kAudioObjectPropertyElementMaster
432 ALLOW_DEPRECATED_DECLARATIONS_END
433#endif
434 };
435 AudioObjectGetPropertyData(defaultDevice(), &muteAddress, 0, nullptr, &muteSize, &mute);
436
437 switch (mute) {
438 case 0:
439 return false;
440 case 1:
441 return true;
442 default:
443 ASSERT_NOT_REACHED();
444 return false;
445 }
446}

456void AudioSessionMac::handleMutedStateChange()
457{
458 bool isCurrentlyMuted = isMuted();
459 if (m_lastMutedState && *m_lastMutedState == isCurrentlyMuted)
460 return;
461
462 for (auto* observer : m_observers)
463 observer->hardwareMutedStateDidChange(this);
464
465 m_lastMutedState = isCurrentlyMuted;
466}

468void AudioSessionMac::addMutedStateObserver(MutedStateObserver* observer)
469{
470 m_observers.add(observer);
471
472 if (m_observers.size() > 1)
473 return;
474
475 AudioObjectPropertyAddress muteAddress = {
476 kAudioDevicePropertyMute,
477 kAudioDevicePropertyScopeOutput,
478#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
479 kAudioObjectPropertyElementMain
480#else
481 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
482 kAudioObjectPropertyElementMaster
483 ALLOW_DEPRECATED_DECLARATIONS_END
484#endif
485 };
486 AudioObjectAddPropertyListener(defaultDevice(), &muteAddress, handleMutePropertyChange, this);
487}
488
489void AudioSessionMac::removeMutedStateObserver(MutedStateObserver* observer)
490{
491 if (m_observers.size() == 1) {
492 AudioObjectPropertyAddress muteAddress = {
493 kAudioDevicePropertyMute,
494 kAudioDevicePropertyScopeOutput,
495#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
496 kAudioObjectPropertyElementMain
497#else
498 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
499 kAudioObjectPropertyElementMaster
500 ALLOW_DEPRECATED_DECLARATIONS_END
501#endif
502 };
503 AudioObjectRemovePropertyListener(defaultDevice(), &muteAddress, handleMutePropertyChange, this);
504 }
505
506 m_observers.remove(observer);
507}
