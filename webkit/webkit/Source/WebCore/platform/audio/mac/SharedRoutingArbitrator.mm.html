48bool SharedRoutingArbitrator::isInRoutingArbitrationForToken(const Token& token)
49{
50 return m_tokens.contains(token);
51}
52
53void SharedRoutingArbitrator::beginRoutingArbitrationForToken(const Token& token, AudioSession::CategoryType requestedCategory, ArbitrationCallback&& callback)
54{
55 ASSERT(!isInRoutingArbitrationForToken(token));
56
57 if (m_setupArbitrationOngoing) {
58 m_enqueuedCallbacks.append([this, weakToken = makeWeakPtr(token), callback = WTFMove(callback)] (RoutingArbitrationError error, DefaultRouteChanged routeChanged) mutable {
59 if (error == RoutingArbitrationError::None && weakToken)
60 m_tokens.add(*weakToken);
61
62 callback(error, routeChanged);
63 });
64
65 return;
66 }
67
68 if (m_currentCategory) {
69 if (*m_currentCategory >= requestedCategory) {
70 m_tokens.add(token);
71 callback(RoutingArbitrationError::None, DefaultRouteChanged::No);
72 return;
73 }
74
75 [[PAL::getAVAudioRoutingArbiterClass() sharedRoutingArbiter] leaveArbitration];
76 }
77
78 m_currentCategory = requestedCategory;
79
80 AVAudioRoutingArbitrationCategory arbitrationCategory = AVAudioRoutingArbitrationCategoryPlayback;
81 switch (requestedCategory) {
82 case AudioSession::CategoryType::MediaPlayback:
83 arbitrationCategory = AVAudioRoutingArbitrationCategoryPlayback;
84 break;
85 case AudioSession::CategoryType::RecordAudio:
86 arbitrationCategory = AVAudioRoutingArbitrationCategoryPlayAndRecord;
87 break;
88 case AudioSession::CategoryType::PlayAndRecord:
89 arbitrationCategory = AVAudioRoutingArbitrationCategoryPlayAndRecordVoice;
90 break;
91 default:
92 ASSERT_NOT_REACHED();
93 }
94
95 m_setupArbitrationOngoing = true;
96 m_enqueuedCallbacks.append([this, weakToken = makeWeakPtr(token), callback = WTFMove(callback)] (RoutingArbitrationError error, DefaultRouteChanged routeChanged) mutable {
97 if (error == RoutingArbitrationError::None && weakToken)
98 m_tokens.add(*weakToken);
99
100 callback(error, routeChanged);
101 });
102
103 [[PAL::getAVAudioRoutingArbiterClass() sharedRoutingArbiter] beginArbitrationWithCategory:arbitrationCategory completionHandler:[this](BOOL defaultDeviceChanged, NSError * _Nullable error) {
104 callOnMainRunLoop([this, defaultDeviceChanged, error = retainPtr(error)] {
105 if (error)
106 RELEASE_LOG_ERROR(Media, "SharedRoutingArbitrator::beginRoutingArbitrationForToken: %s failed with error %s:%s", convertEnumerationToString(*m_currentCategory).ascii().data(), [[error domain] UTF8String], [[error localizedDescription] UTF8String]);
107
108 // FIXME: Do we need to reset sample rate and buffer size for the new default device?
109 if (defaultDeviceChanged)
110 LOG(Media, "AudioSession::setCategory() - defaultDeviceChanged!");
111
112 Vector<ArbitrationCallback> callbacks = WTFMove(m_enqueuedCallbacks);
113 for (auto& callback : callbacks)
114 callback(error ? RoutingArbitrationError::Failed : RoutingArbitrationError::None, defaultDeviceChanged ? DefaultRouteChanged::Yes : DefaultRouteChanged::No);
115
116 m_setupArbitrationOngoing = false;
117 });
118 }];
119}
120
121void SharedRoutingArbitrator::endRoutingArbitrationForToken(const Token& token)
122{
123 m_tokens.remove(token);
124
125 if (!m_tokens.computesEmpty())
126 return;
127
128 for (auto& callback : m_enqueuedCallbacks)
129 callback(RoutingArbitrationError::Cancelled, DefaultRouteChanged::No);
130
131 m_enqueuedCallbacks.clear();
132 m_currentCategory.reset();
133 [[PAL::getAVAudioRoutingArbiterClass() sharedRoutingArbiter] leaveArbitration];
134}
