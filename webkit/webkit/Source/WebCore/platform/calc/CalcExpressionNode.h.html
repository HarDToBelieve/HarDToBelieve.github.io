32enum class CalcExpressionNodeType : uint8_t {
33 Undefined,
34 Number,
35 Length,
36 Operation,
37 Negation,
38 Inversion,
39 BlendLength,
40};
41
42class CalcExpressionNode {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 explicit CalcExpressionNode(CalcExpressionNodeType = CalcExpressionNodeType::Undefined);
46 virtual ~CalcExpressionNode() = default;
47
48 CalcExpressionNodeType type() const { return m_type; }
49
50 virtual float evaluate(float maxValue) const = 0;
51 virtual bool operator==(const CalcExpressionNode&) const = 0;
52 virtual void dump(TextStream&) const = 0;
53
54private:
55 CalcExpressionNodeType m_type;
56};
57
58inline CalcExpressionNode::CalcExpressionNode(CalcExpressionNodeType type)
59 : m_type(type)
60{
61}
62
63TextStream& operator<<(TextStream&, const CalcExpressionNode&);
64
65}
