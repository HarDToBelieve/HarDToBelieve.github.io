26#ifndef KeyedDecoderCF_h
27#define KeyedDecoderCF_h
28
29#include "KeyedCoding.h"
30#include <wtf/RetainPtr.h>
31#include <wtf/Vector.h>
32
33namespace WebCore {
34
35class KeyedDecoderCF final : public KeyedDecoder {
36public:
37 explicit KeyedDecoderCF(const uint8_t* data, size_t);
38 ~KeyedDecoderCF() override;
39
40private:
41 WARN_UNUSED_RETURN bool decodeBytes(const String& key, const uint8_t*&, size_t&) override;
42 WARN_UNUSED_RETURN bool decodeBool(const String& key, bool&) override;
43 WARN_UNUSED_RETURN bool decodeUInt32(const String& key, uint32_t&) override;
44 WARN_UNUSED_RETURN bool decodeUInt64(const String& key, uint64_t&) override;
45 WARN_UNUSED_RETURN bool decodeInt32(const String& key, int32_t&) override;
46 WARN_UNUSED_RETURN bool decodeInt64(const String& key, int64_t&) override;
47 WARN_UNUSED_RETURN bool decodeFloat(const String& key, float&) override;
48 WARN_UNUSED_RETURN bool decodeDouble(const String& key, double&) override;
49 WARN_UNUSED_RETURN bool decodeString(const String& key, String&) override;
50
51 bool beginObject(const String& key) override;
52 void endObject() override;
53
54 bool beginArray(const String& key) override;
55 bool beginArrayElement() override;
56 void endArrayElement() override;
57 void endArray() override;
58
59 RetainPtr<CFDictionaryRef> m_rootDictionary;
60
61 Vector<CFDictionaryRef, 16> m_dictionaryStack;
62 Vector<CFArrayRef, 16> m_arrayStack;
63 Vector<CFIndex> m_arrayIndexStack;
64};
65
66} // namespace WebCore
67
68#endif // KeyedDecoderCF_h
