40static RetainPtr<CFRunLoopTimerRef>& sharedTimer()
41{
42 static NeverDestroyed<RetainPtr<CFRunLoopTimerRef>> sharedTimer;
43 return sharedTimer;
44}
45static void timerFired(CFRunLoopTimerRef, void*);
46
47static const CFTimeInterval kCFTimeIntervalDistantFuture = std::numeric_limits<CFTimeInterval>::max();
48
49bool& MainThreadSharedTimer::shouldSetupPowerObserver()
50{
51 static bool setup = true;
52 return setup;
53}
54
55#if PLATFORM(IOS_FAMILY)
56static void applicationDidBecomeActive(CFNotificationCenterRef, void*, CFStringRef, const void*, CFDictionaryRef)
57{
58 WebThreadRun(^{
59 MainThreadSharedTimer::restartSharedTimer();
60 });
61}
62#endif
63
64static void setupPowerObserver()
65{
66 if (!MainThreadSharedTimer::shouldSetupPowerObserver())
67 return;
68#if PLATFORM(MAC)
69 static PowerObserver* powerObserver;
70 if (!powerObserver)
71 powerObserver = makeUnique<PowerObserver>(MainThreadSharedTimer::restartSharedTimer).release();
72#elif PLATFORM(IOS_FAMILY)
73 static bool registeredForApplicationNotification = false;
74 if (!registeredForApplicationNotification) {
75 registeredForApplicationNotification = true;
76 CFNotificationCenterRef notificationCenter = CFNotificationCenterGetLocalCenter();
77 CFNotificationCenterAddObserver(notificationCenter, nullptr, applicationDidBecomeActive, CFSTR("UIApplicationDidBecomeActiveNotification"), nullptr, CFNotificationSuspensionBehaviorCoalesce);
78 }
79#endif
80}
81
82static void timerFired(CFRunLoopTimerRef, void*)
83{
84 AutodrainedPool pool;
85 MainThreadSharedTimer::singleton().fired();
86}
87
88void MainThreadSharedTimer::restartSharedTimer()
89{
90 if (!sharedTimer())
91 return;
92
93 MainThreadSharedTimer::singleton().stop();
94 timerFired(0, 0);
95}
96
97void MainThreadSharedTimer::invalidate()
98{
99 if (!sharedTimer())
100 return;
101
102 CFRunLoopTimerInvalidate(sharedTimer().get());
103 sharedTimer() = nullptr;
104}
105
106void MainThreadSharedTimer::setFireInterval(Seconds interval)
107{
108 ASSERT(m_firedFunction);
109
110 CFAbsoluteTime fireDate = CFAbsoluteTimeGetCurrent() + interval.value();
111 if (!sharedTimer()) {
112 sharedTimer() = adoptCF(CFRunLoopTimerCreate(nullptr, fireDate, kCFTimeIntervalDistantFuture, 0, 0, timerFired, nullptr));
113#if PLATFORM(IOS_FAMILY)
114 CFRunLoopAddTimer(WebThreadRunLoop(), sharedTimer().get(), kCFRunLoopCommonModes);
115#else
116 CFRunLoopAddTimer(CFRunLoopGetCurrent(), sharedTimer().get(), kCFRunLoopCommonModes);
117#endif
118
119 setupPowerObserver();
120
121 return;
122 }
123
124 CFRunLoopTimerSetNextFireDate(sharedTimer().get(), fireDate);
125}
126
127void MainThreadSharedTimer::stop()
128{
129 if (!sharedTimer())
130 return;
131
132 CFRunLoopTimerSetNextFireDate(sharedTimer().get(), kCFTimeIntervalDistantFuture);
133}
