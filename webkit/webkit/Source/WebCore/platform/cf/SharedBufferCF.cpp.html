36SharedBuffer::SharedBuffer(CFDataRef data)
37{
38 append(data);
39}

41// Using Foundation allows for an even more efficient implementation of this function,
42// so only use this version for non-Foundation.
43#if !USE(FOUNDATION)
44RetainPtr<CFDataRef> SharedBuffer::createCFData() const
45{
46 if (m_segments.size() == 1) {
47 if (auto data = WTF::get_if<RetainPtr<CFDataRef>>(m_segments[0].segment->m_immutableData))
48 return *data;
49 }
50 auto cfData = adoptCF(CFDataCreateMutable(nullptr, size()));
51 forEachSegment([&](auto& segment) {
52 CFDataAppendBytes(cfData.get(), segment.data(), segment.size());
53 });
54 return cfData;
55}
56#endif
57
58Ref<SharedBuffer> SharedBuffer::create(CFDataRef data)
59{
60 return adoptRef(*new SharedBuffer(data));
61}
62
63void SharedBuffer::hintMemoryNotNeededSoon() const
64{
65 for (const auto& entry : m_segments) {
66 if (entry.segment->hasOneRef()) {
67 if (auto data = WTF::get_if<RetainPtr<CFDataRef>>(entry.segment->m_immutableData))
68 OSAllocator::hintMemoryNotNeededSoon(const_cast<UInt8*>(CFDataGetBytePtr(data->get())), CFDataGetLength(data->get()));
69 }
70 }
71}
72
73void SharedBuffer::append(CFDataRef data)
74{
75 ASSERT(!m_hasBeenCombinedIntoOneSegment);
76 if (data) {
77 m_segments.append({m_size, DataSegment::create(data)});
78 m_size += CFDataGetLength(data);
79 }
80 ASSERT(internallyConsistent());
81}
