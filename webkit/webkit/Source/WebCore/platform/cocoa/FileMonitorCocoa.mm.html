26#import "config.h"
27#import "FileMonitor.h"
28
29#import "Logging.h"
30#import <wtf/BlockPtr.h>
31#import <wtf/FileSystem.h>
32
33namespace WebCore {
34
35constexpr unsigned monitorMask = DISPATCH_VNODE_DELETE | DISPATCH_VNODE_WRITE | DISPATCH_VNODE_RENAME | DISPATCH_VNODE_REVOKE;
36constexpr unsigned fileUnavailableMask = DISPATCH_VNODE_DELETE | DISPATCH_VNODE_RENAME | DISPATCH_VNODE_REVOKE;
37
38FileMonitor::FileMonitor(const String& path, Ref<WorkQueue>&& handlerQueue, WTF::Function<void(FileChangeType)>&& modificationHandler)
39{
40 if (path.isEmpty())
41 return;
42
43 if (!modificationHandler)
44 return;
45
46 auto handle = FileSystem::openFile(path, FileSystem::FileOpenMode::EventsOnly);
47 if (handle == FileSystem::invalidPlatformFileHandle) {
48 RELEASE_LOG_ERROR(ResourceLoadStatistics, "Failed to open statistics file for monitoring: %s", path.utf8().data());
49 return;
50 }
51
52 // The source (platformMonitor) retains the dispatch queue.
53 m_platformMonitor = adoptOSObject(dispatch_source_create(DISPATCH_SOURCE_TYPE_VNODE, handle, monitorMask, handlerQueue->dispatchQueue()));
54
55 LOG(ResourceLoadStatistics, "Creating monitor %p", m_platformMonitor.get());
56
57 dispatch_source_set_event_handler(m_platformMonitor.get(), makeBlockPtr([modificationHandler = WTFMove(modificationHandler), fileMonitor = m_platformMonitor] {
58 // If this is getting called after the monitor was cancelled, just drop the notification.
59 if (dispatch_source_testcancel(fileMonitor.get()))
60 return;
61
62 unsigned flag = dispatch_source_get_data(fileMonitor.get());
63 LOG(ResourceLoadStatistics, "File event %#X for monitor %p", flag, fileMonitor.get());
64 if (flag & fileUnavailableMask) {
65 modificationHandler(FileChangeType::Removal);
66 dispatch_source_cancel(fileMonitor.get());
67 } else {
68 ASSERT(flag & DISPATCH_VNODE_WRITE);
69 modificationHandler(FileChangeType::Modification);
70 }
71 }).get());
72
73 dispatch_source_set_cancel_handler(m_platformMonitor.get(), [handle] () mutable {
74 FileSystem::closeFile(handle);
75 });
76
77 dispatch_resume(m_platformMonitor.get());
78}
79
80FileMonitor::~FileMonitor()
81{
82 if (!m_platformMonitor)
83 return;
84
85 LOG(ResourceLoadStatistics, "Stopping monitor %p", m_platformMonitor.get());
86
87 dispatch_source_cancel(m_platformMonitor.get());
88}
