42bool PlatformKeyboardEvent::currentCapsLockState()
43{
44 return currentStateOfModifierKeys().contains(PlatformEvent::Modifier::CapsLockKey);
45}
46
47void PlatformKeyboardEvent::getCurrentModifierState(bool& shiftKey, bool& ctrlKey, bool& altKey, bool& metaKey)
48{
49 auto currentModifiers = currentStateOfModifierKeys();
50 shiftKey = currentModifiers.contains(PlatformEvent::Modifier::ShiftKey);
51 ctrlKey = currentModifiers.contains(PlatformEvent::Modifier::ControlKey);
52 altKey = currentModifiers.contains(PlatformEvent::Modifier::AltKey);
53 metaKey = currentModifiers.contains(PlatformEvent::Modifier::MetaKey);
54}
