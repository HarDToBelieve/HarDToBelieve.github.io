38static MemoryCompactLookupOnlyRobinHoodHashMap<String, MemoryCompactLookupOnlyRobinHoodHashSet<String>>& extensionsForMIMETypeMap()
39{
40 static auto extensionsForMIMETypeMap = makeNeverDestroyed([] {
41 MemoryCompactLookupOnlyRobinHoodHashMap<String, MemoryCompactLookupOnlyRobinHoodHashSet<String>> map;
42
43 auto addExtension = [&](const String& type, const String& extension) {
44 map.add(type, MemoryCompactLookupOnlyRobinHoodHashSet<String>()).iterator->value.add(extension);
45 };
46
47 auto addExtensions = [&](const String& type, NSArray<NSString *> *extensions) {
48 size_t pos = type.reverseFind('/');
49
50 ASSERT(pos != notFound);
51 auto wildcardMIMEType = makeString(StringView(type).left(pos), "/*"_s);
52
53 for (NSString *extension in extensions) {
54 // Add extension to wildcardMIMEType, for example add "png" to "image/*"
55 addExtension(wildcardMIMEType, extension);
56 // Add extension to its mimeType, for example add "png" to "image/png"
57 addExtension(type, extension);
58 }
59 };
60
61 auto allUTIs = adoptCF(_UTCopyDeclaredTypeIdentifiers());
62
63ALLOW_DEPRECATED_DECLARATIONS_BEGIN
64 for (NSString *uti in (__bridge NSArray<NSString *> *)allUTIs.get()) {
65 auto type = adoptCF(UTTypeCopyPreferredTagWithClass((__bridge CFStringRef)uti, kUTTagClassMIMEType));
66 if (!type)
67 continue;
68 auto extensions = adoptCF(UTTypeCopyAllTagsWithClass((__bridge CFStringRef)uti, kUTTagClassFilenameExtension));
69 if (!extensions || !CFArrayGetCount(extensions.get()))
70 continue;
71 addExtensions(type.get(), (__bridge NSArray<NSString *> *)extensions.get());
72 }
73ALLOW_DEPRECATED_DECLARATIONS_END
74
75 return map;
76 }());
77
78 return extensionsForMIMETypeMap;
79}

81static Vector<String> extensionsForWildcardMIMEType(const String& type)
82{
83 Vector<String> extensions;
84
85 auto iterator = extensionsForMIMETypeMap().find(type);
86 if (iterator != extensionsForMIMETypeMap().end())
87 extensions.appendRange(iterator->value.begin(), iterator->value.end());
88
89 return extensions;
90}

92String MIMETypeRegistry::mimeTypeForExtension(const String& extension)
93{
94 return [[NSURLFileTypeMappings sharedMappings] MIMETypeForExtension:(NSString *)extension];
95}
96
97Vector<String> MIMETypeRegistry::extensionsForMIMEType(const String& type)
98{
99 if (type.endsWith('*'))
100 return extensionsForWildcardMIMEType(type);
101 return makeVector<String>([[NSURLFileTypeMappings sharedMappings] extensionsForMIMEType:type]);
102}

104String MIMETypeRegistry::preferredExtensionForMIMEType(const String& type)
105{
106 // System Previews accept some non-standard MIMETypes, so we can't rely on
107 // the file type mappings.
108 if (isSystemPreviewMIMEType(type))
109 return "usdz"_s;
110
111 return [[NSURLFileTypeMappings sharedMappings] preferredExtensionForMIMEType:(NSString *)type];
112}

114bool MIMETypeRegistry::isApplicationPluginMIMEType(const String& MIMEType)
115{
116#if ENABLE(PDFKIT_PLUGIN)
117 // FIXME: This should test if we're actually going to use PDFPlugin,
118 // but we only know that in WebKit2 at the moment. This is not a problem
119 // in practice because if we don't have PDFPlugin and we go to instantiate the
120 // plugin, there won't exist an application plugin supporting these MIME types.
121 if (isPDFOrPostScriptMIMEType(MIMEType))
122 return true;
123#else
124 UNUSED_PARAM(MIMEType);
125#endif
126
127 return false;
128}
