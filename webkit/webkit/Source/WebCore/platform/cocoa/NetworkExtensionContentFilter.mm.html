43static inline NSData *replacementDataFromDecisionInfo(NSDictionary *decisionInfo)
44{
45 ASSERT_WITH_SECURITY_IMPLICATION(!decisionInfo || [decisionInfo isKindOfClass:[NSDictionary class]]);
46 return decisionInfo[NEFilterSourceOptionsPageData];
47}

51NetworkExtensionContentFilter::SandboxExtensionsState NetworkExtensionContentFilter::m_sandboxExtensionsState = SandboxExtensionsState::NotSet;
52
53bool NetworkExtensionContentFilter::enabled()
54{
55 bool enabled = false;
56 switch (m_sandboxExtensionsState) {
57 case SandboxExtensionsState::Consumed:
58 enabled = true;
59 break;
60 case SandboxExtensionsState::NotConsumed:
61 enabled = false;
62 break;
63 case SandboxExtensionsState::NotSet:
64 enabled = isRequired();
65 break;
66 }
67 LOG(ContentFiltering, "NetworkExtensionContentFilter is %s.\n", enabled ? "enabled" : "not enabled");
68 return enabled;
69}

76void NetworkExtensionContentFilter::initialize(const URL* url)
77{
78 ASSERT(!m_queue);
79 ASSERT(!m_neFilterSource);
80 m_queue = adoptOSObject(dispatch_queue_create("WebKit NetworkExtension Filtering", DISPATCH_QUEUE_SERIAL));
81 ASSERT_UNUSED(url, !url);
82 m_neFilterSource = adoptNS([[NEFilterSource alloc] initWithDecisionQueue:m_queue.get()]);
83 [m_neFilterSource setSourceAppIdentifier:applicationBundleIdentifier()];
84 [m_neFilterSource setSourceAppPid:presentingApplicationPID()];
85}

87void NetworkExtensionContentFilter::willSendRequest(ResourceRequest& request, const ResourceResponse& redirectResponse)
88{
89 ASSERT(!request.isNull());
90 if (!request.url().protocolIsInHTTPFamily() || !enabled()) {
91 m_state = State::Allowed;
92 return;
93 }
94
95 initialize();
96
97 if (!redirectResponse.isNull()) {
98 responseReceived(redirectResponse);
99 if (!needsMoreData())
100 return;
101 }
102
103 BinarySemaphore semaphore;
104 RetainPtr<NSString> modifiedRequestURLString;
105 [m_neFilterSource willSendRequest:request.nsURLRequest(DoNotUpdateHTTPBody) decisionHandler:[this, &modifiedRequestURLString, &semaphore](NEFilterSourceStatus status, NSDictionary *decisionInfo) {
106 modifiedRequestURLString = decisionInfo[NEFilterSourceOptionsRedirectURL];
107 ASSERT(!modifiedRequestURLString || [modifiedRequestURLString isKindOfClass:[NSString class]]);
108 handleDecision(status, replacementDataFromDecisionInfo(decisionInfo));
109 semaphore.signal();
110 }];
111
112 // FIXME: We have to block here since DocumentLoader expects to have a
113 // blocked/not blocked answer from the filter immediately after calling
114 // addData(). We should find a way to make this asynchronous.
115 semaphore.wait();
116
117 if (!modifiedRequestURLString)
118 return;
119
120 URL modifiedRequestURL { URL(), modifiedRequestURLString.get() };
121 if (!modifiedRequestURL.isValid()) {
122 LOG(ContentFiltering, "NetworkExtensionContentFilter failed to convert modified URL string %@ to a URL.\n", modifiedRequestURLString.get());
123 return;
124 }
125
126 request.setURL(modifiedRequestURL);
127}

129void NetworkExtensionContentFilter::responseReceived(const ResourceResponse& response)
130{
131 if (!response.url().protocolIsInHTTPFamily()) {
132 m_state = State::Allowed;
133 return;
134 }
135
136 BinarySemaphore semaphore;
137 [m_neFilterSource receivedResponse:response.nsURLResponse() decisionHandler:[this, &semaphore](NEFilterSourceStatus status, NSDictionary *decisionInfo) {
138 handleDecision(status, replacementDataFromDecisionInfo(decisionInfo));
139 semaphore.signal();
140 }];
141
142 // FIXME: We have to block here since DocumentLoader expects to have a
143 // blocked/not blocked answer from the filter immediately after calling
144 // addData(). We should find a way to make this asynchronous.
145 semaphore.wait();
146}
147
148void NetworkExtensionContentFilter::addData(const uint8_t* data, int length)
149{
150 RetainPtr<NSData> copiedData { [NSData dataWithBytes:(void*)data length:length] };
151
152 BinarySemaphore semaphore;
153 [m_neFilterSource receivedData:copiedData.get() decisionHandler:[this, &semaphore](NEFilterSourceStatus status, NSDictionary *decisionInfo) {
154 handleDecision(status, replacementDataFromDecisionInfo(decisionInfo));
155 semaphore.signal();
156 }];
157
158 // FIXME: We have to block here since DocumentLoader expects to have a
159 // blocked/not blocked answer from the filter immediately after calling
160 // addData(). We should find a way to make this asynchronous.
161 semaphore.wait();
162}
163
164void NetworkExtensionContentFilter::finishedAddingData()
165{
166 BinarySemaphore semaphore;
167 [m_neFilterSource finishedLoadingWithDecisionHandler:[this, &semaphore](NEFilterSourceStatus status, NSDictionary *decisionInfo) {
168 handleDecision(status, replacementDataFromDecisionInfo(decisionInfo));
169 semaphore.signal();
170 }];
171
172 // FIXME: We have to block here since DocumentLoader expects to have a
173 // blocked/not blocked answer from the filter immediately after calling
174 // finishedAddingData(). We should find a way to make this asynchronous.
175 semaphore.wait();
176}

184ContentFilterUnblockHandler NetworkExtensionContentFilter::unblockHandler() const
185{
186 using DecisionHandlerFunction = ContentFilterUnblockHandler::DecisionHandlerFunction;
187
188 RetainPtr<NEFilterSource> neFilterSource { m_neFilterSource };
189 return ContentFilterUnblockHandler {
190 "nefilter-unblock"_s, [neFilterSource](DecisionHandlerFunction decisionHandler) {
191 [neFilterSource remediateWithDecisionHandler:[decisionHandler](NEFilterSourceStatus status, NSDictionary *) {
192 LOG(ContentFiltering, "NEFilterSource %s the unblock request.\n", status == NEFilterSourceStatusPass ? "allowed" : "did not allow");
193 decisionHandler(status == NEFilterSourceStatusPass);
194 }];
195 }
196 };
197}
198
199void NetworkExtensionContentFilter::handleDecision(NEFilterSourceStatus status, NSData *replacementData)
200{
201 ASSERT_WITH_SECURITY_IMPLICATION(!replacementData || [replacementData isKindOfClass:[NSData class]]);
202
203 switch (status) {
204 case NEFilterSourceStatusPass:
205 case NEFilterSourceStatusError:
206 case NEFilterSourceStatusWhitelisted:
207 case NEFilterSourceStatusBlacklisted:
208 m_state = State::Allowed;
209 break;
210 case NEFilterSourceStatusBlock:
211 m_state = State::Blocked;
212 break;
213 case NEFilterSourceStatusNeedsMoreData:
214 m_state = State::Filtering;
215 break;
216 }
217
218 if (didBlockData())
219 m_replacementData = replacementData;
220#if !LOG_DISABLED
221 if (!needsMoreData())
222 LOG(ContentFiltering, "NetworkExtensionContentFilter stopped buffering with status %zd and replacement data length %zu.\n", status, replacementData.length);
223#endif
224}

231void NetworkExtensionContentFilter::setHasConsumedSandboxExtensions(bool hasConsumedSandboxExtensions)
232{
233 if (m_sandboxExtensionsState == SandboxExtensionsState::Consumed)
234 return;
235
236 m_sandboxExtensionsState = (hasConsumedSandboxExtensions ? SandboxExtensionsState::Consumed : SandboxExtensionsState::NotConsumed);
237}
