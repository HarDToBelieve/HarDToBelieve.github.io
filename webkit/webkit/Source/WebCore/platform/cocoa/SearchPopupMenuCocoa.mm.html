33static NSString * const dateKey = @"date";
34static NSString * const itemsKey = @"items";
35static NSString * const searchesKey = @"searches";
36static NSString * const searchStringKey = @"searchString";
37
38static NSString *searchFieldRecentSearchesStorageDirectory()
39{
40 NSString *appName = [[NSBundle mainBundle] bundleIdentifier];
41 if (!appName)
42 appName = [[NSProcessInfo processInfo] processName];
43
44 return [[NSHomeDirectory() stringByAppendingPathComponent:@"Library/WebKit"] stringByAppendingPathComponent:appName];
45}
46
47static NSString *searchFieldRecentSearchesPlistPath()
48{
49 return [searchFieldRecentSearchesStorageDirectory() stringByAppendingPathComponent:@"RecentSearches.plist"];
50}
51
52static RetainPtr<NSMutableDictionary> readSearchFieldRecentSearchesPlist()
53{
54 return adoptNS([[NSMutableDictionary alloc] initWithContentsOfFile:searchFieldRecentSearchesPlistPath()]);
55}
56
57static WallTime toSystemClockTime(NSDate *date)
58{
59 ASSERT(date);
60 return WallTime::fromRawSeconds(date.timeIntervalSince1970);
61}
62
63static NSDate *toNSDateFromSystemClock(WallTime time)
64{
65 return [NSDate dateWithTimeIntervalSince1970:time.secondsSinceEpoch().seconds()];
66}

68static NSMutableArray *typeCheckedRecentSearchesArray(NSMutableDictionary *itemsDictionary, NSString *name)
69{
70 NSMutableDictionary *nameDictionary = [itemsDictionary objectForKey:name];
71 if (![nameDictionary isKindOfClass:[NSDictionary class]])
72 return nil;
73
74 NSMutableArray *recentSearches = [nameDictionary objectForKey:searchesKey];
75 if (![recentSearches isKindOfClass:[NSArray class]])
76 return nil;
77
78 return recentSearches;
79}
80
81static NSDate *typeCheckedDateInRecentSearch(NSDictionary *recentSearch)
82{
83 if (![recentSearch isKindOfClass:[NSDictionary class]])
84 return nil;
85
86 NSDate *date = [recentSearch objectForKey:dateKey];
87 if (![date isKindOfClass:[NSDate class]])
88 return nil;
89
90 return date;
91}

93static RetainPtr<NSDictionary> typeCheckedRecentSearchesRemovingRecentSearchesAddedAfterDate(NSDate *date)
94{
95 if ([date isEqualToDate:[NSDate distantPast]])
96 return nil;
97
98 RetainPtr<NSMutableDictionary> recentSearchesPlist = readSearchFieldRecentSearchesPlist();
99 NSMutableDictionary *itemsDictionary = [recentSearchesPlist objectForKey:itemsKey];
100 if (![itemsDictionary isKindOfClass:[NSDictionary class]])
101 return nil;
102
103 RetainPtr<NSMutableArray> keysToRemove = adoptNS([[NSMutableArray alloc] init]);
104 for (NSString *key in itemsDictionary) {
105 if (![key isKindOfClass:[NSString class]])
106 return nil;
107
108 NSMutableArray *recentSearches = typeCheckedRecentSearchesArray(itemsDictionary, key);
109 if (!recentSearches)
110 return nil;
111
112 RetainPtr<NSMutableArray> entriesToRemove = adoptNS([[NSMutableArray alloc] init]);
113 for (NSDictionary *recentSearch in recentSearches) {
114 NSDate *dateAdded = typeCheckedDateInRecentSearch(recentSearch);
115 if (!dateAdded)
116 return nil;
117
118 if ([dateAdded compare:date] == NSOrderedDescending)
119 [entriesToRemove addObject:recentSearch];
120 }
121
122 [recentSearches removeObjectsInArray:entriesToRemove.get()];
123 if (!recentSearches.count)
124 [keysToRemove addObject:key];
125 }
126
127 [itemsDictionary removeObjectsForKeys:keysToRemove.get()];
128 if (!itemsDictionary.count)
129 return nil;
130
131 return recentSearchesPlist;
132}

134static void writeEmptyRecentSearchesPlist()
135{
136 auto emptyItemsDictionary = adoptNS([[NSDictionary alloc] init]);
137 auto emptyRecentSearchesDictionary = adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:emptyItemsDictionary.get(), itemsKey, nil]);
138 [emptyRecentSearchesDictionary writeToFile:searchFieldRecentSearchesPlistPath() atomically:YES];
139}

141void saveRecentSearches(const String& name, const Vector<RecentSearch>& searchItems)
142{
143 if (name.isEmpty())
144 return;
145
146 RetainPtr<NSDictionary> recentSearchesPlist = readSearchFieldRecentSearchesPlist();
147 RetainPtr<NSMutableDictionary> itemsDictionary = [recentSearchesPlist objectForKey:itemsKey];
148
149 // The NSMutableDictionary method we use to read the property list guarantees we get only
150 // mutable containers, but it does not guarantee the file has a dictionary as expected.
151 if (![itemsDictionary isKindOfClass:[NSDictionary class]]) {
152 itemsDictionary = adoptNS([[NSMutableDictionary alloc] init]);
153 recentSearchesPlist = adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:itemsDictionary.get(), itemsKey, nil]);
154 }
155
156 if (searchItems.isEmpty())
157 [itemsDictionary removeObjectForKey:name];
158 else {
159 auto items = createNSArray(searchItems, [] (auto& item) {
160 return adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:item.string, searchStringKey, toNSDateFromSystemClock(item.time), dateKey, nil]);
161 });
162 [itemsDictionary setObject:adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:items.get(), searchesKey, nil]).get() forKey:name];
163 }
164
165 [recentSearchesPlist writeToFile:searchFieldRecentSearchesPlistPath() atomically:YES];
166}

168Vector<RecentSearch> loadRecentSearches(const String& name)
169{
170 Vector<RecentSearch> searchItems;
171
172 if (name.isEmpty())
173 return searchItems;
174
175 RetainPtr<NSMutableDictionary> recentSearchesPlist = readSearchFieldRecentSearchesPlist();
176 if (!recentSearchesPlist)
177 return searchItems;
178
179 NSMutableDictionary *items = [recentSearchesPlist objectForKey:itemsKey];
180 if (![items isKindOfClass:[NSDictionary class]])
181 return searchItems;
182
183 NSArray *recentSearches = typeCheckedRecentSearchesArray(items, name);
184 if (!recentSearches)
185 return searchItems;
186
187 for (NSDictionary *item in recentSearches) {
188 NSDate *date = typeCheckedDateInRecentSearch(item);
189 if (!date)
190 continue;
191
192 NSString *searchString = [item objectForKey:searchStringKey];
193 if (![searchString isKindOfClass:[NSString class]])
194 continue;
195
196 searchItems.append({ String{ searchString }, toSystemClockTime(date) });
197 }
198
199 return searchItems;
200}
201
202void removeRecentlyModifiedRecentSearches(WallTime oldestTimeToRemove)
203{
204 NSDate *date = toNSDateFromSystemClock(oldestTimeToRemove);
205 auto recentSearchesPlist = typeCheckedRecentSearchesRemovingRecentSearchesAddedAfterDate(date);
206 if (recentSearchesPlist)
207 [recentSearchesPlist writeToFile:searchFieldRecentSearchesPlistPath() atomically:YES];
208 else
209 writeEmptyRecentSearchesPlist();
210}
