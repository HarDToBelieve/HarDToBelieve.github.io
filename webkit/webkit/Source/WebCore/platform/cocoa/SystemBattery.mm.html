34static std::optional<bool> hasBattery;
35static std::optional<bool> hasAC;
36
37void setSystemHasBattery(bool battery)
38{
39 hasBattery = battery;
40}
41
42bool systemHasBattery()
43{
44 if (auto overrideForTesting = SystemBatteryStatusTestingOverrides::singleton().hasBattery())
45 return *overrideForTesting;
46
47 if (!hasBattery.has_value()) {
48 hasBattery = [] {
49#if PLATFORM(IOS) || PLATFORM(WATCHOS)
50 // Devices running iOS / WatchOS always have a battery.
51 return true;
52#elif PLATFORM(APPLETV)
53 return false;
54#else
55 RetainPtr<CFTypeRef> powerSourcesInfo = adoptCF(IOPSCopyPowerSourcesInfo());
56 if (!powerSourcesInfo)
57 return false;
58 RetainPtr<CFArrayRef> powerSourcesList = adoptCF(IOPSCopyPowerSourcesList(powerSourcesInfo.get()));
59 if (!powerSourcesList)
60 return false;
61 for (CFIndex i = 0, count = CFArrayGetCount(powerSourcesList.get()); i < count; ++i) {
62 CFDictionaryRef description = IOPSGetPowerSourceDescription(powerSourcesInfo.get(), CFArrayGetValueAtIndex(powerSourcesList.get(), i));
63 CFTypeRef value = CFDictionaryGetValue(description, CFSTR(kIOPSTypeKey));
64 if (!value || CFEqual(value, CFSTR(kIOPSInternalBatteryType)))
65 return true;
66 }
67 return false;
68#endif
69 }();
70 }
71
72 return *hasBattery;
73}

80void setSystemHasAC(bool ac)
81{
82 hasAC = ac;

84
85bool systemHasAC()
86{
87 if (auto overrideForTesting = SystemBatteryStatusTestingOverrides::singleton().hasAC())
88 return *overrideForTesting;
89
90 if (!hasAC.has_value()) {
91 hasAC = [] {
92#if PLATFORM(APPLETV)
93 return true;
94#else
95 RetainPtr<CFTypeRef> powerSourcesInfo = adoptCF(IOPSCopyPowerSourcesInfo());
96 if (!powerSourcesInfo)
97 return false;
98 RetainPtr<CFArrayRef> powerSourcesList = adoptCF(IOPSCopyPowerSourcesList(powerSourcesInfo.get()));
99 if (!powerSourcesList)
100 return false;
101 for (CFIndex i = 0, count = CFArrayGetCount(powerSourcesList.get()); i < count; ++i) {
102 CFDictionaryRef description = IOPSGetPowerSourceDescription(powerSourcesInfo.get(), CFArrayGetValueAtIndex(powerSourcesList.get(), i));
103 if (!description)
104 continue;
105 CFTypeRef value = CFDictionaryGetValue(description, CFSTR(kIOPSPowerSourceStateKey));
106 if (value && CFEqual(value, CFSTR(kIOPSACPowerValue)))
107 return true;
108 }
109 return false;
110#endif
111 }();
112 }
113
114 return *hasAC;
115}

117std::optional<bool> cachedSystemHasAC()
118{
119 return hasAC;
120}
