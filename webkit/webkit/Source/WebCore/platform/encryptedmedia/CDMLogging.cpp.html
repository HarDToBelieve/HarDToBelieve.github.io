43static Ref<JSON::Object> toJSONObject(const CDMMediaCapability& capability)
44{
45 auto object = JSON::Object::create();
46 object->setString("contentType"_s, capability.contentType);
47 object->setString("robustness"_s, capability.robustness);
48 if (capability.encryptionScheme)
49 object->setString("encryptionScheme"_s, convertEnumerationToString(capability.encryptionScheme.value()));
50 return object;
51}
52
53static Ref<JSON::Object> toJSONObject(const CDMRestrictions& restrictions)
54{
55 auto object = JSON::Object::create();
56 object->setBoolean("distinctiveIdentifierDenied"_s, restrictions.distinctiveIdentifierDenied);
57 object->setBoolean("persistentStateDenied"_s, restrictions.persistentStateDenied);
58 auto deniedSessionTypes = JSON::Array::create();
59 for (auto type : restrictions.deniedSessionTypes)
60 deniedSessionTypes->pushString(convertEnumerationToString(type));
61 object->setArray("deniedSessionTypes"_s, WTFMove(deniedSessionTypes));
62 return object;
63}
64
65static Ref<JSON::Object> toJSONObject(const CDMKeySystemConfiguration& configuration)
66{
67 auto object = JSON::Object::create();
68 object->setString("label"_s, configuration.label);
69
70 auto initDataTypes = JSON::Array::create();
71 for (auto initDataType : configuration.initDataTypes)
72 initDataTypes->pushString(initDataType);
73 object->setArray("initDataTypes"_s, WTFMove(initDataTypes));
74
75 auto audioCapabilities = JSON::Array::create();
76 for (auto capability : configuration.audioCapabilities)
77 audioCapabilities->pushObject(toJSONObject(capability));
78 object->setArray("audioCapabilities"_s, WTFMove(audioCapabilities));
79
80 auto videoCapabilities = JSON::Array::create();
81 for (auto capability : configuration.videoCapabilities)
82 videoCapabilities->pushObject(toJSONObject(capability));
83 object->setArray("videoCapabilities"_s, WTFMove(videoCapabilities));
84
85 object->setString("distinctiveIdentifier"_s, convertEnumerationToString(configuration.distinctiveIdentifier));
86 object->setString("persistentState"_s, convertEnumerationToString(configuration.persistentState));
87
88 auto sessionTypes = JSON::Array::create();
89 for (auto type : configuration.sessionTypes)
90 sessionTypes->pushString(convertEnumerationToString(type));
91 object->setArray("sessionTypes"_s, WTFMove(sessionTypes));
92
93 return object;
94}

96static String toJSONString(const CDMKeySystemConfiguration& configuration)
