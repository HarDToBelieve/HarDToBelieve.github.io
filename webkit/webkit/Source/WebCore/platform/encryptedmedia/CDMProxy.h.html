55class KeyHandle : public ThreadSafeRefCounted<KeyHandle> {
56public:
57 using KeyStatus = CDMInstanceSession::KeyStatus;
58
59 static RefPtr<KeyHandle> create(KeyStatus status, KeyIDType&& keyID, KeyHandleValueVariant&& keyHandleValue)
60 {
61 return adoptRef(*new KeyHandle(status, WTFMove(keyID), WTFMove(keyHandleValue)));
62 }
63 Ref<SharedBuffer> idAsSharedBuffer() const { return SharedBuffer::create(m_id.data(), m_id.size()); }
64
65 bool takeValueIfDifferent(KeyHandleValueVariant&&);
66
67 const KeyIDType& id() const { return m_id; }
68 const KeyHandleValueVariant& value() const { return m_value; }
69 KeyHandleValueVariant& value() { return m_value; }
70 KeyStatus status() const { return m_status; }
71 void mergeKeyInto(RefPtr<KeyHandle>&& other)
72 {
73 m_status = other->m_status;
74 m_value = other->m_value;
75 m_numSessionReferences += other->m_numSessionReferences;
76 }
77 bool isStatusCurrentlyValid()
78 {
79 return m_status == CDMInstanceSession::KeyStatus::Usable || m_status == CDMInstanceSession::KeyStatus::OutputRestricted
80 || m_status == CDMInstanceSession::KeyStatus::OutputDownscaled;
81 }
82
83 String idAsString() const;
84
85 // Two keys are equal if they have the same ID, ignoring key value and status.
86 friend bool operator==(const KeyHandle &k1, const KeyHandle &k2) { return k1.m_id == k2.m_id; }
87 friend bool operator==(const KeyHandle &k, const KeyIDType& keyID) { return k.m_id == keyID; }
88 friend bool operator==(const KeyIDType& keyID, const KeyHandle &k) { return k == keyID; }
89 friend bool operator<(const KeyHandle& k1, const KeyHandle& k2)
90 {
91 // Key IDs are compared as follows: For key IDs A of length m and
92 // B of length n, assigned such that m <= n, let A < B if and only
93 // if the m octets of A are less in lexicographical order than the
94 // first m octets of B or those octets are equal and m < n.
95 // 6.1 https://www.w3.org/TR/encrypted-media/
96 int isDifference = memcmp(k1.m_id.data(), k2.m_id.data(), std::min(k1.m_id.size(), k2.m_id.size()));
97 if (isDifference)
98 return isDifference < 0;
99 // The keys are equal to the shared length, the shorter string
100 // is therefore less than the longer one in a lexicographical
101 // ordering.
102 return k1.m_id.size() < k2.m_id.size();
103 }
104
105private:
106 void addSessionReference() { ASSERT(isMainThread()); m_numSessionReferences++; }
107 void removeSessionReference() { ASSERT(isMainThread()); m_numSessionReferences--; }
108 int numSessionReferences() const { ASSERT(isMainThread()); return m_numSessionReferences; }
109 bool hasReferences() const { ASSERT(isMainThread()); return m_numSessionReferences > 0; }
110 friend class KeyStore;
111 friend class ReferenceAwareKeyStore;
112
113 KeyHandle(KeyStatus status, KeyIDType&& keyID, KeyHandleValueVariant&& keyHandleValue)
114 : m_status(status), m_id(WTFMove(keyID)), m_value(WTFMove(keyHandleValue)) { }
115
116 KeyStatus m_status;
117 KeyIDType m_id;
118 KeyHandleValueVariant m_value;
119 int m_numSessionReferences { 0 };
120};

168// Handle to a "real" CDM, not the JavaScript facade. This can be used
169// from background threads (i.e. decryptors).
170class CDMProxy : public ThreadSafeRefCounted<CDMProxy> {
171public:
172 static constexpr Seconds MaxKeyWaitTimeSeconds = 7_s;
173
174 virtual ~CDMProxy() = default;
175
176 void updateKeyStore(const KeyStore&);
177 void unrefAllKeysFrom(const KeyStore&);
178 void setInstance(CDMInstanceProxy*);
179 void abortWaitingForKey() const;
180
181protected:
182 RefPtr<KeyHandle> keyHandle(const KeyIDType&) const;
183 bool keyAvailable(const KeyIDType&) const;
184 bool keyAvailableUnlocked(const KeyIDType&) const WTF_REQUIRES_LOCK(m_keysLock);
185 std::optional<Ref<KeyHandle>> tryWaitForKeyHandle(const KeyIDType&, WeakPtr<CDMProxyDecryptionClient>&&) const;
186 std::optional<Ref<KeyHandle>> getOrWaitForKeyHandle(const KeyIDType&, WeakPtr<CDMProxyDecryptionClient>&&) const;
187 std::optional<KeyHandleValueVariant> getOrWaitForKeyValue(const KeyIDType&, WeakPtr<CDMProxyDecryptionClient>&&) const;
188 void startedWaitingForKey() const;
189 void stoppedWaitingForKey() const;
190 const CDMInstanceProxy* instance() const;
191
192private:
193 mutable Lock m_instanceLock;
194 CDMInstanceProxy* m_instance WTF_GUARDED_BY_LOCK(m_instanceLock);
195
196 mutable Lock m_keysLock;
197 mutable Condition m_keysCondition;
198 // FIXME: Duplicated key stores in the instance and the proxy are probably not needed, but simplified
199 // the initial implementation in terms of threading invariants.
200 ReferenceAwareKeyStore m_keyStore WTF_GUARDED_BY_LOCK(m_keysLock);
201};

236// Base class for common session management code and for communicating messages
237// from "real CDM" state changes to JS.
238class CDMInstanceProxy : public CDMInstance, public CanMakeWeakPtr<CDMInstanceProxy> {
239public:
240 explicit CDMInstanceProxy(const String& keySystem)
241 {
242 ASSERT(isMainThread());
243 m_cdmProxy = CDMProxyFactory::createCDMProxyForKeySystem(keySystem);
244 if (m_cdmProxy)
245 m_cdmProxy->setInstance(this);
246 }
247 virtual ~CDMInstanceProxy() = default;
248
249 // Main-thread only.
250 void mergeKeysFrom(const KeyStore&);
251 void unrefAllKeysFrom(const KeyStore&);
252
253 // Media player query methods - main thread only.
254 const RefPtr<CDMProxy>& proxy() const { ASSERT(isMainThread()); return m_cdmProxy; }
255 virtual bool isWaitingForKey() const { ASSERT(isMainThread()); return m_numDecryptorsWaitingForKey > 0; }
256 void setPlayer(MediaPlayer* player) { ASSERT(isMainThread()); m_player = player; }
257
258 // Proxy methods - must be thread-safe.
259 void startedWaitingForKey();
260 void stoppedWaitingForKey();
261
262private:
263 RefPtr<CDMProxy> m_cdmProxy;
264 // FIXME: WeakPtr for the m_player? This is accessed from background and main threads, it's
265 // concerning we could be accessing it in the middle of a shutdown on the main-thread, eh?
266 // As a CDMProxy, we ***should*** be turned off before this pointer ever goes bad.
267 MediaPlayer* m_player { nullptr }; // FIXME: MainThread<T>?
268
269 std::atomic<int> m_numDecryptorsWaitingForKey { 0 };
270};
