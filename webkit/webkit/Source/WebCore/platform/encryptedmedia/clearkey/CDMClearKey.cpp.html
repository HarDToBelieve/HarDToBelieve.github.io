50static std::optional<Vector<RefPtr<KeyHandle>>> parseLicenseFormat(const JSON::Object& root)
51{
52 // If the 'keys' key is present in the root object, parse the JSON further
53 // according to the specified 'license' format.
54 auto it = root.find("keys");
55 if (it == root.end())
56 return std::nullopt;
57
58 // Retrieve the keys array.
59 auto keysArray = it->value->asArray();
60 if (!keysArray)
61 return std::nullopt;
62
63 Vector<RefPtr<KeyHandle>> decodedKeys;
64 bool validFormat = std::all_of(keysArray->begin(), keysArray->end(),
65 [&decodedKeys] (const auto& value) {
66 auto keyObject = value->asObject();
67 if (!keyObject)
68 return false;
69
70 auto keyType = keyObject->getString("kty");
71 if (!keyType || !equalLettersIgnoringASCIICase(keyType, "oct"))
72 return false;
73
74 auto keyID = keyObject->getString("kid");
75 if (!keyID)
76 return false;
77
78 auto keyValue = keyObject->getString("k");
79 if (!keyValue)
80 return false;
81
82 auto keyIDData = base64URLDecode(keyID);
83 if (!keyIDData)
84 return false;
85
86 auto keyHandleValueData = base64URLDecode(keyValue);
87 if (!keyHandleValueData)
88 return false;
89
90 decodedKeys.append(KeyHandle::create(CDMInstanceSession::KeyStatus::Usable, WTFMove(*keyIDData), WTFMove(*keyHandleValueData)));
91 return true;
92 });
93 if (!validFormat)
94 return std::nullopt;
95 return decodedKeys;
96}

115// https://www.w3.org/TR/eme-initdata-cenc/#common-system
116// 4.1 Definition
117// The SystemID is 1077efec-c0b2-4d02-ace3-3c1e52e2fb4b.
118// The PSSH box format is as follows. It follows version 1 of the 'pssh' box as defined in [CENC].
119// pssh = [
120// 0x00, 0x00, 0x00, 0x4c, 0x70, 0x73, 0x73, 0x68, // BMFF box header (76 bytes, 'pssh')
121// 0x01, 0x00, 0x00, 0x00, // Full box header (version = 1, flags = 0)
122// 0x10, 0x77, 0xef, 0xec, 0xc0, 0xb2, 0x4d, 0x02, // SystemID
123// 0xac, 0xe3, 0x3c, 0x1e, 0x52, 0xe2, 0xfb, 0x4b,
124// 0x00, 0x00, 0x00, 0x02, // KidCount (2)
125// 0x30, 0x31, 0x32, 0x33, 0x34, 0x35, 0x36, 0x37, // First KID ("0123456789012345")
126// 0x38, 0x39, 0x30, 0x31, 0x32, 0x33, 0x34, 0x35,
127// 0x41, 0x42, 0x43, 0x44, 0x45, 0x46, 0x47, 0x48, // Second KID ("ABCDEFGHIJKLMNOP")
128// 0x49, 0x4a, 0x4b, 0x4c, 0x4d, 0x4e, 0x4f, 0x50,
129// 0x00, 0x00, 0x00, 0x00, // Size of Data (0)
130// ];
131
132// This function extracts the KeyIds count and the location of the first KeyId in initData buffer.
133static std::pair<unsigned, unsigned> extractKeyidsLocationFromCencInitData(const SharedBuffer& initData)
134{
135 std::pair<unsigned, unsigned> keyIdsMap(0, 0);
136
137 // Check the initData size.
138 if (initData.isEmpty() || initData.size() > std::numeric_limits<unsigned>::max())
139 return keyIdsMap;
140
141 auto* data = initData.data();
142 unsigned initDataSize = initData.size();
143 unsigned index = 0;
144 unsigned psshSize = 0;
145
146 // Search in the concatenated or the simple InitData, the ClearKey PSSH.
147 bool foundPssh = false;
148 while (true) {
149
150 // Check the overflow InitData.
151 if (index + 12 + ClearKey::cencSystemIdSize >= initDataSize)
152 return keyIdsMap;
153
154 psshSize = data[index + 2] * 256 + data[index + 3];
155
156 // Check the pssh size
157 if (!psshSize)
158 return keyIdsMap;
159
160 // 12 = BMFF box header + Full box header.
161 if (!memcmp(&data[index + 12], ClearKey::cencSystemId, ClearKey::cencSystemIdSize)) {
162 foundPssh = true;
163 break;
164 }
165 index += psshSize;
166 }
167
168 // Check if the InitData contains the ClearKey PSSH.
169 if (!foundPssh)
170 return keyIdsMap;
171
172 index += (12 + ClearKey::cencSystemIdSize); // 12 (BMFF box header + Full box header) + SystemID size.
173
174 // Check the overflow.
175 if (index + 3 >= initDataSize)
176 return keyIdsMap;
177
178 keyIdsMap.first = data[index + 3]; // Read the KeyIdsCount.
179 index += 4; // KeyIdsCount size.
180
181 // Check the overflow.
182 if ((index + (keyIdsMap.first * ClearKey::KeyIDSizeInBytes)) >= initDataSize)
183 return keyIdsMap;
184
185 keyIdsMap.second = index; // The location of the first KeyId in initData.
186
187 return keyIdsMap;
188}

190// This function checks if the initData sharedBuffer is a valid CENC initData.
191static bool isCencInitData(const SharedBuffer& initData)
192{
193 std::pair<unsigned, unsigned> keyIdsMap = extractKeyidsLocationFromCencInitData(initData);
194 return ((keyIdsMap.first) && (keyIdsMap.second));
195}

197static Ref<SharedBuffer> extractKeyidsFromCencInitData(const SharedBuffer& initData)
198{
199 Ref<SharedBuffer> keyIds = SharedBuffer::create();
200
201 std::pair<unsigned, unsigned> keyIdsMap = extractKeyidsLocationFromCencInitData(initData);
202 unsigned keyIdCount = keyIdsMap.first;
203 unsigned index = keyIdsMap.second;
204
205 // Check if initData is a valid CENC initData.
206 if (!keyIdCount || !index)
207 return keyIds;
208
209 auto* data = initData.data();
210
211 auto object = JSON::Object::create();
212 auto keyIdsArray = JSON::Array::create();
213
214 // Read the KeyId
215 // 9.1.3 License Request Format
216 // This section describes the format of the license request provided to the application via the message attribute of the message event.
217 // The format is a JSON object containing the following members:
218 // "kids"
219 // An array of key IDs. Each element of the array is the base64url encoding of the octet sequence containing the key ID value.
220 for (unsigned i = 0; i < keyIdCount; i++) {
221 keyIdsArray->pushString(base64URLEncodeToString(&data[index], ClearKey::KeyIDSizeInBytes));
222 index += ClearKey::KeyIDSizeInBytes;
223 }
224
225 object->setArray("kids", WTFMove(keyIdsArray));
226 CString jsonData = object->toJSONString().utf8();
227 keyIds->append(jsonData.data(), jsonData.length());
228 return keyIds;
229}

231static Ref<SharedBuffer> extractKeyIdFromWebMInitData(const SharedBuffer& initData)
232{
233 Ref<SharedBuffer> keyIds = SharedBuffer::create();
234
235 // Check if initData is a valid WebM initData.
236 if (initData.isEmpty() || initData.size() > std::numeric_limits<unsigned>::max())
237 return keyIds;
238
239 auto object = JSON::Object::create();
240 auto keyIdsArray = JSON::Array::create();
241
242 // Read the KeyId
243 // 9.1.3 License Request Format
244 // This section describes the format of the license request provided to the application via the message attribute of the message event.
245 // The format is a JSON object containing the following members:
246 // "kids"
247 // An array of key IDs. Each element of the array is the base64url encoding of the octet sequence containing the key ID value.
248 keyIdsArray->pushString(base64URLEncodeToString(initData.data(), initData.size()));
249
250 object->setArray("kids", WTFMove(keyIdsArray));
251 CString jsonData = object->toJSONString().utf8();
252 keyIds->append(jsonData.data(), jsonData.length());
253 return keyIds;
254}

299bool CDMPrivateClearKey::supportsConfiguration(const CDMKeySystemConfiguration& configuration) const
300{
301 // Reject any configuration that marks distinctive identifier as required.
302 if (configuration.distinctiveIdentifier == CDMRequirement::Required)
303 return false;
304
305 // Reject any configuration that marks persistent state as required, unless
306 // the 'persistent-license' session type has to be supported.
307 if (configuration.persistentState == CDMRequirement::Required && !containsPersistentLicenseType(configuration.sessionTypes))
308 return false;
309
310 return true;
311}

313bool CDMPrivateClearKey::supportsConfigurationWithRestrictions(const CDMKeySystemConfiguration& configuration, const CDMRestrictions& restrictions) const
314{
315 // Reject any configuration that marks distincitive identifier as required, or that marks
316 // distinctive identifier as optional even when restrictions mark it as denied.
317 if ((configuration.distinctiveIdentifier == CDMRequirement::Optional && restrictions.distinctiveIdentifierDenied)
318 || configuration.distinctiveIdentifier == CDMRequirement::Required)
319 return false;
320
321 // Reject any configuration that marks persistent state as optional even when
322 // restrictions mark it as denied.
323 if (configuration.persistentState == CDMRequirement::Optional && restrictions.persistentStateDenied)
324 return false;
325
326 // Reject any configuration that marks persistent state as required, unless
327 // the 'persistent-license' session type has to be supported.
328 if (configuration.persistentState == CDMRequirement::Required && !containsPersistentLicenseType(configuration.sessionTypes))
329 return false;
330
331 return true;
332}

391bool CDMPrivateClearKey::supportsInitData(const AtomString& initDataType, const SharedBuffer& initData) const
392{
393 // Validate the initData buffer as an JSON object in keyids case.
394 if (equalLettersIgnoringASCIICase(initDataType, "keyids") && CDMUtilities::parseJSONObject(initData))
395 return true;
396
397 // Validate the initData buffer as CENC initData.
398 if (equalLettersIgnoringASCIICase(initDataType, "cenc") && isCencInitData(initData))
399 return true;
400
401 // Validate the initData buffer as WebM initData.
402 if (equalLettersIgnoringASCIICase(initDataType, "webm") && !initData.isEmpty())
403 return true;
404
405 return false;
406}

456void CDMInstanceSessionClearKey::requestLicense(LicenseType, const AtomString& initDataType, Ref<SharedBuffer>&& initData, LicenseCallback&& callback)
457{
458 static uint32_t s_sessionIdValue = 0;
459 ++s_sessionIdValue;
460
461 m_sessionID = String::number(s_sessionIdValue);
462
463 if (equalLettersIgnoringASCIICase(initDataType, "cenc"))
464 initData = extractKeyidsFromCencInitData(initData.get());
465
466 if (equalLettersIgnoringASCIICase(initDataType, "webm"))
467 initData = extractKeyIdFromWebMInitData(initData.get());
468
469 callOnMainThread(
470 [this, weakThis = makeWeakPtr(*this), callback = WTFMove(callback), initData = WTFMove(initData)]() mutable {
471 if (!weakThis)
472 return;
473
474 callback(WTFMove(initData), m_sessionID, false, Succeeded);
475 });
476}

478void CDMInstanceSessionClearKey::updateLicense(const String& sessionId, LicenseType, Ref<SharedBuffer>&& response, LicenseUpdateCallback&& callback)
479{
480#if LOG_DISABLED
481 // We only use the sesion ID for debug logging. The verbose preprocessor checks are because
482 // the Mac port has -Werror -Wunused-parameter.
483 UNUSED_PARAM(sessionId);
484#endif
485 auto dispatchCallback =
486 [weakThis = makeWeakPtr(*this), &callback](bool sessionWasClosed, std::optional<KeyStatusVector>&& changedKeys, SuccessValue succeeded) {
487 callOnMainThread(
488 [weakThis = makeWeakPtr(*weakThis), callback = WTFMove(callback), sessionWasClosed, changedKeys = WTFMove(changedKeys), succeeded] () mutable {
489 if (!weakThis)
490 return;
491
492 callback(sessionWasClosed, WTFMove(changedKeys), std::nullopt, std::nullopt, succeeded);
493 });
494 };
495
496 RefPtr<JSON::Object> root = CDMUtilities::parseJSONObject(response);
497 if (!root) {
498 LOG(EME, "EME - ClearKey - session %s update payload was not valid JSON", sessionId.utf8().data());
499 dispatchCallback(false, std::nullopt, SuccessValue::Failed);
500 return;
501 }
502
503 LOG(EME, "EME - ClearKey - updating license for session %s which currently contains %u keys", sessionId.utf8().data(), m_keyStore.numKeys());
504
505 if (auto decodedKeys = parseLicenseFormat(*root)) {
506 bool keysChanged = m_keyStore.addKeys(WTFMove(*decodedKeys));
507
508 LOG(EME, "EME - ClearKey - session %s has %u keys after update()", sessionId.utf8().data(), m_keyStore.numKeys());
509
510 std::optional<KeyStatusVector> changedKeys;
511 if (keysChanged) {
512 LOG(EME, "EME - ClearKey - session %s has changed keys", sessionId.utf8().data());
513 parentInstance().mergeKeysFrom(m_keyStore);
514 changedKeys = m_keyStore.convertToJSKeyStatusVector();
515 }
516
517 dispatchCallback(false, WTFMove(changedKeys), SuccessValue::Succeeded);
518 return;
519 }
520
521 if (parseLicenseReleaseAcknowledgementFormat(*root)) {
522 LOG(EME, "EME - ClearKey - session %s release acknowledged, clearing all known keys", sessionId.utf8().data());
523 parentInstance().unrefAllKeysFrom(m_keyStore);
524 m_keyStore.unrefAllKeys();
525 dispatchCallback(true, std::nullopt, SuccessValue::Succeeded);
526 return;
527 }
528
529 LOG(EME, "EME - ClearKey - session %s update payload was an unrecognized format", sessionId.utf8().data());
530 dispatchCallback(false, std::nullopt, SuccessValue::Failed);
531}

560void CDMInstanceSessionClearKey::removeSessionData(const String& sessionId, LicenseType, RemoveSessionDataCallback&& callback)
561{
562#ifdef NDEBUG
563 UNUSED_PARAM(sessionId);
564#endif
565
566 ASSERT(sessionId == m_sessionID);
567
568 auto dispatchCallback =
569 [weakThis = makeWeakPtr(*this), &callback](KeyStatusVector&& keyStatusVector, std::optional<Ref<SharedBuffer>>&& message, SuccessValue success) {
570 callOnMainThread(
571 [weakThis = makeWeakPtr(*weakThis), callback = WTFMove(callback), keyStatusVector = WTFMove(keyStatusVector), message = WTFMove(message), success]() mutable {
572 if (!weakThis)
573 return;
574
575 callback(WTFMove(keyStatusVector), WTFMove(message), success);
576 });
577 };
578
579 // Construct the KeyStatusVector object, representing released keys, and the message in the
580 // 'license release' format.
581 KeyStatusVector keyStatusVector = m_keyStore.allKeysAs(CDMInstanceSession::KeyStatus::Released);
582 RefPtr<SharedBuffer> message;
583 {
584 // Construct JSON that represents the 'license release' format, creating a 'kids' array
585 // of base64URL-encoded key IDs for all keys that were associated with this session.
586 auto rootObject = JSON::Object::create();
587 {
588 auto array = JSON::Array::create();
589 for (const auto& key : m_keyStore) {
590 ASSERT(key->id().size() <= std::numeric_limits<unsigned>::max());
591 array->pushString(base64URLEncodeToString(key->id().data(), key->id().size()));
592 }
593 rootObject->setArray("kids", WTFMove(array));
594 }
595
596 // Copy the JSON data into a SharedBuffer object.
597 String messageString = rootObject->toJSONString();
598 CString messageCString = messageString.utf8();
599 message = SharedBuffer::create(messageCString.data(), messageCString.length());
600 }
601
602 m_keyStore.unrefAllKeys();
603 dispatchCallback(WTFMove(keyStatusVector), Ref<SharedBuffer>(*message), SuccessValue::Succeeded);
604}
