45bool GameControllerGamepadProvider::willHandleVendorAndProduct(uint16_t vendorID, uint16_t productID)
46{
47 // Check the vendor/product ID pair against a hard coded list of controllers we expect to work
48 // well with GameController.framework on macOS 10.15.
49 switch ((static_cast<uint32_t>(vendorID) << 16) | productID) {
50 case Dualshock4_1:
51 case Dualshock4_2:
52 case GamesirM2:
53 case HoripadUltimate:
54 case Nimbus1:
55 case Nimbus2:
56 case StratusXL1:
57 case StratusXL2:
58 case StratusXL3:
59 case StratusXL4:
60 case XboxOne1:
61 case XboxOne2:
62 case XboxOne3:
63 return true;
64 default:
65 return false;
66 }
67}
68
69#endif // !HAVE(GCCONTROLLER_HID_DEVICE_CHECK)

84void GameControllerGamepadProvider::controllerDidConnect(GCController *controller, ConnectionVisibility visibility)
85{
86 LOG(Gamepad, "GameControllerGamepadProvider controller %p added", controller);
87
88#if HAVE(MULTIGAMEPADPROVIDER_SUPPORT) && !HAVE(GCCONTROLLER_HID_DEVICE_CHECK)
89 // On macOS 10.15, we use GameController framework for some controllers, but it's much too aggressive in handling devices it shouldn't.
90 // So we check Vendor/Product against an explicit allow-list to determine if we should let GCF handle the device.
91 // (We have the opposite check in HIDGamepadProvider, as well)
92 for (_GCCControllerHIDServiceInfo *serviceInfo in controller.hidServices) {
93 if (!serviceInfo.service)
94 continue;
95
96 auto cfVendorID = adoptCF((CFNumberRef)IOHIDServiceClientCopyProperty(serviceInfo.service, (__bridge CFStringRef)@(kIOHIDVendorIDKey)));
97 auto cfProductID = adoptCF((CFNumberRef)IOHIDServiceClientCopyProperty(serviceInfo.service, (__bridge CFStringRef)@(kIOHIDProductIDKey)));
98
99 int vendorID, productID;
100 CFNumberGetValue(cfVendorID.get(), kCFNumberIntType, &vendorID);
101 CFNumberGetValue(cfProductID.get(), kCFNumberIntType, &productID);
102
103 if (!willHandleVendorAndProduct(vendorID, productID)) {
104 LOG(Gamepad, "GameControllerGamepadProvider::controllerDidConnect - GCController service does not match known VID/PID. Letting the HID provider handle it.");
105 return;
106 }
107 }
108#endif // HAVE(MULTIGAMEPADPROVIDER_SUPPORT) && !HAVE(GCCONTROLLER_HID_DEVICE_CHECK)
109
110
111 // When initially starting up the GameController framework machinery,
112 // we might get the connection notification for an already-connected controller.
113 if (m_gamepadMap.contains(controller))
114 return;
115
116 unsigned index = indexForNewlyConnectedDevice();
117 auto gamepad = makeUnique<GameControllerGamepad>(controller, index);
118
119 if (m_gamepadVector.size() <= index)
120 m_gamepadVector.grow(index + 1);
121
122 m_gamepadVector[index] = gamepad.get();
123 m_gamepadMap.set((__bridge CFTypeRef)controller, WTFMove(gamepad));
124
125
126 if (visibility == ConnectionVisibility::Invisible) {
127 m_invisibleGamepads.add(m_gamepadVector[index]);
128 return;
129 }
130
131 makeInvisibleGamepadsVisible();
132
133 for (auto& client : m_clients)
134 client->platformGamepadConnected(*m_gamepadVector[index], EventMakesGamepadsVisible::Yes);
135}

137void GameControllerGamepadProvider::controllerDidDisconnect(GCController *controller)
138{
139 LOG(Gamepad, "GameControllerGamepadProvider controller %p removed", controller);
140
141 auto removedGamepad = m_gamepadMap.take((__bridge CFTypeRef)controller);
142 ASSERT(removedGamepad);
143
144 auto i = m_gamepadVector.find(removedGamepad.get());
145 if (i != notFound)
146 m_gamepadVector[i] = nullptr;
147
148 m_invisibleGamepads.remove(removedGamepad.get());
149
150 for (auto& client : m_clients)
151 client->platformGamepadDisconnected(*removedGamepad);
152}

154void GameControllerGamepadProvider::prewarmGameControllerDevicesIfNecessary()
155{
156 static bool prewarmed;
157 if (prewarmed)
158 return;
159
160 LOG(Gamepad, "GameControllerGamepadProvider explicitly starting GameController framework monitoring");
161 [getGCControllerClass() __openXPC_and_CBApplicationDidBecomeActive__];
162 prewarmed = true;
163}

165void GameControllerGamepadProvider::startMonitoringGamepads(GamepadProviderClient& client)
166{
167 ASSERT(!m_clients.contains(&client));
168 m_clients.add(&client);
169
170 if (m_connectObserver)
171 return;
172
173 prewarmGameControllerDevicesIfNecessary();
174
175 if (canLoad_GameController_GCControllerDidConnectNotification()) {
176 m_connectObserver = [[NSNotificationCenter defaultCenter] addObserverForName:get_GameController_GCControllerDidConnectNotification() object:nil queue:nil usingBlock:^(NSNotification *notification) {
177 LOG(Gamepad, "GameControllerGamepadProvider notified of new GCController %p", notification.object);
178 GameControllerGamepadProvider::singleton().controllerDidConnect(notification.object, ConnectionVisibility::Visible);
179 }];
180 }
181
182 if (canLoad_GameController_GCControllerDidDisconnectNotification()) {
183 m_disconnectObserver = [[NSNotificationCenter defaultCenter] addObserverForName:get_GameController_GCControllerDidDisconnectNotification() object:nil queue:nil usingBlock:^(NSNotification *notification) {
184 LOG(Gamepad, "GameControllerGamepadProvider notified of disconnected GCController %p", notification.object);
185 GameControllerGamepadProvider::singleton().controllerDidDisconnect(notification.object);
186 }];
187 }
188
189 auto *controllers = [getGCControllerClass() controllers];
190 if (!controllers || !controllers.count)
191 LOG(Gamepad, "GameControllerGamepadProvider has no initial GCControllers attached");
192
193 for (GCController *controller in controllers) {
194 LOG(Gamepad, "GameControllerGamepadProvider has initial GCController %p", controller);
195 controllerDidConnect(controller, ConnectionVisibility::Invisible);
196 }
197}
198
199void GameControllerGamepadProvider::stopMonitoringGamepads(GamepadProviderClient& client)
200{
201 ASSERT(m_clients.contains(&client));
202 m_clients.remove(&client);
203
204 if (!m_connectObserver || !m_clients.isEmpty())
205 return;
206
207 [[NSNotificationCenter defaultCenter] removeObserver:m_connectObserver.get()];
208 [[NSNotificationCenter defaultCenter] removeObserver:m_disconnectObserver.get()];
209}
210
211unsigned GameControllerGamepadProvider::indexForNewlyConnectedDevice()
212{
213 unsigned index = 0;
214 while (index < m_gamepadVector.size() && m_gamepadVector[index])
215 ++index;
216
217 return index;
218}

220void GameControllerGamepadProvider::gamepadHadInput(GameControllerGamepad&, bool hadButtonPresses)
221{
222 if (!m_inputNotificationTimer.isActive())
223 m_inputNotificationTimer.startOneShot(inputNotificationDelay);
224
225 if (hadButtonPresses)
226 m_shouldMakeInvisibleGamepadsVisible = true;
227}
228
229void GameControllerGamepadProvider::makeInvisibleGamepadsVisible()
230{
231 for (auto* gamepad : m_invisibleGamepads) {
232 for (auto& client : m_clients)
233 client->platformGamepadConnected(*gamepad, EventMakesGamepadsVisible::Yes);
234 }
235
236 m_invisibleGamepads.clear();
237}
238
239void GameControllerGamepadProvider::inputNotificationTimerFired()
240{
241 if (m_shouldMakeInvisibleGamepadsVisible) {
242 setShouldMakeGamepadsVisibile();
243 makeInvisibleGamepadsVisible();
244 }
245
246 m_shouldMakeInvisibleGamepadsVisible = false;
247
248 dispatchPlatformGamepadInputActivity();
249}
