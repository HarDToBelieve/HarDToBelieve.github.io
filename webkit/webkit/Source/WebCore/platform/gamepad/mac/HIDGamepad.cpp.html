46std::unique_ptr<HIDGamepad> HIDGamepad::create(IOHIDDeviceRef rawDevice, unsigned index)
47{
48 auto device = HIDDevice { rawDevice };
49
50 std::unique_ptr<HIDGamepad> newGamepad;
51
52 switch ((KnownGamepad)device.fullProductIdentifier()) {
53 case Dualshock3:
54 newGamepad = makeUnique<Dualshock3HIDGamepad>(WTFMove(device), index);
55 break;
56 case LogitechF310:
57 case LogitechF710:
58 newGamepad = makeUnique<LogitechGamepad>(WTFMove(device), index);
59 break;
60 case StadiaA:
61 newGamepad = makeUnique<StadiaHIDGamepad>(WTFMove(device), index);
62 break;
63 default:
64 newGamepad = makeUnique<GenericHIDGamepad>(WTFMove(device), index);
65 }
66
67 newGamepad->initialize();
68 return newGamepad;
69}

71HIDGamepad::HIDGamepad(HIDDevice&& device, unsigned index)
72 : PlatformGamepad(index)
73 , m_device(WTFMove(device))
74{
75 m_connectTime = m_lastUpdateTime = MonotonicTime::now();
76
77 // Currently the spec has no formatting for the id string.
78 // This string formatting matches Firefox.
79 m_id = makeString(hex(m_device.vendorID(), Lowercase), '-', hex(m_device.productID(), Lowercase), '-', m_device.productName());
80}
81
82void HIDGamepad::initialize()
83{
84 for (auto& element : m_elementMap.values())
85 element->refreshCurrentValue();
86}
87
88HIDInputType HIDGamepad::valueChanged(IOHIDValueRef value)
89{
90 IOHIDElementCookie cookie = IOHIDElementGetCookie(IOHIDValueGetElement(value));
91 HIDGamepadElement* element = m_elementMap.get(cookie);
92
93 // This might be an element we don't currently handle as input so we can skip it.
94 if (!element)
95 return HIDInputType::NotAButtonPress;
96
97 m_lastUpdateTime = MonotonicTime::now();
98
99 return element->gamepadValueChanged(value);
100}
