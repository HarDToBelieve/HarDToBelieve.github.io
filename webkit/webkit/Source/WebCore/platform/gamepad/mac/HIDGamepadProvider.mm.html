48static RetainPtr<CFDictionaryRef> deviceMatchingDictionary(uint32_t usagePage, uint32_t usage)
49{
50 ASSERT(usagePage);
51 ASSERT(usage);
52
53 RetainPtr<CFNumberRef> pageNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &usagePage));
54 RetainPtr<CFNumberRef> usageNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &usage));
55
56 CFStringRef keys[] = { CFSTR(kIOHIDDeviceUsagePageKey), CFSTR(kIOHIDDeviceUsageKey) };
57 CFNumberRef values[] = { pageNumber.get(), usageNumber.get() };
58
59 return adoptCF(CFDictionaryCreate(kCFAllocatorDefault, (const void**)keys, (const void**)values, 2, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
60}

62static void deviceAddedCallback(void* context, IOReturn, void*, IOHIDDeviceRef device)
63{
64 HIDGamepadProvider* listener = static_cast<HIDGamepadProvider*>(context);
65 listener->deviceAdded(device);

68static void deviceRemovedCallback(void* context, IOReturn, void*, IOHIDDeviceRef device)
69{
70 HIDGamepadProvider* listener = static_cast<HIDGamepadProvider*>(context);
71 listener->deviceRemoved(device);
72}

74static void deviceValuesChangedCallback(void* context, IOReturn result, void*, IOHIDValueRef value)
75{
76 // A non-zero result value indicates an error that we can do nothing about for input values.
77 if (result)
78 return;
79
80 HIDGamepadProvider* listener = static_cast<HIDGamepadProvider*>(context);
81 listener->valuesChanged(value);
82}

90HIDGamepadProvider::HIDGamepadProvider()
91 : m_initialGamepadsConnectedTimer(*this, &HIDGamepadProvider::initialGamepadsConnectedTimerFired)
92 , m_inputNotificationTimer(*this, &HIDGamepadProvider::inputNotificationTimerFired)
93{
94 m_manager = adoptCF(IOHIDManagerCreate(kCFAllocatorDefault, kIOHIDOptionsTypeNone));
95
96 RetainPtr<CFDictionaryRef> joystickDictionary = deviceMatchingDictionary(kHIDPage_GenericDesktop, kHIDUsage_GD_Joystick);
97 RetainPtr<CFDictionaryRef> gamepadDictionary = deviceMatchingDictionary(kHIDPage_GenericDesktop, kHIDUsage_GD_GamePad);
98
99 CFDictionaryRef devices[] = { joystickDictionary.get(), gamepadDictionary.get() };
100
101 RetainPtr<CFArrayRef> matchingArray = adoptCF(CFArrayCreate(kCFAllocatorDefault, (const void**)devices, 2, &kCFTypeArrayCallBacks));
102
103 IOHIDManagerSetDeviceMatchingMultiple(m_manager.get(), matchingArray.get());
104 IOHIDManagerRegisterDeviceMatchingCallback(m_manager.get(), deviceAddedCallback, this);
105 IOHIDManagerRegisterDeviceRemovalCallback(m_manager.get(), deviceRemovedCallback, this);
106
107 startMonitoringInput();
108}

113 IOHIDManagerRegisterInputValueCallback(m_manager.get(), nullptr, nullptr);
114 IGNORE_NULL_CHECK_WARNINGS_END

117void HIDGamepadProvider::startMonitoringInput()

122unsigned HIDGamepadProvider::indexForNewlyConnectedDevice()
123{
124 unsigned index = 0;
125 while (index < m_gamepadVector.size() && m_gamepadVector[index])
126 ++index;
127
128 return index;
129}

136void HIDGamepadProvider::openAndScheduleManager()
137{
138 LOG(Gamepad, "HIDGamepadProvider opening/scheduling HID manager");
139
140 ASSERT(m_gamepadVector.isEmpty());
141 ASSERT(m_gamepadMap.isEmpty());
142
143 m_initialGamepadsConnected = false;
144
145 IOHIDManagerScheduleWithRunLoop(m_manager.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
146 IOHIDManagerOpen(m_manager.get(), kIOHIDOptionsTypeNone);
147
148 // Any connections we are notified of within the connectionDelayInterval of listening likely represent
149 // devices that were already connected, so we suppress notifying clients of these.
150 m_initialGamepadsConnectedTimer.startOneShot(connectionDelayInterval);
151}
152
153void HIDGamepadProvider::closeAndUnscheduleManager()
154{
155 LOG(Gamepad, "HIDGamepadProvider closing/unscheduling HID manager");
156
157 IOHIDManagerUnscheduleFromRunLoop(m_manager.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
158 IOHIDManagerClose(m_manager.get(), kIOHIDOptionsTypeNone);
159
160 m_gamepadVector.clear();
161 m_gamepadMap.clear();
162
163 m_initialGamepadsConnectedTimer.stop();
164}

166void HIDGamepadProvider::startMonitoringGamepads(GamepadProviderClient& client)
167{
168 bool shouldOpenAndScheduleManager = m_clients.isEmpty();
169
170 ASSERT(!m_clients.contains(&client));
171 m_clients.add(&client);
172
173 if (shouldOpenAndScheduleManager)
174 openAndScheduleManager();
175}
176void HIDGamepadProvider::stopMonitoringGamepads(GamepadProviderClient& client)
177{
178 ASSERT(m_clients.contains(&client));
179
180 bool shouldCloseAndUnscheduleManager = m_clients.remove(&client) && m_clients.isEmpty();
181
182 if (shouldCloseAndUnscheduleManager)
183 closeAndUnscheduleManager();
184}

188enum class GameControllerFrameworkHandlesDevice : uint8_t {
189 No,
190 Yes,
191};
192
193static GameControllerFrameworkHandlesDevice gameControllerFrameworkWillHandleHIDDevice(IOHIDDeviceRef device)
194{
195 if (!isGameControllerFrameworkAvailable())
196 return GameControllerFrameworkHandlesDevice::No;
197
198#if HAVE(GCCONTROLLER_HID_DEVICE_CHECK)
199 return [getGCControllerClass() supportsHIDDevice:device] ? GameControllerFrameworkHandlesDevice::Yes : GameControllerFrameworkHandlesDevice::No;
200#else
201 CFNumberRef cfVendorID = (CFNumberRef)IOHIDDeviceGetProperty(device, CFSTR(kIOHIDVendorIDKey));
202 CFNumberRef cfProductID = (CFNumberRef)IOHIDDeviceGetProperty(device, CFSTR(kIOHIDProductIDKey));
203
204 int vendorID = ((NSNumber *)cfVendorID).intValue;
205 int productID = ((NSNumber *)cfProductID).intValue;
206
207 if (vendorID < 0 || vendorID > std::numeric_limits<uint16_t>::max() || productID < 0 || productID > std::numeric_limits<uint16_t>::max())
208 return GameControllerFrameworkHandlesDevice::No;
209
210 return GameControllerGamepadProvider::willHandleVendorAndProduct((uint16_t)vendorID, (uint16_t)productID) ? GameControllerFrameworkHandlesDevice::Yes : GameControllerFrameworkHandlesDevice::No;
211#endif // HAVE(GCCONTROLLER_HID_DEVICE_CHECK)
212}

214#endif // HAVE(MULTIGAMEPADPROVIDER_SUPPORT)

216void HIDGamepadProvider::deviceAdded(IOHIDDeviceRef device)
217{
218#if HAVE(MULTIGAMEPADPROVIDER_SUPPORT)
219 if (m_ignoresGameControllerFrameworkDevices) {
220 auto result = gameControllerFrameworkWillHandleHIDDevice(device);
221 switch (result) {
222 case GameControllerFrameworkHandlesDevice::Yes:
223 LOG(Gamepad, "GameController framework will handle newly attached device %p - HIDGamepadProvider ignoring it", device);
224 m_gameControllerManagedGamepads.add(device);
225 return;
226 case GameControllerFrameworkHandlesDevice::No:
227 LOG(Gamepad, "GameController framework does not handle attached device %p - HIDGamepadProvider will handle it", device);
228 break;
229 }
230 }
231#endif
232
233 // HID sometimes notifies us multiple times for the same device.
234 if (m_gamepadMap.contains(device))
235 return;
236
237 LOG(Gamepad, "HIDGamepadProvider device %p added", device);
238
239 unsigned index = indexForNewlyConnectedDevice();
240 auto gamepad = HIDGamepad::create(device, index);
241
242 if (m_gamepadVector.size() <= index)
243 m_gamepadVector.grow(index + 1);
244
245 m_gamepadVector[index] = gamepad.get();
246 m_gamepadMap.set(device, WTFMove(gamepad));
247
248 if (!m_initialGamepadsConnected) {
249 // This added device is the result of us starting to monitor gamepads.
250 // We'll get notified of all connected devices during this current spin of the runloop
251 // and the client should be told they were already connected.
252 // The m_initialGamepadsConnectedTimer fires in a subsequent spin of the runloop after which
253 // any connection events are actual new devices and can trigger gamepad visibility.
254 if (!m_initialGamepadsConnectedTimer.isActive())
255 m_initialGamepadsConnectedTimer.startOneShot(0_s);
256 }
257
258 auto eventVisibility = m_initialGamepadsConnected ? EventMakesGamepadsVisible::Yes : EventMakesGamepadsVisible::No;
259 for (auto& client : m_clients)
260 client->platformGamepadConnected(*m_gamepadVector[index], eventVisibility);
261
262 // If we are working together with the GameController provider, let it know
263 // that gamepads should now be visible.
264 if (m_ignoresGameControllerFrameworkDevices && eventVisibility == EventMakesGamepadsVisible::Yes)
265 GameControllerGamepadProvider::singleton().makeInvisibleGamepadsVisible();
266}

268void HIDGamepadProvider::deviceRemoved(IOHIDDeviceRef device)
269{
270 std::unique_ptr<HIDGamepad> removedGamepad = removeGamepadForDevice(device);
271
272 if (!removedGamepad) {
273#if HAVE(MULTIGAMEPADPROVIDER_SUPPORT)
274 auto taken = m_gameControllerManagedGamepads.take(device);
275 ASSERT_UNUSED(taken, taken);
276 LOG(Gamepad, "HIDGamepadProvider informed of removal of device %p, which is managed by GameController framework. Ignoring.", device);
277#endif
278 return;
279 }
280
281 LOG(Gamepad, "HIDGamepadProvider device %p removed", device);
282
283 for (auto& client : m_clients)
284 client->platformGamepadDisconnected(*removedGamepad);
285}

287void HIDGamepadProvider::valuesChanged(IOHIDValueRef value)
288{
289 IOHIDDeviceRef device = IOHIDElementGetDevice(IOHIDValueGetElement(value));
290
291 HIDGamepad* gamepad = m_gamepadMap.get(device);
292
293 // When starting monitoring we might get a value changed callback before we even know the device is connected.
294 if (!gamepad)
295 return;
296
297 if (gamepad->valueChanged(value) == HIDInputType::ButtonPress)
298 setShouldMakeGamepadsVisibile();
299
300 // This isActive check is necessary as we want to delay input notifications from the time of the first input,
301 // and not push the notification out on every subsequent input.
302 if (!m_inputNotificationTimer.isActive())
303 m_inputNotificationTimer.startOneShot(hidInputNotificationDelay);
304}
305
306void HIDGamepadProvider::inputNotificationTimerFired()
307{
308 if (!m_initialGamepadsConnected) {
309 if (!m_inputNotificationTimer.isActive())
310 m_inputNotificationTimer.startOneShot(0_s);
311 return;
312 }
313
314 dispatchPlatformGamepadInputActivity();
315}

317std::unique_ptr<HIDGamepad> HIDGamepadProvider::removeGamepadForDevice(IOHIDDeviceRef device)
318{
319 std::unique_ptr<HIDGamepad> result = m_gamepadMap.take(device);
320 if (!result)
321 return nullptr;
322
323 auto i = m_gamepadVector.find(result.get());
324 if (i != notFound)
325 m_gamepadVector[i] = nullptr;
326
327 return result;
328}
