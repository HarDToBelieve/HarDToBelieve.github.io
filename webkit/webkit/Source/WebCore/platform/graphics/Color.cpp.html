36static constexpr auto lightenedBlack = SRGBA<uint8_t> { 84, 84, 84 };

39Color::Color(const Color& other)
40 : m_colorAndFlags(other.m_colorAndFlags)
41{
42 if (isOutOfLine())

48 *this = WTFMove(other);

51Color& Color::operator=(const Color& other)
52{
53 if (*this == other)
54 return *this;
55
56 if (isOutOfLine())
57 asOutOfLine().deref();
58
59 m_colorAndFlags = other.m_colorAndFlags;
60
61 if (isOutOfLine())
62 asOutOfLine().ref();
63
64 return *this;
65}

67Color& Color::operator=(Color&& other)
68{
69 if (*this == other)
70 return *this;
71
72 if (isOutOfLine())
73 asOutOfLine().deref();
74
75 m_colorAndFlags = other.m_colorAndFlags;
76 other.m_colorAndFlags = invalidColorAndFlags;
77
78 return *this;
79}

81Color Color::lightened() const
82{
83 // Hardcode this common case for speed.
84 if (isInline() && asInline() == black)
85 return lightenedBlack;
86
87 auto [r, g, b, a] = toSRGBALossy<float>();
88 float v = std::max({ r, g, b });
89
90 if (v == 0.0f)
91 return lightenedBlack.colorWithAlphaByte(alphaByte());
92
93 float multiplier = std::min(1.0f, v + 0.33f) / v;
94
95 return convertColor<SRGBA<uint8_t>>(SRGBA<float> { multiplier * r, multiplier * g, multiplier * b, a });
96}
97
98Color Color::darkened() const
99{
100 // Hardcode this common case for speed.
101 if (isInline() && asInline() == white)
102 return darkenedWhite;
103
104 auto [r, g, b, a] = toSRGBALossy<float>();
105
106 float v = std::max({ r, g, b });
107 float multiplier = std::max(0.0f, (v - 0.33f) / v);
108
109 return convertColor<SRGBA<uint8_t>>(SRGBA<float> { multiplier * r, multiplier * g, multiplier * b, a });
110}

112double Color::lightness() const
113{
114 // FIXME: Replace remaining uses with luminance.
115 auto [r, g, b, a] = toSRGBALossy<float>();
116 auto [min, max] = std::minmax({ r, g, b });
117 return 0.5 * (max + min);
118}

120double Color::luminance() const

127Color Color::colorWithAlpha(float alpha) const
128{
129 return callOnUnderlyingType([&] (const auto& underlyingColor) -> Color {
130 auto result = colorWithOverriddenAlpha(underlyingColor, alpha);
131
132 // FIXME: Why is preserving the semantic bit desired and/or correct here?
133 if (isSemantic())
134 return { result, Flags::Semantic };
135
136 return { result };
137 });
138}
139
140Color Color::invertedColorWithAlpha(float alpha) const
141{
142 return callOnUnderlyingType([&] (const auto& underlyingColor) -> Color {
143 using ColorType = std::decay_t<decltype(underlyingColor)>;
144
145 // FIXME: Determine if there is a meaningful understanding of inversion that works
146 // better for non-invertible color types like Lab or consider removing this in favor
147 // of alternatives.
148 if constexpr (ColorType::Model::isInvertible)
149 return invertedColorWithOverriddenAlpha(underlyingColor, alpha);
150 else
151 return invertedColorWithOverriddenAlpha(convertColor<SRGBA<float>>(underlyingColor), alpha);
152 });
153}

155Color Color::semanticColor() const
156{
157 if (isSemantic())
158 return *this;
159
160 if (isOutOfLine())
161 return { asOutOfLineRef(), colorSpace(), Flags::Semantic };
162 return { asInline(), Flags::Semantic };
163}

165ColorComponents<float, 4> Color::toColorComponentsInColorSpace(ColorSpace outputColorSpace) const
166{
167 auto [inputColorSpace, components] = colorSpaceAndComponents();
168 return converColorComponents(inputColorSpace, components, outputColorSpace);
169}

171ColorComponents<float, 4> Color::toColorComponentsInColorSpace(const DestinationColorSpace& outputColorSpace) const
172{
173 auto [inputColorSpace, components] = colorSpaceAndComponents();
174 return converColorComponents(inputColorSpace, components, outputColorSpace);
175}

177std::pair<ColorSpace, ColorComponents<float, 4>> Color::colorSpaceAndComponents() const
178{
179 if (isOutOfLine())
180 return { colorSpace(), asOutOfLine().components() };
181 return { ColorSpace::SRGB, asColorComponents(convertColor<SRGBA<float>>(asInline())) };
182}
