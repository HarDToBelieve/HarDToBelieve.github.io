275inline bool operator==(const Color& a, const Color& b)
276{
277 if (a.isOutOfLine() || b.isOutOfLine())
278 return outOfLineComponentssEqual(a, b);
279 return a.m_colorAndFlags == b.m_colorAndFlags;
280}
281
282inline bool operator!=(const Color& a, const Color& b)
283{
284 return !(a == b);
285}
286
287inline bool outOfLineComponentssEqual(const Color& a, const Color& b)
288{
289 if (a.isOutOfLine() && b.isOutOfLine())
290 return a.asOutOfLine().components() == b.asOutOfLine().components() && a.colorSpace() == b.colorSpace() && a.flags() == b.flags();
291
292 ASSERT(a.isOutOfLine() || b.isOutOfLine());
293 return false;
294}
295
296inline bool outOfLineComponentssEqualIgnoringSemanticColor(const Color& a, const Color& b)
297{
298 if (a.isOutOfLine() && b.isOutOfLine()) {
299 auto aFlags = a.flags() - Color::FlagsIncludingPrivate::Semantic;
300 auto bFlags = b.flags() - Color::FlagsIncludingPrivate::Semantic;
301 return a.asOutOfLine().components() == b.asOutOfLine().components() && a.colorSpace() == b.colorSpace() && aFlags == bFlags;
302 }
303
304 ASSERT(a.isOutOfLine() || b.isOutOfLine());
305 return false;
306}
307
308inline bool equalIgnoringSemanticColor(const Color& a, const Color& b)
309{
310 if (a.isOutOfLine() || b.isOutOfLine())
311 return outOfLineComponentssEqualIgnoringSemanticColor(a, b);
312
313 auto aFlags = a.flags() - Color::FlagsIncludingPrivate::Semantic;
314 auto bFlags = b.flags() - Color::FlagsIncludingPrivate::Semantic;
315 return a.asPackedInline().value == b.asPackedInline().value && aFlags == bFlags;
316}

547template<class Encoder> void Color::encode(Encoder& encoder) const
548{
549 if (!isValid()) {
550 encoder << false;
551 return;
552 }
553 encoder << true;
554
555 encoder << flags().contains(FlagsIncludingPrivate::Semantic);
556 encoder << flags().contains(FlagsIncludingPrivate::UseColorFunctionSerialization);
557
558 if (isOutOfLine()) {
559 encoder << true;
560 encoder << colorSpace();
561
562 auto& outOfLineComponents = asOutOfLine();
563 auto [c1, c2, c3, alpha] = outOfLineComponents.components();
564 encoder << c1;
565 encoder << c2;
566 encoder << c3;
567 encoder << alpha;
568 return;
569 }
570 encoder << false;
571
572 encoder << asPackedInline().value;
573}

575template<class Decoder> std::optional<Color> Color::decode(Decoder& decoder)
576{
577 bool isValid;
578 if (!decoder.decode(isValid))
579 return std::nullopt;
580
581 if (!isValid)
582 return Color { };
583
584 OptionSet<Flags> flags;
585
586 bool isSemantic;
587 if (!decoder.decode(isSemantic))
588 return std::nullopt;
589
590 if (isSemantic)
591 flags.add(Flags::Semantic);
592
593 bool usesColorFunctionSerialization;
594 if (!decoder.decode(usesColorFunctionSerialization))
595 return std::nullopt;
596
597 if (usesColorFunctionSerialization)
598 flags.add(Flags::UseColorFunctionSerialization);
599
600 bool isOutOfLine;
601 if (!decoder.decode(isOutOfLine))
602 return std::nullopt;
603
604 if (isOutOfLine) {
605 ColorSpace colorSpace;
606 if (!decoder.decode(colorSpace))
607 return std::nullopt;
608 float c1;
609 if (!decoder.decode(c1))
610 return std::nullopt;
611 float c2;
612 if (!decoder.decode(c2))
613 return std::nullopt;
614 float c3;
615 if (!decoder.decode(c3))
616 return std::nullopt;
617 float alpha;
618 if (!decoder.decode(alpha))
619 return std::nullopt;
620 return Color { OutOfLineComponents::create({ c1, c2, c3, alpha }), colorSpace, flags };
621 }
622
623 uint32_t value;
624 if (!decoder.decode(value))
625 return std::nullopt;
626
627 return Color { asSRGBA(PackedColor::RGBA { value }), flags };
628}
