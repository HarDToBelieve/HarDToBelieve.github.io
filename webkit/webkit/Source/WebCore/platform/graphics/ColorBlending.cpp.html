34Color blendSourceOver(const Color& backdrop, const Color& source)
35{
36 if (!backdrop.isVisible() || source.isOpaque())
37 return source;
38
39 if (!source.isVisible())
40 return backdrop;
41
42 auto [backdropR, backdropG, backdropB, backdropA] = backdrop.toSRGBALossy<uint8_t>();
43 auto [sourceR, sourceG, sourceB, sourceA] = source.toSRGBALossy<uint8_t>();
44
45 int d = 0xFF * (backdropA + sourceA) - backdropA * sourceA;
46 int a = d / 0xFF;
47 int r = (backdropR * backdropA * (0xFF - sourceA) + 0xFF * sourceA * sourceR) / d;
48 int g = (backdropG * backdropA * (0xFF - sourceA) + 0xFF * sourceA * sourceG) / d;
49 int b = (backdropB * backdropA * (0xFF - sourceA) + 0xFF * sourceA * sourceB) / d;
50
51 return makeFromComponentsClamping<SRGBA<uint8_t>>(r, g, b, a);
52}

54Color blendWithWhite(const Color& color)
55{
56 constexpr int startAlpha = 153; // 60%
57 constexpr int endAlpha = 204; // 80%;
58 constexpr int alphaIncrement = 17;
59
60 auto blendComponent = [](int c, int a) -> int {
61 float alpha = a / 255.0f;
62 int whiteBlend = 255 - a;
63 c -= whiteBlend;
64 return static_cast<int>(c / alpha);
65 };
66
67 // If the color contains alpha already, we leave it alone.
68 if (!color.isOpaque())
69 return color;
70
71 auto [existingR, existingG, existingB, existingAlpha] = color.toSRGBALossy<uint8_t>();
72
73 SRGBA<uint8_t> result;
74 for (int alpha = startAlpha; alpha <= endAlpha; alpha += alphaIncrement) {
75 // We have a solid color. Convert to an equivalent color that looks the same when blended with white
76 // at the current alpha. Try using less transparency if the numbers end up being negative.
77 int r = blendComponent(existingR, alpha);
78 int g = blendComponent(existingG, alpha);
79 int b = blendComponent(existingB, alpha);
80
81 result = makeFromComponentsClamping<SRGBA<uint8_t>>(r, g, b, alpha);
82
83 if (r >= 0 && g >= 0 && b >= 0)
84 break;
85 }
86
87 // FIXME: Why is preserving the semantic bit desired and/or correct here?
88 if (color.isSemantic())
89 return { result, Color::Flags::Semantic };
90 return result;
91}

93Color blend(const Color& from, const Color& to, const BlendingContext& context)
94{
95 // FIXME: ExtendedColor - needs to handle color spaces.
96 // We need to preserve the state of the valid flag at the end of the animation
97 if (context.progress == 1 && !to.isValid())
98 return { };
99
100 auto premultipliedFrom = premultiplied(from.toColorTypeLossy<SRGBA<float>>());
101 auto premultipliedTo = premultiplied(to.toColorTypeLossy<SRGBA<float>>());
102
103 auto premultipliedBlended = makeFromComponentsClamping<SRGBA<float>>(
104 WebCore::blend(premultipliedFrom.red, premultipliedTo.red, context),
105 WebCore::blend(premultipliedFrom.green, premultipliedTo.green, context),
106 WebCore::blend(premultipliedFrom.blue, premultipliedTo.blue, context),
107 WebCore::blend(premultipliedFrom.alpha, premultipliedTo.alpha, context)
108 );
109
110 return convertColor<SRGBA<uint8_t>>(unpremultiplied(premultipliedBlended));
111}

113Color blendWithoutPremultiply(const Color& from, const Color& to, const BlendingContext& context)
114{
115 // FIXME: ExtendedColor - needs to handle color spaces.
116 // We need to preserve the state of the valid flag at the end of the animation
117 if (context.progress == 1 && !to.isValid())
118 return { };
119
120 auto fromSRGB = from.toColorTypeLossy<SRGBA<float>>();
121 auto toSRGB = to.toColorTypeLossy<SRGBA<float>>();
122
123 auto blended = makeFromComponentsClamping<SRGBA<float>>(
124 WebCore::blend(fromSRGB.red, toSRGB.red, context),
125 WebCore::blend(fromSRGB.green, toSRGB.green, context),
126 WebCore::blend(fromSRGB.blue, toSRGB.blue, context),
127 WebCore::blend(fromSRGB.alpha, toSRGB.alpha, context)
128 );
129
130 return convertColor<SRGBA<uint8_t>>(blended);
131}
