38struct HSLHueCalculationResult {
39 float hue;
40 float min;
41 float max;
42 float chroma;
43};
44
45static HSLHueCalculationResult calculateHSLHue(const SRGBA<float>& color)
46{
47 auto [r, g, b, alpha] = color;
48 auto [min, max] = std::minmax({ r, g, b });
49 float chroma = max - min;
50
51 float hue;
52 if (!chroma)
53 hue = 0;
54 else if (max == r)
55 hue = (60.0f * ((g - b) / chroma)) + 360.0f;
56 else if (max == g)
57 hue = (60.0f * ((b - r) / chroma)) + 120.0f;
58 else
59 hue = (60.0f * ((r - g) / chroma)) + 240.0f;
60
61 if (hue >= 360.0f)
62 hue -= 360.0f;
63
64 return { hue, min, max, chroma };
65}

67HSLA<float> ColorConversion<HSLA<float>, SRGBA<float>>::convert(const SRGBA<float>& color)
68{
69 // https://drafts.csswg.org/css-color-4/#hsl-to-rgb
70 auto [r, g, b, alpha] = color;
71 auto [hue, min, max, chroma] = calculateHSLHue(color);
72
73 float lightness = (0.5f * (max + min)) * 100.0f;
74 float saturation;
75 if (!chroma)
76 saturation = 0;
77 else if (lightness <= 50.0f)
78 saturation = (chroma / (max + min)) * 100.0f;
79 else
80 saturation = (chroma / (2.0f - (max + min))) * 100.0f;
81
82 return { hue, saturation, lightness, alpha };
83}

85SRGBA<float> ColorConversion<SRGBA<float>, HSLA<float>>::convert(const HSLA<float>& color)
86{
87 // https://drafts.csswg.org/css-color-4/#hsl-to-rgb
88 auto [hue, saturation, lightness, alpha] = color;
89
90 if (!saturation) {
91 auto grey = lightness / 100.0f;
92 return { grey, grey, grey, alpha };
93 }
94
95 // hueToRGB() wants hue in the 0-6 range.
96 auto normalizedHue = (hue / 360.0f) * 6.0f;
97 auto normalizedLightness = lightness / 100.0f;
98 auto normalizedSaturation = saturation / 100.0f;
99
100 auto hueForRed = normalizedHue + 2.0f;
101 auto hueForGreen = normalizedHue;
102 auto hueForBlue = normalizedHue - 2.0f;
103 if (hueForRed > 6.0f)
104 hueForRed -= 6.0f;
105 else if (hueForBlue < 0.0f)
106 hueForBlue += 6.0f;
107
108 float temp2 = normalizedLightness <= 0.5f ? normalizedLightness * (1.0f + normalizedSaturation) : normalizedLightness + normalizedSaturation - normalizedLightness * normalizedSaturation;
109 float temp1 = 2.0f * normalizedLightness - temp2;
110
111 // Hue is in the range 0-6, other args in 0-1.
112 auto hueToRGB = [](float temp1, float temp2, float hue) {
113 if (hue < 1.0f)
114 return temp1 + (temp2 - temp1) * hue;
115 if (hue < 3.0f)
116 return temp2;
117 if (hue < 4.0f)
118 return temp1 + (temp2 - temp1) * (4.0f - hue);
119 return temp1;
120 };
121
122 return {
123 hueToRGB(temp1, temp2, hueForRed),
124 hueToRGB(temp1, temp2, hueForGreen),
125 hueToRGB(temp1, temp2, hueForBlue),
126 alpha
127 };
128}

132HWBA<float> ColorConversion<HWBA<float>, SRGBA<float>>::convert(const SRGBA<float>& color)
133{
134 // https://drafts.csswg.org/css-color-4/#rgb-to-hwb
135 auto [hue, min, max, chroma] = calculateHSLHue(color);
136 auto whiteness = min * 100.0f;
137 auto blackness = (1.0f - max) * 100.0f;
138
139 return { hue, whiteness, blackness, color.alpha };
140}
141
142SRGBA<float> ColorConversion<SRGBA<float>, HWBA<float>>::convert(const HWBA<float>& color)
143{
144 // https://drafts.csswg.org/css-color-4/#hwb-to-rgb
145 auto [hue, whiteness, blackness, alpha] = color;
146
147 if (whiteness + blackness == 100.0f) {
148 auto grey = whiteness / 100.0f;
149 return { grey, grey, grey, alpha };
150 }
151
152 // hueToRGB() wants hue in the 0-6 range.
153 auto normalizedHue = (hue / 360.0f) * 6.0f;
154
155 auto hueForRed = normalizedHue + 2.0f;
156 auto hueForGreen = normalizedHue;
157 auto hueForBlue = normalizedHue - 2.0f;
158 if (hueForRed > 6.0f)
159 hueForRed -= 6.0f;
160 else if (hueForBlue < 0.0f)
161 hueForBlue += 6.0f;
162
163 auto normalizedWhiteness = whiteness / 100.0f;
164 auto normalizedBlackness = blackness / 100.0f;
165
166 // This is the hueToRGB function in convertColor<SRGBA<float>>(const HSLA&) with temp1 == 0
167 // and temp2 == 1 strength reduced through it.
168 auto hueToRGB = [](float hue) {
169 if (hue < 1.0f)
170 return hue;
171 if (hue < 3.0f)
172 return 1.0f;
173 if (hue < 4.0f)
174 return 4.0f - hue;
175 return 0.0f;
176 };
177
178 auto applyWhitenessBlackness = [](float component, auto whiteness, auto blackness) {
179 return (component * (1.0f - whiteness - blackness)) + whiteness;
180 };
181
182 return {
183 applyWhitenessBlackness(hueToRGB(hueForRed), normalizedWhiteness, normalizedBlackness),
184 applyWhitenessBlackness(hueToRGB(hueForGreen), normalizedWhiteness, normalizedBlackness),
185 applyWhitenessBlackness(hueToRGB(hueForBlue), normalizedWhiteness, normalizedBlackness),
186 alpha
187 };
188}

190// MARK: Lab conversions.
191
192static constexpr float LABe = 216.0f / 24389.0f;
193static constexpr float LABk = 24389.0f / 27.0f;
194static constexpr float D50WhiteValues[] = { 0.96422f, 1.0f, 0.82521f };
195
196XYZA<float, WhitePoint::D50> ColorConversion<XYZA<float, WhitePoint::D50>, Lab<float>>::convert(const Lab<float>& color)
197{
198 float f1 = (color.lightness + 16.0f) / 116.0f;
199 float f0 = f1 + (color.a / 500.0f);
200 float f2 = f1 - (color.b / 200.0f);
201
202 auto computeXAndZ = [](float t) {
203 float tCubed = t * t * t;
204 if (tCubed > LABe)
205 return tCubed;
206
207 return (116.0f * t - 16.0f) / LABk;
208 };
209
210 auto computeY = [](float t) {
211 if (t > (LABk * LABe)) {
212 float value = (t + 16.0) / 116.0;
213 return value * value * value;
214 }
215
216 return t / LABk;
217 };
218
219 float x = D50WhiteValues[0] * computeXAndZ(f0);
220 float y = D50WhiteValues[1] * computeY(color.lightness);
221 float z = D50WhiteValues[2] * computeXAndZ(f2);
222
223 return { x, y, z, color.alpha };
224}
225
226Lab<float> ColorConversion<Lab<float>, XYZA<float, WhitePoint::D50>>::convert(const XYZA<float, WhitePoint::D50>& color)
227{
228 float x = color.x / D50WhiteValues[0];
229 float y = color.y / D50WhiteValues[1];
230 float z = color.z / D50WhiteValues[2];
231
232 auto fTransform = [](float value) {
233 return value > LABe ? std::cbrt(value) : (LABk * value + 16.0f) / 116.0f;
234 };
235
236 float f0 = fTransform(x);
237 float f1 = fTransform(y);
238 float f2 = fTransform(z);
239
240 float lightness = (116.0f * f1) - 16.0f;
241 float a = 500.0f * (f0 - f1);
242 float b = 200.0f * (f1 - f2);
243
244 return { lightness, a, b, color.alpha };
245}

250LCHA<float> ColorConversion<LCHA<float>, Lab<float>>::convert(const Lab<float>& color)
251{
252 // https://www.w3.org/TR/css-color-4/#lab-to-lch
253 float hue = rad2deg(atan2(color.b, color.a));
254
255 return {
256 color.lightness,
257 std::hypot(color.a, color.b),
258 hue >= 0 ? hue : hue + 360,
259 color.alpha
260 };
261}
262
263Lab<float> ColorConversion<Lab<float>, LCHA<float>>::convert(const LCHA<float>& color)
264{
265 // https://www.w3.org/TR/css-color-4/#lch-to-lab
266 float hueAngleRadians = deg2rad(color.hue);
267
268 return {
269 color.lightness,
270 color.chroma * std::cos(hueAngleRadians),
271 color.chroma * std::sin(hueAngleRadians),
272 color.alpha
273 };
274}

278ColorComponents<float, 4> converColorComponents(ColorSpace inputColorSpace, ColorComponents<float, 4> inputColorComponents, ColorSpace outputColorSpace)
279{
280 return callWithColorType(inputColorComponents, inputColorSpace, [outputColorSpace] (const auto& inputColor) {
281 switch (outputColorSpace) {
282 case ColorSpace::A98RGB:
283 return asColorComponents(convertColor<A98RGB<float>>(inputColor));
284 case ColorSpace::DisplayP3:
285 return asColorComponents(convertColor<DisplayP3<float>>(inputColor));
286 case ColorSpace::LCH:
287 return asColorComponents(convertColor<LCHA<float>>(inputColor));
288 case ColorSpace::Lab:
289 return asColorComponents(convertColor<Lab<float>>(inputColor));
290 case ColorSpace::LinearSRGB:
291 return asColorComponents(convertColor<LinearSRGBA<float>>(inputColor));
292 case ColorSpace::ProPhotoRGB:
293 return asColorComponents(convertColor<ProPhotoRGB<float>>(inputColor));
294 case ColorSpace::Rec2020:
295 return asColorComponents(convertColor<Rec2020<float>>(inputColor));
296 case ColorSpace::SRGB:
297 return asColorComponents(convertColor<SRGBA<float>>(inputColor));
298 case ColorSpace::XYZ_D50:
299 return asColorComponents(convertColor<XYZA<float, WhitePoint::D50>>(inputColor));
300 }
301
302 ASSERT_NOT_REACHED();
303 return asColorComponents(convertColor<SRGBA<float>>(inputColor));
304 });
305}

307ColorComponents<float, 4> converColorComponents(ColorSpace inputColorSpace, ColorComponents<float, 4> inputColorComponents, const DestinationColorSpace& outputColorSpace)
308{
309#if USE(CG)
310 return platformConvertColorComponents(inputColorSpace, inputColorComponents, outputColorSpace);
311#else
312 return callWithColorType(inputColorComponents, inputColorSpace, [outputColorSpace] (const auto& inputColor) {
313 switch (outputColorSpace.platformColorSpace()) {
314 case PlatformColorSpace::Name::SRGB:
315 return asColorComponents(convertColor<SRGBA<float>>(inputColor));
316#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
317 case PlatformColorSpace::Name::LinearSRGB:
318 return asColorComponents(convertColor<LinearSRGBA<float>>(inputColor));
319#endif
320#if ENABLE(DESTINATION_COLOR_SPACE_DISPLAY_P3)
321 case PlatformColorSpace::Name::DisplayP3:
322 return asColorComponents(convertColor<DisplayP3<float>>(inputColor));
323#endif
324 }
325
326 ASSERT_NOT_REACHED();
327 return asColorComponents(convertColor<SRGBA<float>>(inputColor));
328 });
329#endif
