100static ASCIILiteral serialization(ColorSpace colorSpace)
101{
102 switch (colorSpace) {
103 case ColorSpace::A98RGB:
104 return "a98-rgb"_s;
105 case ColorSpace::DisplayP3:
106 return "display-p3"_s;
107 case ColorSpace::LCH:
108 return "lch"_s;
109 case ColorSpace::Lab:
110 return "lab"_s;
111 case ColorSpace::LinearSRGB:
112 return "linear-srgb"_s;
113 case ColorSpace::ProPhotoRGB:
114 return "prophoto-rgb"_s;
115 case ColorSpace::Rec2020:
116 return "rec2020"_s;
117 case ColorSpace::SRGB:
118 return "srgb"_s;
119 case ColorSpace::XYZ_D50:
120 return "xyz"_s;
121 }
122
123 ASSERT_NOT_REACHED();
124 return ""_s;
125}

127template<typename ColorType> static String serializationUsingColorFunction(const ColorType& color)
128{
129 static_assert(std::is_same_v<typename ColorType::ComponentType, float>);
130
131 auto [c1, c2, c3, alpha] = color;
132 if (WTF::areEssentiallyEqual(alpha, 1.0f))
133 return makeString("color(", serialization(ColorSpaceFor<ColorType>), ' ', c1, ' ', c2, ' ', c3, ')');
134 return makeString("color(", serialization(ColorSpaceFor<ColorType>), ' ', c1, ' ', c2, ' ', c3, " / ", alpha, ')');
135}
136
137static String serializationUsingColorFunction(const Lab<float>& color)
138{
139 auto [c1, c2, c3, alpha] = color;
140 if (WTF::areEssentiallyEqual(alpha, 1.0f))
141 return makeString("color(", serialization(ColorSpaceFor<Lab<float>>), ' ', c1, "% ", c2, ' ', c3, ')');
142 return makeString("color(", serialization(ColorSpaceFor<Lab<float>>), ' ', c1, "% ", c2, ' ', c3, " / ", alpha, ')');
143}

145static String serializationUsingColorFunction(const SRGBA<uint8_t>& color)
146{

148}

186String serializationForCSS(const LCHA<float>& color, bool useColorFunctionSerialization)
187{
188 // https://www.w3.org/TR/css-color-4/#serializing-lab-lch
189 if (useColorFunctionSerialization)

302static char decimalDigit(unsigned number)
303{
304 ASSERT(number < 10);
305 return '0' + number;
306}
307
308static std::array<char, 4> fractionDigitsForFractionalAlphaValue(uint8_t alpha)
309{
310 ASSERT(alpha > 0);
311 ASSERT(alpha < 0xFF);
312 if (((alpha * 100 + 0x7F) / 0xFF * 0xFF + 50) / 100 != alpha)
313 return { { decimalDigit(alpha * 10 / 0xFF % 10), decimalDigit(alpha * 100 / 0xFF % 10), decimalDigit((alpha * 1000 + 0x7F) / 0xFF % 10), '\0' } };
314 if (int thirdDigit = (alpha * 100 + 0x7F) / 0xFF % 10)
315 return { { decimalDigit(alpha * 10 / 0xFF), decimalDigit(thirdDigit), '\0', '\0' } };
316 return { { decimalDigit((alpha * 10 + 0x7F) / 0xFF), '\0', '\0', '\0' } };
317}
318
319String serializationForCSS(SRGBA<uint8_t> color, bool useColorFunctionSerialization)
320{
321 if (useColorFunctionSerialization)
322 return serializationUsingColorFunction(color);
323
324 auto [red, green, blue, alpha] = color;
325 switch (alpha) {
326 case 0:
327 return makeString("rgba(", red, ", ", green, ", ", blue, ", 0)");
328 case 0xFF:
329 return makeString("rgb(", red, ", ", green, ", ", blue, ')');
330 default:
331 return makeString("rgba(", red, ", ", green, ", ", blue, ", 0.", fractionDigitsForFractionalAlphaValue(alpha).data(), ')');
332 }
333}
334
335String serializationForHTML(SRGBA<uint8_t> color, bool useColorFunctionSerialization)
336{
337 if (useColorFunctionSerialization)
338 return serializationUsingColorFunction(color);
339
340 auto [red, green, blue, alpha] = color;
341 if (alpha == 0xFF)
342 return makeString('#', hex(red, 2, Lowercase), hex(green, 2, Lowercase), hex(blue, 2, Lowercase));
343 return serializationForCSS(color);
344}
345
346String serializationForRenderTreeAsText(SRGBA<uint8_t> color, bool useColorFunctionSerialization)
347{
348 if (useColorFunctionSerialization)
349 return serializationUsingColorFunction(color);
350
351 auto [red, green, blue, alpha] = color;
352 if (alpha < 0xFF)
353 return makeString('#', hex(red, 2), hex(green, 2), hex(blue, 2), hex(alpha, 2));
354 return makeString('#', hex(red, 2), hex(green, 2), hex(blue, 2));
355}
