51template<typename ColorType, typename T> constexpr ColorType makeFromComponents(const ColorComponents<T, 4>& c)
52{
53 return ColorType { c[0], c[1], c[2], c[3] };
54}
55
56template<typename ColorType, unsigned Index, typename T> constexpr auto clampedComponent(T c) -> typename ColorType::ComponentType
57{
58 static_assert(std::is_integral_v<T>);
59
60 constexpr auto componentInfo = ColorType::Model::componentInfo[Index];
61 return std::clamp<T>(c, componentInfo.min, componentInfo.max);
62}
63
64template<typename ColorType, unsigned Index> constexpr float clampedComponent(float c)
65{
66 constexpr auto componentInfo = ColorType::Model::componentInfo[Index];
67
68 if constexpr (componentInfo.min == -std::numeric_limits<float>::infinity() && componentInfo.max == std::numeric_limits<float>::infinity())
69 return c;
70
71 if constexpr (componentInfo.min == -std::numeric_limits<float>::infinity())
72 return std::min(c, componentInfo.max);
73
74 if constexpr (componentInfo.max == std::numeric_limits<float>::infinity())
75 return std::max(c, componentInfo.min);
76
77 return std::clamp(c, componentInfo.min, componentInfo.max);
78}
79
80template<typename ColorType, unsigned Index, typename T> constexpr T clampedComponent(const ColorComponents<T, 4>& c)
81{
82 return clampedComponent<ColorType, Index>(c[Index]);
83}
84
85template<typename T, typename ComponentType = T> constexpr ComponentType clampedAlpha(T alpha)
86{
87 return std::clamp<T>(alpha, AlphaTraits<ComponentType>::transparent, AlphaTraits<ComponentType>::opaque);
88}
89
90template<typename ColorType, typename T> constexpr ColorComponents<T, 4> clampedComponents(const ColorComponents<T, 4>& components)
91{
92 return { clampedComponent<ColorType, 0>(components), clampedComponent<ColorType, 1>(components), clampedComponent<ColorType, 2>(components), clampedAlpha(components[3]) };
93}
94
95template<typename ColorType, typename T> constexpr ColorComponents<T, 4> clampedComponentsExceptAlpha(const ColorComponents<T, 4>& components)
96{
97 return { clampedComponent<ColorType, 0>(components), clampedComponent<ColorType, 1>(components), clampedComponent<ColorType, 2>(components), components[3] };
98}

100template<typename ColorType, typename T> constexpr ColorType makeFromComponentsClamping(const ColorComponents<T, 4>& components)
101{
102 return makeFromComponents<ColorType>(clampedComponents<ColorType>(components));
103}

105template<typename ColorType, typename T> constexpr ColorType makeFromComponentsClamping(T c1, T c2, T c3)
106{
107 return makeFromComponents<ColorType>(ColorComponents { clampedComponent<ColorType, 0>(c1), clampedComponent<ColorType, 1>(c2), clampedComponent<ColorType, 2>(c3), AlphaTraits<typename ColorType::ComponentType>::opaque });
108}
109
110template<typename ColorType, typename T> constexpr ColorType makeFromComponentsClamping(T c1, T c2, T c3, T alpha)
111{
112 return makeFromComponents<ColorType>(ColorComponents { clampedComponent<ColorType, 0>(c1), clampedComponent<ColorType, 1>(c2), clampedComponent<ColorType, 2>(c3), clampedAlpha<T, typename ColorType::ComponentType>(alpha) });
113}
114
115template<typename ColorType, typename T> constexpr ColorType makeFromComponentsClampingExceptAlpha(const ColorComponents<T, 4>& components)
116{
117 return makeFromComponents<ColorType>(clampedComponentsExceptAlpha<ColorType>(components));
118}

120template<typename ColorType, typename T, typename Alpha> constexpr ColorType makeFromComponentsClampingExceptAlpha(T c1, T c2, T c3, Alpha alpha)
121{
122 return makeFromComponents<ColorType>(ColorComponents { clampedComponent<ColorType, 0>(c1), clampedComponent<ColorType, 1>(c2), clampedComponent<ColorType, 2>(c3), alpha });
123}

127template<typename T> constexpr void assertInRange(T color)
128{
129 if constexpr (std::is_same_v<typename T::ComponentType, float>) {
130 auto components = asColorComponents(color);
131 for (unsigned i = 0; i < 3; ++i) {
132 ASSERT_WITH_MESSAGE(components[i] >= T::Model::componentInfo[i].min, "Component at index %d is %f and is less than the allowed minimum %f", i, components[i], T::Model::componentInfo[i].min);
133 ASSERT_WITH_MESSAGE(components[i] <= T::Model::componentInfo[i].max, "Component at index %d is %f and is greater than the allowed maximum %f", i, components[i], T::Model::componentInfo[i].max);
134 }
135 ASSERT_WITH_MESSAGE(color.alpha >= AlphaTraits<typename T::ComponentType>::transparent, "Alpha is %f and is less than the allowed minimum (transparent) %f", color.alpha, AlphaTraits<typename T::ComponentType>::transparent);
136 ASSERT_WITH_MESSAGE(color.alpha <= AlphaTraits<typename T::ComponentType>::opaque, "Alpha is %f and is greater than the allowed maximum (opaque) %f", color.alpha, AlphaTraits<typename T::ComponentType>::opaque);
137 }
138}

164template<typename Parent> struct ColorWithAlphaHelper {
165 // Helper to allow convenient syntax for working with color types.
166 // e.g. auto yellowWith50PercentAlpha = Color::yellow.colorWithAlphaByte(128);
167 constexpr Parent colorWithAlphaByte(uint8_t overrideAlpha) const
168 {
169 static_assert(std::is_same_v<decltype(std::declval<Parent>().alpha), uint8_t>, "Only uint8_t based color types are supported.");
170
171 auto copy = *static_cast<const Parent*>(this);
172 copy.alpha = overrideAlpha;
173 return copy;
174 }
175};

178template<typename ColorType, typename std::enable_if_t<IsConvertibleToColorComponents<ColorType>>* = nullptr>
179constexpr bool operator==(const ColorType& a, const ColorType& b)
180{
181 return asColorComponents(a) == asColorComponents(b);
182}
183
184template<typename ColorType, typename std::enable_if_t<IsConvertibleToColorComponents<ColorType>>* = nullptr>
185constexpr bool operator!=(const ColorType& a, const ColorType& b)
186{
187 return !(a == b);
188}

194 using ComponentType = T;
195 using Model = M;
196 using TransferFunction = TF;
197 using Descriptor = D;

199
200 constexpr RGBAType(T red, T green, T blue, T alpha = AlphaTraits<T>::opaque)
201 : red { red }

203 , blue { blue }
204 , alpha { alpha }
205 {
206 assertInRange(*this);
207 }
208
209 constexpr RGBAType()
210 : RGBAType { 0, 0, 0, 0 }
211 {
212 }
213
214 T red;
215 T green;
216 T blue;
217 T alpha;
218};
