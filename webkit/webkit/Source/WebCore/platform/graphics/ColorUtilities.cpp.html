29namespace WebCore {
30
31SRGBA<float> premultiplied(const SRGBA<float>& color)
32{
33 auto [r, g, b, a] = color;
34 return { r * a, g * a, b * a, a };
35}
36
37SRGBA<float> unpremultiplied(const SRGBA<float>& color)
38{
39 auto [r, g, b, a] = color;
40 if (!a)
41 return color;
42 return makeFromComponentsClampingExceptAlpha<SRGBA<float>>(r / a, g / a, b / a, a);
43}
44
45SRGBA<uint8_t> premultipliedFlooring(SRGBA<uint8_t> color)
46{
47 auto [r, g, b, a] = color;
48 if (!a)
49 return { 0, 0, 0, 0 };
50 if (a == 255)
51 return color;
52 return makeFromComponentsClampingExceptAlpha<SRGBA<uint8_t>>(fastDivideBy255(r * a), fastDivideBy255(g * a), fastDivideBy255(b * a), a);
53}
54
55SRGBA<uint8_t> premultipliedCeiling(SRGBA<uint8_t> color)
56{
57 auto [r, g, b, a] = color;
58 if (!a)
59 return { 0, 0, 0, 0 };
60 if (a == 255)
61 return color;
62 return makeFromComponentsClampingExceptAlpha<SRGBA<uint8_t>>(fastDivideBy255(r * a + 254), fastDivideBy255(g * a + 254), fastDivideBy255(b * a + 254), a);
63}
64
65static inline uint16_t unpremultipliedComponentByte(uint8_t c, uint8_t a)
66{
67 return (fastMultiplyBy255(c) + a - 1) / a;
68}
69
70SRGBA<uint8_t> unpremultiplied(SRGBA<uint8_t> color)
71{
72 auto [r, g, b, a] = color;
73 if (!a || a == 255)
74 return color;
75 return makeFromComponentsClampingExceptAlpha<SRGBA<uint8_t>>(unpremultipliedComponentByte(r, a), unpremultipliedComponentByte(g, a), unpremultipliedComponentByte(b, a), a);
76}
