45class TextLayout {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 static bool isNeeded(RenderText& text, const FontCascade& font)
49 {
50 TextRun run = RenderBlock::constructTextRun(text, text.style());
51 return font.codePath(run) == FontCascade::CodePath::Complex;
52 }
53
54 TextLayout(RenderText& text, const FontCascade& font, float xPos)
55 : m_font(font)
56 , m_run(constructTextRun(text, xPos))
57 , m_controller(makeUnique<ComplexTextController>(m_font, m_run, true))
58 {
59 }
60
61 float width(unsigned from, unsigned len, HashSet<const Font*>* fallbackFonts)
62 {
63 m_controller->advance(from, 0, ByWholeGlyphs, fallbackFonts);
64 float beforeWidth = m_controller->runWidthSoFar();
65 if (m_font.wordSpacing() && from && FontCascade::treatAsSpace(m_run[from]))
66 beforeWidth += m_font.wordSpacing();
67 m_controller->advance(from + len, 0, ByWholeGlyphs, fallbackFonts);
68 float afterWidth = m_controller->runWidthSoFar();
69 return afterWidth - beforeWidth;
70 }
71
72private:
73 static TextRun constructTextRun(RenderText& text, float xPos)
74 {
75 TextRun run = RenderBlock::constructTextRun(text, text.style());
76 run.setXPos(xPos);
77 return run;
78 }
79
80 // ComplexTextController has only references to its FontCascade and TextRun so they must be kept alive here.
81 FontCascade m_font;
82 TextRun m_run;
83 std::unique_ptr<ComplexTextController> m_controller;
84};

103void ComplexTextController::computeExpansionOpportunity()
104{
105 if (!m_expansion)
106 m_expansionPerOpportunity = 0;
107 else {
108 unsigned expansionOpportunityCount = FontCascade::expansionOpportunityCount(m_run.text(), m_run.ltr() ? TextDirection::LTR : TextDirection::RTL, m_run.expansionBehavior()).first;
109
110 if (!expansionOpportunityCount)
111 m_expansionPerOpportunity = 0;
112 else
113 m_expansionPerOpportunity = m_expansion / expansionOpportunityCount;
114 }
115}

147void ComplexTextController::finishConstruction()
148{
149 adjustGlyphsAndAdvances();
150
151 if (!m_isLTROnly) {
152 unsigned length = m_complexTextRuns.size();
153 m_runIndices.reserveInitialCapacity(length);
154 for (unsigned i = 0; i < length; ++i)
155 m_runIndices.uncheckedAppend(length - i - 1);
156 std::sort(m_runIndices.data(), m_runIndices.data() + length,
157 [this](auto a, auto b) {
158 return stringBegin(*m_complexTextRuns[a]) < stringBegin(*m_complexTextRuns[b]);
159 });
160
161 m_glyphCountFromStartToIndex.reserveInitialCapacity(length);
162 unsigned glyphCountSoFar = 0;
163 for (unsigned i = 0; i < length; ++i) {
164 m_glyphCountFromStartToIndex.uncheckedAppend(glyphCountSoFar);
165 glyphCountSoFar += m_complexTextRuns[i]->glyphCount();
166 }
167 }

170unsigned ComplexTextController::offsetForPosition(float h, bool includePartialGlyphs)
171{
172 if (h >= m_totalAdvance.width())
173 return m_run.ltr() ? m_end : 0;
174
175 if (h < 0)
176 return m_run.ltr() ? 0 : m_end;
177
178 float x = h;
179
180 size_t runCount = m_complexTextRuns.size();
181 unsigned offsetIntoAdjustedGlyphs = 0;
182
183 for (size_t r = 0; r < runCount; ++r) {
184 const ComplexTextRun& complexTextRun = *m_complexTextRuns[r];
185 for (unsigned j = 0; j < complexTextRun.glyphCount(); ++j) {
186 unsigned index = offsetIntoAdjustedGlyphs + j;
187 float adjustedAdvance = m_adjustedBaseAdvances[index].width();
188 bool hit = m_run.ltr() ? x < adjustedAdvance : (x <= adjustedAdvance && adjustedAdvance);
189 if (hit) {
190 unsigned hitGlyphStart = complexTextRun.indexAt(j);
191 unsigned hitGlyphEnd;
192 if (m_run.ltr())
193 hitGlyphEnd = std::max(hitGlyphStart, j + 1 < complexTextRun.glyphCount() ? complexTextRun.indexAt(j + 1) : complexTextRun.indexEnd());
194 else
195 hitGlyphEnd = std::max(hitGlyphStart, j > 0 ? complexTextRun.indexAt(j - 1) : complexTextRun.indexEnd());
196
197 // FIXME: Instead of dividing the glyph's advance equally between the characters, this
198 // could use the glyph's "ligature carets". This is available in CoreText via CTFontGetLigatureCaretPositions().
199 unsigned hitIndex;
200 if (m_run.ltr())
201 hitIndex = hitGlyphStart + (hitGlyphEnd - hitGlyphStart) * (x / adjustedAdvance);
202 else {
203 if (hitGlyphStart == hitGlyphEnd)
204 hitIndex = hitGlyphStart;
205 else if (x)
206 hitIndex = hitGlyphEnd - (hitGlyphEnd - hitGlyphStart) * (x / adjustedAdvance);
207 else
208 hitIndex = hitGlyphEnd - 1;
209 }
210
211 unsigned stringLength = complexTextRun.stringLength();
212 CachedTextBreakIterator cursorPositionIterator(StringView(complexTextRun.characters(), stringLength), TextBreakIterator::Mode::Caret, nullAtom());
213 unsigned clusterStart;
214 if (cursorPositionIterator.isBoundary(hitIndex))
215 clusterStart = hitIndex;
216 else
217 clusterStart = cursorPositionIterator.preceding(hitIndex).value_or(0);
218
219 if (!includePartialGlyphs)
220 return complexTextRun.stringLocation() + clusterStart;
221
222 unsigned clusterEnd = cursorPositionIterator.following(hitIndex).value_or(stringLength);
223
224 float clusterWidth;
225 // FIXME: The search stops at the boundaries of complexTextRun. In theory, it should go on into neighboring ComplexTextRuns
226 // derived from the same CTLine. In practice, we do not expect there to be more than one CTRun in a CTLine, as no
227 // reordering and no font fallback should occur within a CTLine.
228 if (clusterEnd - clusterStart > 1) {
229 clusterWidth = adjustedAdvance;
230 if (j) {
231 unsigned firstGlyphBeforeCluster = j - 1;
232 while (complexTextRun.indexAt(firstGlyphBeforeCluster) >= clusterStart && complexTextRun.indexAt(firstGlyphBeforeCluster) < clusterEnd) {
233 float width = m_adjustedBaseAdvances[offsetIntoAdjustedGlyphs + firstGlyphBeforeCluster].width();
234 clusterWidth += width;
235 x += width;
236 if (!firstGlyphBeforeCluster)
237 break;
238 firstGlyphBeforeCluster--;
239 }
240 }
241 unsigned firstGlyphAfterCluster = j + 1;
242 while (firstGlyphAfterCluster < complexTextRun.glyphCount() && complexTextRun.indexAt(firstGlyphAfterCluster) >= clusterStart && complexTextRun.indexAt(firstGlyphAfterCluster) < clusterEnd) {
243 clusterWidth += m_adjustedBaseAdvances[offsetIntoAdjustedGlyphs + firstGlyphAfterCluster].width();
244 firstGlyphAfterCluster++;
245 }
246 } else {
247 clusterWidth = adjustedAdvance / (hitGlyphEnd - hitGlyphStart);
248 x -= clusterWidth * (m_run.ltr() ? hitIndex - hitGlyphStart : hitGlyphEnd - hitIndex - 1);
249 }
250 if (x <= clusterWidth / 2)
251 return complexTextRun.stringLocation() + (m_run.ltr() ? clusterStart : clusterEnd);
252 return complexTextRun.stringLocation() + (m_run.ltr() ? clusterEnd : clusterStart);
253 }
254 x -= adjustedAdvance;
255 }
256 offsetIntoAdjustedGlyphs += complexTextRun.glyphCount();
257 }
258
259 ASSERT_NOT_REACHED();
260 return 0;
261}

263// FIXME: We should consider reimplementing this function using ICU to advance by grapheme.
264// The current implementation only considers explicitly emoji sequences and emoji variations.
265static bool advanceByCombiningCharacterSequence(const UChar*& iterator, const UChar* end, UChar32& baseCharacter, unsigned& markCount)
266{
267 ASSERT(iterator < end);
268
269 markCount = 0;
270
271 unsigned i = 0;
272 unsigned remainingCharacters = end - iterator;
273 U16_NEXT(iterator, i, remainingCharacters, baseCharacter);
274 iterator = iterator + i;
275 if (U_IS_SURROGATE(baseCharacter))
276 return false;
277
278 // Consume marks.
279 bool sawEmojiGroupCandidate = isEmojiGroupCandidate(baseCharacter);
280 bool sawJoiner = false;
281 bool sawRegionalIndicator = isEmojiRegionalIndicator(baseCharacter);
282 while (iterator < end) {
283 UChar32 nextCharacter;
284 unsigned markLength = 0;
285 bool shouldContinue = false;
286 ASSERT(end >= iterator);
287 U16_NEXT(iterator, markLength, static_cast<unsigned>(end - iterator), nextCharacter);
288
289 if (isVariationSelector(nextCharacter) || isEmojiFitzpatrickModifier(nextCharacter))
290 shouldContinue = true;
291
292 if (sawRegionalIndicator && isEmojiRegionalIndicator(nextCharacter)) {
293 shouldContinue = true;
294 sawRegionalIndicator = false;
295 }
296
297 if (sawJoiner && isEmojiGroupCandidate(nextCharacter))
298 shouldContinue = true;
299
300 sawJoiner = false;
301 if (sawEmojiGroupCandidate && nextCharacter == zeroWidthJoiner) {
302 sawJoiner = true;
303 shouldContinue = true;
304 }
305
306 if (!shouldContinue && !(U_GET_GC_MASK(nextCharacter) & U_GC_M_MASK))
307 break;
308
309 markCount += markLength;
310 iterator += markLength;
311 }
312
313 return true;
314}

316// FIXME: Capitalization is language-dependent and context-dependent and should operate on grapheme clusters instead of codepoints.
317static inline std::optional<UChar32> capitalized(UChar32 baseCharacter)
318{
319 if (U_GET_GC_MASK(baseCharacter) & U_GC_M_MASK)
320 return std::nullopt;
321
322 UChar32 uppercaseCharacter = u_toupper(baseCharacter);
323 ASSERT(uppercaseCharacter == baseCharacter || (U_IS_BMP(baseCharacter) == U_IS_BMP(uppercaseCharacter)));
324 if (uppercaseCharacter != baseCharacter)
325 return uppercaseCharacter;
326 return std::nullopt;
327}

329static bool shouldSynthesize(bool dontSynthesizeSmallCaps, const Font* nextFont, UChar32 baseCharacter, std::optional<UChar32> capitalizedBase, FontVariantCaps fontVariantCaps, bool engageAllSmallCapsProcessing)
330{
331 if (dontSynthesizeSmallCaps)
332 return false;
333 if (!nextFont || nextFont == Font::systemFallback())
334 return false;
335 if (engageAllSmallCapsProcessing && isASCIISpace(baseCharacter))
336 return false;
337 if (!engageAllSmallCapsProcessing && !capitalizedBase)
338 return false;
339 return !nextFont->variantCapsSupportsCharacterForSynthesis(fontVariantCaps, baseCharacter);
340}

342void ComplexTextController::collectComplexTextRuns()
343{
344 if (!m_end || !m_font.size())
345 return;
346
347 // We break up glyph run generation for the string by Font.
348 const UChar* cp;
349
350 if (m_run.is8Bit()) {
351 String stringFor8BitRun = String::make16BitFrom8BitSource(m_run.characters8(), m_run.length());
352 m_stringsFor8BitRuns.append(WTFMove(stringFor8BitRun));
353 cp = m_stringsFor8BitRuns.last().characters16();
354 } else
355 cp = m_run.characters16();
356
357 auto fontVariantCaps = m_font.fontDescription().variantCaps();
358 bool dontSynthesizeSmallCaps = !static_cast<bool>(m_font.fontDescription().fontSynthesis() & FontSynthesisSmallCaps);
359 bool engageAllSmallCapsProcessing = fontVariantCaps == FontVariantCaps::AllSmall || fontVariantCaps == FontVariantCaps::AllPetite;
360 bool engageSmallCapsProcessing = engageAllSmallCapsProcessing || fontVariantCaps == FontVariantCaps::Small || fontVariantCaps == FontVariantCaps::Petite;
361
362 if (engageAllSmallCapsProcessing || engageSmallCapsProcessing)
363 m_smallCapsBuffer.resize(m_end);
364
365 unsigned indexOfFontTransition = 0;
366 const UChar* curr = cp;
367 const UChar* end = cp + m_end;
368
369 const Font* font;
370 const Font* nextFont;
371 const Font* synthesizedFont = nullptr;
372 const Font* smallSynthesizedFont = nullptr;
373
374 unsigned markCount;
375 UChar32 baseCharacter;
376 if (!advanceByCombiningCharacterSequence(curr, end, baseCharacter, markCount))
377 return;
378
379 nextFont = m_font.fontForCombiningCharacterSequence(cp, curr - cp);
380
381 bool isSmallCaps = false;
382 bool nextIsSmallCaps = false;
383
384 auto capitalizedBase = capitalized(baseCharacter);
385 if (shouldSynthesize(dontSynthesizeSmallCaps, nextFont, baseCharacter, capitalizedBase, fontVariantCaps, engageAllSmallCapsProcessing)) {
386 synthesizedFont = &nextFont->noSynthesizableFeaturesFont();
387 smallSynthesizedFont = synthesizedFont->smallCapsFont(m_font.fontDescription());
388 UChar32 characterToWrite = capitalizedBase ? capitalizedBase.value() : cp[0];
389 unsigned characterIndex = 0;
390 U16_APPEND_UNSAFE(m_smallCapsBuffer, characterIndex, characterToWrite);
391 for (unsigned i = characterIndex; cp + i < curr; ++i)
392 m_smallCapsBuffer[i] = cp[i];
393 nextIsSmallCaps = true;
394 }
395
396 while (curr < end) {
397 font = nextFont;
398 isSmallCaps = nextIsSmallCaps;
399 unsigned index = curr - cp;
400
401 if (!advanceByCombiningCharacterSequence(curr, end, baseCharacter, markCount))
402 return;
403
404 if (synthesizedFont) {
405 if (auto capitalizedBase = capitalized(baseCharacter)) {
406 unsigned characterIndex = index;
407 U16_APPEND_UNSAFE(m_smallCapsBuffer, characterIndex, capitalizedBase.value());
408 for (unsigned i = 0; i < markCount; ++i)
409 m_smallCapsBuffer[i + characterIndex] = cp[i + characterIndex];
410 nextIsSmallCaps = true;
411 } else {
412 if (engageAllSmallCapsProcessing) {
413 for (unsigned i = 0; i < curr - cp - index; ++i)
414 m_smallCapsBuffer[index + i] = cp[index + i];
415 }
416 nextIsSmallCaps = engageAllSmallCapsProcessing;
417 }
418 }
419
420 if (baseCharacter == zeroWidthJoiner)
421 nextFont = font;
422 else
423 nextFont = m_font.fontForCombiningCharacterSequence(cp + index, curr - cp - index);
424
425 capitalizedBase = capitalized(baseCharacter);
426 if (!synthesizedFont && shouldSynthesize(dontSynthesizeSmallCaps, nextFont, baseCharacter, capitalizedBase, fontVariantCaps, engageAllSmallCapsProcessing)) {
427 // Rather than synthesize each character individually, we should synthesize the entire "run" if any character requires synthesis.
428 synthesizedFont = &nextFont->noSynthesizableFeaturesFont();
429 smallSynthesizedFont = synthesizedFont->smallCapsFont(m_font.fontDescription());
430 nextIsSmallCaps = true;
431 curr = cp + indexOfFontTransition;
432 continue;
433 }
434
435 if (nextFont != font || nextIsSmallCaps != isSmallCaps) {
436 unsigned itemLength = index - indexOfFontTransition;
437 if (itemLength) {
438 unsigned itemStart = indexOfFontTransition;
439 if (synthesizedFont) {
440 if (isSmallCaps)
441 collectComplexTextRunsForCharacters(m_smallCapsBuffer.data() + itemStart, itemLength, itemStart, smallSynthesizedFont);
442 else
443 collectComplexTextRunsForCharacters(cp + itemStart, itemLength, itemStart, synthesizedFont);
444 } else
445 collectComplexTextRunsForCharacters(cp + itemStart, itemLength, itemStart, font);
446 if (nextFont != font) {
447 synthesizedFont = nullptr;
448 smallSynthesizedFont = nullptr;
449 nextIsSmallCaps = false;
450 }
451 }
452 indexOfFontTransition = index;
453 }
454 }
455
456 ASSERT(m_end >= indexOfFontTransition);
457 unsigned itemLength = m_end - indexOfFontTransition;
458 if (itemLength) {
459 unsigned itemStart = indexOfFontTransition;
460 if (synthesizedFont) {
461 if (nextIsSmallCaps)
462 collectComplexTextRunsForCharacters(m_smallCapsBuffer.data() + itemStart, itemLength, itemStart, smallSynthesizedFont);
463 else
464 collectComplexTextRunsForCharacters(cp + itemStart, itemLength, itemStart, synthesizedFont);
465 } else
466 collectComplexTextRunsForCharacters(cp + itemStart, itemLength, itemStart, nextFont);
467 }
468
469 if (!m_run.ltr())
470 m_complexTextRuns.reverse();
471}

480void ComplexTextController::ComplexTextRun::setIsNonMonotonic()
481{
482 ASSERT(m_isMonotonic);
483 m_isMonotonic = false;
484
485 Vector<bool, 64> mappedIndices(m_stringLength, false);
486 for (unsigned i = 0; i < m_glyphCount; ++i) {
487 ASSERT(indexAt(i) < m_stringLength);
488 mappedIndices[indexAt(i)] = true;
489 }
490
491 m_glyphEndOffsets.grow(m_glyphCount);
492 for (unsigned i = 0; i < m_glyphCount; ++i) {
493 unsigned nextMappedIndex = m_indexEnd;
494 for (unsigned j = indexAt(i) + 1; j < m_stringLength; ++j) {
495 if (mappedIndices[j]) {
496 nextMappedIndex = j;
497 break;
498 }
499 }
500 m_glyphEndOffsets[i] = nextMappedIndex;
501 }
502}

504unsigned ComplexTextController::indexOfCurrentRun(unsigned& leftmostGlyph)
505{
506 leftmostGlyph = 0;
507
508 size_t runCount = m_complexTextRuns.size();
509 if (m_currentRun >= runCount)
510 return runCount;
511
512 if (m_isLTROnly) {
513 for (unsigned i = 0; i < m_currentRun; ++i)
514 leftmostGlyph += m_complexTextRuns[i]->glyphCount();
515 return m_currentRun;
516 }
517
518 unsigned currentRunIndex = m_runIndices[m_currentRun];
519 leftmostGlyph = m_glyphCountFromStartToIndex[currentRunIndex];
520 return currentRunIndex;
521}
522
523unsigned ComplexTextController::incrementCurrentRun(unsigned& leftmostGlyph)
524{
525 if (m_isLTROnly) {
526 leftmostGlyph += m_complexTextRuns[m_currentRun++]->glyphCount();
527 return m_currentRun;
528 }
529
530 m_currentRun++;
531 leftmostGlyph = 0;
532 return indexOfCurrentRun(leftmostGlyph);
533}

535float ComplexTextController::runWidthSoFarFraction(unsigned glyphStartOffset, unsigned glyphEndOffset, unsigned oldCharacterInCurrentGlyph, GlyphIterationStyle iterationStyle) const
536{
537 // FIXME: Instead of dividing the glyph's advance equally between the characters, this
538 // could use the glyph's "ligature carets". This is available in CoreText via CTFontGetLigatureCaretPositions().
539 if (glyphStartOffset == glyphEndOffset) {
540 // When there are multiple glyphs per character we need to advance by the full width of the glyph.
541 ASSERT(m_characterInCurrentGlyph == oldCharacterInCurrentGlyph);
542 return 1;
543 }
544
545 if (iterationStyle == ByWholeGlyphs) {
546 if (!oldCharacterInCurrentGlyph)
547 return 1;
548 return 0;
549 }
550
551 return static_cast<float>(m_characterInCurrentGlyph - oldCharacterInCurrentGlyph) / (glyphEndOffset - glyphStartOffset);
552}

554void ComplexTextController::advance(unsigned offset, GlyphBuffer* glyphBuffer, GlyphIterationStyle iterationStyle, HashSet<const Font*>* fallbackFonts)
555{
556 if (offset > m_end)
557 offset = m_end;
558
559 if (offset < m_currentCharacter) {
560 m_runWidthSoFar = 0;
561 m_numGlyphsSoFar = 0;
562 m_currentRun = 0;
563 m_glyphInCurrentRun = 0;
564 m_characterInCurrentGlyph = 0;
565 }
566
567 m_currentCharacter = offset;
568
569 size_t runCount = m_complexTextRuns.size();
570
571 unsigned indexOfLeftmostGlyphInCurrentRun = 0; // Relative to the beginning of ComplexTextController.
572 unsigned currentRunIndex = indexOfCurrentRun(indexOfLeftmostGlyphInCurrentRun);
573 while (m_currentRun < runCount) {
574 const ComplexTextRun& complexTextRun = *m_complexTextRuns[currentRunIndex];
575 bool ltr = complexTextRun.isLTR();
576 unsigned glyphCount = complexTextRun.glyphCount();
577 unsigned glyphIndexIntoCurrentRun = ltr ? m_glyphInCurrentRun : glyphCount - 1 - m_glyphInCurrentRun;
578 unsigned glyphIndexIntoComplexTextController = indexOfLeftmostGlyphInCurrentRun + glyphIndexIntoCurrentRun;
579 if (fallbackFonts && &complexTextRun.font() != &m_font.primaryFont())
580 fallbackFonts->add(&complexTextRun.font());
581
582 // We must store the initial advance for the first glyph we are going to draw.
583 // When leftmostGlyph is 0, it represents the first glyph to draw, taking into
584 // account the text direction.
585 if (!indexOfLeftmostGlyphInCurrentRun && glyphBuffer)
586 glyphBuffer->setInitialAdvance(makeGlyphBufferAdvance(complexTextRun.initialAdvance()));
587
588 while (m_glyphInCurrentRun < glyphCount) {
589 unsigned glyphStartOffset = complexTextRun.indexAt(glyphIndexIntoCurrentRun);
590 unsigned glyphEndOffset;
591 if (complexTextRun.isMonotonic()) {
592 if (ltr)
593 glyphEndOffset = std::max(glyphStartOffset, glyphIndexIntoCurrentRun + 1 < glyphCount ? complexTextRun.indexAt(glyphIndexIntoCurrentRun + 1) : complexTextRun.indexEnd());
594 else
595 glyphEndOffset = std::max(glyphStartOffset, glyphIndexIntoCurrentRun > 0 ? complexTextRun.indexAt(glyphIndexIntoCurrentRun - 1) : complexTextRun.indexEnd());
596 } else
597 glyphEndOffset = complexTextRun.endOffsetAt(glyphIndexIntoCurrentRun);
598
599 FloatSize adjustedBaseAdvance = m_adjustedBaseAdvances[glyphIndexIntoComplexTextController];
600
601 if (glyphStartOffset + complexTextRun.stringLocation() >= m_currentCharacter)
602 return;
603
604 if (glyphBuffer && !m_characterInCurrentGlyph) {
605 auto currentGlyphOrigin = glyphOrigin(glyphIndexIntoComplexTextController);
606 GlyphBufferAdvance paintAdvance = makeGlyphBufferAdvance(adjustedBaseAdvance);
607 if (!glyphIndexIntoCurrentRun) {
608 // The first layout advance of every run includes the "initial layout advance." However, here, we need
609 // paint advances, so subtract it out before transforming the layout advance into a paint advance.
610 setWidth(paintAdvance, width(paintAdvance) - (complexTextRun.initialAdvance().width() - currentGlyphOrigin.x()));
611 setHeight(paintAdvance, height(paintAdvance) - (complexTextRun.initialAdvance().height() - currentGlyphOrigin.y()));
612 }
613 setWidth(paintAdvance, width(paintAdvance) + glyphOrigin(glyphIndexIntoComplexTextController + 1).x() - currentGlyphOrigin.x());
614 setHeight(paintAdvance, height(paintAdvance) + glyphOrigin(glyphIndexIntoComplexTextController + 1).y() - currentGlyphOrigin.y());
615 if (glyphIndexIntoCurrentRun == glyphCount - 1 && currentRunIndex + 1 < runCount) {
616 // Our paint advance points to the end of the run. However, the next run may have an
617 // initial advance, and our paint advance needs to point to the location of the next
618 // glyph. So, we need to add in the next run's initial advance.
619 setWidth(paintAdvance, width(paintAdvance) - glyphOrigin(glyphIndexIntoComplexTextController + 1).x() + m_complexTextRuns[currentRunIndex + 1]->initialAdvance().width());
620 setHeight(paintAdvance, height(paintAdvance) - glyphOrigin(glyphIndexIntoComplexTextController + 1).y() + m_complexTextRuns[currentRunIndex + 1]->initialAdvance().height());
621 }
622 setHeight(paintAdvance, -height(paintAdvance)); // Increasing y points down
623 glyphBuffer->add(m_adjustedGlyphs[glyphIndexIntoComplexTextController], complexTextRun.font(), paintAdvance, complexTextRun.indexAt(m_glyphInCurrentRun));
624 }
625
626 unsigned oldCharacterInCurrentGlyph = m_characterInCurrentGlyph;
627 m_characterInCurrentGlyph = std::min(m_currentCharacter - complexTextRun.stringLocation(), glyphEndOffset) - glyphStartOffset;
628 m_runWidthSoFar += adjustedBaseAdvance.width() * runWidthSoFarFraction(glyphStartOffset, glyphEndOffset, oldCharacterInCurrentGlyph, iterationStyle);
629
630 if (glyphEndOffset + complexTextRun.stringLocation() > m_currentCharacter)
631 return;
632
633 m_numGlyphsSoFar++;
634 m_glyphInCurrentRun++;
635 m_characterInCurrentGlyph = 0;
636 if (ltr) {
637 glyphIndexIntoCurrentRun++;
638 glyphIndexIntoComplexTextController++;
639 } else {
640 glyphIndexIntoCurrentRun--;
641 glyphIndexIntoComplexTextController--;
642 }
643 }
644 currentRunIndex = incrementCurrentRun(indexOfLeftmostGlyphInCurrentRun);
645 m_glyphInCurrentRun = 0;
646 }
647}

649static inline std::pair<bool, bool> expansionLocation(bool ideograph, bool treatAsSpace, bool ltr, bool isAfterExpansion, bool forbidLeftExpansion, bool forbidRightExpansion, bool forceLeftExpansion, bool forceRightExpansion)
650{
651 bool expandLeft = ideograph;
652 bool expandRight = ideograph;
653 if (treatAsSpace) {
654 if (ltr)
655 expandRight = true;
656 else
657 expandLeft = true;
658 }
659 if (isAfterExpansion)
660 expandLeft = false;
661 ASSERT(!forbidLeftExpansion || !forceLeftExpansion);
662 ASSERT(!forbidRightExpansion || !forceRightExpansion);
663 if (forbidLeftExpansion)
664 expandLeft = false;
665 if (forbidRightExpansion)
666 expandRight = false;
667 if (forceLeftExpansion)
668 expandLeft = true;
669 if (forceRightExpansion)
670 expandRight = true;
671 return std::make_pair(expandLeft, expandRight);
672}

674void ComplexTextController::adjustGlyphsAndAdvances()
675{
676 bool afterExpansion = (m_run.expansionBehavior() & LeftExpansionMask) == ForbidLeftExpansion;
677 size_t runCount = m_complexTextRuns.size();
678 bool hasExtraSpacing = (m_font.letterSpacing() || m_font.wordSpacing() || m_expansion) && !m_run.spacingDisabled();
679 bool runForcesLeftExpansion = (m_run.expansionBehavior() & LeftExpansionMask) == ForceLeftExpansion;
680 bool runForcesRightExpansion = (m_run.expansionBehavior() & RightExpansionMask) == ForceRightExpansion;
681 bool runForbidsLeftExpansion = (m_run.expansionBehavior() & LeftExpansionMask) == ForbidLeftExpansion;
682 bool runForbidsRightExpansion = (m_run.expansionBehavior() & RightExpansionMask) == ForbidRightExpansion;
683
684 // We are iterating in glyph order, not string order. Compare this to WidthIterator::advanceInternal()
685 for (size_t runIndex = 0; runIndex < runCount; ++runIndex) {
686 ComplexTextRun& complexTextRun = *m_complexTextRuns[runIndex];
687 unsigned glyphCount = complexTextRun.glyphCount();
688 const Font& font = complexTextRun.font();
689
690 if (!complexTextRun.isLTR())
691 m_isLTROnly = false;
692
693 const CGGlyph* glyphs = complexTextRun.glyphs();
694 const FloatSize* advances = complexTextRun.baseAdvances();
695
696 float spaceWidth = font.spaceWidth() - font.syntheticBoldOffset();
697 const UChar* cp = complexTextRun.characters();
698 FloatPoint glyphOrigin;
699 unsigned lastCharacterIndex = m_run.ltr() ? std::numeric_limits<unsigned>::min() : std::numeric_limits<unsigned>::max();
700 bool isMonotonic = true;
701
702 for (unsigned i = 0; i < glyphCount; i++) {
703 unsigned characterIndex = complexTextRun.indexAt(i);
704 if (m_run.ltr()) {
705 if (characterIndex < lastCharacterIndex)
706 isMonotonic = false;
707 } else {
708 if (characterIndex > lastCharacterIndex)
709 isMonotonic = false;
710 }
711 UChar ch = *(cp + characterIndex);
712
713 bool treatAsSpace = FontCascade::treatAsSpace(ch);
714 CGGlyph glyph = treatAsSpace ? font.spaceGlyph() : glyphs[i];
715 FloatSize advance = treatAsSpace ? FloatSize(spaceWidth, advances[i].height()) : advances[i];
716
717 if (ch == '\t' && m_run.allowTabs())
718 advance.setWidth(m_font.tabWidth(font, m_run.tabSize(), m_run.xPos() + m_totalAdvance.width()));
719 else if (FontCascade::treatAsZeroWidthSpace(ch) && !treatAsSpace) {
720 advance.setWidth(0);
721 glyph = font.spaceGlyph();
722 }
723
724 if (!i) {
725 advance.expand(complexTextRun.initialAdvance().width(), complexTextRun.initialAdvance().height());
726 if (auto* origins = complexTextRun.glyphOrigins())
727 advance.expand(-origins[0].x(), -origins[0].y());
728 }
729
730 advance.expand(font.syntheticBoldOffset(), 0);
731
732 if (hasExtraSpacing) {
733 // If we're a glyph with an advance, add in letter-spacing.
734 // That way we weed out zero width lurkers. This behavior matches the fast text code path.
735 if (advance.width())
736 advance.expand(m_font.letterSpacing(), 0);
737
738 unsigned characterIndexInRun = characterIndex + complexTextRun.stringLocation();
739 bool isFirstCharacter = !(characterIndex + complexTextRun.stringLocation());
740 bool isLastCharacter = characterIndexInRun + 1 == m_run.length() || (U16_IS_LEAD(ch) && characterIndexInRun + 2 == m_run.length() && U16_IS_TRAIL(*(cp + characterIndex + 1)));
741
742 bool forceLeftExpansion = false; // On the left, regardless of m_run.ltr()
743 bool forceRightExpansion = false; // On the right, regardless of m_run.ltr()
744 bool forbidLeftExpansion = false;
745 bool forbidRightExpansion = false;
746 if (runForcesLeftExpansion)
747 forceLeftExpansion = m_run.ltr() ? isFirstCharacter : isLastCharacter;
748 if (runForcesRightExpansion)
749 forceRightExpansion = m_run.ltr() ? isLastCharacter : isFirstCharacter;
750 if (runForbidsLeftExpansion)
751 forbidLeftExpansion = m_run.ltr() ? isFirstCharacter : isLastCharacter;
752 if (runForbidsRightExpansion)
753 forbidRightExpansion = m_run.ltr() ? isLastCharacter : isFirstCharacter;
754 // Handle justification and word-spacing.
755 static bool expandAroundIdeographs = FontCascade::canExpandAroundIdeographsInComplexText();
756 bool ideograph = expandAroundIdeographs && FontCascade::isCJKIdeographOrSymbol(ch);
757 if (treatAsSpace || ideograph || forceLeftExpansion || forceRightExpansion) {
758 // Distribute the run's total expansion evenly over all expansion opportunities in the run.
759 if (m_expansion) {
760 auto [expandLeft, expandRight] = expansionLocation(ideograph, treatAsSpace, m_run.ltr(), afterExpansion, forbidLeftExpansion, forbidRightExpansion, forceLeftExpansion, forceRightExpansion);
761 if (expandLeft) {
762 m_expansion -= m_expansionPerOpportunity;
763 // Increase previous width
764 if (m_adjustedBaseAdvances.isEmpty()) {
765 advance.expand(m_expansionPerOpportunity, 0);
766 complexTextRun.growInitialAdvanceHorizontally(m_expansionPerOpportunity);
767 } else {
768 m_adjustedBaseAdvances.last().expand(m_expansionPerOpportunity, 0);
769 m_totalAdvance.expand(m_expansionPerOpportunity, 0);
770 }
771 }
772 if (expandRight) {
773 m_expansion -= m_expansionPerOpportunity;
774 advance.expand(m_expansionPerOpportunity, 0);
775 afterExpansion = true;
776 }
777 } else
778 afterExpansion = false;
779
780 // Account for word-spacing.
781 if (treatAsSpace && (ch != '\t' || !m_run.allowTabs()) && (characterIndex > 0 || runIndex > 0 || ch == noBreakSpace) && m_font.wordSpacing())
782 advance.expand(m_font.wordSpacing(), 0);
783 } else
784 afterExpansion = false;
785 }
786
787 m_totalAdvance += advance;
788
789 // FIXME: Combining marks should receive a text emphasis mark if they are combine with a space.
790 if (m_forTextEmphasis && (!FontCascade::canReceiveTextEmphasis(ch) || (U_GET_GC_MASK(ch) & U_GC_M_MASK)))
791 glyph = 0;
792
793 m_adjustedBaseAdvances.append(advance);
794 if (auto* origins = complexTextRun.glyphOrigins()) {
795 ASSERT(m_glyphOrigins.size() < m_adjustedBaseAdvances.size());
796 m_glyphOrigins.grow(m_adjustedBaseAdvances.size());
797 m_glyphOrigins[m_glyphOrigins.size() - 1] = origins[i];
798 ASSERT(m_glyphOrigins.size() == m_adjustedBaseAdvances.size());
799 }
800 m_adjustedGlyphs.append(glyph);
801
802 FloatRect glyphBounds = font.boundsForGlyph(glyph);
803 glyphBounds.move(glyphOrigin.x(), glyphOrigin.y());
804 m_minGlyphBoundingBoxX = std::min(m_minGlyphBoundingBoxX, glyphBounds.x());
805 m_maxGlyphBoundingBoxX = std::max(m_maxGlyphBoundingBoxX, glyphBounds.maxX());
806 m_minGlyphBoundingBoxY = std::min(m_minGlyphBoundingBoxY, glyphBounds.y());
807 m_maxGlyphBoundingBoxY = std::max(m_maxGlyphBoundingBoxY, glyphBounds.maxY());
808 glyphOrigin.move(advance);
809
810 lastCharacterIndex = characterIndex;
811 }
812 if (!isMonotonic)
813 complexTextRun.setIsNonMonotonic();
814 }
815}

817// Missing glyphs run constructor. Core Text will not generate a run of missing glyphs, instead falling back on
818// glyphs from LastResort. We want to use the primary font's missing glyph in order to match the fast text code path.
819ComplexTextController::ComplexTextRun::ComplexTextRun(const Font& font, const UChar* characters, unsigned stringLocation, unsigned stringLength, unsigned indexBegin, unsigned indexEnd, bool ltr)
820 : m_font(font)
821 , m_characters(characters)
822 , m_stringLength(stringLength)
823 , m_indexBegin(indexBegin)
824 , m_indexEnd(indexEnd)
825 , m_stringLocation(stringLocation)
826 , m_isLTR(ltr)
827{
828 auto runLengthInCodeUnits = m_indexEnd - m_indexBegin;
829 m_coreTextIndices.reserveInitialCapacity(runLengthInCodeUnits);
830 unsigned r = m_indexBegin;
831 while (r < m_indexEnd) {
832 m_coreTextIndices.uncheckedAppend(r);
833 UChar32 character;
834 U16_NEXT(m_characters, r, m_stringLength, character);
835 }
836 m_glyphCount = m_coreTextIndices.size();
837 if (!ltr) {
838 for (unsigned r = 0, end = m_glyphCount - 1; r < m_glyphCount / 2; ++r, --end)
839 std::swap(m_coreTextIndices[r], m_coreTextIndices[end]);
840 }
841
842 // Synthesize a run of missing glyphs.
843 m_glyphs.fill(0, m_glyphCount);
844 m_baseAdvances.fill(FloatSize(m_font.widthForGlyph(0), 0), m_glyphCount);
845}
