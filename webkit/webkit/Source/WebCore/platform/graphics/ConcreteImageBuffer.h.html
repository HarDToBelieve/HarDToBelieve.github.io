33template<typename BackendType>
34class ConcreteImageBuffer : public ImageBuffer {
35public:
36 template<typename ImageBufferType = ConcreteImageBuffer, typename... Arguments>
37 static RefPtr<ImageBufferType> create(const FloatSize& size, float resolutionScale, const DestinationColorSpace& colorSpace, PixelFormat pixelFormat, const HostWindow* hostWindow, Arguments&&... arguments)
38 {
39 auto parameters = ImageBufferBackend::Parameters { size, resolutionScale, colorSpace, pixelFormat };
40 auto backend = BackendType::create(parameters, hostWindow);
41 if (!backend)
42 return nullptr;
43 return adoptRef(new ImageBufferType(parameters, WTFMove(backend), std::forward<Arguments>(arguments)...));
44 }
45
46 template<typename ImageBufferType = ConcreteImageBuffer, typename... Arguments>
47 static RefPtr<ImageBufferType> create(const FloatSize& size, const GraphicsContext& context, Arguments&&... arguments)
48 {
49 auto parameters = ImageBufferBackend::Parameters { size, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8 };
50 auto backend = BackendType::create(parameters, context);
51 if (!backend)
52 return nullptr;
53 return adoptRef(new ImageBufferType(parameters, WTFMove(backend), std::forward<Arguments>(arguments)...));
54 }
55
56 RenderingMode renderingMode() const override { return BackendType::renderingMode; }
57 bool canMapBackingStore() const override { return BackendType::canMapBackingStore; }
58
59protected:
60 ConcreteImageBuffer(const ImageBufferBackend::Parameters& parameters, std::unique_ptr<BackendType>&& backend = nullptr, RenderingResourceIdentifier renderingResourceIdentifier = RenderingResourceIdentifier::generate())
61 : m_parameters(parameters)
62 , m_backend(WTFMove(backend))
63 , m_renderingResourceIdentifier(renderingResourceIdentifier)
64 {
65 }
66
67 void setBackend(std::unique_ptr<ImageBufferBackend>&& backend) override
68 {
69 ASSERT(!m_backend);
70 m_backend = std::unique_ptr<BackendType> { static_cast<BackendType*>(backend.release()) };
71 }
72
73 void clearBackend() override { m_backend = nullptr; }
74 ImageBufferBackend* ensureBackendCreated() const override { return m_backend.get(); }
75
76 RenderingResourceIdentifier renderingResourceIdentifier() const override { return m_renderingResourceIdentifier; }
77
78 GraphicsContext& context() const override
79 {
80 ASSERT(m_backend);
81 return m_backend->context();
82 }
83
84 void flushContext() override
85 {
86 if (auto* backend = ensureBackendCreated()) {
87 flushDrawingContext();
88 backend->flushContext();
89 }
90 }
91
92 IntSize logicalSize() const override { return IntSize(m_parameters.logicalSize); }
93 float resolutionScale() const override { return m_parameters.resolutionScale; }
94 DestinationColorSpace colorSpace() const override { return m_parameters.colorSpace; }
95 PixelFormat pixelFormat() const override { return m_parameters.pixelFormat; }
96 const ImageBufferBackend::Parameters& parameters() const override { return m_parameters; }
97
98 IntSize backendSize() const override
99 {
100 if (auto* backend = ensureBackendCreated())
101 return backend->backendSize();
102 return { };
103 }
104
105 AffineTransform baseTransform() const override
106 {
107 if (BackendType::isOriginAtBottomLeftCorner)
108 return AffineTransform(1, 0, 0, -1, 0, logicalSize().height());
109 return { };
110 }
111
112 size_t memoryCost() const override
113 {
114 return BackendType::calculateMemoryCost(m_parameters);
115 }
116
117 size_t externalMemoryCost() const override
118 {
119 return BackendType::calculateExternalMemoryCost(m_parameters);
120 }
121
122 RefPtr<NativeImage> copyNativeImage(BackingStoreCopy copyBehavior = CopyBackingStore) const override
123 {
124 if (auto* backend = ensureBackendCreated()) {
125 const_cast<ConcreteImageBuffer&>(*this).flushDrawingContext();
126 return backend->copyNativeImage(copyBehavior);
127 }
128 return nullptr;
129 }
130
131 RefPtr<Image> copyImage(BackingStoreCopy copyBehavior = CopyBackingStore, PreserveResolution preserveResolution = PreserveResolution::No) const override
132 {
133 if (auto* backend = ensureBackendCreated()) {
134 const_cast<ConcreteImageBuffer&>(*this).flushDrawingContext();
135 return backend->copyImage(copyBehavior, preserveResolution);
136 }
137 return nullptr;
138 }

207 String toDataURL(const String& mimeType, std::optional<double> quality, PreserveResolution preserveResolution) const override
208 {
209 if (auto* backend = ensureBackendCreated()) {
210 const_cast<ConcreteImageBuffer&>(*this).flushContext();
211 return backend->toDataURL(mimeType, quality, preserveResolution);
212 }
213 return String();
214 }

221 }
222 return { };
223 }

254 }

274 }

280 return nullptr;
281 }

283 void releaseBufferToPool() override
284 {
285 if (auto* backend = ensureBackendCreated())
286 backend->releaseBufferToPool();
287 }

292};
