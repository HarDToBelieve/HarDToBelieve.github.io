28#include "FloatSize.h"
29#include "GeneratedImage.h"
30#include "Image.h"
31#include <wtf/RefPtr.h>
32
33namespace WebCore {
34
35class CrossfadeGeneratedImage final : public GeneratedImage {
36public:
37 static Ref<CrossfadeGeneratedImage> create(Image& fromImage, Image& toImage, float percentage, const FloatSize& crossfadeSize, const FloatSize& size)
38 {
39 return adoptRef(*new CrossfadeGeneratedImage(fromImage, toImage, percentage, crossfadeSize, size));
40 }
41
42 void setContainerSize(const FloatSize&) override { }
43 bool usesContainerSize() const override { return false; }
44 bool hasRelativeWidth() const override { return false; }
45 bool hasRelativeHeight() const override { return false; }
46
47 FloatSize size(ImageOrientation = ImageOrientation::FromImage) const override { return m_crossfadeSize; }
48
49private:
50 ImageDrawResult draw(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, const ImagePaintingOptions& = { }) override;
51 void drawPattern(GraphicsContext&, const FloatRect& dstRect, const FloatRect& srcRect, const AffineTransform& patternTransform, const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& = { }) override;
52
53 CrossfadeGeneratedImage(Image& fromImage, Image& toImage, float percentage, const FloatSize& crossfadeSize, const FloatSize&);
54
55 bool isCrossfadeGeneratedImage() const override { return true; }
56 void dump(WTF::TextStream&) const override;
57
58 void drawCrossfade(GraphicsContext&);
59
60 Ref<Image> m_fromImage;
61 Ref<Image> m_toImage;
62
63 float m_percentage;
64 FloatSize m_crossfadeSize;
65};
66
67}
