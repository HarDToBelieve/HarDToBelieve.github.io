26#pragma once
27
28#include "PlatformColorSpace.h"
29#include <optional>
30#include <wtf/Forward.h>
31
32namespace WebCore {
33
34class DestinationColorSpace {
35public:
36 WEBCORE_EXPORT static const DestinationColorSpace& SRGB();
37#if ENABLE(DESTINATION_COLOR_SPACE_LINEAR_SRGB)
38 WEBCORE_EXPORT static const DestinationColorSpace& LinearSRGB();
39#endif
40#if ENABLE(DESTINATION_COLOR_SPACE_DISPLAY_P3)
41 WEBCORE_EXPORT static const DestinationColorSpace& DisplayP3();
42#endif
43
44 WEBCORE_EXPORT explicit DestinationColorSpace(PlatformColorSpace);
45 PlatformColorSpaceValue platformColorSpace() const { return m_platformColorSpace.get(); }
46
47 template<class Encoder> void encode(Encoder&) const;
48 template<class Decoder> static std::optional<DestinationColorSpace> decode(Decoder&);
49
50private:
51 PlatformColorSpace m_platformColorSpace;
52};
53
54WEBCORE_EXPORT bool operator==(const DestinationColorSpace&, const DestinationColorSpace&);
55WEBCORE_EXPORT bool operator!=(const DestinationColorSpace&, const DestinationColorSpace&);
56
57WEBCORE_EXPORT TextStream& operator<<(TextStream&, const DestinationColorSpace&);
58
59template<class Encoder> void DestinationColorSpace::encode(Encoder& encoder) const
60{
61 encoder << m_platformColorSpace;
62}
63
64template<class Decoder> std::optional<DestinationColorSpace> DestinationColorSpace::decode(Decoder& decoder)
65{
66 std::optional<PlatformColorSpace> platformColorSpace;
67 decoder >> platformColorSpace;
68 if (!platformColorSpace)
69 return std::nullopt;
70
71 return DestinationColorSpace { WTFMove(*platformColorSpace) };
72}
73
74}
