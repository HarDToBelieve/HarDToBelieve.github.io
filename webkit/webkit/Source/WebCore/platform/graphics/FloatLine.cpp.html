34const FloatPoint FloatLine::pointAtAbsoluteDistance(float absoluteDistance) const
35{
36 if (!length())
37 return m_start;
38 auto relativeDistance = absoluteDistance / length();
39 return pointAtRelativeDistance(relativeDistance);
40}
41
42const FloatPoint FloatLine::pointAtRelativeDistance(float relativeDistance) const
43{
44 return {
45 m_start.x() - (relativeDistance * (m_start.x() - m_end.x())),
46 m_start.y() - (relativeDistance * (m_start.y() - m_end.y())),
47 };
48}

50const FloatLine FloatLine::extendedToBounds(const FloatRect& bounds) const
51{
52 if (fabs(m_start.x() - m_end.x()) <= fabs(m_start.y() - m_end.y())) {
53 // The line is roughly vertical, so construct points at the top and bottom of the bounds.
54 FloatPoint top = { (((bounds.y() - m_start.y()) * (m_end.x() - m_start.x())) / (m_end.y() - m_start.y())) + m_start.x(), bounds.y() };
55 FloatPoint bottom = { (((bounds.y() + bounds.height() - m_start.y()) * (m_end.x() - m_start.x())) / (m_end.y() - m_start.y())) + m_start.x(), bounds.y() + bounds.height() };
56 return { top, bottom };
57 }
58
59 // The line is roughly horizontal, so construct points at the left and right of the bounds.
60 FloatPoint left = { bounds.x(), (((bounds.x() - m_start.x()) * (m_end.y() - m_start.y())) / (m_end.x() - m_start.x())) + m_start.y() };
61 FloatPoint right = { bounds.x() + bounds.width(), (((bounds.x() + bounds.width() - m_start.x()) * (m_end.y() - m_start.y())) / (m_end.x() - m_start.x())) + m_start.y() };
62 return { left, right };
63}

65const std::optional<FloatPoint> FloatLine::intersectionWith(const FloatLine& otherLine) const
66{
67 float denominator = ((m_start.x() - m_end.x()) * (otherLine.start().y() - otherLine.end().y())) - ((m_start.y() - m_end.y()) * (otherLine.start().x() - otherLine.end().x()));
68
69 // A denominator of zero indicates the lines are parallel or coincident, which means there is no true intersection.
70 if (!denominator)
71 return std::nullopt;
72
73 float thisLineCommonNumeratorFactor = (m_start.x() * m_end.y()) - (m_start.y() * m_end.x());
74 float otherLineCommonNumeratorFactor = (otherLine.start().x() * otherLine.end().y()) - (otherLine.start().y() * otherLine.end().x());
75
76 return {{
77 ((thisLineCommonNumeratorFactor * (otherLine.start().x() - otherLine.end().x())) - (otherLineCommonNumeratorFactor * (m_start.x() - m_end.x()))) / denominator,
78 ((thisLineCommonNumeratorFactor * (otherLine.start().y() - otherLine.end().y())) - (otherLineCommonNumeratorFactor * (m_start.y() - m_end.y()))) / denominator,
79 }};
80}
