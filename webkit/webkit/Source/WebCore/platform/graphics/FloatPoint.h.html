61class FloatPoint {
62 WTF_MAKE_FAST_ALLOCATED;
63public:
64 FloatPoint() { }
65 FloatPoint(float x, float y) : m_x(x), m_y(y) { }
66 WEBCORE_EXPORT FloatPoint(const IntPoint&);
67 explicit FloatPoint(const FloatSize& size) : m_x(size.width()), m_y(size.height()) { }
68
69 static FloatPoint zero() { return FloatPoint(); }
70 bool isZero() const { return !m_x && !m_y; }
71
72 WEBCORE_EXPORT static FloatPoint narrowPrecision(double x, double y);
73
74 float x() const { return m_x; }
75 float y() const { return m_y; }
76
77 void setX(float x) { m_x = x; }
78 void setY(float y) { m_y = y; }
79
80 void set(float x, float y)
81 {
82 m_x = x;
83 m_y = y;
84 }
85
86 void move(float dx, float dy)
87 {
88 m_x += dx;
89 m_y += dy;
90 }
91
92 void move(const IntSize& a)
93 {
94 m_x += a.width();
95 m_y += a.height();
96 }
97
98 void move(const FloatSize& a)
99 {
100 m_x += a.width();
101 m_y += a.height();
102 }
103
104 void moveBy(const IntPoint& a)
105 {
106 m_x += a.x();
107 m_y += a.y();
108 }
109
110 void moveBy(const FloatPoint& a)
111 {
112 m_x += a.x();
113 m_y += a.y();
114 }
115
116 void scale(float scale)
117 {
118 m_x *= scale;
119 m_y *= scale;
120 }
121
122 void scale(float scaleX, float scaleY)
123 {
124 m_x *= scaleX;
125 m_y *= scaleY;
126 }
127
128 FloatPoint scaled(float scale) const
129 {
130 return { m_x * scale, m_y * scale };
131 }
132
133 FloatPoint scaled(float scaleX, float scaleY) const
134 {
135 return { m_x * scaleX, m_y * scaleY };
136 }
137
138 void rotate(double angleInRadians, const FloatPoint& aboutPoint = { })
139 {
140 auto sinAngle = sin(angleInRadians);
141 auto cosAngle = cos(angleInRadians);
142 m_x -= aboutPoint.x();
143 m_y -= aboutPoint.y();
144 auto newX = m_x * cosAngle - m_y * sinAngle + aboutPoint.x();
145 m_y = m_x * sinAngle + m_y * cosAngle + aboutPoint.y();
146 m_x = newX;
147 }
148
149 WEBCORE_EXPORT void normalize();
150
151 float dot(const FloatPoint& a) const
152 {
153 return m_x * a.x() + m_y * a.y();
154 }
155
156 float slopeAngleRadians() const;
157
158 float length() const
159 {
160 return std::hypot(m_x, m_y);
161 }
162
163 float lengthSquared() const
164 {
165 return m_x * m_x + m_y * m_y;
166 }
167
168 WEBCORE_EXPORT FloatPoint constrainedBetween(const FloatPoint& min, const FloatPoint& max) const;
169
170 FloatPoint shrunkTo(const FloatPoint& other) const
171 {
172 return { std::min(m_x, other.m_x), std::min(m_y, other.m_y) };
173 }
174
175 FloatPoint expandedTo(const FloatPoint& other) const
176 {
177 return { std::max(m_x, other.m_x), std::max(m_y, other.m_y) };
178 }
179
180 FloatPoint transposedPoint() const
181 {
182 return { m_y, m_x };
183 }
184
185#if USE(CG)
186 WEBCORE_EXPORT FloatPoint(const CGPoint&);
187 WEBCORE_EXPORT operator CGPoint() const;
188#endif
189
190#if PLATFORM(MAC) && !defined(NSGEOMETRY_TYPES_SAME_AS_CGGEOMETRY_TYPES)
191 WEBCORE_EXPORT FloatPoint(const NSPoint&);
192 WEBCORE_EXPORT operator NSPoint() const;
193#endif
194
195#if PLATFORM(WIN)
196 WEBCORE_EXPORT FloatPoint(const D2D1_POINT_2F&);
197 WEBCORE_EXPORT operator D2D1_POINT_2F() const;
198#endif
199
200 WEBCORE_EXPORT FloatPoint matrixTransform(const TransformationMatrix&) const;
201 WEBCORE_EXPORT FloatPoint matrixTransform(const AffineTransform&) const;
202
203private:
204 float m_x { 0 };
205 float m_y { 0 };
206};

209inline FloatPoint& operator+=(FloatPoint& a, const FloatSize& b)
210{
211 a.move(b.width(), b.height());
212 return a;
213}
214
215inline FloatPoint& operator+=(FloatPoint& a, const FloatPoint& b)
216{
217 a.move(b.x(), b.y());
218 return a;
219}
220
221inline FloatPoint& operator-=(FloatPoint& a, const FloatSize& b)
222{
223 a.move(-b.width(), -b.height());
224 return a;
225}
226
227inline FloatPoint operator+(const FloatPoint& a, const FloatSize& b)
228{
229 return FloatPoint(a.x() + b.width(), a.y() + b.height());
230}
231
232inline FloatPoint operator+(const FloatPoint& a, const FloatPoint& b)

241
242inline FloatPoint operator-(const FloatPoint& a, const FloatSize& b)

244 return FloatPoint(a.x() - b.width(), a.y() - b.height());
245}
246
247inline FloatPoint operator-(const FloatPoint& a)
248{
249 return FloatPoint(-a.x(), -a.y());
250}
251
252inline bool operator==(const FloatPoint& a, const FloatPoint& b)
253{
254 return a.x() == b.x() && a.y() == b.y();
255}
256
257inline bool operator!=(const FloatPoint& a, const FloatPoint& b)
258{
259 return a.x() != b.x() || a.y() != b.y();
260}
261
262inline float operator*(const FloatPoint& a, const FloatPoint& b)
263{
264 // dot product
265 return a.dot(b);
266}

268inline IntSize flooredIntSize(const FloatPoint& p)
269{
270 return IntSize(clampToInteger(floorf(p.x())), clampToInteger(floorf(p.y())));
271}
272
273inline IntPoint roundedIntPoint(const FloatPoint& p)
274{
275 return IntPoint(clampToInteger(roundf(p.x())), clampToInteger(roundf(p.y())));
276}
277
278inline IntPoint flooredIntPoint(const FloatPoint& p)
279{
280 return IntPoint(clampToInteger(floorf(p.x())), clampToInteger(floorf(p.y())));
281}
282
283inline IntPoint ceiledIntPoint(const FloatPoint& p)
284{
285 return IntPoint(clampToInteger(ceilf(p.x())), clampToInteger(ceilf(p.y())));
286}

288inline FloatPoint floorPointToDevicePixels(const FloatPoint& p, float deviceScaleFactor)
289{
290 return FloatPoint(floorf(p.x() * deviceScaleFactor) / deviceScaleFactor, floorf(p.y() * deviceScaleFactor) / deviceScaleFactor);
291}
292
293inline FloatPoint ceilPointToDevicePixels(const FloatPoint& p, float deviceScaleFactor)
294{
295 return FloatPoint(ceilf(p.x() * deviceScaleFactor) / deviceScaleFactor, ceilf(p.y() * deviceScaleFactor) / deviceScaleFactor);
296}

298inline FloatSize toFloatSize(const FloatPoint& a)
299{
300 return FloatSize(a.x(), a.y());
301}
302
303inline FloatPoint toFloatPoint(const FloatSize& a)
304{
305 return FloatPoint(a.width(), a.height());
306}
307
308inline bool areEssentiallyEqual(const FloatPoint& a, const FloatPoint& b)
309{
310 return WTF::areEssentiallyEqual(a.x(), b.x()) && WTF::areEssentiallyEqual(a.y(), b.y());
311}
