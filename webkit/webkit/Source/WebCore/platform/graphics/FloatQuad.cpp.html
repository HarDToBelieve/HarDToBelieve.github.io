41static inline float min4(float a, float b, float c, float d)
42{
43 return std::min(std::min(a, b), std::min(c, d));
44}
45
46static inline float max4(float a, float b, float c, float d)
47{
48 return std::max(std::max(a, b), std::max(c, d));
49}

51inline float dot(const FloatSize& a, const FloatSize& b)
52{
53 return a.width() * b.width() + a.height() * b.height();
54}
55
56inline float determinant(const FloatSize& a, const FloatSize& b)
57{
58 return a.width() * b.height() - a.height() * b.width();
59}

61inline bool isPointInTriangle(const FloatPoint& p, const FloatPoint& t1, const FloatPoint& t2, const FloatPoint& t3)
62{
63 // Compute vectors
64 FloatSize v0 = t3 - t1;
65 FloatSize v1 = t2 - t1;
66 FloatSize v2 = p - t1;
67
68 // Compute dot products
69 float dot00 = dot(v0, v0);
70 float dot01 = dot(v0, v1);
71 float dot02 = dot(v0, v2);
72 float dot11 = dot(v1, v1);
73 float dot12 = dot(v1, v2);
74
75 // Compute barycentric coordinates
76 float invDenom = 1.0f / (dot00 * dot11 - dot01 * dot01);
77 float u = (dot11 * dot02 - dot01 * dot12) * invDenom;
78 float v = (dot00 * dot12 - dot01 * dot02) * invDenom;
79
80 // Check if point is in triangle
81 return (u >= 0) && (v >= 0) && (u + v <= 1);
82}

84FloatRect FloatQuad::boundingBox() const
85{
86 float left = min4(m_p1.x(), m_p2.x(), m_p3.x(), m_p4.x());
87 float top = min4(m_p1.y(), m_p2.y(), m_p3.y(), m_p4.y());
88
89 float right = max4(m_p1.x(), m_p2.x(), m_p3.x(), m_p4.x());
90 float bottom = max4(m_p1.y(), m_p2.y(), m_p3.y(), m_p4.y());
91
92 return FloatRect(left, top, right - left, bottom - top);
93}

95bool FloatQuad::isRectilinear() const
96{
97 return (WTF::areEssentiallyEqual(m_p1.x(), m_p2.x()) && WTF::areEssentiallyEqual(m_p2.y(), m_p3.y()) && WTF::areEssentiallyEqual(m_p3.x(), m_p4.x()) && WTF::areEssentiallyEqual(m_p4.y(), m_p1.y()))
98 || (WTF::areEssentiallyEqual(m_p1.y(), m_p2.y()) && WTF::areEssentiallyEqual(m_p2.x(), m_p3.x()) && WTF::areEssentiallyEqual(m_p3.y(), m_p4.y()) && WTF::areEssentiallyEqual(m_p4.x(), m_p1.x()));
99}

101bool FloatQuad::containsPoint(const FloatPoint& p) const
102{
103 return isPointInTriangle(p, m_p1, m_p2, m_p3) || isPointInTriangle(p, m_p1, m_p3, m_p4);
104}

107bool FloatQuad::containsQuad(const FloatQuad& other) const
108{
109 return containsPoint(other.p1()) && containsPoint(other.p2()) && containsPoint(other.p3()) && containsPoint(other.p4());

112static inline FloatPoint rightMostCornerToVector(const FloatRect& rect, const FloatSize& vector)
113{
114 // Return the corner of the rectangle that if it is to the left of the vector
115 // would mean all of the rectangle is to the left of the vector.
116 // The vector here represents the side between two points in a clockwise convex polygon.
117 //
118 // Q XXX
119 // QQQ XXX If the lower left corner of X is left of the vector that goes from the top corner of Q to
120 // QQQ the right corner of Q, then all of X is left of the vector, and intersection impossible.
121 // Q
122 //
123 FloatPoint point;
124 if (vector.width() >= 0)
125 point.setY(rect.maxY());
126 else
127 point.setY(rect.y());
128 if (vector.height() >= 0)
129 point.setX(rect.x());
130 else
131 point.setX(rect.maxX());
132 return point;
133}

135bool FloatQuad::intersectsRect(const FloatRect& rect) const
136{
137 // For each side of the quad clockwise we check if the rectangle is to the left of it
138 // since only content on the right can onlap with the quad.
139 // This only works if the quad is convex.
140 FloatSize v1, v2, v3, v4;
141
142 // Ensure we use clockwise vectors.
143 if (!isCounterclockwise()) {
144 v1 = m_p2 - m_p1;
145 v2 = m_p3 - m_p2;
146 v3 = m_p4 - m_p3;
147 v4 = m_p1 - m_p4;
148 } else {
149 v1 = m_p4 - m_p1;
150 v2 = m_p1 - m_p2;
151 v3 = m_p2 - m_p3;
152 v4 = m_p3 - m_p4;
153 }
154
155 FloatPoint p = rightMostCornerToVector(rect, v1);
156 if (determinant(v1, p - m_p1) < 0)
157 return false;
158
159 p = rightMostCornerToVector(rect, v2);
160 if (determinant(v2, p - m_p2) < 0)
161 return false;
162
163 p = rightMostCornerToVector(rect, v3);
164 if (determinant(v3, p - m_p3) < 0)
165 return false;
166
167 p = rightMostCornerToVector(rect, v4);
168 if (determinant(v4, p - m_p4) < 0)
169 return false;
170
171 // If not all of the rectangle is outside one of the quad's four sides, then that means at least
172 // a part of the rectangle is overlapping the quad.
173 return true;
174}

176// Tests whether the line is contained by or intersected with the circle.
177static inline bool lineIntersectsCircle(const FloatPoint& center, float radius, const FloatPoint& p0, const FloatPoint& p1)
178{
179 float x0 = p0.x() - center.x(), y0 = p0.y() - center.y();
180 float x1 = p1.x() - center.x(), y1 = p1.y() - center.y();
181 float radius2 = radius * radius;
182 if ((x0 * x0 + y0 * y0) <= radius2 || (x1 * x1 + y1 * y1) <= radius2)
183 return true;
184 if (p0 == p1)
185 return false;
186
187 float a = y0 - y1;
188 float b = x1 - x0;
189 float c = x0 * y1 - x1 * y0;
190 float distance2 = c * c / (a * a + b * b);
191 // If distance between the center point and the line > the radius,
192 // the line doesn't cross (or is contained by) the ellipse.
193 if (distance2 > radius2)
194 return false;
195
196 // The nearest point on the line is between p0 and p1?
197 float x = - a * c / (a * a + b * b);
198 float y = - b * c / (a * a + b * b);
199 return (((x0 <= x && x <= x1) || (x0 >= x && x >= x1))
200 && ((y0 <= y && y <= y1) || (y1 <= y && y <= y0)));
201}

203bool FloatQuad::intersectsCircle(const FloatPoint& center, float radius) const
204{
205 return containsPoint(center) // The circle may be totally contained by the quad.
206 || lineIntersectsCircle(center, radius, m_p1, m_p2)
207 || lineIntersectsCircle(center, radius, m_p2, m_p3)
208 || lineIntersectsCircle(center, radius, m_p3, m_p4)
209 || lineIntersectsCircle(center, radius, m_p4, m_p1);
210}

212bool FloatQuad::intersectsEllipse(const FloatPoint& center, const FloatSize& radii) const
213{
214 // Transform the ellipse to an origin-centered circle whose radius is the product of major radius and minor radius.
215 // Here we apply the same transformation to the quad.
216 FloatQuad transformedQuad(*this);
217 transformedQuad.move(-center.x(), -center.y());
218 transformedQuad.scale(radii.height(), radii.width());
219
220 FloatPoint originPoint;
221 return transformedQuad.intersectsCircle(originPoint, radii.height() * radii.width());
222
223}

225bool FloatQuad::isCounterclockwise() const
226{
227 // Return if the two first vectors are turning clockwise. If the quad is convex then all following vectors will turn the same way.
228 return determinant(m_p2 - m_p1, m_p3 - m_p2) < 0;
229}

231bool FloatQuad::isEmpty() const
232{
233 if (areEssentiallyEqual(m_p1, m_p3) || areEssentiallyEqual(m_p2, m_p4)) {
234 // If either diagonal is zero length, then the "quad" either consists of 1 or 2 line segments, or it's just a point.
235 return true;
236 }
237
238 if (areEssentiallyEqual(m_p1, m_p2) && areEssentiallyEqual(m_p3, m_p4)) {
239 // If both top points and both bottom points are equal, then the "quad" is just a single line segment.
240 return true;
241 }
242
243 if (areEssentiallyEqual(m_p1, m_p4) && areEssentiallyEqual(m_p2, m_p3)) {
244 // If both left points and both right points are equal, then the "quad" is just a single line segment.
245 return true;
246 }
247
248 // Fall back to checking whether the 4 points of the quad are colinear (in other words, check whether the three
249 // vectors from one point to each of the other points are capable of forming a 2D basis).
250 auto b1 = m_p1 - m_p2;
251 auto b2 = m_p1 - m_p3;
252 auto b3 = m_p1 - m_p4;
253
254 if (!b1.isZero())
255 b1 = b1 / b1.diagonalLength();
256
257 if (!b2.isZero())
258 b2 = b2 / b2.diagonalLength();
259
260 if (!b3.isZero())
261 b3 = b3 / b3.diagonalLength();
262
263 auto areNormalizedVectorsLinearlyIndependent = [](const FloatSize& u, const FloatSize& v) {
264 if (u.isZero() || v.isZero())
265 return false;
266
267 auto dotProduct = u.width() * v.width() + u.height() * v.height();
268 return !WTF::areEssentiallyEqual<float>(dotProduct, 1) && !WTF::areEssentiallyEqual<float>(dotProduct, -1);
269 };
270
271 return !areNormalizedVectorsLinearlyIndependent(b1, b2) && !areNormalizedVectorsLinearlyIndependent(b2, b3) && !areNormalizedVectorsLinearlyIndependent(b1, b3);
272}

274Vector<FloatRect> boundingBoxes(const Vector<FloatQuad>& quads)
275{
276 Vector<FloatRect> boxes;
277 boxes.reserveInitialCapacity(quads.size());
278 for (const auto& quad : quads)
279 boxes.uncheckedAppend(quad.boundingBox());
280 return boxes;
281}
282
283FloatRect unitedBoundingBoxes(const Vector<FloatQuad>& quads)
284{
285 auto size = quads.size();
286 if (!size)
287 return { };
288 auto result = quads[0].boundingBox();
289 for (size_t i = 1; i < size; ++i)
290 result.uniteEvenIfEmpty(quads[i].boundingBox());
291 return result;
292}
