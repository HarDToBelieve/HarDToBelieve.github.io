63class FloatRect {
64public:
65 enum ContainsMode {
66 InsideOrOnStroke,
67 InsideButNotOnStroke
68 };
69
70 FloatRect() { }
71 FloatRect(const FloatPoint& location, const FloatSize& size)
72 : m_location(location), m_size(size) { }
73 FloatRect(float x, float y, float width, float height)
74 : m_location(FloatPoint(x, y)), m_size(FloatSize(width, height)) { }
75 FloatRect(const FloatPoint& topLeft, const FloatPoint& bottomRight)
76 : m_location(topLeft), m_size(FloatSize(bottomRight.x() - topLeft.x(), bottomRight.y() - topLeft.y())) { }
77 WEBCORE_EXPORT FloatRect(const IntRect&);
78
79 static FloatRect narrowPrecision(double x, double y, double width, double height);
80
81 FloatPoint location() const { return m_location; }
82 FloatSize size() const { return m_size; }
83
84 void setLocation(const FloatPoint& location) { m_location = location; }
85 void setSize(const FloatSize& size) { m_size = size; }
86
87 float x() const { return m_location.x(); }
88 float y() const { return m_location.y(); }
89 float maxX() const { return x() + width(); }
90 float maxY() const { return y() + height(); }
91 float width() const { return m_size.width(); }
92 float height() const { return m_size.height(); }
93
94 float area() const { return m_size.area(); }
95
96 void setX(float x) { m_location.setX(x); }
97 void setY(float y) { m_location.setY(y); }
98 void setWidth(float width) { m_size.setWidth(width); }
99 void setHeight(float height) { m_size.setHeight(height); }
100
101 bool isEmpty() const { return m_size.isEmpty(); }
102 bool isZero() const { return m_size.isZero(); }
103 bool isExpressibleAsIntRect() const;
104
105 FloatPoint center() const { return location() + size() / 2; }
106
107 void move(const FloatSize& delta) { m_location += delta; }
108 void moveBy(const FloatPoint& delta) { m_location.move(delta.x(), delta.y()); }
109 void move(float dx, float dy) { m_location.move(dx, dy); }
110
111 void expand(const FloatSize& size) { m_size += size; }
112 void expand(const FloatBoxExtent& box)
113 {
114 m_location.move(-box.left(), -box.top());
115 m_size.expand(box.left() + box.right(), box.top() + box.bottom());
116 }
117 void expand(float dw, float dh) { m_size.expand(dw, dh); }
118 void contract(const FloatSize& size) { m_size -= size; }
119 void contract(float dw, float dh) { m_size.expand(-dw, -dh); }
120
121 void shiftXEdgeTo(float edge)
122 {
123 float delta = edge - x();
124 setX(edge);
125 setWidth(std::max(0.0f, width() - delta));
126 }
127
128 void shiftMaxXEdgeTo(float edge)
129 {
130 float delta = edge - maxX();
131 setWidth(std::max(0.0f, width() + delta));
132 }
133
134 void shiftYEdgeTo(float edge)
135 {
136 float delta = edge - y();
137 setY(edge);
138 setHeight(std::max(0.0f, height() - delta));
139 }
140
141 void shiftMaxYEdgeTo(float edge)
142 {
143 float delta = edge - maxY();
144 setHeight(std::max(0.0f, height() + delta));
145 }
146
147 void shiftXEdgeBy(float delta)
148 {
149 move(delta, 0);
150 setWidth(std::max(0.0f, width() - delta));
151 }
152
153 void shiftYEdgeBy(float delta)
154 {
155 move(0, delta);
156 setHeight(std::max(0.0f, height() - delta));
157 }

159 FloatPoint minXMinYCorner() const { return m_location; } // typically topLeft
160 FloatPoint maxXMinYCorner() const { return FloatPoint(m_location.x() + m_size.width(), m_location.y()); } // typically topRight
161 FloatPoint minXMaxYCorner() const { return FloatPoint(m_location.x(), m_location.y() + m_size.height()); } // typically bottomLeft
162 FloatPoint maxXMaxYCorner() const { return FloatPoint(m_location.x() + m_size.width(), m_location.y() + m_size.height()); } // typically bottomRight

176 // Note, this doesn't match what IntRect::contains(IntPoint&) does; the int version
177 // is really checking for containment of 1x1 rect, but that doesn't make sense with floats.
178 bool contains(float px, float py) const
179 { return px >= x() && px <= maxX() && py >= y() && py <= maxY(); }
180
181 bool overlapsYRange(float y1, float y2) const { return !isEmpty() && y2 >= y1 && y2 >= y() && y1 <= maxY(); }
182 bool overlapsXRange(float x1, float x2) const { return !isEmpty() && x2 >= x1 && x2 >= x() && x1 <= maxX(); }
183
184 void inflateX(float dx) {
185 m_location.setX(m_location.x() - dx);
186 m_size.setWidth(m_size.width() + dx + dx);
187 }
188 void inflateY(float dy) {
189 m_location.setY(m_location.y() - dy);
190 m_size.setHeight(m_size.height() + dy + dy);
191 }
192 void inflate(float d) { inflateX(d); inflateY(d); }
193 void inflate(FloatSize size) { inflateX(size.width()); inflateY(size.height()); }
194
195 void scale(float s) { scale(s, s); }
196 WEBCORE_EXPORT void scale(float sx, float sy);
197 void scale(FloatSize size) { scale(size.width(), size.height()); }

234 void setLocationAndSizeFromEdges(float left, float top, float right, float bottom)
235 {
236 m_location.set(left, top);
237 m_size.setWidth(right - left);
238 m_size.setHeight(bottom - top);
239 }

242inline FloatRect intersection(const FloatRect& a, const FloatRect& b)
243{
244 FloatRect c = a;
245 c.intersect(b);
246 return c;
247}
248
249inline FloatRect unionRect(const FloatRect& a, const FloatRect& b)
250{
251 FloatRect c = a;
252 c.unite(b);
253 return c;
254}
255
256inline FloatRect& operator+=(FloatRect& a, const FloatRect& b)
257{
258 a.move(b.x(), b.y());
259 a.setWidth(a.width() + b.width());
260 a.setHeight(a.height() + b.height());
261 return a;
262}
263
264inline FloatRect operator+(const FloatRect& a, const FloatRect& b)
265{
266 FloatRect c = a;
267 c += b;
268 return c;
269}
270
271inline bool operator==(const FloatRect& a, const FloatRect& b)
272{
273 return a.location() == b.location() && a.size() == b.size();
274}
275
276inline bool operator!=(const FloatRect& a, const FloatRect& b)
277{
278 return a.location() != b.location() || a.size() != b.size();
279}
280
281inline FloatRect FloatRect::infiniteRect()
282{
283 static FloatRect infiniteRect(-std::numeric_limits<float>::max() / 2, -std::numeric_limits<float>::max() / 2, std::numeric_limits<float>::max(), std::numeric_limits<float>::max());
284 return infiniteRect;
285}
286
287inline bool FloatRect::isInfinite() const
288{
289 return *this == infiniteRect();
290}
