152inline bool operator==(const FloatRoundedRect::Radii& a, const FloatRoundedRect::Radii& b)
153{
154 return a.topLeft() == b.topLeft() && a.topRight() == b.topRight() && a.bottomLeft() == b.bottomLeft() && a.bottomRight() == b.bottomRight();
155}
156
157inline bool operator!=(const FloatRoundedRect::Radii& a, const FloatRoundedRect::Radii& b)
158{
159 return !(a == b);
160}
161
162inline bool operator==(const FloatRoundedRect& a, const FloatRoundedRect& b)
163{
164 return a.rect() == b.rect() && a.radii() == b.radii();
165}
166
167inline bool operator!=(const FloatRoundedRect& a, const FloatRoundedRect& b)
168{
169 return !(a == b);
170}

172inline float calcBorderRadiiConstraintScaleFor(const FloatRect& rect, const FloatRoundedRect::Radii& radii)
173{
174 // Constrain corner radii using CSS3 rules:
175 // http://www.w3.org/TR/css3-background/#the-border-radius
176
177 float factor = 1;
178 float radiiSum;
179
180 // top
181 radiiSum = radii.topLeft().width() + radii.topRight().width(); // Casts to avoid integer overflow.
182 if (radiiSum > rect.width())
183 factor = std::min(rect.width() / radiiSum, factor);
184
185 // bottom
186 radiiSum = radii.bottomLeft().width() + radii.bottomRight().width();
187 if (radiiSum > rect.width())
188 factor = std::min(rect.width() / radiiSum, factor);
189
190 // left
191 radiiSum = radii.topLeft().height() + radii.bottomLeft().height();
192 if (radiiSum > rect.height())
193 factor = std::min(rect.height() / radiiSum, factor);
194
195 // right
196 radiiSum = radii.topRight().height() + radii.bottomRight().height();
197 if (radiiSum > rect.height())
198 factor = std::min(rect.height() / radiiSum, factor);
199
200 ASSERT(factor <= 1);
201 return factor;
202}
