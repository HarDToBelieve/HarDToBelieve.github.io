26#ifndef FloatSizeHash_h
27#define FloatSizeHash_h
28
29#include "FloatSize.h"
30#include <wtf/HashSet.h>
31
32namespace WTF {
33
34template<> struct FloatHash<WebCore::FloatSize> {
35 static unsigned hash(const WebCore::FloatSize& key) { return pairIntHash(DefaultHash<float>::hash(key.width()), DefaultHash<float>::hash(key.height())); }
36 static bool equal(const WebCore::FloatSize& a, const WebCore::FloatSize& b) { return a == b; }
37 static const bool safeToCompareToEmptyOrDeleted = true;
38};
39
40template<> struct DefaultHash<WebCore::FloatSize> : FloatHash<WebCore::FloatSize> { };
41
42template<> struct HashTraits<WebCore::FloatSize> : GenericHashTraits<WebCore::FloatSize> {
43 static const bool emptyValueIsZero = true;
44 static void constructDeletedValue(WebCore::FloatSize& slot) { new (NotNull, &slot) WebCore::FloatSize(-1, -1); }
45 static bool isDeletedValue(const WebCore::FloatSize& value) { return value.width() == -1 && value.height() == -1; }
46};
47
48} // namespace WTF
