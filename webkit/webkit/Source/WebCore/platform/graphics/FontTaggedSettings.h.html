40using FontTag = std::array<char, 4>;
41
42inline FontTag fontFeatureTag(const char characters[4]) { return {{ characters[0], characters[1], characters[2], characters[3] }}; }
43
44inline void add(Hasher& hasher, std::array<char, 4> array)
45{
46 uint32_t integer = (static_cast<uint8_t>(array[0]) << 24) | (static_cast<uint8_t>(array[1]) << 16) | (static_cast<uint8_t>(array[2]) << 8) | static_cast<uint8_t>(array[3]);
47 add(hasher, integer);
48}

50struct FourCharacterTagHash {
51 static unsigned hash(FontTag characters) { return computeHash(characters); }
52 static bool equal(FontTag a, FontTag b) { return a == b; }
53 static const bool safeToCompareToEmptyOrDeleted = true;
54};

56struct FourCharacterTagHashTraits : HashTraits<FontTag> {
57 static const bool emptyValueIsZero = true;
58 static void constructDeletedValue(FontTag& slot) { new (NotNull, std::addressof(slot)) FontTag({{ ff, ff, ff, ff }}); }
59 static bool isDeletedValue(FontTag value) { return value == FontTag({{ ff, ff, ff, ff }}); }

61private:
62 static constexpr char ff = static_cast<char>(0xFF);
63};

65template <typename T>
66class FontTaggedSetting {
67public:
68 FontTaggedSetting() = delete;
69 FontTaggedSetting(FontTag, T value);
70
71 bool operator==(const FontTaggedSetting<T>& other) const;
72 bool operator!=(const FontTaggedSetting<T>& other) const { return !(*this == other); }
73 bool operator<(const FontTaggedSetting<T>& other) const;
74
75 FontTag tag() const { return m_tag; }
76 T value() const { return m_value; }
77 bool enabled() const { return value(); }
78
79 template<class Encoder> void encode(Encoder&) const;
80 template<class Decoder> static std::optional<FontTaggedSetting<T>> decode(Decoder&);
81
82private:
83 FontTag m_tag;
84 T m_value;
85};
86
87template <typename T>
88FontTaggedSetting<T>::FontTaggedSetting(FontTag tag, T value)
89 : m_tag(tag)
90 , m_value(value)
91{
92}
93
94template <typename T>
95bool FontTaggedSetting<T>::operator==(const FontTaggedSetting<T>& other) const
96{
97 return m_tag == other.m_tag && m_value == other.m_value;
98}
99
100template <typename T>
101bool FontTaggedSetting<T>::operator<(const FontTaggedSetting<T>& other) const
102{
103 return (m_tag < other.m_tag) || (m_tag == other.m_tag && m_value < other.m_value);
104}
105
106template <typename T>
107template <class Encoder>
108void FontTaggedSetting<T>::encode(Encoder& encoder) const
109{
110 encoder << static_cast<uint8_t>(m_tag[0]);
111 encoder << static_cast<uint8_t>(m_tag[1]);
112 encoder << static_cast<uint8_t>(m_tag[2]);
113 encoder << static_cast<uint8_t>(m_tag[3]);
114 encoder << m_value;
115}

117template <typename T>
118template <class Decoder>
119std::optional<FontTaggedSetting<T>> FontTaggedSetting<T>::decode(Decoder& decoder)
120{
121 std::optional<uint8_t> char0;
122 decoder >> char0;
123 if (!char0)
124 return std::nullopt;
125
126 std::optional<uint8_t> char1;
127 decoder >> char1;
128 if (!char1)
129 return std::nullopt;
130
131 std::optional<uint8_t> char2;
132 decoder >> char2;
133 if (!char2)
134 return std::nullopt;
135
136 std::optional<uint8_t> char3;
137 decoder >> char3;
138 if (!char3)
139 return std::nullopt;
140
141 std::optional<T> value;
142 decoder >> value;
143 if (!value)
144 return std::nullopt;
145
146 return FontTaggedSetting<T>({{
147 static_cast<char>(*char0),
148 static_cast<char>(*char1),
149 static_cast<char>(*char2),
150 static_cast<char>(*char3)
151 }}, *value);
152}

154template<typename T> void add(Hasher& hasher, const FontTaggedSetting<T>& setting)
155{
156 add(hasher, setting.tag(), setting.value());
157}

159template <typename T>
160class FontTaggedSettings {
161public:
162 void insert(FontTaggedSetting<T>&&);
163 bool operator==(const FontTaggedSettings<T>& other) const { return m_list == other.m_list; }
164 bool operator!=(const FontTaggedSettings<T>& other) const { return !(*this == other); }
165
166 bool isEmpty() const { return !size(); }
167 size_t size() const { return m_list.size(); }
168 const FontTaggedSetting<T>& operator[](int index) const { return m_list[index]; }
169 const FontTaggedSetting<T>& at(size_t index) const { return m_list.at(index); }
170
171 typename Vector<FontTaggedSetting<T>>::const_iterator begin() const { return m_list.begin(); }
172 typename Vector<FontTaggedSetting<T>>::const_iterator end() const { return m_list.end(); }
173
174 unsigned hash() const;
175
176 template<class Encoder> void encode(Encoder&) const;
177 template<class Decoder> static std::optional<FontTaggedSettings<T>> decode(Decoder&);
178
179private:
180 Vector<FontTaggedSetting<T>> m_list;
181};

183template <typename T>
184void FontTaggedSettings<T>::insert(FontTaggedSetting<T>&& feature)
185{
186 // This vector will almost always have 0 or 1 items in it. Don't bother with the overhead of a binary search or a hash set.
187 size_t i;
188 for (i = 0; i < m_list.size(); ++i) {
189 if (!(feature < m_list[i]))
190 break;
191 }
192 if (i < m_list.size() && feature.tag() == m_list[i].tag())
193 m_list.remove(i);
194 m_list.insert(i, WTFMove(feature));
195}
196
197template <typename T>
198template <class Encoder>
199void FontTaggedSettings<T>::encode(Encoder& encoder) const
200{
201 encoder << m_list;
202}
203
204template <typename T>
205template <class Decoder>
206std::optional<FontTaggedSettings<T>> FontTaggedSettings<T>::decode(Decoder& decoder)
207{
208 std::optional<Vector<FontTaggedSetting<T>>> list;
209 decoder >> list;
210 if (!list)
211 return std::nullopt;
212
213 FontTaggedSettings result;
214 result.m_list = WTFMove(*list);
215 return result;
216}
