31std::optional<FourCC> FourCC::fromString(const String& stringValue)
32{
33 auto asciiValue = stringValue.ascii();
34 if (asciiValue.length() != 4)
35 return std::nullopt;
36
37 const char* data = asciiValue.data();
38 ASSERT(asciiValue.data());
39 uint32_t value = data[0] << 24 | data[1] << 16 | data[2] << 8 | data[3];
40 return FourCC(value);
41}
42
43String FourCC::toString() const
44{
45 Vector<LChar, 4> data = {
46 LChar(value >> 24),
47 LChar((value >> 16) & 0xFF),
48 LChar((value >> 8) & 0xFF),
49 LChar(value & 0xFF),
50 };
51
52 return String::adopt(WTFMove(data));
53}
