35float euclidianDistance(const FloatPoint& p1, const FloatPoint& p2)
36{
37 FloatSize delta = p1 - p2;
38 return std::hypot(delta.width(), delta.height());
39}

41float findSlope(const FloatPoint& p1, const FloatPoint& p2, float& c)
42{
43 if (p2.x() == p1.x())
44 return std::numeric_limits<float>::infinity();
45
46 // y = mx + c
47 float slope = (p2.y() - p1.y()) / (p2.x() - p1.x());
48 c = p1.y() - slope * p1.x();
49 return slope;
50}

52bool findIntersection(const FloatPoint& p1, const FloatPoint& p2, const FloatPoint& d1, const FloatPoint& d2, FloatPoint& intersection)
53{
54 float pOffset = 0;
55 float pSlope = findSlope(p1, p2, pOffset);
56
57 float dOffset = 0;
58 float dSlope = findSlope(d1, d2, dOffset);
59
60 if (dSlope == pSlope)
61 return false;
62
63 if (pSlope == std::numeric_limits<float>::infinity()) {
64 intersection.setX(p1.x());
65 intersection.setY(dSlope * intersection.x() + dOffset);
66 return true;
67 }
68 if (dSlope == std::numeric_limits<float>::infinity()) {
69 intersection.setX(d1.x());
70 intersection.setY(pSlope * intersection.x() + pOffset);
71 return true;
72 }
73
74 // Find x at intersection, where ys overlap; x = (c' - c) / (m - m')
75 intersection.setX((dOffset - pOffset) / (pSlope - dSlope));
76 intersection.setY(pSlope * intersection.x() + pOffset);
77 return true;
78}

104FloatPoint mapPoint(FloatPoint p, const FloatRect& srcRect, const FloatRect& destRect)
105{
106 if (!srcRect.width() || !srcRect.height())
107 return p;
108
109 float widthScale = destRect.width() / srcRect.width();
110 float heightScale = destRect.height() / srcRect.height();
111
112 return {
113 destRect.x() + (p.x() - srcRect.x()) * widthScale,
114 destRect.y() + (p.y() - srcRect.y()) * heightScale
115 };
116}

118FloatRect mapRect(const FloatRect& r, const FloatRect& srcRect, const FloatRect& destRect)
119{
120 if (!srcRect.width() || !srcRect.height())
121 return FloatRect();
122
123 float widthScale = destRect.width() / srcRect.width();
124 float heightScale = destRect.height() / srcRect.height();
125 return {
126 destRect.x() + (r.x() - srcRect.x()) * widthScale,
127 destRect.y() + (r.y() - srcRect.y()) * heightScale,
128 r.width() * widthScale,
129 r.height() * heightScale
130 };
131}

133FloatRect largestRectWithAspectRatioInsideRect(float aspectRatio, const FloatRect& srcRect)
134{
135 FloatRect destRect = srcRect;
136
137 if (aspectRatio > srcRect.size().aspectRatio()) {
138 float dy = destRect.width() / aspectRatio - destRect.height();
139 destRect.inflateY(dy / 2);
140 } else {
141 float dx = destRect.height() * aspectRatio - destRect.width();
142 destRect.inflateX(dx / 2);
143 }
144 return destRect;
145}

147FloatRect boundsOfRotatingRect(const FloatRect& r)
148{
149 // Compute the furthest corner from the origin.
150 float maxCornerDistance = euclidianDistance(FloatPoint(), r.minXMinYCorner());
151 maxCornerDistance = std::max(maxCornerDistance, euclidianDistance(FloatPoint(), r.maxXMinYCorner()));
152 maxCornerDistance = std::max(maxCornerDistance, euclidianDistance(FloatPoint(), r.minXMaxYCorner()));
153 maxCornerDistance = std::max(maxCornerDistance, euclidianDistance(FloatPoint(), r.maxXMaxYCorner()));
154
155 return FloatRect(-maxCornerDistance, -maxCornerDistance, 2 * maxCornerDistance, 2 * maxCornerDistance);
156}

178bool ellipseContainsPoint(const FloatPoint& center, const FloatSize& radii, const FloatPoint& point)
179{
180 if (radii.width() <= 0 || radii.height() <= 0)
181 return false;
182
183 // First, offset the query point so that the ellipse is effectively centered at the origin.
184 FloatPoint transformedPoint(point);
185 transformedPoint.move(-center.x(), -center.y());
186
187 // If the point lies outside of the bounding box determined by the radii of the ellipse, it can't possibly
188 // be contained within the ellipse, so bail early.
189 if (transformedPoint.x() < -radii.width() || transformedPoint.x() > radii.width() || transformedPoint.y() < -radii.height() || transformedPoint.y() > radii.height())
190 return false;
191
192 // Let (x, y) represent the translated point, and let (Rx, Ry) represent the radii of an ellipse centered at the origin.
193 // (x, y) is contained within the ellipse if, after scaling the ellipse to be a unit circle, the identically scaled
194 // point lies within that unit circle. In other words, the squared distance (x/Rx)^2 + (y/Ry)^2 of the transformed point
195 // to the origin is no greater than 1. This is equivalent to checking whether or not the point (xRy, yRx) lies within a
196 // circle of radius RxRy.
197 transformedPoint.scale(radii.height(), radii.width());
198 auto transformedRadius = radii.width() * radii.height();
199
200 // We can bail early if |xRy| + |yRx| <= RxRy to avoid additional multiplications, since that means the Manhattan distance
201 // of the transformed point is less than the radius, so the point must lie within the transformed circle.
202 return std::abs(transformedPoint.x()) + std::abs(transformedPoint.y()) <= transformedRadius || transformedPoint.lengthSquared() <= transformedRadius * transformedRadius;
203}

205FloatPoint midPoint(const FloatPoint& first, const FloatPoint& second)
206{
207 return { (first.x() + second.x()) / 2, (first.y() + second.y()) / 2 };

210static float dotProduct(const FloatSize& u, const FloatSize& v)
211{
212 return u.width() * v.width() + u.height() * v.height();
213}

215static float angleBetweenVectors(const FloatSize& u, const FloatSize& v)
216{
217 auto magnitudes = u.diagonalLength() * v.diagonalLength();
218 return magnitudes ? acos(clampTo<float>(dotProduct(u, v) / magnitudes, -1, 1)) : 0;
219}

221RotatedRect rotatedBoundingRectWithMinimumAngleOfRotation(const FloatQuad& quad, std::optional<float> minRotationInRadians)
222{
223 constexpr auto twoPiFloat = 2 * piFloat;
224
225 auto minRotationAmount = minRotationInRadians.value_or(std::numeric_limits<float>::epsilon());
226
227 auto leftMidPoint = midPoint(quad.p1(), quad.p4());
228 auto rightMidPoint = midPoint(quad.p2(), quad.p3());
229 auto widthVector = rightMidPoint - leftMidPoint;
230
231 auto midPointToMidPointDistance = widthVector.diagonalLength();
232 int signOfWidthVectorHeight = widthVector.height() < 0 ? -1 : 1;
233 auto angle = midPointToMidPointDistance ? signOfWidthVectorHeight * acos(widthVector.width() / midPointToMidPointDistance) : 0;
234 if (angle < 0)
235 angle += twoPiFloat;
236
237 if (std::abs(angle) < minRotationAmount || std::abs(twoPiFloat - angle) < minRotationAmount) {
238 auto boundingBox = quad.boundingBox();
239 return { boundingBox.center(), boundingBox.size(), 0 };
240 }
241
242 auto heightVector = FloatSize { widthVector.height(), -widthVector.width() };
243 auto leftPerpendicularAngle = angleBetweenVectors(heightVector, quad.p1() - leftMidPoint);
244 auto rightPerpendicularAngle = angleBetweenVectors(heightVector, quad.p2() - rightMidPoint);
245
246 auto leftHypotenuseLength = (leftMidPoint - quad.p1()).diagonalLength();
247 auto rightHypotenuseLength = (rightMidPoint - quad.p2()).diagonalLength();
248
249 auto leftMargin = leftHypotenuseLength * sin(leftPerpendicularAngle);
250 auto rightMargin = rightHypotenuseLength * sin(rightPerpendicularAngle);
251 auto width = midPointToMidPointDistance + leftMargin + rightMargin;
252 auto height = 2 * std::max(leftHypotenuseLength * cos(leftPerpendicularAngle), rightHypotenuseLength * cos(rightPerpendicularAngle));
253
254 auto leftMidToCenterDistance = (midPointToMidPointDistance + rightMargin - leftMargin) / 2;
255 auto center = leftMidPoint + (widthVector * leftMidToCenterDistance / midPointToMidPointDistance);
256 return { center, { width, height }, angle };
257}
