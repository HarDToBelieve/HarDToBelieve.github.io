165template<typename Encoder> void Gradient::ColorStop::encode(Encoder& encoder) const
166{
167 encoder << offset;
168 encoder << color;
169}
170
171template<typename Decoder> std::optional<Gradient::ColorStop> Gradient::ColorStop::decode(Decoder& decoder)
172{
173 std::optional<float> offset;
174 decoder >> offset;
175 if (!offset)
176 return std::nullopt;
177
178 std::optional<Color> color;
179 decoder >> color;
180 if (!color)
181 return std::nullopt;
182
183 return {{ *offset, *color }};
184}

267template<typename Encoder> void Gradient::encode(Encoder& encoder) const
268{
269 encoder << m_data;
270 encoder << m_stops;
271 encoder << m_stopsSorted;
272 encoder << m_spreadMethod;
273}
274
275template<typename Decoder> std::optional<Ref<Gradient>> Gradient::decode(Decoder& decoder)
276{
277 std::optional<Data> data;
278 decoder >> data;
279 if (!data)
280 return std::nullopt;
281 auto gradient = Gradient::create(WTFMove(*data));
282
283 std::optional<ColorStopVector> stops;
284 decoder >> stops;
285 if (!stops)
286 return std::nullopt;
287 std::optional<bool> stopsSorted;
288 decoder >> stopsSorted;
289 if (!stopsSorted.has_value())
290 return std::nullopt;
291 if (*stopsSorted)
292 gradient->setSortedColorStops(WTFMove(*stops));
293 else {
294 for (auto& stop : *stops)
295 gradient->addColorStop(WTFMove(stop));
296 }
297
298 GradientSpreadMethod spreadMethod;
299 if (!decoder.decode(spreadMethod))
300 return std::nullopt;
301 gradient->setSpreadMethod(spreadMethod);
302
303 return gradient;
304}
