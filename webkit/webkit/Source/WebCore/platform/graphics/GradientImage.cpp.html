43ImageDrawResult GradientImage::draw(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const ImagePaintingOptions& options)
44{
45 GraphicsContextStateSaver stateSaver(destContext);
46 destContext.setCompositeOperation(options.compositeOperator(), options.blendMode());
47 destContext.clip(destRect);
48 destContext.translate(destRect.location());
49 if (destRect.size() != srcRect.size())
50 destContext.scale(destRect.size() / srcRect.size());
51 destContext.translate(-srcRect.location());
52 destContext.fillRect(FloatRect(FloatPoint(), size()), m_gradient.get());
53 return ImageDrawResult::DidDraw;
54}

56void GradientImage::drawPattern(GraphicsContext& destContext, const FloatRect& destRect, const FloatRect& srcRect, const AffineTransform& patternTransform,
57 const FloatPoint& phase, const FloatSize& spacing, const ImagePaintingOptions& options)
58{
59 // Allow the generator to provide visually-equivalent tiling parameters for better performance.
60 FloatSize adjustedSize = size();
61 FloatRect adjustedSrcRect = srcRect;
62 m_gradient->adjustParametersForTiledDrawing(adjustedSize, adjustedSrcRect, spacing);
63
64 // Factor in the destination context's scale to generate at the best resolution
65 AffineTransform destContextCTM = destContext.getCTM(GraphicsContext::DefinitelyIncludeDeviceScale);
66 double xScale = fabs(destContextCTM.xScale());
67 double yScale = fabs(destContextCTM.yScale());
68 AffineTransform adjustedPatternCTM = patternTransform;
69 adjustedPatternCTM.scale(1.0 / xScale, 1.0 / yScale);
70 adjustedSrcRect.scale(xScale, yScale);
71
72 unsigned generatorHash = m_gradient->hash();
73
74 if (!m_cachedImage || m_cachedGeneratorHash != generatorHash || m_cachedAdjustedSize != adjustedSize || !areEssentiallyEqual(destContext.scaleFactor(), m_cachedScaleFactor)) {
75 auto imageBuffer = ImageBuffer::createCompatibleBuffer(adjustedSize, DestinationColorSpace::SRGB(), destContext);
76 if (!imageBuffer)
77 return;
78
79 // Fill with the generated image.
80 imageBuffer->context().fillRect(FloatRect(FloatPoint(), adjustedSize), m_gradient.get());
81
82 m_cachedGeneratorHash = generatorHash;
83 m_cachedAdjustedSize = adjustedSize;
84 m_cachedScaleFactor = destContext.scaleFactor();
85
86 if (destContext.drawLuminanceMask())
87 imageBuffer->convertToLuminanceMask();
88
89 m_cachedImage = ImageBuffer::sinkIntoImage(WTFMove(imageBuffer), PreserveResolution::Yes);
90 if (!m_cachedImage)
91 return;
92 }
93
94 destContext.setDrawLuminanceMask(false);
95
96 // Tile the image buffer into the context.
97 m_cachedImage->drawPattern(destContext, destRect, adjustedSrcRect, adjustedPatternCTM, phase, spacing, options);
98
99}
