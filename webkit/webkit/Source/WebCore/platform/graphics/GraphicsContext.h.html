129struct DocumentMarkerLineStyle {
130 enum class Mode : uint8_t {
131 TextCheckingDictationPhraseWithAlternatives,
132 Spelling,
133 Grammar,
134 AutocorrectionReplacement,
135 DictationAlternatives
136 } mode;
137 bool shouldUseDarkAppearance { false };
138
139 template<class Encoder> void encode(Encoder&) const;
140 template<class Decoder> static std::optional<DocumentMarkerLineStyle> decode(Decoder&);
141};
142
143template<class Encoder>
144void DocumentMarkerLineStyle::encode(Encoder& encoder) const
145{
146 encoder << mode;
147 encoder << shouldUseDarkAppearance;
148}
149
150template<class Decoder>
151std::optional<DocumentMarkerLineStyle> DocumentMarkerLineStyle::decode(Decoder& decoder)
152{
153 std::optional<Mode> mode;
154 decoder >> mode;
155 if (!mode)
156 return std::nullopt;
157
158 std::optional<bool> shouldUseDarkAppearance;
159 decoder >> shouldUseDarkAppearance;
160 if (!shouldUseDarkAppearance)
161 return std::nullopt;
162
163 return {{ *mode, *shouldUseDarkAppearance }};
164}

585class GraphicsContextStateSaver {
586 WTF_MAKE_FAST_ALLOCATED;
587public:
588 GraphicsContextStateSaver(GraphicsContext& context, bool saveAndRestore = true)
589 : m_context(context)
590 , m_saveAndRestore(saveAndRestore)
591 {
592 if (m_saveAndRestore)
593 m_context.save();
594 }
595
596 ~GraphicsContextStateSaver()
597 {
598 if (m_saveAndRestore)
599 m_context.restore();
600 }
601
602 void save()
603 {
