341// Helper for packImageData/extractImageData/extractTextureData which implement packing of pixel
342// data into the specified OpenGL destination format and type.
343// A sourceUnpackAlignment of zero indicates that the source
344// data is tightly packed. Non-zero values may take a slow path.
345// Destination data will have no gaps between rows.
346static bool packPixels(const uint8_t* sourceData, GraphicsContextGL::DataFormat sourceDataFormat, unsigned sourceDataWidth, unsigned sourceDataHeight, const IntRect& sourceDataSubRectangle, int depth, unsigned sourceUnpackAlignment, int unpackImageHeight, unsigned destinationFormat, unsigned destinationType, GraphicsContextGL::AlphaOp alphaOp, void* destinationData, bool flipY)
347{
348 ASSERT(depth >= 1);
349 UNUSED_PARAM(sourceDataHeight); // Derived from sourceDataSubRectangle.height().
350 if (!unpackImageHeight)
351 unpackImageHeight = sourceDataSubRectangle.height();
352 int validSrc = sourceDataWidth * texelBytesForFormat(sourceDataFormat);
353 int remainder = sourceUnpackAlignment ? (validSrc % sourceUnpackAlignment) : 0;
354 int srcStride = remainder ? (validSrc + sourceUnpackAlignment - remainder) : validSrc;
355 int srcRowOffset = sourceDataSubRectangle.x() * texelBytesForFormat(sourceDataFormat);
356
357 GraphicsContextGL::DataFormat dstDataFormat = getDataFormat(destinationFormat, destinationType);
358 int dstStride = sourceDataSubRectangle.width() * texelBytesForFormat(dstDataFormat);
359 if (flipY) {
360 destinationData = static_cast<uint8_t*>(destinationData) + dstStride * ((depth * sourceDataSubRectangle.height()) - 1);
361 dstStride = -dstStride;
362 }
363 if (!GraphicsContextGL::hasAlpha(sourceDataFormat) || !GraphicsContextGL::hasColor(sourceDataFormat) || !GraphicsContextGL::hasColor(dstDataFormat))
364 alphaOp = GraphicsContextGL::AlphaOp::DoNothing;
365
366 if (sourceDataFormat == dstDataFormat && alphaOp == GraphicsContextGL::AlphaOp::DoNothing) {
367 const uint8_t* basePtr = sourceData + srcStride * sourceDataSubRectangle.y();
368 const uint8_t* baseEnd = sourceData + srcStride * sourceDataSubRectangle.maxY();
369
370 // If packing multiple images into a 3D texture, and flipY is true,
371 // then the sub-rectangle is pointing at the start of the
372 // "bottommost" of those images. Since the source pointer strides in
373 // the positive direction, we need to back it up to point at the
374 // last, or "topmost", of these images.
375 if (flipY && depth > 1) {
376 const ptrdiff_t distanceToTopImage = (depth - 1) * srcStride * unpackImageHeight;
377 basePtr -= distanceToTopImage;
378 baseEnd -= distanceToTopImage;
379 }
380
381 unsigned rowSize = (dstStride > 0) ? dstStride: -dstStride;
382 uint8_t* dst = static_cast<uint8_t*>(destinationData);
383
384 for (int i = 0; i < depth; ++i) {
385 const uint8_t* ptr = basePtr;
386 const uint8_t* ptrEnd = baseEnd;
387 while (ptr < ptrEnd) {
388 memcpy(dst, ptr + srcRowOffset, rowSize);
389 ptr += srcStride;
390 dst += dstStride;
391 }
392 basePtr += unpackImageHeight * srcStride;
393 baseEnd += unpackImageHeight * srcStride;
394 }
395 return true;
396 }
397
398 FormatConverter converter(
399 sourceDataSubRectangle, depth,
400 unpackImageHeight, sourceData, destinationData,
401 srcStride, srcRowOffset, dstStride);
402 converter.convert(sourceDataFormat, dstDataFormat, alphaOp);
403 if (!converter.success())
404 return false;
405 return true;
406}

567bool GraphicsContextGL::computeFormatAndTypeParameters(GCGLenum format, GCGLenum type, unsigned* componentsPerPixel, unsigned* bytesPerComponent)
568{
569 switch (format) {
570 case GraphicsContextGL::ALPHA:
571 case GraphicsContextGL::LUMINANCE:
572 case GraphicsContextGL::RED:
573 case GraphicsContextGL::RED_INTEGER:
574 case GraphicsContextGL::DEPTH_COMPONENT:
575 case GraphicsContextGL::DEPTH_STENCIL: // Treat it as one component.
576 *componentsPerPixel = 1;
577 break;
578 case GraphicsContextGL::LUMINANCE_ALPHA:
579 case GraphicsContextGL::RG:
580 case GraphicsContextGL::RG_INTEGER:
581 *componentsPerPixel = 2;
582 break;
583 case GraphicsContextGL::RGB:
584 case GraphicsContextGL::RGB_INTEGER:
585 case ExtensionsGL::SRGB_EXT:
586 *componentsPerPixel = 3;
587 break;
588 case GraphicsContextGL::RGBA:
589 case GraphicsContextGL::RGBA_INTEGER:
590 case ExtensionsGL::BGRA_EXT: // GL_EXT_texture_format_BGRA8888
591 case ExtensionsGL::SRGB_ALPHA_EXT:
592 *componentsPerPixel = 4;
593 break;
594 default:
595 return false;
596 }
597
598 switch (type) {
599 case GraphicsContextGL::BYTE:
600 *bytesPerComponent = sizeof(GCGLbyte);
601 break;
602 case GraphicsContextGL::UNSIGNED_BYTE:
603 *bytesPerComponent = sizeof(GCGLubyte);
604 break;
605 case GraphicsContextGL::SHORT:
606 *bytesPerComponent = sizeof(GCGLshort);
607 break;
608 case GraphicsContextGL::UNSIGNED_SHORT:
609 *bytesPerComponent = sizeof(GCGLushort);
610 break;
611 case GraphicsContextGL::UNSIGNED_SHORT_5_6_5:
612 case GraphicsContextGL::UNSIGNED_SHORT_4_4_4_4:
613 case GraphicsContextGL::UNSIGNED_SHORT_5_5_5_1:
614 *componentsPerPixel = 1;
615 *bytesPerComponent = sizeof(GCGLushort);
616 break;
617 case GraphicsContextGL::INT:
618 *bytesPerComponent = sizeof(GCGLint);
619 break;
620 case GraphicsContextGL::UNSIGNED_INT:
621 *bytesPerComponent = sizeof(GCGLuint);
622 break;
623 case GraphicsContextGL::UNSIGNED_INT_24_8:
624 case GraphicsContextGL::UNSIGNED_INT_10F_11F_11F_REV:
625 case GraphicsContextGL::UNSIGNED_INT_5_9_9_9_REV:
626 case GraphicsContextGL::UNSIGNED_INT_2_10_10_10_REV:
627 *componentsPerPixel = 1;
628 *bytesPerComponent = sizeof(GCGLuint);
629 break;
630 case GraphicsContextGL::FLOAT: // OES_texture_float
631 *bytesPerComponent = sizeof(GCGLfloat);
632 break;
633 case GraphicsContextGL::HALF_FLOAT:
634 case GraphicsContextGL::HALF_FLOAT_OES: // OES_texture_half_float
635 *bytesPerComponent = sizeof(GCGLhalffloat);
636 break;
637 case GraphicsContextGL::FLOAT_32_UNSIGNED_INT_24_8_REV:
638 *bytesPerComponent = sizeof(GCGLfloat) + sizeof(GCGLuint);
639 break;
640 default:
641 return false;
642 }
643 return true;
644}

646GCGLenum GraphicsContextGL::computeImageSizeInBytes(GCGLenum format, GCGLenum type, GCGLsizei width, GCGLsizei height, GCGLsizei depth, const PixelStoreParams& params, unsigned* imageSizeInBytes, unsigned* paddingInBytes, unsigned* skipSizeInBytes)
647{
648 ASSERT(imageSizeInBytes);
649 ASSERT(params.alignment == 1 || params.alignment == 2 || params.alignment == 4 || params.alignment == 8);
650 ASSERT(params.rowLength >= 0);
651 ASSERT(params.imageHeight >= 0);
652 ASSERT(params.skipPixels >= 0);
653 ASSERT(params.skipRows >= 0);
654 ASSERT(params.skipImages >= 0);
655 if (width < 0 || height < 0 || depth < 0)
656 return GraphicsContextGL::INVALID_VALUE;
657 if (!width || !height || !depth) {
658 *imageSizeInBytes = 0;
659 if (paddingInBytes)
660 *paddingInBytes = 0;
661 if (skipSizeInBytes)
662 *skipSizeInBytes = 0;
663 return GraphicsContextGL::NO_ERROR;
664 }
665
666 int rowLength = params.rowLength > 0 ? params.rowLength : width;
667 int imageHeight = params.imageHeight > 0 ? params.imageHeight : height;
668
669 unsigned bytesPerComponent, componentsPerPixel;
670 if (!computeFormatAndTypeParameters(format, type, &componentsPerPixel, &bytesPerComponent))
671 return GraphicsContextGL::INVALID_ENUM;
672 unsigned bytesPerGroup = bytesPerComponent * componentsPerPixel;
673 CheckedUint32 checkedValue = static_cast<uint32_t>(rowLength);
674 checkedValue *= bytesPerGroup;
675 if (checkedValue.hasOverflowed())
676 return GraphicsContextGL::INVALID_VALUE;
677
678 unsigned lastRowSize;
679 if (params.rowLength > 0 && params.rowLength != width) {
680 CheckedUint32 tmp = width;
681 tmp *= bytesPerGroup;
682 if (tmp.hasOverflowed())
683 return GraphicsContextGL::INVALID_VALUE;
684 lastRowSize = tmp;
685 } else
686 lastRowSize = checkedValue;
687
688 unsigned padding = 0;
689 unsigned residual = checkedValue.value() % params.alignment;
690 if (residual) {
691 padding = params.alignment - residual;
692 checkedValue += padding;
693 }
694 if (checkedValue.hasOverflowed())
695 return GraphicsContextGL::INVALID_VALUE;
696 unsigned paddedRowSize = checkedValue;
697
698 CheckedUint32 rows = imageHeight;
699 rows *= (depth - 1);
700 // Last image is not affected by IMAGE_HEIGHT parameter.
701 rows += height;
702 if (rows.hasOverflowed())
703 return GraphicsContextGL::INVALID_VALUE;
704 checkedValue *= (rows - 1);
705 // Last row is not affected by ROW_LENGTH parameter.
706 checkedValue += lastRowSize;
707 if (checkedValue.hasOverflowed())
708 return GraphicsContextGL::INVALID_VALUE;
709 *imageSizeInBytes = checkedValue;
710 if (paddingInBytes)
711 *paddingInBytes = padding;
712
713 CheckedUint32 skipSize = 0;
714 if (params.skipImages > 0) {
715 CheckedUint32 tmp = paddedRowSize;
716 tmp *= imageHeight;
717 tmp *= params.skipImages;
718 if (tmp.hasOverflowed())
719 return GraphicsContextGL::INVALID_VALUE;
720 skipSize += tmp;
721 }
722 if (params.skipRows > 0) {
723 CheckedUint32 tmp = paddedRowSize;
724 tmp *= params.skipRows;
725 if (tmp.hasOverflowed())
726 return GraphicsContextGL::INVALID_VALUE;
727 skipSize += tmp;
728 }
729 if (params.skipPixels > 0) {
730 CheckedUint32 tmp = bytesPerGroup;
731 tmp *= params.skipPixels;
732 if (tmp.hasOverflowed())
733 return GraphicsContextGL::INVALID_VALUE;
734 skipSize += tmp;
735 }
736 if (skipSize.hasOverflowed())
737 return GraphicsContextGL::INVALID_VALUE;
738 if (skipSizeInBytes)
739 *skipSizeInBytes = skipSize;
740
741 checkedValue += skipSize;
742 if (checkedValue.hasOverflowed())
743 return GraphicsContextGL::INVALID_VALUE;
744 return GraphicsContextGL::NO_ERROR;
745}

748{

786bool GraphicsContextGL::extractTextureData(unsigned width, unsigned height, GCGLenum format, GCGLenum type, const PixelStoreParams& unpackParams, bool flipY, bool premultiplyAlpha, const void* pixels, Vector<uint8_t>& data)
787{
788 // Assumes format, type, etc. have already been validated.
789 DataFormat sourceDataFormat = getDataFormat(format, type);
790
791 // Resize the output buffer.
792 unsigned componentsPerPixel, bytesPerComponent;
793 if (!computeFormatAndTypeParameters(format, type, &componentsPerPixel, &bytesPerComponent))
794 return false;
795 unsigned bytesPerPixel = componentsPerPixel * bytesPerComponent;
796 data.resize(width * height * bytesPerPixel);
797
798 unsigned imageSizeInBytes, skipSizeInBytes;
799 computeImageSizeInBytes(format, type, width, height, 1, unpackParams, &imageSizeInBytes, nullptr, &skipSizeInBytes);
800 const uint8_t* srcData = static_cast<const uint8_t*>(pixels);
801 if (skipSizeInBytes)
802 srcData += skipSizeInBytes;
803
804 if (!packPixels(srcData, sourceDataFormat, unpackParams.rowLength ? unpackParams.rowLength : width, height, IntRect(0, 0, width, height), 1, unpackParams.alignment, 0, format, type, (premultiplyAlpha ? AlphaOp::DoPremultiply : AlphaOp::DoNothing), data.data(), flipY))
805 return false;

807 return true;
