32namespace WebCore {
33
34GraphicsLayerUpdater::GraphicsLayerUpdater(GraphicsLayerUpdaterClient& client, PlatformDisplayID displayID)
35 : m_client(client)
36{
37 DisplayRefreshMonitorManager::sharedManager().windowScreenDidChange(displayID, *this);
38 scheduleUpdate();
39}
40
41GraphicsLayerUpdater::~GraphicsLayerUpdater()
42{
43 // ~DisplayRefreshMonitorClient unregisters us as a client.
44}
45
46void GraphicsLayerUpdater::scheduleUpdate()
47{
48 if (m_scheduled)
49 return;
50
51 DisplayRefreshMonitorManager::sharedManager().scheduleAnimation(*this);
52 m_scheduled = true;
53}
54
55void GraphicsLayerUpdater::screenDidChange(PlatformDisplayID displayID)
56{
57 DisplayRefreshMonitorManager::sharedManager().windowScreenDidChange(displayID, *this);
58}
59
60void GraphicsLayerUpdater::displayRefreshFired()
61{
62 m_scheduled = false;
63 m_client.flushLayersSoon(*this);
64}
65
66DisplayRefreshMonitorFactory* GraphicsLayerUpdater::displayRefreshMonitorFactory() const
67{
68 return m_client.displayRefreshMonitorFactory();
69}
