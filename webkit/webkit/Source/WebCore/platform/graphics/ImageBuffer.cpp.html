37static const float MaxClampedLength = 4096;
38static const float MaxClampedArea = MaxClampedLength * MaxClampedLength;
39
40RefPtr<ImageBuffer> ImageBuffer::create(const FloatSize& size, RenderingMode renderingMode, ShouldUseDisplayList shouldUseDisplayList, RenderingPurpose purpose, float resolutionScale, const DestinationColorSpace& colorSpace, PixelFormat pixelFormat, const HostWindow* hostWindow)
41{
42 RefPtr<ImageBuffer> imageBuffer;
43
44 // Give ShouldUseDisplayList a higher precedence since it is a debug option.
45 if (shouldUseDisplayList == ShouldUseDisplayList::Yes) {
46 if (renderingMode == RenderingMode::Accelerated)
47 imageBuffer = DisplayListAcceleratedImageBuffer::create(size, resolutionScale, colorSpace, pixelFormat, hostWindow);
48
49 if (!imageBuffer)
50 imageBuffer = DisplayListUnacceleratedImageBuffer::create(size, resolutionScale, colorSpace, pixelFormat, hostWindow);
51 }
52
53 if (hostWindow && !imageBuffer)
54 imageBuffer = hostWindow->createImageBuffer(size, renderingMode, purpose, resolutionScale, colorSpace, pixelFormat);
55
56 if (!imageBuffer)
57 imageBuffer = ImageBuffer::create(size, renderingMode, resolutionScale, colorSpace, pixelFormat, hostWindow);
58
59 return imageBuffer;
60}

62RefPtr<ImageBuffer> ImageBuffer::create(const FloatSize& size, RenderingMode renderingMode, float resolutionScale, const DestinationColorSpace& colorSpace, PixelFormat pixelFormat, const HostWindow* hostWindow)
63{
64 RefPtr<ImageBuffer> imageBuffer;
65
66 if (renderingMode == RenderingMode::Accelerated)
67 imageBuffer = AcceleratedImageBuffer::create(size, resolutionScale, colorSpace, pixelFormat, hostWindow);
68
69 if (!imageBuffer)
70 imageBuffer = UnacceleratedImageBuffer::create(size, resolutionScale, colorSpace, pixelFormat, hostWindow);
71
72 return imageBuffer;
73}

75RefPtr<ImageBuffer> ImageBuffer::createCompatibleBuffer(const FloatSize& size, const GraphicsContext& context)
76{
77 if (size.isEmpty())
78 return nullptr;
79
80 IntSize scaledSize = ImageBuffer::compatibleBufferSize(size, context);
81
82 RefPtr<ImageBuffer> imageBuffer;
83
84 if (context.renderingMode() == RenderingMode::Accelerated)
85 imageBuffer = AcceleratedImageBuffer::create(scaledSize, context);
86
87 if (!imageBuffer)
88 imageBuffer = UnacceleratedImageBuffer::create(scaledSize, context);
89
90 if (!imageBuffer)
91 return nullptr;
92
93 // Set up a corresponding scale factor on the graphics context.
94 imageBuffer->context().scale(scaledSize / size);
95 return imageBuffer;
96}

98RefPtr<ImageBuffer> ImageBuffer::createCompatibleBuffer(const FloatSize& size, const DestinationColorSpace& colorSpace, const GraphicsContext& context)
99{
100 if (size.isEmpty())
101 return nullptr;
102
103 IntSize scaledSize = ImageBuffer::compatibleBufferSize(size, context);
104
105 auto imageBuffer = ImageBuffer::createCompatibleBuffer(scaledSize, 1, colorSpace, context);
106 if (!imageBuffer)
107 return nullptr;
108
109 // Set up a corresponding scale factor on the graphics context.
110 imageBuffer->context().scale(scaledSize / size);
111 return imageBuffer;
112}

119bool ImageBuffer::sizeNeedsClamping(const FloatSize& size)
120{
121 if (size.isEmpty())
122 return false;
123
124 return floorf(size.height()) * floorf(size.width()) > MaxClampedArea;
125}
126
127bool ImageBuffer::sizeNeedsClamping(const FloatSize& size, FloatSize& scale)
128{
129 FloatSize scaledSize(size);
130 scaledSize.scale(scale.width(), scale.height());
131
132 if (!sizeNeedsClamping(scaledSize))
133 return false;
134
135 // The area of scaled size is bigger than the upper limit, adjust the scale to fit.
136 scale.scale(sqrtf(MaxClampedArea / (scaledSize.width() * scaledSize.height())));
137 ASSERT(!sizeNeedsClamping(size, scale));
138 return true;
139}

141FloatSize ImageBuffer::clampedSize(const FloatSize& size)
142{
143 return size.shrunkTo(FloatSize(MaxClampedLength, MaxClampedLength));
144}
145
146FloatSize ImageBuffer::clampedSize(const FloatSize& size, FloatSize& scale)
147{
148 if (size.isEmpty())
149 return size;
150
151 FloatSize clampedSize = ImageBuffer::clampedSize(size);
152 scale = clampedSize / size;
153 ASSERT(!sizeNeedsClamping(clampedSize));
154 ASSERT(!sizeNeedsClamping(size, scale));
155 return clampedSize;
156}

163IntSize ImageBuffer::compatibleBufferSize(const FloatSize& size, const GraphicsContext& context)
164{
165 // Enlarge the buffer size if the context's transform is scaling it so we need a higher
166 // resolution than one pixel per unit.
167 return expandedIntSize(size * context.scaleFactor());
168}

170RefPtr<ImageBuffer> ImageBuffer::copyRectToBuffer(const FloatRect& rect, const DestinationColorSpace& colorSpace, const GraphicsContext& context)
171{
172 if (rect.isEmpty())
173 return nullptr;
174
175 IntSize scaledSize = ImageBuffer::compatibleBufferSize(rect.size(), context);
176
177 auto buffer = ImageBuffer::createCompatibleBuffer(scaledSize, 1, colorSpace, context);
178 if (!buffer)
179 return nullptr;
180
181 buffer->context().drawImageBuffer(*this, -rect.location());
182 return buffer;
183}
