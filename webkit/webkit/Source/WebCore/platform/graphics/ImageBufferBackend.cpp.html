35IntSize ImageBufferBackend::calculateBackendSize(const Parameters& parameters)
36{
37 FloatSize scaledSize = { ceilf(parameters.resolutionScale * parameters.logicalSize.width()), ceilf(parameters.resolutionScale * parameters.logicalSize.height()) };
38 if (scaledSize.isEmpty() || !scaledSize.isExpressibleAsIntSize())
39 return { };
40
41 return IntSize(scaledSize);
42}

72void ImageBufferBackend::convertToLuminanceMask()
73{
74 PixelBufferFormat format { AlphaPremultiplication::Unpremultiplied, PixelFormat::RGBA8, colorSpace() };
75 auto pixelBuffer = getPixelBuffer(format, logicalRect());
76 if (!pixelBuffer)
77 return;
78
79 auto& pixelArray = pixelBuffer->data();
80 unsigned pixelArrayLength = pixelArray.length();
81 for (unsigned pixelOffset = 0; pixelOffset < pixelArrayLength; pixelOffset += 4) {
82 uint8_t a = pixelArray.item(pixelOffset + 3);
83 if (!a)
84 continue;
85 uint8_t r = pixelArray.item(pixelOffset);
86 uint8_t g = pixelArray.item(pixelOffset + 1);
87 uint8_t b = pixelArray.item(pixelOffset + 2);
88
89 double luma = (r * 0.2125 + g * 0.7154 + b * 0.0721) * ((double)a / 255.0);
90 pixelArray.set(pixelOffset + 3, luma);
91 }
92
93 putPixelBuffer(*pixelBuffer, logicalRect(), IntPoint::zero(), AlphaPremultiplication::Premultiplied);
94}

96std::optional<PixelBuffer> ImageBufferBackend::getPixelBuffer(const PixelBufferFormat& destinationFormat, const IntRect& sourceRect, void* data) const
97{
98 ASSERT(PixelBuffer::supportedPixelFormat(destinationFormat.pixelFormat));
99
100 auto sourceRectScaled = toBackendCoordinates(sourceRect);
101
102 auto pixelBuffer = PixelBuffer::tryCreate(destinationFormat, sourceRectScaled.size());
103 if (!pixelBuffer)
104 return std::nullopt;
105
106 auto sourceRectClipped = intersection(backendRect(), sourceRectScaled);
107 IntRect destinationRect { IntPoint::zero(), sourceRectClipped.size() };
108
109 if (sourceRectScaled.x() < 0)
110 destinationRect.setX(-sourceRectScaled.x());
111
112 if (sourceRectScaled.y() < 0)
113 destinationRect.setY(-sourceRectScaled.y());
114
115 if (destinationRect.size() != sourceRectScaled.size())
116 pixelBuffer->data().zeroFill();
117
118 ConstPixelBufferConversionView source {
119 { AlphaPremultiplication::Premultiplied, pixelFormat(), colorSpace() },
120 bytesPerRow(),
121 static_cast<uint8_t*>(data) + sourceRectClipped.y() * source.bytesPerRow + sourceRectClipped.x() * 4
122 };
123
124 PixelBufferConversionView destination {
125 destinationFormat,
126 static_cast<unsigned>(4 * sourceRectScaled.width()),
127 pixelBuffer->data().data() + destinationRect.y() * destination.bytesPerRow + destinationRect.x() * 4
128 };
129
130 convertImagePixels(source, destination, destinationRect.size());
131
132 return pixelBuffer;
133}

135void ImageBufferBackend::putPixelBuffer(const PixelBuffer& sourcePixelBuffer, const IntRect& sourceRect, const IntPoint& destinationPoint, AlphaPremultiplication destinationAlphaFormat, void* data)
136{
137 auto sourceRectScaled = toBackendCoordinates(sourceRect);
138 auto destinationPointScaled = toBackendCoordinates(destinationPoint);
139
140 auto sourceRectClipped = intersection({ IntPoint::zero(), sourcePixelBuffer.size() }, sourceRectScaled);
141 auto destinationRect = sourceRectClipped;
142 destinationRect.moveBy(destinationPointScaled);
143
144 if (sourceRectScaled.x() < 0)
145 destinationRect.setX(destinationRect.x() - sourceRectScaled.x());
146
147 if (sourceRectScaled.y() < 0)
148 destinationRect.setY(destinationRect.y() - sourceRectScaled.y());
149
150 destinationRect.intersect(backendRect());
151 sourceRectClipped.setSize(destinationRect.size());
152
153 ConstPixelBufferConversionView source {
154 sourcePixelBuffer.format(),
155 static_cast<unsigned>(4 * sourcePixelBuffer.size().width()),
156 sourcePixelBuffer.data().data() + sourceRectClipped.y() * source.bytesPerRow + sourceRectClipped.x() * 4
157 };
158
159 PixelBufferConversionView destination {
160 { destinationAlphaFormat, pixelFormat(), colorSpace() },
161 bytesPerRow(),
162 static_cast<uint8_t*>(data) + destinationRect.y() * destination.bytesPerRow + destinationRect.x() * 4
163 };
164
165 convertImagePixels(source, destination, destinationRect.size());
166}
