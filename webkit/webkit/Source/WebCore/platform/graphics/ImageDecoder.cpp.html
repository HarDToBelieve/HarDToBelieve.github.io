49#if ENABLE(GPU_PROCESS) && HAVE(AVASSETREADER)
50using FactoryVector = Vector<ImageDecoder::ImageDecoderFactory>;
51
52static void platformRegisterFactories(FactoryVector& factories)
53{
54 factories.append({ ImageDecoderAVFObjC::supportsMediaType, ImageDecoderAVFObjC::canDecodeType, ImageDecoderAVFObjC::create });
55}
56
57static FactoryVector& installedFactories()
58{
59 static auto factories = makeNeverDestroyed<FactoryVector>({ });
60 static std::once_flag registerDefaults;
61 std::call_once(registerDefaults, [&] {
62 platformRegisterFactories(factories);
63 });
64
65 return factories;
66}
67
68void ImageDecoder::installFactory(ImageDecoder::ImageDecoderFactory&& factory)
69{
70 installedFactories().append(WTFMove(factory));
71}
72
73void ImageDecoder::resetFactories()
74{
75 installedFactories().clear();
76 platformRegisterFactories(installedFactories());
77}
78
79void ImageDecoder::clearFactories()
80{
81 installedFactories().clear();
82}

85RefPtr<ImageDecoder> ImageDecoder::create(SharedBuffer& data, const String& mimeType, AlphaOption alphaOption, GammaAndColorProfileOption gammaAndColorProfileOption)
86{
87 UNUSED_PARAM(mimeType);
88
89#if HAVE(AVASSETREADER)
90 if (!ImageDecoderCG::canDecodeType(mimeType)) {
91#if ENABLE(GPU_PROCESS)
92 for (auto& factory : installedFactories()) {
93 if (factory.canDecodeType(mimeType))
94 return factory.createImageDecoder(data, mimeType, alphaOption, gammaAndColorProfileOption);
95 }
96#else
97 if (ImageDecoderAVFObjC::canDecodeType(mimeType))
98 return ImageDecoderAVFObjC::create(data, mimeType, alphaOption, gammaAndColorProfileOption);
99#endif
100 }
101#endif
102
103#if USE(GSTREAMER) && ENABLE(VIDEO)
104 if (ImageDecoderGStreamer::canDecodeType(mimeType))
105 return ImageDecoderGStreamer::create(data, mimeType, alphaOption, gammaAndColorProfileOption);
106#endif
107
108#if USE(CG)
109 return ImageDecoderCG::create(data, alphaOption, gammaAndColorProfileOption);
110#elif USE(DIRECT2D)
111 return ImageDecoderDirect2D::create(data, alphaOption, gammaAndColorProfileOption);
112#else
113 return ScalableImageDecoder::create(data, alphaOption, gammaAndColorProfileOption);
114#endif
115}
116
117bool ImageDecoder::supportsMediaType(MediaType type)
118{
119#if USE(CG)
120 if (ImageDecoderCG::supportsMediaType(type))
121 return true;
122#elif USE(DIRECT2D)
123 if (ImageDecoderDirect2D::supportsMediaType(type))
124 return true;
125#else
126 if (ScalableImageDecoder::supportsMediaType(type))
127 return true;
128#endif
129
130#if HAVE(AVASSETREADER)
131#if ENABLE(GPU_PROCESS)
132 for (auto& factory : installedFactories()) {
133 if (factory.supportsMediaType(type))
134 return true;
135 }
136#else
137 if (ImageDecoderAVFObjC::supportsMediaType(type))
138 return true;
139#endif
140#endif
141
142#if USE(GSTREAMER) && ENABLE(VIDEO)
143 if (ImageDecoderGStreamer::supportsMediaType(type))
144 return true;
145#endif
146
147 return false;
148}
