31std::optional<IntSize> ImageResolution::densityCorrectedSize(const FloatSize& sourceSize, const ResolutionMetadata& metadata)
32{
33 if (metadata.resolutionUnit != ResolutionUnit::Inches || metadata.resolution.isEmpty() || metadata.preferredSize.isEmpty())
34 return std::nullopt;
35
36 auto sizeFromResolution = sourceSize * DefaultResolution / metadata.resolution;
37
38 if (sizeFromResolution == metadata.preferredSize)
39 return roundedIntSize(metadata.preferredSize);
40
41 return std::nullopt;
42}
