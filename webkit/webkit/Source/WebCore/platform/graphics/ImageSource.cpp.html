52ImageSource::ImageSource(RefPtr<NativeImage>&& nativeImage)
53 : m_runLoop(RunLoop::current())
54{
55 m_frameCount = 1;
56 m_encodedDataStatus = EncodedDataStatus::Complete;
57 m_cachedMetadata.add({ MetadataType::EncodedDataStatus, MetadataType::FrameCount });
58 growFrames();
59
60 setNativeImage(WTFMove(nativeImage));
61
62 m_decodedSize = m_frames[0].frameBytes();
63
64 m_size = m_frames[0].size();
65 m_orientation = ImageOrientation(ImageOrientation::None);
66 m_cachedMetadata.add({ MetadataType::Orientation, MetadataType::Size });
67}

75bool ImageSource::ensureDecoderAvailable(SharedBuffer* data)
76{
77 if (!data || isDecoderAvailable())
78 return true;
79
80 m_decoder = ImageDecoder::create(*data, mimeType(), m_alphaOption, m_gammaAndColorProfileOption);
81 if (!isDecoderAvailable())
82 return false;
83
84 m_decoder->setEncodedDataStatusChangeCallback([weakThis = makeWeakPtr(this)] (auto status) {
85 if (weakThis)
86 weakThis->encodedDataStatusChanged(status);
87 });
88
89 if (auto expectedContentSize = expectedContentLength())
90 m_decoder->setExpectedContentSize(expectedContentSize);
91
92 // Changing the decoder has to stop the decoding thread. The current frame will
93 // continue decoding safely because the decoding thread has its own
94 // reference of the old decoder.
95 stopAsyncDecodingQueue();
96 return true;
97}

128void ImageSource::destroyDecodedData(size_t frameCount, size_t excludeFrame)
129{
130 unsigned decodedSize = 0;
131
132 ASSERT(frameCount <= m_frames.size());
133
134 for (size_t index = 0; index < frameCount; ++index) {
135 if (index == excludeFrame)
136 continue;
137 decodedSize += m_frames[index].clearImage();
138 }
139
140 decodedSizeReset(decodedSize);
141}
142
143void ImageSource::destroyIncompleteDecodedData()
144{
145 unsigned decodedSize = 0;
146
147 for (auto& frame : m_frames) {
148 if (!frame.hasMetadata() || frame.isComplete())
149 continue;
150
151 decodedSize += frame.clear();
152 }
153
154 decodedSizeDecreased(decodedSize);
155}

186void ImageSource::decodedSizeIncreased(unsigned decodedSize)
187{
188 if (!decodedSize)
189 return;
190
191 m_decodedSize += decodedSize;
192
193 // The fully-decoded frame will subsume the partially decoded data used
194 // to determine image properties.
195 long long changeSize = static_cast<long long>(decodedSize) - m_decodedPropertiesSize;
196 m_decodedPropertiesSize = 0;
197 decodedSizeChanged(changeSize);
198}
199
200void ImageSource::decodedSizeDecreased(unsigned decodedSize)
201{
202 if (!decodedSize)
203 return;
204
205 ASSERT(m_decodedSize >= decodedSize);
206 m_decodedSize -= decodedSize;
207 decodedSizeChanged(-static_cast<long long>(decodedSize));
208}
209
210void ImageSource::decodedSizeReset(unsigned decodedSize)
211{
212 ASSERT(m_decodedSize >= decodedSize);
213 m_decodedSize -= decodedSize;
214
215 // Clearing the ImageSource destroys the extra decoded data used for
216 // determining image properties.
217 decodedSize += m_decodedPropertiesSize;
218 m_decodedPropertiesSize = 0;
219 decodedSizeChanged(-static_cast<long long>(decodedSize));
220}
221
222void ImageSource::didDecodeProperties(unsigned decodedPropertiesSize)
223{
224 if (m_decodedSize)
225 return;
226
227 long long decodedSize = static_cast<long long>(decodedPropertiesSize) - m_decodedPropertiesSize;
228 m_decodedPropertiesSize = decodedPropertiesSize;
229 decodedSizeChanged(decodedSize);
230}

254void ImageSource::cacheMetadataAtIndex(size_t index, SubsamplingLevel subsamplingLevel, DecodingStatus decodingStatus)
255{
256 ASSERT(index < m_frames.size());
257 ImageFrame& frame = m_frames[index];
258
259 ASSERT(isDecoderAvailable());
260 if (decodingStatus == DecodingStatus::Invalid)
261 frame.m_decodingStatus = m_decoder->frameIsCompleteAtIndex(index) ? DecodingStatus::Complete : DecodingStatus::Partial;
262 else
263 frame.m_decodingStatus = decodingStatus;
264
265 if (frame.hasMetadata())
266 return;
267
268 frame.m_subsamplingLevel = subsamplingLevel;
269
270 if (frame.m_decodingOptions.hasSizeForDrawing()) {
271 ASSERT(frame.hasNativeImage());
272 frame.m_size = frame.nativeImage()->size();
273 } else
274 frame.m_size = m_decoder->frameSizeAtIndex(index, subsamplingLevel);
275
276 auto metadata = m_decoder->frameMetadataAtIndex(index);
277 frame.m_orientation = metadata.orientation;
278 frame.m_densityCorrectedSize = metadata.densityCorrectedSize;
279 frame.m_hasAlpha = m_decoder->frameHasAlphaAtIndex(index);
280
281 if (repetitionCount())
282 frame.m_duration = m_decoder->frameDurationAtIndex(index);
283}

285void ImageSource::cachePlatformImageAtIndex(PlatformImagePtr&& platformImage, size_t index, SubsamplingLevel subsamplingLevel, const DecodingOptions& decodingOptions, DecodingStatus decodingStatus)
286{
287 ASSERT(index < m_frames.size());
288 ImageFrame& frame = m_frames[index];
289
290 // Clear the current image frame and update the observer with this clearance.
291 decodedSizeDecreased(frame.clear());
292
293 // Do not cache the NativeImage if adding its frameByes to the MemoryCache will cause numerical overflow.
294 size_t frameBytes = size().unclampedArea() * sizeof(uint32_t);
295 if (!isInBounds<unsigned>(frameBytes + decodedSize()))
296 return;
297
298 // Move the new image to the cache.
299 frame.m_nativeImage = NativeImage::create(WTFMove(platformImage));
300 frame.m_decodingOptions = decodingOptions;
301 cacheMetadataAtIndex(index, subsamplingLevel, decodingStatus);
302
303 // Update the observer with the new image frame bytes.
304 decodedSizeIncreased(frame.frameBytes());
305}

307void ImageSource::cachePlatformImageAtIndexAsync(PlatformImagePtr&& platformImage, size_t index, SubsamplingLevel subsamplingLevel, const DecodingOptions& decodingOptions, DecodingStatus decodingStatus)
308{
309 if (!isDecoderAvailable())
310 return;
311
312 ASSERT(index < m_frames.size());
313
314 // Clean the old native image and set a new one
315 cachePlatformImageAtIndex(WTFMove(platformImage), index, subsamplingLevel, decodingOptions, decodingStatus);
316 LOG(Images, "ImageSource::%s - %p - url: %s [frame %ld has been cached]", __FUNCTION__, this, sourceURL().string().utf8().data(), index);
317
318 // Notify the image with the readiness of the new frame NativeImage.
319 if (m_image)
320 m_image->imageFrameAvailableAtIndex(index);
321}

347void ImageSource::startAsyncDecodingQueue()
348{
349 if (hasAsyncDecodingQueue() || !isDecoderAvailable())
350 return;
351
352 // Async decoding is only enabled for HTMLImageElement and CSS background images.
353 ASSERT(isMainThread());
354
355 // We need to protect this, m_decodingQueue and m_decoder from being deleted while we are in the decoding loop.
356 decodingQueue().dispatch([protectedThis = makeRef(*this), protectedDecodingQueue = makeRef(decodingQueue()), protectedFrameRequestQueue = makeRef(frameRequestQueue()), protectedDecoder = makeRef(*m_decoder), sourceURL = sourceURL().string().isolatedCopy()] () mutable {
357 ImageFrameRequest frameRequest;
358 Seconds minDecodingDuration = protectedThis->frameDecodingDurationForTesting();
359
360 while (protectedFrameRequestQueue->dequeue(frameRequest)) {
361 TraceScope tracingScope(AsyncImageDecodeStart, AsyncImageDecodeEnd);
362
363 MonotonicTime startingTime;
364 if (minDecodingDuration > 0_s)
365 startingTime = MonotonicTime::now();
366
367 // Get the frame NativeImage on the decoding thread.
368 auto platformImage = protectedDecoder->createFrameImageAtIndex(frameRequest.index, frameRequest.subsamplingLevel, frameRequest.decodingOptions);
369 if (platformImage)
370 LOG(Images, "ImageSource::%s - %p - url: %s [frame %ld has been decoded]", __FUNCTION__, protectedThis.ptr(), sourceURL.utf8().data(), frameRequest.index);
371 else {
372 LOG(Images, "ImageSource::%s - %p - url: %s [decoding for frame %ld has failed]", __FUNCTION__, protectedThis.ptr(), sourceURL.utf8().data(), frameRequest.index);
373 continue;
374 }
375
376 // Pretend as if the decoding takes minDecodingDuration.
377 if (minDecodingDuration > 0_s)
378 sleep(minDecodingDuration - (MonotonicTime::now() - startingTime));
379
380 // Update the cached frames on the creation thread to avoid updating the MemoryCache from a different thread.
381 callOnMainThread([protectedThis, protectedDecodingQueue, protectedDecoder, sourceURL = sourceURL.isolatedCopy(), platformImage = WTFMove(platformImage), frameRequest] () mutable {
382 // The queue may have been closed if after we got the frame NativeImage, stopAsyncDecodingQueue() was called.
383 if (protectedDecodingQueue.ptr() == protectedThis->m_decodingQueue && protectedDecoder.ptr() == protectedThis->m_decoder) {
384 ASSERT(protectedThis->m_frameCommitQueue.first() == frameRequest);
385 protectedThis->m_frameCommitQueue.removeFirst();
386 protectedThis->cachePlatformImageAtIndexAsync(WTFMove(platformImage), frameRequest.index, frameRequest.subsamplingLevel, frameRequest.decodingOptions, frameRequest.decodingStatus);
387 } else
388 LOG(Images, "ImageSource::%s - %p - url: %s [frame %ld will not cached]", __FUNCTION__, protectedThis.ptr(), sourceURL.utf8().data(), frameRequest.index);
389 });
390 }
391
392 // Ensure destruction happens on creation thread.
393 callOnMainThread([protectedThis = WTFMove(protectedThis), protectedQueue = WTFMove(protectedDecodingQueue), protectedDecoder = WTFMove(protectedDecoder)] () mutable { });
394 });
395}

397void ImageSource::requestFrameAsyncDecodingAtIndex(size_t index, SubsamplingLevel subsamplingLevel, const std::optional<IntSize>& sizeForDrawing)
398{
399 ASSERT(isDecoderAvailable());
400 if (!hasAsyncDecodingQueue())
401 startAsyncDecodingQueue();
402
403 ASSERT(index < m_frames.size());
404 DecodingStatus decodingStatus = m_decoder->frameIsCompleteAtIndex(index) ? DecodingStatus::Complete : DecodingStatus::Partial;
405
406 LOG(Images, "ImageSource::%s - %p - url: %s [enqueuing frame %ld for decoding]", __FUNCTION__, this, sourceURL().string().utf8().data(), index);
407 m_frameRequestQueue->enqueue({ index, subsamplingLevel, sizeForDrawing, decodingStatus });
408 m_frameCommitQueue.append({ index, subsamplingLevel, sizeForDrawing, decodingStatus });
409}

416void ImageSource::stopAsyncDecodingQueue()
417{
418 if (!hasAsyncDecodingQueue())
419 return;
420
421 std::for_each(m_frameCommitQueue.begin(), m_frameCommitQueue.end(), [this](const ImageFrameRequest& frameRequest) {
422 ImageFrame& frame = m_frames[frameRequest.index];
423 if (!frame.isInvalid()) {
424 LOG(Images, "ImageSource::%s - %p - url: %s [decoding has been cancelled for frame %ld]", __FUNCTION__, this, sourceURL().string().utf8().data(), frameRequest.index);
425 frame.clear();
426 }
427 });
428
429 // Close m_frameRequestQueue then set it to nullptr. A new decoding thread might start and a
430 // new m_frameRequestQueue will be created. So the terminating thread will not have access to it.
431 m_frameRequestQueue->close();
432 m_frameRequestQueue = nullptr;
433 m_frameCommitQueue.clear();
434 m_decodingQueue = nullptr;
435 LOG(Images, "ImageSource::%s - %p - url: %s [decoding has been stopped]", __FUNCTION__, this, sourceURL().string().utf8().data());
436}

438const ImageFrame& ImageSource::frameAtIndexCacheIfNeeded(size_t index, ImageFrame::Caching caching, const std::optional<SubsamplingLevel>& subsamplingLevel)
439{
440 if (index >= m_frames.size())
441 return ImageFrame::defaultFrame();
442
443 ImageFrame& frame = m_frames[index];
444 if (!isDecoderAvailable() || frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(index, DecodingOptions(DecodingMode::Asynchronous)))
445 return frame;
446
447 SubsamplingLevel subsamplingLevelValue = subsamplingLevel ? subsamplingLevel.value() : frame.subsamplingLevel();
448
449 switch (caching) {
450 case ImageFrame::Caching::Metadata:
451 // Retrieve the metadata from ImageDecoder if the ImageFrame isn't complete.
452 if (frame.isComplete())
453 break;
454 cacheMetadataAtIndex(index, subsamplingLevelValue);
455 break;
456
457 case ImageFrame::Caching::MetadataAndImage:
458 // Cache the image and retrieve the metadata from ImageDecoder only if there was not valid image stored.
459 if (frame.hasFullSizeNativeImage(subsamplingLevel))
460 break;
461 // We have to perform synchronous image decoding in this code.
462 auto platformImage = m_decoder->createFrameImageAtIndex(index, subsamplingLevelValue);
463 // Clean the old native image and set a new one.
464 cachePlatformImageAtIndex(WTFMove(platformImage), index, subsamplingLevelValue, DecodingOptions(DecodingMode::Synchronous));
465 break;
466 }
467
468 return frame;
469}

497template<typename T>
498T ImageSource::metadataCacheIfNeeded(T& cachedValue, const T& defaultValue, MetadataType metadataType, T (ImageDecoder::*functor)() const)
499{
500 if (m_cachedMetadata.contains(metadataType))
501 return cachedValue;
502
503 if (!isDecoderAvailable() || !m_decoder->isSizeAvailable())
504 return defaultValue;
505
506 cachedValue = (*m_decoder.*functor)();
507 m_cachedMetadata.add(metadataType);
508 didDecodeProperties(m_decoder->bytesDecodedToDetermineProperties());
509 return cachedValue;
510}

512template<typename T>
513T ImageSource::firstFrameMetadataCacheIfNeeded(T& cachedValue, MetadataType metadataType, T (ImageFrame::*functor)() const, ImageFrame::Caching caching, const std::optional<SubsamplingLevel>& subsamplingLevel)
514{
515 if (m_cachedMetadata.contains(metadataType))
516 return cachedValue;
517
518 const ImageFrame& frame = frameAtIndexCacheIfNeeded(0, caching, subsamplingLevel);
519
520 // Don't cache any unavailable frame metadata.
521 if (!frame.hasMetadata())
522 return (frame.*functor)();
523
524 cachedValue = (frame.*functor)();
525 m_cachedMetadata.add(metadataType);
526 return cachedValue;
527}

591IntSize ImageSource::sourceSize(ImageOrientation orientation)
592{
593 IntSize size;
594#if !USE(CG)
595 // It's possible that we have decoded the metadata, but not frame contents yet. In that case ImageDecoder claims to
596 // have the size available, but the frame cache is empty. Return the decoder size without caching in such case.
597 if (m_frames.isEmpty() && isDecoderAvailable())
598 size = m_decoder->size();
599 else
600#endif
601 size = firstFrameMetadataCacheIfNeeded(m_size, MetadataType::Size, &ImageFrame::size, ImageFrame::Caching::Metadata, SubsamplingLevel::Default);
602
603 if (orientation == ImageOrientation::FromImage)
604 orientation = this->orientation();
605
606 return orientation.usesWidthAsHeight() ? size.transposedSize() : size;
607}

619SubsamplingLevel ImageSource::maximumSubsamplingLevel()
620{
621 if (m_cachedMetadata.contains(MetadataType::MaximumSubsamplingLevel))
622 return m_maximumSubsamplingLevel;
623
624 if (!isDecoderAvailable() || !m_decoder->frameAllowSubsamplingAtIndex(0))
625 return SubsamplingLevel::Default;
626
627 // FIXME: this value was chosen to be appropriate for iOS since the image
628 // subsampling is only enabled by default on iOS. Choose a different value
629 // if image subsampling is enabled on other platform.
630 const int maximumImageAreaBeforeSubsampling = 5 * 1024 * 1024;
631 SubsamplingLevel level = SubsamplingLevel::First;
632
633 for (; level < SubsamplingLevel::Last; ++level) {
634 if (frameSizeAtIndex(0, level).area() < maximumImageAreaBeforeSubsampling)
635 break;
636 }
637
638 m_maximumSubsamplingLevel = level;
639 m_cachedMetadata.add(MetadataType::MaximumSubsamplingLevel);
640 return m_maximumSubsamplingLevel;
641}

696#if USE(DIRECT2D)
697void ImageSource::setTargetContext(const GraphicsContext* targetContext)
