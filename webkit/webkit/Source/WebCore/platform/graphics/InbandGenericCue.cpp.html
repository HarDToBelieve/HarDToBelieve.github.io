42String InbandGenericCue::toJSONString() const
43{
44 auto object = JSON::Object::create();
45
46#if !LOG_DISABLED
47 object->setString("text"_s, m_cueData.m_content);
48#endif
49 object->setInteger("identifier"_s, m_cueData.m_uniqueId.toUInt64());
50 object->setDouble("start"_s, m_cueData.m_startTime.toDouble());
51 object->setDouble("end"_s, m_cueData.m_endTime.toDouble());
52
53 const char* status = "";
54 switch (m_cueData.m_status) {
55 case GenericCueData::Status::Uninitialized:
56 status = "Uninitialized";
57 break;
58 case GenericCueData::Status::Partial:
59 status = "Partial";
60 break;
61 case GenericCueData::Status::Complete:
62 status = "Complete";
63 break;
64 }
65 object->setString("status", status);
66
67 if (!m_cueData.m_id.isEmpty())
68 object->setString("id", m_cueData.m_id);
69
70 if (m_cueData.m_line > 0)
71 object->setDouble("line"_s, m_cueData.m_line);
72
73 if (m_cueData.m_size > 0)
74 object->setDouble("size"_s, m_cueData.m_size);
75
76 if (m_cueData.m_position > 0)
77 object->setDouble("position"_s, m_cueData.m_position);
78
79 if (m_cueData.m_align != GenericCueData::Alignment::None) {
80 const char* align = "";
81 switch (m_cueData.m_align) {
82 case GenericCueData::Alignment::Start:
83 align = "Start";
84 break;
85 case GenericCueData::Alignment::Middle:
86 align = "Middle";
87 break;
88 case GenericCueData::Alignment::End:
89 align = "End";
90 break;
91 case GenericCueData::Alignment::None:
92 align = "None";
93 break;
94 }
95 object->setString("align"_s, align);
96 }
97
98 if (m_cueData.m_foregroundColor.isValid())
99 object->setString("foregroundColor"_s, serializationForHTML(m_cueData.m_foregroundColor));
100
101 if (m_cueData.m_backgroundColor.isValid())
102 object->setString("backgroundColor"_s, serializationForHTML(m_cueData.m_backgroundColor));
103
104 if (m_cueData.m_highlightColor.isValid())
105 object->setString("highlightColor"_s, serializationForHTML(m_cueData.m_highlightColor));
106
107 if (m_cueData.m_baseFontSize)
108 object->setDouble("baseFontSize"_s, m_cueData.m_baseFontSize);
109
110 if (m_cueData.m_relativeFontSize)
111 object->setDouble("relativeFontSize"_s, m_cueData.m_relativeFontSize);
112
113 if (!m_cueData.m_fontName.isEmpty())
114 object->setString("font"_s, m_cueData.m_fontName);
115
116 return object->toJSONString();
117}
118
119bool GenericCueData::equalNotConsideringTimesOrId(const GenericCueData& other) const
120{
121 return m_relativeFontSize == other.m_relativeFontSize
122 && m_baseFontSize == other.m_baseFontSize
123 && m_position == other.m_position
124 && m_line == other.m_line
125 && m_size == other.m_size
126 && m_align == other.m_align
127 && m_foregroundColor == other.m_foregroundColor
128 && m_backgroundColor == other.m_backgroundColor
129 && m_highlightColor == other.m_highlightColor
130 && m_fontName == other.m_fontName
131 && m_id == other.m_id
132 && m_content == other.m_content;
133}
