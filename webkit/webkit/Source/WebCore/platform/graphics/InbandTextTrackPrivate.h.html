30#include "InbandTextTrackPrivateClient.h"
31
32namespace WebCore {
33
34class InbandTextTrackPrivate : public TrackPrivateBase {
35public:
36 enum class CueFormat : uint8_t {
37 Data,
38 Generic,
39 WebVTT
40 };
41 static Ref<InbandTextTrackPrivate> create(CueFormat format) { return adoptRef(*new InbandTextTrackPrivate(format)); }
42 virtual ~InbandTextTrackPrivate() = default;
43
44 InbandTextTrackPrivateClient* client() const override { return m_client; }
45 void setClient(InbandTextTrackPrivateClient* client) { m_client = client; }
46
47 enum class Mode : uint8_t {
48 Disabled,
49 Hidden,
50 Showing
51 };
52 virtual void setMode(Mode mode) { m_mode = mode; };
53 virtual InbandTextTrackPrivate::Mode mode() const { return m_mode; }
54
55 enum class Kind : uint8_t {
56 Subtitles,
57 Captions,
58 Descriptions,
59 Chapters,
60 Metadata,
61 Forced,
62 None
63 };
64 virtual Kind kind() const { return Kind::Subtitles; }
65
66 virtual bool isClosedCaptions() const { return false; }
67 virtual bool isSDH() const { return false; }
68 virtual bool containsOnlyForcedSubtitles() const { return false; }
69 virtual bool isMainProgramContent() const { return true; }
70 virtual bool isEasyToRead() const { return false; }
71 virtual bool isDefault() const { return false; }
72 AtomString label() const override { return emptyAtom(); }
73 AtomString language() const override { return emptyAtom(); }
74 AtomString id() const override { return emptyAtom(); }
75 virtual AtomString inBandMetadataTrackDispatchType() const { return emptyAtom(); }
76
77 virtual CueFormat cueFormat() const { return m_format; }
78
79#if !RELEASE_LOG_DISABLED
80 const char* logClassName() const override { return "InbandTextTrackPrivate"; }
81#endif
82
83protected:
84 InbandTextTrackPrivate(CueFormat format)
85 : m_format(format)
86 {
87 }
88
89private:
90 CueFormat m_format;
91 InbandTextTrackPrivateClient* m_client { nullptr };
92 Mode m_mode { Mode::Disabled };
93};
94
95} // namespace WebCore
