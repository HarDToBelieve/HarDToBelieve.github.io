32namespace WebCore {
33
34IntPoint::IntPoint(const FloatPoint& p)
35 : m_x(clampToInteger(p.x()))
36 , m_y(clampToInteger(p.y()))
37{
38}
39
40IntPoint IntPoint::constrainedBetween(const IntPoint& min, const IntPoint& max) const
41{
42 return {
43 std::max(min.x(), std::min(max.x(), m_x)),
44 std::max(min.y(), std::min(max.y(), m_y))
45 };
46}
47
48TextStream& operator<<(TextStream& ts, const IntPoint& p)
49{
50 return ts << "(" << p.x() << "," << p.y() << ")";
51}
