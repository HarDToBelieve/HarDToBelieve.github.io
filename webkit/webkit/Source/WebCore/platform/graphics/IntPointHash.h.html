20#ifndef IntPointHash_h
21#define IntPointHash_h
22
23#include "IntPoint.h"
24#include <wtf/HashFunctions.h>
25#include <wtf/HashTraits.h>
26
27namespace WTF {
28
29// The empty value is (0, INT_MIN), the deleted value is (INT_MIN, 0)
30struct IntPointHash {
31 static unsigned hash(const WebCore::IntPoint& p) { return pairIntHash(p.x(), p.y()); }
32 static bool equal(const WebCore::IntPoint& a, const WebCore::IntPoint& b) { return a == b; }
33 static const bool safeToCompareToEmptyOrDeleted = true;
34};
35template<> struct HashTraits<WebCore::IntPoint> : GenericHashTraits<WebCore::IntPoint> {
36 static WebCore::IntPoint emptyValue() { return WebCore::IntPoint(0, std::numeric_limits<int>::min()); }
37
38 static void constructDeletedValue(WebCore::IntPoint& slot) { slot.setX(std::numeric_limits<int>::min()); }
39 static bool isDeletedValue(const WebCore::IntPoint& slot) { return slot.x() == std::numeric_limits<int>::min(); }
40};
41template<> struct DefaultHash<WebCore::IntPoint> : IntPointHash { };
42
43}
