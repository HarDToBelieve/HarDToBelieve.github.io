33HashSet<String, ASCIICaseInsensitiveHash>& MIMETypeCache::supportedTypes()
34{
35 if (!m_supportedTypes) {
36 m_supportedTypes = HashSet<String, ASCIICaseInsensitiveHash> { };
37 initializeCache(*m_supportedTypes);
38 }
39
40 return *m_supportedTypes;
41}

43bool MIMETypeCache::supportsContainerType(const String& containerType)
44{
45 if (!isAvailable() || containerType.isEmpty())
46 return false;
47
48 if (isUnsupportedContainerType(containerType))
49 return false;
50
51 return isStaticContainerType(containerType) || supportedTypes().contains(containerType);
52}

54MediaPlayerEnums::SupportsType MIMETypeCache::canDecodeType(const String& mimeType)
55{
56 if (mimeType.isEmpty())
57 return MediaPlayerEnums::SupportsType::IsNotSupported;
58
59 if (m_cachedResults) {
60 auto it = m_cachedResults->find(mimeType);
61 if (it != m_cachedResults->end())
62 return it->value;
63 }
64
65 auto result = MediaPlayerEnums::SupportsType::IsNotSupported;
66 do {
67 if (!isAvailable() || mimeType.isEmpty())
68 break;
69
70 auto contentType = ContentType { mimeType };
71 auto containerType = contentType.containerType();
72 if (!supportsContainerType(containerType))
73 break;
74
75 if (contentType.codecs().isEmpty()) {
76 result = MediaPlayerEnums::SupportsType::MayBeSupported;
77 break;
78 }
79
80 if (shouldOverrideExtendedType(contentType)) {
81 result = MediaPlayerEnums::SupportsType::IsSupported;
82 break;
83 }
84
85 if (canDecodeExtendedType(contentType))
86 result = MediaPlayerEnums::SupportsType::IsSupported;
87
88 } while (0);
89
90 if (!m_cachedResults)
91 m_cachedResults = HashMap<String, MediaPlayerEnums::SupportsType, ASCIICaseInsensitiveHash>();
92 m_cachedResults->add(mimeType, result);
93
94 return result;
95}

97void MIMETypeCache::addSupportedTypes(const Vector<String>& newTypes)
98{
99 if (!m_supportedTypes)
100 m_supportedTypes = HashSet<String, ASCIICaseInsensitiveHash> { };
101
102 for (auto& type : newTypes)
103 m_supportedTypes->add(type);
104}

135bool MIMETypeCache::shouldOverrideExtendedType(const ContentType& type)
136{
137 ASSERT(canDecodeType(type.containerType()) != MediaPlayerEnums::SupportsType::IsNotSupported);
138
139 // Some sites (e.g. Modernizr) use 'audio/mpeg; codecs="mp3"' even though
140 // it is not RFC 3003 compliant.
141 if (equalIgnoringASCIICase(type.containerType(), "audio/mpeg")) {
142 auto codecs = type.codecs();
143 return codecs.size() == 1 && codecs[0] == "mp3";
144 }
145
146 return false;
147}
