30#include "Color.h"

32#include "PlatformImage.h"

37#include <wtf/WeakPtr.h>
38
39namespace WebCore {
40
41class NativeImage : public RefCounted<NativeImage>, public CanMakeWeakPtr<NativeImage> {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 class Observer {
45 public:
46 virtual ~Observer() = default;
47 virtual void releaseNativeImage(RenderingResourceIdentifier) = 0;
48 protected:
49 Observer() = default;
50 };
51
52 static WEBCORE_EXPORT RefPtr<NativeImage> create(PlatformImagePtr&&, RenderingResourceIdentifier = RenderingResourceIdentifier::generate());
53
54 WEBCORE_EXPORT ~NativeImage();
55
56 const PlatformImagePtr& platformImage() const { return m_platformImage; }
57 RenderingResourceIdentifier renderingResourceIdentifier() const { return m_renderingResourceIdentifier; }
58
59 WEBCORE_EXPORT IntSize size() const;
60 bool hasAlpha() const;
61 Color singlePixelSolidColor() const;
62
63 void addObserver(Observer& observer) { m_observers.add(&observer); }
64 void removeObserver(Observer& observer) { m_observers.remove(&observer); }
65
66 void clearSubimages();
67
68private:
69 NativeImage(PlatformImagePtr&&);
70 NativeImage(PlatformImagePtr&&, RenderingResourceIdentifier);
71
72 PlatformImagePtr m_platformImage;
73 HashSet<Observer*> m_observers;
74 RenderingResourceIdentifier m_renderingResourceIdentifier;
75};
76
77using NativeImageHashMap = HashMap<RenderingResourceIdentifier, Ref<NativeImage>>;

79} // namespace WebCore
