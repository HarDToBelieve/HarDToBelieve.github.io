93// The points in the structure are the same as those that would be used with the
94// add... method. For example, a line returns the endpoint, while a cubic returns
95// two tangent points and the endpoint.
96struct PathElement {
97 enum class Type : uint8_t {
98 MoveToPoint, // The points member will contain 1 value.
99 AddLineToPoint, // The points member will contain 1 value.
100 AddQuadCurveToPoint, // The points member will contain 2 values.
101 AddCurveToPoint, // The points member will contain 3 values.
102 CloseSubpath // The points member will contain no values.
103 };
104
105 FloatPoint points[3];
106 Type type;
107};

300template<class Encoder> void Path::encode(Encoder& encoder) const
301{
302#if ENABLE(INLINE_PATH_DATA)
303 bool hasInlineData = this->hasInlineData();
304 encoder << hasInlineData;
305 if (hasInlineData) {
306 encoder << m_inlineData;
307 return;
308 }
309#endif
310
311 encoder << static_cast<uint64_t>(elementCount());
312
313 apply([&](auto& element) {
314 encoder << element.type;
315
316 switch (element.type) {
317 case PathElement::Type::MoveToPoint:
318 encoder << element.points[0];
319 break;
320 case PathElement::Type::AddLineToPoint:
321 encoder << element.points[0];
322 break;
323 case PathElement::Type::AddQuadCurveToPoint:
324 encoder << element.points[0];
325 encoder << element.points[1];
326 break;
327 case PathElement::Type::AddCurveToPoint:
328 encoder << element.points[0];
329 encoder << element.points[1];
330 encoder << element.points[2];
331 break;
332 case PathElement::Type::CloseSubpath:
333 break;
334 }
335 });
336}

338template<class Decoder> std::optional<Path> Path::decode(Decoder& decoder)
339{
340 Path path;
341
342#if ENABLE(INLINE_PATH_DATA)
343 bool hasInlineData;
344 if (!decoder.decode(hasInlineData))
345 return std::nullopt;
346
347 if (hasInlineData) {
348 if (!decoder.decode(path.m_inlineData))
349 return std::nullopt;
350
351 return path;
352 }
353#endif
354
355 uint64_t numPoints;
356 if (!decoder.decode(numPoints))
357 return std::nullopt;
358
359 path.clear();
360
361 for (uint64_t i = 0; i < numPoints; ++i) {
362 PathElement::Type elementType;
363 if (!decoder.decode(elementType))
364 return std::nullopt;
365
366 switch (elementType) {
367 case PathElement::Type::MoveToPoint: {
368 FloatPoint point;
369 if (!decoder.decode(point))
370 return std::nullopt;
371 path.moveTo(point);
372 break;
373 }
374 case PathElement::Type::AddLineToPoint: {
375 FloatPoint point;
376 if (!decoder.decode(point))
377 return std::nullopt;
378 path.addLineTo(point);
379 break;
380 }
381 case PathElement::Type::AddQuadCurveToPoint: {
382 FloatPoint controlPoint;
383 if (!decoder.decode(controlPoint))
384 return std::nullopt;
385
386 FloatPoint endPoint;
387 if (!decoder.decode(endPoint))
388 return std::nullopt;
389
390 path.addQuadCurveTo(controlPoint, endPoint);
391 break;
392 }
393 case PathElement::Type::AddCurveToPoint: {
394 FloatPoint controlPoint1;
395 if (!decoder.decode(controlPoint1))
396 return std::nullopt;
397
398 FloatPoint controlPoint2;
399 if (!decoder.decode(controlPoint2))
400 return std::nullopt;
401
402 FloatPoint endPoint;
403 if (!decoder.decode(endPoint))
404 return std::nullopt;
405
406 path.addBezierCurveTo(controlPoint1, controlPoint2, endPoint);
407 break;
408 }
409 case PathElement::Type::CloseSubpath:
410 path.closeSubpath();
411 break;
412 }
413 }
414
415 return path;
416}
