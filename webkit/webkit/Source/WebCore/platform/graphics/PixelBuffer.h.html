79template<class Encoder> void PixelBuffer::encode(Encoder& encoder) const
80{
81 ASSERT(m_data->byteLength() == (m_size.area() * 4));
82
83 encoder << m_format;
84 encoder << m_size;
85 encoder.encodeFixedLengthData(m_data->data(), m_data->byteLength(), 1);
86}
87
88template<class Decoder> std::optional<PixelBuffer> PixelBuffer::decode(Decoder& decoder)
89{
90 std::optional<PixelBufferFormat> format;
91 decoder >> format;
92 if (!format)
93 return std::nullopt;
94
95 // FIXME: Support non-8 bit formats.
96 if (!(format->pixelFormat == PixelFormat::RGBA8 || format->pixelFormat == PixelFormat::BGRA8))
97 return std::nullopt;
98
99 std::optional<IntSize> size;
100 decoder >> size;
101 if (!size)
102 return std::nullopt;
103
104 auto computedBufferSize = PixelBuffer::computeBufferSize(*format, *size);
105 if (computedBufferSize.hasOverflowed())
106 return std::nullopt;
107
108 auto bufferSize = computedBufferSize;
109 if (!decoder.template bufferIsLargeEnoughToContain<uint8_t>(bufferSize))
110 return std::nullopt;
111
112 auto result = PixelBuffer::tryCreateForDecoding(WTFMove(*format), *size, bufferSize);
113 if (!result)
114 return std::nullopt;
115
116 if (!decoder.decodeFixedLengthData(result->m_data->data(), result->m_data->byteLength(), 1))
117 return std::nullopt;
118
119 return result;
120}
