42static inline vImage_CGImageFormat makeVImageCGImageFormat(const PixelBufferFormat& format)
43{
44 auto [bitsPerComponent, bitsPerPixel, bitmapInfo] = [] (const PixelBufferFormat& format) -> std::tuple<unsigned, unsigned, CGBitmapInfo> {
45 switch (format.pixelFormat) {
46 case PixelFormat::RGBA8:
47 if (format.alphaFormat == AlphaPremultiplication::Premultiplied)
48 return std::make_tuple(8u, 32u, kCGBitmapByteOrder32Big | kCGImageAlphaPremultipliedLast);
49 else
50 return std::make_tuple(8u, 32u, kCGBitmapByteOrder32Big | kCGImageAlphaLast);
51
52 case PixelFormat::BGRA8:
53 if (format.alphaFormat == AlphaPremultiplication::Premultiplied)
54 return std::make_tuple(8u, 32u, kCGBitmapByteOrder32Little | kCGImageAlphaPremultipliedFirst);
55 else
56 return std::make_tuple(8u, 32u, kCGBitmapByteOrder32Little | kCGImageAlphaFirst);
57
58 case PixelFormat::RGB10:
59 case PixelFormat::RGB10A8:
60 break;
61 }
62
63 // We currently only support 8 bit pixel formats for these conversions.
64
65 ASSERT_NOT_REACHED();
66 return std::make_tuple(8u, 32u, kCGBitmapByteOrder32Little | kCGImageAlphaFirst);
67 }(format);
68
69 vImage_CGImageFormat result;
70
71 result.bitsPerComponent = bitsPerComponent;
72 result.bitsPerPixel = bitsPerPixel;
73 result.colorSpace = format.colorSpace.platformColorSpace();
74 result.bitmapInfo = bitmapInfo;
75 result.version = 0;
76 result.decode = nullptr;
77 result.renderingIntent = kCGRenderingIntentDefault;
78
79 return result;
80}

82template<typename View> static vImage_Buffer makeVImageBuffer(const View& view, const IntSize& size)
83{
84 vImage_Buffer result;
85
86 result.height = static_cast<vImagePixelCount>(size.height());
87 result.width = static_cast<vImagePixelCount>(size.width());
88 result.rowBytes = view.bytesPerRow;
89 result.data = const_cast<uint8_t*>(view.rows);
90
91 return result;
92}

94static void convertImagePixelsAccelerated(const ConstPixelBufferConversionView& source, const PixelBufferConversionView& destination, const IntSize& destinationSize)
95{
96 auto sourceVImageBuffer = makeVImageBuffer(source, destinationSize);
97 auto destinationVImageBuffer = makeVImageBuffer(destination, destinationSize);
98
99 if (source.format.colorSpace != destination.format.colorSpace) {
100 // FIXME: Consider using vImageConvert_AnyToAny for all conversions, not just ones that need a color space conversion,
101 // after judiciously performance testing them against each other.
102
103 auto sourceCGImageFormat = makeVImageCGImageFormat(source.format);
104 auto destinationCGImageFormat = makeVImageCGImageFormat(destination.format);
105
106 vImage_Error converterCreateError = kvImageNoError;
107 auto converter = adoptCF(vImageConverter_CreateWithCGImageFormat(&sourceCGImageFormat, &destinationCGImageFormat, nullptr, kvImageNoFlags, &converterCreateError));
108 ASSERT_WITH_MESSAGE_UNUSED(converterCreateError, converterCreateError == kvImageNoError, "vImageConverter creation failed with error: %zd", converterCreateError);
109
110 vImage_Error converterConvertError = vImageConvert_AnyToAny(converter.get(), &sourceVImageBuffer, &destinationVImageBuffer, nullptr, kvImageNoFlags);
111 ASSERT_WITH_MESSAGE_UNUSED(converterConvertError, converterConvertError == kvImageNoError, "vImageConvert_AnyToAny failed conversion with error: %zd", converterConvertError);
112 return;
113 }
114
115 if (source.format.alphaFormat != destination.format.alphaFormat) {
116 if (destination.format.alphaFormat == AlphaPremultiplication::Unpremultiplied) {
117 if (source.format.pixelFormat == PixelFormat::RGBA8)
118 vImageUnpremultiplyData_RGBA8888(&sourceVImageBuffer, &destinationVImageBuffer, kvImageNoFlags);
119 else
120 vImageUnpremultiplyData_BGRA8888(&sourceVImageBuffer, &destinationVImageBuffer, kvImageNoFlags);
121 } else {
122 if (source.format.pixelFormat == PixelFormat::RGBA8)
123 vImagePremultiplyData_RGBA8888(&sourceVImageBuffer, &destinationVImageBuffer, kvImageNoFlags);
124 else
125 vImagePremultiplyData_BGRA8888(&sourceVImageBuffer, &destinationVImageBuffer, kvImageNoFlags);
126 }
127
128 sourceVImageBuffer = destinationVImageBuffer;
129 }
130
131 if (source.format.pixelFormat != destination.format.pixelFormat) {
132 // Swap pixel channels BGRA <-> RGBA.
133 const uint8_t map[4] = { 2, 1, 0, 3 };
134 vImagePermuteChannels_ARGB8888(&sourceVImageBuffer, &destinationVImageBuffer, map, kvImageNoFlags);
135 }
136}

140enum class PixelFormatConversion { None, Permute };
141
142template<PixelFormatConversion pixelFormatConversion>
143static void convertSinglePixelPremultipliedToPremultiplied(const uint8_t* sourcePixel, uint8_t* destinationPixel)
144{
145 uint8_t alpha = sourcePixel[3];
146 if (!alpha) {
147 reinterpret_cast<uint32_t*>(destinationPixel)[0] = 0;
148 return;
149 }
150
151 if constexpr (pixelFormatConversion == PixelFormatConversion::None)
152 reinterpret_cast<uint32_t*>(destinationPixel)[0] = reinterpret_cast<const uint32_t*>(sourcePixel)[0];
153 else {
154 // Swap pixel channels BGRA <-> RGBA.
155 destinationPixel[0] = sourcePixel[2];
156 destinationPixel[1] = sourcePixel[1];
157 destinationPixel[2] = sourcePixel[0];
158 destinationPixel[3] = sourcePixel[3];
159 }
160}
161
162template<PixelFormatConversion pixelFormatConversion>
163static void convertSinglePixelPremultipliedToUnpremultiplied(const uint8_t* sourcePixel, uint8_t* destinationPixel)
164{
165 uint8_t alpha = sourcePixel[3];
166 if (!alpha || alpha == 255) {
167 convertSinglePixelPremultipliedToPremultiplied<pixelFormatConversion>(sourcePixel, destinationPixel);
168 return;
169 }
170
171 if constexpr (pixelFormatConversion == PixelFormatConversion::None) {
172 destinationPixel[0] = (sourcePixel[0] * 255) / alpha;
173 destinationPixel[1] = (sourcePixel[1] * 255) / alpha;
174 destinationPixel[2] = (sourcePixel[2] * 255) / alpha;
175 destinationPixel[3] = alpha;
176 } else {
177 // Swap pixel channels BGRA <-> RGBA.
178 destinationPixel[0] = (sourcePixel[2] * 255) / alpha;
179 destinationPixel[1] = (sourcePixel[1] * 255) / alpha;
180 destinationPixel[2] = (sourcePixel[0] * 255) / alpha;
181 destinationPixel[3] = alpha;
182 }
183}

185template<PixelFormatConversion pixelFormatConversion>
186static void convertSinglePixelUnpremultipliedToPremultiplied(const uint8_t* sourcePixel, uint8_t* destinationPixel)
187{
188 uint8_t alpha = sourcePixel[3];
189 if (!alpha || alpha == 255) {
190 convertSinglePixelPremultipliedToPremultiplied<pixelFormatConversion>(sourcePixel, destinationPixel);
191 return;
192 }
193
194 if constexpr (pixelFormatConversion == PixelFormatConversion::None) {
195 destinationPixel[0] = (sourcePixel[0] * alpha + 254) / 255;
196 destinationPixel[1] = (sourcePixel[1] * alpha + 254) / 255;
197 destinationPixel[2] = (sourcePixel[2] * alpha + 254) / 255;
198 destinationPixel[3] = alpha;
199 } else {
200 // Swap pixel channels BGRA <-> RGBA.
201 destinationPixel[0] = (sourcePixel[2] * alpha + 254) / 255;
202 destinationPixel[1] = (sourcePixel[1] * alpha + 254) / 255;
203 destinationPixel[2] = (sourcePixel[0] * alpha + 254) / 255;
204 destinationPixel[3] = alpha;
205 }
206}

208template<PixelFormatConversion pixelFormatConversion>
209static void convertSinglePixelUnpremultipliedToUnpremultiplied(const uint8_t* sourcePixel, uint8_t* destinationPixel)
210{
211 if constexpr (pixelFormatConversion == PixelFormatConversion::None)
212 reinterpret_cast<uint32_t*>(destinationPixel)[0] = reinterpret_cast<const uint32_t*>(sourcePixel)[0];
213 else {
214 // Swap pixel channels BGRA <-> RGBA.
215 destinationPixel[0] = sourcePixel[2];
216 destinationPixel[1] = sourcePixel[1];
217 destinationPixel[2] = sourcePixel[0];
218 destinationPixel[3] = sourcePixel[3];
219 }

222template<void (*convertFunctor)(const uint8_t*, uint8_t*)>
223static void convertImagePixelsUnaccelerated(const ConstPixelBufferConversionView& source, const PixelBufferConversionView& destination, const IntSize& destinationSize)
224{
225 const uint8_t* sourceRows = source.rows;
226 uint8_t* destinationRows = destination.rows;
227
228 size_t bytesPerRow = destinationSize.width() * 4;
229 for (int y = 0; y < destinationSize.height(); ++y) {
230 for (size_t x = 0; x < bytesPerRow; x += 4)
231 convertFunctor(&sourceRows[x], &destinationRows[x]);
232 sourceRows += source.bytesPerRow;
233 destinationRows += destination.bytesPerRow;
234 }
235}

237void convertImagePixels(const ConstPixelBufferConversionView& source, const PixelBufferConversionView& destination, const IntSize& destinationSize)
238{
239 // We don't currently support converting pixel data with non-8-bit buffers.
240 ASSERT(source.format.pixelFormat == PixelFormat::RGBA8 || source.format.pixelFormat == PixelFormat::BGRA8);
241 ASSERT(destination.format.pixelFormat == PixelFormat::RGBA8 || destination.format.pixelFormat == PixelFormat::BGRA8);
242
243#if USE(ACCELERATE) && USE(CG)
244 if (source.format.alphaFormat == destination.format.alphaFormat && source.format.pixelFormat == destination.format.pixelFormat) {
245 // FIXME: Can thes both just use per-row memcpy?
246 if (source.format.alphaFormat == AlphaPremultiplication::Premultiplied)
247 convertImagePixelsUnaccelerated<convertSinglePixelPremultipliedToPremultiplied<PixelFormatConversion::None>>(source, destination, destinationSize);
248 else
249 convertImagePixelsUnaccelerated<convertSinglePixelUnpremultipliedToUnpremultiplied<PixelFormatConversion::None>>(source, destination, destinationSize);
250 } else
251 convertImagePixelsAccelerated(source, destination, destinationSize);
252#else
253 // FIXME: We don't currently support converting pixel data between different color spaces in the non-accelerated path.
254 // This could be added using conversion functions from ColorConversion.h.
255 ASSERT(source.format.colorSpace == destination.format.colorSpace);
256
257 if (source.format.alphaFormat == destination.format.alphaFormat) {
258 if (source.format.pixelFormat == destination.format.pixelFormat) {
259 if (source.format.alphaFormat == AlphaPremultiplication::Premultiplied)
260 convertImagePixelsUnaccelerated<convertSinglePixelPremultipliedToPremultiplied<PixelFormatConversion::None>>(source, destination, destinationSize);
261 else
262 convertImagePixelsUnaccelerated<convertSinglePixelUnpremultipliedToUnpremultiplied<PixelFormatConversion::None>>(source, destination, destinationSize);
263 } else {
264 if (destination.format.alphaFormat == AlphaPremultiplication::Premultiplied)
265 convertImagePixelsUnaccelerated<convertSinglePixelPremultipliedToPremultiplied<PixelFormatConversion::Permute>>(source, destination, destinationSize);
266 else
267 convertImagePixelsUnaccelerated<convertSinglePixelUnpremultipliedToUnpremultiplied<PixelFormatConversion::Permute>>(source, destination, destinationSize);
268 }
269 } else {
270 if (source.format.pixelFormat == destination.format.pixelFormat) {
271 if (source.format.alphaFormat == AlphaPremultiplication::Premultiplied)
272 convertImagePixelsUnaccelerated<convertSinglePixelPremultipliedToUnpremultiplied<PixelFormatConversion::None>>(source, destination, destinationSize);
273 else
274 convertImagePixelsUnaccelerated<convertSinglePixelUnpremultipliedToPremultiplied<PixelFormatConversion::None>>(source, destination, destinationSize);
275 } else {
276 if (destination.format.alphaFormat == AlphaPremultiplication::Premultiplied)
277 convertImagePixelsUnaccelerated<convertSinglePixelUnpremultipliedToPremultiplied<PixelFormatConversion::Permute>>(source, destination, destinationSize);
278 else
279 convertImagePixelsUnaccelerated<convertSinglePixelPremultipliedToUnpremultiplied<PixelFormatConversion::Permute>>(source, destination, destinationSize);
280 }
281 }
282#endif
