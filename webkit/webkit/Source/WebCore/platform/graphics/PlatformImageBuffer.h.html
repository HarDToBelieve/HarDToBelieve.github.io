63#if HAVE(IOSURFACE)
64class IOSurfaceImageBuffer final : public ConcreteImageBuffer<ImageBufferIOSurfaceBackend> {
65 using Base = ConcreteImageBuffer<ImageBufferIOSurfaceBackend>;
66public:
67 static auto create(const WebCore::FloatSize& size, float resolutionScale, const WebCore::DestinationColorSpace& colorSpace, WebCore::PixelFormat pixelFormat, const HostWindow* hostWindow = nullptr)
68 {
69 return Base::create<IOSurfaceImageBuffer>(size, resolutionScale, colorSpace, pixelFormat, hostWindow);
70 }
71 static auto create(const FloatSize& size, const GraphicsContext& context)
72 {
73 return Base::create<IOSurfaceImageBuffer>(size, context);
74 }
75 using Base::Base;
76
77 IOSurface& surface() { return *m_backend->surface(); }
78};
79using AcceleratedImageBuffer = IOSurfaceImageBuffer;
80#else
81using AcceleratedImageBuffer = ConcreteImageBuffer<UnacceleratedImageBufferBackend>;
82#endif
