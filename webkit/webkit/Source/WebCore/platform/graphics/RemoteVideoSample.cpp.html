45static inline std::unique_ptr<IOSurface> transferBGRAPixelBufferToIOSurface(CVPixelBufferRef pixelBuffer)
46{
47#if USE(ACCELERATE)
48 ASSERT(CVPixelBufferGetPixelFormatType(pixelBuffer) == kCVPixelFormatType_32BGRA);
49
50 auto result = CVPixelBufferLockBaseAddress(pixelBuffer, kCVPixelBufferLock_ReadOnly);
51 ASSERT(result == kCVReturnSuccess);
52 if (result != kCVReturnSuccess) {
53 RELEASE_LOG_ERROR(Media, "transferBGRAPixelBufferToIOSurface CVPixelBufferLockBaseAddress() returned error code %d", result);
54 return nullptr;
55 }
56
57 IntSize size { static_cast<int>(CVPixelBufferGetWidth(pixelBuffer)), static_cast<int>(CVPixelBufferGetHeight(pixelBuffer)) };
58 auto ioSurface = IOSurface::create(size, DestinationColorSpace::SRGB(), IOSurface::Format::BGRA);
59
60 IOSurface::Locker lock(*ioSurface);
61 vImage_Buffer src;
62 src.width = size.width();
63 src.height = size.height();
64 src.rowBytes = CVPixelBufferGetBytesPerRow(pixelBuffer);
65 src.data = CVPixelBufferGetBaseAddress(pixelBuffer);
66
67 vImage_Buffer dest;
68 dest.width = size.width();
69 dest.height = size.height();
70 dest.rowBytes = ioSurface->bytesPerRow();
71 dest.data = lock.surfaceBaseAddress();
72
73 vImageUnpremultiplyData_BGRA8888(&src, &dest, kvImageNoFlags);
74
75 result = CVPixelBufferUnlockBaseAddress(pixelBuffer, kCVPixelBufferLock_ReadOnly);
76 ASSERT(result == kCVReturnSuccess);
77 if (result != kCVReturnSuccess) {
78 RELEASE_LOG_ERROR(Media, "transferBGRAPixelBufferToIOSurface CVPixelBufferUnlockBaseAddress() returned error code %d", result);
79 return nullptr;
80 }
81 return ioSurface;
82#else
83 RELEASE_LOG_ERROR(Media, "transferBGRAPixelBufferToIOSurface cannot convert to IOSurface");
84 return nullptr;
85#endif
86}

88std::unique_ptr<RemoteVideoSample> RemoteVideoSample::create(MediaSample& sample)
89{
90 ASSERT(sample.platformSample().type == PlatformSample::CMSampleBufferType);
91
92 auto imageBuffer = PAL::CMSampleBufferGetImageBuffer(sample.platformSample().sample.cmSampleBuffer);
93 if (!imageBuffer) {
94 RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create: CMSampleBufferGetImageBuffer returned nullptr");
95 return nullptr;
96 }
97
98 std::unique_ptr<IOSurface> ioSurface;
99 auto surface = CVPixelBufferGetIOSurface(imageBuffer);
100 if (!surface) {
101 // Special case for canvas data that is RGBA, not IOSurface backed.
102 auto pixelFormatType = CVPixelBufferGetPixelFormatType(imageBuffer);
103 if (pixelFormatType != kCVPixelFormatType_32BGRA) {
104 RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create does not support non IOSurface backed samples that are not BGRA");
105 return nullptr;
106 }
107
108 ioSurface = transferBGRAPixelBufferToIOSurface(imageBuffer);
109 if (!ioSurface)
110 return nullptr;
111
112 surface = ioSurface->surface();
113 }
114
115 return std::unique_ptr<RemoteVideoSample>(new RemoteVideoSample(surface, DestinationColorSpace::SRGB(), sample.presentationTime(), sample.videoRotation(), sample.videoMirrored()));
116}

118std::unique_ptr<RemoteVideoSample> RemoteVideoSample::create(CVPixelBufferRef imageBuffer, MediaTime&& presentationTime, MediaSample::VideoRotation rotation)
119{
120 auto surface = CVPixelBufferGetIOSurface(imageBuffer);
121 if (!surface) {
122 RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create: CVPixelBufferGetIOSurface returned nullptr");
123 return nullptr;
124 }
125
126 return std::unique_ptr<RemoteVideoSample>(new RemoteVideoSample(surface, DestinationColorSpace::SRGB(), WTFMove(presentationTime), rotation, false));
127}

129RemoteVideoSample::RemoteVideoSample(IOSurfaceRef surface, const DestinationColorSpace& colorSpace, MediaTime&& time, MediaSample::VideoRotation rotation, bool mirrored)
130 : m_ioSurface(WebCore::IOSurface::createFromSurface(surface, colorSpace))
131 , m_rotation(rotation)
132 , m_time(WTFMove(time))
133 , m_videoFormat(IOSurfaceGetPixelFormat(surface))
134 , m_size(IntSize(IOSurfaceGetWidth(surface), IOSurfaceGetHeight(surface)))
135 , m_mirrored(mirrored)
136{
137}
138
139IOSurfaceRef RemoteVideoSample::surface() const
140{
141 if (!m_ioSurface && m_sendRight)
142 const_cast<RemoteVideoSample*>(this)->m_ioSurface = WebCore::IOSurface::createFromSendRight(WTFMove(const_cast<RemoteVideoSample*>(this)->m_sendRight), DestinationColorSpace::SRGB());
143
144 return m_ioSurface ? m_ioSurface->surface() : nullptr;
145}
