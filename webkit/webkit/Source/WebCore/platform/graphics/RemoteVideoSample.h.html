36typedef struct __CVBuffer* CVPixelBufferRef;
37
38namespace WebCore {
39
40class RemoteVideoSample {
41public:
42 RemoteVideoSample() = default;
43 RemoteVideoSample(RemoteVideoSample&&) = default;
44 RemoteVideoSample& operator=(RemoteVideoSample&&) = default;
45 ~RemoteVideoSample() = default;
46
47 WEBCORE_EXPORT static std::unique_ptr<RemoteVideoSample> create(MediaSample&);
48 WEBCORE_EXPORT static std::unique_ptr<RemoteVideoSample> create(CVPixelBufferRef, MediaTime&& presentationTime, MediaSample::VideoRotation = MediaSample::VideoRotation::None);
49 WEBCORE_EXPORT IOSurfaceRef surface() const;
50
51 const MediaTime& time() const { return m_time; }
52 uint32_t videoFormat() const { return m_videoFormat; }
53 IntSize size() const { return m_size; }
54 MediaSample::VideoRotation rotation() const { return m_rotation; }
55 bool mirrored() const { return m_mirrored; }
56
57 template<class Encoder> void encode(Encoder& encoder) const
58 {
59 if (m_ioSurface)
60 encoder << m_ioSurface->createSendRight();
61 else
62 encoder << WTF::MachSendRight();
63 encoder << m_rotation;
64 encoder << m_time;
65 encoder << m_videoFormat;
66 encoder << m_size;
67 encoder << m_mirrored;
68 }
69
70 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder& decoder, RemoteVideoSample& sample)
71 {
72 MachSendRight sendRight;
73 if (!decoder.decode(sendRight))
74 return false;
75 sample.m_sendRight = WTFMove(sendRight);
76
77 MediaSample::VideoRotation rotation;
78 if (!decoder.decode(rotation))
79 return false;
80 sample.m_rotation = rotation;
81
82 MediaTime time;
83 if (!decoder.decode(time))
84 return false;
85 sample.m_time = WTFMove(time);
86
87 uint32_t format;
88 if (!decoder.decode(format))
89 return false;
90 sample.m_videoFormat = format;
91
92 IntSize size;
93 if (!decoder.decode(size))
94 return false;
95 sample.m_size = WTFMove(size);
96
97 bool mirrored;
98 if (!decoder.decode(mirrored))
99 return false;
100 sample.m_mirrored = mirrored;
101
102 return true;
103 }
104
105private:
106 RemoteVideoSample(IOSurfaceRef, const DestinationColorSpace&, MediaTime&&, MediaSample::VideoRotation, bool);
107
108 std::unique_ptr<WebCore::IOSurface> m_ioSurface;
109 WTF::MachSendRight m_sendRight;
110 MediaSample::VideoRotation m_rotation { MediaSample::VideoRotation::None };
111 MediaTime m_time;
112 uint32_t m_videoFormat { 0 };
113 IntSize m_size;
114 bool m_mirrored { false };
115};
116
117}
