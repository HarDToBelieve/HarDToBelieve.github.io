54static inline int roundUpToMultipleOf32(int d)
55{
56 return (1 + (d >> 5)) << 5;

59// ShadowBlur needs a scratch image as the buffer for the blur filter.
60// Instead of creating and destroying the buffer for every operation,
61// we create a buffer which will be automatically purged via a timer.
62class ScratchBuffer {
63 WTF_MAKE_FAST_ALLOCATED;
64public:
65 ScratchBuffer()
66 : m_purgeTimer(RunLoop::main(), this, &ScratchBuffer::purgeTimerFired)
67 {
68 }
69
70 RefPtr<ImageBuffer> getScratchBuffer(const IntSize& size) WTF_REQUIRES_LOCK(lock())
71 {
72 ASSERT(lock().isHeld());
73 auto releaseLayerImage = makeScopeExit([this] {
74 if (m_imageBuffer)
75 scheduleScratchBufferPurge();
76 });
77
78 // We do not need to recreate the buffer if the current buffer is large enough.
79 if (m_imageBuffer && m_imageBuffer->logicalSize().width() >= size.width() && m_imageBuffer->logicalSize().height() >= size.height())
80 return m_imageBuffer;
81
82 // Round to the nearest 32 pixels so we do not grow the buffer for similar sized requests.
83 IntSize roundedSize(roundUpToMultipleOf32(size.width()), roundUpToMultipleOf32(size.height()));
84
85 clearScratchBuffer();
86
87 // ShadowBlur is not used with accelerated drawing, so it's OK to make an unconditionally unaccelerated buffer.
88 m_imageBuffer = ImageBuffer::create(roundedSize, RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
89 return m_imageBuffer;
90 }

92 bool setCachedShadowValues(const FloatSize& radius, const Color& color, const FloatRect& shadowRect, const FloatRoundedRect::Radii& radii, const FloatSize& layerSize) WTF_REQUIRES_LOCK(lock())
93 {
94 ASSERT(lock().isHeld());
95 if (!m_lastWasInset && m_lastRadius == radius && m_lastColor == color && m_lastShadowRect == shadowRect && m_lastRadii == radii && m_lastLayerSize == layerSize)
96 return false;
97
98 m_lastWasInset = false;
99 m_lastRadius = radius;
100 m_lastColor = color;
101 m_lastShadowRect = shadowRect;
102 m_lastRadii = radii;
103 m_lastLayerSize = layerSize;
104
105 return true;
106 }

221void ShadowBlur::updateShadowBlurValues()
222{
223 // Limit blur radius to 128 to avoid lots of very expensive blurring.
224 m_blurRadius = m_blurRadius.shrunkTo(FloatSize(128, 128));
225
226 // The type of shadow is decided by the blur radius, shadow offset, and shadow color.
227 if (!m_color.isVisible()) {
228 // Can't paint the shadow with invalid or invisible color.
229 m_type = NoShadow;
230 } else if (m_blurRadius.width() > 0 || m_blurRadius.height() > 0) {
231 // Shadow is always blurred, even the offset is zero.
232 m_type = BlurShadow;
233 } else if (!m_offset.width() && !m_offset.height()) {
234 // Without blur and zero offset means the shadow is fully hidden.
235 m_type = NoShadow;
236 } else
237 m_type = SolidShadow;
238}
239
240// Instead of integer division, we use 17.15 for fixed-point division.
241static const int blurSumShift = 15;
242
243// Takes a two dimensional array with three rows and two columns for the lobes.
244static void calculateLobes(int lobes[][2], float blurRadius, bool shadowsIgnoreTransforms)
245{
246 int diameter;
247 if (shadowsIgnoreTransforms)
248 diameter = std::max(2, static_cast<int>(floorf((2 / 3.f) * blurRadius))); // Canvas shadow. FIXME: we should adjust the blur radius higher up.
249 else {
250 // http://dev.w3.org/csswg/css3-background/#box-shadow
251 // Approximate a Gaussian blur with a standard deviation equal to half the blur radius,
252 // which http://www.w3.org/TR/SVG/filters.html#feGaussianBlurElement tell us how to do.
253 // However, shadows rendered according to that spec will extend a little further than m_blurRadius,
254 // so we apply a fudge factor to bring the radius down slightly.
255 float stdDev = blurRadius / 2;
256 const float gaussianKernelFactor = 3 / 4.f * sqrtf(2 * piFloat);
257 const float fudgeFactor = 0.88f;
258 diameter = std::max(2, static_cast<int>(floorf(stdDev * gaussianKernelFactor * fudgeFactor + 0.5f)));
259 }
260
261 if (diameter & 1) {
262 // if d is odd, use three box-blurs of size 'd', centered on the output pixel.
263 int lobeSize = (diameter - 1) / 2;
264 lobes[0][LeftLobe] = lobeSize;
265 lobes[0][RightLobe] = lobeSize;
266 lobes[1][LeftLobe] = lobeSize;
267 lobes[1][RightLobe] = lobeSize;
268 lobes[2][LeftLobe] = lobeSize;
269 lobes[2][RightLobe] = lobeSize;
270 } else {
271 // if d is even, two box-blurs of size 'd' (the first one centered on the pixel boundary
272 // between the output pixel and the one to the left, the second one centered on the pixel
273 // boundary between the output pixel and the one to the right) and one box blur of size 'd+1' centered on the output pixel
274 int lobeSize = diameter / 2;
275 lobes[0][LeftLobe] = lobeSize;
276 lobes[0][RightLobe] = lobeSize - 1;
277 lobes[1][LeftLobe] = lobeSize - 1;
278 lobes[1][RightLobe] = lobeSize;
279 lobes[2][LeftLobe] = lobeSize;
280 lobes[2][RightLobe] = lobeSize;
281 }
282}

292void ShadowBlur::blurLayerImage(unsigned char* imageData, const IntSize& size, int rowStride)
293{
294 const int channels[4] = { 3, 0, 1, 3 };
295
296 int lobes[3][2]; // indexed by pass, and left/right lobe
297 calculateLobes(lobes, m_blurRadius.width(), m_shadowsIgnoreTransforms);
298
299 // First pass is horizontal.
300 int stride = 4;
301 int delta = rowStride;
302 int final = size.height();
303 int dim = size.width();
304
305 // Two stages: horizontal and vertical
306 for (int pass = 0; pass < 2; ++pass) {
307 unsigned char* pixels = imageData;
308
309 if (!pass && !m_blurRadius.width())
310 final = 0; // Do no work if horizonal blur is zero.
311
312 for (int j = 0; j < final; ++j, pixels += delta) {
313 // For each step, we blur the alpha in a channel and store the result
314 // in another channel for the subsequent step.
315 // We use sliding window algorithm to accumulate the alpha values.
316 // This is much more efficient than computing the sum of each pixels
317 // covered by the box kernel size for each x.
318 for (int step = 0; step < 3; ++step) {
319 int side1 = lobes[step][LeftLobe];
320 int side2 = lobes[step][RightLobe];
321 int pixelCount = side1 + 1 + side2;
322 int invCount = ((1 << blurSumShift) + pixelCount - 1) / pixelCount;
323 int ofs = 1 + side2;
324 int alpha1 = pixels[channels[step]];
325 int alpha2 = pixels[(dim - 1) * stride + channels[step]];
326
327 unsigned char* ptr = pixels + channels[step + 1];
328 unsigned char* prev = pixels + stride + channels[step];
329 unsigned char* next = pixels + ofs * stride + channels[step];
330
331 int i;
332 int sum = side1 * alpha1 + alpha1;
333 int limit = (dim < side2 + 1) ? dim : side2 + 1;
334
335 for (i = 1; i < limit; ++i, prev += stride)
336 sum += *prev;
337
338 if (limit <= side2)
339 sum += (side2 - limit + 1) * alpha2;
340
341 limit = (side1 < dim) ? side1 : dim;
342 for (i = 0; i < limit; ptr += stride, next += stride, ++i, ++ofs) {
343 *ptr = (sum * invCount) >> blurSumShift;
344 sum += ((ofs < dim) ? *next : alpha2) - alpha1;
345 }
346
347 prev = pixels + channels[step];
348 for (; ofs < dim; ptr += stride, prev += stride, next += stride, ++i, ++ofs) {
349 *ptr = (sum * invCount) >> blurSumShift;
350 sum += (*next) - (*prev);
351 }
352
353 for (; i < dim; ptr += stride, prev += stride, ++i) {
354 *ptr = (sum * invCount) >> blurSumShift;
355 sum += alpha2 - (*prev);
356 }
357 }
358 }
359
360 // Last pass is vertical.
361 stride = rowStride;
362 delta = 4;
363 final = size.width();
364 dim = size.height();
365
366 if (!m_blurRadius.height())
367 break;
368
369 if (m_blurRadius.width() != m_blurRadius.height())
370 calculateLobes(lobes, m_blurRadius.height(), m_shadowsIgnoreTransforms);
371 }
372}

374void ShadowBlur::adjustBlurRadius(const AffineTransform& transform)
375{
376 if (m_shadowsIgnoreTransforms)
377 m_blurRadius.scale(1 / static_cast<float>(transform.xScale()), 1 / static_cast<float>(transform.yScale()));
378}

380IntSize ShadowBlur::blurredEdgeSize() const
381{
382 IntSize edgeSize = expandedIntSize(m_blurRadius);
383
384 // To avoid slowing down blurLayerImage() for radius == 1, we give it two empty pixels on each side.
385 if (edgeSize.width() == 1)
386 edgeSize.setWidth(2);
387
388 if (edgeSize.height() == 1)
389 edgeSize.setHeight(2);
390
391 return edgeSize;
392}

394std::optional<ShadowBlur::LayerImageProperties> ShadowBlur::calculateLayerBoundingRect(const AffineTransform& transform, const FloatRect& shadowedRect, const IntRect& clipRect)
395{
396 LayerImageProperties calculatedLayerImageProperties;
397
398 IntSize edgeSize = blurredEdgeSize();
399
400 // Calculate the destination of the blurred and/or transformed layer.
401 FloatRect layerRect;
402 IntSize inflation;
403
404 if (m_shadowsIgnoreTransforms && !transform.isIdentity()) {
405 FloatQuad transformedPolygon = transform.mapQuad(FloatQuad(shadowedRect));
406 transformedPolygon.move(m_offset);
407 layerRect = transform.inverse().value_or(AffineTransform()).mapQuad(transformedPolygon).boundingBox();
408 } else {
409 layerRect = shadowedRect;
410 layerRect.move(m_offset);
411 }
412
413 // We expand the area by the blur radius to give extra space for the blur transition.
414 if (m_type == BlurShadow) {
415 layerRect.inflateX(edgeSize.width());
416 layerRect.inflateY(edgeSize.height());
417 inflation = edgeSize;
418 }
419
420 FloatRect unclippedLayerRect = layerRect;
421
422 if (!clipRect.contains(enclosingIntRect(layerRect))) {
423 // If we are totally outside the clip region, we aren't painting at all.
424 if (intersection(layerRect, clipRect).isEmpty())
425 return std::nullopt;
426
427 IntRect inflatedClip = clipRect;
428 // Pixels at the edges can be affected by pixels outside the buffer,
429 // so intersect with the clip inflated by the blur.
430 if (m_type == BlurShadow) {
431 inflatedClip.inflateX(edgeSize.width());
432 inflatedClip.inflateY(edgeSize.height());
433 } else {
434 // Enlarge the clipping area 1 pixel so that the fill does not
435 // bleed (due to antialiasing) even if the unaligned clip rect occurred
436 inflatedClip.inflateX(1);
437 inflatedClip.inflateY(1);
438 }
439
440 layerRect.intersect(inflatedClip);
441 }
442
443 IntSize frameSize = inflation;
444 frameSize.scale(2);
445 calculatedLayerImageProperties.shadowedResultSize = FloatSize(shadowedRect.width() + frameSize.width(), shadowedRect.height() + frameSize.height());
446 calculatedLayerImageProperties.layerOrigin = FloatPoint(layerRect.x(), layerRect.y());
447 calculatedLayerImageProperties.layerSize = layerRect.size();
448
449 const FloatPoint unclippedLayerOrigin = FloatPoint(unclippedLayerRect.x(), unclippedLayerRect.y());
450 const FloatSize clippedOut = unclippedLayerOrigin - calculatedLayerImageProperties.layerOrigin;
451
452 // Set the origin as the top left corner of the scratch image, or, in case there's a clipped
453 // out region, set the origin accordingly to the full bounding rect's top-left corner.
454 float translationX = -shadowedRect.x() + inflation.width() - fabsf(clippedOut.width());
455 float translationY = -shadowedRect.y() + inflation.height() - fabsf(clippedOut.height());
456 calculatedLayerImageProperties.layerContextTranslation = FloatSize(translationX, translationY);
457
458 return calculatedLayerImageProperties;
459}

461void ShadowBlur::drawShadowBuffer(GraphicsContext& graphicsContext, ImageBuffer& layerImage, const FloatPoint& layerOrigin, const FloatSize& layerSize)
462{
463 GraphicsContextStateSaver stateSaver(graphicsContext);
464
465 IntSize bufferSize = layerImage.backendSize();
466 if (bufferSize != layerSize) {
467 // The rect passed to clipToImageBuffer() has to be the size of the entire buffer,
468 // but we may not have cleared it all, so clip to the filled part first.
469 graphicsContext.clip(FloatRect(layerOrigin, layerSize));
470 }
471 graphicsContext.clipToImageBuffer(layerImage, FloatRect(layerOrigin, bufferSize));
472 graphicsContext.setFillColor(m_color);
473
474 graphicsContext.clearShadow();
475 graphicsContext.fillRect(FloatRect(layerOrigin, layerSize));
476}

478static void computeSliceSizesFromRadii(const IntSize& twiceRadius, const FloatRoundedRect::Radii& radii, int& leftSlice, int& rightSlice, int& topSlice, int& bottomSlice)
479{
480 leftSlice = twiceRadius.width() + std::max(radii.topLeft().width(), radii.bottomLeft().width());
481 rightSlice = twiceRadius.width() + std::max(radii.topRight().width(), radii.bottomRight().width());
482
483 topSlice = twiceRadius.height() + std::max(radii.topLeft().height(), radii.topRight().height());
484 bottomSlice = twiceRadius.height() + std::max(radii.bottomLeft().height(), radii.bottomRight().height());
485}

487IntSize ShadowBlur::templateSize(const IntSize& radiusPadding, const FloatRoundedRect::Radii& radii) const
488{
489 const int templateSideLength = 1;
490
491 int leftSlice;
492 int rightSlice;
493 int topSlice;
494 int bottomSlice;
495
496 IntSize blurExpansion = radiusPadding;
497 blurExpansion.scale(2);
498
499 computeSliceSizesFromRadii(blurExpansion, radii, leftSlice, rightSlice, topSlice, bottomSlice);
500
501 return IntSize(templateSideLength + leftSlice + rightSlice, templateSideLength + topSlice + bottomSlice);
502}

549void ShadowBlur::drawRectShadow(const AffineTransform& transform, const IntRect& clipBounds, const FloatRoundedRect& shadowedRect, const DrawBufferCallback& drawBuffer, const DrawImageCallback& drawImage, const FillRectCallback& fillRect)
550{
551 auto layerImageProperties = calculateLayerBoundingRect(transform, shadowedRect.rect(), clipBounds);
552 if (!layerImageProperties)
553 return;
554
555 adjustBlurRadius(transform);
556
557 bool canUseTilingTechnique = true;
558
559 // drawRectShadowWithTiling does not work with rotations.
560 // https://bugs.webkit.org/show_bug.cgi?id=45042
561 if (!transform.preservesAxisAlignment() || m_type != BlurShadow)
562 canUseTilingTechnique = false;
563
564 IntSize edgeSize = blurredEdgeSize();
565 IntSize templateSize = this->templateSize(edgeSize, shadowedRect.radii());
566 const FloatRect& rect = shadowedRect.rect();
567
568 if (templateSize.width() > rect.width() || templateSize.height() > rect.height()
569 || (templateSize.unclampedArea() > layerImageProperties->shadowedResultSize.area()))
570 canUseTilingTechnique = false;
571
572 if (canUseTilingTechnique)
573 drawRectShadowWithTiling(transform, shadowedRect, templateSize, edgeSize, drawImage, fillRect, *layerImageProperties);
574 else
575 drawRectShadowWithoutTiling(transform, shadowedRect, *layerImageProperties, drawBuffer);
576}

578void ShadowBlur::drawInsetShadow(const AffineTransform& transform, const IntRect& clipBounds, const FloatRect& fullRect, const FloatRoundedRect& holeRect, const DrawBufferCallback& drawBuffer, const DrawImageCallback& drawImage, const FillRectWithHoleCallback& fillRectWithHole)
579{
580 auto layerImageProperties = calculateLayerBoundingRect(transform, fullRect, clipBounds);
581 if (!layerImageProperties)
582 return;
583
584 adjustBlurRadius(transform);
585
586 bool canUseTilingTechnique = true;
587
588 // drawRectShadowWithTiling does not work with rotations.
589 // https://bugs.webkit.org/show_bug.cgi?id=45042
590 if (!transform.preservesAxisAlignment() || m_type != BlurShadow)
591 canUseTilingTechnique = false;
592
593 IntSize edgeSize = blurredEdgeSize();
594 IntSize templateSize = this->templateSize(edgeSize, holeRect.radii());
595 const FloatRect& hRect = holeRect.rect();
596
597 if (templateSize.width() > hRect.width() || templateSize.height() > hRect.height()
598 || (templateSize.width() * templateSize.height() > hRect.width() * hRect.height()))
599 canUseTilingTechnique = false;
600
601 if (canUseTilingTechnique)
602 drawInsetShadowWithTiling(transform, fullRect, holeRect, templateSize, edgeSize, drawImage, fillRectWithHole);
603 else
604 drawInsetShadowWithoutTiling(transform, fullRect, holeRect, *layerImageProperties, drawBuffer);
605}

607void ShadowBlur::drawRectShadowWithoutTiling(const AffineTransform&, const FloatRoundedRect& shadowedRect, const LayerImageProperties& layerImageProperties, const DrawBufferCallback& drawBuffer)
608{
609 auto layerImage = ImageBuffer::create(expandedIntSize(layerImageProperties.layerSize), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
610 if (!layerImage)
611 return;
612
613 GraphicsContext& shadowContext = layerImage->context();
614 GraphicsContextStateSaver stateSaver(shadowContext);
615 shadowContext.setFillColor(Color::black);
616
617 {
618 GraphicsContext& shadowContext = layerImage->context();
619 GraphicsContextStateSaver stateSaver(shadowContext);
620 shadowContext.translate(layerImageProperties.layerContextTranslation);
621 shadowContext.setFillColor(Color::black);
622 if (shadowedRect.radii().isZero())
623 shadowContext.fillRect(shadowedRect.rect());
624 else {
625 Path path;
626 path.addRoundedRect(shadowedRect);
627 shadowContext.fillPath(path);
628 }
629
630 blurShadowBuffer(*layerImage, expandedIntSize(layerImageProperties.layerSize));
631 }
632 drawBuffer(*layerImage, layerImageProperties.layerOrigin, layerImageProperties.layerSize);
633}

663/*
664 These functions use tiling to improve the performance of the shadow
665 drawing of rounded rectangles. The code basically does the following
666 steps:
667
668 1. Calculate the size of the shadow template, a rectangle that
669 contains all the necessary tiles to draw the complete shadow.
670
671 2. If that size is smaller than the real rectangle render the new
672 template rectangle and its shadow in a new surface, in other case
673 render the shadow of the real rectangle in the destination
674 surface.
675
676 3. Calculate the sizes and positions of the tiles and their
677 destinations and use drawPattern to render the final shadow. The
678 code divides the rendering in 8 tiles:
679
680 1 | 2 | 3
681 -----------
682 4 | | 5
683 -----------
684 6 | 7 | 8
685
686 The corners are directly copied from the template rectangle to the
687 real one and the side tiles are 1 pixel width, we use them as
688 tiles to cover the destination side. The corner tiles are bigger
689 than just the side of the rounded corner, we need to increase it
690 because the modifications caused by the corner over the blur
691 effect. We fill the central or outer part with solid color to complete
692 the shadow.
693 */
694

696{
697 FloatRect templateShadow = FloatRect(edgeSize.width(), edgeSize.height(), templateSize.width() - 2 * edgeSize.width(), templateSize.height() - 2 * edgeSize.height());
698
699#if USE(CG)
700 if (ScratchBuffer::lock().tryLock()) {
701 Locker locker { AdoptLock, ScratchBuffer::lock() };
702 auto layerImageBuffer = ScratchBuffer::singleton().getScratchBuffer(templateSize);
703 if (!layerImageBuffer)
704 return;
705 bool redrawNeeded = ScratchBuffer::singleton().setCachedShadowValues(m_blurRadius, m_color, templateShadow, shadowedRect.radii(), layerImageProperties.layerSize);

707 return;
708 }
709#else
710 UNUSED_PARAM(layerImageProperties);
711#endif
712
713 if (auto layerImageBuffer = ImageBuffer::create(templateSize, RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8))
714 drawRectShadowWithTilingWithLayerImageBuffer(*layerImageBuffer, transform, shadowedRect, templateSize, edgeSize, drawImage, fillRect, templateShadow);
715}

811void ShadowBlur::drawLayerPieces(ImageBuffer& layerImage, const FloatRect& shadowBounds, const FloatRoundedRect::Radii& radii, const IntSize& bufferPadding, const IntSize& templateSize, const DrawImageCallback& drawImage)
812{
813 const IntSize twiceRadius = IntSize(bufferPadding.width() * 2, bufferPadding.height() * 2);
814
815 int leftSlice;
816 int rightSlice;
817 int topSlice;
818 int bottomSlice;
819 computeSliceSizesFromRadii(twiceRadius, radii, leftSlice, rightSlice, topSlice, bottomSlice);
820
821 int centerWidth = shadowBounds.width() - leftSlice - rightSlice;
822 int centerHeight = shadowBounds.height() - topSlice - bottomSlice;
823 FloatRect centerRect(shadowBounds.x() + leftSlice, shadowBounds.y() + topSlice, centerWidth, centerHeight);
824
825 // Top side.
826 FloatRect tileRect = FloatRect(leftSlice, 0, templateSideLength, topSlice);
827 FloatRect destRect = FloatRect(centerRect.x(), centerRect.y() - topSlice, centerRect.width(), topSlice);
828 drawImage(layerImage, destRect, tileRect);
829
830 // Draw the bottom side.
831 tileRect.setY(templateSize.height() - bottomSlice);
832 tileRect.setHeight(bottomSlice);
833 destRect.setY(centerRect.maxY());
834 destRect.setHeight(bottomSlice);
835 drawImage(layerImage, destRect, tileRect);
836
837 // Left side.
838 tileRect = FloatRect(0, topSlice, leftSlice, templateSideLength);
839 destRect = FloatRect(centerRect.x() - leftSlice, centerRect.y(), leftSlice, centerRect.height());
840 drawImage(layerImage, destRect, tileRect);
841
842 // Right side.
843 tileRect.setX(templateSize.width() - rightSlice);
844 tileRect.setWidth(rightSlice);
845 destRect.setX(centerRect.maxX());
846 destRect.setWidth(rightSlice);
847 drawImage(layerImage, destRect, tileRect);
848
849 // Top left corner.
850 tileRect = FloatRect(0, 0, leftSlice, topSlice);
851 destRect = FloatRect(centerRect.x() - leftSlice, centerRect.y() - topSlice, leftSlice, topSlice);
852 drawImage(layerImage, destRect, tileRect);
853
854 // Top right corner.
855 tileRect = FloatRect(templateSize.width() - rightSlice, 0, rightSlice, topSlice);
856 destRect = FloatRect(centerRect.maxX(), centerRect.y() - topSlice, rightSlice, topSlice);
857 drawImage(layerImage, destRect, tileRect);
858
859 // Bottom right corner.
860 tileRect = FloatRect(templateSize.width() - rightSlice, templateSize.height() - bottomSlice, rightSlice, bottomSlice);
861 destRect = FloatRect(centerRect.maxX(), centerRect.maxY(), rightSlice, bottomSlice);
862 drawImage(layerImage, destRect, tileRect);
863
864 // Bottom left corner.
865 tileRect = FloatRect(0, templateSize.height() - bottomSlice, leftSlice, bottomSlice);
866 destRect = FloatRect(centerRect.x() - leftSlice, centerRect.maxY(), leftSlice, bottomSlice);
867 drawImage(layerImage, destRect, tileRect);
868}
