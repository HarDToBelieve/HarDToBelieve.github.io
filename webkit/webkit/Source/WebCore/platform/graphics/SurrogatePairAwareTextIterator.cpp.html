38bool SurrogatePairAwareTextIterator::consumeSlowCase(UChar32& character, unsigned& clusterLength)
39{
40 ASSERT(U16_IS_SURROGATE(character));
41
42 // If we have a surrogate pair, make sure it starts with the high part.
43 if (!U16_IS_SURROGATE_LEAD(character))
44 return false;
45
46 // Do we have a surrogate pair? If so, determine the full Unicode (32 bit) code point before glyph lookup.
47 // Make sure we have another character and it's a low surrogate.
48 if (m_currentIndex + 1 >= m_endIndex)
49 return false;
50
51 UChar low = m_characters[1];
52 if (!U16_IS_TRAIL(low))
53 return false;
54
55 character = U16_GET_SUPPLEMENTARY(character, low);
56 clusterLength = 2;
57 return true;
58}

60UChar32 SurrogatePairAwareTextIterator::normalizeVoicingMarks()
61{
62 // According to http://www.unicode.org/Public/UNIDATA/UCD.html#Canonical_Combining_Class_Values
63 static constexpr uint8_t hiraganaKatakanaVoicingMarksCombiningClass = 8;
64
65 if (m_currentIndex + 1 >= m_endIndex)
66 return 0;
67
68 if (u_getCombiningClass(m_characters[1]) == hiraganaKatakanaVoicingMarksCombiningClass) {
69 UErrorCode status = U_ZERO_ERROR;
70 const UNormalizer2* normalizer = unorm2_getNFCInstance(&status);
71 ASSERT(U_SUCCESS(status));
72 auto composedCharacter = unorm2_composePair(normalizer, m_characters[0], m_characters[1]);
73 if (composedCharacter > 0)
74 return composedCharacter;
75 }
76
77 return 0;
78}
