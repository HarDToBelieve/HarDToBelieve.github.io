34static bool isValidVPLevel(uint8_t level)
35{
36 constexpr uint8_t validLevels[] = {
37 VPConfigurationLevel::Level_1,
38 VPConfigurationLevel::Level_1_1,
39 VPConfigurationLevel::Level_2,
40 VPConfigurationLevel::Level_2_1,
41 VPConfigurationLevel::Level_3,
42 VPConfigurationLevel::Level_3_1,
43 VPConfigurationLevel::Level_4,
44 VPConfigurationLevel::Level_4_1,
45 VPConfigurationLevel::Level_5,
46 VPConfigurationLevel::Level_5_1,
47 VPConfigurationLevel::Level_5_2,
48 VPConfigurationLevel::Level_6,
49 VPConfigurationLevel::Level_6_1,
50 VPConfigurationLevel::Level_6_2,
51 };
52
53 ASSERT(std::is_sorted(std::begin(validLevels), std::end(validLevels)));
54 return std::binary_search(std::begin(validLevels), std::end(validLevels), level);
55}

125std::optional<VPCodecConfigurationRecord> parseVPCodecParameters(StringView codecView)
126{
127 auto codecSplit = codecView.split('.');
128 auto nextElement = codecSplit.begin();
129 if (nextElement == codecSplit.end())
130 return std::nullopt;
131
132 VPCodecConfigurationRecord configuration;
133
134 configuration.codecName = (*nextElement).toString();
135 ++nextElement;
136
137 // Support the legacy identifiers (with no parameters) for VP8 and VP9.
138 if ((configuration.codecName == "vp8" || configuration.codecName == "vp9") && nextElement == codecSplit.end())
139 return configuration;
140
141 // The format of the 'vp09' codec string is specified in the webm GitHub repo:
142 // <https://github.com/webmproject/vp9-dash/blob/master/VPCodecISOMediaFileFormatBinding.md#codecs-parameter-string>
143
144 // "sample entry 4CC, profile, level, and bitDepth are all mandatory fields. If any of these fields are empty, or not
145 // within their allowed range, the processing device SHALL treat it as an error."
146
147 // Codec identifier: legal values are 'vp08' or 'vp09'.
148 if (configuration.codecName != "vp08" && configuration.codecName != "vp09")
149 return std::nullopt;
150
151 if (nextElement == codecSplit.end())
152 return std::nullopt;
153
154 // First element: profile. Legal values are 0-3.
155 auto profile = parseInteger<uint8_t>(*nextElement);
156 if (!profile || *profile > 3)
157 return std::nullopt;
158 configuration.profile = *profile;
159
160 if (++nextElement == codecSplit.end())
161 return std::nullopt;
162
163 // Second element: level. Legal values are enumerated in validVPLevels above.
164 auto level = parseInteger<uint8_t>(*nextElement);
165 if (!level || !isValidVPLevel(*level))
166 return std::nullopt;
167 configuration.level = *level;
168
169 if (++nextElement == codecSplit.end())
170 return std::nullopt;
171
172 // Third element: bitDepth. Legal values are 8, 10, and 12.
173 auto bitDepth = parseInteger<uint8_t>(*nextElement);
174 if (!bitDepth || (*bitDepth != 8 && *bitDepth != 10 && *bitDepth != 12))
175 return std::nullopt;
176 configuration.bitDepth = *bitDepth;
177
178 // "colorPrimaries, transferCharacteristics, matrixCoefficients, videoFullRangeFlag, and chromaSubsampling are OPTIONAL,
179 // mutually inclusive (all or none) fields. If not specified then the processing device MUST use the values listed in the
180 // table below as defaults when deciding if the device is able to decode and potentially render the video."
181
182 if (++nextElement == codecSplit.end())
183 return configuration;
184
185 // Fourth element: chromaSubsampling. Legal values are 0-3.
186 auto chromaSubsampling = parseInteger<uint8_t>(*nextElement);
187 if (!chromaSubsampling || *chromaSubsampling > VPConfigurationChromaSubsampling::Subsampling_444)
188 return std::nullopt;
189 configuration.chromaSubsampling = *chromaSubsampling;
190
191 if (++nextElement == codecSplit.end())
192 return std::nullopt;
193
194 // Fifth element: colorPrimaries. Legal values are defined by ISO/IEC 23001-8:2016, superseded
195 // by ISO/IEC 23091-2:2019.
196 auto colorPrimaries = parseInteger<uint8_t>(*nextElement);
197 if (!colorPrimaries || !isValidVPColorPrimaries(*colorPrimaries))
198 return std::nullopt;
199 configuration.colorPrimaries = *colorPrimaries;
200
201 if (++nextElement == codecSplit.end())
202 return std::nullopt;
203
204 // Sixth element: transferCharacteristics. Legal values are defined by ISO/IEC 23001-8:2016, superseded
205 // by ISO/IEC 23091-2:2019.
206 auto transferCharacteristics = parseInteger<uint8_t>(*nextElement);
207 if (!transferCharacteristics || !isValidVPTransferCharacteristics(*transferCharacteristics))
208 return std::nullopt;
209 configuration.transferCharacteristics = *transferCharacteristics;
210
211 if (++nextElement == codecSplit.end())
212 return std::nullopt;
213
214 // Seventh element: matrixCoefficients. Legal values are defined by ISO/IEC 23001-8:2016, superseded
215 // by ISO/IEC 23091-2:2019.
216 auto matrixCoefficients = parseInteger<uint8_t>(*nextElement);
217 if (!matrixCoefficients || !isValidVPMatrixCoefficients(*matrixCoefficients))
218 return std::nullopt;
219 configuration.matrixCoefficients = *matrixCoefficients;
220
221 // "If matrixCoefficients is 0 (RGB), then chroma subsampling MUST be 3 (4:4:4)."
222 if (!configuration.matrixCoefficients && configuration.chromaSubsampling != 3)
223 return std::nullopt;
224
225 if (++nextElement == codecSplit.end())
226 return std::nullopt;
227
228 // Eighth element: videoFullRangeFlag. Legal values are 0 and 1.
229 auto videoFullRangeFlag = parseInteger<uint8_t>(*nextElement);
230 if (!videoFullRangeFlag || *videoFullRangeFlag > 1)
231 return std::nullopt;
232 configuration.videoFullRangeFlag = *videoFullRangeFlag;
233
234 if (++nextElement != codecSplit.end())
235 return std::nullopt;
236
237 return configuration;
238}
