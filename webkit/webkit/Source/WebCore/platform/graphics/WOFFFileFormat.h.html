26#ifndef WOFFFileFormat_h
27#define WOFFFileFormat_h
28
29#include <wtf/Forward.h>
30
31namespace WebCore {
32
33class SharedBuffer;
34
35// Returns whether the buffer is a WOFF file.
36bool isWOFF(SharedBuffer&);
37
38// Returns false if the WOFF file woff is invalid or could not be converted to sfnt (for example,
39// if conversion ran out of memory). Otherwise returns true and writes the sfnt payload into sfnt.
40bool convertWOFFToSfnt(SharedBuffer& woff, Vector<uint8_t>& sfnt);
41
42// If the given buffer is a WOFF file and the current platform has no native support for WOFF
43// fonts, convert it to sfnt. Returns true if the given buffer was converted. If conversion fails,
44// the buffer will be set to nullptr.
45bool convertWOFFToSfntIfNecessary(RefPtr<SharedBuffer>&);
46
47} // namespace WebCore
48
49#endif // WOFFFileFormat_h
