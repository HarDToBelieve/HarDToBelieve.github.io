38WidthIterator::WidthIterator(const FontCascade& font, const TextRun& run, HashSet<const Font*>* fallbackFonts, bool accountForGlyphBounds, bool forTextEmphasis)
39 : m_font(font)
40 , m_run(run)
41 , m_fallbackFonts(fallbackFonts)
42 , m_expansion(run.expansion())
43 , m_isAfterExpansion((run.expansionBehavior() & LeftExpansionMask) == ForbidLeftExpansion)
44 , m_accountForGlyphBounds(accountForGlyphBounds)
45 , m_enableKerning(font.enableKerning())
46 , m_requiresShaping(font.requiresShaping())
47 , m_forTextEmphasis(forTextEmphasis)
48{
49 // FIXME: Should we clamp m_expansion so it can never be negative?
50
51 if (!m_expansion)
52 m_expansionPerOpportunity = 0;
53 else {
54 unsigned expansionOpportunityCount = FontCascade::expansionOpportunityCount(m_run.text(), m_run.ltr() ? TextDirection::LTR : TextDirection::RTL, run.expansionBehavior()).first;
55
56 if (!expansionOpportunityCount)
57 m_expansionPerOpportunity = 0;
58 else
59 m_expansionPerOpportunity = m_expansion / expansionOpportunityCount;
60 }
61}

78static inline bool isSoftBankEmoji(UChar32 codepoint)
79{
80 return codepoint >= 0xE001 && codepoint <= 0xE537;

83inline auto WidthIterator::shouldApplyFontTransforms(const GlyphBuffer& glyphBuffer, unsigned lastGlyphCount, unsigned currentCharacterIndex) const -> TransformsType
84{
85 ASSERT(currentCharacterIndex <= m_run.length());
86 if (glyphBuffer.size() == (lastGlyphCount + 1) && currentCharacterIndex && isSoftBankEmoji(m_run.text()[currentCharacterIndex - 1]))
87 return TransformsType::Forced;
88 if (m_run.length() <= 1)
89 return TransformsType::None;
90 return TransformsType::NotForced;
91}

93inline float WidthIterator::applyFontTransforms(GlyphBuffer& glyphBuffer, unsigned lastGlyphCount, unsigned currentCharacterIndex, const Font& font, bool force, CharactersTreatedAsSpace& charactersTreatedAsSpace)
94{
95 ASSERT_UNUSED(currentCharacterIndex, shouldApplyFontTransforms(glyphBuffer, lastGlyphCount, currentCharacterIndex) != WidthIterator::TransformsType::None);
96
97 auto glyphBufferSize = glyphBuffer.size();
98 if (!force && glyphBufferSize <= lastGlyphCount + 1)
99 return 0;
100
101 GlyphBufferAdvance* advances = glyphBuffer.advances(0);
102 float beforeWidth = 0;
103 for (unsigned i = lastGlyphCount; i < glyphBufferSize; ++i)
104 beforeWidth += width(advances[i]);
105
106 ASSERT(lastGlyphCount <= glyphBufferSize);
107
108 font.applyTransforms(glyphBuffer, lastGlyphCount, m_currentCharacterIndex, m_enableKerning, m_requiresShaping, m_font.fontDescription().computedLocale(), m_run.text(), m_run.direction());
109 glyphBufferSize = glyphBuffer.size();
110 advances = glyphBuffer.advances(0);
111
112 GlyphBufferOrigin* origins = glyphBuffer.origins(0);
113 for (unsigned i = lastGlyphCount; i < glyphBufferSize; ++i) {
114 setHeight(advances[i], -height(advances[i]));
115 setY(origins[i], -y(origins[i]));
116 }
117
118 for (unsigned i = lastGlyphCount; i < glyphBufferSize; ++i) {
119 auto characterIndex = glyphBuffer.uncheckedStringOffsetAt(i);
120 auto iterator = std::lower_bound(charactersTreatedAsSpace.begin(), charactersTreatedAsSpace.end(), characterIndex, [](const OriginalAdvancesForCharacterTreatedAsSpace& l, GlyphBufferStringOffset r) -> bool {
121 return l.stringOffset < r;
122 });
123 if (iterator == charactersTreatedAsSpace.end() || iterator->stringOffset != characterIndex)
124 continue;
125 const auto& originalAdvances = *iterator;
126 if (i && !originalAdvances.characterIsSpace)
127 setWidth(*glyphBuffer.advances(i - 1), originalAdvances.advanceBeforeCharacter);
128 setWidth(*glyphBuffer.advances(i), originalAdvances.advanceAtCharacter);
129 }
130 charactersTreatedAsSpace.clear();
131
132 float afterWidth = 0;
133 for (unsigned i = lastGlyphCount; i < glyphBufferSize; ++i)
134 afterWidth += width(advances[i]);
135
136 return afterWidth - beforeWidth;
137}

139static inline std::pair<bool, bool> expansionLocation(bool ideograph, bool treatAsSpace, bool ltr, bool isAfterExpansion, bool forbidLeftExpansion, bool forbidRightExpansion, bool forceLeftExpansion, bool forceRightExpansion)
140{
141 bool expandLeft = ideograph;
142 bool expandRight = ideograph;
143 if (treatAsSpace) {
144 if (ltr)
145 expandRight = true;
146 else
147 expandLeft = true;
148 }
149 if (isAfterExpansion) {
150 if (ltr)
151 expandLeft = false;
152 else
153 expandRight = false;
154 }
155 ASSERT(!forbidLeftExpansion || !forceLeftExpansion);
156 ASSERT(!forbidRightExpansion || !forceRightExpansion);
157 if (forbidLeftExpansion)
158 expandLeft = false;
159 if (forbidRightExpansion)
160 expandRight = false;
161 if (forceLeftExpansion)
162 expandLeft = true;
163 if (forceRightExpansion)
164 expandRight = true;
165 return std::make_pair(expandLeft, expandRight);
166}

168void WidthIterator::commitCurrentFontRange(GlyphBuffer& glyphBuffer, unsigned lastGlyphCount, unsigned currentCharacterIndex, const Font& font, const Font& primaryFont, UChar32 character, float widthOfCurrentFontRange, CharactersTreatedAsSpace& charactersTreatedAsSpace)
169{
170 auto transformsType = shouldApplyFontTransforms(glyphBuffer, lastGlyphCount, currentCharacterIndex);
171 if (transformsType != TransformsType::None)
172 m_runWidthSoFar += applyFontTransforms(glyphBuffer, lastGlyphCount, currentCharacterIndex, font, transformsType == TransformsType::Forced, charactersTreatedAsSpace);
173 m_currentCharacterIndex = currentCharacterIndex;
174
175 if (widthOfCurrentFontRange && m_fallbackFonts && &font != &primaryFont) {
176 // FIXME: This does a little extra work that could be avoided if
177 // glyphDataForCharacter() returned whether it chose to use a small caps font.
178 if (!m_font.isSmallCaps() || character == u_toupper(character))
179 m_fallbackFonts->add(&font);
180 else {
181 auto glyphFont = m_font.glyphDataForCharacter(u_toupper(character), m_run.rtl()).font;
182 if (glyphFont != &primaryFont)
183 m_fallbackFonts->add(glyphFont);
184 }
185 }

188bool WidthIterator::hasExtraSpacing() const
189{
190 return (m_font.letterSpacing() || m_font.wordSpacing() || m_expansion) && !m_run.spacingDisabled();

193template <typename TextIterator>
194inline void WidthIterator::advanceInternal(TextIterator& textIterator, GlyphBuffer& glyphBuffer)
195{
196 // The core logic here needs to match FontCascade::widthForSimpleText()
197 bool rtl = m_run.rtl();
198
199 FloatRect bounds;
200
201 const Font& primaryFont = m_font.primaryFont();
202 const Font* lastFontData = &primaryFont;
203 unsigned lastGlyphCount = glyphBuffer.size();
204
205 auto currentCharacterIndex = textIterator.currentIndex();
206 UChar32 character = 0;
207 unsigned clusterLength = 0;
208 CharactersTreatedAsSpace charactersTreatedAsSpace;
209 float widthOfCurrentFontRange = 0;
210 // We are iterating in string order, not glyph order. Compare this to ComplexTextController::adjustGlyphsAndAdvances()
211 while (textIterator.consume(character, clusterLength)) {
212 m_containsTabs |= character == tabCharacter;
213 currentCharacterIndex = textIterator.currentIndex();
214 unsigned advanceLength = clusterLength;
215 if (currentCharacterIndex + advanceLength == m_run.length())
216 m_lastCharacterIndex = currentCharacterIndex;
217 bool characterMustDrawSomething = !isDefaultIgnorableCodePoint(character);
218#if USE(FREETYPE)
219 // Freetype based ports only override the characters with Default_Ignorable unicode property when the font
220 // doesn't support the code point. We should ignore them at this point to ensure they are not displayed.
221 if (!characterMustDrawSomething) {
222 textIterator.advance(advanceLength);
223 currentCharacterIndex = textIterator.currentIndex();
224 continue;
225 }
226#endif
227 const GlyphData& glyphData = m_font.glyphDataForCharacter(character, rtl);
228 Glyph glyph = glyphData.glyph;
229 if (!glyph && !characterMustDrawSomething) {
230 textIterator.advance(advanceLength);
231 currentCharacterIndex = textIterator.currentIndex();
232 continue;
233 }
234 const Font* font = glyphData.font ? glyphData.font : &m_font.primaryFont();
235 ASSERT(font);
236
237 float width = font->widthForGlyph(glyph);
238
239 if (font != lastFontData) {
240 commitCurrentFontRange(glyphBuffer, lastGlyphCount, currentCharacterIndex, *lastFontData, primaryFont, character, widthOfCurrentFontRange, charactersTreatedAsSpace);
241 lastGlyphCount = glyphBuffer.size();
242 lastFontData = font;
243 widthOfCurrentFontRange = width;
244 } else
245 widthOfCurrentFontRange += width;
246
247 auto transformsType = shouldApplyFontTransforms(glyphBuffer, lastGlyphCount, currentCharacterIndex);
248 if (transformsType != TransformsType::None && FontCascade::treatAsSpace(character)) {
249 charactersTreatedAsSpace.constructAndAppend(
250 currentCharacterIndex,
251 character == ' ',
252 glyphBuffer.size() ? WebCore::width(glyphBuffer.advanceAt(glyphBuffer.size() - 1)) : 0,
253 width);
254 }
255
256 if (m_accountForGlyphBounds) {
257 bounds = font->boundsForGlyph(glyph);
258 if (!currentCharacterIndex)
259 m_firstGlyphOverflow = std::max<float>(0, -bounds.x());
260 }
261
262 if (m_forTextEmphasis && !FontCascade::canReceiveTextEmphasis(character))
263 glyph = 0;
264
265 glyphBuffer.add(glyph, *font, width, currentCharacterIndex);
266#if USE(CTFONTSHAPEGLYPHS)
267 // These 0 glyphs are needed by shapers if the source text has surrogate pairs.
268 // However, CTFontTransformGlyphs() can't delete these 0 glyphs from the shaped text,
269 // so we shouldn't add them in the first place if we're using that shaping routine.
270 // Any other shaping routine should delete these glyphs from the shaped text.
271 if (!U_IS_BMP(character))
272 glyphBuffer.add(0, *font, 0, currentCharacterIndex + 1);
273#endif
274
275 // Advance past the character we just dealt with.
276 textIterator.advance(advanceLength);
277 currentCharacterIndex = textIterator.currentIndex();
278
279 m_runWidthSoFar += width;
280
281 if (m_accountForGlyphBounds) {
282 m_maxGlyphBoundingBoxY = std::max(m_maxGlyphBoundingBoxY, bounds.maxY());
283 m_minGlyphBoundingBoxY = std::min(m_minGlyphBoundingBoxY, bounds.y());
284 m_lastGlyphOverflow = std::max<float>(0, bounds.maxX() - width);
285 }
286 }
287
288 commitCurrentFontRange(glyphBuffer, lastGlyphCount, currentCharacterIndex, *lastFontData, primaryFont, character, widthOfCurrentFontRange, charactersTreatedAsSpace);
289}

291auto WidthIterator::calculateAdditionalWidth(GlyphBuffer& glyphBuffer, GlyphBufferStringOffset currentCharacterIndex, unsigned leadingGlyphIndex, unsigned trailingGlyphIndex, float position) const -> AdditionalWidth
292{
293 float leftAdditionalWidth = 0;
294 float rightAdditionalWidth = 0;
295 float leftExpansionAdditionalWidth = 0;
296 float rightExpansionAdditionalWidth = 0;
297
298 auto character = m_run[currentCharacterIndex];
299
300 if (character == tabCharacter && m_run.allowTabs()) {
301 auto& font = glyphBuffer.fontAt(trailingGlyphIndex);
302 auto newWidth = m_font.tabWidth(font, m_run.tabSize(), position);
303 auto currentWidth = width(glyphBuffer.advanceAt(trailingGlyphIndex));
304 rightAdditionalWidth += newWidth - currentWidth;
305 }
306
307 if (hasExtraSpacing()) {
308 bool treatAsSpace = FontCascade::treatAsSpace(character);
309
310 // This is a heuristic to determine if the character is non-visible. Non-visible characters don't get letter-spacing.
311 float baseWidth = 0;
312 for (unsigned i = leadingGlyphIndex; i <= trailingGlyphIndex; ++i)
313 baseWidth += width(glyphBuffer.advanceAt(i));
314 if (baseWidth)
315 rightAdditionalWidth += m_font.letterSpacing();
316
317 if (treatAsSpace && (character != tabCharacter || !m_run.allowTabs()) && (currentCharacterIndex || character == noBreakSpace) && m_font.wordSpacing())
318 rightAdditionalWidth += m_font.wordSpacing();
319
320 if (m_expansion > 0) {
321 bool currentIsLastCharacter = m_lastCharacterIndex && currentCharacterIndex == static_cast<GlyphBufferStringOffset>(*m_lastCharacterIndex);
322
323 bool isLeftmostCharacter = !currentCharacterIndex;
324 bool isRightmostCharacter = currentIsLastCharacter;
325 if (!m_run.ltr())
326 std::swap(isLeftmostCharacter, isRightmostCharacter);
327
328 bool forceLeftExpansion = isLeftmostCharacter && (m_run.expansionBehavior() & LeftExpansionMask) == ForceLeftExpansion;
329 bool forceRightExpansion = isRightmostCharacter && (m_run.expansionBehavior() & RightExpansionMask) == ForceRightExpansion;
330 bool forbidLeftExpansion = isLeftmostCharacter && (m_run.expansionBehavior() & LeftExpansionMask) == ForbidLeftExpansion;
331 bool forbidRightExpansion = isRightmostCharacter && (m_run.expansionBehavior() & RightExpansionMask) == ForbidRightExpansion;
332
333 static const bool expandAroundIdeographs = FontCascade::canExpandAroundIdeographsInComplexText();
334 bool isIdeograph = expandAroundIdeographs && FontCascade::isCJKIdeographOrSymbol(character);
335
336 if (treatAsSpace || isIdeograph || forceLeftExpansion || forceRightExpansion) {
337 auto [expandLeft, expandRight] = expansionLocation(isIdeograph, treatAsSpace, m_run.ltr(), m_isAfterExpansion, forbidLeftExpansion, forbidRightExpansion, forceLeftExpansion, forceRightExpansion);
338
339 if (expandLeft)
340 leftExpansionAdditionalWidth += m_expansionPerOpportunity;
341 if (expandRight)
342 rightExpansionAdditionalWidth += m_expansionPerOpportunity;
343 }
344 }
345 }
346
347 return { leftAdditionalWidth, rightAdditionalWidth, leftExpansionAdditionalWidth, rightExpansionAdditionalWidth };
348}

350struct GlyphIndexRange {
351 // This means the character got expanded to glyphs inside the GlyphBuffer at indices [leadingGlyphIndex, trailingGlyphIndex].
352 unsigned leadingGlyphIndex;
353 unsigned trailingGlyphIndex;
354};
355
356void WidthIterator::applyAdditionalWidth(GlyphBuffer& glyphBuffer, GlyphIndexRange glyphIndexRange, float leftAdditionalWidth, float rightAdditionalWidth, float leftExpansionAdditionalWidth, float rightExpansionAdditionalWidth)
357{
358 m_expansion -= leftExpansionAdditionalWidth + rightExpansionAdditionalWidth;
359
360 leftAdditionalWidth += leftExpansionAdditionalWidth;
361 rightAdditionalWidth += rightExpansionAdditionalWidth;
362
363 m_runWidthSoFar += leftAdditionalWidth;
364 m_runWidthSoFar += rightAdditionalWidth;
365
366 if (leftAdditionalWidth) {
367 if (m_run.ltr()) {
368 // Left additional width in LTR means the previous (leading) glyph's right side gets expanded.
369 auto leadingGlyphIndex = glyphIndexRange.leadingGlyphIndex;
370 if (leadingGlyphIndex)
371 glyphBuffer.expandAdvance(leadingGlyphIndex - 1, leftAdditionalWidth);
372 else
373 glyphBuffer.expandInitialAdvance(leftAdditionalWidth);
374 } else {
375 // Left additional width in RTL means the next (trailing) glyph's right side gets expanded.
376 auto trailingGlyphIndex = glyphIndexRange.trailingGlyphIndex;
377 if (trailingGlyphIndex + 1 < glyphBuffer.size())
378 glyphBuffer.expandAdvance(trailingGlyphIndex + 1, leftAdditionalWidth);
379 else {
380 m_leftoverJustificationWidth = leftAdditionalWidth;
381 // We can't actually add in this width just yet.
382 // Add it in when the client calls advance() again or finalize().
383 m_runWidthSoFar -= m_leftoverJustificationWidth;
384 }
385 }
386 }
387
388 if (rightAdditionalWidth) {
389 // Right additional width means the current glyph's right side gets expanded. This is true for both LTR and RTL.
390 glyphBuffer.expandAdvance(glyphIndexRange.trailingGlyphIndex, rightAdditionalWidth);
391 }
392}

394void WidthIterator::applyExtraSpacingAfterShaping(GlyphBuffer& glyphBuffer, unsigned characterStartIndex, unsigned glyphBufferStartIndex, unsigned characterDestinationIndex, float startingRunWidth)
395{
396 Vector<std::optional<GlyphIndexRange>> characterIndexToGlyphIndexRange(m_run.length(), std::nullopt);
397 Vector<float> advanceWidths(m_run.length(), 0);
398 for (unsigned i = glyphBufferStartIndex; i < glyphBuffer.size(); ++i) {
399 auto stringOffset = glyphBuffer.checkedStringOffsetAt(i, m_run.length());
400 if (!stringOffset)
401 continue;
402 advanceWidths[stringOffset.value()] += width(glyphBuffer.advanceAt(i));
403 auto& glyphIndexRange = characterIndexToGlyphIndexRange[stringOffset.value()];
404 if (glyphIndexRange)
405 glyphIndexRange->trailingGlyphIndex = i;
406 else
407 glyphIndexRange = {{i, i}};
408 }
409
410 // SVG can stretch advances
411 if (m_run.horizontalGlyphStretch() != 1) {
412 for (unsigned i = glyphBufferStartIndex; i < glyphBuffer.size(); ++i) {
413 // All characters' advances get stretched, except apparently tab characters...
414 // This doesn't make much sense, because even tab characters get letter-spacing...
415 auto stringOffset = glyphBuffer.checkedStringOffsetAt(i, m_run.length());
416 if (stringOffset && m_run[stringOffset.value()] == tabCharacter)
417 continue;
418
419 auto currentAdvance = width(glyphBuffer.advanceAt(i));
420 auto newAdvance = currentAdvance * m_run.horizontalGlyphStretch();
421 glyphBuffer.expandAdvance(i, newAdvance - currentAdvance);
422 }
423 }
424
425 float position = m_run.xPos() + startingRunWidth;
426 for (auto i = characterStartIndex; i < characterDestinationIndex; ++i) {
427 auto& glyphIndexRange = characterIndexToGlyphIndexRange[i];
428 if (!glyphIndexRange)
429 continue;
430
431 auto width = calculateAdditionalWidth(glyphBuffer, i, glyphIndexRange->leadingGlyphIndex, glyphIndexRange->trailingGlyphIndex, position);
432 applyAdditionalWidth(glyphBuffer, glyphIndexRange.value(), width.left, width.right, width.leftExpansion, width.rightExpansion);
433
434 m_isAfterExpansion = (m_run.ltr() && width.rightExpansion) || (!m_run.ltr() && width.leftExpansion);
435
436 // This isn't quite perfect, because we may come across a tab character in between two glyphs which both report to correspond to a previous character.
437 // But, the fundamental concept of tabs isn't really compatible with complex text shaping, so this is probably okay.
438 // We can probably just do the best we can here.
439 // The only alternative, to calculate this position in glyph-space rather than character-space,
440 // is O(n^2) because we're iterating across the string here, rather than glyphs, so we can't keep the calculation up-to-date,
441 // which means calculateAdditionalWidth() would have to calculate the result from scratch whenever it's needed.
442 // And we can't do some sort of prefix-sum thing because applyAdditionalWidth() would modify the values,
443 // so updating the data structure each turn of this loop would also end up being O(n^2).
444 // Unfortunately, strings with tabs are more likely to be long data-table kind of strings, which means O(n^2) is not acceptable.
445 // Also, even if we did the O(n^2) thing, there would still be cases that wouldn't be perfect
446 // (because the fundamental concept of tabs isn't really compatible with complex text shaping),
447 // so let's choose the fast-wrong approach here instead of the slow-wrong approach.
448 position += advanceWidths[i]
449 + width.left
450 + width.right
451 + width.leftExpansion
452 + width.rightExpansion;
453 }

456void WidthIterator::finalize(GlyphBuffer& buffer)
457{
458 ASSERT(m_run.rtl() || !m_leftoverJustificationWidth);
459 // In LTR this does nothing. In RTL, this adds left width by moving the whole run to the right.
460 buffer.expandInitialAdvance(m_leftoverJustificationWidth);
461 m_runWidthSoFar += m_leftoverJustificationWidth;
462 m_leftoverJustificationWidth = 0;
463}

465void WidthIterator::advance(unsigned offset, GlyphBuffer& glyphBuffer)
466{
467 m_containsTabs = false;
468 unsigned length = m_run.length();
469
470 if (offset > length)
471 offset = length;
472
473 if (m_currentCharacterIndex >= offset)
474 return;
475
476 unsigned characterStartIndex = m_currentCharacterIndex;
477 unsigned glyphBufferStartIndex = glyphBuffer.size();
478 float startingRunWidth = m_runWidthSoFar;
479
480 if (m_run.is8Bit()) {
481 Latin1TextIterator textIterator(m_run.data8(m_currentCharacterIndex), m_currentCharacterIndex, offset, length);
482 advanceInternal(textIterator, glyphBuffer);
483 } else {
484 SurrogatePairAwareTextIterator textIterator(m_run.data16(m_currentCharacterIndex), m_currentCharacterIndex, offset, length);
485 advanceInternal(textIterator, glyphBuffer);
486 }
487
488 // In general, we have to apply spacing after shaping, because shaping requires its input to be unperturbed (see https://bugs.webkit.org/show_bug.cgi?id=215052).
489 // So, if there's extra spacing to add, do it here after shaping occurs.
490 if (glyphBufferStartIndex < glyphBuffer.size()) {
491 glyphBuffer.expandAdvance(glyphBufferStartIndex, m_leftoverJustificationWidth);
492 m_runWidthSoFar += m_leftoverJustificationWidth;
493 m_leftoverJustificationWidth = 0;
494 }
495
496 if (hasExtraSpacing() || m_containsTabs || m_run.horizontalGlyphStretch() != 1)
497 applyExtraSpacingAfterShaping(glyphBuffer, characterStartIndex, glyphBufferStartIndex, offset, startingRunWidth);
498}

500bool WidthIterator::advanceOneCharacter(float& width, GlyphBuffer& glyphBuffer)
501{
502 unsigned oldSize = glyphBuffer.size();
503 advance(m_currentCharacterIndex + 1, glyphBuffer);
504 float w = 0;
505 for (unsigned i = oldSize; i < glyphBuffer.size(); ++i)
506 w += WebCore::width(glyphBuffer.advanceAt(i));
507 width = w;
508 return glyphBuffer.size() > oldSize;
509}
