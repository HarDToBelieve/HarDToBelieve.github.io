36ExtensionsGLANGLE::ExtensionsGLANGLE(GraphicsContextGLOpenGL* context)
37 : m_initializedAvailableExtensions(false)

40}

42ExtensionsGLANGLE::~ExtensionsGLANGLE() = default;

44bool ExtensionsGLANGLE::supports(const String& name)
45{
46 if (!m_initializedAvailableExtensions) {
47 if (!m_context->makeContextCurrent())
48 return false;
49 initializeAvailableExtensions();
50 }
51 return supportsExtension(name);
52}
53
54void ExtensionsGLANGLE::ensureEnabled(const String& name)
55{
56 // Enable support in ANGLE (if not enabled already).
57 if (m_requestableExtensions.contains(name) && !m_enabledExtensions.contains(name)) {
58 if (!m_context->makeContextCurrent())
59 return;
60 gl::RequestExtensionANGLE(name.ascii().data());
61 m_enabledExtensions.add(name);
62
63 if (name == "GL_CHROMIUM_color_buffer_float_rgba"_s)
64 m_webglColorBufferFloatRGBA = true;
65 else if (name == "GL_CHROMIUM_color_buffer_float_rgb"_s)
66 m_webglColorBufferFloatRGB = true;
67 }
68}
69
70bool ExtensionsGLANGLE::isEnabled(const String& name)
71{
72 return m_availableExtensions.contains(name) || m_enabledExtensions.contains(name);
73}

80String ExtensionsGLANGLE::getTranslatedShaderSourceANGLE(PlatformGLObject shader)
81{
82 if (!m_context->makeContextCurrent())
83 return String();
84
85 int sourceLength = m_context->getShaderi(shader, GL_TRANSLATED_SHADER_SOURCE_LENGTH_ANGLE);
86
87 if (!sourceLength)
88 return emptyString();
89 Vector<GLchar> name(sourceLength); // GL_TRANSLATED_SHADER_SOURCE_LENGTH_ANGLE includes null termination.
90 GCGLint returnedLength = 0;
91 gl::GetTranslatedShaderSourceANGLE(shader, sourceLength, &returnedLength, name.data());
92 if (!returnedLength)
93 return emptyString();
94 // returnedLength does not include the null terminator.
95 ASSERT(returnedLength == sourceLength - 1);
96 return String(name.data(), returnedLength);
97}
98
99void ExtensionsGLANGLE::initializeAvailableExtensions()
100{
101 String extensionsString = getExtensions();
102 for (auto& extension : extensionsString.split(' '))
103 m_availableExtensions.add(extension);
104 extensionsString = String(reinterpret_cast<const char*>(gl::GetString(GL_REQUESTABLE_EXTENSIONS_ANGLE)));
105 for (auto& extension : extensionsString.split(' '))
106 m_requestableExtensions.add(extension);
107 m_initializedAvailableExtensions = true;
108}
109
110bool ExtensionsGLANGLE::supportsExtension(const String& name)
111{
112 return m_availableExtensions.contains(name) || m_requestableExtensions.contains(name);
113}

115void ExtensionsGLANGLE::drawBuffersEXT(GCGLSpan<const GCGLenum> bufs)
116{
117 if (!m_context->makeContextCurrent())
118 return;
119
120 gl::DrawBuffersEXT(bufs.bufSize, bufs.data);
121}

123String ExtensionsGLANGLE::getExtensions()
124{
125 return String(reinterpret_cast<const char*>(gl::GetString(GL_EXTENSIONS)));
126}

128GCGLenum ExtensionsGLANGLE::adjustWebGL1TextureInternalFormat(GCGLenum internalformat, GCGLenum format, GCGLenum type)
129{
130 // The implementation of WEBGL_color_buffer_float for WebGL 1.0 / ES 2.0 requires a sized
131 // internal format. Adjust it if necessary at this lowest level.
132 if (type == GL_FLOAT) {
133 if (m_webglColorBufferFloatRGBA && format == GL_RGBA && internalformat == GL_RGBA)
134 return GL_RGBA32F;
135 if (m_webglColorBufferFloatRGB && format == GL_RGB && internalformat == GL_RGB)
136 return GL_RGB32F;
137 }
138 return internalformat;
139}
