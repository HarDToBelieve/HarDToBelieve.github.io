34namespace WebCore {

36static unsigned defaultDisplayRefCount;
37static void* defaultDisplay = nullptr;
38
39static void refDefaultDisplay(void* display)
40{
41 ASSERT(display != nullptr);
42 ASSERT(defaultDisplay == display || (!defaultDisplay && !defaultDisplayRefCount));
43 defaultDisplayRefCount++;
44 defaultDisplay = display;
45}
46
47static void unrefDefaultDisplayIfNeeded(void* display)
48{
49 if (!display)
50 return;
51 ASSERT(defaultDisplay == display);
52 ASSERT(defaultDisplayRefCount);
53 defaultDisplayRefCount--;
54}
55
56ScopedEGLDefaultDisplay& ScopedEGLDefaultDisplay::operator=(ScopedEGLDefaultDisplay&& other)
57{
58 if (this != &other) {
59 unrefDefaultDisplayIfNeeded(m_display);
60 m_display = std::exchange(other.m_display, nullptr);
61 }
62 return *this;
63}
64
65ScopedEGLDefaultDisplay::ScopedEGLDefaultDisplay(void* display)
66 : m_display(display)
67{
68 refDefaultDisplay(m_display);
69}
70
71ScopedEGLDefaultDisplay::~ScopedEGLDefaultDisplay()
72{
73 unrefDefaultDisplayIfNeeded(m_display);
74}
75
76void ScopedEGLDefaultDisplay::releaseAllResourcesIfUnused()
77{
78 if (defaultDisplayRefCount)
79 return;
80 if (defaultDisplay == nullptr)
81 return;
82 bool result = EGL_Terminate(defaultDisplay);
83 ASSERT_UNUSED(result, result);
84 result = EGL_ReleaseThread();
85 ASSERT_UNUSED(result, result);
86 defaultDisplay = nullptr;
87}
