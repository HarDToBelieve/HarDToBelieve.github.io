36class ScopedRestoreTextureBinding {
37 WTF_MAKE_NONCOPYABLE(ScopedRestoreTextureBinding);
38public:
39 ScopedRestoreTextureBinding(GLenum bindingPointQuery, GLenum bindingPoint, bool condition = true)
40 {
41 ASSERT(bindingPoint != static_cast<GLenum>(0u));
42 if (condition) {
43 m_bindingPoint = bindingPoint;
44 gl::GetIntegerv(bindingPointQuery, reinterpret_cast<GLint*>(&m_bindingValue));
45 }
46 }
47
48 ~ScopedRestoreTextureBinding()
49 {
50 if (m_bindingPoint)
51 gl::BindTexture(m_bindingPoint, m_bindingValue);
52 }
53
54private:
55 GLenum m_bindingPoint { 0 };
56 GLuint m_bindingValue { 0u };
57};

60class ScopedBufferBinding {
61 WTF_MAKE_NONCOPYABLE(ScopedBufferBinding);
62public:
63 ScopedBufferBinding(GLenum bindingPoint, GLuint bindingValue, bool condition = true)
64 {
65 if (!condition)
66 return;
67 gl::GetIntegerv(query(bindingPoint), reinterpret_cast<GLint*>(&m_bindingValue));
68 if (bindingValue == m_bindingValue)
69 return;
70 m_bindingPoint = bindingPoint;
71 gl::BindBuffer(m_bindingPoint, bindingValue);
72 }
73
74 ~ScopedBufferBinding()
75 {
76 if (m_bindingPoint)
77 gl::BindBuffer(m_bindingPoint, m_bindingValue);
78 }
79
80private:
81 static constexpr GLenum query(GLenum bindingPoint)
82 {
83 if (bindingPoint == GL_PIXEL_PACK_BUFFER)
84 return GL_PIXEL_PACK_BUFFER_BINDING;
85 ASSERT(bindingPoint == GL_PIXEL_UNPACK_BUFFER);
86 return GL_PIXEL_UNPACK_BUFFER_BINDING;
87 }
88 GLint m_bindingPoint { 0 };
89 GLuint m_bindingValue { 0u };
90};
91class ScopedRestoreReadFramebufferBinding {
92 WTF_MAKE_NONCOPYABLE(ScopedRestoreReadFramebufferBinding);
93public:
94 ScopedRestoreReadFramebufferBinding(bool isForWebGL2, GLuint restoreValue)
95 : m_framebufferTarget(isForWebGL2 ? GL_READ_FRAMEBUFFER : GL_FRAMEBUFFER)
96 , m_bindingValue(restoreValue)
97 {
98 }
99 ScopedRestoreReadFramebufferBinding(bool isForWebGL2, GLuint restoreValue, GLuint value)
100 : m_framebufferTarget(isForWebGL2 ? GL_READ_FRAMEBUFFER : GL_FRAMEBUFFER)
101 , m_bindingValue(restoreValue)
102 {
103 bindFramebuffer(value);
104 }
105 void markBindingChanged()
106 {
107 m_bindingChanged = true;
108 }
109 void bindFramebuffer(GLuint bindingValue)
110 {
111 if (!m_bindingChanged && m_bindingValue == bindingValue)
112 return;
113 gl::BindFramebuffer(m_framebufferTarget, bindingValue);
114 m_bindingChanged = m_bindingValue != bindingValue;
115 }
116 GLuint framebufferTarget() const { return m_framebufferTarget; }
117 ~ScopedRestoreReadFramebufferBinding()
118 {
119 if (m_bindingChanged)
120 gl::BindFramebuffer(m_framebufferTarget, m_bindingValue);
121 }
122private:
123 const GLenum m_framebufferTarget;
124 GLuint m_bindingValue { 0u };
125 bool m_bindingChanged { false };
126};

128class ScopedPixelStorageMode {
129 WTF_MAKE_NONCOPYABLE(ScopedPixelStorageMode);
130public:
131 explicit ScopedPixelStorageMode(GLenum name, bool condition = true)
132 : m_name(condition ? name : 0)
133 {
134 if (m_name)
135 gl::GetIntegerv(m_name, &m_originalValue);
136 }
137 ScopedPixelStorageMode(GLenum name, GLint value, bool condition = true)
138 : m_name(condition ? name : 0)
139 {
140 if (m_name) {
141 gl::GetIntegerv(m_name, &m_originalValue);
142 pixelStore(value);
143 }
144 }
145 ~ScopedPixelStorageMode()
146 {
147 if (m_name && m_valueChanged)
148 gl::PixelStorei(m_name, m_originalValue);
149 }
150 void pixelStore(GLint value)
151 {
152 ASSERT(m_name);
153 if (!m_valueChanged && m_originalValue == value)
154 return;
155 gl::PixelStorei(m_name, value);
156 m_valueChanged = m_originalValue != value;
157 }
158 operator GLint() const
159 {
160 ASSERT(m_name && !m_valueChanged);
161 return m_originalValue;
162 }
163private:
164 const GLenum m_name;
165 bool m_valueChanged { false };
166 GLint m_originalValue { 0 };
167};
