47auto CDMPrivateMediaSourceAVFObjC::parseKeySystem(const String& keySystem) -> std::optional<KeySystemParameters>
48{
49 static NeverDestroyed<RegularExpression> keySystemRE("^com\\.apple\\.fps\\.[23]_\\d+(?:,\\d+)*$", JSC::Yarr::TextCaseInsensitive);
50
51 if (keySystemRE.get().match(keySystem) < 0)
52 return std::nullopt;
53
54 StringView keySystemView { keySystem };
55
56 int cdmVersion = parseInteger<int>(keySystemView.substring(14, 1)).value();
57 Vector<int> protocolVersions;
58 for (auto protocolVersionString : keySystemView.substring(16).split(','))
59 protocolVersions.append(parseInteger<int>(protocolVersionString).value());
60 return { { cdmVersion, WTFMove(protocolVersions) } };
61}

63CDMPrivateMediaSourceAVFObjC::~CDMPrivateMediaSourceAVFObjC()
64{
65 for (auto& session : m_sessions)
66 session->invalidateCDM();
67}

69static bool queryDecoderAvailability()
70{
71 if (!canLoad_VideoToolbox_VTGetGVADecoderAvailability()) {
72#if HAVE(AVSTREAMSESSION)
73 return false;
74#else
75 return true;
76#endif
77 }
78 uint32_t totalInstanceCount = 0;
79 OSStatus status = VTGetGVADecoderAvailability(&totalInstanceCount, nullptr);
80 return status == noErr && totalInstanceCount;
81}
82
83bool CDMPrivateMediaSourceAVFObjC::supportsKeySystem(const String& keySystem)
84{
85 if (!queryDecoderAvailability())
86 return false;
87
88 auto parameters = parseKeySystem(keySystem);
89 if (!parameters)
90 return false;
91
92 if (parameters.value().version == 3 && !CDMSessionAVContentKeySession::isAvailable())
93 return false;
94
95 return true;
96}
97
98bool CDMPrivateMediaSourceAVFObjC::supportsKeySystemAndMimeType(const String& keySystem, const String& mimeType)
99{
100 if (!supportsKeySystem(keySystem))
101 return false;
102
103 if (mimeType.isEmpty())
104 return true;
105
106 // FIXME: Why is this ignoring case since the check in supportsMIMEType is checking case?
107 if (equalLettersIgnoringASCIICase(mimeType, "keyrelease"))
108 return true;
109
110 MediaEngineSupportParameters parameters;
111 parameters.isMediaSource = true;
112 parameters.type = ContentType(mimeType);
113
114 return MediaPlayerPrivateMediaSourceAVFObjC::supportsTypeAndCodecs(parameters) != MediaPlayer::SupportsType::IsNotSupported;
115}
116
117bool CDMPrivateMediaSourceAVFObjC::supportsMIMEType(const String& mimeType)
118{
119 // FIXME: Why is this checking case since the check in supportsKeySystemAndMimeType is ignoring case?
120 if (mimeType == "keyrelease")
121 return true;
122
123 MediaEngineSupportParameters parameters;
124 parameters.isMediaSource = true;
125 parameters.type = ContentType(mimeType);
126
127 return MediaPlayerPrivateMediaSourceAVFObjC::supportsTypeAndCodecs(parameters) != MediaPlayer::SupportsType::IsNotSupported;
128}

130std::unique_ptr<LegacyCDMSession> CDMPrivateMediaSourceAVFObjC::createSession(LegacyCDMSessionClient* client)
131{
132 String keySystem = m_cdm->keySystem(); // Local copy for StringView usage
133 auto parameters = parseKeySystem(m_cdm->keySystem());
134 ASSERT(parameters);
135 if (!parameters)
136 return nullptr;
137
138 std::unique_ptr<CDMSessionMediaSourceAVFObjC> session;
139
140#if HAVE(AVSTREAMSESSION)
141 bool shouldUseAVContentKeySession = parameters.value().version == 3;
142#else
143 bool shouldUseAVContentKeySession = true;
144#endif
145
146 if (shouldUseAVContentKeySession && CDMSessionAVContentKeySession::isAvailable())
147 session = makeUnique<CDMSessionAVContentKeySession>(WTFMove(parameters.value().protocols), parameters.value().version, *this, client);
148 else
149#if HAVE(AVSTREAMSESSION)
150 session = makeUnique<CDMSessionAVStreamSession>(WTFMove(parameters.value().protocols), *this, client);
151#else
152 return nullptr;
153#endif
154
155 m_sessions.append(session.get());
156 return WTFMove(session);
157}
158
159void CDMPrivateMediaSourceAVFObjC::invalidateSession(CDMSessionMediaSourceAVFObjC* session)
160{
161 ASSERT(m_sessions.contains(session));
162 m_sessions.removeAll(session);
163}
