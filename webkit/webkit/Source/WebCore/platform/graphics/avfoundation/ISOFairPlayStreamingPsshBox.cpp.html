47bool ISOFairPlayStreamingKeyRequestInfoBox::parse(JSC::DataView& view, unsigned& offset)
48{
49 unsigned localOffset = offset;
50 if (!ISOBox::parse(view, localOffset))
51 return false;
52
53 CheckedUint64 remaining = m_size;
54 remaining -= (localOffset - offset);
55 if (remaining.hasOverflowed())
56 return false;
57
58 if (remaining < m_keyID.capacity())
59 return false;
60
61 auto buffer = view.possiblySharedBuffer();
62 if (!buffer)
63 return false;
64
65 auto keyID = buffer->slice(localOffset, localOffset + m_keyID.capacity());
66 localOffset += m_keyID.capacity();
67
68 m_keyID.resize(m_keyID.capacity());
69 memcpy(m_keyID.data(), keyID->data(), m_keyID.capacity());
70
71 offset = localOffset;
72 return true;
73}

75bool ISOFairPlayStreamingKeyAssetIdBox::parse(JSC::DataView& view, unsigned& offset)
76{
77 unsigned localOffset = offset;
78 if (!ISOBox::parse(view, localOffset))
79 return false;
80
81 if (localOffset - offset == m_size) {
82 m_data.clear();
83 offset = localOffset;
84 return true;
85 }
86
87 auto buffer = view.possiblySharedBuffer();
88 if (!buffer)
89 return false;
90
91 size_t dataSize;
92 if (!WTF::safeSub(m_size, localOffset - offset, dataSize))
93 return false;
94
95 auto parsedData = buffer->slice(localOffset, localOffset + dataSize);
96 localOffset += dataSize;
97
98 m_data.resize(dataSize);
99 memcpy(m_data.data(), parsedData->data(), dataSize);
100 offset = localOffset;
101 return true;
102}

133bool ISOFairPlayStreamingKeyVersionListBox::parse(JSC::DataView& view, unsigned& offset)
134{
135 unsigned localOffset = offset;
136 if (!ISOBox::parse(view, localOffset))
137 return false;
138
139 do {
140 if (localOffset - offset == m_size)
141 break;
142
143 uint64_t remaining;
144 if (!WTF::safeSub(m_size, localOffset - offset, remaining))
145 return false;
146
147 if (remaining < sizeof(uint32_t))
148 return false;
149
150 uint32_t version;
151 if (!checkedRead<uint32_t>(version, view, localOffset, BigEndian))
152 return false;
153 m_versions.append(version);
154 } while (true);
155
156 offset = localOffset;
157 return true;
158}

160bool ISOFairPlayStreamingKeyRequestBox::parse(JSC::DataView& view, unsigned& offset)
161{
162 unsigned localOffset = offset;
163 if (!ISOBox::parse(view, localOffset))
164 return false;
165
166 if (!m_requestInfo.read(view, localOffset))
167 return false;
168
169 while (localOffset - offset < m_size) {
170 auto result = peekBox(view, localOffset);
171 if (!result)
172 return false;
173
174 auto name = result.value().first;
175 if (name == ISOFairPlayStreamingKeyAssetIdBox::boxTypeName()) {
176 if (m_assetID)
177 return false;
178
179 ISOFairPlayStreamingKeyAssetIdBox assetID;
180 if (!assetID.read(view, localOffset))
181 return false;
182
183 m_assetID = WTFMove(assetID);
184 continue;
185 }
186
187 if (name == ISOFairPlayStreamingKeyContextBox::boxTypeName()) {
188 if (m_context)
189 return false;
190
191 ISOFairPlayStreamingKeyContextBox context;
192 if (!context.read(view, localOffset))
193 return false;
194
195 m_context = WTFMove(context);
196 continue;
197 }
198
199 if (name == ISOFairPlayStreamingKeyVersionListBox::boxTypeName()) {
200 if (m_versionList)
201 return false;
202
203 ISOFairPlayStreamingKeyVersionListBox versionList;
204 if (!versionList.read(view, localOffset))
205 return false;
206
207 m_versionList = WTFMove(versionList);
208 continue;
209 }
210
211 // Unknown box type; error.
212 return false;
213 }
214
215 offset = localOffset;
216 return true;
217}

219bool ISOFairPlayStreamingInitDataBox::parse(JSC::DataView& view, unsigned& offset)
220{
221 unsigned localOffset = offset;
222 if (!ISOBox::parse(view, localOffset))
223 return false;
224
225 if (!m_info.read(view, localOffset))
226 return false;
227
228 while (localOffset - offset < m_size) {
229 ISOFairPlayStreamingKeyRequestBox request;
230 if (!request.read(view, localOffset))
231 return false;
232
233 m_requests.append(WTFMove(request));
234 }
235
236 offset = localOffset;
237 return true;
238}
