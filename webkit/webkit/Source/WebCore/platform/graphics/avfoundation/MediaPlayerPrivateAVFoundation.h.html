93 enum Type {
94#define DEFINE_TYPE_ENUM(type) type,
95 FOR_EACH_MEDIAPLAYERPRIVATEAVFOUNDATION_NOTIFICATION_TYPE(DEFINE_TYPE_ENUM)
96#undef DEFINE_TYPE_ENUM
97 FunctionType,
98 };
99
100 Notification()
101 : m_type(None)
102 , m_finished(false)
103 {
104 }
105
106 Notification(Type type, const MediaTime& time)
107 : m_type(type)
108 , m_time(time)
109 , m_finished(false)
110 {
111 }
112
113 Notification(Type type, bool finished)
114 : m_type(type)
115 , m_finished(finished)
116 {
117 }
118
119 Notification(WTF::Function<void ()>&& function)
120 : m_type(FunctionType)
121 , m_finished(false)
122 , m_function(WTFMove(function))
123 {
124 }
125
126 Type type() { return m_type; }
127 bool isValid() { return m_type != None; }
128 MediaTime time() { return m_time; }
129 bool finished() { return m_finished; }
130 WTF::Function<void ()>& function() { return m_function; }
131
132 private:
133 Type m_type;
134 MediaTime m_time;
135 bool m_finished;
136 WTF::Function<void ()> m_function;
137 };
