55bool AVAssetMIMETypeCache::canDecodeExtendedType(const ContentType& type)
56{
57#if ENABLE(VIDEO) && USE(AVFOUNDATION)
58 ASSERT(isAvailable());
59 if ([PAL::getAVURLAssetClass() isPlayableExtendedMIMEType:type.raw()])
60 return true;
61
62#if ENABLE(WEBM_FORMAT_READER)
63 if (SourceBufferParserWebM::isContentTypeSupported(type) == MediaPlayerEnums::SupportsType::IsSupported)
64 return true;
65#endif
66
67#endif // ENABLE(VIDEO) && USE(AVFOUNDATION)
68
69 return false;
70}

72bool AVAssetMIMETypeCache::isUnsupportedContainerType(const String& type)
73{
74 if (type.isEmpty())
75 return false;
76
77 String lowerCaseType = type.convertToASCIILowercase();
78
79 // AVFoundation will return non-video MIME types which it claims to support, but which we
80 // do not support in the <video> element. Reject all non video/, audio/, and application/ types.
81 if (!lowerCaseType.startsWith("video/") && !lowerCaseType.startsWith("audio/") && !lowerCaseType.startsWith("application/"))
82 return true;
83
84 // Reject types we know AVFoundation does not support that sites commonly ask about.
85 if (lowerCaseType == "video/x-flv")
86 return true;
87
88 if (lowerCaseType == "audio/ogg" || lowerCaseType == "video/ogg" || lowerCaseType == "application/ogg")
89 return true;
90
91 if (lowerCaseType == "video/h264")
92 return true;
93
94 return false;
95}

97bool AVAssetMIMETypeCache::isStaticContainerType(StringView type)
98{
99 static constexpr ComparableLettersLiteral staticContainerTypesArray[] = {
100 "application/vnd.apple.mpegurl",
101 "application/x-mpegurl",
102 "audio/3gpp",
103 "audio/aac",
104 "audio/aacp",
105 "audio/aiff",
106 "audio/basic",
107 "audio/mp3",
108 "audio/mp4",
109 "audio/mpeg",
110 "audio/mpeg3",
111 "audio/mpegurl",
112 "audio/mpg",
113 "audio/vnd.wave",
114 "audio/wav",
115 "audio/wave",
116 "audio/x-aac",
117 "audio/x-aiff",
118 "audio/x-m4a",
119 "audio/x-mpegurl",
120 "audio/x-wav",
121 "video/3gpp",
122 "video/3gpp2",
123 "video/mp4",
124 "video/mpeg",
125 "video/mpeg2",
126 "video/mpg",
127 "video/quicktime",
128 "video/x-m4v",
129 "video/x-mpeg",
130 "video/x-mpg",
131 };
132 static constexpr SortedArraySet staticContainerTypesSet { staticContainerTypesArray };
133 return staticContainerTypesSet.contains(type);
134}

143void AVAssetMIMETypeCache::initializeCache(HashSet<String, ASCIICaseInsensitiveHash>& cache)
144{
145#if ENABLE(VIDEO) && USE(AVFOUNDATION)
146 if (!isAvailable())
147 return;
148
149 for (NSString *type in [PAL::getAVURLAssetClass() audiovisualMIMETypes])
150 cache.add(type);
151
152#if ENABLE(WEBM_FORMAT_READER)
153 if (SourceBufferParserWebM::isWebMFormatReaderAvailable()) {
154 auto webmTypes = SourceBufferParserWebM::webmMIMETypes();
155 cache.add(webmTypes.begin(), webmTypes.end());
156 }
157#endif
158
159 if (m_cacheTypeCallback)
160 m_cacheTypeCallback(copyToVector(cache));
161#endif
162}
