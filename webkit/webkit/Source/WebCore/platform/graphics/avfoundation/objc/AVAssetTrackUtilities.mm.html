40static Vector<FourCC> contentTypesToCodecs(const Vector<ContentType>& contentTypes)
41{
42 Vector<FourCC> codecs;
43 for (auto& contentType : contentTypes) {
44 auto codecStrings = contentType.codecs();
45 for (auto& codecString : codecStrings) {
46 // https://tools.ietf.org/html/rfc6381
47 // Within a 'codecs' parameter value, "." is reserved as a hierarchy delimiter
48 auto firstPeriod = codecString.find('.');
49 if (firstPeriod != notFound)
50 codecString.truncate(firstPeriod);
51
52 auto codecIdentifier = FourCC::fromString(codecString.left(4));
53 if (codecIdentifier)
54 codecs.append(codecIdentifier.value());
55 }
56 }
57 return codecs;
58}

60bool codecsMeetHardwareDecodeRequirements(const Vector<FourCC>& codecs, const Vector<ContentType>& contentTypesRequiringHardwareDecode)
61{
62 static bool hasBattery = systemHasBattery();
63
64 // If the system is exclusively wall-powered, do not require hardware support.
65 if (!hasBattery)
66 return true;
67
68 // If we can't determine whether a codec has hardware support or not, default to true.
69 if (!canLoad_VideoToolbox_VTIsHardwareDecodeSupported())
70 return true;
71
72 if (contentTypesRequiringHardwareDecode.isEmpty())
73 return true;
74
75 Vector<FourCC> hardwareCodecs = contentTypesToCodecs(contentTypesRequiringHardwareDecode);
76
77 for (auto& codec : codecs) {
78 if (hardwareCodecs.contains(codec) && !VTIsHardwareDecodeSupported(codec.value))
79 return false;
80 }
81 return true;
82}
83
84bool contentTypeMeetsHardwareDecodeRequirements(const ContentType& contentType, const Vector<ContentType>& contentTypesRequiringHardwareDecode)
85{
86 Vector<FourCC> codecs = contentTypesToCodecs({ contentType });
87 return codecsMeetHardwareDecodeRequirements(codecs, contentTypesRequiringHardwareDecode);
88}
89
90bool assetTrackMeetsHardwareDecodeRequirements(AVAssetTrack *track, const Vector<ContentType>& contentTypesRequiringHardwareDecode)
91{
92 Vector<FourCC> codecs;
93 for (NSUInteger i = 0, count = track.formatDescriptions.count; i < count; ++i) {
94 CMFormatDescriptionRef description = (__bridge CMFormatDescriptionRef)track.formatDescriptions[i];
95 if (PAL::CMFormatDescriptionGetMediaType(description) == kCMMediaType_Video)
96 codecs.append(FourCC(PAL::CMFormatDescriptionGetMediaSubType(description)));
97 }
98 return codecsMeetHardwareDecodeRequirements(codecs, contentTypesRequiringHardwareDecode);
99}
