32OBJC_CLASS AVOutputContext;
33OBJC_CLASS NSView;

35namespace WebCore {
36
37class FloatRect;
38
39class AVPlaybackTargetPicker : public CanMakeWeakPtr<AVPlaybackTargetPicker> {
40 WTF_MAKE_FAST_ALLOCATED;
41 WTF_MAKE_NONCOPYABLE(AVPlaybackTargetPicker);
42public:
43 class Client : public CanMakeWeakPtr<Client> {
44 protected:
45 virtual ~Client() = default;
46
47 public:
48 virtual void pickerWasDismissed() = 0;
49 virtual void availableDevicesChanged() = 0;
50 virtual void currentDeviceChanged() = 0;
51 };
52
53 explicit AVPlaybackTargetPicker(Client& client)
54 : m_client(makeWeakPtr(&client))
55 {
56 }
57 virtual ~AVPlaybackTargetPicker() = default;
58
59 virtual void showPlaybackTargetPicker(NSView *, const FloatRect&, bool checkActiveRoute, bool useDarkAppearancebool, bool useiTunesAVOutputContext) = 0;
60 virtual void startingMonitoringPlaybackTargets() = 0;
61 virtual void stopMonitoringPlaybackTargets() = 0;
62 virtual void invalidatePlaybackTargets() = 0;
63 virtual bool externalOutputDeviceAvailable() = 0;
64
65 virtual AVOutputContext* outputContext() = 0;
66
67 WeakPtr<AVPlaybackTargetPicker::Client> client() const { return m_client; }
68
69private:
70 WeakPtr<AVPlaybackTargetPicker::Client> m_client;
71};

73} // namespace WebCore
