39namespace WebCore {
40
41class CDMPrivateMediaSourceAVFObjC;
42
43class CDMSessionAVContentKeySession : public CDMSessionMediaSourceAVFObjC {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 CDMSessionAVContentKeySession(Vector<int>&& protocolVersions, int cdmVersion, CDMPrivateMediaSourceAVFObjC&, LegacyCDMSessionClient*);
47 virtual ~CDMSessionAVContentKeySession();
48
49 static bool isAvailable();
50
51 // LegacyCDMSession
52 LegacyCDMSessionType type() override { return CDMSessionTypeAVContentKeySession; }
53 RefPtr<Uint8Array> generateKeyRequest(const String& mimeType, Uint8Array* initData, String& destinationURL, unsigned short& errorCode, uint32_t& systemCode) override;
54 void releaseKeys() override;
55 bool update(Uint8Array* key, RefPtr<Uint8Array>& nextMessage, unsigned short& errorCode, uint32_t& systemCode) override;
56
57 // CDMSessionMediaSourceAVFObjC
58 void addParser(AVStreamDataParser *) override;
59 void removeParser(AVStreamDataParser *) override;
60
61 void didProvideContentKeyRequest(AVContentKeyRequest *);
62
63protected:
64 RefPtr<Uint8Array> generateKeyReleaseMessage(unsigned short& errorCode, uint32_t& systemCode);
65
66 bool hasContentKeySession() const { return m_contentKeySession; }
67 AVContentKeySession* contentKeySession();
68
69 RetainPtr<AVContentKeySession> m_contentKeySession;
70 RetainPtr<WebCDMSessionAVContentKeySessionDelegate> m_contentKeySessionDelegate;
71 RetainPtr<AVContentKeyRequest> m_keyRequest;
72 RefPtr<Uint8Array> m_identifier;
73 RefPtr<Uint8Array> m_initData;
74 RetainPtr<NSData> m_expiredSession;
75 Vector<int> m_protocolVersions;
76 int m_cdmVersion;
77 int32_t m_protectedTrackID { 1 };
78 enum { Normal, KeyRelease } m_mode;
79};
80
81inline CDMSessionAVContentKeySession* toCDMSessionAVContentKeySession(LegacyCDMSession* session)
82{
83 if (!session || session->type() != CDMSessionTypeAVContentKeySession)
84 return nullptr;
85 return static_cast<CDMSessionAVContentKeySession*>(session);
86}
