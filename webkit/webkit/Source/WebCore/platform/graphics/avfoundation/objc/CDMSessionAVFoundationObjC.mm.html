57RefPtr<Uint8Array> CDMSessionAVFoundationObjC::generateKeyRequest(const String& mimeType, Uint8Array* initData, String& destinationURL, unsigned short& errorCode, uint32_t& systemCode)
58{
59 UNUSED_PARAM(mimeType);
60
61 if (!m_parent) {
62 errorCode = LegacyCDM::UnknownError;
63 return nullptr;
64 }
65
66 String keyURI;
67 String keyID;
68 RefPtr<Uint8Array> certificate;
69 if (!MediaPlayerPrivateAVFoundationObjC::extractKeyURIKeyIDAndCertificateFromInitData(initData, keyURI, keyID, certificate)) {
70 errorCode = LegacyCDM::UnknownError;
71 return nullptr;
72 }
73
74 m_request = m_parent->takeRequestForKeyURI(keyURI);
75 if (!m_request) {
76 errorCode = LegacyCDM::UnknownError;
77 return nullptr;
78 }
79
80 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:certificate->baseAddress() length:certificate->byteLength()]);
81 NSString* assetStr = keyID;
82 RetainPtr<NSData> assetID = [NSData dataWithBytes: [assetStr cStringUsingEncoding:NSUTF8StringEncoding] length:[assetStr lengthOfBytesUsingEncoding:NSUTF8StringEncoding]];
83 NSError* nsError = 0;
84 RetainPtr<NSData> keyRequest = [m_request streamingContentKeyRequestDataForApp:certificateData.get() contentIdentifier:assetID.get() options:nil error:&nsError];
85
86 if (!keyRequest) {
87 errorCode = LegacyCDM::DomainError;
88 systemCode = mediaKeyErrorSystemCode(nsError);
89 return nullptr;
90 }
91
92 errorCode = MediaPlayer::NoError;
93 systemCode = 0;
94 destinationURL = String();
95
96 auto keyRequestBuffer = ArrayBuffer::create([keyRequest.get() bytes], [keyRequest.get() length]);
97 unsigned byteLength = keyRequestBuffer->byteLength();
98 return Uint8Array::tryCreate(WTFMove(keyRequestBuffer), 0, byteLength);
99}

105bool CDMSessionAVFoundationObjC::update(Uint8Array* key, RefPtr<Uint8Array>& nextMessage, unsigned short& errorCode, uint32_t& systemCode)
106{
107 RetainPtr<NSData> keyData = adoptNS([[NSData alloc] initWithBytes:key->baseAddress() length:key->byteLength()]);
108 [[m_request dataRequest] respondWithData:keyData.get()];
109 [m_request finishLoading];
110 errorCode = MediaPlayer::NoError;
111 systemCode = 0;
112 nextMessage = nullptr;
113
114 return true;
115}

117void CDMSessionAVFoundationObjC::playerDidReceiveError(NSError *error)
118{
119 if (!m_client)
120 return;
121
122 unsigned long code = mediaKeyErrorSystemCode(error);
123 m_client->sendError(LegacyCDMSessionClient::MediaKeyErrorDomain, code);
124}
