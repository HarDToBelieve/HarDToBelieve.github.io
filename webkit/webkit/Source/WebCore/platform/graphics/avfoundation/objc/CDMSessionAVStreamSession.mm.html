102RefPtr<Uint8Array> CDMSessionAVStreamSession::generateKeyRequest(const String& mimeType, Uint8Array* initData, String& destinationURL, unsigned short& errorCode, uint32_t& systemCode)
103{
104 UNUSED_PARAM(mimeType);
105 UNUSED_PARAM(destinationURL);
106 ASSERT(initData);
107
108 LOG(Media, "CDMSessionAVStreamSession::generateKeyRequest(%p)", this);
109
110 errorCode = MediaPlayer::NoError;
111 systemCode = 0;
112
113 m_initData = initData;
114
115 if (equalLettersIgnoringASCIICase(mimeType, "keyrelease")) {
116 m_mode = KeyRelease;
117 return generateKeyReleaseMessage(errorCode, systemCode);
118 }
119
120 String certificateString("certificate"_s);
121 auto array = Uint8Array::create(certificateString.length());
122 for (unsigned i = 0, length = certificateString.length(); i < length; ++i)
123 array->set(i, certificateString[i]);
124 return WTFMove(array);
125}

127void CDMSessionAVStreamSession::releaseKeys()
128{
129 if (m_streamSession) {
130 m_stopped = true;
131 for (auto& sourceBuffer : m_sourceBuffers)
132 sourceBuffer->flush();
133
134 LOG(Media, "CDMSessionAVStreamSession::releaseKeys(%p) - expiring stream session", this);
135 [m_streamSession expire];
136
137 if (!m_certificate)
138 return;
139
140 String storagePath = this->storagePath();
141 if (storagePath.isEmpty() || ![PAL::getAVStreamSessionClass() respondsToSelector:@selector(pendingExpiredSessionReportsWithAppIdentifier:storageDirectoryAtURL:)])
142 return;
143
144 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
145 NSArray* expiredSessions = [PAL::getAVStreamSessionClass() pendingExpiredSessionReportsWithAppIdentifier:certificateData.get() storageDirectoryAtURL:[NSURL fileURLWithPath:storagePath]];
146 for (NSData* expiredSessionData in expiredSessions) {
147 NSDictionary *expiredSession = [NSPropertyListSerialization propertyListWithData:expiredSessionData options:kCFPropertyListImmutable format:nullptr error:nullptr];
148 NSString *playbackSessionIdValue = (NSString *)[expiredSession objectForKey:PlaybackSessionIdKey];
149 if (![playbackSessionIdValue isKindOfClass:[NSString class]])
150 continue;
151
152 if (m_sessionId == String(playbackSessionIdValue)) {
153 LOG(Media, "CDMSessionAVStreamSession::releaseKeys(%p) - found session, sending expiration message");
154 m_expiredSession = expiredSessionData;
155 m_client->sendMessage(Uint8Array::create(static_cast<const uint8_t*>([m_expiredSession bytes]), [m_expiredSession length]).ptr(), emptyString());
156 break;
157 }
158 }
159 }
160}

162static bool isEqual(Uint8Array* data, const char* literal)
163{
164 ASSERT(data);
165 ASSERT(literal);
166 unsigned length = data->length();
167
168 for (unsigned i = 0; i < length; ++i) {
169 if (!literal[i])
170 return false;
171
172 if (data->item(i) != static_cast<uint8_t>(literal[i]))
173 return false;
174 }
175 return !literal[length];
176}

178bool CDMSessionAVStreamSession::update(Uint8Array* key, RefPtr<Uint8Array>& nextMessage, unsigned short& errorCode, uint32_t& systemCode)
179{
180 bool shouldGenerateKeyRequest = !m_certificate || isEqual(key, "renew");
181 if (!m_certificate) {
182 LOG(Media, "CDMSessionAVStreamSession::update(%p) - certificate data", this);
183
184 m_certificate = key;
185 }
186
187 if (isEqual(key, "acknowledged")) {
188 LOG(Media, "CDMSessionAVStreamSession::update(%p) - acknowleding secure stop message", this);
189
190 if (!m_expiredSession) {
191 errorCode = MediaPlayer::InvalidPlayerState;
192 return false;
193 }
194
195 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
196
197 IGNORE_WARNINGS_BEGIN("objc-literal-conversion")
198 String storagePath = this->storagePath();
199 if (!storagePath.isEmpty() && [PAL::getAVStreamSessionClass() respondsToSelector:@selector(removePendingExpiredSessionReports:withAppIdentifier:storageDirectoryAtURL:)])
200 [PAL::getAVStreamSessionClass() removePendingExpiredSessionReports:@[m_expiredSession.get()] withAppIdentifier:certificateData.get() storageDirectoryAtURL:[NSURL fileURLWithPath:storagePath]];
201 IGNORE_WARNINGS_END
202 m_expiredSession = nullptr;
203 return true;
204 }
205
206 if (m_mode == KeyRelease)
207 return false;
208
209 RefPtr<SourceBufferPrivateAVFObjC> protectedSourceBuffer;
210 for (auto& sourceBuffer : m_sourceBuffers) {
211 if (sourceBuffer->protectedTrackID() != notFound) {
212 protectedSourceBuffer = sourceBuffer;
213 break;
214 }
215 }
216
217 if (shouldGenerateKeyRequest) {
218 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
219
220 if (m_sourceBuffers.isEmpty())
221 return true;
222
223 if (!protectedSourceBuffer)
224 return true;
225
226 RetainPtr<NSData> initData = adoptNS([[NSData alloc] initWithBytes:m_initData->data() length:m_initData->length()]);
227
228 RetainPtr<NSDictionary> options;
229 if (!m_protocolVersions.isEmpty()) {
230 options = @{ PAL::get_AVFoundation_AVStreamDataParserContentKeyRequestProtocolVersionsKey(): createNSArray(m_protocolVersions, [] (int version) -> NSNumber * {
231 return version ? @(version) : nil;
232 }).get() };
233 }
234
235 NSError* error = nil;
236 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
237 RetainPtr<NSData> request = [protectedSourceBuffer->streamDataParser() streamingContentKeyRequestDataForApp:certificateData.get() contentIdentifier:initData.get() trackID:protectedSourceBuffer->protectedTrackID() options:options.get() error:&error];
238 ALLOW_DEPRECATED_DECLARATIONS_END
239
240 if (![protectedSourceBuffer->streamDataParser() respondsToSelector:@selector(contentProtectionSessionIdentifier)])
241 m_sessionId = createCanonicalUUIDString();
242
243 if (error) {
244 LOG(Media, "CDMSessionAVStreamSession::update(%p) - error:%@", this, [error description]);
245 errorCode = MediaPlayer::InvalidPlayerState;
246 systemCode = mediaKeyErrorSystemCode(error);
247 return false;
248 }
249
250 nextMessage = Uint8Array::create([request length]);
251 [request getBytes:nextMessage->data() length:nextMessage->length()];
252 return false;
253 }
254
255 if (!protectedSourceBuffer) {
256 errorCode = MediaPlayer::InvalidPlayerState;
257 return false;
258 }
259
260 ASSERT(!m_sourceBuffers.isEmpty());
261 LOG(Media, "CDMSessionAVStreamSession::update(%p) - key data", this);
262 errorCode = MediaPlayer::NoError;
263 systemCode = 0;
264 RetainPtr<NSData> keyData = adoptNS([[NSData alloc] initWithBytes:key->data() length:key->length()]);
265 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
266 [protectedSourceBuffer->streamDataParser() processContentKeyResponseData:keyData.get() forTrackID:protectedSourceBuffer->protectedTrackID()];
267 ALLOW_DEPRECATED_DECLARATIONS_END
268
269 return true;
270}

272void CDMSessionAVStreamSession::setStreamSession(AVStreamSession *streamSession)
273{
274 if (m_streamSession)
275 [[NSNotificationCenter defaultCenter] removeObserver:m_dataParserObserver.get() name:AVStreamSessionContentProtectionSessionIdentifierChangedNotification object:m_streamSession.get()];
276
277 m_streamSession = streamSession;
278
279 if (!m_streamSession)
280 return;
281
282 [[NSNotificationCenter defaultCenter] addObserver:m_dataParserObserver.get() selector:@selector(contentProtectionSessionIdentifierChanged:) name:AVStreamSessionContentProtectionSessionIdentifierChangedNotification object:m_streamSession.get()];
283
284 NSData* identifier = [streamSession contentProtectionSessionIdentifier];
285 RetainPtr<NSString> sessionIdentifierString = identifier ? adoptNS([[NSString alloc] initWithData:identifier encoding:(NSUTF8StringEncoding)]) : nil;
286 setSessionId(sessionIdentifierString.get());
287}

301RefPtr<Uint8Array> CDMSessionAVStreamSession::generateKeyReleaseMessage(unsigned short& errorCode, uint32_t& systemCode)
302{
303 ASSERT(m_mode == KeyRelease);
304 m_certificate = m_initData;
305 RetainPtr<NSData> certificateData = adoptNS([[NSData alloc] initWithBytes:m_certificate->data() length:m_certificate->length()]);
306
307 String storagePath = this->storagePath();
308 if (storagePath.isEmpty() || ![PAL::getAVStreamSessionClass() respondsToSelector:@selector(pendingExpiredSessionReportsWithAppIdentifier:storageDirectoryAtURL:)]) {
309 errorCode = MediaPlayer::KeySystemNotSupported;
310 systemCode = '!mor';
311 return nullptr;
312 }
313
314 NSArray* expiredSessions = [PAL::getAVStreamSessionClass() pendingExpiredSessionReportsWithAppIdentifier:certificateData.get() storageDirectoryAtURL:[NSURL fileURLWithPath:storagePath]];
315 if (![expiredSessions count]) {
316 LOG(Media, "CDMSessionAVStreamSession::generateKeyReleaseMessage(%p) - no expired sessions found", this);
317
318 errorCode = MediaPlayer::KeySystemNotSupported;
319 systemCode = '!mor';
320 return nullptr;
321 }
322
323 LOG(Media, "CDMSessionAVStreamSession::generateKeyReleaseMessage(%p) - found %d expired sessions", this, [expiredSessions count]);
324
325 errorCode = 0;
326 systemCode = 0;
327 m_expiredSession = [expiredSessions firstObject];
328 return Uint8Array::tryCreate(static_cast<const uint8_t*>([m_expiredSession bytes]), [m_expiredSession length]);
329}
