42class CDMPrivateMediaSourceAVFObjC;

44class CDMSessionMediaSourceAVFObjC : public LegacyCDMSession, public SourceBufferPrivateAVFObjCErrorClient, public CanMakeWeakPtr<CDMSessionMediaSourceAVFObjC> {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 CDMSessionMediaSourceAVFObjC(CDMPrivateMediaSourceAVFObjC&, LegacyCDMSessionClient*);
48 virtual ~CDMSessionMediaSourceAVFObjC();
49
50 virtual void addParser(AVStreamDataParser*) = 0;
51 virtual void removeParser(AVStreamDataParser*) = 0;
52
53 // LegacyCDMSession
54 void setClient(LegacyCDMSessionClient* client) override { m_client = client; }
55 const String& sessionId() const override { return m_sessionId; }
56
57 // SourceBufferPrivateAVFObjCErrorClient
58 void layerDidReceiveError(AVSampleBufferDisplayLayer *, NSError *, bool& shouldIgnore) override;
59 ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
60 void rendererDidReceiveError(AVSampleBufferAudioRenderer *, NSError *, bool& shouldIgnore) override;
61 ALLOW_NEW_API_WITHOUT_GUARDS_END
62
63 void addSourceBuffer(SourceBufferPrivateAVFObjC*);
64 void removeSourceBuffer(SourceBufferPrivateAVFObjC*);
65 void setSessionId(const String& sessionId) { m_sessionId = sessionId; }
66
67 void invalidateCDM() { m_cdm = nullptr; }
68
69protected:
70 String storagePath() const;
71
72 CDMPrivateMediaSourceAVFObjC* m_cdm;
73 LegacyCDMSessionClient* m_client { nullptr };
74 Vector<RefPtr<SourceBufferPrivateAVFObjC>> m_sourceBuffers;
75 RefPtr<Uint8Array> m_certificate;
76 String m_sessionId;
77 bool m_stopped { false };
78};
79
80inline CDMSessionMediaSourceAVFObjC* toCDMSessionMediaSourceAVFObjC(LegacyCDMSession* session)
81{
82 if (!session || (session->type() != CDMSessionTypeAVStreamSession && session->type() != CDMSessionTypeAVContentKeySession))
83 return nullptr;
84 return static_cast<CDMSessionMediaSourceAVFObjC*>(session);
85}
