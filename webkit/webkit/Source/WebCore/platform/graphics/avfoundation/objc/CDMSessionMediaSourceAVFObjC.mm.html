38CDMSessionMediaSourceAVFObjC::CDMSessionMediaSourceAVFObjC(CDMPrivateMediaSourceAVFObjC& cdm, LegacyCDMSessionClient* client)
39 : m_cdm(&cdm)
40 , m_client(client)
41{
42}
43
44CDMSessionMediaSourceAVFObjC::~CDMSessionMediaSourceAVFObjC()
45{
46 if (m_cdm)
47 m_cdm->invalidateSession(this);
48
49 for (auto& sourceBuffer : m_sourceBuffers)
50 sourceBuffer->unregisterForErrorNotifications(this);
51 m_sourceBuffers.clear();
52}
53
54void CDMSessionMediaSourceAVFObjC::layerDidReceiveError(AVSampleBufferDisplayLayer *, NSError *error, bool& shouldIgnore)
55{
56 if (!m_client)
57 return;
58
59 unsigned long code = mediaKeyErrorSystemCode(error);
60
61 // FIXME(142246): Remove the following once <rdar://problem/20027434> is resolved.
62 shouldIgnore = m_stopped && code == 12785;
63 if (!shouldIgnore)
64 m_client->sendError(LegacyCDMSessionClient::MediaKeyErrorDomain, code);
65}
66
67void CDMSessionMediaSourceAVFObjC::rendererDidReceiveError(AVSampleBufferAudioRenderer *, NSError *error, bool& shouldIgnore)
68{
69 if (!m_client)
70 return;
71
72 unsigned long code = mediaKeyErrorSystemCode(error);
73
74 // FIXME(142246): Remove the following once <rdar://problem/20027434> is resolved.
75 shouldIgnore = m_stopped && code == 12785;
76 if (!shouldIgnore)
77 m_client->sendError(LegacyCDMSessionClient::MediaKeyErrorDomain, code);
78}

81void CDMSessionMediaSourceAVFObjC::addSourceBuffer(SourceBufferPrivateAVFObjC* sourceBuffer)
82{
83 ASSERT(!m_sourceBuffers.contains(sourceBuffer));
84 ASSERT(sourceBuffer);
85
86 addParser(sourceBuffer->streamDataParser());
87
88 m_sourceBuffers.append(sourceBuffer);
89 sourceBuffer->registerForErrorNotifications(this);
90}
91
92void CDMSessionMediaSourceAVFObjC::removeSourceBuffer(SourceBufferPrivateAVFObjC* sourceBuffer)
93{
94 ASSERT(m_sourceBuffers.contains(sourceBuffer));
95 ASSERT(sourceBuffer);
96
97 removeParser(sourceBuffer->streamDataParser());
98
99 sourceBuffer->unregisterForErrorNotifications(this);
100 m_sourceBuffers.remove(m_sourceBuffers.find(sourceBuffer));
101}
102
103String CDMSessionMediaSourceAVFObjC::storagePath() const
104{
105 if (!m_client)
106 return emptyString();
107
108 String storageDirectory = m_client->mediaKeysStorageDirectory();
109 if (storageDirectory.isEmpty())
110 return emptyString();
111
112 return FileSystem::pathByAppendingComponent(storageDirectory, "SecureStop.plist");
113}
