64@interface WebCoreSharedBufferResourceLoaderDelegate : NSObject<AVAssetResourceLoaderDelegate> {
65 WebCore::ImageDecoderAVFObjC* _parent;
66 long long _expectedContentSize;

71}

82@implementation WebCoreSharedBufferResourceLoaderDelegate
83- (id)initWithParent:(WebCore::ImageDecoderAVFObjC*)parent
84{
85 if (!(self = [super init]))
86 return nil;
87 _parent = parent;
88
89 return self;
90}
91
92- (NSData*)data
93{
94 return _data.get();
95}
96
97- (void)setExpectedContentSize:(long long)expectedContentSize
98{
99 Locker locker { _dataLock };
100 _expectedContentSize = expectedContentSize;
101
102 [self fulfillPendingRequests];
103}
104
105- (void)updateData:(NSData *)data complete:(BOOL)complete
106{
107 Locker locker { _dataLock };
108 _data = data;
109 _complete = complete;
110
111 [self fulfillPendingRequests];
112}
113
114- (BOOL)canFulfillRequest:(AVAssetResourceLoadingRequest *)request
115{
116 if (!request)
117 return NO;
118
119 if (request.finished || request.cancelled)
120 return NO;
121
122 // AVURLAsset's resource loader requires knowing the expected content size
123 // to load sucessfully. That requires either having the complete data for
124 // the resource, or knowing the expected content size.
125 if (!_complete && !_expectedContentSize)
126 return NO;
127
128 if (auto dataRequest = request.dataRequest) {
129 if (dataRequest.requestedOffset > static_cast<long long>(_data.get().length))
130 return NO;
131 }
132
133 return YES;
134}
135
136- (void)enqueueRequest:(AVAssetResourceLoadingRequest *)loadingRequest
137{
138 ASSERT(!_requests.contains(loadingRequest));
139 _requests.append(loadingRequest);
140}
141
142- (void)fulfillPendingRequests
143{
144 for (auto& request : _requests) {
145 if ([self canFulfillRequest:request.get()])
146 [self fulfillRequest:request.get()];
147 }
148
149 _requests.removeAllMatching([] (auto& request) {
150 return request.get().finished;
151 });
152}
153
154- (void)fulfillRequest:(AVAssetResourceLoadingRequest *)request
155{
156 if (auto infoRequest = request.contentInformationRequest) {
157 infoRequest.contentType = _parent->uti();
158 infoRequest.byteRangeAccessSupported = YES;
159 infoRequest.contentLength = _complete ? _data.get().length : _expectedContentSize;
160 }
161
162 if (auto dataRequest = request.dataRequest) {
163 long long availableLength = _data.get().length - dataRequest.requestedOffset;
164 if (availableLength <= 0)
165 return;
166
167 long long requestedLength;
168 if (dataRequest.requestsAllDataToEndOfResource)
169 requestedLength = availableLength;
170 else
171 requestedLength = std::min<long long>(availableLength, dataRequest.requestedLength);
172
173 auto range = NSMakeRange(static_cast<NSUInteger>(dataRequest.requestedOffset), static_cast<NSUInteger>(requestedLength));
174 NSData* requestedData = [_data subdataWithRange:range];
175 if (!requestedData)
176 return;
177
178 [dataRequest respondWithData:requestedData];
179
180 if (dataRequest.requestsAllDataToEndOfResource) {
181 if (!_complete)
182 return;
183 } else if (dataRequest.requestedOffset + dataRequest.requestedLength > dataRequest.currentOffset)
184 return;
185 }
186
187 [request finishLoading];
188}
189
190- (BOOL)resourceLoader:(AVAssetResourceLoader *)resourceLoader shouldWaitForLoadingOfRequestedResource:(AVAssetResourceLoadingRequest *)loadingRequest
191{
192 Locker locker { _dataLock };
193
194 UNUSED_PARAM(resourceLoader);
195
196 if ([self canFulfillRequest:loadingRequest]) {
197 [self fulfillRequest:loadingRequest];
198 if (loadingRequest.finished)
199 return YES;
200 }
201
202 [self enqueueRequest:loadingRequest];
203 return YES;
204}
205
206- (void)resourceLoader:(AVAssetResourceLoader *)resourceLoader didCancelLoadingRequest:(AVAssetResourceLoadingRequest *)loadingRequest
207{
208 Locker locker { _dataLock };
209
210 UNUSED_PARAM(resourceLoader);
211 _requests.removeAll(loadingRequest);
212}
213@end

234class ImageDecoderAVFObjCSample : public MediaSampleAVFObjC {
235public:
236 static Ref<ImageDecoderAVFObjCSample> create(RetainPtr<CMSampleBufferRef>&& sampleBuffer)
237 {
238 return adoptRef(*new ImageDecoderAVFObjCSample(WTFMove(sampleBuffer)));
239 }
240
241 CGImageRef image() const { return m_image.get(); }
242 void setImage(RetainPtr<CGImageRef>&& image)
243 {
244 m_image = WTFMove(image);
245 if (!m_image) {
246 m_hasAlpha = false;
247 return;
248 }
249
250 auto alphaInfo = CGImageGetAlphaInfo(m_image.get());
251 m_hasAlpha = alphaInfo != kCGImageAlphaNone && alphaInfo != kCGImageAlphaNoneSkipLast && alphaInfo != kCGImageAlphaNoneSkipFirst;
252 }
253
254 std::optional<ByteRange> byteRange() const final
255 {
256 if (PAL::CMSampleBufferGetDataBuffer(m_sample.get())
257 || PAL::CMSampleBufferGetImageBuffer(m_sample.get())
258 || !PAL::CMSampleBufferDataIsReady(m_sample.get()))
259 return std::nullopt;
260
261 return byteRangeForAttachment(PAL::kCMSampleBufferAttachmentKey_SampleReferenceByteOffset);
262 }
263
264 SampleFlags flags() const override
265 {
266 return (SampleFlags)(MediaSampleAVFObjC::flags() | (m_hasAlpha ? HasAlpha : 0));
267 }
268
269private:
270 ImageDecoderAVFObjCSample(RetainPtr<CMSampleBufferRef>&& sample)
271 : MediaSampleAVFObjC(WTFMove(sample))
272 {
273 }
274
275 RetainPtr<CGImageRef> m_image;
276 bool m_hasAlpha { false };
277};

292RefPtr<ImageDecoderAVFObjC> ImageDecoderAVFObjC::create(SharedBuffer& data, const String& mimeType, AlphaOption alphaOption, GammaAndColorProfileOption gammaAndColorProfileOption)
293{
294 // AVFoundation may not be available at runtime.
295 if (!AVAssetMIMETypeCache::singleton().isAvailable())
296 return nullptr;
297
298 if (!canLoad_VideoToolbox_VTCreateCGImageFromCVPixelBuffer())
299 return nullptr;
300
301 return adoptRef(*new ImageDecoderAVFObjC(data, mimeType, alphaOption, gammaAndColorProfileOption));
302}

304ImageDecoderAVFObjC::ImageDecoderAVFObjC(SharedBuffer& data, const String& mimeType, AlphaOption, GammaAndColorProfileOption)
305 : ImageDecoder()
306 , m_mimeType(mimeType)
307 , m_uti(WebCore::UTIFromMIMEType(mimeType))
308 , m_asset(adoptNS([PAL::allocAVURLAssetInstance() initWithURL:customSchemeURL() options:imageDecoderAssetOptions()]))
309 , m_loader(adoptNS([[WebCoreSharedBufferResourceLoaderDelegate alloc] initWithParent:this]))
310 , m_decompressionSession(WebCoreDecompressionSession::createRGB())
311{
312 [m_loader updateData:data.createNSData().get() complete:NO];
313
314 [m_asset.get().resourceLoader setDelegate:m_loader.get() queue:dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0)];
315 [m_asset loadValuesAsynchronouslyForKeys:@[@"tracks"] completionHandler:[protectedThis = makeRefPtr(this)] () mutable {
316 callOnMainThread([protectedThis = WTFMove(protectedThis)] {
317 protectedThis->setTrack(protectedThis->firstEnabledTrack());
318 });
319 }];
320}

339AVAssetTrack *ImageDecoderAVFObjC::firstEnabledTrack()
340{
341 NSArray<AVAssetTrack *> *videoTracks = [m_asset tracksWithMediaCharacteristic:AVMediaCharacteristicVisual];
342 NSUInteger firstEnabledIndex = [videoTracks indexOfObjectPassingTest:^(AVAssetTrack *track, NSUInteger, BOOL*) {
343 return track.enabled;
344 }];
345
346 if (firstEnabledIndex == NSNotFound) {
347 LOG(Images, "ImageDecoderAVFObjC::firstEnabledTrack(%p) - asset has no enabled video tracks", this);
348 return nil;
349 }
350
351 return [videoTracks objectAtIndex:firstEnabledIndex];
352}

354void ImageDecoderAVFObjC::readSamples()
355{
356 if (!m_sampleData.empty())
357 return;
358
359 auto assetReader = adoptNS([PAL::allocAVAssetReaderInstance() initWithAsset:m_asset.get() error:nil]);
360 auto referenceOutput = adoptNS([PAL::allocAVAssetReaderSampleReferenceOutputInstance() initWithTrack:m_track.get()]);
361
362 referenceOutput.get().alwaysCopiesSampleData = NO;
363 [assetReader addOutput:referenceOutput.get()];
364 [assetReader startReading];
365
366 while (auto sampleBuffer = adoptCF([referenceOutput copyNextSampleBuffer])) {
367 // NOTE: Some samples emitted by the AVAssetReader simply denote the boundary of edits
368 // and do not carry media data.
369 if (!(PAL::CMSampleBufferGetNumSamples(sampleBuffer.get())))
370 continue;
371 m_sampleData.addSample(ImageDecoderAVFObjCSample::create(WTFMove(sampleBuffer)).get());
372 }
373
374 if (m_encodedDataStatusChangedCallback)
375 m_encodedDataStatusChangedCallback(encodedDataStatus());
376}

378void ImageDecoderAVFObjC::readTrackMetadata()
379{
380 AffineTransform finalTransform = CGAffineTransformConcat(m_asset.get().preferredTransform, m_track.get().preferredTransform);
381 auto size = expandedIntSize(FloatSize(m_track.get().naturalSize));
382 if (finalTransform.isIdentity()) {
383 m_size = size;
384 m_imageRotationSession = nullptr;
385 return;
386 }
387
388 if (!m_imageRotationSession
389 || !m_imageRotationSession->transform()
390 || m_imageRotationSession->transform().value() != finalTransform
391 || m_imageRotationSession->size() != size)
392 m_imageRotationSession = makeUnique<ImageRotationSessionVT>(WTFMove(finalTransform), size, ImageRotationSessionVT::IsCGImageCompatible::Yes);
393
394 m_size = expandedIntSize(m_imageRotationSession->rotatedSize());
395}

397bool ImageDecoderAVFObjC::storeSampleBuffer(CMSampleBufferRef sampleBuffer)
398{
399 auto pixelBuffer = m_decompressionSession->decodeSampleSync(sampleBuffer);
400 if (!pixelBuffer) {
401 RELEASE_LOG_ERROR(Images, "ImageDecoderAVFObjC::storeSampleBuffer(%p) - could not decode sampleBuffer", this);
402 return false;
403 }
404
405 auto presentationTime = PAL::toMediaTime(PAL::CMSampleBufferGetOutputPresentationTimeStamp(sampleBuffer));
406 auto iter = m_sampleData.presentationOrder().findSampleWithPresentationTime(presentationTime);
407
408 if (m_imageRotationSession)
409 pixelBuffer = m_imageRotationSession->rotate(pixelBuffer.get());
410
411 CGImageRef rawImage = nullptr;
412 if (noErr != VTCreateCGImageFromCVPixelBuffer(pixelBuffer.get(), nullptr, &rawImage)) {
413 RELEASE_LOG_ERROR(Images, "ImageDecoderAVFObjC::storeSampleBuffer(%p) - could not create CGImage from pixelBuffer", this);
414 return false;
415 }
416
417 if (iter == m_sampleData.presentationOrder().end()) {
418 RELEASE_LOG_ERROR(Images, "ImageDecoderAVFObjC::storeSampleBuffer(%p) - could not find sample buffer entry with specified presentation time", this);
419 return false;
420 }
421
422 toSample(iter)->setImage(adoptCF(rawImage));
423
424 return true;
425}

427void ImageDecoderAVFObjC::advanceCursor()
428{
429 if (m_cursor == m_sampleData.decodeOrder().end() || ++m_cursor == m_sampleData.decodeOrder().end())
430 m_cursor = m_sampleData.decodeOrder().begin();
431}

433void ImageDecoderAVFObjC::setTrack(AVAssetTrack *track)
434{
435 if (m_track == track)
436 return;
437 m_track = track;
438
439 Locker locker { m_sampleGeneratorLock };
440 m_sampleData.clear();
441 m_size.reset();
442 m_cursor = m_sampleData.decodeOrder().end();
443 m_imageRotationSession = nullptr;
444
445 [track loadValuesAsynchronouslyForKeys:@[@"naturalSize", @"preferredTransform"] completionHandler:[protectedThis = makeRefPtr(this)] () mutable {
446 callOnMainThread([protectedThis = WTFMove(protectedThis)] {
447 protectedThis->readTrackMetadata();
448 protectedThis->readSamples();
449 });
450 }];
451}

561PlatformImagePtr ImageDecoderAVFObjC::createFrameImageAtIndex(size_t index, SubsamplingLevel, const DecodingOptions&)
562{
563 Locker locker { m_sampleGeneratorLock };
564
565 auto* sampleData = sampleAtIndex(index);
566 if (!sampleData)
567 return nullptr;
568
569 if (auto image = sampleData->image())
570 return image;
571
572 if (m_cursor == m_sampleData.decodeOrder().end())
573 m_cursor = m_sampleData.decodeOrder().begin();
574
575 auto decodeTime = sampleData->decodeTime();
576
577 if (decodeTime < m_cursor->second->decodeTime()) {
578 // Rewind cursor to the last sync sample to begin decoding
579 m_cursor = m_sampleData.decodeOrder().findSampleWithDecodeKey({decodeTime, sampleData->presentationTime()});
580 do {
581 if (m_cursor->second->isSync())
582 break;
583 } while (--m_cursor != m_sampleData.decodeOrder().begin());
584 }
585
586 while (true) {
587 if (decodeTime < m_cursor->second->decodeTime())
588 return nullptr;
589
590 auto cursorSample = toSample(m_cursor);
591 if (!cursorSample)
592 return nullptr;
593
594 if (!sampleIsComplete(*cursorSample))
595 return nullptr;
596
597 if (auto byteRange = cursorSample->byteRange()) {
598 auto& byteRangeValue = byteRange.value();
599 auto* data = m_loader.get().data;
600 CMBlockBufferCustomBlockSource source {
601 0,
602 nullptr,
603 [](void* refcon, void*, size_t) {
604 [(id)refcon release];
605 },
606 [data retain]
607 };
608 CMBlockBufferRef rawBlockBuffer = nullptr;
609 if (noErr != PAL::CMBlockBufferCreateWithMemoryBlock(kCFAllocatorDefault, const_cast<void*>(data.bytes), data.length, nullptr, &source, byteRangeValue.byteOffset, byteRangeValue.byteLength, 0, &rawBlockBuffer))
610 return nullptr;
611
612 if (!rawBlockBuffer)
613 return nullptr;
614
615 if (noErr != PAL::CMSampleBufferSetDataBuffer(cursorSample->sampleBuffer(), rawBlockBuffer))
616 return nullptr;
617 CFRelease(rawBlockBuffer);
618
619 PAL::CMRemoveAttachment(cursorSample->sampleBuffer(), PAL::kCMSampleBufferAttachmentKey_SampleReferenceByteOffset);
620 PAL::CMRemoveAttachment(cursorSample->sampleBuffer(), PAL::kCMSampleBufferAttachmentKey_SampleReferenceURL);
621 }
622
623 auto cursorSampleBuffer = cursorSample->sampleBuffer();
624 if (!cursorSampleBuffer)
625 break;
626
627 if (!storeSampleBuffer(cursorSampleBuffer))
628 break;
629
630 advanceCursor();
631 if (auto image = sampleData->image())
632 return image;
633 }
634
635 advanceCursor();
636 return nullptr;
637}

663{

672const ImageDecoderAVFObjCSample* ImageDecoderAVFObjC::sampleAtIndex(size_t index) const
673{
674 if (index >= m_sampleData.presentationOrder().size())
675 return nullptr;
676
677 // FIXME: std::map is not random-accessible; this can get expensive if callers repeatedly call
678 // with monotonically increasing indexes. Investigate adding an O(1) side structure to make this
679 // style of access faster.
680 auto iter = m_sampleData.presentationOrder().begin();
681 for (size_t i = 0; i != index; ++i)
682 ++iter;
683
684 return toSample(iter);
685}

687bool ImageDecoderAVFObjC::sampleIsComplete(const ImageDecoderAVFObjCSample& sample) const
688{
689 if (auto byteRange = sample.byteRange()) {
690 auto& byteRangeValue = byteRange.value();
691 return byteRangeValue.byteOffset + byteRangeValue.byteLength <= m_loader.get().data.length;
692 }
693
694 return PAL::CMSampleBufferDataIsReady(sample.sampleBuffer());
695}
