38class InbandTextTrackPrivateAVFObjC : public InbandTextTrackPrivateAVF {

54 AtomString language() const override;
55 bool isDefault() const override;
56
57 void disconnect() override;

59 Category textTrackCategory() const override { return InBand; }

62
63protected:
64 InbandTextTrackPrivateAVFObjC(AVFInbandTrackParent*, AVMediaSelectionOption *, InbandTextTrackPrivate::CueFormat);
65
66 RetainPtr<AVMediaSelectionOption> m_mediaSelectionOption;
