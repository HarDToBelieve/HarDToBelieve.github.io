48@interface WebAVSampleBufferStatusChangeListener : NSObject {
49 WeakPtr<LocalSampleBufferDisplayLayer> _parent;
50}
51
52- (id)initWithParent:(LocalSampleBufferDisplayLayer*)callback;
53- (void)invalidate;
54- (void)begin;
55- (void)stop;
56@end
57
58@implementation WebAVSampleBufferStatusChangeListener
59
60- (id)initWithParent:(LocalSampleBufferDisplayLayer*)parent
61{
62 if (!(self = [super init]))
63 return nil;
64
65 _parent = makeWeakPtr(parent);
66
67 return self;
68}
69
70- (void)dealloc
71{
72 [self invalidate];
73 [super dealloc];
74}
75
76- (void)invalidate
77{
78 [self stop];
79 _parent = nullptr;
80}
81
82- (void)begin
83{
84 ASSERT(_parent);
85 ASSERT(_parent->displayLayer());
86 ASSERT(_parent->rootLayer());
87
88 [_parent->displayLayer() addObserver:self forKeyPath:@"status" options:NSKeyValueObservingOptionNew context:nil];
89 [_parent->displayLayer() addObserver:self forKeyPath:@"error" options:NSKeyValueObservingOptionNew context:nil];
90}
91
92- (void)stop
93{
94 if (!_parent)
95 return;
96
97 if (_parent->displayLayer()) {
98 [_parent->displayLayer() removeObserver:self forKeyPath:@"status"];
99 [_parent->displayLayer() removeObserver:self forKeyPath:@"error"];
100 }
101}
102
103- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
104{
105 UNUSED_PARAM(context);
106 UNUSED_PARAM(keyPath);
107 UNUSED_PARAM(change);
108
109 if (![object isKindOfClass:PAL::getAVSampleBufferDisplayLayerClass()])
110 return;
111
112 if ([keyPath isEqualToString:@"status"]) {
113 callOnMainThread([protectedSelf = RetainPtr<WebAVSampleBufferStatusChangeListener>(self)] {
114 if (protectedSelf->_parent)
115 protectedSelf->_parent->layerStatusDidChange();
116 });
117 return;
118 }
119
120 if ([keyPath isEqualToString:@"error"]) {
121 callOnMainThread([protectedSelf = RetainPtr<WebAVSampleBufferStatusChangeListener>(self)] {
122 if (protectedSelf->_parent)
123 protectedSelf->_parent->layerErrorDidChange();
124 });
125 return;
126 }
127}
128@end

132static void runWithoutAnimations(const WTF::Function<void()>& function)
133{
134 [CATransaction begin];
135 [CATransaction setAnimationDuration:0];
136 [CATransaction setDisableActions:YES];
137 function();
138 [CATransaction commit];
139}

141std::unique_ptr<LocalSampleBufferDisplayLayer> LocalSampleBufferDisplayLayer::create(Client& client)
142{
143 auto sampleBufferDisplayLayer = adoptNS([PAL::allocAVSampleBufferDisplayLayerInstance() init]);
144 if (!sampleBufferDisplayLayer)

150LocalSampleBufferDisplayLayer::LocalSampleBufferDisplayLayer(RetainPtr<AVSampleBufferDisplayLayer>&& sampleBufferDisplayLayer, Client& client)
151 : SampleBufferDisplayLayer(client)
152 , m_statusChangeListener(adoptNS([[WebAVSampleBufferStatusChangeListener alloc] initWithParent:this]))
153 , m_sampleBufferDisplayLayer(WTFMove(sampleBufferDisplayLayer))
154 , m_processingQueue(WorkQueue::create("LocalSampleBufferDisplayLayer queue"))
155#if !RELEASE_LOG_DISABLED

162void LocalSampleBufferDisplayLayer::initialize(bool hideRootLayer, IntSize size, CompletionHandler<void(bool didSucceed)>&& callback)
163{
164 m_sampleBufferDisplayLayer.get().backgroundColor = cachedCGColor(Color::black);
165 m_sampleBufferDisplayLayer.get().anchorPoint = { .5, .5 };
166 m_sampleBufferDisplayLayer.get().needsDisplayOnBoundsChange = YES;
167 m_sampleBufferDisplayLayer.get().videoGravity = AVLayerVideoGravityResizeAspectFill;
168
169 m_rootLayer = adoptNS([[CALayer alloc] init]);
170 m_rootLayer.get().hidden = hideRootLayer;
171
172 m_rootLayer.get().backgroundColor = cachedCGColor(Color::black);
173 m_rootLayer.get().needsDisplayOnBoundsChange = YES;
174
175 m_rootLayer.get().bounds = CGRectMake(0, 0, size.width(), size.height());
176
177 [m_statusChangeListener begin];
178
179 [m_rootLayer addSublayer:m_sampleBufferDisplayLayer.get()];
180
181#ifndef NDEBUG
182 [m_sampleBufferDisplayLayer setName:@"LocalSampleBufferDisplayLayer AVSampleBufferDisplayLayer"];
183 [m_rootLayer setName:@"LocalSampleBufferDisplayLayer AVSampleBufferDisplayLayer parent"];
184#endif
185 callback(true);
186}

188LocalSampleBufferDisplayLayer::~LocalSampleBufferDisplayLayer()
189{
190 ASSERT(isMainThread());
191
192 m_processingQueue->dispatchSync([] { });
193
194 m_processingQueue = nullptr;
195
196 [m_statusChangeListener stop];
197
198 m_pendingVideoSampleQueue.clear();
199
200 [m_sampleBufferDisplayLayer stopRequestingMediaData];
201 [m_sampleBufferDisplayLayer flush];
202 m_sampleBufferDisplayLayer = nullptr;
203
204 m_rootLayer = nullptr;
205}

207void LocalSampleBufferDisplayLayer::layerStatusDidChange()
208{
209 ASSERT(isMainThread());
210 if (m_sampleBufferDisplayLayer.get().status != AVQueuedSampleBufferRenderingStatusRendering)
211 return;
212 if (!m_client)
213 return;
214 m_client->sampleBufferDisplayLayerStatusDidChange(*this);
215}

273void LocalSampleBufferDisplayLayer::updateRootLayerBoundsAndPosition(CGRect bounds, MediaSample::VideoRotation rotation, ShouldUpdateRootLayer shouldUpdateRootLayer)
274{
275 runWithoutAnimations([&] {
276 CGPoint position = { bounds.size.width / 2, bounds.size.height / 2};
277
278 if (shouldUpdateRootLayer == ShouldUpdateRootLayer::Yes) {
279 m_rootLayer.get().position = position;
280 m_rootLayer.get().bounds = bounds;
281 }
282
283 if (rotation == MediaSample::VideoRotation::Right || rotation == MediaSample::VideoRotation::Left)
284 std::swap(bounds.size.width, bounds.size.height);
285
286 m_sampleBufferDisplayLayer.get().position = position;
287 m_sampleBufferDisplayLayer.get().bounds = bounds;
288 });
289}

305static const double rendererLatency = 0.02;
306
307void LocalSampleBufferDisplayLayer::enqueueSample(MediaSample& sample)
308{
309 if (m_paused) {
310#if !RELEASE_LOG_DISABLED
311 m_frameRateMonitor.update();
312#endif
313 return;
314 }
315
316 m_processingQueue->dispatch([this, sample = makeRef(sample)] {
317 if (![m_sampleBufferDisplayLayer isReadyForMoreMediaData]) {
318 WTFLogAlways("LocalSampleBufferDisplayLayer::enqueueSample not ready for more media data");
319 addSampleToPendingQueue(sample);
320 requestNotificationWhenReadyForVideoData();
321 return;
322 }
323 enqueueSampleBuffer(sample);
324 });
325}
326
327void LocalSampleBufferDisplayLayer::enqueueSampleBuffer(MediaSample& sample)
328{
329 ASSERT(!isMainThread());
330
331 auto sampleToEnqueue = sample.platformSample().sample.cmSampleBuffer;
332 auto now = MediaTime::createWithDouble(MonotonicTime::now().secondsSinceEpoch().value() + rendererLatency);
333
334 // If needed, we set the sample buffer to kCMSampleAttachmentKey_DisplayImmediately as a workaround to rdar://problem/49274083.
335 // We clone the sample buffer as modifying the attachments of a sample buffer used elsewhere (encoding e.g.) may not be thread safe.
336 RetainPtr<CMSampleBufferRef> newSampleBuffer;
337 if (m_renderPolicy == RenderPolicy::Immediately || now >= sample.presentationTime()) {
338 newSampleBuffer = MediaSampleAVFObjC::cloneSampleBufferAndSetAsDisplayImmediately(sampleToEnqueue);
339 sampleToEnqueue = newSampleBuffer.get();
340 }
341
342 [m_sampleBufferDisplayLayer enqueueSampleBuffer:sampleToEnqueue];
343
344#if !RELEASE_LOG_DISABLED
345 m_frameRateMonitor.update();
346#endif
347}

349#if !RELEASE_LOG_DISABLED
350void LocalSampleBufferDisplayLayer::onIrregularFrameRateNotification(MonotonicTime frameTime, MonotonicTime lastFrameTime)
351{
352 callOnMainThread([frameTime = frameTime.secondsSinceEpoch().value(), lastFrameTime = lastFrameTime.secondsSinceEpoch().value(), observedFrameRate = m_frameRateMonitor.observedFrameRate(), frameCount = m_frameRateMonitor.frameCount()] {
353 WTFLogAlways("LocalSampleBufferDisplayLayer::enqueueSampleBuffer at %f, previous frame was at %f, observed frame rate is %f, delay since last frame is %f ms, frame count is %lu", frameTime, lastFrameTime, observedFrameRate, (frameTime - lastFrameTime) * 1000, frameCount);
354 });
355}

358void LocalSampleBufferDisplayLayer::removeOldSamplesFromPendingQueue()
359{
360 ASSERT(!isMainThread());
361
362 if (m_pendingVideoSampleQueue.isEmpty())
363 return;
364
365 if (m_renderPolicy == RenderPolicy::Immediately) {
366 m_pendingVideoSampleQueue.clear();
367 return;
368 }
369
370 auto now = MediaTime::createWithDouble(MonotonicTime::now().secondsSinceEpoch().value());
371 while (!m_pendingVideoSampleQueue.isEmpty()) {
372 auto presentationTime = m_pendingVideoSampleQueue.first()->presentationTime();
373 if (presentationTime.isValid() && presentationTime > now)
374 break;
375 m_pendingVideoSampleQueue.removeFirst();
376 }
377}
378
379void LocalSampleBufferDisplayLayer::addSampleToPendingQueue(MediaSample& sample)
380{
381 ASSERT(!isMainThread());
382
383 removeOldSamplesFromPendingQueue();
384 m_pendingVideoSampleQueue.append(sample);
385}

394void LocalSampleBufferDisplayLayer::requestNotificationWhenReadyForVideoData()
395{
396 auto weakThis = makeWeakPtr(*this);
397 [m_sampleBufferDisplayLayer requestMediaDataWhenReadyOnQueue:dispatch_get_main_queue() usingBlock:^{
398 if (!weakThis)
399 return;
400
401 m_processingQueue->dispatch([this] {
402 [m_sampleBufferDisplayLayer stopRequestingMediaData];
403
404 while (!m_pendingVideoSampleQueue.isEmpty()) {
405 if (![m_sampleBufferDisplayLayer isReadyForMoreMediaData]) {
406 requestNotificationWhenReadyForVideoData();
407 return;
408 }
409 auto sample = m_pendingVideoSampleQueue.takeFirst();
410 enqueueSampleBuffer(sample);
411 }
412 });
413 }];
414}
