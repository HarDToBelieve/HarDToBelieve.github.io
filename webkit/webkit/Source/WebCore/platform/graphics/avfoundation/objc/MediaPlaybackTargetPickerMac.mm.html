54MediaPlaybackTargetPickerMac::MediaPlaybackTargetPickerMac(MediaPlaybackTargetPicker::Client& client)

64bool MediaPlaybackTargetPickerMac::externalOutputDeviceAvailable()
65{
66 return routePicker().externalOutputDeviceAvailable();
67}
68
69Ref<MediaPlaybackTarget> MediaPlaybackTargetPickerMac::playbackTarget()
70{
71 return WebCore::MediaPlaybackTargetCocoa::create(routePicker().outputContext());
72}
73
74AVPlaybackTargetPicker& MediaPlaybackTargetPickerMac::routePicker()
75{
76 if (m_routePicker)
77 return *m_routePicker;
78
79#if HAVE(AVROUTEPICKERVIEW)
80 if (AVRoutePickerViewTargetPicker::isAvailable())
81 m_routePicker = makeUnique<AVRoutePickerViewTargetPicker>(*this);
82 else
83#endif
84 m_routePicker = makeUnique<AVOutputDeviceMenuControllerTargetPicker>(*this);
85
86 return *m_routePicker;
87}
