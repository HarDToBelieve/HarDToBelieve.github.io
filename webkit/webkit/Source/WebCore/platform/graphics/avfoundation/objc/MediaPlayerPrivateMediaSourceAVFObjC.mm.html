70String convertEnumerationToString(MediaPlayerPrivateMediaSourceAVFObjC::SeekState enumerationValue)
71{
72 static const NeverDestroyed<String> values[] = {
73 MAKE_STATIC_STRING_IMPL("Seeking"),
74 MAKE_STATIC_STRING_IMPL("WaitingForAvailableFame"),
75 MAKE_STATIC_STRING_IMPL("SeekCompleted"),
76 };
77 static_assert(static_cast<size_t>(MediaPlayerPrivateMediaSourceAVFObjC::SeekState::Seeking) == 0, "MediaPlayerPrivateMediaSourceAVFObjC::SeekState::Seeking is not 0 as expected");
78 static_assert(static_cast<size_t>(MediaPlayerPrivateMediaSourceAVFObjC::SeekState::WaitingForAvailableFame) == 1, "MediaPlayerPrivateMediaSourceAVFObjC::SeekState::WaitingForAvailableFame is not 1 as expected");
79 static_assert(static_cast<size_t>(MediaPlayerPrivateMediaSourceAVFObjC::SeekState::SeekCompleted) == 2, "MediaPlayerPrivateMediaSourceAVFObjC::SeekState::SeekCompleted is not 2 as expected");
80 ASSERT(static_cast<size_t>(enumerationValue) < WTF_ARRAY_LENGTH(values));
81 return values[static_cast<size_t>(enumerationValue)];
82}

105 EffectiveRateChangedListener(MediaPlayerPrivateMediaSourceAVFObjC&, CMTimebaseRef);

129MediaPlayerPrivateMediaSourceAVFObjC::MediaPlayerPrivateMediaSourceAVFObjC(MediaPlayer* player)
130 : m_player(player)
131 , m_synchronizer(adoptNS([PAL::allocAVSampleBufferRenderSynchronizerInstance() init]))
132 , m_seekTimer(*this, &MediaPlayerPrivateMediaSourceAVFObjC::seekInternal)
133 , m_networkState(MediaPlayer::NetworkState::Empty)
134 , m_readyState(MediaPlayer::ReadyState::HaveNothing)
135 , m_rate(1)
136 , m_playing(0)
137 , m_seeking(false)
138 , m_loadingProgressed(false)
139 , m_logger(player->mediaPlayerLogger())
140 , m_logIdentifier(player->mediaPlayerLogIdentifier())
141 , m_videoLayerManager(makeUnique<VideoLayerManagerObjC>(m_logger, m_logIdentifier))
142 , m_effectiveRateChangedListener(EffectiveRateChangedListener::create(*this, [m_synchronizer timebase]))
143{
144 auto logSiteIdentifier = LOGIDENTIFIER;
145 ALWAYS_LOG(logSiteIdentifier);
146 UNUSED_PARAM(logSiteIdentifier);
147
148 // addPeriodicTimeObserverForInterval: throws an exception if you pass a non-numeric CMTime, so just use
149 // an arbitrarily large time value of once an hour:
150 __block auto weakThis = makeWeakPtr(*this);
151 m_timeJumpedObserver = [m_synchronizer addPeriodicTimeObserverForInterval:PAL::toCMTime(MediaTime::createWithDouble(3600)) queue:dispatch_get_main_queue() usingBlock:^(CMTime time) {
152#if LOG_DISABLED
153 UNUSED_PARAM(time);
154#endif
155 // FIXME: Remove the below once <rdar://problem/15798050> is fixed.
156 if (!weakThis)
157 return;
158
159 DEBUG_LOG(logSiteIdentifier, "synchronizer fired for ", PAL::toMediaTime(time), ", seeking = ", m_seeking, ", pending = ", !!m_pendingSeek);
160
161 if (m_seeking && !m_pendingSeek) {
162 m_seeking = false;
163
164 if (shouldBePlaying())
165 [m_synchronizer setRate:m_rate];
166 if (!seeking() && m_seekCompleted == SeekCompleted)
167 m_player->timeChanged();
168 }
169
170 if (m_pendingSeek)
171 seekInternal();
172
173 if (m_currentTimeDidChangeCallback)
174 m_currentTimeDidChangeCallback(CMTIME_IS_NUMERIC(time) ? PAL::toMediaTime(time) : MediaTime::zeroTime());
175 }];
176}
177
178MediaPlayerPrivateMediaSourceAVFObjC::~MediaPlayerPrivateMediaSourceAVFObjC()
179{
180 ALWAYS_LOG(LOGIDENTIFIER);
181
182 m_effectiveRateChangedListener->stop([m_synchronizer timebase]);
183
184 if (m_timeJumpedObserver)
185 [m_synchronizer removeTimeObserver:m_timeJumpedObserver.get()];
186 if (m_timeChangedObserver)
187 [m_synchronizer removeTimeObserver:m_timeChangedObserver.get()];
188 if (m_durationObserver)
189 [m_synchronizer removeTimeObserver:m_durationObserver.get()];
190 flushPendingSizeChanges();
191
192 destroyLayer();
193 destroyDecompressionSession();
194
195 m_seekTimer.stop();
196}

252MediaPlayer::SupportsType MediaPlayerPrivateMediaSourceAVFObjC::supportsTypeAndCodecs(const MediaEngineSupportParameters& parameters)
253{
254 // This engine does not support non-media-source sources.
255 if (!parameters.isMediaSource)
256 return MediaPlayer::SupportsType::IsNotSupported;
257
258 auto supported = SourceBufferParser::isContentTypeSupported(parameters.type);
259
260 if (supported != MediaPlayer::SupportsType::IsSupported)
261 return supported;
262
263 if (!contentTypeMeetsHardwareDecodeRequirements(parameters.type, parameters.contentTypesRequiringHardwareSupport))
264 return MediaPlayer::SupportsType::IsNotSupported;
265
266 return MediaPlayer::SupportsType::IsSupported;
267}

320void MediaPlayerPrivateMediaSourceAVFObjC::playInternal()
321{
322 if (!m_mediaSourcePrivate)
323 return;
324
325 if (currentMediaTime() >= m_mediaSourcePrivate->duration()) {
326 ALWAYS_LOG(LOGIDENTIFIER, "bailing, current time: ", currentMediaTime(), " greater than duration ", m_mediaSourcePrivate->duration());
327 return;
328 }
329
330 ALWAYS_LOG(LOGIDENTIFIER);
331#if PLATFORM(IOS_FAMILY)
332 m_mediaSourcePrivate->flushActiveSourceBuffersIfNeeded();
333#endif
334 m_playing = true;
335 if (shouldBePlaying())
336 [m_synchronizer setRate:m_rate];
337}

453void MediaPlayerPrivateMediaSourceAVFObjC::seekWithTolerance(const MediaTime& time, const MediaTime& negativeThreshold, const MediaTime& positiveThreshold)
454{
455 INFO_LOG(LOGIDENTIFIER, "time = ", time, ", negativeThreshold = ", negativeThreshold, ", positiveThreshold = ", positiveThreshold);
456
457 m_seeking = true;
458 m_pendingSeek = makeUnique<PendingSeek>(time, negativeThreshold, positiveThreshold);
459
460 if (m_seekTimer.isActive())
461 m_seekTimer.stop();
462 m_seekTimer.startOneShot(0_s);
463}

465void MediaPlayerPrivateMediaSourceAVFObjC::seekInternal()
466{
467 std::unique_ptr<PendingSeek> pendingSeek;
468 pendingSeek.swap(m_pendingSeek);
469
470 if (!pendingSeek)
471 return;
472
473 if (!m_mediaSourcePrivate)
474 return;
475
476 if (!pendingSeek->negativeThreshold && !pendingSeek->positiveThreshold)
477 m_lastSeekTime = pendingSeek->targetTime;
478 else
479 m_lastSeekTime = m_mediaSourcePrivate->fastSeekTimeForMediaTime(pendingSeek->targetTime, pendingSeek->positiveThreshold, pendingSeek->negativeThreshold);
480
481 if (m_lastSeekTime.hasDoubleValue())
482 m_lastSeekTime = MediaTime::createWithDouble(m_lastSeekTime.toDouble(), MediaTime::DefaultTimeScale);
483
484 MediaTime synchronizerTime = PAL::toMediaTime(PAL::CMTimebaseGetTime([m_synchronizer timebase]));
485 INFO_LOG(LOGIDENTIFIER, "seekTime = ", m_lastSeekTime, ", synchronizerTime = ", synchronizerTime);
486
487 bool doesNotRequireSeek = synchronizerTime == m_lastSeekTime;
488
489 m_mediaSourcePrivate->willSeek();
490 [m_synchronizer setRate:0 time:PAL::toCMTime(m_lastSeekTime)];
491 m_mediaSourcePrivate->seekToTime(m_lastSeekTime);
492
493 // In cases where the destination seek time precisely matches the synchronizer's existing time
494 // no time jumped notification will be issued. In this case, just notify the MediaPlayer that
495 // the seek completed successfully.
496 if (doesNotRequireSeek) {
497 m_seeking = false;
498
499 if (shouldBePlaying())
500 [m_synchronizer setRate:m_rate];
501 if (!seeking() && m_seekCompleted)
502 m_player->timeChanged();
503 }
504}

514void MediaPlayerPrivateMediaSourceAVFObjC::seekCompleted()
515{
516 if (m_seekCompleted == SeekCompleted)
517 return;
518 if (hasVideo() && !m_hasAvailableVideoFrame) {
519 ALWAYS_LOG(LOGIDENTIFIER, "waiting for video frame");
520 m_seekCompleted = WaitingForAvailableFame;
521 return;
522 }
523 ALWAYS_LOG(LOGIDENTIFIER);
524 m_seekCompleted = SeekCompleted;
525 if (shouldBePlaying())
526 [m_synchronizer setRate:m_rate];
527 if (!m_seeking)
528 m_player->timeChanged();
529}

594
595bool MediaPlayerPrivateMediaSourceAVFObjC::updateLastPixelBuffer()
596{
597#if HAVE(AVSAMPLEBUFFERVIDEOOUTPUT)
598 if (m_videoOutput) {
599 CMTime outputTime;
600 if (auto pixelBuffer = adoptCF([m_videoOutput copyPixelBufferForSourceTime:toCMTime(currentMediaTime()) sourceTimeForDisplay:&outputTime])) {
601 INFO_LOG(LOGIDENTIFIER, "new pixelbuffer found for time ", PAL::toMediaTime(outputTime));
602 m_lastPixelBuffer = WTFMove(pixelBuffer);
603 return true;
604 }
605 }
606#endif
607
608 if (m_sampleBufferDisplayLayer || !m_decompressionSession)
609 return false;
610
611 auto flags = !m_lastPixelBuffer ? WebCoreDecompressionSession::AllowLater : WebCoreDecompressionSession::ExactTime;
612 auto newPixelBuffer = m_decompressionSession->imageForTime(currentMediaTime(), flags);
613 if (!newPixelBuffer)
614 return false;
615
616 m_lastPixelBuffer = newPixelBuffer;
617 return true;
618}
619
620bool MediaPlayerPrivateMediaSourceAVFObjC::updateLastImage()
621{
622 if (!updateLastPixelBuffer())
623 return false;
624
625 ASSERT(m_lastPixelBuffer);
626
627 if (!m_rgbConformer) {
628 auto attributes = @{ (__bridge NSString *)kCVPixelBufferPixelFormatTypeKey: @(kCVPixelFormatType_32BGRA) };
629 m_rgbConformer = makeUnique<PixelBufferConformerCV>((__bridge CFDictionaryRef)attributes);
630 }
631
632 m_lastImage = NativeImage::create(m_rgbConformer->createImageFromPixelBuffer(m_lastPixelBuffer.get()));
633 return true;
634}

655RetainPtr<CVPixelBufferRef> MediaPlayerPrivateMediaSourceAVFObjC::pixelBufferForCurrentTime()
656{
657 // We have been asked to paint into a WebGL canvas, so take that as a signal to create
658 // a decompression session, even if that means the native video can't also be displayed
659 // in page.
660 if (!m_hasBeenAskedToPaintGL) {
661 m_hasBeenAskedToPaintGL = true;
662 acceleratedRenderingStateChanged();
663 }
664
665 if (updateLastPixelBuffer()) {
666 if (!m_lastPixelBuffer)
667 return nullptr;
668 }
669
670 return m_lastPixelBuffer;
671}

709String MediaPlayerPrivateMediaSourceAVFObjC::engineDescription() const

903void MediaPlayerPrivateMediaSourceAVFObjC::updateAllRenderersHaveAvailableSamples()
904{
905 bool allRenderersHaveAvailableSamples = true;
906
907 do {
908 if (hasVideo() && !m_hasAvailableVideoFrame) {
909 allRenderersHaveAvailableSamples = false;
910 break;
911 }
912
913 for (auto& properties : m_sampleBufferAudioRendererMap.values()) {
914 if (!properties.hasAudibleSample) {
915 allRenderersHaveAvailableSamples = false;
916 break;
917 }
918 }
919 } while (0);
920
921 if (m_allRenderersHaveAvailableSamples == allRenderersHaveAvailableSamples)
922 return;
923
924 DEBUG_LOG(LOGIDENTIFIER, allRenderersHaveAvailableSamples);
925 m_allRenderersHaveAvailableSamples = allRenderersHaveAvailableSamples;
926
927 if (shouldBePlaying() && [m_synchronizer rate] != m_rate)
928 [m_synchronizer setRate:m_rate];
929 else if (!shouldBePlaying() && [m_synchronizer rate])
930 [m_synchronizer setRate:0];
931}

933void MediaPlayerPrivateMediaSourceAVFObjC::durationChanged()
934{
935 m_player->durationChanged();
936
937 if (m_durationObserver)
938 [m_synchronizer removeTimeObserver:m_durationObserver.get()];
939
940 if (!m_mediaSourcePrivate)
941 return;
942
943 MediaTime duration = m_mediaSourcePrivate->duration();
944 NSArray* times = @[[NSValue valueWithCMTime:PAL::toCMTime(duration)]];
945
946 auto logSiteIdentifier = LOGIDENTIFIER;
947 DEBUG_LOG(logSiteIdentifier, duration);
948 UNUSED_PARAM(logSiteIdentifier);
949
950 m_durationObserver = [m_synchronizer addBoundaryTimeObserverForTimes:times queue:dispatch_get_main_queue() usingBlock:[weakThis = makeWeakPtr(*this), duration, logSiteIdentifier, this] {
951 if (!weakThis)
952 return;
953
954 MediaTime now = weakThis->currentMediaTime();
955 DEBUG_LOG(logSiteIdentifier, "boundary time observer called, now = ", now);
956
957 weakThis->pauseInternal();
958 if (now < duration) {
959 ERROR_LOG(logSiteIdentifier, "ERROR: boundary time observer called before duration");
960 [weakThis->m_synchronizer setRate:0 time:PAL::toCMTime(duration)];
961 }
962 weakThis->m_player->timeChanged();
963
964 }];
965
966 if (m_playing && duration <= currentMediaTime())
967 pauseInternal();
968}
