53@interface WebRootSampleBufferBoundsChangeListener : NSObject {
54 Function<void()> _callback;
55 RetainPtr<CALayer> _rootLayer;
56}
57
58- (id)initWithCallback:(Function<void()>&&) callback;
59- (void)invalidate;
60- (void)begin:(CALayer*) layer;
61- (void)stop;
62@end
63
64@implementation WebRootSampleBufferBoundsChangeListener
65
66- (id)initWithCallback:(Function<void()>&&) callback
67{
68 if (!(self = [super init]))
69 return nil;
70
71 _callback = WTFMove(callback);
72
73 return self;
74}
75
76- (void)dealloc
77{
78 [self invalidate];
79 [super dealloc];
80}
81
82- (void)invalidate
83{
84 [self stop];
85 _callback = nullptr;
86}
87
88- (void)begin:(CALayer*) layer
89{
90 ASSERT(_callback);
91 _rootLayer = layer;
92 [_rootLayer addObserver:self forKeyPath:@"bounds" options:NSKeyValueObservingOptionNew context:nil];
93}
94
95- (void)stop
96{
97 if (!_rootLayer)
98 return;
99
100 [_rootLayer removeObserver:self forKeyPath:@"bounds"];
101 _rootLayer = nullptr;
102}
103
104- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
105{
106 UNUSED_PARAM(context);
107 UNUSED_PARAM(keyPath);
108
109 if ([[change valueForKey:NSKeyValueChangeNotificationIsPriorKey] boolValue])
110 return;
111
112 if ((CALayer *)object == _rootLayer.get()) {
113 if ([keyPath isEqualToString:@"bounds"]) {
114 ensureOnMainThread([retainedSelf = retainPtr(self)] {
115 if (retainedSelf->_callback)
116 retainedSelf->_callback();
117 });
118 }
119 }
120
121}
122@end

144MediaPlayerPrivateMediaStreamAVFObjC::~MediaPlayerPrivateMediaStreamAVFObjC()
145{
146 INFO_LOG(LOGIDENTIFIER);
147
148 for (const auto& track : m_audioTrackMap.values())
149 track->pause();
150
151 if (m_mediaStreamPrivate)
152 m_mediaStreamPrivate->removeObserver(*this);
153
154 for (auto& track : m_audioTrackMap.values())
155 track->streamTrack().removeObserver(*this);
156
157 for (auto& track : m_videoTrackMap.values())
158 track->streamTrack().removeObserver(*this);
159
160 if (m_activeVideoTrack)
161 m_activeVideoTrack->streamTrack().source().removeVideoSampleObserver(*this);
162
163 [m_boundsChangeListener invalidate];
164
165 destroyLayers();
166
167 auto audioTrackMap = WTFMove(m_audioTrackMap);
168 for (auto& track : audioTrackMap.values())
169 track->clear();
170
171 m_videoTrackMap.clear();
172}

230static inline CGAffineTransform videoTransformationMatrix(MediaSample& sample)
231{
232 CMSampleBufferRef sampleBuffer = sample.platformSample().sample.cmSampleBuffer;
233 CVPixelBufferRef pixelBuffer = static_cast<CVPixelBufferRef>(PAL::CMSampleBufferGetImageBuffer(sampleBuffer));
234 size_t width = CVPixelBufferGetWidth(pixelBuffer);
235 size_t height = CVPixelBufferGetHeight(pixelBuffer);
236 if (!width || !height)
237 return CGAffineTransformIdentity;
238
239 auto videoTransform = CGAffineTransformMakeRotation(static_cast<int>(sample.videoRotation()) * M_PI / 180);
240 if (sample.videoMirrored())
241 videoTransform = CGAffineTransformScale(videoTransform, -1, 1);
242
243 return videoTransform;
244}

252void MediaPlayerPrivateMediaStreamAVFObjC::enqueueVideoSample(MediaSample& sample)
253{
254 if (!m_visible)
255 return;
256
257 if (!m_sampleBufferDisplayLayerLock.tryLock())
258 return;
259 Locker locker { AdoptLock, m_sampleBufferDisplayLayerLock };
260
261 if (!m_canEnqueueDisplayLayer || !m_sampleBufferDisplayLayer || m_sampleBufferDisplayLayer->didFail())
262 return;
263
264 if (sample.videoRotation() != m_videoRotation || sample.videoMirrored() != m_videoMirrored) {
265 m_videoRotation = sample.videoRotation();
266 m_videoMirrored = sample.videoMirrored();
267 m_sampleBufferDisplayLayer->updateAffineTransform(videoTransformationMatrix(sample));
268 m_shouldUpdateDisplayLayer = true;
269 }
270 if (m_shouldUpdateDisplayLayer) {
271 m_sampleBufferDisplayLayer->updateBoundsAndPosition(m_sampleBufferDisplayLayer->rootLayer().bounds, m_videoRotation);
272 m_shouldUpdateDisplayLayer = false;
273 }
274
275 m_sampleBufferDisplayLayer->enqueueSample(sample);
276}

278void MediaPlayerPrivateMediaStreamAVFObjC::processNewVideoSample(MediaSample& sample, bool hasChangedOrientation)
279{
280 if (!isMainThread()) {
281 {
282 Locker locker { m_currentVideoSampleLock };
283 m_currentVideoSample = &sample;
284 }
285 callOnMainThread([weakThis = makeWeakPtr(this), hasChangedOrientation]() mutable {
286 if (!weakThis)
287 return;
288
289 if (hasChangedOrientation)
290 weakThis->m_videoTransform = { };
291
292 RefPtr<MediaSample> sample;
293 {
294 Locker locker { weakThis->m_currentVideoSampleLock };
295 sample = WTFMove(weakThis->m_currentVideoSample);
296 }
297 if (!sample)
298 return;
299
300 weakThis->processNewVideoSample(*sample, hasChangedOrientation);
301 });
302 return;
303 }
304
305 if (!m_activeVideoTrack)
306 return;
307
308 if (hasChangedOrientation)
309 m_videoTransform = { };
310
311 if (!m_imagePainter.mediaSample || m_displayMode != PausedImage) {
312 m_imagePainter.mediaSample = &sample;
313 m_imagePainter.cgImage = nullptr;
314 if (m_readyState < MediaPlayer::ReadyState::HaveEnoughData)
315 updateReadyState();
316 }
317
318 if (m_displayMode != LivePreview && !m_waitingForFirstImage)
319 return;
320
321 if (!m_hasEverEnqueuedVideoFrame) {
322 m_hasEverEnqueuedVideoFrame = true;
323 m_player->firstVideoFrameAvailable();
324 }
325
326 if (m_waitingForFirstImage) {
327 m_waitingForFirstImage = false;
328 updateDisplayMode();
329 }
330}

358void MediaPlayerPrivateMediaStreamAVFObjC::ensureLayers()
359{
360 if (m_sampleBufferDisplayLayer)
361 return;
362
363 auto* activeVideoTrack = this->activeVideoTrack();
364 if (!activeVideoTrack || !activeVideoTrack->enabled())
365 return;
366
367 m_canEnqueueDisplayLayer = false;
368 m_sampleBufferDisplayLayer = SampleBufferDisplayLayer::create(*this);
369 ERROR_LOG_IF(!m_sampleBufferDisplayLayer, LOGIDENTIFIER, "Creating the SampleBufferDisplayLayer failed.");
370 if (!m_sampleBufferDisplayLayer)
371 return;
372
373 if (!playing())
374 m_sampleBufferDisplayLayer->pause();
375
376 if (activeVideoTrack->source().isCaptureSource())
377 m_sampleBufferDisplayLayer->setRenderPolicy(SampleBufferDisplayLayer::RenderPolicy::Immediately);
378
379 auto size = snappedIntRect(m_player->playerContentBoxRect()).size();
380 m_sampleBufferDisplayLayer->initialize(hideRootLayer(), size, [weakThis = makeWeakPtr(this), size](auto didSucceed) {
381 if (weakThis)
382 weakThis->layersAreInitialized(size, didSucceed);
383 });
384}
385
386void MediaPlayerPrivateMediaStreamAVFObjC::layersAreInitialized(IntSize size, bool didSucceed)
387{
388 if (!didSucceed) {
389 ERROR_LOG(LOGIDENTIFIER, "Initializing the SampleBufferDisplayLayer failed.");
390 m_sampleBufferDisplayLayer = nullptr;
391 return;
392 }
393
394 updateRenderingMode();
395 m_shouldUpdateDisplayLayer = true;
396
397 m_videoLayerManager->setVideoLayer(m_sampleBufferDisplayLayer->rootLayer(), size);
398
399 [m_boundsChangeListener begin:m_sampleBufferDisplayLayer->rootLayer()];
400
401 m_canEnqueueDisplayLayer = true;
402}

413

489MediaPlayerPrivateMediaStreamAVFObjC::DisplayMode MediaPlayerPrivateMediaStreamAVFObjC::currentDisplayMode() const
490{
491 if (m_intrinsicSize.isEmpty() || !metaDataAvailable())
492 return None;
493
494 if (auto* track = activeVideoTrack()) {
495 if (!track->enabled() || track->muted() || track->ended())
496 return PaintItBlack;
497 }
498
499 if (m_waitingForFirstImage)
500 return WaitingForFirstImage;
501
502 if (playing() && !m_ended) {
503 if (!m_mediaStreamPrivate->isProducingData())
504 return PausedImage;
505 return LivePreview;
506 }
507
508 if (m_playbackState == PlaybackState::None || m_ended)
509 return PaintItBlack;
510
511 return PausedImage;
512}

530void MediaPlayerPrivateMediaStreamAVFObjC::play()
531{
532 ALWAYS_LOG(LOGIDENTIFIER);
533
534 if (!metaDataAvailable() || playing() || m_ended)
535 return;
536
537 m_playbackState = PlaybackState::Playing;
538 if (m_startTime.isInvalid())
539 m_startTime = MediaTime::createWithDouble(MonotonicTime::now().secondsSinceEpoch().value());
540
541 for (const auto& track : m_audioTrackMap.values())
542 track->play();
543
544 if (m_sampleBufferDisplayLayer)
545 m_sampleBufferDisplayLayer->play();
546 updateDisplayMode();
547
548 scheduleDeferredTask([this] {
549 updateReadyState();
550 if (m_player)
551 m_player->rateChanged();
552 });
553}

647MediaPlayer::ReadyState MediaPlayerPrivateMediaStreamAVFObjC::currentReadyState()
648{
649 if (!m_mediaStreamPrivate || !m_mediaStreamPrivate->active() || !m_mediaStreamPrivate->hasTracks())
650 return MediaPlayer::ReadyState::HaveNothing;
651
652 bool waitingForImage = activeVideoTrack() && !m_imagePainter.mediaSample;
653 if (waitingForImage && (!m_haveSeenMetadata || m_waitingForFirstImage))
654 return MediaPlayer::ReadyState::HaveNothing;
655
656 bool allTracksAreLive = !waitingForImage;
657 if (allTracksAreLive) {
658 m_mediaStreamPrivate->forEachTrack([&](auto& track) {
659 if (!track.enabled() || track.readyState() != MediaStreamTrackPrivate::ReadyState::Live)
660 allTracksAreLive = false;
661 });
662 }
663
664 if (m_waitingForFirstImage || (!allTracksAreLive && !m_haveSeenMetadata))
665 return MediaPlayer::ReadyState::HaveMetadata;
666
667 return MediaPlayer::ReadyState::HaveEnoughData;
668}

713void MediaPlayerPrivateMediaStreamAVFObjC::characteristicsChanged()
714{
715 bool sizeChanged = false;
716
717 FloatSize intrinsicSize = m_mediaStreamPrivate->intrinsicSize();
718 if (intrinsicSize.height() != m_intrinsicSize.height() || intrinsicSize.width() != m_intrinsicSize.width()) {
719 m_intrinsicSize = intrinsicSize;
720 sizeChanged = true;
721 if (m_playbackState == PlaybackState::None)
722 m_playbackState = PlaybackState::Paused;
723 }
724
725 updateTracks();
726 updateDisplayMode();
727
728 scheduleDeferredTask([this, sizeChanged] {
729 updateReadyState();
730
731 if (!m_player)
732 return;
733
734 m_player->characteristicChanged();
735 if (sizeChanged) {
736 m_player->sizeChanged();
737 }
738 });

792template <typename RefT>
793void updateTracksOfType(HashMap<String, RefT>& trackMap, RealtimeMediaSource::Type trackType, MediaStreamTrackPrivateVector& currentTracks, RefT (*itemFactory)(MediaStreamTrackPrivate&), const Function<void(std::reference_wrapper<typename std::remove_pointer<typename RefT::PtrTraits::StorageType>::type>, int, TrackState)>& configureTrack)
794{
795 Vector<RefT> removedTracks;
796 Vector<RefT> addedTracks;
797 Vector<Ref<MediaStreamTrackPrivate>> addedPrivateTracks;
798
799 for (const auto& track : currentTracks) {
800 if (track->type() != trackType)
801 continue;
802
803 if (!trackMap.contains(track->id()))
804 addedPrivateTracks.append(*track);
805 }
806
807 for (const auto& track : trackMap.values()) {
808 auto& streamTrack = track->streamTrack();
809 if (currentTracks.contains(&streamTrack))
810 continue;
811
812 removedTracks.append(track);
813 }
814 for (auto& track : removedTracks)
815 trackMap.remove(track->streamTrack().id());
816
817 for (auto& track : addedPrivateTracks) {
818 RefT newTrack = itemFactory(track.get());
819 trackMap.add(track->id(), newTrack.copyRef());
820 addedTracks.append(WTFMove(newTrack));
821 }
822
823 int index = 0;
824 for (auto& track : removedTracks)
825 configureTrack(track.get(), index++, TrackState::Remove);
826
827 index = 0;
828 for (auto& track : addedTracks)
829 configureTrack(track.get(), index++, TrackState::Add);
830
831 index = 0;
832 for (const auto& track : trackMap.values())
833 configureTrack(track.get(), index++, TrackState::Configure);
834}

836void MediaPlayerPrivateMediaStreamAVFObjC::checkSelectedVideoTrack()
837{
838 auto oldVideoTrack = m_activeVideoTrack;
839 bool hideVideoLayer = true;
840
841 m_activeVideoTrack = nullptr;
842 if (auto* activeVideoTrack = this->activeVideoTrack()) {
843 for (const auto& track : m_videoTrackMap.values()) {
844 if (&track->streamTrack() == activeVideoTrack) {
845 m_activeVideoTrack = track.ptr();
846 if (track->selected())
847 hideVideoLayer = false;
848 break;
849 }
850 }
851 }
852
853 if (oldVideoTrack != m_activeVideoTrack) {
854 m_imagePainter.reset();
855 if (m_displayMode == None)
856 m_waitingForFirstImage = true;
857 }
858 ensureLayers();
859 if (m_sampleBufferDisplayLayer) {
860 if (!m_activeVideoTrack)
861 m_sampleBufferDisplayLayer->clearEnqueuedSamples();
862 m_sampleBufferDisplayLayer->updateDisplayMode(hideVideoLayer || m_displayMode < PausedImage, hideRootLayer());
863 }
864
865 updateDisplayMode();
866
867 if (oldVideoTrack != m_activeVideoTrack) {
868 if (oldVideoTrack)
869 oldVideoTrack->streamTrack().source().removeVideoSampleObserver(*this);
870 if (m_activeVideoTrack) {
871 if (m_sampleBufferDisplayLayer && m_activeVideoTrack->streamTrack().source().isCaptureSource())
872 m_sampleBufferDisplayLayer->setRenderPolicy(SampleBufferDisplayLayer::RenderPolicy::Immediately);
873 m_activeVideoTrack->streamTrack().source().addVideoSampleObserver(*this);
874 }
875 }
876}

878void MediaPlayerPrivateMediaStreamAVFObjC::updateTracks()
879{
880 MediaStreamTrackPrivateVector currentTracks = m_mediaStreamPrivate->tracks();
881
882 auto deviceId = m_player->audioOutputDeviceIdOverride();
883 auto setAudioTrackState = [this, &deviceId](AudioTrackPrivateMediaStream& track, int index, TrackState state)
884 {
885 switch (state) {
886 case TrackState::Remove:
887 track.streamTrack().removeObserver(*this);
888 track.clear();
889 m_player->removeAudioTrack(track);
890 break;
891 case TrackState::Add:
892 track.streamTrack().addObserver(*this);
893 m_player->addAudioTrack(track);
894 break;
895 case TrackState::Configure:
896 track.setTrackIndex(index);
897 track.setVolume(m_volume);
898 track.setMuted(m_muted);
899 track.setEnabled(track.streamTrack().enabled() && !track.streamTrack().muted());
900 if (!deviceId.isNull())
901 track.setAudioOutputDevice(deviceId);
902
903 if (playing())
904 track.play();
905 break;
906 }
907 };
908 updateTracksOfType(m_audioTrackMap, RealtimeMediaSource::Type::Audio, currentTracks, &AudioTrackPrivateMediaStream::create, WTFMove(setAudioTrackState));
909
910 if (!m_player->isVideoPlayer())
911 return;
912
913 auto setVideoTrackState = [this](VideoTrackPrivateMediaStream& track, int index, TrackState state)
914 {
915 switch (state) {
916 case TrackState::Remove:
917 track.streamTrack().removeObserver(*this);
918 m_player->removeVideoTrack(track);
919 checkSelectedVideoTrack();
920 break;
921 case TrackState::Add:
922 track.streamTrack().addObserver(*this);
923 m_player->addVideoTrack(track);
924 break;
925 case TrackState::Configure:
926 track.setTrackIndex(index);
927 bool selected = &track.streamTrack() == activeVideoTrack();
928 track.setSelected(selected);
929 checkSelectedVideoTrack();
930 break;
931 }
932 };
933 updateTracksOfType(m_videoTrackMap, RealtimeMediaSource::Type::Video, currentTracks, &VideoTrackPrivateMediaStream::create, WTFMove(setVideoTrackState));
934}

967void MediaPlayerPrivateMediaStreamAVFObjC::paintCurrentFrameInContext(GraphicsContext& context, const FloatRect& destRect)
968{
969 if (m_displayMode == None || !metaDataAvailable() || context.paintingDisabled())
970 return;
971
972 if (m_displayMode != PaintItBlack && m_imagePainter.mediaSample)
973 updateCurrentFrameImage();
974
975 GraphicsContextStateSaver stateSaver(context);
976 if (m_displayMode == PaintItBlack) {
977 context.fillRect(IntRect(IntPoint(), IntSize(destRect.width(), destRect.height())), Color::black);
978 return;
979 }
980
981 if (!m_imagePainter.cgImage || !m_imagePainter.mediaSample)
982 return;
983
984 auto& image = m_imagePainter.cgImage;
985 FloatRect imageRect { FloatPoint::zero(), image->size() };
986 if (!m_videoTransform)
987 m_videoTransform = videoTransformationMatrix(*m_imagePainter.mediaSample);
988 AffineTransform videoTransform = *m_videoTransform;
989 FloatRect transformedDestRect = videoTransform.inverse().value_or(AffineTransform()).mapRect(destRect);
990 context.concatCTM(videoTransform);
991 context.drawNativeImage(*image, imageRect.size(), transformedDestRect, imageRect);
992}
