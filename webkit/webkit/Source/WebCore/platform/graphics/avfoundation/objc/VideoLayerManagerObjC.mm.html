63void VideoLayerManagerObjC::setVideoLayer(PlatformLayer *videoLayer, IntSize contentSize)
64{
65 ALWAYS_LOG(LOGIDENTIFIER, contentSize.width(), ", ", contentSize.height());
66
67 m_videoLayer = videoLayer;
68 [m_videoLayer web_disableAllActions];
69
70 m_videoInlineLayer = adoptNS([[WebVideoContainerLayer alloc] init]);
71 [m_videoInlineLayer setName:@"WebVideoContainerLayer"];
72 [m_videoInlineLayer setFrame:CGRectMake(0, 0, contentSize.width(), contentSize.height())];
73 [m_videoInlineLayer setContentsGravity:kCAGravityResizeAspect];
74 if (PAL::isAVFoundationFrameworkAvailable() && [videoLayer isKindOfClass:PAL::getAVPlayerLayerClass()])
75 [m_videoInlineLayer setPlayerLayer:(AVPlayerLayer *)videoLayer];
76
77#if ENABLE(VIDEO_PRESENTATION_MODE)
78 if (m_videoFullscreenLayer) {
79 [m_videoLayer setFrame:CGRectMake(0, 0, m_videoFullscreenFrame.width(), m_videoFullscreenFrame.height())];
80 [m_videoFullscreenLayer insertSublayer:m_videoLayer.get() atIndex:0];
81 } else
82#endif
83 {
84 [m_videoLayer setFrame:m_videoInlineLayer.get().bounds];
85 [m_videoInlineLayer insertSublayer:m_videoLayer.get() atIndex:0];
86 }
87}

106void VideoLayerManagerObjC::setVideoFullscreenLayer(PlatformLayer *videoFullscreenLayer, WTF::Function<void()>&& completionHandler, PlatformImagePtr currentImage)
107{
108 if (m_videoFullscreenLayer == videoFullscreenLayer) {
109 completionHandler();
110 return;
111 }
112
113 ALWAYS_LOG(LOGIDENTIFIER);
114
115 m_videoFullscreenLayer = videoFullscreenLayer;
116
117 [CATransaction begin];
118 [CATransaction setDisableActions:YES];
119
120 if (m_videoLayer) {
121 CAContext *oldContext = [m_videoLayer context];
122
123 if (m_videoInlineLayer && currentImage)
124 [m_videoInlineLayer setContents:(__bridge id)currentImage.get()];
125
126 if (m_videoFullscreenLayer) {
127 [m_videoLayer setFrame:CGRectMake(0, 0, m_videoFullscreenFrame.width(), m_videoFullscreenFrame.height())];
128 [m_videoFullscreenLayer insertSublayer:m_videoLayer.get() atIndex:0];
129 } else if (m_videoInlineLayer) {
130 [m_videoLayer setFrame:[m_videoInlineLayer bounds]];
131 [m_videoInlineLayer insertSublayer:m_videoLayer.get() atIndex:0];
132 } else
133 [m_videoLayer removeFromSuperlayer];
134
135 CAContext *newContext = [m_videoLayer context];
136 if (oldContext && newContext && oldContext != newContext) {
137#if PLATFORM(MAC)
138 oldContext.commitPriority = 0;
139 newContext.commitPriority = 1;
140#endif
141 auto fencePort = MachSendRight::adopt([oldContext createFencePort]);
142 [newContext setFencePort:fencePort.sendRight()];
143 }
144 }
145
146 [CATransaction setCompletionBlock:makeBlockPtr([completionHandler = WTFMove(completionHandler)] {
147 completionHandler();
148 }).get()];
149
150 [CATransaction commit];
151}

158void VideoLayerManagerObjC::setVideoFullscreenFrame(FloatRect videoFullscreenFrame)
159{
160 ALWAYS_LOG(LOGIDENTIFIER, videoFullscreenFrame.x(), ", ", videoFullscreenFrame.y(), ", ", videoFullscreenFrame.width(), ", ", videoFullscreenFrame.height());
161
162 m_videoFullscreenFrame = videoFullscreenFrame;
163 if (!m_videoFullscreenLayer)
164 return;
165
166 [m_videoLayer setFrame:m_videoFullscreenFrame];
167 syncTextTrackBounds();
168}

187void VideoLayerManagerObjC::syncTextTrackBounds()
188{
189#if ENABLE(VIDEO_PRESENTATION_MODE)
190 if (!m_videoFullscreenLayer || !m_textTrackRepresentationLayer)
191 return;
192
193 if (m_textTrackRepresentationLayer.get().bounds == m_videoFullscreenFrame)
194 return;
195
196 [CATransaction begin];
197 [CATransaction setDisableActions:YES];
198
199 [m_textTrackRepresentationLayer setFrame:m_videoFullscreenFrame];
200
201 [CATransaction commit];
202#endif
203}
204
205void VideoLayerManagerObjC::setTextTrackRepresentationLayer(PlatformLayer* representationLayer)
206{
207#if !ENABLE(VIDEO_PRESENTATION_MODE)
208 UNUSED_PARAM(representationLayer);
209#else
210 ALWAYS_LOG(LOGIDENTIFIER);
211
212 if (representationLayer == m_textTrackRepresentationLayer) {
213 syncTextTrackBounds();
214 return;
215 }
216
217 [CATransaction begin];
218 [CATransaction setDisableActions:YES];
219
220 if (m_textTrackRepresentationLayer)
221 [m_textTrackRepresentationLayer removeFromSuperlayer];
222
223 m_textTrackRepresentationLayer = representationLayer;
224
225 if (m_videoFullscreenLayer && m_textTrackRepresentationLayer) {
226 syncTextTrackBounds();
227 [m_videoFullscreenLayer addSublayer:m_textTrackRepresentationLayer.get()];
228 }
229
230 [CATransaction commit];
231#endif
232}
