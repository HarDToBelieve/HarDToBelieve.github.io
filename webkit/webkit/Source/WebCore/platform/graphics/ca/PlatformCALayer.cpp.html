82void PlatformCALayer::drawRepaintIndicator(GraphicsContext& graphicsContext, PlatformCALayer* platformCALayer, int repaintCount, Color customBackgroundColor)
83{
84 const float verticalMargin = 2.5;
85 const float horizontalMargin = 5;
86 const unsigned fontSize = 22;
87 constexpr auto backgroundColor = SRGBA<uint8_t> { 128, 64, 255 };
88 constexpr auto acceleratedContextLabelColor = Color::red;
89 constexpr auto unacceleratedContextLabelColor = Color::white;
90 constexpr auto linearGlyphMaskOutlineColor = Color::black.colorWithAlphaByte(192);
91 constexpr auto displayListBorderColor = Color::black.colorWithAlphaByte(166);
92
93 TextRun textRun(String::number(repaintCount));
94
95 FontCascadeDescription fontDescription;
96 fontDescription.setOneFamily("Helvetica");
97 fontDescription.setSpecifiedSize(fontSize);
98 fontDescription.setComputedSize(fontSize);
99
100 FontCascade cascade(WTFMove(fontDescription));
101 cascade.update(nullptr);
102
103 float textWidth = cascade.width(textRun);
104
105 GraphicsContextStateSaver stateSaver(graphicsContext);
106
107 graphicsContext.beginTransparencyLayer(0.5f);
108
109 graphicsContext.setFillColor(customBackgroundColor.isValid() ? customBackgroundColor : backgroundColor);
110 FloatRect indicatorBox(1, 1, horizontalMargin * 2 + textWidth, verticalMargin * 2 + fontSize);
111 if (platformCALayer->isOpaque())
112 graphicsContext.fillRect(indicatorBox);
113 else {
114 Path boundsPath;
115 boundsPath.moveTo(indicatorBox.maxXMinYCorner());
116 boundsPath.addLineTo(indicatorBox.maxXMaxYCorner());
117 boundsPath.addLineTo(indicatorBox.minXMaxYCorner());
118
119 const float cornerChunk = 8;
120 boundsPath.addLineTo(FloatPoint(indicatorBox.x(), indicatorBox.y() + cornerChunk));
121 boundsPath.addLineTo(FloatPoint(indicatorBox.x() + cornerChunk, indicatorBox.y()));
122 boundsPath.closeSubpath();
123
124 graphicsContext.fillPath(boundsPath);
125 }
126
127 if (platformCALayer->owner()->isUsingDisplayListDrawing(platformCALayer)) {
128 graphicsContext.setStrokeColor(displayListBorderColor);
129 graphicsContext.strokeRect(indicatorBox, 2);
130 }
131
132 if (!platformCALayer->isOpaque() && platformCALayer->supportsSubpixelAntialiasedText() && platformCALayer->acceleratesDrawing()) {
133 graphicsContext.setStrokeColor(linearGlyphMaskOutlineColor);
134 graphicsContext.setStrokeThickness(4.5);
135 graphicsContext.setTextDrawingMode(TextDrawingModeFlags { TextDrawingMode::Fill, TextDrawingMode::Stroke });
136 }
137
138 graphicsContext.setFillColor(platformCALayer->acceleratesDrawing() ? acceleratedContextLabelColor : unacceleratedContextLabelColor);
139
140 graphicsContext.drawText(cascade, textRun, FloatPoint(indicatorBox.x() + horizontalMargin, indicatorBox.y() + fontSize));
141 graphicsContext.endTransparencyLayer();
142}

150void PlatformCALayer::drawTextAtPoint(CGContextRef context, CGFloat x, CGFloat y, CGSize scale, CGFloat fontSize, const char* text, size_t length, CGFloat strokeWidthAsPercentageOfFontSize, Color strokeColor) const
151{
152 auto matrix = CGAffineTransformMakeScale(scale.width, scale.height);
153 auto font = adoptCF(CTFontCreateWithName(CFSTR("Helvetica"), fontSize, &matrix));
154 auto strokeWidthNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberCGFloatType, &strokeWidthAsPercentageOfFontSize));
155
156 CFTypeRef keys[] = {
157 kCTFontAttributeName,
158 kCTForegroundColorFromContextAttributeName,
159 kCTStrokeWidthAttributeName,
160 kCTStrokeColorAttributeName,
161 };
162 CFTypeRef values[] = {
163 font.get(),
164 kCFBooleanTrue,
165 strokeWidthNumber.get(),
166 cachedCGColor(strokeColor),
167 };
168
169 auto attributes = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, keys, values, WTF_ARRAY_LENGTH(keys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
170 auto string = adoptCF(CFStringCreateWithBytesNoCopy(kCFAllocatorDefault, reinterpret_cast<const UInt8*>(text), length, kCFStringEncodingUTF8, false, kCFAllocatorNull));
171 auto attributedString = adoptCF(CFAttributedStringCreate(kCFAllocatorDefault, string.get(), attributes.get()));
172 auto line = adoptCF(CTLineCreateWithAttributedString(attributedString.get()));
173 CGContextSetTextPosition(context, x, y);
174 CTLineDraw(line.get(), context);
175}

177Ref<PlatformCALayer> PlatformCALayer::createCompatibleLayerOrTakeFromPool(PlatformCALayer::LayerType layerType, PlatformCALayerClient* client, IntSize size)
178{
179 if (auto layerFromPool = layerPool().takeLayerWithSize(size)) {
180 layerFromPool->setOwner(client);
181 return layerFromPool.releaseNonNull();
182 }
183
184 auto layer = createCompatibleLayer(layerType, client);
185 layer->setBounds(FloatRect(FloatPoint(), size));
186 return layer;
187}

205TextStream& operator<<(TextStream& ts, PlatformCALayer::LayerType layerType)
206{
207 switch (layerType) {
208 case PlatformCALayer::LayerTypeLayer:
209 case PlatformCALayer::LayerTypeWebLayer:
210 case PlatformCALayer::LayerTypeSimpleLayer:
211 ts << "layer";
212 break;
213 case PlatformCALayer::LayerTypeTransformLayer:
214 ts << "transform-layer";
215 break;
216 case PlatformCALayer::LayerTypeTiledBackingLayer:
217 ts << "tiled-backing-layer";
218 break;
219 case PlatformCALayer::LayerTypePageTiledBackingLayer:
220 ts << "page-tiled-backing-layer";
221 break;
222 case PlatformCALayer::LayerTypeTiledBackingTileLayer:
223 ts << "tiled-backing-tile";
224 break;
225 case PlatformCALayer::LayerTypeRootLayer:
226 ts << "root-layer";
227 break;
228 case PlatformCALayer::LayerTypeBackdropLayer:
229 ts << "backdrop-layer";
230 break;
231 case PlatformCALayer::LayerTypeAVPlayerLayer:
232 ts << "av-player-layer";
233 break;
234 case PlatformCALayer::LayerTypeContentsProvidedLayer:
235 ts << "contents-provided-layer";
236 break;
237 case PlatformCALayer::LayerTypeShapeLayer:
238 ts << "shape-layer";
239 break;
240 case PlatformCALayer::LayerTypeScrollContainerLayer:
241 ts << "scroll-container-layer";
242 break;
243 case PlatformCALayer::LayerTypeCustom:
244 ts << "custom-layer";
245 break;
246 case PlatformCALayer::LayerTypeLightSystemBackdropLayer:
247 ts << "light-system-backdrop-layer";
248 break;
249 case PlatformCALayer::LayerTypeDarkSystemBackdropLayer:
250 ts << "dark-system-backdrop-layer";
251 break;
252#if ENABLE(MODEL_ELEMENT)
253 case PlatformCALayer::LayerTypeModelLayer:
254 ts << "model-layer";
255 break;
256#endif
257 }
258 return ts;
259}
