60TileGrid::TileGrid(TileController& controller)
61 : m_controller(controller)
62 , m_containerLayer(controller.rootLayer().createCompatibleLayer(PlatformCALayer::LayerTypeLayer, nullptr))
63 , m_cohortRemovalTimer(*this, &TileGrid::cohortRemovalTimerFired)

102void TileGrid::setNeedsDisplay()
103{
104 for (auto& entry : m_tiles) {
105 TileInfo& tileInfo = entry.value;
106 IntRect tileRect = rectForTileIndex(entry.key);
107
108 if (tileRect.intersects(m_primaryTileCoverageRect) && tileInfo.layer->superlayer())
109 tileInfo.layer->setNeedsDisplay();
110 else
111 tileInfo.hasStaleContent = true;
112 }
113}

115void TileGrid::setNeedsDisplayInRect(const IntRect& rect)
116{
117 if (m_tiles.isEmpty())
118 return;
119
120 FloatRect scaledRect(rect);
121 scaledRect.scale(m_scale);
122 IntRect repaintRectInTileCoords(enclosingIntRect(scaledRect));
123
124 IntSize tileSize = m_tileSize;
125
126 // For small invalidations, lookup the covered tiles.
127 if (repaintRectInTileCoords.height() < 2 * tileSize.height() && repaintRectInTileCoords.width() < 2 * tileSize.width()) {
128 TileIndex topLeft;
129 TileIndex bottomRight;
130 if (getTileIndexRangeForRect(repaintRectInTileCoords, topLeft, bottomRight)) {
131 for (int y = topLeft.y(); y <= bottomRight.y(); ++y) {
132 for (int x = topLeft.x(); x <= bottomRight.x(); ++x) {
133 TileIndex tileIndex(x, y);
134 auto it = m_tiles.find(tileIndex);
135 if (it != m_tiles.end())
136 setTileNeedsDisplayInRect(tileIndex, it->value, repaintRectInTileCoords, m_primaryTileCoverageRect);
137 }
138 }
139 }
140 return;
141 }
142
143 for (auto& entry : m_tiles)
144 setTileNeedsDisplayInRect(entry.key, entry.value, repaintRectInTileCoords, m_primaryTileCoverageRect);
145}

147void TileGrid::dropTilesInRect(const IntRect& rect)
148{
149 if (m_tiles.isEmpty())
150 return;
151
152 FloatRect scaledRect(rect);
153 scaledRect.scale(m_scale);
154 IntRect dropRectInTileCoords(enclosingIntRect(scaledRect));
155
156 Vector<TileIndex> tilesToRemove;
157
158 for (auto& index : m_tiles.keys()) {
159 if (rectForTileIndex(index).intersects(dropRectInTileCoords))
160 tilesToRemove.append(index);
161 }
162
163 removeTiles(tilesToRemove);
164}

166void TileGrid::setTileNeedsDisplayInRect(const TileIndex& tileIndex, TileInfo& tileInfo, const IntRect& repaintRectInTileCoords, const IntRect& coverageRectInTileCoords)
167{
168 PlatformCALayer* tileLayer = tileInfo.layer.get();
169
170 IntRect tileRect = rectForTileIndex(tileIndex);
171 FloatRect tileRepaintRect = tileRect;
172 tileRepaintRect.intersect(repaintRectInTileCoords);
173 if (tileRepaintRect.isEmpty())
174 return;
175
176 tileRepaintRect.moveBy(-tileRect.location());
177
178 // We could test for intersection with the visible rect. This would reduce painting yet more,
179 // but may make scrolling stale tiles into view more frequent.
180 if (tileRect.intersects(coverageRectInTileCoords) && tileLayer->superlayer()) {
181 tileLayer->setNeedsDisplayInRect(tileRepaintRect);
182
183 if (m_controller.rootLayer().owner()->platformCALayerShowRepaintCounter(0)) {
184 FloatRect indicatorRect(0, 0, 52, 27);
185 tileLayer->setNeedsDisplayInRect(indicatorRect);
186 }
187 } else
188 tileInfo.hasStaleContent = true;
189}

191void TileGrid::updateTileLayerProperties()
192{
193 bool acceleratesDrawing = m_controller.acceleratesDrawing();
194 bool deepColor = m_controller.wantsDeepColorBackingStore();
195 bool subpixelAntialiasedText = m_controller.supportsSubpixelAntialiasedText();
196 bool opaque = m_controller.tilesAreOpaque();
197 Color tileDebugBorderColor = m_controller.tileDebugBorderColor();
198 float tileDebugBorderWidth = m_controller.tileDebugBorderWidth();
199 for (auto& tileInfo : m_tiles.values()) {
200 tileInfo.layer->setAcceleratesDrawing(acceleratesDrawing);
201 tileInfo.layer->setWantsDeepColorBackingStore(deepColor);
202 tileInfo.layer->setSupportsSubpixelAntialiasedText(subpixelAntialiasedText);
203 tileInfo.layer->setOpaque(opaque);
204 tileInfo.layer->setBorderColor(tileDebugBorderColor);
205 tileInfo.layer->setBorderWidth(tileDebugBorderWidth);
206 }

209bool TileGrid::tilesWouldChangeForCoverageRect(const FloatRect& coverageRect) const
210{
211 if (coverageRect.isEmpty())
212 return false;
213
214 FloatRect scaledRect(coverageRect);
215 scaledRect.scale(m_scale);
216 IntRect currentCoverageRectInTileCoords(enclosingIntRect(scaledRect));
217
218 IntRect tileCoverageRect;
219 TileIndex topLeft;
220 TileIndex bottomRight;
221 if (getTileIndexRangeForRect(currentCoverageRectInTileCoords, topLeft, bottomRight)) {
222 tileCoverageRect = rectForTileIndex(topLeft);
223 tileCoverageRect.unite(rectForTileIndex(bottomRight));
224 }
225
226 return tileCoverageRect != m_primaryTileCoverageRect;
227}

229bool TileGrid::prepopulateRect(const FloatRect& rect)
230{
231 LOG_WITH_STREAM(Tiling, stream << "TileGrid " << this << " prepopulateRect: " << rect);
232
233 IntRect enclosingCoverageRect = enclosingIntRect(rect);
234 if (m_primaryTileCoverageRect.contains(enclosingCoverageRect))
235 return false;
236
237 m_secondaryTileCoverageRects.append(enclosingCoverageRect);
238 return true;
239}

241IntRect TileGrid::rectForTileIndex(const TileIndex& tileIndex) const
242{
243 // FIXME: calculating the scaled size here should match with the rest of calculated sizes where we use the combination of
244 // enclosingIntRect, expandedIntSize (floor vs ceil).
245 // However enclosing this size could reveal gap on root layer's background. see RenderView::backgroundRect()
246 IntSize tileSize = m_tileSize;
247 IntRect rect(tileIndex.x() * tileSize.width(), tileIndex.y() * tileSize.height(), tileSize.width(), tileSize.height());
248 IntRect scaledBounds(m_controller.bounds());
249 scaledBounds.scale(m_scale);
250 rect.intersect(scaledBounds);
251 return rect;
252}

254bool TileGrid::getTileIndexRangeForRect(const IntRect& rect, TileIndex& topLeft, TileIndex& bottomRight) const
255{
256 IntRect clampedRect = m_controller.bounds();
257 clampedRect.scale(m_scale);
258 clampedRect.intersect(rect);
259
260 if (clampedRect.isEmpty())
261 return false;
262
263 auto tileSize = m_tileSize;
264 if (clampedRect.x() >= 0)
265 topLeft.setX(clampedRect.x() / tileSize.width());
266 else
267 topLeft.setX(floorf((float)clampedRect.x() / tileSize.width()));
268
269 if (clampedRect.y() >= 0)
270 topLeft.setY(clampedRect.y() / tileSize.height());
271 else
272 topLeft.setY(floorf((float)clampedRect.y() / tileSize.height()));
273
274 int bottomXRatio = ceil((float)clampedRect.maxX() / tileSize.width());
275 bottomRight.setX(std::max(bottomXRatio - 1, 0));
276
277 int bottomYRatio = ceil((float)clampedRect.maxY() / tileSize.height());
278 bottomRight.setY(std::max(bottomYRatio - 1, 0));
279
280 return true;
281}

293void TileGrid::removeTiles(Vector<TileGrid::TileIndex>& toRemove)
294{
295 for (size_t i = 0; i < toRemove.size(); ++i) {
296 TileInfo tileInfo = m_tiles.take(toRemove[i]);
297 tileInfo.layer->removeFromSuperlayer();
298 m_tileRepaintCounts.removeAll(tileInfo.layer.get());
299 tileInfo.layer->moveToLayerPool();
300 }
301}

303void TileGrid::removeAllTiles()
304{
305 Vector<TileIndex> tilesToRemove;
306 tilesToRemove.reserveInitialCapacity(m_tiles.size());
307
308 for (auto& entry : m_tiles)
309 tilesToRemove.uncheckedAppend(entry.key);
310
311 removeTiles(tilesToRemove);
312}

314void TileGrid::removeAllSecondaryTiles()
315{
316 Vector<TileIndex> tilesToRemove;
317
318 for (auto& entry : m_tiles) {
319 const TileInfo& tileInfo = entry.value;
320 if (tileInfo.cohort == visibleTileCohort)
321 continue;
322 tilesToRemove.append(entry.key);
323 }
324
325 removeTiles(tilesToRemove);
326}

328void TileGrid::removeTilesInCohort(TileCohort cohort)
329{
330 ASSERT(cohort != visibleTileCohort);
331 Vector<TileIndex> tilesToRemove;
332
333 for (auto& entry : m_tiles) {
334 const TileInfo& tileInfo = entry.value;
335 if (tileInfo.cohort != cohort)
336 continue;
337 tilesToRemove.append(entry.key);
338 }
339
340 removeTiles(tilesToRemove);
341}

343void TileGrid::revalidateTiles(OptionSet<ValidationPolicyFlag> validationPolicy)
344{
345 FloatRect coverageRect = m_controller.coverageRect();
346 IntRect bounds = m_controller.bounds();
347
348 LOG_WITH_STREAM(Tiling, stream << "TileGrid " << this << " (controller " << &m_controller << ") revalidateTiles: bounds " << bounds << " coverageRect" << coverageRect << " validation: " << validationPolicyAsString(validationPolicy));
349
350 FloatRect scaledRect(coverageRect);
351 scaledRect.scale(m_scale);
352 IntRect coverageRectInTileCoords(enclosingIntRect(scaledRect));
353
354 TileCohort currCohort = nextTileCohort();
355 unsigned tilesInCohort = 0;
356
357 Seconds minimumRevalidationTimerDuration = Seconds::infinity();
358 bool needsTileRevalidation = false;
359
360 auto tileSize = m_controller.computeTileSize();
361 if (tileSize != m_tileSize) {
362 removeAllTiles();
363 m_tileSize = tileSize;
364 }
365
366 // Move tiles newly outside the coverage rect into the cohort map.
367 for (auto& entry : m_tiles) {
368 TileInfo& tileInfo = entry.value;
369 TileIndex tileIndex = entry.key;
370
371 PlatformCALayer* tileLayer = tileInfo.layer.get();
372 IntRect tileRect = rectForTileIndex(tileIndex);
373
374 if (tileRect.intersects(coverageRectInTileCoords)) {
375 tileInfo.cohort = visibleTileCohort;
376 if (tileInfo.hasStaleContent) {
377 // FIXME: store a dirty region per layer?
378 tileLayer->setNeedsDisplay();
379 tileInfo.hasStaleContent = false;
380 }
381 } else {
382 // Add to the currentCohort if not already in one.
383 if (tileInfo.cohort == visibleTileCohort) {
384 tileInfo.cohort = currCohort;
385 ++tilesInCohort;
386 tileLayer->removeFromSuperlayer();
387 } else if (m_controller.shouldAggressivelyRetainTiles() && tileLayer->superlayer()) {
388 // Aggressive tile retention means we'll never remove cohorts, but we need to make sure they're unparented.
389 // We can't immediately unparent cohorts comprised of secondary tiles that never touch the primary coverage rect,
390 // because that would defeat the usefulness of prepopulateRect(); instead, age prepopulated tiles out as if they were being removed.
391 for (auto& cohort : m_cohortList) {
392 if (cohort.cohort != tileInfo.cohort)
393 continue;
394 Seconds timeUntilCohortExpires = cohort.timeUntilExpiration();
395 if (timeUntilCohortExpires > 0_s) {
396 minimumRevalidationTimerDuration = std::min(minimumRevalidationTimerDuration, timeUntilCohortExpires);
397 needsTileRevalidation = true;
398 } else
399 tileLayer->removeFromSuperlayer();
400 break;
401 }
402 }
403 }
404 }
405
406 if (needsTileRevalidation)
407 m_controller.scheduleTileRevalidation(minimumRevalidationTimerDuration);
408
409 if (!m_controller.shouldAggressivelyRetainTiles()) {
410 if (m_controller.shouldTemporarilyRetainTileCohorts())
411 scheduleCohortRemoval();
412 else if (tilesInCohort) {
413 removeTilesInCohort(currCohort);
414 tilesInCohort = 0;
415 }
416 }
417
418 if (tilesInCohort)
419 startedNewCohort(currCohort);
420
421 if (validationPolicy & PruneSecondaryTiles) {
422 removeAllSecondaryTiles();
423 m_cohortList.clear();
424 }
425
426 if (validationPolicy & UnparentAllTiles) {
427 for (auto& tile : m_tiles.values())
428 tile.layer->removeFromSuperlayer();
429 }
430
431 auto boundsAtLastRevalidate = m_controller.boundsAtLastRevalidate();
432 if (boundsAtLastRevalidate != bounds) {
433 // If there are margin tiles and the bounds have grown taller or wider, then the tiles that used to
434 // be bottom or right margin tiles need to be invalidated.
435 if (m_controller.hasMargins()) {
436 if (bounds.width() > boundsAtLastRevalidate.width() || bounds.height() > boundsAtLastRevalidate.height()) {
437 IntRect boundsWithoutMargin = m_controller.boundsWithoutMargin();
438 IntRect oldBoundsWithoutMargin = m_controller.boundsAtLastRevalidateWithoutMargin();
439
440 if (bounds.height() > boundsAtLastRevalidate.height()) {
441 IntRect formerBottomMarginRect = IntRect(oldBoundsWithoutMargin.x(), oldBoundsWithoutMargin.height(),
442 oldBoundsWithoutMargin.width(), boundsWithoutMargin.height() - oldBoundsWithoutMargin.height());
443 setNeedsDisplayInRect(formerBottomMarginRect);
444 }
445
446 if (bounds.width() > boundsAtLastRevalidate.width()) {
447 IntRect formerRightMarginRect = IntRect(oldBoundsWithoutMargin.width(), oldBoundsWithoutMargin.y(),
448 boundsWithoutMargin.width() - oldBoundsWithoutMargin.width(), oldBoundsWithoutMargin.height());
449 setNeedsDisplayInRect(formerRightMarginRect);
450 }
451 }
452 }
453
454 FloatRect scaledBounds(bounds);
455 scaledBounds.scale(m_scale);
456 IntRect boundsInTileCoords(enclosingIntRect(scaledBounds));
457
458 TileIndex topLeftForBounds;
459 TileIndex bottomRightForBounds;
460 if (getTileIndexRangeForRect(boundsInTileCoords, topLeftForBounds, bottomRightForBounds)) {
461 Vector<TileIndex> tilesToRemove;
462 for (auto& index : m_tiles.keys()) {
463 if (index.y() < topLeftForBounds.y() || index.y() > bottomRightForBounds.y() || index.x() < topLeftForBounds.x() || index.x() > bottomRightForBounds.x())
464 tilesToRemove.append(index);
465 }
466 removeTiles(tilesToRemove);
467 }
468 }
469
470 // Ensure primary tile coverage tiles.
471 m_primaryTileCoverageRect = ensureTilesForRect(coverageRect, CoverageType::PrimaryTiles);
472
473 // Ensure secondary tiles (requested via prepopulateRect).
474 if (!(validationPolicy & PruneSecondaryTiles)) {
475 for (auto& secondaryCoverageRect : m_secondaryTileCoverageRects) {
476 FloatRect secondaryRectInLayerCoordinates(secondaryCoverageRect);
477 secondaryRectInLayerCoordinates.scale(1 / m_scale);
478 ensureTilesForRect(secondaryRectInLayerCoordinates, CoverageType::SecondaryTiles);
479 }
480 m_secondaryTileCoverageRects.clear();
481 }
482
483 m_controller.didRevalidateTiles();
484}

513void TileGrid::scheduleCohortRemoval()
514{
515 const Seconds cohortRemovalTimerSeconds { 1_s };
516
517 // Start the timer, or reschedule the timer from now if it's already active.
518 if (!m_cohortRemovalTimer.isActive())
519 m_cohortRemovalTimer.startRepeating(cohortRemovalTimerSeconds);

522Seconds TileGrid::TileCohortInfo::timeUntilExpiration()
523{
524 Seconds cohortLifeTimeSeconds = 2_s;
525 MonotonicTime timeThreshold = MonotonicTime::now() - cohortLifeTimeSeconds;
526 return creationTime - timeThreshold;
527}
528
529void TileGrid::cohortRemovalTimerFired()
530{
531 if (m_cohortList.isEmpty()) {
532 m_cohortRemovalTimer.stop();
533 return;
534 }
535
536 while (!m_cohortList.isEmpty() && m_cohortList.first().timeUntilExpiration() < 0_s) {
537 TileCohortInfo firstCohort = m_cohortList.takeFirst();
538 removeTilesInCohort(firstCohort.cohort);
539 }
540
541 m_controller.updateTileCoverageMap();
542}

544IntRect TileGrid::ensureTilesForRect(const FloatRect& rect, CoverageType newTileType)
545{
546 if (!m_controller.isInWindow())
547 return IntRect();
548
549 FloatRect scaledRect(rect);
550 scaledRect.scale(m_scale);
551 IntRect rectInTileCoords(enclosingIntRect(scaledRect));
552
553 TileIndex topLeft;
554 TileIndex bottomRight;
555 if (!getTileIndexRangeForRect(rectInTileCoords, topLeft, bottomRight))
556 return IntRect();
557
558 TileCohort currCohort = nextTileCohort();
559 unsigned tilesInCohort = 0;
560
561 IntRect coverageRect;
562
563 for (int y = topLeft.y(); y <= bottomRight.y(); ++y) {
564 for (int x = topLeft.x(); x <= bottomRight.x(); ++x) {
565 TileIndex tileIndex(x, y);
566
567 IntRect tileRect = rectForTileIndex(tileIndex);
568 TileInfo& tileInfo = m_tiles.add(tileIndex, TileInfo()).iterator->value;
569
570 coverageRect.unite(tileRect);
571
572 bool shouldChangeTileLayerFrame = false;
573
574 if (!tileInfo.layer) {
575 tileInfo.layer = m_controller.createTileLayer(tileRect, *this);
576 ASSERT(!m_tileRepaintCounts.contains(tileInfo.layer.get()));
577 } else {
578 // We already have a layer for this tile. Ensure that its size is correct.
579 FloatSize tileLayerSize(tileInfo.layer->bounds().size());
580 shouldChangeTileLayerFrame = tileLayerSize != FloatSize(tileRect.size());
581
582 if (shouldChangeTileLayerFrame) {
583 tileInfo.layer->setBounds(FloatRect(FloatPoint(), tileRect.size()));
584 tileInfo.layer->setPosition(tileRect.location());
585 tileInfo.layer->setNeedsDisplay();
586 }
587 }
588
589 if (newTileType == CoverageType::SecondaryTiles && !tileRect.intersects(m_primaryTileCoverageRect)) {
590 tileInfo.cohort = currCohort;
591 ++tilesInCohort;
592 }
593
594 if (!tileInfo.layer->superlayer())
595 m_containerLayer.get().appendSublayer(*tileInfo.layer);
596 }
597 }
598
599 if (tilesInCohort)
600 startedNewCohort(currCohort);
601
602 LOG_WITH_STREAM(Tiling, stream << "TileGrid " << this << " (bounds " << m_controller.bounds() << ") ensureTilesForRect: " << rect << " covered " << coverageRect);
603
604 return coverageRect;
605}
606
607IntRect TileGrid::extent() const
608{
609 TileIndex topLeft;
610 TileIndex bottomRight;
611 if (getTileIndexRangeForRect(m_primaryTileCoverageRect, topLeft, bottomRight)) {
612 // Return index of top, left tile and the number of tiles across and down.
613 return IntRect(topLeft.x(), topLeft.y(), bottomRight.x() - topLeft.x() + 1, bottomRight.y() - topLeft.y() + 1);
614 }
615
616 return IntRect();
617}

619double TileGrid::retainedTileBackingStoreMemory() const
620{
621 double totalBytes = 0;
622 for (auto& tileInfo : m_tiles.values()) {
623 if (tileInfo.layer->superlayer()) {
624 FloatRect bounds = tileInfo.layer->bounds();
625 double contentsScale = tileInfo.layer->contentsScale();
626 totalBytes += 4 * bounds.width() * contentsScale * bounds.height() * contentsScale;
627 }
628 }
629 return totalBytes;
630}

640void TileGrid::drawTileMapContents(CGContextRef context, CGRect layerBounds) const
641{
642 CGContextSetRGBFillColor(context, 0.3, 0.3, 0.3, 1);
643 CGContextFillRect(context, layerBounds);
644
645 CGFloat scaleFactor = layerBounds.size.width / m_controller.bounds().width();
646
647 CGFloat contextScale = scaleFactor / m_scale;
648 CGContextScaleCTM(context, contextScale, contextScale);
649
650 for (auto& tileInfo : m_tiles.values()) {
651 PlatformCALayer* tileLayer = tileInfo.layer.get();
652
653 CGFloat red = 1;
654 CGFloat green = 1;
655 CGFloat blue = 1;
656 CGFloat alpha = 1;
657 if (tileInfo.hasStaleContent) {
658 red = 0.25;
659 green = 0.125;
660 blue = 0;
661 } else if (m_controller.shouldAggressivelyRetainTiles() && tileInfo.cohort != visibleTileCohort) {
662 red = 0.8;
663 green = 0.8;
664 blue = 0.8;
665 }
666
667 TileCohort newestCohort = newestTileCohort();
668 TileCohort oldestCohort = oldestTileCohort();
669
670 if (!m_controller.shouldAggressivelyRetainTiles() && tileInfo.cohort != visibleTileCohort && newestCohort > oldestCohort)
671 alpha = 1 - (static_cast<float>((newestCohort - tileInfo.cohort)) / (newestCohort - oldestCohort));
672
673 CGContextSetRGBFillColor(context, red, green, blue, alpha);
674
675 if (tileLayer->superlayer()) {
676 CGContextSetLineWidth(context, 0.5 / contextScale);
677 CGContextSetRGBStrokeColor(context, 0, 0, 0, 1);
678 } else {
679 CGContextSetLineWidth(context, 1 / contextScale);
680 CGContextSetRGBStrokeColor(context, 0.2, 0.1, 0.9, 1);
681 }
682
683 CGRect frame = CGRectMake(tileLayer->position().x(), tileLayer->position().y(), tileLayer->bounds().size().width(), tileLayer->bounds().size().height());
684 CGContextFillRect(context, frame);
685 CGContextStrokeRect(context, frame);
686
687 CGContextSetRGBFillColor(context, 0, 0, 0, 0.5);
688
689 CGContextSaveGState(context);
690
691 auto repaintCount = m_tileRepaintCounts.count(tileLayer);
692 char repaintCountCharacters[lengthOfIntegerAsString(std::numeric_limits<decltype(repaintCount)>::max())];
693 writeIntegerToBuffer(repaintCount, repaintCountCharacters);
694 tileLayer->drawTextAtPoint(context, frame.origin.x + 64, frame.origin.y + 192, CGSizeMake(3, -3), 58,
695 repaintCountCharacters, lengthOfIntegerAsString(repaintCount));
696
697 CGContextRestoreGState(context);
698 }
699}

702{
703#if PLATFORM(IOS_FAMILY)
704 if (pthread_main_np())
705 WebThreadLock();

707
708 if (!platformCALayerRepaintCount(platformCALayer))
709 layerPaintBehavior |= GraphicsLayerPaintFirstTilePaint;

712 GraphicsContextStateSaver stateSaver(context);

719 PlatformCALayer::drawLayerContents(context, &m_controller.rootLayer(), dirtyRects, layerPaintBehavior);

731 m_controller.logFilledVisibleFreshTile(blankPixelCount());
