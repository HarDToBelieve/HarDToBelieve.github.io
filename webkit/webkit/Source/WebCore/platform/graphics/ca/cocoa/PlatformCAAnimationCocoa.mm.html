130CAMediaTimingFunction* toCAMediaTimingFunction(const TimingFunction* timingFunction, bool reverse)
131{
132 ASSERT(timingFunction);
133 if (is<CubicBezierTimingFunction>(timingFunction)) {
134 RefPtr<CubicBezierTimingFunction> reversed;
135 auto* function = downcast<CubicBezierTimingFunction>(timingFunction);
136
137 if (reverse) {
138 reversed = function->createReversed();
139 function = reversed.get();
140 }
141
142 float x1 = static_cast<float>(function->x1());
143 float y1 = static_cast<float>(function->y1());
144 float x2 = static_cast<float>(function->x2());
145 float y2 = static_cast<float>(function->y2());
146 return [CAMediaTimingFunction functionWithControlPoints: x1 : y1 : x2 : y2];
147 }
148
149 ASSERT(timingFunction->type() == TimingFunction::LinearFunction);
150 return [CAMediaTimingFunction functionWithName:kCAMediaTimingFunctionLinear];
151}

163PlatformCAAnimationCocoa::PlatformCAAnimationCocoa(AnimationType type, const String& keyPath)
164 : PlatformCAAnimation(type)
165{
166 switch (type) {
167 case Basic:
168 m_animation = [CABasicAnimation animationWithKeyPath:keyPath];
169 break;
170 case Group:
171 m_animation = [CAAnimationGroup animation];
172 break;
173 case Keyframe:
174 m_animation = [CAKeyframeAnimation animationWithKeyPath:keyPath];
175 break;
176 case Spring:
177 m_animation = [CASpringAnimation animationWithKeyPath:keyPath];
178 break;
179 }
180}

182PlatformCAAnimationCocoa::PlatformCAAnimationCocoa(PlatformAnimationRef animation)
183{
184 auto caAnimation = static_cast<CAAnimation *>(animation);
185 if ([caAnimation isKindOfClass:[CABasicAnimation class]]) {
186 if ([caAnimation isKindOfClass:[CASpringAnimation class]])
187 setType(Spring);
188 else
189 setType(Basic);
190 } else if ([caAnimation isKindOfClass:[CAKeyframeAnimation class]])
191 setType(Keyframe);
192 else if ([caAnimation isKindOfClass:[CAAnimationGroup class]])
193 setType(Group);
194 else {
195 ASSERT_NOT_REACHED();
196 return;
197 }
198
199 m_animation = caAnimation;
200}

206Ref<PlatformCAAnimation> PlatformCAAnimationCocoa::copy() const
207{
208 auto animation = create(animationType(), keyPath());
209
210 animation->setBeginTime(beginTime());
211 animation->setDuration(duration());
212 animation->setSpeed(speed());
213 animation->setTimeOffset(timeOffset());
214 animation->setRepeatCount(repeatCount());
215 animation->setAutoreverses(autoreverses());
216 animation->setFillMode(fillMode());
217 animation->setRemovedOnCompletion(isRemovedOnCompletion());
218 animation->setAdditive(isAdditive());
219 animation->copyTimingFunctionFrom(*this);
220 animation->setValueFunction(valueFunction());
221
222 setHasExplicitBeginTime(downcast<PlatformCAAnimationCocoa>(animation.get()).platformAnimation(), hasExplicitBeginTime(platformAnimation()));
223
224 // Copy the specific Basic or Keyframe values.
225 if (animationType() == Keyframe) {
226 animation->copyValuesFrom(*this);
227 animation->copyKeyTimesFrom(*this);
228 animation->copyTimingFunctionsFrom(*this);
229 } else {
230 animation->copyFromValueFrom(*this);
231 animation->copyToValueFrom(*this);
232 }
233
234 return animation;
235}

256void PlatformCAAnimationCocoa::setBeginTime(CFTimeInterval value)
257{
258 [m_animation setBeginTime:value];
259
260 // Also set a flag to tell us if we've passed in a 0 value.
261 // The flag is needed because later beginTime will get changed
262 // to the time at which it fired and we need to know whether
263 // or not it was 0 to begin with.
264 if (value)
265 setHasExplicitBeginTime(m_animation.get(), true);
266}

328void PlatformCAAnimationCocoa::setTimingFunction(const TimingFunction* value, bool reverse)
329{
330 switch (animationType()) {
331 case Basic:
332 case Keyframe:
333 [m_animation setTimingFunction:toCAMediaTimingFunction(value, reverse)];
334 break;
335 case Spring:
336 if (value->isSpringTimingFunction()) {
337 // FIXME: Handle reverse.
338 auto& function = *static_cast<const SpringTimingFunction*>(value);
339 CASpringAnimation *springAnimation = (CASpringAnimation *)m_animation.get();
340 springAnimation.mass = function.mass();
341 springAnimation.stiffness = function.stiffness();
342 springAnimation.damping = function.damping();
343 springAnimation.initialVelocity = function.initialVelocity();
344 }
345 break;
346 case Group:
347 break;
348 }
349}
