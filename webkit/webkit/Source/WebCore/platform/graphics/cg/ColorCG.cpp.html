57static std::optional<SRGBA<uint8_t>> roundAndClampToSRGBALossy(CGColorRef color)
58{
59 // FIXME: ExtendedColor - needs to handle color spaces.
60
61 if (!color)
62 return std::nullopt;
63
64 size_t numComponents = CGColorGetNumberOfComponents(color);
65 const CGFloat* components = CGColorGetComponents(color);
66
67 float r = 0;
68 float g = 0;
69 float b = 0;
70 float a = 0;
71
72 switch (numComponents) {
73 case 2:
74 r = g = b = components[0];
75 a = components[1];
76 break;
77 case 4:
78 r = components[0];
79 g = components[1];
80 b = components[2];
81 a = components[3];
82 break;
83 default:
84 ASSERT_NOT_REACHED();
85 }
86
87 return convertColor<SRGBA<uint8_t>>(makeFromComponentsClamping<SRGBA<float>>(r, g, b, a ));
88}

95static std::pair<CGColorSpaceRef, ColorComponents<float, 4>> convertToCGCompatibleComponents(ColorSpace colorSpace, ColorComponents<float, 4> components)
96{
97 // Some CG ports don't support all the color spaces required and return
98 // nullptr for unsupported color spaces. In those cases, we eagerly convert
99 // the color into either extended sRGB or normal sRGB, if extended sRGB is
100 // not supported.
101
102 auto cgColorSpace = cachedNullableCGColorSpace(colorSpace);
103 if (!cgColorSpace) {
104#if HAVE(CORE_GRAPHICS_EXTENDED_SRGB_COLOR_SPACE)
105 auto componentsConvertedToExtendedSRGBA = callWithColorType(components, colorSpace, [] (const auto& color) {
106 return asColorComponents(convertColor<ExtendedSRGBA<float>>(color));
107 });
108 return { extendedSRGBColorSpaceRef(), componentsConvertedToExtendedSRGBA };
109#else
110 auto componentsConvertedToSRGBA = callWithColorType(components, colorSpace, [] (const auto& color) {
111 return asColorComponents(convertColor<SRGBA<float>>(color));
112 });
113 return { sRGBColorSpaceRef(), componentsConvertedToSRGBA };
114#endif
115 }
116
117 return { cgColorSpace, components };
118}

120static RetainPtr<CGColorRef> createCGColor(const Color& color)
121{
122 auto [colorSpace, components] = color.colorSpaceAndComponents();
123 auto [cgColorSpace, cgCompatibleComponents] = convertToCGCompatibleComponents(colorSpace, components);
124
125 auto [c1, c2, c3, c4] = cgCompatibleComponents;
126 CGFloat cgFloatComponents[4] { c1, c2, c3, c4 };
127
128 return adoptCF(CGColorCreate(cgColorSpace, cgFloatComponents));
129}

131CGColorRef cachedCGColor(const Color& color)
132{
133 if (auto srgb = color.tryGetAsSRGBABytes()) {
134 switch (PackedColor::RGBA { *srgb }.value) {
135 case PackedColor::RGBA { Color::transparentBlack }.value: {
136 static LazyNeverDestroyed<RetainPtr<CGColorRef>> transparentCGColor;
137 static std::once_flag onceFlag;
138 std::call_once(onceFlag, [] {
139 transparentCGColor.construct(createCGColor(Color::transparentBlack));
140 });
141 return transparentCGColor.get().get();
142 }
143 case PackedColor::RGBA { Color::black }.value: {
144 static LazyNeverDestroyed<RetainPtr<CGColorRef>> blackCGColor;
145 static std::once_flag onceFlag;
146 std::call_once(onceFlag, [] {
147 blackCGColor.construct(createCGColor(Color::black));
148 });
149 return blackCGColor.get().get();
150 }
151 case PackedColor::RGBA { Color::white }.value: {
152 static LazyNeverDestroyed<RetainPtr<CGColorRef>> whiteCGColor;
153 static std::once_flag onceFlag;
154 std::call_once(onceFlag, [] {
155 whiteCGColor.construct(createCGColor(Color::white));
156 });
157 return whiteCGColor.get().get();
158 }
159 }
160 }
161
162 static Lock cachedColorLock;
163 Locker locker { cachedColorLock };
164
165 static NeverDestroyed<TinyLRUCache<Color, RetainPtr<CGColorRef>, 32>> cache;
166 return cache.get().get(color).get();
167}

169ColorComponents<float, 4> platformConvertColorComponents(ColorSpace inputColorSpace, ColorComponents<float, 4> inputColorComponents, const DestinationColorSpace& outputColorSpace)
170{
171 // FIXME: Investigate optimizing this to use the builtin color conversion code for supported color spaces.
172
173 auto [cgInputColorSpace, cgCompatibleComponents] = convertToCGCompatibleComponents(inputColorSpace, inputColorComponents);
174 if (cgInputColorSpace == outputColorSpace.platformColorSpace())
175 return cgCompatibleComponents;
176
177 auto [c1, c2, c3, c4] = cgCompatibleComponents;
178 CGFloat sourceComponents[4] { c1, c2, c3, c4 };
179 CGFloat destinationComponents[4] { };
180
181 auto transform = adoptCF(CGColorTransformCreate(outputColorSpace.platformColorSpace(), nullptr));
182 auto result = CGColorTransformConvertColorComponents(transform.get(), cgInputColorSpace, kCGRenderingIntentDefault, sourceComponents, destinationComponents);
183 ASSERT_UNUSED(result, result);
184 return { static_cast<float>(destinationComponents[0]), static_cast<float>(destinationComponents[1]), static_cast<float>(destinationComponents[2]), static_cast<float>(destinationComponents[3]) };
185}
