42IntSize ImageBufferCGBitmapBackend::calculateSafeBackendSize(const Parameters& parameters)
43{
44 IntSize backendSize = calculateBackendSize(parameters);
45 if (backendSize.isEmpty())
46 return backendSize;
47
48 auto bytesPerRow = 4 * CheckedUint32(backendSize.width());
49 if (bytesPerRow.hasOverflowed())
50 return { };
51
52 CheckedSize numBytes = CheckedUint32(backendSize.height()) * bytesPerRow;
53 if (numBytes.hasOverflowed())
54 return { };
55
56 return backendSize;
57}

59size_t ImageBufferCGBitmapBackend::calculateMemoryCost(const Parameters& parameters)
60{
61 IntSize backendSize = calculateBackendSize(parameters);
62 return ImageBufferBackend::calculateMemoryCost(backendSize, calculateBytesPerRow(backendSize));
63}

65std::unique_ptr<ImageBufferCGBitmapBackend> ImageBufferCGBitmapBackend::create(const Parameters& parameters, const HostWindow*)
66{
67 ASSERT(parameters.pixelFormat == PixelFormat::BGRA8);
68
69 IntSize backendSize = calculateSafeBackendSize(parameters);
70 if (backendSize.isEmpty())
71 return nullptr;
72
73 void* data;
74 unsigned bytesPerRow = 4 * backendSize.width();
75
76 if (!tryFastCalloc(backendSize.height(), bytesPerRow).getValue(data))
77 return nullptr;
78
79 ASSERT(!(reinterpret_cast<intptr_t>(data) & 3));
80
81 size_t numBytes = backendSize.height() * bytesPerRow;
82 verifyImageBufferIsBigEnough(data, numBytes);
83
84 auto cgContext = adoptCF(CGBitmapContextCreate(data, backendSize.width(), backendSize.height(), 8, bytesPerRow, parameters.colorSpace.platformColorSpace(), kCGImageAlphaPremultipliedFirst | kCGBitmapByteOrder32Host));
85 if (!cgContext)
86 return nullptr;
87
88 auto context = makeUnique<GraphicsContextCG>(cgContext.get());
89
90 auto dataProvider = adoptCF(CGDataProviderCreateWithData(nullptr, data, numBytes, [] (void*, const void* data, size_t) {
91 fastFree(const_cast<void*>(data));
92 }));
93
94 return std::unique_ptr<ImageBufferCGBitmapBackend>(new ImageBufferCGBitmapBackend(parameters, data, WTFMove(dataProvider), WTFMove(context)));
95}

97std::unique_ptr<ImageBufferCGBitmapBackend> ImageBufferCGBitmapBackend::create(const Parameters& parameters, const GraphicsContext& context)
98{
99 if (auto cgColorSpace = context.hasPlatformContext() ? contextColorSpace(context) : nullptr) {
100 auto overrideParameters = parameters;
101 overrideParameters.colorSpace = DestinationColorSpace { cgColorSpace };
102
103 return ImageBufferCGBitmapBackend::create(overrideParameters, nullptr);
104 }
105
106 return ImageBufferCGBitmapBackend::create(parameters, nullptr);
107}

138RefPtr<NativeImage> ImageBufferCGBitmapBackend::copyNativeImage(BackingStoreCopy copyBehavior) const
139{
140 switch (copyBehavior) {
141 case CopyBackingStore:
142 return NativeImage::create(adoptCF(CGBitmapContextCreateImage(context().platformContext())));
143
144 case DontCopyBackingStore:
145 auto backendSize = this->backendSize();
146 return NativeImage::create(adoptCF(CGImageCreate(
147 backendSize.width(), backendSize.height(), 8, 32, bytesPerRow(),
148 colorSpace().platformColorSpace(), kCGImageAlphaPremultipliedFirst | kCGBitmapByteOrder32Host, m_dataProvider.get(),
149 0, true, kCGRenderingIntentDefault)));
150 }
151
152 ASSERT_NOT_REACHED();
153 return nullptr;
154}
