62static RetainPtr<CFMutableDictionaryRef> createImageSourceOptions()
63{
64 RetainPtr<CFMutableDictionaryRef> options = adoptCF(CFDictionaryCreateMutable(nullptr, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
65 CFDictionarySetValue(options.get(), kCGImageSourceShouldCache, kCFBooleanTrue);
66 CFDictionarySetValue(options.get(), kCGImageSourceShouldPreferRGB32, kCFBooleanTrue);
67 CFDictionarySetValue(options.get(), kCGImageSourceSkipMetadata, kCFBooleanTrue);
68#if HAVE(IMAGE_RESTRICTED_DECODING)
69 CFDictionarySetValue(options.get(), kCGImageSourceEnableRestrictedDecoding, kCFBooleanTrue);
70#endif
71 return options;
72}

89static RetainPtr<CFMutableDictionaryRef> appendImageSourceOption(RetainPtr<CFMutableDictionaryRef>&& options, SubsamplingLevel subsamplingLevel)
90{
91 subsamplingLevel = std::min(SubsamplingLevel::Last, std::max(SubsamplingLevel::First, subsamplingLevel));
92 int subsampleInt = 1 << static_cast<int>(subsamplingLevel); // [0..3] => [1, 2, 4, 8]
93 auto subsampleNumber = adoptCF(CFNumberCreate(nullptr, kCFNumberIntType, &subsampleInt));
94 CFDictionarySetValue(options.get(), kCGImageSourceSubsampleFactor, subsampleNumber.get());
95 return WTFMove(options);
96}

98static RetainPtr<CFMutableDictionaryRef> appendImageSourceOption(RetainPtr<CFMutableDictionaryRef>&& options, const IntSize& sizeForDrawing)
99{
100 unsigned maxDimension = DecodingOptions::maxDimension(sizeForDrawing);
101 RetainPtr<CFNumberRef> maxDimensionNumber = adoptCF(CFNumberCreate(nullptr, kCFNumberIntType, &maxDimension));
102 CFDictionarySetValue(options.get(), kCGImageSourceThumbnailMaxPixelSize, maxDimensionNumber.get());
103 return WTFMove(options);
104}

106static RetainPtr<CFMutableDictionaryRef> appendImageSourceOptions(RetainPtr<CFMutableDictionaryRef>&& options, SubsamplingLevel subsamplingLevel, const IntSize& sizeForDrawing)
107{
108 if (subsamplingLevel != SubsamplingLevel::Default)
109 options = appendImageSourceOption(WTFMove(options), subsamplingLevel);
110
111 options = appendImageSourceOption(WTFMove(options), sizeForDrawing);
112 return WTFMove(options);

129static CFDictionaryRef animationPropertiesFromProperties(CFDictionaryRef properties)
130{
131 if (!properties)
132 return nullptr;
133
134 if (auto animationProperties = (CFDictionaryRef)CFDictionaryGetValue(properties, kCGImagePropertyGIFDictionary))
135 return animationProperties;
136#if HAVE(WEBP)
137 if (auto animationProperties = (CFDictionaryRef)CFDictionaryGetValue(properties, kCGImagePropertyWebPDictionary))
138 return animationProperties;
139#endif
140 if (auto animationProperties = (CFDictionaryRef)CFDictionaryGetValue(properties, kCGImagePropertyPNGDictionary))
141 return animationProperties;
142
143 return (CFDictionaryRef)CFDictionaryGetValue(properties, WebCoreCGImagePropertyHEICSDictionary);
144}

146static CFDictionaryRef animationHEICSPropertiesFromProperties(CFDictionaryRef properties, size_t index)
147{
148 if (!properties)
149 return nullptr;
150
151 // For HEICS images, ImageIO does not create a properties dictionary for each HEICS frame. Instead it maintains
152 // all frames' information in the image properties dictionary. Here is how ImageIO structures the properties
153 // dictionary for HEICS image:
154 // "{HEICS}" = {
155 // FrameInfo = ( { DelayTime = "0.1"; }, { DelayTime = "0.1"; }, ... );
156 // LoopCount = 0;
157 // ...
158 // };
159 CFDictionaryRef heicsProperties = (CFDictionaryRef)CFDictionaryGetValue(properties, WebCoreCGImagePropertyHEICSDictionary);
160 if (!heicsProperties)
161 return nullptr;
162
163 CFArrayRef frameInfoArray = (CFArrayRef)CFDictionaryGetValue(heicsProperties, WebCoreCGImagePropertyHEICSFrameInfoArray);
164 if (!frameInfoArray)
165 return nullptr;
166
167 return (CFDictionaryRef)CFArrayGetValueAtIndex(frameInfoArray, index);
168}

182static bool mayHaveDensityCorrectedSize(CFDictionaryRef imageProperties)
183{
184 ASSERT(imageProperties);
185 auto resolutionXProperty = (CFNumberRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyDPIWidth);
186 auto resolutionYProperty = (CFNumberRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyDPIHeight);
187 if (!resolutionXProperty || !resolutionYProperty)
188 return false;
189
190 float resolutionX, resolutionY;
191 return CFNumberGetValue(resolutionXProperty, kCFNumberFloat32Type, &resolutionX)
192 && CFNumberGetValue(resolutionYProperty, kCFNumberFloat32Type, &resolutionY)
193 && (resolutionX != ImageResolution::DefaultResolution || resolutionY != ImageResolution::DefaultResolution);
194}

196static std::optional<IntSize> densityCorrectedSizeFromProperties(CFDictionaryRef imageProperties)
197{
198 ASSERT(imageProperties);
199 auto exifDictionary = (CFDictionaryRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyExifDictionary);
200 auto tiffDictionary = (CFDictionaryRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyTIFFDictionary);
201
202 if (!exifDictionary || !tiffDictionary)
203 return std::nullopt;
204
205 auto widthProperty = (CFNumberRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyPixelWidth);
206 auto heightProperty = (CFNumberRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyPixelHeight);
207 auto preferredWidthProperty = (CFNumberRef)CFDictionaryGetValue(exifDictionary, kCGImagePropertyExifPixelXDimension);
208 auto preferredHeightProperty = (CFNumberRef)CFDictionaryGetValue(exifDictionary, kCGImagePropertyExifPixelYDimension);
209 auto resolutionXProperty = (CFNumberRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyDPIWidth);
210 auto resolutionYProperty = (CFNumberRef)CFDictionaryGetValue(imageProperties, kCGImagePropertyDPIHeight);
211 auto resolutionUnitProperty = (CFNumberRef)CFDictionaryGetValue(tiffDictionary, kCGImagePropertyTIFFResolutionUnit);
212
213 if (!preferredWidthProperty || !preferredHeightProperty || !resolutionXProperty || !resolutionYProperty || !resolutionUnitProperty)
214 return std::nullopt;
215
216 int resolutionUnit;
217 float sourceWidth, sourceHeight, preferredWidth, preferredHeight, resolutionWidth, resolutionHeight;
218 if (!CFNumberGetValue(widthProperty, kCFNumberFloat32Type, &sourceWidth)
219 || !CFNumberGetValue(heightProperty, kCFNumberFloat32Type, &sourceHeight)
220 || !CFNumberGetValue(preferredWidthProperty, kCFNumberFloat32Type, &preferredWidth)
221 || !CFNumberGetValue(preferredHeightProperty, kCFNumberFloat32Type, &preferredHeight)
222 || !CFNumberGetValue(resolutionXProperty, kCFNumberFloat32Type, &resolutionWidth)
223 || !CFNumberGetValue(resolutionYProperty, kCFNumberFloat32Type, &resolutionHeight)
224 || !CFNumberGetValue(resolutionUnitProperty, kCFNumberIntType, &resolutionUnit)) {
225 return std::nullopt;
226 }
227
228 return ImageResolution::densityCorrectedSize(FloatSize(sourceWidth, sourceHeight), {
229 { preferredWidth, preferredHeight },
230 { resolutionWidth, resolutionHeight },
231 static_cast<ImageResolution::ResolutionUnit>(resolutionUnit)
232 });
233}

236size_t sharedBufferGetBytesAtPosition(void* info, void* buffer, off_t position, size_t count)
237{
238 SharedBuffer* sharedBuffer = static_cast<SharedBuffer*>(info);
239 size_t sourceSize = sharedBuffer->size();
240 if (position >= sourceSize)
241 return 0;
242
243 auto* source = sharedBuffer->data() + position;
244 size_t amount = std::min<size_t>(count, sourceSize - position);
245 memcpy(buffer, source, amount);
246 return amount;
247}

271size_t ImageDecoderCG::bytesDecodedToDetermineProperties() const
272{
273 // Measured by tracing malloc/calloc calls on Mac OS 10.6.6, x86_64.
274 // A non-zero value ensures cached images with no decoded frames still enter
275 // the live decoded resources list when the CGImageSource decodes image
276 // properties, allowing the cache to prune the partially decoded image.
277 // This value is likely to be inaccurate on other platforms, but the overall
278 // behavior is unchanged.
279 return 13088;
280}

303EncodedDataStatus ImageDecoderCG::encodedDataStatus() const
304{
305 if (m_encodedDataStatus == EncodedDataStatus::Error || m_encodedDataStatus == EncodedDataStatus::Complete)
306 return m_encodedDataStatus;
307
308 // The image source UTI can be changed while receiving more encoded data.
309 String uti = this->uti();
310 if (uti.isEmpty())
311 return EncodedDataStatus::Unknown;
312
313 if (!isSupportedImageType(uti)) {
314 m_encodedDataStatus = EncodedDataStatus::Error;
315 return m_encodedDataStatus;
316 }
317
318 switch (CGImageSourceGetStatus(m_nativeDecoder.get())) {
319 case kCGImageStatusUnknownType:
320 m_encodedDataStatus = EncodedDataStatus::Error;
321 break;
322
323 case kCGImageStatusUnexpectedEOF:
324 case kCGImageStatusInvalidData:
325 case kCGImageStatusReadingHeader:
326 // Ragnaros yells: TOO SOON! You have awakened me TOO SOON, Executus!
327 if (!m_isAllDataReceived)
328 m_encodedDataStatus = EncodedDataStatus::Unknown;
329 else
330 m_encodedDataStatus = EncodedDataStatus::Error;
331 break;
332
333 case kCGImageStatusIncomplete: {
334 if (m_encodedDataStatus == EncodedDataStatus::SizeAvailable)
335 break;
336
337 auto image0Properties = adoptCF(CGImageSourceCopyPropertiesAtIndex(m_nativeDecoder.get(), 0, imageSourceOptions().get()));
338 if (!image0Properties || !CFDictionaryContainsKey(image0Properties.get(), kCGImagePropertyPixelWidth) || !CFDictionaryContainsKey(image0Properties.get(), kCGImagePropertyPixelHeight)) {
339 m_encodedDataStatus = EncodedDataStatus::TypeAvailable;
340 break;
341 }
342
343 m_encodedDataStatus = EncodedDataStatus::SizeAvailable;
344 break;
345 }
346
347 case kCGImageStatusComplete:
348 m_encodedDataStatus = EncodedDataStatus::Complete;
349 break;
350 }
351
352 return m_encodedDataStatus;
353}

360RepetitionCount ImageDecoderCG::repetitionCount() const
361{
362 RetainPtr<CFDictionaryRef> properties = adoptCF(CGImageSourceCopyProperties(m_nativeDecoder.get(), imageSourceOptions().get()));
363 CFDictionaryRef animationProperties = animationPropertiesFromProperties(properties.get());
364
365 // Turns out we're not an animated image after all, so we don't animate.
366 if (!animationProperties)
367 return RepetitionCountNone;
368
369 CFNumberRef num = (CFNumberRef)CFDictionaryGetValue(animationProperties, WebCoreCGImagePropertyLoopCount);
370
371 // No property means loop once.
372 if (!num)
373 return RepetitionCountOnce;
374
375 RepetitionCount loopCount;
376 CFNumberGetValue(num, kCFNumberIntType, &loopCount);
377
378 // A property with value 0 means loop forever.
379 if (!loopCount)
380 return RepetitionCountInfinite;
381
382#if HAVE(CGIMAGESOURCE_WITH_ACCURATE_LOOP_COUNT)
383 return loopCount;
384#else
385 if (!isGIFImageType(uti()))
386 return loopCount;
387
388 // For GIF and loopCount > 0, the specs is not clear about it. But it looks the meaning
389 // is: play once + loop loopCount which is equivalent to play loopCount + 1.
390 return loopCount + 1;
391#endif
392}

394std::optional<IntPoint> ImageDecoderCG::hotSpot() const
395{
396 auto properties = adoptCF(CGImageSourceCopyPropertiesAtIndex(m_nativeDecoder.get(), 0, imageSourceOptions().get()));
397 if (!properties)
398 return std::nullopt;
399
400 int x = -1, y = -1;
401 CFNumberRef num = (CFNumberRef)CFDictionaryGetValue(properties.get(), CFSTR("hotspotX"));
402 if (!num || !CFNumberGetValue(num, kCFNumberIntType, &x))
403 return std::nullopt;
404
405 num = (CFNumberRef)CFDictionaryGetValue(properties.get(), CFSTR("hotspotY"));
406 if (!num || !CFNumberGetValue(num, kCFNumberIntType, &y))
407 return std::nullopt;
408
409 if (x < 0 || y < 0)
410 return std::nullopt;
411
412 return IntPoint(x, y);
413}

435bool ImageDecoderCG::frameIsCompleteAtIndex(size_t index) const
436{
437 ASSERT(frameCount());
438 // CGImageSourceGetStatusAtIndex() changes the return status value from kCGImageStatusIncomplete
439 // to kCGImageStatusComplete only if (index > 1 && index < frameCount() - 1). To get an accurate
440 // result for the last frame (or the single frame of the static image) use CGImageSourceGetStatus()
441 // instead for this frame.
442 if (index == frameCount() - 1)
443 return CGImageSourceGetStatus(m_nativeDecoder.get()) == kCGImageStatusComplete;
444 return CGImageSourceGetStatusAtIndex(m_nativeDecoder.get(), index) == kCGImageStatusComplete;
445}

447ImageDecoder::FrameMetadata ImageDecoderCG::frameMetadataAtIndex(size_t index) const
448{
449 RetainPtr<CFDictionaryRef> properties = adoptCF(CGImageSourceCopyPropertiesAtIndex(m_nativeDecoder.get(), index, imageSourceOptions().get()));
450 if (!properties)
451 return { };
452
453 auto orientation = orientationFromProperties(properties.get());
454 if (!mayHaveDensityCorrectedSize(properties.get()))
455 return { orientation, std::nullopt };
456
457 auto propertiesWithMetadata = adoptCF(CGImageSourceCopyPropertiesAtIndex(m_nativeDecoder.get(), index, createImageSourceMetadataOptions().get()));
458 if (!propertiesWithMetadata)
459 return { orientation, std::nullopt };
460
461 return { orientation, densityCorrectedSizeFromProperties(propertiesWithMetadata.get()) };
462}

464Seconds ImageDecoderCG::frameDurationAtIndex(size_t index) const
465{
466 RetainPtr<CFDictionaryRef> properties = nullptr;
467 RetainPtr<CFDictionaryRef> frameProperties = adoptCF(CGImageSourceCopyPropertiesAtIndex(m_nativeDecoder.get(), index, imageSourceOptions().get()));
468 CFDictionaryRef animationProperties = animationPropertiesFromProperties(frameProperties.get());
469
470 if (frameProperties && !animationProperties) {
471 properties = adoptCF(CGImageSourceCopyProperties(m_nativeDecoder.get(), imageSourceOptions().get()));
472 animationProperties = animationHEICSPropertiesFromProperties(properties.get(), index);
473 }
474
475 // Use the unclamped frame delay if it exists. Otherwise use the clamped frame delay.
476 float value = 0;
477 if (animationProperties) {
478 if (CFNumberRef num = (CFNumberRef)CFDictionaryGetValue(animationProperties, WebCoreCGImagePropertyUnclampedDelayTime))
479 CFNumberGetValue(num, kCFNumberFloatType, &value);
480 else if (CFNumberRef num = (CFNumberRef)CFDictionaryGetValue(animationProperties, WebCoreCGImagePropertyDelayTime))
481 CFNumberGetValue(num, kCFNumberFloatType, &value);
482 }
483
484 Seconds duration(value);
485
486 // Many annoying ads specify a 0 duration to make an image flash as quickly as possible.
487 // We follow Firefox's behavior and use a duration of 100 ms for any frames that specify
488 // a duration of <= 10 ms. See <rdar://problem/7689300> and <http://webkit.org/b/36082>
489 // for more information.
490 if (duration < 11_ms)
491 return 100_ms;
492 return duration;
493}

500bool ImageDecoderCG::frameHasAlphaAtIndex(size_t index) const
501{
502 if (!frameIsCompleteAtIndex(index))
503 return true;
504
505 String uti = this->uti();
506
507 // Return false if there is no image type or the image type is JPEG, because
508 // JPEG does not support alpha transparency.
509 if (uti.isEmpty() || uti == "public.jpeg")
510 return false;
511
512 // FIXME: Could return false for other non-transparent image formats.
513 // FIXME: Could maybe return false for a GIF Frame if we have enough info in the GIF properties dictionary
514 // to determine whether or not a transparent color was defined.
515 return true;
516}

523PlatformImagePtr ImageDecoderCG::createFrameImageAtIndex(size_t index, SubsamplingLevel subsamplingLevel, const DecodingOptions& decodingOptions)
524{
525 LOG(Images, "ImageDecoder %p createFrameImageAtIndex %lu", this, index);
526 RetainPtr<CFDictionaryRef> options;
527 RetainPtr<CGImageRef> image;
528
529 auto size = frameSizeAtIndex(index, SubsamplingLevel::Default);
530
531 if (!decodingOptions.isSynchronous()) {
532 // Don't consider the subsamplingLevel when comparing the image native size with sizeForDrawing.
533
534 if (decodingOptions.hasSizeForDrawing()) {
535 // See which size is smaller: the image native size or the sizeForDrawing.
536 std::optional<IntSize> sizeForDrawing = decodingOptions.sizeForDrawing();
537 if (sizeForDrawing.value().unclampedArea() < size.unclampedArea())
538 size = sizeForDrawing.value();
539 }
540
541 options = imageSourceAsyncOptions(subsamplingLevel, size);
542 image = adoptCF(CGImageSourceCreateThumbnailAtIndex(m_nativeDecoder.get(), index, options.get()));
543 } else {
544 // Decode an image synchronously for its native size.
545 options = imageSourceOptions(subsamplingLevel);
546 image = adoptCF(CGImageSourceCreateImageAtIndex(m_nativeDecoder.get(), index, options.get()));
547 }
548
549#if PLATFORM(IOS_FAMILY)
550 // <rdar://problem/7371198> - CoreGraphics changed the default caching behaviour in iOS 4.0 to kCGImageCachingTransient
551 // which caused a performance regression for us since the images had to be resampled/recreated every time we called
552 // CGContextDrawImage. We now tell CG to cache the drawn images. See also <rdar://problem/14366755> -
553 // CoreGraphics needs to un-deprecate kCGImageCachingTemporary since it's still not the default.
554 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
555 CGImageSetCachingFlags(image.get(), kCGImageCachingTemporary);
556 ALLOW_DEPRECATED_DECLARATIONS_END
557#endif // PLATFORM(IOS_FAMILY)
558
559 String uti = this->uti();
560 if (uti.isEmpty() || uti != "public.xbitmap-image")
561 return image;
562
563 // If it is an xbm image, mask out all the white areas to render them transparent.
564 const CGFloat maskingColors[6] = {255, 255, 255, 255, 255, 255};
565 RetainPtr<CGImageRef> maskedImage = adoptCF(CGImageCreateWithMaskingColors(image.get(), maskingColors));
566 return maskedImage ? maskedImage : image;
567}

569void ImageDecoderCG::setData(SharedBuffer& data, bool allDataReceived)
570{
571 m_isAllDataReceived = allDataReceived;
572
573#if PLATFORM(COCOA)
574 // On Mac the NSData inside the SharedBuffer can be secretly appended to without the SharedBuffer's knowledge.
575 // We use SharedBuffer's ability to wrap itself inside CFData to get around this, ensuring that ImageIO is
576 // really looking at the SharedBuffer.
577 CGImageSourceUpdateData(m_nativeDecoder.get(), data.createCFData().get(), allDataReceived);
578#else
579 // Create a CGDataProvider to wrap the SharedBuffer.
580 data.ref();
581 // We use the GetBytesAtPosition callback rather than the GetBytePointer one because SharedBuffer
582 // does not provide a way to lock down the byte pointer and guarantee that it won't move, which
583 // is a requirement for using the GetBytePointer callback.
584 CGDataProviderDirectCallbacks providerCallbacks = { 0, 0, 0, sharedBufferGetBytesAtPosition, sharedBufferRelease };
585 RetainPtr<CGDataProviderRef> dataProvider = adoptCF(CGDataProviderCreateDirect(&data, data.size(), &providerCallbacks));
586 CGImageSourceUpdateDataProvider(m_nativeDecoder.get(), dataProvider.get(), allDataReceived);
587#endif
588}
