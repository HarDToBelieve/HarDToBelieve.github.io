44static String transcodeImage(const String& path, const String& destinationUTI, const String& destinationExtension)
45{
46 auto sourceURL = adoptCF(CFURLCreateWithFileSystemPath(kCFAllocatorDefault, path.createCFString().get(), kCFURLPOSIXPathStyle, false));
47 auto source = adoptCF(CGImageSourceCreateWithURL(sourceURL.get(), nullptr));
48 if (!source)
49 return nullString();
50
51 auto sourceUTI = String(CGImageSourceGetType(source.get()));
52 if (sourceUTI == destinationUTI)
53 return nullString();
54
55#if !HAVE(IMAGEIO_FIX_FOR_RADAR_59589723)
56 auto sourceMIMEType = MIMETypeFromUTI(sourceUTI);
57 if (sourceMIMEType == "image/heif"_s || sourceMIMEType == "image/heic"_s) {
58 static std::once_flag onceFlag;
59 std::call_once(onceFlag, [&] {
60 // This call will force ImageIO to load the symbols of the HEIF reader. This
61 // bug is already fixed in ImageIO of macOS Big Sur <rdar://problem/59589723>.
62 CGImageSourceGetCount(source.get());
63 });
64 }
65#endif
66
67 // It is important to add the appropriate file extension to the temporary file path.
68 // The File object depends solely on the extension to know the MIME type of the file.
69 auto suffix = makeString('.', destinationExtension);
70
71 FileSystem::PlatformFileHandle destinationFileHandle;
72 String destinationPath = FileSystem::openTemporaryFile("tempImage"_s, destinationFileHandle, suffix);
73 if (destinationFileHandle == FileSystem::invalidPlatformFileHandle) {
74 RELEASE_LOG_ERROR(Images, "transcodeImage: Destination image could not be created: %s %s\n", path.utf8().data(), destinationUTI.utf8().data());
75 return nullString();
76 }
77
78 CGDataConsumerCallbacks callbacks = {
79 [](void* info, const void* buffer, size_t count) -> size_t {
80 auto handle = *static_cast<FileSystem::PlatformFileHandle*>(info);
81 return FileSystem::writeToFile(handle, buffer, count);
82 },
83 nullptr
84 };
85
86 auto consumer = adoptCF(CGDataConsumerCreate(&destinationFileHandle, &callbacks));
87 auto destination = adoptCF(CGImageDestinationCreateWithDataConsumer(consumer.get(), destinationUTI.createCFString().get(), 1, nullptr));
88
89 CGImageDestinationAddImageFromSource(destination.get(), source.get(), 0, nullptr);
90
91 if (!CGImageDestinationFinalize(destination.get())) {
92 RELEASE_LOG_ERROR(Images, "transcodeImage: Image transcoding fails: %s %s\n", path.utf8().data(), destinationUTI.utf8().data());
93 FileSystem::closeFile(destinationFileHandle);
94 FileSystem::deleteFile(destinationPath);
95 return nullString();
96 }
97
98 FileSystem::closeFile(destinationFileHandle);
99 return destinationPath;
100}

102Vector<String> findImagesForTranscoding(const Vector<String>& paths, const Vector<String>& allowedMIMETypes)
103{
104 Vector<String> transcodingPaths;
105 transcodingPaths.reserveInitialCapacity(paths.size());
106
107 bool needsTranscoding = false;
108
109 for (auto& path : paths) {
110 // Append a path of the image which needs transcoding. Otherwise append a null string.
111 if (!allowedMIMETypes.contains(WebCore::MIMETypeRegistry::mimeTypeForPath(path))) {
112 transcodingPaths.uncheckedAppend(path);
113 needsTranscoding = true;
114 } else
115 transcodingPaths.uncheckedAppend(nullString());
116 }
117
118 // If none of the files needs image transcoding, return an empty Vector.
119 return needsTranscoding ? transcodingPaths : Vector<String>();
120}

122Vector<String> transcodeImages(const Vector<String>& paths, const String& destinationUTI, const String& destinationExtension)
123{
124 ASSERT(!destinationUTI.isNull());
125 ASSERT(!destinationExtension.isNull());
126
127 Vector<String> transcodedPaths;
128 transcodedPaths.reserveInitialCapacity(paths.size());
129
130 for (auto& path : paths) {
131 // Append the transcoded path if the image needs transcoding. Otherwise append a null string.
132 if (!path.isNull())
133 transcodedPaths.uncheckedAppend(transcodeImage(path, destinationUTI, destinationExtension));
134 else
135 transcodedPaths.uncheckedAppend(nullString());
136 }
137
138 return transcodedPaths;
139}
