48static RetainPtr<CGContextRef> createScratchContext()
49{
50 CGDataConsumerCallbacks callbacks = { putBytesNowhere, 0 };
51 auto consumer = adoptCF(CGDataConsumerCreate(0, &callbacks));
52 auto context = adoptCF(CGPDFContextCreate(consumer.get(), 0, 0));
53
54 CGFloat black[4] = { 0, 0, 0, 1 };
55 CGContextSetFillColor(context.get(), black);
56 CGContextSetStrokeColor(context.get(), black);
57
58 return context;
59}

64 return context.get().get();

67Path Path::polygonPathFromPoints(const Vector<FloatPoint>& points)
68{
69 Path path;
70 if (points.size() < 2)
71 return path;
72
73 Vector<CGPoint, 32> cgPoints;
74 cgPoints.reserveInitialCapacity(points.size());
75 for (size_t i = 0; i < points.size(); ++i)
76 cgPoints.uncheckedAppend(points[i]);
77
78 CGPathAddLines(path.ensurePlatformPath(), nullptr, cgPoints.data(), cgPoints.size());
79 path.closeSubpath();
80 return path;
81}

83void Path::createCGPath() const
84{
85 if (m_path)
86 return;
87
88 m_path = adoptCF(CGPathCreateMutable());
89
90 WTF::switchOn(m_inlineData,
91 [&](Monostate) { }, // Start with an empty path.
92 [&](const MoveData& move) {
93 CGPathMoveToPoint(m_path.get(), nullptr, move.location.x(), move.location.y());
94 },
95 [&](const LineData& line) {
96 CGPathMoveToPoint(m_path.get(), nullptr, line.start.x(), line.start.y());
97 CGPathAddLineToPoint(m_path.get(), nullptr, line.end.x(), line.end.y());
98 },
99 [&](const ArcData& arc) {
100 if (arc.type == ArcData::Type::LineAndArc || arc.type == ArcData::Type::ClosedLineAndArc)
101 CGPathMoveToPoint(m_path.get(), nullptr, arc.start.x(), arc.start.y());
102 CGPathAddArc(m_path.get(), nullptr, arc.center.x(), arc.center.y(), arc.radius, arc.startAngle, arc.endAngle, arc.clockwise);
103 if (arc.type == ArcData::Type::ClosedLineAndArc)
104 CGPathAddLineToPoint(m_path.get(), nullptr, arc.start.x(), arc.start.y());
105 },
106 [&](const QuadCurveData& curve) {
107 CGPathMoveToPoint(m_path.get(), nullptr, curve.startPoint.x(), curve.startPoint.y());
108 CGPathAddQuadCurveToPoint(m_path.get(), nullptr, curve.controlPoint.x(), curve.controlPoint.y(), curve.endPoint.x(), curve.endPoint.y());
109 },
110 [&](const BezierCurveData& curve) {
111 CGPathMoveToPoint(m_path.get(), nullptr, curve.startPoint.x(), curve.startPoint.y());
112 CGPathAddCurveToPoint(m_path.get(), nullptr, curve.controlPoint1.x(), curve.controlPoint1.y(), curve.controlPoint2.x(), curve.controlPoint2.y(), curve.endPoint.x(), curve.endPoint.y());
113 }
114 );
115}

132PlatformPathPtr Path::ensurePlatformPath()
133{
134 createCGPath();
135 if (m_copyPathBeforeMutation) {
136 if (CFGetRetainCount(m_path.get()) > 1)
137 m_path = adoptCF(CGPathCreateMutableCopy(m_path.get()));
138 m_copyPathBeforeMutation = false;
139 }
140 m_inlineData = Monostate { };
141 return m_path.get();
142}

187static void copyClosingSubpathsApplierFunction(void* info, const CGPathElement* element)
188{
189 CGMutablePathRef path = static_cast<CGMutablePathRef>(info);
190 CGPoint* points = element->points;
191
192 switch (element->type) {
193 case kCGPathElementMoveToPoint:
194 if (!CGPathIsEmpty(path)) // to silence a warning when trying to close an empty path
195 CGPathCloseSubpath(path); // This is the only change from CGPathCreateMutableCopy
196 CGPathMoveToPoint(path, 0, points[0].x, points[0].y);
197 break;
198 case kCGPathElementAddLineToPoint:
199 CGPathAddLineToPoint(path, 0, points[0].x, points[0].y);
200 break;
201 case kCGPathElementAddQuadCurveToPoint:
202 CGPathAddQuadCurveToPoint(path, 0, points[0].x, points[0].y, points[1].x, points[1].y);
203 break;
204 case kCGPathElementAddCurveToPoint:
205 CGPathAddCurveToPoint(path, 0, points[0].x, points[0].y, points[1].x, points[1].y, points[2].x, points[2].y);
206 break;
207 case kCGPathElementCloseSubpath:
208 CGPathCloseSubpath(path);
209 break;
210 }
211}

213static RetainPtr<CGMutablePathRef> copyCGPathClosingSubpaths(CGPathRef originalPath)
214{
215 auto path = adoptCF(CGPathCreateMutable());
216 CGPathApply(originalPath, path.get(), copyClosingSubpathsApplierFunction);
217 CGPathCloseSubpath(path.get());
218 return path;
219}

221bool Path::contains(const FloatPoint &point, WindRule rule) const
222{
223 if (isNull())
224 return false;
225
226 if (!fastBoundingRect().contains(point))
227 return false;
228
229 // CGPathContainsPoint returns false for non-closed paths, as a work-around, we copy and close the path first. Radar 4758998 asks for a better CG API to use
230 auto path = copyCGPathClosingSubpaths(platformPath());
231 return CGPathContainsPoint(path.get(), nullptr, point, rule == WindRule::EvenOdd);
232}

234bool Path::strokeContains(const FloatPoint& point, const Function<void(GraphicsContext&)>& strokeStyleApplier) const
235{
236 ASSERT(strokeStyleApplier);
237
238 if (isNull())
239 return false;
240
241 CGContextRef context = scratchContext();
242
243 CGContextSaveGState(context);
244 CGContextBeginPath(context);
245 CGContextAddPath(context, platformPath());
246
247 GraphicsContextCG graphicsContext(context);
248 strokeStyleApplier(graphicsContext);
249
250 bool hitSuccess = CGContextPathContainsPoint(context, point, kCGPathStroke);
251 CGContextRestoreGState(context);
252
253 return hitSuccess;
254}

261void Path::transform(const AffineTransform& transform)
262{
263 if (transform.isIdentity() || isEmpty())
264 return;
265
266 CGAffineTransform transformCG = transform;
267#if PLATFORM(WIN)
268 auto path = adoptCF(CGPathCreateMutable());
269 CGPathAddPath(path.get(), &transformCG, platformPath());
270#else
271 auto path = adoptCF(CGPathCreateMutableCopyByTransformingPath(platformPath(), &transformCG));
272#endif
273 m_path = WTFMove(path);
274 m_copyPathBeforeMutation = false;
275 m_inlineData = Monostate { };
276}

296FloatRect Path::strokeBoundingRect(const Function<void(GraphicsContext&)>& strokeStyleApplier) const
297{
298 if (isNull())
299 return CGRectZero;
300
301 CGContextRef context = scratchContext();
302
303 CGContextSaveGState(context);
304 CGContextBeginPath(context);
305 CGContextAddPath(context, platformPath());
306
307 if (strokeStyleApplier) {
308 GraphicsContextCG graphicsContext(context);
309 strokeStyleApplier(graphicsContext);
310 }
311
312 CGContextReplacePathWithStrokedPath(context);
313 CGRect box = CGContextIsPathEmpty(context) ? CGRectZero : CGContextGetPathBoundingBox(context);
314 CGContextRestoreGState(context);
315
316 return CGRectIsNull(box) ? CGRectZero : box;
317}

344void Path::platformAddPathForRoundedRect(const FloatRect& rect, const FloatSize& topLeftRadius, const FloatSize& topRightRadius, const FloatSize& bottomLeftRadius, const FloatSize& bottomRightRadius)
345{
346#if PLATFORM(COCOA)
347 bool equalWidths = (topLeftRadius.width() == topRightRadius.width() && topRightRadius.width() == bottomLeftRadius.width() && bottomLeftRadius.width() == bottomRightRadius.width());
348 bool equalHeights = (topLeftRadius.height() == bottomLeftRadius.height() && bottomLeftRadius.height() == topRightRadius.height() && topRightRadius.height() == bottomRightRadius.height());
349
350 if (equalWidths && equalHeights) {
351 // Ensure that CG can render the rounded rect.
352 CGFloat radiusWidth = topLeftRadius.width();
353 CGFloat radiusHeight = topLeftRadius.height();
354 CGRect rectToDraw = rect;
355 CGFloat rectWidth = CGRectGetWidth(rectToDraw);
356 CGFloat rectHeight = CGRectGetHeight(rectToDraw);
357 if (2 * radiusWidth > rectWidth)
358 radiusWidth = rectWidth / 2 - std::numeric_limits<CGFloat>::epsilon();
359 if (2 * radiusHeight > rectHeight)
360 radiusHeight = rectHeight / 2 - std::numeric_limits<CGFloat>::epsilon();
361 CGPathAddRoundedRect(ensurePlatformPath(), nullptr, rectToDraw, radiusWidth, radiusHeight);
362 return;
363 }
364
365#if HAVE(CG_PATH_UNEVEN_CORNERS_ROUNDEDRECT)
366 CGRect rectToDraw = rect;
367
368 enum Corners {
369 BottomLeft,
370 BottomRight,
371 TopRight,
372 TopLeft
373 };
374 CGSize corners[4] = { bottomLeftRadius, bottomRightRadius, topRightRadius, topLeftRadius };
375
376 CGFloat rectWidth = CGRectGetWidth(rectToDraw);
377 CGFloat rectHeight = CGRectGetHeight(rectToDraw);
378
379 // Clamp the radii after conversion to CGFloats.
380 corners[TopRight].width = std::min(corners[TopRight].width, rectWidth - corners[TopLeft].width);
381 corners[BottomRight].width = std::min(corners[BottomRight].width, rectWidth - corners[BottomLeft].width);
382 corners[BottomLeft].height = std::min(corners[BottomLeft].height, rectHeight - corners[TopLeft].height);
383 corners[BottomRight].height = std::min(corners[BottomRight].height, rectHeight - corners[TopRight].height);
384
385 CGPathAddUnevenCornersRoundedRect(ensurePlatformPath(), nullptr, rectToDraw, corners);
386 return;
387#endif
388#endif
389
390 addBeziersForRoundedRect(rect, topLeftRadius, topRightRadius, bottomLeftRadius, bottomRightRadius);
391}

426void Path::addPath(const Path& path, const AffineTransform& transform)
427{
428 if (!path.platformPath())
429 return;
430
431 if (!transform.isInvertible())
432 return;
433
434 CGAffineTransform transformCG = transform;
435 // CG doesn't allow adding a path to itself. Optimize for the common case
436 // and copy the path for the self referencing case.
437 if (ensurePlatformPath() != path.platformPath()) {
438 CGPathAddPath(ensurePlatformPath(), &transformCG, path.platformPath());
439 return;
440 }
441 auto pathCopy = adoptCF(CGPathCreateCopy(path.platformPath()));
442 CGPathAddPath(ensurePlatformPath(), &transformCG, pathCopy.get());
443}

465static void CGPathApplierToPathApplier(void* info, const CGPathElement* element)
466{
467 const PathApplierFunction& function = *(PathApplierFunction*)info;
468 PathElement pathElement;
469 pathElement.type = (PathElement::Type)element->type;
470 CGPoint* cgPoints = element->points;
471 switch (element->type) {
472 case kCGPathElementMoveToPoint:
473 case kCGPathElementAddLineToPoint:
474 pathElement.points[0] = cgPoints[0];
475 break;
476 case kCGPathElementAddQuadCurveToPoint:
477 pathElement.points[0] = cgPoints[0];
478 pathElement.points[1] = cgPoints[1];
479 break;
480 case kCGPathElementAddCurveToPoint:
481 pathElement.points[0] = cgPoints[0];
482 pathElement.points[1] = cgPoints[1];
483 pathElement.points[2] = cgPoints[2];
484 break;
485 case kCGPathElementCloseSubpath:
486 break;
487 }
488 function(pathElement);
489}
