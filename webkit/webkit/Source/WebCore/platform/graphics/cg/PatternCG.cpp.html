40static void patternCallback(void* info, CGContextRef context)
41{
42 CGImageRef platformImage = static_cast<CGImageRef>(info);
43 if (!platformImage)
44 return;
45
46 CGRect rect = GraphicsContextCG(context).roundToDevicePixels(
47 FloatRect(0, 0, CGImageGetWidth(platformImage), CGImageGetHeight(platformImage)));
48 CGContextDrawImage(context, rect, platformImage);
49}
50
51static void patternReleaseCallback(void* info)
52{
53 callOnMainThread([image = adoptCF(static_cast<CGImageRef>(info))] { });
54}
55
56RetainPtr<CGPatternRef> Pattern::createPlatformPattern(const AffineTransform& userSpaceTransform) const
57{
58 FloatRect tileRect = { { }, tileImage().size() };
59
60 AffineTransform patternTransform = userSpaceTransform * patternSpaceTransform();
61 patternTransform.scaleNonUniform(1, -1);
62 patternTransform.translate(0, -tileRect.height());
63
64 PlatformImagePtr platformImage = tileImage().platformImage();
65 if (!platformImage)
66 return nullptr;
67
68 // If we're repeating in both directions, we can use image-backed patterns
69 // instead of custom patterns, and avoid tiling-edge pixel cracks.
70 if (repeatX() && repeatY())
71 return adoptCF(CGPatternCreateWithImage2(platformImage.get(), patternTransform, kCGPatternTilingConstantSpacing));
72
73 // If FLT_MAX should also be used for xStep or yStep, nothing is rendered. Using fractions of FLT_MAX also
74 // result in nothing being rendered.
75 // INT_MAX is almost correct, but there seems to be some number wrapping occurring making the fill
76 // pattern is not filled correctly.
77 // To make error of floating point less than 0.5, we use the half of the number of mantissa of float (1 << 22).
78 CGFloat xStep = repeatX() ? tileRect.width() : (1 << 22);
79 CGFloat yStep = repeatY() ? tileRect.height() : (1 << 22);
80
81 // The pattern will release the CGImageRef when it's done rendering in patternReleaseCallback
82 CGImageRef image = platformImage.leakRef();
83
84 const CGPatternCallbacks patternCallbacks = { 0, patternCallback, patternReleaseCallback };
85 return adoptCF(CGPatternCreate(image, tileRect, patternTransform, xStep, yStep, kCGPatternTilingConstantSpacing, TRUE, &patternCallbacks));
86}
