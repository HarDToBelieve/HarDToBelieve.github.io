37SubimageCacheWithTimer* SubimageCacheWithTimer::s_cache;

39static const Seconds subimageCachePruneDelay { 500_ms };
40static const Seconds subimageCacheEntryLifetime { 500_ms };
41static const int maxSubimageCacheSize = 300;
42
43RetainPtr<CGImageRef> SubimageCacheWithTimer::getSubimage(CGImageRef image, const FloatRect& rect)
44{
45 return subimageCache().subimage(image, rect);
46}

48void SubimageCacheWithTimer::clearImage(CGImageRef image)
49{
50 if (subimageCacheExists())
51 subimageCache().clearImageAndSubimages(image);
52}

54void SubimageCacheWithTimer::clear()
55{
56 if (subimageCacheExists())
57 subimageCache().clearAll();

60struct SubimageRequest {
61 CGImageRef image;
62 const FloatRect& rect;
63 SubimageRequest(CGImageRef image, const FloatRect& rect) : image(image), rect(rect) { }
64};
65
66struct SubimageCacheAdder {
67 static unsigned hash(const SubimageRequest& value)
68 {
69 return SubimageCacheWithTimer::SubimageCacheHash::hash(value.image, value.rect);
70 }
71
72 static bool equal(const SubimageCacheWithTimer::SubimageCacheEntry& a, const SubimageRequest& b)
73 {
74 return a.image == b.image && a.rect == b.rect;
75 }
76
77 static void translate(SubimageCacheWithTimer::SubimageCacheEntry& entry, const SubimageRequest& request, unsigned /*hashCode*/)
78 {
79 entry.image = request.image;
80 entry.subimage = adoptCF(CGImageCreateWithImageInRect(request.image, request.rect));
81 entry.rect = request.rect;
82 }
83};

85SubimageCacheWithTimer::SubimageCacheWithTimer()
86 : m_timer(RunLoop::main(), this, &SubimageCacheWithTimer::pruneCacheTimerFired)
87{
88}
89
90void SubimageCacheWithTimer::pruneCacheTimerFired()
91{
92 Locker locker { m_lock };
93 prune();
94 if (m_cache.isEmpty()) {
95 ASSERT(m_imageCounts.isEmpty());
96 m_timer.stop();
97 }
98}
99
100void SubimageCacheWithTimer::prune()
101{
102 auto now = MonotonicTime::now();
103
104 Vector<SubimageCacheEntry> toBeRemoved;
105
106 for (const auto& entry : m_cache) {
107 if ((now - entry.lastAccessTime) > subimageCacheEntryLifetime)
108 toBeRemoved.append(entry);
109 }
110
111 for (auto& entry : toBeRemoved) {
112 m_imageCounts.remove(entry.image.get());
113 m_cache.remove(entry);
114 }
115}
116
117RetainPtr<CGImageRef> SubimageCacheWithTimer::subimage(CGImageRef image, const FloatRect& rect)
118{
119 Locker locker { m_lock };
120 if (!m_timer.isActive())
121 m_timer.startRepeating(subimageCachePruneDelay);
122
123 if (m_cache.size() == maxSubimageCacheSize) {
124 SubimageCacheEntry entry = *m_cache.begin();
125 m_imageCounts.remove(entry.image.get());
126 m_cache.remove(entry);
127 }
128
129 ASSERT(m_cache.size() < maxSubimageCacheSize);
130 auto result = m_cache.add<SubimageCacheAdder>(SubimageRequest(image, rect));
131 if (result.isNewEntry)
132 m_imageCounts.add(image);
133
134 result.iterator->lastAccessTime = MonotonicTime::now();
135 return result.iterator->subimage;
136}
137
138void SubimageCacheWithTimer::clearImageAndSubimages(CGImageRef image)
139{
140 Locker locker { m_lock };
141 if (m_imageCounts.contains(image)) {
142 Vector<SubimageCacheEntry> toBeRemoved;
143 for (const auto& entry : m_cache) {
144 if (entry.image.get() == image)
145 toBeRemoved.append(entry);
146 }
147
148 for (auto& entry : toBeRemoved)
149 m_cache.remove(entry);
150
151 m_imageCounts.removeAll(image);
152 }
153}
154
155void SubimageCacheWithTimer::clearAll()
156{
157 Locker locker { m_lock };
158 m_imageCounts.clear();
159 m_cache.clear();
160}

162SubimageCacheWithTimer& SubimageCacheWithTimer::subimageCache()
163{
164 static std::once_flag onceKey;
165 std::call_once(onceKey, [&] {
166 s_cache = new SubimageCacheWithTimer;
167 });
168 return *s_cache;
169}
170
171bool SubimageCacheWithTimer::subimageCacheExists()
172{
173 return !!s_cache;
174}
