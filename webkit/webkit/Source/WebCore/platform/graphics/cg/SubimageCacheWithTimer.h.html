41namespace WebCore {
42
43#if CACHE_SUBIMAGES
44
45class SubimageCacheWithTimer {
46 WTF_MAKE_NONCOPYABLE(SubimageCacheWithTimer); WTF_MAKE_FAST_ALLOCATED;
47
48public:
49 struct SubimageCacheEntry {
50 RetainPtr<CGImageRef> image;
51 RetainPtr<CGImageRef> subimage;
52 FloatRect rect;
53 MonotonicTime lastAccessTime;
54 };
55
56 struct SubimageCacheEntryTraits : HashTraits<SubimageCacheEntry> {
57 typedef HashTraits<RetainPtr<CGImageRef>> ImageTraits;
58
59 static const bool emptyValueIsZero = true;
60
61 static const bool hasIsEmptyValueFunction = true;
62 static bool isEmptyValue(const SubimageCacheEntry& value) { return !value.image; }
63
64 static void constructDeletedValue(SubimageCacheEntry& slot) { ImageTraits::constructDeletedValue(slot.image); }
65 static bool isDeletedValue(const SubimageCacheEntry& value) { return ImageTraits::isDeletedValue(value.image); }
66 };
67
68 struct SubimageCacheHash {
69 static unsigned hash(CGImageRef image, const FloatRect& rect)
70 {
71 return WTF::pairIntHash(PtrHash<CGImageRef>::hash(image),
72 (static_cast<unsigned>(rect.x()) << 16) | static_cast<unsigned>(rect.y()));
73 }
74 static unsigned hash(const SubimageCacheEntry& key)
75 {
76 return hash(key.image.get(), key.rect);
77 }
78 static bool equal(const SubimageCacheEntry& a, const SubimageCacheEntry& b)
79 {
80 return a.image == b.image && a.rect == b.rect;
81 }
82 static const bool safeToCompareToEmptyOrDeleted = true;
83 };
84
85 static RetainPtr<CGImageRef> getSubimage(CGImageRef, const FloatRect&);
86 static void clearImage(CGImageRef);
87 static void clear();
88
89private:
90 typedef HashSet<SubimageCacheEntry, SubimageCacheHash, SubimageCacheEntryTraits> SubimageCacheHashSet;
91
92 SubimageCacheWithTimer();
93 void pruneCacheTimerFired();
94
95 RetainPtr<CGImageRef> subimage(CGImageRef, const FloatRect&);
96 void clearImageAndSubimages(CGImageRef);
97 void prune() WTF_REQUIRES_LOCK(m_lock);
98 void clearAll();
99
100 Lock m_lock;
101 HashCountedSet<CGImageRef> m_imageCounts WTF_GUARDED_BY_LOCK(m_lock);
102 SubimageCacheHashSet m_cache WTF_GUARDED_BY_LOCK(m_lock);
103 RunLoop::Timer<SubimageCacheWithTimer> m_timer WTF_GUARDED_BY_LOCK(m_lock);
104
105 static SubimageCacheWithTimer& subimageCache();
106 static bool subimageCacheExists();
107 static SubimageCacheWithTimer* s_cache;
108};

110#endif // CACHE_SUBIMAGES
