41const MemoryCompactLookupOnlyRobinHoodHashSet<String>& defaultSupportedImageTypes()
42{
43 static const auto defaultSupportedImageTypes = makeNeverDestroyed([] {
44 HashSet<String> defaultSupportedImageTypes = {
45 "com.compuserve.gif"_s,
46 "com.microsoft.bmp"_s,
47 "com.microsoft.cur"_s,
48 "com.microsoft.ico"_s,
49 "public.jpeg"_s,
50 "public.png"_s,
51 "public.tiff"_s,
52#if !PLATFORM(WIN)
53 "public.jpeg-2000"_s,
54 "public.mpo-image"_s,
55#endif
56#if HAVE(WEBP)
57 "public.webp"_s,
58 "com.google.webp"_s,
59 "org.webmproject.webp"_s,
60#endif
61 };
62
63 auto systemSupportedCFImageTypes = adoptCF(CGImageSourceCopyTypeIdentifiers());
64 CFIndex count = CFArrayGetCount(systemSupportedCFImageTypes.get());
65
66 HashSet<String> systemSupportedImageTypes;
67 CFArrayApplyFunction(systemSupportedCFImageTypes.get(), CFRangeMake(0, count), [](const void *value, void *context) {
68 String imageType = static_cast<CFStringRef>(value);
69 static_cast<HashSet<String>*>(context)->add(imageType);
70 }, &systemSupportedImageTypes);
71
72 MemoryCompactLookupOnlyRobinHoodHashSet<String> filtered;
73 for (auto& imageType : defaultSupportedImageTypes) {
74 if (systemSupportedImageTypes.contains(imageType))
75 filtered.add(imageType);
76 }
77 return filtered;
78 }());
79
80 return defaultSupportedImageTypes;
81}

83HashSet<String>& additionalSupportedImageTypes()
84{
85 static NeverDestroyed<HashSet<String>> additionalSupportedImageTypes;
86 return additionalSupportedImageTypes;
87}
88
89void setAdditionalSupportedImageTypes(const Vector<String>& imageTypes)
90{
91 MIMETypeRegistry::additionalSupportedImageMIMETypes().clear();
92 for (const auto& imageType : imageTypes) {
93 additionalSupportedImageTypes().add(imageType);
94 auto mimeType = MIMETypeForImageType(imageType);
95 if (!mimeType.isEmpty())
96 MIMETypeRegistry::additionalSupportedImageMIMETypes().add(mimeType);
97 }
98}
99
100void setAdditionalSupportedImageTypesForTesting(const String& imageTypes)
101{
102 setAdditionalSupportedImageTypes(imageTypes.split(';'));
103}
104
105bool isSupportedImageType(const String& imageType)
106{
107 if (imageType.isEmpty())
108 return false;
109 return defaultSupportedImageTypes().contains(imageType) || additionalSupportedImageTypes().contains(imageType);
110}
111
112bool isGIFImageType(StringView imageType)
113{
114 return imageType == "com.compuserve.gif";
115}
