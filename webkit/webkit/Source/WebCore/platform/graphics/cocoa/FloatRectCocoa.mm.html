36id makeNSArrayElement(const WebCore::FloatRect& rect)
37{
38#if PLATFORM(MAC)
39 return [NSValue valueWithRect:rect];
40#else
41 return [NSValue valueWithCGRect:rect];
42#endif
43}

47FloatRect::FloatRect(const NSRect& rect)
48 : m_location(rect.origin)
49 , m_size(rect.size)
50{
51}

53FloatRect::operator NSRect() const
54{
55 return NSMakeRect(x(), y(), width(), height());
56}
