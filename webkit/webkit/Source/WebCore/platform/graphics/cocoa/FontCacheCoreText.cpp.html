50static inline void appendTrueTypeFeature(CFMutableArrayRef features, int type, int selector)
51{
52 auto typeNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &type));
53 auto selectorNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &selector));
54 CFTypeRef featureKeys[] = { kCTFontFeatureTypeIdentifierKey, kCTFontFeatureSelectorIdentifierKey };
55 CFTypeRef featureValues[] = { typeNumber.get(), selectorNumber.get() };
56 auto feature = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, featureKeys, featureValues, WTF_ARRAY_LENGTH(featureKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
57 CFArrayAppendValue(features, feature.get());
58}

65static inline void appendOpenTypeFeature(CFMutableArrayRef features, const FontFeature& feature)
66{
67 auto featureKey = adoptCF(CFStringCreateWithBytes(kCFAllocatorDefault, reinterpret_cast<const UInt8*>(feature.tag().data()), feature.tag().size() * sizeof(FontTag::value_type), kCFStringEncodingASCII, false));
68 int rawFeatureValue = feature.value();
69 auto featureValue = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &rawFeatureValue));
70 CFTypeRef featureDictionaryKeys[] = { kCTFontOpenTypeFeatureTag, kCTFontOpenTypeFeatureValue };
71 CFTypeRef featureDictionaryValues[] = { featureKey.get(), featureValue.get() };
72 auto featureDictionary = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, featureDictionaryKeys, featureDictionaryValues, WTF_ARRAY_LENGTH(featureDictionaryValues), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
73 CFArrayAppendValue(features, featureDictionary.get());
74}

300VariationDefaultsMap defaultVariationValues(CTFontRef font)
301{
302 VariationDefaultsMap result;
303 auto axes = adoptCF(CTFontCopyVariationAxes(font));
304 if (!axes)
305 return result;
306 auto size = CFArrayGetCount(axes.get());
307 for (CFIndex i = 0; i < size; ++i) {
308 CFDictionaryRef axis = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(axes.get(), i));
309 CFNumberRef axisIdentifier = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisIdentifierKey));
310 String axisName = static_cast<CFStringRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisNameKey));
311 CFNumberRef defaultValue = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisDefaultValueKey));
312 CFNumberRef minimumValue = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisMinimumValueKey));
313 CFNumberRef maximumValue = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisMaximumValueKey));
314 uint32_t rawAxisIdentifier = 0;
315 Boolean success = CFNumberGetValue(axisIdentifier, kCFNumberSInt32Type, &rawAxisIdentifier);
316 ASSERT_UNUSED(success, success);
317 float rawDefaultValue = 0;
318 float rawMinimumValue = 0;
319 float rawMaximumValue = 0;
320 CFNumberGetValue(defaultValue, kCFNumberFloatType, &rawDefaultValue);
321 CFNumberGetValue(minimumValue, kCFNumberFloatType, &rawMinimumValue);
322 CFNumberGetValue(maximumValue, kCFNumberFloatType, &rawMaximumValue);
323
324 if (rawMinimumValue > rawMaximumValue)
325 std::swap(rawMinimumValue, rawMaximumValue);
326
327 auto b1 = rawAxisIdentifier >> 24;
328 auto b2 = (rawAxisIdentifier & 0xFF0000) >> 16;
329 auto b3 = (rawAxisIdentifier & 0xFF00) >> 8;
330 auto b4 = rawAxisIdentifier & 0xFF;
331 FontTag resultKey = {{ static_cast<char>(b1), static_cast<char>(b2), static_cast<char>(b3), static_cast<char>(b4) }};
332 VariationDefaults resultValues = { axisName, rawDefaultValue, rawMinimumValue, rawMaximumValue };
333 result.set(resultKey, resultValues);
334 }
335 return result;
336}

347// These values were calculated by performing a linear regression on the CSS weights/widths/slopes and Core Text weights/widths/slopes of San Francisco.
348// FIXME: <rdar://problem/31312602> Get the real values from Core Text.
349static inline float normalizeWeight(float value)
350{
351 return 523.7 * value - 109.3;
352}
353
354static inline float normalizeSlope(float value)
355{
356 return value * 300;
357}
358
359static inline float denormalizeWeight(float value)
360{
361 return (value + 109.3) / 523.7;
362}
363
364static inline float denormalizeSlope(float value)
365{
366 return value / 300;
367}
368
369static inline float denormalizeVariationWidth(float value)
370{
371 if (value <= 125)
372 return value / 100;
373 if (value <= 150)
374 return (value + 125) / 200;
375 return (value + 400) / 400;
376}
377
378static inline float normalizeVariationWidth(float value)
379{
380 if (value <= 1.25)
381 return value * 100;
382 if (value <= 1.375)
383 return value * 200 - 125;
384 return value * 400 - 400;
385}

394struct FontType {
395 FontType(CTFontRef font)
396 {
397 bool foundStat = false;
398 bool foundTrak = false;
399 auto tables = adoptCF(CTFontCopyAvailableTables(font, kCTFontTableOptionNoOptions));
400 if (!tables)
401 return;
402 auto size = CFArrayGetCount(tables.get());
403 for (CFIndex i = 0; i < size; ++i) {
404 auto tableTag = static_cast<CTFontTableTag>(reinterpret_cast<uintptr_t>(CFArrayGetValueAtIndex(tables.get(), i)));
405 switch (tableTag) {
406 case kCTFontTableFvar:
407 if (variationType == VariationType::NotVariable)
408 variationType = VariationType::TrueTypeGX;
409 break;
410 case kCTFontTableSTAT:
411 foundStat = true;
412 variationType = VariationType::OpenType18;
413 break;
414 case kCTFontTableMorx:
415 case kCTFontTableMort:
416 aatShaping = true;
417 break;
418 case kCTFontTableGPOS:
419 case kCTFontTableGSUB:
420 openTypeShaping = true;
421 break;
422 case kCTFontTableTrak:
423 foundTrak = true;
424 break;
425 }
426 }
427 if (foundStat && foundTrak)
428 trackingType = TrackingType::Automatic;
429 else if (foundTrak)
430 trackingType = TrackingType::Manual;
431 }
432
433 enum class VariationType : uint8_t { NotVariable, TrueTypeGX, OpenType18, };
434 VariationType variationType { VariationType::NotVariable };
435 enum class TrackingType : uint8_t { None, Automatic, Manual, };
436 TrackingType trackingType { TrackingType::None };
437 bool openTypeShaping { false };
438 bool aatShaping { false };
439};

441RetainPtr<CTFontRef> preparePlatformFont(CTFontRef originalFont, const FontDescription& fontDescription, const FontFeatureSettings* fontFaceFeatures, FontSelectionSpecifiedCapabilities fontFaceCapabilities, bool applyWeightWidthSlopeVariations)
442{
443 if (!originalFont)
444 return originalFont;
445
446 FontType fontType { originalFont };
447
448 auto fontOpticalSizing = fontDescription.opticalSizing();
449
450 auto defaultValues = defaultVariationValues(originalFont);
451
452 auto fontSelectionRequest = fontDescription.fontSelectionRequest();
453 auto fontStyleAxis = fontDescription.fontStyleAxis();
454
455 bool forceOpticalSizingOn = fontOpticalSizing == FontOpticalSizing::Enabled && fontType.variationType == FontType::VariationType::TrueTypeGX && defaultValues.contains({{'o', 'p', 's', 'z'}});
456 bool forceVariations = defaultValues.contains({{'w', 'g', 'h', 't'}}) || defaultValues.contains({{'w', 'd', 't', 'h'}}) || (fontStyleAxis == FontStyleAxis::ital && defaultValues.contains({{'i', 't', 'a', 'l'}})) || (fontStyleAxis == FontStyleAxis::slnt && defaultValues.contains({{'s', 'l', 'n', 't'}}));
457 const auto& variations = fontDescription.variationSettings();
458
459 const auto& features = fontDescription.featureSettings();
460 const auto& variantSettings = fontDescription.variantSettings();
461 auto textRenderingMode = fontDescription.textRenderingMode();
462 auto shouldDisableLigaturesForSpacing = fontDescription.shouldDisableLigaturesForSpacing();
463
464 // We might want to check fontType.trackingType == FontType::TrackingType::Manual here, but in order to maintain compatibility with the rest of the system, we don't.
465 bool noFontFeatureSettings = features.isEmpty();
466 bool noFontVariationSettings = !forceVariations && variations.isEmpty();
467 bool textRenderingModeIsAuto = textRenderingMode == TextRenderingMode::AutoTextRendering;
468 bool variantSettingsIsNormal = variantSettings.isAllNormal();
469 bool dontNeedToApplyOpticalSizing = fontOpticalSizing == FontOpticalSizing::Enabled && !forceOpticalSizingOn;
470 bool fontFaceDoesntSpecifyFeatures = !fontFaceFeatures || fontFaceFeatures->isEmpty();
471 if (noFontFeatureSettings && noFontVariationSettings && textRenderingModeIsAuto && variantSettingsIsNormal && dontNeedToApplyOpticalSizing && fontFaceDoesntSpecifyFeatures && !shouldDisableLigaturesForSpacing) {
472#if HAVE(CTFONTCREATEFORCHARACTERSWITHLANGUAGEANDOPTION)
473 return originalFont;
474#else
475 return createFontForInstalledFonts(originalFont, fontDescription.shouldAllowUserInstalledFonts());
476#endif
477 }
478
479 // This algorithm is described at https://drafts.csswg.org/css-fonts-4/#feature-variation-precedence
480 FeaturesMap featuresToBeApplied;
481 VariationsMap variationsToBeApplied;
482
483 bool needsConversion = fontType.variationType == FontType::VariationType::TrueTypeGX;
484
485 auto applyVariation = [&](const FontTag& tag, float value) {
486 auto iterator = defaultValues.find(tag);
487 if (iterator == defaultValues.end())
488 return;
489 float valueToApply = clampTo(value, iterator->value.minimumValue, iterator->value.maximumValue);
490 variationsToBeApplied.set(tag, valueToApply);
491 };
492
493 auto applyFeature = [&](const FontTag& tag, int value) {
494 // AAT doesn't differentiate between liga and clig. We need to make sure they always agree.
495 featuresToBeApplied.set(tag, value);
496 if (fontType.aatShaping) {
497 if (tag == fontFeatureTag("liga"))
498 featuresToBeApplied.set(fontFeatureTag("clig"), value);
499 else if (tag == fontFeatureTag("clig"))
500 featuresToBeApplied.set(fontFeatureTag("liga"), value);
501 }
502 };
503
504 // Step 1: CoreText handles default features (such as required ligatures).
505
506 // Step 2: font-weight, font-stretch, and font-style
507 // The system font is somewhat magical. Don't mess with its variations.
508 if (applyWeightWidthSlopeVariations && !fontIsSystemFont(originalFont)) {
509 float weight = fontSelectionRequest.weight;
510 float width = fontSelectionRequest.width;
511 float slope = fontSelectionRequest.slope.value_or(normalItalicValue());
512 if (auto weightValue = fontFaceCapabilities.weight)
513 weight = std::max(std::min(weight, static_cast<float>(weightValue->maximum)), static_cast<float>(weightValue->minimum));
514 if (auto widthValue = fontFaceCapabilities.width)
515 width = std::max(std::min(width, static_cast<float>(widthValue->maximum)), static_cast<float>(widthValue->minimum));
516 if (auto slopeValue = fontFaceCapabilities.weight)
517 slope = std::max(std::min(slope, static_cast<float>(slopeValue->maximum)), static_cast<float>(slopeValue->minimum));
518 if (needsConversion) {
519 weight = denormalizeWeight(weight);
520 width = denormalizeVariationWidth(width);
521 slope = denormalizeSlope(slope);
522 }
523 applyVariation({{'w', 'g', 'h', 't'}}, weight);
524 applyVariation({{'w', 'd', 't', 'h'}}, width);
525 if (fontStyleAxis == FontStyleAxis::ital)
526 applyVariation({{'i', 't', 'a', 'l'}}, 1);
527 else
528 applyVariation({{'s', 'l', 'n', 't'}}, slope);
529 }
530
531 // FIXME: Implement Step 5: font-named-instance
532
533 // FIXME: Implement Step 6: the font-variation-settings descriptor inside @font-face
534
535 // Step 7: Consult with font-feature-settings inside @font-face
536 if (fontFaceFeatures) {
537 for (auto& fontFaceFeature : *fontFaceFeatures)
538 applyFeature(fontFaceFeature.tag(), fontFaceFeature.value());
539 }
540
541 // FIXME: Move font-optical-sizing handling here. It should be step 9.
542
543 // Step 10: Font-variant
544 for (auto& newFeature : computeFeatureSettingsFromVariants(variantSettings))
545 applyFeature(newFeature.key, newFeature.value);
546
547 // Step 11: Other properties
548 if (textRenderingMode == TextRenderingMode::OptimizeSpeed) {
549 applyFeature(fontFeatureTag("liga"), 0);
550 applyFeature(fontFeatureTag("clig"), 0);
551 applyFeature(fontFeatureTag("dlig"), 0);
552 applyFeature(fontFeatureTag("hlig"), 0);
553 applyFeature(fontFeatureTag("calt"), 0);
554 }
555 if (shouldDisableLigaturesForSpacing) {
556 applyFeature(fontFeatureTag("liga"), 0);
557 applyFeature(fontFeatureTag("clig"), 0);
558 applyFeature(fontFeatureTag("dlig"), 0);
559 applyFeature(fontFeatureTag("hlig"), 0);
560 // Core Text doesn't disable calt when letter-spacing is applied, so we won't either.
561 }
562
563 // Step 13: Font-feature-settings
564 for (auto& newFeature : features)
565 applyFeature(newFeature.tag(), newFeature.value());
566
567 // Step 12: font-variation-settings
568 for (auto& newVariation : variations)
569 applyVariation(newVariation.tag(), newVariation.value());
570
571 auto attributes = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 2, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
572 if (!featuresToBeApplied.isEmpty()) {
573 auto featureArray = adoptCF(CFArrayCreateMutable(kCFAllocatorDefault, features.size(), &kCFTypeArrayCallBacks));
574 for (auto& p : featuresToBeApplied) {
575 auto feature = FontFeature(p.key, p.value);
576
577 // CoreText does not map hlig and hist for TrueType fonts.
578 if (fontType.aatShaping && (tagEquals(feature.tag(), "hlig") || tagEquals(feature.tag(), "hist"))) {
579 if (feature.enabled())
580 appendTrueTypeFeature(featureArray.get(), kLigaturesType, kHistoricalLigaturesOnSelector);
581 else if (tagEquals(feature.tag(), "hlig"))
582 appendTrueTypeFeature(featureArray.get(), kLigaturesType, kHistoricalLigaturesOffSelector);
583 continue;
584 }
585
586 appendOpenTypeFeature(featureArray.get(), feature);
587 }
588 CFDictionaryAddValue(attributes.get(), kCTFontFeatureSettingsAttribute, featureArray.get());
589 }
590 if (!variationsToBeApplied.isEmpty()) {
591 auto variationDictionary = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
592 for (auto& p : variationsToBeApplied) {
593 long long bitwiseTag = p.key[0] << 24 | p.key[1] << 16 | p.key[2] << 8 | p.key[3];
594 auto tagNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberLongLongType, &bitwiseTag));
595 auto valueNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberFloatType, &p.value));
596 CFDictionarySetValue(variationDictionary.get(), tagNumber.get(), valueNumber.get());
597 }
598 CFDictionaryAddValue(attributes.get(), kCTFontVariationAttribute, variationDictionary.get());
599 }
600
601 // Step 9: font-optical-sizing
602 // FIXME: Apply this before font-variation-settings
603 if (forceOpticalSizingOn || textRenderingMode == TextRenderingMode::OptimizeLegibility) {
604#if HAVE(CORETEXT_AUTO_OPTICAL_SIZING)
605 CFDictionaryAddValue(attributes.get(), kCTFontOpticalSizeAttribute, CFSTR("auto"));
606#else
607 auto size = CTFontGetSize(originalFont);
608 auto sizeNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberCGFloatType, &size));
609 CFDictionaryAddValue(attributes.get(), kCTFontOpticalSizeAttribute, sizeNumber.get());
610#endif
611 } else if (fontOpticalSizing == FontOpticalSizing::Disabled) {
612#if HAVE(CORETEXT_AUTO_OPTICAL_SIZING)
613 CFDictionaryAddValue(attributes.get(), kCTFontOpticalSizeAttribute, CFSTR("none"));
614#endif
615 }
616
617 addAttributesForInstalledFonts(attributes.get(), fontDescription.shouldAllowUserInstalledFonts());
618
619 auto descriptor = adoptCF(CTFontDescriptorCreateWithAttributes(attributes.get()));
620 return adoptCF(CTFontCreateCopyWithAttributes(originalFont, CTFontGetSize(originalFont), nullptr, descriptor.get()));
621}

623RefPtr<Font> FontCache::similarFont(const FontDescription& description, const String& family)
624{
625 // Attempt to find an appropriate font using a match based on the presence of keywords in
626 // the requested names. For example, we'll match any name that contains "Arabic" to Geeza Pro.
627 if (family.isEmpty())
628 return nullptr;
629
630#if PLATFORM(IOS_FAMILY)
631 // Substitute the default monospace font for well-known monospace fonts.
632 if (equalLettersIgnoringASCIICase(family, "monaco") || equalLettersIgnoringASCIICase(family, "menlo"))
633 return fontForFamily(description, "courier"_s);
634
635 // Substitute Verdana for Lucida Grande.

932static std::optional<MinMax> extractVariationBounds(CFDictionaryRef axis)
933{
934 CFNumberRef minimumValue = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisMinimumValueKey));
935 CFNumberRef maximumValue = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisMaximumValueKey));
936 float rawMinimumValue = 0;
937 float rawMaximumValue = 0;
938 CFNumberGetValue(minimumValue, kCFNumberFloatType, &rawMinimumValue);
939 CFNumberGetValue(maximumValue, kCFNumberFloatType, &rawMaximumValue);
940 if (rawMinimumValue < rawMaximumValue)
941 return {{ rawMinimumValue, rawMaximumValue }};
942 return std::nullopt;
943}

945static VariationCapabilities variationCapabilitiesForFontDescriptor(CTFontDescriptorRef fontDescriptor)
946{
947 VariationCapabilities result;
948
949 if (!adoptCF(CTFontDescriptorCopyAttribute(fontDescriptor, kCTFontVariationAttribute)))
950 return result;
951
952 auto font = adoptCF(CTFontCreateWithFontDescriptor(fontDescriptor, 0, nullptr));
953 auto variations = adoptCF(CTFontCopyVariationAxes(font.get()));
954 if (!variations)
955 return result;
956
957 auto axisCount = CFArrayGetCount(variations.get());
958 if (!axisCount)
959 return result;
960
961 for (CFIndex i = 0; i < axisCount; ++i) {
962 CFDictionaryRef axis = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(variations.get(), i));
963 CFNumberRef axisIdentifier = static_cast<CFNumberRef>(CFDictionaryGetValue(axis, kCTFontVariationAxisIdentifierKey));
964 uint32_t rawAxisIdentifier = 0;
965 Boolean success = CFNumberGetValue(axisIdentifier, kCFNumberSInt32Type, &rawAxisIdentifier);
966 ASSERT_UNUSED(success, success);
967 if (rawAxisIdentifier == 0x77676874) // 'wght'
968 result.weight = extractVariationBounds(axis);
969 else if (rawAxisIdentifier == 0x77647468) // 'wdth'
970 result.width = extractVariationBounds(axis);
971 else if (rawAxisIdentifier == 0x736C6E74) // 'slnt'
972 result.slope = extractVariationBounds(axis);
973 }
974
975 bool optOutFromGXNormalization = false;
976// FIXME: Likely we can remove this special case for watchOS and tvOS.
977#if !PLATFORM(WATCHOS) && !PLATFORM(APPLETV)
978 optOutFromGXNormalization = CTFontDescriptorIsSystemUIFont(fontDescriptor);
979#endif
980
981 if (FontType(font.get()).variationType == FontType::VariationType::TrueTypeGX && !optOutFromGXNormalization) {
982 if (result.weight)
983 result.weight = {{ normalizeWeight(result.weight.value().minimum), normalizeWeight(result.weight.value().maximum) }};
984 if (result.width)
985 result.width = {{ normalizeVariationWidth(result.width.value().minimum), normalizeVariationWidth(result.width.value().maximum) }};
986 if (result.slope)
987 result.slope = {{ normalizeSlope(result.slope.value().minimum), normalizeSlope(result.slope.value().maximum) }};
988 }
989
990 auto minimum = static_cast<float>(FontSelectionValue::minimumValue());
991 auto maximum = static_cast<float>(FontSelectionValue::maximumValue());
992 if (result.weight && (result.weight.value().minimum < minimum || result.weight.value().maximum > maximum))
993 result.weight = { };
994 if (result.width && (result.width.value().minimum < minimum || result.width.value().maximum > maximum))
995 result.width = { };
996 if (result.slope && (result.slope.value().minimum < minimum || result.slope.value().maximum > maximum))
997 result.slope = { };
998
999 return result;
1000}

1013FontSelectionCapabilities capabilitiesForFontDescriptor(CTFontDescriptorRef fontDescriptor)
1014{
1015 if (!fontDescriptor)
1016 return { };
1017
1018 VariationCapabilities variationCapabilities = variationCapabilitiesForFontDescriptor(fontDescriptor);
1019
1020#if !HAS_CORE_TEXT_WIDTH_ATTRIBUTE
1021 bool weightOrWidthComeFromTraits = !variationCapabilities.weight || !variationCapabilities.width;
1022#else
1023 bool weightOrWidthComeFromTraits = false;
1024#endif
1025
1026 if (!variationCapabilities.slope || weightOrWidthComeFromTraits) {
1027 auto traits = adoptCF(static_cast<CFDictionaryRef>(CTFontDescriptorCopyAttribute(fontDescriptor, kCTFontTraitsAttribute)));
1028 if (traits) {
1029#if !HAS_CORE_TEXT_WIDTH_ATTRIBUTE
1030 if (!variationCapabilities.width) {
1031 auto widthValue = stretchFromCoreTextTraits(traits.get());
1032 variationCapabilities.width = {{ widthValue, widthValue }};
1033 }
1034#endif
1035
1036 if (!variationCapabilities.slope) {
1037 auto symbolicTraitsNumber = static_cast<CFNumberRef>(CFDictionaryGetValue(traits.get(), kCTFontSymbolicTrait));
1038 if (symbolicTraitsNumber) {
1039 int32_t symbolicTraits;
1040 auto success = CFNumberGetValue(symbolicTraitsNumber, kCFNumberSInt32Type, &symbolicTraits);
1041 ASSERT_UNUSED(success, success);
1042 auto slopeValue = static_cast<float>(symbolicTraits & kCTFontTraitItalic ? italicValue() : normalItalicValue());
1043 variationCapabilities.slope = {{ slopeValue, slopeValue }};
1044 } else
1045 variationCapabilities.slope = {{ static_cast<float>(normalItalicValue()), static_cast<float>(normalItalicValue()) }};
1046 }
1047 }
1048 }
1049
1050 if (!variationCapabilities.weight) {
1051 auto value = getCSSAttribute(fontDescriptor, kCTFontCSSWeightAttribute, static_cast<float>(normalWeightValue()));
1052 variationCapabilities.weight = {{ value, value }};
1053 }
1054
1055#if HAS_CORE_TEXT_WIDTH_ATTRIBUTE
1056 if (!variationCapabilities.width) {
1057 auto value = getCSSAttribute(fontDescriptor, kCTFontCSSWidthAttribute, static_cast<float>(normalStretchValue()));
1058 variationCapabilities.width = {{ value, value }};
1059 }
1060#endif
1061
1062 FontSelectionCapabilities result = {{ FontSelectionValue(variationCapabilities.weight.value().minimum), FontSelectionValue(variationCapabilities.weight.value().maximum) },
1063 { FontSelectionValue(variationCapabilities.width.value().minimum), FontSelectionValue(variationCapabilities.width.value().maximum) },
1064 { FontSelectionValue(variationCapabilities.slope.value().minimum), FontSelectionValue(variationCapabilities.slope.value().maximum) }};
1065 ASSERT(result.weight.isValid());
1066 ASSERT(result.width.isValid());
1067 ASSERT(result.slope.isValid());
1068 return result;
1069}

1104static FontLookup platformFontLookupWithFamily(const AtomString& family, FontSelectionRequest request, float size, AllowUserInstalledFonts allowUserInstalledFonts)
1105{
1106 const auto& allowlist = fontAllowlist();
1107 if (!isSystemFont(family.string()) && allowlist.size() && !allowlist.contains(family))
1108 return { nullptr };
1109
1110 if (equalLettersIgnoringASCIICase(family, ".applesystemuifontserif")
1111 || equalLettersIgnoringASCIICase(family, ".sf ns mono")
1112 || equalLettersIgnoringASCIICase(family, ".sf ui mono")
1113 || equalLettersIgnoringASCIICase(family, ".applesystemuifontrounded")) {
1114 // If you want to use these fonts, use ui-serif, ui-monospace, and ui-rounded.
1115 return { nullptr };
1116 }
1117
1118 auto& fontDatabase = allowUserInstalledFonts == AllowUserInstalledFonts::Yes ? FontDatabase::singletonAllowingUserInstalledFonts() : FontDatabase::singletonDisallowingUserInstalledFonts();
1119 const auto& familyFonts = fontDatabase.collectionForFamily(family.string());
1120 if (familyFonts.isEmpty()) {
1121 // The CSS spec states that font-family only accepts a name of an actual font family. However, in WebKit, we claim to also
1122 // support supplying a PostScript name instead. However, this creates problems when the other properties (font-weight,
1123 // font-style) disagree with the traits of the PostScript-named font. The solution we have come up with is, when the default
1124 // values for font-weight and font-style are supplied, honor the PostScript name, but if font-weight specifies bold or
1125 // font-style specifies italic, then we run the regular matching algorithm on the family of the PostScript font. This way,
1126 // if content simply states "font-family: PostScriptName;" without specifying the other font properties, it will be honored,
1127 // but if a <b> appears as a descendent element, it will be honored too.
1128 const auto& postScriptFont = fontDatabase.fontForPostScriptName(family);
1129 if (!postScriptFont.fontDescriptor)
1130 return { nullptr };
1131 if ((isItalic(request.slope) && !isItalic(postScriptFont.capabilities.slope.maximum))
1132 || (isFontWeightBold(request.weight) && !isFontWeightBold(postScriptFont.capabilities.weight.maximum))) {
1133 auto postScriptFamilyName = adoptCF(static_cast<CFStringRef>(CTFontDescriptorCopyAttribute(postScriptFont.fontDescriptor.get(), kCTFontFamilyNameAttribute)));
1134 if (!postScriptFamilyName)
1135 return { nullptr };
1136 const auto& familyFonts = fontDatabase.collectionForFamily(String(postScriptFamilyName.get()));
1137 if (familyFonts.isEmpty())
1138 return { nullptr };
1139 if (const auto* installedFont = findClosestFont(familyFonts, request)) {
1140 if (!installedFont->fontDescriptor)
1141 return { nullptr };
1142 return { adoptCF(CTFontCreateWithFontDescriptor(installedFont->fontDescriptor.get(), size, nullptr)), true };
1143 }
1144 return { nullptr };
1145 }
1146 return { adoptCF(CTFontCreateWithFontDescriptor(postScriptFont.fontDescriptor.get(), size, nullptr)), true };
1147 }
1148
1149 if (const auto* installedFont = findClosestFont(familyFonts, request))
1150 return { adoptCF(CTFontCreateWithFontDescriptor(installedFont->fontDescriptor.get(), size, nullptr)), false };
1151
1152 return { nullptr };
1153}
