39struct FontFamilySpecificationKey {
40 RetainPtr<CTFontDescriptorRef> fontDescriptor;
41 FontDescriptionKey fontDescriptionKey;
42
43 FontFamilySpecificationKey() = default;
44
45 FontFamilySpecificationKey(CTFontDescriptorRef fontDescriptor, const FontDescription& fontDescription)
46 : fontDescriptor(fontDescriptor)
47 , fontDescriptionKey(fontDescription)
48 { }
49
50 explicit FontFamilySpecificationKey(WTF::HashTableDeletedValueType deletedValue)
51 : fontDescriptionKey(deletedValue)
52 { }
53
54 bool operator==(const FontFamilySpecificationKey& other) const
55 {
56 return safeCFEqual(fontDescriptor.get(), other.fontDescriptor.get()) && fontDescriptionKey == other.fontDescriptionKey;
57 }
58
59 bool operator!=(const FontFamilySpecificationKey& other) const
60 {
61 return !(*this == other);
62 }
63
64 bool isHashTableDeletedValue() const { return fontDescriptionKey.isHashTableDeletedValue(); }
65};
66
67struct FontFamilySpecificationKeyHash {
68 static unsigned hash(const FontFamilySpecificationKey& key) { return computeHash(safeCFHash(key.fontDescriptor.get()), key.fontDescriptionKey); }
69 static bool equal(const FontFamilySpecificationKey& a, const FontFamilySpecificationKey& b) { return a == b; }
70 static const bool safeToCompareToEmptyOrDeleted = true;
71};

93FontRanges FontFamilySpecificationCoreText::fontRanges(const FontDescription& fontDescription) const
94{
95 auto& fontPlatformData = fontMap().ensure(FontFamilySpecificationKey(m_fontDescriptor.get(), fontDescription), [&] () {
96 auto size = fontDescription.computedSize();
97
98 auto font = adoptCF(CTFontCreateWithFontDescriptor(m_fontDescriptor.get(), size, nullptr));
99
100 font = preparePlatformFont(font.get(), fontDescription, nullptr, { });
101
102 auto [syntheticBold, syntheticOblique] = computeNecessarySynthesis(font.get(), fontDescription, ShouldComputePhysicalTraits::Yes).boldObliquePair();
103
104 return makeUnique<FontPlatformData>(font.get(), size, false, syntheticOblique, fontDescription.orientation(), fontDescription.widthVariant(), fontDescription.textRenderingMode());
105 }).iterator->value;
106
107 ASSERT(fontPlatformData);
108
109 return FontRanges(FontCache::singleton().fontForPlatformData(*fontPlatformData));
110}
