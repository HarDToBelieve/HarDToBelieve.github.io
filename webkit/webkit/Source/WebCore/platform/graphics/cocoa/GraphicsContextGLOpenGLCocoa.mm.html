68#if ASSERT_ENABLED
69// Returns true if we have volatile context extension for the particular API or
70// if the particular API is not used.
71static bool checkVolatileContextSupportIfDeviceExists(EGLDisplay display, const char* deviceContextVolatileExtension,
72 const char* deviceContextExtension, EGLint deviceContextType)
73{
74 const char *clientExtensions = EGL_QueryString(EGL_NO_DISPLAY, EGL_EXTENSIONS);
75 if (clientExtensions && strstr(clientExtensions, deviceContextVolatileExtension))
76 return true;
77 EGLDeviceEXT device = EGL_NO_DEVICE_EXT;
78 if (!EGL_QueryDisplayAttribEXT(display, EGL_DEVICE_EXT, reinterpret_cast<EGLAttrib*>(&device)))
79 return true;
80 if (device == EGL_NO_DEVICE_EXT)
81 return true;
82 const char* deviceExtensions = EGL_QueryDeviceStringEXT(device, EGL_EXTENSIONS);
83 if (!deviceExtensions || !strstr(deviceExtensions, deviceContextExtension))
84 return true;
85 void* deviceContext = nullptr;
86 if (!EGL_QueryDeviceAttribEXT(device, deviceContextType, reinterpret_cast<EGLAttrib*>(&deviceContext)))
87 return true;
88 return !deviceContext;
89}

100static ScopedEGLDefaultDisplay InitializeEGLDisplay(const GraphicsContextGLAttributes& attrs)
101{
102 EGLint majorVersion = 0;
103 EGLint minorVersion = 0;
104 EGLDisplay display;
105
106 Vector<EGLint> displayAttributes;
107
108 // FIXME: This should come in from the GraphicsContextGLAttributes.
109 bool shouldInitializeWithVolatileContextSupport = !isCurrentContextPredictable();
110 if (shouldInitializeWithVolatileContextSupport) {
111 // For WK1 type APIs we need to set "volatile platform context" for specific
112 // APIs, since client code will be able to override the thread-global context
113 // that ANGLE expects.
114 displayAttributes.append(EGL_PLATFORM_ANGLE_DEVICE_CONTEXT_VOLATILE_EAGL_ANGLE);
115 displayAttributes.append(EGL_TRUE);
116 displayAttributes.append(EGL_PLATFORM_ANGLE_DEVICE_CONTEXT_VOLATILE_CGL_ANGLE);
117 displayAttributes.append(EGL_TRUE);
118 }
119 bool canUseMetal = platformSupportsMetal();
120 if (attrs.useMetal && canUseMetal) {
121 displayAttributes.append(EGL_PLATFORM_ANGLE_TYPE_ANGLE);
122 displayAttributes.append(EGL_PLATFORM_ANGLE_TYPE_METAL_ANGLE);
123 }
124
125 LOG(WebGL, "Attempting to use ANGLE's %s backend.\n", attrs.useMetal && canUseMetal ? "Metal" : "OpenGL");
126 if (attrs.powerPreference != GraphicsContextGLAttributes::PowerPreference::Default) {
127 displayAttributes.append(EGL_POWER_PREFERENCE_ANGLE);
128 if (attrs.powerPreference == GraphicsContextGLAttributes::PowerPreference::LowPower)
129 displayAttributes.append(EGL_LOW_POWER_ANGLE);
130 else {
131 ASSERT(attrs.powerPreference == GraphicsContextGLAttributes::PowerPreference::HighPerformance);
132 displayAttributes.append(EGL_HIGH_POWER_ANGLE);
133 }
134 }
135
136 displayAttributes.append(EGL_NONE);
137 display = EGL_GetPlatformDisplayEXT(EGL_PLATFORM_ANGLE_ANGLE, reinterpret_cast<void*>(EGL_DEFAULT_DISPLAY), displayAttributes.data());
138
139 if (EGL_Initialize(display, &majorVersion, &minorVersion) == EGL_FALSE) {
140 LOG(WebGL, "EGLDisplay Initialization failed.");
141 return { };
142 }
143 LOG(WebGL, "ANGLE initialised Major: %d Minor: %d", majorVersion, minorVersion);
144 if (shouldInitializeWithVolatileContextSupport) {
145 // After initialization, EGL_DEFAULT_DISPLAY will return the platform-customized display.
146 ASSERT(display == EGL_GetDisplay(EGL_DEFAULT_DISPLAY));
147 ASSERT(checkVolatileContextSupportIfDeviceExists(display, "EGL_ANGLE_platform_device_context_volatile_eagl", "EGL_ANGLE_device_eagl", EGL_EAGL_CONTEXT_ANGLE));
148 ASSERT(checkVolatileContextSupportIfDeviceExists(display, "EGL_ANGLE_platform_device_context_volatile_cgl", "EGL_ANGLE_device_cgl", EGL_CGL_CONTEXT_ANGLE));
149 }
150 return ScopedEGLDefaultDisplay::adoptInitializedDisplay(display);
151}

171RefPtr<GraphicsContextGLOpenGL> GraphicsContextGLOpenGL::create(GraphicsContextGLAttributes attrs, HostWindow* hostWindow)
172{
173 // If ANGLE is not loaded, we can fail immediately.
174 if (!isANGLEAvailable()) {
175 LOG(WebGL, "ANGLE shared library was not loaded. Can't make GraphicsContextGL.");
176 return nullptr;
177 }
178
179 // Make space for the incoming context if we're full.
180 GraphicsContextGLOpenGLManager::sharedManager().recycleContextIfNecessary();
181 if (GraphicsContextGLOpenGLManager::sharedManager().hasTooManyContexts())
182 return nullptr;
183
184 RefPtr<GraphicsContextGLOpenGL> context = adoptRef(new GraphicsContextGLOpenGL(attrs, hostWindow, nullptr, nullptr));
185
186 if (!context->m_contextObj)
187 return nullptr;
188
189 GraphicsContextGLOpenGLManager::sharedManager().addContext(context.get());
190
191 return context;
192}

209GraphicsContextGLOpenGL::GraphicsContextGLOpenGL(GraphicsContextGLAttributes attrs, HostWindow*, GraphicsContextGLOpenGL* sharedContext, GraphicsContextGLIOSurfaceSwapChain* swapChain)
210 : GraphicsContextGL(attrs, sharedContext)
211{
212 m_isForWebGL2 = attrs.webGLVersion == GraphicsContextGLWebGLVersion::WebGL2;
213
214 m_displayObj = InitializeEGLDisplay(attrs);
215 if (!m_displayObj)
216 return;
217
218 bool supportsPowerPreference = false;
219#if PLATFORM(MAC)
220 const char *displayExtensions = EGL_QueryString(m_displayObj, EGL_EXTENSIONS);
221 m_supportsPowerPreference = strstr(displayExtensions, "EGL_ANGLE_power_preference");
222 supportsPowerPreference = m_supportsPowerPreference;
223#endif
224 if (!supportsPowerPreference && attrs.powerPreference == GraphicsContextGLPowerPreference::HighPerformance) {
225 attrs.powerPreference = GraphicsContextGLPowerPreference::Default;
226 setContextAttributes(attrs);
227 }
228
229 EGLint configAttributes[] = {
230 EGL_RENDERABLE_TYPE, EGL_OPENGL_ES2_BIT,
231 EGL_RED_SIZE, 8,
232 EGL_GREEN_SIZE, 8,
233 EGL_BLUE_SIZE, 8,
234 EGL_ALPHA_SIZE, 8,
235 EGL_NONE
236 };
237 EGLint numberConfigsReturned = 0;
238 EGL_ChooseConfig(m_displayObj, configAttributes, &m_configObj, 1, &numberConfigsReturned);
239 if (numberConfigsReturned != 1) {
240 LOG(WebGL, "EGLConfig Initialization failed.");
241 return;
242 }
243 LOG(WebGL, "Got EGLConfig");
244
245 EGL_BindAPI(EGL_OPENGL_ES_API);
246 if (EGL_GetError() != EGL_SUCCESS) {
247 LOG(WebGL, "Unable to bind to OPENGL_ES_API");
248 return;
249 }
250
251 Vector<EGLint> eglContextAttributes;
252 if (m_isForWebGL2) {
253 eglContextAttributes.append(EGL_CONTEXT_CLIENT_VERSION);
254 eglContextAttributes.append(3);
255 } else {
256 eglContextAttributes.append(EGL_CONTEXT_CLIENT_VERSION);
257 eglContextAttributes.append(2);
258 // ANGLE will upgrade the context to ES3 automatically unless this is specified.
259 eglContextAttributes.append(EGL_CONTEXT_OPENGL_BACKWARDS_COMPATIBLE_ANGLE);
260 eglContextAttributes.append(EGL_FALSE);
261 }
262 if (!sharedContext) {
263 // The shared context is only non-null when creating a context
264 // on behalf of the VideoTextureCopier. WebGL-specific rendering
265 // feedback loop validation does not work in multi-context
266 // scenarios, and must be disabled for the VideoTextureCopier's
267 // context.
268 eglContextAttributes.append(EGL_CONTEXT_WEBGL_COMPATIBILITY_ANGLE);
269 eglContextAttributes.append(EGL_TRUE);
270 }
271 // WebGL requires that all resources are cleared at creation.
272 eglContextAttributes.append(EGL_ROBUST_RESOURCE_INITIALIZATION_ANGLE);
273 eglContextAttributes.append(EGL_TRUE);
274 // WebGL doesn't allow client arrays.
275 eglContextAttributes.append(EGL_CONTEXT_CLIENT_ARRAYS_ENABLED_ANGLE);
276 eglContextAttributes.append(EGL_FALSE);
277 // WebGL doesn't allow implicit creation of objects on bind.
278 eglContextAttributes.append(EGL_CONTEXT_BIND_GENERATES_RESOURCE_CHROMIUM);
279 eglContextAttributes.append(EGL_FALSE);
280
281 eglContextAttributes.append(EGL_NONE);
282
283 m_contextObj = EGL_CreateContext(m_displayObj, m_configObj, sharedContext ? static_cast<EGLContext>(sharedContext->m_contextObj) : EGL_NO_CONTEXT, eglContextAttributes.data());
284 if (m_contextObj == EGL_NO_CONTEXT) {
285 LOG(WebGL, "EGLContext Initialization failed.");
286 return;
287 }
288 LOG(WebGL, "Got EGLContext");
289
290 EGL_MakeCurrent(m_displayObj, EGL_NO_SURFACE, EGL_NO_SURFACE, m_contextObj);
291
292 if (m_isForWebGL2)
293 gl::Enable(GraphicsContextGL::PRIMITIVE_RESTART_FIXED_INDEX);
294
295 Vector<ASCIILiteral, 4> requiredExtensions;
296 if (m_isForWebGL2) {
297 // For WebGL 2.0 occlusion queries to work.
298 requiredExtensions.append("GL_EXT_occlusion_query_boolean"_s);
299 }
300#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
301 if (!needsEAGLOnMac()) {
302 // For IOSurface-backed textures.
303 requiredExtensions.append("GL_ANGLE_texture_rectangle"_s);
304 // For creating the EGL surface from an IOSurface.
305 requiredExtensions.append("GL_EXT_texture_format_BGRA8888"_s);
306 }
307
308#endif // PLATFORM(MAC) || PLATFORM(MACCATALYST)
309 ExtensionsGL& extensions = getExtensions();
310 for (auto& extension : requiredExtensions) {
311 if (!extensions.supports(extension)) {
312 LOG(WebGL, "Missing required extension. %s", extension.characters());
313 return;
314 }
315 extensions.ensureEnabled(extension);
316 }
317 validateAttributes();
318 attrs = contextAttributes(); // They may have changed during validation.
319
320 if (swapChain)
321 m_swapChain = swapChain;
322 else {
323 BEGIN_BLOCK_OBJC_EXCEPTIONS
324 m_webGLLayer = adoptNS([[WebGLLayer alloc] initWithDevicePixelRatio:attrs.devicePixelRatio contentsOpaque:!attrs.alpha]);
325#ifndef NDEBUG
326 [m_webGLLayer setName:@"WebGL Layer"];
327#endif
328 END_BLOCK_OBJC_EXCEPTIONS
329 m_swapChain = &[m_webGLLayer swapChain];
330 }
331
332 // Create the texture that will be used for the framebuffer.
333 GLenum textureTarget = drawingBufferTextureTarget();
334
335 gl::GenTextures(1, &m_texture);
336 gl::BindTexture(textureTarget, m_texture);
337 gl::TexParameteri(textureTarget, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
338 gl::TexParameteri(textureTarget, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
339 gl::TexParameteri(textureTarget, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
340 gl::TexParameteri(textureTarget, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
341 gl::BindTexture(textureTarget, 0);
342
343 gl::GenFramebuffers(1, &m_fbo);
344 gl::BindFramebuffer(GL_FRAMEBUFFER, m_fbo);
345 m_state.boundDrawFBO = m_state.boundReadFBO = m_fbo;
346
347 if (!attrs.antialias && (attrs.stencil || attrs.depth))
348 gl::GenRenderbuffers(1, &m_depthStencilBuffer);
349
350 // If necessary, create another framebuffer for the multisample results.
351 if (attrs.antialias) {
352 gl::GenFramebuffers(1, &m_multisampleFBO);
353 gl::BindFramebuffer(GL_FRAMEBUFFER, m_multisampleFBO);
354 m_state.boundDrawFBO = m_state.boundReadFBO = m_multisampleFBO;
355 gl::GenRenderbuffers(1, &m_multisampleColorBuffer);
356 if (attrs.stencil || attrs.depth)
357 gl::GenRenderbuffers(1, &m_multisampleDepthStencilBuffer);
358 } else if (attrs.preserveDrawingBuffer) {
359 // If necessary, create another texture to handle preserveDrawingBuffer:true without
360 // antialiasing.
361 gl::GenTextures(1, &m_preserveDrawingBufferTexture);
362 gl::BindTexture(GL_TEXTURE_2D, m_preserveDrawingBufferTexture);
363 gl::TexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
364 gl::TexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
365 gl::TexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
366 gl::TexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
367 gl::BindTexture(GL_TEXTURE_2D, 0);
368 // Create an FBO with which to perform BlitFramebuffer from one texture to the other.
369 gl::GenFramebuffers(1, &m_preserveDrawingBufferFBO);
370 }
371
372 gl::ClearColor(0, 0, 0, 0);
373
374 LOG(WebGL, "Created a GraphicsContextGLOpenGL (%p).", this);
375}

377GraphicsContextGLOpenGL::~GraphicsContextGLOpenGL()
378{
379 GraphicsContextGLOpenGLManager::sharedManager().removeContext(this);
380 if (makeContextCurrent()) {
381 GraphicsContextGLAttributes attrs = contextAttributes();
382 gl::DeleteTextures(1, &m_texture);
383
384 if (attrs.antialias) {
385 gl::DeleteRenderbuffers(1, &m_multisampleColorBuffer);
386 if (attrs.stencil || attrs.depth)
387 gl::DeleteRenderbuffers(1, &m_multisampleDepthStencilBuffer);
388 gl::DeleteFramebuffers(1, &m_multisampleFBO);
389 } else {
390 if (attrs.stencil || attrs.depth)
391 gl::DeleteRenderbuffers(1, &m_depthStencilBuffer);
392 }
393 gl::DeleteFramebuffers(1, &m_fbo);
394 if (m_preserveDrawingBufferTexture)
395 gl::DeleteTextures(1, &m_preserveDrawingBufferTexture);
396 if (m_preserveDrawingBufferFBO)
397 gl::DeleteFramebuffers(1, &m_preserveDrawingBufferFBO);
398 }
399 if (m_displayBufferPbuffer) {
400 EGL_DestroySurface(m_displayObj, m_displayBufferPbuffer);
401 auto recycledBuffer = m_swapChain->recycleBuffer();
402 if (recycledBuffer.handle)
403 EGL_DestroySurface(m_displayObj, recycledBuffer.handle);
404 auto contentsHandle = m_swapChain->detachClient();
405 if (contentsHandle)
406 EGL_DestroySurface(m_displayObj, contentsHandle);
407 }
408 if (m_contextObj) {
409 clearCurrentContext();
410 EGL_DestroyContext(m_displayObj, m_contextObj);
411 } else
412 ASSERT(currentContext != this);
413 LOG(WebGL, "Destroyed a GraphicsContextGLOpenGL (%p).", this);
414}

455bool GraphicsContextGLOpenGL::makeContextCurrent()
456{
457 if (!m_contextObj)
458 return false;
459 // If there is no drawing buffer, we failed to allocate one during preparing for display.
460 // The exception is the case when the context is used before reshaping.
461 if (!m_displayBufferBacking && !getInternalFramebufferSize().isEmpty())
462 return false;
463 if (currentContext == this)
464 return true;
465 // Calling MakeCurrent is important to set volatile platform context. See InitializeEGLDisplay().
466 if (!EGL_MakeCurrent(m_displayObj, EGL_NO_SURFACE, EGL_NO_SURFACE, m_contextObj))
467 return false;
468 if (isCurrentContextPredictable())
469 currentContext = this;
470 return true;
471}

473void GraphicsContextGLOpenGL::clearCurrentContext()
474{
475 EGLBoolean result = EGL_MakeCurrent(m_displayObj, EGL_NO_SURFACE, EGL_NO_SURFACE, EGL_NO_CONTEXT);
476 ASSERT_UNUSED(result, result);
477 currentContext = nullptr;
478}

480#if PLATFORM(IOS_FAMILY)
481bool GraphicsContextGLOpenGL::releaseCurrentContext(ReleaseBehavior releaseBehavior)
482{
483 if (!isANGLEAvailable())
484 return true;
485
486 // At the moment this function is relevant only when web thread lock owns the GraphicsContextGLOpenGL current context.
487 ASSERT(!isCurrentContextPredictable());
488
489 if (!EGL_BindAPI(EGL_OPENGL_ES_API))
490 return false;
491 if (EGL_GetCurrentContext() == EGL_NO_CONTEXT)
492 return true;
493
494 // At the time of writing, ANGLE does not flush on MakeCurrent. Since we are
495 // potentially switching threads, we should flush.
496 // Note: Here we assume also that ANGLE has only one platform context -- otherwise
497 // we would need to flush each EGL context that has been used.
498 gl::Flush();
499
500 // Unset the EGL current context, since the next access might be from another thread, and the
501 // context cannot be current on multiple threads.
502
503 if (releaseBehavior == ReleaseBehavior::ReleaseThreadResources) {
504 // Called when we do not know if we will ever see another call from this thread again.
505 // Unset the EGL current context by releasing whole EGL thread state.
506 // Theoretically ReleaseThread can reset the bound API, so the rest of the code mentions BindAPI/QueryAPI.
507 return EGL_ReleaseThread();
508 }
509 // On WebKit owned threads, WebKit is able to choose the time for the EGL cleanup.
510 // This is why we only unset the context.
511 // Note: At the time of writing the EGL cleanup is chosen to be not done at all.
512 EGLDisplay display = EGL_GetDisplay(EGL_DEFAULT_DISPLAY);
513 if (display == EGL_NO_DISPLAY)
514 return false;
515 return EGL_MakeCurrent(display, EGL_NO_SURFACE, EGL_NO_SURFACE, EGL_NO_CONTEXT);
516}
517#endif

519void GraphicsContextGLOpenGL::checkGPUStatus()
520{
521 if (m_failNextStatusCheck) {
522 LOG(WebGL, "Pretending the GPU has reset (%p). Lose the context.", this);
523 m_failNextStatusCheck = false;
524 forceContextLost();
525 clearCurrentContext();
526 return;
527 }
528
529 // Only do the check every statusCheckThreshold calls.
530 if (m_statusCheckCount)
531 return;
532
533 m_statusCheckCount = (m_statusCheckCount + 1) % statusCheckThreshold;
534
535 GLint restartStatus = 0;
536 // FIXME: check via KHR_robustness.
537 restartStatus = 0;
538}

563bool GraphicsContextGLOpenGL::reshapeDisplayBufferBacking()
564{
565 ASSERT(!getInternalFramebufferSize().isEmpty());
566 // Reset the current backbuffer now before allocating a new one in order to slightly reduce memory pressure.
567 if (m_displayBufferBacking) {
568 m_displayBufferBacking.reset();
569 EGL_ReleaseTexImage(m_displayObj, m_displayBufferPbuffer, EGL_BACK_BUFFER);
570 EGL_DestroySurface(m_displayObj, m_displayBufferPbuffer);
571 m_displayBufferPbuffer = EGL_NO_SURFACE;
572 }
573 // Reset the future recycled buffer now, because it most likely will not be reusable at the time it will be reused.
574 auto recycledBuffer = m_swapChain->recycleBuffer();
575 if (recycledBuffer.handle)
576 EGL_DestroySurface(m_displayObj, recycledBuffer.handle);
577 recycledBuffer.surface.reset();
578 return allocateAndBindDisplayBufferBacking();
579}

581bool GraphicsContextGLOpenGL::allocateAndBindDisplayBufferBacking()
582{
583 ASSERT(!getInternalFramebufferSize().isEmpty());
584 auto backing = IOSurface::create(getInternalFramebufferSize(), DestinationColorSpace::SRGB());
585 if (!backing)
586 return false;
587
588 backing->migrateColorSpaceToProperties();
589
590 const bool usingAlpha = contextAttributes().alpha;
591 const auto size = getInternalFramebufferSize();
592 const EGLint surfaceAttributes[] = {
593 EGL_WIDTH, size.width(),
594 EGL_HEIGHT, size.height(),
595 EGL_IOSURFACE_PLANE_ANGLE, 0,
596 EGL_TEXTURE_TARGET, WebCore::GraphicsContextGLOpenGL::EGLDrawingBufferTextureTarget(),
597 EGL_TEXTURE_INTERNAL_FORMAT_ANGLE, usingAlpha ? GL_BGRA_EXT : GL_RGB,
598 EGL_TEXTURE_FORMAT, EGL_TEXTURE_RGBA,
599 EGL_TEXTURE_TYPE_ANGLE, GL_UNSIGNED_BYTE,
600 // Only has an effect on the iOS Simulator.
601 EGL_IOSURFACE_USAGE_HINT_ANGLE, EGL_IOSURFACE_WRITE_HINT_ANGLE,
602 EGL_NONE, EGL_NONE
603 };
604 EGLSurface pbuffer = EGL_CreatePbufferFromClientBuffer(m_displayObj, EGL_IOSURFACE_ANGLE, backing->surface(), m_configObj, surfaceAttributes);
605 if (!pbuffer)
606 return false;
607 return bindDisplayBufferBacking(WTFMove(backing), pbuffer);
608}

610bool GraphicsContextGLOpenGL::bindDisplayBufferBacking(std::unique_ptr<IOSurface> backing, void* pbuffer)
611{
612 GCGLenum textureTarget = drawingBufferTextureTarget();
613 ScopedRestoreTextureBinding restoreBinding(drawingBufferTextureTargetQuery(), textureTarget, textureTarget != TEXTURE_RECTANGLE_ARB);
614 gl::BindTexture(textureTarget, m_texture);
615 if (!EGL_BindTexImage(m_displayObj, pbuffer, EGL_BACK_BUFFER)) {
616 EGL_DestroySurface(m_displayObj, pbuffer);
617 return false;
618 }
619 m_displayBufferPbuffer = pbuffer;
620 m_displayBufferBacking = WTFMove(backing);
621 return true;
622}

624void* GraphicsContextGLOpenGL::createPbufferAndAttachIOSurface(GCGLenum target, PbufferAttachmentUsage usage, GCGLenum internalFormat, GCGLsizei width, GCGLsizei height, GCGLenum type, IOSurfaceRef surface, GCGLuint plane)
625{
626 if (target != GraphicsContextGL::TEXTURE_RECTANGLE_ARB && target != GraphicsContextGL::TEXTURE_2D) {
627 LOG(WebGL, "Unknown texture target %d.", static_cast<int>(target));
628 return nullptr;
629 }
630
631 auto eglTextureTarget = [&] () -> EGLint {
632 if (target == GraphicsContextGL::TEXTURE_RECTANGLE_ARB)
633 return EGL_TEXTURE_RECTANGLE_ANGLE;
634 return EGL_TEXTURE_2D;
635 }();
636
637 if (eglTextureTarget != GraphicsContextGLOpenGL::EGLDrawingBufferTextureTarget()) {
638 LOG(WebGL, "Mismatch in EGL texture target: %d should be %d.", static_cast<int>(target), GraphicsContextGLOpenGL::EGLDrawingBufferTextureTarget());
639 return nullptr;
640 }
641
642 auto usageHintAngle = [&] () -> EGLint {
643 if (usage == PbufferAttachmentUsage::Read)
644 return EGL_IOSURFACE_READ_HINT_ANGLE;
645 if (usage == PbufferAttachmentUsage::Write)
646 return EGL_IOSURFACE_WRITE_HINT_ANGLE;
647 return EGL_IOSURFACE_READ_HINT_ANGLE | EGL_IOSURFACE_WRITE_HINT_ANGLE;
648 }();
649
650 const EGLint surfaceAttributes[] = {
651 EGL_WIDTH, width,
652 EGL_HEIGHT, height,
653 EGL_IOSURFACE_PLANE_ANGLE, static_cast<EGLint>(plane),
654 EGL_TEXTURE_TARGET, static_cast<EGLint>(eglTextureTarget),
655 EGL_TEXTURE_INTERNAL_FORMAT_ANGLE, static_cast<EGLint>(internalFormat),
656 EGL_TEXTURE_FORMAT, EGL_TEXTURE_RGBA,
657 EGL_TEXTURE_TYPE_ANGLE, static_cast<EGLint>(type),
658 // Only has an effect on the iOS Simulator.
659 EGL_IOSURFACE_USAGE_HINT_ANGLE, usageHintAngle,
660 EGL_NONE, EGL_NONE
661 };
662
663 auto display = platformDisplay();
664 EGLSurface pbuffer = EGL_CreatePbufferFromClientBuffer(display, EGL_IOSURFACE_ANGLE, surface, platformConfig(), surfaceAttributes);
665 if (!pbuffer) {
666 LOG(WebGL, "EGL_CreatePbufferFromClientBuffer failed.");
667 return nullptr;
668 }
669
670 if (!EGL_BindTexImage(display, pbuffer, EGL_BACK_BUFFER)) {
671 LOG(WebGL, "EGL_BindTexImage failed.");
672 EGL_DestroySurface(display, pbuffer);
673 return nullptr;
674 }
675
676 return pbuffer;
677}

679void GraphicsContextGLOpenGL::destroyPbufferAndDetachIOSurface(void* handle)
680{
681 auto display = platformDisplay();
682 EGL_ReleaseTexImage(display, handle, EGL_BACK_BUFFER);
683 EGL_DestroySurface(display, handle);
684}

751void GraphicsContextGLOpenGL::prepareForDisplay()
752{
753 if (m_layerComposited)
754 return;
755 if (!makeContextCurrent())
756 return;
757 prepareTextureImpl();
758
759 // The IOSurface will be used from other graphics subsystem, so flush GL commands.
760 gl::Flush();
761
762 auto recycledBuffer = m_swapChain->recycleBuffer();
763
764 EGL_ReleaseTexImage(m_displayObj, m_displayBufferPbuffer, EGL_BACK_BUFFER);
765 m_swapChain->present({ WTFMove(m_displayBufferBacking), m_displayBufferPbuffer });
766 m_displayBufferPbuffer = EGL_NO_SURFACE;
767
768 bool hasNewBacking = false;
769 if (recycledBuffer.surface && recycledBuffer.surface->size() == getInternalFramebufferSize()) {
770 hasNewBacking = bindDisplayBufferBacking(WTFMove(recycledBuffer.surface), recycledBuffer.handle);
771 recycledBuffer.handle = nullptr;
772 }
773 recycledBuffer.surface.reset();
774 if (recycledBuffer.handle)
775 EGL_DestroySurface(m_displayObj, recycledBuffer.handle);
776
777 // Error will be handled by next call to makeContextCurrent() which will notice lack of display buffer.
778 if (!hasNewBacking)
779 allocateAndBindDisplayBufferBacking();
780
781 markLayerComposited();
782}

784std::optional<PixelBuffer> GraphicsContextGLOpenGL::readCompositedResults()
785{
786 auto& displayBuffer = m_swapChain->displayBuffer();
787 if (!displayBuffer.surface || !displayBuffer.handle)
788 return std::nullopt;
789 if (displayBuffer.surface->size() != getInternalFramebufferSize())
790 return std::nullopt;
791 // Note: We are using GL to read the IOSurface. At the time of writing, there are no convinient
792 // functions to convert the IOSurface pixel data to ImageData. The image data ends up being
793 // drawn to a ImageBuffer, but at the time there's no functions to construct a NativeImage
794 // out of an IOSurface in such a way that drawing the NativeImage would be guaranteed leave
795 // the IOSurface be unrefenced after the draw call finishes.
796 ScopedTexture texture;
797 GCGLenum textureTarget = drawingBufferTextureTarget();
798 ScopedRestoreTextureBinding restoreBinding(drawingBufferTextureTargetQuery(), textureTarget, textureTarget != TEXTURE_RECTANGLE_ARB);
799 gl::BindTexture(textureTarget, texture);
800 if (!EGL_BindTexImage(m_displayObj, displayBuffer.handle, EGL_BACK_BUFFER))
801 return std::nullopt;
802 gl::TexParameteri(textureTarget, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
803 ScopedFramebuffer fbo;
804 ScopedRestoreReadFramebufferBinding fboBinding(m_isForWebGL2, m_state.boundReadFBO, fbo);
805 gl::FramebufferTexture2D(fboBinding.framebufferTarget(), GL_COLOR_ATTACHMENT0, textureTarget, texture, 0);
806 ASSERT(gl::CheckFramebufferStatus(fboBinding.framebufferTarget()) == GL_FRAMEBUFFER_COMPLETE);
807
808 auto result = readPixelsForPaintResults();
809
810 EGLBoolean releaseOk = EGL_ReleaseTexImage(m_displayObj, displayBuffer.handle, EGL_BACK_BUFFER);
811 ASSERT_UNUSED(releaseOk, releaseOk);
812 return result;
813}
