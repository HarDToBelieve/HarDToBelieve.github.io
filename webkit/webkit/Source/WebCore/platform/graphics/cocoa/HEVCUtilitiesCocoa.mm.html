41std::optional<MediaCapabilitiesInfo> validateHEVCParameters(const HEVCParameters& parameters, bool hasAlphaChannel, bool hdrSupport)
42{
43 CMVideoCodecType codec = kCMVideoCodecType_HEVC;
44 if (hasAlphaChannel) {
45 if (!PAL::isAVFoundationFrameworkAvailable() || !PAL::canLoad_AVFoundation_AVVideoCodecTypeHEVCWithAlpha())
46 return std::nullopt;
47
48 auto codecCode = FourCC::fromString(AVVideoCodecTypeHEVCWithAlpha);
49 if (!codecCode)
50 return std::nullopt;
51
52 codec = codecCode.value().value;
53 }
54
55 if (hdrSupport) {
56 // Platform supports HDR playback of HEVC Main10 Profile, as defined by ITU-T H.265 v6 (06/2019).
57 bool isMain10 = parameters.generalProfileSpace == 0
58 && (parameters.generalProfileIDC == 2 || parameters.generalProfileCompatibilityFlags == 1);
59 if (!isMain10)
60 return std::nullopt;
61 }
62
63 OSStatus status = VTSelectAndCreateVideoDecoderInstance(codec, kCFAllocatorDefault, nullptr, nullptr);
64 if (status != noErr)
65 return std::nullopt;
66
67 if (!canLoad_VideoToolbox_VTCopyHEVCDecoderCapabilitiesDictionary()
68 || !canLoad_VideoToolbox_kVTHEVCDecoderCapability_SupportedProfiles()
69 || !canLoad_VideoToolbox_kVTHEVCDecoderCapability_PerProfileSupport()
70 || !canLoad_VideoToolbox_kVTHEVCDecoderProfileCapability_IsHardwareAccelerated()
71 || !canLoad_VideoToolbox_kVTHEVCDecoderProfileCapability_MaxDecodeLevel()
72 || !canLoad_VideoToolbox_kVTHEVCDecoderProfileCapability_MaxPlaybackLevel())
73 return std::nullopt;
74
75 auto capabilities = adoptCF(VTCopyHEVCDecoderCapabilitiesDictionary());
76 if (!capabilities)
77 return std::nullopt;
78
79 auto supportedProfiles = (CFArrayRef)CFDictionaryGetValue(capabilities.get(), kVTHEVCDecoderCapability_SupportedProfiles);
80 if (!supportedProfiles || CFGetTypeID(supportedProfiles) != CFArrayGetTypeID())
81 return std::nullopt;
82
83 int16_t generalProfileIDC = parameters.generalProfileIDC;
84 auto cfGeneralProfileIDC = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt16Type, &generalProfileIDC));
85 auto searchRange = CFRangeMake(0, CFArrayGetCount(supportedProfiles));
86 if (!CFArrayContainsValue(supportedProfiles, searchRange, cfGeneralProfileIDC.get()))
87 return std::nullopt;
88
89 auto perProfileSupport = (CFDictionaryRef)CFDictionaryGetValue(capabilities.get(), kVTHEVCDecoderCapability_PerProfileSupport);
90 if (!perProfileSupport || CFGetTypeID(perProfileSupport) != CFDictionaryGetTypeID())
91 return std::nullopt;
92
93 auto generalProfileIDCString = String::number(generalProfileIDC).createCFString();
94 auto profileSupport = (CFDictionaryRef)CFDictionaryGetValue(perProfileSupport, generalProfileIDCString.get());
95 if (!profileSupport || CFGetTypeID(profileSupport) != CFDictionaryGetTypeID())
96 return std::nullopt;
97
98 MediaCapabilitiesInfo info;
99
100 info.supported = true;
101
102 auto isHardwareAccelerated = (CFBooleanRef)CFDictionaryGetValue(profileSupport, kVTHEVCDecoderProfileCapability_IsHardwareAccelerated);
103 if (isHardwareAccelerated && CFGetTypeID(isHardwareAccelerated) == CFBooleanGetTypeID())
104 info.powerEfficient = CFBooleanGetValue(isHardwareAccelerated);
105
106 auto cfMaxDecodeLevel = (CFNumberRef)CFDictionaryGetValue(profileSupport, kVTHEVCDecoderProfileCapability_MaxDecodeLevel);
107 if (cfMaxDecodeLevel && CFGetTypeID(cfMaxDecodeLevel) == CFNumberGetTypeID()) {
108 int16_t maxDecodeLevel = 0;
109 if (!CFNumberGetValue(cfMaxDecodeLevel, kCFNumberSInt16Type, &maxDecodeLevel))
110 return std::nullopt;
111
112 if (parameters.generalLevelIDC > maxDecodeLevel)
113 return std::nullopt;
114 }
115
116 auto cfMaxPlaybackLevel = (CFNumberRef)CFDictionaryGetValue(profileSupport, kVTHEVCDecoderProfileCapability_MaxPlaybackLevel);
117 if (cfMaxPlaybackLevel && CFGetTypeID(cfMaxPlaybackLevel) == CFNumberGetTypeID()) {
118 int16_t maxPlaybackLevel = 0;
119 if (!CFNumberGetValue(cfMaxPlaybackLevel, kCFNumberSInt16Type, &maxPlaybackLevel))
120 return std::nullopt;
121
122 info.smooth = parameters.generalLevelIDC <= maxPlaybackLevel;
123 }
124
125 return info;
126}

128static CMVideoCodecType codecType(DoViParameters::Codec codec)
129{
130 switch (codec) {
131 case DoViParameters::Codec::AVC1:
132 case DoViParameters::Codec::AVC3:
133 return kCMVideoCodecType_H264;
134 case DoViParameters::Codec::HEV1:
135 case DoViParameters::Codec::HVC1:
136 return kCMVideoCodecType_HEVC;
137 }
138}

140static std::optional<Vector<uint16_t>> parseStringArrayFromDictionaryToUInt16Vector(CFDictionaryRef dictionary, const void* key)
141{
142 auto value = CFDictionaryGetValue(dictionary, key);
143 if (!value || CFGetTypeID(value) != CFArrayGetTypeID())
144 return std::nullopt;
145 NSArray *array = (__bridge NSArray *)value;
146 Vector<uint16_t> vector;
147 vector.reserveInitialCapacity(array.count);
148 for (id value in array) {
149 if (![value isKindOfClass:NSString.class])
150 return std::nullopt;
151 auto numericValue = parseInteger<uint16_t>(String((NSString *)value));
152 if (!numericValue)
153 return std::nullopt;
154 vector.uncheckedAppend(*numericValue);
155 }
156 return vector;
157}

159std::optional<MediaCapabilitiesInfo> validateDoViParameters(const DoViParameters& parameters, bool hasAlphaChannel, bool hdrSupport)
160{
161 if (hasAlphaChannel)
162 return std::nullopt;
163
164 if (hdrSupport) {
165 // Platform supports HDR playback of HEVC Main10 Profile, which is signalled by DoVi profiles 4, 5, 7, & 8.
166 switch (parameters.bitstreamProfileID) {
167 case 4:
168 case 5:
169 case 7:
170 case 8:
171 break;
172 default:
173 return std::nullopt;
174 }
175 }
176
177 OSStatus status = VTSelectAndCreateVideoDecoderInstance(codecType(parameters.codec), kCFAllocatorDefault, nullptr, nullptr);
178 if (status != noErr)
179 return std::nullopt;
180
181 if (!canLoad_VideoToolbox_VTCopyHEVCDecoderCapabilitiesDictionary()
182 || !canLoad_VideoToolbox_kVTDolbyVisionDecoderCapability_SupportedProfiles()
183 || !canLoad_VideoToolbox_kVTDolbyVisionDecoderCapability_SupportedLevels()
184 || !canLoad_VideoToolbox_kVTDolbyVisionDecoderCapability_IsHardwareAccelerated())
185 return std::nullopt;
186
187 auto capabilities = adoptCF(VTCopyHEVCDecoderCapabilitiesDictionary());
188 if (!capabilities)
189 return std::nullopt;
190
191 auto supportedProfiles = parseStringArrayFromDictionaryToUInt16Vector(capabilities.get(), kVTDolbyVisionDecoderCapability_SupportedProfiles);
192 if (!supportedProfiles)
193 return std::nullopt;
194
195 auto supportedLevels = parseStringArrayFromDictionaryToUInt16Vector(capabilities.get(), kVTDolbyVisionDecoderCapability_SupportedLevels);
196 if (!supportedLevels)
197 return std::nullopt;
198
199 auto isHardwareAcceleratedCF = (CFBooleanRef)CFDictionaryGetValue(capabilities.get(), kVTDolbyVisionDecoderCapability_IsHardwareAccelerated);
200 if (!isHardwareAcceleratedCF || CFGetTypeID(isHardwareAcceleratedCF) != CFBooleanGetTypeID())
201 return std::nullopt;
202 bool isHardwareAccelerated = CFBooleanGetValue(isHardwareAcceleratedCF);
203
204 if (!supportedProfiles.value().contains(parameters.bitstreamProfileID) || !supportedLevels.value().contains(parameters.bitstreamLevelID))
205 return std::nullopt;
206
207 return { { true, true, isHardwareAccelerated } };
208}
