48inline std::unique_ptr<IOSurface> IOSurface::surfaceFromPool(IntSize size, IntSize contextSize, const DestinationColorSpace& colorSpace, Format pixelFormat)
49{
50 auto cachedSurface = IOSurfacePool::sharedPool().takeSurface(size, colorSpace, pixelFormat);
51 if (!cachedSurface)
52 return nullptr;
53
54 cachedSurface->setContextSize(contextSize);
55 return cachedSurface;
56}

63std::unique_ptr<IOSurface> IOSurface::create(IntSize size, IntSize contextSize, const DestinationColorSpace& colorSpace, Format pixelFormat)
64{
65 if (auto cachedSurface = surfaceFromPool(size, contextSize, colorSpace, pixelFormat)) {
66 LOG_WITH_STREAM(IOSurface, stream << "IOSurface::create took from pool: " << *cachedSurface);
67 return cachedSurface;
68 }
69 bool success = false;
70 auto surface = std::unique_ptr<IOSurface>(new IOSurface(size, contextSize, colorSpace, pixelFormat, success));
71 if (!success) {
72 LOG(IOSurface, "IOSurface::create failed to create %dx%d surface", size.width(), size.height());
73 return nullptr;
74 }
75
76 LOG_WITH_STREAM(IOSurface, stream << "IOSurface::create created " << *surface);
77 return surface;
78}

91std::unique_ptr<IOSurface> IOSurface::createFromImage(CGImageRef image)
92{
93 if (!image)
94 return nullptr;
95
96 size_t width = CGImageGetWidth(image);
97 size_t height = CGImageGetHeight(image);
98
99 auto surface = IOSurface::create(IntSize(width, height), DestinationColorSpace { CGImageGetColorSpace(image) });
100 if (!surface)
101 return nullptr;
102 auto surfaceContext = surface->ensurePlatformContext();
103 CGContextDrawImage(surfaceContext, CGRectMake(0, 0, width, height), image);
104 CGContextFlush(surfaceContext);
105 return surface;
106}

113static NSDictionary *optionsForBiplanarSurface(IntSize size, unsigned pixelFormat, size_t firstPlaneBytesPerPixel, size_t secondPlaneBytesPerPixel)
114{
115 int width = size.width();
116 int height = size.height();
117
118 size_t firstPlaneBytesPerRow = IOSurfaceAlignProperty(kIOSurfaceBytesPerRow, width * firstPlaneBytesPerPixel);
119 size_t firstPlaneTotalBytes = IOSurfaceAlignProperty(kIOSurfaceAllocSize, height * firstPlaneBytesPerRow);
120
121 size_t secondPlaneBytesPerRow = IOSurfaceAlignProperty(kIOSurfaceBytesPerRow, width * secondPlaneBytesPerPixel);
122 size_t secondPlaneTotalBytes = IOSurfaceAlignProperty(kIOSurfaceAllocSize, height * secondPlaneBytesPerRow);
123
124 size_t totalBytes = firstPlaneTotalBytes + secondPlaneTotalBytes;
125 ASSERT(totalBytes);
126
127 NSArray *planeInfo = @[
128 @{
129 (id)kIOSurfacePlaneWidth: @(width),
130 (id)kIOSurfacePlaneHeight: @(height),
131 (id)kIOSurfacePlaneBytesPerRow: @(firstPlaneBytesPerRow),
132 (id)kIOSurfacePlaneOffset: @(0),
133 (id)kIOSurfacePlaneSize: @(firstPlaneTotalBytes)
134 },
135 @{
136 (id)kIOSurfacePlaneWidth: @(width),
137 (id)kIOSurfacePlaneHeight: @(height),
138 (id)kIOSurfacePlaneBytesPerRow: @(secondPlaneBytesPerRow),
139 (id)kIOSurfacePlaneOffset: @(firstPlaneTotalBytes),
140 (id)kIOSurfacePlaneSize: @(secondPlaneTotalBytes)
141 }
142 ];
143
144 return @{
145 (id)kIOSurfaceWidth: @(width),
146 (id)kIOSurfaceHeight: @(height),
147 (id)kIOSurfacePixelFormat: @(pixelFormat),
148 (id)kIOSurfaceAllocSize: @(totalBytes),
149#if PLATFORM(IOS_FAMILY)
150 (id)kIOSurfaceCacheMode: @(kIOMapWriteCombineCache),
151#endif
152 (id)kIOSurfacePlaneInfo: planeInfo,
153 };
154}

156static NSDictionary *optionsFor32BitSurface(IntSize size, unsigned pixelFormat)
157{
158 int width = size.width();
159 int height = size.height();
160
161 unsigned bytesPerElement = 4;
162 unsigned bytesPerPixel = 4;
163
164 size_t bytesPerRow = IOSurfaceAlignProperty(kIOSurfaceBytesPerRow, width * bytesPerPixel);
165 ASSERT(bytesPerRow);
166
167 size_t totalBytes = IOSurfaceAlignProperty(kIOSurfaceAllocSize, height * bytesPerRow);
168 ASSERT(totalBytes);
169
170 return @{
171 (id)kIOSurfaceWidth: @(width),
172 (id)kIOSurfaceHeight: @(height),
173 (id)kIOSurfacePixelFormat: @(pixelFormat),
174 (id)kIOSurfaceBytesPerElement: @(bytesPerElement),
175 (id)kIOSurfaceBytesPerRow: @(bytesPerRow),
176 (id)kIOSurfaceAllocSize: @(totalBytes),
177#if PLATFORM(IOS_FAMILY)
178 (id)kIOSurfaceCacheMode: @(kIOMapWriteCombineCache),
179#endif
180 (id)kIOSurfaceElementHeight: @(1)
181 };
182
183}

185IOSurface::IOSurface(IntSize size, IntSize contextSize, const DestinationColorSpace& colorSpace, Format format, bool& success)
186 : m_colorSpace(colorSpace)
187 , m_size(size)
188 , m_contextSize(contextSize)
189{
190 ASSERT(!success);
191 ASSERT(contextSize.width() <= size.width());
192 ASSERT(contextSize.height() <= size.height());
193 ASSERT(!size.isEmpty());
194
195 NSDictionary *options;
196
197 switch (format) {
198 case Format::BGRA:
199 options = optionsFor32BitSurface(size, 'BGRA');
200 break;
201#if HAVE(IOSURFACE_RGB10)
202 case Format::RGB10:
203 options = optionsFor32BitSurface(size, 'w30r');
204 break;
205 case Format::RGB10A8:
206 options = optionsForBiplanarSurface(size, 'b3a8', 4, 1);
207 break;
208#endif
209 case Format::YUV422:
210 options = optionsForBiplanarSurface(size, '422f', 1, 1);
211 break;
212 }
213 m_surface = adoptCF(IOSurfaceCreate((CFDictionaryRef)options));
214 success = !!m_surface;
215 if (success)
216 m_totalBytes = IOSurfaceGetAllocSize(m_surface.get());
217 else
218 RELEASE_LOG_ERROR(Layers, "IOSurface creation failed for size: (%d %d) and format: (%d)", size.width(), size.height(), format);
219}

231static IntSize computeMaximumSurfaceSize()
232{
233 IntSize maxSize(clampToInteger(IOSurfaceGetPropertyMaximum(kIOSurfaceWidth)), clampToInteger(IOSurfaceGetPropertyMaximum(kIOSurfaceHeight)));
234
235 // Protect against maxSize being { 0, 0 }.
236 const int maxSurfaceDimensionLowerBound = 1024;
237
238#if PLATFORM(IOS_FAMILY)
239 // Match limits imposed by Core Animation. FIXME: should have API for this <rdar://problem/25454148>
240 const int maxSurfaceDimension = 8 * 1024;
241#else
242 // IOSurface::maximumSize() can return { INT_MAX, INT_MAX } when hardware acceleration is unavailable.
243 const int maxSurfaceDimension = 32 * 1024;
244#endif
245
246 return maxSize.constrainedBetween({ maxSurfaceDimensionLowerBound, maxSurfaceDimensionLowerBound }, { maxSurfaceDimension, maxSurfaceDimension });
247}

261IntSize IOSurface::maximumSize()
262{
263 auto size = surfaceMaximumSize().load();
264 if (size.isEmpty()) {

266 surfaceMaximumSize().store(computedSize);
267 return computedSize;

287void IOSurface::setContextSize(IntSize contextSize)
288{
289 if (contextSize == m_contextSize)
290 return;
291
292 // Release the graphics context and update the context size. Next time the graphics context is
293 // accessed, we will construct it again with the right size.
294 releaseGraphicsContext();
295 m_contextSize = contextSize;
296}

298CGContextRef IOSurface::ensurePlatformContext(const HostWindow* hostWindow)
299{
300 if (m_cgContext)
301 return m_cgContext.get();
302
303 CGBitmapInfo bitmapInfo = kCGImageAlphaPremultipliedFirst | kCGBitmapByteOrder32Host;
304
305 size_t bitsPerComponent = 8;
306 size_t bitsPerPixel = 32;
307
308 switch (format()) {
309 case Format::BGRA:
310 break;
311#if HAVE(IOSURFACE_RGB10)
312 case Format::RGB10:
313 case Format::RGB10A8:
314 // A half-float format will be used if CG needs to read back the IOSurface contents,
315 // but for an IOSurface-to-IOSurface copy, there should be no conversion.
316 bitsPerComponent = 16;
317 bitsPerPixel = 64;
318 bitmapInfo = kCGImageAlphaPremultipliedLast | kCGBitmapByteOrder16Host | kCGBitmapFloatComponents;
319 break;
320#endif
321 case Format::YUV422:
322 ASSERT_NOT_REACHED();
323 break;
324 }
325
326 m_cgContext = adoptCF(CGIOSurfaceContextCreate(m_surface.get(), m_contextSize.width(), m_contextSize.height(), bitsPerComponent, bitsPerPixel, m_colorSpace.platformColorSpace(), bitmapInfo));
327
328#if PLATFORM(MAC)
329 if (auto displayMask = primaryOpenGLDisplayMask()) {
330 if (hostWindow && hostWindow->displayID())
331 displayMask = displayMaskForDisplay(hostWindow->displayID());
332ALLOW_DEPRECATED_DECLARATIONS_BEGIN
333 CGIOSurfaceContextSetDisplayMask(m_cgContext.get(), displayMask);
334ALLOW_DEPRECATED_DECLARATIONS_END
335 }
336#else
337 UNUSED_PARAM(hostWindow);
338#endif
339
340 return m_cgContext.get();
341}

354VolatilityState IOSurface::state() const
355{
356 uint32_t previousState = 0;
357 IOReturn ret = IOSurfaceSetPurgeable(m_surface.get(), kIOSurfacePurgeableKeepCurrent, &previousState);
358 ASSERT_UNUSED(ret, ret == kIOReturnSuccess);
359 return previousState == kIOSurfacePurgeableEmpty ? VolatilityState::Empty : VolatilityState::Valid;
360}
361
362bool IOSurface::isVolatile() const
363{
364 uint32_t previousState = 0;
365 IOReturn ret = IOSurfaceSetPurgeable(m_surface.get(), kIOSurfacePurgeableKeepCurrent, &previousState);
366 ASSERT_UNUSED(ret, ret == kIOReturnSuccess);
367 return previousState != kIOSurfacePurgeableNonVolatile;
368}
369
370VolatilityState IOSurface::setVolatile(bool isVolatile)
371{
372 uint32_t previousState = 0;
373 IOReturn ret = IOSurfaceSetPurgeable(m_surface.get(), isVolatile ? kIOSurfacePurgeableVolatile : kIOSurfacePurgeableNonVolatile, &previousState);
374 ASSERT_UNUSED(ret, ret == kIOReturnSuccess);
375
376 if (previousState == kIOSurfacePurgeableEmpty)
377 return VolatilityState::Empty;
378
379 return VolatilityState::Valid;
380}

382IOSurface::Format IOSurface::format() const
383{
384 unsigned pixelFormat = IOSurfaceGetPixelFormat(m_surface.get());
385 if (pixelFormat == 'BGRA')
386 return Format::BGRA;
387
388#if HAVE(IOSURFACE_RGB10)
389 if (pixelFormat == 'w30r')
390 return Format::RGB10;
391
392 if (pixelFormat == 'b3a8')
393 return Format::RGB10A8;
394#endif
395
396 if (pixelFormat == '422f')
397 return Format::YUV422;
398
399 ASSERT_NOT_REACHED();
400 return Format::BGRA;
401}

442void IOSurface::convertToFormat(std::unique_ptr<IOSurface>&& inSurface, Format format, WTF::Function<void(std::unique_ptr<IOSurface>)>&& callback)
443{
444 static IOSurfaceAcceleratorRef accelerator;
445 if (!accelerator) {
446 IOSurfaceAcceleratorCreate(nullptr, nullptr, &accelerator);
447
448 auto runLoopSource = IOSurfaceAcceleratorGetRunLoopSource(accelerator);
449 CFRunLoopAddSource(CFRunLoopGetMain(), runLoopSource, kCFRunLoopDefaultMode);
450 }
451
452 if (inSurface->format() == format) {
453 callback(WTFMove(inSurface));
454 return;
455 }
456
457 auto destinationSurface = IOSurface::create(inSurface->size(), inSurface->colorSpace(), format);
458 if (!destinationSurface) {
459 callback(nullptr);
460 return;
461 }
462
463 IOSurfaceRef destinationIOSurfaceRef = destinationSurface->surface();
464 IOSurfaceAcceleratorCompletion completion;
465 completion.completionRefCon = new WTF::Function<void(std::unique_ptr<IOSurface>)> (WTFMove(callback));
466 completion.completionRefCon2 = destinationSurface.release();
467 completion.completionCallback = [](void *completionRefCon, IOReturn, void * completionRefCon2) {
468 auto* callback = static_cast<WTF::Function<void(std::unique_ptr<IOSurface>)>*>(completionRefCon);
469 auto destinationSurface = std::unique_ptr<IOSurface>(static_cast<IOSurface*>(completionRefCon2));
470
471 (*callback)(WTFMove(destinationSurface));
472 delete callback;
473 };
474
475 NSDictionary *options = @{ (id)kIOSurfaceAcceleratorUnwireSurfaceKey : @YES };
476
477 IOReturn ret = IOSurfaceAcceleratorTransformSurface(accelerator, inSurface->surface(), destinationIOSurfaceRef, (CFDictionaryRef)options, nullptr, &completion, nullptr, nullptr);
478 ASSERT_UNUSED(ret, ret == kIOReturnSuccess);
479}
