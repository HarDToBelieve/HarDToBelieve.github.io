35id makeNSArrayElement(const IntRect& rect)
36{
37#if PLATFORM(MAC)
38 return [NSValue valueWithRect:rect];
39#else
40 return [NSValue valueWithCGRect:rect];
41#endif
42}
43
44#if PLATFORM(MAC) && !defined(NSGEOMETRY_TYPES_SAME_AS_CGGEOMETRY_TYPES)
45
46IntRect::operator NSRect() const
47{
48 return NSMakeRect(x(), y(), width(), height());
49}
50
51IntRect enclosingIntRect(const NSRect& rect)
52{
53 int left = clampTo<int>(std::floor(rect.origin.x));
54 int top = clampTo<int>(std::floor(rect.origin.y));
55 int right = clampTo<int>(std::ceil(NSMaxX(rect)));
56 int bottom = clampTo<int>(std::ceil(NSMaxY(rect)));
57 return { left, top, right - left, bottom - top };
58}
