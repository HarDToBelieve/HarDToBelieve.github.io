30#include <wtf/RetainPtr.h>
31#include <wtf/text/WTFString.h>

33#if PLATFORM(COCOA)
34OBJC_CLASS AVOutputContext;
35OBJC_CLASS NSData;
36#endif
37
38namespace WebCore {
39
40class MediaPlaybackTargetContext {
41public:
42 enum class Type : uint8_t {
43 None,
44 AVOutputContext,
45 SerializedAVOutputContext,
46 Mock,
47 };
48
49 enum class MockState : uint8_t {
50 Unknown = 0,
51 OutputDeviceUnavailable = 1,
52 OutputDeviceAvailable = 2,
53 };
54
55 MediaPlaybackTargetContext() = default;
56 WEBCORE_EXPORT explicit MediaPlaybackTargetContext(RetainPtr<AVOutputContext>&&);
57
58 MediaPlaybackTargetContext(RetainPtr<NSData>&& serializedOutputContext, bool hasActiveRoute)
59 : m_type(Type::SerializedAVOutputContext)
60 , m_serializedOutputContext(WTFMove(serializedOutputContext))
61 , m_cachedHasActiveRoute(hasActiveRoute)
62 {
63 ASSERT(m_serializedOutputContext);
64 }
65
66 MediaPlaybackTargetContext(const String& mockDeviceName, MockState state)
67 : m_type(Type::Mock)
68 , m_mockDeviceName(mockDeviceName)
69 , m_mockState(state)
70 {
71 }
72
73 Type type() const { return m_type; }
74 WEBCORE_EXPORT String deviceName() const;
75 WEBCORE_EXPORT bool hasActiveRoute() const;
76 bool supportsRemoteVideoPlayback() const;
77
78 MockState mockState() const
79 {
80 ASSERT(m_type == Type::Mock);
81 return m_mockState;
82 }
83
84 RetainPtr<AVOutputContext> outputContext() const
85 {
86 ASSERT(m_type == Type::AVOutputContext);
87 return m_outputContext;
88 }
89
90 RetainPtr<NSData> serializedOutputContext() const
91 {
92 ASSERT(m_type == Type::SerializedAVOutputContext);
93 return m_serializedOutputContext;
94 }
95
96 WEBCORE_EXPORT bool serializeOutputContext();
97 WEBCORE_EXPORT bool deserializeOutputContext();
98
99 bool encodingRequiresPlatformData() const { return m_type == Type::AVOutputContext || m_type == Type::SerializedAVOutputContext; }
100
101private:
102 Type m_type { Type::None };
103 RetainPtr<AVOutputContext> m_outputContext;
104 RetainPtr<NSData> m_serializedOutputContext;
105 bool m_cachedHasActiveRoute { false };
106
107 String m_mockDeviceName;
108 MockState m_mockState { MockState::Unknown };
109};
110
111} // namespace WebCore
112
113namespace WTF {

117 WebCore::MediaPlaybackTargetContext::Type,

119 WebCore::MediaPlaybackTargetContext::Type::AVOutputContext,
120 WebCore::MediaPlaybackTargetContext::Type::SerializedAVOutputContext,
121 WebCore::MediaPlaybackTargetContext::Type::Mock
122 >;
123};
124
125template<> struct EnumTraits<WebCore::MediaPlaybackTargetContext::MockState> {
126 using values = EnumValues<
127 WebCore::MediaPlaybackTargetContext::MockState,
128 WebCore::MediaPlaybackTargetContext::MockState::Unknown,
129 WebCore::MediaPlaybackTargetContext::MockState::OutputDeviceUnavailable,
130 WebCore::MediaPlaybackTargetContext::MockState::OutputDeviceAvailable
131 >;
132};
