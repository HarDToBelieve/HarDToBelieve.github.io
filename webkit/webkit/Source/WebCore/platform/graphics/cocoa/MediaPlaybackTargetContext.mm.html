38MediaPlaybackTargetContext::MediaPlaybackTargetContext(RetainPtr<AVOutputContext>&& outputContext)
39 : m_outputContext(outputContext)
40{
41 ASSERT(m_outputContext);
42 m_type = MediaPlaybackTargetContext::Type::AVOutputContext;
43}

45String MediaPlaybackTargetContext::deviceName() const
46{
47 ASSERT(m_type == MediaPlaybackTargetContext::Type::Mock || m_type == MediaPlaybackTargetContext::Type::AVOutputContext);
48
49 if (m_type == MediaPlaybackTargetContext::Type::Mock)
50 return m_mockDeviceName;
51
52 ASSERT(m_type == MediaPlaybackTargetContext::Type::AVOutputContext);
53 String deviceName;
54 if (![m_outputContext supportsMultipleOutputDevices])
55 deviceName = [m_outputContext deviceName];
56 else {
57 auto outputDeviceNames = adoptNS([[NSMutableArray alloc] init]);
58 for (AVOutputDevice *outputDevice in [m_outputContext outputDevices])
59 [outputDeviceNames addObject:[outputDevice deviceName]];
60
61 deviceName = [outputDeviceNames componentsJoinedByString:@" + "];
62 }
63
64 return deviceName;
65}
66bool MediaPlaybackTargetContext::hasActiveRoute() const
67{
68 ASSERT(m_type != MediaPlaybackTargetContext::Type::None);
69 if (m_type == MediaPlaybackTargetContext::Type::Mock)
70 return !m_mockDeviceName.isEmpty();
71
72 // FIXME: It may not be safe to use the cached `hasActiveRoute` of a serialized AVOutputContext
73 if (m_type == MediaPlaybackTargetContext::Type::SerializedAVOutputContext)
74 return m_cachedHasActiveRoute;
75
76 ASSERT(m_type == MediaPlaybackTargetContext::Type::AVOutputContext);
77 bool hasActiveRoute = false;
78 if ([m_outputContext respondsToSelector:@selector(supportsMultipleOutputDevices)] && [m_outputContext supportsMultipleOutputDevices] && [m_outputContext respondsToSelector:@selector(outputDevices)]) {
79 for (AVOutputDevice *outputDevice in [m_outputContext outputDevices]) {
80 if (outputDevice.deviceFeatures & (AVOutputDeviceFeatureVideo | AVOutputDeviceFeatureAudio))
81 hasActiveRoute = true;
82 }
83 } else if ([m_outputContext respondsToSelector:@selector(outputDevice)]) {
84 if (auto *outputDevice = [m_outputContext outputDevice])
85 hasActiveRoute = outputDevice.deviceFeatures & (AVOutputDeviceFeatureVideo | AVOutputDeviceFeatureAudio);
86 } else
87 hasActiveRoute = m_outputContext.get().deviceName;
88
89 return hasActiveRoute;
90}
91bool MediaPlaybackTargetContext::supportsRemoteVideoPlayback() const
92{
93 ASSERT(m_type == MediaPlaybackTargetContext::Type::Mock || m_type == MediaPlaybackTargetContext::Type::AVOutputContext);
94 if (m_type == MediaPlaybackTargetContext::Type::Mock)
95 return !m_mockDeviceName.isEmpty();
96
97 bool supportsRemoteVideoPlayback = false;
98 if (![m_outputContext respondsToSelector:@selector(supportsMultipleOutputDevices)] || ![m_outputContext supportsMultipleOutputDevices] || ![m_outputContext respondsToSelector:@selector(outputDevices)]) {
99 if (auto *outputDevice = [m_outputContext outputDevice]) {
100 if (outputDevice.deviceFeatures & AVOutputDeviceFeatureVideo)
101 supportsRemoteVideoPlayback = true;
102 }
103 } else {
104 for (AVOutputDevice *outputDevice in [m_outputContext outputDevices]) {
105 if (outputDevice.deviceFeatures & AVOutputDeviceFeatureVideo)
106 supportsRemoteVideoPlayback = true;
107 }
108 }
109
110 return supportsRemoteVideoPlayback;
111}

113bool MediaPlaybackTargetContext::serializeOutputContext()
114{
115 ASSERT(m_type != MediaPlaybackTargetContext::Type::SerializedAVOutputContext);
116 if (m_type != MediaPlaybackTargetContext::Type::AVOutputContext)
117 return false;
118
119 auto archiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
120 [archiver encodeObject:m_outputContext.get() forKey:NSKeyedArchiveRootObjectKey];
121 [archiver finishEncoding];
122
123 m_serializedOutputContext = [archiver encodedData];
124 m_type = MediaPlaybackTargetContext::Type::SerializedAVOutputContext;
125 m_outputContext.clear();
126
127 return true;
128}
129
130bool MediaPlaybackTargetContext::deserializeOutputContext()
131{
132 ASSERT(m_type != MediaPlaybackTargetContext::Type::AVOutputContext);
133 if (m_type != MediaPlaybackTargetContext::Type::SerializedAVOutputContext)
134 return false;
135
136 bool success = false;
137 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:m_serializedOutputContext.get() error:nullptr]);
138 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
139 auto allowedClassSet = adoptNS([[NSSet alloc] initWithArray:@[ PAL::getAVOutputContextClass() ]]);
140
141 @try {
142 id result = [unarchiver decodeObjectOfClasses:allowedClassSet.get() forKey:NSKeyedArchiveRootObjectKey];
143 ASSERT(result);
144 if (result) {
145 m_outputContext = result;
146 m_type = MediaPlaybackTargetContext::Type::AVOutputContext;
147 m_serializedOutputContext.clear();
148 success = true;
149 }
150 } @catch (NSException *exception) {
151 LOG_ERROR("Failed to decode object of class AVOutputContext : %@", exception);
152 } @finally {
153 [unarchiver finishDecoding];
154 unarchiver.get().delegate = nil;
155 }
156
157 return success;
158}
