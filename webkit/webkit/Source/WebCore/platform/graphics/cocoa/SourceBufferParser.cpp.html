39MediaPlayerEnums::SupportsType SourceBufferParser::isContentTypeSupported(const ContentType& type)
40{
41 MediaPlayerEnums::SupportsType supports = MediaPlayerEnums::SupportsType::IsNotSupported;
42 supports = std::max(supports, SourceBufferParserWebM::isContentTypeSupported(type));
43 supports = std::max(supports, SourceBufferParserAVFObjC::isContentTypeSupported(type));
44 return supports;
45}
46
47RefPtr<SourceBufferParser> SourceBufferParser::create(const ContentType& type, bool webMParserEnabled)
48{
49 if (SourceBufferParserWebM::isContentTypeSupported(type) != MediaPlayerEnums::SupportsType::IsNotSupported && webMParserEnabled)
50 return adoptRef(new SourceBufferParserWebM());
51
52 if (SourceBufferParserAVFObjC::isContentTypeSupported(type) != MediaPlayerEnums::SupportsType::IsNotSupported)
53 return adoptRef(new SourceBufferParserAVFObjC());
54
55 return nullptr;
56}

92size_t SourceBufferParser::Segment::size() const
93{
94 return WTF::switchOn(m_segment,
95#if HAVE(MT_PLUGIN_FORMAT_READER)
96 [](const RetainPtr<MTPluginByteSourceRef>& byteSource)
97 {
98 return clampTo<size_t>(MTPluginByteSourceGetLength(byteSource.get()));
99 },
100#endif
101 [](const Vector<uint8_t>& vector)
102 {
103 return vector.size();
104 }
105 );
106}
107
108size_t SourceBufferParser::Segment::read(size_t position, size_t sizeToRead, uint8_t* destination) const
109{
110 size_t segmentSize = size();
111 sizeToRead = std::min(sizeToRead, segmentSize - std::min(position, segmentSize));
112 return WTF::switchOn(m_segment,
113#if HAVE(MT_PLUGIN_FORMAT_READER)
114 [&](const RetainPtr<MTPluginByteSourceRef>& byteSource) -> size_t
115 {
116 size_t sizeRead = 0;
117 if (MTPluginByteSourceRead(byteSource.get(), sizeToRead, CheckedInt64(position), destination, &sizeRead) != noErr)
118 return 0;
119 return sizeRead;
120 },
121#endif
122 [&](const Vector<uint8_t>& vector)
123 {
124 memcpy(destination, vector.data() + position, sizeToRead);
125 return sizeToRead;
126 }
127 );
128}
129
130Vector<uint8_t> SourceBufferParser::Segment::takeVector()
131{
132 return WTF::switchOn(m_segment,
133#if HAVE(MT_PLUGIN_FORMAT_READER)
134 [&](RetainPtr<MTPluginByteSourceRef>&)
135 {
136 Vector<uint8_t> vector(size());
137 vector.shrink(read(0, vector.size(), vector.data()));
138 return vector;
139 },
140#endif
141 [](Vector<uint8_t>& vector)
142 {
143 return std::exchange(vector, { });
144 }
145 );
146}
