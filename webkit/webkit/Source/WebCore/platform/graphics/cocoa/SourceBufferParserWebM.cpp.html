274class SourceBufferParserWebM::StreamingVectorReader final : public webm::Reader {
275 WTF_MAKE_FAST_ALLOCATED;
276public:
277 void appendSegment(Segment&& segment)
278 {
279 m_data.push_back(WTFMove(segment));
280 if (m_currentSegment == m_data.end())
281 --m_currentSegment;
282 }
283
284 Status Read(std::size_t numToRead, uint8_t* outputBuffer, uint64_t* numActuallyRead) final
285 {
286 ASSERT(outputBuffer && numActuallyRead);
287 if (!numActuallyRead)
288 return Status(Status::kNotEnoughMemory);
289
290 *numActuallyRead = 0;
291 if (!outputBuffer)
292 return Status(Status::kNotEnoughMemory);
293
294 while (numToRead && m_currentSegment != m_data.end()) {
295 auto& currentSegment = *m_currentSegment;
296
297 if (m_positionWithinSegment >= currentSegment.size()) {
298 advanceToNextSegment();
299 continue;
300 }
301
302 uint64_t lastRead = currentSegment.read(m_positionWithinSegment, numToRead, outputBuffer);
303 m_position += lastRead;
304 *numActuallyRead += lastRead;
305 m_positionWithinSegment += lastRead;
306 numToRead -= lastRead;
307 if (m_positionWithinSegment == currentSegment.size())
308 advanceToNextSegment();
309 }
310 if (!numToRead)
311 return Status(Status::kOkCompleted);
312 if (*numActuallyRead)
313 return Status(Status::kOkPartial);
314 return Status(Status::kWouldBlock);
315 }
316
317 Status Skip(uint64_t numToSkip, uint64_t* numActuallySkipped) final
318 {
319 ASSERT(numActuallySkipped);
320 if (!numActuallySkipped)
321 return Status(Status::kNotEnoughMemory);
322
323 *numActuallySkipped = 0;
324 while (m_currentSegment != m_data.end()) {
325 auto& currentSegment = *m_currentSegment;
326
327 if (m_positionWithinSegment >= currentSegment.size()) {
328 advanceToNextSegment();
329 continue;
330 }
331
332 size_t numAvailable = currentSegment.size() - m_positionWithinSegment;
333
334 if (numToSkip < numAvailable) {
335 *numActuallySkipped += numToSkip;
336 m_position += numToSkip;
337 m_positionWithinSegment += numToSkip;
338 return Status(Status::kOkCompleted);
339 }
340
341 *numActuallySkipped += numAvailable;
342 m_position += numAvailable;
343 m_positionWithinSegment += numAvailable;
344 numToSkip -= numAvailable;
345 advanceToNextSegment();
346 continue;
347 }
348 if (!numToSkip)
349 return Status(Status::kOkCompleted);
350 if (*numActuallySkipped)
351 return Status(Status::kOkPartial);
352 return Status(Status::kWouldBlock);
353 }
354
355 uint64_t Position() const final { return m_position; }
356
357 void reset()
358 {
359 m_position = 0;
360 m_data.clear();
361 m_currentSegment = m_data.end();
362 }
363
364 bool rewindTo(uint64_t rewindToPosition)
365 {
366 ASSERT(rewindToPosition <= m_position);
367 if (rewindToPosition > m_position)
368 return false;
369
370 auto rewindAmount = m_position - rewindToPosition;
371 while (rewindAmount) {
372 if (rewindAmount <= m_positionWithinSegment) {
373 m_positionWithinSegment -= rewindAmount;
374 return true;
375 }
376
377 if (m_currentSegment == m_data.begin())
378 return false;
379
380 rewindAmount -= m_positionWithinSegment;
381 --m_currentSegment;
382 m_positionWithinSegment = m_currentSegment->size();
383 }
384 return true;
385 }
386
387 void reclaimSegments()
388 {
389 m_data.erase(m_data.begin(), m_currentSegment);
390 }
391
392private:
393 void advanceToNextSegment()
394 {
395 ASSERT(m_currentSegment != m_data.end());
396 if (m_currentSegment == m_data.end())
397 return;
398
399 ASSERT(m_positionWithinSegment == m_currentSegment->size());
400
401 ++m_currentSegment;
402 m_positionWithinSegment = 0;
403 }
404
405 StdList<Segment> m_data;
406 StdList<Segment>::iterator m_currentSegment { m_data.end() };
407 size_t m_position { 0 };
408 size_t m_positionWithinSegment { 0 };
409};

411class MediaDescriptionWebM final : public MediaDescription {
412 WTF_MAKE_FAST_ALLOCATED;
413public:
414 static Ref<MediaDescriptionWebM> create(webm::TrackEntry&& track)
415 {
416 ASSERT(isWebmParserAvailable());
417 return adoptRef(*new MediaDescriptionWebM(WTFMove(track)));
418 }
419
420 MediaDescriptionWebM(webm::TrackEntry&& track)
421 : m_track { WTFMove(track) }
422 {
423 }
424
425 AtomString codec() const final
426 {
427 if (m_codec)
428 return *m_codec;
429
430 // From: https://www.matroska.org/technical/codec_specs.html
431 // "Each Codec ID MUST include a Major Codec ID immediately following the Codec ID Prefix.
432 // A Major Codec ID MAY be followed by an OPTIONAL Codec ID Suffix to communicate a refinement
433 // of the Major Codec ID. If a Codec ID Suffix is used, then the Codec ID MUST include a forward
434 // slash (/) as a separator between the Major Codec ID and the Codec ID Suffix. The Major Codec
435 // ID MUST be composed of only capital letters (A-Z) and numbers (0-9). The Codec ID Suffix MUST
436 // be composed of only capital letters (A-Z), numbers (0-9), underscore (_), and forward slash (/)."
437
438 if (!m_track.codec_id.is_present()) {
439 m_codec = emptyAtom();
440 return *m_codec;
441 }
442
443 StringView codecID { m_track.codec_id.value().data(), (unsigned)m_track.codec_id.value().length() };
444 if (!codecID.startsWith("V_") && !codecID.startsWith("A_") && !codecID.startsWith("S_")) {
445 m_codec = emptyAtom();
446 return *m_codec;
447 }
448
449 auto slashLocation = codecID.find('/');
450 auto length = slashLocation == notFound ? codecID.length() - 2 : slashLocation - 2;
451 m_codec = codecID.substring(2, length).convertToASCIILowercaseAtom();
452 return *m_codec;
453 }
454 bool isVideo() const final { return m_track.track_type.is_present() && m_track.track_type.value() == TrackType::kVideo; }
455 bool isAudio() const final { return m_track.track_type.is_present() && m_track.track_type.value() == TrackType::kAudio; }
456 bool isText() const final { return m_track.track_type.is_present() && m_track.track_type.value() == TrackType::kSubtitle; }
457
458 const webm::TrackEntry& track() { return m_track; }
459
460private:
461 mutable std::optional<AtomString> m_codec;
462 webm::TrackEntry m_track;
463};

505MediaPlayerEnums::SupportsType SourceBufferParserWebM::isContentTypeSupported(const ContentType& type)
506{
507#if ENABLE(VP9) || ENABLE(VORBIS) || ENABLE(OPUS)
508 if (!isWebmParserAvailable())
509 return MediaPlayerEnums::SupportsType::IsNotSupported;
510
511 auto containerType = type.containerType();
512 bool isAudioContainerType = WTF::equalIgnoringASCIICase(containerType, "audio/webm");
513 bool isVideoContainerType = WTF::equalIgnoringASCIICase(containerType, "video/webm");
514 if (!isAudioContainerType && !isVideoContainerType)
515 return MediaPlayerEnums::SupportsType::IsNotSupported;
516
517 bool isAnyAudioCodecAvailable = false;
518#if ENABLE(VORBIS)
519 isAnyAudioCodecAvailable |= isVorbisDecoderAvailable();
520#endif
521#if ENABLE(OPUS)
522 isAnyAudioCodecAvailable |= isOpusDecoderAvailable();
523#endif
524
525 if (isAudioContainerType && !isAnyAudioCodecAvailable)
526 return MediaPlayerEnums::SupportsType::IsNotSupported;
527
528 bool isAnyCodecAvailable = isAnyAudioCodecAvailable;
529#if ENABLE(VP9)
530 isAnyCodecAvailable |= isVP9DecoderAvailable();
531#endif
532
533 if (!isAnyCodecAvailable)
534 return MediaPlayerEnums::SupportsType::IsNotSupported;
535
536 auto codecs = type.codecs();
537 if (codecs.isEmpty())
538 return MediaPlayerEnums::SupportsType::MayBeSupported;
539
540 for (auto& codec : codecs) {
541#if ENABLE(VP9)
542 if (codec.startsWith("vp09") || codec.startsWith("vp08") || equal(codec, "vp8") || equal(codec, "vp9")) {
543
544 if (!isVP9DecoderAvailable())
545 return MediaPlayerEnums::SupportsType::IsNotSupported;
546
547 auto codecParameters = parseVPCodecParameters(codec);
548 if (!codecParameters)
549 return MediaPlayerEnums::SupportsType::IsNotSupported;
550
551 if (!isVPCodecConfigurationRecordSupported(*codecParameters))
552 return MediaPlayerEnums::SupportsType::IsNotSupported;
553
554 continue;
555 }
556#endif // ENABLE(VP9)
557
558#if ENABLE(VORBIS)
559 if (codec == "vorbis") {
560 if (!isVorbisDecoderAvailable())
561 return MediaPlayerEnums::SupportsType::IsNotSupported;
562
563 continue;
564 }

606void SourceBufferParserWebM::appendData(Segment&& segment, CompletionHandler<void()>&& completionHandler, AppendFlags appendFlags)
607{
608 if (!m_parser) {
609 completionHandler();
610 return;
611 }
612
613 INFO_LOG_IF_POSSIBLE(LOGIDENTIFIER, "flags(", appendFlags == AppendFlags::Discontinuity ? "Discontinuity" : "", "), size(", segment.size(), ")");
614
615 if (appendFlags == AppendFlags::Discontinuity) {
616 m_reader->reset();
617 m_parser->DidSeek();
618 }
619 m_reader->appendSegment(WTFMove(segment));
620
621 while (true) {
622 m_status = m_parser->Feed(this, &m_reader);
623 if (m_status.ok() || m_status.code == Status::kEndOfFile || m_status.code == Status::kWouldBlock) {
624 m_reader->reclaimSegments();
625
626 // Audio tracks are grouped into meta-samples of a duration no more than m_minimumSampleDuration.
627 // But at the end of a file, no more audio data may be incoming, so flush and emit any pending
628 // audio buffers.
629 flushPendingAudioBuffers();
630
631 completionHandler();
632 return;
633 }
634
635 if (m_status.code != static_cast<int32_t>(ErrorCode::ReceivedEbmlInsideSegment))
636 break;
637
638 // The WebM Byte Stream Format <https://w3c.github.io/media-source/webm-byte-stream-format.html>
639 // states that an "Initialization Segment" starts with an Ebml Element followed by a Segment Element,
640 // and that a "Media Segment" is a single Cluster Element. However, since Cluster Elements are contained
641 // within a Segment Element, this means that a new Ebml Element can be appended at any time while the
642 // parser is still parsing children of a Segment Element. In this scenario, "rewind" the reader to the
643 // position of the incoming Ebml Element, and reset the parser, which will cause the Ebml element to be
644 // parsed as a top-level element, rather than as a child of the Segment.
645 if (!m_reader->rewindTo(*m_rewindToPosition)) {
646 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "failed to rewind reader");
647 break;
648 }
649
650 ALWAYS_LOG_IF_POSSIBLE(LOGIDENTIFIER, "received Ebml element while parsing Segment element. Rewound reader and reset parser, retrying");
651 m_rewindToPosition = std::nullopt;
652 m_parser->DidSeek();
653 m_state = State::None;
654 continue;
655 }
656
657 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "status.code(", m_status.code, ")");
658 m_callOnClientThreadCallback([this, protectedThis = makeRef(*this), code = m_status.code] {
659 if (m_didEncounterErrorDuringParsingCallback)
660 m_didEncounterErrorDuringParsingCallback(code);
661 });
662
663 completionHandler();
664}

761{
762 UNUSED_PARAM(metadata);
763 INFO_LOG_IF_POSSIBLE(LOGIDENTIFIER);
764
765 auto oldState = m_state;
766
767 if (metadata.id == Id::kEbml || metadata.id == Id::kSegment)

795 m_keyIds.clear();
796 }
797
798 return Status(Status::kOkCompleted);
799}

873Status SourceBufferParserWebM::OnTrackEntry(const ElementMetadata& metadata, const TrackEntry& trackEntry)
874{
875 UNUSED_PARAM(metadata);
876 if (!trackEntry.track_type.is_present() || !trackEntry.codec_id.is_present())
877 return Status(Status::kOkCompleted);
878
879 auto trackType = trackEntry.track_type.value();
880 String codecId { trackEntry.codec_id.value().data(), (unsigned)trackEntry.codec_id.value().length() };
881
882 ALWAYS_LOG_IF_POSSIBLE(LOGIDENTIFIER, trackType, ", codec ", codecId);
883
884 if (trackType == TrackType::kVideo && !supportedVideoCodecs().contains(codecId)) {
885 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "Encountered unsupported video codec ID ", codecId);
886 return Status(Status::Code(ErrorCode::UnsupportedVideoCodec));
887 }
888
889 if (trackType == TrackType::kAudio && !supportedAudioCodecs().contains(codecId)) {
890 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "Encountered unsupported audio codec ID ", codecId);
891 return Status(Status::Code(ErrorCode::UnsupportedAudioCodec));
892 }
893
894 if (trackType == TrackType::kVideo) {
895 auto track = VideoTrackPrivateWebM::create(TrackEntry(trackEntry));
896 if (m_logger)
897 track->setLogger(*m_logger, LoggerHelper::childLogIdentifier(m_logIdentifier, ++m_nextChildIdentifier));
898 m_initializationSegment->videoTracks.append({ MediaDescriptionWebM::create(TrackEntry(trackEntry)), WTFMove(track) });
899 } else if (trackType == TrackType::kAudio) {
900 auto track = AudioTrackPrivateWebM::create(TrackEntry(trackEntry));
901 if (m_logger)
902 track->setLogger(*m_logger, LoggerHelper::childLogIdentifier(m_logIdentifier, ++m_nextChildIdentifier));
903 m_initializationSegment->audioTracks.append({ MediaDescriptionWebM::create(TrackEntry(trackEntry)), WTFMove(track) });
904 }
905
906 if (trackEntry.content_encodings.is_present() && !trackEntry.content_encodings.value().encodings.empty()) {
907 ALWAYS_LOG_IF_POSSIBLE(LOGIDENTIFIER, "content_encodings detected:");
908 for (auto& encoding : trackEntry.content_encodings.value().encodings) {
909 if (!encoding.is_present())
910 continue;
911
912 auto& encryption = encoding.value().encryption;
913 if (!encryption.is_present())
914 continue;
915
916 auto& keyIdElement = encryption.value().key_id;
917 if (!keyIdElement.is_present())
918 continue;
919
920 auto& keyId = keyIdElement.value();
921 m_keyIds.append(std::make_pair(trackEntry.track_uid.value(), Uint8Array::create(keyId.data(), keyId.size())));
922 }
923 }
924
925 StringView codecString { trackEntry.codec_id.value().data(), (unsigned)trackEntry.codec_id.value().length() };
926#if ENABLE(VP9)
927 if (codecString == "V_VP9" && isVP9DecoderAvailable()) {
928 m_tracks.append(VideoTrackData::create(CodecType::VP9, trackEntry, *this));
929 return Status(Status::kOkCompleted);
930 }
931 if (codecString == "V_VP8" && isVP8DecoderAvailable()) {
932 m_tracks.append(VideoTrackData::create(CodecType::VP8, trackEntry, *this));
933 return Status(Status::kOkCompleted);
934 }
935#endif
936
937#if ENABLE(VORBIS)
938 if (codecString == "A_VORBIS" && isVorbisDecoderAvailable()) {
939 m_tracks.append(AudioTrackData::create(CodecType::Vorbis, trackEntry, *this, m_minimumAudioSampleDuration));
940 return Status(Status::kOkCompleted);
941 }
942#endif
943
944#if ENABLE(OPUS)
945 if (codecString == "A_OPUS" && isOpusDecoderAvailable()) {
946 m_tracks.append(AudioTrackData::create(CodecType::Opus, trackEntry, *this, m_minimumAudioSampleDuration));
947 return Status(Status::kOkCompleted);
948 }
949#endif
950
951 m_tracks.append(TrackData::create(CodecType::Unsupported, trackEntry, *this));
952 return Status(Status::kOkCompleted);
953}

1031webm::Status SourceBufferParserWebM::OnFrame(const FrameMetadata& metadata, Reader* reader, uint64_t* bytesRemaining)
1032{
1033 ASSERT(reader);
1034 if (!reader)
1035 return Status(Status::kNotEnoughMemory);
1036
1037 if (!m_currentBlock) {
1038 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "no current block!");
1039 return Status(Status::kInvalidElementId);
1040 }
1041
1042 auto* block = WTF::switchOn(*m_currentBlock, [](Block& block) {
1043 return &block;
1044 }, [](SimpleBlock& block) -> Block* {
1045 return &block;
1046 });
1047 if (!block)
1048 return Status(Status::kInvalidElementId);
1049
1050 auto trackNumber = block->track_number;
1051 auto* trackData = trackDataForTrackNumber(trackNumber);
1052 if (!trackData) {
1053 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "Ignoring unknown track number ", trackNumber);
1054 return Status(Status::kInvalidElementId);
1055 }
1056
1057 switch (trackData->codec()) {
1058 case CodecType::VP8:
1059 case CodecType::VP9:
1060 case CodecType::Vorbis:
1061 case CodecType::Opus:
1062 break;
1063
1064 case CodecType::Unsupported:
1065 return Skip(reader, bytesRemaining);
1066 }
1067
1068 return trackData->consumeFrameData(*reader, metadata, bytesRemaining, PAL::CMTimeMake(block->timecode + m_currentTimecode, m_timescale), block->num_frames);
1069}

1094webm::Status SourceBufferParserWebM::VideoTrackData::consumeFrameData(webm::Reader& reader, const FrameMetadata& metadata, uint64_t* bytesRemaining, const CMTime& presentationTime, int sampleCount)
1095{
1096#if ENABLE(VP9)
1097 CMBlockBufferRef rawBlockBuffer = nullptr;
1098
1099 if (m_currentPacketSize && *m_currentPacketSize != metadata.size) {
1100 // The packet's metadata doesn't match the currently pending partial packet; restart.
1101 ASSERT_NOT_REACHED_WITH_MESSAGE("VideoTrackData::consumeFrameData: webm in nonsensical state");
1102 m_partialBytesRead = 0;
1103 m_currentBlockBuffer = nullptr;
1104 m_currentPacketSize = std::nullopt;
1105 }
1106
1107 if (!m_currentPacketSize)
1108 m_currentPacketSize = metadata.size;
1109
1110 if (!m_currentBlockBuffer) {
1111 auto err = PAL::CMBlockBufferCreateWithMemoryBlock(kCFAllocatorDefault, nullptr, *m_currentPacketSize, kCFAllocatorDefault, nullptr, 0, *m_currentPacketSize, 0, &rawBlockBuffer);
1112 if (err) {
1113 PARSER_LOG_ERROR_IF_POSSIBLE("CMBlockBufferCreateWithMemoryBlock failed with error", err);
1114 return Skip(&reader, bytesRemaining);
1115 }
1116
1117 m_currentBlockBuffer = adoptCF(rawBlockBuffer);
1118 m_partialBytesRead = 0;
1119
1120 err = PAL::CMBlockBufferAssureBlockMemory(m_currentBlockBuffer.get());
1121 if (err) {
1122 PARSER_LOG_ERROR_IF_POSSIBLE("CMAudioSampleBufferCreateWithPacketDescriptions failed with error", err);
1123 return Skip(&reader, bytesRemaining);
1124 }
1125 }
1126
1127 if (*m_currentPacketSize < m_partialBytesRead + *bytesRemaining) {
1128 PARSER_LOG_ERROR_IF_POSSIBLE("Invalid frame size allocated, ignoring the entire frame");
1129 m_partialBytesRead += *bytesRemaining;
1130 return Skip(&reader, bytesRemaining);
1131 }
1132 while (*bytesRemaining) {
1133 size_t segmentSizeAtPosition = 0;
1134 uint8_t* blockBufferData = nullptr;
1135 auto err = PAL::CMBlockBufferGetDataPointer(m_currentBlockBuffer.get(), m_partialBytesRead, &segmentSizeAtPosition, nullptr, (char**)&blockBufferData);
1136 if (err) {
1137 PARSER_LOG_ERROR_IF_POSSIBLE("CMBlockBufferGetDataPointer failed with error", err);
1138 return Skip(&reader, bytesRemaining);
1139 }
1140 if (*bytesRemaining < segmentSizeAtPosition) {
1141 PARSER_LOG_ERROR_IF_POSSIBLE("An error occurred, destination buffer too small to contain our frame");
1142 return Skip(&reader, bytesRemaining);
1143 }
1144 uint64_t bytesRead;
1145 auto status = reader.Read(*bytesRemaining, (uint8_t*)blockBufferData, &bytesRead);
1146 *bytesRemaining -= bytesRead;
1147 m_partialBytesRead += bytesRead;
1148
1149 if (!status.completed_ok())
1150 return status;
1151 }
1152
1153 ASSERT(m_partialBytesRead <= *m_currentPacketSize);
1154 if (m_partialBytesRead < *m_currentPacketSize)
1155 return webm::Status(webm::Status::kOkPartial);
1156
1157 createSampleBuffer(presentationTime, sampleCount, metadata);
1158
1159 reset();
1160#else
1161 UNUSED_PARAM(metadata);
1162 UNUSED_PARAM(presentationTime);
1163 UNUSED_PARAM(sampleCount);
1164#endif
1165
1166 return Skip(&reader, bytesRemaining);
1167}

1169void SourceBufferParserWebM::VideoTrackData::createSampleBuffer(const CMTime& presentationTime, int sampleCount, const webm::FrameMetadata& metadata)
1170{
1171#if ENABLE(VP9)

1173 size_t segmentSizeAtPosition = 0;
1174 auto err = PAL::CMBlockBufferGetDataPointer(m_currentBlockBuffer.get(), 0, &segmentSizeAtPosition, nullptr, (char**)&blockBufferData);
1175 if (err) {
1176 PARSER_LOG_ERROR_IF_POSSIBLE("CMBlockBufferGetDataPointer failed with error", err);
1177 return;
1178 }
1179
1180 bool isKey = false;
1181 RetainPtr<CMFormatDescriptionRef> formatDescription;
1182 if (codec() == CodecType::VP9) {

1184 return;
1185
1186 if (m_headerParser.key()) {
1187 isKey = true;
1188 auto formatDescription = createFormatDescriptionFromVP9HeaderParser(m_headerParser, track().video.value().colour);
1189 if (!formatDescription) {
1190 PARSER_LOG_ERROR_IF_POSSIBLE("failed to create format description from VPX header");
1191 return;
1192 }
1193 setFormatDescription(WTFMove(formatDescription));
1194 }
1195 } else if (codec() == CodecType::VP8) {

1197 if (header && header->keyframe) {
1198 isKey = true;
1199 auto formatDescription = createFormatDescriptionFromVP8Header(*header, track().video.value().colour);
1200 if (!formatDescription) {
1201 PARSER_LOG_ERROR_IF_POSSIBLE("failed to create format description from VPX header");
1202 return;
1203 }
1204 setFormatDescription(WTFMove(formatDescription));
1205 }
1206 }
1207
1208 auto track = this->track();
1209 // FIXME: A block might contain more than one frame, but only this frame has been read into `currentBlockBuffer`.
1210 // Below we create sample buffers for each frame, each with the block's timecode and `num_frames` value.
1211 // Shouldn't we create just one sample buffer once all the block's frames have been read into `currentBlockBuffer`?
1212
1213 uint64_t duration = 0;
1214 if (track.default_duration.is_present())
1215 duration = track.default_duration.value() * presentationTime.timescale / 1000000000;
1216
1217 CMSampleBufferRef rawSampleBuffer = nullptr;
1218 size_t frameSize = PAL::CMBlockBufferGetDataLength(m_currentBlockBuffer.get());
1219 CMSampleTimingInfo timing = { PAL::CMTimeMake(duration, presentationTime.timescale), presentationTime, presentationTime };
1220 err = PAL::CMSampleBufferCreateReady(kCFAllocatorDefault, m_currentBlockBuffer.get(), this->formatDescription().get(), sampleCount, 1, &timing, 1, &frameSize, &rawSampleBuffer);
1221 if (err) {
1222 PARSER_LOG_ERROR_IF_POSSIBLE("CMSampleBufferCreateReady failed with error", err);
1223 return;
1224 }
1225 auto sampleBuffer = adoptCF(rawSampleBuffer);
1226
1227 auto attachmentsArray = PAL::CMSampleBufferGetSampleAttachmentsArray(sampleBuffer.get(), true);
1228 ASSERT(attachmentsArray);
1229 if (!attachmentsArray) {
1230 PARSER_LOG_ERROR_IF_POSSIBLE("CMSampleBufferGetSampleAttachmentsArray returned NULL");
1231 return;
1232 }
1233
1234 if (!isKey) {
1235 for (CFIndex i = 0, count = CFArrayGetCount(attachmentsArray); i < count; ++i) {
1236 CFMutableDictionaryRef attachments = checked_cf_cast<CFMutableDictionaryRef>(CFArrayGetValueAtIndex(attachmentsArray, i));
1237 CFDictionarySetValue(attachments, PAL::kCMSampleAttachmentKey_NotSync, kCFBooleanTrue);
1238 }
1239 }
1240
1241 auto trackID = track.track_uid.value();
1242 parser().provideMediaData(WTFMove(sampleBuffer), trackID, metadata.position);
1243#else
1244 UNUSED_PARAM(presentationTime);
1245 UNUSED_PARAM(sampleCount);
1246 UNUSED_PARAM(metadata);
1247#endif // ENABLE(VP9)

1258webm::Status SourceBufferParserWebM::AudioTrackData::consumeFrameData(webm::Reader& reader, const FrameMetadata& metadata, uint64_t* bytesRemaining, const CMTime& presentationTime, int sampleCount)
1259{
1260 ASSERT(sampleCount);
1261
1262 if (m_packetDescriptions.isEmpty()) {
1263 m_byteOffset = metadata.position;
1264 m_samplePresentationTime = presentationTime;
1265 }
1266
1267 if (m_currentPacketSize && *m_currentPacketSize != metadata.size) {
1268 // The packet's metadata doesn't match the currently pending partial packet; restart.
1269 ASSERT_NOT_REACHED_WITH_MESSAGE("AudioTrackData::consumeFrameData: webm in nonsensical state");
1270 m_partialBytesRead = 0;
1271 m_currentPacketSize = std::nullopt;
1272 }
1273
1274 if (!m_currentPacketSize)
1275 m_currentPacketSize = metadata.size;
1276
1277 // Ensure we have room to store the full pending frame.
1278 if (m_packetsData.size() < m_packetsBytesRead + metadata.size)
1279 m_packetsData.grow(m_packetsBytesRead + metadata.size);
1280
1281 if (!m_currentPacketByteOffset)
1282 m_currentPacketByteOffset = m_packetsBytesRead;
1283
1284 while (*bytesRemaining) {
1285 uint64_t bytesRead;
1286 auto status = reader.Read(*bytesRemaining, m_packetsData.data() + m_packetsBytesRead + m_partialBytesRead, &bytesRead);
1287 *bytesRemaining -= bytesRead;
1288 m_partialBytesRead += bytesRead;
1289
1290 if (!status.completed_ok())
1291 return status;
1292 }
1293
1294 ASSERT(m_partialBytesRead <= *m_currentPacketSize);
1295 if (m_partialBytesRead < *m_currentPacketSize)
1296 return webm::Status(webm::Status::kOkPartial);
1297
1298 m_packetsBytesRead += m_partialBytesRead;
1299 m_partialBytesRead = 0;
1300
1301 if (!formatDescription()) {
1302 if (!track().codec_private.is_present()) {
1303 PARSER_LOG_ERROR_IF_POSSIBLE("Audio track missing magic cookie");
1304 return Skip(&reader, bytesRemaining);
1305 }
1306
1307 RetainPtr<CMFormatDescriptionRef> formatDescription;
1308 auto& privateData = track().codec_private.value();
1309 if (codec() == CodecType::Vorbis)
1310 formatDescription = createVorbisAudioFormatDescription(privateData.size(), privateData.data());
1311 else if (codec() == CodecType::Opus) {
1312 OpusCookieContents cookieContents;
1313 if (!parseOpusPrivateData(privateData.size(), privateData.data(), m_packetsBytesRead, m_packetsData.data(), cookieContents)) {
1314 PARSER_LOG_ERROR_IF_POSSIBLE("Failed to parse Opus private data");
1315 return Skip(&reader, bytesRemaining);
1316 }
1317 if (!cookieContents.framesPerPacket) {
1318 PARSER_LOG_ERROR_IF_POSSIBLE("Opus private data indicates 0 frames per packet; bailing");
1319 return Skip(&reader, bytesRemaining);
1320 }
1321 m_framesPerPacket = cookieContents.framesPerPacket;
1322 m_frameDuration = cookieContents.frameDuration;
1323 formatDescription = createOpusAudioFormatDescription(cookieContents);
1324 }
1325
1326 if (!formatDescription) {
1327 PARSER_LOG_ERROR_IF_POSSIBLE("Failed to create format description from audio track header");
1328 return Skip(&reader, bytesRemaining);
1329 }
1330
1331 auto streamDescription = PAL::CMAudioFormatDescriptionGetStreamBasicDescription(formatDescription.get());
1332 if (!streamDescription) {
1333 PARSER_LOG_ERROR_IF_POSSIBLE("CMAudioFormatDescriptionGetStreamBasicDescription failed");
1334 return Skip(&reader, bytesRemaining);
1335 }
1336 m_packetDuration = PAL::CMTimeMake(streamDescription->mFramesPerPacket, streamDescription->mSampleRate);
1337
1338 setFormatDescription(WTFMove(formatDescription));
1339 } else if (codec() == CodecType::Opus) {
1340 // Opus technically allows the frame duration and frames-per-packet values to change from packet to packet.
1341 // CoreAudio doesn't support ASBD values like these to change on a per-packet basis, so throw an error when
1342 // that kind of variablility is encountered.
1343 OpusCookieContents cookieContents;
1344 auto& privateData = track().codec_private.value();
1345 if (!parseOpusPrivateData(privateData.size(), privateData.data(), m_packetsBytesRead, m_packetsData.data(), cookieContents)
1346 || cookieContents.framesPerPacket != m_framesPerPacket
1347 || cookieContents.frameDuration != m_frameDuration) {
1348 PARSER_LOG_ERROR_IF_POSSIBLE("Opus frames-per-packet changed within a track; error");
1349 return Status(Status::Code(ErrorCode::VariableFrameDuration));
1350 }
1351 }
1352
1353 m_packetDescriptions.append({ static_cast<int64_t>(*m_currentPacketByteOffset), 0, static_cast<UInt32>(*m_currentPacketSize) });
1354 m_currentPacketByteOffset = std::nullopt;
1355 m_currentPacketSize = std::nullopt;
1356
1357 auto sampleDuration = PAL::CMTimeGetSeconds(PAL::CMTimeSubtract(presentationTime, m_samplePresentationTime)) + PAL::CMTimeGetSeconds(m_packetDuration) * sampleCount;
1358
1359 if (sampleDuration >= m_minimumSampleDuration)
1360 createSampleBuffer(metadata.position);
1361
1362 ASSERT(!*bytesRemaining);
1363 return webm::Status(webm::Status::kOkCompleted);
1364}

1366void SourceBufferParserWebM::AudioTrackData::createSampleBuffer(std::optional<size_t> latestByteRangeOffset)
1367{
1368 if (m_packetDescriptions.isEmpty() || !m_packetsBytesRead)
1369 return;
1370
1371 ASSERT(!m_packetsData.isEmpty());
1372
1373 CMBlockBufferRef blockBuffer = nullptr;
1374 auto err = PAL::CMBlockBufferCreateWithMemoryBlock(kCFAllocatorDefault, nullptr, m_packetsBytesRead, kCFAllocatorDefault, nullptr, 0, m_packetsBytesRead, kCMBlockBufferAssureMemoryNowFlag, &blockBuffer);
1375 if (err) {
1376 PARSER_LOG_ERROR_IF_POSSIBLE("CMBlockBufferCreateWithMemoryBlock failed with %d", err);
1377 return;
1378 }
1379 auto buffer = adoptCF(blockBuffer);
1380
1381 err = PAL::CMBlockBufferReplaceDataBytes(m_packetsData.data(), buffer.get(), 0, m_packetsBytesRead);
1382 if (err) {
1383 PARSER_LOG_ERROR_IF_POSSIBLE("CMBlockBufferReplaceDataBytes failed with %d", err);
1384 return;
1385 }
1386
1387 CMSampleBufferRef rawSampleBuffer = nullptr;
1388 err = PAL::CMAudioSampleBufferCreateReadyWithPacketDescriptions(kCFAllocatorDefault, buffer.get(), formatDescription().get(), m_packetDescriptions.size(), m_samplePresentationTime, m_packetDescriptions.data(), &rawSampleBuffer);
1389 if (err) {
1390 PARSER_LOG_ERROR_IF_POSSIBLE("CMAudioSampleBufferCreateWithPacketDescriptions failed with %d", err);
1391 return;
1392 }
1393 auto sampleBuffer = adoptCF(rawSampleBuffer);
1394
1395 m_packetsData.remove(0, m_packetsBytesRead);
1396 if (m_currentPacketByteOffset)
1397 *m_currentPacketByteOffset -= m_packetsBytesRead;
1398 m_packetsBytesRead = 0;
1399 m_packetDescriptions.clear();
1400
1401 auto trackID = track().track_uid.value();
1402 parser().provideMediaData(WTFMove(sampleBuffer), trackID, latestByteRangeOffset);
1403}
