122 class TrackData {
123 WTF_MAKE_FAST_ALLOCATED;
124 public:
125 static auto create(CodecType codecType, const webm::TrackEntry& trackEntry, SourceBufferParserWebM& parser) -> UniqueRef<TrackData>
126 {
127 return makeUniqueRef<TrackData>(codecType, trackEntry, Type::Unknown, parser);
128 }
129
130 enum class Type {
131 Unknown,
132 Audio,
133 Video,
134 };
135
136 TrackData(CodecType codecType, const webm::TrackEntry& trackEntry, Type trackType, SourceBufferParserWebM& parser)
137 : m_codec { codecType }
138 , m_track { webm::TrackEntry { trackEntry } }
139 , m_trackType { trackType }
140 , m_parser { parser }
141 {
142 }
143 virtual ~TrackData() = default;
144
145 CodecType codec() const { return m_codec; }
146 webm::TrackEntry& track() { return m_track; }
147 Type trackType() const { return m_trackType; }
148
149 RetainPtr<CMFormatDescriptionRef> formatDescription() { return m_formatDescription; }
150 void setFormatDescription(RetainPtr<CMFormatDescriptionRef>&& description) { m_formatDescription = WTFMove(description); }
151
152 SourceBufferParserWebM& parser() const { return m_parser; }
153
154 virtual webm::Status consumeFrameData(webm::Reader&, const webm::FrameMetadata&, uint64_t*, const CMTime&, int)
155 {
156 ASSERT_NOT_REACHED();
157 return webm::Status(webm::Status::kInvalidElementId);
158 }
159
160 virtual void reset()
161 {
162 m_currentPacketSize = std::nullopt;
163 m_partialBytesRead = 0;
164 }
165
166 protected:
167 std::optional<size_t> m_currentPacketSize;
168 // Size of the currently parsed packet, possibly incomplete.
169 size_t m_partialBytesRead { 0 };
170
171 private:
172 CodecType m_codec;
173 webm::TrackEntry m_track;
174 Type m_trackType;
175 RetainPtr<CMFormatDescriptionRef> m_formatDescription;
176 SourceBufferParserWebM& m_parser;
177 };
