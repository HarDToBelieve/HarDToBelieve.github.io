47RetainPtr<CTFontRef> SystemFontDatabaseCoreText::createSystemUIFont(const CascadeListParameters& parameters, CFStringRef locale)
48{
49 // Work around a quirk of the platform API.
50 // If the passed string is empty instead of null,
51 // CoreText doesn't use the system's locale instead.
52 // We need to use the system locale in this case.
53 if (locale && !CFStringGetLength(locale))
54 locale = nullptr;
55 auto result = adoptCF(CTFontCreateUIFontForLanguage(kCTFontUIFontSystem, parameters.size, locale));
56 ASSERT(result);
57 return createFontByApplyingWeightWidthItalicsAndFallbackBehavior(result.get(), parameters.weight, parameters.width, parameters.italic, parameters.size, parameters.allowUserInstalledFonts);
58}

61RetainPtr<CTFontRef> SystemFontDatabaseCoreText::createSystemDesignFont(SystemFontKind systemFontKind, const CascadeListParameters& parameters)
62{
63 CFStringRef design = kCTFontUIFontDesignDefault;
64 switch (systemFontKind) {
65 case SystemFontKind::UISerif:
66 design = kCTFontUIFontDesignSerif;
67 break;
68 case SystemFontKind::UIMonospace:
69 design = kCTFontUIFontDesignMonospaced;
70 break;
71 case SystemFontKind::UIRounded:
72 design = kCTFontUIFontDesignRounded;
73 break;
74 default:
75 ASSERT_NOT_REACHED();
76 }
77 return createFontByApplyingWeightWidthItalicsAndFallbackBehavior(nullptr, parameters.weight, parameters.width, parameters.italic, parameters.size, parameters.allowUserInstalledFonts, design);
78}

81RetainPtr<CTFontRef> SystemFontDatabaseCoreText::createTextStyleFont(const CascadeListParameters& parameters)
82{
83 RetainPtr<CFStringRef> localeString = parameters.locale.isEmpty() ? nullptr : parameters.locale.string().createCFString();
84 auto descriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(parameters.fontName.string().createCFString().get(), RenderThemeCocoa::singleton().contentSizeCategory(), localeString.get()));
85 // FIXME: Use createFontByApplyingWeightWidthItalicsAndFallbackBehavior().
86 CTFontSymbolicTraits traits = (parameters.weight >= kCTFontWeightSemibold ? kCTFontTraitBold : 0)
87#if HAVE(LEVEL_2_SYSTEM_FONT_WIDTH_VALUES) || HAVE(LEVEL_3_SYSTEM_FONT_WIDTH_VALUES)
88 | (parameters.width >= kCTFontWidthSemiExpanded ? kCTFontTraitExpanded : 0)
89 | (parameters.width <= kCTFontWidthSemiCondensed ? kCTFontTraitCondensed : 0)
90#else
91 | (parameters.width >= kCTFontWidthExpanded ? kCTFontTraitExpanded : 0)
92 | (parameters.width <= kCTFontWidthCondensed ? kCTFontTraitCondensed : 0)
93#endif
94 | (parameters.italic ? kCTFontTraitItalic : 0);
95 if (traits)
96 descriptor = adoptCF(CTFontDescriptorCreateCopyWithSymbolicTraits(descriptor.get(), traits, traits));
97 return createFontForInstalledFonts(descriptor.get(), parameters.size, parameters.allowUserInstalledFonts);
98}

100Vector<RetainPtr<CTFontDescriptorRef>> SystemFontDatabaseCoreText::cascadeList(const CascadeListParameters& parameters, SystemFontKind systemFontKind)
101{
102 ASSERT(!parameters.fontName.isNull());
103 return m_systemFontCache.ensure(parameters, [&] {
104 auto localeString = parameters.locale.string().createCFString();
105 RetainPtr<CTFontRef> systemFont;
106 switch (systemFontKind) {
107 case SystemFontKind::SystemUI:
108 systemFont = createSystemUIFont(parameters, localeString.get());
109 break;
110 case SystemFontKind::UISerif:
111 case SystemFontKind::UIMonospace:
112 case SystemFontKind::UIRounded:
113#if HAVE(DESIGN_SYSTEM_UI_FONTS)
114 systemFont = createSystemDesignFont(systemFontKind, parameters);
115#endif
116 break;
117 case SystemFontKind::TextStyle:
118 systemFont = createTextStyleFont(parameters);
119 break;
120 }
121 ASSERT(systemFont);
122 auto result = computeCascadeList(systemFont.get(), localeString.get());
123 ASSERT(!result.isEmpty());
124 return result;
125 }).iterator->value;
126}

138RetainPtr<CTFontRef> SystemFontDatabaseCoreText::createFontByApplyingWeightWidthItalicsAndFallbackBehavior(CTFontRef font, CGFloat weight, CGFloat width, bool italic, float size, AllowUserInstalledFonts allowUserInstalledFonts, CFStringRef design)
139{
140 auto weightNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberCGFloatType, &weight));
141 auto widthNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberCGFloatType, &width));
142 const float systemFontItalicSlope = 0.07;
143 float italicsRawNumber = italic ? systemFontItalicSlope : 0;
144 auto italicsNumber = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberFloatType, &italicsRawNumber));
145 CFTypeRef traitsKeys[] = { kCTFontWeightTrait, kCTFontWidthTrait, kCTFontSlantTrait, kCTFontUIFontDesignTrait };
146 CFTypeRef traitsValues[] = { weightNumber.get(), widthNumber.get(), italicsNumber.get(), design ? static_cast<CFTypeRef>(design) : static_cast<CFTypeRef>(kCTFontUIFontDesignDefault) };
147 auto traitsDictionary = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, traitsKeys, traitsValues, WTF_ARRAY_LENGTH(traitsKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
148 auto attributes = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
149 CFDictionaryAddValue(attributes.get(), kCTFontTraitsAttribute, traitsDictionary.get());
150 addAttributesForInstalledFonts(attributes.get(), allowUserInstalledFonts);
151 auto modification = adoptCF(CTFontDescriptorCreateWithAttributes(attributes.get()));
152 if (font)
153 return adoptCF(CTFontCreateCopyWithAttributes(font, size, nullptr, modification.get()));
154 return adoptCF(CTFontCreateWithFontDescriptor(modification.get(), size, nullptr));
155}

157RetainPtr<CTFontDescriptorRef> SystemFontDatabaseCoreText::removeCascadeList(CTFontDescriptorRef fontDescriptor)
158{
159 auto emptyArray = adoptCF(CFArrayCreate(kCFAllocatorDefault, nullptr, 0, &kCFTypeArrayCallBacks));
160 CFTypeRef fallbackDictionaryKeys[] = { kCTFontCascadeListAttribute };
161 CFTypeRef fallbackDictionaryValues[] = { emptyArray.get() };
162 auto fallbackDictionary = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, fallbackDictionaryKeys, fallbackDictionaryValues, WTF_ARRAY_LENGTH(fallbackDictionaryKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
163 auto modifiedFontDescriptor = adoptCF(CTFontDescriptorCreateCopyWithAttributes(fontDescriptor, fallbackDictionary.get()));
164 return modifiedFontDescriptor;
165}

167Vector<RetainPtr<CTFontDescriptorRef>> SystemFontDatabaseCoreText::computeCascadeList(CTFontRef font, CFStringRef locale)
168{
169 CFTypeRef arrayValues[] = { locale };
170 auto localeArray = adoptCF(CFArrayCreate(kCFAllocatorDefault, arrayValues, WTF_ARRAY_LENGTH(arrayValues), &kCFTypeArrayCallBacks));
171 auto cascadeList = adoptCF(CTFontCopyDefaultCascadeListForLanguages(font, localeArray.get()));
172 Vector<RetainPtr<CTFontDescriptorRef>> result;
173 // WebKit handles the cascade list, and WebKit 2's IPC code doesn't know how to serialize Core Text's cascade list.
174 result.append(removeCascadeList(adoptCF(CTFontCopyFontDescriptor(font)).get()));
175 if (cascadeList) {
176 CFIndex arrayLength = CFArrayGetCount(cascadeList.get());
177 for (CFIndex i = 0; i < arrayLength; ++i)
178 result.append(static_cast<CTFontDescriptorRef>(CFArrayGetValueAtIndex(cascadeList.get(), i)));
179 }
180 return result;
181}

183static CGFloat mapWeight(FontSelectionValue weight)
184{
185 if (weight < FontSelectionValue(150))
186 return kCTFontWeightUltraLight;
187 if (weight < FontSelectionValue(250))
188 return kCTFontWeightThin;
189 if (weight < FontSelectionValue(350))
190 return kCTFontWeightLight;
191 if (weight < FontSelectionValue(450))
192 return kCTFontWeightRegular;
193 if (weight < FontSelectionValue(550))
194 return kCTFontWeightMedium;
195 if (weight < FontSelectionValue(650))
196 return kCTFontWeightSemibold;
197 if (weight < FontSelectionValue(750))
198 return kCTFontWeightBold;
199 if (weight < FontSelectionValue(850))
200 return kCTFontWeightHeavy;
201 return kCTFontWeightBlack;
202}

204static CGFloat mapWidth(FontSelectionValue width)
205{
206 struct {
207 FontSelectionValue input;
208 CGFloat output;
209 } piecewisePoints[] = {
210#if HAVE(LEVEL_3_SYSTEM_FONT_WIDTH_VALUES)
211 {FontSelectionValue(37.5f), kCTFontWidthUltraCompressed},
212 {FontSelectionValue(50), kCTFontWidthExtraCompressed}, // ultra condensed
213 {FontSelectionValue(62.5f), kCTFontWidthExtraCondensed},
214 {FontSelectionValue(75), kCTFontWidthCondensed},
215 {FontSelectionValue(87.5f), kCTFontWidthSemiCondensed},
216 {FontSelectionValue(100), kCTFontWidthStandard},
217 {FontSelectionValue(112.5f), kCTFontWidthSemiExpanded},
218 {FontSelectionValue(125), kCTFontWidthExpanded},
219 {FontSelectionValue(150), kCTFontWidthExtraExpanded},
220#elif HAVE(LEVEL_2_SYSTEM_FONT_WIDTH_VALUES)
221 {FontSelectionValue(62.5f), kCTFontWidthExtraCondensed},
222 {FontSelectionValue(75), kCTFontWidthCondensed},
223 {FontSelectionValue(87.5f), kCTFontWidthSemiCondensed},
224 {FontSelectionValue(100), kCTFontWidthStandard},
225 {FontSelectionValue(112.5f), kCTFontWidthSemiExpanded},
226 {FontSelectionValue(125), kCTFontWidthExpanded}
227#else // level 1
228 {FontSelectionValue(75), kCTFontWidthCondensed},
229 {FontSelectionValue(100), kCTFontWidthStandard},
230 {FontSelectionValue(125), kCTFontWidthExpanded}
231#endif
232 };
233 for (size_t i = 0; i < std::size(piecewisePoints) - 1; ++i) {
234 auto& previous = piecewisePoints[i];
235 auto& next = piecewisePoints[i + 1];
236 auto middleInput = (previous.input + next.input) / 2;
237 if (width < middleInput)
238 return previous.output;
239 }
240 return piecewisePoints[std::size(piecewisePoints) - 1].output;
241}

243SystemFontDatabaseCoreText::CascadeListParameters SystemFontDatabaseCoreText::systemFontParameters(const FontDescription& description, const AtomString& familyName, SystemFontKind systemFontKind, AllowUserInstalledFonts allowUserInstalledFonts)
244{
245 CascadeListParameters result;
246 result.locale = description.computedLocale();
247 result.size = description.computedSize();
248 result.italic = isItalic(description.italic());
249 result.allowUserInstalledFonts = allowUserInstalledFonts;
250
251 auto weight = description.weight();
252 if (FontCache::singleton().shouldMockBoldSystemFontForAccessibility())
253 weight = weight + FontSelectionValue(200);
254
255 result.weight = mapWeight(weight);
256 result.width = mapWidth(description.stretch());
257
258 switch (systemFontKind) {
259 case SystemFontKind::SystemUI: {
260 static MainThreadNeverDestroyed<const AtomString> systemUI = AtomString("system-ui", AtomString::ConstructFromLiteral);
261 result.fontName = systemUI.get();
262 break;
263 }
264 case SystemFontKind::UISerif: {
265 static MainThreadNeverDestroyed<const AtomString> systemUISerif = AtomString("ui-serif", AtomString::ConstructFromLiteral);
266 result.fontName = systemUISerif.get();
267 break;
268 }
269 case SystemFontKind::UIMonospace: {
270 static MainThreadNeverDestroyed<const AtomString> systemUIMonospace = AtomString("ui-monospace", AtomString::ConstructFromLiteral);
271 result.fontName = systemUIMonospace.get();
272 break;
273 }
274 case SystemFontKind::UIRounded: {
275 static MainThreadNeverDestroyed<const AtomString> systemUIRounded = AtomString("ui-rounded", AtomString::ConstructFromLiteral);
276 result.fontName = systemUIRounded.get();
277 break;
278 }
279 case SystemFontKind::TextStyle:
280 result.fontName = familyName;
281 break;
282 }
283
284 return result;
285}

292static String genericFamily(const String& locale, HashMap<String, String>& map, CFStringRef ctKey)
293{
294 return map.ensure(locale, [&] {
295 auto descriptor = adoptCF(CTFontDescriptorCreateForCSSFamily(ctKey, locale.createCFString().get()));
296 auto value = adoptCF(checked_cf_cast<CFStringRef>(CTFontDescriptorCopyAttribute(descriptor.get(), kCTFontFamilyNameAttribute)));
297 return String { value.get() };
298 }).iterator->value;
299}

321String SystemFontDatabaseCoreText::monospaceFamily(const String& locale)
322{
323 auto result = genericFamily(locale, m_monospaceFamilies, kCTFontCSSFamilyMonospace);
324#if PLATFORM(MAC) && ENABLE(MONOSPACE_FONT_EXCEPTION)
325 // In general, CoreText uses Monaco for monospaced (see: Terminal.app and Xcode.app).
326 // For now, we want to use Courier for web compatibility, until we have more time to do compatibility testing.
327 if (equalLettersIgnoringASCIICase(result, "monaco"))
328 return "Courier"_str;
329#elif PLATFORM(IOS_FAMILY) && ENABLE(MONOSPACE_FONT_EXCEPTION)
330 if (equalLettersIgnoringASCIICase(result, "courier new"))
331 return "Courier"_str;
332#endif
333 return result;
334}
