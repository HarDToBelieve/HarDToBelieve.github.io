37enum class SystemFontKind : uint8_t {
38 SystemUI,
39 UISerif,
40 UIMonospace,
41 UIRounded,
42 TextStyle
43};
44
45class SystemFontDatabaseCoreText {
46public:
47 struct CascadeListParameters {
48 CascadeListParameters()
49 {
50 }
51
52 CascadeListParameters(WTF::HashTableDeletedValueType)
53 : fontName(WTF::HashTableDeletedValue)
54 {
55 }
56
57 bool isHashTableDeletedValue() const
58 {
59 return fontName.isHashTableDeletedValue();
60 }
61
62 bool operator==(const CascadeListParameters& other) const
63 {
64 return fontName == other.fontName
65 && locale == other.locale
66 && weight == other.weight
67 && width == other.width
68 && size == other.size
69 && allowUserInstalledFonts == other.allowUserInstalledFonts
70 && italic == other.italic;
71 }
72
73 unsigned hash() const
74 {
75 return computeHash(fontName, locale, weight, width, size, allowUserInstalledFonts, italic);
76 }
77
78 struct Hash {
79 static unsigned hash(const CascadeListParameters& parameters) { return parameters.hash(); }
80 static bool equal(const CascadeListParameters& a, const CascadeListParameters& b) { return a == b; }
81 static const bool safeToCompareToEmptyOrDeleted = true;
82 };
83
84 AtomString fontName;
85 AtomString locale;
86 CGFloat weight { 0 };
87 CGFloat width { 0 };
88 float size { 0 };
89 AllowUserInstalledFonts allowUserInstalledFonts { AllowUserInstalledFonts::No };
90 bool italic { false };
91 };
92
93 static SystemFontDatabaseCoreText& singleton();
94
95 Vector<RetainPtr<CTFontDescriptorRef>> cascadeList(const FontDescription&, const AtomString& cssFamily, SystemFontKind, AllowUserInstalledFonts);
96
97 String serifFamily(const String& locale);
98 String sansSerifFamily(const String& locale);
99 String cursiveFamily(const String& locale);
100 String fantasyFamily(const String& locale);
101 String monospaceFamily(const String& locale);
102
103 void clear();
104
105private:
106 SystemFontDatabaseCoreText();
107
108 Vector<RetainPtr<CTFontDescriptorRef>> cascadeList(const CascadeListParameters&, SystemFontKind);
109
110 RetainPtr<CTFontRef> createSystemUIFont(const CascadeListParameters&, CFStringRef locale);
111 RetainPtr<CTFontRef> createSystemDesignFont(SystemFontKind, const CascadeListParameters&);
112 RetainPtr<CTFontRef> createTextStyleFont(const CascadeListParameters&);
113
114 static RetainPtr<CTFontRef> createFontByApplyingWeightWidthItalicsAndFallbackBehavior(CTFontRef, CGFloat weight, CGFloat width, bool italic, float size, AllowUserInstalledFonts, CFStringRef design = nullptr);
115 static RetainPtr<CTFontDescriptorRef> removeCascadeList(CTFontDescriptorRef);
116 static Vector<RetainPtr<CTFontDescriptorRef>> computeCascadeList(CTFontRef, CFStringRef locale);
117 static CascadeListParameters systemFontParameters(const FontDescription&, const AtomString& familyName, SystemFontKind, AllowUserInstalledFonts);
118
119 HashMap<CascadeListParameters, Vector<RetainPtr<CTFontDescriptorRef>>, CascadeListParameters::Hash, SimpleClassHashTraits<CascadeListParameters>> m_systemFontCache;
120
121 HashMap<String, String> m_serifFamilies;
122 HashMap<String, String> m_sansSeriferifFamilies;
123 HashMap<String, String> m_cursiveFamilies;
124 HashMap<String, String> m_fantasyFamilies;
125 HashMap<String, String> m_monospaceFamilies;
126};
