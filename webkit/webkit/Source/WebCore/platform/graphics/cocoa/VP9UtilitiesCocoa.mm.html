77enum class ResolutionCategory : uint8_t {
78 R_480p,
79 R_720p,
80 R_1080p,
81 R_2K,
82 R_4K,
83 R_8K,
84 R_12K,
85};
86
87static ResolutionCategory resolutionCategory(const FloatSize& size)
88{
89 auto pixels = size.area();
90 if (pixels > 7680 * 4320)
91 return ResolutionCategory::R_12K;
92 if (pixels > 4096 * 2160)
93 return ResolutionCategory::R_8K;
94 if (pixels > 2048 * 1080)
95 return ResolutionCategory::R_4K;
96 if (pixels > 1920 * 1080)
97 return ResolutionCategory::R_2K;
98 if (pixels > 1280 * 720)
99 return ResolutionCategory::R_1080p;
100 if (pixels > 640 * 480)
101 return ResolutionCategory::R_720p;
102 return ResolutionCategory::R_480p;
103}

150static bool isVP9CodecConfigurationRecordSupported(const VPCodecConfigurationRecord& codecConfiguration)
151{
152 if (!isVP9DecoderAvailable())
153 return false;
154
155 // HW & SW VP9 Decoders support Profile 0 & 2:
156 if (codecConfiguration.profile && codecConfiguration.profile != 2)
157 return false;
158
159 // HW & SW VP9 Decoders support only 420 chroma subsampling:
160 if (codecConfiguration.chromaSubsampling > VPConfigurationChromaSubsampling::Subsampling_420_Colocated)
161 return false;
162
163 // HW & SW VP9 Decoders support 8 or 10 bit color:
164 if (codecConfiguration.bitDepth > 10)
165 return false;
166
167 // HW & SW VP9 Decoders support up to Level 6:
168 if (codecConfiguration.level > VPConfigurationLevel::Level_6)
169 return false;
170
171 // Hardware decoders are always available.
172 if (vp9HardwareDecoderAvailable())
173 return true;
174
175 // For wall-powered devices, always report VP9 as supported, even if not powerEfficient.
176 if (!systemHasBattery())
177 return true;
178
179 // For battery-powered devices, always report VP9 as supported when running on AC power,
180 // but only on battery when there is an attached screen whose resolution is large enough
181 // to support 4K video.
182 if (systemHasAC())
183 return true;
184
185 bool has4kScreen = false;
186 if (auto overrideForTesting = VP9TestingOverrides::singleton().vp9ScreenSizeAndScale()) {
187 auto screenSize = FloatSize(overrideForTesting->width, overrideForTesting->height).scaled(overrideForTesting->scale);
188 has4kScreen = resolutionCategory(screenSize) >= ResolutionCategory::R_4K;
189 } else {
190 for (auto& screenData : getScreenProperties().screenDataMap.values()) {
191 if (resolutionCategory(screenData.screenRect.size().scaled(screenData.scaleFactor)) >= ResolutionCategory::R_4K) {
192 has4kScreen = true;
193 break;
194 }
195 }
196 }
197
198 return has4kScreen;
199}

201static bool isVP8CodecConfigurationRecordSupported(const VPCodecConfigurationRecord& codecConfiguration)
202{
203 if (!isVP8DecoderAvailable())
204 return false;
205
206 // VP8 decoder only supports Profile 0;
207 if (codecConfiguration.profile)
208 return false;
209
210 // VP8 decoder only supports 420 chroma subsampling.
211 if (codecConfiguration.chromaSubsampling > VPConfigurationChromaSubsampling::Subsampling_420_Colocated)
212 return false;
213
214 // VP8 decoder only supports 8-bit color:
215 if (codecConfiguration.bitDepth > 8)
216 return false;
217
218 return true;
219}

221bool isVPCodecConfigurationRecordSupported(const VPCodecConfigurationRecord& codecConfiguration)
222{
223 if (codecConfiguration.codecName == "vp08" || codecConfiguration.codecName == "vp8")
224 return isVP8CodecConfigurationRecordSupported(codecConfiguration);
225
226 if (codecConfiguration.codecName == "vp09" || codecConfiguration.codecName == "vp9")
227 return isVP9CodecConfigurationRecordSupported(codecConfiguration);
228
229 return false;
230}

232std::optional<MediaCapabilitiesInfo> validateVPParameters(const VPCodecConfigurationRecord& codecConfiguration, const VideoConfiguration& videoConfiguration)
233{
234 if (!isVPCodecConfigurationRecordSupported(codecConfiguration))
235 return std::nullopt;
236
237 // VideoConfiguration and VPCodecConfigurationRecord can have conflicting values for HDR properties. If so, reject.
238 if (videoConfiguration.transferFunction) {
239 // Note: Transfer Characteristics are defined by ISO/IEC 23091-2:2019.
240 if (*videoConfiguration.transferFunction == TransferFunction::SRGB && codecConfiguration.transferCharacteristics > 15)
241 return std::nullopt;
242 if (*videoConfiguration.transferFunction == TransferFunction::PQ && codecConfiguration.transferCharacteristics != 16)
243 return std::nullopt;
244 if (*videoConfiguration.transferFunction == TransferFunction::HLG && codecConfiguration.transferCharacteristics != 18)
245 return std::nullopt;
246 }
247
248 if (videoConfiguration.colorGamut) {
249 if (*videoConfiguration.colorGamut == ColorGamut::Rec2020 && codecConfiguration.colorPrimaries != 9)
250 return std::nullopt;
251 }
252
253 MediaCapabilitiesInfo info;
254
255 if (vp9HardwareDecoderAvailable()) {
256 // HW VP9 Decoder does not support alpha channel:
257 if (videoConfiguration.alphaChannel && *videoConfiguration.alphaChannel)
258 return std::nullopt;
259
260 // HW VP9 Decoder can support up to 4K @ 120 or 8K @ 30
261 auto resolution = resolutionCategory({ (float)videoConfiguration.width, (float)videoConfiguration.height });
262 if (resolution > ResolutionCategory::R_8K)
263 return std::nullopt;
264 if (resolution == ResolutionCategory::R_8K && videoConfiguration.framerate > 30)
265 info.smooth = false;
266 else if (resolution <= ResolutionCategory::R_4K && videoConfiguration.framerate > 120)
267 info.smooth = false;
268 else
269 info.smooth = true;
270
271 info.powerEfficient = true;
272 info.supported = true;
273 return info;
274 }
275
276 info.powerEfficient = false;
277
278 // SW VP9 Decoder has much more variable capabilities depending on CPU characteristics.
279 // FIXME: Add a lookup table for device-to-capabilities. For now, assume that the SW VP9
280 // decoder can support 4K @ 30.
281 if (videoConfiguration.height <= 1080 && videoConfiguration.framerate > 60)
282 info.smooth = false;
283 if (videoConfiguration.height <= 2160 && videoConfiguration.framerate > 30)
284 info.smooth = false;
285 else
286 info.smooth = true;
287
288 // For wall-powered devices, always report VP9 as supported, even if not powerEfficient.
289 if (!systemHasBattery()) {
290 info.supported = true;
291 return info;
292 }
293
294 // For battery-powered devices, always report VP9 as supported when running on AC power,
295 // but only on battery when there is an attached screen whose resolution is large enough
296 // to support 4K video.
297 if (systemHasAC()) {
298 info.supported = true;
299 return info;
300 }
301
302 bool has4kScreen = false;
303
304 if (auto overrideForTesting = VP9TestingOverrides::singleton().vp9ScreenSizeAndScale()) {
305 auto screenSize = FloatSize(overrideForTesting->width, overrideForTesting->height).scaled(overrideForTesting->scale);
306 has4kScreen = resolutionCategory(screenSize) >= ResolutionCategory::R_4K;
307 } else {
308 for (auto& screenData : getScreenProperties().screenDataMap.values()) {
309 if (resolutionCategory(screenData.screenRect.size().scaled(screenData.scaleFactor)) >= ResolutionCategory::R_4K) {
310 has4kScreen = true;
311 break;
312 }
313 }
314 }
315
316 if (!has4kScreen)
317 return std::nullopt;
318
319 info.supported = true;
320 return info;
321}

482static uint8_t convertSubsamplingXYToChromaSubsampling(uint64_t x, uint64_t y)
483{
484 if (x & y)
485 return VPConfigurationChromaSubsampling::Subsampling_420_Colocated;
486 if (x & !y)
487 return VPConfigurationChromaSubsampling::Subsampling_422;
488 if (!x & !y)
489 return VPConfigurationChromaSubsampling::Subsampling_444;
490 // This indicates 4:4:0 subsampling, which is not expressable in the 'vpcC' box. Default to 4:2:0.
491 return VPConfigurationChromaSubsampling::Subsampling_420_Colocated;
492}

494static RetainPtr<CMFormatDescriptionRef> createFormatDescriptionFromVPCodecConfigurationRecord(const VPCodecConfigurationRecord& record, int32_t width, int32_t height)
495{
496 // Ref: "VP Codec ISO Media File Format Binding, v1.0, 2017-03-31"
497 // <https://www.webmproject.org/vp9/mp4/>
498 //
499 // class VPCodecConfigurationBox extends FullBox('vpcC', version = 1, 0)
500 // {
501 // VPCodecConfigurationRecord() vpcConfig;
502 // }
503 //
504 // aligned (8) class VPCodecConfigurationRecord {
505 // unsigned int (8) profile;
506 // unsigned int (8) level;
507 // unsigned int (4) bitDepth;
508 // unsigned int (3) chromaSubsampling;
509 // unsigned int (1) videoFullRangeFlag;
510 // unsigned int (8) colourPrimaries;
511 // unsigned int (8) transferCharacteristics;
512 // unsigned int (8) matrixCoefficients;
513 // unsigned int (16) codecIntializationDataSize;
514 // unsigned int (8)[] codecIntializationData;
515 // }
516 //
517 // codecIntializationDataSize​For VP8 and VP9 this field must be 0.
518 // codecIntializationData​binary codec initialization data. Not used for VP8 and VP9.
519 //
520 // FIXME: Convert existing struct to an ISOBox and replace the writing code below
521 // with a subclass of ISOFullBox.
522
523 constexpr size_t VPCodecConfigurationContentsSize = 12;
524
525 uint32_t versionAndFlags = 1 << 24;
526 uint8_t bitDepthChromaAndRange = (0xF & record.bitDepth) << 4 | (0x7 & record.chromaSubsampling) << 1 | (0x1 & record.videoFullRangeFlag);
527 uint16_t codecIntializationDataSize = 0;
528
529 auto view = JSC::DataView::create(ArrayBuffer::create(VPCodecConfigurationContentsSize, 1), 0, VPCodecConfigurationContentsSize);
530 view->set(0, versionAndFlags, false);
531 view->set(4, record.profile, false);
532 view->set(5, record.level, false);
533 view->set(6, bitDepthChromaAndRange, false);
534 view->set(7, record.colorPrimaries, false);
535 view->set(8, record.transferCharacteristics, false);
536 view->set(9, record.matrixCoefficients, false);
537 view->set(10, codecIntializationDataSize, false);
538
539 auto data = adoptCF(CFDataCreate(kCFAllocatorDefault, (const UInt8 *)view->data(), view->byteLength()));
540
541 CFTypeRef configurationKeys[] = { CFSTR("vpcC") };
542 CFTypeRef configurationValues[] = { data.get() };
543 auto configurationDict = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, configurationKeys, configurationValues, WTF_ARRAY_LENGTH(configurationKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
544
545 Vector<CFTypeRef> extensionsKeys { PAL::kCMFormatDescriptionExtension_SampleDescriptionExtensionAtoms };
546 Vector<CFTypeRef> extensionsValues = { configurationDict.get() };
547
548 if (record.videoFullRangeFlag == VPConfigurationRange::FullRange) {
549 extensionsKeys.append(PAL::kCMFormatDescriptionExtension_FullRangeVideo);
550 extensionsValues.append(kCFBooleanTrue);
551 }
552
553 if (auto cmColorPrimaries = convertToCMColorPrimaries(record.colorPrimaries)) {
554 extensionsKeys.append(kCVImageBufferColorPrimariesKey);
555 extensionsValues.append(cmColorPrimaries);
556 }
557
558 if (auto cmTransferFunction = convertToCMTransferFunction(record.transferCharacteristics)) {
559 extensionsKeys.append(kCVImageBufferTransferFunctionKey);
560 extensionsValues.append(cmTransferFunction);
561 }
562
563 if (auto cmMatrix = convertToCMYCbCRMatrix(record.matrixCoefficients)) {
564 extensionsKeys.append(kCVImageBufferYCbCrMatrixKey);
565 extensionsValues.append(cmMatrix);
566 }
567
568 auto extensions = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, extensionsKeys.data(), extensionsValues.data(), extensionsKeys.size(), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
569
570 CMVideoFormatDescriptionRef formatDescription = nullptr;
571 CMVideoCodecType codec = record.codecName == "vp09" ? kCMVideoCodecType_VP9 : 'vp08';
572 if (noErr != CMVideoFormatDescriptionCreate(kCFAllocatorDefault, codec, width, height, extensions.get(), &formatDescription))
573 return nullptr;
574 return adoptCF(formatDescription);
575}

577RetainPtr<CMFormatDescriptionRef> createFormatDescriptionFromVP9HeaderParser(const vp9_parser::Vp9HeaderParser& parser, const webm::Element<Colour>& color)
578{
579 VPCodecConfigurationRecord record;
580
581 record.codecName = "vp09";
582 record.profile = parser.profile();
583 // CoreMedia does nat care about the VP9 codec level; hard-code to Level 1.0 here:
584 record.level = 10;
585 record.bitDepth = parser.bit_depth();
586 record.videoFullRangeFlag = parser.color_range() ? VPConfigurationRange::FullRange : VPConfigurationRange::VideoRange;
587 record.chromaSubsampling = convertSubsamplingXYToChromaSubsampling(parser.subsampling_x(), parser.subsampling_y());
588 record.colorPrimaries = VPConfigurationColorPrimaries::Unspecified;
589 record.transferCharacteristics = VPConfigurationTransferCharacteristics::Unspecified;
590 record.matrixCoefficients = VPConfigurationMatrixCoefficients::Unspecified;
591
592 // Container color values can override per-sample ones:
593 if (color.is_present()) {
594 auto& colorValue = color.value();
595 if (colorValue.chroma_subsampling_x.is_present() && colorValue.chroma_subsampling_y.is_present())
596 record.chromaSubsampling = convertSubsamplingXYToChromaSubsampling(colorValue.chroma_subsampling_x.value(), colorValue.chroma_subsampling_y.value());
597 if (colorValue.range.is_present() && colorValue.range.value() != Range::kUnspecified)
598 record.videoFullRangeFlag = colorValue.range.value() == Range::kFull ? VPConfigurationRange::FullRange : VPConfigurationRange::VideoRange;
599 if (colorValue.bits_per_channel.is_present())
600 record.bitDepth = colorValue.bits_per_channel.value();
601 if (colorValue.transfer_characteristics.is_present())
602 record.transferCharacteristics = convertToTransferCharacteristics(colorValue.transfer_characteristics.value());
603 if (colorValue.matrix_coefficients.is_present())
604 record.matrixCoefficients = convertToMatrixCoefficients(colorValue.matrix_coefficients.value());
605 if (colorValue.primaries.is_present())
606 record.colorPrimaries = convertToColorPrimaries(colorValue.primaries.value());
607 }
608
609 return createFormatDescriptionFromVPCodecConfigurationRecord(record, parser.width(), parser.height());
610}

612std::optional<VP8FrameHeader> parseVP8FrameHeader(uint8_t* frameData, size_t frameSize)
613{
614 // VP8 frame headers are defined in RFC 6386: <https://tools.ietf.org/html/rfc6386>.
615
616 // Bail if the header is below a minimum size
617 if (frameSize < 11)
618 return std::nullopt;
619
620 VP8FrameHeader header;
621 size_t headerSize = 11;
622
623 auto view = JSC::DataView::create(ArrayBuffer::create(frameData, headerSize), 0, headerSize);
624 bool status = true;
625
626 auto uncompressedChunk = view->get<uint32_t>(0, true, &status);
627 if (!status)
628 return std::nullopt;
629
630 header.keyframe = uncompressedChunk & 0x80000000;
631 header.version = (uncompressedChunk >> 28) & 0x7;
632 header.showFrame = uncompressedChunk & 0x8000000;
633 header.partitionSize = (uncompressedChunk >> 8) & 0x7FFFF;
634
635 if (!header.keyframe)
636 return header;
637
638 auto startCode0 = view->get<uint8_t>(3, true, &status);
639 if (!status || startCode0 != 0x9d)
640 return std::nullopt;
641
642 auto startCode1 = view->get<uint8_t>(4, true, &status);
643 if (!status || startCode1 != 0x01)
644 return std::nullopt;
645
646 auto startCode2 = view->get<uint8_t>(5, true, &status);
647 if (!status || startCode2 != 0x2a)
648 return std::nullopt;
649
650 auto horizontalAndWidthField = view->get<uint16_t>(6, true, &status);
651 if (!status)
652 return std::nullopt;
653
654 header.horizontalScale = static_cast<uint8_t>(horizontalAndWidthField >> 14);
655 header.width = static_cast<uint16_t>(horizontalAndWidthField & 0x3FFF);
656
657 auto verticalAndHeightField = view->get<uint16_t>(8, true, &status);
658 if (!status)
659 return std::nullopt;
660
661 header.verticalScale = static_cast<uint8_t>(verticalAndHeightField >> 14);
662 header.height = static_cast<uint16_t>(verticalAndHeightField & 0x3FFF);
663
664 auto colorSpaceAndClampingField = view->get<uint8_t>(10, true, &status);
665 if (!status)
666 return std::nullopt;
667
668 header.colorSpace = colorSpaceAndClampingField & 0x80;
669 header.needsClamping = colorSpaceAndClampingField & 0x40;
670
671 return header;
672}

674RetainPtr<CMFormatDescriptionRef> createFormatDescriptionFromVP8Header(const VP8FrameHeader& header, const webm::Element<Colour>& color)
675{
676 VPCodecConfigurationRecord record;
677 record.codecName = "vp08";
678 record.profile = 0;
679 record.level = 10;
680 record.bitDepth = 8;
681 record.videoFullRangeFlag = VPConfigurationRange::FullRange;
682 record.chromaSubsampling = VPConfigurationChromaSubsampling::Subsampling_420_Colocated;
683 record.colorPrimaries = header.colorSpace ? VPConfigurationColorPrimaries::Unspecified : VPConfigurationColorPrimaries::BT_601_7;
684 record.transferCharacteristics = header.colorSpace ? VPConfigurationTransferCharacteristics::Unspecified : VPConfigurationTransferCharacteristics::BT_601_7;
685 record.matrixCoefficients = header.colorSpace ? VPConfigurationMatrixCoefficients::Unspecified : VPConfigurationMatrixCoefficients::BT_601_7;
686
687 // Container color values can override per-sample ones:
688 if (color.is_present()) {
689 auto& colorValue = color.value();
690 if (colorValue.chroma_subsampling_x.is_present() && colorValue.chroma_subsampling_y.is_present())
691 record.chromaSubsampling = convertSubsamplingXYToChromaSubsampling(colorValue.chroma_subsampling_x.value(), colorValue.chroma_subsampling_y.value());
692 if (colorValue.range.is_present() && colorValue.range.value() != Range::kUnspecified)
693 record.videoFullRangeFlag = colorValue.range.value() == Range::kFull ? VPConfigurationRange::FullRange : VPConfigurationRange::VideoRange;
694 if (colorValue.bits_per_channel.is_present())
695 record.bitDepth = colorValue.bits_per_channel.value();
696 if (colorValue.transfer_characteristics.is_present())
697 record.transferCharacteristics = convertToTransferCharacteristics(colorValue.transfer_characteristics.value());
698 if (colorValue.matrix_coefficients.is_present())
699 record.matrixCoefficients = convertToMatrixCoefficients(colorValue.matrix_coefficients.value());
700 if (colorValue.primaries.is_present())
701 record.colorPrimaries = convertToColorPrimaries(colorValue.primaries.value());
702 }
703
704 return createFormatDescriptionFromVPCodecConfigurationRecord(record, header.width, header.height);
705}
