36class WebGLLayerSwapChain final : public WebCore::GraphicsContextGLIOSurfaceSwapChain {
37public:
38 explicit WebGLLayerSwapChain(WebGLLayer* layer) : m_layer(layer) { };
39 ~WebGLLayerSwapChain() override = default;
40 void present(Buffer&&) override;
41
42 WebCore::IOSurface* displaySurface() { return m_displayBuffer.surface.get(); }
43private:
44 WebGLLayer* const m_layer;
45};
46
47@implementation WebGLLayer {
48 BOOL _preparedForDisplay;
49 std::optional<WebGLLayerSwapChain> _swapChain;
50}
51
52- (id)initWithDevicePixelRatio:(float)devicePixelRatio contentsOpaque:(bool)contentsOpaque
53{
54 self = [super init];
55 self.transform = CATransform3DIdentity;
56 self.contentsOpaque = contentsOpaque;
57 self.contentsScale = devicePixelRatio;
58 _swapChain.emplace(self);
59 return self;
60}
61
62// When using an IOSurface as layer contents, we need to flip the
63// layer to take into account that the IOSurface provides content
64// in Y-up. This means that any incoming transform (unlikely, since
65// this is a contents layer) and anchor point must add a Y scale of
66// -1 and make sure the transform happens from the top.
67
68- (void)setTransform:(CATransform3D)t
69{
70 [super setTransform:CATransform3DScale(t, 1, -1, 1)];
71}
72
73- (void)setAnchorPoint:(CGPoint)p
74{
75 [super setAnchorPoint:CGPointMake(p.x, 1.0 - p.y)];
76}
77
78- (CGImageRef)copyImageSnapshotWithColorSpace:(CGColorSpaceRef)colorSpace
79{
80 // FIXME: implement. https://bugs.webkit.org/show_bug.cgi?id=217377
81 // When implementing, remember to use self.contentsScale.
82 UNUSED_PARAM(colorSpace);
83 return nullptr;
84}
85- (WebCore::GraphicsContextGLIOSurfaceSwapChain&) swapChain
86{
87 return _swapChain.value();
88}
89
90- (void)prepareForDisplay
91{
92 [self setNeedsDisplay];
93 _preparedForDisplay = YES;
94}
95
96- (void)display
97{
98 if (_swapChain->displaySurface() && _preparedForDisplay) {
99 self.contents = _swapChain->displaySurface()->asLayerContents();
100 [self reloadValueForKeyPath:@"contents"];
101 }
102 auto layer = WebCore::PlatformCALayer::platformCALayerForLayer((__bridge void*)self);
103 if (layer && layer->owner())
104 layer->owner()->platformCALayerLayerDidDisplay(layer.get());
105
106 _preparedForDisplay = NO;
107}
108
109@end
110
111void WebGLLayerSwapChain::present(Buffer&& buffer)
112{
113 ASSERT(!m_spareBuffer.surface);
114 GraphicsContextGLIOSurfaceSwapChain::present(WTFMove(buffer));
115 [m_layer prepareForDisplay];
116}
