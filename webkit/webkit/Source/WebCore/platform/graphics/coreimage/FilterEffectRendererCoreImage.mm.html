61static bool isNullOrLinearComponentTransferFunction(const FEComponentTransfer& effect)
62{
63 auto isNullOrLinear = [] (const ComponentTransferFunction& function) {
64 return function.type == FECOMPONENTTRANSFER_TYPE_UNKNOWN
65 || function.type == FECOMPONENTTRANSFER_TYPE_LINEAR;
66 };
67 return isNullOrLinear(effect.redFunction()) && isNullOrLinear(effect.greenFunction())
68 && isNullOrLinear(effect.blueFunction()) && isNullOrLinear(effect.alphaFunction());
69}

71bool FilterEffectRendererCoreImage::supportsCoreImageRendering(FilterEffect& effect)
72{
73 // FIXME: change return value to true once they are implemented
74 switch (effect.filterEffectClassType()) {
75 case FilterEffect::Type::SourceGraphic:
76 return true;
77
78 case FilterEffect::Type::ColorMatrix: {
79 switch (downcast<FEColorMatrix>(effect).type()) {
80 case FECOLORMATRIX_TYPE_UNKNOWN:
81 case FECOLORMATRIX_TYPE_LUMINANCETOALPHA:
82 return false;
83 case FECOLORMATRIX_TYPE_MATRIX:
84 case FECOLORMATRIX_TYPE_SATURATE:
85 case FECOLORMATRIX_TYPE_HUEROTATE:
86 return true;
87 }
88 }
89
90 case FilterEffect::Type::ComponentTransfer:
91 return isNullOrLinearComponentTransferFunction(downcast<FEComponentTransfer>(effect));
92
93 case FilterEffect::Type::Blend:
94 case FilterEffect::Type::Composite:
95 case FilterEffect::Type::ConvolveMatrix:
96 case FilterEffect::Type::DiffuseLighting:
97 case FilterEffect::Type::DisplacementMap:
98 case FilterEffect::Type::DropShadow:
99 case FilterEffect::Type::Flood:
100 case FilterEffect::Type::GaussianBlur:
101 case FilterEffect::Type::Image:
102 case FilterEffect::Type::Lighting:
103 case FilterEffect::Type::Merge:
104 case FilterEffect::Type::Morphology:
105 case FilterEffect::Type::Offset:
106 case FilterEffect::Type::SpecularLighting:
107 case FilterEffect::Type::Tile:
108 case FilterEffect::Type::Turbulence:
109 case FilterEffect::Type::SourceAlpha:
110 return false;
111 }
112 return false;
113}

115void FilterEffectRendererCoreImage::applyEffects(FilterEffect& lastEffect)
116{
117 m_outputImage = connectCIFilters(lastEffect);
118 if (!m_outputImage)
119 return;
120 renderToImageBuffer(lastEffect);

123RetainPtr<CIImage> FilterEffectRendererCoreImage::connectCIFilters(FilterEffect& effect)
124{
125 Vector<RetainPtr<CIImage>> inputImages;
126
127 for (auto in : effect.inputEffects()) {
128 auto inputImage = connectCIFilters(*in);
129 if (!inputImage)
130 return nullptr;
131 inputImages.append(inputImage);
132 }
133 effect.determineAbsolutePaintRect();
134 effect.setResultColorSpace(effect.operatingColorSpace());
135
136 if (effect.absolutePaintRect().isEmpty() || ImageBuffer::sizeNeedsClamping(effect.absolutePaintRect().size()))
137 return nullptr;
138
139 switch (effect.filterEffectClassType()) {
140 case FilterEffect::Type::SourceGraphic:
141 return imageForSourceGraphic(downcast<SourceGraphic>(effect));
142 case FilterEffect::Type::ColorMatrix:
143 return imageForFEColorMatrix(downcast<FEColorMatrix>(effect), inputImages);
144 case FilterEffect::Type::ComponentTransfer:
145 return imageForFEComponentTransfer(downcast<FEComponentTransfer>(effect), inputImages);
146
147 // FIXME: Implement those filters using CIFilter so that the function returns a valid CIImage
148 case FilterEffect::Type::Blend:
149 case FilterEffect::Type::Composite:
150 case FilterEffect::Type::ConvolveMatrix:
151 case FilterEffect::Type::DiffuseLighting:
152 case FilterEffect::Type::DisplacementMap:
153 case FilterEffect::Type::DropShadow:
154 case FilterEffect::Type::Flood:
155 case FilterEffect::Type::GaussianBlur:
156 case FilterEffect::Type::Image:
157 case FilterEffect::Type::Lighting:
158 case FilterEffect::Type::Merge:
159 case FilterEffect::Type::Morphology:
160 case FilterEffect::Type::Offset:
161 case FilterEffect::Type::SpecularLighting:
162 case FilterEffect::Type::Tile:
163 case FilterEffect::Type::Turbulence:
164 case FilterEffect::Type::SourceAlpha:
165 default:
166 return nullptr;
167 }
168 return nullptr;
169}

171RetainPtr<CIImage> FilterEffectRendererCoreImage::imageForSourceGraphic(SourceGraphic& effect)
172{
173 ImageBuffer* sourceImage = effect.filter().sourceImage();
174 if (!sourceImage)
175 return nullptr;
176
177 if (is<IOSurfaceImageBuffer>(*sourceImage))
178 return [CIImage imageWithIOSurface:downcast<IOSurfaceImageBuffer>(*sourceImage).surface().surface()];
179
180 return [CIImage imageWithCGImage:sourceImage->copyNativeImage()->platformImage().get()];
181}

183RetainPtr<CIImage> FilterEffectRendererCoreImage::imageForFEColorMatrix(const FEColorMatrix& effect, const Vector<RetainPtr<CIImage>>& inputImages)
184{
185 auto inputImage = inputImages.at(0);
186
187 auto values = FEColorMatrix::normalizedFloats(effect.values());
188 float components[9];
189
190 switch (effect.type()) {
191 case FECOLORMATRIX_TYPE_SATURATE:
192 FEColorMatrix::calculateSaturateComponents(components, values[0]);
193 break;
194
195 case FECOLORMATRIX_TYPE_HUEROTATE:
196 FEColorMatrix::calculateHueRotateComponents(components, values[0]);
197 break;
198
199 case FECOLORMATRIX_TYPE_MATRIX:
200 break;
201
202 case FECOLORMATRIX_TYPE_UNKNOWN:
203 case FECOLORMATRIX_TYPE_LUMINANCETOALPHA: // FIXME: Add Luminance to Alpha Implementation
204 return nullptr;
205 }
206
207 auto *colorMatrixFilter = [CIFilter filterWithName:@"CIColorMatrix"];
208 [colorMatrixFilter setValue:inputImage.get() forKey:kCIInputImageKey];
209
210 switch (effect.type()) {
211 case FECOLORMATRIX_TYPE_SATURATE:
212 case FECOLORMATRIX_TYPE_HUEROTATE: {
213 [colorMatrixFilter setValue:[CIVector vectorWithX:components[0] Y:components[1] Z:components[2] W:0] forKey:@"inputRVector"];
214 [colorMatrixFilter setValue:[CIVector vectorWithX:components[3] Y:components[4] Z:components[5] W:0] forKey:@"inputGVector"];
215 [colorMatrixFilter setValue:[CIVector vectorWithX:components[6] Y:components[7] Z:components[8] W:0] forKey:@"inputBVector"];
216 [colorMatrixFilter setValue:[CIVector vectorWithX:0 Y:0 Z:0 W:1] forKey:@"inputAVector"];
217 [colorMatrixFilter setValue:[CIVector vectorWithX:0 Y:0 Z:0 W:0] forKey:@"inputBiasVector"];
218 break;
219 }
220 case FECOLORMATRIX_TYPE_MATRIX: {
221 [colorMatrixFilter setValue:[CIVector vectorWithX:values[0] Y:values[1] Z:values[2] W:values[3]] forKey:@"inputRVector"];
222 [colorMatrixFilter setValue:[CIVector vectorWithX:values[5] Y:values[6] Z:values[7] W:values[8]] forKey:@"inputGVector"];
223 [colorMatrixFilter setValue:[CIVector vectorWithX:values[10] Y:values[11] Z:values[12] W:values[13]] forKey:@"inputBVector"];
224 [colorMatrixFilter setValue:[CIVector vectorWithX:values[15] Y:values[16] Z:values[17] W:values[18]] forKey:@"inputAVector"];
225 [colorMatrixFilter setValue:[CIVector vectorWithX:values[4] Y:values[9] Z:values[14] W:values[19]] forKey:@"inputBiasVector"];
226 break;
227 }
228 case FECOLORMATRIX_TYPE_LUMINANCETOALPHA:
229 case FECOLORMATRIX_TYPE_UNKNOWN:
230 return nullptr;
231 }
232 return colorMatrixFilter.outputImage;
233}

235RetainPtr<CIImage> FilterEffectRendererCoreImage::imageForFEComponentTransfer(const FEComponentTransfer& effect, Vector<RetainPtr<CIImage>>& inputImages)
236{
237 // FIXME: Implement the rest of FEComponentTransfer functions
238 ASSERT(isNullOrLinearComponentTransferFunction(effect));
239
240 auto inputImage = inputImages.at(0);
241 auto filter = [CIFilter filterWithName:@"CIColorPolynomial"];
242
243 [filter setValue:inputImage.get() forKey:kCIInputImageKey];
244
245 auto setCoefficients = [&] (NSString *key, const ComponentTransferFunction& function) {
246 if (function.type == FECOMPONENTTRANSFER_TYPE_LINEAR)
247 [filter setValue:[CIVector vectorWithX:function.intercept Y:function.slope Z:0 W:0] forKey:key];
248 };
249 setCoefficients(@"inputRedCoefficients", effect.redFunction());
250 setCoefficients(@"inputGreenCoefficients", effect.greenFunction());
251 setCoefficients(@"inputBlueCoefficients", effect.blueFunction());
252 setCoefficients(@"inputAlphaCoefficients", effect.alphaFunction());
253
254 return filter.outputImage;
255}

257bool FilterEffectRendererCoreImage::canRenderUsingCIFilters(FilterEffect& effect)
258{
259 if (!supportsCoreImageRendering(effect))
260 return false;
261
262 for (auto in : effect.inputEffects()) {
263 if (!supportsCoreImageRendering(*in) || !canRenderUsingCIFilters(*in))
264 return false;
265 }
266 return true;
267}

275void FilterEffectRendererCoreImage::renderToImageBuffer(FilterEffect& lastEffect)
276{
277 FloatSize clampedSize = ImageBuffer::clampedSize(lastEffect.absolutePaintRect().size());
278 m_outputImageBuffer = IOSurfaceImageBuffer::create(clampedSize, lastEffect.filter().filterScale(), lastEffect.resultColorSpace(), PixelFormat::BGRA8);
279 if (!m_outputImageBuffer) {
280 clearResult();
281 return;
282 }
283
284 [sharedCIContext().get() render: m_outputImage.get() toIOSurface: m_outputImageBuffer->surface().surface() bounds:destRect(lastEffect) colorSpace:lastEffect.resultColorSpace().platformColorSpace()];
285}
286
287FloatRect FilterEffectRendererCoreImage::destRect(const FilterEffect& lastEffect) const
288{
289 IntSize destSize = lastEffect.absolutePaintRect().size();
290 destSize.scale(lastEffect.filter().filterScale());
291 FloatRect destRect = FloatRect(FloatPoint(), destSize);
292 return destRect;
293}
