38FontPlatformData::FontPlatformData(RetainPtr<CTFontRef>&& font, float size, bool syntheticBold, bool syntheticOblique, FontOrientation orientation, FontWidthVariant widthVariant, TextRenderingMode textRenderingMode, CreationData* creationData)
39 : FontPlatformData(size, syntheticBold, syntheticOblique, orientation, widthVariant, textRenderingMode, creationData)
40{
41 ASSERT_ARG(font, font);
42#if PLATFORM(WIN)
43 m_ctFont = font;
44#else
45 m_font = font;
46#endif
47 m_isColorBitmapFont = CTFontGetSymbolicTraits(font.get()) & kCTFontColorGlyphsTrait;
48 m_isSystemFont = WebCore::isSystemFont(font.get());
49 auto variations = adoptCF(static_cast<CFDictionaryRef>(CTFontCopyAttribute(font.get(), kCTFontVariationAttribute)));
50 m_hasVariations = variations && CFDictionaryGetCount(variations.get());
51
52#if PLATFORM(IOS_FAMILY)
53 m_isEmoji = CTFontIsAppleColorEmoji(m_font.get());
54#endif
55}

57bool isSystemFont(CTFontRef font)
58{
59#if HAVE(CTFONTISSYSTEMUIFONT)
60 return CTFontIsSystemUIFont(font);
61#else
62 return CTFontDescriptorIsSystemUIFont(adoptCF(CTFontCopyFontDescriptor(font)).get());
63#endif
64}

66CTFontRef FontPlatformData::registeredFont() const
67{
68 CTFontRef platformFont = ctFont();
69 ASSERT(platformFont);
70 if (platformFont && adoptCF(CTFontCopyAttribute(platformFont, kCTFontURLAttribute)))
71 return platformFont;
72 return nullptr;
73}

76inline int mapFontWidthVariantToCTFeatureSelector(FontWidthVariant variant)
77{
78 switch (variant) {
79 case FontWidthVariant::RegularWidth:
80 return kProportionalTextSelector;
81
82 case FontWidthVariant::HalfWidth:
83 return kHalfWidthTextSelector;
84
85 case FontWidthVariant::ThirdWidth:
86 return kThirdWidthTextSelector;
87
88 case FontWidthVariant::QuarterWidth:
89 return kQuarterWidthTextSelector;
90 }
91
92 ASSERT_NOT_REACHED();
93 return kProportionalTextSelector;
94}

96static RetainPtr<CFDictionaryRef> cascadeToLastResortAttributesDictionary()
97{
98 auto lastResort = adoptCF(CTFontDescriptorCreateWithNameAndSize(CFSTR("LastResort"), 0));
99
100 CFTypeRef descriptors[] = { lastResort.get() };
101 RetainPtr<CFArrayRef> array = adoptCF(CFArrayCreate(kCFAllocatorDefault, descriptors, WTF_ARRAY_LENGTH(descriptors), &kCFTypeArrayCallBacks));
102
103 CFTypeRef keys[] = { kCTFontCascadeListAttribute };
104 CFTypeRef values[] = { array.get() };
105 return adoptCF(CFDictionaryCreate(kCFAllocatorDefault, keys, values, WTF_ARRAY_LENGTH(keys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
106}
107
108static CTFontDescriptorRef cascadeToLastResortAndVariationsFontDescriptor()
109{
110 static CTFontDescriptorRef descriptor = CTFontDescriptorCreateWithAttributes(cascadeToLastResortAttributesDictionary().get());
111 return descriptor;
112}
113
114CTFontRef FontPlatformData::ctFont() const
115{
116 if (m_ctFont)
117 return m_ctFont.get();
118
119 ASSERT(m_font);
120 m_ctFont = adoptCF(CTFontCreateCopyWithAttributes(m_font.get(), m_size, nullptr, cascadeToLastResortAndVariationsFontDescriptor()));
121
122 if (m_widthVariant != FontWidthVariant::RegularWidth) {
123 int featureTypeValue = kTextSpacingType;
124 int featureSelectorValue = mapFontWidthVariantToCTFeatureSelector(m_widthVariant);
125 RetainPtr<CTFontDescriptorRef> sourceDescriptor = adoptCF(CTFontCopyFontDescriptor(m_ctFont.get()));
126 RetainPtr<CFNumberRef> featureType = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &featureTypeValue));
127 RetainPtr<CFNumberRef> featureSelector = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberIntType, &featureSelectorValue));
128 RetainPtr<CTFontDescriptorRef> newDescriptor = adoptCF(CTFontDescriptorCreateCopyWithFeature(sourceDescriptor.get(), featureType.get(), featureSelector.get()));
129 RetainPtr<CTFontRef> newFont = adoptCF(CTFontCreateWithFontDescriptor(newDescriptor.get(), m_size, 0));
130
131 if (newFont)
132 m_ctFont = newFont;
133 }
134
135 return m_ctFont.get();
136}

139RetainPtr<CFTypeRef> FontPlatformData::objectForEqualityCheck(CTFontRef ctFont)
140{
141 auto fontDescriptor = adoptCF(CTFontCopyFontDescriptor(ctFont));
142 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=138683 This is a shallow pointer compare for web fonts
143 // because the URL contains the address of the font. This means we might erroneously get false negatives.
144 RetainPtr<CFURLRef> url = adoptCF(static_cast<CFURLRef>(CTFontDescriptorCopyAttribute(fontDescriptor.get(), kCTFontReferenceURLAttribute)));
145 ASSERT(!url || CFGetTypeID(url.get()) == CFURLGetTypeID());
146 return url;
147}
148
149RetainPtr<CFTypeRef> FontPlatformData::objectForEqualityCheck() const
150{
151 return objectForEqualityCheck(ctFont());
152}

154RefPtr<SharedBuffer> FontPlatformData::openTypeTable(uint32_t table) const
155{
156 if (RetainPtr<CFDataRef> data = adoptCF(CTFontCopyTable(ctFont(), table, kCTFontTableOptionNoOptions)))
157 return SharedBuffer::create(data.get());
158
159 return platformOpenTypeTable(table);
160}

164String FontPlatformData::description() const
165{
166 String fontDescription { adoptCF(CFCopyDescription(ctFont())).get() };
167 return makeString(fontDescription, ' ', m_size,
168 (m_syntheticBold ? " synthetic bold" : ""),
169 (m_syntheticOblique ? " synthetic oblique" : ""),
170 (m_orientation == FontOrientation::Vertical ? " vertical orientation" : ""));
171}
