44static bool shouldFillWithVerticalGlyphs(const UChar* buffer, unsigned bufferLength, const Font& font)
45{
46 if (!font.hasVerticalGlyphs())
47 return false;
48 for (unsigned i = 0; i < bufferLength; ++i) {
49 if (!FontCascade::isCJKIdeograph(buffer[i]))
50 return true;
51 }
52 return false;
53}
54
55static const constexpr CGGlyph deletedGlyph = 0xFFFF;
56
57bool GlyphPage::fill(UChar* buffer, unsigned bufferLength)
58{
59 ASSERT(bufferLength == GlyphPage::size || bufferLength == 2 * GlyphPage::size);
60
61 const Font& font = this->font();
62 Vector<CGGlyph, 512> glyphs(bufferLength);
63 unsigned glyphStep = bufferLength / GlyphPage::size;
64
65 if (shouldFillWithVerticalGlyphs(buffer, bufferLength, font))
66 CTFontGetVerticalGlyphsForCharacters(font.platformData().ctFont(), reinterpret_cast<UniChar*>(buffer), glyphs.data(), bufferLength);
67 else
68 CTFontGetGlyphsForCharacters(font.platformData().ctFont(), reinterpret_cast<UniChar*>(buffer), glyphs.data(), bufferLength);
69
70 bool haveGlyphs = false;
71 for (unsigned i = 0; i < GlyphPage::size; ++i) {
72 auto theGlyph = glyphs[i * glyphStep];
73 if (theGlyph && theGlyph != deletedGlyph) {
74 setGlyphForIndex(i, theGlyph);
75 haveGlyphs = true;
76 }
77 }
78 return haveGlyphs;
79}
