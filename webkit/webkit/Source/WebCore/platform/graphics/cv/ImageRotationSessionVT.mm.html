39static ImageRotationSessionVT::RotationProperties transformToRotationProperties(const AffineTransform& inTransform)
40{
41 ImageRotationSessionVT::RotationProperties rotation;
42 if (inTransform.isIdentity())
43 return rotation;
44
45 AffineTransform::DecomposedType decomposed { };
46 if (!inTransform.decompose(decomposed))
47 return rotation;
48
49 rotation.flipY = WTF::areEssentiallyEqual(decomposed.scaleX, -1.);
50 rotation.flipX = WTF::areEssentiallyEqual(decomposed.scaleY, -1.);
51 auto degrees = rad2deg(decomposed.angle);
52 while (degrees < 0)
53 degrees += 360;
54
55 // Only support rotation in multiples of 90ยบ:
56 if (WTF::areEssentiallyEqual(fmod(degrees, 90.), 0.))
57 rotation.angle = clampToUnsigned(degrees);
58
59 return rotation;
60}

73void ImageRotationSessionVT::initialize(const RotationProperties& rotation, FloatSize size, IsCGImageCompatible isCGImageCompatible)
74{
75 m_rotationProperties = rotation;
76 m_size = size;
77 m_isCGImageCompatible = isCGImageCompatible;
78
79 if (m_rotationProperties.angle == 90 || m_rotationProperties.angle == 270)
80 size = size.transposedSize();
81
82 m_rotatedSize = expandedIntSize(size);
83
84 VTImageRotationSessionRef rawRotationSession = nullptr;
85 VTImageRotationSessionCreate(kCFAllocatorDefault, m_rotationProperties.angle, &rawRotationSession);
86 m_rotationSession = rawRotationSession;
87 VTImageRotationSessionSetProperty(m_rotationSession.get(), kVTImageRotationPropertyKey_EnableHighSpeedTransfer, kCFBooleanTrue);
88
89 if (m_rotationProperties.flipY)
90 VTImageRotationSessionSetProperty(m_rotationSession.get(), kVTImageRotationPropertyKey_FlipVerticalOrientation, kCFBooleanTrue);
91 if (m_rotationProperties.flipX)
92 VTImageRotationSessionSetProperty(m_rotationSession.get(), kVTImageRotationPropertyKey_FlipHorizontalOrientation, kCFBooleanTrue);
93}

95RetainPtr<CVPixelBufferRef> ImageRotationSessionVT::rotate(CVPixelBufferRef pixelBuffer)
96{
97 auto pixelFormat = CVPixelBufferGetPixelFormatType(pixelBuffer);
98 if (pixelFormat != m_pixelFormat || !m_rotationPool) {
99 m_pixelFormat = pixelFormat;
100 auto pixelAttributes = @{
101 (__bridge NSString *)kCVPixelBufferWidthKey: @(m_rotatedSize.width()),
102 (__bridge NSString *)kCVPixelBufferHeightKey: @(m_rotatedSize.height()),
103 (__bridge NSString *)kCVPixelBufferPixelFormatTypeKey: @(m_pixelFormat),
104 (__bridge NSString *)kCVPixelBufferCGImageCompatibilityKey: (m_isCGImageCompatible == IsCGImageCompatible::Yes ? @YES : @NO),
105#if PLATFORM(IOS_SIMULATOR) || PLATFORM(MAC)
106 (__bridge NSString *)kCVPixelBufferIOSurfacePropertiesKey : @{ }
107#endif
108 };
109
110 CVPixelBufferPoolRef rawPool = nullptr;
111 if (auto err = CVPixelBufferPoolCreate(kCFAllocatorDefault, nullptr, (__bridge CFDictionaryRef)pixelAttributes, &rawPool); err != noErr) {
112 RELEASE_LOG_ERROR(WebRTC, "ImageRotationSessionVT failed creating buffer pool with error %d", err);
113 return nullptr;
114 }
115
116 m_rotationPool = adoptCF(rawPool);
117 }
118
119 RetainPtr<CVPixelBufferRef> result;
120 CVPixelBufferRef rawRotatedBuffer = nullptr;
121 auto status = CVPixelBufferPoolCreatePixelBuffer(kCFAllocatorDefault, m_rotationPool.get(), &rawRotatedBuffer);
122 if (status != kCVReturnSuccess) {
123 RELEASE_LOG_ERROR(WebRTC, "ImageRotationSessionVT failed creating buffer from pool with error %d", status);
124 return nullptr;
125 }
126 result = adoptCF(rawRotatedBuffer);
127
128 status = VTImageRotationSessionTransferImage(m_rotationSession.get(), pixelBuffer, rawRotatedBuffer);
129 if (status != noErr) {
130 RELEASE_LOG_ERROR(WebRTC, "ImageRotationSessionVT failed rotating buffer with error %d", status);
131 return nullptr;
132 }
133
134 return result;
135}

137RetainPtr<CVPixelBufferRef> ImageRotationSessionVT::rotate(MediaSample& sample, const RotationProperties& rotation, IsCGImageCompatible cgImageCompatible)
138{
139 auto pixelBuffer = static_cast<CVPixelBufferRef>(PAL::CMSampleBufferGetImageBuffer(sample.platformSample().sample.cmSampleBuffer));
140 ASSERT(pixelBuffer);
141 if (!pixelBuffer)
142 return nullptr;
143
144 m_pixelFormat = CVPixelBufferGetPixelFormatType(pixelBuffer);
145 IntSize size { (int)CVPixelBufferGetWidth(pixelBuffer), (int)CVPixelBufferGetHeight(pixelBuffer) };
146 if (rotation != m_rotationProperties || m_size != size)
147 initialize(rotation, size, cgImageCompatible);
148
149 return rotate(pixelBuffer);
150}
