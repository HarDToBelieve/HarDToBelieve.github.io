55ImageTransferSessionVT::ImageTransferSessionVT(uint32_t pixelFormat)
56{
57 VTPixelTransferSessionRef transferSession;
58 VTPixelTransferSessionCreate(kCFAllocatorDefault, &transferSession);
59 ASSERT(transferSession);
60 m_transferSession = adoptCF(transferSession);
61
62 auto status = VTSessionSetProperty(transferSession, kVTPixelTransferPropertyKey_ScalingMode, kVTScalingMode_Trim);
63 if (status != kCVReturnSuccess)
64 RELEASE_LOG(Media, "ImageTransferSessionVT::ImageTransferSessionVT: VTSessionSetProperty(kVTPixelTransferPropertyKey_ScalingMode) failed with error %d", static_cast<int>(status));
65
66 status = VTSessionSetProperty(transferSession, kVTPixelTransferPropertyKey_EnableHighSpeedTransfer, @YES);
67 if (status != kCVReturnSuccess)
68 RELEASE_LOG(Media, "ImageTransferSessionVT::ImageTransferSessionVT: VTSessionSetProperty(kVTPixelTransferPropertyKey_EnableHighSpeedTransfer) failed with error %d", static_cast<int>(status));
69
70 status = VTSessionSetProperty(transferSession, kVTPixelTransferPropertyKey_RealTime, @YES);
71 if (status != kCVReturnSuccess)
72 RELEASE_LOG(Media, "ImageTransferSessionVT::ImageTransferSessionVT: VTSessionSetProperty(kVTPixelTransferPropertyKey_RealTime) failed with error %d", static_cast<int>(status));
73
74#if PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST)
75 status = VTSessionSetProperty(transferSession, kVTPixelTransferPropertyKey_EnableHardwareAcceleratedTransfer, @YES);
76 if (status != kCVReturnSuccess)
77 RELEASE_LOG(Media, "ImageTransferSessionVT::ImageTransferSessionVT: VTSessionSetProperty(kVTPixelTransferPropertyKey_EnableHardwareAcceleratedTransfer) failed with error %d", static_cast<int>(status));
78#endif
79
80 m_pixelFormat = pixelFormat;
81}

83bool ImageTransferSessionVT::setSize(const IntSize& size)
84{
85 if (m_size == size && m_outputBufferPool)
86 return true;
87
88 NSDictionary* pixelBufferOptions = @{
89 (__bridge NSString *)kCVPixelBufferWidthKey : @(size.width()),
90 (__bridge NSString *)kCVPixelBufferHeightKey : @(size.height()),
91 (__bridge NSString *)kCVPixelBufferPixelFormatTypeKey : @(m_pixelFormat),
92 (__bridge NSString *)cvPixelFormatOpenGLKey() : @YES,
93 (__bridge NSString *)kCVPixelBufferIOSurfacePropertiesKey : @{ /*empty dictionary*/ },
94 };
95
96 NSDictionary* pixelBufferPoolOptions = @{
97 (__bridge NSString *)kCVPixelBufferPoolMinimumBufferCountKey: @(6)
98 };
99
100 CVPixelBufferPoolRef bufferPool;
101 auto status = CVPixelBufferPoolCreate(kCFAllocatorDefault, (__bridge CFDictionaryRef)pixelBufferPoolOptions, (__bridge CFDictionaryRef)pixelBufferOptions, &bufferPool);
102 ASSERT(!status);
103 if (status != kCVReturnSuccess)
104 return false;
105
106 m_outputBufferPool = adoptCF(bufferPool);
107 m_size = size;
108 m_ioSurfaceBufferAttributes = nullptr;
109
110 return true;
111}

113RetainPtr<CVPixelBufferRef> ImageTransferSessionVT::convertPixelBuffer(CVPixelBufferRef sourceBuffer, const IntSize& size)
114{
115 if (sourceBuffer && m_size == IntSize(CVPixelBufferGetWidth(sourceBuffer), CVPixelBufferGetHeight(sourceBuffer)) && m_pixelFormat == CVPixelBufferGetPixelFormatType(sourceBuffer))
116 return retainPtr(sourceBuffer);
117
118 if (!sourceBuffer || !setSize(size))
119 return nullptr;
120
121 CVPixelBufferRef outputBuffer = nullptr;
122 auto status = CVPixelBufferPoolCreatePixelBuffer(kCFAllocatorDefault, m_outputBufferPool.get(), &outputBuffer);
123 if (status) {
124 RELEASE_LOG(Media, "ImageTransferSessionVT::convertPixelBuffer, CVPixelBufferPoolCreatePixelBuffer failed with error %d", static_cast<int>(status));
125 return nullptr;
126 }
127 auto result = adoptCF(outputBuffer);
128
129 auto err = VTPixelTransferSessionTransferImage(m_transferSession.get(), sourceBuffer, outputBuffer);
130 if (err) {
131 RELEASE_LOG(Media, "ImageTransferSessionVT::convertPixelBuffer, VTPixelTransferSessionTransferImage failed with error %d", static_cast<int>(err));
132 return nullptr;
133 }
134
135 return result;
136}

143RetainPtr<CMSampleBufferRef> ImageTransferSessionVT::convertCMSampleBuffer(CMSampleBufferRef sourceBuffer, const IntSize& size)
144{
145 if (!sourceBuffer)
146 return nullptr;
147
148 auto description = PAL::CMSampleBufferGetFormatDescription(sourceBuffer);
149 auto sourceSize = FloatSize(PAL::CMVideoFormatDescriptionGetPresentationDimensions(description, true, true));
150 auto pixelBuffer = static_cast<CVPixelBufferRef>(PAL::CMSampleBufferGetImageBuffer(sourceBuffer));
151 if (size == expandedIntSize(sourceSize) && m_pixelFormat == CVPixelBufferGetPixelFormatType(pixelBuffer))
152 return retainPtr(sourceBuffer);
153
154 if (!setSize(size))
155 return nullptr;
156
157 auto convertedPixelBuffer = createPixelBuffer(sourceBuffer, size);
158 if (!convertedPixelBuffer)
159 return nullptr;
160
161 CMItemCount itemCount = 0;
162 auto status = PAL::CMSampleBufferGetSampleTimingInfoArray(sourceBuffer, 1, nullptr, &itemCount);
163 if (status != noErr) {
164 RELEASE_LOG(Media, "ImageTransferSessionVT::convertCMSampleBuffer: CMSampleBufferGetSampleTimingInfoArray failed with error code: %d", static_cast<int>(status));
165 return nullptr;
166 }
167 Vector<CMSampleTimingInfo> timingInfoArray;
168 CMSampleTimingInfo* timeingInfoPtr = nullptr;
169 if (itemCount) {
170 timingInfoArray.grow(itemCount);
171 status = PAL::CMSampleBufferGetSampleTimingInfoArray(sourceBuffer, itemCount, timingInfoArray.data(), nullptr);
172 if (status != noErr) {
173 RELEASE_LOG(Media, "ImageTransferSessionVT::convertCMSampleBuffer: CMSampleBufferGetSampleTimingInfoArray failed with error code: %d", static_cast<int>(status));
174 return nullptr;
175 }
176 timeingInfoPtr = timingInfoArray.data();
177 }
178
179 CMVideoFormatDescriptionRef formatDescription = nullptr;
180 status = PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, convertedPixelBuffer.get(), &formatDescription);
181 if (status != noErr) {
182 RELEASE_LOG(Media, "ImageTransferSessionVT::convertCMSampleBuffer: CMVideoFormatDescriptionCreateForImageBuffer returned: %d", static_cast<int>(status));
183 return nullptr;
184 }
185
186 CMSampleBufferRef resizedSampleBuffer;
187 status = PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, convertedPixelBuffer.get(), formatDescription, timeingInfoPtr, &resizedSampleBuffer);
188 CFRelease(formatDescription);
189 if (status != noErr) {
190 RELEASE_LOG(Media, "ImageTransferSessionVT::convertCMSampleBuffer: failed to create CMSampleBuffer with error code: %d", static_cast<int>(status));
191 return nullptr;
192 }
193
194 return adoptCF(resizedSampleBuffer);
195}

197RetainPtr<CVPixelBufferRef> ImageTransferSessionVT::createPixelBuffer(CGImageRef image, const IntSize& size)
198{
199 if (!image || !setSize(size))
200 return nullptr;
201
202 CVPixelBufferRef rgbBuffer;
203 auto imageSize = IntSize(CGImageGetWidth(image), CGImageGetHeight(image));
204 auto status = CVPixelBufferCreate(kCFAllocatorDefault, imageSize.width(), imageSize.height(), kCVPixelFormatType_32ARGB, nullptr, &rgbBuffer);
205 if (status != kCVReturnSuccess) {
206 RELEASE_LOG(Media, "ImageTransferSessionVT::createPixelBuffer: CVPixelBufferCreate failed with error code: %d", static_cast<int>(status));
207 return nullptr;
208 }
209
210 CVPixelBufferLockBaseAddress(rgbBuffer, 0);
211 void* data = CVPixelBufferGetBaseAddress(rgbBuffer);
212 auto retainedRGBBuffer = adoptCF(rgbBuffer);
213 auto context = adoptCF(CGBitmapContextCreate(data, imageSize.width(), imageSize.height(), 8, CVPixelBufferGetBytesPerRow(rgbBuffer), sRGBColorSpaceRef(), (CGBitmapInfo) kCGImageAlphaNoneSkipFirst));
214 if (!context) {
215 RELEASE_LOG(Media, "ImageTransferSessionVT::createPixelBuffer: CGBitmapContextCreate returned nullptr");
216 return nullptr;
217 }
218
219 CGContextDrawImage(context.get(), CGRectMake(0, 0, imageSize.width(), imageSize.height()), image);
220 CVPixelBufferUnlockBaseAddress(rgbBuffer, 0);
221
222 return convertPixelBuffer(rgbBuffer, size);
223}

225RetainPtr<CMSampleBufferRef> ImageTransferSessionVT::createCMSampleBuffer(CVPixelBufferRef sourceBuffer, const MediaTime& sampleTime, const IntSize& size)
226{
227 if (!sourceBuffer || !setSize(size))
228 return nullptr;
229
230 auto bufferSize = IntSize(CVPixelBufferGetWidth(sourceBuffer), CVPixelBufferGetHeight(sourceBuffer));
231 RetainPtr<CVPixelBufferRef> inputBuffer = sourceBuffer;
232 if (bufferSize != m_size || m_pixelFormat != CVPixelBufferGetPixelFormatType(sourceBuffer)) {
233 inputBuffer = convertPixelBuffer(sourceBuffer, m_size);
234 if (!inputBuffer)
235 return nullptr;
236 }
237
238 CMVideoFormatDescriptionRef formatDescription = nullptr;
239 auto status = PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, (CVImageBufferRef)inputBuffer.get(), &formatDescription);
240 if (status) {
241 RELEASE_LOG(Media, "ImageTransferSessionVT::convertPixelBuffer: failed to initialize CMVideoFormatDescription with error code: %d", static_cast<int>(status));
242 return nullptr;
243 }
244
245 CMSampleBufferRef sampleBuffer;
246 auto cmTime = PAL::toCMTime(sampleTime);
247 CMSampleTimingInfo timingInfo = { PAL::kCMTimeInvalid, cmTime, cmTime };
248 status = PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, (CVImageBufferRef)inputBuffer.get(), formatDescription, &timingInfo, &sampleBuffer);
249 CFRelease(formatDescription);
250 if (status) {
251 RELEASE_LOG(Media, "ImageTransferSessionVT::convertPixelBuffer: failed to initialize CMSampleBuffer with error code: %d", static_cast<int>(status));
252 return nullptr;
253 }
254
255 return adoptCF(sampleBuffer);
256}

276CFDictionaryRef ImageTransferSessionVT::ioSurfacePixelBufferCreationOptions(IOSurfaceRef surface)
277{
278 if (m_ioSurfaceBufferAttributes)
279 return m_ioSurfaceBufferAttributes.get();
280
281 m_ioSurfaceBufferAttributes = (__bridge CFDictionaryRef) @{
282 (__bridge NSString *)cvPixelFormatOpenGLKey() : @YES,
283 };
284
285#if PLATFORM(MAC)
286 auto format = IOSurfaceGetPixelFormat(surface);
287 auto width = IOSurfaceGetWidth(surface);
288 auto height = IOSurfaceGetHeight(surface);
289 auto extendedRight = roundUpToMacroblockMultiple(width) - width;
290 auto extendedBottom = roundUpToMacroblockMultiple(height) - height;
291
292 if ((format == kCVPixelFormatType_420YpCbCr8BiPlanarVideoRange || format == kCVPixelFormatType_420YpCbCr8BiPlanarFullRange)
293 && (IOSurfaceGetBytesPerRowOfPlane(surface, 0) >= width + extendedRight)
294 && (IOSurfaceGetBytesPerRowOfPlane(surface, 1) >= width + extendedRight)
295 && (IOSurfaceGetAllocSize(surface) >= (height + extendedBottom) * IOSurfaceGetBytesPerRowOfPlane(surface, 0) * 3 / 2)) {
296 m_ioSurfaceBufferAttributes = (__bridge CFDictionaryRef) @{
297 (__bridge NSString *)kCVPixelBufferOpenGLCompatibilityKey : @YES,
298 (__bridge NSString *)kCVPixelBufferExtendedPixelsRightKey : @(extendedRight),
299 (__bridge NSString *)kCVPixelBufferExtendedPixelsBottomKey : @(extendedBottom)
300 };
301 }
302#else
303 UNUSED_PARAM(surface);
304#endif
305
306 return m_ioSurfaceBufferAttributes.get();
307}

323RetainPtr<CVPixelBufferRef> ImageTransferSessionVT::createPixelBuffer(IOSurfaceRef surface, const IntSize& size)
324{
325 if (!surface || !setSize(size))
326 return nullptr;
327
328 CVPixelBufferRef pixelBuffer;
329 auto status = CVPixelBufferCreateWithIOSurface(kCFAllocatorDefault, surface, ioSurfacePixelBufferCreationOptions(surface), &pixelBuffer);
330 if (status) {
331 RELEASE_LOG(Media, "CVPixelBufferCreateWithIOSurface failed with error code: %d", static_cast<int>(status));
332 return nullptr;
333 }
334
335 auto retainedBuffer = adoptCF(pixelBuffer);
336 if (m_size == IntSize(CVPixelBufferGetWidth(pixelBuffer), CVPixelBufferGetHeight(pixelBuffer)) && m_pixelFormat == CVPixelBufferGetPixelFormatType(pixelBuffer))
337 return retainedBuffer;

352RefPtr<MediaSample> ImageTransferSessionVT::convertMediaSample(MediaSample& sample, const IntSize& size)
353{
354 ASSERT(sample.platformSample().type == PlatformSample::CMSampleBufferType);
355
356 if (size == expandedIntSize(sample.presentationSize()))
357 return &sample;
358
359 auto resizedBuffer = convertCMSampleBuffer(sample.platformSample().sample.cmSampleBuffer, size);
360 if (!resizedBuffer)
361 return nullptr;
362
363 return MediaSampleAVFObjC::create(resizedBuffer.get(), sample.videoRotation(), sample.videoMirrored());
364}
