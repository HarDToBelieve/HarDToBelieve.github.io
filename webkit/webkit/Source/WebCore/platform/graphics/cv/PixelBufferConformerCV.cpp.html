38PixelBufferConformerCV::PixelBufferConformerCV(CFDictionaryRef attributes)
39{
40 VTPixelBufferConformerRef conformer = nullptr;
41 VTPixelBufferConformerCreateWithAttributes(kCFAllocatorDefault, attributes, &conformer);
42 ASSERT(conformer);
43 m_pixelConformer = adoptCF(conformer);
44}

46struct CVPixelBufferInfo {
47 RetainPtr<CVPixelBufferRef> pixelBuffer;
48 int lockCount { 0 };
49};
50
51static const void* CVPixelBufferGetBytePointerCallback(void* refcon)
52{
53 ASSERT(refcon);
54 if (!refcon) {
55 RELEASE_LOG_ERROR(Media, "CVPixelBufferGetBytePointerCallback() called with NULL refcon");
56 RELEASE_LOG_STACKTRACE(Media);
57 return nullptr;
58 }
59 auto info = static_cast<CVPixelBufferInfo*>(refcon);
60
61 CVReturn result = CVPixelBufferLockBaseAddress(info->pixelBuffer.get(), kCVPixelBufferLock_ReadOnly);
62
63 ASSERT(result == kCVReturnSuccess);
64 if (result != kCVReturnSuccess) {
65 RELEASE_LOG_ERROR(Media, "CVPixelBufferLockBaseAddress() returned error code %d", result);
66 RELEASE_LOG_STACKTRACE(Media);
67 return nullptr;
68 }
69
70 ++info->lockCount;
71 void* address = CVPixelBufferGetBaseAddress(info->pixelBuffer.get());
72 size_t byteLength = CVPixelBufferGetBytesPerRow(info->pixelBuffer.get()) * CVPixelBufferGetHeight(info->pixelBuffer.get());
73
74 verifyImageBufferIsBigEnough(address, byteLength);
75 RELEASE_LOG_INFO(Media, "CVPixelBufferGetBytePointerCallback() returning bytePointer: %p, size: %zu", address, byteLength);
76 return address;
77}

79static void CVPixelBufferReleaseBytePointerCallback(void* refcon, const void*)
80{
81 ASSERT(refcon);
82 if (!refcon) {
83 RELEASE_LOG_ERROR(Media, "CVPixelBufferReleaseBytePointerCallback() called with NULL refcon");
84 RELEASE_LOG_STACKTRACE(Media);
85 return;
86 }
87 auto info = static_cast<CVPixelBufferInfo*>(refcon);
88
89 CVReturn result = CVPixelBufferUnlockBaseAddress(info->pixelBuffer.get(), kCVPixelBufferLock_ReadOnly);
90 ASSERT(result == kCVReturnSuccess);
91 if (result != kCVReturnSuccess) {
92 RELEASE_LOG_ERROR(Media, "CVPixelBufferLockBaseAddress() returned error code %d", result);
93 RELEASE_LOG_STACKTRACE(Media);
94 return;
95 }
96
97 ASSERT(info->lockCount);
98 if (!info->lockCount) {
99 RELEASE_LOG_ERROR(Media, "CVPixelBufferReleaseBytePointerCallback() called without matching CVPixelBufferGetBytePointerCallback()");
100 RELEASE_LOG_STACKTRACE(Media);
101 }
102 --info->lockCount;
103}
104
105static void CVPixelBufferReleaseInfoCallback(void* refcon)
106{
107 ASSERT(refcon);
108 if (!refcon) {
109 RELEASE_LOG_ERROR(Media, "CVPixelBufferReleaseInfoCallback() called with NULL refcon");
110 RELEASE_LOG_STACKTRACE(Media);
111 return;
112 }
113 auto info = static_cast<CVPixelBufferInfo*>(refcon);
114
115 ASSERT(!info->lockCount);
116 if (info->lockCount) {
117 RELEASE_LOG_ERROR(Media, "CVPixelBufferReleaseInfoCallback() called with a non-zero lockCount: %d", info->lockCount);
118 RELEASE_LOG_STACKTRACE(Media);
119
120 CVReturn result = CVPixelBufferUnlockBaseAddress(info->pixelBuffer.get(), kCVPixelBufferLock_ReadOnly);
121 ASSERT(result == kCVReturnSuccess);
122 if (result != kCVReturnSuccess) {
123 RELEASE_LOG_ERROR(Media, "CVPixelBufferLockBaseAddress() returned error code %d", result);
124 RELEASE_LOG_STACKTRACE(Media);
125 }
126 }
127
128 info->pixelBuffer = nullptr;
129 delete info;
130}

132RetainPtr<CVPixelBufferRef> PixelBufferConformerCV::convert(CVPixelBufferRef rawBuffer)
133{
134 RetainPtr<CVPixelBufferRef> buffer { rawBuffer };
135
136 if (!VTPixelBufferConformerIsConformantPixelBuffer(m_pixelConformer.get(), buffer.get())) {
137 CVPixelBufferRef outputBuffer = nullptr;
138 OSStatus status = VTPixelBufferConformerCopyConformedPixelBuffer(m_pixelConformer.get(), buffer.get(), false, &outputBuffer);
139 if (status != noErr || !outputBuffer)
140 return nullptr;
141 return adoptCF(outputBuffer);
142 }
143 return nullptr;
144}

146RetainPtr<CGImageRef> PixelBufferConformerCV::createImageFromPixelBuffer(CVPixelBufferRef rawBuffer)
147{
148 RetainPtr<CVPixelBufferRef> buffer { rawBuffer };
149 size_t width = CVPixelBufferGetWidth(buffer.get());
150 size_t height = CVPixelBufferGetHeight(buffer.get());
151
152 if (!VTPixelBufferConformerIsConformantPixelBuffer(m_pixelConformer.get(), buffer.get())) {
153 CVPixelBufferRef outputBuffer = nullptr;
154 OSStatus status = VTPixelBufferConformerCopyConformedPixelBuffer(m_pixelConformer.get(), buffer.get(), false, &outputBuffer);
155 if (status != noErr || !outputBuffer)
156 return nullptr;
157 buffer = adoptCF(outputBuffer);
158 }
159
160 CGBitmapInfo bitmapInfo = kCGBitmapByteOrder32Little | kCGImageAlphaFirst;
161 size_t bytesPerRow = CVPixelBufferGetBytesPerRow(buffer.get());
162 size_t byteLength = bytesPerRow * height;
163
164 ASSERT(byteLength);
165 if (!byteLength)
166 return nullptr;
167
168 CVPixelBufferInfo* info = new CVPixelBufferInfo();
169 info->pixelBuffer = WTFMove(buffer);
170 info->lockCount = 0;
171
172 CGDataProviderDirectCallbacks providerCallbacks = { 0, CVPixelBufferGetBytePointerCallback, CVPixelBufferReleaseBytePointerCallback, 0, CVPixelBufferReleaseInfoCallback };
173 RetainPtr<CGDataProviderRef> provider = adoptCF(CGDataProviderCreateDirect(info, byteLength, &providerCallbacks));
174
175 return adoptCF(CGImageCreate(width, height, 8, 32, bytesPerRow, sRGBColorSpaceRef(), bitmapInfo, provider.get(), nullptr, false, kCGRenderingIntentDefault));
176}
