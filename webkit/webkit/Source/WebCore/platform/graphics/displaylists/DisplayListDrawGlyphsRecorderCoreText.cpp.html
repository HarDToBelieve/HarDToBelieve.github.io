50static CGContextDelegateRef beginLayer(CGContextDelegateRef delegate, CGRenderingStateRef rstate, CGGStateRef gstate, CGRect rect, CFDictionaryRef, CGContextDelegateRef)
51{
52 DrawGlyphsRecorder& recorder = *static_cast<DrawGlyphsRecorder*>(CGContextDelegateGetInfo(delegate));
53 recorder.recordBeginLayer(rstate, gstate, rect);
54 return delegate;
55}

57static CGContextDelegateRef endLayer(CGContextDelegateRef delegate, CGRenderingStateRef rstate, CGGStateRef gstate)
58{
59 DrawGlyphsRecorder& recorder = *static_cast<DrawGlyphsRecorder*>(CGContextDelegateGetInfo(delegate));
60 recorder.recordEndLayer(rstate, gstate);

64static CGError drawGlyphs(CGContextDelegateRef delegate, CGRenderingStateRef rstate, CGGStateRef gstate, const CGAffineTransform* tm, const CGGlyph glyphs[], const CGPoint positions[], size_t count)
65{
66 if (CGGStateGetAlpha(gstate) > 0) {
67 DrawGlyphsRecorder& recorder = *static_cast<DrawGlyphsRecorder*>(CGContextDelegateGetInfo(delegate));
68 recorder.recordDrawGlyphs(rstate, gstate, tm, glyphs, positions, count);
69 }
70 return kCGErrorSuccess;
71}

80UniqueRef<GraphicsContext> DrawGlyphsRecorder::createInternalContext()
81{
82 auto contextDelegate = adoptCF(CGContextDelegateCreate(this));
83 CGContextDelegateSetCallback(contextDelegate.get(), deBeginLayer, reinterpret_cast<CGContextDelegateCallback>(&beginLayer));
84 CGContextDelegateSetCallback(contextDelegate.get(), deEndLayer, reinterpret_cast<CGContextDelegateCallback>(&endLayer));
85 CGContextDelegateSetCallback(contextDelegate.get(), deDrawGlyphs, reinterpret_cast<CGContextDelegateCallback>(&WebCore::DisplayList::drawGlyphs));
86 CGContextDelegateSetCallback(contextDelegate.get(), deDrawImage, reinterpret_cast<CGContextDelegateCallback>(&drawImage));
87 auto context = adoptCF(CGContextCreateWithDelegate(contextDelegate.get(), kCGContextTypeUnknown, nullptr, nullptr));
88 return makeUniqueRef<GraphicsContextCG>(context.get());
89}

98void DrawGlyphsRecorder::populateInternalState(const GraphicsContextState& contextState)
99{
100 m_originalState.fillStyle.color = contextState.fillColor;
101 m_originalState.fillStyle.gradient = contextState.fillGradient;
102 m_originalState.fillStyle.gradientSpaceTransform = contextState.fillGradientSpaceTransform;
103 m_originalState.fillStyle.pattern = contextState.fillPattern;
104
105 m_originalState.strokeStyle.color = contextState.strokeColor;
106 m_originalState.strokeStyle.gradient = contextState.strokeGradient;
107 m_originalState.strokeStyle.gradientSpaceTransform = contextState.strokeGradientSpaceTransform;
108 m_originalState.strokeStyle.pattern = contextState.strokePattern;
109
110 m_originalState.ctm = m_owner.currentState().ctm; // FIXME: Deal with base CTM.
111
112 m_originalState.shadow.offset = contextState.shadowOffset;
113 m_originalState.shadow.blur = contextState.shadowBlur;
114 m_originalState.shadow.color = contextState.shadowColor;
115 m_originalState.shadow.ignoreTransforms = contextState.shadowsIgnoreTransforms;
116
117 m_currentState = m_originalState;
118}

120void DrawGlyphsRecorder::populateInternalContext(const GraphicsContextState& contextState)
121{
122 if (m_originalState.fillStyle.color.isValid())
123 m_internalContext->setFillColor(m_originalState.fillStyle.color);
124 else if (m_originalState.fillStyle.gradient)
125 m_internalContext->setFillGradient(*m_originalState.fillStyle.gradient, m_originalState.fillStyle.gradientSpaceTransform);
126 else {
127 ASSERT(m_originalState.fillStyle.pattern);
128 if (m_originalState.fillStyle.pattern)
129 m_internalContext->setFillPattern(*m_originalState.fillStyle.pattern);
130 }
131
132 if (m_originalState.strokeStyle.color.isValid())
133 m_internalContext->setStrokeColor(m_originalState.strokeStyle.color);
134 else if (m_originalState.strokeStyle.gradient)
135 m_internalContext->setStrokeGradient(*m_originalState.strokeStyle.gradient, m_originalState.strokeStyle.gradientSpaceTransform);
136 else {
137 ASSERT(m_originalState.strokeStyle.pattern);
138 if (m_originalState.strokeStyle.pattern)
139 m_internalContext->setStrokePattern(*m_originalState.strokeStyle.pattern);
140 }
141
142 m_internalContext->setCTM(m_originalState.ctm);
143
144 m_internalContext->setShadowsIgnoreTransforms(m_originalState.shadow.ignoreTransforms);
145 m_internalContext->setShadow(m_originalState.shadow.offset, m_originalState.shadow.blur, m_originalState.shadow.color, contextState.shadowRadiusMode);
146
147 m_internalContext->setTextDrawingMode(contextState.textDrawingMode);
148}

150void DrawGlyphsRecorder::prepareInternalContext(const Font& font, FontSmoothingMode smoothingMode)
151{
152 ASSERT(CGAffineTransformIsIdentity(CTFontGetMatrix(font.platformData().ctFont())));
153
154 m_originalFont = &font;
155 m_smoothingMode = smoothingMode;
156
157 m_originalTextMatrix = computeOverallTextMatrix(font);
158 if (font.platformData().orientation() == FontOrientation::Vertical)
159 m_originalTextMatrix = computeVerticalTextMatrix(font, m_originalTextMatrix);
160
161 auto& contextState = m_owner.currentState().stateChange.m_state;
162 populateInternalState(contextState);
163 populateInternalContext(contextState);
164}

166void DrawGlyphsRecorder::concludeInternalContext()
167{
168 updateCTM(m_originalState.ctm);
169 updateFillColor(m_originalState.fillStyle.color, m_originalState.fillStyle.gradient.get(), m_originalState.fillStyle.pattern.get());
170 updateStrokeColor(m_originalState.strokeStyle.color, m_originalState.strokeStyle.gradient.get(), m_originalState.strokeStyle.pattern.get());
171 updateShadow(m_originalState.shadow.offset, m_originalState.shadow.blur, m_originalState.shadow.color, m_originalState.shadow.ignoreTransforms ? ShadowsIgnoreTransforms::Yes : ShadowsIgnoreTransforms::No);
172}

174void DrawGlyphsRecorder::updateFillColor(const Color& newColor, Gradient* newGradient, Pattern* newPattern)
175{
176 // This check looks wrong but it actually isn't, for our limited use.
177 // CT will only ever set this to a solid color, which this is the correct check for.
178 // In concludeInternalContext() we set it back to what it was originally, which this check works correctly for too.
179 if (newColor == m_currentState.fillStyle.color)
180 return;
181
182 GraphicsContextState newState;
183 newState.fillColor = newColor;
184 if (newGradient)
185 newState.fillGradient = newGradient;
186 if (newPattern)
187 newState.fillPattern = newPattern;
188 m_owner.updateState(newState, { GraphicsContextState::FillColorChange });
189 m_currentState.fillStyle.color = newColor;
190}

219static bool shadowIsCleared(const FloatSize& shadowOffset, float shadowBlur)
220{
221 return shadowOffset == FloatSize() && !shadowBlur;
222}
223
224void DrawGlyphsRecorder::updateShadow(const FloatSize& shadowOffset, float shadowBlur, const Color& shadowColor, ShadowsIgnoreTransforms shadowsIgnoreTransforms)
225{
226 // We don't need to consider shadowsIgnoreTransforms if nobody has any shadows.
227 if (shadowIsCleared(shadowOffset, shadowBlur) && shadowIsCleared(m_currentState.shadow.offset, m_currentState.shadow.blur))
228 return;
229
230 GraphicsContextState newState;
231 GraphicsContextState::StateChangeFlags stateChangeFlags;
232
233 if (shadowOffset != m_currentState.shadow.offset || shadowBlur != m_currentState.shadow.blur || shadowColor != m_currentState.shadow.color) {
234 newState.shadowOffset = shadowOffset;
235 newState.shadowBlur = shadowBlur;
236 newState.shadowColor = shadowColor;
237 stateChangeFlags.add(GraphicsContextState::ShadowChange);
238 }
239 if (shadowsIgnoreTransforms != ShadowsIgnoreTransforms::Unspecified && (shadowsIgnoreTransforms == ShadowsIgnoreTransforms::Yes) != m_currentState.shadow.ignoreTransforms) {
240 newState.shadowsIgnoreTransforms = (shadowsIgnoreTransforms == ShadowsIgnoreTransforms::Yes);
241 stateChangeFlags.add(GraphicsContextState::ShadowsIgnoreTransformsChange);
242 }
243 if (stateChangeFlags.isEmpty())
244 return;
245 m_owner.updateState(newState, stateChangeFlags);
246
247 m_currentState.shadow.offset = shadowOffset;
248 m_currentState.shadow.blur = shadowBlur;
249 m_currentState.shadow.color = shadowColor;
250 m_currentState.shadow.ignoreTransforms = (shadowsIgnoreTransforms == ShadowsIgnoreTransforms::Yes);
251}

253void DrawGlyphsRecorder::updateShadow(CGStyleRef style)
254{
255 if (CGStyleGetType(style) != kCGStyleShadow) {
256 // FIXME: Support more kinds of CGStyles.
257 updateShadow({0, 0}, 0, Color(), ShadowsIgnoreTransforms::Unspecified);
258 return;
259 }
260
261 const auto& shadowStyle = *static_cast<const CGShadowStyle*>(CGStyleGetData(style));
262 auto rad = deg2rad(shadowStyle.azimuth - 180);
263 auto shadowOffset = FloatSize(std::cos(rad), std::sin(rad)) * shadowStyle.height;
264 updateShadow(shadowOffset, shadowStyle.radius, CGStyleGetColor(style), ShadowsIgnoreTransforms::Yes);
265}

280static Vector<CGSize> computeAdvancesFromPositions(const CGPoint positions[], size_t count, const CGAffineTransform& textMatrix)
281{
282 Vector<CGSize> result;
283 for (size_t i = 0; i < count - 1; ++i) {
284 auto nextPosition = positions[i + 1];
285 auto currentPosition = positions[i];
286 auto advance = CGSizeMake(nextPosition.x - currentPosition.x, nextPosition.y - currentPosition.y);
287 result.append(CGSizeApplyAffineTransform(advance, textMatrix));
288 }
289 result.constructAndAppend(CGSizeMake(0, 0));
290 return result;
291}

293void DrawGlyphsRecorder::recordDrawGlyphs(CGRenderingStateRef, CGGStateRef gstate, const CGAffineTransform*, const CGGlyph glyphs[], const CGPoint positions[], size_t count)
294{
295 if (!count)
296 return;
297
298 CGFontRef usedFont = CGGStateGetFont(gstate);
299 if (usedFont != adoptCF(CTFontCopyGraphicsFont(m_originalFont->platformData().ctFont(), nullptr)).get())
300 return;
301
302#if ASSERT_ENABLED
303 auto textPosition = CGContextGetTextPosition(m_internalContext->platformContext());
304 ASSERT(!textPosition.x);
305 ASSERT(!textPosition.y);
306#endif
307
308 updateCTM(*CGGStateGetCTM(gstate));
309
310 // We want the replayer's CTM and text matrix to match the current CTM and text matrix.
311 // The current text matrix is a concatenation of whatever WebKit sets it to and whatever
312 // Core Text appends to it. So, we have
313 // CTM * m_originalTextMatrix * Core Text's text matrix.
314 // However, CGContextGetTextMatrix() just tells us what the whole text matrix is, so
315 // m_originalTextMatrix * Core Text's text matrix = currentTextMatrix.
316 // The only way we can emulate Core Text's text matrix is by modifying the CTM here.
317 // So, if we do that, the GPU process will have
318 // CTM * X * m_originalTextMatrix
319 // If you set these two equal to each other, and solve for X, you get
320 // CTM * currentTextMatrix = CTM * X * m_originalTextMatrix
321 // currentTextMatrix * inverse(m_originalTextMatrix) = X
322 AffineTransform currentTextMatrix = CGContextGetTextMatrix(m_internalContext->platformContext());
323 AffineTransform ctmFixup;
324 if (auto invertedOriginalTextMatrix = m_originalTextMatrix.inverse())
325 ctmFixup = currentTextMatrix * invertedOriginalTextMatrix.value();
326 AffineTransform inverseCTMFixup;
327 if (auto inverse = ctmFixup.inverse())
328 inverseCTMFixup = inverse.value();
329 else
330 ctmFixup = AffineTransform();
331 m_owner.concatCTM(ctmFixup);
332
333 updateFillColor(CGGStateGetFillColor(gstate));
334 updateStrokeColor(CGGStateGetStrokeColor(gstate));
335 updateShadow(CGGStateGetStyle(gstate));
336
337 m_owner.appendDrawGlyphsItemWithCachedFont(*m_originalFont, glyphs, computeAdvancesFromPositions(positions, count, currentTextMatrix).data(), count, currentTextMatrix.mapPoint(positions[0]), m_smoothingMode);
338
339 m_owner.concatCTM(inverseCTMFixup);
340}

342void DrawGlyphsRecorder::recordDrawImage(CGRenderingStateRef, CGGStateRef gstate, CGRect rect, CGImageRef cgImage)
343{
344 updateCTM(*CGGStateGetCTM(gstate));
345 updateShadow(CGGStateGetStyle(gstate));
346
347 // Core Graphics assumes a "y up" coordinate system, but in WebKit, we use a "y-down" coordinate system.
348 // This means that WebKit's drawing routines (GraphicsContext::drawImage()) intentionally draw images upside-down from Core Graphics's point of view.
349 // (There's a y-flip inside the implementation of GraphicsContext::drawImage().)
350 // The rect has the right bounds, but we need to transform from CG's coordinate system to WebKit's by performing our own y-flip so images are drawn the right-side-up.
351 // We do this at the boundary between the two APIs, which is right here.
352 m_owner.translate(0, rect.size.height + 2 * rect.origin.y);
353 m_owner.scale(FloatSize(1, -1));
354
355 auto image = NativeImage::create(cgImage);
356 m_owner.drawNativeImage(*image, image->size(), FloatRect(rect), FloatRect {{ }, image->size()}, ImagePaintingOptions { ImageOrientation::OriginTopLeft });
357
358 // Undo the above y-flip to restore the context.
359 m_owner.scale(FloatSize(1, -1));
360 m_owner.translate(0, -(rect.size.height + 2 * rect.origin.y));
361}

363struct GlyphsAndAdvancesStorage {
364 Vector<GlyphBufferGlyph> glyphs;
365 Vector<GlyphBufferAdvance> advances;
366};
367
368struct GlyphsAndAdvances {
369 const GlyphBufferGlyph* glyphs;
370 const GlyphBufferAdvance* advances;
371 unsigned numGlyphs;
372 GlyphBufferAdvance initialAdvance;
373 std::optional<GlyphsAndAdvancesStorage> storage;
374};
375
376static GlyphsAndAdvances filterOutOTSVGGlyphs(const Font& font, const GlyphBufferGlyph* glyphs, const GlyphBufferAdvance* advances, unsigned numGlyphs)
377{
378 auto otsvgGlyphs = font.findOTSVGGlyphs(glyphs, numGlyphs);
379 if (!otsvgGlyphs)
380 return { glyphs, advances, numGlyphs, makeGlyphBufferAdvance(), { }};
381
382 ASSERT(otsvgGlyphs->size() >= numGlyphs);
383
384 GlyphsAndAdvances result;
385 result.initialAdvance = makeGlyphBufferAdvance();
386 result.storage = GlyphsAndAdvancesStorage();
387
388 result.storage->glyphs.reserveInitialCapacity(numGlyphs);
389 result.storage->advances.reserveInitialCapacity(numGlyphs);
390
391 for (unsigned i = 0; i < numGlyphs; ++i) {
392 ASSERT(result.storage->glyphs.size() == result.storage->advances.size());
393 if (otsvgGlyphs->quickGet(i)) {
394 if (result.storage->advances.isEmpty())
395 result.initialAdvance = makeGlyphBufferAdvance(size(result.initialAdvance) + size(advances[i]));
396 else
397 result.storage->advances.last() = makeGlyphBufferAdvance(size(result.storage->advances.last()) + size(advances[i]));
398 } else {
399 result.storage->glyphs.uncheckedAppend(glyphs[i]);
400 result.storage->advances.uncheckedAppend(advances[i]);
401 }
402 ASSERT(result.storage->glyphs.size() == result.storage->advances.size());
403 }
404
405 result.glyphs = result.storage->glyphs.data();
406 result.advances = result.storage->advances.data();
407 result.numGlyphs = result.storage->glyphs.size();
408
409 return result;
410}

412void DrawGlyphsRecorder::drawGlyphs(const Font& font, const GlyphBufferGlyph* glyphs, const GlyphBufferAdvance* advances, unsigned numGlyphs, const FloatPoint& startPoint, FontSmoothingMode smoothingMode)
413{
414 if (m_drawGlyphsDeconstruction == DrawGlyphsDeconstruction::DontDeconstruct) {
415 m_owner.appendDrawGlyphsItemWithCachedFont(font, glyphs, advances, numGlyphs, startPoint, smoothingMode);
416 return;
417 }
418
419 ASSERT(m_drawGlyphsDeconstruction == DrawGlyphsDeconstruction::Deconstruct);
420
421 // FIXME: <rdar://problem/70166552> Record OTSVG glyphs.
422 GlyphsAndAdvances glyphsAndAdvancesWithoutOTSVGGlyphs = filterOutOTSVGGlyphs(font, glyphs, advances, numGlyphs);
423 ASSERT(glyphsAndAdvancesWithoutOTSVGGlyphs.glyphs == glyphs || glyphsAndAdvancesWithoutOTSVGGlyphs.glyphs == glyphsAndAdvancesWithoutOTSVGGlyphs.storage->glyphs.data());
424 ASSERT(glyphsAndAdvancesWithoutOTSVGGlyphs.advances == advances || glyphsAndAdvancesWithoutOTSVGGlyphs.advances == glyphsAndAdvancesWithoutOTSVGGlyphs.storage->advances.data());
425
426 prepareInternalContext(font, smoothingMode);
427 FontCascade::drawGlyphs(m_internalContext, font, glyphsAndAdvancesWithoutOTSVGGlyphs.glyphs, glyphsAndAdvancesWithoutOTSVGGlyphs.advances, glyphsAndAdvancesWithoutOTSVGGlyphs.numGlyphs, startPoint + size(glyphsAndAdvancesWithoutOTSVGGlyphs.initialAdvance), smoothingMode);
428 concludeInternalContext();
429}
