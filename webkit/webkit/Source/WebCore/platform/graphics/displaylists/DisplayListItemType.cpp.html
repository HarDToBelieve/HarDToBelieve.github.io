185{

188
189 switch (type) {
190 case ItemType::ApplyDeviceScaleFactor:
191#if USE(CG)
192 case ItemType::ApplyFillPattern:
193 case ItemType::ApplyStrokePattern:
194#endif
195 case ItemType::ClearShadow:
196 case ItemType::Clip:
197 case ItemType::ClipOut:
198 case ItemType::ClipToImageBuffer:
199 case ItemType::ClipOutToPath:
200 case ItemType::ClipPath:
201 case ItemType::BeginClipToDrawingCommands:
202 case ItemType::EndClipToDrawingCommands:
203 case ItemType::ConcatenateCTM:
204 case ItemType::FlushContext:
205 case ItemType::MetaCommandChangeDestinationImageBuffer:
206 case ItemType::MetaCommandChangeItemBuffer:
207 case ItemType::Restore:
208 case ItemType::Rotate:
209 case ItemType::Save:
210 case ItemType::Scale:
211 case ItemType::SetCTM:
212 case ItemType::SetInlineFillColor:
213 case ItemType::SetInlineFillGradient:
214 case ItemType::SetInlineStrokeColor:
215 case ItemType::SetLineCap:
216 case ItemType::SetLineDash:
217 case ItemType::SetLineJoin:

272size_t paddedSizeOfTypeAndItemInBytes(ItemType type)
273{
274 return sizeof(uint64_t) + roundUpToMultipleOf(alignof(uint64_t), sizeOfItemInBytes(type));
275}

277size_t paddedSizeOfTypeAndItemInBytes(const DisplayListItem& displayListItem)
278{
279 auto itemSize = WTF::visit([](const auto& item) {
280 return sizeof(item);
281 }, displayListItem);
282 return sizeof(uint64_t) + roundUpToMultipleOf(alignof(uint64_t), itemSize);
283}

285ItemType displayListItemType(const DisplayListItem& displayListItem)
286{
287 return WTF::visit([](const auto& item) {
288 return item.itemType;
289 }, displayListItem);
290}
