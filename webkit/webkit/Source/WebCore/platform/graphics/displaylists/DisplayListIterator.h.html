26#pragma once
27
28#include "DisplayList.h"
29
30namespace WebCore {
31namespace DisplayList {
32
33class DisplayList::Iterator {
34public:
35 enum class ImmediatelyMoveToEnd { No, Yes };
36 Iterator(const DisplayList& displayList, ImmediatelyMoveToEnd immediatelyMoveToEnd = ImmediatelyMoveToEnd::No)
37 : m_displayList(displayList)
38 {
39 if (immediatelyMoveToEnd == ImmediatelyMoveToEnd::Yes)
40 moveToEnd();
41 else {
42 moveCursorToStartOfCurrentBuffer();
43 updateCurrentItem();
44 }
45 }
46
47 ~Iterator()
48 {
49 clearCurrentItem();
50 }
51
52 bool operator==(const Iterator& other) { return &m_displayList == &other.m_displayList && m_cursor == other.m_cursor; }
53 bool operator!=(const Iterator& other) { return !(*this == other); }
54 void operator++() { advance(); }
55
56 struct Value {
57 ItemHandle item;
58 std::optional<FloatRect> extent;
59 size_t itemSizeInBuffer { 0 };
60 };
61
62 std::optional<Value> operator*() const
63 {
64 if (!m_isValid)
65 return std::nullopt;
66 return {{
67 ItemHandle { m_currentBufferForItem },
68 m_currentExtent,
69 m_currentItemSizeInBuffer,
70 }};
71 }
72
73private:
74 static constexpr size_t sizeOfFixedBufferForCurrentItem = 256;
75
76 WEBCORE_EXPORT void moveCursorToStartOfCurrentBuffer();
77 WEBCORE_EXPORT void moveToEnd();
78 WEBCORE_EXPORT void clearCurrentItem();
79 WEBCORE_EXPORT void updateCurrentItem();
80 WEBCORE_EXPORT void advance();
81
82 enum class ExtentUpdateResult : bool { Failure, Success };
83 ExtentUpdateResult updateCurrentDrawingItemExtent(ItemType);
84
85 bool atEnd() const;
86
87 ItemBuffer* itemBuffer() const { return m_displayList.itemBufferIfExists(); }
88
89 const DisplayList& m_displayList;
90 uint8_t* m_cursor { nullptr };
91 size_t m_readOnlyBufferIndex { 0 };
92 size_t m_drawingItemIndex { 0 };
93 uint8_t* m_currentEndOfBuffer { nullptr };
94
95 uint8_t m_fixedBufferForCurrentItem[sizeOfFixedBufferForCurrentItem] { 0 };
96 uint8_t* m_currentBufferForItem { nullptr };
97 std::optional<FloatRect> m_currentExtent;
98 size_t m_currentItemSizeInBuffer { 0 };
99 bool m_isValid { true };
100};
101
102}
