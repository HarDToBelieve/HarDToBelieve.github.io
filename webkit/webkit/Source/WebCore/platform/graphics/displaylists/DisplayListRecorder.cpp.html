79static bool containsOnlyInlineStateChanges(const GraphicsContextStateChange& changes, GraphicsContextState::StateChangeFlags changeFlags)
80{
81 static constexpr GraphicsContextState::StateChangeFlags inlineStateChangeFlags {
82 GraphicsContextState::StrokeThicknessChange,
83 GraphicsContextState::StrokeColorChange,
84 GraphicsContextState::FillColorChange,
85 GraphicsContextState::FillGradientChange,
86 };
87
88 if (changeFlags != (changeFlags & inlineStateChangeFlags))
89 return false;
90
91 if (changeFlags.contains(GraphicsContextState::StrokeColorChange) && !changes.m_state.strokeColor.tryGetAsSRGBABytes())
92 return false;
93
94 if (changeFlags.contains(GraphicsContextState::FillColorChange) && !changes.m_state.fillColor.tryGetAsSRGBABytes())
95 return false;
96
97 if (changeFlags.contains(GraphicsContextState::FillGradientChange)
98 && (!changes.m_state.fillGradient || !SetInlineFillGradient::isInline(*changes.m_state.fillGradient)))
99 return false;
100
101 return true;
102}

111void Recorder::appendStateChangeItem(const GraphicsContextStateChange& changes, GraphicsContextState::StateChangeFlags changeFlags)
112{
113 if (!containsOnlyInlineStateChanges(changes, changeFlags)) {
114 if (auto pattern = changes.m_state.strokePattern)
115 cacheNativeImage(pattern->tileImage());
116 if (auto pattern = changes.m_state.fillPattern)
117 cacheNativeImage(pattern->tileImage());
118 append<SetState>(changes.m_state, changeFlags);
119 return;
120 }
121
122 if (changeFlags.contains(GraphicsContextState::StrokeColorChange))
123 append<SetInlineStrokeColor>(*changes.m_state.strokeColor.tryGetAsSRGBABytes());
124
125 if (changeFlags.contains(GraphicsContextState::StrokeThicknessChange))
126 append<SetStrokeThickness>(changes.m_state.strokeThickness);
127
128 if (changeFlags.contains(GraphicsContextState::FillColorChange))
129 append<SetInlineFillColor>(*changes.m_state.fillColor.tryGetAsSRGBABytes());
130
131 if (changeFlags.contains(GraphicsContextState::FillGradientChange))
132 append<SetInlineFillGradient>(*changes.m_state.fillGradient, changes.m_state.fillGradientSpaceTransform);
133}

140void Recorder::appendStateChangeItemIfNecessary()
141{
142 auto& stateChanges = currentState().stateChange;
143 auto changesFromLastState = stateChanges.changesFromState(currentState().lastDrawingState);
144 if (!changesFromLastState)
145 return;
146
147 LOG_WITH_STREAM(DisplayLists, stream << "pre-drawing, saving state " << GraphicsContextStateChange(stateChanges.m_state, changesFromLastState));
148 appendStateChangeItem(stateChanges, changesFromLastState);
149 stateChanges.m_changeFlags = { };
150 currentState().lastDrawingState = stateChanges.m_state;
151}

472GraphicsContext::ClipToDrawingCommandsResult Recorder::clipToDrawingCommands(const FloatRect& destination, const DestinationColorSpace& colorSpace, Function<void(GraphicsContext&)>&& drawingFunction)
473{
474 auto initialClip = FloatRect(FloatPoint(), destination.size());
475
476 // The initial CTM matches ImageBuffer's initial CTM.
477 AffineTransform transform = getCTM(GraphicsContext::DefinitelyIncludeDeviceScale);
478 FloatSize scaleFactor(transform.xScale(), transform.yScale());
479 auto scaledSize = expandedIntSize(destination.size() * scaleFactor);
480 AffineTransform initialCTM;
481 initialCTM.scale(1, -1);
482 initialCTM.translate(0, -scaledSize.height());
483 initialCTM.scale(scaledSize / destination.size());
484
485 Recorder nestedContext(*this, GraphicsContextState(), initialClip, initialCTM);
486 append<BeginClipToDrawingCommands>(destination, colorSpace);
487 drawingFunction(nestedContext);
488 append<EndClipToDrawingCommands>(destination);
489
490 return ClipToDrawingCommandsResult::Success;
491}

518// FIXME: share with ShadowData
519static inline float shadowPaintingExtent(float blurRadius)
520{
521 // Blurring uses a Gaussian function whose std. deviation is m_radius/2, and which in theory
522 // extends to infinity. In 8-bit contexts, however, rounding causes the effect to become
523 // undetectable at around 1.4x the radius.
524 const float radiusExtentMultiplier = 1.4;
525 return ceilf(blurRadius * radiusExtentMultiplier);
526}

528FloatRect Recorder::extentFromLocalBounds(const FloatRect& rect) const
529{
530 FloatRect bounds = rect;
531 const ContextState& state = currentState();
532
533 FloatSize shadowOffset;
534 float shadowRadius;
535 Color shadowColor;
536 if (getShadow(shadowOffset, shadowRadius, shadowColor)) {
537 FloatRect shadowExtent = bounds;
538 shadowExtent.move(shadowOffset);
539 shadowExtent.inflate(shadowPaintingExtent(shadowRadius));
540 bounds.unite(shadowExtent);
541 }
542
543 FloatRect clippedExtent = intersection(state.clipBounds, bounds);
544 return state.ctm.mapRect(clippedExtent);
545}

570void Recorder::ContextState::rotate(float angleInRadians)
571{
572 double angleInDegrees = rad2deg(static_cast<double>(angleInRadians));
573 ctm.rotate(angleInDegrees);
574
575 AffineTransform rotation;
576 rotation.rotate(angleInDegrees);
577
578 if (std::optional<AffineTransform> inverse = rotation.inverse())
579 clipBounds = inverse.value().mapRect(clipBounds);
580}

588void Recorder::ContextState::setCTM(const AffineTransform& matrix)
589{
590 std::optional<AffineTransform> inverseTransformForClipBounds;
591 if (auto originalCTMInverse = ctm.inverse())
592 inverseTransformForClipBounds = originalCTMInverse->multiply(matrix).inverse();
593
594 ctm = matrix;
595
596 if (inverseTransformForClipBounds)
597 clipBounds = inverseTransformForClipBounds->mapRect(clipBounds);
598}
599
600void Recorder::ContextState::concatCTM(const AffineTransform& matrix)
601{
602 ctm *= matrix;
603
604 if (std::optional<AffineTransform> inverse = matrix.inverse())
605 clipBounds = inverse.value().mapRect(clipBounds);
606}
