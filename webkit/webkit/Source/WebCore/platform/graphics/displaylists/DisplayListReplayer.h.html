38namespace DisplayList {
39
40enum class StopReplayReason : uint8_t {
41 ReplayedAllItems,
42 MissingCachedResource,
43 ChangeDestinationImageBuffer,
44 InvalidItemOrExtent,
45 OutOfMemory
46};
47
48struct ReplayResult {
49 std::unique_ptr<InMemoryDisplayList> trackedDisplayList;
50 size_t numberOfBytesRead { 0 };
51 std::optional<RenderingResourceIdentifier> nextDestinationImageBuffer;
52 std::optional<RenderingResourceIdentifier> missingCachedResourceIdentifier;
53 StopReplayReason reasonForStopping { StopReplayReason::ReplayedAllItems };
54};
55
56class Replayer {
57 WTF_MAKE_NONCOPYABLE(Replayer);
58public:
59 class Delegate;
60 WEBCORE_EXPORT Replayer(GraphicsContext&, const DisplayList&, const ImageBufferHashMap* = nullptr, const NativeImageHashMap* = nullptr, const FontRenderingResourceMap* = nullptr, WebCore::ImageBuffer* maskImageBuffer = nullptr, Delegate* = nullptr);
61 WEBCORE_EXPORT ~Replayer();
62
63 WEBCORE_EXPORT ReplayResult replay(const FloatRect& initialClip = { }, bool trackReplayList = false);
64
65 class Delegate {
66 public:
67 virtual ~Delegate() { }
68 virtual bool apply(ItemHandle, GraphicsContext&) { return false; }
69 virtual void didCreateMaskImageBuffer(WebCore::ImageBuffer&) { }
70 virtual void didResetMaskImageBuffer() { }
71 };
72
73private:
74 GraphicsContext& context() const;
75 std::pair<std::optional<StopReplayReason>, std::optional<RenderingResourceIdentifier>> applyItem(ItemHandle);
76
77 GraphicsContext& m_context;
78 RefPtr<WebCore::ImageBuffer> m_maskImageBuffer;
79 const DisplayList& m_displayList;
80 const ImageBufferHashMap& m_imageBuffers;
81 const NativeImageHashMap& m_nativeImages;
82 const FontRenderingResourceMap& m_fonts;
83 Delegate* m_delegate;
84};
