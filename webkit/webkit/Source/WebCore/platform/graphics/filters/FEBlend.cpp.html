37FEBlend::FEBlend(Filter& filter, BlendMode mode)
38 : FilterEffect(filter, Type::Blend)
39 , m_mode(mode)
40{
41}
42
43Ref<FEBlend> FEBlend::create(Filter& filter, BlendMode mode)
44{
45 return adoptRef(*new FEBlend(filter, mode));
46}
47
48bool FEBlend::setBlendMode(BlendMode mode)
49{
50 if (m_mode == mode)
51 return false;
52 m_mode = mode;
53 return true;
54}

56#if !HAVE(ARM_NEON_INTRINSICS)
57void FEBlend::platformApplySoftware()
58{
59 FilterEffect* in = inputEffect(0);
60 FilterEffect* in2 = inputEffect(1);
61
62 ImageBuffer* resultImage = createImageBufferResult();
63 if (!resultImage)
64 return;
65 GraphicsContext& filterContext = resultImage->context();
66
67 ImageBuffer* imageBuffer = in->imageBufferResult();
68 ImageBuffer* imageBuffer2 = in2->imageBufferResult();
69 if (!imageBuffer || !imageBuffer2)
70 return;
71
72 filterContext.drawImageBuffer(*imageBuffer2, drawingRegionOfInputImage(in2->absolutePaintRect()));
73 filterContext.drawImageBuffer(*imageBuffer, drawingRegionOfInputImage(in->absolutePaintRect()), IntRect(IntPoint(), imageBuffer->logicalSize()), { CompositeOperator::SourceOver, m_mode });
74}
75#endif

77TextStream& FEBlend::externalRepresentation(TextStream& ts, RepresentationType representation) const
78{
79 ts << indent << "[feBlend";
80 FilterEffect::externalRepresentation(ts, representation);
81 ts << " mode=\"" << (m_mode == BlendMode::Normal ? "normal" : compositeOperatorName(CompositeOperator::SourceOver, m_mode)) << "\"]\n";
82
83 TextStream::IndentScope indentScope(ts);
84 inputEffect(0)->externalRepresentation(ts, representation);
85 inputEffect(1)->externalRepresentation(ts, representation);
86 return ts;
87}
