66inline void matrix(float& red, float& green, float& blue, float& alpha, const Vector<float>& values)
67{
68 float r = red;
69 float g = green;
70 float b = blue;
71 float a = alpha;
72
73 red = values[ 0] * r + values[ 1] * g + values[ 2] * b + values[ 3] * a + values[ 4] * 255;
74 green = values[ 5] * r + values[ 6] * g + values [7] * b + values[ 8] * a + values[ 9] * 255;
75 blue = values[10] * r + values[11] * g + values[12] * b + values[13] * a + values[14] * 255;
76 alpha = values[15] * r + values[16] * g + values[17] * b + values[18] * a + values[19] * 255;
77}
78
79inline void saturateAndHueRotate(float& red, float& green, float& blue, const float* components)
80{
81 float r = red;
82 float g = green;
83 float b = blue;
84
85 red = r * components[0] + g * components[1] + b * components[2];
86 green = r * components[3] + g * components[4] + b * components[5];
87 blue = r * components[6] + g * components[7] + b * components[8];
88}
89
90// FIXME: this should use the luminance(...) function in ColorLuminance.h.
91inline void luminance(float& red, float& green, float& blue, float& alpha)
92{
93 alpha = 0.2125 * red + 0.7154 * green + 0.0721 * blue;
94 red = 0;
95 green = 0;
96 blue = 0;
97}

99#if USE(ACCELERATE)
100template<ColorMatrixType filterType>
101bool effectApplyAccelerated(Uint8ClampedArray& pixelArray, const Vector<float>& values, float components[9], IntSize bufferSize)
102{
103 ASSERT(pixelArray.length() == bufferSize.area() * 4);
104
105 if (filterType == FECOLORMATRIX_TYPE_MATRIX) {
106 // vImageMatrixMultiply_ARGB8888 takes a 4x4 matrix, if any value in the last column of the FEColorMatrix 5x4 matrix
107 // is not zero, fall back to non-vImage code.
108 if (values[4] != 0 || values[9] != 0 || values[14] != 0 || values[19] != 0)
109 return false;
110 }
111
112 const int32_t divisor = 256;
113 uint8_t* data = pixelArray.data();
114
115 vImage_Buffer src;
116 src.width = bufferSize.width();
117 src.height = bufferSize.height();
118 src.rowBytes = bufferSize.width() * 4;
119 src.data = data;
120
121 vImage_Buffer dest;
122 dest.width = bufferSize.width();
123 dest.height = bufferSize.height();
124 dest.rowBytes = bufferSize.width() * 4;
125 dest.data = data;
126
127 switch (filterType) {
128 case FECOLORMATRIX_TYPE_MATRIX: {
129 const int16_t matrix[4 * 4] = {
130 static_cast<int16_t>(roundf(values[ 0] * divisor)),
131 static_cast<int16_t>(roundf(values[ 5] * divisor)),
132 static_cast<int16_t>(roundf(values[10] * divisor)),
133 static_cast<int16_t>(roundf(values[15] * divisor)),
134
135 static_cast<int16_t>(roundf(values[ 1] * divisor)),
136 static_cast<int16_t>(roundf(values[ 6] * divisor)),
137 static_cast<int16_t>(roundf(values[11] * divisor)),
138 static_cast<int16_t>(roundf(values[16] * divisor)),
139
140 static_cast<int16_t>(roundf(values[ 2] * divisor)),
141 static_cast<int16_t>(roundf(values[ 7] * divisor)),
142 static_cast<int16_t>(roundf(values[12] * divisor)),
143 static_cast<int16_t>(roundf(values[17] * divisor)),
144
145 static_cast<int16_t>(roundf(values[ 3] * divisor)),
146 static_cast<int16_t>(roundf(values[ 8] * divisor)),
147 static_cast<int16_t>(roundf(values[13] * divisor)),
148 static_cast<int16_t>(roundf(values[18] * divisor)),
149 };
150 vImageMatrixMultiply_ARGB8888(&src, &dest, matrix, divisor, nullptr, nullptr, kvImageNoFlags);
151 break;
152 }
153
154 case FECOLORMATRIX_TYPE_SATURATE:
155 case FECOLORMATRIX_TYPE_HUEROTATE: {
156 const int16_t matrix[4 * 4] = {
157 static_cast<int16_t>(roundf(components[0] * divisor)),
158 static_cast<int16_t>(roundf(components[3] * divisor)),
159 static_cast<int16_t>(roundf(components[6] * divisor)),
160 0,
161
162 static_cast<int16_t>(roundf(components[1] * divisor)),
163 static_cast<int16_t>(roundf(components[4] * divisor)),
164 static_cast<int16_t>(roundf(components[7] * divisor)),
165 0,
166
167 static_cast<int16_t>(roundf(components[2] * divisor)),
168 static_cast<int16_t>(roundf(components[5] * divisor)),
169 static_cast<int16_t>(roundf(components[8] * divisor)),
170 0,
171
172 0,
173 0,
174 0,
175 divisor,
176 };
177 vImageMatrixMultiply_ARGB8888(&src, &dest, matrix, divisor, nullptr, nullptr, kvImageNoFlags);
178 break;
179 }
180 case FECOLORMATRIX_TYPE_LUMINANCETOALPHA: {
181 const int16_t matrix[4 * 4] = {
182 0,
183 0,
184 0,
185 static_cast<int16_t>(roundf(0.2125 * divisor)),
186
187 0,
188 0,
189 0,
190 static_cast<int16_t>(roundf(0.7154 * divisor)),
191
192 0,
193 0,
194 0,
195 static_cast<int16_t>(roundf(0.0721 * divisor)),
196
197 0,
198 0,
199 0,
200 0,
201 };
202 vImageMatrixMultiply_ARGB8888(&src, &dest, matrix, divisor, nullptr, nullptr, kvImageNoFlags);
203 break;
204 }
205 }
206
207 return true;
208}

211template<ColorMatrixType filterType>
212void effectType(Uint8ClampedArray& pixelArray, const Vector<float>& values, IntSize bufferSize)
213{
214 float components[9];
215
216 if (filterType == FECOLORMATRIX_TYPE_SATURATE)
217 FEColorMatrix::calculateSaturateComponents(components, values[0]);
218 else if (filterType == FECOLORMATRIX_TYPE_HUEROTATE)
219 FEColorMatrix::calculateHueRotateComponents(components, values[0]);
220
221 unsigned pixelArrayLength = pixelArray.length();
222
223#if USE(ACCELERATE)
224 if (effectApplyAccelerated<filterType>(pixelArray, values, components, bufferSize))
225 return;
226#else
227 UNUSED_PARAM(bufferSize);
228#endif
229
230 switch (filterType) {
231 case FECOLORMATRIX_TYPE_MATRIX:
232 for (unsigned pixelByteOffset = 0; pixelByteOffset < pixelArrayLength; pixelByteOffset += 4) {
233 float red = pixelArray.item(pixelByteOffset);
234 float green = pixelArray.item(pixelByteOffset + 1);
235 float blue = pixelArray.item(pixelByteOffset + 2);
236 float alpha = pixelArray.item(pixelByteOffset + 3);
237 matrix(red, green, blue, alpha, values);
238 pixelArray.set(pixelByteOffset, red);
239 pixelArray.set(pixelByteOffset + 1, green);
240 pixelArray.set(pixelByteOffset + 2, blue);
241 pixelArray.set(pixelByteOffset + 3, alpha);
242 }
243 break;
244
245 case FECOLORMATRIX_TYPE_SATURATE:
246 case FECOLORMATRIX_TYPE_HUEROTATE:
247 for (unsigned pixelByteOffset = 0; pixelByteOffset < pixelArrayLength; pixelByteOffset += 4) {
248 float red = pixelArray.item(pixelByteOffset);
249 float green = pixelArray.item(pixelByteOffset + 1);
250 float blue = pixelArray.item(pixelByteOffset + 2);
251 float alpha = pixelArray.item(pixelByteOffset + 3);
252 saturateAndHueRotate(red, green, blue, components);
253 pixelArray.set(pixelByteOffset, red);
254 pixelArray.set(pixelByteOffset + 1, green);
255 pixelArray.set(pixelByteOffset + 2, blue);
256 pixelArray.set(pixelByteOffset + 3, alpha);
257 }
258 break;
259
260 case FECOLORMATRIX_TYPE_LUMINANCETOALPHA:
261 for (unsigned pixelByteOffset = 0; pixelByteOffset < pixelArrayLength; pixelByteOffset += 4) {
262 float red = pixelArray.item(pixelByteOffset);
263 float green = pixelArray.item(pixelByteOffset + 1);
264 float blue = pixelArray.item(pixelByteOffset + 2);
265 float alpha = pixelArray.item(pixelByteOffset + 3);
266 luminance(red, green, blue, alpha);
267 pixelArray.set(pixelByteOffset, red);
268 pixelArray.set(pixelByteOffset + 1, green);
269 pixelArray.set(pixelByteOffset + 2, blue);
270 pixelArray.set(pixelByteOffset + 3, alpha);
271 }
272 break;
273 }
274}

276void FEColorMatrix::platformApplySoftware()
277{
278 FilterEffect* in = inputEffect(0);
279
280 ImageBuffer* resultImage = createImageBufferResult();
281 if (!resultImage)
282 return;
283
284 ImageBuffer* inBuffer = in->imageBufferResult();
285 if (inBuffer)
286 resultImage->context().drawImageBuffer(*inBuffer, drawingRegionOfInputImage(in->absolutePaintRect()));
287
288 PixelBufferFormat format { AlphaPremultiplication::Unpremultiplied, PixelFormat::RGBA8, resultColorSpace() };
289 IntRect imageRect(IntPoint(), resultImage->logicalSize());
290 auto pixelBuffer = resultImage->getPixelBuffer(format, imageRect);
291 if (!pixelBuffer)
292 return;
293
294 auto& pixelArray = pixelBuffer->data();
295 auto pixelArrayDimensions = pixelBuffer->size();
296
297 Vector<float> values = normalizedFloats(m_values);
298
299 switch (m_type) {
300 case FECOLORMATRIX_TYPE_UNKNOWN:
301 break;
302 case FECOLORMATRIX_TYPE_MATRIX:
303 effectType<FECOLORMATRIX_TYPE_MATRIX>(pixelArray, values, pixelArrayDimensions);
304 break;
305 case FECOLORMATRIX_TYPE_SATURATE:
306 effectType<FECOLORMATRIX_TYPE_SATURATE>(pixelArray, values, pixelArrayDimensions);
307 break;
308 case FECOLORMATRIX_TYPE_HUEROTATE:
309 effectType<FECOLORMATRIX_TYPE_HUEROTATE>(pixelArray, values, pixelArrayDimensions);
310 break;
311 case FECOLORMATRIX_TYPE_LUMINANCETOALPHA:
312 effectType<FECOLORMATRIX_TYPE_LUMINANCETOALPHA>(pixelArray, values, pixelArrayDimensions);
313 setIsAlphaImage(true);
314 break;
315 }
316
317 resultImage->putPixelBuffer(*pixelBuffer, imageRect);
318}

338 }
339 return ts;

342TextStream& FEColorMatrix::externalRepresentation(TextStream& ts, RepresentationType representation) const
343{
344 ts << indent << "[feColorMatrix";
345 FilterEffect::externalRepresentation(ts, representation);
346 ts << " type=\"" << m_type << "\"";
347 if (!m_values.isEmpty()) {
348 ts << " values=\"";
349 Vector<float>::const_iterator ptr = m_values.begin();
350 const Vector<float>::const_iterator end = m_values.end();
351 while (ptr < end) {
352 ts << *ptr;
353 ++ptr;
354 if (ptr < end)
355 ts << " ";
356 }
357 ts << "\"";
358 }
359 ts << "]\n";
360
361 TextStream::IndentScope indentScope(ts);
362 inputEffect(0)->externalRepresentation(ts, representation);
363 return ts;
364}
