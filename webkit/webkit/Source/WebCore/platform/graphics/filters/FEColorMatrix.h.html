31enum ColorMatrixType {
32 FECOLORMATRIX_TYPE_UNKNOWN = 0,
33 FECOLORMATRIX_TYPE_MATRIX = 1,
34 FECOLORMATRIX_TYPE_SATURATE = 2,
35 FECOLORMATRIX_TYPE_HUEROTATE = 3,
36 FECOLORMATRIX_TYPE_LUMINANCETOALPHA = 4
37};

39class FEColorMatrix : public FilterEffect {
40public:
41 static Ref<FEColorMatrix> create(Filter&, ColorMatrixType, Vector<float>&&);
42
43 ColorMatrixType type() const { return m_type; }
44 bool setType(ColorMatrixType);
45
46 const Vector<float>& values() const { return m_values; }
47 bool setValues(const Vector<float>&);

58

61 ColorMatrixType m_type;

63};

65inline void FEColorMatrix::calculateSaturateComponents(float* components, float value)
66{
67 components[0] = (0.213 + 0.787 * value);
68 components[1] = (0.715 - 0.715 * value);
69 components[2] = (0.072 - 0.072 * value);
70 components[3] = (0.213 - 0.213 * value);
71 components[4] = (0.715 + 0.285 * value);
72 components[5] = (0.072 - 0.072 * value);
73 components[6] = (0.213 - 0.213 * value);
74 components[7] = (0.715 - 0.715 * value);
75 components[8] = (0.072 + 0.928 * value);
76}
77
78inline void FEColorMatrix::calculateHueRotateComponents(float* components, float value)
79{
80 float cosHue = cos(value * piFloat / 180);
81 float sinHue = sin(value * piFloat / 180);
82 components[0] = 0.213 + cosHue * 0.787 - sinHue * 0.213;
83 components[1] = 0.715 - cosHue * 0.715 - sinHue * 0.715;
84 components[2] = 0.072 - cosHue * 0.072 + sinHue * 0.928;
85 components[3] = 0.213 - cosHue * 0.213 + sinHue * 0.143;
86 components[4] = 0.715 + cosHue * 0.285 + sinHue * 0.140;
87 components[5] = 0.072 - cosHue * 0.072 - sinHue * 0.283;
88 components[6] = 0.213 - cosHue * 0.213 - sinHue * 0.787;
89 components[7] = 0.715 - cosHue * 0.715 + sinHue * 0.715;
90 components[8] = 0.072 + cosHue * 0.928 + sinHue * 0.072;
91}
