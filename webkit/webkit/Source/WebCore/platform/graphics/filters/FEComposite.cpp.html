98static unsigned char clampByte(int c)
99{
100 unsigned char buff[] = { static_cast<unsigned char>(c), 255, 0 };
101 unsigned uc = static_cast<unsigned>(c);
102 return buff[!!(uc & ~0xff) + !!(uc & ~(~0u >> 1))];
103}

105template <int b1, int b4>
106static inline void computeArithmeticPixels(unsigned char* source, unsigned char* destination, int pixelArrayLength, float k1, float k2, float k3, float k4)
107{
108 float scaledK1;
109 float scaledK4;
110 if (b1)
111 scaledK1 = k1 / 255.0f;
112 if (b4)
113 scaledK4 = k4 * 255.0f;
114
115 while (--pixelArrayLength >= 0) {
116 unsigned char i1 = *source;
117 unsigned char i2 = *destination;
118 float result = k2 * i1 + k3 * i2;
119 if (b1)
120 result += scaledK1 * i1 * i2;
121 if (b4)
122 result += scaledK4;
123
124 *destination = clampByte(result);
125 ++source;
126 ++destination;
127 }
128}

130// computeArithmeticPixelsUnclamped is a faster version of computeArithmeticPixels for the common case where clamping
131// is not necessary. This enables aggresive compiler optimizations such as auto-vectorization.
132template <int b1, int b4>
133static inline void computeArithmeticPixelsUnclamped(unsigned char* source, unsigned char* destination, int pixelArrayLength, float k1, float k2, float k3, float k4)
134{
135 float scaledK1;
136 float scaledK4;
137 if (b1)
138 scaledK1 = k1 / 255.0f;
139 if (b4)
140 scaledK4 = k4 * 255.0f;
141
142 while (--pixelArrayLength >= 0) {
143 unsigned char i1 = *source;
144 unsigned char i2 = *destination;
145 float result = k2 * i1 + k3 * i2;
146 if (b1)
147 result += scaledK1 * i1 * i2;
148 if (b4)
149 result += scaledK4;
150
151 *destination = result;
152 ++source;
153 ++destination;
154 }
155}

157#if !HAVE(ARM_NEON_INTRINSICS)
158static inline void arithmeticSoftware(unsigned char* source, unsigned char* destination, int pixelArrayLength, float k1, float k2, float k3, float k4)
159{
160 float upperLimit = std::max(0.0f, k1) + std::max(0.0f, k2) + std::max(0.0f, k3) + k4;
161 float lowerLimit = std::min(0.0f, k1) + std::min(0.0f, k2) + std::min(0.0f, k3) + k4;
162 if ((k4 >= 0.0f && k4 <= 1.0f) && (upperLimit >= 0.0f && upperLimit <= 1.0f) && (lowerLimit >= 0.0f && lowerLimit <= 1.0f)) {
163 if (k4) {
164 if (k1)
165 computeArithmeticPixelsUnclamped<1, 1>(source, destination, pixelArrayLength, k1, k2, k3, k4);
166 else
167 computeArithmeticPixelsUnclamped<0, 1>(source, destination, pixelArrayLength, k1, k2, k3, k4);
168 } else {
169 if (k1)
170 computeArithmeticPixelsUnclamped<1, 0>(source, destination, pixelArrayLength, k1, k2, k3, k4);
171 else
172 computeArithmeticPixelsUnclamped<0, 0>(source, destination, pixelArrayLength, k1, k2, k3, k4);
173 }
174 return;
175 }
176
177 if (k4) {
178 if (k1)
179 computeArithmeticPixels<1, 1>(source, destination, pixelArrayLength, k1, k2, k3, k4);
180 else
181 computeArithmeticPixels<0, 1>(source, destination, pixelArrayLength, k1, k2, k3, k4);
182 } else {
183 if (k1)
184 computeArithmeticPixels<1, 0>(source, destination, pixelArrayLength, k1, k2, k3, k4);
185 else
186 computeArithmeticPixels<0, 0>(source, destination, pixelArrayLength, k1, k2, k3, k4);
187 }
188}
189#endif

191inline void FEComposite::platformArithmeticSoftware(const Uint8ClampedArray& source, Uint8ClampedArray& destination, float k1, float k2, float k3, float k4)
192{
193 int length = source.length();
194 ASSERT(length == static_cast<int>(destination.length()));
195 // The selection here eventually should happen dynamically.
196#if HAVE(ARM_NEON_INTRINSICS)
197 ASSERT(!(length & 0x3));
198 platformArithmeticNeon(source.data(), destination.data(), length, k1, k2, k3, k4);
199#else
200 arithmeticSoftware(source.data(), destination.data(), length, k1, k2, k3, k4);
201#endif
202}

204void FEComposite::determineAbsolutePaintRect()
205{
206 switch (m_type) {
207 case FECOMPOSITE_OPERATOR_IN:
208 case FECOMPOSITE_OPERATOR_ATOP:
209 // For In and Atop the first effect just influences the result of
210 // the second effect. So just use the absolute paint rect of the second effect here.
211 setAbsolutePaintRect(inputEffect(1)->absolutePaintRect());
212 clipAbsolutePaintRect();
213 return;
214 case FECOMPOSITE_OPERATOR_ARITHMETIC:
215 // Arithmetic may influnce the compele filter primitive region. So we can't
216 // optimize the paint region here.
217 setAbsolutePaintRect(enclosingIntRect(maxEffectRect()));
218 return;
219 default:
220 // Take the union of both input effects.
221 FilterEffect::determineAbsolutePaintRect();
222 return;
223 }
224}

226void FEComposite::platformApplySoftware()
227{
228 FilterEffect* in = inputEffect(0);
229 FilterEffect* in2 = inputEffect(1);
230
231 if (m_type == FECOMPOSITE_OPERATOR_ARITHMETIC) {
232 auto& destinationPixelBuffer = createPremultipliedImageResult();
233 if (!destinationPixelBuffer)
234 return;
235
236 auto& destinationPixelArray = destinationPixelBuffer->data();
237
238 IntRect effectADrawingRect = requestedRegionOfInputPixelBuffer(in->absolutePaintRect());
239 auto sourcePixelArray = in->premultipliedResult(effectADrawingRect, operatingColorSpace());
240 if (!sourcePixelArray)
241 return;
242
243 IntRect effectBDrawingRect = requestedRegionOfInputPixelBuffer(in2->absolutePaintRect());
244 in2->copyPremultipliedResult(destinationPixelArray, effectBDrawingRect, operatingColorSpace());
245
246 platformArithmeticSoftware(*sourcePixelArray, destinationPixelArray, m_k1, m_k2, m_k3, m_k4);
247 return;
248 }
249
250 ImageBuffer* resultImage = createImageBufferResult();
251 if (!resultImage)
252 return;
253 GraphicsContext& filterContext = resultImage->context();
254
255 ImageBuffer* imageBuffer = in->imageBufferResult();
256 ImageBuffer* imageBuffer2 = in2->imageBufferResult();
257 if (!imageBuffer || !imageBuffer2)
258 return;
259
260 switch (m_type) {
261 case FECOMPOSITE_OPERATOR_OVER:
262 filterContext.drawImageBuffer(*imageBuffer2, drawingRegionOfInputImage(in2->absolutePaintRect()));
263 filterContext.drawImageBuffer(*imageBuffer, drawingRegionOfInputImage(in->absolutePaintRect()));
264 break;
265 case FECOMPOSITE_OPERATOR_IN: {
266 // Applies only to the intersected region.
267 IntRect destinationRect = in->absolutePaintRect();
268 destinationRect.intersect(in2->absolutePaintRect());
269 destinationRect.intersect(absolutePaintRect());
270 if (destinationRect.isEmpty())
271 break;
272 IntRect adjustedDestinationRect = destinationRect - absolutePaintRect().location();
273 IntRect sourceRect = destinationRect - in->absolutePaintRect().location();
274 IntRect source2Rect = destinationRect - in2->absolutePaintRect().location();
275 filterContext.drawImageBuffer(*imageBuffer2, FloatRect(adjustedDestinationRect), FloatRect(source2Rect));
276 filterContext.drawImageBuffer(*imageBuffer, FloatRect(adjustedDestinationRect), FloatRect(sourceRect), { CompositeOperator::SourceIn });
277 break;
278 }
279 case FECOMPOSITE_OPERATOR_OUT:
280 filterContext.drawImageBuffer(*imageBuffer, drawingRegionOfInputImage(in->absolutePaintRect()));
281 filterContext.drawImageBuffer(*imageBuffer2, drawingRegionOfInputImage(in2->absolutePaintRect()), IntRect(IntPoint(), imageBuffer2->logicalSize()), CompositeOperator::DestinationOut);
282 break;
283 case FECOMPOSITE_OPERATOR_ATOP:
284 filterContext.drawImageBuffer(*imageBuffer2, drawingRegionOfInputImage(in2->absolutePaintRect()));
285 filterContext.drawImageBuffer(*imageBuffer, drawingRegionOfInputImage(in->absolutePaintRect()), IntRect(IntPoint(), imageBuffer->logicalSize()), CompositeOperator::SourceAtop);
286 break;
287 case FECOMPOSITE_OPERATOR_XOR:
288 filterContext.drawImageBuffer(*imageBuffer2, drawingRegionOfInputImage(in2->absolutePaintRect()));
289 filterContext.drawImageBuffer(*imageBuffer, drawingRegionOfInputImage(in->absolutePaintRect()), IntRect(IntPoint(), imageBuffer->logicalSize()), CompositeOperator::XOR);
290 break;
291 case FECOMPOSITE_OPERATOR_LIGHTER:
292 filterContext.drawImageBuffer(*imageBuffer2, drawingRegionOfInputImage(in2->absolutePaintRect()));
293 filterContext.drawImageBuffer(*imageBuffer, drawingRegionOfInputImage(in->absolutePaintRect()), IntRect(IntPoint(), imageBuffer->logicalSize()), CompositeOperator::PlusLighter);
294 break;
295 default:
296 break;
297 }
298}

300static TextStream& operator<<(TextStream& ts, const CompositeOperationType& type)
301{
302 switch (type) {
303 case FECOMPOSITE_OPERATOR_UNKNOWN:
304 ts << "UNKNOWN";
305 break;
306 case FECOMPOSITE_OPERATOR_OVER:
307 ts << "OVER";
308 break;
309 case FECOMPOSITE_OPERATOR_IN:
310 ts << "IN";
311 break;
312 case FECOMPOSITE_OPERATOR_OUT:
313 ts << "OUT";
314 break;
315 case FECOMPOSITE_OPERATOR_ATOP:
316 ts << "ATOP";
317 break;
318 case FECOMPOSITE_OPERATOR_XOR:
319 ts << "XOR";
320 break;
321 case FECOMPOSITE_OPERATOR_ARITHMETIC:
322 ts << "ARITHMETIC";
323 break;
324 case FECOMPOSITE_OPERATOR_LIGHTER:
325 ts << "LIGHTER";
326 break;
327 }
328 return ts;

331TextStream& FEComposite::externalRepresentation(TextStream& ts, RepresentationType representation) const
332{
333 ts << indent << "[feComposite";
334 FilterEffect::externalRepresentation(ts, representation);
335 ts << " operation=\"" << m_type << "\"";
336 if (m_type == FECOMPOSITE_OPERATOR_ARITHMETIC)
337 ts << " k1=\"" << m_k1 << "\" k2=\"" << m_k2 << "\" k3=\"" << m_k3 << "\" k4=\"" << m_k4 << "\"";
338 ts << "]\n";
339
340 TextStream::IndentScope indentScope(ts);
341 inputEffect(0)->externalRepresentation(ts, representation);
342 inputEffect(1)->externalRepresentation(ts, representation);
343 return ts;
344}
