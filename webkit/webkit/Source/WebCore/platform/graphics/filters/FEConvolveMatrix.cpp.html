123/*
124 -----------------------------------
125 ConvolveMatrix implementation
126 -----------------------------------
127
128 The image rectangle is split in the following way:
129
130 +---------------------+
131 | A |
132 +---------------------+
133 | | | |
134 | B | C | D |
135 | | | |
136 +---------------------+
137 | E |
138 +---------------------+
139
140 Where region C contains those pixels, whose values
141 can be calculated without crossing the edge of the rectangle.
142
143 Example:
144 Image size: width: 10, height: 10
145
146 Order (kernel matrix size): width: 3, height 4
147 Target: x:1, y:3
148
149 The following figure shows the target inside the kernel matrix:
150
151 ...
152 ...
153 ...
154 .X.
155
156 The regions in this case are the following:
157 Note: (x1, y1) top-left and (x2, y2) is the bottom-right corner
158 Note: row x2 and column y2 is not part of the region
159 only those (x, y) pixels, where x1 <= x < x2 and y1 <= y < y2
160
161 Region A: x1: 0, y1: 0, x2: 10, y2: 3
162 Region B: x1: 0, y1: 3, x2: 1, y2: 10
163 Region C: x1: 1, y1: 3, x2: 9, y2: 10
164 Region D: x1: 9, y1: 3, x2: 10, y2: 10
165 Region E: x1: 0, y1: 10, x2: 10, y2: 10 (empty region)
166
167 Since region C (often) contains most of the pixels, we implemented
168 a fast algoritm to calculate these values, called fastSetInteriorPixels.
169 For other regions, fastSetOuterPixels is used, which calls getPixelValue,
170 to handle pixels outside of the image. In a rare situations, when
171 kernel matrix is bigger than the image, all pixels are calculated by this
172 function.
173
174 Although these two functions have lot in common, I decided not to make
175 common a template for them, since there are key differences as well,
176 and would make it really hard to understand.
177*/

179static ALWAYS_INLINE unsigned char clampRGBAValue(float channel, unsigned char max = 255)
180{
181 if (channel <= 0)
182 return 0;
183 if (channel >= max)
184 return max;
185 return channel;
186}

188template<bool preserveAlphaValues>
189ALWAYS_INLINE void setDestinationPixels(const Uint8ClampedArray& sourcePixels, Uint8ClampedArray& destPixels, int& pixel, float* totals, float divisor, float bias)
190{
191 unsigned char maxAlpha = preserveAlphaValues ? 255 : clampRGBAValue(totals[3] / divisor + bias);
192 for (int i = 0; i < 3; ++i)
193 destPixels.set(pixel++, clampRGBAValue(totals[i] / divisor + bias, maxAlpha));
194
195 if (preserveAlphaValues) {
196 destPixels.set(pixel, sourcePixels.item(pixel));
197 ++pixel;
198 } else
199 destPixels.set(pixel++, maxAlpha);
200}

208// Only for region C
209template<bool preserveAlphaValues>
210ALWAYS_INLINE void FEConvolveMatrix::fastSetInteriorPixels(PaintingData& paintingData, int clipRight, int clipBottom, int yStart, int yEnd)
211{
212 // edge mode does not affect these pixels
213 int pixel = (m_targetOffset.y() * paintingData.width + m_targetOffset.x()) * 4;
214 int kernelIncrease = clipRight * 4;
215 int xIncrease = (m_kernelSize.width() - 1) * 4;
216 // Contains the sum of rgb(a) components
217 float totals[3 + (preserveAlphaValues ? 0 : 1)];
218
219 // m_divisor cannot be 0, SVGFEConvolveMatrixElement ensures this
220 ASSERT(m_divisor);
221
222 // Skip the first '(clipBottom - yEnd)' lines
223 pixel += (clipBottom - yEnd) * (xIncrease + (clipRight + 1) * 4);
224 int startKernelPixel = (clipBottom - yEnd) * (xIncrease + (clipRight + 1) * 4);
225
226 for (int y = yEnd + 1; y > yStart; --y) {
227 for (int x = clipRight + 1; x > 0; --x) {
228 int kernelValue = paintingData.kernelMatrix.size() - 1;
229 int kernelPixel = startKernelPixel;
230 int width = m_kernelSize.width();
231
232 totals[0] = 0;
233 totals[1] = 0;
234 totals[2] = 0;
235 if (!preserveAlphaValues)
236 totals[3] = 0;
237
238 while (kernelValue >= 0) {
239 totals[0] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(kernelPixel++));
240 totals[1] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(kernelPixel++));
241 totals[2] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(kernelPixel++));
242 if (!preserveAlphaValues)
243 totals[3] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(kernelPixel));
244 ++kernelPixel;
245 --kernelValue;
246 if (!--width) {
247 kernelPixel += kernelIncrease;
248 width = m_kernelSize.width();
249 }
250 }
251
252 setDestinationPixels<preserveAlphaValues>(paintingData.srcPixelArray, paintingData.dstPixelArray, pixel, totals, m_divisor, paintingData.bias);
253 startKernelPixel += 4;
254 }
255 pixel += xIncrease;
256 startKernelPixel += xIncrease;
257 }
258}

260ALWAYS_INLINE int FEConvolveMatrix::getPixelValue(PaintingData& paintingData, int x, int y)
261{
262 if (x >= 0 && x < paintingData.width && y >= 0 && y < paintingData.height)
263 return (y * paintingData.width + x) << 2;
264
265 switch (m_edgeMode) {
266 default: // EDGEMODE_NONE
267 return -1;
268 case EDGEMODE_DUPLICATE:
269 if (x < 0)
270 x = 0;
271 else if (x >= paintingData.width)
272 x = paintingData.width - 1;
273 if (y < 0)
274 y = 0;
275 else if (y >= paintingData.height)
276 y = paintingData.height - 1;
277 return (y * paintingData.width + x) << 2;
278 case EDGEMODE_WRAP:
279 while (x < 0)
280 x += paintingData.width;
281 x %= paintingData.width;
282 while (y < 0)
283 y += paintingData.height;
284 y %= paintingData.height;
285 return (y * paintingData.width + x) << 2;
286 }
287}

289// For other regions than C
290template<bool preserveAlphaValues>
291void FEConvolveMatrix::fastSetOuterPixels(PaintingData& paintingData, int x1, int y1, int x2, int y2)
292{
293 int pixel = (y1 * paintingData.width + x1) * 4;
294 int height = y2 - y1;
295 int width = x2 - x1;
296 int beginKernelPixelX = x1 - m_targetOffset.x();
297 int startKernelPixelX = beginKernelPixelX;
298 int startKernelPixelY = y1 - m_targetOffset.y();
299 int xIncrease = (paintingData.width - width) * 4;
300 // Contains the sum of rgb(a) components
301 float totals[3 + (preserveAlphaValues ? 0 : 1)];
302
303 // m_divisor cannot be 0, SVGFEConvolveMatrixElement ensures this
304 ASSERT(m_divisor);
305
306 for (int y = height; y > 0; --y) {
307 for (int x = width; x > 0; --x) {
308 int kernelValue = paintingData.kernelMatrix.size() - 1;
309 int kernelPixelX = startKernelPixelX;
310 int kernelPixelY = startKernelPixelY;
311 int width = m_kernelSize.width();
312
313 totals[0] = 0;
314 totals[1] = 0;
315 totals[2] = 0;
316 if (!preserveAlphaValues)
317 totals[3] = 0;
318
319 while (kernelValue >= 0) {
320 int pixelIndex = getPixelValue(paintingData, kernelPixelX, kernelPixelY);
321 if (pixelIndex >= 0) {
322 totals[0] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(pixelIndex));
323 totals[1] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(pixelIndex + 1));
324 totals[2] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(pixelIndex + 2));
325 }
326 if (!preserveAlphaValues && pixelIndex >= 0)
327 totals[3] += paintingData.kernelMatrix[kernelValue] * static_cast<float>(paintingData.srcPixelArray.item(pixelIndex + 3));
328 ++kernelPixelX;
329 --kernelValue;
330 if (!--width) {
331 kernelPixelX = startKernelPixelX;
332 ++kernelPixelY;
333 width = m_kernelSize.width();
334 }
335 }
336
337 setDestinationPixels<preserveAlphaValues>(paintingData.srcPixelArray, paintingData.dstPixelArray, pixel, totals, m_divisor, paintingData.bias);
338 ++startKernelPixelX;
339 }
340 pixel += xIncrease;
341 startKernelPixelX = beginKernelPixelX;
342 ++startKernelPixelY;
343 }
344}

350ALWAYS_INLINE void FEConvolveMatrix::setInteriorPixels(PaintingData& paintingData, int clipRight, int clipBottom, int yStart, int yEnd)
351{
352 // Must be implemented here, since it refers another ALWAYS_INLINE
353 // function, which defined in this C++ source file as well
354 if (m_preserveAlpha)
355 fastSetInteriorPixels<true>(paintingData, clipRight, clipBottom, yStart, yEnd);
356 else
357 fastSetInteriorPixels<false>(paintingData, clipRight, clipBottom, yStart, yEnd);
358}
359
360ALWAYS_INLINE void FEConvolveMatrix::setOuterPixels(PaintingData& paintingData, int x1, int y1, int x2, int y2)
361{
362 // Although this function can be moved to the header, it is implemented here
363 // because setInteriorPixels is also implemented here
364 if (m_preserveAlpha)
365 fastSetOuterPixels<true>(paintingData, x1, y1, x2, y2);
366 else
367 fastSetOuterPixels<false>(paintingData, x1, y1, x2, y2);
368}

370void FEConvolveMatrix::platformApplySoftware()
371{
372 FilterEffect* in = inputEffect(0);
373
374 auto& destinationPixelBuffer = m_preserveAlpha ? createUnmultipliedImageResult() : createPremultipliedImageResult();
375 if (!destinationPixelBuffer)
376 return;
377
378 auto& destinationPixelArray = destinationPixelBuffer->data();
379
380 IntRect effectDrawingRect = requestedRegionOfInputPixelBuffer(in->absolutePaintRect());
381
382 RefPtr<Uint8ClampedArray> sourcePixelArray;
383 if (m_preserveAlpha)
384 sourcePixelArray = in->unmultipliedResult(effectDrawingRect, operatingColorSpace());
385 else
386 sourcePixelArray = in->premultipliedResult(effectDrawingRect, operatingColorSpace());
387 if (!sourcePixelArray)
388 return;
389
390 IntSize paintSize = absolutePaintRect().size();
391 paintSize.scale(filter().filterScale());
392
393 PaintingData paintingData = {
394 *sourcePixelArray,
395 destinationPixelArray,
396 paintSize.width(),
397 paintSize.height(),
398 m_bias * 255,
399 normalizedFloats(m_kernelMatrix)
400 };
401
402 // Drawing fully covered pixels
403 int clipRight = paintSize.width() - m_kernelSize.width();
404 int clipBottom = paintSize.height() - m_kernelSize.height();
405
406 if (clipRight < 0 || clipBottom < 0) {
407 // Rare situation, not optimizied for speed
408 setOuterPixels(paintingData, 0, 0, paintSize.width(), paintSize.height());
409 return;
410 }
411
412 if (int iterations = (absolutePaintRect().width() * absolutePaintRect().height()) / s_minimalRectDimension) {
413 int stride = clipBottom / iterations;
414 int chunkCount = (clipBottom + stride - 1) / stride;
415
416 WorkQueue::concurrentApply(chunkCount, [&](size_t index) {
417 int yStart = (stride * index);
418 int yEnd = std::min<int>(stride * (index + 1), clipBottom);
419
420 setInteriorPixels(paintingData, clipRight, clipBottom, yStart, yEnd);
421 });
422 } else
423 setInteriorPixels(paintingData, clipRight, clipBottom, 0, clipBottom);
424
425 clipRight += m_targetOffset.x() + 1;
426 clipBottom += m_targetOffset.y() + 1;
427 if (m_targetOffset.y() > 0)
428 setOuterPixels(paintingData, 0, 0, paintSize.width(), m_targetOffset.y());
429 if (clipBottom < paintSize.height())
430 setOuterPixels(paintingData, 0, clipBottom, paintSize.width(), paintSize.height());
431 if (m_targetOffset.x() > 0)
432 setOuterPixels(paintingData, 0, m_targetOffset.y(), m_targetOffset.x(), clipBottom);
433 if (clipRight < paintSize.width())
434 setOuterPixels(paintingData, clipRight, m_targetOffset.y(), paintSize.width(), clipBottom);
435}
