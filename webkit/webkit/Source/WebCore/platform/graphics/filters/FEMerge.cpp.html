41void FEMerge::platformApplySoftware()
42{
43 unsigned size = numberOfEffectInputs();
44 ASSERT(size > 0);
45
46 ImageBuffer* resultImage = createImageBufferResult();
47 if (!resultImage)
48 return;
49
50
51 GraphicsContext& filterContext = resultImage->context();
52 for (unsigned i = 0; i < size; ++i) {
53 FilterEffect* in = inputEffect(i);
54 if (ImageBuffer* inBuffer = in->imageBufferResult())
55 filterContext.drawImageBuffer(*inBuffer, drawingRegionOfInputImage(in->absolutePaintRect()));
56 }
57}
58
59TextStream& FEMerge::externalRepresentation(TextStream& ts, RepresentationType representation) const
60{
61 ts << indent << "[feMerge";
62 FilterEffect::externalRepresentation(ts, representation);
63 unsigned size = numberOfEffectInputs();
64 ASSERT(size > 0);
65 ts << " mergeNodes=\"" << size << "\"]\n";
66
67 TextStream::IndentScope indentScope(ts);
68 for (unsigned i = 0; i < size; ++i)
69 inputEffect(i)->externalRepresentation(ts, representation);
70 return ts;
71}
