29enum MorphologyOperatorType {

33};

41
42 float radiusX() const { return m_radiusX; }

44
45 float radiusY() const { return m_radiusY; }

48private:
49 FEMorphology(Filter&, MorphologyOperatorType, float radiusX, float radiusY);

53 void platformApplySoftware() override;
