36bool FilterOperations::operator==(const FilterOperations& other) const
37{
38 size_t size = m_operations.size();
39 if (size != other.m_operations.size())
40 return false;
41 for (size_t i = 0; i < size; i++) {
42 if (*m_operations[i] != *other.m_operations[i])
43 return false;
44 }
45 return true;
46}

48bool FilterOperations::operationsMatch(const FilterOperations& other) const
49{
50 size_t size = operations().size();
51 if (size != other.operations().size())
52 return false;
53 for (size_t i = 0; i < size; ++i) {
54 if (!operations()[i]->isSameType(*other.operations()[i]))
55 return false;
56 }
57 return true;
58}

60bool FilterOperations::hasReferenceFilter() const
61{
62 for (auto& operation : m_operations) {
63 if (operation->type() == FilterOperation::REFERENCE)
64 return true;
65 }
66 return false;

69IntOutsets FilterOperations::outsets() const
70{
71 IntOutsets totalOutsets;
72 for (auto& operation : m_operations) {
73 switch (operation->type()) {
74 case FilterOperation::BLUR: {
75 auto& blurOperation = downcast<BlurFilterOperation>(*operation);
76 float stdDeviation = floatValueForLength(blurOperation.stdDeviation(), 0);
77 IntSize outsetSize = FEGaussianBlur::calculateOutsetSize({ stdDeviation, stdDeviation });
78 IntOutsets outsets(outsetSize.height(), outsetSize.width(), outsetSize.height(), outsetSize.width());
79 totalOutsets += outsets;
80 break;
81 }
82 case FilterOperation::DROP_SHADOW: {
83 auto& dropShadowOperation = downcast<DropShadowFilterOperation>(*operation);
84 float stdDeviation = dropShadowOperation.stdDeviation();
85 IntSize outsetSize = FEGaussianBlur::calculateOutsetSize({ stdDeviation, stdDeviation });
86 IntOutsets outsets {
87 std::max(0, outsetSize.height() - dropShadowOperation.y()),
88 std::max(0, outsetSize.width() + dropShadowOperation.x()),
89 std::max(0, outsetSize.height() + dropShadowOperation.y()),
90 std::max(0, outsetSize.width() - dropShadowOperation.x())
91 };
92 totalOutsets += outsets;
93 break;
94 }
95 case FilterOperation::REFERENCE:
96 ASSERT_NOT_REACHED();
97 break;
98 default:
99 break;
100 }
101 }
102 return totalOutsets;
103}

105bool FilterOperations::transformColor(Color& color) const
106{
107 if (isEmpty() || !color.isValid())
108 return false;
109 // Color filter does not apply to semantic CSS colors (like "Windowframe").
110 if (color.isSemantic())
111 return false;
112
113 auto sRGBAColor = color.toSRGBALossy<float>();
114
115 for (auto& operation : m_operations) {
116 if (!operation->transformColor(sRGBAColor))
117 return false;
118 }
119
120 color = convertColor<SRGBA<uint8_t>>(sRGBAColor);
121 return true;
122}

170TextStream& operator<<(TextStream& ts, const FilterOperations& filters)
171{
172 for (size_t i = 0; i < filters.size(); ++i) {
173 auto filter = filters.at(i);
174 if (filter)
175 ts << *filter;
176 else
177 ts << "(null)";
178 if (i < filters.size() - 1)
179 ts << " ";
180 }
181 return ts;
182}
