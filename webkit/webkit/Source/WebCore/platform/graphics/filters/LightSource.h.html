24#pragma once
25
26#include "FloatPoint3D.h"
27#include <wtf/RefCounted.h>
28
29namespace WTF {
30class TextStream;
31}
32
33namespace WebCore {
34
35enum LightType {
36 LS_DISTANT,
37 LS_POINT,
38 LS_SPOT
39};
40
41class FilterEffect;
42
43class LightSource : public RefCounted<LightSource> {
44public:
45 struct ComputedLightingData {
46 FloatPoint3D lightVector;
47 FloatPoint3D colorVector;
48 float lightVectorLength;
49 };
50
51 struct PaintingData {
52 ComputedLightingData initialLightingData;
53 FloatPoint3D directionVector;
54 float coneCutOffLimit;
55 float coneFullLight;
56 int specularExponent;
57 };
58
59 LightSource(LightType type)
60 : m_type(type)
61 { }
62
63 virtual ~LightSource() = default;
64
65 LightType type() const { return m_type; }
66 virtual WTF::TextStream& externalRepresentation(WTF::TextStream&) const = 0;
67
68 virtual void initPaintingData(const FilterEffect&, PaintingData&) = 0;
69 // z is a float number, since it is the alpha value scaled by a user
70 // specified "surfaceScale" constant, which type is <number> in the SVG standard.
71 // x and y are in the coordinates of the FilterEffect's buffer.
72 virtual ComputedLightingData computePixelLightingData(const PaintingData&, int x, int y, float z) const = 0;
73
74 virtual bool setAzimuth(float) { return false; }
75 virtual bool setElevation(float) { return false; }
76
77 // These are in user space coordinates.
78 virtual bool setX(float) { return false; }
79 virtual bool setY(float) { return false; }
80 virtual bool setZ(float) { return false; }
81 virtual bool setPointsAtX(float) { return false; }
82 virtual bool setPointsAtY(float) { return false; }
83 virtual bool setPointsAtZ(float) { return false; }
84
85 virtual bool setSpecularExponent(float) { return false; }
86 virtual bool setLimitingConeAngle(float) { return false; }
87
88private:
89 LightType m_type;
90};
91
92} // namespace WebCore
