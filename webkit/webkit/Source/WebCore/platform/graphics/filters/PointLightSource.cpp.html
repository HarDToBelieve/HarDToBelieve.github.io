39void PointLightSource::initPaintingData(const FilterEffect& filterEffect, PaintingData&)
40{
41 m_bufferPosition.setXY(filterEffect.mapPointFromUserSpaceToBuffer(m_userSpacePosition.xy()));
42 // To scale Z, map a point offset from m_userSpacePosition in the x direction by z.
43 FloatPoint mappedZ = filterEffect.mapPointFromUserSpaceToBuffer({ m_userSpacePosition.x() + m_userSpacePosition.z(), m_userSpacePosition.y() });
44 m_bufferPosition.setZ(mappedZ.x() - m_bufferPosition.x());
45}
46
47LightSource::ComputedLightingData PointLightSource::computePixelLightingData(const PaintingData& paintingData, int x, int y, float z) const
48{
49 FloatPoint3D lightVector = {
50 m_bufferPosition.x() - x,
51 m_bufferPosition.y() - y,
52 m_bufferPosition.z() - z
53 };
54
55 return { lightVector, paintingData.initialLightingData.colorVector, lightVector.length() };
56}
57
58bool PointLightSource::setX(float x)
59{
60 if (m_userSpacePosition.x() == x)
61 return false;
62 m_userSpacePosition.setX(x);
63 return true;
64}
65
66bool PointLightSource::setY(float y)
67{
68 if (m_userSpacePosition.y() == y)
69 return false;
70 m_userSpacePosition.setY(y);
71 return true;
72}
73
74bool PointLightSource::setZ(float z)
75{
76 if (m_userSpacePosition.z() == z)
77 return false;
78 m_userSpacePosition.setZ(z);
79 return true;
80}

82TextStream& PointLightSource::externalRepresentation(TextStream& ts) const
83{
84 ts << "[type=POINT-LIGHT] ";
85 ts << "[position=\"" << position() << "\"]";
86 return ts;
87}
