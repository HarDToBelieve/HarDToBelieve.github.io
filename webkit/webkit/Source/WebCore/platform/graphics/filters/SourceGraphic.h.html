21#ifndef SourceGraphic_h
22#define SourceGraphic_h
23
24#include "FilterEffect.h"
25#include "Filter.h"
26
27namespace WebCore {
28
29class SourceGraphic : public FilterEffect {
30public:
31 static Ref<SourceGraphic> create(Filter&);
32
33 static const AtomString& effectName();
34
35private:
36 SourceGraphic(Filter& filter)
37 : FilterEffect(filter, Type::SourceGraphic)
38 {
39 setOperatingColorSpace(DestinationColorSpace::SRGB());
40 }
41
42 const char* filterName() const final { return "SourceGraphic"; }
43
44 void determineAbsolutePaintRect() override;
45 void platformApplySoftware() override;
46 WTF::TextStream& externalRepresentation(WTF::TextStream&, RepresentationType) const override;
47
48 FilterEffectType filterEffectType() const override { return FilterEffectTypeSourceInput; }
49};
50
51} //namespace WebCore
52
53SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::SourceGraphic)
54 static bool isType(const WebCore::FilterEffect& effect) { return effect.filterEffectClassType() == WebCore::FilterEffect::Type::SourceGraphic; }
55SPECIALIZE_TYPE_TRAITS_END()
