40// spot-light edge darkening depends on an absolute treshold
41// according to the SVG 1.1 SE light regression tests
42static const float antiAliasTreshold = 0.016f;
43
44void SpotLightSource::initPaintingData(const FilterEffect& filterEffect, PaintingData& paintingData)
45{
46 m_bufferPosition.setXY(filterEffect.mapPointFromUserSpaceToBuffer(m_userSpacePosition.xy()));
47 // To scale Z, map a point offset from m_userSpacePosition in the x direction by z.
48 FloatPoint mappedZ = filterEffect.mapPointFromUserSpaceToBuffer({ m_userSpacePosition.x() + m_userSpacePosition.z(), m_userSpacePosition.y() });
49 m_bufferPosition.setZ(mappedZ.x() - m_bufferPosition.x());
50
51 paintingData.directionVector = m_userSpacePointsAt - m_userSpacePosition;
52 paintingData.directionVector.normalize();
53
54 if (!m_limitingConeAngle) {
55 paintingData.coneCutOffLimit = 0.0f;
56 paintingData.coneFullLight = -antiAliasTreshold;
57 } else {
58 float limitingConeAngle = m_limitingConeAngle;
59 if (limitingConeAngle < 0.0f)
60 limitingConeAngle = -limitingConeAngle;
61 if (limitingConeAngle > 90.0f)
62 limitingConeAngle = 90.0f;
63 paintingData.coneCutOffLimit = cosf(deg2rad(180.0f - limitingConeAngle));
64 paintingData.coneFullLight = paintingData.coneCutOffLimit - antiAliasTreshold;
65 }
66
67 // Optimization for common specularExponent values
68 if (!m_specularExponent)
69 paintingData.specularExponent = 0;
70 else if (m_specularExponent == 1.0f)
71 paintingData.specularExponent = 1;
72 else // It is neither 0.0f nor 1.0f
73 paintingData.specularExponent = 2;
74}
75
76LightSource::ComputedLightingData SpotLightSource::computePixelLightingData(const PaintingData& paintingData, int x, int y, float z) const
77{
78 FloatPoint3D lightVector = {
79 m_bufferPosition.x() - x,
80 m_bufferPosition.y() - y,
81 m_bufferPosition.z() - z
82 };
83 float lightVectorLength = lightVector.length();
84
85 float cosineOfAngle = (lightVector * paintingData.directionVector) / lightVectorLength;
86 if (cosineOfAngle > paintingData.coneCutOffLimit) {
87 // No light is produced, scanlines are not updated
88 return { lightVector, { }, lightVectorLength };
89 }
90
91 // Set the color of the pixel
92 float lightStrength;
93 switch (paintingData.specularExponent) {
94 case 0:
95 lightStrength = 1.0f; // -cosineOfAngle ^ 0 == 1
96 break;
97 case 1:
98 lightStrength = -cosineOfAngle; // -cosineOfAngle ^ 1 == -cosineOfAngle
99 break;
100 default:
101 lightStrength = powf(-cosineOfAngle, m_specularExponent);
102 break;
103 }
104
105 if (cosineOfAngle > paintingData.coneFullLight)
106 lightStrength *= (paintingData.coneCutOffLimit - cosineOfAngle) / (paintingData.coneCutOffLimit - paintingData.coneFullLight);
107
108 if (lightStrength > 1.0f)
109 lightStrength = 1.0f;
110
111 return {
112 lightVector,
113 paintingData.initialLightingData.colorVector * lightStrength,
114 lightVectorLength
115 };
116}

182TextStream& SpotLightSource::externalRepresentation(TextStream& ts) const
183{
184 ts << "[type=SPOT-LIGHT] ";
185 ts << "[position=\"" << position() << "\"]";
186 ts << "[direction=\"" << direction() << "\"]";
187 ts << "[specularExponent=\"" << specularExponent() << "\"]";
188 ts << "[limitingConeAngle=\"" << limitingConeAngle() << "\"]";
189 return ts;
190}
