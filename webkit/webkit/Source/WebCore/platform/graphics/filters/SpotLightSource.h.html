23#pragma once
24
25#include "LightSource.h"
26#include <wtf/Ref.h>
27
28namespace WebCore {
29
30class SpotLightSource : public LightSource {
31public:
32 static Ref<SpotLightSource> create(const FloatPoint3D& position,
33 const FloatPoint3D& direction, float specularExponent, float limitingConeAngle)
34 {
35 return adoptRef(*new SpotLightSource(position, direction, specularExponent, limitingConeAngle));
36 }
37
38 const FloatPoint3D& position() const { return m_userSpacePosition; }
39 const FloatPoint3D& direction() const { return m_userSpacePointsAt; }
40 float specularExponent() const { return m_specularExponent; }
41 float limitingConeAngle() const { return m_limitingConeAngle; }
42
43 bool setX(float) override;
44 bool setY(float) override;
45 bool setZ(float) override;
46 bool setPointsAtX(float) override;
47 bool setPointsAtY(float) override;
48 bool setPointsAtZ(float) override;
49
50 bool setSpecularExponent(float) override;
51 bool setLimitingConeAngle(float) override;
52
53 void initPaintingData(const FilterEffect&, PaintingData&) override;
54 ComputedLightingData computePixelLightingData(const PaintingData&, int x, int y, float z) const final;
55
56 WTF::TextStream& externalRepresentation(WTF::TextStream&) const override;
57
58private:
59 SpotLightSource(const FloatPoint3D& position, const FloatPoint3D& direction, float specularExponent, float limitingConeAngle)
60 : LightSource(LS_SPOT)
61 , m_userSpacePosition(position)
62 , m_userSpacePointsAt(direction)
63 , m_specularExponent(specularExponent)
64 , m_limitingConeAngle(limitingConeAngle)
65 {
66 }
67
68 FloatPoint3D m_userSpacePosition;
69 FloatPoint3D m_userSpacePointsAt;
70
71 FloatPoint3D m_bufferPosition;
72
73 float m_specularExponent;
74 float m_limitingConeAngle;
75};
76
77} // namespace WebCore
78
79SPECIALIZE_TYPE_TRAITS_LIGHTSOURCE(SpotLightSource, LS_SPOT)
