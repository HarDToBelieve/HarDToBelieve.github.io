37namespace WebCore {
38
39class GPUDevice;
40
41class GPUBindGroupLayout : public RefCounted<GPUBindGroupLayout> {
42public:
43 static RefPtr<GPUBindGroupLayout> tryCreate(const GPUDevice&, const GPUBindGroupLayoutDescriptor&);
44
45 struct UniformBuffer {
46 unsigned internalLengthName;
47 };
48
49 struct DynamicUniformBuffer {
50 unsigned internalLengthName;
51 };
52
53 struct Sampler {
54 };
55
56 struct SampledTexture {
57 };
58
59 struct StorageBuffer {
60 unsigned internalLengthName;
61 };
62
63 struct DynamicStorageBuffer {
64 unsigned internalLengthName;
65 };
66
67 using InternalBindingDetails = Variant<UniformBuffer, DynamicUniformBuffer, Sampler, SampledTexture, StorageBuffer, DynamicStorageBuffer>;
68
69 struct Binding {
70 GPUBindGroupLayoutBinding externalBinding;
71 unsigned internalName;
72 InternalBindingDetails internalBindingDetails;
73 };
74
75 using BindingsMapType = HashMap<uint64_t, Binding, WTF::IntHash<uint64_t>, WTF::UnsignedWithZeroKeyHashTraits<uint64_t>>;
76 const BindingsMapType& bindingsMap() const { return m_bindingsMap; }
77#if USE(METAL)
78 MTLArgumentEncoder *vertexEncoder() const { return m_vertexEncoder.get(); }
79 MTLArgumentEncoder *fragmentEncoder() const { return m_fragmentEncoder.get(); }
80 MTLArgumentEncoder *computeEncoder() const { return m_computeEncoder.get(); }
81#endif // USE(METAL)
82
83private:
84#if USE(METAL)
85 GPUBindGroupLayout(BindingsMapType&&, RetainPtr<MTLArgumentEncoder>&& vertex, RetainPtr<MTLArgumentEncoder>&& fragment, RetainPtr<MTLArgumentEncoder>&& compute);
86
87 RetainPtr<MTLArgumentEncoder> m_vertexEncoder;
88 RetainPtr<MTLArgumentEncoder> m_fragmentEncoder;
89 RetainPtr<MTLArgumentEncoder> m_computeEncoder;
90#endif // USE(METAL)
91 const BindingsMapType m_bindingsMap;
92};

94} // namespace WebCore
