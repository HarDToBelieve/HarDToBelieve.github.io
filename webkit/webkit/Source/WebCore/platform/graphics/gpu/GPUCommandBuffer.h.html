38namespace WebCore {
39
40class GPUDevice;
41
42struct GPUExtent3D;
43
44struct GPUBufferCopyViewBase {
45 uint64_t offset;
46 unsigned rowPitch;
47 unsigned imageHeight;
48};
49
50struct GPUBufferCopyView final : GPUBufferCopyViewBase {
51 GPUBufferCopyView(Ref<GPUBuffer>&& bufferCopy, const GPUBufferCopyViewBase& base)
52 : GPUBufferCopyViewBase(base)
53 , buffer(WTFMove(bufferCopy))
54 {
55 }
56
57 Ref<GPUBuffer> buffer;
58};
59
60struct GPUTextureCopyViewBase {
61 unsigned mipLevel;
62 unsigned arrayLayer;
63 GPUOrigin3D origin;
64};
65
66struct GPUTextureCopyView final : GPUTextureCopyViewBase {
67 GPUTextureCopyView(Ref<GPUTexture>&& textureCopy, const GPUTextureCopyViewBase& base)
68 : GPUTextureCopyViewBase(base)
69 , texture(WTFMove(textureCopy))
70 {
71 }
72
73 Ref<GPUTexture> texture;
74};
75
76class GPUCommandBuffer : public RefCounted<GPUCommandBuffer> {
77public:
78 static RefPtr<GPUCommandBuffer> tryCreate(const GPUDevice&);
79
80 PlatformCommandBuffer* platformCommandBuffer() const { return m_platformCommandBuffer.get(); }
81 const HashSet<Ref<GPUBuffer>>& usedBuffers() const { return m_usedBuffers; }
82 const HashSet<Ref<GPUTexture>>& usedTextures() const { return m_usedTextures; }
83 bool isEncodingPass() const { return m_isEncodingPass; }
84
85 void setIsEncodingPass(bool isEncoding) { m_isEncodingPass = isEncoding; }
86#if USE(METAL)
87 ~GPUCommandBuffer();
88 void endBlitEncoding();
89#endif
90
91 void copyBufferToBuffer(Ref<GPUBuffer>&&, uint64_t srcOffset, Ref<GPUBuffer>&&, uint64_t dstOffset, uint64_t size);
92 void copyBufferToTexture(GPUBufferCopyView&&, GPUTextureCopyView&&, const GPUExtent3D&);
93 void copyTextureToBuffer(GPUTextureCopyView&&, GPUBufferCopyView&&, const GPUExtent3D&);
94 void copyTextureToTexture(GPUTextureCopyView&&, GPUTextureCopyView&&, const GPUExtent3D&);
95
96 void useBuffer(Ref<GPUBuffer>&& buffer) { m_usedBuffers.addVoid(WTFMove(buffer)); }
97 void useTexture(Ref<GPUTexture>&& texture) { m_usedTextures.addVoid(WTFMove(texture)); }
98
99private:
100 GPUCommandBuffer(PlatformCommandBufferSmartPtr&&);
101
102 PlatformCommandBufferSmartPtr m_platformCommandBuffer;
103 HashSet<Ref<GPUBuffer>> m_usedBuffers;
104 HashSet<Ref<GPUTexture>> m_usedTextures;
105 bool m_isEncodingPass { false };
106#if USE(METAL)
107 MTLBlitCommandEncoder *blitEncoder() const;
108 mutable RetainPtr<MTLBlitCommandEncoder> m_blitEncoder;
109#endif
110};
