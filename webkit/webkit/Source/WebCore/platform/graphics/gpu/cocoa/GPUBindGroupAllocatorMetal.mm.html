50std::optional<GPUBindGroupAllocator::ArgumentBufferOffsets> GPUBindGroupAllocator::allocateAndSetEncoders(MTLArgumentEncoder *vertex, MTLArgumentEncoder *fragment, MTLArgumentEncoder *compute)
51{
52 id<MTLDevice> device = nil;
53 auto checkedOffset = Checked<NSUInteger>(m_lastOffset);
54
55 if (vertex) {
56 device = vertex.device;
57 checkedOffset += vertex.encodedLength;
58 }
59 if (fragment) {
60 device = fragment.device;
61 checkedOffset += fragment.encodedLength;
62 }
63 if (compute) {
64 device = compute.device;
65 checkedOffset += compute.encodedLength;
66 }
67
68 // No arguments; nothing to be done.
69 if (!device)
70 return { };
71
72 if (checkedOffset.hasOverflowed()) {
73 m_errorScopes->generateError("", GPUErrorFilter::OutOfMemory);
74 return { };
75 }
76
77 auto newOffset = checkedOffset.value();
78
79 if (m_argumentBuffer && newOffset > m_argumentBuffer.get().length) {
80 if (!reallocate(newOffset))
81 return { };
82 } else if (!m_argumentBuffer) {
83 // Based off mimimum allocation for a shared-memory MTLBuffer on macOS.
84 NSUInteger minimumSize = 4096;
85 if (minimumSize < newOffset)
86 minimumSize = newOffset;
87
88 BEGIN_BLOCK_OBJC_EXCEPTIONS
89 m_argumentBuffer = adoptNS([device newBufferWithLength:minimumSize options:0]);
90 END_BLOCK_OBJC_EXCEPTIONS
91
92 if (!m_argumentBuffer) {
93 m_errorScopes->generateError("", GPUErrorFilter::OutOfMemory);
94 return { };
95 }
96 }
97
98 ArgumentBufferOffsets offsets;
99
100 // Math in the following section is guarded against overflow by newOffset calculation.
101 BEGIN_BLOCK_OBJC_EXCEPTIONS
102
103 if (vertex) {
104 offsets.vertex = m_lastOffset;
105 [vertex setArgumentBuffer:m_argumentBuffer.get() offset:*offsets.vertex];
106 }
107 if (fragment) {
108 offsets.fragment = (vertex ? vertex.encodedLength : 0) + m_lastOffset;
109 [fragment setArgumentBuffer:m_argumentBuffer.get() offset:*offsets.fragment];
110 }
111 if (compute) {
112 offsets.compute = (vertex ? vertex.encodedLength : 0) + (fragment ? fragment.encodedLength : 0) + m_lastOffset;
113 [compute setArgumentBuffer:m_argumentBuffer.get() offset:*offsets.compute];
114 }
115
116 END_BLOCK_OBJC_EXCEPTIONS
117
118 m_lastOffset = newOffset;
119 ++m_numBindGroups;
120
121 return offsets;
122}

126// FIXME: https://bugs.webkit.org/show_bug.cgi?id=200657, https://bugs.webkit.org/show_bug.cgi?id=200658 Optimize reallocation and reset behavior.
127bool GPUBindGroupAllocator::reallocate(NSUInteger newOffset)
128{
129 MTLBuffer *newBuffer = nil;
130
131 auto newLength = Checked<NSUInteger>(m_argumentBuffer.get().length);
132 while (newLength < newOffset) {
133 newLength += newLength / 4U;
134
135 if (newLength.hasOverflowed()) {
136 newLength = std::numeric_limits<NSUInteger>::max();
137 break;
138 }
139 }
140
141 BEGIN_BLOCK_OBJC_EXCEPTIONS
142
143 newBuffer = [m_argumentBuffer.get().device newBufferWithLength:newLength.value() options:0];
144 memcpy(newBuffer.contents, m_argumentBuffer.get().contents, m_argumentBuffer.get().length);
145
146 END_BLOCK_OBJC_EXCEPTIONS
147
148 if (!newBuffer) {
149 m_errorScopes->generateError("", GPUErrorFilter::OutOfMemory);
150 return false;
151 }
152
153 m_argumentBuffer = adoptNS(newBuffer);
154 return true;
155}
156
157void GPUBindGroupAllocator::tryReset()
158{
159 --m_numBindGroups;
160
161 ASSERT(m_numBindGroups > -1);
162
163 if (!m_numBindGroups) {
164 m_argumentBuffer = nullptr;
165 m_lastOffset = 0;
166 }
167}
