42RefPtr<GPUCommandBuffer> GPUCommandBuffer::tryCreate(const GPUDevice& device)
43{
44 if (!device.platformDevice()) {
45 LOG(WebGPU, "GPUCommandBuffer::create(): Invalid GPUDevice!");
46 return nullptr;
47 }
48
49 auto gpuCommandQueue = device.tryGetQueue();
50 if (!gpuCommandQueue)
51 return nullptr;
52
53 auto mtlQueue = gpuCommandQueue->platformQueue();
54
55 RetainPtr<MTLCommandBuffer> buffer;
56
57 BEGIN_BLOCK_OBJC_EXCEPTIONS
58
59 buffer = [mtlQueue commandBuffer];
60
61 END_BLOCK_OBJC_EXCEPTIONS
62
63 if (!buffer) {
64 LOG(WebGPU, "GPUCommandBuffer::create(): Unable to create MTLCommandBuffer!");
65 return nullptr;
66 }
67
68 return adoptRef(new GPUCommandBuffer(WTFMove(buffer)));
69}

86void GPUCommandBuffer::endBlitEncoding()
87{
88 if (!m_blitEncoder)
89 return;
90 BEGIN_BLOCK_OBJC_EXCEPTIONS
91 [m_blitEncoder endEncoding];
92 END_BLOCK_OBJC_EXCEPTIONS
93 m_blitEncoder = nullptr;
94}

96void GPUCommandBuffer::copyBufferToBuffer(Ref<GPUBuffer>&& src, uint64_t srcOffset, Ref<GPUBuffer>&& dst, uint64_t dstOffset, uint64_t size)
97{
98 if (isEncodingPass() || !src->isCopySource() || !dst->isCopyDestination()) {
99 LOG(WebGPU, "GPUCommandBuffer::copyBufferToBuffer(): Invalid operation!");
100 return;
101 }
102
103#if PLATFORM(MAC)
104 if (size % 4 || srcOffset % 4 || dstOffset % 4) {
105 LOG(WebGPU, "GPUCommandBuffer::copyBufferToBuffer(): Copy must be aligned to a multiple of 4 bytes!");
106 return;
107 }
108#endif
109
110 // This call ensures that size, offset, and size + offset can safely fit in NSUInteger regardless of platform.
111 auto srcLength = checkedSum<NSUInteger>(size, srcOffset);
112 auto dstLength = checkedSum<NSUInteger>(size, dstOffset);
113 if (srcLength.hasOverflowed() || dstLength.hasOverflowed()
114 || srcLength > src->byteLength() || dstLength > dst->byteLength()) {
115 LOG(WebGPU, "GPUCommandBuffer::copyBufferToBuffer(): Invalid offset or copy size!");
116 return;
117 }
118
119 BEGIN_BLOCK_OBJC_EXCEPTIONS
120
121 // These casts are safe due to earlier checkedSum() checks.
122 [blitEncoder()
123 copyFromBuffer:src->platformBuffer()
124 sourceOffset:static_cast<NSUInteger>(srcOffset)
125 toBuffer:dst->platformBuffer()
126 destinationOffset:static_cast<NSUInteger>(dstOffset)
127 size:static_cast<NSUInteger>(size)];
128
129 END_BLOCK_OBJC_EXCEPTIONS
130
131 useBuffer(WTFMove(src));
132 useBuffer(WTFMove(dst));
133}

135void GPUCommandBuffer::copyBufferToTexture(GPUBufferCopyView&& srcBuffer, GPUTextureCopyView&& dstTexture, const GPUExtent3D& size)
136{
137 if (isEncodingPass() || !srcBuffer.buffer->isCopySource() || !dstTexture.texture->isCopyDestination()) {
138 LOG(WebGPU, "GPUCommandBuffer::copyBufferToTexture(): Invalid operation!");
139 return;
140 }
141
142 // MTLBuffer size (NSUInteger) is 32 bits on some platforms.
143 NSUInteger sourceOffset = 0;
144 if (!WTF::convertSafely(srcBuffer.offset, sourceOffset)) {
145 LOG(WebGPU, "GPUCommandBuffer::copyBufferToTexture(): Source offset is too large!");
146 return;
147 }
148
149 // FIXME: Add Metal validation.
150
151 // GPUBufferCopyView::offset: The location must be aligned to the size of the destination texture's pixel format. The value must be a multiple of the destination texture's pixel size, in bytes.
152
153 // GPUBufferCopyView::rowPitch: The value must be a multiple of the destination texture's pixel size, in bytes. The value must be less than or equal to 32,767 multiplied by the destination texture’s pixel size.
154
155 // GPUBufferCopyView::imageHeight: The value must be a multiple of the destination texture's pixel size, in bytes.
156
157 // GPUExtent3D: When you copy to a 1D texture, height and depth must be 1. When you copy to a 2D texture, depth must be 1.
158
159 // GPUTextureCopyView::texture: The value must not be a framebufferOnly texture and must not have a PVRTC pixel format.
160
161 BEGIN_BLOCK_OBJC_EXCEPTIONS
162
163 [blitEncoder()
164 copyFromBuffer:srcBuffer.buffer->platformBuffer()
165 sourceOffset:sourceOffset
166 sourceBytesPerRow:srcBuffer.rowPitch
167 sourceBytesPerImage:srcBuffer.imageHeight
168 sourceSize:MTLSizeMake(size.width, size.height, size.depth)
169 toTexture:dstTexture.texture->platformTexture()
170 destinationSlice:dstTexture.arrayLayer
171 destinationLevel:dstTexture.mipLevel
172 destinationOrigin:MTLOriginMake(dstTexture.origin.x, dstTexture.origin.y, dstTexture.origin.z)];
173
174 END_BLOCK_OBJC_EXCEPTIONS
175
176 useBuffer(WTFMove(srcBuffer.buffer));
177 useTexture(WTFMove(dstTexture.texture));
178}
