38RefPtr<GPUComputePassEncoder> GPUComputePassEncoder::tryCreate(Ref<GPUCommandBuffer>&& buffer)
39{
40 if (buffer->isEncodingPass()) {
41 LOG(WebGPU, "GPUComputePassEncoder::tryCreate(): Existing pass encoder must be ended first!");
42 return nullptr;
43 }
44
45 buffer->endBlitEncoding();
46
47 RetainPtr<MTLComputeCommandEncoder> mtlEncoder;
48
49 BEGIN_BLOCK_OBJC_EXCEPTIONS
50 mtlEncoder = [buffer->platformCommandBuffer() computeCommandEncoder];
51 END_BLOCK_OBJC_EXCEPTIONS
52
53 if (!mtlEncoder) {
54 LOG(WebGPU, "GPUComputePassEncoder::tryCreate(): Unable to create MTLComputeCommandEncoder!");
55 return nullptr;
56 }
57
58 return adoptRef(new GPUComputePassEncoder(WTFMove(buffer), WTFMove(mtlEncoder)));
59}

62 : GPUProgrammablePassEncoder(WTFMove(buffer))

67void GPUComputePassEncoder::setPipeline(Ref<const GPUComputePipeline>&& pipeline)
68{
69 if (!m_platformComputePassEncoder) {
70 LOG(WebGPU, "GPUComputePassEncoder::setPipeline(): Invalid operation!");
71 return;
72 }
73
74 ASSERT(pipeline->platformComputePipeline());
75
76 BEGIN_BLOCK_OBJC_EXCEPTIONS
77 [m_platformComputePassEncoder setComputePipelineState:pipeline->platformComputePipeline()];
78 END_BLOCK_OBJC_EXCEPTIONS
79
80 m_pipeline = WTFMove(pipeline);
81}
82
83void GPUComputePassEncoder::dispatch(unsigned x, unsigned y, unsigned z)
84{
85 if (!m_platformComputePassEncoder) {
86 LOG(WebGPU, "GPUComputePassEncoder::dispatch(): Invalid operation!");
87 return;
88 }
89
90 if (!m_pipeline) {
91 LOG(WebGPU, "GPUComputePassEncoder::dispatch(): No valid GPUComputePipeline found!");
92 return;
93 }
94
95 ASSERT(m_pipeline->platformComputePipeline());
96
97 BEGIN_BLOCK_OBJC_EXCEPTIONS
98
99 auto threadsPerThreadgroup = MTLSizeMake(m_pipeline->computeDimensions().width, m_pipeline->computeDimensions().height, m_pipeline->computeDimensions().depth);
100
101 auto threadgroupsPerGrid = MTLSizeMake(x, y, z);
102
103 [m_platformComputePassEncoder dispatchThreadgroups:threadgroupsPerGrid threadsPerThreadgroup:threadsPerThreadgroup];
104
105 END_BLOCK_OBJC_EXCEPTIONS
106}
107

109{
110 return m_platformComputePassEncoder.get();
111}
