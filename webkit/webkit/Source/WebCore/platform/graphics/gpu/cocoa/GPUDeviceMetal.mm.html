39static bool isAcceptableDevice(id <MTLDevice> device)
40{
41 UNUSED_PARAM(device);
42 return true;
43}

45void GPUDevice::prewarm()
46{
47#if PLATFORM(MAC)
48 // Call MTLCopyAllDevices() on a background thread to avoid hanging the main thread.
49 dispatch_async(dispatch_get_global_queue(0, 0), ^{
50 auto devices = adoptNS(MTLCopyAllDevices());
51 });
52#endif
53}
54
55RefPtr<GPUDevice> GPUDevice::tryCreate(const std::optional<GPURequestAdapterOptions>& options)
56{
57 RetainPtr<MTLDevice> devicePtr;
58
59 BEGIN_BLOCK_OBJC_EXCEPTIONS
60
61#if PLATFORM(MAC)
62 if (!options || !options->powerPreference || options->powerPreference == GPUPowerPreference::LowPower) {
63 auto devices = adoptNS(MTLCopyAllDevices());
64
65 for (id <MTLDevice> device : devices.get()) {
66 if (!isAcceptableDevice(device))
67 continue;
68 if (device.lowPower) {
69 devicePtr = device;
70 break;
71 }
72 }
73 }
74#else
75 UNUSED_PARAM(options);
76#endif // PLATFORM(MAC)
77 if (!devicePtr)
78 devicePtr = adoptNS(MTLCreateSystemDefaultDevice());
79
80 if (!isAcceptableDevice(devicePtr.get()))
81 devicePtr.clear();
82
83 END_BLOCK_OBJC_EXCEPTIONS
84
85 if (!devicePtr) {
86 LOG(WebGPU, "GPUDevice::GPUDevice(): Unable to create GPUDevice!");
87 return nullptr;
88 }
89
90 LOG(WebGPU, "GPUDevice::GPUDevice(): MTLDevice is %p", devicePtr.get());
91 return adoptRef(new GPUDevice(WTFMove(devicePtr)));
92}
