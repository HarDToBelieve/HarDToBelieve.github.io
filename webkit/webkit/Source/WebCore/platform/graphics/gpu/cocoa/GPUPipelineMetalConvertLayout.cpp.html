36static OptionSet<WHLSL::ShaderStage> convertShaderStageFlags(GPUShaderStageFlags flags)
37{
38 OptionSet<WHLSL::ShaderStage> result;
39 if (flags & GPUShaderStage::Flags::Vertex)
40 result.add(WHLSL::ShaderStage::Vertex);
41 if (flags & GPUShaderStage::Flags::Fragment)
42 result.add(WHLSL::ShaderStage::Fragment);
43 if (flags & GPUShaderStage::Flags::Compute)
44 result.add(WHLSL::ShaderStage::Compute);
45 return result;
46}

48static std::optional<WHLSL::Binding::BindingDetails> convertBindingType(GPUBindGroupLayout::InternalBindingDetails internalBindingDetails)
49{
50 return WTF::visit(WTF::makeVisitor([&](GPUBindGroupLayout::UniformBuffer uniformBuffer) -> std::optional<WHLSL::Binding::BindingDetails> {
51 return { WHLSL::UniformBufferBinding { uniformBuffer.internalLengthName } };
52 }, [&](GPUBindGroupLayout::DynamicUniformBuffer) -> std::optional<WHLSL::Binding::BindingDetails> {
53 return std::nullopt;
54 }, [&](GPUBindGroupLayout::Sampler) -> std::optional<WHLSL::Binding::BindingDetails> {
55 return { WHLSL::SamplerBinding { } };
56 }, [&](GPUBindGroupLayout::SampledTexture) -> std::optional<WHLSL::Binding::BindingDetails> {
57 return { WHLSL::TextureBinding { } };
58 }, [&](GPUBindGroupLayout::StorageBuffer storageBuffer) -> std::optional<WHLSL::Binding::BindingDetails> {
59 return { WHLSL::StorageBufferBinding { storageBuffer.internalLengthName } };
60 }, [&](GPUBindGroupLayout::DynamicStorageBuffer) -> std::optional<WHLSL::Binding::BindingDetails> {
61 return std::nullopt;
62 }), internalBindingDetails);
63}

65std::optional<WHLSL::Layout> convertLayout(const GPUPipelineLayout& layout)
66{
67 WHLSL::Layout result;
68 if (layout.bindGroupLayouts().size() > std::numeric_limits<unsigned>::max())
69 return std::nullopt;
70 for (size_t i = 0; i < layout.bindGroupLayouts().size(); ++i) {
71 const auto& bindGroupLayout = layout.bindGroupLayouts()[i];
72 WHLSL::BindGroup bindGroup;
73 bindGroup.name = static_cast<unsigned>(i);
74 for (const auto& keyValuePair : bindGroupLayout->bindingsMap()) {
75 const auto& bindingDetails = keyValuePair.value;
76 WHLSL::Binding binding;
77 binding.visibility = convertShaderStageFlags(bindingDetails.externalBinding.visibility);
78 if (auto bindingType = convertBindingType(bindingDetails.internalBindingDetails))
79 binding.binding = *bindingType;
80 else
81 return std::nullopt;
82 if (bindingDetails.externalBinding.binding > std::numeric_limits<unsigned>::max())
83 return std::nullopt;
84 binding.externalName = bindingDetails.externalBinding.binding;
85 binding.internalName = bindingDetails.internalName;
86 bindGroup.bindings.append(WTFMove(binding));
87 }
88 result.append(WTFMove(bindGroup));
89 }
90 return result;
91}
