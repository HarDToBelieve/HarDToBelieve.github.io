41static NSString * const commandQueueDefaultLabel = @"com.apple.WebKit";
42static NSString * const commandQueueLabelPrefix = @"com.apple.WebKit.";
43
44RefPtr<GPUQueue> GPUQueue::tryCreate(const GPUDevice& device)
45{
46 if (!device.platformDevice()) {
47 LOG(WebGPU, "GPUQueue::tryCreate(): Invalid GPUDevice!");
48 return nullptr;
49 }
50
51 RetainPtr<MTLCommandQueue> queue;
52
53 BEGIN_BLOCK_OBJC_EXCEPTIONS
54
55 queue = adoptNS([device.platformDevice() newCommandQueue]);
56 [queue setLabel:commandQueueDefaultLabel];
57
58 END_BLOCK_OBJC_EXCEPTIONS
59
60 if (!queue) {
61 LOG(WebGPU, "GPUQueue::tryCreate(): Unable to create MTLCommandQueue!");
62 return nullptr;
63 }
64
65 return adoptRef(new GPUQueue(WTFMove(queue), device));
66}

74void GPUQueue::submit(Vector<Ref<GPUCommandBuffer>>&& commandBuffers)
75{
76 for (auto& commandBuffer : commandBuffers) {
77 // Validate resource integrity.
78 for (auto& buffer : commandBuffer->usedBuffers()) {
79 if (buffer->state() != GPUBuffer::State::Unmapped) {
80 LOG(WebGPU, "GPUQueue::submit(): Invalid GPUBuffer set on a GPUCommandBuffer!");
81 return;
82 }
83 }
84 for (auto& texture : commandBuffer->usedTextures()) {
85 if (!texture->platformTexture()) {
86 LOG(WebGPU, "GPUQueue::submit(): Invalid GPUTexture set on a GPUCommandBuffer!");
87 return;
88 }
89 }
90 commandBuffer->endBlitEncoding();
91 // Okay to commit; prevent any buffer mapping callbacks from executing until command buffer is complete.
92 for (auto& buffer : commandBuffer->usedBuffers())
93 buffer->commandBufferCommitted(commandBuffer->platformCommandBuffer());
94 BEGIN_BLOCK_OBJC_EXCEPTIONS
95 [commandBuffer->platformCommandBuffer() commit];
96 END_BLOCK_OBJC_EXCEPTIONS
97 }
98
99 if (m_presentTask.isPending() || !m_device || !m_device->swapChain())
100 return;
101
102 // If a GPUSwapChain exists, ensure that a present is scheduled after all command buffers.
103 m_presentTask.scheduleTask([this, protectedThis = makeRef(*this), swapChain = makeRef(*m_device->swapChain())] () {
104 auto currentDrawable = swapChain->takeDrawable();
105 // If the GPUSwapChain has no drawable, it was invalidated between command submission and this task.
106 if (!currentDrawable)
107 return;
108
109 auto presentCommandBuffer = [m_platformQueue commandBuffer];
110 [presentCommandBuffer presentDrawable:static_cast<id<MTLDrawable>>(currentDrawable.get())];
111 [presentCommandBuffer commit];
112 });
113}

115String GPUQueue::label() const
116{
117 if (!m_platformQueue)
118 return emptyString();
119
120 NSString *prefixedLabel = [m_platformQueue label];
121
122 if ([prefixedLabel isEqualToString:commandQueueDefaultLabel])
123 return emptyString();
124
125 ASSERT(prefixedLabel.length > commandQueueLabelPrefix.length);
126 return [prefixedLabel substringFromIndex:commandQueueLabelPrefix.length];
127}
128
129void GPUQueue::setLabel(const String& label) const
130{
131 if (label.isEmpty())
132 [m_platformQueue setLabel:commandQueueDefaultLabel];
133 else
134 [m_platformQueue setLabel:[commandQueueLabelPrefix stringByAppendingString:label]];
135}
