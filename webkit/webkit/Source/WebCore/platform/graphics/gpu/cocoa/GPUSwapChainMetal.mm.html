43static std::optional<MTLPixelFormat> tryGetSupportedPixelFormat(GPUTextureFormat format)
44{
45 auto mtlFormat = static_cast<MTLPixelFormat>(platformTextureFormatForGPUTextureFormat(format));
46
47 switch (mtlFormat) {
48 case MTLPixelFormatBGRA8Unorm:
49 case MTLPixelFormatBGRA8Unorm_sRGB:
50 case MTLPixelFormatRGBA16Float:
51 return mtlFormat;
52 default: {
53 LOG(WebGPU, "GPUSwapChain::tryCreate(): Unsupported MTLPixelFormat!");
54 return std::nullopt;
55 }
56 }
57}

59static void setLayerShape(WebGPULayer *layer, int width, int height)
60{
61 [layer setBounds:CGRectMake(0, 0, width, height)];
62 [layer setDrawableSize:CGSizeMake(width, height)];
63}

65static RetainPtr<WebGPULayer> tryCreateWebGPULayer(MTLDevice *device, MTLPixelFormat format, OptionSet<GPUTextureUsage::Flags> usage)
66{
67
68 RetainPtr<WebGPULayer> layer;
69
70 BEGIN_BLOCK_OBJC_EXCEPTIONS
71 layer = adoptNS([WebGPULayer new]);
72
73 [layer setName:@"Web GPU Layer"];
74 [layer setOpaque:0];
75
76 [layer setDevice:device];
77 [layer setFramebufferOnly:(usage == GPUTextureUsage::Flags::OutputAttachment)];
78 [layer setPixelFormat:format];
79 END_BLOCK_OBJC_EXCEPTIONS
80
81 if (!layer)
82 LOG(WebGPU, "GPUSwapChain::tryCreate(): Unable to create CAMetalLayer!");
83
84 return layer;
85}

87RefPtr<GPUSwapChain> GPUSwapChain::tryCreate(const GPUSwapChainDescriptor& descriptor, int width, int height)
88{
89 if (!descriptor.device->platformDevice()) {
90 LOG(WebGPU, "GPUSwapChain::tryCreate(): Invalid GPUDevice!");
91 return nullptr;
92 }
93
94 auto pixelFormat = tryGetSupportedPixelFormat(descriptor.format);
95 if (!pixelFormat)
96 return nullptr;
97
98 auto usageOptions = OptionSet<GPUTextureUsage::Flags>::fromRaw(descriptor.usage);
99 if (!usageOptions.contains(GPUTextureUsage::Flags::OutputAttachment)) {
100 LOG(WebGPU, "GPUSwapChain::tryCreate(): Swap chain usage must include OUTPUT_ATTACHMENT!");
101 return nullptr;
102 }
103
104 auto layer = tryCreateWebGPULayer(descriptor.device->platformDevice(), *pixelFormat, usageOptions);
105 if (!layer)
106 return nullptr;
107
108 setLayerShape(layer.get(), width, height);
109
110 auto swapChain = adoptRef(new GPUSwapChain(WTFMove(layer), usageOptions));
111 descriptor.device->setSwapChain(swapChain.copyRef());
112 return swapChain;
113}

115GPUSwapChain::GPUSwapChain(RetainPtr<WebGPULayer>&& platformLayer, OptionSet<GPUTextureUsage::Flags> usageOptions)
116 : m_platformSwapLayer(WTFMove(platformLayer))
117 , m_usage(usageOptions)
118{
119 [m_platformSwapLayer setSwapChain:this];
120}

122RefPtr<GPUTexture> GPUSwapChain::tryGetCurrentTexture()
123{
124 RetainPtr<MTLTexture> mtlTexture;
125
126 BEGIN_BLOCK_OBJC_EXCEPTIONS
127 if (!m_currentDrawable)
128 m_currentDrawable = retainPtr([m_platformSwapLayer nextDrawable]);
129 mtlTexture = [m_currentDrawable texture];
130
131 END_BLOCK_OBJC_EXCEPTIONS
132
133 if (!mtlTexture) {
134 LOG(WebGPU, "GPUSwapChain::getCurrentTexture(): Unable to get current MTLTexture!");
135 return nullptr;
136 }
137
138 return GPUTexture::create(WTFMove(mtlTexture), m_usage);
139}

141void GPUSwapChain::present()
142{
143 if (!m_currentDrawable)
144 return;
145
146 [m_currentDrawable present];
147 m_currentDrawable = nullptr;
148}

150void GPUSwapChain::reshape(int width, int height)
151{
152 setLayerShape(m_platformSwapLayer.get(), width, height);
153}
