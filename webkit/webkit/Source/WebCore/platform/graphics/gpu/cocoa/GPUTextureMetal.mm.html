40static MTLTextureType mtlTextureTypeForGPUTextureDescriptor(const GPUTextureDescriptor& descriptor)
41{
42 switch (descriptor.dimension) {
43 case GPUTextureDimension::_1d:
44 return (descriptor.size.height == 1) ? MTLTextureType1D : MTLTextureType1DArray;
45 case GPUTextureDimension::_2d: {
46 if (descriptor.size.depth == 1)
47 return (descriptor.sampleCount == 1) ? MTLTextureType2D : MTLTextureType2DMultisample;
48 return MTLTextureType2DArray; // MTLTextureType2DMultisampleArray is unavailable on iOS.
49 }
50 case GPUTextureDimension::_3d:
51 return MTLTextureType3D;
52 }
53}

55static std::optional<MTLTextureUsage> mtlTextureUsageForGPUTextureUsageFlags(OptionSet<GPUTextureUsage::Flags> flags, const char* const functionName)
56{
57#if LOG_DISABLED
58 UNUSED_PARAM(functionName);
59#endif
60
61 if (flags.containsAny({ GPUTextureUsage::Flags::CopySource, GPUTextureUsage::Flags::Sampled }) && (flags & GPUTextureUsage::Flags::Storage)) {
62 LOG(WebGPU, "%s: Texture cannot have both STORAGE and a read-only usage!", functionName);
63 return std::nullopt;
64 }
65
66 MTLTextureUsage result = 0;
67 if (flags.contains(GPUTextureUsage::Flags::OutputAttachment))
68 result |= MTLTextureUsageRenderTarget;
69 if (flags.containsAny({ GPUTextureUsage::Flags::Storage, GPUTextureUsage::Flags::Sampled }))
70 result |= MTLTextureUsageShaderRead;
71 if (flags.contains(GPUTextureUsage::Flags::Storage))
72 result |= MTLTextureUsageShaderWrite;
73
74 return result;
75}

78static MTLStorageMode storageModeForPixelFormatAndSampleCount(MTLPixelFormat format, unsigned samples)
79{
80 // Depth, Stencil, DepthStencil, and Multisample textures must be allocated with the MTLStorageModePrivate resource option.
81 if (format == MTLPixelFormatDepth32Float_Stencil8 || samples > 1)
82 return MTLStorageModePrivate;
83
84 return MTLStorageModeShared;
85}

88static RetainPtr<MTLTextureDescriptor> tryCreateMtlTextureDescriptor(const char* const functionName, const GPUTextureDescriptor& descriptor, OptionSet<GPUTextureUsage::Flags> usage)
89{
90 RetainPtr<MTLTextureDescriptor> mtlDescriptor;
91
92 BEGIN_BLOCK_OBJC_EXCEPTIONS
93
94 mtlDescriptor = adoptNS([MTLTextureDescriptor new]);
95
96 END_BLOCK_OBJC_EXCEPTIONS
97
98 if (!mtlDescriptor) {
99 LOG(WebGPU, "%s: Unable to create new MTLTextureDescriptor!", functionName);
100 return nullptr;
101 }
102
103 unsigned width = descriptor.size.width;
104 unsigned height = 1;
105 unsigned depth = 1;
106 unsigned arrayLength = 1;
107 switch (descriptor.dimension) {
108 case GPUTextureDimension::_1d:
109 arrayLength = descriptor.size.height;
110 break;
111 case GPUTextureDimension::_2d:
112 height = descriptor.size.height;
113 arrayLength = descriptor.size.depth;
114 break;
115 case GPUTextureDimension::_3d:
116 height = descriptor.size.height;
117 depth = descriptor.size.depth;
118 break;
119 }
120
121 auto pixelFormat = static_cast<MTLPixelFormat>(platformTextureFormatForGPUTextureFormat(descriptor.format));
122
123 auto mtlUsage = mtlTextureUsageForGPUTextureUsageFlags(usage, functionName);
124 if (!mtlUsage)
125 return nullptr;
126
127#if PLATFORM(MAC)
128 auto storageMode = MTLStorageModePrivate;
129#else
130 auto storageMode = storageModeForPixelFormatAndSampleCount(pixelFormat, descriptor.sampleCount);
131#endif
132
133 BEGIN_BLOCK_OBJC_EXCEPTIONS
134
135 [mtlDescriptor setWidth:width];
136 [mtlDescriptor setHeight:height];
137 [mtlDescriptor setDepth:depth];
138 [mtlDescriptor setArrayLength:arrayLength];
139 [mtlDescriptor setMipmapLevelCount:descriptor.mipLevelCount];
140 [mtlDescriptor setSampleCount:descriptor.sampleCount];
141 [mtlDescriptor setTextureType:mtlTextureTypeForGPUTextureDescriptor(descriptor)];
142 [mtlDescriptor setPixelFormat:pixelFormat];
143 [mtlDescriptor setUsage:*mtlUsage];
144
145 [mtlDescriptor setStorageMode:storageMode];
146
147 END_BLOCK_OBJC_EXCEPTIONS
148
149 return mtlDescriptor;
150}

152RefPtr<GPUTexture> GPUTexture::tryCreate(const GPUDevice& device, const GPUTextureDescriptor& descriptor, GPUErrorScopes& errorScopes)
153{
154 const char* const functionName = "GPUTexture::tryCreate()";
155
156 if (!device.platformDevice()) {
157 LOG(WebGPU, "%s: Invalid GPUDevice!", functionName);
158 return nullptr;
159 }
160
161 if (![device.platformDevice() supportsTextureSampleCount:descriptor.sampleCount]) {
162 errorScopes.generatePrefixedError(makeString("Device does not support ", descriptor.sampleCount, " sample count."));
163 return { };
164 }
165
166 auto usage = OptionSet<GPUTextureUsage::Flags>::fromRaw(descriptor.usage);
167 auto mtlDescriptor = tryCreateMtlTextureDescriptor(functionName, descriptor, usage);
168 if (!mtlDescriptor)
169 return nullptr;
170
171 RetainPtr<MTLTexture> mtlTexture;
172
173 BEGIN_BLOCK_OBJC_EXCEPTIONS
174
175 mtlTexture = adoptNS([device.platformDevice() newTextureWithDescriptor:mtlDescriptor.get()]);
176
177 END_BLOCK_OBJC_EXCEPTIONS
178
179 if (!mtlTexture) {
180 LOG(WebGPU, "%s: Unable to create MTLTexture!", functionName);
181 return nullptr;
182 }
183
184 return adoptRef(new GPUTexture(WTFMove(mtlTexture), usage));
185}

187Ref<GPUTexture> GPUTexture::create(RetainPtr<MTLTexture>&& texture, OptionSet<GPUTextureUsage::Flags> usage)
188{
189 return adoptRef(*new GPUTexture(WTFMove(texture), usage));

192GPUTexture::GPUTexture(RetainPtr<MTLTexture>&& texture, OptionSet<GPUTextureUsage::Flags> usage)
193 : m_platformTexture(WTFMove(texture))
194 , m_usage(usage)
195{
196 m_platformUsage = MTLResourceUsageRead;
197 if (isSampled())
198 m_platformUsage |= MTLResourceUsageSample;
199 else if (isStorage())
200 m_platformUsage |= MTLResourceUsageWrite;
201}

203RefPtr<GPUTexture> GPUTexture::tryCreateDefaultTextureView()
204{
205 RetainPtr<MTLTexture> texture;
206
207 BEGIN_BLOCK_OBJC_EXCEPTIONS
208
209 texture = adoptNS([m_platformTexture newTextureViewWithPixelFormat:m_platformTexture.get().pixelFormat]);
210
211 END_BLOCK_OBJC_EXCEPTIONS
212
213 if (!texture) {
214 LOG(WebGPU, "GPUTexture::createDefaultTextureView(): Unable to create MTLTexture view!");
215 return nullptr;
216 }
217
218 return GPUTexture::create(WTFMove(texture), m_usage);
219}
