33namespace WebCore {

35bool ISOSchemeTypeBox::parse(DataView& view, unsigned& offset)
36{
37 if (!ISOFullBox::parse(view, offset))
38 return false;
39
40 if (!checkedRead<uint32_t>(m_schemeType, view, offset, BigEndian))
41 return false;
42
43 if (!checkedRead<uint32_t>(m_schemeVersion, view, offset, BigEndian))
44 return false;
45
46 return true;
47}
