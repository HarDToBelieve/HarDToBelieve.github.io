26#pragma once

28#include "ISOBox.h"
29#include <wtf/MediaTime.h>
30#include <wtf/text/WTFString.h>

32namespace WebCore {
33
34// 4 bytes : 4CC : identifier = 'vttc'
35// 4 bytes : unsigned : length
36// N bytes : CueSourceIDBox : box : optional
37// N bytes : CueIDBox : box : optional
38// N bytes : CueTimeBox : box : optional
39// N bytes : CueSettingsBox : box : optional
40// N bytes : CuePayloadBox : box : required
41
42class WEBCORE_EXPORT ISOWebVTTCue final : public ISOBox {
43public:
44 ISOWebVTTCue(const MediaTime& presentationTime, const MediaTime& duration);
45 ISOWebVTTCue(MediaTime&& presentationTime, MediaTime&& duration, String&& sourceID, String&& id, String&& originalStartTime, String&& settings, String&& cueText);
46
47 static FourCC boxTypeName() { return "vttc"; }
48
49 const MediaTime& presentationTime() const { return m_presentationTime; }
50 const MediaTime& duration() const { return m_duration; }
51
52 const String& sourceID() const { return m_sourceID; }
53 const String& id() const { return m_identifier; }
54 const String& originalStartTime() const { return m_originalStartTime; }
55 const String& settings() const { return m_settings; }
56 const String& cueText() const { return m_cueText; }
57
58 String toJSONString() const;
59
60 template<class Encoder>
61 void encode(Encoder& encoder) const
62 {
63 encoder << m_presentationTime;
64 encoder << m_duration;
65 encoder << m_sourceID;
66 encoder << m_identifier;
67 encoder << m_originalStartTime;
68 encoder << m_settings;
69 encoder << m_cueText;
70 }
71
72 template <class Decoder>
73 static std::optional<ISOWebVTTCue> decode(Decoder& decoder)
74 {
75 std::optional<MediaTime> presentationTime;
76 decoder >> presentationTime;
77 if (!presentationTime)
78 return std::nullopt;
79
80 std::optional<MediaTime> duration;
81 decoder >> duration;
82 if (!duration)
83 return std::nullopt;
84
85 std::optional<String> sourceID;
86 decoder >> sourceID;
87 if (!sourceID)
88 return std::nullopt;
89
90 std::optional<String> identifier;
91 decoder >> identifier;
92 if (!identifier)
93 return std::nullopt;
94
95 std::optional<String> originalStartTime;
96 decoder >> originalStartTime;
97 if (!originalStartTime)
98 return std::nullopt;
99
100 std::optional<String> settings;
101 decoder >> settings;
102 if (!settings)
103 return std::nullopt;
104
105 std::optional<String> cueText;
106 decoder >> cueText;
107 if (!cueText)
108 return std::nullopt;
109
110 return {{
111 WTFMove(*presentationTime),
112 WTFMove(*duration),
113 WTFMove(*sourceID),
114 WTFMove(*identifier),
115 WTFMove(*originalStartTime),
116 WTFMove(*settings),
117 WTFMove(*cueText)
118 }};
119 }
120
121private:
122 bool parse(JSC::DataView&, unsigned& offset) override;
123
124 MediaTime m_presentationTime;
125 MediaTime m_duration;
126
127 String m_sourceID;
128 String m_identifier;
129 String m_originalStartTime;
130 String m_settings;
131 String m_cueText;
132};
133
134} // namespace WebCore
