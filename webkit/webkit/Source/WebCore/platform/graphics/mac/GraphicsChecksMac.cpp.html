39enum {
40 kAGCOpen,
41 kAGCClose
42};
43
44static io_connect_t attachToAppleGraphicsControl()
45{
46 mach_port_t masterPort = MACH_PORT_NULL;
47
48 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
49 if (IOMasterPort(MACH_PORT_NULL, &masterPort) != KERN_SUCCESS)
50 return IO_OBJECT_NULL;
51 ALLOW_DEPRECATED_DECLARATIONS_END
52
53 CFDictionaryRef classToMatch = IOServiceMatching("AppleGraphicsControl");
54 if (!classToMatch)
55 return IO_OBJECT_NULL;
56
57 kern_return_t kernResult;
58 io_iterator_t iterator;
59 if ((kernResult = IOServiceGetMatchingServices(masterPort, classToMatch, &iterator)) != KERN_SUCCESS)
60 return IO_OBJECT_NULL;
61
62 io_service_t serviceObject = IOIteratorNext(iterator);
63 IOObjectRelease(iterator);
64 if (!serviceObject)
65 return IO_OBJECT_NULL;
66
67 io_connect_t dataPort;
68 IOObjectRetain(serviceObject);
69 kernResult = IOServiceOpen(serviceObject, mach_task_self(), 0, &dataPort);
70 IOObjectRelease(serviceObject);
71
72 return (kernResult == KERN_SUCCESS) ? dataPort : IO_OBJECT_NULL;
73}
74
75static bool hasMuxCapability()
76{
77 io_connect_t dataPort = attachToAppleGraphicsControl();
78
79 if (dataPort == IO_OBJECT_NULL)
80 return false;
81
82 bool result;
83 if (IOConnectCallScalarMethod(dataPort, kAGCOpen, nullptr, 0, nullptr, nullptr) == KERN_SUCCESS) {
84 IOConnectCallScalarMethod(dataPort, kAGCClose, nullptr, 0, nullptr, nullptr);
85 result = true;
86 } else
87 result = false;
88
89 IOServiceClose(dataPort);
90
91 if (result) {
92 // This is detecting Mac hardware with an Intel g575 GPU, which
93 // we don't want to make available to muxing.
94 // Based on information from Apple's OpenGL team, such devices
95 // have four or fewer processors.
96 // <rdar://problem/30060378>
97 int names[2] = { CTL_HW, HW_NCPU };
98 int cpuCount;
99 size_t cpuCountLength = sizeof(cpuCount);
100 sysctl(names, 2, &cpuCount, &cpuCountLength, nullptr, 0);
101 result = cpuCount > 4;
102 }
103
104 return result;
105}

107#endif // HAVE(APPLE_GRAPHICS_CONTROL)

109bool hasLowAndHighPowerGPUs()
110{
111#if HAVE(APPLE_GRAPHICS_CONTROL)
112 static bool canMux = hasMuxCapability();
113 return canMux;
114#else
115 return false;
116#endif
117}
