48// FIXME: Move the code to ChromeClient::iconForFiles().
49RefPtr<Icon> Icon::createIconForFiles(const Vector<String>& filenames)
50{
51 if (filenames.isEmpty())
52 return nullptr;
53
54 bool useIconFromFirstFile;
55 useIconFromFirstFile = filenames.size() == 1;
56 if (useIconFromFirstFile) {
57 // Don't pass relative filenames -- we don't want a result that depends on the current directory.
58 // Need 0U here to disambiguate String::operator[] from operator(NSString*, int)[]
59 if (filenames[0].isEmpty() || filenames[0][0U] != '/')
60 return nullptr;
61
62 NSImage *image = [[NSWorkspace sharedWorkspace] iconForFile:filenames[0]];
63 if (!image)
64 return nullptr;
65
66 return adoptRef(new Icon(image));
67 }
68 NSImage *image = [NSImage imageNamed:NSImageNameMultipleDocuments];
69 if (!image)
70 return nullptr;
71
72 return adoptRef(new Icon(image));
73}

75RefPtr<Icon> Icon::createIconForFileExtension(const String& fileExtension)
76{
77 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
78 NSImage *image = [[NSWorkspace sharedWorkspace] iconForFileType:[@"." stringByAppendingString:fileExtension]];
79 ALLOW_DEPRECATED_DECLARATIONS_END
80 if (!image)
81 return nullptr;
82
83 return adoptRef(new Icon(image));
84}

90 ALLOW_DEPRECATED_DECLARATIONS_END
91 if (!image)
92 return nullptr;
93
94 return adoptRef(new Icon(image));

97void Icon::paint(GraphicsContext& context, const FloatRect& rect)
98{
99 if (context.paintingDisabled())
100 return;
101
102 LocalCurrentGraphicsContext localCurrentGC(context);
103
104 [m_nsImage drawInRect:rect fromRect:NSMakeRect(0, 0, [m_nsImage size].width, [m_nsImage size].height) operation:NSCompositingOperationSourceOver fraction:1.0f];
105}
