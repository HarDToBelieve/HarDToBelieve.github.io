39constexpr unsigned maxUnscheduledFireCount { 20 };
40
41LegacyDisplayRefreshMonitorMac::LegacyDisplayRefreshMonitorMac(PlatformDisplayID displayID)
42 : DisplayRefreshMonitor(displayID)
43{
44 ASSERT(!isInWebProcess());
45 setMaxUnscheduledFireCount(maxUnscheduledFireCount);
46}
47
48LegacyDisplayRefreshMonitorMac::~LegacyDisplayRefreshMonitorMac()
49{
50 ASSERT(!m_displayLink);
51}
52
53void LegacyDisplayRefreshMonitorMac::stop()
54{
55 DisplayRefreshMonitor::stop();
56 LOG_WITH_STREAM(DisplayLink, stream << "LegacyDisplayRefreshMonitorMac::stop for dipslay " << displayID() << " destroying display link");
57 CVDisplayLinkRelease(m_displayLink);
58 m_displayLink = nullptr;
59}
60
61static CVReturn displayLinkCallback(CVDisplayLinkRef, const CVTimeStamp*, const CVTimeStamp*, CVOptionFlags, CVOptionFlags*, void* data)
62{
63 LegacyDisplayRefreshMonitorMac* monitor = static_cast<LegacyDisplayRefreshMonitorMac*>(data);
64 monitor->displayLinkCallbackFired();
65 return kCVReturnSuccess;
66}
67
68void LegacyDisplayRefreshMonitorMac::displayLinkCallbackFired()
69{
70 displayLinkFired(m_currentUpdate);
71 m_currentUpdate = m_currentUpdate.nextUpdate();
72}
73
74void LegacyDisplayRefreshMonitorMac::dispatchDisplayDidRefresh(const DisplayUpdate& displayUpdate)
75{
76 RunLoop::main().dispatch([this, displayUpdate, protectedThis = makeRef(*this)] {
77 if (m_displayLink)
78 displayDidRefresh(displayUpdate);
79 });
80}
81
82WebCore::FramesPerSecond LegacyDisplayRefreshMonitorMac::nominalFramesPerSecondFromDisplayLink(CVDisplayLinkRef displayLink)
83{
84 CVTime refreshPeriod = CVDisplayLinkGetNominalOutputVideoRefreshPeriod(displayLink);
85 return round((double)refreshPeriod.timeScale / (double)refreshPeriod.timeValue);
86}
87
88bool LegacyDisplayRefreshMonitorMac::ensureDisplayLink()
89{
90 if (m_displayLink)
91 return true;
92
93 auto error = CVDisplayLinkCreateWithCGDisplay(displayID(), &m_displayLink);
94 if (error)
95 return false;
96
97 error = CVDisplayLinkSetOutputCallback(m_displayLink, displayLinkCallback, this);
98 if (error)
99 return false;
100
101 return true;
102}
103
104bool LegacyDisplayRefreshMonitorMac::startNotificationMechanism()
105{
106 if (!m_displayLink) {
107 if (!ensureDisplayLink())
108 return false;
109 }
110
111 if (!m_displayLinkIsActive) {
112 LOG_WITH_STREAM(DisplayLink, stream << "LegacyDisplayRefreshMonitorMac::startNotificationMechanism for display " << displayID() << " starting display link");
113
114 auto error = CVDisplayLinkStart(m_displayLink);
115 if (error)
116 return false;
117
118 m_displayLinkIsActive = true;
119 m_currentUpdate = { 0, nominalFramesPerSecondFromDisplayLink(m_displayLink) };
120 }
121
122 return true;
123}
124
125void LegacyDisplayRefreshMonitorMac::stopNotificationMechanism()
126{
127 if (!m_displayLinkIsActive)
128 return;
129
130 if (m_displayLink) {
131 LOG_WITH_STREAM(DisplayLink, stream << "LegacyDisplayRefreshMonitorMac::stopNotificationMechanism for display " << displayID() << " stopping display link");
132 CVDisplayLinkStop(m_displayLink);
133 }
134
135 m_displayLinkIsActive = false;
136}
137
138std::optional<FramesPerSecond> LegacyDisplayRefreshMonitorMac::displayNominalFramesPerSecond()
139{
140 if (!ensureDisplayLink())
141 return std::nullopt;
142
143 return nominalFramesPerSecondFromDisplayLink(m_displayLink);
144}
