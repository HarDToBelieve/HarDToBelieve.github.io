49void PDFDocumentImage::createPDFDocument()
50{
51 m_document = adoptNS([allocPDFDocumentInstance() initWithData:data()->createNSData().get()]);
52}
53
54void PDFDocumentImage::computeBoundsForCurrentPage()
55{
56 PDFPage *pdfPage = [m_document pageAtIndex:0];
57
58 m_cropBox = [pdfPage boundsForBox:kPDFDisplayBoxCropBox];
59 m_rotationDegrees = [pdfPage rotation];
60}
61
62unsigned PDFDocumentImage::pageCount() const
63{
64 return [m_document pageCount];
65}
66
67void PDFDocumentImage::drawPDFPage(GraphicsContext& context)
68{
69 LocalCurrentGraphicsContext localCurrentContext(context);
70
71 // These states can be mutated by PDFKit but are not saved
72 // on the context's state stack. (<rdar://problem/14951759&35738181>)
73 bool allowsSmoothing = CGContextGetAllowsFontSmoothing(context.platformContext());
74 bool allowsSubpixelQuantization = CGContextGetAllowsFontSubpixelQuantization(context.platformContext());
75 bool allowsSubpixelPositioning = CGContextGetAllowsFontSubpixelPositioning(context.platformContext());
76
77 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
78 [[m_document pageAtIndex:0] drawWithBox:kPDFDisplayBoxCropBox];
79 ALLOW_DEPRECATED_DECLARATIONS_END
80
81 CGContextSetAllowsFontSmoothing(context.platformContext(), allowsSmoothing);
82 CGContextSetAllowsFontSubpixelQuantization(context.platformContext(), allowsSubpixelQuantization);
83 CGContextSetAllowsFontSubpixelPositioning(context.platformContext(), allowsSubpixelPositioning);
84}
