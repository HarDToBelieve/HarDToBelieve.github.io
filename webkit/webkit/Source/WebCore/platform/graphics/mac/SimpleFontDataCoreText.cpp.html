35RetainPtr<CFDictionaryRef> Font::getCFStringAttributes(bool enableKerning, FontOrientation orientation, const AtomString& locale) const
36{
37 CFTypeRef keys[5];
38 CFTypeRef values[5];
39
40 keys[0] = kCTFontAttributeName;
41 values[0] = platformData().ctFont();
42 size_t count = 1;
43
44#if USE(CTFONTSHAPEGLYPHS)
45 RetainPtr<CFStringRef> localeString;
46 if (!locale.isEmpty()) {
47 localeString = locale.string().createCFString();
48 keys[count] = kCTLanguageAttributeName;
49 values[count] = localeString.get();
50 ++count;
51 }
52#else
53 UNUSED_PARAM(locale);
54#endif
55 static CTParagraphStyleRef paragraphStyle = [] {
56 auto paragraphStyle = CTParagraphStyleCreate(nullptr, 0);
57 CTParagraphStyleSetCompositionLanguage(paragraphStyle, kCTCompositionLanguageNone);
58 return paragraphStyle;
59 }();
60 keys[count] = kCTParagraphStyleAttributeName;
61 values[count] = paragraphStyle;
62 ++count;
63
64 if (!enableKerning) {
65 const float zero = 0;
66 static NeverDestroyed<RetainPtr<CFNumberRef>> zeroKerningValue = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberFloatType, &zero));
67 keys[count] = kCTKernAttributeName;
68 values[count] = zeroKerningValue.get().get();
69 ++count;
70 }
71
72 if (orientation == FontOrientation::Vertical) {
73 keys[count] = kCTVerticalFormsAttributeName;
74 values[count] = kCFBooleanTrue;
75 ++count;
76 }
77
78 ASSERT(count <= WTF_ARRAY_LENGTH(keys));
79
80 return adoptCF(CFDictionaryCreate(kCFAllocatorDefault, keys, values, count, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
81}
