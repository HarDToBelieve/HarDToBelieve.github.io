33@implementation NSImage (WebKitExtras)
34
35- (void)_web_lockFocusWithDeviceScaleFactor:(CGFloat)deviceScaleFactor
36{
37 NSAffineTransform *transform = [NSAffineTransform transform];
38 [transform scaleBy:deviceScaleFactor];
39
40 NSRect bounds = { NSZeroPoint, self.size };
41 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
42 BOOL flippedOrientation = self.isFlipped;
43 ALLOW_DEPRECATED_DECLARATIONS_END
44 [self lockFocusWithRect:bounds context:nil hints:@{ NSImageHintCTM : transform } flipped:flippedOrientation];
45}
46
47@end
