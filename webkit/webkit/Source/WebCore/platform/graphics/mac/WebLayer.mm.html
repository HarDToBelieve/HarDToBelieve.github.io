50- (void)drawInContext:(CGContextRef)context
51{
52 auto layer = WebCore::PlatformCALayer::platformCALayerForLayer((__bridge void*)self);
53 if (layer) {
54 WebCore::GraphicsContextCG graphicsContext(context);
55 WebCore::PlatformCALayer::RepaintRectList rectsToPaint = WebCore::PlatformCALayer::collectRectsToPaint(graphicsContext, layer.get());
56 WebCore::PlatformCALayer::drawLayerContents(graphicsContext, layer.get(), rectsToPaint, self.isRenderingInContext ? WebCore::GraphicsLayerPaintSnapshotting : WebCore::GraphicsLayerPaintNormal);
57 }
58}

72- (id<CAAction>)actionForKey:(NSString *)key
73{
74 // Fix for <rdar://problem/9015675>: Force the layer content to be updated when the tree is reparented.
75 if ([key isEqualToString:@"onOrderIn"])
76 [self reloadValueForKeyPath:@"contents"];
77
78 return nil;
79}

81- (void)setNeedsDisplay
82{
83 auto layer = WebCore::PlatformCALayer::platformCALayerForLayer((__bridge void*)self);
84 if (layer && layer->owner() && layer->owner()->platformCALayerDrawsContent())
85 [super setNeedsDisplay];
86}
87
88- (void)setNeedsDisplayInRect:(CGRect)dirtyRect
89{
90 auto platformLayer = WebCore::PlatformCALayer::platformCALayerForLayer((__bridge void*)self);
91 if (!platformLayer) {
92 [super setNeedsDisplayInRect:dirtyRect];
93 return;
94 }
95
96 if (WebCore::PlatformCALayerClient* layerOwner = platformLayer->owner()) {
97 if (layerOwner->platformCALayerDrawsContent()) {
98 [super setNeedsDisplayInRect:dirtyRect];
99
100 if (layerOwner->platformCALayerShowRepaintCounter(platformLayer.get())) {
101 CGRect bounds = [self bounds];
102 CGRect indicatorRect = CGRectMake(bounds.origin.x, bounds.origin.y, 52, 27);
103 [super setNeedsDisplayInRect:indicatorRect];
104 }
105 }
106 }
107}

109- (void)display
110{
111#if PLATFORM(IOS_FAMILY)
112 if (pthread_main_np())
113 WebThreadLock();
114#endif
115 ASSERT(isMainThread());
116 [super display];
117 auto layer = WebCore::PlatformCALayer::platformCALayerForLayer((__bridge void*)self);
118 if (layer && layer->owner())
119 layer->owner()->platformCALayerLayerDidDisplay(layer.get());
120}

122- (void)drawInContext:(CGContextRef)context
123{
124#if PLATFORM(IOS_FAMILY)
125 if (pthread_main_np())
126 WebThreadLock();
127#endif
128 ASSERT(isMainThread());
129 auto layer = WebCore::PlatformCALayer::platformCALayerForLayer((__bridge void*)self);
130 if (layer && layer->owner()) {
131 WebCore::GraphicsContextCG graphicsContext(context);
132 graphicsContext.setIsCALayerContext(true);
133 graphicsContext.setIsAcceleratedContext(layer->acceleratesDrawing());
134
135 WebCore::FloatRect clipBounds = CGContextGetClipBoundingBox(context);
136 layer->owner()->platformCALayerPaintContents(layer.get(), graphicsContext, clipBounds, self.isRenderingInContext ? WebCore::GraphicsLayerPaintSnapshotting : WebCore::GraphicsLayerPaintNormal);
137 }
138}

146- (NSString*)_descriptionWithPrefix:(NSString*)inPrefix
147{
148 CGRect aBounds = [self bounds];
149 CGPoint aPos = [self position];
150
151 NSString* selfString = [NSString stringWithFormat:@"%@<%@ 0x%p> \"%@\" bounds(%.1f, %.1f, %.1f, %.1f) pos(%.1f, %.1f), sublayers=%lu masking=%d",
152 inPrefix,
153 [self class],
154 self,
155 [self name],
156 aBounds.origin.x, aBounds.origin.y, aBounds.size.width, aBounds.size.height,
157 aPos.x, aPos.y,
158 static_cast<unsigned long>([[self sublayers] count]),
159 [self masksToBounds]];
160
161 NSMutableString* curDesc = [NSMutableString stringWithString:selfString];
162
163 if ([[self sublayers] count] > 0)
164 [curDesc appendString:@"\n"];
165
166 NSString* sublayerPrefix = [inPrefix stringByAppendingString:@"\t"];
167
168 NSEnumerator* sublayersEnum = [[self sublayers] objectEnumerator];
169 CALayer* curLayer;
170 while ((curLayer = [sublayersEnum nextObject]))
171 [curDesc appendString:[curLayer _descriptionWithPrefix:sublayerPrefix]];
172
173 if ([[self sublayers] count] == 0)
174 [curDesc appendString:@"\n"];
175
176 if (CALayer *mask = [self mask]) {
177 [curDesc appendString:@"mask: "];
178 [curDesc appendString:[mask _descriptionWithPrefix:sublayerPrefix]];
179 }
180
181 return curDesc;
182}
183
184- (NSString*)extendedDescription
185{
186 return [self _descriptionWithPrefix:@""];
187}
