36GraphicsContextGLOpenGLManager& GraphicsContextGLOpenGLManager::sharedManager()
37{
38 static NeverDestroyed<GraphicsContextGLOpenGLManager> s_manager;
39 return s_manager;
40}
41
42#if PLATFORM(MAC)
43void GraphicsContextGLOpenGLManager::displayWasReconfigured(CGDirectDisplayID, CGDisplayChangeSummaryFlags flags, void*)
44{
45 LOG(WebGL, "GraphicsContextGLOpenGLManager::displayWasReconfigured");
46 if (flags & kCGDisplaySetModeFlag)
47 GraphicsContextGLOpenGLManager::sharedManager().displayWasReconfigured();
48}
49#endif
50
51#if PLATFORM(COCOA)
52void GraphicsContextGLOpenGLManager::displayWasReconfigured()
53{
54 for (const auto& context : m_contexts)
55 context->displayWasReconfigured();
56}
57#endif
58
59void GraphicsContextGLOpenGLManager::addContext(GraphicsContextGLOpenGL* context)
60{
61 ASSERT(context);
62 if (!context)
63 return;
64
65 ASSERT(!m_contexts.contains(context));
66 m_contexts.append(context);
67}
68
69void GraphicsContextGLOpenGLManager::removeContext(GraphicsContextGLOpenGL* context)
70{
71 if (!m_contexts.contains(context))
72 return;
73 m_contexts.removeFirst(context);
74}
75
76void GraphicsContextGLOpenGLManager::recycleContextIfNecessary()
77{
78 if (hasTooManyContexts()) {
79 LOG(WebGL, "GraphicsContextGLOpenGLManager recycled context (%p).", m_contexts[0]);
80 m_contexts[0]->recycleContext();
81 }
82}
