40namespace WebCore {
41
42class FontPlatformData;
43class SharedBuffer;
44class Font;
45
46class OpenTypeMathData : public RefCounted<OpenTypeMathData> {
47public:
48 static Ref<OpenTypeMathData> create(const FontPlatformData& font)
49 {
50 return adoptRef(*new OpenTypeMathData(font));
51 }
52 ~OpenTypeMathData();
53
54#if ENABLE(OPENTYPE_MATH)
55 bool hasMathData() const { return m_mathBuffer; }
56#elif USE(HARFBUZZ)
57 bool hasMathData() const { return m_mathFont.get(); }
58#else
59 bool hasMathData() const { return false; }
60#endif
61

65 ScriptPercentScaleDown,
66 ScriptScriptPercentScaleDown,
67 DelimitedSubFormulaMinHeight,

69 MathLeading,
70 AxisHeight,
71 AccentBaseHeight,
72 FlattenedAccentBaseHeight,
73 SubscriptShiftDown,
74 SubscriptTopMax,
75 SubscriptBaselineDropMin,
76 SuperscriptShiftUp,
77 SuperscriptShiftUpCramped,
78 SuperscriptBottomMin,
79 SuperscriptBaselineDropMax,
80 SubSuperscriptGapMin,
81 SuperscriptBottomMaxWithSubscript,
82 SpaceAfterScript,
83 UpperLimitGapMin,
84 UpperLimitBaselineRiseMin,
85 LowerLimitGapMin,
86 LowerLimitBaselineDropMin,
87 StackTopShiftUp,
88 StackTopDisplayStyleShiftUp,
89 StackBottomShiftDown,
90 StackBottomDisplayStyleShiftDown,
91 StackGapMin,
92 StackDisplayStyleGapMin,
93 StretchStackTopShiftUp,
94 StretchStackBottomShiftDown,
95 StretchStackGapAboveMin,
96 StretchStackGapBelowMin,
97 FractionNumeratorShiftUp,
98 FractionNumeratorDisplayStyleShiftUp,
99 FractionDenominatorShiftDown,
100 FractionDenominatorDisplayStyleShiftDown,
101 FractionNumeratorGapMin,
102 FractionNumDisplayStyleGapMin,
103 FractionRuleThickness,
104 FractionDenominatorGapMin,
105 FractionDenomDisplayStyleGapMin,
106 SkewedFractionHorizontalGap,
107 SkewedFractionVerticalGap,
108 OverbarVerticalGap,
109 OverbarRuleThickness,
110 OverbarExtraAscender,
111 UnderbarVerticalGap,
112 UnderbarRuleThickness,
113 UnderbarExtraDescender,
114 RadicalVerticalGap,
115 RadicalDisplayStyleVerticalGap,
116 RadicalRuleThickness,
117 RadicalExtraAscender,
118 RadicalKernBeforeDegree,
119 RadicalKernAfterDegree,
120 RadicalDegreeBottomRaisePercent
121 };
122
123 struct AssemblyPart {
124 Glyph glyph;
125 bool isExtender;
126 };
127
128 float getMathConstant(const Font&, MathConstant) const;
129 float getItalicCorrection(const Font&, Glyph) const;
130 void getMathVariants(Glyph, bool isVertical, Vector<Glyph>& sizeVariants, Vector<AssemblyPart>& assemblyParts) const;
131
132private:
133 explicit OpenTypeMathData(const FontPlatformData&);
134
135#if ENABLE(OPENTYPE_MATH)
136 RefPtr<SharedBuffer> m_mathBuffer;
137#elif USE(HARFBUZZ)
138 HbUniquePtr<hb_font_t> m_mathFont;
139#endif
140};
