37struct BigEndianShort {
38 operator short() const { return (v & 0x00ff) << 8 | v >> 8; }
39 BigEndianShort(short u) : v((u & 0x00ff) << 8 | u >> 8) { }
40 unsigned short v;
41};
42
43struct BigEndianUShort {
44 operator unsigned short() const { return (v & 0x00ff) << 8 | v >> 8; }
45 BigEndianUShort(unsigned short u) : v((u & 0x00ff) << 8 | u >> 8) { }
46 unsigned short v;
47};
48
49struct BigEndianLong {
50 operator int() const { return (v & 0xff) << 24 | (v & 0xff00) << 8 | (v & 0xff0000) >> 8 | v >> 24; }
51 BigEndianLong(int u) : v((u & 0xff) << 24 | (u & 0xff00) << 8 | (u & 0xff0000) >> 8 | u >> 24) { }
52 unsigned v;
53};
54
55struct BigEndianULong {
56 operator unsigned() const { return (v & 0xff) << 24 | (v & 0xff00) << 8 | (v & 0xff0000) >> 8 | v >> 24; }
57 BigEndianULong(unsigned u) : v((u & 0xff) << 24 | (u & 0xff00) << 8 | (u & 0xff0000) >> 8 | u >> 24) { }
58 unsigned v;
59};

71// do endian conversion here but make sure constants are in big-endian order.

76template <typename T> static const T* validateTable(const RefPtr<SharedBuffer>& buffer, size_t count = 1)
77{
78 if (!buffer || buffer->size() < sizeof(T) * count)
79 return 0;
80 return reinterpret_cast<const T*>(buffer->data());
81}
82
83struct TableBase {
84protected:
85 static bool isValidEnd(const SharedBuffer& buffer, const void* position)
86 {
87 if (position < buffer.data())
88 return false;
89 size_t offset = static_cast<const uint8_t*>(position) - buffer.data();
90 return offset <= buffer.size(); // "<=" because end is included as valid
91 }
92
93 template <typename T> static const T* validatePtr(const SharedBuffer& buffer, const void* position)
94 {
95 const T* casted = reinterpret_cast<const T*>(position);
96 if (!isValidEnd(buffer, &casted[1]))
97 return 0;
98 return casted;
99 }
100
101 template <typename T> const T* validateOffset(const SharedBuffer& buffer, uint16_t offset) const
102 {
103 return validatePtr<T>(buffer, reinterpret_cast<const int8_t*>(this) + offset);
104 }
105};

112struct Coverage1Table : CoverageTable {

117struct Coverage2Table : CoverageTable {

127#if ENABLE(OPENTYPE_MATH)
128struct TableWithCoverage : TableBase {
129protected:
130 bool getCoverageIndex(const SharedBuffer& buffer, const CoverageTable* coverage, Glyph glyph, uint32_t& coverageIndex) const
131 {
132 switch (coverage->coverageFormat) {
133 case 1: { // Coverage Format 1
134 const Coverage1Table* coverage1 = validatePtr<Coverage1Table>(buffer, coverage);
135 if (!coverage1)
136 return false;
137 uint16_t glyphCount = coverage1->glyphCount;
138 if (!isValidEnd(buffer, &coverage1->glyphArray[glyphCount]))
139 return false;
140
141 // We do a binary search on the glyph indexes.
142 uint32_t imin = 0, imax = glyphCount;
143 while (imin < imax) {
144 uint32_t imid = (imin + imax) >> 1;
145 uint16_t glyphMid = coverage1->glyphArray[imid];
146 if (glyphMid == glyph) {
147 coverageIndex = imid;
148 return true;
149 }
150 if (glyphMid < glyph)
151 imin = imid + 1;
152 else
153 imax = imid;
154 }
155 break;
156 }
157 case 2: { // Coverage Format 2
158 const Coverage2Table* coverage2 = validatePtr<Coverage2Table>(buffer, coverage);
159 if (!coverage2)
160 return false;
161 uint16_t rangeCount = coverage2->rangeCount;
162 if (!isValidEnd(buffer, &coverage2->ranges[rangeCount]))
163 return false;
164
165 // We do a binary search on the ranges.
166 uint32_t imin = 0, imax = rangeCount;
167 while (imin < imax) {
168 uint32_t imid = (imin + imax) >> 1;
169 uint16_t rStart = coverage2->ranges[imid].start;
170 uint16_t rEnd = coverage2->ranges[imid].end;
171 if (rEnd < glyph)
172 imin = imid + 1;
173 else if (glyph < rStart)
174 imax = imid;
175 else {
176 coverageIndex = coverage2->ranges[imid].startCoverageIndex + glyph - rStart;
177 return true;
178 }
179 }
180 break;
181 }
182 }
183 return false;
184 }
185};
186#endif
