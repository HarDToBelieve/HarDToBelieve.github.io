35bool PerspectiveTransformOperation::operator==(const TransformOperation& other) const
36{
37 if (!isSameType(other))
38 return false;
39 return m_p == downcast<PerspectiveTransformOperation>(other).m_p;
40}
41
42Ref<TransformOperation> PerspectiveTransformOperation::blend(const TransformOperation* from, const BlendingContext& context, bool blendToIdentity)
43{
44 if (from && !from->isSameType(*this))
45 return *this;
46
47 if (blendToIdentity) {
48 double p = floatValueForLength(m_p, 1);
49 p = WebCore::blend(p, 1.0, context); // FIXME: this seems wrong. https://bugs.webkit.org/show_bug.cgi?id=52700
50 return PerspectiveTransformOperation::create(Length(clampToPositiveInteger(p), LengthType::Fixed));
51 }
52
53 const PerspectiveTransformOperation* fromOp = downcast<PerspectiveTransformOperation>(from);
54 Length fromP = fromOp ? fromOp->m_p : Length(m_p.type());
55 Length toP = m_p;
56
57 TransformationMatrix fromT;
58 TransformationMatrix toT;
59 fromT.applyPerspective(floatValueForLength(fromP, 1));
60 toT.applyPerspective(floatValueForLength(toP, 1));
61 toT.blend(fromT, context.progress);
62 TransformationMatrix::Decomposed4Type decomp;
63 toT.decompose4(decomp);
64
65 if (decomp.perspectiveZ) {
66 double val = -1.0 / decomp.perspectiveZ;
67 return PerspectiveTransformOperation::create(Length(clampToPositiveInteger(val), LengthType::Fixed));
68 }
69 return PerspectiveTransformOperation::create(Length(0, LengthType::Fixed));
70}

74 ts << type() << "(" << m_p << ")";
