25#pragma once
26
27#include "TransformOperation.h"
28#include <wtf/Ref.h>
29
30namespace WebCore {
31
32struct BlendingContext;
33
34class SkewTransformOperation final : public TransformOperation {
35public:
36 static Ref<SkewTransformOperation> create(double angleX, double angleY, OperationType type)
37 {
38 return adoptRef(*new SkewTransformOperation(angleX, angleY, type));
39 }
40
41 Ref<TransformOperation> clone() const override
42 {
43 return adoptRef(*new SkewTransformOperation(m_angleX, m_angleY, type()));
44 }
45
46 double angleX() const { return m_angleX; }
47 double angleY() const { return m_angleY; }
48
49private:
50 bool isIdentity() const override { return m_angleX == 0 && m_angleY == 0; }
51 bool isAffectedByTransformOrigin() const override { return !isIdentity(); }
52
53 bool operator==(const TransformOperation&) const override;
54
55 bool apply(TransformationMatrix& transform, const FloatSize&) const override
56 {
57 transform.skew(m_angleX, m_angleY);
58 return false;
59 }
60
61 Ref<TransformOperation> blend(const TransformOperation* from, const BlendingContext&, bool blendToIdentity = false) override;
62
63 void dump(WTF::TextStream&) const final;
64
65 SkewTransformOperation(double angleX, double angleY, OperationType type)
66 : TransformOperation(type)
67 , m_angleX(angleX)
68 , m_angleY(angleY)
69 {
70 ASSERT(isSkewTransformOperationType());
71 }
72
73 double m_angleX;
74 double m_angleY;
75};
76
77} // namespace WebCore
78
79SPECIALIZE_TYPE_TRAITS_TRANSFORMOPERATION(WebCore::SkewTransformOperation, isSkewTransformOperationType())
