30#include <wtf/text/TextStream.h>

32namespace WebCore {
33
34void IdentityTransformOperation::dump(TextStream& ts) const
35{
36 ts << type();
37}
38
39TextStream& operator<<(TextStream& ts, TransformOperation::OperationType type)
40{
41 switch (type) {
42 case TransformOperation::SCALE_X: ts << "scaleX"; break;
43 case TransformOperation::SCALE_Y: ts << "scaleY"; break;
44 case TransformOperation::SCALE: ts << "scale"; break;
45 case TransformOperation::TRANSLATE_X: ts << "translateX"; break;
46 case TransformOperation::TRANSLATE_Y: ts << "translateY"; break;
47 case TransformOperation::TRANSLATE: ts << "translate"; break;
48 case TransformOperation::ROTATE: ts << "rotate"; break;
49 case TransformOperation::SKEW_X: ts << "skewX"; break;
50 case TransformOperation::SKEW_Y: ts << "skewY"; break;
51 case TransformOperation::SKEW: ts << "skew"; break;
52 case TransformOperation::MATRIX: ts << "matrix"; break;
53 case TransformOperation::SCALE_Z: ts << "scaleX"; break;
54 case TransformOperation::SCALE_3D: ts << "scale3d"; break;
55 case TransformOperation::TRANSLATE_Z: ts << "translateZ"; break;
56 case TransformOperation::TRANSLATE_3D: ts << "translate3d"; break;
57 case TransformOperation::ROTATE_X: ts << "rotateX"; break;
58 case TransformOperation::ROTATE_Y: ts << "rotateY"; break;
59 case TransformOperation::ROTATE_3D: ts << "rotate3d"; break;
60 case TransformOperation::MATRIX_3D: ts << "matrix3d"; break;
61 case TransformOperation::PERSPECTIVE: ts << "perspective"; break;
62 case TransformOperation::IDENTITY: ts << "identity"; break;
63 case TransformOperation::NONE: ts << "none"; break;
64 }
65
66 return ts;
67}
68
69TextStream& operator<<(TextStream& ts, const TransformOperation& operation)
70{
71 operation.dump(ts);
72 return ts;
73}
