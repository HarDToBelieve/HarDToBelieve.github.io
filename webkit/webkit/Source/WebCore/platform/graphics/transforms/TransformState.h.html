39namespace WebCore {
40
41class TransformState {
42public:
43 enum TransformDirection { ApplyTransformDirection, UnapplyInverseTransformDirection };
44 enum TransformAccumulation { FlattenTransform, AccumulateTransform };
45
46 TransformState(TransformDirection mappingDirection, const FloatPoint& p, const FloatQuad& quad)
47 : m_lastPlanarPoint(p)
48 , m_lastPlanarQuad(quad)
49 , m_accumulatingTransform(false)
50 , m_mapPoint(true)
51 , m_mapQuad(true)
52 , m_direction(mappingDirection)
53 {
54 }
55
56 TransformState(TransformDirection mappingDirection, const FloatPoint& p)
57 : m_lastPlanarPoint(p)
58 , m_accumulatingTransform(false)
59 , m_mapPoint(true)
60 , m_mapQuad(false)
61 , m_direction(mappingDirection)
62 {
63 }
64
65 TransformState(TransformDirection mappingDirection, const FloatQuad& quad)
66 : m_lastPlanarQuad(quad)
67 , m_accumulatingTransform(false)
68 , m_mapPoint(false)
69 , m_mapQuad(true)
70 , m_direction(mappingDirection)
71 {
72 }
73
74 TransformState(const TransformState& other) { *this = other; }
75
76 TransformState& operator=(const TransformState&);
77
78 void setQuad(const FloatQuad& quad)
79 {
80 // We must be in a flattened state (no accumulated offset) when setting this quad.
81 ASSERT(m_accumulatedOffset == LayoutSize());
82 m_lastPlanarQuad = quad;
83 }
84
85 void setSecondaryQuad(const std::optional<FloatQuad>& quad)
86 {
87 // We must be in a flattened state (no accumulated offset) when setting this secondary quad.
88 ASSERT(m_accumulatedOffset == LayoutSize());
89 m_lastPlanarSecondaryQuad = quad;
90 }
91
92 void setLastPlanarSecondaryQuad(const std::optional<FloatQuad>&);
93
94 void move(LayoutUnit x, LayoutUnit y, TransformAccumulation accumulate = FlattenTransform)
95 {
96 move(LayoutSize(x, y), accumulate);
97 }
98
99 void move(const LayoutSize&, TransformAccumulation = FlattenTransform);
100 void applyTransform(const AffineTransform& transformFromContainer, TransformAccumulation = FlattenTransform, bool* wasClamped = nullptr);
101 void applyTransform(const TransformationMatrix& transformFromContainer, TransformAccumulation = FlattenTransform, bool* wasClamped = nullptr);
102 void flatten(bool* wasClamped = nullptr);
103
104 // Return the coords of the point or quad in the last flattened layer
105 FloatPoint lastPlanarPoint() const { return m_lastPlanarPoint; }
106 FloatQuad lastPlanarQuad() const { return m_lastPlanarQuad; }
107 std::optional<FloatQuad> lastPlanarSecondaryQuad() const { return m_lastPlanarSecondaryQuad; }
108 bool isMappingSecondaryQuad() const { return m_lastPlanarSecondaryQuad.has_value(); }
109
110 // Return the point or quad mapped through the current transform
111 FloatPoint mappedPoint(bool* wasClamped = nullptr) const;
112 FloatQuad mappedQuad(bool* wasClamped = nullptr) const;
113 std::optional<FloatQuad> mappedSecondaryQuad(bool* wasClamped = nullptr) const;
114
115 TransformationMatrix* accumulatedTransform() const { return m_accumulatedTransform.get(); }
116
117private:
118 void translateTransform(const LayoutSize&);
119 void translateMappedCoordinates(const LayoutSize&);
120 void flattenWithTransform(const TransformationMatrix&, bool* wasClamped);
121 void applyAccumulatedOffset();
122
123 TransformDirection direction() const { return m_direction; }
124 TransformDirection inverseDirection() const;
125
126 void mapQuad(FloatQuad&, TransformDirection, bool* clamped = nullptr) const;
127
128 FloatPoint m_lastPlanarPoint;
129 FloatQuad m_lastPlanarQuad;
130 std::optional<FloatQuad> m_lastPlanarSecondaryQuad;
131
132 // We only allocate the transform if we need to
133 std::unique_ptr<TransformationMatrix> m_accumulatedTransform;
134 LayoutSize m_accumulatedOffset;
135 bool m_accumulatingTransform;
136 bool m_mapPoint;
137 bool m_mapQuad;
138 TransformDirection m_direction;
139};
140
141inline TransformState::TransformDirection TransformState::inverseDirection() const
142{
143 return m_direction == ApplyTransformDirection ? UnapplyInverseTransformDirection : ApplyTransformDirection;
144}
145
146WTF::TextStream& operator<<(WTF::TextStream&, const TransformState&);
147
