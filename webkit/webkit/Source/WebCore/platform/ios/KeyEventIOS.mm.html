202int windowsKeyCodeForCharCode(unichar charCode)
203{
204 switch (charCode) {
205 case 8: case 0x7F: return VK_BACK;
206 case 9: return VK_TAB;
207 case 0xD: case 3: return VK_RETURN;
208 case ' ': return VK_SPACE;
209
210 case '0': case ')': return VK_0;
211 case '1': case '!': return VK_1;
212 case '2': case '@': return VK_2;
213 case '3': case '#': return VK_3;
214 case '4': case '$': return VK_4;
215 case '5': case '%': return VK_5;
216 case '6': case '^': return VK_6;
217 case '7': case '&': return VK_7;
218 case '8': case '*': return VK_8;
219 case '9': case '(': return VK_9;
220 case 'a': case 'A': return VK_A;
221 case 'b': case 'B': return VK_B;
222 case 'c': case 'C': return VK_C;
223 case 'd': case 'D': return VK_D;
224 case 'e': case 'E': return VK_E;
225 case 'f': case 'F': return VK_F;
226 case 'g': case 'G': return VK_G;
227 case 'h': case 'H': return VK_H;
228 case 'i': case 'I': return VK_I;
229 case 'j': case 'J': return VK_J;
230 case 'k': case 'K': return VK_K;
231 case 'l': case 'L': return VK_L;
232 case 'm': case 'M': return VK_M;
233 case 'n': case 'N': return VK_N;
234 case 'o': case 'O': return VK_O;
235 case 'p': case 'P': return VK_P;
236 case 'q': case 'Q': return VK_Q;
237 case 'r': case 'R': return VK_R;
238 case 's': case 'S': return VK_S;
239 case 't': case 'T': return VK_T;
240 case 'u': case 'U': return VK_U;
241 case 'v': case 'V': return VK_V;
242 case 'w': case 'W': return VK_W;
243 case 'x': case 'X': return VK_X;
244 case 'y': case 'Y': return VK_Y;
245 case 'z': case 'Z': return VK_Z;
246
247 case 0x1B: return VK_ESCAPE; // WebKit generated code for Escape.
248
249 // WebKit uses Unicode PUA codes in the OpenStep reserve range for some special keys.
250 case NSUpArrowFunctionKey: return VK_UP;
251 case NSDownArrowFunctionKey: return VK_DOWN;
252 case NSLeftArrowFunctionKey: return VK_LEFT;
253 case NSRightArrowFunctionKey: return VK_RIGHT;
254 case NSPageUpFunctionKey: return VK_PRIOR;
255 case NSPageDownFunctionKey: return VK_NEXT;
256
257 // This is for U.S. keyboard mapping, and doesn't necessarily make sense for different keyboard layouts.
258 // For example, '"' on Windows Russian layout is VK_2, not VK_OEM_7.
259 case ';': case ':': return VK_OEM_1;
260 case '=': case '+': return VK_OEM_PLUS;
261 case ',': case '<': return VK_OEM_COMMA;
262 case '-': case '_': return VK_OEM_MINUS;
263 case '.': case '>': return VK_OEM_PERIOD;
264 case '/': case '?': return VK_OEM_2;
265 case '`': case '~': return VK_OEM_3;
266 case '[': case '{': return VK_OEM_4;
267 case '\\': case '|': return VK_OEM_5;
268 case ']': case '}': return VK_OEM_6;
269 case '\'': case '"': return VK_OEM_7;
270 }
271 return 0;
272}

274static bool isFunctionKey(UChar charCode)
275{
276 switch (charCode) {
277#if !USE(UIKIT_KEYBOARD_ADDITIONS)
278 case 1: // Home
279 case 4: // End
280 case 5: // FIXME: For some reason WebKitTestRunner generates this code for F14 (why?).
281 case 0x7F: // Forward Delete
282 case 0x10: // Function key (e.g. F1, F2, ...)
283#endif
284
285 // WebKit uses Unicode PUA codes in the OpenStep reserve range for some special keys.
286 case NSUpArrowFunctionKey:
287 case NSDownArrowFunctionKey:
288 case NSLeftArrowFunctionKey:
289 case NSRightArrowFunctionKey:
290 case NSPageUpFunctionKey:
291 case NSPageDownFunctionKey:
292 case NSClearLineFunctionKey: // Num Lock / Clear
293#if USE(UIKIT_KEYBOARD_ADDITIONS)
294 case NSDeleteFunctionKey: // Forward delete
295 case NSEndFunctionKey:
296 case NSInsertFunctionKey:
297 case NSHomeFunctionKey:
298#endif
299 return true;
300 }
301#if USE(UIKIT_KEYBOARD_ADDITIONS)
302 if (charCode >= NSF1FunctionKey && charCode <= NSF24FunctionKey)
303 return true;
304#endif
305 return false;
306}

308void PlatformKeyboardEvent::disambiguateKeyDownEvent(Type type, bool backwardCompatibilityMode)
309{
310 // Can only change type from KeyDown to RawKeyDown or Char, as we lack information for other conversions.
311 ASSERT(m_type == KeyDown);
312 ASSERT(type == RawKeyDown || type == Char);
313 m_type = type;
314 if (backwardCompatibilityMode)
315 return;
316
317 if (type == PlatformEvent::RawKeyDown) {
318 m_text = String();
319 m_unmodifiedText = String();
320 } else {
321 m_keyIdentifier = String();
322 m_windowsVirtualKeyCode = 0;
323 if (m_text.length() == 1 && isFunctionKey(m_text[0U])) {
324 m_text = String();
325 m_unmodifiedText = String();
326 }
327 }
328}

330OptionSet<PlatformEvent::Modifier> PlatformKeyboardEvent::currentStateOfModifierKeys()
331{
332 // s_currentModifiers is only set in the WebContent process, not in the UI process.
333 if (s_currentModifiers) {
334 ASSERT(isMainThread());
335 return *s_currentModifiers;
336 }
337
338 ::WebEventFlags currentModifiers = [::WebEvent modifierFlags];
339
340 OptionSet<PlatformEvent::Modifier> modifiers;
341 if (currentModifiers & ::WebEventFlagMaskShiftKey)
342 modifiers.add(PlatformEvent::Modifier::ShiftKey);
343 if (currentModifiers & ::WebEventFlagMaskControlKey)
344 modifiers.add(PlatformEvent::Modifier::ControlKey);
345 if (currentModifiers & ::WebEventFlagMaskOptionKey)
346 modifiers.add(PlatformEvent::Modifier::AltKey);
347 if (currentModifiers & ::WebEventFlagMaskCommandKey)
348 modifiers.add(PlatformEvent::Modifier::MetaKey);
349 if (currentModifiers & ::WebEventFlagMaskLeftCapsLockKey)
350 modifiers.add(PlatformEvent::Modifier::CapsLockKey);
351
352 return modifiers;
353}
