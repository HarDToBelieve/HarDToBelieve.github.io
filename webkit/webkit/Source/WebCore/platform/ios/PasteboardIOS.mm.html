164void Pasteboard::read(PasteboardPlainText& text, PlainTextURLReadingPolicy allowURL, std::optional<size_t> itemIndex)
165{
166 auto itemIndexToQuery = itemIndex.value_or(0);
167
168 PasteboardStrategy& strategy = *platformStrategies()->pasteboardStrategy();
169
170 if (allowURL == PlainTextURLReadingPolicy::AllowURL) {
171 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
172 text.text = strategy.readStringFromPasteboard(itemIndexToQuery, kUTTypeURL, m_pasteboardName, context());
173 ALLOW_DEPRECATED_DECLARATIONS_END
174 if (!text.text.isEmpty()) {
175 text.isURL = true;
176 return;
177 }
178 }
179
180 // We ask for the "public.text" representation only as a legacy fallback, in case apps still directly write
181 // plain text for this abstract UTI. In almost all cases, the more correct choice would be to write to
182 // one of the concrete "public.plain-text" representations (e.g. kUTTypeUTF8PlainText). In the future, we
183 // should consider removing support for reading plain text from "public.text".
184ALLOW_DEPRECATED_DECLARATIONS_BEGIN
185 text.text = strategy.readStringFromPasteboard(itemIndexToQuery, kUTTypePlainText, m_pasteboardName, context());
186 if (text.text.isEmpty())
187 text.text = strategy.readStringFromPasteboard(itemIndexToQuery, kUTTypeText, m_pasteboardName, context());
188ALLOW_DEPRECATED_DECLARATIONS_END
189
190 text.isURL = false;
191}

212Pasteboard::ReaderResult Pasteboard::readPasteboardWebContentDataForType(PasteboardWebContentReader& reader, PasteboardStrategy& strategy, NSString *type, const PasteboardItemInfo& itemInfo, int itemIndex)
213{
214ALLOW_DEPRECATED_DECLARATIONS_BEGIN
215 if ([type isEqualToString:WebArchivePboardType] || [type isEqualToString:(__bridge NSString *)kUTTypeWebArchive]) {
216 auto buffer = strategy.readBufferFromPasteboard(itemIndex, type, m_pasteboardName, context());
217 if (m_changeCount != changeCount())
218 return ReaderResult::PasteboardWasChangedExternally;
219 return buffer && reader.readWebArchive(*buffer) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
220 }
221
222 if ([type isEqualToString:(__bridge NSString *)kUTTypeHTML]) {
223 String htmlString = strategy.readStringFromPasteboard(itemIndex, kUTTypeHTML, m_pasteboardName, context());
224 if (m_changeCount != changeCount())
225 return ReaderResult::PasteboardWasChangedExternally;
226 return !htmlString.isNull() && reader.readHTML(htmlString) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
227 }
228
229 if ([type isEqualToString:(__bridge NSString *)kUTTypeVCard]) {
230 // When dropping or pasting a virtual contact file in editable content, there's never a case where we
231 // would want to dump the entire contents of the file as plain text. Instead, fall back on another
232 // appropriate representation, such as a URL or plain text. For instance, in the case of an MKMapItem,
233 // we would prefer to insert an Apple Maps link instead.
234 return ReaderResult::DidNotReadType;
235 }
236
237#if !PLATFORM(MACCATALYST)
238 if ([type isEqualToString:(__bridge NSString *)kUTTypeFlatRTFD]) {
239 RefPtr<SharedBuffer> buffer = strategy.readBufferFromPasteboard(itemIndex, kUTTypeFlatRTFD, m_pasteboardName, context());
240 if (m_changeCount != changeCount())
241 return ReaderResult::PasteboardWasChangedExternally;
242 return buffer && reader.readRTFD(*buffer) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
243 }
244
245 if ([type isEqualToString:(__bridge NSString *)kUTTypeRTF]) {
246 RefPtr<SharedBuffer> buffer = strategy.readBufferFromPasteboard(itemIndex, kUTTypeRTF, m_pasteboardName, context());
247 if (m_changeCount != changeCount())
248 return ReaderResult::PasteboardWasChangedExternally;
249 return buffer && reader.readRTF(*buffer) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
250 }
251#endif // !PLATFORM(MACCATALYST)
252
253 if ([supportedImageTypes() containsObject:type]) {
254 RefPtr<SharedBuffer> buffer = strategy.readBufferFromPasteboard(itemIndex, type, m_pasteboardName, context());
255 if (m_changeCount != changeCount())
256 return ReaderResult::PasteboardWasChangedExternally;
257 return buffer && reader.readImage(buffer.releaseNonNull(), type, itemInfo.preferredPresentationSize) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
258 }
259
260 if ([type isEqualToString:(__bridge NSString *)kUTTypeURL]) {
261 String title;
262 URL url = strategy.readURLFromPasteboard(itemIndex, m_pasteboardName, title, context());
263 if (m_changeCount != changeCount())
264 return ReaderResult::PasteboardWasChangedExternally;
265 return !url.isNull() && reader.readURL(url, title) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
266 }
267
268 if (UTTypeConformsTo((__bridge CFStringRef)type, kUTTypePlainText)) {
269 String string = strategy.readStringFromPasteboard(itemIndex, kUTTypePlainText, m_pasteboardName, context());
270 if (m_changeCount != changeCount())
271 return ReaderResult::PasteboardWasChangedExternally;
272 return !string.isNull() && reader.readPlainText(string) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
273 }
274
275 if (UTTypeConformsTo((__bridge CFStringRef)type, kUTTypeText)) {
276 String string = strategy.readStringFromPasteboard(itemIndex, kUTTypeText, m_pasteboardName, context());
277 if (m_changeCount != changeCount())
278 return ReaderResult::PasteboardWasChangedExternally;
279 return !string.isNull() && reader.readPlainText(string) ? ReaderResult::ReadType : ReaderResult::DidNotReadType;
280 }
281ALLOW_DEPRECATED_DECLARATIONS_END
282
283 return ReaderResult::DidNotReadType;
284}

286static void readURLAlongsideAttachmentIfNecessary(PasteboardWebContentReader& reader, PasteboardStrategy& strategy, const String& typeIdentifier, const String& pasteboardName, int itemIndex, const PasteboardContext* context)
287{
288ALLOW_DEPRECATED_DECLARATIONS_BEGIN
289 if (!UTTypeConformsTo(typeIdentifier.createCFString().get(), kUTTypeVCard))
290 return;
291ALLOW_DEPRECATED_DECLARATIONS_END

295 if (!url.isEmpty())
296 reader.readURL(url, title);
297}

299static bool prefersAttachmentRepresentation(const PasteboardItemInfo& info)
300{
301 auto contentTypeForHighestFidelityItem = info.contentTypeForHighestFidelityItem();
302 if (contentTypeForHighestFidelityItem.isEmpty())
303 return false;
304
305 if (info.preferredPresentationStyle == PasteboardItemPresentationStyle::Inline)
306 return false;
307
308ALLOW_DEPRECATED_DECLARATIONS_BEGIN
309 return info.canBeTreatedAsAttachmentOrFile() || UTTypeConformsTo(contentTypeForHighestFidelityItem.createCFString().get(), kUTTypeVCard);
310ALLOW_DEPRECATED_DECLARATIONS_END
311}

313void Pasteboard::read(PasteboardWebContentReader& reader, WebContentReadingPolicy policy, std::optional<size_t> itemIndex)
314{
315 reader.contentOrigin = readOrigin();
316 if (respectsUTIFidelities()) {
317 readRespectingUTIFidelities(reader, policy, itemIndex);
318 return;
319 }
320
321 PasteboardStrategy& strategy = *platformStrategies()->pasteboardStrategy();
322
323 size_t numberOfItems = strategy.getPasteboardItemsCount(m_pasteboardName, context());
324
325 if (!numberOfItems)
326 return;
327
328 NSArray *types = supportedWebContentPasteboardTypes();
329 int numberOfTypes = [types count];
330
331#if ENABLE(ATTACHMENT_ELEMENT)
332 bool canReadAttachment = policy == WebContentReadingPolicy::AnyType && RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled();
333#else
334 bool canReadAttachment = false;
335#endif
336
337 for (size_t i = 0; i < numberOfItems; i++) {
338 if (itemIndex && i != *itemIndex)
339 continue;
340
341 auto info = strategy.informationForItemAtIndex(i, m_pasteboardName, m_changeCount, context());
342 if (!info)
343 return;
344
345#if ENABLE(ATTACHMENT_ELEMENT)
346 if (canReadAttachment && prefersAttachmentRepresentation(*info)) {
347 auto typeForFileUpload = info->contentTypeForHighestFidelityItem();
348 if (auto buffer = strategy.readBufferFromPasteboard(i, typeForFileUpload, m_pasteboardName, context())) {
349 readURLAlongsideAttachmentIfNecessary(reader, strategy, typeForFileUpload, m_pasteboardName, i, context());
350 reader.readDataBuffer(*buffer, typeForFileUpload, info->suggestedFileName, info->preferredPresentationSize);
351 continue;
352 }
353 }
354#endif
355
356 for (int typeIndex = 0; typeIndex < numberOfTypes; typeIndex++) {
357 NSString *type = [types objectAtIndex:typeIndex];
358 if (!isTypeAllowedByReadingPolicy(type, policy))
359 continue;
360
361 auto itemResult = readPasteboardWebContentDataForType(reader, strategy, type, *info, i);
362 if (itemResult == ReaderResult::PasteboardWasChangedExternally)
363 return;
364
365 if (itemResult == ReaderResult::ReadType)
366 break;
367 }
368 }
369}

381void Pasteboard::readRespectingUTIFidelities(PasteboardWebContentReader& reader, WebContentReadingPolicy policy, std::optional<size_t> itemIndex)
382{
383 ASSERT(respectsUTIFidelities());
384 auto& strategy = *platformStrategies()->pasteboardStrategy();
385 for (NSUInteger index = 0, numberOfItems = strategy.getPasteboardItemsCount(m_pasteboardName, context()); index < numberOfItems; ++index) {
386 if (itemIndex && index != *itemIndex)
387 continue;
388
389#if ENABLE(ATTACHMENT_ELEMENT)
390 auto info = strategy.informationForItemAtIndex(index, m_pasteboardName, m_changeCount, context());
391 if (!info)
392 return;
393
394 auto attachmentFilePath = info->pathForHighestFidelityItem();
395 bool canReadAttachment = policy == WebContentReadingPolicy::AnyType && RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled() && !attachmentFilePath.isEmpty();
396 auto contentType = info->contentTypeForHighestFidelityItem();
397 if (canReadAttachment && prefersAttachmentRepresentation(*info)) {
398 readURLAlongsideAttachmentIfNecessary(reader, strategy, contentType, m_pasteboardName, index, context());
399 reader.readFilePath(WTFMove(attachmentFilePath), info->preferredPresentationSize, contentType);
400 continue;
401 }
402#endif
403 // Try to read data from each type identifier that this pasteboard item supports, and WebKit also recognizes. Type identifiers are
404 // read in order of fidelity, as specified by each pasteboard item.
405 ReaderResult result = ReaderResult::DidNotReadType;
406 for (auto& type : info->platformTypesByFidelity) {
407 if (!isTypeAllowedByReadingPolicy(type, policy))
408 continue;
409
410 result = readPasteboardWebContentDataForType(reader, strategy, type, *info, index);
411 if (result == ReaderResult::PasteboardWasChangedExternally)
412 return;
413 if (result == ReaderResult::ReadType)
414 break;
415 }
416#if ENABLE(ATTACHMENT_ELEMENT)
417 if (canReadAttachment && result == ReaderResult::DidNotReadType)
418 reader.readFilePath(WTFMove(attachmentFilePath), info->preferredPresentationSize, contentType);
419#endif
420 }

499Vector<String> Pasteboard::readPlatformValuesAsStrings(const String& domType, int64_t changeCount, const String& pasteboardName)
500{
501 auto& strategy = *platformStrategies()->pasteboardStrategy();
502
503 // Grab the value off the pasteboard corresponding to the cocoaType.
504 auto cocoaType = cocoaTypeFromHTMLClipboardType(domType);
505 if (!cocoaType)
506 return { };
507
508 auto values = strategy.allStringsForType(cocoaType.get(), pasteboardName, context());
509ALLOW_DEPRECATED_DECLARATIONS_BEGIN
510 if ([cocoaType isEqualToString:(__bridge NSString *)kUTTypePlainText]) {
511 values = values.map([&] (auto& value) -> String {
512 return [value precomposedStringWithCanonicalMapping];
513 });
514 }
515ALLOW_DEPRECATED_DECLARATIONS_END
516
517 // Enforce changeCount ourselves for security. We check after reading instead of before to be
518 // sure it doesn't change between our testing the change count and accessing the data.
519 if (changeCount != changeCountForPasteboard(pasteboardName))
520 return { };
521
522 return values;
523}

525void Pasteboard::addHTMLClipboardTypesForCocoaType(ListHashSet<String>& resultTypes, const String& cocoaType)
526{
527ALLOW_DEPRECATED_DECLARATIONS_BEGIN
528 // UTI may not do these right, so make sure we get the right, predictable result.
529 if ([cocoaType isEqualToString:(NSString *)kUTTypePlainText]
530 || [cocoaType isEqualToString:(NSString *)kUTTypeUTF8PlainText]
531 || [cocoaType isEqualToString:(NSString *)kUTTypeUTF16PlainText]) {
532 resultTypes.add("text/plain"_s);
533 return;
534 }
535 if ([cocoaType isEqualToString:(NSString *)kUTTypeURL]) {
536 resultTypes.add("text/uri-list"_s);
537 return;
538 }
539 if ([cocoaType isEqualToString:(NSString *)kUTTypeHTML]) {
540 resultTypes.add("text/html"_s);
541 // We don't return here for App compatibility.
542 }
543ALLOW_DEPRECATED_DECLARATIONS_END
544 if (Pasteboard::shouldTreatCocoaTypeAsFile(cocoaType))
545 return;
546 String utiType = utiTypeFromCocoaType(cocoaType);
547 if (!utiType.isEmpty()) {
548 resultTypes.add(utiType);
549 return;
550 }
551 // No mapping, just pass the whole string though.
552 resultTypes.add(cocoaType);
553}

564Vector<String> Pasteboard::readFilePaths()
565{
566 Vector<String> filePaths;
567 auto& strategy = *platformStrategies()->pasteboardStrategy();
568 for (NSUInteger index = 0, numberOfItems = strategy.getPasteboardItemsCount(m_pasteboardName, context()); index < numberOfItems; ++index) {
569 // Currently, drag and drop is the only case on iOS where the "pasteboard" may contain file paths.
570 auto info = strategy.informationForItemAtIndex(index, m_pasteboardName, m_changeCount, context());
571 if (!info)
572 return { };
573
574 auto filePath = info->pathForHighestFidelityItem();
575 if (!filePath.isEmpty())
576 filePaths.append(WTFMove(filePath));
577 }
578 return filePaths;
579}
