42static OptionSet<PlatformEvent::Modifier> modifiersForEvent(WebEvent *event)
43{
44 OptionSet<PlatformEvent::Modifier> modifiers;
45
46 if (event.modifierFlags & WebEventFlagMaskShiftKey)
47 modifiers.add(PlatformEvent::Modifier::ShiftKey);
48 if (event.modifierFlags & WebEventFlagMaskControlKey)
49 modifiers.add(PlatformEvent::Modifier::ControlKey);
50 if (event.modifierFlags & WebEventFlagMaskOptionKey)
51 modifiers.add(PlatformEvent::Modifier::AltKey);
52 if (event.modifierFlags & WebEventFlagMaskCommandKey)
53 modifiers.add(PlatformEvent::Modifier::MetaKey);
54 if (event.modifierFlags & WebEventFlagMaskLeftCapsLockKey)
55 modifiers.add(PlatformEvent::Modifier::CapsLockKey);
56
57 return modifiers;
58}

71static PlatformEvent::Type mouseEventType(WebEvent *event)
72{
73 switch (event.type) {
74 case WebEventMouseDown:
75 return PlatformEvent::MousePressed;
76 case WebEventMouseUp:
77 return PlatformEvent::MouseReleased;
78 case WebEventMouseMoved:
79 return PlatformEvent::MouseMoved;
80 default:
81 ASSERT_NOT_REACHED();
82 return PlatformEvent::MousePressed;
83 }
84}

128String keyIdentifierForKeyEvent(WebEvent *event)
129{
130 if (event.keyboardFlags & WebEventKeyboardInputModifierFlagsChanged) {
131 switch (event.keyCode) {
132 case VK_LWIN: // Left Command
133 case VK_APPS: // Right Command
134 return "Meta"_s;
135
136 case VK_CAPITAL: // Caps Lock
137 return "CapsLock"_s;
138
139 case VK_LSHIFT: // Left Shift
140 case VK_RSHIFT: // Right Shift
141 return "Shift"_s;
142
143 case VK_LMENU: // Left Alt
144 case VK_RMENU: // Right Alt
145 return "Alt"_s;
146
147 case VK_LCONTROL: // Left Ctrl
148 case VK_RCONTROL: // Right Ctrl
149 return "Control"_s;
150
151 default:
152 ASSERT_NOT_REACHED();
153 return emptyString();
154 }
155 }
156 NSString *characters = event.charactersIgnoringModifiers;
157 if ([characters length] != 1) {
158 LOG(Events, "received an unexpected number of characters in key event: %u", [characters length]);
159 return "Unidentified"_s;
160 }
161 return keyIdentifierForCharCode([characters characterAtIndex:0]);
162}

164String keyForKeyEvent(WebEvent *event)
165{
166 if (event.keyboardFlags & WebEventKeyboardInputModifierFlagsChanged) {
167 switch (event.keyCode) {
168 case VK_LWIN: // Left Command
169 case VK_APPS: // Right Command
170 return "Meta"_s;
171
172 case VK_CAPITAL: // Caps Lock
173 return "CapsLock"_s;
174
175 case VK_LSHIFT: // Left Shift
176 case VK_RSHIFT: // Right Shift
177 return "Shift"_s;
178
179 case VK_LMENU: // Left Alt
180 case VK_RMENU: // Right Alt
181 return "Alt"_s;
182
183 case VK_LCONTROL: // Left Ctrl
184 case VK_RCONTROL: // Right Ctrl
185 return "Control"_s;
186
187 default:
188 ASSERT_NOT_REACHED();
189 return "Unidentified"_s;
190 }
191 }
192
193 // If more than one key is being pressed and the key combination includes one or more modifier keys
194 // that result in the key no longer producing a printable character (e.g., Control + a), then the
195 // key value should be the printable key value that would have been produced if the key had been
196 // typed with the default keyboard layout with no modifier keys except for Shift and AltGr applied.
197 // See <https://www.w3.org/TR/2015/WD-uievents-20151215/#keys-guidelines>.
198 bool isControlDown = event.modifierFlags & WebEventFlagMaskControlKey;
199 NSString *characters = isControlDown ? event.charactersIgnoringModifiers : event.characters;
200 auto length = [characters length];
201 // characters return an empty string for dead keys.
202 // https://developer.apple.com/reference/appkit/nsevent/1534183-characters
203 // "Dead" is defined here https://w3c.github.io/uievents-key/#keys-composition.
204 if (!length)
205 return "Dead"_s;
206 if (length > 1)
207 return characters;
208 return keyForCharCode([characters characterAtIndex:0]);
209}

423static bool isKeypadEvent(WebEvent* event)
424{
425 // Check that this is the type of event that has a keyCode.
426 if (event.type != WebEventKeyDown && event.type != WebEventKeyUp)
427 return false;
428
429 switch (event.keyCode) {
430 case VK_CLEAR: // Num Pad Clear
431 case VK_OEM_PLUS: // Num Pad =
432 case VK_DIVIDE:
433 case VK_MULTIPLY:
434 case VK_SUBTRACT:
435 case VK_ADD:
436 case VK_RETURN: // Num Pad Enter
437 case VK_DECIMAL: // Num Pad .
438 case VK_SEPARATOR: // Num Pad , (on JIS keyboard)
439 case VK_NUMPAD0:
440 case VK_NUMPAD1:
441 case VK_NUMPAD2:
442 case VK_NUMPAD3:
443 case VK_NUMPAD4:
444 case VK_NUMPAD5:
445 case VK_NUMPAD6:
446 case VK_NUMPAD7:
447 case VK_NUMPAD8:
448 case VK_NUMPAD9:
449 return true;
450 }
451 return false;
452}

454int windowsKeyCodeForKeyEvent(WebEvent* event)
455{
456 if (event.keyboardFlags & WebEventKeyboardInputModifierFlagsChanged)
457 return event.keyCode;
458
459 // There are several kinds of characters for which we produce key code from char code:
460 // 1. Roman letters. Windows keyboard layouts affect both virtual key codes and character codes for these,
461 // so e.g. 'A' gets the same keyCode on QWERTY, AZERTY or Dvorak layouts.
462 // 2. Keys for which there is no known iOS virtual key codes, like PrintScreen.
463 // 3. Certain punctuation keys. On Windows, these are also remapped depending on current keyboard layout,
464 // but see comment in windowsKeyCodeForCharCode().
465 if (!isKeypadEvent(event) && (event.type == WebEventKeyDown || event.type == WebEventKeyUp)) {
466 // Cmd switches Roman letters for Dvorak-QWERTY layout, so try modified characters first.
467 NSString *string = event.characters;
468 int code = string.length > 0 ? windowsKeyCodeForCharCode([string characterAtIndex:0]) : 0;
469 if (code)
470 return code;
471
472 // Ctrl+A on an AZERTY keyboard would get VK_Q keyCode if we relied on -[WebEvent keyCode] below.
473 string = event.charactersIgnoringModifiers;
474 code = string.length > 0 ? windowsKeyCodeForCharCode([string characterAtIndex:0]) : 0;
475 if (code)
476 return code;
477 }
478
479 // Use iOS virtual key code directly for any keys not handled above.
480 // E.g. the key next to Caps Lock has the same Event.keyCode on U.S. keyboard ('A') and on
481 // Russian keyboard (CYRILLIC LETTER EF).
482 return event.keyCode;
483}

485class PlatformKeyboardEventBuilder : public PlatformKeyboardEvent {
486public:
487 PlatformKeyboardEventBuilder(WebEvent *event)
488 {
489 ASSERT(event.type == WebEventKeyDown || event.type == WebEventKeyUp);
490
491 m_type = (event.type == WebEventKeyUp ? PlatformEvent::KeyUp : PlatformEvent::KeyDown);
492 m_modifiers = modifiersForEvent(event);
493 m_timestamp = WallTime::now();
494
495 if (event.keyboardFlags & WebEventKeyboardInputModifierFlagsChanged) {
496 m_text = emptyString();
497 m_unmodifiedText = emptyString();
498 m_autoRepeat = false;
499 } else {
500 m_text = event.characters;
501 m_unmodifiedText = event.charactersIgnoringModifiers;
502 m_autoRepeat = event.isKeyRepeating;
503 }
504 m_key = keyForKeyEvent(event);
505 m_code = codeForKeyEvent(event);
506 m_keyIdentifier = keyIdentifierForKeyEvent(event);
507 m_windowsVirtualKeyCode = windowsKeyCodeForKeyEvent(event);
508 m_isKeypad = false; // iOS does not distinguish the numpad. See <rdar://problem/7190835>.
509 m_isSystemKey = false;
510 m_Event = event;
511
512 // Always use 13 for Enter/Return -- we don't want to use AppKit's different character for Enter.
513 if (m_windowsVirtualKeyCode == '\r') {
514 m_text = "\r";
515 m_unmodifiedText = "\r";
516 }
517
518 // The adjustments below are only needed in backward compatibility mode, but we cannot tell what mode we are in from here.
519
520 // Turn 0x7F into 8, because backspace needs to always be 8.
521 if (m_text == "\x7F")
522 m_text = "\x8";
523 if (m_unmodifiedText == "\x7F")
524 m_unmodifiedText = "\x8";
525 // Always use 9 for tab -- we don't want to use AppKit's different character for shift-tab.
526 if (m_windowsVirtualKeyCode == 9) {
527 m_text = "\x9";
528 m_unmodifiedText = "\x9";
529 }
530 }
531};

540{
541 WebEventTouchPhaseType touchPhase = static_cast<WebEventTouchPhaseType>([touchPhaseNumber unsignedIntValue]);
542 switch (touchPhase) {
543 case WebEventTouchPhaseBegan:
544 return PlatformTouchPoint::TouchPhaseBegan;

551 case WebEventTouchPhaseCancelled:
552 return PlatformTouchPoint::TouchPhaseCancelled;

555 }
556 return PlatformTouchPoint::TouchPhaseBegan;

599class PlatformTouchEventBuilder : public PlatformTouchEvent {
600public:
601 PlatformTouchEventBuilder(WebEvent *event)
602 {
603 m_type = touchEventType(event);
604 m_modifiers = modifiersForEvent(event);
605 m_timestamp = WallTime::fromRawSeconds(event.timestamp);
606
607 m_gestureScale = event.gestureScale;
608 m_gestureRotation = event.gestureRotation;
609 m_isGesture = event.isGesture;
610 m_position = pointForEvent(event);
611 m_globalPosition = globalPointForEvent(event);
612
613 unsigned touchCount = event.touchCount;
614 m_touchPoints.reserveInitialCapacity(touchCount);
615 for (unsigned i = 0; i < touchCount; ++i) {
616 unsigned identifier = [(NSNumber *)[event.touchIdentifiers objectAtIndex:i] unsignedIntValue];
617 IntPoint location = IntPoint([(NSValue *)[event.touchLocations objectAtIndex:i] pointValue]);
618 PlatformTouchPoint::TouchPhaseType touchPhase = convertTouchPhase([event.touchPhases objectAtIndex:i]);
619 m_touchPoints.uncheckedAppend(PlatformTouchPointBuilder(identifier, location, touchPhase));
620 }
621 }
622
623 PlatformTouchEventBuilder(PlatformEvent::Type type, IntPoint location)
624 {
625 m_type = type;
626 m_timestamp = WallTime::now();
627
628 m_gestureScale = 1;
629 m_gestureRotation = 0;
630 m_isGesture = 0;
631 m_position = location;
632 m_globalPosition = location;
633 m_isPotentialTap = true;
634
635 unsigned touchCount = 1;
636 m_touchPoints.reserveInitialCapacity(touchCount);
637 for (unsigned i = 0; i < touchCount; ++i)
638 m_touchPoints.uncheckedAppend(PlatformTouchPointBuilder(1, location, touchPhaseFromPlatformEventType(type)));
639 }
640};
