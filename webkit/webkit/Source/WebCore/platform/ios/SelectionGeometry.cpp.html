63void SelectionGeometry::setLogicalLeft(int left)
64{
65 auto rect = this->rect();
66 if (m_isHorizontal)
67 rect.setX(left);
68 else
69 rect.setY(left);
70 setRect(rect);
71}

74{
75 auto rect = this->rect();

103IntRect SelectionGeometry::rect() const
104{
105 if (!m_cachedEnclosingRect)
106 m_cachedEnclosingRect = m_quad.enclosingBoundingBox();
107 return *m_cachedEnclosingRect;
108}
109
110void SelectionGeometry::setQuad(const FloatQuad& quad)
111{
112 m_quad = quad;
113 m_cachedEnclosingRect.reset();
114}
115
116void SelectionGeometry::setRect(const IntRect& rect)
117{
118 m_quad = FloatQuad { rect };
119 m_cachedEnclosingRect = rect;
120}

122TextStream& operator<<(TextStream& stream, SelectionGeometry rect)
123{
124 TextStream::GroupScope group(stream);
125 stream << "selection geometry";
126
127 stream.dumpProperty("quad", rect.quad());
128 stream.dumpProperty("direction", isLeftToRightDirection(rect.direction()) ? "ltr" : "rtl");
129
130 stream.dumpProperty("min-x", rect.minX());
131 stream.dumpProperty("max-x", rect.maxX());
132 stream.dumpProperty("max-y", rect.maxY());
133
134 stream.dumpProperty("line number", rect.lineNumber());
135 if (rect.isLineBreak())
136 stream.dumpProperty("is line break", true);
137 if (rect.isFirstOnLine())
138 stream.dumpProperty("is first on line", true);
139 if (rect.isLastOnLine())
140 stream.dumpProperty("is last on line", true);
141
142 if (rect.containsStart())
143 stream.dumpProperty("contains start", true);
144
145 if (rect.containsEnd())
146 stream.dumpProperty("contains end", true);
147
148 if (rect.isHorizontal())
149 stream.dumpProperty("is horizontal", true);
150
151 if (rect.isInFixedPosition())
152 stream.dumpProperty("is in fixed position", true);
153
154 if (rect.isRubyText())
155 stream.dumpProperty("is ruby text", true);
156
157 if (rect.behavior() == SelectionRenderingBehavior::UseIndividualQuads)
158 stream.dumpProperty("using individual quads", true);
159
160 stream.dumpProperty("page number", rect.pageNumber());
161 return stream;
162}
