26#pragma once
27
28#include "FloatQuad.h"
29#include "IntRect.h"
30#include "WritingMode.h"
31#include <optional>
32#include <wtf/FastMalloc.h>
33
34namespace WebCore {
35
36enum class SelectionRenderingBehavior : bool { CoalesceBoundingRects, UseIndividualQuads };
37
38class SelectionGeometry {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 WEBCORE_EXPORT explicit SelectionGeometry(const FloatQuad&, SelectionRenderingBehavior, bool isHorizontal, int columnNumber);
42
43 // FIXME: We should move some of these arguments to an auxillary struct.
44 SelectionGeometry(const FloatQuad&, SelectionRenderingBehavior, TextDirection, int, int, int, int, bool, bool, bool, bool, bool, bool, bool, bool, int);
45 SelectionGeometry() = default;
46 ~SelectionGeometry() = default;
47
48 FloatQuad quad() const { return m_quad; }
49 WEBCORE_EXPORT void setQuad(const FloatQuad&);
50
51 WEBCORE_EXPORT IntRect rect() const;
52 WEBCORE_EXPORT void setRect(const IntRect&);
53
54 int logicalLeft() const { return m_isHorizontal ? rect().x() : rect().y(); }
55 int logicalWidth() const { return m_isHorizontal ? rect().width() : rect().height(); }
56 int logicalTop() const { return m_isHorizontal ? rect().y() : rect().x(); }
57 int logicalHeight() const { return m_isHorizontal ? rect().height() : rect().width(); }
58
59 TextDirection direction() const { return m_direction; }
60 int minX() const { return m_minX; }
61 int maxX() const { return m_maxX; }
62 int maxY() const { return m_maxY; }
63 int lineNumber() const { return m_lineNumber; }
64 bool isLineBreak() const { return m_isLineBreak; }
65 bool isFirstOnLine() const { return m_isFirstOnLine; }
66 bool isLastOnLine() const { return m_isLastOnLine; }
67 bool containsStart() const { return m_containsStart; }
68 bool containsEnd() const { return m_containsEnd; }
69 bool isHorizontal() const { return m_isHorizontal; }
70 bool isInFixedPosition() const { return m_isInFixedPosition; }
71 bool isRubyText() const { return m_isRubyText; }
72 int pageNumber() const { return m_pageNumber; }
73 SelectionRenderingBehavior behavior() const { return m_behavior; }
74
75 void setLogicalLeft(int);
76 void setLogicalWidth(int);
77 void setLogicalTop(int);
78 void setLogicalHeight(int);
79
80 void setDirection(TextDirection direction) { m_direction = direction; }
81 void setMinX(int minX) { m_minX = minX; }
82 void setMaxX(int maxX) { m_maxX = maxX; }
83 void setMaxY(int maxY) { m_maxY = maxY; }
84 void setLineNumber(int lineNumber) { m_lineNumber = lineNumber; }
85 void setIsLineBreak(bool isLineBreak) { m_isLineBreak = isLineBreak; }
86 void setIsFirstOnLine(bool isFirstOnLine) { m_isFirstOnLine = isFirstOnLine; }
87 void setIsLastOnLine(bool isLastOnLine) { m_isLastOnLine = isLastOnLine; }
88 void setContainsStart(bool containsStart) { m_containsStart = containsStart; }
89 void setContainsEnd(bool containsEnd) { m_containsEnd = containsEnd; }
90 void setIsHorizontal(bool isHorizontal) { m_isHorizontal = isHorizontal; }
91 void setBehavior(SelectionRenderingBehavior behavior) { m_behavior = behavior; }
92
93private:
94 FloatQuad m_quad;
95 SelectionRenderingBehavior m_behavior { SelectionRenderingBehavior::CoalesceBoundingRects };
96 TextDirection m_direction { TextDirection::LTR };
97 int m_minX { 0 };
98 int m_maxX { 0 };
99 int m_maxY { 0 };
100 int m_lineNumber { 0 };
101 bool m_isLineBreak { false };
102 bool m_isFirstOnLine { false };
103 bool m_isLastOnLine { false };
104 bool m_containsStart { false };
105 bool m_containsEnd { false };
106 bool m_isHorizontal { true };
107 bool m_isInFixedPosition { false };
108 bool m_isRubyText { false };
109 int m_pageNumber { 0 };
110
111 mutable std::optional<IntRect> m_cachedEnclosingRect;
112};
113
114WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, SelectionGeometry);
115
116} // namespace WebCore

124 WebCore::SelectionRenderingBehavior::UseIndividualQuads
125 >;
126};
