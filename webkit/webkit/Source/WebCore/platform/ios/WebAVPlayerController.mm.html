66{

71 return self;
72
73 initAVPlayerController();
74 self.playerControllerProxy = adoptNS([allocAVPlayerControllerInstance() init]).get();
75 _liveStreamEventModePossible = YES;

77 [self addObserver:self forKeyPath:@"seekableTimeRanges" options:(NSKeyValueObservingOptionOld | NSKeyValueObservingOptionNew | NSKeyValueObservingOptionInitial) context:WebAVPlayerControllerSeekableTimeRangesObserverContext];
78 [self addObserver:self forKeyPath:@"hasLiveStreamingContent" options:NSKeyValueObservingOptionInitial context:WebAVPlayerControllerHasLiveStreamingContentObserverContext];

185- (void)setRate:(double)rate
186{
187 if (rate == _rate)
188 return;
189
190 _rate = rate;
191
192 // AVKit doesn't have a separate variable for "paused", instead representing it by a `rate` of
193 // `0`. Unfortunately, `HTMLMediaElement::play` doesn't call `HTMLMediaElement::setPlaybackRate`
194 // so if we propagate a `rate` of `0` along to the `HTMLMediaElement` then any attempt to
195 // `HTMLMediaElement::play` will effectively be a no-op since the `playbackRate` will be `0`.
196 if (!_rate)
197 return;
198
199 // In AVKit, the `defaultPlaybackRate` is used when playback starts, such as resuming after
200 // pausing. In WebKit, however, `defaultPlaybackRate` is only used when first loading and after
201 // ending scanning, with the `playbackRate` being used in all other cases, including when
202 // resuming after pausing. As such, WebKit should return the `playbackRate` instead of the
203 // `defaultPlaybackRate` in these cases when communicating with AVKit.
204 [self setDefaultPlaybackRate:_rate];
205
206 if (self.delegate && self.delegate->playbackRate() != _rate)
207 self.delegate->setPlaybackRate(_rate);
208}

314- (void)skipBackwardThirtySeconds:(id)sender
315{
316 using namespace PAL;
317
318 UNUSED_PARAM(sender);
319 BOOL isTimeWithinSeekableTimeRanges = NO;
320 CMTime currentTime = CMTimeMakeWithSeconds([[self timing] currentValue], 1000);
321 CMTime thirtySecondsBeforeCurrentTime = CMTimeSubtract(currentTime, PAL::CMTimeMake(30, 1));
322
323 for (NSValue *seekableTimeRangeValue in [self seekableTimeRanges]) {
324 if (CMTimeRangeContainsTime([seekableTimeRangeValue CMTimeRangeValue], thirtySecondsBeforeCurrentTime)) {
325 isTimeWithinSeekableTimeRanges = YES;
326 break;
327 }
328 }
329
330 if (isTimeWithinSeekableTimeRanges)
331 [self seekToTime:CMTimeGetSeconds(thirtySecondsBeforeCurrentTime)];
332}

334- (void)gotoEndOfSeekableRanges:(id)sender
335{
336 using namespace PAL;
337
338 UNUSED_PARAM(sender);
339 NSTimeInterval timeAtEndOfSeekableTimeRanges = NAN;
340
341 for (NSValue *seekableTimeRangeValue in [self seekableTimeRanges]) {
342 CMTimeRange seekableTimeRange = [seekableTimeRangeValue CMTimeRangeValue];
343 NSTimeInterval endOfSeekableTimeRange = CMTimeGetSeconds(CMTimeRangeGetEnd(seekableTimeRange));
344 if (isnan(timeAtEndOfSeekableTimeRanges) || endOfSeekableTimeRange > timeAtEndOfSeekableTimeRanges)
345 timeAtEndOfSeekableTimeRanges = endOfSeekableTimeRange;
346 }
347
348 if (!isnan(timeAtEndOfSeekableTimeRanges))
349 [self seekToTime:timeAtEndOfSeekableTimeRanges];
350}

622- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary *)change context:(void *)context
623{
624 using namespace PAL;
625
626 UNUSED_PARAM(object);
627 UNUSED_PARAM(keyPath);
628
629 if (WebAVPlayerControllerSeekableTimeRangesObserverContext == context) {
630 NSArray *oldArray = change[NSKeyValueChangeOldKey];
631 NSArray *newArray = change[NSKeyValueChangeNewKey];
632 if ([oldArray isKindOfClass:[NSArray class]] && [newArray isKindOfClass:[NSArray class]]) {
633 CMTimeRange oldSeekableTimeRange = [oldArray count] > 0 ? [[oldArray firstObject] CMTimeRangeValue] : kCMTimeRangeInvalid;
634 CMTimeRange newSeekableTimeRange = [newArray count] > 0 ? [[newArray firstObject] CMTimeRangeValue] : kCMTimeRangeInvalid;
635 if (!CMTimeRangeEqual(oldSeekableTimeRange, newSeekableTimeRange)) {
636 if (CMTIMERANGE_IS_VALID(newSeekableTimeRange) && CMTIMERANGE_IS_VALID(oldSeekableTimeRange) && _liveStreamEventModePossible && !CMTimeRangeContainsTime(newSeekableTimeRange, oldSeekableTimeRange.start))
637 _liveStreamEventModePossible = NO;
638
639 [self updateMinMaxTiming];
640 }
641 }
642 return;
643 }
644
645 if (WebAVPlayerControllerHasLiveStreamingContentObserverContext == context) {
646 [self updateMinMaxTiming];
647 return;
648 }
649
650 if (WebAVPlayerControllerIsPlayingOnSecondScreenObserverContext == context) {
651 if (auto* delegate = self.delegate)
652 delegate->setPlayingOnSecondScreen(_playingOnSecondScreen);
653 return;
654 }
655
656 ASSERT_NOT_REACHED();
657}

659- (void)updateMinMaxTiming
660{
661 using namespace PAL;
662
663 AVValueTiming *newMinTiming = nil;
664 AVValueTiming *newMaxTiming = nil;
665
666 // For live streams value timings for the min and max times are needed.
667 if ([self hasLiveStreamingContent] && ([[self seekableTimeRanges] count] > 0)) {
668 CMTimeRange seekableTimeRange = [[[self seekableTimeRanges] firstObject] CMTimeRangeValue];
669 if (CMTIMERANGE_IS_VALID(seekableTimeRange)) {
670 NSTimeInterval oldMinTime = [[self minTiming] currentValue];
671 NSTimeInterval oldMaxTime = [[self maxTiming] currentValue];
672 NSTimeInterval newMinTime = CMTimeGetSeconds(seekableTimeRange.start);
673 NSTimeInterval newMaxTime = CMTimeGetSeconds(seekableTimeRange.start) + CMTimeGetSeconds(seekableTimeRange.duration) + (CACurrentMediaTime() - [self seekableTimeRangesLastModifiedTime]);
674 double newMinTimingRate = _liveStreamEventModePossible ? 0.0 : 1.0;
675 BOOL minTimingNeedsUpdate = YES;
676 BOOL maxTimingNeedsUpdate = YES;
677
678 if (isfinite([self liveUpdateInterval]) && [self liveUpdateInterval] > WebAVPlayerControllerLiveStreamMinimumTargetDuration) {
679 // Only update the timing if the new time differs by one segment duration plus the hysteresis delta.
680 minTimingNeedsUpdate = isnan(oldMinTime) || (fabs(oldMinTime - newMinTime) > [self liveUpdateInterval] + WebAVPlayerControllerLiveStreamSeekableTimeRangeDurationHysteresisDelta) || ([[self minTiming] rate] != newMinTimingRate);
681 maxTimingNeedsUpdate = isnan(oldMaxTime) || (fabs(oldMaxTime - newMaxTime) > [self liveUpdateInterval] + WebAVPlayerControllerLiveStreamSeekableTimeRangeDurationHysteresisDelta);
682 }
683
684 if (minTimingNeedsUpdate || maxTimingNeedsUpdate) {
685 newMinTiming = [getAVValueTimingClass() valueTimingWithAnchorValue:newMinTime anchorTimeStamp:[getAVValueTimingClass() currentTimeStamp] rate:newMinTimingRate];
686 newMaxTiming = [getAVValueTimingClass() valueTimingWithAnchorValue:newMaxTime anchorTimeStamp:[getAVValueTimingClass() currentTimeStamp] rate:1.0];
687 } else {
688 newMinTiming = [self minTiming];
689 newMaxTiming = [self maxTiming];
690 }
691 }
692 }
693
694 if (!newMinTiming)
695 newMinTiming = [getAVValueTimingClass() valueTimingWithAnchorValue:[self minTime] anchorTimeStamp:NAN rate:0.0];
696
697 if (!newMaxTiming)
698 newMaxTiming = [getAVValueTimingClass() valueTimingWithAnchorValue:[self maxTime] anchorTimeStamp:NAN rate:0.0];
699
700 [self setMinTiming:newMinTiming];
701 [self setMaxTiming:newMaxTiming];
702}

704- (BOOL)hasSeekableLiveStreamingContent
705{
706 BOOL hasSeekableLiveStreamingContent = NO;
707
708 if ([self hasLiveStreamingContent] && [self minTiming] && [self maxTiming] && isfinite([self liveUpdateInterval]) && [self liveUpdateInterval] > WebAVPlayerControllerLiveStreamMinimumTargetDuration && ([self seekableTimeRangesLastModifiedTime] != 0.0)) {
709 NSTimeInterval timeStamp = [getAVValueTimingClass() currentTimeStamp];
710 NSTimeInterval minTime = [[self minTiming] valueForTimeStamp:timeStamp];
711 NSTimeInterval maxTime = [[self maxTiming] valueForTimeStamp:timeStamp];
712 hasSeekableLiveStreamingContent = ((maxTime - minTime) > WebAVPlayerControllerLiveStreamSeekableTimeRangeMinimumDuration);
713 }
714
715 return hasSeekableLiveStreamingContent;
716}
