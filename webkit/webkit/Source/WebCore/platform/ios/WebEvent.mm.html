120static int windowsKeyCodeForCharCodeIOS(unichar charCode)
121{
122 // iPhone Specific Cases
123 // <rdar://7709408>: We get 10 ('\n') from UIKit when using the software keyboard
124 if (charCode == 10)
125 return 0x0D;
126
127 // General Case
128 return windowsKeyCodeForCharCode(charCode);
129}

131static NSString *normalizedStringWithAppKitCompatibilityMapping(NSString *characters, uint16_t keyCode)
132{
133 auto makeNSStringWithCharacter = [] (unichar c) { return [NSString stringWithCharacters:&c length:1]; };
134
135 switch (keyCode) {
136 case kHIDUsage_KeyboardUpArrow:
137 return makeNSStringWithCharacter(NSUpArrowFunctionKey);
138 case kHIDUsage_KeyboardDownArrow:
139 return makeNSStringWithCharacter(NSDownArrowFunctionKey);
140 case kHIDUsage_KeyboardLeftArrow:
141 return makeNSStringWithCharacter(NSLeftArrowFunctionKey);
142 case kHIDUsage_KeyboardRightArrow:
143 return makeNSStringWithCharacter(NSRightArrowFunctionKey);
144 case kHIDUsage_KeyboardPageUp:
145 return makeNSStringWithCharacter(NSPageUpFunctionKey);
146 case kHIDUsage_KeyboardPageDown:
147 return makeNSStringWithCharacter(NSPageDownFunctionKey);
148 case kHIDUsage_KeyboardEscape:
149 return @"\x1B";
150 case kHIDUsage_KeypadNumLock: // Num Lock / Clear
151 return makeNSStringWithCharacter(NSClearLineFunctionKey);
152 case kHIDUsage_KeyboardDeleteForward:
153 return makeNSStringWithCharacter(NSDeleteFunctionKey);
154 case kHIDUsage_KeyboardEnd:
155 return makeNSStringWithCharacter(NSEndFunctionKey);
156 case kHIDUsage_KeyboardInsert:
157 return makeNSStringWithCharacter(NSInsertFunctionKey);
158 case kHIDUsage_KeyboardHome:
159 return makeNSStringWithCharacter(NSHomeFunctionKey);
160 }
161 if (keyCode >= kHIDUsage_KeyboardF1 && keyCode <= kHIDUsage_KeyboardF12)
162 return makeNSStringWithCharacter(NSF1FunctionKey + (keyCode - kHIDUsage_KeyboardF1));
163 if (keyCode >= kHIDUsage_KeyboardF13 && keyCode <= kHIDUsage_KeyboardF24)
164 return makeNSStringWithCharacter(NSF13FunctionKey + (keyCode - kHIDUsage_KeyboardF13));
165 return characters;
166}

168- (WebEvent *)initWithKeyEventType:(WebEventType)type
169 timeStamp:(CFTimeInterval)timeStamp
170 characters:(NSString *)characters
171 charactersIgnoringModifiers:(NSString *)charactersIgnoringModifiers
172 modifiers:(WebEventFlags)modifiers
173 isRepeating:(BOOL)repeating
174 withFlags:(NSUInteger)flags
175 withInputManagerHint:(NSString *)hint
176 keyCode:(uint16_t)keyCode
177 isTabKey:(BOOL)tabKey
178{
179 self = [super init];
180 if (!self)
181 return nil;
182
183 _type = type;
184 _timestamp = timeStamp;
185 _modifierFlags = modifiers;
186 _keyboardFlags = flags;
187 _inputManagerHint = [hint retain];
188
189 BOOL flagsChanged = _keyboardFlags & WebEventKeyboardInputModifierFlagsChanged;
190 if (!flagsChanged) {
191 // Map Command + . to Escape since Apple Smart Keyboards lack an Escape key.
192 // FIXME: This doesn't work for some keyboard layouts, like French. See <rdar://problem/51047011>.
193 if ([charactersIgnoringModifiers isEqualToString:@"."] && (modifiers & WebEventFlagMaskCommandKey)) {
194 keyCode = kHIDUsage_KeyboardEscape;
195 _modifierFlags &= ~WebEventFlagMaskCommandKey;
196 }
197 }
198
199 if (keyCode)
200 _keyCode = windowsKeyCodeForKeyCode(keyCode);
201 else if ([charactersIgnoringModifiers length] == 1) {
202 // This event is likely for a software keyboard-generated event.
203 _keyCode = windowsKeyCodeForCharCodeIOS([charactersIgnoringModifiers characterAtIndex:0]);
204 }
205
206 if (!flagsChanged) {
207 _characters = [normalizedStringWithAppKitCompatibilityMapping(characters, keyCode) retain];
208 _charactersIgnoringModifiers = [normalizedStringWithAppKitCompatibilityMapping(charactersIgnoringModifiers, keyCode) retain];
209 _tabKey = tabKey;
210 _keyRepeating = repeating;
211 }
212
213 return self;
214}

287- (NSString *)_touchLocationsDescription:(NSArray *)locations
288{
289 BOOL shouldAddComma = NO;
290 NSMutableString *description = [NSMutableString string];
291 for (NSValue *value in locations) {
292 CGPoint point = [value pointValue];
293 [description appendFormat:@"%@(%f, %f)", (shouldAddComma ? @", " : @""), point.x, point.y];
294 shouldAddComma = YES;
295 }
296 return description;
297}

340- (NSString *)_eventDescription
341{
342 switch (_type) {
343 case WebEventMouseDown:
344 case WebEventMouseUp:
345 case WebEventMouseMoved:
346 return [NSString stringWithFormat:@"location: (%f, %f)", _locationInWindow.x, _locationInWindow.y];
347 case WebEventScrollWheel:
348 return [NSString stringWithFormat:@"location: (%f, %f) deltaX: %f deltaY: %f", _locationInWindow.x, _locationInWindow.y, _deltaX, _deltaY];
349 case WebEventKeyDown:
350 case WebEventKeyUp:
351 if (_keyboardFlags & WebEventKeyboardInputModifierFlagsChanged)
352 return [NSString stringWithFormat:@"flags: %d keyboardFlags: %lu keyCode %d", _modifierFlags, static_cast<unsigned long>(_keyboardFlags), _keyCode];
353 return [NSString stringWithFormat:@"chars: %@ charsNoModifiers: %@ flags: %d repeating: %d keyboardFlags: %lu keyCode %d, isTab: %d", _characters, _charactersIgnoringModifiers, _modifierFlags, _keyRepeating, static_cast<unsigned long>(_keyboardFlags), _keyCode, _tabKey];
354 case WebEventTouchBegin:
355 case WebEventTouchChange:
356 case WebEventTouchEnd:
357 case WebEventTouchCancel:
358 return [NSString stringWithFormat:@"location: (%f, %f) count: %d locations: %@ identifiers: %@ phases: %@ isGesture: %d scale: %f rotation: %f", _locationInWindow.x, _locationInWindow.y, _touchCount, [self _touchLocationsDescription:_touchLocations], [self _touchIdentifiersDescription], [self _touchPhasesDescription], (_isGesture ? 1 : 0), _gestureScale, _gestureRotation];
359 default:
360 ASSERT_NOT_REACHED();
361 }
362 return @"Unknown";
363}

371{
372 ASSERT_WITH_MESSAGE(_type == WebEventMouseDown || _type == WebEventMouseUp || _type == WebEventMouseMoved || _type == WebEventScrollWheel
373 // FIXME: <rdar://problem/7185284> TouchEvents may be in more than one window some day.
374 || _type == WebEventTouchBegin || _type == WebEventTouchChange || _type == WebEventTouchEnd || _type == WebEventTouchCancel
375 , "WebEventType: %d", _type);
376 return _locationInWindow;
