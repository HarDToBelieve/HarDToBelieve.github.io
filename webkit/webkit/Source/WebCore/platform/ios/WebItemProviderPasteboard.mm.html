65- (BOOL)web_containsFileURLAndFileUploadContent
66{
67 for (NSString *identifier in self.registeredTypeIdentifiers) {
68ALLOW_DEPRECATED_DECLARATIONS_BEGIN
69 if (UTTypeConformsTo((__bridge CFStringRef)identifier, kUTTypeFileURL))
70 return self.web_fileUploadContentTypes.count;
71ALLOW_DEPRECATED_DECLARATIONS_END
72 }
73 return NO;
74}
75
76- (NSArray<NSString *> *)web_fileUploadContentTypes
77{
78 auto types = adoptNS([NSMutableArray new]);
79 for (NSString *identifier in self.registeredTypeIdentifiers) {
80ALLOW_DEPRECATED_DECLARATIONS_BEGIN
81 if (UTTypeConformsTo((__bridge CFStringRef)identifier, kUTTypeURL))
82 continue;
83ALLOW_DEPRECATED_DECLARATIONS_END
84
85 if ([identifier isEqualToString:@"com.apple.mapkit.map-item"]) {
86 // This type conforms to "public.content", yet the corresponding data is only a serialization of MKMapItem and isn't suitable for file uploads.
87 // Ignore over this type representation for the purposes of file uploads, in favor of "public.vcard" data.
88 continue;
89 }
90
91 if (typeConformsToTypes(identifier, Pasteboard::supportedFileUploadPasteboardTypes()))
92 [types addObject:identifier];
93 }
94
95 return types.autorelease();
96}

543- (NSData *)_preLoadedDataConformingToType:(NSString *)typeIdentifier forItemProviderAtIndex:(NSUInteger)index
544{
545 if (_loadResults.size() != [_itemProviders count]) {
546 ASSERT_NOT_REACHED();
547 return nil;
548 }
549
550 WebItemProviderLoadResult *loadResult = _loadResults[index].get();
551 for (NSString *loadedType in loadResult.loadedTypeIdentifiers) {
552ALLOW_DEPRECATED_DECLARATIONS_BEGIN
553 if (!UTTypeConformsTo((CFStringRef)loadedType, (CFStringRef)typeIdentifier))
554 continue;
555ALLOW_DEPRECATED_DECLARATIONS_END
556
557 // We've already loaded data relevant for this UTI type onto disk, so there's no need to ask the NSItemProvider for the same data again.
558 if (NSData *result = [NSData dataWithContentsOfURL:[loadResult fileURLForType:loadedType] options:NSDataReadingMappedIfSafe error:nil])
559 return result;
560 }
561 return nil;
562}

592static Class classForTypeIdentifier(NSString *typeIdentifier, NSString *&outTypeIdentifierToLoad)
593{
594 outTypeIdentifierToLoad = typeIdentifier;
595
596 // First, try to load a platform NSItemProviderReading-conformant object as-is.
597 for (Class<NSItemProviderReading> loadableClass in allLoadableClasses()) {
598 if ([[loadableClass readableTypeIdentifiersForItemProvider] containsObject:(NSString *)typeIdentifier])
599 return loadableClass;
600 }
601
602 // If we were unable to load any object, check if the given type identifier is still something
603 // WebKit knows how to handle.
604ALLOW_DEPRECATED_DECLARATIONS_BEGIN
605 if ([typeIdentifier isEqualToString:(NSString *)kUTTypeHTML]) {
606 // Load kUTTypeHTML as a plain text HTML string.
607 outTypeIdentifierToLoad = (NSString *)kUTTypePlainText;
608 return [NSString class];
609 }
610ALLOW_DEPRECATED_DECLARATIONS_END
611
612 return nil;
613}

615- (NSArray *)valuesForPasteboardType:(NSString *)pasteboardType inItemSet:(NSIndexSet *)itemSet
616{
617 if (_loadResults.isEmpty())
618 return @[ ];
619
620 auto values = adoptNS([[NSMutableArray alloc] init]);
621 RetainPtr<WebItemProviderPasteboard> retainedSelf = self;
622 [itemSet enumerateIndexesUsingBlock:[retainedSelf, pasteboardType, values] (NSUInteger index, BOOL *) {
623 NSItemProvider *provider = [retainedSelf itemProviderAtIndex:index];
624 if (!provider)
625 return;
626
627 NSString *typeIdentifierToLoad;
628 Class readableClass = classForTypeIdentifier(pasteboardType, typeIdentifierToLoad);
629 if (!readableClass)
630 return;
631
632 NSData *preloadedData = [retainedSelf _preLoadedDataConformingToType:pasteboardType forItemProviderAtIndex:index];
633 if (!preloadedData)
634 return;
635
636 if (id <NSItemProviderReading> readObject = [readableClass objectWithItemProviderData:preloadedData typeIdentifier:(NSString *)typeIdentifierToLoad error:nil])
637 [values addObject:readObject];
638 }];
639
640 return values.autorelease();
641}

702static NSURL *linkTemporaryItemProviderFilesToDropStagingDirectory(NSURL *url, NSString *suggestedName, NSString *typeIdentifier)
703{
704 static NSString *defaultDropFolderName = @"folder";
705 static NSString *defaultDropFileName = @"file";
706 static NSString *droppedDataDirectoryPrefix = @"dropped-data";
707 if (!url)
708 return nil;
709
710 NSString *temporaryDropDataDirectory = FileSystem::createTemporaryDirectory(droppedDataDirectoryPrefix);
711 if (!temporaryDropDataDirectory)
712 return nil;
713
714 NSURL *destination = nil;
715ALLOW_DEPRECATED_DECLARATIONS_BEGIN
716 BOOL isFolder = UTTypeConformsTo((CFStringRef)typeIdentifier, kUTTypeFolder);
717ALLOW_DEPRECATED_DECLARATIONS_END
718 NSFileManager *fileManager = [NSFileManager defaultManager];
719
720 if (!suggestedName)
721 suggestedName = url.lastPathComponent ?: (isFolder ? defaultDropFolderName : defaultDropFileName);
722
723 if ([suggestedName.pathExtension caseInsensitiveCompare:url.pathExtension] != NSOrderedSame && !isFolder)
724 suggestedName = [suggestedName stringByAppendingPathExtension:url.pathExtension];
725
726 destination = [NSURL fileURLWithPath:[temporaryDropDataDirectory stringByAppendingPathComponent:suggestedName]];
727 return [fileManager linkItemAtURL:url toURL:destination error:nil] ? destination : nil;
728}

732ALLOW_DEPRECATED_DECLARATIONS_BEGIN
733 auto typesToLoad = adoptNS([[NSMutableOrderedSet alloc] init]);
734 NSString *highestFidelitySupportedType = nil;
735 NSString *highestFidelityContentType = nil;
736
737 NSArray<NSString *> *registeredTypeIdentifiers = itemProvider.registeredTypeIdentifiers;
738 BOOL containsFile = itemProvider.web_containsFileURLAndFileUploadContent;
739 BOOL containsFlatRTFD = [registeredTypeIdentifiers containsObject:(__bridge NSString *)kUTTypeFlatRTFD];
740 // First, search for the highest fidelity supported type or the highest fidelity generic content type.
741 for (NSString *registeredTypeIdentifier in registeredTypeIdentifiers) {
742 if (containsFlatRTFD && [registeredTypeIdentifier isEqualToString:(__bridge NSString *)kUTTypeRTFD]) {
743 // In the case where attachments are enabled and we're accepting all types of content using attachment
744 // elements as a fallback representation, if the source writes attributed strings to the pasteboard with
745 // com.apple.rtfd at a higher fidelity than com.apple.flat-rtfd, we'll end up loading only com.apple.rtfd
746 // and dropping the text as an attachment element because we cannot convert the dropped content to markup.
747 // Instead, if flat RTFD is present in the item provider, always prefer that over RTFD so that dropping as
748 // regular web content isn't overridden by enabling attachment elements.
749 continue;
750 }
751
752 if (containsFile && UTTypeConformsTo((__bridge CFStringRef)registeredTypeIdentifier, kUTTypeURL))
753 continue;
754
755 if (!highestFidelitySupportedType && typeConformsToTypes(registeredTypeIdentifier, _supportedTypeIdentifiers.get()))
756 highestFidelitySupportedType = registeredTypeIdentifier;
757
758 if (!highestFidelityContentType && UTTypeConformsTo((CFStringRef)registeredTypeIdentifier, kUTTypeContent))
759 highestFidelityContentType = registeredTypeIdentifier;
760 }
761
762 // For compatibility with DataTransfer APIs, additionally load web-exposed types. Since this is the only chance to
763 // fault in any data at all, we need to load up front any information that the page may ask for later down the line.
764 NSString *customPasteboardDataUTI = @(PasteboardCustomData::cocoaType());
765 for (NSString *registeredTypeIdentifier in registeredTypeIdentifiers) {
766 if ([registeredTypeIdentifier isEqualToString:highestFidelityContentType]
767 || [registeredTypeIdentifier isEqualToString:highestFidelitySupportedType]
768 || [registeredTypeIdentifier isEqualToString:customPasteboardDataUTI]
769 || (!containsFile && UTTypeConformsTo((__bridge CFStringRef)registeredTypeIdentifier, kUTTypeURL))
770 || UTTypeConformsTo((__bridge CFStringRef)registeredTypeIdentifier, kUTTypeHTML)
771 || UTTypeConformsTo((__bridge CFStringRef)registeredTypeIdentifier, kUTTypePlainText))
772 [typesToLoad addObject:registeredTypeIdentifier];
773 }
774ALLOW_DEPRECATED_DECLARATIONS_END
775
776 return [typesToLoad array];
777}

784- (void)doAfterLoadingProvidedContentIntoFileURLs:(WebItemProviderFileLoadBlock)action synchronousTimeout:(NSTimeInterval)synchronousTimeout
785{
786 _loadResults.clear();
787
788 auto changeCountBeforeLoading = _changeCount;
789 auto loadResults = adoptNS([[NSMutableArray alloc] initWithCapacity:[_itemProviders count]]);
790
791 // First, figure out which item providers we want to try and load files from.
792 BOOL foundAnyDataToLoad = NO;
793 RetainPtr<WebItemProviderPasteboard> protectedSelf = self;
794 for (NSItemProvider *itemProvider in _itemProviders.get()) {
795 NSArray<NSString *> *typeIdentifiersToLoad = [protectedSelf typeIdentifiersToLoad:itemProvider];
796 foundAnyDataToLoad |= typeIdentifiersToLoad.count;
797 [loadResults addObject:[WebItemProviderLoadResult loadResultWithItemProvider:itemProvider typesToLoad:typeIdentifiersToLoad]];
798 }
799
800 if (!foundAnyDataToLoad) {
801 action(@[ ]);
802 return;
803 }
804
805 auto setFileURLsLock = adoptNS([[NSLock alloc] init]);
806 auto synchronousFileLoadingGroup = adoptOSObject(dispatch_group_create());
807 auto fileLoadingGroup = adoptOSObject(dispatch_group_create());
808 for (WebItemProviderLoadResult *loadResult in loadResults.get()) {
809 for (NSString *typeToLoad in loadResult.typesToLoad) {
810 dispatch_group_enter(fileLoadingGroup.get());
811 dispatch_group_enter(synchronousFileLoadingGroup.get());
812 [loadResult.itemProvider loadFileRepresentationForTypeIdentifier:typeToLoad completionHandler:[protectedLoadResult = retainPtr(loadResult), synchronousFileLoadingGroup, setFileURLsLock, protectedTypeToLoad = retainPtr(typeToLoad), fileLoadingGroup] (NSURL *url, NSError *) {
813 // After executing this completion block, UIKit removes the file at the given URL. However, we need this data to persist longer for the web content process.
814 // To address this, we hard link the given URL to a new temporary file in the temporary directory. This follows the same flow as regular file upload, in
815 // WKFileUploadPanel.mm. The temporary files are cleaned up by the system at a later time.
816 if (NSURL *destination = linkTemporaryItemProviderFilesToDropStagingDirectory(url, [protectedLoadResult itemProvider].suggestedName, protectedTypeToLoad.get())) {
817 [setFileURLsLock lock];
818 [protectedLoadResult setFileURL:destination forType:protectedTypeToLoad.get()];
819 [setFileURLsLock unlock];
820 }
821 dispatch_group_leave(fileLoadingGroup.get());
822 dispatch_group_leave(synchronousFileLoadingGroup.get());
823 }];
824 }
825 }
826
827 RetainPtr<WebItemProviderPasteboard> retainedSelf = self;
828 auto itemLoadCompletion = [retainedSelf, synchronousFileLoadingGroup, fileLoadingGroup, loadResults, completionBlock = makeBlockPtr(action), changeCountBeforeLoading] {
829 if (changeCountBeforeLoading == retainedSelf->_changeCount) {
830 for (WebItemProviderLoadResult *loadResult in loadResults.get())
831 retainedSelf->_loadResults.append(loadResult);
832 }
833
834 completionBlock([retainedSelf allDroppedFileURLs]);
835 };
836
837 if (synchronousTimeout > 0 && !dispatch_group_wait(synchronousFileLoadingGroup.get(), dispatch_time(DISPATCH_TIME_NOW, synchronousTimeout * NSEC_PER_SEC))) {
838 itemLoadCompletion();
839 return;
840 }
841
842 dispatch_group_notify(fileLoadingGroup.get(), dispatch_get_main_queue(), itemLoadCompletion);
843}
