85void Widget::show()
86{

88 return;
89
90 setSelfVisible(true);
91
92 BEGIN_BLOCK_OBJC_EXCEPTIONS
93 [getOuterView() setHidden:NO];
94 END_BLOCK_OBJC_EXCEPTIONS

121void Widget::setFrameRect(const IntRect &rect)
122{
123 m_frame = rect;
124
125 BEGIN_BLOCK_OBJC_EXCEPTIONS
126 NSView *v = getOuterView();
127 NSRect f = rect;
128 if (!NSEqualRects(f, [v frame])) {
129 [v setFrame:f];
130 [v setNeedsDisplay: NO];
131 }
132 END_BLOCK_OBJC_EXCEPTIONS
133}

135NSView* Widget::getOuterView() const
136{
137 NSView* view = platformWidget();
138
139 // If this widget's view is a WebCoreFrameScrollView then we
140 // resize its containing view, a WebFrameView.
141 if ([view conformsToProtocol:@protocol(WebCoreFrameScrollView)]) {
142 view = [view superview];
143 ASSERT(view);
144 }
145
146 return view;
147}

149void Widget::paint(GraphicsContext& p, const IntRect& r, SecurityOriginPaintPolicy, EventRegionContext*)
150{
151 if (p.paintingDisabled())
152 return;
153
154 NSView *view = getOuterView();
155
156 CGContextRef cgContext = p.platformContext();
157 CGContextSaveGState(cgContext);
158
159 NSRect viewFrame = [view frame];
160 NSRect viewBounds = [view bounds];
161 CGContextTranslateCTM(cgContext, viewFrame.origin.x - viewBounds.origin.x, viewFrame.origin.y - viewBounds.origin.y);
162
163 BEGIN_BLOCK_OBJC_EXCEPTIONS
164 [view displayRectIgnoringOpacity:[view convertRect:r fromView:[view superview]] inContext:cgContext];
165 END_BLOCK_OBJC_EXCEPTIONS
166
167 CGContextRestoreGState(cgContext);
168}

174void Widget::addToSuperview(NSView *view)
175{
176 BEGIN_BLOCK_OBJC_EXCEPTIONS
177
178 ASSERT(view);
179 NSView *subview = getOuterView();
180
181 if (!subview)
182 return;
183
184 ASSERT(![view isDescendantOf:subview]);
185
186 if ([subview superview] != view)
187 [view addSubview:subview];
188
189 END_BLOCK_OBJC_EXCEPTIONS
190}

192void Widget::removeFromSuperview()
193{
194 BEGIN_BLOCK_OBJC_EXCEPTIONS
195 safeRemoveFromSuperview(getOuterView());
196 END_BLOCK_OBJC_EXCEPTIONS
197}

199IntRect Widget::convertFromRootToContainingWindow(const Widget* rootWidget, const IntRect& rect)
200{
201 if (!rootWidget->platformWidget())
202 return rect;
203
204 BEGIN_BLOCK_OBJC_EXCEPTIONS
205 ASSERT([rootWidget->platformWidget() isKindOfClass:[NSScrollView class]]);
206 WAKScrollView *view = static_cast<WAKScrollView *>(rootWidget->platformWidget());
207 if (WAKView *documentView = [view documentView])
208 return enclosingIntRect([documentView convertRect:rect toView:nil]);
209 return enclosingIntRect([view convertRect:rect toView:nil]);
210 END_BLOCK_OBJC_EXCEPTIONS
211
212 return rect;
213}

231IntPoint Widget::convertFromRootToContainingWindow(const Widget* rootWidget, const IntPoint& point)
232{
233 if (!rootWidget->platformWidget())
234 return point;
235
236 BEGIN_BLOCK_OBJC_EXCEPTIONS
237 ASSERT([rootWidget->platformWidget() isKindOfClass:[NSScrollView class]]);
238 WAKScrollView *view = static_cast<WAKScrollView *>(rootWidget->platformWidget());
239 NSPoint convertedPoint;
240 if (WAKView *documentView = [view documentView])
241 convertedPoint = [documentView convertPoint:point toView:nil];
242 else
243 convertedPoint = [view convertPoint:point toView:nil];
244 return IntPoint(roundf(convertedPoint.x), roundf(convertedPoint.y));
245 END_BLOCK_OBJC_EXCEPTIONS
246
247 return point;
248}
