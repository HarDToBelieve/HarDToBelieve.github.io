77static NSInvocation* invocationForPostNotification(NSString *name, id object, id userInfo)
78{
79 NSNotificationCenter *target = [NSNotificationCenter defaultCenter];
80 NSInvocation *invocation = WebThreadMakeNSInvocation(target, @selector(postNotificationName:object:userInfo:));
81 [invocation setArgument:&name atIndex:2];
82 [invocation setArgument:&object atIndex:3];
83 [invocation setArgument:&userInfo atIndex:4];
84 return invocation;
85}

87static void notificationCallback (WKViewRef v, WKViewNotificationType type, void *userInfo)
88{
89 UNUSED_PARAM(v);
90 WAKView *view = (WAKView *)userInfo;
91 switch (type){
92 case WKViewNotificationViewDidMoveToWindow: {
93 [view viewDidMoveToWindow];
94 break;
95 }
96 case WKViewNotificationViewFrameSizeChanged: {
97 [view frameSizeChanged];
98 if (WAKScrollView *scrollView = enclosingScrollView(view))
99 [scrollView _adjustScrollers];
100
101 // Posting a notification to the main thread can cause the WebThreadLock to be
102 // relinquished, which gives the main thread a change to run (and possible try
103 // to paint). We don't want this to happen if we've updating view sizes in the middle
104 // of layout, so use an async notification. <rdar://problem/6745974>
105 NSInvocation *invocation = invocationForPostNotification(WAKViewFrameSizeDidChangeNotification, view, nil);
106 WebThreadCallDelegateAsync(invocation);
107 break;
108 }
109 case WKViewNotificationViewDidScroll: {
110 WebThreadRunOnMainThread(^ {
111 [[NSNotificationCenter defaultCenter] postNotificationName:WAKViewDidScrollNotification object:view userInfo:nil];
112 });
113 break;
114 }
115 default: {
116 break;
117 }
118 }
119}

121- (void)handleEvent:(WebEvent *)event
122{
123 ASSERT(event);
124 WAKView *view = self;
125 while (view) {
126 if ([view _selfHandleEvent:event])
127 break;
128 view = [view superview];
129 }

132- (BOOL)_selfHandleEvent:(WebEvent *)event
133{
134 switch (event.type) {
135 case WebEventMouseDown:
136 [self mouseDown:event];
137 return YES;
138 case WebEventMouseUp:
139 [self mouseUp:event];
140 return YES;
141 case WebEventMouseMoved:
142 [self mouseMoved:event];
143 return YES;
144 case WebEventKeyDown:
145 [self keyDown:event];
146 return YES;
147 case WebEventKeyUp:
148 [self keyUp:event];
149 return YES;
150 case WebEventScrollWheel:
151 [self scrollWheel:event];
152 return YES;
153 case WebEventTouchBegin:
154 case WebEventTouchChange:
155 case WebEventTouchEnd:
156 case WebEventTouchCancel:
157#if ENABLE(TOUCH_EVENTS)
158 [self touch:event];
159#endif
160 return YES;
161 }

223- (id)initWithFrame:(CGRect)rect
224{
225 WKViewRef view = WKViewCreateWithFrame(rect, &viewContext);
226 self = [self _initWithViewRef:view];
227 if (self) {
228 viewContext.notificationCallback = notificationCallback;
229 viewContext.notificationUserInfo = self;
230 viewContext.responderCallback = responderCallback;
231 viewContext.responderUserInfo = self;
232 viewContext.willRemoveSubviewCallback = willRemoveSubviewCallback;
233 viewContext.invalidateGStateCallback = invalidateGStateCallback;
234 }
235 WKRelease(view);
236 return self;
237}

271static void _WAKCopyWrapper(const void *value, void *context)
272{
273 if (!value)
274 return;
275 if (!context)
276 return;
277
278 NSMutableArray *array = (NSMutableArray *)context;
279 WAKView *view = WAKViewForWKViewRef(static_cast<WKViewRef>(const_cast<void*>(value)));
280 if (view)
281 [array addObject:view];
282}

284- (NSArray *)subviews
285{
286 CFArrayRef subviews = WKViewGetSubviews([self _viewRef]);
287 if (!subviews)
288 return @[ ];
289
290 CFIndex count = CFArrayGetCount(subviews);
291 if (count == 0)
292 return @[ ];
293
294 NSMutableArray *result = [NSMutableArray arrayWithCapacity:count];
295 if (!result)
296 return @[ ];
297
298 CFArrayApplyFunction(subviews, CFRangeMake(0, count), _WAKCopyWrapper, (void*)result);
299
300 return result;
301}

308- (WAKView *)lastScrollableAncestor
309{
310 WAKView *view = nil;
311 WAKScrollView *scrollView = enclosingScrollView(self);
312
313 while (scrollView) {
314
315 CGSize scrollViewSize = WKViewGetFrame((WKViewRef)scrollView).size;
316
317 WAKView *documentView = [scrollView documentView];
318 scrollView = enclosingScrollView(scrollView);
319
320 // If this document view can be scrolled, and this is NOT the last scroll view.
321 // Our last scroll view is non-existent as far as we're concerned.
322
323 if (scrollView && !CGSizeEqualToSize(scrollViewSize, [documentView frame].size))
324 view = documentView;
325 }
326
327 return view;
328}

450- (void)_lockFocusViewInContext:(CGContextRef)context
451{
452 ASSERT(context);
453 setGlobalFocusView(self);
454 CGContextSaveGState(context);
455
456 WAKView *superview = [self superview];
457 if (superview)
458 CGContextClipToRect(context, [superview bounds]);
459
460 CGContextConcatCTM(context, _WKViewGetTransform(viewRef));
461}

470static CGInterpolationQuality toCGInterpolationQuality(WebCore::InterpolationQuality quality)
471{
472 switch (quality) {
473 case WebCore::InterpolationQuality::Default:
474 return kCGInterpolationDefault;
475 case WebCore::InterpolationQuality::DoNotInterpolate:
476 return kCGInterpolationNone;
477 case WebCore::InterpolationQuality::Low:
478 return kCGInterpolationLow;
479 case WebCore::InterpolationQuality::Medium:
480 return kCGInterpolationMedium;

494- (void)_drawRect:(NSRect)dirtyRect context:(CGContextRef)context lockFocus:(BOOL)lockFocus
495{
496 if (_isHidden)
497 return;
498
499 ASSERT(viewRef);
500 ASSERT(context);
501 CGRect localRect = WKViewGetBounds(viewRef);
502 dirtyRect = CGRectIntersection(localRect, dirtyRect);
503 if (CGRectIsEmpty(dirtyRect))
504 return;
505
506 if (lockFocus)
507 [self _lockFocusViewInContext:context];
508
509 if (viewRef->scale != 1)
510 CGContextSetInterpolationQuality(context, sInterpolationQuality);
511
512 CGContextClipToRect(context, dirtyRect);
513 [self drawRect:dirtyRect];
514
515 if (!_drawsOwnDescendants) {
516 NSArray *subViews = [self subviews];
517 for (WAKView *subView in subViews) {
518 NSRect childDirtyRect = [self convertRect:dirtyRect toView:subView];
519 [subView _drawRect:CGRectIntegral(childDirtyRect) context:context lockFocus:YES];
520 }
521 }
522
523 if (lockFocus)
524 [self _unlockFocusViewInContext:context];
525}

610- (WAKView *)hitTest:(NSPoint)point
611{
612 if (!CGRectContainsPoint([self frame], point))
613 return nil;
614
615 CGPoint subviewPoint = WKViewConvertPointFromSuperview(viewRef, point);
616 for (WAKView *subview in [self subviews]) {
617 if (WAKView *hitView = [subview hitTest: subviewPoint])
618 return hitView;
619 }
620
621 return self;
622}

740- (NSString *)description
741{
742 NSMutableString *description = [NSMutableString stringWithFormat:@"<%@: WAK: %p (WK: %p); ", [self class], self, viewRef];
743
744 float scale = [self scale];
745 if (scale != 1)
746 [description appendFormat:@"scale = %g ", scale];
747
748 CGPoint origin = WKViewGetOrigin(viewRef);
749 if (origin.x || origin.y)
750 [description appendFormat:@"origin = (%g %g) ", origin.x, origin.y];
751
752 CGRect bounds = [self bounds];
753 CGRect frame = [self frame];
754
755 if (frame.origin.x != bounds.origin.x || frame.origin.y != bounds.origin.y || frame.size.width != bounds.size.width || frame.size.height != bounds.size.height)
756 [description appendFormat:@"bounds = (%g %g; %g %g) ", bounds.origin.x, bounds.origin.y, bounds.size.width, bounds.size.height];
757
758 [description appendFormat:@"frame = (%g %g; %g %g)>", frame.origin.x, frame.origin.y, frame.size.width, frame.size.height];
759
760 return description;
761}

763- (void)_appendDescriptionToString:(NSMutableString *)info atLevel:(int)level
764{
765 @autoreleasepool {
766 if ([info length])
767 [info appendString:@"\n"];
768
769 for (int i = 1; i <= level; i++)
770 [info appendString:@" | "];
771
772 [info appendString:[self description]];
773
774 for (WAKView *subview in [self subviews])
775 [subview _appendDescriptionToString:info atLevel:level + 1];
776 }
