200{
201 [invocation invoke];

204 return;

206 RunLoop::main().dispatch([invocation = WTFMove(invocation)] { });

209static void HandleDelegateSource(void*)
210{
211 ASSERT(!WebThreadIsCurrent());
212
213#if LOG_MAIN_THREAD_LOCKING
214 sendingDelegateMessage = YES;
215#endif
216
217 _WebThreadAutoLock();
218
219 {
220 Locker locker { delegateLock };
221
222#if LOG_MESSAGES
223 if ([[delegateInvocation() target] isKindOfClass:[NSNotificationCenter class]]) {
224 id argument0;
225 [delegateInvocation() getArgument:&argument0 atIndex:0];
226 NSLog(@"notification receive: %@", argument0);
227 } else
228 NSLog(@"delegate receive: %@", NSStringFromSelector([delegateInvocation() selector]));
229#endif
230
231 SendMessage(WTFMove(delegateInvocation()));
232
233 delegateHandled = YES;
234 delegateCondition.notifyOne();
235 }
236
237#if LOG_MAIN_THREAD_LOCKING
238 sendingDelegateMessage = NO;
239#endif
240}

252static void SendDelegateMessage(RetainPtr<NSInvocation>&& invocation)
253{
254 if (!WebThreadIsCurrent()) {
255 SendMessage(WTFMove(invocation));
256 return;
257 }
258
259 ASSERT(delegateSource());
260 delegateLock.lock();
261
262 delegateInvocation() = WTFMove(invocation);
263 delegateHandled = NO;
264
265#if LOG_MESSAGES
266 if ([[delegateInvocation() target] isKindOfClass:[NSNotificationCenter class]]) {
267 id argument0;
268 [delegateInvocation() getArgument:&argument0 atIndex:0];
269 NSLog(@"notification send: %@", argument0);
270 } else
271 NSLog(@"delegate send: %@", NSStringFromSelector([delegateInvocation() selector]));
272#endif
273
274 {
275 WebThreadDelegateMessageScope delegateScope;
276 // Code block created to scope JSC::JSLock::DropAllLocks outside of WebThreadLock()
277 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
278 _WebThreadUnlock();
279
280 CFRunLoopSourceSignal(delegateSource().get());
281 CFRunLoopWakeUp(CFRunLoopGetMain());
282
283 while (!delegateHandled) {
284 if (!delegateCondition.waitFor(delegateLock, DelegateWaitInterval)) {
285 id delegateInformation;
286 if ([[delegateInvocation() target] isKindOfClass:[NSNotificationCenter class]])
287 [delegateInvocation() getArgument:&delegateInformation atIndex:0];
288 else
289 delegateInformation = NSStringFromSelector([delegateInvocation() selector]);
290
291 auto mode = adoptCF(CFRunLoopCopyCurrentMode(CFRunLoopGetMain()));
292 NSLog(@"%s: delegate (%@) failed to return after waiting %f seconds. main run loop mode: %@", __PRETTY_FUNCTION__, delegateInformation, DelegateWaitInterval.seconds(), mode.get());
293 }
294 }
295 delegateLock.unlock();
296 _WebThreadLock();
297 }
298}

300void WebThreadRunOnMainThread(void(^delegateBlock)())
301{
302 if (!WebThreadIsCurrent()) {
303 ASSERT(pthread_main_np());
304 delegateBlock();
305 return;

309 _WebThreadUnlock();

311 WorkQueue::main().dispatchSync(makeBlockPtr(delegateBlock).get());
312
313 _WebThreadLock();
314}

334void WebCoreObjCDeallocOnWebThread(Class cls)
335{
336 SEL releaseSEL = @selector(release);
337 SEL webThreadReleaseSEL = @selector(_webcore_releaseOnWebThread);
338
339 // get the existing release method
340 Method releaseMethod = class_getInstanceMethod(cls, releaseSEL);
341 if (!releaseMethod) {
342 ASSERT_WITH_MESSAGE(releaseMethod, "WebCoreObjCDeallocOnWebThread() failed to find %s for %@", releaseSEL, NSStringFromClass(cls));
343 return;
344 }
345
346 // add the implementation that ensures release WebThread release/deallocation
347 if (!class_addMethod(cls, webThreadReleaseSEL, (IMP)WebCoreObjCDeallocOnWebThreadImpl, method_getTypeEncoding(releaseMethod))) {
348 ASSERT_WITH_MESSAGE(releaseMethod, "WebCoreObjCDeallocOnWebThread() failed to add %s for %@", webThreadReleaseSEL, NSStringFromClass(cls));
349 return;
350 }
351
352 // ensure the implementation exists at cls in the class hierarchy
353 if (class_addMethod(cls, releaseSEL, class_getMethodImplementation(cls, releaseSEL), method_getTypeEncoding(releaseMethod)))
354 releaseMethod = class_getInstanceMethod(cls, releaseSEL);
355
356 // swizzle the old release for the new implementation
357 method_exchangeImplementations(releaseMethod, class_getInstanceMethod(cls, webThreadReleaseSEL));
358}

386void WebCoreObjCDeallocOnWebThreadImpl(id self, SEL)
387{
388 if (!WebThreadIsEnabled()) {
389 [self _webcore_releaseOnWebThread];
390 return;
391 }
392
393 {
394 Locker locker { webCoreReleaseLock };
395 if ([self retainCount] != 1) {
396 // This is not the only reference retaining the object, so another
397 // thread could also call release - hold the lock whilst calling
398 // release to avoid a race condition.
399 [self _webcore_releaseOnWebThread];
400 return;
401 }
402 // This is the only reference retaining the object, so we can
403 // safely release the webCoreReleaseLock now.
404 }
405 if (WebThreadIsCurrent())
406 [self _webcore_releaseOnWebThread];
407 else
408 WebThreadAdoptAndRelease(self);
409}
410
411void WebCoreObjCDeallocWithWebThreadLockImpl(id self, SEL)
412{
413 Locker locker { webCoreReleaseLock };
414 if (WebThreadIsLockedOrDisabled() || 1 != [self retainCount])
415 [self _webcore_releaseWithWebThreadLock];
416 else
417 WebThreadAdoptAndRelease(self);
418}

421{

423
424 RetainPtr<CFMutableArrayRef> objects;
425 {

427 if (CFArrayGetCount(webThreadReleaseObjArray().get())) {

433 if (!objects)

435
436 for (unsigned i = 0, count = CFArrayGetCount(objects.get()); i < count; ++i) {

440#endif
441 }

485static void MainRunLoopAutoUnlock(CFRunLoopObserverRef, CFRunLoopActivity, void*)
486{
487 ASSERT(!WebThreadIsCurrent());
488

490 return;
491
492 if (!mainThreadHasPendingAutoUnlock)
493 return;
494
495 mainThreadHasPendingAutoUnlock = NO;

498 _WebThreadUnlock();

506 mainThreadHasPendingAutoUnlock = YES;
507 CFRunLoopAddObserver(CFRunLoopGetCurrent(), mainRunLoopAutoUnlockObserver().get(), kCFRunLoopCommonModes);

509 CFRunLoopWakeUp(CFRunLoopGetMain());

513static void WebRunLoopLockInternal(AutoreleasePoolOperation poolOperation)
514{
515 _WebThreadLock();
516 if (poolOperation == PushOrPopAutoreleasePool && !perCalloutAutoreleasepoolEnabled)
517 autoreleasePoolMark = objc_autoreleasePoolPush();
518 isWebThreadLocked = YES;
519}
520
521static void WebRunLoopUnlockInternal(AutoreleasePoolOperation poolOperation)
522{
523 ASSERT(sAsyncDelegates());
524 if ([sAsyncDelegates() count]) {
525 for (NSInvocation* invocation in sAsyncDelegates().get())
526 SendDelegateMessage(invocation);
527 [sAsyncDelegates() removeAllObjects];
528 }
529
530 if (poolOperation == PushOrPopAutoreleasePool && !perCalloutAutoreleasepoolEnabled)
531 objc_autoreleasePoolPop(autoreleasePoolMark);
532
533 _WebThreadUnlock();
534 isWebThreadLocked = NO;
535}

537static void WebRunLoopLock(CFRunLoopObserverRef, CFRunLoopActivity activity, void*)
538{
539 ASSERT(WebThreadIsCurrent());
540 ASSERT_UNUSED(activity, activity == kCFRunLoopAfterWaiting || activity == kCFRunLoopBeforeTimers || activity == kCFRunLoopBeforeSources);
541
542 // If the WebThread is locked by the main thread then we want to
543 // grab the lock ourselves when the main thread releases the lock.
544 if (isWebThreadLocked && !mainThreadLockCount)
545 return;
546 WebRunLoopLockInternal(PushOrPopAutoreleasePool);
547}
548
549static void WebRunLoopUnlock(CFRunLoopObserverRef, CFRunLoopActivity activity, void*)
550{
551 ASSERT(WebThreadIsCurrent());
552 ASSERT_UNUSED(activity, activity == kCFRunLoopBeforeWaiting || activity == kCFRunLoopExit);
553 ASSERT(!mainThreadLockCount);
554
555 if (!isWebThreadLocked)
556 return;
557 WebRunLoopUnlockInternal(PushOrPopAutoreleasePool);
558}

560static void MainRunLoopUnlockGuard(CFRunLoopObserverRef observer, CFRunLoopActivity activity, void* context)
561{
562 ASSERT(!WebThreadIsCurrent());
563
564 // We shouldn't have the web lock at this point. However, MobileMail sometimes
565 // get to a state where the main thread has web lock but it didn't release it on last
566 // runloop exit, and web thread gets stuck at waiting for the lock. If this happens,
567 // we need to help release the lock. See <rdar://problem/8005192>.
568 if (mainThreadLockCount && !sMainThreadModalCount) {
569 NSLog(@"WARNING: Main thread didn't release the lock at last runloop exit!");
570
571 MainRunLoopAutoUnlock(observer, activity, context);
572 if (mainThreadLockCount)
573 mainThreadLockCount = 0;
574 }
575}

633static void* RunWebThread(void*)
634{
635 FloatingPointEnvironment::singleton().propagateMainThreadEnvironment();
636
637 // WTF::initializeWebThread() needs to be called before JSC::initialize() since the
638 // code invoked by the latter needs to know if it's running on the WebThread. See
639 // <rdar://problem/8502487>.
640 WTF::initializeWebThread();
641 JSC::initialize();
642 WebCore::populateJITOperations();
643
644 // Make sure that the WebThread and the main thread share the same ThreadGlobalData objects.
645 WebCore::threadGlobalData().setWebCoreThreadData();
646
647#if HAVE(PTHREAD_SETNAME_NP)
648 pthread_setname_np("WebThread");
649#endif
650
651 webThreadContext = CurrentThreadContext();
652 webThreadRunLoop = CFRunLoopGetCurrent();
653 webThreadNSRunLoop() = [NSRunLoop currentRunLoop];
654
655 auto webRunLoopLockObserverRef = adoptCF(CFRunLoopObserverCreate(nullptr, kCFRunLoopBeforeTimers | kCFRunLoopBeforeSources | kCFRunLoopAfterWaiting, YES, 0, WebRunLoopLock, nullptr));
656 CFRunLoopAddObserver(webThreadRunLoop, webRunLoopLockObserverRef.get(), kCFRunLoopCommonModes);
657
658 WebThreadInitRunQueue();
659
660 // We must have the lock when CA paints in the web thread. CA commits at 2000000 so we use larger order number than that to free the lock.
661 auto webRunLoopUnlockObserverRef = adoptCF(CFRunLoopObserverCreate(nullptr, kCFRunLoopBeforeWaiting | kCFRunLoopExit, YES, 2500000, WebRunLoopUnlock, nullptr));
662 CFRunLoopAddObserver(webThreadRunLoop, webRunLoopUnlockObserverRef.get(), kCFRunLoopCommonModes);
663
664 CFRunLoopSourceContext ReleaseSourceContext = {0, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, HandleWebThreadReleaseSource};
665 webThreadReleaseSource() = adoptCF(CFRunLoopSourceCreate(nullptr, -1, &ReleaseSourceContext));
666 CFRunLoopAddSource(webThreadRunLoop, webThreadReleaseSource().get(), kCFRunLoopDefaultMode);
667
668 perCalloutAutoreleasepoolEnabled = _CFRunLoopSetPerCalloutAutoreleasepoolEnabled(YES);
669
670 {
671 Locker locker { startupLock };
672 startupCondition.notifyOne();
673 }
674
675 while (1)
676 CFRunLoopRunInMode(kCFRunLoopDefaultMode, DistantFuture, true);
677
678 return nullptr;
679}

681static void StartWebThread()
682{
683 webThreadStarted = TRUE;
684
685 // ThreadGlobalData touches AtomString, which requires Threading initialization.
686 WTF::initializeMainThread();
687
688 // Initialize AtomString on the main thread.
689 WTF::AtomString::init();
690
691 // Initialize ThreadGlobalData on the main UI thread so that the WebCore thread
692 // can later set it's thread-specific data to point to the same objects.
693 WebCore::ThreadGlobalData& unused = WebCore::threadGlobalData();
694 UNUSED_PARAM(unused);
695
696 // register class for WebThread deallocation
697 WebCoreObjCDeallocOnWebThread([WAKWindow class]);
698 WebCoreObjCDeallocWithWebThreadLock([WAKView class]);
699
700 CFRunLoopRef runLoop = CFRunLoopGetCurrent();
701 CFRunLoopSourceContext delegateSourceContext = {0, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, nullptr, HandleDelegateSource};
702 delegateSource() = adoptCF(CFRunLoopSourceCreate(nullptr, 0, &delegateSourceContext));
703
704 // We shouldn't get delegate callbacks while scrolling, but there might be
705 // one outstanding when we start. Add the source for all common run loop
706 // modes so we don't block the web thread while scrolling.
707 CFRunLoopAddSource(runLoop, delegateSource().get(), kCFRunLoopCommonModes);
708
709 sAsyncDelegates() = adoptNS([[NSMutableArray alloc] init]);
710
711 mainRunLoopAutoUnlockObserver() = adoptCF(CFRunLoopObserverCreate(nullptr, kCFRunLoopBeforeWaiting | kCFRunLoopExit, YES, 3000001, MainRunLoopAutoUnlock, nullptr));
712
713 pthread_attr_t tattr;
714 pthread_attr_init(&tattr);
715 pthread_attr_setscope(&tattr, PTHREAD_SCOPE_SYSTEM);
716 // The web thread is a secondary thread, and secondary threads are usually given
717 // a 512 kb stack, but we need more space in order to have room for the JavaScriptCore
718 // reentrancy limit. This limit works on both the simulator and the device.
719 pthread_attr_setstacksize(&tattr, 800 * KB);
720
721 pthread_attr_set_qos_class_np(&tattr, QOS_CLASS_USER_INTERACTIVE, -10);
722
723 // Wait for the web thread to startup completely before we continue.
724 {
725 Locker locker { startupLock };
726
727 // Propagate the mainThread's fenv to workers & the web thread.
728 FloatingPointEnvironment::singleton().saveMainThreadEnvironment();
729
730 pthread_create(&webThread, &tattr, RunWebThread, nullptr);
731 pthread_attr_destroy(&tattr);
732
733 startupCondition.wait(startupLock);
734 }
735
736 initializeApplicationUIThread();
737}

740static unsigned lockCount;
741#endif
742
743static void _WebThreadLock()
744{
745 // Suspend the web thread if the main thread is trying to lock.
746 bool onMainThread = pthread_main_np();
747 if (onMainThread)
748 webThreadShouldYield = true;
749 else if (!WebThreadIsCurrent()) {
750 NSLog(@"%s, %p: Tried to obtain the web lock from a thread other than the main thread or the web thread. This may be a result of calling to UIKit from a secondary thread. Crashing now...", __PRETTY_FUNCTION__, CurrentThreadContext());
751 CRASH();
752 }
753
754 webLock.lock();
755
756#if LOG_WEB_LOCK || LOG_MAIN_THREAD_LOCKING
757 lockCount++;
758#if LOG_WEB_LOCK
759 NSLog(@"lock %d, web-thread: %d", lockCount, WebThreadIsCurrent());
760#endif
761#endif
762 if (onMainThread) {
763 ASSERT(CFRunLoopGetCurrent() == CFRunLoopGetMain());
764 webThreadShouldYield = false;
765 mainThreadLockCount++;
766#if LOG_MAIN_THREAD_LOCKING
767 if (!sendingDelegateMessage && lockCount == 1)
768 NSLog(@"Main thread locking outside of delegate messages.");
769#endif
770 } else {
771 webThreadLockCount++;
772 if (webThreadLockCount > 1) {
773 NSLog(@"%s, %p: Multiple locks on web thread not allowed! Please file a bug. Crashing now...", __PRETTY_FUNCTION__, CurrentThreadContext());
774 CRASH();
775 }
776 }
777}

847{

851 lockCount--;
