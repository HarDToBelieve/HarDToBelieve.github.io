37// The time after which we'll check the blocklist data.
38static time_t blocklistNextCheckTime;
39
40// The number of seconds before we'll check if the blocklist data has changed.
41const time_t blocklistCheckTimeInterval = 60 * 10;
42
43// The time when we last re-parsed the blocklist file.
44static time_t blocklistUpdateTime;
45
46#if HAVE(READ_ONLY_SYSTEM_VOLUME)
47NSString * const blocklistPath = @"/Library/Apple/System/Library/CoreServices/XProtect.bundle/Contents/Resources/XProtect.meta.plist";
48#else
49NSString * const blocklistPath = @"/System/Library/CoreServices/XProtect.bundle/Contents/Resources/XProtect.meta.plist";
50#endif

57WorkQueue& BlocklistUpdater::queue()
58{
59 static NeverDestroyed<RefPtr<WorkQueue>> queue;
60 static dispatch_once_t once;
61 dispatch_once(&once, ^{
62 queue.get() = WorkQueue::create("com.apple.WebKit.Blocklist");
63 });
64
65 return *queue.get();
66}
67
68NSDictionary * BlocklistUpdater::readBlocklistData()
69{
70 NSData *data = [NSData dataWithContentsOfFile:blocklistPath];
71 if (!data)
72 return nil;
73
74 return dynamic_objc_cast<NSDictionary>([NSPropertyListSerialization propertyListWithData:data options:NSPropertyListImmutable format:nullptr error:nullptr]);
75}
76
77void BlocklistUpdater::reloadIfNecessary()
78{
79 struct timeval timeVal;
80 if (!gettimeofday(&timeVal, NULL)) {
81 if (timeVal.tv_sec < blocklistNextCheckTime)
82 return;
83 }
84
85 blocklistNextCheckTime = timeVal.tv_sec + blocklistCheckTimeInterval;
86
87 struct stat statBuf;
88 if (stat([blocklistPath fileSystemRepresentation], &statBuf) == -1)
89 return;
90
91 if (statBuf.st_mtimespec.tv_sec == blocklistUpdateTime)
92 return;
93 NSDictionary *propertyList = readBlocklistData();
94 if (!propertyList)
95 return;
96
97 if (s_pluginBlocklist) {
98 delete s_pluginBlocklist;
99 s_pluginBlocklist = 0;
100 }
101
102 if (s_webGLBlocklist) {
103 delete s_webGLBlocklist;
104 s_webGLBlocklist = 0;
105 }
106
107 s_pluginBlocklist = PluginBlocklist::create(propertyList).release();
108 s_webGLBlocklist = WebGLBlocklist::create(propertyList).release();
109
110 blocklistUpdateTime = statBuf.st_mtimespec.tv_sec;
111}
