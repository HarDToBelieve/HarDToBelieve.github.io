71static NSInteger WKCoreCursor_coreCursorType(id self, SEL)
72{
73 if (self == busyButClickableNSCursor)
74 return kCoreCursorBusyButClickable;
75 if (self == makeAliasNSCursor)
76 return kCoreCursorMakeAlias;
77 if (self == moveNSCursor)
78 return kCoreCursorWindowMove;
79 if (self == resizeEastNSCursor)
80 return kCoreCursorWindowResizeEast;
81 if (self == resizeEastWestNSCursor)
82 return kCoreCursorWindowResizeEastWest;
83 if (self == resizeNorthNSCursor)
84 return kCoreCursorWindowResizeNorth;
85 if (self == resizeNorthSouthNSCursor)
86 return kCoreCursorWindowResizeNorthSouth;
87 if (self == resizeNortheastNSCursor)
88 return kCoreCursorWindowResizeNorthEast;
89 if (self == resizeNortheastSouthwestNSCursor)
90 return kCoreCursorWindowResizeNorthEastSouthWest;
91 if (self == resizeNorthwestNSCursor)
92 return kCoreCursorWindowResizeNorthWest;
93 if (self == resizeNorthwestSoutheastNSCursor)
94 return kCoreCursorWindowResizeNorthWestSouthEast;
95 if (self == resizeSouthNSCursor)
96 return kCoreCursorWindowResizeSouth;
97 if (self == resizeSoutheastNSCursor)
98 return kCoreCursorWindowResizeSouthEast;
99 if (self == resizeSouthwestNSCursor)
100 return kCoreCursorWindowResizeSouthWest;
101 if (self == resizeWestNSCursor)
102 return kCoreCursorWindowResizeWest;
103 if (self == cellNSCursor)
104 return kCoreCursorCell;
105 if (self == helpNSCursor)
106 return kCoreCursorHelp;
107 if (self == zoomInNSCursor)
108 return kCoreCursorZoomIn;
109 if (self == zoomOutNSCursor)
110 return kCoreCursorZoomOut;
111
112 return NSNotFound;
113}

115static Class createCoreCursorClass()
116{
117 Class coreCursorClass = objc_allocateClassPair([NSCursor class], "WKCoreCursor", 0);
118 SEL coreCursorType = NSSelectorFromString(@"_coreCursorType");
119 class_addMethod(coreCursorClass, coreCursorType, (IMP)WKCoreCursor_coreCursorType, method_getTypeEncoding(class_getInstanceMethod([NSCursor class], coreCursorType)));
120 objc_registerClassPair(coreCursorClass);
121 return coreCursorClass;
122}
123
124static Class coreCursorClass()
125{
126 Class coreCursorClass = objc_lookUpClass("WKCoreCursor");
127 if (!coreCursorClass)
128 coreCursorClass = createCoreCursorClass();
129 return coreCursorClass;
130}

132static NSCursor *cursor(const char *name)
133{
134 __strong NSCursor **slot = nullptr;
135
136 if (!strcmp(name, "BusyButClickable"))
137 slot = &busyButClickableNSCursor;
138 else if (!strcmp(name, "MakeAlias"))
139 slot = &makeAliasNSCursor;
140 else if (!strcmp(name, "Move"))
141 slot = &moveNSCursor;
142 else if (!strcmp(name, "ResizeEast"))
143 slot = &resizeEastNSCursor;
144 else if (!strcmp(name, "ResizeEastWest"))
145 slot = &resizeEastWestNSCursor;
146 else if (!strcmp(name, "ResizeNorth"))
147 slot = &resizeNorthNSCursor;
148 else if (!strcmp(name, "ResizeNorthSouth"))
149 slot = &resizeNorthSouthNSCursor;
150 else if (!strcmp(name, "ResizeNortheast"))
151 slot = &resizeNortheastNSCursor;
152 else if (!strcmp(name, "ResizeNortheastSouthwest"))
153 slot = &resizeNortheastSouthwestNSCursor;
154 else if (!strcmp(name, "ResizeNorthwest"))
155 slot = &resizeNorthwestNSCursor;
156 else if (!strcmp(name, "ResizeNorthwestSoutheast"))
157 slot = &resizeNorthwestSoutheastNSCursor;
158 else if (!strcmp(name, "ResizeSouth"))
159 slot = &resizeSouthNSCursor;
160 else if (!strcmp(name, "ResizeSoutheast"))
161 slot = &resizeSoutheastNSCursor;
162 else if (!strcmp(name, "ResizeSouthwest"))
163 slot = &resizeSouthwestNSCursor;
164 else if (!strcmp(name, "ResizeWest"))
165 slot = &resizeWestNSCursor;
166 else if (!strcmp(name, "Cell"))
167 slot = &cellNSCursor;
168 else if (!strcmp(name, "Help"))
169 slot = &helpNSCursor;
170 else if (!strcmp(name, "ZoomIn"))
171 slot = &zoomInNSCursor;
172 else if (!strcmp(name, "ZoomOut"))
173 slot = &zoomOutNSCursor;
174
175 if (!slot)
176 return nil;
177
178 if (!*slot)
179 *slot = [[coreCursorClass() alloc] init];
180 return *slot;
181}

192// Simple NSCursor calls shouldn't need protection,
193// but creating a cursor with a bad image might throw.
194
195#if ENABLE(CUSTOM_CURSOR_SUPPORT)
196#if ENABLE(MOUSE_CURSOR_SCALE)
197static RetainPtr<NSCursor> createCustomCursor(Image* image, const IntPoint& hotSpot, float scale)
198#else
199static RetainPtr<NSCursor> createCustomCursor(Image* image, const IntPoint& hotSpot)
200#endif
201{
202 // FIXME: The cursor won't animate. Not sure if that's a big deal.
203 auto nsImage = image->snapshotNSImage();
204 if (!nsImage)
205 return nullptr;
206 BEGIN_BLOCK_OBJC_EXCEPTIONS
207
208#if ENABLE(MOUSE_CURSOR_SCALE)
209 NSSize size = NSMakeSize(image->width() / scale, image->height() / scale);
210 NSSize expandedSize = NSMakeSize(ceil(size.width), ceil(size.height));
211
212 // Pad the image with transparent pixels so it has an integer boundary.
213 if (size.width != expandedSize.width || size.height != expandedSize.height) {
214 RetainPtr<NSImage> expandedImage = adoptNS([[NSImage alloc] initWithSize:expandedSize]);
215 NSRect toRect = NSMakeRect(0, expandedSize.height - size.height, size.width, size.height);
216 NSRect fromRect = NSMakeRect(0, 0, image->width(), image->height());
217
218 [expandedImage lockFocus];
219 [nsImage drawInRect:toRect fromRect:fromRect operation:NSCompositingOperationSourceOver fraction:1];
220 [expandedImage unlockFocus];
221
222 return adoptNS([[NSCursor alloc] initWithImage:expandedImage.get() hotSpot:hotSpot]);
223 }
224
225 // Scale the image and its representation to match retina resolution.
226 [nsImage setSize:expandedSize];
227 [[[nsImage representations] objectAtIndex:0] setSize:expandedSize];
228#endif
229
230 return adoptNS([[NSCursor alloc] initWithImage:nsImage.get() hotSpot:hotSpot]);
231 END_BLOCK_OBJC_EXCEPTIONS
232 return nullptr;
233}
234#endif // ENABLE(CUSTOM_CURSOR_SUPPORT)

236void Cursor::ensurePlatformCursor() const
237{
238 if (m_platformCursor)
239 return;
240
241 switch (m_type) {
242 case Cursor::Pointer:
243 m_platformCursor = [NSCursor arrowCursor];
244 break;
245
246 case Cursor::Cross:
247 m_platformCursor = [NSCursor crosshairCursor];
248 break;
249
250 case Cursor::Hand:
251 m_platformCursor = [NSCursor pointingHandCursor];
252 break;
253
254 case Cursor::IBeam:
255 m_platformCursor = [NSCursor IBeamCursor];
256 break;
257
258 case Cursor::Wait:
259 m_platformCursor = cursor("BusyButClickable");
260 break;
261
262 case Cursor::Help:
263 m_platformCursor = cursor("Help");
264 break;
265
266 case Cursor::Move:
267 case Cursor::MiddlePanning:
268 m_platformCursor = cursor("Move");
269 break;
270
271 case Cursor::EastResize:
272 case Cursor::EastPanning:
273 m_platformCursor = cursor("ResizeEast");
274 break;
275
276 case Cursor::NorthResize:
277 case Cursor::NorthPanning:
278 m_platformCursor = cursor("ResizeNorth");
279 break;
280
281 case Cursor::NorthEastResize:
282 case Cursor::NorthEastPanning:
283 m_platformCursor = cursor("ResizeNortheast");
284 break;
285
286 case Cursor::NorthWestResize:
287 case Cursor::NorthWestPanning:
288 m_platformCursor = cursor("ResizeNorthwest");
289 break;
290
291 case Cursor::SouthResize:
292 case Cursor::SouthPanning:
293 m_platformCursor = cursor("ResizeSouth");
294 break;
295
296 case Cursor::SouthEastResize:
297 case Cursor::SouthEastPanning:
298 m_platformCursor = cursor("ResizeSoutheast");
299 break;
300
301 case Cursor::SouthWestResize:
302 case Cursor::SouthWestPanning:
303 m_platformCursor = cursor("ResizeSouthwest");
304 break;
305
306 case Cursor::WestResize:
307 case Cursor::WestPanning:
308 m_platformCursor = cursor("ResizeWest");
309 break;
310
311 case Cursor::NorthSouthResize:
312 m_platformCursor = cursor("ResizeNorthSouth");
313 break;
314
315 case Cursor::EastWestResize:
316 m_platformCursor = cursor("ResizeEastWest");
317 break;
318
319 case Cursor::NorthEastSouthWestResize:
320 m_platformCursor = cursor("ResizeNortheastSouthwest");
321 break;
322
323 case Cursor::NorthWestSouthEastResize:
324 m_platformCursor = cursor("ResizeNorthwestSoutheast");
325 break;
326
327 case Cursor::ColumnResize:
328 m_platformCursor = [NSCursor resizeLeftRightCursor];
329 break;
330
331 case Cursor::RowResize:
332 m_platformCursor = [NSCursor resizeUpDownCursor];
333 break;
334
335 case Cursor::VerticalText:
336 m_platformCursor = [NSCursor IBeamCursorForVerticalLayout];
337 break;
338
339 case Cursor::Cell:
340 m_platformCursor = cursor("Cell");
341 break;
342
343 case Cursor::ContextMenu:
344 m_platformCursor = [NSCursor contextualMenuCursor];
345 break;
346
347 case Cursor::Alias:
348 m_platformCursor = cursor("MakeAlias");
349 break;
350
351 case Cursor::Progress:
352 m_platformCursor = cursor("BusyButClickable");
353 break;
354
355 case Cursor::NoDrop:
356 m_platformCursor = [NSCursor operationNotAllowedCursor];
357 break;
358
359 case Cursor::Copy:
360 m_platformCursor = [NSCursor dragCopyCursor];
361 break;
362
363 case Cursor::None:
364#if ENABLE(CUSTOM_CURSOR_SUPPORT)
365 m_platformCursor = adoptNS([[NSCursor alloc] initWithImage:adoptNS([[NSImage alloc] initWithSize:NSMakeSize(1, 1)]).get() hotSpot:NSZeroPoint]);
366#else
367 m_platformCursor = [NSCursor arrowCursor];
368#endif
369 break;
370
371 case Cursor::NotAllowed:
372 m_platformCursor = [NSCursor operationNotAllowedCursor];
373 break;
374
375 case Cursor::ZoomIn:
376 m_platformCursor = cursor("ZoomIn");
377 break;
378
379 case Cursor::ZoomOut:
380 m_platformCursor = cursor("ZoomOut");
381 break;
382
383 case Cursor::Grab:
384 m_platformCursor = [NSCursor openHandCursor];
385 break;
386
387 case Cursor::Grabbing:
388 m_platformCursor = [NSCursor closedHandCursor];
389 break;
390
391 case Cursor::Custom:
392#if ENABLE(CUSTOM_CURSOR_SUPPORT)
393#if ENABLE(MOUSE_CURSOR_SCALE)
394 m_platformCursor = createCustomCursor(m_image.get(), m_hotSpot, m_imageScaleFactor);
395#else
396 m_platformCursor = createCustomCursor(m_image.get(), m_hotSpot);
397#endif // ENABLE(MOUSE_CURSOR_SCALE)
398#endif // ENABLE(CUSTOM_CURSOR_SUPPORT)
399 break;
400 }
401}

403NSCursor *Cursor::platformCursor() const
404{
405 ensurePlatformCursor();
406 return m_platformCursor.get();
407}
408
409void Cursor::setAsPlatformCursor() const
410{
411 NSCursor *cursor = platformCursor();
412 if ([NSCursor currentCursor] == cursor)
413 return;
414 [cursor set];
415}
