64DataDetectorHighlight::DataDetectorHighlight(Page& page, DataDetectorHighlightClient& client, Type type, RetainPtr<DDHighlightRef>&& ddHighlight, SimpleRange&& range)
65 : m_client(makeWeakPtr(client))
66 , m_page(makeWeakPtr(page))
67 , m_range(WTFMove(range))
68 , m_graphicsLayer(GraphicsLayer::create(page.chrome().client().graphicsLayerFactory(), *this))
69 , m_type(type)
70{
71 ASSERT(ddHighlight);
72
73 m_graphicsLayer->setDrawsContent(true);
74
75 setHighlight(ddHighlight.get());
76
77 // Set directly on the PlatformCALayer so that when we leave the 'from' value implicit
78 // in our animations, we get the right initial value regardless of flush timing.
79 downcast<GraphicsLayerCA>(layer()).platformCALayer()->setOpacity(0);
80}
81
82void DataDetectorHighlight::setHighlight(DDHighlightRef highlight)
83{
84 if (!PAL::isDataDetectorsFrameworkAvailable())
85 return;
86
87 if (!m_client)
88 return;
89
90 m_highlight = highlight;
91
92 if (!m_highlight)
93 return;
94
95 CGRect highlightBoundingRect = PAL::softLink_DataDetectors_DDHighlightGetBoundingRect(m_highlight.get());
96 m_graphicsLayer->setPosition(FloatPoint(highlightBoundingRect.origin));
97 m_graphicsLayer->setSize(FloatSize(highlightBoundingRect.size));
98
99 m_graphicsLayer->setNeedsDisplay();
100}

117void DataDetectorHighlight::paintContents(const GraphicsLayer*, GraphicsContext& graphicsContext, const FloatRect&, GraphicsLayerPaintBehavior)
118{
119 if (!PAL::isDataDetectorsFrameworkAvailable())
120 return;
121
122 // FIXME: This needs to be moved into GraphicsContext as a DisplayList-compatible drawing command.
123 if (!graphicsContext.hasPlatformContext()) {
124 ASSERT_NOT_REACHED();
125 return;
126 }
127
128 CGContextRef cgContext = graphicsContext.platformContext();
129
130 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
131 CGLayerRef highlightLayer = PAL::softLink_DataDetectors_DDHighlightGetLayerWithContext(highlight(), cgContext);
132 ALLOW_DEPRECATED_DECLARATIONS_END
133 CGRect highlightBoundingRect = PAL::softLink_DataDetectors_DDHighlightGetBoundingRect(highlight());
134 highlightBoundingRect.origin = CGPointZero;
135
136 CGContextDrawLayerInRect(cgContext, highlightBoundingRect, highlightLayer);
137}

147void DataDetectorHighlight::fadeIn()
148{
149 RetainPtr<CABasicAnimation> animation = [CABasicAnimation animationWithKeyPath:@"opacity"];
150 [animation setDuration:highlightFadeAnimationDuration.seconds()];
151 [animation setFillMode:kCAFillModeForwards];
152 [animation setRemovedOnCompletion:false];
153 [animation setToValue:@1];
154
155 auto platformAnimation = PlatformCAAnimationCocoa::create(animation.get());
156 downcast<GraphicsLayerCA>(layer()).platformCALayer()->addAnimationForKey("FadeHighlightIn", platformAnimation.get());
157}
158
159void DataDetectorHighlight::fadeOut()
160{
161 RetainPtr<CABasicAnimation> animation = [CABasicAnimation animationWithKeyPath:@"opacity"];
162 [animation setDuration:highlightFadeAnimationDuration.seconds()];
163 [animation setFillMode:kCAFillModeForwards];
164 [animation setRemovedOnCompletion:false];
165 [animation setToValue:@0];
166
167 [CATransaction begin];
168 [CATransaction setCompletionBlock:[protectedSelf = makeRef(*this)]() mutable {
169 protectedSelf->didFinishFadeOutAnimation();
170 }];
171
172 auto platformAnimation = PlatformCAAnimationCocoa::create(animation.get());
173 downcast<GraphicsLayerCA>(layer()).platformCALayer()->addAnimationForKey("FadeHighlightOut", platformAnimation.get());
174 [CATransaction commit];
175}
176
177void DataDetectorHighlight::didFinishFadeOutAnimation()
178{
179 if (!m_client)
180 return;
181
182 if (m_client->activeHighlight() == this)
183 return;
184
185 layer().removeFromParent();
186}

188bool areEquivalent(const DataDetectorHighlight* a, const DataDetectorHighlight* b)
189{
190 if (a == b)
191 return true;
192
193 if (!a || !b)
194 return false;
195
196 return a->type() == b->type() && a->range() == b->range();
197}
