37HIDElement::HIDElement(IOHIDElementRef element)
38 : m_physicalMin(IOHIDElementGetPhysicalMin(element))
39 , m_physicalMax(IOHIDElementGetPhysicalMax(element))
40 , m_physicalValue(m_physicalMin)
41 , m_usage(IOHIDElementGetUsage(element))
42 , m_usagePage(IOHIDElementGetUsagePage(element))
43 , m_cookie(IOHIDElementGetCookie(element))
44 , m_rawElement(element)
45{
46}
47
48void HIDElement::valueChanged(IOHIDValueRef value)
49{
50 if (IOHIDValueGetElement(value) != m_rawElement.get()) {
51 LOG(HID, "HIDElement: Changed value whose IOHIDElement %p doesn't match %p", IOHIDValueGetElement(value), m_rawElement.get());
52 return;
53 }
54
55 m_physicalValue = IOHIDValueGetScaledValue(value, kIOHIDValueScaleTypePhysical);
56}
