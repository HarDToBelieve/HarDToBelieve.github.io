38{
39 WebCore::PlatformSpeechSynthesizer* m_synthesizerObject;
40 // Hold a Ref to the utterance so that it won't disappear until the synth is done with it.
41 WebCore::PlatformSpeechSynthesisUtterance* m_utterance;

45}

52@implementation WebSpeechSynthesisWrapper

54- (WebSpeechSynthesisWrapper *)initWithSpeechSynthesizer:(WebCore::PlatformSpeechSynthesizer *)synthesizer
55{
56 if (!(self = [super init]))
57 return nil;
58
59 m_synthesizerObject = synthesizer;
60 [self updateBasePitchForSynthesizer];
61 return self;
62}
63
64// NSSpeechSynthesizer expects a Words per Minute (WPM) rate. There is no preset default
65// but they recommend that normal speaking is 180-220 WPM
66- (float)convertRateToWPM:(float)rate
67{
68 // We'll say 200 WPM is the default 1x value.
69 return 200.0f * rate;
70}
71
72- (float)convertPitchToNSSpeechValue:(float)pitch
73{
74 // This allows the base pitch to range from 0% - 200% of the normal pitch.
75 return m_basePitch * pitch;
76}
77
78- (void)updateBasePitchForSynthesizer
79{
80 // Reset the base pitch whenever we change voices, since the base pitch is different for each voice.
81 [m_synthesizer setObject:nil forProperty:NSSpeechResetProperty error:nil];
82 m_basePitch = [[m_synthesizer objectForProperty:NSSpeechPitchBaseProperty error:nil] floatValue];
83}
84
85- (void)speakUtterance:(WebCore::PlatformSpeechSynthesisUtterance *)utterance
86{
87 // When speak is called we should not have an existing speech utterance outstanding.
88 ASSERT(!m_utterance);
89 ASSERT(utterance);
90
91 if (!m_synthesizer) {
92 m_synthesizer = adoptNS([[NSSpeechSynthesizer alloc] initWithVoice:nil]);
93 [m_synthesizer setDelegate:self];
94 }
95
96 // Find if we should use a specific voice based on the voiceURI in utterance.
97 // Otherwise, find the voice that matches the language. The Mac doesn't have a default voice per language, so the first
98 // one will have to do.
99
100 WebCore::PlatformSpeechSynthesisVoice* utteranceVoice = utterance->voice();
101 // If no voice was specified, try to match by language.
102 if (!utteranceVoice && !utterance->lang().isEmpty()) {
103 for (auto& voice : m_synthesizerObject->voiceList()) {
104 if (equalIgnoringASCIICase(utterance->lang(), voice->lang())) {
105 utteranceVoice = voice.get();
106 if (voice->isDefault())
107 break;
108 }
109 }
110 }
111
112 NSString *voiceURI = nil;
113 if (utteranceVoice)
114 voiceURI = utteranceVoice->voiceURI();
115 else
116 voiceURI = [NSSpeechSynthesizer defaultVoice];
117
118 // Don't set the voice unless necessary. There's a bug in NSSpeechSynthesizer such that
119 // setting the voice for the first time will cause the first speechDone callback to report it was unsuccessful.
120 BOOL updatePitch = NO;
121 if (![[m_synthesizer voice] isEqualToString:voiceURI]) {
122 [m_synthesizer setVoice:voiceURI];
123 // Reset the base pitch whenever we change voices.
124 updatePitch = YES;
125 }
126
127 if (m_basePitch == 0 || updatePitch)
128 [self updateBasePitchForSynthesizer];
129
130 [m_synthesizer setObject:[NSNumber numberWithFloat:[self convertPitchToNSSpeechValue:utterance->pitch()]] forProperty:NSSpeechPitchBaseProperty error:nil];
131 [m_synthesizer setRate:[self convertRateToWPM:utterance->rate()]];
132 [m_synthesizer setVolume:utterance->volume()];
133
134 m_utterance = utterance;
135 [m_synthesizer startSpeakingString:utterance->text()];
136 m_synthesizerObject->client()->didStartSpeaking(*m_utterance);
137}
138
139- (void)pause
140{
141 if (!m_utterance)
142 return;
143
144 [m_synthesizer pauseSpeakingAtBoundary:NSSpeechImmediateBoundary];
145 m_synthesizerObject->client()->didPauseSpeaking(*m_utterance);
146}
147
148- (void)resume
149{
150 if (!m_utterance)
151 return;
152
153 [m_synthesizer continueSpeaking];
154 m_synthesizerObject->client()->didResumeSpeaking(*m_utterance);
155}
156
157- (void)cancel
158{
159 if (!m_utterance)
160 return;
161
162 [m_synthesizer stopSpeakingAtBoundary:NSSpeechImmediateBoundary];
163 m_synthesizerObject->client()->speakingErrorOccurred(*m_utterance);
164 m_utterance = 0;
165}
166
167- (void)invalidate
168{
169 m_utterance = 0;
170 [m_synthesizer setDelegate:nil];
171 [m_synthesizer stopSpeakingAtBoundary:NSSpeechImmediateBoundary];
172}
173
174- (void)speechSynthesizer:(NSSpeechSynthesizer *)sender didFinishSpeaking:(BOOL)finishedSpeaking
175{
176 if (!m_utterance)
177 return;
178
179 UNUSED_PARAM(sender);
180
181 // Clear the m_utterance variable in case finish speaking kicks off a new speaking job immediately.
182 WebCore::PlatformSpeechSynthesisUtterance* utterance = m_utterance;
183 m_utterance = 0;
184
185 if (finishedSpeaking)
186 m_synthesizerObject->client()->didFinishSpeaking(*utterance);
187 else
188 m_synthesizerObject->client()->speakingErrorOccurred(*utterance);
189}
190
191- (void)speechSynthesizer:(NSSpeechSynthesizer *)sender willSpeakWord:(NSRange)characterRange ofString:(NSString *)string
192{
193 UNUSED_PARAM(sender);
194 UNUSED_PARAM(string);
195
196 if (!m_utterance)
197 return;
198
199 // Mac platform only supports word boundaries.
200 m_synthesizerObject->client()->boundaryEventOccurred(*m_utterance, WebCore::SpeechBoundary::SpeechWordBoundary, characterRange.location);
201}

217static RetainPtr<CFArrayRef> speechSynthesisGetVoiceIdentifiers()
218{
219 // Get all the voices offered by TTS.
220 // By default speech only returns "premium" voices, which does not include all the
221 // international voices. This allows us to offer speech synthesis for all supported languages.
222 return adoptCF(CopySpeechSynthesisVoicesForMode((__bridge CFArrayRef)@[ @"VoiceGroupDefault", @"VoiceGroupCompact" ]));
223}
224
225static NSString *speechSynthesisGetDefaultVoiceIdentifierForLocale(NSLocale *userLocale)
226{
227 if (!userLocale)
228 return nil;
229
230 return (__bridge NSString *)GetIdentifierStringForPreferredVoiceInListWithLocale(speechSynthesisGetVoiceIdentifiers().get(), (__bridge CFLocaleRef)userLocale);
231}

233void PlatformSpeechSynthesizer::initializeVoiceList()
234{
235 auto availableVoices = speechSynthesisGetVoiceIdentifiers();
236 NSUInteger count = [(__bridge NSArray *)availableVoices.get() count];
237 for (NSUInteger k = 0; k < count; k++) {
238 NSString *voiceName = [(__bridge NSArray *)availableVoices.get() objectAtIndex:k];
239 NSDictionary *attributes = [NSSpeechSynthesizer attributesForVoice:voiceName];
240
241 NSString *voiceURI = [attributes objectForKey:NSVoiceIdentifier];
242 NSString *name = [attributes objectForKey:NSVoiceName];
243 NSString *language = [attributes objectForKey:NSVoiceLocaleIdentifier];
244 NSString *defaultVoiceURI = speechSynthesisGetDefaultVoiceIdentifierForLocale(adoptNS([[NSLocale alloc] initWithLocaleIdentifier:language]).get());
245
246 // Change to BCP-47 format as defined by spec.
247 language = [language stringByReplacingOccurrencesOfString:@"_" withString:@"-"];
248
249 bool isDefault = [defaultVoiceURI isEqualToString:voiceURI];
250
251 m_voiceList.append(PlatformSpeechSynthesisVoice::create(voiceURI, name, language, true, isDefault));
252 }
