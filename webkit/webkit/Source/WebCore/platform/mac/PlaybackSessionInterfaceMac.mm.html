95void PlaybackSessionInterfaceMac::rateChanged(OptionSet<PlaybackSessionModel::PlaybackState> playbackState, double playbackRate, double defaultPlaybackRate)
96{
97#if ENABLE(WEB_PLAYBACK_CONTROLS_MANAGER)
98 auto isPlaying = playbackState.contains(PlaybackSessionModel::PlaybackState::Playing);
99 WebPlaybackControlsManager* controlsManager = playBackControlsManager();
100 [controlsManager setRate:isPlaying ? playbackRate : 0.];
101 [controlsManager setDefaultPlaybackRate:defaultPlaybackRate];
102 [controlsManager setPlaying:isPlaying];
103 updatePlaybackControlsManagerTiming(m_playbackSessionModel ? m_playbackSessionModel->currentTime() : 0, [[NSProcessInfo processInfo] systemUptime], playbackRate, isPlaying);
104#else
105 UNUSED_PARAM(isPlaying);
106 UNUSED_PARAM(playbackRate);
107#endif
108}

130#if ENABLE(WEB_PLAYBACK_CONTROLS_MANAGER)
131static RetainPtr<NSMutableArray> timeRangesToArray(const TimeRanges& timeRanges)
132{
133 RetainPtr<NSMutableArray> rangeArray = adoptNS([[NSMutableArray alloc] init]);
134
135 for (unsigned i = 0; i < timeRanges.length(); i++) {
136 const PlatformTimeRanges& ranges = timeRanges.ranges();
137 CMTimeRange range = PAL::CMTimeRangeMake(PAL::toCMTime(ranges.start(i)), PAL::toCMTime(ranges.end(i)));
138 [rangeArray addObject:[NSValue valueWithCMTimeRange:range]];
139 }
140
141 return rangeArray;
142}
143#endif

229void PlaybackSessionInterfaceMac::setPlayBackControlsManager(WebPlaybackControlsManager *manager)
230{
231 m_playbackControlsManager = manager;
232
233 if (!manager || !m_playbackSessionModel)
234 return;
235
236 NSTimeInterval anchorTimeStamp = ![manager rate] ? NAN : [[NSProcessInfo processInfo] systemUptime];
237 manager.timing = [getAVValueTimingClass() valueTimingWithAnchorValue:m_playbackSessionModel->currentTime() anchorTimeStamp:anchorTimeStamp rate:0];
238 double duration = m_playbackSessionModel->duration();
239 manager.contentDuration = duration;
240 manager.hasEnabledAudio = duration > 0;
241 manager.hasEnabledVideo = duration > 0;
242 manager.defaultPlaybackRate = m_playbackSessionModel->defaultPlaybackRate();
243 manager.rate = m_playbackSessionModel->isPlaying() ? m_playbackSessionModel->playbackRate() : 0.;
244 manager.seekableTimeRanges = timeRangesToArray(m_playbackSessionModel->seekableRanges()).get();
245 manager.canTogglePlayback = YES;
246 manager.playing = m_playbackSessionModel->isPlaying();
247 [manager setAudioMediaSelectionOptions:m_playbackSessionModel->audioMediaSelectionOptions() withSelectedIndex:static_cast<NSUInteger>(m_playbackSessionModel->audioMediaSelectedIndex())];
248 [manager setLegibleMediaSelectionOptions:m_playbackSessionModel->legibleMediaSelectionOptions() withSelectedIndex:static_cast<NSUInteger>(m_playbackSessionModel->legibleMediaSelectedIndex())];
249}

251void PlaybackSessionInterfaceMac::updatePlaybackControlsManagerCanTogglePictureInPicture()
252{
253 PlaybackSessionModel* model = playbackSessionModel();
254 if (!model) {
255 [playBackControlsManager() setCanTogglePictureInPicture:NO];
256 return;
257 }
258
259 [playBackControlsManager() setCanTogglePictureInPicture:model->isPictureInPictureSupported() && !model->externalPlaybackEnabled()];
260}

262void PlaybackSessionInterfaceMac::updatePlaybackControlsManagerTiming(double currentTime, double anchorTime, double playbackRate, bool isPlaying)
263{
264 WebPlaybackControlsManager *manager = playBackControlsManager();
265 if (!manager)
266 return;
267
268 PlaybackSessionModel *model = playbackSessionModel();
269 if (!model)
270 return;
271
272 double effectiveAnchorTime = playbackRate ? anchorTime : NAN;
273 double effectivePlaybackRate = playbackRate;
274 if (!isPlaying
275 || model->isScrubbing()
276 || (manager.rate > 0 && model->playbackStartedTime() >= currentTime)
277 || (manager.rate < 0 && model->playbackStartedTime() <= currentTime))
278 effectivePlaybackRate = 0;
279
280 manager.timing = [getAVValueTimingClass() valueTimingWithAnchorValue:currentTime anchorTimeStamp:effectiveAnchorTime rate:effectivePlaybackRate];
281}
