38auto PluginBlocklist::loadPolicyForPluginVersion(NSString *bundleIdentifier, NSString *bundleVersionString) -> LoadPolicy
39{
40 auto loadPolicy = LoadPolicy::LoadNormally;
41 BlocklistUpdater::queue().dispatchSync([&] {
42 BlocklistUpdater::reloadIfNecessary();
43 if (auto* pluginBlocklist = BlocklistUpdater::pluginBlocklist())
44 loadPolicy = pluginBlocklist->loadPolicyForPlugin(bundleIdentifier, bundleVersionString);
45 });
46
47 return loadPolicy;
48}

62std::unique_ptr<PluginBlocklist> PluginBlocklist::create(NSDictionary *propertyList)
63{
64 auto systemVersionDictionary = adoptCF(_CFCopySystemVersionDictionary());
65 CFStringRef osVersion = static_cast<CFStringRef>(CFDictionaryGetValue(systemVersionDictionary.get(), _kCFSystemVersionProductVersionKey));
66
67 NSDictionary *dictionary = [propertyList objectForKey:@"PlugInBlocklist"];
68
69 NSMutableDictionary *bundleIDToMinimumSecureVersion = [NSMutableDictionary dictionary];
70 NSMutableDictionary *bundleIDToMinimumCompatibleVersion = [NSMutableDictionary dictionary];
71 NSMutableDictionary *bundleIDToBlockedVersions = [NSMutableDictionary dictionary];
72 NSMutableSet *bundleIDsWithAvailableUpdates = [NSMutableSet set];
73
74 for (NSString *osVersionComponent in splitOSVersion((__bridge NSString *)osVersion)) {
75 NSDictionary *bundleIDs = [dictionary objectForKey:osVersionComponent];
76 if (!bundleIDs)
77 continue;
78
79 for (NSString *bundleID in bundleIDs) {
80 NSDictionary *versionInfo = [bundleIDs objectForKey:bundleID];
81 assert(versionInfo);
82
83 if (![versionInfo isKindOfClass:[NSDictionary class]])
84 continue;
85
86 [bundleIDToMinimumSecureVersion removeObjectForKey:bundleID];
87 [bundleIDToMinimumCompatibleVersion removeObjectForKey:bundleID];
88 [bundleIDToBlockedVersions removeObjectForKey:bundleID];
89
90 if (NSArray *blockedVersions = [versionInfo objectForKey:@"BlockedPlugInBundleVersions"])
91 [bundleIDToBlockedVersions setObject:blockedVersions forKey:bundleID];
92
93 if (NSString *minimumSecureVersion = [versionInfo objectForKey:@"MinimumPlugInBundleVersion"])
94 [bundleIDToMinimumSecureVersion setObject:minimumSecureVersion forKey:bundleID];
95
96 if (NSString *minimumCompatibleVersion = [versionInfo objectForKey:@"MinimumCompatiblePlugInBundleVersion"])
97 [bundleIDToMinimumCompatibleVersion setObject:minimumCompatibleVersion forKey:bundleID];
98
99 if (NSNumber *updateAvailable = [versionInfo objectForKey:@"PlugInUpdateAvailable"]) {
100 // A missing PlugInUpdateAvailable key means that there is a plug-in update available.
101 if (!updateAvailable || [updateAvailable boolValue])
102 [bundleIDsWithAvailableUpdates addObject:bundleID];
103 }
104 }
105 }
106
107 return std::unique_ptr<PluginBlocklist>(new PluginBlocklist(bundleIDToMinimumSecureVersion, bundleIDToMinimumCompatibleVersion, bundleIDToBlockedVersions, bundleIDsWithAvailableUpdates));
108}

114NSArray *PluginBlocklist::splitOSVersion(NSString *osVersion)
115{
116 NSArray *components = [osVersion componentsSeparatedByString:@"."];
117
118 NSMutableArray *result = [NSMutableArray array];
119
120 for (NSUInteger i = 0; i < [components count]; ++i) {
121 NSString *versionString = [[components subarrayWithRange:NSMakeRange(0, i + 1)] componentsJoinedByString:@"."];
122
123 [result addObject:versionString];
124 }
125
126 return result;
127}

130PluginBlocklist::LoadPolicy PluginBlocklist::loadPolicyForPlugin(NSString *bundleIdentifier, NSString *bundleVersionString) const
131{
132 if (!bundleIdentifier || !bundleVersionString)
133 return LoadPolicy::LoadNormally;
134
135 // First, check for explicitly blocked versions.
136 for (NSString *blockedVersion in [m_bundleIDToBlockedVersions objectForKey:bundleIdentifier]) {
137 if ([blockedVersion isEqualToString:bundleVersionString])
138 return LoadPolicy::BlockedForSecurity;
139 }
140
141 // Then, check if there's a forced minimum version for security issues.
142 if (NSString *minimumSecureVersion = [m_bundleIDToMinimumSecureVersion objectForKey:bundleIdentifier]) {
143 if ([bundleVersionString compare:minimumSecureVersion options:NSNumericSearch] == NSOrderedAscending)
144 return LoadPolicy::BlockedForSecurity;
145 }
146
147 // Then, check if there's a forced minimum version for compatibility issues.
148 if (NSString *minimumCompatibleVersion = [m_bundleIDToMinimumCompatibleVersion objectForKey:bundleIdentifier]) {
149 if ([bundleVersionString compare:minimumCompatibleVersion options:NSNumericSearch] == NSOrderedAscending)
150 return LoadPolicy::BlockedForCompatibility;
151 }
152
153 return LoadPolicy::LoadNormally;
154}

156bool PluginBlocklist::isUpdateAvailable(NSString *bundleIdentifier) const
157{
158 return [m_bundleIDsWithAvailableUpdates containsObject:bundleIdentifier];
159}
