28#if PLATFORM(MAC)
29#import "PowerObserverMac.h"

31namespace WebCore {
32
33PowerObserver::PowerObserver(WTF::Function<void()>&& powerOnHander)
34 : m_powerOnHander(WTFMove(powerOnHander))
35 , m_powerConnection(0)
36 , m_notificationPort(nullptr)
37 , m_notifierReference(0)
38 , m_dispatchQueue(adoptOSObject(dispatch_queue_create("com.apple.WebKit.PowerObserver", 0)))
39{
40 m_powerConnection = IORegisterForSystemPower(this, &m_notificationPort, [](void* context, io_service_t service, uint32_t messageType, void* messageArgument) {
41 static_cast<PowerObserver*>(context)->didReceiveSystemPowerNotification(service, messageType, messageArgument);
42 }, &m_notifierReference);
43 if (!m_powerConnection)
44 return;
45
46 IONotificationPortSetDispatchQueue(m_notificationPort, m_dispatchQueue.get());
47}
48
49PowerObserver::~PowerObserver()
50{
51 if (!m_powerConnection)
52 return;
53
54 IODeregisterForSystemPower(&m_notifierReference);
55 IOServiceClose(m_powerConnection);
56 IONotificationPortDestroy(m_notificationPort);
57}
58
59void PowerObserver::didReceiveSystemPowerNotification(io_service_t, uint32_t messageType, void* messageArgument)
60{
61 IOAllowPowerChange(m_powerConnection, reinterpret_cast<long>(messageArgument));
62
63 // We only care about the "wake from sleep" message.
64 if (messageType != kIOMessageSystemWillPowerOn)
65 return;
66
67 // We need to restart the timer on the main thread.
68 CFRunLoopPerformBlock(CFRunLoopGetMain(), kCFRunLoopCommonModes, ^() {
69 m_powerOnHander();
70 });
71}
