233- (NSPoint)scrollerImpPair:(NSScrollerImpPair *)scrollerImpPair convertContentPoint:(NSPoint)pointInContentArea toScrollerImp:(NSScrollerImp *)scrollerImp
234{
235 UNUSED_PARAM(scrollerImpPair);
236
237 if (!_scrollableArea || !scrollerImp)
238 return NSZeroPoint;
239
240 WebCore::Scrollbar* scrollbar = 0;
241 if ([scrollerImp isHorizontal])
242 scrollbar = _scrollableArea->horizontalScrollbar();
243 else
244 scrollbar = _scrollableArea->verticalScrollbar();
245
246 // It is possible to have a null scrollbar here since it is possible for this delegate
247 // method to be called between the moment when a scrollbar has been set to 0 and the
248 // moment when its destructor has been called. We should probably de-couple some
249 // of the clean-up work in ScrollbarThemeMac::unregisterScrollbar() to avoid this
250 // issue.
251 if (!scrollbar)
252 return NSZeroPoint;
253
254 ASSERT(scrollerImp == scrollerImpForScrollbar(*scrollbar));
255
256 return scrollbar->convertFromContainingView(WebCore::roundedIntPoint(pointInContentArea));
257}

323@implementation WebScrollbarPartAnimation
324
325- (id)initWithScrollbar:(Scrollbar*)scrollbar featureToAnimate:(FeatureToAnimate)featureToAnimate animateFrom:(CGFloat)startValue animateTo:(CGFloat)endValue duration:(NSTimeInterval)duration
326{
327 self = [super init];
328 if (!self)
329 return nil;
330
331 const NSTimeInterval timeInterval = 0.01;
332 _timer = adoptNS([[NSTimer alloc] initWithFireDate:[NSDate dateWithTimeIntervalSinceNow:0] interval:timeInterval target:self selector:@selector(setCurrentProgress:) userInfo:nil repeats:YES]);
333 _duration = duration;
334 _timingFunction = CubicBezierTimingFunction::create(CubicBezierTimingFunction::EaseInOut);
335
336 LOG_WITH_STREAM(OverlayScrollbars, stream << "Creating WebScrollbarPartAnimation for " << featureToAnimate << " from " << startValue << " to " << endValue);
337
338 _scrollbar = scrollbar;
339 _featureToAnimate = featureToAnimate;
340 _startValue = startValue;
341 _endValue = endValue;
342
343 return self;
344}
345
346- (void)startAnimation
347{
348 ASSERT(_scrollbar);
349
350 _scrollerImp = scrollerImpForScrollbar(*_scrollbar);
351
352 LOG_WITH_STREAM(OverlayScrollbars, stream << "-[WebScrollbarPartAnimation " << self << "startAnimation] for " << _featureToAnimate);
353
354 [[NSRunLoop mainRunLoop] addTimer:_timer.get() forMode:NSDefaultRunLoopMode];
355 _startDate = adoptNS([[NSDate alloc] initWithTimeIntervalSinceNow:0]);
356}
357
358- (void)setStartValue:(CGFloat)startValue
359{
360 _startValue = startValue;
361}
362
363- (void)setEndValue:(CGFloat)endValue
364{
365 _endValue = endValue;
366}
367
368- (void)setCurrentProgress:(NSTimer *)timer
369{
370 CGFloat progress = 0;
371 NSDate *now = [NSDate dateWithTimeIntervalSinceNow:0];
372 NSTimeInterval elapsed = [now timeIntervalSinceDate:_startDate.get()];
373 if (elapsed > _duration) {
374 progress = 1;
375 [timer invalidate];
376 } else {
377 NSTimeInterval t = 1;
378 if (_duration)
379 t = elapsed / _duration;
380 progress = _timingFunction->transformTime(t, _duration);
381 }
382 ASSERT(_scrollbar);
383
384 LOG_WITH_STREAM(OverlayScrollbars, stream << "-[WebScrollbarPartAnimation " << self << "setCurrentProgress:" << progress <<"] for " << _featureToAnimate);
385
386 CGFloat currentValue;
387 if (_startValue > _endValue)
388 currentValue = 1 - progress;
389 else
390 currentValue = progress;
391
392 switch (_featureToAnimate) {
393 case ThumbAlpha:
394 [_scrollerImp setKnobAlpha:currentValue];
395 break;
396 case TrackAlpha:
397 [_scrollerImp setTrackAlpha:currentValue];
398 break;
399 case UIStateTransition:
400 [_scrollerImp setUiStateTransitionProgress:currentValue];
401 break;
402 case ExpansionTransition:
403 [_scrollerImp setExpansionTransitionProgress:currentValue];
404 break;
405 }
406
407 if (!_scrollbar->supportsUpdateOnSecondaryThread())
408 _scrollbar->invalidate();
409}
410
411- (void)invalidate
412{
413 BEGIN_BLOCK_OBJC_EXCEPTIONS
414 [self stopAnimation];
415 END_BLOCK_OBJC_EXCEPTIONS
416 _scrollbar = 0;
417}
418
419- (void)setDuration:(NSTimeInterval)duration
420{
421 _duration = duration;
422}
423
424- (void)stopAnimation
425{
426 [_timer invalidate];
427}

546- (void)setUpAlphaAnimation:(RetainPtr<WebScrollbarPartAnimation>&)scrollbarPartAnimation scrollerPainter:(NSScrollerImp *)scrollerPainter part:(WebCore::ScrollbarPart)part animateAlphaTo:(CGFloat)newAlpha duration:(NSTimeInterval)duration
547{
548 // If the user has scrolled the page, then the scrollbars must be animated here.
549 // This overrides the early returns.
550 bool mustAnimate = [self scrollAnimator]->haveScrolledSincePageLoad();
551
552 if ([self scrollAnimator]->scrollbarPaintTimerIsActive() && !mustAnimate)
553 return;
554
555 if (_scrollbar->scrollableArea().shouldSuspendScrollAnimations() && !mustAnimate) {
556 [self scrollAnimator]->startScrollbarPaintTimer();
557 return;
558 }
559
560 // At this point, we are definitely going to animate now, so stop the timer.
561 [self scrollAnimator]->stopScrollbarPaintTimer();
562
563 // If we are currently animating, stop
564 if (scrollbarPartAnimation) {
565 [scrollbarPartAnimation stopAnimation];
566 scrollbarPartAnimation = nil;
567 }
568
569 if (ScrollbarThemeMac* macTheme = macScrollbarTheme())
570 macTheme->setPaintCharacteristicsForScrollbar(*_scrollbar);
571
572 if (part == WebCore::ThumbPart && _scrollbar->orientation() == VerticalScrollbar) {
573 if (newAlpha == 1) {
574 IntRect thumbRect = IntRect([scrollerPainter rectForPart:NSScrollerKnob]);
575 [self scrollAnimator]->setVisibleScrollerThumbRect(thumbRect);
576 } else
577 [self scrollAnimator]->setVisibleScrollerThumbRect(IntRect());
578 }
579
580 scrollbarPartAnimation = adoptNS([[WebScrollbarPartAnimation alloc] initWithScrollbar:_scrollbar
581 featureToAnimate:part == ThumbPart ? ThumbAlpha : TrackAlpha
582 animateFrom:part == ThumbPart ? [scrollerPainter knobAlpha] : [scrollerPainter trackAlpha]
583 animateTo:newAlpha
584 duration:duration]);
585 [scrollbarPartAnimation startAnimation];
586}

588- (void)scrollerImp:(NSScrollerImp *)scrollerImp animateKnobAlphaTo:(CGFloat)newKnobAlpha duration:(NSTimeInterval)duration
589{
590 if (!_scrollbar)
591 return;
592
593 ASSERT(scrollerImp == scrollerImpForScrollbar(*_scrollbar));
594
595 NSScrollerImp *scrollerPainter = (NSScrollerImp *)scrollerImp;
596 if (![self scrollAnimator]->scrollbarsCanBeActive()) {
597 [scrollerImp setKnobAlpha:0];
598 _scrollbar->invalidate();
599 return;
600 }
601
602 // If we are fading the scrollbar away, that is a good indication that we are no longer going to
603 // be moving it around on the scrolling thread. Calling [scrollerPainter setUsePresentationValue:NO]
604 // will pass that information on to the NSScrollerImp API.
605 if (newKnobAlpha == 0 && _scrollbar->supportsUpdateOnSecondaryThread())
606 [scrollerPainter setUsePresentationValue:NO];
607
608 [self setUpAlphaAnimation:_knobAlphaAnimation scrollerPainter:scrollerPainter part:WebCore::ThumbPart animateAlphaTo:newKnobAlpha duration:duration];
609}

611- (void)scrollerImp:(NSScrollerImp *)scrollerImp animateTrackAlphaTo:(CGFloat)newTrackAlpha duration:(NSTimeInterval)duration
612{
613 if (!_scrollbar)
614 return;
615
616 ASSERT(scrollerImp == scrollerImpForScrollbar(*_scrollbar));

619 [self setUpAlphaAnimation:_trackAlphaAnimation scrollerPainter:scrollerPainter part:WebCore::BackTrackPart animateAlphaTo:newTrackAlpha duration:duration];
620}

622- (void)scrollerImp:(NSScrollerImp *)scrollerImp animateUIStateTransitionWithDuration:(NSTimeInterval)duration
623{
624 if (!_scrollbar)
625 return;
626
627 ASSERT(scrollerImp == scrollerImpForScrollbar(*_scrollbar));
628
629 // UIStateTransition always animates to 1. In case an animation is in progress this avoids a hard transition.
630 [scrollerImp setUiStateTransitionProgress:1 - [scrollerImp uiStateTransitionProgress]];
631
632 // If the UI state transition is happening, then we are no longer moving the scrollbar on the scrolling thread.
633 if (_scrollbar->supportsUpdateOnSecondaryThread())
634 [scrollerImp setUsePresentationValue:NO];
635
636 if (!_uiStateTransitionAnimation)
637 _uiStateTransitionAnimation = adoptNS([[WebScrollbarPartAnimation alloc] initWithScrollbar:_scrollbar
638 featureToAnimate:UIStateTransition
639 animateFrom:[scrollerImp uiStateTransitionProgress]
640 animateTo:1.0
641 duration:duration]);
642 else {
643 // If we don't need to initialize the animation, just reset the values in case they have changed.
644 [_uiStateTransitionAnimation setStartValue:[scrollerImp uiStateTransitionProgress]];
645 [_uiStateTransitionAnimation setEndValue:1.0];
646 [_uiStateTransitionAnimation setDuration:duration];
647 }
648 [_uiStateTransitionAnimation startAnimation];
649}

658 // ExpansionTransition always animates to 1. In case an animation is in progress this avoids a hard transition.

736#if ENABLE(RUBBER_BANDING)
737static bool rubberBandingEnabledForSystem()
738{
739 static bool initialized = false;
740 static bool enabled = true;
741 // Caches the result, which is consistent with other apps like the Finder, which all
742 // require a restart after changing this default.
743 if (!initialized) {
744 // Uses -objectForKey: and not -boolForKey: in order to default to true if the value wasn't set.

754bool ScrollAnimatorMac::scroll(ScrollbarOrientation orientation, ScrollGranularity granularity, float step, float multiplier, OptionSet<ScrollBehavior> behavior)
755{
756 m_haveScrolledSincePageLoad = true;
757
758 // This method doesn't do directional snapping, but our base class does. It will call into
759 // ScrollAnimatorMac::scroll again with the snapped positions and ScrollBehavior::Default.
760 if (behavior.contains(ScrollBehavior::DoDirectionalSnapping))
761 return ScrollAnimator::scroll(orientation, granularity, step, multiplier, behavior);
762
763 bool shouldAnimate = scrollAnimationEnabledForSystem() && m_scrollableArea.scrollAnimatorEnabled() && granularity != ScrollByPixel
764 && !behavior.contains(ScrollBehavior::NeverAnimate);
765 FloatPoint newPosition = this->currentPosition() + deltaFromStep(orientation, step, multiplier);
766 newPosition = newPosition.constrainedBetween(scrollableArea().minimumScrollPosition(), scrollableArea().maximumScrollPosition());
767
768 LOG_WITH_STREAM(Scrolling, stream << "ScrollAnimatorMac::scroll from " << currentPosition() << " to " << newPosition);
769
770 if (!shouldAnimate)
771 return scrollToPositionWithoutAnimation(newPosition);
772
773 if ([m_scrollAnimationHelper _isAnimating]) {
774 NSPoint targetOrigin = [m_scrollAnimationHelper targetOrigin];
775 if (orientation == HorizontalScrollbar)
776 newPosition.setY(targetOrigin.y);
777 else
778 newPosition.setX(targetOrigin.x);
779 }
780
781 LOG_WITH_STREAM(Scrolling, stream << "ScrollAnimatorMac::scroll " << " from " << currentPosition() << " to " << newPosition);
782 [m_scrollAnimationHelper scrollToPoint:newPosition];
783 return true;
784}

1201void ScrollAnimatorMac::handleWheelEventPhase(PlatformWheelEventPhase phase)
1202{
1203 LOG_WITH_STREAM(OverlayScrollbars, stream << "ScrollAnimatorMac " << this << " scrollableArea " << m_scrollableArea << " handleWheelEventPhase " << phase);
1204
1205 // This may not have been set to true yet if the wheel event was handled by the ScrollingTree,
1206 // So set it to true here.
1207 m_haveScrolledSincePageLoad = true;
1208
1209 // FIXME: Need to ensure we get PlatformWheelEventPhase::Ended.
1210 if (phase == PlatformWheelEventPhase::Began)
1211 didBeginScrollGesture();
1212 else if (phase == PlatformWheelEventPhase::Ended || phase == PlatformWheelEventPhase::Cancelled)
1213 didEndScrollGesture();
1214 else if (phase == PlatformWheelEventPhase::MayBegin)
1215 mayBeginScrollGesture();
1216}

1220bool ScrollAnimatorMac::shouldForwardWheelEventsToParent(const PlatformWheelEvent& wheelEvent) const
1221{
1222 if (std::abs(wheelEvent.deltaY()) >= std::abs(wheelEvent.deltaX()))
1223 return !allowsVerticalStretching(wheelEvent);

1228bool ScrollAnimatorMac::handleWheelEvent(const PlatformWheelEvent& wheelEvent)
1229{
1230 m_haveScrolledSincePageLoad = true;
1231
1232 if (!wheelEvent.hasPreciseScrollingDeltas() || !rubberBandingEnabledForSystem())
1233 return ScrollAnimator::handleWheelEvent(wheelEvent);
1234
1235 m_scrollController.updateGestureInProgressState(wheelEvent);
1236
1237 // FIXME: This is somewhat roundabout hack to allow forwarding wheel events
1238 // up to the parent scrollable area. It takes advantage of the fact that
1239 // the base class implementation of handleWheelEvent will not accept the
1240 // wheel event if there is nowhere to scroll.
1241 if (shouldForwardWheelEventsToParent(wheelEvent)) {
1242 bool didHandleEvent = ScrollAnimator::handleWheelEvent(wheelEvent);
1243 if (didHandleEvent || (!wheelEvent.deltaX() && !wheelEvent.deltaY()))
1244 handleWheelEventPhase(wheelEvent.phase());
1245 return didHandleEvent;
1246 }
1247
1248 bool didHandleEvent = m_scrollController.handleWheelEvent(wheelEvent);
1249
1250 if (didHandleEvent)
1251 handleWheelEventPhase(wheelEvent.phase());
1252
1253 return didHandleEvent;
1254}

1286bool ScrollAnimatorMac::allowsVerticalStretching(const PlatformWheelEvent& wheelEvent) const
1287{
1288 switch (m_scrollableArea.verticalScrollElasticity()) {
1289 case ScrollElasticityAutomatic: {
1290 Scrollbar* hScroller = m_scrollableArea.horizontalScrollbar();
1291 Scrollbar* vScroller = m_scrollableArea.verticalScrollbar();
1292 bool scrollbarsAllowStretching = ((vScroller && vScroller->enabled()) || (!hScroller || !hScroller->enabled()));
1293 bool eventPreventsStretching = m_scrollableArea.hasScrollableOrRubberbandableAncestor() && wheelEvent.isGestureStart() && m_scrollableArea.isPinnedForScrollDeltaOnAxis(-wheelEvent.deltaY(), ScrollEventAxis::Vertical);
1294 if (!eventPreventsStretching)
1295 eventPreventsStretching = gestureShouldBeginSnap(wheelEvent, ScrollEventAxis::Vertical, m_scrollableArea.snapOffsetsInfo());
1296 return scrollbarsAllowStretching && !eventPreventsStretching;
1297 }
1298 case ScrollElasticityNone:
1299 return false;
1300 case ScrollElasticityAllowed:
1301 return true;
1302 }
1303
1304 ASSERT_NOT_REACHED();
1305 return false;
1306}

1371void ScrollAnimatorMac::updateScrollerStyle()
1372{
1373 if ([m_scrollerImpPair overlayScrollerStateIsLocked]) {
1374 m_needsScrollerStyleUpdate = true;
1375 return;
1376 }
1377
1378 ScrollbarThemeMac* macTheme = macScrollbarTheme();
1379 if (!macTheme) {
1380 m_needsScrollerStyleUpdate = false;
1381 return;
1382 }
1383
1384 macTheme->usesOverlayScrollbarsChanged();
1385
1386 NSScrollerStyle newStyle = [m_scrollerImpPair scrollerStyle];
1387
1388 if (Scrollbar* verticalScrollbar = scrollableArea().verticalScrollbar()) {
1389 verticalScrollbar->invalidate();
1390
1391 NSScrollerImp *oldVerticalPainter = [m_scrollerImpPair verticalScrollerImp];
1392 auto newVerticalPainter = retainPtr([NSScrollerImp scrollerImpWithStyle:newStyle controlSize:(NSControlSize)verticalScrollbar->controlSize() horizontal:NO replacingScrollerImp:oldVerticalPainter]);
1393
1394 [m_scrollerImpPair setVerticalScrollerImp:newVerticalPainter.get()];
1395 macTheme->setNewPainterForScrollbar(*verticalScrollbar, WTFMove(newVerticalPainter));
1396 macTheme->didCreateScrollerImp(*verticalScrollbar);
1397
1398 // The different scrollbar styles have different thicknesses, so we must re-set the
1399 // frameRect to the new thickness, and the re-layout below will ensure the position
1400 // and length are properly updated.
1401 int thickness = macTheme->scrollbarThickness(verticalScrollbar->controlSize());
1402 verticalScrollbar->setFrameRect(IntRect(0, 0, thickness, thickness));
1403 }
1404
1405 if (Scrollbar* horizontalScrollbar = scrollableArea().horizontalScrollbar()) {
1406 horizontalScrollbar->invalidate();
1407
1408 NSScrollerImp *oldHorizontalPainter = [m_scrollerImpPair horizontalScrollerImp];
1409 auto newHorizontalPainter = retainPtr([NSScrollerImp scrollerImpWithStyle:newStyle controlSize:(NSControlSize)horizontalScrollbar->controlSize() horizontal:YES replacingScrollerImp:oldHorizontalPainter]);
1410
1411 [m_scrollerImpPair setHorizontalScrollerImp:newHorizontalPainter.get()];
1412 macTheme->setNewPainterForScrollbar(*horizontalScrollbar, WTFMove(newHorizontalPainter));
1413 macTheme->didCreateScrollerImp(*horizontalScrollbar);
1414
1415 // The different scrollbar styles have different thicknesses, so we must re-set the
1416 // frameRect to the new thickness, and the re-layout below will ensure the position
1417 // and length are properly updated.
1418 int thickness = macTheme->scrollbarThickness(horizontalScrollbar->controlSize());
1419 horizontalScrollbar->setFrameRect(IntRect(0, 0, thickness, thickness));
1420 }
1421
1422 // If m_needsScrollerStyleUpdate is true, then the page is restoring from the back/forward cache, and
1423 // a relayout will happen on its own. Otherwise, we must initiate a re-layout ourselves.
1424 scrollableArea().scrollbarStyleChanged(newStyle == NSScrollerStyleOverlay ? ScrollbarStyle::Overlay : ScrollbarStyle::AlwaysVisible, !m_needsScrollerStyleUpdate);
1425
1426 m_needsScrollerStyleUpdate = false;
1427}
