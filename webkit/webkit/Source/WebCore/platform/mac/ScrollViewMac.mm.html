50inline NSScrollView<WebCoreFrameScrollView> *ScrollView::scrollView() const

65void ScrollView::platformAddChild(Widget* child)
66{
67 BEGIN_BLOCK_OBJC_EXCEPTIONS
68 NSView *parentView = documentView();
69 NSView *childView = child->getOuterView();
70 ASSERT(![parentView isDescendantOf:childView]);
71
72 // Suppress the resetting of drag margins since we know we can't affect them.
73 NSWindow *window = [parentView window];
74 BOOL resetDragMargins = [window _needsToResetDragMargins];
75 [window _setNeedsToResetDragMargins:NO];
76 if ([childView superview] != parentView)
77 [parentView addSubview:childView];
78 [window _setNeedsToResetDragMargins:resetDragMargins];
79 END_BLOCK_OBJC_EXCEPTIONS
80}

126void ScrollView::platformSetTopContentInset(float topContentInset)
127{
128 BEGIN_BLOCK_OBJC_EXCEPTIONS
129 if (topContentInset)
130 scrollView().automaticallyAdjustsContentInsets = NO;
131 else
132 scrollView().automaticallyAdjustsContentInsets = YES;

140IntRect ScrollView::platformVisibleContentRect(bool includeScrollbars) const
141{
142 BEGIN_BLOCK_OBJC_EXCEPTIONS
143 IntRect visibleContentRect = platformVisibleContentRectIncludingObscuredArea(includeScrollbars);
144
145 visibleContentRect.move(scrollView().contentInsets.left, scrollView().contentInsets.top);
146 visibleContentRect.contract(scrollView().contentInsets.left + scrollView().contentInsets.right, scrollView().contentInsets.top + scrollView().contentInsets.bottom);
147
148 return visibleContentRect;
149 END_BLOCK_OBJC_EXCEPTIONS
150
151 return IntRect();
152}

159IntRect ScrollView::platformVisibleContentRectIncludingObscuredArea(bool includeScrollbars) const
160{
161 BEGIN_BLOCK_OBJC_EXCEPTIONS
162 IntRect visibleContentRectIncludingObscuredArea = enclosingIntRect([scrollView() documentVisibleRect]);
163
164 if (includeScrollbars) {

203void ScrollView::platformSetScrollPosition(const IntPoint& scrollPoint)
204{
205 BEGIN_BLOCK_OBJC_EXCEPTIONS
206 NSPoint floatPoint = scrollPoint;
207 NSPoint tempPoint = { std::max(-[scrollView() scrollOrigin].x, floatPoint.x), std::max(-[scrollView() scrollOrigin].y, floatPoint.y) }; // Don't use NSMakePoint to work around 4213314.
208
209 // AppKit has the inset factored into all of its scroll positions. In WebCore, we use positions that ignore
210 // the insets so that they are equivalent whether or not there is an inset.
211 tempPoint.x = tempPoint.x - scrollView().contentInsets.left;
212 tempPoint.y = tempPoint.y - scrollView().contentInsets.top;
213
214 [documentView() scrollPoint:tempPoint];
215 END_BLOCK_OBJC_EXCEPTIONS
216}

234// "Containing Window" means the NSWindow's coord system, which is origin lower left
235
236IntRect ScrollView::platformContentsToScreen(const IntRect& rect) const
237{
238 BEGIN_BLOCK_OBJC_EXCEPTIONS
239 if (NSView* documentView = this->documentView()) {
240 NSRect tempRect = rect;
241 tempRect = [documentView convertRect:tempRect toView:nil];
242 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
243 tempRect.origin = [[documentView window] convertBaseToScreen:tempRect.origin];
244 ALLOW_DEPRECATED_DECLARATIONS_END
245 return enclosingIntRect(tempRect);
246 }
247 END_BLOCK_OBJC_EXCEPTIONS
248 return IntRect();
249}

251IntPoint ScrollView::platformScreenToContents(const IntPoint& point) const
252{
253 BEGIN_BLOCK_OBJC_EXCEPTIONS
254 if (NSView* documentView = this->documentView()) {
255 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
256 NSPoint windowCoord = [[documentView window] convertScreenToBase: point];
257 ALLOW_DEPRECATED_DECLARATIONS_END
258 return IntPoint([documentView convertPoint:windowCoord fromView:nil]);
259 }
260 END_BLOCK_OBJC_EXCEPTIONS
261 return IntPoint();
262}
