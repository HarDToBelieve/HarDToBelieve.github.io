32namespace WebCore {
33
34static String wildcardRegexPatternString(const String& string)
35{
36 String metaCharacters = ".|+?()[]{}^$";
37 UChar escapeCharacter = '\\';
38 UChar wildcardCharacter = '*';
39
40 StringBuilder stringBuilder;
41
42 stringBuilder.append('^');
43 for (unsigned i = 0; i < string.length(); i++) {
44 auto character = string[i];
45 if (metaCharacters.contains(character) || character == escapeCharacter)
46 stringBuilder.append(escapeCharacter);
47 else if (character == wildcardCharacter)
48 stringBuilder.append('.');
49
50 stringBuilder.append(character);
51 }
52 stringBuilder.append('$');
53
54 return stringBuilder.toString();
55}
56
57bool stringMatchesWildcardString(const String& string, const String& wildcardString)
58{
59 return JSC::Yarr::RegularExpression(wildcardRegexPatternString(wildcardString), JSC::Yarr::TextCaseInsensitive).match(string) != -1;
60}
