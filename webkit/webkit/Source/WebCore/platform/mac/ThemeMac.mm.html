137static NSControlSize controlSizeForFont(const FontCascade& font)
138{
139 int fontSize = font.pixelSize();
140#if HAVE(LARGE_CONTROL_SIZE)
141 if (fontSize >= 21 && ThemeMac::supportsLargeFormControls())
142 return NSControlSizeLarge;
143#endif
144 if (fontSize >= 16)
145 return NSControlSizeRegular;
146 if (fontSize >= 11)
147 return NSControlSizeSmall;
148 return NSControlSizeMini;
149}
150
151static LengthSize sizeFromNSControlSize(NSControlSize nsControlSize, const LengthSize& zoomedSize, float zoomFactor, const std::array<IntSize, 4>& sizes)
152{
153 IntSize controlSize = sizes[nsControlSize];
154 if (zoomFactor != 1.0f)
155 controlSize = IntSize(controlSize.width() * zoomFactor, controlSize.height() * zoomFactor);
156 LengthSize result = zoomedSize;
157 if (zoomedSize.width.isIntrinsicOrAuto() && controlSize.width() > 0)
158 result.width = { controlSize.width(), LengthType::Fixed };
159 if (zoomedSize.height.isIntrinsicOrAuto() && controlSize.height() > 0)
160 result.height = { controlSize.height(), LengthType::Fixed };
161 return result;
162}
163
164static LengthSize sizeFromFont(const FontCascade& font, const LengthSize& zoomedSize, float zoomFactor, const std::array<IntSize, 4>& sizes)
165{
166 return sizeFromNSControlSize(controlSizeForFont(font), zoomedSize, zoomFactor, sizes);
167}

169static NSControlSize controlSizeFromPixelSize(const std::array<IntSize, 4>& sizes, const IntSize& minZoomedSize, float zoomFactor)
170{
171#if HAVE(LARGE_CONTROL_SIZE)
172 if (ThemeMac::supportsLargeFormControls()
173 && minZoomedSize.width() >= static_cast<int>(sizes[NSControlSizeLarge].width() * zoomFactor)
174 && minZoomedSize.height() >= static_cast<int>(sizes[NSControlSizeLarge].height() * zoomFactor))
175 return NSControlSizeLarge;
176#endif
177 if (minZoomedSize.width() >= static_cast<int>(sizes[NSControlSizeRegular].width() * zoomFactor)
178 && minZoomedSize.height() >= static_cast<int>(sizes[NSControlSizeRegular].height() * zoomFactor))
179 return NSControlSizeRegular;
180 if (minZoomedSize.width() >= static_cast<int>(sizes[NSControlSizeSmall].width() * zoomFactor)
181 && minZoomedSize.height() >= static_cast<int>(sizes[NSControlSizeSmall].height() * zoomFactor))
182 return NSControlSizeSmall;
183 return NSControlSizeMini;
184}

186static void setControlSize(NSCell* cell, const std::array<IntSize, 4>& sizes, const IntSize& minZoomedSize, float zoomFactor)
187{
188 auto size = controlSizeFromPixelSize(sizes, minZoomedSize, zoomFactor);
189 if (size != [cell controlSize]) // Only update if we have to, since AppKit does work even if the size is the same.
190 [cell setControlSize:size];
191}

193static void updateStates(NSCell* cell, const ControlStates& controlStates, bool useAnimation = false)
194{
195 // The animated state cause this thread to start and stop repeatedly on CoreAnimation synchronize calls.
196 // This short burts of activity in between are not long enough for VoiceOver to retrieve accessibility attributes and makes the process appear unresponsive.
197 if (AXObjectCache::accessibilityEnhancedUserInterfaceEnabled())
198 useAnimation = false;
199
200 auto states = controlStates.states();
201
202 // Hover state is not supported by Aqua.
203
204 // Pressed state
205 bool oldPressed = [cell isHighlighted];
206 bool pressed = states.contains(ControlStates::States::Pressed);
207 if (pressed != oldPressed) {
208 [(NSButtonCell*)cell _setHighlighted:pressed animated:useAnimation];
209 }
210
211 // Enabled state
212 bool oldEnabled = [cell isEnabled];
213 bool enabled = states.contains(ControlStates::States::Enabled);
214 if (enabled != oldEnabled)
215 [cell setEnabled:enabled];
216
217 // Checked and Indeterminate
218 bool oldIndeterminate = [cell state] == NSControlStateValueMixed;
219 bool indeterminate = states.contains(ControlStates::States::Indeterminate);
220 bool checked = states.contains(ControlStates::States::Checked);
221 bool oldChecked = [cell state] == NSControlStateValueOn;
222 if (oldIndeterminate != indeterminate || checked != oldChecked) {
223 NSControlStateValue newState = indeterminate ? NSControlStateValueMixed : (checked ? NSControlStateValueOn : NSControlStateValueOff);
224 [(NSButtonCell*)cell _setState:newState animated:useAnimation];
225 }
226
227 // Presenting state
228 if (states.contains(ControlStates::States::Presenting))
229 [(NSButtonCell*)cell _setHighlighted:YES animated:NO];
230
231 // Window inactive state does not need to be checked explicitly, since we paint parented to
232 // a view in a window whose key state can be detected.
233}

235static FloatRect inflateRect(const FloatRect& zoomedRect, const IntSize& zoomedSize, const int* margins, float zoomFactor)
236{
237 // Only do the inflation if the available width/height are too small.
238 // Otherwise try to fit the glow/check space into the available box's width/height.
239 int widthDelta = zoomedRect.width() - (zoomedSize.width() + margins[leftMargin] * zoomFactor + margins[rightMargin] * zoomFactor);
240 int heightDelta = zoomedRect.height() - (zoomedSize.height() + margins[topMargin] * zoomFactor + margins[bottomMargin] * zoomFactor);
241 FloatRect result(zoomedRect);
242 if (widthDelta < 0) {
243 result.setX(result.x() - margins[leftMargin] * zoomFactor);
244 result.setWidth(result.width() - widthDelta);
245 }
246 if (heightDelta < 0) {
247 result.setY(result.y() - margins[topMargin] * zoomFactor);
248 result.setHeight(result.height() - heightDelta);
249 }
250 return result;
251}

261static const int* checkboxMargins(NSControlSize controlSize)
262{
263 static const int margins[4][4] =
264 {
265 // top right bottom left
266 { 2, 2, 2, 2 },
267 { 2, 1, 2, 1 },
268 { 0, 0, 1, 0 },
269 { 2, 2, 2, 2 },
270 };
271 return margins[controlSize];
272}

274static LengthSize checkboxSize(const FontCascade& font, const LengthSize& zoomedSize, float zoomFactor)
275{
276 // If the width and height are both specified, then we have nothing to do.
277 if (!zoomedSize.width.isIntrinsicOrAuto() && !zoomedSize.height.isIntrinsicOrAuto())
278 return zoomedSize;
279
280 // Use the font size to determine the intrinsic width of the control.
281 return sizeFromFont(font, zoomedSize, zoomFactor, checkboxSizes());
282}

325static void configureToggleButton(NSCell* cell, ControlPart buttonType, const ControlStates& states, const IntSize& zoomedSize, float zoomFactor, bool isStateChange)
326{
327 // Set the control size based off the rectangle we're painting into.
328 setControlSize(cell, buttonType == CheckboxPart ? checkboxSizes() : radioSizes(), zoomedSize, zoomFactor);
329
330 // Update the various states we respond to.
331 updateStates(cell, states, isStateChange);
332}

334static RetainPtr<NSButtonCell> createToggleButtonCell(ControlPart buttonType)
335{
336 RetainPtr<NSButtonCell> toggleButtonCell = adoptNS([[NSButtonCell alloc] init]);
337
338 if (buttonType == CheckboxPart) {
339 [toggleButtonCell setButtonType:NSButtonTypeSwitch];
340 [toggleButtonCell setAllowsMixedState:YES];
341 } else {
342 ASSERT(buttonType == RadioPart);
343 [toggleButtonCell setButtonType:NSButtonTypeRadio];
344 }
345
346 [toggleButtonCell setTitle:nil];
347 [toggleButtonCell setFocusRingType:NSFocusRingTypeExterior];
348 return toggleButtonCell;

367static bool drawCellFocusRingWithFrameAtTime(NSCell *cell, NSRect cellFrame, NSView *controlView, NSTimeInterval timeOffset)
368{
369 CGContextRef cgContext = [[NSGraphicsContext currentContext] CGContext];
370
371 CGContextStateSaver stateSaver(cgContext);
372
373 CGFocusRingStyle focusRingStyle;
374 bool needsRepaint = NSInitializeCGFocusRingStyleForTime(NSFocusRingOnly, &focusRingStyle, timeOffset);
375
376 // We want to respect the CGContext clipping and also not overpaint any
377 // existing focus ring. The way to do this is set accumulate to
378 // -1. According to CoreGraphics, the reasoning for this behavior has been
379 // lost in time.
380 focusRingStyle.accumulate = -1;
381
382 // FIXME: This color should be shared with RenderThemeMac. For now just use the same NSColor color.
383 // The color is expected to be opaque, since CoreGraphics will apply opacity when drawing (because opacity is normally animated).
384 auto color = colorFromNSColor([NSColor keyboardFocusIndicatorColor]).opaqueColor();
385 auto style = adoptCF(CGStyleCreateFocusRingWithColor(&focusRingStyle, cachedCGColor(color)));
386 CGContextSetStyle(cgContext, style.get());
387
388 CGContextBeginTransparencyLayerWithRect(cgContext, NSRectToCGRect(cellFrame), nullptr);
389 [cell drawFocusRingMaskWithFrame:cellFrame inView:controlView];
390 CGContextEndTransparencyLayer(cgContext);
391
392 return needsRepaint;
393}

395static bool drawCellFocusRing(NSCell *cell, NSRect cellFrame, NSView *controlView)
396{
397 drawCellFocusRingWithFrameAtTime(cell, cellFrame, controlView, std::numeric_limits<double>::max());
398 return false;
399}

401static void paintToggleButton(ControlPart buttonType, ControlStates& controlStates, GraphicsContext& context, const FloatRect& zoomedRect, float zoomFactor, ScrollView* scrollView, float deviceScaleFactor)
402{
403 BEGIN_BLOCK_OBJC_EXCEPTIONS
404
405 RetainPtr<NSButtonCell> toggleButtonCell = static_cast<NSButtonCell *>(controlStates.platformControl());
406 IntSize zoomedRectSize = IntSize(zoomedRect.size());
407
408 if (controlStates.isDirty()) {
409 if (!toggleButtonCell)
410 toggleButtonCell = createToggleButtonCell(buttonType);
411 configureToggleButton(toggleButtonCell.get(), buttonType, controlStates, zoomedRectSize, zoomFactor, true);
412 } else {
413 if (!toggleButtonCell) {
414 if (buttonType == CheckboxPart)
415 toggleButtonCell = sharedCheckboxCell(controlStates, zoomedRectSize, zoomFactor);
416 else {
417 ASSERT(buttonType == RadioPart);
418 toggleButtonCell = sharedRadioCell(controlStates, zoomedRectSize, zoomFactor);
419 }
420 }
421 configureToggleButton(toggleButtonCell.get(), buttonType, controlStates, zoomedRectSize, zoomFactor, false);
422 }
423 controlStates.setDirty(false);
424
425 GraphicsContextStateSaver stateSaver(context);
426
427 NSControlSize controlSize = [toggleButtonCell controlSize];
428 IntSize zoomedSize = buttonType == CheckboxPart ? checkboxSizes()[controlSize] : radioSizes()[controlSize];
429 zoomedSize.setWidth(zoomedSize.width() * zoomFactor);
430 zoomedSize.setHeight(zoomedSize.height() * zoomFactor);
431 const int* controlMargins = buttonType == CheckboxPart ? checkboxMargins(controlSize) : radioMargins(controlSize);
432 FloatRect inflatedRect = inflateRect(zoomedRect, zoomedSize, controlMargins, zoomFactor);
433
434 if (zoomFactor != 1.0f) {
435 inflatedRect.setWidth(inflatedRect.width() / zoomFactor);
436 inflatedRect.setHeight(inflatedRect.height() / zoomFactor);
437 context.translate(inflatedRect.location());
438 context.scale(zoomFactor);
439 context.translate(-inflatedRect.location());
440 }
441 LocalCurrentGraphicsContext localContext(context);
442
443 NSView *view = ThemeMac::ensuredView(scrollView, controlStates, true /* useUnparentedView */);
444
445 bool needsRepaint = false;
446 bool isCellFocused = controlStates.states().contains(ControlStates::States::Focused);
447
448 if ([toggleButtonCell _stateAnimationRunning]) {
449 context.translate(inflatedRect.location());
450 context.scale(FloatSize(1, -1));
451 context.translate(0, -inflatedRect.height());
452
453 [toggleButtonCell _renderCurrentAnimationFrameInContext:context.platformContext() atLocation:NSMakePoint(0, 0)];
454 if (![toggleButtonCell _stateAnimationRunning] && isCellFocused)
455 needsRepaint = ThemeMac::drawCellOrFocusRingWithViewIntoContext(toggleButtonCell.get(), context, inflatedRect, view, false, true, deviceScaleFactor);
456 } else
457 needsRepaint = ThemeMac::drawCellOrFocusRingWithViewIntoContext(toggleButtonCell.get(), context, inflatedRect, view, true, isCellFocused, deviceScaleFactor);
458
459 [toggleButtonCell setControlView:nil];
460
461 needsRepaint |= [toggleButtonCell _stateAnimationRunning];
462 controlStates.setNeedsRepaint(needsRepaint);
463 if (needsRepaint)
464 controlStates.setPlatformControl(toggleButtonCell.get());
465
466 END_BLOCK_OBJC_EXCEPTIONS
467}

478static const int* buttonMargins(NSControlSize controlSize)
479{
480 // FIXME: These values may need to be reevaluated. They appear to have been originally chosen
481 // to reflect the size of shadows around native form controls on macOS, but as of macOS 10.15,
482 // these margins extend well past the boundaries of a native button cell's shadows.
483 static const int margins[4][4] =
484 {
485 { 4, 6, 7, 6 },
486 { 4, 5, 6, 5 },
487 { 0, 1, 1, 1 },
488 { 4, 6, 7, 6 },
489 };
490 return margins[controlSize];
491}

497 auto cell = adoptNS([[NSButtonCell alloc] init]);

505static void setUpButtonCell(NSButtonCell *cell, ControlPart part, const ControlStates& states, const IntSize& zoomedSize, float zoomFactor)
506{
507 // Set the control size based off the rectangle we're painting into.
508 const std::array<IntSize, 4>& sizes = buttonSizes();
509 switch (part) {
510 case SquareButtonPart:
511 [cell setBezelStyle:NSBezelStyleShadowlessSquare];
512 break;
513#if ENABLE(INPUT_TYPE_COLOR)
514 case ColorWellPart:
515 [cell setBezelStyle:NSBezelStyleTexturedSquare];
516 break;
517#endif
518 default:
519#if HAVE(LARGE_CONTROL_SIZE)
520 auto largestControlSize = ThemeMac::supportsLargeFormControls() ? NSControlSizeLarge : NSControlSizeRegular;
521#else
522 auto largestControlSize = NSControlSizeRegular;
523#endif
524 NSBezelStyle style = (zoomedSize.height() > buttonSizes()[largestControlSize].height() * zoomFactor) ? NSBezelStyleShadowlessSquare : NSBezelStyleRounded;
525 [cell setBezelStyle:style];
526 break;
527 }
528
529 setControlSize(cell, sizes, zoomedSize, zoomFactor);
530
531 // Update the various states we respond to.
532 updateStates(cell, states);
533}

535static NSButtonCell *button(ControlPart part, const ControlStates& controlStates, const IntSize& zoomedSize, float zoomFactor)
536{
537 auto states = controlStates.states();
538 NSButtonCell *cell;
539 if (states.contains(ControlStates::States::Default)) {
540 static NeverDestroyed<RetainPtr<NSButtonCell>> defaultCell = buttonCell(DefaultButtonCell);
541 cell = defaultCell.get().get();
542 } else {
543 static NeverDestroyed<RetainPtr<NSButtonCell>> normalCell = buttonCell(NormalButtonCell);
544 cell = normalCell.get().get();
545 }
546 setUpButtonCell(cell, part, controlStates, zoomedSize, zoomFactor);
547 return cell;
548}

550static void paintButton(ControlPart part, ControlStates& controlStates, GraphicsContext& context, const FloatRect& zoomedRect, float zoomFactor, ScrollView* scrollView, float deviceScaleFactor)
551{
552 BEGIN_BLOCK_OBJC_EXCEPTIONS
553
554 // Determine the width and height needed for the control and prepare the cell for painting.
555 auto states = controlStates.states();
556 NSButtonCell *buttonCell = button(part, controlStates, IntSize(zoomedRect.size()), zoomFactor);
557 GraphicsContextStateSaver stateSaver(context);
558
559 NSControlSize controlSize = [buttonCell controlSize];
560 IntSize zoomedSize = buttonSizes()[controlSize];
561 zoomedSize.setWidth(zoomedRect.width()); // Buttons don't ever constrain width, so the zoomed width can just be honored.
562 zoomedSize.setHeight(zoomedSize.height() * zoomFactor);
563 FloatRect inflatedRect = zoomedRect;
564 if ([buttonCell bezelStyle] == NSBezelStyleRounded) {
565 // Center the button within the available space.
566 if (inflatedRect.height() > zoomedSize.height()) {
567 inflatedRect.setY(inflatedRect.y() + (inflatedRect.height() - zoomedSize.height()) / 2);
568 inflatedRect.setHeight(zoomedSize.height());
569 }
570
571 // Now inflate it to account for the shadow.
572 inflatedRect = inflateRect(inflatedRect, zoomedSize, buttonMargins(controlSize), zoomFactor);
573
574 if (zoomFactor != 1.0f) {
575 inflatedRect.setWidth(inflatedRect.width() / zoomFactor);
576 inflatedRect.setHeight(inflatedRect.height() / zoomFactor);
577 context.translate(inflatedRect.location());
578 context.scale(zoomFactor);
579 context.translate(-inflatedRect.location());
580 }
581 }
582
583 LocalCurrentGraphicsContext localContext(context);
584
585 NSView *view = ThemeMac::ensuredView(scrollView, controlStates);
586 NSWindow *window = [view window];
587 NSButtonCell *previousDefaultButtonCell = [window defaultButtonCell];
588
589 bool needsRepaint = ThemeMac::drawCellOrFocusRingWithViewIntoContext(buttonCell, context, inflatedRect, view, true, states.contains(ControlStates::States::Focused), deviceScaleFactor);
590 if (states.contains(ControlStates::States::Default))
591 [window setDefaultButtonCell:buttonCell];
592 else if ([previousDefaultButtonCell isEqual:buttonCell])
593 [window setDefaultButtonCell:nil];
594
595 controlStates.setNeedsRepaint(needsRepaint);
596
597 [buttonCell setControlView:nil];
598
599 if (![previousDefaultButtonCell isEqual:buttonCell])
600 [window setDefaultButtonCell:previousDefaultButtonCell];
601
602 END_BLOCK_OBJC_EXCEPTIONS
603}

629static void paintStepper(ControlStates& controlStates, GraphicsContext& context, const FloatRect& zoomedRect, float zoomFactor, ScrollView*)
630{
631 // We don't use NSStepperCell because there are no ways to draw an
632 // NSStepperCell with the up button highlighted.
633
634 NSString *coreUIState;
635 auto states = controlStates.states();
636 if (!states.contains(ControlStates::States::Enabled))
637 coreUIState = (__bridge NSString *)kCUIStateDisabled;
638 else if (states.contains(ControlStates::States::Pressed))
639 coreUIState = (__bridge NSString *)kCUIStatePressed;
640 else
641 coreUIState = (__bridge NSString *)kCUIStateActive;
642
643 NSString *coreUISize;
644 auto controlSize = controlSizeFromPixelSize(stepperSizes(), IntSize(zoomedRect.size()), zoomFactor);
645 if (controlSize == NSControlSizeMini)
646 coreUISize = (__bridge NSString *)kCUISizeMini;
647 else if (controlSize == NSControlSizeSmall)
648 coreUISize = (__bridge NSString *)kCUISizeSmall;
649 else
650 coreUISize = (__bridge NSString *)kCUISizeRegular;
651
652 IntRect rect(zoomedRect);
653 GraphicsContextStateSaver stateSaver(context);
654 if (zoomFactor != 1.0f) {

719bool ThemeMac::drawCellOrFocusRingWithViewIntoContext(NSCell *cell, GraphicsContext& context, const FloatRect& rect, NSView *view, bool drawButtonCell, bool drawFocusRing, float deviceScaleFactor)
720{
721 ASSERT(drawButtonCell || drawFocusRing);
722 bool needsRepaint = false;
723 auto platformContext = context.platformContext();
724 auto userCTM = AffineTransform(CGAffineTransformConcat(CGContextGetCTM(platformContext), CGAffineTransformInvert(CGContextGetBaseCTM(platformContext))));
725 bool useImageBuffer = userCTM.xScale() != 1.0 || userCTM.yScale() != 1.0;
726
727 if (useImageBuffer) {
728 NSRect imageBufferDrawRect = NSRect(FloatRect(buttonFocusRectOutlineWidth, buttonFocusRectOutlineWidth, rect.width(), rect.height()));
729 auto imageBuffer = ImageBuffer::createCompatibleBuffer(rect.size() + 2 * FloatSize(buttonFocusRectOutlineWidth, buttonFocusRectOutlineWidth), deviceScaleFactor, DestinationColorSpace::SRGB(), context);
730 if (!imageBuffer)
731 return needsRepaint;
732 {
733 LocalCurrentGraphicsContext localContext(imageBuffer->context());
734 needsRepaint = drawCellOrFocusRingIntoRectWithView(cell, imageBufferDrawRect, view, drawButtonCell, drawFocusRing);
735 }
736 context.drawConsumingImageBuffer(WTFMove(imageBuffer), rect.location() - FloatSize(buttonFocusRectOutlineWidth, buttonFocusRectOutlineWidth));
737 return needsRepaint;
738 }
739 if (drawButtonCell)
740 needsRepaint = drawCellOrFocusRingIntoRectWithView(cell, NSRect(rect), view, drawButtonCell, drawFocusRing);
741
742 return needsRepaint;
743}

882void ThemeMac::inflateControlPaintRect(ControlPart part, const ControlStates& states, FloatRect& zoomedRect, float zoomFactor) const
883{
884 BEGIN_BLOCK_OBJC_EXCEPTIONS
885 IntSize zoomRectSize = IntSize(zoomedRect.size());
886 switch (part) {
887 case CheckboxPart: {
888 // We inflate the rect as needed to account for padding included in the cell to accommodate the checkbox
889 // shadow" and the check. We don't consider this part of the bounds of the control in WebKit.
890 NSCell *cell = sharedCheckboxCell(states, zoomRectSize, zoomFactor);
891 NSControlSize controlSize = [cell controlSize];
892 IntSize zoomedSize = checkboxSizes()[controlSize];
893 zoomedSize.setHeight(zoomedSize.height() * zoomFactor);
894 zoomedSize.setWidth(zoomedSize.width() * zoomFactor);
895 zoomedRect = inflateRect(zoomedRect, zoomedSize, checkboxMargins(controlSize), zoomFactor);
896 break;
897 }
898 case RadioPart: {
899 // We inflate the rect as needed to account for padding included in the cell to accommodate the radio button
900 // shadow". We don't consider this part of the bounds of the control in WebKit.
901 NSCell *cell = sharedRadioCell(states, zoomRectSize, zoomFactor);
902 NSControlSize controlSize = [cell controlSize];
903 IntSize zoomedSize = radioSizes()[controlSize];
904 zoomedSize.setHeight(zoomedSize.height() * zoomFactor);
905 zoomedSize.setWidth(zoomedSize.width() * zoomFactor);
906 zoomedRect = inflateRect(zoomedRect, zoomedSize, radioMargins(controlSize), zoomFactor);
907 break;
908 }
909 case PushButtonPart:
910 case DefaultButtonPart:
911 case ButtonPart: {
912 NSButtonCell *cell = button(part, states, zoomRectSize, zoomFactor);
913 NSControlSize controlSize = [cell controlSize];
914
915 // We inflate the rect as needed to account for the Aqua button's shadow.
916 if ([cell bezelStyle] == NSBezelStyleRounded) {
917 IntSize zoomedSize = buttonSizes()[controlSize];
918 zoomedSize.setHeight(zoomedSize.height() * zoomFactor);
919 zoomedSize.setWidth(zoomedRect.width()); // Buttons don't ever constrain width, so the zoomed width can just be honored.
920 zoomedRect = inflateRect(zoomedRect, zoomedSize, buttonMargins(controlSize), zoomFactor);
921 }
922 break;
923 }
924 case InnerSpinButtonPart: {
925 static const int stepperMargin[4] = { 0, 0, 0, 0 };
926 auto controlSize = controlSizeFromPixelSize(stepperSizes(), zoomRectSize, zoomFactor);
927 IntSize zoomedSize = stepperSizes()[controlSize];
928 zoomedSize.setHeight(zoomedSize.height() * zoomFactor);
929 zoomedSize.setWidth(zoomedSize.width() * zoomFactor);
930 zoomedRect = inflateRect(zoomedRect, zoomedSize, stepperMargin, zoomFactor);
931 break;
932 }
933 default:
934 break;
935 }
936 END_BLOCK_OBJC_EXCEPTIONS
937}
