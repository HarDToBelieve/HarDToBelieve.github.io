47namespace WebCore {
48
49static const CGFloat horizontalPadding = 5;
50static const CGFloat verticalPadding = 5;
51static const CGFloat maxLabelWidth = 300;
52
53ValidationBubble::ValidationBubble(NSView* view, const String& message, const Settings& settings)
54 : m_view(view)
55 , m_message(message)
56{
57 RetainPtr<NSViewController> controller = adoptNS([[NSViewController alloc] init]);
58
59 RetainPtr<NSView> popoverView = adoptNS([[NSView alloc] initWithFrame:NSZeroRect]);
60 [controller setView:popoverView.get()];
61
62 RetainPtr<NSTextField> label = adoptNS([[NSTextField alloc] init]);
63 [label setEditable:NO];
64 [label setDrawsBackground:NO];
65 [label setBordered:NO];
66 [label setStringValue:message];
67 m_fontSize = std::max(settings.minimumFontSize, 13.0);
68 [label setFont:[NSFont systemFontOfSize:m_fontSize]];
69 [label setMaximumNumberOfLines:4];
70 [[label cell] setTruncatesLastVisibleLine:YES];
71 [popoverView addSubview:label.get()];
72 NSSize labelSize = [label sizeThatFits:NSMakeSize(maxLabelWidth, CGFLOAT_MAX)];
73 [label setFrame:NSMakeRect(horizontalPadding, verticalPadding, labelSize.width, labelSize.height)];
74 [popoverView setFrame:NSMakeRect(0, 0, labelSize.width + horizontalPadding * 2, labelSize.height + verticalPadding * 2)];
75
76 m_popover = adoptNS([[WebValidationPopover alloc] init]);
77 [m_popover setContentViewController:controller.get()];
78 [m_popover setBehavior:NSPopoverBehaviorTransient];
79 [m_popover setAnimates:NO];
80}
81
82ValidationBubble::~ValidationBubble()
83{
84 [m_popover close];
85}
86
87void ValidationBubble::showRelativeTo(const IntRect& anchorRect)
88{
89 NSRect rect = NSMakeRect(anchorRect.x(), anchorRect.y(), anchorRect.width(), anchorRect.height());
90 [m_popover showRelativeToRect:rect ofView:m_view preferredEdge:NSMinYEdge];
91}
