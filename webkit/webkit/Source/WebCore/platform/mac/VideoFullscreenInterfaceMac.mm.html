188- (void)setUpPIPForVideoView:(NSView *)videoView withFrame:(NSRect)frame inWindow:(NSWindow *)window
189{
190 ASSERT(!_pipViewController);
191 ASSERT(!_videoViewContainerController);
192 ASSERT(!_videoViewContainer);
193
194 _pipViewController = adoptNS([allocPIPViewControllerInstance() init]);
195 [_pipViewController setDelegate:self];
196 [_pipViewController setUserCanResize:YES];
197 [_pipViewController setPlaying:_playing];
198 [self setVideoDimensions:NSEqualSizes(_videoDimensions, NSZeroSize) ? frame.size : _videoDimensions];
199 if (_videoFullscreenInterfaceMac && _videoFullscreenInterfaceMac->videoFullscreenModel())
200 _videoFullscreenInterfaceMac->videoFullscreenModel()->setVideoLayerGravity(MediaPlayerEnums::VideoGravity::ResizeAspectFill);
201
202 _videoViewContainer = adoptNS([[WebVideoViewContainer alloc] initWithFrame:frame]);
203 [_videoViewContainer setVideoViewContainerDelegate:self];
204 [_videoViewContainer addSubview:videoView];
205 videoView.frame = [_videoViewContainer bounds];
206 videoView.autoresizingMask = NSViewWidthSizable | NSViewHeightSizable;
207
208 _videoViewContainerController = adoptNS([[NSViewController alloc] init]);
209 [_videoViewContainerController setView:_videoViewContainer.get()];
210 [window.contentView addSubview:_videoViewContainer.get() positioned:NSWindowAbove relativeTo:nil];
211}
212
213- (void)enterPIP
214{
215 if (_pipState == PIPState::EnteringPIP || _pipState == PIPState::InPIP)
216 return;
217
218 [_videoViewContainerController view].layer.backgroundColor = CGColorGetConstantColor(kCGColorBlack);
219 [_pipViewController presentViewControllerAsPictureInPicture:_videoViewContainerController.get()];
220 _pipState = PIPState::EnteringPIP;
221}
222
223- (void)exitPIP
224{
225 if (_pipState != PIPState::InPIP || !_pipViewController || !_videoViewContainerController)
226 return;
227
228 [_videoViewContainerController view].layer.backgroundColor = CGColorGetConstantColor(kCGColorClear);
229 [_pipViewController dismissViewController:_videoViewContainerController.get()];
230 _pipState = PIPState::ExitingPIP;
231}
232
233- (void)exitPIPAnimatingToRect:(NSRect)rect inWindow:(NSWindow *)window
234{
235 _returningWindow = window;
236 _returningRect = rect;
237
238 [_pipViewController setReplacementRect:rect];
239 [_pipViewController setReplacementWindow:window];
240
241 [self exitPIP];
242}

246- (void)boundsDidChangeForVideoViewContainer:(WebVideoViewContainer *)videoViewContainer
247{
248 if (!_videoViewContainer || !_pipViewController)
249 return;
250
251 ASSERT_UNUSED(videoViewContainer, videoViewContainer == _videoViewContainer);
252
253 if (!_videoFullscreenInterfaceMac)
254 return;
255
256 if (_pipState == PIPState::EnteringPIP) {
257 // FIXME(rdar://problem/42250952)
258 // Currently, -[PIPViewController presentViewControllerAsPictureInPicture:] does not
259 // take a completionHandler parameter, so we use the first bounds change event
260 // as an indication that entering picture-in-picture is completed.
261 _pipState = PIPState::InPIP;
262
263 if (auto* model = _videoFullscreenInterfaceMac->videoFullscreenModel())
264 model->didEnterPictureInPicture();
265
266 if (auto* observer = _videoFullscreenInterfaceMac->videoFullscreenChangeObserver())
267 observer->didEnterFullscreen((WebCore::FloatSize)[_videoViewContainer bounds].size);
268 }
269
270 if (auto* model = _videoFullscreenInterfaceMac->videoFullscreenModel())
271 model->setVideoLayerFrame([_videoViewContainer bounds]);
272}
273
274- (void)superviewDidChangeForVideoViewContainer:(WebVideoViewContainer *)videoViewContainer
275{
276 if (!_videoViewContainer || !_pipViewController)
277 return;
278
279 ASSERT(videoViewContainer == _videoViewContainer);
280
281 if (![videoViewContainer isDescendantOf:[_pipViewController view]])
282 return;
283
284 // Once the view is moved into the pip view, make sure it resizes with the pip view.
285 videoViewContainer.frame = [videoViewContainer superview].bounds;
286 videoViewContainer.autoresizingMask = NSViewWidthSizable | NSViewHeightSizable;
287}

289// PIPViewControllerDelegate
290
291- (BOOL)pipShouldClose:(PIPViewController *)pip
292{
293 ASSERT_UNUSED(pip, pip == _pipViewController);
294
295 if (!_videoFullscreenInterfaceMac)
296 return YES;
297
298 if (_videoFullscreenInterfaceMac->videoFullscreenChangeObserver())
299 _videoFullscreenInterfaceMac->videoFullscreenChangeObserver()->fullscreenMayReturnToInline();
300
301 _videoFullscreenInterfaceMac->requestHideAndExitPiP();
302
303 return NO;
304}
305
306- (void)pipDidClose:(PIPViewController *)pip
307{
308 ASSERT_UNUSED(pip, pip == _pipViewController);
309
310 if (!_videoFullscreenInterfaceMac)
311 return;
312
313 if (_pipState != PIPState::ExitingPIP) {
314 // We got told to close without going through -pipActionStop, nor by exlicitly being asked to in -exitPiP:.
315 // Call -pipActionStop: here in order to set the fullscreen state to an expected value.
316 [self pipActionStop:pip];
317 }
318
319 if (_videoFullscreenInterfaceMac->videoFullscreenModel() && _videoViewContainer && _returningWindow && !NSEqualRects(_returningRect, NSZeroRect)) {
320 [NSAnimationContext runAnimationGroup:^(NSAnimationContext *context) {
321 context.allowsImplicitAnimation = NO;
322 [_videoViewContainer setFrame:_returningRect];
323 _videoFullscreenInterfaceMac->videoFullscreenModel()->setVideoLayerFrame([_videoViewContainer bounds]);
324 _videoFullscreenInterfaceMac->videoFullscreenModel()->setVideoLayerGravity(MediaPlayerEnums::VideoGravity::ResizeAspect);
325
326 [[_returningWindow contentView] addSubview:_videoViewContainer.get() positioned:NSWindowAbove relativeTo:nil];
327 } completionHandler:nil];
328 }
329
330 if (!self.isExitingToStandardFullscreen) {
331 if (VideoFullscreenModel* videoFullscreenModel = _videoFullscreenInterfaceMac->videoFullscreenModel()) {
332 videoFullscreenModel->didExitPictureInPicture();
333 videoFullscreenModel->setVideoLayerGravity(MediaPlayerEnums::VideoGravity::ResizeAspect);
334 }
335 }
336
337 if (VideoFullscreenChangeObserver* fullscreenChangeObserver = _videoFullscreenInterfaceMac->videoFullscreenChangeObserver())
338 fullscreenChangeObserver->didExitFullscreen();
339
340 _videoFullscreenInterfaceMac->clearMode(HTMLMediaElementEnums::VideoFullscreenModePictureInPicture);
341}
342
343- (void)pipActionPlay:(PIPViewController *)pip
344{
345 ASSERT_UNUSED(pip, pip == _pipViewController);
346
347 if (_videoFullscreenInterfaceMac && _videoFullscreenInterfaceMac->playbackSessionModel())
348 _videoFullscreenInterfaceMac->playbackSessionModel()->play();
349}
350
351- (void)pipActionPause:(PIPViewController *)pip
352{
353 ASSERT_UNUSED(pip, pip == _pipViewController);
354
355 if (_videoFullscreenInterfaceMac && _videoFullscreenInterfaceMac->playbackSessionModel())
356 _videoFullscreenInterfaceMac->playbackSessionModel()->pause();
357}
358
359- (void)pipActionStop:(PIPViewController *)pip
360{
361 ASSERT_UNUSED(pip, pip == _pipViewController);
362
363 if (!_videoFullscreenInterfaceMac)
364 return;
365
366 if (PlaybackSessionModel* playbackSessionModel = _videoFullscreenInterfaceMac->playbackSessionModel())
367 playbackSessionModel->pause();
368
369 _videoFullscreenInterfaceMac->requestHideAndExitPiP();
370 _pipState = PIPState::ExitingPIP;
371}

448void VideoFullscreenInterfaceMac::setupFullscreen(NSView& layerHostedView, const IntRect& initialRect, NSWindow *parentWindow, HTMLMediaElementEnums::VideoFullscreenMode mode, bool allowsPictureInPicturePlayback)
449{
450 LOG(Fullscreen, "VideoFullscreenInterfaceMac::setupFullscreen(%p), initialRect:{%d, %d, %d, %d}, parentWindow:%p, mode:%d", this, initialRect.x(), initialRect.y(), initialRect.width(), initialRect.height(), parentWindow, mode);
451
452 UNUSED_PARAM(allowsPictureInPicturePlayback);
453 ASSERT(mode == HTMLMediaElementEnums::VideoFullscreenModePictureInPicture);
454
455 m_mode = mode;
456
457 [videoFullscreenInterfaceObjC() setUpPIPForVideoView:&layerHostedView withFrame:(NSRect)initialRect inWindow:parentWindow];
458
459 RunLoop::main().dispatch([protectedThis = makeRefPtr(this), this] {
460 if (m_fullscreenChangeObserver)
461 m_fullscreenChangeObserver->didSetupFullscreen();
462 });
463}

465void VideoFullscreenInterfaceMac::enterFullscreen()
466{
467 LOG(Fullscreen, "VideoFullscreenInterfaceMac::enterFullscreen(%p)", this);
468
469 RELEASE_ASSERT(m_videoFullscreenModel);
470 if (mode() == HTMLMediaElementEnums::VideoFullscreenModePictureInPicture) {
471 m_videoFullscreenModel->willEnterPictureInPicture();
472 [m_webVideoFullscreenInterfaceObjC enterPIP];
473
474#if ENABLE(WEB_PLAYBACK_CONTROLS_MANAGER)
475 [m_playbackSessionInterface->playBackControlsManager() setPictureInPictureActive:YES];
476#endif
477 }

480bool VideoFullscreenInterfaceMac::exitFullscreen(const IntRect& finalRect, NSWindow *parentWindow)
481{
482 LOG(Fullscreen, "VideoFullscreenInterfaceMac::exitFullscreen(%p), finalRect:{%d, %d, %d, %d}, parentWindow:%p", this, finalRect.x(), finalRect.y(), finalRect.width(), finalRect.height(), parentWindow);
483
484#if ENABLE(WEB_PLAYBACK_CONTROLS_MANAGER)
485 [m_playbackSessionInterface->playBackControlsManager() setPictureInPictureActive:NO];
486#endif
487
488 if (finalRect.isEmpty())
489 [m_webVideoFullscreenInterfaceObjC exitPIP];
490 else
491 [m_webVideoFullscreenInterfaceObjC exitPIPAnimatingToRect:finalRect inWindow:parentWindow];
492
493 return true;
494}

496void VideoFullscreenInterfaceMac::exitFullscreenWithoutAnimationToMode(HTMLMediaElementEnums::VideoFullscreenMode mode)
497{
498 LOG(Fullscreen, "VideoFullscreenInterfaceMac::exitFullscreenWithoutAnimationToMode(%p), mode:%d", this, mode);
499
500#if ENABLE(WEB_PLAYBACK_CONTROLS_MANAGER)
501 [m_playbackSessionInterface->playBackControlsManager() setPictureInPictureActive:NO];
502#endif
503
504 bool isExitingToStandardFullscreen = mode == HTMLMediaElementEnums::VideoFullscreenModeStandard;
505 // On Mac, standard fullscreen is handled by the Fullscreen API and not by VideoFullscreenManager.
506 // Just update m_mode directly to HTMLMediaElementEnums::VideoFullscreenModeStandard in that case to keep
507 // m_mode in sync with the fullscreen mode in HTMLMediaElement.
508 if (isExitingToStandardFullscreen)
509 m_mode = HTMLMediaElementEnums::VideoFullscreenModeStandard;
510
511 [m_webVideoFullscreenInterfaceObjC setExitingToStandardFullscreen:isExitingToStandardFullscreen];
512 [m_webVideoFullscreenInterfaceObjC exitPIP];
513}
