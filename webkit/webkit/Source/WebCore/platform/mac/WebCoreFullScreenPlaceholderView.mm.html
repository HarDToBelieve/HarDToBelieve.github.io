27#import "WebCoreFullScreenPlaceholderView.h"
28
29#if PLATFORM(MAC)
30
31#import "LocalizedStrings.h"
32#import "WebCoreFullScreenWarningView.h"
33#import <wtf/text/WTFString.h>
34
35@implementation WebCoreFullScreenPlaceholderView
36
37- (id)initWithFrame:(NSRect)frameRect
38{
39 self = [super initWithFrame:frameRect];
40 if (!self)
41 return nil;
42
43 self.wantsLayer = YES;
44 self.autoresizesSubviews = YES;
45 self.layerContentsPlacement = NSViewLayerContentsPlacementScaleProportionallyToFit;
46 self.layerContentsRedrawPolicy = NSViewLayerContentsRedrawNever;
47
48 _effectView = adoptNS([[NSVisualEffectView alloc] initWithFrame:frameRect]);
49 _effectView.get().wantsLayer = YES;
50 _effectView.get().autoresizesSubviews = YES;
51 _effectView.get().autoresizingMask = NSViewWidthSizable | NSViewHeightSizable;
52 _effectView.get().blendingMode = NSVisualEffectBlendingModeWithinWindow;
53 _effectView.get().hidden = YES;
54 _effectView.get().state = NSVisualEffectStateActive;
55 _effectView.get().material = NSVisualEffectMaterialPopover;
56 [self addSubview:_effectView.get()];
57
58 _exitWarning = adoptNS([[NSTextField alloc] initWithFrame:NSZeroRect]);
59 _exitWarning.get().autoresizingMask = NSViewMinXMargin | NSViewMaxXMargin | NSViewMinYMargin | NSViewMaxYMargin;
60 _exitWarning.get().bordered = NO;
61 _exitWarning.get().drawsBackground = NO;
62 _exitWarning.get().editable = NO;
63 _exitWarning.get().font = [NSFont systemFontOfSize:27];
64 _exitWarning.get().selectable = NO;
65 _exitWarning.get().stringValue = WebCore::clickToExitFullScreenText();
66 _exitWarning.get().textColor = [NSColor tertiaryLabelColor];
67 [_exitWarning sizeToFit];
68
69 NSRect warningFrame = [_exitWarning.get() frame];
70 warningFrame.origin = NSMakePoint((frameRect.size.width - warningFrame.size.width) / 2, frameRect.size.height / 2);
71 _exitWarning.get().frame = warningFrame;
72 [_effectView addSubview:_exitWarning.get()];
73
74 return self;
75}
76
77- (NSResponder *)target
78{
79 return _target.get().get();
80}
81
82- (void)setTarget:(NSResponder *)target
83{
84 _target = target;
85}
86
87@dynamic contents;
88
89- (void)setContents:(id)contents
90{
91 [[self layer] setContents:contents];
92}
93
94- (id)contents
95{
96 return [[self layer] contents];
97}
98
99- (void)setExitWarningVisible:(BOOL)visible
100{
101 [_effectView setHidden:!visible];
102}
103
104- (void)mouseDown:(NSEvent *)event
105{
106 UNUSED_PARAM(event);
107 [_target cancelOperation:self];
108}
109
110@end
