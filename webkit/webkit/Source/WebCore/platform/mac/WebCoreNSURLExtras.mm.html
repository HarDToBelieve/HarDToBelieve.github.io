42NSURL *URLByCanonicalizingURL(NSURL *URL)
43{
44 RetainPtr<NSURLRequest> request = adoptNS([[NSURLRequest alloc] initWithURL:URL]);
45#if HAVE(NSURLPROTOCOL_WITH_SKIPAPPSSO)
46 Class concreteClass = [NSURLProtocol _protocolClassForRequest:request.get() skipAppSSO:YES];
47#else
48 Class concreteClass = [NSURLProtocol _protocolClassForRequest:request.get()];
49#endif
50 if (!concreteClass) {
51 return URL;
52 }
53
54 // This applies NSURL's concept of canonicalization, but not URL's concept. It would
55 // make sense to apply both, but when we tried that it caused a performance degradation
56 // (see 5315926). It might make sense to apply only the URL concept and not the NSURL
57 // concept, but it's too risky to make that change for WebKit 3.0.
58 NSURLRequest *newRequest = [concreteClass canonicalRequestForRequest:request.get()];
59 return retainPtr([newRequest URL]).autorelease();
60}
