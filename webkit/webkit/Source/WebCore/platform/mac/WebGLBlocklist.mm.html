37struct OSBuildInfo {
38 OSBuildInfo()
39 : major(0)
40 , minor(0)
41 , build(0)
42 {
43 }
44
45 OSBuildInfo(int major, int minor, int build)
46 : major(major)
47 , minor(minor)
48 , build(build)
49 {
50 }
51
52 bool operator==(const OSBuildInfo& other) const
53 {
54 return major == other.major && minor == other.minor && build == other.build;
55 }
56
57 bool operator>(const OSBuildInfo& other) const
58 {
59 return std::tie(major, minor, build) > std::tie(other.major, other.minor, other.build);
60 }
61
62 bool operator<=(const OSBuildInfo& other) const
63 {
64 return std::tie(major, minor, build) <= std::tie(other.major, other.minor, other.build);
65 }
66
67 bool operator<(const OSBuildInfo& other) const
68 {
69 return std::tie(major, minor, build) < std::tie(other.major, other.minor, other.build);
70 }
71
72 int major; // Represents the 13 in 13C64.
73 int minor; // Represents the C in 13C64 (as a number where A == 1, i.e. 3).
74 int build; // Represents the 64 in 13C64.
75};

77static OSBuildInfo buildInfoFromOSBuildString(NSString *buildString)
78{
79 NSError *error = NULL;
80 NSRegularExpression *regex = [NSRegularExpression regularExpressionWithPattern:@"^(\\d+)([A-Z])(\\d+)" options:0 error:&error];
81 NSArray *matches = [regex matchesInString:buildString options:0 range:NSMakeRange(0, [buildString length])];
82 if (!matches || matches.count != 1) {
83#ifndef NDEBUG
84 NSLog(@"WebGLBlocklist could not parse OSBuild entry: %@", buildString);
85#endif
86 return OSBuildInfo();
87 }
88
89 NSTextCheckingResult *matchResult = [matches objectAtIndex:0];
90
91 if (matchResult.numberOfRanges != 4) {
92#ifndef NDEBUG
93 NSLog(@"WebGLBlocklist could not parse OSBuild entry: %@", buildString);
94#endif
95 return OSBuildInfo();
96 }
97
98 int majorVersion = [[buildString substringWithRange:[matchResult rangeAtIndex:1]] intValue];
99 int minorVersion = [[buildString substringWithRange:[matchResult rangeAtIndex:2]] characterAtIndex:0] - 'A' + 1;
100 int buildVersion = [[buildString substringWithRange:[matchResult rangeAtIndex:3]] intValue];
101
102 return OSBuildInfo(majorVersion, minorVersion, buildVersion);
103}

105bool WebGLBlocklist::shouldBlockWebGL()
106{
107 bool shouldBlock = false;
108 BlocklistUpdater::queue().dispatchSync([&shouldBlock] {
109 BlocklistUpdater::reloadIfNecessary();
110 if (auto* webGLBlocklist = BlocklistUpdater::webGLBlocklist())
111 shouldBlock = webGLBlocklist->shouldBlock();
112 });
113
114 return shouldBlock;
115}

129static bool matchesGPU(GLint machineId, GLint rendererMask)
130{
131 // If the last two bytes of the rendererMask are not zero, then we're
132 // looking for an exact GPU match. Otherwise we're matching against
133 // a class of GPUs.
134
135 if (rendererMask & 0xFF)
136 return machineId == rendererMask;
137
138 return (machineId & kCGLRendererIDMatchingMask) == rendererMask;
139}
140
141static GLint gpuMaskFromString(NSString *input)
142{
143 NSScanner* scanner = [NSScanner scannerWithString:input];
144 unsigned maskValue;
145 [scanner scanHexInt:&maskValue];
146 return static_cast<GLint>(maskValue & (kCGLRendererIDMatchingMask | 0xFF));
147}

149static bool matchesBuildInfo(OSBuildInfo machineInfo, OSBuildInfo blockInfo, WebGLBlocklist::BlockComparison comparison)
150{
151 switch (comparison) {
152 case WebGLBlocklist::BlockComparison::Equals:
153 return machineInfo == blockInfo;
154 case WebGLBlocklist::BlockComparison::LessThan:
155 return machineInfo < blockInfo;
156 case WebGLBlocklist::BlockComparison::LessThanEquals:
157 return machineInfo <= blockInfo;
158 }
159}

161std::unique_ptr<WebGLBlocklist> WebGLBlocklist::create(NSDictionary *propertyList)
162{
163 auto systemVersionDictionary = adoptCF(_CFCopySystemVersionDictionary());
164 CFStringRef osBuild = static_cast<CFStringRef>(CFDictionaryGetValue(systemVersionDictionary.get(), _kCFSystemVersionBuildVersionKey));
165 OSBuildInfo buildInfo = buildInfoFromOSBuildString((__bridge NSString *)osBuild);
166
167 if (!buildInfo.major)
168 return nullptr;
169
170 NSArray *blockEntries = [propertyList objectForKey:@"WebGLBlocklist"];
171
172 if (![blockEntries isKindOfClass:[NSArray class]] || !blockEntries.count)
173 return nullptr;
174
175 CGLPixelFormatAttribute attribs[12] = {
176 kCGLPFAColorSize, (CGLPixelFormatAttribute)32,
177 kCGLPFADepthSize, (CGLPixelFormatAttribute)32,
178 kCGLPFAAccelerated,
179 kCGLPFASupersample,
180 kCGLPFAMultisample,
181 kCGLPFASampleBuffers, (CGLPixelFormatAttribute)1,
182 kCGLPFASamples, (CGLPixelFormatAttribute)4,
183 (CGLPixelFormatAttribute)0
184 };
185
186 CGLPixelFormatObj pix;
187 GLint npix;
188 CGLChoosePixelFormat(attribs, &pix, &npix);
189 CGLContextObj ctx;
190 CGLCreateContext(pix, 0, &ctx);
191 GLint rendererId = 0;
192 CGLGetParameter(ctx, kCGLCPCurrentRendererID, &rendererId);
193 GLint supportsSeparateAddressSpace = 0;
194 CGLGetParameter(ctx, kCGLCPSupportSeparateAddressSpace, &supportsSeparateAddressSpace);
195 CGLDestroyContext(ctx);
196 CGLReleasePixelFormat(pix);
197
198 rendererId &= kCGLRendererIDMatchingMask | 0xFF;
199
200 BlockCommand globalCommand = BlockCommand::Allow;
201
202 for (NSDictionary *blockData in blockEntries) {
203
204 GLint gpuMask = gpuMaskFromString([blockData objectForKey:@"GPU"]);
205
206 OSBuildInfo blockedBuildInfo = buildInfoFromOSBuildString(static_cast<NSString*>([blockData objectForKey:@"OSBuild"]));
207
208 NSString *comparisonString = [blockData objectForKey:@"Comparison"];
209 BlockComparison comparison = BlockComparison::Equals;
210 if ([comparisonString isEqualToString:@"LessThan"])
211 comparison = BlockComparison::LessThan;
212 else if ([comparisonString isEqualToString:@"LessThanEquals"])
213 comparison = BlockComparison::LessThanEquals;
214
215 NSString *commandString = [blockData objectForKey:@"Command"];
216 BlockCommand command = BlockCommand::Allow;
217 if ([commandString isEqualToString:@"Block"])
218 command = BlockCommand::Block;
219 else if ([commandString isEqualToString:@"SuggestBlocking"])
220 command = BlockCommand::SuggestBlocking;
221
222 if (matchesGPU(rendererId, gpuMask) && matchesBuildInfo(buildInfo, blockedBuildInfo, comparison)) {
223 globalCommand = command;
224 break;
225 }
226 }
227
228 if (!supportsSeparateAddressSpace && globalCommand == BlockCommand::Allow)
229 globalCommand = BlockCommand::SuggestBlocking;
230
231 return std::unique_ptr<WebGLBlocklist>(new WebGLBlocklist(globalCommand));
232}
