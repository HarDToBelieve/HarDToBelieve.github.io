33enum {
34 NSAttachmentCharacter = 0xfffc /* To denote attachments. */
35};

39namespace WebCore {
40
41NSAttributedString *attributedStringByStrippingAttachmentCharacters(NSAttributedString *attributedString)
42{
43 NSRange attachmentRange;
44 NSString *originalString = [attributedString string];
45 static auto attachmentCharString = makeNeverDestroyed([] {
46 unichar chars[2] = { NSAttachmentCharacter, 0 };
47 return adoptNS([[NSString alloc] initWithCharacters:chars length:1]);
48 }());
49
50 attachmentRange = [originalString rangeOfString:attachmentCharString.get().get()];
51 if (attachmentRange.location != NSNotFound && attachmentRange.length > 0) {
52 auto newAttributedString = adoptNS([attributedString mutableCopyWithZone:NULL]);
53
54 while (attachmentRange.location != NSNotFound && attachmentRange.length > 0) {
55 [newAttributedString replaceCharactersInRange:attachmentRange withString:@""];
56 attachmentRange = [[newAttributedString string] rangeOfString:attachmentCharString.get().get()];
57 }
58 return newAttributedString.autorelease();
59 }
60
61 return attributedString;
62}
