56using FactoryVector = Vector<MediaEngineConfigurationFactory::MediaEngineFactory>;
57static FactoryVector defaultFactories()
58{
59 FactoryVector factories;
60#if PLATFORM(COCOA)
61 factories.append({ &createMediaPlayerDecodingConfigurationCocoa, nullptr });
62#endif
63#if USE(GSTREAMER)
64 factories.append({ &createMediaPlayerDecodingConfigurationGStreamer, &createMediaPlayerEncodingConfigurationGStreamer });
65#endif
66 return factories;
67}

69static FactoryVector& factories()
70{
71 static auto factories = makeNeverDestroyed<FactoryVector>(defaultFactories());
72 return factories;
73}

90bool MediaEngineConfigurationFactory::hasDecodingConfigurationFactory()
91{
92 return mockEnabled() || WTF::anyOf(factories(), [] (auto& factory) { return (bool)factory.createDecodingConfiguration; });
93}

95bool MediaEngineConfigurationFactory::hasEncodingConfigurationFactory()

100void MediaEngineConfigurationFactory::createDecodingConfiguration(MediaDecodingConfiguration&& config, MediaEngineConfigurationFactory::DecodingConfigurationCallback&& callback)
101{
102 if (mockEnabled()) {
103 MediaEngineConfigurationFactoryMock::createDecodingConfiguration(WTFMove(config), WTFMove(callback));
104 return;
105 }
106
107 auto factoryCallback = [] (auto factoryCallback, auto nextFactory, auto&& config, auto&& callback) mutable {
108 if (nextFactory == factories().end()) {
109 callback({{ }, WTFMove(config)});
110 return;
111 }
112
113 auto& factory = *nextFactory;
114 if (!factory.createDecodingConfiguration) {
115 callback({{ }, WTFMove(config)});
116 return;
117 }
118
119 factory.createDecodingConfiguration(WTFMove(config), [factoryCallback, nextFactory, config, callback = WTFMove(callback)] (auto&& info) mutable {
120 if (info.supported) {
121 callback(WTFMove(info));
122 return;
123 }
124
125 factoryCallback(factoryCallback, ++nextFactory, WTFMove(info.supportedConfiguration), WTFMove(callback));
126 });
127 };
128 factoryCallback(factoryCallback, factories().begin(), config, WTFMove(callback));
129}

131void MediaEngineConfigurationFactory::createEncodingConfiguration(MediaEncodingConfiguration&& config, MediaEngineConfigurationFactory::EncodingConfigurationCallback&& callback)
132{
133 if (mockEnabled()) {
134 MediaEngineConfigurationFactoryMock::createEncodingConfiguration(WTFMove(config), WTFMove(callback));
135 return;
136 }
137
138 auto factoryCallback = [] (auto factoryCallback, auto nextFactory, auto&& config, auto&& callback) mutable {
139 if (nextFactory == factories().end()) {
140 callback({ });
141 return;
142 }
143
144 auto& factory = *nextFactory;
145 if (!factory.createEncodingConfiguration) {
146 callback({ });
147 return;
148 }
149
150 factory.createEncodingConfiguration(WTFMove(config), [factoryCallback, nextFactory, callback = WTFMove(callback)] (auto&& info) mutable {
151 if (info.supported) {
152 callback(WTFMove(info));
153 return;
154 }
155
156 factoryCallback(factoryCallback, ++nextFactory, WTFMove(info.supportedConfiguration), WTFMove(callback));
157 });
158 };
159 factoryCallback(factoryCallback, factories().begin(), WTFMove(config), WTFMove(callback));
160}
