28#pragma once
29
30#include <wtf/Function.h>
31
32namespace WebCore {
33
34struct MediaCapabilitiesDecodingInfo;
35struct MediaCapabilitiesEncodingInfo;
36struct MediaDecodingConfiguration;
37struct MediaEncodingConfiguration;
38
39class MediaEngineConfigurationFactory {
40public:
41 using DecodingConfigurationCallback = WTF::Function<void(MediaCapabilitiesDecodingInfo&&)>;
42 using EncodingConfigurationCallback = WTF::Function<void(MediaCapabilitiesEncodingInfo&&)>;
43
44 static bool hasDecodingConfigurationFactory();
45 static bool hasEncodingConfigurationFactory();
46
47 WEBCORE_EXPORT static void createDecodingConfiguration(MediaDecodingConfiguration&&, DecodingConfigurationCallback&&);
48 WEBCORE_EXPORT static void createEncodingConfiguration(MediaEncodingConfiguration&&, EncodingConfigurationCallback&&);
49
50 using CreateDecodingConfiguration = WTF::Function<void(MediaDecodingConfiguration&&, DecodingConfigurationCallback&&)>;
51 using CreateEncodingConfiguration = WTF::Function<void(MediaEncodingConfiguration&&, EncodingConfigurationCallback&&)>;
52
53 struct MediaEngineFactory {
54 CreateDecodingConfiguration createDecodingConfiguration;
55 CreateEncodingConfiguration createEncodingConfiguration;
56 };
57
58 WEBCORE_EXPORT static void clearFactories();
59 WEBCORE_EXPORT static void resetFactories();
60 WEBCORE_EXPORT static void installFactory(MediaEngineFactory&&);
61
62 WEBCORE_EXPORT static void enableMock();
63 WEBCORE_EXPORT static void disableMock();
64};
65
66} // namespace WebCore
