46std::unique_ptr<MediaRecorderPrivateAVFImpl> MediaRecorderPrivateAVFImpl::create(MediaStreamPrivate& stream, const MediaRecorderPrivateOptions& options)
47{
48 // FIXME: we will need to implement support for multiple audio/video tracks
49 // Currently we only choose the first track as the recorded track.
50 // FIXME: We would better to throw an exception to JavaScript if writer creation fails.
51
52 auto selectedTracks = MediaRecorderPrivate::selectTracks(stream);
53
54 auto writer = MediaRecorderPrivateWriter::create(!!selectedTracks.audioTrack, !!selectedTracks.videoTrack, options);
55 if (!writer)
56 return nullptr;
57
58 auto recorder = std::unique_ptr<MediaRecorderPrivateAVFImpl>(new MediaRecorderPrivateAVFImpl(writer.releaseNonNull()));
59 if (selectedTracks.audioTrack) {
60 recorder->setAudioSource(&selectedTracks.audioTrack->source());
61 recorder->checkTrackState(*selectedTracks.audioTrack);
62 }
63 if (selectedTracks.videoTrack) {
64 recorder->setVideoSource(&selectedTracks.videoTrack->source());
65 recorder->checkTrackState(*selectedTracks.videoTrack);
66 }
67 return recorder;
68}

86void MediaRecorderPrivateAVFImpl::videoSampleAvailable(MediaSample& sampleBuffer)
87{
88 if (shouldMuteVideo()) {
89 if (!m_blackFrame) {
90 m_blackFrameDescription = PAL::CMSampleBufferGetFormatDescription(sampleBuffer.platformSample().sample.cmSampleBuffer);
91 auto dimensions = PAL::CMVideoFormatDescriptionGetDimensions(m_blackFrameDescription.get());
92 m_blackFrame = createBlackPixelBuffer(dimensions.width, dimensions.height);
93
94 CMVideoFormatDescriptionRef formatDescription = nullptr;
95 auto status = PAL::CMVideoFormatDescriptionCreateForImageBuffer(kCFAllocatorDefault, m_blackFrame.get(), &formatDescription);
96 if (status != noErr) {
97 RELEASE_LOG_ERROR(Media, "MediaRecorderPrivateAVFImpl::videoSampleAvailable ::unable to create a black frame description: %d", static_cast<int>(status));
98 m_blackFrame = nullptr;
99 return;
100 }
101 m_blackFrameDescription = adoptCF(formatDescription);
102 }
103
104 CMSampleBufferRef sample = nullptr;
105 CMSampleTimingInfo timingInfo { PAL::kCMTimeInvalid, PAL::toCMTime(sampleBuffer.presentationTime()), PAL::toCMTime(sampleBuffer.decodeTime()) };
106 auto status = PAL::CMSampleBufferCreateReadyWithImageBuffer(kCFAllocatorDefault, (CVImageBufferRef)m_blackFrame.get(), m_blackFrameDescription.get(), &timingInfo, &sample);
107
108 if (status != noErr) {
109 RELEASE_LOG_ERROR(MediaStream, "MediaRecorderPrivateAVFImpl::videoSampleAvailable - unable to create a black frame: %d", static_cast<int>(status));
110 return;
111 }
112 auto newSample = adoptCF(sample);
113 m_writer->appendVideoSampleBuffer(MediaSampleAVFObjC::create(newSample.get(), sampleBuffer.videoRotation(), sampleBuffer.videoMirrored()));
114 return;
115 }
116
117 m_blackFrame = nullptr;
118 m_blackFrameDescription = nullptr;
119 m_writer->appendVideoSampleBuffer(sampleBuffer);
120}

122void MediaRecorderPrivateAVFImpl::audioSamplesAvailable(const WTF::MediaTime& mediaTime, const PlatformAudioData& data, const AudioStreamDescription& description, size_t sampleCount)
123{
124 ASSERT(is<WebAudioBufferList>(data));
125 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
126
127 if (shouldMuteAudio()) {
128 if (!m_audioBuffer || m_description != toCAAudioStreamDescription(description)) {
129 m_description = toCAAudioStreamDescription(description);
130 m_audioBuffer = makeUnique<WebAudioBufferList>(m_description, sampleCount);
131 } else
132 m_audioBuffer->setSampleCount(sampleCount);
133 m_audioBuffer->zeroFlatBuffer();
134 m_writer->appendAudioSampleBuffer(*m_audioBuffer, description, mediaTime, sampleCount);
135 return;
136 }
137
138 m_writer->appendAudioSampleBuffer(data, description, mediaTime, sampleCount);
139}
