42std::unique_ptr<VideoSampleBufferCompressor> VideoSampleBufferCompressor::create(String mimeType, CMBufferQueueTriggerCallback callback, void* callbackObject)
43{
44 auto profile = Profile::Baseline;
45 for (auto codec : ContentType(mimeType).codecs()) {
46 if (startsWithLettersIgnoringASCIICase(codec, "avc1.") && codec.length() >= 11) {
47 if (codec[5] == '6' && codec[6] == '4')
48 profile = Profile::High;
49 else if (codec[5] == '4' && (codec[6] == 'd' || codec[6] == 'D'))
50 profile = Profile::Main;
51 break;
52 }
53 }
54
55 auto compressor = std::unique_ptr<VideoSampleBufferCompressor>(new VideoSampleBufferCompressor(kCMVideoCodecType_H264, profile));
56 if (!compressor->initialize(callback, callbackObject))
57 return nullptr;
58 return compressor;
59}

72 m_vtSession = nullptr;

76bool VideoSampleBufferCompressor::initialize(CMBufferQueueTriggerCallback callback, void* callbackObject)
77{
78 CMBufferQueueRef outputBufferQueue;
79 if (auto error = PAL::CMBufferQueueCreate(kCFAllocatorDefault, 0, PAL::CMBufferQueueGetCallbacksForUnsortedSampleBuffers(), &outputBufferQueue)) {
80 RELEASE_LOG_ERROR(MediaStream, "VideoSampleBufferCompressor unable to create buffer queue %d", error);
81 return false;
82 }
83 m_outputBufferQueue = outputBufferQueue;
84 PAL::CMBufferQueueInstallTrigger(m_outputBufferQueue.get(), callback, callbackObject, kCMBufferQueueTrigger_WhenDataBecomesReady, PAL::kCMTimeZero, NULL);
85
86 m_isEncoding = true;
87 return true;
88}

95void VideoSampleBufferCompressor::finish()
96{
97 m_serialDispatchQueue->dispatchSync([this] {
98 auto error = PAL::VTCompressionSessionCompleteFrames(m_vtSession.get(), PAL::kCMTimeInvalid);
99 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTCompressionSessionCompleteFrames failed with %d", error);
100
101 error = PAL::CMBufferQueueMarkEndOfData(m_outputBufferQueue.get());
102 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor CMBufferQueueMarkEndOfData failed with %d", error);
103
104 m_isEncoding = false;
105 });
106}

108void VideoSampleBufferCompressor::videoCompressionCallback(void *refCon, void*, OSStatus status, VTEncodeInfoFlags, CMSampleBufferRef buffer)
109{
110 RELEASE_LOG_ERROR_IF(status, MediaStream, "VideoSampleBufferCompressor videoCompressionCallback status is %d", status);
111 if (status != noErr)
112 return;
113
114 VideoSampleBufferCompressor *compressor = static_cast<VideoSampleBufferCompressor*>(refCon);
115
116 auto error = PAL::CMBufferQueueEnqueue(compressor->m_outputBufferQueue.get(), buffer);
117 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor CMBufferQueueEnqueue failed with %d", error);
118}

120static inline OSStatus setCompressionSessionProperty(VTCompressionSessionRef vtSession, CFStringRef key, uint32_t value)
121{
122 int64_t value64 = value;
123 auto cfValue = adoptCF(CFNumberCreate(kCFAllocatorDefault, kCFNumberSInt64Type, &value64));
124 OSStatus status = VTSessionSetProperty(vtSession, key, cfValue.get());
125 return status;
126}

140bool VideoSampleBufferCompressor::initCompressionSession(CMVideoFormatDescriptionRef formatDescription)
141{
142 CMVideoDimensions dimensions = PAL::CMVideoFormatDescriptionGetDimensions(formatDescription);
143#if PLATFORM(IOS)
144 NSDictionary *encoderSpecifications = nil;
145#else
146 NSDictionary *encoderSpecifications = @{(__bridge NSString *)PAL::kVTVideoEncoderSpecification_EnableHardwareAcceleratedVideoEncoder:@YES};
147#endif
148
149 VTCompressionSessionRef vtSession;
150 auto error = PAL::VTCompressionSessionCreate(kCFAllocatorDefault, dimensions.width, dimensions.height, m_outputCodecType, (__bridge CFDictionaryRef)encoderSpecifications, NULL, NULL, videoCompressionCallback, this, &vtSession);
151 if (error) {
152 RELEASE_LOG_ERROR(MediaStream, "VideoSampleBufferCompressor VTCompressionSessionCreate failed with %d", error);
153 return NO;
154 }
155 m_vtSession = adoptCF(vtSession);
156
157 error = VTSessionSetProperty(m_vtSession.get(), PAL::kVTCompressionPropertyKey_RealTime, kCFBooleanTrue);
158 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTSessionSetProperty kVTCompressionPropertyKey_RealTime failed with %d", error);
159 error = setCompressionSessionProperty(m_vtSession.get(), PAL::kVTCompressionPropertyKey_MaxKeyFrameIntervalDuration, m_maxKeyFrameIntervalDuration);
160 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTSessionSetProperty kVTCompressionPropertyKey_MaxKeyFrameIntervalDuration failed with %d", error);
161 error = setCompressionSessionProperty(m_vtSession.get(), PAL::kVTCompressionPropertyKey_ExpectedFrameRate, m_expectedFrameRate);
162 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTSessionSetProperty kVTCompressionPropertyKey_ExpectedFrameRate failed with %d", error);
163
164 error = VTSessionSetProperty(m_vtSession.get(), PAL::kVTCompressionPropertyKey_ProfileLevel, vtProfileLevel());
165 if (error) {
166 RELEASE_LOG_ERROR(MediaStream, "VideoSampleBufferCompressor VTSessionSetProperty kVTCompressionPropertyKey_ProfileLevel failed with %d for profile %d", error, m_profile);
167 if (m_profile != Profile::Baseline) {
168 error = VTSessionSetProperty(m_vtSession.get(), PAL::kVTCompressionPropertyKey_ProfileLevel, PAL::kVTProfileLevel_H264_Baseline_AutoLevel);
169 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTSessionSetProperty kVTCompressionPropertyKey_ProfileLevel failed with %d for default profile", error);
170 }
171 }
172
173 if (m_outputBitRate) {
174 error = setCompressionSessionProperty(m_vtSession.get(), PAL::kVTCompressionPropertyKey_AverageBitRate, *m_outputBitRate);
175 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTSessionSetProperty kVTCompressionPropertyKey_AverageBitRate failed with %d", error);
176 }
177
178 error = PAL::VTCompressionSessionPrepareToEncodeFrames(m_vtSession.get());
179 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTCompressionSessionPrepareToEncodeFrames failed with %d", error);
180
181 return YES;
182}

184void VideoSampleBufferCompressor::processSampleBuffer(CMSampleBufferRef buffer)
185{
186 if (!m_vtSession) {
187 if (!initCompressionSession(PAL::CMSampleBufferGetFormatDescription(buffer)))
188 return;
189 }
190
191 auto imageBuffer = PAL::CMSampleBufferGetImageBuffer(buffer);
192 auto presentationTimeStamp = PAL::CMSampleBufferGetPresentationTimeStamp(buffer);
193 auto duration = PAL::CMSampleBufferGetDuration(buffer);
194 auto error = PAL::VTCompressionSessionEncodeFrame(m_vtSession.get(), imageBuffer, presentationTimeStamp, duration, NULL, this, NULL);
195 RELEASE_LOG_ERROR_IF(error, MediaStream, "VideoSampleBufferCompressor VTCompressionSessionEncodeFrame failed with %d", error);
196}

198void VideoSampleBufferCompressor::addSampleBuffer(CMSampleBufferRef buffer)
199{
200 m_serialDispatchQueue->dispatchSync([this, buffer] {
201 if (!m_isEncoding)
202 return;
203
204 processSampleBuffer(buffer);
205 });
206}
