36AudioTrackPrivateMediaStream::AudioTrackPrivateMediaStream(MediaStreamTrackPrivate& track)
37 : m_streamTrack(track)
38 , m_audioSource(track.source())
39 , m_id(track.id())
40 , m_label(track.label())
41 , m_renderer(createRenderer(*this))
42{
43 track.addObserver(*this);
44}
45
46AudioTrackPrivateMediaStream::~AudioTrackPrivateMediaStream()
47{
48 clear();
49}
50
51std::unique_ptr<AudioMediaStreamTrackRenderer> AudioTrackPrivateMediaStream::createRenderer(AudioTrackPrivateMediaStream& stream)
52{
53 auto renderer = AudioMediaStreamTrackRenderer::create();
54 if (!renderer)
55 return nullptr;
56#if !RELEASE_LOG_DISABLED
57 auto& track = stream.m_streamTrack.get();
58 renderer->setLogger(track.logger(), track.logIdentifier());
59#endif
60 renderer->setCrashCallback([stream = makeWeakPtr(stream)] {
61 if (stream)
62 stream->createNewRenderer();
63 });
64 return renderer;
65}
66
67void AudioTrackPrivateMediaStream::clear()
68{
69 if (m_isCleared)
70 return;
71
72 m_isCleared = true;
73
74 if (m_isPlaying)
75 m_audioSource->removeAudioSampleObserver(*this);
76
77 streamTrack().removeObserver(*this);
78 if (m_renderer)
79 m_renderer->clear();
80}
81
82void AudioTrackPrivateMediaStream::play()
83{
84 if (m_shouldPlay)
85 return;
86
87 m_shouldPlay = true;
88 updateRenderer();
89}
90
91void AudioTrackPrivateMediaStream::pause()
92{
93 m_shouldPlay = false;
94 updateRenderer();
95}

116float AudioTrackPrivateMediaStream::volume() const
117{
118 if (m_renderer)

145void AudioTrackPrivateMediaStream::updateRenderer()
146{
147 if (!m_shouldPlay || !volume() || m_muted || streamTrack().muted() || streamTrack().ended() || !streamTrack().enabled()) {
148 stopRenderer();
149 return;
150 }
151 startRenderer();
152}
153
154void AudioTrackPrivateMediaStream::startRenderer()
155{
156 ASSERT(isMainThread());
157 if (m_isPlaying || !m_renderer)
158 return;
159
160 m_isPlaying = true;
161 m_renderer->start([protectedThis = makeRef(*this)] {
162 if (protectedThis->m_isPlaying)
163 protectedThis->m_audioSource->addAudioSampleObserver(protectedThis.get());
164 });
165}
166
167void AudioTrackPrivateMediaStream::stopRenderer()
168{
169 ASSERT(isMainThread());
170 if (!m_isPlaying)
171 return;
172
173 m_isPlaying = false;
174 m_audioSource->removeAudioSampleObserver(*this);
175 if (m_renderer)
176 m_renderer->stop();
177}
178
179void AudioTrackPrivateMediaStream::createNewRenderer()
180{
181 bool isPlaying = m_isPlaying;
182 stopRenderer();
183
184 float volume = this->volume();
185 m_renderer = createRenderer(*this);
186 if (m_renderer)
187 m_renderer->setVolume(volume);
188
189 if (isPlaying)
190 startRenderer();
191}
