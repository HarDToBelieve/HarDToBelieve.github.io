26#pragma once
27
28#if ENABLE(MEDIA_STREAM)
29
30#include "AudioTrackPrivate.h"
31#include "MediaStreamTrackPrivate.h"
32
33namespace WebCore {
34
35class AudioMediaStreamTrackRenderer;
36
37class AudioTrackPrivateMediaStream final
38 : public AudioTrackPrivate
39 , public MediaStreamTrackPrivate::Observer
40 , private RealtimeMediaSource::AudioSampleObserver {
41 WTF_MAKE_NONCOPYABLE(AudioTrackPrivateMediaStream)
42public:
43 static Ref<AudioTrackPrivateMediaStream> create(MediaStreamTrackPrivate& streamTrack)
44 {
45 return adoptRef(*new AudioTrackPrivateMediaStream(streamTrack));
46 }
47 ~AudioTrackPrivateMediaStream();
48
49 void setTrackIndex(int index) { m_index = index; }
50 void setAudioOutputDevice(const String&);
51
52 MediaStreamTrackPrivate& streamTrack() { return m_streamTrack.get(); }
53
54 void clear();
55
56 void play();
57 void pause();
58 bool isPlaying() const { return m_isPlaying; }
59 bool shouldPlay() const { return m_shouldPlay; }
60
61 void setVolume(float);
62 float volume() const;
63
64 void setMuted(bool);
65 bool muted() const { return m_muted; }
66
67#if !RELEASE_LOG_DISABLED
68 const char* logClassName() const final { return "AudioTrackPrivateMediaStream"; }
69#endif
70
71private:
72 explicit AudioTrackPrivateMediaStream(MediaStreamTrackPrivate&);
73
74 static std::unique_ptr<AudioMediaStreamTrackRenderer> createRenderer(AudioTrackPrivateMediaStream&);
75
76 // AudioTrackPrivate
77 Kind kind() const final { return Kind::Main; }
78 AtomString id() const final { return m_id; }
79 AtomString label() const final { return m_label; }
80 int trackIndex() const final { return m_index; }
81 bool isBackedByMediaStreamTrack() const final { return true; }
82
83 // MediaStreamTrackPrivate::Observer
84 void trackEnded(MediaStreamTrackPrivate&) final;
85 void trackMutedChanged(MediaStreamTrackPrivate&) final;
86 void trackEnabledChanged(MediaStreamTrackPrivate&) final;
87 void trackSettingsChanged(MediaStreamTrackPrivate&) final { }
88
89 // RealtimeMediaSource::AudioSampleObserver
90 void audioSamplesAvailable(const MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t) final;
91
92 void startRenderer();
93 void stopRenderer();
94 void updateRenderer();
95 void createNewRenderer();
96
97 // Main thread writable members
98 bool m_isPlaying { false };
99 bool m_shouldPlay { false };
100 bool m_muted { false };
101 bool m_isCleared { false };
102
103 Ref<MediaStreamTrackPrivate> m_streamTrack;
104 Ref<RealtimeMediaSource> m_audioSource;
105 AtomString m_id;
106 AtomString m_label;
107 int m_index { 0 };
108
109 // Audio thread members
110 std::unique_ptr<AudioMediaStreamTrackRenderer> m_renderer;
111};
112
113}
114
115SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::AudioTrackPrivateMediaStream)
116 static bool isType(const WebCore::AudioTrackPrivate& track) { return track.isBackedByMediaStreamTrack(); }
117SPECIALIZE_TYPE_TRAITS_END()

119#endif // ENABLE(MEDIA_STREAM)
