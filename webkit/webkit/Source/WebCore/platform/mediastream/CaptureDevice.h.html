30namespace WebCore {

32class CaptureDevice {
33public:
34 enum class DeviceType { Unknown, Microphone, Speaker, Camera, Screen, Window };
35
36 CaptureDevice(const String& persistentId, DeviceType type, const String& label, const String& groupId = emptyString(), bool isEnabled = false, bool isDefault = false, bool isMock = false)
37 : m_persistentId(persistentId)
38 , m_type(type)
39 , m_label(label)
40 , m_groupId(groupId)
41 , m_enabled(isEnabled)
42 , m_default(isDefault)
43 , m_isMockDevice(isMock)
44 {
45 }
46
47 CaptureDevice() = default;
48
49 const String& persistentId() const { return m_persistentId; }
50
51 const String& label() const
52 {
53 static NeverDestroyed<String> airPods(MAKE_STATIC_STRING_IMPL("AirPods"));
54
55 if ((m_type == DeviceType::Microphone || m_type == DeviceType::Speaker) && m_label.contains(airPods))
56 return airPods;
57
58 return m_label;
59 }
60
61 void setGroupId(const String& groupId) { m_groupId = groupId; }
62 const String& groupId() const { return m_groupId; }
63
64 DeviceType type() const { return m_type; }
65
66 bool enabled() const { return m_enabled; }
67 void setEnabled(bool enabled) { m_enabled = enabled; }
68
69 bool isDefault() const { return m_default; }
70 void setIsDefault(bool isDefault) { m_default = isDefault; }
71
72 bool isMockDevice() const { return m_isMockDevice; }
73 void setIsMockDevice(bool isMockDevice) { m_isMockDevice = isMockDevice; }
74
75 explicit operator bool() const { return m_type != DeviceType::Unknown; }
76
77 CaptureDevice isolatedCopy() &&;
78
79#if ENABLE(MEDIA_STREAM)
80 template<class Encoder>
81 void encode(Encoder& encoder) const
82 {
83 encoder << m_persistentId;
84 encoder << m_label;
85 encoder << m_groupId;
86 encoder << m_enabled;
87 encoder << m_default;
88 encoder << m_type;
89 encoder << m_isMockDevice;
90 }
91
92 template <class Decoder>
93 static std::optional<CaptureDevice> decode(Decoder& decoder)
94 {
95 std::optional<String> persistentId;
96 decoder >> persistentId;
97 if (!persistentId)
98 return std::nullopt;
99
100 std::optional<String> label;
101 decoder >> label;
102 if (!label)
103 return std::nullopt;
104
105 std::optional<String> groupId;
106 decoder >> groupId;
107 if (!groupId)
108 return std::nullopt;
109
110 std::optional<bool> enabled;
111 decoder >> enabled;
112 if (!enabled)
113 return std::nullopt;
114
115 std::optional<bool> isDefault;
116 decoder >> isDefault;
117 if (!isDefault)
118 return std::nullopt;
119
120 std::optional<CaptureDevice::DeviceType> type;
121 decoder >> type;
122 if (!type)
123 return std::nullopt;
124
125 std::optional<bool> isMockDevice;
126 decoder >> isMockDevice;
127 if (!isMockDevice)
128 return std::nullopt;
129
130 std::optional<CaptureDevice> device = {{ WTFMove(*persistentId), WTFMove(*type), WTFMove(*label), WTFMove(*groupId) }};
131 device->setEnabled(*enabled);
132 device->setIsDefault(*isDefault);
133 device->setIsMockDevice(*isMockDevice);
134 return device;
135 }
136#endif
137
138protected:
139 String m_persistentId;
140 DeviceType m_type { DeviceType::Unknown };
141 String m_label;
142 String m_groupId;
143 bool m_enabled { false };
144 bool m_default { false };
145 bool m_isMockDevice { false };
146};
147
148inline bool haveDevicesChanged(const Vector<CaptureDevice>& oldDevices, const Vector<CaptureDevice>& newDevices)
149{
150 if (oldDevices.size() != newDevices.size())
151 return true;
152
153 for (auto& newDevice : newDevices) {
154 if (newDevice.type() != CaptureDevice::DeviceType::Camera && newDevice.type() != CaptureDevice::DeviceType::Microphone)
155 continue;
156
157 auto index = oldDevices.findMatching([&newDevice](auto& oldDevice) {
158 return newDevice.persistentId() == oldDevice.persistentId() && newDevice.enabled() != oldDevice.enabled();
159 });
160
161 if (index == notFound)
162 return true;
163 }
164
165 return false;
166}

168inline CaptureDevice CaptureDevice::isolatedCopy() &&
169{
170 return {
171 WTFMove(m_persistentId).isolatedCopy(),
172 m_type,
173 WTFMove(m_label).isolatedCopy(),
174 WTFMove(m_groupId).isolatedCopy(),
175 m_enabled,
176 m_default,
177 m_isMockDevice
178 };
179}
