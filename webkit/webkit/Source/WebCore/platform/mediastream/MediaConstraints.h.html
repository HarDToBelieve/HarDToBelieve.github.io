44class MediaConstraint {
45public:
46 enum class DataType : uint8_t { None, Integer, Double, Boolean, String };
47
48 bool isInt() const { return m_dataType == DataType::Integer; }
49 bool isDouble() const { return m_dataType == DataType::Double; }
50 bool isBoolean() const { return m_dataType == DataType::Boolean; }
51 bool isString() const { return m_dataType == DataType::String; }
52
53 DataType dataType() const { return m_dataType; }
54 MediaConstraintType constraintType() const { return m_constraintType; }
55 const String& name() const { return m_name; }
56
57 template <class Encoder> void encode(Encoder& encoder) const
58 {
59 encoder << m_constraintType;
60 encoder << m_name;
61 encoder << m_dataType;
62 }
63
64 template <class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder& decoder, MediaConstraint& constraint)
65 {
66 if (!decoder.decode(constraint.m_constraintType))
67 return false;
68
69 if (!decoder.decode(constraint.m_name))
70 return false;
71
72 if (!decoder.decode(constraint.m_dataType))
73 return false;
74
75 return true;
76 }
77

80protected:
81 MediaConstraint(const String& name, MediaConstraintType constraintType, DataType dataType)
82 : m_name(name)
83 , m_constraintType(constraintType)
84 , m_dataType(dataType)

89 ~MediaConstraint() = default;

97template<class ValueType>
98class NumericConstraint : public MediaConstraint {
99public:
100 void setMin(ValueType value) { m_min = value; }
101 void setMax(ValueType value) { m_max = value; }
102 void setExact(ValueType value) { m_exact = value; }
103 void setIdeal(ValueType value) { m_ideal = value; }
104
105 bool getMin(ValueType& min) const
106 {
107 if (!m_min)
108 return false;
109
110 min = m_min.value();
111 return true;
112 }
113
114 bool getMax(ValueType& max) const
115 {
116 if (!m_max)
117 return false;
118
119 max = m_max.value();
120 return true;
121 }
122
123 bool getExact(ValueType& exact) const
124 {
125 if (!m_exact)
126 return false;
127
128 exact = m_exact.value();
129 return true;
130 }
131
132 bool getIdeal(ValueType& ideal) const
133 {
134 if (!m_ideal)
135 return false;
136
137 ideal = m_ideal.value();
138 return true;
139 }
140
141 bool nearlyEqual(double a, double b) const
142 {
143 // Don't require strict equality when comparing constraints, or many floating point constraint values,
144 // e.g. "aspectRatio: 1.333", will never match.
145 const double epsilon = 0.00001;
146 return std::abs(a - b) <= epsilon;
147 }
148
149 double fitnessDistance(ValueType rangeMin, ValueType rangeMax) const
150 {
151 // https://w3c.github.io/mediacapture-main/#dfn-applyconstraints
152 // 1. If the constraint is not supported by the browser, the fitness distance is 0.
153 if (isEmpty())
154 return 0;
155
156 // 2. If the constraint is required ('min', 'max', or 'exact'), and the settings
157 // dictionary's value for the constraint does not satisfy the constraint, the
158 // fitness distance is positive infinity.
159 bool valid = validForRange(rangeMin, rangeMax);
160 if (m_exact) {
161 if (valid && m_min && m_exact.value() < m_min.value())
162 valid = false;
163 if (valid && m_max && m_exact.value() > m_max.value())
164 valid = false;
165 if (!valid)
166 return std::numeric_limits<double>::infinity();
167 }
168
169 if (m_min) {
170 if (valid && m_max && m_min.value() > m_max.value())
171 valid = false;
172 if (!valid)
173 return std::numeric_limits<double>::infinity();
174 }
175
176 if (m_max) {
177 if (valid && m_min && m_max.value() < m_min.value())
178 valid = false;
179 if (!valid)
180 return std::numeric_limits<double>::infinity();
181 }
182
183 // 3. If no ideal value is specified, the fitness distance is 0.
184 if (!m_ideal)
185 return 0;
186
187 // 4. For all positive numeric non-required constraints (such as height, width, frameRate,
188 // aspectRatio, sampleRate and sampleSize), the fitness distance is the result of the formula
189 //
190 // (actual == ideal) ? 0 : |actual - ideal| / max(|actual|,|ideal|)
191 ValueType ideal = m_ideal.value();
192 if (ideal >= rangeMin && ideal <= rangeMax)
193 return 0;
194
195 ideal = ideal > std::max(rangeMin, rangeMax) ? rangeMax : rangeMin;
196 return static_cast<double>(std::abs(ideal - m_ideal.value())) / std::max(std::abs(ideal), std::abs(m_ideal.value()));
197 }
198
199 double fitnessDistance(const Vector<ValueType>& discreteCapabilityValues) const
200 {
201 double minDistance = std::numeric_limits<double>::infinity();
202
203 for (auto& value : discreteCapabilityValues) {
204 auto distance = fitnessDistance(value, value);
205 if (distance < minDistance)
206 minDistance = distance;
207 }
208
209 return minDistance;
210 }
211
212 bool validForRange(ValueType rangeMin, ValueType rangeMax) const
213 {
214 if (isEmpty())
215 return false;
216
217 if (m_exact) {
218 const ValueType exact = m_exact.value();
219 if (exact < rangeMin && !nearlyEqual(exact, rangeMin))
220 return false;
221 if (exact > rangeMax && !nearlyEqual(exact, rangeMax))
222 return false;
223 }
224
225 if (m_min) {
226 const ValueType constraintMin = m_min.value();
227 if (constraintMin > rangeMax && !nearlyEqual(constraintMin, rangeMax))
228 return false;
229 }
230
231 if (m_max) {
232 const ValueType constraintMax = m_max.value();
233 if (constraintMax < rangeMin && !nearlyEqual(constraintMax, rangeMin))
234 return false;
235 }
236
237 return true;
238 }
239
240 ValueType find(const WTF::Function<bool(ValueType)>& function) const
241 {
242 if (m_min && function(m_min.value()))
243 return m_min.value();
244
245 if (m_max && function(m_max.value()))
246 return m_max.value();
247
248 if (m_exact && function(m_exact.value()))
249 return m_exact.value();
250
251 if (m_ideal && function(m_ideal.value()))
252 return m_ideal.value();
253
254 return 0;
255 }
256
257 ValueType valueForCapabilityRange(ValueType current, ValueType capabilityMin, ValueType capabilityMax) const
258 {
259 ValueType value { 0 };
260 ValueType min { capabilityMin };
261 ValueType max { capabilityMax };
262
263 if (m_exact) {
264 ASSERT(validForRange(capabilityMin, capabilityMax));
265 return m_exact.value();
266 }
267
268 if (m_min) {
269 value = m_min.value();
270 ASSERT(validForRange(value, capabilityMax));
271 if (value > min)
272 min = value;
273 if (value < min)
274 value = min;
275
276 // If there is no ideal, don't change if minimum is smaller than current.
277 if (!m_ideal && value < current)
278 value = current;
279 }
280
281 if (m_max) {
282 value = m_max.value();
283 ASSERT(validForRange(capabilityMin, value));
284 if (value < max)
285 max = value;
286 if (value > max)
287 value = max;
288 }
289
290 if (m_ideal)
291 value = std::max(min, std::min(max, m_ideal.value()));
292
293 return value;
294 }
295
296 std::optional<ValueType> valueForDiscreteCapabilityValues(ValueType current, const Vector<ValueType>& discreteCapabilityValues) const
297 {
298 std::optional<ValueType> value;
299 std::optional<ValueType> min;
300 std::optional<ValueType> max;
301
302 if (m_exact) {
303 ASSERT(discreteCapabilityValues.contains(m_exact.value()));
304 return m_exact.value();
305 }
306
307 if (m_min) {
308 auto index = discreteCapabilityValues.findMatching([&](ValueType value) { return m_min.value() >= value; });
309 if (index != notFound) {
310 min = value = discreteCapabilityValues[index];
311
312 // If there is no ideal, don't change if minimum is smaller than current.
313 if (!m_ideal && *value < current)
314 value = current;
315 }
316 }
317
318 if (m_max && m_max.value() >= discreteCapabilityValues[0]) {
319 for (auto& discreteValue : discreteCapabilityValues) {
320 if (m_max.value() <= discreteValue)
321 max = value = discreteValue;
322 }
323 }
324
325 if (m_ideal && discreteCapabilityValues.contains(m_ideal.value())) {
326 value = m_ideal.value();
327 if (max)
328 value = std::min(max.value(), *value);
329 if (min)
330 value = std::max(min.value(), *value);
331 }
332
333 return value;
334 }
335
336 bool isEmpty() const { return !m_min && !m_max && !m_exact && !m_ideal; }
337 bool isMandatory() const { return m_min || m_max || m_exact; }
338
339 template <class Encoder> void encode(Encoder& encoder) const
340 {
341 MediaConstraint::encode(encoder);
342
343 encoder << m_min;
344 encoder << m_max;
345 encoder << m_exact;
346 encoder << m_ideal;
347 }
348
349 template <class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder& decoder, NumericConstraint& constraint)
350 {
351 if (!MediaConstraint::decode(decoder, constraint))
352 return false;
353
354 if (!decoder.decode(constraint.m_min))
355 return false;
356 if (!decoder.decode(constraint.m_max))
357 return false;
358 if (!decoder.decode(constraint.m_exact))
359 return false;
360 if (!decoder.decode(constraint.m_ideal))
361 return false;
362
363 return true;
364 }
365
366protected:
367 NumericConstraint(const String& name, MediaConstraintType type, DataType dataType)
368 : MediaConstraint(name, type, dataType)
369 {
370 }
371
372 NumericConstraint() = default;
373
374 void innerMerge(const NumericConstraint& other)
375 {
376 if (other.isEmpty())
377 return;
378
379 ValueType value;
380 if (other.getExact(value))
381 m_exact = value;
382
383 if (other.getMin(value))
384 m_min = value;
385
386 if (other.getMax(value))
387 m_max = value;
388
389 // https://w3c.github.io/mediacapture-main/#constrainable-interface
390 // When processing advanced constraints:
391 // ... the User Agent must attempt to apply, individually, any 'ideal' constraints or
392 // a constraint given as a bare value for the property. Of these properties, it must
393 // satisfy the largest number that it can, in any order.
394 if (other.getIdeal(value)) {
395 if (!m_ideal || value > m_ideal.value())
396 m_ideal = value;
397 }
398 }
399
400 std::optional<ValueType> m_min;
401 std::optional<ValueType> m_max;
402 std::optional<ValueType> m_exact;
403 std::optional<ValueType> m_ideal;
404};

442class BooleanConstraint final : public MediaConstraint {
443public:
444 BooleanConstraint(const String& name, MediaConstraintType type)
445 : MediaConstraint(name, type, DataType::Boolean)
446 {
447 }
448
449 BooleanConstraint() = default;
450
451 void setExact(bool value) { m_exact = value; }
452 void setIdeal(bool value) { m_ideal = value; }
453
454 bool getExact(bool& exact) const
455 {
456 if (!m_exact)
457 return false;
458
459 exact = m_exact.value();
460 return true;
461 }
462
463 bool getIdeal(bool& ideal) const
464 {
465 if (!m_ideal)
466 return false;
467
468 ideal = m_ideal.value();
469 return true;
470 }
471
472 double fitnessDistance(bool value) const
473 {
474 // https://w3c.github.io/mediacapture-main/#dfn-applyconstraints
475 // 1. If the constraint is not supported by the browser, the fitness distance is 0.
476 if (isEmpty())
477 return 0;
478
479 // 2. If the constraint is required ('min', 'max', or 'exact'), and the settings
480 // dictionary's value for the constraint does not satisfy the constraint, the
481 // fitness distance is positive infinity.
482 if (m_exact && value != m_exact.value())
483 return std::numeric_limits<double>::infinity();
484
485 // 3. If no ideal value is specified, the fitness distance is 0.
486 if (!m_ideal || m_ideal.value() == value)
487 return 0;
488
489 // 5. For all string and enum non-required constraints (deviceId, groupId, facingMode,
490 // echoCancellation), the fitness distance is the result of the formula
491 // (actual == ideal) ? 0 : 1
492 return 1;
493 }
494
495 void merge(const MediaConstraint& other)
496 {
497 ASSERT(other.isBoolean());
498 const BooleanConstraint& typedOther = downcast<BooleanConstraint>(other);
499
500 if (typedOther.isEmpty())
501 return;
502
503 bool value;
504 if (typedOther.getExact(value))
505 m_exact = value;
506
507 if (typedOther.getIdeal(value)) {
508 if (!m_ideal || (value && !m_ideal.value()))
509 m_ideal = value;
510 }
511 }
512
513 bool isEmpty() const { return !m_exact && !m_ideal; };
514 bool isMandatory() const { return bool(m_exact); }
515
516 template <class Encoder> void encode(Encoder& encoder) const
517 {
518 MediaConstraint::encode(encoder);
519 encoder << m_exact;
520 encoder << m_ideal;
521 }
522
523 template <class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder& decoder, BooleanConstraint& constraint)
524 {
525 if (!MediaConstraint::decode(decoder, constraint))
526 return false;
527
528 if (!decoder.decode(constraint.m_exact))
529 return false;
530 if (!decoder.decode(constraint.m_ideal))
531 return false;
532
533 return true;
534 }
535
536 void logAsBoolean() const;
537
538private:
539 std::optional<bool> m_exact;
540 std::optional<bool> m_ideal;
541};

802private:
803 class ConstraintHolder {
804 public:
805 static ConstraintHolder create(const MediaConstraint& value) { return ConstraintHolder(value); }
806
807 ~ConstraintHolder()
808 {
809 if (m_value.asRaw) {
810 switch (dataType()) {
811 case MediaConstraint::DataType::Integer:
812 delete m_value.asInteger;
813 break;
814 case MediaConstraint::DataType::Double:
815 delete m_value.asDouble;
816 break;
817 case MediaConstraint::DataType::Boolean:
818 delete m_value.asBoolean;
819 break;
820 case MediaConstraint::DataType::String:
821 delete m_value.asString;
822 break;
823 case MediaConstraint::DataType::None:
824 ASSERT_NOT_REACHED();
825 break;
826 }
827 }
828#ifndef NDEBUG
