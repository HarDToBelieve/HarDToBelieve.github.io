49Ref<MediaStreamPrivate> MediaStreamPrivate::create(Ref<const Logger>&& logger, Ref<RealtimeMediaSource>&& source)
50{
51 auto loggerCopy = logger.copyRef();
52 return MediaStreamPrivate::create(WTFMove(logger), MediaStreamTrackPrivateVector::from(MediaStreamTrackPrivate::create(WTFMove(loggerCopy), WTFMove(source))));
53}
54
55Ref<MediaStreamPrivate> MediaStreamPrivate::create(Ref<const Logger>&& logger, RefPtr<RealtimeMediaSource>&& audioSource, RefPtr<RealtimeMediaSource>&& videoSource)
56{
57 MediaStreamTrackPrivateVector tracks;
58 tracks.reserveInitialCapacity(2);
59
60 if (audioSource)
61 tracks.uncheckedAppend(MediaStreamTrackPrivate::create(logger.copyRef(), audioSource.releaseNonNull()));
62 if (videoSource)
63 tracks.uncheckedAppend(MediaStreamTrackPrivate::create(logger.copyRef(), videoSource.releaseNonNull()));
64
65 return MediaStreamPrivate::create(WTFMove(logger), tracks);
66}

68MediaStreamPrivate::MediaStreamPrivate(Ref<const Logger>&& logger, const MediaStreamTrackPrivateVector& tracks, String&& id)
69 : m_id(WTFMove(id))
70#if !RELEASE_LOG_DISABLED
71 , m_logger(WTFMove(logger))
72 , m_logIdentifier(uniqueLogIdentifier())
73#endif
74{
75 UNUSED_PARAM(logger);
76 ASSERT(!m_id.isEmpty());
77
78 for (auto& track : tracks) {
79 track->addObserver(*this);
80 m_trackSet.add(track->id(), track);
81 }
82 m_isActive = computeActiveState();
83 updateActiveVideoTrack();
84}
85
86MediaStreamPrivate::~MediaStreamPrivate()
87{
88 for (auto& track : m_trackSet.values())
89 track->removeObserver(*this);
90}

128bool MediaStreamPrivate::computeActiveState()
129{
130 return WTF::anyOf(m_trackSet, [](auto& track) { return !track.value->ended(); });
131}
132
133void MediaStreamPrivate::updateActiveState()
134{
135 bool newActiveState = computeActiveState();
136
137 updateActiveVideoTrack();
138
139 // A stream is active if it has at least one un-ended track.
140 if (newActiveState == m_isActive)
141 return;
142
143 m_isActive = newActiveState;
144
145 forEachObserver([](auto& observer) {
146 observer.activeStatusChanged();
147 });
148}
149
150void MediaStreamPrivate::addTrack(Ref<MediaStreamTrackPrivate>&& track)
151{
152 if (m_trackSet.contains(track->id()))
153 return;
154
155 ALWAYS_LOG(LOGIDENTIFIER, track->logIdentifier());
156
157 auto& trackRef = track.get();
158 track->addObserver(*this);
159 m_trackSet.add(track->id(), WTFMove(track));
160
161 forEachObserver([&trackRef](auto& observer) {
162 observer.didAddTrack(trackRef);
163 });
164
165 updateActiveState();
166 characteristicsChanged();
167}
168
169void MediaStreamPrivate::removeTrack(MediaStreamTrackPrivate& track)
170{
171 if (!m_trackSet.remove(track.id()))
172 return;
173
174 ALWAYS_LOG(LOGIDENTIFIER, track.logIdentifier());
175 track.removeObserver(*this);
176
177 forEachObserver([&track](auto& observer) {
178 observer.didRemoveTrack(track);
179 });
180
181 updateActiveState();
182 characteristicsChanged();
183}

235FloatSize MediaStreamPrivate::intrinsicSize() const
236{
237 FloatSize size;
238
239 if (m_activeVideoTrack) {
240 const RealtimeMediaSourceSettings& setting = m_activeVideoTrack->settings();
241 size.setWidth(setting.width());
242 size.setHeight(setting.height());
243 }
244
245 return size;
246}
247
248void MediaStreamPrivate::updateActiveVideoTrack()
249{
250 m_activeVideoTrack = nullptr;
251 for (auto& track : m_trackSet.values()) {
252 if (!track->ended() && track->type() == RealtimeMediaSource::Type::Video) {
253 m_activeVideoTrack = track.get();
254 break;
255 }
256 }
257}
