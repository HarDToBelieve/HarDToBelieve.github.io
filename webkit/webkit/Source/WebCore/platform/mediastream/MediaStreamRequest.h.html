28#include <wtf/EnumTraits.h>

30#if ENABLE(MEDIA_STREAM)
31
32#include "MediaConstraints.h"
33
34namespace WebCore {
35
36struct MediaStreamRequest {
37 enum class Type { UserMedia, DisplayMedia };
38 Type type { Type::UserMedia };
39 MediaConstraints audioConstraints;
40 MediaConstraints videoConstraints;
41 bool isUserGesturePriviledged { false };
42
43 template<class Encoder>
44 void encode(Encoder& encoder) const
45 {
46 encoder << type;
47 encoder << audioConstraints;
48 encoder << videoConstraints;
49 encoder << isUserGesturePriviledged;
50 }
51
52 template <class Decoder> static std::optional<MediaStreamRequest> decode(Decoder& decoder)
53 {
54 MediaStreamRequest request;
55 if (decoder.decode(request.type) && decoder.decode(request.audioConstraints) && decoder.decode(request.videoConstraints) && decoder.decode(request.isUserGesturePriviledged))
56 return request;
57
58 return std::nullopt;
59 }
60};
61
62} // namespace WebCore
