28#pragma once
29
30#if ENABLE(MEDIA_STREAM)
31
32#include "RealtimeMediaSource.h"
33#include <wtf/LoggerHelper.h>
34#include <wtf/RefCounted.h>
35#include <wtf/WeakHashSet.h>
36
37namespace WebCore {
38
39class GraphicsContext;
40class MediaSample;
41class RealtimeMediaSourceCapabilities;
42class WebAudioSourceProvider;
43
44class MediaStreamTrackPrivate final
45 : public RefCounted<MediaStreamTrackPrivate>
46 , public RealtimeMediaSource::Observer
47#if !RELEASE_LOG_DISABLED
48 , public LoggerHelper
49#endif
50{
51public:
52 class Observer : public CanMakeWeakPtr<Observer> {
53 public:
54 virtual ~Observer() = default;
55
56 virtual void trackStarted(MediaStreamTrackPrivate&) { };
57 virtual void trackEnded(MediaStreamTrackPrivate&) = 0;
58 virtual void trackMutedChanged(MediaStreamTrackPrivate&) = 0;
59 virtual void trackSettingsChanged(MediaStreamTrackPrivate&) = 0;
60 virtual void trackEnabledChanged(MediaStreamTrackPrivate&) = 0;
61 virtual void readyStateChanged(MediaStreamTrackPrivate&) { };
62 };
63
64 static Ref<MediaStreamTrackPrivate> create(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&);
65 static Ref<MediaStreamTrackPrivate> create(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&, String&& id);
66
67 WEBCORE_EXPORT virtual ~MediaStreamTrackPrivate();
68
69 const String& id() const { return m_id; }
70 const String& label() const;
71
72 bool isActive() const { return enabled() && !ended() && !muted(); }
73
74 bool ended() const { return m_isEnded; }
75
76 enum class HintValue { Empty, Speech, Music, Motion, Detail, Text };
77 HintValue contentHint() const { return m_contentHint; }
78 void setContentHint(HintValue);
79
80 void startProducingData() { m_source->start(); }
81 void stopProducingData() { m_source->stop(); }
82 bool isProducingData() { return m_source->isProducingData(); }
83
84 bool isIsolated() const { return m_source->isIsolated(); }
85
86 bool muted() const;
87 void setMuted(bool muted) { m_source->setMuted(muted); }
88
89 bool isCaptureTrack() const;
90
91 bool enabled() const { return m_isEnabled; }
92 void setEnabled(bool);
93
94 Ref<MediaStreamTrackPrivate> clone();
95
96 RealtimeMediaSource& source() { return m_source.get(); }
97 const RealtimeMediaSource& source() const { return m_source.get(); }
98 WEBCORE_EXPORT RealtimeMediaSource::Type type() const;
99
100 void endTrack();
101
102 void addObserver(Observer&);
103 void removeObserver(Observer&);
104 bool hasObserver(Observer& observer) const { return m_observers.contains(observer); }
105
106 WEBCORE_EXPORT const RealtimeMediaSourceSettings& settings() const;
107 const RealtimeMediaSourceCapabilities& capabilities() const;
108
109 void applyConstraints(const MediaConstraints&, RealtimeMediaSource::ApplyConstraintsHandler&&);
110
111 RefPtr<WebAudioSourceProvider> createAudioSourceProvider();
112
113 void paintCurrentFrameInContext(GraphicsContext&, const FloatRect&);
114
115 enum class ReadyState { None, Live, Ended };
116 ReadyState readyState() const { return m_readyState; }
117
118 void setIdForTesting(String&& id) { m_id = WTFMove(id); }
119
120#if !RELEASE_LOG_DISABLED
121 const Logger& logger() const final { return m_logger; }
122 const void* logIdentifier() const final { return m_logIdentifier; }
123#endif
124
125private:
126 MediaStreamTrackPrivate(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&, String&& id);
127
128 // RealtimeMediaSource::Observer
129 void sourceStarted() final;
130 void sourceStopped() final;
131 void sourceMutedChanged() final;
132 void sourceSettingsChanged() final;
133 bool preventSourceFromStopping() final;
134 void audioUnitWillStart() final;
135 void hasStartedProducingData() final;
136
137 void updateReadyState();
138
139 void forEachObserver(const Function<void(Observer&)>&);
140
141#if !RELEASE_LOG_DISABLED
142 const char* logClassName() const final { return "MediaStreamTrackPrivate"; }
143 WTFLogChannel& logChannel() const final;
144#endif
145
146 WeakHashSet<Observer> m_observers;
147 Ref<RealtimeMediaSource> m_source;
148
149 String m_id;
150 ReadyState m_readyState { ReadyState::None };
151 bool m_isEnabled { true };
152 bool m_isEnded { false };
153 bool m_hasStartedProducingData { false };
154 HintValue m_contentHint { HintValue::Empty };
155 Ref<const Logger> m_logger;
156#if !RELEASE_LOG_DISABLED
157 const void* m_logIdentifier;
158#endif
159};
160
161typedef Vector<RefPtr<MediaStreamTrackPrivate>> MediaStreamTrackPrivateVector;
162
163} // namespace WebCore
164
165#endif // ENABLE(MEDIA_STREAM)
