42class RTCSessionDescriptionDescriptor;

44class RTCSessionDescriptionRequest : public RefCounted<RTCSessionDescriptionRequest> {
45public:
46 class ExtraData : public RefCounted<ExtraData> {

48 virtual ~ExtraData() = default;
49 };
50
51 virtual ~RTCSessionDescriptionRequest() = default;
52
53 virtual void requestSucceeded(RTCSessionDescriptionDescriptor&) = 0;
54 virtual void requestFailed(const String& error) = 0;
55
56 ExtraData* extraData() const { return m_extraData.get(); }
57 void setExtraData(RefPtr<ExtraData>&& extraData) { m_extraData = extraData; }
58
59protected:
60 RTCSessionDescriptionRequest() = default;
61
62private:
63 RefPtr<ExtraData> m_extraData;
64};

66} // namespace WebCore
