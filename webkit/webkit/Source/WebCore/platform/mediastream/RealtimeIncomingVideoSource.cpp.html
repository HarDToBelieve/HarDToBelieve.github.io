38RealtimeIncomingVideoSource::RealtimeIncomingVideoSource(rtc::scoped_refptr<webrtc::VideoTrackInterface>&& videoTrack, String&& videoTrackId)
39 : RealtimeMediaSource(Type::Video, "remote video"_s, WTFMove(videoTrackId))
40 , m_videoTrack(WTFMove(videoTrack))
41{
42 ASSERT(m_videoTrack);
43
44 RealtimeMediaSourceSupportedConstraints constraints;
45 constraints.setSupportsWidth(true);
46 constraints.setSupportsHeight(true);
47 m_currentSettings = RealtimeMediaSourceSettings { };
48 m_currentSettings->setSupportedConstraints(WTFMove(constraints));
49
50 m_videoTrack->RegisterObserver(this);
51}
52
53RealtimeIncomingVideoSource::~RealtimeIncomingVideoSource()
54{
55 stop();
56 m_videoTrack->UnregisterObserver(this);
57}
58
59void RealtimeIncomingVideoSource::startProducingData()
60{
61 m_videoTrack->AddOrUpdateSink(this, rtc::VideoSinkWants());
62}
63
64void RealtimeIncomingVideoSource::stopProducingData()
65{
66 m_videoTrack->RemoveSink(this);
67}
68
69void RealtimeIncomingVideoSource::OnChanged()
70{
71 callOnMainThread([protectedThis = makeRef(*this)] {
72 if (protectedThis->m_videoTrack->state() == webrtc::MediaStreamTrackInterface::kEnded)
73 protectedThis->end();
74 });
75}
76
77const RealtimeMediaSourceCapabilities& RealtimeIncomingVideoSource::capabilities()
78{
79 return RealtimeMediaSourceCapabilities::emptyCapabilities();
80}
81
82const RealtimeMediaSourceSettings& RealtimeIncomingVideoSource::settings()
83{
84 if (m_currentSettings)
85 return m_currentSettings.value();
86
87 RealtimeMediaSourceSupportedConstraints constraints;
88 constraints.setSupportsWidth(true);
89 constraints.setSupportsHeight(true);
90
91 RealtimeMediaSourceSettings settings;
92 auto& size = this->size();
93 settings.setWidth(size.width());
94 settings.setHeight(size.height());
95 settings.setSupportedConstraints(constraints);
96
97 m_currentSettings = WTFMove(settings);
98 return m_currentSettings.value();
99}
100
101void RealtimeIncomingVideoSource::settingsDidChange(OptionSet<RealtimeMediaSourceSettings::Flag> settings)
102{
103 if (settings.containsAny({ RealtimeMediaSourceSettings::Flag::Width, RealtimeMediaSourceSettings::Flag::Height }))
104 m_currentSettings = std::nullopt;
105}
