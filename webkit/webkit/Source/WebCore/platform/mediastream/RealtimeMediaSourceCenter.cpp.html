72void RealtimeMediaSourceCenter::createMediaStream(Ref<const Logger>&& logger, NewMediaStreamHandler&& completionHandler, String&& hashSalt, CaptureDevice&& audioDevice, CaptureDevice&& videoDevice, const MediaStreamRequest& request)
73{
74 Vector<Ref<RealtimeMediaSource>> audioSources;
75 Vector<Ref<RealtimeMediaSource>> videoSources;
76 String invalidConstraint;
77
78 RefPtr<RealtimeMediaSource> audioSource;
79 if (audioDevice) {
80 auto source = audioCaptureFactory().createAudioCaptureSource(WTFMove(audioDevice), String { hashSalt }, &request.audioConstraints);
81 if (!source) {
82 completionHandler(makeUnexpected(makeString("Failed to create MediaStream audio source: ", source.errorMessage)));
83 return;
84 }
85 audioSource = source.source();
86 }
87
88 RefPtr<RealtimeMediaSource> videoSource;
89 if (videoDevice) {
90 CaptureSourceOrError source;
91 if (videoDevice.type() == CaptureDevice::DeviceType::Camera)
92 source = videoCaptureFactory().createVideoCaptureSource(WTFMove(videoDevice), WTFMove(hashSalt), &request.videoConstraints);
93 else
94 source = displayCaptureFactory().createDisplayCaptureSource(WTFMove(videoDevice), &request.videoConstraints);
95
96 if (!source) {
97 completionHandler(makeUnexpected(makeString("Failed to create MediaStream video source: ", source.errorMessage)));
98 return;
99 }
100 videoSource = source.source();
101 }
102
103 CompletionHandler<void(String&&)> whenAudioSourceReady = [audioSource, videoSource = WTFMove(videoSource), logger = WTFMove(logger), completionHandler = WTFMove(completionHandler)](auto&& errorMessage) mutable {
104 if (!errorMessage.isEmpty())
105 return completionHandler(makeUnexpected(makeString("Failed to create MediaStream audio source: ", errorMessage)));
106 if (!videoSource)
107 return completionHandler(MediaStreamPrivate::create(WTFMove(logger), WTFMove(audioSource), WTFMove(videoSource)));
108
109 CompletionHandler<void(String&&)> whenVideoSourceReady = [audioSource = WTFMove(audioSource), videoSource, logger = WTFMove(logger), completionHandler = WTFMove(completionHandler)](auto&& errorMessage) mutable {
110 if (!errorMessage.isEmpty())
111 return completionHandler(makeUnexpected(makeString("Failed to create MediaStream video source: ", errorMessage)));
112 completionHandler(MediaStreamPrivate::create(WTFMove(logger), WTFMove(audioSource), WTFMove(videoSource)));
113 };
114 videoSource->whenReady(WTFMove(whenVideoSourceReady));
115 };
116 if (!audioSource)
117 return whenAudioSourceReady({ });
118 audioSource->whenReady(WTFMove(whenAudioSourceReady));
119}

121void RealtimeMediaSourceCenter::getMediaStreamDevices(CompletionHandler<void(Vector<CaptureDevice>&&)>&& completion)
122{
123 enumerateDevices(true, true, true, true, [this, completion = WTFMove(completion)]() mutable {
124 Vector<CaptureDevice> results;
125
126 results.appendVector(audioCaptureFactory().audioCaptureDeviceManager().captureDevices());
127 results.appendVector(videoCaptureFactory().videoCaptureDeviceManager().captureDevices());
128 results.appendVector(displayCaptureFactory().displayCaptureDeviceManager().captureDevices());
129 results.appendVector(audioCaptureFactory().speakerDevices());
130
131 completion(WTFMove(results));
132 });
133}

135static void addStringToSHA1(SHA1& sha1, const String& string)
136{
137 if (string.isEmpty())
138 return;
139
140 if (string.is8Bit() && string.isAllASCII()) {
141 const uint8_t nullByte = 0;
142 sha1.addBytes(string.characters8(), string.length());
143 sha1.addBytes(&nullByte, 1);
144 return;
145 }
146
147 auto utf8 = string.utf8();
148 sha1.addBytes(utf8.dataAsUInt8Ptr(), utf8.length() + 1); // Include terminating null byte.
149}

151String RealtimeMediaSourceCenter::hashStringWithSalt(const String& id, const String& hashSalt)
152{
153 if (id.isEmpty() || hashSalt.isEmpty())
154 return emptyString();
155
156 SHA1 sha1;
157
158 addStringToSHA1(sha1, id);
159 addStringToSHA1(sha1, hashSalt);
160
161 SHA1::Digest digest;
162 sha1.computeHash(digest);
163
164 return SHA1::hexDigest(digest).data();
165}

179void RealtimeMediaSourceCenter::captureDevicesChanged()
180{
181 ASSERT(isMainThread());
182
183 // When a device with camera and microphone is attached or detached, the CaptureDevice notification for
184 // the different devices won't arrive at the same time so delay a bit so we can coalesce the callbacks.
185 if (!m_debounceTimer.isActive())
186 m_debounceTimer.startOneShot(deviceChangeDebounceTimerInterval);
187}

190{
191 auto protectedThis = makeRef(*this);
192 m_observers.forEach([](auto& observer) {
193 observer.devicesChanged();
194 });
195}

197void RealtimeMediaSourceCenter::getDisplayMediaDevices(const MediaStreamRequest& request, Vector<DeviceInfo>& diaplayDeviceInfo, String& firstInvalidConstraint)
198{
199 if (!request.videoConstraints.isValid)
200 return;
201
202 String invalidConstraint;
203 for (auto& device : displayCaptureFactory().displayCaptureDeviceManager().captureDevices()) {
204 if (!device.enabled())
205 return;
206
207 auto sourceOrError = displayCaptureFactory().createDisplayCaptureSource(device, { });
208 if (sourceOrError && sourceOrError.captureSource->supportsConstraints(request.videoConstraints, invalidConstraint))
209 diaplayDeviceInfo.append({sourceOrError.captureSource->fitnessScore(), device});
210
211 if (!invalidConstraint.isEmpty() && firstInvalidConstraint.isEmpty())
212 firstInvalidConstraint = invalidConstraint;
213 }
214}

216void RealtimeMediaSourceCenter::getUserMediaDevices(const MediaStreamRequest& request, String&& hashSalt, Vector<DeviceInfo>& audioDeviceInfo, Vector<DeviceInfo>& videoDeviceInfo, String& firstInvalidConstraint)
217{
218 String invalidConstraint;
219 if (request.audioConstraints.isValid) {
220 for (auto& device : audioCaptureFactory().audioCaptureDeviceManager().captureDevices()) {
221 if (!device.enabled())
222 continue;
223
224 auto sourceOrError = audioCaptureFactory().createAudioCaptureSource(device, String { hashSalt }, { });
225 if (sourceOrError && sourceOrError.captureSource->supportsConstraints(request.audioConstraints, invalidConstraint))
226 audioDeviceInfo.append({sourceOrError.captureSource->fitnessScore(), device});
227
228 if (!invalidConstraint.isEmpty() && firstInvalidConstraint.isEmpty())
229 firstInvalidConstraint = invalidConstraint;
230 }
231 }
232
233 if (request.videoConstraints.isValid) {
234 for (auto& device : videoCaptureFactory().videoCaptureDeviceManager().captureDevices()) {
235 if (!device.enabled())
236 continue;
237
238 auto sourceOrError = videoCaptureFactory().createVideoCaptureSource(device, String { hashSalt }, { });
239 if (sourceOrError && sourceOrError.captureSource->supportsConstraints(request.videoConstraints, invalidConstraint))
240 videoDeviceInfo.append({sourceOrError.captureSource->fitnessScore(), device});
241
242 if (!invalidConstraint.isEmpty() && firstInvalidConstraint.isEmpty())
243 firstInvalidConstraint = invalidConstraint;
244 }
245 }

248void RealtimeMediaSourceCenter::enumerateDevices(bool shouldEnumerateCamera, bool shouldEnumerateDisplay, bool shouldEnumerateMicrophone, bool shouldEnumerateSpeakers, CompletionHandler<void()>&& callback)
249{
250 auto callbackAggregator = CallbackAggregator::create(WTFMove(callback));
251 if (shouldEnumerateCamera)
252 videoCaptureFactory().videoCaptureDeviceManager().computeCaptureDevices([callbackAggregator] { });
253 if (shouldEnumerateDisplay)
254 displayCaptureFactory().displayCaptureDeviceManager().computeCaptureDevices([callbackAggregator] { });
255 if (shouldEnumerateMicrophone)
256 audioCaptureFactory().audioCaptureDeviceManager().computeCaptureDevices([callbackAggregator] { });
257 if (shouldEnumerateSpeakers)
258 audioCaptureFactory().computeSpeakerDevices([callbackAggregator] { });
259}

261void RealtimeMediaSourceCenter::validateRequestConstraints(ValidConstraintsHandler&& validHandler, InvalidConstraintsHandler&& invalidHandler, const MediaStreamRequest& request, String&& deviceIdentifierHashSalt)
262{
263 bool shouldEnumerateCamera = request.videoConstraints.isValid;
264 bool shouldEnumerateDisplay = request.type == MediaStreamRequest::Type::DisplayMedia;
265 bool shouldEnumerateMicrophone = request.audioConstraints.isValid;
266 bool shouldEnumerateSpeakers = false;
267 enumerateDevices(shouldEnumerateCamera, shouldEnumerateDisplay, shouldEnumerateMicrophone, shouldEnumerateSpeakers, [this, validHandler = WTFMove(validHandler), invalidHandler = WTFMove(invalidHandler), request, deviceIdentifierHashSalt = WTFMove(deviceIdentifierHashSalt)]() mutable {
268 validateRequestConstraintsAfterEnumeration(WTFMove(validHandler), WTFMove(invalidHandler), request, WTFMove(deviceIdentifierHashSalt));
269 });
270}

272void RealtimeMediaSourceCenter::validateRequestConstraintsAfterEnumeration(ValidConstraintsHandler&& validHandler, InvalidConstraintsHandler&& invalidHandler, const MediaStreamRequest& request, String&& deviceIdentifierHashSalt)
273{
274 struct {
275 bool operator()(const DeviceInfo& a, const DeviceInfo& b)
276 {
277 return a.fitnessScore > b.fitnessScore;
278 }
279 } sortBasedOnFitnessScore;
280
281 Vector<DeviceInfo> audioDeviceInfo;
282 Vector<DeviceInfo> videoDeviceInfo;
283 String firstInvalidConstraint;
284
285 if (request.type == MediaStreamRequest::Type::DisplayMedia)
286 getDisplayMediaDevices(request, videoDeviceInfo, firstInvalidConstraint);
287 else
288 getUserMediaDevices(request, String { deviceIdentifierHashSalt }, audioDeviceInfo, videoDeviceInfo, firstInvalidConstraint);
289
290 if (request.audioConstraints.isValid && audioDeviceInfo.isEmpty()) {
291 WTFLogAlways("Audio capture was requested but no device was found amongst %zu devices", audioCaptureFactory().audioCaptureDeviceManager().captureDevices().size());
292 request.audioConstraints.mandatoryConstraints.forEach([](auto& constraint) { constraint.log(); });
293
294 invalidHandler(firstInvalidConstraint);
295 return;
296 }
297
298 if (request.videoConstraints.isValid && videoDeviceInfo.isEmpty()) {
299 WTFLogAlways("Video capture was requested but no device was found amongst %zu devices", videoCaptureFactory().videoCaptureDeviceManager().captureDevices().size());
300 request.videoConstraints.mandatoryConstraints.forEach([](auto& constraint) { constraint.log(); });
301
302 invalidHandler(firstInvalidConstraint);
303 return;
304 }
305
306 Vector<CaptureDevice> audioDevices;
307 if (!audioDeviceInfo.isEmpty()) {
308 std::sort(audioDeviceInfo.begin(), audioDeviceInfo.end(), sortBasedOnFitnessScore);
309 audioDevices = WTF::map(audioDeviceInfo, [] (auto& info) {
310 return info.device;
311 });
312 }
313
314 Vector<CaptureDevice> videoDevices;
315 if (!videoDeviceInfo.isEmpty()) {
316 std::sort(videoDeviceInfo.begin(), videoDeviceInfo.end(), sortBasedOnFitnessScore);
317 videoDevices = WTF::map(videoDeviceInfo, [] (auto& info) {
318 return info.device;
319 });
320 }
321
322 validHandler(WTFMove(audioDevices), WTFMove(videoDevices), WTFMove(deviceIdentifierHashSalt));
323}
