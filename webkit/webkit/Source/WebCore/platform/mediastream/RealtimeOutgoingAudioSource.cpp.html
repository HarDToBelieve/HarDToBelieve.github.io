41RealtimeOutgoingAudioSource::RealtimeOutgoingAudioSource(Ref<MediaStreamTrackPrivate>&& source)
42 : m_audioSource(WTFMove(source))
43{
44}
45
46RealtimeOutgoingAudioSource::~RealtimeOutgoingAudioSource()
47{
48 ASSERT(!m_audioSource->hasObserver(*this));
49#if ASSERT_ENABLED
50 Locker locker { m_sinksLock };
51#endif
52 ASSERT(m_sinks.isEmpty());
53
54 stop();
55}
56
57void RealtimeOutgoingAudioSource::observeSource()
58{
59 ASSERT(!m_audioSource->hasObserver(*this));
60 m_audioSource->addObserver(*this);
61 m_audioSource->source().addAudioSampleObserver(*this);
62 initializeConverter();
63}
64
65void RealtimeOutgoingAudioSource::unobserveSource()
66{
67 m_audioSource->source().removeAudioSampleObserver(*this);
68 m_audioSource->removeObserver(*this);
69}
70
71void RealtimeOutgoingAudioSource::setSource(Ref<MediaStreamTrackPrivate>&& newSource)
72{
73 ALWAYS_LOG("Changing source to ", newSource->logIdentifier());
74
75 ASSERT(!m_audioSource->hasObserver(*this));
76 m_audioSource = WTFMove(newSource);
77 sourceUpdated();
78}
79
80void RealtimeOutgoingAudioSource::initializeConverter()
81{
82 m_muted = m_audioSource->muted();
83 m_enabled = m_audioSource->enabled();
84}
85
86void RealtimeOutgoingAudioSource::sourceMutedChanged()
87{
88 m_muted = m_audioSource->muted();
89}
90
91void RealtimeOutgoingAudioSource::sourceEnabledChanged()
92{
93 m_enabled = m_audioSource->enabled();
94}
95
96void RealtimeOutgoingAudioSource::AddSink(webrtc::AudioTrackSinkInterface* sink)
97{
98 Locker locker { m_sinksLock };
99 m_sinks.add(sink);
100}
101
102void RealtimeOutgoingAudioSource::RemoveSink(webrtc::AudioTrackSinkInterface* sink)
103{
104 Locker locker { m_sinksLock };
105 m_sinks.remove(sink);
106}
107
108void RealtimeOutgoingAudioSource::sendAudioFrames(const void* audioData, int bitsPerSample, int sampleRate, size_t numberOfChannels, size_t numberOfFrames)
109{
110#if !RELEASE_LOG_DISABLED
111 if (!(++m_chunksSent % 200))
112 ALWAYS_LOG(LOGIDENTIFIER, "chunk ", m_chunksSent);
113#endif
114
115 Locker locker { m_sinksLock };
116 for (auto sink : m_sinks)
117 sink->OnData(audioData, bitsPerSample, sampleRate, numberOfChannels, numberOfFrames);
118}
