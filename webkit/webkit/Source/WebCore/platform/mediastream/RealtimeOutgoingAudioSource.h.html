40#include <webrtc/api/media_stream_interface.h>

44#include <wtf/Lock.h>
45#include <wtf/LoggerHelper.h>
46#include <wtf/ThreadSafeRefCounted.h>
47
48namespace webrtc {
49class AudioTrackInterface;
50class AudioTrackSinkInterface;
51}
52
53namespace WebCore {
54
55class RealtimeOutgoingAudioSource
56 : public ThreadSafeRefCounted<RealtimeOutgoingAudioSource, WTF::DestructionThread::Main>
57 , public webrtc::AudioSourceInterface
58 , private MediaStreamTrackPrivate::Observer
59 , private RealtimeMediaSource::AudioSampleObserver
60#if !RELEASE_LOG_DISABLED
61 , private LoggerHelper
62#endif
63{
64public:
65 static Ref<RealtimeOutgoingAudioSource> create(Ref<MediaStreamTrackPrivate>&& audioSource);
66
67 ~RealtimeOutgoingAudioSource();
68
69 void start() { observeSource(); }
70 void stop() { unobserveSource(); }
71
72 void setSource(Ref<MediaStreamTrackPrivate>&&);
73 MediaStreamTrackPrivate& source() const { return m_audioSource.get(); }
74
75protected:
76 explicit RealtimeOutgoingAudioSource(Ref<MediaStreamTrackPrivate>&&);
77
78 bool isSilenced() const { return m_muted || !m_enabled; }
79
80 void sendAudioFrames(const void* audioData, int bitsPerSample, int sampleRate, size_t numberOfChannels, size_t numberOfFrames);
81
82#if !RELEASE_LOG_DISABLED
83 // LoggerHelper API
84 const Logger& logger() const final { return m_audioSource->logger(); }
85 const void* logIdentifier() const final { return m_audioSource->logIdentifier(); }
86 const char* logClassName() const final { return "RealtimeOutgoingAudioSource"; }
87 WTFLogChannel& logChannel() const final;
88#endif
89
90private:
91 // webrtc::AudioSourceInterface API
92 void AddSink(webrtc::AudioTrackSinkInterface*) final;
93 void RemoveSink(webrtc::AudioTrackSinkInterface*) final;
94
95 void AddRef() const final { ref(); }
96 rtc::RefCountReleaseStatus Release() const final
97 {
98 auto result = refCount() - 1;
99 deref();
100 return result ? rtc::RefCountReleaseStatus::kOtherRefsRemained : rtc::RefCountReleaseStatus::kDroppedLastRef;
101 }
102
103 SourceState state() const final { return kLive; }
104 bool remote() const final { return false; }
105 void RegisterObserver(webrtc::ObserverInterface*) final { }
106 void UnregisterObserver(webrtc::ObserverInterface*) final { }
107
108 void observeSource();
109 void unobserveSource();
110
111 void sourceMutedChanged();
112 void sourceEnabledChanged();
113
114 virtual bool isReachingBufferedAudioDataHighLimit() { return false; };
115 virtual bool isReachingBufferedAudioDataLowLimit() { return false; };
116 virtual bool hasBufferedEnoughData() { return false; };
117 virtual void sourceUpdated() { }
118
119 // MediaStreamTrackPrivate::Observer API
120 void trackMutedChanged(MediaStreamTrackPrivate&) final { sourceMutedChanged(); }
121 void trackEnabledChanged(MediaStreamTrackPrivate&) final { sourceEnabledChanged(); }
122 void trackEnded(MediaStreamTrackPrivate&) final { }
123 void trackSettingsChanged(MediaStreamTrackPrivate&) final { }
124
125 void initializeConverter();
126
127 Ref<MediaStreamTrackPrivate> m_audioSource;
128 bool m_muted { false };
129 bool m_enabled { true };
130
131 mutable Lock m_sinksLock;
132 HashSet<webrtc::AudioTrackSinkInterface*> m_sinks WTF_GUARDED_BY_LOCK(m_sinksLock);
133
134#if !RELEASE_LOG_DISABLED
135 size_t m_chunksSent { 0 };
136#endif
137};

139} // namespace WebCore

141#endif // USE(LIBWEBRTC)
