39
40#include <webrtc/api/media_stream_interface.h>

44#include <wtf/LoggerHelper.h>
45#include <wtf/ThreadSafeRefCounted.h>
46
47namespace WebCore {
48
49class RealtimeOutgoingVideoSource
50 : public ThreadSafeRefCounted<RealtimeOutgoingVideoSource, WTF::DestructionThread::Main>
51 , public webrtc::VideoTrackSourceInterface
52 , private MediaStreamTrackPrivate::Observer
53 , private RealtimeMediaSource::VideoSampleObserver
54#if !RELEASE_LOG_DISABLED
55 , private LoggerHelper
56#endif
57{
58public:
59 static Ref<RealtimeOutgoingVideoSource> create(Ref<MediaStreamTrackPrivate>&& videoSource);
60 ~RealtimeOutgoingVideoSource();
61
62 void start() { observeSource(); }
63 void stop();
64 void setSource(Ref<MediaStreamTrackPrivate>&&);
65 MediaStreamTrackPrivate& source() const { return m_videoSource.get(); }
66
67 void AddRef() const final { ref(); }
68 rtc::RefCountReleaseStatus Release() const final
69 {
70 deref();
71 return rtc::RefCountReleaseStatus::kOtherRefsRemained;
72 }
73
74 void applyRotation();
75
76protected:
77 explicit RealtimeOutgoingVideoSource(Ref<MediaStreamTrackPrivate>&&);
78
79 void sendFrame(rtc::scoped_refptr<webrtc::VideoFrameBuffer>&&);
80 bool isSilenced() const { return m_muted || !m_enabled; }
81
82 virtual rtc::scoped_refptr<webrtc::VideoFrameBuffer> createBlackFrame(size_t width, size_t height) = 0;
83
84 bool m_shouldApplyRotation { false };
85 webrtc::VideoRotation m_currentRotation { webrtc::kVideoRotation_0 };
86
87#if !RELEASE_LOG_DISABLED
88 // LoggerHelper API
89 const Logger& logger() const final { return m_logger.get(); }
90 const void* logIdentifier() const final { return m_logIdentifier; }
91 const char* logClassName() const final { return "RealtimeOutgoingVideoSource"; }
92 WTFLogChannel& logChannel() const final;
93#endif
94
95private:
96 void sendBlackFramesIfNeeded();
97 void sendOneBlackFrame();
98 void initializeFromSource();
99 void updateBlackFramesSending();
100
101 void observeSource();
102 void unobserveSource();
103
104 using MediaStreamTrackPrivate::Observer::weakPtrFactory;
105 using WeakValueType = MediaStreamTrackPrivate::Observer::WeakValueType;
106
107 // Notifier API
108 void RegisterObserver(webrtc::ObserverInterface*) final { }
109 void UnregisterObserver(webrtc::ObserverInterface*) final { }
110
111 // VideoTrackSourceInterface API
112 bool is_screencast() const final { return false; }
113 absl::optional<bool> needs_denoising() const final { return absl::optional<bool>(); }
114 bool GetStats(Stats*) final { return false; };
115 bool SupportsEncodedOutput() const final { return false; }
116 void GenerateKeyFrame() final { }
117 void AddEncodedSink(rtc::VideoSinkInterface<webrtc::RecordableEncodedFrame>*) final { }
118 void RemoveEncodedSink(rtc::VideoSinkInterface<webrtc::RecordableEncodedFrame>*) final { }
119
120 // MediaSourceInterface API
121 SourceState state() const final { return SourceState(); }
122 bool remote() const final { return true; }
123
124 // rtc::VideoSourceInterface<webrtc::VideoFrame> API
125 void AddOrUpdateSink(rtc::VideoSinkInterface<webrtc::VideoFrame>*, const rtc::VideoSinkWants&) final;
126 void RemoveSink(rtc::VideoSinkInterface<webrtc::VideoFrame>*) final;
127
128 void sourceMutedChanged();
129 void sourceEnabledChanged();
130
131 // MediaStreamTrackPrivate::Observer API
132 void trackMutedChanged(MediaStreamTrackPrivate&) final { sourceMutedChanged(); }
133 void trackEnabledChanged(MediaStreamTrackPrivate&) final { sourceEnabledChanged(); }
134 void trackSettingsChanged(MediaStreamTrackPrivate&) final { initializeFromSource(); }
135 void trackEnded(MediaStreamTrackPrivate&) final { }
136
137 // RealtimeMediaSource::VideoSampleObserver API
138 void videoSampleAvailable(MediaSample&) override { }
139
140 Ref<MediaStreamTrackPrivate> m_videoSource;
141 Timer m_blackFrameTimer;
142 rtc::scoped_refptr<webrtc::VideoFrameBuffer> m_blackFrame;
143
144 mutable Lock m_sinksLock;
145 HashSet<rtc::VideoSinkInterface<webrtc::VideoFrame>*> m_sinks WTF_GUARDED_BY_LOCK(m_sinksLock);
146 bool m_areSinksAskingToApplyRotation { false };
147
148 bool m_enabled { true };
149 bool m_muted { false };
150 uint32_t m_width { 0 };
151 uint32_t m_height { 0 };
152
153#if !RELEASE_LOG_DISABLED
154 Ref<const Logger> m_logger;
155 const void* m_logIdentifier;
156 MonotonicTime m_lastFrameLogTime;
157 unsigned m_frameCount { 0 };
158#endif
159};
