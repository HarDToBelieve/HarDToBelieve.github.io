86void RealtimeVideoCaptureSource::setSupportedPresets(const Vector<Ref<VideoPreset>>& presets)
87{
88 m_presets = WTF::map(presets, [](auto& preset) {
89 return preset.copyRef();
90 });
91
92 for (auto& preset : m_presets) {
93 std::sort(preset->frameRateRanges.begin(), preset->frameRateRanges.end(),
94 [&] (const auto& a, const auto& b) -> bool {
95 return a.minimum < b.minimum;
96 });
97 }

148template <typename ValueType>
149static void updateMinMax(ValueType& min, ValueType& max, ValueType value)
150{
151 min = std::min<ValueType>(min, value);
152 max = std::max<ValueType>(max, value);

155void RealtimeVideoCaptureSource::updateCapabilities(RealtimeMediaSourceCapabilities& capabilities)
156{
157 ASSERT(!presets().isEmpty());
158
159 int minimumWidth = std::numeric_limits<int>::max();
160 int maximumWidth = 0;
161 int minimumHeight = std::numeric_limits<int>::max();
162 int maximumHeight = 0;
163 double minimumAspectRatio = std::numeric_limits<double>::max();
164 double maximumAspectRatio = 0;
165 // RealtimeVideoSource will decimate frame rate if the source cannot go below a given value.
166 double minimumFrameRate = 1;
167 double maximumFrameRate = 0;
168 for (const auto& preset : presets()) {
169 const auto& size = preset->size;
170 updateMinMax(minimumWidth, maximumWidth, size.width());
171 updateMinMax(minimumHeight, maximumHeight, size.height());
172 updateMinMax(minimumAspectRatio, maximumAspectRatio, static_cast<double>(size.width()) / size.height());
173
174 for (const auto& rate : preset->frameRateRanges)
175 maximumFrameRate = std::max(maximumFrameRate, rate.maximum);
176 }
177
178 if (canResizeVideoFrames()) {
179 minimumWidth = 1;
180 minimumHeight = 1;
181 minimumAspectRatio = 1.0 / maximumHeight;
182 maximumAspectRatio = maximumWidth;
183 }
184
185 capabilities.setWidth({ minimumWidth, maximumWidth });
186 capabilities.setHeight({ minimumHeight, maximumHeight });
187 capabilities.setAspectRatio({ minimumAspectRatio, maximumAspectRatio });
188 capabilities.setFrameRate({ minimumFrameRate, maximumFrameRate });
189}

191bool RealtimeVideoCaptureSource::supportsSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate)
192{
193 if (!width && !height && !frameRate)
194 return true;
195
196 return !!bestSupportedSizeAndFrameRate(width, height, frameRate);
197}

199bool RealtimeVideoCaptureSource::frameRateRangeIncludesRate(const FrameRateRange& range, double frameRate)
200{
201 const double epsilon = 0.001;
202 return frameRate + epsilon >= range.minimum && frameRate - epsilon <= range.maximum;
203}
204
205bool RealtimeVideoCaptureSource::presetSupportsFrameRate(RefPtr<VideoPreset> preset, double frameRate)
206{
207 for (const auto& range : preset->frameRateRanges) {
208 if (frameRateRangeIncludesRate(range, frameRate))
209 return true;
210 }
211
212 return false;
213}

215bool RealtimeVideoCaptureSource::supportsCaptureSize(std::optional<int> width, std::optional<int> height, const Function<bool(const IntSize&)>&& function)
216{
217 if (width && height)
218 return function({ width.value(), height.value() });
219
220 if (width) {
221 for (auto& size : standardVideoSizes()) {
222 if (width.value() == size.width() && function({ size.width(), size.height() }))
223 return true;
224 }
225
226 return false;
227 }
228
229 for (auto& size : standardVideoSizes()) {
230 if (height.value() == size.height() && function({ size.width(), size.height() }))
231 return true;
232 }
233
234 return false;
235}

237bool RealtimeVideoCaptureSource::shouldUsePreset(VideoPreset& current, VideoPreset& candidate)
238{
239 return candidate.size.width() <= current.size.width() && candidate.size.height() <= current.size.height() && prefersPreset(candidate);
240}
241
242static inline double frameRateFromPreset(const VideoPreset& preset, double currentFrameRate)
243{
244 auto minFrameRate = preset.minFrameRate();
245 auto maxFrameRate = preset.maxFrameRate();
246 return currentFrameRate >= minFrameRate && currentFrameRate <= maxFrameRate ? currentFrameRate : maxFrameRate;
247}

249std::optional<RealtimeVideoCaptureSource::CaptureSizeAndFrameRate> RealtimeVideoCaptureSource::bestSupportedSizeAndFrameRate(std::optional<int> requestedWidth, std::optional<int> requestedHeight, std::optional<double> requestedFrameRate)
250{
251 if (!requestedWidth && !requestedHeight && !requestedFrameRate)
252 return { };
253
254 if (!requestedWidth && !requestedHeight && !size().isEmpty()) {
255 requestedWidth = size().width();
256 requestedHeight = size().height();
257 }
258
259 RefPtr<VideoPreset> exactSizePreset;
260 RefPtr<VideoPreset> aspectRatioPreset;
261 IntSize aspectRatioMatchSize;
262 RefPtr<VideoPreset> resizePreset;
263 IntSize resizeSize;
264
265 for (const auto& preset : presets()) {
266 const auto& presetSize = preset->size;
267
268 if (requestedFrameRate && !presetSupportsFrameRate(&preset.get(), requestedFrameRate.value()))
269 continue;
270
271 if (!requestedWidth && !requestedHeight) {
272 exactSizePreset = preset.ptr();
273 break;
274 }
275
276 // Don't look at presets smaller than the requested resolution because we never want to resize larger.
277 if ((requestedWidth && presetSize.width() < requestedWidth.value()) || (requestedHeight && presetSize.height() < requestedHeight.value()))
278 continue;
279
280 auto lookForExactSizeMatch = [&] (const IntSize& size) -> bool {
281 return preset->size == size;
282 };
283 if (supportsCaptureSize(requestedWidth, requestedHeight, WTFMove(lookForExactSizeMatch))) {
284 if (!exactSizePreset || prefersPreset(preset))
285 exactSizePreset = &preset.get();
286 continue;
287 }
288
289 IntSize encodingSize;
290 auto lookForAspectRatioMatch = [this, &preset, &encodingSize] (const IntSize& size) -> bool {
291 auto aspectRatio = [] (const IntSize size) -> double {
292 return size.width() / static_cast<double>(size.height());
293 };
294 if (std::abs(aspectRatio(preset->size) - aspectRatio(size)) > 10e-7 || !canResizeVideoFrames())
295 return false;
296
297 encodingSize = size;
298 return true;
299 };
300 if (supportsCaptureSize(requestedWidth, requestedHeight, WTFMove(lookForAspectRatioMatch))) {
301 if (!aspectRatioPreset || shouldUsePreset(*aspectRatioPreset, preset)) {
302 aspectRatioPreset = &preset.get();
303 aspectRatioMatchSize = encodingSize;
304 }
305 }
306
307 if (exactSizePreset || aspectRatioPreset)
308 continue;
309
310 if ((requestedWidth && requestedWidth.value() > preset->size.width()) || (requestedHeight && requestedHeight.value() > preset->size.height()))
311 continue;
312
313 if (requestedWidth && requestedHeight) {
314 if (!resizePreset || shouldUsePreset(*resizePreset, preset)) {
315 resizePreset = &preset.get();
316 resizeSize = { requestedWidth.value(), requestedHeight.value() };
317 }
318 } else {
319 for (auto& standardSize : standardVideoSizes()) {
320 if (standardSize.width() > preset->size.width() || standardSize.height() > preset->size.height())
321 break;
322 if ((requestedWidth && requestedWidth.value() != standardSize.width()) || (requestedHeight && requestedHeight.value() != standardSize.height()))
323 continue;
324
325 if (!resizePreset || shouldUsePreset(*resizePreset, preset)) {
326 resizePreset = &preset.get();
327 resizeSize = standardSize;
328 }
329 }
330
331 if (!resizePreset || shouldUsePreset(*resizePreset, preset)) {
332 resizePreset = &preset.get();
333 if (requestedWidth)
334 resizeSize = { requestedWidth.value(), requestedWidth.value() * preset->size.height() / preset->size.width()};
335 else
336 resizeSize = { requestedHeight.value() * preset->size.width() / preset->size.height(), requestedHeight.value() };
337 }
338 }
339 }
340
341 if (!exactSizePreset && !aspectRatioPreset && !resizePreset) {
342 WTFLogAlways("RealtimeVideoCaptureSource::bestSupportedSizeAndFrameRate failed supporting constraints %d %d %f", requestedWidth ? *requestedWidth : -1, requestedHeight ? *requestedHeight : -1, requestedFrameRate ? *requestedFrameRate : -1);
343 for (const auto& preset : presets())
344 preset->log();
345
346 return { };
347 }
348
349 if (exactSizePreset) {
350 auto size = exactSizePreset->size;
351 auto captureFrameRate = requestedFrameRate ? *requestedFrameRate : frameRateFromPreset(*exactSizePreset, frameRate());
352 return CaptureSizeAndFrameRate { WTFMove(exactSizePreset), size, captureFrameRate };
353 }
354
355 if (aspectRatioPreset) {
356 auto captureFrameRate = requestedFrameRate ? *requestedFrameRate : frameRateFromPreset(*aspectRatioPreset, frameRate());
357 return CaptureSizeAndFrameRate { WTFMove(aspectRatioPreset), aspectRatioMatchSize, captureFrameRate };
358 }
359
360 auto captureFrameRate = requestedFrameRate ? *requestedFrameRate : frameRateFromPreset(*resizePreset, frameRate());
361 return CaptureSizeAndFrameRate { WTFMove(resizePreset), resizeSize, captureFrameRate };
362}

364void RealtimeVideoCaptureSource::setSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate)
365{
366 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, SizeAndFrameRate { width, height, frameRate });
367
368 auto size = this->size();
369 if (!width && !height && !size.isEmpty()) {
370 width = size.width();
371 height = size.height();
372 }
373
374 auto match = bestSupportedSizeAndFrameRate(width, height, frameRate);
375 if (!match) {
376 match = bestSupportedSizeAndFrameRate(width, height, { });
377 ASSERT(match);
378 if (!match)
379 return;
380 }
381
382 setFrameRateWithPreset(match->requestedFrameRate, match->encodingPreset);
383
384 if (!match->requestedSize.isEmpty())
385 setSize(match->requestedSize);
386 setFrameRate(match->requestedFrameRate);
387}

389void RealtimeVideoCaptureSource::dispatchMediaSampleToObservers(MediaSample& sample)
390{
391 MediaTime sampleTime = sample.presentationTime();
392
393 auto frameTime = sampleTime.toDouble();
394 m_observedFrameTimeStamps.append(frameTime);
395 m_observedFrameTimeStamps.removeAllMatching([&](auto time) {
396 return time <= frameTime - 2;
397 });
398
399 auto interval = m_observedFrameTimeStamps.last() - m_observedFrameTimeStamps.first();
400 if (interval > 1)
401 m_observedFrameRate = (m_observedFrameTimeStamps.size() / interval);
402
403 videoSampleAvailable(sample);
404}

406void RealtimeVideoCaptureSource::clientUpdatedSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate)
407{
408 // FIXME: We only change settings if capture resolution is below requested one. We should get the best preset for all clients.
409 auto& settings = this->settings();
410 if (width && *width <= static_cast<int>(settings.width()))
411 width = { };
412 if (height && *height <= static_cast<int>(settings.height()))
413 height = { };
414 if (frameRate && *frameRate <= static_cast<double>(settings.frameRate()))
415 frameRate = { };
416
417 if (!width && !height && !frameRate)
418 return;
419
420 auto match = bestSupportedSizeAndFrameRate(width, height, frameRate);
421 ERROR_LOG_IF(loggerPtr() && !match, LOGIDENTIFIER, "unable to find a preset that would match the size and frame rate");
422 if (!match)
423 return;
424
425 setFrameRateWithPreset(match->requestedFrameRate, match->encodingPreset);
426 setSize(match->encodingPreset->size);
427 setFrameRate(match->requestedFrameRate);
428}
