41class WEBCORE_EXPORT RealtimeVideoCaptureSource : public RealtimeMediaSource {
42public:
43 virtual ~RealtimeVideoCaptureSource();
44
45 void clientUpdatedSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double> frameRate);
46
47 bool supportsSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double>) override;
48 virtual void generatePresets() = 0;
49 virtual MediaSample::VideoRotation sampleRotation() const { return MediaSample::VideoRotation::None; }
50
51 double observedFrameRate() const { return m_observedFrameRate; }
52 Vector<VideoPresetData> presetsData();
53
54protected:
55 RealtimeVideoCaptureSource(String&& name, String&& id, String&& hashSalt);
56
57 void prepareToProduceData();
58 void setSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double>) override;
59
60 virtual bool prefersPreset(VideoPreset&) { return true; }
61 virtual void setFrameRateWithPreset(double, RefPtr<VideoPreset>) { };
62 virtual bool canResizeVideoFrames() const { return false; }
63 bool shouldUsePreset(VideoPreset& current, VideoPreset& candidate);
64
65 void setSupportedPresets(const Vector<Ref<VideoPreset>>&);
66 void setSupportedPresets(Vector<VideoPresetData>&&);
67 const Vector<Ref<VideoPreset>>& presets();
68
69 bool frameRateRangeIncludesRate(const FrameRateRange&, double);
70
71 void updateCapabilities(RealtimeMediaSourceCapabilities&);
72
73 void dispatchMediaSampleToObservers(MediaSample&);
74 const Vector<IntSize>& standardVideoSizes();
75
76private:
77 struct CaptureSizeAndFrameRate {
78 RefPtr<VideoPreset> encodingPreset;
79 IntSize requestedSize;
80 double requestedFrameRate { 0 };
81 };
82 bool supportsCaptureSize(std::optional<int>, std::optional<int>, const Function<bool(const IntSize&)>&&);
83 std::optional<CaptureSizeAndFrameRate> bestSupportedSizeAndFrameRate(std::optional<int> width, std::optional<int> height, std::optional<double>);
84 bool presetSupportsFrameRate(RefPtr<VideoPreset>, double);
85
86#if !RELEASE_LOG_DISABLED
87 const char* logClassName() const override { return "RealtimeVideoCaptureSource"; }
88#endif
89
90 Vector<Ref<VideoPreset>> m_presets;
91 Deque<double> m_observedFrameTimeStamps;
92 double m_observedFrameRate { 0 };
93};
94
95struct SizeAndFrameRate {
96 std::optional<int> width;
97 std::optional<int> height;
98 std::optional<double> frameRate;
99
100 String toJSONString() const;
101 Ref<JSON::Object> toJSONObject() const;
102};
