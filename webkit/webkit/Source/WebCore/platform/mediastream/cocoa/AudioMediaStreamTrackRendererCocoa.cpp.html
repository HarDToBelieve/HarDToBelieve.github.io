43void AudioMediaStreamTrackRendererCocoa::start(CompletionHandler<void()>&& callback)
44{
45 clear();
46
47 AudioMediaStreamTrackRendererUnit::singleton().retrieveFormatDescription([weakThis = makeWeakPtr(this), callback = WTFMove(callback)](auto* formatDescription) mutable {
48 if (weakThis && formatDescription)
49 weakThis->m_outputDescription = makeUnique<CAAudioStreamDescription>(*formatDescription);
50 callback();
51 });
52}
53
54void AudioMediaStreamTrackRendererCocoa::stop()
55{
56 if (m_dataSource)
57 AudioMediaStreamTrackRendererUnit::singleton().removeSource(*m_dataSource);
58}
59
60void AudioMediaStreamTrackRendererCocoa::clear()
61{
62 stop();
63
64 m_dataSource = nullptr;
65 m_outputDescription = { };
66}

82static unsigned pollSamplesCount()
83{
84#if USE(LIBWEBRTC)
85 return LibWebRTCAudioModule::PollSamplesCount + 1;
86#else
87 return 2;
88#endif
89}
90
91void AudioMediaStreamTrackRendererCocoa::pushSamples(const MediaTime& sampleTime, const PlatformAudioData& audioData, const AudioStreamDescription& description, size_t sampleCount)
92{
93 ASSERT(!isMainThread());
94 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
95 if (!m_dataSource || m_shouldReset || !m_dataSource->inputDescription() || *m_dataSource->inputDescription() != description) {
96 DisableMallocRestrictionsForCurrentThreadScope scope;
97
98 // FIXME: For non libwebrtc sources, we can probably reduce poll samples count to 2.
99
100 auto dataSource = AudioSampleDataSource::create(description.sampleRate() * 0.5, *this, pollSamplesCount());
101
102 if (dataSource->setInputFormat(toCAAudioStreamDescription(description))) {
103 ERROR_LOG(LOGIDENTIFIER, "Unable to set the input format of data source");
104 return;
105 }
106
107 if (!m_outputDescription || dataSource->setOutputFormat(*m_outputDescription)) {
108 ERROR_LOG(LOGIDENTIFIER, "Unable to set the output format of data source");
109 return;
110 }
111
112 callOnMainThread([this, weakThis = makeWeakPtr(this), oldSource = m_dataSource, newSource = dataSource]() mutable {
113 if (!weakThis)
114 return;
115
116#if !RELEASE_LOG_DISABLED
117 newSource->setLogger(logger(), logIdentifier());
118#endif
119 if (oldSource)
120 AudioMediaStreamTrackRendererUnit::singleton().removeSource(*oldSource);
121
122 newSource->setVolume(volume());
123 AudioMediaStreamTrackRendererUnit::singleton().addSource(WTFMove(newSource));
124 });
125 m_dataSource = WTFMove(dataSource);
126 m_shouldReset = false;
127 }
128
129 m_dataSource->pushSamples(sampleTime, audioData, sampleCount);
130}
