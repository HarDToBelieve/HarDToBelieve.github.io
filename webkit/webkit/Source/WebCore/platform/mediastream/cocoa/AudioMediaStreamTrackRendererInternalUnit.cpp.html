91void LocalAudioMediaStreamTrackRendererInternalUnit::setAudioOutputDevice(const String& deviceID)
92{
93#if PLATFORM(MAC)
94 auto device = CoreAudioCaptureDeviceManager::singleton().coreAudioDeviceWithUID(deviceID);
95
96 if (!device && !deviceID.isEmpty()) {
97 RELEASE_LOG(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::setAudioOutputDeviceId - did not find device");
98 return;
99 }
100
101 auto audioUnitDeviceID = device ? device->deviceID() : 0;
102 if (m_deviceID == audioUnitDeviceID)
103 return;
104
105 bool shouldRestart = m_isStarted;
106 if (m_isStarted)
107 stop();
108
109 m_deviceID = audioUnitDeviceID;
110
111 if (shouldRestart)
112 start();
113#else
114 UNUSED_PARAM(deviceID);
115#endif
116}

118void LocalAudioMediaStreamTrackRendererInternalUnit::start()
119{
120 RELEASE_LOG_INFO(WebRTC, "LocalAudioMediaStreamTrackRendererInternalUnit::start");
121 if (m_isStarted)
122 return;
123
124 createAudioUnitIfNeeded();
125 if (!m_remoteIOUnit)
126 return;
127
128 if (auto error = PAL::AudioOutputUnitStart(m_remoteIOUnit)) {
129 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::start AudioOutputUnitStart failed, error = %d", error);
130 PAL::AudioComponentInstanceDispose(m_remoteIOUnit);
131 m_remoteIOUnit = nullptr;
132 return;
133 }
134 m_isStarted = true;
135 RELEASE_LOG(WebRTC, "AudioMediaStreamTrackRendererInternalUnit is started");
136}
137
138void LocalAudioMediaStreamTrackRendererInternalUnit::stop()
139{
140 RELEASE_LOG_INFO(WebRTC, "LocalAudioMediaStreamTrackRendererInternalUnit::stop");
141 if (!m_remoteIOUnit)
142 return;
143
144 if (m_isStarted) {
145 PAL::AudioOutputUnitStop(m_remoteIOUnit);
146 m_isStarted = false;
147 }
148
149 PAL::AudioComponentInstanceDispose(m_remoteIOUnit);
150 m_remoteIOUnit = nullptr;
151}

153void LocalAudioMediaStreamTrackRendererInternalUnit::createAudioUnitIfNeeded()
154{
155 ASSERT(!m_remoteIOUnit || m_outputDescription);
156 if (m_remoteIOUnit)
157 return;
158
159 CAAudioStreamDescription outputDescription;
160 AudioComponentInstance remoteIOUnit { nullptr };
161
162 AudioComponentDescription ioUnitDescription { kAudioUnitType_Output, 0, kAudioUnitManufacturer_Apple, 0, 0 };
163#if PLATFORM(IOS_FAMILY)
164 ioUnitDescription.componentSubType = kAudioUnitSubType_RemoteIO;
165#else
166 ioUnitDescription.componentSubType = kAudioUnitSubType_DefaultOutput;
167#endif
168
169 AudioComponent ioComponent = PAL::AudioComponentFindNext(nullptr, &ioUnitDescription);
170 ASSERT(ioComponent);
171 if (!ioComponent) {
172 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to find remote IO unit component");
173 return;
174 }
175
176 auto error = PAL::AudioComponentInstanceNew(ioComponent, &remoteIOUnit);
177 if (error) {
178 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to open vpio unit, error = %d", error);
179 return;
180 }
181
182#if PLATFORM(IOS_FAMILY)
183 UInt32 param = 1;
184 error = PAL::AudioUnitSetProperty(remoteIOUnit, kAudioOutputUnitProperty_EnableIO, kAudioUnitScope_Output, 0, &param, sizeof(param));
185 if (error) {
186 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to enable vpio unit output, error = %d", error);
187 return;
188 }
189#endif
190
191#if PLATFORM(MAC)
192 if (m_deviceID) {
193 error = PAL::AudioUnitSetProperty(remoteIOUnit, kAudioOutputUnitProperty_CurrentDevice, kAudioUnitScope_Global, 0, &m_deviceID, sizeof(m_deviceID));
194 if (error) {
195 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to set unit device ID %d, error %d (%.4s)", (int)m_deviceID, (int)error, (char*)&error);
196 return;
197 }
198 }
199#endif
200
201 AURenderCallbackStruct callback = { LocalAudioMediaStreamTrackRendererInternalUnit::renderingCallback, this };
202 error = PAL::AudioUnitSetProperty(remoteIOUnit, kAudioUnitProperty_SetRenderCallback, kAudioUnitScope_Global, 0, &callback, sizeof(callback));
203 if (error) {
204 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to set vpio unit speaker proc, error = %d", error);
205 return;
206 }
207
208 UInt32 size = sizeof(outputDescription.streamDescription());
209 error = PAL::AudioUnitGetProperty(remoteIOUnit, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Input, 0, &outputDescription.streamDescription(), &size);
210 if (error) {
211 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to get input stream format, error = %d", error);
212 return;
213 }
214
215 outputDescription.streamDescription().mSampleRate = AudioSession::sharedSession().sampleRate();
216
217 error = PAL::AudioUnitSetProperty(remoteIOUnit, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Input, 0, &outputDescription.streamDescription(), sizeof(outputDescription.streamDescription()));
218 if (error) {
219 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit unable to set input stream format, error = %d", error);
220 return;
221 }
222
223 error = PAL::AudioUnitInitialize(remoteIOUnit);
224 if (error) {
225 RELEASE_LOG_ERROR(WebRTC, "AudioMediaStreamTrackRendererInternalUnit::createAudioUnit AudioUnitInitialize() failed, error = %d", error);
226 return;
227 }
228
229 m_outputDescription = makeUnique<CAAudioStreamDescription>(outputDescription);
230 m_remoteIOUnit = remoteIOUnit;
231}

233OSStatus LocalAudioMediaStreamTrackRendererInternalUnit::renderingCallback(void* processor, AudioUnitRenderActionFlags* actionFlags, const AudioTimeStamp* timeStamp, UInt32, UInt32 sampleCount, AudioBufferList* ioData)
234{
235 return static_cast<LocalAudioMediaStreamTrackRendererInternalUnit*>(processor)->m_renderCallback(sampleCount, *ioData, timeStamp->mSampleTime, timeStamp->mHostTime, *actionFlags);
236}
