37LibWebRTCAudioModule::LibWebRTCAudioModule()
38 : m_queue(WorkQueue::create("WebKitWebRTCAudioModule", WorkQueue::Type::Serial, WorkQueue::QOS::UserInteractive))
39 , m_logTimer(*this, &LibWebRTCAudioModule::logTimerFired)
40{
41}

43int32_t LibWebRTCAudioModule::RegisterAudioCallback(webrtc::AudioTransport* audioTransport)
44{
45 RELEASE_LOG(WebRTC, "LibWebRTCAudioModule::RegisterAudioCallback %d", !!audioTransport);
46
47 m_audioTransport = audioTransport;
48 return 0;
49}

51int32_t LibWebRTCAudioModule::StartPlayout()
52{
53 RELEASE_LOG(WebRTC, "LibWebRTCAudioModule::StartPlayout %d", m_isPlaying);
54
55 if (m_isPlaying)
56 return 0;
57
58 m_isPlaying = true;
59 callOnMainThread([this, protectedThis = rtc::scoped_refptr<webrtc::AudioDeviceModule>(this)] {
60 m_logTimer.startRepeating(logTimerInterval);
61 });
62
63 m_queue->dispatch([this, protectedThis = rtc::scoped_refptr<webrtc::AudioDeviceModule>(this)] {
64 m_pollingTime = MonotonicTime::now();
65 pollAudioData();
66 });
67 return 0;
68}
69
70int32_t LibWebRTCAudioModule::StopPlayout()
71{
72 RELEASE_LOG(WebRTC, "LibWebRTCAudioModule::StopPlayout %d", m_isPlaying);
73
74 m_isPlaying = false;
75 callOnMainThread([this, protectedThis = rtc::scoped_refptr<webrtc::AudioDeviceModule>(this)] {
76 m_logTimer.stop();
77 });
78 return 0;
79}

81void LibWebRTCAudioModule::logTimerFired()
82{
83 RELEASE_LOG_IF(m_timeSpent, WebRTC, "LibWebRTCAudioModule::pollAudioData, polling took too much time: %d ms", m_timeSpent);
84 m_timeSpent = 0;
85}
86
87// libwebrtc uses 10ms frames.
88const unsigned frameLengthMs = 1000 * LibWebRTCAudioFormat::chunkSampleCount / LibWebRTCAudioFormat::sampleRate;
89const unsigned pollInterval = LibWebRTCAudioModule::PollSamplesCount * frameLengthMs;
90const unsigned channels = 2;
91
92Seconds LibWebRTCAudioModule::computeDelayUntilNextPolling()
93{
94 auto now = MonotonicTime::now();
95 auto delayUntilNextPolling = m_pollingTime + Seconds::fromMilliseconds(pollInterval) - now;
96 if (delayUntilNextPolling.milliseconds() < 0) {
97 m_timeSpent = (now - m_pollingTime).milliseconds();
98 delayUntilNextPolling = 0_s;
99 }
100 m_pollingTime = now + delayUntilNextPolling;
101 return delayUntilNextPolling;
102}
103
104void LibWebRTCAudioModule::pollAudioData()
105{
106 if (!m_isPlaying)
107 return;
108
109 Function<void()> nextPollFunction = [this, protectedThis = rtc::scoped_refptr<webrtc::AudioDeviceModule>(this)] {
110 pollAudioData();
111 };
112
113 {
114 // For performance reasons, we forbid heap allocations while doing rendering on the webrtc audio thread.
115 ForbidMallocUseForCurrentThreadScope forbidMallocUse;
116
117 pollFromSource();
118 }
119 m_queue->dispatchAfter(computeDelayUntilNextPolling(), WTFMove(nextPollFunction));
120}
121
122void LibWebRTCAudioModule::pollFromSource()
123{
124 if (!m_audioTransport)
125 return;
126
127 for (unsigned i = 0; i < PollSamplesCount; i++) {
128 int64_t elapsedTime = -1;
129 int64_t ntpTime = -1;
130 char data[LibWebRTCAudioFormat::sampleByteSize * channels * LibWebRTCAudioFormat::chunkSampleCount];
131 m_audioTransport->PullRenderData(LibWebRTCAudioFormat::sampleByteSize * 8, LibWebRTCAudioFormat::sampleRate, channels, LibWebRTCAudioFormat::chunkSampleCount, data, &elapsedTime, &ntpTime);
132 }
133}
