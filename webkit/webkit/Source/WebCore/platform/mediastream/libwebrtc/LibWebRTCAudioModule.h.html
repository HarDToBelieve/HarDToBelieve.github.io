41namespace WebCore {
42
43// LibWebRTCAudioModule is pulling streamed data to ensure audio data is passed to the audio track.
44class LibWebRTCAudioModule : public webrtc::AudioDeviceModule {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 LibWebRTCAudioModule();
48
49 static constexpr unsigned PollSamplesCount = 3;
50
51private:
52 template<typename U> U shouldNotBeCalled(U value) const
53 {
54 ASSERT_NOT_REACHED();
55 return value;
56 }
57
58 // webrtc::AudioDeviceModule API
59 int32_t StartPlayout() final;
60 int32_t StopPlayout() final;
61 int32_t RegisterAudioCallback(webrtc::AudioTransport*) final;
62 bool Playing() const final { return m_isPlaying; }
63
64 int32_t ActiveAudioLayer(AudioLayer*) const final { return shouldNotBeCalled(-1); }
65 int32_t Init() final { return 0; }
66 int32_t Terminate() final { return 0; }
67 bool Initialized() const final { return true; }
68 int16_t PlayoutDevices() final { return 0; }
69 int16_t RecordingDevices() final { return 0; }
70 int32_t PlayoutDeviceName(uint16_t, char[webrtc::kAdmMaxDeviceNameSize], char[webrtc::kAdmMaxGuidSize]) final { return 0; }
71 int32_t RecordingDeviceName(uint16_t, char[webrtc::kAdmMaxDeviceNameSize], char[webrtc::kAdmMaxGuidSize]) final { return 0; }
72 int32_t SetPlayoutDevice(uint16_t) final { return 0; }
73 int32_t SetPlayoutDevice(WindowsDeviceType) final { return 0; }
74 int32_t SetRecordingDevice(uint16_t) final { return 0; }
75 int32_t SetRecordingDevice(WindowsDeviceType) final { return 0; }
76 int32_t PlayoutIsAvailable(bool*) final { return shouldNotBeCalled(-1); }
77 int32_t InitPlayout() final { return 0; }
78 bool PlayoutIsInitialized() const final { return true; }
79 int32_t RecordingIsAvailable(bool*) final { return shouldNotBeCalled(-1); }
80 int32_t InitRecording() final { return 0; }
81 bool RecordingIsInitialized() const final { return false; }
82 int32_t StartRecording() final { return 0; }
83 int32_t StopRecording() final { return 0; }
84 bool Recording() const final { return 0; }
85 int32_t InitSpeaker() final { return 0; }
86 bool SpeakerIsInitialized() const final { return false; }
87 int32_t InitMicrophone() final { return 0; }
88 bool MicrophoneIsInitialized() const final { return false; }
89 int32_t MicrophoneVolumeIsAvailable(bool*) final { return shouldNotBeCalled(-1); }
90 int32_t SpeakerVolumeIsAvailable(bool*) final { return shouldNotBeCalled(-1); }
91 int32_t SetSpeakerVolume(uint32_t) final { return shouldNotBeCalled(-1); }
92 int32_t SpeakerVolume(uint32_t*) const final { return shouldNotBeCalled(-1); }
93 int32_t MaxSpeakerVolume(uint32_t*) const final { return shouldNotBeCalled(-1); }
94 int32_t MinSpeakerVolume(uint32_t*) const final { return shouldNotBeCalled(-1); }
95 int32_t SetMicrophoneVolume(uint32_t) final { return shouldNotBeCalled(-1); }
96 int32_t MicrophoneVolume(uint32_t*) const final { return shouldNotBeCalled(-1); }
97 int32_t MaxMicrophoneVolume(uint32_t*) const final { return shouldNotBeCalled(-1); }
98 int32_t MinMicrophoneVolume(uint32_t*) const final { return shouldNotBeCalled(-1); }
99 int32_t SpeakerMuteIsAvailable(bool*) final { return shouldNotBeCalled(-1); }
100 int32_t SetSpeakerMute(bool) final { return shouldNotBeCalled(-1); }
101 int32_t SpeakerMute(bool*) const final { return shouldNotBeCalled(-1); }
102 int32_t MicrophoneMuteIsAvailable(bool*) final { return shouldNotBeCalled(-1); }
103 int32_t SetMicrophoneMute(bool) final { return shouldNotBeCalled(-1); }
104 int32_t MicrophoneMute(bool*) const final { return shouldNotBeCalled(-1); }

106 int32_t SetStereoPlayout(bool) final { return 0; }
107 int32_t StereoPlayout(bool*) const final { return shouldNotBeCalled(-1); }

109 int32_t SetStereoRecording(bool) final { return 0; }

113 bool BuiltInAGCIsAvailable() const final { return false; }
114 bool BuiltInNSIsAvailable() const final { return false; }
115 int32_t EnableBuiltInAEC(bool) final { return shouldNotBeCalled(-1); }
116 int32_t EnableBuiltInAGC(bool) final { return shouldNotBeCalled(-1); }
117 int32_t EnableBuiltInNS(bool) final { return shouldNotBeCalled(-1); }
118
119#if defined(WEBRTC_IOS)
120 int GetPlayoutAudioParameters(webrtc::AudioParameters*) const final { return shouldNotBeCalled(-1); }
121 int GetRecordAudioParameters(webrtc::AudioParameters*) const final { return shouldNotBeCalled(-1); }
122#endif

138};
