35static inline String toWTFString(const std::string& value)
36{
37 return String::fromUTF8(value.data(), value.length());
38}
39
40LibWebRTCDTMFSenderBackend::LibWebRTCDTMFSenderBackend(rtc::scoped_refptr<webrtc::DtmfSenderInterface>&& sender)
41 : m_sender(WTFMove(sender))
42{
43 m_sender->RegisterObserver(this);
44}
45
46LibWebRTCDTMFSenderBackend::~LibWebRTCDTMFSenderBackend()
47{
48 m_sender->UnregisterObserver();
49}
50
51bool LibWebRTCDTMFSenderBackend::canInsertDTMF()
52{
53 return m_sender->CanInsertDtmf();
54}
55
56void LibWebRTCDTMFSenderBackend::playTone(const String& tone, size_t duration, size_t interToneGap)
57{
58 bool ok = m_sender->InsertDtmf(tone.utf8().data(), duration, interToneGap);
59 ASSERT_UNUSED(ok, ok);
60}
61
62String LibWebRTCDTMFSenderBackend::tones() const
63{
64 return toWTFString(m_sender->tones());
65}
66
67size_t LibWebRTCDTMFSenderBackend::duration() const
68{
69 return m_sender->duration();
70}
71
72size_t LibWebRTCDTMFSenderBackend::interToneGap() const
73{
74 return m_sender->inter_tone_gap();
75}
76
77void LibWebRTCDTMFSenderBackend::OnToneChange(const std::string& tone, const std::string&)
78{
79 // We are just interested in notifying the end of the tone, which corresponds to the empty string.
80 if (!tone.empty())
81 return;
82 callOnMainThread([this, weakThis = makeWeakPtr(*this), tone = toWTFString(tone)] {
83 if (!weakThis)
84 return;
85 if (m_onTonePlayed)
86 m_onTonePlayed(tone);
87 });
88}
89
90void LibWebRTCDTMFSenderBackend::onTonePlayed(Function<void(const String&)>&& onTonePlayed)
91{
92 m_onTonePlayed = WTFMove(onTonePlayed);
93}
