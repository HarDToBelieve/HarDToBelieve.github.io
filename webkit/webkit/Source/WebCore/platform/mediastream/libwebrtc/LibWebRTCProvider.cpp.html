83static inline rtc::SocketAddress prepareSocketAddress(const rtc::SocketAddress& address, bool disableNonLocalhostConnections)
84{
85 auto result = address;
86 if (disableNonLocalhostConnections)
87 result.SetIP("127.0.0.1");
88 return result;

91class BasicPacketSocketFactory : public rtc::BasicPacketSocketFactory {
92 WTF_MAKE_FAST_ALLOCATED;
93public:
94 explicit BasicPacketSocketFactory(rtc::Thread& networkThread)
95 : m_socketFactory(makeUniqueRefWithoutFastMallocCheck<rtc::BasicPacketSocketFactory>(&networkThread))
96 {
97 }
98
99 void setDisableNonLocalhostConnections(bool disableNonLocalhostConnections) { m_disableNonLocalhostConnections = disableNonLocalhostConnections; }
100
101 rtc::AsyncPacketSocket* CreateUdpSocket(const rtc::SocketAddress& address, uint16_t minPort, uint16_t maxPort) final
102 {
103 return m_socketFactory->CreateUdpSocket(prepareSocketAddress(address, m_disableNonLocalhostConnections), minPort, maxPort);
104 }
105
106 rtc::AsyncPacketSocket* CreateServerTcpSocket(const rtc::SocketAddress& address, uint16_t minPort, uint16_t maxPort, int options) final
107 {
108 return m_socketFactory->CreateServerTcpSocket(prepareSocketAddress(address, m_disableNonLocalhostConnections), minPort, maxPort, options);
109 }
110
111 rtc::AsyncPacketSocket* CreateClientTcpSocket(const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress, const rtc::ProxyInfo& info, const std::string& name, const rtc::PacketSocketTcpOptions& options)
112 {
113 return m_socketFactory->CreateClientTcpSocket(prepareSocketAddress(localAddress, m_disableNonLocalhostConnections), remoteAddress, info, name, options);
114 }
115
116private:
117 bool m_disableNonLocalhostConnections { false };
118 UniqueRef<rtc::BasicPacketSocketFactory> m_socketFactory;
119};

133static void doReleaseLogging(rtc::LoggingSeverity severity, const char* message)
134{
135#if RELEASE_LOG_DISABLED
136 UNUSED_PARAM(severity);
137 UNUSED_PARAM(message);
138#else
139 if (severity == rtc::LS_ERROR)
140 RELEASE_LOG_ERROR(WebRTC, "LibWebRTC error: %" PUBLIC_LOG_STRING, message);
141 else
142 RELEASE_LOG(WebRTC, "LibWebRTC message: %" PUBLIC_LOG_STRING, message);
143#endif
144}

146static rtc::LoggingSeverity computeLogLevel(WTFLogLevel level)
147{
148#if !RELEASE_LOG_DISABLED
149 switch (level) {
150 case WTFLogLevel::Always:
151 case WTFLogLevel::Error:
152 return rtc::LS_ERROR;
153 case WTFLogLevel::Warning:
154 return rtc::LS_WARNING;
155 case WTFLogLevel::Info:
156 return rtc::LS_INFO;
157 case WTFLogLevel::Debug:
158 return rtc::LS_VERBOSE;
159 }
160 ASSERT_NOT_REACHED();
161#else
162 UNUSED_PARAM(level);
163#endif
164 return rtc::LS_NONE;
165}

167void LibWebRTCProvider::setRTCLogging(WTFLogLevel level)
168{
169 auto rtcLevel = computeLogLevel(level);
170 rtc::LogMessage::SetLogOutput(rtcLevel, (rtcLevel == rtc::LS_NONE) ? nullptr : doReleaseLogging);
171}

173static void initializePeerConnectionFactoryAndThreads(PeerConnectionFactoryAndThreads& factoryAndThreads)
174{
175 ASSERT(!factoryAndThreads.networkThread);
176
177 factoryAndThreads.networkThread = factoryAndThreads.networkThreadWithSocketServer ? rtc::Thread::CreateWithSocketServer() : rtc::Thread::Create();
178 factoryAndThreads.networkThread->SetName("WebKitWebRTCNetwork", nullptr);
179 bool result = factoryAndThreads.networkThread->Start();
180 ASSERT_UNUSED(result, result);
181
182 factoryAndThreads.signalingThread = rtc::Thread::Create();
183 factoryAndThreads.signalingThread->SetName("WebKitWebRTCSignaling", nullptr);
184
185 result = factoryAndThreads.signalingThread->Start();
186 ASSERT(result);
187}

189static inline PeerConnectionFactoryAndThreads& staticFactoryAndThreads()
190{
191 static NeverDestroyed<PeerConnectionFactoryAndThreads> factoryAndThreads;
192 return factoryAndThreads.get();
193}
194
195PeerConnectionFactoryAndThreads& LibWebRTCProvider::getStaticFactoryAndThreads(bool useNetworkThreadWithSocketServer)
196{
197 auto& factoryAndThreads = staticFactoryAndThreads();
198
199 ASSERT(!factoryAndThreads.networkThread || factoryAndThreads.networkThreadWithSocketServer == useNetworkThreadWithSocketServer);
200
201 if (!factoryAndThreads.networkThread) {
202 factoryAndThreads.networkThreadWithSocketServer = useNetworkThreadWithSocketServer;
203 initializePeerConnectionFactoryAndThreads(factoryAndThreads);
204 startedNetworkThread();
205 }
206 return factoryAndThreads;
207}

216void PeerConnectionFactoryAndThreads::OnMessage(rtc::Message* message)
217{
218 ASSERT(message->message_id == 1);
219 auto* data = static_cast<ThreadMessageData*>(message->pdata);
220 data->callback();
221 delete data;
222}

252webrtc::PeerConnectionFactoryInterface* LibWebRTCProvider::factory()
253{
254 if (m_factory)
255 return m_factory.get();
256
257 if (!webRTCAvailable()) {
258 RELEASE_LOG_ERROR(WebRTC, "LibWebRTC is not available to create a factory");
259 return nullptr;
260 }
261
262 auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
263
264 m_factory = createPeerConnectionFactory(factoryAndThreads.networkThread.get(), factoryAndThreads.signalingThread.get());
265
266 return m_factory;
267}

302rtc::scoped_refptr<webrtc::PeerConnectionInterface> LibWebRTCProvider::createPeerConnection(webrtc::PeerConnectionObserver& observer, rtc::PacketSocketFactory*, webrtc::PeerConnectionInterface::RTCConfiguration&& configuration)
303{
304 // Default WK1 implementation.
305 ASSERT(m_useNetworkThreadWithSocketServer);
306 auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
307
308 if (!factoryAndThreads.networkManager)
309 factoryAndThreads.networkManager = makeUniqueWithoutFastMallocCheck<rtc::BasicNetworkManager>();
310
311 if (!factoryAndThreads.packetSocketFactory)
312 factoryAndThreads.packetSocketFactory = makeUnique<BasicPacketSocketFactory>(*factoryAndThreads.networkThread);
313 factoryAndThreads.packetSocketFactory->setDisableNonLocalhostConnections(m_disableNonLocalhostConnections);
314
315 return createPeerConnection(observer, *factoryAndThreads.networkManager, *factoryAndThreads.packetSocketFactory, WTFMove(configuration), nullptr);
316}

343rtc::scoped_refptr<webrtc::PeerConnectionInterface> LibWebRTCProvider::createPeerConnection(webrtc::PeerConnectionObserver& observer, rtc::NetworkManager& networkManager, rtc::PacketSocketFactory& packetSocketFactory, webrtc::PeerConnectionInterface::RTCConfiguration&& configuration, std::unique_ptr<webrtc::AsyncResolverFactory>&& asyncResolveFactory)
344{
345 auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
346
347 std::unique_ptr<cricket::BasicPortAllocator> portAllocator;
348 factoryAndThreads.signalingThread->Invoke<void>(RTC_FROM_HERE, [&]() {
349 auto basicPortAllocator = makeUniqueWithoutFastMallocCheck<cricket::BasicPortAllocator>(&networkManager, &packetSocketFactory);
350
351 basicPortAllocator->set_allow_tcp_listen(false);
352 portAllocator = WTFMove(basicPortAllocator);
353 });
354
355 auto* factory = this->factory();
356 if (!factory)
357 return nullptr;
358
359 webrtc::PeerConnectionDependencies dependencies { &observer };
360 dependencies.allocator = WTFMove(portAllocator);
361 dependencies.async_resolver_factory = WTFMove(asyncResolveFactory);
362
363 auto peerConnectionOrError = m_factory->CreatePeerConnectionOrError(configuration, WTFMove(dependencies));
364 if (!peerConnectionOrError.ok())
365 return nullptr;
366
367 return peerConnectionOrError.MoveValue();
368}

370void LibWebRTCProvider::prepareCertificateGenerator(Function<void(rtc::RTCCertificateGenerator&)>&& callback)
371{
372 auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
373 if (!factoryAndThreads.certificateGenerator)
374 factoryAndThreads.certificateGenerator = makeUniqueWithoutFastMallocCheck<rtc::RTCCertificateGenerator>(factoryAndThreads.signalingThread.get(), factoryAndThreads.networkThread.get());
375
376 auto& generator = *factoryAndThreads.certificateGenerator;
377 callOnWebRTCSignalingThread([&generator, callback = WTFMove(callback)]() mutable {
378 callback(generator);
379 });
380}

403static inline RTCRtpCapabilities toRTCRtpCapabilities(const webrtc::RtpCapabilities& rtpCapabilities)
404{
405 RTCRtpCapabilities capabilities;
406
407 capabilities.codecs.reserveInitialCapacity(rtpCapabilities.codecs.size());
408 for (auto& codec : rtpCapabilities.codecs) {
409 StringBuilder sdpFmtpLineBuilder;
410 bool hasParameter = false;
411 for (auto& parameter : codec.parameters) {
412 sdpFmtpLineBuilder.append(hasParameter ? ";" : "", StringView(parameter.first.data(), parameter.first.length()), '=', StringView(parameter.second.data(), parameter.second.length()));
413 hasParameter = true;
414 }
415 String sdpFmtpLine;
416 if (sdpFmtpLineBuilder.length())
417 sdpFmtpLine = sdpFmtpLineBuilder.toString();
418 capabilities.codecs.uncheckedAppend(RTCRtpCodecCapability { fromStdString(codec.mime_type()), static_cast<uint32_t>(codec.clock_rate ? *codec.clock_rate : 0), toChannels(codec.num_channels), WTFMove(sdpFmtpLine) });
419 }
420
421 capabilities.headerExtensions.reserveInitialCapacity(rtpCapabilities.header_extensions.size());
422 for (auto& header : rtpCapabilities.header_extensions)
423 capabilities.headerExtensions.uncheckedAppend(RTCRtpCapabilities::HeaderExtensionCapability { fromStdString(header.uri) });
424
425 return capabilities;
426}

428std::optional<RTCRtpCapabilities> LibWebRTCProvider::receiverCapabilities(const String& kind)
429{
430 auto mediaType = typeFromKind(kind);
431 if (!mediaType)
432 return { };
433
434 auto* factory = this->factory();
435 if (!factory)
436 return { };
437
438 return toRTCRtpCapabilities(factory->GetRtpReceiverCapabilities(*mediaType));
439}
