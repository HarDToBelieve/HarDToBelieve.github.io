110CaptureSourceOrError AVVideoCaptureSource::create(String&& id, String&& hashSalt, const MediaConstraints* constraints)
111{
112 AVCaptureDevice *device = [PAL::getAVCaptureDeviceClass() deviceWithUniqueID:id];
113 if (!device)
114 return { "No AVVideoCaptureSource device"_s };
115
116 auto source = adoptRef(*new AVVideoCaptureSource(device, WTFMove(id), WTFMove(hashSalt)));
117 if (constraints) {
118 auto result = source->applyConstraints(*constraints);
119 if (result)
120 return WTFMove(result.value().badConstraint);
121 }
122
123 return CaptureSourceOrError(RealtimeVideoSource::create(WTFMove(source)));
124}

154void AVVideoCaptureSource::verifyIsCapturing()
155{
156 ASSERT(m_isRunning);
157 if (m_lastFramesCount != m_framesCount) {
158 m_lastFramesCount = m_framesCount;
159 return;
160 }
161
162 RELEASE_LOG_ERROR(WebRTC, "AVVideoCaptureSource::verifyIsCapturing - no frame received in %d seconds, failing", static_cast<int>(m_verifyCapturingTimer.repeatInterval().value()));
163 captureFailed();
164}

166void AVVideoCaptureSource::updateVerifyCapturingTimer()
167{
168 if (!m_isRunning || m_interrupted) {
169 m_verifyCapturingTimer.stop();
170 return;
171 }
172
173 if (m_verifyCapturingTimer.isActive())
174 return;
175
176 m_verifyCapturingTimer.startRepeating(verifyCaptureInterval);
177 m_framesCount = 0;
178 m_lastFramesCount = 0;
179}

238const RealtimeMediaSourceSettings& AVVideoCaptureSource::settings()
239{
240 if (m_currentSettings)
241 return *m_currentSettings;
242
243 RealtimeMediaSourceSettings settings;
244 if ([device() position] == AVCaptureDevicePositionFront)
245 settings.setFacingMode(RealtimeMediaSourceSettings::User);
246 else if ([device() position] == AVCaptureDevicePositionBack)
247 settings.setFacingMode(RealtimeMediaSourceSettings::Environment);
248 else
249 settings.setFacingMode(RealtimeMediaSourceSettings::Unknown);
250
251 settings.setLabel(name());
252 settings.setFrameRate(frameRate());
253
254 auto size = this->size();
255 if (m_sampleRotation == MediaSample::VideoRotation::Left || m_sampleRotation == MediaSample::VideoRotation::Right)
256 size = size.transposedSize();
257
258 settings.setWidth(size.width());
259 settings.setHeight(size.height());
260 settings.setDeviceId(hashedId());
261
262 RealtimeMediaSourceSupportedConstraints supportedConstraints;
263 supportedConstraints.setSupportsDeviceId(true);
264 supportedConstraints.setSupportsFacingMode([device() position] != AVCaptureDevicePositionUnspecified);
265 supportedConstraints.setSupportsWidth(true);
266 supportedConstraints.setSupportsHeight(true);
267 supportedConstraints.setSupportsAspectRatio(true);
268 supportedConstraints.setSupportsFrameRate(true);
269
270 settings.setSupportedConstraints(supportedConstraints);
271
272 m_currentSettings = WTFMove(settings);
273
274 return *m_currentSettings;
275}

283 capabilities.setDeviceId(hashedId());

286 if ([videoDevice position] == AVCaptureDevicePositionFront)

290
291 updateCapabilities(capabilities);
292
293 m_capabilities = WTFMove(capabilities);
294
295 return *m_capabilities;

318void AVVideoCaptureSource::setSessionSizeAndFrameRate()
319{
320 if (!m_session)
321 return;
322
323 auto* avPreset = m_currentPreset.get();
324 if (!avPreset)
325 return;
326
327 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, SizeAndFrameRate { m_currentPreset->size.width(), m_currentPreset->size.height(), m_currentFrameRate });
328
329 ASSERT(avPreset->format);
330
331 NSError *error = nil;
332 [m_session beginConfiguration];
333 @try {
334 if ([device() lockForConfiguration:&error]) {
335 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, "setting preset to ", m_currentSize);
336 [device() setActiveFormat:avPreset->format.get()];
337
338#if PLATFORM(MAC)
339 auto settingsDictionary = @{
340 (__bridge NSString *)kCVPixelBufferPixelFormatTypeKey: @(avVideoCapturePixelBufferFormat()),
341 (__bridge NSString *)kCVPixelBufferWidthKey: @(avPreset->size.width()),
342 (__bridge NSString *)kCVPixelBufferHeightKey: @(avPreset->size.height()),
343 (__bridge NSString *)kCVPixelBufferIOSurfacePropertiesKey : @{ }
344 };
345 [m_videoOutput setVideoSettings:settingsDictionary];
346#endif
347
348 auto* frameRateRange = frameDurationForFrameRate(m_currentFrameRate);
349 ASSERT(frameRateRange);
350 if (frameRateRange) {
351 m_currentFrameRate = clampTo(m_currentFrameRate, frameRateRange.minFrameRate, frameRateRange.maxFrameRate);
352
353 auto frameDuration = PAL::CMTimeMake(1, m_currentFrameRate);
354 if (PAL::CMTimeCompare(frameDuration, frameRateRange.minFrameDuration) < 0)
355 frameDuration = frameRateRange.minFrameDuration;
356 else if (PAL::CMTimeCompare(frameDuration, frameRateRange.maxFrameDuration) > 0)
357 frameDuration = frameRateRange.maxFrameDuration;
358
359 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, "setting frame rate to ", m_currentFrameRate, ", duration ", PAL::toMediaTime(frameDuration));
360
361 [device() setActiveVideoMinFrameDuration: frameDuration];
362 [device() setActiveVideoMaxFrameDuration: frameDuration];
363 } else
364 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "cannot find proper frame rate range for the selected preset\n");
365
366 [device() unlockForConfiguration];
367 }
368 } @catch(NSException *exception) {
369 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "error configuring device ", [[exception name] UTF8String], ", reason : ", [[exception reason] UTF8String]);
370 [device() unlockForConfiguration];
371 ASSERT_NOT_REACHED();
372 }
373 [m_session commitConfiguration];
374
375 ERROR_LOG_IF(error && loggerPtr(), LOGIDENTIFIER, [[error localizedDescription] UTF8String]);
376}

412{

434AVFrameRateRange* AVVideoCaptureSource::frameDurationForFrameRate(double rate)
435{
436 using namespace PAL; // For CMTIME_COMPARE_INLINE
437
438 AVFrameRateRange *bestFrameRateRange = nil;
439 for (AVFrameRateRange *frameRateRange in [[device() activeFormat] videoSupportedFrameRateRanges]) {
440 if (frameRateRangeIncludesRate({ [frameRateRange minFrameRate], [frameRateRange maxFrameRate] }, rate)) {
441 if (!bestFrameRateRange || CMTIME_COMPARE_INLINE([frameRateRange minFrameDuration], >, [bestFrameRateRange minFrameDuration]))
442 bestFrameRateRange = frameRateRange;
443 }
444 }
445
446 if (!bestFrameRateRange)
447 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "no frame rate range for rate ", rate);
448
449 return bestFrameRateRange;
450}

452bool AVVideoCaptureSource::setupCaptureSession()
453{
454 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER);
455
456#if PLATFORM(IOS_FAMILY)
457 RealtimeMediaSourceCenter::singleton().videoCaptureFactory().setActiveSource(*this);
458#endif
459
460 NSError *error = nil;
461 RetainPtr<AVCaptureDeviceInput> videoIn = adoptNS([PAL::allocAVCaptureDeviceInputInstance() initWithDevice:device() error:&error]);
462 if (error) {
463 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "failed to allocate AVCaptureDeviceInput ", [[error localizedDescription] UTF8String]);
464 return false;
465 }
466
467 if (![session() canAddInput:videoIn.get()]) {
468 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "unable to add video input device");
469 return false;
470 }
471 [session() addInput:videoIn.get()];
472
473 m_videoOutput = adoptNS([PAL::allocAVCaptureVideoDataOutputInstance() init]);
474 auto settingsDictionary = adoptNS([[NSMutableDictionary alloc] initWithObjectsAndKeys: @(avVideoCapturePixelBufferFormat()), kCVPixelBufferPixelFormatTypeKey, nil]);
475
476 [m_videoOutput setVideoSettings:settingsDictionary.get()];
477 [m_videoOutput setAlwaysDiscardsLateVideoFrames:YES];
478 [m_videoOutput setSampleBufferDelegate:m_objcObserver.get() queue:globaVideoCaptureSerialQueue()];
479
480 if (![session() canAddOutput:m_videoOutput.get()]) {
481 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "unable to add video output device");
482 return false;
483 }
484 [session() addOutput:m_videoOutput.get()];
485
486 setSessionSizeAndFrameRate();
487
488 m_sensorOrientation = sensorOrientationFromVideoOutput(m_videoOutput.get());
489 computeSampleRotation();
490
491 return true;
492}

518void AVVideoCaptureSource::computeSampleRotation()
519{
520 bool frontCamera = [device() position] == AVCaptureDevicePositionFront;
521 MediaSample::VideoRotation sampleRotation;
522 switch (m_sensorOrientation - m_deviceOrientation) {
523 case 0:
524 sampleRotation = MediaSample::VideoRotation::None;
525 break;
526 case 180:
527 case -180:
528 sampleRotation = MediaSample::VideoRotation::UpsideDown;
529 break;
530 case 90:
531 case -270:
532 sampleRotation = frontCamera ? MediaSample::VideoRotation::Left : MediaSample::VideoRotation::Right;
533 break;
534 case -90:
535 case 270:
536 sampleRotation = frontCamera ? MediaSample::VideoRotation::Right : MediaSample::VideoRotation::Left;
537 break;
538 default:
539 ASSERT_NOT_REACHED();
540 sampleRotation = MediaSample::VideoRotation::None;
541 }
542 if (sampleRotation == m_sampleRotation)
543 return;
544
545 m_sampleRotation = sampleRotation;
546 notifySettingsDidChangeObservers({ RealtimeMediaSourceSettings::Flag::Width, RealtimeMediaSourceSettings::Flag::Height });
547}

597void AVVideoCaptureSource::generatePresets()
598{
599 Vector<Ref<VideoPreset>> presets;
600 for (AVCaptureDeviceFormat* format in [device() formats]) {
601
602 CMVideoDimensions dimensions = PAL::CMVideoFormatDescriptionGetDimensions(format.formatDescription);
603 IntSize size = { dimensions.width, dimensions.height };
604 auto index = presets.findMatching([&size](auto& preset) {
605 return size == preset->size;
606 });
607 if (index != notFound)
608 continue;
609
610 Vector<FrameRateRange> frameRates;
611 for (AVFrameRateRange* range in [format videoSupportedFrameRateRanges])
612 frameRates.append({ range.minFrameRate, range.maxFrameRate});
613
614 presets.append(AVVideoPreset::create(size, WTFMove(frameRates), format));
615 }
616
617 setSupportedPresets(WTFMove(presets));
618}

632 [m_session startRunning];
633 m_isRunning = [m_session isRunning];
634 });
635}

708- (void)observeValueForKeyPath:keyPath ofObject:(id)object change:(NSDictionary*)change context:(void*)context
709{
710 UNUSED_PARAM(object);
711 UNUSED_PARAM(context);
712
713 if (!m_callback)
714 return;
715
716 id newValue = [change valueForKey:NSKeyValueChangeNewKey];
717 bool willChange = [[change valueForKey:NSKeyValueChangeNotificationIsPriorKey] boolValue];
718
719#if !RELEASE_LOG_DISABLED
720 if (m_callback->loggerPtr()) {
721 auto identifier = Logger::LogSiteIdentifier("AVVideoCaptureSource", "observeValueForKeyPath", m_callback->logIdentifier());
722 RetainPtr<NSString> valueString = adoptNS([[NSString alloc] initWithFormat:@"%@", newValue]);
