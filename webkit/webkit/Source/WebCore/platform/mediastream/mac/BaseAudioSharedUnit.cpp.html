40BaseAudioSharedUnit::BaseAudioSharedUnit()

45void BaseAudioSharedUnit::addClient(CoreAudioCaptureSource& client)
46{
47 ASSERT(isMainThread());
48 Locker locker { m_clientsLock };
49 m_clients.add(&client);
50}
51
52void BaseAudioSharedUnit::removeClient(CoreAudioCaptureSource& client)
53{
54 ASSERT(isMainThread());
55 Locker locker { m_clientsLock };
56 m_clients.remove(&client);
57}

59void BaseAudioSharedUnit::forEachClient(const Function<void(CoreAudioCaptureSource&)>& apply) const
60{
61 Vector<CoreAudioCaptureSource*> clientsCopy;
62 {
63 Locker locker { m_clientsLock };
64 clientsCopy = copyToVector(m_clients);
65 }
66 for (auto* client : clientsCopy) {
67 Locker locker { m_clientsLock };
68 // Make sure the client has not been destroyed.
69 if (!m_clients.contains(client))
70 continue;
71 apply(*client);
72 }
73}

75void BaseAudioSharedUnit::clearClients()
76{
77 Locker locker { m_clientsLock };
78 m_clients.clear();
79}

81void BaseAudioSharedUnit::startProducingData()
82{
83 ASSERT(isMainThread());
84
85 if (m_suspended)
86 resume();
87
88 if (++m_producingCount != 1)
89 return;
90
91 if (isProducingData())
92 return;
93
94 if (hasAudioUnit()) {
95 cleanupAudioUnit();
96 ASSERT(!hasAudioUnit());
97 }
98 startUnit();
99}
100
101OSStatus BaseAudioSharedUnit::startUnit()
102{
103 forEachClient([](auto& client) {
104 client.audioUnitWillStart();
105 });
106 ASSERT(!DeprecatedGlobalSettings::shouldManageAudioSessionCategory() || AudioSession::sharedSession().category() == AudioSession::CategoryType::PlayAndRecord);
107
108 if (auto error = startInternal()) {
109 captureFailed();
110 return error;
111 }
112 return 0;
113}

115void BaseAudioSharedUnit::resetSampleRate()
116{
117 m_sampleRate = AudioSession::sharedSession().sampleRate();
118}
119
120void BaseAudioSharedUnit::prepareForNewCapture()
121{
122 m_volume = 1;
123 resetSampleRate();
124
125 if (!m_suspended)
126 return;
127 m_suspended = false;
128
129 if (!m_producingCount)
130 return;
131
132 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::prepareForNewCapture, notifying suspended sources of capture failure");
133 captureFailed();
134}
135
136void BaseAudioSharedUnit::captureFailed()
137{
138 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::captureFailed - capture failed");
139 forEachClient([](auto& client) {
140 client.captureFailed();
141 });
142
143 m_producingCount = 0;
144
145 clearClients();
146
147 stopInternal();
148 cleanupAudioUnit();
149}
150
151void BaseAudioSharedUnit::stopProducingData()
152{
153 ASSERT(isMainThread());
154 ASSERT(m_producingCount);
155
156 if (m_producingCount && --m_producingCount)
157 return;
158
159 stopInternal();
160 cleanupAudioUnit();
161}
162
163void BaseAudioSharedUnit::reconfigure()
164{
165 ASSERT(isMainThread());
166 if (m_suspended) {
167 m_needsReconfiguration = true;
168 return;
169 }
170 reconfigureAudioUnit();
171}
172
173OSStatus BaseAudioSharedUnit::resume()
174{
175 ASSERT(isMainThread());
176 if (!m_suspended)
177 return 0;
178
179 ASSERT(!isProducingData());
180
181 RELEASE_LOG_INFO(WebRTC, "BaseAudioSharedUnit::resume");
182
183 m_suspended = false;
184
185 if (m_needsReconfiguration) {
186 m_needsReconfiguration = false;
187 reconfigure();
188 }
189
190 ASSERT(!m_producingCount);
191
192 forEachClient([](auto& client) {
193 client.setMuted(false);
194 });
195
196 return 0;
197}
198
199OSStatus BaseAudioSharedUnit::suspend()
200{
201 ASSERT(isMainThread());
202
203 RELEASE_LOG_INFO(WebRTC, "BaseAudioSharedUnit::suspend");
204
205 m_suspended = true;
206 stopInternal();
207
208 forEachClient([](auto& client) {
209 client.setMuted(true);
210 });
211
212 ASSERT(!m_producingCount);
213
214 return 0;
215}

217void BaseAudioSharedUnit::audioSamplesAvailable(const MediaTime& time, const PlatformAudioData& data, const AudioStreamDescription& description, size_t numberOfFrames)
218{
219 // We hold the lock here since adding/removing clients can only happen in main thread.
220 Locker locker { m_clientsLock };
221
222 // For performance reasons, we forbid heap allocations while doing rendering on the capture audio thread.
223 ForbidMallocUseForCurrentThreadScope forbidMallocUse;
224
225 for (auto* client : m_clients) {
226 if (client->isProducingData())
227 client->audioSamplesAvailable(time, data, description, numberOfFrames);
228 }
229}
