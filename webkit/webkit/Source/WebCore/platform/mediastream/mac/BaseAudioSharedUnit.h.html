30#include "RealtimeMediaSourceCapabilities.h"

32#include <wtf/HashSet.h>

37namespace WebCore {
38
39class AudioStreamDescription;
40class CoreAudioCaptureSource;
41class PlatformAudioData;
42
43class BaseAudioSharedUnit {
44public:
45 BaseAudioSharedUnit();
46 virtual ~BaseAudioSharedUnit() = default;
47
48 void startProducingData();
49 void stopProducingData();
50 void reconfigure();
51 virtual bool isProducingData() const = 0;
52
53 virtual void delaySamples(Seconds) { }
54
55 void prepareForNewCapture();
56
57 OSStatus resume();
58 OSStatus suspend();
59
60 bool isSuspended() const { return m_suspended; }
61
62 double volume() const { return m_volume; }
63 int sampleRate() const { return m_sampleRate; }
64 bool enableEchoCancellation() const { return m_enableEchoCancellation; }
65
66 void setVolume(double volume) { m_volume = volume; }
67 void setSampleRate(int sampleRate) { m_sampleRate = sampleRate; }
68 void setEnableEchoCancellation(bool enableEchoCancellation) { m_enableEchoCancellation = enableEchoCancellation; }
69
70 void addClient(CoreAudioCaptureSource&);
71 void removeClient(CoreAudioCaptureSource&);
72 void clearClients();
73
74 virtual bool hasAudioUnit() const = 0;
75 virtual void setCaptureDevice(String&&, uint32_t) = 0;
76
77 virtual CapabilityValueOrRange sampleRateCapacities() const = 0;
78
79protected:
80 void forEachClient(const Function<void(CoreAudioCaptureSource&)>&) const;
81 bool hasClients() const { return !m_clients.isEmpty(); }
82 void captureFailed();
83
84 virtual void cleanupAudioUnit() = 0;
85 virtual OSStatus startInternal() = 0;
86 virtual void stopInternal() = 0;
87 virtual OSStatus reconfigureAudioUnit() = 0;
88 virtual void resetSampleRate();
89
90 void setSuspended(bool value) { m_suspended = value; }
91
92 void audioSamplesAvailable(const MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t /*numberOfFrames*/);
93
94private:
95 OSStatus startUnit();
96
97 bool m_enableEchoCancellation { true };
98 double m_volume { 1 };
99 int m_sampleRate;
100 bool m_suspended { false };
101 bool m_needsReconfiguration { false };
102
103 int32_t m_producingCount { 0 };
104
105 HashSet<CoreAudioCaptureSource*> m_clients;
106 mutable RecursiveLock m_clientsLock;
107};
108
109} // namespace WebCore

111#endif // ENABLE(MEDIA_STREAM)
