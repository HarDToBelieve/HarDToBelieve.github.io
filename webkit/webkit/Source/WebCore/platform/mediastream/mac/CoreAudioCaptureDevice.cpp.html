40static bool getDeviceInfo(uint32_t deviceID, CaptureDevice::DeviceType type, String& persistentID, String& label)
41{
42 CFStringRef uniqueID;
43 AudioObjectPropertyAddress address {
44 kAudioDevicePropertyDeviceUID,
45 kAudioDevicePropertyScopeInput,
46#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
47 kAudioObjectPropertyElementMain
48#else
49 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
50 kAudioObjectPropertyElementMaster
51 ALLOW_DEPRECATED_DECLARATIONS_END
52#endif
53 };
54 UInt32 dataSize = sizeof(uniqueID);
55 auto err = AudioObjectGetPropertyData(static_cast<UInt32>(deviceID), &address, 0, nullptr, &dataSize, &uniqueID);
56 if (err) {
57 RELEASE_LOG_ERROR(WebRTC, "CoreAudioCaptureDevice::getDeviceInfo failed to get device unique id with error %d (%.4s)", (int)err, (char*)&err);
58 return false;
59 }
60 persistentID = uniqueID;
61 CFRelease(uniqueID);
62
63 CFStringRef localizedName = nullptr;
64 AudioObjectPropertyScope scope = type == CaptureDevice::DeviceType::Microphone ? kAudioDevicePropertyScopeInput : kAudioDevicePropertyScopeOutput;
65 address = {
66 kAudioDevicePropertyDataSource,
67 scope,
68#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
69 kAudioObjectPropertyElementMain
70#else
71 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
72 kAudioObjectPropertyElementMaster
73 ALLOW_DEPRECATED_DECLARATIONS_END
74#endif
75 };
76 uint32_t sourceID;
77 dataSize = sizeof(sourceID);
78 err = AudioObjectGetPropertyData(static_cast<UInt32>(deviceID), &address, 0, nullptr, &dataSize, &sourceID);
79 if (!err) {
80 AudioValueTranslation translation = { &deviceID, sizeof(deviceID), &localizedName, sizeof(localizedName) };
81 address = {
82 kAudioDevicePropertyDataSourceNameForIDCFString,
83 scope,
84#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
85 kAudioObjectPropertyElementMain
86#else
87 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
88 kAudioObjectPropertyElementMaster
89 ALLOW_DEPRECATED_DECLARATIONS_END
90#endif
91 };
92 dataSize = sizeof(translation);
93 err = AudioObjectGetPropertyData(static_cast<UInt32>(deviceID), &address, 0, nullptr, &dataSize, &translation);
94 }
95
96 if (err || !localizedName || !CFStringGetLength(localizedName)) {
97 address = {
98 kAudioObjectPropertyName,
99 kAudioObjectPropertyScopeGlobal,
100#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
101 kAudioObjectPropertyElementMain
102#else
103 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
104 kAudioObjectPropertyElementMaster
105 ALLOW_DEPRECATED_DECLARATIONS_END
106#endif
107 };
108 dataSize = sizeof(localizedName);
109 err = AudioObjectGetPropertyData(static_cast<UInt32>(deviceID), &address, 0, nullptr, &dataSize, &localizedName);
110 }
111
112 if (err) {
113 RELEASE_LOG_ERROR(WebRTC, "CoreAudioCaptureDevice::getDeviceInfo failed to get device name with error %d (%.4s)", (int)err, (char*)&err);
114 return false;
115 }
116
117 label = localizedName;
118 CFRelease(localizedName);
119
120 return true;
121}

123std::optional<CoreAudioCaptureDevice> CoreAudioCaptureDevice::create(uint32_t deviceID, DeviceType type, const String& groupID)
124{
125 ASSERT(type == CaptureDevice::DeviceType::Microphone || type == CaptureDevice::DeviceType::Speaker);
126 String persistentID;
127 String label;
128 if (!getDeviceInfo(deviceID, type, persistentID, label))
129 return std::nullopt;
130
131 return CoreAudioCaptureDevice(deviceID, persistentID, type, label, groupID.isNull() ? persistentID : groupID);
132}

134CoreAudioCaptureDevice::CoreAudioCaptureDevice(uint32_t deviceID, const String& persistentID, DeviceType deviceType, const String& label, const String& groupID)
135 : CaptureDevice(persistentID, deviceType, label, groupID)
136 , m_deviceID(deviceID)
137{
138 AudioObjectPropertyAddress address {
139 kAudioDevicePropertyDeviceIsAlive,
140 kAudioObjectPropertyScopeGlobal,
141#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
142 kAudioObjectPropertyElementMain
143#else
144 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
145 kAudioObjectPropertyElementMaster
146 ALLOW_DEPRECATED_DECLARATIONS_END
147#endif
148 };
149 UInt32 state = 0;
150 UInt32 dataSize = sizeof(state);
151 auto err = AudioObjectGetPropertyData(static_cast<UInt32>(m_deviceID), &address, 0, nullptr, &dataSize, &state);
152 if (err)
153 RELEASE_LOG_ERROR(WebRTC, "CoreAudioCaptureDevice::CoreAudioCaptureDevice(%p) failed to get \"is alive\" with error %d (%.4s)", this, (int)err, (char*)&err);
154 setEnabled(!err && state);
155
156 UInt32 property = deviceType == CaptureDevice::DeviceType::Microphone ? kAudioHardwarePropertyDefaultInputDevice : kAudioHardwarePropertyDefaultOutputDevice;
157 address = {
158 property,
159 kAudioObjectPropertyScopeGlobal,
160#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
161 kAudioObjectPropertyElementMain
162#else
163 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
164 kAudioObjectPropertyElementMaster
165 ALLOW_DEPRECATED_DECLARATIONS_END
166#endif
167 };
168 AudioDeviceID defaultID = kAudioDeviceUnknown;
169 dataSize = sizeof(defaultID);
170 err = AudioObjectGetPropertyData(kAudioObjectSystemObject, &address, 0, nullptr, &dataSize, &defaultID);
171 if (err)
172 RELEASE_LOG_ERROR(WebRTC, "CoreAudioCaptureDevice::CoreAudioCaptureDevice(%p) failed to get \"is default\" with error %d (%.4s)", this, (int)err, (char*)&err);
173 setIsDefault(!err && defaultID == m_deviceID);
174}

176Vector<AudioDeviceID> CoreAudioCaptureDevice::relatedAudioDeviceIDs(AudioDeviceID deviceID)
177{
178 UInt32 size = 0;
179 AudioObjectPropertyAddress property = {
180 kAudioDevicePropertyRelatedDevices,
181 kAudioDevicePropertyScopeOutput,
182#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
183 kAudioObjectPropertyElementMain
184#else
185 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
186 kAudioObjectPropertyElementMaster
187 ALLOW_DEPRECATED_DECLARATIONS_END
188#endif
189 };
190 OSStatus error = AudioObjectGetPropertyDataSize(deviceID, &property, 0, 0, &size);
191 if (error || !size)
192 return { };
193
194 Vector<AudioDeviceID> devices(size / sizeof(AudioDeviceID));
195 error = AudioObjectGetPropertyData(deviceID, &property, 0, nullptr, &size, devices.data());
196 if (error)
197 return { };
198 return devices;
199}

201RetainPtr<CMClockRef> CoreAudioCaptureDevice::deviceClock()
202{
203 if (m_deviceClock)
204 return m_deviceClock;
205
206 CMClockRef clock;
207 auto err = PAL::CMAudioDeviceClockCreate(kCFAllocatorDefault, persistentId().createCFString().get(), &clock);
208 if (err) {
209 RELEASE_LOG_ERROR(WebRTC, "CoreAudioCaptureDevice::CMAudioDeviceClockCreate(%p) CMAudioDeviceClockCreate failed with error %d (%.4s)", this, (int)err, (char*)&err);
210 return nullptr;
211 }
212
213 m_deviceClock = adoptCF(clock);
214
215 return m_deviceClock;
216}
