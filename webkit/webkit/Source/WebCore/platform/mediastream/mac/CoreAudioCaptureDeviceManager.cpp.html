58std::optional<CaptureDevice> CoreAudioCaptureDeviceManager::captureDeviceWithPersistentID(CaptureDevice::DeviceType type, const String& deviceID)

68static bool deviceHasStreams(AudioObjectID deviceID, const AudioObjectPropertyAddress& address)
69{
70 UInt32 dataSize = 0;
71 auto err = AudioObjectGetPropertyDataSize(deviceID, &address, 0, nullptr, &dataSize);
72 if (err || !dataSize)
73 return false;
74
75 auto bufferList = std::unique_ptr<AudioBufferList>((AudioBufferList*) ::operator new (dataSize));
76 memset(bufferList.get(), 0, dataSize);
77 err = AudioObjectGetPropertyData(deviceID, &address, 0, nullptr, &dataSize, bufferList.get());
78
79 return !err && bufferList->mNumberBuffers;
80}

114static bool isValidCaptureDevice(const CoreAudioCaptureDevice& device)
115{
116 // Ignore output devices that have input only for echo cancellation.
117 AudioObjectPropertyAddress address = {
118 kAudioDevicePropertyTapEnabled,
119 kAudioDevicePropertyScopeOutput,
120#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
121 kAudioObjectPropertyElementMain
122#else
123 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
124 kAudioObjectPropertyElementMaster
125 ALLOW_DEPRECATED_DECLARATIONS_END
126#endif
127 };
128 if (AudioObjectHasProperty(device.deviceID(), &address)) {
129 RELEASE_LOG(WebRTC, "Ignoring output device that have input only for echo cancellation");
130 return false;
131 }
132
133 // Ignore non-aggregable devices.
134 UInt32 dataSize = 0;
135 address = {
136 kAudioObjectPropertyCreator,
137 kAudioObjectPropertyScopeGlobal,
138#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
139 kAudioObjectPropertyElementMain
140#else
141 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
142 kAudioObjectPropertyElementMaster
143 ALLOW_DEPRECATED_DECLARATIONS_END
144#endif
145 };
146 CFStringRef name = nullptr;
147 dataSize = sizeof(name);
148 AudioObjectGetPropertyData(device.deviceID(), &address, 0, nullptr, &dataSize, &name);
149 bool isNonAggregable = !name || !String(name).startsWith("com.apple.audio.CoreAudio");
150 if (name)
151 CFRelease(name);
152 if (isNonAggregable) {
153 RELEASE_LOG(WebRTC, "Ignoring output device that is non aggregable");
154 return false;
155 }
156
157 // Ignore unnamed devices and aggregate devices created by VPIO.
158 if (device.label().isEmpty()) {
159 RELEASE_LOG(WebRTC, "Ignoring output device that is unnamed");
160 return false;
161 }
162
163 if (device.label().startsWith("VPAUAggregateAudioDevice")) {
164 RELEASE_LOG(WebRTC, "Ignoring output VPAUAggregateAudioDevice device");
165 return false;
166 }
167
168 if (device.label().contains("WebexMediaAudioDevice")) {
169 RELEASE_LOG(WebRTC, "Ignoring webex audio device");
170 return false;
171 }
172
173 return true;
174}

176void CoreAudioCaptureDeviceManager::scheduleUpdateCaptureDevices()
177{
178 if (m_wasRefreshAudioCaptureDevicesScheduled)
179 return;
180
181 m_wasRefreshAudioCaptureDevicesScheduled = true;
182 callOnMainThread([this] {
183 refreshAudioCaptureDevices(NotifyIfDevicesHaveChanged::Notify);
184 m_wasRefreshAudioCaptureDevicesScheduled = false;
185 });
186}

188Vector<CoreAudioCaptureDevice>& CoreAudioCaptureDeviceManager::coreAudioCaptureDevices()
189{
190 static bool initialized;
191 if (!initialized) {
192 initialized = true;
193 refreshAudioCaptureDevices(NotifyIfDevicesHaveChanged::DoNotNotify);
194
195 auto listener = ^(UInt32 count, const AudioObjectPropertyAddress properties[]) {
196 bool notify = false;
197 for (UInt32 i = 0; i < count; ++i)
198 notify |= (properties[i].mSelector == kAudioHardwarePropertyDevices || properties[i].mSelector == kAudioHardwarePropertyDefaultInputDevice || properties[i].mSelector == kAudioHardwarePropertyDefaultOutputDevice);
199
200 if (notify)
201 CoreAudioCaptureDeviceManager::singleton().scheduleUpdateCaptureDevices();
202 };
203
204 AudioObjectPropertyAddress address = {
205 kAudioHardwarePropertyDevices,
206 kAudioObjectPropertyScopeGlobal,
207#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
208 kAudioObjectPropertyElementMain
209#else
210 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
211 kAudioObjectPropertyElementMaster
212 ALLOW_DEPRECATED_DECLARATIONS_END
213#endif
214 };
215 auto err = AudioObjectAddPropertyListenerBlock(kAudioObjectSystemObject, &address, dispatch_get_main_queue(), listener);
216 if (err)
217 LOG_ERROR("CoreAudioCaptureDeviceManager::devices(%p) AudioObjectAddPropertyListener for kAudioHardwarePropertyDevices returned error %d (%.4s)", this, (int)err, (char*)&err);
218
219 address = {
220 kAudioHardwarePropertyDefaultInputDevice,
221 kAudioObjectPropertyScopeGlobal,
222#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
223 kAudioObjectPropertyElementMain
224#else
225 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
226 kAudioObjectPropertyElementMaster
227 ALLOW_DEPRECATED_DECLARATIONS_END
228#endif
229 };
230 err = AudioObjectAddPropertyListenerBlock(kAudioObjectSystemObject, &address, dispatch_get_main_queue(), listener);
231 if (err)
232 LOG_ERROR("CoreAudioCaptureDeviceManager::devices(%p) AudioObjectAddPropertyListener for kAudioHardwarePropertyDefaultInputDevice returned error %d (%.4s)", this, (int)err, (char*)&err);
233
234 address = {
235 kAudioHardwarePropertyDefaultOutputDevice,
236 kAudioObjectPropertyScopeGlobal,
237#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
238 kAudioObjectPropertyElementMain
239#else
240 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
241 kAudioObjectPropertyElementMaster
242 ALLOW_DEPRECATED_DECLARATIONS_END
243#endif
244 };
245 err = AudioObjectAddPropertyListenerBlock(kAudioObjectSystemObject, &address, dispatch_get_main_queue(), listener);
246 if (err)
247 LOG_ERROR("CoreAudioCaptureDeviceManager::devices(%p) AudioObjectAddPropertyListener for kAudioHardwarePropertyDefaultOutputDevice returned error %d (%.4s)", this, (int)err, (char*)&err);
248 }
249
250 return m_coreAudioCaptureDevices;
251}

262static inline bool hasDevice(const Vector<CoreAudioCaptureDevice>& devices, uint32_t deviceID, CaptureDevice::DeviceType deviceType)
263{
264 return std::any_of(devices.begin(), devices.end(), [&deviceID, deviceType](auto& device) {
265 return device.deviceID() == deviceID && device.type() == deviceType;
266 });

269static inline Vector<CoreAudioCaptureDevice> computeAudioDeviceList()
270{
271 AudioObjectPropertyAddress address = {
272 kAudioHardwarePropertyDevices,
273 kAudioObjectPropertyScopeGlobal,
274#if HAVE(AUDIO_OBJECT_PROPERTY_ELEMENT_MAIN)
275 kAudioObjectPropertyElementMain
276#else
277 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
278 kAudioObjectPropertyElementMaster
279 ALLOW_DEPRECATED_DECLARATIONS_END
280#endif
281 };
282 UInt32 dataSize = 0;
283 auto err = AudioObjectGetPropertyDataSize(kAudioObjectSystemObject, &address, 0, nullptr, &dataSize);
284 if (err) {
285 RELEASE_LOG(WebRTC, "computeAudioDeviceList failed to get size of device list %d (%.4s)", (int)err, (char*)&err);
286 return { };
287 }
288
289 size_t deviceCount = dataSize / sizeof(AudioObjectID);
290 Vector<AudioObjectID> deviceIDs(deviceCount);
291 err = AudioObjectGetPropertyData(kAudioObjectSystemObject, &address, 0, nullptr, &dataSize, deviceIDs.data());
292 if (err) {
293 RELEASE_LOG(WebRTC, "computeAudioDeviceList failed to get device list %d (%.4s)", (int)err, (char*)&err);
294 return { };
295 }
296
297 Vector<CoreAudioCaptureDevice> audioDevices;
298
299 // Microphones
300 for (size_t i = 0; i < deviceCount; i++) {
301 AudioObjectID deviceID = deviceIDs[i];
302
303 if (!deviceHasInputStreams(deviceID) || hasDevice(audioDevices, deviceID, CaptureDevice::DeviceType::Microphone))
304 continue;
305
306 auto microphoneDevice = CoreAudioCaptureDevice::create(deviceID, CaptureDevice::DeviceType::Microphone, { });
307 if (microphoneDevice && isValidCaptureDevice(microphoneDevice.value()))
308 audioDevices.append(WTFMove(microphoneDevice.value()));
309 }
310
311 // Speakers
312 for (size_t i = 0; i < deviceCount; i++) {
313 AudioObjectID deviceID = deviceIDs[i];
314
315 if (!deviceHasOutputStreams(deviceID) || hasDevice(audioDevices, deviceID, CaptureDevice::DeviceType::Speaker))
316 continue;
317
318 String groupID;
319 for (auto relatedDeviceID : CoreAudioCaptureDevice::relatedAudioDeviceIDs(deviceID)) {
320 for (auto& device : audioDevices) {
321 if (device.deviceID() == relatedDeviceID && device.type() == CaptureDevice::DeviceType::Microphone) {
322 groupID = device.persistentId();
323 break;
324 }
325 }
326 }
327
328 auto device = CoreAudioCaptureDevice::create(deviceID, CaptureDevice::DeviceType::Speaker, groupID);
329 if (device) {
330 // If there is no groupID, relate devices if the label is matching.
331 if (groupID.isNull()) {
332 for (auto& existingDevice : audioDevices) {
333 if (existingDevice.label() == device->label() && existingDevice.type() == CaptureDevice::DeviceType::Microphone) {
334 device->setGroupId(existingDevice.persistentId());
335 break;
336 }
337 }
338 }
339 audioDevices.append(WTFMove(device.value()));
340 }
341 }
342 return audioDevices;
343}

345void CoreAudioCaptureDeviceManager::refreshAudioCaptureDevices(NotifyIfDevicesHaveChanged notify)
346{
347 ASSERT(isMainThread());
348
349 auto audioDevices = computeAudioDeviceList();
350 bool haveDeviceChanges = audioDevices.size() != m_coreAudioCaptureDevices.size();
351 if (!haveDeviceChanges) {
352 for (size_t cptr = 0; cptr < audioDevices.size(); ++cptr) {
353 auto& oldDevice = m_coreAudioCaptureDevices[cptr];
354 auto& newDevice = audioDevices[cptr];
355 if (newDevice.type() != oldDevice.type() || newDevice.deviceID() != oldDevice.deviceID() || newDevice.isDefault() != oldDevice.isDefault() || newDevice.enabled() != oldDevice.enabled() || newDevice.isDefault() != oldDevice.isDefault())
356 haveDeviceChanges = true;
357 }
358 }
359 if (!haveDeviceChanges)
360 return;
361
362 std::sort(audioDevices.begin(), audioDevices.end(), [] (auto& first, auto& second) -> bool {
363 return first.isDefault() && !second.isDefault();
364 });
365 m_coreAudioCaptureDevices = WTFMove(audioDevices);
366
367 m_captureDevices.clear();
368 m_speakerDevices.clear();
369 for (auto& device : m_coreAudioCaptureDevices) {
370 if (device.type() == CaptureDevice::DeviceType::Microphone)
371 m_captureDevices.append(device);
372 else
373 m_speakerDevices.append(device);
374 }
375
376 if (notify == NotifyIfDevicesHaveChanged::Notify) {
377 deviceChanged();
378 CoreAudioCaptureSourceFactory::singleton().devicesChanged(m_captureDevices);
379 }
380}
