195OSStatus CoreAudioSharedUnit::setupAudioUnit()
196{
197 if (m_ioUnit)
198 return 0;
199
200 ASSERT(hasClients());
201
202 mach_timebase_info_data_t timebaseInfo;
203 mach_timebase_info(&timebaseInfo);
204 m_DTSConversionRatio = 1e-9 * static_cast<double>(timebaseInfo.numer) / static_cast<double>(timebaseInfo.denom);
205
206 AudioComponentDescription ioUnitDescription = { kAudioUnitType_Output, kAudioUnitSubType_VoiceProcessingIO, kAudioUnitManufacturer_Apple, 0, 0 };
207 AudioComponent ioComponent = PAL::AudioComponentFindNext(nullptr, &ioUnitDescription);
208 ASSERT(ioComponent);
209 if (!ioComponent) {
210 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) unable to find vpio unit component", this);
211 return -1;
212 }
213
214#if !LOG_DISABLED
215 CFStringRef name = nullptr;
216 PAL::AudioComponentCopyName(ioComponent, &name);
217 if (name) {
218 m_ioUnitName = name;
219 CFRelease(name);
220 RELEASE_LOG(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) created \"%s\" component", this, m_ioUnitName.utf8().data());
221 }
222#endif
223
224 auto err = PAL::AudioComponentInstanceNew(ioComponent, &m_ioUnit);
225 if (err) {
226 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) unable to open vpio unit, error %d (%.4s)", this, (int)err, (char*)&err);
227 return err;
228 }
229
230 if (!enableEchoCancellation()) {
231 uint32_t param = 0;
232 err = PAL::AudioUnitSetProperty(m_ioUnit, kAUVoiceIOProperty_VoiceProcessingEnableAGC, kAudioUnitScope_Global, inputBus, &param, sizeof(param));
233 if (err) {
234 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) unable to set vpio automatic gain control, error %d (%.4s)", this, (int)err, (char*)&err);
235 return err;
236 }
237 param = 1;
238 err = PAL::AudioUnitSetProperty(m_ioUnit, kAUVoiceIOProperty_BypassVoiceProcessing, kAudioUnitScope_Global, inputBus, &param, sizeof(param));
239 if (err) {
240 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) unable to set vpio unit echo cancellation, error %d (%.4s)", this, (int)err, (char*)&err);
241 return err;
242 }
243 }
244
245#if PLATFORM(IOS_FAMILY)
246 uint32_t param = 1;
247 err = PAL::AudioUnitSetProperty(m_ioUnit, kAudioOutputUnitProperty_EnableIO, kAudioUnitScope_Input, inputBus, &param, sizeof(param));
248 if (err) {
249 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) unable to enable vpio unit input, error %d (%.4s)", this, (int)err, (char*)&err);
250 return err;
251 }
252#else
253 if (!m_captureDeviceID) {
254 err = defaultInputDevice(&m_captureDeviceID);
255 if (err)
256 return err;
257 }
258
259 err = PAL::AudioUnitSetProperty(m_ioUnit, kAudioOutputUnitProperty_CurrentDevice, kAudioUnitScope_Global, inputBus, &m_captureDeviceID, sizeof(m_captureDeviceID));
260 if (err) {
261 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) unable to set vpio unit capture device ID %d, error %d (%.4s)", this, (int)m_captureDeviceID, (int)err, (char*)&err);
262 return err;
263 }
264#endif
265
266 err = configureMicrophoneProc();
267 if (err)
268 return err;
269
270 err = configureSpeakerProc();
271 if (err)
272 return err;
273
274 err = PAL::AudioUnitInitialize(m_ioUnit);
275 if (err) {
276 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::setupAudioUnit(%p) AudioUnitInitialize() failed, error %d (%.4s)", this, (int)err, (char*)&err);
277 return err;
278 }
279 m_ioUnitInitialized = true;
280
281 unduck();
282
283 return err;
284}

293OSStatus CoreAudioSharedUnit::configureMicrophoneProc()
294{
295 AURenderCallbackStruct callback = { microphoneCallback, this };
296 auto err = PAL::AudioUnitSetProperty(m_ioUnit, kAudioOutputUnitProperty_SetInputCallback, kAudioUnitScope_Global, inputBus, &callback, sizeof(callback));
297 if (err) {
298 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::configureMicrophoneProc(%p) unable to set vpio unit mic proc, error %d (%.4s)", this, (int)err, (char*)&err);
299 return err;
300 }
301
302 AudioStreamBasicDescription microphoneProcFormat = { };
303
304 UInt32 size = sizeof(microphoneProcFormat);
305 err = PAL::AudioUnitGetProperty(m_ioUnit, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Output, inputBus, &microphoneProcFormat, &size);
306 if (err) {
307 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::configureMicrophoneProc(%p) unable to get output stream format, error %d (%.4s)", this, (int)err, (char*)&err);
308 return err;
309 }
310
311 microphoneProcFormat.mSampleRate = sampleRate();
312 err = PAL::AudioUnitSetProperty(m_ioUnit, kAudioUnitProperty_StreamFormat, kAudioUnitScope_Output, inputBus, &microphoneProcFormat, size);
313 if (err) {
314 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::configureMicrophoneProc(%p) unable to set output stream format, error %d (%.4s)", this, (int)err, (char*)&err);
315 return err;
316 }
317
318 m_microphoneSampleBuffer = AudioSampleBufferList::create(microphoneProcFormat, preferredIOBufferSize() * 2);
319 m_microphoneProcFormat = microphoneProcFormat;
320
321 return err;
322}

356void CoreAudioSharedUnit::checkTimestamps(const AudioTimeStamp& timeStamp, uint64_t sampleTime, double hostTime)
357{
358 if (!timeStamp.mSampleTime || sampleTime == m_latestMicTimeStamp || !hostTime)
359 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::checkTimestamps: unusual timestamps, sample time = %lld, previous sample time = %lld, hostTime %f", sampleTime, m_latestMicTimeStamp, hostTime);

363OSStatus CoreAudioSharedUnit::provideSpeakerData(AudioUnitRenderActionFlags& /*ioActionFlags*/, const AudioTimeStamp& timeStamp, UInt32 /*inBusNumber*/, UInt32 inNumberFrames, AudioBufferList* ioData)
364{
365 // Called when the audio unit needs data to play through the speakers.
366#if !LOG_DISABLED
367 ++m_speakerProcsCalled;
368#endif
369
370 if (m_speakerSampleBuffer->sampleCapacity() < inNumberFrames) {
371 if (m_activeSources.isEmpty())
372 return 0;
373 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::provideSpeakerData: speaker sample buffer size (%d) too small for amount of sample data requested (%d)!", m_speakerSampleBuffer->sampleCapacity(), (int)inNumberFrames);
374 // FIXME: This fails the capture, we should thus either reconfigure the audio unit or notify all clients that capture is failing.
375 return kAudio_ParamError;
376 }
377
378 if (m_activeSources.isEmpty())
379 return 0;
380
381 double adjustedHostTime = m_DTSConversionRatio * timeStamp.mHostTime;
382 uint64_t sampleTime = timeStamp.mSampleTime;
383#if !LOG_DISABLED
384 checkTimestamps(timeStamp, sampleTime, adjustedHostTime);
385#endif
386 m_speakerSampleBuffer->setTimes(adjustedHostTime, sampleTime);
387
388 AudioBufferList& bufferList = m_speakerSampleBuffer->bufferList();
389 for (uint32_t i = 0; i < bufferList.mNumberBuffers; ++i)
390 bufferList.mBuffers[i] = ioData->mBuffers[i];
391
392 bool firstSource = true;
393 for (auto& source : m_activeSources) {
394 source->pullSamples(*m_speakerSampleBuffer.get(), inNumberFrames, adjustedHostTime, sampleTime, firstSource ? AudioSampleDataSource::Copy : AudioSampleDataSource::Mix);
395 firstSource = false;
396 }
397
398 return noErr;
399}

409OSStatus CoreAudioSharedUnit::processMicrophoneSamples(AudioUnitRenderActionFlags& ioActionFlags, const AudioTimeStamp& timeStamp, UInt32 inBusNumber, UInt32 inNumberFrames, AudioBufferList* /*ioData*/)
410{
411 // Pull through the vpio unit to our mic buffer.
412 m_microphoneSampleBuffer->reset();
413 AudioBufferList& bufferList = m_microphoneSampleBuffer->bufferList();
414 auto err = AudioUnitRender(m_ioUnit, &ioActionFlags, &timeStamp, inBusNumber, inNumberFrames, &bufferList);
415 if (err) {
416 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::processMicrophoneSamples(%p) AudioUnitRender failed with error %d (%.4s), bufferList size %d, inNumberFrames %d ", this, (int)err, (char*)&err, (int)bufferList.mBuffers[0].mDataByteSize, (int)inNumberFrames);
417 if (err == kAudio_ParamError) {
418 // Our buffer might be too small, the preferred buffer size or sample rate might have changed.
419 callOnMainThread([] {
420 CoreAudioSharedUnit::singleton().reconfigure();
421 });
422 }
423 // We return early so that if this error happens, we do not increment m_microphoneProcsCalled and fail the capture once timer kicks in.
424 return err;
425 }
426
427 ++m_microphoneProcsCalled;
428
429 double adjustedHostTime = m_DTSConversionRatio * timeStamp.mHostTime;
430 uint64_t sampleTime = timeStamp.mSampleTime;
431#if !LOG_DISABLED
432 checkTimestamps(timeStamp, sampleTime, adjustedHostTime);
433#endif
434 m_latestMicTimeStamp = sampleTime;
435 m_microphoneSampleBuffer->setTimes(adjustedHostTime, sampleTime);
436
437 if (volume() != 1.0)
438 m_microphoneSampleBuffer->applyGain(volume());
439
440 audioSamplesAvailable(MediaTime(sampleTime, m_microphoneProcFormat.sampleRate()), m_microphoneSampleBuffer->bufferList(), m_microphoneProcFormat, inNumberFrames);
441 return noErr;
442}

452void CoreAudioSharedUnit::cleanupAudioUnit()
453{
454 if (m_ioUnitInitialized) {
455 ASSERT(m_ioUnit);
456 auto err = AudioUnitUninitialize(m_ioUnit);
457 if (err)
458 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::cleanupAudioUnit(%p) AudioUnitUninitialize failed with error %d (%.4s)", this, (int)err, (char*)&err);
459 m_ioUnitInitialized = false;
460 }
461
462 if (m_ioUnit) {
463 PAL::AudioComponentInstanceDispose(m_ioUnit);
464 m_ioUnit = nullptr;
465 }
466
467 m_microphoneSampleBuffer = nullptr;
468 m_speakerSampleBuffer = nullptr;
469#if !LOG_DISABLED
470 m_ioUnitName = emptyString();
471#endif
472}

474OSStatus CoreAudioSharedUnit::reconfigureAudioUnit()
475{
476 OSStatus err;
477 if (!hasAudioUnit())
478 return 0;
479
480 if (m_ioUnitStarted) {
481 err = PAL::AudioOutputUnitStop(m_ioUnit);
482 if (err) {
483 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::reconfigureAudioUnit(%p) AudioOutputUnitStop failed with error %d (%.4s)", this, (int)err, (char*)&err);
484 return err;
485 }
486 }
487
488 cleanupAudioUnit();
489 err = setupAudioUnit();
490 if (err)
491 return err;
492
493 if (m_ioUnitStarted) {
494 err = PAL::AudioOutputUnitStart(m_ioUnit);
495 if (err) {
496 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::reconfigureAudioUnit(%p) AudioOutputUnitStart failed with error %d (%.4s)", this, (int)err, (char*)&err);
497 return err;
498 }
499 }
500 return err;
501}

503OSStatus CoreAudioSharedUnit::startInternal()
504{
505 OSStatus err;
506 if (!m_ioUnit) {
507 err = setupAudioUnit();
508 if (err) {
509 cleanupAudioUnit();
510 ASSERT(!m_ioUnit);
511 return err;
512 }
513 ASSERT(m_ioUnit);
514 }
515
516 unduck();
517
518 err = PAL::AudioOutputUnitStart(m_ioUnit);
519 if (err) {
520 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::start(%p) AudioOutputUnitStart failed with error %d (%.4s)", this, (int)err, (char*)&err);
521 cleanupAudioUnit();
522 ASSERT(!m_ioUnit);
523 return err;
524 }
525
526 m_ioUnitStarted = true;
527
528 m_verifyCapturingTimer.startRepeating(10_s);
529 m_microphoneProcsCalled = 0;
530 m_microphoneProcsCalledLastTime = 0;
531
532 return 0;
533}
534
535void CoreAudioSharedUnit::verifyIsCapturing()
536{
537 if (m_microphoneProcsCalledLastTime != m_microphoneProcsCalled) {
538 m_microphoneProcsCalledLastTime = m_microphoneProcsCalled;
539 if (m_verifyCapturingTimer.repeatInterval() == 10_s)
540 m_verifyCapturingTimer.startRepeating(2_s);
541 return;
542 }
543
544 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::verifyIsCapturing - no audio received in %d seconds, failing", static_cast<int>(m_verifyCapturingTimer.repeatInterval().value()));
545 captureFailed();
546}

548void CoreAudioSharedUnit::stopInternal()
549{
550 m_verifyCapturingTimer.stop();
551
552 if (!m_ioUnit || !m_ioUnitStarted)
553 return;
554
555 auto err = PAL::AudioOutputUnitStop(m_ioUnit);
556 if (err) {
557 RELEASE_LOG_ERROR(WebRTC, "CoreAudioSharedUnit::stop(%p) AudioOutputUnitStop failed with error %d (%.4s)", this, (int)err, (char*)&err);
558 return;
559 }
560
561 m_ioUnitStarted = false;
562}

602static CaptureSourceOrError initializeCoreAudioCaptureSource(Ref<CoreAudioCaptureSource>&& source, const MediaConstraints* constraints)

611CaptureSourceOrError CoreAudioCaptureSource::create(String&& deviceID, String&& hashSalt, const MediaConstraints* constraints)
612{
613#if PLATFORM(MAC)
614 auto device = CoreAudioCaptureDeviceManager::singleton().coreAudioDeviceWithUID(deviceID);
615 if (!device)
616 return { "No CoreAudioCaptureSource device"_s };
617
618 auto source = adoptRef(*new CoreAudioCaptureSource(WTFMove(deviceID), String { device->label() }, WTFMove(hashSalt), device->deviceID()));
619#elif PLATFORM(IOS_FAMILY)
620 auto device = AVAudioSessionCaptureDeviceManager::singleton().audioSessionDeviceWithUID(WTFMove(deviceID));
621 if (!device)
622 return { };
623
624 auto source = adoptRef(*new CoreAudioCaptureSource(WTFMove(deviceID), String { device->label() }, WTFMove(hashSalt), 0));
625#endif
626 return initializeCoreAudioCaptureSource(WTFMove(source), constraints);
627}

720void CoreAudioCaptureSource::initializeToStartProducingData()
721{
722 if (m_isReadyToStart)
723 return;
724
725 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER);
726 m_isReadyToStart = true;
727
728 auto& unit = this->unit();
729 unit.setCaptureDevice(String { persistentID() }, m_captureDeviceID);
730
731 bool shouldReconfigure = echoCancellation() != unit.enableEchoCancellation() || sampleRate() != unit.sampleRate() || volume() != unit.volume();
732 unit.setEnableEchoCancellation(echoCancellation());
733 unit.setSampleRate(sampleRate());
734 unit.setVolume(volume());
735
736 if (shouldReconfigure)
737 unit.reconfigure();
738
739 unit.addClient(*this);
740}
