60CaptureSourceOrError DisplayCaptureSourceCocoa::create(const CaptureDevice& device, const MediaConstraints* constraints)
61{
62#if PLATFORM(MAC)
63 switch (device.type()) {
64 case CaptureDevice::DeviceType::Screen:
65 return create(ScreenDisplayCapturerMac::create(device.persistentId()), device, constraints);
66 case CaptureDevice::DeviceType::Window:
67 return create(WindowDisplayCapturerMac::create(device.persistentId()), device, constraints);
68 case CaptureDevice::DeviceType::Microphone:
69 case CaptureDevice::DeviceType::Speaker:
70 case CaptureDevice::DeviceType::Camera:
71 case CaptureDevice::DeviceType::Unknown:
72 ASSERT_NOT_REACHED();
73 break;
74 }
75#else
76 UNUSED_PARAM(device);
77 UNUSED_PARAM(constraints);
78#endif
79 return { };
80}

82CaptureSourceOrError DisplayCaptureSourceCocoa::create(Expected<UniqueRef<Capturer>, String>&& capturer, const CaptureDevice& device, const MediaConstraints* constraints)
83{
84 if (!capturer.has_value())
85 return CaptureSourceOrError { WTFMove(capturer.error()) };
86
87 auto source = adoptRef(*new DisplayCaptureSourceCocoa(WTFMove(capturer.value()), String { device.label() }));
88 if (constraints) {
89 auto result = source->applyConstraints(*constraints);
90 if (result)
91 return WTFMove(result.value().badConstraint);
92 }
93
94 return CaptureSourceOrError(WTFMove(source));
95}

111const RealtimeMediaSourceCapabilities& DisplayCaptureSourceCocoa::capabilities()
112{
113 if (!m_capabilities) {
114 RealtimeMediaSourceCapabilities capabilities(settings().supportedConstraints());
115
116 // FIXME: what should these be?

118 capabilities.setHeight(CapabilityValueOrRange(1, 2160));

120
121 m_capabilities = WTFMove(capabilities);
122 }
123 return m_capabilities.value();

126const RealtimeMediaSourceSettings& DisplayCaptureSourceCocoa::settings()
127{
128 if (!m_currentSettings) {
129 RealtimeMediaSourceSettings settings;
130 settings.setFrameRate(frameRate());
131
132 auto size = this->size();
133 settings.setWidth(size.width());
134 settings.setHeight(size.height());
135
136 settings.setDisplaySurface(m_capturer->surfaceType());
137 settings.setLogicalSurface(false);
138
139 RealtimeMediaSourceSupportedConstraints supportedConstraints;
140 supportedConstraints.setSupportsFrameRate(true);
141 supportedConstraints.setSupportsWidth(true);
142 supportedConstraints.setSupportsHeight(true);
143 supportedConstraints.setSupportsDisplaySurface(true);
144 supportedConstraints.setSupportsLogicalSurface(true);
145
146 settings.setSupportedConstraints(supportedConstraints);
147
148 m_currentSettings = WTFMove(settings);
149 }
150 return m_currentSettings.value();
151}

153void DisplayCaptureSourceCocoa::settingsDidChange(OptionSet<RealtimeMediaSourceSettings::Flag> settings)
154{
155 if (settings.contains(RealtimeMediaSourceSettings::Flag::FrameRate) && m_timer.isActive())
156 m_timer.startRepeating(1_s / frameRate());
157
158 if (settings.containsAny({ RealtimeMediaSourceSettings::Flag::Width, RealtimeMediaSourceSettings::Flag::Height }))
159 m_bufferAttributes = nullptr;
160
161 m_currentSettings = { };
162}

164void DisplayCaptureSourceCocoa::startProducingData()
165{
166#if PLATFORM(IOS_FAMILY)
167 RealtimeMediaSourceCenter::singleton().displayCaptureFactory().setActiveSource(*this);
168#endif
169
170 m_startTime = MonotonicTime::now();
171 m_timer.startRepeating(1_s / frameRate());
172
173 if (!m_capturer->start(frameRate()))
174 captureFailed();
175}
176
177void DisplayCaptureSourceCocoa::stopProducingData()
178{
179 m_timer.stop();
180 m_elapsedTime += MonotonicTime::now() - m_startTime;
181 m_startTime = MonotonicTime::nan();
182
183 m_capturer->stop();
184}
185
186Seconds DisplayCaptureSourceCocoa::elapsedTime()
187{
188 if (std::isnan(m_startTime))
189 return m_elapsedTime;
190
191 return m_elapsedTime + (MonotonicTime::now() - m_startTime);
192}

194// We keep the aspect ratio of the intrinsic size for the frame size as getDisplayMedia allows max constraints only.
195void DisplayCaptureSourceCocoa::updateFrameSize()
196{
197 auto intrinsicSize = this->intrinsicSize();
198
199 auto frameSize = size();
200 if (!frameSize.height())
201 frameSize.setHeight(intrinsicSize.height());
202 if (!frameSize.width())
203 frameSize.setWidth(intrinsicSize.width());
204
205 auto maxHeight = std::min(frameSize.height(), intrinsicSize.height());
206 auto maxWidth = std::min(frameSize.width(), intrinsicSize.width());
207
208 auto heightForMaxWidth = maxWidth * intrinsicSize.height() / intrinsicSize.width();
209 auto widthForMaxHeight = maxHeight * intrinsicSize.width() / intrinsicSize.height();
210
211 if (heightForMaxWidth <= maxHeight) {
212 setSize({ maxWidth, heightForMaxWidth });
213 return;
214 }
215 if (widthForMaxHeight <= maxWidth) {
216 setSize({ widthForMaxHeight, maxHeight });
217 return;
218 }
219 setSize(intrinsicSize);
220}

222void DisplayCaptureSourceCocoa::emitFrame()
223{
224#if PLATFORM(COCOA) && !PLATFORM(IOS_FAMILY_SIMULATOR)
225 if (muted())
226 return;
227
228 if (!m_imageTransferSession)
229 m_imageTransferSession = ImageTransferSessionVT::create(preferedPixelBufferFormat());
230
231 auto sampleTime = MediaTime::createWithDouble((elapsedTime() + 100_ms).seconds());
232
233 auto frame = m_capturer->generateFrame();
234 auto imageSize = WTF::switchOn(frame,
235 [](RetainPtr<IOSurfaceRef> surface) -> IntSize {
236 if (!surface)
237 return { };
238
239 return IntSize(IOSurfaceGetWidth(surface.get()), IOSurfaceGetHeight(surface.get()));
240 },
241 [](RefPtr<NativeImage> image) -> IntSize {
242 if (!image)
243 return { };
244
245 return image->size();
246 }
247 );
248
249 ASSERT(!imageSize.isEmpty());
250 if (imageSize.isEmpty())
251 return;
252
253 if (intrinsicSize() != imageSize) {
254 setIntrinsicSize(imageSize);
255 updateFrameSize();
256 }
257
258 auto sample = WTF::switchOn(frame,
259 [this, sampleTime](RetainPtr<IOSurfaceRef> surface) -> RefPtr<MediaSample> {
260 if (!surface)
261 return nullptr;
262
263 return m_imageTransferSession->createMediaSample(surface.get(), sampleTime, size());
264 },
265 [this, sampleTime](RefPtr<NativeImage> image) -> RefPtr<MediaSample> {
266 if (!image)
267 return nullptr;
268
269 return m_imageTransferSession->createMediaSample(image->platformImage().get(), sampleTime, size());
270 }
271 );
272
273 if (!sample) {
274 ASSERT_NOT_REACHED();
275 return;
276 }
277
278 videoSampleAvailable(*sample.get());
279#endif
280}
