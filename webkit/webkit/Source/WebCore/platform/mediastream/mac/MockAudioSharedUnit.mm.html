52static inline size_t alignTo16Bytes(size_t size)
53{
54 return (size + 15) & ~15;
55}
56
57static const double Tau = 2 * M_PI;
58static const double BipBopDuration = 0.07;
59static const double BipBopVolume = 0.5;
60static const double BipFrequency = 1500;
61static const double BopFrequency = 500;
62static const double HumFrequency = 150;
63static const double HumVolume = 0.1;
64static const double NoiseFrequency = 3000;
65static const double NoiseVolume = 0.05;
66
67template <typename AudioSampleType>
68static void writeHum(float amplitude, float frequency, float sampleRate, AudioSampleType *p, uint64_t count)
69{
70 float humPeriod = sampleRate / frequency;
71 for (uint64_t i = 0; i < count; ++i)
72 *p++ = amplitude * sin(i * Tau / humPeriod);
73}
74
75template <typename AudioSampleType>
76static void addHum(float amplitude, float frequency, float sampleRate, uint64_t start, AudioSampleType *p, uint64_t count)
77{
78 float humPeriod = sampleRate / frequency;
79 for (uint64_t i = start, end = start + count; i < end; ++i) {
80 AudioSampleType a = amplitude * sin(i * Tau / humPeriod);
81 a += *p;
82 *p++ = a;
83 }
84}

86CaptureSourceOrError MockRealtimeAudioSource::create(String&& deviceID, String&& name, String&& hashSalt, const MediaConstraints* constraints)
87{
88 auto device = MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(deviceID);
89 ASSERT(device);
90 if (!device)
91 return { "No mock microphone device"_s };
92
93 MockAudioSharedUnit::singleton().setDeviceID(deviceID);
94 return CoreAudioCaptureSource::createForTesting(WTFMove(deviceID), WTFMove(name), WTFMove(hashSalt), constraints, MockAudioSharedUnit::singleton());

126OSStatus MockAudioSharedUnit::reconfigureAudioUnit()
127{
128 if (!hasAudioUnit())
129 return 0;
130
131 m_timer.stop();
132 m_lastRenderTime = MonotonicTime::nan();
133 m_workQueue->dispatch([this] {
134 reconfigure();
135 callOnMainThread([this] {
136 m_lastRenderTime = MonotonicTime::now();
137 m_timer.startRepeating(renderInterval());
138 });
139 });
140 return 0;
141}

173void MockAudioSharedUnit::tick()
174{
175 if (std::isnan(m_lastRenderTime))
176 m_lastRenderTime = MonotonicTime::now();
177
178 MonotonicTime now = MonotonicTime::now();
179
180 if (m_delayUntil) {
181 if (m_delayUntil < now)
182 return;
183 m_delayUntil = MonotonicTime();
184 }
185
186 Seconds delta = now - m_lastRenderTime;
187 m_lastRenderTime = now;
188
189 m_workQueue->dispatch([this, delta] {
190 render(delta);
191 });
192}

199void MockAudioSharedUnit::reconfigure()
200{
201 ASSERT(!isMainThread());
202
203 auto rate = sampleRate();
204 ASSERT(rate);
205
206 m_maximiumFrameCount = WTF::roundUpToPowerOfTwo(renderInterval().seconds() * rate * 2);
207 ASSERT(m_maximiumFrameCount);
208
209 const int bytesPerFloat = sizeof(Float32);
210 const int bitsPerByte = 8;
211 const int channelCount = m_channelCount;
212 const bool isFloat = true;
213 const bool isBigEndian = false;
214 const bool isNonInterleaved = true;
215 FillOutASBDForLPCM(m_streamFormat, rate, channelCount, bitsPerByte * bytesPerFloat, bitsPerByte * bytesPerFloat, isFloat, isBigEndian, isNonInterleaved);
216
217 m_audioBufferList = makeUnique<WebAudioBufferList>(m_streamFormat, m_maximiumFrameCount);
218
219 CMFormatDescriptionRef formatDescription;
220 PAL::CMAudioFormatDescriptionCreate(NULL, &m_streamFormat, 0, NULL, 0, NULL, NULL, &formatDescription);
221 m_formatDescription = adoptCF(formatDescription);
222
223 size_t sampleCount = 2 * rate;
224 m_bipBopBuffer.resize(sampleCount);
225 m_bipBopBuffer.fill(0);
226
227 size_t bipBopSampleCount = ceil(BipBopDuration * rate);
228 size_t bipStart = 0;
229 size_t bopStart = rate;
230
231 addHum(BipBopVolume, BipFrequency, rate, 0, m_bipBopBuffer.data() + bipStart, bipBopSampleCount);
232 addHum(BipBopVolume, BopFrequency, rate, 0, m_bipBopBuffer.data() + bopStart, bipBopSampleCount);
233 if (!enableEchoCancellation())
234 addHum(NoiseVolume, NoiseFrequency, rate, 0, m_bipBopBuffer.data(), sampleCount);
235}

237void MockAudioSharedUnit::emitSampleBuffers(uint32_t frameCount)
238{
239 ASSERT(!isMainThread());
240 ASSERT(m_formatDescription);
241
242 CMTime startTime = PAL::CMTimeMake(m_samplesEmitted, sampleRate());
243 m_samplesEmitted += frameCount;
244
245 audioSamplesAvailable(PAL::toMediaTime(startTime), *m_audioBufferList, CAAudioStreamDescription(m_streamFormat), frameCount);
246}
247
248void MockAudioSharedUnit::render(Seconds delta)
249{
250 ASSERT(!isMainThread());
251 if (!m_audioBufferList || !m_bipBopBuffer.size())
252 reconfigure();
253
254 uint32_t totalFrameCount = alignTo16Bytes(delta.seconds() * sampleRate());
255 uint32_t frameCount = std::min(totalFrameCount, m_maximiumFrameCount);
256
257 while (frameCount) {
258 uint32_t bipBopStart = m_samplesRendered % m_bipBopBuffer.size();
259 uint32_t bipBopRemain = m_bipBopBuffer.size() - bipBopStart;
260 uint32_t bipBopCount = std::min(frameCount, bipBopRemain);
261 for (auto& audioBuffer : m_audioBufferList->buffers()) {
262 audioBuffer.mDataByteSize = frameCount * m_streamFormat.mBytesPerFrame;
263 memcpy(audioBuffer.mData, &m_bipBopBuffer[bipBopStart], sizeof(Float32) * bipBopCount);
264 addHum(HumVolume, HumFrequency, sampleRate(), m_samplesRendered, static_cast<float*>(audioBuffer.mData), bipBopCount);
265 }
266 emitSampleBuffers(bipBopCount);
267 m_samplesRendered += bipBopCount;
268 totalFrameCount -= bipBopCount;
269 frameCount = std::min(totalFrameCount, m_maximiumFrameCount);
270 }
