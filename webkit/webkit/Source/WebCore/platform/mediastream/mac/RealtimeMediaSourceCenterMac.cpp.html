45namespace WebCore {
46
47class VideoCaptureSourceFactoryMac final : public VideoCaptureFactory {
48public:
49 CaptureSourceOrError createVideoCaptureSource(const CaptureDevice& device, String&& hashSalt, const MediaConstraints* constraints) final
50 {
51 ASSERT(device.type() == CaptureDevice::DeviceType::Camera);
52 return AVVideoCaptureSource::create(String { device.persistentId() }, WTFMove(hashSalt), constraints);
53 }
54
55private:
56 CaptureDeviceManager& videoCaptureDeviceManager() { return AVCaptureDeviceManager::singleton(); }
57};
58
59class DisplayCaptureSourceFactoryMac final : public DisplayCaptureFactory {
60public:
61 CaptureSourceOrError createDisplayCaptureSource(const CaptureDevice& device, const MediaConstraints* constraints) final
62 {
63#if PLATFORM(IOS_FAMILY)
64 UNUSED_PARAM(device);
65 UNUSED_PARAM(constraints);
66#endif
67#if PLATFORM(MAC)
68 return DisplayCaptureSourceCocoa::create(device, constraints);
69#else
70 return { };
71#endif
72 }
73private:
74 CaptureDeviceManager& displayCaptureDeviceManager() { return DisplayCaptureManagerCocoa::singleton(); }
75};
76
77AudioCaptureFactory& RealtimeMediaSourceCenter::defaultAudioCaptureFactory()
78{
79 return CoreAudioCaptureSource::factory();
80}
81
82VideoCaptureFactory& RealtimeMediaSourceCenter::defaultVideoCaptureFactory()
83{
84 static NeverDestroyed<VideoCaptureSourceFactoryMac> factory;
85 return factory.get();
86}
87
88DisplayCaptureFactory& RealtimeMediaSourceCenter::defaultDisplayCaptureFactory()
89{
90 static NeverDestroyed<DisplayCaptureSourceFactoryMac> factory;
91 return factory.get();
92}
