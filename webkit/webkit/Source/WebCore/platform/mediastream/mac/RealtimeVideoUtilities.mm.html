33RetainPtr<CVPixelBufferPoolRef> createPixelBufferPool(size_t width, size_t height, OSType videoCaptureFormat)
34{
35 auto pixelAttributes = @{
36 (__bridge NSString *)kCVPixelBufferWidthKey: @(width),
37 (__bridge NSString *)kCVPixelBufferHeightKey: @(height),
38 (__bridge NSString *)kCVPixelBufferPixelFormatTypeKey: @(videoCaptureFormat),
39 (__bridge NSString *)kCVPixelBufferCGImageCompatibilityKey: @NO,
40#if PLATFORM(IOS_FAMILY)
41 (__bridge NSString *)kCVPixelFormatOpenGLESCompatibility : @YES,
42#else
43 (__bridge NSString *)kCVPixelBufferOpenGLCompatibilityKey : @YES,
44#endif
45 (__bridge NSString *)kCVPixelBufferIOSurfacePropertiesKey : @{ }
46 };
47
48 CVPixelBufferPoolRef pool = nullptr;
49 auto status = CVPixelBufferPoolCreate(kCFAllocatorDefault, nullptr, (__bridge CFDictionaryRef)pixelAttributes, &pool);
50
51 if (status != kCVReturnSuccess)
52 return nullptr;
53
54 return adoptCF(pool);
55}
56
57RetainPtr<CVPixelBufferRef> createPixelBufferFromPool(CVPixelBufferPoolRef pixelBufferPool)
58{
59 CVPixelBufferRef pixelBuffer = nullptr;
60 auto status = CVPixelBufferPoolCreatePixelBuffer(kCFAllocatorDefault, pixelBufferPool, &pixelBuffer);
61
62 if (status != kCVReturnSuccess)
63 return nullptr;
64 return adoptCF(pixelBuffer);
65}
