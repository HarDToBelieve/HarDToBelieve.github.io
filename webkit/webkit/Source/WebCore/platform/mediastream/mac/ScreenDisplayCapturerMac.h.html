70 class DisplaySurface {
71 public:
72 DisplaySurface() = default;
73 explicit DisplaySurface(IOSurfaceRef surface)
74 : m_surface(surface)
75 {
76 if (m_surface)
77 IOSurfaceIncrementUseCount(m_surface.get());
78 }
79
80 ~DisplaySurface()
81 {
82 if (m_surface)
83 IOSurfaceDecrementUseCount(m_surface.get());
84 }
85
86 DisplaySurface& operator=(IOSurfaceRef surface)
87 {
88 if (m_surface)
89 IOSurfaceDecrementUseCount(m_surface.get());
90 if (surface)
91 IOSurfaceIncrementUseCount(surface);
92 m_surface = surface;
93 return *this;
94 }
95
96 IOSurfaceRef ioSurface() const { return m_surface.get(); }
97
98 private:
99 RetainPtr<IOSurfaceRef> m_surface;
100 };
