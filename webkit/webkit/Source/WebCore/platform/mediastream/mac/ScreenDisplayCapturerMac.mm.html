52static std::optional<CGDirectDisplayID> updateDisplayID(CGDirectDisplayID displayID)
53{
54 uint32_t displayCount = 0;
55 auto err = CGGetActiveDisplayList(0, nullptr, &displayCount);
56 if (err) {
57 RELEASE_LOG(WebRTC, "CGGetActiveDisplayList() returned error %d when trying to get display count", static_cast<int>(err));
58 return std::nullopt;
59 }
60
61 if (!displayCount) {
62 RELEASE_LOG(WebRTC, "CGGetActiveDisplayList() returned a display count of 0");
63 return std::nullopt;
64 }
65
66 Vector<CGDirectDisplayID> activeDisplays(displayCount);
67 err = CGGetActiveDisplayList(displayCount, activeDisplays.data(), &displayCount);
68 if (err) {
69 RELEASE_LOG(WebRTC, "CGGetActiveDisplayList() returned error %d when trying to get the active display list", static_cast<int>(err));
70 return std::nullopt;
71 }
72
73 auto displayMask = CGDisplayIDToOpenGLDisplayMask(displayID);
74 for (auto display : activeDisplays) {
75 if (displayMask == CGDisplayIDToOpenGLDisplayMask(display))
76 return display;
77 }
78
79 return std::nullopt;
80}

82Expected<UniqueRef<DisplayCaptureSourceCocoa::Capturer>, String> ScreenDisplayCapturerMac::create(const String& deviceID)
83{
84 auto displayID = parseInteger<uint32_t>(deviceID);
85 if (!displayID)
86 return makeUnexpected("Invalid display device ID"_s);
87 auto actualDisplayID = updateDisplayID(*displayID);
88 if (!actualDisplayID)
89 return makeUnexpected("Invalid display ID"_s);
90 return UniqueRef<DisplayCaptureSourceCocoa::Capturer>(makeUniqueRef<ScreenDisplayCapturerMac>(actualDisplayID.value()));
91}

106bool ScreenDisplayCapturerMac::createDisplayStream(float frameRate)
107{
108 static const int screenQueueMaximumLength = 6;
109
110 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER);
111
112 auto actualDisplayID = updateDisplayID(m_displayID);
113 if (!actualDisplayID) {
114 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "invalid display ID: ", m_displayID);
115 return false;
116 }
117
118 if (m_displayID != actualDisplayID.value()) {
119 m_displayID = actualDisplayID.value();
120 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, "display ID changed to ", static_cast<int>(m_displayID));
121 m_displayStream = nullptr;
122 }
123
124 if (!m_displayStream) {
125 auto displayMode = adoptCF(CGDisplayCopyDisplayMode(m_displayID));
126 auto screenWidth = CGDisplayModeGetPixelsWide(displayMode.get());
127 auto screenHeight = CGDisplayModeGetPixelsHigh(displayMode.get());
128 if (!screenWidth || !screenHeight) {
129 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "unable to get screen width/height");
130 return false;
131 }
132
133 if (!m_captureQueue)
134 m_captureQueue = adoptOSObject(dispatch_queue_create("ScreenDisplayCapturerMac Capture Queue", DISPATCH_QUEUE_SERIAL));
135
136 NSDictionary* streamOptions = @{
137 (__bridge NSString *)kCGDisplayStreamMinimumFrameTime : @(1 / frameRate),
138 (__bridge NSString *)kCGDisplayStreamQueueDepth : @(screenQueueMaximumLength),
139 (__bridge NSString *)kCGDisplayStreamColorSpace : (__bridge id)sRGBColorSpaceRef(),
140 (__bridge NSString *)kCGDisplayStreamShowCursor : @YES,
141 };
142
143 auto weakThis = makeWeakPtr(*this);
144 auto frameAvailableBlock = ^(CGDisplayStreamFrameStatus status, uint64_t displayTime, IOSurfaceRef frameSurface, CGDisplayStreamUpdateRef updateRef) {
145
146 if (!frameSurface || !displayTime)
147 return;
148
149 size_t count;
150 auto* rects = CGDisplayStreamUpdateGetRects(updateRef, kCGDisplayStreamUpdateDirtyRects, &count);
151 if (!rects || !count)
152 return;
153
154 RunLoop::main().dispatch([weakThis, status, frame = DisplaySurface { frameSurface }]() mutable {
155 if (!weakThis)
156 return;
157 weakThis->newFrame(status, WTFMove(frame));
158 });
159 };
160
161 m_displayStream = adoptCF(CGDisplayStreamCreateWithDispatchQueue(m_displayID, screenWidth, screenHeight, preferedPixelBufferFormat(), (__bridge CFDictionaryRef)streamOptions, m_captureQueue.get(), frameAvailableBlock));
162 if (!m_displayStream) {
163 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "CGDisplayStreamCreate failed");
164 return false;
165 }
166 }
167
168 if (!m_observingDisplayChanges) {
169 CGDisplayRegisterReconfigurationCallback(displayReconfigurationCallBack, this);
170 m_observingDisplayChanges = true;
171 }
172
173 return true;
174}

204bool ScreenDisplayCapturerMac::startDisplayStream(float frameRate)
205{
206 auto actualDisplayID = updateDisplayID(m_displayID);
207 if (!actualDisplayID)
208 return false;
209
210 if (m_displayID != actualDisplayID.value()) {
211 m_displayID = actualDisplayID.value();
212 ALWAYS_LOG_IF(loggerPtr(), LOGIDENTIFIER, "display ID changed to ", static_cast<int>(m_displayID));
213 }
214
215 if (!m_displayStream && !createDisplayStream(frameRate))
216 return false;
217
218 auto err = CGDisplayStreamStart(m_displayStream.get());
219 if (err) {
220 ERROR_LOG_IF(loggerPtr(), LOGIDENTIFIER, "CGDisplayStreamStart failed with error ", static_cast<int>(err));
221 return false;
222 }
223
224 m_isRunning = true;
225 return true;
226}

231 startDisplayStream(frameRate);

245void ScreenDisplayCapturerMac::newFrame(CGDisplayStreamFrameStatus status, DisplaySurface&& newFrame)
246{
247 switch (status) {
248 case kCGDisplayStreamFrameStatusFrameComplete:
249 break;
250
251 case kCGDisplayStreamFrameStatusFrameIdle:
252 break;
253
254 case kCGDisplayStreamFrameStatusFrameBlank:
255 RELEASE_LOG(WebRTC, "ScreenDisplayCapturerMac::frameAvailable: kCGDisplayStreamFrameStatusFrameBlank");
256 break;
257
258 case kCGDisplayStreamFrameStatusStopped:
259 RELEASE_LOG(WebRTC, "ScreenDisplayCapturerMac::frameAvailable: kCGDisplayStreamFrameStatusStopped");
260 break;
261 }
262
263 m_currentFrame = WTFMove(newFrame);
264}

266std::optional<CaptureDevice> ScreenDisplayCapturerMac::screenCaptureDeviceWithPersistentID(const String& deviceID)
267{
268 auto displayID = parseInteger<uint32_t>(deviceID);
269 if (!displayID) {
270 RELEASE_LOG(WebRTC, "ScreenDisplayCapturerMac::screenCaptureDeviceWithPersistentID: display ID does not convert to 32-bit integer");
271 return std::nullopt;
272 }
273
274 auto actualDisplayID = updateDisplayID(*displayID);
275 if (!actualDisplayID)
276 return std::nullopt;
277
278 auto device = CaptureDevice(String::number(*actualDisplayID), CaptureDevice::DeviceType::Screen, "ScreenCaptureDevice"_s);
279 device.setEnabled(true);
280 return device;
281}

283void ScreenDisplayCapturerMac::screenCaptureDevices(Vector<CaptureDevice>& displays)
284{
285 auto screenID = displayID([NSScreen mainScreen]);
286 if (CGDisplayIDToOpenGLDisplayMask(screenID)) {
287 CaptureDevice displayDevice(String::number(screenID), CaptureDevice::DeviceType::Screen, makeString("Screen 0"));
288 displayDevice.setEnabled(true);
289 displays.append(WTFMove(displayDevice));
290 return;
291 }
292
293 uint32_t displayCount = 0;
294 auto err = CGGetActiveDisplayList(0, nullptr, &displayCount);
295 if (err) {
296 RELEASE_LOG(WebRTC, "CGGetActiveDisplayList() returned error %d when trying to get display count", (int)err);
297 return;
298 }
299
300 if (!displayCount) {
301 RELEASE_LOG(WebRTC, "CGGetActiveDisplayList() returned a display count of 0");
302 return;
303 }
304
305 Vector<CGDirectDisplayID> activeDisplays(displayCount);
306 err = CGGetActiveDisplayList(displayCount, activeDisplays.data(), &displayCount);
307 if (err) {
308 RELEASE_LOG(WebRTC, "CGGetActiveDisplayList() returned error %d when trying to get the active display list", (int)err);
309 return;
310 }
311
312 int count = 0;
313 for (auto displayID : activeDisplays) {
314 CaptureDevice displayDevice(String::number(displayID), CaptureDevice::DeviceType::Screen, makeString("Screen ", String::number(count++)));
315 displayDevice.setEnabled(CGDisplayIDToOpenGLDisplayMask(displayID));
316 displays.append(WTFMove(displayDevice));
317 }
318}
