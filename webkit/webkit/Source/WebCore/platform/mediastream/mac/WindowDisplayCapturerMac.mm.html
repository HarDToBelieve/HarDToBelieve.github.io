50static bool anyOfCGWindow(const Function<bool(CFDictionaryRef info, unsigned id, const String& title)>& predicate)
51{
52 auto windows = adoptCF(CGWindowListCopyWindowInfo(kCGWindowListOptionOnScreenOnly | kCGWindowListExcludeDesktopElements, kCGNullWindowID));
53 if (!windows) {
54 RELEASE_LOG(WebRTC, "CGWindowListCopyWindowInfo returned NULL");
55 return false;
56 }
57
58 auto windowCount = CFArrayGetCount(windows.get());
59 for (auto i = 0; i < windowCount; i++) {
60 auto windowInfo = checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(windows.get(), i));
61 if (!windowInfo)
62 continue;
63
64 // Menus, the dock, etc have layers greater than 0, skip them.
65 auto windowLayerRef = checked_cf_cast<CFNumberRef>(CFDictionaryGetValue(windowInfo, kCGWindowLayer));
66 if (!windowLayerRef)
67 continue;
68 unsigned windowLayer;
69 CFNumberGetValue(windowLayerRef, kCFNumberIntType, &windowLayer);
70 if (windowLayer)
71 continue;
72
73 auto onScreen = checked_cf_cast<CFBooleanRef>(CFDictionaryGetValue(windowInfo, kCGWindowIsOnscreen));
74 if (!CFBooleanGetValue(onScreen))
75 continue;
76
77 auto windowIDRef = checked_cf_cast<CFNumberRef>(CFDictionaryGetValue(windowInfo, kCGWindowNumber));
78 if (!windowIDRef)
79 continue;
80 unsigned windowID;
81 CFNumberGetValue(windowIDRef, kCFNumberIntType, &windowID);
82 if (!windowID)
83 continue;
84
85 auto windowTitle = checked_cf_cast<CFStringRef>(CFDictionaryGetValue(windowInfo, kCGWindowName));
86
87 if (predicate(windowInfo, windowID, windowTitle))
88 return true;
89 }
90
91 return false;
92}

94static RetainPtr<CFDictionaryRef> windowDescription(CGWindowID id)
95{
96 auto ids = adoptCF(CFArrayCreate(nullptr, reinterpret_cast<const void**>(&id), 1, nullptr));
97 auto windows = adoptCF(CGWindowListCreateDescriptionFromArray(ids.get()));
98 if (!windows)
99 return nullptr;
100
101 return checked_cf_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(windows.get(), 0));
102}

104Expected<UniqueRef<DisplayCaptureSourceCocoa::Capturer>, String> WindowDisplayCapturerMac::create(const String& deviceID)
105{
106 auto displayID = parseInteger<uint32_t>(deviceID);
107 if (!displayID)
108 return makeUnexpected("Invalid window device ID"_s);
109
110 auto windowInfo = windowDescription(*displayID);
111 if (!windowInfo)
112 return makeUnexpected("Invalid window ID"_s);
113
114 return UniqueRef<DisplayCaptureSourceCocoa::Capturer>(makeUniqueRef<WindowDisplayCapturerMac>(*displayID));
115}

122RetainPtr<CGImageRef> WindowDisplayCapturerMac::windowImage()
123{
124 auto image = adoptCF(CGWindowListCreateImage(CGRectNull, kCGWindowListOptionIncludingWindow, m_windowID, kCGWindowImageBoundsIgnoreFraming | kCGWindowImageShouldBeOpaque));
125 if (!image)
126 RELEASE_LOG(WebRTC, "WindowDisplayCapturerMac::windowImage: failed to capture window image");
127
128 return image;
129}

136std::optional<CaptureDevice> WindowDisplayCapturerMac::windowCaptureDeviceWithPersistentID(const String& idString)
137{
138 auto windowID = parseInteger<uint32_t>(idString);
139 if (!windowID) {
140 RELEASE_LOG(WebRTC, "WindowDisplayCapturerMac::windowCaptureDeviceWithPersistentID: window ID does not convert to 32-bit integer");
141 return std::nullopt;
142 }
143
144 String windowTitle;
145 if (!anyOfCGWindow([&windowTitle, windowID] (CFDictionaryRef, CGWindowID id, const String& title) {
146 if (windowID != id)
147 return false;
148
149 windowTitle = title;
150 return true;
151
152 })) {
153 RELEASE_LOG(WebRTC, "WindowDisplayCapturerMac::windowCaptureDeviceWithPersistentID: window ID is not valid");
154 return std::nullopt;
155 }
156
157 auto device = CaptureDevice(String::number(*windowID), CaptureDevice::DeviceType::Window, windowTitle);
158 device.setEnabled(true);
159
160 return device;
161}

163void WindowDisplayCapturerMac::windowCaptureDevices(Vector<CaptureDevice>& windows)
164{
165 anyOfCGWindow([&] (CFDictionaryRef, int id, const String& title) mutable {
166 CaptureDevice device(String::number(id), CaptureDevice::DeviceType::Window, title);
167 device.setEnabled(true);
168 windows.append(WTFMove(device));
169 return false;
170 });
171}
