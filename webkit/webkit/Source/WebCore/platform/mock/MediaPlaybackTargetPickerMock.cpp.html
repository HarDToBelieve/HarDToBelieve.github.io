52bool MediaPlaybackTargetPickerMock::externalOutputDeviceAvailable()
53{
54 LOG(Media, "MediaPlaybackTargetPickerMock::externalOutputDeviceAvailable");
55 return m_state == MediaPlaybackTargetContext::MockState::OutputDeviceAvailable;

64void MediaPlaybackTargetPickerMock::showPlaybackTargetPicker(PlatformView*, const FloatRect&, bool checkActiveRoute, bool useDarkAppearance, bool)
65{
66 if (!client() || m_showingMenu)
67 return;
68
69#if LOG_DISABLED
70 UNUSED_PARAM(checkActiveRoute);
71 UNUSED_PARAM(useDarkAppearance);
72#endif
73
74 LOG(Media, "MediaPlaybackTargetPickerMock::showPlaybackTargetPicker - checkActiveRoute = %i, useDarkAppearance = %i", (int)checkActiveRoute, (int)useDarkAppearance);
75
76 m_showingMenu = true;
77 callOnMainThread([this, weakThis = makeWeakPtr(*this)] {
78 if (!weakThis)
79 return;
80
81 m_showingMenu = false;
82 currentDeviceDidChange();
83 });
84}
85
86void MediaPlaybackTargetPickerMock::startingMonitoringPlaybackTargets()
87{
88 LOG(Media, "MediaPlaybackTargetPickerMock::startingMonitoringPlaybackTargets");
89
90 callOnMainThread([this, weakThis = makeWeakPtr(*this)] {
91 if (!weakThis)
92 return;
93
94 if (m_state == MediaPlaybackTargetContext::MockState::OutputDeviceAvailable)
95 availableDevicesDidChange();
96
97 if (!m_deviceName.isEmpty() && m_state != MediaPlaybackTargetContext::MockState::Unknown)
98 currentDeviceDidChange();
99 });
100}

105}

107void MediaPlaybackTargetPickerMock::invalidatePlaybackTargets()
108{
109 LOG(Media, "MediaPlaybackTargetPickerMock::invalidatePlaybackTargets");
110 setState(emptyString(), MediaPlaybackTargetContext::MockState::Unknown);
111}
112
113void MediaPlaybackTargetPickerMock::setState(const String& deviceName, MediaPlaybackTargetContext::MockState state)
114{
115 LOG(Media, "MediaPlaybackTargetPickerMock::setState - name = %s, state = 0x%x", deviceName.utf8().data(), (unsigned)state);
116
117 callOnMainThread([this, weakThis = makeWeakPtr(*this), state, deviceName] {
118 if (!weakThis)
119 return;
120
121 if (deviceName != m_deviceName && state != MediaPlaybackTargetContext::MockState::Unknown) {
122 m_deviceName = deviceName;
123 currentDeviceDidChange();
124 }
125
126 if (m_state != state) {
127 m_state = state;
128 availableDevicesDidChange();
129 }
130 });
131}

133void MediaPlaybackTargetPickerMock::dismissPopup()
134{
135 if (!m_showingMenu)
136 return;
137
138 m_showingMenu = false;
139 currentDeviceDidChange();
140}
