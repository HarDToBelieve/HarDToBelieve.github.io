27
28#if ENABLE(WEB_AUDIO) && PLATFORM(COCOA)
29
30#include "AudioDestinationCocoa.h"
31#include <wtf/RunLoop.h>
32#include <wtf/WorkQueue.h>
33
34namespace WebCore {
35
36class AudioIOCallback;
37
38class MockAudioDestinationCocoa final : public AudioDestinationCocoa {
39 WTF_MAKE_FAST_ALLOCATED;
40public:
41 static Ref<AudioDestination> create(AudioIOCallback& callback, float sampleRate)
42 {
43 return adoptRef(*new MockAudioDestinationCocoa(callback, sampleRate));
44 }
45
46 WEBCORE_EXPORT MockAudioDestinationCocoa(AudioIOCallback&, float sampleRate);
47 WEBCORE_EXPORT virtual ~MockAudioDestinationCocoa();
48
49private:
50 void startRendering(CompletionHandler<void(bool)>&&) final;
51 void stopRendering(CompletionHandler<void(bool)>&&) final;
52
53 void tick();
54
55 Ref<WorkQueue> m_workQueue;
56 RunLoop::Timer<MockAudioDestinationCocoa> m_timer;
57 uint32_t m_numberOfFramesToProcess { 384 };
58};
59
60} // namespace WebCore
61
62#endif // ENABLE(WEB_AUDIO) && PLATFORM(COCOA)
