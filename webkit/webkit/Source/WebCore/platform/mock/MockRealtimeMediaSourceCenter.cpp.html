98class MockRealtimeVideoSourceFactory : public VideoCaptureFactory {
99public:
100 CaptureSourceOrError createVideoCaptureSource(const CaptureDevice& device, String&& hashSalt, const MediaConstraints* constraints) final
101 {
102 ASSERT(device.type() == CaptureDevice::DeviceType::Camera);
103 if (!MockRealtimeMediaSourceCenter::captureDeviceWithPersistentID(CaptureDevice::DeviceType::Camera, device.persistentId()))
104 return { "Unable to find mock camera device with given persistentID"_s };
105
106 return MockRealtimeVideoSource::create(String { device.persistentId() }, String { device.label() }, WTFMove(hashSalt), constraints);
107 }
108
109private:
110 CaptureDeviceManager& videoCaptureDeviceManager() final { return MockRealtimeMediaSourceCenter::singleton().videoCaptureDeviceManager(); }
111};

151class MockRealtimeDisplaySourceFactory : public DisplayCaptureFactory {
152public:
153 CaptureSourceOrError createDisplayCaptureSource(const CaptureDevice& device, const MediaConstraints* constraints) final
154 {
155 if (!MockRealtimeMediaSourceCenter::captureDeviceWithPersistentID(device.type(), device.persistentId()))
156 return { "Unable to find mock display device with given persistentID"_s };
157
158 switch (device.type()) {
159 case CaptureDevice::DeviceType::Screen:
160 case CaptureDevice::DeviceType::Window:
161#if PLATFORM(MAC)
162 return DisplayCaptureSourceCocoa::create(UniqueRef<DisplayCaptureSourceCocoa::Capturer>(makeUniqueRef<MockDisplayCapturer>(device)), device, constraints);
163#else
164 return MockRealtimeVideoSource::create(String { device.persistentId() }, String { device.label() }, String { }, constraints);
165#endif
166 break;
167 case CaptureDevice::DeviceType::Microphone:
168 case CaptureDevice::DeviceType::Speaker:
169 case CaptureDevice::DeviceType::Camera:
170 case CaptureDevice::DeviceType::Unknown:
171 ASSERT_NOT_REACHED();
172 break;
173 }
174
175 return { };
176 }
177private:
178 CaptureDeviceManager& displayCaptureDeviceManager() final { return MockRealtimeMediaSourceCenter::singleton().displayCaptureDeviceManager(); }
179};

209static HashMap<String, MockMediaDevice>& deviceMap()
210{
211 static auto map = makeNeverDestroyed([] {
212 HashMap<String, MockMediaDevice> map;

216 return map;

221static inline Vector<CaptureDevice>& deviceListForDevice(const MockMediaDevice& device)
222{
223 if (device.isMicrophone())
224 return MockRealtimeMediaSourceCenter::microphoneDevices();
225 if (device.isSpeaker())
226 return MockRealtimeMediaSourceCenter::speakerDevices();
227 if (device.isCamera())
228 return MockRealtimeMediaSourceCenter::videoDevices();
229
230 ASSERT(device.isDisplay());
231 return MockRealtimeMediaSourceCenter::displayDevices();
232}

240void MockRealtimeMediaSourceCenter::setMockRealtimeMediaSourceCenterEnabled(bool enabled)
241{
242 MockRealtimeMediaSourceCenter& mock = singleton();
243
244 if (mock.m_isEnabled == enabled)
245 return;
246
247 mock.m_isEnabled = enabled;
248 RealtimeMediaSourceCenter& center = RealtimeMediaSourceCenter::singleton();
249
250 if (mock.m_isEnabled) {
251 if (mock.m_isMockAudioCaptureEnabled)
252 center.setAudioCaptureFactory(mock.audioCaptureFactory());
253 if (mock.m_isMockVideoCaptureEnabled)
254 center.setVideoCaptureFactory(mock.videoCaptureFactory());
255 if (mock.m_isMockDisplayCaptureEnabled)
256 center.setDisplayCaptureFactory(mock.displayCaptureFactory());
257 return;
258 }
259
260 if (mock.m_isMockAudioCaptureEnabled)
261 center.unsetAudioCaptureFactory(mock.audioCaptureFactory());
262 if (mock.m_isMockVideoCaptureEnabled)
263 center.unsetVideoCaptureFactory(mock.videoCaptureFactory());
264 if (mock.m_isMockDisplayCaptureEnabled)
265 center.unsetDisplayCaptureFactory(mock.displayCaptureFactory());
266}

273static CaptureDevice toCaptureDevice(const MockMediaDevice& device)
274{
275 auto captureDevice = device.captureDevice();
276 captureDevice.setEnabled(true);
277 captureDevice.setIsMockDevice(true);
278 return captureDevice;

281static void createMockDevice(const MockMediaDevice& device)
282{
283 deviceListForDevice(device).append(toCaptureDevice(device));

292void MockRealtimeMediaSourceCenter::setDevices(Vector<MockMediaDevice>&& newMockDevices)
293{
294 microphoneDevices().clear();
295 speakerDevices().clear();
296 videoDevices().clear();
297 displayDevices().clear();
298
299 auto& mockDevices = devices();
300 mockDevices = WTFMove(newMockDevices);
301
302 auto& map = deviceMap();
303 map.clear();
304
305 for (const auto& device : mockDevices) {
306 map.add(device.persistentId, device);
307 createMockDevice(device);
308 }
309 RealtimeMediaSourceCenter::singleton().captureDevicesChanged();
310}
311
312void MockRealtimeMediaSourceCenter::addDevice(const MockMediaDevice& device)
313{
314 devices().append(device);
315 deviceMap().set(device.persistentId, device);
316 createMockDevice(device);
317 RealtimeMediaSourceCenter::singleton().captureDevicesChanged();
318}

320void MockRealtimeMediaSourceCenter::removeDevice(const String& persistentId)
321{
322 auto& map = deviceMap();
323 auto iterator = map.find(persistentId);
324 if (iterator == map.end())
325 return;
326
327 devices().removeFirstMatching([&persistentId](const auto& device) {
328 return device.persistentId == persistentId;
329 });
330
331 deviceListForDevice(iterator->value).removeFirstMatching([&persistentId](const auto& device) {
332 return device.persistentId() == persistentId;
333 });
334
335 map.remove(iterator);
336 RealtimeMediaSourceCenter::singleton().captureDevicesChanged();
337}

339std::optional<MockMediaDevice> MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(const String& id)
340{
341 ASSERT(!id.isEmpty());
342
343 auto& map = deviceMap();
344 auto iterator = map.find(id);
345 if (iterator == map.end())
346 return std::nullopt;
347
348 return iterator->value;
349}
350
351std::optional<CaptureDevice> MockRealtimeMediaSourceCenter::captureDeviceWithPersistentID(CaptureDevice::DeviceType type, const String& id)
352{
353 ASSERT(!id.isEmpty());
354
355 auto& map = deviceMap();
356 auto iterator = map.find(id);
357 if (iterator == map.end() || iterator->value.type() != type)
358 return std::nullopt;
359
360 return toCaptureDevice(iterator->value);
361}
