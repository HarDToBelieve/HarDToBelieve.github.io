36namespace WebCore {
37
38ScrollAnimatorMock::ScrollAnimatorMock(ScrollableArea& scrollableArea, WTF::Function<void(const String&)>&& logger)
39 : ScrollAnimator(scrollableArea)
40 , m_logger(WTFMove(logger))
41{
42}
43
44ScrollAnimatorMock::~ScrollAnimatorMock() = default;
45
46void ScrollAnimatorMock::didAddVerticalScrollbar(Scrollbar* scrollbar)
47{
48 m_verticalScrollbar = scrollbar;
49 ScrollAnimator::didAddVerticalScrollbar(scrollbar);
50}
51
52void ScrollAnimatorMock::didAddHorizontalScrollbar(Scrollbar* scrollbar)
53{
54 m_horizontalScrollbar = scrollbar;
55 ScrollAnimator::didAddHorizontalScrollbar(scrollbar);
56}
57
58void ScrollAnimatorMock::willRemoveVerticalScrollbar(Scrollbar* scrollbar)
59{
60 ScrollAnimator::willRemoveVerticalScrollbar(scrollbar);
61 m_verticalScrollbar = nullptr;
62}
63
64void ScrollAnimatorMock::willRemoveHorizontalScrollbar(Scrollbar* scrollbar)
65{
66 ScrollAnimator::willRemoveHorizontalScrollbar(scrollbar);
67 m_horizontalScrollbar = nullptr;
68}
69
70void ScrollAnimatorMock::mouseEnteredContentArea()
71{
72 m_logger("mouseEnteredContentArea");
73 ScrollAnimator::mouseEnteredContentArea();
74}
75
76void ScrollAnimatorMock::mouseMovedInContentArea()
77{
78 m_logger("mouseMovedInContentArea");
79 ScrollAnimator::mouseMovedInContentArea();
80}
81
82void ScrollAnimatorMock::mouseExitedContentArea()
83{
84 m_logger("mouseExitedContentArea");
85 ScrollAnimator::mouseExitedContentArea();
86}
87
88const char* ScrollAnimatorMock::scrollbarPrefix(Scrollbar* scrollbar) const
89{
90 return scrollbar == m_verticalScrollbar ? "Vertical" : scrollbar == m_horizontalScrollbar ? "Horizontal" : "Unknown";
91}
92
93void ScrollAnimatorMock::mouseEnteredScrollbar(Scrollbar* scrollbar) const
94{
95 m_logger(makeString("mouseEntered", scrollbarPrefix(scrollbar), "Scrollbar"));
96 ScrollAnimator::mouseEnteredScrollbar(scrollbar);
97}
98
99void ScrollAnimatorMock::mouseExitedScrollbar(Scrollbar* scrollbar) const
100{
101 m_logger(makeString("mouseExited", scrollbarPrefix(scrollbar), "Scrollbar"));
102 ScrollAnimator::mouseExitedScrollbar(scrollbar);
103}
104
105void ScrollAnimatorMock::mouseIsDownInScrollbar(Scrollbar* scrollbar, bool isPressed) const
106{
107 m_logger(makeString(isPressed ? "mouseIsDownIn" : "mouseIsUpIn", scrollbarPrefix(scrollbar), "Scrollbar"));
108 ScrollAnimator::mouseIsDownInScrollbar(scrollbar, isPressed);
109}
