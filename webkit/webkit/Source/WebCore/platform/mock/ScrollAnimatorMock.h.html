35namespace WebCore {
36
37// A Mock implementation of ScrollAnimator used to test the scroll events
38// received by the scroll animator. Tests can enable this scroll animator using
39// the internal setting enableMockScrollAnimator.
40class ScrollAnimatorMock final : public ScrollAnimator {
41public:
42 ScrollAnimatorMock(ScrollableArea&, WTF::Function<void(const String&)>&&);
43 virtual ~ScrollAnimatorMock();
44
45private:
46
47#if ENABLE(RUBBER_BANDING)
48 bool allowsHorizontalStretching(const PlatformWheelEvent&) const override { return false; }
49 bool allowsVerticalStretching(const PlatformWheelEvent&) const override { return false; }
50 IntSize stretchAmount() const override { return IntSize(); }
51 bool isPinnedForScrollDelta(const FloatSize&) const override { return false; }
52 RectEdges<bool> edgePinnedState() const override { return { }; }
53 bool allowsHorizontalScrolling() const override { return false; }
54 bool allowsVerticalScrolling() const override { return false; }
55 bool shouldRubberBandInDirection(ScrollDirection) const override { return false; }
56 void immediateScrollBy(const FloatSize&) override { }
57 void immediateScrollByWithoutContentEdgeConstraints(const FloatSize&) override { }
58 void adjustScrollPositionToBoundsIfNecessary() override { }
59#endif
60
61 void didAddVerticalScrollbar(Scrollbar*) override;
62 void didAddHorizontalScrollbar(Scrollbar*) override;
63 void willRemoveVerticalScrollbar(Scrollbar*) override;
64 void willRemoveHorizontalScrollbar(Scrollbar*) override;
65 void mouseEnteredContentArea() override;
66 void mouseMovedInContentArea() override;
67 void mouseExitedContentArea() override;
68 void mouseEnteredScrollbar(Scrollbar*) const override;
69 void mouseExitedScrollbar(Scrollbar*) const override;
70 void mouseIsDownInScrollbar(Scrollbar*, bool) const override;
71 const char* scrollbarPrefix(Scrollbar*) const;
72
73 WTF::Function<void(const String&)> m_logger;
74 Scrollbar* m_verticalScrollbar { nullptr };
75 Scrollbar* m_horizontalScrollbar { nullptr };
76};
77
78} // namespace WebCore
