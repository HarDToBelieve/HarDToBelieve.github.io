26#ifndef ScrollbarThemeMock_h
27#define ScrollbarThemeMock_h
28
29#include "ScrollbarThemeComposite.h"
30
31namespace WebCore {
32
33// Scrollbar theme used in image snapshots, to eliminate appearance differences between platforms.
34class ScrollbarThemeMock : public ScrollbarThemeComposite {
35public:
36 int scrollbarThickness(ScrollbarControlSize = ScrollbarControlSize::Regular, ScrollbarExpansionState = ScrollbarExpansionState::Expanded) override;
37
38protected:
39 bool hasButtons(Scrollbar&) override { return false; }
40 bool hasThumb(Scrollbar&) override { return true; }
41
42 IntRect backButtonRect(Scrollbar&, ScrollbarPart, bool /*painting*/ = false) override { return IntRect(); }
43 IntRect forwardButtonRect(Scrollbar&, ScrollbarPart, bool /*painting*/ = false) override { return IntRect(); }
44 IntRect trackRect(Scrollbar&, bool painting = false) override;
45
46 void paintTrackBackground(GraphicsContext&, Scrollbar&, const IntRect&) override;
47 void paintThumb(GraphicsContext&, Scrollbar&, const IntRect&) override;
48 int maxOverlapBetweenPages() override { return 40; }
49
50 bool usesOverlayScrollbars() const override;
51private:
52 bool isMockTheme() const override { return true; }
53};
54
55}
56#endif // ScrollbarThemeMock_h
