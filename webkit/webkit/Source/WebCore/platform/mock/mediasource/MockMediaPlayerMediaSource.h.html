26#ifndef MockMediaPlayerMediaSource_h
27#define MockMediaPlayerMediaSource_h
28
29#if ENABLE(MEDIA_SOURCE)
30
31#include "MediaPlayerPrivate.h"
32#include <wtf/Logger.h>
33#include <wtf/MediaTime.h>
34
35namespace WebCore {
36
37class MediaSource;
38class MockMediaSourcePrivate;
39
40class MockMediaPlayerMediaSource : public MediaPlayerPrivateInterface {
41public:
42 explicit MockMediaPlayerMediaSource(MediaPlayer*);
43
44 // MediaPlayer Engine Support
45 WEBCORE_EXPORT static void registerMediaEngine(MediaEngineRegistrar);
46 static void getSupportedTypes(HashSet<String, ASCIICaseInsensitiveHash>& types);
47 static MediaPlayer::SupportsType supportsType(const MediaEngineSupportParameters&);
48
49 virtual ~MockMediaPlayerMediaSource();
50
51 void advanceCurrentTime();
52 MediaTime currentMediaTime() const override;
53 void updateDuration(const MediaTime&);
54
55 MediaPlayer::ReadyState readyState() const override;
56 void setReadyState(MediaPlayer::ReadyState);
57 void setNetworkState(MediaPlayer::NetworkState);
58 void waitForSeekCompleted();
59 void seekCompleted();
60
61#if !RELEASE_LOG_DISABLED
62 const void* mediaPlayerLogIdentifier() { return m_player->mediaPlayerLogIdentifier(); }
63 const Logger& mediaPlayerLogger() { return m_player->mediaPlayerLogger(); }
64#endif
65
66private:
67 // MediaPlayerPrivate Overrides
68 void load(const String& url) override;
69 void load(const URL&, const ContentType&, MediaSourcePrivateClient*) override;
70#if ENABLE(MEDIA_STREAM)
71 void load(MediaStreamPrivate&) override { }
72#endif
73 void cancelLoad() override;
74 void play() override;
75 void pause() override;
76 FloatSize naturalSize() const override;
77 bool hasVideo() const override;
78 bool hasAudio() const override;
79 void setVisible(bool) override;
80 bool seeking() const override;
81 bool paused() const override;
82 MediaPlayer::NetworkState networkState() const override;
83 MediaTime maxMediaTimeSeekable() const override;
84 std::unique_ptr<PlatformTimeRanges> buffered() const override;
85 bool didLoadingProgress() const override;
86 void setSize(const IntSize&) override;
87 void paint(GraphicsContext&, const FloatRect&) override;
88 MediaTime durationMediaTime() const override;
89 void seekWithTolerance(const MediaTime&, const MediaTime&, const MediaTime&) override;
90 std::optional<VideoPlaybackQualityMetrics> videoPlaybackQualityMetrics() override;
91
92 MediaPlayer* m_player;
93 RefPtr<MockMediaSourcePrivate> m_mediaSourcePrivate;
94
95 MediaTime m_currentTime;
96 MediaTime m_duration;
97 MediaPlayer::ReadyState m_readyState;
98 MediaPlayer::NetworkState m_networkState;
99 bool m_playing;
100 bool m_seekCompleted;
101};
102
103}

105#endif // ENABLE(MEDIA_SOURCE)

107#endif // MockMediaPlayerMediaSource_h
