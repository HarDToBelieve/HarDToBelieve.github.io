85bool AuthenticationChallengeBase::compare(const AuthenticationChallenge& a, const AuthenticationChallenge& b)
86{
87 if (a.isNull() && b.isNull())
88 return true;
89
90 if (a.isNull() || b.isNull())
91 return false;
92
93 if (a.protectionSpace() != b.protectionSpace())
94 return false;
95
96 if (a.proposedCredential() != b.proposedCredential())
97 return false;
98
99 if (a.previousFailureCount() != b.previousFailureCount())
100 return false;
101
102 if (a.failureResponse() != b.failureResponse())
103 return false;
104
105 if (a.error() != b.error())
106 return false;
107
108 return AuthenticationChallenge::platformCompare(a, b);
109}
