25#ifndef AuthenticationChallengeBase_h
26#define AuthenticationChallengeBase_h
27
28#include "Credential.h"
29#include "ProtectionSpace.h"
30#include "ResourceResponse.h"
31#include "ResourceError.h"
32
33namespace WebCore {
34
35class AuthenticationChallenge;
36
37class AuthenticationChallengeBase {
38public:
39 WEBCORE_EXPORT AuthenticationChallengeBase();
40 WEBCORE_EXPORT AuthenticationChallengeBase(const ProtectionSpace& protectionSpace, const Credential& proposedCredential, unsigned previousFailureCount, const ResourceResponse& response, const ResourceError& error);
41
42 WEBCORE_EXPORT unsigned previousFailureCount() const;
43 WEBCORE_EXPORT const Credential& proposedCredential() const;
44 WEBCORE_EXPORT const ProtectionSpace& protectionSpace() const;
45 WEBCORE_EXPORT const ResourceResponse& failureResponse() const;
46 WEBCORE_EXPORT const ResourceError& error() const;
47
48 WEBCORE_EXPORT bool isNull() const;
49 WEBCORE_EXPORT void nullify();
50
51 WEBCORE_EXPORT static bool compare(const AuthenticationChallenge& a, const AuthenticationChallenge& b);
52
53protected:
54 // The AuthenticationChallenge subclass may "shadow" this method to compare platform specific fields
55 static bool platformCompare(const AuthenticationChallengeBase&, const AuthenticationChallengeBase&) { return true; }
56
57 bool m_isNull;
58 ProtectionSpace m_protectionSpace;
59 Credential m_proposedCredential;
60 unsigned m_previousFailureCount;
61 ResourceResponse m_failureResponse;
62 ResourceError m_error;
63};
64
65inline bool operator==(const AuthenticationChallenge& a, const AuthenticationChallenge& b) { return AuthenticationChallengeBase::compare(a, b); }
66inline bool operator!=(const AuthenticationChallenge& a, const AuthenticationChallenge& b) { return !(a == b); }
67
68}

70#endif
