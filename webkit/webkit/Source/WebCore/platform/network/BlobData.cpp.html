39BlobData::BlobData(const String& contentType)
40 : m_contentType(contentType)
41{
42 ASSERT(Blob::isNormalizedContentType(contentType));
43}
44
45const long long BlobDataItem::toEndOfFile = -1;
46
47long long BlobDataItem::length() const
48{
49 if (m_length != toEndOfFile)
50 return m_length;
51
52 switch (m_type) {
53 case Type::Data:
54 ASSERT_NOT_REACHED();
55 return m_length;
56 case Type::File:
57 return m_file->size();
58 }
59
60 ASSERT_NOT_REACHED();
61 return m_length;
62}
63
64void BlobData::appendData(const ThreadSafeDataBuffer& data)
65{
66 size_t dataSize = data.data() ? data.data()->size() : 0;
67 appendData(data, 0, dataSize);
68}
69
70void BlobData::appendData(const ThreadSafeDataBuffer& data, long long offset, long long length)
71{
72 m_items.append(BlobDataItem(data, offset, length));
73}
74
75void BlobData::appendFile(Ref<BlobDataFileReference>&& file)
76{
77 file->startTrackingModifications();
78 m_items.append(BlobDataItem(WTFMove(file)));
79}
80
81void BlobData::appendFile(BlobDataFileReference* file, long long offset, long long length)
82{
83 m_items.append(BlobDataItem(file, offset, length));
84}

86void BlobData::swapItems(BlobDataItemList& items)
87{
88 m_items.swap(items);
89}
