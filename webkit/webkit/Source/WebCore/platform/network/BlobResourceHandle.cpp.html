113void BlobResourceSynchronousLoader::didReceiveResponseAsync(ResourceHandle* handle, ResourceResponse&& response, CompletionHandler<void()>&& completionHandler)
114{
115 // We cannot handle the size that is more than maximum integer.
116 if (response.expectedContentLength() > INT_MAX) {
117 m_error = ResourceError(webKitBlobResourceDomain, static_cast<int>(BlobResourceHandle::Error::NotReadableError), response.url(), "File is too large");
118 completionHandler();
119 return;
120 }
121
122 m_response = response;
123
124 // Read all the data.
125 m_data.resize(static_cast<size_t>(response.expectedContentLength()));
126 static_cast<BlobResourceHandle*>(handle)->readSync(m_data.data(), static_cast<int>(m_data.size()));
127 completionHandler();
128}

130void BlobResourceSynchronousLoader::didFail(ResourceHandle*, const ResourceError& error)
131{
132 m_error = error;
133}

145void BlobResourceHandle::loadResourceSynchronously(BlobData* blobData, const ResourceRequest& request, ResourceError& error, ResourceResponse& response, Vector<uint8_t>& data)
146{
147 if (!equalLettersIgnoringASCIICase(request.httpMethod(), "get")) {
148 error = ResourceError(webKitBlobResourceDomain, static_cast<int>(Error::MethodNotAllowed), response.url(), "Request method must be GET");
149 return;
150 }
151
152 BlobResourceSynchronousLoader loader(error, response, data);
153 RefPtr<BlobResourceHandle> handle = adoptRef(new BlobResourceHandle(blobData, request, &loader, false));
154 handle->start();
155}

180void BlobResourceHandle::start()
181{
182 if (!m_async) {
183 doStart();
184 return;
185 }
186
187 // Finish this async call quickly and return.
188 callOnMainThread([protectedThis = makeRef(*this)]() mutable {
189 protectedThis->doStart();
190 });
191}
192
193void BlobResourceHandle::doStart()
194{
195 ASSERT(isMainThread());
196
197 // Do not continue if the request is aborted or an error occurs.
198 if (erroredOrAborted())
199 return;
200
201 if (!equalLettersIgnoringASCIICase(firstRequest().httpMethod(), "get")) {
202 notifyFail(Error::MethodNotAllowed);
203 return;
204 }
205
206 // If the blob data is not found, fail now.
207 if (!m_blobData) {
208 notifyFail(Error::NotFoundError);
209 return;
210 }
211
212 // Parse the "Range" header we care about.
213 String range = firstRequest().httpHeaderField(HTTPHeaderName::Range);
214 if (!range.isEmpty() && !parseRange(range, m_rangeOffset, m_rangeEnd, m_rangeSuffixLength)) {
215 m_errorCode = Error::RangeError;
216 notifyResponse();
217 return;
218 }
219
220 if (m_async)
221 getSizeForNext();
222 else {
223 Ref<BlobResourceHandle> protectedThis(*this); // getSizeForNext calls the client
224 for (size_t i = 0; i < m_blobData->items().size() && !erroredOrAborted(); ++i)
225 getSizeForNext();
226 notifyResponse();
227 }
228}
229
230void BlobResourceHandle::getSizeForNext()
231{
232 ASSERT(isMainThread());
233
234 // Do we finish validating and counting size for all items?
235 if (m_sizeItemCount >= m_blobData->items().size()) {
236 seek();
237
238 // Start reading if in asynchronous mode.
239 if (m_async) {
240 Ref<BlobResourceHandle> protectedThis(*this);
241 notifyResponse();
242 }
243 return;
244 }
245
246 const BlobDataItem& item = m_blobData->items().at(m_sizeItemCount);
247 switch (item.type()) {
248 case BlobDataItem::Type::Data:
249 didGetSize(item.length());
250 break;
251 case BlobDataItem::Type::File:
252 // Files know their sizes, but asking the stream to verify that the file wasn't modified.
253 if (m_async)
254 m_asyncStream->getSize(item.file()->path(), item.file()->expectedModificationTime());
255 else
256 didGetSize(m_stream->getSize(item.file()->path(), item.file()->expectedModificationTime()));
257 break;
258 default:
259 ASSERT_NOT_REACHED();
260 }
261}
262
263void BlobResourceHandle::didGetSize(long long size)
264{
265 ASSERT(isMainThread());
266
267 // Do not continue if the request is aborted or an error occurs.
268 if (erroredOrAborted())
269 return;
270
271 // If the size is -1, it means the file has been moved or changed. Fail now.
272 if (size == -1) {
273 notifyFail(Error::NotFoundError);
274 return;
275 }
276
277 // The size passed back is the size of the whole file. If the underlying item is a sliced file, we need to use the slice length.
278 const BlobDataItem& item = m_blobData->items().at(m_sizeItemCount);
279 size = item.length();
280
281 // Cache the size.
282 m_itemLengthList.append(size);
283
284 // Count the size.
285 m_totalSize += size;
286 m_totalRemainingSize += size;
287 m_sizeItemCount++;
288
289 // Continue with the next item.
290 getSizeForNext();
291}

293void BlobResourceHandle::seek()
294{
295 ASSERT(isMainThread());
296
297 // Convert from the suffix length to the range.
298 if (m_rangeSuffixLength != kPositionNotSpecified) {
299 m_rangeOffset = m_totalRemainingSize - m_rangeSuffixLength;
300 m_rangeEnd = m_rangeOffset + m_rangeSuffixLength - 1;
301 }
302
303 // Bail out if the range is not provided.
304 if (m_rangeOffset == kPositionNotSpecified)
305 return;
306
307 // Skip the initial items that are not in the range.
308 long long offset = m_rangeOffset;
309 for (m_readItemCount = 0; m_readItemCount < m_blobData->items().size() && offset >= m_itemLengthList[m_readItemCount]; ++m_readItemCount)
310 offset -= m_itemLengthList[m_readItemCount];
311
312 // Set the offset that need to jump to for the first item in the range.
313 m_currentItemReadSize = offset;
314
315 // Adjust the total remaining size in order not to go beyond the range.
316 if (m_rangeEnd != kPositionNotSpecified) {
317 long long rangeSize = m_rangeEnd - m_rangeOffset + 1;
318 if (m_totalRemainingSize > rangeSize)
319 m_totalRemainingSize = rangeSize;
320 } else
321 m_totalRemainingSize -= m_rangeOffset;
322}

324int BlobResourceHandle::readSync(uint8_t* buf, int length)
325{
326 ASSERT(isMainThread());
327
328 ASSERT(!m_async);
329 Ref<BlobResourceHandle> protectedThis(*this);
330
331 int offset = 0;
332 int remaining = length;
333 while (remaining) {
334 // Do not continue if the request is aborted or an error occurs.
335 if (erroredOrAborted())
336 break;
337
338 // If there is no more remaining data to read, we are done.
339 if (!m_totalRemainingSize || m_readItemCount >= m_blobData->items().size())
340 break;
341
342 const BlobDataItem& item = m_blobData->items().at(m_readItemCount);
343 int bytesRead = 0;
344 if (item.type() == BlobDataItem::Type::Data)
345 bytesRead = readDataSync(item, buf + offset, remaining);
346 else if (item.type() == BlobDataItem::Type::File)
347 bytesRead = readFileSync(item, buf + offset, remaining);
348 else
349 ASSERT_NOT_REACHED();
350
351 if (bytesRead > 0) {
352 offset += bytesRead;
353 remaining -= bytesRead;
354 }
355 }
356
357 int result;
358 if (erroredOrAborted())
359 result = -1;
360 else
361 result = length - remaining;
362
363 if (result > 0)
364 notifyReceiveData(buf, result);
365
366 if (!result)
367 notifyFinish();
368
369 return result;
370}

372int BlobResourceHandle::readDataSync(const BlobDataItem& item, void* buf, int length)
373{
374 ASSERT(isMainThread());
375
376 ASSERT(!m_async);
377
378 long long remaining = item.length() - m_currentItemReadSize;
379 int bytesToRead = (length > remaining) ? static_cast<int>(remaining) : length;
380 if (bytesToRead > m_totalRemainingSize)
381 bytesToRead = static_cast<int>(m_totalRemainingSize);
382 memcpy(buf, item.data().data() + item.offset() + m_currentItemReadSize, bytesToRead);
383 m_totalRemainingSize -= bytesToRead;
384
385 m_currentItemReadSize += bytesToRead;
386 if (m_currentItemReadSize == item.length()) {
387 m_readItemCount++;
388 m_currentItemReadSize = 0;
389 }
390
391 return bytesToRead;
392}

394int BlobResourceHandle::readFileSync(const BlobDataItem& item, void* buf, int length)
395{
396 ASSERT(isMainThread());
397
398 ASSERT(!m_async);
399
400 if (!m_fileOpened) {
401 long long bytesToRead = m_itemLengthList[m_readItemCount] - m_currentItemReadSize;
402 if (bytesToRead > m_totalRemainingSize)
403 bytesToRead = m_totalRemainingSize;
404 bool success = m_stream->openForRead(item.file()->path(), item.offset() + m_currentItemReadSize, bytesToRead);
405 m_currentItemReadSize = 0;
406 if (!success) {
407 m_errorCode = Error::NotReadableError;
408 return 0;
409 }
410
411 m_fileOpened = true;
412 }
413
414 int bytesRead = m_stream->read(buf, length);
415 if (bytesRead < 0) {
416 m_errorCode = Error::NotReadableError;
417 return 0;
418 }
419 if (!bytesRead) {
420 m_stream->close();
421 m_fileOpened = false;
422 m_readItemCount++;
423 } else
424 m_totalRemainingSize -= bytesRead;
425
426 return bytesRead;
427}

429void BlobResourceHandle::readAsync()
430{
431 ASSERT(isMainThread());
432
433 // Do not continue if the request is aborted or an error occurs.
434 if (erroredOrAborted())
435 return;
436
437 // If there is no more remaining data to read, we are done.
438 if (!m_totalRemainingSize || m_readItemCount >= m_blobData->items().size()) {
439 notifyFinish();
440 return;
441 }
442
443 const BlobDataItem& item = m_blobData->items().at(m_readItemCount);
444 if (item.type() == BlobDataItem::Type::Data)
445 readDataAsync(item);
446 else if (item.type() == BlobDataItem::Type::File)
447 readFileAsync(item);
448 else
449 ASSERT_NOT_REACHED();
450}
451
452void BlobResourceHandle::readDataAsync(const BlobDataItem& item)
453{
454 ASSERT(isMainThread());
455 ASSERT(item.data().data());
456
457 Ref<BlobResourceHandle> protectedThis(*this);
458
459 long long bytesToRead = item.length() - m_currentItemReadSize;
460 ASSERT(bytesToRead >= 0);
461 if (bytesToRead > m_totalRemainingSize)
462 bytesToRead = m_totalRemainingSize;
463
464 auto* data = item.data().data()->data() + item.offset() + m_currentItemReadSize;
465 m_currentItemReadSize = 0;
466
467 consumeData(data, static_cast<int>(bytesToRead));
468}
469
470void BlobResourceHandle::readFileAsync(const BlobDataItem& item)
471{
472 ASSERT(isMainThread());
473
474 if (m_fileOpened) {
475 m_asyncStream->read(m_buffer.data(), m_buffer.size());
476 return;
477 }
478
479 long long bytesToRead = m_itemLengthList[m_readItemCount] - m_currentItemReadSize;
480 if (bytesToRead > m_totalRemainingSize)
481 bytesToRead = static_cast<int>(m_totalRemainingSize);
482 m_asyncStream->openForRead(item.file()->path(), item.offset() + m_currentItemReadSize, bytesToRead);
483 m_fileOpened = true;
484 m_currentItemReadSize = 0;
485}

510void BlobResourceHandle::consumeData(const uint8_t* data, int bytesRead)
511{
512 ASSERT(m_async);
513 Ref<BlobResourceHandle> protectedThis(*this);
514
515 m_totalRemainingSize -= bytesRead;
516
517 // Notify the client.
518 if (bytesRead)
519 notifyReceiveData(data, bytesRead);
520
521 if (m_fileOpened) {
522 // When the current item is a file item, the reading is completed only if bytesRead is 0.
523 if (!bytesRead) {
524 // Close the file.
525 m_fileOpened = false;
526 m_asyncStream->close();
527
528 // Move to the next item.
529 m_readItemCount++;
530 }
531 } else {
532 // Otherwise, we read the current text item as a whole and move to the next item.
533 m_readItemCount++;
534 }
535
536 // Continue the reading.
537 readAsync();
538}

540void BlobResourceHandle::failed(Error errorCode)
541{
542 ASSERT(m_async);
543 Ref<BlobResourceHandle> protectedThis(*this);
544
545 // Notify the client.
546 notifyFail(errorCode);
547
548 // Close the file if needed.
549 if (m_fileOpened) {
550 m_fileOpened = false;
551 m_asyncStream->close();
552 }
553}

555void BlobResourceHandle::notifyResponse()
556{
557 if (!client())
558 return;
559
560 if (m_errorCode != Error::NoError) {
561 Ref<BlobResourceHandle> protectedThis(*this);
562 notifyResponseOnError();
563 notifyFinish();
564 } else
565 notifyResponseOnSuccess();
566}
567
568void BlobResourceHandle::notifyResponseOnSuccess()
569{
570 ASSERT(isMainThread());
571
572 bool isRangeRequest = m_rangeOffset != kPositionNotSpecified;
573 ResourceResponse response(firstRequest().url(), extractMIMETypeFromMediaType(m_blobData->contentType()), m_totalRemainingSize, String());
574 response.setHTTPStatusCode(isRangeRequest ? httpPartialContent : httpOK);
575 response.setHTTPStatusText(isRangeRequest ? httpPartialContentText : httpOKText);
576
577 response.setHTTPHeaderField(HTTPHeaderName::ContentType, m_blobData->contentType());
578 response.setHTTPHeaderField(HTTPHeaderName::ContentLength, String::number(m_totalRemainingSize));
579
580 if (isRangeRequest)
581 response.setHTTPHeaderField(HTTPHeaderName::ContentRange, ParsedContentRange(m_rangeOffset, m_rangeEnd, m_totalSize).headerValue());
582 // FIXME: If a resource identified with a blob: URL is a File object, user agents must use that file's name attribute,
583 // as if the response had a Content-Disposition header with the filename parameter set to the File's name attribute.
584 // Notably, this will affect a name suggested in "File Save As".
585
586 client()->didReceiveResponseAsync(this, WTFMove(response), [this, protectedThis = makeRef(*this)] {
587 m_buffer.resize(bufferSize);
588 readAsync();
589 });
590}
591
592void BlobResourceHandle::notifyResponseOnError()
593{
594 ASSERT(m_errorCode != Error::NoError);
595
596 ResourceResponse response(firstRequest().url(), "text/plain", 0, String());
597 switch (m_errorCode) {
598 case Error::RangeError:
599 response.setHTTPStatusCode(httpRequestedRangeNotSatisfiable);
600 response.setHTTPStatusText(httpRequestedRangeNotSatisfiableText);
601 break;
602 case Error::SecurityError:
603 response.setHTTPStatusCode(httpNotAllowed);
604 response.setHTTPStatusText(httpNotAllowedText);
605 break;
606 default:
607 response.setHTTPStatusCode(httpInternalError);
608 response.setHTTPStatusText(httpInternalErrorText);
609 break;
610 }
611
612 client()->didReceiveResponseAsync(this, WTFMove(response), [this, protectedThis = makeRef(*this)] {
613 m_buffer.resize(bufferSize);
614 readAsync();
615 });
616}

618void BlobResourceHandle::notifyReceiveData(const uint8_t* data, int bytesRead)
619{
620 if (client())
621 client()->didReceiveBuffer(this, SharedBuffer::create(data, bytesRead), bytesRead);
622}

624void BlobResourceHandle::notifyFail(Error errorCode)
625{
626 if (client())
627 client()->didFail(this, ResourceError(webKitBlobResourceDomain, static_cast<int>(errorCode), firstRequest().url(), String()));
628}

630static void doNotifyFinish(BlobResourceHandle& handle)
631{
632 if (handle.aborted())
633 return;
634
635 if (!handle.client())
636 return;
637
638 handle.client()->didFinishLoading(&handle, { });
639}
640
641void BlobResourceHandle::notifyFinish()
642{
643 if (!m_async) {
644 doNotifyFinish(*this);
645 return;
646 }
647
648 // Schedule to notify the client from a standalone function because the client might dispose the handle immediately from the callback function
649 // while we still have BlobResourceHandle calls in the stack.
650 callOnMainThread([protectedThis = makeRef(*this)]() mutable {
651 doNotifyFinish(protectedThis);
652 });
