34// Need to enforce empty, non-null strings due to the pickiness of the String == String operator
35// combined with the semantics of the String(NSString*) constructor
36CredentialBase::CredentialBase()
37 : m_user(emptyString())
38 , m_password(emptyString())
39 , m_persistence(CredentialPersistenceNone)
40{
41}
42
43// Need to enforce empty, non-null strings due to the pickiness of the String == String operator
44// combined with the semantics of the String(NSString*) constructor
45CredentialBase::CredentialBase(const String& user, const String& password, CredentialPersistence persistence)
46 : m_user(user.length() ? user : emptyString())
47 , m_password(password.length() ? password : emptyString())
48 , m_persistence(persistence)
49{
50}
51
52CredentialBase::CredentialBase(const Credential& original, CredentialPersistence persistence)
53 : m_user(original.user())
54 , m_password(original.password())
55 , m_persistence(persistence)
56{
57}

59bool CredentialBase::isEmpty() const
60{
61 return m_user.isEmpty() && m_password.isEmpty();
62}

64const String& CredentialBase::user() const

74bool CredentialBase::hasPassword() const
75{
76 return !m_password.isEmpty();
77}
78
79CredentialPersistence CredentialBase::persistence() const
80{
81 return m_persistence;
82}
83
84bool CredentialBase::compare(const Credential& a, const Credential& b)
85{
86 // Check persistence first since all credential types
87 // have the persistence property.
88 if (a.persistence() != b.persistence())
89 return false;
90 if (a.user() != b.user())
91 return false;
92 if (a.password() != b.password())
93 return false;
94
95 return Credential::platformCompare(a, b);
96}
97
98String CredentialBase::serializationForBasicAuthorizationHeader() const
99{
100 auto credentialStringData = makeString(m_user, ':', m_password).utf8();
101 return makeString("Basic ", base64Encoded(credentialStringData));
102}
