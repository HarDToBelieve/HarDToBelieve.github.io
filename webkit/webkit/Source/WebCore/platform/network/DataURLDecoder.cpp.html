46static bool shouldRemoveFragmentIdentifier(const String& mediaType)
47{
48#if PLATFORM(COCOA)
49 if (!linkedOnOrAfter(SDKVersion::FirstWithDataURLFragmentRemoval))
50 return false;
51
52 // HLS uses # in the middle of the manifests.
53 return !equalLettersIgnoringASCIICase(mediaType, "video/mpegurl")
54 && !equalLettersIgnoringASCIICase(mediaType, "audio/mpegurl")
55 && !equalLettersIgnoringASCIICase(mediaType, "application/x-mpegurl")
56 && !equalLettersIgnoringASCIICase(mediaType, "vnd.apple.mpegurl");
57#else
58 UNUSED_PARAM(mediaType);
59 return true;
60#endif
61}

69static Result parseMediaType(const String& mediaType)
70{
71 if (std::optional<ParsedContentType> parsedContentType = ParsedContentType::create(mediaType))
72 return { parsedContentType->mimeType(), parsedContentType->charset(), parsedContentType->serialize(), { } };
73 return { "text/plain"_s, "US-ASCII"_s, "text/plain;charset=US-ASCII"_s, { } };
74}

76struct DecodeTask {
77 WTF_MAKE_FAST_ALLOCATED;
78public:
79 DecodeTask(const URL& url, const ScheduleContext& scheduleContext, DecodeCompletionHandler&& completionHandler)
80 : url(url.isolatedCopy())
81 , scheduleContext(scheduleContext)
82 , completionHandler(WTFMove(completionHandler))
83 {
84 }
85
86 bool process()
87 {
88 // Syntax:
89 // url := data:<header>,<encodedData>
90 // header := [<mediatype>][;base64]
91 // mediatype := [<mimetype>][;charset=<charsettype>]
92
93 const char dataString[] = "data:";
94 ASSERT(url.string().startsWith(dataString));
95
96 size_t headerStart = strlen(dataString);
97 size_t headerEnd = url.string().find(',', headerStart);
98 if (headerEnd == notFound)
99 return false;
100 if (size_t fragmentInHeader = url.string().reverseFind('#', headerEnd); fragmentInHeader != notFound)
101 return false;
102 size_t encodedDataStart = headerEnd == notFound ? headerEnd : headerEnd + 1;
103
104 auto header = StringView(url.string()).substring(headerStart, headerEnd - headerStart);
105
106 // There might one or two semicolons in the header, find the last one.
107 size_t mediaTypeEnd = header.reverseFind(';');
108 mediaTypeEnd = mediaTypeEnd == notFound ? header.length() : mediaTypeEnd;
109
110 // formatTypeStart might be at the begining of "base64" or "charset=...".
111 size_t formatTypeStart = mediaTypeEnd + 1;
112 auto formatType = header.substring(formatTypeStart, header.length() - formatTypeStart);
113 formatType = stripLeadingAndTrailingHTTPSpaces(formatType);
114
115 isBase64 = equalLettersIgnoringASCIICase(formatType, "base64");
116
117 // If header does not end with "base64", mediaType should be the whole header.
118 auto mediaType = (isBase64 ? header.substring(0, mediaTypeEnd) : header).toString();
119 mediaType = stripLeadingAndTrailingHTTPSpaces(mediaType);
120 if (mediaType.startsWith(';'))
121 mediaType.insert("text/plain", 0);
122
123 if (shouldRemoveFragmentIdentifier(mediaType))
124 url.removeFragmentIdentifier();
125
126 encodedData = StringView(url.string()).substring(encodedDataStart);
127
128 result = parseMediaType(mediaType);
129 return true;
130 }
131
132 URL url;
133 StringView encodedData;
134 bool isBase64 { false };
135 const ScheduleContext scheduleContext;
136 DecodeCompletionHandler completionHandler;
137
138 Result result;
139};

141static std::unique_ptr<DecodeTask> createDecodeTask(const URL& url, const ScheduleContext& scheduleContext, DecodeCompletionHandler&& completionHandler)

150static std::optional<Vector<uint8_t>> decodeBase64(const DecodeTask& task, Mode mode)
151{
152 switch (mode) {
153 case Mode::ForgivingBase64:
154 return base64Decode(decodeURLEscapeSequences(task.encodedData), { Base64DecodeOptions::ValidatePadding, Base64DecodeOptions::IgnoreSpacesAndNewLines, Base64DecodeOptions::DiscardVerticalTab});
155
156 case Mode::Legacy:
157 // First try base64url.
158 if (auto decodedData = base64URLDecode(task.encodedData))
159 return decodedData;
160 // Didn't work, try unescaping and decoding as base64.
161 return base64Decode(decodeURLEscapeSequences(task.encodedData), { Base64DecodeOptions::IgnoreSpacesAndNewLines, Base64DecodeOptions::DiscardVerticalTab });
162 }
163
164 RELEASE_ASSERT_NOT_REACHED();
165}

167static Vector<uint8_t> decodeEscaped(const DecodeTask& task)
168{
169 TextEncoding encodingFromCharset(task.result.charset);
170 auto& encoding = encodingFromCharset.isValid() ? encodingFromCharset : UTF8Encoding();
171 return decodeURLEscapeSequencesAsData(task.encodedData, encoding);
172}

174static std::optional<Result> decodeSynchronously(DecodeTask& task, Mode mode)
175{
176 if (!task.process())
177 return std::nullopt;
178
179 if (task.isBase64) {
180 auto decodedData = decodeBase64(task, mode);
181 if (!decodedData)
182 return std::nullopt;
183 task.result.data = WTFMove(*decodedData);
184 } else
185 task.result.data = decodeEscaped(task);
186
187 task.result.data.shrinkToFit();
188 return WTFMove(task.result);
189}

191void decode(const URL& url, const ScheduleContext& scheduleContext, Mode mode, DecodeCompletionHandler&& completionHandler)
192{
193 ASSERT(url.protocolIsData());
194
195 decodeQueue().dispatch([decodeTask = createDecodeTask(url, scheduleContext, WTFMove(completionHandler)), mode]() mutable {
196 auto result = decodeSynchronously(*decodeTask, mode);
197
198#if USE(COCOA_EVENT_LOOP)
199 auto scheduledPairs = decodeTask->scheduleContext.scheduledPairs;
200#endif
201
202 auto callCompletionHandler = [result = WTFMove(result), completionHandler = WTFMove(decodeTask->completionHandler)]() mutable {
203 completionHandler(WTFMove(result));
204 };
205
206#if USE(COCOA_EVENT_LOOP)
207 RunLoop::dispatch(scheduledPairs, WTFMove(callCompletionHandler));
208#else
209 RunLoop::main().dispatch(WTFMove(callCompletionHandler));
210#endif
211 });
212}
213
214std::optional<Result> decode(const URL& url, Mode mode)
215{
216 ASSERT(url.protocolIsData());
217
218 auto task = createDecodeTask(url, { }, nullptr);
219 return decodeSynchronously(*task, mode);
220}
