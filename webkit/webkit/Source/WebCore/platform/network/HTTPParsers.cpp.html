46// True if characters which satisfy the predicate are present, incrementing
47// "pos" to the next character which does not satisfy the predicate.
48// Note: might return pos == str.length().
49static inline bool skipWhile(const String& str, unsigned& pos, const WTF::Function<bool(const UChar)>& predicate)
50{
51 const unsigned start = pos;
52 const unsigned len = str.length();
53 while (pos < len && predicate(str[pos]))
54 ++pos;
55 return pos != start;
56}
57
58// true if there is more to parse, after incrementing pos past whitespace.
59// Note: Might return pos == str.length()
60static inline bool skipWhiteSpace(const String& str, unsigned& pos)
61{
62 skipWhile(str, pos, RFC7230::isWhitespace);
63 return pos < str.length();
64}
65
66// Returns true if the function can match the whole token (case insensitive)
67// incrementing pos on match, otherwise leaving pos unchanged.
68// Note: Might return pos == str.length()
69static inline bool skipToken(const String& str, unsigned& pos, const char* token)
70{
71 unsigned len = str.length();
72 unsigned current = pos;
73
74 while (current < len && *token) {
75 if (toASCIILower(str[current]) != *token++)
76 return false;
77 ++current;
78 }
79
80 if (*token)
81 return false;
82
83 pos = current;
84 return true;
85}

87// True if the expected equals sign is seen and there is more to follow.
88static inline bool skipEquals(const String& str, unsigned &pos)
89{
90 return skipWhiteSpace(str, pos) && str[pos++] == '=' && skipWhiteSpace(str, pos);
91}

93// True if a value present, incrementing pos to next space or semicolon, if any.
94// Note: might return pos == str.length().
95static inline bool skipValue(const String& str, unsigned& pos)
96{
97 unsigned start = pos;
98 unsigned len = str.length();
99 while (pos < len) {
100 if (str[pos] == ' ' || str[pos] == '\t' || str[pos] == ';')
101 break;
102 ++pos;
103 }
104 return pos != start;
105}

107// See RFC 7230, Section 3.1.2.
108bool isValidReasonPhrase(const String& value)
109{
110 for (unsigned i = 0; i < value.length(); ++i) {
111 UChar c = value[i];
112 if (c == 0x7F || !isLatin1(c) || (c < 0x20 && c != '\t'))
113 return false;
114 }
115 return true;
116}

118// See https://fetch.spec.whatwg.org/#concept-header
119bool isValidHTTPHeaderValue(const String& value)
120{
121 UChar c = value[0];
122 if (c == ' ' || c == '\t')
123 return false;
124 c = value[value.length() - 1];
125 if (c == ' ' || c == '\t')
126 return false;
127 for (unsigned i = 0; i < value.length(); ++i) {
128 c = value[i];
129 if (c == 0x00 || c == 0x0A || c == 0x0D)
130 return false;
131 }
132 return true;
133}

135// See RFC 7231, Section 5.3.2.
136bool isValidAcceptHeaderValue(const String& value)
137{
138 for (unsigned i = 0; i < value.length(); ++i) {
139 UChar c = value[i];
140
141 // First check for alphanumeric for performance reasons then allowlist four delimiter characters.
142 if (isASCIIAlphanumeric(c) || c == ',' || c == '/' || c == ';' || c == '=')
143 continue;
144
145 ASSERT(isLatin1(c));
146 if (c == 0x7F || (c < 0x20 && c != '\t'))
147 return false;
148
149 if (RFC7230::isDelimiter(c))
150 return false;
151 }
152
153 return true;
154}

156static bool containsCORSUnsafeRequestHeaderBytes(const String& value)
157{
158 for (unsigned i = 0; i < value.length(); ++i) {
159 UChar c = value[i];
160 // https://fetch.spec.whatwg.org/#cors-unsafe-request-header-byte
161 if ((c < 0x20 && c != '\t') || (c == '"' || c == '(' || c == ')' || c == ':' || c == '<' || c == '>' || c == '?'
162 || c == '@' || c == '[' || c == '\\' || c == ']' || c == 0x7B || c == '{' || c == '}' || c == 0x7F))
163 return true;
164 }
165
166 return false;
167}

170bool isValidLanguageHeaderValue(const String& value)
171{
172 for (unsigned i = 0; i < value.length(); ++i) {
173 UChar c = value[i];
174 if (isASCIIAlphanumeric(c) || c == ' ' || c == '*' || c == ',' || c == '-' || c == '.' || c == ';' || c == '=')

186// See RFC 7230, Section 3.2.6.
187bool isValidHTTPToken(StringView value)
188{
189 if (value.isEmpty())
190 return false;
191 for (UChar c : value.codeUnits()) {
192 if (!RFC7230::isTokenCharacter(c))
193 return false;
194 }
195 return true;
196}

204// True if the character at the given position satisifies a predicate, incrementing "pos" by one.
205// Note: Might return pos == str.length()
206static inline bool skipCharacter(const String& value, unsigned& pos, WTF::Function<bool(const UChar)>&& predicate)
207{
208 if (pos < value.length() && predicate(value[pos])) {
209 ++pos;
210 return true;
211 }
212 return false;
213}

215// True if the "expected" character is at the given position, incrementing "pos" by one.
216// Note: Might return pos == str.length()
217static inline bool skipCharacter(const String& value, unsigned& pos, const UChar expected)
218{
219 return skipCharacter(value, pos, [expected](const UChar c) {
220 return c == expected;
221 });
222}

224// True if a quoted pair is present, incrementing "pos" to the position after the quoted pair.
225// Note: Might return pos == str.length()
226// See RFC 7230, Section 3.2.6.
227static constexpr auto QuotedPairStartCharacter = '\\';
228static bool skipQuotedPair(const String& value, unsigned& pos)
229{
230 // quoted-pair = "\" ( HTAB / SP / VCHAR / obs-text )
231 return skipCharacter(value, pos, QuotedPairStartCharacter)
232 && skipCharacter(value, pos, RFC7230::isQuotedPairSecondOctet);
233}
234
235// True if a comment is present, incrementing "pos" to the position after the comment.
236// Note: Might return pos == str.length()
237// See RFC 7230, Section 3.2.6.
238static constexpr auto CommentStartCharacter = '(';
239static constexpr auto CommentEndCharacter = ')';
240static bool skipComment(const String& value, unsigned& pos)
241{
242 // comment = "(" *( ctext / quoted-pair / comment ) ")"
243 // ctext = HTAB / SP / %x21-27 / %x2A-5B / %x5D-7E / obs-text
244 if (!skipCharacter(value, pos, CommentStartCharacter))
245 return false;
246
247 const unsigned end = value.length();
248 while (pos < end && value[pos] != CommentEndCharacter) {
249 switch (value[pos]) {
250 case CommentStartCharacter:
251 if (!skipComment(value, pos))
252 return false;
253 break;
254 case QuotedPairStartCharacter:
255 if (!skipQuotedPair(value, pos))
256 return false;
257 break;
258 default:
259 if (!skipWhile(value, pos, RFC7230::isCommentText))
260 return false;
261 }
262 }
263 return skipCharacter(value, pos, CommentEndCharacter);
264}

266// True if an HTTP header token is present, incrementing "pos" to the position after it.
267// Note: Might return pos == str.length()
268// See RFC 7230, Section 3.2.6.
269static bool skipHTTPToken(const String& value, unsigned& pos)
270{
271 return skipWhile(value, pos, RFC7230::isTokenCharacter);
272}

274// True if a product specifier (as in an User-Agent header) is present, incrementing "pos" to the position after it.
275// Note: Might return pos == str.length()
276// See RFC 7231, Section 5.5.3.
277static bool skipUserAgentProduct(const String& value, unsigned& pos)
278{
279 // product = token ["/" product-version]
280 // product-version = token
281 if (!skipHTTPToken(value, pos))
282 return false;
283 if (skipCharacter(value, pos, '/'))
284 return skipHTTPToken(value, pos);
285 return true;
286}
287
288// See RFC 7231, Section 5.5.3
289bool isValidUserAgentHeaderValue(const String& value)
290{
291 // User-Agent = product *( RWS ( product / comment ) )
292 unsigned pos = 0;
293 if (!skipUserAgentProduct(value, pos))
294 return false;
295
296 while (pos < value.length()) {
297 if (!skipWhiteSpace(value, pos))
298 return false;
299 if (value[pos] == CommentStartCharacter) {
300 if (!skipComment(value, pos))
301 return false;
302 } else {
303 if (!skipUserAgentProduct(value, pos))
304 return false;
305 }
306 }
307
308 return pos == value.length();
309}

312static const size_t maxInputSampleSize = 128;
313template<typename CharType>
314static String trimInputSample(CharType* p, size_t length)
315{
316 String s = String(p, std::min<size_t>(length, maxInputSampleSize));
317 if (length > maxInputSampleSize)
318 s.append(horizontalEllipsis);
319 return s;
320}

322std::optional<WallTime> parseHTTPDate(const String& value)
323{
324 double dateInMillisecondsSinceEpoch = parseDateFromNullTerminatedCharacters(value.utf8().data());
325 if (!std::isfinite(dateInMillisecondsSinceEpoch))
326 return std::nullopt;
327 // This assumes system_clock epoch equals Unix epoch which is true for all implementations but unspecified.
328 // FIXME: The parsing function should be switched to WallTime too.
329 return WallTime::fromRawSeconds(dateInMillisecondsSinceEpoch / 1000.0);
330}

332// FIXME: This function doesn't comply with RFC 6266.
333// For example, this function doesn't handle the interaction between " and ;
334// that arises from quoted-string, nor does this function properly unquote
335// attribute values. Further this function appears to process parameter names
336// in a case-sensitive manner. (There are likely other bugs as well.)
337String filenameFromHTTPContentDisposition(const String& value)
338{
339 for (auto& keyValuePair : value.split(';')) {
340 size_t valueStartPos = keyValuePair.find('=');
341 if (valueStartPos == notFound)
342 continue;
343
344 String key = keyValuePair.left(valueStartPos).stripWhiteSpace();
345
346 if (key.isEmpty() || key != "filename")
347 continue;
348
349 String value = keyValuePair.substring(valueStartPos + 1).stripWhiteSpace();
350
351 // Remove quotes if there are any
352 if (value[0] == '\"')
353 value = value.substring(1, value.length() - 2);
354
355 return value;
356 }
357
358 return String();
359}

361String extractMIMETypeFromMediaType(const String& mediaType)
362{
363 unsigned position = 0;
364 unsigned length = mediaType.length();
365
366 for (; position < length; ++position) {
367 UChar c = mediaType[position];
368 if (c != '\t' && c != ' ')
369 break;
370 }
371
372 if (position == length)
373 return mediaType;
374
375 unsigned typeStart = position;
376
377 unsigned typeEnd = position;
378 for (; position < length; ++position) {
379 UChar c = mediaType[position];
380
381 // While RFC 2616 does not allow it, other browsers allow multiple values in the HTTP media
382 // type header field, Content-Type. In such cases, the media type string passed here may contain
383 // the multiple values separated by commas. For now, this code ignores text after the first comma,
384 // which prevents it from simply failing to parse such types altogether. Later for better
385 // compatibility we could consider using the first or last valid MIME type instead.
386 // See https://bugs.webkit.org/show_bug.cgi?id=25352 for more discussion.
387 if (c == ',')
388 break;
389
390 if (c == '\t' || c == ' ' || c == ';')
391 break;
392
393 typeEnd = position + 1;
394 }
395
396 return mediaType.substring(typeStart, typeEnd - typeStart);
397}

399String extractCharsetFromMediaType(const String& mediaType)
400{
401 unsigned charsetPos = 0, charsetLen = 0;
402 size_t pos = 0;
403 unsigned length = mediaType.length();
404
405 while (pos < length) {
406 pos = mediaType.findIgnoringASCIICase("charset", pos);
407 if (pos == notFound || pos == 0) {
408 charsetLen = 0;
409 break;
410 }
411
412 // is what we found a beginning of a word?
413 if (mediaType[pos-1] > ' ' && mediaType[pos-1] != ';') {
414 pos += 7;
415 continue;
416 }
417
418 pos += 7;
419
420 // skip whitespace
421 while (pos != length && mediaType[pos] <= ' ')
422 ++pos;
423
424 if (mediaType[pos++] != '=') // this "charset" substring wasn't a parameter name, but there may be others
425 continue;
426
427 while (pos != length && (mediaType[pos] <= ' ' || mediaType[pos] == '"' || mediaType[pos] == '\''))
428 ++pos;
429
430 // we don't handle spaces within quoted parameter values, because charset names cannot have any
431 unsigned endpos = pos;
432 while (pos != length && mediaType[endpos] > ' ' && mediaType[endpos] != '"' && mediaType[endpos] != '\'' && mediaType[endpos] != ';')
433 ++endpos;
434
435 charsetPos = pos;
436 charsetLen = endpos - pos;
437 break;
438 }
439 return mediaType.substring(charsetPos, charsetLen);
440}

442XSSProtectionDisposition parseXSSProtectionHeader(const String& header, String& failureReason, unsigned& failurePosition, String& reportURL)
443{
444 static NeverDestroyed<String> failureReasonInvalidToggle(MAKE_STATIC_STRING_IMPL("expected 0 or 1"));
445 static NeverDestroyed<String> failureReasonInvalidSeparator(MAKE_STATIC_STRING_IMPL("expected semicolon"));
446 static NeverDestroyed<String> failureReasonInvalidEquals(MAKE_STATIC_STRING_IMPL("expected equals sign"));
447 static NeverDestroyed<String> failureReasonInvalidMode(MAKE_STATIC_STRING_IMPL("invalid mode directive"));
448 static NeverDestroyed<String> failureReasonInvalidReport(MAKE_STATIC_STRING_IMPL("invalid report directive"));
449 static NeverDestroyed<String> failureReasonDuplicateMode(MAKE_STATIC_STRING_IMPL("duplicate mode directive"));
450 static NeverDestroyed<String> failureReasonDuplicateReport(MAKE_STATIC_STRING_IMPL("duplicate report directive"));
451 static NeverDestroyed<String> failureReasonInvalidDirective(MAKE_STATIC_STRING_IMPL("unrecognized directive"));
452
453 unsigned pos = 0;
454
455 if (!skipWhiteSpace(header, pos))
456 return XSSProtectionDisposition::Enabled;
457
458 if (header[pos] == '0')
459 return XSSProtectionDisposition::Disabled;
460
461 if (header[pos++] != '1') {
462 failureReason = failureReasonInvalidToggle;
463 return XSSProtectionDisposition::Invalid;
464 }
465
466 XSSProtectionDisposition result = XSSProtectionDisposition::Enabled;
467 bool modeDirectiveSeen = false;
468 bool reportDirectiveSeen = false;
469
470 while (1) {
471 // At end of previous directive: consume whitespace, semicolon, and whitespace.
472 if (!skipWhiteSpace(header, pos))
473 return result;
474
475 if (header[pos++] != ';') {
476 failureReason = failureReasonInvalidSeparator;
477 failurePosition = pos;
478 return XSSProtectionDisposition::Invalid;
479 }
480
481 if (!skipWhiteSpace(header, pos))
482 return result;
483
484 // At start of next directive.
485 if (skipToken(header, pos, "mode")) {
486 if (modeDirectiveSeen) {
487 failureReason = failureReasonDuplicateMode;
488 failurePosition = pos;
489 return XSSProtectionDisposition::Invalid;
490 }
491 modeDirectiveSeen = true;
492 if (!skipEquals(header, pos)) {
493 failureReason = failureReasonInvalidEquals;
494 failurePosition = pos;
495 return XSSProtectionDisposition::Invalid;
496 }
497 if (!skipToken(header, pos, "block")) {
498 failureReason = failureReasonInvalidMode;
499 failurePosition = pos;
500 return XSSProtectionDisposition::Invalid;
501 }
502 result = XSSProtectionDisposition::BlockEnabled;
503 } else if (skipToken(header, pos, "report")) {
504 if (reportDirectiveSeen) {
505 failureReason = failureReasonDuplicateReport;
506 failurePosition = pos;
507 return XSSProtectionDisposition::Invalid;
508 }
509 reportDirectiveSeen = true;
510 if (!skipEquals(header, pos)) {
511 failureReason = failureReasonInvalidEquals;
512 failurePosition = pos;
513 return XSSProtectionDisposition::Invalid;
514 }
515 size_t startPos = pos;
516 if (!skipValue(header, pos)) {
517 failureReason = failureReasonInvalidReport;
518 failurePosition = pos;
519 return XSSProtectionDisposition::Invalid;
520 }
521 reportURL = header.substring(startPos, pos - startPos);
522 failurePosition = startPos; // If later semantic check deems unacceptable.
523 } else {
524 failureReason = failureReasonInvalidDirective;
525 failurePosition = pos;
526 return XSSProtectionDisposition::Invalid;
527 }
528 }
529}

539// For example: "HTTP/1.1 200 OK" => "OK".
540// Note that HTTP/2 does not include a reason phrase, so we return the empty atom.
541AtomString extractReasonPhraseFromHTTPStatusLine(const String& statusLine)
542{

544 size_t spacePos = view.find(' ');
545
546 // Remove status code from the status line.

554XFrameOptionsDisposition parseXFrameOptionsHeader(const String& header)
555{
556 XFrameOptionsDisposition result = XFrameOptionsDisposition::None;
557
558 if (header.isEmpty())
559 return result;
560
561 for (auto& currentHeader : header.split(',')) {
562 currentHeader = currentHeader.stripWhiteSpace();
563 XFrameOptionsDisposition currentValue = XFrameOptionsDisposition::None;
564 if (equalLettersIgnoringASCIICase(currentHeader, "deny"))
565 currentValue = XFrameOptionsDisposition::Deny;
566 else if (equalLettersIgnoringASCIICase(currentHeader, "sameorigin"))
567 currentValue = XFrameOptionsDisposition::SameOrigin;
568 else if (equalLettersIgnoringASCIICase(currentHeader, "allowall"))
569 currentValue = XFrameOptionsDisposition::AllowAll;
570 else
571 currentValue = XFrameOptionsDisposition::Invalid;
572
573 if (result == XFrameOptionsDisposition::None)
574 result = currentValue;
575 else if (result != currentValue)
576 return XFrameOptionsDisposition::Conflict;
577 }
578 return result;
579}

581bool parseRange(const String& range, long long& rangeOffset, long long& rangeEnd, long long& rangeSuffixLength)
582{
583 // The format of "Range" header is defined in RFC 2616 Section 14.35.1.
584 // http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.35.1
585 // We don't support multiple range requests.
586
587 rangeOffset = rangeEnd = rangeSuffixLength = -1;
588
589 // The "bytes" unit identifier should be present.
590 static const unsigned bytesLength = 6;
591 if (!startsWithLettersIgnoringASCIICase(range, "bytes="))
592 return false;
593 // FIXME: The rest of this should use StringView.
594 String byteRange = range.substring(bytesLength);
595
596 // The '-' character needs to be present.
597 int index = byteRange.find('-');
598 if (index == -1)
599 return false;
600
601 // If the '-' character is at the beginning, the suffix length, which specifies the last N bytes, is provided.
602 // Example:
603 // -500
604 if (!index) {
605 if (auto value = parseInteger<long long>(StringView { byteRange }.substring(index + 1)))
606 rangeSuffixLength = *value;
607 return true;
608 }
609
610 // Otherwise, the first-byte-position and the last-byte-position are provied.
611 // Examples:
612 // 0-499
613 // 500-
614 auto firstBytePos = parseInteger<long long>(StringView { byteRange }.left(index));
615 if (!firstBytePos)
616 return false;
617
618 auto lastBytePosStr = stripLeadingAndTrailingHTTPSpaces(StringView { byteRange }.substring(index + 1));
619 long long lastBytePos = -1;
620 if (!lastBytePosStr.isEmpty()) {
621 auto value = parseInteger<long long>(lastBytePosStr);
622 if (!value)
623 return false;
624 lastBytePos = *value;
625 }
626
627 if (*firstBytePos < 0 || !(lastBytePos == -1 || lastBytePos >= *firstBytePos))
628 return false;
629
630 rangeOffset = *firstBytePos;
631 rangeEnd = lastBytePos;
632 return true;
633}

635template<typename CharacterType>
636static inline bool isValidHeaderNameCharacter(CharacterType character)
637{
638 // https://tools.ietf.org/html/rfc7230#section-3.2
639 // A header name should only contain one or more of
640 // alphanumeric or ! # $ % & ' * + - . ^ _ ` | ~
641 if (isASCIIAlphanumeric(character))
642 return true;
643 switch (character) {
644 case '!':
645 case '#':
646 case '$':
647 case '%':
648 case '&':
649 case '\'':
650 case '*':
651 case '+':
652 case '-':
653 case '.':
654 case '^':
655 case '_':
656 case '`':
657 case '|':
658 case '~':
659 return true;
660 default:
661 return false;
662 }
663}

665size_t parseHTTPHeader(const uint8_t* start, size_t length, String& failureReason, StringView& nameStr, String& valueStr, bool strict)
666{
667 auto p = start;
668 auto end = start + length;
669
670 Vector<uint8_t> name;
671 Vector<uint8_t> value;
672
673 bool foundFirstNameChar = false;
674 const uint8_t* namePtr = nullptr;
675 size_t nameSize = 0;
676
677 nameStr = StringView();
678 valueStr = String();
679
680 for (; p < end; p++) {
681 switch (*p) {
682 case '\r':
683 if (name.isEmpty()) {
684 if (p + 1 < end && *(p + 1) == '\n')
685 return (p + 2) - start;
686 failureReason = makeString("CR doesn't follow LF in header name at ", trimInputSample(p, end - p));
687 return 0;
688 }
689 failureReason = makeString("Unexpected CR in header name at ", trimInputSample(name.data(), name.size()));
690 return 0;
691 case '\n':
692 failureReason = makeString("Unexpected LF in header name at ", trimInputSample(name.data(), name.size()));
693 return 0;
694 case ':':
695 break;
696 default:
697 if (!isValidHeaderNameCharacter(*p)) {
698 if (name.size() < 1)
699 failureReason = "Unexpected start character in header name";
700 else
701 failureReason = makeString("Unexpected character in header name at ", trimInputSample(name.data(), name.size()));
702 return 0;
703 }
704 name.append(*p);
705 if (!foundFirstNameChar) {
706 namePtr = p;
707 foundFirstNameChar = true;
708 }
709 continue;
710 }
711 if (*p == ':') {
712 ++p;
713 break;
714 }
715 }
716
717 nameSize = name.size();
718 nameStr = StringView(namePtr, nameSize);
719
720 for (; p < end && *p == 0x20; p++) { }
721
722 for (; p < end; p++) {
723 switch (*p) {
724 case '\r':
725 break;
726 case '\n':
727 if (strict) {
728 failureReason = makeString("Unexpected LF in header value at ", trimInputSample(value.data(), value.size()));
729 return 0;
730 }
731 break;
732 default:
733 value.append(*p);
734 }
735 if (*p == '\r' || (!strict && *p == '\n')) {
736 ++p;
737 break;
738 }
739 }
740 if (p >= end || (strict && *p != '\n')) {
741 failureReason = makeString("CR doesn't follow LF after header value at ", trimInputSample(p, end - p));
742 return 0;
743 }
744 valueStr = String::fromUTF8(value.data(), value.size());
745 if (valueStr.isNull()) {
746 failureReason = "Invalid UTF-8 sequence in header value"_s;
747 return 0;
748 }
749 return p - start;
750}

868// Implements https://fetch.spec.whatwg.org/#cors-safelisted-request-header
869bool isCrossOriginSafeRequestHeader(HTTPHeaderName name, const String& value)
870{
871 switch (name) {
872 case HTTPHeaderName::Accept:
873 if (!isValidAcceptHeaderValue(value))
874 return false;
875 break;
876 case HTTPHeaderName::AcceptLanguage:
877 case HTTPHeaderName::ContentLanguage:
878 if (!isValidLanguageHeaderValue(value))
879 return false;
880 break;
881 case HTTPHeaderName::ContentType: {
882 // Preflight is required for MIME types that can not be sent via form submission.
883 if (containsCORSUnsafeRequestHeaderBytes(value))
884 return false;
885 auto parsedContentType = ParsedContentType::create(value);
886 if (!parsedContentType)
887 return false;
888 String mimeType = parsedContentType->mimeType();
889 if (!(equalLettersIgnoringASCIICase(mimeType, "application/x-www-form-urlencoded") || equalLettersIgnoringASCIICase(mimeType, "multipart/form-data") || equalLettersIgnoringASCIICase(mimeType, "text/plain")))
890 return false;
891 break;
892 }
893 default:
894 // FIXME: Should we also make safe other headers (DPR, Downlink, Save-Data...)? That would require validating their values.
895 return false;
896 }
897 return value.length() <= 128;
898}

900// Implements <https://fetch.spec.whatwg.org/#concept-method-normalize>.
901String normalizeHTTPMethod(const String& method)
902{
903 const ASCIILiteral methods[] = { "DELETE"_s, "GET"_s, "HEAD"_s, "OPTIONS"_s, "POST"_s, "PUT"_s };
904 for (auto value : methods) {
905 if (equalIgnoringASCIICase(method, value.characters())) {
906 // Don't bother allocating a new string if it's already all uppercase.
907 if (method == value)
908 break;
909 return value;
910 }
911 }
912 return method;
913}
