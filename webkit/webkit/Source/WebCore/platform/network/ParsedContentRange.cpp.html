35static bool areContentRangeValuesValid(int64_t firstBytePosition, int64_t lastBytePosition, int64_t instanceLength)
36{
37 // From <http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html>
38 // 14.16 Content-Range
39 // A byte-content-range-spec with a byte-range-resp-spec whose last- byte-pos value is less than its first-byte-pos value,
40 // or whose instance-length value is less than or equal to its last-byte-pos value, is invalid.
41 if (firstBytePosition < 0)
42 return false;
43 ASSERT(firstBytePosition >= 0);
44
45 if (lastBytePosition < firstBytePosition)
46 return false;
47 ASSERT(lastBytePosition >= 0);
48
49 if (instanceLength == ParsedContentRange::unknownLength)
50 return true;
51
52 return lastBytePosition < instanceLength;
53}
54
55static bool parseContentRange(StringView headerValue, int64_t& firstBytePosition, int64_t& lastBytePosition, int64_t& instanceLength)
56{
57 // From <http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html>
58 // 14.16 Content-Range
59 //
60 // Content-Range = "Content-Range" ":" content-range-spec
61 // content-range-spec = byte-content-range-spec
62 // byte-content-range-spec = bytes-unit SP
63 // byte-range-resp-spec "/"
64 // ( instance-length | "*" )
65 // byte-range-resp-spec = (first-byte-pos "-" last-byte-pos)
66 // | "*"
67 // instance-length = 1*DIGIT
68
69 static const char* prefix = "bytes ";
70 static const size_t prefixLength = 6;
71
72 if (!headerValue.startsWith(prefix))
73 return false;
74
75 size_t byteSeparatorTokenLoc = headerValue.find('-', prefixLength);
76 if (byteSeparatorTokenLoc == notFound)
77 return false;
78
79 size_t instanceLengthSeparatorToken = headerValue.find('/', byteSeparatorTokenLoc + 1);
80 if (instanceLengthSeparatorToken == notFound)
81 return false;
82
83 auto firstByteString = headerValue.substring(prefixLength, byteSeparatorTokenLoc - prefixLength);
84 if (!firstByteString.isAllSpecialCharacters<isASCIIDigit>())
85 return false;
86
87 auto optionalFirstBytePosition = parseInteger<int64_t>(firstByteString);
88 if (!optionalFirstBytePosition)
89 return false;
90 firstBytePosition = *optionalFirstBytePosition;
91
92 auto lastByteString = headerValue.substring(byteSeparatorTokenLoc + 1, instanceLengthSeparatorToken - (byteSeparatorTokenLoc + 1));
93 if (!lastByteString.isAllSpecialCharacters<isASCIIDigit>())
94 return false;
95
96 auto optionalLastBytePosition = parseInteger<int64_t>(lastByteString);
97 if (!optionalLastBytePosition)
98 return false;
99 lastBytePosition = *optionalLastBytePosition;
100
101 auto instanceString = headerValue.substring(instanceLengthSeparatorToken + 1);
102 if (instanceString == "*")
103 instanceLength = ParsedContentRange::unknownLength;
104 else {
105 if (!instanceString.isAllSpecialCharacters<isASCIIDigit>())
106 return false;
107
108 auto optionalInstanceLength = parseInteger<int64_t>(instanceString);
109 if (!optionalInstanceLength)
110 return false;
111 instanceLength = *optionalInstanceLength;
112 }
113
114 return areContentRangeValuesValid(firstBytePosition, lastBytePosition, instanceLength);
115}
116
117ParsedContentRange::ParsedContentRange(const String& headerValue)
118{
119 if (!parseContentRange(StringView(headerValue), m_firstBytePosition, m_lastBytePosition, m_instanceLength))
120 m_instanceLength = invalidLength;
121}
122
123ParsedContentRange::ParsedContentRange(int64_t firstBytePosition, int64_t lastBytePosition, int64_t instanceLength)
124 : m_firstBytePosition(firstBytePosition)
125 , m_lastBytePosition(lastBytePosition)
126 , m_instanceLength(instanceLength)
127{
128 if (!areContentRangeValuesValid(m_firstBytePosition, m_lastBytePosition, m_instanceLength))
129 m_instanceLength = invalidLength;
130}
131
132String ParsedContentRange::headerValue() const
133{
134 if (!isValid())
135 return String();
136 if (m_instanceLength == unknownLength)
137 return makeString("bytes ", m_firstBytePosition, '-', m_lastBytePosition, "/*");
138 return makeString("bytes ", m_firstBytePosition, '-', m_lastBytePosition, '/', m_instanceLength);
139}
