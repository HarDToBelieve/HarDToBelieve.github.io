41static void skipSpaces(StringView input, unsigned& startIndex)
42{
43 while (startIndex < input.length() && isHTTPSpace(input[startIndex]))
44 ++startIndex;
45}

47static bool isQuotedStringTokenCharacter(UChar c)
48{
49 return (c >= ' ' && c <= '~') || (c >= 0x80 && c <= 0xFF) || c == '\t';
50}
51
52static bool isTokenCharacter(UChar c)
53{
54 return isASCII(c) && c > ' ' && c != '"' && c != '(' && c != ')' && c != ',' && c != '/' && (c < ':' || c > '@') && (c < '[' || c > ']');
55}
56
57using CharacterMeetsCondition = bool (*)(UChar);
58
59static StringView parseToken(StringView input, unsigned& startIndex, CharacterMeetsCondition characterMeetsCondition, Mode mode, bool skipTrailingWhitespace = false)
60{
61 unsigned inputLength = input.length();
62 unsigned tokenStart = startIndex;
63 unsigned& tokenEnd = startIndex;
64
65 if (tokenEnd >= inputLength)
66 return StringView();
67
68 while (tokenEnd < inputLength && characterMeetsCondition(input[tokenEnd])) {
69 if (mode == Mode::Rfc2045 && !isTokenCharacter(input[tokenEnd]))
70 break;
71 ++tokenEnd;
72 }
73
74 if (tokenEnd == tokenStart)
75 return StringView();
76 if (skipTrailingWhitespace) {
77 if (mode == Mode::Rfc2045) {
78 while (input[tokenEnd - 1] == ' ')
79 --tokenEnd;
80 } else {
81 while (isHTTPSpace(input[tokenEnd - 1]))
82 --tokenEnd;
83 }
84 }
85 return input.substring(tokenStart, tokenEnd - tokenStart);
86}

88static bool isNotQuoteOrBackslash(UChar ch)
89{
90 return ch != '"' && ch != '\\';
91}
92
93static String collectHTTPQuotedString(StringView input, unsigned& startIndex)
94{
95 ASSERT(input[startIndex] == '"');
96 unsigned inputLength = input.length();
97 unsigned& position = startIndex;
98 position++;
99 StringBuilder builder;
100 while (true) {
101 unsigned positionStart = position;
102 parseToken(input, position, isNotQuoteOrBackslash, Mode::MimeSniff);
103 builder.append(input.substring(positionStart, position - positionStart));
104 if (position >= inputLength)
105 break;
106 UChar quoteOrBackslash = input[position++];
107 if (quoteOrBackslash == '\\') {
108 if (position >= inputLength) {
109 builder.append(quoteOrBackslash);
110 break;
111 }
112 builder.append(input[position++]);
113 } else {
114 ASSERT(quoteOrBackslash == '"');
115 break;
116 }
117
118 }
119 return builder.toString();
120}

122static bool containsNonTokenCharacters(StringView input, Mode mode)
123{
124 if (mode == Mode::MimeSniff)
125 return !isValidHTTPToken(input);
126 for (unsigned index = 0; index < input.length(); ++index) {
127 if (!isTokenCharacter(input[index]))
128 return true;
129 }
130 return false;
131}

133static StringView parseQuotedString(StringView input, unsigned& startIndex)
134{
135 unsigned inputLength = input.length();
136 unsigned quotedStringStart = startIndex + 1;
137 unsigned& quotedStringEnd = startIndex;
138
139 if (quotedStringEnd >= inputLength)
140 return StringView();
141
142 if (input[quotedStringEnd++] != '"' || quotedStringEnd >= inputLength)
143 return StringView();
144
145 bool lastCharacterWasBackslash = false;
146 char currentCharacter;
147 while ((currentCharacter = input[quotedStringEnd++]) != '"' || lastCharacterWasBackslash) {
148 if (quotedStringEnd >= inputLength)
149 return StringView();
150 if (currentCharacter == '\\' && !lastCharacterWasBackslash) {
151 lastCharacterWasBackslash = true;
152 continue;
153 }
154 if (lastCharacterWasBackslash)
155 lastCharacterWasBackslash = false;
156 }
157 if (input[quotedStringEnd - 1] == '"')
158 quotedStringEnd++;
159 return input.substring(quotedStringStart, quotedStringEnd - quotedStringStart);
160}

162// From http://tools.ietf.org/html/rfc2045#section-5.1:
163//
164// content := "Content-Type" ":" type "/" subtype
165// *(";" parameter)
166// ; Matching of media type and subtype
167// ; is ALWAYS case-insensitive.
168//
169// type := discrete-type / composite-type
170//
171// discrete-type := "text" / "image" / "audio" / "video" /
172// "application" / extension-token
173//
174// composite-type := "message" / "multipart" / extension-token
175//
176// extension-token := ietf-token / x-token
177//
178// ietf-token := <An extension token defined by a
179// standards-track RFC and registered
180// with IANA.>
181//
182// x-token := <The two characters "X-" or "x-" followed, with
183// no intervening white space, by any token>
184//
185// subtype := extension-token / iana-token
186//
187// iana-token := <A publicly-defined extension token. Tokens
188// of this form must be registered with IANA
189// as specified in RFC 2048.>
190//
191// parameter := attribute "=" value
192//
193// attribute := token
194// ; Matching of attributes
195// ; is ALWAYS case-insensitive.
196//
197// value := token / quoted-string
198//
199// token := 1*<any (US-ASCII) CHAR except SPACE, CTLs,
200// or tspecials>
201//
202// tspecials := "(" / ")" / "<" / ">" / "@" /
203// "," / ";" / ":" / "\" / <">
204// "/" / "[" / "]" / "?" / "="
205// ; Must be in quoted-string,
206// ; to use within parameter values
207
208static bool isNotForwardSlash(UChar ch)
209{
210 return ch != '/';
211}
212
213static bool isNotSemicolon(UChar ch)
214{
215 return ch != ';';
216}
217
218static bool isNotSemicolonOrEqualSign(UChar ch)
219{
220 return ch != ';' && ch != '=';
221}
222
223static bool containsNewline(UChar ch)
224{
225 return ch == '\r' || ch == '\n';
226}

228bool ParsedContentType::parseContentType(Mode mode)
229{
230 if (mode == Mode::Rfc2045 && m_contentType.find(containsNewline) != notFound)
231 return false;
232 unsigned index = 0;
233 unsigned contentTypeLength = m_contentType.length();
234 skipSpaces(m_contentType, index);
235 if (index >= contentTypeLength) {
236 LOG_ERROR("Invalid Content-Type string '%s'", m_contentType.ascii().data());
237 return false;
238 }
239
240 unsigned contentTypeStart = index;
241 auto typeRange = parseToken(m_contentType, index, isNotForwardSlash, mode);
242 if (typeRange.isNull() || containsNonTokenCharacters(typeRange, mode)) {
243 LOG_ERROR("Invalid Content-Type, invalid type value.");
244 return false;
245 }
246
247 if (index >= contentTypeLength || m_contentType[index++] != '/') {
248 LOG_ERROR("Invalid Content-Type, missing '/'.");
249 return false;
250 }
251
252 auto subTypeRange = parseToken(m_contentType, index, isNotSemicolon, mode, mode == Mode::MimeSniff);
253 if (subTypeRange.isNull() || containsNonTokenCharacters(subTypeRange, mode)) {
254 LOG_ERROR("Invalid Content-Type, invalid subtype value.");
255 return false;
256 }
257
258 // There should not be any quoted strings until we reach the parameters.
259 size_t semiColonIndex = m_contentType.find(';', contentTypeStart);
260 if (semiColonIndex == notFound) {
261 setContentType(m_contentType.substring(contentTypeStart, contentTypeLength - contentTypeStart), mode);
262 return true;
263 }
264
265 setContentType(m_contentType.substring(contentTypeStart, semiColonIndex - contentTypeStart), mode);
266 index = semiColonIndex + 1;
267 while (true) {
268 skipSpaces(m_contentType, index);
269 auto keyRange = parseToken(m_contentType, index, isNotSemicolonOrEqualSign, mode);
270 if (mode == Mode::Rfc2045 && (keyRange.isNull() || index >= contentTypeLength)) {
271 LOG_ERROR("Invalid Content-Type parameter name.");
272 return false;
273 }
274
275 // Should we tolerate spaces here?
276 if (mode == Mode::Rfc2045) {
277 if (index >= contentTypeLength || m_contentType[index++] != '=') {
278 LOG_ERROR("Invalid Content-Type malformed parameter.");
279 return false;
280 }
281 } else {
282 if (index >= contentTypeLength)
283 break;
284 if (m_contentType[index] != '=' && m_contentType[index] != ';') {
285 LOG_ERROR("Invalid Content-Type malformed parameter.");
286 return false;
287 }
288 if (m_contentType[index++] == ';')
289 continue;
290 }
291
292 // Should we tolerate spaces here?
293 String parameterValue;
294 StringView valueRange;
295 if (index < contentTypeLength && m_contentType[index] == '"') {
296 if (mode == Mode::MimeSniff) {
297 parameterValue = collectHTTPQuotedString(m_contentType, index);
298 parseToken(m_contentType, index, isNotSemicolon, mode);
299 } else
300 valueRange = parseQuotedString(m_contentType, index);
301 } else
302 valueRange = parseToken(m_contentType, index, isNotSemicolon, mode, mode == Mode::MimeSniff);
303
304 if (parameterValue.isNull()) {
305 if (valueRange.isNull()) {
306 if (mode == Mode::MimeSniff)
307 continue;
308 LOG_ERROR("Invalid Content-Type, invalid parameter value.");
309 return false;
310 }
311 parameterValue = valueRange.toString();
312 }
313
314 // Should we tolerate spaces here?
315 if (mode == Mode::Rfc2045 && index < contentTypeLength && m_contentType[index++] != ';') {
316 LOG_ERROR("Invalid Content-Type, invalid character at the end of key/value parameter.");
317 return false;
318 }
319
320 if (!keyRange.isNull())
321 setContentTypeParameter(keyRange.toString(), parameterValue, mode);
322
323 if (index >= contentTypeLength)
324 return true;
325 }
326
327 return true;
328}

330std::optional<ParsedContentType> ParsedContentType::create(const String& contentType, Mode mode)
331{
332 ParsedContentType parsedContentType(mode == Mode::Rfc2045 ? contentType : stripLeadingAndTrailingHTTPSpaces(contentType));
333 if (!parsedContentType.parseContentType(mode))
334 return std::nullopt;
335 return { WTFMove(parsedContentType) };
336}

338bool isValidContentType(const String& contentType, Mode mode)
339{
340 return ParsedContentType::create(contentType, mode) != std::nullopt;
341}

368void ParsedContentType::setContentType(StringView contentRange, Mode mode)
369{
370 m_mimeType = contentRange.toString();
371 if (mode == Mode::MimeSniff)
372 m_mimeType = stripLeadingAndTrailingHTTPSpaces(m_mimeType).convertToASCIILowercase();
373 else
374 m_mimeType = m_mimeType.stripWhiteSpace();
375}

377static bool containsNonQuoteStringTokenCharacters(const String& input)
378{
379 for (unsigned index = 0; index < input.length(); ++index) {
380 if (!isQuotedStringTokenCharacter(input[index]))
381 return true;
382 }
383 return false;

386void ParsedContentType::setContentTypeParameter(const String& keyName, const String& keyValue, Mode mode)
387{
388 String name = keyName;
389 if (mode == Mode::MimeSniff) {
390 if (m_parameterValues.contains(name) || !isValidHTTPToken(name) || containsNonQuoteStringTokenCharacters(keyValue))
391 return;
392 name = name.convertToASCIILowercase();
393 }
394 m_parameterValues.set(name, keyValue);
395 m_parameterNames.append(name);
396}

398String ParsedContentType::serialize() const
399{
400 StringBuilder builder;
401 builder.append(m_mimeType);
402 for (auto& name : m_parameterNames) {
403 builder.append(';');
404 builder.append(name);
405 builder.append('=');
406 String value = m_parameterValues.get(name);
407 if (value.isEmpty() || !isValidHTTPToken(value)) {
408 builder.append('"');
409 for (unsigned index = 0; index < value.length(); ++index) {
410 auto ch = value[index];
411 if (ch == '\\' || ch =='"')
412 builder.append('\\');
413 builder.append(ch);
414 }
415 builder.append('"');
416 } else
417 builder.append(value);
418 }
419 return builder.toString();
420}
