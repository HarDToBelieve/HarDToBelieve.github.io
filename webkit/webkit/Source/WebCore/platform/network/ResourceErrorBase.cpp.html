38inline const ResourceError& ResourceErrorBase::asResourceError() const
39{
40 return *static_cast<const ResourceError*>(this);
41}
42
43ResourceError ResourceErrorBase::isolatedCopy() const
44{
45 lazyInit();
46
47 ResourceError errorCopy;
48 errorCopy.m_domain = m_domain.isolatedCopy();
49 errorCopy.m_errorCode = m_errorCode;
50 errorCopy.m_failingURL = m_failingURL.isolatedCopy();
51 errorCopy.m_localizedDescription = m_localizedDescription.isolatedCopy();
52 errorCopy.m_type = m_type;
53
54 errorCopy.doPlatformIsolatedCopy(asResourceError());
55
56 return errorCopy;
57}
58
59void ResourceErrorBase::lazyInit() const
60{
61 const_cast<ResourceError*>(static_cast<const ResourceError*>(this))->platformLazyInit();
62}
63
64void ResourceErrorBase::setType(Type type)
65{
66 // setType should only be used to specialize the error type.
67 ASSERT(m_type == type || m_type == Type::General || m_type == Type::Null || (m_type == Type::Cancellation && type == Type::AccessControl));
68 m_type = type;
69}
70
71bool ResourceErrorBase::compare(const ResourceError& a, const ResourceError& b)
72{
73 if (a.isNull() && b.isNull())
74 return true;
75
76 if (a.type() != b.type())
77 return false;
78
79 if (a.domain() != b.domain())
80 return false;
81
82 if (a.errorCode() != b.errorCode())
83 return false;
84
85 if (a.failingURL() != b.failingURL())
86 return false;
87
88 if (a.localizedDescription() != b.localizedDescription())
89 return false;
90
91 return ResourceError::platformCompare(a, b);
92}

94ResourceError internalError(const URL& url)
95{
96 RELEASE_LOG_ERROR(Loading, "Internal error called");
97 RELEASE_LOG_STACKTRACE(Loading);
98
99 return ResourceError("WebKitErrorDomain"_s, 300, url, WEB_UI_STRING("WebKit encountered an internal error", "WebKitErrorInternal description"));
100}
