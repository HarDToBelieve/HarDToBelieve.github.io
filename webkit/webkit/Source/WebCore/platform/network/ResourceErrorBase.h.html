27#pragma once
28
29#include <wtf/EnumTraits.h>
30#include <wtf/URL.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35class ResourceError;
36
37WEBCORE_EXPORT extern const char* const errorDomainWebKitInternal; // Used for errors that won't be exposed to clients.
38WEBCORE_EXPORT extern const char* const errorDomainWebKitServiceWorker; // Used for errors that happen when loading a resource from a service worker.
39
40class ResourceErrorBase {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 WEBCORE_EXPORT ResourceError isolatedCopy() const;
44
45 const String& domain() const { lazyInit(); return m_domain; }
46 int errorCode() const { lazyInit(); return m_errorCode; }
47 const URL& failingURL() const { lazyInit(); return m_failingURL; }
48 const String& localizedDescription() const { lazyInit(); return m_localizedDescription; }
49
50 enum class Type : uint8_t {
51 Null,
52 General,
53 AccessControl,
54 Cancellation,
55 Timeout
56 };
57
58 bool isNull() const { return m_type == Type::Null; }
59 bool isGeneral() const { return m_type == Type::General; }
60 bool isAccessControl() const { return m_type == Type::AccessControl; }
61 bool isCancellation() const { return m_type == Type::Cancellation; }
62 bool isTimeout() const { return m_type == Type::Timeout; }
63
64 static bool compare(const ResourceError&, const ResourceError&);
65
66 WEBCORE_EXPORT void setType(Type);
67 Type type() const { return m_type; }
68
69protected:
70 ResourceErrorBase(Type type) : m_type(type) { }
71
72 ResourceErrorBase(const String& domain, int errorCode, const URL& failingURL, const String& localizedDescription, Type type)
73 : m_domain(domain)
74 , m_failingURL(failingURL)
75 , m_localizedDescription(localizedDescription)
76 , m_errorCode(errorCode)
77 , m_type(type)
78 {
79 }
80
81 WEBCORE_EXPORT void lazyInit() const;
82
83 // The ResourceError subclass may "shadow" this method to lazily initialize platform specific fields
84 void platformLazyInit() {}
85
86 // The ResourceError subclass may "shadow" this method to compare platform specific fields
87 static bool platformCompare(const ResourceError&, const ResourceError&) { return true; }
88
89 String m_domain;
90 URL m_failingURL;
91 String m_localizedDescription;
92 int m_errorCode { 0 };
93 Type m_type { Type::General };
94
95private:
96 const ResourceError& asResourceError() const;
97};
98
99WEBCORE_EXPORT ResourceError internalError(const URL&);
100
101inline bool operator==(const ResourceError& a, const ResourceError& b) { return ResourceErrorBase::compare(a, b); }
102inline bool operator!=(const ResourceError& a, const ResourceError& b) { return !(a == b); }
103
104} // namespace WebCore

115 WebCore::ResourceErrorBase::Type::Timeout
