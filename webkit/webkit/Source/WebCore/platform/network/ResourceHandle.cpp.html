80ResourceHandle::ResourceHandle(NetworkingContext* context, const ResourceRequest& request, ResourceHandleClient* client, bool defersLoading, bool shouldContentSniff, bool shouldContentEncodingSniff, RefPtr<SecurityOrigin>&& sourceOrigin, bool isMainFrameNavigation)
81 : d(makeUnique<ResourceHandleInternal>(this, context, request, client, defersLoading, shouldContentSniff && shouldContentSniffURL(request.url()), shouldContentEncodingSniff, WTFMove(sourceOrigin), isMainFrameNavigation))
82{
83 if (!request.url().isValid()) {
84 scheduleFailure(InvalidURLFailure);
85 return;
86 }
87
88 if (!portAllowed(request.url())) {
89 scheduleFailure(BlockedFailure);
90 return;
91 }
92}

94RefPtr<ResourceHandle> ResourceHandle::create(NetworkingContext* context, const ResourceRequest& request, ResourceHandleClient* client, bool defersLoading, bool shouldContentSniff, bool shouldContentEncodingSniff, RefPtr<SecurityOrigin>&& sourceOrigin, bool isMainFrameNavigation)
95{
96 if (auto constructor = builtinResourceHandleConstructorMap().get(request.url().protocol().toStringWithoutCopying()))
97 return constructor(request, client);
98
99 auto newHandle = adoptRef(*new ResourceHandle(context, request, client, defersLoading, shouldContentSniff, shouldContentEncodingSniff, WTFMove(sourceOrigin), isMainFrameNavigation));
100
101 if (newHandle->d->m_scheduledFailureType != NoFailure)
102 return newHandle;
103
104 if (newHandle->start())
105 return newHandle;
106
107 return nullptr;
108}

115
116void ResourceHandle::failureTimerFired()
117{
118 if (!client())
119 return;
120
121 switch (d->m_scheduledFailureType) {
122 case NoFailure:
123 ASSERT_NOT_REACHED();
124 return;
125 case BlockedFailure:
126 d->m_scheduledFailureType = NoFailure;
127 client()->wasBlocked(this);
128 return;
129 case InvalidURLFailure:
130 d->m_scheduledFailureType = NoFailure;
131 client()->cannotShowURL(this);
132 return;
133 }
134
135 ASSERT_NOT_REACHED();
136}

158void ResourceHandle::didReceiveResponse(ResourceResponse&& response, CompletionHandler<void()>&& completionHandler)
159{
160 if (response.isHTTP09()) {
161 auto url = response.url();
162 std::optional<uint16_t> port = url.port();
163 if (port && !WTF::isDefaultPortForProtocol(port.value(), url.protocol())) {
164 cancel();
165 String message = "Cancelled load from '" + url.stringCenterEllipsizedToLength() + "' because it is using HTTP/0.9.";
166 d->m_client->didFail(this, { String(), 0, url, message });
167 completionHandler();
168 return;
169 }
170 }
171 client()->didReceiveResponseAsync(this, WTFMove(response), WTFMove(completionHandler));
172}

214void ResourceHandle::checkTAO(const ResourceResponse& response)
215{
216 if (d->m_failsTAOCheck)
217 return;
218
219 RefPtr<SecurityOrigin> origin;
220 if (d->m_isMainFrameNavigation)
221 origin = SecurityOrigin::create(firstRequest().url());
222 else
223 origin = d->m_sourceOrigin;
224
225 if (origin)
226 d->m_failsTAOCheck = !passesTimingAllowOriginCheck(response, *origin);
227}

274bool ResourceHandle::shouldContentSniffURL(const URL& url)
275{
276#if PLATFORM(COCOA)
277 if (shouldForceContentSniffing)
278 return true;
279#endif
280 // We shouldn't content sniff file URLs as their MIME type should be established via their extension.
281 return !url.protocolIs("file");
282}

289void ResourceHandle::setDefersLoading(bool defers)
290{
291 LOG(Network, "Handle %p setDefersLoading(%s)", this, defers ? "true" : "false");
292
293 ASSERT(d->m_defersLoading != defers); // Deferring is not counted, so calling setDefersLoading() repeatedly is likely to be in error.
294 d->m_defersLoading = defers;
295
296 if (defers) {
297 ASSERT(d->m_failureTimer.isActive() == (d->m_scheduledFailureType != NoFailure));
298 if (d->m_failureTimer.isActive())
299 d->m_failureTimer.stop();
300 } else if (d->m_scheduledFailureType != NoFailure) {
301 ASSERT(!d->m_failureTimer.isActive());
302 d->m_failureTimer.startOneShot(0_s);
303 }
304
305 platformSetDefersLoading(defers);
306}
