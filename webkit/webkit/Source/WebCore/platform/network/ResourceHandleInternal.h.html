57// The allocations and releases in ResourceHandleInternal are
58// Cocoa-exception-free (either simple Foundation classes or
59// WebCoreResourceLoaderImp which avoids doing work in dealloc).
60
61namespace WebCore {
62
63DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(ResourceHandleInternal);
64class ResourceHandleInternal {
65 WTF_MAKE_NONCOPYABLE(ResourceHandleInternal);
66 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(ResourceHandleInternal);
67public:
68 ResourceHandleInternal(ResourceHandle* loader, NetworkingContext* context, const ResourceRequest& request, ResourceHandleClient* client, bool defersLoading, bool shouldContentSniff, bool shouldContentEncodingSniff, RefPtr<SecurityOrigin>&& sourceOrigin, bool isMainFrameNavigation)
69 : m_context(context)
70 , m_client(client)
71 , m_firstRequest(request)
72 , m_lastHTTPMethod(request.httpMethod())
73 , m_partition(request.cachePartition())
74 , m_defersLoading(defersLoading)
75 , m_shouldContentSniff(shouldContentSniff)
76 , m_shouldContentEncodingSniff(shouldContentEncodingSniff)
77#if USE(CFURLCONNECTION)
78 , m_currentRequest(request)
79#endif
80 , m_failureTimer(*loader, &ResourceHandle::failureTimerFired)
81 , m_sourceOrigin(WTFMove(sourceOrigin))
82 , m_isMainFrameNavigation(isMainFrameNavigation)
83 {
84 const URL& url = m_firstRequest.url();
85 m_user = url.user();
86 m_password = url.password();
87 m_firstRequest.removeCredentials();
88 }
89
90 ~ResourceHandleInternal();
91
92 ResourceHandleClient* client() { return m_client; }
93
94 RefPtr<NetworkingContext> m_context;
95 ResourceHandleClient* m_client;
96 ResourceRequest m_firstRequest;
97 String m_lastHTTPMethod;
98 String m_partition;
99
100 // Suggested credentials for the current redirection step.
101 String m_user;
102 String m_password;
103
104 Credential m_initialCredential;
105
106 int status { 0 };
107
108 bool m_defersLoading;
109 bool m_shouldContentSniff;
110 bool m_shouldContentEncodingSniff;
111#if USE(CFURLCONNECTION)
112 RetainPtr<CFURLConnectionRef> m_connection;
113 ResourceRequest m_currentRequest;
114 RefPtr<ResourceHandleCFURLConnectionDelegate> m_connectionDelegate;
115#endif
116#if PLATFORM(COCOA)
117 RetainPtr<NSURLConnection> m_connection;
118 RetainPtr<id> m_delegate;
119#endif
120#if PLATFORM(COCOA)
121 bool m_startWhenScheduled { false };
122#endif
123#if PLATFORM(COCOA) || USE(CFURLCONNECTION)
124 RetainPtr<CFURLStorageSessionRef> m_storageSession;
125#endif
126#if USE(CURL)
127 std::unique_ptr<CurlResourceHandleDelegate> m_delegate;
128
129 bool m_cancelled { false };
130 unsigned m_authFailureCount { 0 };
131 bool m_addedCacheValidationHeaders { false };
132 RefPtr<CurlRequest> m_curlRequest;
133 RefPtr<SynchronousLoaderMessageQueue> m_messageQueue;
134#endif
135 Box<NetworkLoadMetrics> m_networkLoadMetrics;
136 MonotonicTime m_startTime;
137 uint16_t m_redirectCount { 0 };
138 bool m_failsTAOCheck { false };
139 bool m_hasCrossOriginRedirect { false };
140 bool m_isCrossOrigin { false };
141
142#if PLATFORM(COCOA)
143 // We need to keep a reference to the original challenge to be able to cancel it.
144 // It is almost identical to m_currentWebChallenge.nsURLAuthenticationChallenge(), but has a different sender.
145 NSURLAuthenticationChallenge *m_currentMacChallenge { nil };
146#endif
147
148 AuthenticationChallenge m_currentWebChallenge;
149 ResourceHandle::FailureType m_scheduledFailureType { ResourceHandle::NoFailure };
150 Timer m_failureTimer;
151 RefPtr<SecurityOrigin> m_sourceOrigin;
152 bool m_isMainFrameNavigation { false };
153};
