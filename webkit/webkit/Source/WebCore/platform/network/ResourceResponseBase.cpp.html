43bool isScriptAllowedByNosniff(const ResourceResponse& response)
44{
45 if (parseContentTypeOptionsHeader(response.httpHeaderField(HTTPHeaderName::XContentTypeOptions)) != ContentTypeOptionsDisposition::Nosniff)
46 return true;
47 String mimeType = extractMIMETypeFromMediaType(response.httpHeaderField(HTTPHeaderName::ContentType));
48 return MIMETypeRegistry::isSupportedJavaScriptMIMEType(mimeType);
49}

114ResourceResponse ResourceResponseBase::fromCrossThreadData(CrossThreadData&& data)
115{
116 ResourceResponse response;
117
118 response.setURL(data.url);
119 response.setMimeType(data.mimeType);
120 response.setExpectedContentLength(data.expectedContentLength);
121 response.setTextEncodingName(data.textEncodingName);
122
123 response.setHTTPStatusCode(data.httpStatusCode);
124 response.setHTTPStatusText(data.httpStatusText);
125 response.setHTTPVersion(data.httpVersion);
126
127 response.m_httpHeaderFields = WTFMove(data.httpHeaderFields);
128 if (data.networkLoadMetrics)

132 response.m_type = data.type;
133 response.m_tainting = data.tainting;
134 response.m_isRedirected = data.isRedirected;
135 response.m_isRangeRequested = data.isRangeRequested;
136
137 return response;

163ResourceResponse ResourceResponseBase::filter(const ResourceResponse& response, PerformExposeAllHeadersCheck performCheck)
164{
165 if (response.tainting() == Tainting::Opaque) {
166 ResourceResponse opaqueResponse;
167 opaqueResponse.setTainting(Tainting::Opaque);
168 opaqueResponse.setType(Type::Opaque);
169 return opaqueResponse;
170 }
171
172 if (response.tainting() == Tainting::Opaqueredirect) {
173 ResourceResponse opaqueResponse;
174 opaqueResponse.setTainting(Tainting::Opaqueredirect);
175 opaqueResponse.setType(Type::Opaqueredirect);
176 opaqueResponse.setURL(response.url());
177 return opaqueResponse;
178 }
179
180 ResourceResponse filteredResponse = response;
181 // Let's initialize filteredResponse to remove some header fields.
182 filteredResponse.lazyInit(AllFields);
183
184 filteredResponse.m_httpHeaderFields.remove(HTTPHeaderName::SetCookie);
185 filteredResponse.m_httpHeaderFields.remove(HTTPHeaderName::SetCookie2);
186
187 if (response.tainting() == Tainting::Basic) {
188 filteredResponse.setType(Type::Basic);
189 return filteredResponse;
190 }
191
192 ASSERT(response.tainting() == Tainting::Cors);
193 filteredResponse.setType(Type::Cors);
194
195 auto accessControlExposeHeaderSet = parseAccessControlAllowList<ASCIICaseInsensitiveHash>(response.httpHeaderField(HTTPHeaderName::AccessControlExposeHeaders)).value_or(HashSet<String, ASCIICaseInsensitiveHash> { });
196 if (performCheck == PerformExposeAllHeadersCheck::Yes && accessControlExposeHeaderSet.contains("*"))
197 return filteredResponse;
198
199 filteredResponse.m_httpHeaderFields.uncommonHeaders().removeAllMatching([&](auto& entry) {
200 return !isCrossOriginSafeHeader(entry.key, accessControlExposeHeaderSet);
201 });
202 filteredResponse.m_httpHeaderFields.commonHeaders().removeAllMatching([&](auto& entry) {
203 return !isCrossOriginSafeHeader(entry.key, accessControlExposeHeaderSet);
204 });
205
206 return filteredResponse;
207}

305String ResourceResponseBase::sanitizeSuggestedFilename(const String& suggestedFilename)
306{
307 if (suggestedFilename.isEmpty())
308 return suggestedFilename;
309
310 ResourceResponse response { { { }, "http://example.com/"_s }, { }, -1, { } };
311 response.setHTTPStatusCode(200);
312 String escapedSuggestedFilename = String(suggestedFilename).replace('\\', "\\\\").replace('"', "\\\"");
313 response.setHTTPHeaderField(HTTPHeaderName::ContentDisposition, makeString("attachment; filename=\"", escapedSuggestedFilename, '"'));
314 return response.suggestedFilename();
315}

383 || name == HTTPHeaderName::CacheControl
384 || name == HTTPHeaderName::Date
385 || name == HTTPHeaderName::Expires
386 || name == HTTPHeaderName::ETag
387 || name == HTTPHeaderName::LastModified
388 || name == HTTPHeaderName::Age
389 || name == HTTPHeaderName::Pragma
390 || name == HTTPHeaderName::ReferrerPolicy
391 || name == HTTPHeaderName::Refresh
392 || name == HTTPHeaderName::Vary
393 || name == HTTPHeaderName::AccessControlAllowCredentials
394 || name == HTTPHeaderName::AccessControlAllowHeaders
395 || name == HTTPHeaderName::AccessControlAllowMethods
396 || name == HTTPHeaderName::AccessControlAllowOrigin
397 || name == HTTPHeaderName::AccessControlExposeHeaders
398 || name == HTTPHeaderName::AccessControlMaxAge

451void ResourceResponseBase::sanitizeHTTPHeaderFieldsAccordingToTainting()
452{
453 switch (m_tainting) {
454 case ResourceResponse::Tainting::Basic:
455 break;
456 case ResourceResponse::Tainting::Cors: {
457 auto corsSafeHeaderSet = parseAccessControlAllowList<ASCIICaseInsensitiveHash>(httpHeaderField(HTTPHeaderName::AccessControlExposeHeaders)).value_or(HashSet<String, ASCIICaseInsensitiveHash> { });
458 if (corsSafeHeaderSet.contains("*"_str))
459 return;
460
461 m_httpHeaderFields.commonHeaders().removeAllMatching([&corsSafeHeaderSet](auto& header) {
462 return !isSafeCrossOriginResponseHeader(header.key) && !corsSafeHeaderSet.contains(httpHeaderNameString(header.key).toStringWithoutCopying());
463 });
464 m_httpHeaderFields.uncommonHeaders().removeAllMatching([&corsSafeHeaderSet](auto& header) { return !corsSafeHeaderSet.contains(header.key); });
465 break;
466 }
467 case ResourceResponse::Tainting::Opaque:
468 case ResourceResponse::Tainting::Opaqueredirect:
469 m_httpHeaderFields.commonHeaders().removeAllMatching([](auto& header) { return !isSafeCrossOriginResponseHeader(header.key); });
470 m_httpHeaderFields.uncommonHeaders().clear();
471 break;
472 }
473}
474
475void ResourceResponseBase::sanitizeHTTPHeaderFields(SanitizationType type)
476{
477 lazyInit(AllFields);
478
479 m_httpHeaderFields.remove(HTTPHeaderName::SetCookie);
480 m_httpHeaderFields.remove(HTTPHeaderName::SetCookie2);
481
482 switch (type) {
483 case SanitizationType::RemoveCookies:
484 return;
485 case SanitizationType::Redirection: {
486 m_httpHeaderFields.commonHeaders().removeAllMatching([](auto& header) { return !isSafeRedirectionResponseHeader(header.key); });
487 m_httpHeaderFields.uncommonHeaders().clear();
488 return;
489 }
490 case SanitizationType::CrossOriginSafe:
491 sanitizeHTTPHeaderFieldsAccordingToTainting();
492 }
493}

678static std::optional<WallTime> parseDateValueInHeader(const HTTPHeaderMap& headers, HTTPHeaderName headerName)
679{
680 String headerValue = headers.get(headerName);
681 if (headerValue.isEmpty())
682 return std::nullopt;
683 // This handles all date formats required by RFC2616:
684 // Sun, 06 Nov 1994 08:49:37 GMT ; RFC 822, updated by RFC 1123
685 // Sunday, 06-Nov-94 08:49:37 GMT ; RFC 850, obsoleted by RFC 1036
686 // Sun Nov 6 08:49:37 1994 ; ANSI C's asctime() format
687 return parseHTTPDate(headerValue);
688}

702{

727std::optional<WallTime> ResourceResponseBase::lastModified() const
728{
729 lazyInit(CommonFieldsOnly);
730
731 if (!m_haveParsedLastModifiedHeader) {
732 m_lastModified = parseDateValueInHeader(m_httpHeaderFields, HTTPHeaderName::LastModified);
733#if PLATFORM(COCOA)
734 // CFNetwork converts malformed dates into Epoch so we need to treat Epoch as
735 // an invalid value (rdar://problem/22352838).
736 const WallTime epoch = WallTime::fromRawSeconds(0);
737 if (m_lastModified && m_lastModified.value() == epoch)
738 m_lastModified = std::nullopt;
739#endif
740 m_haveParsedLastModifiedHeader = true;
741 }
742 return m_lastModified;
743}
