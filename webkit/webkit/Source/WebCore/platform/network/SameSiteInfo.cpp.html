34SameSiteInfo SameSiteInfo::create(const ResourceRequest& request, IsForDOMCookieAccess isForDOMAccess)
35{
36 // SameSite=strict cookies should be returned in document.cookie.
37 // See https://github.com/httpwg/http-extensions/issues/769
38 // and https://github.com/httpwg/http-extensions/pull/1428/files.
39 auto isSameSite = request.isSameSite() || (isForDOMAccess == IsForDOMCookieAccess::Yes && request.isTopSite());
40 return { isSameSite, request.isTopSite(), isSafeMethod(request.httpMethod()) };
41}
