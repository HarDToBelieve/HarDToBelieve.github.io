32#pragma once
33
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38class SocketStreamError {
39public:
40 SocketStreamError() = default;
41
42 explicit SocketStreamError(int errorCode)
43 : m_errorCode(errorCode)
44 , m_isNull(false)
45 {
46 }
47
48 SocketStreamError(int errorCode, const String& failingURL, const String& localizedDescription)
49 : m_errorCode(errorCode)
50 , m_failingURL(failingURL)
51 , m_localizedDescription(localizedDescription)
52 , m_isNull(false)
53 {
54 }
55
56 bool isNull() const { return m_isNull; }
57 int errorCode() const { return m_errorCode; }
58 const String& failingURL() const { return m_failingURL; }
59 const String& localizedDescription() const { return m_localizedDescription; }
60
61 template<class Encoder> void encode(Encoder&) const;
62 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, SocketStreamError&);
63
64private:
65 int m_errorCode { 0 };
66 String m_failingURL;
67 String m_localizedDescription;
68 bool m_isNull { true };
69};
70
71template<class Encoder>
72void SocketStreamError::encode(Encoder& encoder) const
73{
74 encoder << m_isNull;
75 if (m_isNull)
76 return;
77 encoder << m_errorCode;
78 encoder << m_failingURL;
79 encoder << m_localizedDescription;
80}
81
82template<class Decoder>
83bool SocketStreamError::decode(Decoder& decoder, SocketStreamError& error)
84{
85 if (!decoder.decode(error.m_isNull))
86 return false;
87 if (error.m_isNull)
88 return true;
89 if (!decoder.decode(error.m_errorCode))
90 return false;
91 if (!decoder.decode(error.m_failingURL))
92 return false;
93 if (!decoder.decode(error.m_localizedDescription))
94 return false;
95 return true;
96}
97
98}
