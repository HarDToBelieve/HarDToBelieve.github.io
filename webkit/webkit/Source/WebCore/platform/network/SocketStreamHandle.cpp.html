40SocketStreamHandle::SocketStreamHandle(const URL& url, SocketStreamHandleClient& client)
41 : m_url(url)
42 , m_client(client)
43 , m_state(Connecting)
44{
45 ASSERT(isMainThread());
46}
47
48SocketStreamHandle::SocketStreamState SocketStreamHandle::state() const
49{
50 return m_state;
51}
52
53void SocketStreamHandle::sendData(const uint8_t* data, size_t length, Function<void(bool)> completionHandler)
54{
55 if (m_state == Connecting || m_state == Closing)
56 return completionHandler(false);
57 platformSend(data, length, WTFMove(completionHandler));
58}
59
60void SocketStreamHandle::sendHandshake(CString&& handshake, std::optional<CookieRequestHeaderFieldProxy>&& headerFieldProxy, Function<void(bool, bool)> completionHandler)
61{
62 if (m_state == Connecting || m_state == Closing)
63 return completionHandler(false, false);
64 platformSendHandshake(handshake.dataAsUInt8Ptr(), handshake.length(), WTFMove(headerFieldProxy), WTFMove(completionHandler));
65}
66
67void SocketStreamHandle::close()
68{
69 if (m_state == Closed)
70 return;
71 m_state = Closing;
72 if (bufferedAmount())
73 return;
74 disconnect();
75}
76
77void SocketStreamHandle::disconnect()
78{
79 auto protect = makeRef(static_cast<SocketStreamHandle&>(*this)); // platformClose calls the client, which may make the handle get deallocated immediately.
80
81 platformClose();
82 m_state = Closed;
83}
