26#pragma once
27
28#include "AuthenticationChallengeBase.h"
29#include "AuthenticationClient.h"
30#include <wtf/RefPtr.h>
31
32#if USE(CFURLCONNECTION)
33typedef struct _CFURLAuthChallenge* CFURLAuthChallengeRef;
34#else
35OBJC_CLASS NSURLAuthenticationChallenge;
36#endif
37
38namespace WebCore {
39
40class AuthenticationChallenge : public AuthenticationChallengeBase {
41public:
42 AuthenticationChallenge() {}
43 WEBCORE_EXPORT AuthenticationChallenge(const ProtectionSpace&, const Credential& proposedCredential, unsigned previousFailureCount, const ResourceResponse&, const ResourceError&);
44
45#if USE(CFURLCONNECTION)
46 AuthenticationChallenge(CFURLAuthChallengeRef, AuthenticationClient*);
47
48 CFURLAuthChallengeRef cfURLAuthChallengeRef() const { return m_cfChallenge.get(); }
49#else
50 WEBCORE_EXPORT AuthenticationChallenge(NSURLAuthenticationChallenge *);
51
52 id sender() const { return m_sender.get(); }
53 NSURLAuthenticationChallenge *nsURLAuthenticationChallenge() const { return m_nsChallenge.get(); }
54#endif
55
56 WEBCORE_EXPORT void setAuthenticationClient(AuthenticationClient*); // Changes sender to one that invokes client methods.
57 WEBCORE_EXPORT AuthenticationClient* authenticationClient() const;
58
59private:
60 friend class AuthenticationChallengeBase;
61 static bool platformCompare(const AuthenticationChallenge& a, const AuthenticationChallenge& b);
62
63 // Platform challenge may be null. If it's non-null, it's always up to date with other fields.
64#if USE(CFURLCONNECTION)
65 RefPtr<AuthenticationClient> m_authenticationClient;
66 RetainPtr<CFURLAuthChallengeRef> m_cfChallenge;
67#else
68 RetainPtr<id> m_sender;
69 RetainPtr<NSURLAuthenticationChallenge> m_nsChallenge;
70#endif
71};
72
73}
