42RetainPtr<CFURLStorageSessionRef> NetworkStorageSession::createCFStorageSessionForIdentifier(CFStringRef identifier)
43{
44 auto storageSession = adoptCF(_CFURLStorageSessionCreate(kCFAllocatorDefault, identifier, nullptr));
45
46 if (!storageSession)
47 return nullptr;
48
49 auto cache = adoptCF(_CFURLStorageSessionCopyCache(kCFAllocatorDefault, storageSession.get()));
50 if (!cache)
51 return nullptr;
52
53 CFURLCacheSetDiskCapacity(cache.get(), 0);
54
55 auto sharedCache = adoptCF(CFURLCacheCopySharedURLCache());
56 CFURLCacheSetMemoryCapacity(cache.get(), CFURLCacheMemoryCapacity(sharedCache.get()));
57
58 if (!NetworkStorageSession::processMayUseCookieAPI())
59 return storageSession;
60
61 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
62
63 auto cookieStorage = adoptCF(_CFURLStorageSessionCopyCookieStorage(kCFAllocatorDefault, storageSession.get()));
64 if (!cookieStorage)
65 return nullptr;
66
67 auto sharedCookieStorage = _CFHTTPCookieStorageGetDefault(kCFAllocatorDefault);
68 auto sharedPolicy = CFHTTPCookieStorageGetCookieAcceptPolicy(sharedCookieStorage);
69 CFHTTPCookieStorageSetCookieAcceptPolicy(cookieStorage.get(), sharedPolicy);
70
71 return storageSession;
72}

74NetworkStorageSession::NetworkStorageSession(PAL::SessionID sessionID, RetainPtr<CFURLStorageSessionRef>&& platformSession, RetainPtr<CFHTTPCookieStorageRef>&& platformCookieStorage, IsInMemoryCookieStore isInMemoryCookieStore)
75 : m_sessionID(sessionID)
76 , m_platformSession(WTFMove(platformSession))
77 , m_isInMemoryCookieStore(isInMemoryCookieStore == IsInMemoryCookieStore::Yes)
78{
79 ASSERT(processMayUseCookieAPI() || !platformCookieStorage || m_isInMemoryCookieStore);
80 m_platformCookieStorage = platformCookieStorage ? WTFMove(platformCookieStorage) : cookieStorage();
81}

88RetainPtr<CFHTTPCookieStorageRef> NetworkStorageSession::cookieStorage() const
89{
90 if (!processMayUseCookieAPI() && !m_isInMemoryCookieStore)
91 return nullptr;
92
93 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies) || m_isInMemoryCookieStore);
94
95 if (m_platformCookieStorage)
96 return m_platformCookieStorage;
97
98 if (m_platformSession)
99 return adoptCF(_CFURLStorageSessionCopyCookieStorage(kCFAllocatorDefault, m_platformSession.get()));
100
101#if USE(CFURLCONNECTION)
102 return _CFHTTPCookieStorageGetDefault(kCFAllocatorDefault);
103#else
104 // When using NSURLConnection, we also use its shared cookie storage.
105 return nullptr;
106#endif
107}
