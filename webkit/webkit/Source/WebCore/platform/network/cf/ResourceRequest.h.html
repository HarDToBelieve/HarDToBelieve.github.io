29#include "ResourceRequestBase.h"
30#include <wtf/RetainPtr.h>

32OBJC_CLASS NSCachedURLResponse;
33OBJC_CLASS NSURLRequest;
34
35#if PLATFORM(COCOA) || USE(CFURLCONNECTION)
36typedef const struct _CFURLRequest* CFURLRequestRef;
37typedef const struct __CFURLStorageSession* CFURLStorageSessionRef;
38#endif
39
40namespace WebCore {
41
42class ResourceRequest : public ResourceRequestBase {
43public:
44 explicit ResourceRequest(const String& url)
45 : ResourceRequestBase(URL({ }, url), ResourceRequestCachePolicy::UseProtocolCachePolicy)
46 {
47 }
48
49 ResourceRequest(const URL& url)
50 : ResourceRequestBase(url, ResourceRequestCachePolicy::UseProtocolCachePolicy)
51 {
52 }
53
54 ResourceRequest(const URL& url, const String& referrer, ResourceRequestCachePolicy policy = ResourceRequestCachePolicy::UseProtocolCachePolicy)
55 : ResourceRequestBase(url, policy)
56 {
57 setHTTPReferrer(referrer);
58 }
59
60 ResourceRequest()
61 : ResourceRequestBase(URL(), ResourceRequestCachePolicy::UseProtocolCachePolicy)
62 {
63 }
64
65#if USE(CFURLCONNECTION)
66 ResourceRequest(CFURLRequestRef cfRequest)
67 : ResourceRequestBase()
68 , m_cfRequest(cfRequest)
69 {
70 }
71#else
72 ResourceRequest(NSURLRequest *nsRequest)
73 : ResourceRequestBase()
74 , m_nsRequest(nsRequest)
75 {
76 }
77#endif
78
79 WEBCORE_EXPORT void updateFromDelegatePreservingOldProperties(const ResourceRequest&);
80
81#if PLATFORM(COCOA)
82 bool encodingRequiresPlatformData() const { return m_httpBody || m_nsRequest; }
83 WEBCORE_EXPORT NSURLRequest *nsURLRequest(HTTPBodyUpdatePolicy) const;
84
85 WEBCORE_EXPORT static CFStringRef isUserInitiatedKey();
86#endif
87
88#if PLATFORM(COCOA) || USE(CFURLCONNECTION)
89 WEBCORE_EXPORT CFURLRequestRef cfURLRequest(HTTPBodyUpdatePolicy) const;
90 void setStorageSession(CFURLStorageSessionRef);
91#endif
92
93 WEBCORE_EXPORT static bool httpPipeliningEnabled();
94 WEBCORE_EXPORT static void setHTTPPipeliningEnabled(bool);
95
96 static bool resourcePrioritiesEnabled();
97
98 WEBCORE_EXPORT void replacePlatformRequest(HTTPBodyUpdatePolicy);
99
100private:
101 friend class ResourceRequestBase;
102
103 void doUpdatePlatformRequest();
104 void doUpdateResourceRequest();
105 void doUpdatePlatformHTTPBody();
106 void doUpdateResourceHTTPBody();
107
108 void doPlatformSetAsIsolatedCopy(const ResourceRequest&);
109
110#if USE(CFURLCONNECTION)
111 RetainPtr<CFURLRequestRef> m_cfRequest;
112#endif
113#if PLATFORM(COCOA)
114 RetainPtr<NSURLRequest> m_nsRequest;
115#endif
116
117 static bool s_httpPipeliningEnabled;
118};
119
120inline bool ResourceRequest::resourcePrioritiesEnabled()
121{
122#if PLATFORM(MAC)
123 return true;
124#elif PLATFORM(IOS_FAMILY)
125 return true;
126#elif PLATFORM(WIN)
127 return false;
128#endif
129}

134#endif
