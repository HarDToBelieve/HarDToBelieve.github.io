110static inline void setHeaderFields(CFMutableURLRequestRef request, const HTTPHeaderMap& requestHeaders)
111{
112 // Remove existing headers first, as some of them may no longer be present in the map.
113 RetainPtr<CFDictionaryRef> oldHeaderFields = adoptCF(CFURLRequestCopyAllHTTPHeaderFields(request));
114 CFIndex oldHeaderFieldCount = CFDictionaryGetCount(oldHeaderFields.get());
115 if (oldHeaderFieldCount) {
116 Vector<CFStringRef> oldHeaderFieldNames(oldHeaderFieldCount);
117 CFDictionaryGetKeysAndValues(oldHeaderFields.get(), reinterpret_cast<const void**>(&oldHeaderFieldNames[0]), 0);
118 for (CFIndex i = 0; i < oldHeaderFieldCount; ++i)
119 CFURLRequestSetHTTPHeaderFieldValue(request, oldHeaderFieldNames[i], 0);
120 }
121
122 for (const auto& header : requestHeaders)
123 CFURLRequestSetHTTPHeaderFieldValue(request, header.key.createCFString().get(), httpHeaderValueUsingSuitableEncoding(header).get());
124}

126static inline CFURLRequestCachePolicy toPlatformRequestCachePolicy(ResourceRequestCachePolicy policy)
127{
128 switch (policy) {
129 case ResourceRequestCachePolicy::UseProtocolCachePolicy:
130 return kCFURLRequestCachePolicyProtocolDefault;
131 case ResourceRequestCachePolicy::ReturnCacheDataElseLoad:
132 return kCFURLRequestCachePolicyReturnCacheDataElseLoad;
133 case ResourceRequestCachePolicy::ReturnCacheDataDontLoad:
134 return kCFURLRequestCachePolicyReturnCacheDataDontLoad;
135 case ResourceRequestCachePolicy::ReloadIgnoringCacheData:
136 case ResourceRequestCachePolicy::DoNotUseAnyCache:
137 case ResourceRequestCachePolicy::RefreshAnyCacheData:
138 return kCFURLRequestCachePolicyReloadIgnoringCache;
139 }
140
141 ASSERT_NOT_REACHED();
142 return kCFURLRequestCachePolicyReloadIgnoringCache;
143}
144
145static inline ResourceRequestCachePolicy fromPlatformRequestCachePolicy(CFURLRequestCachePolicy policy)
146{
147 switch (policy) {
148 case kCFURLRequestCachePolicyProtocolDefault:
149 return ResourceRequestCachePolicy::UseProtocolCachePolicy;
150 case kCFURLRequestCachePolicyReloadIgnoringCache:
151 return ResourceRequestCachePolicy::ReloadIgnoringCacheData;
152 case kCFURLRequestCachePolicyReturnCacheDataElseLoad:
153 return ResourceRequestCachePolicy::ReturnCacheDataElseLoad;
154 case kCFURLRequestCachePolicyReturnCacheDataDontLoad:
155 return ResourceRequestCachePolicy::ReturnCacheDataDontLoad;
156 default:
157 return ResourceRequestCachePolicy::ReloadIgnoringCacheData;
158 }
159}

176void ResourceRequest::doUpdatePlatformRequest()
177{
178 RetainPtr<CFMutableURLRequestRef> cfRequest;
179
180 RetainPtr<CFURLRef> url = ResourceRequest::url().createCFURL();
181 RetainPtr<CFURLRef> firstPartyForCookies = ResourceRequest::firstPartyForCookies().createCFURL();
182 double timeoutInterval = ResourceRequestBase::timeoutInterval() ? ResourceRequestBase::timeoutInterval() : ResourceRequestBase::defaultTimeoutInterval();
183 if (m_cfRequest) {
184 cfRequest = adoptCF(CFURLRequestCreateMutableCopy(0, m_cfRequest.get()));
185 CFURLRequestSetURL(cfRequest.get(), url.get());
186 CFURLRequestSetMainDocumentURL(cfRequest.get(), firstPartyForCookies.get());
187 CFURLRequestSetCachePolicy(cfRequest.get(), toPlatformRequestCachePolicy(cachePolicy()));
188 CFURLRequestSetTimeoutInterval(cfRequest.get(), timeoutInterval);
189 } else
190 cfRequest = adoptCF(CFURLRequestCreateMutable(0, url.get(), toPlatformRequestCachePolicy(cachePolicy()), timeoutInterval, firstPartyForCookies.get()));
191
192 CFURLRequestSetHTTPRequestMethod(cfRequest.get(), httpMethod().createCFString().get());
193
194 if (httpPipeliningEnabled())
195 CFURLRequestSetShouldPipelineHTTP(cfRequest.get(), true, true);
196
197 if (resourcePrioritiesEnabled()) {
198 CFURLRequestSetRequestPriority(cfRequest.get(), toPlatformRequestPriority(priority()));
199
200 // Used by PLT to ignore very low priority beacon and ping loads.
201 if (priority() == ResourceLoadPriority::VeryLow)
202 _CFURLRequestSetProtocolProperty(cfRequest.get(), CFSTR("WKVeryLowLoadPriority"), kCFBooleanTrue);
203 }
204
205 setHeaderFields(cfRequest.get(), httpHeaderFields());
206
207 CFURLRequestSetShouldHandleHTTPCookies(cfRequest.get(), allowCookies());
208
209#if PLATFORM(IOS_FAMILY)
210 _CFURLRequestSetProtocolProperty(cfRequest.get(), CFSTR("_kCFHTTPCookiePolicyPropertySiteForCookies"), siteForCookies(m_sameSiteDisposition, url.get()));
211
212 int isTopSite = m_isTopSite;
213 RetainPtr<CFNumberRef> isTopSiteCF = adoptCF(CFNumberCreate(nullptr, kCFNumberIntType, &isTopSite));
214 _CFURLRequestSetProtocolProperty(cfRequest.get(), CFSTR("_kCFHTTPCookiePolicyPropertyisTopSite"), isTopSiteCF.get());
215#endif
216
217 unsigned fallbackCount = m_responseContentDispositionEncodingFallbackArray.size();
218 RetainPtr<CFMutableArrayRef> encodingFallbacks = adoptCF(CFArrayCreateMutable(kCFAllocatorDefault, fallbackCount, 0));
219 for (unsigned i = 0; i != fallbackCount; ++i) {
220 RetainPtr<CFStringRef> encodingName = m_responseContentDispositionEncodingFallbackArray[i].createCFString();
221 CFStringEncoding encoding = CFStringConvertIANACharSetNameToEncoding(encodingName.get());
222 if (encoding != kCFStringEncodingInvalidId)
223 CFArrayAppendValue(encodingFallbacks.get(), reinterpret_cast<const void*>(encoding));
224 }
225 setContentDispositionEncodingFallbackArray(cfRequest.get(), encodingFallbacks.get());
226
227#if ENABLE(CACHE_PARTITIONING)
228 String partition = cachePartition();
229 if (!partition.isNull() && !partition.isEmpty()) {
230 CString utf8String = partition.utf8();
231 RetainPtr<CFStringRef> partitionValue = adoptCF(CFStringCreateWithBytes(0, utf8String.dataAsUInt8Ptr(), utf8String.length(), kCFStringEncodingUTF8, false));
232 _CFURLRequestSetProtocolProperty(cfRequest.get(), _kCFURLCachePartitionKey, partitionValue.get());
233 }
234#endif
235
236 m_cfRequest = WTFMove(cfRequest);
237}

239void ResourceRequest::doUpdatePlatformHTTPBody()
240{
241 RetainPtr<CFMutableURLRequestRef> cfRequest;
242
243 RetainPtr<CFURLRef> url = ResourceRequest::url().createCFURL();
244 RetainPtr<CFURLRef> firstPartyForCookies = ResourceRequest::firstPartyForCookies().createCFURL();
245 double timeoutInterval = ResourceRequestBase::timeoutInterval() ? ResourceRequestBase::timeoutInterval() : ResourceRequestBase::defaultTimeoutInterval();
246 if (m_cfRequest) {
247 cfRequest = adoptCF(CFURLRequestCreateMutableCopy(0, m_cfRequest.get()));
248 CFURLRequestSetURL(cfRequest.get(), url.get());
249 CFURLRequestSetMainDocumentURL(cfRequest.get(), firstPartyForCookies.get());
250 CFURLRequestSetCachePolicy(cfRequest.get(), toPlatformRequestCachePolicy(cachePolicy()));
251 CFURLRequestSetTimeoutInterval(cfRequest.get(), timeoutInterval);
252 } else
253 cfRequest = adoptCF(CFURLRequestCreateMutable(0, url.get(), toPlatformRequestCachePolicy(cachePolicy()), timeoutInterval, firstPartyForCookies.get()));
254
255 FormData* formData = httpBody();
256 if (formData && !formData->isEmpty())
257 WebCore::setHTTPBody(cfRequest.get(), formData);
258
259 if (RetainPtr<CFReadStreamRef> bodyStream = adoptCF(CFURLRequestCopyHTTPRequestBodyStream(cfRequest.get()))) {
260 // For streams, provide a Content-Length to avoid using chunked encoding, and to get accurate total length in callbacks.
261 if (RetainPtr<CFStringRef> lengthString = adoptCF(static_cast<CFStringRef>(CFReadStreamCopyProperty(bodyStream.get(), formDataStreamLengthPropertyName())))) {
262 CFURLRequestSetHTTPHeaderFieldValue(cfRequest.get(), CFSTR("Content-Length"), lengthString.get());
263 // Since resource request is already marked updated, we need to keep it up to date too.
264 ASSERT(m_resourceRequestUpdated);
265 m_httpHeaderFields.set(HTTPHeaderName::ContentLength, lengthString.get());
266 }
267 }
268
269 m_cfRequest = WTFMove(cfRequest);
270}

272void ResourceRequest::doUpdateResourceRequest()
273{
274 if (!m_cfRequest) {
275 *this = ResourceRequest();
276 return;
277 }
278
279 m_url = CFURLRequestGetURL(m_cfRequest.get());
280
281 if (m_cachePolicy == ResourceRequestCachePolicy::UseProtocolCachePolicy)
282 m_cachePolicy = fromPlatformRequestCachePolicy(CFURLRequestGetCachePolicy(m_cfRequest.get()));
283 m_timeoutInterval = CFURLRequestGetTimeoutInterval(m_cfRequest.get());
284 m_firstPartyForCookies = CFURLRequestGetMainDocumentURL(m_cfRequest.get());
285 if (auto method = adoptCF(CFURLRequestCopyHTTPRequestMethod(m_cfRequest.get())))
286 m_httpMethod = method.get();
287 m_allowCookies = CFURLRequestShouldHandleHTTPCookies(m_cfRequest.get());
288
289 if (resourcePrioritiesEnabled())
290 m_priority = toResourceLoadPriority(CFURLRequestGetRequestPriority(m_cfRequest.get()));
291
292#if PLATFORM(IOS_FAMILY)
293 RetainPtr<CFURLRef> siteForCookies = adoptCF(checked_cf_cast<CFURLRef>(_CFURLRequestCopyProtocolPropertyForKey(m_cfRequest.get(), CFSTR("_kCFHTTPCookiePolicyPropertySiteForCookies"))));
294 m_sameSiteDisposition = !siteForCookies ? SameSiteDisposition::Unspecified : (areRegistrableDomainsEqual(siteForCookies.get(), m_url) ? SameSiteDisposition::SameSite : SameSiteDisposition::CrossSite);
295
296 RetainPtr<CFNumberRef> isTopSiteCF = adoptCF(checked_cf_cast<CFNumber>(_CFURLRequestCopyProtocolPropertyForKey(m_cfRequest.get(), CFSTR("_kCFHTTPCookiePolicyPropertyisTopSite"))));
297 if (!isTopSiteCF)
298 m_isTopSite = false;
299 else {
300 int isTopSite = 0;
301 CFNumberGetValue(isTopSiteCF.get(), kCFNumberIntType, &isTopSite);
302 m_isTopSite = isTopSite;
303 }
304#endif
305
306 m_httpHeaderFields.clear();
307 if (auto headers = adoptCF(CFURLRequestCopyAllHTTPHeaderFields(m_cfRequest.get()))) {
308 CFIndex headerCount = CFDictionaryGetCount(headers.get());
309 Vector<const void*, 128> keys(headerCount);
310 Vector<const void*, 128> values(headerCount);
311 CFDictionaryGetKeysAndValues(headers.get(), keys.data(), values.data());
312 for (int i = 0; i < headerCount; ++i)
313 m_httpHeaderFields.set((CFStringRef)keys[i], (CFStringRef)values[i]);
314 }
315
316 m_responseContentDispositionEncodingFallbackArray.clear();
317 RetainPtr<CFArrayRef> encodingFallbacks = adoptCF(copyContentDispositionEncodingFallbackArray(m_cfRequest.get()));
318 if (encodingFallbacks) {
319 CFIndex count = CFArrayGetCount(encodingFallbacks.get());
320 for (CFIndex i = 0; i < count; ++i) {
321 CFStringEncoding encoding = reinterpret_cast<CFIndex>(CFArrayGetValueAtIndex(encodingFallbacks.get(), i));
322 if (encoding != kCFStringEncodingInvalidId)
323 m_responseContentDispositionEncodingFallbackArray.append(CFStringConvertEncodingToIANACharSetName(encoding));
324 }
325 }
326
327#if ENABLE(CACHE_PARTITIONING)
328 RetainPtr<CFStringRef> cachePartition = adoptCF(static_cast<CFStringRef>(_CFURLRequestCopyProtocolPropertyForKey(m_cfRequest.get(), _kCFURLCachePartitionKey)));
329 if (cachePartition)
330 m_cachePartition = cachePartition.get();
331#endif
332}

334void ResourceRequest::doUpdateResourceHTTPBody()
335{
336 if (!m_cfRequest) {
337 m_httpBody = nullptr;
338 return;
339 }
340
341 if (RetainPtr<CFDataRef> bodyData = adoptCF(CFURLRequestCopyHTTPRequestBody(m_cfRequest.get())))
342 m_httpBody = FormData::create(CFDataGetBytePtr(bodyData.get()), CFDataGetLength(bodyData.get()));
343 else if (RetainPtr<CFReadStreamRef> bodyStream = adoptCF(CFURLRequestCopyHTTPRequestBodyStream(m_cfRequest.get()))) {
344 FormData* formData = httpBodyFromStream(bodyStream.get());
345 // There is no FormData object if a client provided a custom data stream.
346 // We shouldn't be looking at http body after client callbacks.
347 ASSERT(formData);
348 if (formData)
349 m_httpBody = formData;
350 }
351}

366void ResourceRequest::updateFromDelegatePreservingOldProperties(const ResourceRequest& delegateProvidedRequest)
367{
368 // These are things we don't want willSendRequest delegate to mutate or reset.
369 ResourceLoadPriority oldPriority = priority();
370 RefPtr<FormData> oldHTTPBody = httpBody();
371 bool isHiddenFromInspector = hiddenFromInspector();
372 auto oldRequester = requester();
373 auto oldInitiatorIdentifier = initiatorIdentifier();
374 auto oldInspectorInitiatorNodeIdentifier = inspectorInitiatorNodeIdentifier();
375
376 *this = delegateProvidedRequest;
377
378 setPriority(oldPriority);
379 setHTTPBody(WTFMove(oldHTTPBody));
380 setHiddenFromInspector(isHiddenFromInspector);
381 setRequester(oldRequester);
382 setInitiatorIdentifier(oldInitiatorIdentifier);
383 if (oldInspectorInitiatorNodeIdentifier)
384 setInspectorInitiatorNodeIdentifier(*oldInspectorInitiatorNodeIdentifier);
385}

397// FIXME: It is confusing that this function both sets connection count and determines maximum request count at network layer. This can and should be done separately.
398unsigned initializeMaximumHTTPConnectionCountPerHost()
399{
400 static const unsigned preferredConnectionCount = 6;
401 static const unsigned unlimitedRequestCount = 10000;
402
403 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPLoadWidth, preferredConnectionCount);
404 unsigned maximumHTTPConnectionCountPerHost = _CFNetworkHTTPConnectionCacheGetLimit(kHTTPLoadWidth);
405
406 Boolean keyExistsAndHasValidFormat = false;
407 Boolean prefValue = CFPreferencesGetAppBooleanValue(CFSTR("WebKitEnableHTTPPipelining"), kCFPreferencesCurrentApplication, &keyExistsAndHasValidFormat);
408 if (keyExistsAndHasValidFormat)
409 ResourceRequest::setHTTPPipeliningEnabled(prefValue);
410
411 // Use WebCore scheduler when we can't use request priorities with CFNetwork.
412 if (!ResourceRequest::resourcePrioritiesEnabled())
413 return maximumHTTPConnectionCountPerHost;
414
415 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPPriorityNumLevels, resourceLoadPriorityCount);
416#if !PLATFORM(WIN)
417 // FIXME: <rdar://problem/9375609> Implement minimum fast lane priority setting on Windows
418 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPMinimumFastLanePriority, toPlatformRequestPriority(ResourceLoadPriority::Medium));
419#endif
420
421 return unlimitedRequestCount;
422}

425void initializeHTTPConnectionSettingsOnStartup()
426{
427 // This need to be called from WebKitInitialize so the calls happen early enough, before any requests are made. <rdar://problem/9691871>
428 // Desktop doesn't have early initialization so it is not clear how this should be done there. The CFNetwork SPI probably
429 // needs to become more forgiving.
430 // We can't read settings here as this is called too early for that. All values need to be constants.
431 static const unsigned preferredConnectionCount = 6;
432 static const unsigned fastLaneConnectionCount = 1;
433 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPLoadWidth, preferredConnectionCount);
434 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPPriorityNumLevels, resourceLoadPriorityCount);
435 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPMinimumFastLanePriority, toPlatformRequestPriority(ResourceLoadPriority::Medium));
436 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPNumFastLanes, fastLaneConnectionCount);
437}
