99SocketStreamHandleImpl::SocketStreamHandleImpl(const URL& url, SocketStreamHandleClient& client, PAL::SessionID sessionID, const String& credentialPartition, SourceApplicationAuditToken&& auditData, const StorageSessionProvider* provider)
100 : SocketStreamHandle(url, client)
101 , m_connectingSubstate(New)
102 , m_connectionType(Unknown)
103 , m_sentStoredCredentials(false)
104 , m_credentialPartition(credentialPartition)
105 , m_auditData(WTFMove(auditData))
106 , m_storageSessionProvider(provider)
107{
108 LOG(Network, "SocketStreamHandle %p new client %p", this, &m_client);
109
110 ASSERT(url.protocolIs("ws") || url.protocolIs("wss"));
111
112 URL httpsURL(URL(), "https://" + m_url.host());
113 m_httpsURL = httpsURL.createCFURL();
114
115#if PLATFORM(COCOA)
116 // Don't check for HSTS violation for ephemeral sessions since
117 // HSTS state should not transfer between regular and private browsing.
118 if (url.protocolIs("ws")
119 && !sessionID.isEphemeral()
120 && _CFNetworkIsKnownHSTSHostWithSession(m_httpsURL.get(), nullptr)) {
121 // Call this asynchronously because the socket stream is not fully constructed at this point.
122 callOnMainThread([this, protectedThis = makeRef(*this)] {
123 m_client.didFailSocketStream(*this, SocketStreamError(0, m_url.string(), "WebSocket connection failed because it violates HTTP Strict Transport Security."));
124 });
125 return;
126 }
127#endif
128
129 createStreams();
130 ASSERT(!m_readStream == !m_writeStream);
131 if (!m_readStream) // Doing asynchronous PAC file processing, streams will be created later.
132 return;
133
134 scheduleStreams();
135}

137void SocketStreamHandleImpl::scheduleStreams()
138{
139 ASSERT(m_readStream);
140 ASSERT(m_writeStream);
141
142 CFStreamClientContext clientContext = { 0, this, retainSocketStreamHandle, releaseSocketStreamHandle, copyCFStreamDescription };
143 // FIXME: Pass specific events we're interested in instead of -1.
144 CFReadStreamSetClient(m_readStream.get(), static_cast<CFOptionFlags>(-1), readStreamCallback, &clientContext);
145 CFWriteStreamSetClient(m_writeStream.get(), static_cast<CFOptionFlags>(-1), writeStreamCallback, &clientContext);
146
147 CFReadStreamScheduleWithRunLoop(m_readStream.get(), callbacksRunLoop(), callbacksRunLoopMode());
148 CFWriteStreamScheduleWithRunLoop(m_writeStream.get(), callbacksRunLoop(), callbacksRunLoopMode());
149
150 CFReadStreamOpen(m_readStream.get());
151 CFWriteStreamOpen(m_writeStream.get());
152
153 if (m_pacRunLoopSource)
154 removePACRunLoopSource();
155
156 m_connectingSubstate = WaitingForConnect;
157 RELEASE_LOG(Network, "SocketStreamHandleImpl::scheduleStreams - m_connectionSubState is WaitingForConnect");
158}

186void SocketStreamHandleImpl::pacExecutionCallback(void* client, CFArrayRef proxyList, CFErrorRef)
187{
188 SocketStreamHandleImpl* handle = static_cast<SocketStreamHandleImpl*>(client);
189
190 RefPtr<SocketStreamHandle> protector(handle);
191 callOnMainThreadAndWait([&] {
192 ASSERT(handle->m_connectingSubstate == ExecutingPACFile);
193 // This time, the array won't have PAC as a first entry.

196 handle->chooseProxyFromArray(proxyList);
197 handle->createStreams();
198 handle->scheduleStreams();
199 });

220void SocketStreamHandleImpl::chooseProxy()
221{
222 RetainPtr<CFDictionaryRef> proxyDictionary = adoptCF(CFNetworkCopySystemProxySettings());
223
224 // SOCKS or HTTPS (AKA CONNECT) proxies are supported.
225 // WebSocket protocol relies on handshake being transferred unchanged, so we need a proxy that will not modify headers.
226 // Since HTTP proxies must add Via headers, they are highly unlikely to work.
227 // Many CONNECT proxies limit connectivity to port 443, so we prefer SOCKS, if configured.
228
229 if (!proxyDictionary) {
230 m_connectionType = Direct;
231 return;
232 }
233
234 // CFNetworkCopyProxiesForURL doesn't know about WebSocket schemes, so pretend to use http.
235 // Always use "https" to get HTTPS proxies in result - we'll try to use those for ws:// even though many are configured to reject connections to ports other than 443.
236 RetainPtr<CFArrayRef> proxyArray = adoptCF(CFNetworkCopyProxiesForURL(m_httpsURL.get(), proxyDictionary.get()));
237
238 chooseProxyFromArray(proxyArray.get());
239}

241void SocketStreamHandleImpl::chooseProxyFromArray(CFArrayRef proxyArray)
242{
243 if (!proxyArray) {
244 m_connectionType = Direct;
245 return;
246 }
247
248 CFIndex proxyArrayCount = CFArrayGetCount(proxyArray);
249
250 // PAC is always the first entry, if present.
251 if (proxyArrayCount) {
252 CFDictionaryRef proxyInfo = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(proxyArray, 0));
253 CFTypeRef proxyType = CFDictionaryGetValue(proxyInfo, kCFProxyTypeKey);
254 if (proxyType && CFGetTypeID(proxyType) == CFStringGetTypeID()) {
255 if (CFEqual(proxyType, kCFProxyTypeAutoConfigurationURL)) {
256 CFTypeRef pacFileURL = CFDictionaryGetValue(proxyInfo, kCFProxyAutoConfigurationURLKey);
257 if (pacFileURL && CFGetTypeID(pacFileURL) == CFURLGetTypeID()) {
258 executePACFileURL(static_cast<CFURLRef>(pacFileURL));
259 return;
260 }
261 }
262 }
263 }
264
265 CFDictionaryRef chosenProxy = 0;
266 for (CFIndex i = 0; i < proxyArrayCount; ++i) {
267 CFDictionaryRef proxyInfo = static_cast<CFDictionaryRef>(CFArrayGetValueAtIndex(proxyArray, i));
268 CFTypeRef proxyType = CFDictionaryGetValue(proxyInfo, kCFProxyTypeKey);
269 if (proxyType && CFGetTypeID(proxyType) == CFStringGetTypeID()) {
270 if (CFEqual(proxyType, kCFProxyTypeSOCKS)) {
271 m_connectionType = SOCKSProxy;
272 chosenProxy = proxyInfo;
273 break;
274 }
275 if (CFEqual(proxyType, kCFProxyTypeHTTPS)) {
276 m_connectionType = CONNECTProxy;
277 chosenProxy = proxyInfo;
278 // Keep looking for proxies, as a SOCKS one is preferable.
279 }
280 }
281 }
282
283 if (chosenProxy) {
284 ASSERT(m_connectionType != Unknown);
285 ASSERT(m_connectionType != Direct);
286
287 CFTypeRef proxyHost = CFDictionaryGetValue(chosenProxy, kCFProxyHostNameKey);
288 CFTypeRef proxyPort = CFDictionaryGetValue(chosenProxy, kCFProxyPortNumberKey);
289
290 if (proxyHost && CFGetTypeID(proxyHost) == CFStringGetTypeID() && proxyPort && CFGetTypeID(proxyPort) == CFNumberGetTypeID()) {
291 m_proxyHost = static_cast<CFStringRef>(proxyHost);
292 m_proxyPort = static_cast<CFNumberRef>(proxyPort);
293 return;
294 }
295 }
296
297 m_connectionType = Direct;
298}

300static void setCONNECTProxyForStream(CFReadStreamRef stream, CFStringRef proxyHost, CFNumberRef proxyPort)
301{
302 const void* proxyKeys[] = { kCFStreamPropertyCONNECTProxyHost, kCFStreamPropertyCONNECTProxyPort };
303 const void* proxyValues[] = { proxyHost, proxyPort };
304 auto connectDictionary = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, proxyKeys, proxyValues, sizeof(proxyKeys) / sizeof(*proxyKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
305 CFReadStreamSetProperty(stream, kCFStreamPropertyCONNECTProxy, connectDictionary.get());
306}

315void SocketStreamHandleImpl::createStreams()
316{
317 if (m_connectionType == Unknown)
318 chooseProxy();
319
320 // If it's still unknown, then we're resolving a PAC file asynchronously.
321 if (m_connectionType == Unknown)
322 return;
323
324 RetainPtr<CFStringRef> host = m_url.host().createCFString();
325
326 // Creating streams to final destination, not to proxy.
327 CFReadStreamRef readStream = 0;
328 CFWriteStreamRef writeStream = 0;
329 CFStreamCreatePairWithSocketToHost(0, host.get(), port(), &readStream, &writeStream);
330#if PLATFORM(COCOA)
331 // <rdar://problem/12855587> _kCFStreamSocketSetNoDelay is not exported on Windows
332 CFWriteStreamSetProperty(writeStream, _kCFStreamSocketSetNoDelay, kCFBooleanTrue);
333 if (m_auditData.sourceApplicationAuditData && m_auditData.sourceApplicationAuditData.get()) {
334 CFReadStreamSetProperty(readStream, kCFStreamPropertySourceApplication, m_auditData.sourceApplicationAuditData.get());
335 CFWriteStreamSetProperty(writeStream, kCFStreamPropertySourceApplication, m_auditData.sourceApplicationAuditData.get());
336 }
337#endif
338
339 m_readStream = adoptCF(readStream);
340 m_writeStream = adoptCF(writeStream);
341
342 switch (m_connectionType) {
343 case Unknown:
344 ASSERT_NOT_REACHED();
345 break;
346 case Direct:
347 break;
348 case SOCKSProxy: {
349 // FIXME: SOCKS5 doesn't do challenge-response, should we try to apply credentials from Keychain right away?
350 // But SOCKS5 credentials don't work at the time of this writing anyway, see <rdar://6776698>.
351 const void* proxyKeys[] = { kCFStreamPropertySOCKSProxyHost, kCFStreamPropertySOCKSProxyPort };
352 const void* proxyValues[] = { m_proxyHost.get(), m_proxyPort.get() };
353 RetainPtr<CFDictionaryRef> connectDictionary = adoptCF(CFDictionaryCreate(0, proxyKeys, proxyValues, WTF_ARRAY_LENGTH(proxyKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
354 CFReadStreamSetProperty(m_readStream.get(), kCFStreamPropertySOCKSProxy, connectDictionary.get());
355 break;
356 }
357 case CONNECTProxy:
358 setCONNECTProxyForStream(m_readStream.get(), m_proxyHost.get(), m_proxyPort.get());
359 break;
360 }
361
362 if (shouldUseSSL()) {
363 CFBooleanRef validateCertificateChain = DeprecatedGlobalSettings::allowsAnySSLCertificate() ? kCFBooleanFalse : kCFBooleanTrue;
364 const void* keys[] = {
365 kCFStreamSSLPeerName,
366 kCFStreamSSLLevel,
367 kCFStreamSSLValidatesCertificateChain
368 };
369 const void* values[] = {
370 host.get(),
371#if PLATFORM(COCOA)
372 gLegacyTLSEnabled ? kCFStreamSocketSecurityLevelNegotiatedSSL : kCFStreamSocketSecurityLevelTLSv1_2,
373#else
374 kCFStreamSocketSecurityLevelNegotiatedSSL,
375#endif
376 validateCertificateChain
377 };
378 RetainPtr<CFDictionaryRef> settings = adoptCF(CFDictionaryCreate(0, keys, values, WTF_ARRAY_LENGTH(keys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
379 CFReadStreamSetProperty(m_readStream.get(), kCFStreamPropertySSLSettings, settings.get());
380 CFWriteStreamSetProperty(m_writeStream.get(), kCFStreamPropertySSLSettings, settings.get());
381 }
382}

384bool SocketStreamHandleImpl::getStoredCONNECTProxyCredentials(const ProtectionSpace& protectionSpace, String& login, String& password)
385{
386 // FIXME (<rdar://problem/10416495>): Proxy credentials should be retrieved from AuthBrokerAgent.
387
388 // Try system credential storage first, matching HTTP behavior (CFNetwork only asks the client for password if it couldn't find it in Keychain).
389 Credential storedCredential;
390 if (auto* storageSession = m_storageSessionProvider ? m_storageSessionProvider->storageSession() : nullptr) {
391 storedCredential = CredentialStorage::getFromPersistentStorage(protectionSpace);
392 if (storedCredential.isEmpty())
393 storedCredential = storageSession->credentialStorage().get(m_credentialPartition, protectionSpace);
394 }
395
396 if (storedCredential.isEmpty())
397 return false;
398
399 login = storedCredential.user();
400 password = storedCredential.password();
401
402 return true;
403}

405static ProtectionSpaceAuthenticationScheme authenticationSchemeFromAuthenticationMethod(CFStringRef method)
406{
407 if (CFEqual(method, kCFHTTPAuthenticationSchemeBasic))
408 return ProtectionSpaceAuthenticationSchemeHTTPBasic;
409 if (CFEqual(method, kCFHTTPAuthenticationSchemeDigest))
410 return ProtectionSpaceAuthenticationSchemeHTTPDigest;
411 if (CFEqual(method, kCFHTTPAuthenticationSchemeNTLM))
412 return ProtectionSpaceAuthenticationSchemeNTLM;
413 if (CFEqual(method, kCFHTTPAuthenticationSchemeNegotiate))
414 return ProtectionSpaceAuthenticationSchemeNegotiate;
415 ASSERT_NOT_REACHED();
416 return ProtectionSpaceAuthenticationSchemeUnknown;
417}

420{
421 auto originalCONNECTDictionary = adoptCF((CFDictionaryRef)CFReadStreamCopyProperty(stream, kCFStreamPropertyCONNECTProxy));
422 auto connectDictionary = adoptCF(CFDictionaryCreateMutableCopy(kCFAllocatorDefault, 0, originalCONNECTDictionary.get()));
423

425 const void* headerFieldValues[] = { proxyAuthorizationString };
426 auto additionalHeaderFields = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, headerFieldNames, headerFieldValues, sizeof(headerFieldNames) / sizeof(*headerFieldValues), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
427
428 CFDictionarySetValue(connectDictionary.get(), kCFStreamPropertyCONNECTAdditionalHeaders, additionalHeaderFields.get());
429 CFReadStreamSetProperty(stream, kCFStreamPropertyCONNECTProxy, connectDictionary.get());
430}

432void SocketStreamHandleImpl::addCONNECTCredentials(CFHTTPMessageRef proxyResponse)
433{
434 RetainPtr<CFHTTPAuthenticationRef> authentication = adoptCF(CFHTTPAuthenticationCreateFromResponse(0, proxyResponse));
435
436 if (!CFHTTPAuthenticationRequiresUserNameAndPassword(authentication.get())) {
437 // That's all we can offer...
438 m_client.didFailSocketStream(*this, SocketStreamError(0, m_url.string(), "Proxy authentication scheme is not supported for WebSockets"));
439 return;
440 }
441
442 int port = 0;
443 CFNumberGetValue(m_proxyPort.get(), kCFNumberIntType, &port);
444 RetainPtr<CFStringRef> methodCF = adoptCF(CFHTTPAuthenticationCopyMethod(authentication.get()));
445 RetainPtr<CFStringRef> realmCF = adoptCF(CFHTTPAuthenticationCopyRealm(authentication.get()));
446
447 if (!methodCF || !realmCF) {
448 // This shouldn't happen, but on some OS versions we get incomplete authentication data, see <rdar://problem/10416316>.
449 m_client.didFailSocketStream(*this, SocketStreamError(0, m_url.string(), "WebSocket proxy authentication couldn't be handled"));
450 return;
451 }
452
453 ProtectionSpace protectionSpace(String(m_proxyHost.get()), port, ProtectionSpaceProxyHTTPS, String(realmCF.get()), authenticationSchemeFromAuthenticationMethod(methodCF.get()));
454 String login;
455 String password;
456 if (!m_sentStoredCredentials && getStoredCONNECTProxyCredentials(protectionSpace, login, password)) {
457 // Try to apply stored credentials, if we haven't tried those already.
458 // Create a temporary request to make CFNetwork apply credentials to it. Unfortunately, this cannot work with NTLM authentication.
459 RetainPtr<CFHTTPMessageRef> dummyRequest = adoptCF(CFHTTPMessageCreateRequest(0, CFSTR("GET"), m_httpsURL.get(), kCFHTTPVersion1_1));
460
461 Boolean appliedCredentials = CFHTTPMessageApplyCredentials(dummyRequest.get(), authentication.get(), login.createCFString().get(), password.createCFString().get(), 0);
462 ASSERT_UNUSED(appliedCredentials, appliedCredentials);
463
464 RetainPtr<CFStringRef> proxyAuthorizationString = adoptCF(CFHTTPMessageCopyHeaderFieldValue(dummyRequest.get(), CFSTR("Proxy-Authorization")));
465
466 if (!proxyAuthorizationString) {
467 // Fails e.g. for NTLM auth.
468 m_client.didFailSocketStream(*this, SocketStreamError(0, m_url.string(), "Proxy authentication scheme is not supported for WebSockets"));
469 return;
470 }
471
472 // Setting the authorization results in a new connection attempt.
473 setCONNECTProxyAuthorizationForStream(m_readStream.get(), proxyAuthorizationString.get());
474 m_sentStoredCredentials = true;
475 return;
476 }
477
478 // FIXME: On platforms where AuthBrokerAgent is not available, ask the client if credentials could not be found.
479
480 m_client.didFailSocketStream(*this, SocketStreamError(0, m_url.string(), "Proxy credentials are not available"));
481}

532{
533 auto message = adoptCF(checked_cf_cast<CFHTTPMessageRef>(CFReadStreamCopyProperty(stream, kCFStreamPropertyCONNECTResponse)));

536 _CFHTTPMessageSetResponseURL(message.get(), responseURL);

538#if !PLATFORM(IOS_FAMILY)

542 setResponseProxyURL(message.get(), proxyURL.get());
543#else
544 UNUSED_PARAM(proxyHost);
545 UNUSED_PARAM(proxyPort);
546#endif
547
548 return message;
549}

551void SocketStreamHandleImpl::readStreamCallback(CFStreamEventType type)
552{
553 switch (type) {
554 case kCFStreamEventNone:
555 return;
556 case kCFStreamEventOpenCompleted:
557 return;
558 case kCFStreamEventHasBytesAvailable: {
559 if (m_connectingSubstate == WaitingForCredentials)
560 return;
561
562 if (m_connectingSubstate == WaitingForConnect) {
563 if (m_connectionType == CONNECTProxy) {
564 RetainPtr<CFHTTPMessageRef> proxyResponse = copyCONNECTProxyResponse(m_readStream.get(), m_httpsURL.get(), m_proxyHost.get(), m_proxyPort.get());
565 if (!proxyResponse)
566 return;
567
568 CFIndex proxyResponseCode = CFHTTPMessageGetResponseStatusCode(proxyResponse.get());
569 switch (proxyResponseCode) {
570 case 200:
571 // Successful connection.
572 break;
573 case 407:
574 addCONNECTCredentials(proxyResponse.get());
575 return;
576 default:
577 m_client.didFailSocketStream(*this, SocketStreamError(static_cast<int>(proxyResponseCode), m_url.string(), "Proxy connection could not be established, unexpected response code"));
578 platformClose();
579 return;
580 }
581 }
582 RELEASE_LOG(Network, "SocketStreamHandleImpl::readStreamCallback - m_connectionSubState is Connected");
583 m_connectingSubstate = Connected;
584 m_state = Open;
585 m_client.didOpenSocketStream(*this);
586 }
587
588 // Not an "else if", we could have made a client call above, and it could close the connection.
589 if (m_state == Closed)
590 return;
591
592 ASSERT(m_state == Open);
593 ASSERT(m_connectingSubstate == Connected);
594
595 CFIndex length;
596 UInt8 localBuffer[1024]; // Used if CFReadStreamGetBuffer couldn't return anything.
597 const UInt8* ptr = CFReadStreamGetBuffer(m_readStream.get(), 0, &length);
598 if (!ptr) {
599 length = CFReadStreamRead(m_readStream.get(), localBuffer, sizeof(localBuffer));
600 ptr = localBuffer;
601 }
602
603 if (!length)
604 return;
605
606 if (length == -1)
607 m_client.didFailToReceiveSocketStreamData(*this);
608 else
609 m_client.didReceiveSocketStreamData(*this, ptr, length);
610
611 return;
612 }
613 case kCFStreamEventCanAcceptBytes:
614 ASSERT_NOT_REACHED();
615 return;
616 case kCFStreamEventErrorOccurred: {
617 RetainPtr<CFErrorRef> error = adoptCF(CFReadStreamCopyError(m_readStream.get()));
618 reportErrorToClient(error.get());
619 return;
620 }
621 case kCFStreamEventEndEncountered:
622 platformClose();
623 return;
624 }
625}
626
627void SocketStreamHandleImpl::writeStreamCallback(CFStreamEventType type)
628{
629 switch (type) {
630 case kCFStreamEventNone:
631 return;
632 case kCFStreamEventOpenCompleted:
633 return;
634 case kCFStreamEventHasBytesAvailable:
635 ASSERT_NOT_REACHED();
636 return;
637 case kCFStreamEventCanAcceptBytes: {
638 // Can be false if read stream callback just decided to retry a CONNECT with credentials.
639 if (!CFWriteStreamCanAcceptBytes(m_writeStream.get()))
640 return;
641
642 if (m_connectingSubstate == WaitingForCredentials)
643 return;
644
645 if (m_connectingSubstate == WaitingForConnect) {
646 if (m_connectionType == CONNECTProxy) {
647 RetainPtr<CFHTTPMessageRef> proxyResponse = copyCONNECTProxyResponse(m_readStream.get(), m_httpsURL.get(), m_proxyHost.get(), m_proxyPort.get());
648 if (!proxyResponse)
649 return;
650
651 // Don't write anything until read stream callback has dealt with CONNECT credentials.
652 // The order of callbacks is not defined, so this can be called before readStreamCallback's kCFStreamEventHasBytesAvailable.
653 CFIndex proxyResponseCode = CFHTTPMessageGetResponseStatusCode(proxyResponse.get());
654 if (proxyResponseCode != 200)
655 return;
656 }
657 m_connectingSubstate = Connected;
658 m_state = Open;
659 m_client.didOpenSocketStream(*this);
660 }
661
662 // Not an "else if", we could have made a client call above, and it could close the connection.
663 if (m_state == Closed)
664 return;
665
666 ASSERT(m_state == Open);
667 ASSERT(m_connectingSubstate == Connected);
668
669 sendPendingData();
670 return;
671 }
672 case kCFStreamEventErrorOccurred: {
673 RetainPtr<CFErrorRef> error = adoptCF(CFWriteStreamCopyError(m_writeStream.get()));
674 reportErrorToClient(error.get());
675 return;
676 }
677 case kCFStreamEventEndEncountered:
678 // FIXME: Currently, we handle closing in read callback, but these can come independently (e.g. a server can stop listening, but keep sending data).
679 return;
680 }
681}

683void SocketStreamHandleImpl::reportErrorToClient(CFErrorRef error)
684{
685 CFIndex errorCode = CFErrorGetCode(error);
686 String description;
687
688#if PLATFORM(MAC)
689
690 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
691
692 if (CFEqual(CFErrorGetDomain(error), kCFErrorDomainOSStatus)) {
693 const char* descriptionOSStatus = GetMacOSStatusCommentString(static_cast<OSStatus>(errorCode));
694 if (descriptionOSStatus && descriptionOSStatus[0] != '\0')
695 description = makeString("OSStatus Error ", errorCode, ": ", descriptionOSStatus);
696 }
697
698 ALLOW_DEPRECATED_DECLARATIONS_END
699
700#endif
701
702 if (description.isNull()) {
703 RetainPtr<CFStringRef> descriptionCF = adoptCF(CFErrorCopyDescription(error));
704 description = String(descriptionCF.get());
705 }
706
707 m_client.didFailSocketStream(*this, SocketStreamError(static_cast<int>(errorCode), m_url.string(), description));
708}

717std::optional<size_t> SocketStreamHandleImpl::platformSendInternal(const uint8_t* data, size_t length)
718{
719 if (!m_writeStream)
720 return 0;
721
722 if (!CFWriteStreamCanAcceptBytes(m_writeStream.get()))
723 return 0;
724
725 CFIndex result = CFWriteStreamWrite(m_writeStream.get(), data, length);
726 if (result == -1)
727 return std::nullopt;
728
729 ASSERT(result >= 0);
730 return static_cast<size_t>(result);
731}

733void SocketStreamHandleImpl::platformClose()
734{
735 LOG(Network, "SocketStreamHandle %p platformClose", this);
736
737 if (m_pacRunLoopSource)
738 removePACRunLoopSource();
739
740 ASSERT(!m_readStream == !m_writeStream);
741 if (!m_readStream) {
742 if (m_connectingSubstate == New || m_connectingSubstate == ExecutingPACFile)
743 m_client.didCloseSocketStream(*this);
744 return;
745 }
746
747 CFReadStreamUnscheduleFromRunLoop(m_readStream.get(), callbacksRunLoop(), callbacksRunLoopMode());
748 CFWriteStreamUnscheduleFromRunLoop(m_writeStream.get(), callbacksRunLoop(), callbacksRunLoopMode());
749
750 CFReadStreamClose(m_readStream.get());
751 CFWriteStreamClose(m_writeStream.get());
752
753 m_readStream = nullptr;
754 m_writeStream = nullptr;
755
756 m_client.didCloseSocketStream(*this);
757}
