46@interface WebCookieObserverAdapter : NSObject {
47 WebCore::CookieStorageObserver* observer;
48}
49- (instancetype)initWithObserver:(WebCore::CookieStorageObserver&)theObserver;
50- (void)cookiesChangedNotificationHandler:(NSNotification *)notification;
51
52@end
53
54@implementation WebCookieObserverAdapter
55
56- (instancetype)initWithObserver:(WebCore::CookieStorageObserver&)theObserver
57{
58 self = [super init];
59 if (!self)
60 return nil;
61
62 observer = &theObserver;
63
64 return self;
65}
66
67- (void)cookiesChangedNotificationHandler:(NSNotification *)notification
68{
69 UNUSED_PARAM(notification);
70 observer->cookiesDidChange();
71}
72
73@end

75namespace WebCore {

77CookieStorageObserver::CookieStorageObserver(NSHTTPCookieStorage *cookieStorage)
78 : m_cookieStorage(cookieStorage)
79{
80 ASSERT(isMainThread());
81 ASSERT(m_cookieStorage);
82 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
83}
84
85CookieStorageObserver::~CookieStorageObserver()
86{
87 ASSERT(isMainThread());
88
89 if (m_cookieChangeCallback) {
90 ASSERT(m_observerAdapter);
91 stopObserving();
92 }
93}
94
95void CookieStorageObserver::startObserving(WTF::Function<void()>&& callback)
96{
97 ASSERT(isMainThread());
98 ASSERT(!m_cookieChangeCallback);
99 ASSERT(!m_observerAdapter);
100 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
101
102 m_cookieChangeCallback = WTFMove(callback);
103 m_observerAdapter = adoptNS([[WebCookieObserverAdapter alloc] initWithObserver:*this]);
104
105 if (!m_hasRegisteredInternalsForNotifications) {
106 if (m_cookieStorage.get() != [NSHTTPCookieStorage sharedHTTPCookieStorage]) {
107 auto internalObject = (static_cast<WebNSHTTPCookieStorageDummyForInternalAccess *>(m_cookieStorage.get()))->_internal;
108 [internalObject registerForPostingNotificationsWithContext:m_cookieStorage.get()];
109 }
110
111 m_hasRegisteredInternalsForNotifications = true;
112 }
113
114 [[NSNotificationCenter defaultCenter] addObserver:m_observerAdapter.get() selector:@selector(cookiesChangedNotificationHandler:) name:NSHTTPCookieManagerCookiesChangedNotification object:m_cookieStorage.get()];
115}
116
117void CookieStorageObserver::stopObserving()
118{
119 ASSERT(isMainThread());
120 ASSERT(m_cookieChangeCallback);
121 ASSERT(m_observerAdapter);
122 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
123
124 [[NSNotificationCenter defaultCenter] removeObserver:m_observerAdapter.get() name:NSHTTPCookieManagerCookiesChangedNotification object:nil];
125
126 m_cookieChangeCallback = nullptr;
127 m_observerAdapter = nil;
128}
129
130void CookieStorageObserver::cookiesDidChange()
131{
132 callOnMainThread([weakThis = makeWeakPtr(*this)] {
133 if (weakThis && weakThis->m_cookieChangeCallback)
134 weakThis->m_cookieChangeCallback();
135 });
136}
