84static Vector<Cookie> nsCookiesToCookieVector(NSArray<NSHTTPCookie *> *nsCookies, const Function<bool(NSHTTPCookie *)>& filter = { })
85{
86 Vector<Cookie> cookies;
87 cookies.reserveInitialCapacity(nsCookies.count);
88 for (NSHTTPCookie *nsCookie in nsCookies) {
89 if (!filter || filter(nsCookie))
90 cookies.uncheckedAppend(nsCookie);
91 }
92 return cookies;
93}

121void NetworkStorageSession::setAllCookiesToSameSiteStrict(const RegistrableDomain& domain, CompletionHandler<void()>&& completionHandler)
122{
123 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
124
125#if HAVE(CFNETWORK_SAMESITE_COOKIE_API)
126 RetainPtr<NSMutableArray<NSHTTPCookie *>> oldCookiesToDelete = adoptNS([[NSMutableArray alloc] init]);
127 RetainPtr<NSMutableArray<NSHTTPCookie *>> newCookiesToAdd = adoptNS([[NSMutableArray alloc] init]);
128
129 for (NSHTTPCookie *nsCookie in nsCookieStorage().cookies) {
130 if (RegistrableDomain::uncheckedCreateFromHost(nsCookie.domain) == domain && nsCookie.sameSitePolicy != NSHTTPCookieSameSiteStrict) {
131 [oldCookiesToDelete addObject:nsCookie];
132 RetainPtr<NSMutableDictionary<NSHTTPCookiePropertyKey, id>> mutableProperties = adoptNS([[nsCookie properties] mutableCopy]);
133 mutableProperties.get()[NSHTTPCookieSameSitePolicy] = NSHTTPCookieSameSiteStrict;
134 NSHTTPCookie *strictCookie = [NSHTTPCookie cookieWithProperties:mutableProperties.get()];
135 [newCookiesToAdd addObject:strictCookie];
136 }
137 }
138
139 BEGIN_BLOCK_OBJC_EXCEPTIONS
140 for (NSHTTPCookie *oldCookie in oldCookiesToDelete.get())
141 deleteHTTPCookie(cookieStorage().get(), oldCookie);
142
143 for (NSHTTPCookie *newCookie in newCookiesToAdd.get())
144 [nsCookieStorage() setCookie:newCookie];
145 END_BLOCK_OBJC_EXCEPTIONS
146#else
147 UNUSED_PARAM(domain);
148#endif
149 completionHandler();
150}

177RetainPtr<CFURLStorageSessionRef> createPrivateStorageSession(CFStringRef identifier, std::optional<HTTPCookieAcceptPolicy> cookieAcceptPolicy)
178{
179 const void* sessionPropertyKeys[] = { _kCFURLStorageSessionIsPrivate };
180 const void* sessionPropertyValues[] = { kCFBooleanTrue };
181 auto sessionProperties = adoptCF(CFDictionaryCreate(kCFAllocatorDefault, sessionPropertyKeys, sessionPropertyValues, sizeof(sessionPropertyKeys) / sizeof(*sessionPropertyKeys), &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
182 auto storageSession = adoptCF(_CFURLStorageSessionCreate(kCFAllocatorDefault, identifier, sessionProperties.get()));
183
184 if (!storageSession)
185 return nullptr;
186
187 // The private storage session should have the same properties as the default storage session,
188 // with the exception that it should be in-memory only storage.
189
190 // FIXME 9199649: If any of the storages do not exist, do no use the storage session.
191 // This could occur if there is an issue figuring out where to place a storage on disk (e.g. the
192 // sandbox does not allow CFNetwork access).
193
194 auto cache = adoptCF(_CFURLStorageSessionCopyCache(kCFAllocatorDefault, storageSession.get()));
195 if (!cache)
196 return nullptr;
197
198 CFURLCacheSetDiskCapacity(cache.get(), 0); // Setting disk cache size should not be necessary once <rdar://problem/12656814> is fixed.
199 CFURLCacheSetMemoryCapacity(cache.get(), [[NSURLCache sharedURLCache] memoryCapacity]);
200
201 auto cookieStorage = adoptCF(_CFURLStorageSessionCopyCookieStorage(kCFAllocatorDefault, storageSession.get()));
202 if (!cookieStorage)
203 return nullptr;
204
205 NSHTTPCookieAcceptPolicy nsCookieAcceptPolicy;
206 if (cookieAcceptPolicy)
207 nsCookieAcceptPolicy = toNSHTTPCookieAcceptPolicy(*cookieAcceptPolicy);
208 else
209 nsCookieAcceptPolicy = [[NSHTTPCookieStorage sharedHTTPCookieStorage] cookieAcceptPolicy];
210
211 // FIXME: Use _CFHTTPCookieStorageGetDefault when USE(CFNETWORK) is defined in WebKit for consistency.
212 CFHTTPCookieStorageSetCookieAcceptPolicy(cookieStorage.get(), nsCookieAcceptPolicy);
213
214 return storageSession;
215}

253static RetainPtr<NSArray> cookiesForURL(NSHTTPCookieStorage *storage, NSURL *url, NSURL *mainDocumentURL, const std::optional<SameSiteInfo>& sameSiteInfo, NSString *partition = nullptr)
254{
255 // The _getCookiesForURL: method calls the completionHandler synchronously. We use std::optional<> to check this invariant and crash if it's not met.
256 std::optional<RetainPtr<NSArray>> cookiesPtr;
257 auto completionHandler = [&cookiesPtr] (NSArray *cookies) {
258 cookiesPtr = retainPtr(cookies);
259 };
260// FIXME: Seems like this newer code path can be used for watchOS and tvOS too.
261#if !PLATFORM(WATCHOS) && !PLATFORM(APPLETV)
262 if ([storage respondsToSelector:@selector(_getCookiesForURL:mainDocumentURL:partition:policyProperties:completionHandler:)])
263 [storage _getCookiesForURL:url mainDocumentURL:mainDocumentURL partition:partition policyProperties:sameSiteInfo ? policyProperties(sameSiteInfo.value(), url).get() : nullptr completionHandler:completionHandler];
264 else
265 [storage _getCookiesForURL:url mainDocumentURL:mainDocumentURL partition:partition completionHandler:completionHandler];
266#else
267 [storage _getCookiesForURL:url mainDocumentURL:mainDocumentURL partition:partition completionHandler:completionHandler];
268 UNUSED_PARAM(sameSiteInfo);
269#endif
270 RELEASE_ASSERT(!!cookiesPtr);
271 return WTFMove(*cookiesPtr);
272}

274void NetworkStorageSession::setHTTPCookiesForURL(CFHTTPCookieStorageRef cookieStorage, NSArray *cookies, NSURL *url, NSURL *mainDocumentURL, const SameSiteInfo& sameSiteInfo) const
275{
276 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies) || m_isInMemoryCookieStore);
277 if (!cookieStorage) {
278// FIXME: Seems like this newer code path can be used for watchOS and tvOS too.
279#if !PLATFORM(WATCHOS) && !PLATFORM(APPLETV)
280 if ([NSHTTPCookieStorage instancesRespondToSelector:@selector(_setCookies:forURL:mainDocumentURL:policyProperties:)])
281 [[NSHTTPCookieStorage sharedHTTPCookieStorage] _setCookies:cookies forURL:url mainDocumentURL:mainDocumentURL policyProperties:policyProperties(sameSiteInfo, url).get()];
282 else
283#endif
284 [[NSHTTPCookieStorage sharedHTTPCookieStorage] setCookies:cookies forURL:url mainDocumentURL:mainDocumentURL];
285 return;
286 }
287// FIXME: Seems like this newer code path can be used for watchOS and tvOS too.
288#if !PLATFORM(WATCHOS) && !PLATFORM(APPLETV)
289 if ([NSHTTPCookieStorage instancesRespondToSelector:@selector(_setCookies:forURL:mainDocumentURL:policyProperties:)]) {
290 // FIXME: Stop creating a new NSHTTPCookieStorage object each time we want to query the cookie jar.
291 // NetworkStorageSession could instead keep a NSHTTPCookieStorage object for us.
292 RetainPtr<NSHTTPCookieStorage> nsCookieStorage = adoptNS([[NSHTTPCookieStorage alloc] _initWithCFHTTPCookieStorage:cookieStorage]);
293 [nsCookieStorage _setCookies:cookies forURL:url mainDocumentURL:mainDocumentURL policyProperties:policyProperties(sameSiteInfo, url).get()];
294 } else {
295#endif
296 auto cfCookies = adoptCF([NSHTTPCookie _ns2cfCookies:cookies]);
297 CFHTTPCookieStorageSetCookies(cookieStorage, cfCookies.get(), [url _cfurl], [mainDocumentURL _cfurl]);
298#if !PLATFORM(WATCHOS) && !PLATFORM(APPLETV)
299 }
300#else
301 UNUSED_PARAM(sameSiteInfo);
302#endif
303}

319NSHTTPCookie *NetworkStorageSession::capExpiryOfPersistentCookie(NSHTTPCookie *cookie, Seconds cap)
320{
321 if ([cookie isSessionOnly])
322 return cookie;
323
324 if (!cookie.expiresDate || cookie.expiresDate.timeIntervalSinceNow > cap.seconds()) {
325 auto properties = adoptNS([[cookie properties] mutableCopy]);
326 auto date = adoptNS([[NSDate alloc] initWithTimeIntervalSinceNow:cap.seconds()]);
327 [properties setObject:date.get() forKey:NSHTTPCookieExpires];
328 cookie = [NSHTTPCookie cookieWithProperties:properties.get()];
329 }
330 return cookie;
331}

346std::pair<String, bool> NetworkStorageSession::cookiesForSession(const URL& firstParty, const SameSiteInfo& sameSiteInfo, const URL& url, std::optional<FrameIdentifier> frameID, std::optional<PageIdentifier> pageID, IncludeHTTPOnlyOrNot includeHTTPOnly, IncludeSecureCookies includeSecureCookies, ShouldAskITP shouldAskITP, ShouldRelaxThirdPartyCookieBlocking shouldRelaxThirdPartyCookieBlocking) const
347{
348 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies) || m_isInMemoryCookieStore);
349
350 BEGIN_BLOCK_OBJC_EXCEPTIONS
351
352 auto cookies = cookiesForURL(firstParty, sameSiteInfo, url, frameID, pageID, shouldAskITP, shouldRelaxThirdPartyCookieBlocking);
353 if (![cookies count])
354 return { String(), false }; // Return a null string; StringBuilder below would create an empty one.
355
356 StringBuilder cookiesBuilder;
357 bool didAccessSecureCookies = false;
358 for (NSHTTPCookie *cookie in cookies.get()) {
359 if (![[cookie name] length])
360 continue;
361 if (!includeHTTPOnly && [cookie isHTTPOnly])
362 continue;
363 if ([cookie isSecure]) {
364 didAccessSecureCookies = true;
365 if (includeSecureCookies == IncludeSecureCookies::No)
366 continue;
367 }
368 cookiesBuilder.append(cookiesBuilder.isEmpty() ? "" : "; ", [cookie name], '=', [cookie value]);
369 }
370 return { cookiesBuilder.toString(), didAccessSecureCookies };
371
372 END_BLOCK_OBJC_EXCEPTIONS
373 return { String(), false };
374}

376static void deleteAllHTTPCookies(CFHTTPCookieStorageRef cookieStorage)
377{
378 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
379
380 if (!cookieStorage) {
381 NSHTTPCookieStorage *cookieStorage = [NSHTTPCookieStorage sharedHTTPCookieStorage];
382 NSArray *cookies = [cookieStorage cookies];
383 if (!cookies)
384 return;
385
386 for (NSHTTPCookie *cookie in cookies)
387 [cookieStorage deleteCookie:cookie];
388 return;
389 }
390
391 CFHTTPCookieStorageDeleteAllCookies(cookieStorage);
392}

409static NSHTTPCookie *parseDOMCookie(String cookieString, NSURL* cookieURL, std::optional<Seconds> cappedLifetime)
410{
411 // <rdar://problem/5632883> On 10.5, NSHTTPCookieStorage would store an empty cookie,
412 // which would be sent as "Cookie: =".
413 if (cookieString.isEmpty())
414 return nil;
415
416 // <http://bugs.webkit.org/show_bug.cgi?id=6531>, <rdar://4409034>
417 // cookiesWithResponseHeaderFields doesn't parse cookies without a value
418 cookieString = cookieString.contains('=') ? cookieString : cookieString + "=";
419
420 NSHTTPCookie *cookie = [NSHTTPCookie _cookieForSetCookieString:cookieString forURL:cookieURL partition:nil];
421 if (!cookie)
422 return nil;
423
424 // <rdar://problem/5632883> On 10.5, NSHTTPCookieStorage would store an empty cookie,
425 // which would be sent as "Cookie: =". We have a workaround in setCookies() to prevent
426 // that, but we also need to avoid sending cookies that were previously stored, and
427 // there's no harm to doing this check because such a cookie is never valid.
428 if (![[cookie name] length])
429 return nil;
430
431 if ([cookie isHTTPOnly])
432 return nil;
433
434 // Cap lifetime of persistent, client-side cookies.
435 if (cappedLifetime)
436 return NetworkStorageSession::capExpiryOfPersistentCookie(cookie, *cappedLifetime);
437
438 return cookie;
439}

442{
443 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies) || m_isInMemoryCookieStore);
444
445 BEGIN_BLOCK_OBJC_EXCEPTIONS
446
447#if ENABLE(RESOURCE_LOAD_STATISTICS)
448 if (shouldAskITP == ShouldAskITP::Yes && shouldBlockCookies(firstParty, url, frameID, pageID, shouldRelaxThirdPartyCookieBlocking))
449 return;

452 UNUSED_PARAM(pageID);

464 if (!cookie)
465 return;
466
467 setHTTPCookiesForURL(cookieStorage().get(), @[cookie], cookieURL, firstParty, sameSiteInfo);

469 END_BLOCK_OBJC_EXCEPTIONS
470}

492bool NetworkStorageSession::getRawCookies(const URL& firstParty, const SameSiteInfo& sameSiteInfo, const URL& url, std::optional<FrameIdentifier> frameID, std::optional<PageIdentifier> pageID, ShouldAskITP shouldAskITP, ShouldRelaxThirdPartyCookieBlocking shouldRelaxThirdPartyCookieBlocking, Vector<Cookie>& rawCookies) const
493{
494 rawCookies.clear();
495 BEGIN_BLOCK_OBJC_EXCEPTIONS
496
497 RetainPtr<NSArray> cookies = cookiesForURL(firstParty, sameSiteInfo, url, frameID, pageID, shouldAskITP, shouldRelaxThirdPartyCookieBlocking);
498 NSUInteger count = [cookies count];
499 rawCookies.reserveCapacity(count);
500 for (NSUInteger i = 0; i < count; ++i) {
501 NSHTTPCookie *cookie = (NSHTTPCookie *)[cookies objectAtIndex:i];
502 rawCookies.uncheckedAppend({ cookie });
503 }
504
505 END_BLOCK_OBJC_EXCEPTIONS
506 return true;
507}

509void NetworkStorageSession::deleteCookie(const URL& url, const String& cookieName) const
510{
511 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
512
513 BEGIN_BLOCK_OBJC_EXCEPTIONS
514
515 RetainPtr<CFHTTPCookieStorageRef> cookieStorage = this->cookieStorage();
516 RetainPtr<NSArray> cookies = httpCookiesForURL(cookieStorage.get(), nil, std::nullopt, url);
517
518 NSString *cookieNameString = cookieName;
519
520 NSUInteger count = [cookies count];
521 for (NSUInteger i = 0; i < count; ++i) {
522 NSHTTPCookie *cookie = (NSHTTPCookie *)[cookies objectAtIndex:i];
523 if ([[cookie name] isEqualToString:cookieNameString])
524 deleteHTTPCookie(cookieStorage.get(), cookie);
525 }
526
527 END_BLOCK_OBJC_EXCEPTIONS
528}

561
562 RetainPtr<CFHTTPCookieStorageRef> cookieStorage = this->cookieStorage();
563 RetainPtr<NSArray> cookies = httpCookies(cookieStorage.get());
564 if (!cookies)
565 return;
566
567 HashMap<String, Vector<RetainPtr<NSHTTPCookie>>> cookiesByDomain;
568 for (NSHTTPCookie *cookie in cookies.get()) {
569 if (!cookie.domain || (includeHttpOnlyCookies == IncludeHttpOnlyCookies::No && cookie.isHTTPOnly))
570 continue;
571 cookiesByDomain.ensure(cookie.domain, [] {
572 return Vector<RetainPtr<NSHTTPCookie>>();
573 }).iterator->value.append(cookie);
574 }
575
576 for (const auto& hostname : hostnames) {
577 auto it = cookiesByDomain.find(hostname);
578 if (it == cookiesByDomain.end())
579 continue;
580
581 for (auto& cookie : it->value)
582 deleteHTTPCookie(cookieStorage.get(), cookie.get());
583 }
584
585 [nsCookieStorage() _saveCookies];
586
587 END_BLOCK_OBJC_EXCEPTIONS
588}

614void NetworkStorageSession::registerCookieChangeListenersIfNecessary()
615{
616 if (m_didRegisterCookieListeners)
617 return;
618
619 m_didRegisterCookieListeners = true;
620
621 [nsCookieStorage() _setCookiesChangedHandler:makeBlockPtr([this, weakThis = makeWeakPtr(*this)](NSArray<NSHTTPCookie *> *addedCookies, NSString *domainForChangedCookie) {
622 if (!weakThis)
623 return;
624 String host = domainForChangedCookie;
625 auto it = m_cookieChangeObservers.find(host);
626 if (it == m_cookieChangeObservers.end())
627 return;
628 auto cookies = nsCookiesToCookieVector(addedCookies, [](NSHTTPCookie *cookie) { return !cookie.HTTPOnly; });
629 if (cookies.isEmpty())
630 return;
631 for (auto* observer : it->value)
632 observer->cookiesAdded(host, cookies);
633 }).get() onQueue:dispatch_get_main_queue()];
634
635 [nsCookieStorage() _setCookiesRemovedHandler:makeBlockPtr([this, weakThis = makeWeakPtr(*this)](NSArray<NSHTTPCookie *> *removedCookies, NSString *domainForRemovedCookies, bool removeAllCookies) {
636 if (!weakThis)
637 return;
638 if (removeAllCookies) {
639 for (auto& observers : m_cookieChangeObservers.values()) {
640 for (auto* observer : observers)
641 observer->allCookiesDeleted();
642 }
643 return;
644 }
645
646 String host = domainForRemovedCookies;
647 auto it = m_cookieChangeObservers.find(host);
648 if (it == m_cookieChangeObservers.end())
649 return;
650
651 auto cookies = nsCookiesToCookieVector(removedCookies, [](NSHTTPCookie *cookie) { return !cookie.HTTPOnly; });
652 if (cookies.isEmpty())
653 return;
654 for (auto* observer : it->value)
655 observer->cookiesDeleted(host, cookies);
656 }).get() onQueue:dispatch_get_main_queue()];
657}

671void NetworkStorageSession::startListeningForCookieChangeNotifications(CookieChangeObserver& observer, const String& host)
672{
673 registerCookieChangeListenersIfNecessary();
674
675 auto& observers = m_cookieChangeObservers.ensure(host, [] {
676 return HashSet<CookieChangeObserver*> { };
677 }).iterator->value;
678 ASSERT(!observers.contains(&observer));
679 observers.add(&observer);
680
681 if (!m_subscribedDomainsForCookieChanges)
682 m_subscribedDomainsForCookieChanges = adoptNS([[NSMutableSet alloc] init]);
683 else if ([m_subscribedDomainsForCookieChanges containsObject:(NSString *)host])
684 return;
685
686 [m_subscribedDomainsForCookieChanges addObject:(NSString *)host];
687 [nsCookieStorage() _setSubscribedDomainsForCookieChanges:m_subscribedDomainsForCookieChanges.get()];
688}

690void NetworkStorageSession::stopListeningForCookieChangeNotifications(CookieChangeObserver& observer, const HashSet<String>& hosts)
691{
692 bool subscribedURLsChanged = false;
693 for (auto& host : hosts) {
694 auto it = m_cookieChangeObservers.find(host);
695 ASSERT(it != m_cookieChangeObservers.end());
696 if (it == m_cookieChangeObservers.end())
697 continue;
698
699 auto& observers = it->value;
700 ASSERT(observers.contains(&observer));
701 observers.remove(&observer);
702 if (observers.isEmpty()) {
703 m_cookieChangeObservers.remove(it);
704 ASSERT([m_subscribedDomainsForCookieChanges containsObject:(NSString *)host]);
705 [m_subscribedDomainsForCookieChanges removeObject:(NSString *)host];
706 subscribedURLsChanged = true;
707 }
708 }
709 if (subscribedURLsChanged)
710 [nsCookieStorage() _setSubscribedDomainsForCookieChanges:m_subscribedDomainsForCookieChanges.get()];
711}
