32typedef struct _CFURLProtectionSpace* CFURLProtectionSpaceRef;

34
35OBJC_CLASS NSURLProtectionSpace;
36
37namespace WebCore {
38
39class ProtectionSpace : public ProtectionSpaceBase {
40public:
41 ProtectionSpace() : ProtectionSpaceBase() { }
42 ProtectionSpace(const String& host, int port, ProtectionSpaceServerType serverType, const String& realm, ProtectionSpaceAuthenticationScheme authenticationScheme)
43 : ProtectionSpaceBase(host, port, serverType, realm, authenticationScheme)
44 {
45 }
46
47 ProtectionSpace(WTF::HashTableDeletedValueType deletedValue) : ProtectionSpaceBase(deletedValue) { }
48
49#if USE(CFURLCONNECTION)
50 explicit ProtectionSpace(CFURLProtectionSpaceRef);
51#endif
52 WEBCORE_EXPORT explicit ProtectionSpace(NSURLProtectionSpace *);
53
54 static bool platformCompare(const ProtectionSpace& a, const ProtectionSpace& b);
55
56 bool encodingRequiresPlatformData() const { return m_nsSpace && encodingRequiresPlatformData(m_nsSpace.get()); }
57
58 WEBCORE_EXPORT bool receivesCredentialSecurely() const;
59
60#if USE(CFURLCONNECTION)
61 CFURLProtectionSpaceRef cfSpace() const;
62#endif
63 WEBCORE_EXPORT NSURLProtectionSpace *nsSpace() const;
64
65private:
66 WEBCORE_EXPORT static bool encodingRequiresPlatformData(NSURLProtectionSpace *);
67
68 mutable RetainPtr<NSURLProtectionSpace> m_nsSpace;
69};

71} // namespace WebCore
