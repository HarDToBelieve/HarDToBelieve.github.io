33static ProtectionSpaceServerType type(NSURLProtectionSpace *space)
34{
35 if ([space isProxy]) {
36 NSString *proxyType = space.proxyType;
37 if ([proxyType isEqualToString:NSURLProtectionSpaceHTTPProxy])
38 return ProtectionSpaceProxyHTTP;
39 if ([proxyType isEqualToString:NSURLProtectionSpaceHTTPSProxy])
40 return ProtectionSpaceProxyHTTPS;
41 if ([proxyType isEqualToString:NSURLProtectionSpaceFTPProxy])
42 return ProtectionSpaceProxyFTP;
43 if ([proxyType isEqualToString:NSURLProtectionSpaceSOCKSProxy])
44 return ProtectionSpaceProxySOCKS;
45
46 ASSERT_NOT_REACHED();
47 return ProtectionSpaceProxyHTTP;
48 }
49
50 NSString *protocol = space.protocol;
51 if ([protocol caseInsensitiveCompare:@"http"] == NSOrderedSame)
52 return ProtectionSpaceServerHTTP;
53 if ([protocol caseInsensitiveCompare:@"https"] == NSOrderedSame)
54 return ProtectionSpaceServerHTTPS;
55 if ([protocol caseInsensitiveCompare:@"ftp"] == NSOrderedSame)
56 return ProtectionSpaceServerFTP;
57 if ([protocol caseInsensitiveCompare:@"ftps"] == NSOrderedSame)
58 return ProtectionSpaceServerFTPS;
59
60 ASSERT_NOT_REACHED();
61 return ProtectionSpaceServerHTTP;
62}
63
64static ProtectionSpaceAuthenticationScheme scheme(NSURLProtectionSpace *space)
65{
66 NSString *method = space.authenticationMethod;
67 if ([method isEqualToString:NSURLAuthenticationMethodDefault])
68 return ProtectionSpaceAuthenticationSchemeDefault;
69 if ([method isEqualToString:NSURLAuthenticationMethodHTTPBasic])
70 return ProtectionSpaceAuthenticationSchemeHTTPBasic;
71 if ([method isEqualToString:NSURLAuthenticationMethodHTTPDigest])
72 return ProtectionSpaceAuthenticationSchemeHTTPDigest;
73 if ([method isEqualToString:NSURLAuthenticationMethodHTMLForm])
74 return ProtectionSpaceAuthenticationSchemeHTMLForm;
75 if ([method isEqualToString:NSURLAuthenticationMethodNTLM])
76 return ProtectionSpaceAuthenticationSchemeNTLM;
77 if ([method isEqualToString:NSURLAuthenticationMethodNegotiate])
78 return ProtectionSpaceAuthenticationSchemeNegotiate;
79#if USE(PROTECTION_SPACE_AUTH_CALLBACK)
80 if ([method isEqualToString:NSURLAuthenticationMethodClientCertificate])
81 return ProtectionSpaceAuthenticationSchemeClientCertificateRequested;
82 if ([method isEqualToString:NSURLAuthenticationMethodServerTrust])
83 return ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested;
84#endif
85 if ([method isEqualToString:NSURLAuthenticationMethodOAuth])
86 return ProtectionSpaceAuthenticationSchemeOAuth;
87
88 ASSERT_NOT_REACHED();
89 return ProtectionSpaceAuthenticationSchemeUnknown;
90}

92ProtectionSpace::ProtectionSpace(NSURLProtectionSpace *space)
93 : ProtectionSpace(space.host, space.port, type(space), space.realm, scheme(space))
94{
95 m_nsSpace = space;
96}

98NSURLProtectionSpace *ProtectionSpace::nsSpace() const
99{
100 if (m_nsSpace)
101 return m_nsSpace.get();
102
103 NSString *proxyType = nil;
104 NSString *protocol = nil;
105 switch (serverType()) {
106 case ProtectionSpaceServerHTTP:
107 protocol = @"http";
108 break;
109 case ProtectionSpaceServerHTTPS:
110 protocol = @"https";
111 break;
112 case ProtectionSpaceServerFTP:
113 protocol = @"ftp";
114 break;
115 case ProtectionSpaceServerFTPS:
116 protocol = @"ftps";
117 break;
118 case ProtectionSpaceProxyHTTP:
119 proxyType = NSURLProtectionSpaceHTTPProxy;
120 break;
121 case ProtectionSpaceProxyHTTPS:
122 proxyType = NSURLProtectionSpaceHTTPSProxy;
123 break;
124 case ProtectionSpaceProxyFTP:
125 proxyType = NSURLProtectionSpaceFTPProxy;
126 break;
127 case ProtectionSpaceProxySOCKS:
128 proxyType = NSURLProtectionSpaceSOCKSProxy;
129 break;
130 default:
131 ASSERT_NOT_REACHED();
132 }
133
134 NSString *method = nil;
135 switch (authenticationScheme()) {
136 case ProtectionSpaceAuthenticationSchemeDefault:
137 method = NSURLAuthenticationMethodDefault;
138 break;
139 case ProtectionSpaceAuthenticationSchemeHTTPBasic:
140 method = NSURLAuthenticationMethodHTTPBasic;
141 break;
142 case ProtectionSpaceAuthenticationSchemeHTTPDigest:
143 method = NSURLAuthenticationMethodHTTPDigest;
144 break;
145 case ProtectionSpaceAuthenticationSchemeHTMLForm:
146 method = NSURLAuthenticationMethodHTMLForm;
147 break;
148 case ProtectionSpaceAuthenticationSchemeNTLM:
149 method = NSURLAuthenticationMethodNTLM;
150 break;
151 case ProtectionSpaceAuthenticationSchemeNegotiate:
152 method = NSURLAuthenticationMethodNegotiate;
153 break;
154#if USE(PROTECTION_SPACE_AUTH_CALLBACK)
155 case ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested:
156 method = NSURLAuthenticationMethodServerTrust;
157 break;
158 case ProtectionSpaceAuthenticationSchemeClientCertificateRequested:
159 method = NSURLAuthenticationMethodClientCertificate;
160 break;
161#endif
162 case ProtectionSpaceAuthenticationSchemeOAuth:
163 method = NSURLAuthenticationMethodOAuth;
164 break;
165 default:
166 ASSERT_NOT_REACHED();
167 }
168
169 m_nsSpace = adoptNS(proxyType
170 ? [[NSURLProtectionSpace alloc] initWithProxyHost:host() port:port() type:proxyType realm:realm() authenticationMethod:method]
171 : [[NSURLProtectionSpace alloc] initWithHost:host() port:port() protocol:protocol realm:realm() authenticationMethod:method]);
172
173 return m_nsSpace.get();
174}
175
176bool ProtectionSpace::platformCompare(const ProtectionSpace& a, const ProtectionSpace& b)
177{
178 if (!a.m_nsSpace && !b.m_nsSpace)
179 return true;
180
181 return [a.nsSpace() isEqual:b.nsSpace()];
182}

184bool ProtectionSpace::receivesCredentialSecurely() const
185{
186 return nsSpace().receivesCredentialSecurely;
187}

189bool ProtectionSpace::encodingRequiresPlatformData(NSURLProtectionSpace *space)
190{
191 return space.distinguishedNames || space.serverTrust;
192}
