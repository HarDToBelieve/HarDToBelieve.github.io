76static ResourceResponse synthesizedResponseForRange(const ResourceResponse& originalResponse, const ParsedRequestRange& parsedRequestRange, std::optional<size_t> totalContentLength)
77{
78 ASSERT(isMainThread());
79 auto begin = parsedRequestRange.begin;
80 auto end = parsedRequestRange.end;
81
82 auto newContentRange = makeString("bytes ", begin, "-", end, "/", (totalContentLength ? makeString(*totalContentLength) : "*"));
83 auto newContentLength = makeString(end - begin + 1);
84
85 ResourceResponse newResponse = originalResponse;
86 newResponse.setHTTPHeaderField(HTTPHeaderName::ContentRange, newContentRange);
87 newResponse.setHTTPHeaderField(HTTPHeaderName::ContentLength, newContentLength);
88 constexpr auto partialContent = 206;
89 newResponse.setHTTPStatusCode(partialContent);
90
91 return newResponse;
92}

103void RangeResponseGenerator::giveResponseToTaskIfBytesInRangeReceived(WebCoreNSURLSessionDataTask *task, const ParsedRequestRange& range, std::optional<size_t> expectedContentLength, const Data& data)
104{
105 ASSERT(isMainThread());
106 auto buffer = data.buffer;
107 auto bufferSize = buffer->size();
108
109 if (bufferSize < range.begin)
110 return;
111
112 auto* taskData = data.taskData.get(task);
113 if (!taskData)
114 return;
115
116 auto giveBytesToTask = [task = retainPtr(task), buffer, taskData = makeWeakPtr(*taskData), generator = makeWeakPtr(*this)] {
117 ASSERT(isMainThread());
118 if ([task state] != NSURLSessionTaskStateRunning)
119 return;
120 if (!taskData)
121 return;
122 auto& range = taskData->range;
123 auto& byteIndex = taskData->nextByteToGiveBufferIndex;
124 while (true) {
125 if (byteIndex >= buffer->size())
126 break;
127 auto bufferView = buffer->getSomeData(byteIndex);
128 if (!bufferView.size() || byteIndex > range.end)
129 break;
130
131 size_t bytesFromThisViewToDeliver = std::min(bufferView.size(), range.end - byteIndex + 1);
132 byteIndex += bytesFromThisViewToDeliver;
133 [task resource:nullptr receivedData:bufferView.data() length:bytesFromThisViewToDeliver];
134 }
135 if (byteIndex >= range.end) {
136 [task resourceFinished:nullptr metrics:NetworkLoadMetrics { }];
137 callOnMainThread([generator, task] {
138 if (generator)
139 generator->removeTask(task.get());
140 });
141 }
142 };
143
144 switch (taskData->responseState) {
145 case Data::TaskData::ResponseState::NotSynthesizedYet: {
146 auto response = synthesizedResponseForRange(data.originalResponse, range, expectedContentLength);
147 [task resource:nullptr receivedResponse:response completionHandler:[giveBytesToTask = WTFMove(giveBytesToTask), taskData = makeWeakPtr(taskData), task = retainPtr(task)] (WebCore::ShouldContinuePolicyCheck shouldContinue) {
148 if (taskData)
149 taskData->responseState = Data::TaskData::ResponseState::SessionCalledCompletionHandler;
150 if (shouldContinue == ShouldContinuePolicyCheck::Yes)
151 giveBytesToTask();
152 else
153 [task cancel];
154 }];
155 taskData->responseState = Data::TaskData::ResponseState::WaitingForSession;
156 break;
157 }
158 case Data::TaskData::ResponseState::WaitingForSession:
159 break;
160 case Data::TaskData::ResponseState::SessionCalledCompletionHandler:
161 giveBytesToTask();
162 break;
163 }
164}

166std::optional<size_t> RangeResponseGenerator::expectedContentLengthFromData(const Data& data)
167{
168 ASSERT(isMainThread());
169 if (data.successfullyFinishedLoading == Data::SuccessfullyFinishedLoading::Yes)
170 return data.buffer->size();
171
172 // FIXME: ResourceResponseBase::expectedContentLength() should return std::optional<size_t> instead of us doing this check here.
173 auto expectedContentLength = data.originalResponse.expectedContentLength();
174 if (expectedContentLength == NSURLResponseUnknownLength)
175 return std::nullopt;
176 return expectedContentLength;
177}

179void RangeResponseGenerator::giveResponseToTasksWithFinishedRanges(Data& data)
180{
181 ASSERT(isMainThread());
182 auto expectedContentLength = expectedContentLengthFromData(data);
183
184 for (auto& pair : data.taskData)
185 giveResponseToTaskIfBytesInRangeReceived(pair.key.get(), pair.value->range, expectedContentLength, data);
186}

188bool RangeResponseGenerator::willHandleRequest(WebCoreNSURLSessionDataTask *task, NSURLRequest *request)
189{
190 ASSERT(isMainThread());
191 auto* data = m_map.get(request.URL.absoluteString);
192 if (!data)
193 return false;
194
195 auto range = ParsedRequestRange::parse([request valueForHTTPHeaderField:@"Range"]);
196 if (!range)
197 return false;
198
199 auto expectedContentLength = expectedContentLengthFromData(*data);
200 data->taskData.add(task, makeUnique<Data::TaskData>(WTFMove(*range)));
201 giveResponseToTaskIfBytesInRangeReceived(task, *range, expectedContentLength, *data);
202
203 return true;
204}

206class RangeResponseGenerator::MediaResourceClient : public PlatformMediaResourceClient {
207public:
208 MediaResourceClient(RangeResponseGenerator& generator, URL&& url)
209 : m_generator(makeWeakPtr(generator))

220 {
221 RELEASE_ASSERT_NOT_REACHED();
222 completionHandler({ });
223 }

225 {
226 RELEASE_ASSERT_NOT_REACHED();
227 }
228 void accessControlCheckFailed(PlatformMediaResource&, const ResourceError&) final
229 {
230 RELEASE_ASSERT_NOT_REACHED();
231 }
232
233 bool shouldCacheResponse(PlatformMediaResource&, const ResourceResponse&) final
234 {
235 ASSERT(isMainThread());
236 return false;
237 }
238
239 void dataReceived(PlatformMediaResource&, const uint8_t* bytes, int length) final
240 {
241 ASSERT(isMainThread());
242 if (!m_generator)
243 return;
244 auto* data = m_generator->m_map.get(m_urlString);
245 if (!data)
246 return;
247 data->buffer->append(bytes, length);
248 m_generator->giveResponseToTasksWithFinishedRanges(*data);
249 }
250
251 void loadFailed(PlatformMediaResource&, const ResourceError& error) final
252 {
253 ASSERT(isMainThread());
254 if (!m_generator)
255 return;
256 auto data = m_generator->m_map.take(m_urlString);
257 if (!data)
258 return;
259 for (auto& task : data->taskData.keys())
260 [task resource:nullptr loadFailedWithError:error];
261 }
262
263 void loadFinished(PlatformMediaResource&, const NetworkLoadMetrics&) final
264 {
265 ASSERT(isMainThread());
266 auto generator = makeRefPtr(m_generator.get());
267 if (!generator)
268 return;
269 auto* data = generator->m_map.get(m_urlString);
270 if (!data)
271 return;
272 data->successfullyFinishedLoading = Data::SuccessfullyFinishedLoading::Yes;
273 data->resource = nullptr; // This line can delete this MediaResourceClient.
274 generator->giveResponseToTasksWithFinishedRanges(*data);
275 }
276
277 WeakPtr<RangeResponseGenerator> m_generator;
278 const String m_urlString;
279};

281bool RangeResponseGenerator::willSynthesizeRangeResponses(WebCoreNSURLSessionDataTask *task, PlatformMediaResource& resource, const ResourceResponse& response)
282{
283 ASSERT(isMainThread());
284 NSURLRequest *originalRequest = task.originalRequest;
285 if (!originalRequest.URL)
286 return false;
287 if (response.httpStatusCode() != 200)
288 return false;
289 if (!response.httpHeaderField(HTTPHeaderName::ContentRange).isEmpty())
290 return false;
291
292 auto parsedRequestRange = ParsedRequestRange::parse([originalRequest valueForHTTPHeaderField:@"Range"]);
293 if (!parsedRequestRange)
294 return false;
295
296 resource.setClient(adoptRef(new MediaResourceClient(*this, originalRequest.URL)));
297
298 m_map.ensure(originalRequest.URL.absoluteString, [&] {
299 return makeUnique<Data>(response, resource);
300 }).iterator->value->taskData.add(task, makeUnique<Data::TaskData>(WTFMove(*parsedRequestRange)));
301
302 return true;
303}
