34namespace WebCore {
35
36void BlobDataFileReference::generateReplacementFile()
37{
38 ASSERT(m_replacementPath.isNull());
39 ASSERT(m_replacementShouldBeGenerated);
40
41 prepareForFileAccess();
42
43 auto generatedFile = FileSystem::createTemporaryZipArchive(m_path);
44 if (!generatedFile.isNull())
45 m_replacementPath = WTFMove(generatedFile);
46
47 m_replacementShouldBeGenerated = false;
48 if (!m_replacementPath.isNull()) {
49 if (auto fileSize = FileSystem::fileSize(m_replacementPath))
50 m_size = *fileSize;
51 }
52
53 revokeFileAccess();
54}
55
56}
