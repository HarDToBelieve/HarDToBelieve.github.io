34void startObservingCookieChanges(const NetworkStorageSession& storageSession, WTF::Function<void()>&& callback)
35{
36 storageSession.cookieStorageObserver().startObserving(WTFMove(callback));
37}
38
39void stopObservingCookieChanges(const NetworkStorageSession& storageSession)
40{
41 storageSession.cookieStorageObserver().stopObserving();
42}
