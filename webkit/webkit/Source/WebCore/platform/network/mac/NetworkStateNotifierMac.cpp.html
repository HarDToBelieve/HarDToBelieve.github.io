35void NetworkStateNotifier::updateStateWithoutNotifying()
36{
37 auto key = adoptCF(SCDynamicStoreKeyCreateNetworkInterface(0, kSCDynamicStoreDomainState));
38 auto propertyList = adoptCF(SCDynamicStoreCopyValue(m_store.get(), key.get()));
39 if (!propertyList || CFGetTypeID(propertyList.get()) != CFDictionaryGetTypeID())
40 return;
41
42 auto netInterfaces = CFDictionaryGetValue((CFDictionaryRef)propertyList.get(), kSCDynamicStorePropNetInterfaces);
43 if (!netInterfaces || CFGetTypeID(netInterfaces) != CFArrayGetTypeID())
44 return;
45
46 for (CFIndex i = 0; i < CFArrayGetCount((CFArrayRef)netInterfaces); i++) {
47 auto interfaceName = (CFStringRef)CFArrayGetValueAtIndex((CFArrayRef)netInterfaces, i);
48 if (CFGetTypeID(interfaceName) != CFStringGetTypeID())
49 continue;
50
51 // Ignore the loopback interface.
52 if (CFStringHasPrefix(interfaceName, CFSTR("lo")))
53 continue;
54
55 // Ignore Parallels virtual interfaces on host machine as these are always up.
56 if (CFStringHasPrefix(interfaceName, CFSTR("vnic")))
57 continue;
58
59 // Ignore VMWare virtual interfaces on host machine as these are always up.
60 if (CFStringHasPrefix(interfaceName, CFSTR("vmnet")))
61 continue;
62
63 auto key = adoptCF(SCDynamicStoreKeyCreateNetworkInterfaceEntity(0, kSCDynamicStoreDomainState, interfaceName, kSCEntNetIPv4));
64 if (auto value = adoptCF(SCDynamicStoreCopyValue(m_store.get(), key.get()))) {
65 m_isOnLine = true;
66 return;
67 }
68 }
69
70 m_isOnLine = false;
71}

73void NetworkStateNotifier::startObserving()
74{
75 SCDynamicStoreContext context = { 0, this, 0, 0, 0 };
76 m_store = adoptCF(SCDynamicStoreCreate(0, CFSTR("com.apple.WebCore"), [] (SCDynamicStoreRef, CFArrayRef, void*) {
77 // Calling updateState() could be expensive so we coalesce calls with a timer.
78 singleton().updateStateSoon();
79 }, &context));
80 if (!m_store)
81 return;
82
83 auto source = adoptCF(SCDynamicStoreCreateRunLoopSource(0, m_store.get(), 0));
84 if (!source)
85 return;
86
87 CFRunLoopAddSource(CFRunLoopGetMain(), source.get(), kCFRunLoopCommonModes);
88
89 auto keys = adoptCF(CFArrayCreateMutable(0, 0, &kCFTypeArrayCallBacks));
90 CFArrayAppendValue(keys.get(), adoptCF(SCDynamicStoreKeyCreateNetworkGlobalEntity(0, kSCDynamicStoreDomainState, kSCEntNetIPv4)).get());
91 CFArrayAppendValue(keys.get(), adoptCF(SCDynamicStoreKeyCreateNetworkGlobalEntity(0, kSCDynamicStoreDomainState, kSCEntNetDNS)).get());
92
93 auto patterns = adoptCF(CFArrayCreateMutable(0, 0, &kCFTypeArrayCallBacks));
94 CFArrayAppendValue(patterns.get(), adoptCF(SCDynamicStoreKeyCreateNetworkInterfaceEntity(0, kSCDynamicStoreDomainState, kSCCompAnyRegex, kSCEntNetIPv4)).get());
95
96 SCDynamicStoreSetNotificationKeys(m_store.get(), keys.get(), patterns.get());
97}
