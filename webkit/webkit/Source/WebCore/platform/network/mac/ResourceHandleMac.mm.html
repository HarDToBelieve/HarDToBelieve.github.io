73static NSOperationQueue *operationQueueForAsyncClients()
74{
75 static auto queue = makeNeverDestroyed([] {
76 auto queue = adoptNS([[NSOperationQueue alloc] init]);
77 // Default concurrent operation count depends on current system workload, but delegate methods are mostly idling in IPC, so we can run as many as needed.
78 [queue setMaxConcurrentOperationCount:NSOperationQueueDefaultMaxConcurrentOperationCount];
79 return queue;
80 }());
81 return queue.get().get();
82}

108NSURLRequest *ResourceHandle::applySniffingPoliciesIfNeeded(NSURLRequest *request, bool shouldContentSniff, bool shouldContentEncodingSniff)
109{
110#if !USE(CFNETWORK_CONTENT_ENCODING_SNIFFING_OVERRIDE)
111 UNUSED_PARAM(shouldContentEncodingSniff);
112#endif
113
114 if (shouldContentSniff
115#if USE(CFNETWORK_CONTENT_ENCODING_SNIFFING_OVERRIDE)
116 && shouldContentEncodingSniff
117#endif
118 )
119 return request;
120
121 auto mutableRequest = adoptNS([request mutableCopy]);
122
123#if USE(CFNETWORK_CONTENT_ENCODING_SNIFFING_OVERRIDE)
124 if (!shouldContentEncodingSniff)
125 [mutableRequest _setProperty:@YES forKey:(__bridge NSString *)kCFURLRequestContentDecoderSkipURLCheck];
126#endif
127
128 if (!shouldContentSniff)
129 [mutableRequest _setProperty:@NO forKey:(__bridge NSString *)_kCFURLConnectionPropertyShouldSniff];
130
131 return mutableRequest.autorelease();
132}

136#if !PLATFORM(IOS_FAMILY)
137void ResourceHandle::createNSURLConnection(id delegate, bool shouldUseCredentialStorage, bool shouldContentSniff, bool shouldContentEncodingSniff, SchedulingBehavior schedulingBehavior)
138#else
139void ResourceHandle::createNSURLConnection(id delegate, bool shouldUseCredentialStorage, bool shouldContentSniff, bool shouldContentEncodingSniff, SchedulingBehavior schedulingBehavior, NSDictionary *connectionProperties)
140#endif
141{
142 // Credentials for ftp can only be passed in URL, the connection:didReceiveAuthenticationChallenge: delegate call won't be made.
143 if ((!d->m_user.isEmpty() || !d->m_password.isEmpty()) && !firstRequest().url().protocolIsInHTTPFamily()) {
144 URL urlWithCredentials(firstRequest().url());
145 urlWithCredentials.setUser(d->m_user);
146 urlWithCredentials.setPassword(d->m_password);
147 firstRequest().setURL(urlWithCredentials);
148 }
149
150 if (shouldUseCredentialStorage && firstRequest().url().protocolIsInHTTPFamily()) {
151 if (d->m_user.isEmpty() && d->m_password.isEmpty()) {
152 // <rdar://problem/7174050> - For URLs that match the paths of those previously challenged for HTTP Basic authentication,
153 // try and reuse the credential preemptively, as allowed by RFC 2617.
154 if (auto* networkStorageSession = d->m_context->storageSession())
155 d->m_initialCredential = networkStorageSession->credentialStorage().get(firstRequest().cachePartition(), firstRequest().url());
156 } else {
157 // If there is already a protection space known for the URL, update stored credentials before sending a request.
158 // This makes it possible to implement logout by sending an XMLHttpRequest with known incorrect credentials, and aborting it immediately
159 // (so that an authentication dialog doesn't pop up).
160 if (auto* networkStorageSession = d->m_context->storageSession())
161 networkStorageSession->credentialStorage().set(firstRequest().cachePartition(), Credential(d->m_user, d->m_password, CredentialPersistenceNone), firstRequest().url());
162 }
163 }
164
165 if (!d->m_initialCredential.isEmpty()) {
166 // FIXME: Support Digest authentication, and Proxy-Authorization.
167 applyBasicAuthorizationHeader(firstRequest(), d->m_initialCredential);
168 }
169
170 auto nsRequest = retainPtr(firstRequest().nsURLRequest(HTTPBodyUpdatePolicy::UpdateHTTPBody));
171 nsRequest = applySniffingPoliciesIfNeeded(nsRequest.get(), shouldContentSniff, shouldContentEncodingSniff);
172
173 if (d->m_storageSession)
174 nsRequest = copyRequestWithStorageSession(d->m_storageSession.get(), nsRequest.get());
175
176 ASSERT([NSURLConnection instancesRespondToSelector:@selector(_initWithRequest:delegate:usesCache:maxContentLength:startImmediately:connectionProperties:)]);
177
178#if PLATFORM(IOS_FAMILY)
179 // FIXME: This code is different from iOS code in ResourceHandleCFNet.cpp in that here we respect stream properties that were present in client properties.
180 NSDictionary *streamPropertiesFromClient = [connectionProperties objectForKey:@"kCFURLConnectionSocketStreamProperties"];
181 RetainPtr<NSMutableDictionary> streamProperties;
182 if (streamPropertiesFromClient)
183 streamProperties = adoptNS([streamPropertiesFromClient mutableCopy]);
184 else
185 streamProperties = retainPtr([NSMutableDictionary dictionary]);
186#else
187 auto streamProperties = retainPtr([NSMutableDictionary dictionary]);
188#endif
189
190 if (!shouldUseCredentialStorage) {
191 // Avoid using existing connections, because they may be already authenticated.
192 [streamProperties setObject:@"WebKitPrivateSession" forKey:@"_kCFURLConnectionSessionID"];
193 }
194
195 if (schedulingBehavior == SchedulingBehavior::Synchronous) {
196 // Synchronous requests should not be subject to regular connection count limit to avoid deadlocks.
197 // If we are using all available connections for async requests, and make a sync request, then prior
198 // requests may get stuck waiting for delegate calls while we are in nested run loop, and the sync
199 // request won't start because there are no available connections.
200 // Connections are grouped by their socket stream properties, with each group having a separate count.
201 [streamProperties setObject:@TRUE forKey:@"_WebKitSynchronousRequest"];
202 }
203
204 RetainPtr<CFDataRef> sourceApplicationAuditData = d->m_context->sourceApplicationAuditData();
205 if (sourceApplicationAuditData)
206 [streamProperties setObject:(__bridge NSData *)sourceApplicationAuditData.get() forKey:@"kCFStreamPropertySourceApplication"];
207
208#if PLATFORM(IOS_FAMILY)
209 NSMutableDictionary *propertyDictionary = [NSMutableDictionary dictionaryWithDictionary:connectionProperties];
210 [propertyDictionary setObject:streamProperties.get() forKey:@"kCFURLConnectionSocketStreamProperties"];
211 const bool usesCache = false;
212#if !PLATFORM(MACCATALYST)
213 if (synchronousWillSendRequestEnabled())
214 CFURLRequestSetShouldStartSynchronously([nsRequest _CFURLRequest], 1);
215#endif
216#else
217 NSMutableDictionary *propertyDictionary = [NSMutableDictionary dictionaryWithObject:streamProperties.get() forKey:@"kCFURLConnectionSocketStreamProperties"];
218 const bool usesCache = true;
219#endif
220 [propertyDictionary setObject:@{@"_kCFURLConnectionPropertyTimingDataOptions": @(_TimingDataOptionsEnableW3CNavigationTiming)} forKey:@"kCFURLConnectionURLConnectionProperties"];
221
222 // This is used to signal that to CFNetwork that this connection should be considered
223 // web content for purposes of App Transport Security.
224 [propertyDictionary setObject:@{@"NSAllowsArbitraryLoadsInWebContent": @YES} forKey:@"_kCFURLConnectionPropertyATSFrameworkOverrides"];
225
226ALLOW_DEPRECATED_DECLARATIONS_BEGIN
227 d->m_connection = adoptNS([[NSURLConnection alloc] _initWithRequest:nsRequest.get() delegate:delegate usesCache:usesCache maxContentLength:0 startImmediately:NO connectionProperties:propertyDictionary]);
228ALLOW_DEPRECATED_DECLARATIONS_END
229}

231bool ResourceHandle::start()
232{
233 if (!d->m_context)
234 return false;
235
236 BEGIN_BLOCK_OBJC_EXCEPTIONS
237
238 // If NetworkingContext is invalid then we are no longer attached to a Page,
239 // this must be an attempted load from an unload event handler, so let's just block it.
240 if (!d->m_context->isValid())
241 return false;
242
243 if (auto* networkStorageSession = d->m_context->storageSession())
244 d->m_storageSession = networkStorageSession->platformSession();
245
246 // FIXME: Do not use the sync version of shouldUseCredentialStorage when the client returns true from usesAsyncCallbacks.
247 bool shouldUseCredentialStorage = !client() || client()->shouldUseCredentialStorage(this);
248
249 SchedulingBehavior schedulingBehavior = client() && client()->loadingSynchronousXHR() ? SchedulingBehavior::Synchronous : SchedulingBehavior::Asynchronous;
250
251#if !PLATFORM(IOS_FAMILY)
252 createNSURLConnection(
253 ResourceHandle::makeDelegate(shouldUseCredentialStorage, nullptr),
254 shouldUseCredentialStorage,
255 d->m_shouldContentSniff || d->m_context->localFileContentSniffingEnabled(),
256 d->m_shouldContentEncodingSniff,
257 schedulingBehavior);
258#else
259 createNSURLConnection(
260 ResourceHandle::makeDelegate(shouldUseCredentialStorage, nullptr),
261 shouldUseCredentialStorage,
262 d->m_shouldContentSniff || d->m_context->localFileContentSniffingEnabled(),
263 d->m_shouldContentEncodingSniff,
264 schedulingBehavior,
265 (NSDictionary *)client()->connectionProperties(this).get());
266#endif
267
268 [connection() setDelegateQueue:operationQueueForAsyncClients()];
269 [connection() start];
270 d->m_startTime = MonotonicTime::now();
271
272 LOG(Network, "Handle %p starting connection %p for %@", this, connection(), firstRequest().nsURLRequest(HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody));
273
274 if (d->m_connection) {
275 if (d->m_defersLoading)
276 connection().defersCallbacks = YES;
277
278 return true;
279 }
280
281 END_BLOCK_OBJC_EXCEPTIONS
282
283 return false;
284}

286void ResourceHandle::cancel()
287{
288 LOG(Network, "Handle %p cancel connection %p", this, d->m_connection.get());
289
290 // Leaks were seen on HTTP tests without this; can be removed once <rdar://problem/6886937> is fixed.
291 if (d->m_currentMacChallenge)
292 [[d->m_currentMacChallenge sender] cancelAuthenticationChallenge:d->m_currentMacChallenge];
293
294 [d->m_connection.get() cancel];

354void ResourceHandle::platformLoadResourceSynchronously(NetworkingContext* context, const ResourceRequest& request, StoredCredentialsPolicy storedCredentialsPolicy, SecurityOrigin* sourceOrigin, ResourceError& error, ResourceResponse& response, Vector<uint8_t>& data)
355{
356 LOG(Network, "ResourceHandle::platformLoadResourceSynchronously:%@ storedCredentialsPolicy:%u", request.nsURLRequest(HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody), static_cast<unsigned>(storedCredentialsPolicy));
357
358 ASSERT(!request.isEmpty());
359
360 SynchronousLoaderClient client;
361 client.setAllowStoredCredentials(storedCredentialsPolicy == StoredCredentialsPolicy::Use);
362
363 bool defersLoading = false;
364 bool shouldContentSniff = true;
365 bool shouldContentEncodingSniff = true;
366 RefPtr<ResourceHandle> handle = adoptRef(new ResourceHandle(context, request, &client, defersLoading, shouldContentSniff, shouldContentEncodingSniff, sourceOrigin, false));
367
368 handle->d->m_storageSession = context->storageSession()->platformSession();
369
370 if (context && handle->d->m_scheduledFailureType != NoFailure) {
371 error = context->blockedError(request);
372 return;
373 }
374
375 bool shouldUseCredentialStorage = storedCredentialsPolicy == StoredCredentialsPolicy::Use;
376#if !PLATFORM(IOS_FAMILY)
377 handle->createNSURLConnection(
378 handle->makeDelegate(shouldUseCredentialStorage, &client.messageQueue()),
379 shouldUseCredentialStorage,
380 handle->shouldContentSniff() || context->localFileContentSniffingEnabled(),
381 handle->shouldContentEncodingSniff(),
382 SchedulingBehavior::Synchronous);
383#else
384 handle->createNSURLConnection(
385 handle->makeDelegate(shouldUseCredentialStorage, &client.messageQueue()), // A synchronous request cannot turn into a download, so there is no need to proxy the delegate.
386 shouldUseCredentialStorage,
387 handle->shouldContentSniff() || (context && context->localFileContentSniffingEnabled()),
388 handle->shouldContentEncodingSniff(),
389 SchedulingBehavior::Synchronous,
390 (NSDictionary *)handle->client()->connectionProperties(handle.get()).get());
391#endif
392
393 [handle->connection() setDelegateQueue:operationQueueForAsyncClients()];
394 [handle->connection() start];
395
396 do {
397 if (auto task = client.messageQueue().waitForMessage())
398 (*task)();
399 } while (!client.messageQueue().killed());
400
401 error = client.error();
402
403 [handle->connection() cancel];
404
405 if (error.isNull())
406 response = client.response();
407
408 data.swap(client.mutableData());
409}

411void ResourceHandle::willSendRequest(ResourceRequest&& request, ResourceResponse&& redirectResponse, CompletionHandler<void(ResourceRequest&&)>&& completionHandler)
412{
413 ASSERT(!redirectResponse.isNull());
414
415 if (redirectResponse.httpStatusCode() == 307) {
416 String lastHTTPMethod = d->m_lastHTTPMethod;
417 if (!equalIgnoringASCIICase(lastHTTPMethod, request.httpMethod())) {
418 request.setHTTPMethod(lastHTTPMethod);
419
420 FormData* body = d->m_firstRequest.httpBody();
421 if (!equalLettersIgnoringASCIICase(lastHTTPMethod, "get") && body && !body->isEmpty())
422 request.setHTTPBody(body);
423
424 String originalContentType = d->m_firstRequest.httpContentType();
425 if (!originalContentType.isEmpty())
426 request.setHTTPHeaderField(HTTPHeaderName::ContentType, originalContentType);
427 }
428 } else if (redirectResponse.httpStatusCode() == 303 && equalLettersIgnoringASCIICase(d->m_firstRequest.httpMethod(), "head")) // FIXME: (rdar://problem/13706454).
429 request.setHTTPMethod("HEAD"_s);
430
431 // Should not set Referer after a redirect from a secure resource to non-secure one.
432 if (!request.url().protocolIs("https") && protocolIs(request.httpReferrer(), "https") && d->m_context->shouldClearReferrerOnHTTPSToHTTPRedirect())
433 request.clearHTTPReferrer();
434
435 const URL& url = request.url();
436 d->m_user = url.user();
437 d->m_password = url.password();
438 d->m_lastHTTPMethod = request.httpMethod();
439 request.removeCredentials();
440
441 if (!protocolHostAndPortAreEqual(request.url(), redirectResponse.url())) {
442 // The network layer might carry over some headers from the original request that
443 // we want to strip here because the redirect is cross-origin.
444 request.clearHTTPAuthorization();
445 request.clearHTTPOrigin();
446 } else {
447 // Only consider applying authentication credentials if this is actually a redirect and the redirect
448 // URL didn't include credentials of its own.
449 if (d->m_user.isEmpty() && d->m_password.isEmpty() && !redirectResponse.isNull()) {
450 Credential credential;
451 if (auto* networkStorageSession = d->m_context->storageSession())
452 credential = networkStorageSession->credentialStorage().get(request.cachePartition(), request.url());
453 if (!credential.isEmpty()) {
454 d->m_initialCredential = credential;
455
456 // FIXME: Support Digest authentication, and Proxy-Authorization.
457 applyBasicAuthorizationHeader(request, d->m_initialCredential);
458 }
459 }
460 }
461
462 client()->willSendRequestAsync(this, WTFMove(request), WTFMove(redirectResponse), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (ResourceRequest&& request) mutable {
463 // Client call may not preserve the session, especially if the request is sent over IPC.
464 if (!request.isNull())
465 request.setStorageSession(d->m_storageSession.get());
466 completionHandler(WTFMove(request));
467 });
468}

470void ResourceHandle::didReceiveAuthenticationChallenge(const AuthenticationChallenge& challenge)
471{
472 ASSERT(!d->m_currentMacChallenge);
473 ASSERT(d->m_currentWebChallenge.isNull());
474 // Since NSURLConnection networking relies on keeping a reference to the original NSURLAuthenticationChallenge,
475 // we make sure that is actually present
476 ASSERT(challenge.nsURLAuthenticationChallenge());
477
478 // Proxy authentication is handled by CFNetwork internally. We can get here if the user cancels
479 // CFNetwork authentication dialog, and we shouldn't ask the client to display another one in that case.
480 if (challenge.protectionSpace().isProxy()) {
481 // Cannot use receivedRequestToContinueWithoutCredential(), because current challenge is not yet set.
482 [challenge.sender() continueWithoutCredentialForAuthenticationChallenge:challenge.nsURLAuthenticationChallenge()];
483 return;
484 }
485
486 if (tryHandlePasswordBasedAuthentication(challenge))
487 return;
488
489#if PLATFORM(IOS_FAMILY)
490 // If the challenge is for a proxy protection space, look for default credentials in
491 // the keychain. CFNetwork used to handle this until WebCore was changed to always
492 // return NO to -connectionShouldUseCredentialStorage: for <rdar://problem/7704943>.
493 if (!challenge.previousFailureCount() && challenge.protectionSpace().isProxy()) {
494 RELEASE_ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessCredentials));
495 NSURLAuthenticationChallenge *macChallenge = mac(challenge);
496 if (NSURLCredential *credential = [[NSURLCredentialStorage sharedCredentialStorage] defaultCredentialForProtectionSpace:[macChallenge protectionSpace]]) {
497 [challenge.sender() useCredential:credential forAuthenticationChallenge:macChallenge];
498 return;
499 }
500 }
501#endif // PLATFORM(IOS_FAMILY)
502
503 d->m_currentMacChallenge = challenge.nsURLAuthenticationChallenge();
504 d->m_currentWebChallenge = core(d->m_currentMacChallenge);
505 d->m_currentWebChallenge.setAuthenticationClient(this);
506
507 // FIXME: Several concurrent requests can return with the an authentication challenge for the same protection space.
508 // We should avoid making additional client calls for the same protection space when already waiting for the user,
509 // because typing the same credentials several times is annoying.
510 if (client())
511 client()->didReceiveAuthenticationChallenge(this, d->m_currentWebChallenge);
512 else {
513 clearAuthentication();
514 [challenge.sender() performDefaultHandlingForAuthenticationChallenge:challenge.nsURLAuthenticationChallenge()];
515 }
516}
517
518bool ResourceHandle::tryHandlePasswordBasedAuthentication(const AuthenticationChallenge& challenge)
519{
520 if (!challenge.protectionSpace().isPasswordBased())
521 return false;
522
523 if (!d->m_user.isNull() && !d->m_password.isNull()) {
524 auto credential = adoptNS([[NSURLCredential alloc] initWithUser:d->m_user
525 password:d->m_password persistence:NSURLCredentialPersistenceForSession]);
526 d->m_currentMacChallenge = challenge.nsURLAuthenticationChallenge();
527 d->m_currentWebChallenge = challenge;
528 receivedCredential(challenge, Credential(credential.get()));
529 // FIXME: Per the specification, the user shouldn't be asked for credentials if there were incorrect ones provided explicitly.
530 d->m_user = String();
531 d->m_password = String();
532 return true;
533 }
534
535 // FIXME: Do not use the sync version of shouldUseCredentialStorage when the client returns true from usesAsyncCallbacks.
536 if (!client() || client()->shouldUseCredentialStorage(this)) {
537 if (!d->m_initialCredential.isEmpty() || challenge.previousFailureCount()) {
538 // The stored credential wasn't accepted, stop using it.
539 // There is a race condition here, since a different credential might have already been stored by another ResourceHandle,
540 // but the observable effect should be very minor, if any.
541 if (auto* networkStorageSession = d->m_context->storageSession())
542 networkStorageSession->credentialStorage().remove(d->m_partition, challenge.protectionSpace());
543 }
544
545 if (!challenge.previousFailureCount()) {
546 Credential credential;
547 if (auto* networkStorageSession = d->m_context->storageSession())
548 credential = networkStorageSession->credentialStorage().get(d->m_partition, challenge.protectionSpace());
549 if (!credential.isEmpty() && credential != d->m_initialCredential) {
550 ASSERT(credential.persistence() == CredentialPersistenceNone);
551 if (challenge.failureResponse().httpStatusCode() == 401) {
552 // Store the credential back, possibly adding it as a default for this directory.
553 if (auto* networkStorageSession = d->m_context->storageSession())
554 networkStorageSession->credentialStorage().set(d->m_partition, credential, challenge.protectionSpace(), challenge.failureResponse().url());
555 }
556 [challenge.sender() useCredential:credential.nsCredential() forAuthenticationChallenge:mac(challenge)];
557 return true;
558 }
559 }
560 }
561
562 return false;
563}

575void ResourceHandle::receivedCredential(const AuthenticationChallenge& challenge, const Credential& credential)
576{
577 LOG(Network, "Handle %p receivedCredential", this);
578
579 ASSERT(!challenge.isNull());
580 if (challenge != d->m_currentWebChallenge)
581 return;
582
583 // FIXME: Support empty credentials. Currently, an empty credential cannot be stored in WebCore credential storage, as that's empty value for its map.
584 if (credential.isEmpty()) {
585 receivedRequestToContinueWithoutCredential(challenge);
586 return;
587 }
588
589 if (credential.persistence() == CredentialPersistenceForSession && challenge.protectionSpace().authenticationScheme() != ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested) {
590 // Manage per-session credentials internally, because once NSURLCredentialPersistenceForSession is used, there is no way
591 // to ignore it for a particular request (short of removing it altogether).
592 Credential webCredential(credential, CredentialPersistenceNone);
593 URL urlToStore;
594 if (challenge.failureResponse().httpStatusCode() == 401)
595 urlToStore = challenge.failureResponse().url();
596 if (auto* networkStorageSession = d->m_context->storageSession())
597 networkStorageSession->credentialStorage().set(d->m_partition, webCredential, ProtectionSpace([d->m_currentMacChallenge protectionSpace]), urlToStore);
598 [[d->m_currentMacChallenge sender] useCredential:webCredential.nsCredential() forAuthenticationChallenge:d->m_currentMacChallenge];
599 } else
600 [[d->m_currentMacChallenge sender] useCredential:credential.nsCredential() forAuthenticationChallenge:d->m_currentMacChallenge];
601
602 clearAuthentication();
603}
604
