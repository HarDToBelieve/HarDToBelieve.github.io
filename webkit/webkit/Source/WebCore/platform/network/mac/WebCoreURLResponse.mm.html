294void adjustMIMETypeIfNecessary(CFURLResponseRef cfResponse, bool isMainResourceLoad)
295{
296 UNUSED_PARAM(isMainResourceLoad);
297 RetainPtr<CFStringRef> result = CFURLResponseGetMIMEType(cfResponse);
298 RetainPtr<CFStringRef> originalResult = result;
299
300 if (!result) {
301 auto url = CFURLResponseGetURL(cfResponse);
302 if ([(__bridge NSURL *)url isFileURL]) {
303 RetainPtr<CFStringRef> extension = adoptCF(CFURLCopyPathExtension(url));
304 if (extension) {
305 // <rdar://problem/7007389> CoreTypes UTI map is missing 100+ file extensions that GateKeeper knew about
306 // When this radar is resolved, we can remove this file:// url specific code.
307 static NeverDestroyed<RetainPtr<CFDictionaryRef>> extensionMap = createExtensionToMIMETypeMap();
308 auto mutableExtension = adoptCF(CFStringCreateMutableCopy(kCFAllocatorDefault, 0, extension.get()));
309 CFStringLowercase(mutableExtension.get(), NULL);
310 extension = WTFMove(mutableExtension);
311 result = (CFStringRef)CFDictionaryGetValue(extensionMap.get().get(), extension.get());
312
313 if (!result) {
314 // If the Gatekeeper-based map doesn't have a MIME type, we'll try to figure out what it should be by
315 // looking up the file extension in the UTI maps.
316ALLOW_DEPRECATED_DECLARATIONS_BEGIN
317 RetainPtr<CFStringRef> uti = adoptCF(UTTypeCreatePreferredIdentifierForTag(kUTTagClassFilenameExtension, extension.get(), 0));
318ALLOW_DEPRECATED_DECLARATIONS_END
319 String MIMEType = MIMETypeFromUTITree(uti.get());
320 if (!MIMEType.isEmpty())
321 result = MIMEType.createCFString();
322 }
323 }
324 }
325 }
326
327 if (!result) {
328 static CFStringRef defaultMIMETypeString = WebCore::defaultMIMEType().createCFString().leakRef();
329 result = defaultMIMETypeString;
330 }
331
332 if (result != originalResult)
333 CFURLResponseSetMIMEType(cfResponse, result.get());
334}

337NSURLResponse *synthesizeRedirectResponseIfNecessary(NSURLRequest *currentRequest, NSURLRequest *newRequest, NSURLResponse *redirectResponse)
338{
339 if (redirectResponse)
340 return redirectResponse;
341
342 if ([[[newRequest URL] scheme] isEqualToString:[[currentRequest URL] scheme]] && ![newRequest _schemeWasUpgradedDueToDynamicHSTS])
343 return nil;
344
345 return retainPtr(ResourceResponse::syntheticRedirectResponse(URL([currentRequest URL]), URL([newRequest URL])).nsURLResponse()).autorelease();
346}
