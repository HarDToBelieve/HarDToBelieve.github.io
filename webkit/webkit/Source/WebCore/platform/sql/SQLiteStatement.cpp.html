59SQLiteStatement::~SQLiteStatement()

65int SQLiteStatement::step()
66{
67 Locker databaseLock { m_database.databaseMutex() };
68
69 int error = sqlite3_step(m_statement);
70 if (error != SQLITE_DONE && error != SQLITE_ROW)
71 LOG(SQLDatabase, "sqlite3_step failed (%i)\nError - %s", error, sqlite3_errmsg(m_database.sqlite3Handle()));
72
73 return error;
74}

76int SQLiteStatement::reset()
77{
78 int status = sqlite3_reset(m_statement);
79 sqlite3_clear_bindings(m_statement);
80 return status;
81}

83bool SQLiteStatement::executeCommand()

88int SQLiteStatement::bindBlob(int index, Span<const uint8_t> blob)
89{
90 ASSERT(index > 0);
91 ASSERT(static_cast<unsigned>(index) <= bindParameterCount());
92 ASSERT(blob.data() || !blob.size());
93 ASSERT(blob.size() >= 0);
94
95 return sqlite3_bind_blob(m_statement, index, blob.data(), blob.size(), SQLITE_TRANSIENT);
96}

98int SQLiteStatement::bindBlob(int index, const String& text)
99{
100 // String::characters() returns 0 for the empty string, which SQLite
101 // treats as a null, so we supply a non-null pointer for that case.
102 auto upconvertedCharacters = StringView(text).upconvertedCharacters();
103 UChar anyCharacter = 0;
104 const UChar* characters;
105 if (text.isEmpty() && !text.isNull())
106 characters = &anyCharacter;
107 else
108 characters = upconvertedCharacters;
109
110 return bindBlob(index, Span { reinterpret_cast<const uint8_t*>(characters), text.length() * sizeof(UChar) });
111}

113int SQLiteStatement::bindText(int index, StringView text)
114{
115 ASSERT(index > 0);
116 ASSERT(static_cast<unsigned>(index) <= bindParameterCount());
117
118 // Fast path when the input text is all ASCII.
119 if (text.is8Bit() && text.isAllASCII())
120 return sqlite3_bind_text(m_statement, index, text.length() ? reinterpret_cast<const char*>(text.characters8()) : "", text.length(), SQLITE_TRANSIENT);
121
122 auto utf8Text = text.utf8();
123 return sqlite3_bind_text(m_statement, index, utf8Text.data(), utf8Text.length(), SQLITE_TRANSIENT);
124}

158int SQLiteStatement::bindValue(int index, const SQLValue& value)
159{
160 return WTF::switchOn(value,
161 [&] (const std::nullptr_t&) { return bindNull(index); },
162 [&] (const String& string) { return bindText(index, string); },
163 [&] (double number) { return bindDouble(index, number); }
164 );
165}

183String SQLiteStatement::columnName(int col)
184{
185 ASSERT(col >= 0);
186 if (!hasStartedStepping() && step() != SQLITE_ROW)
187 return String();
188 if (columnCount() <= col)
189 return String();
190 return String::fromUTF8(sqlite3_column_name(m_statement, col));
191}
192
193SQLValue SQLiteStatement::columnValue(int col)
194{
195 ASSERT(col >= 0);
196 if (!hasStartedStepping() && step() != SQLITE_ROW)
197 return nullptr;
198 if (columnCount() <= col)
199 return nullptr;
200
201 // SQLite is typed per value. optional column types are
202 // "(mostly) ignored"
203 sqlite3_value* value = sqlite3_column_value(m_statement, col);
204 switch (sqlite3_value_type(value)) {
205 case SQLITE_INTEGER: // SQLValue and JS don't represent integers, so use FLOAT -case
206 case SQLITE_FLOAT:
207 return sqlite3_value_double(value);
208 case SQLITE_BLOB: // SQLValue and JS don't represent blobs, so use TEXT -case
209 case SQLITE_TEXT:
210 return String::fromUTF8(sqlite3_value_text(value), sqlite3_value_bytes(value));
211 case SQLITE_NULL:
212 return nullptr;
213 default:
214 break;
215 }
216
217 ASSERT_NOT_REACHED();
218 return nullptr;
219}

261String SQLiteStatement::columnBlobAsString(int col)
262{
263 ASSERT(col >= 0);
264
265 if (!hasStartedStepping() && step() != SQLITE_ROW)
266 return String();
267
268 if (columnCount() <= col)
269 return String();
270
271 const void* blob = sqlite3_column_blob(m_statement, col);
272 if (!blob)
273 return emptyString();
274
275 int size = sqlite3_column_bytes(m_statement, col);
276 if (size < 0)
277 return String();
278
279 ASSERT(!(size % sizeof(UChar)));
280 return String(static_cast<const UChar*>(blob), size / sizeof(UChar));
281}

283Vector<uint8_t> SQLiteStatement::columnBlob(int col)
284{
285 auto span = columnBlobAsSpan(col);

289Span<const uint8_t> SQLiteStatement::columnBlobAsSpan(int col)
290{
291 ASSERT(col >= 0);
292
293 if (!hasStartedStepping() && step() != SQLITE_ROW)
294 return { };
295
296 if (columnCount() <= col)
297 return { };
298
299 const void* blob = sqlite3_column_blob(m_statement, col);
300 if (!blob)
301 return { };
302
303 int blobSize = sqlite3_column_bytes(m_statement, col);
304 if (blobSize <= 0)
305 return { };
306
307 return { static_cast<const uint8_t*>(blob), static_cast<size_t>(blobSize) };
308}
