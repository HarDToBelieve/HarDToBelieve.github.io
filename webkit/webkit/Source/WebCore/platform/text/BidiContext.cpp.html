52Ref<BidiContext> BidiContext::create(unsigned char level, UCharDirection direction, bool override, BidiEmbeddingSource source, BidiContext* parent)
53{
54 ASSERT(direction == (level % 2 ? U_RIGHT_TO_LEFT : U_LEFT_TO_RIGHT));
55
56 if (parent)
57 return createUncached(level, direction, override, source, parent);
58
59 ASSERT(level <= 1);
60 if (!level) {
61 if (!override) {
62 static NeverDestroyed<RefPtr<BidiContext>> ltrContext;
63 static std::once_flag ltrContextOnceFlag;
64 std::call_once(ltrContextOnceFlag, [&]() {
65 ltrContext.get() = createUncached(0, U_LEFT_TO_RIGHT, false, FromStyleOrDOM, 0);
66 });
67 return *ltrContext.get();
68 }
69
70 static NeverDestroyed<RefPtr<BidiContext>> ltrOverrideContext;
71 static std::once_flag ltrOverrideContextOnceFlag;
72 std::call_once(ltrOverrideContextOnceFlag, [&]() {
73 ltrOverrideContext.get() = createUncached(0, U_LEFT_TO_RIGHT, true, FromStyleOrDOM, 0);
74 });
75 return *ltrOverrideContext.get();
76 }
77
78 if (!override) {
79 static NeverDestroyed<RefPtr<BidiContext>> rtlContext;
80 static std::once_flag rtlContextOnceFlag;
81 std::call_once(rtlContextOnceFlag, [&]() {
82 rtlContext.get() = createUncached(1, U_RIGHT_TO_LEFT, false, FromStyleOrDOM, 0);
83 });
84 return *rtlContext.get();
85 }
86
87 static NeverDestroyed<RefPtr<BidiContext>> rtlOverrideContext;
88 static std::once_flag rtlOverrideContextOnceFlag;
89 std::call_once(rtlOverrideContextOnceFlag, [&]() {
90 rtlOverrideContext.get() = createUncached(1, U_RIGHT_TO_LEFT, true, FromStyleOrDOM, 0);
91 });
92 return *rtlOverrideContext.get();
93}

95static inline Ref<BidiContext> copyContextAndRebaselineLevel(BidiContext& context, BidiContext* parent)
96{
97 auto newLevel = parent ? parent->level() : 0;
98 if (context.dir() == U_RIGHT_TO_LEFT)
99 newLevel = nextGreaterOddLevel(newLevel);
100 else if (parent)
101 newLevel = nextGreaterEvenLevel(newLevel);
102 return BidiContext::create(newLevel, context.dir(), context.override(), context.source(), parent);
103}

105// The BidiContext stack must be immutable -- they're re-used for re-layout after
106// DOM modification/editing -- so we copy all the non-Unicode contexts, and
107// recalculate their levels.
108Ref<BidiContext> BidiContext::copyStackRemovingUnicodeEmbeddingContexts()
109{
110 Vector<BidiContext*, 64> contexts;
111 for (auto* ancestor = this; ancestor; ancestor = ancestor->parent()) {
112 if (ancestor->source() != FromUnicode)
113 contexts.append(ancestor);
114 }
115 ASSERT(contexts.size());
116 auto topContext = copyContextAndRebaselineLevel(*contexts.last(), nullptr);
117 for (unsigned i = contexts.size() - 1; i; --i)
118 topContext = copyContextAndRebaselineLevel(*contexts[i - 1], topContext.ptr());
119 return topContext;
120}

122bool operator==(const BidiContext& c1, const BidiContext& c2)
123{
124 if (&c1 == &c2)
125 return true;
126 if (c1.level() != c2.level() || c1.override() != c2.override() || c1.dir() != c2.dir() || c1.source() != c2.source())
127 return false;
128 if (!c1.parent())
129 return !c2.parent();
130 return c2.parent() && *c1.parent() == *c2.parent();
131}
