32static inline bool isEmojiGroupCandidate(UChar32 character)
33{
34#define SYMBOLS_AND_PICTOGRAPHS_EXTENDED_A 298
35#if U_ICU_VERSION_MAJOR_NUM < 64
36#define UBLOCK_SYMBOLS_AND_PICTOGRAPHS_EXTENDED_A SYMBOLS_AND_PICTOGRAPHS_EXTENDED_A
37#else
38static_assert(UBLOCK_SYMBOLS_AND_PICTOGRAPHS_EXTENDED_A == SYMBOLS_AND_PICTOGRAPHS_EXTENDED_A);
39#endif
40
41 switch (static_cast<int>(ublock_getCode(character))) {
42 case UBLOCK_MISCELLANEOUS_SYMBOLS:
43 case UBLOCK_DINGBATS:
44 case UBLOCK_MISCELLANEOUS_SYMBOLS_AND_PICTOGRAPHS:
45 case UBLOCK_EMOTICONS:
46 case UBLOCK_TRANSPORT_AND_MAP_SYMBOLS:
47 case UBLOCK_SUPPLEMENTAL_SYMBOLS_AND_PICTOGRAPHS:
48 case UBLOCK_SYMBOLS_AND_PICTOGRAPHS_EXTENDED_A:
49 return true;
50 default:
51 return false;
52 }
53}

55static inline bool isEmojiFitzpatrickModifier(UChar32 character)
56{
57 // U+1F3FB - EMOJI MODIFIER FITZPATRICK TYPE-1-2
58 // U+1F3FC - EMOJI MODIFIER FITZPATRICK TYPE-3
59 // U+1F3FD - EMOJI MODIFIER FITZPATRICK TYPE-4
60 // U+1F3FE - EMOJI MODIFIER FITZPATRICK TYPE-5
61 // U+1F3FF - EMOJI MODIFIER FITZPATRICK TYPE-6
62
63 return character >= 0x1F3FB && character <= 0x1F3FF;
64}
65
66inline bool isVariationSelector(UChar32 character)
67{
68 return character >= 0xFE00 && character <= 0xFE0F;
69}
70
71inline bool isEmojiKeycapBase(UChar32 character)
72{
73 return (character >= '0' && character <= '9') || character == '#' || character == '*';
74}
75
76inline bool isEmojiRegionalIndicator(UChar32 character)
77{
78 return character >= 0x1F1E6 && character <= 0x1F1FF;
79}
80
81inline bool isEmojiWithPresentationByDefault(UChar32 character)
82{
83 return u_hasBinaryProperty(character, UCHAR_EMOJI_PRESENTATION);
84}
85
86inline bool isEmojiModifierBase(UChar32 character)
87{
88 return u_hasBinaryProperty(character, UCHAR_EMOJI_MODIFIER_BASE);
89}
90
91inline bool isDefaultIgnorableCodePoint(UChar32 character)
92{
93 return u_hasBinaryProperty(character, UCHAR_DEFAULT_IGNORABLE_CODE_POINT);
94}
95
96inline bool isControlCharacter(UChar32 character)
97{
98 return u_getIntPropertyValue(character, UCHAR_GENERAL_CATEGORY) == U_CONTROL_CHAR;
99}
