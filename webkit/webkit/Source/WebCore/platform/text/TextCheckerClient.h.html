28#pragma once
29
30#include "TextChecking.h"
31
32namespace WebCore {
33
34class VisibleSelection;
35
36class TextCheckerClient {
37public:
38 virtual ~TextCheckerClient() = default;
39
40 virtual bool shouldEraseMarkersAfterChangeSelection(TextCheckingType) const = 0;
41 virtual void ignoreWordInSpellDocument(const String&) = 0;
42 virtual void learnWord(const String&) = 0;
43 virtual void checkSpellingOfString(StringView, int* misspellingLocation, int* misspellingLength) = 0;
44 virtual String getAutoCorrectSuggestionForMisspelledWord(const String& misspelledWord) = 0;
45 virtual void checkGrammarOfString(StringView, Vector<GrammarDetail>&, int* badGrammarLocation, int* badGrammarLength) = 0;
46
47#if USE(UNIFIED_TEXT_CHECKING)
48 virtual Vector<TextCheckingResult> checkTextOfParagraph(StringView, OptionSet<TextCheckingType> checkingTypes, const VisibleSelection& currentSelection) = 0;
49#endif
50
51 // For spellcheckers that support multiple languages, it's often important to be able to identify the language in order to
52 // provide more accurate correction suggestions. Caller can pass in more text in "context" to aid such spellcheckers on language
53 // identification. Noramlly it's the text surrounding the "word" for which we are getting correction suggestions.
54 virtual void getGuessesForWord(const String& word, const String& context, const VisibleSelection& currentSelection, Vector<String>& guesses) = 0;
55 virtual void requestCheckingOfString(TextCheckingRequest&, const VisibleSelection& currentSelection) = 0;
56};

58}
