104void TextCodecICU::registerEncodingNames(EncodingNameRegistrar registrar)
105{
106 for (auto& encodingName : encodingNames) {
107 registrar(encodingName.name, encodingName.name);
108 for (size_t i = 0; i < encodingName.aliasCount; ++i)
109 registrar(encodingName.aliases[i], encodingName.name);
110 }
111}

113void TextCodecICU::registerCodecs(TextCodecRegistrar registrar)
114{
115 for (auto& encodingName : encodingNames) {
116 const char* name = encodingName.name;
117
118 // These encodings currently don't have standard names, so we need to register encoders manually.
119 // http://demo.icu-project.org/icu-bin/convexp
120 if (!strcmp(name, "windows-949")) {
121 registrar(name, [name] {
122 return makeUnique<TextCodecICU>(name, "windows-949-2000");
123 });
124 continue;
125 }
126 if (!strcmp(name, "x-mac-cyrillic")) {
127 registrar(name, [name] {
128 return makeUnique<TextCodecICU>(name, "macos-7_3-10.2");
129 });
130 continue;
131 }
132 if (!strcmp(name, "x-mac-greek")) {
133 registrar(name, [name] {
134 return makeUnique<TextCodecICU>(name, "macos-6_2-10.4");
135 });
136 continue;
137 }
138 if (!strcmp(name, "x-mac-centraleurroman")) {
139 registrar(name, [name] {
140 return makeUnique<TextCodecICU>(name, "macos-29-10.2");
141 });
142 continue;
143 }
144 if (!strcmp(name, "x-mac-turkish")) {
145 registrar(name, [name] {
146 return makeUnique<TextCodecICU>(name, "macos-35-10.2");
147 });
148 continue;
149 }
150
151 UErrorCode error = U_ZERO_ERROR;
152 const char* canonicalConverterName = ucnv_getCanonicalName(name, "IANA", &error);
153 ASSERT(U_SUCCESS(error));
154 registrar(name, [name, canonicalConverterName] {
155 return makeUnique<TextCodecICU>(name, canonicalConverterName);
156 });
157 }

174void TextCodecICU::createICUConverter() const
175{
176 ASSERT(!m_converter);
177
178 auto& cachedConverter = threadGlobalData().cachedConverterICU().converter;
179 if (cachedConverter) {
180 UErrorCode error = U_ZERO_ERROR;
181 const char* cachedConverterName = ucnv_getName(cachedConverter.get(), &error);
182 if (U_SUCCESS(error) && !strcmp(m_canonicalConverterName, cachedConverterName)) {
183 m_converter = WTFMove(cachedConverter);
184 return;
185 }
186 }
187
188 UErrorCode error = U_ZERO_ERROR;
189 m_converter = ICUConverterPtr { ucnv_open(m_canonicalConverterName, &error) };
190 if (m_converter)
191 ucnv_setFallback(m_converter.get(), true);
192}

194int TextCodecICU::decodeToBuffer(UChar* target, UChar* targetLimit, const char*& source, const char* sourceLimit, int32_t* offsets, bool flush, UErrorCode& error)
195{
196 UChar* targetStart = target;
197 error = U_ZERO_ERROR;
198 ucnv_toUnicode(m_converter.get(), &target, targetLimit, &source, sourceLimit, offsets, flush, &error);
199 return target - targetStart;
200}

202class ErrorCallbackSetter {
203public:
204 ErrorCallbackSetter(UConverter& converter, bool stopOnError)
205 : m_converter(converter)
206 , m_shouldStopOnEncodingErrors(stopOnError)
207 {
208 if (m_shouldStopOnEncodingErrors) {
209 UErrorCode err = U_ZERO_ERROR;
210 ucnv_setToUCallBack(&m_converter, UCNV_TO_U_CALLBACK_SUBSTITUTE, UCNV_SUB_STOP_ON_ILLEGAL, &m_savedAction, &m_savedContext, &err);
211 ASSERT(err == U_ZERO_ERROR);
212 }
213 }
214 ~ErrorCallbackSetter()
215 {
216 if (m_shouldStopOnEncodingErrors) {
217 UErrorCode err = U_ZERO_ERROR;
218 const void* oldContext;
219 UConverterToUCallback oldAction;
220 ucnv_setToUCallBack(&m_converter, m_savedAction, m_savedContext, &oldAction, &oldContext, &err);
221 ASSERT(oldAction == UCNV_TO_U_CALLBACK_SUBSTITUTE);
222 ASSERT(!strcmp(static_cast<const char*>(oldContext), UCNV_SUB_STOP_ON_ILLEGAL));
223 ASSERT(err == U_ZERO_ERROR);
224 }
225 }
226
227private:
228 UConverter& m_converter;
229 bool m_shouldStopOnEncodingErrors;
230 const void* m_savedContext;
231 UConverterToUCallback m_savedAction;
232};

234String TextCodecICU::decode(const char* bytes, size_t length, bool flush, bool stopOnError, bool& sawError)
235{
236 // Get a converter for the passed-in encoding.
237 if (!m_converter) {
238 createICUConverter();
239 if (!m_converter) {
240 LOG_ERROR("error creating ICU encoder even though encoding was in table");
241 sawError = true;
242 return { };
243 }
244 }
245
246 ErrorCallbackSetter callbackSetter(*m_converter, stopOnError);
247
248 StringBuilder result;
249
250 UChar buffer[ConversionBufferSize];
251 UChar* bufferLimit = buffer + ConversionBufferSize;
252 const char* source = reinterpret_cast<const char*>(bytes);
253 const char* sourceLimit = source + length;
254 int32_t* offsets = NULL;
255 UErrorCode err = U_ZERO_ERROR;
256
257 do {
258 int ucharsDecoded = decodeToBuffer(buffer, bufferLimit, source, sourceLimit, offsets, flush, err);
259 result.appendCharacters(buffer, ucharsDecoded);
260 } while (needsToGrowToProduceBuffer(err));
261
262 if (U_FAILURE(err)) {
263 // flush the converter so it can be reused, and not be bothered by this error.
264 do {
265 decodeToBuffer(buffer, bufferLimit, source, sourceLimit, offsets, true, err);
266 } while (source < sourceLimit);
267 sawError = true;
268 }
269
270 String resultString = result.toString();
271
272 return resultString;
273}

275// Invalid character handler when writing escaped entities for unrepresentable
276// characters. See the declaration of TextCodec::encode for more.
277static void urlEscapedEntityCallback(const void* context, UConverterFromUnicodeArgs* fromUArgs, const UChar* codeUnits, int32_t length,
278 UChar32 codePoint, UConverterCallbackReason reason, UErrorCode* error)
279{
280 if (reason == UCNV_UNASSIGNED) {
281 *error = U_ZERO_ERROR;
282 UnencodableReplacementArray entity;
283 int entityLen = TextCodec::getUnencodableReplacement(codePoint, UnencodableHandling::URLEncodedEntities, entity);
284 ucnv_cbFromUWriteBytes(fromUArgs, entity.data(), entityLen, 0, error);
285 } else
286 UCNV_FROM_U_CALLBACK_ESCAPE(context, fromUArgs, codeUnits, length, codePoint, reason, error);
287}

289Vector<uint8_t> TextCodecICU::encode(StringView string, UnencodableHandling handling) const
290{
291 if (string.isEmpty())
292 return { };
293
294 if (!m_converter) {
295 createICUConverter();
296 if (!m_converter)
297 return { };
298 }
299
300 // FIXME: We should see if there is "force ASCII range" mode in ICU;
301 // until then, we change the backslash into a yen sign.
302 // Encoding will change the yen sign back into a backslash.
303 String copy;
304 if (shouldShowBackslashAsCurrencySymbolIn(m_encodingName)) {
305 copy = string.toStringWithoutCopying();
306 copy.replace('\\', yenSign);
307 string = copy;
308 }
309
310 UErrorCode error;
311 switch (handling) {
312 case UnencodableHandling::QuestionMarks:
313 error = U_ZERO_ERROR;
314 ucnv_setSubstChars(m_converter.get(), "?", 1, &error);
315 if (U_FAILURE(error))
316 return { };
317 error = U_ZERO_ERROR;
318 ucnv_setFromUCallBack(m_converter.get(), UCNV_FROM_U_CALLBACK_SUBSTITUTE, 0, 0, 0, &error);
319 if (U_FAILURE(error))
320 return { };
321 break;
322 case UnencodableHandling::Entities:
323 error = U_ZERO_ERROR;
324 ucnv_setFromUCallBack(m_converter.get(), UCNV_FROM_U_CALLBACK_ESCAPE, UCNV_ESCAPE_XML_DEC, 0, 0, &error);
325 if (U_FAILURE(error))
326 return { };
327 break;
328 case UnencodableHandling::URLEncodedEntities:
329 error = U_ZERO_ERROR;
330 ucnv_setFromUCallBack(m_converter.get(), urlEscapedEntityCallback, 0, 0, 0, &error);
331 if (U_FAILURE(error))
332 return { };
333 break;
334 }
335
336 auto upconvertedCharacters = string.upconvertedCharacters();
337 auto* source = upconvertedCharacters.get();
338 auto* sourceLimit = source + string.length();
339
340 Vector<uint8_t> result;
341 do {
342 char buffer[ConversionBufferSize];
343 char* target = buffer;
344 char* targetLimit = target + ConversionBufferSize;
345 error = U_ZERO_ERROR;
346 ucnv_fromUnicode(m_converter.get(), &target, targetLimit, &source, sourceLimit, 0, true, &error);
347 result.append(reinterpret_cast<uint8_t*>(buffer), target - buffer);
348 } while (needsToGrowToProduceBuffer(error));
349 return result;
350}
