180 static std::once_flag once;
181 std::call_once(once, [&] {
182 auto* mutableEntries = new SingleByteEncodeTableEntry[size];
183 size_t j = 0;
184 for (uint8_t i = 0; i < std::size(decodeTable); i++) {
185 if (decodeTable[i] != replacementCharacter)
186 mutableEntries[j++] = { decodeTable[i], i + 0x80 };
187 }
188 ASSERT(j == size);
189 auto collection = WTF::makeIteratorRange(&mutableEntries[0], &mutableEntries[size]);
190 sortByFirst(collection);
191 ASSERT(sortedFirstsAreUnique(collection));
192 entries = mutableEntries;
193 });
194 return WTF::makeIteratorRange(&entries[0], &entries[size]);
195}

251// https://encoding.spec.whatwg.org/#single-byte-encoder
252static Vector<uint8_t> encode(const SingleByteEncodeTable& table, StringView string, Function<void(UChar32, Vector<uint8_t>&)>&& unencodableHandler)
253{
254 // FIXME: Consider adding an ASCII fast path like the one in TextCodecLatin1::decode.
255 Vector<uint8_t> result;
256 result.reserveInitialCapacity(string.length());
257 for (auto codePoint : string.codePoints()) {
258 if (isASCII(codePoint)) {
259 result.append(codePoint);
260 continue;
261 }
262 auto byte = findFirstInSortedPairs(table, codePoint);
263 if (!byte) {
264 unencodableHandler(codePoint, result);
265 continue;
266 }
267 result.append(*byte);
268 }
269 return result;
270}

272// https://encoding.spec.whatwg.org/#single-byte-decoder
273static String decode(const SingleByteDecodeTable& table, const uint8_t* bytes, size_t length, bool, bool stopOnError, bool& sawError)
274{
275 StringBuilder result;
276 result.reserveCapacity(length);
277 auto parseByte = [&] (uint8_t byte) {
278 if (isASCII(byte)) {
279 result.append(byte);
280 return;
281 }
282 UChar codePoint = table[byte - 0x80];
283 if (codePoint == replacementCharacter)
284 sawError = true;
285 result.append(codePoint);
286 };
287 if (stopOnError) {
288 for (size_t i = 0; i < length; i++) {
289 parseByte(bytes[i]);
290 if (sawError)
291 return result.toString();
292 }
293 } else {
294 for (size_t i = 0; i < length; i++)
295 parseByte(bytes[i]);
296 }
297 return result.toString();
298}
