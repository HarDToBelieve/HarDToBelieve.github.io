66// https://encoding.spec.whatwg.org/#shared-utf-16-decoder
67String TextCodecUTF16::decode(const char* bytes, size_t length, bool flush, bool, bool& sawError)
68{
69 const auto* p = reinterpret_cast<const uint8_t*>(bytes);
70 const auto* const end = p + length;
71 const auto* const endMinusOneOrNull = end ? end - 1 : nullptr;
72
73 StringBuilder result;
74 result.reserveCapacity(length / 2);
75
76 auto processCodeUnit = [&] (UChar codeUnit) {
77 if (std::exchange(m_shouldStripByteOrderMark, false) && codeUnit == byteOrderMark)
78 return;
79 if (m_leadSurrogate) {
80 auto leadSurrogate = *std::exchange(m_leadSurrogate, std::nullopt);
81 if (U16_IS_TRAIL(codeUnit)) {
82 result.appendCharacter(U16_GET_SUPPLEMENTARY(leadSurrogate, codeUnit));
83 return;
84 }
85 sawError = true;
86 result.append(replacementCharacter);
87 }
88 if (U16_IS_LEAD(codeUnit)) {
89 m_leadSurrogate = codeUnit;
90 return;
91 }
92 if (U16_IS_TRAIL(codeUnit)) {
93 sawError = true;
94 result.append(replacementCharacter);
95 return;
96 }
97 result.append(codeUnit);
98 };
99 auto processBytesLE = [&] (uint8_t first, uint8_t second) {
100 processCodeUnit(first | (second << 8));
101 };
102 auto processBytesBE = [&] (uint8_t first, uint8_t second) {
103 processCodeUnit((first << 8) | second);
104 };
105
106 if (m_leadByte && p < end) {
107 auto leadByte = *std::exchange(m_leadByte, std::nullopt);
108 if (m_littleEndian)
109 processBytesLE(leadByte, p[0]);
110 else
111 processBytesBE(leadByte, p[0]);
112 p++;
113 }
114
115 if (m_littleEndian) {
116 while (p < endMinusOneOrNull) {
117 processBytesLE(p[0], p[1]);
118 p += 2;
119 }
120 } else {
121 while (p < endMinusOneOrNull) {
122 processBytesBE(p[0], p[1]);
123 p += 2;
124 }
125 }
126
127 if (p && p == endMinusOneOrNull) {
128 ASSERT(!m_leadByte);
129 m_leadByte = p[0];
130 } else
131 ASSERT(!p || p == end);
132
133 if (flush) {
134 m_shouldStripByteOrderMark = false;
135 if (m_leadByte || m_leadSurrogate) {
136 m_leadByte = std::nullopt;
137 m_leadSurrogate = std::nullopt;
138 sawError = true;
139 result.append(replacementCharacter);
140 }
141 }
142
143 return result.toString();
144}

146Vector<uint8_t> TextCodecUTF16::encode(StringView string, UnencodableHandling) const
147{
148 Vector<uint8_t> result(WTF::checkedProduct<size_t>(string.length(), 2));
149 auto* bytes = result.data();
150
151 if (m_littleEndian) {
152 for (auto character : string.codeUnits()) {
153 *bytes++ = character;
154 *bytes++ = character >> 8;
155 }
156 } else {
157 for (auto character : string.codeUnits()) {
158 *bytes++ = character >> 8;
159 *bytes++ = character;
160 }
161 }
162
163 return result;
164}
