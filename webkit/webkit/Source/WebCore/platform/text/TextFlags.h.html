192 bool isAllNormal() const
193 {
194 return commonLigatures == FontVariantLigatures::Normal
195 && discretionaryLigatures == FontVariantLigatures::Normal
196 && historicalLigatures == FontVariantLigatures::Normal
197 && contextualAlternates == FontVariantLigatures::Normal
198 && position == FontVariantPosition::Normal
199 && caps == FontVariantCaps::Normal
200 && numericFigure == FontVariantNumericFigure::Normal
201 && numericSpacing == FontVariantNumericSpacing::Normal
202 && numericFraction == FontVariantNumericFraction::Normal
203 && numericOrdinal == FontVariantNumericOrdinal::Normal
204 && numericSlashedZero == FontVariantNumericSlashedZero::Normal
205 && alternates == FontVariantAlternates::Normal
206 && eastAsianVariant == FontVariantEastAsianVariant::Normal
207 && eastAsianWidth == FontVariantEastAsianWidth::Normal
208 && eastAsianRuby == FontVariantEastAsianRuby::Normal;
209 }
210
211 bool operator==(const FontVariantSettings& other) const
212 {
213 return commonLigatures == other.commonLigatures
214 && discretionaryLigatures == other.discretionaryLigatures
215 && historicalLigatures == other.historicalLigatures
216 && contextualAlternates == other.contextualAlternates
217 && position == other.position
218 && caps == other.caps
219 && numericFigure == other.numericFigure
220 && numericSpacing == other.numericSpacing
221 && numericFraction == other.numericFraction
222 && numericOrdinal == other.numericOrdinal
223 && numericSlashedZero == other.numericSlashedZero
224 && alternates == other.alternates
225 && eastAsianVariant == other.eastAsianVariant
226 && eastAsianWidth == other.eastAsianWidth
227 && eastAsianRuby == other.eastAsianRuby;
228 }
229
230 bool operator!=(const FontVariantSettings& other) const { return !(*this == other); }
231
232 unsigned uniqueValue() const
233 {
234 return static_cast<unsigned>(commonLigatures) << 26
235 | static_cast<unsigned>(discretionaryLigatures) << 24
236 | static_cast<unsigned>(historicalLigatures) << 22
237 | static_cast<unsigned>(contextualAlternates) << 20
238 | static_cast<unsigned>(position) << 18
239 | static_cast<unsigned>(caps) << 15
240 | static_cast<unsigned>(numericFigure) << 13
241 | static_cast<unsigned>(numericSpacing) << 11
242 | static_cast<unsigned>(numericFraction) << 9
243 | static_cast<unsigned>(numericOrdinal) << 8
244 | static_cast<unsigned>(numericSlashedZero) << 7
245 | static_cast<unsigned>(alternates) << 6
246 | static_cast<unsigned>(eastAsianVariant) << 3
247 | static_cast<unsigned>(eastAsianWidth) << 1
248 | static_cast<unsigned>(eastAsianRuby) << 0;
249 }
