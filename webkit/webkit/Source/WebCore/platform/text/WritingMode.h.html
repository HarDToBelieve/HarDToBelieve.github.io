37namespace WebCore {

39enum class WritingMode : uint8_t {
40 TopToBottom = 0, // horizontal-tb
41 BottomToTop = 1, // horizontal-bt
42 LeftToRight = 2, // vertical-lr
43 RightToLeft = 3, // vertical-rl
44};
45
46constexpr unsigned makeTextFlowInitalizer(WritingMode writingMode, TextDirection direction)
47{
48 return static_cast<unsigned>(writingMode) << 1 | static_cast<unsigned>(direction);
49}
50
51// Define the text flow in terms of the writing mode and the text direction. The first
52// part is the line growing direction and the second part is the block growing direction.
53enum TextFlow {
54 InlineEastBlockSouth = makeTextFlowInitalizer(WritingMode::TopToBottom, TextDirection::LTR),
55 InlineWestBlockSouth = makeTextFlowInitalizer(WritingMode::TopToBottom, TextDirection::RTL),
56 InlineEastBlockNorth = makeTextFlowInitalizer(WritingMode::BottomToTop, TextDirection::LTR),
57 InlineWestBlockNorth = makeTextFlowInitalizer(WritingMode::BottomToTop, TextDirection::RTL),
58 InlineSouthBlockEast = makeTextFlowInitalizer(WritingMode::LeftToRight, TextDirection::LTR),
59 InlineSouthBlockWest = makeTextFlowInitalizer(WritingMode::LeftToRight, TextDirection::RTL),
60 InlineNorthBlockEast = makeTextFlowInitalizer(WritingMode::RightToLeft, TextDirection::LTR),
61 InlineNorthBlockWest = makeTextFlowInitalizer(WritingMode::RightToLeft, TextDirection::RTL)
62};
63
64constexpr inline TextFlow makeTextFlow(WritingMode writingMode, TextDirection direction)
65{
66 return static_cast<TextFlow>(makeTextFlowInitalizer(writingMode, direction));
67}
68
69constexpr unsigned TextFlowReversedMask = 1;
70constexpr unsigned TextFlowFlippedMask = 2;
71constexpr unsigned TextFlowVerticalMask = 4;
72
73constexpr inline bool isReversedTextFlow(TextFlow textflow)
74{
75 return textflow & TextFlowReversedMask;
76}
77
78constexpr inline bool isFlippedTextFlow(TextFlow textflow)
79{
80 return textflow & TextFlowFlippedMask;
81}
82
83constexpr inline bool isVerticalTextFlow(TextFlow textflow)
84{
85 return textflow & TextFlowVerticalMask;
86}
87
88constexpr inline bool isFlippedLinesTextFlow(TextFlow textflow)
89{
90 return isFlippedTextFlow(textflow) != isVerticalTextFlow(textflow);
91}
92
93// Lines have vertical orientation; modes vertical-lr or vertical-rl.
94constexpr inline bool isVerticalWritingMode(WritingMode writingMode)
95{
96 return isVerticalTextFlow(makeTextFlow(writingMode, TextDirection::LTR));
97}
98
99// Block progression increases in the opposite direction to normal; modes vertical-rl or horizontal-bt.
100constexpr inline bool isFlippedWritingMode(WritingMode writingMode)
101{
102 return isFlippedTextFlow(makeTextFlow(writingMode, TextDirection::LTR));
103}
104
105// Lines have horizontal orientation; modes horizontal-tb or horizontal-bt.
106constexpr inline bool isHorizontalWritingMode(WritingMode writingMode)
107{
108 return !isVerticalWritingMode(writingMode);
109}
110
111// Bottom of the line occurs earlier in the block; modes vertical-lr or horizontal-bt.
112constexpr inline bool isFlippedLinesWritingMode(WritingMode writingMode)
113{
114 return isFlippedLinesTextFlow(makeTextFlow(writingMode, TextDirection::LTR));
115}

117enum class LogicalBoxSide : uint8_t {
118 BlockStart,
119 InlineEnd,
120 BlockEnd,
121 InlineStart
122};

124enum class BoxSide : uint8_t {
125 Top,
126 Right,
127 Bottom,
128 Left
129};
130
131enum class LogicalBoxCorner : uint8_t {
132 StartStart,
133 StartEnd,
134 EndStart,
135 EndEnd
136};
137
138// The mapping is with the first start/end giving the block axis side,
139// and the second the inline-axis side, i.e. patterned as 'border-block-inline-radius'.
140enum class BoxCorner : uint8_t {
141 TopLeft,
142 TopRight,
143 BottomRight,
144 BottomLeft
145};
146
147enum class LogicalBoxAxis : uint8_t {
148 Inline,
149 Block
150};
151
152enum class BoxAxis : uint8_t {
153 Horizontal,
154 Vertical
155};
156
157constexpr std::array<BoxSide, 4> allBoxSides = { BoxSide::Top, BoxSide::Right, BoxSide::Bottom, BoxSide::Left };
158
159constexpr inline bool isHorizontalPhysicalSide(BoxSide physicalSide)
160{
161 return physicalSide == BoxSide::Left || physicalSide == BoxSide::Right;
162}
163
164constexpr inline BoxSide mirrorPhysicalSide(BoxSide physicalSide)
165{
166 // top <-> bottom and left <-> right conversion
167 return static_cast<BoxSide>((static_cast<int>(physicalSide) + 2) % 4);
168}
169
170constexpr inline BoxSide rotatePhysicalSide(BoxSide physicalSide)
171{
172 // top <-> left and right <-> bottom conversion
173 bool horizontalSide = isHorizontalPhysicalSide(physicalSide);
174 return static_cast<BoxSide>((static_cast<int>(physicalSide) + (horizontalSide ? 1 : 3)) % 4);
175}
176
177constexpr inline BoxSide mapLogicalSideToPhysicalSide(TextFlow textflow, LogicalBoxSide logicalSide)
178{
179 BoxSide physicalSide = static_cast<BoxSide>(logicalSide);
180 bool horizontalSide = isHorizontalPhysicalSide(physicalSide);
181
182 if (isVerticalTextFlow(textflow))
183 physicalSide = rotatePhysicalSide(physicalSide);
184
185 if ((horizontalSide && isReversedTextFlow(textflow)) || (!horizontalSide && isFlippedTextFlow(textflow)))
186 physicalSide = mirrorPhysicalSide(physicalSide);
187
188 return physicalSide;
189}
190
191constexpr inline BoxSide mapLogicalSideToPhysicalSide(WritingMode writingMode, LogicalBoxSide logicalSide)
192{
193 // Set the direction such that side is mirrored if isFlippedWritingMode() is true
194 TextDirection direction = isFlippedWritingMode(writingMode) ? TextDirection::RTL : TextDirection::LTR;
195 return mapLogicalSideToPhysicalSide(makeTextFlow(writingMode, direction), logicalSide);
196}
197
198constexpr inline BoxCorner mapLogicalCornerToPhysicalCorner(TextFlow textflow, LogicalBoxCorner logicalBoxCorner)
199{
200 bool isBlockStart = logicalBoxCorner == LogicalBoxCorner::StartStart || logicalBoxCorner == LogicalBoxCorner::StartEnd;
201 bool isInlineStart = logicalBoxCorner == LogicalBoxCorner::StartStart || logicalBoxCorner == LogicalBoxCorner::EndStart;
202 if (isBlockStart == isFlippedTextFlow(textflow)) {
203 if (isInlineStart == isReversedTextFlow(textflow))
204 return BoxCorner::BottomRight;
205 } else if (isInlineStart != isReversedTextFlow(textflow))
206 return BoxCorner::TopLeft;
207 if (isBlockStart == isFlippedLinesTextFlow(textflow))
208 return BoxCorner::BottomLeft;
209 return BoxCorner::TopRight;
210}
211
212constexpr inline BoxAxis mapLogicalAxisToPhysicalAxis(TextFlow textflow, LogicalBoxAxis logicalAxis)
213{
214 if (isVerticalTextFlow(textflow))
215 return logicalAxis == LogicalBoxAxis::Inline ? BoxAxis::Vertical : BoxAxis::Horizontal;
216 return logicalAxis == LogicalBoxAxis::Inline ? BoxAxis::Horizontal : BoxAxis::Vertical;
217}
