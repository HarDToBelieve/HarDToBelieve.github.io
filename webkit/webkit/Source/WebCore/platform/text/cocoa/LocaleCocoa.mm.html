52static RetainPtr<NSDateFormatter> createDateTimeFormatter(NSLocale* locale, NSCalendar* calendar, NSDateFormatterStyle dateStyle, NSDateFormatterStyle timeStyle)
53{
54 auto formatter = adoptNS([[NSDateFormatter alloc] init]);
55 [formatter setLocale:locale];
56 [formatter setDateStyle:dateStyle];
57 [formatter setTimeStyle:timeStyle];
58 [formatter setTimeZone:[NSTimeZone timeZoneWithAbbreviation:@"UTC"]];
59 [formatter setCalendar:calendar];
60 return formatter;
61}

63LocaleCocoa::LocaleCocoa(const AtomString& locale)
64 : m_locale(adoptNS([[NSLocale alloc] initWithLocaleIdentifier:locale]))
65 , m_gregorianCalendar(adoptNS([[NSCalendar alloc] initWithCalendarIdentifier:NSCalendarIdentifierGregorian]))
66 , m_didInitializeNumberData(false)
67{
68 NSArray* availableLanguages = [NSLocale ISOLanguageCodes];
69 // NSLocale returns a lower case NSLocaleLanguageCode so we don't have care about case.
70 NSString* language = [m_locale.get() objectForKey:NSLocaleLanguageCode];
71 if ([availableLanguages indexOfObject:language] == NSNotFound)
72 m_locale = adoptNS([[NSLocale alloc] initWithLocaleIdentifier:defaultLanguage()]);
73 [m_gregorianCalendar.get() setLocale:m_locale.get()];
74}

86String LocaleCocoa::formatDateTime(const DateComponents& dateComponents, FormatType)
87{
88 double msec = dateComponents.millisecondsSinceEpoch();
89 DateComponentsType type = dateComponents.type();
90
91 // "week" type not supported.
92 ASSERT(type != DateComponentsType::Invalid);
93 if (type == DateComponentsType::Week)
94 return String();
95
96 // Incoming msec value is milliseconds since 1970-01-01 00:00:00 UTC. The 1970 epoch.
97 NSTimeInterval secondsSince1970 = (msec / 1000);
98 NSDate *date = [NSDate dateWithTimeIntervalSince1970:secondsSince1970];
99
100 // Return a formatted string.
101 NSDateFormatter *dateFormatter = localizedDateCache().formatterForDateType(type);
102 return [dateFormatter stringFromDate:date];
103}

105const Vector<String>& LocaleCocoa::monthLabels()
106{
107 if (!m_monthLabels.isEmpty())
108 return m_monthLabels;
109 m_monthLabels.reserveCapacity(12);
110 NSArray *array = [shortDateFormatter().get() monthSymbols];
111 if ([array count] == 12) {
112 for (unsigned i = 0; i < 12; ++i)
113 m_monthLabels.append(String([array objectAtIndex:i]));
114 return m_monthLabels;
115 }
116 for (auto& name : WTF::monthFullName)
117 m_monthLabels.append(name);
118 return m_monthLabels;
119}

266AtomString LocaleCocoa::canonicalLanguageIdentifierFromString(const AtomString& string)
267{
268 if (string.isEmpty())
269 return string;
270 return canonicalLocaleMap().ensure(string, [&] {
271 return [NSLocale canonicalLanguageIdentifierFromString:string];
272 }).iterator->value;

280void LocaleCocoa::initializeLocaleData()
281{
282 if (m_didInitializeNumberData)
283 return;
284 m_didInitializeNumberData = true;
285
286 RetainPtr<NSNumberFormatter> formatter = adoptNS([[NSNumberFormatter alloc] init]);
287 [formatter.get() setLocale:m_locale.get()];
288 [formatter.get() setNumberStyle:NSNumberFormatterDecimalStyle];
289 [formatter.get() setUsesGroupingSeparator:NO];
290
291 RetainPtr<NSNumber> sampleNumber = adoptNS([[NSNumber alloc] initWithDouble:9876543210]);
292 String nineToZero([formatter.get() stringFromNumber:sampleNumber.get()]);
293 if (nineToZero.length() != 10)
294 return;
295 Vector<String, DecimalSymbolsSize> symbols;
296 for (unsigned i = 0; i < 10; ++i)
297 symbols.append(nineToZero.substring(9 - i, 1));
298 ASSERT(symbols.size() == DecimalSeparatorIndex);
299 symbols.append([formatter.get() decimalSeparator]);
300 ASSERT(symbols.size() == GroupSeparatorIndex);
301 symbols.append([formatter.get() groupingSeparator]);
302 ASSERT(symbols.size() == DecimalSymbolsSize);
303
304 String positivePrefix([formatter.get() positivePrefix]);
305 String positiveSuffix([formatter.get() positiveSuffix]);
306 String negativePrefix([formatter.get() negativePrefix]);
307 String negativeSuffix([formatter.get() negativeSuffix]);
308 setLocaleData(symbols, positivePrefix, positiveSuffix, negativePrefix, negativeSuffix);
309}
