89 using FeatureList = Vector<ReferenceSpaceType>;
90 bool supports(SessionMode mode) const { return m_enabledFeaturesMap.contains(mode); }
91 void setSupportedFeatures(SessionMode mode, const FeatureList& features) { m_enabledFeaturesMap.set(mode, features); }
92 FeatureList supportedFeatures(SessionMode mode) const { return m_enabledFeaturesMap.get(mode); }
93 void setEnabledFeatures(SessionMode mode, const FeatureList& features) { m_enabledFeaturesMap.set(mode, features); }
94 FeatureList enabledFeatures(SessionMode mode) const { return m_enabledFeaturesMap.get(mode); }
95
96 virtual WebCore::IntSize recommendedResolution(SessionMode) { return { 1, 1 }; }
97
98 bool supportsOrientationTracking() const { return m_supportsOrientationTracking; }
99 bool supportsViewportScaling() const { return m_supportsViewportScaling; }
100
101 // Returns the value that the device's recommended resolution must be multiplied by
102 // to yield the device's native framebuffer resolution.
103 virtual double nativeFramebufferScalingFactor() const { return 1.0; }
104 // Returns the value that the device's recommended resolution must be multiplied by
105 // to yield the device's max framebuffer resolution. This resolution can be larger than
106 // the native resolution if the device supports supersampling.
107 virtual double maxFramebufferScalingFactor() const { return nativeFramebufferScalingFactor(); }

111 virtual void shutDownTrackingAndRendering() = 0;

115 // If this method returns true, that means the device will notify TrackingAndRenderingClient

360template<class Encoder>
361void Device::FrameData::View::encode(Encoder& encoder) const
362{
363 encoder << offset;
364
365 bool hasFov = WTF::holds_alternative<PlatformXR::Device::FrameData::Fov>(projection);
366 encoder << hasFov;
367 if (hasFov) {
368 encoder << WTF::get<PlatformXR::Device::FrameData::Fov>(projection);
369 return;
370 }
371
372 bool hasProjectionMatrix = WTF::holds_alternative<PlatformXR::Device::FrameData::ProjectionMatrix>(projection);
373 encoder << hasProjectionMatrix;
374 if (hasProjectionMatrix) {
375 for (float f : WTF::get<PlatformXR::Device::FrameData::ProjectionMatrix>(projection))
376 encoder << f;
377 return;
378 }
379
380 ASSERT(WTF::holds_alternative<std::nullptr_t>(projection));
381}
382
383template<class Decoder>
384std::optional<Device::FrameData::View> Device::FrameData::View::decode(Decoder& decoder)
385{
386 PlatformXR::Device::FrameData::View view;
387 if (!decoder.decode(view.offset))
388 return std::nullopt;
389
390 bool hasFov;
391 if (!decoder.decode(hasFov))
392 return std::nullopt;
393
394 if (hasFov) {
395 PlatformXR::Device::FrameData::Fov fov;
396 if (!decoder.decode(fov))
397 return std::nullopt;
398 view.projection = { WTFMove(fov) };
399 return view;
400 }
401
402 bool hasProjectionMatrix;
403 if (!decoder.decode(hasProjectionMatrix))
404 return std::nullopt;
405
406 if (hasProjectionMatrix) {
407 PlatformXR::Device::FrameData::ProjectionMatrix projectionMatrix;
408 for (size_t i = 0; i < PlatformXR::Device::FrameData::projectionMatrixSize; ++i) {
409 float f;
410 if (!decoder.decode(f))
411 return std::nullopt;
412 projectionMatrix[i] = f;
413 }
414 view.projection = { WTFMove(projectionMatrix) };
415 return view;
416 }
417
418 view.projection = { nullptr };
419 return view;
420}

441void Device::FrameData::LayerData::encode(Encoder& encoder) const
442{
443#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
444 WTF::MachSendRight surfaceSendRight = surface ? surface->createSendRight() : WTF::MachSendRight();

452template<class Decoder>
453std::optional<Device::FrameData::LayerData> Device::FrameData::LayerData::decode(Decoder& decoder)
454{
455 PlatformXR::Device::FrameData::LayerData layerData;
456#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
457 WTF::MachSendRight surfaceSendRight;
458 if (!decoder.decode(surfaceSendRight))
459 return std::nullopt;
460 layerData.surface = WebCore::IOSurface::createFromSendRight(WTFMove(surfaceSendRight), WebCore::DestinationColorSpace::SRGB());
461 if (!decoder.decode(layerData.isShared))
462 return std::nullopt;
463#else
464 if (!decoder.decode(layerData.opaqueTexture))
465 return std::nullopt;
466#endif
467 return layerData;
468}
