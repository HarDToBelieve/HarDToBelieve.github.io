36static bool isSessionReady(XrSessionState state)
37{
38 return state >= XR_SESSION_STATE_READY && state < XR_SESSION_STATE_STOPPING;
39}

41Ref<OpenXRDevice> OpenXRDevice::create(XrInstance instance, XrSystemId system, Ref<WorkQueue>&& queue, const OpenXRExtensions& extensions, CompletionHandler<void()>&& callback)
42{
43 auto device = adoptRef(*new OpenXRDevice(instance, system, WTFMove(queue), extensions));

56void OpenXRDevice::initialize(CompletionHandler<void()>&& callback)
57{
58 ASSERT(isMainThread());
59 m_queue.dispatch([this, protectedThis = makeRef(*this), callback = WTFMove(callback)]() mutable {
60 auto systemProperties = createStructure<XrSystemProperties, XR_TYPE_SYSTEM_PROPERTIES>();
61 auto result = xrGetSystemProperties(m_instance, m_systemId, &systemProperties);
62 if (XR_SUCCEEDED(result))
63 m_supportsOrientationTracking = systemProperties.trackingProperties.orientationTracking == XR_TRUE;
64#if !LOG_DISABLED
65 else
66 LOG(XR, "xrGetSystemProperties(): error %s\n", resultToString(result, m_instance).utf8().data());
67 LOG(XR, "Found XRSystem %lu: \"%s\", vendor ID %d\n", systemProperties.systemId, systemProperties.systemName, systemProperties.vendorId);
68#endif
69
70 collectSupportedSessionModes();
71 collectConfigurationViews();
72
73 callOnMainThread(WTFMove(callback));
74 });
75}

77WebCore::IntSize OpenXRDevice::recommendedResolution(SessionMode mode)
78{
79 auto configType = toXrViewConfigurationType(mode);
80 auto viewsIterator = m_configurationViews.find(configType);
81 if (viewsIterator != m_configurationViews.end())
82 return { static_cast<int>(2 * viewsIterator->value[0].recommendedImageRectWidth), static_cast<int>(viewsIterator->value[0].recommendedImageRectHeight) };
83 return Device::recommendedResolution(mode);
84}

86void OpenXRDevice::initializeTrackingAndRendering(SessionMode mode)
87{
88 m_queue.dispatch([this, protectedThis = makeRef(*this), mode]() {
89 ASSERT(m_instance != XR_NULL_HANDLE);
90 ASSERT(m_session == XR_NULL_HANDLE);
91 ASSERT(m_extensions.methods().xrGetOpenGLGraphicsRequirementsKHR);
92
93 m_currentViewConfigurationType = toXrViewConfigurationType(mode);
94 ASSERT(m_configurationViews.contains(m_currentViewConfigurationType));
95
96 // https://www.khronos.org/registry/OpenXR/specs/1.0/man/html/xrGetOpenGLGraphicsRequirementsKHR.html
97 // OpenXR requires to call xrGetOpenGLGraphicsRequirementsKHR before creating a session.
98 auto requirements = createStructure<XrGraphicsRequirementsOpenGLKHR, XR_TYPE_GRAPHICS_REQUIREMENTS_OPENGL_KHR>();
99 auto result = m_extensions.methods().xrGetOpenGLGraphicsRequirementsKHR(m_instance, m_systemId, &requirements);
100 RETURN_IF_FAILED(result, "xrGetOpenGLGraphicsRequirementsKHR", m_instance);
101
102 m_graphicsBinding = createStructure<XrGraphicsBindingEGLMNDX, XR_TYPE_GRAPHICS_BINDING_EGL_MNDX>();
103 m_egl = GLContextEGL::createSharingContext(PlatformDisplay::sharedDisplay());
104 if (!m_egl) {
105 LOG(XR, "Failed to create EGL context");
106 return;
107 }
108
109 auto& context = static_cast<GLContext&>(*m_egl);
110 context.makeContextCurrent();
111
112 GraphicsContextGLAttributes attributes;
113 attributes.depth = false;
114 attributes.stencil = false;
115 attributes.antialias = false;
116
117 m_gl = GraphicsContextGL::create(attributes, nullptr);
118 if (!m_gl) {
119 LOG(XR, "Failed to create a valid GraphicsContextGL");
120 return;
121 }
122
123 m_graphicsBinding.display = PlatformDisplay::sharedDisplay().eglDisplay();
124 m_graphicsBinding.context = context.platformContext();
125 m_graphicsBinding.config = m_egl->config();
126 m_graphicsBinding.getProcAddress = m_extensions.methods().getProcAddressFunc;
127
128 // Create the session.
129 auto sessionCreateInfo = createStructure<XrSessionCreateInfo, XR_TYPE_SESSION_CREATE_INFO>();
130 sessionCreateInfo.systemId = m_systemId;
131 sessionCreateInfo.next = &m_graphicsBinding;
132
133 result = xrCreateSession(m_instance, &sessionCreateInfo, &m_session);
134 RETURN_IF_FAILED(result, "xrCreateSession", m_instance);
135
136 // Create the default reference spaces
137 m_localSpace = createReferenceSpace(XR_REFERENCE_SPACE_TYPE_LOCAL);
138 m_viewSpace = createReferenceSpace(XR_REFERENCE_SPACE_TYPE_VIEW);
139
140 // Initialize input tracking
141 m_input = OpenXRInput::create(m_instance, m_session, m_localSpace);
142 });
143}

145void OpenXRDevice::shutDownTrackingAndRendering()
146{
147 m_queue.dispatch([this, protectedThis = makeRef(*this)]() {
148 if (m_session == XR_NULL_HANDLE)
149 return;
150
151 // xrRequestExitSession() will transition the session to STOPPED state.
152 // If the session was not running we have to reset the session ourselves.
153 if (XR_FAILED(xrRequestExitSession(m_session))) {
154 resetSession();
155 return;
156 }
157
158 // OpenXR needs to wait for the XR_SESSION_STATE_STOPPING state to properly end the session.
159 waitUntilStopping();
160 });
161}

163void OpenXRDevice::initializeReferenceSpace(PlatformXR::ReferenceSpaceType spaceType)
164{
165 if ((spaceType == ReferenceSpaceType::LocalFloor || spaceType == ReferenceSpaceType::BoundedFloor) && m_stageSpace == XR_NULL_HANDLE)
166 m_stageSpace = createReferenceSpace(XR_REFERENCE_SPACE_TYPE_STAGE);
167 if (spaceType == ReferenceSpaceType::BoundedFloor)
168 updateStageParameters();

171void OpenXRDevice::requestFrame(RequestFrameCallback&& callback)
172{
173 m_queue.dispatch([this, protectedThis = makeRef(*this), callback = WTFMove(callback)]() mutable {
174 pollEvents();
175 if (!isSessionReady(m_sessionState)) {
176 callOnMainThread([callback = WTFMove(callback)]() mutable {
177 // Device not ready or stopping. Report frameData with invalid tracking.
178 callback({ });
179 });
180 return;
181 }
182
183 m_frameState = createStructure<XrFrameState, XR_TYPE_FRAME_STATE>();
184 auto frameWaitInfo = createStructure<XrFrameWaitInfo, XR_TYPE_FRAME_WAIT_INFO>();
185 auto result = xrWaitFrame(m_session, &frameWaitInfo, &m_frameState);
186 RETURN_IF_FAILED(result, "xrWaitFrame", m_instance);
187
188 auto frameBeginInfo = createStructure<XrFrameBeginInfo, XR_TYPE_FRAME_BEGIN_INFO>();
189 result = xrBeginFrame(m_session, &frameBeginInfo);
190 RETURN_IF_FAILED(result, "xrBeginFrame", m_instance);
191
192 Device::FrameData frameData;
193 frameData.predictedDisplayTime = m_frameState.predictedDisplayTime;
194 frameData.shouldRender = m_frameState.shouldRender;
195 frameData.stageParameters = m_stageParameters;
196
197 ASSERT(m_configurationViews.contains(m_currentViewConfigurationType));
198 const auto& configurationView = m_configurationViews.get(m_currentViewConfigurationType);
199
200 uint32_t viewCount = configurationView.size();
201 m_frameViews.fill(createStructure<XrView, XR_TYPE_VIEW>(), viewCount);
202
203 if (m_frameState.shouldRender) {
204 // Query head location
205 auto location = createStructure<XrSpaceLocation, XR_TYPE_SPACE_LOCATION>();
206 xrLocateSpace(m_viewSpace, m_localSpace, m_frameState.predictedDisplayTime, &location);
207 frameData.isTrackingValid = location.locationFlags & XR_SPACE_LOCATION_ORIENTATION_VALID_BIT;
208 frameData.isPositionValid = location.locationFlags & XR_SPACE_LOCATION_POSITION_VALID_BIT;
209 frameData.isPositionEmulated = location.locationFlags & XR_SPACE_LOCATION_POSITION_TRACKED_BIT;
210
211 if (frameData.isTrackingValid)
212 frameData.origin = XrPosefToPose(location.pose);
213
214 auto viewLocateInfo = createStructure<XrViewLocateInfo, XR_TYPE_VIEW_LOCATE_INFO>();
215 viewLocateInfo.displayTime = m_frameState.predictedDisplayTime;
216 viewLocateInfo.space = m_viewSpace;
217
218 auto viewState = createStructure<XrViewState, XR_TYPE_VIEW_STATE>();
219 uint32_t viewCountOutput;
220 result = xrLocateViews(m_session, &viewLocateInfo, &viewState, viewCount, &viewCountOutput, m_frameViews.data());
221 if (!XR_FAILED(result)) {
222 for (auto& view : m_frameViews)
223 frameData.views.append(xrViewToPose(view));
224 }
225
226 // Query floor transform
227 if (m_stageSpace != XR_NULL_HANDLE) {
228 auto floorLocation = createStructure<XrSpaceLocation, XR_TYPE_SPACE_LOCATION>();
229 xrLocateSpace(m_stageSpace, m_localSpace, m_frameState.predictedDisplayTime, &floorLocation);
230 frameData.floorTransform = { XrPosefToPose(floorLocation.pose) };
231 }
232
233 for (auto& layer : m_layers) {
234 auto layerData = layer.value->startFrame();
235 if (layerData)
236 frameData.layers.add(layer.key, *layerData);
237 }
238
239 if (m_input)
240 frameData.inputSources = m_input->collectInputSources(m_frameState);
241
242 } else {
243 // https://www.khronos.org/registry/OpenXR/specs/1.0/man/html/XrFrameState.html
244 // When shouldRender is false the application should avoid heavy GPU work where possible,
245 // for example by skipping layer rendering and then omitting those layers when calling xrEndFrame.
246 // In this case we don't need to wait for OpenXRDevice::submitFrame to finish the frame.
247 auto frameEndInfo = createStructure<XrFrameEndInfo, XR_TYPE_FRAME_END_INFO>();
248 frameEndInfo.displayTime = m_frameState.predictedDisplayTime;
249 frameEndInfo.environmentBlendMode = XR_ENVIRONMENT_BLEND_MODE_OPAQUE;
250 frameEndInfo.layerCount = 0;
251 frameEndInfo.layers = nullptr;
252 LOG_IF_FAILED(xrEndFrame(m_session, &frameEndInfo), "xrEndFrame", m_instance);
253 }
254
255 callOnMainThread([frameData = WTFMove(frameData), callback = WTFMove(callback)]() mutable {
256 callback(WTFMove(frameData));
257 });
258 });
259}

261void OpenXRDevice::submitFrame(Vector<Device::Layer>&& layers)
262{
263 m_queue.dispatch([this, protectedThis = makeRef(*this), layers = WTFMove(layers)]() mutable {
264 ASSERT(m_frameState.shouldRender);
265 Vector<const XrCompositionLayerBaseHeader*> frameEndLayers;
266 if (m_frameState.shouldRender) {
267 for (auto& layer : layers) {
268 auto it = m_layers.find(layer.handle);
269 if (it == m_layers.end()) {
270 LOG(XR, "Didn't find a OpenXRLayer with %d handle", layer.handle);
271 continue;
272 }
273 auto header = it->value->endFrame(layer, m_viewSpace, m_frameViews);
274 if (!header) {
275 LOG(XR, "endFrame() call failed in OpenXRLayer with %d handle", layer.handle);
276 continue;
277 }
278
279 frameEndLayers.append(header);
280 }
281 }
282
283 auto frameEndInfo = createStructure<XrFrameEndInfo, XR_TYPE_FRAME_END_INFO>();
284 frameEndInfo.displayTime = m_frameState.predictedDisplayTime;
285 frameEndInfo.environmentBlendMode = XR_ENVIRONMENT_BLEND_MODE_OPAQUE;
286 frameEndInfo.layerCount = frameEndLayers.size();
287 frameEndInfo.layers = frameEndLayers.data();
288 auto result = xrEndFrame(m_session, &frameEndInfo);
289 RETURN_IF_FAILED(result, "xrEndFrame", m_instance);
290 });
291}

293Vector<Device::ViewData> OpenXRDevice::views(SessionMode mode) const
294{
295 Vector<Device::ViewData> views;
296 auto configurationType = toXrViewConfigurationType(mode);
297
298 if (configurationType == XR_VIEW_CONFIGURATION_TYPE_PRIMARY_MONO)
299 views.append({ .active = true, .eye = Eye::None });
300 else {
301 ASSERT(configurationType == XR_VIEW_CONFIGURATION_TYPE_PRIMARY_STEREO);
302 views.append({ .active = true, .eye = Eye::Left });
303 views.append({ .active = true, .eye = Eye::Right });
304 }
305 return views;
306}

308std::optional<LayerHandle> OpenXRDevice::createLayerProjection(uint32_t width, uint32_t height, bool alpha)
309{
310 std::optional<LayerHandle> handle;
311
312 BinarySemaphore semaphore;
313 m_queue.dispatch([this, width, height, alpha, &handle, &semaphore]() mutable {
314 auto format = alpha ? GL_RGBA8 : GL_RGB8;
315 int64_t sampleCount = 1;
316
317 auto it = m_configurationViews.find(m_currentViewConfigurationType);
318 if (it != m_configurationViews.end())
319 sampleCount = it->value[0].recommendedSwapchainSampleCount;
320
321 auto layer = OpenXRLayerProjection::create(m_instance, m_session, width, height, format, sampleCount);
322 if (layer) {
323 handle = generateLayerHandle();
324 m_layers.add(*handle, WTFMove(layer));
325 }
326
327 semaphore.signal();
328 });
329 semaphore.wait();
330
331 return handle;
332}

339Device::FeatureList OpenXRDevice::collectSupportedFeatures() const
340{
341 Device::FeatureList features;
342
343 // https://www.khronos.org/registry/OpenXR/specs/1.0/man/html/XrReferenceSpaceType.html
344 // OpenXR runtimes must support Viewer and Local spaces.
345 features.append(ReferenceSpaceType::Viewer);
346 features.append(ReferenceSpaceType::Local);
347
348 // Mark LocalFloor as supported regardless if XR_REFERENCE_SPACE_TYPE_STAGE is available.
349 // The spec uses a estimated height if we don't provide a floor transform in frameData.
350 features.append(ReferenceSpaceType::LocalFloor);
351
352 // Mark BoundedFloor as supported regardless if XR_REFERENCE_SPACE_TYPE_STAGE is available.
353 // The spec allows reporting an empty array if xrGetReferenceSpaceBoundsRect fails.
354 features.append(ReferenceSpaceType::BoundedFloor);
355
356 if (m_extensions.isExtensionSupported(XR_MSFT_UNBOUNDED_REFERENCE_SPACE_EXTENSION_NAME))
357 features.append(ReferenceSpaceType::Unbounded);
358
359 return features;
360}

362void OpenXRDevice::collectSupportedSessionModes()
363{
364 ASSERT(&RunLoop::current() == &m_queue.runLoop());
365 uint32_t viewConfigurationCount;
366 auto result = xrEnumerateViewConfigurations(m_instance, m_systemId, 0, &viewConfigurationCount, nullptr);
367 RETURN_IF_FAILED(result, "xrEnumerateViewConfigurations", m_instance);
368
369 XrViewConfigurationType viewConfigurations[viewConfigurationCount];
370 result = xrEnumerateViewConfigurations(m_instance, m_systemId, viewConfigurationCount, &viewConfigurationCount, viewConfigurations);
371 RETURN_IF_FAILED(result, "xrEnumerateViewConfigurations", m_instance);
372
373 FeatureList features = collectSupportedFeatures();
374 for (uint32_t i = 0; i < viewConfigurationCount; ++i) {
375 auto viewConfigurationProperties = createStructure<XrViewConfigurationProperties, XR_TYPE_VIEW_CONFIGURATION_PROPERTIES>();
376 result = xrGetViewConfigurationProperties(m_instance, m_systemId, viewConfigurations[i], &viewConfigurationProperties);
377 if (result != XR_SUCCESS) {
378 LOG(XR, "xrGetViewConfigurationProperties(): error %s\n", resultToString(result, m_instance).utf8().data());
379 continue;
380 }
381 auto configType = viewConfigurationProperties.viewConfigurationType;
382 switch (configType) {
383 case XR_VIEW_CONFIGURATION_TYPE_PRIMARY_MONO:
384 setSupportedFeatures(SessionMode::Inline, features);
385 break;
386 case XR_VIEW_CONFIGURATION_TYPE_PRIMARY_STEREO:
387 setSupportedFeatures(SessionMode::ImmersiveVr, features);
388 break;
389 default:
390 continue;
391 };
392 m_viewConfigurationProperties.add(configType, WTFMove(viewConfigurationProperties));
393 }
394}

396void OpenXRDevice::collectConfigurationViews()
397{
398 ASSERT(&RunLoop::current() == &m_queue.runLoop());
399 for (auto& config : m_viewConfigurationProperties.values()) {
400 uint32_t viewCount;
401 auto configType = config.viewConfigurationType;
402 auto result = xrEnumerateViewConfigurationViews(m_instance, m_systemId, configType, 0, &viewCount, nullptr);
403 if (result != XR_SUCCESS) {
404 LOG(XR, "%s %s: %s\n", __func__, "xrEnumerateViewConfigurationViews", resultToString(result, m_instance).utf8().data());
405 continue;
406 }
407
408 Vector<XrViewConfigurationView> configViews(viewCount,
409 [] {
410 XrViewConfigurationView object;
411 std::memset(&object, 0, sizeof(XrViewConfigurationView));
412 object.type = XR_TYPE_VIEW_CONFIGURATION_VIEW;
413 return object;
414 }());
415 result = xrEnumerateViewConfigurationViews(m_instance, m_systemId, configType, viewCount, &viewCount, configViews.data());
416 if (result != XR_SUCCESS)
417 continue;
418 m_configurationViews.add(configType, WTFMove(configViews));
419 }
420}

422XrSpace OpenXRDevice::createReferenceSpace(XrReferenceSpaceType type)
423{
424 ASSERT(&RunLoop::current() == &m_queue.runLoop());
425 ASSERT(m_session != XR_NULL_HANDLE);
426 ASSERT(m_instance != XR_NULL_HANDLE);
427
428 XrPosef identityPose {
429 .orientation = { .x = 0, .y = 0, .z = 0, .w = 1.0 },
430 .position = { .x = 0, .y = 0, .z = 0 }
431 };
432
433 auto spaceCreateInfo = createStructure<XrReferenceSpaceCreateInfo, XR_TYPE_REFERENCE_SPACE_CREATE_INFO>();
434 spaceCreateInfo.referenceSpaceType = type;
435 spaceCreateInfo.poseInReferenceSpace = identityPose;
436
437 XrSpace space;
438 auto result = xrCreateReferenceSpace(m_session, &spaceCreateInfo, &space);
439 RETURN_IF_FAILED(result, "xrCreateReferenceSpace", m_instance, XR_NULL_HANDLE);
440
441 return space;
442}

444void OpenXRDevice::pollEvents()
445{
446 ASSERT(!isMainThread());
447 auto runtimeEvent = createStructure<XrEventDataBuffer, XR_TYPE_EVENT_DATA_BUFFER>();
448 while (xrPollEvent(m_instance, &runtimeEvent) == XR_SUCCESS) {
449 switch (runtimeEvent.type) {
450 case XR_TYPE_EVENT_DATA_SESSION_STATE_CHANGED: {
451 auto* event = reinterpret_cast<XrEventDataSessionStateChanged*>(&runtimeEvent);
452 m_sessionState = event->state;
453 handleSessionStateChange();
454 break;
455 }
456 case XR_TYPE_EVENT_DATA_EVENTS_LOST:
457 case XR_TYPE_EVENT_DATA_INSTANCE_LOSS_PENDING:
458 case XR_TYPE_EVENT_DATA_REFERENCE_SPACE_CHANGE_PENDING: {
459 auto* event = reinterpret_cast<XrEventDataReferenceSpaceChangePending*>(&runtimeEvent);
460 if (event->referenceSpaceType == XR_REFERENCE_SPACE_TYPE_STAGE)
461 updateStageParameters();
462 break;
463 }
464 case XR_TYPE_EVENT_DATA_INTERACTION_PROFILE_CHANGED:
465 updateInteractionProfile();
466 break;
467 case XR_TYPE_EVENT_DATA_MAIN_SESSION_VISIBILITY_CHANGED_EXTX:
468 case XR_TYPE_EVENT_DATA_VISIBILITY_MASK_CHANGED_KHR:
469 case XR_TYPE_EVENT_DATA_PERF_SETTINGS_EXT:
470 break;
471 default:
472 ASSERT_NOT_REACHED("Unhandled event type %d\n", runtimeEvent.type);
473 }
474 }
475}

477XrResult OpenXRDevice::beginSession()
478{
479 ASSERT(!isMainThread());
480 ASSERT(m_sessionState == XR_SESSION_STATE_READY);
481
482 auto sessionBeginInfo = createStructure<XrSessionBeginInfo, XR_TYPE_SESSION_BEGIN_INFO>();
483 sessionBeginInfo.primaryViewConfigurationType = XR_VIEW_CONFIGURATION_TYPE_PRIMARY_STEREO;
484 auto result = xrBeginSession(m_session, &sessionBeginInfo);
485#if !LOG_DISABLED
486 if (XR_FAILED(result))
487 LOG(XR, "%s %s: %s\n", __func__, "xrBeginSession", resultToString(result, m_instance).utf8().data());
488#endif
489 return result;
490}

492void OpenXRDevice::endSession()
493{
494 ASSERT(m_session != XR_NULL_HANDLE);
495 xrEndSession(m_session);
496 resetSession();
497 if (!m_trackingAndRenderingClient)
498 return;
499
500 // Notify did end event
501 callOnMainThread([this, weakThis = makeWeakPtr(*this)]() {
502 if (!weakThis)
503 return;
504 if (m_trackingAndRenderingClient)
505 m_trackingAndRenderingClient->sessionDidEnd();
506 });
507}

509void OpenXRDevice::resetSession()
510{
511 ASSERT(&RunLoop::current() == &m_queue.runLoop());
512 m_layers.clear();
513 m_input.reset();
514 if (m_session != XR_NULL_HANDLE) {
515 xrDestroySession(m_session);
516 m_session = XR_NULL_HANDLE;
517 }
518 m_sessionState = XR_SESSION_STATE_UNKNOWN;
519 m_localSpace = XR_NULL_HANDLE;
520 m_viewSpace = XR_NULL_HANDLE;
521 m_stageSpace = XR_NULL_HANDLE;
522 m_stageParameters = { };
523
524 // deallocate graphic resources
525 m_gl = nullptr;
526 m_egl.reset();
527}
528
529void OpenXRDevice::handleSessionStateChange()
530{
531 ASSERT(&RunLoop::current() == &m_queue.runLoop());
532 if (m_sessionState == XR_SESSION_STATE_STOPPING) {
533 // The application should exit the render loop and call xrEndSession
534 endSession();
535 } else if (m_sessionState == XR_SESSION_STATE_READY) {
536 // The application is ready to call xrBeginSession.
537 beginSession();
538 }
539}

541void OpenXRDevice::waitUntilStopping()
542{
543 ASSERT(&RunLoop::current() == &m_queue.runLoop());
544 pollEvents();
545 if (m_sessionState >= XR_SESSION_STATE_STOPPING)
546 return;
547 m_queue.dispatch([this, protectedThis = makeRef(*this)]() {
548 waitUntilStopping();
549 });
550}

552void OpenXRDevice::updateStageParameters()
553{
554 ASSERT(&RunLoop::current() == &m_queue.runLoop());
555 if (m_stageSpace == XR_NULL_HANDLE)
556 return; // Stage space not requested.
557
558 m_stageParameters.id++;
559
560 XrExtent2Df bounds;
561 if (XR_SUCCEEDED(xrGetReferenceSpaceBoundsRect(m_session, XR_REFERENCE_SPACE_TYPE_STAGE, &bounds))) {
562 // https://immersive-web.github.io/webxr/#xrboundedreferencespace-native-bounds-geometry
563 // Points MUST be given in a clockwise order as viewed from above, looking towards the negative end of the Y axis.
564 m_stageParameters.bounds = Vector<WebCore::FloatPoint> {
565 { bounds.width * 0.5f, -bounds.height * 0.5f },
566 { bounds.width * 0.5f, bounds.height * 0.5f },
567 { -bounds.width * 0.5f, bounds.height * 0.5f },
568 { -bounds.width * 0.5f, -bounds.height * 0.5f }
569 };
570 } else {
571 // https://immersive-web.github.io/webxr/#dom-xrboundedreferencespace-boundsgeometry
572 // If the native bounds geometry is temporarily unavailable, which may occur for several reasons such as during XR device initialization,
573 // extended periods of tracking loss, or movement between pre-configured spaces, the boundsGeometry MUST report an empty array.
574 m_stageParameters.bounds.clear();
575 }
576}

578void OpenXRDevice::updateInteractionProfile()
579{
580 if (!m_input)
581 return;
582
583 m_input->updateInteractionProfile();
584
585 if (didNotifyInputInitialization)
586 return;
587
588 didNotifyInputInitialization = true;
589 auto inputSources = m_input->collectInputSources(m_frameState);
590 callOnMainThread([this, weakThis = makeWeakPtr(*this), inputSources = WTFMove(inputSources)]() mutable {
591 if (!weakThis)
592 return;
593 if (m_trackingAndRenderingClient)
594 m_trackingAndRenderingClient->sessionDidInitializeInputSources(WTFMove(inputSources));
595 });
596}
