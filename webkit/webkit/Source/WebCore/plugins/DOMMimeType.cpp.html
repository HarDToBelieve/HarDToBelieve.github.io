28Ref<DOMMimeType> DOMMimeType::create(Navigator& navigator, const MimeClassInfo& info, DOMPlugin& enabledPlugin)
29{
30 return adoptRef(*new DOMMimeType(navigator, info, enabledPlugin));
31}
32
33DOMMimeType::DOMMimeType(Navigator& navigator, const MimeClassInfo& info, DOMPlugin& enabledPlugin)
34 : m_navigator(makeWeakPtr(navigator))
35 , m_info(info)
36 , m_enabledPlugin(makeWeakPtr(enabledPlugin))
37{
38}
39
40DOMMimeType::~DOMMimeType() = default;
41
42String DOMMimeType::type() const
43{
44 return m_info.type;
45}
46
47String DOMMimeType::suffixes() const
48{
49 StringBuilder builder;
50 for (size_t i = 0; i < m_info.extensions.size(); ++i) {
51 if (i)
52 builder.append(',');
53 builder.append(m_info.extensions[i]);
54 }
55 return builder.toString();
56}
57
58String DOMMimeType::description() const
59{
60 return m_info.desc;
61}
62
63RefPtr<DOMPlugin> DOMMimeType::enabledPlugin() const
64{
65 return m_enabledPlugin.get();
66}
