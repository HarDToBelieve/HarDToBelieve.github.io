31Ref<DOMPlugin> DOMPlugin::create(Navigator& navigator, const PluginInfo& info)
32{
33 return adoptRef(*new DOMPlugin(navigator, info));
34}
35
36static Vector<Ref<DOMMimeType>> makeMimeTypes(Navigator& navigator, const PluginInfo& info, DOMPlugin& self)
37{
38 Vector<Ref<DOMMimeType>> types;
39 types.reserveInitialCapacity(info.mimes.size());
40 for (auto& type : info.mimes)
41 types.uncheckedAppend(DOMMimeType::create(navigator, type, self));
42
43 std::sort(types.begin(), types.end(), [](const Ref<DOMMimeType>& a, const Ref<DOMMimeType>& b) {
44 return codePointCompareLessThan(a->type(), b->type());
45 });
46
47 return types;
48}
49
50DOMPlugin::DOMPlugin(Navigator& navigator, const PluginInfo& info)
51 : m_navigator(makeWeakPtr(navigator))
52 , m_info(info)
53 , m_mimeTypes(makeMimeTypes(navigator, info, *this))
54{
55}

57DOMPlugin::~DOMPlugin() = default;

64String DOMPlugin::filename() const

69String DOMPlugin::description() const

73
74unsigned DOMPlugin::length() const
75{
76 return m_mimeTypes.size();
77}
78
79RefPtr<DOMMimeType> DOMPlugin::item(unsigned index)
80{
81 if (index >= m_mimeTypes.size())
82 return nullptr;
83 return m_mimeTypes[index].ptr();
84}
85
86RefPtr<DOMMimeType> DOMPlugin::namedItem(const AtomString& propertyName)
87{
88 for (auto& type : m_mimeTypes) {
89 if (type->type() == propertyName)
90 return type.ptr();
91 }
92 return nullptr;
93}
94
95Vector<AtomString> DOMPlugin::supportedPropertyNames()
96{
97 Vector<AtomString> result;
98 result.reserveInitialCapacity(m_mimeTypes.size());
99 for (auto& type : m_mimeTypes)
100 result.uncheckedAppend(type->type());
101 return result;
102}
