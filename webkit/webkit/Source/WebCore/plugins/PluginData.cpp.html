41void PluginData::initPlugins()

47const Vector<PluginInfo>& PluginData::webVisiblePlugins() const
48{
49 auto documentURL = m_page.mainFrame().document() ? m_page.mainFrame().document()->url() : URL { };
50 if (!documentURL.isNull() && !protocolHostAndPortAreEqual(m_cachedVisiblePlugins.pageURL, documentURL)) {
51 m_cachedVisiblePlugins.pageURL = WTFMove(documentURL);
52 m_cachedVisiblePlugins.pluginList = std::nullopt;
53 }
54
55 if (!m_cachedVisiblePlugins.pluginList)
56 m_cachedVisiblePlugins.pluginList = m_page.pluginInfoProvider().webVisiblePluginInfo(m_page, m_cachedVisiblePlugins.pageURL);
57
58 return *m_cachedVisiblePlugins.pluginList;
59}

73static bool shouldBePubliclyVisible(const PluginInfo& plugin)
74{
75 // We can greatly reduce fingerprinting opportunities by only advertising plug-ins
76 // that are widely needed for general website compatibility. Since many users
77 // will have these plug-ins, we are not revealing much user-specific information.
78 //
79 // Web compatibility data indicate that Flash, QuickTime, Java, and PDF support
80 // are frequently accessed through the bad practice of iterating over the contents
81 // of the navigator.plugins list. Luckily, these plug-ins happen to be the least
82 // user-specific.
83 return plugin.name.containsIgnoringASCIICase("Shockwave")
84 || plugin.name.containsIgnoringASCIICase("QuickTime")
85 || plugin.name.containsIgnoringASCIICase("Java")
86 || isBuiltInPDFPlugIn(plugin);
87}

89std::pair<Vector<PluginInfo>, Vector<PluginInfo>> PluginData::publiclyVisiblePluginsAndAdditionalWebVisiblePlugins() const
90{
91 auto plugins = webVisiblePlugins();
92
93 if (m_page.showAllPlugins())
94 return { plugins, { } };
95
96 Vector<PluginInfo> additionalWebVisiblePlugins;
97 plugins.removeAllMatching([&](auto& plugin) {
98 if (!shouldBePubliclyVisible(plugin)) {
99 additionalWebVisiblePlugins.append(plugin);
100 return true;
101 }
102 return false;
103 });
104
105 return { plugins, additionalWebVisiblePlugins };
106}

108Vector<MimeClassInfo> PluginData::webVisibleMimeTypes() const
109{
110 Vector<MimeClassInfo> result;
111 for (auto& plugin : webVisiblePlugins())
112 result.appendVector(plugin.mimes);
113 return result;
114}

116static bool supportsMimeTypeForPlugins(const String& mimeType, const PluginData::AllowedPluginTypes allowedPluginTypes, const Vector<PluginInfo>& plugins)
117{
118 for (auto& plugin : plugins) {
119 for (auto& type : plugin.mimes) {
120 if (type.type == mimeType && (allowedPluginTypes == PluginData::AllPlugins || plugin.isApplicationPlugin))
121 return true;
122 }
123 }
124 return false;
125}
126
127bool PluginData::supportsMimeType(const String& mimeType, const AllowedPluginTypes allowedPluginTypes) const
128{
129 return supportsMimeTypeForPlugins(mimeType, allowedPluginTypes, plugins());
130}
131
132bool PluginData::supportsWebVisibleMimeType(const String& mimeType, const AllowedPluginTypes allowedPluginTypes) const
133{
134 return supportsMimeTypeForPlugins(mimeType, allowedPluginTypes, webVisiblePlugins());
135}
136
137bool PluginData::supportsWebVisibleMimeTypeForURL(const String& mimeType, const AllowedPluginTypes allowedPluginTypes, const URL& url) const
138{
139 if (!protocolHostAndPortAreEqual(m_cachedVisiblePlugins.pageURL, url))
140 m_cachedVisiblePlugins = { url, m_page.pluginInfoProvider().webVisiblePluginInfo(m_page, url) };
141 if (!m_cachedVisiblePlugins.pluginList)
142 return false;
143 return supportsMimeTypeForPlugins(mimeType, allowedPluginTypes, *m_cachedVisiblePlugins.pluginList);
144}
145
146String PluginData::pluginFileForWebVisibleMimeType(const String& mimeType) const
147{
148 for (auto& plugin : webVisiblePlugins()) {
149 for (auto& type : plugin.mimes) {
150 if (type.type == mimeType)
151 return plugin.file;
152 }
153 }
154 return { };
155}
