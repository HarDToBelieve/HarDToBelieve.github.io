33class AutoTableLayout final : public TableLayout {
34public:

38 void computeIntrinsicLogicalWidths(LayoutUnit& minWidth, LayoutUnit& maxWidth, TableIntrinsics) override;

42
43private:

54 float minLogicalWidth { 0 };
55 float maxLogicalWidth { 0 };

57 float effectiveMaxLogicalWidth { 0 };
58 float computedLogicalWidth { 0 };
59 bool emptyCellsOnly { true };

62 Vector<Layout> m_layoutStruct;

64 bool m_hasPercent : 1;

66 LayoutUnit m_scaledWidthFromPercentColumns;
67};
