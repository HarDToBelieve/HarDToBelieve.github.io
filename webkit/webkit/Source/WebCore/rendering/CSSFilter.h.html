46class CSSFilter final : public Filter {
47 WTF_MAKE_FAST_ALLOCATED;
48 friend class RenderLayerFilters;
49public:

52 void setSourceImageRect(const FloatRect&);

54
55 ImageBuffer* output() const;
56
57 bool build(RenderElement&, const FilterOperations&, FilterConsumer);
58 void clearIntermediateResults();
59 void apply();
60
61 bool hasFilterThatMovesPixels() const { return m_hasFilterThatMovesPixels; }
62 bool hasFilterThatShouldBeRestrictedBySecurityOrigin() const { return m_hasFilterThatShouldBeRestrictedBySecurityOrigin; }
63
64 void determineFilterPrimitiveSubregion();
65 IntOutsets outsets() const;
66
67private:
68 CSSFilter();

70
71 bool isCSSFilter() const final { return true; }
72
73 FloatRect sourceImageRect() const final { return m_sourceDrawingRegion; }

75 FloatRect filterRegion() const final { return m_filterRegion; }

80 void setMaxEffectRects(const FloatRect&);
81
82 GraphicsContext* inputContext();

84 bool updateBackingStoreRect(const FloatRect& filterRect);

86
87 IntRect outputRect() const;
88
89 LayoutRect computeSourceImageRectForDirtyRect(const LayoutRect& filterBoxRect, const LayoutRect& dirtyRect);
90
91 FloatRect m_sourceDrawingRegion;
92 FloatRect m_filterRegion;
93
94 Vector<Ref<FilterEffect>> m_effects;

98 mutable IntOutsets m_outsets;

102 bool m_hasFilterThatShouldBeRestrictedBySecurityOrigin { false };
103
104 std::unique_ptr<FilterEffectRenderer> m_filterRenderer;
105};
