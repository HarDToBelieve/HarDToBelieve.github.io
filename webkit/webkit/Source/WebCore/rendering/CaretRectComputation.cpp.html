40static LayoutRect computeCaretRectForEmptyElement(const RenderBoxModelObject& renderer, LayoutUnit width, LayoutUnit textIndentOffset, CaretRectMode caretRectMode)
41{
42 ASSERT(!renderer.firstChild());
43
44 // FIXME: This does not take into account either :first-line or :first-letter
45 // However, as soon as some content is entered, the line boxes will be
46 // constructed and this kludge is not called any more. So only the caret size
47 // of an empty :first-line'd block is wrong. I think we can live with that.
48 const RenderStyle& currentStyle = renderer.firstLineStyle();
49
50 enum CaretAlignment { AlignLeft, AlignRight, AlignCenter };
51
52 CaretAlignment alignment = AlignLeft;
53
54 switch (currentStyle.textAlign()) {
55 case TextAlignMode::Left:
56 case TextAlignMode::WebKitLeft:
57 break;
58 case TextAlignMode::Center:
59 case TextAlignMode::WebKitCenter:
60 alignment = AlignCenter;
61 break;
62 case TextAlignMode::Right:
63 case TextAlignMode::WebKitRight:
64 alignment = AlignRight;
65 break;
66 case TextAlignMode::Justify:
67 case TextAlignMode::Start:
68 if (!currentStyle.isLeftToRightDirection())
69 alignment = AlignRight;
70 break;
71 case TextAlignMode::End:
72 if (currentStyle.isLeftToRightDirection())
73 alignment = AlignRight;
74 break;
75 }
76
77 LayoutUnit x = renderer.borderLeft() + renderer.paddingLeft();
78 LayoutUnit maxX = width - renderer.borderRight() - renderer.paddingRight();
79
80 switch (alignment) {
81 case AlignLeft:
82 if (currentStyle.isLeftToRightDirection())
83 x += textIndentOffset;
84 break;
85 case AlignCenter:
86 x = (x + maxX) / 2;
87 if (currentStyle.isLeftToRightDirection())
88 x += textIndentOffset / 2;
89 else
90 x -= textIndentOffset / 2;
91 break;
92 case AlignRight:
93 x = maxX - caretWidth;
94 if (!currentStyle.isLeftToRightDirection())
95 x -= textIndentOffset;
96 break;
97 }
98 x = std::min(x, std::max<LayoutUnit>(maxX - caretWidth, 0));
99
100 auto lineHeight = renderer.lineHeight(true, currentStyle.isHorizontalWritingMode() ? HorizontalLine : VerticalLine, PositionOfInteriorLineBoxes);
101 auto height = std::min(lineHeight, LayoutUnit { currentStyle.fontMetrics().height() });
102 auto y = renderer.paddingTop() + renderer.borderTop() + (lineHeight > height ? (lineHeight - height) / 2 : LayoutUnit { });
103
104 auto rect = LayoutRect(x, y, caretWidth, height);
105
106 if (caretRectMode == CaretRectMode::ExpandToEndOfLine)
107 rect.shiftMaxXEdgeTo(width);
108
109 return currentStyle.isHorizontalWritingMode() ? rect : rect.transposedRect();
110}

112static LayoutRect computeCaretRectForLinePosition(const LayoutIntegration::LineIterator& line, float logicalLeftPosition, CaretRectMode caretRectMode)
113{
114 auto& containingBlock = line->containingBlock();
115 auto lineSelectionRect = line->selectionRect();
116
117 int height = lineSelectionRect.height();
118 int top = lineSelectionRect.y();
119
120 // Distribute the caret's width to either side of the offset.
121 float left = logicalLeftPosition;
122 int caretWidthLeftOfOffset = caretWidth / 2;
123 left -= caretWidthLeftOfOffset;
124 int caretWidthRightOfOffset = caretWidth - caretWidthLeftOfOffset;
125 left = roundf(left);
126
127 float lineLeft = lineSelectionRect.x();
128 float lineRight = lineSelectionRect.maxX();
129
130 bool rightAligned = false;
131 switch (containingBlock.style().textAlign()) {
132 case TextAlignMode::Right:
133 case TextAlignMode::WebKitRight:
134 rightAligned = true;
135 break;
136 case TextAlignMode::Left:
137 case TextAlignMode::WebKitLeft:
138 case TextAlignMode::Center:
139 case TextAlignMode::WebKitCenter:
140 break;
141 case TextAlignMode::Justify:
142 case TextAlignMode::Start:
143 rightAligned = !containingBlock.style().isLeftToRightDirection();
144 break;
145 case TextAlignMode::End:
146 rightAligned = containingBlock.style().isLeftToRightDirection();
147 break;
148 }
149
150 float leftEdge = std::min<float>(0, lineLeft);
151 float rightEdge = std::max<float>(containingBlock.logicalWidth(), lineRight);
152
153 if (rightAligned) {
154 left = std::max(left, leftEdge);
155 left = std::min(left, lineRight - caretWidth);
156 } else {
157 left = std::min(left, rightEdge - caretWidthRightOfOffset);
158 left = std::max(left, lineLeft);
159 }
160
161 auto rect = IntRect(left, top, caretWidth, height);
162
163 if (caretRectMode == CaretRectMode::ExpandToEndOfLine)
164 rect.shiftMaxXEdgeTo(lineRight);
165
166 return containingBlock.style().isHorizontalWritingMode() ? rect : rect.transposedRect();
167}

169static LayoutRect computeCaretRectForText(const InlineRunAndOffset& runAndOffset, CaretRectMode caretRectMode)
170{
171 if (!runAndOffset.run)
172 return { };
173
174 auto& textRun = downcast<LayoutIntegration::TextRunIterator>(runAndOffset.run);
175 auto line = textRun.line();
176
177 float position = textRun->positionForOffset(runAndOffset.offset);
178 return computeCaretRectForLinePosition(line, position, caretRectMode);
179}

181static LayoutRect computeCaretRectForLineBreak(const InlineRunAndOffset& runAndOffset, CaretRectMode caretRectMode)
182{
183 ASSERT(!runAndOffset.offset);
184
185 if (!runAndOffset.run)
186 return { };
187
188 auto line = runAndOffset.run.line();
189 return computeCaretRectForLinePosition(line, line->contentLogicalLeft(), caretRectMode);
190}

192static LayoutRect computeCaretRectForSVGInlineText(const InlineRunAndOffset& runAndOffset, CaretRectMode)
193{
194 auto* box = runAndOffset.run ? runAndOffset.run->legacyInlineBox() : nullptr;
195 auto caretOffset = runAndOffset.offset;
196
197 if (!is<LegacyInlineTextBox>(box))
198 return { };
199
200 auto& textBox = downcast<LegacyInlineTextBox>(*box);
201 if (caretOffset < textBox.start() || caretOffset > textBox.start() + textBox.len())
202 return { };
203
204 // Use the edge of the selection rect to determine the caret rect.
205 if (caretOffset < textBox.start() + textBox.len()) {
206 LayoutRect rect = textBox.localSelectionRect(caretOffset, caretOffset + 1);
207 LayoutUnit x = textBox.isLeftToRightDirection() ? rect.x() : rect.maxX();
208 return LayoutRect(x, rect.y(), caretWidth, rect.height());
209 }
210
211 LayoutRect rect = textBox.localSelectionRect(caretOffset - 1, caretOffset);
212 LayoutUnit x = textBox.isLeftToRightDirection() ? rect.maxX() : rect.x();
213 return { x, rect.y(), caretWidth, rect.height() };
214}

216static LayoutRect computeCaretRectForBox(const RenderBox& renderer, const InlineRunAndOffset& runAndOffset, CaretRectMode caretRectMode)
217{
218 // VisiblePositions at offsets inside containers either a) refer to the positions before/after
219 // those containers (tables and select elements) or b) refer to the position inside an empty block.
220 // They never refer to children.
221 // FIXME: Paint the carets inside empty blocks differently than the carets before/after elements.
222
223 LayoutRect rect(renderer.location(), LayoutSize(caretWidth, renderer.height()));
224 bool ltr = runAndOffset.run ? runAndOffset.run->isLeftToRightDirection() : renderer.style().isLeftToRightDirection();
225
226 if ((!runAndOffset.offset) ^ ltr)
227 rect.move(LayoutSize(renderer.width() - caretWidth, 0_lu));
228
229 if (runAndOffset.run) {
230 auto line = runAndOffset.run.line();
231 LayoutUnit top = line->top();
232 rect.setY(top);
233 rect.setHeight(line->bottom() - top);
234 }
235
236 // If height of box is smaller than font height, use the latter one,
237 // otherwise the caret might become invisible.
238 //
239 // Also, if the box is not a replaced element, always use the font height.
240 // This prevents the "big caret" bug described in:
241 // <rdar://problem/3777804> Deleting all content in a document can result in giant tall-as-window insertion point
242 //
243 // FIXME: ignoring :first-line, missing good reason to take care of
244 LayoutUnit fontHeight = renderer.style().fontMetrics().height();
245 if (fontHeight > rect.height() || (!renderer.isReplaced() && !renderer.isTable()))
246 rect.setHeight(fontHeight);
247
248 // Move to local coords
249 rect.moveBy(-renderer.location());
250
251 // FIXME: Border/padding should be added for all elements but this workaround
252 // is needed because we use offsets inside an "atomic" element to represent
253 // positions before and after the element in deprecated editing offsets.
254 if (renderer.element() && !(editingIgnoresContent(*renderer.element()) || isRenderedTable(renderer.element()))) {
255 rect.setX(rect.x() + renderer.borderLeft() + renderer.paddingLeft());
256 rect.setY(rect.y() + renderer.paddingTop() + renderer.borderTop());
257 }
258
259 if (caretRectMode == CaretRectMode::ExpandToEndOfLine)
260 rect.shiftMaxXEdgeTo(renderer.x() + renderer.width());
261
262 return renderer.isHorizontalWritingMode() ? rect : rect.transposedRect();
263}

265static LayoutRect computeCaretRectForBlock(const RenderBlock& renderer, const InlineRunAndOffset& runAndOffset, CaretRectMode caretRectMode)
266{
267 // Do the normal calculation in most cases.
268 if (renderer.firstChild())
269 return computeCaretRectForBox(renderer, runAndOffset, caretRectMode);
270
271 return computeCaretRectForEmptyElement(renderer, renderer.width(), renderer.textIndentOffset(), caretRectMode);
272}
273
274static LayoutRect computeCaretRectForInline(const RenderInline& renderer)
275{
276 if (renderer.firstChild()) {
277 // This condition is possible if the RenderInline is at an editing boundary,
278 // i.e. the VisiblePosition is:
279 // <RenderInline editingBoundary=true>|<RenderText> </RenderText></RenderInline>
280 // FIXME: need to figure out how to make this return a valid rect, note that
281 // there are no line boxes created in the above case.
282 return { };
283 }
284
285 LayoutRect caretRect = computeCaretRectForEmptyElement(renderer, renderer.horizontalBorderAndPaddingExtent(), 0, CaretRectMode::Normal);
286
287 if (LegacyInlineBox* firstBox = renderer.firstLineBox())
288 caretRect.moveBy(LayoutPoint(firstBox->topLeft()));
289
290 return caretRect;
291}

293LayoutRect computeLocalCaretRect(const RenderObject& renderer, const InlineRunAndOffset& runAndOffset, CaretRectMode caretRectMode)
294{
295 if (is<RenderSVGInlineText>(renderer))
296 return computeCaretRectForSVGInlineText(runAndOffset, caretRectMode);
297
298 if (is<RenderText>(renderer))
299 return computeCaretRectForText(runAndOffset, caretRectMode);
300
301 if (is<RenderLineBreak>(renderer))
302 return computeCaretRectForLineBreak(runAndOffset, caretRectMode);
303
304 if (is<RenderBlock>(renderer))
305 return computeCaretRectForBlock(downcast<RenderBlock>(renderer), runAndOffset, caretRectMode);
306
307 if (is<RenderBox>(renderer))
308 return computeCaretRectForBox(downcast<RenderBox>(renderer), runAndOffset, caretRectMode);
309
310 if (is<RenderInline>(renderer))
311 return computeCaretRectForInline(downcast<RenderInline>(renderer));
312
313 return { };
314}
