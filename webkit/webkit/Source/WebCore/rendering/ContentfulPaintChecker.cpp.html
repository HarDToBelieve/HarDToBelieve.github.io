31bool ContentfulPaintChecker::qualifiesForContentfulPaint(FrameView& frameView)
32{
33 ASSERT(!frameView.needsLayout());
34 ASSERT(frameView.renderView());
35
36 auto oldPaintBehavior = frameView.paintBehavior();
37 auto oldEntireContents = frameView.paintsEntireContents();
38
39 frameView.setPaintBehavior(PaintBehavior::FlattenCompositingLayers);
40 frameView.setPaintsEntireContents(true);
41
42 NullGraphicsContext checkerContext(NullGraphicsContext::PaintInvalidationReasons::DetectingContentfulPaint);
43 frameView.paint(checkerContext, frameView.renderView()->documentRect());
44
45 frameView.setPaintsEntireContents(oldEntireContents);
46 frameView.setPaintBehavior(oldPaintBehavior);
47
48 return checkerContext.contenfulPaintDetected();
49}
