40void EventRegionContext::pushTransform(const AffineTransform& transform)
41{
42 if (m_transformStack.isEmpty())
43 m_transformStack.append(transform);
44 else
45 m_transformStack.append(m_transformStack.last() * transform);
46}
47
48void EventRegionContext::popTransform()
49{
50 if (m_transformStack.isEmpty()) {
51 ASSERT_NOT_REACHED();
52 return;
53 }
54 m_transformStack.removeLast();
55}
56
57void EventRegionContext::pushClip(const IntRect& clipRect)
58{
59 auto transformedClip = m_transformStack.isEmpty() ? clipRect : m_transformStack.last().mapRect(clipRect);
60
61 if (m_clipStack.isEmpty())
62 m_clipStack.append(transformedClip);
63 else
64 m_clipStack.append(intersection(m_clipStack.last(), transformedClip));
65}
66
67void EventRegionContext::popClip()
68{
69 if (m_clipStack.isEmpty()) {
70 ASSERT_NOT_REACHED();
71 return;
72 }
73 m_clipStack.removeLast();
74}

76void EventRegionContext::unite(const Region& region, const RenderStyle& style, bool overrideUserModifyIsEditable)
77{
78 if (m_transformStack.isEmpty() && m_clipStack.isEmpty()) {
79 m_eventRegion.unite(region, style, overrideUserModifyIsEditable);
80 return;
81 }
82
83 auto transformedAndClippedRegion = m_transformStack.isEmpty() ? region : m_transformStack.last().mapRegion(region);
84
85 if (!m_clipStack.isEmpty())
86 transformedAndClippedRegion.intersect(m_clipStack.last());
87
88 m_eventRegion.unite(transformedAndClippedRegion, style, overrideUserModifyIsEditable);
89}
90
91bool EventRegionContext::contains(const IntRect& rect) const
92{
93 if (m_transformStack.isEmpty())
94 return m_eventRegion.contains(rect);
95
96 return m_eventRegion.contains(m_transformStack.last().mapRect(rect));
97}

101bool EventRegion::operator==(const EventRegion& other) const
102{
103#if ENABLE(TOUCH_ACTION_REGIONS)
104 if (m_touchActionRegions != other.m_touchActionRegions)
105 return false;
106#endif
107
108#if ENABLE(WHEEL_EVENT_REGIONS)
109 if (m_wheelEventListenerRegion != other.m_wheelEventListenerRegion)
110 return false;
111 if (m_nonPassiveWheelEventListenerRegion != other.m_nonPassiveWheelEventListenerRegion)
112 return false;
113#endif
114
115#if ENABLE(EDITABLE_REGION)
116 if (m_editableRegion != other.m_editableRegion)
117 return false;
118#endif
119 return m_region == other.m_region;
120}
121
122void EventRegion::unite(const Region& region, const RenderStyle& style, bool overrideUserModifyIsEditable)
123{
124 m_region.unite(region);
125
126#if ENABLE(TOUCH_ACTION_REGIONS)
127 uniteTouchActions(region, style.effectiveTouchActions());
128#endif
129
130#if ENABLE(WHEEL_EVENT_REGIONS)
131 uniteEventListeners(region, style.eventListenerRegionTypes());
132#endif
133
134#if ENABLE(EDITABLE_REGION)
135 if (m_editableRegion && (overrideUserModifyIsEditable || style.userModify() != UserModify::ReadOnly)) {
136 m_editableRegion->unite(region);
137 LOG_WITH_STREAM(EventRegions, stream << " uniting editable region");
138 }
139#else
140 UNUSED_PARAM(overrideUserModifyIsEditable);
141#endif
142
143#if !ENABLE(TOUCH_ACTION_REGIONS) && !ENABLE(WHEEL_EVENT_REGIONS) && !ENABLE(EDITABLE_REGION)
144 UNUSED_PARAM(style);
145#endif
146}

148void EventRegion::translate(const IntSize& offset)
149{
150 m_region.translate(offset);
151
152#if ENABLE(TOUCH_ACTION_REGIONS)
153 for (auto& touchActionRegion : m_touchActionRegions)
154 touchActionRegion.translate(offset);
155#endif
156
157#if ENABLE(WHEEL_EVENT_REGIONS)
158 m_wheelEventListenerRegion.translate(offset);
159 m_nonPassiveWheelEventListenerRegion.translate(offset);
160#endif
161
162#if ENABLE(EDITABLE_REGION)
163 if (m_editableRegion)
164 m_editableRegion->translate(offset);
165#endif
166}

168static inline unsigned toIndex(TouchAction touchAction)
169{
170 switch (touchAction) {
171 case TouchAction::None:
172 return 0;
173 case TouchAction::Manipulation:
174 return 1;
175 case TouchAction::PanX:
176 return 2;
177 case TouchAction::PanY:
178 return 3;
179 case TouchAction::PinchZoom:
180 return 4;
181 case TouchAction::Auto:
182 break;
183 }
184 ASSERT_NOT_REACHED();
185 return 0;
186}
187
188static inline TouchAction toTouchAction(unsigned index)
189{
190 switch (index) {
191 case 0:
192 return TouchAction::None;
193 case 1:
194 return TouchAction::Manipulation;
195 case 2:
196 return TouchAction::PanX;
197 case 3:
198 return TouchAction::PanY;
199 case 4:
200 return TouchAction::PinchZoom;
201 default:
202 break;
203 }
204 ASSERT_NOT_REACHED();
205 return TouchAction::Auto;
206}

208#if ENABLE(TOUCH_ACTION_REGIONS)
209void EventRegion::uniteTouchActions(const Region& touchRegion, OptionSet<TouchAction> touchActions)
210{
211 for (auto touchAction : touchActions) {
212 if (touchAction == TouchAction::Auto)
213 break;
214 auto index = toIndex(touchAction);
215 if (m_touchActionRegions.size() < index + 1)
216 m_touchActionRegions.grow(index + 1);
217 }
218
219 for (unsigned i = 0; i < m_touchActionRegions.size(); ++i) {
220 auto regionTouchAction = toTouchAction(i);
221 if (touchActions.contains(regionTouchAction)) {
222 m_touchActionRegions[i].unite(touchRegion);
223 LOG_WITH_STREAM(EventRegions, stream << " uniting for TouchAction " << regionTouchAction);
224 } else {
225 m_touchActionRegions[i].subtract(touchRegion);
226 LOG_WITH_STREAM(EventRegions, stream << " subtracting for TouchAction " << regionTouchAction);
227 }
228 }
229}
230
231const Region* EventRegion::regionForTouchAction(TouchAction action) const
232{
233 unsigned actionIndex = toIndex(action);
234 if (actionIndex >= m_touchActionRegions.size())
235 return nullptr;
236
237 return &m_touchActionRegions[actionIndex];
238}
239
240OptionSet<TouchAction> EventRegion::touchActionsForPoint(const IntPoint& point) const
241{
242 OptionSet<TouchAction> actions;
243
244 for (unsigned i = 0; i < m_touchActionRegions.size(); ++i) {
245 if (m_touchActionRegions[i].contains(point)) {
246 auto action = toTouchAction(i);
247 actions.add(action);
248 if (action == TouchAction::None || action == TouchAction::Manipulation)
249 break;
250 }
251 }
252
253 if (actions.isEmpty())
254 return { TouchAction::Auto };
255
256 return actions;
257}
258#endif

260#if ENABLE(WHEEL_EVENT_REGIONS)
261void EventRegion::uniteEventListeners(const Region& region, OptionSet<EventListenerRegionType> eventListenerRegionTypes)
262{
263 if (eventListenerRegionTypes.contains(EventListenerRegionType::Wheel)) {
264 m_wheelEventListenerRegion.unite(region);
265 LOG_WITH_STREAM(EventRegions, stream << " uniting for passive wheel event listener");
266 }
267 if (eventListenerRegionTypes.contains(EventListenerRegionType::NonPassiveWheel)) {
268 m_nonPassiveWheelEventListenerRegion.unite(region);
269 LOG_WITH_STREAM(EventRegions, stream << " uniting for active wheel event listener");
270 }
271}

273OptionSet<EventListenerRegionType> EventRegion::eventListenerRegionTypesForPoint(const IntPoint& point) const
274{
275 OptionSet<EventListenerRegionType> regionTypes;
276 if (m_wheelEventListenerRegion.contains(point))
277 regionTypes.add(EventListenerRegionType::Wheel);
278 if (m_nonPassiveWheelEventListenerRegion.contains(point))
279 regionTypes.add(EventListenerRegionType::NonPassiveWheel);
280
281 return regionTypes;
282}

306void EventRegion::dump(TextStream& ts) const
307{
308 ts << m_region;
309
310#if ENABLE(TOUCH_ACTION_REGIONS)
311 if (!m_touchActionRegions.isEmpty()) {
312 TextStream::IndentScope indentScope(ts);
313 ts << indent << "(touch-action\n";
314 for (unsigned i = 0; i < m_touchActionRegions.size(); ++i) {
315 if (m_touchActionRegions[i].isEmpty())
316 continue;
317 TextStream::IndentScope indentScope(ts);
318 ts << indent << "(" << toTouchAction(i);
319 ts << indent << m_touchActionRegions[i];
320 ts << indent << ")\n";
321 }
322 ts << indent << ")\n";
323 }
324#endif
325
326#if ENABLE(WHEEL_EVENT_REGIONS)
327 if (!m_wheelEventListenerRegion.isEmpty()) {
328 ts << indent << "(wheel event listener region" << m_wheelEventListenerRegion;
329 if (!m_nonPassiveWheelEventListenerRegion.isEmpty()) {
330 TextStream::IndentScope indentScope(ts);
331 ts << indent << "(non-passive" << m_nonPassiveWheelEventListenerRegion;
332 ts << indent << ")\n";
333 }
334 ts << indent << ")\n";
335 }
336#endif
337
338#if ENABLE(EDITABLE_REGION)
339 if (m_editableRegion && !m_editableRegion->isEmpty()) {
340 ts << indent << "(editable region" << *m_editableRegion;
341 ts << indent << ")\n";
342 }
343#endif
344}

346TextStream& operator<<(TextStream& ts, TouchAction touchAction)
347{
348 switch (touchAction) {
349 case TouchAction::None:
350 return ts << "none";
351 case TouchAction::Manipulation:
352 return ts << "manipulation";
353 case TouchAction::PanX:
354 return ts << "pan-x";
355 case TouchAction::PanY:
356 return ts << "pan-y";
357 case TouchAction::PinchZoom:
358 return ts << "pinch-zoom";
359 case TouchAction::Auto:
360 return ts << "auto";
361 }
362 ASSERT_NOT_REACHED();
363 return ts;
364}

366TextStream& operator<<(TextStream& ts, const EventRegion& eventRegion)
