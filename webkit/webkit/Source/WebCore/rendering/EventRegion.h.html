59class EventRegion {
60public:
61 WEBCORE_EXPORT EventRegion();
62
63 EventRegionContext makeContext() { return EventRegionContext(*this); }
64
65 bool isEmpty() const { return m_region.isEmpty(); }
66
67 WEBCORE_EXPORT bool operator==(const EventRegion&) const;
68
69 void unite(const Region&, const RenderStyle&, bool overrideUserModifyIsEditable = false);
70 void translate(const IntSize&);
71
72 bool contains(const IntPoint& point) const { return m_region.contains(point); }
73 bool contains(const IntRect& rect) const { return m_region.contains(rect); }
74 bool intersects(const IntRect& rect) const { return m_region.intersects(rect); }
75
76 const Region& region() const { return m_region; }
77
78#if ENABLE(TOUCH_ACTION_REGIONS)
79 bool hasTouchActions() const { return !m_touchActionRegions.isEmpty(); }

84#if ENABLE(WHEEL_EVENT_REGIONS)

124template<class Encoder>
125void EventRegion::encode(Encoder& encoder) const
126{
127 encoder << m_region;
128#if ENABLE(WHEEL_EVENT_REGIONS)
129 encoder << m_wheelEventListenerRegion;
130 encoder << m_nonPassiveWheelEventListenerRegion;
131#endif
132#if ENABLE(TOUCH_ACTION_REGIONS)
133 encoder << m_touchActionRegions;
134#endif
135#if ENABLE(EDITABLE_REGION)
136 encoder << m_editableRegion;
137#endif
138}

140template<class Decoder>
141std::optional<EventRegion> EventRegion::decode(Decoder& decoder)
142{
143 std::optional<Region> region;
144 decoder >> region;
145 if (!region)
146 return std::nullopt;
147
148 EventRegion eventRegion;
149 eventRegion.m_region = WTFMove(*region);
150
151#if ENABLE(WHEEL_EVENT_REGIONS)
152 std::optional<Region> wheelEventListenerRegion;
153 decoder >> wheelEventListenerRegion;
154 if (!wheelEventListenerRegion)
155 return std::nullopt;
156
157 eventRegion.m_wheelEventListenerRegion = WTFMove(*wheelEventListenerRegion);
158
159 std::optional<Region> nonPassiveWheelEventListenerRegion;
160 decoder >> nonPassiveWheelEventListenerRegion;
161 if (!nonPassiveWheelEventListenerRegion)
162 return std::nullopt;
163
164 eventRegion.m_nonPassiveWheelEventListenerRegion = WTFMove(*nonPassiveWheelEventListenerRegion);
165#endif
166
167#if ENABLE(TOUCH_ACTION_REGIONS)
168 std::optional<Vector<Region>> touchActionRegions;
169 decoder >> touchActionRegions;
170 if (!touchActionRegions)
171 return std::nullopt;
172
173 eventRegion.m_touchActionRegions = WTFMove(*touchActionRegions);
174#endif
175
176#if ENABLE(EDITABLE_REGION)
177 std::optional<std::optional<Region>> editableRegion;
178 decoder >> editableRegion;
179 if (!editableRegion)
180 return std::nullopt;
181 eventRegion.m_editableRegion = WTFMove(*editableRegion);
182#endif
183
184 return eventRegion;
185}
186
187template<class Decoder>
188bool EventRegion::decode(Decoder& decoder, EventRegion& eventRegion)
189{
190 std::optional<EventRegion> decodedEventRegion;
191 decoder >> decodedEventRegion;
192 if (!decodedEventRegion)
193 return false;
194 eventRegion = WTFMove(*decodedEventRegion);
195 return true;
196}
