38namespace WebCore {
39
40class RenderBox;
41
42class FlexItem {
43public:
44 FlexItem(RenderBox&, LayoutUnit flexBaseContentSize, LayoutUnit mainAxisBorderAndPadding, LayoutUnit mainAxisMargin, std::pair<LayoutUnit, LayoutUnit> minMaxSizes, bool everHadLayout);
45
46 LayoutUnit hypotheticalMainAxisMarginBoxSize() const
47 {
48 return hypotheticalMainContentSize + mainAxisBorderAndPadding + mainAxisMargin;
49 }
50
51 LayoutUnit flexBaseMarginBoxSize() const
52 {
53 return flexBaseContentSize + mainAxisBorderAndPadding + mainAxisMargin;
54 }
55
56 LayoutUnit flexedMarginBoxSize() const
57 {
58 return flexedContentSize + mainAxisBorderAndPadding + mainAxisMargin;
59 }
60
61 LayoutUnit constrainSizeByMinMax(const LayoutUnit) const;
62
63 RenderBox& box;
64 const LayoutUnit flexBaseContentSize;
65 const LayoutUnit mainAxisBorderAndPadding;
66 const LayoutUnit mainAxisMargin;
67 const std::pair<LayoutUnit, LayoutUnit> minMaxSizes;
68 const LayoutUnit hypotheticalMainContentSize;
69 LayoutUnit flexedContentSize;
70 bool frozen { false };
71 bool everHadLayout { false };
72};
73
74class FlexLayoutAlgorithm {
75 WTF_MAKE_NONCOPYABLE(FlexLayoutAlgorithm);
76
77public:
78 FlexLayoutAlgorithm(const RenderStyle&, LayoutUnit lineBreakLength, const Vector<FlexItem>& allItems, LayoutUnit gapBetweenItems, LayoutUnit gapBetweenLines);
79
80 // The hypothetical main size of an item is the flex base size clamped
81 // according to its min and max main size properties
82 bool computeNextFlexLine(size_t& nextIndex, Vector<FlexItem>& lineItems, LayoutUnit& sumFlexBaseSize, double& totalFlexGrow, double& totalFlexShrink, double& totalWeightedFlexShrink, LayoutUnit& sumHypotheticalMainSize);
83
84private:
85 bool isMultiline() const { return m_style.flexWrap() != FlexWrap::NoWrap; }
86
87 const RenderStyle& m_style;
88 LayoutUnit m_lineBreakLength;
89 const Vector<FlexItem>& m_allItems;
90
91 const LayoutUnit m_gapBetweenItems;
92 const LayoutUnit m_gapBetweenLines;
93};
