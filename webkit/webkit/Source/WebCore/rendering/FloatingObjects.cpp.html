129inline static bool rangesIntersect(LayoutUnit floatTop, LayoutUnit floatBottom, LayoutUnit objectTop, LayoutUnit objectBottom)
130{
131 if (objectTop >= floatBottom || objectBottom < floatTop)
132 return false;
133
134 // The top of the object overlaps the float
135 if (objectTop >= floatTop)
136 return true;
137
138 // The object encloses the float
139 if (objectTop < floatTop && objectBottom > floatBottom)
140 return true;
141
142 // The bottom of the object overlaps the float
143 if (objectBottom > objectTop && objectBottom > floatTop && objectBottom <= floatBottom)
144 return true;
145
146 return false;
147}

211class FindNextFloatLogicalBottomAdapter {
212public:
213 typedef FloatingObjectInterval IntervalType;
214
215 FindNextFloatLogicalBottomAdapter(const RenderBlockFlow& renderer, LayoutUnit belowLogicalHeight)
216 : m_renderer(makeWeakPtr(renderer))
217 , m_belowLogicalHeight(belowLogicalHeight)
218 {
219 }
220
221 LayoutUnit lowValue() const { return m_belowLogicalHeight; }
222 LayoutUnit highValue() const { return LayoutUnit::max(); }
223 void collectIfNeeded(const IntervalType&);
224
225 LayoutUnit nextLogicalBottom() const { return m_nextLogicalBottom.value_or(0); }
226 LayoutUnit nextShapeLogicalBottom() const { return m_nextShapeLogicalBottom.value_or(nextLogicalBottom()); }
227
228private:
229 WeakPtr<const RenderBlockFlow> m_renderer;
230 LayoutUnit m_belowLogicalHeight;
231 std::optional<LayoutUnit> m_nextLogicalBottom;
232 std::optional<LayoutUnit> m_nextShapeLogicalBottom;
233};
234
235inline void FindNextFloatLogicalBottomAdapter::collectIfNeeded(const IntervalType& interval)
236{
237 const auto& floatingObject = *interval.data();
238 if (!rangesIntersect(interval.low(), interval.high(), m_belowLogicalHeight, LayoutUnit::max()))
239 return;
240
241 // All the objects returned from the tree should be already placed.
242 ASSERT(floatingObject.isPlaced());
243 ASSERT(rangesIntersect(m_renderer->logicalTopForFloat(floatingObject), m_renderer->logicalBottomForFloat(floatingObject), m_belowLogicalHeight, LayoutUnit::max()));
244
245 LayoutUnit floatBottom = m_renderer->logicalBottomForFloat(floatingObject);
246 if (m_nextLogicalBottom && m_nextLogicalBottom.value() < floatBottom)
247 return;
248
249 if (ShapeOutsideInfo* shapeOutside = floatingObject.renderer().shapeOutsideInfo()) {
250 LayoutUnit shapeBottom = m_renderer->logicalTopForFloat(floatingObject) + m_renderer->marginBeforeForChild(floatingObject.renderer()) + shapeOutside->shapeLogicalBottom();
251 // Use the shapeBottom unless it extends outside of the margin box, in which case it is clipped.
252 m_nextShapeLogicalBottom = std::min(shapeBottom, floatBottom);
253 } else
254 m_nextShapeLogicalBottom = floatBottom;
255 m_nextLogicalBottom = floatBottom;
256}

301void FloatingObjects::moveAllToFloatInfoMap(RendererToFloatInfoMap& map)
302{
303 for (auto it = m_set.begin(), end = m_set.end(); it != end; ++it) {
304 auto& renderer = it->get()->renderer();
305 // FIXME: The only reason it is safe to move these out of the set is that
306 // we are about to clear it. Otherwise it would break the hash table invariant.
307 // A clean way to do this would be to add a takeAll function to HashSet.
308 map.add(&renderer, WTFMove(*it));
309 }
310 clear();
311}

329FloatingObjectInterval FloatingObjects::intervalForFloatingObject(FloatingObject* floatingObject)
330{
331 // FIXME The endpoints of the floating object interval shouldn't need to be
332 // floored. See http://wkb.ug/125831 for more details.
333 if (m_horizontalWritingMode)
334 return FloatingObjectInterval(floatingObject->frameRect().y().floor(), floatingObject->frameRect().maxY().floor(), floatingObject);
335 return FloatingObjectInterval(floatingObject->frameRect().x().floor(), floatingObject->frameRect().maxX().floor(), floatingObject);
336}

338void FloatingObjects::addPlacedObject(FloatingObject* floatingObject)
339{
340 ASSERT(!floatingObject->isInPlacedTree());
341
342 floatingObject->setIsPlaced(true);
343 if (m_placedFloatsTree)
344 m_placedFloatsTree->add(intervalForFloatingObject(floatingObject));
345
346#if ASSERT_ENABLED
347 floatingObject->setIsInPlacedTree(true);
348#endif
349}
350
351void FloatingObjects::removePlacedObject(FloatingObject* floatingObject)
352{
353 ASSERT(floatingObject->isPlaced() && floatingObject->isInPlacedTree());
354
355 if (m_placedFloatsTree) {
356 bool removed = m_placedFloatsTree->remove(intervalForFloatingObject(floatingObject));
357 ASSERT_UNUSED(removed, removed);
358 }
359
360 floatingObject->setIsPlaced(false);
361#if ASSERT_ENABLED
362 floatingObject->setIsInPlacedTree(false);
363#endif
364}

366FloatingObject* FloatingObjects::add(std::unique_ptr<FloatingObject> floatingObject)
367{
368 increaseObjectsCount(floatingObject->type());
369 if (floatingObject->isPlaced())
370 addPlacedObject(floatingObject.get());
371 return m_set.add(WTFMove(floatingObject)).iterator->get();
372}
373
374void FloatingObjects::remove(FloatingObject* floatingObject)
375{
376 ASSERT((m_set.contains(floatingObject)));
377 decreaseObjectsCount(floatingObject->type());
378 ASSERT(floatingObject->isPlaced() || !floatingObject->isInPlacedTree());
379 if (floatingObject->isPlaced())
380 removePlacedObject(floatingObject);
381 ASSERT(!floatingObject->originatingLine());
382 m_set.remove(floatingObject);
383}
384
385void FloatingObjects::computePlacedFloatsTree()
386{
387 ASSERT(!m_placedFloatsTree);
388 if (m_set.isEmpty())
389 return;
390
391 m_placedFloatsTree = makeUnique<FloatingObjectTree>();
392 for (auto it = m_set.begin(), end = m_set.end(); it != end; ++it) {
393 FloatingObject* floatingObject = it->get();
394 if (floatingObject->isPlaced())
395 m_placedFloatsTree->add(intervalForFloatingObject(floatingObject));
396 }
397}

399inline const FloatingObjectTree* FloatingObjects::placedFloatsTree()

401 if (!m_placedFloatsTree)
402 computePlacedFloatsTree();
403 return m_placedFloatsTree.get();

476template <FloatingObject::Type FloatTypeValue>
477inline void ComputeFloatOffsetAdapter<FloatTypeValue>::collectIfNeeded(const IntervalType& interval)
478{
479 const auto& floatingObject = *interval.data();
480 if (floatingObject.type() != FloatTypeValue || !rangesIntersect(interval.low(), interval.high(), m_lineTop, m_lineBottom))
481 return;
482
483 // All the objects returned from the tree should be already placed.
484 ASSERT(floatingObject.isPlaced());
485 ASSERT(rangesIntersect(m_renderer->logicalTopForFloat(floatingObject), m_renderer->logicalBottomForFloat(floatingObject), m_lineTop, m_lineBottom));
486
487 bool floatIsNewExtreme = updateOffsetIfNeeded(floatingObject);
488 if (floatIsNewExtreme)
489 m_outermostFloat = &floatingObject;
490}

492template<>
493inline bool ComputeFloatOffsetForLineLayoutAdapter<FloatingObject::FloatLeft>::updateOffsetIfNeeded(const FloatingObject& floatingObject)
494{
495 LayoutUnit logicalRight = m_renderer->logicalRightForFloat(floatingObject);
496 if (ShapeOutsideInfo* shapeOutside = floatingObject.renderer().shapeOutsideInfo()) {
497 ShapeOutsideDeltas shapeDeltas = shapeOutside->computeDeltasForContainingBlockLine(*m_renderer, floatingObject, m_lineTop, m_lineBottom - m_lineTop);
498 if (!shapeDeltas.isValid() || !shapeDeltas.lineOverlapsShape())
499 return false;
500
501 logicalRight += shapeDeltas.rightMarginBoxDelta();
502 }
503 if (logicalRight > m_offset) {
504 m_offset = logicalRight;
505 return true;
506 }
507
508 return false;
509}

511template<>
512inline bool ComputeFloatOffsetForLineLayoutAdapter<FloatingObject::FloatRight>::updateOffsetIfNeeded(const FloatingObject& floatingObject)
513{
514 LayoutUnit logicalLeft = m_renderer->logicalLeftForFloat(floatingObject);
515 if (ShapeOutsideInfo* shapeOutside = floatingObject.renderer().shapeOutsideInfo()) {
516 ShapeOutsideDeltas shapeDeltas = shapeOutside->computeDeltasForContainingBlockLine(*m_renderer, floatingObject, m_lineTop, m_lineBottom - m_lineTop);
517 if (!shapeDeltas.isValid() || !shapeDeltas.lineOverlapsShape())
518 return false;
519
520 logicalLeft += shapeDeltas.leftMarginBoxDelta();
521 }
522 if (logicalLeft < m_offset) {
523 m_offset = logicalLeft;
524 return true;
525 }
526
527 return false;
528}
