38class FloatingObject {
39 WTF_MAKE_NONCOPYABLE(FloatingObject); WTF_MAKE_FAST_ALLOCATED;
40public:
41 // Note that Type uses bits so you can use FloatLeftRight as a mask to query for both left and right.
42 enum Type { FloatLeft = 1, FloatRight = 2, FloatLeftRight = 3 };
43
44 static std::unique_ptr<FloatingObject> create(RenderBox&);
45 std::unique_ptr<FloatingObject> copyToNewContainer(LayoutSize, bool shouldPaint = false, bool isDescendant = false) const;
46 std::unique_ptr<FloatingObject> cloneForNewParent() const;
47
48 explicit FloatingObject(RenderBox&);
49 FloatingObject(RenderBox&, Type, const LayoutRect&, const LayoutSize&, bool shouldPaint, bool isDescendant);
50
51 Type type() const { return static_cast<Type>(m_type); }
52 RenderBox& renderer() const { ASSERT(m_renderer); return *m_renderer; }
53
54 bool isPlaced() const { return m_isPlaced; }
55 void setIsPlaced(bool placed = true) { m_isPlaced = placed; }
56
57 LayoutUnit x() const { ASSERT(isPlaced()); return m_frameRect.x(); }
58 LayoutUnit maxX() const { ASSERT(isPlaced()); return m_frameRect.maxX(); }
59 LayoutUnit y() const { ASSERT(isPlaced()); return m_frameRect.y(); }
60 LayoutUnit maxY() const { ASSERT(isPlaced()); return m_frameRect.maxY(); }
61 LayoutUnit width() const { return m_frameRect.width(); }
62 LayoutUnit height() const { return m_frameRect.height(); }
63
64 void setX(LayoutUnit x) { ASSERT(!isInPlacedTree()); m_frameRect.setX(x); }
65 void setY(LayoutUnit y) { ASSERT(!isInPlacedTree()); m_frameRect.setY(y); }
66 void setWidth(LayoutUnit width) { ASSERT(!isInPlacedTree()); m_frameRect.setWidth(width); }
67 void setHeight(LayoutUnit height) { ASSERT(!isInPlacedTree()); m_frameRect.setHeight(height); }
68
69 void setMarginOffset(LayoutSize offset) { ASSERT(!isInPlacedTree()); m_marginOffset = offset; }
70
71 const LayoutRect& frameRect() const { ASSERT(isPlaced()); return m_frameRect; }
72 void setFrameRect(const LayoutRect& frameRect) { ASSERT(!isInPlacedTree()); m_frameRect = frameRect; }
73
74 LayoutUnit paginationStrut() const { return m_paginationStrut; }
75 void setPaginationStrut(LayoutUnit strut) { m_paginationStrut = strut; }
76
77#if ASSERT_ENABLED
78 bool isInPlacedTree() const { return m_isInPlacedTree; }
79 void setIsInPlacedTree(bool value) { m_isInPlacedTree = value; }
80#endif
81
82 bool shouldPaint() const;
83
84 bool paintsFloat() const { return m_paintsFloat; }
85 void setPaintsFloat(bool paintsFloat) { m_paintsFloat = paintsFloat; }
86
87 bool isDescendant() const { return m_isDescendant; }
88 void setIsDescendant(bool isDescendant) { m_isDescendant = isDescendant; }
89
90 // FIXME: Callers of these methods are dangerous and should be allowed explicitly or removed.
91 LegacyRootInlineBox* originatingLine() const { return m_originatingLine.get(); }
92 void clearOriginatingLine() { m_originatingLine = nullptr; }
93 void setOriginatingLine(LegacyRootInlineBox& line) { m_originatingLine = makeWeakPtr(line); }
94
95 LayoutSize locationOffsetOfBorderBox() const
96 {
97 ASSERT(isPlaced());
98 return LayoutSize(m_frameRect.location().x() + m_marginOffset.width(), m_frameRect.location().y() + m_marginOffset.height());
99 }
100 LayoutSize marginOffset() const { ASSERT(isPlaced()); return m_marginOffset; }
101 LayoutSize translationOffsetToAncestor() const;
102
103private:
104 WeakPtr<RenderBox> m_renderer;
105 WeakPtr<LegacyRootInlineBox> m_originatingLine;
106 LayoutRect m_frameRect;
107 LayoutUnit m_paginationStrut;
108 LayoutSize m_marginOffset;
109 unsigned m_type : 2; // Type (left or right aligned)
110 unsigned m_paintsFloat : 1;
111 unsigned m_isDescendant : 1;
112 unsigned m_isPlaced : 1;
113#if ASSERT_ENABLED
114 unsigned m_isInPlacedTree : 1;
115#endif
116};

118// FIXME: This could be simplified if we made it inherit from PtrHash<std::unique_ptr<FloatingObject>> and
119// changed PtrHashBase to have all of its hash and equal functions bottleneck through single functions (as
120// is done here). That would allow us to only override those master hash and equal functions.
121struct FloatingObjectHashFunctions {
122 typedef std::unique_ptr<FloatingObject> T;
123 typedef typename WTF::GetPtrHelper<T>::PtrType PtrType;
124
125 static unsigned hash(PtrType key) { return PtrHash<RenderBox*>::hash(&key->renderer()); }
126 static bool equal(PtrType a, PtrType b) { return &a->renderer() == &b->renderer(); }
127 static const bool safeToCompareToEmptyOrDeleted = true;
128
129 static unsigned hash(const T& key) { return hash(WTF::getPtr(key)); }
130 static bool equal(const T& a, const T& b) { return equal(WTF::getPtr(a), WTF::getPtr(b)); }
131 static bool equal(PtrType a, const T& b) { return equal(a, WTF::getPtr(b)); }
132 static bool equal(const T& a, PtrType b) { return equal(WTF::getPtr(a), b); }
133};
134struct FloatingObjectHashTranslator {
135 static unsigned hash(const RenderBox& key) { return PtrHash<const RenderBox*>::hash(&key); }
136 static bool equal(const std::unique_ptr<FloatingObject>& a, const RenderBox& b) { return &a->renderer() == &b; }
137};
