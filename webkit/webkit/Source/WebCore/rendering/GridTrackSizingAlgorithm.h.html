61 const LayoutUnit& growthLimitIfNotInfinite() const;

63 const LayoutUnit& plannedSize() const { return m_plannedSize; }

75
76 const GridTrackSize& cachedTrackSize() const { return *m_cachedTrackSize; }

78
79private:

105 void run();
106 void reset();
107
108 // Required by RenderGrid. Try to minimize the exposed surface.
109 const Grid& grid() const { return m_grid; }

111 // and encapsulate any access in the algorithm class.

115 LayoutUnit maxContentSize() const { return m_maxContentSize; };
116
117 LayoutUnit baselineOffsetForChild(const RenderBox&, GridAxis) const;

120 void cacheBaselineAlignedItem(const RenderBox&, GridAxis);
121 void copyBaselineItemsCache(const GridTrackSizingAlgorithm&, GridAxis);
122 void clearBaselineItemsCache();
123
124 Vector<GridTrack>& tracks(GridTrackSizingDirection direction) { return direction == ForColumns ? m_columns : m_rows; }

133 LayoutUnit computeTrackBasedSize() const;

213 // computation. We're normally just interested in intrinsic inline sizes

248class GridTrackSizingAlgorithmStrategy {
249 WTF_MAKE_FAST_ALLOCATED;
250public:
251 virtual LayoutUnit minContentForChild(RenderBox&) const;
252 LayoutUnit maxContentForChild(RenderBox&) const;
253 LayoutUnit minSizeForChild(RenderBox&) const;
254
255 virtual ~GridTrackSizingAlgorithmStrategy() = default;
256
257 virtual void maximizeTracks(Vector<GridTrack>&, std::optional<LayoutUnit>& freeSpace) = 0;
258 virtual double findUsedFlexFraction(Vector<unsigned>& flexibleSizedTracksIndex, GridTrackSizingDirection, std::optional<LayoutUnit> initialFreeSpace) const = 0;
259 virtual bool recomputeUsedFlexFractionIfNeeded(double& flexFraction, LayoutUnit& totalGrowth) const = 0;
260 virtual LayoutUnit freeSpaceForStretchAutoTracksStep() const = 0;
261 virtual bool isComputingSizeContainment() const = 0;
262
263protected:
264 GridTrackSizingAlgorithmStrategy(GridTrackSizingAlgorithm& algorithm)
265 : m_algorithm(algorithm) { }
266
267 virtual LayoutUnit minLogicalSizeForChild(RenderBox&, const Length& childMinSize, std::optional<LayoutUnit> availableSize) const;
268 virtual void layoutGridItemForMinSizeComputation(RenderBox&, bool overrideSizeHasChanged) const = 0;
269
270 LayoutUnit logicalHeightForChild(RenderBox&) const;
271 bool updateOverridingContainingBlockContentSizeForChild(RenderBox&, GridTrackSizingDirection, std::optional<LayoutUnit> = std::nullopt) const;
272
273 // GridTrackSizingAlgorithm accessors for subclasses.
274 LayoutUnit computeTrackBasedSize() const { return m_algorithm.computeTrackBasedSize(); }
275 GridTrackSizingDirection direction() const { return m_algorithm.m_direction; }
276 double findFrUnitSize(const GridSpan& tracksSpan, LayoutUnit leftOverSpace) const { return m_algorithm.findFrUnitSize(tracksSpan, leftOverSpace); }
277 void distributeSpaceToTracks(Vector<GridTrack*>& tracks, LayoutUnit& availableLogicalSpace) const { m_algorithm.distributeSpaceToTracks<MaximizeTracks>(tracks, nullptr, availableLogicalSpace); }
278 const RenderGrid* renderGrid() const { return m_algorithm.m_renderGrid; }
279 std::optional<LayoutUnit> availableSpace() const { return m_algorithm.availableSpace(); }

281 GridTrackSizingAlgorithm& m_algorithm;
