52RenderRangeIterator::RenderRangeIterator(RenderObject* start)
53 : m_current(start)
54{
55 checkForSpanner();
56}
57
58RenderObject* RenderRangeIterator::current() const
59{
60 return m_current;
61}
62
63RenderObject* RenderRangeIterator::next()
64{
65 RenderObject* currentSpan = m_spannerStack.isEmpty() ? nullptr : m_spannerStack.last()->spanner();
66 m_current = m_current->nextInPreOrder(currentSpan);
67 checkForSpanner();
68 if (!m_current && currentSpan) {
69 RenderObject* placeholder = m_spannerStack.last();
70 m_spannerStack.removeLast();
71 m_current = placeholder->nextInPreOrder();
72 checkForSpanner();
73 }
74 return m_current;
75}
76
77void RenderRangeIterator::checkForSpanner()
78{
79 if (!is<RenderMultiColumnSpannerPlaceholder>(m_current))
80 return;
81 auto& placeholder = downcast<RenderMultiColumnSpannerPlaceholder>(*m_current);
82 m_spannerStack.append(&placeholder);
83 m_current = placeholder.spanner();
84}

86static RenderObject* rendererAfterOffset(const RenderObject& renderer, unsigned offset)
87{
88 auto* child = renderer.childAt(offset);
89 return child ? child : renderer.nextInPreOrderAfterChildren();
90}

98bool HighlightData::setRenderRange(const HighlightRangeData& rangeData)
99{
100 if (!rangeData.startPosition || !rangeData.endPosition)
101 return false;
102
103 auto startPosition = rangeData.startPosition.value();
104 auto endPosition = rangeData.endPosition.value();
105
106 if (!startPosition.containerNode() || !endPosition.containerNode())
107 return false;
108
109 auto* startRenderer = startPosition.containerNode()->renderer();
110 auto* endRenderer = endPosition.containerNode()->renderer();
111
112 if (!startRenderer || !endRenderer)
113 return false;
114
115 unsigned startOffset = startPosition.computeOffsetInContainerNode();
116 unsigned endOffset = endPosition.computeOffsetInContainerNode();
117
118 setRenderRange({ startRenderer, endRenderer, startOffset, endOffset });
119 return true;
120}

122RenderObject::HighlightState HighlightData::highlightStateForRenderer(const RenderObject& renderer)
123{
124 if (m_isSelection)
125 return renderer.selectionState();
126
127 if (&renderer == m_renderRange.start()) {
128 if (m_renderRange.start() && m_renderRange.end() && m_renderRange.start() == m_renderRange.end())
129 return RenderObject::HighlightState::Both;
130 if (m_renderRange.start())
131 return RenderObject::HighlightState::Start;
132 }
133 if (&renderer == m_renderRange.end())
134 return RenderObject::HighlightState::End;
135
136 auto* highlightEnd = rendererAfterOffset(*m_renderRange.end(), m_renderRange.endOffset());
137
138 RenderRangeIterator highlightIterator(m_renderRange.start());
139 for (auto* currentRenderer = m_renderRange.start(); currentRenderer && currentRenderer != highlightEnd; currentRenderer = highlightIterator.next()) {
140 if (currentRenderer == m_renderRange.start())
141 continue;
142 if (!currentRenderer->canBeSelectionLeaf())
143 continue;
144 if (&renderer == currentRenderer)
145 return RenderObject::HighlightState::Inside;
146 }
147 return RenderObject::HighlightState::None;
148}

150RenderObject::HighlightState HighlightData::highlightStateForTextBox(const RenderText& renderer, const TextBoxSelectableRange& textBoxRange)
151{
152 auto state = highlightStateForRenderer(renderer);
153
154 if (state == RenderObject::HighlightState::None || state == RenderObject::HighlightState::Inside)
155 return state;
156
157 auto startOffset = this->startOffset();
158 auto endOffset = this->endOffset();
159
160 // The position after a hard line break is considered to be past its end.
161 ASSERT(textBoxRange.start + textBoxRange.length >= (textBoxRange.isLineBreak ? 1 : 0));
162 unsigned lastSelectable = textBoxRange.start + textBoxRange.length - (textBoxRange.isLineBreak ? 1 : 0);
163
164 bool containsStart = state != RenderObject::HighlightState::End && startOffset >= textBoxRange.start && startOffset < textBoxRange.start + textBoxRange.length;
165 bool containsEnd = state != RenderObject::HighlightState::Start && endOffset > textBoxRange.start && endOffset <= lastSelectable;
166 if (containsStart && containsEnd)
167 return RenderObject::HighlightState::Both;
168 if (containsStart)
169 return RenderObject::HighlightState::Start;
170 if (containsEnd)
171 return RenderObject::HighlightState::End;
172 if ((state == RenderObject::HighlightState::End || startOffset < textBoxRange.start) && (state == RenderObject::HighlightState::Start || endOffset > lastSelectable))
173 return RenderObject::HighlightState::Inside;
174
175 return RenderObject::HighlightState::None;
176}
177
178std::pair<unsigned, unsigned> HighlightData::rangeForTextBox(const RenderText& renderer, const TextBoxSelectableRange& textBoxRange)
179{
180 auto state = highlightStateForTextBox(renderer, textBoxRange);
181
182 switch (state) {
183 case RenderObject::HighlightState::Inside:
184 return textBoxRange.clamp(0, std::numeric_limits<unsigned>::max());
185 case RenderObject::HighlightState::Start:
186 return textBoxRange.clamp(startOffset(), std::numeric_limits<unsigned>::max());
187 case RenderObject::HighlightState::End:
188 return textBoxRange.clamp(0, endOffset());
189 case RenderObject::HighlightState::Both:
190 return textBoxRange.clamp(startOffset(), endOffset());
191 case RenderObject::HighlightState::None:
192 return { 0, 0 };
193 };
194
195 RELEASE_ASSERT_NOT_REACHED();
196}
