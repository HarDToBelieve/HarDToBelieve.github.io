27HitTestLocation::HitTestLocation() = default;

29HitTestLocation::HitTestLocation(const LayoutPoint& point)
30 : m_point(point)
31 , m_boundingBox(LayoutRect { flooredIntPoint(point), LayoutSize { 1, 1 } })
32 , m_transformedPoint(point)
33 , m_transformedRect(m_boundingBox)
34{
35}

37HitTestLocation::HitTestLocation(const FloatPoint& point, const FloatQuad& quad)
38 : m_point { flooredLayoutPoint(point) }
39 , m_boundingBox { quad.enclosingBoundingBox() }
40 , m_transformedPoint { point }
41 , m_transformedRect { quad }
42 , m_isRectBased { true }
43 , m_isRectilinear { quad.isRectilinear() }
44{
45}

47HitTestLocation::HitTestLocation(const LayoutRect& rect)
48 : m_point { rect.center() } // Rounded to an integer point; it may not be the exact center.
49 , m_boundingBox { rect }
50 , m_transformedPoint { rect.center() }

91void HitTestLocation::move(const LayoutSize& offset)
92{
93 m_point.move(offset);
94 m_transformedPoint.move(offset);
95 m_transformedRect.move(offset);
96 m_boundingBox = enclosingIntRect(m_transformedRect.boundingBox());
97}

99template<typename RectType>
100bool HitTestLocation::intersectsRect(const RectType& rect) const
101{
102 // FIXME: When the hit test is not rect based we should use rect.contains(m_point).
103 // That does change some corner case tests though.
104
105 // First check if rect even intersects our bounding box.
106 if (!rect.intersects(m_boundingBox))
107 return false;
108
109 // If the transformed rect is rectilinear the bounding box intersection was accurate.
110 if (m_isRectilinear)
111 return true;
112
113 // If rect fully contains our bounding box, we are also sure of an intersection.
114 if (rect.contains(m_boundingBox))
115 return true;
116
117 // Otherwise we need to do a slower quad based intersection test.
118 return m_transformedRect.intersectsRect(rect);
119}
120
121bool HitTestLocation::intersects(const LayoutRect& rect) const
122{
123 return intersectsRect(rect);
124}
125
126bool HitTestLocation::intersects(const FloatRect& rect) const
127{
128 return intersectsRect(rect);
129}
130
131bool HitTestLocation::intersects(const RoundedRect& rect) const
132{
133 return rect.intersectsQuad(m_transformedRect);
134}
