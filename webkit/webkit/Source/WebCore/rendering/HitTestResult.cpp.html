58static inline void appendToNodeSet(const HitTestResult::NodeSet& source, HitTestResult::NodeSet& destination)
59{
60 for (auto& node : source)
61 destination.add(node.copyRef());

123static Node* moveOutOfUserAgentShadowTree(Node& node)
124{
125 if (node.isInShadowTree()) {
126 if (ShadowRoot* root = node.containingShadowRoot()) {
127 if (root->mode() == ShadowRootMode::UserAgent)
128 return root->host();
129 }
130 }
131 return &node;
132}

203String HitTestResult::selectedText() const
204{
205 if (!m_innerNonSharedNode)
206 return emptyString();
207
208 Frame* frame = m_innerNonSharedNode->document().frame();
209 if (!frame)
210 return emptyString();
211
212 auto range = frame->selection().selection().toNormalizedRange();
213 if (!range)
214 return emptyString();
215
216 // Look for a character that's not just a separator.
217 for (TextIterator it(*range); !it.atEnd(); it.advance()) {
218 int length = it.text().length();
219 for (int i = 0; i < length; ++i) {
220 if (!(U_GET_GC_MASK(it.text()[i]) & U_GC_Z_MASK))
221 return frame->displayStringModifiedByEncoding(frame->editor().selectedText());
222 }
223 }
224 return emptyString();
225}

227String HitTestResult::spellingToolTip(TextDirection& dir) const
228{
229 dir = TextDirection::LTR;
230 // Return the tool tip string associated with this point, if any. Only markers associated with bad grammar
231 // currently supply strings, but maybe someday markers associated with misspelled words will also.
232 if (!m_innerNonSharedNode)
233 return String();
234
235 DocumentMarker* marker = m_innerNonSharedNode->document().markers().markerContainingPoint(m_hitTestLocation.point(), DocumentMarker::Grammar);
236 if (!marker)
237 return String();
238
239 if (auto renderer = m_innerNonSharedNode->renderer())
240 dir = renderer->style().direction();
241 return marker->description();
242}

258String HitTestResult::title(TextDirection& dir) const
259{
260 dir = TextDirection::LTR;
261 // Find the title in the nearest enclosing DOM node.
262 // For <area> tags in image maps, walk the tree for the <area>, not the <img> using it.
263 for (Node* titleNode = m_innerNode.get(); titleNode; titleNode = titleNode->parentInComposedTree()) {
264 if (is<Element>(*titleNode)) {
265 Element& titleElement = downcast<Element>(*titleNode);
266 String title = titleElement.title();
267 if (!title.isEmpty()) {
268 if (auto renderer = titleElement.renderer())
269 dir = renderer->style().direction();
270 return title;
271 }
272 }
273 }
274 return String();
275}
276
277String HitTestResult::innerTextIfTruncated(TextDirection& dir) const
278{
279 for (Node* truncatedNode = m_innerNode.get(); truncatedNode; truncatedNode = truncatedNode->parentInComposedTree()) {
280 if (!is<Element>(*truncatedNode))
281 continue;
282
283 if (auto renderer = downcast<Element>(*truncatedNode).renderer()) {
284 if (is<RenderBlockFlow>(*renderer)) {
285 RenderBlockFlow& block = downcast<RenderBlockFlow>(*renderer);
286 if (block.style().textOverflow() == TextOverflow::Ellipsis) {
287 for (auto* line = block.firstRootBox(); line; line = line->nextRootBox()) {
288 if (line->hasEllipsisBox()) {
289 dir = block.style().direction();
290 return downcast<Element>(*truncatedNode).innerText();
291 }
292 }
293 }
294 break;
295 }
296 }
297 }
298
299 dir = TextDirection::LTR;
300 return String();
301}

366URL HitTestResult::absoluteImageURL() const
367{
368 auto imageNode = nodeForImageData();
369 if (!imageNode)
370 return { };
371
372 auto renderer = imageNode->renderer();
373 if (!renderer || !renderer->isImage())
374 return { };
375
376 if (is<HTMLEmbedElement>(*imageNode)
377 || is<HTMLImageElement>(*imageNode)
378 || is<HTMLInputElement>(*imageNode)
379 || is<HTMLObjectElement>(*imageNode)
380 || is<SVGImageElement>(*imageNode))
381 return imageNode->document().completeURL(downcast<Element>(*imageNode).imageSourceURL());
382
383 return { };
384}

386URL HitTestResult::absolutePDFURL() const
387{
388 if (!m_innerNonSharedNode)
389 return URL();
390
391 if (!is<HTMLEmbedElement>(*m_innerNonSharedNode) && !is<HTMLObjectElement>(*m_innerNonSharedNode))
392 return URL();
393
394 HTMLPlugInImageElement& element = downcast<HTMLPlugInImageElement>(*m_innerNonSharedNode);
395 URL url = m_innerNonSharedNode->document().completeURL(stripLeadingAndTrailingHTMLSpaces(element.url()));
396 if (!url.isValid())
397 return URL();
398
399 if (element.serviceType() == "application/pdf" || (element.serviceType().isEmpty() && url.path().endsWithIgnoringASCIICase(".pdf")))
400 return url;
401 return URL();

571bool HitTestResult::isOverTextInsideFormControlElement() const
572{
573 Node* node = innerNode();
574 if (!node)
575 return false;
576
577 if (!is<Element>(*node) || !downcast<Element>(*node).isTextField())
578 return false;
579
580 Frame* frame = node->document().frame();
581 if (!frame)
582 return false;
583
584 IntPoint framePoint = roundedPointInInnerNodeFrame();
585 if (!frame->rangeForPoint(framePoint))
586 return false;
587
588 VisiblePosition position = frame->visiblePositionForPoint(framePoint);
589 if (position.isNull())
590 return false;
591
592 auto wordRange = enclosingTextUnitOfGranularity(position, TextGranularity::WordGranularity, SelectionDirection::Forward);
593 return wordRange && hasAnyPlainText(*wordRange);
594}

641template<typename RectType>
642inline HitTestProgress HitTestResult::addNodeToListBasedTestResultCommon(Node* node, const HitTestRequest& request, const HitTestLocation& locationInContainer, const RectType& rect)
643{
644 // If it is not a list-based hit test, this method has to be no-op.
645 if (!request.resultIsElementList()) {
646 ASSERT(!isRectBasedTest());
647 return HitTestProgress::Stop;
648 }
649
650 if (!node)
651 return HitTestProgress::Continue;
652
653 if ((request.disallowsUserAgentShadowContent() && node->isInUserAgentShadowTree())
654 || (request.disallowsUserAgentShadowContentExceptForImageOverlays() && !HTMLElement::isInsideImageOverlay(*node) && node->isInUserAgentShadowTree()))
655 node = node->document().ancestorNodeInThisScope(node);
656
657 mutableListBasedTestResult().add(*node);
658
659 if (request.includesAllElementsUnderPoint())
660 return HitTestProgress::Continue;
661
662 bool regionFilled = rect.contains(locationInContainer.boundingBox());
663 return regionFilled ? HitTestProgress::Stop : HitTestProgress::Continue;
664}
665
666HitTestProgress HitTestResult::addNodeToListBasedTestResult(Node* node, const HitTestRequest& request, const HitTestLocation& locationInContainer, const LayoutRect& rect)
667{
668 return addNodeToListBasedTestResultCommon(node, request, locationInContainer, rect);
669}
670
671HitTestProgress HitTestResult::addNodeToListBasedTestResult(Node* node, const HitTestRequest& request, const HitTestLocation& locationInContainer, const FloatRect& rect)
672{
673 return addNodeToListBasedTestResultCommon(node, request, locationInContainer, rect);
674}

676void HitTestResult::append(const HitTestResult& other, const HitTestRequest& request)
677{
678 ASSERT_UNUSED(request, request.resultIsElementList());
679
680 if (!m_innerNode && other.innerNode()) {
681 m_innerNode = other.innerNode();
682 m_innerNonSharedNode = other.innerNonSharedNode();
683 m_localPoint = other.localPoint();
684 m_pointInInnerNodeFrame = other.m_pointInInnerNodeFrame;
685 m_innerURLElement = other.URLElement();
686 m_scrollbar = other.scrollbar();
687 m_isOverWidget = other.isOverWidget();
688 }
689
690 if (other.m_listBasedTestResult)
691 appendToNodeSet(*other.m_listBasedTestResult, mutableListBasedTestResult());
692}

709{

711 if (!m_innerNonSharedNode)
