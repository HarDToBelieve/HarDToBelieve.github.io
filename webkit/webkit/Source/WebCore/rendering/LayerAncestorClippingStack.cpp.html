36LayerAncestorClippingStack::LayerAncestorClippingStack(Vector<CompositedClipData>&& clipDataStack)
37{
38 m_stack.reserveInitialCapacity(clipDataStack.size());
39 for (auto& clipDataEntry : clipDataStack)
40 m_stack.uncheckedAppend({ WTFMove(clipDataEntry), 0, nullptr });
41}
42
43bool LayerAncestorClippingStack::equalToClipData(const Vector<CompositedClipData>& clipDataStack) const
44{
45 if (m_stack.size() != clipDataStack.size())
46 return false;
47
48 for (unsigned i = 0; i < m_stack.size(); ++i) {
49 if (m_stack[i].clipData != clipDataStack[i])
50 return false;
51 }
52
53 return true;
54}
55
56bool LayerAncestorClippingStack::hasAnyScrollingLayers() const
57{
58 for (const auto& entry : m_stack) {
59 if (entry.clipData.isOverflowScroll)
60 return true;
61 }
62
63 return false;
64}

66void LayerAncestorClippingStack::clear(ScrollingCoordinator* scrollingCoordinator)
67{
68 for (auto& entry : m_stack) {
69 if (entry.overflowScrollProxyNodeID) {
70 ASSERT(scrollingCoordinator);
71 scrollingCoordinator->unparentChildrenAndDestroyNode(entry.overflowScrollProxyNodeID);
72 entry.overflowScrollProxyNodeID = 0;
73 }
74
75 GraphicsLayer::unparentAndClear(entry.clippingLayer);
76 }
77}
78
79void LayerAncestorClippingStack::detachFromScrollingCoordinator(ScrollingCoordinator& scrollingCoordinator)
80{
81 for (auto& entry : m_stack) {
82 if (entry.overflowScrollProxyNodeID) {
83 scrollingCoordinator.unparentChildrenAndDestroyNode(entry.overflowScrollProxyNodeID);
84 entry.overflowScrollProxyNodeID = 0;
85 }
86 }
87}

99ScrollingNodeID LayerAncestorClippingStack::lastOverflowScrollProxyNodeID() const
100{
101 for (auto& entry : WTF::makeReversedRange(m_stack)) {
102 if (entry.overflowScrollProxyNodeID)
103 return entry.overflowScrollProxyNodeID;
104 }
105
106 return 0;
107}

109void LayerAncestorClippingStack::updateScrollingNodeLayers(ScrollingCoordinator& scrollingCoordinator)
110{
111 for (const auto& entry : m_stack) {
112 if (!entry.clipData.isOverflowScroll)
113 continue;
114
115 scrollingCoordinator.setNodeLayers(entry.overflowScrollProxyNodeID, { entry.clippingLayer.get() });
116 }
117}

119bool LayerAncestorClippingStack::updateWithClipData(ScrollingCoordinator* scrollingCoordinator, Vector<CompositedClipData>&& clipDataStack)
120{
121 bool stackChanged = false;
122
123 int clipEntryCount = clipDataStack.size();
124 int stackEntryCount = m_stack.size();
125 for (int i = 0; i < clipEntryCount; ++i) {
126 auto& clipDataEntry = clipDataStack[i];
127
128 if (i >= stackEntryCount) {
129 m_stack.append({ WTFMove(clipDataEntry), 0, nullptr });
130 stackChanged = true;
131 continue;
132 }
133
134 auto& existingEntry = m_stack[i];
135
136 if (existingEntry.clipData != clipDataEntry)
137 stackChanged = true;
138
139 if (existingEntry.clipData.isOverflowScroll && !clipDataEntry.isOverflowScroll) {
140 ASSERT(scrollingCoordinator);
141 scrollingCoordinator->unparentChildrenAndDestroyNode(existingEntry.overflowScrollProxyNodeID);
142 existingEntry.overflowScrollProxyNodeID = 0;
143 }
144
145 existingEntry.clipData = WTFMove(clipDataEntry);
146 }
147
148 if (stackEntryCount > clipEntryCount) {
149 for (int i = clipEntryCount; i < stackEntryCount; ++i) {
150 auto& entry = m_stack[i];
151 if (entry.overflowScrollProxyNodeID) {
152 ASSERT(scrollingCoordinator);
153 scrollingCoordinator->unparentChildrenAndDestroyNode(entry.overflowScrollProxyNodeID);
154 }
155 GraphicsLayer::unparentAndClear(entry.clippingLayer);
156 }
157
158 m_stack.shrink(clipEntryCount);
159 stackChanged = true;
160 } else
161 m_stack.shrinkToFit();
162
163 return stackChanged;
164}

166Vector<CompositedClipData> LayerAncestorClippingStack::compositedClipData() const
167{
168 Vector<CompositedClipData> clipData;
169 clipData.reserveInitialCapacity(m_stack.size());
170
171 for (const auto& entry : m_stack)
172 clipData.uncheckedAppend(entry.clipData);
173
174 return clipData;
175}

177static TextStream& operator<<(TextStream& ts, const LayerAncestorClippingStack::ClippingStackEntry& entry)
178{
179 ts.dumpProperty("layer", entry.clipData.clippingLayer.get());
180 ts.dumpProperty("clip", entry.clipData.clipRect);
181 ts.dumpProperty("isOverflowScroll", entry.clipData.isOverflowScroll);
182 if (entry.overflowScrollProxyNodeID)
183 ts.dumpProperty("overflowScrollProxyNodeID", entry.overflowScrollProxyNodeID);
184 if (entry.clippingLayer)
185 ts.dumpProperty("clippingLayer", entry.clippingLayer->primaryLayerID());
186 return ts;
187}

189TextStream& operator<<(TextStream& ts, const LayerAncestorClippingStack& clipStack)
190{
191 TextStream multilineStream;
192 multilineStream.setIndent(ts.indent() + 2);
193
194 {
195 TextStream::GroupScope scope(ts);
196 multilineStream << "LayerAncestorClippingStack";
197
198 for (unsigned i = 0; i < clipStack.stack().size(); ++i) {
199 auto& entry = clipStack.stack()[i];
200 TextStream::GroupScope entryScope(multilineStream);
201 multilineStream << "entry " << i;
202 multilineStream << entry;
203 }
204
205 ts << multilineStream.release();
206 }
207 return ts;
208}
