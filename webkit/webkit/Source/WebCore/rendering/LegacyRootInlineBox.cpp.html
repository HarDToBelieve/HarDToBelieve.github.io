118bool LegacyRootInlineBox::lineCanAccommodateEllipsis(bool ltr, int blockEdge, int lineBoxEdge, int ellipsisWidth)
119{
120 // First sanity-check the unoverflowed width of the whole line to see if there is sufficient room.
121 int delta = ltr ? lineBoxEdge - blockEdge : blockEdge - lineBoxEdge;
122 if (logicalWidth() - delta < ellipsisWidth)
123 return false;
124
125 // Next iterate over all the line boxes on the line. If we find a replaced element that intersects
126 // then we refuse to accommodate the ellipsis. Otherwise we're ok.
127 return LegacyInlineFlowBox::canAccommodateEllipsis(ltr, blockEdge, ellipsisWidth);
128}

130float LegacyRootInlineBox::placeEllipsis(const AtomString& ellipsisStr, bool ltr, float blockLeftEdge, float blockRightEdge, float ellipsisWidth, LegacyInlineBox* markupBox)
131{
132 if (!gEllipsisBoxMap)
133 gEllipsisBoxMap = new EllipsisBoxMap();
134
135 ASSERT(!hasEllipsisBox());
136 auto* ellipsisBox = gEllipsisBoxMap->set(this, makeUnique<LegacyEllipsisBox>(blockFlow(), ellipsisStr, this, ellipsisWidth - (markupBox ? markupBox->logicalWidth() : 0), logicalHeight(), y(), !prevRootBox(), isHorizontal(), markupBox)).iterator->value.get();
137 setHasEllipsisBox(true);
138 // FIXME: Do we need an RTL version of this?
139 if (ltr && (x() + logicalWidth() + ellipsisWidth) <= blockRightEdge) {
140 ellipsisBox->setX(x() + logicalWidth());
141 return logicalWidth() + ellipsisWidth;
142 }
143
144 // Now attempt to find the nearest glyph horizontally and place just to the right (or left in RTL)
145 // of that glyph. Mark all of the objects that intersect the ellipsis box as not painting (as being
146 // truncated).
147 bool foundBox = false;
148 float truncatedWidth = 0;
149 float position = placeEllipsisBox(ltr, blockLeftEdge, blockRightEdge, ellipsisWidth, truncatedWidth, foundBox);
150 ellipsisBox->setX(position);
151 return truncatedWidth;
152}

154float LegacyRootInlineBox::placeEllipsisBox(bool ltr, float blockLeftEdge, float blockRightEdge, float ellipsisWidth, float &truncatedWidth, bool& foundBox)
155{
156 float result = LegacyInlineFlowBox::placeEllipsisBox(ltr, blockLeftEdge, blockRightEdge, ellipsisWidth, truncatedWidth, foundBox);
157 if (result == -1) {
158 result = ltr ? blockRightEdge - ellipsisWidth : blockLeftEdge;
159 truncatedWidth = blockRightEdge - blockLeftEdge;
160 }
161 return result;
162}

244LayoutUnit LegacyRootInlineBox::alignBoxesInBlockDirection(LayoutUnit heightOfBlock, GlyphOverflowAndFallbackFontsMap& textBoxDataMap, VerticalPositionCache& verticalPositionCache)
245{
246 // SVG will handle vertical alignment on its own.
247 if (isSVGRootInlineBox())
248 return 0;
249
250 LayoutUnit maxPositionTop;
251 LayoutUnit maxPositionBottom;
252 LayoutUnit maxAscent;
253 LayoutUnit maxDescent;
254 bool setMaxAscent = false;
255 bool setMaxDescent = false;
256
257 // Figure out if we're in no-quirks mode.
258 bool noQuirksMode = renderer().document().inNoQuirksMode();
259
260 m_baselineType = requiresIdeographicBaseline(textBoxDataMap) ? IdeographicBaseline : AlphabeticBaseline;
261
262 computeLogicalBoxHeights(*this, maxPositionTop, maxPositionBottom, maxAscent, maxDescent, setMaxAscent, setMaxDescent, noQuirksMode,
263 textBoxDataMap, baselineType(), verticalPositionCache);
264
265 if (maxAscent + maxDescent < std::max(maxPositionTop, maxPositionBottom))
266 adjustMaxAscentAndDescent(maxAscent, maxDescent, maxPositionTop, maxPositionBottom);
267
268 LayoutUnit maxHeight = maxAscent + maxDescent;
269 LayoutUnit lineTop = heightOfBlock;
270 LayoutUnit lineBottom = heightOfBlock;
271 LayoutUnit lineTopIncludingMargins = heightOfBlock;
272 LayoutUnit lineBottomIncludingMargins = heightOfBlock;
273 bool setLineTop = false;
274 bool hasAnnotationsBefore = false;
275 bool hasAnnotationsAfter = false;
276 placeBoxesInBlockDirection(heightOfBlock, maxHeight, maxAscent, noQuirksMode, lineTop, lineBottom, setLineTop,
277 lineTopIncludingMargins, lineBottomIncludingMargins, hasAnnotationsBefore, hasAnnotationsAfter, baselineType());
278 m_hasAnnotationsBefore = hasAnnotationsBefore;
279 m_hasAnnotationsAfter = hasAnnotationsAfter;
280
281 maxHeight = std::max<LayoutUnit>(0, maxHeight); // FIXME: Is this really necessary?
282
283 LayoutUnit lineBoxTop = heightOfBlock;
284 LayoutUnit lineBoxBottom = heightOfBlock + maxHeight;
285 setLineTopBottomPositions(lineTop, lineBottom, lineBoxTop, lineBoxBottom);
286 setPaginatedLineWidth(blockFlow().availableLogicalWidthForContent(heightOfBlock));
287
288 LayoutUnit annotationsAdjustment = beforeAnnotationsAdjustment();
289 if (annotationsAdjustment) {
290 // FIXME: Need to handle pagination here. We might have to move to the next page/column as a result of the
291 // ruby expansion.
292 adjustBlockDirectionPosition(annotationsAdjustment);
293 heightOfBlock += annotationsAdjustment;
294 }
295
296 LayoutUnit gridSnapAdjustment = lineSnapAdjustment();
297 if (gridSnapAdjustment) {
298 adjustBlockDirectionPosition(gridSnapAdjustment);
299 heightOfBlock += gridSnapAdjustment;
300 }
301
302 return heightOfBlock + maxHeight;
303}
304
305LayoutUnit LegacyRootInlineBox::beforeAnnotationsAdjustment() const
306{
307 LayoutUnit result;
308
309 if (!renderer().style().isFlippedLinesWritingMode()) {
310 // Annotations under the previous line may push us down.
311 if (prevRootBox() && prevRootBox()->hasAnnotationsAfter())
312 result = prevRootBox()->computeUnderAnnotationAdjustment(lineTop());
313
314 if (!hasAnnotationsBefore())
315 return result;
316
317 // Annotations over this line may push us further down.
318 LayoutUnit highestAllowedPosition = prevRootBox() ? std::min(prevRootBox()->lineBottom(), lineTop()) + result : blockFlow().borderBefore();
319 result = computeOverAnnotationAdjustment(highestAllowedPosition);
320 } else {
321 // Annotations under this line may push us up.
322 if (hasAnnotationsBefore())
323 result = computeUnderAnnotationAdjustment(prevRootBox() ? prevRootBox()->lineBottom() : blockFlow().borderBefore());
324
325 if (!prevRootBox() || !prevRootBox()->hasAnnotationsAfter())
326 return result;
327
328 // We have to compute the expansion for annotations over the previous line to see how much we should move.
329 LayoutUnit lowestAllowedPosition = std::max(prevRootBox()->lineBottom(), lineTop()) - result;
330 result = prevRootBox()->computeOverAnnotationAdjustment(lowestAllowedPosition);
331 }
332
333 return result;
334}

336LayoutUnit LegacyRootInlineBox::lineSnapAdjustment(LayoutUnit delta) const
337{
338 // If our block doesn't have snapping turned on, do nothing.
339 // FIXME: Implement bounds snapping.
340 if (blockFlow().style().lineSnap() == LineSnap::None)
341 return 0;
342
343 // Get the current line grid and offset.
344 auto* layoutState = blockFlow().view().frameView().layoutContext().layoutState();
345 RenderBlockFlow* lineGrid = layoutState->lineGrid();
346 LayoutSize lineGridOffset = layoutState->lineGridOffset();
347 if (!lineGrid || lineGrid->style().writingMode() != blockFlow().style().writingMode())
348 return 0;
349
350 // Get the hypothetical line box used to establish the grid.
351 LegacyRootInlineBox* lineGridBox = lineGrid->lineGridBox();
352 if (!lineGridBox)
353 return 0;
354
355 LayoutUnit lineGridBlockOffset = lineGrid->isHorizontalWritingMode() ? lineGridOffset.height() : lineGridOffset.width();
356 LayoutUnit blockOffset = blockFlow().isHorizontalWritingMode() ? layoutState->layoutOffset().height() : layoutState->layoutOffset().width();
357
358 // Now determine our position on the grid. Our baseline needs to be adjusted to the nearest baseline multiple
359 // as established by the line box.
360 // FIXME: Need to handle crazy line-box-contain values that cause the root line box to not be considered. I assume
361 // the grid should honor line-box-contain.
362 LayoutUnit gridLineHeight = lineGridBox->lineBoxBottom() - lineGridBox->lineBoxTop();
363 if (!gridLineHeight)
364 return 0;
365
366 LayoutUnit lineGridFontAscent = lineGrid->style().fontMetrics().ascent(baselineType());
367 LayoutUnit lineGridFontHeight { lineGridBox->logicalHeight() };
368 LayoutUnit firstTextTop { lineGridBlockOffset + lineGridBox->logicalTop() };
369 LayoutUnit firstLineTopWithLeading = lineGridBlockOffset + lineGridBox->lineBoxTop();
370 LayoutUnit firstBaselinePosition = firstTextTop + lineGridFontAscent;
371
372 LayoutUnit currentTextTop { blockOffset + logicalTop() + delta };
373 LayoutUnit currentFontAscent = blockFlow().style().fontMetrics().ascent(baselineType());
374 LayoutUnit currentBaselinePosition = currentTextTop + currentFontAscent;
375
376 LayoutUnit lineGridPaginationOrigin = isHorizontal() ? layoutState->lineGridPaginationOrigin().height() : layoutState->lineGridPaginationOrigin().width();
377
378 // If we're paginated, see if we're on a page after the first one. If so, the grid resets on subsequent pages.
379 // FIXME: If the grid is an ancestor of the pagination establisher, then this is incorrect.
380 LayoutUnit pageLogicalTop;
381 if (layoutState->isPaginated() && layoutState->pageLogicalHeight()) {
382 pageLogicalTop = blockFlow().pageLogicalTopForOffset(lineBoxTop() + delta);
383 if (pageLogicalTop > firstLineTopWithLeading)
384 firstTextTop = pageLogicalTop + lineGridBox->logicalTop() - lineGrid->borderAndPaddingBefore() + lineGridPaginationOrigin;
385 }
386
387 if (blockFlow().style().lineSnap() == LineSnap::Contain) {
388 // Compute the desired offset from the text-top of a grid line.
389 // Look at our height (logicalHeight()).
390 // Look at the total available height. It's going to be (textBottom - textTop) + (n-1)*(multiple with leading)
391 // where n is number of grid lines required to enclose us.
392 if (logicalHeight() <= lineGridFontHeight)
393 firstTextTop += (lineGridFontHeight - logicalHeight()) / 2;
394 else {
395 LayoutUnit numberOfLinesWithLeading { ceilf(static_cast<float>(logicalHeight() - lineGridFontHeight) / gridLineHeight) };
396 LayoutUnit totalHeight = lineGridFontHeight + numberOfLinesWithLeading * gridLineHeight;
397 firstTextTop += (totalHeight - logicalHeight()) / 2;
398 }
399 firstBaselinePosition = firstTextTop + currentFontAscent;
400 } else
401 firstBaselinePosition = firstTextTop + lineGridFontAscent;
402
403 // If we're above the first line, just push to the first line.
404 if (currentBaselinePosition < firstBaselinePosition)
405 return delta + firstBaselinePosition - currentBaselinePosition;
406
407 // Otherwise we're in the middle of the grid somewhere. Just push to the next line.
408 LayoutUnit baselineOffset = currentBaselinePosition - firstBaselinePosition;
409 LayoutUnit remainder = roundToInt(baselineOffset) % roundToInt(gridLineHeight);
410 LayoutUnit result = delta;
411 if (remainder)
412 result += gridLineHeight - remainder;
413
414 // If we aren't paginated we can return the result.
415 if (!layoutState->isPaginated() || !layoutState->pageLogicalHeight() || result == delta)
416 return result;
417
418 // We may end up shifted to a new page. We need to do a re-snap when that happens.
419 LayoutUnit newPageLogicalTop = blockFlow().pageLogicalTopForOffset(lineBoxBottom() + result);
420 if (newPageLogicalTop == pageLogicalTop)
421 return result;
422
423 auto pageLogicalHeight = blockFlow().pageLogicalHeightForOffset(lineBoxBottom() + result);
424 // Put ourselves at the top of the next page to force a snap onto the new grid established by that page unless
425 // the line is taller than the page. In such cases let's just leave the line overflow.
426 if (pageLogicalHeight < lineBoxBottom() + result)
427 return result;
428 return lineSnapAdjustment(newPageLogicalTop - (blockOffset + lineBoxTop()));
429}

431GapRects LegacyRootInlineBox::lineSelectionGap(RenderBlock& rootBlock, const LayoutPoint& rootBlockPhysicalPosition, const LayoutSize& offsetFromRootBlock,
432 LayoutUnit selTop, LayoutUnit selHeight, const LogicalSelectionOffsetCaches& cache, const PaintInfo* paintInfo)
433{
434 RenderObject::HighlightState lineState = selectionState();
435
436 bool leftGap, rightGap;
437 blockFlow().getSelectionGapInfo(lineState, leftGap, rightGap);
438
439 GapRects result;
440
441 LegacyInlineBox* firstBox = firstSelectedBox();
442 LegacyInlineBox* lastBox = lastSelectedBox();
443 if (leftGap) {
444 result.uniteLeft(blockFlow().logicalLeftSelectionGap(rootBlock, rootBlockPhysicalPosition, offsetFromRootBlock, &firstBox->parent()->renderer(), LayoutUnit(firstBox->logicalLeft()),
445 selTop, selHeight, cache, paintInfo));
446 }
447 if (rightGap) {
448 result.uniteRight(blockFlow().logicalRightSelectionGap(rootBlock, rootBlockPhysicalPosition, offsetFromRootBlock, &lastBox->parent()->renderer(), LayoutUnit(lastBox->logicalRight()),
449 selTop, selHeight, cache, paintInfo));
450 }
451
452 // When dealing with bidi text, a non-contiguous selection region is possible.
453 // e.g. The logical text aaaAAAbbb (capitals denote RTL text and non-capitals LTR) is layed out
454 // visually as 3 text runs |aaa|bbb|AAA| if we select 4 characters from the start of the text the
455 // selection will look like (underline denotes selection):
456 // |aaa|bbb|AAA|
457 // ___ _
458 // We can see that the |bbb| run is not part of the selection while the runs around it are.
459 if (firstBox && firstBox != lastBox) {
460 // Now fill in any gaps on the line that occurred between two selected elements.
461 LayoutUnit lastLogicalLeft { firstBox->logicalRight() };
462 bool isPreviousBoxSelected = firstBox->selectionState() != RenderObject::HighlightState::None;
463 for (auto* box = firstBox->nextLeafOnLine(); box; box = box->nextLeafOnLine()) {
464 if (box->selectionState() != RenderObject::HighlightState::None) {
465 LayoutRect logicalRect { lastLogicalLeft, selTop, LayoutUnit(box->logicalLeft() - lastLogicalLeft), selHeight };
466 logicalRect.move(renderer().isHorizontalWritingMode() ? offsetFromRootBlock : LayoutSize(offsetFromRootBlock.height(), offsetFromRootBlock.width()));
467 LayoutRect gapRect = rootBlock.logicalRectToPhysicalRect(rootBlockPhysicalPosition, logicalRect);
468 if (isPreviousBoxSelected && gapRect.width() > 0 && gapRect.height() > 0) {
469 if (paintInfo && box->parent()->renderer().style().visibility() == Visibility::Visible)
470 paintInfo->context().fillRect(gapRect, box->parent()->renderer().selectionBackgroundColor());
471 // VisibleSelection may be non-contiguous, see comment above.
472 result.uniteCenter(gapRect);
473 }
474 lastLogicalLeft = box->logicalRight();
475 }
476 if (box == lastBox)
477 break;
478 isPreviousBoxSelected = box->selectionState() != RenderObject::HighlightState::None;
479 }
480 }
481
482 return result;
483}

485RenderObject::HighlightState LegacyRootInlineBox::selectionState()
486{
487 // Walk over all of the selected boxes.
488 RenderObject::HighlightState state = RenderObject::HighlightState::None;
489 for (auto* box = firstLeafDescendant(); box; box = box->nextLeafOnLine()) {
490 RenderObject::HighlightState boxState = box->selectionState();
491 if ((boxState == RenderObject::HighlightState::Start && state == RenderObject::HighlightState::End)
492 || (boxState == RenderObject::HighlightState::End && state == RenderObject::HighlightState::Start))
493 state = RenderObject::HighlightState::Both;
494 else if (state == RenderObject::HighlightState::None || ((boxState == RenderObject::HighlightState::Start || boxState == RenderObject::HighlightState::End)
495 && (state == RenderObject::HighlightState::None || state == RenderObject::HighlightState::Inside)))
496 state = boxState;
497 else if (boxState == RenderObject::HighlightState::None && state == RenderObject::HighlightState::Start) {
498 // We are past the end of the selection.
499 state = RenderObject::HighlightState::Both;
500 }
501 if (state == RenderObject::HighlightState::Both)
502 break;
503 }
504
505 return state;
506}

526LayoutUnit LegacyRootInlineBox::selectionTop(ForHitTesting forHitTesting) const
527{
528 LayoutUnit selectionTop = m_lineTop;
529
530 if (m_hasAnnotationsBefore)
531 selectionTop -= !renderer().style().isFlippedLinesWritingMode() ? computeOverAnnotationAdjustment(m_lineTop) : computeUnderAnnotationAdjustment(m_lineTop);
532
533 if (renderer().style().isFlippedLinesWritingMode())
534 return selectionTop;
535
536#if !PLATFORM(IOS_FAMILY)
537 // See rdar://problem/19692206 ... don't want to do this adjustment for iOS where overlap is ok and handled.
538 if (renderer().isRubyBase()) {
539 // The ruby base selection should avoid intruding into the ruby text. This is only the case if there is an actual ruby text above us.
540 RenderRubyBase* base = &downcast<RenderRubyBase>(renderer());
541 RenderRubyRun* run = base->rubyRun();
542 if (run) {
543 RenderRubyText* text = run->rubyText();
544 if (text && text->logicalTop() < base->logicalTop()) {
545 // The ruby text is above the ruby base. Just return now in order to avoid painting on top of the ruby text.
546 return selectionTop;
547 }
548 }
549 } else if (renderer().isRubyText()) {
550 // The ruby text selection should go all the way to the selection top of the containing line.
551 RenderRubyText* text = &downcast<RenderRubyText>(renderer());
552 RenderRubyRun* run = text->rubyRun();
553 if (run && run->inlineBoxWrapper()) {
554 RenderRubyBase* base = run->rubyBase();
555 if (base && text->logicalTop() < base->logicalTop()) {
556 // The ruby text is above the ruby base.
557 const LegacyRootInlineBox& containingLine = run->inlineBoxWrapper()->root();
558 LayoutUnit enclosingSelectionTop = containingLine.selectionTop();
559 LayoutUnit deltaBetweenObjects = text->logicalTop() + run->logicalTop();
560 LayoutUnit selectionTopInRubyTextCoords = enclosingSelectionTop - deltaBetweenObjects;
561 return std::min(selectionTop, selectionTopInRubyTextCoords);
562 }
563 }
564 }
565#endif
566
567 LayoutUnit prevBottom;
568 if (auto* previousBox = prevRootBox())
569 prevBottom = previousBox->selectionBottom();
570 else {
571 auto borderAndPaddingBefore = blockFlow().borderAndPaddingBefore();
572 prevBottom = forHitTesting == ForHitTesting::Yes ? borderAndPaddingBefore : std::max(borderAndPaddingBefore, selectionTop);
573 }
574
575 if (prevBottom < selectionTop && blockFlow().containsFloats()) {
576 // This line has actually been moved further down, probably from a large line-height, but possibly because the
577 // line was forced to clear floats. If so, let's check the offsets, and only be willing to use the previous
578 // line's bottom if the offsets are greater on both sides.
579 LayoutUnit prevLeft = blockFlow().logicalLeftOffsetForLine(prevBottom, DoNotIndentText);
580 LayoutUnit prevRight = blockFlow().logicalRightOffsetForLine(prevBottom, DoNotIndentText);
581 LayoutUnit newLeft = blockFlow().logicalLeftOffsetForLine(selectionTop, DoNotIndentText);
582 LayoutUnit newRight = blockFlow().logicalRightOffsetForLine(selectionTop, DoNotIndentText);
583 if (prevLeft > newLeft || prevRight < newRight)
584 return selectionTop;
585 }
586
587 return prevBottom;
588}

590static RenderBlock* blockBeforeWithinSelectionRoot(const RenderBlockFlow& blockFlow, LayoutSize& offset)
591{
592 if (blockFlow.isSelectionRoot())
593 return nullptr;
594
595 const RenderElement* object = &blockFlow;
596 RenderObject* sibling;
597 do {
598 sibling = object->previousSibling();
599 while (sibling && (!is<RenderBlock>(*sibling) || downcast<RenderBlock>(*sibling).isSelectionRoot()))
600 sibling = sibling->previousSibling();
601
602 offset -= LayoutSize(downcast<RenderBlock>(*object).logicalLeft(), downcast<RenderBlock>(*object).logicalTop());
603 object = object->parent();
604 } while (!sibling && is<RenderBlock>(object) && !downcast<RenderBlock>(*object).isSelectionRoot());
605
606 if (!sibling)
607 return nullptr;
608
609 RenderBlock* beforeBlock = downcast<RenderBlock>(sibling);
610
611 offset += LayoutSize(beforeBlock->logicalLeft(), beforeBlock->logicalTop());
612
613 RenderObject* child = beforeBlock->lastChild();
614 while (is<RenderBlock>(child)) {
615 beforeBlock = downcast<RenderBlock>(child);
616 offset += LayoutSize(beforeBlock->logicalLeft(), beforeBlock->logicalTop());
617 child = beforeBlock->lastChild();
618 }
619 return beforeBlock;
620}

622LayoutUnit LegacyRootInlineBox::selectionTopAdjustedForPrecedingBlock() const
623{
624 const LegacyRootInlineBox& rootBox = root();
625 LayoutUnit top = selectionTop();
626
627 auto blockSelectionState = rootBox.blockFlow().selectionState();
628 if (blockSelectionState != RenderObject::HighlightState::Inside && blockSelectionState != RenderObject::HighlightState::End)
629 return top;
630
631 LayoutSize offsetToBlockBefore;
632 auto* blockBefore = blockBeforeWithinSelectionRoot(rootBox.blockFlow(), offsetToBlockBefore);
633 if (!is<RenderBlockFlow>(blockBefore))
634 return top;
635
636 // Do not adjust blocks sharing the same line.
637 if (!offsetToBlockBefore.height())
638 return top;
639
640 if (auto* lastLine = downcast<RenderBlockFlow>(*blockBefore).lastRootBox()) {
641 RenderObject::HighlightState lastLineSelectionState = lastLine->selectionState();
642 if (lastLineSelectionState != RenderObject::HighlightState::Inside && lastLineSelectionState != RenderObject::HighlightState::Start)
643 return top;
644
645 LayoutUnit lastLineSelectionBottom = lastLine->selectionBottom() + offsetToBlockBefore.height();
646 top = std::max(top, lastLineSelectionBottom);
647 }
648 return top;
649}
650
651LayoutUnit LegacyRootInlineBox::selectionBottom() const
652{
653 LayoutUnit selectionBottom = m_lineBottom;
654
655 if (m_hasAnnotationsAfter)
656 selectionBottom += !renderer().style().isFlippedLinesWritingMode() ? computeUnderAnnotationAdjustment(m_lineBottom) : computeOverAnnotationAdjustment(m_lineBottom);
657
658 if (!renderer().style().isFlippedLinesWritingMode() || !nextRootBox())
659 return selectionBottom;
660
661#if !PLATFORM(IOS_FAMILY)
662 // See rdar://problem/19692206 ... don't want to do this adjustment for iOS where overlap is ok and handled.
663 if (renderer().isRubyBase()) {
664 // The ruby base selection should avoid intruding into the ruby text. This is only the case if there is an actual ruby text below us.
665 RenderRubyBase* base = &downcast<RenderRubyBase>(renderer());
666 RenderRubyRun* run = base->rubyRun();
667 if (run) {
668 RenderRubyText* text = run->rubyText();
669 if (text && text->logicalTop() > base->logicalTop()) {
670 // The ruby text is below the ruby base. Just return now in order to avoid painting on top of the ruby text.
671 return selectionBottom;
672 }
673 }
674 } else if (renderer().isRubyText()) {
675 // The ruby text selection should go all the way to the selection bottom of the containing line.
676 RenderRubyText* text = &downcast<RenderRubyText>(renderer());
677 RenderRubyRun* run = text->rubyRun();
678 if (run && run->inlineBoxWrapper()) {
679 RenderRubyBase* base = run->rubyBase();
680 if (base && text->logicalTop() > base->logicalTop()) {
681 // The ruby text is above the ruby base.
682 const LegacyRootInlineBox& containingLine = run->inlineBoxWrapper()->root();
683 LayoutUnit enclosingSelectionBottom = containingLine.selectionBottom();
684 LayoutUnit deltaBetweenObjects = text->logicalTop() + run->logicalTop();
685 LayoutUnit selectionBottomInRubyTextCoords = enclosingSelectionBottom - deltaBetweenObjects;
686 return std::min(selectionBottom, selectionBottomInRubyTextCoords);
687 }
688 }
689 }
690#endif
691
692 LayoutUnit nextTop = nextRootBox()->selectionTop();
693 if (nextTop > selectionBottom && blockFlow().containsFloats()) {
694 // The next line has actually been moved further over, probably from a large line-height, but possibly because the
695 // line was forced to clear floats. If so, let's check the offsets, and only be willing to use the next
696 // line's top if the offsets are greater on both sides.
697 LayoutUnit nextLeft = blockFlow().logicalLeftOffsetForLine(nextTop, DoNotIndentText);
698 LayoutUnit nextRight = blockFlow().logicalRightOffsetForLine(nextTop, DoNotIndentText);
699 LayoutUnit newLeft = blockFlow().logicalLeftOffsetForLine(selectionBottom, DoNotIndentText);
700 LayoutUnit newRight = blockFlow().logicalRightOffsetForLine(selectionBottom, DoNotIndentText);
701 if (nextLeft > newLeft || nextRight < newRight)
702 return selectionBottom;
703 }
704
705 return nextTop;
706}

773static void setAscentAndDescent(LayoutUnit& ascent, LayoutUnit& descent, LayoutUnit newAscent, LayoutUnit newDescent, bool& ascentDescentSet)
774{
775 if (!ascentDescentSet) {
776 ascentDescentSet = true;
777 ascent = newAscent;
778 descent = newDescent;
779 } else {
780 ascent = std::max(ascent, newAscent);
781 descent = std::max(descent, newDescent);
782 }

785void LegacyRootInlineBox::ascentAndDescentForBox(LegacyInlineBox& box, GlyphOverflowAndFallbackFontsMap& textBoxDataMap, LayoutUnit& ascent, LayoutUnit& descent,
786 bool& affectsAscent, bool& affectsDescent) const
787{
788 bool ascentDescentSet = false;
789
790 // Replaced boxes will return 0 for the line-height if line-box-contain says they are
791 // not to be included.
792 if (box.renderer().isReplaced()) {
793 if (lineStyle().lineBoxContain().contains(LineBoxContain::Replaced)) {
794 ascent = box.baselinePosition(baselineType());
795 descent = roundToInt(box.lineHeight()) - ascent;
796
797 // Replaced elements always affect both the ascent and descent.
798 affectsAscent = true;
799 affectsDescent = true;
800 }
801 return;
802 }
803
804 Vector<const Font*>* usedFonts = nullptr;
805 GlyphOverflow* glyphOverflow = nullptr;
806 if (is<LegacyInlineTextBox>(box)) {
807 GlyphOverflowAndFallbackFontsMap::iterator it = textBoxDataMap.find(&downcast<LegacyInlineTextBox>(box));
808 usedFonts = it == textBoxDataMap.end() ? nullptr : &it->value.first;
809 glyphOverflow = it == textBoxDataMap.end() ? nullptr : &it->value.second;
810 }
811
812 bool includeLeading = includeLeadingForBox(box);
813 bool includeFont = includeFontForBox(box);
814
815 bool setUsedFont = false;
816 bool setUsedFontWithLeading = false;
817
818 const RenderStyle& boxLineStyle = box.lineStyle();
819 if (usedFonts && !usedFonts->isEmpty() && (includeFont || (boxLineStyle.lineHeight().isNegative() && includeLeading))) {
820 usedFonts->append(&boxLineStyle.fontCascade().primaryFont());
821 for (auto& font : *usedFonts) {
822 auto& fontMetrics = font->fontMetrics();
823 LayoutUnit usedFontAscent { fontMetrics.ascent(baselineType()) };
824 LayoutUnit usedFontDescent { fontMetrics.descent(baselineType()) };
825 LayoutUnit halfLeading { (fontMetrics.lineSpacing() - fontMetrics.height()) / 2 };
826 LayoutUnit usedFontAscentAndLeading { usedFontAscent + halfLeading };
827 LayoutUnit usedFontDescentAndLeading { fontMetrics.lineSpacing() - usedFontAscentAndLeading };
828 if (includeFont) {
829 setAscentAndDescent(ascent, descent, usedFontAscent, usedFontDescent, ascentDescentSet);
830 setUsedFont = true;
831 }
832 if (includeLeading) {
833 setAscentAndDescent(ascent, descent, usedFontAscentAndLeading, usedFontDescentAndLeading, ascentDescentSet);
834 setUsedFontWithLeading = true;
835 }
836 if (!affectsAscent)
837 affectsAscent = usedFontAscent - box.logicalTop() > 0;
838 if (!affectsDescent)
839 affectsDescent = usedFontDescent + box.logicalTop() > 0;
840 }
841 }
842
843 // If leading is included for the box, then we compute that box.
844 if (includeLeading && !setUsedFontWithLeading) {
845 LayoutUnit ascentWithLeading { box.baselinePosition(baselineType()) };
846 LayoutUnit descentWithLeading { box.lineHeight() - ascentWithLeading };
847 setAscentAndDescent(ascent, descent, ascentWithLeading, descentWithLeading, ascentDescentSet);
848
849 // Examine the font box for inline flows and text boxes to see if any part of it is above the baseline.
850 // If the top of our font box relative to the root box baseline is above the root box baseline, then
851 // we are contributing to the maxAscent value. Descent is similar. If any part of our font box is below
852 // the root box's baseline, then we contribute to the maxDescent value.
853 affectsAscent = ascentWithLeading - box.logicalTop() > 0;
854 affectsDescent = descentWithLeading + box.logicalTop() > 0;
855 }
856
857 if (includeFontForBox(box) && !setUsedFont) {
858 LayoutUnit fontAscent { boxLineStyle.fontMetrics().ascent(baselineType()) };
859 LayoutUnit fontDescent { boxLineStyle.fontMetrics().descent(baselineType()) };
860 setAscentAndDescent(ascent, descent, fontAscent, fontDescent, ascentDescentSet);
861 affectsAscent = fontAscent - box.logicalTop() > 0;
862 affectsDescent = fontDescent + box.logicalTop() > 0;
863 }
864
865 if (includeGlyphsForBox(box) && glyphOverflow && glyphOverflow->computeBounds) {
866 setAscentAndDescent(ascent, descent, glyphOverflow->top, glyphOverflow->bottom, ascentDescentSet);
867 affectsAscent = glyphOverflow->top - box.logicalTop() > 0;
868 affectsDescent = glyphOverflow->bottom + box.logicalTop() > 0;
869 glyphOverflow->top = std::min(glyphOverflow->top, std::max(0, glyphOverflow->top - boxLineStyle.fontMetrics().ascent(baselineType())));
870 glyphOverflow->bottom = std::min(glyphOverflow->bottom, std::max(0, glyphOverflow->bottom - boxLineStyle.fontMetrics().descent(baselineType())));
871 }
872
873 if (includeInitialLetterForBox(box)) {
874 bool canUseGlyphs = glyphOverflow && glyphOverflow->computeBounds;
875 LayoutUnit letterAscent { baselineType() == AlphabeticBaseline ? boxLineStyle.fontMetrics().capHeight() : (canUseGlyphs ? glyphOverflow->top : boxLineStyle.fontMetrics().ascent(baselineType())) };
876 LayoutUnit letterDescent { canUseGlyphs ? glyphOverflow->bottom : (box.isRootInlineBox() ? 0 : boxLineStyle.fontMetrics().descent(baselineType())) };
877 setAscentAndDescent(ascent, descent, letterAscent, letterDescent, ascentDescentSet);
878 affectsAscent = letterAscent - box.logicalTop() > 0;
879 affectsDescent = letterDescent + box.logicalTop() > 0;
880 if (canUseGlyphs) {
881 glyphOverflow->top = std::min(glyphOverflow->top, std::max(0, glyphOverflow->top - boxLineStyle.fontMetrics().ascent(baselineType())));
882 glyphOverflow->bottom = std::min(glyphOverflow->bottom, std::max(0, glyphOverflow->bottom - boxLineStyle.fontMetrics().descent(baselineType())));
883 }
884 }
885
886 if (includeMarginForBox(box)) {
887 LayoutUnit ascentWithMargin = boxLineStyle.fontMetrics().ascent(baselineType());
888 LayoutUnit descentWithMargin = boxLineStyle.fontMetrics().descent(baselineType());
889 if (box.parent() && !box.renderer().isTextOrLineBreak()) {
890 ascentWithMargin += box.boxModelObject()->borderAndPaddingBefore() + box.boxModelObject()->marginBefore();
891 descentWithMargin += box.boxModelObject()->borderAndPaddingAfter() + box.boxModelObject()->marginAfter();
892 }
893 setAscentAndDescent(ascent, descent, ascentWithMargin, descentWithMargin, ascentDescentSet);
894
895 // Treat like a replaced element, since we're using the margin box.
896 affectsAscent = true;
897 affectsDescent = true;
898 }
899}

901LayoutUnit LegacyRootInlineBox::verticalPositionForBox(LegacyInlineBox* box, VerticalPositionCache& verticalPositionCache)
902{
903 if (box->renderer().isTextOrLineBreak())
904 return LayoutUnit(box->parent()->logicalTop());
905
906 RenderBoxModelObject* renderer = box->boxModelObject();
907 ASSERT(renderer->isInline());
908 if (!renderer->isInline())
909 return 0;
910
911 // This method determines the vertical position for inline elements.
912 bool firstLine = isFirstLine();
913 if (firstLine && !blockFlow().view().usesFirstLineRules())
914 firstLine = false;
915
916 // Check the cache.
917 bool isRenderInline = renderer->isRenderInline();
918 if (isRenderInline && !firstLine) {
919 LayoutUnit cachedPosition;
920 if (verticalPositionCache.get(renderer, baselineType(), cachedPosition))
921 return cachedPosition;
922 }
923
924 LayoutUnit verticalPosition;
925 VerticalAlign verticalAlign = renderer->style().verticalAlign();
926 if (verticalAlign == VerticalAlign::Top || verticalAlign == VerticalAlign::Bottom)
927 return 0;
928
929 RenderElement* parent = renderer->parent();
930 if (parent->isRenderInline() && parent->style().verticalAlign() != VerticalAlign::Top && parent->style().verticalAlign() != VerticalAlign::Bottom)
931 verticalPosition = box->parent()->logicalTop();
932
933 if (verticalAlign != VerticalAlign::Baseline) {
934 const RenderStyle& parentLineStyle = firstLine ? parent->firstLineStyle() : parent->style();
935 const FontCascade& font = parentLineStyle.fontCascade();
936 const FontMetrics& fontMetrics = font.fontMetrics();
937 int fontSize = font.pixelSize();
938
939 LineDirectionMode lineDirection = parent->isHorizontalWritingMode() ? HorizontalLine : VerticalLine;
940
941 if (verticalAlign == VerticalAlign::Sub)
942 verticalPosition += fontSize / 5 + 1;
943 else if (verticalAlign == VerticalAlign::Super)
944 verticalPosition -= fontSize / 3 + 1;
945 else if (verticalAlign == VerticalAlign::TextTop)
946 verticalPosition += renderer->baselinePosition(baselineType(), firstLine, lineDirection) - fontMetrics.ascent(baselineType());
947 else if (verticalAlign == VerticalAlign::Middle)
948 verticalPosition = (verticalPosition - LayoutUnit(fontMetrics.xHeight() / 2) - renderer->lineHeight(firstLine, lineDirection) / 2 + renderer->baselinePosition(baselineType(), firstLine, lineDirection));
949 else if (verticalAlign == VerticalAlign::TextBottom) {
950 verticalPosition += fontMetrics.descent(baselineType());
951 // lineHeight - baselinePosition is always 0 for replaced elements (except inline blocks), so don't bother wasting time in that case.
952 if (!renderer->isReplaced() || renderer->isInlineBlockOrInlineTable())
953 verticalPosition -= (renderer->lineHeight(firstLine, lineDirection) - renderer->baselinePosition(baselineType(), firstLine, lineDirection));
954 } else if (verticalAlign == VerticalAlign::BaselineMiddle)
955 verticalPosition += -renderer->lineHeight(firstLine, lineDirection) / 2 + renderer->baselinePosition(baselineType(), firstLine, lineDirection);
956 else if (verticalAlign == VerticalAlign::Length) {
957 LayoutUnit lineHeight;
958 // Per http://www.w3.org/TR/CSS21/visudet.html#propdef-vertical-align: 'Percentages: refer to the 'line-height' of the element itself'.
959 if (renderer->style().verticalAlignLength().isPercentOrCalculated())
960 lineHeight = renderer->style().computedLineHeight();
961 else
962 lineHeight = renderer->lineHeight(firstLine, lineDirection);
963 verticalPosition -= valueForLength(renderer->style().verticalAlignLength(), lineHeight);
964 }
965 }
966
967 // Store the cached value.
968 if (isRenderInline && !firstLine)
969 verticalPositionCache.set(renderer, baselineType(), verticalPosition);
970
971 return verticalPosition;
972}
