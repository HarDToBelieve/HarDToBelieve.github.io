26#pragma once
27
28#include <wtf/EnumTraits.h>
29
30namespace WebCore {
31
32struct Pagination {
33 enum Mode : uint8_t { Unpaginated, LeftToRightPaginated, RightToLeftPaginated, TopToBottomPaginated, BottomToTopPaginated };
34
35 bool operator==(const Pagination& other) const
36 {
37 return mode == other.mode && behavesLikeColumns == other.behavesLikeColumns && pageLength == other.pageLength && gap == other.gap;
38 }
39
40 bool operator!=(const Pagination& other) const
41 {
42 return !(*this == other);
43 }
44
45 Mode mode { Unpaginated };
46 bool behavesLikeColumns { false };
47 unsigned pageLength { 0 };
48 unsigned gap { 0 };
49};
50
51} // namespace WebCore
52
53namespace WTF {
54
55template<> struct EnumTraits<WebCore::Pagination::Mode> {
56 using values = EnumValues<
57 WebCore::Pagination::Mode,
58 WebCore::Pagination::Mode::Unpaginated,
59 WebCore::Pagination::Mode::LeftToRightPaginated,
60 WebCore::Pagination::Mode::RightToLeftPaginated,
61 WebCore::Pagination::Mode::TopToBottomPaginated,
62 WebCore::Pagination::Mode::BottomToTopPaginated
63 >;
64};
65
66} // namespace WTF
