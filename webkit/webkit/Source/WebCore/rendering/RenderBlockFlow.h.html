116 struct RenderBlockFlowRareData {
117 WTF_MAKE_NONCOPYABLE(RenderBlockFlowRareData); WTF_MAKE_FAST_ALLOCATED;
118 public:
119 RenderBlockFlowRareData(const RenderBlockFlow& block)
120 : m_margins(positiveMarginBeforeDefault(block), negativeMarginBeforeDefault(block), positiveMarginAfterDefault(block), negativeMarginAfterDefault(block))
121 , m_lineBreakToAvoidWidow(-1)
122 , m_discardMarginBefore(false)
123 , m_discardMarginAfter(false)
124 , m_didBreakAtLineToAvoidWidow(false)
125 {
126 }
127
128 static LayoutUnit positiveMarginBeforeDefault(const RenderBlock& block)
129 {
130 return std::max<LayoutUnit>(block.marginBefore(), 0);
131 }
132 static LayoutUnit negativeMarginBeforeDefault(const RenderBlock& block)
133 {
134 return std::max<LayoutUnit>(-block.marginBefore(), 0);
135 }
136 static LayoutUnit positiveMarginAfterDefault(const RenderBlock& block)
137 {
138 return std::max<LayoutUnit>(block.marginAfter(), 0);
139 }
140 static LayoutUnit negativeMarginAfterDefault(const RenderBlock& block)
141 {
142 return std::max<LayoutUnit>(-block.marginAfter(), 0);
143 }

150
151 bool m_discardMarginBefore : 1;
152 bool m_discardMarginAfter : 1;
153 bool m_didBreakAtLineToAvoidWidow : 1;
154 };

156 class MarginInfo {
157 // Collapsing flags for whether we can collapse our margins with our children's margins.
158 bool m_canCollapseWithChildren : 1;
159 bool m_canCollapseMarginBeforeWithChildren : 1;
160 bool m_canCollapseMarginAfterWithChildren : 1;
161
162 // Whether or not we are a quirky container, i.e., do we collapse away top and bottom
163 // margins in our container. Table cells and the body are the common examples. We
164 // also have a custom style property for Safari RSS to deal with TypePad blog articles.
165 bool m_quirkContainer : 1;
166
167 // This flag tracks whether we are still looking at child margins that can all collapse together at the beginning of a block.
168 // They may or may not collapse with the top margin of the block (|m_canCollapseTopWithChildren| tells us that), but they will
169 // always be collapsing with one another. This variable can remain set to true through multiple iterations
170 // as long as we keep encountering self-collapsing blocks.
171 bool m_atBeforeSideOfBlock : 1;
172
173 // This flag is set when we know we're examining bottom margins and we know we're at the bottom of the block.
174 bool m_atAfterSideOfBlock : 1;
175
176 // These variables are used to detect quirky margins that we need to collapse away (in table cells
177 // and in the body element).
178 bool m_hasMarginBeforeQuirk : 1;
179 bool m_hasMarginAfterQuirk : 1;
180 bool m_determinedMarginBeforeQuirk : 1;
181
182 bool m_discardMargin : 1;
183
184 // These flags track the previous maximal positive and negative margins.
185 LayoutUnit m_positiveMargin;
186 LayoutUnit m_negativeMargin;
187
188 public:
189 MarginInfo(const RenderBlockFlow&, LayoutUnit beforeBorderPadding, LayoutUnit afterBorderPadding);
190
191 void setAtBeforeSideOfBlock(bool b) { m_atBeforeSideOfBlock = b; }
192 void setAtAfterSideOfBlock(bool b) { m_atAfterSideOfBlock = b; }
193 void clearMargin()
194 {
195 m_positiveMargin = 0;
196 m_negativeMargin = 0;
197 }
198 void setHasMarginBeforeQuirk(bool b) { m_hasMarginBeforeQuirk = b; }
199 void setHasMarginAfterQuirk(bool b) { m_hasMarginAfterQuirk = b; }
200 void setDeterminedMarginBeforeQuirk(bool b) { m_determinedMarginBeforeQuirk = b; }
201 void setPositiveMargin(LayoutUnit p) { ASSERT(!m_discardMargin); m_positiveMargin = p; }
202 void setNegativeMargin(LayoutUnit n) { ASSERT(!m_discardMargin); m_negativeMargin = n; }
203 void setPositiveMarginIfLarger(LayoutUnit p)
204 {
205 ASSERT(!m_discardMargin);
206 if (p > m_positiveMargin)
207 m_positiveMargin = p;
208 }
209 void setNegativeMarginIfLarger(LayoutUnit n)
210 {
211 ASSERT(!m_discardMargin);
212 if (n > m_negativeMargin)
213 m_negativeMargin = n;
214 }
215
216 void setMargin(LayoutUnit p, LayoutUnit n) { ASSERT(!m_discardMargin); m_positiveMargin = p; m_negativeMargin = n; }
217 void setCanCollapseMarginAfterWithChildren(bool collapse) { m_canCollapseMarginAfterWithChildren = collapse; }
218 void setDiscardMargin(bool value) { m_discardMargin = value; }
219
220 bool atBeforeSideOfBlock() const { return m_atBeforeSideOfBlock; }
221 bool canCollapseWithMarginBefore() const { return m_atBeforeSideOfBlock && m_canCollapseMarginBeforeWithChildren; }
222 bool canCollapseWithMarginAfter() const { return m_atAfterSideOfBlock && m_canCollapseMarginAfterWithChildren; }
223 bool canCollapseMarginBeforeWithChildren() const { return m_canCollapseMarginBeforeWithChildren; }
224 bool canCollapseMarginAfterWithChildren() const { return m_canCollapseMarginAfterWithChildren; }
225 bool quirkContainer() const { return m_quirkContainer; }
226 bool determinedMarginBeforeQuirk() const { return m_determinedMarginBeforeQuirk; }
227 bool hasMarginBeforeQuirk() const { return m_hasMarginBeforeQuirk; }
228 bool hasMarginAfterQuirk() const { return m_hasMarginAfterQuirk; }
229 LayoutUnit positiveMargin() const { return m_positiveMargin; }
230 LayoutUnit negativeMargin() const { return m_negativeMargin; }
231 bool discardMargin() const { return m_discardMargin; }
232 LayoutUnit margin() const { return m_positiveMargin - m_negativeMargin; }
233 };
