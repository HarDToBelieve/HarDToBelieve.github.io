26#pragma once
27
28#include "RenderOverflow.h"
29
30namespace WebCore {
31
32class RenderBoxFragmentInfo {
33 WTF_MAKE_FAST_ALLOCATED;
34 WTF_MAKE_NONCOPYABLE(RenderBoxFragmentInfo);
35public:
36 RenderBoxFragmentInfo(LayoutUnit logicalLeft, LayoutUnit logicalWidth, bool isShifted)
37 : m_logicalLeft(logicalLeft)
38 , m_logicalWidth(logicalWidth)
39 , m_isShifted(isShifted)
40 { }
41
42 LayoutUnit logicalLeft() const { return m_logicalLeft; }
43 LayoutUnit logicalWidth() const { return m_logicalWidth; }
44
45 void shiftLogicalLeft(LayoutUnit delta) { m_logicalLeft += delta; m_isShifted = true; }
46
47 bool isShifted() const { return m_isShifted; }
48
49 void createOverflow(const LayoutRect& layoutOverflow, const LayoutRect& visualOverflow) { m_overflow = adoptRef(new RenderOverflow(layoutOverflow, visualOverflow)); }
50 RenderOverflow* overflow() const { return m_overflow.get(); }
51 void clearOverflow()
52 {
53 if (m_overflow)
54 m_overflow = nullptr;
55 }
56
57private:
58 LayoutUnit m_logicalLeft;
59 LayoutUnit m_logicalWidth;
60 bool m_isShifted;
61 RefPtr<RenderOverflow> m_overflow;
62};
63
64} // namespace WebCore
