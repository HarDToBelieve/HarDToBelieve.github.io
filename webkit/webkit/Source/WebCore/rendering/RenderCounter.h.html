62inline void RenderCounter::rendererStyleChanged(RenderElement& renderer, const RenderStyle* oldStyle, const RenderStyle& newStyle)
63{
64 if ((!oldStyle || !oldStyle->counterDirectives()) && !newStyle.counterDirectives())
65 return;
66
67 rendererStyleChangedSlowCase(renderer, oldStyle, newStyle);
68}
