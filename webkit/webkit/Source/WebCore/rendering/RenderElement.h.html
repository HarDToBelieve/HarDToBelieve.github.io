442inline bool RenderElement::canContainFixedPositionObjects() const
443{
444 return isRenderView()
445 || (isRenderBlock() && hasTransform())
446 // FIXME: will-change should create containing blocks on inline boxes (bug 225035)
447 || (isRenderBlock() && style().willChange() && style().willChange()->createsContainingBlockForOutOfFlowPositioned())
448 || isSVGForeignObject()
449 || shouldApplyLayoutContainment(*this);
450}
451
452inline bool RenderElement::canContainAbsolutelyPositionedObjects() const
453{
454 return style().position() != PositionType::Static
455 || (isRenderBlock() && hasTransformRelatedProperty())
456 // FIXME: will-change should create containing blocks on inline boxes (bug 225035)
457 || (isRenderBlock() && style().willChange() && style().willChange()->createsContainingBlockForAbsolutelyPositioned())
458 || isSVGForeignObject()
459 || shouldApplyLayoutContainment(*this)
460 || isRenderView();
461}

527inline RenderObject* RenderElement::firstInFlowChild() const
528{
529 if (auto* firstChild = this->firstChild()) {
530 if (firstChild->isInFlow())
531 return firstChild;
532 return firstChild->nextInFlowSibling();
533 }
534 return nullptr;
535}
536
537inline RenderObject* RenderElement::lastInFlowChild() const
538{
539 if (auto* lastChild = this->lastChild()) {
540 if (lastChild->isInFlow())
541 return lastChild;
542 return lastChild->previousInFlowSibling();
543 }
544 return nullptr;
545}
