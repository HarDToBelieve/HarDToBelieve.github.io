108bool RenderEmbeddedObject::allowsAcceleratedCompositing() const
109{
110#if PLATFORM(IOS_FAMILY)
111 // The timing of layer creation is different on the phone, since the plugin can only be manipulated from the main thread.
112 return is<PluginViewBase>(widget()) && downcast<PluginViewBase>(*widget()).willProvidePluginLayer();
113#else
114 return is<PluginViewBase>(widget()) && downcast<PluginViewBase>(*widget()).platformLayer();
115#endif
116}

119static String unavailablePluginReplacementText(RenderEmbeddedObject::PluginUnavailabilityReason pluginUnavailabilityReason)
120{
121 switch (pluginUnavailabilityReason) {
122 case RenderEmbeddedObject::PluginMissing:
123 return missingPluginText();
124 case RenderEmbeddedObject::PluginCrashed:
125 return crashedPluginText();
126 case RenderEmbeddedObject::PluginBlockedByContentSecurityPolicy:
127 return blockedPluginByContentSecurityPolicyText();
128 case RenderEmbeddedObject::InsecurePluginVersion:
129 return insecurePluginVersionText();
130 case RenderEmbeddedObject::UnsupportedPlugin:
131 return unsupportedPluginText();
132 case RenderEmbeddedObject::PluginTooSmall:
133 return pluginTooSmallText();
134 }
135
136 ASSERT_NOT_REACHED();
137 return String();
138}

155void RenderEmbeddedObject::setPluginUnavailabilityReasonWithDescription(PluginUnavailabilityReason pluginUnavailabilityReason, const String& description)
156{
157#if PLATFORM(IOS_FAMILY)

180void RenderEmbeddedObject::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
181{
182 // The relevant repainted object heuristic is not tuned for plugin documents.
183 bool countsTowardsRelevantObjects = !document().isPluginDocument() && paintInfo.phase == PaintPhase::Foreground;
184
185 if (isPluginUnavailable()) {
186 if (countsTowardsRelevantObjects)
187 page().addRelevantUnpaintedObject(this, visualOverflowRect());
188 RenderReplaced::paint(paintInfo, paintOffset);
189 return;
190 }
191
192 if (countsTowardsRelevantObjects)
193 page().addRelevantRepaintedObject(this, visualOverflowRect());
194
195 RenderWidget::paint(paintInfo, paintOffset);
196}

198static void drawReplacementArrow(GraphicsContext& context, const FloatRect& insideRect)
199{
200 GraphicsContextStateSaver stateSaver(context);
201
202 FloatRect rect(insideRect);
203 rect.inflate(-replacementArrowPadding);
204
205 FloatPoint center(rect.center());
206 FloatPoint arrowTip(rect.maxX(), center.y());
207
208 context.setStrokeThickness(2);
209 context.setLineCap(RoundCap);
210 context.setLineJoin(RoundJoin);
211
212 Path path;
213 path.moveTo(FloatPoint(rect.x(), center.y()));
214 path.addLineTo(arrowTip);
215 path.addLineTo(FloatPoint(center.x(), rect.y()));
216 path.moveTo(arrowTip);
217 path.addLineTo(FloatPoint(center.x(), rect.maxY()));
218 context.strokePath(path);
219}

221void RenderEmbeddedObject::paintReplaced(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
222{
223 if (!showsUnavailablePluginIndicator())
224 return;
225
226 if (paintInfo.phase == PaintPhase::Selection)
227 return;
228
229 GraphicsContext& context = paintInfo.context();
230 if (context.paintingDisabled())
231 return;
232
233 FloatRect contentRect;
234 FloatRect indicatorRect;
235 FloatRect replacementTextRect;
236 FloatRect arrowRect;
237 FontCascade font;
238 TextRun run(emptyString());
239 float textWidth;
240 getReplacementTextGeometry(paintOffset, contentRect, indicatorRect, replacementTextRect, arrowRect, font, run, textWidth);
241
242 Path background;
243 background.addRoundedRect(indicatorRect, FloatSize(replacementTextRoundedRectRadius, replacementTextRoundedRectRadius));
244
245 GraphicsContextStateSaver stateSaver(context);
246 context.clip(contentRect);
247 context.setFillColor(m_unavailablePluginIndicatorIsPressed ? replacementTextRoundedRectPressedColor : replacementTextRoundedRectColor);
248 context.fillPath(background);
249
250 Path strokePath;
251 FloatRect strokeRect(indicatorRect);
252 strokeRect.inflate(1);
253 strokePath.addRoundedRect(strokeRect, FloatSize(replacementTextRoundedRectRadius + 1, replacementTextRoundedRectRadius + 1));
254
255 context.setStrokeColor(unavailablePluginBorderColor);
256 context.setStrokeThickness(2);
257 context.strokePath(strokePath);
258
259 const FontMetrics& fontMetrics = font.fontMetrics();
260 float labelX = roundf(replacementTextRect.location().x() + replacementTextRoundedRectLeftTextMargin);
261 float labelY = roundf(replacementTextRect.location().y() + (replacementTextRect.size().height() - fontMetrics.height()) / 2 + fontMetrics.ascent() + replacementTextRoundedRectTopTextMargin);
262 context.setFillColor(replacementTextColor);
263 context.drawBidiText(font, run, FloatPoint(labelX, labelY));
264
265 if (shouldUnavailablePluginMessageBeButton(page(), m_pluginUnavailabilityReason)) {
266 arrowRect.inflate(-replacementArrowCirclePadding);
267
268 context.beginTransparencyLayer(1.0);
269 context.setFillColor(replacementTextColor);
270 context.fillEllipse(arrowRect);
271
272 context.setCompositeOperation(CompositeOperator::Clear);
273 drawReplacementArrow(context, arrowRect);
274 context.endTransparencyLayer();
275 }
276}

300void RenderEmbeddedObject::getReplacementTextGeometry(const LayoutPoint& accumulatedOffset, FloatRect& contentRect, FloatRect& indicatorRect, FloatRect& replacementTextRect, FloatRect& arrowRect, FontCascade& font, TextRun& run, float& textWidth) const
301{
302 bool includesArrow = shouldUnavailablePluginMessageBeButton(page(), m_pluginUnavailabilityReason);
303
304 contentRect = contentBoxRect();
305 contentRect.moveBy(roundedIntPoint(accumulatedOffset));
306
307 FontCascadeDescription fontDescription;
308 RenderTheme::singleton().systemFont(CSSValueWebkitSmallControl, fontDescription);
309 fontDescription.setWeight(boldWeightValue());
310 fontDescription.setRenderingMode(settings().fontRenderingMode());
311 fontDescription.setComputedSize(12);
312 font = FontCascade(WTFMove(fontDescription), 0, 0);
313 font.update(0);
314
315 run = TextRun(m_unavailablePluginReplacementText);
316 textWidth = font.width(run);
317
318 replacementTextRect.setSize(FloatSize(textWidth + replacementTextRoundedRectLeftTextMargin + (includesArrow ? replacementTextRoundedRectRightTextMarginWithArrow : replacementTextRoundedRectRightTextMargin), replacementTextRoundedRectHeight));
319 replacementTextRect.setLocation(contentRect.location() + (contentRect.size() / 2 - replacementTextRect.size() / 2));
320
321 indicatorRect = replacementTextRect;
322
323 // Expand the background rect to include the arrow, if it will be used.
324 if (includesArrow) {
325 arrowRect = indicatorRect;
326 arrowRect.setX(ceilf(arrowRect.maxX() + replacementArrowLeftMargin));
327 arrowRect.setWidth(arrowRect.height());
328 indicatorRect.unite(arrowRect);
329 }
330}

358bool RenderEmbeddedObject::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
359{
360 if (!RenderWidget::nodeAtPoint(request, result, locationInContainer, accumulatedOffset, hitTestAction))
361 return false;
362
363 if (!is<PluginViewBase>(widget()))
364 return true;
365
366 PluginViewBase& view = downcast<PluginViewBase>(*widget());
367 IntPoint roundedPoint = locationInContainer.roundedPoint();
368
369 if (Scrollbar* horizontalScrollbar = view.horizontalScrollbar()) {
370 if (horizontalScrollbar->shouldParticipateInHitTesting() && horizontalScrollbar->frameRect().contains(roundedPoint)) {
371 result.setScrollbar(horizontalScrollbar);
372 return true;
373 }
374 }
375
376 if (Scrollbar* verticalScrollbar = view.verticalScrollbar()) {
377 if (verticalScrollbar->shouldParticipateInHitTesting() && verticalScrollbar->frameRect().contains(roundedPoint)) {
378 result.setScrollbar(verticalScrollbar);
379 return true;
380 }
381 }
382
383 return true;
384}

410void RenderEmbeddedObject::handleUnavailablePluginIndicatorEvent(Event* event)
411{
412 if (!shouldUnavailablePluginMessageBeButton(page(), m_pluginUnavailabilityReason))
413 return;
414
415 if (!is<MouseEvent>(*event))
416 return;
417
418 MouseEvent& mouseEvent = downcast<MouseEvent>(*event);
419 HTMLPlugInElement& element = downcast<HTMLPlugInElement>(frameOwnerElement());
420 if (mouseEvent.type() == eventNames().mousedownEvent && mouseEvent.button() == LeftButton) {
421 m_mouseDownWasInUnavailablePluginIndicator = isInUnavailablePluginIndicator(mouseEvent);
422 if (m_mouseDownWasInUnavailablePluginIndicator) {
423 frame().eventHandler().setCapturingMouseEventsElement(&element);
424 element.setIsCapturingMouseEvents(true);
425 setUnavailablePluginIndicatorIsPressed(true);
426 }
427 mouseEvent.setDefaultHandled();
428 }
429 if (mouseEvent.type() == eventNames().mouseupEvent && mouseEvent.button() == LeftButton) {
430 if (m_unavailablePluginIndicatorIsPressed) {
431 frame().eventHandler().setCapturingMouseEventsElement(nullptr);
432 element.setIsCapturingMouseEvents(false);
433 setUnavailablePluginIndicatorIsPressed(false);
434 }
435 if (m_mouseDownWasInUnavailablePluginIndicator && isInUnavailablePluginIndicator(mouseEvent)) {
436 page().chrome().client().unavailablePluginButtonClicked(element, m_pluginUnavailabilityReason);
437 }
438 m_mouseDownWasInUnavailablePluginIndicator = false;
439 event->setDefaultHandled();
440 }
441 if (mouseEvent.type() == eventNames().mousemoveEvent) {
442 setUnavailablePluginIndicatorIsPressed(m_mouseDownWasInUnavailablePluginIndicator && isInUnavailablePluginIndicator(mouseEvent));
443 mouseEvent.setDefaultHandled();
444 }
445}

447CursorDirective RenderEmbeddedObject::getCursor(const LayoutPoint& point, Cursor& cursor) const
448{
449 if (showsUnavailablePluginIndicator() && shouldUnavailablePluginMessageBeButton(page(), m_pluginUnavailabilityReason) && isInUnavailablePluginIndicator(point)) {
450 cursor = handCursor();
451 return SetCursor;
452 }
453 if (widget() && widget()->isPluginViewBase()) {
454 // A plug-in is responsible for setting the cursor when the pointer is over it.
455 return DoNotSetCursor;
456 }
457 return RenderWidget::getCursor(point, cursor);
458}
