23#pragma once
24
25#include "RenderWidget.h"
26
27namespace WebCore {
28
29class MouseEvent;
30class TextRun;
31
32// Renderer for embeds and objects, often, but not always, rendered via plug-ins.
33// For example, <embed src="foo.html"> does not invoke a plug-in.
34class RenderEmbeddedObject final : public RenderWidget {
35 WTF_MAKE_ISO_ALLOCATED(RenderEmbeddedObject);
36public:
37 RenderEmbeddedObject(HTMLFrameOwnerElement&, RenderStyle&&);
38 virtual ~RenderEmbeddedObject();
39
40 enum PluginUnavailabilityReason {
41 PluginMissing,
42 PluginCrashed,
43 PluginBlockedByContentSecurityPolicy,
44 InsecurePluginVersion,
45 UnsupportedPlugin,
46 PluginTooSmall
47 };
48 PluginUnavailabilityReason pluginUnavailabilityReason() const { return m_pluginUnavailabilityReason; };
49 WEBCORE_EXPORT void setPluginUnavailabilityReason(PluginUnavailabilityReason);
50 WEBCORE_EXPORT void setPluginUnavailabilityReasonWithDescription(PluginUnavailabilityReason, const String& description);
51
52 bool isPluginUnavailable() const { return m_isPluginUnavailable; }
53
54 WEBCORE_EXPORT void setUnavailablePluginIndicatorIsHidden(bool);
55
56 void handleUnavailablePluginIndicatorEvent(Event*);
57
58 bool allowsAcceleratedCompositing() const;
59
60 LayoutRect unavailablePluginIndicatorBounds(const LayoutPoint& accumulatedOffset) const;
61
62 const String& pluginReplacementTextIfUnavailable() const { return m_unavailablePluginReplacementText; }
63
64private:
65 void paintReplaced(PaintInfo&, const LayoutPoint&) final;
66 void paint(PaintInfo&, const LayoutPoint&) final;
67
68 CursorDirective getCursor(const LayoutPoint&, Cursor&) const final;
69
70 void layout() final;
71 void willBeDestroyed() final;
72
73 const char* renderName() const final { return "RenderEmbeddedObject"; }
74 bool isEmbeddedObject() const final { return true; }
75
76 bool showsUnavailablePluginIndicator() const { return isPluginUnavailable() && m_isUnavailablePluginIndicatorState != UnavailablePluginIndicatorState::Hidden; }
77
78 bool requiresLayer() const final;
79
80 bool nodeAtPoint(const HitTestRequest&, HitTestResult&, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction) final;
81
82 bool scroll(ScrollDirection, ScrollGranularity, float multiplier = 1, Element** stopElement = nullptr, RenderBox* startBox = nullptr, const IntPoint& wheelEventAbsolutePoint = IntPoint()) final;
83 bool logicalScroll(ScrollLogicalDirection, ScrollGranularity, float multiplier, Element** stopElement) final;
84
85 void setUnavailablePluginIndicatorIsPressed(bool);
86 bool isInUnavailablePluginIndicator(const MouseEvent&) const;
87 bool isInUnavailablePluginIndicator(const FloatPoint&) const;
88 void getReplacementTextGeometry(const LayoutPoint& accumulatedOffset, FloatRect& contentRect, FloatRect& indicatorRect, FloatRect& replacementTextRect, FloatRect& arrowRect, FontCascade&, TextRun&, float& textWidth) const;
89 LayoutRect getReplacementTextGeometry(const LayoutPoint& accumulatedOffset) const;
90
91 bool m_isPluginUnavailable;
92 enum class UnavailablePluginIndicatorState { Uninitialized, Hidden, Visible };
93 UnavailablePluginIndicatorState m_isUnavailablePluginIndicatorState { UnavailablePluginIndicatorState::Uninitialized };
94 PluginUnavailabilityReason m_pluginUnavailabilityReason;
95 String m_unavailablePluginReplacementText;
96 bool m_unavailablePluginIndicatorIsPressed;
97 bool m_mouseDownWasInUnavailablePluginIndicator;
98 String m_unavailabilityDescription;
99};
100
101} // namespace WebCore
102
103SPECIALIZE_TYPE_TRAITS_RENDER_OBJECT(RenderEmbeddedObject, isEmbeddedObject())
