31namespace WebCore {
32
33WTF_MAKE_ISO_ALLOCATED_IMPL(RenderFrame);
34
35RenderFrame::RenderFrame(HTMLFrameElement& frame, RenderStyle&& style)
36 : RenderFrameBase(frame, WTFMove(style))
37{
38}
39
40HTMLFrameElement& RenderFrame::frameElement() const
41{
42 return downcast<HTMLFrameElement>(RenderFrameBase::frameOwnerElement());
43}
44
45FrameEdgeInfo RenderFrame::edgeInfo() const
46{
47 return FrameEdgeInfo(frameElement().noResize(), frameElement().hasFrameBorder());
48}
49
50void RenderFrame::updateFromElement()
51{
52 if (is<RenderFrameSet>(parent()))
53 downcast<RenderFrameSet>(*parent()).notifyFrameEdgeInfoChanged();
54}
