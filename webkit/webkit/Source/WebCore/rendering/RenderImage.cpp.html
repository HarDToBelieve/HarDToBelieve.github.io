78// FIXME: This doesn't behave correctly for floating or positioned images, but WebCore doesn't handle those well
79// during selection creation yet anyway.
80// FIXME: We can't tell whether or not we contain the start or end of the selected Range using only the offsets
81// of the start and end, we need to know the whole Position.
82void RenderImage::collectSelectionGeometries(Vector<SelectionGeometry>& geometries, unsigned, unsigned)
83{
84 RenderBlock* containingBlock = this->containingBlock();
85
86 IntRect imageRect;
87 // FIXME: It doesn't make sense to package line bounds into SelectionGeometry. We should find
88 // the right and left extent of the selection once for the entire selected Range, perhaps
89 // using the Range's common ancestor.
90 IntRect lineExtentRect;
91 bool isFirstOnLine = false;
92 bool isLastOnLine = false;
93
94 auto run = LayoutIntegration::runFor(*this);
95 if (!run) {
96 // This is a block image.
97 imageRect = IntRect(0, 0, width(), height());
98 isFirstOnLine = true;
99 isLastOnLine = true;
100 lineExtentRect = imageRect;
101 if (containingBlock->isHorizontalWritingMode()) {
102 lineExtentRect.setX(containingBlock->x());
103 lineExtentRect.setWidth(containingBlock->width());
104 } else {
105 lineExtentRect.setY(containingBlock->y());
106 lineExtentRect.setHeight(containingBlock->height());
107 }
108 } else {
109 auto line = run.line();
110 LayoutUnit selectionTop = !containingBlock->style().isFlippedBlocksWritingMode() ? line->selectionTop() - logicalTop() : logicalBottom() - line->selectionBottom();
111 imageRect = IntRect(0, selectionTop, logicalWidth(), line->selectionHeight());
112 isFirstOnLine = !run.previousOnLine();
113 isLastOnLine = !run.nextOnLine();
114 LogicalSelectionOffsetCaches cache(*containingBlock);
115 LayoutUnit leftOffset = containingBlock->logicalLeftSelectionOffset(*containingBlock, LayoutUnit(run->logicalTop()), cache);
116 LayoutUnit rightOffset = containingBlock->logicalRightSelectionOffset(*containingBlock, LayoutUnit(run->logicalTop()), cache);
117 lineExtentRect = IntRect(leftOffset - logicalLeft(), imageRect.y(), rightOffset - leftOffset, imageRect.height());
118 if (!run->isHorizontal()) {
119 imageRect = imageRect.transposedRect();
120 lineExtentRect = lineExtentRect.transposedRect();
121 }
122 }
123
124 bool isFixed = false;
125 auto absoluteQuad = localToAbsoluteQuad(FloatRect(imageRect), UseTransforms, &isFixed);
126 auto lineExtentBounds = localToAbsoluteQuad(FloatRect(lineExtentRect)).enclosingBoundingBox();
127 if (!containingBlock->isHorizontalWritingMode())
128 lineExtentBounds = lineExtentBounds.transposedRect();
129
130 // FIXME: We should consider either making SelectionGeometry a struct or better organize its optional fields into
131 // an auxiliary struct to simplify its initialization.
132 geometries.append(SelectionGeometry(absoluteQuad, SelectionRenderingBehavior::CoalesceBoundingRects, containingBlock->style().direction(), lineExtentBounds.x(), lineExtentBounds.maxX(), lineExtentBounds.maxY(), 0, false /* line break */, isFirstOnLine, isLastOnLine, false /* contains start */, false /* contains end */, containingBlock->style().isHorizontalWritingMode(), isFixed, false /* ruby text */, view().pageNumberForBlockProgressionOffset(absoluteQuad.enclosingBoundingBox().x())));
133}

164// If we'll be displaying either alt text or an image, add some padding.
165static const unsigned short paddingWidth = 4;
166static const unsigned short paddingHeight = 4;
167
168// Alt text is restricted to this maximum size, in pixels. These are
169// signed integers because they are compared with other signed values.
170static const float maxAltTextWidth = 1024;
171static const int maxAltTextHeight = 256;
172
173IntSize RenderImage::imageSizeForError(CachedImage* newImage) const
174{
175 ASSERT_ARG(newImage, newImage);
176 ASSERT_ARG(newImage, newImage->imageForRenderer(this));
177
178 FloatSize imageSize;
179 if (newImage->willPaintBrokenImage()) {
180 std::pair<Image*, float> brokenImageAndImageScaleFactor = newImage->brokenImage(document().deviceScaleFactor());
181 imageSize = brokenImageAndImageScaleFactor.first->size();
182 imageSize.scale(1 / brokenImageAndImageScaleFactor.second);
183 } else
184 imageSize = newImage->imageForRenderer(this)->size();
185
186 // imageSize() returns 0 for the error image. We need the true size of the
187 // error image, so we have to get it by grabbing image() directly.
188 return IntSize(paddingWidth + imageSize.width() * style().effectiveZoom(), paddingHeight + imageSize.height() * style().effectiveZoom());
189}
190
191// Sets the image height and width to fit the alt text. Returns true if the
192// image size changed.
193ImageSizeChangeType RenderImage::setImageSizeForAltText(CachedImage* newImage /* = 0 */)
194{
195 IntSize imageSize;
196 if (newImage && newImage->imageForRenderer(this))
197 imageSize = imageSizeForError(newImage);
198 else if (!m_altText.isEmpty() || newImage) {
199 // If we'll be displaying either text or an image, add a little padding.
200 imageSize = IntSize(paddingWidth, paddingHeight);
201 }
202
203 // we have an alt and the user meant it (its not a text we invented)
204 if (!m_altText.isEmpty()) {
205 const FontCascade& font = style().fontCascade();
206 IntSize paddedTextSize(paddingWidth + std::min(ceilf(font.width(RenderBlock::constructTextRun(m_altText, style()))), maxAltTextWidth), paddingHeight + std::min(font.fontMetrics().height(), maxAltTextHeight));
207 imageSize = imageSize.expandedTo(paddedTextSize);
208 }
209
210 if (imageSize == intrinsicSize())
211 return ImageSizeChangeNone;
212
213 setIntrinsicSize(imageSize);
214 return ImageSizeChangeForAltText;
215}

224void RenderImage::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
225{
226 RenderReplaced::styleDidChange(diff, oldStyle);
227 if (m_needsToSetSizeForAltText) {
228 if (!m_altText.isEmpty() && setImageSizeForAltText(cachedImage()))
229 repaintOrMarkForLayout(ImageSizeChangeForAltText);
230 m_needsToSetSizeForAltText = false;
231 }
232 if (diff == StyleDifference::Layout && oldStyle && oldStyle->imageOrientation() != style().imageOrientation())
233 return repaintOrMarkForLayout(ImageSizeChangeNone);
234
235#if ENABLE(CSS_IMAGE_RESOLUTION)
236 if (diff == StyleDifference::Layout && oldStyle
237 && (oldStyle->imageResolution() != style().imageResolution()
238 || oldStyle->imageResolutionSnap() != style().imageResolutionSnap()
239 || oldStyle->imageResolutionSource() != style().imageResolutionSource()))
240 repaintOrMarkForLayout(ImageSizeChangeNone);
241#endif
242}

244bool RenderImage::shouldCollapseToEmpty() const
245{
246 auto imageRepresentsNothing = [&] {
247 if (!element()->hasAttribute(HTMLNames::altAttr))
248 return false;
249 return imageResource().errorOccurred() && m_altText.isEmpty();
250 };
251 if (!element()) {
252 // Images with no associated elements do not fall under the category of unwanted content.
253 return false;
254 }
255 if (!isInline())
256 return false;
257 if (!imageRepresentsNothing())
258 return false;
259 return document().inNoQuirksMode() || (style().logicalWidth().isAuto() && style().logicalHeight().isAuto());
260}

276void RenderImage::imageChanged(WrappedImagePtr newImage, const IntRect* rect)
277{
278 if (renderTreeBeingDestroyed())
279 return;
280
281 if (hasVisibleBoxDecorations() || hasMask() || hasShapeOutside())
282 RenderReplaced::imageChanged(newImage, rect);
283
284 if (shouldCollapseToEmpty()) {
285 // Image might need resizing when we are at the final state.
286 setNeedsLayout();
287 }
288
289 if (newImage != imageResource().imagePtr() || !newImage)
290 return;
291
292 // At a zoom level of 1 the image is guaranteed to have an integer size.
293 incrementVisuallyNonEmptyPixelCountIfNeeded(flooredIntSize(imageResource().imageSize(1.0f)));
294
295 ImageSizeChangeType imageSizeChange = ImageSizeChangeNone;
296
297 // Set image dimensions, taking into account the size of the alt text.
298 if (imageResource().errorOccurred()) {
299 if (!m_altText.isEmpty() && document().hasPendingStyleRecalc()) {
300 ASSERT(element());
301 if (element()) {
302 m_needsToSetSizeForAltText = true;
303 element()->invalidateStyle();
304 }
305 return;
306 }
307 imageSizeChange = setImageSizeForAltText(cachedImage());
308 }
309 repaintOrMarkForLayout(imageSizeChange, rect);
310 if (AXObjectCache* cache = document().existingAXObjectCache())
311 cache->deferRecomputeIsIgnoredIfNeeded(element());
312}

321void RenderImage::updateInnerContentRect()
322{
323 // Propagate container size to image resource.

327 if (HTMLImageElement* imageElement = is<HTMLImageElement>(element()) ? downcast<HTMLImageElement>(element()) : nullptr)

333void RenderImage::repaintOrMarkForLayout(ImageSizeChangeType imageSizeChange, const IntRect* rect)
334{
335#if ENABLE(CSS_IMAGE_RESOLUTION)
336 double scale = style().imageResolution();
337 if (style().imageResolutionSnap() == ImageResolutionSnap::Pixels)
338 scale = roundForImpreciseConversion<int>(scale);
339 if (scale <= 0)
340 scale = 1;
341 LayoutSize newIntrinsicSize = imageResource().intrinsicSize(style().effectiveZoom() / scale);
342#else
343 LayoutSize newIntrinsicSize = imageResource().intrinsicSize(style().effectiveZoom());
344#endif
345 LayoutSize oldIntrinsicSize = intrinsicSize();
346
347 updateIntrinsicSizeIfNeeded(newIntrinsicSize);
348
349 // In the case of generated image content using :before/:after/content, we might not be
350 // in the render tree yet. In that case, we just need to update our intrinsic size.
351 // layout() will be called after we are inserted in the tree which will take care of
352 // what we are doing here.
353 if (!containingBlock())
354 return;
355
356 bool imageSourceHasChangedSize = oldIntrinsicSize != newIntrinsicSize || imageSizeChange != ImageSizeChangeNone;
357
358 if (imageSourceHasChangedSize && setNeedsLayoutIfNeededAfterIntrinsicSizeChange())
359 return;
360
361 if (everHadLayout() && !selfNeedsLayout()) {
362 // The inner content rectangle is calculated during layout, but may need an update now
363 // (unless the box has already been scheduled for layout). In order to calculate it, we
364 // may need values from the containing block, though, so make sure that we're not too
365 // early. It may be that layout hasn't even taken place once yet.
366
367 // FIXME: we should not have to trigger another call to setContainerContextForRenderer()
368 // from here, since it's already being done during layout.
369 updateInnerContentRect();
370 }
371
372 LayoutRect repaintRect = contentBoxRect();
373 if (rect) {
374 // The image changed rect is in source image coordinates (pre-zooming),
375 // so map from the bounds of the image to the contentsBox.
376 repaintRect.intersect(enclosingIntRect(mapRect(*rect, FloatRect(FloatPoint(), imageResource().imageSize(1.0f)), repaintRect)));
377 }
378
379 repaintRectangle(repaintRect);
380
381 // Tell any potential compositing layers that the image needs updating.
382 contentChanged(ImageChanged);
383}

385void RenderImage::notifyFinished(CachedResource& newImage, const NetworkLoadMetrics& metrics)
386{
387 if (renderTreeBeingDestroyed())
388 return;
389
390 invalidateBackgroundObscurationStatus();
391
392 if (&newImage == cachedImage()) {
393 // tell any potential compositing layers
394 // that the image is done and they can reference it directly.
395 contentChanged(ImageChanged);
396 }
397
398 if (is<HTMLImageElement>(element()))
399 page().didFinishLoadingImageForElement(downcast<HTMLImageElement>(*element()));
400
401 RenderReplaced::notifyFinished(newImage, metrics);
402}

437void RenderImage::paintIncompleteImageOutline(PaintInfo& paintInfo, LayoutPoint paintOffset, LayoutUnit borderWidth) const
438{
439 auto contentSize = this->contentSize();
440 if (contentSize.width() <= 2 || contentSize.height() <= 2)
441 return;
442
443 auto leftBorder = borderLeft();
444 auto topBorder = borderTop();
445 auto leftPadding = paddingLeft();
446 auto topPadding = paddingTop();
447
448 // Draw an outline rect where the image should be.
449 GraphicsContext& context = paintInfo.context();
450 context.setStrokeStyle(SolidStroke);
451 context.setStrokeColor(Color::lightGray);
452 context.setFillColor(Color::transparentBlack);
453 context.drawRect(snapRectToDevicePixels(LayoutRect({ paintOffset.x() + leftBorder + leftPadding, paintOffset.y() + topBorder + topPadding }, contentSize), document().deviceScaleFactor()), borderWidth);
454}

461void RenderImage::paintReplaced(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
462{
463 GraphicsContext& context = paintInfo.context();
464 if (context.invalidatingImagesWithAsyncDecodes()) {
465 if (cachedImage() && cachedImage()->isClientWaitingForAsyncDecoding(*this))
466 cachedImage()->removeAllClientsWaitingForAsyncDecoding();
467 return;
468 }
469
470 auto contentSize = this->contentSize();
471 float deviceScaleFactor = document().deviceScaleFactor();
472 LayoutUnit missingImageBorderWidth(1 / deviceScaleFactor);
473
474 if (context.detectingContentfulPaint()) {
475 if (!context.contenfulPaintDetected() && !isDeferredImage(element()) && cachedImage() && cachedImage()->canRender(this, deviceScaleFactor) && !contentSize.isEmpty())
476 context.setContentfulPaintDetected();
477
478 return;
479 }
480
481 if (!imageResource().cachedImage() || isDeferredImage(element()) || shouldDisplayBrokenImageIcon()) {
482 if (paintInfo.phase == PaintPhase::Selection)
483 return;
484
485 if (paintInfo.phase == PaintPhase::Foreground)
486 page().addRelevantUnpaintedObject(this, visualOverflowRect());
487
488 paintIncompleteImageOutline(paintInfo, paintOffset, missingImageBorderWidth);
489
490 if (contentSize.width() > 2 && contentSize.height() > 2) {
491 LayoutUnit leftBorder = borderLeft();
492 LayoutUnit topBorder = borderTop();
493 LayoutUnit leftPad = paddingLeft();
494 LayoutUnit topPad = paddingTop();
495
496 bool errorPictureDrawn = false;
497 LayoutSize imageOffset;
498 // When calculating the usable dimensions, exclude the pixels of
499 // the outline rect so the error image/alt text doesn't draw on it.
500 LayoutSize usableSize = contentSize - LayoutSize(2 * missingImageBorderWidth, 2 * missingImageBorderWidth);
501
502 RefPtr<Image> image = imageResource().image();
503
504 if (shouldDisplayBrokenImageIcon() && !image->isNull() && usableSize.width() >= image->width() && usableSize.height() >= image->height()) {
505 // Call brokenImage() explicitly to ensure we get the broken image icon at the appropriate resolution.
506 std::pair<Image*, float> brokenImageAndImageScaleFactor = cachedImage()->brokenImage(deviceScaleFactor);
507 image = brokenImageAndImageScaleFactor.first;
508 FloatSize imageSize = image->size();
509 imageSize.scale(1 / brokenImageAndImageScaleFactor.second);
510
511 // Center the error image, accounting for border and padding.
512 LayoutUnit centerX { (usableSize.width() - imageSize.width()) / 2 };
513 if (centerX < 0)
514 centerX = 0;
515 LayoutUnit centerY { (usableSize.height() - imageSize.height()) / 2 };
516 if (centerY < 0)
517 centerY = 0;
518 imageOffset = LayoutSize(leftBorder + leftPad + centerX + missingImageBorderWidth, topBorder + topPad + centerY + missingImageBorderWidth);
519
520 context.drawImage(*image, snapRectToDevicePixels(LayoutRect(paintOffset + imageOffset, imageSize), deviceScaleFactor), { imageOrientation() });
521 errorPictureDrawn = true;
522 }
523
524 if (!m_altText.isEmpty()) {
525 String text = document().displayStringModifiedByEncoding(m_altText);
526 context.setFillColor(style().visitedDependentColorWithColorFilter(CSSPropertyColor));
527 const FontCascade& font = style().fontCascade();
528 const FontMetrics& fontMetrics = font.fontMetrics();
529 LayoutUnit ascent = fontMetrics.ascent();
530 LayoutPoint altTextOffset = paintOffset;
531 altTextOffset.move(leftBorder + leftPad + (paddingWidth / 2) - missingImageBorderWidth, topBorder + topPad + ascent + (paddingHeight / 2) - missingImageBorderWidth);
532
533 // Only draw the alt text if it'll fit within the content box,
534 // and only if it fits above the error image.
535 TextRun textRun = RenderBlock::constructTextRun(text, style());
536 LayoutUnit textWidth { font.width(textRun) };
537 if (errorPictureDrawn) {
538 if (usableSize.width() >= textWidth && fontMetrics.height() <= imageOffset.height())
539 context.drawBidiText(font, textRun, altTextOffset);
540 } else if (usableSize.width() >= textWidth && usableSize.height() >= fontMetrics.height())
541 context.drawBidiText(font, textRun, altTextOffset);
542 }
543 }
544 return;
545 }
546
547 if (contentSize.isEmpty())
548 return;
549
550 bool showBorderForIncompleteImage = settings().incompleteImageBorderEnabled();
551
552 RefPtr<Image> img = imageResource().image(flooredIntSize(contentSize));
553 if (!img || img->isNull()) {
554 if (showBorderForIncompleteImage)
555 paintIncompleteImageOutline(paintInfo, paintOffset, missingImageBorderWidth);
556
557 if (paintInfo.phase == PaintPhase::Foreground)
558 page().addRelevantUnpaintedObject(this, visualOverflowRect());
559 return;
560 }
561
562 LayoutRect contentBoxRect = this->contentBoxRect();
563 contentBoxRect.moveBy(paintOffset);
564 LayoutRect replacedContentRect = this->replacedContentRect();
565 replacedContentRect.moveBy(paintOffset);
566 bool clip = !contentBoxRect.contains(replacedContentRect);
567 GraphicsContextStateSaver stateSaver(context, clip);
568 if (clip)
569 context.clip(contentBoxRect);
570
571 ImageDrawResult result = paintIntoRect(paintInfo, snapRectToDevicePixels(replacedContentRect, deviceScaleFactor));
572
573 if (showBorderForIncompleteImage && (result != ImageDrawResult::DidDraw || (cachedImage() && cachedImage()->isLoading())))
574 paintIncompleteImageOutline(paintInfo, paintOffset, missingImageBorderWidth);
575
576 if (cachedImage() && paintInfo.phase == PaintPhase::Foreground) {
577 // For now, count images as unpainted if they are still progressively loading. We may want
578 // to refine this in the future to account for the portion of the image that has painted.
579 LayoutRect visibleRect = intersection(replacedContentRect, contentBoxRect);
580 if (cachedImage()->isLoading() || result == ImageDrawResult::DidRequestDecoding)
581 page().addRelevantUnpaintedObject(this, visibleRect);
582 else
583 page().addRelevantRepaintedObject(this, visibleRect);
584 }
585}

595void RenderImage::paintAreaElementFocusRing(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
596{
597 if (document().printing() || !frame().selection().isFocusedAndActive())
598 return;
599
600 if (paintInfo.context().paintingDisabled() && !paintInfo.context().performingPaintInvalidation())
601 return;
602
603 Element* focusedElement = document().focusedElement();
604 if (!is<HTMLAreaElement>(focusedElement))
605 return;
606
607 HTMLAreaElement& areaElement = downcast<HTMLAreaElement>(*focusedElement);
608 if (areaElement.imageElement() != element())
609 return;
610
611 auto* areaElementStyle = areaElement.computedStyle();
612 if (!areaElementStyle)
613 return;
614
615 float outlineWidth = areaElementStyle->outlineWidth();
616 if (!outlineWidth)
617 return;
618
619 // Even if the theme handles focus ring drawing for entire elements, it won't do it for
620 // an area within an image, so we don't call RenderTheme::supportsFocusRing here.
621 auto path = areaElement.computePathForFocusRing(size());
622 if (path.isEmpty())
623 return;
624
625 AffineTransform zoomTransform;
626 zoomTransform.scale(style().effectiveZoom());
627 path.transform(zoomTransform);
628
629 auto adjustedOffset = paintOffset;
630 adjustedOffset.moveBy(location());
631 path.translate(toFloatSize(adjustedOffset));
632
633#if PLATFORM(MAC)
634 bool needsRepaint;
635 paintInfo.context().drawFocusRing(path, page().focusController().timeSinceFocusWasSet().seconds(), needsRepaint, RenderTheme::singleton().focusRingColor(styleColorOptions()));
636 if (needsRepaint)
637 page().focusController().setFocusedElementNeedsRepaint();
638#else
639 paintInfo.context().drawFocusRing(path, outlineWidth, areaElementStyle->outlineOffset(), areaElementStyle->visitedDependentColorWithColorFilter(CSSPropertyOutlineColor));
640#endif // PLATFORM(MAC)
641}

643void RenderImage::areaElementFocusChanged(HTMLAreaElement* element)
644{
645 ASSERT_UNUSED(element, element->imageElement() == this->element());
646
647 // It would be more efficient to only repaint the focus ring rectangle
648 // for the passed-in area element. That would require adding functions
649 // to the area element class.
650 repaint();

653ImageDrawResult RenderImage::paintIntoRect(PaintInfo& paintInfo, const FloatRect& rect)
654{
655 if (!imageResource().cachedImage() || imageResource().errorOccurred() || rect.width() <= 0 || rect.height() <= 0)
656 return ImageDrawResult::DidNothing;
657
658 RefPtr<Image> img = imageResource().image(flooredIntSize(rect.size()));
659 if (!img || img->isNull())
660 return ImageDrawResult::DidNothing;
661
662 HTMLImageElement* imageElement = is<HTMLImageElement>(element()) ? downcast<HTMLImageElement>(element()) : nullptr;
663
664 // FIXME: Document when image != img.get().
665 Image* image = imageResource().image().get();
666
667#if USE(CG)
668 if (is<PDFDocumentImage>(image))
669 downcast<PDFDocumentImage>(*image).setPdfImageCachingPolicy(settings().pdfImageCachingPolicy());
670#endif
671
672 if (is<BitmapImage>(image))
673 downcast<BitmapImage>(*image).updateFromSettings(settings());
674
675 ImagePaintingOptions options = {
676 imageElement ? imageElement->compositeOperator() : CompositeOperator::SourceOver,
677 decodingModeForImageDraw(*image, paintInfo),
678 imageOrientation(),
679 image ? chooseInterpolationQuality(paintInfo.context(), *image, image, LayoutSize(rect.size())) : InterpolationQuality::Default
680 };
681
682 auto drawResult = paintInfo.context().drawImage(*img, rect, options);
683 if (drawResult == ImageDrawResult::DidRequestDecoding)
684 imageResource().cachedImage()->addClientWaitingForAsyncDecoding(*this);
685
686#if USE(SYSTEM_PREVIEW)
687 if (imageElement && imageElement->isSystemPreviewImage() && drawResult == ImageDrawResult::DidDraw && imageElement->document().settings().systemPreviewEnabled())
688 theme().paintSystemPreviewBadge(*img, paintInfo, rect);
689#endif
690
691 return drawResult;
692}

702bool RenderImage::foregroundIsKnownToBeOpaqueInRect(const LayoutRect& localRect, unsigned maxDepthToTest) const
703{
704 UNUSED_PARAM(maxDepthToTest);
705 if (!imageResource().cachedImage() || imageResource().errorOccurred())
706 return false;
707 if (cachedImage() && !cachedImage()->isLoaded())
708 return false;
709 if (!contentBoxRect().contains(localRect))
710 return false;
711 FillBox backgroundClip = style().backgroundClip();
712 // Background paints under borders.
713 if (backgroundClip == FillBox::Border && style().hasBorder() && !borderObscuresBackground())
714 return false;
715 // Background shows in padding area.
716 if ((backgroundClip == FillBox::Border || backgroundClip == FillBox::Padding) && style().hasPadding())
717 return false;
718 // Object-fit may leave parts of the content box empty.
719 ObjectFit objectFit = style().objectFit();
720 if (objectFit != ObjectFit::Fill && objectFit != ObjectFit::Cover)
721 return false;
722
723 LengthPoint objectPosition = style().objectPosition();
724 if (objectPosition != RenderStyle::initialObjectPosition())
725 return false;
726
727 // Check for image with alpha.
728 return cachedImage() && cachedImage()->currentFrameKnownToBeOpaque(this);
729}

755bool RenderImage::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
756{
757 HitTestResult tempResult(result.hitTestLocation());
758 bool inside = RenderReplaced::nodeAtPoint(request, tempResult, locationInContainer, accumulatedOffset, hitTestAction);
759
760 if (tempResult.innerNode() && element()) {
761 if (HTMLMapElement* map = imageMap()) {
762 LayoutRect contentBox = contentBoxRect();
763 float scaleFactor = 1 / style().effectiveZoom();
764 LayoutPoint mapLocation = locationInContainer.point() - toLayoutSize(accumulatedOffset) - locationOffset() - toLayoutSize(contentBox.location());
765 mapLocation.scale(scaleFactor);
766
767 if (map->mapMouseEvent(mapLocation, contentBox.size(), tempResult))
768 tempResult.setInnerNonSharedNode(element());
769 }
770 }
771
772 if (!inside && request.resultIsElementList())
773 result.append(tempResult, request);
774 if (inside)
775 result = tempResult;
776 return inside;
777}

814void RenderImage::layoutShadowContent(const LayoutSize& oldSize)
815{
816 for (auto& renderBox : childrenOfType<RenderBox>(*this)) {
817 bool childNeedsLayout = renderBox.needsLayout();
818 // If the region chain has changed we also need to relayout the children to update the region box info.
819 // FIXME: We can do better once we compute region box info for RenderReplaced, not only for RenderBlock.
820 auto* fragmentedFlow = enclosingFragmentedFlow();
821 if (fragmentedFlow && !childNeedsLayout) {
822 if (fragmentedFlow->pageLogicalSizeChanged())
823 childNeedsLayout = true;
824 }
825
826 auto newSize = contentBoxRect().size();
827 if (newSize == oldSize && !childNeedsLayout)
828 continue;
829
830 // When calling layout() on a child node, a parent must either push a LayoutStateMaintainer, or
831 // instantiate LayoutStateDisabler. Since using a LayoutStateMaintainer is slightly more efficient,
832 // and this method might be called many times per second during video playback, use a LayoutStateMaintainer:
833 LayoutStateMaintainer statePusher(*this, locationOffset(), hasTransform() || hasReflection() || style().isFlippedBlocksWritingMode());
834 renderBox.setLocation(LayoutPoint(borderLeft(), borderTop()) + LayoutSize(paddingLeft(), paddingTop()));
835 renderBox.mutableStyle().setHeight(Length(newSize.height(), LengthType::Fixed));
836 renderBox.mutableStyle().setWidth(Length(newSize.width(), LengthType::Fixed));
837 renderBox.setNeedsLayout(MarkOnlyThis);
838 renderBox.layout();
839 }
840
841 clearChildNeedsLayout();
842}

844void RenderImage::computeIntrinsicRatioInformation(FloatSize& intrinsicSize, double& intrinsicRatio) const
845{
846 ASSERT(!shouldApplySizeContainment(*this));
847 RenderReplaced::computeIntrinsicRatioInformation(intrinsicSize, intrinsicRatio);
848
849 // Our intrinsicSize is empty if we're rendering generated images with relative width/height. Figure out the right intrinsic size to use.
850 if (intrinsicSize.isEmpty() && (imageResource().imageHasRelativeWidth() || imageResource().imageHasRelativeHeight())) {
851 RenderObject* containingBlock = isOutOfFlowPositioned() ? container() : this->containingBlock();
852 if (is<RenderBox>(*containingBlock)) {
853 auto& box = downcast<RenderBox>(*containingBlock);
854 intrinsicSize.setWidth(box.availableLogicalWidth());
855 intrinsicSize.setHeight(box.availableLogicalHeight(IncludeMarginBorderPadding));
856 }
857 }
858
859 // Don't compute an intrinsic ratio to preserve historical WebKit behavior if we're painting alt text and/or a broken image.
860 if (shouldDisplayBrokenImageIcon()) {
861 if (settings().aspectRatioOfImgFromWidthAndHeightEnabled()
862 && style().aspectRatioType() == AspectRatioType::AutoAndRatio && !isShowingAltText())
863 intrinsicRatio = style().logicalAspectRatio();
864 else
865 intrinsicRatio = 1;
866 return;
867 }
868}
