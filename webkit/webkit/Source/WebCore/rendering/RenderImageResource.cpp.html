40WTF_MAKE_ISO_ALLOCATED_IMPL(RenderImageResource);
41
42RenderImageResource::RenderImageResource() = default;
43
44void RenderImageResource::initialize(RenderElement& renderer, CachedImage* styleCachedImage)
45{
46 ASSERT(!m_renderer);
47 ASSERT(!m_cachedImage);
48 m_renderer = makeWeakPtr(renderer);
49 m_cachedImage = styleCachedImage;
50 m_cachedImageRemoveClientIsNeeded = !styleCachedImage;
51}
52
53void RenderImageResource::shutdown()
54{
55 image()->stopAnimation();
56 setCachedImage(nullptr);
57}
58
59void RenderImageResource::setCachedImage(CachedImage* newImage)
60{
61 if (m_cachedImage == newImage)
62 return;
63
64 if (m_cachedImage && m_renderer && m_cachedImageRemoveClientIsNeeded)
65 m_cachedImage->removeClient(*m_renderer);
66 m_cachedImage = newImage;
67 m_cachedImageRemoveClientIsNeeded = true;
68 if (!m_cachedImage)
69 return;
70
71 m_cachedImage->addClient(*renderer());
72 if (m_cachedImage->errorOccurred())
73 renderer()->imageChanged(m_cachedImage.get());
74}
75
76void RenderImageResource::resetAnimation()
77{
78 if (!m_cachedImage)
79 return;
80
81 image()->resetAnimation();
82
83 if (m_renderer && !m_renderer->needsLayout())
84 m_renderer->repaint();
85}
86
87RefPtr<Image> RenderImageResource::image(const IntSize&) const
88{
89 if (!m_cachedImage)
90 return &Image::nullImage();
91 if (auto image = m_cachedImage->imageForRenderer(m_renderer.get()))
92 return image;
93 return &Image::nullImage();
94}
95
96void RenderImageResource::setContainerContext(const IntSize& imageContainerSize, const URL& imageURL)
97{
98 if (!m_cachedImage || !m_renderer)
99 return;
100 m_cachedImage->setContainerContextForClient(*m_renderer, imageContainerSize, m_renderer->style().effectiveZoom(), imageURL);
101}
102
103LayoutSize RenderImageResource::imageSize(float multiplier, CachedImage::SizeType type) const
104{
105 if (!m_cachedImage)
106 return LayoutSize();
107 LayoutSize size = m_cachedImage->imageSizeForRenderer(m_renderer.get(), multiplier, type);
108 if (is<RenderImage>(m_renderer.get()))
109 size.scale(downcast<RenderImage>(*m_renderer).imageDevicePixelRatio());
110 return size;
111}
