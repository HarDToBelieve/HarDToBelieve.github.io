26#pragma once
27
28#include "RenderImageResource.h"
29#include "StyleImage.h"
30#include <wtf/RefPtr.h>
31
32namespace WebCore {
33
34class RenderElement;
35
36class RenderImageResourceStyleImage final : public RenderImageResource {
37 WTF_MAKE_ISO_ALLOCATED(RenderImageResourceStyleImage);
38public:
39 explicit RenderImageResourceStyleImage(StyleImage&);
40
41private:
42 void initialize(RenderElement&) final;
43 void shutdown() final;
44
45 RefPtr<Image> image(const IntSize& = { }) const final;
46 bool errorOccurred() const final { return m_styleImage->errorOccurred(); }
47
48 void setContainerContext(const IntSize&, const URL&) final;
49
50 bool imageHasRelativeWidth() const final { return m_styleImage->imageHasRelativeWidth(); }
51 bool imageHasRelativeHeight() const final { return m_styleImage->imageHasRelativeHeight(); }
52
53 WrappedImagePtr imagePtr() const final { return m_styleImage->data(); }
54 LayoutSize imageSize(float multiplier, CachedImage::SizeType) const final { return LayoutSize(m_styleImage->imageSize(renderer(), multiplier)); }
55
56 Ref<StyleImage> m_styleImage;
57};
58
59} // namespace WebCore
