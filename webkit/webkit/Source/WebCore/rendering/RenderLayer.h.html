246 // These flags propagate in paint order (z-order tree).
247 enum class Compositing {
248 HasDescendantNeedingRequirementsTraversal = 1 << 0, // Need to do the overlap-testing tree walk because hierarchy or geometry changed.
249 HasDescendantNeedingBackingOrHierarchyTraversal = 1 << 1, // Need to update geometry, configuration and update the GraphicsLayer tree.
250
251 // Things that trigger HasDescendantNeedingRequirementsTraversal
252 NeedsPaintOrderChildrenUpdate = 1 << 2, // The paint order children of this layer changed (gained/lost child, order change).
253 NeedsPostLayoutUpdate = 1 << 3, // Needs compositing to be re-evaluated after layout (it depends on geometry).
254 DescendantsNeedRequirementsTraversal = 1 << 4, // Something changed that forces computeCompositingRequirements to traverse all descendant layers.
255 SubsequentLayersNeedRequirementsTraversal = 1 << 5, // Something changed that forces computeCompositingRequirements to traverse all layers later in paint order.
256
257 // Things that trigger HasDescendantNeedingBackingOrHierarchyTraversal
258 NeedsGeometryUpdate = 1 << 6, // This layer needs a geometry update.
259 NeedsConfigurationUpdate = 1 << 7, // This layer needs a configuration update (updating its internal compositing hierarchy).
260 NeedsScrollingTreeUpdate = 1 << 8, // Something changed that requires this layer's scrolling tree node to be updated.
261 NeedsLayerConnection = 1 << 9, // This layer needs hookup with its parents or children.
262 ChildrenNeedGeometryUpdate = 1 << 10, // This layer's composited children need a geometry update.
263 DescendantsNeedBackingAndHierarchyTraversal = 1 << 11, // Something changed that forces us to traverse all descendant layers in updateBackingAndHierarchy.
264 };
