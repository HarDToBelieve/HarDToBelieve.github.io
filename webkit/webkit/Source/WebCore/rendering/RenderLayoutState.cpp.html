38RenderLayoutState::RenderLayoutState(RenderElement& renderer, IsPaginated isPaginated)
39 : m_clipped(false)
40 , m_isPaginated(isPaginated == IsPaginated::Yes)
41 , m_pageLogicalHeightChanged(false)
42#if ASSERT_ENABLED
43 , m_layoutDeltaXSaturated(false)
44 , m_layoutDeltaYSaturated(false)
45 , m_renderer(&renderer)
46#endif
47{
48 if (RenderElement* container = renderer.container()) {
49 FloatPoint absContentPoint = container->localToAbsolute(FloatPoint(), UseTransforms);
50 m_paintOffset = LayoutSize(absContentPoint.x(), absContentPoint.y());
51
52 if (container->hasOverflowClip()) {
53 m_clipped = true;
54 auto& containerBox = downcast<RenderBox>(*container);
55 m_clipRect = LayoutRect(toLayoutPoint(m_paintOffset), containerBox.cachedSizeForOverflowClip());
56 m_paintOffset -= toLayoutSize(containerBox.scrollPosition());
57 }
58 }
59 if (m_isPaginated) {
60 // This is just a flag for known page height (see RenderBlockFlow::checkForPaginationLogicalHeightChange).
61 m_pageLogicalHeight = 1;
62 }
63}

65RenderLayoutState::RenderLayoutState(const FrameViewLayoutContext::LayoutStateStack& layoutStateStack, RenderBox& renderer, const LayoutSize& offset, LayoutUnit pageLogicalHeight, bool pageLogicalHeightChanged)
66 : m_clipped(false)
67 , m_isPaginated(false)
68 , m_pageLogicalHeightChanged(false)
69#if ASSERT_ENABLED
70 , m_layoutDeltaXSaturated(false)
71 , m_layoutDeltaYSaturated(false)
72 , m_renderer(&renderer)
73#endif
74{
75 if (!layoutStateStack.isEmpty()) {
76 auto& ancestor = *layoutStateStack.last().get();
77 computeOffsets(ancestor, renderer, offset);
78 computeClipRect(ancestor, renderer);
79 }
80 computePaginationInformation(layoutStateStack, renderer, pageLogicalHeight, pageLogicalHeightChanged);
81}

83void RenderLayoutState::computeOffsets(const RenderLayoutState& ancestor, RenderBox& renderer, LayoutSize offset)
84{
85 bool fixed = renderer.isFixedPositioned();
86 if (fixed) {
87 FloatPoint fixedOffset = renderer.view().localToAbsolute(FloatPoint(), IsFixed);
88 m_paintOffset = LayoutSize(fixedOffset.x(), fixedOffset.y()) + offset;
89 } else
90 m_paintOffset = ancestor.paintOffset() + offset;
91
92 if (renderer.isOutOfFlowPositioned() && !fixed) {
93 if (auto* container = renderer.container()) {
94 if (container->isInFlowPositioned() && is<RenderInline>(*container))
95 m_paintOffset += downcast<RenderInline>(*container).offsetForInFlowPositionedInline(&renderer);
96 }
97 }
98
99 m_layoutOffset = m_paintOffset;
100
101 if (renderer.isInFlowPositioned() && renderer.hasLayer())
102 m_paintOffset += renderer.layer()->offsetForInFlowPosition();
103
104 if (renderer.hasOverflowClip())
105 m_paintOffset -= toLayoutSize(renderer.scrollPosition());
106
107 m_layoutDelta = ancestor.layoutDelta();
108#if ASSERT_ENABLED
109 m_layoutDeltaXSaturated = ancestor.m_layoutDeltaXSaturated;
110 m_layoutDeltaYSaturated = ancestor.m_layoutDeltaYSaturated;
111#endif
112}

114void RenderLayoutState::computeClipRect(const RenderLayoutState& ancestor, RenderBox& renderer)
115{
116 m_clipped = !renderer.isFixedPositioned() && ancestor.isClipped();
117 if (m_clipped)
118 m_clipRect = ancestor.clipRect();
119 if (!renderer.hasOverflowClip())
120 return;
121
122 auto paintOffsetForClipRect = toLayoutPoint(m_paintOffset + toLayoutSize(renderer.scrollPosition()));
123 LayoutRect clipRect(paintOffsetForClipRect + renderer.view().frameView().layoutContext().layoutDelta(), renderer.cachedSizeForOverflowClip());
124 if (m_clipped)
125 m_clipRect.intersect(clipRect);
126 else
127 m_clipRect = clipRect;
128 m_clipped = true;
129 // FIXME: <http://bugs.webkit.org/show_bug.cgi?id=13443> Apply control clip if present.
130}

132void RenderLayoutState::computePaginationInformation(const FrameViewLayoutContext::LayoutStateStack& layoutStateStack, RenderBox& renderer, LayoutUnit pageLogicalHeight, bool pageLogicalHeightChanged)
133{
134 auto* ancestor = layoutStateStack.isEmpty() ? nullptr : layoutStateStack.last().get();
135 // If we establish a new page height, then cache the offset to the top of the first page.
136 // We can compare this later on to figure out what part of the page we're actually on.
137 if (pageLogicalHeight || renderer.isRenderFragmentedFlow()) {
138 m_pageLogicalHeight = pageLogicalHeight;
139 bool isFlipped = renderer.style().isFlippedBlocksWritingMode();
140 m_pageOffset = LayoutSize(m_layoutOffset.width() + (!isFlipped ? renderer.borderLeft() + renderer.paddingLeft() : renderer.borderRight() + renderer.paddingRight()), m_layoutOffset.height() + (!isFlipped ? renderer.borderTop() + renderer.paddingTop() : renderer.borderBottom() + renderer.paddingBottom()));
141 m_pageLogicalHeightChanged = pageLogicalHeightChanged;
142 m_isPaginated = true;
143 } else if (ancestor) {
144 // If we don't establish a new page height, then propagate the old page height and offset down.
145 m_pageLogicalHeight = ancestor->pageLogicalHeight();
146 m_pageLogicalHeightChanged = ancestor->pageLogicalHeightChanged();
147 m_pageOffset = ancestor->pageOffset();
148
149 // Disable pagination for objects we don't support. For now this includes overflow:scroll/auto, inline blocks and writing mode roots.
150 if (renderer.isUnsplittableForPagination()) {
151 m_pageLogicalHeight = 0;
152 m_isPaginated = false;
153 } else
154 m_isPaginated = m_pageLogicalHeight || renderer.enclosingFragmentedFlow();
155 }
156
157 // Propagate line grid information.
158 if (ancestor)
159 propagateLineGridInfo(*ancestor, renderer);
160
161 if (lineGrid() && (lineGrid()->style().writingMode() == renderer.style().writingMode()) && is<RenderMultiColumnFlow>(renderer))
162 computeLineGridPaginationOrigin(downcast<RenderMultiColumnFlow>(renderer));
163
164 // If we have a new grid to track, then add it to our set.
165 if (renderer.style().lineGrid() != RenderStyle::initialLineGrid() && is<RenderBlockFlow>(renderer))
166 establishLineGrid(layoutStateStack, downcast<RenderBlockFlow>(renderer));
167}

169LayoutUnit RenderLayoutState::pageLogicalOffset(RenderBox* child, LayoutUnit childLogicalOffset) const
170{
171 if (child->isHorizontalWritingMode())
172 return m_layoutOffset.height() + childLogicalOffset - m_pageOffset.height();
173 return m_layoutOffset.width() + childLogicalOffset - m_pageOffset.width();
174}

176void RenderLayoutState::computeLineGridPaginationOrigin(const RenderMultiColumnFlow& multicol)
177{
178 if (!isPaginated() || !pageLogicalHeight())
179 return;
180
181 if (!multicol.progressionIsInline())
182 return;
183 // We need to cache a line grid pagination origin so that we understand how to reset the line grid
184 // at the top of each column.
185 // Get the current line grid and offset.
186 ASSERT(m_lineGrid);
187 // Get the hypothetical line box used to establish the grid.
188 auto* lineGridBox = m_lineGrid->lineGridBox();
189 if (!lineGridBox)
190 return;
191
192 // Now determine our position on the grid. Our baseline needs to be adjusted to the nearest baseline multiple
193 // as established by the line box.
194 // FIXME: Need to handle crazy line-box-contain values that cause the root line box to not be considered. I assume
195 // the grid should honor line-box-contain.
196 bool isHorizontalWritingMode = m_lineGrid->isHorizontalWritingMode();
197 LayoutUnit lineGridBlockOffset = isHorizontalWritingMode ? m_lineGridOffset.height() : m_lineGridOffset.width();
198 LayoutUnit firstLineTopWithLeading = lineGridBlockOffset + lineGridBox->lineBoxTop();
199 LayoutUnit pageLogicalTop = isHorizontalWritingMode ? m_pageOffset.height() : m_pageOffset.width();
200 if (pageLogicalTop <= firstLineTopWithLeading)
201 return;
202
203 // Shift to the next highest line grid multiple past the page logical top. Cache the delta
204 // between this new value and the page logical top as the pagination origin.
205 auto lineBoxHeight = lineGridBox->lineBoxHeight();
206 if (!roundToInt(lineBoxHeight))
207 return;
208 LayoutUnit remainder = roundToInt(pageLogicalTop - firstLineTopWithLeading) % roundToInt(lineBoxHeight);
209 LayoutUnit paginationDelta = lineBoxHeight - remainder;
210 if (isHorizontalWritingMode)
211 m_lineGridPaginationOrigin.setHeight(paginationDelta);
212 else
213 m_lineGridPaginationOrigin.setWidth(paginationDelta);
214}

216void RenderLayoutState::propagateLineGridInfo(const RenderLayoutState& ancestor, RenderBox& renderer)
217{
218 // Disable line grids for objects we don't support. For now this includes overflow:scroll/auto, inline blocks and
219 // writing mode roots.
220 if (renderer.isUnsplittableForPagination())
221 return;
222
223 m_lineGrid = makeWeakPtr(ancestor.lineGrid());
224 m_lineGridOffset = ancestor.lineGridOffset();
225 m_lineGridPaginationOrigin = ancestor.lineGridPaginationOrigin();
226}

228void RenderLayoutState::establishLineGrid(const FrameViewLayoutContext::LayoutStateStack& layoutStateStack, RenderBlockFlow& renderer)
229{
230 // First check to see if this grid has been established already.
231 if (m_lineGrid) {
232 if (m_lineGrid->style().lineGrid() == renderer.style().lineGrid())
233 return;
234 auto* currentGrid = m_lineGrid.get();
235 for (int i = layoutStateStack.size() - 1; i >= 0; --i) {
236 auto& currentState = *layoutStateStack[i].get();
237 if (currentState.m_lineGrid == currentGrid)
238 continue;
239 currentGrid = currentState.lineGrid();
240 if (!currentGrid)
241 break;
242 if (currentGrid->style().lineGrid() == renderer.style().lineGrid()) {
243 m_lineGrid = makeWeakPtr(currentGrid);
244 m_lineGridOffset = currentState.m_lineGridOffset;
245 return;
246 }
247 }
248 }
249
250 // We didn't find an already-established grid with this identifier. Our render object establishes the grid.
251 m_lineGrid = makeWeakPtr(renderer);
252 m_lineGridOffset = m_layoutOffset;
253}

255void RenderLayoutState::addLayoutDelta(LayoutSize delta)
256{
257 m_layoutDelta += delta;
258#if ASSERT_ENABLED
259 m_layoutDeltaXSaturated |= m_layoutDelta.width() == LayoutUnit::max() || m_layoutDelta.width() == LayoutUnit::min();
260 m_layoutDeltaYSaturated |= m_layoutDelta.height() == LayoutUnit::max() || m_layoutDelta.height() == LayoutUnit::min();
261#endif
262}
263
264#if ASSERT_ENABLED
265bool RenderLayoutState::layoutDeltaMatches(LayoutSize delta) const
266{
267 return (delta.width() == m_layoutDelta.width() || m_layoutDeltaXSaturated) && (delta.height() == m_layoutDelta.height() || m_layoutDeltaYSaturated);
268}
269#endif

271LayoutStateMaintainer::LayoutStateMaintainer(RenderBox& root, LayoutSize offset, bool disablePaintOffsetCache, LayoutUnit pageHeight, bool pageHeightChanged)
272 : m_context(root.view().frameView().layoutContext())
273 , m_paintOffsetCacheIsDisabled(disablePaintOffsetCache)
274{
275 m_didPushLayoutState = m_context.pushLayoutState(root, offset, pageHeight, pageHeightChanged);
276 if (m_didPushLayoutState && m_paintOffsetCacheIsDisabled)
277 m_context.disablePaintOffsetCache();
278}
279
280LayoutStateMaintainer::~LayoutStateMaintainer()
281{
282 if (!m_didPushLayoutState)
283 return;
284 m_context.popLayoutState();
285 if (m_paintOffsetCacheIsDisabled)
286 m_context.enablePaintOffsetCache();
287}

300static bool shouldDisablePaintOffsetCacheForSubtree(RenderElement& subtreeLayoutRoot)
301{
302 for (auto* renderer = &subtreeLayoutRoot; renderer; renderer = renderer->container()) {
303 if (renderer->hasTransform() || renderer->hasReflection())
304 return true;
305 }
306 return false;
307}
308
309SubtreeLayoutStateMaintainer::SubtreeLayoutStateMaintainer(RenderElement* subtreeLayoutRoot)
310{
311 if (subtreeLayoutRoot) {
312 m_context = &subtreeLayoutRoot->view().frameView().layoutContext();
313 m_context->pushLayoutState(*subtreeLayoutRoot);
314 if (shouldDisablePaintOffsetCacheForSubtree(*subtreeLayoutRoot)) {
315 m_context->disablePaintOffsetCache();
316 m_didDisablePaintOffsetCache = true;
317 }
318 }
319}
320
321SubtreeLayoutStateMaintainer::~SubtreeLayoutStateMaintainer()
322{
323 if (m_context) {
324 m_context->popLayoutState();
325 if (m_didDisablePaintOffsetCache)
326 m_context->enablePaintOffsetCache();
327 }
328}
