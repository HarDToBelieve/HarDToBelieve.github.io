22#pragma once
23
24#include "RenderBoxModelObject.h"
25
26namespace WebCore {
27
28class LegacyInlineElementBox;
29class HTMLElement;
30class Position;
31
32class RenderLineBreak final : public RenderBoxModelObject {
33 WTF_MAKE_ISO_ALLOCATED(RenderLineBreak);
34public:
35 RenderLineBreak(HTMLElement&, RenderStyle&&);
36 virtual ~RenderLineBreak();
37
38 // FIXME: The lies here keep render tree dump based test results unchanged.
39 const char* renderName() const final { return m_isWBR ? "RenderWordBreak" : "RenderBR"; }
40
41 bool isWBR() const final { return m_isWBR; }
42
43 std::unique_ptr<LegacyInlineElementBox> createInlineBox();
44 LegacyInlineElementBox* inlineBoxWrapper() const { return m_inlineBoxWrapper; }
45 void setInlineBoxWrapper(LegacyInlineElementBox*);
46 void deleteInlineBoxWrapper();
47 void replaceInlineBoxWrapper(LegacyInlineElementBox&);
48 void dirtyLineBoxes(bool fullLayout);
49
50 IntRect linesBoundingBox() const;
51
52 void absoluteRects(Vector<IntRect>&, const LayoutPoint& accumulatedOffset) const final;
53 void absoluteQuads(Vector<FloatQuad>&, bool* wasFixed = nullptr) const final;
54#if PLATFORM(IOS_FAMILY)
55 void collectSelectionGeometries(Vector<SelectionGeometry>&, unsigned startOffset = 0, unsigned endOffset = std::numeric_limits<unsigned>::max()) final;
56#endif
57
58private:
59 void node() const = delete;
60
61 bool canHaveChildren() const final { return false; }
62 void paint(PaintInfo&, const LayoutPoint&) final { }
63
64 VisiblePosition positionForPoint(const LayoutPoint&, const RenderFragmentContainer*) final;
65 int caretMinOffset() const final;
66 int caretMaxOffset() const final;
67 bool canBeSelectionLeaf() const final;
68
69 LayoutUnit lineHeight(bool firstLine, LineDirectionMode, LinePositionMode) const final;
70 LayoutUnit baselinePosition(FontBaseline, bool firstLine, LineDirectionMode, LinePositionMode) const final;
71
72 LayoutUnit marginTop() const final { return 0; }
73 LayoutUnit marginBottom() const final { return 0; }
74 LayoutUnit marginLeft() const final { return 0; }
75 LayoutUnit marginRight() const final { return 0; }
76 LayoutUnit marginBefore(const RenderStyle*) const final { return 0; }
77 LayoutUnit marginAfter(const RenderStyle*) const final { return 0; }
78 LayoutUnit marginStart(const RenderStyle*) const final { return 0; }
79 LayoutUnit marginEnd(const RenderStyle*) const final { return 0; }
80 LayoutUnit offsetWidth() const final { return linesBoundingBox().width(); }
81 LayoutUnit offsetHeight() const final { return linesBoundingBox().height(); }
82 LayoutRect borderBoundingBox() const final { return LayoutRect(LayoutPoint(), linesBoundingBox().size()); }
83 LayoutRect frameRectForStickyPositioning() const final { ASSERT_NOT_REACHED(); return LayoutRect(); }
84 LayoutRect clippedOverflowRect(const RenderLayerModelObject*, VisibleRectContext) const final { return LayoutRect(); }
85
86 void updateFromStyle() final;
87 bool requiresLayer() const final { return false; }
88
89 LegacyInlineElementBox* m_inlineBoxWrapper;
90 mutable int m_cachedLineHeight;
91 bool m_isWBR;
92};
93
94} // namespace WebCore
