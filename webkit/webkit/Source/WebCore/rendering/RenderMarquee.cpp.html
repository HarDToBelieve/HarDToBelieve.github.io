60RenderMarquee::RenderMarquee(RenderLayer* layer)
61 : m_layer(layer)
62 , m_timer(*this, &RenderMarquee::timerFired)
63{
64 ASSERT(layer);
65 ASSERT(layer->scrollableArea());
66 layer->scrollableArea()->setConstrainsScrollingToContentEdge(false);
67}

71int RenderMarquee::marqueeSpeed() const
72{
73 int result = m_layer->renderer().style().marqueeSpeed();
74 Element* element = m_layer->renderer().element();
75 if (is<HTMLMarqueeElement>(element))
76 result = std::max(result, downcast<HTMLMarqueeElement>(*element).minimumDelay());
77 return result;
78}

80static MarqueeDirection reverseDirection(MarqueeDirection direction)
81{
82 switch (direction) {
83 case MarqueeDirection::Auto:
84 return MarqueeDirection::Auto;
85 case MarqueeDirection::Left:
86 return MarqueeDirection::Right;
87 case MarqueeDirection::Right:
88 return MarqueeDirection::Left;
89 case MarqueeDirection::Up:
90 return MarqueeDirection::Down;
91 case MarqueeDirection::Down:
92 return MarqueeDirection::Up;
93 case MarqueeDirection::Backward:
94 return MarqueeDirection::Forward;
95 case MarqueeDirection::Forward:
96 return MarqueeDirection::Backward;
97 }
98 return MarqueeDirection::Auto;
99}

101MarqueeDirection RenderMarquee::direction() const
102{
103 // FIXME: Support the CSS3 "auto" value for determining the direction of the marquee.
104 // For now just map MarqueeDirection::Auto to MarqueeDirection::Backward
105 MarqueeDirection result = m_layer->renderer().style().marqueeDirection();
106 TextDirection dir = m_layer->renderer().style().direction();
107 if (result == MarqueeDirection::Auto)
108 result = MarqueeDirection::Backward;
109 if (result == MarqueeDirection::Forward)
110 result = (dir == TextDirection::LTR) ? MarqueeDirection::Right : MarqueeDirection::Left;
111 if (result == MarqueeDirection::Backward)
112 result = (dir == TextDirection::LTR) ? MarqueeDirection::Left : MarqueeDirection::Right;
113
114 // Now we have the real direction. Next we check to see if the increment is negative.
115 // If so, then we reverse the direction.
116 Length increment = m_layer->renderer().style().marqueeIncrement();
117 if (increment.isNegative())
118 result = reverseDirection(result);
119
120 return result;
121}

123bool RenderMarquee::isHorizontal() const
124{
125 return direction() == MarqueeDirection::Left || direction() == MarqueeDirection::Right;
126}

128int RenderMarquee::computePosition(MarqueeDirection dir, bool stopAtContentEdge)
129{
130 RenderBox* box = m_layer->renderBox();
131 ASSERT(box);
132 auto& boxStyle = box->style();
133 if (isHorizontal()) {
134 bool ltr = boxStyle.isLeftToRightDirection();
135 LayoutUnit clientWidth = box->clientWidth();
136 LayoutUnit contentWidth = ltr ? box->maxPreferredLogicalWidth() : box->minPreferredLogicalWidth();
137 if (ltr)
138 contentWidth += (box->paddingRight() - box->borderLeft());
139 else {
140 contentWidth = box->width() - contentWidth;
141 contentWidth += (box->paddingLeft() - box->borderRight());
142 }
143 if (dir == MarqueeDirection::Right) {
144 if (stopAtContentEdge)
145 return std::max<LayoutUnit>(0, ltr ? (contentWidth - clientWidth) : (clientWidth - contentWidth));
146
147 return ltr ? contentWidth : clientWidth;
148 }
149
150 if (stopAtContentEdge)
151 return std::min<LayoutUnit>(0, ltr ? (contentWidth - clientWidth) : (clientWidth - contentWidth));
152
153 return ltr ? -clientWidth : -contentWidth;
154 }
155
156 // Vertical
157 int contentHeight = box->layoutOverflowRect().maxY() - box->borderTop() + box->paddingBottom();
158 int clientHeight = roundToInt(box->clientHeight());
159 if (dir == MarqueeDirection::Up) {
160 if (stopAtContentEdge)
161 return std::min(contentHeight - clientHeight, 0);
162
163 return -clientHeight;
164 }
165
166 if (stopAtContentEdge)
167 return std::max(contentHeight - clientHeight, 0);
168
169 return contentHeight;
170}

172void RenderMarquee::start()
173{
174 if (m_timer.isActive() || m_layer->renderer().style().marqueeIncrement().isZero())
175 return;
176
177 auto* scrollableArea = m_layer->scrollableArea();
178 ASSERT(scrollableArea);
179
180 auto details = ScrollPositionChangeOptions::createProgrammaticUnclamped();
181 if (!m_suspended && !m_stopped) {
182 if (isHorizontal())
183 scrollableArea->scrollToOffset(ScrollOffset(m_start, 0), details);
184 else
185 scrollableArea->scrollToOffset(ScrollOffset(0, m_start), details);
186 } else {
187 m_suspended = false;
188 m_stopped = false;
189 }
190
191 m_timer.startRepeating(1_ms * speed());
192}

194void RenderMarquee::suspend()
195{
196 m_timer.stop();
197 m_suspended = true;
198}

200void RenderMarquee::stop()

202 m_timer.stop();
203 m_stopped = true;

206void RenderMarquee::updateMarqueePosition()
207{
208 bool activate = (m_totalLoops <= 0 || m_currentLoop < m_totalLoops);
209 if (activate) {
210 MarqueeBehavior behavior = m_layer->renderer().style().marqueeBehavior();
211 m_start = computePosition(direction(), behavior == MarqueeBehavior::Alternate);
212 m_end = computePosition(reverseDirection(direction()), behavior == MarqueeBehavior::Alternate || behavior == MarqueeBehavior::Slide);
213 if (!m_stopped)
214 start();
215 }
216}

218void RenderMarquee::updateMarqueeStyle()
219{
220 auto& style = m_layer->renderer().style();
221
222 if (m_direction != style.marqueeDirection() || (m_totalLoops != style.marqueeLoopCount() && m_currentLoop >= m_totalLoops))
223 m_currentLoop = 0; // When direction changes or our loopCount is a smaller number than our current loop, reset our loop.
224
225 m_totalLoops = style.marqueeLoopCount();
226 m_direction = style.marqueeDirection();
227
228 if (m_layer->renderer().isHTMLMarquee()) {
229 // Hack for WinIE. In WinIE, a value of 0 or lower for the loop count for SLIDE means to only do
230 // one loop.
231 if (m_totalLoops <= 0 && style.marqueeBehavior() == MarqueeBehavior::Slide)
232 m_totalLoops = 1;
233 }
234
235 if (speed() != marqueeSpeed()) {
236 m_speed = marqueeSpeed();
237 if (m_timer.isActive())
238 m_timer.startRepeating(1_ms * speed());
239 }
240
241 // Check the loop count to see if we should now stop.
242 bool activate = (m_totalLoops <= 0 || m_currentLoop < m_totalLoops);
243 if (activate && !m_timer.isActive())
244 m_layer->renderer().setNeedsLayout();
245 else if (!activate && m_timer.isActive())
246 m_timer.stop();
247}

249void RenderMarquee::timerFired()
250{
251 if (m_layer->renderer().view().needsLayout())
252 return;
253
254 auto* scrollableArea = m_layer->scrollableArea();
255 ASSERT(scrollableArea);
256
257 if (m_reset) {
258 m_reset = false;
259 if (isHorizontal())
260 scrollableArea->scrollToXOffset(m_start);
261 else
262 scrollableArea->scrollToYOffset(m_start);
263 return;
264 }
265
266 const RenderStyle& style = m_layer->renderer().style();
267
268 int endPoint = m_end;
269 int range = m_end - m_start;
270 int newPos;
271 if (range == 0)
272 newPos = m_end;
273 else {
274 bool addIncrement = direction() == MarqueeDirection::Up || direction() == MarqueeDirection::Left;
275 bool isReversed = style.marqueeBehavior() == MarqueeBehavior::Alternate && m_currentLoop % 2;
276 if (isReversed) {
277 // We're going in the reverse direction.
278 endPoint = m_start;
279 range = -range;
280 addIncrement = !addIncrement;
281 }
282 bool positive = range > 0;
283 int clientSize = (isHorizontal() ? roundToInt(m_layer->renderBox()->clientWidth()) : roundToInt(m_layer->renderBox()->clientHeight()));
284 int increment = abs(intValueForLength(m_layer->renderer().style().marqueeIncrement(), clientSize));
285 int currentPos = (isHorizontal() ? scrollableArea->scrollOffset().x() : scrollableArea->scrollOffset().y());
286 newPos = currentPos + (addIncrement ? increment : -increment);
287 if (positive)
288 newPos = std::min(newPos, endPoint);
289 else
290 newPos = std::max(newPos, endPoint);
291 }
292
293 if (newPos == endPoint) {
294 m_currentLoop++;
295 if (m_totalLoops > 0 && m_currentLoop >= m_totalLoops)
296 m_timer.stop();
297 else if (style.marqueeBehavior() != MarqueeBehavior::Alternate)
298 m_reset = true;
299 }
300
301 if (isHorizontal())
302 scrollableArea->scrollToXOffset(newPos);
303 else
304 scrollableArea->scrollToYOffset(newPos);
305}
