66RenderMultiColumnSet* RenderMultiColumnFlow::firstMultiColumnSet() const
67{
68 for (RenderObject* sibling = nextSibling(); sibling; sibling = sibling->nextSibling()) {
69 if (is<RenderMultiColumnSet>(*sibling))
70 return downcast<RenderMultiColumnSet>(sibling);
71 }
72 return nullptr;
73}

75RenderMultiColumnSet* RenderMultiColumnFlow::lastMultiColumnSet() const
76{
77 ASSERT(multiColumnBlockFlow());
78
79 for (RenderObject* sibling = multiColumnBlockFlow()->lastChild(); sibling; sibling = sibling->previousSibling()) {
80 if (is<RenderMultiColumnSet>(*sibling))
81 return downcast<RenderMultiColumnSet>(sibling);
82 }
83 return nullptr;
84}

86RenderBox* RenderMultiColumnFlow::firstColumnSetOrSpanner() const
87{
88 if (RenderObject* sibling = nextSibling()) {
89 ASSERT(is<RenderBox>(*sibling));
90 ASSERT(is<RenderMultiColumnSet>(*sibling) || findColumnSpannerPlaceholder(downcast<RenderBox>(sibling)));
91 return downcast<RenderBox>(sibling);
92 }
93 return nullptr;
94}

101RenderBox* RenderMultiColumnFlow::previousColumnSetOrSpannerSiblingOf(const RenderBox* child)

117void RenderMultiColumnFlow::layout()
118{
119 ASSERT(!m_inLayout);
120 m_inLayout = true;
121 m_lastSetWorkedOn = nullptr;
122 if (RenderBox* first = firstColumnSetOrSpanner()) {
123 if (is<RenderMultiColumnSet>(*first)) {
124 m_lastSetWorkedOn = downcast<RenderMultiColumnSet>(first);
125 m_lastSetWorkedOn->beginFlow(this);
126 }
127 }
128 RenderFragmentedFlow::layout();
129 if (RenderMultiColumnSet* lastSet = lastMultiColumnSet()) {
130 if (!nextColumnSetOrSpannerSiblingOf(lastSet))
131 lastSet->endFlow(this, logicalHeight());
132 lastSet->expandToEncompassFragmentedFlowContentsIfNeeded();
133 }
134 m_inLayout = false;
135 m_lastSetWorkedOn = nullptr;
136}

138void RenderMultiColumnFlow::addFragmentToThread(RenderFragmentContainer* fragmentContainer)
139{
140 auto* columnSet = downcast<RenderMultiColumnSet>(fragmentContainer);
141 if (RenderMultiColumnSet* nextSet = columnSet->nextSiblingMultiColumnSet()) {
142 RenderFragmentContainerList::iterator it = m_fragmentList.find(nextSet);
143 ASSERT(it != m_fragmentList.end());
144 m_fragmentList.insertBefore(it, columnSet);
145 } else
146 m_fragmentList.add(columnSet);
147 fragmentContainer->setIsValid(true);
148}

150void RenderMultiColumnFlow::willBeRemovedFromTree(IsInternalMove isInternalMove)
151{
152 // Detach all column sets from the flow thread. Cannot destroy them at this point, since they
153 // are siblings of this object, and there may be pointers to this object's sibling somewhere
154 // further up on the call stack.
155 for (RenderMultiColumnSet* columnSet = firstMultiColumnSet(); columnSet; columnSet = columnSet->nextSiblingMultiColumnSet())
156 columnSet->detachFragment();
157 RenderFragmentedFlow::willBeRemovedFromTree(isInternalMove);
158}

160void RenderMultiColumnFlow::fragmentedFlowDescendantBoxLaidOut(RenderBox* descendant)
161{
162 if (!is<RenderMultiColumnSpannerPlaceholder>(*descendant))
163 return;
164 auto& placeholder = downcast<RenderMultiColumnSpannerPlaceholder>(*descendant);
165 RenderBlock* container = placeholder.containingBlock();
166
167 for (RenderBox* prev = previousColumnSetOrSpannerSiblingOf(placeholder.spanner()); prev; prev = previousColumnSetOrSpannerSiblingOf(prev)) {
168 if (is<RenderMultiColumnSet>(*prev)) {
169 downcast<RenderMultiColumnSet>(*prev).endFlow(container, placeholder.logicalTop());
170 break;
171 }
172 }
173
174 for (RenderBox* next = nextColumnSetOrSpannerSiblingOf(placeholder.spanner()); next; next = nextColumnSetOrSpannerSiblingOf(next)) {
175 if (is<RenderMultiColumnSet>(*next)) {
176 m_lastSetWorkedOn = downcast<RenderMultiColumnSet>(next);
177 m_lastSetWorkedOn->beginFlow(container);
178 break;
179 }
180 }
181}

212RenderFragmentContainer* RenderMultiColumnFlow::fragmentAtBlockOffset(const RenderBox* box, LayoutUnit offset, bool extendLastFragment) const
213{
214 if (!m_inLayout)
215 return RenderFragmentedFlow::fragmentAtBlockOffset(box, offset, extendLastFragment);
216
217 // Layout in progress. We are calculating the set heights as we speak, so the fragment range
218 // information is not up-to-date.
219
220 RenderMultiColumnSet* columnSet = m_lastSetWorkedOn ? m_lastSetWorkedOn : firstMultiColumnSet();
221 if (!columnSet) {
222 // If there's no set, bail. This multicol is empty or only consists of spanners. There
223 // are no fragments.
224 return nullptr;
225 }
226 // The last set worked on is a good guess. But if we're not within the bounds, search for the
227 // right one.
228 if (offset < columnSet->logicalTopInFragmentedFlow()) {
229 do {
230 if (RenderMultiColumnSet* prev = columnSet->previousSiblingMultiColumnSet())
231 columnSet = prev;
232 else
233 break;
234 } while (offset < columnSet->logicalTopInFragmentedFlow());
235 } else {
236 while (offset >= columnSet->logicalBottomInFragmentedFlow()) {
237 RenderMultiColumnSet* next = columnSet->nextSiblingMultiColumnSet();
238 if (!next || !next->hasBeenFlowed())
239 break;
240 columnSet = next;
241 }
242 }
243 return columnSet;
244}

246void RenderMultiColumnFlow::setFragmentRangeForBox(const RenderBox& box, RenderFragmentContainer* startFragment, RenderFragmentContainer* endFragment)
247{
248 // Some column sets may have zero height, which means that two or more sets may start at the
249 // exact same flow thread position, which means that some parts of the code may believe that a
250 // given box lives in sets that it doesn't really live in. Make some adjustments here and
251 // include such sets if they are adjacent to the start and/or end fragments.
252 for (RenderMultiColumnSet* columnSet = downcast<RenderMultiColumnSet>(*startFragment).previousSiblingMultiColumnSet(); columnSet; columnSet = columnSet->previousSiblingMultiColumnSet()) {
253 if (columnSet->logicalHeightInFragmentedFlow())
254 break;
255 startFragment = columnSet;
256 }
257 for (RenderMultiColumnSet* columnSet = downcast<RenderMultiColumnSet>(*startFragment).nextSiblingMultiColumnSet(); columnSet; columnSet = columnSet->nextSiblingMultiColumnSet()) {
258 if (columnSet->logicalHeightInFragmentedFlow())
259 break;
260 endFragment = columnSet;
261 }
262
263 RenderFragmentedFlow::setFragmentRangeForBox(box, startFragment, endFragment);
264}

266bool RenderMultiColumnFlow::addForcedFragmentBreak(const RenderBlock* block, LayoutUnit offset, RenderBox* /*breakChild*/, bool /*isBefore*/, LayoutUnit* offsetBreakAdjustment)
267{
268 if (auto* multicolSet = downcast<RenderMultiColumnSet>(fragmentAtBlockOffset(block, offset))) {
269 multicolSet->addForcedBreak(offset);
270 if (offsetBreakAdjustment)
271 *offsetBreakAdjustment = pageLogicalHeightForOffset(offset) ? pageRemainingLogicalHeightForOffset(offset, IncludePageBoundary) : 0_lu;
272 return true;
273 }
274 return false;

277LayoutSize RenderMultiColumnFlow::offsetFromContainer(RenderElement& enclosingContainer, const LayoutPoint& physicalPoint, bool* offsetDependsOnPoint) const
278{
279 ASSERT(&enclosingContainer == container());
280
281 if (offsetDependsOnPoint)
282 *offsetDependsOnPoint = true;
283
284 LayoutPoint translatedPhysicalPoint(physicalPoint);
285 if (RenderFragmentContainer* fragment = physicalTranslationFromFlowToFragment(translatedPhysicalPoint))
286 translatedPhysicalPoint.moveBy(fragment->topLeftLocation());
287
288 LayoutSize offset(translatedPhysicalPoint.x(), translatedPhysicalPoint.y());
289 if (is<RenderBox>(enclosingContainer))
290 offset -= toLayoutSize(downcast<RenderBox>(enclosingContainer).scrollPosition());
291 return offset;
292}

294void RenderMultiColumnFlow::mapAbsoluteToLocalPoint(MapCoordinatesFlags mode, TransformState& transformState) const
295{
296 // First get the transform state's point into the block flow thread's physical coordinate space.
297 parent()->mapAbsoluteToLocalPoint(mode, transformState);
298 LayoutPoint transformPoint(transformState.mappedPoint());
299
300 // Now walk through each fragment.
301 const RenderMultiColumnSet* candidateColumnSet = nullptr;
302 LayoutPoint candidatePoint;
303 LayoutSize candidateContainerOffset;
304
305 for (const auto& columnSet : childrenOfType<RenderMultiColumnSet>(*parent())) {
306 candidateContainerOffset = columnSet.offsetFromContainer(*parent(), LayoutPoint());
307
308 candidatePoint = transformPoint - candidateContainerOffset;
309 candidateColumnSet = &columnSet;
310
311 // We really have no clue what to do with overflow. We'll just use the closest fragment to the point in that case.
312 LayoutUnit pointOffset = isHorizontalWritingMode() ? candidatePoint.y() : candidatePoint.x();
313 LayoutUnit fragmentOffset = isHorizontalWritingMode() ? columnSet.topLeftLocation().y() : columnSet.topLeftLocation().x();
314 if (pointOffset < fragmentOffset + columnSet.logicalHeight())
315 break;
316 }
317
318 // Once we have a good guess as to which fragment we hit tested through (and yes, this was just a heuristic, but it's
319 // the best we could do), then we can map from the fragment into the flow thread.
320 LayoutSize translationOffset = physicalTranslationFromFragmentToFlow(candidateColumnSet, candidatePoint) + candidateContainerOffset;
321 bool preserve3D = mode & UseTransforms && (parent()->style().preserves3D() || style().preserves3D());
322 if (mode & UseTransforms && shouldUseTransformFromContainer(parent())) {
323 TransformationMatrix t;
324 getTransformFromContainer(parent(), translationOffset, t);
325 transformState.applyTransform(t, preserve3D ? TransformState::AccumulateTransform : TransformState::FlattenTransform);
326 } else
327 transformState.move(translationOffset.width(), translationOffset.height(), preserve3D ? TransformState::AccumulateTransform : TransformState::FlattenTransform);
328}

330LayoutSize RenderMultiColumnFlow::physicalTranslationFromFragmentToFlow(const RenderMultiColumnSet* columnSet, const LayoutPoint& physicalPoint) const
331{
332 LayoutPoint logicalPoint = columnSet->flipForWritingMode(physicalPoint);
333 LayoutPoint translatedPoint = columnSet->translateFragmentPointToFragmentedFlow(logicalPoint);
334 LayoutPoint physicalTranslatedPoint = columnSet->flipForWritingMode(translatedPoint);
335 return physicalPoint - physicalTranslatedPoint;
336}

338RenderFragmentContainer* RenderMultiColumnFlow::mapFromFlowToFragment(TransformState& transformState) const
339{
340 if (!hasValidFragmentInfo())
341 return nullptr;
342
343 // Get back into our local flow thread space.
344 LayoutRect boxRect = transformState.mappedQuad().enclosingBoundingBox();
345 flipForWritingMode(boxRect);
346
347 // FIXME: We need to refactor RenderObject::absoluteQuads to be able to split the quads across fragments,
348 // for now we just take the center of the mapped enclosing box and map it to a column.
349 LayoutPoint centerPoint = boxRect.center();
350 LayoutUnit centerLogicalOffset = isHorizontalWritingMode() ? centerPoint.y() : centerPoint.x();
351 auto* fragmentContainer = fragmentAtBlockOffset(this, centerLogicalOffset, true);
352 if (!fragmentContainer)
353 return nullptr;
354 transformState.move(physicalTranslationOffsetFromFlowToFragment(fragmentContainer, centerLogicalOffset));
355 return fragmentContainer;
356}

358LayoutSize RenderMultiColumnFlow::physicalTranslationOffsetFromFlowToFragment(const RenderFragmentContainer* fragmentContainer, const LayoutUnit logicalOffset) const
359{
360 // Now that we know which multicolumn set we hit, we need to get the appropriate translation offset for the column.
361 const auto* columnSet = downcast<RenderMultiColumnSet>(fragmentContainer);
362 LayoutPoint translationOffset = columnSet->columnTranslationForOffset(logicalOffset);
363
364 // Now we know how we want the rect to be translated into the fragment. At this point we're converting
365 // back to physical coordinates.
366 if (style().isFlippedBlocksWritingMode()) {
367 LayoutRect portionRect(columnSet->fragmentedFlowPortionRect());
368 LayoutRect columnRect = columnSet->columnRectAt(0);
369 LayoutUnit physicalDeltaFromPortionBottom = logicalHeight() - columnSet->logicalBottomInFragmentedFlow();
370 if (isHorizontalWritingMode())
371 columnRect.setHeight(portionRect.height());
372 else
373 columnRect.setWidth(portionRect.width());
374 columnSet->flipForWritingMode(columnRect);
375 if (isHorizontalWritingMode())
376 translationOffset.move(0_lu, columnRect.y() - portionRect.y() - physicalDeltaFromPortionBottom);
377 else
378 translationOffset.move(columnRect.x() - portionRect.x() - physicalDeltaFromPortionBottom, 0_lu);
379 }
380
381 return LayoutSize(translationOffset.x(), translationOffset.y());
382}
383
384RenderFragmentContainer* RenderMultiColumnFlow::physicalTranslationFromFlowToFragment(LayoutPoint& physicalPoint) const
385{
386 if (!hasValidFragmentInfo())
387 return nullptr;
388
389 // Put the physical point into the flow thread's coordinate space.
390 LayoutPoint logicalPoint = flipForWritingMode(physicalPoint);
391
392 // Now get the fragment that we are in.
393 LayoutUnit logicalOffset = isHorizontalWritingMode() ? logicalPoint.y() : logicalPoint.x();
394 RenderFragmentContainer* fragmentContainer = fragmentAtBlockOffset(this, logicalOffset, true);
395 if (!fragmentContainer)
396 return nullptr;
397
398 // Translate to the coordinate space of the fragment.
399 LayoutSize translationOffset = physicalTranslationOffsetFromFlowToFragment(fragmentContainer, logicalOffset);
400
401 // Now shift the physical point into the fragment's coordinate space.
402 physicalPoint += translationOffset;
403
404 return fragmentContainer;
405}

414bool RenderMultiColumnFlow::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
415{
416 // You cannot be inside an in-flow RenderFragmentedFlow without a corresponding DOM node. It's better to
417 // just let the ancestor figure out where we are instead.
418 if (hitTestAction == HitTestBlockBackground)
419 return false;
420 bool inside = RenderFragmentedFlow::nodeAtPoint(request, result, locationInContainer, accumulatedOffset, hitTestAction);
421 if (inside && !result.innerNode())
422 return false;
423 return inside;
424}
