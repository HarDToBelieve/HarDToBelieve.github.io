75RenderObject* RenderMultiColumnSet::firstRendererInFragmentedFlow() const
76{
77 if (RenderBox* sibling = RenderMultiColumnFlow::previousColumnSetOrSpannerSiblingOf(this)) {
78 // Adjacent sets should not occur. Currently we would have no way of figuring out what each
79 // of them contains then.
80 ASSERT(!sibling->isRenderMultiColumnSet());
81 if (RenderMultiColumnSpannerPlaceholder* placeholder = multiColumnFlow()->findColumnSpannerPlaceholder(sibling))
82 return placeholder->nextInPreOrderAfterChildren();
83 ASSERT_NOT_REACHED();
84 }
85 return fragmentedFlow()->firstChild();
86}

88RenderObject* RenderMultiColumnSet::lastRendererInFragmentedFlow() const
89{
90 if (RenderBox* sibling = RenderMultiColumnFlow::nextColumnSetOrSpannerSiblingOf(this)) {
91 // Adjacent sets should not occur. Currently we would have no way of figuring out what each
92 // of them contains then.
93 ASSERT(!sibling->isRenderMultiColumnSet());
94 if (RenderMultiColumnSpannerPlaceholder* placeholder = multiColumnFlow()->findColumnSpannerPlaceholder(sibling))
95 return placeholder->previousInPreOrder();
96 ASSERT_NOT_REACHED();
97 }
98 return fragmentedFlow()->lastLeafChild();
99}

101static bool precedesRenderer(const RenderObject* renderer, const RenderObject* boundary)
102{
103 for (; renderer; renderer = renderer->nextInPreOrder()) {
104 if (renderer == boundary)
105 return true;
106 }
107 return false;
108}
109
110bool RenderMultiColumnSet::containsRendererInFragmentedFlow(const RenderObject& renderer) const
111{
112 if (!previousSiblingMultiColumnSet() && !nextSiblingMultiColumnSet()) {
113 // There is only one set. This is easy, then.
114 return renderer.isDescendantOf(m_fragmentedFlow);
115 }
116
117 RenderObject* firstRenderer = firstRendererInFragmentedFlow();
118 RenderObject* lastRenderer = lastRendererInFragmentedFlow();
119 ASSERT(firstRenderer);
120 ASSERT(lastRenderer);
121
122 // This is SLOW! But luckily very uncommon.
123 return precedesRenderer(firstRenderer, &renderer) && precedesRenderer(&renderer, lastRenderer);
124}

161void RenderMultiColumnSet::setAndConstrainColumnHeight(LayoutUnit newHeight)
162{
163 m_computedColumnHeight = newHeight;
164 if (m_computedColumnHeight > m_maxColumnHeight)
165 m_computedColumnHeight = m_maxColumnHeight;
166
167 // FIXME: The available column height is not the same as the constrained height specified
168 // by the pagination API. The column set in this case is allowed to be bigger than the
169 // height of a single column. We cache available column height in order to use it
170 // in computeLogicalHeight later. This is pretty gross, and maybe there's a better way
171 // to formalize the idea of clamped column heights without having a view dependency
172 // here.
173 m_availableColumnHeight = m_computedColumnHeight;
174 if (multiColumnFlow() && !multiColumnFlow()->progressionIsInline() && parent()->isRenderView()) {
175 int pageLength = view().frameView().pagination().pageLength;
176 if (pageLength)
177 m_computedColumnHeight = pageLength;
178 }
179
180 m_columnHeightComputed = true;
181
182 // FIXME: the height may also be affected by the enclosing pagination context, if any.
183}

185unsigned RenderMultiColumnSet::findRunWithTallestColumns() const
186{
187 unsigned indexWithLargestHeight = 0;
188 LayoutUnit largestHeight;
189 LayoutUnit previousOffset;
190 size_t runCount = m_contentRuns.size();
191 ASSERT(runCount);
192 for (size_t i = 0; i < runCount; i++) {
193 const ContentRun& run = m_contentRuns[i];
194 LayoutUnit height = run.columnLogicalHeight(previousOffset);
195 if (largestHeight < height) {
196 largestHeight = height;
197 indexWithLargestHeight = i;
198 }
199 previousOffset = run.breakOffset();
200 }
201 return indexWithLargestHeight;
202}

204void RenderMultiColumnSet::distributeImplicitBreaks()
205{
206#ifndef NDEBUG
207 // There should be no implicit breaks assumed at this point.
208 for (unsigned i = 0; i < forcedBreaksCount(); i++)
209 ASSERT(!m_contentRuns[i].assumedImplicitBreaks());
210#endif // NDEBUG
211
212 // Insert a final content run to encompass all content. This will include overflow if this is
213 // the last set.
214 addForcedBreak(logicalBottomInFragmentedFlow());
215 unsigned breakCount = forcedBreaksCount();
216
217 // If there is room for more breaks (to reach the used value of column-count), imagine that we
218 // insert implicit breaks at suitable locations. At any given time, the content run with the
219 // currently tallest columns will get another implicit break "inserted", which will increase its
220 // column count by one and shrink its columns' height. Repeat until we have the desired total
221 // number of breaks. The largest column height among the runs will then be the initial column
222 // height for the balancer to use.
223 while (breakCount < m_computedColumnCount) {
224 unsigned index = findRunWithTallestColumns();
225 m_contentRuns[index].assumeAnotherImplicitBreak();
226 breakCount++;
227 }
228}

230LayoutUnit RenderMultiColumnSet::calculateBalancedHeight(bool initial) const
231{
232 if (initial) {
233 // Start with the lowest imaginable column height.
234 unsigned index = findRunWithTallestColumns();
235 LayoutUnit startOffset = index > 0 ? m_contentRuns[index - 1].breakOffset() : logicalTopInFragmentedFlow();
236 return std::max<LayoutUnit>(m_contentRuns[index].columnLogicalHeight(startOffset), m_minimumColumnHeight);
237 }
238
239 LayoutUnit sizeContainmentShortage = std::max<LayoutUnit>(LayoutUnit(), m_spaceShortageForSizeContainment);
240 if (columnCount() <= computedColumnCount()) {
241 // With the current column height, the content fits without creating overflowing columns. We're done.
242 return m_computedColumnHeight + sizeContainmentShortage;
243 }
244
245 if (forcedBreaksCount() >= computedColumnCount()) {
246 // Too many forced breaks to allow any implicit breaks. Initial balancing should already
247 // have set a good height. There's nothing more we should do.
248 return m_computedColumnHeight + sizeContainmentShortage;
249 }
250
251 // If the initial guessed column height wasn't enough, stretch it now. Stretch by the lowest
252 // amount of space shortage found during layout.
253
254 ASSERT(m_minSpaceShortage > 0); // We should never _shrink_ the height!
255 // ASSERT(m_minSpaceShortage != RenderFragmentedFlow::maxLogicalHeight()); // If this happens, we probably have a bug.
256 if (m_minSpaceShortage == RenderFragmentedFlow::maxLogicalHeight())
257 return m_computedColumnHeight + sizeContainmentShortage; // So bail out rather than looping infinitely.
258
259 auto toAdd = std::max<LayoutUnit>(sizeContainmentShortage, m_minSpaceShortage);
260 return m_computedColumnHeight + toAdd;
261}

265 m_contentRuns.clear();

268void RenderMultiColumnSet::addForcedBreak(LayoutUnit offsetFromFirstPage)
269{
270 if (!requiresBalancing())
271 return;
272 if (!m_contentRuns.isEmpty() && offsetFromFirstPage <= m_contentRuns.last().breakOffset())
273 return;
274 // Append another item as long as we haven't exceeded used column count. What ends up in the
275 // overflow area shouldn't affect column balancing.
276 if (m_contentRuns.size() < m_computedColumnCount)
277 m_contentRuns.append(ContentRun(offsetFromFirstPage));
278}

280bool RenderMultiColumnSet::recalculateColumnHeight(bool initial)
281{
282 LayoutUnit oldColumnHeight = m_computedColumnHeight;
283 if (requiresBalancing()) {
284 if (initial)
285 distributeImplicitBreaks();
286 LayoutUnit newColumnHeight = calculateBalancedHeight(initial);
287 setAndConstrainColumnHeight(newColumnHeight);
288 // After having calculated an initial column height, the multicol container typically needs at
289 // least one more layout pass with a new column height, but if a height was specified, we only
290 // need to do this if we think that we need less space than specified. Conversely, if we
291 // determined that the columns need to be as tall as the specified height of the container, we
292 // have already laid it out correctly, and there's no need for another pass.
293 } else {
294 // The position of the column set may have changed, in which case height available for
295 // columns may have changed as well.
296 setAndConstrainColumnHeight(m_computedColumnHeight);
297 }
298 if (m_computedColumnHeight == oldColumnHeight)
299 return false; // No change. We're done.
300
301 m_minSpaceShortage = RenderFragmentedFlow::maxLogicalHeight();
302 return true; // Need another pass.
303}
304
305void RenderMultiColumnSet::recordSpaceShortage(LayoutUnit spaceShortage)
306{
307 if (spaceShortage >= m_minSpaceShortage)
308 return;
309
310 // The space shortage is what we use as our stretch amount. We need a positive number here in
311 // order to get anywhere. Some lines actually have zero height. Ignore them.
312 if (spaceShortage > 0)
313 m_minSpaceShortage = spaceShortage;
314}

325bool RenderMultiColumnSet::requiresBalancing() const
326{
327 if (!multiColumnFlow()->progressionIsInline())
328 return false;
329
330 if (RenderBox* next = RenderMultiColumnFlow::nextColumnSetOrSpannerSiblingOf(this)) {
331 if (!next->isRenderMultiColumnSet() && !next->isLegend()) {
332 // If we're followed by a spanner, we need to balance.
333 ASSERT(multiColumnFlow()->findColumnSpannerPlaceholder(next));
334 return true;
335 }
336 }
337 RenderBlockFlow* container = multiColumnBlockFlow();
338 if (container->style().columnFill() == ColumnFill::Balance)
339 return true;
340 return !multiColumnFlow()->columnHeightAvailable();
341}

343void RenderMultiColumnSet::prepareForLayout(bool initial)
344{
345 // Guess box logical top. This might eliminate the need for another layout pass.
346 if (RenderBox* previous = RenderMultiColumnFlow::previousColumnSetOrSpannerSiblingOf(this))
347 setLogicalTop(previous->logicalBottom() + previous->marginAfter());
348 else
349 setLogicalTop(multiColumnBlockFlow()->borderAndPaddingBefore());
350
351 if (initial)
352 m_maxColumnHeight = calculateMaxColumnHeight();
353 if (requiresBalancing()) {
354 if (initial) {
355 m_computedColumnHeight = 0;
356 m_availableColumnHeight = 0;
357 m_columnHeightComputed = false;
358 }
359 } else
360 setAndConstrainColumnHeight(heightAdjustedForSetOffset(multiColumnFlow()->columnHeightAvailable()));
361
362 // Set box width.
363 updateLogicalWidth();
364
365 // Any breaks will be re-inserted during layout, so get rid of what we already have.
366 clearForcedBreaks();
367
368 // Nuke previously stored minimum column height. Contents may have changed for all we know.
369 m_minimumColumnHeight = 0;
370
371 m_spaceShortageForSizeContainment = 0;
372
373 // Start with "infinite" flow thread portion height until height is known.
374 setLogicalBottomInFragmentedFlow(RenderFragmentedFlow::maxLogicalHeight());
375
376 setNeedsLayout(MarkOnlyThis);
377}

379void RenderMultiColumnSet::beginFlow(RenderBlock* container)
380{
381 RenderMultiColumnFlow* fragmentedFlow = multiColumnFlow();
382
383 // At this point layout is exactly at the beginning of this set. Store block offset from flow
384 // thread start.
385 LayoutUnit logicalTopInFragmentedFlow = fragmentedFlow->offsetFromLogicalTopOfFirstFragment(container) + container->logicalHeight();
386 setLogicalTopInFragmentedFlow(logicalTopInFragmentedFlow);
387}
388
389void RenderMultiColumnSet::endFlow(RenderBlock* container, LayoutUnit bottomInContainer)
390{
391 RenderMultiColumnFlow* fragmentedFlow = multiColumnFlow();
392
393 // At this point layout is exactly at the end of this set. Store block offset from flow thread
394 // start. Also note that a new column height may have affected the height used in the flow
395 // thread (because of struts), which may affect the number of columns. So we also need to update
396 // the flow thread portion height in order to be able to calculate actual column-count.
397 LayoutUnit logicalBottomInFragmentedFlow = fragmentedFlow->offsetFromLogicalTopOfFirstFragment(container) + bottomInContainer;
398 setLogicalBottomInFragmentedFlow(logicalBottomInFragmentedFlow);
399 container->setLogicalHeight(bottomInContainer);
400}

402void RenderMultiColumnSet::layout()
403{
404 RenderBlockFlow::layout();
405
406 // At this point the logical top and bottom of the column set are known. Update maximum column
407 // height (multicol height may be constrained).
408 m_maxColumnHeight = calculateMaxColumnHeight();
409
410 if (!nextSiblingMultiColumnSet()) {
411 // This is the last set, i.e. the last fragment. Seize the opportunity to validate them.
412 multiColumnFlow()->validateFragments();
413 }
414}

421LayoutUnit RenderMultiColumnSet::calculateMaxColumnHeight() const
422{
423 RenderBlockFlow* multicolBlock = multiColumnBlockFlow();
424 const RenderStyle& multicolStyle = multicolBlock->style();
425 LayoutUnit availableHeight = multiColumnFlow()->columnHeightAvailable();
426 LayoutUnit maxColumnHeight = availableHeight ? availableHeight : RenderFragmentedFlow::maxLogicalHeight();
427 if (!multicolStyle.logicalMaxHeight().isUndefined())
428 maxColumnHeight = std::min(maxColumnHeight, multicolBlock->computeContentLogicalHeight(MaxSize, multicolStyle.logicalMaxHeight(), std::nullopt).value_or(maxColumnHeight));
429 return heightAdjustedForSetOffset(maxColumnHeight);
430}

432LayoutUnit RenderMultiColumnSet::columnGap() const
433{
434 // FIXME: Eventually we will cache the column gap when the widths of columns start varying, but for now we just
435 // go to the parent block to get the gap.
436 RenderBlockFlow& parentBlock = downcast<RenderBlockFlow>(*parent());
437 if (parentBlock.style().columnGap().isNormal())
438 return parentBlock.style().fontDescription().computedPixelSize(); // "1em" is recommended as the normal gap setting. Matches <p> margins.
439 return valueForLength(parentBlock.style().columnGap().length(), parentBlock.availableLogicalWidth());
440}

442unsigned RenderMultiColumnSet::columnCount() const
443{
444 // We must always return a value of 1 or greater. Column count = 0 is a meaningless situation,
445 // and will confuse and cause problems in other parts of the code.
446 auto computedColumnHeight = this->computedColumnHeight();
447 if (computedColumnHeight <= 0)
448 return 1;
449
450 // Our portion rect determines our column count. We have as many columns as needed to fit all the content.
451 auto logicalHeightInColumns = fragmentedFlow()->isHorizontalWritingMode() ? fragmentedFlowPortionRect().height() : fragmentedFlowPortionRect().width();
452 if (logicalHeightInColumns <= 0)
453 return 1;
454
455 unsigned count = ceilf(static_cast<float>(logicalHeightInColumns) / computedColumnHeight);
456 ASSERT(count >= 1);
457 return count;
458}

460LayoutUnit RenderMultiColumnSet::columnLogicalLeft(unsigned index) const
461{
462 LayoutUnit colLogicalWidth = computedColumnWidth();
463 LayoutUnit colLogicalLeft = borderAndPaddingLogicalLeft();
464 LayoutUnit colGap = columnGap();
465
466 bool progressionReversed = multiColumnFlow()->progressionIsReversed();
467 bool progressionInline = multiColumnFlow()->progressionIsInline();
468
469 if (progressionInline) {
470 if (style().isLeftToRightDirection() ^ progressionReversed)
471 colLogicalLeft += index * (colLogicalWidth + colGap);
472 else
473 colLogicalLeft += contentLogicalWidth() - colLogicalWidth - index * (colLogicalWidth + colGap);
474 }
475
476 return colLogicalLeft;
477}

479LayoutUnit RenderMultiColumnSet::columnLogicalTop(unsigned index) const
480{
481 LayoutUnit colLogicalHeight = computedColumnHeight();
482 LayoutUnit colLogicalTop = borderAndPaddingBefore();
483 LayoutUnit colGap = columnGap();
484
485 bool progressionReversed = multiColumnFlow()->progressionIsReversed();
486 bool progressionInline = multiColumnFlow()->progressionIsInline();
487
488 if (!progressionInline) {
489 if (!progressionReversed)
490 colLogicalTop += index * (colLogicalHeight + colGap);
491 else
492 colLogicalTop += contentLogicalHeight() - colLogicalHeight - index * (colLogicalHeight + colGap);
493 }
494
495 return colLogicalTop;
496}

508unsigned RenderMultiColumnSet::columnIndexAtOffset(LayoutUnit offset, ColumnIndexCalculationMode mode) const
509{
510 LayoutRect portionRect(fragmentedFlowPortionRect());
511
512 // Handle the offset being out of range.
513 LayoutUnit fragmentedFlowLogicalTop = isHorizontalWritingMode() ? portionRect.y() : portionRect.x();
514 if (offset < fragmentedFlowLogicalTop)
515 return 0;
516 // If we're laying out right now, we cannot constrain against some logical bottom, since it
517 // isn't known yet. Otherwise, just return the last column if we're past the logical bottom.
518 if (mode == ClampToExistingColumns) {
519 LayoutUnit fragmentedFlowLogicalBottom = isHorizontalWritingMode() ? portionRect.maxY() : portionRect.maxX();
520 if (offset >= fragmentedFlowLogicalBottom)
521 return columnCount() - 1;
522 }
523
524 // Sometimes computedColumnHeight() is 0 here: see https://bugs.webkit.org/show_bug.cgi?id=132884
525 if (!computedColumnHeight())
526 return 0;
527
528 // Just divide by the column height to determine the correct column.
529 return static_cast<float>(offset - fragmentedFlowLogicalTop) / computedColumnHeight();
530}

532LayoutRect RenderMultiColumnSet::fragmentedFlowPortionRectAt(unsigned index) const
533{
534 LayoutRect portionRect = fragmentedFlowPortionRect();
535 if (isHorizontalWritingMode())
536 portionRect = LayoutRect(portionRect.x(), portionRect.y() + index * computedColumnHeight(), portionRect.width(), computedColumnHeight());
537 else
538 portionRect = LayoutRect(portionRect.x() + index * computedColumnHeight(), portionRect.y(), computedColumnHeight(), portionRect.height());
539 return portionRect;
540}

542LayoutRect RenderMultiColumnSet::fragmentedFlowPortionOverflowRect(const LayoutRect& portionRect, unsigned index, unsigned colCount, LayoutUnit colGap)
543{
544 // This function determines the portion of the flow thread that paints for the column. Along the inline axis, columns are
545 // unclipped at outside edges (i.e., the first and last column in the set), and they clip to half the column
546 // gap along interior edges.
547 //
548 // In the block direction, we will not clip overflow out of the top of the first column, or out of the bottom of
549 // the last column. This applies only to the true first column and last column across all column sets.
550 //
551 // FIXME: Eventually we will know overflow on a per-column basis, but we can't do this until we have a painting
552 // mode that understands not to paint contents from a previous column in the overflow area of a following column.
553 // This problem applies to fragments and pages as well and is not unique to columns.
554
555 bool progressionReversed = multiColumnFlow()->progressionIsReversed();
556
557 bool isFirstColumn = !index;
558 bool isLastColumn = index == colCount - 1;
559 bool isLeftmostColumn = style().isLeftToRightDirection() ^ progressionReversed ? isFirstColumn : isLastColumn;
560 bool isRightmostColumn = style().isLeftToRightDirection() ^ progressionReversed ? isLastColumn : isFirstColumn;
561
562 // Calculate the overflow rectangle, based on the flow thread's, clipped at column logical
563 // top/bottom unless it's the first/last column.
564 LayoutRect overflowRect = overflowRectForFragmentedFlowPortion(portionRect, isFirstColumn && isFirstFragment(), isLastColumn && isLastFragment(), VisualOverflow);
565
566 // For RenderViews only (i.e., iBooks), avoid overflowing into neighboring columns, by clipping in the middle of adjacent column gaps. Also make sure that we avoid rounding errors.
567 if (&view() == parent()) {
568 if (isHorizontalWritingMode()) {
569 if (!isLeftmostColumn)
570 overflowRect.shiftXEdgeTo(portionRect.x() - colGap / 2);
571 if (!isRightmostColumn)
572 overflowRect.shiftMaxXEdgeTo(portionRect.maxX() + colGap - colGap / 2);
573 } else {
574 if (!isLeftmostColumn)
575 overflowRect.shiftYEdgeTo(portionRect.y() - colGap / 2);
576 if (!isRightmostColumn)
577 overflowRect.shiftMaxYEdgeTo(portionRect.maxY() + colGap - colGap / 2);
578 }
579 }
580 return overflowRect;
581}

583void RenderMultiColumnSet::paintColumnRules(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
584{
585 if (paintInfo.context().paintingDisabled())
586 return;
587
588 RenderMultiColumnFlow* fragmentedFlow = multiColumnFlow();
589 const RenderStyle& blockStyle = parent()->style();
590 const Color& ruleColor = blockStyle.visitedDependentColorWithColorFilter(CSSPropertyColumnRuleColor);
591 bool ruleTransparent = blockStyle.columnRuleIsTransparent();
592 BorderStyle ruleStyle = collapsedBorderStyle(blockStyle.columnRuleStyle());
593 LayoutUnit ruleThickness { blockStyle.columnRuleWidth() };
594 LayoutUnit colGap = columnGap();
595 bool renderRule = ruleStyle > BorderStyle::Hidden && !ruleTransparent;
596 if (!renderRule)
597 return;
598
599 unsigned colCount = columnCount();
600 if (colCount <= 1)
601 return;
602
603 bool antialias = shouldAntialiasLines(paintInfo.context());
604
605 if (fragmentedFlow->progressionIsInline()) {
606 bool leftToRight = style().isLeftToRightDirection() ^ fragmentedFlow->progressionIsReversed();
607 LayoutUnit currLogicalLeftOffset = leftToRight ? 0_lu : contentLogicalWidth();
608 LayoutUnit ruleAdd = logicalLeftOffsetForContent();
609 LayoutUnit ruleLogicalLeft = leftToRight ? 0_lu : contentLogicalWidth();
610 LayoutUnit inlineDirectionSize = computedColumnWidth();
611 BoxSide boxSide = isHorizontalWritingMode()
612 ? leftToRight ? BoxSide::Left : BoxSide::Right
613 : leftToRight ? BoxSide::Top : BoxSide::Bottom;
614
615 for (unsigned i = 0; i < colCount; i++) {
616 // Move to the next position.
617 if (leftToRight) {
618 ruleLogicalLeft += inlineDirectionSize + colGap / 2;
619 currLogicalLeftOffset += inlineDirectionSize + colGap;
620 } else {
621 ruleLogicalLeft -= (inlineDirectionSize + colGap / 2);
622 currLogicalLeftOffset -= (inlineDirectionSize + colGap);
623 }
624
625 // Now paint the column rule.
626 if (i < colCount - 1) {
627 LayoutUnit ruleLeft = isHorizontalWritingMode() ? paintOffset.x() + ruleLogicalLeft - ruleThickness / 2 + ruleAdd : paintOffset.x() + borderLeft() + paddingLeft();
628 LayoutUnit ruleRight = isHorizontalWritingMode() ? ruleLeft + ruleThickness : ruleLeft + contentWidth();
629 LayoutUnit ruleTop = isHorizontalWritingMode() ? paintOffset.y() + borderTop() + paddingTop() : paintOffset.y() + ruleLogicalLeft - ruleThickness / 2 + ruleAdd;
630 LayoutUnit ruleBottom = isHorizontalWritingMode() ? ruleTop + contentHeight() : ruleTop + ruleThickness;
631 IntRect pixelSnappedRuleRect = snappedIntRect(ruleLeft, ruleTop, ruleRight - ruleLeft, ruleBottom - ruleTop);
632 drawLineForBoxSide(paintInfo.context(), pixelSnappedRuleRect, boxSide, ruleColor, ruleStyle, 0, 0, antialias);
633 }
634
635 ruleLogicalLeft = currLogicalLeftOffset;
636 }
637 } else {
638 bool topToBottom = !style().isFlippedBlocksWritingMode() ^ fragmentedFlow->progressionIsReversed();
639 LayoutUnit ruleLeft = isHorizontalWritingMode() ? 0_lu : colGap / 2 - colGap - ruleThickness / 2;
640 LayoutUnit ruleWidth = isHorizontalWritingMode() ? contentWidth() : ruleThickness;
641 LayoutUnit ruleTop = isHorizontalWritingMode() ? colGap / 2 - colGap - ruleThickness / 2 : 0_lu;
642 LayoutUnit ruleHeight = isHorizontalWritingMode() ? ruleThickness : contentHeight();
643 LayoutRect ruleRect(ruleLeft, ruleTop, ruleWidth, ruleHeight);
644
645 if (!topToBottom) {
646 if (isHorizontalWritingMode())
647 ruleRect.setY(height() - ruleRect.maxY());
648 else
649 ruleRect.setX(width() - ruleRect.maxX());
650 }
651
652 ruleRect.moveBy(paintOffset);
653
654 BoxSide boxSide = isHorizontalWritingMode() ? topToBottom ? BoxSide::Top : BoxSide::Bottom : topToBottom ? BoxSide::Left : BoxSide::Right;
655
656 LayoutSize step(0_lu, topToBottom ? computedColumnHeight() + colGap : -(computedColumnHeight() + colGap));
657 if (!isHorizontalWritingMode())
658 step = step.transposedSize();
659
660 for (unsigned i = 1; i < colCount; i++) {
661 ruleRect.move(step);
662 IntRect pixelSnappedRuleRect = snappedIntRect(ruleRect);
663 drawLineForBoxSide(paintInfo.context(), pixelSnappedRuleRect, boxSide, ruleColor, ruleStyle, 0, 0, antialias);
664 }
665 }
666}

668void RenderMultiColumnSet::repaintFragmentedFlowContent(const LayoutRect& repaintRect)
669{
670 // Figure out the start and end columns and only check within that range so that we don't walk the
671 // entire column set. Put the repaint rect into flow thread coordinates by flipping it first.
672 LayoutRect fragmentedFlowRepaintRect(repaintRect);
673 fragmentedFlow()->flipForWritingMode(fragmentedFlowRepaintRect);
674
675 // Now we can compare this rect with the flow thread portions owned by each column. First let's
676 // just see if the repaint rect intersects our flow thread portion at all.
677 LayoutRect clippedRect(fragmentedFlowRepaintRect);
678 clippedRect.intersect(RenderFragmentContainer::fragmentedFlowPortionOverflowRect());
679 if (clippedRect.isEmpty())
680 return;
681
682 // Now we know we intersect at least one column. Let's figure out the logical top and logical
683 // bottom of the area we're repainting.
684 LayoutUnit repaintLogicalTop = isHorizontalWritingMode() ? fragmentedFlowRepaintRect.y() : fragmentedFlowRepaintRect.x();
685 LayoutUnit repaintLogicalBottom = (isHorizontalWritingMode() ? fragmentedFlowRepaintRect.maxY() : fragmentedFlowRepaintRect.maxX()) - 1;
686
687 unsigned startColumn = columnIndexAtOffset(repaintLogicalTop);
688 unsigned endColumn = columnIndexAtOffset(repaintLogicalBottom);
689
690 LayoutUnit colGap = columnGap();
691 unsigned colCount = columnCount();
692 for (unsigned i = startColumn; i <= endColumn; i++) {
693 LayoutRect colRect = columnRectAt(i);
694
695 // Get the portion of the flow thread that corresponds to this column.
696 LayoutRect fragmentedFlowPortion = fragmentedFlowPortionRectAt(i);
697
698 // Now get the overflow rect that corresponds to the column.
699 LayoutRect fragmentedFlowOverflowPortion = fragmentedFlowPortionOverflowRect(fragmentedFlowPortion, i, colCount, colGap);
700
701 // Do a repaint for this specific column.
702 flipForWritingMode(colRect);
703 repaintFragmentedFlowContentRectangle(repaintRect, fragmentedFlowPortion, colRect.location(), &fragmentedFlowOverflowPortion);
704 }

720void RenderMultiColumnSet::collectLayerFragments(LayerFragments& fragments, const LayoutRect& layerBoundingBox, const LayoutRect& dirtyRect)
721{
722 // Let's start by introducing the different coordinate systems involved here. They are different
723 // in how they deal with writing modes and columns. RenderLayer rectangles tend to be more
724 // physical than the rectangles used in RenderObject & co.
725 //
726 // The two rectangles passed to this method are physical, except that we pretend that there's
727 // only one long column (that's the flow thread). They are relative to the top left corner of
728 // the flow thread. All rectangles being compared to the dirty rect also need to be in this
729 // coordinate system.
730 //
731 // Then there's the output from this method - the stuff we put into the list of fragments. The
732 // translationOffset point is the actual physical translation required to get from a location in
733 // the flow thread to a location in some column. The paginationClip rectangle is in the same
734 // coordinate system as the two rectangles passed to this method (i.e. physical, in flow thread
735 // coordinates, pretending that there's only one long column).
736 //
737 // All other rectangles in this method are slightly less physical, when it comes to how they are
738 // used with different writing modes, but they aren't really logical either. They are just like
739 // RenderBox::frameRect(). More precisely, the sizes are physical, and the inline direction
740 // coordinate is too, but the block direction coordinate is always "logical top". These
741 // rectangles also pretend that there's only one long column, i.e. they are for the flow thread.
742 //
743 // To sum up: input and output from this method are "physical" RenderLayer-style rectangles and
744 // points, while inside this method we mostly use the RenderObject-style rectangles (with the
745 // block direction coordinate always being logical top).
746
747 // Put the layer bounds into flow thread-local coordinates by flipping it first. Since we're in
748 // a renderer, most rectangles are represented this way.
749 LayoutRect layerBoundsInFragmentedFlow(layerBoundingBox);
750 fragmentedFlow()->flipForWritingMode(layerBoundsInFragmentedFlow);
751
752 // Now we can compare with the flow thread portions owned by each column. First let's
753 // see if the rect intersects our flow thread portion at all.
754 LayoutRect clippedRect(layerBoundsInFragmentedFlow);
755 clippedRect.intersect(RenderFragmentContainer::fragmentedFlowPortionOverflowRect());
756 if (clippedRect.isEmpty())
757 return;
758
759 // Now we know we intersect at least one column. Let's figure out the logical top and logical
760 // bottom of the area we're checking.
761 LayoutUnit layerLogicalTop = isHorizontalWritingMode() ? layerBoundsInFragmentedFlow.y() : layerBoundsInFragmentedFlow.x();
762 LayoutUnit layerLogicalBottom = (isHorizontalWritingMode() ? layerBoundsInFragmentedFlow.maxY() : layerBoundsInFragmentedFlow.maxX()) - 1;
763
764 // Figure out the start and end columns and only check within that range so that we don't walk the
765 // entire column set.
766 unsigned startColumn = columnIndexAtOffset(layerLogicalTop);
767 unsigned endColumn = columnIndexAtOffset(layerLogicalBottom);
768
769 LayoutUnit colLogicalWidth = computedColumnWidth();
770 LayoutUnit colGap = columnGap();
771 unsigned colCount = columnCount();
772
773 bool progressionReversed = multiColumnFlow()->progressionIsReversed();
774 bool progressionIsInline = multiColumnFlow()->progressionIsInline();
775
776 LayoutUnit initialBlockOffset = initialBlockOffsetForPainting();
777
778 for (unsigned i = startColumn; i <= endColumn; i++) {
779 // Get the portion of the flow thread that corresponds to this column.
780 LayoutRect fragmentedFlowPortion = fragmentedFlowPortionRectAt(i);
781
782 // Now get the overflow rect that corresponds to the column.
783 LayoutRect fragmentedFlowOverflowPortion = fragmentedFlowPortionOverflowRect(fragmentedFlowPortion, i, colCount, colGap);
784
785 // In order to create a fragment we must intersect the portion painted by this column.
786 LayoutRect clippedRect(layerBoundsInFragmentedFlow);
787 clippedRect.intersect(fragmentedFlowOverflowPortion);
788 if (clippedRect.isEmpty())
789 continue;
790
791 // We also need to intersect the dirty rect. We have to apply a translation and shift based off
792 // our column index.
793 LayoutSize translationOffset;
794 LayoutUnit inlineOffset = progressionIsInline ? i * (colLogicalWidth + colGap) : 0_lu;
795
796 bool leftToRight = style().isLeftToRightDirection() ^ progressionReversed;
797 if (!leftToRight) {
798 inlineOffset = -inlineOffset;
799 if (progressionReversed)
800 inlineOffset += contentLogicalWidth() - colLogicalWidth;
801 }
802 translationOffset.setWidth(inlineOffset);
803
804 LayoutUnit blockOffset = initialBlockOffset + logicalTop() - fragmentedFlow()->logicalTop() + (isHorizontalWritingMode() ? -fragmentedFlowPortion.y() : -fragmentedFlowPortion.x());
805 if (!progressionIsInline) {
806 if (!progressionReversed)
807 blockOffset = i * colGap;
808 else
809 blockOffset -= i * (computedColumnHeight() + colGap);
810 }
811 if (isFlippedWritingMode(style().writingMode()))
812 blockOffset = -blockOffset;
813 translationOffset.setHeight(blockOffset);
814 if (!isHorizontalWritingMode())
815 translationOffset = translationOffset.transposedSize();
816
817 // Shift the dirty rect to be in flow thread coordinates with this translation applied.
818 LayoutRect translatedDirtyRect(dirtyRect);
819 translatedDirtyRect.move(-translationOffset);
820
821 // See if we intersect the dirty rect.
822 clippedRect = layerBoundingBox;
823 clippedRect.intersect(translatedDirtyRect);
824 if (clippedRect.isEmpty())
825 continue;
826
827 // Something does need to paint in this column. Make a fragment now and supply the physical translation
828 // offset and the clip rect for the column with that offset applied.
829 LayerFragment fragment;
830 fragment.paginationOffset = translationOffset;
831
832 LayoutRect flippedFragmentedFlowOverflowPortion(fragmentedFlowOverflowPortion);
833 // Flip it into more a physical (RenderLayer-style) rectangle.
834 fragmentedFlow()->flipForWritingMode(flippedFragmentedFlowOverflowPortion);
835 fragment.paginationClip = flippedFragmentedFlowOverflowPortion;
836 fragments.append(fragment);
837 }

840LayoutPoint RenderMultiColumnSet::columnTranslationForOffset(const LayoutUnit& offset) const
841{
842 unsigned startColumn = columnIndexAtOffset(offset);
843
844 LayoutUnit colGap = columnGap();
845
846 LayoutRect fragmentedFlowPortion = fragmentedFlowPortionRectAt(startColumn);
847 LayoutPoint translationOffset;
848
849 bool progressionReversed = multiColumnFlow()->progressionIsReversed();
850 bool progressionIsInline = multiColumnFlow()->progressionIsInline();
851
852 LayoutUnit initialBlockOffset = initialBlockOffsetForPainting();
853
854 translationOffset.setX(columnLogicalLeft(startColumn));
855
856 LayoutUnit blockOffset = initialBlockOffset - (isHorizontalWritingMode() ? fragmentedFlowPortion.y() : fragmentedFlowPortion.x());
857 if (!progressionIsInline) {
858 if (!progressionReversed)
859 blockOffset = startColumn * colGap;
860 else
861 blockOffset -= startColumn * (computedColumnHeight() + colGap);
862 }
863 if (isFlippedWritingMode(style().writingMode()))
864 blockOffset = -blockOffset;
865 translationOffset.setY(blockOffset);
866
867 if (!isHorizontalWritingMode())
868 translationOffset = translationOffset.transposedPoint();
869
870 return translationOffset;
871}

897LayoutPoint RenderMultiColumnSet::translateFragmentPointToFragmentedFlow(const LayoutPoint & logicalPoint, ColumnHitTestTranslationMode clampMode) const
898{
899 // Determine which columns we intersect.
900 LayoutUnit colGap = columnGap();
901 LayoutUnit halfColGap = colGap / 2;
902
903 bool progressionIsInline = multiColumnFlow()->progressionIsInline();
904
905 LayoutPoint point = logicalPoint;
906
907 for (unsigned i = 0; i < columnCount(); i++) {
908 // Add in half the column gap to the left and right of the rect.
909 LayoutRect colRect = columnRectAt(i);
910 if (isHorizontalWritingMode() == progressionIsInline) {
911 LayoutRect gapAndColumnRect(colRect.x() - halfColGap, colRect.y(), colRect.width() + colGap, colRect.height());
912 if (point.x() >= gapAndColumnRect.x() && point.x() < gapAndColumnRect.maxX()) {
913 if (clampMode == ClampHitTestTranslationToColumns) {
914 if (progressionIsInline) {
915 // FIXME: The clamping that follows is not completely right for right-to-left
916 // content.
917 // Clamp everything above the column to its top left.
918 if (point.y() < gapAndColumnRect.y())
919 point = gapAndColumnRect.location();
920 // Clamp everything below the column to the next column's top left. If there is
921 // no next column, this still maps to just after this column.
922 else if (point.y() >= gapAndColumnRect.maxY()) {
923 point = gapAndColumnRect.location();
924 point.move(0_lu, gapAndColumnRect.height());
925 }
926 } else {
927 if (point.x() < colRect.x())
928 point.setX(colRect.x());
929 else if (point.x() >= colRect.maxX())
930 point.setX(colRect.maxX() - 1);
931 }
932 }
933
934 LayoutSize offsetInColumn = point - colRect.location();
935 LayoutRect fragmentedFlowPortion = fragmentedFlowPortionRectAt(i);
936
937 return fragmentedFlowPortion.location() + offsetInColumn;
938 }
939 } else {
940 LayoutRect gapAndColumnRect(colRect.x(), colRect.y() - halfColGap, colRect.width(), colRect.height() + colGap);
941 if (point.y() >= gapAndColumnRect.y() && point.y() < gapAndColumnRect.maxY()) {
942 if (clampMode == ClampHitTestTranslationToColumns) {
943 if (progressionIsInline) {
944 // FIXME: The clamping that follows is not completely right for right-to-left
945 // content.
946 // Clamp everything above the column to its top left.
947 if (point.x() < gapAndColumnRect.x())
948 point = gapAndColumnRect.location();
949 // Clamp everything below the column to the next column's top left. If there is
950 // no next column, this still maps to just after this column.
951 else if (point.x() >= gapAndColumnRect.maxX()) {
952 point = gapAndColumnRect.location();
953 point.move(gapAndColumnRect.width(), 0_lu);
954 }
955 } else {
956 if (point.y() < colRect.y())
957 point.setY(colRect.y());
958 else if (point.y() >= colRect.maxY())
959 point.setY(colRect.maxY() - 1);
960 }
961 }
962
963 LayoutSize offsetInColumn = point - colRect.location();
964 LayoutRect fragmentedFlowPortion = fragmentedFlowPortionRectAt(i);
965 return fragmentedFlowPortion.location() + offsetInColumn;
966 }
967 }
968 }
969
970 return logicalPoint;
