36namespace WebCore {
37
38WTF_MAKE_ISO_ALLOCATED_IMPL(RenderMultiColumnSpannerPlaceholder);
39
40RenderPtr<RenderMultiColumnSpannerPlaceholder> RenderMultiColumnSpannerPlaceholder::createAnonymous(RenderMultiColumnFlow& fragmentedFlow, RenderBox& spanner, const RenderStyle& parentStyle)
41{
42 auto newStyle = RenderStyle::createAnonymousStyleWithDisplay(parentStyle, DisplayType::Block);
43 newStyle.setClear(Clear::Both); // We don't want floats in the row preceding the spanner to continue on the other side.
44 auto placeholder = createRenderer<RenderMultiColumnSpannerPlaceholder>(fragmentedFlow, spanner, WTFMove(newStyle));
45 placeholder->initializeStyle();
46 return placeholder;
47}
48
49RenderMultiColumnSpannerPlaceholder::RenderMultiColumnSpannerPlaceholder(RenderMultiColumnFlow& fragmentedFlow, RenderBox& spanner, RenderStyle&& style)
50 : RenderBox(fragmentedFlow.document(), WTFMove(style), RenderBoxModelObjectFlag)
51 , m_spanner(makeWeakPtr(spanner))
52 , m_fragmentedFlow(makeWeakPtr(fragmentedFlow))
53{
54}
55
56const char* RenderMultiColumnSpannerPlaceholder::renderName() const
57{
58 return "RenderMultiColumnSpannerPlaceholder";
59}
