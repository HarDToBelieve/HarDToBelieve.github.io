195void RenderObject::setFragmentedFlowStateIncludingDescendants(FragmentedFlowState state, const RenderElement* fragmentedFlowRoot)
196{
197 setFragmentedFlowState(state);
198
199 if (!is<RenderElement>(*this))
200 return;
201
202 for (auto& child : childrenOfType<RenderObject>(downcast<RenderElement>(*this))) {
203 // If the child is a fragmentation context it already updated the descendants flag accordingly.
204 if (child.isRenderFragmentedFlow())
205 continue;
206 if (fragmentedFlowRoot && child.isOutOfFlowPositioned()) {
207 // Fragmented status propagation stops at out-of-flow boundary.
208 auto isInsideMulticolumnFlow = [&] {
209 auto* containingBlock = child.containingBlock();
210 if (!containingBlock) {
211 ASSERT_NOT_REACHED();
212 return false;
213 }
214 // It's ok to only check the first level containing block (as opposed to the containing block chain) as setFragmentedFlowStateIncludingDescendants is top to down.
215 return containingBlock->isDescendantOf(fragmentedFlowRoot);
216 };
217 if (!isInsideMulticolumnFlow())
218 continue;
219 }
220 ASSERT(state != child.fragmentedFlowState());
221 child.setFragmentedFlowStateIncludingDescendants(state, fragmentedFlowRoot);
222 }
223}

225RenderObject::FragmentedFlowState RenderObject::computedFragmentedFlowState(const RenderObject& renderer)
226{
227 if (!renderer.parent())
228 return renderer.fragmentedFlowState();
229
230 if (is<RenderMultiColumnFlow>(renderer)) {
231 // Multicolumn flows do not inherit the flow state.
232 return InsideInFragmentedFlow;
233 }
234
235 auto inheritedFlowState = RenderObject::NotInsideFragmentedFlow;
236 if (is<RenderText>(renderer))
237 inheritedFlowState = renderer.parent()->fragmentedFlowState();
238 else if (is<RenderSVGBlock>(renderer) || is<RenderSVGInline>(renderer) || is<RenderSVGModelObject>(renderer)) {
239 // containingBlock() skips svg boundary (SVG root is a RenderReplaced).
240 if (auto* svgRoot = SVGRenderSupport::findTreeRootObject(downcast<RenderElement>(renderer)))
241 inheritedFlowState = svgRoot->fragmentedFlowState();
242 } else if (auto* container = renderer.container())
243 inheritedFlowState = container->fragmentedFlowState();
244 else {
245 // Splitting lines or doing continuation, so just keep the current state.
246 inheritedFlowState = renderer.fragmentedFlowState();
247 }
248 return inheritedFlowState;
249}

251void RenderObject::initializeFragmentedFlowStateOnInsertion()
252{
253 ASSERT(parent());
254
255 // A RenderFragmentedFlow is always considered to be inside itself, so it never has to change its state in response to parent changes.
256 if (isRenderFragmentedFlow())
257 return;
258
259 auto computedState = computedFragmentedFlowState(*this);
260 if (fragmentedFlowState() == computedState)
261 return;
262
263 auto* enclosingFragmentedFlow = locateEnclosingFragmentedFlow();
264 setFragmentedFlowStateIncludingDescendants(computedState, enclosingFragmentedFlow ? enclosingFragmentedFlow->parent() : nullptr);
265}

267void RenderObject::resetFragmentedFlowStateOnRemoval()
268{
269 if (fragmentedFlowState() == NotInsideFragmentedFlow)
270 return;
271
272 if (!renderTreeBeingDestroyed() && is<RenderElement>(*this)) {
273 downcast<RenderElement>(*this).removeFromRenderFragmentedFlow();
274 return;
275 }
276
277 // A RenderFragmentedFlow is always considered to be inside itself, so it never has to change its state in response to parent changes.
278 if (isRenderFragmentedFlow())
279 return;
280
281 auto* enclosingFragmentedFlow = this->enclosingFragmentedFlow();
282 setFragmentedFlowStateIncludingDescendants(NotInsideFragmentedFlow, enclosingFragmentedFlow ? enclosingFragmentedFlow->parent() : nullptr);
283}

390// Non-recursive version of the DFS search.
391RenderObject* RenderObject::traverseNext(const RenderObject* stayWithin, HeightTypeTraverseNextInclusionFunction inclusionFunction, int& currentDepth, int& newFixedDepth) const
392{
393 BlockContentHeightType overflowType;
394
395 // Check for suitable children.
396 for (RenderObject* child = firstChildSlow(); child; child = child->nextSibling()) {
397 overflowType = inclusionFunction(*child);
398 if (overflowType != FixedHeight) {
399 currentDepth++;
400 if (overflowType == OverflowHeight)
401 newFixedDepth = currentDepth;
402 ASSERT(!stayWithin || child->isDescendantOf(stayWithin));
403 return child;
404 }
405 }
406
407 if (this == stayWithin)
408 return nullptr;
409
410 // Now we traverse other nodes if they exist, otherwise
411 // we go to the parent node and try doing the same.
412 const RenderObject* n = this;
413 while (n) {
414 while (n && !n->nextSibling() && (!stayWithin || n->parent() != stayWithin)) {
415 n = n->parent();
416 currentDepth--;
417 }
418 if (!n)
419 return nullptr;
420 for (RenderObject* sibling = n->nextSibling(); sibling; sibling = sibling->nextSibling()) {
421 overflowType = inclusionFunction(*sibling);
422 if (overflowType != FixedHeight) {
423 if (overflowType == OverflowHeight)
424 newFixedDepth = currentDepth;
425 ASSERT(!stayWithin || !n->nextSibling() || n->nextSibling()->isDescendantOf(stayWithin));
426 return sibling;
427 }
428 }
429 if (!stayWithin || n->parent() != stayWithin) {
430 n = n->parent();
431 currentDepth--;
432 } else
433 return nullptr;
434 }
435 return nullptr;
436}

472RenderBox* RenderObject::enclosingScrollableContainerForSnapping() const
473{
474 // Walk up the container chain to find the scrollable container that contains
475 // this RenderObject. The important thing here is that `container()` respects
476 // the containing block chain for positioned elements. This is important because
477 // scrollable overflow does not establish a new containing block for children.
478 for (auto* candidate = container(); candidate; candidate = candidate->container()) {
479 // Currently the RenderView can look like it has scrollable overflow, but we never
480 // want to return this as our container. Instead we should use the root element.
481 if (candidate->isRenderView())
482 break;
483 if (candidate->hasOverflowClip())
484 return downcast<RenderBox>(candidate);
485 }
486
487 // If we reach the root, then the root element is the scrolling container.
488 return document().documentElement() ? document().documentElement()->renderBox() : nullptr;
489}

496static inline bool objectIsRelayoutBoundary(const RenderElement* object)
497{
498 // FIXME: In future it may be possible to broaden these conditions in order to improve performance.
499 if (object->isRenderView())
500 return true;
501
502 if (object->isTextControl())
503 return true;
504
505 if (shouldApplyLayoutContainment(*object))
506 return true;
507
508 if (object->isSVGRoot())
509 return true;
510
511 if (!object->hasOverflowClip())
512 return false;
513
514 if (object->style().width().isIntrinsicOrAuto() || object->style().height().isIntrinsicOrAuto() || object->style().height().isPercentOrCalculated())
515 return false;
516
517 // Table parts can't be relayout roots since the table is responsible for layouting all the parts.
518 if (object->isTablePart())
519 return false;
520
521 return true;
522}

550void RenderObject::markContainingBlocksForLayout(ScheduleRelayout scheduleRelayout, RenderElement* newRoot)
551{
552 ASSERT(scheduleRelayout == ScheduleRelayout::No || !newRoot);
553 ASSERT(!isSetNeedsLayoutForbidden());
554
555 auto ancestor = container();
556
557 bool simplifiedNormalFlowLayout = needsSimplifiedNormalFlowLayout() && !selfNeedsLayout() && !normalChildNeedsLayout();
558 bool hasOutOfFlowPosition = !isText() && style().hasOutOfFlowPosition();
559
560 while (ancestor) {
561 // FIXME: Remove this once we remove the special cases for counters, quotes and mathml calling setNeedsLayout during preferred width computation.
562 SetLayoutNeededForbiddenScope layoutForbiddenScope(*ancestor, isSetNeedsLayoutForbidden());
563
564 // Don't mark the outermost object of an unrooted subtree. That object will be
565 // marked when the subtree is added to the document.
566 auto container = ancestor->container();
567 if (!container && !ancestor->isRenderView())
568 return;
569 if (hasOutOfFlowPosition) {
570 bool willSkipRelativelyPositionedInlines = !ancestor->isRenderBlock() || ancestor->isAnonymousBlock();
571 // Skip relatively positioned inlines and anonymous blocks to get to the enclosing RenderBlock.
572 while (ancestor && (!ancestor->isRenderBlock() || ancestor->isAnonymousBlock()))
573 ancestor = ancestor->container();
574 if (!ancestor || ancestor->posChildNeedsLayout())
575 return;
576 if (willSkipRelativelyPositionedInlines)
577 container = ancestor->container();
578 ancestor->setPosChildNeedsLayoutBit(true);
579 simplifiedNormalFlowLayout = true;
580 } else if (simplifiedNormalFlowLayout) {
581 if (ancestor->needsSimplifiedNormalFlowLayout())
582 return;
583 ancestor->setNeedsSimplifiedNormalFlowLayoutBit(true);
584 } else {
585 if (ancestor->normalChildNeedsLayout())
586 return;
587 ancestor->setNormalChildNeedsLayoutBit(true);
588 }
589 ASSERT(!ancestor->isSetNeedsLayoutForbidden());
590
591 if (ancestor == newRoot)
592 return;
593
594 if (scheduleRelayout == ScheduleRelayout::Yes && objectIsRelayoutBoundary(ancestor))
595 break;
596
597 hasOutOfFlowPosition = ancestor->style().hasOutOfFlowPosition();
598 ancestor = container;
599 }
600
601 if (scheduleRelayout == ScheduleRelayout::Yes && ancestor)
602 scheduleRelayoutForSubtree(*ancestor);
603}

623void RenderObject::invalidateContainerPreferredLogicalWidths()
624{
625 // In order to avoid pathological behavior when inlines are deeply nested, we do include them
626 // in the chain that we mark dirty (even though they're kind of irrelevant).
627 auto o = isTableCell() ? containingBlock() : container();
628 while (o && !o->preferredLogicalWidthsDirty()) {
629 // Don't invalidate the outermost object of an unrooted subtree. That object will be
630 // invalidated when the subtree is added to the document.
631 auto container = o->isTableCell() ? o->containingBlock() : o->container();
632 if (!container && !o->isRenderView())
633 break;
634
635 o->m_bitfields.setPreferredLogicalWidthsDirty(true);
636 if (o->style().hasOutOfFlowPosition())
637 // A positioned object has no effect on the min/max width of its containing block ever.
638 // We can optimize this case and not go up any further.
639 break;
640 o = container;
641 }
642}

686void RenderObject::addPDFURLRect(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
687{
688 Vector<LayoutRect> focusRingRects;
689 addFocusRingRects(focusRingRects, paintOffset, paintInfo.paintContainer);
690 LayoutRect urlRect = unionRect(focusRingRects);
691
692 if (urlRect.isEmpty())
693 return;
694 Node* node = this->node();
695 if (!is<Element>(node) || !node->isLink())
696 return;
697 Element& element = downcast<Element>(*node);
698 const AtomString& href = element.getAttribute(hrefAttr);
699 if (href.isNull())
700 return;
701
702 if (paintInfo.context().supportsInternalLinks()) {
703 String outAnchorName;
704 Element* linkTarget = element.findAnchorElementForLink(outAnchorName);
705 if (linkTarget) {
706 paintInfo.context().setDestinationForRect(outAnchorName, urlRect);
707 return;
708 }
709 }
710
711 paintInfo.context().setURLForRect(element.document().completeURL(href), urlRect);
712
713}

715#if PLATFORM(IOS_FAMILY)
716// This function is similar in spirit to RenderText::absoluteRectsForRange, but returns rectangles
717// which are annotated with additional state which helps iOS draw selections in its unique way.
718// No annotations are added in this class.
719// FIXME: Move to RenderText with absoluteRectsForRange()?
720void RenderObject::collectSelectionGeometries(Vector<SelectionGeometry>& geometries, unsigned start, unsigned end)
721{
722 Vector<FloatQuad> quads;
723
724 if (!firstChildSlow()) {
725 // FIXME: WebKit's position for an empty span after a BR is incorrect, so we can't trust
726 // quads for them. We don't need selection geometries for those anyway though, since they
727 // are just empty containers. See <https://bugs.webkit.org/show_bug.cgi?id=49358>.
728 RenderObject* previous = previousSibling();
729 Node* node = this->node();
730 if (!previous || !previous->isBR() || !node || !node->isContainerNode() || !isInline()) {
731 // For inline elements we don't use absoluteQuads, since it takes into account continuations and leads to wrong results.
732 absoluteQuadsForSelection(quads);
733 }
734 } else {
735 unsigned offset = start;
736 for (RenderObject* child = childAt(start); child && offset < end; child = child->nextSibling(), ++offset)
737 child->absoluteQuads(quads);
738 }
739
740 for (auto& quad : quads)

807RenderLayerModelObject* RenderObject::containerForRepaint() const
808{
809 RenderLayerModelObject* repaintContainer = nullptr;
810
811 if (view().usesCompositing()) {
812 if (RenderLayer* parentLayer = enclosingLayer()) {
813 RenderLayer* compLayer = parentLayer->enclosingCompositingLayerForRepaint();
814 if (compLayer)
815 repaintContainer = &compLayer->renderer();
816 }
817 }
818 if (view().hasSoftwareFilters()) {
819 if (RenderLayer* parentLayer = enclosingLayer()) {
820 RenderLayer* enclosingFilterLayer = parentLayer->enclosingFilterLayer();
821 if (enclosingFilterLayer)
822 return &enclosingFilterLayer->renderer();
823 }
824 }
825
826 // If we have a flow thread, then we need to do individual repaints within the RenderFragmentContainers instead.
827 // Return the flow thread as a repaint container in order to create a chokepoint that allows us to change
828 // repainting to do individual region repaints.
829 RenderFragmentedFlow* parentRenderFragmentedFlow = enclosingFragmentedFlow();
830 if (parentRenderFragmentedFlow) {
831 // If we have already found a repaint container then we will repaint into that container only if it is part of the same
832 // flow thread. Otherwise we will need to catch the repaint call and send it to the flow thread.
833 RenderFragmentedFlow* repaintContainerFragmentedFlow = repaintContainer ? repaintContainer->enclosingFragmentedFlow() : nullptr;
834 if (!repaintContainerFragmentedFlow || repaintContainerFragmentedFlow != parentRenderFragmentedFlow)
835 repaintContainer = parentRenderFragmentedFlow;
836 }
837 return repaintContainer;
838}

840void RenderObject::propagateRepaintToParentWithOutlineAutoIfNeeded(const RenderLayerModelObject& repaintContainer, const LayoutRect& repaintRect) const
841{
842 if (!hasOutlineAutoAncestor())
843 return;
844
845 // FIXME: We should really propagate only when the child renderer sticks out.
846 bool repaintRectNeedsConverting = false;
847 // Issue repaint on the renderer with outline: auto.
848 for (const auto* renderer = this; renderer; renderer = renderer->parent()) {
849 bool rendererHasOutlineAutoAncestor = renderer->hasOutlineAutoAncestor();
850 ASSERT(rendererHasOutlineAutoAncestor
851 || renderer->outlineStyleForRepaint().outlineStyleIsAuto() == OutlineIsAuto::On
852 || (is<RenderBoxModelObject>(*renderer) && downcast<RenderBoxModelObject>(*renderer).isContinuation()));
853 if (renderer == &repaintContainer && rendererHasOutlineAutoAncestor)
854 repaintRectNeedsConverting = true;
855 if (rendererHasOutlineAutoAncestor)
856 continue;
857 // Issue repaint on the correct repaint container.
858 LayoutRect adjustedRepaintRect = repaintRect;
859 adjustedRepaintRect.inflate(renderer->outlineStyleForRepaint().outlineSize());
860 if (!repaintRectNeedsConverting)
861 repaintContainer.repaintRectangle(adjustedRepaintRect);
862 else if (is<RenderLayerModelObject>(renderer)) {
863 const auto& rendererWithOutline = downcast<RenderLayerModelObject>(*renderer);
864 adjustedRepaintRect = LayoutRect(repaintContainer.localToContainerQuad(FloatRect(adjustedRepaintRect), &rendererWithOutline).boundingBox());
865 rendererWithOutline.repaintRectangle(adjustedRepaintRect);
866 }
867 return;
868 }
869 ASSERT_NOT_REACHED();
870}

872void RenderObject::repaintUsingContainer(const RenderLayerModelObject* repaintContainer, const LayoutRect& r, bool shouldClipToLayer) const
873{
874 if (r.isEmpty())
875 return;
876
877 if (!repaintContainer)
878 repaintContainer = &view();
879
880 if (is<RenderFragmentedFlow>(*repaintContainer)) {
881 downcast<RenderFragmentedFlow>(*repaintContainer).repaintRectangleInFragments(r);
882 return;
883 }
884
885 propagateRepaintToParentWithOutlineAutoIfNeeded(*repaintContainer, r);
886
887 if (repaintContainer->hasFilter() && repaintContainer->layer() && repaintContainer->layer()->requiresFullLayerImageForFilters()) {
888 repaintContainer->layer()->setFilterBackendNeedsRepaintingInRect(r);
889 return;
890 }
891
892 if (repaintContainer->isRenderView()) {
893 RenderView& view = this->view();
894 ASSERT(repaintContainer == &view);
895 bool viewHasCompositedLayer = view.isComposited();
896 if (!viewHasCompositedLayer || view.layer()->backing()->paintsIntoWindow()) {
897 LayoutRect rect = r;
898 if (viewHasCompositedLayer && view.layer()->transform())
899 rect = LayoutRect(view.layer()->transform()->mapRect(snapRectToDevicePixels(rect, document().deviceScaleFactor())));
900 view.repaintViewRectangle(rect);
901 return;
902 }
903 }
904
905 if (view().usesCompositing()) {
906 ASSERT(repaintContainer->isComposited());
907 repaintContainer->layer()->setBackingNeedsRepaintInRect(r, shouldClipToLayer ? GraphicsLayer::ClipToLayer : GraphicsLayer::DoNotClipToLayer);
908 }

1664RenderBoxModelObject* RenderObject::offsetParent() const
1665{
1666 // If any of the following holds true return null and stop this algorithm:
1667 // A is the root element.
1668 // A is the HTML body element.
1669 // The computed value of the position property for element A is fixed.
1670 if (isDocumentElementRenderer() || isBody() || isFixedPositioned())
1671 return nullptr;
1672
1673 // If A is an area HTML element which has a map HTML element somewhere in the ancestor
1674 // chain return the nearest ancestor map HTML element and stop this algorithm.
1675 // FIXME: Implement!
1676
1677 // Return the nearest ancestor element of A for which at least one of the following is
1678 // true and stop this algorithm if such an ancestor is found:
1679 // * The computed value of the position property is not static.
1680 // * It is the HTML body element.
1681 // * The computed value of the position property of A is static and the ancestor
1682 // is one of the following HTML elements: td, th, or table.
1683 // * Our own extension: if there is a difference in the effective zoom
1684
1685 bool skipTables = isPositioned();
1686 float currZoom = style().effectiveZoom();
1687 auto current = parent();
1688 while (current && (!current->element() || (!current->isPositioned() && !current->isBody()))) {
1689 Element* element = current->element();
1690 if (!skipTables && element && (is<HTMLTableElement>(*element) || is<HTMLTableCellElement>(*element)))
1691 break;
1692
1693 float newZoom = current->style().effectiveZoom();
1694 if (currZoom != newZoom)
1695 break;
1696 currZoom = newZoom;
1697 current = current->parent();
1698 }
1699
1700 return is<RenderBoxModelObject>(current) ? downcast<RenderBoxModelObject>(current) : nullptr;
1701}

1703VisiblePosition RenderObject::createVisiblePosition(int offset, Affinity affinity) const
1704{
1705 // If this is a non-anonymous renderer in an editable area, then it's simple.
1706 if (Node* node = nonPseudoNode()) {
1707 if (!node->hasEditableStyle()) {
1708 // If it can be found, we prefer a visually equivalent position that is editable.
1709 Position position = makeDeprecatedLegacyPosition(node, offset);
1710 Position candidate = position.downstream(CanCrossEditingBoundary);
1711 if (candidate.deprecatedNode()->hasEditableStyle())
1712 return VisiblePosition(candidate, affinity);
1713 candidate = position.upstream(CanCrossEditingBoundary);
1714 if (candidate.deprecatedNode()->hasEditableStyle())
1715 return VisiblePosition(candidate, affinity);
1716 }
1717 // FIXME: Eliminate legacy editing positions
1718 return VisiblePosition(makeDeprecatedLegacyPosition(node, offset), affinity);
1719 }
1720
1721 // We don't want to cross the boundary between editable and non-editable
1722 // regions of the document, but that is either impossible or at least
1723 // extremely unlikely in any normal case because we stop as soon as we
1724 // find a single non-anonymous renderer.
1725
1726 // Find a nearby non-anonymous renderer.
1727 const RenderObject* child = this;
1728 while (const auto parent = child->parent()) {
1729 // Find non-anonymous content after.
1730 const RenderObject* renderer = child;
1731 while ((renderer = renderer->nextInPreOrder(parent))) {
1732 if (Node* node = renderer->nonPseudoNode())
1733 return firstPositionInOrBeforeNode(node);
1734 }
1735
1736 // Find non-anonymous content before.
1737 renderer = child;
1738 while ((renderer = renderer->previousInPreOrder())) {
1739 if (renderer == parent)
1740 break;
1741 if (Node* node = renderer->nonPseudoNode())
1742 return lastPositionInOrAfterNode(node);
1743 }
1744
1745 // Use the parent itself unless it too is anonymous.
1746 if (Element* element = parent->nonPseudoElement())
1747 return firstPositionInOrBeforeNode(element);
1748
1749 // Repeat at the next level up.
1750 child = parent;
1751 }
1752
1753 // Everything was anonymous. Give up.
1754 return VisiblePosition();
1755}

2343{
2344 auto result = RenderObject::collectSelectionGeometriesInternal(range);
2345 auto numberOfGeometries = result.geometries.size();
2346
2347 // Union all the rectangles on interior lines (i.e. not first or last).
2348 // On first and last lines, just avoid having overlaps by merging intersecting rectangles.
2349 Vector<SelectionGeometry> coalescedGeometries;
2350 IntRect interiorUnionRect;
2351 for (size_t i = 0; i < numberOfGeometries; ++i) {
2352 auto& currentGeometry = result.geometries[i];

2354 if (currentGeometry.quad().isEmpty())
2355 continue;
2356
2357 if (coalescedGeometries.isEmpty() || !coalesceSelectionGeometryWithAdjacentQuadsIfPossible(coalescedGeometries.last(), currentGeometry))
2358 coalescedGeometries.append(currentGeometry);
2359 continue;
2360 }
2361
2362 if (currentGeometry.lineNumber() == 1) {
2363 ASSERT(interiorUnionRect.isEmpty());

2365 auto& previousRect = coalescedGeometries.last();

2367 previousRect = coalesceSelectionGeometries(currentGeometry, previousRect);
2368 continue;
2369 }
2370 }
2371 // Couldn't merge with previous rect, so just appending.
2372 coalescedGeometries.append(currentGeometry);
2373 } else if (currentGeometry.lineNumber() < result.maxLineNumber) {
2374 if (interiorUnionRect.isEmpty()) {
2375 // Start collecting interior rects.
2376 interiorUnionRect = currentGeometry.rect();
2377 } else if (interiorUnionRect.intersects(currentGeometry.rect())
2378 || interiorUnionRect.maxX() == currentGeometry.rect().x()
2379 || interiorUnionRect.maxY() == currentGeometry.rect().y()
2380 || interiorUnionRect.x() == currentGeometry.rect().maxX()
2381 || interiorUnionRect.y() == currentGeometry.rect().maxY()) {
2382 // Only union the lines that are attached.
2383 // For iBooks, the interior lines may cross multiple horizontal pages.
2384 interiorUnionRect.unite(currentGeometry.rect());

2386 coalescedGeometries.append(SelectionGeometry({ interiorUnionRect }, SelectionRenderingBehavior::CoalesceBoundingRects, currentGeometry.isHorizontal(), currentGeometry.pageNumber()));

2388 }
2389 } else {
2390 // Processing last line.
2391 if (!interiorUnionRect.isEmpty()) {
2392 coalescedGeometries.append(SelectionGeometry({ interiorUnionRect }, SelectionRenderingBehavior::CoalesceBoundingRects, currentGeometry.isHorizontal(), currentGeometry.pageNumber()));
