142Color RenderReplaced::calculateHighlightColor() const
143{
144 HighlightData highlightData;
145#if ENABLE(APP_HIGHLIGHTS)
146 if (auto appHighlightRegister = document().appHighlightRegisterIfExists()) {
147 if (appHighlightRegister->highlightsVisibility() == HighlightVisibility::Visible) {
148 for (auto& highlight : appHighlightRegister->map()) {
149 for (auto& rangeData : highlight.value->rangesData()) {
150 if (!highlightData.setRenderRange(rangeData))
151 continue;
152
153 auto state = highlightData.highlightStateForRenderer(*this);
154 if (!isHighlighted(state, highlightData))
155 continue;
156
157 OptionSet<StyleColor::Options> styleColorOptions = { StyleColor::Options::UseSystemAppearance };
158 return theme().appHighlightColor(styleColorOptions);
159 }
160 }
161 }
162 }
163#endif
164 if (RuntimeEnabledFeatures::sharedFeatures().highlightAPIEnabled()) {
165 if (auto highlightRegister = document().highlightRegisterIfExists()) {
166 for (auto& highlight : highlightRegister->map()) {
167 for (auto& rangeData : highlight.value->rangesData()) {
168 if (!highlightData.setRenderRange(rangeData))
169 continue;
170
171 auto state = highlightData.highlightStateForRenderer(*this);
172 if (!isHighlighted(state, highlightData))
173 continue;
174
175 if (auto highlightStyle = getUncachedPseudoStyle({ PseudoId::Highlight, highlight.key }, &style()))
176 return highlightStyle->backgroundColor();
177 }
178 }
179 }
180 }
181 return Color();
182}

184void RenderReplaced::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
185{
186 if (!shouldPaint(paintInfo, paintOffset))
187 return;
188
189 LayoutPoint adjustedPaintOffset = paintOffset + location();
190
191 if (paintInfo.phase == PaintPhase::EventRegion) {
192 if (visibleToHitTesting()) {
193 auto borderRect = LayoutRect(adjustedPaintOffset, size());
194 auto borderRegion = approximateAsRegion(style().getRoundedBorderFor(borderRect));
195 paintInfo.eventRegionContext->unite(borderRegion, style());
196 }
197 return;
198 }
199
200 SetLayoutNeededForbiddenScope scope(*this);
201
202 GraphicsContextStateSaver savedGraphicsContext(paintInfo.context(), false);
203 if (element() && element()->parentOrShadowHostElement()) {
204 auto* parentContainer = element()->parentOrShadowHostElement();
205 ASSERT(parentContainer);
206 if (draggedContentContainsReplacedElement(document().markers().markersFor(*parentContainer, DocumentMarker::DraggedContent), *element())) {
207 savedGraphicsContext.save();
208 paintInfo.context().setAlpha(0.25);
209 }
210 }
211
212 if (hasVisibleBoxDecorations() && paintInfo.phase == PaintPhase::Foreground)
213 paintBoxDecorations(paintInfo, adjustedPaintOffset);
214
215 if (paintInfo.phase == PaintPhase::Mask) {
216 paintMask(paintInfo, adjustedPaintOffset);
217 return;
218 }
219
220 LayoutRect paintRect = LayoutRect(adjustedPaintOffset, size());
221 if (paintInfo.phase == PaintPhase::Outline || paintInfo.phase == PaintPhase::SelfOutline) {
222 if (style().outlineWidth())
223 paintOutline(paintInfo, paintRect);
224 return;
225 }
226
227 if (paintInfo.phase != PaintPhase::Foreground && paintInfo.phase != PaintPhase::Selection)
228 return;
229
230 if (!paintInfo.shouldPaintWithinRoot(*this))
231 return;
232
233 Color highlightColor;
234 if (!document().printing() && !paintInfo.paintBehavior.contains(PaintBehavior::ExcludeSelection))
235 highlightColor = calculateHighlightColor();
236
237 bool drawSelectionTint = shouldDrawSelectionTint();
238 if (paintInfo.phase == PaintPhase::Selection) {
239 if (selectionState() == HighlightState::None)
240 return;
241 drawSelectionTint = false;
242 }
243
244 bool completelyClippedOut = false;
245 if (style().hasBorderRadius()) {
246 completelyClippedOut = size().isEmpty();
247 if (!completelyClippedOut) {
248 // Push a clip if we have a border radius, since we want to round the foreground content that gets painted.
249 paintInfo.context().save();
250 auto pixelSnappedRoundedRect = style().getRoundedInnerBorderFor(paintRect,
251 paddingTop() + borderTop(), paddingBottom() + borderBottom(), paddingLeft() + borderLeft(), paddingRight() + borderRight(), true, true).pixelSnappedRoundedRectForPainting(document().deviceScaleFactor());
252 clipRoundedInnerRect(paintInfo.context(), paintRect, pixelSnappedRoundedRect);
253 }
254 }
255
256 if (!completelyClippedOut) {
257 paintReplaced(paintInfo, adjustedPaintOffset);
258
259 if (style().hasBorderRadius())
260 paintInfo.context().restore();
261 }
262
263 // The selection tint never gets clipped by border-radius rounding, since we want it to run right up to the edges of
264 // surrounding content.
265 if (drawSelectionTint) {
266 LayoutRect selectionPaintingRect = localSelectionRect();
267 selectionPaintingRect.moveBy(adjustedPaintOffset);
268 paintInfo.context().fillRect(snappedIntRect(selectionPaintingRect), selectionBackgroundColor());
269 }
270
271 if (highlightColor.isVisible()) {
272 auto selectionPaintingRect = localSelectionRect(false);
273 selectionPaintingRect.moveBy(adjustedPaintOffset);
274 paintInfo.context().fillRect(snappedIntRect(selectionPaintingRect), highlightColor);
275 }
276}

278bool RenderReplaced::shouldPaint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
279{
280 if ((paintInfo.paintBehavior.contains(PaintBehavior::ExcludeSelection)) && isSelected())
281 return false;
282
283 if (paintInfo.phase != PaintPhase::Foreground
284 && paintInfo.phase != PaintPhase::Outline
285 && paintInfo.phase != PaintPhase::SelfOutline
286 && paintInfo.phase != PaintPhase::Selection
287 && paintInfo.phase != PaintPhase::Mask
288 && paintInfo.phase != PaintPhase::EventRegion)
289 return false;
290
291 if (!paintInfo.shouldPaintWithinRoot(*this))
292 return false;
293
294 // if we're invisible or haven't received a layout yet, then just bail.
295 if (style().visibility() != Visibility::Visible)
296 return false;
297
298 LayoutPoint adjustedPaintOffset = paintOffset + location();
299
300 // Early exit if the element touches the edges.
301 LayoutUnit top = adjustedPaintOffset.y() + visualOverflowRect().y();
302 LayoutUnit bottom = adjustedPaintOffset.y() + visualOverflowRect().maxY();
303 if (isSelected() && m_inlineBoxWrapper) {
304 const LegacyRootInlineBox& rootBox = m_inlineBoxWrapper->root();
305 LayoutUnit selTop = paintOffset.y() + rootBox.selectionTop();
306 LayoutUnit selBottom = paintOffset.y() + selTop + rootBox.selectionHeight();
307 top = std::min(selTop, top);
308 bottom = std::max(selBottom, bottom);
309 }
310
311 LayoutRect localRepaintRect = paintInfo.rect;
312 if (adjustedPaintOffset.x() + visualOverflowRect().x() >= localRepaintRect.maxX() || adjustedPaintOffset.x() + visualOverflowRect().maxX() <= localRepaintRect.x())
313 return false;
314
315 if (top >= localRepaintRect.maxY() || bottom <= localRepaintRect.y())
316 return false;
317
318 return true;
319}

321static inline RenderBlock* firstContainingBlockWithLogicalWidth(const RenderReplaced* replaced)
322{
323 // We have to lookup the containing block, which has an explicit width, which must not be equal to our direct containing block.
324 // If the embedded document appears _after_ we performed the initial layout, our intrinsic size is 300x150. If our containing
325 // block doesn't provide an explicit width, it's set to the 300 default, coming from the initial layout run.
326 RenderBlock* containingBlock = replaced->containingBlock();
327 if (!containingBlock)
328 return 0;
329
330 for (; containingBlock && !is<RenderView>(*containingBlock) && !containingBlock->isBody(); containingBlock = containingBlock->containingBlock()) {
331 if (containingBlock->style().logicalWidth().isSpecified())
332 return containingBlock;
333 }
334
335 return 0;
336}

366bool RenderReplaced::setNeedsLayoutIfNeededAfterIntrinsicSizeChange()
367{
368 setPreferredLogicalWidthsDirty(true);
369
370 // If the actual area occupied by the image has changed and it is not constrained by style then a layout is required.
371 bool imageSizeIsConstrained = style().logicalWidth().isSpecified() && style().logicalHeight().isSpecified();
372
373 // FIXME: We only need to recompute the containing block's preferred size
374 // if the containing block's size depends on the image's size (i.e., the container uses shrink-to-fit sizing).
375 // There's no easy way to detect that shrink-to-fit is needed, always force a layout.
376 bool containingBlockNeedsToRecomputePreferredSize =
377 style().logicalWidth().isPercentOrCalculated()
378 || style().logicalMaxWidth().isPercentOrCalculated()
379 || style().logicalMinWidth().isPercentOrCalculated();
380
381 // Flex layout algorithm uses the intrinsic image width/height even if width/height are specified.
382 if (!imageSizeIsConstrained || containingBlockNeedsToRecomputePreferredSize || isFlexItem()) {
383 setNeedsLayout();
384 return true;
385 }
386
387 return false;
388}

390void RenderReplaced::computeAspectRatioInformationForRenderBox(RenderBox* contentRenderer, FloatSize& constrainedSize, double& intrinsicRatio) const
391{
392 FloatSize intrinsicSize;
393 if (shouldApplySizeContainment(*this))
394 RenderReplaced::computeIntrinsicRatioInformation(intrinsicSize, intrinsicRatio);
395 else if (contentRenderer) {
396 contentRenderer->computeIntrinsicRatioInformation(intrinsicSize, intrinsicRatio);
397
398 if (style().aspectRatioType() == AspectRatioType::Ratio || (style().aspectRatioType() == AspectRatioType::AutoAndRatio && !intrinsicRatio))
399 intrinsicRatio = style().aspectRatioWidth() / style().aspectRatioHeight();
400
401 // Handle zoom & vertical writing modes here, as the embedded document doesn't know about them.
402 intrinsicSize.scale(style().effectiveZoom());
403
404 if (is<RenderImage>(*this))
405 intrinsicSize.scale(downcast<RenderImage>(*this).imageDevicePixelRatio());
406
407 // Update our intrinsic size to match what the content renderer has computed, so that when we
408 // constrain the size below, the correct intrinsic size will be obtained for comparison against
409 // min and max widths.
410 if (intrinsicRatio && !intrinsicSize.isZero())
411 m_intrinsicSize = LayoutSize(intrinsicSize);
412
413 if (!isHorizontalWritingMode()) {
414 if (intrinsicRatio)
415 intrinsicRatio = 1 / intrinsicRatio;
416 intrinsicSize = intrinsicSize.transposedSize();
417 }
418 } else {
419 computeIntrinsicRatioInformation(intrinsicSize, intrinsicRatio);
420 if (intrinsicRatio && !intrinsicSize.isZero())
421 m_intrinsicSize = LayoutSize(isHorizontalWritingMode() ? intrinsicSize : intrinsicSize.transposedSize());
422 }
423
424 // Now constrain the intrinsic size along each axis according to minimum and maximum width/heights along the
425 // opposite axis. So for example a maximum width that shrinks our width will result in the height we compute here
426 // having to shrink in order to preserve the aspect ratio. Because we compute these values independently along
427 // each axis, the final returned size may in fact not preserve the aspect ratio.
428 // FIXME: In the long term, it might be better to just return this code more to the way it used to be before this
429 // function was added, since all it has done is make the code more unclear.
430 constrainedSize = intrinsicSize;
431 if (intrinsicRatio && !intrinsicSize.isEmpty() && style().logicalWidth().isAuto() && style().logicalHeight().isAuto()) {
432 // We can't multiply or divide by 'intrinsicRatio' here, it breaks tests, like fast/images/zoomed-img-size.html, which
433 // can only be fixed once subpixel precision is available for things like intrinsicWidth/Height - which include zoom!
434 constrainedSize.setWidth(RenderBox::computeReplacedLogicalHeight() * intrinsicSize.width() / intrinsicSize.height());
435 constrainedSize.setHeight(RenderBox::computeReplacedLogicalWidth() * intrinsicSize.height() / intrinsicSize.width());
436 }
437}

439LayoutRect RenderReplaced::replacedContentRect(const LayoutSize& intrinsicSize) const
440{
441 LayoutRect contentRect = contentBoxRect();
442 if (intrinsicSize.isEmpty())
443 return contentRect;
444
445 ObjectFit objectFit = style().objectFit();
446
447 LayoutRect finalRect = contentRect;
448 switch (objectFit) {
449 case ObjectFit::Contain:
450 case ObjectFit::ScaleDown:
451 case ObjectFit::Cover:
452 finalRect.setSize(finalRect.size().fitToAspectRatio(intrinsicSize, objectFit == ObjectFit::Cover ? AspectRatioFitGrow : AspectRatioFitShrink));
453 if (objectFit != ObjectFit::ScaleDown || finalRect.width() <= intrinsicSize.width())
454 break;
455 FALLTHROUGH;
456 case ObjectFit::None:
457 finalRect.setSize(intrinsicSize);
458 break;
459 case ObjectFit::Fill:
460 break;
461 }
462
463 LengthPoint objectPosition = style().objectPosition();
464
465 LayoutUnit xOffset = minimumValueForLength(objectPosition.x(), contentRect.width() - finalRect.width());
466 LayoutUnit yOffset = minimumValueForLength(objectPosition.y(), contentRect.height() - finalRect.height());
467
468 finalRect.move(xOffset, yOffset);
469
470 return finalRect;
471}

480void RenderReplaced::computeIntrinsicRatioInformation(FloatSize& intrinsicSize, double& intrinsicRatio) const
481{
482 // If there's an embeddedContentBox() of a remote, referenced document available, this code-path should never be used.
483 ASSERT(!embeddedContentBox() || shouldApplySizeContainment(*this));
484 intrinsicSize = FloatSize(intrinsicLogicalWidth(), intrinsicLogicalHeight());
485
486 if (style().hasAspectRatio()) {
487 intrinsicRatio = style().logicalAspectRatio();
488 if (style().aspectRatioType() == AspectRatioType::Ratio)
489 return;
490 }
491 // Figure out if we need to compute an intrinsic ratio.
492 if (!hasIntrinsicAspectRatio())
493 return;
494
495 if (intrinsicSize.isEmpty())
496 return;
497
498 intrinsicRatio = intrinsicSize.width() / intrinsicSize.height();
499}

501LayoutUnit RenderReplaced::computeConstrainedLogicalWidth(ShouldComputePreferred shouldComputePreferred) const
502{
503 if (shouldComputePreferred == ComputePreferred)
504 return computeReplacedLogicalWidthRespectingMinMaxWidth(0_lu, ComputePreferred);
505
506 // The aforementioned 'constraint equation' used for block-level, non-replaced
507 // elements in normal flow:
508 // 'margin-left' + 'border-left-width' + 'padding-left' + 'width' +
509 // 'padding-right' + 'border-right-width' + 'margin-right' = width of
510 // containing block
511 LayoutUnit logicalWidth = containingBlock()->availableLogicalWidth();
512
513 // This solves above equation for 'width' (== logicalWidth).
514 LayoutUnit marginStart = minimumValueForLength(style().marginStart(), logicalWidth);
515 LayoutUnit marginEnd = minimumValueForLength(style().marginEnd(), logicalWidth);
516
517 // FIXME: This expression does not align with the comment above, which is quoting https://www.w3.org/TR/CSS22/visudet.html#blockwidth.
518 logicalWidth = std::max(0_lu, (logicalWidth - (marginStart + marginEnd + borderLeft() + borderRight())));
519 return computeReplacedLogicalWidthRespectingMinMaxWidth(logicalWidth, shouldComputePreferred);
520}

522static inline LayoutUnit resolveWidthForRatio(LayoutUnit borderAndPaddingLogicalHeight, LayoutUnit borderAndPaddingLogicalWidth, LayoutUnit logicalHeight, double aspectRatio, BoxSizing boxSizing)
523{
524 if (boxSizing == BoxSizing::BorderBox)
525 return LayoutUnit(round((logicalHeight + borderAndPaddingLogicalHeight) * aspectRatio)) - borderAndPaddingLogicalWidth;
526 return LayoutUnit(round(logicalHeight * aspectRatio));
527}
528
529LayoutUnit RenderReplaced::computeReplacedLogicalWidth(ShouldComputePreferred shouldComputePreferred) const
530{
531 if (style().logicalWidth().isSpecified() || style().logicalWidth().isIntrinsic())
532 return computeReplacedLogicalWidthRespectingMinMaxWidth(computeReplacedLogicalWidthUsing(MainOrPreferredSize, style().logicalWidth()), shouldComputePreferred);
533
534 RenderBox* contentRenderer = embeddedContentBox();
535
536 // 10.3.2 Inline, replaced elements: http://www.w3.org/TR/CSS21/visudet.html#inline-replaced-width
537 double intrinsicRatio = 0;
538 FloatSize constrainedSize;
539 computeAspectRatioInformationForRenderBox(contentRenderer, constrainedSize, intrinsicRatio);
540
541 if (style().logicalWidth().isAuto()) {
542 bool computedHeightIsAuto = style().logicalHeight().isAuto();
543 // FIXME: The hasIntrinsicWidth and hasIntrinsicHeight flags here are only updated for SVG elements. Ideally, it can be used in
544 // all the cases so we don't need constrainedSize.width() > 0 and constrainedSize.height() > 0 clauses.
545 bool hasIntrinsicWidth = constrainedSize.hasIntrinsicWidth || constrainedSize.width() > 0;
546 bool hasIntrinsicHeight = constrainedSize.hasIntrinsicHeight || constrainedSize.height() > 0;
547
548 // For flex items where the logical height has been overriden then we should use that size to compute the replaced width as long as the flex item has
549 // an intrinsic size. It is possible (indeed, common) for an SVG graphic to have an intrinsic aspect ratio but not to have an intrinsic width or height.
550 // There are also elements with intrinsic sizes but without intrinsic ratio (like an iframe). We cannot apply this rule to grid items because the grid
551 // container can distort aspect ratios in case of "align-self: stretch" for example (see https://drafts.csswg.org/css-grid/#grid-item-sizing).
552 if (intrinsicRatio && isFlexItem() && hasOverridingLogicalHeight() && hasIntrinsicWidth && hasIntrinsicHeight)
553 return computeReplacedLogicalWidthRespectingMinMaxWidth(roundToInt(round(overridingContentLogicalHeight() * intrinsicRatio)), shouldComputePreferred);
554
555 // If 'height' and 'width' both have computed values of 'auto' and the element also has an intrinsic width, then that intrinsic width is the used value of 'width'.
556 if (computedHeightIsAuto && hasIntrinsicWidth)
557 return computeReplacedLogicalWidthRespectingMinMaxWidth(constrainedSize.width(), shouldComputePreferred);
558
559 if (intrinsicRatio) {
560 // If 'height' and 'width' both have computed values of 'auto' and the element has no intrinsic width, but does have an intrinsic height and intrinsic ratio;
561 // or if 'width' has a computed value of 'auto', 'height' has some other computed value, and the element does have an intrinsic ratio; then the used value
562 // of 'width' is: (used height) * (intrinsic ratio)
563 if (!computedHeightIsAuto || (!hasIntrinsicWidth && hasIntrinsicHeight)) {
564 LayoutUnit estimatedUsedWidth = hasIntrinsicWidth ? LayoutUnit(constrainedSize.width()) : computeConstrainedLogicalWidth(shouldComputePreferred);
565 LayoutUnit logicalHeight = computeReplacedLogicalHeight(std::optional<LayoutUnit>(estimatedUsedWidth));
566 BoxSizing boxSizing = style().hasAspectRatio() ? style().boxSizingForAspectRatio() : BoxSizing::ContentBox;
567 return computeReplacedLogicalWidthRespectingMinMaxWidth(resolveWidthForRatio(borderAndPaddingLogicalHeight(), borderAndPaddingLogicalWidth(), logicalHeight, intrinsicRatio, boxSizing), shouldComputePreferred);
568 }
569
570 // If 'height' and 'width' both have computed values of 'auto' and the
571 // element has an intrinsic ratio but no intrinsic height or width, then
572 // the used value of 'width' is undefined in CSS 2.1. However, it is
573 // suggested that, if the containing block's width does not itself depend
574 // on the replaced element's width, then the used value of 'width' is
575 // calculated from the constraint equation used for block-level,
576 // non-replaced elements in normal flow.
577 if (computedHeightIsAuto && !hasIntrinsicWidth && !hasIntrinsicHeight)
578 return computeConstrainedLogicalWidth(shouldComputePreferred);
579 }
580
581 // Otherwise, if 'width' has a computed value of 'auto', and the element has an intrinsic width, then that intrinsic width is the used value of 'width'.
582 if (hasIntrinsicWidth)
583 return computeReplacedLogicalWidthRespectingMinMaxWidth(constrainedSize.width(), shouldComputePreferred);
584
585 // Otherwise, if 'width' has a computed value of 'auto', but none of the conditions above are met, then the used value of 'width' becomes 300px. If 300px is too
586 // wide to fit the device, UAs should use the width of the largest rectangle that has a 2:1 ratio and fits the device instead.
587 // Note: We fall through and instead return intrinsicLogicalWidth() here - to preserve existing WebKit behavior, which might or might not be correct, or desired.
588 // Changing this to return cDefaultWidth, will affect lots of test results. Eg. some tests assume that a blank <img> tag (which implies width/height=auto)
589 // has no intrinsic size, which is wrong per CSS 2.1, but matches our behavior since a long time.
590 }
591
592 return computeReplacedLogicalWidthRespectingMinMaxWidth(intrinsicLogicalWidth(), shouldComputePreferred);
593}

595static inline LayoutUnit resolveHeightForRatio(LayoutUnit borderAndPaddingLogicalWidth, LayoutUnit borderAndPaddingLogicalHeight, LayoutUnit logicalWidth, double aspectRatio, BoxSizing boxSizing)
596{
597 if (boxSizing == BoxSizing::BorderBox)
598 return LayoutUnit(round((logicalWidth + borderAndPaddingLogicalWidth) / aspectRatio)) - borderAndPaddingLogicalHeight;
599 return LayoutUnit(round(logicalWidth / aspectRatio));

602LayoutUnit RenderReplaced::computeReplacedLogicalHeight(std::optional<LayoutUnit> estimatedUsedWidth) const
603{
604 // 10.5 Content height: the 'height' property: http://www.w3.org/TR/CSS21/visudet.html#propdef-height
605 if (hasReplacedLogicalHeight())
606 return computeReplacedLogicalHeightRespectingMinMaxHeight(computeReplacedLogicalHeightUsing(MainOrPreferredSize, style().logicalHeight()));
607
608 RenderBox* contentRenderer = embeddedContentBox();
609
610 // 10.6.2 Inline, replaced elements: http://www.w3.org/TR/CSS21/visudet.html#inline-replaced-height
611 double intrinsicRatio = 0;
612 FloatSize constrainedSize;
613 computeAspectRatioInformationForRenderBox(contentRenderer, constrainedSize, intrinsicRatio);
614
615 bool widthIsAuto = style().logicalWidth().isAuto();
616 // FIXME: The hasIntrinsicHeight and hasIntrinsicWidth flags here are only updated for SVG elements. Ideally, it can be used in
617 // all the cases so we don't need constrainedSize.height() > 0 and constrainedSize.width() > 0 clauses.
618 bool hasIntrinsicHeight = constrainedSize.hasIntrinsicHeight || constrainedSize.height() > 0;
619 bool hasIntrinsicWidth = constrainedSize.hasIntrinsicWidth || constrainedSize.width() > 0;
620
621 // See computeReplacedLogicalHeight() for a similar check for heights.
622 if (intrinsicRatio && isFlexItem() && hasOverridingLogicalWidth() && hasIntrinsicHeight && hasIntrinsicWidth)
623 return computeReplacedLogicalHeightRespectingMinMaxHeight(roundToInt(round(overridingContentLogicalWidth() / intrinsicRatio)));
624
625 // If 'height' and 'width' both have computed values of 'auto' and the element also has an intrinsic height, then that intrinsic height is the used value of 'height'.
626 if (widthIsAuto && hasIntrinsicHeight)
627 return computeReplacedLogicalHeightRespectingMinMaxHeight(constrainedSize.height());
628
629 // Otherwise, if 'height' has a computed value of 'auto', and the element has an intrinsic ratio then the used value of 'height' is:
630 // (used width) / (intrinsic ratio)
631 if (intrinsicRatio) {
632 LayoutUnit usedWidth = estimatedUsedWidth ? estimatedUsedWidth.value() : availableLogicalWidth();
633 BoxSizing boxSizing = BoxSizing::ContentBox;
634 if (style().hasAspectRatio())
635 boxSizing = style().boxSizingForAspectRatio();
636 return computeReplacedLogicalHeightRespectingMinMaxHeight(resolveHeightForRatio(borderAndPaddingLogicalWidth(), borderAndPaddingLogicalHeight(), usedWidth, intrinsicRatio, boxSizing));
637 }
638
639 // Otherwise, if 'height' has a computed value of 'auto', and the element has an intrinsic height, then that intrinsic height is the used value of 'height'.
640 if (hasIntrinsicHeight)
641 return computeReplacedLogicalHeightRespectingMinMaxHeight(constrainedSize.height());
642
643 // Otherwise, if 'height' has a computed value of 'auto', but none of the conditions above are met, then the used value of 'height' must be set to the height
644 // of the largest rectangle that has a 2:1 ratio, has a height not greater than 150px, and has a width not greater than the device width.
645 return computeReplacedLogicalHeightRespectingMinMaxHeight(intrinsicLogicalHeight());
646}

650 minLogicalWidth = maxLogicalWidth = intrinsicLogicalWidth();

653void RenderReplaced::computePreferredLogicalWidths()
654{
655 ASSERT(preferredLogicalWidthsDirty());
656
657 // We cannot resolve any percent logical width here as the available logical
658 // width may not be set on our containing block.
659 if (style().logicalWidth().isPercentOrCalculated())
660 computeIntrinsicLogicalWidths(m_minPreferredLogicalWidth, m_maxPreferredLogicalWidth);
661 else
662 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = computeReplacedLogicalWidth(ComputePreferred);
663
664 const RenderStyle& styleToUse = style();
665 if (styleToUse.logicalWidth().isPercentOrCalculated() || styleToUse.logicalMaxWidth().isPercentOrCalculated())
666 m_minPreferredLogicalWidth = 0;
667
668 if (styleToUse.logicalMinWidth().isFixed() && styleToUse.logicalMinWidth().value() > 0) {
669 m_maxPreferredLogicalWidth = std::max(m_maxPreferredLogicalWidth, adjustContentBoxLogicalWidthForBoxSizing(styleToUse.logicalMinWidth()));
670 m_minPreferredLogicalWidth = std::max(m_minPreferredLogicalWidth, adjustContentBoxLogicalWidthForBoxSizing(styleToUse.logicalMinWidth()));
671 }
672
673 if (styleToUse.logicalMaxWidth().isFixed()) {
674 m_maxPreferredLogicalWidth = std::min(m_maxPreferredLogicalWidth, adjustContentBoxLogicalWidthForBoxSizing(styleToUse.logicalMaxWidth()));
675 m_minPreferredLogicalWidth = std::min(m_minPreferredLogicalWidth, adjustContentBoxLogicalWidthForBoxSizing(styleToUse.logicalMaxWidth()));
676 }
677
678 LayoutUnit borderAndPadding = borderAndPaddingLogicalWidth();
679 m_minPreferredLogicalWidth += borderAndPadding;
680 m_maxPreferredLogicalWidth += borderAndPadding;
681
682 setPreferredLogicalWidthsDirty(false);
683}

685VisiblePosition RenderReplaced::positionForPoint(const LayoutPoint& point, const RenderFragmentContainer* fragment)
686{
687 auto [top, bottom] = [&] {
688 if (auto run = LayoutIntegration::runFor(*this)) {
689 auto line = run.line();
690 return std::make_pair(line->selectionTopForHitTesting(), line->selectionBottom());
691 }
692 return std::make_pair(logicalTop(), logicalBottom());
693 }();
694
695 LayoutUnit blockDirectionPosition = isHorizontalWritingMode() ? point.y() + y() : point.x() + x();
696 LayoutUnit lineDirectionPosition = isHorizontalWritingMode() ? point.x() + x() : point.y() + y();
697
698 if (blockDirectionPosition < top)
699 return createVisiblePosition(caretMinOffset(), Affinity::Downstream); // coordinates are above
700
701 if (blockDirectionPosition >= bottom)
702 return createVisiblePosition(caretMaxOffset(), Affinity::Downstream); // coordinates are below
703
704 if (element()) {
705 if (lineDirectionPosition <= logicalLeft() + (logicalWidth() / 2))
706 return createVisiblePosition(0, Affinity::Downstream);
707 return createVisiblePosition(1, Affinity::Downstream);
708 }
709
710 return RenderBox::positionForPoint(point, fragment);
711}

726LayoutRect RenderReplaced::localSelectionRect(bool checkWhetherSelected) const
727{
728 if (checkWhetherSelected && !isSelected())
729 return LayoutRect();
730
731 if (!m_inlineBoxWrapper)
732 // We're a block-level replaced element. Just return our own dimensions.
733 return LayoutRect(LayoutPoint(), size());
734
735 const LegacyRootInlineBox& rootBox = m_inlineBoxWrapper->root();
736 LayoutUnit newLogicalTop { rootBox.blockFlow().style().isFlippedBlocksWritingMode() ? m_inlineBoxWrapper->logicalBottom() - rootBox.selectionBottom() : rootBox.selectionTop() - m_inlineBoxWrapper->logicalTop() };
737 if (rootBox.blockFlow().style().isHorizontalWritingMode())
738 return LayoutRect(0_lu, newLogicalTop, width(), rootBox.selectionHeight());
739 return LayoutRect(newLogicalTop, 0_lu, rootBox.selectionHeight(), height());
740}

747bool RenderReplaced::isHighlighted(HighlightState state, const HighlightData& rangeData) const
748{
749 if (state == HighlightState::None)
750 return false;
751 if (state == HighlightState::Inside)
752 return true;
753
754 auto selectionStart = rangeData.startOffset();
755 auto selectionEnd = rangeData.endOffset();
756 if (state == HighlightState::Start)
757 return !selectionStart;
758
759 unsigned end = element()->hasChildNodes() ? element()->countChildNodes() : 1;
760 if (state == HighlightState::End)
761 return selectionEnd == end;
762 if (state == HighlightState::Both)
763 return !selectionStart && selectionEnd == end;
764 ASSERT_NOT_REACHED();
765 return false;
766}

768LayoutRect RenderReplaced::clippedOverflowRect(const RenderLayerModelObject* repaintContainer, VisibleRectContext context) const
769{
770 if (style().visibility() != Visibility::Visible && !enclosingLayer()->hasVisibleContent())
771 return LayoutRect();

782bool RenderReplaced::isContentLikelyVisibleInViewport()
783{
784 if (!isVisibleIgnoringGeometry())
785 return false;
786
787 auto& frameView = view().frameView();
788 auto visibleRect = LayoutRect(frameView.windowToContents(frameView.windowClipRect()));
789 auto contentRect = computeRectForRepaint(replacedContentRect(), nullptr);
790
791 // Content rectangle may be empty because it is intrinsically sized and the content has not loaded yet.
792 if (contentRect.isEmpty() && (style().logicalWidth().isAuto() || style().logicalHeight().isAuto()))
793 return visibleRect.contains(contentRect.location());
794
795 return visibleRect.intersects(contentRect);
796}
