22#pragma once
23
24#include "RenderBox.h"
25
26namespace WebCore {
27
28class RenderReplaced : public RenderBox {
29 WTF_MAKE_ISO_ALLOCATED(RenderReplaced);
30public:
31 virtual ~RenderReplaced();
32
33 LayoutUnit computeReplacedLogicalWidth(ShouldComputePreferred = ComputeActual) const override;
34 LayoutUnit computeReplacedLogicalHeight(std::optional<LayoutUnit> estimatedUsedWidth = std::nullopt) const override;
35
36 LayoutRect replacedContentRect(const LayoutSize& intrinsicSize) const;
37 LayoutRect replacedContentRect() const { return replacedContentRect(intrinsicSize()); }
38
39 bool hasReplacedLogicalWidth() const;
40 bool hasReplacedLogicalHeight() const;
41 bool setNeedsLayoutIfNeededAfterIntrinsicSizeChange();
42
43 LayoutSize intrinsicSize() const final
44 {
45 if (shouldApplySizeContainment(*this))
46 return LayoutSize();
47 return m_intrinsicSize;
48 }
49
50 RoundedRect roundedContentBoxRect() const;
51
52 bool isContentLikelyVisibleInViewport();
53 bool needsPreferredWidthsRecalculation() const override;
54
55protected:
56 RenderReplaced(Element&, RenderStyle&&);
57 RenderReplaced(Element&, RenderStyle&&, const LayoutSize& intrinsicSize);
58 RenderReplaced(Document&, RenderStyle&&, const LayoutSize& intrinsicSize);
59
60 void layout() override;
61
62 void computeIntrinsicRatioInformation(FloatSize& intrinsicSize, double& intrinsicRatio) const override;
63
64 void computeIntrinsicLogicalWidths(LayoutUnit& minLogicalWidth, LayoutUnit& maxLogicalWidth) const final;
65
66 virtual LayoutUnit minimumReplacedHeight() const { return 0_lu; }
67
68 bool isSelected() const;
69
70 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
71
72 void setIntrinsicSize(const LayoutSize& intrinsicSize) { m_intrinsicSize = intrinsicSize; }
73 virtual void intrinsicSizeChanged();
74 virtual bool hasRelativeIntrinsicLogicalWidth() const { return false; }
75
76 void paint(PaintInfo&, const LayoutPoint&) override;
77 bool shouldPaint(PaintInfo&, const LayoutPoint&);
78 LayoutRect localSelectionRect(bool checkWhetherSelected = true) const; // This is in local coordinates, but it's a physical rect (so the top left corner is physical top left).
79
80 void willBeDestroyed() override;
81
82private:
83 LayoutUnit computeConstrainedLogicalWidth(ShouldComputePreferred) const;
84
85 virtual RenderBox* embeddedContentBox() const { return 0; }
86 const char* renderName() const override { return "RenderReplaced"; }
87
88 bool canHaveChildren() const override { return false; }
89
90 void computePreferredLogicalWidths() final;
91 virtual void paintReplaced(PaintInfo&, const LayoutPoint&) { }
92
93 LayoutRect clippedOverflowRect(const RenderLayerModelObject* repaintContainer, VisibleRectContext) const override;
94
95 VisiblePosition positionForPoint(const LayoutPoint&, const RenderFragmentContainer*) final;
96
97 bool canBeSelectionLeaf() const override { return true; }
98
99 LayoutRect selectionRectForRepaint(const RenderLayerModelObject* repaintContainer, bool clipToVisibleContent = true) final;
100 void computeAspectRatioInformationForRenderBox(RenderBox*, FloatSize& constrainedSize, double& intrinsicRatio) const;
101
102 virtual bool shouldDrawSelectionTint() const;
103
104 Color calculateHighlightColor() const;
105 bool isHighlighted(HighlightState, const HighlightData&) const;
106
107 mutable LayoutSize m_intrinsicSize;
108};
109
110} // namespace WebCore
