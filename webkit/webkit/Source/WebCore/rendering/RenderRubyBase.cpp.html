68void RenderRubyBase::adjustInlineDirectionLineBounds(int expansionOpportunityCount, float& logicalLeft, float& logicalWidth) const
69{
70 if (rubyRun()->hasOverridingLogicalWidth() && firstRootBox() && !firstRootBox()->nextRootBox()) {
71 logicalLeft += m_initialOffset;
72 logicalWidth -= 2 * m_initialOffset;
73 return;
74 }
75
76 LayoutUnit maxPreferredLogicalWidth = rubyRun() && rubyRun()->hasOverridingLogicalWidth() ? rubyRun()->overridingLogicalWidth() : this->maxPreferredLogicalWidth();
77 if (maxPreferredLogicalWidth >= logicalWidth)
78 return;
79
80 // Inset the ruby base by half the inter-ideograph expansion amount.
81 float inset = (logicalWidth - maxPreferredLogicalWidth) / (expansionOpportunityCount + 1);
82
83 logicalLeft += inset / 2;
84 logicalWidth -= inset;
85}

87void RenderRubyBase::cachePriorCharactersIfNeeded(const LazyLineBreakIterator& lineBreakIterator)
88{
89 auto* run = rubyRun();
90 if (run)
91 run->setCachedPriorCharacters(lineBreakIterator.lastCharacter(), lineBreakIterator.secondToLastCharacter());
92}
