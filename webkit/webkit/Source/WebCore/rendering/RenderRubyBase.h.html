35namespace WebCore {
36
37class RenderRubyRun;
38
39class RenderRubyBase final : public RenderBlockFlow {
40 WTF_MAKE_ISO_ALLOCATED(RenderRubyBase);
41public:
42 RenderRubyBase(Document&, RenderStyle&&);
43 virtual ~RenderRubyBase();
44
45 const char* renderName() const override { return "RenderRubyBase (anonymous)"; }
46
47 RenderRubyRun* rubyRun() const;
48
49 void setIsAfterExpansion(bool isAfterExpansion) { m_isAfterExpansion = isAfterExpansion; }
50 bool isAfterExpansion() { return m_isAfterExpansion; }
51
52 void setInitialOffset(float initialOffset) { m_initialOffset = initialOffset; }
53
54 void reset()
55 {
56 m_initialOffset = 0;
57 m_isAfterExpansion = true;
58 }
59
60 void cachePriorCharactersIfNeeded(const LazyLineBreakIterator&) override;
61
62private:
63 bool isRubyBase() const override { return true; }
64 bool isChildAllowed(const RenderObject&, const RenderStyle&) const override;
65 std::optional<TextAlignMode> overrideTextAlignmentForLine(bool endsWithSoftBreak) const override;
66 void adjustInlineDirectionLineBounds(int expansionOpportunityCount, float& logicalLeft, float& logicalWidth) const override;
67
68 float m_initialOffset;
69 unsigned m_isAfterExpansion : 1;
70};
71
72} // namespace WebCore
