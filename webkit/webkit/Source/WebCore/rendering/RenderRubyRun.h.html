31#pragma once
32
33#include "RenderBlockFlow.h"
34
35namespace WebCore {
36
37class RenderRubyBase;
38class RenderRubyText;
39
40// RenderRubyRuns are 'inline-block/table' like objects, and wrap a single pairing of a ruby base with its ruby text(s).
41// See RenderRuby.h for further comments on the structure
42
43class RenderRubyRun final : public RenderBlockFlow {
44 WTF_MAKE_ISO_ALLOCATED(RenderRubyRun);
45public:
46 RenderRubyRun(Document&, RenderStyle&&);
47 virtual ~RenderRubyRun();
48
49 bool hasRubyText() const;
50 bool hasRubyBase() const;
51 RenderRubyText* rubyText() const;
52 RenderRubyBase* rubyBase() const;
53
54 void layoutExcludedChildren(bool relayoutChildren) override;
55 void layout() override;
56 void layoutBlock(bool relayoutChildren, LayoutUnit pageHeight = 0_lu) override;
57
58 bool isChildAllowed(const RenderObject&, const RenderStyle&) const override;
59
60 RenderBlock* firstLineBlock() const override;
61
62 void getOverhang(bool firstLine, RenderObject* startRenderer, RenderObject* endRenderer, float& startOverhang, float& endOverhang) const;
63
64 static RenderPtr<RenderRubyRun> staticCreateRubyRun(const RenderObject* parentRuby);
65
66 void updatePriorContextFromCachedBreakIterator(LazyLineBreakIterator&) const;
67 void setCachedPriorCharacters(UChar last, UChar secondToLast)
68 {
69 m_lastCharacter = last;
70 m_secondToLastCharacter = secondToLast;
71 }
72 bool canBreakBefore(const LazyLineBreakIterator&) const;
73
74 RenderPtr<RenderRubyBase> createRubyBase() const;
75
76private:
77 bool isRubyRun() const override { return true; }
78 const char* renderName() const override { return "RenderRubyRun (anonymous)"; }
79 bool createsAnonymousWrapper() const override { return true; }
80 bool canDropAnonymousBlockChild() const override { return false; }
81
82private:
83 UChar m_lastCharacter;
84 UChar m_secondToLastCharacter;
85};
86
87} // namespace WebCore
88
89SPECIALIZE_TYPE_TRAITS_RENDER_OBJECT(RenderRubyRun, isRubyRun())
