26#pragma once
27
28#include "RenderPtr.h"
29#include "RenderStyleConstants.h"
30#include "Scrollbar.h"
31#include <wtf/HashMap.h>
32
33namespace WebCore {
34
35class Frame;
36class Element;
37class RenderBox;
38class RenderScrollbarPart;
39class RenderStyle;
40
41class RenderScrollbar final : public Scrollbar {
42public:
43 friend class Scrollbar;
44 static Ref<Scrollbar> createCustomScrollbar(ScrollableArea&, ScrollbarOrientation, Element*, Frame* owningFrame = nullptr);
45 virtual ~RenderScrollbar();
46
47 RenderBox* owningRenderer() const;
48
49 void paintPart(GraphicsContext&, ScrollbarPart, const IntRect&);
50
51 IntRect buttonRect(ScrollbarPart) const;
52 IntRect trackRect(int startLength, int endLength) const;
53 IntRect trackPieceRectWithMargins(ScrollbarPart, const IntRect&) const;
54
55 int minimumThumbLength() const;
56
57 float opacity() const;
58
59 bool isHiddenByStyle() const final;
60
61 std::unique_ptr<RenderStyle> getScrollbarPseudoStyle(ScrollbarPart, PseudoId) const;
62
63private:
64 RenderScrollbar(ScrollableArea&, ScrollbarOrientation, Element*, Frame*);
65
66 bool isOverlayScrollbar() const override { return false; }
67
68 void setParent(ScrollView*) override;
69 void setEnabled(bool) override;
70
71 void paint(GraphicsContext&, const IntRect& damageRect, Widget::SecurityOriginPaintPolicy, EventRegionContext*) override;
72
73 void setHoveredPart(ScrollbarPart) override;
74 void setPressedPart(ScrollbarPart) override;
75
76 void styleChanged() override;
77
78 void updateScrollbarParts();
79
80 void updateScrollbarPart(ScrollbarPart);
81
82 // This Scrollbar(Widget) may outlive the DOM which created it (during tear down),
83 // so we keep a reference to the Element which caused this custom scrollbar creation.
84 // This will not create a reference cycle as the Widget tree is owned by our containing
85 // FrameView which this Element pointer can in no way keep alive. See webkit bug 80610.
86 RefPtr<Element> m_ownerElement;
87
88 WeakPtr<Frame> m_owningFrame;
89 HashMap<unsigned, RenderPtr<RenderScrollbarPart>> m_parts;
90};
91
92} // namespace WebCore
93
94SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::RenderScrollbar)
95 static bool isType(const WebCore::Scrollbar& scrollbar) { return scrollbar.isCustomScrollbar(); }
96SPECIALIZE_TYPE_TRAITS_END()
