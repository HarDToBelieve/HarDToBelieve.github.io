26#pragma once
27
28#include "RenderBlock.h"
29#include "ScrollTypes.h"
30
31namespace WebCore {
32
33class RenderScrollbar;
34
35class RenderScrollbarPart final : public RenderBlock {
36 WTF_MAKE_ISO_ALLOCATED(RenderScrollbarPart);
37public:
38 RenderScrollbarPart(Document&, RenderStyle&&, RenderScrollbar* = nullptr, ScrollbarPart = NoPart);
39
40 virtual ~RenderScrollbarPart();
41
42 const char* renderName() const override { return "RenderScrollbarPart"; }
43
44 bool requiresLayer() const override { return false; }
45
46 void layout() override;
47
48 void paintIntoRect(GraphicsContext&, const LayoutPoint&, const LayoutRect&);
49
50 // Scrollbar parts needs to be rendered at device pixel boundaries.
51 LayoutUnit marginTop() const override { ASSERT(isIntegerValue(m_marginBox.top())); return m_marginBox.top(); }
52 LayoutUnit marginBottom() const override { ASSERT(isIntegerValue(m_marginBox.bottom())); return m_marginBox.bottom(); }
53 LayoutUnit marginLeft() const override { ASSERT(isIntegerValue(m_marginBox.left())); return m_marginBox.left(); }
54 LayoutUnit marginRight() const override { ASSERT(isIntegerValue(m_marginBox.right())); return m_marginBox.right(); }
55
56 RenderBox* rendererOwningScrollbar() const;
57
58private:
59 void styleDidChange(StyleDifference, const RenderStyle* oldStyle) override;
60 void imageChanged(WrappedImagePtr, const IntRect* = nullptr) override;
61
62 bool isRenderScrollbarPart() const override { return true; }
63 void computePreferredLogicalWidths() override;
64
65 void layoutHorizontalPart();
66 void layoutVerticalPart();
67
68 void computeScrollbarWidth();
69 void computeScrollbarHeight();
70
71 RenderScrollbar* m_scrollbar;
72 ScrollbarPart m_part;
73};
74
75} // namespace WebCore
76
77SPECIALIZE_TYPE_TRAITS_RENDER_OBJECT(RenderScrollbarPart, isRenderScrollbarPart())
