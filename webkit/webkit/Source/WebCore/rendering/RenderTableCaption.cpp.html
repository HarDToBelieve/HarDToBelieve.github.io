30RenderTableCaption::RenderTableCaption(Element& element, RenderStyle&& style)
31 : RenderBlockFlow(element, WTFMove(style))
32{
33}
34
35RenderTableCaption::~RenderTableCaption() = default;
36
37void RenderTableCaption::insertedIntoTree(IsInternalMove isInternalMove)
38{
39 RenderBlockFlow::insertedIntoTree(isInternalMove);
40 table()->addCaption(*this);
41}
42
43void RenderTableCaption::willBeRemovedFromTree(IsInternalMove isInternalMove)
44{
45 RenderBlockFlow::willBeRemovedFromTree(isInternalMove);
46 table()->removeCaption(*this);
47}
48
49RenderTable* RenderTableCaption::table() const
50{
51 return downcast<RenderTable>(parent());
52}
