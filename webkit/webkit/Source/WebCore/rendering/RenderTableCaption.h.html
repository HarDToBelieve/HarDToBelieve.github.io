22#include "RenderBlockFlow.h"

24namespace WebCore {

26class RenderTable;
27
28class RenderTableCaption final : public RenderBlockFlow {
29 WTF_MAKE_ISO_ALLOCATED(RenderTableCaption);
30public:
31 RenderTableCaption(Element&, RenderStyle&&);
32 virtual ~RenderTableCaption();
33
34 Element& element() const { return downcast<Element>(nodeForNonAnonymous()); }
35
36 LayoutUnit containingBlockLogicalWidthForContent() const override;
37
38private:
39 bool isTableCaption() const override { return true; }
40
41 void insertedIntoTree(IsInternalMove) override;
42 void willBeRemovedFromTree(IsInternalMove) override;
43
44 RenderTable* table() const;
45};
46
47inline LayoutUnit RenderTableCaption::containingBlockLogicalWidthForContent() const
48{
49 if (auto* containingBlock = this->containingBlock())
50 return containingBlock->logicalWidth();
51 return { };
52}
