151String capitalize(const String& string, UChar previousCharacter)
152{
153 // FIXME: Change this to use u_strToTitle instead of u_totitle and to consider locale.
154
155 unsigned length = string.length();
156 auto& stringImpl = *string.impl();
157
158 static_assert(String::MaxLength < std::numeric_limits<unsigned>::max(), "Must be able to add one without overflowing unsigned");
159
160 // Replace NO BREAK SPACE with a normal spaces since ICU does not treat it as a word separator.
161 Vector<UChar> stringWithPrevious(length + 1);
162 stringWithPrevious[0] = convertNoBreakSpaceToSpace(previousCharacter);
163 for (unsigned i = 1; i < length + 1; i++)
164 stringWithPrevious[i] = convertNoBreakSpaceToSpace(stringImpl[i - 1]);
165
166 auto* breakIterator = wordBreakIterator(StringView { stringWithPrevious.data(), length + 1 });
167 if (!breakIterator)
168 return string;
169
170 StringBuilder result;
171 result.reserveCapacity(length);
172
173 int32_t startOfWord = ubrk_first(breakIterator);
174 int32_t endOfWord;
175 for (endOfWord = ubrk_next(breakIterator); endOfWord != UBRK_DONE; startOfWord = endOfWord, endOfWord = ubrk_next(breakIterator)) {
176 if (startOfWord) // Do not append the first character, since it's the previous character, not from this string.
177 result.appendCharacter(u_totitle(stringImpl[startOfWord - 1]));
178 for (int i = startOfWord + 1; i < endOfWord; i++)
179 result.append(stringImpl[i - 1]);
180 }
181
182 return result == string ? string : result.toString();
183}

233bool RenderText::computeUseBackslashAsYenSymbol() const
234{
235 const RenderStyle& style = this->style();
236 const auto& fontDescription = style.fontDescription();
237 if (style.fontCascade().useBackslashAsYenSymbol())
238 return true;
239 if (fontDescription.isSpecifiedFont())
240 return false;
241 const TextEncoding* encoding = document().decoder() ? &document().decoder()->encoding() : 0;
242 if (encoding && encoding->backslashAsCurrencySymbol() != '\\')
243 return true;
244 return false;
245}

247void RenderText::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
248{
249 // There is no need to ever schedule repaints from a style change of a text run, since
250 // we already did this for the parent of the text run.
251 // We do have to schedule layouts, though, since a style change can force us to
252 // need to relayout.
253 if (diff == StyleDifference::Layout) {
254 setNeedsLayoutAndPrefWidthsRecalc();
255 m_knownToHaveNoOverflowAndNoFallbackFonts = false;
256 }
257
258 const RenderStyle& newStyle = style();
259 bool needsResetText = false;
260 if (!oldStyle) {
261 m_useBackslashAsYenSymbol = computeUseBackslashAsYenSymbol();
262 needsResetText = m_useBackslashAsYenSymbol;
263 } else if (oldStyle->fontCascade().useBackslashAsYenSymbol() != newStyle.fontCascade().useBackslashAsYenSymbol()) {
264 m_useBackslashAsYenSymbol = computeUseBackslashAsYenSymbol();
265 needsResetText = true;
266 }
267
268 if (!oldStyle || oldStyle->fontCascade() != newStyle.fontCascade())
269 m_canUseSimplifiedTextMeasuring = computeCanUseSimplifiedTextMeasuring();
270
271 TextTransform oldTransform = oldStyle ? oldStyle->textTransform() : TextTransform::None;
272 TextSecurity oldSecurity = oldStyle ? oldStyle->textSecurity() : TextSecurity::None;
273 if (needsResetText || oldTransform != newStyle.textTransform() || oldSecurity != newStyle.textSecurity())
274 RenderText::setText(originalText(), true);
275}

323// This function is similar in spirit to addLineBoxRects, but returns rectangles
324// which are annotated with additional state which helps the iPhone draw selections in its unique way.
325// Full annotations are added in this class.
326void RenderText::collectSelectionGeometries(Vector<SelectionGeometry>& rects, unsigned start, unsigned end)
327{
328 for (auto run = LayoutIntegration::firstTextRunFor(*this); run; run = run.traverseNextTextRun()) {
329 LayoutRect rect;
330 if (start <= run->start() && run->end() <= end)
331 rect = run->selectionRect(start, end);
332 else {
333 unsigned realEnd = std::min(run->end(), end);
334 rect = run->selectionRect(start, realEnd);
335 if (rect.isEmpty())
336 continue;
337 }
338
339 if (run.line()->legacyRootInlineBox() && run.line()->legacyRootInlineBox()->isFirstAfterPageBreak()) {
340 if (run->isHorizontal())
341 rect.shiftYEdgeTo(run.line()->lineBoxTop());
342 else
343 rect.shiftXEdgeTo(run.line()->lineBoxTop());
344 }
345
346 RenderBlock* containingBlock = this->containingBlock();
347 // Map rect, extended left to leftOffset, and right to rightOffset, through transforms to get minX and maxX.
348 LogicalSelectionOffsetCaches cache(*containingBlock);
349 LayoutUnit leftOffset = containingBlock->logicalLeftSelectionOffset(*containingBlock, LayoutUnit(run->logicalTop()), cache);
350 LayoutUnit rightOffset = containingBlock->logicalRightSelectionOffset(*containingBlock, LayoutUnit(run->logicalTop()), cache);
351 LayoutRect extentsRect = rect;
352 if (run->isHorizontal()) {
353 extentsRect.setX(leftOffset);
354 extentsRect.setWidth(rightOffset - leftOffset);
355 } else {
356 extentsRect.setY(leftOffset);
357 extentsRect.setHeight(rightOffset - leftOffset);
358 }
359 extentsRect = localToAbsoluteQuad(FloatRect(extentsRect)).enclosingBoundingBox();
360 if (!run->isHorizontal())
361 extentsRect = extentsRect.transposedRect();
362 bool isFirstOnLine = !run.previousOnLine();
363 bool isLastOnLine = !run.nextOnLine();
364 if (containingBlock->isRubyBase() || containingBlock->isRubyText())
365 isLastOnLine = !containingBlock->containingBlock()->inlineBoxWrapper()->nextOnLineExists();
366
367 bool containsStart = run->start() <= start && run->end() >= start;
368 bool containsEnd = run->start() <= end && run->end() >= end;
369
370 bool isFixed = false;
371 auto absoluteQuad = localToAbsoluteQuad(FloatRect(rect), UseTransforms, &isFixed);
372 bool boxIsHorizontal = !is<SVGInlineTextBox>(run->legacyInlineBox()) ? run->isHorizontal() : !style().isVerticalWritingMode();
373 // If the containing block is an inline element, we want to check the inlineBoxWrapper orientation
374 // to determine the orientation of the block. In this case we also use the inlineBoxWrapper to
375 // determine if the element is the last on the line.
376 if (containingBlock->inlineBoxWrapper()) {
377 if (containingBlock->inlineBoxWrapper()->isHorizontal() != boxIsHorizontal) {
378 boxIsHorizontal = containingBlock->inlineBoxWrapper()->isHorizontal();
379 isLastOnLine = !containingBlock->inlineBoxWrapper()->nextOnLineExists();
380 }
381 }
382
383 rects.append(SelectionGeometry(absoluteQuad, HTMLElement::selectionRenderingBehavior(textNode()), run->direction(), extentsRect.x(), extentsRect.maxX(), extentsRect.maxY(), 0, run->isLineBreak(), isFirstOnLine, isLastOnLine, containsStart, containsEnd, boxIsHorizontal, isFixed, containingBlock->isRubyText(), view().pageNumberForBlockProgressionOffset(absoluteQuad.enclosingBoundingBox().x())));
384 }
385}

396static IntRect ellipsisRectForTextRun(const LayoutIntegration::PathTextRun& run, unsigned start, unsigned end)
397{
398 // FIXME: No ellipsis support in modern path yet.
399 if (!run.legacyInlineBox())
400 return { };
401
402 auto& box = *run.legacyInlineBox();
403
404 unsigned short truncation = box.truncation();
405 if (truncation == cNoTruncation)
406 return { };
407
408 auto ellipsis = box.root().ellipsisBox();
409 if (!ellipsis)
410 return { };
411
412 int ellipsisStartPosition = std::max<int>(start - box.start(), 0);
413 int ellipsisEndPosition = std::min<int>(end - box.start(), box.len());
414
415 // The ellipsis should be considered to be selected if the end of
416 // the selection is past the beginning of the truncation and the
417 // beginning of the selection is before or at the beginning of the truncation.
418 if (ellipsisEndPosition < truncation && ellipsisStartPosition > truncation)
419 return { };
420
421 return ellipsis->selectionRect();
422}

424enum class ClippingOption { NoClipping, ClipToEllipsis };

426// FIXME: Unify with absoluteQuadsForRange.
427static Vector<FloatQuad> collectAbsoluteQuads(const RenderText& textRenderer, bool* wasFixed, ClippingOption clipping)
428{
429 Vector<FloatQuad> quads;
430 for (auto& run : LayoutIntegration::textRunsFor(textRenderer)) {
431 auto boundaries = boundariesForTextRun(run);
432
433 // Shorten the width of this text box if it ends in an ellipsis.
434 if (clipping == ClippingOption::ClipToEllipsis) {
435 auto ellipsisRect = ellipsisRectForTextRun(run, 0, textRenderer.text().length());
436 if (!ellipsisRect.isEmpty()) {
437 if (textRenderer.style().isHorizontalWritingMode())
438 boundaries.setWidth(ellipsisRect.maxX() - boundaries.x());
439 else
440 boundaries.setHeight(ellipsisRect.maxY() - boundaries.y());
441 }
442 }
443
444 quads.append(textRenderer.localToAbsoluteQuad(boundaries, UseTransforms, wasFixed));
445 }
446 return quads;
447}

459static FloatRect localQuadForTextRun(const LayoutIntegration::PathTextRun& run, unsigned start, unsigned end, bool useSelectionHeight)
460{
461 unsigned realEnd = std::min(run.end(), end);
462 LayoutRect boxSelectionRect = run.selectionRect(start, realEnd);
463 if (!boxSelectionRect.height())
464 return { };
465 if (useSelectionHeight)
466 return boxSelectionRect;
467
468 auto rect = run.rect();
469 if (run.isHorizontal()) {
470 boxSelectionRect.setHeight(rect.height());
471 boxSelectionRect.setY(rect.y());
472 } else {
473 boxSelectionRect.setWidth(rect.width());
474 boxSelectionRect.setX(rect.x());
475 }
476 return boxSelectionRect;
477}

479Vector<FloatQuad> RenderText::absoluteQuadsForRange(unsigned start, unsigned end, bool useSelectionHeight, bool ignoreEmptyTextSelections, bool* wasFixed) const
480{
481 // Work around signed/unsigned issues. This function takes unsigneds, and is often passed UINT_MAX
482 // to mean "all the way to the end". LegacyInlineTextBox coordinates are unsigneds, so changing this
483 // function to take ints causes various internal mismatches. But selectionRect takes ints, and
484 // passing UINT_MAX to it causes trouble. Ideally we'd change selectionRect to take unsigneds, but
485 // that would cause many ripple effects, so for now we'll just clamp our unsigned parameters to INT_MAX.
486 ASSERT(end == UINT_MAX || end <= INT_MAX);
487 ASSERT(start <= INT_MAX);
488 start = std::min(start, static_cast<unsigned>(INT_MAX));
489 end = std::min(end, static_cast<unsigned>(INT_MAX));
490
491 Vector<FloatQuad> quads;
492 for (auto& run : LayoutIntegration::textRunsFor(*this)) {
493 if (ignoreEmptyTextSelections && !run.isSelectable(start, end))
494 continue;
495 if (start <= run.start() && run.end() <= end) {
496 auto boundaries = boundariesForTextRun(run);
497
498 if (useSelectionHeight) {
499 LayoutRect selectionRect = run.selectionRect(start, end);
500 if (run.isHorizontal()) {
501 boundaries.setHeight(selectionRect.height());
502 boundaries.setY(selectionRect.y());
503 } else {
504 boundaries.setWidth(selectionRect.width());
505 boundaries.setX(selectionRect.x());
506 }
507 }
508 quads.append(localToAbsoluteQuad(boundaries, UseTransforms, wasFixed));
509 continue;
510 }
511 FloatRect rect = localQuadForTextRun(run, start, end, useSelectionHeight);
512 if (!rect.isZero())
513 quads.append(localToAbsoluteQuad(rect, UseTransforms, wasFixed));
514 }
515 return quads;
516}

523enum ShouldAffinityBeDownstream { AlwaysDownstream, AlwaysUpstream, UpstreamIfPositionIsNotAtStart };
524
525static bool lineDirectionPointFitsInBox(int pointLineDirection, const LayoutIntegration::TextRunIterator& textRun, ShouldAffinityBeDownstream& shouldAffinityBeDownstream)
526{
527 shouldAffinityBeDownstream = AlwaysDownstream;
528
529 // the x coordinate is equal to the left edge of this box
530 // the affinity must be downstream so the position doesn't jump back to the previous line
531 // except when box is the first box in the line
532 if (pointLineDirection <= textRun->logicalLeft()) {
533 shouldAffinityBeDownstream = !textRun.previousOnLine() ? UpstreamIfPositionIsNotAtStart : AlwaysDownstream;
534 return true;
535 }
536
537#if !PLATFORM(IOS_FAMILY)
538 // and the x coordinate is to the left of the right edge of this box
539 // check to see if position goes in this box
540 if (pointLineDirection < textRun->logicalRight()) {
541 shouldAffinityBeDownstream = UpstreamIfPositionIsNotAtStart;
542 return true;
543 }
544#endif
545
546 // box is first on line
547 // and the x coordinate is to the left of the first text box left edge
548 if (!textRun.previousOnLineIgnoringLineBreak() && pointLineDirection < textRun->logicalLeft())
549 return true;
550
551 if (!textRun.nextOnLineIgnoringLineBreak()) {
552 // box is last on line
553 // and the x coordinate is to the right of the last text box right edge
554 // generate VisiblePosition, use Affinity::Upstream affinity if possible
555 shouldAffinityBeDownstream = UpstreamIfPositionIsNotAtStart;
556 return true;
557 }
558
559 return false;
560}

562static VisiblePosition createVisiblePositionForBox(const LayoutIntegration::RunIterator& run, unsigned offset, ShouldAffinityBeDownstream shouldAffinityBeDownstream)
563{
564 auto affinity = VisiblePosition::defaultAffinity;
565 switch (shouldAffinityBeDownstream) {
566 case AlwaysDownstream:
567 affinity = Affinity::Downstream;
568 break;
569 case AlwaysUpstream:
570 affinity = Affinity::Upstream;
571 break;
572 case UpstreamIfPositionIsNotAtStart:
573 affinity = offset > run->minimumCaretOffset() ? Affinity::Upstream : Affinity::Downstream;
574 break;
575 }
576 return run->renderer().createVisiblePosition(offset, affinity);
577}

579static VisiblePosition createVisiblePositionAfterAdjustingOffsetForBiDi(const LayoutIntegration::TextRunIterator& run, unsigned offset, ShouldAffinityBeDownstream shouldAffinityBeDownstream)
580{
581 if (offset && offset < run->length())
582 return createVisiblePositionForBox(run, run->start() + offset, shouldAffinityBeDownstream);
583
584 bool positionIsAtStartOfBox = !offset;
585 if (positionIsAtStartOfBox == run->isLeftToRightDirection()) {
586 // offset is on the left edge
587
588 auto previousRun = run.previousOnLineIgnoringLineBreak();
589 if ((previousRun && previousRun->bidiLevel() == run->bidiLevel())
590 || run->renderer().containingBlock()->style().direction() == run->direction()) // FIXME: left on 12CBA
591 return createVisiblePositionForBox(run, run->leftmostCaretOffset(), shouldAffinityBeDownstream);
592
593 if (previousRun && previousRun->bidiLevel() > run->bidiLevel()) {
594 // e.g. left of B in aDC12BAb
595 auto leftmostRun = previousRun;
596 for (; previousRun; previousRun.traversePreviousOnLineIgnoringLineBreak()) {
597 if (previousRun->bidiLevel() <= run->bidiLevel())
598 break;
599 leftmostRun = previousRun;
600 }
601 return createVisiblePositionForBox(leftmostRun, leftmostRun->rightmostCaretOffset(), shouldAffinityBeDownstream);
602 }
603
604 if (!previousRun || previousRun->bidiLevel() < run->bidiLevel()) {
605 // e.g. left of D in aDC12BAb
606 LayoutIntegration::RunIterator rightmostRun = run;
607 for (auto nextRun = run.nextOnLineIgnoringLineBreak(); nextRun; nextRun.traverseNextOnLineIgnoringLineBreak()) {
608 if (nextRun->bidiLevel() < run->bidiLevel())
609 break;
610 rightmostRun = nextRun;
611 }
612 return createVisiblePositionForBox(rightmostRun,
613 run->isLeftToRightDirection() ? rightmostRun->maximumCaretOffset() : rightmostRun->minimumCaretOffset(), shouldAffinityBeDownstream);
614 }
615
616 return createVisiblePositionForBox(run, run->rightmostCaretOffset(), shouldAffinityBeDownstream);
617 }
618
619 auto nextRun = run.nextOnLineIgnoringLineBreak();
620 if ((nextRun && nextRun->bidiLevel() == run->bidiLevel())
621 || run->renderer().containingBlock()->style().direction() == run->direction())
622 return createVisiblePositionForBox(run, run->rightmostCaretOffset(), shouldAffinityBeDownstream);
623
624 // offset is on the right edge
625 if (nextRun && nextRun->bidiLevel() > run->bidiLevel()) {
626 // e.g. right of C in aDC12BAb
627 auto rightmostRun = nextRun;
628 for (; nextRun; nextRun.traverseNextOnLineIgnoringLineBreak()) {
629 if (nextRun->bidiLevel() <= run->bidiLevel())
630 break;
631 rightmostRun = nextRun;
632 }
633
634 return createVisiblePositionForBox(rightmostRun, rightmostRun->leftmostCaretOffset(), shouldAffinityBeDownstream);
635 }
636
637 if (!nextRun || nextRun->bidiLevel() < run->bidiLevel()) {
638 // e.g. right of A in aDC12BAb
639 LayoutIntegration::RunIterator leftmostRun = run;
640 for (auto previousRun = run.previousOnLineIgnoringLineBreak(); previousRun; previousRun.traversePreviousOnLineIgnoringLineBreak()) {
641 if (previousRun->bidiLevel() < run->bidiLevel())
642 break;
643 leftmostRun = previousRun;
644 }
645
646 return createVisiblePositionForBox(leftmostRun,
647 run->isLeftToRightDirection() ? leftmostRun->minimumCaretOffset() : leftmostRun->maximumCaretOffset(), shouldAffinityBeDownstream);
648 }
649
650 return createVisiblePositionForBox(run, run->leftmostCaretOffset(), shouldAffinityBeDownstream);
651}

654VisiblePosition RenderText::positionForPoint(const LayoutPoint& point, const RenderFragmentContainer*)
655{
656 auto firstRun = LayoutIntegration::firstTextRunFor(*this);
657
658 if (!firstRun || !text().length())
659 return createVisiblePosition(0, Affinity::Downstream);
660
661 LayoutUnit pointLineDirection = firstRun->isHorizontal() ? point.x() : point.y();
662 LayoutUnit pointBlockDirection = firstRun->isHorizontal() ? point.y() : point.x();
663 bool blocksAreFlipped = style().isFlippedBlocksWritingMode();
664
665 LayoutIntegration::TextRunIterator lastRun;
666 for (auto run = firstRun; run; run.traverseNextTextRun()) {
667 if (run->isLineBreak() && !run.previousOnLine() && run.nextOnLine() && !run.nextOnLine()->isLineBreak())
668 run.traverseNextTextRun();
669
670 auto line = run.line();
671 LayoutUnit top = std::min(line->selectionTopForHitTesting(), line->top());
672 if (pointBlockDirection > top || (!blocksAreFlipped && pointBlockDirection == top)) {
673 LayoutUnit bottom = line->selectionBottom();
674 if (auto nextLine = line.next())
675 bottom = std::min(bottom, nextLine->top());
676
677 if (pointBlockDirection < bottom || (blocksAreFlipped && pointBlockDirection == bottom)) {
678 ShouldAffinityBeDownstream shouldAffinityBeDownstream;
679#if PLATFORM(IOS_FAMILY)
680 if (pointLineDirection != run->logicalLeft() && point.x() < run->rect().x() + run->logicalWidth()) {
681 int half = run->rect().x() + run->logicalWidth() / 2;
682 auto affinity = point.x() < half ? Affinity::Downstream : Affinity::Upstream;
683 return createVisiblePosition(run->offsetForPosition(pointLineDirection) + run->start(), affinity);
684 }
685#endif
686 if (lineDirectionPointFitsInBox(pointLineDirection, run, shouldAffinityBeDownstream))
687 return createVisiblePositionAfterAdjustingOffsetForBiDi(run, run->offsetForPosition(pointLineDirection), shouldAffinityBeDownstream);
688 }
689 }
690 lastRun = run;
691 }
692
693 if (lastRun) {
694 ShouldAffinityBeDownstream shouldAffinityBeDownstream;
695 lineDirectionPointFitsInBox(pointLineDirection, lastRun, shouldAffinityBeDownstream);
696 return createVisiblePositionAfterAdjustingOffsetForBiDi(lastRun, lastRun->offsetForPosition(pointLineDirection) + lastRun->start(), shouldAffinityBeDownstream);
697 }
698 return createVisiblePosition(0, Affinity::Downstream);
699}

788RenderText::Widths RenderText::trimmedPreferredWidths(float leadWidth, bool& stripFrontSpaces)
789{
790 auto& style = this->style();
791 bool collapseWhiteSpace = style.collapseWhiteSpace();
792
793 if (!collapseWhiteSpace)
794 stripFrontSpaces = false;
795
796 if (m_hasTab || preferredLogicalWidthsDirty())
797 computePreferredLogicalWidths(leadWidth);
798
799 Widths widths;
800
801 widths.beginWS = !stripFrontSpaces && m_hasBeginWS;
802 widths.endWS = m_hasEndWS;
803
804 unsigned length = this->length();
805
806 if (!length || (stripFrontSpaces && text().isAllSpecialCharacters<isHTMLSpace>()))
807 return widths;
808
809 widths.min = m_minWidth;
810 widths.max = m_maxWidth;
811
812 widths.beginMin = m_beginMinWidth;
813 widths.endMin = m_endMinWidth;
814
815 widths.hasBreakableChar = m_hasBreakableChar;
816 widths.hasBreak = m_hasBreak;
817
818 if (text()[0] == ' ' || (text()[0] == '\n' && !style.preserveNewline()) || text()[0] == '\t') {
819 auto& font = style.fontCascade(); // FIXME: This ignores first-line.
820 if (stripFrontSpaces)
821 widths.max -= font.width(RenderBlock::constructTextRun(&space, 1, style));
822 else
823 widths.max += font.wordSpacing();
824 }
825
826 stripFrontSpaces = collapseWhiteSpace && m_hasEndWS;
827
828 if (!style.autoWrap() || widths.min > widths.max)
829 widths.min = widths.max;
830
831 // Compute our max widths by scanning the string for newlines.
832 if (widths.hasBreak) {
833 auto& font = style.fontCascade(); // FIXME: This ignores first-line.
834 bool firstLine = true;
835 widths.beginMax = widths.max;
836 widths.endMax = widths.max;
837 for (unsigned i = 0; i < length; i++) {
838 unsigned lineLength = 0;
839 while (i + lineLength < length && text()[i + lineLength] != '\n')
840 lineLength++;
841
842 if (lineLength) {
843 widths.endMax = widthFromCache(font, i, lineLength, leadWidth + widths.endMax, 0, 0, style);
844 if (firstLine) {
845 firstLine = false;
846 leadWidth = 0;
847 widths.beginMax = widths.endMax;
848 }
849 i += lineLength;
850 } else if (firstLine) {
851 widths.beginMax = 0;
852 firstLine = false;
853 leadWidth = 0;
854 }
855
856 if (i == length - 1) {
857 // A <pre> run that ends with a newline, as in, e.g.,
858 // <pre>Some text\n\n<span>More text</pre>
859 widths.endMax = 0;
860 }
861 }
862 }
863
864 return widths;
865}

922static float maxWordFragmentWidth(RenderText& renderer, const RenderStyle& style, const FontCascade& font, StringView word, unsigned minimumPrefixLength, unsigned minimumSuffixLength, unsigned& suffixStart, HashSet<const Font*>& fallbackFonts, GlyphOverflow& glyphOverflow)
923{
924 suffixStart = 0;
925 if (word.length() <= minimumSuffixLength)
926 return 0;
927
928 Vector<int, 8> hyphenLocations;
929 ASSERT(word.length() >= minimumSuffixLength);
930 unsigned hyphenLocation = word.length() - minimumSuffixLength;
931 while ((hyphenLocation = lastHyphenLocation(word, hyphenLocation, style.computedLocale())) >= std::max(minimumPrefixLength, 1U))
932 hyphenLocations.append(hyphenLocation);
933
934 if (hyphenLocations.isEmpty())
935 return 0;
936
937 hyphenLocations.reverse();
938
939 // FIXME: Breaking the string at these places in the middle of words is completely broken with complex text.
940 float minimumFragmentWidthToConsider = font.pixelSize() * 5 / 4 + hyphenWidth(renderer, font);
941 float maxFragmentWidth = 0;
942 for (size_t k = 0; k < hyphenLocations.size(); ++k) {
943 int fragmentLength = hyphenLocations[k] - suffixStart;
944 StringBuilder fragmentWithHyphen;
945 fragmentWithHyphen.append(word.substring(suffixStart, fragmentLength));
946 fragmentWithHyphen.append(style.hyphenString());
947
948 TextRun run = RenderBlock::constructTextRun(fragmentWithHyphen.toString(), style);
949 run.setCharacterScanForCodePath(!renderer.canUseSimpleFontCodePath());
950 float fragmentWidth = font.width(run, &fallbackFonts, &glyphOverflow);
951
952 // Narrow prefixes are ignored. See tryHyphenating in RenderBlockLineLayout.cpp.
953 if (fragmentWidth <= minimumFragmentWidthToConsider)
954 continue;
955
956 suffixStart += fragmentLength;
957 maxFragmentWidth = std::max(maxFragmentWidth, fragmentWidth);
958 }
959
960 return maxFragmentWidth;
961}

963void RenderText::computePreferredLogicalWidths(float leadWidth, HashSet<const Font*>& fallbackFonts, GlyphOverflow& glyphOverflow)
964{
965 ASSERT(m_hasTab || preferredLogicalWidthsDirty() || !m_knownToHaveNoOverflowAndNoFallbackFonts);
966
967 m_minWidth = 0;
968 m_beginMinWidth = 0;
969 m_endMinWidth = 0;
970 m_maxWidth = 0;
971
972 float currMaxWidth = 0;
973 m_hasBreakableChar = false;
974 m_hasBreak = false;
975 m_hasTab = false;
976 m_hasBeginWS = false;
977 m_hasEndWS = false;
978
979 auto& style = this->style();
980 auto& font = style.fontCascade(); // FIXME: This ignores first-line.
981 float wordSpacing = font.wordSpacing();
982 auto& string = text();
983 unsigned length = string.length();
984 auto iteratorMode = mapLineBreakToIteratorMode(style.lineBreak());
985 LazyLineBreakIterator breakIterator(string, style.computedLocale(), iteratorMode);
986 bool needsWordSpacing = false;
987 bool ignoringSpaces = false;
988 bool isSpace = false;
989 bool firstWord = true;
990 bool firstLine = true;
991 std::optional<unsigned> nextBreakable;
992 unsigned lastWordBoundary = 0;
993
994 WordTrailingSpace wordTrailingSpace(style);
995 // If automatic hyphenation is allowed, we keep track of the width of the widest word (or word
996 // fragment) encountered so far, and only try hyphenating words that are wider.
997 float maxWordWidth = std::numeric_limits<float>::max();
998 unsigned minimumPrefixLength = 0;
999 unsigned minimumSuffixLength = 0;
1000 if (style.hyphens() == Hyphens::Auto && canHyphenate(style.computedLocale())) {
1001 maxWordWidth = 0;
1002
1003 // Map 'hyphenate-limit-{before,after}: auto;' to 2.
1004 auto before = style.hyphenationLimitBefore();
1005 minimumPrefixLength = before < 0 ? 2 : before;
1006
1007 auto after = style.hyphenationLimitAfter();
1008 minimumSuffixLength = after < 0 ? 2 : after;
1009 }
1010
1011 std::optional<int> firstGlyphLeftOverflow;
1012
1013 bool breakNBSP = style.autoWrap() && style.nbspMode() == NBSPMode::Space;
1014
1015 // Note the deliberate omission of word-wrap and overflow-wrap from this breakAll check. Those
1016 // do not affect minimum preferred sizes. Note that break-word is a non-standard value for
1017 // word-break, but we support it as though it means break-all.
1018 bool breakAnywhere = style.lineBreak() == LineBreak::Anywhere && style.autoWrap();
1019 bool breakAll = (style.wordBreak() == WordBreak::BreakAll || style.wordBreak() == WordBreak::BreakWord) && style.autoWrap();
1020 bool keepAllWords = style.wordBreak() == WordBreak::KeepAll;
1021 bool canUseLineBreakShortcut = iteratorMode == LineBreakIteratorMode::Default;
1022
1023 for (unsigned i = 0; i < length; i++) {
1024 UChar c = string[i];
1025
1026 bool previousCharacterIsSpace = isSpace;
1027
1028 bool isNewline = false;
1029 if (c == '\n') {
1030 if (style.preserveNewline()) {
1031 m_hasBreak = true;
1032 isNewline = true;
1033 isSpace = false;
1034 } else
1035 isSpace = true;
1036 } else if (c == '\t') {
1037 if (!style.collapseWhiteSpace()) {
1038 m_hasTab = true;
1039 isSpace = false;
1040 } else
1041 isSpace = true;
1042 } else
1043 isSpace = c == ' ';
1044
1045 if ((isSpace || isNewline) && !i)
1046 m_hasBeginWS = true;
1047 if ((isSpace || isNewline) && i == length - 1)
1048 m_hasEndWS = true;
1049
1050 ignoringSpaces |= style.collapseWhiteSpace() && previousCharacterIsSpace && isSpace;
1051 ignoringSpaces &= isSpace;
1052
1053 // Ignore spaces and soft hyphens
1054 if (ignoringSpaces) {
1055 ASSERT(lastWordBoundary == i);
1056 lastWordBoundary++;
1057 continue;
1058 } else if (c == softHyphen && style.hyphens() != Hyphens::None) {
1059 ASSERT(i >= lastWordBoundary);
1060 currMaxWidth += widthFromCache(font, lastWordBoundary, i - lastWordBoundary, leadWidth + currMaxWidth, &fallbackFonts, &glyphOverflow, style);
1061 if (!firstGlyphLeftOverflow)
1062 firstGlyphLeftOverflow = glyphOverflow.left;
1063 lastWordBoundary = i + 1;
1064 continue;
1065 }
1066
1067 bool hasBreak = breakAll || isBreakable(breakIterator, i, nextBreakable, breakNBSP, canUseLineBreakShortcut, keepAllWords, breakAnywhere);
1068 bool betweenWords = true;
1069 unsigned j = i;
1070 while (c != '\n' && !isSpaceAccordingToStyle(c, style) && c != '\t' && (c != softHyphen || style.hyphens() == Hyphens::None)) {
1071 j++;
1072 if (j == length)
1073 break;
1074 c = string[j];
1075 if (isBreakable(breakIterator, j, nextBreakable, breakNBSP, canUseLineBreakShortcut, keepAllWords, breakAnywhere) && characterAt(j - 1) != softHyphen)
1076 break;
1077 if (breakAll) {
1078 betweenWords = false;
1079 break;
1080 }
1081 }
1082
1083 unsigned wordLen = j - i;
1084 if (wordLen) {
1085 float currMinWidth = 0;
1086 bool isSpace = (j < length) && isSpaceAccordingToStyle(c, style);
1087 float w;
1088 std::optional<float> wordTrailingSpaceWidth;
1089 if (isSpace)
1090 wordTrailingSpaceWidth = wordTrailingSpace.width(fallbackFonts);
1091 if (wordTrailingSpaceWidth)
1092 w = widthFromCache(font, i, wordLen + 1, leadWidth + currMaxWidth, &fallbackFonts, &glyphOverflow, style) - wordTrailingSpaceWidth.value();
1093 else {
1094 w = widthFromCache(font, i, wordLen, leadWidth + currMaxWidth, &fallbackFonts, &glyphOverflow, style);
1095 if (c == softHyphen && style.hyphens() != Hyphens::None)
1096 currMinWidth = hyphenWidth(*this, font);
1097 }
1098
1099 if (w > maxWordWidth) {
1100 unsigned suffixStart;
1101 float maxFragmentWidth = maxWordFragmentWidth(*this, style, font, StringView(string).substring(i, wordLen), minimumPrefixLength, minimumSuffixLength, suffixStart, fallbackFonts, glyphOverflow);
1102
1103 if (suffixStart) {
1104 float suffixWidth;
1105 std::optional<float> wordTrailingSpaceWidth;
1106 if (isSpace)
1107 wordTrailingSpaceWidth = wordTrailingSpace.width(fallbackFonts);
1108 if (wordTrailingSpaceWidth)
1109 suffixWidth = widthFromCache(font, i + suffixStart, wordLen - suffixStart + 1, leadWidth + currMaxWidth, 0, 0, style) - wordTrailingSpaceWidth.value();
1110 else
1111 suffixWidth = widthFromCache(font, i + suffixStart, wordLen - suffixStart, leadWidth + currMaxWidth, 0, 0, style);
1112
1113 maxFragmentWidth = std::max(maxFragmentWidth, suffixWidth);
1114
1115 currMinWidth += maxFragmentWidth - w;
1116 maxWordWidth = std::max(maxWordWidth, maxFragmentWidth);
1117 } else
1118 maxWordWidth = w;
1119 }
1120
1121 if (!firstGlyphLeftOverflow)
1122 firstGlyphLeftOverflow = glyphOverflow.left;
1123 currMinWidth += w;
1124 if (betweenWords) {
1125 if (lastWordBoundary == i)
1126 currMaxWidth += w;
1127 else {
1128 ASSERT(j >= lastWordBoundary);
1129 currMaxWidth += widthFromCache(font, lastWordBoundary, j - lastWordBoundary, leadWidth + currMaxWidth, &fallbackFonts, &glyphOverflow, style);
1130 }
1131 lastWordBoundary = j;
1132 }
1133
1134 bool isCollapsibleWhiteSpace = (j < length) && style.isCollapsibleWhiteSpace(c);
1135 if (j < length && style.autoWrap())
1136 m_hasBreakableChar = true;
1137
1138 // Add in wordSpacing to our currMaxWidth, but not if this is the last word on a line or the
1139 // last word in the run.
1140 if ((isSpace || isCollapsibleWhiteSpace) && !containsOnlyHTMLWhitespace(j, length - j))
1141 currMaxWidth += wordSpacing;
1142
1143 if (firstWord) {
1144 firstWord = false;
1145 // If the first character in the run is breakable, then we consider ourselves to have a beginning
1146 // minimum width of 0, since a break could occur right before our run starts, preventing us from ever
1147 // being appended to a previous text run when considering the total minimum width of the containing block.
1148 if (hasBreak)
1149 m_hasBreakableChar = true;
1150 m_beginMinWidth = hasBreak ? 0 : currMinWidth;
1151 }
1152 m_endMinWidth = currMinWidth;
1153
1154 m_minWidth = std::max(currMinWidth, m_minWidth);
1155
1156 i += wordLen - 1;
1157 } else {
1158 // Nowrap can never be broken, so don't bother setting the
1159 // breakable character boolean. Pre can only be broken if we encounter a newline.
1160 if (style.autoWrap() || isNewline)
1161 m_hasBreakableChar = true;
1162
1163 if (isNewline) { // Only set if preserveNewline was true and we saw a newline.
1164 if (firstLine) {
1165 firstLine = false;
1166 leadWidth = 0;
1167 if (!style.autoWrap())
1168 m_beginMinWidth = currMaxWidth;
1169 }
1170
1171 if (currMaxWidth > m_maxWidth)
1172 m_maxWidth = currMaxWidth;
1173 currMaxWidth = 0;
1174 } else {
1175 TextRun run = RenderBlock::constructTextRun(*this, i, 1, style);
1176 run.setTabSize(!style.collapseWhiteSpace(), style.tabSize());
1177 run.setXPos(leadWidth + currMaxWidth);
1178
1179 currMaxWidth += font.width(run, &fallbackFonts);
1180 glyphOverflow.right = 0;
1181 needsWordSpacing = isSpace && !previousCharacterIsSpace && i == length - 1;
1182 }
1183 ASSERT(lastWordBoundary == i);
1184 lastWordBoundary++;
1185 }
1186 }
1187
1188 glyphOverflow.left = firstGlyphLeftOverflow.value_or(glyphOverflow.left);
1189
1190 if ((needsWordSpacing && length > 1) || (ignoringSpaces && !firstWord))
1191 currMaxWidth += wordSpacing;
1192
1193 m_maxWidth = std::max(currMaxWidth, m_maxWidth);
1194
1195 if (!style.autoWrap())
1196 m_minWidth = m_maxWidth;
1197
1198 if (style.whiteSpace() == WhiteSpace::Pre) {
1199 if (firstLine)
1200 m_beginMinWidth = m_maxWidth;
1201 m_endMinWidth = currMaxWidth;
1202 }
1203
1204 setPreferredLogicalWidthsDirty(false);
1205}

1387void RenderText::secureText(UChar maskingCharacter)
1388{
1389 // This hides the text by replacing all the characters with the masking character.
1390 // Offsets within the hidden text have to match offsets within the original text
1391 // to handle things like carets and selection, so this won't work right if any
1392 // of the characters are surrogate pairs or combining marks. Thus, this function
1393 // does not attempt to handle either of those.
1394
1395 unsigned length = text().length();
1396 if (!length)
1397 return;
1398
1399 UChar characterToReveal = 0;
1400 unsigned revealedCharactersOffset = 0;
1401
1402 if (SecureTextTimer* timer = secureTextTimers().get(this)) {
1403 // We take the offset out of the timer to make this one-shot. We count on this being called only once.
1404 // If it's called a second time we assume the text is different and a character should not be revealed.
1405 revealedCharactersOffset = timer->takeOffsetAfterLastTypedCharacter();
1406 if (revealedCharactersOffset && revealedCharactersOffset <= length)
1407 characterToReveal = text()[--revealedCharactersOffset];
1408 }
1409
1410 UChar* characters;
1411 m_text = String::createUninitialized(length, characters);
1412
1413 for (unsigned i = 0; i < length; ++i)
1414 characters[i] = maskingCharacter;
1415 if (characterToReveal)
1416 characters[revealedCharactersOffset] = characterToReveal;
1417}
