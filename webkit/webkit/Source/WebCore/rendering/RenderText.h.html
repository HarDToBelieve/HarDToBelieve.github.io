275inline const RenderStyle* RenderText::getCachedPseudoStyle(PseudoId pseudoId, const RenderStyle* parentStyle) const
276{
277 // Pseudostyle is associated with an element, so ascend the tree until we find a non-anonymous ancestor.
278 if (auto* ancestor = firstNonAnonymousAncestor())
279 return ancestor->getCachedPseudoStyle(pseudoId, parentStyle);
280 return nullptr;
281}
