88void RenderThemeCocoa::adjustApplePayButtonStyle(RenderStyle& style, const Element*) const
89{
90 if (style.applePayButtonType() == ApplePayButtonType::Plain)
91 style.setMinWidth(Length(applePayButtonPlainMinimumWidth, LengthType::Fixed));
92 else
93 style.setMinWidth(Length(applePayButtonMinimumWidth, LengthType::Fixed));
94 style.setMinHeight(Length(applePayButtonMinimumHeight, LengthType::Fixed));
95
96 if (!style.hasExplicitlySetBorderRadius()) {
97 auto cornerRadius = PAL::get_PassKit_PKApplePayButtonDefaultCornerRadius();
98 style.setBorderRadius({ { cornerRadius, LengthType::Fixed }, { cornerRadius, LengthType::Fixed } });
99 }
100}

152bool RenderThemeCocoa::paintApplePayButton(const RenderObject& renderer, const PaintInfo& paintInfo, const IntRect& paintRect)
153{
154 GraphicsContextStateSaver stateSaver(paintInfo.context());
155
156 paintInfo.context().setShouldSmoothFonts(true);
157 paintInfo.context().scale(FloatSize(1, -1));
158
159 auto& style = renderer.style();
160 auto largestCornerRadius = std::max<CGFloat>({
161 floatValueForLength(style.borderTopLeftRadius().height, paintRect.height()),
162 floatValueForLength(style.borderTopLeftRadius().width, paintRect.width()),
163 floatValueForLength(style.borderTopRightRadius().height, paintRect.height()),
164 floatValueForLength(style.borderTopRightRadius().width, paintRect.width()),
165 floatValueForLength(style.borderBottomLeftRadius().height, paintRect.height()),
166 floatValueForLength(style.borderBottomLeftRadius().width, paintRect.width()),
167 floatValueForLength(style.borderBottomRightRadius().height, paintRect.height()),
168 floatValueForLength(style.borderBottomRightRadius().width, paintRect.width())
169 });
170
171 PKDrawApplePayButtonWithCornerRadius(paintInfo.context().platformContext(), CGRectMake(paintRect.x(), -paintRect.maxY(), paintRect.width(), paintRect.height()), 1.0, largestCornerRadius, toPKPaymentButtonType(style.applePayButtonType()), toPKPaymentButtonStyle(style.applePayButtonStyle()), style.computedLocale());
172 return false;
173}

210String RenderThemeCocoa::mediaControlsFormattedStringForDuration(const double durationInSeconds)
211{
212 if (!std::isfinite(durationInSeconds))
213 return WEB_UI_STRING("indefinite time", "accessibility help text for an indefinite media controller time value");
214
215 BEGIN_BLOCK_OBJC_EXCEPTIONS
216 if (!m_durationFormatter) {
217 m_durationFormatter = adoptNS([NSDateComponentsFormatter new]);
218 m_durationFormatter.get().unitsStyle = NSDateComponentsFormatterUnitsStyleFull;
219 m_durationFormatter.get().allowedUnits = NSCalendarUnitHour | NSCalendarUnitMinute | NSCalendarUnitSecond;
220 m_durationFormatter.get().formattingContext = NSFormattingContextStandalone;
221 m_durationFormatter.get().maximumUnitCount = 2;
222 }
223 return [m_durationFormatter.get() stringFromTimeInterval:durationInSeconds];
224 END_BLOCK_OBJC_EXCEPTIONS
225}

229FontCascadeDescription& RenderThemeCocoa::cachedSystemFontDescription(CSSValueID valueID) const
230{
231 static auto fontDescriptions = makeNeverDestroyed<std::array<FontCascadeDescription, 17>>({ });
232
233 ASSERT(std::all_of(std::begin(fontDescriptions.get()), std::end(fontDescriptions.get()), [](auto& description) {
234 return !description.isAbsoluteSize();
235 }));
236
237 switch (valueID) {
238 case CSSValueAppleSystemHeadline:
239 return fontDescriptions.get()[0];
240 case CSSValueAppleSystemBody:
241 return fontDescriptions.get()[1];
242 case CSSValueAppleSystemTitle0:
243 return fontDescriptions.get()[2];
244 case CSSValueAppleSystemTitle1:
245 return fontDescriptions.get()[3];
246 case CSSValueAppleSystemTitle2:
247 return fontDescriptions.get()[4];
248 case CSSValueAppleSystemTitle3:
249 return fontDescriptions.get()[5];
250 case CSSValueAppleSystemTitle4:
251 return fontDescriptions.get()[6];
252 case CSSValueAppleSystemSubheadline:
253 return fontDescriptions.get()[7];
254 case CSSValueAppleSystemFootnote:
255 return fontDescriptions.get()[8];
256 case CSSValueAppleSystemCaption1:
257 return fontDescriptions.get()[9];
258 case CSSValueAppleSystemCaption2:
259 return fontDescriptions.get()[10];
260 case CSSValueAppleSystemShortHeadline:
261 return fontDescriptions.get()[11];
262 case CSSValueAppleSystemShortBody:
263 return fontDescriptions.get()[12];
264 case CSSValueAppleSystemShortSubheadline:
265 return fontDescriptions.get()[13];
266 case CSSValueAppleSystemShortFootnote:
267 return fontDescriptions.get()[14];
268 case CSSValueAppleSystemShortCaption1:
269 return fontDescriptions.get()[15];
270 case CSSValueAppleSystemTallBody:
271 return fontDescriptions.get()[16];
272 default:
273 return RenderTheme::cachedSystemFontDescription(valueID);
274 }

277static inline FontSelectionValue cssWeightOfSystemFont(CTFontRef font)
278{
279 auto traits = adoptCF(CTFontCopyTraits(font));
280 CFNumberRef resultRef = (CFNumberRef)CFDictionaryGetValue(traits.get(), kCTFontWeightTrait);
281 float result = 0;
282 CFNumberGetValue(resultRef, kCFNumberFloatType, &result);
283 // These numbers were experimentally gathered from weights of the system font.
284 static constexpr float weightThresholds[] = { -0.6, -0.365, -0.115, 0.130, 0.235, 0.350, 0.5, 0.7 };
285 for (unsigned i = 0; i < WTF_ARRAY_LENGTH(weightThresholds); ++i) {
286 if (result < weightThresholds[i])
287 return FontSelectionValue((static_cast<int>(i) + 1) * 100);
288 }
289 return FontSelectionValue(900);
290}

292void RenderThemeCocoa::updateCachedSystemFontDescription(CSSValueID valueID, FontCascadeDescription& fontDescription) const
293{
294 auto cocoaFontClass = [] {
295#if PLATFORM(IOS_FAMILY)
296 return PAL::getUIFontClass();
297#else
298 return NSFont.class;
299#endif
300 };
301 // FIXME: Hook up locale strings.
302 RetainPtr<CTFontDescriptorRef> fontDescriptor;
303 CFStringRef textStyle = nullptr;
304 AtomString style;
305 switch (valueID) {
306 case CSSValueAppleSystemHeadline:
307 textStyle = kCTUIFontTextStyleHeadline;
308 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
309 break;
310 case CSSValueAppleSystemBody:
311 textStyle = kCTUIFontTextStyleBody;
312 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
313 break;
314 case CSSValueAppleSystemTitle0:
315 textStyle = kCTUIFontTextStyleTitle0;
316 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
317 break;
318 case CSSValueAppleSystemTitle1:
319 textStyle = kCTUIFontTextStyleTitle1;
320 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
321 break;
322 case CSSValueAppleSystemTitle2:
323 textStyle = kCTUIFontTextStyleTitle2;
324 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
325 break;
326 case CSSValueAppleSystemTitle3:
327 textStyle = kCTUIFontTextStyleTitle3;
328 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
329 break;
330 case CSSValueAppleSystemTitle4:
331 textStyle = kCTUIFontTextStyleTitle4;
332 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
333 break;
334 case CSSValueAppleSystemSubheadline:
335 textStyle = kCTUIFontTextStyleSubhead;
336 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
337 break;
338 case CSSValueAppleSystemFootnote:
339 textStyle = kCTUIFontTextStyleFootnote;
340 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
341 break;
342 case CSSValueAppleSystemCaption1:
343 textStyle = kCTUIFontTextStyleCaption1;
344 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
345 break;
346 case CSSValueAppleSystemCaption2:
347 textStyle = kCTUIFontTextStyleCaption2;
348 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
349 break;
350 case CSSValueAppleSystemShortHeadline:
351 textStyle = kCTUIFontTextStyleShortHeadline;
352 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
353 break;
354 case CSSValueAppleSystemShortBody:
355 textStyle = kCTUIFontTextStyleShortBody;
356 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
357 break;
358 case CSSValueAppleSystemShortSubheadline:
359 textStyle = kCTUIFontTextStyleShortSubhead;
360 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
361 break;
362 case CSSValueAppleSystemShortFootnote:
363 textStyle = kCTUIFontTextStyleShortFootnote;
364 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
365 break;
366 case CSSValueAppleSystemShortCaption1:
367 textStyle = kCTUIFontTextStyleShortCaption1;
368 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
369 break;
370 case CSSValueAppleSystemTallBody:
371 textStyle = kCTUIFontTextStyleTallBody;
372 fontDescriptor = adoptCF(CTFontDescriptorCreateWithTextStyle(textStyle, contentSizeCategory(), nullptr));
373 break;
374 case CSSValueSmallCaption: {
375 style = AtomString("system-ui", AtomString::ConstructFromLiteral);
376 auto font = [cocoaFontClass() systemFontOfSize:[cocoaFontClass() smallSystemFontSize]];
377 fontDescriptor = static_cast<CTFontDescriptorRef>(font.fontDescriptor);
378 break;
379 }
380 case CSSValueMenu:
381 style = AtomString("-apple-menu", AtomString::ConstructFromLiteral);
382 fontDescriptor = adoptCF(CTFontDescriptorCreateForUIType(kCTFontUIFontMenuItem, [cocoaFontClass() systemFontSize], nullptr));
383 break;
384 case CSSValueStatusBar: {
385 style = AtomString("-apple-status-bar", AtomString::ConstructFromLiteral);
386 fontDescriptor = adoptCF(CTFontDescriptorCreateForUIType(kCTFontUIFontSystem, [cocoaFontClass() labelFontSize], nullptr));
387 break;
388 }
389 case CSSValueWebkitMiniControl: {
390 style = AtomString("system-ui", AtomString::ConstructFromLiteral);
391#if PLATFORM(IOS_FAMILY)
392 fontDescriptor = adoptCF(CTFontDescriptorCreateForUIType(kCTFontUIFontMiniSystem, 0, nullptr));
393#else
394 auto font = [cocoaFontClass() systemFontOfSize:[cocoaFontClass() systemFontSizeForControlSize:NSControlSizeMini]];
395 fontDescriptor = static_cast<CTFontDescriptorRef>(font.fontDescriptor);
396#endif
397 break;
398 }
399 case CSSValueWebkitSmallControl: {
400 style = AtomString("system-ui", AtomString::ConstructFromLiteral);
401#if PLATFORM(IOS_FAMILY)
402 fontDescriptor = adoptCF(CTFontDescriptorCreateForUIType(kCTFontUIFontSmallSystem, 0, nullptr));
403#else
404 auto font = [cocoaFontClass() systemFontOfSize:[cocoaFontClass() systemFontSizeForControlSize:NSControlSizeSmall]];
405 fontDescriptor = static_cast<CTFontDescriptorRef>(font.fontDescriptor);
406#endif
407 break;
408 }
409 case CSSValueWebkitControl: {
410 style = AtomString("system-ui", AtomString::ConstructFromLiteral);
411#if PLATFORM(IOS_FAMILY)
412 fontDescriptor = adoptCF(CTFontDescriptorCreateForUIType(kCTFontUIFontSystem, 0, nullptr));
413#else
414 auto font = [cocoaFontClass() systemFontOfSize:[cocoaFontClass() systemFontSizeForControlSize:NSControlSizeRegular]];
415 fontDescriptor = static_cast<CTFontDescriptorRef>(font.fontDescriptor);
416#endif
417 break;
418 }
419 default:
420 style = AtomString("system-ui", AtomString::ConstructFromLiteral);
421 fontDescriptor = adoptCF(CTFontDescriptorCreateForUIType(kCTFontUIFontSystem, 0, nullptr));
422 }
423
424 if (style.isNull())
425 style = textStyle;
426
427 ASSERT(fontDescriptor);
428 auto font = adoptCF(CTFontCreateWithFontDescriptor(fontDescriptor.get(), 0, nullptr));
429 fontDescription.setIsAbsoluteSize(true);
430 fontDescription.setOneFamily(style);
431 fontDescription.setSpecifiedSize(CTFontGetSize(font.get()));
432 fontDescription.setWeight(cssWeightOfSystemFont(font.get()));
433 fontDescription.setItalic(normalItalicValue());
434}
