103struct IOSGradient {
104 float* start; // points to static float[4]
105 float* end; // points to static float[4]
106 IOSGradient(float start[4], float end[4])
107 : start(start)
108 , end(end)
109 {
110 }
111};

115enum Interpolation
116{
117 LinearInterpolation,
118 ExponentialInterpolation
119};

121static void interpolateLinearGradient(void *info, const CGFloat *inData, CGFloat *outData)
122{
123 IOSGradientRef gradient = static_cast<IOSGradientRef>(info);
124 float alpha = inData[0];
125 float inverse = 1.0f - alpha;
126
127 outData[0] = inverse * gradient->start[0] + alpha * gradient->end[0];
128 outData[1] = inverse * gradient->start[1] + alpha * gradient->end[1];
129 outData[2] = inverse * gradient->start[2] + alpha * gradient->end[2];
130 outData[3] = inverse * gradient->start[3] + alpha * gradient->end[3];
131}
132
133static void interpolateExponentialGradient(void *info, const CGFloat *inData, CGFloat *outData)
134{
135 IOSGradientRef gradient = static_cast<IOSGradientRef>(info);
136 float a = inData[0];
137 for (int paintInfo = 0; paintInfo < 4; ++paintInfo) {
138 float end = logf(std::max(gradient->end[paintInfo], 0.01f));
139 float start = logf(std::max(gradient->start[paintInfo], 0.01f));
140 outData[paintInfo] = expf(start - (end + start) * a);
141 }
142}

144static CGFunctionRef getSharedFunctionRef(IOSGradientRef gradient, Interpolation interpolation)
145{
146 static NeverDestroyed<HashMap<IOSGradientRef, RetainPtr<CGFunctionRef>>> linearFunctionRefs;
147 static NeverDestroyed<HashMap<IOSGradientRef, RetainPtr<CGFunctionRef>>> exponentialFunctionRefs;
148
149 if (interpolation == LinearInterpolation) {
150 auto function = linearFunctionRefs->get(gradient);
151 if (!function) {
152 static struct CGFunctionCallbacks linearFunctionCallbacks = { 0, interpolateLinearGradient, 0 };
153 linearFunctionRefs->set(gradient, function = adoptCF(CGFunctionCreate(gradient, 1, nullptr, 4, nullptr, &linearFunctionCallbacks)));
154 }
155
156 return function.get();
157 }
158
159 auto function = exponentialFunctionRefs->get(gradient);
160 if (!function) {
161 static struct CGFunctionCallbacks exponentialFunctionCallbacks = { 0, interpolateExponentialGradient, 0 };
162 exponentialFunctionRefs->set(gradient, function = adoptCF(CGFunctionCreate(gradient, 1, 0, 4, 0, &exponentialFunctionCallbacks)));
163 }
164 return function.get();
165}

315FloatRect RenderThemeIOS::addRoundedBorderClip(const RenderObject& box, GraphicsContext& context, const IntRect& rect)
316{
317 // To fix inner border bleeding issues <rdar://problem/9812507>, we clip to the outer border and assert that
318 // the border is opaque or transparent, unless we're checked because checked radio/checkboxes show no bleeding.
319 auto& style = box.style();
320 RoundedRect border = isChecked(box) ? style.getRoundedInnerBorderFor(rect) : style.getRoundedBorderFor(rect);
321
322 if (border.isRounded())
323 context.clipRoundedRect(FloatRoundedRect(border));
324 else
325 context.clip(border.rect());
326
327 if (isChecked(box)) {
328 ASSERT(style.visitedDependentColor(CSSPropertyBorderTopColor).alphaByte() % 255 == 0);
329 ASSERT(style.visitedDependentColor(CSSPropertyBorderRightColor).alphaByte() % 255 == 0);
330 ASSERT(style.visitedDependentColor(CSSPropertyBorderBottomColor).alphaByte() % 255 == 0);
331 ASSERT(style.visitedDependentColor(CSSPropertyBorderLeftColor).alphaByte() % 255 == 0);
332 }
333
334 return border.rect();
335}

347static CGPoint shortened(CGPoint start, CGPoint end, float width)
348{
349 float x = end.x - start.x;
350 float y = end.y - start.y;
351 float ratio = (!x && !y) ? 0 : width / std::hypot(x, y);
352 return CGPointMake(start.x + x * ratio, start.y + y * ratio);
353}

355static void drawJoinedLines(CGContextRef context, const Vector<CGPoint>& points, CGLineCap lineCap, float lineWidth, Color strokeColor)
356{
357 CGContextSetLineWidth(context, lineWidth);
358 CGContextSetStrokeColorWithColor(context, cachedCGColor(strokeColor));
359 CGContextSetShouldAntialias(context, true);
360 CGContextBeginPath(context);
361 CGContextSetLineCap(context, lineCap);
362 CGContextMoveToPoint(context, points[0].x, points[0].y);
363
364 for (unsigned i = 1; i < points.size(); ++i)
365 CGContextAddLineToPoint(context, points[i].x, points[i].y);
366
367 CGContextStrokePath(context);
368}

381void RenderThemeIOS::paintCheckboxDecorations(const RenderObject& box, const PaintInfo& paintInfo, const IntRect& rect)
382{
383#if ENABLE(IOS_FORM_CONTROL_REFRESH)
384 if (box.settings().iOSFormControlRefreshEnabled())
385 return;
386#endif
387
388 bool checked = isChecked(box);
389 bool indeterminate = isIndeterminate(box);
390 CGContextRef cgContext = paintInfo.context().platformContext();
391 GraphicsContextStateSaver stateSaver { paintInfo.context() };
392
393 if (checked || indeterminate) {
394 auto border = box.style().getRoundedBorderFor(rect);
395 paintInfo.context().fillRoundedRect(border.pixelSnappedRoundedRectForPainting(box.document().deviceScaleFactor()), Color::black.colorWithAlphaByte(204));
396
397 auto clip = addRoundedBorderClip(box, paintInfo.context(), rect);
398 auto width = clip.width();
399 auto height = clip.height();
400 drawAxialGradient(cgContext, gradientWithName(ConcaveGradient), clip.location(), FloatPoint { clip.x(), clip.maxY() }, LinearInterpolation);
401
402 constexpr float thicknessRatio = 2 / 14.0;
403 float lineWidth = std::min(width, height) * 2.0f * thicknessRatio;
404
405 Vector<CGPoint, 3> line;
406 Vector<CGPoint, 3> shadow;
407 if (checked) {
408 constexpr CGSize size { 14.0f, 14.0f };
409 constexpr CGPoint pathRatios[] = {
410 { 2.5f / size.width, 7.5f / size.height },
411 { 5.5f / size.width, 10.5f / size.height },
412 { 11.5f / size.width, 2.5f / size.height }
413 };
414
415 line.uncheckedAppend(CGPointMake(clip.x() + width * pathRatios[0].x, clip.y() + height * pathRatios[0].y));
416 line.uncheckedAppend(CGPointMake(clip.x() + width * pathRatios[1].x, clip.y() + height * pathRatios[1].y));
417 line.uncheckedAppend(CGPointMake(clip.x() + width * pathRatios[2].x, clip.y() + height * pathRatios[2].y));
418
419 shadow.uncheckedAppend(shortened(line[0], line[1], lineWidth / 4.0f));
420 shadow.uncheckedAppend(line[1]);
421 shadow.uncheckedAppend(shortened(line[2], line[1], lineWidth / 4.0f));
422 } else {
423 line.uncheckedAppend(CGPointMake(clip.x() + 3.5, clip.center().y()));
424 line.uncheckedAppend(CGPointMake(clip.maxX() - 3.5, clip.center().y()));
425
426 shadow.uncheckedAppend(shortened(line[0], line[1], lineWidth / 4.0f));
427 shadow.uncheckedAppend(shortened(line[1], line[0], lineWidth / 4.0f));
428 }
429
430 lineWidth = std::max<float>(lineWidth, 1);
431 drawJoinedLines(cgContext, Vector<CGPoint> { WTFMove(shadow) }, kCGLineCapSquare, lineWidth, Color::black.colorWithAlphaByte(179));
432
433 lineWidth = std::max<float>(std::min(width, height) * thicknessRatio, 1);
434 drawJoinedLines(cgContext, Vector<CGPoint> { WTFMove(line) }, kCGLineCapButt, lineWidth, Color::white.colorWithAlphaByte(240));
435 } else {
436 auto clip = addRoundedBorderClip(box, paintInfo.context(), rect);
437 auto width = clip.width();
438 auto height = clip.height();
439 FloatPoint bottomCenter { clip.x() + width / 2.0f, clip.maxY() };
440
441 drawAxialGradient(cgContext, gradientWithName(ShadeGradient), clip.location(), FloatPoint { clip.x(), clip.maxY() }, LinearInterpolation);
442 drawRadialGradient(cgContext, gradientWithName(ShineGradient), bottomCenter, 0, bottomCenter, sqrtf((width * width) / 4.0f + height * height), ExponentialInterpolation);
443 }
444}

446LayoutRect RenderThemeIOS::adjustedPaintRect(const RenderBox& box, const LayoutRect& paintRect) const
447{
448 // Workaround for <rdar://problem/6209763>. Force the painting bounds of checkboxes and radio controls to be square.
449 if (box.style().appearance() == CheckboxPart || box.style().appearance() == RadioPart) {
450 float width = std::min(paintRect.width(), paintRect.height());
451 float height = width;
452 return enclosingLayoutRect(FloatRect(paintRect.x(), paintRect.y() + (box.height() - height) / 2, width, height)); // Vertically center the checkbox, like on desktop
453 }
454
455 return paintRect;
456}

490void RenderThemeIOS::paintRadioDecorations(const RenderObject& box, const PaintInfo& paintInfo, const IntRect& rect)
491{
492#if ENABLE(IOS_FORM_CONTROL_REFRESH)
493 if (box.settings().iOSFormControlRefreshEnabled())
494 return;
495#endif
496
497 GraphicsContextStateSaver stateSaver(paintInfo.context());
498 CGContextRef cgContext = paintInfo.context().platformContext();
499
500 auto drawShadeAndShineGradients = [&](auto clip) {
501 FloatPoint bottomCenter(clip.x() + clip.width() / 2.0, clip.maxY());
502 drawAxialGradient(cgContext, gradientWithName(ShadeGradient), clip.location(), FloatPoint(clip.x(), clip.maxY()), LinearInterpolation);
503 drawRadialGradient(cgContext, gradientWithName(ShineGradient), bottomCenter, 0, bottomCenter, std::max(clip.width(), clip.height()), ExponentialInterpolation);
504 };
505
506 if (isChecked(box)) {
507 auto border = box.style().getRoundedBorderFor(rect);
508 paintInfo.context().fillRoundedRect(border.pixelSnappedRoundedRectForPainting(box.document().deviceScaleFactor()), Color::black.colorWithAlphaByte(204));
509
510 auto clip = addRoundedBorderClip(box, paintInfo.context(), rect);
511 drawAxialGradient(cgContext, gradientWithName(ConcaveGradient), clip.location(), FloatPoint(clip.x(), clip.maxY()), LinearInterpolation);
512
513 // The inner circle is 6 / 14 the size of the surrounding circle,
514 // leaving 8 / 14 around it. (8 / 14) / 2 = 2 / 7.
515
516 static const float InnerInverseRatio = 2 / 7.0;
517
518 clip.inflateX(-clip.width() * InnerInverseRatio);
519 clip.inflateY(-clip.height() * InnerInverseRatio);
520
521 constexpr auto shadowColor = Color::black.colorWithAlphaByte(179);
522 paintInfo.context().drawRaisedEllipse(clip, Color::white, shadowColor);
523
524 FloatSize radius(clip.width() / 2.0f, clip.height() / 2.0f);
525 paintInfo.context().clipRoundedRect(FloatRoundedRect(clip, radius, radius, radius, radius));
526 drawShadeAndShineGradients(clip);
527 } else {
528 auto clip = addRoundedBorderClip(box, paintInfo.context(), rect);
529 drawShadeAndShineGradients(clip);
530 }
531}

552}

1548void RenderAttachmentInfo::buildWrappedLines(const String& text, CTFontRef font, UIColor *color, unsigned maximumLineCount)
1549{
1550 if (text.isEmpty())
1551 return;
1552
1553 NSDictionary *textAttributes = @{
1554 (id)kCTFontAttributeName: (id)font,
1555 (id)kCTForegroundColorAttributeName: color
1556 };
1557 RetainPtr<NSAttributedString> attributedText = adoptNS([[NSAttributedString alloc] initWithString:text attributes:textAttributes]);
1558 RetainPtr<CTFramesetterRef> framesetter = adoptCF(CTFramesetterCreateWithAttributedString((CFAttributedStringRef)attributedText.get()));
1559
1560 CFRange fitRange;
1561 CGFloat wrappingWidth = attachmentWrappingTextMaximumWidth * attachmentDynamicTypeScaleFactor();
1562 CGSize textSize = CTFramesetterSuggestFrameSizeWithConstraints(framesetter.get(), CFRangeMake(0, 0), nullptr, CGSizeMake(wrappingWidth, CGFLOAT_MAX), &fitRange);
1563
1564 RetainPtr<CGPathRef> textPath = adoptCF(CGPathCreateWithRect(CGRectMake(0, 0, textSize.width, textSize.height), nullptr));
1565 RetainPtr<CTFrameRef> textFrame = adoptCF(CTFramesetterCreateFrame(framesetter.get(), fitRange, textPath.get(), nullptr));
1566
1567 CFArrayRef ctLines = CTFrameGetLines(textFrame.get());
1568 CFIndex lineCount = CFArrayGetCount(ctLines);
1569 if (!lineCount)
1570 return;
1571
1572 // Lay out and record the first (maximumLineCount - 1) lines.
1573 CFIndex lineIndex = 0;
1574 CFIndex nonTruncatedLineCount = std::min<CFIndex>(maximumLineCount - 1, lineCount);
1575 for (; lineIndex < nonTruncatedLineCount; ++lineIndex)
1576 addLine((CTLineRef)CFArrayGetValueAtIndex(ctLines, lineIndex));
1577
1578 if (lineIndex == lineCount)
1579 return;
1580
1581 // We had text that didn't fit in the first (maximumLineCount - 1) lines.
1582 // Combine it into one last line, and center-truncate it.
1583 CTLineRef firstRemainingLine = (CTLineRef)CFArrayGetValueAtIndex(ctLines, lineIndex);
1584 CFIndex remainingRangeStart = CTLineGetStringRange(firstRemainingLine).location;
1585 CFRange remainingRange = CFRangeMake(remainingRangeStart, [attributedText length] - remainingRangeStart);
1586 RetainPtr<CGPathRef> remainingPath = adoptCF(CGPathCreateWithRect(CGRectMake(0, 0, CGFLOAT_MAX, CGFLOAT_MAX), nullptr));
1587 RetainPtr<CTFrameRef> remainingFrame = adoptCF(CTFramesetterCreateFrame(framesetter.get(), remainingRange, remainingPath.get(), nullptr));
1588 RetainPtr<NSAttributedString> ellipsisString = adoptNS([[NSAttributedString alloc] initWithString:@"\u2026" attributes:textAttributes]);
1589 RetainPtr<CTLineRef> ellipsisLine = adoptCF(CTLineCreateWithAttributedString((CFAttributedStringRef)ellipsisString.get()));
1590 CTLineRef remainingLine = (CTLineRef)CFArrayGetValueAtIndex(CTFrameGetLines(remainingFrame.get()), 0);
1591 RetainPtr<CTLineRef> truncatedLine = adoptCF(CTLineCreateTruncatedLine(remainingLine, wrappingWidth, kCTLineTruncationMiddle, ellipsisLine.get()));
1592
1593 if (!truncatedLine)
1594 truncatedLine = remainingLine;
1595
1596 addLine(truncatedLine.get());
1597}
