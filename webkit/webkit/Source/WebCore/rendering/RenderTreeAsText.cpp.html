126static String getTagName(Node* n)
127{
128 if (n->isDocumentNode())
129 return "";
130 if (n->nodeType() == Node::COMMENT_NODE)
131 return "COMMENT";
132 return n->nodeName();

135static bool isEmptyOrUnstyledAppleStyleSpan(const Node* node)
136{
137 if (!is<HTMLSpanElement>(node))
138 return false;
139
140 const HTMLElement& element = downcast<HTMLSpanElement>(*node);
141 if (element.getAttribute(classAttr) != "Apple-style-span")
142 return false;
143
144 if (!node->hasChildNodes())
145 return true;
146
147 const StyleProperties* inlineStyleDecl = element.inlineStyle();
148 return (!inlineStyleDecl || inlineStyleDecl->isEmpty());
149}

151String quoteAndEscapeNonPrintables(StringView s)
152{
153 StringBuilder result;
154 result.append('"');
155 for (unsigned i = 0; i != s.length(); ++i) {
156 UChar c = s[i];
157 if (c == '\\') {
158 result.append("\\\\");
159 } else if (c == '"') {
160 result.append("\\\"");
161 } else if (c == '\n' || c == noBreakSpace)
162 result.append(' ');
163 else {
164 if (c >= 0x20 && c < 0x7F)
165 result.append(c);
166 else
167 result.append("\\x{", hex(c), '}');
168 }
169 }
170 result.append('"');
171 return result.toString();
172}

174static inline bool isRenderInlineEmpty(const RenderInline& inlineRenderer)
175{
176 if (isEmptyInline(inlineRenderer))
177 return true;
178
179 for (auto& child : childrenOfType<RenderObject>(inlineRenderer)) {
180 if (child.isFloatingOrOutOfFlowPositioned())
181 continue;
182 auto isChildEmpty = false;
183 if (is<RenderInline>(child))
184 isChildEmpty = isRenderInlineEmpty(downcast<RenderInline>(child));
185 else if (is<RenderText>(child))
186 isChildEmpty = !downcast<RenderText>(child).linesBoundingBox().height();
187 if (!isChildEmpty)
188 return false;
189 }
190 return true;
191}
192
193static inline bool hasNonEmptySibling(const RenderInline& inlineRenderer)
194{
195 auto* parent = inlineRenderer.parent();
196 if (!parent)
197 return false;
198
199 for (auto& sibling : childrenOfType<RenderObject>(*parent)) {
200 if (&sibling == &inlineRenderer || sibling.isFloatingOrOutOfFlowPositioned())
201 continue;
202 if (!is<RenderInline>(sibling))
203 return true;
204 auto& siblingRendererInline = downcast<RenderInline>(sibling);
205 if (siblingRendererInline.shouldCreateLineBoxes() || !isRenderInlineEmpty(siblingRendererInline))
206 return true;
207 }
208 return false;
209}

211void RenderTreeAsText::writeRenderObject(TextStream& ts, const RenderObject& o, OptionSet<RenderAsTextFlag> behavior)
212{
213 ts << o.renderName();
214
215 if (behavior.contains(RenderAsTextFlag::ShowAddresses))
216 ts << " " << &o;
217
218 if (o.style().usedZIndex()) // FIXME: This should use !hasAutoUsedZIndex().
219 ts << " zI: " << o.style().usedZIndex();
220
221 if (o.node()) {
222 String tagName = getTagName(o.node());
223 // FIXME: Temporary hack to make tests pass by simulating the old generated content output.
224 if (o.isPseudoElement() || (o.parent() && o.parent()->isPseudoElement()))
225 tagName = emptyAtom();
226 if (!tagName.isEmpty()) {
227 ts << " {" << tagName << "}";
228 // flag empty or unstyled AppleStyleSpan because we never
229 // want to leave them in the DOM
230 if (isEmptyOrUnstyledAppleStyleSpan(o.node()))
231 ts << " *empty or unstyled AppleStyleSpan*";
232 }
233 }
234
235 RenderBlock* cb = o.containingBlock();
236 bool adjustForTableCells = cb ? cb->isTableCell() : false;
237
238 LayoutRect r;
239 if (is<RenderText>(o)) {
240 // FIXME: Would be better to dump the bounding box x and y rather than the first run's x and y, but that would involve updating
241 // many test results.
242 const RenderText& text = downcast<RenderText>(o);
243 r = IntRect(text.firstRunLocation(), text.linesBoundingBox().size());
244 if (!LayoutIntegration::firstTextRunFor(text))
245 adjustForTableCells = false;
246 } else if (o.isBR()) {
247 const RenderLineBreak& br = downcast<RenderLineBreak>(o);
248 IntRect linesBox = br.linesBoundingBox();
249 r = IntRect(linesBox.x(), linesBox.y(), linesBox.width(), linesBox.height());
250 if (!br.inlineBoxWrapper() && !LayoutIntegration::runFor(br))
251 adjustForTableCells = false;
252 } else if (is<RenderInline>(o)) {
253 const RenderInline& inlineFlow = downcast<RenderInline>(o);
254 // FIXME: Would be better not to just dump 0, 0 as the x and y here.
255 auto width = inlineFlow.linesBoundingBox().width();
256 auto inlineHeight = [&] {
257 // Let's match legacy line layout's RenderInline behavior and report 0 height when the inline box is "empty".
258 // FIXME: Remove and rebaseline when LFC inline boxes are enabled (see webkit.org/b/220722)
259 auto height = inlineFlow.linesBoundingBox().height();
260 if (width)
261 return height;
262 if (is<RenderQuote>(inlineFlow) || is<RenderRubyAsInline>(inlineFlow))
263 return height;
264 if (inlineFlow.marginStart() || inlineFlow.marginEnd())
265 return height;
266 // This is mostly pre/post continuation content. Also see webkit.org/b/220735
267 if (hasNonEmptySibling(inlineFlow))
268 return height;
269 if (isRenderInlineEmpty(inlineFlow))
270 return 0;
271 return height;
272 };
273 r = IntRect(0, 0, width, inlineHeight());
274 adjustForTableCells = false;
275 } else if (is<RenderTableCell>(o)) {
276 // FIXME: Deliberately dump the "inner" box of table cells, since that is what current results reflect. We'd like
277 // to clean up the results to dump both the outer box and the intrinsic padding so that both bits of information are
278 // captured by the results.
279 const RenderTableCell& cell = downcast<RenderTableCell>(o);
280 r = LayoutRect(cell.x(), cell.y() + cell.intrinsicPaddingBefore(), cell.width(), cell.height() - cell.intrinsicPaddingBefore() - cell.intrinsicPaddingAfter());
281 } else if (is<RenderBox>(o))
282 r = downcast<RenderBox>(o).frameRect();
283
284 // FIXME: Temporary in order to ensure compatibility with existing layout test results.
285 if (adjustForTableCells)
286 r.move(0_lu, -downcast<RenderTableCell>(*o.containingBlock()).intrinsicPaddingBefore());
287
288 // FIXME: Convert layout test results to report sub-pixel values, in the meantime using enclosingIntRect
289 // for consistency with old results.
290 ts << " " << enclosingIntRect(r);
291
292 if (!is<RenderText>(o)) {
293 if (is<RenderFileUploadControl>(o))
294 ts << " " << quoteAndEscapeNonPrintables(downcast<RenderFileUploadControl>(o).fileTextValue());
295
296 if (o.parent()) {
297 Color color = o.style().visitedDependentColor(CSSPropertyColor);
298 if (!equalIgnoringSemanticColor(o.parent()->style().visitedDependentColor(CSSPropertyColor), color))
299 ts << " [color=" << serializationForRenderTreeAsText(color) << "]";
300
301 // Do not dump invalid or transparent backgrounds, since that is the default.
302 Color backgroundColor = o.style().visitedDependentColor(CSSPropertyBackgroundColor);
303 if (!equalIgnoringSemanticColor(o.parent()->style().visitedDependentColor(CSSPropertyBackgroundColor), backgroundColor)
304 && backgroundColor != Color::transparentBlack)
305 ts << " [bgcolor=" << serializationForRenderTreeAsText(backgroundColor) << "]";
306
307 Color textFillColor = o.style().visitedDependentColor(CSSPropertyWebkitTextFillColor);
308 if (!equalIgnoringSemanticColor(o.parent()->style().visitedDependentColor(CSSPropertyWebkitTextFillColor), textFillColor)
309 && textFillColor != color && textFillColor != Color::transparentBlack)
310 ts << " [textFillColor=" << serializationForRenderTreeAsText(textFillColor) << "]";
311
312 Color textStrokeColor = o.style().visitedDependentColor(CSSPropertyWebkitTextStrokeColor);
313 if (!equalIgnoringSemanticColor(o.parent()->style().visitedDependentColor(CSSPropertyWebkitTextStrokeColor), textStrokeColor)
314 && textStrokeColor != color && textStrokeColor != Color::transparentBlack)
315 ts << " [textStrokeColor=" << serializationForRenderTreeAsText(textStrokeColor) << "]";
316
317 if (o.parent()->style().textStrokeWidth() != o.style().textStrokeWidth() && o.style().textStrokeWidth() > 0)
318 ts << " [textStrokeWidth=" << o.style().textStrokeWidth() << "]";
319 }
320
321 if (!is<RenderBoxModelObject>(o) || is<RenderLineBreak>(o))
322 return;
323
324 const RenderBoxModelObject& box = downcast<RenderBoxModelObject>(o);
325 LayoutUnit borderTop = box.borderTop();
326 LayoutUnit borderRight = box.borderRight();
327 LayoutUnit borderBottom = box.borderBottom();
328 LayoutUnit borderLeft = box.borderLeft();
329 if (box.isFieldset()) {
330 const auto& block = downcast<RenderBlock>(box);
331 if (o.style().writingMode() == WritingMode::TopToBottom)
332 borderTop -= block.intrinsicBorderForFieldset();
333 else if (o.style().writingMode() == WritingMode::BottomToTop)
334 borderBottom -= block.intrinsicBorderForFieldset();
335 else if (o.style().writingMode() == WritingMode::LeftToRight)
336 borderLeft -= block.intrinsicBorderForFieldset();
337 else if (o.style().writingMode() == WritingMode::RightToLeft)
338 borderRight -= block.intrinsicBorderForFieldset();
339
340 }
341 if (borderTop || borderRight || borderBottom || borderLeft) {
342 ts << " [border:";
343
344 BorderValue prevBorder = o.style().borderTop();
345 if (!borderTop)
346 ts << " none";
347 else {
348 ts << " (" << borderTop << "px ";
349 printBorderStyle(ts, o.style().borderTopStyle());
350 auto color = o.style().borderTopColor();
351 if (!color.isValid())
352 color = o.style().color();
353 ts << serializationForRenderTreeAsText(color) << ")";
354 }
355
356 if (o.style().borderRight() != prevBorder) {
357 prevBorder = o.style().borderRight();
358 if (!borderRight)
359 ts << " none";
360 else {
361 ts << " (" << borderRight << "px ";
362 printBorderStyle(ts, o.style().borderRightStyle());
363 auto color = o.style().borderRightColor();
364 if (!color.isValid())
365 color = o.style().color();
366 ts << serializationForRenderTreeAsText(color) << ")";
367 }
368 }
369
370 if (o.style().borderBottom() != prevBorder) {
371 prevBorder = box.style().borderBottom();
372 if (!borderBottom)
373 ts << " none";
374 else {
375 ts << " (" << borderBottom << "px ";
376 printBorderStyle(ts, o.style().borderBottomStyle());
377 auto color = o.style().borderBottomColor();
378 if (!color.isValid())
379 color = o.style().color();
380 ts << serializationForRenderTreeAsText(color) << ")";
381 }
382 }
383
384 if (o.style().borderLeft() != prevBorder) {
385 prevBorder = o.style().borderLeft();
386 if (!borderLeft)
387 ts << " none";
388 else {
389 ts << " (" << borderLeft << "px ";
390 printBorderStyle(ts, o.style().borderLeftStyle());
391 auto color = o.style().borderLeftColor();
392 if (!color.isValid())
393 color = o.style().color();
394 ts << serializationForRenderTreeAsText(color) << ")";
395 }
396 }
397
398 ts << "]";
399 }
400
401#if ENABLE(MATHML)
402 // We want to show any layout padding, both CSS padding and intrinsic padding, so we can't just check o.style().hasPadding().
403 if (o.isRenderMathMLBlock() && (box.paddingTop() || box.paddingRight() || box.paddingBottom() || box.paddingLeft())) {
404 ts << " [";
405 LayoutUnit cssTop = box.computedCSSPaddingTop();
406 LayoutUnit cssRight = box.computedCSSPaddingRight();
407 LayoutUnit cssBottom = box.computedCSSPaddingBottom();
408 LayoutUnit cssLeft = box.computedCSSPaddingLeft();
409 if (box.paddingTop() != cssTop || box.paddingRight() != cssRight || box.paddingBottom() != cssBottom || box.paddingLeft() != cssLeft) {
410 ts << "intrinsic ";
411 if (cssTop || cssRight || cssBottom || cssLeft)
412 ts << "+ CSS ";
413 }
414 ts << "padding: " << roundToInt(box.paddingTop()) << " " << roundToInt(box.paddingRight()) << " " << roundToInt(box.paddingBottom()) << " " << roundToInt(box.paddingLeft()) << "]";
415 }
416#endif
417 }
418
419 if (is<RenderTableCell>(o)) {
420 const RenderTableCell& c = downcast<RenderTableCell>(o);
421 ts << " [r=" << c.rowIndex() << " c=" << c.col() << " rs=" << c.rowSpan() << " cs=" << c.colSpan() << "]";
422 }
423
424 if (is<RenderDetailsMarker>(o)) {
425 ts << ": ";
426 switch (downcast<RenderDetailsMarker>(o).orientation()) {
427 case RenderDetailsMarker::Left:
428 ts << "left";
429 break;
430 case RenderDetailsMarker::Right:
431 ts << "right";
432 break;
433 case RenderDetailsMarker::Up:
434 ts << "up";
435 break;
436 case RenderDetailsMarker::Down:
437 ts << "down";
438 break;
439 }
440 }
441
442 if (is<RenderListMarker>(o)) {
443 String text = downcast<RenderListMarker>(o).textWithoutSuffix().toString();
444 if (!text.isEmpty()) {
445 if (text.length() != 1)
446 text = quoteAndEscapeNonPrintables(text);
447 else {
448 switch (text[0]) {
449 case bullet:
450 text = "bullet";
451 break;
452 case blackSquare:
453 text = "black square";
454 break;
455 case whiteBullet:
456 text = "white bullet";
457 break;
458 default:
459 text = quoteAndEscapeNonPrintables(text);
460 }
461 }
462 ts << ": " << text;
463 }
464 }
465
466 writeDebugInfo(ts, o, behavior);
467}

469void writeDebugInfo(TextStream& ts, const RenderObject& object, OptionSet<RenderAsTextFlag> behavior)
470{
471 if (behavior.contains(RenderAsTextFlag::ShowIDAndClass)) {
472 if (Element* element = is<Element>(object.node()) ? downcast<Element>(object.node()) : nullptr) {
473 if (element->hasID())
474 ts << " id=\"" << element->getIdAttribute() << "\"";
475
476 if (element->hasClass()) {
477 ts << " class=\"";
478 for (size_t i = 0; i < element->classNames().size(); ++i) {
479 if (i > 0)
480 ts << " ";
481 ts << element->classNames()[i];
482 }
483 ts << "\"";
484 }
485 }
486 }
487
488 if (behavior.contains(RenderAsTextFlag::ShowLayoutState)) {
489 bool needsLayout = object.selfNeedsLayout() || object.needsPositionedMovementLayout() || object.posChildNeedsLayout() || object.normalChildNeedsLayout();
490 if (needsLayout)
491 ts << " (needs layout:";
492
493 bool havePrevious = false;
494 if (object.selfNeedsLayout()) {
495 ts << " self";
496 havePrevious = true;
497 }
498
499 if (object.needsPositionedMovementLayout()) {
500 if (havePrevious)
501 ts << ",";
502 havePrevious = true;
503 ts << " positioned movement";
504 }
505
506 if (object.normalChildNeedsLayout()) {
507 if (havePrevious)
508 ts << ",";
509 havePrevious = true;
510 ts << " child";
511 }
512
513 if (object.posChildNeedsLayout()) {
514 if (havePrevious)
515 ts << ",";
516 ts << " positioned child";
517 }
518
519 if (needsLayout)
520 ts << ")";
521 }
522
523 if (behavior.contains(RenderAsTextFlag::ShowOverflow) && is<RenderBox>(object)) {
524 const auto& box = downcast<RenderBox>(object);
525 if (box.hasRenderOverflow()) {
526 LayoutRect layoutOverflow = box.layoutOverflowRect();
527 ts << " (layout overflow " << layoutOverflow.x().toInt() << "," << layoutOverflow.y().toInt() << " " << layoutOverflow.width().toInt() << "x" << layoutOverflow.height().toInt() << ")";
528
529 if (box.hasVisualOverflow()) {
530 LayoutRect visualOverflow = box.visualOverflowRect();
531 ts << " (visual overflow " << visualOverflow.x().toInt() << "," << visualOverflow.y().toInt() << " " << visualOverflow.width().toInt() << "x" << visualOverflow.height().toInt() << ")";
532 }
533 }
534 }

537void write(TextStream& ts, const RenderObject& o, OptionSet<RenderAsTextFlag> behavior)
538{
539 auto writeTextRun = [&](auto& textRenderer, auto& textRun)
540 {
541 auto rect = textRun.rect();
542 int x = rect.x();
543 int y = rect.y();
544 // FIXME: Use non-logical width. webkit.org/b/206809.
545 int logicalWidth = ceilf(rect.x() + (textRun.isHorizontal() ? rect.width() : rect.height())) - x;
546 // FIXME: Table cell adjustment is temporary until results can be updated.
547 if (is<RenderTableCell>(*o.containingBlock()))
548 y -= floorToInt(downcast<RenderTableCell>(*o.containingBlock()).intrinsicPaddingBefore());
549
550 ts << "text run at (" << x << "," << y << ") width " << logicalWidth;
551 if (!textRun.isLeftToRightDirection() || textRun.dirOverride()) {
552 ts << (!textRun.isLeftToRightDirection() ? " RTL" : " LTR");
553 if (textRun.dirOverride())
554 ts << " override";
555 }
556 ts << ": "
557 << quoteAndEscapeNonPrintables(textRun.text());
558 if (textRun.hasHyphen())
559 ts << " + hyphen string " << quoteAndEscapeNonPrintables(textRenderer.style().hyphenString().string());
560 ts << "\n";
561 };
562
563 if (is<RenderSVGShape>(o)) {
564 write(ts, downcast<RenderSVGShape>(o), behavior);
565 return;
566 }
567 if (is<RenderSVGGradientStop>(o)) {
568 writeSVGGradientStop(ts, downcast<RenderSVGGradientStop>(o), behavior);
569 return;
570 }
571 if (is<RenderSVGResourceContainer>(o)) {
572 writeSVGResourceContainer(ts, downcast<RenderSVGResourceContainer>(o), behavior);
573 return;
574 }
575 if (is<RenderSVGContainer>(o)) {
576 writeSVGContainer(ts, downcast<RenderSVGContainer>(o), behavior);
577 return;
578 }
579 if (is<RenderSVGRoot>(o)) {
580 write(ts, downcast<RenderSVGRoot>(o), behavior);
581 return;
582 }
583 if (is<RenderSVGText>(o)) {
584 writeSVGText(ts, downcast<RenderSVGText>(o), behavior);
585 return;
586 }
587 if (is<RenderSVGInlineText>(o)) {
588 writeSVGInlineText(ts, downcast<RenderSVGInlineText>(o), behavior);
589 return;
590 }
591 if (is<RenderSVGImage>(o)) {
592 writeSVGImage(ts, downcast<RenderSVGImage>(o), behavior);
593 return;
594 }
595
596 ts << indent;
597
598 RenderTreeAsText::writeRenderObject(ts, o, behavior);
599 ts << "\n";
600
601 TextStream::IndentScope indentScope(ts);
602
603 if (is<RenderText>(o)) {
604 auto& text = downcast<RenderText>(o);
605 for (auto& run : LayoutIntegration::textRunsFor(text)) {
606 ts << indent;
607 writeTextRun(text, run);
608 }
609 } else {
610 for (auto& child : childrenOfType<RenderObject>(downcast<RenderElement>(o))) {
611 if (child.hasLayer())
612 continue;
613 write(ts, child, behavior);
614 }
615 }
616
617 if (is<RenderWidget>(o)) {
618 Widget* widget = downcast<RenderWidget>(o).widget();
619 if (is<FrameView>(widget)) {
620 FrameView& view = downcast<FrameView>(*widget);
621 if (RenderView* root = view.frame().contentRenderer()) {
622 if (!(behavior.contains(RenderAsTextFlag::DontUpdateLayout)))
623 view.layoutContext().layout();
624 if (RenderLayer* layer = root->layer())
625 writeLayers(ts, *layer, *layer, layer->rect(), behavior);
626 }
627 }
628 }
629}

637static void writeLayer(TextStream& ts, const RenderLayer& layer, const LayoutRect& layerBounds, const LayoutRect& backgroundClipRect, const LayoutRect& clipRect,
638 LayerPaintPhase paintPhase = LayerPaintPhaseAll, OptionSet<RenderAsTextFlag> behavior = { })
639{
640 IntRect adjustedLayoutBounds = snappedIntRect(layerBounds);
641 IntRect adjustedBackgroundClipRect = snappedIntRect(backgroundClipRect);
642 IntRect adjustedClipRect = snappedIntRect(clipRect);
643
644 ts << indent << "layer ";
645
646 if (behavior.contains(RenderAsTextFlag::ShowAddresses)) {
647 ts << &layer << " ";
648 if (auto* scrollableArea = layer.scrollableArea())
649 ts << "scrollableArea " << scrollableArea << " ";
650 }
651
652 ts << adjustedLayoutBounds;
653
654 if (!adjustedLayoutBounds.isEmpty()) {
655 if (!adjustedBackgroundClipRect.contains(adjustedLayoutBounds))
656 ts << " backgroundClip " << adjustedBackgroundClipRect;
657 if (!adjustedClipRect.contains(adjustedLayoutBounds))
658 ts << " clip " << adjustedClipRect;
659 }
660
661 if (layer.renderer().hasOverflowClip()) {
662 if (auto* scrollableArea = layer.scrollableArea()) {
663 if (scrollableArea->scrollOffset().x())
664 ts << " scrollX " << scrollableArea->scrollOffset().x();
665 if (scrollableArea->scrollOffset().y())
666 ts << " scrollY " << scrollableArea->scrollOffset().y();
667 if (layer.renderBox() && roundToInt(layer.renderBox()->clientWidth()) != scrollableArea->scrollWidth())
668 ts << " scrollWidth " << scrollableArea->scrollWidth();
669 if (layer.renderBox() && roundToInt(layer.renderBox()->clientHeight()) != scrollableArea->scrollHeight())
670 ts << " scrollHeight " << scrollableArea->scrollHeight();
671 }
672#if PLATFORM(MAC)
673 ScrollbarTheme& scrollbarTheme = ScrollbarTheme::theme();
674 if (!scrollbarTheme.isMockTheme() && layer.scrollableArea() && layer.scrollableArea()->hasVerticalScrollbar()) {
675 ScrollbarThemeMac& macTheme = *static_cast<ScrollbarThemeMac*>(&scrollbarTheme);
676 if (macTheme.isLayoutDirectionRTL(*layer.scrollableArea()->verticalScrollbar()))
677 ts << " scrollbarHasRTLLayoutDirection";
678 }
679#endif
680 }
681
682 if (paintPhase == LayerPaintPhaseBackground)
683 ts << " layerType: background only";
684 else if (paintPhase == LayerPaintPhaseForeground)
685 ts << " layerType: foreground only";
686
687 if (behavior.contains(RenderAsTextFlag::ShowCompositedLayers)) {
688 if (layer.isComposited()) {
689 ts << " (composited " << layer.compositor().reasonsForCompositing(layer)
690 << ", bounds=" << layer.backing()->compositedBounds()
691 << ", drawsContent=" << layer.backing()->graphicsLayer()->drawsContent()
692 << ", paints into ancestor=" << layer.backing()->paintsIntoCompositedAncestor() << ")";
693 } else if (layer.paintsIntoProvidedBacking())
694 ts << " (shared backing of " << layer.backingProviderLayer() << ")";
695 }
696
697#if ENABLE(CSS_COMPOSITING)
698 if (layer.isolatesBlending())
699 ts << " isolatesBlending";
700 if (layer.hasBlendMode())
701 ts << " blendMode: " << compositeOperatorName(CompositeOperator::SourceOver, layer.blendMode());
702#endif
703
704 ts << "\n";
705}

721static void writeLayers(TextStream& ts, const RenderLayer& rootLayer, RenderLayer& layer, const LayoutRect& paintRect, OptionSet<RenderAsTextFlag> behavior)
722{
723 // FIXME: Apply overflow to the root layer to not break every test. Complete hack. Sigh.
724 LayoutRect paintDirtyRect(paintRect);
725 if (&rootLayer == &layer) {
726 paintDirtyRect.setWidth(std::max<LayoutUnit>(paintDirtyRect.width(), rootLayer.renderBox()->layoutOverflowRect().maxX()));
727 paintDirtyRect.setHeight(std::max<LayoutUnit>(paintDirtyRect.height(), rootLayer.renderBox()->layoutOverflowRect().maxY()));
728 layer.setSize(layer.size().expandedTo(snappedIntSize(maxLayoutOverflow(layer.renderBox()), LayoutPoint(0, 0))));
729 }
730
731 // Calculate the clip rects we should use.
732 LayoutRect layerBounds;
733 ClipRect damageRect;
734 ClipRect clipRectToApply;
735 LayoutSize offsetFromRoot = layer.offsetFromAncestor(&rootLayer);
736 layer.calculateRects(RenderLayer::ClipRectsContext(&rootLayer, TemporaryClipRects), paintDirtyRect, layerBounds, damageRect, clipRectToApply, offsetFromRoot);
737
738 // Ensure our lists are up-to-date.
739 layer.updateLayerListsIfNeeded();
740 layer.updateDescendantDependentFlags();
741
742 bool shouldPaint = (behavior.contains(RenderAsTextFlag::ShowAllLayers)) ? true : layer.intersectsDamageRect(layerBounds, damageRect.rect(), &rootLayer, layer.offsetFromAncestor(&rootLayer));
743 auto negativeZOrderLayers = layer.negativeZOrderLayers();
744 bool paintsBackgroundSeparately = negativeZOrderLayers.size() > 0;
745 if (shouldPaint && paintsBackgroundSeparately) {
746 writeLayer(ts, layer, layerBounds, damageRect.rect(), clipRectToApply.rect(), LayerPaintPhaseBackground, behavior);
747 writeLayerRenderers(ts, layer, LayerPaintPhaseBackground, behavior);
748 }
749
750 if (negativeZOrderLayers.size()) {
751 if (behavior.contains(RenderAsTextFlag::ShowLayerNesting)) {
752 ts << indent << " negative z-order list (" << negativeZOrderLayers.size() << ")\n";
753 ts.increaseIndent();
754 }
755
756 for (auto* currLayer : negativeZOrderLayers)
757 writeLayers(ts, rootLayer, *currLayer, paintDirtyRect, behavior);
758
759 if (behavior.contains(RenderAsTextFlag::ShowLayerNesting))
760 ts.decreaseIndent();
761 }
762
763 if (shouldPaint) {
764 writeLayer(ts, layer, layerBounds, damageRect.rect(), clipRectToApply.rect(), paintsBackgroundSeparately ? LayerPaintPhaseForeground : LayerPaintPhaseAll, behavior);
765
766 if (behavior.contains(RenderAsTextFlag::ShowLayerFragments)) {
767 LayerFragments layerFragments;
768 layer.collectFragments(layerFragments, &rootLayer, paintDirtyRect, RenderLayer::PaginationInclusionMode::ExcludeCompositedPaginatedLayers, TemporaryClipRects, IgnoreOverlayScrollbarSize, RespectOverflowClip, offsetFromRoot);
769
770 if (layerFragments.size() > 1) {
771 TextStream::IndentScope indentScope(ts, 2);
772 for (unsigned i = 0; i < layerFragments.size(); ++i) {
773 const auto& fragment = layerFragments[i];
774 ts << indent << " fragment " << i << ": bounds in layer " << fragment.layerBounds << " fragment bounds " << fragment.boundingBox << "\n";
775 }
776 }
777 }
778
779 writeLayerRenderers(ts, layer, paintsBackgroundSeparately ? LayerPaintPhaseForeground : LayerPaintPhaseAll, behavior);
780 }
781
782 auto normalFlowLayers = layer.normalFlowLayers();
783 if (normalFlowLayers.size()) {
784 if (behavior.contains(RenderAsTextFlag::ShowLayerNesting)) {
785 ts << indent << " normal flow list (" << normalFlowLayers.size() << ")\n";
786 ts.increaseIndent();
787 }
788
789 for (auto* currLayer : normalFlowLayers)
790 writeLayers(ts, rootLayer, *currLayer, paintDirtyRect, behavior);
791
792 if (behavior.contains(RenderAsTextFlag::ShowLayerNesting))
793 ts.decreaseIndent();
794 }
795
796 auto positiveZOrderLayers = layer.positiveZOrderLayers();
797 if (positiveZOrderLayers.size()) {
798 size_t layerCount = positiveZOrderLayers.size();
799
800 if (layerCount) {
801 if (behavior.contains(RenderAsTextFlag::ShowLayerNesting)) {
802 ts << indent << " positive z-order list (" << layerCount << ")\n";
803 ts.increaseIndent();
804 }
805
806 for (auto* currLayer : positiveZOrderLayers)
807 writeLayers(ts, rootLayer, *currLayer, paintDirtyRect, behavior);
808
809 if (behavior.contains(RenderAsTextFlag::ShowLayerNesting))
810 ts.decreaseIndent();
811 }
812 }
813}

815static String nodePosition(Node* node)
816{
817 StringBuilder result;
818
819 auto* body = node->document().bodyOrFrameset();
820 Node* parent;
821 for (Node* n = node; n; n = parent) {
822 parent = n->parentOrShadowHostNode();
823 if (n != node)
824 result.append(" of ");
825 if (parent) {
826 if (body && n == body) {
827 // We don't care what offset body may be in the document.
828 result.append("body");
829 break;
830 }
831 if (n->isShadowRoot())
832 result.append('{', getTagName(n), '}');
833 else
834 result.append("child ", n->computeNodeIndex(), " {", getTagName(n), '}');
835 } else
836 result.append("document");
837 }
838
839 return result.toString();
840}

842static void writeSelection(TextStream& ts, const RenderBox& renderer)
843{
844 if (!renderer.isRenderView())
845 return;
846
847 Frame* frame = renderer.document().frame();
848 if (!frame)
849 return;
850
851 VisibleSelection selection = frame->selection().selection();
852 if (selection.isCaret()) {
853 ts << "caret: position " << selection.start().deprecatedEditingOffset() << " of " << nodePosition(selection.start().deprecatedNode());
854 if (selection.affinity() == Affinity::Upstream)
855 ts << " (upstream affinity)";
856 ts << "\n";
857 } else if (selection.isRange())
858 ts << "selection start: position " << selection.start().deprecatedEditingOffset() << " of " << nodePosition(selection.start().deprecatedNode()) << "\n"
859 << "selection end: position " << selection.end().deprecatedEditingOffset() << " of " << nodePosition(selection.end().deprecatedNode()) << "\n";
