51void RenderVTTCue::layout()
52{
53 StackStats::LayoutCheckPoint layoutCheckPoint;
54 RenderBlockFlow::layout();
55
56 // If WebVTT Regions are used, the regular WebVTT layout algorithm is no
57 // longer necessary, since cues having the region parameter set do not have
58 // any positioning parameters. Also, in this case, the regions themselves
59 // have positioning information.
60 if (!m_cue->regionId().isEmpty())
61 return;
62
63 LayoutStateMaintainer statePusher(*this, locationOffset(), hasTransform() || hasReflection() || style().isFlippedBlocksWritingMode());
64
65 if (m_cue->cueType()== TextTrackCue::WebVTT) {
66 if (m_cue->snapToLines())
67 repositionCueSnapToLinesSet();
68 else
69 repositionCueSnapToLinesNotSet();
70 } else
71 repositionGenericCue();
72}

74bool RenderVTTCue::initializeLayoutParameters(LegacyInlineFlowBox*& firstLineBox, LayoutUnit& step, LayoutUnit& position)
75{
76 ASSERT(firstChild());
77 if (!firstChild())
78 return false;
79
80 RenderBlock* parentBlock = containingBlock();
81
82 // firstChild() returns the wrapping (backdrop) <div>. The cue object is
83 // the <div>'s first child.
84 RenderObject& firstChild = *this->firstChild();
85 RenderElement& backdropElement = downcast<RenderElement>(firstChild);
86
87 firstLineBox = downcast<RenderInline>(*backdropElement.firstChild()).firstLineBox();
88 if (!firstLineBox)
89 firstLineBox = this->firstRootBox();
90
91 // 1. Horizontal: Let step be the height of the first line box in boxes.
92 // Vertical: Let step be the width of the first line box in boxes.
93 step = m_cue->getWritingDirection() == VTTCue::Horizontal ? firstLineBox->height() : firstLineBox->width();
94
95 // 2. If step is zero, then jump to the step labeled done positioning below.
96 if (!step)
97 return false;
98
99 // 3. Let line position be the text track cue computed line position.
100 int linePosition = m_cue->calculateComputedLinePosition();
101
102 // 4. Vertical Growing Left: Add one to line position then negate it.
103 if (m_cue->getWritingDirection() == VTTCue::VerticalGrowingLeft)
104 linePosition = -(linePosition + 1);
105
106 // 5. Let position be the result of multiplying step and line position.
107 position = step * linePosition;
108
109 // 6. Vertical Growing Left: Decrease position by the width of the
110 // bounding box of the boxes in boxes, then increase position by step.
111 if (m_cue->getWritingDirection() == VTTCue::VerticalGrowingLeft) {
112 position -= width();
113 position += step;
114 }
115
116 // 7. If line position is less than zero...
117 if (linePosition < 0) {
118 // Horizontal / Vertical: ... then increase position by the
119 // height / width of the video's rendering area ...
120 position += m_cue->getWritingDirection() == VTTCue::Horizontal ? parentBlock->height() : parentBlock->width();
121
122 // ... and negate step.
123 step = -step;
124 }
125
126 return true;
127}

129void RenderVTTCue::placeBoxInDefaultPosition(LayoutUnit position, bool& switched)
130{
131 // 8. Move all boxes in boxes ...
132 if (m_cue->getWritingDirection() == VTTCue::Horizontal)
133 // Horizontal: ... down by the distance given by position
134 setY(y() + position);
135 else
136 // Vertical: ... right by the distance given by position
137 setX(x() + position);
138
139 // 9. Default: Remember the position of all the boxes in boxes as their
140 // default position.
141 m_fallbackPosition = FloatPoint(x(), y());
142
143 // 10. Let switched be false.
144 switched = false;
145}

168RenderObject* RenderVTTCue::overlappingObjectForRect(const IntRect& rect) const
169{
170 for (RenderObject* box = previousSibling(); box; box = box->previousSibling()) {
171 IntRect boxRect = box->absoluteBoundingBoxRect();
172
173 if (rect.intersects(boxRect))
174 return box;
175 }
176
177 return 0;
178}

180bool RenderVTTCue::shouldSwitchDirection(LegacyInlineFlowBox* firstLineBox, LayoutUnit step) const
181{
182 LayoutUnit top = y();
183 LayoutUnit left = x();
184 LayoutUnit bottom { top + firstLineBox->height() };
185 LayoutUnit right { left + firstLineBox->width() };
186
187 // 12. Horizontal: If step is negative and the top of the first line
188 // box in boxes is now above the top of the video's rendering area,
189 // or if step is positive and the bottom of the first line box in
190 // boxes is now below the bottom of the video's rendering area, jump
191 // to the step labeled switch direction.
192 LayoutUnit parentHeight = containingBlock()->height();
193 if (m_cue->getWritingDirection() == VTTCue::Horizontal && ((step < 0 && top < 0) || (step > 0 && bottom > parentHeight)))
194 return true;
195
196 // 12. Vertical: If step is negative and the left edge of the first line
197 // box in boxes is now to the left of the left edge of the video's
198 // rendering area, or if step is positive and the right edge of the
199 // first line box in boxes is now to the right of the right edge of
200 // the video's rendering area, jump to the step labeled switch direction.
201 LayoutUnit parentWidth = containingBlock()->width();
202 if (m_cue->getWritingDirection() != VTTCue::Horizontal && ((step < 0 && left < 0) || (step > 0 && right > parentWidth)))
203 return true;
204
205 return false;
206}

208void RenderVTTCue::moveBoxesByStep(LayoutUnit step)
209{
210 // 13. Horizontal: Move all the boxes in boxes down by the distance
211 // given by step. (If step is negative, then this will actually
212 // result in an upwards movement of the boxes in absolute terms.)
213 if (m_cue->getWritingDirection() == VTTCue::Horizontal)
214 setY(y() + step);
215
216 // 13. Vertical: Move all the boxes in boxes right by the distance
217 // given by step. (If step is negative, then this will actually
218 // result in a leftwards movement of the boxes in absolute terms.)
219 else
220 setX(x() + step);
221}
222
223bool RenderVTTCue::switchDirection(bool& switched, LayoutUnit& step)
224{
225 // 15. Switch direction: Move all the boxes in boxes back to their
226 // default position as determined in the step above labeled default.
227 setX(m_fallbackPosition.x());
228 setY(m_fallbackPosition.y());
229
230 // 16. If switched is true, jump to the step labeled done
231 // positioning below.
232 if (switched)
233 return false;
234
235 // 17. Negate step.
236 step = -step;
237
238 // 18. Set switched to true.
239 switched = true;
240 return true;
241}

243void RenderVTTCue::moveIfNecessaryToKeepWithinContainer()
244{
245 IntRect containerRect = containingBlock()->absoluteBoundingBoxRect();
246 IntRect cueRect = absoluteBoundingBoxRect();
247
248 int topOverflow = cueRect.y() - containerRect.y();
249 int bottomOverflow = containerRect.maxY() - cueRect.maxY();
250
251 int verticalAdjustment = 0;
252 if (topOverflow < 0)
253 verticalAdjustment = -topOverflow;
254 else if (bottomOverflow < 0)
255 verticalAdjustment = bottomOverflow;
256
257 if (verticalAdjustment)
258 setY(y() + verticalAdjustment);
259
260 int leftOverflow = cueRect.x() - containerRect.x();
261 int rightOverflow = containerRect.maxX() - cueRect.maxX();
262
263 int horizontalAdjustment = 0;
264 if (leftOverflow < 0)
265 horizontalAdjustment = -leftOverflow;
266 else if (rightOverflow < 0)
267 horizontalAdjustment = rightOverflow;
268
269 if (horizontalAdjustment)
270 setX(x() + horizontalAdjustment);
271}

273bool RenderVTTCue::findNonOverlappingPosition(int& newX, int& newY) const
274{
275 newX = x();
276 newY = y();
277 IntRect srcRect = absoluteBoundingBoxRect();
278 IntRect destRect = srcRect;
279
280 // Move the box up, looking for a non-overlapping position:
281 while (RenderObject* box = overlappingObjectForRect(destRect)) {
282 if (m_cue->getWritingDirection() == VTTCue::Horizontal)
283 destRect.setY(box->absoluteBoundingBoxRect().y() - destRect.height());
284 else
285 destRect.setX(box->absoluteBoundingBoxRect().x() - destRect.width());
286 }
287
288 if (rectIsWithinContainer(destRect)) {
289 newX += destRect.x() - srcRect.x();
290 newY += destRect.y() - srcRect.y();
291 return true;
292 }
293
294 destRect = srcRect;
295
296 // Move the box down, looking for a non-overlapping position:
297 while (RenderObject* box = overlappingObjectForRect(destRect)) {
298 if (m_cue->getWritingDirection() == VTTCue::Horizontal)
299 destRect.setY(box->absoluteBoundingBoxRect().maxY());
300 else
301 destRect.setX(box->absoluteBoundingBoxRect().maxX());
302 }
303
304 if (rectIsWithinContainer(destRect)) {
305 newX += destRect.x() - srcRect.x();
306 newY += destRect.y() - srcRect.y();
307 return true;
308 }
309
310 return false;
311}

313void RenderVTTCue::repositionCueSnapToLinesSet()
314{
315 LegacyInlineFlowBox* firstLineBox;
316 LayoutUnit step;
317 LayoutUnit position;
318 if (!initializeLayoutParameters(firstLineBox, step, position))
319 return;
320
321 bool switched;
322 placeBoxInDefaultPosition(position, switched);
323
324 // 11. Step loop: If none of the boxes in boxes would overlap any of the boxes
325 // in output and all the boxes in output are within the video's rendering area
326 // then jump to the step labeled done positioning.
327 while (isOutside() || isOverlapping()) {
328 if (!shouldSwitchDirection(firstLineBox, step))
329 // 13. Move all the boxes in boxes ...
330 // 14. Jump back to the step labeled step loop.
331 moveBoxesByStep(step);
332 else if (!switchDirection(switched, step))
333 break;
334
335 // 19. Jump back to the step labeled step loop.
336 }
337
338 // Acommodate extra top and bottom padding, border or margin.
339 // Note: this is supported only for internal UA styling, not through the cue selector.
340 if (hasInlineDirectionBordersPaddingOrMargin())
341 moveIfNecessaryToKeepWithinContainer();
342}

344void RenderVTTCue::repositionGenericCue()
345{
346 ASSERT(firstChild());
347
348 // firstChild() returns the wrapping (backdrop) <div>. The cue object is
349 // the <div>'s first child.
350 RenderObject& firstChild = *this->firstChild();
351 RenderElement& backdropElement = downcast<RenderElement>(firstChild);
352
353 LegacyInlineFlowBox* firstLineBox = downcast<RenderInline>(*backdropElement.firstChild()).firstLineBox();
354 if (downcast<TextTrackCueGeneric>(*m_cue).useDefaultPosition() && firstLineBox) {
355 LayoutUnit parentWidth = containingBlock()->logicalWidth();
356 LayoutUnit width { firstLineBox->width() };
357 LayoutUnit right = (parentWidth / 2) - (width / 2);
358 setX(right);
359 }
360 repositionCueSnapToLinesNotSet();
361}
362
363void RenderVTTCue::repositionCueSnapToLinesNotSet()
364{
365 // 3. If none of the boxes in boxes would overlap any of the boxes in output, and all the boxes in
366 // output are within the video's rendering area, then jump to the step labeled done positioning below.
367 if (!isOutside() && !isOverlapping())
368 return;
369
370 // 4. If there is a position to which the boxes in boxes can be moved while maintaining the relative
371 // positions of the boxes in boxes to each other such that none of the boxes in boxes would overlap
372 // any of the boxes in output, and all the boxes in output would be within the video's rendering area,
373 // then move the boxes in boxes to the closest such position to their current position, and then jump
374 // to the step labeled done positioning below. If there are multiple such positions that are equidistant
375 // from their current position, use the highest one amongst them; if there are several at that height,
376 // then use the leftmost one amongst them.
377 moveIfNecessaryToKeepWithinContainer();
378 int x = 0;
379 int y = 0;
380 if (!findNonOverlappingPosition(x, y))
381 return;
382
383 setX(x);
384 setY(y);
385}
