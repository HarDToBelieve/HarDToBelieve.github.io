102bool RenderVideo::updateIntrinsicSize()
103{
104 LayoutSize size = calculateIntrinsicSize();
105 size.scale(style().effectiveZoom());
106
107 // Never set the element size to zero when in a media document.
108 if (size.isEmpty() && document().isMediaDocument())
109 return false;
110
111 // Treat the media player's natural size as visually non-empty.
112 if (videoElement().readyState() >= HTMLMediaElementEnums::HAVE_METADATA)
113 incrementVisuallyNonEmptyPixelCountIfNeeded(roundedIntSize(size));
114
115 if (size == intrinsicSize())
116 return false;
117
118 setIntrinsicSize(size);
119 setPreferredLogicalWidthsDirty(true);
120 setNeedsLayout();
121 return true;
122}
123
124LayoutSize RenderVideo::calculateIntrinsicSize()
125{
126 if (shouldApplySizeContainment(*this))
127 return LayoutSize();
128
129 // Spec text from 4.8.6
130 //
131 // The intrinsic width of a video element's playback area is the intrinsic width
132 // of the video resource, if that is available; otherwise it is the intrinsic
133 // width of the poster frame, if that is available; otherwise it is 300 CSS pixels.
134 //
135 // The intrinsic height of a video element's playback area is the intrinsic height
136 // of the video resource, if that is available; otherwise it is the intrinsic
137 // height of the poster frame, if that is available; otherwise it is 150 CSS pixels.
138 auto player = videoElement().player();
139 if (player && videoElement().readyState() >= HTMLVideoElement::HAVE_METADATA) {
140 LayoutSize size(player->naturalSize());
141 if (!size.isEmpty())
142 return size;
143 }
144
145 if (videoElement().shouldDisplayPosterImage() && !m_cachedImageSize.isEmpty() && !imageResource().errorOccurred())
146 return m_cachedImageSize;
147
148 // <video> in standalone media documents should not use the default 300x150
149 // size since they also have audio-only files. By setting the intrinsic
150 // size to 300x1 the video will resize itself in these cases, and audio will
151 // have the correct height (it needs to be > 0 for controls to render properly).
152 if (videoElement().document().isMediaDocument())
153 return LayoutSize(defaultSize().width(), 1);
154
155 return defaultSize();
156}

158void RenderVideo::imageChanged(WrappedImagePtr newImage, const IntRect* rect)
159{
160 RenderMedia::imageChanged(newImage, rect);
161
162 // Cache the image intrinsic size so we can continue to use it to draw the image correctly
163 // even if we know the video intrinsic size but aren't able to draw video frames yet
164 // (we don't want to scale the poster to the video size without keeping aspect ratio).
165 if (videoElement().shouldDisplayPosterImage())
166 m_cachedImageSize = intrinsicSize();
167
168 // The intrinsic size is now that of the image, but in case we already had the
169 // intrinsic size of the video we call this here to restore the video size.
170 updateIntrinsicSize();
171}

173IntRect RenderVideo::videoBox() const
174{
175 auto mediaPlayer = videoElement().player();
176 if (mediaPlayer && mediaPlayer->shouldIgnoreIntrinsicSize())
177 return snappedIntRect(contentBoxRect());
178
179 LayoutSize intrinsicSize = this->intrinsicSize();
180
181 if (videoElement().shouldDisplayPosterImage())
182 intrinsicSize = m_cachedImageSize;
183
184 return snappedIntRect(replacedContentRect(intrinsicSize));
185}

197void RenderVideo::paintReplaced(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
198{
199 auto mediaPlayer = videoElement().player();
200 bool displayingPoster = videoElement().shouldDisplayPosterImage();
201
202 if (!displayingPoster && !mediaPlayer) {
203 if (paintInfo.phase == PaintPhase::Foreground)
204 page().addRelevantUnpaintedObject(this, visualOverflowRect());
205 return;
206 }
207
208 LayoutRect rect = videoBox();
209 if (rect.isEmpty()) {
210 if (paintInfo.phase == PaintPhase::Foreground)
211 page().addRelevantUnpaintedObject(this, visualOverflowRect());
212 return;
213 }
214 rect.moveBy(paintOffset);
215
216 if (paintInfo.phase == PaintPhase::Foreground)
217 page().addRelevantRepaintedObject(this, rect);
218
219 LayoutRect contentRect = contentBoxRect();
220 contentRect.moveBy(paintOffset);
221 GraphicsContext& context = paintInfo.context();
222
223 if (context.detectingContentfulPaint()) {
224 context.setContentfulPaintDetected();
225 return;
226 }
227
228 bool clip = !contentRect.contains(rect);
229 GraphicsContextStateSaver stateSaver(context, clip);
230 if (clip)
231 context.clip(contentRect);
232
233 if (displayingPoster)
234 paintIntoRect(paintInfo, rect);
235 else if (!videoElement().isFullscreen() || !videoElement().supportsAcceleratedRendering()) {
236 if (paintInfo.paintBehavior.contains(PaintBehavior::FlattenCompositingLayers))
237 context.paintFrameForMedia(*mediaPlayer, rect);
238 else
239 mediaPlayer->paint(context, rect);
240 }
241}

262void RenderVideo::updatePlayer()
263{
264 if (renderTreeBeingDestroyed())
265 return;
266
267 bool intrinsicSizeChanged;
268 intrinsicSizeChanged = updateIntrinsicSize();
269 ASSERT_UNUSED(intrinsicSizeChanged, !intrinsicSizeChanged || !view().frameView().layoutContext().isInRenderTreeLayout());
270
271 auto mediaPlayer = videoElement().player();
272 if (!mediaPlayer)
273 return;
274
275 if (!videoElement().inActiveDocument()) {
276 mediaPlayer->setVisible(false);
277 return;
278 }
279
280 contentChanged(VideoChanged);
281
282 IntRect videoBounds = videoBox();
283 mediaPlayer->setSize(IntSize(videoBounds.width(), videoBounds.height()));
284 mediaPlayer->setVisible(!videoElement().elementIsHidden());
285 mediaPlayer->setShouldMaintainAspectRatio(style().objectFit() != ObjectFit::Fill);
286}

353bool RenderVideo::foregroundIsKnownToBeOpaqueInRect(const LayoutRect& localRect, unsigned maxDepthToTest) const
354{
355 if (videoElement().shouldDisplayPosterImage())
356 return RenderImage::foregroundIsKnownToBeOpaqueInRect(localRect, maxDepthToTest);
357
358 if (!videoBox().contains(enclosingIntRect(localRect)))
359 return false;
360
361 if (auto player = videoElement().player())
362 return player->hasAvailableVideoFrame();
363
364 return false;
365}
