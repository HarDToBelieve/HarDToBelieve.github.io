89void RenderView::scheduleLazyRepaint(RenderBox& renderer)
90{
91 if (renderer.renderBoxNeedsLazyRepaint())
92 return;
93 renderer.setRenderBoxNeedsLazyRepaint(true);
94 m_renderersNeedingLazyRepaint.add(&renderer);
95 if (!m_lazyRepaintTimer.isActive())
96 m_lazyRepaintTimer.startOneShot(0_s);
97}
98
99void RenderView::unscheduleLazyRepaint(RenderBox& renderer)
100{
101 if (!renderer.renderBoxNeedsLazyRepaint())
102 return;
103 renderer.setRenderBoxNeedsLazyRepaint(false);
104 m_renderersNeedingLazyRepaint.remove(&renderer);
105 if (m_renderersNeedingLazyRepaint.isEmpty())
106 m_lazyRepaintTimer.stop();
107}
108
109void RenderView::lazyRepaintTimerFired()
110{
111 for (auto& renderer : m_renderersNeedingLazyRepaint) {
112 renderer->repaint();
113 renderer->setRenderBoxNeedsLazyRepaint(false);
114 }
115 m_renderersNeedingLazyRepaint.clear();
116}

148void RenderView::layout()
149{
150 StackStats::LayoutCheckPoint layoutCheckPoint;
151 if (!document().paginated())
152 m_pageLogicalSize = { };
153
154 if (shouldUsePrintingLayout()) {
155 if (!m_pageLogicalSize)
156 m_pageLogicalSize = LayoutSize(logicalWidth(), 0_lu);
157 m_minPreferredLogicalWidth = m_pageLogicalSize->width();
158 m_maxPreferredLogicalWidth = m_minPreferredLogicalWidth;
159 }
160
161 // Use calcWidth/Height to get the new width/height, since this will take the full page zoom factor into account.
162 bool relayoutChildren = !shouldUsePrintingLayout() && (width() != viewWidth() || height() != viewHeight());
163 if (relayoutChildren) {
164 setChildNeedsLayout(MarkOnlyThis);
165
166 for (auto& box : childrenOfType<RenderBox>(*this)) {
167 if (box.hasRelativeLogicalHeight()
168 || box.style().logicalHeight().isPercentOrCalculated()
169 || box.style().logicalMinHeight().isPercentOrCalculated()
170 || box.style().logicalMaxHeight().isPercentOrCalculated()
171 || box.isSVGRoot()
172 )
173 box.setChildNeedsLayout(MarkOnlyThis);
174 }
175 }
176
177 ASSERT(!frameView().layoutContext().layoutState());
178 if (!needsLayout())
179 return;
180
181 LayoutStateMaintainer statePusher(*this, { }, false, m_pageLogicalSize.value_or(LayoutSize()).height(), m_pageLogicalHeightChanged);
182
183 m_pageLogicalHeightChanged = false;
184
185 RenderBlockFlow::layout();
186
187#ifndef NDEBUG
188 frameView().layoutContext().checkLayoutState();
189#endif
190 clearNeedsLayout();
191}

240void RenderView::mapLocalToContainer(const RenderLayerModelObject* ancestorContainer, TransformState& transformState, MapCoordinatesFlags mode, bool* wasFixed) const
241{
242 // If a container was specified, and was not nullptr or the RenderView,
243 // then we should have found it by now.
244 ASSERT_ARG(ancestorContainer, !ancestorContainer || ancestorContainer == this);
245 ASSERT_UNUSED(wasFixed, !wasFixed || *wasFixed == (mode & IsFixed));
246
247 if (mode & IsFixed)
248 transformState.move(toLayoutSize(frameView().scrollPositionRespectingCustomFixedPosition()));
249
250 if (!ancestorContainer && mode & UseTransforms && shouldUseTransformFromContainer(nullptr)) {
251 TransformationMatrix t;
252 getTransformFromContainer(nullptr, LayoutSize(), t);
253 transformState.applyTransform(t);
254 }
255}

257const RenderObject* RenderView::pushMappingToContainer(const RenderLayerModelObject* ancestorToStopAt, RenderGeometryMap& geometryMap) const
258{
259 // If a container was specified, and was not nullptr or the RenderView,
260 // then we should have found it by now.
261 ASSERT_ARG(ancestorToStopAt, !ancestorToStopAt || ancestorToStopAt == this);
262
263 LayoutPoint scrollPosition = frameView().scrollPositionRespectingCustomFixedPosition();
264
265 if (!ancestorToStopAt && shouldUseTransformFromContainer(nullptr)) {
266 TransformationMatrix t;
267 getTransformFromContainer(nullptr, LayoutSize(), t);
268 geometryMap.pushView(this, toLayoutSize(scrollPosition), &t);
269 } else
270 geometryMap.pushView(this, toLayoutSize(scrollPosition));
271
272 return nullptr;
273}

302void RenderView::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
303{
304 // If we ever require layout but receive a paint anyway, something has gone horribly wrong.
305 ASSERT(!needsLayout());
306 // RenderViews should never be called to paint with an offset not on device pixels.
307 ASSERT(LayoutPoint(IntPoint(paintOffset.x(), paintOffset.y())) == paintOffset);
308
309 // This avoids painting garbage between columns if there is a column gap.
310 if (frameView().pagination().mode != Pagination::Unpaginated && paintInfo.shouldPaintWithinRoot(*this))
311 paintInfo.context().fillRect(paintInfo.rect, frameView().baseBackgroundColor());
312
313 paintObject(paintInfo, paintOffset);
314}

316RenderElement* RenderView::rendererForRootBackground() const
317{
318 auto* firstChild = this->firstChild();
319 if (!firstChild)
320 return nullptr;
321 ASSERT(is<RenderElement>(*firstChild));
322 auto& documentRenderer = downcast<RenderElement>(*firstChild);
323
324 if (documentRenderer.hasBackground())
325 return &documentRenderer;
326
327 // We propagate the background only for HTML content.
328 if (!is<HTMLHtmlElement>(documentRenderer.element()))
329 return &documentRenderer;
330
331 if (auto* body = document().body()) {
332 if (auto* renderer = body->renderer())
333 return renderer;
334 }
335 return &documentRenderer;
336}

338static inline bool rendererObscuresBackground(const RenderElement& rootElement)
339{
340 auto& style = rootElement.style();
341 if (style.visibility() != Visibility::Visible || style.opacity() != 1 || style.hasTransform())
342 return false;
343
344 if (style.hasBorderRadius())
345 return false;
346
347 if (rootElement.isComposited())
348 return false;
349
350 auto* rendererForBackground = rootElement.view().rendererForRootBackground();
351 if (!rendererForBackground)
352 return false;
353
354 if (rendererForBackground->style().backgroundClip() == FillBox::Text)
355 return false;
356
357 return true;
358}

360void RenderView::paintBoxDecorations(PaintInfo& paintInfo, const LayoutPoint&)
361{
362 if (!paintInfo.shouldPaintWithinRoot(*this))
363 return;
364
365 // Check to see if we are enclosed by a layer that requires complex painting rules. If so, we cannot blit
366 // when scrolling, and we need to use slow repaints. Examples of layers that require this are transparent layers,
367 // layers with reflections, or transformed layers.
368 // FIXME: This needs to be dynamic. We should be able to go back to blitting if we ever stop being inside
369 // a transform, transparency layer, etc.
370 for (HTMLFrameOwnerElement* element = document().ownerElement(); element && element->renderer(); element = element->document().ownerElement()) {
371 RenderLayer* layer = element->renderer()->enclosingLayer();
372 if (layer->cannotBlitToWindow()) {
373 frameView().setCannotBlitToWindow();
374 break;
375 }
376
377 if (RenderLayer* compositingLayer = layer->enclosingCompositingLayerForRepaint()) {
378 if (!compositingLayer->backing()->paintsIntoWindow()) {
379 frameView().setCannotBlitToWindow();
380 break;
381 }
382 }
383 }
384
385 if (document().ownerElement())
386 return;
387
388 if (paintInfo.skipRootBackground())
389 return;
390
391 bool rootFillsViewport = false;
392 bool rootObscuresBackground = false;
393 Element* documentElement = document().documentElement();
394 if (RenderElement* rootRenderer = documentElement ? documentElement->renderer() : nullptr) {
395 // The document element's renderer is currently forced to be a block, but may not always be.
396 RenderBox* rootBox = is<RenderBox>(*rootRenderer) ? downcast<RenderBox>(rootRenderer) : nullptr;
397 rootFillsViewport = rootBox && !rootBox->x() && !rootBox->y() && rootBox->width() >= width() && rootBox->height() >= height();
398 rootObscuresBackground = rendererObscuresBackground(*rootRenderer);
399 }
400
401 compositor().rootBackgroundColorOrTransparencyChanged();
402
403 Page* page = document().page();
404 float pageScaleFactor = page ? page->pageScaleFactor() : 1;
405
406 // If painting will entirely fill the view, no need to fill the background.
407 if (rootFillsViewport && rootObscuresBackground && pageScaleFactor >= 1)
408 return;
409
410 // This code typically only executes if the root element's visibility has been set to hidden,
411 // if there is a transform on the <html>, or if there is a page scale factor less than 1.
412 // Only fill with a background color (typically white) if we're the root document,
413 // since iframes/frames with no background in the child document should show the parent's background.
414 // We use the base background color unless the backgroundShouldExtendBeyondPage setting is set,
415 // in which case we use the document's background color.
416 if (frameView().isTransparent()) // FIXME: This needs to be dynamic. We should be able to go back to blitting if we ever stop being transparent.
417 frameView().setCannotBlitToWindow(); // The parent must show behind the child.
418 else {
419 const Color& documentBackgroundColor = frameView().documentBackgroundColor();
420 const Color& backgroundColor = (settings().backgroundShouldExtendBeyondPage() && documentBackgroundColor.isValid()) ? documentBackgroundColor : frameView().baseBackgroundColor();
421 if (backgroundColor.isVisible()) {
422 CompositeOperator previousOperator = paintInfo.context().compositeOperation();
423 paintInfo.context().setCompositeOperation(CompositeOperator::Copy);
424 paintInfo.context().fillRect(paintInfo.rect, backgroundColor);
425 paintInfo.context().setCompositeOperation(previousOperator);
426 } else
427 paintInfo.context().clearRect(paintInfo.rect);
428 }
429}

436void RenderView::repaintRootContents()
437{
438 if (layer()->isComposited()) {
439 layer()->setBackingNeedsRepaint(GraphicsLayer::DoNotClipToLayer);
440 return;
441 }
442
443 // Always use layoutOverflowRect() to fix rdar://problem/27182267.
444 // This should be cleaned up via webkit.org/b/159913 and webkit.org/b/159914.
445 RenderLayerModelObject* repaintContainer = containerForRepaint();
446 repaintUsingContainer(repaintContainer, computeRectForRepaint(layoutOverflowRect(), repaintContainer));
447}

449void RenderView::repaintViewRectangle(const LayoutRect& repaintRect) const
450{
451 if (!shouldRepaint(repaintRect))
452 return;
453
454 // FIXME: enclosingRect is needed as long as we integral snap ScrollView/FrameView/RenderWidget size/position.
455 IntRect enclosingRect = enclosingIntRect(repaintRect);
456 if (auto ownerElement = document().ownerElement()) {
457 RenderBox* ownerBox = ownerElement->renderBox();
458 if (!ownerBox)
459 return;
460 LayoutRect viewRect = this->viewRect();
461#if PLATFORM(IOS_FAMILY)
462 // Don't clip using the visible rect since clipping is handled at a higher level on iPhone.
463 LayoutRect adjustedRect = enclosingRect;
464#else
465 LayoutRect adjustedRect = intersection(enclosingRect, viewRect);
466#endif
467 adjustedRect.moveBy(-viewRect.location());
468 adjustedRect.moveBy(ownerBox->contentBoxRect().location());
469
470 // A dirty rect in an iframe is relative to the contents of that iframe.
471 // When we traverse between parent frames and child frames, we need to make sure
472 // that the coordinate system is mapped appropriately between the iframe's contents
473 // and the Renderer that contains the iframe. This transformation must account for a
474 // left scrollbar (if one exists).
475 FrameView& frameView = this->frameView();
476 if (frameView.shouldPlaceVerticalScrollbarOnLeft() && frameView.verticalScrollbar())
477 adjustedRect.move(LayoutSize(frameView.verticalScrollbar()->occupiedWidth(), 0));
478
479 ownerBox->repaintRectangle(adjustedRect);
480 return;
481 }
482
483 frameView().addTrackedRepaintRect(snapRectToDevicePixels(repaintRect, document().deviceScaleFactor()));
484 if (!m_accumulatedRepaintRegion) {
485 frameView().repaintContentRectangle(enclosingRect);
486 return;
487 }
488 m_accumulatedRepaintRegion->unite(enclosingRect);
489
490 // Region will get slow if it gets too complex. Merge all rects so far to bounds if this happens.
491 // FIXME: Maybe there should be a region type that does this automatically.
492 static const unsigned maximumRepaintRegionGridSize = 16 * 16;
493 if (m_accumulatedRepaintRegion->gridSize() > maximumRepaintRegionGridSize)
494 m_accumulatedRepaintRegion = makeUnique<Region>(m_accumulatedRepaintRegion->bounds());
495}

516std::optional<LayoutRect> RenderView::computeVisibleRectInContainer(const LayoutRect& rect, const RenderLayerModelObject* container, VisibleRectContext context) const
517{
518 // If a container was specified, and was not nullptr or the RenderView,
519 // then we should have found it by now.
520 ASSERT_ARG(container, !container || container == this);
521
522 if (printing())
523 return rect;
524
525 LayoutRect adjustedRect = rect;
526 if (style().isFlippedBlocksWritingMode()) {
527 // We have to flip by hand since the view's logical height has not been determined. We
528 // can use the viewport width and height.
529 if (style().isHorizontalWritingMode())
530 adjustedRect.setY(viewHeight() - adjustedRect.maxY());
531 else
532 adjustedRect.setX(viewWidth() - adjustedRect.maxX());
533 }
534
535 if (context.hasPositionFixedDescendant)
536 adjustedRect.moveBy(frameView().scrollPositionRespectingCustomFixedPosition());
537
538 // Apply our transform if we have one (because of full page zooming).
539 if (!container && layer() && layer()->transform())
540 adjustedRect = LayoutRect(layer()->transform()->mapRect(snapRectToDevicePixels(adjustedRect, document().deviceScaleFactor())));
541 return adjustedRect;
542}

669void RenderView::updateHitTestResult(HitTestResult& result, const LayoutPoint& point)
670{
671 if (result.innerNode())
672 return;
673
674 if (multiColumnFlow() && multiColumnFlow()->firstMultiColumnSet())
675 return multiColumnFlow()->firstMultiColumnSet()->updateHitTestResult(result, point);
676
677 if (auto* node = nodeForHitTest()) {
678 result.setInnerNode(node);
679 if (!result.innerNonSharedNode())
680 result.setInnerNonSharedNode(node);
681
682 LayoutPoint adjustedPoint = point;
683 offsetForContents(adjustedPoint);
684
685 result.setLocalPoint(adjustedPoint);
686 }

689// FIXME: This function is obsolete and only used by embedded WebViews inside AppKit NSViews.
690// Do not add callers of this function!
691// The idea here is to take into account what object is moving the pagination point, and
692// thus choose the best place to chop it.
693void RenderView::setBestTruncatedAt(int y, RenderBoxModelObject* forRenderer, bool forcedBreak)
694{
695 // Nobody else can set a page break once we have a forced break.
696 if (m_legacyPrinting.m_forcedPageBreak)
697 return;
698
699 // Forced breaks always win over unforced breaks.
700 if (forcedBreak) {
701 m_legacyPrinting.m_forcedPageBreak = true;
702 m_legacyPrinting.m_bestTruncatedAt = y;
703 return;
704 }
705
706 // Prefer the widest object that tries to move the pagination point
707 LayoutRect boundingBox = forRenderer->borderBoundingBox();
708 if (boundingBox.width() > m_legacyPrinting.m_truncatorWidth) {
709 m_legacyPrinting.m_truncatorWidth = boundingBox.width();
710 m_legacyPrinting.m_bestTruncatedAt = y;
711 }
712}

800void RenderView::resumePausedImageAnimationsIfNeeded(const IntRect& visibleRect)
801{
802 Vector<std::pair<RenderElement*, CachedImage*>, 10> toRemove;
803 for (auto& it : m_renderersWithPausedImageAnimation) {
804 auto* renderer = it.key;
805 for (auto* image : it.value) {
806 if (renderer->repaintForPausedImageAnimationsIfNeeded(visibleRect, *image))
807 toRemove.append(std::make_pair(renderer, image));
808 }
809 }
810 for (auto& pair : toRemove)
811 removeRendererWithPausedImageAnimations(*pair.first, *pair.second);
812}

838unsigned RenderView::pageNumberForBlockProgressionOffset(int offset) const
839{
840 int columnNumber = 0;
841 const Pagination& pagination = page().pagination();
842 if (pagination.mode == Pagination::Unpaginated)
843 return columnNumber;
844
845 bool progressionIsInline = false;
846 bool progressionIsReversed = false;
847
848 if (multiColumnFlow()) {
849 progressionIsInline = multiColumnFlow()->progressionIsInline();
850 progressionIsReversed = multiColumnFlow()->progressionIsReversed();
851 } else
852 return columnNumber;
853
854 if (!progressionIsInline) {
855 if (!progressionIsReversed)
856 columnNumber = (pagination.pageLength + pagination.gap - offset) / (pagination.pageLength + pagination.gap);
857 else
858 columnNumber = offset / (pagination.pageLength + pagination.gap);
859 }
860
861 return columnNumber;
862}

864unsigned RenderView::pageCount() const
865{
866 const Pagination& pagination = page().pagination();
867 if (pagination.mode == Pagination::Unpaginated)
868 return 0;
869
870 if (multiColumnFlow() && multiColumnFlow()->firstMultiColumnSet())
871 return multiColumnFlow()->firstMultiColumnSet()->columnCount();
872
873 return 0;
874}
