58void WidgetHierarchyUpdatesSuspensionScope::moveWidgets()
59{
60 while (!widgetNewParentMap().isEmpty()) {
61 auto map = std::exchange(widgetNewParentMap(), { });
62 for (auto& entry : map) {
63 auto& child = *entry.key;
64 auto* currentParent = child.parent();
65 auto* newParent = entry.value;
66 if (newParent != currentParent) {
67 if (currentParent)
68 currentParent->removeChild(child);
69 if (newParent)
70 newParent->addChild(child);
71 }
72 }
73 }
74}

76static void moveWidgetToParentSoon(Widget& child, FrameView* parent)
77{
78 if (!WidgetHierarchyUpdatesSuspensionScope::isSuspended()) {
79 if (parent)
80 parent->addChild(child);
81 else
82 child.removeFromParent();
83 return;
84 }
85 WidgetHierarchyUpdatesSuspensionScope::scheduleWidgetToMove(child, parent);
86}

116// Widgets are always placed on integer boundaries, so rounding the size is actually
117// the desired behavior. This function is here because it's otherwise seldom what we
118// want to do with a LayoutRect.
119static inline IntRect roundedIntRect(const LayoutRect& rect)
120{
121 return IntRect(roundedIntPoint(rect.location()), roundedIntSize(rect.size()));
122}
123
124bool RenderWidget::setWidgetGeometry(const LayoutRect& frame)
125{
126 IntRect clipRect = roundedIntRect(enclosingLayer()->childrenClipRect());
127 IntRect newFrameRect = roundedIntRect(frame);
128 IntRect oldFrameRect = m_widget->frameRect();
129 bool clipChanged = m_clipRect != clipRect;
130 bool boundsChanged = oldFrameRect != newFrameRect;
131
132 if (!boundsChanged && !clipChanged)
133 return false;
134
135 m_clipRect = clipRect;
136
137 auto weakThis = makeWeakPtr(*this);
138 // These calls *may* cause this renderer to disappear from underneath...
139 if (boundsChanged)
140 m_widget->setFrameRect(newFrameRect);
141 else if (clipChanged)
142 m_widget->clipRectChanged();
143 // ...so we follow up with a sanity check.
144 if (!weakThis)
145 return true;
146
147 if (boundsChanged)
148 view().compositor().widgetDidChangeSize(*this);
149
150 return oldFrameRect.size() != newFrameRect.size();
151}

153bool RenderWidget::updateWidgetGeometry()
154{
155 if (!m_widget->transformsAffectFrameRect())
156 return setWidgetGeometry(absoluteContentBox());
157
158 LayoutRect contentBox = contentBoxRect();
159 LayoutRect absoluteContentBox(localToAbsoluteQuad(FloatQuad(contentBox)).boundingBox());
160 if (m_widget->isFrameView()) {
161 contentBox.setLocation(absoluteContentBox.location());
162 return setWidgetGeometry(contentBox);
163 }
164
165 return setWidgetGeometry(absoluteContentBox);
166}

168void RenderWidget::setWidget(RefPtr<Widget>&& widget)
169{
170 if (widget == m_widget)
171 return;
172
173 if (m_widget) {
174 moveWidgetToParentSoon(*m_widget, nullptr);
175 view().frameView().willRemoveWidgetFromRenderTree(*m_widget);
176 widgetRendererMap().remove(m_widget.get());
177 m_widget = nullptr;
178 }
179 m_widget = widget;
180 if (m_widget) {
181 widgetRendererMap().add(m_widget.get(), this);
182 view().frameView().didAddWidgetToRenderTree(*m_widget);
183 // If we've already received a layout, apply the calculated space to the
184 // widget immediately, but we have to have really been fully constructed.
185 if (hasInitializedStyle()) {
186 if (!needsLayout()) {
187 auto weakThis = makeWeakPtr(*this);
188 updateWidgetGeometry();
189 if (!weakThis)
190 return;
191 }
192
193 if (style().visibility() != Visibility::Visible)
194 m_widget->hide();
195 else {
196 m_widget->show();
197 repaint();
198 }
199 }
200 moveWidgetToParentSoon(*m_widget, &view().frameView());
201 }
202}

223void RenderWidget::paintContents(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
224{
225 if (paintInfo.requireSecurityOriginAccessForWidgets) {
226 if (auto contentDocument = frameOwnerElement().contentDocument()) {
227 if (!document().securityOrigin().isSameOriginDomain(contentDocument->securityOrigin()))
228 return;
229 }
230 }
231
232 IntPoint contentPaintOffset = roundedIntPoint(paintOffset + location() + contentBoxRect().location());
233 // Tell the widget to paint now. This is the only time the widget is allowed
234 // to paint itself. That way it will composite properly with z-indexed layers.
235 LayoutRect paintRect = paintInfo.rect;
236
237 OptionSet<PaintBehavior> oldBehavior = PaintBehavior::Normal;
238 if (is<FrameView>(*m_widget) && (paintInfo.paintBehavior & PaintBehavior::TileFirstPaint)) {
239 FrameView& frameView = downcast<FrameView>(*m_widget);
240 oldBehavior = frameView.paintBehavior();
241 frameView.setPaintBehavior(oldBehavior | PaintBehavior::TileFirstPaint);
242 }
243
244 IntPoint widgetLocation = m_widget->frameRect().location();
245 IntSize widgetPaintOffset = contentPaintOffset - widgetLocation;
246 // When painting widgets into compositing layers, tx and ty are relative to the enclosing compositing layer,
247 // not the root. In this case, shift the CTM and adjust the paintRect to be root-relative to fix plug-in drawing.
248 if (!widgetPaintOffset.isZero()) {
249 paintInfo.context().translate(widgetPaintOffset);
250 paintRect.move(-widgetPaintOffset);
251 }
252
253 if (paintInfo.eventRegionContext) {
254 AffineTransform transform;
255 transform.translate(contentPaintOffset);
256 paintInfo.eventRegionContext->pushTransform(transform);
257 }
258
259 // FIXME: Remove repaintrect enclosing/integral snapping when RenderWidget becomes device pixel snapped.
260 m_widget->paint(paintInfo.context(), snappedIntRect(paintRect), paintInfo.requireSecurityOriginAccessForWidgets ? Widget::SecurityOriginPaintPolicy::AccessibleOriginOnly : Widget::SecurityOriginPaintPolicy::AnyOrigin, paintInfo.eventRegionContext);
261
262 if (paintInfo.eventRegionContext)
263 paintInfo.eventRegionContext->popTransform();
264
265 if (!widgetPaintOffset.isZero())
266 paintInfo.context().translate(-widgetPaintOffset);
267
268 if (is<FrameView>(*m_widget)) {
269 FrameView& frameView = downcast<FrameView>(*m_widget);
270 bool runOverlapTests = !frameView.useSlowRepaintsIfNotOverlapped();
271 if (paintInfo.overlapTestRequests && runOverlapTests) {
272 ASSERT(!paintInfo.overlapTestRequests->contains(this) || (paintInfo.overlapTestRequests->get(this) == m_widget->frameRect()));
273 paintInfo.overlapTestRequests->set(this, m_widget->frameRect());
274 }
275 if (paintInfo.paintBehavior & PaintBehavior::TileFirstPaint)
276 frameView.setPaintBehavior(oldBehavior);
277 }
278}

280void RenderWidget::paint(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
281{
282 if (!shouldPaint(paintInfo, paintOffset))
283 return;
284
285 if (paintInfo.context().detectingContentfulPaint())
286 return;
287
288 LayoutPoint adjustedPaintOffset = paintOffset + location();
289
290 if (hasVisibleBoxDecorations() && (paintInfo.phase == PaintPhase::Foreground || paintInfo.phase == PaintPhase::Selection))
291 paintBoxDecorations(paintInfo, adjustedPaintOffset);
292
293 if (paintInfo.phase == PaintPhase::Mask) {
294 paintMask(paintInfo, adjustedPaintOffset);
295 return;
296 }
297
298 if ((paintInfo.phase == PaintPhase::Outline || paintInfo.phase == PaintPhase::SelfOutline) && hasOutline())
299 paintOutline(paintInfo, LayoutRect(adjustedPaintOffset, size()));
300
301 // FIXME: Shouldn't check if the frame view needs layout during event region painting. This is a workaround
302 // for the fact that non-composited frames depend on their enclosing compositing layer to perform an event
303 // region update on their behalf. See <https://webkit.org/b/210311> for more details.
304 bool needsEventRegionContentPaint = paintInfo.phase == PaintPhase::EventRegion && is<FrameView>(m_widget) && !downcast<FrameView>(*m_widget).needsLayout();
305 if (paintInfo.phase != PaintPhase::Foreground && !needsEventRegionContentPaint)
306 return;
307
308 if (style().hasBorderRadius()) {
309 LayoutRect borderRect = LayoutRect(adjustedPaintOffset, size());
310
311 if (borderRect.isEmpty())
312 return;
313
314 // Push a clip if we have a border radius, since we want to round the foreground content that gets painted.
315 paintInfo.context().save();
316 FloatRoundedRect roundedInnerRect = FloatRoundedRect(style().getRoundedInnerBorderFor(borderRect,
317 paddingTop() + borderTop(), paddingBottom() + borderBottom(), paddingLeft() + borderLeft(), paddingRight() + borderRight(), true, true));
318 clipRoundedInnerRect(paintInfo.context(), borderRect, roundedInnerRect);
319 }
320
321 if (m_widget)
322 paintContents(paintInfo, paintOffset);
323
324 if (style().hasBorderRadius())
325 paintInfo.context().restore();
326
327 if (paintInfo.phase == PaintPhase::EventRegion)
328 return;
329
330 // Paint a partially transparent wash over selected widgets.
331 if (isSelected() && !document().printing()) {
332 // FIXME: selectionRect() is in absolute, not painting coordinates.
333 paintInfo.context().fillRect(snappedIntRect(selectionRect()), selectionBackgroundColor());
334 }
335
336 if (hasLayer() && layer()->canResize()) {
337 ASSERT(layer()->scrollableArea());
338 layer()->scrollableArea()->paintResizer(paintInfo.context(), roundedIntPoint(adjustedPaintOffset), paintInfo.rect);
339 }
340}

348RenderWidget::ChildWidgetState RenderWidget::updateWidgetPosition()
349{
350 if (!m_widget)
351 return ChildWidgetState::Destroyed;
352
353 auto weakThis = makeWeakPtr(*this);
354 bool widgetSizeChanged = updateWidgetGeometry();
355 if (!weakThis || !m_widget)
356 return ChildWidgetState::Destroyed;
357
358 // if the frame size got changed, or if view needs layout (possibly indicating
359 // content size is wrong) we have to do a layout to set the right widget size.
360 if (is<FrameView>(*m_widget)) {
361 FrameView& frameView = downcast<FrameView>(*m_widget);
362 // Check the frame's page to make sure that the frame isn't in the process of being destroyed.
363 if ((widgetSizeChanged || frameView.needsLayout()) && frameView.frame().page() && frameView.frame().document())
364 frameView.layoutContext().layout();
365 }
366 return ChildWidgetState::Valid;
367}

388bool RenderWidget::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction action)
389{
390 auto shouldHitTestChildFrameContent = request.allowsChildFrameContent() || (request.allowsVisibleChildFrameContent() && visibleToHitTesting(request));
391 auto hasRenderView = is<FrameView>(widget()) && downcast<FrameView>(*widget()).renderView();
392 if (shouldHitTestChildFrameContent && hasRenderView) {
393 FrameView& childFrameView = downcast<FrameView>(*widget());
394
395 LayoutPoint adjustedLocation = accumulatedOffset + location();
396 LayoutPoint contentOffset = LayoutPoint(borderLeft() + paddingLeft(), borderTop() + paddingTop()) - toIntSize(childFrameView.scrollPosition());
397 HitTestLocation newHitTestLocation(locationInContainer, -adjustedLocation - contentOffset);
398 HitTestRequest newHitTestRequest(request.type() | HitTestRequest::Type::ChildFrameHitTest);
399 HitTestResult childFrameResult(newHitTestLocation);
400
401 auto* document = childFrameView.frame().document();
402 if (!document)
403 return false;
404 bool isInsideChildFrame = document->hitTest(newHitTestRequest, newHitTestLocation, childFrameResult);
405
406 if (request.resultIsElementList())
407 result.append(childFrameResult, request);
408 else if (isInsideChildFrame)
409 result = childFrameResult;
410
411 if (isInsideChildFrame)
412 return true;
413 }
414
415 bool hadResult = result.innerNode();
416 bool inside = RenderReplaced::nodeAtPoint(request, result, locationInContainer, accumulatedOffset, action);
417
418 // Check to see if we are really over the widget itself (and not just in the border/padding area).
419 if ((inside || result.isRectBasedTest()) && !hadResult && result.innerNode() == &frameOwnerElement())
420 result.setIsOverWidget(contentBoxRect().contains(result.localPoint()));
421 return inside;
422}

424bool RenderWidget::requiresLayer() const
425{
426 return RenderReplaced::requiresLayer() || requiresAcceleratedCompositing();
427}
428
429bool RenderWidget::requiresAcceleratedCompositing() const
430{
431 // If this is a renderer with a contentDocument and that document needs a layer, then we need a layer.
432 if (Document* contentDocument = frameOwnerElement().contentDocument()) {
433 if (RenderView* view = contentDocument->renderView())
434 return view->usesCompositing();
435 }
436
437 return false;
438}
