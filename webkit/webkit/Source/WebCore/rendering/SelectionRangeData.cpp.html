49struct SelectionContext {
50
51 using RendererMap = HashMap<RenderObject*, std::unique_ptr<RenderSelectionInfo>>;
52 using RenderBlockMap = HashMap<const RenderBlock*, std::unique_ptr<RenderBlockSelectionInfo>>;
53
54 unsigned startOffset;
55 unsigned endOffset;
56 RendererMap renderers;
57 RenderBlockMap blocks;
58};
59
60}

62static RenderObject* rendererAfterOffset(const RenderObject& renderer, unsigned offset)
63{
64 auto* child = renderer.childAt(offset);
65 return child ? child : renderer.nextInPreOrderAfterChildren();
66}
67
68static bool isValidRendererForSelection(const RenderObject& renderer, const RenderRange& selection)
69{
70 return (renderer.canBeSelectionLeaf() || &renderer == selection.start() || &renderer == selection.end())
71 && renderer.selectionState() != RenderObject::HighlightState::None
72 && renderer.containingBlock();
73}
74
75static RenderBlock* containingBlockBelowView(const RenderObject& renderer)
76{
77 auto* containingBlock = renderer.containingBlock();
78 return is<RenderView>(containingBlock) ? nullptr : containingBlock;
79}

81static SelectionContext collectSelectionData(const RenderRange& selection, bool repaintDifference)
82{
83 SelectionContext oldSelectionData { selection.startOffset(), selection.endOffset(), { }, { } };
84 // Blocks contain selected objects and fill gaps between them, either on the left, right, or in between lines and blocks.
85 // In order to get the repaint rect right, we have to examine left, middle, and right rects individually, since otherwise
86 // the union of those rects might remain the same even when changes have occurred.
87 auto* start = selection.start();
88 RenderObject* stop = nullptr;
89 if (selection.end())
90 stop = rendererAfterOffset(*selection.end(), selection.endOffset());
91 RenderRangeIterator selectionIterator(start);
92 while (start && start != stop) {
93 if (isValidRendererForSelection(*start, selection)) {
94 // Blocks are responsible for painting line gaps and margin gaps. They must be examined as well.
95 oldSelectionData.renderers.set(start, makeUnique<RenderSelectionInfo>(*start, true));
96 if (repaintDifference) {
97 for (auto* block = containingBlockBelowView(*start); block; block = containingBlockBelowView(*block)) {
98 auto& blockInfo = oldSelectionData.blocks.add(block, nullptr).iterator->value;
99 if (blockInfo)
100 break;
101 blockInfo = makeUnique<RenderBlockSelectionInfo>(*block);
102 }
103 }
104 }
105 start = selectionIterator.next();
106 }
107 return oldSelectionData;
108}

119void SelectionRangeData::set(const RenderRange& selection, RepaintMode blockRepaintMode)
120{
121 if ((selection.start() && !selection.end()) || (selection.end() && !selection.start()))
122 return;
123 // Just return if the selection hasn't changed.
124 auto isCaret = m_renderView.frame().selection().isCaret();
125 if (selection == m_renderRange && m_selectionWasCaret == isCaret)
126 return;
127#if ENABLE(SERVICE_CONTROLS)
128 // Clear the current rects and create a notifier for the new rects we are about to gather.
129 // The Notifier updates the Editor when it goes out of scope and is destroyed.
130 auto notifier = m_selectionGeometryGatherer.clearAndCreateNotifier();
131#endif
132 m_selectionWasCaret = isCaret;
133 apply(selection, blockRepaintMode);
134}

136void SelectionRangeData::clear()
137{
138 m_renderView.layer()->repaintBlockSelectionGaps();
139 set({ }, SelectionRangeData::RepaintMode::NewMinusOld);

142void SelectionRangeData::repaint() const
143{
144 HashSet<RenderBlock*> processedBlocks;
145 RenderObject* end = nullptr;
146 if (m_renderRange.end())
147 end = rendererAfterOffset(*m_renderRange.end(), m_renderRange.endOffset());
148 RenderRangeIterator highlightIterator(m_renderRange.start());
149 for (auto* renderer = highlightIterator.current(); renderer && renderer != end; renderer = highlightIterator.next()) {
150 if (!renderer->canBeSelectionLeaf() && renderer != m_renderRange.start() && renderer != m_renderRange.end())
151 continue;
152 if (renderer->selectionState() == RenderObject::HighlightState::None)
153 continue;
154 RenderSelectionInfo(*renderer, true).repaint();
155 // Blocks are responsible for painting line gaps and margin gaps. They must be examined as well.
156 for (auto* block = containingBlockBelowView(*renderer); block; block = containingBlockBelowView(*block)) {
157 if (!processedBlocks.add(block).isNewEntry)
158 break;
159 RenderSelectionInfo(*block, true).repaint();
160 }
161 }

164IntRect SelectionRangeData::collectBounds(ClipToVisibleContent clipToVisibleContent) const
165{
166 LOG_WITH_STREAM(Selection, stream << "SelectionData::collectBounds (clip to visible " << (clipToVisibleContent == ClipToVisibleContent::Yes ? "yes" : "no"));
167
168 SelectionContext::RendererMap renderers;
169 auto* start = m_renderRange.start();
170 RenderObject* stop = nullptr;
171 if (m_renderRange.end())
172 stop = rendererAfterOffset(*m_renderRange.end(), m_renderRange.endOffset());
173
174 RenderRangeIterator selectionIterator(start);
175 while (start && start != stop) {
176 if ((start->canBeSelectionLeaf() || start == m_renderRange.start() || start == m_renderRange.end())
177 && start->selectionState() != RenderObject::HighlightState::None) {
178 // Blocks are responsible for painting line gaps and margin gaps. They must be examined as well.
179 renderers.set(start, makeUnique<RenderSelectionInfo>(*start, clipToVisibleContent == ClipToVisibleContent::Yes));
180 LOG_WITH_STREAM(Selection, stream << " added start " << *start << " with rect " << renderers.get(start)->rect());
181
182 auto* block = start->containingBlock();
183 while (block && !is<RenderView>(*block)) {
184 LOG_WITH_STREAM(Scrolling, stream << " added block " << *block);
185 std::unique_ptr<RenderSelectionInfo>& blockInfo = renderers.add(block, nullptr).iterator->value;
186 if (blockInfo)
187 break;
188 blockInfo = makeUnique<RenderSelectionInfo>(*block, clipToVisibleContent == ClipToVisibleContent::Yes);
189 LOG_WITH_STREAM(Selection, stream << " added containing block " << *block << " with rect " << blockInfo->rect());
190 block = block->containingBlock();
191 }
192 }
193 start = selectionIterator.next();
194 }
195
196 // Now create a single bounding box rect that encloses the whole selection.
197 LayoutRect selectionRect;
198 for (auto& info : renderers.values()) {
199 // RenderSelectionInfo::rect() is in the coordinates of the repaintContainer, so map to page coordinates.
200 LayoutRect currentRect = info->rect();
201 if (currentRect.isEmpty())
202 continue;
203
204 if (auto* repaintContainer = info->repaintContainer()) {
205 FloatRect localRect = currentRect;
206 FloatQuad absQuad = repaintContainer->localToAbsoluteQuad(localRect);
207 currentRect = absQuad.enclosingBoundingBox();
208 LOG_WITH_STREAM(Selection, stream << " rect " << localRect << " mapped to " << currentRect << " in container " << *repaintContainer);
209 }
210 selectionRect.unite(currentRect);
211 }
212
213 LOG_WITH_STREAM(Selection, stream << " final rect " << selectionRect);
214 return snappedIntRect(selectionRect);
215}

217void SelectionRangeData::apply(const RenderRange& newSelection, RepaintMode blockRepaintMode)
218{
219 auto oldSelectionData = collectSelectionData(m_renderRange, blockRepaintMode == RepaintMode::NewXOROld);
220 // Remove current selection.
221 for (auto* renderer : oldSelectionData.renderers.keys())
222 renderer->setSelectionStateIfNeeded(RenderObject::HighlightState::None);
223 m_renderRange = newSelection;
224 auto* selectionStart = m_renderRange.start();
225 // Update the selection status of all objects between selectionStart and selectionEnd
226 if (selectionStart && selectionStart == m_renderRange.end())
227 selectionStart->setSelectionStateIfNeeded(RenderObject::HighlightState::Both);
228 else {
229 if (selectionStart)
230 selectionStart->setSelectionStateIfNeeded(RenderObject::HighlightState::Start);
231 if (auto* end = m_renderRange.end())
232 end->setSelectionStateIfNeeded(RenderObject::HighlightState::End);
233 }
234
235 RenderObject* selectionEnd = nullptr;
236 auto* selectionDataEnd = m_renderRange.end();
237 if (selectionDataEnd)
238 selectionEnd = rendererAfterOffset(*selectionDataEnd, m_renderRange.endOffset());
239 RenderRangeIterator selectionIterator(selectionStart);
240 for (auto* currentRenderer = selectionStart; currentRenderer && currentRenderer != selectionEnd; currentRenderer = selectionIterator.next()) {
241 if (currentRenderer == selectionStart || currentRenderer == m_renderRange.end())
242 continue;
243 if (!currentRenderer->canBeSelectionLeaf())
244 continue;
245 currentRenderer->setSelectionStateIfNeeded(RenderObject::HighlightState::Inside);
246 }
247
248 if (blockRepaintMode != RepaintMode::Nothing)
249 m_renderView.layer()->clearBlockSelectionGapsBounds();
250
251 // Now that the selection state has been updated for the new objects, walk them again and
252 // put them in the new objects list.
253 SelectionContext::RendererMap newSelectedRenderers;
254 SelectionContext::RenderBlockMap newSelectedBlocks;
255 selectionIterator = RenderRangeIterator(selectionStart);
256 for (auto* currentRenderer = selectionStart; currentRenderer && currentRenderer != selectionEnd; currentRenderer = selectionIterator.next()) {
257 if (isValidRendererForSelection(*currentRenderer, m_renderRange)) {
258 std::unique_ptr<RenderSelectionInfo> selectionInfo = makeUnique<RenderSelectionInfo>(*currentRenderer, true);
259#if ENABLE(SERVICE_CONTROLS)
260 for (auto& quad : selectionInfo->collectedSelectionQuads())
261 m_selectionGeometryGatherer.addQuad(selectionInfo->repaintContainer(), quad);
262 if (!currentRenderer->isTextOrLineBreak())
263 m_selectionGeometryGatherer.setTextOnly(false);
264#endif
265 newSelectedRenderers.set(currentRenderer, WTFMove(selectionInfo));
266 auto* containingBlock = currentRenderer->containingBlock();
267 while (containingBlock && !is<RenderView>(*containingBlock)) {
268 std::unique_ptr<RenderBlockSelectionInfo>& blockInfo = newSelectedBlocks.add(containingBlock, nullptr).iterator->value;
269 if (blockInfo)
270 break;
271 blockInfo = makeUnique<RenderBlockSelectionInfo>(*containingBlock);
272 containingBlock = containingBlock->containingBlock();
273#if ENABLE(SERVICE_CONTROLS)
274 m_selectionGeometryGatherer.addGapRects(blockInfo->repaintContainer(), blockInfo->rects());
275#endif
276 }
277 }
278 }
279
280 if (blockRepaintMode == RepaintMode::Nothing)
281 return;
282
283 // Have any of the old selected objects changed compared to the new selection?
284 for (auto& selectedRendererInfo : oldSelectionData.renderers) {
285 auto* renderer = selectedRendererInfo.key;
286 auto* newInfo = newSelectedRenderers.get(renderer);
287 auto* oldInfo = selectedRendererInfo.value.get();
288 if (!newInfo || oldInfo->rect() != newInfo->rect() || oldInfo->state() != newInfo->state()
289 || (m_renderRange.start() == renderer && oldSelectionData.startOffset != m_renderRange.startOffset())
290 || (m_renderRange.end() == renderer && oldSelectionData.endOffset != m_renderRange.endOffset())) {
291 oldInfo->repaint();
292 if (newInfo) {
293 newInfo->repaint();
294 newSelectedRenderers.remove(renderer);
295 }
296 }
297 }
298
299 // Any new objects that remain were not found in the old objects dict, and so they need to be updated.
300 for (auto& selectedRendererInfo : newSelectedRenderers)
301 selectedRendererInfo.value->repaint();
302
303 // Have any of the old blocks changed?
304 for (auto& selectedBlockInfo : oldSelectionData.blocks) {
305 auto* block = selectedBlockInfo.key;
306 auto* newInfo = newSelectedBlocks.get(block);
307 auto* oldInfo = selectedBlockInfo.value.get();
308 if (!newInfo || oldInfo->rects() != newInfo->rects() || oldInfo->state() != newInfo->state()) {
309 oldInfo->repaint();
310 if (newInfo) {
311 newInfo->repaint();
312 newSelectedBlocks.remove(block);
313 }
314 }
315 }
316
317 // Any new blocks that remain were not found in the old blocks dict, and so they need to be updated.
318 for (auto& selectedBlockInfo : newSelectedBlocks)
319 selectedBlockInfo.value->repaint();
320}
