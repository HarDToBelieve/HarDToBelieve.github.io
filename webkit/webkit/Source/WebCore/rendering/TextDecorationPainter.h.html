25#include "Color.h"

30namespace WebCore {
31
32class FilterOperations;
33class FontCascade;
34class FloatRect;
35class GraphicsContext;
36class LegacyInlineTextBox;
37class RenderObject;
38class RenderStyle;
39class RenderText;
40class ShadowData;
41class TextRun;
42
43class TextDecorationPainter {
44public:
45 struct Styles;
46 TextDecorationPainter(GraphicsContext&, OptionSet<TextDecoration> decorations, const RenderText&, bool isFirstLine, const FontCascade&, std::optional<Styles> = std::nullopt);
47
48 void setInlineTextBox(const LegacyInlineTextBox* inlineTextBox) { m_inlineTextBox = inlineTextBox; }
49 void setIsHorizontal(bool isHorizontal) { m_isHorizontal = isHorizontal; }
50 void setWidth(float width) { m_width = width; }
51 void setTextShadow(const ShadowData* textShadow) { m_shadow = textShadow; }
52 void setShadowColorFilter(const FilterOperations* colorFilter) { m_shadowColorFilter = colorFilter; }
53
54 void paintTextDecoration(const TextRun&, const FloatPoint& textOrigin, const FloatPoint& boxOrigin);
55
56 struct Styles {
57 bool operator==(const Styles&) const;
58 bool operator!=(const Styles& other) const { return !(*this == other); }
59
60 Color underlineColor;
61 Color overlineColor;
62 Color linethroughColor;
63 TextDecorationStyle underlineStyle;
64 TextDecorationStyle overlineStyle;
65 TextDecorationStyle linethroughStyle;
66 };
67 static Color decorationColor(const RenderStyle&);
68 static OptionSet<TextDecoration> textDecorationsInEffectForStyle(const Styles&);
69 static Styles stylesForRenderer(const RenderObject&, OptionSet<TextDecoration> requestedDecorations, bool firstLineStyle = false, PseudoId = PseudoId::None);
70
71private:
72 GraphicsContext& m_context;
73 OptionSet<TextDecoration> m_decorations;
74 float m_wavyOffset;
75 float m_width { 0 };
76 FloatPoint m_boxOrigin;
77 bool m_isPrinting;
78 bool m_isHorizontal { true };
79 const ShadowData* m_shadow { nullptr };
80 const FilterOperations* m_shadowColorFilter { nullptr };
81 const LegacyInlineTextBox* m_inlineTextBox { nullptr };
82 const FontCascade& m_font;
83
84 Styles m_styles;
85 const RenderStyle& m_lineStyle;
86};
