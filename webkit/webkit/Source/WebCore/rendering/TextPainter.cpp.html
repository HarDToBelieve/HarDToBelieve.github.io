40ShadowApplier::ShadowApplier(GraphicsContext& context, const ShadowData* shadow, const FilterOperations* colorFilter, const FloatRect& textRect, bool lastShadowIterationShouldDrawText, bool opaque, FontOrientation orientation)
41 : m_context { context }
42 , m_shadow { shadow }
43 , m_onlyDrawsShadow { !isLastShadowIteration() || !lastShadowIterationShouldDrawText }
44 , m_avoidDrawingShadow { shadowIsCompletelyCoveredByText(opaque) }
45 , m_nothingToDraw { shadow && m_avoidDrawingShadow && m_onlyDrawsShadow }
46 , m_didSaveContext { false }
47{
48 if (!shadow || m_nothingToDraw) {
49 m_shadow = nullptr;
50 return;
51 }
52
53 int shadowX = orientation == FontOrientation::Horizontal ? shadow->x() : shadow->y();
54 int shadowY = orientation == FontOrientation::Horizontal ? shadow->y() : -shadow->x();
55 FloatSize shadowOffset(shadowX, shadowY);
56 int shadowRadius = shadow->radius();
57 Color shadowColor = shadow->color();
58 if (colorFilter)
59 colorFilter->transformColor(shadowColor);
60
61 // When drawing shadows, we usually clip the context to the area the shadow will reside, and then
62 // draw the text itself outside the clipped area (so only the shadow shows up). However, we can
63 // often draw the *last* shadow and the text itself in a single call.
64 if (m_onlyDrawsShadow) {
65 FloatRect shadowRect(textRect);
66 shadowRect.inflate(shadow->paintingExtent() + 3 * textRect.height());
67 shadowRect.move(shadowOffset);
68 context.save();
69 context.clip(shadowRect);
70
71 m_didSaveContext = true;
72 m_extraOffset = FloatSize(0, 2 * shadowRect.height() + std::max(0.0f, shadowOffset.height()) + shadowRadius);
73 shadowOffset -= m_extraOffset;
74 }
75
76 if (!m_avoidDrawingShadow)
77 context.setShadow(shadowOffset, shadowRadius, shadowColor);
78}
79
80inline bool ShadowApplier::isLastShadowIteration()
81{
82 return m_shadow && !m_shadow->next();
83}
84
85inline bool ShadowApplier::shadowIsCompletelyCoveredByText(bool textIsOpaque)
86{
87 return textIsOpaque && m_shadow && m_shadow->location().isZero() && !m_shadow->radius();
88}
89
90ShadowApplier::~ShadowApplier()
91{
92 if (!m_shadow)
93 return;
94 if (m_onlyDrawsShadow)
95 m_context.restore();
96 else if (!m_avoidDrawingShadow)
97 m_context.clearShadow();
98}

105void TextPainter::paintTextOrEmphasisMarks(const FontCascade& font, const TextRun& textRun, const AtomString& emphasisMark,
106 float emphasisMarkOffset, const FloatPoint& textOrigin, unsigned startOffset, unsigned endOffset)
107{
108 ASSERT(startOffset < endOffset);
109
110 if (m_context.detectingContentfulPaint()) {
111 if (!textRun.text().isAllSpecialCharacters<isHTMLSpace>())
112 m_context.setContentfulPaintDetected();
113 return;
114 }
115
116 if (!emphasisMark.isEmpty())
117 m_context.drawEmphasisMarks(font, textRun, emphasisMark, textOrigin + FloatSize(0, emphasisMarkOffset), startOffset, endOffset);
118 else if (startOffset || endOffset < textRun.length() || !m_glyphDisplayList)
119 m_context.drawText(font, textRun, textOrigin, startOffset, endOffset);
120 else {
121 // Replaying back a whole cached glyph run to the GraphicsContext.
122 m_context.translate(textOrigin);
123 DisplayList::Replayer replayer(m_context, *m_glyphDisplayList);
124 replayer.replay();
125 m_context.translate(-textOrigin);
126 }
127 m_glyphDisplayList = nullptr;
128}

130void TextPainter::paintTextWithShadows(const ShadowData* shadow, const FilterOperations* colorFilter, const FontCascade& font, const TextRun& textRun, const FloatRect& boxRect, const FloatPoint& textOrigin,
131 unsigned startOffset, unsigned endOffset, const AtomString& emphasisMark, float emphasisMarkOffset, bool stroked)
132{
133 if (!shadow) {
134 paintTextOrEmphasisMarks(font, textRun, emphasisMark, emphasisMarkOffset, textOrigin, startOffset, endOffset);
135 return;
136 }
137
138 Color fillColor = m_context.fillColor();
139 bool opaque = fillColor.isOpaque();
140 bool lastShadowIterationShouldDrawText = !stroked && opaque;
141 if (!opaque)
142 m_context.setFillColor(Color::black);
143 while (shadow) {
144 ShadowApplier shadowApplier(m_context, shadow, colorFilter, boxRect, lastShadowIterationShouldDrawText, opaque, m_textBoxIsHorizontal ? FontOrientation::Horizontal : FontOrientation::Vertical);
145 if (!shadowApplier.nothingToDraw())
146 paintTextOrEmphasisMarks(font, textRun, emphasisMark, emphasisMarkOffset, textOrigin + shadowApplier.extraOffset(), startOffset, endOffset);
147 shadow = shadow->next();
148 }
149
150 if (!lastShadowIterationShouldDrawText) {
151 if (!opaque)
152 m_context.setFillColor(fillColor);
153 paintTextOrEmphasisMarks(font, textRun, emphasisMark, emphasisMarkOffset, textOrigin, startOffset, endOffset);
154 }
155}

157void TextPainter::paintTextAndEmphasisMarksIfNeeded(const TextRun& textRun, const FloatRect& boxRect, const FloatPoint& textOrigin, unsigned startOffset, unsigned endOffset,
158 const TextPaintStyle& paintStyle, const ShadowData* shadow, const FilterOperations* shadowColorFilter)
159{
160 if (paintStyle.paintOrder == PaintOrder::Normal) {
161 // FIXME: Truncate right-to-left text correctly.
162 paintTextWithShadows(shadow, shadowColorFilter, *m_font, textRun, boxRect, textOrigin, startOffset, endOffset, nullAtom(), 0, paintStyle.strokeWidth > 0);
163 } else {
164 auto textDrawingMode = m_context.textDrawingMode();
165 auto paintOrder = RenderStyle::paintTypesForPaintOrder(paintStyle.paintOrder);
166 auto shadowToUse = shadow;
167
168 for (auto order : paintOrder) {
169 switch (order) {
170 case PaintType::Fill: {
171 auto textDrawingModeWithoutStroke = textDrawingMode;
172 textDrawingModeWithoutStroke.remove(TextDrawingMode::Stroke);
173 m_context.setTextDrawingMode(textDrawingModeWithoutStroke);
174 paintTextWithShadows(shadowToUse, shadowColorFilter, *m_font, textRun, boxRect, textOrigin, startOffset, endOffset, nullAtom(), 0, false);
175 shadowToUse = nullptr;
176 m_context.setTextDrawingMode(textDrawingMode);
177 break;
178 }
179 case PaintType::Stroke: {
180 auto textDrawingModeWithoutFill = textDrawingMode;
181 textDrawingModeWithoutFill.remove(TextDrawingMode::Fill);
182 m_context.setTextDrawingMode(textDrawingModeWithoutFill);
183 paintTextWithShadows(shadowToUse, shadowColorFilter, *m_font, textRun, boxRect, textOrigin, startOffset, endOffset, nullAtom(), 0, paintStyle.strokeWidth > 0);
184 shadowToUse = nullptr;
185 m_context.setTextDrawingMode(textDrawingMode);
186 }
187 break;
188 case PaintType::Markers:
189 continue;
190 }
191 }
192 }
193
194 if (m_emphasisMark.isEmpty())
195 return;
196
197 FloatPoint boxOrigin = boxRect.location();
198 updateGraphicsContext(m_context, paintStyle, UseEmphasisMarkColor);
199 static NeverDestroyed<TextRun> objectReplacementCharacterTextRun(StringView(&objectReplacementCharacter, 1));
200 const TextRun& emphasisMarkTextRun = m_combinedText ? objectReplacementCharacterTextRun.get() : textRun;
201 FloatPoint emphasisMarkTextOrigin = m_combinedText ? FloatPoint(boxOrigin.x() + boxRect.width() / 2, boxOrigin.y() + m_font->fontMetrics().ascent()) : textOrigin;
202 if (m_combinedText)
203 m_context.concatCTM(rotation(boxRect, Clockwise));
204
205 // FIXME: Truncate right-to-left text correctly.
206 paintTextWithShadows(shadow, shadowColorFilter, m_combinedText ? m_combinedText->originalFont() : *m_font, emphasisMarkTextRun, boxRect, emphasisMarkTextOrigin, startOffset, endOffset,
207 m_emphasisMark, m_emphasisMarkOffset, paintStyle.strokeWidth > 0);
208
209 if (m_combinedText)
210 m_context.concatCTM(rotation(boxRect, Counterclockwise));
211}

213void TextPainter::paint(const TextRun& textRun, const FloatRect& boxRect, const FloatPoint& textOrigin)
214{
215 paintRange(textRun, boxRect, textOrigin, 0, textRun.length());
216}

218void TextPainter::paintRange(const TextRun& textRun, const FloatRect& boxRect, const FloatPoint& textOrigin, unsigned start, unsigned end)
219{
220 ASSERT(m_font);
221 ASSERT(start < end);
222
223 GraphicsContextStateSaver stateSaver(m_context, m_style.strokeWidth > 0);
224 updateGraphicsContext(m_context, m_style);
225 paintTextAndEmphasisMarksIfNeeded(textRun, boxRect, textOrigin, start, end, m_style, m_shadow, m_shadowColorFilter);
226}

231#if ENABLE(LAYOUT_FORMATTING_CONTEXT)
232 if (RuntimeEnabledFeatures::sharedFeatures().layoutFormattingContextIntegrationEnabled())
233 GlyphDisplayListCache<LayoutIntegration::Run>::singleton().clear();
234#endif
235}

237bool TextPainter::shouldUseGlyphDisplayList(const PaintInfo& paintInfo)
238{
239 return !paintInfo.context().paintingDisabled() && paintInfo.enclosingSelfPaintingLayer() && paintInfo.enclosingSelfPaintingLayer()->paintingFrequently();
240}
