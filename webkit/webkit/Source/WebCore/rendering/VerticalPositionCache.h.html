26#pragma once
27
28#include "FontBaseline.h"
29#include <wtf/HashMap.h>
30
31namespace WebCore {
32
33class RenderObject;
34
35class VerticalPositionCache {
36 WTF_MAKE_NONCOPYABLE(VerticalPositionCache);
37public:
38 VerticalPositionCache()
39 { }
40
41 bool get(RenderObject* renderer, FontBaseline baselineType, LayoutUnit& result) const
42 {
43 const HashMap<RenderObject*, LayoutUnit>& mapToCheck = baselineType == AlphabeticBaseline ? m_alphabeticPositions : m_ideographicPositions;
44 const HashMap<RenderObject*, LayoutUnit>::const_iterator it = mapToCheck.find(renderer);
45 if (it == mapToCheck.end())
46 return false;
47
48 result = it->value;
49 return true;
50 }
51
52 void set(RenderObject* renderer, FontBaseline baselineType, LayoutUnit position)
53 {
54 if (baselineType == AlphabeticBaseline)
55 m_alphabeticPositions.set(renderer, position);
56 else
57 m_ideographicPositions.set(renderer, position);
58 }
59
60private:
61 HashMap<RenderObject*, LayoutUnit> m_alphabeticPositions;
62 HashMap<RenderObject*, LayoutUnit> m_ideographicPositions;
63};
64
65} // namespace WebCore
