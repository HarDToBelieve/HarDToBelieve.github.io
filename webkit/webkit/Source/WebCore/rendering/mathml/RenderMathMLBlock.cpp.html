73static LayoutUnit axisHeight(const RenderStyle& style)
74{
75 // If we have a MATH table we just return the AxisHeight constant.
76 const auto& primaryFont = style.fontCascade().primaryFont();
77 if (auto* mathData = primaryFont.mathData())
78 return LayoutUnit(mathData->getMathConstant(primaryFont, OpenTypeMathData::AxisHeight));
79
80 // Otherwise, the idea is to try and use the middle of operators as the math axis which we thus approximate by "half of the x-height".
81 // Note that Gecko has a slower but more accurate version that measures half of the height of U+2212 MINUS SIGN.
82 return LayoutUnit(style.fontMetrics().xHeight() / 2);
83}

85LayoutUnit RenderMathMLBlock::mathAxisHeight() const
86{

98LayoutUnit RenderMathMLBlock::baselinePosition(FontBaseline baselineType, bool firstLine, LineDirectionMode direction, LinePositionMode linePositionMode) const
99{
100 // mathml.css sets math { -webkit-line-box-contain: glyphs replaced; line-height: 0; }, so when linePositionMode == PositionOfInteriorLineBoxes we want to
101 // return 0 here to match our line-height. This matters when LegacyRootInlineBox::ascentAndDescentForBox is called on a RootInlineBox for an inline-block.
102 if (linePositionMode == PositionOfInteriorLineBoxes)
103 return 0;
104
105 return firstLineBaseline().value_or(RenderBlock::baselinePosition(baselineType, firstLine, direction, linePositionMode));
106}

109void RenderMathMLBlock::paint(PaintInfo& info, const LayoutPoint& paintOffset)
110{
111 RenderBlock::paint(info, paintOffset);
112
113 if (info.context().paintingDisabled() || info.phase != PaintPhase::Foreground)
114 return;
115
116 IntPoint adjustedPaintOffset = roundedIntPoint(paintOffset + location());
117
118 GraphicsContextStateSaver stateSaver(info.context());
119
120 info.context().setStrokeThickness(1.0f);
121 info.context().setStrokeStyle(SolidStroke);
122 info.context().setStrokeColor(Color::blue);
123
124 info.context().drawLine(adjustedPaintOffset, IntPoint(adjustedPaintOffset.x() + pixelSnappedOffsetWidth(), adjustedPaintOffset.y()));
125 info.context().drawLine(IntPoint(adjustedPaintOffset.x() + pixelSnappedOffsetWidth(), adjustedPaintOffset.y()), IntPoint(adjustedPaintOffset.x() + pixelSnappedOffsetWidth(), adjustedPaintOffset.y() + pixelSnappedOffsetHeight()));
126 info.context().drawLine(IntPoint(adjustedPaintOffset.x(), adjustedPaintOffset.y() + pixelSnappedOffsetHeight()), IntPoint(adjustedPaintOffset.x() + pixelSnappedOffsetWidth(), adjustedPaintOffset.y() + pixelSnappedOffsetHeight()));
127 info.context().drawLine(adjustedPaintOffset, IntPoint(adjustedPaintOffset.x(), adjustedPaintOffset.y() + pixelSnappedOffsetHeight()));
128
129 int topStart = paddingTop();
130
131 info.context().setStrokeColor(Color::green);
132
133 info.context().drawLine(IntPoint(adjustedPaintOffset.x(), adjustedPaintOffset.y() + topStart), IntPoint(adjustedPaintOffset.x() + pixelSnappedOffsetWidth(), adjustedPaintOffset.y() + topStart));
134
135 int baseline = roundToInt(baselinePosition(AlphabeticBaseline, true, HorizontalLine));
136
137 info.context().setStrokeColor(Color::red);
138
139 info.context().drawLine(IntPoint(adjustedPaintOffset.x(), adjustedPaintOffset.y() + baseline), IntPoint(adjustedPaintOffset.x() + pixelSnappedOffsetWidth(), adjustedPaintOffset.y() + baseline));
140}

143LayoutUnit toUserUnits(const MathMLElement::Length& length, const RenderStyle& style, const LayoutUnit& referenceValue)
144{
145 switch (length.type) {
146 // Zoom for physical units needs to be accounted for.
147 case MathMLElement::LengthType::Cm:
148 return LayoutUnit(style.effectiveZoom() * length.value * cssPixelsPerInch / 2.54f);
149 case MathMLElement::LengthType::In:
150 return LayoutUnit(style.effectiveZoom() * length.value * cssPixelsPerInch);
151 case MathMLElement::LengthType::Mm:
152 return LayoutUnit(style.effectiveZoom() * length.value * cssPixelsPerInch / 25.4f);
153 case MathMLElement::LengthType::Pc:
154 return LayoutUnit(style.effectiveZoom() * length.value * cssPixelsPerInch / 6);
155 case MathMLElement::LengthType::Pt:
156 return LayoutUnit(style.effectiveZoom() * length.value * cssPixelsPerInch / 72);
157 case MathMLElement::LengthType::Px:
158 return LayoutUnit(style.effectiveZoom() * length.value);
159
160 // Zoom for logical units is accounted for either in the font info or referenceValue.
161 case MathMLElement::LengthType::Em:
162 return LayoutUnit(length.value * style.fontCascade().size());
163 case MathMLElement::LengthType::Ex:
164 return LayoutUnit(length.value * style.fontMetrics().xHeight());
165 case MathMLElement::LengthType::MathUnit:
166 return LayoutUnit(length.value * style.fontCascade().size() / 18);
167 case MathMLElement::LengthType::Percentage:
168 return LayoutUnit(referenceValue * length.value / 100);
169 case MathMLElement::LengthType::UnitLess:
170 return LayoutUnit(referenceValue * length.value);
171 case MathMLElement::LengthType::ParsingFailed:
172 return referenceValue;
173 default:
174 ASSERT_NOT_REACHED();
175 return referenceValue;
176 }
177}

179std::optional<LayoutUnit> RenderMathMLTable::firstLineBaseline() const
180{
181 // By default the vertical center of <mtable> is aligned on the math axis.
182 // This is different than RenderTable::firstLineBoxBaseline, which returns the baseline of the first row of a <table>.
183 return LayoutUnit { (logicalHeight() / 2 + axisHeight(style())).toInt() };
184}

186void RenderMathMLBlock::layoutItems(bool relayoutChildren)
187{
188 LayoutUnit verticalOffset = borderBefore() + paddingBefore();
189 LayoutUnit horizontalOffset = borderStart() + paddingStart();
190
191 LayoutUnit preferredHorizontalExtent;
192 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
193 LayoutUnit childHorizontalExtent = child->maxPreferredLogicalWidth() - child->horizontalBorderAndPaddingExtent();
194 LayoutUnit childHorizontalMarginBoxExtent = child->horizontalBorderAndPaddingExtent() + childHorizontalExtent;
195 childHorizontalMarginBoxExtent += child->horizontalMarginExtent();
196
197 preferredHorizontalExtent += childHorizontalMarginBoxExtent;
198 }
199
200 LayoutUnit currentHorizontalExtent = contentLogicalWidth();
201 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
202 LayoutUnit childSize = child->maxPreferredLogicalWidth() - child->horizontalBorderAndPaddingExtent();
203
204 if (preferredHorizontalExtent > currentHorizontalExtent)
205 childSize = currentHorizontalExtent;
206
207 LayoutUnit childPreferredSize = childSize + child->horizontalBorderAndPaddingExtent();
208
209 if (childPreferredSize != child->width())
210 child->setChildNeedsLayout(MarkOnlyThis);
211
212 updateBlockChildDirtyBitsBeforeLayout(relayoutChildren, *child);
213 child->layoutIfNeeded();
214
215 LayoutUnit childVerticalMarginBoxExtent;
216 childVerticalMarginBoxExtent = child->height() + child->verticalMarginExtent();
217
218 setLogicalHeight(std::max(logicalHeight(), verticalOffset + borderAfter() + paddingAfter() + childVerticalMarginBoxExtent + horizontalScrollbarHeight()));
219
220 horizontalOffset += child->marginStart();
221
222 LayoutUnit childHorizontalExtent = child->width();
223 LayoutPoint childLocation(style().isLeftToRightDirection() ? horizontalOffset : width() - horizontalOffset - childHorizontalExtent,
224 verticalOffset + child->marginBefore());
225
226 child->setLocation(childLocation);
227 horizontalOffset += childHorizontalExtent + child->marginEnd();
228 }
229}

231void RenderMathMLBlock::layoutBlock(bool relayoutChildren, LayoutUnit)
232{
233 ASSERT(needsLayout());
234
235 if (!relayoutChildren && simplifiedLayout())
236 return;
237
238 LayoutRepainter repainter(*this, checkForRepaintDuringLayout());
239
240 if (recomputeLogicalWidth())
241 relayoutChildren = true;
242
243 setLogicalHeight(borderAndPaddingLogicalHeight() + scrollbarLogicalHeight());
244
245 layoutItems(relayoutChildren);
246
247 updateLogicalHeight();
248
249 layoutPositionedObjects(relayoutChildren);
250
251 repainter.repaintAfterLayout();
252
253 updateScrollInfoAfterLayout();
254
255 clearNeedsLayout();
256}

258void RenderMathMLBlock::layoutInvalidMarkup(bool relayoutChildren)
259{
260 // Invalid MathML subtrees are just renderered as empty boxes.
261 // FIXME: https://webkit.org/b/135460 - Should we display some "invalid" markup message instead?
262 ASSERT(needsLayout());
263 for (auto child = firstChildBox(); child; child = child->nextSiblingBox())
264 child->layoutIfNeeded();
265 setLogicalWidth(0);
266 setLogicalHeight(0);
267 layoutPositionedObjects(relayoutChildren);
268 updateScrollInfoAfterLayout();
269 clearNeedsLayout();
270}
