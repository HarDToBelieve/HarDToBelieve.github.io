55std::optional<LayoutUnit> RenderMathMLRow::firstLineBaseline() const
56{
57 auto* baselineChild = firstChildBox();
58 if (!baselineChild)
59 return std::optional<LayoutUnit>();
60
61 return LayoutUnit { static_cast<int>(lroundf(ascentForChild(*baselineChild) + baselineChild->logicalTop())) };
62}

64static RenderMathMLOperator* toVerticalStretchyOperator(RenderBox* box)
65{
66 if (is<RenderMathMLBlock>(box)) {
67 auto* renderOperator = downcast<RenderMathMLBlock>(*box).unembellishedOperator();
68 if (renderOperator && renderOperator->isStretchy() && renderOperator->isVertical())
69 return renderOperator;
70 }
71 return nullptr;
72}

74void RenderMathMLRow::stretchVerticalOperatorsAndLayoutChildren()
75{
76 // First calculate stretch ascent and descent.
77 LayoutUnit stretchAscent, stretchDescent;
78 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
79 if (child->isOutOfFlowPositioned()) {
80 child->containingBlock()->insertPositionedObject(*child);
81 continue;
82 }
83 if (toVerticalStretchyOperator(child))
84 continue;
85 child->layoutIfNeeded();
86 LayoutUnit childAscent = ascentForChild(*child);
87 LayoutUnit childDescent = child->logicalHeight() - childAscent;
88 stretchAscent = std::max(stretchAscent, childAscent);
89 stretchDescent = std::max(stretchDescent, childDescent);
90 }
91 if (stretchAscent + stretchDescent <= 0) {
92 // We ensure a minimal stretch size.
93 stretchAscent = style().computedFontPixelSize();
94 stretchDescent = 0;
95 }
96
97 // Next, we stretch the vertical operators.
98 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
99 if (child->isOutOfFlowPositioned())
100 continue;
101 if (auto renderOperator = toVerticalStretchyOperator(child)) {
102 renderOperator->stretchTo(stretchAscent, stretchDescent);
103 renderOperator->layoutIfNeeded();
104 child->layoutIfNeeded();
105 }
106 }
107}

109void RenderMathMLRow::getContentBoundingBox(LayoutUnit& width, LayoutUnit& ascent, LayoutUnit& descent) const
110{
111 ascent = 0;
112 descent = 0;
113 width = borderAndPaddingStart();
114 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
115 if (child->isOutOfFlowPositioned())
116 continue;
117
118 width += child->marginStart() + child->logicalWidth() + child->marginEnd();
119 LayoutUnit childAscent = ascentForChild(*child);
120 LayoutUnit childDescent = child->logicalHeight() - childAscent;
121 ascent = std::max(ascent, childAscent + child->marginTop());
122 descent = std::max(descent, childDescent + child->marginBottom());
123 }
124 width += borderEnd() + paddingEnd();
125}

127void RenderMathMLRow::computePreferredLogicalWidths()
128{
129 ASSERT(preferredLogicalWidthsDirty());
130
131 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = 0;
132
133 LayoutUnit preferredWidth;
134 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
135 if (child->isOutOfFlowPositioned())
136 continue;
137 preferredWidth += child->maxPreferredLogicalWidth() + child->marginLogicalWidth();
138 }
139
140 m_minPreferredLogicalWidth = m_maxPreferredLogicalWidth = preferredWidth + borderAndPaddingLogicalWidth();
141
142 setPreferredLogicalWidthsDirty(false);
143}

145void RenderMathMLRow::layoutRowItems(LayoutUnit width, LayoutUnit ascent)
146{
147 LayoutUnit horizontalOffset = borderAndPaddingStart();
148 for (auto* child = firstChildBox(); child; child = child->nextSiblingBox()) {
149 if (child->isOutOfFlowPositioned())
150 continue;
151 horizontalOffset += child->marginStart();
152 LayoutUnit childAscent = ascentForChild(*child);
153 LayoutUnit childVerticalOffset = borderTop() + paddingTop() + child->marginTop() + ascent - childAscent;
154 LayoutUnit childWidth = child->logicalWidth();
155 LayoutUnit childHorizontalOffset = style().isLeftToRightDirection() ? horizontalOffset : width - horizontalOffset - childWidth;
156 child->setLocation(LayoutPoint(childHorizontalOffset, childVerticalOffset));
157 horizontalOffset += childWidth + child->marginEnd();
158 }
159}

161void RenderMathMLRow::layoutBlock(bool relayoutChildren, LayoutUnit)
162{
163 ASSERT(needsLayout());
164
165 if (!relayoutChildren && simplifiedLayout())
166 return;
167
168 recomputeLogicalWidth();
169
170 setLogicalHeight(borderAndPaddingLogicalHeight() + scrollbarLogicalHeight());
171
172 LayoutUnit width, ascent, descent;
173 stretchVerticalOperatorsAndLayoutChildren();
174 getContentBoundingBox(width, ascent, descent);
175 layoutRowItems(width, ascent);
176 setLogicalWidth(width);
177 setLogicalHeight(borderTop() + paddingTop() + ascent + descent + borderBottom() + paddingBottom() + horizontalScrollbarHeight());
178 updateLogicalHeight();
179
180 layoutPositionedObjects(relayoutChildren);
181
182 updateScrollInfoAfterLayout();
183
184 clearNeedsLayout();
185}
