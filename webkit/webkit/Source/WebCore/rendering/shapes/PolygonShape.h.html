38class OffsetPolygonEdge final : public VertexPair {
39public:
40 OffsetPolygonEdge(const FloatPolygonEdge& edge, const FloatSize& offset)
41 : m_vertex1(edge.vertex1() + offset)
42 , m_vertex2(edge.vertex2() + offset)
43 {
44 }
45
46 const FloatPoint& vertex1() const override { return m_vertex1; }
47 const FloatPoint& vertex2() const override { return m_vertex2; }
48
49 bool isWithinYRange(float y1, float y2) const { return y1 <= minY() && y2 >= maxY(); }
50 bool overlapsYRange(float y1, float y2) const { return y2 >= minY() && y1 <= maxY(); }
51 float xIntercept(float y) const;
52 FloatShapeInterval clippedEdgeXRange(float y1, float y2) const;
53
54private:
55 FloatPoint m_vertex1;
56 FloatPoint m_vertex2;
57};
58
59class PolygonShape : public Shape {
60 WTF_MAKE_NONCOPYABLE(PolygonShape);
61public:
62 PolygonShape(Vector<FloatPoint>&& vertices, WindRule fillRule)
63 : m_polygon(WTFMove(vertices), fillRule)
64 {
65 }
66
67 LayoutRect shapeMarginLogicalBoundingBox() const override;
68 bool isEmpty() const override { return m_polygon.isEmpty(); }
69 LineSegment getExcludedInterval(LayoutUnit logicalTop, LayoutUnit logicalHeight) const override;
70
71 void buildDisplayPaths(DisplayPaths&) const override;
72
73private:
74 FloatPolygon m_polygon;
75};
