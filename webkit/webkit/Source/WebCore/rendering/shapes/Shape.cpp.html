47static std::unique_ptr<Shape> createInsetShape(const FloatRoundedRect& bounds)
48{
49 ASSERT(bounds.rect().width() >= 0 && bounds.rect().height() >= 0);
50 return makeUnique<BoxShape>(bounds);
51}

53static std::unique_ptr<Shape> createCircleShape(const FloatPoint& center, float radius)
54{
55 ASSERT(radius >= 0);
56 return makeUnique<RectangleShape>(FloatRect(center.x() - radius, center.y() - radius, radius*2, radius*2), FloatSize(radius, radius));
57}
58
59static std::unique_ptr<Shape> createEllipseShape(const FloatPoint& center, const FloatSize& radii)
60{
61 ASSERT(radii.width() >= 0 && radii.height() >= 0);
62 return makeUnique<RectangleShape>(FloatRect(center.x() - radii.width(), center.y() - radii.height(), radii.width()*2, radii.height()*2), radii);
63}
64
65static std::unique_ptr<Shape> createPolygonShape(Vector<FloatPoint>&& vertices, WindRule fillRule)
66{
67 return makeUnique<PolygonShape>(WTFMove(vertices), fillRule);
68}

70static inline FloatRect physicalRectToLogical(const FloatRect& rect, float logicalBoxHeight, WritingMode writingMode)
71{
72 if (isHorizontalWritingMode(writingMode))
73 return rect;
74 if (isFlippedWritingMode(writingMode))
75 return FloatRect(rect.y(), logicalBoxHeight - rect.maxX(), rect.height(), rect.width());
76 return rect.transposedRect();
77}
78
79static inline FloatPoint physicalPointToLogical(const FloatPoint& point, float logicalBoxHeight, WritingMode writingMode)
80{
81 if (isHorizontalWritingMode(writingMode))
82 return point;
83 if (isFlippedWritingMode(writingMode))
84 return FloatPoint(point.y(), logicalBoxHeight - point.x());
85 return point.transposedPoint();
86}
87
88static inline FloatSize physicalSizeToLogical(const FloatSize& size, WritingMode writingMode)
89{
90 if (isHorizontalWritingMode(writingMode))
91 return size;
92 return size.transposedSize();
93}

95std::unique_ptr<Shape> Shape::createShape(const BasicShape& basicShape, const LayoutSize& logicalBoxSize, WritingMode writingMode, float margin)
96{
97 bool horizontalWritingMode = isHorizontalWritingMode(writingMode);
98 float boxWidth = horizontalWritingMode ? logicalBoxSize.width() : logicalBoxSize.height();
99 float boxHeight = horizontalWritingMode ? logicalBoxSize.height() : logicalBoxSize.width();
100 std::unique_ptr<Shape> shape;
101
102 switch (basicShape.type()) {
103
104 case BasicShape::Type::Circle: {
105 const auto& circle = downcast<BasicShapeCircle>(basicShape);
106 float centerX = floatValueForCenterCoordinate(circle.centerX(), boxWidth);
107 float centerY = floatValueForCenterCoordinate(circle.centerY(), boxHeight);
108 float radius = circle.floatValueForRadiusInBox(boxWidth, boxHeight);
109 FloatPoint logicalCenter = physicalPointToLogical(FloatPoint(centerX, centerY), logicalBoxSize.height(), writingMode);
110
111 shape = createCircleShape(logicalCenter, radius);
112 break;
113 }
114
115 case BasicShape::Type::Ellipse: {
116 const auto& ellipse = downcast<BasicShapeEllipse>(basicShape);
117 float centerX = floatValueForCenterCoordinate(ellipse.centerX(), boxWidth);
118 float centerY = floatValueForCenterCoordinate(ellipse.centerY(), boxHeight);
119 float radiusX = ellipse.floatValueForRadiusInBox(ellipse.radiusX(), centerX, boxWidth);
120 float radiusY = ellipse.floatValueForRadiusInBox(ellipse.radiusY(), centerY, boxHeight);
121 FloatPoint logicalCenter = physicalPointToLogical(FloatPoint(centerX, centerY), logicalBoxSize.height(), writingMode);
122
123 shape = createEllipseShape(logicalCenter, FloatSize(radiusX, radiusY));
124 break;
125 }
126
127 case BasicShape::Type::Polygon: {
128 const auto& polygon = downcast<BasicShapePolygon>(basicShape);
129 const Vector<Length>& values = polygon.values();
130 size_t valuesSize = values.size();
131 ASSERT(!(valuesSize % 2));
132 Vector<FloatPoint> vertices(valuesSize / 2);
133 for (unsigned i = 0; i < valuesSize; i += 2) {
134 FloatPoint vertex(
135 floatValueForLength(values.at(i), boxWidth),
136 floatValueForLength(values.at(i + 1), boxHeight));
137 vertices[i / 2] = physicalPointToLogical(vertex, logicalBoxSize.height(), writingMode);
138 }
139 shape = createPolygonShape(WTFMove(vertices), polygon.windRule());
140 break;
141 }
142
143 case BasicShape::Type::Inset: {
144 const auto& inset = downcast<BasicShapeInset>(basicShape);
145 float left = floatValueForLength(inset.left(), boxWidth);
146 float top = floatValueForLength(inset.top(), boxHeight);
147 FloatRect rect(left,
148 top,
149 std::max<float>(boxWidth - left - floatValueForLength(inset.right(), boxWidth), 0),
150 std::max<float>(boxHeight - top - floatValueForLength(inset.bottom(), boxHeight), 0));
151 FloatRect logicalRect = physicalRectToLogical(rect, logicalBoxSize.height(), writingMode);
152
153 FloatSize boxSize(boxWidth, boxHeight);
154 FloatSize topLeftRadius = physicalSizeToLogical(floatSizeForLengthSize(inset.topLeftRadius(), boxSize), writingMode);
155 FloatSize topRightRadius = physicalSizeToLogical(floatSizeForLengthSize(inset.topRightRadius(), boxSize), writingMode);
156 FloatSize bottomLeftRadius = physicalSizeToLogical(floatSizeForLengthSize(inset.bottomLeftRadius(), boxSize), writingMode);
157 FloatSize bottomRightRadius = physicalSizeToLogical(floatSizeForLengthSize(inset.bottomRightRadius(), boxSize), writingMode);
158 FloatRoundedRect::Radii cornerRadii(topLeftRadius, topRightRadius, bottomLeftRadius, bottomRightRadius);
159
160 cornerRadii.scale(calcBorderRadiiConstraintScaleFor(logicalRect, cornerRadii));
161
162 shape = createInsetShape(FloatRoundedRect(logicalRect, cornerRadii));
163 break;
164 }
165
166 default:
167 ASSERT_NOT_REACHED();
168 }
169
170 shape->m_writingMode = writingMode;
171 shape->m_margin = margin;
172
173 return shape;
174}

176std::unique_ptr<Shape> Shape::createRasterShape(Image* image, float threshold, const LayoutRect& imageR, const LayoutRect& marginR, WritingMode writingMode, float margin)
177{
178 ASSERT(marginR.height() >= 0);
179
180 IntRect imageRect = snappedIntRect(imageR);
181 IntRect marginRect = snappedIntRect(marginR);
182 auto intervals = makeUnique<RasterShapeIntervals>(marginRect.height(), -marginRect.y());
183 // FIXME (149420): This buffer should not be unconditionally unaccelerated.
184 auto imageBuffer = ImageBuffer::create(imageRect.size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
185
186 auto createShape = [&]() {
187 auto rasterShape = makeUnique<RasterShape>(WTFMove(intervals), marginRect.size());
188 rasterShape->m_writingMode = writingMode;
189 rasterShape->m_margin = margin;
190 return rasterShape;
191 };
192
193 if (!imageBuffer)
194 return createShape();
195
196 GraphicsContext& graphicsContext = imageBuffer->context();
197 if (image)
198 graphicsContext.drawImage(*image, IntRect(IntPoint(), imageRect.size()));
199
200 PixelBufferFormat format { AlphaPremultiplication::Unpremultiplied, PixelFormat::RGBA8, DestinationColorSpace::SRGB() };
201 auto pixelBuffer = imageBuffer->getPixelBuffer(format, { IntPoint(), imageRect.size() });
202
203 // We could get to a value where PixelBuffer could be nullopt. A case where ImageRect.size() is huge, PixelBuffer::tryCreate
204 // can return a nullopt because data size has overflowed. Refer rdar://problem/61793884
205 if (!pixelBuffer)
206 return createShape();
207
208 auto& pixelArray = pixelBuffer->data();
209 unsigned pixelArrayLength = pixelArray.length();
210 unsigned pixelArrayOffset = 3; // Each pixel is four bytes: RGBA.
211 uint8_t alphaPixelThreshold = static_cast<uint8_t>(lroundf(clampTo<float>(threshold, 0, 1) * 255.0f));
212
213 int minBufferY = std::max(0, marginRect.y() - imageRect.y());
214 int maxBufferY = std::min(imageRect.height(), marginRect.maxY() - imageRect.y());
215
216 if ((imageRect.area() * 4) == pixelArrayLength) {
217 for (int y = minBufferY; y < maxBufferY; ++y) {
218 int startX = -1;
219 for (int x = 0; x < imageRect.width(); ++x, pixelArrayOffset += 4) {
220 uint8_t alpha = pixelArray.item(pixelArrayOffset);
221 bool alphaAboveThreshold = alpha > alphaPixelThreshold;
222 if (startX == -1 && alphaAboveThreshold) {
223 startX = x;
224 } else if (startX != -1 && (!alphaAboveThreshold || x == imageRect.width() - 1)) {
225 // We're creating "end-point exclusive" intervals here. The value of an interval's x1 is
226 // the first index of an above-threshold pixel for y, and the value of x2 is 1+ the index
227 // of the last above-threshold pixel.
228 int endX = alphaAboveThreshold ? x + 1 : x;
229 intervals->intervalAt(y + imageRect.y()).unite(IntShapeInterval(startX + imageRect.x(), endX + imageRect.x()));
230 startX = -1;
231 }
232 }
233 }
234 }
235
236 return createShape();
237}

239std::unique_ptr<Shape> Shape::createBoxShape(const RoundedRect& roundedRect, WritingMode writingMode, float margin)
240{
241 ASSERT(roundedRect.rect().width() >= 0 && roundedRect.rect().height() >= 0);
242
243 FloatRect rect(0, 0, roundedRect.rect().width(), roundedRect.rect().height());
244 FloatRoundedRect bounds(rect, roundedRect.radii());
245 auto shape = makeUnique<BoxShape>(bounds);
246 shape->m_writingMode = writingMode;
247 shape->m_margin = margin;
248
249 return shape;
250}
