50void BasicShapeCenterCoordinate::updateComputedLength()
51{
52 if (m_direction == TopLeft) {
53 m_computedLength = m_length.isUndefined() ? Length(0, LengthType::Fixed) : m_length;
54 return;
55 }
56
57 if (m_length.isUndefined()) {
58 m_computedLength = Length(100, LengthType::Percent);
59 return;
60 }
61
62 m_computedLength = convertTo100PercentMinusLength(m_length);
63}

65struct SVGPathTransformedByteStream {
66 bool operator==(const SVGPathTransformedByteStream& other) const
67 {
68 return other.offset == offset && other.zoom == zoom && other.rawStream == rawStream;
69 }
70 bool operator!=(const SVGPathTransformedByteStream& other) const { return !(*this == other); }
71 bool isEmpty() const { return rawStream.isEmpty(); }
72
73 Path path() const
74 {
75 Path path = buildPathFromByteStream(rawStream);
76 if (zoom != 1)
77 path.transform(AffineTransform().scale(zoom));
78 path.translate(toFloatSize(offset));
79 return path;
80 }
81
82 SVGPathByteStream rawStream;
83 float zoom;
84 FloatPoint offset;
85};

159float BasicShapeCircle::floatValueForRadiusInBox(float boxWidth, float boxHeight) const
160{
161 if (m_radius.type() == BasicShapeRadius::Value)
162 return floatValueForLength(m_radius.value(), std::hypot(boxWidth, boxHeight) / sqrtOfTwoFloat);
163
164 float centerX = floatValueForCenterCoordinate(m_centerX, boxWidth);
165 float centerY = floatValueForCenterCoordinate(m_centerY, boxHeight);
166
167 float widthDelta = std::abs(boxWidth - centerX);
168 float heightDelta = std::abs(boxHeight - centerY);
169 if (m_radius.type() == BasicShapeRadius::ClosestSide)
170 return std::min(std::min(std::abs(centerX), widthDelta), std::min(std::abs(centerY), heightDelta));
171
172 // If radius.type() == BasicShapeRadius::FarthestSide.
173 return std::max(std::max(std::abs(centerX), widthDelta), std::max(std::abs(centerY), heightDelta));
174}

176const Path& BasicShapeCircle::path(const FloatRect& boundingBox)
177{
178 float centerX = floatValueForCenterCoordinate(m_centerX, boundingBox.width());
179 float centerY = floatValueForCenterCoordinate(m_centerY, boundingBox.height());
180 float radius = floatValueForRadiusInBox(boundingBox.width(), boundingBox.height());
181
182 return cachedEllipsePath(FloatRect(centerX - radius + boundingBox.x(), centerY - radius + boundingBox.y(), radius * 2, radius * 2));
183}

224float BasicShapeEllipse::floatValueForRadiusInBox(const BasicShapeRadius& radius, float center, float boxWidthOrHeight) const
225{
226 if (radius.type() == BasicShapeRadius::Value)
227 return floatValueForLength(radius.value(), std::abs(boxWidthOrHeight));
228
229 float widthOrHeightDelta = std::abs(boxWidthOrHeight - center);
230 if (radius.type() == BasicShapeRadius::ClosestSide)
231 return std::min(std::abs(center), widthOrHeightDelta);
232
233 ASSERT(radius.type() == BasicShapeRadius::FarthestSide);
234 return std::max(std::abs(center), widthOrHeightDelta);
235}

256Ref<BasicShape> BasicShapeEllipse::blend(const BasicShape& other, const BlendingContext& context) const
257{
258 ASSERT(type() == other.type());
259 auto& otherEllipse = downcast<BasicShapeEllipse>(other);
260 auto result = BasicShapeEllipse::create();
261
262 if (m_radiusX.type() != BasicShapeRadius::Value || otherEllipse.radiusX().type() != BasicShapeRadius::Value
263 || m_radiusY.type() != BasicShapeRadius::Value || otherEllipse.radiusY().type() != BasicShapeRadius::Value) {
264 result->setCenterX(otherEllipse.centerX());
265 result->setCenterY(otherEllipse.centerY());
266 result->setRadiusX(otherEllipse.radiusX());
267 result->setRadiusY(otherEllipse.radiusY());
268 return result;
269 }
270
271 result->setCenterX(m_centerX.blend(otherEllipse.centerX(), context));
272 result->setCenterY(m_centerY.blend(otherEllipse.centerY(), context));
273 result->setRadiusX(m_radiusX.blend(otherEllipse.radiusX(), context));
274 result->setRadiusY(m_radiusY.blend(otherEllipse.radiusY(), context));
275 return result;
276}

296const Path& BasicShapePolygon::path(const FloatRect& boundingBox)
297{
298 ASSERT(!(m_values.size() % 2));
299 size_t length = m_values.size();
300
301 Vector<FloatPoint> points(length / 2);
302 for (size_t i = 0; i < points.size(); ++i) {
303 points[i].setX(floatValueForLength(m_values.at(i * 2), boundingBox.width()) + boundingBox.x());
304 points[i].setY(floatValueForLength(m_values.at(i * 2 + 1), boundingBox.height()) + boundingBox.y());
305 }
306
307 return cachedPolygonPath(points);
308}

319Ref<BasicShape> BasicShapePolygon::blend(const BasicShape& other, const BlendingContext& context) const
320{
321 ASSERT(type() == other.type());
322
323 auto& otherPolygon = downcast<BasicShapePolygon>(other);
324 ASSERT(m_values.size() == otherPolygon.values().size());
325 ASSERT(!(m_values.size() % 2));
326
327 size_t length = m_values.size();
328 auto result = BasicShapePolygon::create();
329 if (!length)
330 return result;
331
332 result->setWindRule(otherPolygon.windRule());
333
334 for (size_t i = 0; i < length; i = i + 2) {
335 result->appendPoint(
336 WebCore::blend(otherPolygon.values().at(i), m_values.at(i), context),
337 WebCore::blend(otherPolygon.values().at(i + 1), m_values.at(i + 1), context));
338 }
339
340 return result;
341}

397bool BasicShapeInset::operator==(const BasicShape& other) const
398{

404 && m_top == otherInset.m_top

407 && m_topLeftRadius == otherInset.m_topLeftRadius

413const Path& BasicShapeInset::path(const FloatRect& boundingBox)
414{
415 float left = floatValueForLength(m_left, boundingBox.width());
416 float top = floatValueForLength(m_top, boundingBox.height());
417 auto rect = FloatRect(left + boundingBox.x(), top + boundingBox.y(),
418 std::max<float>(boundingBox.width() - left - floatValueForLength(m_right, boundingBox.width()), 0),
419 std::max<float>(boundingBox.height() - top - floatValueForLength(m_bottom, boundingBox.height()), 0));
420 auto radii = FloatRoundedRect::Radii(floatSizeForLengthSize(m_topLeftRadius, boundingBox.size()),
421 floatSizeForLengthSize(m_topRightRadius, boundingBox.size()),
422 floatSizeForLengthSize(m_bottomLeftRadius, boundingBox.size()),
423 floatSizeForLengthSize(m_bottomRightRadius, boundingBox.size()));
424 radii.scale(calcBorderRadiiConstraintScaleFor(rect, radii));
425
426 return cachedRoundedRectPath(FloatRoundedRect(rect, radii));
427}

434Ref<BasicShape> BasicShapeInset::blend(const BasicShape& from, const BlendingContext& context) const
435{
436 ASSERT(type() == from.type());
437
438 auto& fromInset = downcast<BasicShapeInset>(from);
439 auto result = BasicShapeInset::create();
440 result->setTop(WebCore::blend(fromInset.top(), top(), context));
441 result->setRight(WebCore::blend(fromInset.right(), right(), context));
442 result->setBottom(WebCore::blend(fromInset.bottom(), bottom(), context));
443 result->setLeft(WebCore::blend(fromInset.left(), left(), context));
444
445 result->setTopLeftRadius(WebCore::blend(fromInset.topLeftRadius(), topLeftRadius(), context));
446 result->setTopRightRadius(WebCore::blend(fromInset.topRightRadius(), topRightRadius(), context));
447 result->setBottomRightRadius(WebCore::blend(fromInset.bottomRightRadius(), bottomRightRadius(), context));
448 result->setBottomLeftRadius(WebCore::blend(fromInset.bottomLeftRadius(), bottomLeftRadius(), context));
449
450 return result;
451}
