25#pragma once
26
27#include "RenderStyleConstants.h"
28#include <wtf/text/AtomString.h>
29
30namespace WebCore {
31
32class CounterContent {
33 WTF_MAKE_FAST_ALLOCATED;
34public:
35 CounterContent(const AtomString& identifier, ListStyleType style, const AtomString& separator)
36 : m_identifier(identifier)
37 , m_listStyle(style)
38 , m_separator(separator)
39 {
40 ASSERT(style != ListStyleType::String);
41 }
42
43 const AtomString& identifier() const { return m_identifier; }
44 ListStyleType listStyle() const { return m_listStyle; }
45 const AtomString& separator() const { return m_separator; }
46
47private:
48 AtomString m_identifier;
49 ListStyleType m_listStyle;
50 AtomString m_separator;
51};
52
53static inline bool operator==(const CounterContent& a, const CounterContent& b)
54{
55 return a.identifier() == b.identifier()
56 && a.listStyle() == b.listStyle()
57 && a.separator() == b.separator();
58}
59
60} // namespace WebCore
