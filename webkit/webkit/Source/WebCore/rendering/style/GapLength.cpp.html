34#include <wtf/text/TextStream.h>

36namespace WebCore {
37
38TextStream& operator<<(TextStream& ts, const GapLength& gapLength)
39{
40 if (gapLength.isNormal())
41 ts << "normal";
42 else
43 ts << gapLength.length();
44 return ts;
45}
