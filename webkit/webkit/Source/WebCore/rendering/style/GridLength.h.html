32#pragma once
33
34#include "Length.h"
35
36namespace WebCore {
37
38// This class wraps the <track-breadth> which can be either a <percentage>, <length>, min-content, max-content
39// or <flex>. This class avoids spreading the knowledge of <flex> throughout the rendering directory by adding
40// an new unit to Length.h.
41class GridLength {
42public:
43 GridLength(const Length& length)
44 : m_length(length)
45 , m_flex(0)
46 , m_type(LengthType)
47 {
48 ASSERT(!length.isUndefined());
49 }
50
51 explicit GridLength(double flex)
52 : m_flex(flex)
53 , m_type(FlexType)
54 {
55 }
56
57 bool isLength() const { return m_type == LengthType; }
58 bool isFlex() const { return m_type == FlexType; }
59
60 const Length& length() const { ASSERT(isLength()); return m_length; }
61
62 double flex() const { ASSERT(isFlex()); return m_flex; }
63
64 bool isPercentage() const { return m_type == LengthType && m_length.isPercentOrCalculated(); }
65
66 bool operator==(const GridLength& o) const
67 {
68 return m_type == o.m_type && m_flex == o.m_flex && m_length == o.m_length;
69 }
70
71 bool isContentSized() const { return m_type == LengthType && (m_length.isAuto() || m_length.isMinContent() || m_length.isMaxContent()); }
72 bool isAuto() const { return m_type == LengthType && m_length.isAuto(); }
73
74private:
75 // Ideally we would put the 2 following fields in a union, but Length has a constructor,
76 // a destructor and a copy assignment which isn't allowed.
77 Length m_length;
78 double m_flex;
79 enum GridLengthType {
80 LengthType,
81 FlexType
82 };
83 GridLengthType m_type;
84};
85
86} // namespace WebCore
