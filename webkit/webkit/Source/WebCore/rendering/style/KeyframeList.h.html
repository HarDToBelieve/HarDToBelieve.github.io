42class KeyframeValue {
43public:
44 KeyframeValue(double key, std::unique_ptr<RenderStyle> style)
45 : m_key(key)
46 , m_style(WTFMove(style))
47 {
48 }
49
50 void addProperty(CSSPropertyID prop) { m_properties.add(prop); }
51 bool containsProperty(CSSPropertyID prop) const { return m_properties.contains(prop); }
52 const HashSet<CSSPropertyID>& properties() const { return m_properties; }
53
54 double key() const { return m_key; }
55 void setKey(double key) { m_key = key; }
56
57 const RenderStyle* style() const { return m_style.get(); }
58 void setStyle(std::unique_ptr<RenderStyle> style) { m_style = WTFMove(style); }
59
60 TimingFunction* timingFunction() const { return m_timingFunction.get(); }
61 void setTimingFunction(const RefPtr<TimingFunction>& timingFunction) { m_timingFunction = timingFunction; }
62
63private:
64 double m_key;
65 HashSet<CSSPropertyID> m_properties; // The properties specified in this keyframe.
66 std::unique_ptr<RenderStyle> m_style;
67 RefPtr<TimingFunction> m_timingFunction;
68};
69
70class KeyframeList {
71public:
72 explicit KeyframeList(const AtomString& animationName)
73 : m_animationName(animationName)
74 {
75 }
76 ~KeyframeList();
77
78 KeyframeList& operator=(KeyframeList&&) = default;
79 bool operator==(const KeyframeList& o) const;
80 bool operator!=(const KeyframeList& o) const { return !(*this == o); }
81
82 const AtomString& animationName() const { return m_animationName; }
83
84 void insert(KeyframeValue&&);
85
86 void addProperty(CSSPropertyID prop) { m_properties.add(prop); }
87 bool containsProperty(CSSPropertyID prop) const { return m_properties.contains(prop); }
88 const HashSet<CSSPropertyID>& properties() const { return m_properties; }
89
90 void clear();
91 bool isEmpty() const { return m_keyframes.isEmpty(); }
92 size_t size() const { return m_keyframes.size(); }
93 const KeyframeValue& operator[](size_t index) const { return m_keyframes[index]; }
94 const Vector<KeyframeValue>& keyframes() const { return m_keyframes; }
95
96 void copyKeyframes(KeyframeList&);
97 bool hasImplicitKeyframes() const;
98 void fillImplicitKeyframes(const Element&, Style::Resolver&, const RenderStyle* elementStyle, const RenderStyle* parentElementStyle);
99
100private:
101 AtomString m_animationName;
102 Vector<KeyframeValue> m_keyframes; // Kept sorted by key.
103 HashSet<CSSPropertyID> m_properties; // The properties being animated.
104};
