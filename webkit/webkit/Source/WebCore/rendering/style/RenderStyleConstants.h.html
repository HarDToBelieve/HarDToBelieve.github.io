48// The difference between two styles. The following values are used:
49// - StyleDifference::Equal - The two styles are identical
50// - StyleDifference::RecompositeLayer - The layer needs its position and transform updated, but no repaint
51// - StyleDifference::Repaint - The object just needs to be repainted.
52// - StyleDifference::RepaintIfTextOrBorderOrOutline - The object needs to be repainted if it contains text or a border or outline.
53// - StyleDifference::RepaintLayer - The layer and its descendant layers needs to be repainted.
54// - StyleDifference::LayoutPositionedMovementOnly - Only the position of this positioned object has been updated
55// - StyleDifference::SimplifiedLayout - Only overflow needs to be recomputed
56// - StyleDifference::SimplifiedLayoutAndPositionedMovement - Both positioned movement and simplified layout updates are required.
57// - StyleDifference::Layout - A full layout is required.
58enum class StyleDifference {
59 Equal,
60 RecompositeLayer,
61 Repaint,
62 RepaintIfTextOrBorderOrOutline,
63 RepaintLayer,
64 LayoutPositionedMovementOnly,
65 SimplifiedLayout,
66 SimplifiedLayoutAndPositionedMovement,
67 Layout,
68 NewStyle
69};

73// A simple StyleDifference does not provide enough information so we return a bit mask of
74// StyleDifferenceContextSensitiveProperties from RenderStyle::diff() too.
75enum class StyleDifferenceContextSensitiveProperty {

77 Transform = 1 << 0,
78 Opacity = 1 << 1,
79 Filter = 1 << 2,
80 ClipRect = 1 << 3,
81 ClipPath = 1 << 4,
82 WillChange = 1 << 5,
83};

115class PseudoIdSet {
116public:
117 PseudoIdSet()
118 : m_data(0)
119 {
120 }
121
122 PseudoIdSet(std::initializer_list<PseudoId> initializerList)
123 : m_data(0)
124 {
125 for (PseudoId pseudoId : initializerList)
126 add(pseudoId);
127 }
128
129 static PseudoIdSet fromMask(unsigned rawPseudoIdSet)
130 {
131 return PseudoIdSet(rawPseudoIdSet);
132 }
133
134 bool has(PseudoId pseudoId) const
135 {
136 ASSERT((sizeof(m_data) * 8) > static_cast<unsigned>(pseudoId));
137 return m_data & (1U << static_cast<unsigned>(pseudoId));
138 }
139
140 void add(PseudoId pseudoId)
141 {
142 ASSERT((sizeof(m_data) * 8) > static_cast<unsigned>(pseudoId));
143 m_data |= (1U << static_cast<unsigned>(pseudoId));
144 }
145
146 void remove(PseudoId pseudoId)
147 {
148 ASSERT((sizeof(m_data) * 8) > static_cast<unsigned>(pseudoId));
149 m_data &= ~(1U << static_cast<unsigned>(pseudoId));
150 }
151
152 void merge(PseudoIdSet source)
153 {
154 m_data |= source.m_data;
155 }
156
157 PseudoIdSet operator &(const PseudoIdSet& pseudoIdSet) const
158 {
159 return PseudoIdSet(m_data & pseudoIdSet.m_data);
160 }
161
162 PseudoIdSet operator |(const PseudoIdSet& pseudoIdSet) const
163 {
164 return PseudoIdSet(m_data | pseudoIdSet.m_data);
165 }
166
167 explicit operator bool() const
168 {
169 return m_data;
170 }
171
172 unsigned data() const { return m_data; }
173
174 static ptrdiff_t dataMemoryOffset() { return OBJECT_OFFSETOF(PseudoIdSet, m_data); }
175
176private:
177 explicit PseudoIdSet(unsigned rawPseudoIdSet)
178 : m_data(rawPseudoIdSet)
179 {
180 }
181
182 unsigned m_data;
183};
