48bool ShadowData::operator==(const ShadowData& o) const
49{
50 if (!arePointingToEqualData(m_next, o.m_next))
51 return false;
52
53 return m_location == o.m_location
54 && m_radius == o.m_radius
55 && m_spread == o.m_spread
56 && m_style == o.m_style
57 && m_color == o.m_color
58 && m_isWebkitBoxShadow == o.m_isWebkitBoxShadow;
59}

61static inline void calculateShadowExtent(const ShadowData* shadow, LayoutUnit additionalOutlineSize, LayoutUnit& shadowLeft, LayoutUnit& shadowRight, LayoutUnit& shadowTop, LayoutUnit& shadowBottom)
62{
63 do {
64 LayoutUnit extentAndSpread = shadow->paintingExtent() + shadow->spread() + additionalOutlineSize;
65 if (shadow->style() == ShadowStyle::Normal) {
66 shadowLeft = std::min(shadow->x() - extentAndSpread, shadowLeft);
67 shadowRight = std::max(shadow->x() + extentAndSpread, shadowRight);
68 shadowTop = std::min(shadow->y() - extentAndSpread, shadowTop);
69 shadowBottom = std::max(shadow->y() + extentAndSpread, shadowBottom);
70 }
71
72 shadow = shadow->next();
73 } while (shadow);
74}

76void ShadowData::adjustRectForShadow(LayoutRect& rect, int additionalOutlineSize) const
77{
78 LayoutUnit shadowLeft;
79 LayoutUnit shadowRight;
80 LayoutUnit shadowTop;
81 LayoutUnit shadowBottom;
82 calculateShadowExtent(this, additionalOutlineSize, shadowLeft, shadowRight, shadowTop, shadowBottom);
83
84 rect.move(shadowLeft, shadowTop);
85 rect.setWidth(rect.width() - shadowLeft + shadowRight);
86 rect.setHeight(rect.height() - shadowTop + shadowBottom);
87}

89void ShadowData::adjustRectForShadow(FloatRect& rect, int additionalOutlineSize) const

96
97 rect.move(shadowLeft, shadowTop);
98 rect.setWidth(rect.width() - shadowLeft + shadowRight);
99 rect.setHeight(rect.height() - shadowTop + shadowBottom);
100}
