30StyleBackgroundData::StyleBackgroundData()
31 : background(FillLayer::create(FillLayerType::Background))
32 , color(RenderStyle::initialBackgroundColor())
33{
34}
35
36inline StyleBackgroundData::StyleBackgroundData(const StyleBackgroundData& other)
37 : RefCounted<StyleBackgroundData>()
38 , background(other.background)
39 , color(other.color)
40 , outline(other.outline)
41{
42}
43
44Ref<StyleBackgroundData> StyleBackgroundData::copy() const
45{
46 return adoptRef(*new StyleBackgroundData(*this));
47}
48
49bool StyleBackgroundData::operator==(const StyleBackgroundData& other) const
50{
51 return background == other.background && color == other.color && outline == other.outline;
52}
53
54bool StyleBackgroundData::isEquivalentForPainting(const StyleBackgroundData& other, bool currentColorDiffers) const
55{
56 if (background != other.background || color != other.color)
57 return false;
58 if (currentColorDiffers && color == RenderStyle::currentColor())
59 return false;
60 if (!outline.isVisible() && !other.outline.isVisible())
61 return true;
62 if (currentColorDiffers && outline.color() == RenderStyle::currentColor())
63 return false;
64 return outline == other.outline;
65}
66
67void StyleBackgroundData::dump(TextStream& ts, DumpStyleValues behavior) const
68{
69 if (behavior == DumpStyleValues::All || *background != FillLayer::create(FillLayerType::Background).get())
70 ts.dumpProperty("background-image", background);
71 if (behavior == DumpStyleValues::All || color != RenderStyle::initialBackgroundColor())
72 ts.dumpProperty("background-color", color);
73 if (behavior == DumpStyleValues::All || outline != OutlineValue())
74 ts.dumpProperty("outline", outline);
75}
76
77TextStream& operator<<(TextStream& ts, const StyleBackgroundData& backgroundData)
78{
79 backgroundData.dump(ts);
80 return ts;
81}
