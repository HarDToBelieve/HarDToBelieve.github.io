30struct SameSizeAsStyleBoxData : public RefCounted<SameSizeAsStyleBoxData> {

40StyleBoxData::StyleBoxData()

44 , m_maxHeight(RenderStyle::initialMaxSize())
45 , m_specifiedZIndex(0)

47 , m_hasAutoSpecifiedZIndex(true)

81bool StyleBoxData::operator==(const StyleBoxData& o) const
82{
83 return m_width == o.m_width
84 && m_height == o.m_height
85 && m_minWidth == o.m_minWidth
86 && m_maxWidth == o.m_maxWidth
87 && m_minHeight == o.m_minHeight
88 && m_maxHeight == o.m_maxHeight
89 && m_verticalAlign == o.m_verticalAlign
90 && m_specifiedZIndex == o.m_specifiedZIndex
91 && m_hasAutoSpecifiedZIndex == o.m_hasAutoSpecifiedZIndex
92 && m_usedZIndex == o.m_usedZIndex
93 && m_hasAutoUsedZIndex == o.m_hasAutoUsedZIndex
94 && m_boxSizing == o.m_boxSizing
95#if ENABLE(CSS_BOX_DECORATION_BREAK)
96 && m_boxDecorationBreak == o.m_boxDecorationBreak
97#endif
98 ;
99}
