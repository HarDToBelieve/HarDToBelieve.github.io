33StyleFlexibleBoxData::StyleFlexibleBoxData()
34 : flexGrow(RenderStyle::initialFlexGrow())
35 , flexShrink(RenderStyle::initialFlexShrink())
36 , flexBasis(RenderStyle::initialFlexBasis())
37 , flexDirection(static_cast<unsigned>(RenderStyle::initialFlexDirection()))
38 , flexWrap(static_cast<unsigned>(RenderStyle::initialFlexWrap()))
39{
40}
41
42inline StyleFlexibleBoxData::StyleFlexibleBoxData(const StyleFlexibleBoxData& other)
43 : RefCounted<StyleFlexibleBoxData>()
44 , flexGrow(other.flexGrow)
45 , flexShrink(other.flexShrink)
46 , flexBasis(other.flexBasis)
47 , flexDirection(other.flexDirection)
48 , flexWrap(other.flexWrap)
49{
50}
51
52Ref<StyleFlexibleBoxData> StyleFlexibleBoxData::copy() const
53{
54 return adoptRef(*new StyleFlexibleBoxData(*this));
55}
56
57bool StyleFlexibleBoxData::operator==(const StyleFlexibleBoxData& other) const
58{
59 return flexGrow == other.flexGrow && flexShrink == other.flexShrink && flexBasis == other.flexBasis
60 && flexDirection == other.flexDirection && flexWrap == other.flexWrap;
61}
