62FloatSize StyleGeneratedImage::imageSize(const RenderElement* renderer, float multiplier) const
63{
64 if (!m_fixedSize)
65 return m_containerSize;
66
67 if (!renderer)
68 return { };
69
70 FloatSize fixedSize = m_imageGeneratorValue->fixedSize(*renderer);
71 if (multiplier == 1.0f)
72 return fixedSize;
73
74 float width = fixedSize.width() * multiplier;
75 float height = fixedSize.height() * multiplier;
76
77 // Don't let images that have a width/height >= 1 shrink below 1 device pixel when zoomed.
78 float deviceScaleFactor = renderer->document().deviceScaleFactor();
79 if (fixedSize.width() > 0)
80 width = std::max<float>(1 / deviceScaleFactor, width);
81 if (fixedSize.height() > 0)
82 height = std::max<float>(1 / deviceScaleFactor, height);
83
84 return { width, height };
85}

87void StyleGeneratedImage::computeIntrinsicDimensions(const RenderElement* renderer, Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio)
88{
89 // At a zoom level of 1 the image is guaranteed to have a device pixel size.
90 FloatSize size = floorSizeToDevicePixels(LayoutSize(imageSize(renderer, 1)), renderer ? renderer->document().deviceScaleFactor() : 1);
91 intrinsicWidth = Length(size.width(), LengthType::Fixed);
92 intrinsicHeight = Length(size.height(), LengthType::Fixed);
93 intrinsicRatio = size;
94}
