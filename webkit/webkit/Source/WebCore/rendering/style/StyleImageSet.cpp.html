30namespace WebCore {

32Ref<StyleImageSet> StyleImageSet::create(CSSImageSetValue& cssValue)
33{
34 return adoptRef(*new StyleImageSet(cssValue));
35}
36
37StyleImageSet::StyleImageSet(CSSImageSetValue& cssValue)
38 : m_cssValue(cssValue)
39{
40 m_isImageSet = true;
41}
42
43bool StyleImageSet::operator==(const StyleImage& other) const
44{
45 return is<StyleImageSet>(other) && equals(downcast<StyleImageSet>(other));
46}
47
48StyleImageSet::~StyleImageSet() = default;
49
50Ref<CSSValue> StyleImageSet::cssValue() const
51{
52 return m_cssValue.copyRef();
53}
54
55ImageWithScale StyleImageSet::selectBestFitImage(const Document& document) const
56{
57 return m_cssValue->selectBestFitImage(document);
58}
