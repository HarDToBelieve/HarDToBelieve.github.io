29DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleInheritedData);

31StyleInheritedData::StyleInheritedData()
32 : horizontalBorderSpacing(RenderStyle::initialHorizontalBorderSpacing())
33 , verticalBorderSpacing(RenderStyle::initialVerticalBorderSpacing())
34 , lineHeight(RenderStyle::initialLineHeight())
35#if ENABLE(TEXT_AUTOSIZING)
36 , specifiedLineHeight(RenderStyle::initialLineHeight())
37#endif
38 , color(RenderStyle::initialColor())
39 , visitedLinkColor(RenderStyle::initialColor())
40{
41}
42
43inline StyleInheritedData::StyleInheritedData(const StyleInheritedData& o)
44 : RefCounted<StyleInheritedData>()
45 , horizontalBorderSpacing(o.horizontalBorderSpacing)
46 , verticalBorderSpacing(o.verticalBorderSpacing)
47 , lineHeight(o.lineHeight)
48#if ENABLE(TEXT_AUTOSIZING)
49 , specifiedLineHeight(o.specifiedLineHeight)
50#endif
51 , fontCascade(o.fontCascade)
52 , color(o.color)
53 , visitedLinkColor(o.visitedLinkColor)
54{
55}
56
57Ref<StyleInheritedData> StyleInheritedData::copy() const
58{
59 return adoptRef(*new StyleInheritedData(*this));
60}
61
62bool StyleInheritedData::operator==(const StyleInheritedData& o) const
63{
64 return lineHeight == o.lineHeight
65#if ENABLE(TEXT_AUTOSIZING)
66 && specifiedLineHeight == o.specifiedLineHeight
67#endif
68 && fontCascade == o.fontCascade
69 && color == o.color
70 && visitedLinkColor == o.visitedLinkColor
71 && horizontalBorderSpacing == o.horizontalBorderSpacing
72 && verticalBorderSpacing == o.verticalBorderSpacing;
73}
