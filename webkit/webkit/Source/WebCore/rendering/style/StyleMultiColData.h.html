25#pragma once
26
27#include "BorderValue.h"
28#include "GapLength.h"
29#include "RenderStyleConstants.h"
30#include <wtf/RefCounted.h>
31
32namespace WebCore {
33
34// CSS3 Multi Column Layout
35
36class StyleMultiColData : public RefCounted<StyleMultiColData> {
37public:
38 static Ref<StyleMultiColData> create() { return adoptRef(*new StyleMultiColData); }
39 Ref<StyleMultiColData> copy() const;
40
41 bool operator==(const StyleMultiColData&) const;
42 bool operator!=(const StyleMultiColData& other) const { return !(*this == other); }
43
44 unsigned short ruleWidth() const
45 {
46 if (rule.style() == BorderStyle::None || rule.style() == BorderStyle::Hidden)
47 return 0;
48 return rule.width();
49 }
50
51 float width { 0 };
52 unsigned short count;
53 BorderValue rule;
54 Color visitedLinkColumnRuleColor;
55
56 bool autoWidth : 1;
57 bool autoCount : 1;
58 unsigned fill : 1; // ColumnFill
59 unsigned columnSpan : 1; // ColumnSpan
60 unsigned axis : 2; // ColumnAxis
61 unsigned progression : 2; // ColumnProgression
62
63private:
64 StyleMultiColData();
65 StyleMultiColData(const StyleMultiColData&);
66};
67
68} // namespace WebCore
