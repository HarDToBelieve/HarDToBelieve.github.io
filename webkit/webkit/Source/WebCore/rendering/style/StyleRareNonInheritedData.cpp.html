324bool StyleRareNonInheritedData::contentDataEquivalent(const StyleRareNonInheritedData& other) const
325{
326 auto* a = content.get();
327 auto* b = other.content.get();
328 while (a && b && *a == *b) {
329 a = a->next();
330 b = b->next();
331 }
332 return !a && !b;
333}
