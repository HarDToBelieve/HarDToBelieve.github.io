33namespace WebCore {
34
35DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleTransformData);
36class StyleTransformData : public RefCounted<StyleTransformData> {
37 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(StyleTransformData);
38public:
39 static Ref<StyleTransformData> create() { return adoptRef(*new StyleTransformData); }
40 Ref<StyleTransformData> copy() const;
41
42 bool operator==(const StyleTransformData&) const;
43 bool operator!=(const StyleTransformData& other) const
44 {
45 return !(*this == other);
46 }
47
48 bool hasTransform() const { return operations.size(); }
49
50 LengthPoint originXY() const { return { x, y }; }
51
52 TransformOperations operations;
53 Length x;
54 Length y;
55 float z;
56 TransformBox transformBox;
57
58private:
59 StyleTransformData();
60 StyleTransformData(const StyleTransformData&);
61};
