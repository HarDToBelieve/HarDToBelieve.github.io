31bool WillChangeData::operator==(const WillChangeData& other) const
32{
33 return m_animatableFeatures == other.m_animatableFeatures;

36bool WillChangeData::containsScrollPosition() const
37{
38 for (const auto& feature : m_animatableFeatures) {
39 if (feature.feature() == ScrollPosition)
40 return true;
41 }
42 return false;
43}

45bool WillChangeData::containsContents() const

54bool WillChangeData::containsProperty(CSSPropertyID property) const
55{

63bool WillChangeData::createsContainingBlockForAbsolutelyPositioned() const
64{
65 return createsContainingBlockForOutOfFlowPositioned()
66 || containsProperty(CSSPropertyPosition);
67}
68
69bool WillChangeData::createsContainingBlockForOutOfFlowPositioned() const
70{
71 return containsProperty(CSSPropertyPerspective)
72 // CSS transforms
73 || containsProperty(CSSPropertyTransform)
74 || containsProperty(CSSPropertyTransformStyle)
75 || containsProperty(CSSPropertyTranslate)
76 || containsProperty(CSSPropertyRotate)
77 || containsProperty(CSSPropertyScale)
78 || containsProperty(CSSPropertyContain)
79 // CSS filter & backdrop-filter
80 // FIXME: exclude root element for those properties (bug 225034)
81#if ENABLE(FILTERS_LEVEL_2)
82 || containsProperty(CSSPropertyWebkitBackdropFilter)
83#endif
84 || containsProperty(CSSPropertyFilter);
85}

87// "If any non-initial value of a property would create a stacking context on the element,
88// specifying that property in will-change must create a stacking context on the element."
89bool WillChangeData::propertyCreatesStackingContext(CSSPropertyID property)
90{
91 switch (property) {
92 case CSSPropertyPerspective:
93 case CSSPropertyScale:
94 case CSSPropertyRotate:
95 case CSSPropertyTranslate:
96 case CSSPropertyTransform:
97 case CSSPropertyTransformStyle:
98 case CSSPropertyWebkitTransformStyle:
99 case CSSPropertyClipPath:
100 case CSSPropertyMask:
101 case CSSPropertyOpacity:
102 case CSSPropertyPosition:
103 case CSSPropertyZIndex:
104 case CSSPropertyWebkitBoxReflect:
105#if ENABLE(CSS_COMPOSITING)
106 case CSSPropertyMixBlendMode:
107 case CSSPropertyIsolation:
108#endif
109 case CSSPropertyFilter:
110#if ENABLE(FILTERS_LEVEL_2)
111 case CSSPropertyWebkitBackdropFilter:
112#endif
113 case CSSPropertyWebkitMask:
114 case CSSPropertyWebkitMaskImage:
115 case CSSPropertyWebkitMaskBoxImage:
116#if ENABLE(OVERFLOW_SCROLLING_TOUCH)
117 case CSSPropertyWebkitOverflowScrolling:
118#endif
119 case CSSPropertyContain:
120 return true;
121 default:
122 return false;
123 }
124}

126static bool propertyTriggersCompositing(CSSPropertyID property)
127{
128 switch (property) {
129 case CSSPropertyOpacity:
130 case CSSPropertyFilter:
131#if ENABLE(FILTERS_LEVEL_2)
132 case CSSPropertyWebkitBackdropFilter:
133#endif
134 return true;
135 default:
136 return false;
137 }
138}
139
140static bool propertyTriggersCompositingOnBoxesOnly(CSSPropertyID property)
141{
142 // Don't trigger for perspective and transform-style, because those
143 // only do compositing if they have a 3d-transformed descendant and
144 // we don't want to do compositing all the time.
145 // Similarly, we don't want -webkit-overflow-scrolling-touch to
146 // always composite if there's no scrollable overflow.
147 switch (property) {
148 case CSSPropertyScale:
149 case CSSPropertyRotate:
150 case CSSPropertyTranslate:
151 case CSSPropertyTransform:
152 return true;
153 default:
154 return false;
155 }
156}

158void WillChangeData::addFeature(Feature feature, CSSPropertyID propertyID)
159{
160 ASSERT(feature == Property || propertyID == CSSPropertyInvalid);
161 m_animatableFeatures.append(AnimatableFeature(feature, propertyID));
162
163 m_canCreateStackingContext |= propertyCreatesStackingContext(propertyID);
164
165 m_canTriggerCompositingOnInline |= propertyTriggersCompositing(propertyID);
166 m_canTriggerCompositing |= m_canTriggerCompositingOnInline | propertyTriggersCompositingOnBoxesOnly(propertyID);
167}
168
169WillChangeData::FeaturePropertyPair WillChangeData::featureAt(size_t index) const
170{
171 if (index >= m_animatableFeatures.size())
172 return FeaturePropertyPair(Invalid, CSSPropertyInvalid);
173
174 return m_animatableFeatures[index].featurePropertyPair();
175}
