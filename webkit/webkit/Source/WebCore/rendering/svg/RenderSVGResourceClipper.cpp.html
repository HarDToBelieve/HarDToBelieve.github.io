70bool RenderSVGResourceClipper::applyResource(RenderElement& renderer, const RenderStyle&, GraphicsContext*& context, OptionSet<RenderSVGResourceMode> resourceMode)
71{
72 ASSERT(context);
73 ASSERT_UNUSED(resourceMode, !resourceMode);
74
75 auto repaintRect = renderer.repaintRectInLocalCoordinates();
76 if (repaintRect.isEmpty())
77 return true;
78
79 return applyClippingToContext(renderer, renderer.objectBoundingBox(), *context);
80}

82bool RenderSVGResourceClipper::pathOnlyClipping(GraphicsContext& context, const AffineTransform& animatedLocalTransform, const FloatRect& objectBoundingBox)
83{
84 // If the current clip-path gets clipped itself, we have to fall back to masking.
85 if (style().clipPath())
86 return false;
87 WindRule clipRule = WindRule::NonZero;
88 Path clipPath;
89
90 // If clip-path only contains one visible shape or path, we can use path-based clipping. Invisible
91 // shapes don't affect the clipping and can be ignored. If clip-path contains more than one
92 // visible shape, the additive clipping may not work, caused by the clipRule. EvenOdd
93 // as well as NonZero can cause self-clipping of the elements.
94 // See also http://www.w3.org/TR/SVG/painting.html#FillRuleProperty
95 for (Node* childNode = clipPathElement().firstChild(); childNode; childNode = childNode->nextSibling()) {
96 auto* renderer = childNode->renderer();
97 if (!renderer)
98 continue;
99 // Only shapes or paths are supported for direct clipping. We need to fall back to masking for texts.
100 if (is<RenderSVGText>(renderer))
101 return false;
102 if (!is<SVGGraphicsElement>(*childNode))
103 continue;
104 auto& style = renderer->style();
105 if (style.display() == DisplayType::None || style.visibility() != Visibility::Visible)
106 continue;
107 // Current shape in clip-path gets clipped too. Fall back to masking.
108 if (style.clipPath())
109 return false;
110 // Fall back to masking if there is more than one clipping path.
111 if (!clipPath.isEmpty())
112 return false;
113 clipPath = downcast<SVGGraphicsElement>(*childNode).toClipPath();
114 clipRule = style.svgStyle().clipRule();
115 }
116
117 // Only one visible shape/path was found. Directly continue clipping and transform the content to userspace if necessary.
118 if (clipPathElement().clipPathUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) {
119 AffineTransform transform;
120 transform.translate(objectBoundingBox.location());
121 transform.scale(objectBoundingBox.size());
122 clipPath.transform(transform);
123 }
124
125 // Transform path by animatedLocalTransform.
126 clipPath.transform(animatedLocalTransform);
127
128 // The SVG specification wants us to clip everything, if clip-path doesn't have a child.
129 if (clipPath.isEmpty())
130 clipPath.addRect(FloatRect());
131 context.clipPath(clipPath, clipRule);
132 return true;
133}

135bool RenderSVGResourceClipper::applyClippingToContext(RenderElement& renderer, const FloatRect& objectBoundingBox, GraphicsContext& context)
136{
137 ClipperData& clipperData = addRendererToClipper(renderer);
138
139 LOG_WITH_STREAM(SVG, stream << "RenderSVGResourceClipper " << this << " applyClippingToContext: renderer " << &renderer << " objectBoundingBox " << objectBoundingBox << " (existing image buffer " << clipperData.imageBuffer.get() << ")");
140
141 AffineTransform animatedLocalTransform = clipPathElement().animatedLocalTransform();
142
143 if (!clipperData.imageBuffer && pathOnlyClipping(context, animatedLocalTransform, objectBoundingBox))
144 return true;
145
146 AffineTransform absoluteTransform = SVGRenderingContext::calculateTransformationToOutermostCoordinateSystem(renderer);
147 if (!clipperData.isValidForGeometry(objectBoundingBox, absoluteTransform)) {
148 // FIXME (149469): This image buffer should not be unconditionally unaccelerated. Making it match the context breaks nested clipping, though.
149 auto maskImage = SVGRenderingContext::createImageBuffer(objectBoundingBox, absoluteTransform, DestinationColorSpace::SRGB(), RenderingMode::Unaccelerated, &context);
150 if (!maskImage)
151 return false;
152
153 clipperData = { WTFMove(maskImage), objectBoundingBox, absoluteTransform };
154
155 GraphicsContext& maskContext = clipperData.imageBuffer->context();
156 maskContext.concatCTM(animatedLocalTransform);
157
158 // clipPath can also be clipped by another clipPath.
159 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(*this);
160 RenderSVGResourceClipper* clipper;
161 bool succeeded;
162 if (resources && (clipper = resources->clipper())) {
163 GraphicsContextStateSaver stateSaver(maskContext);
164
165 if (!clipper->applyClippingToContext(*this, objectBoundingBox, maskContext))
166 return false;
167
168 succeeded = drawContentIntoMaskImage(*clipperData.imageBuffer, objectBoundingBox);
169 // The context restore applies the clipping on non-CG platforms.
170 } else
171 succeeded = drawContentIntoMaskImage(*clipperData.imageBuffer, objectBoundingBox);
172
173 if (!succeeded)
174 clipperData = { };
175 }
176
177 if (!clipperData.imageBuffer)
178 return false;
179
180 SVGRenderingContext::clipToImageBuffer(context, absoluteTransform, objectBoundingBox, clipperData.imageBuffer, true);
181 return true;
182}

184bool RenderSVGResourceClipper::drawContentIntoMaskImage(ImageBuffer& maskImageBuffer, const FloatRect& objectBoundingBox)
185{
186 GraphicsContext& maskContext = maskImageBuffer.context();
187
188 AffineTransform maskContentTransformation;
189 if (clipPathElement().clipPathUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) {
190 maskContentTransformation.translate(objectBoundingBox.location());
191 maskContentTransformation.scale(objectBoundingBox.size());
192 maskContext.concatCTM(maskContentTransformation);
193 }
194
195 // Switch to a paint behavior where all children of this <clipPath> will be rendered using special constraints:
196 // - fill-opacity/stroke-opacity/opacity set to 1
197 // - masker/filter not applied when rendering the children
198 // - fill is set to the initial fill paint server (solid, black)
199 // - stroke is set to the initial stroke paint server (none)
200 auto oldBehavior = view().frameView().paintBehavior();
201 view().frameView().setPaintBehavior(oldBehavior | PaintBehavior::RenderingSVGMask);
202
203 // Draw all clipPath children into a global mask.
204 for (auto& child : childrenOfType<SVGElement>(clipPathElement())) {
205 auto renderer = child.renderer();
206 if (!renderer)
207 continue;
208 if (renderer->needsLayout()) {
209 view().frameView().setPaintBehavior(oldBehavior);
210 return false;
211 }
212 const RenderStyle& style = renderer->style();
213 if (style.display() == DisplayType::None || style.visibility() != Visibility::Visible)
214 continue;
215
216 WindRule newClipRule = style.svgStyle().clipRule();
217 bool isUseElement = child.hasTagName(SVGNames::useTag);
218 if (isUseElement) {
219 SVGUseElement& useElement = downcast<SVGUseElement>(child);
220 renderer = useElement.rendererClipChild();
221 if (!renderer)
222 continue;
223 if (!useElement.hasAttributeWithoutSynchronization(SVGNames::clip_ruleAttr))
224 newClipRule = renderer->style().svgStyle().clipRule();
225 }
226
227 // Only shapes, paths and texts are allowed for clipping.
228 if (!renderer->isSVGShape() && !renderer->isSVGText())
229 continue;
230
231 maskContext.setFillRule(newClipRule);
232
233 // In the case of a <use> element, we obtained its renderere above, to retrieve its clipRule.
234 // We have to pass the <use> renderer itself to renderSubtreeToContext() to apply it's x/y/transform/etc. values when rendering.
235 // So if isUseElement is true, refetch the childNode->renderer(), as renderer got overridden above.
236 SVGRenderingContext::renderSubtreeToContext(maskContext, isUseElement ? *child.renderer() : *renderer, maskContentTransformation);
237 }
238
239 view().frameView().setPaintBehavior(oldBehavior);
240 return true;
241}

243void RenderSVGResourceClipper::calculateClipContentRepaintRect()
244{
245 // This is a rough heuristic to appraise the clip size and doesn't consider clip on clip.
246 for (Node* childNode = clipPathElement().firstChild(); childNode; childNode = childNode->nextSibling()) {
247 RenderObject* renderer = childNode->renderer();
248 if (!childNode->isSVGElement() || !renderer)
249 continue;
250 if (!renderer->isSVGShape() && !renderer->isSVGText() && !childNode->hasTagName(SVGNames::useTag))
251 continue;
252 const RenderStyle& style = renderer->style();
253 if (style.display() == DisplayType::None || style.visibility() != Visibility::Visible)
254 continue;
255 m_clipBoundaries.unite(renderer->localToParentTransform().mapRect(renderer->repaintRectInLocalCoordinates()));
256 }
257 m_clipBoundaries = clipPathElement().animatedLocalTransform().mapRect(m_clipBoundaries);
258}

265bool RenderSVGResourceClipper::hitTestClipContent(const FloatRect& objectBoundingBox, const FloatPoint& nodeAtPoint)
266{
267 FloatPoint point = nodeAtPoint;
268 if (!SVGRenderSupport::pointInClippingArea(*this, point))
269 return false;
270
271 SVGHitTestCycleDetectionScope hitTestScope(*this);
272
273 if (clipPathElement().clipPathUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) {
274 AffineTransform transform;
275 transform.translate(objectBoundingBox.location());
276 transform.scale(objectBoundingBox.size());
277 point = transform.inverse().value_or(AffineTransform()).mapPoint(point);
278 }
279
280 point = clipPathElement().animatedLocalTransform().inverse().value_or(AffineTransform()).mapPoint(point);
281
282 for (Node* childNode = clipPathElement().firstChild(); childNode; childNode = childNode->nextSibling()) {
283 RenderObject* renderer = childNode->renderer();
284 if (!childNode->isSVGElement() || !renderer)
285 continue;
286 if (!renderer->isSVGShape() && !renderer->isSVGText() && !childNode->hasTagName(SVGNames::useTag))
287 continue;
288
289 IntPoint hitPoint;
290 HitTestResult result(hitPoint);
291 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::SVGClipContent, HitTestRequest::Type::DisallowUserAgentShadowContent };
292 if (renderer->nodeAtFloatPoint(hitType, result, point, HitTestForeground))
293 return true;
294 }
295
296 return false;
297}

299FloatRect RenderSVGResourceClipper::resourceBoundingBox(const RenderObject& object)
300{
301 // Resource was not layouted yet. Give back the boundingBox of the object.
302 if (selfNeedsLayout()) {
303 addRendererToClipper(object); // For selfNeedsClientInvalidation().
304 return object.objectBoundingBox();
305 }
306
307 if (m_clipBoundaries.isEmpty())
308 calculateClipContentRepaintRect();
309
310 if (clipPathElement().clipPathUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) {
311 FloatRect objectBoundingBox = object.objectBoundingBox();
312 AffineTransform transform;
313 transform.translate(objectBoundingBox.location());
314 transform.scale(objectBoundingBox.size());
315 return transform.mapRect(m_clipBoundaries);
316 }
317
318 return m_clipBoundaries;
319}
