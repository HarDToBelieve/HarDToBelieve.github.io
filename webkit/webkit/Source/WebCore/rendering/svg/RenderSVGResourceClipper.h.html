20#pragma once
21
22#include "RenderSVGResourceContainer.h"
23#include "SVGClipPathElement.h"
24#include "SVGUnitTypes.h"
25
26#include <wtf/HashMap.h>
27
28namespace WebCore {
29
30class GraphicsContext;
31class ImageBuffer;
32
33class RenderSVGResourceClipper final : public RenderSVGResourceContainer {
34 WTF_MAKE_ISO_ALLOCATED(RenderSVGResourceClipper);
35public:
36 RenderSVGResourceClipper(SVGClipPathElement&, RenderStyle&&);
37 virtual ~RenderSVGResourceClipper();
38
39 SVGClipPathElement& clipPathElement() const { return downcast<SVGClipPathElement>(nodeForNonAnonymous()); }
40
41 void removeAllClientsFromCache(bool markForInvalidation = true) override;
42 void removeClientFromCache(RenderElement&, bool markForInvalidation = true) override;
43
44 bool applyResource(RenderElement&, const RenderStyle&, GraphicsContext*&, OptionSet<RenderSVGResourceMode>) override;
45 // clipPath can be clipped too, but don't have a boundingBox or repaintRect. So we can't call
46 // applyResource directly and use the rects from the object, since they are empty for RenderSVGResources
47 // FIXME: We made applyClippingToContext public because we cannot call applyResource on HTML elements (it asserts on RenderObject::objectBoundingBox)
48 bool applyClippingToContext(RenderElement&, const FloatRect&, GraphicsContext&);
49 FloatRect resourceBoundingBox(const RenderObject&) override;
50
51 RenderSVGResourceType resourceType() const override { return ClipperResourceType; }
52
53 bool hitTestClipContent(const FloatRect&, const FloatPoint&);
54
55 SVGUnitTypes::SVGUnitType clipPathUnits() const { return clipPathElement().clipPathUnits(); }
56
57private:
58 bool selfNeedsClientInvalidation() const override { return (everHadLayout() || m_clipper.size()) && selfNeedsLayout(); }
59
60 struct ClipperData {
61 FloatRect objectBoundingBox;
62 AffineTransform absoluteTransform;
63 RefPtr<ImageBuffer> imageBuffer;
64
65 ClipperData() = default;
66 ClipperData(RefPtr<ImageBuffer>&& buffer, const FloatRect& boundingBox, const AffineTransform& transform)
67 : objectBoundingBox(boundingBox)
68 , absoluteTransform(transform)
69 , imageBuffer(WTFMove(buffer))
70 {
71 }
72
73 bool isValidForGeometry(const FloatRect& boundingBox, const AffineTransform& transform) const
74 {
75 return imageBuffer && objectBoundingBox == boundingBox && absoluteTransform == transform;
76 }
77 };
78
79 void element() const = delete;
80
81 const char* renderName() const override { return "RenderSVGResourceClipper"; }
82 bool isSVGResourceClipper() const override { return true; }
83
84 bool pathOnlyClipping(GraphicsContext&, const AffineTransform&, const FloatRect&);
85 bool drawContentIntoMaskImage(ImageBuffer&, const FloatRect& objectBoundingBox);
86 void calculateClipContentRepaintRect();
87 ClipperData& addRendererToClipper(const RenderObject&);
88
89 FloatRect m_clipBoundaries;
90 HashMap<const RenderObject*, ClipperData> m_clipper;
91};
92
93}
94
95SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::RenderSVGResourceClipper)
96static bool isType(const WebCore::RenderObject& renderer) { return renderer.isSVGResourceClipper(); }
97static bool isType(const WebCore::RenderSVGResource& resource) { return resource.resourceType() == WebCore::ClipperResourceType; }
98SPECIALIZE_TYPE_TRAITS_END()
