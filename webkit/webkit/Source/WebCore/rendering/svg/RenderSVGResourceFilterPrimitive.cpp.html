43RenderSVGResourceFilterPrimitive::RenderSVGResourceFilterPrimitive(SVGFilterPrimitiveStandardAttributes& filterPrimitiveElement, RenderStyle&& style)
44 : RenderSVGHiddenContainer(filterPrimitiveElement, WTFMove(style))
45{
46}
47
48SVGFilterPrimitiveStandardAttributes& RenderSVGResourceFilterPrimitive::filterPrimitiveElement() const
49{
50 return static_cast<SVGFilterPrimitiveStandardAttributes&>(RenderSVGHiddenContainer::element());
51}
52
53void RenderSVGResourceFilterPrimitive::styleDidChange(StyleDifference diff, const RenderStyle* oldStyle)
54{
55 RenderSVGHiddenContainer::styleDidChange(diff, oldStyle);
56
57 auto* filter = parent();
58 if (!filter)
59 return;
60
61 if (diff == StyleDifference::Equal || !oldStyle)
62 return;
63
64 const SVGRenderStyle& newStyle = style().svgStyle();
65 if (is<SVGFEFloodElement>(filterPrimitiveElement())) {
66 if (newStyle.floodColor() != oldStyle->svgStyle().floodColor())
67 downcast<RenderSVGResourceFilter>(*filter).primitiveAttributeChanged(this, SVGNames::flood_colorAttr);
68 if (newStyle.floodOpacity() != oldStyle->svgStyle().floodOpacity())
69 downcast<RenderSVGResourceFilter>(*filter).primitiveAttributeChanged(this, SVGNames::flood_opacityAttr);
70 } else if (is<SVGFEDiffuseLightingElement>(filterPrimitiveElement()) || is<SVGFESpecularLightingElement>(filterPrimitiveElement())) {
71 if (newStyle.lightingColor() != oldStyle->svgStyle().lightingColor())
72 downcast<RenderSVGResourceFilter>(*filter).primitiveAttributeChanged(this, SVGNames::lighting_colorAttr);
73 }
74}
