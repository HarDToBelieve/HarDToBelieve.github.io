30namespace WebCore {
31
32class GraphicsContext;
33
34class RenderSVGResourceGradient : public RenderSVGResourceContainer {
35 WTF_MAKE_ISO_ALLOCATED(RenderSVGResourceGradient);
36public:
37 SVGGradientElement& gradientElement() const { return static_cast<SVGGradientElement&>(RenderSVGResourceContainer::element()); }
38
39 void removeAllClientsFromCache(bool markForInvalidation = true) final;
40 void removeClientFromCache(RenderElement&, bool markForInvalidation = true) final;
41
42 bool applyResource(RenderElement&, const RenderStyle&, GraphicsContext*&, OptionSet<RenderSVGResourceMode>) final;
43 void postApplyResource(RenderElement&, GraphicsContext*&, OptionSet<RenderSVGResourceMode>, const Path*, const RenderSVGShape*) final;
44 FloatRect resourceBoundingBox(const RenderObject&) final { return FloatRect(); }
45
46protected:
47 RenderSVGResourceGradient(SVGGradientElement&, RenderStyle&&);
48
49 static void addStops(Gradient&, const Gradient::ColorStopVector&, const RenderStyle&);
50 static GradientSpreadMethod platformSpreadMethodFromSVGType(SVGSpreadMethodType);
51
52private:
53 void element() const = delete;
54
55 virtual SVGUnitTypes::SVGUnitType gradientUnits() const = 0;
56 virtual AffineTransform gradientTransform() const = 0;
57 virtual bool collectGradientAttributes() = 0;
58 virtual Ref<Gradient> buildGradient(const RenderStyle&) const = 0;
59
60 struct GradientData {
61 RefPtr<Gradient> gradient;
62 AffineTransform userspaceTransform;
63 };
64 HashMap<RenderObject*, GradientData> m_gradientMap;
65
66#if USE(CG)
67 GraphicsContext* m_savedContext { nullptr };
68 RefPtr<ImageBuffer> m_imageBuffer;
69#endif
70
71 bool m_shouldCollectGradientAttributes { true };
72};
