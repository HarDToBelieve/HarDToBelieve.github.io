61void RenderSVGResourcePattern::collectPatternAttributes(PatternAttributes& attributes) const
62{
63 const RenderSVGResourcePattern* current = this;
64
65 while (current) {
66 const SVGPatternElement& pattern = current->patternElement();
67 pattern.collectPatternAttributes(attributes);
68
69 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(*current);
70 ASSERT_IMPLIES(resources && resources->linkedResource(), is<RenderSVGResourcePattern>(resources->linkedResource()));
71 current = resources ? downcast<RenderSVGResourcePattern>(resources->linkedResource()) : nullptr;
72 }
73}

75PatternData* RenderSVGResourcePattern::buildPattern(RenderElement& renderer, OptionSet<RenderSVGResourceMode> resourceMode, GraphicsContext& context)
76{
77 ASSERT(!m_shouldCollectPatternAttributes);
78
79 PatternData* currentData = m_patternMap.get(&renderer);
80 if (currentData && currentData->pattern)
81 return currentData;
82
83 // If we couldn't determine the pattern content element root, stop here.
84 if (!m_attributes.patternContentElement())
85 return nullptr;
86
87 // An empty viewBox disables rendering.
88 if (m_attributes.hasViewBox() && m_attributes.viewBox().isEmpty())
89 return nullptr;
90
91 // Compute all necessary transformations to build the tile image & the pattern.
92 FloatRect tileBoundaries;
93 AffineTransform tileImageTransform;
94 if (!buildTileImageTransform(renderer, m_attributes, patternElement(), tileBoundaries, tileImageTransform))
95 return nullptr;
96
97 AffineTransform absoluteTransformIgnoringRotation = SVGRenderingContext::calculateTransformationToOutermostCoordinateSystem(renderer);
98
99 // Ignore 2D rotation, as it doesn't affect the size of the tile.
100 SVGRenderingContext::clear2DRotation(absoluteTransformIgnoringRotation);
101 FloatRect absoluteTileBoundaries = absoluteTransformIgnoringRotation.mapRect(tileBoundaries);
102 FloatRect clampedAbsoluteTileBoundaries;
103
104 // Scale the tile size to match the scale level of the patternTransform.
105 absoluteTileBoundaries.scale(static_cast<float>(m_attributes.patternTransform().xScale()),
106 static_cast<float>(m_attributes.patternTransform().yScale()));
107
108 // Build tile image.
109 auto tileImage = createTileImage(m_attributes, tileBoundaries, absoluteTileBoundaries, tileImageTransform, clampedAbsoluteTileBoundaries, context.renderingMode());
110 if (!tileImage)
111 return nullptr;
112
113 const IntSize tileImageSize = tileImage->logicalSize();
114
115 auto copiedImage = ImageBuffer::sinkIntoNativeImage(WTFMove(tileImage));
116 if (!copiedImage)
117 return nullptr;
118
119 // Compute pattern space transformation.
120 auto patternData = makeUnique<PatternData>();
121 patternData->transform.translate(tileBoundaries.location());
122 patternData->transform.scale(tileBoundaries.size() / tileImageSize);
123
124 AffineTransform patternTransform = m_attributes.patternTransform();
125 if (!patternTransform.isIdentity())
126 patternData->transform = patternTransform * patternData->transform;
127
128 // Account for text drawing resetting the context to non-scaled, see SVGInlineTextBox::paintTextWithShadows.
129 if (resourceMode.contains(RenderSVGResourceMode::ApplyToText)) {
130 AffineTransform additionalTextTransformation;
131 if (shouldTransformOnTextPainting(renderer, additionalTextTransformation))
132 patternData->transform *= additionalTextTransformation;
133 }
134
135 // Build pattern.
136 patternData->pattern = Pattern::create(copiedImage.releaseNonNull(), { true, true, patternData->transform });
137
138 // Various calls above may trigger invalidations in some fringe cases (ImageBuffer allocation
139 // failures in the SVG image cache for example). To avoid having our PatternData deleted by
140 // removeAllClientsFromCache(), we only make it visible in the cache at the very end.
141 return m_patternMap.set(&renderer, WTFMove(patternData)).iterator->value.get();
142}

144bool RenderSVGResourcePattern::applyResource(RenderElement& renderer, const RenderStyle& style, GraphicsContext*& context, OptionSet<RenderSVGResourceMode> resourceMode)
145{
146 ASSERT(context);
147 ASSERT(!resourceMode.isEmpty());
148
149 if (m_shouldCollectPatternAttributes) {
150 patternElement().synchronizeAllAttributes();
151
152 m_attributes = PatternAttributes();
153 collectPatternAttributes(m_attributes);
154 m_shouldCollectPatternAttributes = false;
155 }
156
157 // Spec: When the geometry of the applicable element has no width or height and objectBoundingBox is specified,
158 // then the given effect (e.g. a gradient or a filter) will be ignored.
159 FloatRect objectBoundingBox = renderer.objectBoundingBox();
160 if (m_attributes.patternUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX && objectBoundingBox.isEmpty())
161 return false;
162
163 PatternData* patternData = buildPattern(renderer, resourceMode, *context);
164 if (!patternData)
165 return false;
166
167 // Draw pattern
168 context->save();
169
170 const SVGRenderStyle& svgStyle = style.svgStyle();
171
172 if (resourceMode.contains(RenderSVGResourceMode::ApplyToFill)) {
173 context->setAlpha(svgStyle.fillOpacity());
174 context->setFillPattern(*patternData->pattern);
175 context->setFillRule(svgStyle.fillRule());
176 } else if (resourceMode.contains(RenderSVGResourceMode::ApplyToStroke)) {
177 if (svgStyle.vectorEffect() == VectorEffect::NonScalingStroke)
178 patternData->pattern->setPatternSpaceTransform(transformOnNonScalingStroke(&renderer, patternData->transform));
179 context->setAlpha(svgStyle.strokeOpacity());
180 context->setStrokePattern(*patternData->pattern);
181 SVGRenderSupport::applyStrokeStyleToContext(*context, style, renderer);
182 }
183
184 if (resourceMode.contains(RenderSVGResourceMode::ApplyToText)) {
185 if (resourceMode.contains(RenderSVGResourceMode::ApplyToFill)) {
186 context->setTextDrawingMode(TextDrawingMode::Fill);
187
188#if USE(CG)
189 context->applyFillPattern();
190#endif
191 } else if (resourceMode.contains(RenderSVGResourceMode::ApplyToStroke)) {
192 context->setTextDrawingMode(TextDrawingMode::Stroke);
193
194#if USE(CG)
195 context->applyStrokePattern();
196#endif
197 }
198 }
199
200 return true;
201}

203void RenderSVGResourcePattern::postApplyResource(RenderElement&, GraphicsContext*& context, OptionSet<RenderSVGResourceMode> resourceMode, const Path* path, const RenderSVGShape* shape)
204{
205 ASSERT(context);
206 ASSERT(!resourceMode.isEmpty());
207
208 if (resourceMode.contains(RenderSVGResourceMode::ApplyToFill)) {
209 if (path)
210 context->fillPath(*path);
211 else if (shape)
212 shape->fillShape(*context);
213 }
214 if (resourceMode.contains(RenderSVGResourceMode::ApplyToStroke)) {
215 if (path)
216 context->strokePath(*path);
217 else if (shape)
218 shape->strokeShape(*context);
219 }
220
221 context->restore();
222}

224static inline FloatRect calculatePatternBoundaries(const PatternAttributes& attributes,
225 const FloatRect& objectBoundingBox,
226 const SVGPatternElement& patternElement)
227{
228 return SVGLengthContext::resolveRectangle(&patternElement, attributes.patternUnits(), objectBoundingBox, attributes.x(), attributes.y(), attributes.width(), attributes.height());
229}

231bool RenderSVGResourcePattern::buildTileImageTransform(RenderElement& renderer,
232 const PatternAttributes& attributes,
233 const SVGPatternElement& patternElement,
234 FloatRect& patternBoundaries,
235 AffineTransform& tileImageTransform) const
236{
237 FloatRect objectBoundingBox = renderer.objectBoundingBox();
238 patternBoundaries = calculatePatternBoundaries(attributes, objectBoundingBox, patternElement);
239 if (patternBoundaries.width() <= 0 || patternBoundaries.height() <= 0)
240 return false;
241
242 AffineTransform viewBoxCTM = SVGFitToViewBox::viewBoxToViewTransform(attributes.viewBox(), attributes.preserveAspectRatio(), patternBoundaries.width(), patternBoundaries.height());
243
244 // Apply viewBox/objectBoundingBox transformations.
245 if (!viewBoxCTM.isIdentity())
246 tileImageTransform = viewBoxCTM;
247 else if (attributes.patternContentUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX)
248 tileImageTransform.scale(objectBoundingBox.width(), objectBoundingBox.height());
249
250 return true;
251}

253RefPtr<ImageBuffer> RenderSVGResourcePattern::createTileImage(const PatternAttributes& attributes, const FloatRect& tileBoundaries, const FloatRect& absoluteTileBoundaries, const AffineTransform& tileImageTransform, FloatRect& clampedAbsoluteTileBoundaries, RenderingMode renderingMode) const
254{
255 clampedAbsoluteTileBoundaries = ImageBuffer::clampedRect(absoluteTileBoundaries);
256 auto tileImage = SVGRenderingContext::createImageBuffer(absoluteTileBoundaries, clampedAbsoluteTileBoundaries, DestinationColorSpace::SRGB(), renderingMode);
257 if (!tileImage)
258 return nullptr;
259
260 GraphicsContext& tileImageContext = tileImage->context();
261
262 // The image buffer represents the final rendered size, so the content has to be scaled (to avoid pixelation).
263 tileImageContext.scale(clampedAbsoluteTileBoundaries.size() / tileBoundaries.size());
264
265 // Apply tile image transformations.
266 if (!tileImageTransform.isIdentity())
267 tileImageContext.concatCTM(tileImageTransform);
268
269 AffineTransform contentTransformation;
270 if (attributes.patternContentUnits() == SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX)
271 contentTransformation = tileImageTransform;
272
273 // Draw the content into the ImageBuffer.
274 for (auto& child : childrenOfType<SVGElement>(*attributes.patternContentElement())) {
275 if (!child.renderer())
276 continue;
277 if (child.renderer()->needsLayout())
278 return nullptr;
279 SVGRenderingContext::renderSubtreeToContext(tileImageContext, *child.renderer(), contentTransformation);
280 }
281
282 return tileImage;
283}
