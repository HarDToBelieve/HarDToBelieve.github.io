39 FloatPoint centerPoint(const RadialGradientAttributes&) const;

44private:

48 AffineTransform gradientTransform() const final { return m_attributes.gradientTransform(); }
49 Ref<Gradient> buildGradient(const RenderStyle&) const final;

51 void gradientElement() const = delete;

53 const char* renderName() const final { return "RenderSVGResourceRadialGradient"; }
54 bool collectGradientAttributes() final;

56 RadialGradientAttributes m_attributes;
