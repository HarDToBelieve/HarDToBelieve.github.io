71void RenderSVGRoot::computeIntrinsicRatioInformation(FloatSize& intrinsicSize, double& intrinsicRatio) const
72{
73 ASSERT(!shouldApplySizeContainment(*this));
74
75 // Spec: http://www.w3.org/TR/SVG/coords.html#IntrinsicSizing
76 // SVG needs to specify how to calculate some intrinsic sizing properties to enable inclusion within other languages.
77
78 // The intrinsic aspect ratio of the viewport of SVG content is necessary for example, when including SVG from an ‘object’
79 // element in HTML styled with CSS. It is possible (indeed, common) for an SVG graphic to have an intrinsic aspect ratio but
80 // not to have an intrinsic width or height. The intrinsic aspect ratio must be calculated based upon the following rules:
81 // - The aspect ratio is calculated by dividing a width by a height.
82 // - If the ‘width’ and ‘height’ of the rootmost ‘svg’ element are both specified with unit identifiers (in, mm, cm, pt, pc,
83 // px, em, ex) or in user units, then the aspect ratio is calculated from the ‘width’ and ‘height’ attributes after
84 // resolving both values to user units.
85 intrinsicSize.hasIntrinsicWidth = svgSVGElement().hasIntrinsicWidth();
86 intrinsicSize.hasIntrinsicHeight = svgSVGElement().hasIntrinsicHeight();
87 intrinsicSize.setWidth(floatValueForLength(svgSVGElement().intrinsicWidth(), 0));
88 intrinsicSize.setHeight(floatValueForLength(svgSVGElement().intrinsicHeight(), 0));
89
90 if (style().aspectRatioType() == AspectRatioType::Ratio) {
91 intrinsicRatio = style().logicalAspectRatio();
92 return;
93 }
94
95 std::optional<double> intrinsicRatioValue;
96 if (!intrinsicSize.isEmpty())
97 intrinsicRatioValue = intrinsicSize.width() / static_cast<double>(intrinsicSize.height());
98 else {
99 // - If either/both of the ‘width’ and ‘height’ of the rootmost ‘svg’ element are in percentage units (or omitted), the
100 // aspect ratio is calculated from the width and height values of the ‘viewBox’ specified for the current SVG document
101 // fragment. If the ‘viewBox’ is not correctly specified, or set to 'none', the intrinsic aspect ratio cannot be
102 // calculated and is considered unspecified.
103 FloatSize viewBoxSize = svgSVGElement().viewBox().size();
104 if (!viewBoxSize.isEmpty()) {
105 // The viewBox can only yield an intrinsic ratio, not an intrinsic size.
106 intrinsicRatioValue = viewBoxSize.width() / static_cast<double>(viewBoxSize.height());
107 }
108 }
109
110 if (intrinsicRatioValue)
111 intrinsicRatio = *intrinsicRatioValue;
112 else if (style().aspectRatioType() == AspectRatioType::AutoAndRatio)
113 intrinsicRatio = style().logicalAspectRatio();
114}

121bool RenderSVGRoot::isEmbeddedThroughFrameContainingSVGDocument() const
122{
123 // If our frame has an owner renderer, we're embedded through eg. object/embed/iframe,
124 // but we only negotiate if we're in an SVG document.
125 if (!frame().ownerRenderer())
126 return false;
127 return frame().document()->isSVGDocument();
128}

130LayoutUnit RenderSVGRoot::computeReplacedLogicalWidth(ShouldComputePreferred shouldComputePreferred) const
131{
132 // When we're embedded through SVGImage (border-image/background-image/<html:img>/...) we're forced to resize to a specific size.
133 if (!m_containerSize.isEmpty())
134 return m_containerSize.width();
135
136 if (isEmbeddedThroughFrameContainingSVGDocument())
137 return containingBlock()->availableLogicalWidth();
138
139 // SVG embedded via SVGImage (background-image/border-image/etc) / Inline SVG.
140 return RenderReplaced::computeReplacedLogicalWidth(shouldComputePreferred);
141}

143LayoutUnit RenderSVGRoot::computeReplacedLogicalHeight(std::optional<LayoutUnit> estimatedUsedWidth) const
144{
145 // When we're embedded through SVGImage (border-image/background-image/<html:img>/...) we're forced to resize to a specific size.
146 if (!m_containerSize.isEmpty())
147 return m_containerSize.height();
148
149 if (isEmbeddedThroughFrameContainingSVGDocument())
150 return containingBlock()->availableLogicalHeight(IncludeMarginBorderPadding);
151
152 // SVG embedded via SVGImage (background-image/border-image/etc) / Inline SVG.
153 return RenderReplaced::computeReplacedLogicalHeight(estimatedUsedWidth);
154}

156void RenderSVGRoot::layout()
157{
158 SetForScope<bool> change(m_inLayout, true);
159 StackStats::LayoutCheckPoint layoutCheckPoint;
160 ASSERT(needsLayout());
161
162 m_resourcesNeedingToInvalidateClients.clear();
163
164 // Arbitrary affine transforms are incompatible with RenderLayoutState.
165 LayoutStateDisabler layoutStateDisabler(view().frameView().layoutContext());
166
167 bool needsLayout = selfNeedsLayout();
168 LayoutRepainter repainter(*this, checkForRepaintDuringLayout() && needsLayout);
169
170 LayoutSize oldSize = size();
171 updateLogicalWidth();
172 updateLogicalHeight();
173 buildLocalToBorderBoxTransform();
174
175 m_isLayoutSizeChanged = needsLayout || (svgSVGElement().hasRelativeLengths() && oldSize != size());
176 SVGRenderSupport::layoutChildren(*this, needsLayout || SVGRenderSupport::filtersForceContainerLayout(*this));
177
178 if (!m_resourcesNeedingToInvalidateClients.isEmpty()) {
179 // Invalidate resource clients, which may mark some nodes for layout.
180 for (auto& resource : m_resourcesNeedingToInvalidateClients) {
181 resource->removeAllClientsFromCache();
182 SVGResourcesCache::clientStyleChanged(*resource, StyleDifference::Layout, resource->style());
183 }
184
185 m_isLayoutSizeChanged = false;
186 SVGRenderSupport::layoutChildren(*this, false);
187 }
188
189 // At this point LayoutRepainter already grabbed the old bounds,
190 // recalculate them now so repaintAfterLayout() uses the new bounds.
191 if (m_needsBoundariesOrTransformUpdate) {
192 updateCachedBoundaries();
193 m_needsBoundariesOrTransformUpdate = false;
194 }
195
196 clearOverflow();
197 if (!shouldApplyViewportClip()) {
198 FloatRect contentRepaintRect = repaintRectInLocalCoordinates();
199 contentRepaintRect = m_localToBorderBoxTransform.mapRect(contentRepaintRect);
200 addVisualOverflow(enclosingLayoutRect(contentRepaintRect));
201 }
202
203 updateLayerTransform();
204 m_hasBoxDecorations = isDocumentElementRenderer() ? hasVisibleBoxDecorationStyle() : hasVisibleBoxDecorations();
205 invalidateBackgroundObscurationStatus();
206
207 repainter.repaintAfterLayout();
208
209 clearNeedsLayout();
210}

212bool RenderSVGRoot::shouldApplyViewportClip() const
213{
214 // the outermost svg is clipped if auto, and svg document roots are always clipped
215 // When the svg is stand-alone (isDocumentElement() == true) the viewport clipping should always
216 // be applied, noting that the window scrollbars should be hidden if overflow=hidden.
217 return style().overflowX() == Overflow::Hidden
218 || style().overflowX() == Overflow::Auto
219 || style().overflowX() == Overflow::Scroll
220 || this->isDocumentElementRenderer();
221}

223void RenderSVGRoot::paintReplaced(PaintInfo& paintInfo, const LayoutPoint& paintOffset)
224{
225 // An empty viewport disables rendering.
226 if (borderBoxRect().isEmpty())
227 return;
228
229 // Don't paint, if the context explicitly disabled it.
230 if (paintInfo.context().paintingDisabled() && !paintInfo.context().detectingContentfulPaint())
231 return;
232
233 // SVG outlines are painted during PaintPhase::Foreground.
234 if (paintInfo.phase == PaintPhase::Outline || paintInfo.phase == PaintPhase::SelfOutline)
235 return;
236
237 // An empty viewBox also disables rendering.
238 // (http://www.w3.org/TR/SVG/coords.html#ViewBoxAttribute)
239 if (svgSVGElement().hasEmptyViewBox())
240 return;
241
242 GraphicsContext& context = paintInfo.context();
243 if (context.detectingContentfulPaint()) {
244 for (auto& current : childrenOfType<RenderObject>(*this)) {
245 if (!current.isSVGHiddenContainer()) {
246 context.setContentfulPaintDetected();
247 return;
248 }
249 }
250 return;
251 }
252
253 // Don't paint if we don't have kids, except if we have filters we should paint those.
254 if (!firstChild()) {
255 auto* resources = SVGResourcesCache::cachedResourcesForRenderer(*this);
256 if (!resources || !resources->filter()) {
257 if (paintInfo.phase == PaintPhase::Foreground)
258 page().addRelevantUnpaintedObject(this, visualOverflowRect());
259 return;
260 }
261 }
262
263 if (paintInfo.phase == PaintPhase::Foreground)
264 page().addRelevantRepaintedObject(this, visualOverflowRect());
265
266 // Make a copy of the PaintInfo because applyTransform will modify the damage rect.
267 PaintInfo childPaintInfo(paintInfo);
268 childPaintInfo.context().save();
269
270 // Apply initial viewport clip
271 if (shouldApplyViewportClip())
272 childPaintInfo.context().clip(snappedIntRect(overflowClipRect(paintOffset)));
273
274 // Convert from container offsets (html renderers) to a relative transform (svg renderers).
275 // Transform from our paint container's coordinate system to our local coords.
276 IntPoint adjustedPaintOffset = roundedIntPoint(paintOffset);
277 childPaintInfo.applyTransform(AffineTransform::translation(adjustedPaintOffset.x(), adjustedPaintOffset.y()) * localToBorderBoxTransform());
278
279 // SVGRenderingContext must be destroyed before we restore the childPaintInfo.context(), because a filter may have
280 // changed the context and it is only reverted when the SVGRenderingContext destructor finishes applying the filter.
281 {
282 SVGRenderingContext renderingContext;
283 bool continueRendering = true;
284 if (childPaintInfo.phase == PaintPhase::Foreground) {
285 renderingContext.prepareToRenderSVGContent(*this, childPaintInfo);
286 continueRendering = renderingContext.isRenderingPrepared();
287 }
288
289 if (continueRendering) {
290 childPaintInfo.updateSubtreePaintRootForChildren(this);
291 for (auto& child : childrenOfType<RenderElement>(*this))
292 child.paint(childPaintInfo, location());
293 }
294 }
295
296 childPaintInfo.context().restore();
297}

320{
321 if (diff == StyleDifference::Layout)

323

330}

332// RenderBox methods will expect coordinates w/o any transforms in coordinates
333// relative to our borderBox origin. This method gives us exactly that.
334void RenderSVGRoot::buildLocalToBorderBoxTransform()
335{
336 float scale = style().effectiveZoom();
337 FloatPoint translate = svgSVGElement().currentTranslateValue();
338 LayoutSize borderAndPadding(borderLeft() + paddingLeft(), borderTop() + paddingTop());
339 m_localToBorderBoxTransform = svgSVGElement().viewBoxToViewTransform(contentWidth() / scale, contentHeight() / scale);
340 if (borderAndPadding.isZero() && scale == 1 && translate == FloatPoint::zero())
341 return;
342 m_localToBorderBoxTransform = AffineTransform(scale, 0, 0, scale, borderAndPadding.width() + translate.x(), borderAndPadding.height() + translate.y()) * m_localToBorderBoxTransform;
343}

345const AffineTransform& RenderSVGRoot::localToParentTransform() const
346{
347 // Slightly optimized version of m_localToParentTransform = AffineTransform::translation(x(), y()) * m_localToBorderBoxTransform;
348 m_localToParentTransform = m_localToBorderBoxTransform;
349 if (x())
350 m_localToParentTransform.setE(m_localToParentTransform.e() + roundToInt(x()));
351 if (y())
352 m_localToParentTransform.setF(m_localToParentTransform.f() + roundToInt(y()));
353 return m_localToParentTransform;
354}

356LayoutRect RenderSVGRoot::clippedOverflowRect(const RenderLayerModelObject* repaintContainer, VisibleRectContext context) const
357{
358 if (style().visibility() != Visibility::Visible && !enclosingLayer()->hasVisibleContent())
359 return LayoutRect();
360
361 FloatRect contentRepaintRect = m_localToBorderBoxTransform.mapRect(repaintRectInLocalCoordinates());
362 contentRepaintRect.intersect(snappedIntRect(borderBoxRect()));
363
364 LayoutRect repaintRect = enclosingLayoutRect(contentRepaintRect);
365 if (m_hasBoxDecorations || hasRenderOverflow())
366 repaintRect.unite(unionRect(localSelectionRect(false), visualOverflowRect()));
367
368 return RenderReplaced::computeRect(enclosingIntRect(repaintRect), repaintContainer, context);
369}

371std::optional<FloatRect> RenderSVGRoot::computeFloatVisibleRectInContainer(const FloatRect& rect, const RenderLayerModelObject* container, VisibleRectContext context) const
372{
373 // Apply our local transforms (except for x/y translation), then our shadow,
374 // and then call RenderBox's method to handle all the normal CSS Box model bits
375 FloatRect adjustedRect = m_localToBorderBoxTransform.mapRect(rect);
376
377 const SVGRenderStyle& svgStyle = style().svgStyle();
378 if (const ShadowData* shadow = svgStyle.shadow())
379 shadow->adjustRectForShadow(adjustedRect);
380
381 // Apply initial viewport clip
382 if (shouldApplyViewportClip()) {
383 if (context.options.contains(VisibleRectContextOption::UseEdgeInclusiveIntersection)) {
384 if (!adjustedRect.edgeInclusiveIntersect(snappedIntRect(borderBoxRect())))
385 return std::nullopt;
386 } else
387 adjustedRect.intersect(snappedIntRect(borderBoxRect()));
388 }
389
390 if (m_hasBoxDecorations || hasRenderOverflow()) {
391 // The selectionRect can project outside of the overflowRect, so take their union
392 // for repainting to avoid selection painting glitches.
393 LayoutRect decoratedRepaintRect = unionRect(localSelectionRect(false), visualOverflowRect());
394 adjustedRect.unite(decoratedRepaintRect);
395 }
396
397 if (std::optional<LayoutRect> rectInContainer = RenderReplaced::computeVisibleRectInContainer(enclosingIntRect(adjustedRect), container, context))
398 return FloatRect(*rectInContainer);
399 return std::nullopt;
400}

427bool RenderSVGRoot::nodeAtPoint(const HitTestRequest& request, HitTestResult& result, const HitTestLocation& locationInContainer, const LayoutPoint& accumulatedOffset, HitTestAction hitTestAction)
428{
429 LayoutPoint pointInParent = locationInContainer.point() - toLayoutSize(accumulatedOffset);
430 LayoutPoint pointInBorderBox = pointInParent - toLayoutSize(location());
431
432 ASSERT(SVGHitTestCycleDetectionScope::isEmpty());
433
434 // Test SVG content if the point is in our content box or it is inside the visualOverflowRect and the overflow is visible.
435 // FIXME: This should be an intersection when rect-based hit tests are supported by nodeAtFloatPoint.
436 if (contentBoxRect().contains(pointInBorderBox) || (!shouldApplyViewportClip() && visualOverflowRect().contains(pointInParent))) {
437 FloatPoint localPoint = localToParentTransform().inverse().value_or(AffineTransform()).mapPoint(FloatPoint(pointInParent));
438
439 for (RenderObject* child = lastChild(); child; child = child->previousSibling()) {
440 // FIXME: nodeAtFloatPoint() doesn't handle rect-based hit tests yet.
441 if (child->nodeAtFloatPoint(request, result, localPoint, hitTestAction)) {
442 updateHitTestResult(result, pointInBorderBox);
443 if (result.addNodeToListBasedTestResult(child->node(), request, locationInContainer) == HitTestProgress::Stop) {
444 ASSERT(SVGHitTestCycleDetectionScope::isEmpty());
445 return true;
446 }
447 }
448 }
449 }
450
451 // If we didn't early exit above, we've just hit the container <svg> element. Unlike SVG 1.1, 2nd Edition allows container elements to be hit.
452 if ((hitTestAction == HitTestBlockBackground || hitTestAction == HitTestChildBlockBackground) && visibleToHitTesting(request)) {
453 // Only return true here, if the last hit testing phase 'BlockBackground' is executed. If we'd return true in the 'Foreground' phase,
454 // hit testing would stop immediately. For SVG only trees this doesn't matter. Though when we have a <foreignObject> subtree we need
455 // to be able to detect hits on the background of a <div> element. If we'd return true here in the 'Foreground' phase, we are not able
456 // to detect these hits anymore.
457 LayoutRect boundsRect(accumulatedOffset + location(), size());
458 if (locationInContainer.intersects(boundsRect)) {
459 updateHitTestResult(result, pointInBorderBox);
460 if (result.addNodeToListBasedTestResult(nodeForHitTest(), request, locationInContainer, boundsRect) == HitTestProgress::Stop)
461 return true;
462 }
463 }
464
465 ASSERT(SVGHitTestCycleDetectionScope::isEmpty());
466
467 return false;
468}
