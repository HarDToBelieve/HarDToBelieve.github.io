22
23#include "RenderSVGContainer.h"
24#include "SVGGraphicsElement.h"

26namespace WebCore {
27
28class SVGGraphicsElement;
29class RenderSVGTransformableContainer final : public RenderSVGContainer {
30 WTF_MAKE_ISO_ALLOCATED(RenderSVGTransformableContainer);
31public:
32 RenderSVGTransformableContainer(SVGGraphicsElement&, RenderStyle&&);
33 SVGGraphicsElement& graphicsElement() { return downcast<SVGGraphicsElement>(RenderSVGContainer::element()); }
34
35 bool isSVGTransformableContainer() const override { return true; }
36 const AffineTransform& localToParentTransform() const override { return m_localTransform; }
37 void setNeedsTransformUpdate() override { m_needsTransformUpdate = true; }
38 bool didTransformToRootUpdate() override { return m_didTransformToRootUpdate; }
39
40private:
41 void element() const = delete;
42 bool calculateLocalTransform() override;
43 AffineTransform localTransform() const override { return m_localTransform; }
44
45 bool m_needsTransformUpdate : 1;
46 bool m_didTransformToRootUpdate : 1;
47 AffineTransform m_localTransform;
48 FloatSize m_lastTranslation;
49};
50
51} // namespace WebCore
