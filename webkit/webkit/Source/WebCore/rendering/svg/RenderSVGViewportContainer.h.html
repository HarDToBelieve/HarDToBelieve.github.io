23#pragma once
24
25#include "RenderSVGContainer.h"
26
27namespace WebCore {
28
29// This is used for non-root <svg> elements and <marker> elements, neither of which are SVGTransformable
30// thus we inherit from RenderSVGContainer instead of RenderSVGTransformableContainer
31class RenderSVGViewportContainer final : public RenderSVGContainer {
32 WTF_MAKE_ISO_ALLOCATED(RenderSVGViewportContainer);
33public:
34 RenderSVGViewportContainer(SVGSVGElement&, RenderStyle&&);
35
36 SVGSVGElement& svgSVGElement() const;
37
38 FloatRect viewport() const { return m_viewport; }
39
40 bool isLayoutSizeChanged() const { return m_isLayoutSizeChanged; }
41 bool didTransformToRootUpdate() override { return m_didTransformToRootUpdate; }
42
43 void determineIfLayoutSizeChanged() override;
44 void setNeedsTransformUpdate() override { m_needsTransformUpdate = true; }
45
46 void paint(PaintInfo&, const LayoutPoint&) override;
47
48private:
49 void element() const = delete;
50
51 bool isSVGViewportContainer() const override { return true; }
52 const char* renderName() const override { return "RenderSVGViewportContainer"; }
53
54 AffineTransform viewportTransform() const;
55 const AffineTransform& localToParentTransform() const override { return m_localToParentTransform; }
56
57 void calcViewport() override;
58 bool calculateLocalTransform() override;
59
60 void applyViewportClip(PaintInfo&) override;
61 bool pointIsInsideViewportClip(const FloatPoint& pointInParent) override;
62
63 bool m_didTransformToRootUpdate : 1;
64 bool m_isLayoutSizeChanged : 1;
65 bool m_needsTransformUpdate : 1;
66
67 FloatRect m_viewport;
68 mutable AffineTransform m_localToParentTransform;
69};
70
71} // namespace WebCore
72
73SPECIALIZE_TYPE_TRAITS_RENDER_OBJECT(RenderSVGViewportContainer, isSVGViewportContainer())
