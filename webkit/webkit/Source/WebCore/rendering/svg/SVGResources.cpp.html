47static const MemoryCompactLookupOnlyRobinHoodHashSet<AtomString>& clipperFilterMaskerTags()
48{
49 static NeverDestroyed<MemoryCompactLookupOnlyRobinHoodHashSet<AtomString>> s_tagList;
50 if (s_tagList.get().isEmpty()) {
51 // "container elements": http://www.w3.org/TR/SVG11/intro.html#TermContainerElement
52 // "graphics elements" : http://www.w3.org/TR/SVG11/intro.html#TermGraphicsElement
53 s_tagList.get().add(SVGNames::aTag->localName());
54 s_tagList.get().add(SVGNames::circleTag->localName());
55 s_tagList.get().add(SVGNames::ellipseTag->localName());
56 s_tagList.get().add(SVGNames::glyphTag->localName());
57 s_tagList.get().add(SVGNames::gTag->localName());
58 s_tagList.get().add(SVGNames::imageTag->localName());
59 s_tagList.get().add(SVGNames::lineTag->localName());
60 s_tagList.get().add(SVGNames::markerTag->localName());
61 s_tagList.get().add(SVGNames::maskTag->localName());
62 s_tagList.get().add(SVGNames::missing_glyphTag->localName());
63 s_tagList.get().add(SVGNames::pathTag->localName());
64 s_tagList.get().add(SVGNames::polygonTag->localName());
65 s_tagList.get().add(SVGNames::polylineTag->localName());
66 s_tagList.get().add(SVGNames::rectTag->localName());
67 s_tagList.get().add(SVGNames::svgTag->localName());
68 s_tagList.get().add(SVGNames::textTag->localName());
69 s_tagList.get().add(SVGNames::useTag->localName());
70
71 // Not listed in the definitions is the clipPath element, the SVG spec says though:
72 // The "clipPath" element or any of its children can specify property "clip-path".
73 // So we have to add clipPathTag here, otherwhise clip-path on clipPath will fail.
74 // (Already mailed SVG WG, waiting for a solution)
75 s_tagList.get().add(SVGNames::clipPathTag->localName());
76
77 // Not listed in the definitions are the text content elements, though filter/clipper/masker on tspan/text/.. is allowed.
78 // (Already mailed SVG WG, waiting for a solution)
79 s_tagList.get().add(SVGNames::altGlyphTag->localName());
80 s_tagList.get().add(SVGNames::textPathTag->localName());
81 s_tagList.get().add(SVGNames::trefTag->localName());
82 s_tagList.get().add(SVGNames::tspanTag->localName());
83
84 // Not listed in the definitions is the foreignObject element, but clip-path
85 // is a supported attribute.
86 s_tagList.get().add(SVGNames::foreignObjectTag->localName());
87
88 // Elements that we ignore, as it doesn't make any sense.
89 // defs, pattern, switch (FIXME: Mail SVG WG about these)
90 // symbol (is converted to a svg element, when referenced by use, we can safely ignore it.)
91 }
92
93 return s_tagList;
94}

143static inline String targetReferenceFromResource(SVGElement& element)
144{
145 String target;
146 if (is<SVGPatternElement>(element))
147 target = downcast<SVGPatternElement>(element).href();
148 else if (is<SVGGradientElement>(element))
149 target = downcast<SVGGradientElement>(element).href();
150 else if (is<SVGFilterElement>(element))
151 target = downcast<SVGFilterElement>(element).href();
152 else
153 ASSERT_NOT_REACHED();
154
155 return SVGURIReference::fragmentIdentifierFromIRIString(target, element.document());
156}
157
158static inline bool isChainableResource(const SVGElement& element, const SVGElement& linkedResource)
159{
160 if (is<SVGPatternElement>(element))
161 return is<SVGPatternElement>(linkedResource);
162
163 if (is<SVGGradientElement>(element))
164 return is<SVGGradientElement>(linkedResource);
165
166 if (is<SVGFilterElement>(element))
167 return is<SVGFilterElement>(linkedResource);
168
169 ASSERT_NOT_REACHED();
170 return false;
171}

173static inline RenderSVGResourceContainer* paintingResourceFromSVGPaint(Document& document, const SVGPaintType& paintType, const String& paintUri, AtomString& id, bool& hasPendingResource)
174{
175 if (paintType != SVGPaintType::URI && paintType != SVGPaintType::URIRGBColor && paintType != SVGPaintType::URICurrentColor)
176 return nullptr;
177
178 id = SVGURIReference::fragmentIdentifierFromIRIString(paintUri, document);
179 RenderSVGResourceContainer* container = getRenderSVGResourceContainerById(document, id);
180 if (!container) {
181 hasPendingResource = true;
182 return nullptr;
183 }
184
185 RenderSVGResourceType resourceType = container->resourceType();
186 if (resourceType != PatternResourceType && resourceType != LinearGradientResourceType && resourceType != RadialGradientResourceType)
187 return nullptr;
188
189 return container;
190}

192static inline void registerPendingResource(SVGDocumentExtensions& extensions, const AtomString& id, SVGElement& element)
193{
194 extensions.addPendingResource(id, element);

197bool SVGResources::buildCachedResources(const RenderElement& renderer, const RenderStyle& style)
198{
199 ASSERT(renderer.element());
200 ASSERT_WITH_SECURITY_IMPLICATION(renderer.element()->isSVGElement());
201
202 if (!renderer.element())
203 return false;
204
205 auto& element = downcast<SVGElement>(*renderer.element());
206
207 Document& document = element.document();
208
209 SVGDocumentExtensions& extensions = document.accessSVGExtensions();
210
211 const AtomString& tagName = element.localName();
212 if (tagName.isNull())
213 return false;
214
215 const SVGRenderStyle& svgStyle = style.svgStyle();
216
217 bool foundResources = false;
218 if (clipperFilterMaskerTags().contains(tagName)) {
219 if (is<ReferenceClipPathOperation>(style.clipPath())) {
220 // FIXME: -webkit-clip-path should support external resources
221 // https://bugs.webkit.org/show_bug.cgi?id=127032
222 auto& clipPath = downcast<ReferenceClipPathOperation>(*style.clipPath());
223 AtomString id(clipPath.fragment());
224 if (setClipper(getRenderSVGResourceById<RenderSVGResourceClipper>(document, id)))
225 foundResources = true;
226 else
227 registerPendingResource(extensions, id, element);
228 }
229
230 if (style.hasFilter()) {
231 const FilterOperations& filterOperations = style.filter();
232 if (filterOperations.size() == 1) {
233 const FilterOperation& filterOperation = *filterOperations.at(0);
234 if (filterOperation.type() == FilterOperation::REFERENCE) {
235 const auto& referenceFilterOperation = downcast<ReferenceFilterOperation>(filterOperation);
236 AtomString id = SVGURIReference::fragmentIdentifierFromIRIString(referenceFilterOperation.url(), element.document());
237 if (setFilter(getRenderSVGResourceById<RenderSVGResourceFilter>(document, id)))
238 foundResources = true;
239 else
240 registerPendingResource(extensions, id, element);
241 }
242 }
243 }
244
245 if (svgStyle.hasMasker()) {
246 AtomString id(svgStyle.maskerResource());
247 if (setMasker(getRenderSVGResourceById<RenderSVGResourceMasker>(document, id)))
248 foundResources = true;
249 else
250 registerPendingResource(extensions, id, element);
251 }
252 }
253
254 if (markerTags().contains(tagName) && svgStyle.hasMarkers()) {
255 AtomString markerStartId(svgStyle.markerStartResource());
256 if (setMarkerStart(getRenderSVGResourceById<RenderSVGResourceMarker>(document, markerStartId)))
257 foundResources = true;
258 else
259 registerPendingResource(extensions, markerStartId, element);
260
261 AtomString markerMidId(svgStyle.markerMidResource());
262 if (setMarkerMid(getRenderSVGResourceById<RenderSVGResourceMarker>(document, markerMidId)))
263 foundResources = true;
264 else
265 registerPendingResource(extensions, markerMidId, element);
266
267 AtomString markerEndId(svgStyle.markerEndResource());
268 if (setMarkerEnd(getRenderSVGResourceById<RenderSVGResourceMarker>(document, markerEndId)))
269 foundResources = true;
270 else
271 registerPendingResource(extensions, markerEndId, element);
272 }
273
274 if (fillAndStrokeTags().contains(tagName)) {
275 if (svgStyle.hasFill()) {
276 bool hasPendingResource = false;
277 AtomString id;
278 if (setFill(paintingResourceFromSVGPaint(document, svgStyle.fillPaintType(), svgStyle.fillPaintUri(), id, hasPendingResource)))
279 foundResources = true;
280 else if (hasPendingResource)
281 registerPendingResource(extensions, id, element);
282 }
283
284 if (svgStyle.hasStroke()) {
285 bool hasPendingResource = false;
286 AtomString id;
287 if (setStroke(paintingResourceFromSVGPaint(document, svgStyle.strokePaintType(), svgStyle.strokePaintUri(), id, hasPendingResource)))
288 foundResources = true;
289 else if (hasPendingResource)
290 registerPendingResource(extensions, id, element);
291 }
292 }
293
294 if (chainableResourceTags().contains(tagName)) {
295 AtomString id(targetReferenceFromResource(element));
296 auto* linkedResource = getRenderSVGResourceContainerById(document, id);
297 if (!linkedResource)
298 registerPendingResource(extensions, id, element);
299 else if (isChainableResource(element, linkedResource->element())) {
300 setLinkedResource(linkedResource);
301 foundResources = true;
302 }
303 }
304
305 return foundResources;
306}

308void SVGResources::layoutDifferentRootIfNeeded(const RenderSVGRoot* svgRoot)
309{
310 if (clipper() && svgRoot != SVGRenderSupport::findTreeRootObject(*clipper()))
311 clipper()->layoutIfNeeded();
312
313 if (masker() && svgRoot != SVGRenderSupport::findTreeRootObject(*masker()))
314 masker()->layoutIfNeeded();
315
316 if (filter() && svgRoot != SVGRenderSupport::findTreeRootObject(*filter()))
317 filter()->layoutIfNeeded();
318
319 if (markerStart() && svgRoot != SVGRenderSupport::findTreeRootObject(*markerStart()))
320 markerStart()->layoutIfNeeded();
321
322 if (markerMid() && svgRoot != SVGRenderSupport::findTreeRootObject(*markerMid()))
323 markerMid()->layoutIfNeeded();
324
325 if (markerEnd() && svgRoot != SVGRenderSupport::findTreeRootObject(*markerEnd()))
326 markerEnd()->layoutIfNeeded();
327}

336void SVGResources::removeClientFromCache(RenderElement& renderer, bool markForInvalidation) const
337{
338 if (isEmpty())
339 return;
340
341 if (m_linkedResource) {
342 ASSERT(!m_clipperFilterMaskerData);
343 ASSERT(!m_markerData);
344 ASSERT(!m_fillStrokeData);
345 m_linkedResource->removeClientFromCache(renderer, markForInvalidation);
346 return;
347 }
348
349 if (m_clipperFilterMaskerData) {
350 if (m_clipperFilterMaskerData->clipper)
351 m_clipperFilterMaskerData->clipper->removeClientFromCache(renderer, markForInvalidation);
352 if (m_clipperFilterMaskerData->filter)
353 m_clipperFilterMaskerData->filter->removeClientFromCache(renderer, markForInvalidation);
354 if (m_clipperFilterMaskerData->masker)
355 m_clipperFilterMaskerData->masker->removeClientFromCache(renderer, markForInvalidation);
356 }
357
358 if (m_markerData) {
359 if (m_markerData->markerStart)
360 m_markerData->markerStart->removeClientFromCache(renderer, markForInvalidation);
361 if (m_markerData->markerMid)
362 m_markerData->markerMid->removeClientFromCache(renderer, markForInvalidation);
363 if (m_markerData->markerEnd)
364 m_markerData->markerEnd->removeClientFromCache(renderer, markForInvalidation);
365 }
366
367 if (m_fillStrokeData) {
368 if (m_fillStrokeData->fill)
369 m_fillStrokeData->fill->removeClientFromCache(renderer, markForInvalidation);
370 if (m_fillStrokeData->stroke)
371 m_fillStrokeData->stroke->removeClientFromCache(renderer, markForInvalidation);
372 }
373}

375bool SVGResources::resourceDestroyed(RenderSVGResourceContainer& resource)
376{
377 if (isEmpty())
378 return false;
379
380 if (m_linkedResource == &resource) {
381 ASSERT(!m_clipperFilterMaskerData);
382 ASSERT(!m_markerData);
383 ASSERT(!m_fillStrokeData);
384 m_linkedResource->removeAllClientsFromCache();
385 m_linkedResource = nullptr;
386 return true;
387 }
388
389 bool foundResources = false;
390 switch (resource.resourceType()) {
391 case MaskerResourceType:
392 if (!m_clipperFilterMaskerData)
393 break;
394 if (m_clipperFilterMaskerData->masker == &resource) {
395 m_clipperFilterMaskerData->masker->removeAllClientsFromCache();
396 m_clipperFilterMaskerData->masker = nullptr;
397 foundResources = true;
398 }
399 break;
400 case MarkerResourceType:
401 if (!m_markerData)
402 break;
403 if (m_markerData->markerStart == &resource) {
404 m_markerData->markerStart->removeAllClientsFromCache();
405 m_markerData->markerStart = nullptr;
406 foundResources = true;
407 }
408 if (m_markerData->markerMid == &resource) {
409 m_markerData->markerMid->removeAllClientsFromCache();
410 m_markerData->markerMid = nullptr;
411 foundResources = true;
412 }
413 if (m_markerData->markerEnd == &resource) {
414 m_markerData->markerEnd->removeAllClientsFromCache();
415 m_markerData->markerEnd = nullptr;
416 foundResources = true;
417 }
418 break;
419 case PatternResourceType:
420 case LinearGradientResourceType:
421 case RadialGradientResourceType:
422 if (!m_fillStrokeData)
423 break;
424 if (m_fillStrokeData->fill == &resource) {
425 m_fillStrokeData->fill->removeAllClientsFromCache();
426 m_fillStrokeData->fill = nullptr;
427 foundResources = true;
428 }
429 if (m_fillStrokeData->stroke == &resource) {
430 m_fillStrokeData->stroke->removeAllClientsFromCache();
431 m_fillStrokeData->stroke = nullptr;
432 foundResources = true;
433 }
434 break;
435 case FilterResourceType:
436 if (!m_clipperFilterMaskerData)
437 break;
438 if (m_clipperFilterMaskerData->filter == &resource) {
439 m_clipperFilterMaskerData->filter->removeAllClientsFromCache();
440 m_clipperFilterMaskerData->filter = nullptr;
441 foundResources = true;
442 }
443 break;
444 case ClipperResourceType:
445 if (!m_clipperFilterMaskerData)
446 break;
447 if (m_clipperFilterMaskerData->clipper == &resource) {
448 m_clipperFilterMaskerData->clipper->removeAllClientsFromCache();
449 m_clipperFilterMaskerData->clipper = nullptr;
450 foundResources = true;
451 }
452 break;
453 case SolidColorResourceType:
454 ASSERT_NOT_REACHED();
455 }
456 return foundResources;
457}

459void SVGResources::buildSetOfResources(HashSet<RenderSVGResourceContainer*>& set)
460{
461 if (isEmpty())
462 return;
463
464 if (m_linkedResource) {
465 ASSERT(!m_clipperFilterMaskerData);
466 ASSERT(!m_markerData);
467 ASSERT(!m_fillStrokeData);
468 set.add(m_linkedResource);
469 return;
470 }
471
472 if (m_clipperFilterMaskerData) {
473 if (m_clipperFilterMaskerData->clipper)
474 set.add(m_clipperFilterMaskerData->clipper);
475 if (m_clipperFilterMaskerData->filter)
476 set.add(m_clipperFilterMaskerData->filter);
477 if (m_clipperFilterMaskerData->masker)
478 set.add(m_clipperFilterMaskerData->masker);
479 }
480
481 if (m_markerData) {
482 if (m_markerData->markerStart)
483 set.add(m_markerData->markerStart);
484 if (m_markerData->markerMid)
485 set.add(m_markerData->markerMid);
486 if (m_markerData->markerEnd)
487 set.add(m_markerData->markerEnd);
488 }
489
490 if (m_fillStrokeData) {
491 if (m_fillStrokeData->fill)
492 set.add(m_fillStrokeData->fill);
493 if (m_fillStrokeData->stroke)
494 set.add(m_fillStrokeData->stroke);
495 }
496}

686void SVGResources::dump(const RenderObject* object)
687{
688 ASSERT(object);
689 ASSERT(object->node());
690
691 fprintf(stderr, "-> this=%p, SVGResources(renderer=%p, node=%p)\n", this, object, object->node());
692 fprintf(stderr, " | DOM Tree:\n");
693 object->node()->showTreeForThis();
694
695 fprintf(stderr, "\n | List of resources:\n");
