20#pragma once
21
22#include "SVGTextMetrics.h"
23#include <wtf/HashMap.h>
24#include <wtf/Noncopyable.h>
25#include <wtf/Vector.h>
26
27namespace WebCore {
28
29class RenderSVGInlineText;
30
31struct SVGCharacterData {
32 SVGCharacterData();
33
34 float x;
35 float y;
36 float dx;
37 float dy;
38 float rotate;
39};
40
41typedef HashMap<unsigned, SVGCharacterData> SVGCharacterDataMap;
42
43class SVGTextLayoutAttributes {
44 WTF_MAKE_NONCOPYABLE(SVGTextLayoutAttributes);
45public:
46 explicit SVGTextLayoutAttributes(RenderSVGInlineText&);
47
48 void clear();
49 void dump() const;
50 static float emptyValue();
51
52 RenderSVGInlineText& context() { return m_context; }
53 const RenderSVGInlineText& context() const { return m_context; }
54
55 SVGCharacterDataMap& characterDataMap() { return m_characterDataMap; }
56 const SVGCharacterDataMap& characterDataMap() const { return m_characterDataMap; }
57
58 Vector<SVGTextMetrics>& textMetricsValues() { return m_textMetricsValues; }
59
60private:
61 RenderSVGInlineText& m_context;
62 SVGCharacterDataMap m_characterDataMap;
63 Vector<SVGTextMetrics> m_textMetricsValues;
64};
65
66inline SVGCharacterData::SVGCharacterData()
67 : x(SVGTextLayoutAttributes::emptyValue())
68 , y(SVGTextLayoutAttributes::emptyValue())
69 , dx(SVGTextLayoutAttributes::emptyValue())
70 , dy(SVGTextLayoutAttributes::emptyValue())
71 , rotate(SVGTextLayoutAttributes::emptyValue())
72{
73}
