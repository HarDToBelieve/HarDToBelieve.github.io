22#include "SVGTextMetricsBuilder.h"

24namespace WebCore {
25
26class RenderBoxModelObject;
27class RenderObject;
28class RenderSVGInlineText;
29class RenderSVGText;
30class SVGTextPositioningElement;
31
32// SVGTextLayoutAttributesBuilder performs the first layout phase for SVG text.
33//
34// It extracts the x/y/dx/dy/rotate values from the SVGTextPositioningElements in the DOM.
35// These values are propagated to the corresponding RenderSVGInlineText renderers.
36// The first layout phase only extracts the relevant information needed in RenderBlockLineLayout
37// to create the InlineBox tree based on text chunk boundaries & BiDi information.
38// The second layout phase is carried out by SVGTextLayoutEngine.
39
40class SVGTextLayoutAttributesBuilder {
41 WTF_MAKE_NONCOPYABLE(SVGTextLayoutAttributesBuilder);
42public:
43 SVGTextLayoutAttributesBuilder();
44 bool buildLayoutAttributesForForSubtree(RenderSVGText&);
45 void buildLayoutAttributesForTextRenderer(RenderSVGInlineText&);
46
47 void rebuildMetricsForTextRenderer(RenderSVGInlineText&);
48
49 // Invoked whenever the underlying DOM tree changes, so that m_textPositions is rebuild.
50 void clearTextPositioningElements() { m_textPositions.clear(); }
51 unsigned numberOfTextPositioningElements() const { return m_textPositions.size(); }
52
53private:
54 struct TextPosition {
55 TextPosition(SVGTextPositioningElement* newElement = 0, unsigned newStart = 0, unsigned newLength = 0)
56 : element(newElement)
57 , start(newStart)
58 , length(newLength)
59 {
60 }
61
62 SVGTextPositioningElement* element;
63 unsigned start;
64 unsigned length;
65 };
66
67 void buildCharacterDataMap(RenderSVGText&);
68 void collectTextPositioningElements(RenderBoxModelObject&, bool& lastCharacterWasSpace);
69 void fillCharacterDataMap(const TextPosition&);
70
71private:
72 unsigned m_textLength;
73 Vector<TextPosition> m_textPositions;
74 SVGCharacterDataMap m_characterDataMap;
75 SVGTextMetricsBuilder m_metricsBuilder;
76};
