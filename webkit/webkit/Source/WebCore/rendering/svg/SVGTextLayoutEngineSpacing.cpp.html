31SVGTextLayoutEngineSpacing::SVGTextLayoutEngineSpacing(const FontCascade& font)
32 : m_font(font)
33 , m_lastCharacter(0)
34{
35}
36
37float SVGTextLayoutEngineSpacing::calculateCSSKerningAndSpacing(const SVGRenderStyle* style, SVGElement* contextElement, const UChar* currentCharacter)
38{
39 float kerning = 0;
40 auto kerningLength = style->kerning();
41 if (kerningLength.lengthType() == SVGLengthType::Percentage)
42 kerning = kerningLength.valueAsPercentage() * m_font.pixelSize();
43 else {
44 SVGLengthContext lengthContext(contextElement);
45 kerning = kerningLength.value(lengthContext);
46 }
47
48 const UChar* lastCharacter = m_lastCharacter;
49 m_lastCharacter = currentCharacter;
50
51 if (!kerning && !m_font.letterSpacing() && !m_font.wordSpacing())
52 return 0;
53
54 float spacing = m_font.letterSpacing() + kerning;
55 if (currentCharacter && lastCharacter && m_font.wordSpacing()) {
56 if (FontCascade::treatAsSpace(*currentCharacter) && !FontCascade::treatAsSpace(*lastCharacter))
57 spacing += m_font.wordSpacing();
58 }
59
60 return spacing;
61}
