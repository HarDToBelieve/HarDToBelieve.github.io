41SVGTextMetrics::SVGTextMetrics(RenderSVGInlineText& textRenderer, const TextRun& run)
42{
43 float scalingFactor = textRenderer.scalingFactor();
44 ASSERT(scalingFactor);
45
46 const FontCascade& scaledFont = textRenderer.scaledFont();
47 int length = 0;
48
49 // Calculate width/height using the scaled font, divide this result by the scalingFactor afterwards.
50 m_width = scaledFont.width(run) / scalingFactor;
51 length = run.length();
52 m_glyph.name = emptyString();
53 m_height = scaledFont.fontMetrics().floatHeight() / scalingFactor;
54
55 m_glyph.unicodeString = run.is8Bit() ? String(run.characters8(), length) : String(run.characters16(), length);
56 m_glyph.isValid = true;
57
58 ASSERT(length >= 0);
59 m_length = static_cast<unsigned>(length);
60}

62TextRun SVGTextMetrics::constructTextRun(RenderSVGInlineText& text, unsigned position, unsigned length)
63{
64 const RenderStyle& style = text.style();
65
66 TextRun run(StringView(text.text()).substring(position, length)
67 , 0 /* xPos, only relevant with allowTabs=true */
68 , 0 /* padding, only relevant for justified text, not relevant for SVG */
69 , AllowRightExpansion
70 , style.direction()
71 , isOverride(style.unicodeBidi()) /* directionalOverride */);
72
73 // We handle letter & word spacing ourselves.
74 run.disableSpacing();
75 return run;
76}
77
78SVGTextMetrics SVGTextMetrics::measureCharacterRange(RenderSVGInlineText& text, unsigned position, unsigned length)
79{
80 return SVGTextMetrics(text, constructTextRun(text, position, length));
81}
82
83SVGTextMetrics::SVGTextMetrics(RenderSVGInlineText& text, unsigned length, float width)
84{
85 float scalingFactor = text.scalingFactor();
86 ASSERT(scalingFactor);
87
88 m_width = width / scalingFactor;
89 m_height = text.scaledFont().fontMetrics().floatHeight() / scalingFactor;
90
91 m_length = length;
92}
