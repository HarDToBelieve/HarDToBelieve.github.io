30SVGTextMetricsBuilder::SVGTextMetricsBuilder()
31 : m_text(0)
32 , m_run(StringView())
33 , m_textPosition(0)
34 , m_isComplexText(false)
35 , m_totalWidth(0)
36{
37}

39inline bool SVGTextMetricsBuilder::currentCharacterStartsSurrogatePair() const
40{
41 return U16_IS_LEAD(m_run[m_textPosition]) && (m_textPosition + 1) < m_run.length() && U16_IS_TRAIL(m_run[m_textPosition + 1]);
42}
43
44bool SVGTextMetricsBuilder::advance()
45{
46 m_textPosition += m_currentMetrics.length();
47 if (m_textPosition >= m_run.length())
48 return false;
49
50 if (m_isComplexText)
51 advanceComplexText();
52 else
53 advanceSimpleText();
54
55 return m_currentMetrics.length() > 0;
56}
57
58void SVGTextMetricsBuilder::advanceSimpleText()
59{
60 GlyphBuffer glyphBuffer;
61 auto before = m_simpleWidthIterator->currentCharacterIndex();
62 m_simpleWidthIterator->advance(m_textPosition + 1, glyphBuffer);
63 auto after = m_simpleWidthIterator->currentCharacterIndex();
64 if (before == after) {
65 m_currentMetrics = SVGTextMetrics();
66 return;
67 }
68
69 float currentWidth = m_simpleWidthIterator->runWidthSoFar() - m_totalWidth;
70 m_totalWidth = m_simpleWidthIterator->runWidthSoFar();
71
72 m_currentMetrics = SVGTextMetrics(*m_text, after - before, currentWidth);
73}
74
75void SVGTextMetricsBuilder::advanceComplexText()
76{
77 unsigned metricsLength = currentCharacterStartsSurrogatePair() ? 2 : 1;
78 m_currentMetrics = SVGTextMetrics::measureCharacterRange(*m_text, m_textPosition, metricsLength);
79 m_complexStartToCurrentMetrics = SVGTextMetrics::measureCharacterRange(*m_text, 0, m_textPosition + metricsLength);
80 ASSERT(m_currentMetrics.length() == metricsLength);
81
82 // Frequent case for Arabic text: when measuring a single character the arabic isolated form is taken
83 // when rendering the glyph "in context" (with it's surrounding characters) it changes due to shaping.
84 // So whenever currentWidth != currentMetrics.width(), we are processing a text run whose length is
85 // not equal to the sum of the individual lengths of the glyphs, when measuring them isolated.
86 float currentWidth = m_complexStartToCurrentMetrics.width() - m_totalWidth;
87 if (currentWidth != m_currentMetrics.width())
88 m_currentMetrics.setWidth(currentWidth);
89
90 m_totalWidth = m_complexStartToCurrentMetrics.width();
91}

93void SVGTextMetricsBuilder::initializeMeasurementWithTextRenderer(RenderSVGInlineText& text)
94{
95 m_text = &text;
96 m_textPosition = 0;
97 m_currentMetrics = SVGTextMetrics();
98 m_complexStartToCurrentMetrics = SVGTextMetrics();
99 m_totalWidth = 0;
100
101 const FontCascade& scaledFont = text.scaledFont();
102 m_run = SVGTextMetrics::constructTextRun(text);
103 m_isComplexText = scaledFont.codePath(m_run) == FontCascade::CodePath::Complex;
104
105 if (m_isComplexText)
106 m_simpleWidthIterator = nullptr;
107 else
108 m_simpleWidthIterator = makeUnique<WidthIterator>(scaledFont, m_run);
109}

111struct MeasureTextData {
112 MeasureTextData(SVGCharacterDataMap* characterDataMap)
113 : allCharactersMap(characterDataMap)
114 , lastCharacter(0)
115 , processRenderer(false)
116 , valueListPosition(0)
117 , skippedCharacters(0)
118 {
119 }
120
121 SVGCharacterDataMap* allCharactersMap;
122 UChar lastCharacter;
123 bool processRenderer;
124 unsigned valueListPosition;
125 unsigned skippedCharacters;
126};

128void SVGTextMetricsBuilder::measureTextRenderer(RenderSVGInlineText& text, MeasureTextData* data)
129{
130 SVGTextLayoutAttributes* attributes = text.layoutAttributes();
131 Vector<SVGTextMetrics>* textMetricsValues = &attributes->textMetricsValues();
132 if (data->processRenderer) {
133 if (data->allCharactersMap)
134 attributes->clear();
135 else
136 textMetricsValues->clear();
137 }
138
139 initializeMeasurementWithTextRenderer(text);
140 bool preserveWhiteSpace = text.style().whiteSpace() == WhiteSpace::Pre;
141 int surrogatePairCharacters = 0;
142
143 while (advance()) {
144 UChar currentCharacter = m_run[m_textPosition];
145 if (currentCharacter == ' ' && !preserveWhiteSpace && (!data->lastCharacter || data->lastCharacter == ' ')) {
146 if (data->processRenderer)
147 textMetricsValues->append(SVGTextMetrics(SVGTextMetrics::SkippedSpaceMetrics));
148 if (data->allCharactersMap)
149 data->skippedCharacters += m_currentMetrics.length();
150 continue;
151 }
152
153 if (data->processRenderer) {
154 if (data->allCharactersMap) {
155 const SVGCharacterDataMap::const_iterator it = data->allCharactersMap->find(data->valueListPosition + m_textPosition - data->skippedCharacters - surrogatePairCharacters + 1);
156 if (it != data->allCharactersMap->end())
157 attributes->characterDataMap().set(m_textPosition + 1, it->value);
158 }
159 textMetricsValues->append(m_currentMetrics);
160 }
161
162 if (data->allCharactersMap && currentCharacterStartsSurrogatePair())
163 surrogatePairCharacters++;
164
165 data->lastCharacter = currentCharacter;
166 }
167
168 if (m_simpleWidthIterator) {
169 GlyphBuffer glyphBuffer;
170 m_simpleWidthIterator->finalize(glyphBuffer);
171 }
172
173 if (!data->allCharactersMap)
174 return;
175
176 data->valueListPosition += m_textPosition - data->skippedCharacters;
177 data->skippedCharacters = 0;
178}

180void SVGTextMetricsBuilder::walkTree(RenderElement& start, RenderSVGInlineText* stopAtLeaf, MeasureTextData* data)
181{
182 for (auto& child : childrenOfType<RenderObject>(start)) {
183 if (is<RenderSVGInlineText>(child)) {
184 auto& text = downcast<RenderSVGInlineText>(child);
185 if (stopAtLeaf && stopAtLeaf != &text) {
186 data->processRenderer = false;
187 measureTextRenderer(text, data);
188 continue;
189 }
190
191 data->processRenderer = true;
192 measureTextRenderer(text, data);
193 if (stopAtLeaf)
194 return;
195
196 continue;
197 }
198
199 if (!is<RenderSVGInline>(child))
200 continue;
201
202 walkTree(downcast<RenderSVGInline>(child), stopAtLeaf, data);
203 }
204}

206void SVGTextMetricsBuilder::measureTextRenderer(RenderSVGInlineText& text)
207{
208 auto* textRoot = RenderSVGText::locateRenderSVGTextAncestor(text);
209 if (!textRoot)
210 return;
211
212 MeasureTextData data(nullptr);
213 walkTree(*textRoot, &text, &data);
214}

216void SVGTextMetricsBuilder::buildMetricsAndLayoutAttributes(RenderSVGText& textRoot, RenderSVGInlineText* stopAtLeaf, SVGCharacterDataMap& allCharactersMap)
217{
218 MeasureTextData data(&allCharactersMap);
219 walkTree(textRoot, stopAtLeaf, &data);
220}
