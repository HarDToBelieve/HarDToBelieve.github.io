40void RenderTreeBuilder::BlockFlow::attach(RenderBlockFlow& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
41{
42 if (auto* multicolumnFlow = parent.multiColumnFlow()) {
43 auto legendAvoidsMulticolumn = parent.isFieldset() && child->isLegend();
44 if (legendAvoidsMulticolumn)
45 return m_builder.blockBuilder().attach(parent, WTFMove(child), nullptr);
46
47 auto legendBeforeChildIsIncorrect = parent.isFieldset() && beforeChild && beforeChild->isLegend();
48 if (legendBeforeChildIsIncorrect)
49 return m_builder.blockBuilder().attach(*multicolumnFlow, WTFMove(child), nullptr);
50
51 // When the before child is set to be the first child of the RenderBlockFlow, we need to readjust it to be the first
52 // child of the multicol conainter.
53 return m_builder.attach(*multicolumnFlow, WTFMove(child), beforeChild == multicolumnFlow ? multicolumnFlow->firstChild() : beforeChild);
54 }
55
56 auto* beforeChildOrPlaceholder = beforeChild;
57 if (auto* containingFragmentedFlow = parent.enclosingFragmentedFlow())
58 beforeChildOrPlaceholder = m_builder.multiColumnBuilder().resolveMovedChild(*containingFragmentedFlow, beforeChild);
59 m_builder.blockBuilder().attach(parent, WTFMove(child), beforeChildOrPlaceholder);
60}

62void RenderTreeBuilder::BlockFlow::moveAllChildrenIncludingFloats(RenderBlockFlow& from, RenderBlock& to, RenderTreeBuilder::NormalizeAfterInsertion normalizeAfterInsertion)
63{
64 m_builder.moveAllChildren(from, to, normalizeAfterInsertion);
65 from.addFloatsToNewParent(downcast<RenderBlockFlow>(to));
66}
