40void RenderTreeBuilder::FormControls::attach(RenderButton& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
41{
42 m_builder.blockBuilder().attach(findOrCreateParentForChild(parent), WTFMove(child), beforeChild);
43}
44
45void RenderTreeBuilder::FormControls::attach(RenderMenuList& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
46{
47 auto& newChild = *child.get();
48 m_builder.blockBuilder().attach(findOrCreateParentForChild(parent), WTFMove(child), beforeChild);
49 parent.didAttachChild(newChild, beforeChild);
50}
51
52RenderPtr<RenderObject> RenderTreeBuilder::FormControls::detach(RenderMenuList& parent, RenderObject& child)
53{
54 auto* innerRenderer = parent.innerRenderer();
55 if (!innerRenderer || &child == innerRenderer)
56 return m_builder.blockBuilder().detach(parent, child);
57 return m_builder.detach(*innerRenderer, child);
58}
59
60RenderPtr<RenderObject> RenderTreeBuilder::FormControls::detach(RenderButton& parent, RenderObject& child)
61{
62 auto* innerRenderer = parent.innerRenderer();
63 if (!innerRenderer || &child == innerRenderer || child.parent() == &parent) {
64 ASSERT(&child == innerRenderer || !innerRenderer);
65 return m_builder.blockBuilder().detach(parent, child);
66 }
67 return m_builder.detach(*innerRenderer, child);
68}

71RenderBlock& RenderTreeBuilder::FormControls::findOrCreateParentForChild(RenderButton& parent)
72{
73 auto* innerRenderer = parent.innerRenderer();
74 if (innerRenderer)
75 return *innerRenderer;
76
77 auto wrapper = parent.createAnonymousBlock(parent.style().display());
78 innerRenderer = wrapper.get();
79 m_builder.blockBuilder().attach(parent, WTFMove(wrapper), nullptr);
80 parent.setInnerRenderer(*innerRenderer);
81 return *innerRenderer;
82}
83
84RenderBlock& RenderTreeBuilder::FormControls::findOrCreateParentForChild(RenderMenuList& parent)
85{
86 auto* innerRenderer = parent.innerRenderer();
87 if (innerRenderer)
88 return *innerRenderer;
89
90 auto wrapper = parent.createAnonymousBlock();
91 innerRenderer = wrapper.get();
92 m_builder.blockBuilder().attach(parent, WTFMove(wrapper), nullptr);
93 parent.setInnerRenderer(*innerRenderer);
94 return *innerRenderer;
95}
