39static RenderMultiColumnSet* findSetRendering(const RenderMultiColumnFlow& fragmentedFlow, const RenderObject& renderer)
40{
41 // Find the set inside which the specified renderer would be rendered.
42 for (auto* multicolSet = fragmentedFlow.firstMultiColumnSet(); multicolSet; multicolSet = multicolSet->nextSiblingMultiColumnSet()) {
43 if (multicolSet->containsRendererInFragmentedFlow(renderer))
44 return multicolSet;
45 }
46 return nullptr;
47}

49static RenderObject* spannerPlacehoderCandidate(const RenderObject& renderer, const RenderMultiColumnFlow& stayWithin)
50{
51 // Spanner candidate is a next sibling/ancestor's next child within the flow thread and
52 // it is in the same inflow/out-of-flow layout context.
53 if (renderer.isOutOfFlowPositioned())
54 return nullptr;
55
56 ASSERT(renderer.isDescendantOf(&stayWithin));
57 auto* current = &renderer;
58 while (true) {
59 // Skip to the first in-flow sibling.
60 auto* nextSibling = current->nextSibling();
61 while (nextSibling && nextSibling->isOutOfFlowPositioned())
62 nextSibling = nextSibling->nextSibling();
63 if (nextSibling)
64 return nextSibling;
65 // No sibling candidate, jump to the parent and check its siblings.
66 current = current->parent();
67 if (!current || current == &stayWithin || current->isOutOfFlowPositioned())
68 return nullptr;
69 }
70 return nullptr;
71}

73static bool isValidColumnSpanner(const RenderMultiColumnFlow& fragmentedFlow, const RenderObject& descendant)
74{
75 // We assume that we're inside the flow thread. This function is not to be called otherwise.
76 ASSERT(descendant.isDescendantOf(&fragmentedFlow));
77 // First make sure that the renderer itself has the right properties for becoming a spanner.
78 if (!is<RenderBox>(descendant))
79 return false;
80
81 auto& descendantBox = downcast<RenderBox>(descendant);
82 if (descendantBox.isFloatingOrOutOfFlowPositioned())
83 return false;
84
85 if (descendantBox.style().columnSpan() != ColumnSpan::All)
86 return false;
87
88 auto* parent = descendantBox.parent();
89 if (!is<RenderBlockFlow>(*parent) || parent->childrenInline()) {
90 // Needs to be block-level.
91 return false;
92 }
93
94 // We need to have the flow thread as the containing block. A spanner cannot break out of the flow thread.
95 auto* enclosingFragmentedFlow = descendantBox.enclosingFragmentedFlow();
96 if (enclosingFragmentedFlow != &fragmentedFlow)
97 return false;
98
99 // This looks like a spanner, but if we're inside something unbreakable, it's not to be treated as one.
100 for (auto* ancestor = descendantBox.containingBlock(); ancestor; ancestor = ancestor->containingBlock()) {
101 if (is<RenderView>(*ancestor))
102 return false;
103 if (ancestor->isLegend())
104 return false;
105 if (is<RenderTextControl>(*ancestor))
106 return false;
107 if (is<RenderFragmentedFlow>(*ancestor)) {
108 // Don't allow any intervening non-multicol fragmentation contexts. The spec doesn't say
109 // anything about disallowing this, but it's just going to be too complicated to
110 // implement (not to mention specify behavior).
111 return ancestor == &fragmentedFlow;
112 }
113 if (is<RenderBlockFlow>(*ancestor)) {
114 auto& blockFlowAncestor = downcast<RenderBlockFlow>(*ancestor);
115 if (blockFlowAncestor.willCreateColumns()) {
116 // This ancestor (descendent of the fragmentedFlow) will create columns later. The spanner belongs to it.
117 return false;
118 }
119 if (blockFlowAncestor.multiColumnFlow()) {
120 // While this ancestor (descendent of the fragmentedFlow) has a fragmented flow context, this context is being destroyed.
121 // However the spanner still belongs to it (will most likely be moved to the parent fragmented context as the next step).
122 return false;
123 }
124 }
125 ASSERT(ancestor->style().columnSpan() != ColumnSpan::All || !isValidColumnSpanner(fragmentedFlow, *ancestor));
126 if (ancestor->isUnsplittableForPagination())
127 return false;
128 }
129 ASSERT_NOT_REACHED();
130 return false;
131}

138void RenderTreeBuilder::MultiColumn::updateAfterDescendants(RenderBlockFlow& flow)
139{
140 bool needsFragmentedFlow = flow.requiresColumns(flow.style().columnCount());
141 bool hasFragmentedFlow = flow.multiColumnFlow();
142
143 if (!hasFragmentedFlow && needsFragmentedFlow) {
144 createFragmentedFlow(flow);
145 return;
146 }
147 if (hasFragmentedFlow && !needsFragmentedFlow) {
148 destroyFragmentedFlow(flow);
149 return;
150 }
151}

153void RenderTreeBuilder::MultiColumn::createFragmentedFlow(RenderBlockFlow& flow)
154{
155 flow.setChildrenInline(false); // Do this to avoid wrapping inline children that are just going to move into the flow thread.
156 flow.deleteLines();
157 // If this soon-to-be multicolumn flow is already part of a multicolumn context, we need to move back the descendant spanners
158 // to their original position before moving subtrees around.
159 if (auto* enclosingflow = flow.enclosingFragmentedFlow(); is<RenderMultiColumnFlow>(enclosingflow))
160 restoreColumnSpannersForContainer(flow, downcast<RenderMultiColumnFlow>(*enclosingflow));
161
162 auto newFragmentedFlow = WebCore::createRenderer<RenderMultiColumnFlow>(flow.document(), RenderStyle::createAnonymousStyleWithDisplay(flow.style(), DisplayType::Block));
163 newFragmentedFlow->initializeStyle();
164 auto& fragmentedFlow = *newFragmentedFlow;
165 m_builder.blockBuilder().attach(flow, WTFMove(newFragmentedFlow), nullptr);
166
167 // Reparent children preceding the fragmented flow into the fragmented flow.
168 m_builder.moveChildren(flow, fragmentedFlow, flow.firstChild(), &fragmentedFlow, RenderTreeBuilder::NormalizeAfterInsertion::Yes);
169 if (flow.isFieldset()) {
170 // Keep legends out of the flow thread.
171 for (auto& box : childrenOfType<RenderBox>(fragmentedFlow)) {
172 if (box.isLegend())
173 m_builder.move(fragmentedFlow, flow, box, RenderTreeBuilder::NormalizeAfterInsertion::Yes);
174 }
175 }
176
177 flow.setMultiColumnFlow(fragmentedFlow);
178}

180void RenderTreeBuilder::MultiColumn::restoreColumnSpannersForContainer(const RenderElement& container, RenderMultiColumnFlow& multiColumnFlow)
181{
182 auto& spanners = multiColumnFlow.spannerMap();
183 Vector<RenderMultiColumnSpannerPlaceholder*> placeholdersToRestore;
184 for (auto& spannerAndPlaceholder : spanners) {
185 auto& placeholder = *spannerAndPlaceholder.value;
186 if (!placeholder.isDescendantOf(&container))
187 continue;
188 placeholdersToRestore.append(&placeholder);
189 }
190 for (auto* placeholder : placeholdersToRestore) {
191 auto* spanner = placeholder->spanner();
192 if (!spanner) {
193 ASSERT_NOT_REACHED();
194 continue;
195 }
196 // Move the spanner back to its original position.
197 auto& spannerOriginalParent = *placeholder->parent();
198 // Detaching the spanner takes care of removing the placeholder (and merges the RenderMultiColumnSets).
199 auto spannerToReInsert = m_builder.detach(*spanner->parent(), *spanner);
200 m_builder.attach(spannerOriginalParent, WTFMove(spannerToReInsert));
201 }
202}

204void RenderTreeBuilder::MultiColumn::destroyFragmentedFlow(RenderBlockFlow& flow)
205{
206 auto& multiColumnFlow = *flow.multiColumnFlow();
207 multiColumnFlow.deleteLines();
208
209 // Move spanners back to their original DOM position in the tree, and destroy the placeholders.
210 auto& spanners = multiColumnFlow.spannerMap();
211 Vector<RenderMultiColumnSpannerPlaceholder*> placeholdersToDelete;
212 for (auto& spannerAndPlaceholder : spanners)
213 placeholdersToDelete.append(spannerAndPlaceholder.value.get());
214 Vector<std::pair<RenderElement*, RenderPtr<RenderObject>>> parentAndSpannerList;
215 for (auto* placeholder : placeholdersToDelete) {
216 auto* spannerOriginalParent = placeholder->parent();
217 if (spannerOriginalParent == &multiColumnFlow)
218 spannerOriginalParent = &flow;
219 // Detaching the spanner takes care of removing the placeholder (and merges the RenderMultiColumnSets).
220 auto* spanner = placeholder->spanner();
221 parentAndSpannerList.append(std::make_pair(spannerOriginalParent, m_builder.detach(*spanner->parent(), *spanner, CanCollapseAnonymousBlock::No)));
222 }
223 while (auto* columnSet = multiColumnFlow.firstMultiColumnSet())
224 m_builder.destroy(*columnSet);
225
226 flow.clearMultiColumnFlow();
227 auto hasInitialBlockChild = [&] {
228 if (!flow.isFieldset())
229 return false;
230 // We don't move the legend under the multicolumn flow (see MultiColumn::createFragmentedFlow), so when the multicolumn context is destroyed
231 // the fieldset already has a legend block level box.
232 for (auto& box : childrenOfType<RenderBox>(flow)) {
233 if (box.isLegend())
234 return true;
235 }
236 return false;
237 }();
238 flow.setChildrenInline(!hasInitialBlockChild);
239 m_builder.moveAllChildren(multiColumnFlow, flow, RenderTreeBuilder::NormalizeAfterInsertion::Yes);
240 m_builder.destroy(multiColumnFlow);
241 for (auto& parentAndSpanner : parentAndSpannerList)
242 m_builder.attach(*parentAndSpanner.first, WTFMove(parentAndSpanner.second));
243}

246RenderObject* RenderTreeBuilder::MultiColumn::resolveMovedChild(RenderFragmentedFlow& enclosingFragmentedFlow, RenderObject* beforeChild)
247{
248 if (!beforeChild)
249 return nullptr;
250
251 if (!is<RenderBox>(*beforeChild))
252 return beforeChild;
253
254 if (!is<RenderMultiColumnFlow>(enclosingFragmentedFlow))
255 return beforeChild;
256
257 // We only need to resolve for column spanners.
258 if (beforeChild->style().columnSpan() != ColumnSpan::All)
259 return beforeChild;
260
261 // The renderer for the actual DOM node that establishes a spanner is moved from its original
262 // location in the render tree to becoming a sibling of the column sets. In other words, it's
263 // moved out from the flow thread (and becomes a sibling of it). When we for instance want to
264 // create and insert a renderer for the sibling node immediately preceding the spanner, we need
265 // to map that spanner renderer to the spanner's placeholder, which is where the new inserted
266 // renderer belongs.
267 if (auto* placeholder = downcast<RenderMultiColumnFlow>(enclosingFragmentedFlow).findColumnSpannerPlaceholder(downcast<RenderBox>(beforeChild)))
268 return placeholder;
269
270 // This is an invalid spanner, or its placeholder hasn't been created yet. This happens when
271 // moving an entire subtree into the flow thread, when we are processing the insertion of this
272 // spanner's preceding sibling, and we obviously haven't got as far as processing this spanner
273 // yet.
274 return beforeChild;
275}

277static bool gShiftingSpanner = false;
278
279void RenderTreeBuilder::MultiColumn::multiColumnDescendantInserted(RenderMultiColumnFlow& flow, RenderObject& newDescendant)
280{
281 if (gShiftingSpanner || newDescendant.isRenderFragmentedFlow())
282 return;
283
284 auto* subtreeRoot = &newDescendant;
285 auto* descendant = subtreeRoot;
286 while (descendant) {
287 // Skip nested multicolumn flows.
288 if (is<RenderMultiColumnFlow>(*descendant)) {
289 descendant = descendant->nextSibling();
290 continue;
291 }
292 if (is<RenderMultiColumnSpannerPlaceholder>(*descendant)) {
293 // A spanner's placeholder has been inserted. The actual spanner renderer is moved from
294 // where it would otherwise occur (if it weren't a spanner) to becoming a sibling of the
295 // column sets.
296 RenderMultiColumnSpannerPlaceholder& placeholder = downcast<RenderMultiColumnSpannerPlaceholder>(*descendant);
297 ASSERT(!flow.spannerMap().get(placeholder.spanner()));
298 flow.spannerMap().add(placeholder.spanner(), makeWeakPtr(downcast<RenderMultiColumnSpannerPlaceholder>(descendant)));
299 ASSERT(!placeholder.firstChild()); // There should be no children here, but if there are, we ought to skip them.
300 } else
301 descendant = processPossibleSpannerDescendant(flow, subtreeRoot, *descendant);
302 if (descendant)
303 descendant = descendant->nextInPreOrder(subtreeRoot);
304 }
305}

307RenderObject* RenderTreeBuilder::MultiColumn::processPossibleSpannerDescendant(RenderMultiColumnFlow& flow, RenderObject*& subtreeRoot, RenderObject& descendant)
308{
309 RenderBlockFlow* multicolContainer = flow.multiColumnBlockFlow();
310 RenderObject* nextRendererInFragmentedFlow = spannerPlacehoderCandidate(descendant, flow);
311 RenderObject* insertBeforeMulticolChild = nullptr;
312 RenderObject* nextDescendant = &descendant;
313
314 if (!multicolContainer)
315 return nullptr;
316
317 if (isValidColumnSpanner(flow, descendant)) {
318 // This is a spanner (column-span:all). Such renderers are moved from where they would
319 // otherwise occur in the render tree to becoming a direct child of the multicol container,
320 // so that they live among the column sets. This simplifies the layout implementation, and
321 // basically just relies on regular block layout done by the RenderBlockFlow that
322 // establishes the multicol container.
323 RenderBlockFlow* container = downcast<RenderBlockFlow>(descendant.parent());
324 RenderMultiColumnSet* setToSplit = nullptr;
325 if (nextRendererInFragmentedFlow) {
326 setToSplit = findSetRendering(flow, descendant);
327 if (setToSplit) {
328 setToSplit->setNeedsLayout();
329 insertBeforeMulticolChild = setToSplit->nextSibling();
330 }
331 }
332 // Moving a spanner's renderer so that it becomes a sibling of the column sets requires us
333 // to insert an anonymous placeholder in the tree where the spanner's renderer otherwise
334 // would have been. This is needed for a two reasons: We need a way of separating inline
335 // content before and after the spanner, so that it becomes separate line boxes. Secondly,
336 // this placeholder serves as a break point for column sets, so that, when encountered, we
337 // end flowing one column set and move to the next one.
338 auto newPlaceholder = RenderMultiColumnSpannerPlaceholder::createAnonymous(flow, downcast<RenderBox>(descendant), container->style());
339 auto& placeholder = *newPlaceholder;
340 m_builder.attach(*container, WTFMove(newPlaceholder), descendant.nextSibling());
341 auto takenDescendant = m_builder.detach(*container, descendant);
342
343 // This is a guard to stop an ancestor flow thread from processing the spanner.
344 gShiftingSpanner = true;
345 m_builder.blockBuilder().attach(*multicolContainer, WTFMove(takenDescendant), insertBeforeMulticolChild);
346 gShiftingSpanner = false;
347
348 // The spanner has now been moved out from the flow thread, but we don't want to
349 // examine its children anyway. They are all part of the spanner and shouldn't trigger
350 // creation of column sets or anything like that. Continue at its original position in
351 // the tree, i.e. where the placeholder was just put.
352 if (subtreeRoot == &descendant)
353 subtreeRoot = &placeholder;
354 nextDescendant = &placeholder;
355 } else {
356 // This is regular multicol content, i.e. not part of a spanner.
357 if (is<RenderMultiColumnSpannerPlaceholder>(nextRendererInFragmentedFlow)) {
358 // Inserted right before a spanner. Is there a set for us there?
359 RenderMultiColumnSpannerPlaceholder& placeholder = downcast<RenderMultiColumnSpannerPlaceholder>(*nextRendererInFragmentedFlow);
360 if (RenderObject* previous = placeholder.spanner()->previousSibling()) {
361 if (is<RenderMultiColumnSet>(*previous))
362 return nextDescendant; // There's already a set there. Nothing to do.
363 }
364 insertBeforeMulticolChild = placeholder.spanner();
365 } else if (RenderMultiColumnSet* lastSet = flow.lastMultiColumnSet()) {
366 // This child is not an immediate predecessor of a spanner, which means that if this
367 // child precedes a spanner at all, there has to be a column set created for us there
368 // already. If it doesn't precede any spanner at all, on the other hand, we need a
369 // column set at the end of the multicol container. We don't really check here if the
370 // child inserted precedes any spanner or not (as that's an expensive operation). Just
371 // make sure we have a column set at the end. It's no big deal if it remains unused.
372 if (!lastSet->nextSibling())
373 return nextDescendant;
374 }
375 }
376 // Need to create a new column set when there's no set already created. We also always insert
377 // another column set after a spanner. Even if it turns out that there are no renderers
378 // following the spanner, there may be bottom margins there, which take up space.
379 auto newSet = createRenderer<RenderMultiColumnSet>(flow, RenderStyle::createAnonymousStyleWithDisplay(multicolContainer->style(), DisplayType::Block));
380 newSet->initializeStyle();
381 auto& set = *newSet;
382 m_builder.blockBuilder().attach(*multicolContainer, WTFMove(newSet), insertBeforeMulticolChild);
383 flow.invalidateFragments();
384
385 // We cannot handle immediate column set siblings at the moment (and there's no need for
386 // it, either). There has to be at least one spanner separating them.
387 ASSERT_UNUSED(set, !RenderMultiColumnFlow::previousColumnSetOrSpannerSiblingOf(&set)
388 || !RenderMultiColumnFlow::previousColumnSetOrSpannerSiblingOf(&set)->isRenderMultiColumnSet());
389 ASSERT(!RenderMultiColumnFlow::nextColumnSetOrSpannerSiblingOf(&set)
390 || !RenderMultiColumnFlow::nextColumnSetOrSpannerSiblingOf(&set)->isRenderMultiColumnSet());
391
392 return nextDescendant;
393}

395void RenderTreeBuilder::MultiColumn::handleSpannerRemoval(RenderMultiColumnFlow& flow, RenderObject& spanner, RenderTreeBuilder::CanCollapseAnonymousBlock canCollapseAnonymousBlock)
396{
397 // The placeholder may already have been removed, but if it hasn't, do so now.
398 if (auto placeholder = flow.spannerMap().take(&downcast<RenderBox>(spanner)))
399 m_builder.destroy(*placeholder, canCollapseAnonymousBlock);
400
401 if (auto* next = spanner.nextSibling()) {
402 if (auto* previous = spanner.previousSibling()) {
403 if (previous->isRenderMultiColumnSet() && next->isRenderMultiColumnSet()) {
404 // Merge two sets that no longer will be separated by a spanner.
405 m_builder.destroy(*next);
406 previous->setNeedsLayout();
407 }
408 }
409 }
410}

412void RenderTreeBuilder::MultiColumn::multiColumnRelativeWillBeRemoved(RenderMultiColumnFlow& flow, RenderObject& relative, RenderTreeBuilder::CanCollapseAnonymousBlock canCollapseAnonymousBlock)
413{
414 flow.invalidateFragments();
415 if (is<RenderMultiColumnSpannerPlaceholder>(relative)) {
416 // Remove the map entry for this spanner, but leave the actual spanner renderer alone. Also
417 // keep the reference to the spanner, since the placeholder may be about to be re-inserted
418 // in the tree.
419 ASSERT(relative.isDescendantOf(&flow));
420 flow.spannerMap().remove(downcast<RenderMultiColumnSpannerPlaceholder>(relative).spanner());
421 return;
422 }
423 if (relative.style().columnSpan() == ColumnSpan::All) {
424 if (relative.parent() != flow.parent())
425 return; // not a valid spanner.
426
427 handleSpannerRemoval(flow, relative, canCollapseAnonymousBlock);
428 }
429 // Note that we might end up with empty column sets if all column content is removed. That's no
430 // big deal though (and locating them would be expensive), and they will be found and re-used if
431 // content is added again later.
432}

434RenderObject* RenderTreeBuilder::MultiColumn::adjustBeforeChildForMultiColumnSpannerIfNeeded(RenderObject& beforeChild)
435{
436 if (!is<RenderBox>(beforeChild))
437 return &beforeChild;
438
439 auto* nextSibling = beforeChild.nextSibling();
440 if (!nextSibling)
441 return &beforeChild;
442
443 if (!is<RenderMultiColumnSet>(*nextSibling))
444 return &beforeChild;
445
446 auto* multiColumnFlow = downcast<RenderMultiColumnSet>(*nextSibling).multiColumnFlow();
447 if (!multiColumnFlow)
448 return &beforeChild;
449
450 return multiColumnFlow->findColumnSpannerPlaceholder(downcast<RenderBox>(&beforeChild));
451}
