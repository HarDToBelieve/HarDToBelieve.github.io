42RenderElement& RenderTreeBuilder::Table::findOrCreateParentForChild(RenderTableRow& parent, const RenderObject& child, RenderObject*& beforeChild)
43{
44 if (is<RenderTableCell>(child))
45 return parent;
46
47 if (beforeChild && !beforeChild->isAnonymous() && beforeChild->parent() == &parent) {
48 auto* previousSibling = beforeChild->previousSibling();
49 if (is<RenderTableCell>(previousSibling) && previousSibling->isAnonymous()) {
50 beforeChild = nullptr;
51 return downcast<RenderElement>(*previousSibling);
52 }
53 }
54
55 auto createAnonymousTableCell = [&] (auto& parent) -> RenderTableCell& {
56 auto newCell = RenderTableCell::createAnonymousWithParentRenderer(parent);
57 auto& cell = *newCell;
58 m_builder.attach(parent, WTFMove(newCell), beforeChild);
59 beforeChild = nullptr;
60 return cell;
61 };
62
63 auto* lastChild = beforeChild ? beforeChild : parent.lastCell();
64 if (lastChild) {
65 if (is<RenderTableCell>(*lastChild) && lastChild->isAnonymous() && !lastChild->isBeforeOrAfterContent()) {
66 if (beforeChild == lastChild)
67 beforeChild = downcast<RenderElement>(*lastChild).firstChild();
68 return downcast<RenderElement>(*lastChild);
69 }
70
71 // Try to find an anonymous container for the child.
72 if (auto* lastChildParent = lastChild->parent()) {
73 if (lastChildParent->isAnonymous() && !lastChildParent->isBeforeOrAfterContent()) {
74 // If beforeChild is inside an anonymous COLGROUP, create a cell for the new renderer.
75 if (is<RenderTableCol>(*lastChildParent))
76 return createAnonymousTableCell(parent);
77 // If beforeChild is inside an anonymous cell, insert into the cell.
78 if (!is<RenderTableCell>(*lastChild))
79 return *lastChildParent;
80 // If beforeChild is inside an anonymous row, insert into the row.
81 if (is<RenderTableRow>(*lastChildParent))
82 return createAnonymousTableCell(downcast<RenderTableRow>(*lastChildParent));
83 }
84 }
85 }
86 return createAnonymousTableCell(parent);
87}

89RenderElement& RenderTreeBuilder::Table::findOrCreateParentForChild(RenderTableSection& parent, const RenderObject& child, RenderObject*& beforeChild)
90{
91 if (is<RenderTableRow>(child))
92 return parent;
93
94 auto* lastChild = beforeChild ? beforeChild : parent.lastRow();
95 if (is<RenderTableRow>(lastChild) && lastChild->isAnonymous() && !lastChild->isBeforeOrAfterContent()) {
96 if (beforeChild == lastChild)
97 beforeChild = downcast<RenderTableRow>(*lastChild).firstCell();
98 return downcast<RenderElement>(*lastChild);
99 }
100
101 if (beforeChild && !beforeChild->isAnonymous() && beforeChild->parent() == &parent) {
102 auto* row = beforeChild->previousSibling();
103 if (is<RenderTableRow>(row) && row->isAnonymous()) {
104 beforeChild = nullptr;
105 return downcast<RenderElement>(*row);
106 }
107 }
108
109 // If beforeChild is inside an anonymous cell/row, insert into the cell or into
110 // the anonymous row containing it, if there is one.
111 auto* parentCandidate = lastChild;
112 while (parentCandidate && parentCandidate->parent() && parentCandidate->parent()->isAnonymous() && !is<RenderTableRow>(*parentCandidate))
113 parentCandidate = parentCandidate->parent();
114 if (is<RenderTableRow>(parentCandidate) && parentCandidate->isAnonymous() && !parentCandidate->isBeforeOrAfterContent())
115 return downcast<RenderElement>(*parentCandidate);
116
117 auto newRow = RenderTableRow::createAnonymousWithParentRenderer(parent);
118 auto& row = *newRow;
119 m_builder.attach(parent, WTFMove(newRow), beforeChild);
120 beforeChild = nullptr;
121 return row;
122}

124RenderElement& RenderTreeBuilder::Table::findOrCreateParentForChild(RenderTable& parent, const RenderObject& child, RenderObject*& beforeChild)
125{
126 if (is<RenderTableCaption>(child) || is<RenderTableSection>(child))
127 return parent;
128
129 if (is<RenderTableCol>(child)) {
130 if (!child.node() || child.style().display() == DisplayType::TableColumnGroup) {
131 // COLGROUPs and anonymous RenderTableCols (generated wrappers for COLs) are direct children of the table renderer.
132 return parent;
133 }
134 auto newColGroup = createRenderer<RenderTableCol>(parent.document(), RenderStyle::createAnonymousStyleWithDisplay(parent.style(), DisplayType::TableColumnGroup));
135 newColGroup->initializeStyle();
136 auto& colGroup = *newColGroup;
137 m_builder.attach(parent, WTFMove(newColGroup), beforeChild);
138 beforeChild = nullptr;
139 return colGroup;
140 }
141
142 auto* lastChild = parent.lastChild();
143 if (!beforeChild && is<RenderTableSection>(lastChild) && lastChild->isAnonymous() && !lastChild->isBeforeContent())
144 return downcast<RenderElement>(*lastChild);
145
146 if (beforeChild && !beforeChild->isAnonymous() && beforeChild->parent() == &parent) {
147 auto* section = beforeChild->previousSibling();
148 if (is<RenderTableSection>(section) && section->isAnonymous()) {
149 beforeChild = nullptr;
150 return downcast<RenderElement>(*section);
151 }
152 }
153
154 auto* parentCandidate = beforeChild;
155 while (parentCandidate && parentCandidate->parent()->isAnonymous()
156 && !is<RenderTableSection>(*parentCandidate)
157 && parentCandidate->style().display() != DisplayType::TableCaption
158 && parentCandidate->style().display() != DisplayType::TableColumnGroup)
159 parentCandidate = parentCandidate->parent();
160
161 if (parentCandidate) {
162 if (beforeChild && !beforeChild->isAnonymous() && parentCandidate->parent() == &parent) {
163 auto* section = parentCandidate->previousSibling();
164 if (is<RenderTableSection>(section) && section->isAnonymous()) {
165 beforeChild = nullptr;
166 return downcast<RenderElement>(*section);
167 }
168 }
169
170 if (is<RenderTableSection>(*parentCandidate) && parentCandidate->isAnonymous() && !parent.isAfterContent(parentCandidate)) {
171 if (beforeChild == parentCandidate)
172 beforeChild = downcast<RenderTableSection>(*parentCandidate).firstRow();
173 return downcast<RenderElement>(*parentCandidate);
174 }
175 }
176
177 if (beforeChild && !is<RenderTableSection>(*beforeChild)
178 && beforeChild->style().display() != DisplayType::TableCaption
179 && beforeChild->style().display() != DisplayType::TableColumnGroup)
180 beforeChild = nullptr;
181
182 auto newSection = RenderTableSection::createAnonymousWithParentRenderer(parent);
183 auto& section = *newSection;
184 m_builder.attach(parent, WTFMove(newSection), beforeChild);
185 beforeChild = nullptr;
186 return section;
187}

189void RenderTreeBuilder::Table::attach(RenderTableRow& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
190{
191 if (beforeChild && beforeChild->parent() != &parent)
192 beforeChild = m_builder.splitAnonymousBoxesAroundChild(parent, *beforeChild);
193
194 auto& newChild = *child.get();
195 ASSERT(!beforeChild || is<RenderTableCell>(*beforeChild));
196 m_builder.attachToRenderElement(parent, WTFMove(child), beforeChild);
197 // FIXME: child should always be a RenderTableCell at this point.
198 if (is<RenderTableCell>(newChild))
199 parent.didInsertTableCell(downcast<RenderTableCell>(newChild), beforeChild);
200}

202void RenderTreeBuilder::Table::attach(RenderTableSection& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
203{
204 if (beforeChild && beforeChild->parent() != &parent)
205 beforeChild = m_builder.splitAnonymousBoxesAroundChild(parent, *beforeChild);
206
207 // FIXME: child should always be a RenderTableRow at this point.
208 if (is<RenderTableRow>(*child.get()))
209 parent.willInsertTableRow(downcast<RenderTableRow>(*child.get()), beforeChild);
210 ASSERT(!beforeChild || is<RenderTableRow>(*beforeChild));
211 m_builder.attachToRenderElement(parent, WTFMove(child), beforeChild);
212}

214void RenderTreeBuilder::Table::attach(RenderTable& parent, RenderPtr<RenderObject> child, RenderObject* beforeChild)
215{
216 if (beforeChild && beforeChild->parent() != &parent)
217 beforeChild = m_builder.splitAnonymousBoxesAroundChild(parent, *beforeChild);
218
219 auto& newChild = *child.get();
220 if (is<RenderTableSection>(newChild))
221 parent.willInsertTableSection(downcast<RenderTableSection>(newChild), beforeChild);
222 else if (is<RenderTableCol>(newChild))
223 parent.willInsertTableColumn(downcast<RenderTableCol>(newChild), beforeChild);
224
225 m_builder.attachToRenderElement(parent, WTFMove(child), beforeChild);
226}

228bool RenderTreeBuilder::Table::childRequiresTable(const RenderElement& parent, const RenderObject& child)
229{
230 if (is<RenderTableCol>(child)) {
231 const RenderTableCol& newTableColumn = downcast<RenderTableCol>(child);
232 bool isColumnInColumnGroup = newTableColumn.isTableColumn() && is<RenderTableCol>(parent);
233 return !is<RenderTable>(parent) && !isColumnInColumnGroup;
234 }
235 if (is<RenderTableCaption>(child))
236 return !is<RenderTable>(parent);
237
238 if (is<RenderTableSection>(child))
239 return !is<RenderTable>(parent);
240
241 if (is<RenderTableRow>(child))
242 return !is<RenderTableSection>(parent);
243
244 if (is<RenderTableCell>(child))
245 return !is<RenderTableRow>(parent);
246
247 return false;
248}

250static inline bool canCollapseNextSibling(const RenderBox& previousSibling, const RenderBox& nextSibling)
251{
252 if (!previousSibling.isAnonymous() || !nextSibling.isAnonymous())
253 return false;
254 auto* previousSiblingFirstInFlowChild = previousSibling.firstInFlowChild();
255 auto* nextSiblingFirstInFlowChild = nextSibling.firstInFlowChild();
256 // Do not try to collapse and move inline level boxes over to a container with block level boxes (and vice versa).
257 return !previousSiblingFirstInFlowChild || !nextSiblingFirstInFlowChild || previousSiblingFirstInFlowChild->isInline() == nextSiblingFirstInFlowChild->isInline();
258}

260template <typename Parent, typename Child>
261RenderPtr<RenderObject> RenderTreeBuilder::Table::collapseAndDetachAnonymousNextSibling(Parent* parent, Child* previousSibling, Child* nextSibling)
262{
263 if (!parent || !previousSibling || !nextSibling)
264 return { };
265 if (!canCollapseNextSibling(*previousSibling, *nextSibling))
266 return { };
267 m_builder.moveAllChildren(*nextSibling, *previousSibling, RenderTreeBuilder::NormalizeAfterInsertion::No);
268 previousSibling->setChildrenInline(!previousSibling->firstInFlowChild() || previousSibling->firstInFlowChild()->isInline());
269 return m_builder.detach(*parent, *nextSibling);
270}

272void RenderTreeBuilder::Table::collapseAndDestroyAnonymousSiblingCells(const RenderTableCell& willBeDestroyed)
273{
274 if (auto nextCellToDestroy = collapseAndDetachAnonymousNextSibling(willBeDestroyed.row(), willBeDestroyed.previousCell(), willBeDestroyed.nextCell()))
275 downcast<RenderTableCell>(*nextCellToDestroy).deleteLines();
276}
277
278void RenderTreeBuilder::Table::collapseAndDestroyAnonymousSiblingRows(const RenderTableRow& willBeDestroyed)
279{
280 auto toDestroy = collapseAndDetachAnonymousNextSibling(willBeDestroyed.section(), willBeDestroyed.previousRow(), willBeDestroyed.nextRow());
281}
