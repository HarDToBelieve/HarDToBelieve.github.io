89static ContainerNode* findRenderingRoot(ContainerNode& node)
90{
91 if (node.renderer())
92 return &node;
93 for (auto& ancestor : composedTreeAncestors(node)) {
94 if (ancestor.renderer())
95 return &ancestor;
96 if (!ancestor.hasDisplayContents())
97 return nullptr;
98 }
99 return &node.document();
100}

102static ListHashSet<ContainerNode*> findRenderingRoots(const Style::Update& update)
103{
104 ListHashSet<ContainerNode*> renderingRoots;
105 for (auto& root : update.roots()) {
106 auto* renderingRoot = findRenderingRoot(*root);
107 if (!renderingRoot)
108 continue;
109 renderingRoots.add(renderingRoot);
110 }
111 return renderingRoots;
112}

114void RenderTreeUpdater::commit(std::unique_ptr<const Style::Update> styleUpdate)
115{
116 ASSERT(&m_document == &styleUpdate->document());
117
118 if (!m_document.shouldCreateRenderers() || !m_document.renderView())
119 return;
120
121 TraceScope scope(RenderTreeBuildStart, RenderTreeBuildEnd);
122
123 m_styleUpdate = WTFMove(styleUpdate);
124
125 for (auto* root : findRenderingRoots(*m_styleUpdate))
126 updateRenderTree(*root);
127
128 generatedContent().updateRemainingQuotes();
129
130 m_builder.updateAfterDescendants(renderView());
131
132 m_styleUpdate = nullptr;
133}

135static bool shouldCreateRenderer(const Element& element, const RenderElement& parentRenderer)
136{
137 if (!parentRenderer.canHaveChildren() && !(element.isPseudoElement() && parentRenderer.canHaveGeneratedChildren()))
138 return false;
139 if (parentRenderer.element() && !parentRenderer.element()->childShouldCreateRenderer(element))
140 return false;
141 return true;
142}

144void RenderTreeUpdater::updateRenderTree(ContainerNode& root)
145{
146 ASSERT(root.renderer());
147 ASSERT(m_parentStack.isEmpty());
148
149 m_parentStack.append(Parent(root));
150
151 auto descendants = composedTreeDescendants(root);
152 auto it = descendants.begin();
153 auto end = descendants.end();
154
155 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=156172
156 it.dropAssertions();
157
158 while (it != end) {
159 popParentsToDepth(it.depth());
160
161 auto& node = *it;
162
163 if (auto* renderer = node.renderer())
164 renderTreePosition().invalidateNextSibling(*renderer);
165 else if (is<Element>(node) && downcast<Element>(node).hasDisplayContents())
166 renderTreePosition().invalidateNextSibling();
167
168 if (is<Text>(node)) {
169 auto& text = downcast<Text>(node);
170 auto* textUpdate = m_styleUpdate->textUpdate(text);
171 bool didCreateParent = parent().updates && parent().updates->update.change == Style::Change::Renderer;
172 bool mayNeedUpdateWhitespaceOnlyRenderer = renderingParent().didCreateOrDestroyChildRenderer && text.data().isAllSpecialCharacters<isHTMLSpace>();
173 if (didCreateParent || textUpdate || mayNeedUpdateWhitespaceOnlyRenderer)
174 updateTextRenderer(text, textUpdate);
175
176 storePreviousRenderer(text);
177 it.traverseNextSkippingChildren();
178 continue;
179 }
180
181 auto& element = downcast<Element>(node);
182
183 auto* elementUpdates = m_styleUpdate->elementUpdates(element);
184
185 // We hop through display: contents elements in findRenderingRoot, so
186 // there may be other updates down the tree.
187 if (!elementUpdates && !element.hasDisplayContents()) {
188 storePreviousRenderer(element);
189 it.traverseNextSkippingChildren();
190 continue;
191 }
192
193 if (elementUpdates)
194 updateElementRenderer(element, *elementUpdates);
195
196 storePreviousRenderer(element);
197
198 bool mayHaveRenderedDescendants = element.renderer() || (element.hasDisplayContents() && shouldCreateRenderer(element, renderTreePosition().parent()));
199 if (!mayHaveRenderedDescendants) {
200 it.traverseNextSkippingChildren();
201 continue;
202 }
203
204 pushParent(element, elementUpdates);
205
206 it.traverseNext();
207 }
208
209 popParentsToDepth(0);
210}

212auto RenderTreeUpdater::renderingParent() -> Parent&
213{
214 for (unsigned i = m_parentStack.size(); i--;) {
215 if (m_parentStack[i].renderTreePosition)
216 return m_parentStack[i];
217 }
218 ASSERT_NOT_REACHED();
219 return m_parentStack.last();
220}

227void RenderTreeUpdater::pushParent(Element& element, const Style::ElementUpdates* updates)
228{
229 m_parentStack.append(Parent(element, updates));
230
231 updateBeforeDescendants(element, updates);
232}
233
234void RenderTreeUpdater::popParent()
235{
236 auto& parent = m_parentStack.last();
237 if (parent.element)
238 updateAfterDescendants(*parent.element, parent.updates);
239
240 if (&parent != &renderingParent())
241 renderTreePosition().invalidateNextSibling();
242
243 m_parentStack.removeLast();
244}
245
246void RenderTreeUpdater::popParentsToDepth(unsigned depth)
247{
248 ASSERT(m_parentStack.size() >= depth);
249
250 while (m_parentStack.size() > depth)
251 popParent();
252}

260void RenderTreeUpdater::updateAfterDescendants(Element& element, const Style::ElementUpdates* updates)
261{
262 if (updates)
263 generatedContent().updatePseudoElement(element, *updates, PseudoId::After);
264
265 auto* renderer = element.renderer();
266 if (!renderer)
267 return;
268
269 m_builder.updateAfterDescendants(*renderer);
270
271 if (element.hasCustomStyleResolveCallbacks() && updates && updates->update.change == Style::Change::Renderer)
272 element.didAttachRenderers();

275static bool pseudoStyleCacheIsInvalid(RenderElement* renderer, RenderStyle* newStyle)
276{
277 const RenderStyle& currentStyle = renderer->style();
278
279 const PseudoStyleCache* pseudoStyleCache = currentStyle.cachedPseudoStyles();
280 if (!pseudoStyleCache)
281 return false;
282
283 for (auto& cache : *pseudoStyleCache) {
284 PseudoId pseudoId = cache->styleType();
285 std::unique_ptr<RenderStyle> newPseudoStyle = renderer->getUncachedPseudoStyle({ pseudoId }, newStyle, newStyle);
286 if (!newPseudoStyle)
287 return true;
288 if (*newPseudoStyle != *cache) {
289 newStyle->addCachedPseudoStyle(WTFMove(newPseudoStyle));
290 return true;
291 }
292 }
293 return false;
294}

303void RenderTreeUpdater::updateElementRenderer(Element& element, const Style::ElementUpdates& updates)
304{
305#if PLATFORM(IOS_FAMILY)
306 ContentChangeObserver::StyleChangeScope observingScope(m_document, element);
307#endif
308
309 auto& elementUpdate = updates.update;
310 auto elementUpdateStyle = RenderStyle::clonePtr(*elementUpdate.style);
311
312 for (auto& it : updates.pseudoElementUpdates) {
313 auto pseudoId = it.key;
314 if (pseudoId == PseudoId::Before || pseudoId == PseudoId::After)
315 continue;
316 elementUpdateStyle->addCachedPseudoStyle(RenderStyle::clonePtr(*it.value.style));
317 }
318
319 bool shouldTearDownRenderers = elementUpdate.change == Style::Change::Renderer && (element.renderer() || element.hasDisplayContents());
320 if (shouldTearDownRenderers) {
321 if (!element.renderer()) {
322 // We may be tearing down a descendant renderer cached in renderTreePosition.
323 renderTreePosition().invalidateNextSibling();
324 }
325
326 // display:none cancels animations.
327 auto teardownType = elementUpdate.style->display() == DisplayType::None ? TeardownType::RendererUpdateCancelingAnimations : TeardownType::RendererUpdate;
328 tearDownRenderers(element, teardownType, m_builder);
329
330 renderingParent().didCreateOrDestroyChildRenderer = true;
331 }
332
333 bool hasDisplayContents = elementUpdate.style->display() == DisplayType::Contents;
334 if (hasDisplayContents)
335 element.storeDisplayContentsStyle(WTFMove(elementUpdateStyle));
336 else
337 element.resetComputedStyle();
338
339 bool shouldCreateNewRenderer = !element.renderer() && !hasDisplayContents;
340 if (shouldCreateNewRenderer) {
341 if (element.hasCustomStyleResolveCallbacks())
342 element.willAttachRenderers();
343 createRenderer(element, WTFMove(*elementUpdateStyle));
344
345 renderingParent().didCreateOrDestroyChildRenderer = true;
346 return;
347 }
348
349 if (!element.renderer())
350 return;
351 auto& renderer = *element.renderer();
352
353 if (elementUpdate.recompositeLayer) {
354 updateRendererStyle(renderer, WTFMove(*elementUpdateStyle), StyleDifference::RecompositeLayer);
355 return;
356 }
357
358 if (elementUpdate.change == Style::Change::None) {
359 if (pseudoStyleCacheIsInvalid(&renderer, elementUpdateStyle.get())) {
360 updateRendererStyle(renderer, WTFMove(*elementUpdateStyle), StyleDifference::Equal);
361 return;
362 }
363 return;
364 }
365
366 updateRendererStyle(renderer, WTFMove(*elementUpdateStyle), StyleDifference::Equal);
367}

369void RenderTreeUpdater::createRenderer(Element& element, RenderStyle&& style)
370{
371 auto computeInsertionPosition = [this, &element] () {
372 renderTreePosition().computeNextSibling(element);
373 return renderTreePosition();
374 };
375
376 if (!shouldCreateRenderer(element, renderTreePosition().parent()))
377 return;
378
379 if (!element.rendererIsNeeded(style))
380 return;
381
382 RenderTreePosition insertionPosition = computeInsertionPosition();
383 auto newRenderer = element.createElementRenderer(WTFMove(style), insertionPosition);
384 if (!newRenderer)
385 return;
386
387 if (!insertionPosition.parent().isChildAllowed(*newRenderer, newRenderer->style()))
388 return;
389
390 element.setRenderer(newRenderer.get());
391
392 newRenderer->initializeStyle();
393
394#if ENABLE(FULLSCREEN_API)
395 if (m_document.fullscreenManager().isFullscreen() && m_document.fullscreenManager().currentFullscreenElement() == &element) {
396 newRenderer = RenderFullScreen::wrapNewRenderer(m_builder, WTFMove(newRenderer), insertionPosition.parent(), m_document);
397 if (!newRenderer)
398 return;
399 }
400#endif
401
402 m_builder.attach(insertionPosition.parent(), WTFMove(newRenderer), insertionPosition.nextSibling());
403
404 auto* textManipulationController = m_document.textManipulationControllerIfExists();
405 if (UNLIKELY(textManipulationController))
406 textManipulationController->didCreateRendererForElement(element);
407
408 if (AXObjectCache* cache = m_document.axObjectCache())
409 cache->updateCacheAfterNodeIsAttached(&element);
410}

412bool RenderTreeUpdater::textRendererIsNeeded(const Text& textNode)
413{
414 auto& renderingParent = this->renderingParent();
415 auto& parentRenderer = renderingParent.renderTreePosition->parent();
416 if (!parentRenderer.canHaveChildren())
417 return false;
418 if (parentRenderer.element() && !parentRenderer.element()->childShouldCreateRenderer(textNode))
419 return false;
420 if (textNode.isEditingText())
421 return true;
422 if (!textNode.length())
423 return false;
424 if (!textNode.data().isAllSpecialCharacters<isHTMLSpace>())
425 return true;
426 if (is<RenderText>(renderingParent.previousChildRenderer))
427 return true;
428 // This text node has nothing but white space. We may still need a renderer in some cases.
429 if (parentRenderer.isTable() || parentRenderer.isTableRow() || parentRenderer.isTableSection() || parentRenderer.isRenderTableCol() || parentRenderer.isFrameSet() || parentRenderer.isRenderGrid() || (parentRenderer.isFlexibleBox() && !parentRenderer.isRenderButton()))
430 return false;
431 if (parentRenderer.style().preserveNewline()) // pre/pre-wrap/pre-line always make renderers.
432 return true;
433
434 auto* previousRenderer = renderingParent.previousChildRenderer;
435 if (previousRenderer && previousRenderer->isBR()) // <span><br/> <br/></span>
436 return false;
437
438 if (parentRenderer.isRenderInline()) {
439 // <span><div/> <div/></span>
440 if (previousRenderer && !previousRenderer->isInline())
441 return false;
442 } else {
443 if (parentRenderer.isRenderBlock() && !parentRenderer.childrenInline() && (!previousRenderer || !previousRenderer->isInline()))
444 return false;
445
446 RenderObject* first = parentRenderer.firstChild();
447 while (first && first->isFloatingOrOutOfFlowPositioned())
448 first = first->nextSibling();
449 RenderObject* nextRenderer = textNode.renderer() ? textNode.renderer() : renderTreePosition().nextSiblingRenderer(textNode);
450 if (!first || nextRenderer == first) {
451 // Whitespace at the start of a block just goes away. Don't even make a render object for this text.
452 return false;
453 }
454 }
455 return true;
456}

458void RenderTreeUpdater::createTextRenderer(Text& textNode, const Style::TextUpdate* textUpdate)
459{
460 ASSERT(!textNode.renderer());
461
462 auto& renderTreePosition = this->renderTreePosition();
463 auto textRenderer = textNode.createTextRenderer(renderTreePosition.parent().style());
464
465 renderTreePosition.computeNextSibling(textNode);
466
467 if (!renderTreePosition.parent().isChildAllowed(*textRenderer, renderTreePosition.parent().style()))
468 return;
469
470 textNode.setRenderer(textRenderer.get());
471
472 if (textUpdate && textUpdate->inheritedDisplayContentsStyle && *textUpdate->inheritedDisplayContentsStyle) {
473 // Wrap text renderer into anonymous inline so we can give it a style.
474 // This is to support "<div style='display:contents;color:green'>text</div>" type cases
475 auto newDisplayContentsAnonymousWrapper = WebCore::createRenderer<RenderInline>(textNode.document(), RenderStyle::clone(**textUpdate->inheritedDisplayContentsStyle));
476 newDisplayContentsAnonymousWrapper->initializeStyle();
477 auto& displayContentsAnonymousWrapper = *newDisplayContentsAnonymousWrapper;
478 m_builder.attach(renderTreePosition.parent(), WTFMove(newDisplayContentsAnonymousWrapper), renderTreePosition.nextSibling());
479
480 textRenderer->setInlineWrapperForDisplayContents(&displayContentsAnonymousWrapper);
481 m_builder.attach(displayContentsAnonymousWrapper, WTFMove(textRenderer));
482 return;
483 }
484
485 m_builder.attach(renderTreePosition.parent(), WTFMove(textRenderer), renderTreePosition.nextSibling());
486
487 auto* textManipulationController = m_document.textManipulationControllerIfExists();
488 if (UNLIKELY(textManipulationController))
489 textManipulationController->didCreateRendererForTextNode(textNode);
490}

492void RenderTreeUpdater::updateTextRenderer(Text& text, const Style::TextUpdate* textUpdate)
493{
494 auto* existingRenderer = text.renderer();
495 bool needsRenderer = textRendererIsNeeded(text);
496
497 if (existingRenderer && textUpdate && textUpdate->inheritedDisplayContentsStyle) {
498 if (existingRenderer->inlineWrapperForDisplayContents() || *textUpdate->inheritedDisplayContentsStyle) {
499 // FIXME: We could update without teardown.
500 tearDownTextRenderer(text, m_builder);
501 existingRenderer = nullptr;
502 }
503 }
504
505 if (existingRenderer) {
506 if (needsRenderer) {
507 if (textUpdate)
508 existingRenderer->setTextWithOffset(text.data(), textUpdate->offset, textUpdate->length);
509 return;
510 }
511 tearDownTextRenderer(text, m_builder);
512 renderingParent().didCreateOrDestroyChildRenderer = true;
513 return;
514 }
515 if (!needsRenderer)
516 return;
517 createTextRenderer(text, textUpdate);
518 renderingParent().didCreateOrDestroyChildRenderer = true;
519}

548void RenderTreeUpdater::tearDownRenderers(Element& root, TeardownType teardownType, RenderTreeBuilder& builder)
549{
550 Vector<Element*, 30> teardownStack;
551
552 auto push = [&] (Element& element) {
553 if (element.hasCustomStyleResolveCallbacks())
554 element.willDetachRenderers();
555 teardownStack.append(&element);
556 };
557
558 auto& document = root.document();
559
560 auto pop = [&] (unsigned depth) {
561 while (teardownStack.size() > depth) {
562 auto& element = *teardownStack.takeLast();
563
564 // Make sure we don't leave any renderers behind in nodes outside the composed tree.
565 if (element.shadowRoot())
566 tearDownLeftoverShadowHostChildren(element, builder);
567
568 switch (teardownType) {
569 case TeardownType::Full:
570 case TeardownType::RendererUpdateCancelingAnimations:
571 if (document.renderTreeBeingDestroyed() || teardownType == TeardownType::RendererUpdateCancelingAnimations)
572 Styleable::fromElement(element).cancelDeclarativeAnimations();
573 break;
574 case TeardownType::RendererUpdate:
575 Styleable::fromElement(element).willChangeRenderer();
576 break;
577 }
578
579 if (teardownType == TeardownType::Full)
580 element.clearHoverAndActiveStatusBeforeDetachingRenderer();
581
582 GeneratedContent::removeBeforePseudoElement(element, builder);
583 GeneratedContent::removeAfterPseudoElement(element, builder);
584
585 if (auto* renderer = element.renderer()) {
586 builder.destroyAndCleanUpAnonymousWrappers(*renderer);
587 element.setRenderer(nullptr);
588 }
589
590 if (element.hasCustomStyleResolveCallbacks())
591 element.didDetachRenderers();
592 }
593 };
594
595 push(root);
596
597 auto descendants = composedTreeDescendants(root);
598 for (auto it = descendants.begin(), end = descendants.end(); it != end; ++it) {
599 pop(it.depth());
600
601 if (is<Text>(*it)) {
602 tearDownTextRenderer(downcast<Text>(*it), builder);
603 continue;
604 }
605
606 push(downcast<Element>(*it));
607 }
608
609 pop(0);
610
611 tearDownLeftoverPaginationRenderersIfNeeded(root, builder);
612}

614void RenderTreeUpdater::tearDownTextRenderer(Text& text, RenderTreeBuilder& builder)

616 auto* renderer = text.renderer();

618 return;
619 builder.destroyAndCleanUpAnonymousWrappers(*renderer);
620 text.setRenderer(nullptr);
621}

623void RenderTreeUpdater::tearDownLeftoverPaginationRenderersIfNeeded(Element& root, RenderTreeBuilder& builder)
624{
625 if (&root != root.document().documentElement())
626 return;
627 for (auto* child = root.document().renderView()->firstChild(); child;) {
628 auto* nextSibling = child->nextSibling();
629 if (is<RenderMultiColumnFlow>(*child) || is<RenderMultiColumnSet>(*child))
630 builder.destroyAndCleanUpAnonymousWrappers(*child);
631 child = nextSibling;
632 }
633}
634
635void RenderTreeUpdater::tearDownLeftoverShadowHostChildren(Element& host, RenderTreeBuilder& builder)
636{
637 for (auto* hostChild = host.firstChild(); hostChild; hostChild = hostChild->nextSibling()) {
638 if (!hostChild->renderer())
639 continue;
640 if (is<Text>(*hostChild)) {
641 tearDownTextRenderer(downcast<Text>(*hostChild), builder);
642 continue;
643 }
644 if (is<Element>(*hostChild))
645 tearDownRenderers(downcast<Element>(*hostChild), TeardownType::Full, builder);
646 }
