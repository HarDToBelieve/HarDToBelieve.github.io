28#pragma once
29
30#include "FrameLoaderTypes.h"
31#include "ScrollTypes.h"
32#include <wtf/Forward.h>
33#include <wtf/Noncopyable.h>
34#include <wtf/OptionSet.h>
35
36namespace WebCore {
37
38class Frame;
39class FrameLoadRequest;
40class Page;
41class PlatformKeyboardEvent;
42class PlatformMouseEvent;
43class PlatformWheelEvent;
44
45enum class WheelEventProcessingSteps : uint8_t;
46
47// Real user inputs come from WebKit or WebKit2.
48// Synthetic inputs come from within WebCore (i.e., from web replay or fake mouse moves).
49enum class InputSource {
50 User,
51 Synthetic
52};
53
54class UserInputBridge {
55 WTF_MAKE_FAST_ALLOCATED;
56 WTF_MAKE_NONCOPYABLE(UserInputBridge);
57public:
58 UserInputBridge(Page&);
59
60 // User input APIs.
61#if ENABLE(CONTEXT_MENU_EVENT)
62 WEBCORE_EXPORT bool handleContextMenuEvent(const PlatformMouseEvent&, Frame&, InputSource = InputSource::User);
63#endif
64 WEBCORE_EXPORT bool handleMousePressEvent(const PlatformMouseEvent&, InputSource = InputSource::User);
65 WEBCORE_EXPORT bool handleMouseReleaseEvent(const PlatformMouseEvent&, InputSource = InputSource::User);
66 WEBCORE_EXPORT bool handleMouseMoveEvent(const PlatformMouseEvent&, InputSource = InputSource::User);
67 WEBCORE_EXPORT bool handleMouseMoveOnScrollbarEvent(const PlatformMouseEvent&, InputSource = InputSource::User);
68 WEBCORE_EXPORT bool handleMouseForceEvent(const PlatformMouseEvent&, InputSource = InputSource::User);
69 WEBCORE_EXPORT bool handleWheelEvent(const PlatformWheelEvent&, OptionSet<WheelEventProcessingSteps>, InputSource = InputSource::User);
70 WEBCORE_EXPORT bool handleKeyEvent(const PlatformKeyboardEvent&, InputSource = InputSource::User);
71 WEBCORE_EXPORT bool handleAccessKeyEvent(const PlatformKeyboardEvent&, InputSource = InputSource::User);
72 void focusSetActive(bool active, InputSource = InputSource::User);
73 void focusSetFocused(bool focused, InputSource = InputSource::User);
74 WEBCORE_EXPORT bool scrollRecursively(ScrollDirection, ScrollGranularity, InputSource = InputSource::User);
75 WEBCORE_EXPORT bool logicalScrollRecursively(ScrollLogicalDirection, ScrollGranularity, InputSource = InputSource::User);
76
77 // Navigation APIs.
78 WEBCORE_EXPORT void loadRequest(FrameLoadRequest&&, InputSource = InputSource::User);
79 WEBCORE_EXPORT void reloadFrame(Frame&, OptionSet<ReloadOption>, InputSource = InputSource::User);
80 WEBCORE_EXPORT void stopLoadingFrame(Frame&, InputSource = InputSource::User);
81 WEBCORE_EXPORT bool tryClosePage(InputSource = InputSource::User);
82
83private:
84 Page& m_page;
85};
86
87} // namespace WebCore
