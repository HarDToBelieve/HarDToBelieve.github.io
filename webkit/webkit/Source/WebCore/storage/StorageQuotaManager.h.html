34namespace WebCore {
35
36class StorageQuotaManager : public ThreadSafeRefCounted<StorageQuotaManager>, public CanMakeWeakPtr<StorageQuotaManager> {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 using UsageGetter = Function<uint64_t()>;
40 using QuotaIncreaseRequester = Function<void(uint64_t currentQuota, uint64_t currentUsage, uint64_t requestedIncrease, CompletionHandler<void(std::optional<uint64_t>)>&&)>;
41 WEBCORE_EXPORT static Ref<StorageQuotaManager> create(uint64_t quota, UsageGetter&&, QuotaIncreaseRequester&&);
42
43 static constexpr uint64_t defaultThirdPartyQuotaFromPerOriginQuota(uint64_t quota) { return quota / 10; }
44 static constexpr uint64_t defaultQuota() { return 1000 * MB; }
45 static constexpr uint64_t defaultThirdPartyQuota() { return defaultThirdPartyQuotaFromPerOriginQuota(defaultQuota()); }
46
47 enum class Decision { Deny, Grant };
48 using RequestCallback = CompletionHandler<void(Decision)>;
49 WEBCORE_EXPORT void requestSpaceOnMainThread(uint64_t, RequestCallback&&);
50 WEBCORE_EXPORT Decision requestSpaceOnBackgroundThread(uint64_t);
51
52 WEBCORE_EXPORT void resetQuotaUpdatedBasedOnUsageForTesting();
53 WEBCORE_EXPORT void resetQuotaForTesting();
54private:
55 StorageQuotaManager(uint64_t quota, UsageGetter&&, QuotaIncreaseRequester&&);
56 bool tryGrantRequest(uint64_t) WTF_REQUIRES_LOCK(m_quotaCountDownLock);
57
58 void updateQuotaBasedOnUsage();
59
60 Lock m_quotaCountDownLock;
61 uint64_t m_quotaCountDown WTF_GUARDED_BY_LOCK(m_quotaCountDownLock) { 0 };
62 uint64_t m_quota { 0 };
63 uint64_t m_usage { 0 };
64
65 UsageGetter m_usageGetter;
66 QuotaIncreaseRequester m_quotaIncreaseRequester;
67
68 Ref<WorkQueue> m_workQueue;
69
70 bool m_quotaUpdatedBasedOnUsage { false };
71
72 // Test only.
73 uint64_t m_initialQuota { 0 };
74};

76} // namespace WebCore
