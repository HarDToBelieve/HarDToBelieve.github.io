26#pragma once
27
28#include "Element.h"
29#include "StyleInvalidator.h"
30
31namespace WebCore {
32namespace Style {
33
34class AttributeChangeInvalidation {
35public:
36 AttributeChangeInvalidation(Element&, const QualifiedName&, const AtomString& oldValue, const AtomString& newValue);
37 ~AttributeChangeInvalidation();
38
39private:
40 void invalidateStyle(const QualifiedName&, const AtomString& oldValue, const AtomString& newValue);
41 void invalidateStyleWithRuleSets();
42
43 const bool m_isEnabled;
44 Element& m_element;
45
46 Invalidator::MatchElementRuleSets m_matchElementRuleSets;
47};
48
49inline AttributeChangeInvalidation::AttributeChangeInvalidation(Element& element, const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
50 : m_isEnabled(element.needsStyleInvalidation())
51 , m_element(element)
52{
53 if (!m_isEnabled)
54 return;
55 invalidateStyle(attributeName, oldValue, newValue);
56 invalidateStyleWithRuleSets();
57}
58
59inline AttributeChangeInvalidation::~AttributeChangeInvalidation()
60{
61 if (!m_isEnabled)
62 return;
63 invalidateStyleWithRuleSets();
64}
65
66}
