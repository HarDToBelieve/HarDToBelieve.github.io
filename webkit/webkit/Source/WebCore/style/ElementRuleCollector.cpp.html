53static const StyleProperties& leftToRightDeclaration()
54{
55 static auto& declaration = [] () -> const StyleProperties& {
56 auto properties = MutableStyleProperties::create();
57 properties->setProperty(CSSPropertyDirection, CSSValueLtr);
58 return properties.leakRef();
59 }();
60 return declaration;
61}

63static const StyleProperties& rightToLeftDeclaration()
64{
65 static auto& declaration = [] () -> const StyleProperties& {
66 auto properties = MutableStyleProperties::create();
67 properties->setProperty(CSSPropertyDirection, CSSValueRtl);
68 return properties.leakRef();
69 }();
70 return declaration;
71}

137void ElementRuleCollector::collectMatchingRules(const MatchRequest& matchRequest)
138{
139 ASSERT(matchRequest.ruleSet);
140 ASSERT_WITH_MESSAGE(!(m_mode == SelectorChecker::Mode::CollectingRulesIgnoringVirtualPseudoElements && m_pseudoElementRequest.pseudoId != PseudoId::None), "When in StyleInvalidation or SharingRules, SelectorChecker does not try to match the pseudo ID. While ElementRuleCollector supports matching a particular pseudoId in this case, this would indicate a error at the call site since matching a particular element should be unnecessary.");
141
142 auto* shadowRoot = element().containingShadowRoot();
143 if (shadowRoot && shadowRoot->mode() == ShadowRootMode::UserAgent)
144 collectMatchingShadowPseudoElementRules(matchRequest);
145
146 // We need to collect the rules for id, class, tag, and everything else into a buffer and
147 // then sort the buffer.
148 auto& id = element().idForStyleResolution();
149 if (!id.isNull())
150 collectMatchingRulesForList(matchRequest.ruleSet->idRules(id), matchRequest);
151 if (element().hasClass()) {
152 for (size_t i = 0; i < element().classNames().size(); ++i)
153 collectMatchingRulesForList(matchRequest.ruleSet->classRules(element().classNames()[i]), matchRequest);
154 }
155
156 if (element().isLink())
157 collectMatchingRulesForList(matchRequest.ruleSet->linkPseudoClassRules(), matchRequest);
158 if (matchesFocusPseudoClass(element()))
159 collectMatchingRulesForList(matchRequest.ruleSet->focusPseudoClassRules(), matchRequest);
160 collectMatchingRulesForList(matchRequest.ruleSet->tagRules(element().localName(), element().isHTMLElement() && element().document().isHTMLDocument()), matchRequest);
161 collectMatchingRulesForList(matchRequest.ruleSet->universalRules(), matchRequest);
162}

186void ElementRuleCollector::transferMatchedRules(DeclarationOrigin declarationOrigin, std::optional<ScopeOrdinal> fromScope)
187{
188 if (m_mode != SelectorChecker::Mode::CollectingRules)
189 declarationsForOrigin(m_result, declarationOrigin).reserveCapacity(m_matchedRules.size());
190
191 for (; m_matchedRuleTransferIndex < m_matchedRules.size(); ++m_matchedRuleTransferIndex) {
192 auto& matchedRule = m_matchedRules[m_matchedRuleTransferIndex];
193 if (fromScope && matchedRule.styleScopeOrdinal < *fromScope)
194 break;
195
196 if (m_mode == SelectorChecker::Mode::CollectingRules) {
197 m_matchedRuleList.append(&matchedRule.ruleData->styleRule());
198 continue;
199 }
200
201 addMatchedProperties({
202 &matchedRule.ruleData->styleRule().properties(),
203 static_cast<uint16_t>(matchedRule.ruleData->linkMatchType()),
204 static_cast<uint16_t>(matchedRule.ruleData->propertyAllowlistType()),
205 matchedRule.styleScopeOrdinal
206 }, declarationOrigin);
207 }

229void ElementRuleCollector::collectMatchingAuthorRules()
230{
231 {
232 MatchRequest matchRequest(m_authorStyle.ptr());
233 collectMatchingRules(matchRequest);
234 }
235
236 auto* parent = element().parentElement();
237 if (parent && parent->shadowRoot())
238 matchSlottedPseudoElementRules();
239
240 if (element().shadowRoot())
241 matchHostPseudoClassRules();
242
243 if (element().isInShadowTree()) {
244 matchAuthorShadowPseudoElementRules();
245 matchPartPseudoElementRules();
246 }

249void ElementRuleCollector::matchAuthorShadowPseudoElementRules()
250{
251 ASSERT(element().isInShadowTree());
252 auto& shadowRoot = *element().containingShadowRoot();
253 if (shadowRoot.mode() != ShadowRootMode::UserAgent)
254 return;
255 // Look up shadow pseudo elements also from the host scope author style as they are web-exposed.
256 auto& hostAuthorRules = Scope::forNode(*shadowRoot.host()).resolver().ruleSets().authorStyle();
257 MatchRequest hostAuthorRequest { &hostAuthorRules, ScopeOrdinal::ContainingHost };
258 collectMatchingShadowPseudoElementRules(hostAuthorRequest);
259}

261void ElementRuleCollector::matchHostPseudoClassRules()
262{
263 ASSERT(element().shadowRoot());
264
265 auto& shadowAuthorStyle = element().shadowRoot()->styleScope().resolver().ruleSets().authorStyle();
266 auto& shadowHostRules = shadowAuthorStyle.hostPseudoClassRules();
267 if (shadowHostRules.isEmpty())
268 return;
269
270 SetForScope<bool> change(m_isMatchingHostPseudoClass, true);
271
272 MatchRequest hostMatchRequest { nullptr, ScopeOrdinal::Shadow };
273 collectMatchingRulesForList(&shadowHostRules, hostMatchRequest);
274}

276void ElementRuleCollector::matchSlottedPseudoElementRules()
277{
278 auto* slot = element().assignedSlot();
279 auto styleScopeOrdinal = ScopeOrdinal::FirstSlot;
280
281 for (; slot; slot = slot->assignedSlot(), ++styleScopeOrdinal) {
282 auto& styleScope = Scope::forNode(*slot);
283 if (!styleScope.resolver().ruleSets().isAuthorStyleDefined())
284 continue;
285 // Find out if there are any ::slotted rules in the shadow tree matching the current slot.
286 // FIXME: This is really part of the slot style and could be cached when resolving it.
287 ElementRuleCollector collector(*slot, styleScope.resolver().ruleSets().authorStyle(), nullptr);
288 auto slottedPseudoElementRules = collector.collectSlottedPseudoElementRulesForSlot();
289 if (!slottedPseudoElementRules)
290 continue;
291 // Match in the current scope.
292 SetForScope<bool> change(m_isMatchingSlottedPseudoElements, true);
293
294 MatchRequest scopeMatchRequest(nullptr, styleScopeOrdinal);
295 collectMatchingRulesForList(slottedPseudoElementRules.get(), scopeMatchRequest);
296
297 m_keepAliveSlottedPseudoElementRules.append(WTFMove(slottedPseudoElementRules));
298 }
299}

301void ElementRuleCollector::matchPartPseudoElementRules()
302{
303 ASSERT(element().isInShadowTree());
304
305 bool isUAShadowPseudoElement = element().containingShadowRoot()->mode() == ShadowRootMode::UserAgent && !element().shadowPseudoId().isNull();
306
307 auto& partMatchingElement = isUAShadowPseudoElement ? *element().shadowHost() : element();
308 if (partMatchingElement.partNames().isEmpty() || !partMatchingElement.isInShadowTree())
309 return;
310
311 matchPartPseudoElementRulesForScope(*partMatchingElement.containingShadowRoot());
312}

314void ElementRuleCollector::matchPartPseudoElementRulesForScope(const ShadowRoot& scopeShadowRoot)
315{
316 auto& shadowHost = *scopeShadowRoot.host();
317 {
318 SetForScope<RefPtr<const Element>> partMatchingScope(m_shadowHostInPartRuleScope, &shadowHost);
319
320 auto& hostAuthorRules = Scope::forNode(shadowHost).resolver().ruleSets().authorStyle();
321 MatchRequest hostAuthorRequest { &hostAuthorRules, ScopeOrdinal::ContainingHost };
322 collectMatchingRulesForList(&hostAuthorRules.partPseudoElementRules(), hostAuthorRequest);
323 }
324
325 // Element may be exposed to styling from enclosing scopes via exportparts attributes.
326 if (scopeShadowRoot.partMappings().isEmpty())
327 return;
328
329 if (auto* parentScopeShadowRoot = shadowHost.containingShadowRoot())
330 matchPartPseudoElementRulesForScope(*parentScopeShadowRoot);
331}

333void ElementRuleCollector::collectMatchingShadowPseudoElementRules(const MatchRequest& matchRequest)
334{
335 ASSERT(matchRequest.ruleSet);
336 ASSERT(element().containingShadowRoot()->mode() == ShadowRootMode::UserAgent);
337
338 auto& rules = *matchRequest.ruleSet;
339#if ENABLE(VIDEO)
340 // FXIME: WebVTT should not be done by styling UA shadow trees like this.
341 if (element().isWebVTTElement())
342 collectMatchingRulesForList(&rules.cuePseudoRules(), matchRequest);
343#endif
344 auto& pseudoId = element().shadowPseudoId();
345 if (!pseudoId.isEmpty())
346 collectMatchingRulesForList(rules.shadowPseudoElementRules(pseudoId), matchRequest);
347}

349std::unique_ptr<RuleSet::RuleDataVector> ElementRuleCollector::collectSlottedPseudoElementRulesForSlot()
350{
351 ASSERT(is<HTMLSlotElement>(element()));
352
353 clearMatchedRules();
354
355 m_mode = SelectorChecker::Mode::CollectingRules;
356
357 // Match global author rules.
358 MatchRequest matchRequest(m_authorStyle.ptr());
359 collectMatchingRulesForList(&m_authorStyle->slottedPseudoElementRules(), matchRequest);
360
361 if (m_matchedRules.isEmpty())
362 return { };
363
364 auto ruleDataVector = makeUnique<RuleSet::RuleDataVector>();
365 ruleDataVector->reserveInitialCapacity(m_matchedRules.size());
366 for (auto& matchedRule : m_matchedRules)
367 ruleDataVector->uncheckedAppend(*matchedRule.ruleData);
368
369 return ruleDataVector;
370}

385void ElementRuleCollector::matchUARules()
386{
387 // First we match rules from the user agent sheet.
388 auto* userAgentStyleSheet = m_isPrintStyle
389 ? UserAgentStyle::defaultPrintStyle : UserAgentStyle::defaultStyle;
390 matchUARules(*userAgentStyleSheet);
391

409static const CSSSelector* findSlottedPseudoElementSelector(const CSSSelector* selector)
410{
411 for (; selector; selector = selector->tagHistory()) {
412 if (selector->match() == CSSSelector::PseudoElement && selector->pseudoElementType() == CSSSelector::PseudoElementSlotted) {
413 if (auto* list = selector->selectorList())
414 return list->first();
415 break;
416 }
417 };
418 return nullptr;
419}

421inline bool ElementRuleCollector::ruleMatches(const RuleData& ruleData, unsigned& specificity)
422{
423 // We know a sufficiently simple single part selector matches simply because we found it from the rule hash when filtering the RuleSet.
424 // This is limited to HTML only so we don't need to check the namespace (because of tag name match).
425 auto matchBasedOnRuleHash = ruleData.matchBasedOnRuleHash();
426 if (matchBasedOnRuleHash != MatchBasedOnRuleHash::None && element().isHTMLElement()) {
427 ASSERT_WITH_MESSAGE(m_pseudoElementRequest.pseudoId == PseudoId::None, "If we match based on the rule hash while collecting for a particular pseudo element ID, we would add incorrect rules for that pseudo element ID. We should never end in ruleMatches() with a pseudo element if the ruleData cannot match any pseudo element.");
428
429 switch (matchBasedOnRuleHash) {
430 case MatchBasedOnRuleHash::None:
431 ASSERT_NOT_REACHED();
432 break;
433 case MatchBasedOnRuleHash::Universal:
434 specificity = 0;
435 break;
436 case MatchBasedOnRuleHash::ClassA:
437 specificity = static_cast<unsigned>(SelectorSpecificityIncrement::ClassA);
438 break;
439 case MatchBasedOnRuleHash::ClassB:
440 specificity = static_cast<unsigned>(SelectorSpecificityIncrement::ClassB);
441 break;
442 case MatchBasedOnRuleHash::ClassC:
443 specificity = static_cast<unsigned>(SelectorSpecificityIncrement::ClassC);
444 break;
445 }
446 return true;
447 }
448
449#if ENABLE(CSS_SELECTOR_JIT)
450 auto& compiledSelector = ruleData.compiledSelector();
451
452 if (compiledSelector.status == SelectorCompilationStatus::NotCompiled)
453 SelectorCompiler::compileSelector(compiledSelector, ruleData.selector(), SelectorCompiler::SelectorContext::RuleCollector);
454
455 if (compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker) {
456 compiledSelector.wasUsed();
457
458#if !ASSERT_MSG_DISABLED
459 unsigned ignoreSpecificity;
460 ASSERT_WITH_MESSAGE(!SelectorCompiler::ruleCollectorSimpleSelectorChecker(compiledSelector, &element(), &ignoreSpecificity) || m_pseudoElementRequest.pseudoId == PseudoId::None, "When matching pseudo elements, we should never compile a selector checker without context unless it cannot match anything.");
461#endif
462 bool selectorMatches = SelectorCompiler::ruleCollectorSimpleSelectorChecker(compiledSelector, &element(), &specificity);
463
464 if (selectorMatches && ruleData.containsUncommonAttributeSelector())
465 m_didMatchUncommonAttributeSelector = true;
466
467 return selectorMatches;
468 }
469#endif // ENABLE(CSS_SELECTOR_JIT)
470
471 SelectorChecker::CheckingContext context(m_mode);
472 context.pseudoId = m_pseudoElementRequest.pseudoId;
473 context.scrollbarState = m_pseudoElementRequest.scrollbarState;
474 context.nameForHightlightPseudoElement = m_pseudoElementRequest.highlightName;
475 context.isMatchingHostPseudoClass = m_isMatchingHostPseudoClass;
476 context.shadowHostInPartRuleScope = m_shadowHostInPartRuleScope.get();
477
478 bool selectorMatches;
479#if ENABLE(CSS_SELECTOR_JIT)
480 if (compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext) {
481 compiledSelector.wasUsed();
482 selectorMatches = SelectorCompiler::ruleCollectorSelectorCheckerWithCheckingContext(compiledSelector, &element(), &context, &specificity);
483 } else
484#endif // ENABLE(CSS_SELECTOR_JIT)
485 {
486 auto* selector = ruleData.selector();
487 if (m_isMatchingSlottedPseudoElements) {
488 selector = findSlottedPseudoElementSelector(ruleData.selector());
489 if (!selector)
490 return false;
491 }
492 // Slow path.
493 SelectorChecker selectorChecker(element().document());
494 selectorMatches = selectorChecker.match(*selector, element(), context);
495 if (selectorMatches)
496 specificity = selector->computeSpecificity();
497 }
498
499 if (ruleData.containsUncommonAttributeSelector()) {
500 if (selectorMatches || context.pseudoIDSet)
501 m_didMatchUncommonAttributeSelector = true;
502 }
503 m_matchedPseudoElementIds.merge(context.pseudoIDSet);
504 m_styleRelations.appendVector(context.styleRelations);
505
506 return selectorMatches;
507}

509void ElementRuleCollector::collectMatchingRulesForList(const RuleSet::RuleDataVector* rules, const MatchRequest& matchRequest)
510{
511 if (!rules)
512 return;
513
514 for (unsigned i = 0, size = rules->size(); i < size; ++i) {
515 const auto& ruleData = rules->data()[i];
516
517 if (UNLIKELY(!ruleData.isEnabled()))
518 continue;
519
520 if (!ruleData.canMatchPseudoElement() && m_pseudoElementRequest.pseudoId != PseudoId::None)
521 continue;
522
523 if (m_selectorFilter && m_selectorFilter->fastRejectSelector(ruleData.descendantSelectorIdentifierHashes()))
524 continue;
525
526 auto& rule = ruleData.styleRule();
527
528 // If the rule has no properties to apply, then ignore it in the non-debug mode.
529 // Note that if we get null back here, it means we have a rule with deferred properties,
530 // and that means we always have to consider it.
531 const StyleProperties* properties = rule.propertiesWithoutDeferredParsing();
532 if (properties && properties->isEmpty() && !m_shouldIncludeEmptyRules)
533 continue;
534
535 unsigned specificity;
536 if (ruleMatches(ruleData, specificity))
537 addMatchedRule(ruleData, specificity, matchRequest.styleScopeOrdinal);
538 }
539}

541static inline bool compareRules(MatchedRule r1, MatchedRule r2)
542{
543 // For normal properties the earlier scope wins. This may be reversed by !important which is handled when resolving cascade.
544 if (r1.styleScopeOrdinal != r2.styleScopeOrdinal)
545 return r1.styleScopeOrdinal > r2.styleScopeOrdinal;
546
547 if (r1.specificity != r2.specificity)
548 return r1.specificity < r2.specificity;
549
550 return r1.ruleData->position() < r2.ruleData->position();
551}

558void ElementRuleCollector::matchAllRules(bool matchAuthorAndUserStyles, bool includeSMILProperties)
559{
560 matchUARules();
561
562 // Now we check user sheet rules.
563 if (matchAuthorAndUserStyles)
564 matchUserRules();
565
566 if (is<StyledElement>(element())) {
567 auto& styledElement = downcast<StyledElement>(element());
568 // https://html.spec.whatwg.org/#presentational-hints
569 addElementStyleProperties(styledElement.presentationalHintStyle());
570
571 // Tables and table cells share an additional presentation style that must be applied
572 // after all attributes, since their style depends on the values of multiple attributes.
573 addElementStyleProperties(styledElement.additionalPresentationalHintStyle());
574
575 if (is<HTMLElement>(styledElement)) {
576 bool isAuto;
577 auto textDirection = downcast<HTMLElement>(styledElement).directionalityIfhasDirAutoAttribute(isAuto);
578 auto& properties = textDirection == TextDirection::LTR ? leftToRightDeclaration() : rightToLeftDeclaration();
579 if (isAuto)
580 addMatchedProperties({ &properties }, DeclarationOrigin::Author);
581 }
582 }
583
584 if (matchAuthorAndUserStyles) {
585 clearMatchedRules();
586
587 collectMatchingAuthorRules();
588 sortMatchedRules();
589
590 transferMatchedRules(DeclarationOrigin::Author, ScopeOrdinal::Element);
591
592 // Inline style behaves as if it has higher specificity than any rule.
593 addElementInlineStyleProperties(includeSMILProperties);
594
595 // Rules from the host scope override inline style.
596 transferMatchedRules(DeclarationOrigin::Author, ScopeOrdinal::ContainingHost);
597 }
598}

600void ElementRuleCollector::addElementInlineStyleProperties(bool includeSMILProperties)
601{
602 if (!is<StyledElement>(element()))
603 return;
604
605 if (auto* inlineStyle = downcast<StyledElement>(element()).inlineStyle()) {
606 // FIXME: Media control shadow trees seem to have problems with caching.
607 bool isInlineStyleCacheable = !inlineStyle->isMutable() && !element().isInShadowTree();
608 addElementStyleProperties(inlineStyle, isInlineStyleCacheable);
609 }
610
611 if (includeSMILProperties && is<SVGElement>(element()))
612 addElementStyleProperties(downcast<SVGElement>(element()).animatedSMILStyleProperties(), false /* isCacheable */);
613}

625void ElementRuleCollector::addMatchedProperties(MatchedProperties&& matchedProperties, DeclarationOrigin declarationOrigin)
626{
627 // FIXME: This should be moved to the matched properties cache code.
628 auto computeIsCacheable = [&] {
629 if (!m_result.isCacheable)
630 return false;
631
632 if (matchedProperties.styleScopeOrdinal != ScopeOrdinal::Element)
633 return false;
634
635 auto& properties = *matchedProperties.properties;
636 for (unsigned i = 0, count = properties.propertyCount(); i < count; ++i) {
637 // Currently the property cache only copy the non-inherited values and resolve
638 // the inherited ones.
639 // Here we define some exception were we have to resolve some properties that are not inherited
640 // by default. If those exceptions become too common on the web, it should be possible
641 // to build a list of exception to resolve instead of completely disabling the cache.
642 StyleProperties::PropertyReference current = properties.propertyAt(i);
643 if (current.isInherited())
644 continue;
645
646 // If the property value is explicitly inherited, we need to apply further non-inherited properties
647 // as they might override the value inherited here. For this reason we don't allow declarations with
648 // explicitly inherited properties to be cached.
649 const CSSValue& value = *current.value();
650 if (value.isInheritedValue())
651 return false;
652
653 // The value currentColor has implicitely the same side effect. It depends on the value of color,
654 // which is an inherited value, making the non-inherited property implicitly inherited.
655 if (is<CSSPrimitiveValue>(value) && downcast<CSSPrimitiveValue>(value).valueID() == CSSValueCurrentcolor)
656 return false;
657
658 if (value.hasVariableReferences())
659 return false;
660 }
661
662 return true;
663 };
664
665 m_result.isCacheable = computeIsCacheable();
666
667 declarationsForOrigin(m_result, declarationOrigin).append(WTFMove(matchedProperties));
668}
