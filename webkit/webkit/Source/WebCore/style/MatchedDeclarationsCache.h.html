27
28#include "ElementRuleCollector.h"
29#include "RenderStyle.h"
30#include "Timer.h"

32namespace WebCore {
33
34namespace Style {
35
36class MatchedDeclarationsCache {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 MatchedDeclarationsCache();
40 ~MatchedDeclarationsCache();
41
42 static bool isCacheable(const Element&, const RenderStyle&, const RenderStyle& parentStyle);
43 static unsigned computeHash(const MatchResult&);
44
45 struct Entry {
46 MatchResult matchResult;
47 std::unique_ptr<const RenderStyle> renderStyle;
48 std::unique_ptr<const RenderStyle> parentRenderStyle;
49
50 bool isUsableAfterHighPriorityProperties(const RenderStyle&) const;
51 };
52
53 const Entry* find(unsigned hash, const MatchResult&);
54 void add(const RenderStyle&, const RenderStyle& parentStyle, unsigned hash, const MatchResult&);
55
56 // Every N additions to the matched declaration cache trigger a sweep where entries holding
57 // the last reference to a style declaration are garbage collected.
58 void invalidate();
59 void clearEntriesAffectedByViewportUnits();
60
61private:
62 void sweep();
63
64 HashMap<unsigned, Entry> m_entries;
65 Timer m_sweepTimer;
66 unsigned m_additionsSinceLastSweep { 0 };
67};
68
69}
