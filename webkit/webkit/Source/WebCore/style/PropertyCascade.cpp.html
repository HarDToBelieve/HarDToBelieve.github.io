180void PropertyCascade::buildCascade(OptionSet<CascadeLevel> cascadeLevels)
181{
182 OptionSet<CascadeLevel> cascadeLevelsWithImportant;
183
184 for (auto cascadeLevel : cascadeLevels) {
185 bool hasImportant = addNormalMatches(cascadeLevel);
186 if (hasImportant)
187 cascadeLevelsWithImportant.add(cascadeLevel);
188 }
189
190 for (auto cascadeLevel : { CascadeLevel::Author, CascadeLevel::User, CascadeLevel::UserAgent }) {
191 if (!cascadeLevelsWithImportant.contains(cascadeLevel))
192 continue;
193 addImportantMatches(cascadeLevel);
194 }
195}

197void PropertyCascade::setPropertyInternal(Property& property, CSSPropertyID id, CSSValue& cssValue, unsigned linkMatchType, CascadeLevel cascadeLevel, ScopeOrdinal styleScopeOrdinal)
198{
199 ASSERT(linkMatchType <= SelectorChecker::MatchAll);
200 property.id = id;
201 property.level = cascadeLevel;
202 property.styleScopeOrdinal = styleScopeOrdinal;
203 if (linkMatchType == SelectorChecker::MatchAll) {
204 property.cssValue[0] = &cssValue;
205 property.cssValue[SelectorChecker::MatchLink] = &cssValue;
206 property.cssValue[SelectorChecker::MatchVisited] = &cssValue;
207 } else
208 property.cssValue[linkMatchType] = &cssValue;
209}

211void PropertyCascade::set(CSSPropertyID id, CSSValue& cssValue, unsigned linkMatchType, CascadeLevel cascadeLevel, ScopeOrdinal styleScopeOrdinal)
212{
213 if (CSSProperty::isDirectionAwareProperty(id)) {
214 auto direction = this->direction();
215 id = CSSProperty::resolveDirectionAwareProperty(id, direction.textDirection, direction.writingMode);
216 }
217
218 ASSERT(!shouldApplyPropertyInParseOrder(id));
219
220 auto& property = m_properties[id];
221 ASSERT(id < m_propertyIsPresent.size());
222 if (id == CSSPropertyCustom) {
223 m_propertyIsPresent.set(id);
224 const auto& customValue = downcast<CSSCustomPropertyValue>(cssValue);
225 bool hasValue = m_customProperties.contains(customValue.name());
226 if (!hasValue) {
227 Property property;
228 property.id = id;
229 memset(property.cssValue, 0, sizeof(property.cssValue));
230 setPropertyInternal(property, id, cssValue, linkMatchType, cascadeLevel, styleScopeOrdinal);
231 m_customProperties.set(customValue.name(), property);
232 } else {
233 Property property = customProperty(customValue.name());
234 setPropertyInternal(property, id, cssValue, linkMatchType, cascadeLevel, styleScopeOrdinal);
235 m_customProperties.set(customValue.name(), property);
236 }
237 return;
238 }
239
240 if (!m_propertyIsPresent[id])
241 memset(property.cssValue, 0, sizeof(property.cssValue));
242 m_propertyIsPresent.set(id);
243 setPropertyInternal(property, id, cssValue, linkMatchType, cascadeLevel, styleScopeOrdinal);
244}

246void PropertyCascade::setDeferred(CSSPropertyID id, CSSValue& cssValue, unsigned linkMatchType, CascadeLevel cascadeLevel, ScopeOrdinal styleScopeOrdinal)
247{
248 ASSERT(!CSSProperty::isDirectionAwareProperty(id));
249 ASSERT(shouldApplyPropertyInParseOrder(id));
250
251 Property property;
252 memset(property.cssValue, 0, sizeof(property.cssValue));
253 setPropertyInternal(property, id, cssValue, linkMatchType, cascadeLevel, styleScopeOrdinal);
254 m_deferredProperties.append(property);
255}

258bool PropertyCascade::addMatch(const MatchedProperties& matchedProperties, CascadeLevel cascadeLevel, bool important)
259{
260 auto& styleProperties = *matchedProperties.properties;
261 auto propertyAllowlistType = static_cast<PropertyAllowlistType>(matchedProperties.allowlistType);
262 bool hasImportantProperties = false;
263
264 for (unsigned i = 0, count = styleProperties.propertyCount(); i < count; ++i) {
265 auto current = styleProperties.propertyAt(i);
266
267 if (current.isImportant())
268 hasImportantProperties = true;
269 if (important != current.isImportant())
270 continue;
271
272 if (m_includedProperties == IncludedProperties::InheritedOnly && !current.isInherited()) {
273 // Inherited only mode is used after matched properties cache hit.
274 // A match with a value that is explicitly inherited should never have been cached.
275 ASSERT(!current.value()->isInheritedValue());
276 continue;
277 }
278 CSSPropertyID propertyID = current.id();
279
280#if ENABLE(VIDEO)
281 if (propertyAllowlistType == PropertyAllowlistCue && !isValidCueStyleProperty(propertyID))
282 continue;
283#endif
284 if (propertyAllowlistType == PropertyAllowlistMarker && !isValidMarkerStyleProperty(propertyID))
285 continue;
286
287 if (shouldApplyPropertyInParseOrder(propertyID))
288 setDeferred(propertyID, *current.value(), matchedProperties.linkMatchType, cascadeLevel, matchedProperties.styleScopeOrdinal);
289 else
290 set(propertyID, *current.value(), matchedProperties.linkMatchType, cascadeLevel, matchedProperties.styleScopeOrdinal);
291 }
292
293 return hasImportantProperties;
294}

296static auto& declarationsForCascadeLevel(const MatchResult& matchResult, CascadeLevel cascadeLevel)
297{
298 switch (cascadeLevel) {
299 case CascadeLevel::UserAgent: return matchResult.userAgentDeclarations;
300 case CascadeLevel::User: return matchResult.userDeclarations;
301 case CascadeLevel::Author: return matchResult.authorDeclarations;
302 }
303 ASSERT_NOT_REACHED();
304 return matchResult.authorDeclarations;
305}

307bool PropertyCascade::addNormalMatches(CascadeLevel cascadeLevel)
308{
309 bool hasImportant = false;
310 for (auto& matchedDeclarations : declarationsForCascadeLevel(m_matchResult, cascadeLevel))
311 hasImportant |= addMatch(matchedDeclarations, cascadeLevel, false);
312
313 return hasImportant;
314}

316static bool hasImportantProperties(const StyleProperties& properties)
317{
318 for (unsigned i = 0, count = properties.propertyCount(); i < count; ++i) {
319 if (properties.propertyAt(i).isImportant())
320 return true;
321 }
322 return false;
323}

325void PropertyCascade::addImportantMatches(CascadeLevel cascadeLevel)
326{
327 struct IndexAndOrdinal {
328 unsigned index;
329 ScopeOrdinal ordinal;
330 };
331 Vector<IndexAndOrdinal> importantMatches;
332 bool hasMatchesFromOtherScopes = false;
333
334 auto& matchedDeclarations = declarationsForCascadeLevel(m_matchResult, cascadeLevel);
335
336 for (unsigned i = 0; i < matchedDeclarations.size(); ++i) {
337 const MatchedProperties& matchedProperties = matchedDeclarations[i];
338
339 if (!hasImportantProperties(*matchedProperties.properties))
340 continue;
341
342 importantMatches.append({ i, matchedProperties.styleScopeOrdinal });
343
344 if (matchedProperties.styleScopeOrdinal != ScopeOrdinal::Element)
345 hasMatchesFromOtherScopes = true;
346 }
347
348 if (importantMatches.isEmpty())
349 return;
350
351 if (hasMatchesFromOtherScopes) {
352 // For !important properties a later shadow tree wins.
353 // Match results are sorted in reverse tree context order so this is not needed for normal properties.
354 std::stable_sort(importantMatches.begin(), importantMatches.end(), [] (const IndexAndOrdinal& a, const IndexAndOrdinal& b) {
355 return a.ordinal < b.ordinal;
356 });
357 }
358
359 for (auto& match : importantMatches)
360 addMatch(matchedDeclarations[match.index], cascadeLevel, true);
361}

363const PropertyCascade* PropertyCascade::propertyCascadeForRollback(CascadeLevel cascadeLevel) const
364{
365 switch (cascadeLevel) {
366 case CascadeLevel::Author:
367 if (!m_authorRollbackCascade) {
368 auto cascadeLevels = OptionSet<CascadeLevel> { CascadeLevel::UserAgent, CascadeLevel::User };
369 m_authorRollbackCascade = makeUnique<const PropertyCascade>(*this, cascadeLevels);
370 }
371 return m_authorRollbackCascade.get();
372
373 case CascadeLevel::User:
374 if (!m_userRollbackCascade) {
375 auto cascadeLevels = OptionSet<CascadeLevel> { CascadeLevel::UserAgent };
376 m_userRollbackCascade = makeUnique<const PropertyCascade>(*this, cascadeLevels);
377 }
378 return m_userRollbackCascade.get();
379
380 case CascadeLevel::UserAgent:
381 return nullptr;
382 }
383 ASSERT_NOT_REACHED();
384 return nullptr;
385}

387PropertyCascade::Direction PropertyCascade::resolveDirectionAndWritingMode(Direction inheritedDirection) const
388{
389 Direction result = inheritedDirection;
390
391 bool hadImportantWritingMode = false;
392 bool hadImportantDirection = false;
393
394 for (auto cascadeLevel : { CascadeLevel::UserAgent, CascadeLevel::User, CascadeLevel::Author }) {
395 for (const auto& matchedProperties : declarationsForCascadeLevel(m_matchResult, cascadeLevel)) {
396 for (unsigned i = 0, count = matchedProperties.properties->propertyCount(); i < count; ++i) {
397 auto property = matchedProperties.properties->propertyAt(i);
398 if (!property.value()->isPrimitiveValue())
399 continue;
400 switch (property.id()) {
401 case CSSPropertyWritingMode:
402 if (!hadImportantWritingMode || property.isImportant()) {
403 result.writingMode = downcast<CSSPrimitiveValue>(*property.value());
404 hadImportantWritingMode = property.isImportant();
405 }
406 break;
407 case CSSPropertyDirection:
408 if (!hadImportantDirection || property.isImportant()) {
409 result.textDirection = downcast<CSSPrimitiveValue>(*property.value());
410 hadImportantDirection = property.isImportant();
411 }
412 break;
413 default:
414 break;
415 }
416 }
417 }
418 }
419
420 return result;
421}
422
423PropertyCascade::Direction PropertyCascade::direction() const
424{
425 if (m_directionIsUnresolved) {
426 m_direction = resolveDirectionAndWritingMode(m_direction);
427 m_directionIsUnresolved = false;
428 }
429 return m_direction;
430}
