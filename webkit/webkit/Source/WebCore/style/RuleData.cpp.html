64static inline MatchBasedOnRuleHash computeMatchBasedOnRuleHash(const CSSSelector& selector)
65{
66 if (selector.tagHistory())
67 return MatchBasedOnRuleHash::None;
68
69 if (selector.match() == CSSSelector::Tag) {
70 const QualifiedName& tagQualifiedName = selector.tagQName();
71 const AtomString& selectorNamespace = tagQualifiedName.namespaceURI();
72 if (selectorNamespace == starAtom() || selectorNamespace == xhtmlNamespaceURI) {
73 if (tagQualifiedName == anyQName())
74 return MatchBasedOnRuleHash::Universal;
75 return MatchBasedOnRuleHash::ClassC;
76 }
77 return MatchBasedOnRuleHash::None;
78 }
79 if (SelectorChecker::isCommonPseudoClassSelector(&selector))
80 return MatchBasedOnRuleHash::ClassB;
81 if (selector.match() == CSSSelector::Id)
82 return MatchBasedOnRuleHash::ClassA;
83 if (selector.match() == CSSSelector::Class)
84 return MatchBasedOnRuleHash::ClassB;
85 return MatchBasedOnRuleHash::None;
86}

88static bool selectorCanMatchPseudoElement(const CSSSelector& rootSelector)
89{
90 const CSSSelector* selector = &rootSelector;
91 do {
92 if (selector->matchesPseudoElement())
93 return true;
94
95 if (const CSSSelectorList* selectorList = selector->selectorList()) {
96 for (const CSSSelector* subSelector = selectorList->first(); subSelector; subSelector = CSSSelectorList::next(subSelector)) {
97 if (selectorCanMatchPseudoElement(*subSelector))
98 return true;
99 }
100 }
101
102 selector = selector->tagHistory();
103 } while (selector);
104 return false;
105}

107static inline bool isCommonAttributeSelectorAttribute(const QualifiedName& attribute)
108{
109 // These are explicitly tested for equality in canShareStyleWithElement.
110 return attribute == typeAttr || attribute == readonlyAttr;
111}

113static bool computeContainsUncommonAttributeSelector(const CSSSelector& rootSelector, bool matchesRightmostElement = true)
114{
115 const CSSSelector* selector = &rootSelector;
116 do {
117 if (selector->isAttributeSelector()) {
118 // FIXME: considering non-rightmost simple selectors is necessary because of the style sharing of cousins.
119 // It is a primitive solution which disable a lot of style sharing on pages that rely on attributes for styling.
120 // We should investigate better ways of doing this.
121 if (!isCommonAttributeSelectorAttribute(selector->attribute()) || !matchesRightmostElement)
122 return true;
123 }
124
125 if (const CSSSelectorList* selectorList = selector->selectorList()) {
126 for (const CSSSelector* subSelector = selectorList->first(); subSelector; subSelector = CSSSelectorList::next(subSelector)) {
127 if (computeContainsUncommonAttributeSelector(*subSelector, matchesRightmostElement))
128 return true;
129 }
130 }
131
132 if (selector->relation() != CSSSelector::Subselector)
133 matchesRightmostElement = false;
134
135 selector = selector->tagHistory();
136 } while (selector);
137 return false;
138}

140static inline PropertyAllowlistType determinePropertyAllowlistType(const CSSSelector* selector)
141{
142 for (const CSSSelector* component = selector; component; component = component->tagHistory()) {
143#if ENABLE(VIDEO)
144 if (component->match() == CSSSelector::PseudoElement && (component->pseudoElementType() == CSSSelector::PseudoElementCue || component->value() == TextTrackCue::cueShadowPseudoId()))
145 return PropertyAllowlistCue;
146#endif
147 if (component->match() == CSSSelector::PseudoElement && component->pseudoElementType() == CSSSelector::PseudoElementMarker)
148 return PropertyAllowlistMarker;
149
150 if (const auto* selectorList = selector->selectorList()) {
151 for (const auto* subSelector = selectorList->first(); subSelector; subSelector = CSSSelectorList::next(subSelector)) {
152 auto allowlistType = determinePropertyAllowlistType(subSelector);
153 if (allowlistType != PropertyAllowlistNone)
154 return allowlistType;
155 }
156 }
157 }
158 return PropertyAllowlistNone;
159}

161RuleData::RuleData(const StyleRule& styleRule, unsigned selectorIndex, unsigned selectorListIndex, unsigned position)
162 : m_styleRule(&styleRule)
163 , m_selectorIndex(selectorIndex)
164 , m_selectorListIndex(selectorListIndex)
165 , m_position(position)
166 , m_matchBasedOnRuleHash(static_cast<unsigned>(computeMatchBasedOnRuleHash(*selector())))
167 , m_canMatchPseudoElement(selectorCanMatchPseudoElement(*selector()))
168 , m_containsUncommonAttributeSelector(computeContainsUncommonAttributeSelector(*selector()))
169 , m_linkMatchType(SelectorChecker::determineLinkMatchType(selector()))
170 , m_propertyAllowlistType(determinePropertyAllowlistType(selector()))
171 , m_isEnabled(true)
172 , m_descendantSelectorIdentifierHashes(SelectorFilter::collectHashes(*selector()))
173{
174 ASSERT(m_position == position);
175 ASSERT(m_selectorIndex == selectorIndex);
176}
