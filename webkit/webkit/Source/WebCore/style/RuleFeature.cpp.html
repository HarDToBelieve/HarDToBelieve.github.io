39static bool isSiblingOrSubject(MatchElement matchElement)
40{
41 switch (matchElement) {
42 case MatchElement::Subject:
43 case MatchElement::IndirectSibling:
44 case MatchElement::DirectSibling:
45 case MatchElement::AnySibling:
46 case MatchElement::Host:
47 return true;
48 case MatchElement::Parent:
49 case MatchElement::Ancestor:
50 case MatchElement::ParentSibling:
51 case MatchElement::AncestorSibling:
52 return false;
53 }
54 ASSERT_NOT_REACHED();
55 return false;
56}

68MatchElement RuleFeatureSet::computeNextMatchElement(MatchElement matchElement, CSSSelector::RelationType relation)
69{
70 if (isSiblingOrSubject(matchElement)) {
71 switch (relation) {
72 case CSSSelector::Subselector:
73 return matchElement;
74 case CSSSelector::DescendantSpace:
75 return MatchElement::Ancestor;
76 case CSSSelector::Child:
77 return MatchElement::Parent;
78 case CSSSelector::IndirectAdjacent:
79 if (matchElement == MatchElement::AnySibling)
80 return MatchElement::AnySibling;
81 return MatchElement::IndirectSibling;
82 case CSSSelector::DirectAdjacent:
83 if (matchElement == MatchElement::AnySibling)
84 return MatchElement::AnySibling;
85 return matchElement == MatchElement::Subject ? MatchElement::DirectSibling : MatchElement::IndirectSibling;
86 case CSSSelector::ShadowDescendant:
87 return MatchElement::Host;
88 };
89 }
90 switch (relation) {
91 case CSSSelector::Subselector:
92 return matchElement;
93 case CSSSelector::DescendantSpace:
94 case CSSSelector::Child:
95 return MatchElement::Ancestor;
96 case CSSSelector::IndirectAdjacent:
97 case CSSSelector::DirectAdjacent:
98 return matchElement == MatchElement::Parent ? MatchElement::ParentSibling : MatchElement::AncestorSibling;
99 case CSSSelector::ShadowDescendant:
100 return MatchElement::Host;
101 };
102 ASSERT_NOT_REACHED();
103 return matchElement;
104};

106MatchElement RuleFeatureSet::computeSubSelectorMatchElement(MatchElement matchElement, const CSSSelector& selector)
107{
108 ASSERT(selector.selectorList());
109
110 if (selector.match() == CSSSelector::PseudoClass) {
111 auto type = selector.pseudoClassType();
112 // For :nth-child(n of .some-subselector) where an element change may affect other elements similar to sibling combinators.
113 if (type == CSSSelector::PseudoClassNthChild || type == CSSSelector::PseudoClassNthLastChild)
114 return MatchElement::AnySibling;
115
116 // Similarly for :host().
117 if (type == CSSSelector::PseudoClassHost)
118 return MatchElement::Host;
119 }
120 if (selector.match() == CSSSelector::PseudoElement) {
121 // Similarly for ::slotted().
122 if (selector.pseudoElementType() == CSSSelector::PseudoElementSlotted)
123 return MatchElement::Host;
124 }
125
126 return matchElement;
127};

129void RuleFeatureSet::recursivelyCollectFeaturesFromSelector(SelectorFeatures& selectorFeatures, const CSSSelector& firstSelector, MatchElement matchElement)
130{
131 const CSSSelector* selector = &firstSelector;
132 do {
133 if (selector->match() == CSSSelector::Id) {
134 idsInRules.add(selector->value());
135 if (matchElement == MatchElement::Parent || matchElement == MatchElement::Ancestor)
136 idsMatchingAncestorsInRules.add(selector->value());
137 } else if (selector->match() == CSSSelector::Class)
138 selectorFeatures.classes.append(std::make_pair(selector->value(), matchElement));
139 else if (selector->isAttributeSelector()) {
140 auto& canonicalLocalName = selector->attributeCanonicalLocalName();
141 auto& localName = selector->attribute().localName();
142 attributeCanonicalLocalNamesInRules.add(canonicalLocalName);
143 attributeLocalNamesInRules.add(localName);
144 selectorFeatures.attributes.append(std::make_pair(selector, matchElement));
145 } else if (selector->match() == CSSSelector::PseudoElement) {
146 switch (selector->pseudoElementType()) {
147 case CSSSelector::PseudoElementFirstLine:
148 usesFirstLineRules = true;
149 break;
150 case CSSSelector::PseudoElementFirstLetter:
151 usesFirstLetterRules = true;
152 break;
153 default:
154 break;
155 }
156 } else if (selector->match() == CSSSelector::PseudoClass)
157 selectorFeatures.pseudoClasses.append(std::make_pair(selector->pseudoClassType(), matchElement));
158
159 if (!selectorFeatures.hasSiblingSelector && selector->isSiblingSelector())
160 selectorFeatures.hasSiblingSelector = true;
161
162 if (const CSSSelectorList* selectorList = selector->selectorList()) {
163 auto subSelectorMatchElement = computeSubSelectorMatchElement(matchElement, *selector);
164
165 for (const CSSSelector* subSelector = selectorList->first(); subSelector; subSelector = CSSSelectorList::next(subSelector)) {
166 if (!selectorFeatures.hasSiblingSelector && selector->isSiblingSelector())
167 selectorFeatures.hasSiblingSelector = true;
168 recursivelyCollectFeaturesFromSelector(selectorFeatures, *subSelector, subSelectorMatchElement);
169 }
170 }
171
172 matchElement = computeNextMatchElement(matchElement, selector->relation());
173
174 selector = selector->tagHistory();
175 } while (selector);
176}

178void RuleFeatureSet::collectFeatures(const RuleData& ruleData)
179{
180 SelectorFeatures selectorFeatures;
181 recursivelyCollectFeaturesFromSelector(selectorFeatures, *ruleData.selector());
182 if (selectorFeatures.hasSiblingSelector)
183 siblingRules.append({ ruleData });
184 if (ruleData.containsUncommonAttributeSelector())
185 uncommonAttributeRules.append({ ruleData });
186
187 for (auto& nameAndMatch : selectorFeatures.classes) {
188 classRules.ensure(nameAndMatch.first, [] {
189 return makeUnique<RuleFeatureVector>();
190 }).iterator->value->append({ ruleData, nameAndMatch.second });
191 if (nameAndMatch.second == MatchElement::Host)
192 classesAffectingHost.add(nameAndMatch.first);
193 }
194 for (auto& selectorAndMatch : selectorFeatures.attributes) {
195 auto* selector = selectorAndMatch.first;
196 auto matchElement = selectorAndMatch.second;
197 attributeRules.ensure(selector->attribute().localName().convertToASCIILowercase(), [] {
198 return makeUnique<Vector<RuleFeatureWithInvalidationSelector>>();
199 }).iterator->value->append({ ruleData, matchElement, selector });
200 if (matchElement == MatchElement::Host)
201 attributesAffectingHost.add(selector->attribute().localName().convertToASCIILowercase());
202 }
203 for (auto& keyAndMatch : selectorFeatures.pseudoClasses) {
204 pseudoClassRules.ensure(keyAndMatch.first, [] {
205 return makeUnique<RuleFeatureVector>();
206 }).iterator->value->append({ ruleData, keyAndMatch.second });
207 if (keyAndMatch.second == MatchElement::Host)
208 pseudoClassesAffectingHost.add(keyAndMatch.first);
209 }
210}

212void RuleFeatureSet::add(const RuleFeatureSet& other)
213{
214 idsInRules.add(other.idsInRules.begin(), other.idsInRules.end());
215 idsMatchingAncestorsInRules.add(other.idsMatchingAncestorsInRules.begin(), other.idsMatchingAncestorsInRules.end());
216 attributeCanonicalLocalNamesInRules.add(other.attributeCanonicalLocalNamesInRules.begin(), other.attributeCanonicalLocalNamesInRules.end());
217 attributeLocalNamesInRules.add(other.attributeLocalNamesInRules.begin(), other.attributeLocalNamesInRules.end());
218 contentAttributeNamesInRules.add(other.contentAttributeNamesInRules.begin(), other.contentAttributeNamesInRules.end());
219 siblingRules.appendVector(other.siblingRules);
220 uncommonAttributeRules.appendVector(other.uncommonAttributeRules);
221 for (auto& keyValuePair : other.classRules) {
222 classRules.ensure(keyValuePair.key, [] {
223 return makeUnique<RuleFeatureVector>();
224 }).iterator->value->appendVector(*keyValuePair.value);
225 }
226 classesAffectingHost.add(other.classesAffectingHost.begin(), other.classesAffectingHost.end());
227
228 for (auto& keyValuePair : other.attributeRules) {
229 attributeRules.ensure(keyValuePair.key, [] {
230 return makeUnique<Vector<RuleFeatureWithInvalidationSelector>>();
231 }).iterator->value->appendVector(*keyValuePair.value);
232 }
233 attributesAffectingHost.add(other.attributesAffectingHost.begin(), other.attributesAffectingHost.end());
234
235 for (auto& keyValuePair : other.pseudoClassRules) {
236 pseudoClassRules.ensure(keyValuePair.key, [] {
237 return makeUnique<RuleFeatureVector>();
238 }).iterator->value->appendVector(*keyValuePair.value);
239 }
240 pseudoClassesAffectingHost.add(other.pseudoClassesAffectingHost.begin(), other.pseudoClassesAffectingHost.end());
241
242 usesFirstLineRules = usesFirstLineRules || other.usesFirstLineRules;
243 usesFirstLetterRules = usesFirstLetterRules || other.usesFirstLetterRules;
244}

246void RuleFeatureSet::registerContentAttribute(const AtomString& attributeName)
247{
248 contentAttributeNamesInRules.add(attributeName.convertToASCIILowercase());
249 attributeCanonicalLocalNamesInRules.add(attributeName);
250 attributeLocalNamesInRules.add(attributeName);
251}

253void RuleFeatureSet::clear()
254{
255 idsInRules.clear();
256 idsMatchingAncestorsInRules.clear();
257 attributeCanonicalLocalNamesInRules.clear();
258 attributeLocalNamesInRules.clear();
259 contentAttributeNamesInRules.clear();
260 siblingRules.clear();
261 uncommonAttributeRules.clear();
262 classRules.clear();
263 classesAffectingHost.clear();
264 attributeRules.clear();
265 attributesAffectingHost.clear();
266 pseudoClassRules.clear();
267 pseudoClassesAffectingHost.clear();
268 usesFirstLineRules = false;
269 usesFirstLetterRules = false;
270}

272void RuleFeatureSet::shrinkToFit()
273{
274 siblingRules.shrinkToFit();
275 uncommonAttributeRules.shrinkToFit();
276 for (auto& rules : classRules.values())
277 rules->shrinkToFit();
278 for (auto& rules : attributeRules.values())
279 rules->shrinkToFit();
280 for (auto& rules : pseudoClassRules.values())
281 rules->shrinkToFit();
282}
