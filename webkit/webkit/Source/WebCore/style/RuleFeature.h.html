34class StyleRule;

36namespace Style {
37
38class RuleData;
39
40enum class MatchElement : uint8_t { Subject, Parent, Ancestor, DirectSibling, IndirectSibling, AnySibling, ParentSibling, AncestorSibling, Host };
41constexpr unsigned matchElementCount = static_cast<unsigned>(MatchElement::Host) + 1;
42
43#if HAVE(PROBABILISTIC_GUARD_MALLOC)
44// Help finding rdar://77450900
45using RuleMalloc = ProbabilisticGuardMalloc;
46#else
47using RuleMalloc = FastMalloc;
48#endif
49
50struct RuleFeature {
51 RuleFeature(const RuleData&, std::optional<MatchElement> = std::nullopt);
52
53 RefPtr<const StyleRule> styleRule;
54 uint16_t selectorIndex; // Keep in sync with RuleData's selectorIndex size.
55 uint16_t selectorListIndex; // Keep in sync with RuleData's selectorListIndex size.
56 std::optional<MatchElement> matchElement { };
57};
58static_assert(sizeof(RuleFeature) <= 16, "RuleFeature is a frquently alocated object. Keep it small.");
59
60struct RuleFeatureWithInvalidationSelector : public RuleFeature {
61 RuleFeatureWithInvalidationSelector(const RuleData& data, std::optional<MatchElement> matchElement = std::nullopt, const CSSSelector* invalidationSelector = nullptr)
62 : RuleFeature(data, WTFMove(matchElement))
63 , invalidationSelector(invalidationSelector)
64 { }
65
66 const CSSSelector* invalidationSelector { nullptr };
67};
68
69using RuleFeatureVector = Vector<RuleFeature, 0, CrashOnOverflow, 16, RuleMalloc>;
70
71struct RuleFeatureSet {
72 void add(const RuleFeatureSet&);
73 void clear();
74 void shrinkToFit();
75 void collectFeatures(const RuleData&);
76 void registerContentAttribute(const AtomString&);
77
78 HashSet<AtomString> idsInRules;
79 HashSet<AtomString> idsMatchingAncestorsInRules;
80 HashSet<AtomString> attributeCanonicalLocalNamesInRules;
81 HashSet<AtomString> attributeLocalNamesInRules;
82 HashSet<AtomString> contentAttributeNamesInRules;
83 RuleFeatureVector siblingRules;
84 RuleFeatureVector uncommonAttributeRules;
85
86 HashMap<AtomString, std::unique_ptr<RuleFeatureVector>> classRules;
87 HashMap<AtomString, std::unique_ptr<Vector<RuleFeatureWithInvalidationSelector>>> attributeRules;
88 HashMap<CSSSelector::PseudoClassType, std::unique_ptr<RuleFeatureVector>, WTF::IntHash<CSSSelector::PseudoClassType>, WTF::StrongEnumHashTraits<CSSSelector::PseudoClassType>> pseudoClassRules;
89 HashSet<AtomString> classesAffectingHost;
90 HashSet<AtomString> attributesAffectingHost;
91 HashSet<CSSSelector::PseudoClassType, WTF::IntHash<CSSSelector::PseudoClassType>, WTF::StrongEnumHashTraits<CSSSelector::PseudoClassType>> pseudoClassesAffectingHost;
92
93 bool usesFirstLineRules { false };
94 bool usesFirstLetterRules { false };
95
96private:
97 static MatchElement computeNextMatchElement(MatchElement, CSSSelector::RelationType);
98 static MatchElement computeSubSelectorMatchElement(MatchElement, const CSSSelector&);
99
100 struct SelectorFeatures {
101 bool hasSiblingSelector { false };
102
103 Vector<std::pair<AtomString, MatchElement>, 32> classes;
104 Vector<std::pair<const CSSSelector*, MatchElement>, 32> attributes;
105 Vector<std::pair<CSSSelector::PseudoClassType, MatchElement>, 32> pseudoClasses;
106 };
107 void recursivelyCollectFeaturesFromSelector(SelectorFeatures&, const CSSSelector&, MatchElement = MatchElement::Subject);
108};
