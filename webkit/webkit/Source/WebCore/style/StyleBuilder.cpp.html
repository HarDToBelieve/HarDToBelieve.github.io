56inline bool isValidVisitedLinkProperty(CSSPropertyID id)
57{
58 switch (id) {
59 case CSSPropertyBackgroundColor:
60 case CSSPropertyBorderLeftColor:
61 case CSSPropertyBorderRightColor:
62 case CSSPropertyBorderTopColor:
63 case CSSPropertyBorderBottomColor:
64 case CSSPropertyCaretColor:
65 case CSSPropertyColor:
66 case CSSPropertyOutlineColor:
67 case CSSPropertyColumnRuleColor:
68 case CSSPropertyTextDecorationColor:
69 case CSSPropertyWebkitTextEmphasisColor:
70 case CSSPropertyWebkitTextFillColor:
71 case CSSPropertyWebkitTextStrokeColor:
72 case CSSPropertyFill:
73 case CSSPropertyStroke:
74 case CSSPropertyStrokeColor:
75 return true;
76 default:
77 break;
78 }
79
80 return false;
81}

97// High priority properties may affect resolution of other properties (they are mostly font related).
98void Builder::applyHighPriorityProperties()
99{
100 applyProperties(CSSPropertyWebkitRubyPosition, CSSPropertyWebkitRubyPosition);
101 m_state.adjustStyleForInterCharacterRuby();
102
103#if ENABLE(DARK_MODE_CSS)
104 // Supported color schemes can affect resolved colors, so we need to apply that property before any color properties.
105 applyProperties(CSSPropertyColorScheme, CSSPropertyColorScheme);
106#endif
107
108 applyProperties(firstCSSProperty, lastHighPriorityProperty);
109
110 m_state.updateFont();
111}
112
113void Builder::applyLowPriorityProperties()
114{
115 ASSERT(!m_state.fontDirty());
116
117 applyCustomProperties();
118 applyProperties(firstLowPriorityProperty, lastCSSProperty);
119 applyDeferredProperties();
120
121 ASSERT(!m_state.fontDirty());
122}

140void Builder::applyProperties(int firstProperty, int lastProperty)
141{
142 if (LIKELY(m_cascade.customProperties().isEmpty()))
143 return applyPropertiesImpl<CustomPropertyCycleTracking::Disabled>(firstProperty, lastProperty);
144
145 return applyPropertiesImpl<CustomPropertyCycleTracking::Enabled>(firstProperty, lastProperty);
146}
147
148template<Builder::CustomPropertyCycleTracking trackCycles>
149inline void Builder::applyPropertiesImpl(int firstProperty, int lastProperty)
150{
151 for (int id = firstProperty; id <= lastProperty; ++id) {
152 CSSPropertyID propertyID = static_cast<CSSPropertyID>(id);
153 if (!m_cascade.hasProperty(propertyID))
154 continue;
155 ASSERT(propertyID != CSSPropertyCustom);
156 auto& property = m_cascade.property(propertyID);
157
158 if (trackCycles == CustomPropertyCycleTracking::Enabled) {
159 if (UNLIKELY(m_state.m_inProgressProperties.get(propertyID))) {
160 // We are in a cycle (eg. setting font size using registered custom property value containing em).
161 // So this value should be unset.
162 m_state.m_appliedProperties.set(propertyID);
163 // This property is in a cycle, and only the root of the call stack will have firstProperty != lastProperty.
164 ASSERT(firstProperty == lastProperty);
165 continue;
166 }
167 m_state.m_inProgressProperties.set(propertyID);
168 applyCascadeProperty(property);
169 m_state.m_appliedProperties.set(propertyID);
170 m_state.m_inProgressProperties.set(propertyID, false);
171 continue;
172 }
173
174 // If we don't have any custom properties, then there can't be any cycles.
175 applyCascadeProperty(property);
176 }
177}

184
185void Builder::applyCustomProperty(const String& name)
186{
187 if (m_state.m_appliedCustomProperties.contains(name) || !m_cascade.customProperties().contains(name))
188 return;
189
190 auto property = m_cascade.customProperty(name);
191 bool inCycle = m_state.m_inProgressPropertiesCustom.contains(name);
192
193 for (auto index : { SelectorChecker::MatchDefault, SelectorChecker::MatchLink, SelectorChecker::MatchVisited }) {
194 if (!property.cssValue[index])
195 continue;
196 if (index != SelectorChecker::MatchDefault && m_state.style().insideLink() == InsideLink::NotInside)
197 continue;
198
199 auto valueToApply = makeRef(downcast<CSSCustomPropertyValue>(*property.cssValue[index]));
200
201 if (inCycle) {
202 m_state.m_appliedCustomProperties.add(name); // Make sure we do not try to apply this property again while resolving it.
203 valueToApply = CSSCustomPropertyValue::createWithID(name, CSSValueInvalid);
204 }
205
206 m_state.m_inProgressPropertiesCustom.add(name);
207
208 if (WTF::holds_alternative<Ref<CSSVariableReferenceValue>>(valueToApply->value())) {
209 RefPtr<CSSValue> parsedValue = resolvedVariableValue(CSSPropertyCustom, valueToApply.get());
210
211 if (m_state.m_appliedCustomProperties.contains(name))
212 return; // There was a cycle and the value was reset, so bail.
213
214 if (!parsedValue)
215 parsedValue = CSSCustomPropertyValue::createWithID(name, CSSValueUnset);
216
217 valueToApply = downcast<CSSCustomPropertyValue>(*parsedValue);
218 }
219
220 if (m_state.m_inProgressPropertiesCustom.contains(name)) {
221 SetForScope<SelectorChecker::LinkMatchMask> scopedLinkMatchMutation(m_state.m_linkMatch, index);
222 applyProperty(CSSPropertyCustom, valueToApply.get(), index);
223 }
224 }
225
226 m_state.m_inProgressPropertiesCustom.remove(name);
227 m_state.m_appliedCustomProperties.add(name);
228
229 for (auto index : { SelectorChecker::MatchDefault, SelectorChecker::MatchLink, SelectorChecker::MatchVisited }) {
230 if (!property.cssValue[index])
231 continue;
232 if (index != SelectorChecker::MatchDefault && m_state.style().insideLink() == InsideLink::NotInside)
233 continue;
234
235 auto valueToApply = makeRef(downcast<CSSCustomPropertyValue>(*property.cssValue[index]));
236
237 if (inCycle && WTF::holds_alternative<Ref<CSSVariableReferenceValue>>(valueToApply->value())) {
238 // Resolve this value so that we reset its dependencies.
239 resolvedVariableValue(CSSPropertyCustom, valueToApply.get());
240 }
241 }
242}

244inline void Builder::applyCascadeProperty(const PropertyCascade::Property& property)
245{
246 m_state.m_cascadeLevel = property.level;
247 m_state.m_styleScopeOrdinal = property.styleScopeOrdinal;
248
249 auto applyWithLinkMatch = [&](SelectorChecker::LinkMatchMask linkMatch) {
250 if (property.cssValue[linkMatch]) {
251 SetForScope<SelectorChecker::LinkMatchMask> scopedLinkMatchMutation(m_state.m_linkMatch, linkMatch);
252 applyProperty(property.id, *property.cssValue[linkMatch], linkMatch);
253 }
254 };
255
256 applyWithLinkMatch(SelectorChecker::MatchDefault);
257
258 if (m_state.style().insideLink() == InsideLink::NotInside)
259 return;
260
261 applyWithLinkMatch(SelectorChecker::MatchLink);
262 applyWithLinkMatch(SelectorChecker::MatchVisited);
263
264 m_state.m_linkMatch = SelectorChecker::MatchDefault;
265}

267void Builder::applyProperty(CSSPropertyID id, CSSValue& value, SelectorChecker::LinkMatchMask linkMatchMask)
268{
269 ASSERT_WITH_MESSAGE(!isShorthandCSSProperty(id), "Shorthand property id = %d wasn't expanded at parsing time", id);
270
271 auto valueToApply = resolveValue(id, value);
272
273 if (CSSProperty::isDirectionAwareProperty(id)) {
274 auto direction = m_cascade.direction();
275 CSSPropertyID newId = CSSProperty::resolveDirectionAwareProperty(id, direction.textDirection, direction.writingMode);
276 ASSERT(newId != id);
277 return applyProperty(newId, valueToApply.get(), linkMatchMask);
278 }
279
280 CSSCustomPropertyValue* customPropertyValue = nullptr;
281 CSSValueID customPropertyValueID = CSSValueInvalid;
282 CSSRegisteredCustomProperty* customPropertyRegistered = nullptr;
283
284 if (id == CSSPropertyCustom) {
285 customPropertyValue = downcast<CSSCustomPropertyValue>(valueToApply.ptr());
286 ASSERT(customPropertyValue->isResolved());
287 if (WTF::holds_alternative<CSSValueID>(customPropertyValue->value()))
288 customPropertyValueID = WTF::get<CSSValueID>(customPropertyValue->value());
289 auto& name = customPropertyValue->name();
290 customPropertyRegistered = m_state.document().getCSSRegisteredCustomPropertySet().get(name);
291 }
292
293 bool isInherit = valueToApply->isInheritedValue() || customPropertyValueID == CSSValueInherit;
294 bool isInitial = valueToApply->isInitialValue() || customPropertyValueID == CSSValueInitial;
295
296 bool isUnset = valueToApply->isUnsetValue() || customPropertyValueID == CSSValueUnset;
297 bool isRevert = valueToApply->isRevertValue() || customPropertyValueID == CSSValueRevert;
298
299 if (isRevert) {
300 if (auto* rollback = m_cascade.propertyCascadeForRollback(m_state.m_cascadeLevel)) {
301 // With the rollback cascade built, we need to obtain the property and apply it. If the property is
302 // not present, then we behave like "unset." Otherwise we apply the property instead of
303 // our own.
304 if (customPropertyValue) {
305 if (customPropertyRegistered && customPropertyRegistered->inherits && rollback->hasCustomProperty(customPropertyValue->name())) {
306 auto property = rollback->customProperty(customPropertyValue->name());
307 if (property.cssValue[linkMatchMask])
308 applyProperty(property.id, *property.cssValue[linkMatchMask], linkMatchMask);
309 return;
310 }
311 } else if (rollback->hasProperty(id)) {
312 auto& property = rollback->property(id);
313 if (property.cssValue[linkMatchMask])
314 applyProperty(property.id, *property.cssValue[linkMatchMask], linkMatchMask);
315 return;
316 }
317 }
318
319 isUnset = true;
320 }
321
322 if (isUnset) {
323 if (CSSProperty::isInheritedProperty(id))
324 isInherit = true;
325 else
326 isInitial = true;
327 }
328
329 ASSERT(!isInherit || !isInitial); // isInherit -> !isInitial && isInitial -> !isInherit
330
331 if (m_state.applyPropertyToVisitedLinkStyle() && !isValidVisitedLinkProperty(id)) {
332 // Limit the properties that can be applied to only the ones honored by :visited.
333 return;
334 }
335
336 if (isInherit && !CSSProperty::isInheritedProperty(id))
337 m_state.style().setHasExplicitlyInheritedProperties();
338
339#if ENABLE(CSS_PAINTING_API)
340 if (is<CSSPaintImageValue>(valueToApply)) {
341 auto& name = downcast<CSSPaintImageValue>(valueToApply.get()).name();
342 if (auto* paintWorklet = const_cast<Document&>(m_state.document()).paintWorkletGlobalScopeForName(name)) {
343 Locker locker { paintWorklet->paintDefinitionLock() };
344 if (auto* registration = paintWorklet->paintDefinitionMap().get(name)) {
345 for (auto& property : registration->inputProperties)
346 m_state.style().addCustomPaintWatchProperty(property);
347 }
348 }
349 }
350#endif
351
352 BuilderGenerated::applyProperty(id, m_state, valueToApply.get(), isInitial, isInherit, customPropertyRegistered);
353}

355Ref<CSSValue> Builder::resolveValue(CSSPropertyID propertyID, CSSValue& value)
356{
357 if (!value.hasVariableReferences())
358 return value;
359
360 auto variableValue = resolvedVariableValue(propertyID, value);
361 // If the cascade has already applied this id, then we detected a cycle, and this value should be unset.
362 if (!variableValue || m_state.m_appliedProperties.get(propertyID)) {
363 if (CSSProperty::isInheritedProperty(propertyID))
364 return CSSValuePool::singleton().createInheritedValue();
365 return CSSValuePool::singleton().createExplicitInitialValue();
366 }
367
368 return *variableValue;
369}
370
371RefPtr<CSSValue> Builder::resolvedVariableValue(CSSPropertyID propID, const CSSValue& value)
372{
373 if (value.isPendingSubstitutionValue()) {
374 auto& substitutionValue = downcast<CSSPendingSubstitutionValue>(value);
375 CSSParser parser(CSSParserContext(m_state.document(), substitutionValue.baseURL()));
376 return parser.parseValueWithVariableReferences(propID, value, m_state);
377 }
378
379 CSSParser parser(m_state.document());
380 return parser.parseValueWithVariableReferences(propID, value, m_state);
381}
