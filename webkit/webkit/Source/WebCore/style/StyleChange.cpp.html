34Change determineChange(const RenderStyle& s1, const RenderStyle& s2)
35{
36 if (s1.display() != s2.display())
37 return Change::Renderer;
38 if (s1.hasPseudoStyle(PseudoId::FirstLetter) != s2.hasPseudoStyle(PseudoId::FirstLetter))
39 return Change::Renderer;
40 // We just detach if a renderer acquires or loses a column-span, since spanning elements
41 // typically won't contain much content.
42 auto columnSpanNeedsNewRenderer = [&] {
43 if (s1.columnSpan() != s2.columnSpan())
44 return true;
45 if (s1.columnSpan() != ColumnSpan::All)
46 return false;
47 // Spanning in ignored for floating and out-of-flow boxes.
48 return s1.isFloating() != s2.isFloating() || s1.hasOutOfFlowPosition() != s2.hasOutOfFlowPosition();
49 }();
50 if (columnSpanNeedsNewRenderer)
51 return Change::Renderer;
52 if (!s1.contentDataEquivalent(&s2))
53 return Change::Renderer;
54 // When text-combine property has been changed, we need to prepare a separate renderer object.
55 // When text-combine is on, we use RenderCombineText, otherwise RenderText.
56 // https://bugs.webkit.org/show_bug.cgi?id=55069
57 if (s1.hasTextCombine() != s2.hasTextCombine())
58 return Change::Renderer;
59
60 if (!s1.inheritedEqual(s2))
61 return Change::Inherited;
62
63 if (!s1.descendantAffectingNonInheritedPropertiesEqual(s2))
64 return Change::Inherited;
65
66 if (s1 != s2)
67 return Change::NonInherited;
68
69 // If the pseudoStyles have changed, we want any StyleChange that is not NoChange
70 // because setStyle will do the right thing with anything else.
71 if (s1.hasAnyPublicPseudoStyles()) {
72 for (PseudoId pseudoId = PseudoId::FirstPublicPseudoId; pseudoId < PseudoId::FirstInternalPseudoId; pseudoId = static_cast<PseudoId>(static_cast<unsigned>(pseudoId) + 1)) {
73 if (s1.hasPseudoStyle(pseudoId)) {
74 RenderStyle* ps2 = s2.getCachedPseudoStyle(pseudoId);
75 if (!ps2)
76 return Change::NonInherited;
77 RenderStyle* ps1 = s1.getCachedPseudoStyle(pseudoId);
78 if (!ps1 || *ps1 != *ps2)
79 return Change::NonInherited;
80 }
81 }
82 }
83
84 return Change::None;
85}
