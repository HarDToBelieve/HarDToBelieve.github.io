42float computedFontSizeFromSpecifiedSize(float specifiedSize, bool isAbsoluteSize, float zoomFactor, MinimumFontSizeRule minimumSizeRule, const Settings::Values& settings)
43{
44 // Text with a 0px font size should not be visible and therefore needs to be
45 // exempt from minimum font size rules. Acid3 relies on this for pixel-perfect
46 // rendering. This is also compatible with other browsers that have minimum
47 // font size settings (e.g. Firefox).
48 if (fabsf(specifiedSize) < std::numeric_limits<float>::epsilon())
49 return 0.0f;
50
51 // We support two types of minimum font size. The first is a hard override that applies to
52 // all fonts. This is "minSize." The second type of minimum font size is a "smart minimum"
53 // that is applied only when the Web page can't know what size it really asked for, e.g.,
54 // when it uses logical sizes like "small" or expresses the font-size as a percentage of
55 // the user's default font setting.
56
57 // With the smart minimum, we never want to get smaller than the minimum font size to keep fonts readable.
58 // However we always allow the page to set an explicit pixel size that is smaller,
59 // since sites will mis-render otherwise (e.g., http://www.gamespot.com with a 9px minimum).
60
61 if (minimumSizeRule == MinimumFontSizeRule::None)
62 return specifiedSize;
63
64 int minSize = settings.minimumFontSize;
65 int minLogicalSize = settings.minimumLogicalFontSize;
66 float zoomedSize = specifiedSize * zoomFactor;
67
68 // Apply the hard minimum first. We only apply the hard minimum if after zooming we're still too small.
69 zoomedSize = std::max(zoomedSize, static_cast<float>(minSize));
70
71 // Now apply the smart minimum. This minimum is also only applied if we're still too small
72 // after zooming. The font size must either be relative to the user default or the original size
73 // must have been acceptable. In other words, we only apply the smart minimum whenever we're positive
74 // doing so won't disrupt the layout.
75 if (minimumSizeRule == MinimumFontSizeRule::AbsoluteAndRelative && (specifiedSize >= minLogicalSize || !isAbsoluteSize))
76 zoomedSize = std::max(zoomedSize, static_cast<float>(minLogicalSize));
77
78 // Also clamp to a reasonable maximum to prevent insane font sizes from causing crashes on various
79 // platforms. (I'm looking at you, Windows.)
80 return std::min(maximumAllowedFontSize, zoomedSize);
81}

83float computedFontSizeFromSpecifiedSize(float specifiedSize, bool isAbsoluteSize, bool useSVGZoomRules, const RenderStyle* style, const Document& document)
84{
85 float zoomFactor = 1.0f;
86 if (!useSVGZoomRules) {
87 zoomFactor = style->effectiveZoom();
88 Frame* frame = document.frame();
89 if (frame && style->textZoom() != TextZoom::Reset)
90 zoomFactor *= frame->textZoomFactor();
91 }
92 return computedFontSizeFromSpecifiedSize(specifiedSize, isAbsoluteSize, zoomFactor, useSVGZoomRules ? MinimumFontSizeRule::None : MinimumFontSizeRule::AbsoluteAndRelative, document.settingsValues());
93}

95float computedFontSizeFromSpecifiedSizeForSVGInlineText(float specifiedSize, bool isAbsoluteSize, float zoomFactor, const Document& document)
96{
97 return computedFontSizeFromSpecifiedSize(specifiedSize, isAbsoluteSize, zoomFactor, MinimumFontSizeRule::Absolute, document.settingsValues());
98}

100const int fontSizeTableMax = 16;
101const int fontSizeTableMin = 9;
102const int totalKeywords = 8;
103
104// WinIE/Nav4 table for font sizes. Designed to match the legacy font mapping system of HTML.
105static const int quirksFontSizeTable[fontSizeTableMax - fontSizeTableMin + 1][totalKeywords] =
106{
107 { 9, 9, 9, 9, 11, 14, 18, 28 },
108 { 9, 9, 9, 10, 12, 15, 20, 31 },
109 { 9, 9, 9, 11, 13, 17, 22, 34 },
110 { 9, 9, 10, 12, 14, 18, 24, 37 },
111 { 9, 9, 10, 13, 16, 20, 26, 40 }, // fixed font default (13)
112 { 9, 9, 11, 14, 17, 21, 28, 42 },
113 { 9, 10, 12, 15, 17, 23, 30, 45 },
114 { 9, 10, 13, 16, 18, 24, 32, 48 } // proportional font default (16)
115};
116// HTML 1 2 3 4 5 6 7
117// CSS xxs xs s m l xl xxl
118// |
119// user pref
120
121// Strict mode table matches MacIE and Mozilla's settings exactly.
122static const int strictFontSizeTable[fontSizeTableMax - fontSizeTableMin + 1][totalKeywords] =
123{
124 { 9, 9, 9, 9, 11, 14, 18, 27 },
125 { 9, 9, 9, 10, 12, 15, 20, 30 },
126 { 9, 9, 10, 11, 13, 17, 22, 33 },
127 { 9, 9, 10, 12, 14, 18, 24, 36 },
128 { 9, 10, 12, 13, 16, 20, 26, 39 }, // fixed font default (13)
129 { 9, 10, 12, 14, 17, 21, 28, 42 },
130 { 9, 10, 13, 15, 18, 23, 30, 45 },
131 { 9, 10, 13, 16, 18, 24, 32, 48 } // proportional font default (16)
132};
133// HTML 1 2 3 4 5 6 7
134// CSS xxs xs s m l xl xxl
135// |
136// user pref
137
138// For values outside the range of the table, we use Todd Fahrner's suggested scale
139// factors for each keyword value.
140static const float fontSizeFactors[totalKeywords] = { 0.60f, 0.75f, 0.89f, 1.0f, 1.2f, 1.5f, 2.0f, 3.0f };
141
142float fontSizeForKeyword(unsigned keywordID, bool shouldUseFixedDefaultSize, const Settings::Values& settings, bool inQuirksMode)
143{
144 int mediumSize = shouldUseFixedDefaultSize ? settings.defaultFixedFontSize : settings.defaultFontSize;
145 if (mediumSize >= fontSizeTableMin && mediumSize <= fontSizeTableMax) {
146 // Look up the entry in the table.
147 int row = mediumSize - fontSizeTableMin;
148 int col = (keywordID - CSSValueXxSmall);
149 return inQuirksMode ? quirksFontSizeTable[row][col] : strictFontSizeTable[row][col];
150 }
151
152 // Value is outside the range of the table. Apply the scale factor instead.
153 float minLogicalSize = std::max<float>(settings.minimumLogicalFontSize, 1);
154 return std::max(fontSizeFactors[keywordID - CSSValueXxSmall] * mediumSize, minLogicalSize);
155}

157float fontSizeForKeyword(unsigned keywordID, bool shouldUseFixedDefaultSize, const Document& document)
158{
159 return fontSizeForKeyword(keywordID, shouldUseFixedDefaultSize, document.settingsValues(), document.inQuirksMode());
160}

162template<typename T>
163static int findNearestLegacyFontSize(int pixelFontSize, const T* table, int multiplier)
164{
165 // Ignore table[0] because xx-small does not correspond to any legacy font size.
166 for (int i = 1; i < totalKeywords - 1; i++) {
167 if (pixelFontSize * 2 < (table[i] + table[i + 1]) * multiplier)
168 return i;
169 }
170 return totalKeywords - 1;
171}

173int legacyFontSizeForPixelSize(int pixelFontSize, bool shouldUseFixedDefaultSize, const Document& document)
174{
175 bool quirksMode = document.inQuirksMode();
176 int mediumSize = shouldUseFixedDefaultSize ? document.settings().defaultFixedFontSize() : document.settings().defaultFontSize();
177 if (mediumSize >= fontSizeTableMin && mediumSize <= fontSizeTableMax) {
178 int row = mediumSize - fontSizeTableMin;
179 return findNearestLegacyFontSize<int>(pixelFontSize, quirksMode ? quirksFontSizeTable[row] : strictFontSizeTable[row], 1);
180 }
181
182 return findNearestLegacyFontSize<float>(pixelFontSize, fontSizeFactors, mediumSize);
183}
