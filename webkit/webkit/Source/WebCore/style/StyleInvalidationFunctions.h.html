37template <typename TraverseFunction>
38inline void traverseRuleFeaturesInShadowTree(Element& element, TraverseFunction&& function)
39{
40 if (!element.shadowRoot())
41 return;
42 auto& shadowRuleSets = element.shadowRoot()->styleScope().resolver().ruleSets();
43 auto& authorStyle = shadowRuleSets.authorStyle();
44 bool hasHostPseudoClassRulesMatchingInShadowTree = authorStyle.hasHostPseudoClassRulesMatchingInShadowTree();
45 if (authorStyle.hostPseudoClassRules().isEmpty() && !hasHostPseudoClassRulesMatchingInShadowTree)
46 return;
47 function(shadowRuleSets.features(), hasHostPseudoClassRulesMatchingInShadowTree);
48}
49
50template <typename TraverseFunction>
51inline void traverseRuleFeaturesForSlotted(Element& element, TraverseFunction&& function)
52{
53 auto assignedShadowRoots = assignedShadowRootsIfSlotted(element);
54 for (auto& assignedShadowRoot : assignedShadowRoots) {
55 auto& ruleSets = assignedShadowRoot->styleScope().resolver().ruleSets();
56 if (ruleSets.authorStyle().slottedPseudoElementRules().isEmpty())
57 continue;
58 function(ruleSets.features(), false);
59 }
60}
61
62template <typename TraverseFunction>
63inline void traverseRuleFeatures(Element& element, TraverseFunction&& function)
64{
65 auto& ruleSets = element.styleResolver().ruleSets();
66
67 auto mayAffectShadowTree = [&] {
68 if (element.shadowRoot() && element.shadowRoot()->isUserAgentShadowRoot()) {
69 if (ruleSets.authorStyle().hasShadowPseudoElementRules())
70 return true;
71#if ENABLE(VIDEO)
72 if (element.isMediaElement() && !ruleSets.authorStyle().cuePseudoRules().isEmpty())
73 return true;
74#endif
75 }
76 if (is<HTMLSlotElement>(element) && !ruleSets.authorStyle().slottedPseudoElementRules().isEmpty())
77 return true;
78 return false;
79 };
80
81 function(ruleSets.features(), mayAffectShadowTree());
82
83 traverseRuleFeaturesInShadowTree(element, function);
84 traverseRuleFeaturesForSlotted(element, function);
85
86 // Ensure that the containing tree resolver also exists so it doesn't get created in the middle of invalidation.
87 if (element.isInShadowTree())
88 Style::Scope::forNode(*element.containingShadowRoot()->host()).resolver();
89}
