45// <https://html.spec.whatwg.org/multipage/urls-and-fetching.html#cors-settings-attributes>
46enum class LoadPolicy { CORS, NoCORS, Anonymous };
47static void loadPendingImage(Document& document, const StyleImage* styleImage, const Element* element, LoadPolicy loadPolicy = LoadPolicy::NoCORS)
48{
49 if (!styleImage || !styleImage->isPending())
50 return;
51
52 bool isInUserAgentShadowTree = element && element->isInUserAgentShadowTree();
53 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
54 options.contentSecurityPolicyImposition = isInUserAgentShadowTree ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
55
56 if (!isInUserAgentShadowTree && document.settings().useAnonymousModeWhenFetchingMaskImages()) {
57 switch (loadPolicy) {
58 case LoadPolicy::Anonymous:
59 options.storedCredentialsPolicy = StoredCredentialsPolicy::DoNotUse;
60 FALLTHROUGH;
61 case LoadPolicy::CORS:
62 options.mode = FetchOptions::Mode::Cors;
63 options.credentials = FetchOptions::Credentials::SameOrigin;
64 options.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
65 break;
66 case LoadPolicy::NoCORS:
67 break;
68 }
69 }
70
71 const_cast<StyleImage&>(*styleImage).load(document.cachedResourceLoader(), options);
72}

74void loadPendingResources(RenderStyle& style, Document& document, const Element* element)
75{
76 for (auto* backgroundLayer = &style.backgroundLayers(); backgroundLayer; backgroundLayer = backgroundLayer->next())
77 loadPendingImage(document, backgroundLayer->image(), element);
78
79 for (auto* contentData = style.contentData(); contentData; contentData = contentData->next()) {
80 if (is<ImageContentData>(*contentData)) {
81 auto& styleImage = downcast<ImageContentData>(*contentData).image();
82 loadPendingImage(document, &styleImage, element);
83 }
84 }
85
86 if (auto* cursorList = style.cursors()) {
87 for (size_t i = 0; i < cursorList->size(); ++i)
88 loadPendingImage(document, cursorList->at(i).image(), element);
89 }
90
91 loadPendingImage(document, style.listStyleImage(), element);
92 loadPendingImage(document, style.borderImageSource(), element);
93 loadPendingImage(document, style.maskBoxImageSource(), element);
94
95 if (auto* reflection = style.boxReflect())
96 loadPendingImage(document, reflection->mask().image(), element);
97
98 // Masking operations may be sensitive to timing attacks that can be used to reveal the pixel data of
99 // the image used as the mask. As a means to mitigate such attacks CSS mask images and shape-outside
100 // images are retreived in "Anonymous" mode, which uses a potentially CORS-enabled fetch.
101 for (auto* maskLayer = &style.maskLayers(); maskLayer; maskLayer = maskLayer->next())
102 loadPendingImage(document, maskLayer->image(), element, LoadPolicy::CORS);
103
104 if (style.shapeOutside())
105 loadPendingImage(document, style.shapeOutside()->image(), element, LoadPolicy::Anonymous);
106}
