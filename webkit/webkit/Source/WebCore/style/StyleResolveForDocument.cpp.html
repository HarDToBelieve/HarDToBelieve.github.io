53RenderStyle resolveForDocument(const Document& document)
54{
55 ASSERT(document.hasLivingRenderTree());
56
57 RenderView& renderView = *document.renderView();
58
59 auto documentStyle = RenderStyle::create();
60
61 documentStyle.setDisplay(DisplayType::Block);
62 documentStyle.setRTLOrdering(document.visuallyOrdered() ? Order::Visual : Order::Logical);
63 documentStyle.setZoom(!document.printing() ? renderView.frame().pageZoomFactor() : 1);
64 documentStyle.setPageScaleTransform(renderView.frame().frameScaleFactor());
65 FontCascadeDescription documentFontDescription = documentStyle.fontDescription();
66 documentFontDescription.setSpecifiedLocale(document.contentLanguage());
67 documentStyle.setFontDescription(WTFMove(documentFontDescription));
68
69 // This overrides any -webkit-user-modify inherited from the parent iframe.
70 documentStyle.setUserModify(document.inDesignMode() ? UserModify::ReadWrite : UserModify::ReadOnly);
71#if PLATFORM(IOS_FAMILY)
72 if (document.inDesignMode())
73 documentStyle.setTextSizeAdjust(TextSizeAdjustment(NoTextSizeAdjustment));
74#endif
75
76 Adjuster::adjustEventListenerRegionTypesForRootStyle(documentStyle, document);
77
78 Element* docElement = document.documentElement();
79 RenderObject* docElementRenderer = docElement ? docElement->renderer() : nullptr;
80 if (docElementRenderer) {
81 // Use the direction and writing-mode of the body to set the
82 // viewport's direction and writing-mode unless the property is set on the document element.
83 // If there is no body, then use the document element.
84 auto* body = document.bodyOrFrameset();
85 RenderObject* bodyRenderer = body ? body->renderer() : nullptr;
86 if (bodyRenderer && !docElementRenderer->style().hasExplicitlySetWritingMode())
87 documentStyle.setWritingMode(bodyRenderer->style().writingMode());
88 else
89 documentStyle.setWritingMode(docElementRenderer->style().writingMode());
90 if (bodyRenderer && !docElementRenderer->style().hasExplicitlySetDirection())
91 documentStyle.setDirection(bodyRenderer->style().direction());
92 else
93 documentStyle.setDirection(docElementRenderer->style().direction());
94 }
95
96 const Pagination& pagination = renderView.frameView().pagination();
97 if (pagination.mode != Pagination::Unpaginated) {
98 documentStyle.setColumnStylesFromPaginationMode(pagination.mode);
99 documentStyle.setColumnGap(GapLength(Length((int) pagination.gap, LengthType::Fixed)));
100 if (renderView.multiColumnFlow())
101 renderView.updateColumnProgressionFromStyle(documentStyle);
102 if (renderView.page().paginationLineGridEnabled()) {
103 documentStyle.setLineGrid("-webkit-default-pagination-grid");
104 documentStyle.setLineSnap(LineSnap::Contain);
105 }
106 }
107
108 const Settings& settings = renderView.frame().settings();
109
110 FontCascadeDescription fontDescription;
111 fontDescription.setSpecifiedLocale(document.contentLanguage());
112 fontDescription.setRenderingMode(settings.fontRenderingMode());
113 fontDescription.setOneFamily(standardFamily);
114 fontDescription.setShouldAllowUserInstalledFonts(settings.shouldAllowUserInstalledFonts() ? AllowUserInstalledFonts::Yes : AllowUserInstalledFonts::No);
115
116 fontDescription.setKeywordSizeFromIdentifier(CSSValueMedium);
117 int size = fontSizeForKeyword(CSSValueMedium, false, document);
118 fontDescription.setSpecifiedSize(size);
119 bool useSVGZoomRules = document.isSVGDocument();
120 fontDescription.setComputedSize(computedFontSizeFromSpecifiedSize(size, fontDescription.isAbsoluteSize(), useSVGZoomRules, &documentStyle, document));
121
122 auto [fontOrientation, glyphOrientation] = documentStyle.fontAndGlyphOrientation();
123 fontDescription.setOrientation(fontOrientation);
124 fontDescription.setNonCJKGlyphOrientation(glyphOrientation);
125
126 documentStyle.setFontDescription(WTFMove(fontDescription));
127
128 documentStyle.fontCascade().update(&const_cast<Document&>(document).fontSelector());
129
130 return documentStyle;
131}
