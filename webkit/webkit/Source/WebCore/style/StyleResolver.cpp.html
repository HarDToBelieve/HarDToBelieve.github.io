95Resolver::Resolver(Document& document)
96 : m_ruleSets(*this)
97 , m_document(document)
98 , m_matchAuthorAndUserStyles(m_document.settings().authorAndUserStylesEnabled())
99{
100 UserAgentStyle::initDefaultStyleSheet();
101
102 // construct document root element default style. this is needed
103 // to evaluate media queries that contain relative constraints, like "screen and (max-width: 10em)"
104 // This is here instead of constructor, because when constructor is run,
105 // document doesn't have documentElement
106 // NOTE: this assumes that element that gets passed to styleForElement -call
107 // is always from the document that owns the style selector
108 FrameView* view = m_document.view();
109 if (view)
110 m_mediaQueryEvaluator = MediaQueryEvaluator { view->mediaType() };
111 else
112 m_mediaQueryEvaluator = MediaQueryEvaluator { };
113
114 if (auto* documentElement = m_document.documentElement()) {
115 m_rootDefaultStyle = styleForElement(*documentElement, m_document.renderStyle(), nullptr, RuleMatchingBehavior::MatchOnlyUserAgentRules).renderStyle;
116 // Turn off assertion against font lookups during style resolver initialization. We may need root style font for media queries.
117 m_document.fontSelector().incrementIsComputingRootStyleFont();
118 m_rootDefaultStyle->fontCascade().update(&m_document.fontSelector());
119 m_rootDefaultStyle->fontCascade().primaryFont();
120 m_document.fontSelector().decrementIsComputingRootStyleFont();
121 }
122
123 if (m_rootDefaultStyle && view)
124 m_mediaQueryEvaluator = MediaQueryEvaluator { view->mediaType(), m_document, m_rootDefaultStyle.get() };
125
126 m_ruleSets.resetAuthorStyle();
127 m_ruleSets.resetUserAgentMediaQueryStyle();
128}

161Resolver::State::State(const Element& element, const RenderStyle* parentStyle, const RenderStyle* documentElementStyle)
162 : m_element(&element)
163 , m_parentStyle(parentStyle)
164{
165 bool resetStyleInheritance = hasShadowRootParent(element) && downcast<ShadowRoot>(element.parentNode())->resetStyleInheritance();
166 if (resetStyleInheritance)
167 m_parentStyle = nullptr;
168
169 auto& document = element.document();
170 auto* documentElement = document.documentElement();
171 if (!documentElement || documentElement == &element)
172 m_rootElementStyle = document.renderStyle();
173 else
174 m_rootElementStyle = documentElementStyle ? documentElementStyle : documentElement->renderStyle();
175}

203ElementStyle Resolver::styleForElement(const Element& element, const RenderStyle* parentStyle, const RenderStyle* parentBoxStyle, RuleMatchingBehavior matchingBehavior, const SelectorFilter* selectorFilter)
204{
205 RELEASE_ASSERT(!m_isDeleted);
206
207 auto state = State(element, parentStyle, m_overrideDocumentElementStyle);
208
209 if (state.parentStyle()) {
210 state.setStyle(RenderStyle::createPtr());
211 state.style()->inheritFrom(*state.parentStyle());
212 } else {
213 state.setStyle(defaultStyleForElement(&element));
214 state.setParentStyle(RenderStyle::clonePtr(*state.style()));
215 }
216
217 auto& style = *state.style();
218
219 if (element.isLink()) {
220 style.setIsLink(true);
221 InsideLink linkState = document().visitedLinkState().determineLinkState(element);
222 if (linkState != InsideLink::NotInside) {
223 bool forceVisited = InspectorInstrumentation::forcePseudoState(element, CSSSelector::PseudoClassVisited);
224 if (forceVisited)
225 linkState = InsideLink::InsideVisited;
226 }
227 style.setInsideLink(linkState);
228 }
229
230 UserAgentStyle::ensureDefaultStyleSheetsForElement(element);
231
232 ElementRuleCollector collector(element, m_ruleSets, selectorFilter);
233 collector.setMedium(&m_mediaQueryEvaluator);
234
235 if (matchingBehavior == RuleMatchingBehavior::MatchOnlyUserAgentRules)
236 collector.matchUARules();
237 else
238 collector.matchAllRules(m_matchAuthorAndUserStyles, matchingBehavior != RuleMatchingBehavior::MatchAllRulesExcludingSMIL);
239
240 if (collector.matchedPseudoElementIds())
241 style.setHasPseudoStyles(collector.matchedPseudoElementIds());
242
243 // This is required for style sharing.
244 if (collector.didMatchUncommonAttributeSelector())
245 style.setUnique();
246
247 auto elementStyleRelations = commitRelationsToRenderStyle(style, element, collector.styleRelations());
248
249 applyMatchedProperties(state, collector.matchResult());
250
251 Adjuster adjuster(document(), *state.parentStyle(), parentBoxStyle, &element);
252 adjuster.adjust(*state.style(), state.userAgentAppearanceStyle());
253
254 if (state.style()->hasViewportUnits())
255 document().setHasStyleWithViewportUnits();
256
257 return { state.takeStyle(), WTFMove(elementStyleRelations) };
258}

260std::unique_ptr<RenderStyle> Resolver::styleForKeyframe(const Element& element, const RenderStyle* elementStyle, const RenderStyle* parentElementStyle, const StyleRuleKeyframe* keyframe, KeyframeValue& keyframeValue)
261{
262 RELEASE_ASSERT(!m_isDeleted);
263
264 MatchResult result;
265 result.authorDeclarations.append({ &keyframe->properties() });
266
267 auto state = State(element, nullptr, m_overrideDocumentElementStyle);
268
269 state.setStyle(RenderStyle::clonePtr(*elementStyle));
270 state.setParentStyle(RenderStyle::clonePtr(parentElementStyle ? *parentElementStyle : *elementStyle));
271
272 Builder builder(*state.style(), builderContext(state), result, { CascadeLevel::Author });
273 builder.applyAllProperties();
274
275 Adjuster adjuster(document(), *state.parentStyle(), nullptr, nullptr);
276 adjuster.adjust(*state.style(), state.userAgentAppearanceStyle());
277
278 // Add all the animating properties to the keyframe.
279 unsigned propertyCount = keyframe->properties().propertyCount();
280 for (unsigned i = 0; i < propertyCount; ++i) {
281 CSSPropertyID property = keyframe->properties().propertyAt(i).id();
282 // Timing-function within keyframes is special, because it is not animated; it just
283 // describes the timing function between this keyframe and the next.
284 if (property != CSSPropertyAnimationTimingFunction)
285 keyframeValue.addProperty(property);
286 }
287
288 return state.takeStyle();
289}

296void Resolver::keyframeStylesForAnimation(const Element& element, const RenderStyle* elementStyle, const RenderStyle* parentElementStyle, KeyframeList& list)
297{
298 list.clear();
299
300 // Get the keyframesRule for this name.
301 if (list.animationName().isEmpty())
302 return;
303
304 m_keyframesRuleMap.checkConsistency();
305
306 KeyframesRuleMap::iterator it = m_keyframesRuleMap.find(list.animationName().impl());
307 if (it == m_keyframesRuleMap.end())
308 return;
309
310 const StyleRuleKeyframes* keyframesRule = it->value.get();
311
312 auto* keyframes = &keyframesRule->keyframes();
313 Vector<Ref<StyleRuleKeyframe>> newKeyframesIfNecessary;
314
315 bool hasDuplicateKeys = false;
316 HashSet<double> keyframeKeys;
317 for (auto& keyframe : *keyframes) {
318 for (auto key : keyframe->keys()) {
319 if (!keyframeKeys.add(key)) {
320 hasDuplicateKeys = true;
321 break;
322 }
323 }
324 if (hasDuplicateKeys)
325 break;
326 }
327
328 // FIXME: If HashMaps could have Ref<> as value types, we wouldn't need
329 // to copy the HashMap into a Vector.
330 if (hasDuplicateKeys) {
331 // Merge duplicate key times.
332 HashMap<double, RefPtr<StyleRuleKeyframe>> keyframesMap;
333
334 for (auto& originalKeyframe : keyframesRule->keyframes()) {
335 for (auto key : originalKeyframe->keys()) {
336 if (auto keyframe = keyframesMap.get(key))
337 keyframe->mutableProperties().mergeAndOverrideOnConflict(originalKeyframe->properties());
338 else {
339 auto StyleRuleKeyframe = StyleRuleKeyframe::create(MutableStyleProperties::create());
340 StyleRuleKeyframe.ptr()->setKey(key);
341 StyleRuleKeyframe.ptr()->mutableProperties().mergeAndOverrideOnConflict(originalKeyframe->properties());
342 keyframesMap.set(key, StyleRuleKeyframe.ptr());
343 }
344 }
345 }
346
347 for (auto& keyframe : keyframesMap.values())
348 newKeyframesIfNecessary.append(*keyframe.get());
349
350 keyframes = &newKeyframesIfNecessary;
351 }
352
353 // Construct and populate the style for each keyframe.
354 for (auto& keyframe : *keyframes) {
355 // Add this keyframe style to all the indicated key times
356 for (auto key : keyframe->keys()) {
357 KeyframeValue keyframeValue(0, nullptr);
358 keyframeValue.setStyle(styleForKeyframe(element, elementStyle, parentElementStyle, keyframe.ptr(), keyframeValue));
359 keyframeValue.setKey(key);
360 if (auto timingFunctionCSSValue = keyframe->properties().getPropertyCSSValue(CSSPropertyAnimationTimingFunction))
361 keyframeValue.setTimingFunction(TimingFunction::createFromCSSValue(*timingFunctionCSSValue.get()));
362 list.insert(WTFMove(keyframeValue));
363 }
364 }
365
366 list.fillImplicitKeyframes(element, *this, elementStyle, parentElementStyle);
367}

369std::unique_ptr<RenderStyle> Resolver::pseudoStyleForElement(const Element& element, const PseudoElementRequest& pseudoElementRequest, const RenderStyle& parentStyle, const RenderStyle* parentBoxStyle, const SelectorFilter* selectorFilter)
370{
371 auto state = State(element, &parentStyle, m_overrideDocumentElementStyle);
372
373 if (state.parentStyle()) {
374 state.setStyle(RenderStyle::createPtr());
375 state.style()->inheritFrom(*state.parentStyle());
376 } else {
377 state.setStyle(defaultStyleForElement(&element));
378 state.setParentStyle(RenderStyle::clonePtr(*state.style()));
379 }
380
381 ElementRuleCollector collector(element, m_ruleSets, selectorFilter);
382 collector.setPseudoElementRequest(pseudoElementRequest);
383 collector.setMedium(&m_mediaQueryEvaluator);
384 collector.matchUARules();
385
386 if (m_matchAuthorAndUserStyles) {
387 collector.matchUserRules();
388 collector.matchAuthorRules();
389 }
390
391 ASSERT(!collector.matchedPseudoElementIds());
392
393 if (collector.matchResult().isEmpty())
394 return nullptr;
395
396 state.style()->setStyleType(pseudoElementRequest.pseudoId);
397
398 applyMatchedProperties(state, collector.matchResult());
399
400 Adjuster adjuster(document(), *state.parentStyle(), parentBoxStyle, nullptr);
401 adjuster.adjust(*state.style(), state.userAgentAppearanceStyle());
402
403 if (state.style()->hasViewportUnits())
404 document().setHasStyleWithViewportUnits();
405
406 return state.takeStyle();
407}

409std::unique_ptr<RenderStyle> Resolver::styleForPage(int pageIndex)
410{
411 RELEASE_ASSERT(!m_isDeleted);
412
413 auto* documentElement = m_document.documentElement();
414 if (!documentElement)
415 return RenderStyle::createPtr();
416
417 auto state = State(*documentElement, m_document.renderStyle());
418
419 state.setStyle(RenderStyle::createPtr());
420 state.style()->inheritFrom(*state.rootElementStyle());
421
422 PageRuleCollector collector(state, m_ruleSets);
423 collector.matchAllPageRules(pageIndex);
424
425 auto& result = collector.matchResult();
426
427 Builder builder(*state.style(), builderContext(state), result, { CascadeLevel::Author });
428 builder.applyAllProperties();
429
430 // Now return the style.
431 return state.takeStyle();
432}

434std::unique_ptr<RenderStyle> Resolver::defaultStyleForElement(const Element* element)
435{
436 auto style = RenderStyle::createPtr();
437
438 FontCascadeDescription fontDescription;
439 fontDescription.setRenderingMode(settings().fontRenderingMode());
440 fontDescription.setOneFamily(standardFamily);
441 fontDescription.setKeywordSizeFromIdentifier(CSSValueMedium);
442
443 auto size = fontSizeForKeyword(CSSValueMedium, false, document());
444 fontDescription.setSpecifiedSize(size);
445 fontDescription.setComputedSize(computedFontSizeFromSpecifiedSize(size, fontDescription.isAbsoluteSize(), is<SVGElement>(element), style.get(), document()));
446
447 fontDescription.setShouldAllowUserInstalledFonts(settings().shouldAllowUserInstalledFonts() ? AllowUserInstalledFonts::Yes : AllowUserInstalledFonts::No);
448 style->setFontDescription(WTFMove(fontDescription));
449
450 style->fontCascade().update(&document().fontSelector());
451
452 return style;
453}

460Vector<RefPtr<const StyleRule>> Resolver::pseudoStyleRulesForElement(const Element* element, PseudoId pseudoId, unsigned rulesToInclude)
461{
462 if (!element)
463 return { };
464
465 auto state = State(*element, nullptr);
466
467 ElementRuleCollector collector(*element, m_ruleSets, nullptr);
468 collector.setMode(SelectorChecker::Mode::CollectingRules);
469 collector.setPseudoElementRequest({ pseudoId });
470 collector.setMedium(&m_mediaQueryEvaluator);
471 collector.setIncludeEmptyRules(rulesToInclude & EmptyCSSRules);
472
473 if (rulesToInclude & UAAndUserCSSRules) {
474 // First we match rules from the user agent sheet.
475 collector.matchUARules();
476
477 // Now we check user sheet rules.
478 if (m_matchAuthorAndUserStyles)
479 collector.matchUserRules();
480 }
481
482 if (m_matchAuthorAndUserStyles && (rulesToInclude & AuthorCSSRules))
483 collector.matchAuthorRules();
484
485 return collector.matchedRuleList();
486}

488static bool elementTypeHasAppearanceFromUAStyle(const Element& element)
489{
490 // NOTE: This is just a hard-coded list of elements that have some -webkit-appearance value in html.css
491 const auto& localName = element.localName();
492 return localName == HTMLNames::inputTag
493 || localName == HTMLNames::textareaTag
494 || localName == HTMLNames::buttonTag
495 || localName == HTMLNames::progressTag
496 || localName == HTMLNames::selectTag
497 || localName == HTMLNames::meterTag;
498}

510void Resolver::applyMatchedProperties(State& state, const MatchResult& matchResult, UseMatchedDeclarationsCache useMatchedDeclarationsCache)
511{
512 unsigned cacheHash = useMatchedDeclarationsCache == UseMatchedDeclarationsCache::Yes ? MatchedDeclarationsCache::computeHash(matchResult) : 0;
513 auto includedProperties = PropertyCascade::IncludedProperties::All;
514
515 auto& style = *state.style();
516 auto& parentStyle = *state.parentStyle();
517 auto& element = *state.element();
518
519 auto* cacheEntry = m_matchedDeclarationsCache.find(cacheHash, matchResult);
520 if (cacheEntry && MatchedDeclarationsCache::isCacheable(element, style, parentStyle)) {
521 // We can build up the style by copying non-inherited properties from an earlier style object built using the same exact
522 // style declarations. We then only need to apply the inherited properties, if any, as their values can depend on the
523 // element context. This is fast and saves memory by reusing the style data structures.
524 style.copyNonInheritedFrom(*cacheEntry->renderStyle);
525
526 if (parentStyle.inheritedEqual(*cacheEntry->parentRenderStyle) && !isAtShadowBoundary(element)) {
527 InsideLink linkStatus = state.style()->insideLink();
528 // If the cache item parent style has identical inherited properties to the current parent style then the
529 // resulting style will be identical too. We copy the inherited properties over from the cache and are done.
530 style.inheritFrom(*cacheEntry->renderStyle);
531
532 // Unfortunately the link status is treated like an inherited property. We need to explicitly restore it.
533 style.setInsideLink(linkStatus);
534 return;
535 }
536
537 includedProperties = PropertyCascade::IncludedProperties::InheritedOnly;
538 }
539
540 if (elementTypeHasAppearanceFromUAStyle(element)) {
541 // Find out if there's a -webkit-appearance property in effect from the UA sheet.
542 // If so, we cache the border and background styles so that RenderTheme::adjustStyle()
543 // can look at them later to figure out if this is a styled form control or not.
544 auto userAgentStyle = RenderStyle::clonePtr(style);
545 Builder builder(*userAgentStyle, builderContext(state), matchResult, { CascadeLevel::UserAgent });
546 builder.applyAllProperties();
547
548 state.setUserAgentAppearanceStyle(WTFMove(userAgentStyle));
549 }
550
551 Builder builder(*state.style(), builderContext(state), matchResult, allCascadeLevels(), includedProperties);
552
553 // High priority properties may affect resolution of other properties (they are mostly font related).
554 builder.applyHighPriorityProperties();
555
556 if (cacheEntry && !cacheEntry->isUsableAfterHighPriorityProperties(style)) {
557 // We need to resolve all properties without caching.
558 applyMatchedProperties(state, matchResult, UseMatchedDeclarationsCache::No);
559 return;
560 }
561
562 builder.applyLowPriorityProperties();
563
564 for (auto& contentAttribute : builder.state().registeredContentAttributes())
565 ruleSets().mutableFeatures().registerContentAttribute(contentAttribute);
566
567 if (cacheEntry || !cacheHash)
568 return;
569
570 if (MatchedDeclarationsCache::isCacheable(element, style, parentStyle))
571 m_matchedDeclarationsCache.add(style, parentStyle, cacheHash, matchResult);
572}
