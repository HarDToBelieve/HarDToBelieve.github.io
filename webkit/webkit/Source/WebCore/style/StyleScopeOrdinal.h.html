27namespace WebCore {
28namespace Style {
29
30// This is used to identify style scopes that can affect an element.
31// Scopes are in tree-of-trees order. Styles from earlier scopes win over later ones (modulo !important).
32enum class ScopeOrdinal : int {
33 ContainingHost = -1, // ::part rules and author-exposed UA pseudo classes from the host tree scope.
34 Element = 0, // Normal rules in the same tree where the element is.
35 FirstSlot = 1, // ::slotted rules in the parent's shadow tree. Values greater than FirstSlot indicate subsequent slots in the chain.
36 Shadow = std::numeric_limits<int>::max(), // :host rules in element's own shadow tree.
37};
38
39inline ScopeOrdinal& operator++(ScopeOrdinal& ordinal)
40{
41 ASSERT(ordinal < ScopeOrdinal::Shadow);
42 return ordinal = static_cast<ScopeOrdinal>(static_cast<int>(ordinal) + 1);
43}
