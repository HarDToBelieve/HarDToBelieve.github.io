45struct InvalidationRuleSet {
46 MatchElement matchElement;
47 Ref<RuleSet> ruleSet;
48 Vector<const CSSSelector*> invalidationSelectors;

53class ScopeRuleSets {
54public:
55 ScopeRuleSets(Resolver&);
56 ~ScopeRuleSets();
57
58 bool isAuthorStyleDefined() const { return m_isAuthorStyleDefined; }
59 RuleSet* userAgentMediaQueryStyle() const;
60 RuleSet& authorStyle() const { return *m_authorStyle; }
61 RuleSet* userStyle() const;
62 const RuleFeatureSet& features() const;
63 RuleSet* sibling() const { return m_siblingRuleSet.get(); }
64 RuleSet* uncommonAttribute() const { return m_uncommonAttributeRuleSet.get(); }
65
66 const Vector<InvalidationRuleSet>* classInvalidationRuleSets(const AtomString& className) const;
67 const Vector<InvalidationRuleSet>* attributeInvalidationRuleSets(const AtomString& attributeName) const;
68 const Vector<InvalidationRuleSet>* pseudoClassInvalidationRuleSets(CSSSelector::PseudoClassType) const;
69
70 bool hasComplexSelectorsForStyleAttribute() const;
71
72 void setUsesSharedUserStyle(bool b) { m_usesSharedUserStyle = b; }
73 void initializeUserStyle();
74
75 void resetAuthorStyle();
76 void appendAuthorStyleSheets(const Vector<RefPtr<CSSStyleSheet>>&, MediaQueryEvaluator*, Style::InspectorCSSOMWrappers&);
77
78 void resetUserAgentMediaQueryStyle();
79
80 bool hasViewportDependentMediaQueries() const;
81
82 std::optional<DynamicMediaQueryEvaluationChanges> evaluateDynamicMediaQueryRules(const MediaQueryEvaluator&);
83
84 RuleFeatureSet& mutableFeatures();
85
86 bool& isInvalidatingStyleWithRuleSets() { return m_isInvalidatingStyleWithRuleSets; }
87
88private:
89 void collectFeatures() const;
90 void collectRulesFromUserStyleSheets(const Vector<RefPtr<CSSStyleSheet>>&, RuleSet& userStyle, const MediaQueryEvaluator&);
91 void updateUserAgentMediaQueryStyleIfNeeded() const;
92
93 RefPtr<RuleSet> m_authorStyle;
94 mutable RefPtr<RuleSet> m_userAgentMediaQueryStyle;
95 RefPtr<RuleSet> m_userStyle;
96
97 Resolver& m_styleResolver;
98 mutable RuleFeatureSet m_features;
99 mutable RefPtr<RuleSet> m_siblingRuleSet;
100 mutable RefPtr<RuleSet> m_uncommonAttributeRuleSet;
101 mutable HashMap<AtomString, std::unique_ptr<Vector<InvalidationRuleSet>>> m_classInvalidationRuleSets;
102 mutable HashMap<AtomString, std::unique_ptr<Vector<InvalidationRuleSet>>> m_attributeInvalidationRuleSets;
103 mutable HashMap<CSSSelector::PseudoClassType, std::unique_ptr<Vector<InvalidationRuleSet>>, WTF::IntHash<CSSSelector::PseudoClassType>, WTF::StrongEnumHashTraits<CSSSelector::PseudoClassType>> m_pseudoClassInvalidationRuleSets;
104
105 mutable std::optional<bool> m_cachedHasComplexSelectorsForStyleAttribute;
106
107 mutable unsigned m_defaultStyleVersionOnFeatureCollection { 0 };
108 mutable unsigned m_userAgentMediaQueryRuleCountOnUpdate { 0 };
109
110 bool m_usesSharedUserStyle { false };
111 bool m_isAuthorStyleDefined { false };
112
113 // For catching <rdar://problem/53413013>
114 bool m_isInvalidatingStyleWithRuleSets { false };
115};
116
117inline const RuleFeatureSet& ScopeRuleSets::features() const
118{
119 if (m_defaultStyleVersionOnFeatureCollection < UserAgentStyle::defaultStyleVersion)
120 collectFeatures();
121 return m_features;
122}
123
124// FIXME: There should be just the const version.
125inline RuleFeatureSet& ScopeRuleSets::mutableFeatures()
126{
127 if (m_defaultStyleVersionOnFeatureCollection < UserAgentStyle::defaultStyleVersion)
128 collectFeatures();
129 return m_features;
130}
