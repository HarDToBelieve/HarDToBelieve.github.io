75std::unique_ptr<RenderStyle> SharingResolver::resolve(const Styleable& searchStyleable, const Update& update)
76{
77 if (!is<StyledElement>(searchStyleable.element))
78 return nullptr;
79 auto& element = downcast<StyledElement>(searchStyleable.element);
80 if (!element.parentElement())
81 return nullptr;
82 auto& parentElement = *element.parentElement();
83 if (parentElement.shadowRoot())
84 return nullptr;
85 if (!update.elementStyle(parentElement))
86 return nullptr;
87 // If the element has inline style it is probably unique.
88 if (element.inlineStyle())
89 return nullptr;
90 if (element.isSVGElement() && downcast<SVGElement>(element).animatedSMILStyleProperties())
91 return nullptr;
92 // Ids stop style sharing if they show up in the stylesheets.
93 auto& id = element.idForStyleResolution();
94 if (!id.isNull() && m_ruleSets.features().idsInRules.contains(id))
95 return nullptr;
96 if (parentElementPreventsSharing(parentElement))
97 return nullptr;
98 if (&element == m_document.cssTarget())
99 return nullptr;
100 if (elementHasDirectionAuto(element))
101 return nullptr;
102 if (element.shadowRoot() && !element.shadowRoot()->styleScope().resolver().ruleSets().authorStyle().hostPseudoClassRules().isEmpty())
103 return nullptr;
104 if (auto* keyframeEffectStack = searchStyleable.keyframeEffectStack()) {
105 if (keyframeEffectStack->hasEffectWithImplicitKeyframes())
106 return nullptr;
107 }
108
109 Context context {
110 update,
111 element,
112 element.hasClass() && classNamesAffectedByRules(element.classNames()),
113 m_document.visitedLinkState().determineLinkState(element)
114 };
115
116 // Check previous siblings and their cousins.
117 unsigned count = 0;
118 StyledElement* shareElement = nullptr;
119 Node* cousinList = element.previousSibling();
120 while (cousinList) {
121 shareElement = findSibling(context, cousinList, count);
122 if (shareElement)
123 break;
124 if (count >= cStyleSearchThreshold)
125 break;
126 cousinList = locateCousinList(cousinList->parentElement());
127 }
128
129 // If we have exhausted all our budget or our cousins.
130 if (!shareElement)
131 return nullptr;
132
133 // Can't share if sibling rules apply. This is checked at the end as it should rarely fail.
134 if (styleSharingCandidateMatchesRuleSet(element, m_ruleSets.sibling()))
135 return nullptr;
136 // Can't share if attribute rules apply.
137 if (styleSharingCandidateMatchesRuleSet(element, m_ruleSets.uncommonAttribute()))
138 return nullptr;
139 // Tracking child index requires unique style for each node. This may get set by the sibling rule match above.
140 if (parentElementPreventsSharing(parentElement))
141 return nullptr;
142
143 m_elementsSharingStyle.add(&element, shareElement);
144
145 return RenderStyle::clonePtr(*update.elementStyle(*shareElement));
146}
147
148StyledElement* SharingResolver::findSibling(const Context& context, Node* node, unsigned& count) const
149{
150 for (; node; node = node->previousSibling()) {
151 if (!is<StyledElement>(*node))
152 continue;
153 if (canShareStyleWithElement(context, downcast<StyledElement>(*node)))
154 break;
155 if (count++ >= cStyleSearchThreshold)
156 return nullptr;
157 }
158 return downcast<StyledElement>(node);
159}
160
161Node* SharingResolver::locateCousinList(const Element* parent) const
162{
163 for (unsigned count = 0; count < cStyleSearchThreshold; ++count) {
164 auto* elementSharingParentStyle = m_elementsSharingStyle.get(parent);
165 if (!elementSharingParentStyle)
166 return nullptr;
167 if (!parentElementPreventsSharing(*elementSharingParentStyle)) {
168 if (auto* cousin = elementSharingParentStyle->lastChild())
169 return cousin;
170 }
171 parent = elementSharingParentStyle;
172 }
173
174 return nullptr;
175}

177static bool canShareStyleWithControl(const HTMLFormControlElement& element, const HTMLFormControlElement& formElement)
178{
179 if (!is<HTMLInputElement>(formElement) || !is<HTMLInputElement>(element))
180 return false;
181
182 auto& thisInputElement = downcast<HTMLInputElement>(formElement);
183 auto& otherInputElement = downcast<HTMLInputElement>(element);
184
185 if (thisInputElement.isAutoFilled() != otherInputElement.isAutoFilled())
186 return false;
187 if (thisInputElement.shouldAppearChecked() != otherInputElement.shouldAppearChecked())
188 return false;
189 if (thisInputElement.isRequired() != otherInputElement.isRequired())
190 return false;
191
192 if (formElement.isDisabledFormControl() != element.isDisabledFormControl())
193 return false;
194
195 if (formElement.isInRange() != element.isInRange())
196 return false;
197
198 if (formElement.isOutOfRange() != element.isOutOfRange())
199 return false;
200
201 return true;
202}

204bool SharingResolver::canShareStyleWithElement(const Context& context, const StyledElement& candidateElement) const
205{
206 auto& element = context.element;
207 auto* style = context.update.elementStyle(candidateElement);
208 if (!style)
209 return false;
210 if (style->unique())
211 return false;
212 if (style->hasUniquePseudoStyle())
213 return false;
214 if (candidateElement.tagQName() != element.tagQName())
215 return false;
216 if (candidateElement.inlineStyle())
217 return false;
218 if (candidateElement.needsStyleRecalc())
219 return false;
220 if (candidateElement.isSVGElement() && downcast<SVGElement>(candidateElement).animatedSMILStyleProperties())
221 return false;
222 if (candidateElement.isLink() != element.isLink())
223 return false;
224 if (candidateElement.hovered() != element.hovered())
225 return false;
226 if (candidateElement.active() != element.active())
227 return false;
228 if (candidateElement.focused() != element.focused())
229 return false;
230 if (candidateElement.hasFocusVisible() != element.hasFocusVisible())
231 return false;
232 if (candidateElement.hasFocusWithin() != element.hasFocusWithin())
233 return false;
234 if (candidateElement.isBeingDragged() != element.isBeingDragged())
235 return false;
236 if (candidateElement.shadowPseudoId() != element.shadowPseudoId())
237 return false;
238 if (element.isInShadowTree() && candidateElement.partNames() != element.partNames())
239 return false;
240 if (&candidateElement == m_document.cssTarget())
241 return false;
242 if (!sharingCandidateHasIdenticalStyleAffectingAttributes(context, candidateElement))
243 return false;
244 if (const_cast<StyledElement&>(candidateElement).additionalPresentationalHintStyle() != const_cast<StyledElement&>(element).additionalPresentationalHintStyle())
245 return false;
246 if (candidateElement.affectsNextSiblingElementStyle() || candidateElement.styleIsAffectedByPreviousSibling())
247 return false;
248
249 auto& candidateElementId = candidateElement.idForStyleResolution();
250 if (!candidateElementId.isNull() && m_ruleSets.features().idsInRules.contains(candidateElementId))
251 return false;
252
253 bool isControl = is<HTMLFormControlElement>(candidateElement);
254
255 if (isControl != is<HTMLFormControlElement>(element))
256 return false;
257
258 if (isControl && !canShareStyleWithControl(downcast<HTMLFormControlElement>(element), downcast<HTMLFormControlElement>(candidateElement)))
259 return false;
260
261 if (style->transitions() || style->animations())
262 return false;
263
264 // Turn off style sharing for elements that can gain layers for reasons outside of the style system.
265 // See comments in RenderObject::setStyle().
266 if (candidateElement.hasTagName(HTMLNames::iframeTag) || candidateElement.hasTagName(HTMLNames::frameTag))
267 return false;
268
269 if (candidateElement.hasTagName(HTMLNames::embedTag) || candidateElement.hasTagName(HTMLNames::objectTag) || candidateElement.hasTagName(HTMLNames::appletTag) || candidateElement.hasTagName(HTMLNames::canvasTag))
270 return false;
271
272 if (elementHasDirectionAuto(candidateElement))
273 return false;
274
275 if (candidateElement.isLink() && context.elementLinkState != style->insideLink())
276 return false;
277
278 if (candidateElement.elementData() != element.elementData()) {
279 if (candidateElement.attributeWithoutSynchronization(HTMLNames::readonlyAttr) != element.attributeWithoutSynchronization(HTMLNames::readonlyAttr))
280 return false;
281 if (candidateElement.isSVGElement()) {
282 if (candidateElement.getAttribute(HTMLNames::typeAttr) != element.getAttribute(HTMLNames::typeAttr))
283 return false;
284 } else {
285 if (candidateElement.attributeWithoutSynchronization(HTMLNames::typeAttr) != element.attributeWithoutSynchronization(HTMLNames::typeAttr))
286 return false;
287 }
288 }
289
290 if (candidateElement.matchesValidPseudoClass() != element.matchesValidPseudoClass())
291 return false;
292
293 if (element.matchesInvalidPseudoClass() != element.matchesValidPseudoClass())
294 return false;
295
296 if (candidateElement.matchesIndeterminatePseudoClass() != element.matchesIndeterminatePseudoClass())
297 return false;
298
299 if (candidateElement.matchesDefaultPseudoClass() != element.matchesDefaultPseudoClass())
300 return false;
301
302 if (candidateElement.shadowRoot() && !candidateElement.shadowRoot()->styleScope().resolver().ruleSets().authorStyle().hostPseudoClassRules().isEmpty())
303 return false;
304
305#if ENABLE(WHEEL_EVENT_REGIONS)
306 if (candidateElement.hasEventListeners() || element.hasEventListeners())
307 return false;
308#endif
309
310#if ENABLE(FULLSCREEN_API)
311 if (&candidateElement == m_document.fullscreenManager().currentFullscreenElement() || &element == m_document.fullscreenManager().currentFullscreenElement())
312 return false;
313#endif
314 return true;
315}

317bool SharingResolver::styleSharingCandidateMatchesRuleSet(const StyledElement& element, const RuleSet* ruleSet) const
318{
319 if (!ruleSet)
320 return false;
321
322 ElementRuleCollector collector(const_cast<StyledElement&>(element), m_ruleSets, &m_selectorFilter);
323 return collector.hasAnyMatchingRules(ruleSet);
324}

326bool SharingResolver::sharingCandidateHasIdenticalStyleAffectingAttributes(const Context& context, const StyledElement& sharingCandidate) const
327{
328 auto& element = context.element;
329 if (element.elementData() == sharingCandidate.elementData())
330 return true;
331 if (element.attributeWithoutSynchronization(XMLNames::langAttr) != sharingCandidate.attributeWithoutSynchronization(XMLNames::langAttr))
332 return false;
333 if (element.attributeWithoutSynchronization(HTMLNames::langAttr) != sharingCandidate.attributeWithoutSynchronization(HTMLNames::langAttr))
334 return false;
335
336 if (context.elementAffectedByClassRules) {
337 if (!sharingCandidate.hasClass())
338 return false;
339 // SVG elements require a (slow!) getAttribute comparision because "class" is an animatable attribute for SVG.
340 if (element.isSVGElement()) {
341 if (element.getAttribute(HTMLNames::classAttr) != sharingCandidate.getAttribute(HTMLNames::classAttr))
342 return false;
343 } else {
344 if (element.classNames() != sharingCandidate.classNames())
345 return false;
346 }
347 } else if (sharingCandidate.hasClass() && classNamesAffectedByRules(sharingCandidate.classNames()))
348 return false;
349
350 if (const_cast<StyledElement&>(element).presentationalHintStyle() != const_cast<StyledElement&>(sharingCandidate).presentationalHintStyle())
351 return false;
352
353 return true;
354}

356bool SharingResolver::classNamesAffectedByRules(const SpaceSplitString& classNames) const
357{
358 for (unsigned i = 0; i < classNames.size(); ++i) {
359 if (m_ruleSets.features().classRules.contains(classNames[i]))
360 return true;
361 }
362 return false;
363}
