56static inline bool removeCSSTransitionFromMap(CSSTransition& transition, PropertyToTransitionMap& cssTransitionsByProperty)
57{
58 auto transitionIterator = cssTransitionsByProperty.find(transition.property());
59 if (transitionIterator == cssTransitionsByProperty.end() || transitionIterator->value != &transition)
60 return false;
61
62 cssTransitionsByProperty.remove(transitionIterator);
63 return true;
64}

77void Styleable::animationWasRemoved(WebAnimation& animation) const
78{
79 ensureAnimations().remove(&animation);
80
81 // Now, if we're dealing with a CSS Transition, we remove it from the m_elementToRunningCSSTransitionByCSSPropertyID map.
82 // We don't need to do this for CSS Animations because their timing can be set via CSS to end, which would cause this
83 // function to be called, but they should remain associated with their owning element until this is changed via a call
84 // to the JS API or changing the target element's animation-name property.
85 if (is<CSSTransition>(animation))
86 removeDeclarativeAnimationFromListsForOwningElement(animation);
87}

89static void removeCSSAnimationCreatedByMarkup(const Styleable& styleable, CSSAnimation& cssAnimation)
90{
91 styleable.animationsCreatedByMarkup().remove(&cssAnimation);
92
93 if (!styleable.hasKeyframeEffects())
94 return;
95
96 auto& keyframeEffectStack = styleable.ensureKeyframeEffectStack();
97 auto* cssAnimationList = keyframeEffectStack.cssAnimationList();
98 if (!cssAnimationList || cssAnimationList->isEmpty())
99 return;
100
101 auto& backingAnimation = cssAnimation.backingAnimation();
102 for (size_t i = 0; i < cssAnimationList->size(); ++i) {
103 if (cssAnimationList->animation(i) == backingAnimation) {
104 auto newAnimationList = cssAnimationList->copy();
105 newAnimationList->remove(i);
106 keyframeEffectStack.setCSSAnimationList(WTFMove(newAnimationList));
107 return;
108 }
109 }

125void Styleable::cancelDeclarativeAnimations() const
126{
127 if (auto* animations = this->animations()) {
128 for (auto& animation : *animations) {
129 if (is<DeclarativeAnimation>(animation)) {
130 if (is<CSSAnimation>(animation))
131 removeCSSAnimationCreatedByMarkup(*this, downcast<CSSAnimation>(*animation));
132 downcast<DeclarativeAnimation>(*animation).cancelFromStyle();
133 }
134 }
135 }

138static bool shouldConsiderAnimation(Element& element, const Animation& animation)
139{
140 if (!animation.isValidAnimation())
141 return false;
142
143 auto& name = animation.name().string;
144 if (name == "none" || name.isEmpty())
145 return false;
146
147 if (auto* styleScope = Style::Scope::forOrdinal(element, animation.nameStyleScopeOrdinal()))
148 return styleScope->resolver().isAnimationNameValid(name);
149
150 return false;
151}

153void Styleable::updateCSSAnimations(const RenderStyle* currentStyle, const RenderStyle& newStyle, const RenderStyle* parentElementStyle) const
154{
155 auto& keyframeEffectStack = ensureKeyframeEffectStack();
156
157 // In case this element is newly getting a "display: none" we need to cancel all of its animations and disregard new ones.
158 if (currentStyle && currentStyle->display() != DisplayType::None && newStyle.display() == DisplayType::None) {
159 for (auto& cssAnimation : animationsCreatedByMarkup())
160 cssAnimation->cancelFromStyle();
161 keyframeEffectStack.setCSSAnimationList(nullptr);
162 return;
163 }
164
165 auto* currentAnimationList = newStyle.animations();
166 auto* previousAnimationList = keyframeEffectStack.cssAnimationList();
167 if (previousAnimationList && !previousAnimationList->isEmpty() && newStyle.hasAnimations() && *(previousAnimationList) == *(newStyle.animations()))
168 return;
169
170 CSSAnimationCollection newAnimations;
171 auto& previousAnimations = animationsCreatedByMarkup();
172
173 // https://www.w3.org/TR/css-animations-1/#animations
174 // The same @keyframes rule name may be repeated within an animation-name. Changes to the animation-name update existing
175 // animations by iterating over the new list of animations from last to first, and, for each animation, finding the last
176 // matching animation in the list of existing animations. If a match is found, the existing animation is updated using the
177 // animation properties corresponding to its position in the new list of animations, whilst maintaining its current playback
178 // time as described above. The matching animation is removed from the existing list of animations such that it will not match
179 // twice. If a match is not found, a new animation is created. As a result, updating animation-name from ‘a’ to ‘a, a’ will
180 // cause the existing animation for ‘a’ to become the second animation in the list and a new animation will be created for the
181 // first item in the list.
182 if (currentAnimationList) {
183 for (size_t i = currentAnimationList->size(); i > 0; --i) {
184 auto& currentAnimation = currentAnimationList->animation(i - 1);
185 if (!shouldConsiderAnimation(this->element, currentAnimation))
186 continue;
187
188 bool foundMatchingAnimation = false;
189 for (auto& previousAnimation : previousAnimations) {
190 if (previousAnimation->animationName() == currentAnimation.name().string) {
191 // Timing properties or play state may have changed so we need to update the backing animation with
192 // the Animation found in the current style.
193 previousAnimation->setBackingAnimation(currentAnimation);
194 newAnimations.add(previousAnimation);
195 // Remove the matched animation from the list of previous animations so we may not match it again.
196 previousAnimations.remove(previousAnimation);
197 foundMatchingAnimation = true;
198 break;
199 }
200 }
201
202 if (!foundMatchingAnimation)
203 newAnimations.add(CSSAnimation::create(*this, currentAnimation, currentStyle, newStyle, parentElementStyle));
204 }
205 }
206
207 // Any animation found in previousAnimations but not found in newAnimations is not longer current and should be canceled.
208 for (auto& previousAnimation : previousAnimations) {
209 if (!newAnimations.contains(previousAnimation)) {
210 if (previousAnimation->owningElement())
211 previousAnimation->cancelFromStyle();
212 }
213 }
214
215 setAnimationsCreatedByMarkup(WTFMove(newAnimations));
216
217 keyframeEffectStack.setCSSAnimationList(currentAnimationList);
218}

220static KeyframeEffect* keyframeEffectForElementAndProperty(const Styleable& styleable, CSSPropertyID property)
221{
222 if (auto* keyframeEffectStack = styleable.keyframeEffectStack()) {
223 auto effects = keyframeEffectStack->sortedEffects();
224 for (const auto& effect : makeReversedRange(effects)) {
225 if (effect->animatesProperty(property))
226 return effect.get();
227 }
228 }
229
230 return nullptr;
231}

242static double transitionCombinedDuration(const Animation* transition)
243{
244 return std::max(0.0, transition->duration()) + transition->delay();

247static bool transitionMatchesProperty(const Animation& transition, CSSPropertyID property)
248{
249 auto mode = transition.property().mode;
250 if (mode == Animation::TransitionMode::None || mode == Animation::TransitionMode::UnknownProperty)
251 return false;
252 if (mode == Animation::TransitionMode::SingleProperty) {
253 auto transitionProperty = transition.property().id;
254 if (transitionProperty != property) {
255 for (auto longhand : shorthandForProperty(transitionProperty)) {
256 if (longhand == property)
257 return true;
258 }
259 return false;
260 }
261 }
262 return true;
263}

265static void compileTransitionPropertiesInStyle(const RenderStyle& style, HashSet<CSSPropertyID>& transitionProperties, bool& transitionPropertiesContainAll)
266{
267 if (transitionPropertiesContainAll)
268 return;
269
270 auto* transitions = style.transitions();
271 if (!transitions)
272 return;
273
274 for (size_t i = 0; i < transitions->size(); ++i) {
275 const auto& animation = transitions->animation(i);
276 auto mode = animation.property().mode;
277 if (mode == Animation::TransitionMode::SingleProperty) {
278 auto property = animation.property().id;
279 if (isShorthandCSSProperty(property)) {
280 for (auto longhand : shorthandForProperty(property))
281 transitionProperties.add(longhand);
282 } else if (property != CSSPropertyInvalid)
283 transitionProperties.add(property);
284 } else if (mode == Animation::TransitionMode::All) {
285 transitionPropertiesContainAll = true;
286 return;
287 }
288 }

291static void updateCSSTransitionsForStyleableAndProperty(const Styleable& styleable, CSSPropertyID property, const RenderStyle& currentStyle, const RenderStyle& newStyle, const MonotonicTime generationTime)
292{
293 auto* keyframeEffect = keyframeEffectForElementAndProperty(styleable, property);
294 auto* animation = keyframeEffect ? keyframeEffect->animation() : nullptr;
295
296 bool isDeclarative = false;
297 if (is<DeclarativeAnimation>(animation)) {
298 if (auto owningElement = downcast<DeclarativeAnimation>(*animation).owningElement())
299 isDeclarative = *owningElement == styleable;
300 }
301
302 if (animation && !isDeclarative)
303 return;
304
305 const Animation* matchingBackingAnimation = nullptr;
306 if (auto* transitions = newStyle.transitions()) {
307 for (size_t i = 0; i < transitions->size(); ++i) {
308 auto& backingAnimation = transitions->animation(i);
309 if (transitionMatchesProperty(backingAnimation, property))
310 matchingBackingAnimation = &backingAnimation;
311 }
312 }
313
314 // A CSS Transition might have completed since the last time animations were updated so we must
315 // update the running and completed transitions membership in that case.
316 if (is<CSSTransition>(animation) && matchingBackingAnimation && styleable.hasRunningTransitionForProperty(property) && animation->playState() == WebAnimation::PlayState::Finished) {
317 styleable.ensureCompletedTransitionsByProperty().set(property, styleable.ensureRunningTransitionsByProperty().take(property));
318 animation = nullptr;
319 }
320
321 // https://drafts.csswg.org/css-transitions-1/#before-change-style
322 // Define the before-change style as the computed values of all properties on the element as of the previous style change event, except with
323 // any styles derived from declarative animations such as CSS Transitions, CSS Animations, and SMIL Animations updated to the current time.
324 auto beforeChangeStyle = [&]() -> const RenderStyle {
325 if (animation && animation->isRelevant()) {
326 auto animatedStyle = RenderStyle::clone(currentStyle);
327 // If a transition has not yet started or started when animations were last updated, use the timeline time at its creation
328 // as its start time to ensure that it will produce a style with progress > 0.
329 bool shouldUseTimelineTimeAtCreation = is<CSSTransition>(animation) && (!animation->startTime() || *animation->startTime() == styleable.element.document().timeline().currentTime());
330 animation->resolve(animatedStyle, nullptr, shouldUseTimelineTimeAtCreation ? downcast<CSSTransition>(*animation).timelineTimeAtCreation() : std::nullopt);
331 return animatedStyle;
332 }
333
334 // If it exists, use the recorded RenderStyle for this element during a previous call to Style::TreeResolver::createAnimatedElementUpdate().
335 if (auto* lastStyleChangeEventStyle = styleable.lastStyleChangeEventStyle())
336 return RenderStyle::clone(*lastStyleChangeEventStyle);
337
338 // If we haven't computed styles from animations for this element, the before-change style is the previously resolved style for this element.
339 return RenderStyle::clone(currentStyle);
340 }();
341
342 // https://drafts.csswg.org/css-transitions-1/#after-change-style
343 // Likewise, define the after-change style as the computed values of all properties on the element based on the information known at the start
344 // of that style change event, but using the computed values of the animation-* properties from the before-change style, excluding any styles
345 // from CSS Transitions in the computation, and inheriting from the after-change style of the parent. Note that this means the after-change
346 // style does not differ from the before-change style due to newly created or canceled CSS Animations.
347 auto afterChangeStyle = [&]() -> const RenderStyle {
348 if (is<CSSAnimation>(animation) && animation->isRelevant()) {
349 auto animatedStyle = RenderStyle::clone(newStyle);
350 animation->resolve(animatedStyle, nullptr);
351 return animatedStyle;
352 }
353
354 return RenderStyle::clone(newStyle);
355 }();
356
357 if (!styleable.hasRunningTransitionForProperty(property)
358 && !CSSPropertyAnimation::propertiesEqual(property, beforeChangeStyle, afterChangeStyle)
359 && CSSPropertyAnimation::canPropertyBeInterpolated(property, beforeChangeStyle, afterChangeStyle)
360 && !propertyInStyleMatchesValueForTransitionInMap(property, afterChangeStyle, styleable.ensureCompletedTransitionsByProperty())
361 && matchingBackingAnimation && transitionCombinedDuration(matchingBackingAnimation) > 0) {
362 // 1. If all of the following are true:
363 // - the element does not have a running transition for the property,
364 // - the before-change style is different from and can be interpolated with the after-change style for that property,
365 // - the element does not have a completed transition for the property or the end value of the completed transition is different from the after-change style for the property,
366 // - there is a matching transition-property value, and
367 // - the combined duration is greater than 0s,
368
369 // then implementations must remove the completed transition (if present) from the set of completed transitions
370 styleable.ensureCompletedTransitionsByProperty().remove(property);
371
372 // and start a transition whose:
373 // - start time is the time of the style change event plus the matching transition delay,
374 // - end time is the start time plus the matching transition duration,
375 // - start value is the value of the transitioning property in the before-change style,
376 // - end value is the value of the transitioning property in the after-change style,
377 // - reversing-adjusted start value is the same as the start value, and
378 // - reversing shortening factor is 1.
379 auto delay = Seconds(matchingBackingAnimation->delay());
380 auto duration = Seconds(matchingBackingAnimation->duration());
381 auto& reversingAdjustedStartStyle = beforeChangeStyle;
382 auto reversingShorteningFactor = 1;
383 styleable.ensureRunningTransitionsByProperty().set(property, CSSTransition::create(styleable, property, generationTime, *matchingBackingAnimation, &beforeChangeStyle, afterChangeStyle, delay, duration, reversingAdjustedStartStyle, reversingShorteningFactor));
384 } else if (styleable.hasCompletedTransitionForProperty(property) && !propertyInStyleMatchesValueForTransitionInMap(property, afterChangeStyle, styleable.ensureCompletedTransitionsByProperty())) {
385 // 2. Otherwise, if the element has a completed transition for the property and the end value of the completed transition is different from
386 // the after-change style for the property, then implementations must remove the completed transition from the set of completed transitions.
387 styleable.ensureCompletedTransitionsByProperty().remove(property);
388 }
389
390 bool hasRunningTransition = styleable.hasRunningTransitionForProperty(property);
391 if ((hasRunningTransition || styleable.hasCompletedTransitionForProperty(property)) && !matchingBackingAnimation) {
392 // 3. If the element has a running transition or completed transition for the property, and there is not a matching transition-property
393 // value, then implementations must cancel the running transition or remove the completed transition from the set of completed transitions.
394 if (hasRunningTransition)
395 styleable.ensureRunningTransitionsByProperty().take(property)->cancel();
396 else
397 styleable.ensureCompletedTransitionsByProperty().remove(property);
398 }
399
400 if (matchingBackingAnimation && styleable.hasRunningTransitionForProperty(property) && !propertyInStyleMatchesValueForTransitionInMap(property, afterChangeStyle, styleable.ensureRunningTransitionsByProperty())) {
401 auto previouslyRunningTransition = styleable.ensureRunningTransitionsByProperty().take(property);
402 auto& previouslyRunningTransitionCurrentStyle = previouslyRunningTransition->currentStyle();
403 // 4. If the element has a running transition for the property, there is a matching transition-property value, and the end value of the running
404 // transition is not equal to the value of the property in the after-change style, then:
405 if (CSSPropertyAnimation::propertiesEqual(property, previouslyRunningTransitionCurrentStyle, afterChangeStyle) || !CSSPropertyAnimation::canPropertyBeInterpolated(property, currentStyle, afterChangeStyle)) {
406 // 1. If the current value of the property in the running transition is equal to the value of the property in the after-change style,
407 // or if these two values cannot be interpolated, then implementations must cancel the running transition.
408 previouslyRunningTransition->cancelFromStyle();
409 } else if (transitionCombinedDuration(matchingBackingAnimation) <= 0.0 || !CSSPropertyAnimation::canPropertyBeInterpolated(property, previouslyRunningTransitionCurrentStyle, afterChangeStyle)) {
410 // 2. Otherwise, if the combined duration is less than or equal to 0s, or if the current value of the property in the running transition
411 // cannot be interpolated with the value of the property in the after-change style, then implementations must cancel the running transition.
412 previouslyRunningTransition->cancelFromStyle();
413 } else if (CSSPropertyAnimation::propertiesEqual(property, previouslyRunningTransition->reversingAdjustedStartStyle(), afterChangeStyle)) {
414 // 3. Otherwise, if the reversing-adjusted start value of the running transition is the same as the value of the property in the after-change
415 // style (see the section on reversing of transitions for why these case exists), implementations must cancel the running transition
416 previouslyRunningTransition->cancelFromStyle();
417
418 // and start a new transition whose:
419 // - reversing-adjusted start value is the end value of the running transition,
420 // - reversing shortening factor is the absolute value, clamped to the range [0, 1], of the sum of:
421 // 1. the output of the timing function of the old transition at the time of the style change event, times the reversing shortening factor of the old transition
422 // 2. 1 minus the reversing shortening factor of the old transition.
423 // - start time is the time of the style change event plus:
424 // 1. if the matching transition delay is nonnegative, the matching transition delay, or
425 // 2. if the matching transition delay is negative, the product of the new transition’s reversing shortening factor and the matching transition delay,
426 // - end time is the start time plus the product of the matching transition duration and the new transition’s reversing shortening factor,
427 // - start value is the current value of the property in the running transition,
428 // - end value is the value of the property in the after-change style
429 auto& reversingAdjustedStartStyle = previouslyRunningTransition->targetStyle();
430 double transformedProgress = 1;
431 if (auto* effect = previouslyRunningTransition->effect()) {
432 if (auto computedTimingProgress = effect->getComputedTiming().progress)
433 transformedProgress = *computedTimingProgress;
434 }
435 auto reversingShorteningFactor = std::max(std::min(((transformedProgress * previouslyRunningTransition->reversingShorteningFactor()) + (1 - previouslyRunningTransition->reversingShorteningFactor())), 1.0), 0.0);
436 auto delay = matchingBackingAnimation->delay() < 0 ? Seconds(matchingBackingAnimation->delay()) * reversingShorteningFactor : Seconds(matchingBackingAnimation->delay());
437 auto duration = Seconds(matchingBackingAnimation->duration()) * reversingShorteningFactor;
438
439 styleable.ensureRunningTransitionsByProperty().set(property, CSSTransition::create(styleable, property, generationTime, *matchingBackingAnimation, &previouslyRunningTransitionCurrentStyle, afterChangeStyle, delay, duration, reversingAdjustedStartStyle, reversingShorteningFactor));
440 } else {
441 // 4. Otherwise, implementations must cancel the running transition
442 previouslyRunningTransition->cancelFromStyle();
443
444 // and start a new transition whose:
445 // - start time is the time of the style change event plus the matching transition delay,
446 // - end time is the start time plus the matching transition duration,
447 // - start value is the current value of the property in the running transition,
448 // - end value is the value of the property in the after-change style,
449 // - reversing-adjusted start value is the same as the start value, and
450 // - reversing shortening factor is 1.
451 auto delay = Seconds(matchingBackingAnimation->delay());
452 auto duration = Seconds(matchingBackingAnimation->duration());
453 auto& reversingAdjustedStartStyle = currentStyle;
454 auto reversingShorteningFactor = 1;
455 styleable.ensureRunningTransitionsByProperty().set(property, CSSTransition::create(styleable, property, generationTime, *matchingBackingAnimation, &previouslyRunningTransitionCurrentStyle, afterChangeStyle, delay, duration, reversingAdjustedStartStyle, reversingShorteningFactor));
456 }
457 }

460void Styleable::updateCSSTransitions(const RenderStyle& currentStyle, const RenderStyle& newStyle) const
461{
462 // In case this element is newly getting a "display: none" we need to cancel all of its transitions and disregard new ones.
463 if (currentStyle.hasTransitions() && currentStyle.display() != DisplayType::None && newStyle.display() == DisplayType::None) {
464 if (hasRunningTransitions()) {
465 auto runningTransitions = ensureRunningTransitionsByProperty();
466 for (const auto& cssTransitionsByCSSPropertyIDMapItem : runningTransitions)
467 cssTransitionsByCSSPropertyIDMapItem.value->cancelFromStyle();
468 }
469 return;
470 }
471
472 // Section 3 "Starting of transitions" from the CSS Transitions Level 1 specification.
473 // https://drafts.csswg.org/css-transitions-1/#starting
474
475 auto generationTime = MonotonicTime::now();
476
477 // First, let's compile the list of all CSS properties found in the current style and the after-change style.
478 bool transitionPropertiesContainAll = false;
479 HashSet<CSSPropertyID> transitionProperties;
480 compileTransitionPropertiesInStyle(currentStyle, transitionProperties, transitionPropertiesContainAll);
481 compileTransitionPropertiesInStyle(newStyle, transitionProperties, transitionPropertiesContainAll);
482
483 if (transitionPropertiesContainAll) {
484 auto numberOfProperties = CSSPropertyAnimation::getNumProperties();
485 for (int propertyIndex = 0; propertyIndex < numberOfProperties; ++propertyIndex) {
486 std::optional<bool> isShorthand;
487 auto property = CSSPropertyAnimation::getPropertyAtIndex(propertyIndex, isShorthand);
488 if (isShorthand && *isShorthand)
489 continue;
490 updateCSSTransitionsForStyleableAndProperty(*this, property, currentStyle, newStyle, generationTime);
491 }
492 return;
493 }
494
495 for (auto property : transitionProperties)
496 updateCSSTransitionsForStyleableAndProperty(*this, property, currentStyle, newStyle, generationTime);
497}
