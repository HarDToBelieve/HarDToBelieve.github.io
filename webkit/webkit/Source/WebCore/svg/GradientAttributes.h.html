20#pragma once
21
22#include "SVGGradientElement.h"
23#include "SVGUnitTypes.h"
24
25namespace WebCore {
26
27struct GradientAttributes {
28 GradientAttributes()
29 : m_spreadMethod(SVGSpreadMethodPad)
30 , m_gradientUnits(SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX)
31 , m_spreadMethodSet(false)
32 , m_gradientUnitsSet(false)
33 , m_gradientTransformSet(false)
34 {
35 }
36
37 SVGSpreadMethodType spreadMethod() const { return static_cast<SVGSpreadMethodType>(m_spreadMethod); }
38 SVGUnitTypes::SVGUnitType gradientUnits() const { return static_cast<SVGUnitTypes::SVGUnitType>(m_gradientUnits); }
39 AffineTransform gradientTransform() const { return m_gradientTransform; }
40 const Gradient::ColorStopVector& stops() const { return m_stops; }
41
42 void setSpreadMethod(SVGSpreadMethodType value)
43 {
44 m_spreadMethod = value;
45 m_spreadMethodSet = true;
46 }
47
48 void setGradientUnits(SVGUnitTypes::SVGUnitType unitType)
49 {
50 m_gradientUnits = unitType;
51 m_gradientUnitsSet = true;
52 }
53
54 void setGradientTransform(const AffineTransform& value)
55 {
56 m_gradientTransform = value;
57 m_gradientTransformSet = true;
58 }
59
60 void setStops(Gradient::ColorStopVector&& value)
61 {
62 m_stops = WTFMove(value);
63 }
64
65 bool hasSpreadMethod() const { return m_spreadMethodSet; }
66 bool hasGradientUnits() const { return m_gradientUnitsSet; }
67 bool hasGradientTransform() const { return m_gradientTransformSet; }
68 bool hasStops() const { return !m_stops.isEmpty(); }
69
70private:
71 // Properties
72 AffineTransform m_gradientTransform;
73 Gradient::ColorStopVector m_stops;
74
75 unsigned m_spreadMethod : 2;
76 unsigned m_gradientUnits : 2;
77
78 // Property states
79 unsigned m_spreadMethodSet : 1;
80 unsigned m_gradientUnitsSet : 1;
81 unsigned m_gradientTransformSet : 1;
82};
83
84struct SameSizeAsGradientAttributes {
85 AffineTransform a;
86 Gradient::ColorStopVector b;
87 unsigned c : 7;
88};
89
90COMPILE_ASSERT(sizeof(GradientAttributes) == sizeof(SameSizeAsGradientAttributes), GradientAttributes_size_guard);
