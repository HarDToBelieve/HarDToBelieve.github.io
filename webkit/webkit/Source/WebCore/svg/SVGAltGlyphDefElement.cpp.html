44bool SVGAltGlyphDefElement::hasValidGlyphElements(Vector<String>& glyphNames) const
45{
46 // Spec: http://www.w3.org/TR/SVG/text.html#AltGlyphDefElement
47 // An 'altGlyphDef' can contain either of the following:
48 //
49 // 1. In the simplest case, an 'altGlyphDef' contains one or more 'glyphRef' elements.
50 // Each 'glyphRef' element references a single glyph within a particular font.
51 // If all of the referenced glyphs are available, then these glyphs are rendered
52 // instead of the character(s) inside of the referencing 'altGlyph' element.
53 // If any of the referenced glyphs are unavailable, then the character(s) that are
54 // inside of the 'altGlyph' element are rendered as if there were not an 'altGlyph'
55 // element surrounding those characters.
56 //
57 // 2. In the more complex case, an 'altGlyphDef' contains one or more 'altGlyphItem' elements.
58 // Each 'altGlyphItem' represents a candidate set of substitute glyphs. Each 'altGlyphItem'
59 // contains one or more 'glyphRef' elements. Each 'glyphRef' element references a single
60 // glyph within a particular font. The first 'altGlyphItem' in which all referenced glyphs
61 // are available is chosen. The glyphs referenced from this 'altGlyphItem' are rendered
62 // instead of the character(s) that are inside of the referencing 'altGlyph' element.
63 // If none of the 'altGlyphItem' elements result in a successful match (i.e., none of the
64 // 'altGlyphItem' elements has all of its referenced glyphs available), then the character(s)
65 // that are inside of the 'altGlyph' element are rendered as if there were not an 'altGlyph'
66 // element surrounding those characters.
67 //
68 // The spec doesn't tell how to deal with the mixing of <glyphRef> and <altGlyItem>.
69 // However, we determine content model by the type of the first appearing element
70 // just like Opera 11 does. After the content model is determined we skip elements
71 // which don't comform to it. For example:
72 // a. <altGlyphDef>
73 // <glyphRef id="g1" />
74 // <altGlyphItem id="i1"> ... </altGlyphItem>
75 // <glyphRef id="g2" />
76 // </altGlyphDef>
77 //
78 // b. <altGlyphDef>
79 // <altGlyphItem id="i1"> ... </altGlyphItem>
80 // <altGlyphItem id="i2"> ... </altGlyphItem>
81 // <glyphRef id="g1" />
82 // <glyphRef id="g2" />
83 // </altGlyphDef>
84 // For a), the content model is 1), so we will use "g1" and "g2" if they are all valid
85 // and "i1" is skipped.
86 // For b), the content model is 2), so we will use <glyphRef> elements contained in
87 // "i1" if they are valid and "g1" and "g2" are skipped.
88
89 // These 2 variables are used to determine content model.
90 bool fountFirstGlyphRef = false;
91 bool foundFirstAltGlyphItem = false;
92
93 for (auto& child : childrenOfType<SVGElement>(*this)) {
94 if (!foundFirstAltGlyphItem && is<SVGGlyphRefElement>(child)) {
95 fountFirstGlyphRef = true;
96 String referredGlyphName;
97
98 if (downcast<SVGGlyphRefElement>(child).hasValidGlyphElement(referredGlyphName))
99 glyphNames.append(referredGlyphName);
100 else {
101 // As the spec says "If any of the referenced glyphs are unavailable,
102 // then the character(s) that are inside of the 'altGlyph' element are
103 // rendered as if there were not an 'altGlyph' element surrounding
104 // those characters.".
105 glyphNames.clear();
106 return false;
107 }
108 } else if (!fountFirstGlyphRef && is<SVGAltGlyphItemElement>(child)) {
109 foundFirstAltGlyphItem = true;
110
111 // As the spec says "The first 'altGlyphItem' in which all referenced glyphs
112 // are available is chosen."
113 if (downcast<SVGAltGlyphItemElement>(child).hasValidGlyphElements(glyphNames) && !glyphNames.isEmpty())
114 return true;
115 }
116 }
117 return !glyphNames.isEmpty();
118}
