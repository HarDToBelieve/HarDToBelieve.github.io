44bool SVGAltGlyphItemElement::hasValidGlyphElements(Vector<String>& glyphNames) const
45{
46 // Spec: http://www.w3.org/TR/SVG/text.html#AltGlyphItemElement
47 // The ‘altGlyphItem’ element defines a candidate set of possible glyph substitutions.
48 // The first ‘altGlyphItem’ element whose referenced glyphs are all available is chosen.
49 // Its glyphs are rendered instead of the character(s) that are inside of the referencing
50 // ‘altGlyph’ element.
51 //
52 // Here we fill glyphNames and return true only if all referenced glyphs are valid and
53 // there is at least one glyph.
54
55 for (auto& glyphRef : childrenOfType<SVGGlyphRefElement>(*this)) {
56 String referredGlyphName;
57 if (glyphRef.hasValidGlyphElement(referredGlyphName))
58 glyphNames.append(referredGlyphName);
59 else {
60 glyphNames.clear();
61 return false;
62 }
63 }
64 return !glyphNames.isEmpty();
65}
