59bool SVGAnimateMotionElement::hasValidAttributeType() const
60{
61 auto targetElement = makeRefPtr(this->targetElement());
62 if (!targetElement)
63 return false;
64
65 // We don't have a special attribute name to verify the animation type. Check the element name instead.
66 if (!targetElement->isSVGGraphicsElement())
67 return false;
68 // Spec: SVG 1.1 section 19.2.15
69 // FIXME: svgTag is missing. Needs to be checked, if transforming <svg> could cause problems.
70 if (targetElement->hasTagName(gTag)
71 || targetElement->hasTagName(defsTag)
72 || targetElement->hasTagName(useTag)
73 || is<SVGImageElement>(*targetElement)
74 || targetElement->hasTagName(switchTag)
75 || targetElement->hasTagName(pathTag)
76 || targetElement->hasTagName(rectTag)
77 || targetElement->hasTagName(circleTag)
78 || targetElement->hasTagName(ellipseTag)
79 || targetElement->hasTagName(lineTag)
80 || targetElement->hasTagName(polylineTag)
81 || targetElement->hasTagName(polygonTag)
82 || targetElement->hasTagName(textTag)
83 || targetElement->hasTagName(clipPathTag)
84 || targetElement->hasTagName(maskTag)
85 || targetElement->hasTagName(SVGNames::aTag)
86 || targetElement->hasTagName(foreignObjectTag)
87 )
88 return true;
89 return false;
90}

98void SVGAnimateMotionElement::parseAttribute(const QualifiedName& name, const AtomString& value)
99{
100 if (name == SVGNames::pathAttr) {
101 m_path = buildPathFromString(value);
102 updateAnimationPath();
103 return;
104 }
105
106 SVGAnimationElement::parseAttribute(name, value);
107}

109SVGAnimateMotionElement::RotateMode SVGAnimateMotionElement::rotateMode() const
110{
111 static MainThreadNeverDestroyed<const AtomString> autoVal("auto", AtomString::ConstructFromLiteral);
112 static MainThreadNeverDestroyed<const AtomString> autoReverse("auto-reverse", AtomString::ConstructFromLiteral);
113 const AtomString& rotate = getAttribute(SVGNames::rotateAttr);
114 if (rotate == autoVal)
115 return RotateAuto;
116 if (rotate == autoReverse)
117 return RotateAutoReverse;
118 return RotateAngle;
119}

121void SVGAnimateMotionElement::updateAnimationPath()
122{
123 m_animationPath = Path();
124 bool foundMPath = false;
125
126 for (auto& mPath : childrenOfType<SVGMPathElement>(*this)) {
127 auto pathElement = mPath.pathElement();
128 if (pathElement) {
129 m_animationPath = pathFromGraphicsElement(pathElement.get());
130 foundMPath = true;
131 break;
132 }
133 }
134
135 if (!foundMPath && hasAttributeWithoutSynchronization(SVGNames::pathAttr))
136 m_animationPath = m_path;
137
138 updateAnimationMode();
139}

141void SVGAnimateMotionElement::startAnimation()
142{
143 if (!hasValidAttributeType())
144 return;
145 auto targetElement = makeRefPtr(this->targetElement());
146 if (!targetElement)
147 return;
148 if (AffineTransform* transform = targetElement->supplementalTransform())
149 transform->makeIdentity();
150}

152void SVGAnimateMotionElement::stopAnimation(SVGElement* targetElement)
153{
154 if (!targetElement)
155 return;
156 if (AffineTransform* transform = targetElement->supplementalTransform())
157 transform->makeIdentity();
158 applyResultsToTarget();
159}

175bool SVGAnimateMotionElement::calculateFromAndByValues(const String& fromString, const String& byString)
176{
177 m_toPointAtEndOfDuration = std::nullopt;
178 if (animationMode() == AnimationMode::By && !isAdditive())
179 return false;
180 m_fromPoint = parsePoint(fromString).value_or(FloatPoint { });
181 auto byPoint = parsePoint(byString).value_or(FloatPoint { });
182 m_toPoint = FloatPoint(m_fromPoint.x() + byPoint.x(), m_fromPoint.y() + byPoint.y());
183 return true;
184}

186void SVGAnimateMotionElement::buildTransformForProgress(AffineTransform* transform, float percentage)
187{
188 ASSERT(!m_animationPath.isEmpty());
189
190 float positionOnPath = m_animationPath.length() * percentage;
191 auto traversalState(m_animationPath.traversalStateAtLength(positionOnPath));
192 if (!traversalState.success())
193 return;
194
195 FloatPoint position = traversalState.current();
196 float angle = traversalState.normalAngle();
197
198 transform->translate(position);
199 RotateMode rotateMode = this->rotateMode();
200 if (rotateMode != RotateAuto && rotateMode != RotateAutoReverse)
201 return;
202 if (rotateMode == RotateAutoReverse)
203 angle += 180;
204 transform->rotate(angle);
205}

207void SVGAnimateMotionElement::calculateAnimatedValue(float percentage, unsigned repeatCount)
208{
209 auto targetElement = makeRefPtr(this->targetElement());
210 if (!targetElement)
211 return;
212
213 AffineTransform* transform = targetElement->supplementalTransform();
214 if (!transform)
215 return;
216
217 if (!isAdditive())
218 transform->makeIdentity();
219
220 if (animationMode() != AnimationMode::Path) {
221 FloatPoint toPointAtEndOfDuration = m_toPoint;
222 if (isAccumulated() && repeatCount && m_toPointAtEndOfDuration)
223 toPointAtEndOfDuration = *m_toPointAtEndOfDuration;
224
225 float animatedX = 0;
226 animateAdditiveNumber(percentage, repeatCount, m_fromPoint.x(), m_toPoint.x(), toPointAtEndOfDuration.x(), animatedX);
227
228 float animatedY = 0;
229 animateAdditiveNumber(percentage, repeatCount, m_fromPoint.y(), m_toPoint.y(), toPointAtEndOfDuration.y(), animatedY);
230
231 transform->translate(animatedX, animatedY);
232 return;
233 }
234
235 buildTransformForProgress(transform, percentage);
236
237 // Handle accumulate="sum".
238 if (isAccumulated() && repeatCount) {
239 for (unsigned i = 0; i < repeatCount; ++i)
240 buildTransformForProgress(transform, 1);
241 }
242}

244void SVGAnimateMotionElement::applyResultsToTarget()
245{
246 // We accumulate to the target element transform list so there is not much to do here.
247 auto targetElement = makeRefPtr(this->targetElement());
248 if (!targetElement)
249 return;
250
251 if (RenderElement* renderer = targetElement->renderer()) {
252 renderer->setNeedsTransformUpdate();
253 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
254 }
255
256 AffineTransform* targetSupplementalTransform = targetElement->supplementalTransform();
257 if (!targetSupplementalTransform)
258 return;
259
260 // ...except in case where we have additional instances in <use> trees.
261 for (auto& instance : copyToVectorOf<Ref<SVGElement>>(targetElement->instances())) {
262 AffineTransform* transform = instance->supplementalTransform();
263 if (!transform || *transform == *targetSupplementalTransform)
264 continue;
265 *transform = *targetSupplementalTransform;
266 if (RenderElement* renderer = instance->renderer()) {
267 renderer->setNeedsTransformUpdate();
268 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
269 }
270 }

273std::optional<float> SVGAnimateMotionElement::calculateDistance(const String& fromString, const String& toString)
274{
275 auto from = parsePoint(fromString);
276 if (!from)
277 return { };
278 auto to = parsePoint(toString);
279 if (!to)
280 return { };
281 auto diff = *to - *from;
282 return std::hypot(diff.width(), diff.height());
283}
