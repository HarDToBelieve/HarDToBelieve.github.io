58static void parseKeyTimes(const String& parse, Vector<float>& result, bool verifyOrder)
59{
60 result.clear();
61 bool isFirst = true;
62 for (StringView timeString : StringView(parse).split(';')) {
63 bool ok;
64 float time = timeString.toFloat(ok);
65 if (!ok || time < 0 || time > 1)
66 goto fail;
67 if (verifyOrder) {
68 if (isFirst) {
69 if (time)
70 goto fail;
71 isFirst = false;
72 } else if (time < result.last())
73 goto fail;
74 }
75 result.append(time);
76 }
77 return;
78fail:
79 result.clear();
80}

82static std::optional<Vector<UnitBezier>> parseKeySplines(StringView string)
83{
84 if (string.isEmpty())
85 return std::nullopt;
86
87 return readCharactersForParsing(string, [&](auto buffer) -> std::optional<Vector<UnitBezier>> {
88 skipOptionalSVGSpaces(buffer);
89
90 Vector<UnitBezier> result;
91
92 bool delimParsed = false;
93 while (buffer.hasCharactersRemaining()) {
94 delimParsed = false;
95 auto posA = parseNumber(buffer);
96 if (!posA || !isInRange<float>(*posA, 0, 1))
97 return std::nullopt;
98
99 auto posB = parseNumber(buffer);
100 if (!posB || !isInRange<float>(*posB, 0, 1))
101 return std::nullopt;
102
103 auto posC = parseNumber(buffer);
104 if (!posC || !isInRange<float>(*posC, 0, 1))
105 return std::nullopt;
106
107 auto posD = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
108 if (!posD || !isInRange<float>(*posD, 0, 1))
109 return std::nullopt;
110
111 skipOptionalSVGSpaces(buffer);
112
113 if (skipExactly(buffer, ';'))
114 delimParsed = true;
115
116 skipOptionalSVGSpaces(buffer);
117
118 result.append(UnitBezier { *posA, *posB, *posC, *posD });
119 }
120
121 if (!(buffer.atEnd() && !delimParsed))
122 return std::nullopt;
123
124 return result;
125 });
126}

149void SVGAnimationElement::parseAttribute(const QualifiedName& name, const AtomString& value)
150{
151 if (name == SVGNames::valuesAttr) {
152 // Per the SMIL specification, leading and trailing white space,
153 // and white space before and after semicolon separators, is allowed and will be ignored.
154 // http://www.w3.org/TR/SVG11/animate.html#ValuesAttribute
155 m_values = value.string().split(';');
156 for (auto& value : m_values)
157 value = value.stripWhiteSpace();
158
159 updateAnimationMode();
160 return;
161 }
162
163 if (name == SVGNames::keyTimesAttr) {
164 parseKeyTimes(value, m_keyTimesFromAttribute, true);
165 return;
166 }
167
168 if (name == SVGNames::keyPointsAttr) {
169 if (hasTagName(SVGNames::animateMotionTag)) {
170 // This is specified to be an animateMotion attribute only but it is simpler to put it here
171 // where the other timing calculatations are.
172 parseKeyTimes(value, m_keyPoints, false);
173 }
174 return;
175 }
176
177 if (name == SVGNames::keySplinesAttr) {
178 if (auto keySplines = parseKeySplines(value))
179 m_keySplines = WTFMove(*keySplines);
180 else
181 m_keySplines.clear();
182 return;
183 }
184
185 if (name == SVGNames::attributeTypeAttr) {
186 setAttributeType(value);
187 return;
188 }
189
190 if (name == SVGNames::calcModeAttr) {
191 setCalcMode(value);
192 return;
193 }
194
195 if (name == SVGNames::fromAttr || name == SVGNames::toAttr || name == SVGNames::byAttr) {
196 updateAnimationMode();
197 return;
198 }
199
200 SVGSMILElement::parseAttribute(name, value);
201 SVGTests::parseAttribute(name, value);
202}

262void SVGAnimationElement::updateAnimationMode()
263{
264 // http://www.w3.org/TR/2001/REC-smil-animation-20010904/#AnimFuncValues
265 if (hasAttribute(SVGNames::valuesAttr))
266 setAnimationMode(AnimationMode::Values);
267 else if (!toValue().isEmpty())
268 setAnimationMode(fromValue().isEmpty() ? AnimationMode::To : AnimationMode::FromTo);
269 else if (!byValue().isEmpty())
270 setAnimationMode(fromValue().isEmpty() ? AnimationMode::By : AnimationMode::FromBy);
271 else
272 setAnimationMode(AnimationMode::None);
273}

275void SVGAnimationElement::setCalcMode(const AtomString& calcMode)
276{
277 static MainThreadNeverDestroyed<const AtomString> discrete("discrete", AtomString::ConstructFromLiteral);
278 static MainThreadNeverDestroyed<const AtomString> linear("linear", AtomString::ConstructFromLiteral);
279 static MainThreadNeverDestroyed<const AtomString> paced("paced", AtomString::ConstructFromLiteral);
280 static MainThreadNeverDestroyed<const AtomString> spline("spline", AtomString::ConstructFromLiteral);
281 if (calcMode == discrete)
282 setCalcMode(CalcMode::Discrete);
283 else if (calcMode == linear)
284 setCalcMode(CalcMode::Linear);
285 else if (calcMode == paced)
286 setCalcMode(CalcMode::Paced);
287 else if (calcMode == spline)
288 setCalcMode(CalcMode::Spline);
289 else
290 setCalcMode(hasTagName(SVGNames::animateMotionTag) ? CalcMode::Paced : CalcMode::Linear);
291}

339void SVGAnimationElement::calculateKeyTimesForCalcModePaced()
340{
341 ASSERT(calcMode() == CalcMode::Paced);
342 ASSERT(animationMode() == AnimationMode::Values);
343
344 unsigned valuesCount = m_values.size();
345 ASSERT(valuesCount >= 1);
346 if (valuesCount == 1)
347 return;
348
349 m_keyTimesForPaced.clear();
350
351 Vector<float> keyTimesForPaced;
352 float totalDistance = 0;
353 keyTimesForPaced.append(0);
354 for (unsigned n = 0; n < valuesCount - 1; ++n) {
355 // Distance in any units
356 auto distance = calculateDistance(m_values[n], m_values[n + 1]);
357 if (!distance)
358 return;
359 totalDistance += *distance;
360 keyTimesForPaced.append(*distance);
361 }
362 if (!totalDistance)
363 return;
364
365 // Normalize.
366 for (unsigned n = 1; n < keyTimesForPaced.size() - 1; ++n)
367 keyTimesForPaced[n] = keyTimesForPaced[n - 1] + keyTimesForPaced[n] / totalDistance;
368 keyTimesForPaced[keyTimesForPaced.size() - 1] = 1;
369
370 m_keyTimesForPaced = WTFMove(keyTimesForPaced);
371}

380unsigned SVGAnimationElement::calculateKeyTimesIndex(float percent) const
381{
382 const auto& keyTimes = this->keyTimes();
383 unsigned index;
384 unsigned keyTimesCount = keyTimes.size();
385 // Compare index + 1 to keyTimesCount because the last keyTimes entry is
386 // required to be 1, and percent can never exceed 1; i.e., the second last
387 // keyTimes entry defines the beginning of the final interval
388 for (index = 1; index + 1 < keyTimesCount; ++index) {
389 if (keyTimes[index] > percent)
390 break;
391 }
392 return --index;
393}

395float SVGAnimationElement::calculatePercentForSpline(float percent, unsigned splineIndex) const
396{
397 ASSERT(calcMode() == CalcMode::Spline);
398 ASSERT_WITH_SECURITY_IMPLICATION(splineIndex < m_keySplines.size());
399 UnitBezier bezier = m_keySplines[splineIndex];
400 SMILTime duration = simpleDuration();
401 if (!duration.isFinite())
402 duration = 100.0;
403 return narrowPrecisionToFloat(bezier.solve(percent, solveEpsilon(duration.value())));
404}

406float SVGAnimationElement::calculatePercentFromKeyPoints(float percent) const
407{
408 const auto& keyTimes = this->keyTimes();
409
410 ASSERT(!m_keyPoints.isEmpty());
411 ASSERT(calcMode() != CalcMode::Paced);
412 ASSERT(keyTimes.size() > 1);
413 ASSERT(m_keyPoints.size() == keyTimes.size());
414
415 if (percent == 1)
416 return m_keyPoints[m_keyPoints.size() - 1];
417
418 unsigned index = calculateKeyTimesIndex(percent);
419 float fromPercent = keyTimes[index];
420 float toPercent = keyTimes[index + 1];
421 float fromKeyPoint = m_keyPoints[index];
422 float toKeyPoint = m_keyPoints[index + 1];
423
424 if (calcMode() == CalcMode::Discrete)
425 return fromKeyPoint;
426
427 float keyPointPercent = (percent - fromPercent) / (toPercent - fromPercent);
428
429 if (calcMode() == CalcMode::Spline) {
430 ASSERT(m_keySplines.size() == m_keyPoints.size() - 1);
431 keyPointPercent = calculatePercentForSpline(keyPointPercent, index);
432 }
433 return (toKeyPoint - fromKeyPoint) * keyPointPercent + fromKeyPoint;
434}

436float SVGAnimationElement::calculatePercentForFromTo(float percent) const
437{
438 const auto& keyTimes = this->keyTimes();
439 if (calcMode() == CalcMode::Discrete && keyTimes.size() == 2)
440 return percent > keyTimes[1] ? 1 : 0;
441
442 return percent;
443}

445void SVGAnimationElement::currentValuesFromKeyPoints(float percent, float& effectivePercent, String& from, String& to) const
446{
447 ASSERT(!m_keyPoints.isEmpty());
448 ASSERT(m_keyPoints.size() == keyTimes().size());
449 ASSERT(calcMode() != CalcMode::Paced);
450 effectivePercent = calculatePercentFromKeyPoints(percent);
451 unsigned index = effectivePercent == 1 ? m_values.size() - 2 : static_cast<unsigned>(effectivePercent * (m_values.size() - 1));
452 from = m_values[index];
453 to = m_values[index + 1];
454}

456void SVGAnimationElement::currentValuesForValuesAnimation(float percent, float& effectivePercent, String& from, String& to)
457{
458 unsigned valuesCount = m_values.size();
459 ASSERT(m_animationValid);
460 ASSERT(valuesCount >= 1);
461
462 if (percent == 1 || valuesCount == 1) {
463 from = m_values[valuesCount - 1];
464 to = m_values[valuesCount - 1];
465 effectivePercent = 1;
466 return;
467 }
468
469 CalcMode calcMode = this->calcMode();
470 if (is<SVGAnimateElement>(*this) || is<SVGAnimateColorElement>(*this)) {
471 ASSERT(targetElement());
472 if (downcast<SVGAnimateElementBase>(*this).isDiscreteAnimator())
473 calcMode = CalcMode::Discrete;
474 }
475 if (!m_keyPoints.isEmpty() && calcMode != CalcMode::Paced)
476 return currentValuesFromKeyPoints(percent, effectivePercent, from, to);
477
478 const auto& keyTimes = this->keyTimes();
479 unsigned keyTimesCount = keyTimes.size();
480 ASSERT(!keyTimesCount || valuesCount == keyTimesCount);
481 ASSERT(!keyTimesCount || (keyTimesCount > 1 && !keyTimes[0]));
482
483 unsigned index = calculateKeyTimesIndex(percent);
484 if (calcMode == CalcMode::Discrete) {
485 if (!keyTimesCount)
486 index = static_cast<unsigned>(percent * valuesCount);
487 from = m_values[index];
488 to = m_values[index];
489 effectivePercent = 0;
490 return;
491 }
492
493 float fromPercent;
494 float toPercent;
495 if (keyTimesCount) {
496 fromPercent = keyTimes[index];
497 toPercent = keyTimes[index + 1];
498 } else {
499 index = static_cast<unsigned>(floorf(percent * (valuesCount - 1)));
500 fromPercent = static_cast<float>(index) / (valuesCount - 1);
501 toPercent = static_cast<float>(index + 1) / (valuesCount - 1);
502 }
503
504 if (index == valuesCount - 1)
505 --index;
506 from = m_values[index];
507 to = m_values[index + 1];
508 ASSERT(toPercent > fromPercent);
509 effectivePercent = (percent - fromPercent) / (toPercent - fromPercent);
510
511 if (calcMode == CalcMode::Spline) {
512 ASSERT(m_keySplines.size() == m_values.size() - 1);
513 effectivePercent = calculatePercentForSpline(effectivePercent, index);
514 }
515}

517void SVGAnimationElement::startedActiveInterval()
518{
519 m_animationValid = false;
520
521 if (!hasValidAttributeType())
522 return;
523
524 const auto& keyTimes = this->keyTimes();
525
526 // These validations are appropriate for all animation modes.
527 if (hasAttributeWithoutSynchronization(SVGNames::keyPointsAttr) && m_keyPoints.size() != keyTimes.size())
528 return;
529
530 AnimationMode animationMode = this->animationMode();
531 CalcMode calcMode = this->calcMode();
532 if (calcMode == CalcMode::Spline) {
533 unsigned splinesCount = m_keySplines.size();
534 if (!splinesCount
535 || (hasAttributeWithoutSynchronization(SVGNames::keyPointsAttr) && m_keyPoints.size() - 1 != splinesCount)
536 || (animationMode == AnimationMode::Values && m_values.size() - 1 != splinesCount)
537 || (hasAttributeWithoutSynchronization(SVGNames::keyTimesAttr) && keyTimes.size() - 1 != splinesCount))
538 return;
539 }
540
541 String from = fromValue();
542 String to = toValue();
543 String by = byValue();
544 if (animationMode == AnimationMode::None)
545 return;
546 if ((animationMode == AnimationMode::FromTo || animationMode == AnimationMode::FromBy || animationMode == AnimationMode::To || animationMode == AnimationMode::By)
547 && (hasAttributeWithoutSynchronization(SVGNames::keyPointsAttr) && hasAttributeWithoutSynchronization(SVGNames::keyTimesAttr) && (keyTimes.size() < 2 || keyTimes.size() != m_keyPoints.size())))
548 return;
549 if (animationMode == AnimationMode::FromTo)
550 m_animationValid = calculateFromAndToValues(from, to);
551 else if (animationMode == AnimationMode::To) {
552 // For to-animations the from value is the current accumulated value from lower priority animations.
553 // The value is not static and is determined during the animation.
554 m_animationValid = calculateFromAndToValues(emptyString(), to);
555 } else if (animationMode == AnimationMode::FromBy)
556 m_animationValid = calculateFromAndByValues(from, by);
557 else if (animationMode == AnimationMode::By)
558 m_animationValid = calculateFromAndByValues(emptyString(), by);
559 else if (animationMode == AnimationMode::Values) {
560 m_animationValid = m_values.size() >= 1
561 && (calcMode == CalcMode::Paced || !hasAttributeWithoutSynchronization(SVGNames::keyTimesAttr) || hasAttributeWithoutSynchronization(SVGNames::keyPointsAttr) || (m_values.size() == keyTimes.size()))
562 && (calcMode == CalcMode::Discrete || !keyTimes.size() || keyTimes.last() == 1)
563 && (calcMode != CalcMode::Spline || ((m_keySplines.size() && (m_keySplines.size() == m_values.size() - 1)) || m_keySplines.size() == m_keyPoints.size() - 1))
564 && (!hasAttributeWithoutSynchronization(SVGNames::keyPointsAttr) || (keyTimes.size() > 1 && keyTimes.size() == m_keyPoints.size()));
565 if (m_animationValid)
566 m_animationValid = calculateToAtEndOfDurationValue(m_values.last());
567 if (calcMode == CalcMode::Paced && m_animationValid)
568 calculateKeyTimesForCalcModePaced();
569 } else if (animationMode == AnimationMode::Path)
570 m_animationValid = calcMode == CalcMode::Paced || !hasAttributeWithoutSynchronization(SVGNames::keyPointsAttr) || (keyTimes.size() > 1 && keyTimes.size() == m_keyPoints.size());
571}

573void SVGAnimationElement::updateAnimation(float percent, unsigned repeatCount)
574{
575 if (!m_animationValid)
576 return;
577
578 float effectivePercent;
579 CalcMode calcMode = this->calcMode();
580 AnimationMode animationMode = this->animationMode();
581 if (animationMode == AnimationMode::Values) {
582 String from;
583 String to;
584 currentValuesForValuesAnimation(percent, effectivePercent, from, to);
585 if (from != m_lastValuesAnimationFrom || to != m_lastValuesAnimationTo) {
586 m_animationValid = calculateFromAndToValues(from, to);
587 if (!m_animationValid)
588 return;
589 m_lastValuesAnimationFrom = from;
590 m_lastValuesAnimationTo = to;
591 }
592 } else if (!m_keyPoints.isEmpty() && calcMode != CalcMode::Paced)
593 effectivePercent = calculatePercentFromKeyPoints(percent);
594 else if (m_keyPoints.isEmpty() && calcMode == CalcMode::Spline && keyTimes().size() > 1)
595 effectivePercent = calculatePercentForSpline(percent, calculateKeyTimesIndex(percent));
596 else if (animationMode == AnimationMode::FromTo || animationMode == AnimationMode::To)
597 effectivePercent = calculatePercentForFromTo(percent);
598 else
599 effectivePercent = percent;
600
601 calculateAnimatedValue(effectivePercent, repeatCount);
602}
