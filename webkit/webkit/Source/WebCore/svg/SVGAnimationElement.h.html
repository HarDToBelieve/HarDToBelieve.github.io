62 void animateAdditiveNumber(float percentage, unsigned repeatCount, float fromNumber, float toNumber, float toAtEndOfDurationNumber, float& animatedNumber)
63 {
64 float number;
65 if (calcMode() == CalcMode::Discrete)
66 number = percentage < 0.5 ? fromNumber : toNumber;
67 else
68 number = (toNumber - fromNumber) * percentage + fromNumber;
69
70 if (isAccumulated() && repeatCount)
71 number += toAtEndOfDurationNumber * repeatCount;
72
73 if (isAdditive() && animationMode() != AnimationMode::To)
74 animatedNumber += number;
75 else
76 animatedNumber = number;
77 }
