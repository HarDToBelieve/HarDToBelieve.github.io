107void SVGDocumentExtensions::startAnimations()
108{
109 // FIXME: Eventually every "Time Container" will need a way to latch on to some global timer
110 // starting animations for a document will do this "latching"
111 // FIXME: We hold a ref pointers to prevent a shadow tree from getting removed out from underneath us.
112 // In the future we should refactor the use-element to avoid this. See https://webkit.org/b/53704
113 auto timeContainers = copyToVectorOf<Ref<SVGSVGElement>>(m_timeContainers);
114 for (auto& element : timeContainers)
115 element->timeContainer().begin();
116}

132void SVGDocumentExtensions::dispatchLoadEventToOutermostSVGElements()
133{

139 }

177bool SVGDocumentExtensions::isElementWithPendingResources(Element& element) const
178{
179 // This algorithm takes time proportional to the number of pending resources and need not.
180 // If performance becomes an issue we can keep a counted set of elements and answer the question efficiently.
181 return WTF::anyOf(m_pendingResources.values(), [&] (auto& elements) {
182 return elements.contains(element);
183 });
184}

186bool SVGDocumentExtensions::isPendingResource(Element& element, const AtomString& id) const
187{
188 if (id.isEmpty())
189 return false;
190
191 auto it = m_pendingResources.find(id);
192 if (it == m_pendingResources.end())
193 return false;
194
195 return it->value.contains(element);
196}

204void SVGDocumentExtensions::removeElementFromPendingResources(Element& element)
205{
206 // Remove the element from pending resources.
207 if (!m_pendingResources.isEmpty() && element.hasPendingResources()) {
208 Vector<AtomString> toBeRemoved;
209 for (auto& resource : m_pendingResources) {
210 auto& elements = resource.value;
211 elements.remove(element);
212 if (elements.computesEmpty())
213 toBeRemoved.append(resource.key);
214 }
215
216 clearHasPendingResourcesIfPossible(element);
217
218 // We use the removePendingResource function here because it deals with set lifetime correctly.
219 for (auto& resource : toBeRemoved)
220 removePendingResource(resource);
221 }
222
223 // Remove the element from pending resources that were scheduled for removal.
224 if (!m_pendingResourcesForRemoval.isEmpty()) {
225 Vector<AtomString> toBeRemoved;
226 for (auto& resource : m_pendingResourcesForRemoval) {
227 auto& elements = resource.value;
228 elements.remove(element);
229 if (elements.computesEmpty())
230 toBeRemoved.append(resource.key);
231 }
232
233 // We use the removePendingResourceForRemoval function here because it deals with set lifetime correctly.
234 for (auto& resource : toBeRemoved)
235 m_pendingResourcesForRemoval.remove(resource);
236 }
237}

239void SVGDocumentExtensions::markPendingResourcesForRemoval(const AtomString& id)
240{
241 if (id.isEmpty())
242 return;
243
244 ASSERT(!m_pendingResourcesForRemoval.contains(id));
245
246 auto existing = m_pendingResources.take(id);
247 if (!existing.computesEmpty())
248 m_pendingResourcesForRemoval.add(id, WTFMove(existing));
249}

251RefPtr<Element> SVGDocumentExtensions::takeElementFromPendingResourcesForRemovalMap(const AtomString& id)
252{
253 if (id.isEmpty())
254 return nullptr;
255
256 auto it = m_pendingResourcesForRemoval.find(id);
257 if (it == m_pendingResourcesForRemoval.end())
258 return nullptr;
259
260 auto& resourceSet = it->value;
261 auto firstElement = makeRefPtr(resourceSet.begin().get());
262 if (!firstElement)
263 return nullptr;
264
265 resourceSet.remove(*firstElement);
266
267 if (resourceSet.computesEmpty())
268 m_pendingResourcesForRemoval.remove(id);
269
270 return firstElement;
271}

283void SVGDocumentExtensions::rebuildElements()
284{
285 auto shadowRebuildElements = std::exchange(m_rebuildElements, { });
286 for (auto& element : shadowRebuildElements)
287 element->svgAttributeChanged(SVGNames::hrefAttr);
288}
