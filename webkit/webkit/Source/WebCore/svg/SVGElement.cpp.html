169SVGElement::~SVGElement()
170{
171 if (m_svgRareData) {
172 RELEASE_ASSERT(m_svgRareData->referencingElements().computesEmpty());
173 for (SVGElement& instance : copyToVectorOf<Ref<SVGElement>>(instances()))
174 instance.m_svgRareData->setCorrespondingElement(nullptr);

177 }
178 document().accessSVGExtensions().removeElementToRebuild(*this);

198bool SVGElement::isOutermostSVGSVGElement() const
199{
200 if (!is<SVGSVGElement>(*this))
201 return false;
202
203 // Element may not be in the document, pretend we're outermost for viewport(), getCTM(), etc.
204 if (!parentNode())
205 return true;
206
207 // We act like an outermost SVG element, if we're a direct child of a <foreignObject> element.
208 if (is<SVGForeignObjectElement>(*parentNode()))
209 return true;
210
211 // If we're inside the shadow tree of a <use> element, we're always an inner <svg> element.
212 if (isInShadowTree() && is<SVGUseElement>(shadowHost()))
213 return false;
214
215 // This is true whenever this is the outermost SVG, even if there are HTML elements outside it
216 return !is<SVGElement>(*parentNode());
217}

219void SVGElement::reportAttributeParsingError(SVGParsingError error, const QualifiedName& name, const AtomString& value)
220{
221 if (error == NoError)
222 return;
223
224 String errorString = "<" + tagName() + "> attribute " + name.toString() + "=\"" + value + "\"";
225 SVGDocumentExtensions& extensions = document().accessSVGExtensions();
226
227 if (error == NegativeValueForbiddenError) {
228 extensions.reportError("Invalid negative value for " + errorString);
229 return;
230 }
231
232 if (error == ParsingAttributeFailedError) {
233 extensions.reportError("Invalid value for " + errorString);
234 return;
235 }
236
237 ASSERT_NOT_REACHED();
238}

240void SVGElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
241{
242 if (removalType.disconnectedFromDocument)
243 updateRelativeLengthsInformation(false, *this);
244
245 StyledElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
246
247 if (removalType.disconnectedFromDocument) {
248 auto& extensions = document().accessSVGExtensions();
249 if (m_svgRareData) {
250 for (auto& element : m_svgRareData->takeReferencingElements()) {
251 extensions.addElementToRebuild(element);
252 makeRef(element)->clearTarget();
253 }
254 RELEASE_ASSERT(m_svgRareData->referencingElements().computesEmpty());
255 }
256 extensions.removeElementToRebuild(*this);
257 }
258 invalidateInstances();
259
260 if (removalType.treeScopeChanged && oldParentOfRemovedTree.isUserAgentShadowRoot())
261 setCorrespondingElement(nullptr);
262}

264SVGSVGElement* SVGElement::ownerSVGElement() const
265{
266 ContainerNode* node = parentOrShadowHostNode();
267 while (node) {
268 if (is<SVGSVGElement>(*node))
269 return downcast<SVGSVGElement>(node);
270
271 node = node->parentOrShadowHostNode();
272 }
273
274 return nullptr;
275}

277SVGElement* SVGElement::viewportElement() const
278{
279 // This function needs shadow tree support - as RenderSVGContainer uses this function
280 // to determine the "overflow" property. <use> on <symbol> wouldn't work otherwhise.
281 ContainerNode* node = parentOrShadowHostNode();
282 while (node) {
283 if (is<SVGSVGElement>(*node) || is<SVGImageElement>(*node) || node->hasTagName(SVGNames::symbolTag))
284 return downcast<SVGElement>(node);
285
286 node = node->parentOrShadowHostNode();
287 }
288
289 return nullptr;
290}

369void SVGElement::parseAttribute(const QualifiedName& name, const AtomString& value)
370{
371 if (name == HTMLNames::classAttr) {
372 m_className->setBaseValInternal(value);
373 return;
374 }
375
376 if (name == HTMLNames::tabindexAttr) {
377 if (value.isEmpty())
378 setTabIndexExplicitly(std::nullopt);
379 else if (auto optionalTabIndex = parseHTMLInteger(value))
380 setTabIndexExplicitly(optionalTabIndex.value());
381 return;
382 }
383
384 auto& eventName = HTMLElement::eventNameForEventHandlerAttribute(name);
385 if (!eventName.isNull()) {
386 setAttributeEventListener(eventName, name, value);
387 return;
388 }
389}

400bool SVGElement::addEventListener(const AtomString& eventType, Ref<EventListener>&& listener, const AddEventListenerOptions& options)
401{
402 // Add event listener to regular DOM element
403 if (!Node::addEventListener(eventType, listener.copyRef(), options))
404 return false;
405
406 if (containingShadowRoot())
407 return true;
408
409 // Add event listener to all shadow tree DOM element instances
410 ASSERT(!instanceUpdatesBlocked());
411 for (auto& instance : copyToVectorOf<Ref<SVGElement>>(instances())) {
412 ASSERT(instance->correspondingElement() == this);
413 ASSERT(instance->isInUserAgentShadowTree());
414 bool result = instance->Node::addEventListener(eventType, listener.copyRef(), options);
415 ASSERT_UNUSED(result, result);
416 }
417
418 return true;
419}
420
421bool SVGElement::removeEventListener(const AtomString& eventType, EventListener& listener, const EventListenerOptions& options)
422{
423 if (containingShadowRoot())
424 return Node::removeEventListener(eventType, listener, options);
425
426 // EventTarget::removeEventListener creates a Ref around the given EventListener
427 // object when creating a temporary RegisteredEventListener object used to look up the
428 // event listener in a cache. If we want to be able to call removeEventListener() multiple
429 // times on different nodes, we have to delay its immediate destruction, which would happen
430 // after the first call below.
431 Ref<EventListener> protector(listener);
432
433 // Remove event listener from regular DOM element
434 if (!Node::removeEventListener(eventType, listener, options))
435 return false;
436
437 // Remove event listener from all shadow tree DOM element instances
438 ASSERT(!instanceUpdatesBlocked());
439 for (auto& instance : copyToVectorOf<Ref<SVGElement>>(instances())) {
440 ASSERT(instance->correspondingElement() == this);
441 ASSERT(instance->isInUserAgentShadowTree());
442
443 if (instance->Node::removeEventListener(eventType, listener, options))
444 continue;
445
446 // This case can only be hit for event listeners created from markup
447 ASSERT(listener.wasCreatedFromMarkup());
448
449 // If the event listener 'listener' has been created from markup and has been fired before
450 // then JSLazyEventListener::parseCode() has been called and m_jsFunction of that listener
451 // has been created (read: it's not 0 anymore). During shadow tree creation, the event
452 // listener DOM attribute has been cloned, and another event listener has been setup in
453 // the shadow tree. If that event listener has not been used yet, m_jsFunction is still 0,
454 // and tryRemoveEventListener() above will fail. Work around that very rare problem.
455 ASSERT(instance->eventTargetData());
456 instance->eventTargetData()->eventListenerMap.removeFirstEventListenerCreatedFromMarkup(eventType);
457 }
458
459 return true;
460}

462static bool hasLoadListener(Element* element)
463{
464 if (element->hasEventListeners(eventNames().loadEvent))
465 return true;
466
467 for (element = element->parentOrShadowHostElement(); element; element = element->parentOrShadowHostElement()) {
468 if (element->hasCapturingEventListeners(eventNames().loadEvent))
469 return true;
470 }
471
472 return false;
473}

509bool SVGElement::childShouldCreateRenderer(const Node& child) const
510{
511 if (!child.isSVGElement())
512 return false;
513 auto& svgChild = downcast<SVGElement>(child);
514
515 static const QualifiedName* const invalidTextContent[] {
516 &SVGNames::altGlyphTag.get(),
517 &SVGNames::textPathTag.get(),
518 &SVGNames::trefTag.get(),
519 &SVGNames::tspanTag.get(),
520 };
521 auto& name = svgChild.localName();
522 for (auto* tag : invalidTextContent) {
523 if (name == tag->localName())
524 return false;
525 }
526
527 return svgChild.isValid();
528}

530void SVGElement::attributeChanged(const QualifiedName& name, const AtomString& oldValue, const AtomString& newValue, AttributeModificationReason)
531{
532 StyledElement::attributeChanged(name, oldValue, newValue);
533
534 if (name == HTMLNames::idAttr)
535 document().accessSVGExtensions().rebuildAllElementReferencesForTarget(*this);
536
537 // Changes to the style attribute are processed lazily (see Element::getAttribute() and related methods),
538 // so we don't want changes to the style attribute to result in extra work here except invalidateInstances().
539 if (name == HTMLNames::styleAttr)
540 invalidateInstances();
541 else
542 svgAttributeChanged(name);
543}

561void SVGElement::commitPropertyChange(SVGProperty* property)
562{
563 // We want to dirty the top-level property when a descendant changes. For example
564 // a change in an SVGLength item in SVGLengthList should set the dirty flag on
565 // SVGLengthList and not the SVGLength.
566 property->setDirty();
567
568 invalidateSVGAttributes();
569 svgAttributeChanged(propertyRegistry().propertyAttributeName(*property));
570}
571
572void SVGElement::commitPropertyChange(SVGAnimatedProperty& animatedProperty)
573{
574 QualifiedName attributeName = propertyRegistry().animatedPropertyAttributeName(animatedProperty);
575 ASSERT(attributeName != nullQName());
576
577 // A change in a style property, e.g SVGRectElement::x should be serialized to
578 // the attribute immediately. Otherwise it is okay to be lazy in this regard.
579 if (!propertyRegistry().isAnimatedStylePropertyAttribute(attributeName))
580 propertyRegistry().setAnimatedPropertDirty(attributeName, animatedProperty);
581 else
582 setSynchronizedLazyAttribute(attributeName, animatedProperty.baseValAsString());
583
584 invalidateSVGAttributes();
585 svgAttributeChanged(attributeName);
586}

603RefPtr<SVGAttributeAnimator> SVGElement::createAnimator(const QualifiedName& attributeName, AnimationMode animationMode, CalcMode calcMode, bool isAccumulated, bool isAdditive)
604{
605 // Property animator, e.g. "fill" or "fill-opacity".
606 if (auto animator = propertyAnimatorFactory().createAnimator(attributeName, animationMode, calcMode, isAccumulated, isAdditive))
607 return animator;
608
609 // Animated property animator.
610 auto animator = propertyRegistry().createAnimator(attributeName, animationMode, calcMode, isAccumulated, isAdditive);
611 if (!animator)
612 return animator;
613 for (auto& instance : copyToVectorOf<Ref<SVGElement>>(instances()))
614 instance->propertyRegistry().appendAnimatedInstance(attributeName, *animator);
615 return animator;
616}

838void SVGElement::svgAttributeChanged(const QualifiedName& attrName)
839{
840 CSSPropertyID propId = cssPropertyIdForSVGAttributeName(attrName);
841 if (propId > 0) {
842 invalidateInstances();
843 return;
844 }
845
846 if (attrName == HTMLNames::classAttr) {
847 classAttributeChanged(className());
848 invalidateInstances();
849 return;
850 }
851
852 if (attrName == HTMLNames::idAttr) {
853 auto renderer = this->renderer();
854 // Notify resources about id changes, this is important as we cache resources by id in SVGDocumentExtensions
855 if (is<RenderSVGResourceContainer>(renderer))
856 downcast<RenderSVGResourceContainer>(*renderer).idChanged();
857 if (isConnected())
858 buildPendingResourcesIfNeeded();
859 invalidateInstances();
860 return;
861 }

884void SVGElement::buildPendingResourcesIfNeeded()
885{
886 if (!needsPendingResourceHandling() || !isConnected() || isInShadowTree())
887 return;
888
889 SVGDocumentExtensions& extensions = document().accessSVGExtensions();
890 String resourceId = getIdAttribute();
891 if (!extensions.isIdOfPendingResource(resourceId))
892 return;
893
894 // Mark pending resources as pending for removal.
895 extensions.markPendingResourcesForRemoval(resourceId);
896
897 // Rebuild pending resources for each client of a pending resource that is being removed.
898 while (auto clientElement = extensions.takeElementFromPendingResourcesForRemovalMap(resourceId)) {
899 ASSERT(clientElement->hasPendingResources());
900 if (clientElement->hasPendingResources()) {
901 clientElement->buildPendingResource();
902 if (auto renderer = clientElement->renderer()) {
903 for (auto& ancestor : ancestorsOfType<RenderSVGResourceContainer>(*renderer))
904 ancestor.markAllClientsForRepaint();
905 }
906 extensions.clearHasPendingResourcesIfPossible(*clientElement);
907 }
908 }

941void SVGElement::updateRelativeLengthsInformation(bool hasRelativeLengths, SVGElement& element)
942{
943 // If we're not yet in a document, this function will be called again from insertedIntoAncestor(). Do nothing now.
944 if (!isConnected())
945 return;
946
947 // An element wants to notify us that its own relative lengths state changed.
948 // Register it in the relative length map, and register us in the parent relative length map.
949 // Register the parent in the grandparents map, etc. Repeat procedure until the root of the SVG tree.
950
951 if (hasRelativeLengths)
952 m_elementsWithRelativeLengths.add(element);
953 else {
954 bool neverRegistered = !m_elementsWithRelativeLengths.contains(element);
955 if (neverRegistered)
956 return;
957
958 m_elementsWithRelativeLengths.remove(element);
959 }
960
961 if (is<SVGGraphicsElement>(element)) {
962 if (auto parent = makeRefPtr(parentNode()); is<SVGElement>(parent))
963 downcast<SVGElement>(*parent).updateRelativeLengthsInformation(hasRelativeLengths, *this);
964 }
