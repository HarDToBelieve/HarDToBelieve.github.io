22#include "StyleProperties.h"

28namespace WebCore {
29
30class CSSCursorImageValue;
31class SVGCursorElement;
32class SVGElement;
33
34class SVGElementRareData {
35 WTF_MAKE_NONCOPYABLE(SVGElementRareData); WTF_MAKE_FAST_ALLOCATED;
36public:
37 SVGElementRareData()
38 : m_instancesUpdatesBlocked(false)
39 , m_useOverrideComputedStyle(false)
40 , m_needsOverrideComputedStyleUpdate(false)
41 {
42 }
43
44 void addInstance(SVGElement& element) { m_instances.add(element); }
45 void removeInstance(SVGElement& element) { m_instances.remove(element); }
46 const WeakHashSet<SVGElement>& instances() const { return m_instances; }
47
48 bool instanceUpdatesBlocked() const { return m_instancesUpdatesBlocked; }
49 void setInstanceUpdatesBlocked(bool value) { m_instancesUpdatesBlocked = value; }
50
51 void addReferencingElement(SVGElement& element) { m_referencingElements.add(element); }
52 void removeReferencingElement(SVGElement& element) { m_referencingElements.remove(element); }
53 const WeakHashSet<SVGElement>& referencingElements() const { return m_referencingElements; }
54 WeakHashSet<SVGElement> takeReferencingElements() { return std::exchange(m_referencingElements, { }); }
55 SVGElement* referenceTarget() const { return m_referenceTarget.get(); }
56 void setReferenceTarget(WeakPtr<SVGElement>&& element) { m_referenceTarget = WTFMove(element); }
57
58 SVGElement* correspondingElement() { return m_correspondingElement.get(); }
59 void setCorrespondingElement(SVGElement* correspondingElement) { m_correspondingElement = makeWeakPtr(correspondingElement); }
60
61 MutableStyleProperties* animatedSMILStyleProperties() const { return m_animatedSMILStyleProperties.get(); }
62 MutableStyleProperties& ensureAnimatedSMILStyleProperties()
63 {
64 if (!m_animatedSMILStyleProperties)
65 m_animatedSMILStyleProperties = MutableStyleProperties::create(SVGAttributeMode);
66 return *m_animatedSMILStyleProperties;
67 }
68
69 const RenderStyle* overrideComputedStyle(Element& element, const RenderStyle* parentStyle)
70 {
71 if (!m_useOverrideComputedStyle)
72 return nullptr;
73 if (!m_overrideComputedStyle || m_needsOverrideComputedStyleUpdate) {
74 // The style computed here contains no CSS Animations/Transitions or SMIL induced rules - this is needed to compute the "base value" for the SMIL animation sandwhich model.
75 m_overrideComputedStyle = element.styleResolver().styleForElement(element, parentStyle, nullptr, RuleMatchingBehavior::MatchAllRulesExcludingSMIL).renderStyle;
76 m_needsOverrideComputedStyleUpdate = false;
77 }
78 ASSERT(m_overrideComputedStyle);
79 return m_overrideComputedStyle.get();
80 }
81
82 bool useOverrideComputedStyle() const { return m_useOverrideComputedStyle; }
83 void setUseOverrideComputedStyle(bool value) { m_useOverrideComputedStyle = value; }
84 void setNeedsOverrideComputedStyleUpdate() { m_needsOverrideComputedStyleUpdate = true; }
85
86private:
87 WeakHashSet<SVGElement> m_referencingElements;
88 WeakPtr<SVGElement> m_referenceTarget;
89 WeakHashSet<SVGElement> m_instances;
90 WeakPtr<SVGElement> m_correspondingElement;
91 bool m_instancesUpdatesBlocked : 1;
92 bool m_useOverrideComputedStyle : 1;
93 bool m_needsOverrideComputedStyleUpdate : 1;
94 RefPtr<MutableStyleProperties> m_animatedSMILStyleProperties;
95 std::unique_ptr<RenderStyle> m_overrideComputedStyle;
96};
