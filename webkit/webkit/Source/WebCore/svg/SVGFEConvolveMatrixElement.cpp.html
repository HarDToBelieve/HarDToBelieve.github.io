63void SVGFEConvolveMatrixElement::parseAttribute(const QualifiedName& name, const AtomString& value)
64{
65 if (name == SVGNames::inAttr) {
66 m_in1->setBaseValInternal(value);
67 return;
68 }
69
70 if (name == SVGNames::orderAttr) {
71 auto result = parseNumberOptionalNumber(value);
72 if (result && result->first >= 1 && result->second >= 1) {
73 m_orderX->setBaseValInternal(result->first);
74 m_orderY->setBaseValInternal(result->second);
75 } else
76 document().accessSVGExtensions().reportWarning("feConvolveMatrix: problem parsing order=\"" + value + "\". Filtered element will not be displayed.");
77 return;
78 }
79
80 if (name == SVGNames::edgeModeAttr) {
81 EdgeModeType propertyValue = SVGPropertyTraits<EdgeModeType>::fromString(value);
82 if (propertyValue > 0)
83 m_edgeMode->setBaseValInternal<EdgeModeType>(propertyValue);
84 else
85 document().accessSVGExtensions().reportWarning("feConvolveMatrix: problem parsing edgeMode=\"" + value + "\". Filtered element will not be displayed.");
86 return;
87 }
88
89 if (name == SVGNames::kernelMatrixAttr) {
90 m_kernelMatrix->baseVal()->parse(value);
91 return;
92 }
93
94 if (name == SVGNames::divisorAttr) {
95 float divisor = value.toFloat();
96 if (divisor)
97 m_divisor->setBaseValInternal(divisor);
98 else
99 document().accessSVGExtensions().reportWarning("feConvolveMatrix: problem parsing divisor=\"" + value + "\". Filtered element will not be displayed.");
100 return;
101 }
102
103 if (name == SVGNames::biasAttr) {
104 m_bias->setBaseValInternal(value.toFloat());
105 return;
106 }
107
108 if (name == SVGNames::targetXAttr) {
109 m_targetX->setBaseValInternal(parseInteger<unsigned>(value).value_or(0));
110 return;
111 }
112
113 if (name == SVGNames::targetYAttr) {
114 m_targetY->setBaseValInternal(parseInteger<unsigned>(value).value_or(0));
115 return;
116 }
117
118 if (name == SVGNames::kernelUnitLengthAttr) {
119 auto result = parseNumberOptionalNumber(value);
120 if (result && result->first > 0 && result->second > 0) {
121 m_kernelUnitLengthX->setBaseValInternal(result->first);
122 m_kernelUnitLengthY->setBaseValInternal(result->second);
123 } else
124 document().accessSVGExtensions().reportWarning("feConvolveMatrix: problem parsing kernelUnitLength=\"" + value + "\". Filtered element will not be displayed.");
125 return;
126 }
127
128 if (name == SVGNames::preserveAlphaAttr) {
129 if (value == "true")
130 m_preserveAlpha->setBaseValInternal(true);
131 else if (value == "false")
132 m_preserveAlpha->setBaseValInternal(false);
133 else
134 document().accessSVGExtensions().reportWarning("feConvolveMatrix: problem parsing preserveAlphaAttr=\"" + value + "\". Filtered element will not be displayed.");
135 return;
136 }
137
138 SVGFilterPrimitiveStandardAttributes::parseAttribute(name, value);
139}

141bool SVGFEConvolveMatrixElement::setFilterEffectAttribute(FilterEffect* effect, const QualifiedName& attrName)
142{
143 FEConvolveMatrix* convolveMatrix = static_cast<FEConvolveMatrix*>(effect);
144 if (attrName == SVGNames::edgeModeAttr)
145 return convolveMatrix->setEdgeMode(edgeMode());
146 if (attrName == SVGNames::divisorAttr)
147 return convolveMatrix->setDivisor(divisor());
148 if (attrName == SVGNames::biasAttr)
149 return convolveMatrix->setBias(bias());
150 if (attrName == SVGNames::targetXAttr)
151 return convolveMatrix->setTargetOffset(IntPoint(targetX(), targetY()));
152 if (attrName == SVGNames::targetYAttr)
153 return convolveMatrix->setTargetOffset(IntPoint(targetX(), targetY()));
154 if (attrName == SVGNames::kernelUnitLengthAttr)
155 return convolveMatrix->setKernelUnitLength(FloatPoint(kernelUnitLengthX(), kernelUnitLengthY()));
156 if (attrName == SVGNames::preserveAlphaAttr)
157 return convolveMatrix->setPreserveAlpha(preserveAlpha());
158
159 ASSERT_NOT_REACHED();
160 return false;
161}

194RefPtr<FilterEffect> SVGFEConvolveMatrixElement::build(SVGFilterBuilder* filterBuilder, Filter& filter) const
195{
196 auto input1 = filterBuilder->getEffectById(in1());
197
198 if (!input1)
199 return nullptr;
200
201 int orderXValue = orderX();
202 int orderYValue = orderY();
203 if (!hasAttribute(SVGNames::orderAttr)) {
204 orderXValue = 3;
205 orderYValue = 3;
206 }
207 // Spec says order must be > 0. Bail if it is not.
208 if (orderXValue < 1 || orderYValue < 1)
209 return nullptr;
210 auto& kernelMatrix = this->kernelMatrix();
211 int kernelMatrixSize = kernelMatrix.items().size();
212 // The spec says this is a requirement, and should bail out if fails
213 if (orderXValue * orderYValue != kernelMatrixSize)
214 return nullptr;
215
216 int targetXValue = targetX();
217 int targetYValue = targetY();
218 if (hasAttribute(SVGNames::targetXAttr) && (targetXValue < 0 || targetXValue >= orderXValue))
219 return nullptr;
220 // The spec says the default value is: targetX = floor ( orderX / 2 ))
221 if (!hasAttribute(SVGNames::targetXAttr))
222 targetXValue = static_cast<int>(floorf(orderXValue / 2));
223 if (hasAttribute(SVGNames::targetYAttr) && (targetYValue < 0 || targetYValue >= orderYValue))
224 return nullptr;
225 // The spec says the default value is: targetY = floor ( orderY / 2 ))
226 if (!hasAttribute(SVGNames::targetYAttr))
227 targetYValue = static_cast<int>(floorf(orderYValue / 2));
228
229 // Spec says default kernelUnitLength is 1.0, and a specified length cannot be 0.
230 int kernelUnitLengthXValue = kernelUnitLengthX();
231 int kernelUnitLengthYValue = kernelUnitLengthY();
232 if (!hasAttribute(SVGNames::kernelUnitLengthAttr)) {
233 kernelUnitLengthXValue = 1;
234 kernelUnitLengthYValue = 1;
235 }
236 if (kernelUnitLengthXValue <= 0 || kernelUnitLengthYValue <= 0)
237 return nullptr;
238
239 float divisorValue = divisor();
240 if (hasAttribute(SVGNames::divisorAttr) && !divisorValue)
241 return nullptr;
242 if (!hasAttribute(SVGNames::divisorAttr)) {
243 for (int i = 0; i < kernelMatrixSize; ++i)
244 divisorValue += kernelMatrix.items()[i]->value();
245 if (!divisorValue)
246 divisorValue = 1;
247 }
248
249 auto effect = FEConvolveMatrix::create(filter, IntSize(orderXValue, orderYValue), divisorValue, bias(), IntPoint(targetXValue, targetYValue), edgeMode(), FloatPoint(kernelUnitLengthXValue, kernelUnitLengthYValue), preserveAlpha(), kernelMatrix);
250 effect->inputEffects() = { input1 };
251 return effect;
252}
