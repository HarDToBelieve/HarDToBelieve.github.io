37inline SVGFEDiffuseLightingElement::SVGFEDiffuseLightingElement(const QualifiedName& tagName, Document& document)
38 : SVGFilterPrimitiveStandardAttributes(tagName, document)
39{
40 ASSERT(hasTagName(SVGNames::feDiffuseLightingTag));
41
42 static std::once_flag onceFlag;
43 std::call_once(onceFlag, [] {
44 PropertyRegistry::registerProperty<SVGNames::inAttr, &SVGFEDiffuseLightingElement::m_in1>();
45 PropertyRegistry::registerProperty<SVGNames::diffuseConstantAttr, &SVGFEDiffuseLightingElement::m_diffuseConstant>();
46 PropertyRegistry::registerProperty<SVGNames::surfaceScaleAttr, &SVGFEDiffuseLightingElement::m_surfaceScale>();
47 PropertyRegistry::registerProperty<SVGNames::kernelUnitLengthAttr, &SVGFEDiffuseLightingElement::m_kernelUnitLengthX, &SVGFEDiffuseLightingElement::m_kernelUnitLengthY>();
48 });
49}

56void SVGFEDiffuseLightingElement::parseAttribute(const QualifiedName& name, const AtomString& value)
57{
58 if (name == SVGNames::inAttr) {
59 m_in1->setBaseValInternal(value);
60 return;
61 }
62
63 if (name == SVGNames::surfaceScaleAttr) {
64 m_surfaceScale->setBaseValInternal(value.toFloat());
65 return;
66 }
67
68 if (name == SVGNames::diffuseConstantAttr) {
69 m_diffuseConstant->setBaseValInternal(value.toFloat());
70 return;
71 }
72
73 if (name == SVGNames::kernelUnitLengthAttr) {
74 if (auto result = parseNumberOptionalNumber(value)) {
75 m_kernelUnitLengthX->setBaseValInternal(result->first);
76 m_kernelUnitLengthY->setBaseValInternal(result->second);
77 }
78 return;
79 }
80
81 SVGFilterPrimitiveStandardAttributes::parseAttribute(name, value);
82}

84bool SVGFEDiffuseLightingElement::setFilterEffectAttribute(FilterEffect* effect, const QualifiedName& attrName)
85{
86 FEDiffuseLighting* diffuseLighting = static_cast<FEDiffuseLighting*>(effect);
87
88 if (attrName == SVGNames::lighting_colorAttr) {
89 RenderObject* renderer = this->renderer();
90 ASSERT(renderer);
91 Color color = renderer->style().colorByApplyingColorFilter(renderer->style().svgStyle().lightingColor());
92 return diffuseLighting->setLightingColor(color);
93 }
94 if (attrName == SVGNames::surfaceScaleAttr)
95 return diffuseLighting->setSurfaceScale(surfaceScale());
96 if (attrName == SVGNames::diffuseConstantAttr)
97 return diffuseLighting->setDiffuseConstant(diffuseConstant());
98
99 auto& lightSource = const_cast<LightSource&>(diffuseLighting->lightSource());
100 const SVGFELightElement* lightElement = SVGFELightElement::findLightElement(this);
101 ASSERT(lightElement);
102
103 if (attrName == SVGNames::azimuthAttr)
104 return lightSource.setAzimuth(lightElement->azimuth());
105 if (attrName == SVGNames::elevationAttr)
106 return lightSource.setElevation(lightElement->elevation());
107 if (attrName == SVGNames::xAttr)
108 return lightSource.setX(lightElement->x());
109 if (attrName == SVGNames::yAttr)
110 return lightSource.setY(lightElement->y());
111 if (attrName == SVGNames::zAttr)
112 return lightSource.setZ(lightElement->z());
113 if (attrName == SVGNames::pointsAtXAttr)
114 return lightSource.setPointsAtX(lightElement->pointsAtX());
115 if (attrName == SVGNames::pointsAtYAttr)
116 return lightSource.setPointsAtY(lightElement->pointsAtY());
117 if (attrName == SVGNames::pointsAtZAttr)
118 return lightSource.setPointsAtZ(lightElement->pointsAtZ());
119 if (attrName == SVGNames::specularExponentAttr)
120 return lightSource.setSpecularExponent(lightElement->specularExponent());
121 if (attrName == SVGNames::limitingConeAngleAttr)
122 return lightSource.setLimitingConeAngle(lightElement->limitingConeAngle());
123
124 ASSERT_NOT_REACHED();
125 return false;
126}

128void SVGFEDiffuseLightingElement::svgAttributeChanged(const QualifiedName& attrName)
129{
130 if (attrName == SVGNames::surfaceScaleAttr || attrName == SVGNames::diffuseConstantAttr || attrName == SVGNames::kernelUnitLengthAttr || attrName == SVGNames::lighting_colorAttr) {
131 InstanceInvalidationGuard guard(*this);
132 primitiveAttributeChanged(attrName);
133 return;
134 }
135
136 if (attrName == SVGNames::inAttr) {
137 InstanceInvalidationGuard guard(*this);
138 invalidate();
139 return;
140 }
141
142 SVGFilterPrimitiveStandardAttributes::svgAttributeChanged(attrName);
143}

145void SVGFEDiffuseLightingElement::lightElementAttributeChanged(const SVGFELightElement* lightElement, const QualifiedName& attrName)
146{
147 if (SVGFELightElement::findLightElement(this) != lightElement)
148 return;
149
150 // The light element has different attribute names.
151 primitiveAttributeChanged(attrName);
152}

154RefPtr<FilterEffect> SVGFEDiffuseLightingElement::build(SVGFilterBuilder* filterBuilder, Filter& filter) const
155{
156 auto input1 = filterBuilder->getEffectById(in1());
157
158 if (!input1)
159 return nullptr;
160
161 auto lightElement = makeRefPtr(SVGFELightElement::findLightElement(this));
162 if (!lightElement)
163 return nullptr;
164
165 auto lightSource = lightElement->lightSource(*filterBuilder);
166
167 RenderObject* renderer = this->renderer();
168 if (!renderer)
169 return nullptr;
170
171 Color color = renderer->style().colorByApplyingColorFilter(renderer->style().svgStyle().lightingColor());
172
173 auto effect = FEDiffuseLighting::create(filter, color, surfaceScale(), diffuseConstant(), kernelUnitLengthX(), kernelUnitLengthY(), WTFMove(lightSource));
174 effect->inputEffects() = { input1 };
175 return effect;
176}
