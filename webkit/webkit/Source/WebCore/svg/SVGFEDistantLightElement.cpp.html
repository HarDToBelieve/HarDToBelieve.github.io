28inline SVGFEDistantLightElement::SVGFEDistantLightElement(const QualifiedName& tagName, Document& document)
29 : SVGFELightElement(tagName, document)
30{
31 ASSERT(hasTagName(SVGNames::feDistantLightTag));
32}
33
34Ref<SVGFEDistantLightElement> SVGFEDistantLightElement::create(const QualifiedName& tagName, Document& document)
35{
36 return adoptRef(*new SVGFEDistantLightElement(tagName, document));
37}

39Ref<LightSource> SVGFEDistantLightElement::lightSource(SVGFilterBuilder&) const
40{
41 return DistantLightSource::create(azimuth(), elevation());
42}
