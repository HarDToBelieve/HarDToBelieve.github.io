35inline SVGFEGaussianBlurElement::SVGFEGaussianBlurElement(const QualifiedName& tagName, Document& document)
36 : SVGFilterPrimitiveStandardAttributes(tagName, document)
37{
38 ASSERT(hasTagName(SVGNames::feGaussianBlurTag));
39
40 static std::once_flag onceFlag;
41 std::call_once(onceFlag, [] {
42 PropertyRegistry::registerProperty<SVGNames::inAttr, &SVGFEGaussianBlurElement::m_in1>();
43 PropertyRegistry::registerProperty<SVGNames::stdDeviationAttr, &SVGFEGaussianBlurElement::m_stdDeviationX, &SVGFEGaussianBlurElement::m_stdDeviationY>();
44 PropertyRegistry::registerProperty<SVGNames::edgeModeAttr, EdgeModeType, &SVGFEGaussianBlurElement::m_edgeMode>();
45 });
46}

53void SVGFEGaussianBlurElement::setStdDeviation(float x, float y)
54{
55 m_stdDeviationX->setBaseValInternal(x);
56 m_stdDeviationY->setBaseValInternal(y);
57 invalidate();
58}
59
60void SVGFEGaussianBlurElement::parseAttribute(const QualifiedName& name, const AtomString& value)
61{
62 if (name == SVGNames::stdDeviationAttr) {
63 if (auto result = parseNumberOptionalNumber(value)) {
64 m_stdDeviationX->setBaseValInternal(result->first);
65 m_stdDeviationY->setBaseValInternal(result->second);
66 }
67 return;
68 }
69
70 if (name == SVGNames::inAttr) {
71 m_in1->setBaseValInternal(value);
72 return;
73 }
74
75 if (name == SVGNames::edgeModeAttr) {
76 auto propertyValue = SVGPropertyTraits<EdgeModeType>::fromString(value);
77 if (propertyValue > 0)
78 m_edgeMode->setBaseValInternal<EdgeModeType>(propertyValue);
79 else
80 document().accessSVGExtensions().reportWarning("feGaussianBlur: problem parsing edgeMode=\"" + value + "\". Filtered element will not be displayed.");
81 return;
82 }
83
84 SVGFilterPrimitiveStandardAttributes::parseAttribute(name, value);
85}

98RefPtr<FilterEffect> SVGFEGaussianBlurElement::build(SVGFilterBuilder* filterBuilder, Filter& filter) const
99{
100 auto input1 = filterBuilder->getEffectById(in1());
101 if (!input1)
102 return nullptr;
103
104 if (stdDeviationX() < 0 || stdDeviationY() < 0)
105 return nullptr;
106
107 auto effect = FEGaussianBlur::create(filter, stdDeviationX(), stdDeviationY(), edgeMode());
108 effect->inputEffects() = { input1 };
109 return effect;
110}
