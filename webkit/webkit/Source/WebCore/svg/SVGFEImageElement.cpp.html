41inline SVGFEImageElement::SVGFEImageElement(const QualifiedName& tagName, Document& document)
42 : SVGFilterPrimitiveStandardAttributes(tagName, document)
43 , SVGURIReference(this)
44{
45 ASSERT(hasTagName(SVGNames::feImageTag));
46
47 static std::once_flag onceFlag;
48 std::call_once(onceFlag, [] {
49 PropertyRegistry::registerProperty<SVGNames::preserveAspectRatioAttr, &SVGFEImageElement::m_preserveAspectRatio>();
50 });
51}

63bool SVGFEImageElement::hasSingleSecurityOrigin() const
64{
65 if (!m_cachedImage)
66 return true;
67 auto* image = m_cachedImage->image();
68 return !image || image->hasSingleSecurityOrigin();
69}

71void SVGFEImageElement::clearResourceReferences()
72{
73 if (m_cachedImage) {
74 m_cachedImage->removeClient(*this);
75 m_cachedImage = nullptr;
76 }
77
78 removeElementReference();
79}

81void SVGFEImageElement::requestImageResource()
82{
83 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
84 options.contentSecurityPolicyImposition = isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
85
86 CachedResourceRequest request(ResourceRequest(document().completeURL(href())), options);
87 request.setInitiator(*this);
88 m_cachedImage = document().cachedResourceLoader().requestImage(WTFMove(request)).value_or(nullptr);
89
90 if (m_cachedImage)
91 m_cachedImage->addClient(*this);
92}
93
94void SVGFEImageElement::buildPendingResource()
95{
96 clearResourceReferences();
97 if (!isConnected())
98 return;
99
100 auto target = SVGURIReference::targetElementFromIRIString(href(), treeScope());
101 if (!target.element) {
102 if (target.identifier.isEmpty())
103 requestImageResource();
104 else {
105 document().accessSVGExtensions().addPendingResource(target.identifier, *this);
106 ASSERT(hasPendingResources());
107 }
108 } else if (is<SVGElement>(*target.element))
109 downcast<SVGElement>(*target.element).addReferencingElement(*this);
110
111 invalidate();
112}

114void SVGFEImageElement::parseAttribute(const QualifiedName& name, const AtomString& value)
115{
116 if (name == SVGNames::preserveAspectRatioAttr) {
117 m_preserveAspectRatio->setBaseValInternal(SVGPreserveAspectRatioValue { value });
118 return;
119 }
120
121 SVGFilterPrimitiveStandardAttributes::parseAttribute(name, value);
122 SVGURIReference::parseAttribute(name, value);
123}

125void SVGFEImageElement::svgAttributeChanged(const QualifiedName& attrName)
126{
127 if (attrName == SVGNames::preserveAspectRatioAttr) {
128 InstanceInvalidationGuard guard(*this);
129 invalidate();
130 return;
131 }
132
133 if (SVGURIReference::isKnownAttribute(attrName)) {
134 InstanceInvalidationGuard guard(*this);
135 buildPendingResource();
136 return;
137 }
138
139 SVGFilterPrimitiveStandardAttributes::svgAttributeChanged(attrName);
140}

160void SVGFEImageElement::notifyFinished(CachedResource&, const NetworkLoadMetrics&)
161{
162 if (!isConnected())
163 return;
164
165 auto parent = makeRefPtr(parentElement());
166
167 if (!parent || !parent->hasTagName(SVGNames::filterTag))
168 return;
169
170 RenderElement* parentRenderer = parent->renderer();
171 if (!parentRenderer)
172 return;
173
174 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*parentRenderer);
175}
176
177RefPtr<FilterEffect> SVGFEImageElement::build(SVGFilterBuilder*, Filter& filter) const
178{
179 if (m_cachedImage)
180 return FEImage::createWithImage(filter, m_cachedImage->imageForRenderer(renderer()), preserveAspectRatio());
181
182 auto target = SVGURIReference::targetElementFromIRIString(href(), treeScope());
183 if (isDescendantOrShadowDescendantOf(target.element.get()))
184 return nullptr;
185
186 return FEImage::createWithIRIReference(filter, treeScope(), href(), preserveAspectRatio());
187}
