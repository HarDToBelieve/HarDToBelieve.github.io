36inline SVGFEMergeElement::SVGFEMergeElement(const QualifiedName& tagName, Document& document)
37 : SVGFilterPrimitiveStandardAttributes(tagName, document)

39 ASSERT(hasTagName(SVGNames::feMergeTag));
40}

42Ref<SVGFEMergeElement> SVGFEMergeElement::create(const QualifiedName& tagName, Document& document)
43{
44 return adoptRef(*new SVGFEMergeElement(tagName, document));
45}
46
47RefPtr<FilterEffect> SVGFEMergeElement::build(SVGFilterBuilder* filterBuilder, Filter& filter) const
48{
49 auto effect = FEMerge::create(filter);
50 auto& mergeInputs = effect->inputEffects();
51
52 for (auto& mergeNode : childrenOfType<SVGFEMergeNodeElement>(*this)) {
53 auto mergeEffect = filterBuilder->getEffectById(mergeNode.in1());
54 if (!mergeEffect)
55 return nullptr;
56 mergeInputs.append(WTFMove(mergeEffect));
57 }
58
59 mergeInputs.shrinkToFit();
60
61 if (mergeInputs.isEmpty())
62 return nullptr;
63
64 return effect;
65}
