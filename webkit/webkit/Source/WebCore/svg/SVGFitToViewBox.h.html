22#pragma once
23
24#include "FloatRect.h"
25#include "QualifiedName.h"
26#include "SVGAnimatedPropertyImpl.h"
27#include "SVGNames.h"
28#include "SVGPreserveAspectRatio.h"
29#include "SVGPropertyOwnerRegistry.h"
30
31namespace WebCore {
32
33class AffineTransform;
34
35class SVGFitToViewBox {
36 WTF_MAKE_NONCOPYABLE(SVGFitToViewBox);
37public:
38 static AffineTransform viewBoxToViewTransform(const FloatRect& viewBoxRect, const SVGPreserveAspectRatioValue&, float viewWidth, float viewHeight);
39
40 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGFitToViewBox>;
41
42 const FloatRect& viewBox() const { return m_viewBox->currentValue(); }
43 const SVGPreserveAspectRatioValue& preserveAspectRatio() const { return m_preserveAspectRatio->currentValue(); }
44
45 SVGAnimatedRect& viewBoxAnimated() { return m_viewBox; }
46 SVGAnimatedPreserveAspectRatio& preserveAspectRatioAnimated() { return m_preserveAspectRatio; }
47
48 void setViewBox(const FloatRect&);
49 void resetViewBox();
50
51 void setPreserveAspectRatio(const SVGPreserveAspectRatioValue& preserveAspectRatio) { m_preserveAspectRatio->setBaseValInternal(preserveAspectRatio); }
52 void resetPreserveAspectRatio() { m_preserveAspectRatio->setBaseValInternal({ }); }
53
54 String viewBoxString() const { return SVGPropertyTraits<FloatRect>::toString(viewBox()); }
55 String preserveAspectRatioString() const { return preserveAspectRatio().valueAsString(); }
56
57 bool hasValidViewBox() const { return m_isViewBoxValid; }
58 bool hasEmptyViewBox() const { return m_isViewBoxValid && viewBox().isEmpty(); }
59
60protected:
61 SVGFitToViewBox(SVGElement* contextElement, SVGPropertyAccess = SVGPropertyAccess::ReadWrite);
62
63 static bool isKnownAttribute(const QualifiedName& attributeName) { return PropertyRegistry::isKnownAttribute(attributeName); }
64
65 void reset();
66 bool parseAttribute(const QualifiedName&, const AtomString&);
67 std::optional<FloatRect> parseViewBox(StringView);
68 std::optional<FloatRect> parseViewBox(StringParsingBuffer<LChar>&, bool validate = true);
69 std::optional<FloatRect> parseViewBox(StringParsingBuffer<UChar>&, bool validate = true);
70
71private:
72 template<typename CharacterType> std::optional<FloatRect> parseViewBoxGeneric(StringParsingBuffer<CharacterType>&, bool validate = true);
73
74 Ref<SVGAnimatedRect> m_viewBox;
75 Ref<SVGAnimatedPreserveAspectRatio> m_preserveAspectRatio;
76 bool m_isViewBoxValid { false };
77};
78
79} // namespace WebCore
