73void SVGFontFaceElement::parseAttribute(const QualifiedName& name, const AtomString& value)
74{
75 CSSPropertyID propertyId = cssPropertyIdForSVGAttributeName(name);
76 if (propertyId > 0) {
77 // FIXME: Parse using the @font-face descriptor grammars, not the property grammars.
78 auto& properties = m_fontFaceRule->mutableProperties();
79 bool valueChanged = properties.setProperty(propertyId, value);
80
81 if (valueChanged) {
82 // The above parser is designed for the font-face properties, not descriptors, and the properties accept the global keywords, but descriptors don't.
83 // Rather than invasively modifying the parser for the properties to have a special mode, we can simply detect the error condition after-the-fact and
84 // avoid it explicitly.
85 if (auto parsedValue = properties.getPropertyCSSValue(propertyId)) {
86 if (parsedValue->isGlobalKeyword())
87 properties.removeProperty(propertyId);
88 }
89 }
90
91 rebuildFontFace();
92 return;
93 }
94
95 SVGElement::parseAttribute(name, value);
96}

98unsigned SVGFontFaceElement::unitsPerEm() const
99{
100 const AtomString& valueString = attributeWithoutSynchronization(units_per_emAttr);
101 if (valueString.isEmpty())
102 return FontMetrics::defaultUnitsPerEm;
103
104 auto value = static_cast<unsigned>(ceilf(valueString.toFloat()));
105 if (!value)
106 return FontMetrics::defaultUnitsPerEm;
107
108 return value;
109}

121float SVGFontFaceElement::horizontalOriginX() const
122{
123 if (!m_fontElement)
124 return 0.0f;
125
126 // Spec: The X-coordinate in the font coordinate system of the origin of a glyph to be used when
127 // drawing horizontally oriented text. (Note that the origin applies to all glyphs in the font.)
128 // If the attribute is not specified, the effect is as if a value of "0" were specified.
129 return m_fontElement->attributeWithoutSynchronization(horiz_origin_xAttr).toFloat();
130}

143float SVGFontFaceElement::horizontalAdvanceX() const
144{
145 if (!m_fontElement)
146 return 0.0f;
147
148 // Spec: The default horizontal advance after rendering a glyph in horizontal orientation. Glyph
149 // widths are required to be non-negative, even if the glyph is typically rendered right-to-left,

151 return m_fontElement->attributeWithoutSynchronization(horiz_adv_xAttr).toFloat();

154float SVGFontFaceElement::verticalOriginX() const
155{
156 if (!m_fontElement)
157 return 0.0f;
158
159 // Spec: The default X-coordinate in the font coordinate system of the origin of a glyph to be used when
160 // drawing vertically oriented text. If the attribute is not specified, the effect is as if the attribute
161 // were set to half of the effective value of attribute horiz-adv-x.
162 const AtomString& value = m_fontElement->attributeWithoutSynchronization(vert_origin_xAttr);
163 if (value.isEmpty())
164 return horizontalAdvanceX() / 2.0f;
165
166 return value.toFloat();
167}
168
169float SVGFontFaceElement::verticalOriginY() const
170{
171 if (!m_fontElement)
172 return 0.0f;
173
174 // Spec: The default Y-coordinate in the font coordinate system of the origin of a glyph to be used when
175 // drawing vertically oriented text. If the attribute is not specified, the effect is as if the attribute
176 // were set to the position specified by the font's ascent attribute.
177 const AtomString& value = m_fontElement->attributeWithoutSynchronization(vert_origin_yAttr);
178 if (value.isEmpty())
179 return ascent();
180
181 return value.toFloat();
182}
183
184float SVGFontFaceElement::verticalAdvanceY() const
185{
186 if (!m_fontElement)
187 return 0.0f;
188
189 // Spec: The default vertical advance after rendering a glyph in vertical orientation. If the attribute is
190 // not specified, the effect is as if a value equivalent of one em were specified (see units-per-em).
191 const AtomString& value = m_fontElement->attributeWithoutSynchronization(vert_adv_yAttr);
192 if (value.isEmpty())
193 return 1.0f;
194
195 return value.toFloat();
196}

198int SVGFontFaceElement::ascent() const
199{
200 // Spec: Same syntax and semantics as the 'ascent' descriptor within an @font-face rule. The maximum
201 // unaccented height of the font within the font coordinate system. If the attribute is not specified,
202 // the effect is as if the attribute were set to the difference between the units-per-em value and the
203 // vert-origin-y value for the corresponding font.
204 const AtomString& ascentValue = attributeWithoutSynchronization(ascentAttr);
205 if (!ascentValue.isEmpty())
206 return static_cast<int>(ceilf(ascentValue.toFloat()));
207
208 if (m_fontElement) {
209 const AtomString& vertOriginY = m_fontElement->attributeWithoutSynchronization(vert_origin_yAttr);
210 if (!vertOriginY.isEmpty())
211 return static_cast<int>(unitsPerEm()) - static_cast<int>(ceilf(vertOriginY.toFloat()));
212 }
213
214 // Match Batiks default value
215 return static_cast<int>(ceilf(unitsPerEm() * 0.8f));
216}
217
218int SVGFontFaceElement::descent() const
219{
220 // Spec: Same syntax and semantics as the 'descent' descriptor within an @font-face rule. The maximum
221 // unaccented depth of the font within the font coordinate system. If the attribute is not specified,
222 // the effect is as if the attribute were set to the vert-origin-y value for the corresponding font.
223 const AtomString& descentValue = attributeWithoutSynchronization(descentAttr);
224 if (!descentValue.isEmpty()) {
225 // 14 different W3C SVG 1.1 testcases use a negative descent value,
226 // where a positive was meant to be used Including:
227 // animate-elem-24-t.svg, fonts-elem-01-t.svg, fonts-elem-02-t.svg (and 11 others)
228 int descent = static_cast<int>(ceilf(descentValue.toFloat()));
229 return descent < 0 ? -descent : descent;
230 }
231
232 if (m_fontElement) {
233 const AtomString& vertOriginY = m_fontElement->attributeWithoutSynchronization(vert_origin_yAttr);
234 if (!vertOriginY.isEmpty())
235 return static_cast<int>(ceilf(vertOriginY.toFloat()));
236 }
237
238 // Match Batiks default value
239 return static_cast<int>(ceilf(unitsPerEm() * 0.2f));
240}

254void SVGFontFaceElement::rebuildFontFace()
255{
256 if (!isConnected()) {
257 ASSERT(!m_fontElement);
258 return;
259 }
260
261 // we currently ignore all but the first src element, alternatively we could concat them
262 auto srcElement = childrenOfType<SVGFontFaceSrcElement>(*this).first();
263
264 bool describesParentFont = is<SVGFontElement>(*parentNode());
265 RefPtr<CSSValueList> list;
266
267 if (describesParentFont) {
268 m_fontElement = downcast<SVGFontElement>(parentNode());
269
270 list = CSSValueList::createCommaSeparated();
271 list->append(CSSFontFaceSrcValue::createLocal(fontFamily()));
272 } else {
273 m_fontElement = nullptr;
274 if (srcElement)
275 list = srcElement->srcValue();
276 }
277
278 if (!list || !list->length())
279 return;
280
281 // Parse in-memory CSS rules
282 m_fontFaceRule->mutableProperties().addParsedProperty(CSSProperty(CSSPropertySrc, list));
283
284 if (describesParentFont) {
285 // Traverse parsed CSS values and associate CSSFontFaceSrcValue elements with ourselves.
286 RefPtr<CSSValue> src = m_fontFaceRule->properties().getPropertyCSSValue(CSSPropertySrc);
287 CSSValueList* srcList = downcast<CSSValueList>(src.get());
288
289 unsigned srcLength = srcList ? srcList->length() : 0;
290 for (unsigned i = 0; i < srcLength; ++i) {
291 if (auto item = makeRefPtr(downcast<CSSFontFaceSrcValue>(srcList->itemWithoutBoundsCheck(i))))
292 item->setSVGFontFaceElement(this);
293 }
294 }
295
296 document().styleScope().didChangeStyleSheetEnvironment();
297}

299Node::InsertedIntoAncestorResult SVGFontFaceElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
300{
301 SVGElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
302 if (!insertionType.connectedToDocument) {
303 ASSERT(!m_fontElement);
304 return InsertedIntoAncestorResult::Done;
305 }
306 document().accessSVGExtensions().registerSVGFontFaceElement(*this);
307
308 rebuildFontFace();
309 return InsertedIntoAncestorResult::Done;
310}

312void SVGFontFaceElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
313{
314 SVGElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
315
316 if (removalType.disconnectedFromDocument) {
317 m_fontElement = nullptr;
318 document().accessSVGExtensions().unregisterSVGFontFaceElement(*this);
319 m_fontFaceRule->mutableProperties().clear();
320
321 document().styleScope().didChangeStyleSheetEnvironment();
322 } else
323 ASSERT(!m_fontElement);
324}
