57Ref<CSSFontFaceSrcValue> SVGFontFaceUriElement::srcValue() const
58{
59 auto src = CSSFontFaceSrcValue::create(getAttribute(SVGNames::hrefAttr, XLinkNames::hrefAttr), LoadedFromOpaqueSource::No);
60 AtomString value(attributeWithoutSynchronization(formatAttr));
61 src.get().setFormat(value.isEmpty() ? "svg" : value); // Default format
62 return src;
63}
64
65void SVGFontFaceUriElement::parseAttribute(const QualifiedName& name, const AtomString& value)
66{
67 if (name == SVGNames::hrefAttr || name == XLinkNames::hrefAttr)
68 loadFont();
69 else
70 SVGElement::parseAttribute(name, value);
71}

73void SVGFontFaceUriElement::childrenChanged(const ChildChange& change)
74{
75 SVGElement::childrenChanged(change);
76
77 if (!parentNode() || !parentNode()->hasTagName(font_face_srcTag))
78 return;
79
80 auto grandparent = makeRefPtr(parentNode()->parentNode());
81 if (grandparent && grandparent->hasTagName(font_faceTag))
82 downcast<SVGFontFaceElement>(*grandparent).rebuildFontFace();
83}

85Node::InsertedIntoAncestorResult SVGFontFaceUriElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
86{

91static bool isSVGFontTarget(const SVGFontFaceUriElement& element)
92{
93 Ref<CSSFontFaceSrcValue> srcValue(element.srcValue());
94 return srcValue->isSVGFontTarget();
95}
96
97void SVGFontFaceUriElement::loadFont()
98{
99 if (m_cachedFont)
100 m_cachedFont->removeClient(*this);
101
102 const AtomString& href = getAttribute(SVGNames::hrefAttr, XLinkNames::hrefAttr);
103 if (!href.isNull()) {
104 ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
105 options.contentSecurityPolicyImposition = isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
106
107 CachedResourceLoader& cachedResourceLoader = document().cachedResourceLoader();
108 CachedResourceRequest request(ResourceRequest(document().completeURL(href)), options);
109 request.setInitiator(*this);
110 m_cachedFont = cachedResourceLoader.requestFont(WTFMove(request), isSVGFontTarget(*this)).value_or(nullptr);
111 if (m_cachedFont) {
112 m_cachedFont->addClient(*this);
113 m_cachedFont->beginLoadIfNeeded(cachedResourceLoader);
114 }
115 } else
116 m_cachedFont = nullptr;
117}
