35inline SVGGlyphRefElement::SVGGlyphRefElement(const QualifiedName& tagName, Document& document)
36 : SVGElement(tagName, document)

42Ref<SVGGlyphRefElement> SVGGlyphRefElement::create(const QualifiedName& tagName, Document& document)
43{
44 return adoptRef(*new SVGGlyphRefElement(tagName, document));
45}

47bool SVGGlyphRefElement::hasValidGlyphElement(String& glyphName) const
48{
49 // FIXME: We only support xlink:href so far.
50 // https://bugs.webkit.org/show_bug.cgi?id=64787
51 // No need to support glyphRef referencing another node inside a shadow tree.
52 auto target = targetElementFromIRIString(getAttribute(SVGNames::hrefAttr, XLinkNames::hrefAttr), document());
53 glyphName = target.identifier;
54 return is<SVGGlyphElement>(target.element);
55}

57static float parseFloat(const AtomString& value)
58{
59 return parseNumber(value).value_or(0);
60}
61
62void SVGGlyphRefElement::parseAttribute(const QualifiedName& name, const AtomString& value)
63{
64 // FIXME: Is the error handling in parseFloat correct for these attributes?
65 if (name == SVGNames::xAttr)
66 m_x = parseFloat(value);
67 else if (name == SVGNames::yAttr)
68 m_y = parseFloat(value);
69 else if (name == SVGNames::dxAttr)
70 m_dx = parseFloat(value);
71 else if (name == SVGNames::dyAttr)
72 m_dy = parseFloat(value);
73 else {
74 SVGURIReference::parseAttribute(name, value);
75 SVGElement::parseAttribute(name, value);
76 }
77}
