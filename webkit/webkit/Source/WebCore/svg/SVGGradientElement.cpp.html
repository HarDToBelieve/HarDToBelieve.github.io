41SVGGradientElement::SVGGradientElement(const QualifiedName& tagName, Document& document)
42 : SVGElement(tagName, document)
43 , SVGURIReference(this)
44{
45 static std::once_flag onceFlag;
46 std::call_once(onceFlag, [] {
47 PropertyRegistry::registerProperty<SVGNames::spreadMethodAttr, SVGSpreadMethodType, &SVGGradientElement::m_spreadMethod>();
48 PropertyRegistry::registerProperty<SVGNames::gradientUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGGradientElement::m_gradientUnits>();
49 PropertyRegistry::registerProperty<SVGNames::gradientTransformAttr, &SVGGradientElement::m_gradientTransform>();
50 });
51}

53void SVGGradientElement::parseAttribute(const QualifiedName& name, const AtomString& value)
54{
55 if (name == SVGNames::gradientUnitsAttr) {
56 auto propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
57 if (propertyValue > 0)
58 m_gradientUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
59 return;
60 }
61
62 if (name == SVGNames::gradientTransformAttr) {
63 m_gradientTransform->baseVal()->parse(value);
64 return;
65 }
66
67 if (name == SVGNames::spreadMethodAttr) {
68 auto propertyValue = SVGPropertyTraits<SVGSpreadMethodType>::fromString(value);
69 if (propertyValue > 0)
70 m_spreadMethod->setBaseValInternal<SVGSpreadMethodType>(propertyValue);
71 return;
72 }
73
74 SVGElement::parseAttribute(name, value);
75 SVGURIReference::parseAttribute(name, value);
76}

78void SVGGradientElement::svgAttributeChanged(const QualifiedName& attrName)
79{
80 if (PropertyRegistry::isKnownAttribute(attrName) || SVGURIReference::isKnownAttribute(attrName)) {
81 InstanceInvalidationGuard guard(*this);
82 if (RenderObject* object = renderer())
83 object->setNeedsLayout();
84 return;
85 }
86
87 SVGElement::svgAttributeChanged(attrName);
88}

101Gradient::ColorStopVector SVGGradientElement::buildStops()
102{
103 Gradient::ColorStopVector stops;
104 float previousOffset = 0.0f;
105 for (auto& stop : childrenOfType<SVGStopElement>(*this)) {
106 auto monotonicallyIncreasingOffset = std::clamp(stop.offset(), previousOffset, 1.0f);
107 previousOffset = monotonicallyIncreasingOffset;
108 stops.append({ monotonicallyIncreasingOffset, stop.stopColorIncludingOpacity() });
109 }
110 return stops;
111}
