39void SVGImageLoader::dispatchLoadEvent()
40{
41 if (image()->errorOccurred())
42 element().dispatchEvent(Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::No));
43 else
44 downcast<SVGImageElement>(ImageLoader::element()).sendLoadEventIfPossible();
45}

47String SVGImageLoader::sourceURI(const AtomString& attribute) const
48{
49 URL base = element().baseURI();
50 if (base != aboutBlankURL())
51 return URL(base, stripLeadingAndTrailingHTMLSpaces(attribute)).string();
52 return element().document().completeURL(stripLeadingAndTrailingHTMLSpaces(attribute)).string();
53}
