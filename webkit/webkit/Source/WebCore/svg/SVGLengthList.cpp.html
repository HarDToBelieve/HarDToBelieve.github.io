34bool SVGLengthList::parse(StringView value)
35{
36 clearItems();
37
38 return readCharactersForParsing(value, [&](auto buffer) {
39 skipOptionalSVGSpaces(buffer);
40
41 while (buffer.hasCharactersRemaining()) {
42 auto start = buffer.position();
43
44 skipUntil<isSVGSpaceOrComma>(buffer);
45
46 if (buffer.position() == start)
47 break;
48
49 auto value = SVGLengthValue::construct(m_lengthMode, StringView(start, buffer.position() - start));
50 if (!value)
51 break;
52
53 append(SVGLength::create(WTFMove(*value)));
54 skipOptionalSVGSpacesOrDelimiter(buffer);
55 }
56
57 // FIXME: Should this clearItems() on failure like SVGTransformList does?
58
59 return buffer.atEnd();
60 });
61}
62
63String SVGLengthList::valueAsString() const
64{
65 StringBuilder builder;
66
67 for (const auto& length : m_items) {
68 if (builder.length())
69 builder.append(' ');
70
71 builder.append(length->value().valueAsString());
72 }
73
74 return builder.toString();
75}
