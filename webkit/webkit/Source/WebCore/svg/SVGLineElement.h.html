24#include "SVGGeometryElement.h"
25#include "SVGNames.h"

27namespace WebCore {
28
29class SVGLineElement final : public SVGGeometryElement {
30 WTF_MAKE_ISO_ALLOCATED(SVGLineElement);
31public:
32 static Ref<SVGLineElement> create(const QualifiedName&, Document&);
33
34 const SVGLengthValue& x1() const { return m_x1->currentValue(); }
35 const SVGLengthValue& y1() const { return m_y1->currentValue(); }
36 const SVGLengthValue& x2() const { return m_x2->currentValue(); }
37 const SVGLengthValue& y2() const { return m_y2->currentValue(); }
38
39 SVGAnimatedLength& x1Animated() { return m_x1; }
40 SVGAnimatedLength& y1Animated() { return m_y1; }
41 SVGAnimatedLength& x2Animated() { return m_x2; }
42 SVGAnimatedLength& y2Animated() { return m_y2; }
43
44private:
45 SVGLineElement(const QualifiedName&, Document&);
46
47 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGLineElement, SVGGeometryElement>;
48 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
49
50 void parseAttribute(const QualifiedName&, const AtomString&) final;
51 void svgAttributeChanged(const QualifiedName&) final;
52
53 bool isValid() const final { return SVGTests::isValid(); }
54 bool supportsMarkers() const final { return true; }
55 bool selfHasRelativeLengths() const final;
56
57 PropertyRegistry m_propertyRegistry { *this };
58 Ref<SVGAnimatedLength> m_x1 { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
59 Ref<SVGAnimatedLength> m_y1 { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
60 Ref<SVGAnimatedLength> m_x2 { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
61 Ref<SVGAnimatedLength> m_y2 { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
62};
