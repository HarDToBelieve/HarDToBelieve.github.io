34inline SVGMarkerElement::SVGMarkerElement(const QualifiedName& tagName, Document& document)
35 : SVGElement(tagName, document)
36 , SVGFitToViewBox(this)
37{
38 // Spec: If the markerWidth/markerHeight attribute is not specified, the effect is as if a value of "3" were specified.
39 ASSERT(hasTagName(SVGNames::markerTag));
40
41 static std::once_flag onceFlag;
42 std::call_once(onceFlag, [] {
43 PropertyRegistry::registerProperty<SVGNames::refXAttr, &SVGMarkerElement::m_refX>();
44 PropertyRegistry::registerProperty<SVGNames::refYAttr, &SVGMarkerElement::m_refY>();
45 PropertyRegistry::registerProperty<SVGNames::markerWidthAttr, &SVGMarkerElement::m_markerWidth>();
46 PropertyRegistry::registerProperty<SVGNames::markerHeightAttr, &SVGMarkerElement::m_markerHeight>();
47 PropertyRegistry::registerProperty<SVGNames::markerUnitsAttr, SVGMarkerUnitsType, &SVGMarkerElement::m_markerUnits>();
48 PropertyRegistry::registerProperty<SVGNames::orientAttr, &SVGMarkerElement::m_orientAngle, &SVGMarkerElement::m_orientType>();
49 });
50}

62void SVGMarkerElement::parseAttribute(const QualifiedName& name, const AtomString& value)
63{
64 if (name == SVGNames::markerUnitsAttr) {
65 auto propertyValue = SVGPropertyTraits<SVGMarkerUnitsType>::fromString(value);
66 if (propertyValue > 0)
67 m_markerUnits->setBaseValInternal<SVGMarkerUnitsType>(propertyValue);
68 return;
69 }
70
71 if (name == SVGNames::orientAttr) {
72 auto pair = SVGPropertyTraits<std::pair<SVGAngleValue, SVGMarkerOrientType>>::fromString(value);
73 m_orientAngle->setBaseValInternal(pair.first);
74 m_orientType->setBaseValInternal(pair.second);
75 return;
76 }
77
78 SVGParsingError parseError = NoError;
79
80 if (name == SVGNames::refXAttr)
81 m_refX->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
82 else if (name == SVGNames::refYAttr)
83 m_refY->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
84 else if (name == SVGNames::markerWidthAttr)
85 m_markerWidth->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
86 else if (name == SVGNames::markerHeightAttr)
87 m_markerHeight->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
88
89 reportAttributeParsingError(parseError, name, value);
90
91 SVGElement::parseAttribute(name, value);
92 SVGFitToViewBox::parseAttribute(name, value);
93}

95void SVGMarkerElement::svgAttributeChanged(const QualifiedName& attrName)
96{
97 if (PropertyRegistry::isKnownAttribute(attrName)) {
98 InstanceInvalidationGuard guard(*this);
99 if (PropertyRegistry::isAnimatedLengthAttribute(attrName))
100 updateRelativeLengthsInformation();
101 if (RenderObject* object = renderer())
102 object->setNeedsLayout();
103 return;
104 }
105
106 if (SVGFitToViewBox::isKnownAttribute(attrName)) {
107 if (RenderObject* object = renderer())
108 object->setNeedsLayout();
109 return;
110 }
111
112 SVGElement::svgAttributeChanged(attrName);
113}

151bool SVGMarkerElement::selfHasRelativeLengths() const
152{
153 return refX().isRelative()
154 || refY().isRelative()
155 || markerWidth().isRelative()
156 || markerHeight().isRelative();
