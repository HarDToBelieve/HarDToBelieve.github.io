22
23#include "SVGElement.h"
24#include "SVGNames.h"
25#include "SVGTests.h"
26#include "SVGUnitTypes.h"

28namespace WebCore {
29
30class SVGMaskElement final : public SVGElement, public SVGTests {
31 WTF_MAKE_ISO_ALLOCATED(SVGMaskElement);
32public:
33 static Ref<SVGMaskElement> create(const QualifiedName&, Document&);
34
35 const SVGLengthValue& x() const { return m_x->currentValue(); }
36 const SVGLengthValue& y() const { return m_y->currentValue(); }
37 const SVGLengthValue& width() const { return m_width->currentValue(); }
38 const SVGLengthValue& height() const { return m_height->currentValue(); }
39 SVGUnitTypes::SVGUnitType maskUnits() const { return m_maskUnits->currentValue<SVGUnitTypes::SVGUnitType>(); }
40 SVGUnitTypes::SVGUnitType maskContentUnits() const { return m_maskContentUnits->currentValue<SVGUnitTypes::SVGUnitType>(); }
41
42 SVGAnimatedLength& xAnimated() { return m_x; }
43 SVGAnimatedLength& yAnimated() { return m_y; }
44 SVGAnimatedLength& widthAnimated() { return m_width; }
45 SVGAnimatedLength& heightAnimated() { return m_height; }
46 SVGAnimatedEnumeration& maskUnitsAnimated() { return m_maskUnits; }
47 SVGAnimatedEnumeration& maskContentUnitsAnimated() { return m_maskContentUnits; }
48
49private:
50 SVGMaskElement(const QualifiedName&, Document&);
51
52 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGMaskElement, SVGElement, SVGTests>;
53 const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
54
55 void parseAttribute(const QualifiedName&, const AtomString&) final;
56 void svgAttributeChanged(const QualifiedName&) final;
57 void childrenChanged(const ChildChange&) final;
58
59 RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
60
61 bool isValid() const final { return SVGTests::isValid(); }
62 bool needsPendingResourceHandling() const final { return false; }
63 bool selfHasRelativeLengths() const final { return true; }
64
65 PropertyRegistry m_propertyRegistry { *this };
66 Ref<SVGAnimatedLength> m_x { SVGAnimatedLength::create(this, SVGLengthMode::Width, "-10%") };
67 Ref<SVGAnimatedLength> m_y { SVGAnimatedLength::create(this, SVGLengthMode::Height, "-10%") };
68 Ref<SVGAnimatedLength> m_width { SVGAnimatedLength::create(this, SVGLengthMode::Width, "120%") };
69 Ref<SVGAnimatedLength> m_height { SVGAnimatedLength::create(this, SVGLengthMode::Height, "120%") };
70 Ref<SVGAnimatedEnumeration> m_maskUnits { SVGAnimatedEnumeration::create(this, SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) };
71 Ref<SVGAnimatedEnumeration> m_maskContentUnits { SVGAnimatedEnumeration::create(this, SVGUnitTypes::SVG_UNIT_TYPE_USERSPACEONUSE) };
72};
