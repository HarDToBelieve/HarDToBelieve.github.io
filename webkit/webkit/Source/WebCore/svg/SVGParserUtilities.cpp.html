35template <typename FloatType> static inline bool isValidRange(const FloatType& x)
36{
37 static const FloatType max = std::numeric_limits<FloatType>::max();
38 return x >= -max && x <= max;
39}

41// We use this generic parseNumber function to allow the Path parsing code to work
42// at a higher precision internally, without any unnecessary runtime cost or code
43// complexity.
44// FIXME: Can this be shared/replaced with number parsing in WTF?
45template <typename CharacterType, typename FloatType = float> static std::optional<FloatType> genericParseNumber(StringParsingBuffer<CharacterType>& buffer, SuffixSkippingPolicy skip = SuffixSkippingPolicy::Skip)
46{
47 FloatType number = 0;
48 FloatType integer = 0;
49 FloatType decimal = 0;
50 FloatType frac = 1;
51 FloatType exponent = 0;
52 int sign = 1;
53 int expsign = 1;
54 auto start = buffer.position();
55
56 // read the sign
57 if (buffer.hasCharactersRemaining() && *buffer == '+')
58 ++buffer;
59 else if (buffer.hasCharactersRemaining() && *buffer == '-') {
60 ++buffer;
61 sign = -1;
62 }
63
64 if (buffer.atEnd() || (!isASCIIDigit(*buffer) && *buffer != '.'))
65 return std::nullopt;
66
67 // read the integer part, build right-to-left
68 auto ptrStartIntPart = buffer.position();
69
70 // Advance to first non-digit.
71 skipWhile<isASCIIDigit>(buffer);
72
73 if (buffer.position() != ptrStartIntPart) {
74 auto ptrScanIntPart = buffer.position() - 1;
75 FloatType multiplier = 1;
76 while (ptrScanIntPart >= ptrStartIntPart) {
77 integer += multiplier * static_cast<FloatType>(*(ptrScanIntPart--) - '0');
78 multiplier *= 10;
79 }
80 // Bail out early if this overflows.
81 if (!isValidRange(integer))
82 return std::nullopt;
83 }
84
85 // read the decimals
86 if (buffer.hasCharactersRemaining() && *buffer == '.') {
87 ++buffer;
88
89 // There must be a least one digit following the .
90 if (buffer.atEnd() || !isASCIIDigit(*buffer))
91 return std::nullopt;
92
93 while (buffer.hasCharactersRemaining() && isASCIIDigit(*buffer))
94 decimal += (*(buffer++) - '0') * (frac *= static_cast<FloatType>(0.1));
95 }
96
97 // read the exponent part
98 if (buffer.position() != start && buffer.position() + 1 < buffer.end() && (*buffer == 'e' || *buffer == 'E')
99 && (buffer[1] != 'x' && buffer[1] != 'm')) {
100 ++buffer;
101
102 // read the sign of the exponent
103 if (*buffer == '+')
104 ++buffer;
105 else if (*buffer == '-') {
106 ++buffer;
107 expsign = -1;
108 }
109
110 // There must be an exponent
111 if (buffer.atEnd() || !isASCIIDigit(*buffer))
112 return std::nullopt;
113
114 while (buffer.hasCharactersRemaining() && isASCIIDigit(*buffer)) {
115 exponent *= static_cast<FloatType>(10);
116 exponent += *buffer++ - '0';
117 }
118 // Make sure exponent is valid.
119 if (!isValidRange(exponent) || exponent > std::numeric_limits<FloatType>::max_exponent)
120 return std::nullopt;
121 }
122
123 number = integer + decimal;
124 number *= sign;
125
126 if (exponent)
127 number *= static_cast<FloatType>(pow(10.0, expsign * static_cast<int>(exponent)));
128
129 // Don't return Infinity() or NaN().
130 if (!isValidRange(number))
131 return std::nullopt;
132
133 if (start == buffer.position())
134 return std::nullopt;
135
136 if (skip == SuffixSkippingPolicy::Skip)
137 skipOptionalSVGSpacesOrDelimiter(buffer);
138
139 return number;
140}

143{
144 return genericParseNumber(buffer, skip);

152std::optional<float> parseNumber(const StringView& string, SuffixSkippingPolicy skip)
153{
154 return readCharactersForParsing(string, [skip](auto buffer) -> std::optional<float> {
155 auto result = genericParseNumber(buffer, skip);
156 if (!buffer.atEnd())
157 return std::nullopt;
158 return result;
159 });
160}

162// only used to parse largeArcFlag and sweepFlag which must be a "0" or "1"
163// and might not have any whitespace/comma after it
164template <typename CharacterType> std::optional<bool> genericParseArcFlag(StringParsingBuffer<CharacterType>& buffer)
165{
166 if (buffer.atEnd())
167 return std::nullopt;
168
169 const CharacterType flagChar = *buffer;
170 ++buffer;
171
172 bool flag;
173 if (flagChar == '0')
174 flag = false;
175 else if (flagChar == '1')
176 flag = true;
177 else
178 return std::nullopt;
179
180 skipOptionalSVGSpacesOrDelimiter(buffer);
181
182 return flag;
183}

195std::optional<std::pair<float, float>> parseNumberOptionalNumber(const StringView& string)
196{
197 if (string.isEmpty())
198 return std::nullopt;
199
200 return readCharactersForParsing(string, [](auto buffer) -> std::optional<std::pair<float, float>> {
201 auto x = parseNumber(buffer);
202 if (!x)
203 return std::nullopt;
204
205 if (buffer.atEnd())
206 return std::make_pair(*x, *x);
207
208 auto y = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
209 if (!y)
210 return std::nullopt;
211
212 if (!buffer.atEnd())
213 return std::nullopt;
214
215 return std::make_pair(*x, *y);
216 });
217}

219std::optional<FloatPoint> parsePoint(const StringView& string)
220{
221 if (string.isEmpty())
222 return std::nullopt;
223
224 return readCharactersForParsing(string, [](auto buffer) -> std::optional<FloatPoint> {
225 if (!skipOptionalSVGSpaces(buffer))
226 return std::nullopt;
227
228 auto point = parseFloatPoint(buffer);
229 if (!point)
230 return std::nullopt;
231
232 // Disallow anything except spaces at the end.
233 skipOptionalSVGSpaces(buffer);
234
235 return point;
236 });
237}

239std::optional<FloatRect> parseRect(const StringView& string)
240{
241 return readCharactersForParsing(string, [](auto buffer) -> std::optional<FloatRect> {
242 skipOptionalSVGSpaces(buffer);
243
244 auto x = parseNumber(buffer);
245 if (!x)
246 return std::nullopt;
247 auto y = parseNumber(buffer);
248 if (!y)
249 return std::nullopt;
250 auto width = parseNumber(buffer);
251 if (!width)
252 return std::nullopt;
253 auto height = parseNumber(buffer, SuffixSkippingPolicy::DontSkip);
254 if (!height)
255 return std::nullopt;
256
257 return FloatRect { *x, *y, *width, *height };
258 });
259}

261std::optional<HashSet<String>> parseGlyphName(const StringView& string)
262{
263 // FIXME: Parsing error detection is missing.
264
265 return readCharactersForParsing(string, [](auto buffer) -> HashSet<String> {
266 skipOptionalSVGSpaces(buffer);
267
268 HashSet<String> values;
269
270 while (buffer.hasCharactersRemaining()) {
271 // Leading and trailing white space, and white space before and after separators, will be ignored.
272 auto inputStart = buffer.position();
273
274 skipUntil(buffer, ',');
275
276 if (buffer.position() == inputStart)
277 break;
278
279 // walk backwards from the ; to ignore any whitespace
280 auto inputEnd = buffer.position() - 1;
281 while (inputStart < inputEnd && isSVGSpace(*inputEnd))
282 --inputEnd;
283
284 values.add(String(inputStart, inputEnd - inputStart + 1));
285 skipOptionalSVGSpacesOrDelimiter(buffer, ',');
286 }
287 return values;
288 });

292template<typename CharacterType> static std::optional<UnicodeRange> parseUnicodeRange(StringParsingBuffer<CharacterType> buffer)
293{
294 unsigned length = buffer.lengthRemaining();
295 if (length < 2 || buffer[0] != 'U' || buffer[1] != '+')
296 return std::nullopt;
297
298 buffer += 2;
299
300 // Parse the starting hex number (or its prefix).
301 unsigned startRange = 0;
302 unsigned startLength = 0;
303
304 while (buffer.hasCharactersRemaining()) {
305 if (!isASCIIHexDigit(*buffer))
306 break;
307 ++startLength;
308 if (startLength > 6)
309 return std::nullopt;
310 startRange = (startRange << 4) | toASCIIHexValue(*buffer);
311 ++buffer;
312 }
313
314 // Handle the case of ranges separated by "-" sign.
315 if (2 + startLength < length && *buffer == '-') {
316 if (!startLength)
317 return std::nullopt;
318
319 // Parse the ending hex number (or its prefix).
320 unsigned endRange = 0;
321 unsigned endLength = 0;
322 ++buffer;
323 while (buffer.hasCharactersRemaining()) {
324 if (!isASCIIHexDigit(*buffer))
325 break;
326 ++endLength;
327 if (endLength > 6)
328 return std::nullopt;
329 endRange = (endRange << 4) | toASCIIHexValue(*buffer);
330 ++buffer;
331 }
332
333 if (!endLength)
334 return std::nullopt;
335
336 UnicodeRange range;
337 range.first = startRange;
338 range.second = endRange;
339 return range;
340 }
341
342 // Handle the case of a number with some optional trailing question marks.
343 unsigned endRange = startRange;
344 while (buffer.hasCharactersRemaining()) {
345 if (*buffer != '?')
346 break;
347 ++startLength;
348 if (startLength > 6)
349 return std::nullopt;
350 startRange <<= 4;
351 endRange = (endRange << 4) | 0xF;
352 ++buffer;
353 }
354
355 if (!startLength)
356 return std::nullopt;
357
358 UnicodeRange range;
359 range.first = startRange;
360 range.second = endRange;
361 return range;
362}

364std::optional<std::pair<UnicodeRanges, HashSet<String>>> parseKerningUnicodeString(const StringView& string)
365{
366 // FIXME: Parsing error detection is missing.
367
368 return readCharactersForParsing(string, [](auto buffer) -> std::pair<UnicodeRanges, HashSet<String>> {
369 UnicodeRanges rangeList;
370 HashSet<String> stringList;
371
372 while (1) {
373 auto inputStart = buffer.position();
374
375 skipUntil(buffer, ',');
376
377 if (buffer.position() == inputStart)
378 break;
379
380 // Try to parse unicode range first
381 if (auto range = parseUnicodeRange(StringParsingBuffer { inputStart, buffer.position() }))
382 rangeList.append(WTFMove(*range));
383 else
384 stringList.add(String(inputStart, buffer.position() - inputStart));
385
386 if (buffer.atEnd())
387 break;
388
389 ++buffer;
390 }
391
392 return std::make_pair(WTFMove(rangeList), WTFMove(stringList));
393 });
394}

396template <typename CharacterType> static std::optional<FloatPoint> genericParseFloatPoint(StringParsingBuffer<CharacterType>& buffer)
397{
398 auto x = parseNumber(buffer);
399 if (!x)
400 return std::nullopt;
401
402 auto y = parseNumber(buffer);
403 if (!y)
404 return std::nullopt;
405
406 return FloatPoint { *x, *y };
407}
