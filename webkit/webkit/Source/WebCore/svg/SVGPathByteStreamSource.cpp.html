26SVGPathByteStreamSource::SVGPathByteStreamSource(const SVGPathByteStream& stream)
27{
28 m_streamCurrent = stream.begin();
29 m_streamEnd = stream.end();
30}
31
32bool SVGPathByteStreamSource::hasMoreData() const
33{
34 return m_streamCurrent < m_streamEnd;
35}
