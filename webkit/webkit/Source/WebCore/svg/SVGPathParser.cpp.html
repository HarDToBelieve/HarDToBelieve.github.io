37bool SVGPathParser::parse(SVGPathSource& source, SVGPathConsumer& consumer, PathParsingMode mode, bool checkForInitialMoveTo)
38{
39 SVGPathParser parser(consumer, source, mode);
40 return parser.parsePathData(checkForInitialMoveTo);

43bool SVGPathParser::parseToByteStream(SVGPathSource& source, SVGPathByteStream& byteStream, PathParsingMode mode, bool checkForInitialMoveTo)
44{
45 SVGPathByteStreamBuilder builder(byteStream);
46 auto result = parse(source, builder, mode, checkForInitialMoveTo);

51bool SVGPathParser::parseToString(SVGPathSource& source, String& result, PathParsingMode mode, bool checkForInitialMoveTo)
52{
53 SVGPathStringBuilder builder;
54 SVGPathParser parser(builder, source, mode);
55 bool ok = parser.parsePathData(checkForInitialMoveTo);
56 result = builder.result();

67void SVGPathParser::parseClosePathSegment()
68{
69 // Reset m_currentPoint for the next path.
70 if (m_pathParsingMode == NormalizedParsing)
71 m_currentPoint = m_subPathPoint;
72 m_closePath = true;
73 m_consumer.closePath();
74}

76bool SVGPathParser::parseMoveToSegment()
77{
78 auto result = m_source.parseMoveToSegment();
79 if (!result)
80 return false;
81
82 if (m_pathParsingMode == NormalizedParsing) {
83 if (m_mode == RelativeCoordinates)
84 m_currentPoint += result->targetPoint;
85 else
86 m_currentPoint = result->targetPoint;
87 m_subPathPoint = m_currentPoint;
88 m_consumer.moveTo(m_currentPoint, m_closePath, AbsoluteCoordinates);
89 } else
90 m_consumer.moveTo(result->targetPoint, m_closePath, m_mode);
91 m_closePath = false;
92 return true;
93}

146bool SVGPathParser::parseCurveToCubicSegment()
147{
148 auto result = m_source.parseCurveToCubicSegment();
149 if (!result)
150 return false;
151
152 if (m_pathParsingMode == NormalizedParsing) {
153 if (m_mode == RelativeCoordinates) {
154 result->point1 += m_currentPoint;
155 result->point2 += m_currentPoint;
156 result->targetPoint += m_currentPoint;
157 }
158 m_consumer.curveToCubic(result->point1, result->point2, result->targetPoint, AbsoluteCoordinates);
159
160 m_controlPoint = result->point2;
161 m_currentPoint = result->targetPoint;
162 } else
163 m_consumer.curveToCubic(result->point1, result->point2, result->targetPoint, m_mode);
164 return true;
165}

167bool SVGPathParser::parseCurveToCubicSmoothSegment()
168{
169 auto result = m_source.parseCurveToCubicSmoothSegment();
170 if (!result)
171 return false;
172
173 if (m_lastCommand != PathSegCurveToCubicAbs
174 && m_lastCommand != PathSegCurveToCubicRel
175 && m_lastCommand != PathSegCurveToCubicSmoothAbs
176 && m_lastCommand != PathSegCurveToCubicSmoothRel)
177 m_controlPoint = m_currentPoint;
178
179 if (m_pathParsingMode == NormalizedParsing) {
180 FloatPoint point1 = m_currentPoint;
181 point1.scale(2);
182 point1.move(-m_controlPoint.x(), -m_controlPoint.y());
183 if (m_mode == RelativeCoordinates) {
184 result->point2 += m_currentPoint;
185 result->targetPoint += m_currentPoint;
186 }
187
188 m_consumer.curveToCubic(point1, result->point2, result->targetPoint, AbsoluteCoordinates);
189
190 m_controlPoint = result->point2;
191 m_currentPoint = result->targetPoint;
192 } else
193 m_consumer.curveToCubicSmooth(result->point2, result->targetPoint, m_mode);
194 return true;
195}

197bool SVGPathParser::parseCurveToQuadraticSegment()
198{
199 auto result = m_source.parseCurveToQuadraticSegment();
200 if (!result)
201 return false;
202
203 if (m_pathParsingMode == NormalizedParsing) {
204 m_controlPoint = result->point1;
205
206 FloatPoint point1 = m_currentPoint;
207 point1.move(2 * m_controlPoint.x(), 2 * m_controlPoint.y());
208 FloatPoint point2(result->targetPoint.x() + 2 * m_controlPoint.x(), result->targetPoint.y() + 2 * m_controlPoint.y());
209 if (m_mode == RelativeCoordinates) {
210 point1.move(2 * m_currentPoint.x(), 2 * m_currentPoint.y());
211 point2.move(3 * m_currentPoint.x(), 3 * m_currentPoint.y());
212 result->targetPoint += m_currentPoint;
213 }
214 point1.scale(gOneOverThree);
215 point2.scale(gOneOverThree);
216
217 m_consumer.curveToCubic(point1, point2, result->targetPoint, AbsoluteCoordinates);
218
219 if (m_mode == RelativeCoordinates)
220 m_controlPoint += m_currentPoint;
221 m_currentPoint = result->targetPoint;
222 } else
223 m_consumer.curveToQuadratic(result->point1, result->targetPoint, m_mode);
224 return true;
225}

245 if (m_mode == RelativeCoordinates) {

261bool SVGPathParser::parseArcToSegment()
262{
263 auto result = m_source.parseArcToSegment();
264 if (!result)
265 return false;
266
267 // If rx = 0 or ry = 0 then this arc is treated as a straight line segment (a "lineto") joining the endpoints.
268 // http://www.w3.org/TR/SVG/implnote.html#ArcOutOfRangeParameters
269 // If the current point and target point for the arc are identical, it should be treated as a zero length
270 // path. This ensures continuity in animations.
271 result->rx = std::abs(result->rx);
272 result->ry = std::abs(result->ry);
273 bool arcIsZeroLength = false;
274 if (m_pathParsingMode == NormalizedParsing) {
275 if (m_mode == RelativeCoordinates)
276 arcIsZeroLength = result->targetPoint == FloatPoint::zero();
277 else
278 arcIsZeroLength = result->targetPoint == m_currentPoint;
279 }
280 if (!result->rx || !result->ry || arcIsZeroLength) {
281 if (m_pathParsingMode == NormalizedParsing) {
282 if (m_mode == RelativeCoordinates)
283 m_currentPoint += result->targetPoint;
284 else
285 m_currentPoint = result->targetPoint;
286 m_consumer.lineTo(m_currentPoint, AbsoluteCoordinates);
287 } else
288 m_consumer.lineTo(result->targetPoint, m_mode);
289 return true;
290 }
291
292 if (m_pathParsingMode == NormalizedParsing) {
293 FloatPoint point1 = m_currentPoint;
294 if (m_mode == RelativeCoordinates)
295 result->targetPoint += m_currentPoint;
296 m_currentPoint = result->targetPoint;
297 return decomposeArcToCubic(result->angle, result->rx, result->ry, point1, result->targetPoint, result->largeArc, result->sweep);
298 }
299 m_consumer.arcTo(result->rx, result->ry, result->angle, result->largeArc, result->sweep, result->targetPoint, m_mode);
300 return true;
301}

303bool SVGPathParser::parsePathData(bool checkForInitialMoveTo)
304{
305 // Skip any leading spaces.
306 if (!m_source.moveToNextToken())
307 return true;
308
309 auto parsedCommand = m_source.parseSVGSegmentType();
310 if (!parsedCommand)
311 return false;
312
313 auto command = *parsedCommand;
314
315 // Path must start with moveto.
316 if (checkForInitialMoveTo && command != PathSegMoveToAbs && command != PathSegMoveToRel)
317 return false;
318
319 while (true) {
320 // Skip spaces between command and first coordinate.
321 m_source.moveToNextToken();
322 m_mode = AbsoluteCoordinates;
323 switch (command) {
324 case PathSegMoveToRel:
325 m_mode = RelativeCoordinates;
326 FALLTHROUGH;
327 case PathSegMoveToAbs:
328 if (!parseMoveToSegment())
329 return false;
330 break;
331 case PathSegLineToRel:
332 m_mode = RelativeCoordinates;
333 FALLTHROUGH;
334 case PathSegLineToAbs:
335 if (!parseLineToSegment())
336 return false;
337 break;
338 case PathSegLineToHorizontalRel:
339 m_mode = RelativeCoordinates;
340 FALLTHROUGH;
341 case PathSegLineToHorizontalAbs:
342 if (!parseLineToHorizontalSegment())
343 return false;
344 break;
345 case PathSegLineToVerticalRel:
346 m_mode = RelativeCoordinates;
347 FALLTHROUGH;
348 case PathSegLineToVerticalAbs:
349 if (!parseLineToVerticalSegment())
350 return false;
351 break;
352 case PathSegClosePath:
353 parseClosePathSegment();
354 break;
355 case PathSegCurveToCubicRel:
356 m_mode = RelativeCoordinates;
357 FALLTHROUGH;
358 case PathSegCurveToCubicAbs:
359 if (!parseCurveToCubicSegment())
360 return false;
361 break;
362 case PathSegCurveToCubicSmoothRel:
363 m_mode = RelativeCoordinates;
364 FALLTHROUGH;
365 case PathSegCurveToCubicSmoothAbs:
366 if (!parseCurveToCubicSmoothSegment())
367 return false;
368 break;
369 case PathSegCurveToQuadraticRel:
370 m_mode = RelativeCoordinates;
371 FALLTHROUGH;
372 case PathSegCurveToQuadraticAbs:
373 if (!parseCurveToQuadraticSegment())
374 return false;
375 break;
376 case PathSegCurveToQuadraticSmoothRel:
377 m_mode = RelativeCoordinates;
378 FALLTHROUGH;
379 case PathSegCurveToQuadraticSmoothAbs:
380 if (!parseCurveToQuadraticSmoothSegment())
381 return false;
382 break;
383 case PathSegArcRel:
384 m_mode = RelativeCoordinates;
385 FALLTHROUGH;
386 case PathSegArcAbs:
387 if (!parseArcToSegment())
388 return false;
389 break;
390 default:
391 return false;
392 }
393 if (!m_consumer.continueConsuming())
394 return true;
395
396 m_lastCommand = command;
397
398 if (!m_source.hasMoreData())
399 return true;
400
401 command = m_source.nextCommand(command);
402
403 if (m_lastCommand != PathSegCurveToCubicAbs
404 && m_lastCommand != PathSegCurveToCubicRel
405 && m_lastCommand != PathSegCurveToCubicSmoothAbs
406 && m_lastCommand != PathSegCurveToCubicSmoothRel
407 && m_lastCommand != PathSegCurveToQuadraticAbs
408 && m_lastCommand != PathSegCurveToQuadraticRel
409 && m_lastCommand != PathSegCurveToQuadraticSmoothAbs
410 && m_lastCommand != PathSegCurveToQuadraticSmoothRel)
411 m_controlPoint = m_currentPoint;
412
413 m_consumer.incrementPathSegmentCount();
414 }
415
416 return false;
417}

419// This works by converting the SVG arc to "simple" beziers.
420// Partly adapted from Niko's code in kdelibs/kdecore/svgicons.
421// See also SVG implementation notes: http://www.w3.org/TR/SVG/implnote.html#ArcConversionEndpointToCenter
422bool SVGPathParser::decomposeArcToCubic(float angle, float rx, float ry, FloatPoint& point1, FloatPoint& point2, bool largeArcFlag, bool sweepFlag)
423{
424 FloatSize midPointDistance = point1 - point2;
425 midPointDistance.scale(0.5f);
426
427 AffineTransform pointTransform;
428 pointTransform.rotate(-angle);
429
430 FloatPoint transformedMidPoint = pointTransform.mapPoint(FloatPoint(midPointDistance.width(), midPointDistance.height()));
431 float squareRx = rx * rx;
432 float squareRy = ry * ry;
433 float squareX = transformedMidPoint.x() * transformedMidPoint.x();
434 float squareY = transformedMidPoint.y() * transformedMidPoint.y();
435
436 // Check if the radii are big enough to draw the arc, scale radii if not.
437 // http://www.w3.org/TR/SVG/implnote.html#ArcCorrectionOutOfRangeRadii
438 float radiiScale = squareX / squareRx + squareY / squareRy;
439 if (radiiScale > 1) {
440 rx *= sqrtf(radiiScale);
441 ry *= sqrtf(radiiScale);
442 }
443
444 pointTransform.makeIdentity();
445 pointTransform.scale(1 / rx, 1 / ry);
446 pointTransform.rotate(-angle);
447
448 point1 = pointTransform.mapPoint(point1);
449 point2 = pointTransform.mapPoint(point2);
450 FloatSize delta = point2 - point1;
451
452 float d = delta.width() * delta.width() + delta.height() * delta.height();
453 float scaleFactorSquared = std::max(1 / d - 0.25f, 0.f);
454
455 float scaleFactor = sqrtf(scaleFactorSquared);
456 if (sweepFlag == largeArcFlag)
457 scaleFactor = -scaleFactor;
458
459 delta.scale(scaleFactor);
460 FloatPoint centerPoint = point1 + point2;
461 centerPoint.scale(0.5f);
462 centerPoint.move(-delta.height(), delta.width());
463
464 float theta1 = FloatPoint(point1 - centerPoint).slopeAngleRadians();
465 float theta2 = FloatPoint(point2 - centerPoint).slopeAngleRadians();
466
467 float thetaArc = theta2 - theta1;
468 if (thetaArc < 0 && sweepFlag)
469 thetaArc += 2 * piFloat;
470 else if (thetaArc > 0 && !sweepFlag)
471 thetaArc -= 2 * piFloat;
472
473 pointTransform.makeIdentity();
474 pointTransform.rotate(angle);
475 pointTransform.scale(rx, ry);
476
477 // Some results of atan2 on some platform implementations are not exact enough. So that we get more
478 // cubic curves than expected here. Adding 0.001f reduces the count of sgements to the correct count.
479 int segments = ceilf(fabsf(thetaArc / (piOverTwoFloat + 0.001f)));
480 for (int i = 0; i < segments; ++i) {
481 float startTheta = theta1 + i * thetaArc / segments;
482 float endTheta = theta1 + (i + 1) * thetaArc / segments;
483
484 float t = (8 / 6.f) * tanf(0.25f * (endTheta - startTheta));
485 if (!std::isfinite(t))
486 return false;
487 float sinStartTheta = sinf(startTheta);
488 float cosStartTheta = cosf(startTheta);
489 float sinEndTheta = sinf(endTheta);
490 float cosEndTheta = cosf(endTheta);
491
492 point1 = FloatPoint(cosStartTheta - t * sinStartTheta, sinStartTheta + t * cosStartTheta);
493 point1.move(centerPoint.x(), centerPoint.y());
494 FloatPoint targetPoint = FloatPoint(cosEndTheta, sinEndTheta);
495 targetPoint.move(centerPoint.x(), centerPoint.y());
496 point2 = targetPoint;
497 point2.move(t * sinEndTheta, -t * cosEndTheta);
498
499 m_consumer.curveToCubic(pointTransform.mapPoint(point1), pointTransform.mapPoint(point2),
500 pointTransform.mapPoint(targetPoint), AbsoluteCoordinates);
501 }
502 return true;
503}
