31SVGPathSegListSource::SVGPathSegListSource(const SVGPathSegList& pathSegList)
32 : m_pathSegList(pathSegList)
33{
34 m_itemCurrent = 0;
35 m_itemEnd = m_pathSegList.size();
36}
37
38bool SVGPathSegListSource::hasMoreData() const
39{
40 return m_itemCurrent < m_itemEnd;
41}
42
43SVGPathSegType SVGPathSegListSource::nextCommand(SVGPathSegType)
44{
45 m_segment = m_pathSegList.at(m_itemCurrent);
46 SVGPathSegType pathSegType = static_cast<SVGPathSegType>(m_segment->pathSegType());
47 ++m_itemCurrent;
48 return pathSegType;
49}
50
51std::optional<SVGPathSegType> SVGPathSegListSource::parseSVGSegmentType()
52{
53 m_segment = m_pathSegList.at(m_itemCurrent);
54 SVGPathSegType pathSegType = static_cast<SVGPathSegType>(m_segment->pathSegType());
55 ++m_itemCurrent;
56 return pathSegType;
57}

59std::optional<SVGPathSource::MoveToSegment> SVGPathSegListSource::parseMoveToSegment()
60{
61 ASSERT(m_segment);
62 ASSERT(m_segment->pathSegType() == PathSegMoveToAbs || m_segment->pathSegType() == PathSegMoveToRel);
63 SVGPathSegSingleCoordinate* moveTo = static_cast<SVGPathSegSingleCoordinate*>(m_segment.get());
64
65 MoveToSegment segment;
66 segment.targetPoint = FloatPoint(moveTo->x(), moveTo->y());
67 return segment;
68}

103std::optional<SVGPathSource::CurveToCubicSegment> SVGPathSegListSource::parseCurveToCubicSegment()

151std::optional<SVGPathSource::ArcToSegment> SVGPathSegListSource::parseArcToSegment()
152{
153 ASSERT(m_segment);
154 ASSERT(m_segment->pathSegType() == PathSegArcAbs || m_segment->pathSegType() == PathSegArcRel);
155 SVGPathSegArc* arcTo = static_cast<SVGPathSegArc*>(m_segment.get());
156
157 ArcToSegment segment;
158 segment.rx = arcTo->r1();
159 segment.ry = arcTo->r2();
160 segment.angle = arcTo->angle();
161 segment.largeArc = arcTo->largeArcFlag();
162 segment.sweep = arcTo->sweepFlag();
163 segment.targetPoint = FloatPoint(arcTo->x(), arcTo->y());
164 return segment;
165}
