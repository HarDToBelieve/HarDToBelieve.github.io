31inline SVGSetElement::SVGSetElement(const QualifiedName& tagName, Document& document)
32 : SVGAnimateElementBase(tagName, document)
33{
34 setAnimationMode(AnimationMode::To);
35 ASSERT(hasTagName(SVGNames::setTag));
36}
37
38Ref<SVGSetElement> SVGSetElement::create(const QualifiedName& tagName, Document& document)
39{
40 return adoptRef(*new SVGSetElement(tagName, document));
41}
42
43void SVGSetElement::updateAnimationMode()
44{
45 // No-op, as <set> has a constant animation mode of AnimationMode::To.
46 // See: http://www.w3.org/TR/SVG/single-page.html#animate-SetElement
47}
