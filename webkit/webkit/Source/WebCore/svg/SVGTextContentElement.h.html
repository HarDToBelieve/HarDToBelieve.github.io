28struct DOMPointInit;

30enum SVGLengthAdjustType {
31 SVGLengthAdjustUnknown,
32 SVGLengthAdjustSpacing,
33 SVGLengthAdjustSpacingAndGlyphs
34};
35
36template<> struct SVGPropertyTraits<SVGLengthAdjustType> {
37 static unsigned highestEnumValue() { return SVGLengthAdjustSpacingAndGlyphs; }
38
39 static String toString(SVGLengthAdjustType type)
40 {
41 switch (type) {
42 case SVGLengthAdjustUnknown:
43 return emptyString();
44 case SVGLengthAdjustSpacing:
45 return "spacing"_s;
46 case SVGLengthAdjustSpacingAndGlyphs:
47 return "spacingAndGlyphs"_s;
48 }
49
50 ASSERT_NOT_REACHED();
51 return emptyString();
52 }
53
54 static SVGLengthAdjustType fromString(const String& value)
55 {
56 if (value == "spacingAndGlyphs")
57 return SVGLengthAdjustSpacingAndGlyphs;
58 if (value == "spacing")
59 return SVGLengthAdjustSpacing;
60 return SVGLengthAdjustUnknown;
61 }
62};

64class SVGTextContentElement : public SVGGraphicsElement {
65 WTF_MAKE_ISO_ALLOCATED(SVGTextContentElement);
66public:
67 enum {
68 LENGTHADJUST_UNKNOWN = SVGLengthAdjustUnknown,
69 LENGTHADJUST_SPACING = SVGLengthAdjustSpacing,
70 LENGTHADJUST_SPACINGANDGLYPHS = SVGLengthAdjustSpacingAndGlyphs
71 };
72
73 unsigned getNumberOfChars();
74 float getComputedTextLength();
75 ExceptionOr<float> getSubStringLength(unsigned charnum, unsigned nchars);
76 ExceptionOr<Ref<SVGPoint>> getStartPositionOfChar(unsigned charnum);
77 ExceptionOr<Ref<SVGPoint>> getEndPositionOfChar(unsigned charnum);
78 ExceptionOr<Ref<SVGRect>> getExtentOfChar(unsigned charnum);
79 ExceptionOr<float> getRotationOfChar(unsigned charnum);
80 int getCharNumAtPosition(DOMPointInit&&);
81 ExceptionOr<void> selectSubString(unsigned charnum, unsigned nchars);
82
83 static SVGTextContentElement* elementFromRenderer(RenderObject*);
84
85 using PropertyRegistry = SVGPropertyOwnerRegistry<SVGTextContentElement, SVGGraphicsElement>;
86
87 const SVGLengthValue& specifiedTextLength() const { return m_specifiedTextLength; }
88 const SVGLengthValue& textLength() const { return m_textLength->currentValue(); }
89 SVGLengthAdjustType lengthAdjust() const { return m_lengthAdjust->currentValue<SVGLengthAdjustType>(); }
90
91 SVGAnimatedLength& textLengthAnimated();
92 SVGAnimatedEnumeration& lengthAdjustAnimated() { return m_lengthAdjust; }
93
94protected:
95 SVGTextContentElement(const QualifiedName&, Document&);
96
97 bool isValid() const override { return SVGTests::isValid(); }
98
99 void parseAttribute(const QualifiedName&, const AtomString&) override;
100 bool hasPresentationalHintsForAttribute(const QualifiedName&) const override;
101 void collectPresentationalHintsForAttribute(const QualifiedName&, const AtomString&, MutableStyleProperties&) override;
102 void svgAttributeChanged(const QualifiedName&) override;
103
104 bool selfHasRelativeLengths() const override;
105
106private:
107 bool isTextContent() const final { return true; }
108
109 Ref<SVGAnimatedLength> m_textLength { SVGAnimatedLength::create(this, SVGLengthMode::Other) };
110 Ref<SVGAnimatedEnumeration> m_lengthAdjust { SVGAnimatedEnumeration::create(this, SVGLengthAdjustSpacing) };
111 SVGLengthValue m_specifiedTextLength { SVGLengthMode::Other };
112};
