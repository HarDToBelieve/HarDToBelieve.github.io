31WTF_MAKE_ISO_ALLOCATED_IMPL(SVGTitleElement);
32
33inline SVGTitleElement::SVGTitleElement(const QualifiedName& tagName, Document& document)
34 : SVGElement(tagName, document)
35{
36 ASSERT(hasTagName(SVGNames::titleTag));
37}
38
39Ref<SVGTitleElement> SVGTitleElement::create(const QualifiedName& tagName, Document& document)
40{
41 return adoptRef(*new SVGTitleElement(tagName, document));
42}
43
44Node::InsertedIntoAncestorResult SVGTitleElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
45{
46 SVGElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
47 document().titleElementAdded(*this);
48 return InsertedIntoAncestorResult::Done;
49}
50
51void SVGTitleElement::removedFromAncestor(RemovalType removalType, ContainerNode& oldParentOfRemovedTree)
52{
53 SVGElement::removedFromAncestor(removalType, oldParentOfRemovedTree);
54 document().titleElementRemoved(*this);
55}
56
57void SVGTitleElement::childrenChanged(const ChildChange& change)
58{
59 SVGElement::childrenChanged(change);
60 document().titleElementTextChanged(*this);
61}
