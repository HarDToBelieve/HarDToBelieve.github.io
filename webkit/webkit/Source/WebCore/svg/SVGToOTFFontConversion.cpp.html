273static uint16_t roundDownToPowerOfTwo(uint16_t x)
274{
275 x |= x >> 1;
276 x |= x >> 2;
277 x |= x >> 4;
278 x |= x >> 8;
279 return (x >> 1) + 1;
280}

282static uint16_t integralLog2(uint16_t x)
283{
284 uint16_t result = 0;
285 while (x >>= 1)
286 ++result;
287 return result;
288}

290void SVGToOTFFontConverter::appendFormat12CMAPTable(const Vector<std::pair<UChar32, Glyph>>& mappings)
291{
292 // Braindead scheme: One segment for each character
293 ASSERT(m_glyphs.size() < 0xFFFF);
294 auto subtableLocation = m_result.size();
295 append32(12 << 16); // Format 12
296 append32(0); // Placeholder for byte length
297 append32(0); // Language independent
298 append32(0); // Placeholder for nGroups
299 for (auto& mapping : mappings) {
300 append32(mapping.first); // startCharCode
301 append32(mapping.first); // endCharCode
302 append32(mapping.second); // startGlyphCode
303 }
304 overwrite32(subtableLocation + 4, m_result.size() - subtableLocation);
305 overwrite32(subtableLocation + 12, mappings.size());
306}

308void SVGToOTFFontConverter::appendFormat4CMAPTable(const Vector<std::pair<UChar32, Glyph>>& bmpMappings)
309{
310 auto subtableLocation = m_result.size();
311 append16(4); // Format 4
312 append16(0); // Placeholder for length in bytes
313 append16(0); // Language independent
314 uint16_t segCount = bmpMappings.size() + 1;
315 append16(clampTo<uint16_t>(2 * segCount)); // segCountX2: "2 x segCount"
316 uint16_t originalSearchRange = roundDownToPowerOfTwo(segCount);
317 uint16_t searchRange = clampTo<uint16_t>(2 * originalSearchRange); // searchRange: "2 x (2**floor(log2(segCount)))"
318 append16(searchRange);
319 append16(integralLog2(originalSearchRange)); // entrySelector: "log2(searchRange/2)"
320 append16(clampTo<uint16_t>((2 * segCount) - searchRange)); // rangeShift: "2 x segCount - searchRange"
321
322 // Ending character codes
323 for (auto& mapping : bmpMappings)
324 append16(mapping.first); // startCharCode
325 append16(0xFFFF);
326
327 append16(0); // reserved
328
329 // Starting character codes
330 for (auto& mapping : bmpMappings)
331 append16(mapping.first); // startCharCode
332 append16(0xFFFF);
333
334 // idDelta
335 for (auto& mapping : bmpMappings)
336 append16(static_cast<uint16_t>(mapping.second) - static_cast<uint16_t>(mapping.first)); // startCharCode
337 append16(0x0001);
338
339 // idRangeOffset
340 for (size_t i = 0; i < bmpMappings.size(); ++i)
341 append16(0); // startCharCode
342 append16(0);
343
344 // Fonts strive to hold 2^16 glyphs, but with the current encoding scheme, we write 8 bytes per codepoint into this subtable.
345 // Because the size of this subtable must be represented as a 16-bit number, we are limiting the number of glyphs we support to 2^13.
346 // FIXME: If we hit this limit in the wild, use a more compact encoding scheme for this subtable.
347 overwrite16(subtableLocation + 2, clampTo<uint16_t>(m_result.size() - subtableLocation));
348}

350void SVGToOTFFontConverter::appendCMAPTable()
351{
352 auto startingOffset = m_result.size();
353 append16(0);
354 append16(3); // Number of subtables
355
356 append16(0); // Unicode
357 append16(3); // Unicode version 2.2+
358 append32(28); // Byte offset of subtable
359
360 append16(3); // Microsoft
361 append16(1); // Unicode BMP
362 auto format4OffsetLocation = m_result.size();
363 append32(0); // Byte offset of subtable
364
365 append16(3); // Microsoft
366 append16(10); // Unicode
367 append32(28); // Byte offset of subtable
368
369 Vector<std::pair<UChar32, Glyph>> mappings;
370 UChar32 previousCodepoint = std::numeric_limits<UChar32>::max();
371 for (size_t i = 0; i < m_glyphs.size(); ++i) {
372 auto& glyph = m_glyphs[i];
373 UChar32 codepoint;
374 auto codePoints = StringView(glyph.codepoints).codePoints();
375 auto iterator = codePoints.begin();
376 if (iterator == codePoints.end())
377 codepoint = 0;
378 else {
379 codepoint = *iterator;
380 ++iterator;
381 // Don't map ligatures here.
382 if (iterator != codePoints.end() || codepoint == previousCodepoint)
383 continue;
384 }
385
386 mappings.append(std::make_pair(codepoint, Glyph(i)));
387 previousCodepoint = codepoint;
388 }
389
390 appendFormat12CMAPTable(mappings);
391
392 Vector<std::pair<UChar32, Glyph>> bmpMappings;
393 for (auto& mapping : mappings) {
394 if (mapping.first < 0x10000)
395 bmpMappings.append(mapping);
396 }
397 overwrite32(format4OffsetLocation, m_result.size() - startingOffset);
398 appendFormat4CMAPTable(bmpMappings);
399}

401void SVGToOTFFontConverter::appendHEADTable()
402{
403 append32(0x00010000); // Version
404 append32(0x00010000); // Revision
405 append32(0); // Checksum placeholder; to be overwritten by the caller.
406 append32(0x5F0F3CF5); // Magic number.
407 append16((1 << 9) | 1);
408
409 append16(s_outputUnitsPerEm);
410 append32(0); // First half of creation date
411 append32(0); // Last half of creation date
412 append32(0); // First half of modification date
413 append32(0); // Last half of modification date
414 append16(clampTo<int16_t>(m_boundingBox.x()));
415 append16(clampTo<int16_t>(m_boundingBox.y()));
416 append16(clampTo<int16_t>(m_boundingBox.maxX()));
417 append16(clampTo<int16_t>(m_boundingBox.maxY()));
418 append16((m_italic ? 1 << 1 : 0) | (m_weight >= 7 ? 1 : 0));
419 append16(3); // Smallest readable size in pixels
420 append16(0); // Might contain LTR or RTL glyphs
421 append16(0); // Short offsets in the 'loca' table. However, CFF fonts don't have a 'loca' table so this is irrelevant
422 append16(0); // Glyph data format
423}

425void SVGToOTFFontConverter::appendHHEATable()
426{
427 append32(0x00010000); // Version
428 append16(clampTo<int16_t>(m_ascent));
429 append16(clampTo<int16_t>(-m_descent));
430 // WebKit SVG font rendering has hard coded the line gap to be 1/10th of the font size since 2008 (see r29719).
431 append16(clampTo<int16_t>(m_lineGap));
432 append16(clampTo<uint16_t>(m_advanceWidthMax));
433 append16(clampTo<int16_t>(m_boundingBox.x())); // Minimum left side bearing
434 append16(clampTo<int16_t>(m_minRightSideBearing)); // Minimum right side bearing
435 append16(clampTo<int16_t>(m_boundingBox.maxX())); // X maximum extent
436 // Since WebKit draws the caret and ignores the following values, it doesn't matter what we set them to.
437 append16(1); // Vertical caret
438 append16(0); // Vertical caret
439 append16(0); // "Set value to 0 for non-slanted fonts"
440 append32(0); // Reserved
441 append32(0); // Reserved
442 append16(0); // Current format
443 append16(m_glyphs.size()); // Number of advance widths in HMTX table
444}

490void SVGToOTFFontConverter::appendOS2Table()
491{
492 int16_t averageAdvance = s_outputUnitsPerEm;
493 auto horizAdvX = parseHTMLInteger(m_fontElement.attributeWithoutSynchronization(SVGNames::horiz_adv_xAttr));
494 if (!horizAdvX && m_missingGlyphElement)
495 horizAdvX = parseHTMLInteger(m_missingGlyphElement->attributeWithoutSynchronization(SVGNames::horiz_adv_xAttr));
496 if (horizAdvX)
497 averageAdvance = clampTo<int16_t>(scaleUnitsPerEm(*horizAdvX));
498
499 append16(2); // Version
500 append16(clampTo<int16_t>(averageAdvance));
501 append16(m_weight); // Weight class
502 append16(5); // Width class
503 append16(0); // Protected font
504 // WebKit handles these superscripts and subscripts
505 append16(0); // Subscript X Size
506 append16(0); // Subscript Y Size
507 append16(0); // Subscript X Offset
508 append16(0); // Subscript Y Offset
509 append16(0); // Superscript X Size
510 append16(0); // Superscript Y Size
511 append16(0); // Superscript X Offset
512 append16(0); // Superscript Y Offset
513 append16(0); // Strikeout width
514 append16(0); // Strikeout Position
515 append16(0); // No classification
516
517 unsigned numPanoseBytes = 0;
518 const unsigned panoseSize = 10;
519 char panoseBytes[panoseSize];
520 if (m_fontFaceElement) {
521 Vector<String> segments = m_fontFaceElement->attributeWithoutSynchronization(SVGNames::panose_1Attr).string().split(' ');
522 if (segments.size() == panoseSize) {
523 for (auto& segment : segments) {
524 if (auto value = parseIntegerAllowingTrailingJunk<uint8_t>(segment))
525 panoseBytes[numPanoseBytes++] = *value;
526 }
527 }
528 }
529 if (numPanoseBytes != panoseSize)
530 memset(panoseBytes, 0, panoseSize);
531 m_result.append(panoseBytes, panoseSize);
532
533 for (int i = 0; i < 4; ++i)
534 append32(0); // "Bit assignments are pending. Set to 0"
535 append32(0x544B4257); // Font Vendor. "WBKT"
536 append16((m_weight >= 7 ? 1 << 5 : 0) | (m_italic ? 1 : 0)); // Font Patterns.
537 append16(0); // First unicode index
538 append16(0xFFFF); // Last unicode index
539 append16(clampTo<int16_t>(m_ascent)); // Typographical ascender
540 append16(clampTo<int16_t>(-m_descent)); // Typographical descender
541 append16(clampTo<int16_t>(m_lineGap)); // Typographical line gap
542 append16(clampTo<uint16_t>(m_ascent)); // Windows-specific ascent
543 append16(clampTo<uint16_t>(m_descent)); // Windows-specific descent
544 append32(0xFF10FC07); // Bitmask for supported codepages (Part 1). Report all pages as supported.
545 append32(0x0000FFFF); // Bitmask for supported codepages (Part 2). Report all pages as supported.
546 append16(clampTo<int16_t>(m_xHeight)); // x-height
547 append16(clampTo<int16_t>(m_capHeight)); // Cap-height
548 append16(0); // Default char
549 append16(' '); // Break character
550 append16(3); // Maximum context needed to perform font features
551 append16(3); // Smallest optical point size
552 append16(0xFFFF); // Largest optical point size
553}

584void SVGToOTFFontConverter::appendCFFTable()
585{
586 auto startingOffset = m_result.size();
587
588 // Header
589 m_result.append(1); // Major version
590 m_result.append(0); // Minor version
591 m_result.append(4); // Header size
592 m_result.append(4); // Offsets within CFF table are 4 bytes long
593
594 // Name INDEX
595 String fontName;
596 if (m_fontFaceElement) {
597 // FIXME: fontFamily() here might not be quite what we want.
598 String potentialFontName = m_fontFamily;
599 if (isValidStringForCFF(potentialFontName))
600 fontName = potentialFontName;
601 }
602 append16(1); // INDEX contains 1 element
603 m_result.append(4); // Offsets in this INDEX are 4 bytes long
604 append32(1); // 1-index offset of name data
605 append32(fontName.length() + 1); // 1-index offset just past end of name data
606 appendValidCFFString(fontName);
607
608 String weight;
609 if (m_fontFaceElement) {
610 auto& potentialWeight = m_fontFaceElement->attributeWithoutSynchronization(SVGNames::font_weightAttr);
611 if (isValidStringForCFF(potentialWeight))
612 weight = potentialWeight;
613 }
614
615 bool hasWeight = !weight.isNull();
616
617 const char operand32Bit = 29;
618 const char fullNameKey = 2;
619 const char familyNameKey = 3;
620 const char weightKey = 4;
621 const char fontBBoxKey = 5;
622 const char charsetIndexKey = 15;
623 const char charstringsIndexKey = 17;
624 const char privateDictIndexKey = 18;
625 const uint32_t userDefinedStringStartIndex = 391;
626 const unsigned sizeOfTopIndex = 56 + (hasWeight ? 6 : 0);
627
628 // Top DICT INDEX.
629 append16(1); // INDEX contains 1 element
630 m_result.append(4); // Offsets in this INDEX are 4 bytes long
631 append32(1); // 1-index offset of DICT data
632 append32(1 + sizeOfTopIndex); // 1-index offset just past end of DICT data
633
634 // DICT information
635#if ASSERT_ENABLED
636 unsigned topDictStart = m_result.size();
637#endif
638 m_result.append(operand32Bit);
639 append32(userDefinedStringStartIndex);
640 m_result.append(fullNameKey);
641 m_result.append(operand32Bit);
642 append32(userDefinedStringStartIndex);
643 m_result.append(familyNameKey);
644 if (hasWeight) {
645 m_result.append(operand32Bit);
646 append32(userDefinedStringStartIndex + 2);
647 m_result.append(weightKey);
648 }
649 m_result.append(operand32Bit);
650 append32(clampTo<int32_t>(m_boundingBox.x()));
651 m_result.append(operand32Bit);
652 append32(clampTo<int32_t>(m_boundingBox.y()));
653 m_result.append(operand32Bit);
654 append32(clampTo<int32_t>(m_boundingBox.width()));
655 m_result.append(operand32Bit);
656 append32(clampTo<int32_t>(m_boundingBox.height()));
657 m_result.append(fontBBoxKey);
658 m_result.append(operand32Bit);
659 unsigned charsetOffsetLocation = m_result.size();
660 append32(0); // Offset of Charset info. Will be overwritten later.
661 m_result.append(charsetIndexKey);
662 m_result.append(operand32Bit);
663 unsigned charstringsOffsetLocation = m_result.size();
664 append32(0); // Offset of CharStrings INDEX. Will be overwritten later.
665 m_result.append(charstringsIndexKey);
666 m_result.append(operand32Bit);
667 append32(0); // 0-sized private dict
668 m_result.append(operand32Bit);
669 append32(0); // no location for private dict
670 m_result.append(privateDictIndexKey); // Private dict size and offset
671 ASSERT(m_result.size() == topDictStart + sizeOfTopIndex);
672
673 // String INDEX
674 String unknownCharacter = "UnknownChar"_s;
675 append16(2 + (hasWeight ? 1 : 0)); // Number of elements in INDEX
676 m_result.append(4); // Offsets in this INDEX are 4 bytes long
677 uint32_t offset = 1;
678 append32(offset);
679 offset += fontName.length();
680 append32(offset);
681 offset += unknownCharacter.length();
682 append32(offset);
683 if (hasWeight) {
684 offset += weight.length();
685 append32(offset);
686 }
687 appendValidCFFString(fontName);
688 appendValidCFFString(unknownCharacter);
689 appendValidCFFString(weight);
690
691 append16(0); // Empty subroutine INDEX
692
693 // Charset info
694 overwrite32(charsetOffsetLocation, m_result.size() - startingOffset);
695 m_result.append(0);
696 for (Glyph i = 1; i < m_glyphs.size(); ++i)
697 append16(userDefinedStringStartIndex + 1);
698
699 // CharStrings INDEX
700 overwrite32(charstringsOffsetLocation, m_result.size() - startingOffset);
701 append16(m_glyphs.size());
702 m_result.append(4); // Offsets in this INDEX are 4 bytes long
703 offset = 1;
704 append32(offset);
705 for (auto& glyph : m_glyphs) {
706 offset += glyph.charString.size();
707 append32(offset);
708 }
709 for (auto& glyph : m_glyphs)
710 m_result.appendVector(glyph.charString);
711}

730void SVGToOTFFontConverter::appendLigatureSubtable(size_t subtableRecordLocation)
731{
732 typedef std::pair<Vector<Glyph, 3>, Glyph> LigaturePair;
733 Vector<LigaturePair> ligaturePairs;
734 for (Glyph glyphIndex = 0; glyphIndex < m_glyphs.size(); ++glyphIndex) {
735 ligaturePairs.append(LigaturePair(Vector<Glyph, 3>(), glyphIndex));
736 Vector<Glyph, 3>& ligatureGlyphs = ligaturePairs.last().first;
737 auto codePoints = StringView(m_glyphs[glyphIndex].codepoints).codePoints();
738 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=138592 This needs to be done in codepoint space, not glyph space
739 for (auto codePoint : codePoints)
740 ligatureGlyphs.append(firstGlyph(glyphsForCodepoint(codePoint), codePoint));
741 if (ligatureGlyphs.size() < 2)
742 ligaturePairs.removeLast();
743 }
744 if (ligaturePairs.size() > std::numeric_limits<uint16_t>::max())
745 ligaturePairs.clear();
746 std::sort(ligaturePairs.begin(), ligaturePairs.end(), [](auto& lhs, auto& rhs) {
747 return lhs.first[0] < rhs.first[0];
748 });
749 Vector<size_t> overlappingFirstGlyphSegmentLengths;
750 if (!ligaturePairs.isEmpty()) {
751 Glyph previousFirstGlyph = ligaturePairs[0].first[0];
752 size_t segmentStart = 0;
753 for (size_t i = 0; i < ligaturePairs.size(); ++i) {
754 auto& ligaturePair = ligaturePairs[i];
755 if (ligaturePair.first[0] != previousFirstGlyph) {
756 overlappingFirstGlyphSegmentLengths.append(i - segmentStart);
757 segmentStart = i;
758 previousFirstGlyph = ligaturePairs[0].first[0];
759 }
760 }
761 overlappingFirstGlyphSegmentLengths.append(ligaturePairs.size() - segmentStart);
762 }
763
764 overwrite16(subtableRecordLocation + 6, m_result.size() - subtableRecordLocation);
765 auto subtableLocation = m_result.size();
766 append16(1); // Format 1
767 append16(0); // Placeholder for offset to coverage table, relative to beginning of substitution table
768 append16(ligaturePairs.size()); // Number of LigatureSet tables
769 grow(overlappingFirstGlyphSegmentLengths.size() * 2); // Placeholder for offset to LigatureSet table
770
771 Vector<size_t> ligatureSetTableLocations;
772 for (size_t i = 0; i < overlappingFirstGlyphSegmentLengths.size(); ++i) {
773 overwrite16(subtableLocation + 6 + 2 * i, m_result.size() - subtableLocation);
774 ligatureSetTableLocations.append(m_result.size());
775 append16(overlappingFirstGlyphSegmentLengths[i]); // LigatureCount
776 grow(overlappingFirstGlyphSegmentLengths[i] * 2); // Placeholder for offset to Ligature table
777 }
778 ASSERT(ligatureSetTableLocations.size() == overlappingFirstGlyphSegmentLengths.size());
779
780 size_t ligaturePairIndex = 0;
781 for (size_t i = 0; i < overlappingFirstGlyphSegmentLengths.size(); ++i) {
782 for (size_t j = 0; j < overlappingFirstGlyphSegmentLengths[i]; ++j) {
783 overwrite16(ligatureSetTableLocations[i] + 2 + 2 * j, m_result.size() - ligatureSetTableLocations[i]);
784 auto& ligaturePair = ligaturePairs[ligaturePairIndex];
785 append16(ligaturePair.second);
786 append16(ligaturePair.first.size());
787 for (size_t k = 1; k < ligaturePair.first.size(); ++k)
788 append16(ligaturePair.first[k]);
789 ++ligaturePairIndex;
790 }
791 }
792 ASSERT(ligaturePairIndex == ligaturePairs.size());
793
794 // Coverage table
795 overwrite16(subtableLocation + 2, m_result.size() - subtableLocation);
796 append16(1); // CoverageFormat
797 append16(ligatureSetTableLocations.size()); // GlyphCount
798 ligaturePairIndex = 0;
799 for (auto segmentLength : overlappingFirstGlyphSegmentLengths) {
800 auto& ligaturePair = ligaturePairs[ligaturePairIndex];
801 ASSERT(ligaturePair.first.size() > 1);
802 append16(ligaturePair.first[0]);
803 ligaturePairIndex += segmentLength;
804 }
805}

850void SVGToOTFFontConverter::appendGSUBTable()
851{
852 auto tableLocation = m_result.size();
853 auto headerSize = 10;
854
855 append32(0x00010000); // Version
856 append16(headerSize); // Offset to ScriptList
857 Placeholder toFeatureList = placeholder(tableLocation);
858 Placeholder toLookupList = placeholder(tableLocation);
859 ASSERT(tableLocation + headerSize == m_result.size());
860
861 // ScriptList
862 auto scriptListLocation = m_result.size();
863 append16(2); // Number of ScriptRecords
864 append32BitCode("DFLT");
865 append16(0); // Placeholder for offset of Script table, relative to beginning of ScriptList
866 append32BitCode("arab");
867 append16(0); // Placeholder for offset of Script table, relative to beginning of ScriptList
868
869 overwrite16(scriptListLocation + 6, m_result.size() - scriptListLocation);
870 appendScriptSubtable(1);
871 overwrite16(scriptListLocation + 12, m_result.size() - scriptListLocation);
872 appendScriptSubtable(4);
873
874 const unsigned featureCount = 5;
875
876 // FeatureList
877 toFeatureList.populate();
878 auto featureListLocation = m_result.size();
879 size_t featureListSize = 2 + 6 * featureCount;
880 size_t featureTableSize = 6;
881 append16(featureCount); // FeatureCount
882 append32BitCode("liga");
883 append16(featureListSize + featureTableSize * 0); // Offset of feature table, relative to beginning of FeatureList table
884 append32BitCode("fina");
885 append16(featureListSize + featureTableSize * 1); // Offset of feature table, relative to beginning of FeatureList table
886 append32BitCode("medi");
887 append16(featureListSize + featureTableSize * 2); // Offset of feature table, relative to beginning of FeatureList table
888 append32BitCode("init");
889 append16(featureListSize + featureTableSize * 3); // Offset of feature table, relative to beginning of FeatureList table
890 append32BitCode("rlig");
891 append16(featureListSize + featureTableSize * 4); // Offset of feature table, relative to beginning of FeatureList table
892 ASSERT_UNUSED(featureListLocation, featureListLocation + featureListSize == m_result.size());
893
894 for (unsigned i = 0; i < featureCount; ++i) {
895 auto featureTableStart = m_result.size();
896 append16(0); // FeatureParams "= NULL ... reserved"
897 append16(1); // LookupCount
898 append16(i); // LookupListIndex
899 ASSERT_UNUSED(featureTableStart, featureTableStart + featureTableSize == m_result.size());
900 }
901
902 // LookupList
903 toLookupList.populate();
904 auto lookupListLocation = m_result.size();
905 append16(featureCount); // LookupCount
906 for (unsigned i = 0; i < featureCount; ++i)
907 append16(0); // Placeholder for offset to feature table, relative to beginning of LookupList
908 size_t subtableRecordLocations[featureCount];
909 for (unsigned i = 0; i < featureCount; ++i) {
910 subtableRecordLocations[i] = m_result.size();
911 overwrite16(lookupListLocation + 2 + 2 * i, m_result.size() - lookupListLocation);
912 switch (i) {
913 case 4:
914 append16(3); // Type 3: "Replace one glyph with one of many glyphs"
915 break;
916 case 0:
917 append16(4); // Type 4: "Replace multiple glyphs with one glyph"
918 break;
919 default:
920 append16(1); // Type 1: "Replace one glyph with one glyph"
921 break;
922 }
923 append16(0); // LookupFlag
924 append16(1); // SubTableCount
925 append16(0); // Placeholder for offset to subtable, relative to beginning of Lookup table
926 }
927
928 appendLigatureSubtable(subtableRecordLocations[0]);
929 appendArabicReplacementSubtable(subtableRecordLocations[1], "terminal");
930 appendArabicReplacementSubtable(subtableRecordLocations[2], "medial");
931 appendArabicReplacementSubtable(subtableRecordLocations[3], "initial");
932
933 // Manually append empty "rlig" subtable
934 overwrite16(subtableRecordLocations[4] + 6, m_result.size() - subtableRecordLocations[4]);
935 append16(1); // Format 1
936 append16(6); // offset to coverage table, relative to beginning of substitution table
937 append16(0); // AlternateSetCount
938 append16(1); // CoverageFormat
939 append16(0); // GlyphCount
940}

1055template<typename T> inline size_t SVGToOTFFontConverter::appendKERNSubtable(std::optional<SVGKerningPair> (T::*buildKerningPair)() const, uint16_t coverage)
1056{
1057 Vector<KerningData> kerningData;
1058 for (auto& element : childrenOfType<T>(m_fontElement)) {
1059 if (auto kerningPair = (element.*buildKerningPair)())
1060 addKerningPair(kerningData, WTFMove(*kerningPair));
1061 }
1062 return finishAppendingKERNSubtable(WTFMove(kerningData), coverage);
1063}
1064
1065size_t SVGToOTFFontConverter::finishAppendingKERNSubtable(Vector<KerningData> kerningData, uint16_t coverage)
1066{
1067 std::sort(kerningData.begin(), kerningData.end(), [](auto& a, auto& b) {
1068 return a.glyph1 < b.glyph1 || (a.glyph1 == b.glyph1 && a.glyph2 < b.glyph2);
1069 });
1070
1071 size_t sizeOfKerningDataTable = 14 + 6 * kerningData.size();
1072 if (sizeOfKerningDataTable > std::numeric_limits<uint16_t>::max()) {
1073 kerningData.clear();
1074 sizeOfKerningDataTable = 14;
1075 }
1076
1077 append16(0); // Version of subtable
1078 append16(sizeOfKerningDataTable); // Length of this subtable
1079 append16(coverage); // Table coverage bitfield
1080
1081 uint16_t roundedNumKerningPairs = roundDownToPowerOfTwo(kerningData.size());
1082
1083 append16(kerningData.size());
1084 append16(roundedNumKerningPairs * 6); // searchRange: "The largest power of two less than or equal to the value of nPairs, multiplied by the size in bytes of an entry in the table."
1085 append16(integralLog2(roundedNumKerningPairs)); // entrySelector: "log2 of the largest power of two less than or equal to the value of nPairs."
1086 append16((kerningData.size() - roundedNumKerningPairs) * 6); // rangeShift: "The value of nPairs minus the largest power of two less than or equal to nPairs,
1087 // and then multiplied by the size in bytes of an entry in the table."
1088
1089 for (auto& kerningDataElement : kerningData) {
1090 append16(kerningDataElement.glyph1);
1091 append16(kerningDataElement.glyph2);
1092 append16(kerningDataElement.adjustment);
1093 }
1094
1095 return sizeOfKerningDataTable;
1096}

1326bool SVGToOTFFontConverter::compareCodepointsLexicographically(const GlyphData& data1, const GlyphData& data2)
1327{
1328 auto codePoints1 = StringView(data1.codepoints).codePoints();
1329 auto codePoints2 = StringView(data2.codepoints).codePoints();
1330 auto iterator1 = codePoints1.begin();
1331 auto iterator2 = codePoints2.begin();
1332 while (iterator1 != codePoints1.end() && iterator2 != codePoints2.end()) {
1333 UChar32 codepoint1, codepoint2;
1334 codepoint1 = *iterator1;
1335 codepoint2 = *iterator2;
1336
1337 if (codepoint1 < codepoint2)
1338 return true;
1339 if (codepoint1 > codepoint2)
1340 return false;
1341
1342 ++iterator1;
1343 ++iterator2;
1344 }
1345
1346 if (iterator1 == codePoints1.end() && iterator2 == codePoints2.end()) {
1347 bool firstIsIsolated = data1.glyphElement && equalLettersIgnoringASCIICase(data1.glyphElement->attributeWithoutSynchronization(SVGNames::arabic_formAttr), "isolated");
1348 bool secondIsIsolated = data2.glyphElement && equalLettersIgnoringASCIICase(data2.glyphElement->attributeWithoutSynchronization(SVGNames::arabic_formAttr), "isolated");
1349 return firstIsIsolated && !secondIsIsolated;
1350 }
1351 return iterator1 == codePoints1.end();
1352}

1502 overwrite32(directoryEntryOffset + 4, calculateChecksum(offset, m_result.size()));

1508bool SVGToOTFFontConverter::convertSVGToOTFFont()
1509{
1510 if (m_glyphs.isEmpty())
1511 return false;
1512
1513 uint16_t numTables = 14;
1514 uint16_t roundedNumTables = roundDownToPowerOfTwo(numTables);
1515 uint16_t searchRange = roundedNumTables * 16; // searchRange: "(Maximum power of 2 <= numTables) x 16."
1516
1517 m_result.append('O');
1518 m_result.append('T');
1519 m_result.append('T');
1520 m_result.append('O');
1521 append16(numTables);
1522 append16(searchRange);
1523 append16(integralLog2(roundedNumTables)); // entrySelector: "Log2(maximum power of 2 <= numTables)."
1524 append16(numTables * 16 - searchRange); // rangeShift: "NumTables x 16-searchRange."
1525
1526 ASSERT(m_result.size() == headerSize);
1527
1528 // Leave space for the directory entries.
1529 for (size_t i = 0; i < directoryEntrySize * numTables; ++i)
1530 m_result.append(0);
1531
1532 appendTable("CFF ", &SVGToOTFFontConverter::appendCFFTable);
1533 appendTable("GSUB", &SVGToOTFFontConverter::appendGSUBTable);
1534 appendTable("OS/2", &SVGToOTFFontConverter::appendOS2Table);
1535 appendTable("VORG", &SVGToOTFFontConverter::appendVORGTable);
1536 appendTable("cmap", &SVGToOTFFontConverter::appendCMAPTable);
1537 auto headTableOffset = m_result.size();
1538 appendTable("head", &SVGToOTFFontConverter::appendHEADTable);
1539 appendTable("hhea", &SVGToOTFFontConverter::appendHHEATable);
1540 appendTable("hmtx", &SVGToOTFFontConverter::appendHMTXTable);
1541 appendTable("kern", &SVGToOTFFontConverter::appendKERNTable);
1542 appendTable("maxp", &SVGToOTFFontConverter::appendMAXPTable);
1543 appendTable("name", &SVGToOTFFontConverter::appendNAMETable);
1544 appendTable("post", &SVGToOTFFontConverter::appendPOSTTable);
1545 appendTable("vhea", &SVGToOTFFontConverter::appendVHEATable);
1546 appendTable("vmtx", &SVGToOTFFontConverter::appendVMTXTable);
1547
1548 ASSERT(numTables == m_tablesAppendedCount);
1549
1550 // checksumAdjustment: "To compute: set it to 0, calculate the checksum for the 'head' table and put it in the table directory,
1551 // sum the entire font as uint32, then store B1B0AFBA - sum. The checksum for the 'head' table will now be wrong. That is OK."
1552 overwrite32(headTableOffset + 8, 0xB1B0AFBAU - calculateChecksum(0, m_result.size()));
1553 return true;
1554}
