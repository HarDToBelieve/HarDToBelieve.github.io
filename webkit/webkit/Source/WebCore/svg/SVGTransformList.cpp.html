35ExceptionOr<RefPtr<SVGTransform>> SVGTransformList::consolidate()
36{
37 auto result = canAlterList();
38 if (result.hasException())
39 return result.releaseException();
40 ASSERT(result.releaseReturnValue());
41
42 // Spec: If the list was empty, then a value of null is returned.
43 if (m_items.isEmpty())
44 return nullptr;
45
46 if (m_items.size() == 1)
47 return makeRefPtr(at(0).get());
48
49 auto newItem = SVGTransform::create(concatenate());
50 clearItems();
51
52 auto item = append(WTFMove(newItem));
53 commitChange();
54 return makeRefPtr(item.get());
55}
56
57AffineTransform SVGTransformList::concatenate() const
58{
59 AffineTransform result;
60 for (auto& transform : m_items)
61 result *= transform->matrix()->value();
62 return result;
63}

65template<typename CharacterType> bool SVGTransformList::parseGeneric(StringParsingBuffer<CharacterType>& buffer)
66{
67 bool delimParsed = false;
68 skipOptionalSVGSpaces(buffer);
69
70 while (buffer.hasCharactersRemaining()) {
71 delimParsed = false;
72
73 auto parsedTransformType = SVGTransformable::parseTransformType(buffer);
74 if (!parsedTransformType)
75 return false;
76
77 auto parsedTransformValue = SVGTransformable::parseTransformValue(*parsedTransformType, buffer);
78 if (!parsedTransformValue)
79 return false;
80
81 append(SVGTransform::create(*parsedTransformValue));
82
83 skipOptionalSVGSpaces(buffer);
84
85 if (skipExactly(buffer, ','))
86 delimParsed = true;
87
88 skipOptionalSVGSpaces(buffer);
89 }
90 return !delimParsed;
91}

93void SVGTransformList::parse(StringView value)
94{
95 clearItems();
96
97 bool parsingSucceeded = readCharactersForParsing(value, [&](auto buffer) {
98 return parseGeneric(buffer);
99 });
100
101 if (!parsingSucceeded)
102 clearItems();
103}
104
105bool SVGTransformList::parse(StringParsingBuffer<LChar>& buffer)
106{
107 return parseGeneric(buffer);
108}
109
110bool SVGTransformList::parse(StringParsingBuffer<UChar>& buffer)
111{
112 return parseGeneric(buffer);
113}

115String SVGTransformList::valueAsString() const
116{
117 StringBuilder builder;
118 for (const auto& transfrom : m_items) {
119 if (builder.length())
120 builder.append(' ');
121
122 builder.append(transfrom->value().valueAsString());
123 }
124 return builder.toString();
125}
