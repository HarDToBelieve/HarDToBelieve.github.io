35inline SVGViewElement::SVGViewElement(const QualifiedName& tagName, Document& document)
36 : SVGElement(tagName, document)
37 , SVGFitToViewBox(this)
38{
39 ASSERT(hasTagName(SVGNames::viewTag));
40}

42Ref<SVGViewElement> SVGViewElement::create(const QualifiedName& tagName, Document& document)
43{
44 return adoptRef(*new SVGViewElement(tagName, document));
45}

47void SVGViewElement::parseAttribute(const QualifiedName& name, const AtomString& value)
48{
49 SVGElement::parseAttribute(name, value);
50 SVGFitToViewBox::parseAttribute(name, value);
51 SVGZoomAndPan::parseAttribute(name, value);
52}
53
54void SVGViewElement::svgAttributeChanged(const QualifiedName& attrName)
55{
56 // We ignore changes to SVGNames::viewTargetAttr, which is deprecated and unused in WebCore.
57 if (PropertyRegistry::isKnownAttribute(attrName))
58 return;
59
60 if (SVGFitToViewBox::isKnownAttribute(attrName)) {
61 if (m_targetElement) {
62 m_targetElement->inheritViewAttributes(*this);
63 if (auto* renderer = m_targetElement->renderer())
64 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
65 }
66
67 return;
68 }
69
70 SVGElement::svgAttributeChanged(attrName);
71}
