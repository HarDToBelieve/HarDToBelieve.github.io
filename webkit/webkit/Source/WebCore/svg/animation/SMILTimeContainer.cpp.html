43SMILTimeContainer::SMILTimeContainer(SVGSVGElement& owner)
44 : m_timer(*this, &SMILTimeContainer::timerFired)
45 , m_ownerSVGElement(owner)
46{
47}

49void SMILTimeContainer::schedule(SVGSMILElement* animation, SVGElement* target, const QualifiedName& attributeName)
50{
51 ASSERT(animation->timeContainer() == this);
52 ASSERT(target);
53 ASSERT(animation->hasValidAttributeName());
54
55 ElementAttributePair key(target, attributeName);
56 auto& animations = m_scheduledAnimations.add(key, AnimationsVector()).iterator->value;
57 ASSERT(!animations.contains(animation));
58 animations.append(animation);
59
60 SMILTime nextFireTime = animation->nextProgressTime();
61 if (nextFireTime.isFinite())
62 notifyIntervalsChanged();
63}
64
65void SMILTimeContainer::unschedule(SVGSMILElement* animation, SVGElement* target, const QualifiedName& attributeName)
66{
67 ASSERT(animation->timeContainer() == this);
68
69 ElementAttributePair key(target, attributeName);
70 auto& animations = m_scheduledAnimations.find(key)->value;
71 ASSERT(animations.contains(animation));
72 animations.removeFirst(animation);
73}

75void SMILTimeContainer::notifyIntervalsChanged()
76{
77 // Schedule updateAnimations() to be called asynchronously so multiple intervals
78 // can change with updateAnimations() only called once at the end.
79 startTimer(elapsed(), 0);
80}

82Seconds SMILTimeContainer::animationFrameDelay() const
83{
84 auto* page = m_ownerSVGElement.document().page();
85 if (!page)
86 return SMILAnimationFrameDelay;
87 return page->isLowPowerModeEnabled() ? SMILAnimationFrameThrottledDelay : SMILAnimationFrameDelay;
88}

90SMILTime SMILTimeContainer::elapsed() const
91{
92 if (!m_beginTime)
93 return 0_s;
94 if (isPaused())
95 return m_accumulatedActiveTime;
96 return MonotonicTime::now() + m_accumulatedActiveTime - m_resumeTime;
97}

114void SMILTimeContainer::begin()
115{
116 ASSERT(!m_beginTime);
117 MonotonicTime now = MonotonicTime::now();
118
119 // If 'm_presetStartTime' is set, the timeline was modified via setElapsed() before the document began.
120 // In this case pass on 'seekToTime=true' to updateAnimations().
121 m_beginTime = m_resumeTime = now - m_presetStartTime;
122 updateAnimations(SMILTime(m_presetStartTime), m_presetStartTime ? true : false);
123 m_presetStartTime = 0_s;
124
125 if (m_pauseTime) {
126 m_pauseTime = now;
127 m_timer.stop();
128 }
129}
130
131void SMILTimeContainer::pause()
132{
133 ASSERT(!isPaused());
134
135 m_pauseTime = MonotonicTime::now();
136 if (m_beginTime) {
137 m_accumulatedActiveTime += m_pauseTime - m_resumeTime;
138 m_timer.stop();
139 }
140}
141
142void SMILTimeContainer::resume()
143{
144 ASSERT(isPaused());
145
146 m_resumeTime = MonotonicTime::now();
147 m_pauseTime = MonotonicTime();
148 startTimer(elapsed(), 0);
149}

151void SMILTimeContainer::setElapsed(SMILTime time)
152{
153 // If the documment didn't begin yet, record a new start time, we'll seek to once its possible.
154 if (!m_beginTime) {
155 m_presetStartTime = Seconds(time.value());
156 return;
157 }
158
159 if (m_beginTime)
160 m_timer.stop();
161
162 MonotonicTime now = MonotonicTime::now();
163 m_beginTime = now - Seconds { time.value() };
164
165 if (m_pauseTime) {
166 m_resumeTime = m_pauseTime = now;
167 m_accumulatedActiveTime = Seconds(time.value());
168 } else
169 m_resumeTime = m_beginTime;
170
171 processScheduledAnimations([](auto& animation) {
172 animation.reset();
173 });
174
175 updateAnimations(time, true);
176}
177
178void SMILTimeContainer::startTimer(SMILTime elapsed, SMILTime fireTime, SMILTime minimumDelay)
179{
180 if (!m_beginTime || isPaused())
181 return;
182
183 if (!fireTime.isFinite())
184 return;
185
186 SMILTime delay = std::max(fireTime - elapsed, minimumDelay);
187 m_timer.startOneShot(1_s * delay.value());
188}
189
190void SMILTimeContainer::timerFired()
191{
192 ASSERT(!!m_beginTime);
193 ASSERT(!m_pauseTime);
194 updateAnimations(elapsed());
195}

197void SMILTimeContainer::updateDocumentOrderIndexes()
198{
199 unsigned timingElementCount = 0;
200
201 for (auto& smilElement : descendantsOfType<SVGSMILElement>(m_ownerSVGElement))
202 smilElement.setDocumentOrderIndex(timingElementCount++);
203
204 m_documentOrderIndexesDirty = false;
205}

207struct PriorityCompare {
208 PriorityCompare(SMILTime elapsed) : m_elapsed(elapsed) {}
209 bool operator()(SVGSMILElement* a, SVGSMILElement* b)
210 {
211 // FIXME: This should also consider possible timing relations between the elements.
212 SMILTime aBegin = a->intervalBegin();
213 SMILTime bBegin = b->intervalBegin();
214 // Frozen elements need to be prioritized based on their previous interval.
215 aBegin = a->isFrozen() && m_elapsed < aBegin ? a->previousIntervalBegin() : aBegin;
216 bBegin = b->isFrozen() && m_elapsed < bBegin ? b->previousIntervalBegin() : bBegin;
217 if (aBegin == bBegin)
218 return a->documentOrderIndex() < b->documentOrderIndex();
219 return aBegin < bBegin;
220 }
221 SMILTime m_elapsed;
222};

224void SMILTimeContainer::sortByPriority(AnimationsVector& animations, SMILTime elapsed)
225{
226 if (m_documentOrderIndexesDirty)
227 updateDocumentOrderIndexes();
228 std::sort(animations.begin(), animations.end(), PriorityCompare(elapsed));
229}

231void SMILTimeContainer::processScheduledAnimations(const Function<void(SVGSMILElement&)>& callback)
232{
233 for (auto& animations : copyToVector(m_scheduledAnimations.values())) {
234 for (auto* animation : animations)
235 callback(*animation);
236 }
237}

239void SMILTimeContainer::updateAnimations(SMILTime elapsed, bool seekToTime)
240{
241 // Don't mutate the DOM while updating the animations.
242 EventQueueScope scope;
243
244 processScheduledAnimations([](auto& animation) {
245 if (!animation.hasConditionsConnected())
246 animation.connectConditions();
247 });
248
249 AnimationsVector animationsToApply;
250 SMILTime earliestFireTime = SMILTime::unresolved();
251
252 for (auto& animations : copyToVector(m_scheduledAnimations.values())) {
253 // Sort according to priority. Elements with later begin time have higher priority.
254 // In case of a tie, document order decides.
255 // FIXME: This should also consider timing relationships between the elements. Dependents
256 // have higher priority.
257 sortByPriority(animations, elapsed);
258
259 RefPtr<SVGSMILElement> firstAnimation;
260 for (auto* animation : animations) {
261 ASSERT(animation->timeContainer() == this);
262 ASSERT(animation->targetElement());
263 ASSERT(animation->hasValidAttributeName());
264
265 // Results are accumulated to the first animation that animates and contributes to a particular element/attribute pair.
266 if (!firstAnimation) {
267 if (!animation->hasValidAttributeType())
268 return;
269 firstAnimation = animation;
270 }
271
272 // This will calculate the contribution from the animation and add it to the resultsElement.
273 if (!animation->progress(elapsed, *firstAnimation, seekToTime) && firstAnimation == animation)
274 firstAnimation = nullptr;
275
276 SMILTime nextFireTime = animation->nextProgressTime();
277 if (nextFireTime.isFinite())
278 earliestFireTime = std::min(nextFireTime, earliestFireTime);
279 }
280
281 if (firstAnimation)
282 animationsToApply.append(firstAnimation.get());
283 }
284
285 // Apply results to target elements.
286 for (auto& animation : animationsToApply)
287 animation->applyResultsToTarget();
288
289 startTimer(elapsed, earliestFireTime, animationFrameDelay());
290}
