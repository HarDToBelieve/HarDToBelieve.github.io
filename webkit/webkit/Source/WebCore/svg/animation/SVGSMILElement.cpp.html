80 {
81 return listener->type() == ConditionEventListenerType

86 bool operator==(const EventListener& other) const final;

88 void disconnectAnimation()
89 {
90 m_animation = nullptr;
91 }

94 ConditionEventListener(SVGSMILElement* animation, SVGSMILElement::Condition* condition)
95 : EventListener(ConditionEventListenerType)

97 , m_condition(condition)

101 void handleEvent(ScriptExecutionContext&, Event&) final;
102
103 SVGSMILElement* m_animation;
104 SVGSMILElement::Condition* m_condition;

175void SVGSMILElement::buildPendingResource()
176{
177 clearResourceReferences();
178
179 if (!isConnected()) {
180 // Reset the target element if we are no longer in the document.
181 setTargetElement(nullptr);
182 return;
183 }
184
185 String id;
186 RefPtr<Element> target;
187 auto& href = getAttribute(SVGNames::hrefAttr, XLinkNames::hrefAttr);
188 if (href.isEmpty())
189 target = parentElement();
190 else {
191 auto result = SVGURIReference::targetElementFromIRIString(href.string(), treeScope());
192 target = WTFMove(result.element);
193 id = WTFMove(result.identifier);
194 }
195 auto svgTarget = makeRefPtr(is<SVGElement>(target) && target->isConnected() ? downcast<SVGElement>(target.get()) : nullptr);
196
197 if (svgTarget != targetElement())
198 setTargetElement(svgTarget.get());
199
200 if (!svgTarget) {
201 // Do not register as pending if we are already pending this resource.
202 if (document().accessSVGExtensions().isPendingResource(*this, id))
203 return;
204
205 if (!id.isEmpty()) {
206 document().accessSVGExtensions().addPendingResource(id, *this);
207 ASSERT(hasPendingResources());
208 }
209 } else
210 svgTarget->addReferencingElement(*this);
211}

213inline QualifiedName SVGSMILElement::constructAttributeName() const
214{
215 auto parseResult = Document::parseQualifiedName(attributeWithoutSynchronization(SVGNames::attributeNameAttr));
216 if (parseResult.hasException())
217 return anyQName();
218
219 auto [prefix, localName] = parseResult.releaseReturnValue();
220
221 if (prefix.isNull())
222 return { nullAtom(), localName, nullAtom() };
223
224 auto namespaceURI = lookupNamespaceURI(prefix);
225 if (namespaceURI.isEmpty())
226 return anyQName();
227
228 return { nullAtom(), localName, namespaceURI };
229}

258Node::InsertedIntoAncestorResult SVGSMILElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
259{
260 SVGElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
261 if (!insertionType.connectedToDocument)
262 return InsertedIntoAncestorResult::Done;
263
264 // Verify we are not in <use> instance tree.
265 ASSERT(!isInShadowTree() || !is<SVGUseElement>(shadowHost()));
266
267 updateAttributeName();
268
269 auto owner = makeRefPtr(ownerSVGElement());
270 if (!owner)
271 return InsertedIntoAncestorResult::Done;
272
273 m_timeContainer = &owner->timeContainer();
274 m_timeContainer->setDocumentOrderIndexesDirty();
275
276 // "If no attribute is present, the default begin value (an offset-value of 0) must be evaluated."
277 if (!hasAttributeWithoutSynchronization(SVGNames::beginAttr))
278 m_beginTimes.append(SMILTimeWithOrigin());
279
280 if (m_isWaitingForFirstInterval)
281 resolveFirstInterval();
282
283 if (m_timeContainer)
284 m_timeContainer->notifyIntervalsChanged();
285
286 return InsertedIntoAncestorResult::NeedsPostInsertionCallback;
287}

313SMILTime SVGSMILElement::parseOffsetValue(const String& data)
314{
315 bool ok;
316 double result = 0;
317 String parse = data.stripWhiteSpace();
318 if (parse.endsWith('h'))
319 result = parse.left(parse.length() - 1).toDouble(&ok) * 60 * 60;
320 else if (parse.endsWith("min"))
321 result = parse.left(parse.length() - 3).toDouble(&ok) * 60;
322 else if (parse.endsWith("ms"))
323 result = parse.left(parse.length() - 2).toDouble(&ok) / 1000;
324 else if (parse.endsWith('s'))
325 result = parse.left(parse.length() - 1).toDouble(&ok);
326 else
327 result = parse.toDouble(&ok);
328 if (!ok || !SMILTime(result).isFinite())
329 return SMILTime::unresolved();
330 return result;
331}
332
333SMILTime SVGSMILElement::parseClockValue(const String& data)
334{
335 if (data.isNull())
336 return SMILTime::unresolved();
337
338 String parse = data.stripWhiteSpace();
339
340 static MainThreadNeverDestroyed<const AtomString> indefiniteValue("indefinite", AtomString::ConstructFromLiteral);
341 if (parse == indefiniteValue)
342 return SMILTime::indefinite();
343
344 double result = 0;
345 bool ok;
346 size_t doublePointOne = parse.find(':');
347 size_t doublePointTwo = parse.find(':', doublePointOne + 1);
348 if (doublePointOne == 2 && doublePointTwo == 5 && parse.length() >= 8) {
349 auto hour = parseInteger<uint8_t>(StringView { parse }.substring(0, 2));
350 auto minute = parseInteger<uint8_t>(StringView { parse }.substring(3, 2));
351 if (!hour || !minute)
352 return SMILTime::unresolved();
353 result = *hour * 60 * 60 + *minute * 60 + parse.substring(6).toDouble(&ok);
354 } else if (doublePointOne == 2 && doublePointTwo == notFound && parse.length() >= 5) {
355 auto minute = parseInteger<uint8_t>(StringView { parse }.substring(0, 2));
356 if (!minute)
357 return SMILTime::unresolved();
358 result = *minute * 60 + parse.substring(3).toDouble(&ok);
359 } else
360 return parseOffsetValue(parse);
361 if (!ok || !SMILTime(result).isFinite())
362 return SMILTime::unresolved();
363 return result;
364}

371bool SVGSMILElement::parseCondition(const String& value, BeginOrEnd beginOrEnd)
372{
373 String parseString = value.stripWhiteSpace();
374
375 double sign = 1.;
376 size_t pos = parseString.find('+');
377 if (pos == notFound) {
378 pos = parseString.find('-');
379 if (pos != notFound)
380 sign = -1.;
381 }
382 String conditionString;
383 SMILTime offset = 0;
384 if (pos == notFound)
385 conditionString = parseString;
386 else {
387 conditionString = parseString.left(pos).stripWhiteSpace();
388 String offsetString = parseString.substring(pos + 1).stripWhiteSpace();
389 offset = parseOffsetValue(offsetString);
390 if (offset.isUnresolved())
391 return false;
392 offset = offset * sign;
393 }
394 if (conditionString.isEmpty())
395 return false;
396 pos = conditionString.find('.');
397
398 String baseID;
399 String nameString;
400 if (pos == notFound)
401 nameString = conditionString;
402 else {
403 baseID = conditionString.left(pos);
404 nameString = conditionString.substring(pos + 1);
405 }
406 if (nameString.isEmpty())
407 return false;
408
409 Condition::Type type;
410 int repeats = -1;
411 if (nameString.startsWith("repeat(") && nameString.endsWith(')')) {
412 // FIXME: For repeat events we just need to add the data carrying TimeEvent class and fire the events at appropiate times.
413 auto parsedRepeat = parseInteger<unsigned>(StringView { nameString }.substring(7, nameString.length() - 8));
414 if (!parsedRepeat)
415 return false;
416 // FIXME: By assigning an unsigned to a signed, this can turn large integers into negative numbers.
417 repeats = *parsedRepeat;
418 nameString = "repeat"_s;
419 type = Condition::EventBase;
420 } else if (nameString == "begin" || nameString == "end") {
421 if (baseID.isEmpty())
422 return false;
423 type = Condition::Syncbase;
424 } else if (nameString.startsWith("accesskey(")) {
425 // FIXME: accesskey() support.
426 type = Condition::AccessKey;
427 } else
428 type = Condition::EventBase;
429
430 m_conditions.append(Condition(type, beginOrEnd, baseID, nameString, offset, repeats));
431
432 if (type == Condition::EventBase && beginOrEnd == End)
433 m_hasEndEventConditions = true;
434
435 return true;
436}

438void SVGSMILElement::parseBeginOrEnd(const String& parseString, BeginOrEnd beginOrEnd)
439{
440 Vector<SMILTimeWithOrigin>& timeList = beginOrEnd == Begin ? m_beginTimes : m_endTimes;
441 if (beginOrEnd == End)
442 m_hasEndEventConditions = false;
443 HashSet<double> existing;
444 for (auto& time : timeList)
445 existing.add(time.time().value());
446 for (auto& string : parseString.split(';')) {
447 SMILTime value = parseClockValue(string);
448 if (value.isUnresolved())
449 parseCondition(string, beginOrEnd);
450 else if (!existing.contains(value.value()))
451 timeList.append(SMILTimeWithOrigin(value, SMILTimeWithOrigin::ParserOrigin));
452 }
453 sortTimeList(timeList);
454}

473void SVGSMILElement::parseAttribute(const QualifiedName& name, const AtomString& value)
474{
475 if (name == SVGNames::beginAttr) {
476 if (!m_conditions.isEmpty()) {
477 disconnectConditions();
478 m_conditions.clear();
479 parseBeginOrEnd(attributeWithoutSynchronization(SVGNames::endAttr), End);
480 }
481 parseBeginOrEnd(value.string(), Begin);
482 if (isConnected())
483 connectConditions();
484 } else if (name == SVGNames::endAttr) {
485 if (!m_conditions.isEmpty()) {
486 disconnectConditions();
487 m_conditions.clear();
488 parseBeginOrEnd(attributeWithoutSynchronization(SVGNames::beginAttr), Begin);
489 }
490 parseBeginOrEnd(value.string(), End);
491 if (isConnected())
492 connectConditions();
493 } else if (name == SVGNames::onendAttr)
494 setAttributeEventListener(eventNames().endEventEvent, name, value);
495 else if (name == SVGNames::onbeginAttr)
496 setAttributeEventListener(eventNames().beginEventEvent, name, value);
497 else
498 SVGElement::parseAttribute(name, value);
499}

501void SVGSMILElement::svgAttributeChanged(const QualifiedName& attrName)
502{
503 if (!isSupportedAttribute(attrName)) {
504 SVGElement::svgAttributeChanged(attrName);
505 return;
506 }
507
508 if (attrName == SVGNames::durAttr)
509 m_cachedDur = invalidCachedTime;
510 else if (attrName == SVGNames::repeatDurAttr)
511 m_cachedRepeatDur = invalidCachedTime;
512 else if (attrName == SVGNames::repeatCountAttr)
513 m_cachedRepeatCount = invalidCachedTime;
514 else if (attrName == SVGNames::minAttr)
515 m_cachedMin = invalidCachedTime;
516 else if (attrName == SVGNames::maxAttr)
517 m_cachedMax = invalidCachedTime;
518 else if (attrName == SVGNames::attributeNameAttr)
519 updateAttributeName();
520 else if (attrName.matches(SVGNames::hrefAttr) || attrName.matches(XLinkNames::hrefAttr)) {
521 InstanceInvalidationGuard guard(*this);
522 buildPendingResource();
523 } else if (isConnected()) {
524 if (attrName == SVGNames::beginAttr)
525 beginListChanged(elapsed());
526 else if (attrName == SVGNames::endAttr)
527 endListChanged(elapsed());
528 }
529
530 animationAttributeChanged();
531}

535 return condition.m_baseID.isEmpty() ? targetElement() : treeScope().getElementById(condition.m_baseID);
536}
537
538void SVGSMILElement::connectConditions()
539{
540 if (m_conditionsConnected)
541 disconnectConditions();
542 m_conditionsConnected = true;
543 for (auto& condition : m_conditions) {
544 if (condition.m_type == Condition::EventBase) {
545 ASSERT(!condition.m_syncbase);
546 auto eventBase = makeRefPtr(eventBaseFor(condition));
547 if (!eventBase)
548 continue;
549 ASSERT(!condition.m_eventListener);
550 condition.m_eventListener = ConditionEventListener::create(this, &condition);
551 eventBase->addEventListener(condition.m_name, *condition.m_eventListener, false);
552 } else if (condition.m_type == Condition::Syncbase) {
553 ASSERT(!condition.m_baseID.isEmpty());
554 condition.m_syncbase = treeScope().getElementById(condition.m_baseID);
555 if (!condition.m_syncbase)
556 continue;
557 if (!is<SVGSMILElement>(*condition.m_syncbase)) {
558 condition.m_syncbase = nullptr;
559 continue;
560 }
561 downcast<SVGSMILElement>(*condition.m_syncbase).addTimeDependent(this);
562 }
563 }
564}
565
566void SVGSMILElement::disconnectConditions()
567{
568 if (!m_conditionsConnected)
569 return;
570 m_conditionsConnected = false;
571 for (auto& condition : m_conditions) {
572 if (condition.m_type == Condition::EventBase) {
573 ASSERT(!condition.m_syncbase);
574 if (!condition.m_eventListener)
575 continue;
576 // Note: It's a memory optimization to try to remove our condition
577 // event listener, but it's not guaranteed to work, since we have
578 // no guarantee that eventBaseFor() will be able to find our condition's
579 // original eventBase. So, we also have to disconnect ourselves from
580 // our condition event listener, in case it later fires.
581 auto eventBase = makeRefPtr(eventBaseFor(condition));
582 if (eventBase)
583 eventBase->removeEventListener(condition.m_name, *condition.m_eventListener, false);
584 condition.m_eventListener->disconnectAnimation();
585 condition.m_eventListener = nullptr;
586 } else if (condition.m_type == Condition::Syncbase) {
587 if (condition.m_syncbase)
588 downcast<SVGSMILElement>(condition.m_syncbase.get())->removeTimeDependent(this);
589 }
590 condition.m_syncbase = nullptr;
591 }
592}

594void SVGSMILElement::setAttributeName(const QualifiedName& attributeName)
595{
596 if (m_timeContainer && m_targetElement && m_attributeName != attributeName) {
597 if (hasValidAttributeName())
598 m_timeContainer->unschedule(this, m_targetElement, m_attributeName);
599 m_attributeName = attributeName;
600 if (hasValidAttributeName())
601 m_timeContainer->schedule(this, m_targetElement, m_attributeName);
602 } else
603 m_attributeName = attributeName;
604
605 // Only clear the animated type, if we had a target before.
606 if (m_targetElement)
607 stopAnimation(m_targetElement);
608}
609
610void SVGSMILElement::setTargetElement(SVGElement* target)
611{
612 if (m_timeContainer && hasValidAttributeName()) {
613 if (m_targetElement)
614 m_timeContainer->unschedule(this, m_targetElement, m_attributeName);
615 if (target)
616 m_timeContainer->schedule(this, target, m_attributeName);
617 }
618
619 if (m_targetElement) {
620 // Clear values that may depend on the previous target.
621 stopAnimation(m_targetElement);
622 disconnectConditions();
623 }
624
625 // If the animation state is not Inactive, always reset to a clear state before leaving the old target element.
626 if (m_activeState != Inactive)
627 endedActiveInterval();
628
629 m_targetElement = target;
630}

685// So a count is not really a time but let just all pretend we did not notice.
686SMILTime SVGSMILElement::repeatCount() const
687{
688 if (m_cachedRepeatCount != invalidCachedTime)
689 return m_cachedRepeatCount;
690 const AtomString& value = attributeWithoutSynchronization(SVGNames::repeatCountAttr);
691 if (value.isNull())
692 return SMILTime::unresolved();
693
694 static MainThreadNeverDestroyed<const AtomString> indefiniteValue("indefinite", AtomString::ConstructFromLiteral);
695 if (value == indefiniteValue)
696 return SMILTime::indefinite();
697 bool ok;
698 double result = value.string().toDouble(&ok);
699 return m_cachedRepeatCount = ok && result > 0 ? result : SMILTime::unresolved();
700}

702SMILTime SVGSMILElement::maxValue() const
703{
704 if (m_cachedMax != invalidCachedTime)
705 return m_cachedMax;
706 const AtomString& value = attributeWithoutSynchronization(SVGNames::maxAttr);
707 SMILTime result = parseClockValue(value);
708 return m_cachedMax = (result.isUnresolved() || result <= 0) ? SMILTime::indefinite() : result;
709}
710
711SMILTime SVGSMILElement::minValue() const
712{
713 if (m_cachedMin != invalidCachedTime)
714 return m_cachedMin;
715 const AtomString& value = attributeWithoutSynchronization(SVGNames::minAttr);
716 SMILTime result = parseClockValue(value);

746SMILTime SVGSMILElement::findInstanceTime(BeginOrEnd beginOrEnd, SMILTime minimumTime, bool equalsMinimumOK) const
747{
748 const Vector<SMILTimeWithOrigin>& list = beginOrEnd == Begin ? m_beginTimes : m_endTimes;
749 int sizeOfList = list.size();
750
751 if (!sizeOfList)
752 return beginOrEnd == Begin ? SMILTime::unresolved() : SMILTime::indefinite();
753
754 const SMILTimeWithOrigin* result = approximateBinarySearch<const SMILTimeWithOrigin, SMILTime>(list, sizeOfList, minimumTime, extractTimeFromVector);
755 int indexOfResult = result - list.begin();
756 ASSERT_WITH_SECURITY_IMPLICATION(indexOfResult < sizeOfList);
757
758 if (list[indexOfResult].time() < minimumTime && indexOfResult < sizeOfList - 1)
759 ++indexOfResult;
760
761 const SMILTime& currentTime = list[indexOfResult].time();
762
763 // The special value "indefinite" does not yield an instance time in the begin list.
764 if (currentTime.isIndefinite() && beginOrEnd == Begin)
765 return SMILTime::unresolved();
766
767 if (currentTime < minimumTime)
768 return beginOrEnd == Begin ? SMILTime::unresolved() : SMILTime::indefinite();
769 if (currentTime > minimumTime)
770 return currentTime;
771
772 ASSERT(currentTime == minimumTime);
773 if (equalsMinimumOK)
774 return currentTime;
775
776 // If the equals is not accepted, return the next bigger item in the list.
777 SMILTime nextTime = currentTime;
778 while (indexOfResult < sizeOfList - 1) {
779 nextTime = list[indexOfResult + 1].time();
780 if (nextTime > minimumTime)
781 return nextTime;
782 ++indexOfResult;
783 }
784
785 return beginOrEnd == Begin ? SMILTime::unresolved() : SMILTime::indefinite();
786}

788SMILTime SVGSMILElement::repeatingDuration() const
789{
790 // Computing the active duration
791 // http://www.w3.org/TR/SMIL2/smil-timing.html#Timing-ComputingActiveDur
792 SMILTime repeatCount = this->repeatCount();
793 SMILTime repeatDur = this->repeatDur();
794 SMILTime simpleDuration = this->simpleDuration();
795 if (!simpleDuration || (repeatDur.isUnresolved() && repeatCount.isUnresolved()))
796 return simpleDuration;
797 SMILTime repeatCountDuration = simpleDuration * repeatCount;
798 return std::min(repeatCountDuration, std::min(repeatDur, SMILTime::indefinite()));
799}

801SMILTime SVGSMILElement::resolveActiveEnd(SMILTime resolvedBegin, SMILTime resolvedEnd) const
802{
803 // Computing the active duration
804 // http://www.w3.org/TR/SMIL2/smil-timing.html#Timing-ComputingActiveDur
805 SMILTime preliminaryActiveDuration;
806 if (!resolvedEnd.isUnresolved() && dur().isUnresolved() && repeatDur().isUnresolved() && repeatCount().isUnresolved())
807 preliminaryActiveDuration = resolvedEnd - resolvedBegin;
808 else if (!resolvedEnd.isFinite())
809 preliminaryActiveDuration = repeatingDuration();
810 else
811 preliminaryActiveDuration = std::min(repeatingDuration(), resolvedEnd - resolvedBegin);
812
813 SMILTime minValue = this->minValue();
814 SMILTime maxValue = this->maxValue();
815 if (minValue > maxValue) {
816 // Ignore both.
817 // http://www.w3.org/TR/2001/REC-smil-animation-20010904/#MinMax
818 minValue = 0;
819 maxValue = SMILTime::indefinite();
820 }
821 return resolvedBegin + std::min(maxValue, std::max(minValue, preliminaryActiveDuration));
822}

824void SVGSMILElement::resolveInterval(bool first, SMILTime& beginResult, SMILTime& endResult) const
825{
826 // See the pseudocode in http://www.w3.org/TR/SMIL3/smil-timing.html#q90.
827 SMILTime beginAfter = first ? -std::numeric_limits<double>::infinity() : m_intervalEnd;
828 SMILTime lastIntervalTempEnd = std::numeric_limits<double>::infinity();
829 while (true) {
830 bool equalsMinimumOK = !first || m_intervalEnd > m_intervalBegin;
831 SMILTime tempBegin = findInstanceTime(Begin, beginAfter, equalsMinimumOK);
832 if (tempBegin.isUnresolved())
833 break;
834 SMILTime tempEnd;
835 if (m_endTimes.isEmpty())
836 tempEnd = resolveActiveEnd(tempBegin, SMILTime::indefinite());
837 else {
838 tempEnd = findInstanceTime(End, tempBegin, true);
839 if ((first && tempBegin == tempEnd && tempEnd == lastIntervalTempEnd) || (!first && tempEnd == m_intervalEnd))
840 tempEnd = findInstanceTime(End, tempBegin, false);
841 if (tempEnd.isUnresolved()) {
842 if (!m_endTimes.isEmpty() && !m_hasEndEventConditions)
843 break;
844 }
845 tempEnd = resolveActiveEnd(tempBegin, tempEnd);
846 }
847 if (!first || (tempEnd > 0 || (!tempBegin.value() && !tempEnd.value()))) {
848 beginResult = tempBegin;
849 endResult = tempEnd;
850 return;
851 }
852
853 beginAfter = tempEnd;
854 lastIntervalTempEnd = tempEnd;
855 }
856 beginResult = SMILTime::unresolved();
857 endResult = SMILTime::unresolved();
858}

860void SVGSMILElement::resolveFirstInterval()
861{
862 SMILTime begin;
863 SMILTime end;
864 resolveInterval(true, begin, end);
865 ASSERT(!begin.isIndefinite());
866
867 if (!begin.isUnresolved() && (begin != m_intervalBegin || end != m_intervalEnd)) {
868 bool wasUnresolved = m_intervalBegin.isUnresolved();
869 m_intervalBegin = begin;
870 m_intervalEnd = end;
871 notifyDependentsIntervalChanged(wasUnresolved ? NewInterval : ExistingInterval);
872 m_nextProgressTime = std::min(m_nextProgressTime, m_intervalBegin);
873
874 if (m_timeContainer)
875 m_timeContainer->notifyIntervalsChanged();
876 }

900void SVGSMILElement::beginListChanged(SMILTime eventTime)
901{
902 if (m_isWaitingForFirstInterval)
903 resolveFirstInterval();
904 else {
905 SMILTime newBegin = findInstanceTime(Begin, eventTime, true);
906 if (newBegin.isFinite() && (m_intervalEnd <= eventTime || newBegin < m_intervalBegin)) {
907 // Begin time changed, re-resolve the interval.
908 SMILTime oldBegin = m_intervalBegin;
909 m_intervalEnd = eventTime;
910 resolveInterval(false, m_intervalBegin, m_intervalEnd);
911 ASSERT(!m_intervalBegin.isUnresolved());
912 if (m_intervalBegin != oldBegin) {
913 if (m_activeState == Active && m_intervalBegin > eventTime) {
914 m_activeState = determineActiveState(eventTime);
915 if (m_activeState != Active)
916 endedActiveInterval();
917 }
918 notifyDependentsIntervalChanged(ExistingInterval);
919 }
920 }
921 }
922 m_nextProgressTime = elapsed();
923
924 if (m_timeContainer)
925 m_timeContainer->notifyIntervalsChanged();
926}

949void SVGSMILElement::checkRestart(SMILTime elapsed)
950{
951 ASSERT(!m_isWaitingForFirstInterval);
952 ASSERT(elapsed >= m_intervalBegin);
953
954 Restart restart = this->restart();
955 if (restart == RestartNever)
956 return;
957
958 if (elapsed < m_intervalEnd) {
959 if (restart != RestartAlways)
960 return;
961 SMILTime nextBegin = findInstanceTime(Begin, m_intervalBegin, false);
962 if (nextBegin < m_intervalEnd) {
963 m_intervalEnd = nextBegin;
964 notifyDependentsIntervalChanged(ExistingInterval);
965 }
966 }
967
968 if (elapsed >= m_intervalEnd)
969 resolveNextInterval(true);
970}

972void SVGSMILElement::seekToIntervalCorrespondingToTime(SMILTime elapsed)
973{
974 ASSERT(!m_isWaitingForFirstInterval);
975 ASSERT(elapsed >= m_intervalBegin);
976
977 // Manually seek from interval to interval, just as if the animation would run regulary.
978 while (true) {
979 // Figure out the next value in the begin time list after the current interval begin.
980 SMILTime nextBegin = findInstanceTime(Begin, m_intervalBegin, false);
981
982 // If the 'nextBegin' time is unresolved (eg. just one defined interval), we're done seeking.
983 if (nextBegin.isUnresolved())
984 return;
985
986 // If the 'nextBegin' time is larger than or equal to the current interval end time, we're done seeking.
987 // If the 'elapsed' time is smaller than the next begin interval time, we're done seeking.
988 if (nextBegin < m_intervalEnd && elapsed >= nextBegin) {
989 // End current interval, and start a new interval from the 'nextBegin' time.
990 m_intervalEnd = nextBegin;
991 resolveNextInterval(false);
992 continue;
993 }
994
995 // If the desired 'elapsed' time is past the current interval, advance to the next.
996 if (elapsed >= m_intervalEnd) {
997 resolveNextInterval(false);
998 continue;
999 }
1000
1001 return;
1002 }

1005float SVGSMILElement::calculateAnimationPercentAndRepeat(SMILTime elapsed, unsigned& repeat) const
1006{
1007 SMILTime simpleDuration = this->simpleDuration();
1008 repeat = 0;
1009 if (simpleDuration.isIndefinite()) {
1010 repeat = 0;
1011 return 0.f;
1012 }
1013 if (!simpleDuration) {
1014 repeat = 0;
1015 return 1.f;
1016 }
1017 ASSERT(m_intervalBegin.isFinite());
1018 ASSERT(simpleDuration.isFinite());
1019 SMILTime activeTime = elapsed - m_intervalBegin;
1020 SMILTime repeatingDuration = this->repeatingDuration();
1021 if (elapsed >= m_intervalEnd || activeTime > repeatingDuration) {
1022 repeat = static_cast<unsigned>(repeatingDuration.value() / simpleDuration.value()) - 1;
1023
1024 double percent = (m_intervalEnd.value() - m_intervalBegin.value()) / simpleDuration.value();
1025 percent = percent - floor(percent);
1026 if (percent < std::numeric_limits<float>::epsilon() || 1 - percent < std::numeric_limits<float>::epsilon())
1027 return 1.0f;
1028 return narrowPrecisionToFloat(percent);
1029 }
1030 repeat = static_cast<unsigned>(activeTime.value() / simpleDuration.value());
1031 SMILTime simpleTime = fmod(activeTime.value(), simpleDuration.value());
1032 return narrowPrecisionToFloat(simpleTime.value() / simpleDuration.value());
1033}

1035SMILTime SVGSMILElement::calculateNextProgressTime(SMILTime elapsed) const
1036{
1037 if (m_timeContainer && m_activeState == Active) {
1038 // If duration is indefinite the value does not actually change over time. Same is true for <set>.
1039 SMILTime simpleDuration = this->simpleDuration();
1040 if (simpleDuration.isIndefinite() || hasTagName(SVGNames::setTag)) {
1041 SMILTime repeatingDurationEnd = m_intervalBegin + repeatingDuration();
1042 // We are supposed to do freeze semantics when repeating ends, even if the element is still active.
1043 // Take care that we get a timer callback at that point.
1044 if (elapsed < repeatingDurationEnd && repeatingDurationEnd < m_intervalEnd && repeatingDurationEnd.isFinite())
1045 return repeatingDurationEnd;
1046 return m_intervalEnd;
1047 }
1048 return elapsed + m_timeContainer->animationFrameDelay();
1049 }
1050 return m_intervalBegin >= elapsed ? m_intervalBegin : SMILTime::unresolved();
1051}

1067bool SVGSMILElement::progress(SMILTime elapsed, SVGSMILElement& firstAnimation, bool seekToTime)
1068{
1069 ASSERT(m_timeContainer);
1070 ASSERT(m_isWaitingForFirstInterval || m_intervalBegin.isFinite());
1071
1072 if (!m_intervalBegin.isFinite()) {
1073 ASSERT(m_activeState == Inactive);
1074 m_nextProgressTime = SMILTime::unresolved();
1075 return false;
1076 }
1077
1078 if (elapsed < m_intervalBegin) {
1079 ASSERT(m_activeState != Active);
1080 if (m_activeState == Frozen) {
1081 if (this == &firstAnimation)
1082 startAnimation();
1083 updateAnimation(m_lastPercent, m_lastRepeat);
1084 }
1085 m_nextProgressTime = m_intervalBegin;
1086 return false;
1087 }
1088
1089 m_previousIntervalBegin = m_intervalBegin;
1090
1091 if (m_isWaitingForFirstInterval) {
1092 m_isWaitingForFirstInterval = false;
1093 resolveFirstInterval();
1094 }
1095
1096 // This call may obtain a new interval -- never call calculateAnimationPercentAndRepeat() before!
1097 if (seekToTime) {
1098 seekToIntervalCorrespondingToTime(elapsed);
1099 if (elapsed < m_intervalBegin) {
1100 // elapsed is not within an interval.
1101 m_nextProgressTime = m_intervalBegin;
1102 return false;
1103 }
1104 }
1105
1106 unsigned repeat = 0;
1107 float percent = calculateAnimationPercentAndRepeat(elapsed, repeat);
1108 checkRestart(elapsed);
1109
1110 ActiveState oldActiveState = m_activeState;
1111 m_activeState = determineActiveState(elapsed);
1112 bool animationIsContributing = isContributing(elapsed);
1113
1114 if (animationIsContributing) {
1115 // Only start the animation of the lowest priority animation that animates and contributes to a particular element/attribute pair.
1116 if (this == &firstAnimation)
1117 startAnimation();
1118
1119 if (oldActiveState == Inactive)
1120 startedActiveInterval();
1121
1122 updateAnimation(percent, repeat);
1123 m_lastPercent = percent;
1124 m_lastRepeat = repeat;
1125 }
1126
1127 if (oldActiveState == Active && m_activeState != Active) {
1128 smilEndEventSender().dispatchEventSoon(*this);
1129 endedActiveInterval();
1130 if (m_activeState != Frozen)
1131 stopAnimation(m_targetElement);
1132 } else if (oldActiveState != Active && m_activeState == Active)
1133 smilBeginEventSender().dispatchEventSoon(*this);
1134
1135 // Triggering all the pending events if the animation timeline is changed.
1136 if (seekToTime) {
1137 if (m_activeState == Inactive || m_activeState == Frozen)
1138 smilEndEventSender().dispatchEventSoon(*this);
1139 }
1140
1141 m_nextProgressTime = calculateNextProgressTime(elapsed);
1142 return animationIsContributing;
1143}

1164 for (auto& condition : m_conditions) {

1166 ASSERT(condition.m_name == "begin" || condition.m_name == "end");
