29#pragma once
30
31#include "EmptyClients.h"
32#include "SVGImage.h"
33
34namespace WebCore {
35
36class SVGImageChromeClient final : public EmptyChromeClient {
37 WTF_MAKE_NONCOPYABLE(SVGImageChromeClient); WTF_MAKE_FAST_ALLOCATED;
38public:
39 SVGImageChromeClient(SVGImage* image)
40 : m_image(image)
41 {
42 }
43
44 bool isSVGImageChromeClient() const final { return true; }
45 SVGImage* image() const { return m_image; }
46
47private:
48 void chromeDestroyed() final
49 {
50 m_image = nullptr;
51 }
52
53 void invalidateContentsAndRootView(const IntRect& r) final
54 {
55 // If m_image->m_page is null, we're being destroyed.
56 if (!m_image || !m_image->m_page)
57 return;
58
59 auto* imageObserver = m_image->imageObserver();
60 if (!imageObserver)
61 return;
62
63 imageObserver->imageFrameAvailable(*m_image, m_image->isAnimating() ? ImageAnimatingState::Yes : ImageAnimatingState::No, &r);
64 }
65
66 bool scheduleRenderingUpdate() final
67 {
68 if (m_image && m_image->imageObserver())
69 m_image->imageObserver()->scheduleRenderingUpdate(*m_image);
70 return true;
71 }
72
73 SVGImage* m_image;
74};
75
76} // namespace WebCore
