25namespace WebCore {
26
27SVGFilter::SVGFilter(const AffineTransform& absoluteTransform, const FloatRect& absoluteSourceDrawingRegion, const FloatRect& targetBoundingBox, const FloatRect& filterRegion, bool effectBBoxMode)
28 : Filter(absoluteTransform)
29 , m_absoluteSourceDrawingRegion(absoluteSourceDrawingRegion)
30 , m_targetBoundingBox(targetBoundingBox)
31 , m_filterRegion(filterRegion)
32 , m_effectBBoxMode(effectBBoxMode)
33{
34 m_absoluteFilterRegion = absoluteTransform.mapRect(filterRegion);
35}
36
37FloatSize SVGFilter::scaledByFilterResolution(FloatSize size) const
38{
39 if (m_effectBBoxMode)
40 size = size * m_targetBoundingBox.size();
41
42 return Filter::scaledByFilterResolution(size) * m_absoluteFilterRegion.size() / m_filterRegion.size();
43}
44
45Ref<SVGFilter> SVGFilter::create(const AffineTransform& absoluteTransform, const FloatRect& absoluteSourceDrawingRegion, const FloatRect& targetBoundingBox, const FloatRect& filterRegion, bool effectBBoxMode)
46{
47 return adoptRef(*new SVGFilter(absoluteTransform, absoluteSourceDrawingRegion, targetBoundingBox, filterRegion, effectBBoxMode));
48}
