26#pragma once
27
28#include "SVGAnimatedProperty.h"
29#include "SVGSharedPrimitiveProperty.h"
30
31namespace WebCore {
32
33template<typename PropertyType>
34class SVGAnimatedPrimitiveProperty : public SVGAnimatedProperty {
35public:
36 using ValueType = PropertyType;
37
38 static Ref<SVGAnimatedPrimitiveProperty> create(SVGElement* contextElement)
39 {
40 return adoptRef(*new SVGAnimatedPrimitiveProperty(contextElement));
41 }
42
43 static Ref<SVGAnimatedPrimitiveProperty> create(SVGElement* contextElement, const PropertyType& value)
44 {
45 return adoptRef(*new SVGAnimatedPrimitiveProperty(contextElement, value));
46 }
47
48 // Used by the DOM.
49 ExceptionOr<void> setBaseVal(const PropertyType& baseVal)
50 {
51 m_baseVal->setValue(baseVal);
52 commitPropertyChange(nullptr);
53 return { };
54 }
55
56 // Used by SVGElement::parseAttribute().
57 void setBaseValInternal(const PropertyType& baseVal) { m_baseVal->setValue(baseVal); }
58 const PropertyType& baseVal() const { return m_baseVal->value(); }
59
60 // Used by SVGAttributeAnimator::progress.
61 void setAnimVal(const PropertyType& animVal)
62 {
63 ASSERT(isAnimating() && m_animVal);
64 m_animVal->setValue(animVal);
65 }
66
67 const PropertyType& animVal() const
68 {
69 ASSERT_IMPLIES(isAnimating(), m_animVal);
70 return isAnimating() ? m_animVal->value() : m_baseVal->value();
71 }
72
73 PropertyType& animVal()
74 {
75 ASSERT_IMPLIES(isAnimating(), m_animVal);
76 return isAnimating() ? m_animVal->value() : m_baseVal->value();
77 }
78
79 // Used when committing a change from the SVGAnimatedProperty to the attribute.
80 String baseValAsString() const override { return m_baseVal->valueAsString(); }
81
82 // Used to apply the SVGAttributeAnimator change to the target element.
83 String animValAsString() const override
84 {
85 ASSERT(isAnimating() && m_animVal);
86 return m_animVal->valueAsString();
87 }
88
89 // Managing the relationship with the owner.
90 void setDirty() override { m_baseVal->setDirty(); }
91 bool isDirty() const override { return m_baseVal->isDirty(); }
92 std::optional<String> synchronize() override { return m_baseVal->synchronize(); }
93
94 // Used by RenderSVGElements and DumpRenderTree.
95 const PropertyType& currentValue() const
96 {
97 ASSERT_IMPLIES(isAnimating(), m_animVal);
98 return isAnimating() ? m_animVal->value() : m_baseVal->value();
99 }
100
101 // Controlling the animation.
102 void startAnimation(SVGAttributeAnimator& animator) override
103 {
104 if (m_animVal)
105 m_animVal->setValue(m_baseVal->value());
106 else
107 ensureAnimVal();
108 SVGAnimatedProperty::startAnimation(animator);
109 }
110
111 void stopAnimation(SVGAttributeAnimator& animator) override
112 {
113 SVGAnimatedProperty::stopAnimation(animator);
114 if (!isAnimating())
115 m_animVal = nullptr;
116 else if (m_animVal)
117 m_animVal->setValue(m_baseVal->value());
118 }
119
120 // Controlling the instance animation.
121 void instanceStartAnimation(SVGAttributeAnimator& animator, SVGAnimatedProperty& animated) override
122 {
123 if (isAnimating())
124 return;
125 m_animVal = static_cast<SVGAnimatedPrimitiveProperty&>(animated).m_animVal;
126 SVGAnimatedProperty::instanceStartAnimation(animator, animated);
127 }
128
129 void instanceStopAnimation(SVGAttributeAnimator& animator) override
130 {
131 if (!isAnimating())
132 return;
133 m_animVal = nullptr;
134 SVGAnimatedProperty::instanceStopAnimation(animator);
135 }
136
137protected:
138 SVGAnimatedPrimitiveProperty(SVGElement* contextElement)
139 : SVGAnimatedProperty(contextElement)
140 , m_baseVal(SVGSharedPrimitiveProperty<PropertyType>::create())
141 {
142 }
143
144 SVGAnimatedPrimitiveProperty(SVGElement* contextElement, const PropertyType& value)
145 : SVGAnimatedProperty(contextElement)
146 , m_baseVal(SVGSharedPrimitiveProperty<PropertyType>::create(value))
147 {
148 }
149
150 RefPtr<SVGSharedPrimitiveProperty<PropertyType>>& ensureAnimVal()
151 {
152 if (!m_animVal)
153 m_animVal = SVGSharedPrimitiveProperty<PropertyType>::create(m_baseVal->value());
154 return m_animVal;
155 }
156
157 Ref<SVGSharedPrimitiveProperty<PropertyType>> m_baseVal;
158 mutable RefPtr<SVGSharedPrimitiveProperty<PropertyType>> m_animVal;
159};
160
161}
