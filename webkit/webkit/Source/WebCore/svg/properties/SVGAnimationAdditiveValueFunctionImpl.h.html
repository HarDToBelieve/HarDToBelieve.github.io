39class SVGAnimationAngleFunction : public SVGAnimationAdditiveValueFunction<SVGAngleValue> {
40public:
41 using Base = SVGAnimationAdditiveValueFunction<SVGAngleValue>;
42 using Base::Base;
43
44 void setFromAndToValues(SVGElement*, const String&, const String&) override
45 {
46 // Values will be set by SVGAnimatedAngleOrientAnimator.
47 ASSERT_NOT_REACHED();
48 }
49
50 void animate(SVGElement*, float progress, unsigned repeatCount, SVGAngleValue& animated)
51 {
52 float number = animated.value();
53 number = Base::animate(progress, repeatCount, m_from.value(), m_to.value(), toAtEndOfDuration().value(), number);
54 animated.setValue(number);
55 }
56
57private:
58 friend class SVGAnimatedAngleOrientAnimator;
59
60 void addFromAndToValues(SVGElement*) override
61 {
62 m_to.setValue(m_to.value() + m_from.value());
63 }
64};

66class SVGAnimationColorFunction : public SVGAnimationAdditiveValueFunction<Color> {
67public:
68 using Base = SVGAnimationAdditiveValueFunction<Color>;
69 using Base::Base;
70
71 void setFromAndToValues(SVGElement* targetElement, const String& from, const String& to) override
72 {
73 m_from = colorFromString(targetElement, from);
74 m_to = colorFromString(targetElement, to);
75 }
76
77 void setToAtEndOfDurationValue(const String& toAtEndOfDuration) override
78 {
79 m_toAtEndOfDuration = SVGPropertyTraits<Color>::fromString(toAtEndOfDuration);
80 }
81
82 void animate(SVGElement*, float progress, unsigned repeatCount, Color& animated)
83 {
84 auto simpleAnimated = animated.toSRGBALossy<uint8_t>();
85 auto simpleFrom = m_animationMode == AnimationMode::To ? simpleAnimated : m_from.toSRGBALossy<uint8_t>();
86 auto simpleTo = m_to.toSRGBALossy<uint8_t>();
87 auto simpleToAtEndOfDuration = toAtEndOfDuration().toSRGBALossy<uint8_t>();
88
89 float red = Base::animate(progress, repeatCount, simpleFrom.red, simpleTo.red, simpleToAtEndOfDuration.red, simpleAnimated.red);
90 float green = Base::animate(progress, repeatCount, simpleFrom.green, simpleTo.green, simpleToAtEndOfDuration.green, simpleAnimated.green);
91 float blue = Base::animate(progress, repeatCount, simpleFrom.blue, simpleTo.blue, simpleToAtEndOfDuration.blue, simpleAnimated.blue);
92 float alpha = Base::animate(progress, repeatCount, simpleFrom.alpha, simpleTo.alpha, simpleToAtEndOfDuration.alpha, simpleAnimated.alpha);
93
94 animated = makeFromComponentsClamping<SRGBA<uint8_t>>(std::lround(red), std::lround(green), std::lround(blue), std::lround(alpha));
95 }
96
97 std::optional<float> calculateDistance(SVGElement*, const String& from, const String& to) const override
98 {
99 Color fromColor = CSSParser::parseColor(from.stripWhiteSpace());
100 if (!fromColor.isValid())
101 return { };
102 Color toColor = CSSParser::parseColor(to.stripWhiteSpace());
103 if (!toColor.isValid())
104 return { };
105
106 auto simpleFrom = fromColor.toSRGBALossy<uint8_t>();
107 auto simpleTo = toColor.toSRGBALossy<uint8_t>();
108
109 float red = simpleFrom.red - simpleTo.red;
110 float green = simpleFrom.green - simpleTo.green;
111 float blue = simpleFrom.blue - simpleTo.blue;
112
113 return std::hypot(red, green, blue);
114 }
115
116private:
117 void addFromAndToValues(SVGElement*) override
118 {
119 auto simpleFrom = m_from.toSRGBALossy<uint8_t>();
120 auto simpleTo = m_to.toSRGBALossy<uint8_t>();
121
122 // Ignores any alpha and sets alpha on result to 100% opaque.
123 m_to = makeFromComponentsClamping<SRGBA<uint8_t>>(simpleTo.red + simpleFrom.red, simpleTo.green + simpleFrom.green, simpleTo.blue + simpleFrom.blue);
124 }
125
126 static Color colorFromString(SVGElement*, const String&);
127};

161class SVGAnimationLengthFunction : public SVGAnimationAdditiveValueFunction<SVGLengthValue> {
162 using Base = SVGAnimationAdditiveValueFunction<SVGLengthValue>;
163
164public:
165 SVGAnimationLengthFunction(AnimationMode animationMode, CalcMode calcMode, bool isAccumulated, bool isAdditive, SVGLengthMode lengthMode)
166 : Base(animationMode, calcMode, isAccumulated, isAdditive)
167 , m_lengthMode(lengthMode)
168 {
169 }
170
171 void setFromAndToValues(SVGElement*, const String& from, const String& to) override
172 {
173 m_from = SVGLengthValue(m_lengthMode, from);
174 m_to = SVGLengthValue(m_lengthMode, to);
175 }
176
177 void setToAtEndOfDurationValue(const String& toAtEndOfDuration) override
178 {
179 m_toAtEndOfDuration = SVGLengthValue(m_lengthMode, toAtEndOfDuration);
180 }
181
182 void animate(SVGElement* targetElement, float progress, unsigned repeatCount, SVGLengthValue& animated)
183 {
184 SVGLengthContext lengthContext(targetElement);
185 SVGLengthType lengthType = progress < 0.5 ? m_from.lengthType() : m_to.lengthType();
186
187 float from = (m_animationMode == AnimationMode::To ? animated : m_from).value(lengthContext);
188 float to = m_to.value(lengthContext);
189 float toAtEndOfDuration = this->toAtEndOfDuration().value(lengthContext);
190 float value = animated.value(lengthContext);
191
192 value = Base::animate(progress, repeatCount, from, to, toAtEndOfDuration, value);
193 animated = { lengthContext, value, lengthType, m_lengthMode };
194 }
195
196 std::optional<float> calculateDistance(SVGElement* targetElement, const String& from, const String& to) const override
197 {
198 SVGLengthContext lengthContext(targetElement);
199 auto fromLength = SVGLengthValue(m_lengthMode, from);
200 auto toLength = SVGLengthValue(m_lengthMode, to);
201 return fabsf(toLength.value(lengthContext) - fromLength.value(lengthContext));
202 }
203
204private:
205 void addFromAndToValues(SVGElement* targetElement) override
206 {
207 SVGLengthContext lengthContext(targetElement);
208 m_to.setValue(lengthContext, m_to.value(lengthContext) + m_from.value(lengthContext));
209 }
210
211 SVGLengthMode m_lengthMode;
212};

250class SVGAnimationPathSegListFunction : public SVGAnimationAdditiveValueFunction<SVGPathByteStream> {
251public:
252 using Base = SVGAnimationAdditiveValueFunction<SVGPathByteStream>;
253 using Base::Base;
254
255 void setFromAndToValues(SVGElement*, const String& from, const String& to) override
256 {
257 m_from = SVGPathByteStream(from);
258 m_to = SVGPathByteStream(to);
259 }
260
261 void setToAtEndOfDurationValue(const String& toAtEndOfDuration) override
262 {
263 m_toAtEndOfDuration = SVGPathByteStream(toAtEndOfDuration);
264 }
265
266 void animate(SVGElement*, float progress, unsigned repeatCount, SVGPathByteStream& animated)
267 {
268 SVGPathByteStream underlyingPath;
269 if (m_animationMode == AnimationMode::To)
270 underlyingPath = animated;
271
272 const SVGPathByteStream& from = m_animationMode == AnimationMode::To ? underlyingPath : m_from;
273
274 // Cache the current animated value before the buildAnimatedSVGPathByteStream() clears animatedPath.
275 SVGPathByteStream lastAnimated;
276 if (!from.size() || (m_isAdditive && m_animationMode != AnimationMode::To))
277 lastAnimated = animated;
278
279 buildAnimatedSVGPathByteStream(from, m_to, animated, progress);
280
281 // Handle additive='sum'.
282 if (!lastAnimated.isEmpty())
283 addToSVGPathByteStream(animated, lastAnimated);
284
285 // Handle accumulate='sum'.
286 if (m_isAccumulated && repeatCount)
287 addToSVGPathByteStream(animated, toAtEndOfDuration(), repeatCount);
288 }
289
290private:
291 void addFromAndToValues(SVGElement*) override
292 {
293 if (!m_from.size() || m_from.size() != m_to.size())
294 return;
295 addToSVGPathByteStream(m_to, m_from);
296 }
297};

299class SVGAnimationRectFunction : public SVGAnimationAdditiveValueFunction<FloatRect> {
300public:
301 using Base = SVGAnimationAdditiveValueFunction<FloatRect>;
302 using Base::Base;
303
304 void setFromAndToValues(SVGElement*, const String& from, const String& to) override
305 {
306 m_from = SVGPropertyTraits<FloatRect>::fromString(from);
307 m_to = SVGPropertyTraits<FloatRect>::fromString(to);
308 }
309
310 void setToAtEndOfDurationValue(const String& toAtEndOfDuration) override
311 {
312 m_toAtEndOfDuration = SVGPropertyTraits<FloatRect>::fromString(toAtEndOfDuration);
313 }
314
315 void animate(SVGElement*, float progress, unsigned repeatCount, FloatRect& animated)
316 {
317 FloatRect from = m_animationMode == AnimationMode::To ? animated : m_from;
318
319 float x = Base::animate(progress, repeatCount, from.x(), m_to.x(), toAtEndOfDuration().x(), animated.x());
320 float y = Base::animate(progress, repeatCount, from.y(), m_to.y(), toAtEndOfDuration().y(), animated.y());
321 float width = Base::animate(progress, repeatCount, from.width(), m_to.width(), toAtEndOfDuration().width(), animated.width());
322 float height = Base::animate(progress, repeatCount, from.height(), m_to.height(), toAtEndOfDuration().height(), animated.height());
323
324 animated = { x, y, width, height };
325 }
326
327private:
328 void addFromAndToValues(SVGElement*) override
329 {
330 m_to += m_from;
331 }
