42 static bool isKnownAttribute(const QualifiedName& attributeName)
43 {
44 return attributeAnimatorCreator().contains(attributeName.impl());
45 }
46
47 RefPtr<SVGAttributeAnimator> createAnimator(const QualifiedName& attributeName, AnimationMode animationMode, CalcMode calcMode, bool isAccumulated, bool isAdditive)
48 {
49 auto iterator = attributeAnimatorCreator().find(attributeName.impl());
50 if (iterator == attributeAnimatorCreator().end())
51 return nullptr;
52
53 auto addResult = m_attributeProperty.ensure(attributeName, [&iterator]() {
54 return iterator->value.first();
55 });
56
57 return iterator->value.second(attributeName, addResult.iterator->value.copyRef(), animationMode, calcMode, isAccumulated, isAdditive);
58 }
59
60 void animatorWillBeDeleted(const QualifiedName& attributeName)
61 {
62 auto iterator = m_attributeProperty.find(attributeName);
63 if (iterator == m_attributeProperty.end())
64 return;
65
66 // If refCount = 1 (in the animator) + 1 (in m_attributeProperty) = 2, the entry can be deleted.
67 if (iterator->value->refCount() == 2)
68 m_attributeProperty.remove(iterator);
69 }

71private:
72 // This HashMap maps an attribute name to a pair of static methods. The first one creates a shared
73 // Ref<SVGProperty> for the value type of this attribute. The second creates the animator given the

78 Function<Ref<SVGProperty>()>,

80 >
81 >;
