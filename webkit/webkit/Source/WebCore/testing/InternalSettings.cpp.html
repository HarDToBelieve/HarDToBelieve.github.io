78void InternalSettings::Backup::restoreTo(Settings& settings)
79{
80 for (const auto& standardFont : m_standardFontFamilies)
81 settings.setStandardFontFamily(standardFont.value, static_cast<UScriptCode>(standardFont.key));
82 m_standardFontFamilies.clear();
83
84 for (const auto& fixedFont : m_fixedFontFamilies)
85 settings.setFixedFontFamily(fixedFont.value, static_cast<UScriptCode>(fixedFont.key));
86 m_fixedFontFamilies.clear();
87
88 for (const auto& serifFont : m_serifFontFamilies)
89 settings.setSerifFontFamily(serifFont.value, static_cast<UScriptCode>(serifFont.key));
90 m_serifFontFamilies.clear();
91
92 for (const auto& sansSerifFont : m_sansSerifFontFamilies)
93 settings.setSansSerifFontFamily(sansSerifFont.value, static_cast<UScriptCode>(sansSerifFont.key));
94 m_sansSerifFontFamilies.clear();
95
96 for (const auto& cursiveFont : m_cursiveFontFamilies)
97 settings.setCursiveFontFamily(cursiveFont.value, static_cast<UScriptCode>(cursiveFont.key));
98 m_cursiveFontFamilies.clear();
99
100 for (const auto& fantasyFont : m_fantasyFontFamilies)
101 settings.setFantasyFontFamily(fantasyFont.value, static_cast<UScriptCode>(fantasyFont.key));
102 m_fantasyFontFamilies.clear();
103
104 for (const auto& pictographFont : m_pictographFontFamilies)
105 settings.setPictographFontFamily(pictographFont.value, static_cast<UScriptCode>(pictographFont.key));
106 m_pictographFontFamilies.clear();
107
108 settings.setMinimumDOMTimerInterval(m_minimumDOMTimerInterval);
109 settings.setTimeWithoutMouseMovementBeforeHidingControls(m_originalTimeWithoutMouseMovementBeforeHidingControls);
110 settings.setEditingBehaviorType(m_originalEditingBehavior);
111 settings.setStorageBlockingPolicy(m_storageBlockingPolicy);
112 settings.setUserInterfaceDirectionPolicy(m_userInterfaceDirectionPolicy);
113 settings.setSystemLayoutDirection(m_systemLayoutDirection);
114 settings.setPdfImageCachingPolicy(m_pdfImageCachingPolicy);
115 settings.setForcedColorsAreInvertedAccessibilityValue(m_forcedColorsAreInvertedAccessibilityValue);
116 settings.setForcedDisplayIsMonochromeAccessibilityValue(m_forcedDisplayIsMonochromeAccessibilityValue);
117 settings.setForcedPrefersContrastAccessibilityValue(m_forcedPrefersContrastAccessibilityValue);
118 settings.setForcedPrefersReducedMotionAccessibilityValue(m_forcedPrefersReducedMotionAccessibilityValue);
119 settings.setFontLoadTimingOverride(m_fontLoadTimingOverride);
120 settings.setFrameFlattening(m_frameFlattening);
121
122 RuntimeEnabledFeatures::sharedFeatures().setFetchAPIKeepAliveEnabled(m_fetchAPIKeepAliveAPIEnabled);
123 RuntimeEnabledFeatures::sharedFeatures().setCustomPasteboardDataEnabled(m_customPasteboardDataEnabled);
124
125#if USE(AUDIO_SESSION)
126 DeprecatedGlobalSettings::setShouldManageAudioSessionCategory(m_shouldManageAudioSessionCategory);
127#endif
128
129#if ENABLE(VIDEO) || ENABLE(WEB_AUDIO)
130 PlatformMediaSessionManager::setShouldDeactivateAudioSession(m_shouldDeactivateAudioSession);
131#endif
132
133 RenderTheme::singleton().setShouldMockBoldSystemFontForAccessibility(m_shouldMockBoldSystemFontForAccessibility);
134 FontCache::singleton().setShouldMockBoldSystemFontForAccessibility(m_shouldMockBoldSystemFontForAccessibility);
135
136#if ENABLE(WEB_AUDIO)
137 AudioContext::setDefaultSampleRateForTesting(std::nullopt);
138#endif
139}

141class InternalSettingsWrapper : public Supplement<Page> {
142 WTF_MAKE_FAST_ALLOCATED;
143public:
144 explicit InternalSettingsWrapper(Page* page)
145 : m_internalSettings(InternalSettings::create(page)) { }
146 virtual ~InternalSettingsWrapper() { m_internalSettings->hostDestroyed(); }
147#if ASSERT_ENABLED
148 bool isRefCountedWrapper() const override { return true; }
149#endif
150 InternalSettings* internalSettings() const { return m_internalSettings.get(); }
151
152private:
153 RefPtr<InternalSettings> m_internalSettings;
154};

156const char* InternalSettings::supplementName()
157{
158 return "InternalSettings";
159}

161InternalSettings* InternalSettings::from(Page* page)
162{
163 if (!Supplement<Page>::from(page, supplementName()))
164 Supplement<Page>::provideTo(page, supplementName(), makeUnique<InternalSettingsWrapper>(page));
165 return static_cast<InternalSettingsWrapper*>(Supplement<Page>::from(page, supplementName()))->internalSettings();
166}

185void InternalSettings::resetToConsistentState()
186{
187 m_page->setPageScaleFactor(1, { 0, 0 });
188 m_page->mainFrame().setPageAndTextZoomFactors(1, 1);
189 m_page->setCanStartMedia(true);
190 setUseDarkAppearanceInternal(false);
191
192 m_backup.restoreTo(settings());
193 m_backup = Backup { settings() };
194
195 InternalSettingsGenerated::resetToConsistentState();
196}

204ExceptionOr<void> InternalSettings::setStandardFontFamily(const String& family, const String& script)
205{
206 if (!m_page)
207 return Exception { InvalidAccessError };
208 UScriptCode code = scriptNameToCode(script);
209 if (code == USCRIPT_INVALID_CODE)
210 return { };
211 m_backup.m_standardFontFamilies.add(code, settings().standardFontFamily(code));
212 settings().setStandardFontFamily(family, code);
213 return { };
214}

460ExceptionOr<void> InternalSettings::setShouldDisplayTrackKind(TrackKind kind, bool enabled)
461{
462 if (!m_page)
463 return Exception { InvalidAccessError };
464#if ENABLE(VIDEO)
465 auto& captionPreferences = m_page->group().captionPreferences();
466 switch (kind) {
467 case TrackKind::Subtitles:
468 captionPreferences.setUserPrefersSubtitles(enabled);
469 break;
470 case TrackKind::Captions:
471 captionPreferences.setUserPrefersCaptions(enabled);
472 break;
473 case TrackKind::TextDescriptions:
474 captionPreferences.setUserPrefersTextDescriptions(enabled);
475 break;
476 }
477#else
478 UNUSED_PARAM(kind);
479 UNUSED_PARAM(enabled);
480#endif
481 return { };
482}
483
484ExceptionOr<bool> InternalSettings::shouldDisplayTrackKind(TrackKind kind)
485{
486 if (!m_page)
487 return Exception { InvalidAccessError };
488#if ENABLE(VIDEO)
489 auto& captionPreferences = m_page->group().captionPreferences();
490 switch (kind) {
491 case TrackKind::Subtitles:

493 case TrackKind::Captions:
494 return captionPreferences.userPrefersCaptions();
495 case TrackKind::TextDescriptions:
496 return captionPreferences.userPrefersTextDescriptions();
497 }
498#else
499 UNUSED_PARAM(kind);
500#endif
501 return false;
502}
