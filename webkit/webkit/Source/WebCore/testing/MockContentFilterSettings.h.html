26#pragma once
27
28#include <wtf/text/WTFString.h>
29
30namespace WebCore {
31
32class MockContentFilterSettings {
33 friend class NeverDestroyed<MockContentFilterSettings>;
34public:
35 enum class DecisionPoint {
36 AfterWillSendRequest,
37 AfterRedirect,
38 AfterResponse,
39 AfterAddData,
40 AfterFinishedAddingData,
41 Never
42 };
43
44 enum class Decision {
45 Allow,
46 Block
47 };
48
49 WEBCORE_TESTSUPPORT_EXPORT static MockContentFilterSettings& singleton();
50 static void reset();
51 static const char* unblockURLHost() { return "mock-unblock"; }
52
53 // Trick the generated bindings into thinking we're RefCounted.
54 void ref() { }
55 void deref() { }
56
57 bool enabled() const { return m_enabled; }
58 WEBCORE_TESTSUPPORT_EXPORT void setEnabled(bool);
59
60 const String& blockedString() const { return m_blockedString; }
61 void setBlockedString(const String& blockedString) { m_blockedString = blockedString; }
62
63 DecisionPoint decisionPoint() const { return m_decisionPoint; }
64 void setDecisionPoint(DecisionPoint decisionPoint) { m_decisionPoint = decisionPoint; }
65
66 Decision decision() const { return m_decision; }
67 void setDecision(Decision decision) { m_decision = decision; }
68
69 Decision unblockRequestDecision() const { return m_unblockRequestDecision; }
70 void setUnblockRequestDecision(Decision unblockRequestDecision) { m_unblockRequestDecision = unblockRequestDecision; }
71
72 const String& unblockRequestURL() const;
73
74 const String& modifiedRequestURL() const { return m_modifiedRequestURL; }
75 void setModifiedRequestURL(const String& modifiedRequestURL) { m_modifiedRequestURL = modifiedRequestURL; }
76
77private:
78 MockContentFilterSettings() = default;
79 MockContentFilterSettings(const MockContentFilterSettings&) = delete;
80 MockContentFilterSettings& operator=(const MockContentFilterSettings&) = default;
81
82 bool m_enabled { false };
83 DecisionPoint m_decisionPoint { DecisionPoint::AfterResponse };
84 Decision m_decision { Decision::Allow };
85 Decision m_unblockRequestDecision { Decision::Block };
86 String m_blockedString;
87 String m_modifiedRequestURL;
88};
89
90} // namespace WebCore
