36class MockGamepadProvider : public GamepadProvider {

38 friend class NeverDestroyed<MockGamepadProvider>;
39public:
40 WEBCORE_TESTSUPPORT_EXPORT static MockGamepadProvider& singleton();
41
42 WEBCORE_TESTSUPPORT_EXPORT void startMonitoringGamepads(GamepadProviderClient&) final;
43 WEBCORE_TESTSUPPORT_EXPORT void stopMonitoringGamepads(GamepadProviderClient&) final;
44 const Vector<PlatformGamepad*>& platformGamepads() final { return m_connectedGamepadVector; }
45 bool isMockGamepadProvider() const final { return true; }
46
47 void setMockGamepadDetails(unsigned index, const String& gamepadID, const String& mapping, unsigned axisCount, unsigned buttonCount);
48 bool setMockGamepadAxisValue(unsigned index, unsigned axisIndex, double value);
49 bool setMockGamepadButtonValue(unsigned index, unsigned buttonIndex, double value);
50 bool connectMockGamepad(unsigned index);
51 bool disconnectMockGamepad(unsigned index);
52
53private:
54 MockGamepadProvider();
55
56 void gamepadInputActivity();
57
58 Vector<PlatformGamepad*> m_connectedGamepadVector;
59 Vector<std::unique_ptr<MockGamepad>> m_mockGamepadVector;
60
61 bool m_shouldScheduleActivityCallback { true };
62};
