40static inline rtc::scoped_refptr<webrtc::PeerConnectionFactoryInterface>& getRealPeerConnectionFactory()
41{
42 static NeverDestroyed<rtc::scoped_refptr<webrtc::PeerConnectionFactoryInterface>> realPeerConnectionFactory;
43 return realPeerConnectionFactory;
44}
45
46static inline webrtc::PeerConnectionFactoryInterface* realPeerConnectionFactory()
47{
48 return getRealPeerConnectionFactory().get();
49}
50
51void useRealRTCPeerConnectionFactory(LibWebRTCProvider& provider)
52{
53 auto& factory = getRealPeerConnectionFactory();
54 if (!factory)
55 return;
56 provider.setPeerConnectionFactory(factory.get());
57 factory = nullptr;
58}

60void useMockRTCPeerConnectionFactory(LibWebRTCProvider* provider, const String& testCase)
61{
62 if (!provider)
63 return;
64
65 if (!realPeerConnectionFactory()) {
66 auto& factory = getRealPeerConnectionFactory();
67 factory = provider->factory();
68 }
69 provider->setPeerConnectionFactory(MockLibWebRTCPeerConnectionFactory::create(testCase));
70}

72MockLibWebRTCPeerConnection::~MockLibWebRTCPeerConnection()
73{
74 // Free senders and receivers in a different thread like an actual peer connection would probably do.
75 Thread::create("MockLibWebRTCPeerConnection thread", [transceivers = WTFMove(m_transceivers)] { });
76}

110void MockLibWebRTCPeerConnectionForIceCandidates::sendCandidates()
111{
112 if (m_delayCount > 0) {
113 m_delayCount--;
114 callOnMainThread([this] {
115 LibWebRTCProvider::callOnWebRTCNetworkThread([this] {
116 sendCandidates();
117 });
118 });
119 return;
120 }
121
122 // Let's gather candidates
123 LibWebRTCProvider::callOnWebRTCSignalingThread([this]() {
124 MockLibWebRTCIceCandidate candidate("candidate:2013266431 1 udp 2013266432 192.168.0.100 38838 typ host generation 0", "1");
125 m_observer.OnIceCandidate(&candidate);
126 });
127
128 LibWebRTCProvider::callOnWebRTCSignalingThread([this]() {
129 MockLibWebRTCIceCandidate candidate("candidate:1019216383 1 tcp 1019216384 192.168.0.100 9 typ host tcptype passive generation 0", "1");
130 m_observer.OnIceCandidate(&candidate);
131 MockLibWebRTCIceCandidate candidateSSLTcp("candidate:1019216384 1 ssltcp 1019216385 192.168.0.100 49888 typ host generation 0", "1");
132 m_observer.OnIceCandidate(&candidateSSLTcp);
133 });
134
135 LibWebRTCProvider::callOnWebRTCSignalingThread([this]() {
136 MockLibWebRTCIceCandidate candidate("candidate:1677722111 1 tcp 1677722112 172.18.0.1 47989 typ srflx raddr 192.168.0.100 rport 47989 generation 0", "1");
137 m_observer.OnIceCandidate(&candidate);
138 });
139
140 LibWebRTCProvider::callOnWebRTCSignalingThread([this]() {
141 m_observer.OnIceGatheringChange(webrtc::PeerConnectionInterface::kIceGatheringComplete);
142 });
143
144 Release();
145}

166template<typename U> static inline void releaseInNetworkThread(MockLibWebRTCPeerConnection& mock, U& observer)
167{
168 mock.AddRef();
169 observer.AddRef();
170 callOnMainThread([&mock, &observer] {
171 LibWebRTCProvider::callOnWebRTCNetworkThread([&mock, &observer]() {
172 observer.Release();
173 mock.Release();
174 });
175 });
176}

220static rtc::scoped_refptr<webrtc::PeerConnectionInterface> createConnection(const String& testCase, webrtc::PeerConnectionObserver& observer)
221{
222 if (testCase == "ICECandidates")
223 return new rtc::RefCountedObject<MockLibWebRTCPeerConnectionForIceCandidates>(observer);
224
225 if (testCase == "ICECandidatesWithDelay")
226 return new rtc::RefCountedObject<MockLibWebRTCPeerConnectionForIceCandidates>(observer, 1000);
227
228 if (testCase == "ICEConnectionState")
229 return new rtc::RefCountedObject<MockLibWebRTCPeerConnectionForIceConnectionState>(observer);
230
231 if (testCase == "LibWebRTCReleasingWhileCreatingOffer")
232 return new rtc::RefCountedObject<MockLibWebRTCPeerConnectionReleasedInNetworkThreadWhileCreatingOffer>(observer);
233
234 if (testCase == "LibWebRTCReleasingWhileGettingStats")
235 return new rtc::RefCountedObject<MockLibWebRTCPeerConnectionReleasedInNetworkThreadWhileGettingStats>(observer);
236
237 if (testCase == "LibWebRTCReleasingWhileSettingDescription")
238 return new rtc::RefCountedObject<MockLibWebRTCPeerConnectionReleasedInNetworkThreadWhileSettingDescription>(observer);
239
240 return new rtc::RefCountedObject<MockLibWebRTCPeerConnection>(observer);
241}

272void MockLibWebRTCPeerConnection::SetRemoteDescription(webrtc::SetSessionDescriptionObserver* observer, webrtc::SessionDescriptionInterface* sessionDescription)
273{
274 std::unique_ptr<webrtc::SessionDescriptionInterface> toBeFreed(sessionDescription);
275 LibWebRTCProvider::callOnWebRTCSignalingThread([observer] {
276 observer->OnSuccess();
277 });
278 ASSERT(sessionDescription);
279 if (sessionDescription->type() == "offer") {
280 std::string sdp;
281 sessionDescription->ToString(&sdp);
282
283 m_isInitiator = false;
284 m_isReceivingAudio = sdp.find("m=audio") != std::string::npos;
285 m_isReceivingVideo = sdp.find("m=video") != std::string::npos;
286 }
287}

297webrtc::RTCErrorOr<rtc::scoped_refptr<webrtc::RtpSenderInterface>> MockLibWebRTCPeerConnection::AddTrack(rtc::scoped_refptr<webrtc::MediaStreamTrackInterface> track, const std::vector<std::string>& streamIds)
298{
299 LibWebRTCProvider::callOnWebRTCSignalingThread([observer = &m_observer] {
300 observer->OnRenegotiationNeeded();
301 });
302
303 if (!streamIds.empty())
304 m_streamLabel = streamIds.front();
305
306 rtc::scoped_refptr<webrtc::RtpSenderInterface> sender = new rtc::RefCountedObject<MockRtpSender>(WTFMove(track));
307 rtc::scoped_refptr<webrtc::RtpReceiverInterface> receiver = new rtc::RefCountedObject<MockRtpReceiver>();
308 rtc::scoped_refptr<MockRtpTransceiver> transceiver = new rtc::RefCountedObject<MockRtpTransceiver>(WTFMove(sender), WTFMove(receiver));
309
310 m_transceivers.append(WTFMove(transceiver));
311 return rtc::scoped_refptr<webrtc::RtpSenderInterface>(m_transceivers.last()->sender());
312}

314bool MockLibWebRTCPeerConnection::RemoveTrack(webrtc::RtpSenderInterface* sender)
315{
316 LibWebRTCProvider::callOnWebRTCSignalingThread([observer = &m_observer] {
317 observer->OnRenegotiationNeeded();
318 });
319 bool isRemoved = false;
320 return m_transceivers.removeFirstMatching([&](auto& transceiver) {
321 if (transceiver->sender().get() != sender)
322 return false;
323 isRemoved = true;
324 return true;
325 });
326}
