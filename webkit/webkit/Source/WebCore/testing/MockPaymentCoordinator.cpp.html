68std::optional<String> MockPaymentCoordinator::validatedPaymentNetwork(const String& paymentNetwork)
69{
70 auto result = m_availablePaymentNetworks.find(paymentNetwork);
71 if (result == m_availablePaymentNetworks.end())
72 return std::nullopt;
73 return *result;
74}

95static uint64_t showCount;
96static uint64_t hideCount;
97
98void MockPaymentCoordinator::dispatchIfShowing(Function<void()>&& function)
99{
100 ASSERT(showCount > hideCount);
101 RunLoop::main().dispatch([currentShowCount = showCount, function = WTFMove(function)]() {
102 if (showCount > hideCount && showCount == currentShowCount)
103 function();
104 });
105}

107bool MockPaymentCoordinator::showPaymentUI(const URL&, const Vector<URL>&, const ApplePaySessionPaymentRequest& request)
108{
109 if (request.shippingContact().pkContact())
110 m_shippingAddress = request.shippingContact().toApplePayPaymentContact(request.version());
111 m_shippingMethods = request.shippingMethods();
112 m_requiredBillingContactFields = request.requiredBillingContactFields();
113 m_requiredShippingContactFields = request.requiredShippingContactFields();
114#if ENABLE(APPLE_PAY_INSTALLMENTS)
115 m_installmentConfiguration = request.installmentConfiguration().applePayInstallmentConfiguration();
116#endif
117#if ENABLE(APPLE_PAY_COUPON_CODE)
118 m_supportsCouponCode = request.supportsCouponCode();
119 m_couponCode = request.couponCode();
120#endif
121#if ENABLE(APPLE_PAY_SHIPPING_CONTACT_EDITING_MODE)
122 m_shippingContactEditingMode = request.shippingContactEditingMode();
123#endif
124#if defined(MockPaymentCoordinatorAdditions_showPaymentUI)
125 MockPaymentCoordinatorAdditions_showPaymentUI
126#endif
127
128 ASSERT(showCount == hideCount);
129 ++showCount;
130 dispatchIfShowing([page = &m_page]() {
131 page->paymentCoordinator().validateMerchant({ URL(), "https://webkit.org/"_s });
132 });
133 return true;
134}

155static Vector<MockPaymentError> convert(Vector<RefPtr<ApplePayError>>&& errors)
156{
157 Vector<MockPaymentError> result;
158 for (auto& error : errors) {
159 if (error)
160 result.append({ error->code(), error->message(), error->contactField() });
161 }
162 return result;
163}

245void MockPaymentCoordinator::cancelPayment()
246{
247 dispatchIfShowing([page = &m_page] {
248 page->paymentCoordinator().didCancelPaymentSession({ });
249 ++hideCount;
250 ASSERT(showCount == hideCount);
251 });
252}

254void MockPaymentCoordinator::completePaymentSession(std::optional<PaymentAuthorizationResult>&& result)
255{
256 auto isFinalState = isFinalStateResult(result);
257 m_errors = convert(WTFMove(result->errors));
258
259 if (!isFinalState)
260 return;
261
262 ++hideCount;
263 ASSERT(showCount == hideCount);
264}
