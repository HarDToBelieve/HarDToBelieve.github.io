35struct MockWebAuthenticationConfiguration {
36 enum class HidStage : bool {
37 Info,
38 Request
39 };
40
41 enum class HidSubStage : bool {
42 Init,
43 Msg
44 };
45
46 enum class HidError : uint8_t {
47 Success,
48 DataNotSent,
49 EmptyReport,
50 WrongChannelId,
51 MaliciousPayload,
52 UnsupportedOptions,
53 WrongNonce
54 };
55
56 enum class NfcError : uint8_t {
57 Success,
58 NoTags,
59 WrongTagType,
60 NoConnections,
61 MaliciousPayload
62 };
63
64 enum class UserVerification : uint8_t {
65 No,
66 Yes,
67 Cancel,
68 Presence
69 };
70
71 struct LocalConfiguration {
72 UserVerification userVerification { UserVerification::No };
73 bool acceptAttestation { false };
74 String privateKeyBase64;
75 String userCertificateBase64;
76 String intermediateCACertificateBase64;
77 String preferredCredentialIdBase64;
78
79 template<class Encoder> void encode(Encoder&) const;
80 template<class Decoder> static std::optional<LocalConfiguration> decode(Decoder&);
81 };
82
83 struct HidConfiguration {
84 Vector<String> payloadBase64;
85 HidStage stage { HidStage::Info };
86 HidSubStage subStage { HidSubStage::Init };
87 HidError error { HidError::Success };
88 bool isU2f { false };
89 bool keepAlive { false };
90 bool fastDataArrival { false };
91 bool continueAfterErrorData { false };
92 bool canDowngrade { false };
93 bool expectCancel { false };
94 bool supportClientPin { false };
95
96 template<class Encoder> void encode(Encoder&) const;
97 template<class Decoder> static std::optional<HidConfiguration> decode(Decoder&);
98 };
99
100 struct NfcConfiguration {
101 NfcError error { NfcError::Success };
102 Vector<String> payloadBase64;
103 bool multipleTags { false };
104 bool multiplePhysicalTags { false };
105
106 template<class Encoder> void encode(Encoder&) const;
107 template<class Decoder> static std::optional<NfcConfiguration> decode(Decoder&);
108 };
109
110 bool silentFailure { false };
111 std::optional<LocalConfiguration> local;
112 std::optional<HidConfiguration> hid;
113 std::optional<NfcConfiguration> nfc;
114
115 template<class Encoder> void encode(Encoder&) const;
116 template<class Decoder> static std::optional<MockWebAuthenticationConfiguration> decode(Decoder&);
117};
118
119template<class Encoder>
120void MockWebAuthenticationConfiguration::LocalConfiguration::encode(Encoder& encoder) const
121{
122 encoder << userVerification << acceptAttestation << privateKeyBase64 << userCertificateBase64 << intermediateCACertificateBase64 << preferredCredentialIdBase64;
123}
124
125template<class Decoder>
126std::optional<MockWebAuthenticationConfiguration::LocalConfiguration> MockWebAuthenticationConfiguration::LocalConfiguration::decode(Decoder& decoder)
127{
128 MockWebAuthenticationConfiguration::LocalConfiguration result;
129
130 std::optional<UserVerification> userVerification;
131 decoder >> userVerification;
132 if (!userVerification)
133 return std::nullopt;
134 result.userVerification = *userVerification;
135
136 std::optional<bool> acceptAttestation;
137 decoder >> acceptAttestation;
138 if (!acceptAttestation)
139 return std::nullopt;
140 result.acceptAttestation = *acceptAttestation;
141
142 std::optional<String> privateKeyBase64;
143 decoder >> privateKeyBase64;
144 if (!privateKeyBase64)
145 return std::nullopt;
146 result.privateKeyBase64 = WTFMove(*privateKeyBase64);
147
148 std::optional<String> userCertificateBase64;
149 decoder >> userCertificateBase64;
150 if (!userCertificateBase64)
151 return std::nullopt;
152 result.userCertificateBase64 = WTFMove(*userCertificateBase64);
153
154 std::optional<String> intermediateCACertificateBase64;
155 decoder >> intermediateCACertificateBase64;
156 if (!intermediateCACertificateBase64)
157 return std::nullopt;
158 result.intermediateCACertificateBase64 = WTFMove(*intermediateCACertificateBase64);
159
160 std::optional<String> preferredCredentialIdBase64;
161 decoder >> preferredCredentialIdBase64;
162 if (!preferredCredentialIdBase64)
163 return std::nullopt;
164 result.preferredCredentialIdBase64 = WTFMove(*preferredCredentialIdBase64);
165
166 return result;
167}

169template<class Encoder>

172 encoder << payloadBase64 << stage << subStage << error << isU2f << keepAlive << fastDataArrival << continueAfterErrorData << canDowngrade << expectCancel << supportClientPin;
173}

175template<class Decoder>
176std::optional<MockWebAuthenticationConfiguration::HidConfiguration> MockWebAuthenticationConfiguration::HidConfiguration::decode(Decoder& decoder)
177{
178 MockWebAuthenticationConfiguration::HidConfiguration result;
179 if (!decoder.decode(result.payloadBase64))
180 return std::nullopt;
181 if (!decoder.decode(result.stage))
182 return std::nullopt;
183 if (!decoder.decode(result.subStage))
184 return std::nullopt;
185 if (!decoder.decode(result.error))
186 return std::nullopt;
187 if (!decoder.decode(result.isU2f))
188 return std::nullopt;
189 if (!decoder.decode(result.keepAlive))
190 return std::nullopt;
191 if (!decoder.decode(result.fastDataArrival))
192 return std::nullopt;
193 if (!decoder.decode(result.continueAfterErrorData))
194 return std::nullopt;
195 if (!decoder.decode(result.canDowngrade))
196 return std::nullopt;
197 if (!decoder.decode(result.expectCancel))
198 return std::nullopt;
199 if (!decoder.decode(result.supportClientPin))
200 return std::nullopt;
201 return result;
202}

225template<class Encoder>
226void MockWebAuthenticationConfiguration::encode(Encoder& encoder) const
227{
228 encoder << silentFailure << local << hid << nfc;
229}
230
231template<class Decoder>
232std::optional<MockWebAuthenticationConfiguration> MockWebAuthenticationConfiguration::decode(Decoder& decoder)
233{
234 MockWebAuthenticationConfiguration result;
235
236 std::optional<bool> silentFailure;
237 decoder >> silentFailure;
238 if (!silentFailure)
239 return std::nullopt;
240 result.silentFailure = *silentFailure;
241
242 std::optional<std::optional<LocalConfiguration>> local;
243 decoder >> local;
244 if (!local)
245 return std::nullopt;
246 result.local = WTFMove(*local);
247
248 std::optional<std::optional<HidConfiguration>> hid;
249 decoder >> hid;
250 if (!hid)
251 return std::nullopt;
252 result.hid = WTFMove(*hid);
253
254 std::optional<std::optional<NfcConfiguration>> nfc;
255 decoder >> nfc;
256 if (!nfc)
257 return std::nullopt;
258 result.nfc = WTFMove(*nfc);
259
260 return result;
261}
