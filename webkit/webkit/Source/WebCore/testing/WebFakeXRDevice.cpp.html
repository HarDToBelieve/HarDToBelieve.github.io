106void SimulatedXRDevice::initializeTrackingAndRendering(PlatformXR::SessionMode)
107{
108 GraphicsContextGLAttributes attributes;
109 attributes.depth = false;
110 attributes.stencil = false;
111 attributes.antialias = false;
112 m_gl = GraphicsContextGL::create(attributes, nullptr);
113
114 if (m_trackingAndRenderingClient) {
115 // WebXR FakeDevice waits for simulateInputConnection calls to add input sources-
116 // There is no way to know how many simulateInputConnection calls will the device receive,
117 // so notify the input sources have been initialized with an empty list. This is not a problem because
118 // WPT tests rely on requestAnimationFrame updates to test the input sources.
119 callOnMainThread([this, weakThis = makeWeakPtr(*this)]() {
120 if (!weakThis)
121 return;
122 if (m_trackingAndRenderingClient)
123 m_trackingAndRenderingClient->sessionDidInitializeInputSources({ });
124 });
125 }
126}
127
128void SimulatedXRDevice::shutDownTrackingAndRendering()
129{
130 if (m_supportsShutdownNotification)
131 simulateShutdownCompleted();
132 stopTimer();
133 if (m_gl) {
134 for (auto layer : m_layers)
135 m_gl->deleteTexture(layer.value);
136 m_gl = nullptr;
137 }
138 m_layers.clear();
139}

147void SimulatedXRDevice::frameTimerFired()
148{
149 FrameData data = m_frameData.copy();
150 data.shouldRender = true;
151
152 for (auto& layer : m_layers) {
153#if USE(IOSURFACE_FOR_XR_LAYER_DATA)
154 data.layers.add(layer.key, FrameData::LayerData { .surface = IOSurface::create(recommendedResolution(PlatformXR::SessionMode::ImmersiveVr), DestinationColorSpace::SRGB()) });
155#else
156 data.layers.add(layer.key, FrameData::LayerData { .opaqueTexture = layer.value });
157#endif
158 }
159
160 for (auto& input : m_inputConnections) {
161 if (input->isConnected())
162 data.inputSources.append(input->getFrameData());
163 }
164
165 if (m_FrameCallback)
166 m_FrameCallback(WTFMove(data));
167}

176std::optional<PlatformXR::LayerHandle> SimulatedXRDevice::createLayerProjection(uint32_t width, uint32_t height, bool alpha)
177{
178 using GL = GraphicsContextGL;
179 if (!m_gl)
180 return std::nullopt;
181 PlatformXR::LayerHandle handle = ++m_layerIndex;
182 auto texture = m_gl->createTexture();
183 auto colorFormat = alpha ? GL::RGBA8 : GL::RGB8;
184
185 m_gl->bindTexture(GL::TEXTURE_2D, texture);
186 m_gl->texParameteri(GL::TEXTURE_2D, GL::TEXTURE_WRAP_S, GL::CLAMP_TO_EDGE);
187 m_gl->texParameteri(GL::TEXTURE_2D, GL::TEXTURE_WRAP_T, GL::CLAMP_TO_EDGE);
188 m_gl->texParameteri(GL::TEXTURE_2D, GL::TEXTURE_MIN_FILTER, GL::LINEAR);
189 m_gl->texParameteri(GL::TEXTURE_2D, GL::TEXTURE_MAG_FILTER, GL::LINEAR);
190 m_gl->texStorage2D(GL::TEXTURE_2D, 1, colorFormat, width, height);
191
192 m_layers.add(handle, texture);
193 return handle;
194}
195
196void SimulatedXRDevice::deleteLayer(PlatformXR::LayerHandle handle)
197{
198 auto it = m_layers.find(handle);
199 if (it != m_layers.end()) {
200 if (m_gl)
201 m_gl->deleteTexture(it->value);
202 m_layers.remove(it);
203 }
204}

216void WebFakeXRDevice::setViews(const Vector<FakeXRViewInit>& views)
217{
218 Vector<PlatformXR::Device::FrameData::View> deviceViews;
219
220 for (auto& viewInit : views) {
221 auto parsedView = parseView(viewInit);
222 if (!parsedView.hasException()) {
223 auto fakeView = parsedView.releaseReturnValue();
224 PlatformXR::Device::FrameData::View view;
225 view.offset = fakeView->offset();
226 if (fakeView->fieldOfView())
227 view.projection = { *fakeView->fieldOfView() };
228 else
229 view.projection = { fakeView->projection() };
230
231 deviceViews.append(view);
232 }
233 }
234
235 m_device.setViews(WTFMove(deviceViews));
236}

278ExceptionOr<PlatformXR::Device::FrameData::Pose> WebFakeXRDevice::parseRigidTransform(const FakeXRRigidTransformInit& init)
279{
280 if (init.position.size() != 3 || init.orientation.size() != 4)
281 return Exception { TypeError };
282
283 PlatformXR::Device::FrameData::Pose pose;
284 pose.position = { init.position[0], init.position[1], init.position[2] };
285 pose.orientation = { init.orientation[0], init.orientation[1], init.orientation[2], init.orientation[3] };
286
287 return pose;
288}
289
290ExceptionOr<Ref<FakeXRView>> WebFakeXRDevice::parseView(const FakeXRViewInit& init)
291{
292 // https://immersive-web.github.io/webxr-test-api/#parse-a-view
293 auto fakeView = FakeXRView::create(init.eye);
294
295 if (init.projectionMatrix.size() != 16)
296 return Exception { TypeError };
297 fakeView->setProjection(init.projectionMatrix);
298
299 auto viewOffset = parseRigidTransform(init.viewOffset);
300 if (viewOffset.hasException())
301 return viewOffset.releaseException();
302 fakeView->setOffset(viewOffset.releaseReturnValue());
303
304 fakeView->setResolution(init.resolution);
305
306 if (init.fieldOfView) {
307 fakeView->setFieldOfView(init.fieldOfView.value());
308 }
309
310 return fakeView;
311}
