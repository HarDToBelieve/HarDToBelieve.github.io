43class FakeXRView final : public RefCounted<FakeXRView> {
44public:
45 static Ref<FakeXRView> create(XREye eye) { return adoptRef(*new FakeXRView(eye)); }
46 using Pose = PlatformXR::Device::FrameData::Pose;
47 using Fov = PlatformXR::Device::FrameData::Fov;
48
49 XREye eye() const { return m_eye; }
50 const Pose& offset() const { return m_offset; }
51 const std::array<float, 16>& projection() const { return m_projection; }
52 const std::optional<Fov>& fieldOfView() const { return m_fov;}
53
54 void setResolution(FakeXRViewInit::DeviceResolution resolution) { m_resolution = resolution; }
55 void setOffset(Pose&& offset) { m_offset = WTFMove(offset); }
56 void setProjection(const Vector<float>&);
57 void setFieldOfView(const FakeXRViewInit::FieldOfViewInit&);
58private:
59 FakeXRView(XREye eye)
60 : m_eye(eye) { }
61
62
63 XREye m_eye;
64 FakeXRViewInit::DeviceResolution m_resolution;
65 Pose m_offset;
66 std::array<float, 16> m_projection;
67 std::optional<Fov> m_fov;
68};

70class SimulatedXRDevice final : public PlatformXR::Device {
71 WTF_MAKE_FAST_ALLOCATED;
72public:
73 SimulatedXRDevice();
74 ~SimulatedXRDevice();
75 void setViews(Vector<FrameData::View>&&);
76 void setNativeBoundsGeometry(const Vector<FakeXRBoundsPoint>&);
77 void setViewerOrigin(const std::optional<FrameData::Pose>&);
78 void setFloorOrigin(std::optional<FrameData::Pose>&& origin) { m_frameData.floorTransform = WTFMove(origin); }
79 void setEmulatedPosition(bool emulated) { m_frameData.isPositionEmulated = emulated; }
80 void setSupportsShutdownNotification(bool supportsShutdownNotification) { m_supportsShutdownNotification = supportsShutdownNotification; }
81 void simulateShutdownCompleted();
82 void scheduleOnNextFrame(Function<void()>&&);
83 void addInputConnection(Ref<WebFakeXRInputController>&& input) { m_inputConnections.append(WTFMove(input)); };
84private:
85 WebCore::IntSize recommendedResolution(PlatformXR::SessionMode) final;
86 void initializeTrackingAndRendering(PlatformXR::SessionMode) final;
87 void shutDownTrackingAndRendering() final;
88 bool supportsSessionShutdownNotification() const final { return m_supportsShutdownNotification; }
89 void initializeReferenceSpace(PlatformXR::ReferenceSpaceType) final { }
90 Vector<PlatformXR::Device::ViewData> views(PlatformXR::SessionMode) const final;
91 void requestFrame(RequestFrameCallback&&) final;
92 std::optional<PlatformXR::LayerHandle> createLayerProjection(uint32_t width, uint32_t height, bool alpha) final;
93 void deleteLayer(PlatformXR::LayerHandle) final;
94
95 void stopTimer();
96 void frameTimerFired();
97
98 PlatformXR::Device::FrameData m_frameData;
99 bool m_supportsShutdownNotification { false };
100 Timer m_frameTimer;
101 RequestFrameCallback m_FrameCallback;
102 RefPtr<WebCore::GraphicsContextGL> m_gl;
103 HashMap<PlatformXR::LayerHandle, PlatformGLObject> m_layers;
104 uint32_t m_layerIndex { 0 };
105 Vector<Ref<WebFakeXRInputController>> m_inputConnections;
106};

108class WebFakeXRDevice final : public RefCounted<WebFakeXRDevice> {
109public:
110 static Ref<WebFakeXRDevice> create() { return adoptRef(*new WebFakeXRDevice()); }

112 void setViews(const Vector<FakeXRViewInit>&);
113 void disconnect(DOMPromiseDeferred<void>&&);
114 void setViewerOrigin(FakeXRRigidTransformInit origin, bool emulatedPosition = false);
115 void clearViewerOrigin() { m_device.setViewerOrigin(std::nullopt); }
116 void simulateVisibilityChange(XRVisibilityState);
117 void setBoundsGeometry(Vector<FakeXRBoundsPoint>&& bounds) { m_device.setNativeBoundsGeometry(WTFMove(bounds)); }
118 void setFloorOrigin(FakeXRRigidTransformInit);
119 void clearFloorOrigin() { m_device.setFloorOrigin(std::nullopt); }
120 void simulateResetPose();
121 Ref<WebFakeXRInputController> simulateInputSourceConnection(const FakeXRInputSourceInit&);

124 void setSupportsShutdownNotification();
125 void simulateShutdown();

129private:

134};
