34using InputSource = PlatformXR::Device::FrameData::InputSource;
35using InputSourceButton = PlatformXR::Device::FrameData::InputSourceButton;
36using InputSourcePose = PlatformXR::Device::FrameData::InputSourcePose;
37using ButtonType = FakeXRButtonStateInit::Type;

39// https://immersive-web.github.io/webxr-gamepads-module/#xr-standard-gamepad-mapping
40constexpr std::array<ButtonType, 5> XR_STANDARD_BUTTONS = { ButtonType::Grip, ButtonType::Touchpad, ButtonType::Thumbstick, ButtonType::OptionalButton, ButtonType::OptionalThumbstick };

42Ref<WebFakeXRInputController> WebFakeXRInputController::create(PlatformXR::InputSourceHandle handle, const FakeXRInputSourceInit& init)
43{
44 return adoptRef(*new WebFakeXRInputController(handle, init));
45}

47WebFakeXRInputController::WebFakeXRInputController(PlatformXR::InputSourceHandle handle, const FakeXRInputSourceInit& init)
48 : m_handle(handle)
49 , m_handeness(init.handedness)
50 , m_targetRayMode(init.targetRayMode)
51 , m_profiles(init.profiles)
52 , m_primarySelected(init.selectionStarted)
53 , m_simulateSelect(init.selectionClicked)
54{
55 setPointerOrigin(init.pointerOrigin, false);
56 setGripOrigin(init.gripOrigin, false);
57 setSupportedButtons(init.supportedButtons);
58}

60void WebFakeXRInputController::setGripOrigin(FakeXRRigidTransformInit gripOrigin, bool emulatedPosition)
61{
62 auto transform = WebFakeXRDevice::parseRigidTransform(gripOrigin);
63 if (transform.hasException())
64 return;
65 m_gripOrigin = InputSourcePose { transform.releaseReturnValue(), emulatedPosition };
66}
67
68void WebFakeXRInputController::setPointerOrigin(FakeXRRigidTransformInit pointerOrigin, bool emulatedPosition)
69{
70 auto transform = WebFakeXRDevice::parseRigidTransform(pointerOrigin);
71 if (transform.hasException())
72 return;
73 m_pointerOrigin = { transform.releaseReturnValue(), emulatedPosition };
74}

86void WebFakeXRInputController::setSupportedButtons(const Vector<FakeXRButtonStateInit>& buttons)
87{
88 m_buttons.clear();
89 for (auto& button : buttons)
90 m_buttons.add(button.buttonType, button);
91}

93void WebFakeXRInputController::updateButtonState(const FakeXRButtonStateInit& init)
94{
95 auto it = m_buttons.find(init.buttonType);
96 if (it != m_buttons.end())
97 it->value = init;

100InputSource WebFakeXRInputController::getFrameData()
101{
102 InputSource state;
103 state.handle = m_handle;
104 state.handeness = m_handeness;
105 state.targetRayMode = m_targetRayMode;
106 state.profiles = m_profiles;
107 state.pointerOrigin = m_pointerOrigin;
108 state.gripOrigin = m_gripOrigin;
109
110 if (m_simulateSelect)
111 m_primarySelected = true;
112
113 // https://immersive-web.github.io/webxr-gamepads-module/#xr-standard-gamepad-mapping
114 // Mimic xr-standard gamepad layout
115
116 // Primary trigger is required and must be at index 0
117 state.buttons.append({
118 .touched = m_primarySelected,
119 .pressed = m_primarySelected,
120 .pressedValue = m_primarySelected ? 1.0f : 0.0f
121 });
122
123 // Next buttons in xr-standard order
124 for (auto buttonType : XR_STANDARD_BUTTONS) {
125 auto data = getButtonOrPlaceholder(buttonType);
126 if (data.button)
127 state.buttons.append(*data.button);
128 if (data.axes)
129 state.axes.appendVector(*data.axes);
130
131 }
132
133 if (m_simulateSelect) {
134 m_primarySelected = false;
135 m_simulateSelect = false;
136 }
137
138 return state;
139}

141WebFakeXRInputController::ButtonOrPlaceholder WebFakeXRInputController::getButtonOrPlaceholder(FakeXRButtonStateInit::Type buttonType) const
142{
143 ButtonOrPlaceholder result;
144
145 auto it = m_buttons.find(buttonType);
146 if (it != m_buttons.end()) {
147 result.button = InputSourceButton {
148 .touched = it->value.touched,
149 .pressed = it->value.pressed,
150 .pressedValue = it->value.pressedValue
151 };
152
153 if (buttonType == ButtonType::Touchpad || buttonType == ButtonType::Thumbstick)
154 result.axes = Vector<float> { it->value.xValue, it->value.yValue };
155
156 } else {
157 // Add a placeholder if needed
158 // Devices that lack one of the optional inputs listed in the tables above MUST preserve their place in the
159 // buttons or axes array, reporting a placeholder button or placeholder axis, respectively.
160 if (buttonType != ButtonType::OptionalButton && buttonType != ButtonType::OptionalThumbstick) {
161 auto priority = std::find(XR_STANDARD_BUTTONS.begin(), XR_STANDARD_BUTTONS.end(), buttonType);
162 ASSERT(priority != XR_STANDARD_BUTTONS.end());
163
164 for (auto it = priority + 1; it != XR_STANDARD_BUTTONS.end(); ++it) {
165 if (m_buttons.contains(*it)) {
166 result.button = InputSourceButton();
167 break;
168 }
169 }
170 }
171
172 if (buttonType == ButtonType::Touchpad && m_buttons.contains(ButtonType::Thumbstick))
173 result.axes = Vector<float> { 0.0, 0.0 };
174 }
175
176 return result;
177}
