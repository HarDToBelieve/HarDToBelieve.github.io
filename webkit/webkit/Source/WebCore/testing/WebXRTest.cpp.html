41void WebXRTest::simulateDeviceConnection(ScriptExecutionContext& context, const FakeXRDeviceInit& init, WebFakeXRDevicePromise&& promise)
42{
43 // https://immersive-web.github.io/webxr-test-api/#dom-xrtest-simulatedeviceconnection
44 context.postTask([this, protectedThis = makeRef(*this), init, promise = WTFMove(promise)] (ScriptExecutionContext& context) mutable {
45 auto device = WebFakeXRDevice::create();
46 auto& simulatedDevice = device->simulatedXRDevice();
47
48 device->setViews(init.views);
49
50 Vector<XRReferenceSpaceType> features;
51 if (init.supportedFeatures) {
52 if (auto* globalObject = context.globalObject()) {
53 for (auto& feature : init.supportedFeatures.value()) {
54 if (auto referenceSpaceType = parseEnumeration<XRReferenceSpaceType>(*globalObject, feature))
55 features.append(referenceSpaceType.value());
56 }
57 }
58 }
59
60 if (init.boundsCoordinates) {
61 if (init.boundsCoordinates->size() < 3) {
62 promise.reject(Exception { TypeError });
63 return;
64 }
65 simulatedDevice.setNativeBoundsGeometry(init.boundsCoordinates.value());
66 }
67
68 if (init.viewerOrigin)
69 device->setViewerOrigin(init.viewerOrigin.value());
70
71 if (init.floorOrigin)
72 device->setFloorOrigin(init.floorOrigin.value());
73
74 Vector<XRSessionMode> supportedModes;
75 if (init.supportedModes) {
76 supportedModes = init.supportedModes.value();
77 if (supportedModes.isEmpty())
78 supportedModes.append(XRSessionMode::Inline);
79 } else {
80 supportedModes.append(XRSessionMode::Inline);
81 if (init.supportsImmersive)
82 supportedModes.append(XRSessionMode::ImmersiveVr);
83 }
84
85 for (auto& mode : supportedModes)
86 simulatedDevice.setSupportedFeatures(mode, features);
87
88 m_context->registerSimulatedXRDeviceForTesting(simulatedDevice);
89
90 promise.resolve(device.get());
91 m_devices.append(WTFMove(device));
92 });
93}

95void WebXRTest::simulateUserActivation(Document& document, XRSimulateUserActivationFunction& function)
96{
97 // https://immersive-web.github.io/webxr-test-api/#dom-xrtest-simulateuseractivation
98 // Invoke function as if it had transient activation.
99 UserGestureIndicator gestureIndicator(ProcessingUserGesture, &document);
100 function.handleEvent();
101}
102
103void WebXRTest::disconnectAllDevices(DOMPromiseDeferred<void>&& promise)
104{
105 for (auto& device : m_devices)
106 m_context->unregisterSimulatedXRDeviceForTesting(device->simulatedXRDevice());
107 m_devices.clear();
108 promise.resolve();
109}
