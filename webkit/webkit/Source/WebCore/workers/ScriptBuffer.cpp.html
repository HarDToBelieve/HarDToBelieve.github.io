31#include "config.h"
32#include "ScriptBuffer.h"

34#include <wtf/text/StringBuilder.h>

36namespace WebCore {
37
38ScriptBuffer::ScriptBuffer(const String& string)
39{
40 auto utf8 = string.utf8();
41 m_buffer = SharedBuffer::create(utf8.data(), utf8.length());
42}
43
44ScriptBuffer ScriptBuffer::empty()
45{
46 return ScriptBuffer { SharedBuffer::create() };
47}
48
49
50String ScriptBuffer::toString() const
51{
52 if (!m_buffer)
53 return String();
54
55 StringBuilder builder;
56 m_buffer->forEachSegment([&](auto& segment) {
57 builder.append(String::fromUTF8(segment.data(), segment.size()));
58 });
59 return builder.toString();
60}
61
62bool ScriptBuffer::containsSingleFileMappedSegment() const
63{
64 return m_buffer && m_buffer->hasOneSegment() && m_buffer->begin()->segment->containsMappedFileData();
65}
66
67void ScriptBuffer::append(const String& string)
68{
69 if (!m_buffer)
70 m_buffer = SharedBuffer::create();
71 auto utf8 = string.utf8();
72 m_buffer->append(utf8.data(), utf8.length());
73}
74
75bool operator==(const ScriptBuffer& a, const ScriptBuffer& b)
76{
77 if (a.buffer() == b.buffer())
78 return true;
79 if (!a.buffer() || !b.buffer())
80 return false;
81 return *a.buffer() == *b.buffer();
82}
83
84bool operator!=(const ScriptBuffer& a, const ScriptBuffer& b)
85{
86 return !(a == b);
87}
