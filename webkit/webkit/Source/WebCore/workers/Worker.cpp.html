71inline Worker::Worker(ScriptExecutionContext& context, JSC::RuntimeFlags runtimeFlags, const Options& options)
72 : ActiveDOMObject(&context)
73 , m_name(options.name)
74 , m_identifier("worker:" + Inspector::IdentifiersFactory::createIdentifier())
75 , m_contextProxy(WorkerGlobalScopeProxy::create(*this))
76 , m_runtimeFlags(runtimeFlags)
77 , m_type(options.type)
78 , m_credentials(options.credentials)
79{
80 static bool addedListener;
81 if (!addedListener) {
82 platformStrategies()->loaderStrategy()->addOnlineStateChangeListener(&networkStateChanged);
83 addedListener = true;
84 }
85
86 auto addResult = allWorkers().add(this);
87 ASSERT_UNUSED(addResult, addResult.isNewEntry);
88}

90ExceptionOr<Ref<Worker>> Worker::create(ScriptExecutionContext& context, JSC::RuntimeFlags runtimeFlags, const String& url, const Options& options)
91{
92 ASSERT(isMainThread());
93
94 // We don't currently support nested workers, so workers can only be created from documents.
95 ASSERT_WITH_SECURITY_IMPLICATION(context.isDocument());
96
97 auto worker = adoptRef(*new Worker(context, runtimeFlags, options));
98
99 worker->suspendIfNeeded();
100
101 bool shouldBypassMainWorldContentSecurityPolicy = context.shouldBypassMainWorldContentSecurityPolicy();
102 auto scriptURL = worker->resolveURL(url, shouldBypassMainWorldContentSecurityPolicy);
103 if (scriptURL.hasException())
104 return scriptURL.releaseException();
105
106 worker->m_shouldBypassMainWorldContentSecurityPolicy = shouldBypassMainWorldContentSecurityPolicy;
107
108 // https://html.spec.whatwg.org/multipage/workers.html#official-moment-of-creation
109 worker->m_workerCreationTime = MonotonicTime::now();
110
111 worker->m_scriptLoader = WorkerScriptLoader::create();
112 auto contentSecurityPolicyEnforcement = shouldBypassMainWorldContentSecurityPolicy ? ContentSecurityPolicyEnforcement::DoNotEnforce : ContentSecurityPolicyEnforcement::EnforceChildSrcDirective;
113
114 ResourceRequest request { scriptURL.releaseReturnValue() };
115 request.setInitiatorIdentifier(worker->m_identifier);
116
117 FetchOptions fetchOptions;
118 fetchOptions.mode = FetchOptions::Mode::SameOrigin;
119 if (worker->m_type == WorkerType::Module)
120 fetchOptions.credentials = worker->m_credentials;
121 else
122 fetchOptions.credentials = FetchOptions::Credentials::SameOrigin;
123 fetchOptions.cache = FetchOptions::Cache::Default;
124 fetchOptions.redirect = FetchOptions::Redirect::Follow;
125 fetchOptions.destination = FetchOptions::Destination::Worker;
126 worker->m_scriptLoader->loadAsynchronously(context, WTFMove(request), WTFMove(fetchOptions), contentSecurityPolicyEnforcement, ServiceWorkersMode::All, worker.get(), WorkerRunLoop::defaultMode());
127
128 return worker;
129}

131Worker::~Worker()
132{
133 ASSERT(isMainThread());
134 ASSERT(scriptExecutionContext()); // The context is protected by worker context proxy, so it cannot be destroyed while a Worker exists.
135 allWorkers().remove(this);
136 m_contextProxy.workerObjectDestroyed();
137}

139ExceptionOr<void> Worker::postMessage(JSC::JSGlobalObject& state, JSC::JSValue messageValue, PostMessageOptions&& options)
140{
141 Vector<RefPtr<MessagePort>> ports;
142 auto message = SerializedScriptValue::create(state, messageValue, WTFMove(options.transfer), ports, SerializationContext::WorkerPostMessage);
143 if (message.hasException())
144 return message.releaseException();
145
146 // Disentangle the port in preparation for sending it to the remote context.
147 auto channels = MessagePort::disentanglePorts(WTFMove(ports));
148 if (channels.hasException())
149 return channels.releaseException();
150
151 m_contextProxy.postMessageToWorkerGlobalScope({ message.releaseReturnValue(), channels.releaseReturnValue() });
152 return { };
153}

197void Worker::didReceiveResponse(unsigned long identifier, const ResourceResponse& response)
198{
199 const URL& responseURL = response.url();
200 if (!responseURL.protocolIsBlob() && !responseURL.protocolIs("file") && !SecurityOrigin::create(responseURL)->isUnique())
201 m_contentSecurityPolicyResponseHeaders = ContentSecurityPolicyResponseHeaders(response);
202 InspectorInstrumentation::didReceiveScriptResponse(scriptExecutionContext(), identifier);
203}

205void Worker::notifyFinished()
206{
207 auto clearLoader = makeScopeExit([this] {
208 m_scriptLoader = nullptr;
209 });
210
211 auto* context = scriptExecutionContext();
212 if (!context)
213 return;
214
215 if (m_scriptLoader->failed()) {
216 queueTaskToDispatchEvent(*this, TaskSource::DOMManipulation, Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::Yes));
217 return;
218 }
219
220 bool isOnline = platformStrategies()->loaderStrategy()->isOnLine();
221 const ContentSecurityPolicyResponseHeaders& contentSecurityPolicyResponseHeaders = m_contentSecurityPolicyResponseHeaders ? m_contentSecurityPolicyResponseHeaders.value() : context->contentSecurityPolicy()->responseHeaders();
222 ReferrerPolicy referrerPolicy = ReferrerPolicy::EmptyString;
223 if (auto policy = parseReferrerPolicy(m_scriptLoader->referrerPolicy(), ReferrerPolicySource::HTTPHeader))
224 referrerPolicy = *policy;
225
226 URL responseURL = m_scriptLoader->responseURL();
227 if (!m_scriptLoader->isRedirected() && m_scriptLoader->responseSource() != ResourceResponse::Source::ServiceWorker) {
228 if (m_scriptLoader->url().hasFragmentIdentifier())
229 responseURL.setFragmentIdentifier(m_scriptLoader->url().fragmentIdentifier());
230 }
231 m_contextProxy.startWorkerGlobalScope(responseURL, m_name, context->userAgent(responseURL), isOnline, m_scriptLoader->script(), contentSecurityPolicyResponseHeaders, m_shouldBypassMainWorldContentSecurityPolicy, m_workerCreationTime, referrerPolicy, m_type, m_credentials, m_runtimeFlags);
232 InspectorInstrumentation::scriptImported(*context, m_scriptLoader->identifier(), m_scriptLoader->script().toString());
233}

235void Worker::dispatchEvent(Event& event)
236{
237 if (m_wasTerminated)
238 return;
239
240 AbstractWorker::dispatchEvent(event);
241 if (is<ErrorEvent>(event) && !event.defaultPrevented() && event.isTrusted() && scriptExecutionContext()) {
242 auto& errorEvent = downcast<ErrorEvent>(event);
243 scriptExecutionContext()->reportException(errorEvent.message(), errorEvent.lineno(), errorEvent.colno(), errorEvent.filename(), nullptr, nullptr);
244 }
245}

256 });
