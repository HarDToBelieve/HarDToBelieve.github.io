48
49class RTCRtpScriptTransform;
50class RTCRtpScriptTransformer;

55class Worker final : public AbstractWorker, public ActiveDOMObject, private WorkerScriptLoaderClient {
56 WTF_MAKE_ISO_ALLOCATED(Worker);
57public:
58 struct Options {
59 WorkerType type;
60 FetchRequestCredentials credentials;
61 String name;
62 };
63 static ExceptionOr<Ref<Worker>> create(ScriptExecutionContext&, JSC::RuntimeFlags, const String& url, const Options&);
64 virtual ~Worker();
65
66 ExceptionOr<void> postMessage(JSC::JSGlobalObject&, JSC::JSValue message, PostMessageOptions&&);
67
68 void terminate();
69 bool wasTerminated() const { return m_wasTerminated; }
70
71 String identifier() const { return m_identifier; }
72 const String& name() const { return m_name; }
73
74 ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
75
76 void dispatchEvent(Event&) final;
77
78#if ENABLE(WEB_RTC)
79 void createRTCRtpScriptTransformer(RTCRtpScriptTransform&, MessageWithMessagePorts&&);
80 void postTaskToWorkerGlobalScope(Function<void(ScriptExecutionContext&)>&&);
81#endif
82
83 WorkerType type() const { return m_type; }
84
85private:
86 explicit Worker(ScriptExecutionContext&, JSC::RuntimeFlags, const Options&);
87
88 EventTargetInterface eventTargetInterface() const final { return WorkerEventTargetInterfaceType; }
89
90 void notifyNetworkStateChange(bool isOnline);
91
92 void didReceiveResponse(unsigned long identifier, const ResourceResponse&) final;
93 void notifyFinished() final;
94
95 // ActiveDOMObject.
96 void stop() final;
97 void suspend(ReasonForSuspension) final;
98 void resume() final;
99 const char* activeDOMObjectName() const final;
100 bool virtualHasPendingActivity() const final;
101
102 static void networkStateChanged(bool isOnLine);
103
104 RefPtr<WorkerScriptLoader> m_scriptLoader;
105 String m_name;
106 String m_identifier;
107 WorkerGlobalScopeProxy& m_contextProxy; // The proxy outlives the worker to perform thread shutdown.
108 std::optional<ContentSecurityPolicyResponseHeaders> m_contentSecurityPolicyResponseHeaders;
109 MonotonicTime m_workerCreationTime;
110 bool m_shouldBypassMainWorldContentSecurityPolicy { false };
111 bool m_isSuspendedForBackForwardCache { false };
112 JSC::RuntimeFlags m_runtimeFlags;
113 Deque<RefPtr<Event>> m_pendingEvents;
114 bool m_wasTerminated { false };
115#if ENABLE(WEB_RTC)
116 HashSet<String> m_transformers;
117#endif
118 WorkerType m_type { WorkerType::Classic };
119 FetchRequestCredentials m_credentials { FetchRequestCredentials::SameOrigin };
120};
