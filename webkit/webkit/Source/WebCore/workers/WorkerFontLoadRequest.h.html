26#pragma once
27
28#include "FontLoadRequest.h"
29#include "ResourceLoaderOptions.h"
30#include "ThreadableLoaderClient.h"
31#include <wtf/URL.h>
32#include <wtf/WeakPtr.h>
33
34namespace WebCore {
35
36class ScriptExecutionContext;
37class SharedBuffer;
38class WorkerGlobalScope;
39
40struct FontCustomPlatformData;
41
42class WorkerFontLoadRequest : public FontLoadRequest, public ThreadableLoaderClient {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 WorkerFontLoadRequest(URL&&, LoadedFromOpaqueSource);
46 ~WorkerFontLoadRequest() = default;
47
48 void load(WorkerGlobalScope&);
49
50private:
51 const URL& url() const final { return m_url; }
52 bool isPending() const final { return !m_isLoading && !m_errorOccurred && !m_data; }
53 bool isLoading() const final { return m_isLoading; }
54 bool errorOccurred() const final { return m_errorOccurred; }
55
56 bool ensureCustomFontData(const AtomString& remoteURI) final;
57 RefPtr<Font> createFont(const FontDescription&, const AtomString& remoteURI, bool syntheticBold, bool syntheticItalic, const FontFeatureSettings&, FontSelectionSpecifiedCapabilities) final;
58
59 void setClient(FontLoadRequestClient*) final;
60
61 bool isWorkerFontLoadRequest() const final { return true; }
62
63 void didReceiveResponse(unsigned long identifier, const ResourceResponse&) final;
64 void didReceiveData(const uint8_t* data, int dataLength) final;
65 void didFinishLoading(unsigned long identifier) final;
66 void didFail(const ResourceError&) final;
67
68 URL m_url;
69 LoadedFromOpaqueSource m_loadedFromOpaqueSource;
70
71 bool m_isLoading { false };
72 bool m_notifyOnClientSet { false };
73 bool m_errorOccurred { false };
74 FontLoadRequestClient* m_fontLoadRequestClient { nullptr };
75
76 WeakPtr<ScriptExecutionContext> m_context;
77 RefPtr<SharedBuffer> m_data;
78 std::unique_ptr<FontCustomPlatformData> m_fontCustomPlatformData;
79};
80
81} // namespace WebCore
