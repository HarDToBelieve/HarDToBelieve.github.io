40namespace WebCore {
41
42class ContentSecurityPolicyResponseHeaders;
43class ScriptBuffer;
44class ScriptExecutionContext;
45class Worker;
46enum class ReferrerPolicy : uint8_t;
47enum class WorkerType : uint8_t;

49// A proxy to talk to the worker context.
50class WorkerGlobalScopeProxy {
51public:
52 static WorkerGlobalScopeProxy& create(Worker&);
53
54 virtual void startWorkerGlobalScope(const URL& scriptURL, const String& name, const String& userAgent, bool isOnline, const ScriptBuffer& sourceCode, const ContentSecurityPolicyResponseHeaders&, bool shouldBypassMainWorldContentSecurityPolicy, MonotonicTime timeOrigin, ReferrerPolicy, WorkerType, FetchRequestCredentials, JSC::RuntimeFlags) = 0;
55 virtual void terminateWorkerGlobalScope() = 0;
56 virtual void postMessageToWorkerGlobalScope(MessageWithMessagePorts&&) = 0;
57 virtual void postTaskToWorkerGlobalScope(Function<void(ScriptExecutionContext&)>&&) = 0;
58 virtual bool hasPendingActivity() const = 0;
59 virtual void workerObjectDestroyed() = 0;
60 virtual void notifyNetworkStateChange(bool isOnline) = 0;
61
62 virtual void suspendForBackForwardCache() = 0;
63 virtual void resumeForBackForwardCache() = 0;
64
65protected:
66 virtual ~WorkerGlobalScopeProxy() = default;
67};
