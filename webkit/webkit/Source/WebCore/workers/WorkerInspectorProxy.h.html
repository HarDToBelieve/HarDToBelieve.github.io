28#include <wtf/HashSet.h>
29#include <wtf/RefPtr.h>
30#include <wtf/URL.h>
31#include <wtf/text/WTFString.h>

33// All of these methods should be called on the Main Thread.
34// Used to send messages to the WorkerInspector on the WorkerThread.
35
36namespace WebCore {
37
38class ScriptExecutionContext;
39class WorkerThread;
40
41enum class WorkerThreadStartMode;
42
43class WorkerInspectorProxy {
44 WTF_MAKE_NONCOPYABLE(WorkerInspectorProxy);
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 WorkerInspectorProxy(const String& identifier);
48 ~WorkerInspectorProxy();
49
50 // A Worker's inspector messages come in and go out through the Page's WorkerAgent.
51 class PageChannel {
52 public:
53 virtual ~PageChannel() = default;
54 virtual void sendMessageFromWorkerToFrontend(WorkerInspectorProxy&, const String&) = 0;
55 };
56
57 static HashSet<WorkerInspectorProxy*>& allWorkerInspectorProxies();
58
59 const URL& url() const { return m_url; }
60 const String& name() const { return m_name; }
61 const String& identifier() const { return m_identifier; }
62 ScriptExecutionContext* scriptExecutionContext() const { return m_scriptExecutionContext.get(); }
63
64 WorkerThreadStartMode workerStartMode(ScriptExecutionContext&);
65 void workerStarted(ScriptExecutionContext*, WorkerThread*, const URL&, const String& name);
66 void workerTerminated();
67
68 void resumeWorkerIfPaused();
69 void connectToWorkerInspectorController(PageChannel&);
70 void disconnectFromWorkerInspectorController();
71 void sendMessageToWorkerInspectorController(const String&);
72 void sendMessageFromWorkerToFrontend(const String&);
73
74private:
75 RefPtr<ScriptExecutionContext> m_scriptExecutionContext;
76 RefPtr<WorkerThread> m_workerThread;
77 String m_identifier;
78 URL m_url;
79 String m_name;
80 PageChannel* m_pageChannel { nullptr };
81};
82
83} // namespace WebCore
