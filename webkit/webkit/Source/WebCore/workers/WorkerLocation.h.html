26#pragma once
27
28#include <wtf/RefCounted.h>
29#include <wtf/URL.h>
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34 class WorkerLocation : public RefCounted<WorkerLocation> {
35 public:
36 static Ref<WorkerLocation> create(URL&& url, String&& origin) { return adoptRef(*new WorkerLocation(WTFMove(url), WTFMove(origin))); }
37
38 const URL& url() const { return m_url; }
39 String href() const;
40
41 // URI decomposition attributes
42 String protocol() const;
43 String host() const;
44 String hostname() const;
45 String port() const;
46 String pathname() const;
47 String search() const;
48 String hash() const;
49 String origin() const;
50
51 private:
52 WorkerLocation(URL&& url, String&& origin)
53 : m_url(WTFMove(url))
54 , m_origin(WTFMove(origin))
55 {
56 }
57
58 URL m_url;
59 String m_origin;
60 };
61
62} // namespace WebCore
