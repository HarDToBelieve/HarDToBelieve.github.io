59WorkerMessagingProxy::WorkerMessagingProxy(Worker& workerObject)
60 : m_scriptExecutionContext(workerObject.scriptExecutionContext())
61 , m_inspectorProxy(makeUnique<WorkerInspectorProxy>(workerObject.identifier()))
62 , m_workerObject(&workerObject)
63{
64 ASSERT((is<Document>(*m_scriptExecutionContext) && isMainThread())
65 || (is<WorkerGlobalScope>(*m_scriptExecutionContext) && downcast<WorkerGlobalScope>(*m_scriptExecutionContext).thread().thread() == &Thread::current()));
66
67 // Nobody outside this class ref counts this object. The original ref
68 // is balanced by the deref in workerGlobalScopeDestroyedInternal.
69}

71WorkerMessagingProxy::~WorkerMessagingProxy()
72{
73 ASSERT(!m_workerObject);
74 ASSERT((is<Document>(*m_scriptExecutionContext) && isMainThread())
75 || (is<WorkerGlobalScope>(*m_scriptExecutionContext) && downcast<WorkerGlobalScope>(*m_scriptExecutionContext).thread().thread() == &Thread::current()));
76}

78void WorkerMessagingProxy::startWorkerGlobalScope(const URL& scriptURL, const String& name, const String& userAgent, bool isOnline, const ScriptBuffer& sourceCode, const ContentSecurityPolicyResponseHeaders& contentSecurityPolicyResponseHeaders, bool shouldBypassMainWorldContentSecurityPolicy, MonotonicTime timeOrigin, ReferrerPolicy referrerPolicy, WorkerType workerType, FetchRequestCredentials credentials, JSC::RuntimeFlags runtimeFlags)
79{
80 // FIXME: This need to be revisited when we support nested worker one day
81 ASSERT(m_scriptExecutionContext);
82 Document& document = downcast<Document>(*m_scriptExecutionContext);
83 WorkerThreadStartMode startMode = m_inspectorProxy->workerStartMode(*m_scriptExecutionContext.get());
84 String identifier = m_inspectorProxy->identifier();
85
86 IDBClient::IDBConnectionProxy* proxy = document.idbConnectionProxy();
87
88 SocketProvider* socketProvider = document.socketProvider();
89
90 WorkerParameters params = { scriptURL, name, identifier, userAgent, isOnline, contentSecurityPolicyResponseHeaders, shouldBypassMainWorldContentSecurityPolicy, timeOrigin, referrerPolicy, workerType, credentials, document.settingsValues() };
91 auto thread = DedicatedWorkerThread::create(params, sourceCode, *this, *this, *this, startMode, document.topOrigin(), proxy, socketProvider, runtimeFlags);
92
93 workerThreadCreated(thread.get());
94 thread->start();
95
96 m_inspectorProxy->workerStarted(m_scriptExecutionContext.get(), thread.ptr(), scriptURL, name);
97}

99void WorkerMessagingProxy::postMessageToWorkerObject(MessageWithMessagePorts&& message)
100{
101 m_scriptExecutionContext->postTask([this, message = WTFMove(message)] (ScriptExecutionContext& context) mutable {
102 Worker* workerObject = this->workerObject();
103 if (!workerObject || askedToTerminate())
104 return;
105
106 auto ports = MessagePort::entanglePorts(context, WTFMove(message.transferredPorts));
107 ActiveDOMObject::queueTaskToDispatchEvent(*workerObject, TaskSource::PostedMessageQueue, MessageEvent::create(WTFMove(ports), message.message.releaseNonNull()));
108 });
109}

111void WorkerMessagingProxy::postTaskToWorkerObject(Function<void(Worker&)>&& function)
112{
113 m_scriptExecutionContext->postTask([this, function = WTFMove(function)](auto&) mutable {
114 auto* workerObject = this->workerObject();
115 if (!workerObject || askedToTerminate())
116 return;
117 function(*workerObject);
118 });
119}

121void WorkerMessagingProxy::postMessageToWorkerGlobalScope(MessageWithMessagePorts&& message)
122{
123 postTaskToWorkerGlobalScope([message = WTFMove(message)](auto& scriptContext) mutable {
124 ASSERT_WITH_SECURITY_IMPLICATION(scriptContext.isWorkerGlobalScope());
125 auto& context = static_cast<DedicatedWorkerGlobalScope&>(scriptContext);
126 auto ports = MessagePort::entanglePorts(scriptContext, WTFMove(message.transferredPorts));
127 context.dispatchEvent(MessageEvent::create(WTFMove(ports), message.message.releaseNonNull()));
128 context.thread().workerObjectProxy().confirmMessageFromWorkerObject(context.hasPendingActivity());
129 });
130}

132void WorkerMessagingProxy::postTaskToWorkerGlobalScope(Function<void(ScriptExecutionContext&)>&& task)
133{
134 if (m_askedToTerminate)
135 return;
136
137 if (!m_workerThread) {
138 m_queuedEarlyTasks.append(makeUnique<ScriptExecutionContext::Task>(WTFMove(task)));
139 return;
140 }
141 ++m_unconfirmedMessageCount;
142 m_workerThread->runLoop().postTask(WTFMove(task));
143}

194void WorkerMessagingProxy::postExceptionToWorkerObject(const String& errorMessage, int lineNumber, int columnNumber, const String& sourceURL)
195{
196 m_scriptExecutionContext->postTask([this, errorMessage = errorMessage.isolatedCopy(), sourceURL = sourceURL.isolatedCopy(), lineNumber, columnNumber] (ScriptExecutionContext&) {
197 Worker* workerObject = this->workerObject();
198 if (!workerObject)
199 return;
200
201 // We don't bother checking the askedToTerminate() flag here, because exceptions should *always* be reported even if the thread is terminated.
202 // This is intentionally different than the behavior in MessageWorkerTask, because terminated workers no longer deliver messages (section 4.6 of the WebWorker spec), but they do report exceptions.
203 ActiveDOMObject::queueTaskToDispatchEvent(*workerObject, TaskSource::DOMManipulation, ErrorEvent::create(errorMessage, sourceURL, lineNumber, columnNumber, { }));
204 });
205}

224void WorkerMessagingProxy::workerThreadCreated(DedicatedWorkerThread& workerThread)
225{
226 m_workerThread = &workerThread;
227
228 if (m_askedToTerminate) {
229 // Worker.terminate() could be called from JS before the thread was created.
230 m_workerThread->stop(nullptr);
231 } else {
232 if (m_askedToSuspend) {
233 m_askedToSuspend = false;
234 m_workerThread->suspend();
235 }
236
237 ASSERT(!m_unconfirmedMessageCount);
238 m_unconfirmedMessageCount = m_queuedEarlyTasks.size();
239 m_workerThreadHadPendingActivity = true; // Worker initialization means a pending activity.
240
241 auto queuedEarlyTasks = WTFMove(m_queuedEarlyTasks);
242 for (auto& task : queuedEarlyTasks)
243 m_workerThread->runLoop().postTask(WTFMove(*task));
244 }
245}

247void WorkerMessagingProxy::workerObjectDestroyed()
248{
249 m_workerObject = nullptr;
250 m_scriptExecutionContext->postTask([this] (ScriptExecutionContext&) {
251 m_mayBeDestroyed = true;
252 if (m_workerThread)
253 terminateWorkerGlobalScope();
254 else
255 workerGlobalScopeDestroyedInternal();
256 });
257}

259void WorkerMessagingProxy::notifyNetworkStateChange(bool isOnline)
260{
261 if (m_askedToTerminate)
262 return;
263
264 if (!m_workerThread)
265 return;
266
267 m_workerThread->runLoop().postTask([isOnline] (ScriptExecutionContext& context) {
268 auto& globalScope = downcast<WorkerGlobalScope>(context);
269 globalScope.setIsOnline(isOnline);
270 globalScope.dispatchEvent(Event::create(isOnline ? eventNames().onlineEvent : eventNames().offlineEvent, Event::CanBubble::No, Event::IsCancelable::No));
271 });
272}

288void WorkerMessagingProxy::workerGlobalScopeDestroyedInternal()
289{
290 // This is always the last task to be performed, so the proxy is not needed for communication
291 // in either side any more. However, the Worker object may still exist, and it assumes that the proxy exists, too.
292 m_askedToTerminate = true;
293 m_workerThread = nullptr;
294
295 m_inspectorProxy->workerTerminated();
296
297 // This balances the original ref in construction.
298 if (m_mayBeDestroyed)
299 deref();
300}
301
302void WorkerMessagingProxy::terminateWorkerGlobalScope()
303{
304 if (m_askedToTerminate)
305 return;
306 m_askedToTerminate = true;
307
308 m_inspectorProxy->workerTerminated();
309
310 if (m_workerThread)
311 m_workerThread->stop(nullptr);
312}

314void WorkerMessagingProxy::confirmMessageFromWorkerObject(bool hasPendingActivity)
315{
316 m_scriptExecutionContext->postTask([this, hasPendingActivity] (ScriptExecutionContext&) {
317 reportPendingActivityInternal(true, hasPendingActivity);
318 });
319}
320
321void WorkerMessagingProxy::reportPendingActivity(bool hasPendingActivity)
322{
323 m_scriptExecutionContext->postTask([this, hasPendingActivity] (ScriptExecutionContext&) {
324 reportPendingActivityInternal(false, hasPendingActivity);
325 });
326}
327
328void WorkerMessagingProxy::reportPendingActivityInternal(bool confirmingMessage, bool hasPendingActivity)
329{
330 if (confirmingMessage && !m_askedToTerminate) {
331 ASSERT(m_unconfirmedMessageCount);
332 --m_unconfirmedMessageCount;
333 }
334
335 m_workerThreadHadPendingActivity = hasPendingActivity;
336}
337
338bool WorkerMessagingProxy::hasPendingActivity() const
339{
340 return (m_unconfirmedMessageCount || m_workerThreadHadPendingActivity) && !m_askedToTerminate;
341}
