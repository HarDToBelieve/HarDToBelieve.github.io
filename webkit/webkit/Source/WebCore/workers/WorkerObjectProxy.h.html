33#include "MessageWithMessagePorts.h"
34#include "WorkerReportingProxy.h"

36#include <memory>
37#include <wtf/Function.h>
38
39namespace WebCore {
40
41class Worker;
42
43// A proxy to talk to the worker object.
44class WorkerObjectProxy : public WorkerReportingProxy {
45public:
46 virtual void postMessageToWorkerObject(MessageWithMessagePorts&&) = 0;
47 virtual void postTaskToWorkerObject(Function<void(Worker&)>&&) { };
48
49 virtual void confirmMessageFromWorkerObject(bool hasPendingActivity) = 0;
50 virtual void reportPendingActivity(bool hasPendingActivity) = 0;
51
52 // No need to notify the parent page context when dedicated workers are closing.
53 void workerGlobalScopeClosed() override { }
54};
55
56} // namespace WebCore
