43WorkerOrWorkletGlobalScope::WorkerOrWorkletGlobalScope(WorkerThreadType type, Ref<JSC::VM>&& vm, WorkerOrWorkletThread* thread)

53void WorkerOrWorkletGlobalScope::prepareForDestruction()
54{
55 if (m_defaultTaskGroup) {
56 m_defaultTaskGroup->markAsReadyToStop();
57 ASSERT(m_defaultTaskGroup->isStoppedPermanently());
58 }
59
60 stopActiveDOMObjects();
61
62 // Event listeners would keep DOMWrapperWorld objects alive for too long. Also, they have references to JS objects,
63 // which become dangling once Heap is destroyed.
64 removeAllEventListeners();
65
66 // MicrotaskQueue and RejectedPromiseTracker reference Heap.
67 if (m_eventLoop)
68 m_eventLoop->clearMicrotaskQueue();
69 removeRejectedPromiseTracker();
70
71 m_inspectorController->workerTerminating();
72}

99EventLoopTaskGroup& WorkerOrWorkletGlobalScope::eventLoop()
100{
101 ASSERT(isContextThread());
102 if (UNLIKELY(!m_defaultTaskGroup)) {
103 m_eventLoop = WorkerEventLoop::create(*this);
104 m_defaultTaskGroup = makeUnique<EventLoopTaskGroup>(*m_eventLoop);
105 if (activeDOMObjectsAreStopped())
106 m_defaultTaskGroup->stopAndDiscardAllTasks();
107 }
108 return *m_defaultTaskGroup;
109}
110
111bool WorkerOrWorkletGlobalScope::isContextThread() const
112{
113 auto* thread = workerOrWorkletThread();
114 return thread ? thread->thread() == &Thread::current() : isMainThread();
115}

117void WorkerOrWorkletGlobalScope::postTask(Task&& task)
118{
119 ASSERT(workerOrWorkletThread());
120 workerOrWorkletThread()->runLoop().postTask(WTFMove(task));
121}
