33namespace WebCore {
34
35class EventLoopTaskGroup;
36class ScriptModuleLoader;
37class WorkerEventLoop;
38class WorkerInspectorController;
39class WorkerOrWorkletScriptController;
40class WorkerOrWorkletThread;
41
42class WorkerOrWorkletGlobalScope : public ScriptExecutionContext, public RefCounted<WorkerOrWorkletGlobalScope>, public EventTargetWithInlineData {
43 WTF_MAKE_ISO_ALLOCATED(WorkerOrWorkletGlobalScope);
44 WTF_MAKE_NONCOPYABLE(WorkerOrWorkletGlobalScope);
45public:
46 virtual ~WorkerOrWorkletGlobalScope();
47
48 bool isClosing() const { return m_isClosing; }
49 WorkerOrWorkletThread* workerOrWorkletThread() const { return m_thread; }
50
51 WorkerOrWorkletScriptController* script() const { return m_script.get(); }
52 void clearScript();
53
54 JSC::VM& vm() final;
55 WorkerInspectorController& inspectorController() const { return *m_inspectorController; }
56
57 unsigned long createUniqueIdentifier() { return m_uniqueIdentifier++; }
58
59 ScriptModuleLoader& moduleLoader() { return *m_moduleLoader; }
60
61 // ScriptExecutionContext.
62 EventLoopTaskGroup& eventLoop() final;
63 bool isContextThread() const final;
64 void postTask(Task&&) final; // Executes the task on context's thread asynchronously.
65
66 virtual void prepareForDestruction();
67
68 using RefCounted::ref;
69 using RefCounted::deref;
70
71 virtual void suspend() { }
72 virtual void resume() { }
73
74 virtual FetchOptions::Destination destination() const = 0;
75
76protected:
77 WorkerOrWorkletGlobalScope(WorkerThreadType, Ref<JSC::VM>&&, WorkerOrWorkletThread*);
78
79 // ScriptExecutionContext.
80 bool isJSExecutionForbidden() const final;
81
82 void markAsClosing() { m_isClosing = true; }
83
84private:
85 // ScriptExecutionContext.
86 void disableEval(const String& errorMessage) final;
87 void disableWebAssembly(const String& errorMessage) final;
88 void refScriptExecutionContext() final { ref(); }
89 void derefScriptExecutionContext() final { deref(); }
90
91 // EventTarget.
92 ScriptExecutionContext* scriptExecutionContext() const final { return const_cast<WorkerOrWorkletGlobalScope*>(this); }
93 void refEventTarget() final { ref(); }
94 void derefEventTarget() final { deref(); }
95
96 std::unique_ptr<WorkerOrWorkletScriptController> m_script;
97 std::unique_ptr<ScriptModuleLoader> m_moduleLoader;
98 WorkerOrWorkletThread* m_thread;
99 RefPtr<WorkerEventLoop> m_eventLoop;
100 std::unique_ptr<EventLoopTaskGroup> m_defaultTaskGroup;
101 std::unique_ptr<WorkerInspectorController> m_inspectorController;
102 unsigned long m_uniqueIdentifier { 1 };
103 bool m_isClosing { false };
104};
105
106} // namespace WebCore

108SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::WorkerOrWorkletGlobalScope)
109 static bool isType(const WebCore::ScriptExecutionContext& context) { return context.isWorkerGlobalScope() || context.isWorkletGlobalScope(); }
110SPECIALIZE_TYPE_TRAITS_END()
