39class Exception;
40class JSGlobalObject;
41class VM;
42}
43
44namespace WebCore {
45
46class Exception;
47class JSDOMGlobalObject;
48class ScriptSourceCode;
49class WorkerConsoleClient;
50class WorkerOrWorkletGlobalScope;
51class WorkerScriptFetcher;
52
53class WorkerOrWorkletScriptController {
54 WTF_MAKE_NONCOPYABLE(WorkerOrWorkletScriptController);
55 WTF_MAKE_FAST_ALLOCATED;
56public:
57 WorkerOrWorkletScriptController(WorkerThreadType, Ref<JSC::VM>&&, WorkerOrWorkletGlobalScope*);
58 explicit WorkerOrWorkletScriptController(WorkerThreadType, WorkerOrWorkletGlobalScope*);
59 ~WorkerOrWorkletScriptController();
60
61 void releaseHeapAccess();
62 void acquireHeapAccess();
63
64 void addTimerSetNotification(JSC::JSRunLoopTimer::TimerNotificationCallback);
65 void removeTimerSetNotification(JSC::JSRunLoopTimer::TimerNotificationCallback);
66
67 JSDOMGlobalObject* globalScopeWrapper()
68 {
69 initScriptIfNeeded();
70 return m_globalScopeWrapper.get();
71 }
72
73 void attachDebugger(JSC::Debugger*);
74 void detachDebugger(JSC::Debugger*);
75
76 // Async request to terminate a JS run execution. Eventually causes termination
77 // exception raised during JS execution, if the worker thread happens to run JS.
78 // After JS execution was terminated in this way, the Worker thread has to use
79 // forbidExecution()/isExecutionForbidden() to guard against reentry into JS.
80 // Can be called from any thread.
81 void scheduleExecutionTermination();
82 bool isTerminatingExecution() const;
83
84 // Called on Worker thread when JS exits with termination exception caused by forbidExecution() request,
85 // or by Worker thread termination code to prevent future entry into JS.
86 void forbidExecution();
87 bool isExecutionForbidden() const;
88
89 JSC::VM& vm() { return *m_vm; }
90
91 void setException(JSC::Exception*);
92
93 void disableEval(const String& errorMessage);
94 void disableWebAssembly(const String& errorMessage);
95
96 void evaluate(const ScriptSourceCode&, String* returnedExceptionMessage = nullptr);
97 void evaluate(const ScriptSourceCode&, NakedPtr<JSC::Exception>& returnedException, String* returnedExceptionMessage = nullptr);
98
99 JSC::JSValue evaluateModule(JSC::JSModuleRecord&, JSC::JSValue awaitedValue, JSC::JSValue resumeMode);
100
101 void linkAndEvaluateModule(WorkerScriptFetcher&, const ScriptSourceCode&, String* returnedExceptionMessage = nullptr);
102 MessageQueueWaitResult loadModuleSynchronously(WorkerScriptFetcher&, const ScriptSourceCode&);
103
104 void loadAndEvaluateModule(const URL& moduleURL, FetchOptions::Credentials, CompletionHandler<void(std::optional<Exception>&&)>&&);
105
106protected:
107 WorkerOrWorkletGlobalScope* globalScope() const { return m_globalScope; }
108
109 void initScriptIfNeeded()
110 {
111 if (!m_globalScopeWrapper)
112 initScript();
113 }
114 WEBCORE_EXPORT void initScript();
115
116private:
117 template<typename JSGlobalScopePrototype, typename JSGlobalScope, typename GlobalScope>
118 void initScriptWithSubclass();
119
120 RefPtr<JSC::VM> m_vm;
121 WorkerOrWorkletGlobalScope* m_globalScope;
122 JSC::Strong<JSDOMGlobalObject> m_globalScopeWrapper;
123 std::unique_ptr<WorkerConsoleClient> m_consoleClient;
124 mutable Lock m_scheduledTerminationLock;
125 bool m_isTerminatingExecution WTF_GUARDED_BY_LOCK(m_scheduledTerminationLock) { false };
126};
