50class WorkerSharedTimer final : public SharedTimer {
51public:
52 // SharedTimer interface.
53 void setFiredFunction(WTF::Function<void()>&& function) final { m_sharedTimerFunction = WTFMove(function); }
54 void setFireInterval(Seconds interval) final { m_nextFireTime = MonotonicTime::now() + interval; }
55 void stop() final { m_nextFireTime = MonotonicTime { }; }
56
57 bool isActive() { return m_sharedTimerFunction && m_nextFireTime; }
58 Seconds fireTimeDelay() { return std::max(0_s, m_nextFireTime - MonotonicTime::now()); }
59 void fire() { m_sharedTimerFunction(); }
60
61private:
62 WTF::Function<void()> m_sharedTimerFunction;
63 MonotonicTime m_nextFireTime;
64};

66class ModePredicate {
67public:
68 explicit ModePredicate(String&& mode)
69 : m_mode(WTFMove(mode))
70 , m_defaultMode(m_mode == WorkerRunLoop::defaultMode())
71 {
72 }
73
74 bool isDefaultMode() const
75 {
76 return m_defaultMode;
77 }
78
79 bool operator()(const WorkerRunLoop::Task& task) const
80 {
81 return m_defaultMode || m_mode == task.mode();
82 }
83
84private:
85 String m_mode;
86 bool m_defaultMode;
87};

109class RunLoopSetup {
110 WTF_MAKE_NONCOPYABLE(RunLoopSetup);
111public:
112 enum class IsForDebugging { No, Yes };
113 RunLoopSetup(WorkerRunLoop& runLoop, IsForDebugging isForDebugging)
114 : m_runLoop(runLoop)
115 , m_isForDebugging(isForDebugging)
116 {
117 if (!m_runLoop.m_nestedCount)
118 threadGlobalData().threadTimers().setSharedTimer(m_runLoop.m_sharedTimer.get());
119 m_runLoop.m_nestedCount++;
120 if (m_isForDebugging == IsForDebugging::Yes)
121 m_runLoop.m_debugCount++;
122 }
123
124 ~RunLoopSetup()
125 {
126 m_runLoop.m_nestedCount--;
127 if (!m_runLoop.m_nestedCount)
128 threadGlobalData().threadTimers().setSharedTimer(nullptr);
129 if (m_isForDebugging == IsForDebugging::Yes)
130 m_runLoop.m_debugCount--;
131 }
132private:
133 WorkerRunLoop& m_runLoop;
134 IsForDebugging m_isForDebugging { IsForDebugging::No };
135};

137void WorkerRunLoop::run(WorkerOrWorkletGlobalScope* context)
138{
139 RunLoopSetup setup(*this, RunLoopSetup::IsForDebugging::No);
140 ModePredicate modePredicate(defaultMode());
141 MessageQueueWaitResult result;
142 do {
143 result = runInMode(context, modePredicate, WaitForMessage);
144 } while (result != MessageQueueTerminated);
145 runCleanupTasks(context);
146}
147
148MessageQueueWaitResult WorkerRunLoop::runInDebuggerMode(WorkerOrWorkletGlobalScope& context)
149{
150 RunLoopSetup setup(*this, RunLoopSetup::IsForDebugging::Yes);
151 return runInMode(&context, ModePredicate { debuggerMode() }, WaitForMessage);
152}

154MessageQueueWaitResult WorkerRunLoop::runInMode(WorkerOrWorkletGlobalScope* context, const String& mode, WaitMode waitMode)
155{
156 ASSERT(mode != debuggerMode());
157 RunLoopSetup setup(*this, RunLoopSetup::IsForDebugging::No);
158 ModePredicate modePredicate(String { mode });
159 MessageQueueWaitResult result = runInMode(context, modePredicate, waitMode);
160 return result;
161}

163MessageQueueWaitResult WorkerRunLoop::runInMode(WorkerOrWorkletGlobalScope* context, const ModePredicate& predicate, WaitMode waitMode)
164{
165 ASSERT(context);
166 ASSERT(context->workerOrWorkletThread()->thread() == &Thread::current());
167
168 JSC::JSRunLoopTimer::TimerNotificationCallback timerAddedTask = WTF::createSharedTask<JSC::JSRunLoopTimer::TimerNotificationType>([this] {
169 // We don't actually do anything here, we just want to loop around runInMode
170 // to both recalculate our deadline and to potentially run the run loop.
171 this->postTask([](ScriptExecutionContext&) { });
172 });
173
174#if USE(GLIB)
175 GMainContext* mainContext = g_main_context_get_thread_default();
176 if (g_main_context_pending(mainContext))
177 g_main_context_iteration(mainContext, FALSE);
178#endif
179
180 Seconds timeoutDelay = Seconds::infinity();
181
182#if USE(CF)
183 CFAbsoluteTime nextCFRunLoopTimerFireDate = CFRunLoopGetNextTimerFireDate(CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
184 double timeUntilNextCFRunLoopTimerInSeconds = nextCFRunLoopTimerFireDate - CFAbsoluteTimeGetCurrent();
185 timeoutDelay = std::max(0_s, Seconds(timeUntilNextCFRunLoopTimerInSeconds));
186#endif
187
188 if (waitMode == WaitForMessage && predicate.isDefaultMode() && m_sharedTimer->isActive())
189 timeoutDelay = std::min(timeoutDelay, m_sharedTimer->fireTimeDelay());
190
191 if (auto* script = context->script()) {
192 script->releaseHeapAccess();
193 script->addTimerSetNotification(timerAddedTask);
194 }
195 MessageQueueWaitResult result;
196 auto task = m_messageQueue.waitForMessageFilteredWithTimeout(result, predicate, timeoutDelay);
197 if (auto* script = context->script()) {
198 script->acquireHeapAccess();
199 script->removeTimerSetNotification(timerAddedTask);
200 }
201
202 // If the context is closing, don't execute any further JavaScript tasks (per section 4.1.1 of the Web Workers spec). However, there may be implementation cleanup tasks in the queue, so keep running through it.
203
204 switch (result) {
205 case MessageQueueTerminated:
206 break;
207
208 case MessageQueueMessageReceived:
209 task->performTask(context);
210 break;
211
212 case MessageQueueTimeout:
213 if (!context->isClosing() && !isBeingDebugged())
214 m_sharedTimer->fire();
215 break;
216 }
217
218#if USE(CF)
219 if (result != MessageQueueTerminated) {
220 if (nextCFRunLoopTimerFireDate <= CFAbsoluteTimeGetCurrent())
221 CFRunLoopRunInMode(kCFRunLoopDefaultMode, 0, /*returnAfterSourceHandled*/ false);
222 }
223#endif
224
225 return result;
226}

228void WorkerRunLoop::runCleanupTasks(WorkerOrWorkletGlobalScope* context)
229{
230 ASSERT(context);
231 ASSERT(context->workerOrWorkletThread()->thread() == &Thread::current());
232 ASSERT(m_messageQueue.killed());
233
234 while (true) {
235 auto task = m_messageQueue.tryGetMessageIgnoringKilled();
236 if (!task)
237 return;
238 task->performTask(context);
239 }
240}

242void WorkerRunLoop::terminate()
243{
244 m_messageQueue.kill();
245}

247void WorkerRunLoop::postTask(ScriptExecutionContext::Task&& task)
248{
249 postTaskForMode(WTFMove(task), defaultMode());
250}
251
252void WorkerRunLoop::postTaskAndTerminate(ScriptExecutionContext::Task&& task)
253{
254 m_messageQueue.appendAndKill(makeUnique<Task>(WTFMove(task), defaultMode()));
255}
256
257void WorkerRunLoop::postTaskForMode(ScriptExecutionContext::Task&& task, const String& mode)
258{
259 m_messageQueue.append(makeUnique<Task>(WTFMove(task), mode));
260}
261
262void WorkerRunLoop::postDebuggerTask(ScriptExecutionContext::Task&& task)
263{
264 postTaskForMode(WTFMove(task), debuggerMode());
265}
266
267void WorkerRunLoop::Task::performTask(WorkerOrWorkletGlobalScope* context)
268{
269 if ((!context->isClosing() && context->script() && !context->script()->isTerminatingExecution()) || m_task.isCleanupTask())
270 m_task.performTask(*context);
271}
