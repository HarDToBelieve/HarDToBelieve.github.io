44namespace WebCore {
45
46class Frame;

49class ServiceWorker final : public RefCounted<ServiceWorker>, public EventTargetWithInlineData, public ActiveDOMObject {
50 WTF_MAKE_ISO_ALLOCATED(ServiceWorker);
51public:
52 using State = ServiceWorkerState;
53 static Ref<ServiceWorker> getOrCreate(ScriptExecutionContext&, ServiceWorkerData&&);
54
55 virtual ~ServiceWorker();
56
57 const URL& scriptURL() const { return m_data.scriptURL; }
58
59 State state() const { return m_data.state; }
60
61 void updateState(State);
62
63 ExceptionOr<void> postMessage(JSC::JSGlobalObject&, JSC::JSValue message, PostMessageOptions&&);
64
65 ServiceWorkerIdentifier identifier() const { return m_data.identifier; }
66 ServiceWorkerRegistrationIdentifier registrationIdentifier() const { return m_data.registrationIdentifier; }
67 WorkerType workerType() const { return m_data.type; }
68
69 using RefCounted::ref;
70 using RefCounted::deref;
71
72private:
73 ServiceWorker(ScriptExecutionContext&, ServiceWorkerData&&);
74 void updatePendingActivityForEventDispatch();
75
76 EventTargetInterface eventTargetInterface() const final;
77 ScriptExecutionContext* scriptExecutionContext() const final;
78 void refEventTarget() final { ref(); }
79 void derefEventTarget() final { deref(); }
80
81 // ActiveDOMObject.
82 const char* activeDOMObjectName() const final;
83 void stop() final;
84
85 SWClientConnection& swConnection();
86
87 ServiceWorkerData m_data;
88 bool m_isStopped { false };
89 RefPtr<PendingActivity<ServiceWorker>> m_pendingActivityForEventDispatch;
90};
