34namespace WebCore {
35
36struct ServiceWorkerClientIdentifier {
37 SWServerConnectionIdentifier serverConnectionIdentifier;
38 DocumentIdentifier contextIdentifier;
39
40 unsigned hash() const;
41
42 String toString() const { return makeString(serverConnectionIdentifier.toUInt64(), '-', contextIdentifier.toUInt64()); }
43 static std::optional<ServiceWorkerClientIdentifier> fromString(StringView);
44
45 template<class Encoder> void encode(Encoder&) const;
46 template<class Decoder> static std::optional<ServiceWorkerClientIdentifier> decode(Decoder&);
47};
48
49inline bool operator==(const ServiceWorkerClientIdentifier& a, const ServiceWorkerClientIdentifier& b)
50{
51 return a.serverConnectionIdentifier == b.serverConnectionIdentifier && a.contextIdentifier == b.contextIdentifier;
52}
53
54template<class Encoder>
55void ServiceWorkerClientIdentifier::encode(Encoder& encoder) const
56{
57 encoder << serverConnectionIdentifier << contextIdentifier;
58}
59
60template<class Decoder>
61std::optional<ServiceWorkerClientIdentifier> ServiceWorkerClientIdentifier::decode(Decoder& decoder)
62{
63 std::optional<SWServerConnectionIdentifier> serverConnectionIdentifier;
64 decoder >> serverConnectionIdentifier;
65 if (!serverConnectionIdentifier)
66 return std::nullopt;
67
68 std::optional<DocumentIdentifier> contextIdentifier;
69 decoder >> contextIdentifier;
70 if (!contextIdentifier)
71 return std::nullopt;
72
73 return { { *serverConnectionIdentifier, *contextIdentifier } };
74}
75
76inline unsigned ServiceWorkerClientIdentifier::hash() const
77{
78 return computeHash(serverConnectionIdentifier, contextIdentifier);
79}
80
81struct ServiceWorkerClientIdentifierHash {
82 static unsigned hash(const WebCore::ServiceWorkerClientIdentifier& key) { return key.hash(); }
83 static bool equal(const WebCore::ServiceWorkerClientIdentifier& a, const WebCore::ServiceWorkerClientIdentifier& b) { return a == b; }
84 static const bool safeToCompareToEmptyOrDeleted = true;
85};
