99auto ServiceWorkerContainer::ready() -> ReadyPromise&
100{
101 if (!m_readyPromise) {
102 m_readyPromise = makeUnique<ReadyPromise>();
103
104 if (m_isStopped)
105 return *m_readyPromise;
106
107 auto& context = *scriptExecutionContext();
108 ensureSWClientConnection().whenRegistrationReady(context.topOrigin().data(), context.url(), [this, protectedThis = makeRef(*this)](auto&& registrationData) mutable {
109 queueTaskKeepingObjectAlive(*this, TaskSource::DOMManipulation, [this, registrationData = WTFMove(registrationData)]() mutable {
110 auto* context = scriptExecutionContext();
111 if (!context || !m_readyPromise)
112 return;
113 auto registration = ServiceWorkerRegistration::getOrCreate(*context, *this, WTFMove(registrationData));
114 m_readyPromise->resolve(WTFMove(registration));
115 });
116 });
117 }
118 return *m_readyPromise;
119}

128void ServiceWorkerContainer::addRegistration(const String& relativeScriptURL, const RegistrationOptions& options, Ref<DeferredPromise>&& promise)
129{
130 auto* context = scriptExecutionContext();
131 if (m_isStopped) {
132 promise->reject(Exception(InvalidStateError));
133 return;
134 }
135
136 if (relativeScriptURL.isEmpty()) {
137 promise->reject(Exception { TypeError, "serviceWorker.register() cannot be called with an empty script URL"_s });
138 return;
139 }
140
141 ServiceWorkerJobData jobData(ensureSWClientConnection().serverConnectionIdentifier(), contextIdentifier());
142
143 jobData.scriptURL = context->completeURL(relativeScriptURL);
144 if (!jobData.scriptURL.isValid()) {
145 CONTAINER_RELEASE_LOG_ERROR("addRegistration: Invalid scriptURL");
146 promise->reject(Exception { TypeError, "serviceWorker.register() must be called with a valid relative script URL"_s });
147 return;
148 }
149
150 if (!jobData.scriptURL.protocolIsInHTTPFamily()) {
151 CONTAINER_RELEASE_LOG_ERROR("addRegistration: Invalid scriptURL scheme is not HTTP or HTTPS");
152 promise->reject(Exception { TypeError, "serviceWorker.register() must be called with a script URL whose protocol is either HTTP or HTTPS"_s });
153 return;
154 }
155
156 auto path = jobData.scriptURL.path();
157 if (path.containsIgnoringASCIICase("%2f") || path.containsIgnoringASCIICase("%5c")) {
158 CONTAINER_RELEASE_LOG_ERROR("addRegistration: scriptURL contains invalid character");
159 promise->reject(Exception { TypeError, "serviceWorker.register() must be called with a script URL whose path does not contain '%2f' or '%5c'"_s });
160 return;
161 }
162
163 if (!options.scope.isEmpty())
164 jobData.scopeURL = context->completeURL(options.scope);
165 else
166 jobData.scopeURL = URL(jobData.scriptURL, "./");
167
168 if (!jobData.scopeURL.isNull() && !jobData.scopeURL.protocolIsInHTTPFamily()) {
169 CONTAINER_RELEASE_LOG_ERROR("addRegistration: scopeURL scheme is not HTTP or HTTPS");
170 promise->reject(Exception { TypeError, "Scope URL provided to serviceWorker.register() must be either HTTP or HTTPS"_s });
171 return;
172 }
173
174 path = jobData.scopeURL.path();
175 if (path.containsIgnoringASCIICase("%2f") || path.containsIgnoringASCIICase("%5c")) {
176 CONTAINER_RELEASE_LOG_ERROR("addRegistration: scopeURL contains invalid character");
177 promise->reject(Exception { TypeError, "Scope URL provided to serviceWorker.register() cannot have a path that contains '%2f' or '%5c'"_s });
178 return;
179 }
180
181 CONTAINER_RELEASE_LOG("addRegistration: Registering service worker. jobID=%" PRIu64, jobData.identifier().jobIdentifier.toUInt64());
182
183 jobData.clientCreationURL = context->url();
184 jobData.topOrigin = context->topOrigin().data();
185 jobData.workerType = options.type;
186 jobData.type = ServiceWorkerJobType::Register;
187 jobData.registrationOptions = options;
188
189 scheduleJob(makeUnique<ServiceWorkerJob>(*this, WTFMove(promise), WTFMove(jobData)));
190}

192void ServiceWorkerContainer::unregisterRegistration(ServiceWorkerRegistrationIdentifier registrationIdentifier, DOMPromiseDeferred<IDLBoolean>&& promise)
193{
194 ASSERT(!m_isStopped);
195 if (!m_swConnection) {
196 ASSERT_NOT_REACHED();

198 return;
199 }
200
201 CONTAINER_RELEASE_LOG("unregisterRegistration: Unregistering service worker.");
202 m_swConnection->scheduleUnregisterJobInServer(registrationIdentifier, contextIdentifier(), [promise = WTFMove(promise)](auto&& result) mutable {
203 promise.settle(WTFMove(result));
204 });

207void ServiceWorkerContainer::updateRegistration(const URL& scopeURL, const URL& scriptURL, WorkerType workerType, RefPtr<DeferredPromise>&& promise)
208{
209 ASSERT(!m_isStopped);
210
211 auto& context = *scriptExecutionContext();
212
213 if (!m_swConnection) {
214 ASSERT_NOT_REACHED();
215 if (promise)
216 promise->reject(Exception(InvalidStateError));
217 return;
218 }
219
220 ServiceWorkerJobData jobData(m_swConnection->serverConnectionIdentifier(), contextIdentifier());
221 jobData.clientCreationURL = context.url();
222 jobData.topOrigin = context.topOrigin().data();
223 jobData.workerType = workerType;
224 jobData.type = ServiceWorkerJobType::Update;
225 jobData.scopeURL = scopeURL;
226 jobData.scriptURL = scriptURL;
227
228 CONTAINER_RELEASE_LOG("removeRegistration: Updating service worker. jobID=%" PRIu64, jobData.identifier().jobIdentifier.toUInt64());
229
230 scheduleJob(makeUnique<ServiceWorkerJob>(*this, WTFMove(promise), WTFMove(jobData)));
231}

233void ServiceWorkerContainer::scheduleJob(std::unique_ptr<ServiceWorkerJob>&& job)
234{
235 ASSERT(m_creationThread.ptr() == &Thread::current());
236 ASSERT(m_swConnection);
237 ASSERT(!isStopped());
238
239 auto& jobData = job->data();
240 auto jobIdentifier = job->identifier();
241 ASSERT(!m_jobMap.contains(jobIdentifier));
242 m_jobMap.add(jobIdentifier, OngoingJob { WTFMove(job), makePendingActivity(*this) });
243
244 m_swConnection->scheduleJob(contextIdentifier(), jobData);
245}

247void ServiceWorkerContainer::getRegistration(const String& clientURL, Ref<DeferredPromise>&& promise)
248{
249 if (m_isStopped) {
250 promise->reject(Exception { InvalidStateError });
251 return;
252 }
253
254 auto& context = *scriptExecutionContext();
255 URL parsedURL = context.completeURL(clientURL);
256 if (!protocolHostAndPortAreEqual(parsedURL, context.url())) {
257 promise->reject(Exception { SecurityError, "Origin of clientURL is not client's origin"_s });
258 return;
259 }
260
261 ensureSWClientConnection().matchRegistration(SecurityOriginData { context.topOrigin().data() }, parsedURL, [this, protectedThis = makeRef(*this), promise = WTFMove(promise)](auto&& result) mutable {
262 queueTaskKeepingObjectAlive(*this, TaskSource::DOMManipulation, [this, promise = WTFMove(promise), result = WTFMove(result)]() mutable {
263 if (!result) {
264 promise->resolve();
265 return;
266 }
267 promise->resolve<IDLInterface<ServiceWorkerRegistration>>(ServiceWorkerRegistration::getOrCreate(*scriptExecutionContext(), *this, WTFMove(result.value())));
268 });
269 });
270}

272void ServiceWorkerContainer::updateRegistrationState(ServiceWorkerRegistrationIdentifier identifier, ServiceWorkerRegistrationState state, const std::optional<ServiceWorkerData>& serviceWorkerData)
273{
274 if (m_isStopped)
275 return;
276
277 queueTaskKeepingObjectAlive(*this, TaskSource::DOMManipulation, [this, identifier, state, serviceWorkerData = std::optional<ServiceWorkerData> { serviceWorkerData }]() mutable {
278 RefPtr<ServiceWorker> serviceWorker;
279 if (serviceWorkerData)
280 serviceWorker = ServiceWorker::getOrCreate(*scriptExecutionContext(), WTFMove(*serviceWorkerData));
281
282 if (auto* registration = m_registrations.get(identifier))
283 registration->updateStateFromServer(state, WTFMove(serviceWorker));
284 });
285}

298void ServiceWorkerContainer::getRegistrations(Ref<DeferredPromise>&& promise)
299{
300 if (m_isStopped) {
301 promise->reject(Exception { InvalidStateError });
302 return;
303 }
304
305 auto& context = *scriptExecutionContext();
306 ensureSWClientConnection().getRegistrations(SecurityOriginData { context.topOrigin().data() }, context.url(), [this, protectedThis = makeRef(*this), promise = WTFMove(promise)] (auto&& registrationDatas) mutable {
307 queueTaskKeepingObjectAlive(*this, TaskSource::DOMManipulation, [this, promise = WTFMove(promise), registrationDatas = WTFMove(registrationDatas)]() mutable {

352void ServiceWorkerContainer::jobResolvedWithRegistration(ServiceWorkerJob& job, ServiceWorkerRegistrationData&& data, ShouldNotifyWhenResolved shouldNotifyWhenResolved)
353{
354 ASSERT(m_creationThread.ptr() == &Thread::current());
355 ASSERT_WITH_MESSAGE(job.hasPromise() || job.data().type == ServiceWorkerJobType::Update, "Only soft updates have no promise");
356
357 if (job.data().type == ServiceWorkerJobType::Register)
358 CONTAINER_RELEASE_LOG("jobResolvedWithRegistration: Registration job %" PRIu64 " succeeded", job.identifier().toUInt64());
359 else {
360 ASSERT(job.data().type == ServiceWorkerJobType::Update);
361 CONTAINER_RELEASE_LOG("jobResolvedWithRegistration: Update job %" PRIu64 " succeeded", job.identifier().toUInt64());
362 }
363
364 auto guard = WTF::makeScopeExit([this, &job] {
365 destroyJob(job);
366 });
367
368 auto notifyIfExitEarly = WTF::makeScopeExit([this, protectedThis = makeRef(*this), key = data.key, shouldNotifyWhenResolved] {
369 if (shouldNotifyWhenResolved == ShouldNotifyWhenResolved::Yes)
370 notifyRegistrationIsSettled(key);
371 });
372
373 if (isStopped())
374 return;
375
376 auto promise = job.takePromise();
377 if (!promise)
378 return;
379
380 queueTaskKeepingObjectAlive(*this, TaskSource::DOMManipulation, [this, protectedThis = makeRef(*this), promise = WTFMove(promise), jobIdentifier = job.identifier(), data = WTFMove(data), shouldNotifyWhenResolved, notifyIfExitEarly = WTFMove(notifyIfExitEarly)]() mutable {
381 notifyIfExitEarly.release();
382
383 auto registration = ServiceWorkerRegistration::getOrCreate(*scriptExecutionContext(), *this, WTFMove(data));
384
385 CONTAINER_RELEASE_LOG("jobResolvedWithRegistration: Resolving promise for job %" PRIu64 ". registrationID=%" PRIu64, jobIdentifier.toUInt64(), registration->identifier().toUInt64());
386
387 if (shouldNotifyWhenResolved == ShouldNotifyWhenResolved::Yes) {
388 m_ongoingSettledRegistrations.add(++m_lastOngoingSettledRegistrationIdentifier, registration->data().key);
389 promise->whenSettled([this, protectedThis = WTFMove(protectedThis), identifier = m_lastOngoingSettledRegistrationIdentifier] {
390 auto iterator = m_ongoingSettledRegistrations.find(identifier);
391 if (iterator == m_ongoingSettledRegistrations.end())
392 return;
393 notifyRegistrationIsSettled(iterator->value);
394 m_ongoingSettledRegistrations.remove(iterator);
395 });
396 }
397
398 promise->resolve<IDLInterface<ServiceWorkerRegistration>>(WTFMove(registration));
399 });
400}

402void ServiceWorkerContainer::postMessage(MessageWithMessagePorts&& message, ServiceWorkerData&& sourceData, String&& sourceOrigin)
403{
404 auto& context = *scriptExecutionContext();
405 MessageEventSource source = RefPtr<ServiceWorker> { ServiceWorker::getOrCreate(context, WTFMove(sourceData)) };
406
407 auto messageEvent = MessageEvent::create(MessagePort::entanglePorts(context, WTFMove(message.transferredPorts)), message.message.releaseNonNull(), sourceOrigin, { }, WTFMove(source));
408 if (m_shouldDeferMessageEvents)
409 m_deferredMessageEvents.append(WTFMove(messageEvent));
410 else {
411 ASSERT(m_deferredMessageEvents.isEmpty());
412 queueTaskToDispatchEvent(*this, TaskSource::DOMManipulation, WTFMove(messageEvent));
413 }
414}

539void ServiceWorkerContainer::stop()
540{
541 m_isStopped = true;
542 removeAllEventListeners();
543 m_readyPromise = nullptr;
544 auto jobMap = WTFMove(m_jobMap);
545 for (auto& ongoingJob : jobMap.values()) {
546 if (ongoingJob.job->cancelPendingLoad())
547 notifyFailedFetchingScript(*ongoingJob.job.get(), ResourceError { errorDomainWebKitInternal, 0, ongoingJob.job->data().scriptURL, "Job cancelled"_s, ResourceError::Type::Cancellation });
548 }
549
550 auto registrationMap = WTFMove(m_ongoingSettledRegistrations);
551 for (auto& registration : registrationMap.values())
552 notifyRegistrationIsSettled(registration);
553}

572bool ServiceWorkerContainer::addEventListener(const AtomString& eventType, Ref<EventListener>&& eventListener, const AddEventListenerOptions& options)
573{
574 // Setting the onmessage EventHandler attribute on the ServiceWorkerContainer should start the messages
575 // automatically.
576 if (eventListener->isAttribute() && eventType == eventNames().messageEvent)
577 startMessages();
578
579 return EventTargetWithInlineData::addEventListener(eventType, WTFMove(eventListener), options);
580}
