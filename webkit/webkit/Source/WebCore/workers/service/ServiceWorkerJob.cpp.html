97void ServiceWorkerJob::fetchScriptWithContext(ScriptExecutionContext& context, FetchOptions::Cache cachePolicy)
98{
99 ASSERT(m_creationThread.ptr() == &Thread::current());
100 ASSERT(!m_completed);
101
102 // FIXME: WorkerScriptLoader is the wrong loader class to use here, but there's nothing else better right now.
103 m_scriptLoader = WorkerScriptLoader::create();
104
105 ResourceRequest request { m_jobData.scriptURL };
106 request.setInitiatorIdentifier(context.resourceRequestIdentifier());
107 request.addHTTPHeaderField("Service-Worker"_s, "script"_s);
108
109 FetchOptions options;
110 options.mode = FetchOptions::Mode::SameOrigin;
111 options.cache = cachePolicy;
112 options.redirect = FetchOptions::Redirect::Error;
113 options.destination = FetchOptions::Destination::Serviceworker;
114 options.credentials = FetchOptions::Credentials::SameOrigin;
115 m_scriptLoader->loadAsynchronously(context, WTFMove(request), WTFMove(options), ContentSecurityPolicyEnforcement::DoNotEnforce, ServiceWorkersMode::None, *this, WorkerRunLoop::defaultMode());
116}

118ResourceError ServiceWorkerJob::validateServiceWorkerResponse(const ServiceWorkerJobData& jobData, const ResourceResponse& response)
119{
120 // Extract a MIME type from the response's header list. If this MIME type (ignoring parameters) is not a JavaScript MIME type, then:
121 if (!MIMETypeRegistry::isSupportedJavaScriptMIMEType(response.mimeType()))
122 return { errorDomainWebKitInternal, 0, response.url(), "MIME Type is not a JavaScript MIME type"_s };
123
124 auto serviceWorkerAllowed = response.httpHeaderField(HTTPHeaderName::ServiceWorkerAllowed);
125 String maxScopeString;
126 if (serviceWorkerAllowed.isNull()) {
127 auto path = jobData.scriptURL.path();
128 // Last part of the path is the script's filename.
129 maxScopeString = path.substring(0, path.reverseFind('/') + 1).toString();
130 } else {
131 auto maxScope = URL(jobData.scriptURL, serviceWorkerAllowed);
132 if (SecurityOrigin::create(maxScope)->isSameOriginAs(SecurityOrigin::create(jobData.scriptURL)))
133 maxScopeString = maxScope.path().toString();
134 }
135
136 auto scopeString = jobData.scopeURL.path();
137 if (maxScopeString.isNull() || !scopeString.startsWith(maxScopeString))
138 return { errorDomainWebKitInternal, 0, response.url(), "Scope URL should start with the given script URL"_s };
139
140 return { };
141}

143void ServiceWorkerJob::didReceiveResponse(unsigned long, const ResourceResponse& response)
144{
145 ASSERT(m_creationThread.ptr() == &Thread::current());
146 ASSERT(!m_completed);
147 ASSERT(m_scriptLoader);
148
149 auto error = validateServiceWorkerResponse(m_jobData, response);
150 if (error.isNull())
151 return;
152
153 m_scriptLoader->cancel();
154 m_scriptLoader = nullptr;
155
156 Exception exception { SecurityError, error.localizedDescription() };
157 m_client.jobFailedLoadingScript(*this, WTFMove(error), WTFMove(exception));
158}
159
160void ServiceWorkerJob::notifyFinished()
161{
162 ASSERT(m_creationThread.ptr() == &Thread::current());
163 ASSERT(m_scriptLoader);
164
165 auto scriptLoader = WTFMove(m_scriptLoader);
166
167 if (!scriptLoader->failed()) {
168 m_client.jobFinishedLoadingScript(*this, scriptLoader->script(), scriptLoader->certificateInfo(), scriptLoader->contentSecurityPolicy(), scriptLoader->referrerPolicy());
169 return;
170 }
171
172 auto& error = scriptLoader->error();
173 ASSERT(!error.isNull());
174
175 m_client.jobFailedLoadingScript(*this, error, Exception { error.isAccessControl() ? SecurityError : TypeError, makeString("Script ", scriptLoader->url().string(), " load failed") });
176}
177
178bool ServiceWorkerJob::cancelPendingLoad()
179{
180 if (!m_scriptLoader)
181 return false;
182
183 m_scriptLoader->cancel();
184 m_scriptLoader = nullptr;
185 return true;
186}
