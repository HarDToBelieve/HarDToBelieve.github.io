33static inline ServiceWorkerOrClientIdentifier serviceWorkerOrClientIdentifier(SWServerConnectionIdentifier connectionIdentifier, const DocumentOrWorkerIdentifier& localSourceContext)
34{
35 return WTF::switchOn(localSourceContext, [&](DocumentIdentifier documentIdentifier) -> ServiceWorkerOrClientIdentifier {
36 return ServiceWorkerClientIdentifier { connectionIdentifier, documentIdentifier };
37 }, [&](ServiceWorkerIdentifier serviceWorkerIdentifier) -> ServiceWorkerOrClientIdentifier {
38 return serviceWorkerIdentifier;
39 });
40}
41
42ServiceWorkerJobData::ServiceWorkerJobData(SWServerConnectionIdentifier connectionIdentifier, const DocumentOrWorkerIdentifier& localSourceContext)
43 : sourceContext(serviceWorkerOrClientIdentifier(connectionIdentifier, localSourceContext))
44 , m_identifier { connectionIdentifier, ServiceWorkerJobIdentifier::generateThreadSafe() }
45{
46}
47
48ServiceWorkerJobData::ServiceWorkerJobData(Identifier identifier, const DocumentOrWorkerIdentifier& localSourceContext)
49 : sourceContext(serviceWorkerOrClientIdentifier(identifier.connectionIdentifier, localSourceContext))
50 , m_identifier { identifier }
51{
52}

54ServiceWorkerRegistrationKey ServiceWorkerJobData::registrationKey() const
55{
56 URL scope = scopeURL;
57 scope.removeFragmentIdentifier();
58 return { SecurityOriginData { topOrigin }, WTFMove(scope) };
59}

61ServiceWorkerJobData ServiceWorkerJobData::isolatedCopy() const
62{
63 ServiceWorkerJobData result;
64 result.m_identifier = identifier();
65 result.sourceContext = sourceContext;
66 result.workerType = workerType;
67 result.type = type;
68
69 result.scriptURL = scriptURL.isolatedCopy();
70 result.clientCreationURL = clientCreationURL.isolatedCopy();
71 result.topOrigin = topOrigin.isolatedCopy();
72 result.scopeURL = scopeURL.isolatedCopy();
73 result.registrationOptions = registrationOptions.isolatedCopy();
74
75 return result;
76}

78// https://w3c.github.io/ServiceWorker/#dfn-job-equivalent
79bool ServiceWorkerJobData::isEquivalent(const ServiceWorkerJobData& job) const
80{
81 if (type != job.type)
82 return false;
83
84 switch (type) {
85 case ServiceWorkerJobType::Register:
86 case ServiceWorkerJobType::Update:
87 return scopeURL == job.scopeURL
88 && scriptURL == job.scriptURL
89 && workerType == job.workerType
90 && registrationOptions.updateViaCache == job.registrationOptions.updateViaCache;
91 case ServiceWorkerJobType::Unregister:
92 return scopeURL == job.scopeURL;
93 }
94 return false;
95}
