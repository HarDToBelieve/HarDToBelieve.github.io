26#pragma once
27
28#if ENABLE(SERVICE_WORKER)
29
30#include "ServiceWorkerTypes.h"
31
32namespace WebCore {
33
34struct ServiceWorkerJobDataIdentifier {
35 SWServerConnectionIdentifier connectionIdentifier;
36 ServiceWorkerJobIdentifier jobIdentifier;
37
38 String loggingString() const
39 {
40 return connectionIdentifier.loggingString() + "-" + jobIdentifier.loggingString();
41 }
42
43 template<class Encoder> void encode(Encoder&) const;
44 template<class Decoder> static std::optional<ServiceWorkerJobDataIdentifier> decode(Decoder&);
45};
46
47inline bool operator==(const ServiceWorkerJobDataIdentifier& a, const ServiceWorkerJobDataIdentifier& b)
48{
49 return a.connectionIdentifier == b.connectionIdentifier && a.jobIdentifier == b.jobIdentifier;
50}
51
52template<class Encoder>
53void ServiceWorkerJobDataIdentifier::encode(Encoder& encoder) const
54{
55 encoder << connectionIdentifier << jobIdentifier;
56}
57
58template<class Decoder>
59std::optional<ServiceWorkerJobDataIdentifier> ServiceWorkerJobDataIdentifier::decode(Decoder& decoder)
60{
61 std::optional<SWServerConnectionIdentifier> connectionIdentifier;
62 decoder >> connectionIdentifier;
63 if (!connectionIdentifier)
64 return std::nullopt;
65
66 std::optional<ServiceWorkerJobIdentifier> jobIdentifier;
67 decoder >> jobIdentifier;
68 if (!jobIdentifier)
69 return std::nullopt;
70
71 return { { WTFMove(*connectionIdentifier), WTFMove(*jobIdentifier) } };
72}
73
74} // namespace WebCore
75
76#endif // ENABLE(SERVICE_WORKER)
