55void SWContextManager::registerServiceWorkerThreadForInstall(Ref<ServiceWorkerThreadProxy>&& serviceWorkerThreadProxy)
56{
57 auto serviceWorkerIdentifier = serviceWorkerThreadProxy->identifier();
58 auto jobDataIdentifier = serviceWorkerThreadProxy->thread().jobDataIdentifier();
59 auto* threadProxy = serviceWorkerThreadProxy.ptr();
60 auto result = m_workerMap.add(serviceWorkerIdentifier, WTFMove(serviceWorkerThreadProxy));
61 ASSERT_UNUSED(result, result.isNewEntry);
62
63 threadProxy->thread().start([jobDataIdentifier, serviceWorkerIdentifier](const String& exceptionMessage, bool doesHandleFetch) {
64 SWContextManager::singleton().startedServiceWorker(jobDataIdentifier, serviceWorkerIdentifier, exceptionMessage, doesHandleFetch);
65 });
66}

68void SWContextManager::startedServiceWorker(std::optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, const String& exceptionMessage, bool doesHandleFetch)
69{
70 if (m_serviceWorkerCreationCallback)
71 m_serviceWorkerCreationCallback(serviceWorkerIdentifier.toUInt64());
72 if (!exceptionMessage.isEmpty()) {
73 connection()->serviceWorkerFailedToStart(jobDataIdentifier, serviceWorkerIdentifier, exceptionMessage);
74 return;
75 }
76 connection()->serviceWorkerStarted(jobDataIdentifier, serviceWorkerIdentifier, doesHandleFetch);
77}

112void SWContextManager::terminateWorker(ServiceWorkerIdentifier identifier, Seconds timeout, Function<void()>&& completionHandler)
113{
114 auto serviceWorker = m_workerMap.take(identifier);
115 if (!serviceWorker) {
116 if (completionHandler)
117 completionHandler();
118 return;
119 }
120 stopWorker(*serviceWorker, timeout, WTFMove(completionHandler));
121}

129void SWContextManager::stopWorker(ServiceWorkerThreadProxy& serviceWorker, Seconds timeout, Function<void()>&& completionHandler)
130{
131 auto identifier = serviceWorker.identifier();
132 serviceWorker.setAsTerminatingOrTerminated();
133
134 m_pendingServiceWorkerTerminationRequests.add(identifier, makeUnique<ServiceWorkerTerminationRequest>(*this, identifier, timeout));
135
136 auto& thread = serviceWorker.thread();
137 thread.stop([this, identifier, serviceWorker = makeRef(serviceWorker), completionHandler = WTFMove(completionHandler)]() mutable {
138 m_pendingServiceWorkerTerminationRequests.remove(identifier);
139
140 if (auto* connection = SWContextManager::singleton().connection())
141 connection->workerTerminated(identifier);
142
143 if (completionHandler)
144 completionHandler();
145
146 if (m_pendingServiceWorkerTerminationRequests.isEmpty()) {
147 for (auto& handler : std::exchange(m_whenTerminationRequestsAreDoneHandlers, { }))
148 handler();
149 }
150
151 // Spin the runloop before releasing the worker thread proxy, as there would otherwise be
152 // a race towards its destruction.
153 callOnMainThread([serviceWorker = WTFMove(serviceWorker)] { });
154 });
155}
156
157void SWContextManager::whenTerminationRequestsAreDone(CompletionHandler<void()>&& completionHandler)
158{
159 if (m_pendingServiceWorkerTerminationRequests.isEmpty())
160 return completionHandler();
161
162 m_whenTerminationRequestsAreDoneHandlers.append(WTFMove(completionHandler));
163}

171bool SWContextManager::postTaskToServiceWorker(ServiceWorkerIdentifier identifier, WTF::Function<void(ServiceWorkerGlobalScope&)>&& task)
172{
173 auto* serviceWorker = m_workerMap.get(identifier);
174 if (!serviceWorker)
175 return false;
176
177 serviceWorker->thread().runLoop().postTask([task = WTFMove(task)] (auto& context) {
178 task(downcast<ServiceWorkerGlobalScope>(context));
179 });
180 return true;
181}

183void SWContextManager::serviceWorkerFailedToTerminate(ServiceWorkerIdentifier serviceWorkerIdentifier)
184{
185 UNUSED_PARAM(serviceWorkerIdentifier);
186 RELEASE_LOG_ERROR(ServiceWorker, "Failed to terminate service worker with identifier %s, killing the service worker process", serviceWorkerIdentifier.loggingString().utf8().data());
187 ASSERT_NOT_REACHED();
188 _exit(EXIT_FAILURE);
189}

191SWContextManager::ServiceWorkerTerminationRequest::ServiceWorkerTerminationRequest(SWContextManager& manager, ServiceWorkerIdentifier serviceWorkerIdentifier, Seconds timeout)
192 : m_timeoutTimer([&manager, serviceWorkerIdentifier] { manager.serviceWorkerFailedToTerminate(serviceWorkerIdentifier); })
193{
194 m_timeoutTimer.startOneShot(timeout);
195}

197void SWContextManager::stopAllServiceWorkers()
198{
199 auto serviceWorkers = WTFMove(m_workerMap);
200 for (auto& serviceWorker : serviceWorkers.values())
201 stopWorker(serviceWorker, workerTerminationTimeout, [] { });
202}
