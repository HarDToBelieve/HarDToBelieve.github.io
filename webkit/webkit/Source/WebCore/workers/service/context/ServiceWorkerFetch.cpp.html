48// https://fetch.spec.whatwg.org/#http-fetch step 3.3
49static inline std::optional<ResourceError> validateResponse(const ResourceResponse& response, FetchOptions::Mode mode, FetchOptions::Redirect redirect)
50{
51 if (response.type() == ResourceResponse::Type::Error)
52 return ResourceError { errorDomainWebKitInternal, 0, response.url(), "Response served by service worker is an error"_s, ResourceError::Type::General };
53
54 if (mode != FetchOptions::Mode::NoCors && response.tainting() == ResourceResponse::Tainting::Opaque)
55 return ResourceError { errorDomainWebKitInternal, 0, response.url(), "Response served by service worker is opaque"_s, ResourceError::Type::AccessControl };
56
57 // Navigate mode induces manual redirect.
58 if (redirect != FetchOptions::Redirect::Manual && mode != FetchOptions::Mode::Navigate && response.tainting() == ResourceResponse::Tainting::Opaqueredirect)
59 return ResourceError { errorDomainWebKitInternal, 0, response.url(), "Response served by service worker is opaque redirect"_s, ResourceError::Type::AccessControl };
60
61 if ((redirect != FetchOptions::Redirect::Follow || mode == FetchOptions::Mode::Navigate) && response.isRedirected())
62 return ResourceError { errorDomainWebKitInternal, 0, response.url(), "Response served by service worker has redirections"_s, ResourceError::Type::AccessControl };
63
64 return { };
65}

67static void processResponse(Ref<Client>&& client, Expected<Ref<FetchResponse>, std::optional<ResourceError>>&& result, FetchOptions::Mode mode, FetchOptions::Redirect redirect, const URL& requestURL, CertificateInfo&& certificateInfo)
68{
69 if (!result.has_value()) {
70 auto& error = result.error();
71 if (!error) {
72 client->didNotHandle();
73 return;
74 }
75 client->didFail(*error);
76 return;
77 }
78 auto response = WTFMove(result.value());
79
80 auto loadingError = response->loadingError();
81 if (!loadingError.isNull()) {
82 client->didFail(loadingError);
83 return;
84 }
85
86 auto resourceResponse = response->resourceResponse();
87 if (auto error = validateResponse(resourceResponse, mode, redirect)) {
88 client->didFail(error.value());
89 return;
90 }
91
92 if (resourceResponse.isRedirection() && resourceResponse.httpHeaderFields().contains(HTTPHeaderName::Location)) {
93 client->didReceiveRedirection(resourceResponse);
94 return;
95 }
96
97 // In case of main resource and mime type is the default one, we set it to text/html to pass more service worker WPT tests.
98 // FIXME: We should refine our MIME type sniffing strategy for synthetic responses.
99 if (mode == FetchOptions::Mode::Navigate) {
100 if (resourceResponse.mimeType() == defaultMIMEType()) {
101 resourceResponse.setMimeType("text/html"_s);
102 resourceResponse.setTextEncodingName("UTF-8"_s);
103 }
104
105 if (!resourceResponse.certificateInfo())
106 resourceResponse.setCertificateInfo(WTFMove(certificateInfo));
107 }
108
109 // As per https://fetch.spec.whatwg.org/#main-fetch step 9, copy request's url list in response's url list if empty.
110 if (resourceResponse.url().isNull())
111 resourceResponse.setURL(requestURL);
112
113 client->didReceiveResponse(resourceResponse);
114
115 if (response->isBodyReceivedByChunk()) {
116 response->consumeBodyReceivedByChunk([client = WTFMove(client)] (auto&& result) mutable {
117 if (result.hasException()) {
118 client->didFail(FetchEvent::createResponseError(URL { }, result.exception().message()));
119 return;
120 }
121
122 if (auto* chunk = result.returnValue())
123 client->didReceiveData(SharedBuffer::create(chunk->data(), chunk->size()));
124 else
125 client->didFinish();
126 });
127 return;
128 }
129
130 auto body = response->consumeBody();
131 WTF::switchOn(body, [&] (Ref<FormData>& formData) {
132 client->didReceiveFormDataAndFinish(WTFMove(formData));
133 }, [&] (Ref<SharedBuffer>& buffer) {
134 client->didReceiveData(WTFMove(buffer));
135 client->didFinish();
136 }, [&] (std::nullptr_t&) {
137 client->didFinish();
138 });
139}

141void dispatchFetchEvent(Ref<Client>&& client, ServiceWorkerGlobalScope& globalScope, std::optional<ServiceWorkerClientIdentifier> clientId, ResourceRequest&& request, String&& referrer, FetchOptions&& options)
142{
143 auto requestHeaders = FetchHeaders::create(FetchHeaders::Guard::Immutable, HTTPHeaderMap { request.httpHeaderFields() });
144
145 FetchOptions::Mode mode = options.mode;
146 FetchOptions::Redirect redirect = options.redirect;
147
148 bool isNavigation = options.mode == FetchOptions::Mode::Navigate;
149
150 ASSERT(globalScope.registration().active());
151 ASSERT(globalScope.registration().active()->identifier() == globalScope.thread().identifier());
152 ASSERT(globalScope.registration().active()->state() == ServiceWorkerState::Activated);
153
154 auto* formData = request.httpBody();
155 std::optional<FetchBody> body;
156 if (formData && !formData->isEmpty()) {
157 body = FetchBody::fromFormData(globalScope, *formData);
158 if (!body) {
159 client->didNotHandle();
160 return;
161 }
162 }
163 // FIXME: loading code should set redirect mode to manual.
164 if (isNavigation)
165 options.redirect = FetchOptions::Redirect::Manual;
166
167 URL requestURL = request.url();
168 auto fetchRequest = FetchRequest::create(globalScope, WTFMove(body), WTFMove(requestHeaders), WTFMove(request), WTFMove(options), WTFMove(referrer));
169
170 FetchEvent::Init init;
171 init.request = WTFMove(fetchRequest);
172 if (isNavigation) {
173 // FIXME: Set reservedClientId.
174 if (clientId)
175 init.targetClientId = clientId->toString();
176 } else if (clientId)
177 init.clientId = clientId->toString();
178 init.cancelable = true;
179 auto event = FetchEvent::create(eventNames().fetchEvent, WTFMove(init), Event::IsTrusted::Yes);
180
181 CertificateInfo certificateInfo = globalScope.certificateInfo();
182
183 event->onResponse([client, mode, redirect, requestURL, certificateInfo = WTFMove(certificateInfo)] (auto&& result) mutable {
184 processResponse(WTFMove(client), WTFMove(result), mode, redirect, requestURL, WTFMove(certificateInfo));
185 });
186
187 globalScope.dispatchEvent(event);
188
189 if (!event->respondWithEntered()) {
190 if (event->defaultPrevented()) {
191 client->didFail(ResourceError { errorDomainWebKitInternal, 0, requestURL, "Fetch event was canceled"_s });
192 return;
193 }
194 client->didNotHandle();
195 }
196
197 globalScope.updateExtendedEventsSet(event.ptr());
198}
