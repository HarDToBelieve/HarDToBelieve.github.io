41namespace WebCore {
42
43class SWServer;
44class SWServerRegistration;
45
46class RegistrationStore : public CanMakeWeakPtr<RegistrationStore> {
47WTF_MAKE_FAST_ALLOCATED;
48public:
49 RegistrationStore(SWServer&, String&& databaseDirectory);
50 ~RegistrationStore();
51
52 void clearAll(CompletionHandler<void()>&&);
53 void flushChanges(CompletionHandler<void()>&&);
54
55 void closeDatabase(CompletionHandler<void()>&&);
56 void startSuspension(CompletionHandler<void()>&&);
57 void endSuspension();
58
59 // Callbacks from the SWServer
60 void updateRegistration(const ServiceWorkerContextData&);
61 void removeRegistration(const ServiceWorkerRegistrationKey&);
62
63 // Callbacks from the database
64 void addRegistrationFromDatabase(ServiceWorkerContextData&&);
65 void databaseFailedToOpen();
66 void databaseOpenedAndRecordsImported();
67 void didSaveWorkerScriptsToDisk(ServiceWorkerIdentifier, ScriptBuffer&& mainScript, HashMap<URL, ScriptBuffer>&& importedScripts);
68
69 SWServer& server() { return m_server; };
70
71private:
72 void scheduleDatabasePushIfNecessary();
73 void pushChangesToDatabase(CompletionHandler<void()>&&);
74 void pushChangesToDatabase() { pushChangesToDatabase({ }); }
75
76 SWServer& m_server;
77 Ref<RegistrationDatabase> m_database;
78
79 HashMap<ServiceWorkerRegistrationKey, std::optional<ServiceWorkerContextData>> m_updatedRegistrations;
80 Timer m_databasePushTimer;
81
82 bool m_isSuspended { false };
83 bool m_needsPushingChanges { false };
84};
