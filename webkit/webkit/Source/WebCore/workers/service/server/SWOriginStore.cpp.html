33namespace WebCore {
34
35void SWOriginStore::add(const SecurityOriginData& origin)
36{
37 ++m_originCounts.ensure(origin, [&] {
38 addToStore(origin);
39 return 0;
40 }).iterator->value;
41}
42
43void SWOriginStore::remove(const SecurityOriginData& origin)
44{
45 auto iterator = m_originCounts.find(origin);
46 ASSERT(iterator != m_originCounts.end());
47 if (iterator == m_originCounts.end())
48 return;
49
50 if (--iterator->value)
51 return;
52
53 m_originCounts.remove(iterator);
54 removeFromStore(origin);
55}
56
57void SWOriginStore::clear(const SecurityOriginData& origin)
58{
59 m_originCounts.remove(origin);
60 removeFromStore(origin);
61}
62
63void SWOriginStore::clearAll()
64{
65 m_originCounts.clear();
66 clearStore();
67}
