40static bool shouldUseFileMapping(uint64_t fileSize)
41{
42 return fileSize >= pageSize();
43}

45SWScriptStorage::SWScriptStorage(const String& directory)
46 : m_directory(directory)
47 , m_salt(FileSystem::readOrMakeSalt(saltPath()).value_or(FileSystem::Salt()))
48{
49 ASSERT(!isMainThread());
50}

52String SWScriptStorage::sha2Hash(const String& input) const
53{
54 auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
55 crypto->addBytes(m_salt.data(), m_salt.size());
56 auto inputUtf8 = input.utf8();
57 crypto->addBytes(inputUtf8.data(), inputUtf8.length());
58 auto hash = crypto->computeHash();
59 return base64URLEncodeToString(hash.data(), hash.size());
60}
61
62String SWScriptStorage::sha2Hash(const URL& input) const
63{
64 return sha2Hash(input.string());
65}
66
67String SWScriptStorage::saltPath() const
68{
69 return FileSystem::pathByAppendingComponent(m_directory, "salt");
70}

72String SWScriptStorage::registrationDirectory(const ServiceWorkerRegistrationKey& registrationKey) const
73{
74 return FileSystem::pathByAppendingComponents(m_directory, { sha2Hash(registrationKey.topOrigin().toString()), sha2Hash(registrationKey.scope()) });
75}

77String SWScriptStorage::scriptPath(const ServiceWorkerRegistrationKey& registrationKey, const URL& scriptURL) const
78{
79 return FileSystem::pathByAppendingComponent(registrationDirectory(registrationKey), sha2Hash(scriptURL));
80}

82ScriptBuffer SWScriptStorage::store(const ServiceWorkerRegistrationKey& registrationKey, const URL& scriptURL, const ScriptBuffer& script)
83{
84 ASSERT(!isMainThread());
85
86 auto scriptPath = this->scriptPath(registrationKey, scriptURL);
87 FileSystem::makeAllDirectories(FileSystem::parentPath(scriptPath));
88
89 auto iterateOverBufferAndWriteData = [&](const Function<bool(const uint8_t*, size_t)>& writeData) {
90 for (auto it = script.buffer()->begin(); it != script.buffer()->end(); ++it)
91 writeData(it->segment->data(), it->segment->size());
92 };
93
94 if (!shouldUseFileMapping(script.buffer()->size())) {
95 auto handle = FileSystem::openFile(scriptPath, FileSystem::FileOpenMode::Write);
96 if (!FileSystem::isHandleValid(handle)) {
97 RELEASE_LOG_ERROR(ServiceWorker, "SWScriptStorage::store: Failure to store %s, FileSystem::openFile() failed", scriptPath.utf8().data());
98 return { };
99 }
100 iterateOverBufferAndWriteData([&](const uint8_t* data, size_t size) {
101 FileSystem::writeToFile(handle, data, size);
102 return true;
103 });
104 FileSystem::closeFile(handle);
105 return script;
106 }
107
108 // Make sure we delete the file before writing as there may be code using a mmap'd version of this file.
109 FileSystem::deleteFile(scriptPath);
110 auto mappedFile = FileSystem::mapToFile(scriptPath, script.buffer()->size(), WTFMove(iterateOverBufferAndWriteData));
111 if (!mappedFile) {
112 RELEASE_LOG_ERROR(ServiceWorker, "SWScriptStorage::store: Failure to store %s, FileSystem::mapToFile() failed", scriptPath.utf8().data());
113 return { };
114 }
115 return ScriptBuffer { SharedBuffer::create(WTFMove(mappedFile)) };
116}
117
118ScriptBuffer SWScriptStorage::retrieve(const ServiceWorkerRegistrationKey& registrationKey, const URL& scriptURL)
119{
120 ASSERT(!isMainThread());
121
122 auto scriptPath = this->scriptPath(registrationKey, scriptURL);
123 auto fileSize = FileSystem::fileSize(scriptPath);
124 if (!fileSize) {
125 RELEASE_LOG_ERROR(ServiceWorker, "SWScriptStorage::retrieve: Failure to retrieve %s, FileSystem::fileSize() failed", scriptPath.utf8().data());
126 return { };
127 }
128
129 // FIXME: Do we need to disable file mapping in more cases to avoid having too many file descriptors open?
130 return SharedBuffer::createWithContentsOfFile(scriptPath, FileSystem::MappedFileMode::Private, shouldUseFileMapping(*fileSize) ? SharedBuffer::MayUseFileMapping::Yes : SharedBuffer::MayUseFileMapping::No);
131}

133void SWScriptStorage::clear(const ServiceWorkerRegistrationKey& registrationKey)
134{
135 auto registrationDirectory = this->registrationDirectory(registrationKey);
136 bool result = FileSystem::deleteNonEmptyDirectory(registrationDirectory);
137 RELEASE_LOG_ERROR_IF(!result, ServiceWorker, "SWScriptStorage::clear: Failure to clear scripts for registration %s", registrationKey.toDatabaseKey().utf8().data());
138}
