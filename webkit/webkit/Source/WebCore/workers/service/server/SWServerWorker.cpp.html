52// FIXME: Use r-value references for script and contentSecurityPolicy
53SWServerWorker::SWServerWorker(SWServer& server, SWServerRegistration& registration, const URL& scriptURL, const ScriptBuffer& script, const CertificateInfo& certificateInfo, const ContentSecurityPolicyResponseHeaders& contentSecurityPolicy, String&& referrerPolicy, WorkerType type, ServiceWorkerIdentifier identifier, HashMap<URL, ServiceWorkerContextData::ImportedScript>&& scriptResourceMap)
54 : m_server(makeWeakPtr(server))
55 , m_registrationKey(registration.key())
56 , m_registration(makeWeakPtr(registration))
57 , m_data { identifier, scriptURL, ServiceWorkerState::Redundant, type, registration.identifier() }
58 , m_script(script)
59 , m_certificateInfo(certificateInfo)
60 , m_contentSecurityPolicy(contentSecurityPolicy)
61 , m_referrerPolicy(WTFMove(referrerPolicy))
62 , m_registrableDomain(m_data.scriptURL)
63 , m_scriptResourceMap(WTFMove(scriptResourceMap))
64 , m_terminationTimer(*this, &SWServerWorker::terminationTimerFired)
65 , m_lastNavigationWasAppBound(m_server->clientIsAppBoundForRegistrableDomain(m_registrableDomain))
66{
67 m_data.scriptURL.removeFragmentIdentifier();
68
69 auto result = allWorkers().add(identifier, this);
70 ASSERT_UNUSED(result, result.isNewEntry);
71
72 ASSERT(m_server->getRegistration(m_registrationKey) == &registration);
73}

75SWServerWorker::~SWServerWorker()
76{
77 ASSERT(m_whenActivatedHandlers.isEmpty());
78 callWhenActivatedHandler(false);
79
80 auto taken = allWorkers().take(identifier());
81 ASSERT_UNUSED(taken, taken == this);
82
83 callTerminationCallbacks();
84}

104void SWServerWorker::terminate(CompletionHandler<void()>&& callback)
105{
106 if (!m_server)
107 return callback();
108
109 switch (m_state) {
110 case State::Running:
111 startTermination(WTFMove(callback));
112 return;
113 case State::Terminating:
114 m_terminationCallbacks.append(WTFMove(callback));
115 return;
116 case State::NotRunning:
117 return callback();
118 }
119}
120
121void SWServerWorker::whenTerminated(CompletionHandler<void()>&& callback)
122{
123 ASSERT(isRunning() || isTerminating());
124 m_terminationCallbacks.append(WTFMove(callback));
125}
126
127void SWServerWorker::startTermination(CompletionHandler<void()>&& callback)
128{
129 auto* contextConnection = this->contextConnection();
130 ASSERT(contextConnection);
131 if (!contextConnection) {
132 RELEASE_LOG_ERROR(ServiceWorker, "Request to terminate a worker %" PRIu64 " whose context connection does not exist", identifier().toUInt64());
133 setState(State::NotRunning);
134 callback();
135 m_server->workerContextTerminated(*this);
136 return;
137 }
138
139 setState(State::Terminating);
140
141 m_terminationCallbacks.append(WTFMove(callback));
142 m_terminationTimer.startOneShot(SWServer::defaultTerminationDelay);
143
144 contextConnection->terminateWorker(identifier());
145}
146
147void SWServerWorker::terminationCompleted()
148{
149 m_terminationTimer.stop();
150 callTerminationCallbacks();
151}
152
153void SWServerWorker::callTerminationCallbacks()
154{
155 auto callbacks = WTFMove(m_terminationCallbacks);
156 for (auto& callback : callbacks)
157 callback();
158}

160void SWServerWorker::terminationTimerFired()

194void SWServerWorker::didFinishInstall(const std::optional<ServiceWorkerJobDataIdentifier>& jobDataIdentifier, bool wasSuccessful)
195{
196 auto state = this->state();

198 return;
199
200 ASSERT(m_server);

202 if (m_server)

254void SWServerWorker::didSaveScriptsToDisk(ScriptBuffer&& mainScript, HashMap<URL, ScriptBuffer>&& importedScripts)
255{
256 // Send mmap'd version of the scripts to the ServiceWorker process so we can save dirty memory.
257 if (auto* contextConnection = this->contextConnection())
258 contextConnection->didSaveScriptsToDisk(identifier(), mainScript, importedScripts);
259
260 // The scripts were saved to disk, replace our scripts with the mmap'd version to save dirty memory.
261 ASSERT(mainScript == m_script); // Do a memcmp to make sure the scripts are identical.
262 m_script = WTFMove(mainScript);
263 for (auto& pair : importedScripts) {
264 auto it = m_scriptResourceMap.find(pair.key);
265 ASSERT(it != m_scriptResourceMap.end());
266 if (it == m_scriptResourceMap.end())
267 continue;
268 ASSERT(it->value.script == pair.value); // Do a memcmp to make sure the scripts are identical.
269 it->value.script = WTFMove(pair.value);
270 }
271}

282void SWServerWorker::setHasPendingEvents(bool hasPendingEvents)
283{
284 if (m_hasPendingEvents == hasPendingEvents)
285 return;
286
287 m_hasPendingEvents = hasPendingEvents;
288 if (m_hasPendingEvents)
289 return;
290
291 // Do tryClear/tryActivate, as per https://w3c.github.io/ServiceWorker/#wait-until-method.
292 if (!m_registration)
293 return;
294
295 if (m_registration->isUnregistered() && m_registration->tryClear())
296 return;
297 m_registration->tryActivate();
298}

300void SWServerWorker::whenActivated(CompletionHandler<void(bool)>&& handler)
301{
302 if (state() == ServiceWorkerState::Activating) {
303 m_whenActivatedHandlers.append(WTFMove(handler));
304 return;
305 }
306 ASSERT(state() == ServiceWorkerState::Activated);
307 handler(state() == ServiceWorkerState::Activated);
308}

310void SWServerWorker::setState(ServiceWorkerState state)
311{
312 if (state == ServiceWorkerState::Redundant)
313 terminate();
314
315 m_data.state = state;
316
317 ASSERT(m_registration || state == ServiceWorkerState::Redundant);
318 if (m_registration) {
319 m_registration->forEachConnection([&](auto& connection) {
320 connection.updateWorkerStateInClient(this->identifier(), state);
321 });
322 }
323
324 if (state == ServiceWorkerState::Activated || state == ServiceWorkerState::Redundant)
325 callWhenActivatedHandler(state == ServiceWorkerState::Activated);
326}
327
328void SWServerWorker::callWhenActivatedHandler(bool success)
329{
330 auto whenActivatedHandlers = WTFMove(m_whenActivatedHandlers);
331 for (auto& handler : whenActivatedHandlers)
332 handler(success);
333}
334
335void SWServerWorker::setState(State state)
336{
337 ASSERT(state != State::Running || m_registration);
338 ASSERT(state != State::Running || m_state != State::Terminating);
339 m_state = state;
340
341 switch (state) {
342 case State::Running:
343 m_shouldSkipHandleFetch = false;
344 break;
345 case State::Terminating:
346 callWhenActivatedHandler(false);
347 break;
348 case State::NotRunning:
349 terminationCompleted();
350
351 callWhenActivatedHandler(false);
352 // As per https://w3c.github.io/ServiceWorker/#activate, a worker goes to activated even if activating fails.
353 if (m_data.state == ServiceWorkerState::Activating)
354 didFinishActivation();
355 break;
356 }
357}
