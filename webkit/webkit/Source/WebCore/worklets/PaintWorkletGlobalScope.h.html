44class PaintWorkletGlobalScope final : public WorkletGlobalScope {
45 WTF_MAKE_ISO_ALLOCATED(PaintWorkletGlobalScope);
46public:
47 static RefPtr<PaintWorkletGlobalScope> tryCreate(Document&, ScriptSourceCode&&);
48
49 ExceptionOr<void> registerPaint(JSC::JSGlobalObject&, const String& name, JSC::Strong<JSC::JSObject> paintConstructor);
50 double devicePixelRatio() const;
51
52 // All paint definitions must be destroyed before the vm is destroyed, because otherwise they will point to freed memory.
53 struct PaintDefinition : public CanMakeWeakPtr<PaintDefinition> {
54 WTF_MAKE_STRUCT_FAST_ALLOCATED;
55 PaintDefinition(const AtomString& name, JSC::JSObject* paintConstructor, Ref<CSSPaintCallback>&&, Vector<String>&& inputProperties, Vector<String>&& inputArguments);
56
57 const AtomString name;
58 const JSC::JSObject* const paintConstructor;
59 const Ref<CSSPaintCallback> paintCallback;
60 const Vector<String> inputProperties;
61 const Vector<String> inputArguments;
62 };
63
64 HashMap<String, std::unique_ptr<PaintDefinition>>& paintDefinitionMap() WTF_REQUIRES_LOCK(m_paintDefinitionLock);
65 Lock& paintDefinitionLock() WTF_RETURNS_LOCK(m_paintDefinitionLock) { return m_paintDefinitionLock; }
66
67 void prepareForDestruction() final
68 {
69 if (m_hasPreparedForDestruction)
70 return;
71 m_hasPreparedForDestruction = true;
72
73 {
74 Locker locker { paintDefinitionLock() };
75 paintDefinitionMap().clear();
76 }
77 WorkletGlobalScope::prepareForDestruction();
78 }
79
80 FetchOptions::Destination destination() const final { return FetchOptions::Destination::Paintworklet; }
81
82private:
83 PaintWorkletGlobalScope(Document&, Ref<JSC::VM>&&, ScriptSourceCode&&);
84
85 ~PaintWorkletGlobalScope()
86 {
87#if ASSERT_ENABLED
88 Locker locker { paintDefinitionLock() };
89 ASSERT(paintDefinitionMap().isEmpty());
90#endif
91 }
92
93 bool isPaintWorkletGlobalScope() const final { return true; }
94
95 HashMap<String, std::unique_ptr<PaintDefinition>> m_paintDefinitionMap WTF_GUARDED_BY_LOCK(m_paintDefinitionLock);
96 Lock m_paintDefinitionLock;
97 bool m_hasPreparedForDestruction { false };
98};
99
100inline auto PaintWorkletGlobalScope::paintDefinitionMap() -> HashMap<String, std::unique_ptr<PaintDefinition>>&
101{
102 ASSERT(m_paintDefinitionLock.isLocked());
103 return m_paintDefinitionMap;
104}
105
106} // namespace WebCore

110static bool isType(const WebCore::WorkletGlobalScope& context) { return context.isPaintWorkletGlobalScope(); }
