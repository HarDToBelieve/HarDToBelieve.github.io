31namespace WebCore {
32
33WorkletPendingTasks::WorkletPendingTasks(Worklet& worklet, DOMPromiseDeferred<void>&& promise, int counter)
34 : m_worklet(makeWeakPtr(worklet))
35 , m_promise(WTFMove(promise))
36 , m_counter(counter)
37{
38 ASSERT(isMainThread());
39}
40
41void WorkletPendingTasks::abort(Exception&& exception)
42{
43 ASSERT(isMainThread());
44
45 if (m_counter == -1)
46 return;
47
48 m_counter = -1;
49 m_promise.reject(WTFMove(exception));
50 if (m_worklet)
51 m_worklet->finishPendingTasks(*this);
52}
53
54void WorkletPendingTasks::decrementCounter()
55{
56 ASSERT(isMainThread());
57
58 if (m_counter == -1)
59 return;
60
61 --m_counter;
62 if (!m_counter) {
63 m_promise.resolve();
64 if (m_worklet)
65 m_worklet->finishPendingTasks(*this);
66 }
67}
