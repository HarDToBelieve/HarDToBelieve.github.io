28#include "JSDOMPromiseDeferred.h"
29#include <wtf/RefCounted.h>
30#include <wtf/WeakPtr.h>
31
32namespace WebCore {
33
34class Worklet;
35
36// https://drafts.css-houdini.org/worklets/#pending-tasks-struct
37class WorkletPendingTasks : public ThreadSafeRefCounted<WorkletPendingTasks> {
38public:
39 static Ref<WorkletPendingTasks> create(Worklet& worklet, DOMPromiseDeferred<void>&& promise, int counter)
40 {
41 return adoptRef(*new WorkletPendingTasks(worklet, WTFMove(promise), counter));
42 }
43
44 void abort(Exception&&);
45 void decrementCounter();
46
47private:
48 WorkletPendingTasks(Worklet&, DOMPromiseDeferred<void>&&, int counter);
49
50 WeakPtr<Worklet> m_worklet;
51 DOMPromiseDeferred<void> m_promise;
52 int m_counter;
53};

55} // namespace WebCore
