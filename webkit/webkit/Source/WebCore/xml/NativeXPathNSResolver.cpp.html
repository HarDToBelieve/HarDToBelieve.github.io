41
42String NativeXPathNSResolver::lookupNamespaceURI(const String& prefix)
43{
44 // This is not done by Node::lookupNamespaceURI as per the DOM3 Core spec,
45 // but the XPath spec says that we should do it for XPathNSResolver.
46 if (prefix == "xml")
47 return XMLNames::xmlNamespaceURI.get();
48
49 return m_node->lookupNamespaceURI(prefix).string();
50}
