56void XMLErrors::handleError(ErrorType type, const char* message, int lineNumber, int columnNumber)
57{
58 handleError(type, message, TextPosition(OrdinalNumber::fromOneBasedInt(lineNumber), OrdinalNumber::fromOneBasedInt(columnNumber)));
59}
60
61void XMLErrors::handleError(ErrorType type, const char* message, TextPosition position)
62{
63 if (type == fatal || (m_errorCount < maxErrors && (!m_lastErrorPosition || (m_lastErrorPosition->m_line != position.m_line && m_lastErrorPosition->m_column != position.m_column)))) {
64 switch (type) {
65 case warning:
66 appendErrorMessage("warning", position, message);
67 break;
68 case fatal:
69 case nonFatal:
70 appendErrorMessage("error", position, message);
71 }
72
73 m_lastErrorPosition = position;
74 ++m_errorCount;
75 }
76}
77
78void XMLErrors::appendErrorMessage(const String& typeString, TextPosition position, const char* message)
79{
80 // <typeString> on line <lineNumber> at column <columnNumber>: <message>
81 m_errorMessages.append(typeString, " on line ", position.m_line.oneBasedInt(), " at column ", position.m_column.oneBasedInt(), ": ", message);
82}

84static inline Ref<Element> createXHTMLParserErrorHeader(Document& document, const String& errorMessages)
85{
86 Ref<Element> reportElement = document.createElement(QualifiedName(nullAtom(), "parsererror", xhtmlNamespaceURI), true);
87
88 Vector<Attribute> reportAttributes;
89 reportAttributes.append(Attribute(styleAttr, "display: block; white-space: pre; border: 2px solid #c77; padding: 0 1em 0 1em; margin: 1em; background-color: #fdd; color: black"));
90 reportElement->parserSetAttributes(reportAttributes);
91
92 auto h3 = HTMLHeadingElement::create(h3Tag, document);
93 reportElement->parserAppendChild(h3);
94 h3->parserAppendChild(Text::create(document, "This page contains the following errors:"_s));
95
96 auto fixed = HTMLDivElement::create(document);
97 Vector<Attribute> fixedAttributes;
98 fixedAttributes.append(Attribute(styleAttr, "font-family:monospace;font-size:12px"));
99 fixed->parserSetAttributes(fixedAttributes);
100 reportElement->parserAppendChild(fixed);
101
102 fixed->parserAppendChild(Text::create(document, errorMessages));
103
104 h3 = HTMLHeadingElement::create(h3Tag, document);
105 reportElement->parserAppendChild(h3);
106 h3->parserAppendChild(Text::create(document, "Below is a rendering of the page up to the first error."_s));
107
108 return reportElement;
109}

111void XMLErrors::insertErrorMessageBlock()
112{
113 // One or more errors occurred during parsing of the code. Display an error block to the user above
114 // the normal content (the DOM tree is created manually and includes line/col info regarding
115 // where the errors are located)
116
117 // Create elements for display
118 RefPtr<Element> documentElement = m_document.documentElement();
119 if (!documentElement) {
120 auto rootElement = HTMLHtmlElement::create(m_document);
121 auto body = HTMLBodyElement::create(m_document);
122 rootElement->parserAppendChild(body);
123 m_document.parserAppendChild(rootElement);
124 documentElement = WTFMove(body);
125 } else if (documentElement->namespaceURI() == SVGNames::svgNamespaceURI) {
126 auto rootElement = HTMLHtmlElement::create(m_document);
127 auto head = HTMLHeadElement::create(m_document);
128 auto style = HTMLStyleElement::create(m_document);
129 head->parserAppendChild(style);
130 style->parserAppendChild(m_document.createTextNode("html, body { height: 100% } parsererror + svg { width: 100%; height: 100% }"_s));
131 style->finishParsingChildren();
132 rootElement->parserAppendChild(head);
133 auto body = HTMLBodyElement::create(m_document);
134 rootElement->parserAppendChild(body);
135
136 m_document.parserRemoveChild(*documentElement);
137 if (!documentElement->parentNode())
138 body->parserAppendChild(*documentElement);
139
140 m_document.parserAppendChild(rootElement);
141
142 documentElement = WTFMove(body);
143 }
144
145 String errorMessages = m_errorMessages.toString();
146 auto reportElement = createXHTMLParserErrorHeader(m_document, errorMessages);
147
148#if ENABLE(XSLT)
149 if (m_document.transformSourceDocument()) {
150 Vector<Attribute> attributes;
151 attributes.append(Attribute(styleAttr, "white-space: normal"));
152 auto paragraph = HTMLParagraphElement::create(m_document);
153 paragraph->parserSetAttributes(attributes);
154 paragraph->parserAppendChild(m_document.createTextNode("This document was created as the result of an XSL transformation. The line and column numbers given are from the transformed result."_s));
155 reportElement->parserAppendChild(paragraph);
156 }
157#endif
158
159 Node* firstChild = documentElement->firstChild();
160 if (firstChild)
161 documentElement->parserInsertBefore(reportElement, *firstChild);
162 else
163 documentElement->parserAppendChild(reportElement);
164
165 m_document.updateStyleIfNeeded();
166}
