82static void replaceCharsetInMediaTypeIfNeeded(String& mediaType)
83{
84 auto parsedContentType = ParsedContentType::create(mediaType);
85 if (!parsedContentType || parsedContentType->charset().isEmpty() || equalIgnoringASCIICase(parsedContentType->charset(), "UTF-8"))
86 return;
87
88 parsedContentType->setCharset("UTF-8");
89 mediaType = parsedContentType->serialize();
90}

161ExceptionOr<Document*> XMLHttpRequest::responseXML()
162{
163 ASSERT(scriptExecutionContext()->isDocument());
164
165 if (responseType() != ResponseType::EmptyString && responseType() != ResponseType::Document)
166 return Exception { InvalidStateError };
167
168 if (!doneWithoutErrors())
169 return nullptr;
170
171 if (!m_createdDocument) {
172 auto& context = downcast<Document>(*scriptExecutionContext());
173
174 String mimeType = responseMIMEType();
175 bool isHTML = equalLettersIgnoringASCIICase(mimeType, "text/html");
176 bool isXML = MIMETypeRegistry::isXMLMIMEType(mimeType);
177
178 // The W3C spec requires the final MIME type to be some valid XML type, or text/html.
179 // If it is text/html, then the responseType of "document" must have been supplied explicitly.
180 if ((m_response.isInHTTPFamily() && !isXML && !isHTML)
181 || (isHTML && responseType() == ResponseType::EmptyString)) {
182 m_responseDocument = nullptr;
183 } else {
184 if (isHTML)
185 m_responseDocument = HTMLDocument::create(nullptr, context.settings(), m_response.url());
186 else
187 m_responseDocument = XMLDocument::create(nullptr, context.settings(), m_response.url());
188 m_responseDocument->overrideLastModified(m_response.lastModified());
189 m_responseDocument->setContextDocument(context);
190 m_responseDocument->setSecurityOriginPolicy(context.securityOriginPolicy());
191 m_responseDocument->overrideMIMEType(mimeType);
192 m_responseDocument->setContent(m_responseBuilder.toStringPreserveCapacity());
193
194 if (!m_responseDocument->wellFormed())
195 m_responseDocument = nullptr;
196 }
197 m_createdDocument = true;
198 }
199
200 return m_responseDocument.get();
201}

203Ref<Blob> XMLHttpRequest::createResponseBlob()
204{
205 ASSERT(responseType() == ResponseType::Blob);
206 ASSERT(doneWithoutErrors());
207
208 // FIXME: We just received the data from NetworkProcess, and are sending it back. This is inefficient.
209 Vector<uint8_t> data;
210 if (m_binaryResponseBuilder)
211 data = std::exchange(m_binaryResponseBuilder, nullptr)->takeData();
212 String normalizedContentType = Blob::normalizedContentType(responseMIMEType(FinalMIMEType::Yes)); // responseMIMEType defaults to text/xml which may be incorrect.
213 return Blob::create(scriptExecutionContext(), WTFMove(data), normalizedContentType);
214}

226ExceptionOr<void> XMLHttpRequest::setTimeout(unsigned timeout)
227{
228 if (scriptExecutionContext()->isDocument() && !m_async) {
229 logConsoleError(scriptExecutionContext(), "XMLHttpRequest.timeout cannot be set for synchronous HTTP(S) requests made from the window context.");
230 return Exception { InvalidAccessError };
231 }
232 m_timeoutMilliseconds = timeout;
233 if (!m_timeoutTimer.isActive())
234 return { };
235
236 // If timeout is zero, we should use the default network timeout. But we disabled it so let's mimic it with a 60 seconds timeout value.
237 Seconds interval = Seconds { m_timeoutMilliseconds ? m_timeoutMilliseconds / 1000. : 60. } - (MonotonicTime::now() - m_sendingTime);
238 m_timeoutTimer.startOneShot(std::max(interval, 0_s));
239 return { };
240}

242ExceptionOr<void> XMLHttpRequest::setResponseType(ResponseType type)
243{
244 if (!scriptExecutionContext()->isDocument() && type == ResponseType::Document)
245 return { };
246
247 if (readyState() >= LOADING)
248 return Exception { InvalidStateError };
249
250 // Newer functionality is not available to synchronous requests in window contexts, as a spec-mandated
251 // attempt to discourage synchronous XHR use. responseType is one such piece of functionality.
252 // We'll only disable this functionality for HTTP(S) requests since sync requests for local protocols
253 // such as file: and data: still make sense to allow.
254 if (!m_async && scriptExecutionContext()->isDocument() && m_url.protocolIsInHTTPFamily()) {
255 logConsoleError(scriptExecutionContext(), "XMLHttpRequest.responseType cannot be changed for synchronous HTTP(S) requests made from the window context.");
256 return Exception { InvalidAccessError };
257 }
258
259 m_responseType = static_cast<unsigned>(type);
260 return { };

278void XMLHttpRequest::changeState(State newState)
279{
280 if (readyState() != newState) {
281 // Setting the readyState to DONE could get the wrapper collected before we get a chance to fire the JS
282 // events in callReadyStateChangeListener() below so we extend the lifetime of the JS wrapper until the
283 // of this scope.
284 auto eventFiringActivity = makePendingActivity(*this);
285
286 m_readyState = static_cast<State>(newState);
287 if (readyState() == DONE) {
288 // The XHR object itself holds on to the responseText, and
289 // thus has extra cost even independent of any
290 // responseText or responseXML objects it has handed
291 // out. But it is protected from GC while loading, so this
292 // can't be recouped until the load is done, so only
293 // report the extra cost at that point.
294 if (auto* context = scriptExecutionContext()) {
295 JSC::VM& vm = context->vm();
296 JSC::JSLockHolder lock(vm);
297 vm.heap.reportExtraMemoryAllocated(memoryCost());
298 }
299 }
300 callReadyStateChangeListener();
301 }
302}
303
304void XMLHttpRequest::callReadyStateChangeListener()
305{
306 if (!scriptExecutionContext())
307 return;
308
309 // Check whether sending load and loadend events before sending readystatechange event, as it may change m_error/m_readyState values.
310 bool shouldSendLoadEvent = (readyState() == DONE && !m_error);
311
312 if (m_async || (readyState() <= OPENED || readyState() == DONE)) {
313 m_progressEventThrottle.dispatchReadyStateChangeEvent(Event::create(eventNames().readystatechangeEvent, Event::CanBubble::No, Event::IsCancelable::No),
314 readyState() == DONE ? FlushProgressEvent : DoNotFlushProgressEvent);
315 }
316
317 if (shouldSendLoadEvent) {
318 m_progressEventThrottle.dispatchProgressEvent(eventNames().loadEvent);
319 m_progressEventThrottle.dispatchProgressEvent(eventNames().loadendEvent);
320 }
321}

338ExceptionOr<void> XMLHttpRequest::open(const String& method, const URL& url, bool async)
339{
340 auto* context = scriptExecutionContext();
341 bool contextIsDocument = is<Document>(*context);
342 if (contextIsDocument && !downcast<Document>(*context).isFullyActive())
343 return Exception { InvalidStateError, "Document is not fully active"_s };
344
345 if (!isValidHTTPToken(method))
346 return Exception { SyntaxError };
347
348 if (isForbiddenMethod(method))
349 return Exception { SecurityError };
350
351 if (!url.isValid())
352 return Exception { SyntaxError };
353
354 if (!async && contextIsDocument) {
355 // Newer functionality is not available to synchronous requests in window contexts, as a spec-mandated
356 // attempt to discourage synchronous XHR use. responseType is one such piece of functionality.
357 // We'll only disable this functionality for HTTP(S) requests since sync requests for local protocols
358 // such as file: and data: still make sense to allow.
359 if (url.protocolIsInHTTPFamily() && responseType() != ResponseType::EmptyString) {
360 logConsoleError(context, "Synchronous HTTP(S) requests made from the window context cannot have XMLHttpRequest.responseType set.");
361 return Exception { InvalidAccessError };
362 }
363
364 // Similarly, timeouts are disabled for synchronous requests as well.
365 if (m_timeoutMilliseconds > 0) {
366 logConsoleError(context, "Synchronous XMLHttpRequests must not have a timeout value set.");
367 return Exception { InvalidAccessError };
368 }
369 }
370
371 if (!internalAbort())
372 return { };
373
374 m_sendFlag = false;
375 m_uploadListenerFlag = false;
376 m_method = normalizeHTTPMethod(method);
377 m_error = false;
378 m_uploadComplete = false;
379 m_wasAbortedByClient = false;
380
381 // clear stuff from possible previous load
382 clearResponse();
383 clearRequest();
384
385 m_url = url;
386 context->contentSecurityPolicy()->upgradeInsecureRequestIfNeeded(m_url, ContentSecurityPolicy::InsecureRequestType::Load);
387
388 m_async = async;
389
390 ASSERT(!m_loadingActivity);
391
392 changeState(OPENED);
393
394 return { };
395}

408std::optional<ExceptionOr<void>> XMLHttpRequest::prepareToSend()
409{
410 // A return value other than std::nullopt means we should not try to send, and we should return that value to the caller.
411 // std::nullopt means we are ready to send and should continue with the send algorithm.
412
413 if (!scriptExecutionContext())
414 return ExceptionOr<void> { };
415
416 auto& context = *scriptExecutionContext();
417
418 if (is<Document>(context) && downcast<Document>(context).shouldIgnoreSyncXHRs()) {
419 logConsoleError(scriptExecutionContext(), makeString("Ignoring XMLHttpRequest.send() call for '", m_url.string(), "' because the maximum number of synchronous failures was reached."));
420 return ExceptionOr<void> { };
421 }
422
423 if (readyState() != OPENED || m_sendFlag)
424 return ExceptionOr<void> { Exception { InvalidStateError } };
425 ASSERT(!m_loadingActivity);
426
427 // FIXME: Convert this to check the isolated world's Content Security Policy once webkit.org/b/104520 is solved.
428 if (!context.shouldBypassMainWorldContentSecurityPolicy() && !context.contentSecurityPolicy()->allowConnectToSource(m_url)) {
429 if (!m_async)
430 return ExceptionOr<void> { Exception { NetworkError } };
431 m_timeoutTimer.stop();
432 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this] {
433 networkError();
434 });
435 return ExceptionOr<void> { };
436 }
437
438 m_error = false;
439 return std::nullopt;
440}

442ExceptionOr<void> XMLHttpRequest::send(std::optional<SendTypes>&& sendType)
443{
444 InspectorInstrumentation::willSendXMLHttpRequest(scriptExecutionContext(), url().string());
445 m_userGestureToken = UserGestureIndicator::currentUserGesture();
446
447 ExceptionOr<void> result;
448 if (!sendType)
449 result = send();
450 else {
451 result = WTF::switchOn(sendType.value(),
452 [this] (const RefPtr<Document>& document) -> ExceptionOr<void> { return send(*document); },
453 [this] (const RefPtr<Blob>& blob) -> ExceptionOr<void> { return send(*blob); },
454 [this] (const RefPtr<JSC::ArrayBufferView>& arrayBufferView) -> ExceptionOr<void> { return send(*arrayBufferView); },
455 [this] (const RefPtr<JSC::ArrayBuffer>& arrayBuffer) -> ExceptionOr<void> { return send(*arrayBuffer); },
456 [this] (const RefPtr<DOMFormData>& formData) -> ExceptionOr<void> { return send(*formData); },
457 [this] (const RefPtr<URLSearchParams>& searchParams) -> ExceptionOr<void> { return send(*searchParams); },
458 [this] (const String& string) -> ExceptionOr<void> { return send(string); }
459 );
460 }
461
462 return result;
463}

465ExceptionOr<void> XMLHttpRequest::send(Document& document)
466{
467 if (auto result = prepareToSend())
468 return WTFMove(result.value());
469
470 if (m_method != "GET" && m_method != "HEAD") {
471 if (!m_requestHeaders.contains(HTTPHeaderName::ContentType)) {
472 // FIXME: this should include the charset used for encoding.
473 m_requestHeaders.set(HTTPHeaderName::ContentType, document.isHTMLDocument() ? "text/html;charset=UTF-8"_s : "application/xml;charset=UTF-8"_s);
474 } else {
475 String contentType = m_requestHeaders.get(HTTPHeaderName::ContentType);
476 replaceCharsetInMediaTypeIfNeeded(contentType);
477 m_requestHeaders.set(HTTPHeaderName::ContentType, contentType);
478 }
479
480 // FIXME: According to XMLHttpRequest Level 2, this should use the Document.innerHTML algorithm
481 // from the HTML5 specification to serialize the document.
482 m_requestEntityBody = FormData::create(UTF8Encoding().encode(serializeFragment(document, SerializedNodes::SubtreeIncludingNode), UnencodableHandling::Entities));
483 if (m_upload)
484 m_requestEntityBody->setAlwaysStream(true);
485 }
486
487 return createRequest();
488}

490ExceptionOr<void> XMLHttpRequest::send(const String& body)
491{
492 if (auto result = prepareToSend())
493 return WTFMove(result.value());
494
495 if (!body.isNull() && m_method != "GET" && m_method != "HEAD") {
496 String contentType = m_requestHeaders.get(HTTPHeaderName::ContentType);
497 if (contentType.isNull()) {
498 m_requestHeaders.set(HTTPHeaderName::ContentType, HTTPHeaderValues::textPlainContentType());
499 } else {
500 replaceCharsetInMediaTypeIfNeeded(contentType);
501 m_requestHeaders.set(HTTPHeaderName::ContentType, contentType);
502 }
503
504 m_requestEntityBody = FormData::create(UTF8Encoding().encode(body, UnencodableHandling::Entities));
505 if (m_upload)
506 m_requestEntityBody->setAlwaysStream(true);
507 }
508
509 return createRequest();
510}

512ExceptionOr<void> XMLHttpRequest::send(Blob& body)
513{
514 if (auto result = prepareToSend())
515 return WTFMove(result.value());
516
517 if (m_method != "GET" && m_method != "HEAD") {
518 if (!m_url.protocolIsInHTTPFamily()) {
519 // FIXME: We would like to support posting Blobs to non-http URLs (e.g. custom URL schemes)
520 // but because of the architecture of blob-handling that will require a fair amount of work.
521
522 ASCIILiteral consoleMessage { "POST of a Blob to non-HTTP protocols in XMLHttpRequest.send() is currently unsupported."_s };
523 scriptExecutionContext()->addConsoleMessage(MessageSource::JS, MessageLevel::Warning, consoleMessage);
524
525 return createRequest();
526 }
527
528 if (!m_requestHeaders.contains(HTTPHeaderName::ContentType)) {
529 const String& blobType = body.type();
530 if (!blobType.isEmpty() && isValidContentType(blobType))
531 m_requestHeaders.set(HTTPHeaderName::ContentType, blobType);
532 }
533
534 m_requestEntityBody = FormData::create();
535 m_requestEntityBody->appendBlob(body.url());
536 }
537
538 return createRequest();
539}

588ExceptionOr<void> XMLHttpRequest::createRequest()
589{
590 // Only GET request is supported for blob URL.
591 if (!m_async && m_url.protocolIsBlob() && m_method != "GET")
592 return Exception { NetworkError };
593
594 if (m_async && m_upload && m_upload->hasEventListeners())
595 m_uploadListenerFlag = true;
596
597 ResourceRequest request(m_url);
598 request.setRequester(ResourceRequest::Requester::XHR);
599 request.setInitiatorIdentifier(scriptExecutionContext()->resourceRequestIdentifier());
600 request.setHTTPMethod(m_method);
601
602 if (m_requestEntityBody) {
603 ASSERT(m_method != "GET");
604 ASSERT(m_method != "HEAD");
605 request.setHTTPBody(WTFMove(m_requestEntityBody));
606 }
607
608 if (!m_requestHeaders.isEmpty())
609 request.setHTTPHeaderFields(m_requestHeaders);
610
611 ThreadableLoaderOptions options;
612 options.sendLoadCallbacks = SendCallbackPolicy::SendCallbacks;
613 // The presence of upload event listeners forces us to use preflighting because POSTing to an URL that does not
614 // permit cross origin requests should look exactly like POSTing to an URL that does not respond at all.
615 options.preflightPolicy = m_uploadListenerFlag ? PreflightPolicy::Force : PreflightPolicy::Consider;
616 options.credentials = m_includeCredentials ? FetchOptions::Credentials::Include : FetchOptions::Credentials::SameOrigin;
617 options.mode = FetchOptions::Mode::Cors;
618 options.contentSecurityPolicyEnforcement = scriptExecutionContext()->shouldBypassMainWorldContentSecurityPolicy() ? ContentSecurityPolicyEnforcement::DoNotEnforce : ContentSecurityPolicyEnforcement::EnforceConnectSrcDirective;
619 options.initiator = cachedResourceRequestInitiators().xmlhttprequest;
620 options.sameOriginDataURLFlag = SameOriginDataURLFlag::Set;
621 options.filteringPolicy = ResponseFilteringPolicy::Enable;
622 options.sniffContentEncoding = ContentEncodingSniffingPolicy::DoNotSniff;
623
624 if (m_timeoutMilliseconds) {
625 if (!m_async)
626 request.setTimeoutInterval(m_timeoutMilliseconds / 1000.0);
627 else {
628 request.setTimeoutInterval(std::numeric_limits<double>::infinity());
629 m_sendingTime = MonotonicTime::now();
630 m_timeoutTimer.startOneShot(1_ms * m_timeoutMilliseconds);
631 }
632 }
633
634 m_exceptionCode = std::nullopt;
635 m_error = false;
636 m_uploadComplete = !request.httpBody();
637 m_sendFlag = true;
638
639 if (m_async) {
640 m_progressEventThrottle.dispatchProgressEvent(eventNames().loadstartEvent);
641 if (!m_uploadComplete && m_uploadListenerFlag)
642 m_upload->dispatchProgressEvent(eventNames().loadstartEvent, 0, request.httpBody()->lengthInBytes());
643
644 if (readyState() != OPENED || !m_sendFlag || m_loadingActivity)
645 return { };
646
647 // ThreadableLoader::create can return null here, for example if we're no longer attached to a page or if a content blocker blocks the load.
648 // This is true while running onunload handlers.
649 // FIXME: Maybe we need to be able to send XMLHttpRequests from onunload, <http://bugs.webkit.org/show_bug.cgi?id=10904>.
650 auto loader = ThreadableLoader::create(*scriptExecutionContext(), *this, WTFMove(request), options);
651 if (loader)
652 m_loadingActivity = LoadingActivity { makeRef(*this), loader.releaseNonNull() };
653
654 // Either loader is null or some error was synchronously sent to us.
655 ASSERT(m_loadingActivity || !m_sendFlag);
656 } else {
657 if (scriptExecutionContext()->isDocument() && !isFeaturePolicyAllowedByDocumentAndAllOwners(FeaturePolicy::Type::SyncXHR, *document()))
658 return Exception { NetworkError };
659
660 request.setDomainForCachePartition(scriptExecutionContext()->domainForCachePartition());
661 InspectorInstrumentation::willLoadXHRSynchronously(scriptExecutionContext());

673void XMLHttpRequest::abort()
674{

676
677 m_wasAbortedByClient = true;
678 if (!internalAbort())
679 return;
680
681 clearResponseBuffers();
682
683 m_requestHeaders.clear();
684 if ((readyState() == OPENED && m_sendFlag) || readyState() == HEADERS_RECEIVED || readyState() == LOADING) {
685 ASSERT(!m_loadingActivity);
686 m_sendFlag = false;

688 dispatchErrorEvents(eventNames().abortEvent);
689 }
690 if (readyState() == DONE)
691 m_readyState = static_cast<State>(UNSENT);

694bool XMLHttpRequest::internalAbort()
695{
696 m_error = true;
697
698 // FIXME: when we add the support for multi-part XHR, we will have to think be careful with this initialization.
699 m_receivedLength = 0;
700
701 m_decoder = nullptr;
702
703 m_timeoutTimer.stop();
704
705 if (!m_loadingActivity)
706 return true;
707
708 // Cancelling m_loadingActivity may trigger a window.onload callback which can call open() on the same xhr.
709 // This would create internalAbort reentrant call.
710 // m_loadingActivity is set to std::nullopt before being cancelled to exit early in any reentrant internalAbort() call.
711 auto loadingActivity = std::exchange(m_loadingActivity, std::nullopt);
712 loadingActivity->loader->cancel();
713
714 // If window.onload callback calls open() and send() on the same xhr, m_loadingActivity is now set to a new value.
715 // The function calling internalAbort() should abort to let the open() and send() calls continue properly.
716 // We ask the function calling internalAbort() to exit by returning false.
717 // Save this information to a local variable since we are going to drop protection.
718 bool newLoadStarted = !!m_loadingActivity;
719
720 return !newLoadStarted;
721}

810String XMLHttpRequest::getAllResponseHeaders() const
811{
812 if (readyState() < HEADERS_RECEIVED || m_error)
813 return emptyString();
814
815 if (!m_allResponseHeaders) {
816 Vector<std::pair<String, String>> headers;
817 headers.reserveInitialCapacity(m_response.httpHeaderFields().size());
818
819 for (auto& header : m_response.httpHeaderFields())
820 headers.uncheckedAppend(std::make_pair(header.key, header.value));
821
822 std::sort(headers.begin(), headers.end(), [] (const std::pair<String, String>& x, const std::pair<String, String>& y) {
823 unsigned xLength = x.first.length();
824 unsigned yLength = y.first.length();
825 unsigned commonLength = std::min(xLength, yLength);
826 for (unsigned i = 0; i < commonLength; ++i) {
827 auto xCharacter = toASCIIUpper(x.first[i]);
828 auto yCharacter = toASCIIUpper(y.first[i]);
829 if (xCharacter != yCharacter)
830 return xCharacter < yCharacter;
831 }
832 return xLength < yLength;
833 });
834
835 StringBuilder stringBuilder;
836 for (auto& header : headers)
837 stringBuilder.append(lowercase(header.first), ": ", header.second, "\r\n");
838
839 m_allResponseHeaders = stringBuilder.toString();
840 }
841
842 return m_allResponseHeaders;
843}

884void XMLHttpRequest::didFail(const ResourceError& error)
885{
886 auto protectedThis = makeRef(*this);
887
888 // If we are already in an error state, for instance we called abort(), bail out early.
889 if (m_error)
890 return;
891
892 // The XHR specification says we should only fire an abort event if the cancelation was requested by the client.
893 if (m_wasAbortedByClient && error.isCancellation()) {
894 m_exceptionCode = AbortError;
895 abortError();
896 return;
897 }
898
899 // In case of worker sync timeouts.
900 if (error.isTimeout()) {
901 didReachTimeout();
902 return;
903 }
904
905 // In case didFail is called synchronously on an asynchronous XHR call, let's dispatch network error asynchronously
906 if (m_async && m_sendFlag && !m_loadingActivity) {
907 m_sendFlag = false;
908 m_timeoutTimer.stop();
909 queueTaskKeepingObjectAlive(*this, TaskSource::Networking, [this] {
910 networkError();
911 });
912 return;
913 }
914 m_exceptionCode = NetworkError;
915 networkError();
916}

993Ref<TextResourceDecoder> XMLHttpRequest::createDecoder() const
994{
995 TextEncoding finalResponseCharset = this->finalResponseCharset();
996 if (finalResponseCharset.isValid())
997 return TextResourceDecoder::create("text/plain", finalResponseCharset);
998
999 switch (responseType()) {
1000 case ResponseType::EmptyString:
1001 if (MIMETypeRegistry::isXMLMIMEType(responseMIMEType())) {
1002 auto decoder = TextResourceDecoder::create("application/xml");
1003 // Don't stop on encoding errors, unlike it is done for other kinds of XML resources. This matches the behavior of previous WebKit versions, Firefox and Opera.
1004 decoder->useLenientXMLDecoding();
1005 return decoder;
1006 }
1007 FALLTHROUGH;
1008 case ResponseType::Text:
1009 case ResponseType::Json:
1010 return TextResourceDecoder::create("text/plain", "UTF-8");
1011 case ResponseType::Document: {
1012 if (equalLettersIgnoringASCIICase(responseMIMEType(), "text/html"))
1013 return TextResourceDecoder::create("text/html", "UTF-8");
1014 auto decoder = TextResourceDecoder::create("application/xml");
1015 // Don't stop on encoding errors, unlike it is done for other kinds of XML resources. This matches the behavior of previous WebKit versions, Firefox and Opera.
1016 decoder->useLenientXMLDecoding();
1017 return decoder;
1018 }
1019 case ResponseType::Arraybuffer:
1020 case ResponseType::Blob:
1021 ASSERT_NOT_REACHED();
1022 break;
1023 }
1024 return TextResourceDecoder::create("text/plain", "UTF-8");
1025}

1027void XMLHttpRequest::didReceiveData(const uint8_t* data, int len)
1028{
1029 if (m_error)
1030 return;
1031
1032 if (readyState() < HEADERS_RECEIVED)
1033 changeState(HEADERS_RECEIVED);
1034
1035 if (!m_mimeTypeOverride.isEmpty())
1036 m_responseEncoding = extractCharsetFromMediaType(m_mimeTypeOverride);
1037 if (m_responseEncoding.isEmpty())
1038 m_responseEncoding = m_response.textEncodingName();
1039
1040 bool useDecoder = shouldDecodeResponse(responseType());
1041
1042 if (useDecoder && !m_decoder)
1043 m_decoder = createDecoder();
1044
1045 if (!len)
1046 return;
1047
1048 if (len == -1)
1049 len = strlen(reinterpret_cast<const char*>(data));
1050
1051 if (useDecoder)
1052 m_responseBuilder.append(m_decoder->decode(data, len));
1053 else {
1054 // Buffer binary data.
1055 if (!m_binaryResponseBuilder)
1056 m_binaryResponseBuilder = SharedBuffer::create();
1057 m_binaryResponseBuilder->append(data, len);
1058 }
1059
1060 if (!m_error) {
1061 m_receivedLength += len;
1062
1063 if (readyState() != LOADING)
1064 changeState(LOADING);
1065 else {
1066 // Firefox calls readyStateChanged every time it receives data, 4449442
1067 callReadyStateChangeListener();
1068 }
1069
1070 if (m_async) {
1071 long long expectedLength = m_response.expectedContentLength();
1072 bool lengthComputable = expectedLength > 0 && m_receivedLength <= expectedLength;
1073 unsigned long long total = lengthComputable ? expectedLength : 0;
1074 m_progressEventThrottle.dispatchThrottledProgressEvent(lengthComputable, m_receivedLength, total);
1075 }
1076 }
1077}
