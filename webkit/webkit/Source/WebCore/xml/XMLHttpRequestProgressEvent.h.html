27#pragma once
28
29#include "ProgressEvent.h"
30
31namespace WebCore {
32
33class XMLHttpRequestProgressEvent final : public ProgressEvent {
34 WTF_MAKE_ISO_ALLOCATED(XMLHttpRequestProgressEvent);
35public:
36 static Ref<XMLHttpRequestProgressEvent> create(const AtomString& type, bool lengthComputable = false, unsigned long long loaded = 0, unsigned long long total = 0)
37 {
38 return adoptRef(*new XMLHttpRequestProgressEvent(type, lengthComputable, loaded, total));
39 }
40 // Those 2 synonyms are included for compatibility with Firefox.
41 unsigned long long position() const { return loaded(); }
42 unsigned long long totalSize() const { return total(); }
43
44 EventInterface eventInterface() const override { return XMLHttpRequestProgressEventInterfaceType; }
45
46private:
47 XMLHttpRequestProgressEvent(const AtomString& type, bool lengthComputable, unsigned long long loaded, unsigned long long total)
48 : ProgressEvent(type, lengthComputable, loaded, total)
49 {
50 }
51};
52
53} // namespace WebCore
