42Expression::Expression()
43 : m_isContextNodeSensitive(false)
44 , m_isContextPositionSensitive(false)
45 , m_isContextSizeSensitive(false)
46{
47}
48
49void Expression::setSubexpressions(Vector<std::unique_ptr<Expression>> subexpressions)
50{
51 ASSERT(m_subexpressions.isEmpty());
52 m_subexpressions = WTFMove(subexpressions);
53 for (auto& subexpression : m_subexpressions) {
54 m_isContextNodeSensitive |= subexpression->m_isContextNodeSensitive;
55 m_isContextPositionSensitive |= subexpression->m_isContextPositionSensitive;
56 m_isContextSizeSensitive |= subexpression->m_isContextSizeSensitive;
57 }
58}
