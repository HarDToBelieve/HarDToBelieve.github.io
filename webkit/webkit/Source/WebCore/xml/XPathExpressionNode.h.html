27#pragma once
28
29#include "XPathValue.h"

31namespace WebCore {
32namespace XPath {
33
34struct EvaluationContext {
35 RefPtr<Node> node;
36 unsigned size;
37 unsigned position;
38 HashMap<String, String> variableBindings;
39
40 bool hadTypeConversionError;
41};
42
43class Expression {
44 WTF_MAKE_NONCOPYABLE(Expression); WTF_MAKE_FAST_ALLOCATED;
45public:
46 static EvaluationContext& evaluationContext();
47
48 virtual ~Expression() = default;
49
50 virtual Value evaluate() const = 0;
51 virtual Value::Type resultType() const = 0;
52
53 bool isContextNodeSensitive() const { return m_isContextNodeSensitive; }
54 bool isContextPositionSensitive() const { return m_isContextPositionSensitive; }
55 bool isContextSizeSensitive() const { return m_isContextSizeSensitive; }
56
57protected:
58 Expression();
59
60 unsigned subexpressionCount() const { return m_subexpressions.size(); }
61 const Expression& subexpression(unsigned i) const { return *m_subexpressions[i]; }
62
63 void addSubexpression(std::unique_ptr<Expression> expression)
64 {
65 m_isContextNodeSensitive |= expression->m_isContextNodeSensitive;
66 m_isContextPositionSensitive |= expression->m_isContextPositionSensitive;
67 m_isContextSizeSensitive |= expression->m_isContextSizeSensitive;
68 m_subexpressions.append(WTFMove(expression));
69 }
70
71 void setSubexpressions(Vector<std::unique_ptr<Expression>>);
72
73 void setIsContextNodeSensitive(bool value) { m_isContextNodeSensitive = value; }
74 void setIsContextPositionSensitive(bool value) { m_isContextPositionSensitive = value; }
75 void setIsContextSizeSensitive(bool value) { m_isContextSizeSensitive = value; }
76
77private:
78 Vector<std::unique_ptr<Expression>> m_subexpressions;
79
80 // Evaluation details that can be used for optimization.
81 bool m_isContextNodeSensitive;
82 bool m_isContextPositionSensitive;
83 bool m_isContextSizeSensitive;
84};
85
86} // namespace XPath
87} // namespace WebCore
