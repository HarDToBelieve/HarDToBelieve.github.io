49struct Parser::Token {
50 int type;
51 String string;
52 Step::Axis axis;
53 NumericOp::Opcode numericOpcode;
54 EqTestOp::Opcode equalityTestOpcode;
55
56 Token(int type) : type(type) { }
57 Token(int type, const String& string) : type(type), string(string) { }
58 Token(int type, Step::Axis axis) : type(type), axis(axis) { }
59 Token(int type, NumericOp::Opcode opcode) : type(type), numericOpcode(opcode) { }
60 Token(int type, EqTestOp::Opcode opcode) : type(type), equalityTestOpcode(opcode) { }
61};

63enum XMLCat { NameStart, NameCont, NotPartOfName };
64
65static XMLCat charCat(UChar character)
66{
67 if (character == '_')
68 return NameStart;
69
70 if (character == '.' || character == '-')
71 return NameCont;
72 unsigned characterTypeMask = U_GET_GC_MASK(character);
73 if (characterTypeMask & (U_GC_LU_MASK | U_GC_LL_MASK | U_GC_LO_MASK | U_GC_LT_MASK | U_GC_NL_MASK))
74 return NameStart;
75 if (characterTypeMask & (U_GC_M_MASK | U_GC_LM_MASK | U_GC_ND_MASK))
76 return NameCont;
77 return NotPartOfName;
78}

80static MemoryCompactLookupOnlyRobinHoodHashMap<String, Step::Axis> createAxisNamesMap()
81{
82 struct AxisName {
83 ASCIILiteral name;
84 Step::Axis axis;
85 };
86 const AxisName axisNameList[] = {
87 { "ancestor"_s, Step::AncestorAxis },
88 { "ancestor-or-self"_s, Step::AncestorOrSelfAxis },
89 { "attribute"_s, Step::AttributeAxis },
90 { "child"_s, Step::ChildAxis },
91 { "descendant"_s, Step::DescendantAxis },
92 { "descendant-or-self"_s, Step::DescendantOrSelfAxis },
93 { "following"_s, Step::FollowingAxis },
94 { "following-sibling"_s, Step::FollowingSiblingAxis },
95 { "namespace"_s, Step::NamespaceAxis },
96 { "parent"_s, Step::ParentAxis },
97 { "preceding"_s, Step::PrecedingAxis },
98 { "preceding-sibling"_s, Step::PrecedingSiblingAxis },
99 { "self"_s, Step::SelfAxis }
100 };
101 MemoryCompactLookupOnlyRobinHoodHashMap<String, Step::Axis> map;
102 for (auto& axisName : axisNameList)
103 map.add(axisName.name, axisName.axis);
104 return map;
105}

107static bool parseAxisName(const String& name, Step::Axis& type)
108{
109 static const auto axisNames = makeNeverDestroyed(createAxisNamesMap());
110 auto it = axisNames.get().find(name);
111 if (it == axisNames.get().end())
112 return false;
113 type = it->value;
114 return true;
115}

117// Returns whether the current token can possibly be a binary operator, given
118// the previous token. Necessary to disambiguate some of the operators
119// (* (multiply), div, and, or, mod) in the [32] Operator rule
120// (check http://www.w3.org/TR/xpath#exprlex).
121bool Parser::isBinaryOperatorContext() const
122{
123 switch (m_lastTokenType) {
124 case 0:
125 case '@': case AXISNAME: case '(': case '[': case ',':
126 case AND: case OR: case MULOP:
127 case '/': case SLASHSLASH: case '|': case PLUS: case MINUS:
128 case EQOP: case RELOP:
129 return false;
130 default:
131 return true;
132 }
133}

135void Parser::skipWS()

159// Returns next char if it's there and interesting, 0 otherwise.
160char Parser::peekAheadHelper()
161{
162 if (m_nextPos + 1 >= m_data.length())
163 return 0;
164 UChar next = m_data[m_nextPos + 1];
165 if (next >= 0xff)
166 return 0;
167 return next;
168}
169
170char Parser::peekCurHelper()
171{
172 if (m_nextPos >= m_data.length())
173 return 0;
174 UChar next = m_data[m_nextPos];
175 if (next >= 0xff)
176 return 0;
177 return next;
178}

180Parser::Token Parser::lexString()
181{
182 UChar delimiter = m_data[m_nextPos];
183 int startPos = m_nextPos + 1;
184
185 for (m_nextPos = startPos; m_nextPos < m_data.length(); ++m_nextPos) {
186 if (m_data[m_nextPos] == delimiter) {
187 String value = m_data.substring(startPos, m_nextPos - startPos);
188 if (value.isNull())
189 value = emptyString();
190 ++m_nextPos; // Consume the char.
191 return Token(LITERAL, value);
192 }
193 }
194
195 // Ouch, went off the end -- report error.
196 return Token(XPATH_ERROR);
197}

199Parser::Token Parser::lexNumber()
200{
201 int startPos = m_nextPos;
202 bool seenDot = false;
203
204 // Go until end or a non-digits character.
205 for (; m_nextPos < m_data.length(); ++m_nextPos) {
206 UChar aChar = m_data[m_nextPos];
207 if (aChar >= 0xff) break;
208
209 if (!isASCIIDigit(aChar)) {
210 if (aChar == '.' && !seenDot)
211 seenDot = true;
212 else
213 break;
214 }
215 }
216
217 return Token(NUMBER, m_data.substring(startPos, m_nextPos - startPos));
218}

220bool Parser::lexNCName(String& name)
221{
222 int startPos = m_nextPos;
223 if (m_nextPos >= m_data.length())
224 return false;
225
226 if (charCat(m_data[m_nextPos]) != NameStart)
227 return false;
228
229 // Keep going until we get a character that's not good for names.
230 for (; m_nextPos < m_data.length(); ++m_nextPos)
231 if (charCat(m_data[m_nextPos]) == NotPartOfName)
232 break;
233
234 name = m_data.substring(startPos, m_nextPos - startPos);
235 return true;
236}
237
238bool Parser::lexQName(String& name)
239{
240 String n1;
241 if (!lexNCName(n1))
242 return false;
243
244 skipWS();
245
246 // If the next character is :, what we just got it the prefix, if not,
247 // it's the whole thing.
248 if (peekAheadHelper() != ':') {
249 name = n1;
250 return true;
251 }
252
253 String n2;
254 if (!lexNCName(n2))
255 return false;
256
257 name = n1 + ":" + n2;
258 return true;
259}

261inline Parser::Token Parser::nextTokenInternal()
262{
263 skipWS();
264
265 if (m_nextPos >= m_data.length())
266 return Token(0);
267
268 char code = peekCurHelper();
269 switch (code) {
270 case '(': case ')': case '[': case ']':
271 case '@': case ',': case '|':
272 return makeTokenAndAdvance(code);
273 case '\'':
274 case '\"':
275 return lexString();
276 case '0': case '1': case '2': case '3': case '4':
277 case '5': case '6': case '7': case '8': case '9':
278 return lexNumber();
279 case '.': {
280 char next = peekAheadHelper();
281 if (next == '.')
282 return makeTokenAndAdvance(DOTDOT, 2);
283 if (isASCIIDigit(next))
284 return lexNumber();
285 return makeTokenAndAdvance('.');
286 }
287 case '/':
288 if (peekAheadHelper() == '/')
289 return makeTokenAndAdvance(SLASHSLASH, 2);
290 return makeTokenAndAdvance('/');
291 case '+':
292 return makeTokenAndAdvance(PLUS);
293 case '-':
294 return makeTokenAndAdvance(MINUS);
295 case '=':
296 return makeTokenAndAdvance(EQOP, EqTestOp::OP_EQ);
297 case '!':
298 if (peekAheadHelper() == '=')
299 return makeTokenAndAdvance(EQOP, EqTestOp::OP_NE, 2);
300 return Token(XPATH_ERROR);
301 case '<':
302 if (peekAheadHelper() == '=')
303 return makeTokenAndAdvance(RELOP, EqTestOp::OP_LE, 2);
304 return makeTokenAndAdvance(RELOP, EqTestOp::OP_LT);
305 case '>':
306 if (peekAheadHelper() == '=')
307 return makeTokenAndAdvance(RELOP, EqTestOp::OP_GE, 2);
308 return makeTokenAndAdvance(RELOP, EqTestOp::OP_GT);
309 case '*':
310 if (isBinaryOperatorContext())
311 return makeTokenAndAdvance(MULOP, NumericOp::OP_Mul);
312 ++m_nextPos;
313 return Token(NAMETEST, "*");
314 case '$': { // $ QName
315 m_nextPos++;
316 String name;
317 if (!lexQName(name))
318 return Token(XPATH_ERROR);
319 return Token(VARIABLEREFERENCE, name);
320 }
321 }
322
323 String name;
324 if (!lexNCName(name))
325 return Token(XPATH_ERROR);
326
327 skipWS();
328 // If we're in an operator context, check for any operator names
329 if (isBinaryOperatorContext()) {
330 if (name == "and") //### hash?
331 return Token(AND);
332 if (name == "or")
333 return Token(OR);
334 if (name == "mod")
335 return Token(MULOP, NumericOp::OP_Mod);
336 if (name == "div")
337 return Token(MULOP, NumericOp::OP_Div);
338 }
339
340 // See whether we are at a :
341 if (peekCurHelper() == ':') {
342 m_nextPos++;
343 // Any chance it's an axis name?
344 if (peekCurHelper() == ':') {
345 m_nextPos++;
346
347 //It might be an axis name.
348 Step::Axis axis;
349 if (parseAxisName(name, axis))
350 return Token(AXISNAME, axis);
351 // Ugh, :: is only valid in axis names -> error
352 return Token(XPATH_ERROR);
353 }
354
355 // Seems like this is a fully qualified qname, or perhaps the * modified one from NameTest
356 skipWS();
357 if (peekCurHelper() == '*') {
358 m_nextPos++;
359 return Token(NAMETEST, name + ":*");
360 }
361
362 // Make a full qname.
363 String n2;
364 if (!lexNCName(n2))
365 return Token(XPATH_ERROR);
366
367 name = name + ":" + n2;
368 }
369
370 skipWS();
371
372 if (peekCurHelper() == '(') {
373 // note: we don't swallow the '(' here!
374
375 // Either node type oor function name.
376
377 if (name == "processing-instruction")
378 return Token(PI);
379 if (name == "node")
380 return Token(NODE);
381 if (name == "text")
382 return Token(TEXT_);
383 if (name == "comment")
384 return Token(COMMENT);
385
386 return Token(FUNCTIONNAME, name);
387 }
388
389 // At this point, it must be NAMETEST.
390 return Token(NAMETEST, name);
391}

393inline Parser::Token Parser::nextToken()
394{
395 Token token = nextTokenInternal();
396 m_lastTokenType = token.type;
397 return token;
398}

406int Parser::lex(YYSTYPE& yylval)
407{
408 Token token = nextToken();
409
410 switch (token.type) {
411 case AXISNAME:
412 yylval.axis = token.axis;
413 break;
414 case MULOP:
415 yylval.numericOpcode = token.numericOpcode;
416 break;
417 case RELOP:
418 case EQOP:
419 yylval.equalityTestOpcode = token.equalityTestOpcode;
420 break;
421 case NODETYPE:
422 case FUNCTIONNAME:
423 case LITERAL:
424 case VARIABLEREFERENCE:
425 case NUMBER:
426 case NAMETEST:
427 yylval.string = token.string.releaseImpl().leakRef();
428 break;
429 }
430
431 return token.type;
432}

434bool Parser::expandQualifiedName(const String& qualifiedName, String& localName, String& namespaceURI)
435{
436 size_t colon = qualifiedName.find(':');
437 if (colon != notFound) {
438 if (!m_resolver) {
439 m_sawNamespaceError = true;
440 return false;
441 }
442 namespaceURI = m_resolver->lookupNamespaceURI(qualifiedName.left(colon));
443 if (namespaceURI.isNull()) {
444 m_sawNamespaceError = true;
445 return false;
446 }
447 localName = qualifiedName.substring(colon + 1);
448 } else
449 localName = qualifiedName;
450 return true;
451}

453ExceptionOr<std::unique_ptr<Expression>> Parser::parseStatement(const String& statement, RefPtr<XPathNSResolver>&& resolver)
454{
455 Parser parser { statement, WTFMove(resolver) };
456
457 int parseError = xpathyyparse(parser);
458
459 if (parser.m_sawNamespaceError)
460 return Exception { NamespaceError };
461
462 if (parseError)
463 return Exception { SyntaxError };
464
465 return WTFMove(parser.m_result);
466}
