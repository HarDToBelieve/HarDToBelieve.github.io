27#pragma once
28
29#include "XPathNodeSet.h"
30
31namespace WebCore {
32 namespace XPath {
33
34 class Value {
35 public:
36 enum Type { NodeSetValue, BooleanValue, NumberValue, StringValue };
37
38 Value(bool value) : m_type(BooleanValue), m_bool(value) { }
39 Value(unsigned value) : m_type(NumberValue), m_number(value) { }
40 Value(double value) : m_type(NumberValue), m_number(value) { }
41
42 Value(const String& value) : m_type(StringValue), m_data(Data::create(value)) { }
43 Value(const char* value) : m_type(StringValue), m_data(Data::create(value)) { }
44
45 explicit Value(NodeSet&& value)
46 : m_type(NodeSetValue), m_data(Data::create(WTFMove(value)))
47 { }
48 explicit Value(Node* value)
49 : m_type(NodeSetValue), m_data(Data::create(value))
50 { }
51 explicit Value(RefPtr<Node>&& value)
52 : m_type(NodeSetValue), m_data(Data::create(WTFMove(value)))
53 { }
54
55 Type type() const { return m_type; }
56
57 bool isNodeSet() const { return m_type == NodeSetValue; }
58 bool isBoolean() const { return m_type == BooleanValue; }
59 bool isNumber() const { return m_type == NumberValue; }
60 bool isString() const { return m_type == StringValue; }
61
62 const NodeSet& toNodeSet() const;
63 bool toBoolean() const;
64 double toNumber() const;
65 String toString() const;
66
67 // Note that the NodeSet is shared with other Values that this one was copied from or that are copies of this one.
68 NodeSet& modifiableNodeSet();
69
70 private:
71 // This constructor creates ambiguity so that we don't accidentally call the boolean overload for pointer types.
72 Value(void*) = delete;
73
74 struct Data : public RefCounted<Data> {
75 static Ref<Data> create() { return adoptRef(*new Data); }
76 static Ref<Data> create(const String& string) { return adoptRef(*new Data(string)); }
77 static Ref<Data> create(NodeSet&& nodeSet) { return adoptRef(*new Data(WTFMove(nodeSet))); }
78 static Ref<Data> create(RefPtr<Node>&& node) { return adoptRef(*new Data(WTFMove(node))); }
79
80 String string;
81 NodeSet nodeSet;
82
83 private:
84 Data() { }
85 explicit Data(const String& string)
86 : string(string)
87 { }
88 explicit Data(NodeSet&& nodeSet)
89 : nodeSet(WTFMove(nodeSet))
90 { }
91 explicit Data(RefPtr<Node>&& node)
92 : nodeSet(WTFMove(node))
93 { }
94 };
95
96 Type m_type;
97 bool m_bool;
98 double m_number;
99 RefPtr<Data> m_data;
100 };
101
102 } // namespace XPath
103} // namespace WebCore
