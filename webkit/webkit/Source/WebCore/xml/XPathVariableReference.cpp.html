34VariableReference::VariableReference(const String& name)
35 : m_name(name)
36{
37}
38
39Value VariableReference::evaluate() const
40{
41 HashMap<String, String>& bindings = evaluationContext().variableBindings;
42 if (!bindings.contains(m_name))
43 // FIXME: Is this the right thing to do if an unknown variable is referenced?
44 return "";
45 return bindings.get(m_name);
46}
