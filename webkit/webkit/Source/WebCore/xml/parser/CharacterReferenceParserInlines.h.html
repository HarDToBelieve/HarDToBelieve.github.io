29#include <wtf/text/StringBuilder.h>

31namespace WebCore {
32
33inline void unconsumeCharacters(SegmentedString& source, StringBuilder& consumedCharacters)
34{
35 source.pushBack(consumedCharacters.toString());
36}
37
38template <typename ParserFunctions>
39bool consumeCharacterReference(SegmentedString& source, StringBuilder& decodedCharacter, bool& notEnoughCharacters, UChar additionalAllowedCharacter)
40{
41 ASSERT(!additionalAllowedCharacter || additionalAllowedCharacter == '"' || additionalAllowedCharacter == '\'' || additionalAllowedCharacter == '>');
42 ASSERT(!notEnoughCharacters);
43 ASSERT(decodedCharacter.isEmpty());
44
45 enum {
46 Initial,
47 Number,
48 MaybeHexLowerCaseX,
49 MaybeHexUpperCaseX,
50 Hex,
51 Decimal,
52 Named
53 } state = Initial;
54 Checked<UChar32, RecordOverflow> result = 0;
55 StringBuilder consumedCharacters;
56
57 while (!source.isEmpty()) {
58 UChar character = source.currentCharacter();
59 switch (state) {
60 case Initial:
61 if (character == '\x09' || character == '\x0A' || character == '\x0C' || character == ' ' || character == '<' || character == '&')
62 return false;
63 if (additionalAllowedCharacter && character == additionalAllowedCharacter)
64 return false;
65 if (character == '#') {
66 state = Number;
67 break;
68 }
69 if (isASCIIAlpha(character)) {
70 state = Named;
71 goto Named;
72 }
73 return false;
74 case Number:
75 if (character == 'x') {
76 state = MaybeHexLowerCaseX;
77 break;
78 }
79 if (character == 'X') {
80 state = MaybeHexUpperCaseX;
81 break;
82 }
83 if (isASCIIDigit(character)) {
84 state = Decimal;
85 goto Decimal;
86 }
87 source.pushBack("#"_s);
88 return false;
89 case MaybeHexLowerCaseX:
90 if (isASCIIHexDigit(character)) {
91 state = Hex;
92 goto Hex;
93 }
94 source.pushBack("#x"_s);
95 return false;
96 case MaybeHexUpperCaseX:
97 if (isASCIIHexDigit(character)) {
98 state = Hex;
99 goto Hex;
100 }
101 source.pushBack("#X"_s);
102 return false;
103 case Hex:
104 Hex:
105 if (isASCIIHexDigit(character)) {
106 result *= 16;
107 result += static_cast<UChar32>(toASCIIHexValue(character));
108 break;
109 }
110 if (character == ';') {
111 source.advancePastNonNewline();
112 decodedCharacter.appendCharacter(ParserFunctions::legalEntityFor(result.hasOverflowed() ? 0 : result.value()));
113 return true;
114 }
115 if (ParserFunctions::acceptMalformed()) {
116 decodedCharacter.appendCharacter(ParserFunctions::legalEntityFor(result.hasOverflowed() ? 0 : result.value()));
117 return true;
118 }
119 unconsumeCharacters(source, consumedCharacters);
120 return false;
121 case Decimal:
122 Decimal:
123 if (isASCIIDigit(character)) {
124 result *= 10;
125 result += (character - '0');
126 break;
127 }
128 if (character == ';') {
129 source.advancePastNonNewline();
130 decodedCharacter.appendCharacter(ParserFunctions::legalEntityFor(result.hasOverflowed() ? 0 : result.value()));
131 return true;
132 }
133 if (ParserFunctions::acceptMalformed()) {
134 decodedCharacter.appendCharacter(ParserFunctions::legalEntityFor(result.hasOverflowed() ? 0 : result.value()));
135 return true;
136 }
137 unconsumeCharacters(source, consumedCharacters);
138 return false;
139 case Named:
140 Named:
141 return ParserFunctions::consumeNamedEntity(source, decodedCharacter, notEnoughCharacters, additionalAllowedCharacter, character);
142 }
143 consumedCharacters.append(character);
144 source.advancePastNonNewline();
145 }
146 ASSERT(source.isEmpty());
147 notEnoughCharacters = true;
148 unconsumeCharacters(source, consumedCharacters);
149 return false;
150}
