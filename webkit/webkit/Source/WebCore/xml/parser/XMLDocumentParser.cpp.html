64void XMLDocumentParser::pushCurrentNode(ContainerNode* n)
65{
66 ASSERT(n);
67 ASSERT(m_currentNode);
68 if (n != document())
69 n->ref();
70 m_currentNodeStack.append(m_currentNode);
71 m_currentNode = n;
72 if (m_currentNodeStack.size() > maxDOMTreeDepth)
73 handleError(XMLErrors::fatal, "Excessive node nesting.", textPosition());
74}
75
76void XMLDocumentParser::popCurrentNode()
77{
78 if (!m_currentNode)
79 return;
80 ASSERT(m_currentNodeStack.size());
81
82 if (m_currentNode != document())
83 m_currentNode->deref();
84
85 m_currentNode = m_currentNodeStack.last();
86 m_currentNodeStack.removeLast();
87}
88
89void XMLDocumentParser::clearCurrentNodeStack()
90{
91 if (m_currentNode && m_currentNode != document())
92 m_currentNode->deref();
93 m_currentNode = nullptr;
94 m_leafTextNode = nullptr;
95
96 if (m_currentNodeStack.size()) { // Aborted parsing.
97 for (size_t i = m_currentNodeStack.size() - 1; i != 0; --i)
98 m_currentNodeStack[i]->deref();
99 if (m_currentNodeStack[0] && m_currentNodeStack[0] != document())
100 m_currentNodeStack[0]->deref();
101 m_currentNodeStack.clear();
102 }
103}

110void XMLDocumentParser::append(RefPtr<StringImpl>&& inputSource)
111{
112 String source { WTFMove(inputSource) };
113
114 if (m_sawXSLTransform || !m_sawFirstElement)
115 m_originalSourceForTransform.append(source);
116
117 if (isStopped() || m_sawXSLTransform)
118 return;
119
120 if (m_parserPaused) {
121 m_pendingSrc.append(source);
122 return;
123 }
124
125 doWrite(source);
126
127 // After parsing, dispatch image beforeload events.
128 ImageLoader::dispatchPendingBeforeLoadEvents(nullptr);
129}

131void XMLDocumentParser::handleError(XMLErrors::ErrorType type, const char* m, TextPosition position)
132{
133 if (!m_xmlErrors)
134 m_xmlErrors = makeUnique<XMLErrors>(*document());
135 m_xmlErrors->handleError(type, m, position);
136 if (type != XMLErrors::warning)
137 m_sawError = true;
138 if (type == XMLErrors::fatal)
139 stopParsing();
140}

142void XMLDocumentParser::createLeafTextNode()
143{
144 if (m_leafTextNode)
145 return;
146
147 ASSERT(m_bufferedText.size() == 0);
148 ASSERT(!m_leafTextNode);
149 m_leafTextNode = Text::create(m_currentNode->document(), "");
150 m_currentNode->parserAppendChild(*m_leafTextNode);
151}
152
153bool XMLDocumentParser::updateLeafTextNode()
154{
155 if (isStopped())
156 return false;
157
158 if (!m_leafTextNode)
159 return true;
160
161 // This operation might fire mutation event, see below.
162 m_leafTextNode->appendData(String::fromUTF8(m_bufferedText.data(), m_bufferedText.size()));
163 m_bufferedText = { };
164
165 m_leafTextNode = nullptr;
166
167 // Hence, we need to check again whether the parser is stopped, since mutation
168 // event handlers executed by appendData might have detached this parser.
169 return !isStopped();
170}

178void XMLDocumentParser::end()
179{
180 // XMLDocumentParserLibxml2 will do bad things to the document if doEnd() is called.
181 // I don't believe XMLDocumentParserQt needs doEnd called in the fragment case.
182 ASSERT(!m_parsingFragment);
183
184 doEnd();
185
186 // doEnd() call above can detach the parser and null out its document.
187 // In that case, we just bail out.
188 if (isDetached())
189 return;
190
191 // doEnd() could process a script tag, thus pausing parsing.
192 if (m_parserPaused)
193 return;
194
195 if (m_sawError && !isStopped()) {
196 insertErrorMessageBlock();
197 if (isDetached()) // Inserting an error message may have ran arbitrary scripts.
198 return;
199 } else {
200 updateLeafTextNode();
201 document()->styleScope().didChangeStyleSheetEnvironment();
202 }
203
204 if (isParsing())
205 prepareToStopParsing();
206 document()->setReadyState(Document::Interactive);
207 clearCurrentNodeStack();
208 document()->finishedParsing();
209}

211void XMLDocumentParser::finish()
212{
213 // FIXME: We should ASSERT(!m_parserStopped) here, since it does not
214 // makes sense to call any methods on DocumentParser once it's been stopped.
215 // However, FrameLoader::stop calls DocumentParser::finish unconditionally.
216
217 Ref<XMLDocumentParser> protectedThis(*this);
218
219 if (m_parserPaused)
220 m_finishCalled = true;
221 else
222 end();
223}

231void XMLDocumentParser::notifyFinished(PendingScript& pendingScript)
232{
233 ASSERT(&pendingScript == m_pendingScript.get());
234
235 // JavaScript can detach this parser, make sure it's kept alive even if detached.
236 Ref<XMLDocumentParser> protectedThis(*this);
237
238 m_pendingScript = nullptr;
239 pendingScript.clearClient();
240
241 pendingScript.element().executePendingScript(pendingScript);
242
243 if (!isDetached() && !m_requestingScript)
244 resumeParsing();
245}

247void XMLDocumentParser::pauseParsing()
248{
249 ASSERT(!m_parserPaused);
250
251 if (m_parsingFragment)
252 return;
253
254 m_parserPaused = true;
255}

257struct XMLParsingNamespaces {
258 AtomString defaultNamespace;
259 HashMap<AtomString, AtomString> prefixNamespaces;
260};
261
262static XMLParsingNamespaces findXMLParsingNamespaces(Element* contextElement)
263{
264 if (!contextElement)
265 return { };
266
267 XMLParsingNamespaces result;
268
269 bool stopLookingForDefaultNamespace = false;
270
271 for (auto& element : lineageOfType<Element>(*contextElement)) {
272 if (is<SVGForeignObjectElement>(element))
273 stopLookingForDefaultNamespace = true;
274 else if (!stopLookingForDefaultNamespace)
275 result.defaultNamespace = element.namespaceURI();
276
277 if (!element.hasAttributes())
278 continue;
279
280 for (auto& attribute : element.attributesIterator()) {
281 if (attribute.prefix() == xmlnsAtom())
282 result.prefixNamespaces.set(attribute.localName(), attribute.value());
283 }
284 }
285
286 return result;
287}

289bool XMLDocumentParser::parseDocumentFragment(const String& chunk, DocumentFragment& fragment, Element* contextElement, ParserContentPolicy parserContentPolicy)
290{
291 if (!chunk.length())
292 return true;
293
294 // FIXME: We need to implement the HTML5 XML Fragment parsing algorithm:
295 // http://www.whatwg.org/specs/web-apps/current-work/multipage/the-xhtml-syntax.html#xml-fragment-parsing-algorithm
296 // For now we have a hack for script/style innerHTML support:
297 if (contextElement && (contextElement->hasLocalName(HTMLNames::scriptTag->localName()) || contextElement->hasLocalName(HTMLNames::styleTag->localName()))) {
298 fragment.parserAppendChild(fragment.document().createTextNode(chunk));
299 return true;
300 }
301
302 auto namespaces = findXMLParsingNamespaces(contextElement);
303 auto parser = XMLDocumentParser::create(fragment, WTFMove(namespaces.prefixNamespaces), namespaces.defaultNamespace, parserContentPolicy);
304 bool wellFormed = parser->appendFragmentSource(chunk);
305 // Do not call finish(). The finish() and doEnd() implementations touch the main document and loader and can cause crashes in the fragment case.
306 parser->detach(); // Allows ~DocumentParser to assert it was detached before destruction.
307 return wellFormed; // appendFragmentSource()'s wellFormed is more permissive than Document::wellFormed().
308}
