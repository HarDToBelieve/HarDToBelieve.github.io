74static inline bool shouldRenderInXMLTreeViewerMode(Document& document)
75{
76 if (document.sawElementsInKnownNamespaces())
77 return false;
78
79 if (document.transformSourceDocument())
80 return false;
81
82 auto* frame = document.frame();
83 if (!frame)
84 return false;
85
86 if (!frame->settings().developerExtrasEnabled())
87 return false;
88
89 if (frame->tree().parent())
90 return false; // This document is not in a top frame
91
92 return true;
93}

97class PendingCallbacks {
98 WTF_MAKE_FAST_ALLOCATED;
99public:
100 void appendStartElementNSCallback(const xmlChar* xmlLocalName, const xmlChar* xmlPrefix, const xmlChar* xmlURI, int numNamespaces, const xmlChar** namespaces, int numAttributes, int numDefaulted, const xmlChar** attributes)
101 {
102 auto callback = makeUnique<PendingStartElementNSCallback>();
103
104 callback->xmlLocalName = xmlStrdup(xmlLocalName);
105 callback->xmlPrefix = xmlStrdup(xmlPrefix);
106 callback->xmlURI = xmlStrdup(xmlURI);
107 callback->numNamespaces = numNamespaces;
108 callback->namespaces = static_cast<xmlChar**>(xmlMalloc(sizeof(xmlChar*) * numNamespaces * 2));
109 for (int i = 0; i < numNamespaces * 2 ; i++)
110 callback->namespaces[i] = xmlStrdup(namespaces[i]);
111 callback->numAttributes = numAttributes;
112 callback->numDefaulted = numDefaulted;
113 callback->attributes = static_cast<xmlChar**>(xmlMalloc(sizeof(xmlChar*) * numAttributes * 5));
114 for (int i = 0; i < numAttributes; i++) {
115 // Each attribute has 5 elements in the array:
116 // name, prefix, uri, value and an end pointer.
117
118 for (int j = 0; j < 3; j++)
119 callback->attributes[i * 5 + j] = xmlStrdup(attributes[i * 5 + j]);
120
121 int len = attributes[i * 5 + 4] - attributes[i * 5 + 3];
122
123 callback->attributes[i * 5 + 3] = xmlStrndup(attributes[i * 5 + 3], len);
124 callback->attributes[i * 5 + 4] = callback->attributes[i * 5 + 3] + len;
125 }
126
127 m_callbacks.append(WTFMove(callback));
128 }
129
130 void appendEndElementNSCallback()
131 {
132 m_callbacks.append(makeUnique<PendingEndElementNSCallback>());
133 }

359class OffsetBuffer {
360 WTF_MAKE_FAST_ALLOCATED;
361public:
362 OffsetBuffer(Vector<uint8_t>&& buffer)
363 : m_buffer(WTFMove(buffer))
364 {
365 }
366
367 int readOutBytes(char* outputBuffer, unsigned askedToRead)
368 {
369 unsigned bytesLeft = m_buffer.size() - m_currentOffset;
370 unsigned lenToCopy = std::min(askedToRead, bytesLeft);
371 if (lenToCopy) {
372 memcpy(outputBuffer, m_buffer.data() + m_currentOffset, lenToCopy);
373 m_currentOffset += lenToCopy;
374 }
375 return lenToCopy;
376 }
377
378private:
379 Vector<uint8_t> m_buffer;
380 unsigned m_currentOffset { 0 };
381};

383static bool externalEntityMimeTypeAllowed(const ResourceResponse& response)
384{
385 String contentType = response.httpHeaderField(HTTPHeaderName::ContentType);
386 String mimeType = extractMIMETypeFromMediaType(contentType);
387 if (mimeType.isEmpty()) {
388 // Same logic as XMLHttpRequest::responseMIMEType(). Keep them in sync.
389 if (response.isInHTTPFamily())
390 mimeType = contentType;
391 else
392 mimeType = response.mimeType();
393 }
394 return MIMETypeRegistry::isXMLMIMEType(mimeType) || MIMETypeRegistry::isXMLEntityMIMEType(mimeType);
395}

404static void switchToUTF16(xmlParserCtxtPtr ctxt)
405{
406 // Hack around libxml2's lack of encoding overide support by manually
407 // resetting the encoding to UTF-16 before every chunk. Otherwise libxml
408 // will detect <?xml version="1.0" encoding="<encoding name>"?> blocks
409 // and switch encodings, causing the parse to fail.
410
411 // FIXME: Can we just use XML_PARSE_IGNORE_ENC now?
412
413 const unsigned char BOMHighByte = *reinterpret_cast<const unsigned char*>(&byteOrderMark);
414 xmlSwitchEncoding(ctxt, BOMHighByte == 0xFF ? XML_CHAR_ENCODING_UTF16LE : XML_CHAR_ENCODING_UTF16BE);
415}

417static bool shouldAllowExternalLoad(const URL& url)
418{
419 String urlString = url.string();
420
421 // On non-Windows platforms libxml asks for this URL, the "XML_XML_DEFAULT_CATALOG", on initialization.
422 if (urlString == "file:///etc/xml/catalog")
423 return false;
424
425 // On Windows, libxml computes a URL relative to where its DLL resides.
426 if (startsWithLettersIgnoringASCIICase(urlString, "file:///") && urlString.endsWithIgnoringASCIICase("/etc/catalog"))
427 return false;
428
429 // The most common DTD. There isn't much point in hammering www.w3c.org by requesting this for every XHTML document.
430 if (startsWithLettersIgnoringASCIICase(urlString, "http://www.w3.org/tr/xhtml"))
431 return false;
432
433 // Similarly, there isn't much point in requesting the SVG DTD.
434 if (startsWithLettersIgnoringASCIICase(urlString, "http://www.w3.org/graphics/svg"))
435 return false;
436
437 // The libxml doesn't give us a lot of context for deciding whether to
438 // allow this request. In the worst case, this load could be for an
439 // external entity and the resulting document could simply read the
440 // retrieved content. If we had more context, we could potentially allow
441 // the parser to load a DTD. As things stand, we take the conservative
442 // route and allow same-origin requests only.
443 if (!XMLDocumentParserScope::currentCachedResourceLoader->document()->securityOrigin().canRequest(url)) {
444 XMLDocumentParserScope::currentCachedResourceLoader->printAccessDeniedMessage(url);
445 return false;
446 }
447
448 return true;
449}

451static void* openFunc(const char* uri)
452{
453 ASSERT(XMLDocumentParserScope::currentCachedResourceLoader);
454 ASSERT(libxmlLoaderThread == &Thread::current());
455
456 CachedResourceLoader& cachedResourceLoader = *XMLDocumentParserScope::currentCachedResourceLoader;
457 Document* document = cachedResourceLoader.document();
458 // Same logic as HTMLBaseElement::href(). Keep them in sync.
459 auto* encoding = (document && document->decoder()) ? document->decoder()->encodingForURLParsing() : nullptr;
460 URL url(document ? document->url() : URL(), stripLeadingAndTrailingHTMLSpaces(uri), encoding);
461
462 if (!shouldAllowExternalLoad(url))
463 return &globalDescriptor;
464
465 ResourceResponse response;
466 RefPtr<SharedBuffer> data;
467
468 {
469 ResourceError error;
470 XMLDocumentParserScope scope(nullptr);
471 // FIXME: We should restore the original global error handler as well.
472
473 if (cachedResourceLoader.frame()) {
474 FetchOptions options;
475 options.mode = FetchOptions::Mode::SameOrigin;
476 options.credentials = FetchOptions::Credentials::Include;
477 cachedResourceLoader.frame()->loader().loadResourceSynchronously(url, ClientCredentialPolicy::MayAskClientForCredentials, options, { }, error, response, data);
478
479 if (response.url().isEmpty()) {
480 if (Page* page = document ? document->page() : nullptr)
481 page->console().addMessage(MessageSource::Security, MessageLevel::Error, makeString("Did not parse external entity resource at '", url.stringCenterEllipsizedToLength(), "' because cross-origin loads are not allowed."));
482 return &globalDescriptor;
483 }
484 if (!externalEntityMimeTypeAllowed(response)) {
485 if (Page* page = document ? document->page() : nullptr)
486 page->console().addMessage(MessageSource::Security, MessageLevel::Error, makeString("Did not parse external entity resource at '", url.stringCenterEllipsizedToLength(), "' because only XML MIME types are allowed."));
487 return &globalDescriptor;
488 }
489 }
490 }
491
492 if (!data)
493 return &globalDescriptor;
494
495 return new OffsetBuffer({ data->data(), data->size() });
496}

498static int readFunc(void* context, char* buffer, int len)
499{
500 // Do 0-byte reads in case of a null descriptor
501 if (context == &globalDescriptor)
502 return 0;
503
504 OffsetBuffer* data = static_cast<OffsetBuffer*>(context);
505 return data->readOutBytes(buffer, len);
506}

530static void initializeXMLParser()
531{
532 static std::once_flag flag;
533 std::call_once(flag, [&] {
534 xmlInitParser();
535 xmlRegisterInputCallbacks(matchFunc, openFunc, readFunc, closeFunc);
536 xmlRegisterOutputCallbacks(matchFunc, openFunc, writeFunc, closeFunc);
537 libxmlLoaderThread = &Thread::current();
538 });
539}

541Ref<XMLParserContext> XMLParserContext::createStringParser(xmlSAXHandlerPtr handlers, void* userData)
542{
543 initializeXMLParser();
544
545 xmlParserCtxtPtr parser = xmlCreatePushParserCtxt(handlers, 0, 0, 0, 0);
546 parser->_private = userData;
547
548 // Substitute entities.
549 xmlCtxtUseOptions(parser, XML_PARSE_NOENT | XML_PARSE_HUGE);
550
551 switchToUTF16(parser);
552
553 return adoptRef(*new XMLParserContext(parser));
554}

557// Chunk should be encoded in UTF-8
558RefPtr<XMLParserContext> XMLParserContext::createMemoryParser(xmlSAXHandlerPtr handlers, void* userData, const CString& chunk)
559{
560 initializeXMLParser();
561
562 // appendFragmentSource() checks that the length doesn't overflow an int.
563 xmlParserCtxtPtr parser = xmlCreateMemoryParserCtxt(chunk.data(), chunk.length());
564
565 if (!parser)
566 return 0;
567
568 memcpy(parser->sax, handlers, sizeof(xmlSAXHandler));
569
570 // Substitute entities.
571 // FIXME: Why is XML_PARSE_NODICT needed? This is different from what createStringParser does.
572 xmlCtxtUseOptions(parser, XML_PARSE_NODICT | XML_PARSE_NOENT | XML_PARSE_HUGE);
573
574 // Internal initialization
575 parser->sax2 = 1;
576 parser->instate = XML_PARSER_CONTENT; // We are parsing a CONTENT
577 parser->depth = 0;
578 parser->str_xml = xmlDictLookup(parser->dict, reinterpret_cast<xmlChar*>(const_cast<char*>("xml")), 3);
579 parser->str_xmlns = xmlDictLookup(parser->dict, reinterpret_cast<xmlChar*>(const_cast<char*>("xmlns")), 5);
580 parser->str_xml_ns = xmlDictLookup(parser->dict, XML_XML_NAMESPACE, 36);
581 parser->_private = userData;
582
583 return adoptRef(*new XMLParserContext(parser));
584}

632void XMLDocumentParser::doWrite(const String& parseString)
633{
634 ASSERT(!isDetached());
635 if (!m_context)
636 initializeParserContext();
637
638 // Protect the libxml context from deletion during a callback
639 RefPtr<XMLParserContext> context = m_context;
640
641 // libXML throws an error if you try to switch the encoding for an empty string.
642 if (parseString.length()) {
643 // JavaScript may cause the parser to detach during xmlParseChunk
644 // keep this alive until this function is done.
645 Ref<XMLDocumentParser> protectedThis(*this);
646
647 XMLDocumentParserScope scope(&document()->cachedResourceLoader());
648
649 // FIXME: Can we parse 8-bit strings directly as Latin-1 instead of upconverting to UTF-16?
650 switchToUTF16(context->context());
651 xmlParseChunk(context->context(), reinterpret_cast<const char*>(StringView(parseString).upconvertedCharacters().get()), sizeof(UChar) * parseString.length(), 0);
652
653 // JavaScript (which may be run under the xmlParseChunk callstack) may
654 // cause the parser to be stopped or detached.
655 if (isStopped())
656 return;
657 }
658
659 // FIXME: Why is this here? And why is it after we process the passed source?
660 if (document()->decoder() && document()->decoder()->sawError()) {
661 // If the decoder saw an error, report it as fatal (stops parsing)
662 TextPosition position(OrdinalNumber::fromOneBasedInt(context->context()->input->line), OrdinalNumber::fromOneBasedInt(context->context()->input->col));
663 handleError(XMLErrors::fatal, "Encoding error", position);
664 }
665}

693static inline bool handleNamespaceAttributes(Vector<Attribute>& prefixedAttributes, const xmlChar** libxmlNamespaces, int numNamespaces)
694{
695 xmlSAX2Namespace* namespaces = reinterpret_cast<xmlSAX2Namespace*>(libxmlNamespaces);
696 for (int i = 0; i < numNamespaces; i++) {
697 AtomString namespaceQName = xmlnsAtom();
698 AtomString namespaceURI = toAtomString(namespaces[i].uri);
699 if (namespaces[i].prefix)
700 namespaceQName = "xmlns:" + toString(namespaces[i].prefix);
701
702 auto result = Element::parseAttributeName(XMLNSNames::xmlnsNamespaceURI, namespaceQName);
703 if (result.hasException())
704 return false;
705
706 prefixedAttributes.append(Attribute(result.releaseReturnValue(), namespaceURI));
707 }
708 return true;
709}

711struct _xmlSAX2Attributes {
712 const xmlChar* localname;
713 const xmlChar* prefix;
714 const xmlChar* uri;
715 const xmlChar* value;
716 const xmlChar* end;
717};
718typedef struct _xmlSAX2Attributes xmlSAX2Attributes;
719
720static inline bool handleElementAttributes(Vector<Attribute>& prefixedAttributes, const xmlChar** libxmlAttributes, int numAttributes)
721{
722 xmlSAX2Attributes* attributes = reinterpret_cast<xmlSAX2Attributes*>(libxmlAttributes);
723 for (int i = 0; i < numAttributes; i++) {
724 int valueLength = static_cast<int>(attributes[i].end - attributes[i].value);
725 AtomString attrValue = toAtomString(attributes[i].value, valueLength);
726 String attrPrefix = toString(attributes[i].prefix);
727 AtomString attrURI = attrPrefix.isEmpty() ? nullAtom() : toAtomString(attributes[i].uri);
728 AtomString attrQName = attrPrefix.isEmpty() ? toAtomString(attributes[i].localname) : attrPrefix + ":" + toString(attributes[i].localname);
729
730 auto result = Element::parseAttributeName(attrURI, attrQName);
731 if (result.hasException())
732 return false;
733
734 prefixedAttributes.append(Attribute(result.releaseReturnValue(), attrValue));
735 }
736 return true;
737}

752void XMLDocumentParser::startElementNs(const xmlChar* xmlLocalName, const xmlChar* xmlPrefix, const xmlChar* xmlURI, int numNamespaces, const xmlChar** libxmlNamespaces, int numAttributes, int numDefaulted, const xmlChar** libxmlAttributes)
753{
754 if (isStopped())
755 return;
756
757 if (m_parserPaused) {
758 m_pendingCallbacks->appendStartElementNSCallback(xmlLocalName, xmlPrefix, xmlURI, numNamespaces, libxmlNamespaces, numAttributes, numDefaulted, libxmlAttributes);
759 return;
760 }
761
762 if (!updateLeafTextNode())
763 return;
764
765 AtomString localName = toAtomString(xmlLocalName);
766 AtomString uri = toAtomString(xmlURI);
767 AtomString prefix = toAtomString(xmlPrefix);
768
769 if (m_parsingFragment && uri.isNull()) {
770 if (!prefix.isNull())
771 uri = m_prefixToNamespaceMap.get(prefix);
772 else if (is<SVGElement>(m_currentNode) || localName == SVGNames::svgTag->localName())
773 uri = SVGNames::svgNamespaceURI;
774 else
775 uri = m_defaultNamespaceURI;
776 }
777
778 // If libxml entity parsing is broken, transfer the currentNodes' namespaceURI to the new node,
779 // if we're currently expanding elements which originate from an entity declaration.
780 if (hackAroundLibXMLEntityParsingBug() && depthTriggeringEntityExpansion() != -1 && context()->depth > depthTriggeringEntityExpansion() && uri.isNull() && prefix.isNull())
781 uri = m_currentNode->namespaceURI();
782
783 bool isFirstElement = !m_sawFirstElement;
784 m_sawFirstElement = true;
785
786 QualifiedName qName(prefix, localName, uri);
787 auto newElement = m_currentNode->document().createElement(qName, true);
788
789 Vector<Attribute> prefixedAttributes;
790 if (!handleNamespaceAttributes(prefixedAttributes, libxmlNamespaces, numNamespaces)) {
791 setAttributes(newElement.ptr(), prefixedAttributes, parserContentPolicy());
792 stopParsing();
793 return;
794 }
795
796 bool success = handleElementAttributes(prefixedAttributes, libxmlAttributes, numAttributes);
797 setAttributes(newElement.ptr(), prefixedAttributes, parserContentPolicy());
798 if (!success) {
799 stopParsing();
800 return;
801 }
802
803 newElement->beginParsingChildren();
804
805 if (isScriptElement(newElement.get()))
806 m_scriptStartPosition = textPosition();
807
808 m_currentNode->parserAppendChild(newElement);
809 if (!m_currentNode) // Synchronous DOM events may have removed the current node.
810 return;
811
812 if (is<HTMLTemplateElement>(newElement))
813 pushCurrentNode(&downcast<HTMLTemplateElement>(newElement.get()).content());
814 else
815 pushCurrentNode(newElement.ptr());
816
817 if (is<HTMLHtmlElement>(newElement))
818 downcast<HTMLHtmlElement>(newElement.get()).insertedByParser();
819
820 if (!m_parsingFragment && isFirstElement && document()->frame())
821 document()->frame()->injectUserScripts(UserScriptInjectionTime::DocumentStart);
822}
823
824void XMLDocumentParser::endElementNs()
825{
826 if (isStopped())
827 return;
828
829 if (m_parserPaused) {
830 m_pendingCallbacks->appendEndElementNSCallback();
831 return;
832 }
833
834 // JavaScript can detach the parser. Make sure this is not released
835 // before the end of this method.
836 Ref<XMLDocumentParser> protectedThis(*this);
837
838 if (!updateLeafTextNode())
839 return;
840
841 RefPtr<ContainerNode> node = m_currentNode;
842 node->finishParsingChildren();
843
844 // Once we reach the depth again where entity expansion started, stop executing the work-around.
845 if (hackAroundLibXMLEntityParsingBug() && context()->depth <= depthTriggeringEntityExpansion())
846 setDepthTriggeringEntityExpansion(-1);
847
848 if (!scriptingContentIsAllowed(parserContentPolicy()) && is<Element>(*node) && isScriptElement(downcast<Element>(*node))) {
849 popCurrentNode();
850 node->remove();
851 return;
852 }
853
854 if (!node->isElementNode() || !m_view) {
855 popCurrentNode();
856 return;
857 }
858
859 auto& element = downcast<Element>(*node);
860
861 // The element's parent may have already been removed from document.
862 // Parsing continues in this case, but scripts aren't executed.
863 if (!element.isConnected()) {
864 popCurrentNode();
865 return;
866 }
867
868 if (!isScriptElement(element)) {
869 popCurrentNode();
870 return;
871 }
872
873 // Don't load external scripts for standalone documents (for now).
874 ASSERT(!m_pendingScript);
875 m_requestingScript = true;
876
877 auto& scriptElement = downcastScriptElement(element);
878 if (scriptElement.prepareScript(m_scriptStartPosition, ScriptElement::AllowLegacyTypeInTypeAttribute)) {
879 // FIXME: Script execution should be shared between
880 // the libxml2 and Qt XMLDocumentParser implementations.
881
882 if (scriptElement.readyToBeParserExecuted())
883 scriptElement.executeClassicScript(ScriptSourceCode(scriptElement.scriptContent(), URL(document()->url()), m_scriptStartPosition, JSC::SourceProviderSourceType::Program, InlineClassicScript::create(scriptElement)));
884 else if (scriptElement.willBeParserExecuted() && scriptElement.loadableScript()) {
885 m_pendingScript = PendingScript::create(scriptElement, *scriptElement.loadableScript());
886 m_pendingScript->setClient(*this);
887
888 // m_pendingScript will be nullptr if script was already loaded and setClient() executed it.
889 if (m_pendingScript)
890 pauseParsing();
891 }
892
893 // JavaScript may have detached the parser
894 if (isDetached())
895 return;
896 }
897 m_requestingScript = false;
898 popCurrentNode();
899}

1156static size_t convertUTF16EntityToUTF8(const UChar* utf16Entity, size_t numberOfCodeUnits, char* target, size_t targetSize)
1157{
1158 const char* originalTarget = target;
1159 auto conversionResult = WTF::Unicode::convertUTF16ToUTF8(&utf16Entity, utf16Entity + numberOfCodeUnits, &target, target + targetSize);
1160 if (conversionResult != WTF::Unicode::ConversionOK)
1161 return 0;
1162
1163 // Even though we must pass the length, libxml expects the entity string to be null terminated.
1164 ASSERT(target >= originalTarget + 1);
1165 *target = '\0';
1166 return target - originalTarget;
1167}

1169static xmlEntityPtr getXHTMLEntity(const xmlChar* name)
1170{
1171 UChar utf16DecodedEntity[4];
1172 size_t numberOfCodeUnits = decodeNamedEntityToUCharArray(reinterpret_cast<const char*>(name), utf16DecodedEntity);
1173 if (!numberOfCodeUnits)
1174 return 0;
1175
1176 ASSERT(numberOfCodeUnits <= 4);
1177 size_t entityLengthInUTF8 = convertUTF16EntityToUTF8(utf16DecodedEntity, numberOfCodeUnits,
1178 reinterpret_cast<char*>(sharedXHTMLEntityResult), WTF_ARRAY_LENGTH(sharedXHTMLEntityResult));
1179 if (!entityLengthInUTF8)
1180 return 0;
1181
1182 xmlEntityPtr entity = sharedXHTMLEntity();
1183 entity->length = entityLengthInUTF8;
1184 entity->name = name;
1185 return entity;

1188static void entityDeclarationHandler(void* closure, const xmlChar* name, int type, const xmlChar* publicId, const xmlChar* systemId, xmlChar* content)
1189{
1190 // Prevent the next call to getEntityHandler() to record the entity expansion depth.
1191 // We're parsing the entity declaration, so there's no need to record anything.
1192 // We only need to record the depth, if we're actually expanding the entity, when it's referenced.
1193 if (hackAroundLibXMLEntityParsingBug())

1195 xmlSAX2EntityDecl(closure, name, type, publicId, systemId, content);

1198static xmlEntityPtr getEntityHandler(void* closure, const xmlChar* name)
1199{
1200 xmlParserCtxtPtr ctxt = static_cast<xmlParserCtxtPtr>(closure);
1201
1202 XMLDocumentParser* parser = getParser(closure);
1203 if (hackAroundLibXMLEntityParsingBug()) {
1204 if (parser->isParsingEntityDeclaration()) {
1205 // We're parsing the entity declarations (not an entity reference), no need to do anything special.
1206 parser->setIsParsingEntityDeclaration(false);
1207 ASSERT(parser->depthTriggeringEntityExpansion() == -1);
1208 } else {
1209 // The entity will be used and eventually expanded. Record the current parser depth
1210 // so the next call to startElementNs() knows that the new element originates from
1211 // an entity declaration.
1212 parser->setDepthTriggeringEntityExpansion(ctxt->depth);
1213 }
1214 }
1215
1216 xmlEntityPtr ent = xmlGetPredefinedEntity(name);
1217 if (ent) {
1218 ent->etype = XML_INTERNAL_PREDEFINED_ENTITY;
1219 return ent;
1220 }
1221
1222 ent = xmlGetDocEntity(ctxt->myDoc, name);
1223 if (!ent && parser->isXHTMLDocument()) {
1224 ent = getXHTMLEntity(name);
1225 if (ent)
1226 ent->etype = XML_INTERNAL_GENERAL_ENTITY;
1227 }
1228
1229 return ent;
1230}

1313void XMLDocumentParser::doEnd()
1314{
1315 if (!isStopped()) {
1316 if (m_context) {
1317 // Tell libxml we're done.
1318 {
1319 XMLDocumentParserScope scope(&document()->cachedResourceLoader());
1320 xmlParseChunk(context(), 0, 0, 1);
1321 }
1322
1323 m_context = nullptr;
1324 }
1325 }
1326
1327#if ENABLE(XSLT)

1329 if (xmlViewerMode) {

1346 }
1347#endif
1348}

1400void XMLDocumentParser::resumeParsing()
1401{
1402 ASSERT(!isDetached());
1403 ASSERT(m_parserPaused);
1404
1405 m_parserPaused = false;
1406
1407 // First, execute any pending callbacks
1408 while (!m_pendingCallbacks->isEmpty()) {
1409 m_pendingCallbacks->callAndRemoveFirstCallback(this);
1410
1411 // A callback paused the parser
1412 if (m_parserPaused)
1413 return;
1414 }
1415
1416 // There is normally only one string left, so toString() shouldn't copy.
1417 // In any case, the XML parser runs on the main thread and it's OK if
1418 // the passed string has more than one reference.
1419 auto rest = m_pendingSrc.toString();
1420 m_pendingSrc.clear();
1421 append(rest.impl());
1422
1423 // Finally, if finish() has been called and write() didn't result
1424 // in any further callbacks being queued, call end()
1425 if (m_finishCalled && m_pendingCallbacks->isEmpty())
1426 end();
1427}

1429bool XMLDocumentParser::appendFragmentSource(const String& chunk)
1430{
1431 ASSERT(!m_context);
1432 ASSERT(m_parsingFragment);
1433
1434 CString chunkAsUtf8 = chunk.utf8();
1435
1436 // libxml2 takes an int for a length, and therefore can't handle XML chunks larger than 2 GiB.
1437 if (chunkAsUtf8.length() > INT_MAX)
1438 return false;
1439
1440 initializeParserContext(chunkAsUtf8);
1441 xmlParseContent(context());
1442 endDocument(); // Close any open text nodes.
1443
1444 // FIXME: If this code is actually needed, it should probably move to finish()
1445 // XMLDocumentParserQt has a similar check (m_stream.error() == QXmlStreamReader::PrematureEndOfDocumentError) in doEnd().
1446 // Check if all the chunk has been processed.
1447 long bytesProcessed = xmlByteConsumed(context());
1448 if (bytesProcessed == -1 || ((unsigned long)bytesProcessed) != chunkAsUtf8.length()) {
1449 // FIXME: I don't believe we can hit this case without also having seen an error or a null byte.
1450 // If we hit this ASSERT, we've found a test case which demonstrates the need for this code.
1451 ASSERT(m_sawError || (bytesProcessed >= 0 && !chunkAsUtf8.data()[bytesProcessed]));
1452 return false;
1453 }
1454
1455 // No error if the chunk is well formed or it is not but we have no error.
1456 return context()->wellFormed || !xmlCtxtGetLastError(context());
1457}

1463static void attributesStartElementNsHandler(void* closure, const xmlChar* xmlLocalName, const xmlChar* /*xmlPrefix*/, const xmlChar* /*xmlURI*/, int /*numNamespaces*/, const xmlChar** /*namespaces*/, int numAttributes, int /*numDefaulted*/, const xmlChar** libxmlAttributes)
1464{
1465 if (strcmp(reinterpret_cast<const char*>(xmlLocalName), "attrs") != 0)
1466 return;
1467
1468 auto& state = *static_cast<AttributeParseState*>(static_cast<xmlParserCtxtPtr>(closure)->_private);
1469
1470 state = HashMap<String, String> { };
1471
1472 xmlSAX2Attributes* attributes = reinterpret_cast<xmlSAX2Attributes*>(libxmlAttributes);
1473 for (int i = 0; i < numAttributes; i++) {
1474 String attrLocalName = toString(attributes[i].localname);
1475 int valueLength = (int) (attributes[i].end - attributes[i].value);
1476 String attrValue = toString(attributes[i].value, valueLength);
1477 String attrPrefix = toString(attributes[i].prefix);
1478 String attrQName = attrPrefix.isEmpty() ? attrLocalName : attrPrefix + ":" + attrLocalName;
1479
1480 state->set(attrQName, attrValue);
1481 }

1484std::optional<HashMap<String, String>> parseAttributes(const String& string)
1485{
1486 String parseString = "<?xml version=\"1.0\"?><attrs " + string + " />";
1487
1488 AttributeParseState attributes;
1489
1490 xmlSAXHandler sax;
1491 memset(&sax, 0, sizeof(sax));
1492 sax.startElementNs = attributesStartElementNsHandler;
1493 sax.initialized = XML_SAX2_MAGIC;

1496
1497 // FIXME: Can we parse 8-bit strings directly as Latin-1 instead of upconverting to UTF-16?
