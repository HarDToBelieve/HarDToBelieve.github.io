78// We wouldn't want the GPUProcess to repeatedly exit then relaunch when under memory pressure. In particular, we need to make sure the
79// WebProcess has a change to schedule work after the GPUProcess get launched. For this reason, we make sure that the GPUProcess never
80// idle-exits less than 5 seconds after getting launched. This amount of time should be sufficient for the WebProcess to schedule work
81// work in the GPUProcess.
82constexpr Seconds minimumLifetimeBeforeIdleExit { 5_s };

108void GPUProcess::createGPUConnectionToWebProcess(ProcessIdentifier identifier, PAL::SessionID sessionID, GPUProcessConnectionParameters&& parameters, CompletionHandler<void(std::optional<IPC::Attachment>&&)>&& completionHandler)
109{
110 RELEASE_LOG(Process, "%p - GPUProcess::createGPUConnectionToWebProcess: processIdentifier=%" PRIu64, this, identifier.toUInt64());
111 auto ipcConnection = createIPCConnectionPair();
112 if (!ipcConnection) {
113 completionHandler({ });
114 return;
115 }
116
117 auto newConnection = GPUConnectionToWebProcess::create(*this, identifier, ipcConnection->first, sessionID, WTFMove(parameters));
118
119#if ENABLE(MEDIA_STREAM)
120 // FIXME: We should refactor code to go from WebProcess -> GPUProcess -> UIProcess when getUserMedia is called instead of going from WebProcess -> UIProcess directly.
121 auto access = m_mediaCaptureAccessMap.take(identifier);
122 newConnection->updateCaptureAccess(access.allowAudioCapture, access.allowVideoCapture, access.allowDisplayCapture);
123 newConnection->setOrientationForMediaCapture(m_orientation);
124#endif
125
126#if ENABLE(IPC_TESTING_API)
127 if (parameters.ignoreInvalidMessageForTesting)
128 newConnection->connection().setIgnoreInvalidMessageForTesting();
129#endif
130
131 ASSERT(!m_webProcessConnections.contains(identifier));
132 m_webProcessConnections.add(identifier, WTFMove(newConnection));
133
134 completionHandler(WTFMove(ipcConnection->second));
135}

154bool GPUProcess::canExitUnderMemoryPressure() const
155{
156 ASSERT(isMainRunLoop());
157 for (auto& webProcessConnection : m_webProcessConnections.values()) {
158 if (!webProcessConnection->allowsExitUnderMemoryPressure())
159 return false;
160 }
161 return true;
162}

164void GPUProcess::tryExitIfUnusedAndUnderMemoryPressure()
165{
166 ASSERT(isMainRunLoop());
167 if (!MemoryPressureHandler::singleton().isUnderMemoryPressure())
168 return;
169
170 tryExitIfUnused();
171}
172
173void GPUProcess::tryExitIfUnused()
174{
175 ASSERT(isMainRunLoop());
176 if (!canExitUnderMemoryPressure()) {
177 m_idleExitTimer.stop();
178 return;
179 }
180
181 // To avoid exiting the GPUProcess too aggressively while under memory pressure and make sure the WebProcess gets a
182 // change to schedule work, we don't exit if we've been running for less than |minimumLifetimeBeforeIdleExit|.
183 // In case of simulated memory pressure, we ignore this rule to avoid flakiness in our benchmarks and tests.
184 auto lifetime = MonotonicTime::now() - m_creationTime;
185 if (lifetime < minimumLifetimeBeforeIdleExit && !MemoryPressureHandler::singleton().isSimulatingMemoryPressure()) {
186 RELEASE_LOG(Process, "GPUProcess::tryExitIfUnused: GPUProcess is idle and under memory pressure but it is not exiting because it has just launched");
187 // Check again after the process have lived long enough (minimumLifetimeBeforeIdleExit) to see if the GPUProcess
188 // can idle-exit then.
189 if (!m_idleExitTimer.isActive())
190 m_idleExitTimer.startOneShot(minimumLifetimeBeforeIdleExit - lifetime);
191 return;
192 }
193 m_idleExitTimer.stop();
194
195 RELEASE_LOG(Process, "GPUProcess::tryExitIfUnused: GPUProcess is exiting because we are under memory pressure and the process is no longer useful.");
196 parentProcessConnection()->send(Messages::GPUProcessProxy::ProcessIsReadyToExit(), 0);
197}

199void GPUProcess::lowMemoryHandler(Critical critical, Synchronous synchronous)
200{
201 RELEASE_LOG(Process, "GPUProcess::lowMemoryHandler: critical=%d, synchronous=%d", critical == Critical::Yes, synchronous == Synchronous::Yes);
202 tryExitIfUnused();
203
204 WebCore::releaseGraphicsMemory(critical, synchronous);
205}

207void GPUProcess::initializeGPUProcess(GPUProcessCreationParameters&& parameters)
208{
209 RELEASE_LOG(Process, "%p - GPUProcess::initializeGPUProcess:", this);
210 WTF::Thread::setCurrentThreadIsUserInitiated();
211 AtomString::init();
212
213 auto& memoryPressureHandler = MemoryPressureHandler::singleton();
214 memoryPressureHandler.setLowMemoryHandler([this] (Critical critical, Synchronous synchronous) {
215 lowMemoryHandler(critical, synchronous);
216 });
217 memoryPressureHandler.install();
218
219#if PLATFORM(IOS_FAMILY) || ENABLE(ROUTING_ARBITRATION)
220 DeprecatedGlobalSettings::setShouldManageAudioSessionCategory(true);
221#endif
222
223#if ENABLE(MEDIA_STREAM)
224 setMockCaptureDevicesEnabled(parameters.useMockCaptureDevices);
225#if PLATFORM(MAC)
226 SandboxExtension::consumePermanently(parameters.microphoneSandboxExtensionHandle);
227#endif
228#endif // ENABLE(MEDIA_STREAM)
229
230#if USE(SANDBOX_EXTENSIONS_FOR_CACHE_AND_TEMP_DIRECTORY_ACCESS)
231 SandboxExtension::consumePermanently(parameters.containerCachesDirectoryExtensionHandle);
232 SandboxExtension::consumePermanently(parameters.containerTemporaryDirectoryExtensionHandle);
233#endif
234#if PLATFORM(IOS_FAMILY)
235 SandboxExtension::consumePermanently(parameters.compilerServiceExtensionHandles);
236 SandboxExtension::consumePermanently(parameters.dynamicIOKitExtensionHandles);
237 SandboxExtension::consumePermanently(parameters.dynamicMachExtensionHandles);
238#endif
239
240#if HAVE(CGIMAGESOURCE_WITH_SET_ALLOWABLE_TYPES)
241 auto emptyArray = adoptCF(CFArrayCreate(kCFAllocatorDefault, nullptr, 0, &kCFTypeArrayCallBacks));
242 CGImageSourceSetAllowableTypes(emptyArray.get());
243#endif
244
245#if !LOG_DISABLED || !RELEASE_LOG_DISABLED
246 WebCore::initializeLogChannelsIfNecessary(parameters.webCoreLoggingChannels);
247 WebKit::initializeLogChannelsIfNecessary(parameters.webKitLoggingChannels);
248#endif
249
250 // Match the QoS of the UIProcess since the GPU process is doing rendering on its behalf.
251 WTF::Thread::setCurrentThreadIsUserInteractive(0);
252
253 WebCore::setPresentingApplicationPID(parameters.parentPID);
254}

300void GPUProcess::updateCaptureAccess(bool allowAudioCapture, bool allowVideoCapture, bool allowDisplayCapture, ProcessIdentifier processID, CompletionHandler<void()>&& completionHandler)
301{
302 if (auto* connection = webProcessConnection(processID)) {
303 connection->updateCaptureAccess(allowAudioCapture, allowVideoCapture, allowDisplayCapture);
304 return completionHandler();
305 }
306
307 auto& access = m_mediaCaptureAccessMap.add(processID, MediaCaptureAccess { allowAudioCapture, allowVideoCapture, allowDisplayCapture }).iterator->value;
308 access.allowAudioCapture |= allowAudioCapture;
309 access.allowVideoCapture |= allowVideoCapture;
310 access.allowDisplayCapture |= allowDisplayCapture;
311
312 completionHandler();
313}

350void GPUProcess::addSession(PAL::SessionID sessionID, GPUProcessSessionParameters&& parameters)
351{
352 ASSERT(!m_sessions.contains(sessionID));
353 SandboxExtension::consumePermanently(parameters.mediaCacheDirectorySandboxExtensionHandle);
354#if ENABLE(LEGACY_ENCRYPTED_MEDIA)
355 SandboxExtension::consumePermanently(parameters.mediaKeysStorageDirectorySandboxExtensionHandle);
356#endif
357
358 m_sessions.add(sessionID, GPUSession {
359 WTFMove(parameters.mediaCacheDirectory)
360#if ENABLE(LEGACY_ENCRYPTED_MEDIA)
361 , WTFMove(parameters.mediaKeysStorageDirectory)
362#endif
363 });
364}
