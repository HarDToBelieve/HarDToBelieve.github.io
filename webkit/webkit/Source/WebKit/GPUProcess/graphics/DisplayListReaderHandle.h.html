26#pragma once
27
28#include "SharedDisplayListHandle.h"
29#include <wtf/Expected.h>
30#include <wtf/FastMalloc.h>
31
32namespace WebKit {
33
34class DisplayListReaderHandle : public SharedDisplayListHandle {
35 WTF_MAKE_NONCOPYABLE(DisplayListReaderHandle); WTF_MAKE_FAST_ALLOCATED;
36public:
37 static RefPtr<DisplayListReaderHandle> create(WebCore::DisplayList::ItemBufferIdentifier identifier, Ref<SharedMemory>&& sharedMemory)
38 {
39 if (sharedMemory->size() <= headerSize())
40 return nullptr;
41 return adoptRef(new DisplayListReaderHandle(identifier, WTFMove(sharedMemory)));
42 }
43
44 std::optional<size_t> advance(size_t amount);
45 std::unique_ptr<WebCore::DisplayList::DisplayList> displayListForReading(size_t offset, size_t capacity, WebCore::DisplayList::ItemBufferReadingClient&) const;
46
47 void startWaiting()
48 {
49 header().waitingStatus.store(static_cast<SharedDisplayListHandle::WaitingStatusStorageType>(SharedDisplayListHandle::WaitingStatus::Waiting));
50 }
51
52 enum class StopWaitingFailureReason : uint8_t { InvalidWaitingStatus };
53 Expected<std::optional<SharedDisplayListHandle::ResumeReadingInformation>, StopWaitingFailureReason> stopWaiting()
54 {
55 auto& header = this->header();
56 auto previousStatus = header.waitingStatus.exchange(static_cast<SharedDisplayListHandle::WaitingStatusStorageType>(SharedDisplayListHandle::WaitingStatus::NotWaiting));
57 if (!isValidEnum<SharedDisplayListHandle::WaitingStatus>(previousStatus))
58 return makeUnexpected(StopWaitingFailureReason::InvalidWaitingStatus);
59
60 if (static_cast<SharedDisplayListHandle::WaitingStatus>(previousStatus) == SharedDisplayListHandle::WaitingStatus::Resuming)
61 return { header.resumeReadingInfo };
62
63 return { std::nullopt };
64 }
65
66private:
67 DisplayListReaderHandle(WebCore::DisplayList::ItemBufferIdentifier identifier, Ref<SharedMemory>&& sharedMemory)
68 : SharedDisplayListHandle(identifier, WTFMove(sharedMemory))
69 {
70 }
71};
