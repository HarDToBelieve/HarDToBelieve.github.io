28#if ENABLE(GPU_PROCESS)

30#include "IPCSemaphore.h"
31#include "RenderingBackendIdentifier.h"
32#include "WebPageProxyIdentifier.h"
33#include <WebCore/PageIdentifier.h>
34
35namespace WebKit {
36
37struct RemoteRenderingBackendCreationParameters {
38 RenderingBackendIdentifier identifier;
39 IPC::Semaphore resumeDisplayListSemaphore;
40 WebPageProxyIdentifier pageProxyID;
41 WebCore::PageIdentifier pageID;
42
43 void encode(IPC::Encoder& encoder) const
44 {
45 encoder << identifier << resumeDisplayListSemaphore << pageProxyID << pageID;
46 }
47
48 static std::optional<RemoteRenderingBackendCreationParameters> decode(IPC::Decoder& decoder)
49 {
50 std::optional<RenderingBackendIdentifier> identifier;
51 decoder >> identifier;
52 if (!identifier)
53 return std::nullopt;
54
55 std::optional<IPC::Semaphore> resumeDisplayListSemaphore;
56 decoder >> resumeDisplayListSemaphore;
57 if (!resumeDisplayListSemaphore)
58 return std::nullopt;
59
60 std::optional<WebPageProxyIdentifier> pageProxyID;
61 decoder >> pageProxyID;
62 if (!pageProxyID)
63 return std::nullopt;
64
65 std::optional<WebCore::PageIdentifier> pageID;
66 decoder >> pageID;
67 if (!pageID)
68 return std::nullopt;
69
70 return {{ *identifier, WTFMove(*resumeDisplayListSemaphore), *pageProxyID, *pageID }};
71 }
72};
73
74} // namespace WebKit
75
76#endif // ENABLE(GPU_PROCESS)
