36namespace WebKit {

38class RemoteRenderingBackend;
39
40class RemoteResourceCache {
41public:
42 RemoteResourceCache() = default;
43
44 void cacheImageBuffer(Ref<WebCore::ImageBuffer>&&);
45 WebCore::ImageBuffer* cachedImageBuffer(WebCore::RenderingResourceIdentifier);
46 void cacheNativeImage(Ref<WebCore::NativeImage>&&);
47 void cacheFont(Ref<WebCore::Font>&&);
48 void deleteAllFonts();
49 void releaseRemoteResource(WebCore::RenderingResourceIdentifier);
50
51 const WebCore::ImageBufferHashMap& imageBuffers() const { return m_imageBuffers; }
52 const WebCore::NativeImageHashMap& nativeImages() const { return m_nativeImages; }
53 const WebCore::FontRenderingResourceMap& fonts() const { return m_fonts; }
54
55private:
56 WebCore::ImageBufferHashMap m_imageBuffers;
57 WebCore::NativeImageHashMap m_nativeImages;
58 WebCore::FontRenderingResourceMap m_fonts;
59};
