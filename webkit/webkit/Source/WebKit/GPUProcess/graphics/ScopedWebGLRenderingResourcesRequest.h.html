26#pragma once
27
28#include "ScopedRenderingResourcesRequest.h"
29#include <atomic>
30#include <wtf/StdLibExtras.h>
31
32namespace WebKit {
33
34class ScopedWebGLRenderingResourcesRequest {
35public:
36 ScopedWebGLRenderingResourcesRequest() = default;
37 ScopedWebGLRenderingResourcesRequest(ScopedWebGLRenderingResourcesRequest&& other)
38 : m_requested(std::exchange(other.m_requested, false))
39 , m_renderingResourcesRequest(WTFMove(other.m_renderingResourcesRequest))
40 {
41 }
42 ~ScopedWebGLRenderingResourcesRequest()
43 {
44 reset();
45 }
46 ScopedWebGLRenderingResourcesRequest& operator=(ScopedWebGLRenderingResourcesRequest&& other)
47 {
48 if (this != &other) {
49 reset();
50 m_requested = std::exchange(other.m_requested, false);
51 m_renderingResourcesRequest = WTFMove(other.m_renderingResourcesRequest);
52 }
53 return *this;
54 }
55 bool isRequested() const { return m_requested; }
56 void reset()
57 {
58 if (!m_requested)
59 return;
60 ASSERT(s_requests);
61 --s_requests;
62 if (!s_requests)
63 scheduleFreeWebGLRenderingResources();
64 }
65 static ScopedWebGLRenderingResourcesRequest acquire()
66 {
67 return { DidRequest };
68 }
69private:
70 static void scheduleFreeWebGLRenderingResources();
71 static void freeWebGLRenderingResources();
72 enum RequestState { DidRequest };
73 ScopedWebGLRenderingResourcesRequest(RequestState)
74 : m_requested(true)
75 , m_renderingResourcesRequest(ScopedRenderingResourcesRequest::acquire())
76 {
77 ++s_requests;
78 }
79 bool m_requested { false };
80 ScopedRenderingResourcesRequest m_renderingResourcesRequest;
81 static std::atomic<unsigned> s_requests;
82};
