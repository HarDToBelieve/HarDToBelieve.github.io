44namespace WebKit {
45
46class RemoteAudioDestination final
47#if PLATFORM(COCOA)
48 : public WebCore::AudioUnitRenderer
49#endif
50{
51 WTF_MAKE_FAST_ALLOCATED;
52public:
53 RemoteAudioDestination(GPUConnectionToWebProcess&, RemoteAudioDestinationIdentifier identifier, const String& inputDeviceId, uint32_t numberOfInputChannels, uint32_t numberOfOutputChannels, float sampleRate, float hardwareSampleRate, IPC::Semaphore&& renderSemaphore)
54 : m_id(identifier)
55#if PLATFORM(COCOA)
56 , m_audioOutputUnitAdaptor(*this)
57 , m_ringBuffer(makeUniqueRef<WebCore::CARingBuffer>())
58#endif
59 , m_renderSemaphore(WTFMove(renderSemaphore))
60 {
61 ASSERT(isMainRunLoop());
62#if PLATFORM(COCOA)
63 m_audioOutputUnitAdaptor.configure(hardwareSampleRate, numberOfOutputChannels);
64#endif
65 }
66
67 ~RemoteAudioDestination()
68 {
69 ASSERT(isMainRunLoop());
70 // Make sure we stop audio rendering and wait for it to finish before destruction.
71 if (m_isPlaying)
72 stop();
73 }
74
75#if PLATFORM(COCOA)
76 void audioSamplesStorageChanged(const SharedMemory::IPCHandle& ipcHandle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames)
77 {
78 m_ringBuffer = WebCore::CARingBuffer::adoptStorage(makeUniqueRef<ReadOnlySharedRingBufferStorage>(ipcHandle.handle), description, numberOfFrames);
79 }
80#endif
81
82 void start()
83 {
84#if PLATFORM(COCOA)
85 if (m_audioOutputUnitAdaptor.start())
86 return;
87
88 m_isPlaying = true;
89#endif
90 }
91
92 void stop()
93 {
94#if PLATFORM(COCOA)
95 if (m_audioOutputUnitAdaptor.stop())
96 return;
97
98 m_isPlaying = false;
99#endif
100 }
101
102 bool isPlaying() const { return m_isPlaying; }
103
104private:
105#if PLATFORM(COCOA)
106 OSStatus render(double sampleTime, uint64_t hostTime, UInt32 numberOfFrames, AudioBufferList* ioData)
107 {
108 ASSERT(!isMainRunLoop());
109
110 OSStatus status = -1;
111 if (m_ringBuffer->fetchIfHasEnoughData(ioData, numberOfFrames, m_startFrame)) {
112 m_startFrame += numberOfFrames;
113 status = noErr;
114 }
115
116 for (unsigned i = 0; i < numberOfFrames; i += WebCore::AudioUtilities::renderQuantumSize) {
117 // Ask the audio thread in the WebContent process to render a quantum.
118 m_renderSemaphore.signal();
119 }
120
121 return status;
122 }
123#endif
124
125 RemoteAudioDestinationIdentifier m_id;
126
127#if PLATFORM(COCOA)
128 WebCore::AudioOutputUnitAdaptor m_audioOutputUnitAdaptor;
129
130 UniqueRef<WebCore::CARingBuffer> m_ringBuffer;
131 uint64_t m_startFrame { 0 };
132#endif
133 IPC::Semaphore m_renderSemaphore;
134
135 bool m_isPlaying { false };
136};

138RemoteAudioDestinationManager::RemoteAudioDestinationManager(GPUConnectionToWebProcess& connection)
139 : m_gpuConnectionToWebProcess(connection)
140{
141}
142
143RemoteAudioDestinationManager::~RemoteAudioDestinationManager() = default;
144
145void RemoteAudioDestinationManager::createAudioDestination(const String& inputDeviceId, uint32_t numberOfInputChannels, uint32_t numberOfOutputChannels, float sampleRate, float hardwareSampleRate, IPC::Semaphore&& renderSemaphore, CompletionHandler<void(const WebKit::RemoteAudioDestinationIdentifier)>&& completionHandler)
146{
147 auto newID = RemoteAudioDestinationIdentifier::generateThreadSafe();
148 auto destination = makeUniqueRef<RemoteAudioDestination>(m_gpuConnectionToWebProcess, newID, inputDeviceId, numberOfInputChannels, numberOfOutputChannels, sampleRate, hardwareSampleRate, WTFMove(renderSemaphore));
149 m_audioDestinations.add(newID, WTFMove(destination));
150 completionHandler(newID);
151}
152
153void RemoteAudioDestinationManager::deleteAudioDestination(RemoteAudioDestinationIdentifier identifier, CompletionHandler<void()>&& completionHandler)
154{
155 m_audioDestinations.remove(identifier);
156 completionHandler();
157
158 if (allowsExitUnderMemoryPressure())
159 m_gpuConnectionToWebProcess.gpuProcess().tryExitIfUnusedAndUnderMemoryPressure();
160}
161
162void RemoteAudioDestinationManager::startAudioDestination(RemoteAudioDestinationIdentifier identifier, CompletionHandler<void(bool)>&& completionHandler)
163{
164 bool isPlaying = false;
165 if (auto* item = m_audioDestinations.get(identifier)) {
166 item->start();
167 isPlaying = item->isPlaying();
168 }
169 completionHandler(isPlaying);
170}
171
172void RemoteAudioDestinationManager::stopAudioDestination(RemoteAudioDestinationIdentifier identifier, CompletionHandler<void(bool)>&& completionHandler)
173{
174 bool isPlaying = false;
175 if (auto* item = m_audioDestinations.get(identifier)) {
176 item->stop();
177 isPlaying = item->isPlaying();
178 }
179 completionHandler(isPlaying);
180}
181
182#if PLATFORM(COCOA)
183void RemoteAudioDestinationManager::audioSamplesStorageChanged(RemoteAudioDestinationIdentifier identifier, const SharedMemory::IPCHandle& ipcHandle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames)
184{
185 if (auto* item = m_audioDestinations.get(identifier))
186 item->audioSamplesStorageChanged(ipcHandle, description, numberOfFrames);
187}
188#endif
189
190bool RemoteAudioDestinationManager::allowsExitUnderMemoryPressure() const
191{
192 for (auto& audioDestination : m_audioDestinations.values()) {
193 if (audioDestination->isPlaying())
194 return false;
195 }
196 return true;
197}
