46class GPUConnectionToWebProcess;

49class RemoteAudioDestinationManager : private IPC::MessageReceiver {
50 WTF_MAKE_FAST_ALLOCATED;
51 WTF_MAKE_NONCOPYABLE(RemoteAudioDestinationManager);
52public:
53 RemoteAudioDestinationManager(GPUConnectionToWebProcess&);
54 ~RemoteAudioDestinationManager();
55
56 bool didReceiveSyncMessageFromWebProcess(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& encoder) { return didReceiveSyncMessage(connection, decoder, encoder); }
57 void didReceiveMessageFromWebProcess(IPC::Connection& connection, IPC::Decoder& decoder) { didReceiveMessage(connection, decoder); }
58
59 bool allowsExitUnderMemoryPressure() const;
60
61private:
62 void didReceiveMessage(IPC::Connection&, IPC::Decoder&);
63 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&);
64
65 void createAudioDestination(const String& inputDeviceId, uint32_t numberOfInputChannels, uint32_t numberOfOutputChannels, float sampleRate, float hardwareSampleRate, IPC::Semaphore&& renderSemaphore, CompletionHandler<void(const WebKit::RemoteAudioDestinationIdentifier)>&&);
66
67 void deleteAudioDestination(RemoteAudioDestinationIdentifier, CompletionHandler<void()>&&);
68 void startAudioDestination(RemoteAudioDestinationIdentifier, CompletionHandler<void(bool)>&&);
69 void stopAudioDestination(RemoteAudioDestinationIdentifier, CompletionHandler<void(bool)>&&);
70#if PLATFORM(COCOA)
71 void audioSamplesStorageChanged(RemoteAudioDestinationIdentifier, const SharedMemory::IPCHandle&, const WebCore::CAAudioStreamDescription&, uint64_t numberOfFrames);
72#endif
73
74 HashMap<RemoteAudioDestinationIdentifier, UniqueRef<RemoteAudioDestination>> m_audioDestinations;
75 GPUConnectionToWebProcess& m_gpuConnectionToWebProcess;
76};
