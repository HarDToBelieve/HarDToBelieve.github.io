42UniqueRef<RemoteAudioSessionProxy> RemoteAudioSessionProxy::create(GPUConnectionToWebProcess& gpuConnection)
43{
44 return makeUniqueRef<RemoteAudioSessionProxy>(gpuConnection);
45}

47RemoteAudioSessionProxy::RemoteAudioSessionProxy(GPUConnectionToWebProcess& gpuConnection)
48 : m_gpuConnection(gpuConnection)
49{
50}
51
52RemoteAudioSessionProxy::~RemoteAudioSessionProxy() = default;
53
54WebCore::ProcessIdentifier RemoteAudioSessionProxy::processIdentifier()
55{
56 return m_gpuConnection.webProcessIdentifier();
57}
58
59RemoteAudioSessionConfiguration RemoteAudioSessionProxy::configuration()
60{
61 auto& session = audioSessionManager().session();
62 return {
63 session.routingContextUID(),
64 session.sampleRate(),
65 session.bufferSize(),
66 session.numberOfOutputChannels(),
67 session.maximumNumberOfOutputChannels(),
68 session.preferredBufferSize(),
69 session.isMuted(),
70 m_active
71 };
72}
73
74void RemoteAudioSessionProxy::setCategory(AudioSession::CategoryType category, RouteSharingPolicy policy)
75{
76 if (m_category == category && m_routeSharingPolicy == policy)
77 return;
78
79 m_category = category;
80 m_routeSharingPolicy = policy;
81 audioSessionManager().updateCategory();
82}
83
84void RemoteAudioSessionProxy::setPreferredBufferSize(uint64_t size)
85{
86 m_preferredBufferSize = size;
87 audioSessionManager().setPreferredBufferSizeForProcess(*this, size);
88}
89
90void RemoteAudioSessionProxy::tryToSetActive(bool active, SetActiveCompletion&& completion)
91{
92 m_active = audioSessionManager().tryToSetActiveForProcess(*this, active);
93 completion(m_active);
94}
95
96void RemoteAudioSessionProxy::beginInterruption()
97{
98 connection().send(Messages::RemoteAudioSession::BeginInterruption(), { });
99}
100
101void RemoteAudioSessionProxy::endInterruption(AudioSession::MayResume mayResume)
102{
103 connection().send(Messages::RemoteAudioSession::EndInterruption(mayResume), { });
104}
105
106RemoteAudioSessionProxyManager& RemoteAudioSessionProxy::audioSessionManager()
107{
108 return m_gpuConnection.gpuProcess().audioSessionManager();
109}

111IPC::Connection& RemoteAudioSessionProxy::connection()

113 return m_gpuConnection.connection();
