36Ref<RemoteAudioSourceProviderProxy> RemoteAudioSourceProviderProxy::create(WebCore::MediaPlayerIdentifier identifier, Ref<IPC::Connection>&& connection, AudioSourceProviderAVFObjC& localProvider)
37{
38 auto remoteProvider = adoptRef(*new RemoteAudioSourceProviderProxy(identifier, WTFMove(connection)));
39
40 localProvider.setRingBufferCreationCallback([remoteProvider]() {
41 return remoteProvider->createRingBuffer();
42 });
43 localProvider.setAudioCallback([remoteProvider](auto startFrame, auto numberOfFrames) {
44 remoteProvider->newAudioSamples(startFrame, numberOfFrames);
45 });
46
47 return remoteProvider;
48}
49
50RemoteAudioSourceProviderProxy::RemoteAudioSourceProviderProxy(WebCore::MediaPlayerIdentifier identifier, Ref<IPC::Connection>&& connection)
51 : m_identifier(identifier)
52 , m_connection(WTFMove(connection))
53{
54}
55
56RemoteAudioSourceProviderProxy::~RemoteAudioSourceProviderProxy() = default;
57
58UniqueRef<CARingBuffer> RemoteAudioSourceProviderProxy::createRingBuffer()
59{
60 return makeUniqueRef<CARingBuffer>(makeUniqueRef<SharedRingBufferStorage>([protectedThis = makeRef(*this)](SharedMemory* memory, const CAAudioStreamDescription& format, size_t frameCount) mutable {
61 protectedThis->storageChanged(memory, format, frameCount);
62 }));
63}
64
65void RemoteAudioSourceProviderProxy::newAudioSamples(uint64_t startFrame, uint64_t numberOfFrames)
66{
67 m_connection->send(Messages::RemoteAudioSourceProviderManager::AudioSamplesAvailable { m_identifier, startFrame, numberOfFrames }, 0);
68}
69
70void RemoteAudioSourceProviderProxy::storageChanged(SharedMemory* memory, const CAAudioStreamDescription& format, size_t frameCount)
71{
72 SharedMemory::Handle handle;
73 if (memory)
74 memory->createHandle(handle, SharedMemory::Protection::ReadOnly);
75
76 // FIXME: Send the actual data size with IPCHandle.
77#if OS(DARWIN) || OS(WINDOWS)
78 uint64_t dataSize = handle.size();
79#else
80 uint64_t dataSize = 0;
81#endif
82 m_connection->send(Messages::RemoteAudioSourceProviderManager::AudioStorageChanged { m_identifier, SharedMemory::IPCHandle { WTFMove(handle), dataSize }, format, frameCount }, 0);
83}
