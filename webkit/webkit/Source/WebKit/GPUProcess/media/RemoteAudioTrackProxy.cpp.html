42RemoteAudioTrackProxy::RemoteAudioTrackProxy(GPUConnectionToWebProcess& connectionToWebProcess, TrackPrivateRemoteIdentifier identifier, AudioTrackPrivate& trackPrivate, MediaPlayerIdentifier mediaPlayerIdentifier)
43 : m_connectionToWebProcess(makeWeakPtr(connectionToWebProcess))
44 , m_identifier(identifier)
45 , m_trackPrivate(trackPrivate)
46 , m_mediaPlayerIdentifier(mediaPlayerIdentifier)
47{
48 m_trackPrivate->setClient(this);
49 m_connectionToWebProcess->connection().send(Messages::MediaPlayerPrivateRemote::AddRemoteAudioTrack(m_identifier, configuration()), m_mediaPlayerIdentifier);
50}
51
52RemoteAudioTrackProxy::~RemoteAudioTrackProxy()
53{
54}
55
56TrackPrivateRemoteConfiguration& RemoteAudioTrackProxy::configuration()
57{
58 static NeverDestroyed<TrackPrivateRemoteConfiguration> configuration;
59
60 configuration->trackId = m_trackPrivate->id();
61 configuration->label = m_trackPrivate->label();
62 configuration->language = m_trackPrivate->language();
63 configuration->trackIndex = m_trackPrivate->trackIndex();
64 configuration->startTimeVariance = m_trackPrivate->startTimeVariance();
65 configuration->enabled = m_trackPrivate->enabled();
66 configuration->audioKind = m_trackPrivate->kind();
67
68 return configuration.get();
69}
70
71void RemoteAudioTrackProxy::configurationChanged()
72{
73 if (!m_connectionToWebProcess)
74 return;
75
76 m_connectionToWebProcess->connection().send(Messages::MediaPlayerPrivateRemote::RemoteAudioTrackConfigurationChanged(m_identifier, configuration()), m_mediaPlayerIdentifier);
77}
78
79void RemoteAudioTrackProxy::willRemove()
80{
81 ASSERT_NOT_REACHED();
82}
83
84void RemoteAudioTrackProxy::enabledChanged(bool enabled)
85{
86 if (enabled == m_enabled)
87 return;
88 m_enabled = enabled;
89 configurationChanged();
90}

92void RemoteAudioTrackProxy::idChanged(const AtomString&)

99 configurationChanged();

102void RemoteAudioTrackProxy::languageChanged(const AtomString&)

104 configurationChanged();
105}
