42std::unique_ptr<RemoteCDMInstanceProxy> RemoteCDMInstanceProxy::create(WeakPtr<RemoteCDMProxy>&& cdm, Ref<CDMInstance>&& priv, RemoteCDMInstanceIdentifier identifier)
43{
44 auto configuration = makeUniqueRefWithoutFastMallocCheck<RemoteCDMInstanceConfiguration, RemoteCDMInstanceConfiguration&&>({
45 priv->keySystem(),
46 });
47 return std::unique_ptr<RemoteCDMInstanceProxy>(new RemoteCDMInstanceProxy(WTFMove(cdm), WTFMove(priv), WTFMove(configuration), identifier));
48}

50RemoteCDMInstanceProxy::RemoteCDMInstanceProxy(WeakPtr<RemoteCDMProxy>&& cdm, Ref<CDMInstance>&& priv, UniqueRef<RemoteCDMInstanceConfiguration>&& configuration, RemoteCDMInstanceIdentifier identifier)
51 : m_cdm(WTFMove(cdm))

64void RemoteCDMInstanceProxy::unrequestedInitializationDataReceived(const String& type, Ref<SharedBuffer>&& initData)
65{
66 if (!m_cdm)
67 return;
68
69 auto* factory = m_cdm->factory();
70 if (!factory)
71 return;
72
73 auto* gpuConnectionToWebProcess = factory->gpuConnectionToWebProcess();
74 if (!gpuConnectionToWebProcess)
75 return;
76
77 gpuConnectionToWebProcess->connection().send(Messages::RemoteCDMInstance::UnrequestedInitializationDataReceived(type, WTFMove(initData)), m_identifier);
78}

85void RemoteCDMInstanceProxy::setServerCertificate(IPC::SharedBufferCopy&& certificate, CompletionHandler<void(SuccessValue)>&& completion)
86{
87 if (!certificate.buffer()) {
88 completion(CDMInstance::Failed);

91
92 m_instance->setServerCertificate(certificate.buffer().releaseNonNull(), WTFMove(completion));
93}

100void RemoteCDMInstanceProxy::createSession(CompletionHandler<void(const RemoteCDMInstanceSessionIdentifier&)>&& completion)
101{
102 auto privSession = m_instance->createSession();
103 if (!privSession || !m_cdm || !m_cdm->factory()) {
104 completion({ });
105 return;
106 }
107 auto identifier = RemoteCDMInstanceSessionIdentifier::generate();
108 auto session = RemoteCDMInstanceSessionProxy::create(makeWeakPtr(m_cdm.get()), privSession.releaseNonNull(), identifier);
109 m_cdm->factory()->addSession(identifier, WTFMove(session));
110 completion(identifier);
111}
