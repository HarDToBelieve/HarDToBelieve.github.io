45class RemoteCDMInstanceSessionProxy final : private IPC::MessageReceiver, private WebCore::CDMInstanceSessionClient {

50private:

57 // Types

59 using KeyStatusVector = WebCore::CDMInstanceSession::KeyStatusVector;

69 // Messages
70 void requestLicense(LicenseType, AtomString initDataType, IPC::SharedBufferCopy&& initData, LicenseCallback&&);
71 void updateLicense(String sessionId, LicenseType, IPC::SharedBufferCopy&& response, LicenseUpdateCallback&&);

73 void closeSession(const String& sessionId, CloseSessionCallback&&);
74 void removeSessionData(const String& sessionId, LicenseType, RemoveSessionDataCallback&&);
75 void storeRecordOfKeyUsage(const String& sessionId);
76 void displayIDChanged(PlatformDisplayID);
77
78 // CDMInstanceSessionClient
79 void updateKeyStatuses(KeyStatusVector&&) final;
80 void sendMessage(WebCore::CDMMessageType, Ref<WebCore::SharedBuffer>&& message) final;
81 void sessionIdChanged(const String&) final;
82 PlatformDisplayID displayID() final { return m_displayID; }
83
84 WeakPtr<RemoteCDMProxy> m_cdm;
85 Ref<WebCore::CDMInstanceSession> m_session;
86 RemoteCDMInstanceSessionIdentifier m_identifier;
87 PlatformDisplayID m_displayID { 0 };
