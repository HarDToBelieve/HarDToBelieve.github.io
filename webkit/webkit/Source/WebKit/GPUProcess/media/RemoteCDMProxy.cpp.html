42std::unique_ptr<RemoteCDMProxy> RemoteCDMProxy::create(WeakPtr<RemoteCDMFactoryProxy>&& factory, std::unique_ptr<WebCore::CDMPrivate>&& priv)
43{
44 if (!priv)
45 return nullptr;
46
47 auto configuration = makeUniqueRefWithoutFastMallocCheck<RemoteCDMConfiguration, RemoteCDMConfiguration&&>({
48 priv->supportedInitDataTypes(),
49 priv->supportedRobustnesses(),
50 priv->supportsServerCertificates(),
51 priv->supportsSessions()
52 });
53 // Use new() to access CDMPrivate's private constructor.
54 return std::unique_ptr<RemoteCDMProxy>(new RemoteCDMProxy(WTFMove(factory), WTFMove(priv), WTFMove(configuration)));
55}

86void RemoteCDMProxy::createInstance(CompletionHandler<void(RemoteCDMInstanceIdentifier, RemoteCDMInstanceConfiguration&&)>&& completion)
87{
88 auto privateInstance = m_private->createInstance();
89 if (!privateInstance || !m_factory) {
90 completion({ }, { });
91 return;
92 }
93 auto identifier = RemoteCDMInstanceIdentifier::generate();
94 auto instance = RemoteCDMInstanceProxy::create(makeWeakPtr(this), privateInstance.releaseNonNull(), identifier);
95 RemoteCDMInstanceConfiguration configuration = instance->configuration();
96 m_factory->addInstance(identifier, WTFMove(instance));
97 completion(identifier, WTFMove(configuration));
98}
