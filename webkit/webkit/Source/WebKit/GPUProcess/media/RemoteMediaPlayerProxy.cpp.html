74RemoteMediaPlayerProxy::RemoteMediaPlayerProxy(RemoteMediaPlayerManagerProxy& manager, MediaPlayerIdentifier identifier, Ref<IPC::Connection>&& connection, MediaPlayerEnums::MediaEngineIdentifier engineIdentifier, RemoteMediaPlayerProxyConfiguration&& configuration)
75 : m_id(identifier)
76 , m_webProcessConnection(WTFMove(connection))
77 , m_manager(makeWeakPtr(manager))
78 , m_engineIdentifier(engineIdentifier)
79 , m_updateCachedStateMessageTimer(RunLoop::main(), this, &RemoteMediaPlayerProxy::timerFired)
80 , m_configuration(configuration)
81 , m_renderingResourcesRequest(ScopedRenderingResourcesRequest::acquire())
82#if !RELEASE_LOG_DISABLED
83 , m_logger(manager.logger())
84#endif
85{
86 m_typesRequiringHardwareSupport = m_configuration.mediaContentTypesRequiringHardwareSupport;
87 m_player = MediaPlayer::create(*this, m_engineIdentifier);
88}

97void RemoteMediaPlayerProxy::invalidate()
98{
99 m_updateCachedStateMessageTimer.stop();
100 m_player->invalidate();
101 if (m_sandboxExtension) {
102 m_sandboxExtension->revoke();
103 m_sandboxExtension = nullptr;
104 }
105 m_renderingResourcesRequest = { };
106}

108void RemoteMediaPlayerProxy::getConfiguration(RemoteMediaPlayerConfiguration& configuration)
109{
110 configuration.engineDescription = m_player->engineDescription();
111 auto maxDuration = m_player->maximumDurationToCacheMediaTime();
112 configuration.maximumDurationToCacheMediaTime = maxDuration ? maxDuration : 0.2;
113 configuration.supportsScanning = m_player->supportsScanning();
114 configuration.supportsFullscreen = m_player->supportsFullscreen();
115 configuration.supportsPictureInPicture = m_player->supportsPictureInPicture();
116 configuration.supportsAcceleratedRendering = m_player->supportsAcceleratedRendering();
117#if ENABLE(WIRELESS_PLAYBACK_TARGET)
118 configuration.canPlayToWirelessPlaybackTarget = m_player->canPlayToWirelessPlaybackTarget();
119#endif
120 configuration.shouldIgnoreIntrinsicSize = m_player->shouldIgnoreIntrinsicSize();
121
122 m_observingTimeChanges = m_player->setCurrentTimeDidChangeCallback([this, weakThis = makeWeakPtr(this)] (auto currentTime) mutable {
123 if (!weakThis)
124 return;
125
126 currentTimeChanged(currentTime);
127 });
128}

130void RemoteMediaPlayerProxy::load(URL&& url, std::optional<SandboxExtension::Handle>&& sandboxExtensionHandle, const ContentType& contentType, const String& keySystem, CompletionHandler<void(RemoteMediaPlayerConfiguration&&)>&& completionHandler)
131{
132 RemoteMediaPlayerConfiguration configuration;
133 if (sandboxExtensionHandle) {
134 m_sandboxExtension = SandboxExtension::create(WTFMove(sandboxExtensionHandle.value()));
135 if (m_sandboxExtension)
136 m_sandboxExtension->consume();
137 else
138 WTFLogAlways("Unable to create sandbox extension for media url.\n");
139 }
140
141 m_player->load(url, contentType, keySystem);
142 getConfiguration(configuration);
143 completionHandler(WTFMove(configuration));
144}

274RefPtr<PlatformMediaResource> RemoteMediaPlayerProxy::requestResource(ResourceRequest&& request, PlatformMediaResourceLoader::LoadOptions options)
275{
276 ASSERT(m_manager && m_manager->gpuConnectionToWebProcess());
277 if (!m_manager || !m_manager->gpuConnectionToWebProcess())
278 return nullptr;
279
280 auto& remoteMediaResourceManager = m_manager->gpuConnectionToWebProcess()->remoteMediaResourceManager();
281 auto remoteMediaResourceIdentifier = RemoteMediaResourceIdentifier::generate();
282 auto remoteMediaResource = RemoteMediaResource::create(remoteMediaResourceManager, *this, remoteMediaResourceIdentifier);
283 remoteMediaResourceManager.addMediaResource(remoteMediaResourceIdentifier, remoteMediaResource);
284
285 m_webProcessConnection->sendWithAsyncReply(Messages::MediaPlayerPrivateRemote::RequestResource(remoteMediaResourceIdentifier, request, options), [remoteMediaResource]() {
286 remoteMediaResource->setReady(true);
287 }, m_id);
288
289 return remoteMediaResource;
290}

985void RemoteMediaPlayerProxy::performTaskAtMediaTime(const MediaTime& taskTime, MonotonicTime messageTime, PerformTaskAtMediaTimeCompletionHandler&& completionHandler)
986{
987 if (m_performTaskAtMediaTimeCompletionHandler) {
988 // A media player is only expected to track one pending task-at-time at once (e.g. see
989 // MediaPlayerPrivateAVFoundationObjC::performTaskAtMediaTime), so cancel the existing
990 // CompletionHandler.
991 auto handler = WTFMove(m_performTaskAtMediaTimeCompletionHandler);
992 handler(std::nullopt, std::nullopt);
993 }
994
995 auto now = MonotonicTime::now();
996 auto transmissionTime = MediaTime::createWithDouble((now - messageTime).value(), 1);
997 auto adjustedTaskTime = taskTime - transmissionTime;
998 auto currentTime = m_player->currentTime();
999 if (adjustedTaskTime <= currentTime) {
1000 completionHandler(currentTime, now);
1001 return;
1002 }
1003
1004 m_performTaskAtMediaTimeCompletionHandler = WTFMove(completionHandler);
1005 m_player->performTaskAtMediaTime([this, weakThis = makeWeakPtr(this)]() mutable {
1006 if (!weakThis || !m_performTaskAtMediaTimeCompletionHandler)
1007 return;
1008
1009 auto completionHandler = WTFMove(m_performTaskAtMediaTimeCompletionHandler);
1010 completionHandler(m_player->currentTime(), MonotonicTime::now());
1011 }, adjustedTaskTime);
1012}
