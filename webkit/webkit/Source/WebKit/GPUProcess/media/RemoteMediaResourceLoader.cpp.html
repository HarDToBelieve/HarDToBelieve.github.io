36using namespace WebCore;
37
38RemoteMediaResourceLoader::RemoteMediaResourceLoader(RemoteMediaPlayerProxy& remoteMediaPlayerProxy)
39 : m_remoteMediaPlayerProxy(makeWeakPtr(remoteMediaPlayerProxy))
40{
41}
42
43RemoteMediaResourceLoader::~RemoteMediaResourceLoader()
44{
45}
46
47RefPtr<PlatformMediaResource> RemoteMediaResourceLoader::requestResource(ResourceRequest&& request, LoadOptions options)
48{
49 if (!m_remoteMediaPlayerProxy)
50 return nullptr;
51
52 return m_remoteMediaPlayerProxy->requestResource(WTFMove(request), options);
53}
54
55void RemoteMediaResourceLoader::sendH2Ping(const URL& url, CompletionHandler<void(Expected<Seconds, ResourceError>&&)>&& completionHandler)
56{
57 if (!m_remoteMediaPlayerProxy)
58 return completionHandler(makeUnexpected(internalError(url)));
59
60 m_remoteMediaPlayerProxy->sendH2Ping(url, WTFMove(completionHandler));
61}
