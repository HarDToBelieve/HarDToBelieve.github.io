49void RemoteMediaResourceManager::addMediaResource(RemoteMediaResourceIdentifier remoteMediaResourceIdentifier, RemoteMediaResource& remoteMediaResource)
50{
51 ASSERT(!m_remoteMediaResources.contains(remoteMediaResourceIdentifier));
52 m_remoteMediaResources.add(remoteMediaResourceIdentifier, &remoteMediaResource);
53}
54
55void RemoteMediaResourceManager::removeMediaResource(RemoteMediaResourceIdentifier remoteMediaResourceIdentifier)
56{
57 ASSERT(m_remoteMediaResources.contains(remoteMediaResourceIdentifier));
58 m_remoteMediaResources.remove(remoteMediaResourceIdentifier);
59}
60
61void RemoteMediaResourceManager::responseReceived(RemoteMediaResourceIdentifier identifier, const ResourceResponse& response, bool didPassAccessControlCheck, CompletionHandler<void(ShouldContinuePolicyCheck)>&& completionHandler)
62{
63 auto* resource = m_remoteMediaResources.get(identifier);
64 if (!resource || !resource->ready()) {
65 completionHandler(ShouldContinuePolicyCheck::No);
66 return;
67 }
68
69 resource->responseReceived(response, didPassAccessControlCheck, WTFMove(completionHandler));
70}
71
72void RemoteMediaResourceManager::redirectReceived(RemoteMediaResourceIdentifier identifier, ResourceRequest&& request, const ResourceResponse& response, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler)
73{
74 auto* resource = m_remoteMediaResources.get(identifier);
75 if (!resource || !resource->ready()) {
76 completionHandler({ });
77 return;
78 }
79
80 resource->redirectReceived(WTFMove(request), response, WTFMove(completionHandler));
81}
