50RemoteSourceBufferProxy::RemoteSourceBufferProxy(GPUConnectionToWebProcess& connectionToWebProcess, RemoteSourceBufferIdentifier identifier, Ref<SourceBufferPrivate>&& sourceBufferPrivate, RemoteMediaPlayerProxy& remoteMediaPlayerProxy)
51 : m_connectionToWebProcess(makeWeakPtr(connectionToWebProcess))
52 , m_identifier(identifier)
53 , m_sourceBufferPrivate(WTFMove(sourceBufferPrivate))
54 , m_remoteMediaPlayerProxy(makeWeakPtr(remoteMediaPlayerProxy))
55{
56 m_connectionToWebProcess->messageReceiverMap().addMessageReceiver(Messages::RemoteSourceBufferProxy::messageReceiverName(), m_identifier.toUInt64(), *this);
57 m_sourceBufferPrivate->setClient(this);
58 m_sourceBufferPrivate->setIsAttached(true);
59}
60
61RemoteSourceBufferProxy::~RemoteSourceBufferProxy()
62{
63 m_sourceBufferPrivate->setIsAttached(false);
64 m_connectionToWebProcess->messageReceiverMap().removeMessageReceiver(Messages::RemoteSourceBufferProxy::messageReceiverName(), m_identifier.toUInt64());
65}

67void RemoteSourceBufferProxy::sourceBufferPrivateDidReceiveInitializationSegment(InitializationSegment&& segment, CompletionHandler<void()>&& completionHandler)
68{
69 if (!m_remoteMediaPlayerProxy) {
70 completionHandler();
71 return;
72 }
73
74 InitializationSegmentInfo segmentInfo;
75 segmentInfo.duration = segment.duration;
76 for (auto& audioTrackInfo : segment.audioTracks) {
77 auto identifier = m_remoteMediaPlayerProxy->addRemoteAudioTrackProxy(*audioTrackInfo.track);
78 segmentInfo.audioTracks.append({ MediaDescriptionInfo(*audioTrackInfo.description), identifier });
79
80 ASSERT(!m_trackIds.contains(identifier));
81 ASSERT(!m_mediaDescriptions.contains(identifier));
82 m_trackIds.add(identifier, audioTrackInfo.track->id());
83 m_mediaDescriptions.add(identifier, *audioTrackInfo.description);
84 }
85
86 for (auto& videoTrackInfo : segment.videoTracks) {
87 auto identifier = m_remoteMediaPlayerProxy->addRemoteVideoTrackProxy(*videoTrackInfo.track);
88 segmentInfo.videoTracks.append({ MediaDescriptionInfo(*videoTrackInfo.description), identifier });
89
90 ASSERT(!m_trackIds.contains(identifier));
91 ASSERT(!m_mediaDescriptions.contains(identifier));
92 m_trackIds.add(identifier, videoTrackInfo.track->id());
93 m_mediaDescriptions.add(identifier, *videoTrackInfo.description);
94 }
95
96 for (auto& textTrackInfo : segment.textTracks) {
97 auto identifier = m_remoteMediaPlayerProxy->addRemoteTextTrackProxy(*textTrackInfo.track);
98 segmentInfo.textTracks.append({ MediaDescriptionInfo(*textTrackInfo.description), identifier });
99
100 ASSERT(!m_trackIds.contains(identifier));
101 ASSERT(!m_mediaDescriptions.contains(identifier));
102 m_trackIds.add(identifier, textTrackInfo.track->id());
103 m_mediaDescriptions.add(identifier, *textTrackInfo.description);
104 }
105
106 if (!m_connectionToWebProcess) {
107 completionHandler();
108 return;
109 }
110
111 m_connectionToWebProcess->connection().sendWithAsyncReply(Messages::SourceBufferPrivateRemote::SourceBufferPrivateDidReceiveInitializationSegment(segmentInfo), WTFMove(completionHandler), m_identifier);
112}

333void RemoteSourceBufferProxy::updateTrackIds(Vector<std::pair<TrackPrivateRemoteIdentifier, TrackPrivateRemoteIdentifier>>&& identifierPairs)
334{
335 Vector<std::pair<AtomString, AtomString>> trackIdPairs;
336
337 for (auto& identifierPair : identifierPairs) {
338 ASSERT(m_trackIds.contains(identifierPair.first));
339 ASSERT(m_trackIds.contains(identifierPair.second));
340
341 auto oldId = m_trackIds.take(identifierPair.first);
342 auto newId = m_trackIds.get(identifierPair.second);
343 trackIdPairs.append(std::make_pair(oldId, newId));
344 }
345
346 if (!trackIdPairs.isEmpty())
347 m_sourceBufferPrivate->updateTrackIds(WTFMove(trackIdPairs));
348}
