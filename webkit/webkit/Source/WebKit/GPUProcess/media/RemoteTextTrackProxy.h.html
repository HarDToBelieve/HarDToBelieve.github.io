49class RemoteTextTrackProxy final
50 : public ThreadSafeRefCounted<RemoteTextTrackProxy, WTF::DestructionThread::Main>
51 , private WebCore::InbandTextTrackPrivateClient {
52public:
53 static Ref<RemoteTextTrackProxy> create(GPUConnectionToWebProcess& connectionToWebProcess, TrackPrivateRemoteIdentifier identifier, WebCore::InbandTextTrackPrivate& trackPrivate, WebCore::MediaPlayerIdentifier mediaPlayerIdentifier)
54 {
55 return adoptRef(*new RemoteTextTrackProxy(connectionToWebProcess, identifier, trackPrivate, mediaPlayerIdentifier));
56 }
57
58 virtual ~RemoteTextTrackProxy();
59
60 TrackPrivateRemoteIdentifier identifier() const { return m_identifier; }
61 void setMode(WebCore::InbandTextTrackPrivate::Mode mode) { m_trackPrivate->setMode(mode); }
62
63private:
64 RemoteTextTrackProxy(GPUConnectionToWebProcess&, TrackPrivateRemoteIdentifier, WebCore::InbandTextTrackPrivate&, WebCore::MediaPlayerIdentifier);

66 // InbandTextTrackPrivateClient

69#if ENABLE(DATACUE_VALUE)

71 virtual void updateDataCue(const MediaTime& start, const MediaTime& end, WebCore::SerializedPlatformDataCue&);
72 virtual void removeDataCue(const MediaTime& start, const MediaTime& end, WebCore::SerializedPlatformDataCue&);
73#endif
74
75 virtual void addGenericCue(WebCore::InbandGenericCue&);
76 virtual void updateGenericCue(WebCore::InbandGenericCue&);
77 virtual void removeGenericCue(WebCore::InbandGenericCue&);
78
79 virtual void parseWebVTTFileHeader(String&&);
80 virtual void parseWebVTTCueData(const uint8_t* data, unsigned length);
81 virtual void parseWebVTTCueData(WebCore::ISOWebVTTCue&&);

84 void idChanged(const AtomString&) final;

87 void willRemove() final;

89 TextTrackPrivateRemoteConfiguration& configuration();

91
92 WeakPtr<GPUConnectionToWebProcess> m_connectionToWebProcess;
93 TrackPrivateRemoteIdentifier m_identifier;
