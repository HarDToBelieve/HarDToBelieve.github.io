26#pragma once
27
28#if ENABLE(GPU_PROCESS)
29
30#include <WebCore/AudioTrackPrivate.h>
31#include <WebCore/VideoTrackPrivate.h>
32#include <wtf/MediaTime.h>
33
34namespace WebKit {
35
36struct TrackPrivateRemoteConfiguration {
37 AtomString trackId;
38 AtomString label;
39 AtomString language;
40 MediaTime startTimeVariance { MediaTime::zeroTime() };
41 int trackIndex;
42
43 bool enabled;
44 WebCore::AudioTrackPrivate::Kind audioKind { WebCore::AudioTrackPrivate::Kind::None };
45
46 bool selected;
47 WebCore::VideoTrackPrivate::Kind videoKind { WebCore::VideoTrackPrivate::Kind::None };
48
49 template<class Encoder>
50 void encode(Encoder& encoder) const
51 {
52 encoder << trackId;
53 encoder << label;
54 encoder << language;
55 encoder << startTimeVariance;
56 encoder << trackIndex;
57 encoder << enabled;
58 encoder << audioKind;
59 encoder << selected;
60 encoder << videoKind;
61 }
62
63 template <class Decoder>
64 static std::optional<TrackPrivateRemoteConfiguration> decode(Decoder& decoder)
65 {
66 std::optional<AtomString> trackId;
67 decoder >> trackId;
68 if (!trackId)
69 return std::nullopt;
70
71 std::optional<AtomString> label;
72 decoder >> label;
73 if (!label)
74 return std::nullopt;
75
76 std::optional<AtomString> language;
77 decoder >> language;
78 if (!language)
79 return std::nullopt;
80
81 std::optional<MediaTime> startTimeVariance;
82 decoder >> startTimeVariance;
83 if (!startTimeVariance)
84 return std::nullopt;
85
86 std::optional<int> trackIndex;
87 decoder >> trackIndex;
88 if (!trackIndex)
89 return std::nullopt;
90
91 std::optional<bool> enabled;
92 decoder >> enabled;
93 if (!enabled)
94 return std::nullopt;
95
96 std::optional<WebCore::AudioTrackPrivate::Kind> audioKind;
97 decoder >> audioKind;
98 if (!audioKind)
99 return std::nullopt;
100
101 std::optional<bool> selected;
102 decoder >> selected;
103 if (!selected)
104 return std::nullopt;
105
106 std::optional<WebCore::VideoTrackPrivate::Kind> videoKind;
107 decoder >> videoKind;
108 if (!videoKind)
109 return std::nullopt;
110
111 return {{
112 WTFMove(*trackId),
113 WTFMove(*label),
114 WTFMove(*language),
115 WTFMove(*startTimeVariance),
116 *trackIndex,
117 *enabled,
118 *audioKind,
119 *selected,
120 *videoKind,
121 }};
122 }
123};
124
125} // namespace WebKit
