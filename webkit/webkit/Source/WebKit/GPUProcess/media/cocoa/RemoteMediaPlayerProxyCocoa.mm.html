40static void setVideoInlineSizeIfPossible(LayerHostingContext& context, const WebCore::FloatSize& size)
41{
42 if (!context.rootLayer() || size.isEmpty())
43 return;
44
45 // We do not want animations here.
46 [CATransaction begin];
47 [CATransaction setDisableActions:YES];
48 [context.rootLayer() setFrame:CGRectMake(0, 0, size.width(), size.height())];
49 [CATransaction commit];
50}

52void RemoteMediaPlayerProxy::prepareForPlayback(bool privateMode, WebCore::MediaPlayerEnums::Preload preload, bool preservesPitch, bool prepareForRendering, float videoContentScale, WebCore::DynamicRangeMode preferredDynamicRangeMode, CompletionHandler<void(std::optional<LayerHostingContextID>&& inlineLayerHostingContextId)>&& completionHandler)
53{
54 m_player->setPrivateBrowsingMode(privateMode);
55 m_player->setPreload(preload);
56 m_player->setPreservesPitch(preservesPitch);
57 m_player->setPreferredDynamicRangeMode(preferredDynamicRangeMode);
58 if (prepareForRendering)
59 m_player->prepareForRendering();
60 m_videoContentScale = videoContentScale;
61 if (!m_inlineLayerHostingContext)
62 m_inlineLayerHostingContext = LayerHostingContext::createForExternalHostingProcess();
63 completionHandler(m_inlineLayerHostingContext->contextID());
64}

66void RemoteMediaPlayerProxy::mediaPlayerFirstVideoFrameAvailable()
67{
68 // Initially the size of the platformLayer may be 0x0 because we do not provide mediaPlayerContentBoxRect() in this class.
69 m_inlineLayerHostingContext->setRootLayer(m_player->platformLayer());
70 setVideoInlineSizeIfPossible(*m_inlineLayerHostingContext, m_videoInlineSize);
71 m_webProcessConnection->send(Messages::MediaPlayerPrivateRemote::FirstVideoFrameAvailable(), m_id);
72}
73
74void RemoteMediaPlayerProxy::setVideoInlineSizeFenced(const WebCore::FloatSize& size, const WTF::MachSendRight& machSendRight)
75{
76 m_inlineLayerHostingContext->setFencePort(machSendRight.sendRight());
77
78 m_videoInlineSize = size;
79 setVideoInlineSizeIfPossible(*m_inlineLayerHostingContext, size);
80}

82void RemoteMediaPlayerProxy::nativeImageForCurrentTime(CompletionHandler<void(std::optional<WTF::MachSendRight>&&)>&& completionHandler)
83{
84 if (!m_player) {
85 completionHandler(std::nullopt);
86 return;
87 }
88
89 auto nativeImage = m_player->nativeImageForCurrentTime();
90 if (!nativeImage) {
91 completionHandler(std::nullopt);
92 return;
93 }
94
95 auto platformImage = nativeImage->platformImage();
96 if (!platformImage) {
97 completionHandler(std::nullopt);
98 return;
99 }
100
101 auto surface = WebCore::IOSurface::createFromImage(platformImage.get());
102 if (!surface) {
103 completionHandler(std::nullopt);
104 return;
105 }
106
107 completionHandler(surface->createSendRight());
108}

110void RemoteMediaPlayerProxy::pixelBufferForCurrentTime(CompletionHandler<void(RetainPtr<CVPixelBufferRef>&&)>&& completionHandler)
111{
112 RetainPtr<CVPixelBufferRef> result;
113 if (m_player)
114 result = m_player->pixelBufferForCurrentTime();
115 completionHandler(WTFMove(result));
116}
