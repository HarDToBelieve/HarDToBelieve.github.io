55class LibWebRTCCodecsProxy : public IPC::Connection::ThreadMessageReceiverRefCounted {
56 WTF_MAKE_FAST_ALLOCATED;
57public:
58 static Ref<LibWebRTCCodecsProxy> create(GPUConnectionToWebProcess& process) { return adoptRef(*new LibWebRTCCodecsProxy(process)); }
59 ~LibWebRTCCodecsProxy();
60
61 void close();
62
63 bool allowsExitUnderMemoryPressure() const;
64
65private:
66 explicit LibWebRTCCodecsProxy(GPUConnectionToWebProcess&);
67
68 // IPC::Connection::ThreadMessageReceiver
69 void dispatchToThread(Function<void()>&&) final;
70
71 // IPC::MessageReceiver
72 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
73 void createH264Decoder(RTCDecoderIdentifier);
74 void createH265Decoder(RTCDecoderIdentifier);
75 void createVP9Decoder(RTCDecoderIdentifier);
76 void releaseDecoder(RTCDecoderIdentifier);
77 void decodeFrame(RTCDecoderIdentifier, uint32_t timeStamp, const IPC::DataReference&);
78 void setFrameSize(RTCDecoderIdentifier, uint16_t width, uint16_t height);
79
80 void createEncoder(RTCEncoderIdentifier, const String&, const Vector<std::pair<String, String>>&, bool useLowLatency);
81 void releaseEncoder(RTCEncoderIdentifier);
82 void initializeEncoder(RTCEncoderIdentifier, uint16_t width, uint16_t height, unsigned startBitrate, unsigned maxBitrate, unsigned minBitrate, uint32_t maxFramerate);
83 void encodeFrame(RTCEncoderIdentifier, WebCore::RemoteVideoSample&&, uint32_t timeStamp, bool shouldEncodeAsKeyFrame);
84 void setEncodeRates(RTCEncoderIdentifier, uint32_t bitRate, uint32_t frameRate);
85 void setRTCLoggingLevel(WTFLogLevel);
86
87 CFDictionaryRef ioSurfacePixelBufferCreationOptions(IOSurfaceRef);
88
89 GPUConnectionToWebProcess& m_gpuConnectionToWebProcess;
90
91 mutable Lock m_lock;
92 HashMap<RTCDecoderIdentifier, webrtc::LocalDecoder> m_decoders WTF_GUARDED_BY_LOCK(m_lock); // Only modified on the libWebRTCCodecsQueue but may get accessed from the main thread.
93 HashMap<RTCEncoderIdentifier, webrtc::LocalEncoder> m_encoders WTF_GUARDED_BY_LOCK(m_lock); // Only modified on the libWebRTCCodecsQueue but may get accessed from the main thread.
94
95 Ref<WorkQueue> m_queue;
96 std::unique_ptr<WebCore::ImageTransferSessionVT> m_imageTransferSession;
97};
