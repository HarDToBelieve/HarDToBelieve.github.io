61void LibWebRTCCodecsProxy::close()
62{
63 m_gpuConnectionToWebProcess.connection().removeThreadMessageReceiver(Messages::LibWebRTCCodecsProxy::messageReceiverName());
64
65 dispatchToThread([this, protectedThis = makeRef(*this)] {
66 Locker locker { m_lock };
67 auto decoders = WTFMove(m_decoders);
68 for (auto decoder : decoders.values())
69 webrtc::releaseLocalDecoder(decoder);
70 auto encoders = WTFMove(m_encoders);
71 for (auto encoder : encoders.values())
72 webrtc::releaseLocalEncoder(encoder);
73 });
74}

76void LibWebRTCCodecsProxy::createH264Decoder(RTCDecoderIdentifier identifier)
77{
78 ASSERT(!isMainRunLoop());
79 Locker locker { m_lock };
80 ASSERT(!m_decoders.contains(identifier));
81 m_decoders.add(identifier, webrtc::createLocalH264Decoder(makeBlockPtr([connection = makeRef(m_gpuConnectionToWebProcess.connection()), identifier](CVPixelBufferRef pixelBuffer, uint32_t timeStampNs, uint32_t timeStamp) {
82 if (auto sample = WebCore::RemoteVideoSample::create(pixelBuffer, MediaTime(timeStampNs, 1)))
83 connection->send(Messages::LibWebRTCCodecs::CompletedDecoding { identifier, timeStamp, *sample }, 0);
84 }).get()));
85}

118// For performance reasons, this function accesses m_decoders without locking. This is safe because this function runs on the libWebRTCCodecsQueue
119// and m_decoders only get modified on this queue.
120void LibWebRTCCodecsProxy::decodeFrame(RTCDecoderIdentifier identifier, uint32_t timeStamp, const IPC::DataReference& data) WTF_IGNORES_THREAD_SAFETY_ANALYSIS
121{
122 ASSERT(!isMainRunLoop());
123 ASSERT(m_decoders.contains(identifier));
124 auto decoder = m_decoders.get(identifier);
125 if (!decoder)
126 return;
127
128 if (webrtc::decodeFrame(decoder, timeStamp, data.data(), data.size()))
129 m_gpuConnectionToWebProcess.connection().send(Messages::LibWebRTCCodecs::FailedDecoding { identifier }, 0);
130}

145void LibWebRTCCodecsProxy::createEncoder(RTCEncoderIdentifier identifier, const String& formatName, const Vector<std::pair<String, String>>& parameters, bool useLowLatency)
146{
147 ASSERT(!isMainRunLoop());
148 Locker locker { m_lock };
149 ASSERT(!m_encoders.contains(identifier));
150
151 std::map<std::string, std::string> rtcParameters;
152 for (auto& parameter : parameters)
153 rtcParameters.emplace(parameter.first.utf8().data(), parameter.second.utf8().data());
154
155 auto* encoder = webrtc::createLocalEncoder(webrtc::SdpVideoFormat { formatName.utf8().data(), rtcParameters }, makeBlockPtr([connection = makeRef(m_gpuConnectionToWebProcess.connection()), identifier](const uint8_t* buffer, size_t size, const webrtc::WebKitEncodedFrameInfo& info) {
156 connection->send(Messages::LibWebRTCCodecs::CompletedEncoding { identifier, IPC::DataReference { buffer, size }, info }, 0);
157 }).get());
158 webrtc::setLocalEncoderLowLatency(encoder, useLowLatency);
159 m_encoders.add(identifier, encoder);
160}

184static inline webrtc::VideoRotation toWebRTCVideoRotation(WebCore::MediaSample::VideoRotation rotation)
185{
186 switch (rotation) {
187 case WebCore::MediaSample::VideoRotation::None:
188 return webrtc::kVideoRotation_0;
189 case WebCore::MediaSample::VideoRotation::UpsideDown:
190 return webrtc::kVideoRotation_180;
191 case WebCore::MediaSample::VideoRotation::Right:
192 return webrtc::kVideoRotation_90;
193 case WebCore::MediaSample::VideoRotation::Left:
194 return webrtc::kVideoRotation_270;
195 }
196 ASSERT_NOT_REACHED();
197 return webrtc::kVideoRotation_0;
198}

200// For performance reasons, this function accesses m_encoders without locking. This is safe because this function runs on the libWebRTCCodecsQueue
201// and m_encoders only get modified on this queue.
202void LibWebRTCCodecsProxy::encodeFrame(RTCEncoderIdentifier identifier, WebCore::RemoteVideoSample&& sample, uint32_t timeStamp, bool shouldEncodeAsKeyFrame) WTF_IGNORES_THREAD_SAFETY_ANALYSIS
203{
204 ASSERT(!isMainRunLoop());
205 ASSERT(m_encoders.contains(identifier));
206 auto encoder = m_encoders.get(identifier);
207 if (!encoder)
208 return;
209
210 if (!m_imageTransferSession || m_imageTransferSession->pixelFormat() != sample.videoFormat())
211 m_imageTransferSession = WebCore::ImageTransferSessionVT::create(sample.videoFormat());
212
213#if !PLATFORM(MACCATALYST)
214 auto pixelBuffer = m_imageTransferSession->createPixelBuffer(sample.surface());
215 webrtc::encodeLocalEncoderFrame(encoder, pixelBuffer.get(), sample.time().toTimeScale(1000000).timeValue(), timeStamp, toWebRTCVideoRotation(sample.rotation()), shouldEncodeAsKeyFrame);
216#endif
217}

231bool LibWebRTCCodecsProxy::allowsExitUnderMemoryPressure() const
232{
233 ASSERT(isMainRunLoop());
234 Locker locker { m_lock };
235 return m_encoders.isEmpty() && m_decoders.isEmpty();
