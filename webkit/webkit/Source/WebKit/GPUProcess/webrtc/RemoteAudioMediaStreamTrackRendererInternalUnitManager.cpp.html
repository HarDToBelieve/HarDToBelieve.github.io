118static WebCore::AudioMediaStreamTrackRendererInternalUnit::RenderCallback renderCallback(RemoteAudioMediaStreamTrackRendererInternalUnitManager::Unit& unit)
119{
120 return [&unit](auto sampleCount, auto& list, auto sampleTime, auto hostTime, auto& flags) {

125RemoteAudioMediaStreamTrackRendererInternalUnitManager::Unit::Unit(AudioMediaStreamTrackRendererInternalUnitIdentifier identifier, Ref<IPC::Connection>&& connection, CompletionHandler<void(const WebCore::CAAudioStreamDescription&, size_t)>&& callback)
126 : m_identifier(identifier)
127 , m_connection(WTFMove(connection))
128 , m_localUnit(WebCore::AudioMediaStreamTrackRendererInternalUnit::createLocalInternalUnit(renderCallback(*this)))
129{
130 m_localUnit->retrieveFormatDescription([weakThis = makeWeakPtr(this), this, callback = WTFMove(callback)](auto&& description) mutable {
131 if (!weakThis || !description) {
132 RELEASE_LOG_IF(!description, WebRTC, "RemoteAudioMediaStreamTrackRendererInternalUnitManager::Unit unable to get format description");
133 callback({ }, 0);
134 return;
135 }
136 m_frameChunkSize = std::max(WebCore::AudioUtilities::renderQuantumSize, WebCore::AudioSession::sharedSession().preferredBufferSize());
137 callback(*description, m_frameChunkSize);
138 });
139}

146void RemoteAudioMediaStreamTrackRendererInternalUnitManager::Unit::start(const SharedMemory::Handle& handle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames, IPC::Semaphore&& semaphore)
147{
148 if (m_isPlaying)
149 stop();
150
151 m_readOffset = 0;
152 m_isPlaying = true;
153 m_ringBuffer = WebCore::CARingBuffer::adoptStorage(makeUniqueRef<ReadOnlySharedRingBufferStorage>(handle), description, numberOfFrames).moveToUniquePtr();
154 m_renderSemaphore = WTFMove(semaphore);
155 m_localUnit->start();
156}
157
158void RemoteAudioMediaStreamTrackRendererInternalUnitManager::Unit::stop()
159{
160 m_isPlaying = false;
161 m_localUnit->stop();
162}

166 m_localUnit->setAudioOutputDevice(deviceId);
167}

169OSStatus RemoteAudioMediaStreamTrackRendererInternalUnitManager::Unit::render(size_t sampleCount, AudioBufferList& list, uint64_t, double, AudioUnitRenderActionFlags&)
170{
171 ASSERT(!isMainRunLoop());
172
173 OSStatus status = -1;
174
175 if (m_ringBuffer->fetchIfHasEnoughData(&list, sampleCount, m_readOffset)) {
176 m_readOffset += sampleCount;
177 status = noErr;
178 }
179
180 for (unsigned i = 0; i < sampleCount; i += m_frameChunkSize)
181 m_renderSemaphore.signal();
182
183 return status;
184}
