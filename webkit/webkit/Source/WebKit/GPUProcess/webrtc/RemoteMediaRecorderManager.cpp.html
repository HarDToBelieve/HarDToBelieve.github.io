50void RemoteMediaRecorderManager::didReceiveRemoteMediaRecorderMessage(IPC::Connection& connection, IPC::Decoder& decoder)
51{
52 if (auto* recorder = m_recorders.get(makeObjectIdentifier<MediaRecorderIdentifierType>(decoder.destinationID())))
53 recorder->didReceiveMessage(connection, decoder);
54}
55
56void RemoteMediaRecorderManager::createRecorder(MediaRecorderIdentifier identifier, bool recordAudio, bool recordVideo, const MediaRecorderPrivateOptions& options, CompletionHandler<void(std::optional<ExceptionData>&&, String&&, unsigned, unsigned)>&& completionHandler)
57{
58 ASSERT(!m_recorders.contains(identifier));
59 auto recorder = RemoteMediaRecorder::create(m_gpuConnectionToWebProcess, identifier, recordAudio, recordVideo, options);
60 if (!recorder)
61 return completionHandler(ExceptionData { NotSupportedError, "Unable to create a recorder with the provided stream"_s }, { }, 0, 0);
62
63 completionHandler({ }, recorder->mimeType(), recorder->audioBitRate(), recorder->videoBitRate());
64 m_recorders.add(identifier, WTFMove(recorder));
65}
66
67void RemoteMediaRecorderManager::releaseRecorder(MediaRecorderIdentifier identifier)
68{
69 m_recorders.remove(identifier);
70 if (allowsExitUnderMemoryPressure())
71 m_gpuConnectionToWebProcess.gpuProcess().tryExitIfUnusedAndUnderMemoryPressure();
72}
73
74bool RemoteMediaRecorderManager::allowsExitUnderMemoryPressure() const
75{
76 return m_recorders.isEmpty();
77}
