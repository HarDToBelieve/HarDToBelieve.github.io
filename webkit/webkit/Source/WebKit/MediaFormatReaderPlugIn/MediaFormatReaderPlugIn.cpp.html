26#include <WebKit/WKMediaFormatReader.h>
27
28extern "C" WK_EXPORT OSStatus CreateMediaFormatReaderInstance(MTPluginByteSourceRef, CFAllocatorRef, CFDictionaryRef, MTPluginFormatReaderRef*);
29
30OSStatus CreateMediaFormatReaderInstance(MTPluginByteSourceRef byteSource, CFAllocatorRef allocator, CFDictionaryRef, MTPluginFormatReaderRef* formatReader)
31{
32 OSStatus status = WKMediaFormatReaderCreate(allocator, formatReader);
33 if (status != noErr)
34 return status;
35 return WKMediaFormatReaderStartOnMainThread(*formatReader, byteSource);
36}
