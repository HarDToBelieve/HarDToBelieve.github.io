56 RegistrableDomain domain;
57 WallTime lastSeen;
58};

60static void pruneResources(HashMap<RegistrableDomain, UniqueRef<ResourceLoadStatistics>>& statisticsMap, Vector<StatisticsLastSeen>& statisticsToPrune, size_t& numberOfEntriesToPrune)
61{
62 if (statisticsToPrune.size() > numberOfEntriesToPrune) {
63 std::sort(statisticsToPrune.begin(), statisticsToPrune.end(), [](const StatisticsLastSeen& a, const StatisticsLastSeen& b) {
64 return a.lastSeen < b.lastSeen;
65 });
66 }
67
68 for (size_t i = 0, end = std::min(numberOfEntriesToPrune, statisticsToPrune.size()); i != end; ++i, --numberOfEntriesToPrune)
69 statisticsMap.remove(statisticsToPrune[i].domain);
70}

92static Vector<WebResourceLoadStatisticsStore::ThirdPartyDataForSpecificFirstParty> getThirdPartyDataForSpecificFirstPartyDomains(const ResourceLoadStatistics& thirdPartyStatistic)
93{
94 Vector<WebResourceLoadStatisticsStore::ThirdPartyDataForSpecificFirstParty> thirdPartyDataForSpecificFirstPartyDomains;
95
96 for (auto firstPartyDomain : thirdPartyStatistic.subframeUnderTopFrameDomains)
97 ensureThirdPartyDataForSpecificFirstPartyDomain(thirdPartyDataForSpecificFirstPartyDomains, firstPartyDomain, thirdPartyStatistic.storageAccessUnderTopFrameDomains.contains(firstPartyDomain));
98 for (auto firstPartyDomain : thirdPartyStatistic.subresourceUnderTopFrameDomains)
99 ensureThirdPartyDataForSpecificFirstPartyDomain(thirdPartyDataForSpecificFirstPartyDomains, firstPartyDomain, thirdPartyStatistic.storageAccessUnderTopFrameDomains.contains(firstPartyDomain));
100 for (auto firstPartyDomain : thirdPartyStatistic.subresourceUniqueRedirectsTo)
101 ensureThirdPartyDataForSpecificFirstPartyDomain(thirdPartyDataForSpecificFirstPartyDomains, firstPartyDomain, thirdPartyStatistic.storageAccessUnderTopFrameDomains.contains(firstPartyDomain));
102
103 return thirdPartyDataForSpecificFirstPartyDomains;
104}

113Vector<WebResourceLoadStatisticsStore::ThirdPartyData> ResourceLoadStatisticsMemoryStore::aggregatedThirdPartyData() const
114{
115 ASSERT(!RunLoop::isMain());
116
117 Vector<WebResourceLoadStatisticsStore::ThirdPartyData> thirdPartyDataList;
118 for (const auto& statistics : m_resourceStatisticsMap.values()) {
119 if (hasBeenThirdParty(statistics) && (thirdPartyCookieBlockingMode() == ThirdPartyCookieBlockingMode::All || statistics->isPrevalentResource))
120 thirdPartyDataList.append(WebResourceLoadStatisticsStore::ThirdPartyData { statistics->registrableDomain, getThirdPartyDataForSpecificFirstPartyDomains(statistics) });
121 }
122 std::sort(thirdPartyDataList.rbegin(), thirdPartyDataList.rend());
123 return thirdPartyDataList;

136unsigned ResourceLoadStatisticsMemoryStore::recursivelyGetAllDomainsThatHaveRedirectedToThisDomain(const ResourceLoadStatistics& resourceStatistic, HashSet<RegistrableDomain>& domainsThatHaveRedirectedTo, unsigned numberOfRecursiveCalls) const
137{
138 ASSERT(!RunLoop::isMain());
139
140 if (numberOfRecursiveCalls >= maxNumberOfRecursiveCallsInRedirectTraceBack) {
141 // Model version 14 invokes a deliberate re-classification of the whole set.
142 if (statisticsModelVersion != 14)
143 ASSERT_NOT_REACHED();
144 RELEASE_LOG(ResourceLoadStatistics, "Hit %u recursive calls in redirect backtrace. Returning early.", maxNumberOfRecursiveCallsInRedirectTraceBack);
145 return numberOfRecursiveCalls;
146 }
147
148 numberOfRecursiveCalls++;
149
150 for (auto& subresourceUniqueRedirectFromDomain : resourceStatistic.subresourceUniqueRedirectsFrom) {
151 auto* statistics = m_resourceStatisticsMap.get(RegistrableDomain { subresourceUniqueRedirectFromDomain });
152 if (!statistics || statistics->isPrevalentResource)
153 continue;
154 if (domainsThatHaveRedirectedTo.add(statistics->registrableDomain).isNewEntry)
155 numberOfRecursiveCalls = recursivelyGetAllDomainsThatHaveRedirectedToThisDomain(*statistics, domainsThatHaveRedirectedTo, numberOfRecursiveCalls);
156 }
157 for (auto& topFrameUniqueRedirectFromDomain : resourceStatistic.topFrameUniqueRedirectsFrom) {
158 auto* statistics = m_resourceStatisticsMap.get(RegistrableDomain { topFrameUniqueRedirectFromDomain });
159 if (!statistics || statistics->isPrevalentResource)
160 continue;
161 if (domainsThatHaveRedirectedTo.add(statistics->registrableDomain).isNewEntry)
162 numberOfRecursiveCalls = recursivelyGetAllDomainsThatHaveRedirectedToThisDomain(*statistics, domainsThatHaveRedirectedTo, numberOfRecursiveCalls);
163 }
164
165 return numberOfRecursiveCalls;
166}

168void ResourceLoadStatisticsMemoryStore::markAsPrevalentIfHasRedirectedToPrevalent(ResourceLoadStatistics& resourceStatistic)
169{
170 ASSERT(!RunLoop::isMain());
171
172 if (resourceStatistic.isPrevalentResource)
173 return;
174
175 for (auto& subresourceDomainRedirectedTo : resourceStatistic.subresourceUniqueRedirectsTo) {
176 auto* statistics = m_resourceStatisticsMap.get(RegistrableDomain { subresourceDomainRedirectedTo });
177 if (statistics && statistics->isPrevalentResource) {
178 setPrevalentResource(resourceStatistic, ResourceLoadPrevalence::High);
179 return;
180 }
181 }
182
183 for (auto& topFrameDomainRedirectedTo : resourceStatistic.topFrameUniqueRedirectsTo) {
184 auto* statistics = m_resourceStatisticsMap.get(RegistrableDomain { topFrameDomainRedirectedTo });
185 if (statistics && statistics->isPrevalentResource) {
186 setPrevalentResource(resourceStatistic, ResourceLoadPrevalence::High);
187 return;
188 }
189 }

200void ResourceLoadStatisticsMemoryStore::classifyPrevalentResources()
201{
202 ASSERT(!RunLoop::isMain());
203
204 for (auto& resourceStatistic : m_resourceStatisticsMap.values()) {
205 if (shouldSkip(resourceStatistic->registrableDomain))
206 continue;
207 if (isPrevalentDueToDebugMode(resourceStatistic))
208 setPrevalentResource(resourceStatistic, ResourceLoadPrevalence::High);
209 else if (!resourceStatistic->isVeryPrevalentResource) {
210 markAsPrevalentIfHasRedirectedToPrevalent(resourceStatistic);
211 auto currentPrevalence = resourceStatistic->isPrevalentResource ? ResourceLoadPrevalence::High : ResourceLoadPrevalence::Low;
212 auto newPrevalence = classifier().calculateResourcePrevalence(resourceStatistic, currentPrevalence);
213 if (newPrevalence != currentPrevalence)
214 setPrevalentResource(resourceStatistic, newPrevalence);
215 }
216 }

230CookieAccess ResourceLoadStatisticsMemoryStore::cookieAccess(const ResourceLoadStatistics& resourceStatistic, const TopFrameDomain& topFrameDomain)
231{
232 bool isPrevalent = resourceStatistic.isPrevalentResource;
233 bool hadUserInteraction = resourceStatistic.hadUserInteraction;
234
235 if (!areAllThirdPartyCookiesBlockedUnder(topFrameDomain) && !isPrevalent)
236 return CookieAccess::BasedOnCookiePolicy;
237
238 if (!hadUserInteraction)
239 return CookieAccess::CannotRequest;
240
241 return CookieAccess::OnlyIfGranted;

244void ResourceLoadStatisticsMemoryStore::hasStorageAccess(const SubFrameDomain& subFrameDomain, const TopFrameDomain& topFrameDomain, std::optional<FrameIdentifier> frameID, PageIdentifier pageID, CompletionHandler<void(bool)>&& completionHandler)
245{
246 ASSERT(!RunLoop::isMain());
247
248 auto& subFrameStatistic = ensureResourceStatisticsForRegistrableDomain(subFrameDomain);
249 switch (cookieAccess(subFrameStatistic, topFrameDomain)) {
250 case CookieAccess::CannotRequest:
251 completionHandler(false);
252 return;
253 case CookieAccess::BasedOnCookiePolicy:
254 RunLoop::main().dispatch([store = makeRef(store()), subFrameDomain = subFrameDomain.isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {
255 store->hasCookies(subFrameDomain, [store, completionHandler = WTFMove(completionHandler)](bool result) mutable {
256 store->statisticsQueue().dispatch([completionHandler = WTFMove(completionHandler), result] () mutable {
257 completionHandler(result);
258 });
259 });
260 });
261 return;
262 case CookieAccess::OnlyIfGranted:
263 // Handled below.
264 break;
265 }
266
267 RunLoop::main().dispatch([store = makeRef(store()), subFrameDomain = subFrameDomain.isolatedCopy(), topFrameDomain = topFrameDomain.isolatedCopy(), frameID, pageID, completionHandler = WTFMove(completionHandler)]() mutable {
268 store->callHasStorageAccessForFrameHandler(subFrameDomain, topFrameDomain, frameID.value(), pageID, [store, completionHandler = WTFMove(completionHandler)](bool result) mutable {
269 store->statisticsQueue().dispatch([completionHandler = WTFMove(completionHandler), result] () mutable {
270 completionHandler(result);
271 });
272 });
273 });
274}

276void ResourceLoadStatisticsMemoryStore::requestStorageAccess(SubFrameDomain&& subFrameDomain, TopFrameDomain&& topFrameDomain, FrameIdentifier frameID, PageIdentifier pageID, StorageAccessScope scope, CompletionHandler<void(StorageAccessStatus)>&& completionHandler)
277{
278 ASSERT(!RunLoop::isMain());
279
280 auto& subFrameStatistic = ensureResourceStatisticsForRegistrableDomain(subFrameDomain);
281 switch (cookieAccess(subFrameStatistic, topFrameDomain)) {
282 case CookieAccess::CannotRequest:
283 if (UNLIKELY(debugLoggingEnabled())) {
284 RELEASE_LOG_INFO(ITPDebug, "Cannot grant storage access to %" PUBLIC_LOG_STRING " since its cookies are blocked in third-party contexts and it has not received user interaction as first-party.", subFrameDomain.string().utf8().data());
285 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] Cannot grant storage access to '"_s, subFrameDomain.string(), "' since its cookies are blocked in third-party contexts and it has not received user interaction as first-party."_s));
286 }
287 completionHandler(StorageAccessStatus::CannotRequestAccess);
288 return;
289
290 case CookieAccess::BasedOnCookiePolicy:
291 if (UNLIKELY(debugLoggingEnabled())) {
292 RELEASE_LOG_INFO(ITPDebug, "No need to grant storage access to %" PUBLIC_LOG_STRING " since its cookies are not blocked in third-party contexts. Note that the underlying cookie policy may still block this third-party from setting cookies.", subFrameDomain.string().utf8().data());
293 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] No need to grant storage access to '"_s, subFrameDomain.string(), "' since its cookies are not blocked in third-party contexts. Note that the underlying cookie policy may still block this third-party from setting cookies."_s));
294 }
295 completionHandler(StorageAccessStatus::HasAccess);
296 return;
297
298 case CookieAccess::OnlyIfGranted:
299 // Handled below.
300 break;
301 }
302
303 auto userWasPromptedEarlier = hasUserGrantedStorageAccessThroughPrompt(subFrameStatistic, topFrameDomain);
304 if (userWasPromptedEarlier == StorageAccessPromptWasShown::No) {
305 if (UNLIKELY(debugLoggingEnabled())) {
306 RELEASE_LOG_INFO(ITPDebug, "About to ask the user whether they want to grant storage access to %" PUBLIC_LOG_STRING " under %" PUBLIC_LOG_STRING " or not.", subFrameDomain.string().utf8().data(), topFrameDomain.string().utf8().data());
307 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] About to ask the user whether they want to grant storage access to '"_s, subFrameDomain.string(), "' under '"_s, topFrameDomain.string(), "' or not."_s));
308 }
309 completionHandler(StorageAccessStatus::RequiresUserPrompt);
310 return;
311 }
312
313 if (userWasPromptedEarlier == StorageAccessPromptWasShown::Yes) {
314 if (UNLIKELY(debugLoggingEnabled())) {
315 RELEASE_LOG_INFO(ITPDebug, "Storage access was granted to %" PUBLIC_LOG_STRING " under %" PUBLIC_LOG_STRING ".", subFrameDomain.string().utf8().data(), topFrameDomain.string().utf8().data());
316 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] Storage access was granted to '"_s, subFrameDomain.string(), "' under '"_s, topFrameDomain.string(), "'."_s));
317 }
318 }
319
320 subFrameStatistic.timesAccessedAsFirstPartyDueToStorageAccessAPI++;
321
322 grantStorageAccessInternal(WTFMove(subFrameDomain), WTFMove(topFrameDomain), frameID, pageID, userWasPromptedEarlier, scope, [completionHandler = WTFMove(completionHandler)] (StorageAccessWasGranted wasGranted) mutable {
323 completionHandler(wasGranted == StorageAccessWasGranted::Yes ? StorageAccessStatus::HasAccess : StorageAccessStatus::CannotRequestAccess);
324 });
325}

355void ResourceLoadStatisticsMemoryStore::grantStorageAccessInternal(SubFrameDomain&& subFrameDomain, TopFrameDomain&& topFrameDomain, std::optional<FrameIdentifier> frameID, PageIdentifier pageID, StorageAccessPromptWasShown promptWasShownNowOrEarlier, StorageAccessScope scope, CompletionHandler<void(StorageAccessWasGranted)>&& completionHandler)
356{
357 ASSERT(!RunLoop::isMain());
358
359 if (subFrameDomain == topFrameDomain) {
360 completionHandler(StorageAccessWasGranted::Yes);
361 return;
362 }
363
364 if (promptWasShownNowOrEarlier == StorageAccessPromptWasShown::Yes) {
365 auto& subFrameStatistic = ensureResourceStatisticsForRegistrableDomain(subFrameDomain);
366 ASSERT(subFrameStatistic.hadUserInteraction);
367 ASSERT(subFrameStatistic.storageAccessUnderTopFrameDomains.contains(topFrameDomain));
368 subFrameStatistic.mostRecentUserInteractionTime = WallTime::now();
369 }
370
371 RunLoop::main().dispatch([subFrameDomain = subFrameDomain.isolatedCopy(), topFrameDomain = topFrameDomain.isolatedCopy(), frameID, pageID, store = makeRef(store()), scope, completionHandler = WTFMove(completionHandler)]() mutable {
372 store->callGrantStorageAccessHandler(subFrameDomain, topFrameDomain, frameID, pageID, scope, [completionHandler = WTFMove(completionHandler), store](StorageAccessWasGranted wasGranted) mutable {
373 store->statisticsQueue().dispatch([wasGranted, completionHandler = WTFMove(completionHandler)] () mutable {
374 completionHandler(wasGranted);
375 });
376 });
377 });

390Vector<RegistrableDomain> ResourceLoadStatisticsMemoryStore::ensurePrevalentResourcesForDebugMode()
391{
392 ASSERT(!RunLoop::isMain());
393
394 if (!debugModeEnabled())
395 return { };
396
397 Vector<RegistrableDomain> domainsToBlock;
398 domainsToBlock.reserveInitialCapacity(2);
399
400 auto& staticSesourceStatistic = ensureResourceStatisticsForRegistrableDomain(debugStaticPrevalentResource());
401 setPrevalentResource(staticSesourceStatistic, ResourceLoadPrevalence::High);
402 domainsToBlock.uncheckedAppend(debugStaticPrevalentResource());
403
404 if (!debugManualPrevalentResource().isEmpty()) {
405 auto& manualResourceStatistic = ensureResourceStatisticsForRegistrableDomain(debugManualPrevalentResource());
406 setPrevalentResource(manualResourceStatistic, ResourceLoadPrevalence::High);
407 domainsToBlock.uncheckedAppend(debugManualPrevalentResource());
408
409 if (debugLoggingEnabled()) {
410 RELEASE_LOG_INFO(ITPDebug, "Did set %" PUBLIC_LOG_STRING " as prevalent resource for the purposes of ITP Debug Mode.", debugManualPrevalentResource().string().utf8().data());
411 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] Did set '"_s, debugManualPrevalentResource().string(), "' as prevalent resource for the purposes of ITP Debug Mode."_s));
412 }
413 }
414
415 return domainsToBlock;
416}

418void ResourceLoadStatisticsMemoryStore::logFrameNavigation(const RegistrableDomain& targetDomain, const RegistrableDomain& topFrameDomain, const RegistrableDomain& sourceDomain, bool isRedirect, bool isMainFrame, Seconds delayAfterMainFrameDocumentLoad, bool wasPotentiallyInitiatedByUser)
419{
420 ASSERT(!RunLoop::isMain());
421
422 bool areTargetAndTopFrameDomainsSameSite = targetDomain == topFrameDomain;
423 bool areTargetAndSourceDomainsSameSite = targetDomain == sourceDomain;
424
425 bool statisticsWereUpdated = false;
426 if (!isMainFrame && !(areTargetAndTopFrameDomainsSameSite || areTargetAndSourceDomainsSameSite)) {
427 auto& targetStatistics = ensureResourceStatisticsForRegistrableDomain(targetDomain);
428 targetStatistics.lastSeen = ResourceLoadStatistics::reduceTimeResolution(WallTime::now());
429 if (targetStatistics.subframeUnderTopFrameDomains.add(topFrameDomain).isNewEntry)
430 statisticsWereUpdated = true;
431 }
432
433 if (!areTargetAndSourceDomainsSameSite) {
434 if (isMainFrame) {
435 auto& redirectingDomainStatistics = ensureResourceStatisticsForRegistrableDomain(sourceDomain);
436 bool wasNavigatedAfterShortDelayWithoutUserInteraction = !wasPotentiallyInitiatedByUser && delayAfterMainFrameDocumentLoad < parameters().minDelayAfterMainFrameDocumentLoadToNotBeARedirect;
437 if (isRedirect || wasNavigatedAfterShortDelayWithoutUserInteraction) {
438 if (redirectingDomainStatistics.topFrameUniqueRedirectsTo.add(targetDomain).isNewEntry)
439 statisticsWereUpdated = true;
440 if (isRedirect && redirectingDomainStatistics.topFrameUniqueRedirectsToSinceSameSiteStrictEnforcement.add(targetDomain).isNewEntry) {
441 statisticsWereUpdated = true;
442
443 if (UNLIKELY(debugLoggingEnabled())) {
444 RELEASE_LOG_INFO(ITPDebug, "Did set %" PUBLIC_LOG_STRING " as making a top frame redirect to %" PUBLIC_LOG_STRING ".", sourceDomain.string().utf8().data(), targetDomain.string().utf8().data());
445 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("Did set '", sourceDomain.string(), "' as making a top frame redirect to '", targetDomain.string(), "'."));
446 }
447 }
448 auto& targetStatistics = ensureResourceStatisticsForRegistrableDomain(targetDomain);
449 if (targetStatistics.topFrameUniqueRedirectsFrom.add(sourceDomain).isNewEntry)
450 statisticsWereUpdated = true;
451 }
452 } else if (isRedirect) {
453 auto& redirectingDomainStatistics = ensureResourceStatisticsForRegistrableDomain(sourceDomain);
454 if (redirectingDomainStatistics.subresourceUniqueRedirectsTo.add(targetDomain).isNewEntry)
455 statisticsWereUpdated = true;
456 auto& targetStatistics = ensureResourceStatisticsForRegistrableDomain(targetDomain);
457 if (targetStatistics.subresourceUniqueRedirectsFrom.add(sourceDomain).isNewEntry)
458 statisticsWereUpdated = true;
459 }
460 }
461
462 if (statisticsWereUpdated)
463 scheduleStatisticsProcessingRequestIfNecessary();
464}

517void ResourceLoadStatisticsMemoryStore::setPrevalentResource(ResourceLoadStatistics& resourceStatistic, ResourceLoadPrevalence newPrevalence)
518{
519 ASSERT(!RunLoop::isMain());
520
521 if (shouldSkip(resourceStatistic.registrableDomain))
522 return;
523
524 resourceStatistic.isPrevalentResource = true;
525 resourceStatistic.isVeryPrevalentResource = newPrevalence == ResourceLoadPrevalence::VeryHigh;
526 HashSet<RegistrableDomain> domainsThatHaveRedirectedTo;
527 recursivelyGetAllDomainsThatHaveRedirectedToThisDomain(resourceStatistic, domainsThatHaveRedirectedTo, 0);
528 for (auto& domain : domainsThatHaveRedirectedTo) {
529 auto* statistics = m_resourceStatisticsMap.get(domain);
530 if (!statistics)
531 continue;
532 ASSERT(!statistics->isPrevalentResource);
533 statistics->isPrevalentResource = true;
534 }

730void ResourceLoadStatisticsMemoryStore::mergeWithDataFromDecoder(KeyedDecoder& decoder)
731{
732 ASSERT(!RunLoop::isMain());
733
734 unsigned versionOnDisk;
735 if (!decoder.decodeUInt32("version", versionOnDisk))
736 return;
737
738 if (versionOnDisk > statisticsModelVersion) {
739 WTFLogAlways("Found resource load statistics on disk with model version %u whereas the highest supported version is %u. Resetting.", versionOnDisk, statisticsModelVersion);
740 return;
741 }
742
743 double endOfGrandfatheringTimestamp;
744 if (decoder.decodeDouble("endOfGrandfatheringTimestamp", endOfGrandfatheringTimestamp))
745 setEndOfGrandfatheringTimestamp(WallTime::fromRawSeconds(endOfGrandfatheringTimestamp));
746 else
747 clearEndOfGrandfatheringTimeStamp();
748
749 Vector<ResourceLoadStatistics> loadedStatistics;
750 bool succeeded = decoder.decodeObjects("browsingStatistics", loadedStatistics, [versionOnDisk](KeyedDecoder& decoderInner, ResourceLoadStatistics& statistics) {
751 return statistics.decode(decoderInner, versionOnDisk);
752 });
753
754 if (!succeeded)
755 return;
756
757 mergeStatistics(WTFMove(loadedStatistics));
758 updateCookieBlocking([]() { });
759
760 Vector<OperatingDate> operatingDates;
761 succeeded = decoder.decodeObjects("operatingDates", operatingDates, [](KeyedDecoder& decoder, OperatingDate& date) {
762 double value;
763 if (!decoder.decodeDouble("date", value))
764 return false;
765
766 date = OperatingDate::fromWallTime(WallTime::fromRawSeconds(value));
767 return true;
768 });
769
770 if (!succeeded)
771 return;
772
773 mergeOperatingDates(WTFMove(operatingDates));
774}

820void ResourceLoadStatisticsMemoryStore::updateCookieBlocking(CompletionHandler<void()>&& completionHandler)
821{
822 ASSERT(!RunLoop::isMain());
823
824 Vector<RegistrableDomain> domainsToBlockAndDeleteCookiesFor;
825 Vector<RegistrableDomain> domainsToBlockButKeepCookiesFor;
826 Vector<RegistrableDomain> domainsWithUserInteractionAsFirstParty;
827 for (auto& resourceStatistic : m_resourceStatisticsMap.values()) {
828 if (hasHadUnexpiredRecentUserInteraction(resourceStatistic, OperatingDatesWindow::Long)) {
829 if (resourceStatistic->isPrevalentResource)
830 domainsToBlockButKeepCookiesFor.append(resourceStatistic->registrableDomain.isolatedCopy());
831 domainsWithUserInteractionAsFirstParty.append(resourceStatistic->registrableDomain);
832 } else if (resourceStatistic->isPrevalentResource)
833 domainsToBlockAndDeleteCookiesFor.append(resourceStatistic->registrableDomain);
834 }
835
836 if (domainsToBlockAndDeleteCookiesFor.isEmpty() && domainsToBlockButKeepCookiesFor.isEmpty() && domainsWithUserInteractionAsFirstParty.isEmpty() && !debugModeEnabled()) {
837 completionHandler();
838 return;
839 }
840
841 RegistrableDomainsToBlockCookiesFor domainsToBlock { domainsToBlockAndDeleteCookiesFor, domainsToBlockButKeepCookiesFor, domainsWithUserInteractionAsFirstParty, { } };
842
843 if (debugLoggingEnabled() && (!domainsToBlockAndDeleteCookiesFor.isEmpty() || !domainsToBlockButKeepCookiesFor.isEmpty()))
844 debugLogDomainsInBatches("Applying cross-site tracking restrictions", domainsToBlock);
845
846 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this), store = makeRef(store()), domainsToBlock = crossThreadCopy(domainsToBlock), completionHandler = WTFMove(completionHandler)] () mutable {
847 store->callUpdatePrevalentDomainsToBlockCookiesForHandler(domainsToBlock, [weakThis = WTFMove(weakThis), store, completionHandler = WTFMove(completionHandler)]() mutable {
848 store->statisticsQueue().dispatch([weakThis = WTFMove(weakThis), completionHandler = WTFMove(completionHandler)]() mutable {
849 completionHandler();
850
851 if (!weakThis)
852 return;
853
854 if (UNLIKELY(weakThis->debugLoggingEnabled())) {
855 RELEASE_LOG_INFO(ITPDebug, "Done applying cross-site tracking restrictions.");
856 weakThis->debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, "[ITP] Done applying cross-site tracking restrictions."_s);
857 }
858 });
859 });
860 });
861}

940RegistrableDomainsToDeleteOrRestrictWebsiteDataFor ResourceLoadStatisticsMemoryStore::registrableDomainsToDeleteOrRestrictWebsiteDataFor()
941{
942 ASSERT(!RunLoop::isMain());
943
944 bool shouldCheckForGrandfathering = endOfGrandfatheringTimestamp() > WallTime::now();
945 bool shouldClearGrandfathering = !shouldCheckForGrandfathering && endOfGrandfatheringTimestamp();
946
947 if (shouldClearGrandfathering)
948 clearEndOfGrandfatheringTimeStamp();
949
950 auto now = WallTime::now();
951 auto oldestUserInteraction = now;
952 RegistrableDomainsToDeleteOrRestrictWebsiteDataFor toDeleteOrRestrictFor;
953 for (auto& statistic : m_resourceStatisticsMap.values()) {
954 if (shouldExemptFromWebsiteDataDeletion(statistic->registrableDomain))
955 continue;
956 if (auto mostRecentUserInteractionTime = this->mostRecentUserInteractionTime(statistic))
957 oldestUserInteraction = std::min(oldestUserInteraction, *mostRecentUserInteractionTime);
958 if (shouldRemoveAllWebsiteDataFor(statistic, shouldCheckForGrandfathering)) {
959 toDeleteOrRestrictFor.domainsToDeleteAllCookiesFor.append(statistic->registrableDomain);
960 toDeleteOrRestrictFor.domainsToDeleteAllNonCookieWebsiteDataFor.append(statistic->registrableDomain);
961 } else {
962 if (shouldRemoveAllButCookiesFor(statistic, shouldCheckForGrandfathering)) {
963 toDeleteOrRestrictFor.domainsToDeleteAllNonCookieWebsiteDataFor.append(statistic->registrableDomain);
964 statistic->gotLinkDecorationFromPrevalentResource = false;
965 }
966 if (shouldEnforceSameSiteStrictFor(statistic, shouldCheckForGrandfathering)) {
967 toDeleteOrRestrictFor.domainsToEnforceSameSiteStrictFor.append(statistic->registrableDomain);
968
969 if (UNLIKELY(debugLoggingEnabled())) {
970 RELEASE_LOG_INFO(ITPDebug, "Scheduled %" PUBLIC_LOG_STRING " to have its cookies set to SameSite=strict.", statistic->registrableDomain.string().utf8().data());
971 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("Scheduled '", statistic->registrableDomain.string(), "' to have its cookies set to SameSite=strict'."));
972 }
973 }
974 }
975
976 if (shouldClearGrandfathering && statistic->grandfathered)
977 statistic->grandfathered = false;
978 }
979
980 // Give the user enough time to interact with websites until we remove non-cookie website data.
981 if (!parameters().isRunningTest && now - oldestUserInteraction < parameters().minimumTimeBetweenDataRecordsRemoval)
982 toDeleteOrRestrictFor.domainsToDeleteAllNonCookieWebsiteDataFor.clear();
983
984 return toDeleteOrRestrictFor;
985}

987void ResourceLoadStatisticsMemoryStore::pruneStatisticsIfNeeded()
988{
989 ASSERT(!RunLoop::isMain());
990
991 if (m_resourceStatisticsMap.size() <= parameters().maxStatisticsEntries)
992 return;
993
994 ASSERT(parameters().pruneEntriesDownTo <= parameters().maxStatisticsEntries);
995
996 size_t numberOfEntriesLeftToPrune = m_resourceStatisticsMap.size() - parameters().pruneEntriesDownTo;
997 ASSERT(numberOfEntriesLeftToPrune);
998
999 Vector<StatisticsLastSeen> resourcesToPrunePerImportance[maxImportance + 1];
1000 for (auto& resourceStatistic : m_resourceStatisticsMap.values())
1001 resourcesToPrunePerImportance[computeImportance(resourceStatistic)].append({ resourceStatistic->registrableDomain, resourceStatistic->lastSeen });
1002
1003 for (unsigned importance = 0; numberOfEntriesLeftToPrune && importance <= maxImportance; ++importance)
1004 pruneResources(m_resourceStatisticsMap, resourcesToPrunePerImportance[importance], numberOfEntriesLeftToPrune);
1005
1006 ASSERT(!numberOfEntriesLeftToPrune);
1007}

1063Vector<OperatingDate> ResourceLoadStatisticsMemoryStore::mergeOperatingDates(const Vector<OperatingDate>& existingDates, Vector<OperatingDate>&& newDates)
1064{
1065 if (existingDates.isEmpty())
1066 return WTFMove(newDates);
1067
1068 Vector<OperatingDate> mergedDates(existingDates.size() + newDates.size());
1069
1070 // Merge the two sorted vectors of dates.
1071 std::merge(existingDates.begin(), existingDates.end(), newDates.begin(), newDates.end(), mergedDates.begin());
1072 // Remove duplicate dates.
1073 removeRepeatedElements(mergedDates);
1074
1075 // Drop old dates until the Vector size reaches operatingDatesWindowLong.
1076 while (mergedDates.size() > operatingDatesWindowLong)
1077 mergedDates.remove(0);
1078
1079 return mergedDates;
1080}

1103bool ResourceLoadStatisticsMemoryStore::hasStatisticsExpired(WallTime mostRecentUserInteractionTime, OperatingDatesWindow operatingDatesWindow) const
1104{
1105 ASSERT(!RunLoop::isMain());
1106
1107 unsigned operatingDatesWindowInDays = 0;
1108 switch (operatingDatesWindow) {
1109 case OperatingDatesWindow::Long:
1110 operatingDatesWindowInDays = operatingDatesWindowLong;
1111 break;
1112 case OperatingDatesWindow::Short:
1113 operatingDatesWindowInDays = operatingDatesWindowShort;
1114 break;
1115 case OperatingDatesWindow::ForLiveOnTesting:
1116 return WallTime::now() > mostRecentUserInteractionTime + operatingTimeWindowForLiveOnTesting;
1117 case OperatingDatesWindow::ForReproTesting:
1118 return true;
1119 }
1120
1121 if (m_operatingDates.size() >= operatingDatesWindowInDays) {
1122 if (OperatingDate::fromWallTime(mostRecentUserInteractionTime) < m_operatingDates.first())
1123 return true;
1124 }
1125
1126 // If we don't meet the real criteria for an expired statistic, check the user setting for a tighter restriction (mainly for testing).
1127 if (this->parameters().timeToLiveUserInteraction) {
1128 if (WallTime::now() > mostRecentUserInteractionTime + this->parameters().timeToLiveUserInteraction.value())
1129 return true;
1130 }
1131
1132 return false;
1133}

1140void ResourceLoadStatisticsMemoryStore::insertExpiredStatisticForTesting(const RegistrableDomain& domain, unsigned numberOfOperatingDaysPassed, bool hasUserInteraction, bool isScheduledForAllButCookieDataRemoval, bool isPrevalent)
1141{
1142 // Populate the Operating Dates table with enough days to require pruning.
1143 double daysAgoInSeconds = 0;
1144 for (unsigned i = 1; i <= numberOfOperatingDaysPassed; i++) {
1145 double daysToSubtract = Seconds::fromHours(24 * i).value();
1146 daysAgoInSeconds = WallTime::now().secondsSinceEpoch().value() - daysToSubtract;
1147 auto dateToInsert = OperatingDate::fromWallTime(WallTime::fromRawSeconds(daysAgoInSeconds));
1148 m_operatingDates.append(OperatingDate(dateToInsert.year(), dateToInsert.month(), dateToInsert.monthDay()));
1149 }
