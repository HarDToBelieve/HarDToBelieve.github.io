54static String domainsToString(const Vector<RegistrableDomain>& domains)
55{
56 StringBuilder builder;
57 for (auto& domain : domains)
58 builder.append(builder.isEmpty() ? "" : ", ", domain.string());
59 return builder.toString();
60}
61
62static String domainsToString(const RegistrableDomainsToDeleteOrRestrictWebsiteDataFor& domainsToRemoveOrRestrictWebsiteDataFor)
63{
64 StringBuilder builder;
65 for (auto& domain : domainsToRemoveOrRestrictWebsiteDataFor.domainsToDeleteAllCookiesFor)
66 builder.append(builder.isEmpty() ? "" : ", ", domain.string(), "(all data)");
67 for (auto& domain : domainsToRemoveOrRestrictWebsiteDataFor.domainsToDeleteAllButHttpOnlyCookiesFor)
68 builder.append(builder.isEmpty() ? "" : ", ", domain.string(), "(all but HttpOnly cookies)");
69 for (auto& domain : domainsToRemoveOrRestrictWebsiteDataFor.domainsToDeleteAllNonCookieWebsiteDataFor)
70 builder.append(builder.isEmpty() ? "" : ", ", domain.string(), "(all but cookies)");
71 return builder.toString();
72}

74OperatingDate OperatingDate::fromWallTime(WallTime time)
75{
76 double ms = time.secondsSinceEpoch().milliseconds();
77 int year = msToYear(ms);
78 int yearDay = dayInYear(ms, year);
79 int month = monthFromDayInYear(yearDay, isLeapYear(year));
80 int monthDay = dayInMonthFromDayInYear(yearDay, isLeapYear(year));
81 return OperatingDate { year, month, monthDay };
82}
83
84OperatingDate OperatingDate::today()
85{
86 return OperatingDate::fromWallTime(WallTime::now());
87}
88
89Seconds OperatingDate::secondsSinceEpoch() const
90{
91 return Seconds { dateToDaysFrom1970(m_year, m_month, m_monthDay) * secondsPerDay };
92}
93
94bool OperatingDate::operator==(const OperatingDate& other) const
95{
96 return m_monthDay == other.m_monthDay && m_month == other.m_month && m_year == other.m_year;
97}
98
99bool OperatingDate::operator<(const OperatingDate& other) const
100{
101 return secondsSinceEpoch() < other.secondsSinceEpoch();
102}
103
104bool OperatingDate::operator<=(const OperatingDate& other) const
105{
106 return secondsSinceEpoch() <= other.secondsSinceEpoch();
107}

122unsigned ResourceLoadStatisticsStore::computeImportance(const ResourceLoadStatistics& resourceStatistic)
123{
124 unsigned importance = ResourceLoadStatisticsStore::maxImportance;
125 if (!resourceStatistic.isPrevalentResource)
126 importance -= 1;
127 if (!resourceStatistic.hadUserInteraction)
128 importance -= 2;
129 return importance;
130}

157void ResourceLoadStatisticsStore::removeDataRecords(CompletionHandler<void()>&& completionHandler)
158{
159 ASSERT(!RunLoop::isMain());
160
161 if (!shouldRemoveDataRecords()) {
162 completionHandler();
163 return;
164 }
165
166#if ENABLE(NETSCAPE_PLUGIN_API)
167 m_activePluginTokens.clear();
168 for (const auto& plugin : PluginProcessManager::singleton().pluginProcesses())
169 m_activePluginTokens.add(plugin->pluginProcessToken());
170#endif
171
172 auto domainsToDeleteOrRestrictWebsiteDataFor = registrableDomainsToDeleteOrRestrictWebsiteDataFor();
173 if (domainsToDeleteOrRestrictWebsiteDataFor.isEmpty()) {
174 completionHandler();
175 return;
176 }
177
178 if (UNLIKELY(m_debugLoggingEnabled)) {
179 RELEASE_LOG_INFO(ITPDebug, "About to remove data records for %" PUBLIC_LOG_STRING ".", domainsToString(domainsToDeleteOrRestrictWebsiteDataFor).utf8().data());
180 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] About to remove data records for: ["_s, domainsToString(domainsToDeleteOrRestrictWebsiteDataFor), "]."_s));
181 }
182
183 setDataRecordsBeingRemoved(true);
184
185 RunLoop::main().dispatch([store = makeRef(m_store), domainsToDeleteOrRestrictWebsiteDataFor = crossThreadCopy(domainsToDeleteOrRestrictWebsiteDataFor), completionHandler = WTFMove(completionHandler), weakThis = makeWeakPtr(*this), shouldNotifyPagesWhenDataRecordsWereScanned = m_parameters.shouldNotifyPagesWhenDataRecordsWereScanned, workQueue = m_workQueue] () mutable {
186 store->deleteAndRestrictWebsiteDataForRegistrableDomains(WebResourceLoadStatisticsStore::monitoredDataTypes(), WTFMove(domainsToDeleteOrRestrictWebsiteDataFor), shouldNotifyPagesWhenDataRecordsWereScanned, [completionHandler = WTFMove(completionHandler), weakThis = WTFMove(weakThis), workQueue](const HashSet<RegistrableDomain>& domainsWithDeletedWebsiteData) mutable {
187 workQueue->dispatch([domainsWithDeletedWebsiteData = crossThreadCopy(domainsWithDeletedWebsiteData), completionHandler = WTFMove(completionHandler), weakThis = WTFMove(weakThis)] () mutable {
188 if (!weakThis) {
189 completionHandler();
190 return;
191 }
192
193 weakThis->incrementRecordsDeletedCountForDomains(WTFMove(domainsWithDeletedWebsiteData));
194 weakThis->setDataRecordsBeingRemoved(false);
195
196 auto dataRecordRemovalCompletionHandlers = WTFMove(weakThis->m_dataRecordRemovalCompletionHandlers);
197 completionHandler();
198
199 for (auto& dataRecordRemovalCompletionHandler : dataRecordRemovalCompletionHandlers)
200 dataRecordRemovalCompletionHandler();
201
202 if (UNLIKELY(weakThis->m_debugLoggingEnabled)) {
203 RELEASE_LOG_INFO(ITPDebug, "Done removing data records.");
204 weakThis->debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, "[ITP] Done removing data records"_s);
205 }
206 });
207 });
208 });
209}

211void ResourceLoadStatisticsStore::processStatisticsAndDataRecords()
212{
213 ASSERT(!RunLoop::isMain());
214
215 if (parameters().isRunningTest && !m_parameters.shouldNotifyPagesWhenDataRecordsWereScanned)
216 return;
217
218 if (m_parameters.shouldClassifyResourcesBeforeDataRecordsRemoval)
219 classifyPrevalentResources();
220
221 removeDataRecords([this, weakThis = makeWeakPtr(*this)] () mutable {
222 ASSERT(!RunLoop::isMain());
223 if (!weakThis)
224 return;
225
226 pruneStatisticsIfNeeded();
227
228 logTestingEvent("Storage Synced"_s);
229
230 if (!m_parameters.shouldNotifyPagesWhenDataRecordsWereScanned)
231 return;
232
233 RunLoop::main().dispatch([store = makeRef(m_store)] {
234 store->notifyResourceLoadStatisticsProcessed();
235 });
236 });
237}

239void ResourceLoadStatisticsStore::grandfatherExistingWebsiteData(CompletionHandler<void()>&& callback)
240{
241 ASSERT(!RunLoop::isMain());
242
243 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this), callback = WTFMove(callback), shouldNotifyPagesWhenDataRecordsWereScanned = m_parameters.shouldNotifyPagesWhenDataRecordsWereScanned, workQueue = m_workQueue, store = makeRef(m_store)] () mutable {
244 store->registrableDomainsWithWebsiteData(WebResourceLoadStatisticsStore::monitoredDataTypes(), shouldNotifyPagesWhenDataRecordsWereScanned, [weakThis = WTFMove(weakThis), callback = WTFMove(callback), workQueue] (HashSet<RegistrableDomain>&& domainsWithWebsiteData) mutable {
245 workQueue->dispatch([weakThis = WTFMove(weakThis), domainsWithWebsiteData = crossThreadCopy(domainsWithWebsiteData), callback = WTFMove(callback)] () mutable {
246 if (!weakThis) {
247 callback();
248 return;
249 }
250
251 weakThis->grandfatherDataForDomains(domainsWithWebsiteData);
252 weakThis->m_endOfGrandfatheringTimestamp = WallTime::now() + weakThis->m_parameters.grandfatheringTime;
253 callback();
254 weakThis->logTestingEvent("Grandfathered"_s);
255 });
256 });
257 });
258}

296void ResourceLoadStatisticsStore::scheduleStatisticsProcessingRequestIfNecessary()
297{
298 ASSERT(!RunLoop::isMain());
299
300 m_pendingStatisticsProcessingRequestIdentifier = ++m_lastStatisticsProcessingRequestIdentifier;
301 m_workQueue->dispatchAfter(minimumStatisticsProcessingInterval, [this, weakThis = makeWeakPtr(*this), statisticsProcessingRequestIdentifier = *m_pendingStatisticsProcessingRequestIdentifier] {
302 if (!weakThis)
303 return;
304
305 if (!m_pendingStatisticsProcessingRequestIdentifier || *m_pendingStatisticsProcessingRequestIdentifier != statisticsProcessingRequestIdentifier) {
306 // This request has been canceled.
307 return;
308 }
309
310 updateCookieBlocking([]() { });
311 processStatisticsAndDataRecords();
312 });
313}

387bool ResourceLoadStatisticsStore::shouldRemoveDataRecords() const
388{
389 ASSERT(!RunLoop::isMain());
390
391 if (m_dataRecordsBeingRemoved)
392 return false;
393
394#if ENABLE(NETSCAPE_PLUGIN_API)
395 for (const auto& plugin : PluginProcessManager::singleton().pluginProcesses()) {
396 if (!m_activePluginTokens.contains(plugin->pluginProcessToken()))
397 return true;
398 }
399#endif
400
401 return !m_lastTimeDataRecordsWereRemoved || MonotonicTime::now() >= (m_lastTimeDataRecordsWereRemoved + m_parameters.minimumTimeBetweenDataRecordsRemoval) || parameters().isRunningTest;
402}

501void ResourceLoadStatisticsStore::debugLogDomainsInBatches(const char* action, const RegistrableDomainsToBlockCookiesFor& domainsToBlock)
502{
503 ASSERT(debugLoggingEnabled());
504
505 Vector<RegistrableDomain> domains;
506 domains.appendVector(domainsToBlock.domainsToBlockAndDeleteCookiesFor);
507 domains.appendVector(domainsToBlock.domainsToBlockButKeepCookiesFor);
508 if (domains.isEmpty())
509 return;
510
511 debugBroadcastConsoleMessage(MessageSource::ITPDebug, MessageLevel::Info, makeString("[ITP] "_s, action, " to: ["_s, domainsToString(domains), "]."_s));
512
513 static const auto maxNumberOfDomainsInOneLogStatement = 50;
514
515 if (domains.size() <= maxNumberOfDomainsInOneLogStatement) {
516 RELEASE_LOG_INFO(ITPDebug, "%" PUBLIC_LOG_STRING " to: %" PUBLIC_LOG_STRING ".", action, domainsToString(domains).utf8().data());
517 return;
518 }
519
520 Vector<RegistrableDomain> batch;
521 batch.reserveInitialCapacity(maxNumberOfDomainsInOneLogStatement);
522 auto batchNumber = 1;
523#if !RELEASE_LOG_DISABLED
524 unsigned numberOfBatches = std::ceil(domains.size() / static_cast<float>(maxNumberOfDomainsInOneLogStatement));
525#endif
526
527 for (auto& domain : domains) {
528 if (batch.size() == maxNumberOfDomainsInOneLogStatement) {
529 RELEASE_LOG_INFO(ITPDebug, "%" PUBLIC_LOG_STRING " to (%{public}d of %u): %" PUBLIC_LOG_STRING ".", action, batchNumber, numberOfBatches, domainsToString(batch).utf8().data());
530 batch.shrink(0);
531 ++batchNumber;
532 }
533 batch.append(domain);
534 }
535 if (!batch.isEmpty())
536 RELEASE_LOG_INFO(ITPDebug, "%" PUBLIC_LOG_STRING " to (%{public}d of %u): %" PUBLIC_LOG_STRING ".", action, batchNumber, numberOfBatches, domainsToString(batch).utf8().data());
537}
