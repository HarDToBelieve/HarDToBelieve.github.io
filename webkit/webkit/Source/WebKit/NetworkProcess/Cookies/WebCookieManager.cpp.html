50WebCookieManager::WebCookieManager(NetworkProcess& process)
51 : m_process(process)
52{
53 m_process.addMessageReceiver(Messages::WebCookieManager::messageReceiverName(), *this);
54}

56WebCookieManager::~WebCookieManager() = default;

58void WebCookieManager::getHostnamesWithCookies(PAL::SessionID sessionID, CompletionHandler<void(Vector<String>&&)>&& completionHandler)
59{
60 HashSet<String> hostnames;
61 if (auto* storageSession = m_process.storageSession(sessionID))
62 storageSession->getHostnamesWithCookies(hostnames);
63 completionHandler(copyToVector(hostnames));
64}

92void WebCookieManager::getAllCookies(PAL::SessionID sessionID, CompletionHandler<void(Vector<WebCore::Cookie>&&)>&& completionHandler)
93{
94 Vector<Cookie> cookies;
95 if (auto* storageSession = m_process.storageSession(sessionID))
96 cookies = storageSession->getAllCookies();
97 completionHandler(WTFMove(cookies));
98}

100void WebCookieManager::getCookies(PAL::SessionID sessionID, const URL& url, CompletionHandler<void(Vector<WebCore::Cookie>&&)>&& completionHandler)
101{
102 Vector<Cookie> cookies;
103 if (auto* storageSession = m_process.storageSession(sessionID))
104 cookies = storageSession->getCookies(url);
105 completionHandler(WTFMove(cookies));
106}

108void WebCookieManager::setCookie(PAL::SessionID sessionID, const Vector<Cookie>& cookies, CompletionHandler<void()>&& completionHandler)
109{
110 if (auto* storageSession = m_process.storageSession(sessionID)) {
111 for (auto& cookie : cookies)
112 storageSession->setCookie(cookie);
113 }
114 completionHandler();
115}

117void WebCookieManager::setCookies(PAL::SessionID sessionID, const Vector<Cookie>& cookies, const URL& url, const URL& mainDocumentURL, CompletionHandler<void()>&& completionHandler)

124void WebCookieManager::notifyCookiesDidChange(PAL::SessionID sessionID)
125{
126 ASSERT(RunLoop::isMain());
127 m_process.send(Messages::WebCookieManagerProxy::CookiesDidChange(sessionID), 0);
128}
129
130void WebCookieManager::startObservingCookieChanges(PAL::SessionID sessionID)
131{
132 if (auto* storageSession = m_process.storageSession(sessionID)) {
133 WebCore::startObservingCookieChanges(*storageSession, [this, sessionID] {
134 notifyCookiesDidChange(sessionID);
135 });
136 }
137}
138
139void WebCookieManager::stopObservingCookieChanges(PAL::SessionID sessionID)
140{
141 if (auto* storageSession = m_process.storageSession(sessionID))
142 WebCore::stopObservingCookieChanges(*storageSession);
143}

145void WebCookieManager::setHTTPCookieAcceptPolicy(HTTPCookieAcceptPolicy policy, CompletionHandler<void()>&& completionHandler)
146{
147 platformSetHTTPCookieAcceptPolicy(policy, [policy, process = makeRef(m_process), completionHandler = WTFMove(completionHandler)] () mutable {
148 process->cookieAcceptPolicyChanged(policy);
149 completionHandler();
150 });
151}
152
153void WebCookieManager::getHTTPCookieAcceptPolicy(PAL::SessionID sessionID, CompletionHandler<void(HTTPCookieAcceptPolicy)>&& completionHandler)
154{
155 if (auto* storageSession = m_process.storageSession(sessionID))
156 completionHandler(storageSession->cookieAcceptPolicy());
157 else
158 completionHandler(HTTPCookieAcceptPolicy::Never);
159}
