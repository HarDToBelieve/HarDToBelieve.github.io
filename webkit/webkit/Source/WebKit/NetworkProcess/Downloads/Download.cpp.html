88void Download::cancel(CompletionHandler<void(const IPC::DataReference&)>&& completionHandler, IgnoreDidFailCallback ignoreDidFailCallback)
89{
90 RELEASE_ASSERT(isMainRunLoop());
91
92 // URLSession:task:didCompleteWithError: is still called after cancelByProducingResumeData's completionHandler.
93 // If this cancel request came from the API, we do not want to send DownloadProxy::DidFail because the
94 // completionHandler will inform the API that the cancellation succeeded.
95 m_ignoreDidFailCallback = ignoreDidFailCallback;
96
97 auto completionHandlerWrapper = [this, weakThis = makeWeakPtr(*this), completionHandler = WTFMove(completionHandler)] (const IPC::DataReference& resumeData) mutable {
98 completionHandler(resumeData);
99 if (!weakThis || m_ignoreDidFailCallback == IgnoreDidFailCallback::No)
100 return;
101 DOWNLOAD_RELEASE_LOG("didCancel: (id = %" PRIu64 ")", downloadID().toUInt64());
102 if (auto extension = std::exchange(m_sandboxExtension, nullptr))
103 extension->revoke();
104 m_downloadManager.downloadFinished(*this);
105 };
106
107 if (m_download) {
108 m_download->cancel();
109 completionHandlerWrapper({ });
110 return;
111 }
112 platformCancelNetworkLoad(WTFMove(completionHandlerWrapper));
113}

115void Download::didReceiveChallenge(const WebCore::AuthenticationChallenge& challenge, ChallengeCompletionHandler&& completionHandler)
116{
117 if (challenge.protectionSpace().isPasswordBased() && !challenge.proposedCredential().isEmpty() && !challenge.previousFailureCount()) {
118 completionHandler(AuthenticationChallengeDisposition::UseCredential, challenge.proposedCredential());
119 return;
120 }
121
122 m_client->downloadsAuthenticationManager().didReceiveAuthenticationChallenge(*this, challenge, WTFMove(completionHandler));
123}

130void Download::didReceiveData(uint64_t bytesWritten, uint64_t totalBytesWritten, uint64_t totalBytesExpectedToWrite)
131{
132 if (!m_hasReceivedData) {
133 DOWNLOAD_RELEASE_LOG("didReceiveData: Started receiving data (id = %" PRIu64 ")", downloadID().toUInt64());
134 m_hasReceivedData = true;
135 }
136
137 m_monitor.downloadReceivedBytes(bytesWritten);
138
139 send(Messages::DownloadProxy::DidReceiveData(bytesWritten, totalBytesWritten, totalBytesExpectedToWrite));
140}
141
142void Download::didFinish()
143{
144 DOWNLOAD_RELEASE_LOG("didFinish: (id = %" PRIu64 ")", downloadID().toUInt64());
145
146 send(Messages::DownloadProxy::DidFinish());
147
148 if (m_sandboxExtension) {
149 m_sandboxExtension->revoke();
150 m_sandboxExtension = nullptr;
151 }
152
153 m_downloadManager.downloadFinished(*this);
154}
155
156void Download::didFail(const ResourceError& error, const IPC::DataReference& resumeData)
157{
158 if (m_ignoreDidFailCallback == IgnoreDidFailCallback::Yes)
159 return;
160
161 DOWNLOAD_RELEASE_LOG("didFail: (id = %" PRIu64 ", isTimeout = %d, isCancellation = %d, errCode = %d)",
162 downloadID().toUInt64(), error.isTimeout(), error.isCancellation(), error.errorCode());
163
164 send(Messages::DownloadProxy::DidFail(error, resumeData));
165
166 if (m_sandboxExtension) {
167 m_sandboxExtension->revoke();
168 m_sandboxExtension = nullptr;
169 }
170 m_downloadManager.downloadFinished(*this);
171}
