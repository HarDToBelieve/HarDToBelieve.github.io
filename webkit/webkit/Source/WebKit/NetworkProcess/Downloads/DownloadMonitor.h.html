31namespace WebKit {
32
33class Download;
34
35class DownloadMonitor {
36 WTF_MAKE_NONCOPYABLE(DownloadMonitor); WTF_MAKE_FAST_ALLOCATED;
37public:
38 DownloadMonitor(Download&);
39
40 void applicationDidEnterBackground();
41 void applicationWillEnterForeground();
42 void downloadReceivedBytes(uint64_t);
43 void timerFired();
44
45private:
46 Download& m_download;
47
48 double measuredThroughputRate() const;
49 uint32_t testSpeedMultiplier() const;
50
51 struct Timestamp {
52 MonotonicTime time;
53 uint64_t bytesReceived;
54 };
55 static constexpr size_t timestampCapacity = 10;
56 Deque<Timestamp, timestampCapacity> m_timestamps;
57 WebCore::Timer m_timer { *this, &DownloadMonitor::timerFired };
58 size_t m_interval { 0 };
59};
