123template<class MessageType> void WebIDBConnectionToClient::handleGetResult(const WebCore::IDBResultData& resultData)
124{
125 if (resultData.type() == IDBResultType::Error) {
126 send(MessageType(resultData));
127 return;
128 }
129
130 if (resultData.type() == IDBResultType::GetAllRecordsSuccess && resultData.getAllResult().type() == IndexedDB::GetAllType::Keys) {
131 send(MessageType(resultData));
132 return;
133 }
134
135 auto blobFilePaths = resultData.type() == IDBResultType::GetAllRecordsSuccess ? resultData.getAllResult().allBlobFilePaths() : resultData.getResult().value().blobFilePaths();
136 if (blobFilePaths.isEmpty()) {
137 send(MessageType(resultData));
138 return;
139 }
140
141 send(MessageType(resultData));
142}
