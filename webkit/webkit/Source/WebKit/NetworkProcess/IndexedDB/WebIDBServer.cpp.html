40WebIDBServer::WebIDBServer(PAL::SessionID sessionID, const String& directory, WebCore::IDBServer::IDBServer::StorageQuotaManagerSpaceRequester&& spaceRequester)
41 : m_queue(WorkQueue::create("com.apple.WebKit.IndexedDBServer"))
42{
43 ASSERT(RunLoop::isMain());
44
45 postTask([this, protectedThis = makeRef(*this), sessionID, directory = directory.isolatedCopy(), spaceRequester = WTFMove(spaceRequester)] () mutable {
46 ASSERT(!RunLoop::isMain());
47
48 Locker locker { m_serverLock };
49 m_server = makeUnique<WebCore::IDBServer::IDBServer>(sessionID, directory, WTFMove(spaceRequester), m_serverLock);
50 });
51}

118void WebIDBServer::suspend() WTF_IGNORES_THREAD_SAFETY_ANALYSIS
119{
120 ASSERT(RunLoop::isMain());
121
122 if (m_isSuspended)
123 return;
124
125 m_isSuspended = true;
126 m_serverLock.lock();
127 if (m_server)
128 m_server->stopDatabaseActivitiesOnMainThread();
129}
130
131void WebIDBServer::resume() WTF_IGNORES_THREAD_SAFETY_ANALYSIS
132{
133 ASSERT(RunLoop::isMain());
134
135 if (!m_isSuspended)
136 return;
137
138 m_isSuspended = false;
139 m_serverLock.unlock();
140}

353void WebIDBServer::addConnection(IPC::Connection& connection, WebCore::ProcessIdentifier processIdentifier)
354{
355 ASSERT(RunLoop::isMain());
356
357 postTask([this, protectedThis = makeRef(*this), protectedConnection = makeRefPtr(connection), processIdentifier] {
358 auto[iter, isNewEntry] = m_connectionMap.ensure(protectedConnection->uniqueID(), [&] {
359 return makeUnique<WebIDBConnectionToClient>(*protectedConnection, processIdentifier);
360 });
361
362 ASSERT_UNUSED(isNewEntry, isNewEntry);
363
364 Locker locker { m_serverLock };
365 m_server->registerConnection(iter->value->connectionToClient());
366 });
367 m_connections.add(connection);
368 connection.addWorkQueueMessageReceiver(Messages::WebIDBServer::messageReceiverName(), m_queue.get(), this);
369}
370
371void WebIDBServer::removeConnection(IPC::Connection& connection)
372{
373 ASSERT(RunLoop::isMain());
374
375 if (!m_connections.remove(connection))
376 return;
377
378 connection.removeWorkQueueMessageReceiver(Messages::WebIDBServer::messageReceiverName());
379 postTask([this, protectedThis = makeRef(*this), connectionID = connection.uniqueID()] {
380 auto connection = m_connectionMap.take(connectionID);
381
382 ASSERT(connection);
383
384 Locker locker { m_serverLock };
385 m_server->unregisterConnection(connection->connectionToClient());
386 });
387}

403void WebIDBServer::close(CompletionHandler<void()>&& completionHandler)
404{
405 ASSERT(RunLoop::isMain());
406
407 // Remove the references held by IPC::Connection.
408 for (auto& connection : m_connections)
409 connection.removeWorkQueueMessageReceiver(Messages::WebIDBServer::messageReceiverName());
410
411 // Dispatch last task to clean up.
412 postTask([this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)]() mutable {
413 m_connectionMap.clear();
414
415 {
416 Locker locker { m_serverLock };
417 m_server = nullptr;
418 }
419
420 postTaskReply([protectedThis = WTFMove(protectedThis), completionHandler = WTFMove(completionHandler)]() mutable {
421 if (completionHandler)
422 completionHandler();
423 });
424 });
425}
