40namespace WebKit {
41
42class NetworkActivityTracker {
43public:
44 enum class Domain {
45 // These are defined to match analogous values used in the Darwin implementation.
46 // If they are renumbered, platform-specific code will need to be added to map
47 // them to the Darwin-specific values.
48
49 Invalid = 0,
50 WebKit = 16,
51 };
52
53 enum class Label {
54 // These are ours to define, but once defined, they shouldn't change. They can
55 // be obsolesced and replaced with other codes, but previously-defined codes
56 // should not be renumbered. Previously assigned values should not be re-used.
57
58 Invalid = 0,
59 LoadPage = 1,
60 LoadResource = 2,
61 };
62
63 enum class CompletionCode {
64 Undefined,
65 None,
66 Success,
67 Failure,
68 Cancel,
69 };
70
71 NetworkActivityTracker() = default;
72 explicit NetworkActivityTracker(Label, Domain = Domain::WebKit);
73 ~NetworkActivityTracker();
74
75 void setParent(NetworkActivityTracker&);
76 void start();
77 void complete(CompletionCode);
78
79#if HAVE(NW_ACTIVITY)
80 nw_activity_t getPlatformObject() const { return m_networkActivity.get(); }
81#endif
82
83private:
84#if HAVE(NW_ACTIVITY)
85 Domain m_domain { Domain::Invalid };
86 Label m_label { Label::Invalid };
87 bool m_isCompleted { false };
88 RetainPtr<nw_activity_t> m_networkActivity;
89#endif
90};

92} // namespace WebKit
