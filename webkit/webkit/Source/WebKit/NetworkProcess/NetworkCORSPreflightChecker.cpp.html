44NetworkCORSPreflightChecker::NetworkCORSPreflightChecker(NetworkProcess& networkProcess, NetworkResourceLoader* networkResourceLoader, Parameters&& parameters, bool shouldCaptureExtraNetworkLoadMetrics, CompletionCallback&& completionCallback)

48 , m_shouldCaptureExtraNetworkLoadMetrics(shouldCaptureExtraNetworkLoadMetrics)
49 , m_networkResourceLoader(makeWeakPtr(networkResourceLoader))

53NetworkCORSPreflightChecker::~NetworkCORSPreflightChecker()
54{
55 if (m_task) {
56 ASSERT(m_task->client() == this);
57 m_task->clearClient();
58 m_task->cancel();
59 }
60 if (m_completionCallback)
61 m_completionCallback(ResourceError { ResourceError::Type::Cancellation });
62}

64void NetworkCORSPreflightChecker::startPreflight()
65{
66 CORS_CHECKER_RELEASE_LOG("startPreflight");
67
68 NetworkLoadParameters loadParameters;
69 loadParameters.request = createAccessControlPreflightRequest(m_parameters.originalRequest, m_parameters.sourceOrigin, m_parameters.referrer);
70 if (!m_parameters.userAgent.isNull())
71 loadParameters.request.setHTTPHeaderField(HTTPHeaderName::UserAgent, m_parameters.userAgent);
72
73 if (m_shouldCaptureExtraNetworkLoadMetrics)
74 m_loadInformation = NetworkTransactionInformation { NetworkTransactionInformation::Type::Preflight, loadParameters.request, { }, { } };
75
76 if (auto* networkSession = m_networkProcess->networkSession(m_parameters.sessionID)) {
77 m_task = NetworkDataTask::create(*networkSession, *this, WTFMove(loadParameters));
78 m_task->resume();
79 } else
80 ASSERT_NOT_REACHED();
81}

83void NetworkCORSPreflightChecker::willPerformHTTPRedirection(WebCore::ResourceResponse&& response, WebCore::ResourceRequest&&, RedirectCompletionHandler&& completionHandler)
84{
85 if (m_shouldCaptureExtraNetworkLoadMetrics)
86 m_loadInformation.response = WTFMove(response);
87
88 CORS_CHECKER_RELEASE_LOG("willPerformHTTPRedirection");
89 completionHandler({ });
90 m_completionCallback(ResourceError { errorDomainWebKitInternal, 0, m_parameters.originalRequest.url(), "Preflight response is not successful"_s, ResourceError::Type::AccessControl });
91}

93void NetworkCORSPreflightChecker::didReceiveChallenge(WebCore::AuthenticationChallenge&& challenge, NegotiatedLegacyTLS negotiatedLegacyTLS, ChallengeCompletionHandler&& completionHandler)
94{
95 CORS_CHECKER_RELEASE_LOG("didReceiveChallenge, authentication scheme: %u", challenge.protectionSpace().authenticationScheme());
96
97 auto scheme = challenge.protectionSpace().authenticationScheme();
98 bool isTLSHandshake = scheme == ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested
99 || scheme == ProtectionSpaceAuthenticationSchemeClientCertificateRequested;
100
101 if (!isTLSHandshake) {
102 completionHandler(AuthenticationChallengeDisposition::UseCredential, { });
103 return;
104 }
105
106 m_networkProcess->authenticationManager().didReceiveAuthenticationChallenge(m_parameters.sessionID, m_parameters.webPageProxyID, m_parameters.topOrigin ? &m_parameters.topOrigin->data() : nullptr, challenge, negotiatedLegacyTLS, WTFMove(completionHandler));
107}

125void NetworkCORSPreflightChecker::didCompleteWithError(const WebCore::ResourceError& preflightError, const WebCore::NetworkLoadMetrics& metrics)
126{
127 if (m_shouldCaptureExtraNetworkLoadMetrics)
128 m_loadInformation.metrics = metrics;
129
130 if (!preflightError.isNull()) {
131 CORS_CHECKER_RELEASE_LOG("didCompleteWithError");
132 auto error = preflightError;
133 if (error.isNull() || error.isGeneral())
134 error.setType(ResourceError::Type::AccessControl);
135
136 m_completionCallback(WTFMove(error));
137 return;
138 }
139
140 CORS_CHECKER_RELEASE_LOG("didComplete http_status_code=%d", m_response.httpStatusCode());
141
142 auto result = validatePreflightResponse(m_parameters.sessionID, m_parameters.originalRequest, m_response, m_parameters.storedCredentialsPolicy, m_parameters.sourceOrigin, m_networkResourceLoader.get());
143 if (!result) {
144 CORS_CHECKER_RELEASE_LOG("didComplete, AccessControl error: %s", result.error().utf8().data());
145 m_completionCallback(ResourceError { errorDomainWebKitInternal, 0, m_parameters.originalRequest.url(), result.error(), ResourceError::Type::AccessControl });
146 return;
147 }
148 m_completionCallback(ResourceError { });
149}
