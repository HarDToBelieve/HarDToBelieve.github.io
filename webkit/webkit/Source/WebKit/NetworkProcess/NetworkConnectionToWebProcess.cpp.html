105NetworkConnectionToWebProcess::NetworkConnectionToWebProcess(NetworkProcess& networkProcess, WebCore::ProcessIdentifier webProcessIdentifier, PAL::SessionID sessionID, IPC::Connection::Identifier connectionIdentifier)
106 : m_connection(IPC::Connection::createServerConnection(connectionIdentifier, *this))
107 , m_networkProcess(networkProcess)
108 , m_sessionID(sessionID)
109 , m_networkResourceLoaders([this](bool hasUpload) { hasUploadStateChanged(hasUpload); })
110#if ENABLE(WEB_RTC)
111 , m_mdnsRegister(*this)
112#endif
113 , m_webProcessIdentifier(webProcessIdentifier)
114 , m_schemeRegistry(NetworkSchemeRegistry::create())
115{
116 RELEASE_ASSERT(RunLoop::isMain());
117
118 // Use this flag to force synchronous messages to be treated as asynchronous messages in the WebProcess.
119 // Otherwise, the WebProcess would process incoming synchronous IPC while waiting for a synchronous IPC
120 // reply from the Network process, which would be unsafe.
121 m_connection->setOnlySendMessagesAsDispatchWhenWaitingForSyncReplyWhenProcessingSuchAMessage(true);
122 m_connection->open();
123
124#if ENABLE(SERVICE_WORKER)
125 establishSWServerConnection();
126#endif
127}

129NetworkConnectionToWebProcess::~NetworkConnectionToWebProcess()
130{
131 RELEASE_ASSERT(RunLoop::isMain());
132
133 m_connection->invalidate();
134
135 // This may call hasUploadStateChanged().
136 m_networkResourceLoaders.clear();
137
138 for (auto& port : m_processEntangledPorts)
139 networkProcess().messagePortChannelRegistry().didCloseMessagePort(port);
140
141#if HAVE(COOKIE_CHANGE_LISTENER_API)
142 if (auto* networkStorageSession = storageSession())
143 networkStorageSession->stopListeningForCookieChangeNotifications(*this, m_hostsWithCookieListeners);
144#endif
145
146#if USE(LIBWEBRTC)
147 if (m_rtcProvider)
148 m_rtcProvider->close();
149#endif
150#if ENABLE(WEB_RTC)
151 unregisterToRTCDataChannelProxy();
152#endif
153
154#if ENABLE(SERVICE_WORKER)
155 unregisterSWConnection();
156#endif
157}

188void NetworkConnectionToWebProcess::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
189{
190 // For security reasons, Messages::NetworkProcess IPC is only supposed to come from the UIProcess.
191 ASSERT(decoder.messageReceiverName() != Messages::NetworkProcess::messageReceiverName());
192
193 if (decoder.messageReceiverName() == Messages::NetworkConnectionToWebProcess::messageReceiverName()) {
194 didReceiveNetworkConnectionToWebProcessMessage(connection, decoder);
195 return;
196 }
197
198 if (decoder.messageReceiverName() == Messages::NetworkResourceLoader::messageReceiverName()) {
199 RELEASE_ASSERT(RunLoop::isMain());
200 RELEASE_ASSERT(decoder.destinationID());
201 if (auto* loader = m_networkResourceLoaders.get(decoder.destinationID()))
202 loader->didReceiveNetworkResourceLoaderMessage(connection, decoder);
203 return;
204 }
205
206 if (decoder.messageReceiverName() == Messages::NetworkSocketStream::messageReceiverName()) {
207 if (auto* socketStream = m_networkSocketStreams.get(makeObjectIdentifier<WebSocketIdentifierType>(decoder.destinationID()))) {
208 socketStream->didReceiveMessage(connection, decoder);
209 if (decoder.messageName() == Messages::NetworkSocketStream::Close::name())
210 m_networkSocketStreams.remove(makeObjectIdentifier<WebSocketIdentifierType>(decoder.destinationID()));
211 }
212 return;
213 }
214
215 if (decoder.messageReceiverName() == Messages::NetworkSocketChannel::messageReceiverName()) {
216 if (auto* channel = m_networkSocketChannels.get(makeObjectIdentifier<WebSocketIdentifierType>(decoder.destinationID())))
217 channel->didReceiveMessage(connection, decoder);
218 return;
219 }
220
221#if USE(LIBWEBRTC)
222 if (decoder.messageReceiverName() == Messages::NetworkRTCMonitor::messageReceiverName()) {
223 rtcProvider().didReceiveNetworkRTCMonitorMessage(connection, decoder);
224 return;
225 }
226#endif
227#if ENABLE(WEB_RTC)
228 if (decoder.messageReceiverName() == Messages::NetworkMDNSRegister::messageReceiverName()) {
229 mdnsRegister().didReceiveMessage(connection, decoder);
230 return;
231 }
232#endif
233
234 if (decoder.messageReceiverName() == Messages::CacheStorageEngineConnection::messageReceiverName()) {
235 cacheStorageConnection().didReceiveMessage(connection, decoder);
236 return;
237 }
238
239#if ENABLE(SERVICE_WORKER)
240 if (decoder.messageReceiverName() == Messages::WebSWServerConnection::messageReceiverName()) {
241 if (m_swConnection)
242 m_swConnection->didReceiveMessage(connection, decoder);
243 return;
244 }
245 if (decoder.messageReceiverName() == Messages::WebSWServerToContextConnection::messageReceiverName()) {
246 if (m_swContextConnection)
247 m_swContextConnection->didReceiveMessage(connection, decoder);
248 return;
249 }
250
251 if (decoder.messageReceiverName() == Messages::ServiceWorkerFetchTask::messageReceiverName()) {
252 if (m_swContextConnection)
253 m_swContextConnection->didReceiveFetchTaskMessage(connection, decoder);
254 return;
255 }
256#endif
257
258#if ENABLE(APPLE_PAY_REMOTE_UI)
259 if (decoder.messageReceiverName() == Messages::WebPaymentCoordinatorProxy::messageReceiverName())
260 return paymentCoordinator().didReceiveMessage(connection, decoder);
261#endif
262
263 WTFLogAlways("Unhandled network process message '%s'", description(decoder.messageName()));
264 ASSERT_NOT_REACHED();
265}

319bool NetworkConnectionToWebProcess::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& reply)
320{
321 // For security reasons, Messages::NetworkProcess IPC is only supposed to come from the UIProcess.
322 ASSERT(decoder.messageReceiverName() != Messages::NetworkProcess::messageReceiverName());
323
324 if (decoder.messageReceiverName() == Messages::NetworkConnectionToWebProcess::messageReceiverName())
325 return didReceiveSyncNetworkConnectionToWebProcessMessage(connection, decoder, reply);
326
327#if ENABLE(SERVICE_WORKER)
328 if (decoder.messageReceiverName() == Messages::WebSWServerConnection::messageReceiverName()) {
329 if (m_swConnection)
330 return m_swConnection->didReceiveSyncMessage(connection, decoder, reply);
331 return false;
332 }
333#endif
334
335#if ENABLE(APPLE_PAY_REMOTE_UI)
336 if (decoder.messageReceiverName() == Messages::WebPaymentCoordinatorProxy::messageReceiverName())
337 return paymentCoordinator().didReceiveSyncMessage(connection, decoder, reply);
338#endif
339
340 WTFLogAlways("Unhandled network process message '%s'", description(decoder.messageName()));
341 ASSERT_NOT_REACHED();
342 return false;
343}

351void NetworkConnectionToWebProcess::didClose(IPC::Connection& connection)
352{
353#if ENABLE(SERVICE_WORKER)
354 m_swContextConnection = nullptr;
355#else
356 UNUSED_PARAM(connection);
357#endif
358
359 // Protect ourself as we might be otherwise be deleted during this function.
360 Ref<NetworkConnectionToWebProcess> protector(*this);
361
362#if OS(DARWIN)
363 CONNECTION_RELEASE_LOG(Loading, "didClose: WebProcess (%d) closed its connection. Aborting related loaders.", connection.remoteProcessID());
364#else
365 CONNECTION_RELEASE_LOG(Loading, "didClose: WebProcess closed its connection. Aborting related loaders.");
366#endif
367
368 while (!m_networkResourceLoaders.isEmpty())
369 m_networkResourceLoaders.begin()->value->abort();
370
371 // All trackers of resources that were in the middle of being loaded were
372 // stopped with the abort() calls above, but we still need to sweep up the
373 // root activity trackers.
374 stopAllNetworkActivityTracking();
375
376 m_networkProcess->connectionToWebProcessClosed(connection, m_sessionID);
377
378 m_networkProcess->removeNetworkConnectionToWebProcess(*this);
379
380#if USE(LIBWEBRTC)
381 if (m_rtcProvider) {
382 m_rtcProvider->close();
383 m_rtcProvider = nullptr;
384 }
385#endif
386
387#if ENABLE(SERVICE_WORKER)
388 unregisterSWConnection();
389#endif
390
391#if ENABLE(APPLE_PAY_REMOTE_UI)
392 m_paymentCoordinator = nullptr;
393#endif
394}

449Vector<RefPtr<WebCore::BlobDataFileReference>> NetworkConnectionToWebProcess::resolveBlobReferences(const NetworkResourceLoadParameters& loadParameters)
450{
451 CONNECTION_RELEASE_LOG(Loading, "resolveBlobReferences: (parentPID=%d, pageProxyID=%" PRIu64 ", webPageID=%" PRIu64 ", frameID=%" PRIu64 ", resourceID=%" PRIu64 ")", loadParameters.parentPID, loadParameters.webPageProxyID.toUInt64(), loadParameters.webPageID.toUInt64(), loadParameters.webFrameID.toUInt64(), loadParameters.identifier);
452
453 auto* session = networkSession();
454 if (!session)
455 return { };
456
457 auto& blobRegistry = session->blobRegistry();
458
459 Vector<RefPtr<WebCore::BlobDataFileReference>> files;
460 if (auto* body = loadParameters.request.httpBody()) {
461 for (auto& element : body->elements()) {
462 if (auto* blobData = WTF::get_if<FormDataElement::EncodedBlobData>(element.data))
463 files.appendVector(blobRegistry.filesInBlob(blobData->url));
464 }
465 const_cast<WebCore::ResourceRequest&>(loadParameters.request).setHTTPBody(body->resolveBlobReferences(&blobRegistry));
466 }
467
468 return files;
469}

478void NetworkConnectionToWebProcess::scheduleResourceLoad(NetworkResourceLoadParameters&& loadParameters)
479{
480 CONNECTION_RELEASE_LOG(Loading, "scheduleResourceLoad: (parentPID=%d, pageProxyID=%" PRIu64 ", webPageID=%" PRIu64 ", frameID=%" PRIu64 ", resourceID=%" PRIu64 ")", loadParameters.parentPID, loadParameters.webPageProxyID.toUInt64(), loadParameters.webPageID.toUInt64(), loadParameters.webFrameID.toUInt64(), loadParameters.identifier);
481
482#if ENABLE(SERVICE_WORKER)
483 auto& server = m_networkProcess->swServerForSession(m_sessionID);
484 if (!server.isImportCompleted()) {
485 server.whenImportIsCompleted([this, protectedThis = makeRef(*this), loadParameters = WTFMove(loadParameters)]() mutable {
486 if (!m_networkProcess->webProcessConnection(webProcessIdentifier()))
487 return;
488 ASSERT(m_networkProcess->swServerForSession(m_sessionID).isImportCompleted());
489 scheduleResourceLoad(WTFMove(loadParameters));
490 });
491 return;
492 }
493#endif
494 auto identifier = loadParameters.identifier;
495 RELEASE_ASSERT(identifier);
496 RELEASE_ASSERT(RunLoop::isMain());
497 ASSERT(!m_networkResourceLoaders.contains(identifier));
498
499 auto& loader = m_networkResourceLoaders.add(identifier, NetworkResourceLoader::create(WTFMove(loadParameters), *this)).iterator->value;
500
501#if ENABLE(SERVICE_WORKER)
502 loader->startWithServiceWorker();
503#else
504 loader->start();
505#endif
506}

552void NetworkConnectionToWebProcess::removeLoadIdentifier(ResourceLoadIdentifier identifier)
553{
554 RELEASE_ASSERT(identifier);
555 RELEASE_ASSERT(RunLoop::isMain());
556
557 RefPtr<NetworkResourceLoader> loader = m_networkResourceLoaders.get(identifier);
558
559 // It's possible we have no loader for this identifier if the NetworkProcess crashed and this was a respawned NetworkProcess.
560 if (!loader)
561 return;
562
563 // Abort the load now, as the WebProcess won't be able to respond to messages any more which might lead
564 // to leaked loader resources (connections, threads, etc).
565 CONNECTION_RELEASE_LOG(Loading, "removeLoadIdentifier: Removing identifier %" PRIu64 " and aborting corresponding loader", identifier);
566 loader->abort();
567 ASSERT(!m_networkResourceLoaders.contains(identifier));
568}

588void NetworkConnectionToWebProcess::sendH2Ping(NetworkResourceLoadParameters&& parameters, CompletionHandler<void(Expected<Seconds, ResourceError>&&)>&& completionHandler)
589{
590#if ENABLE(SERVER_PRECONNECT)

594
595 URL url = parameters.request.url();
596 auto* task = new PreconnectTask(*networkSession, WTFMove(parameters), [] (const ResourceError&) { });
597 task->setH2PingCallback(url, WTFMove(completionHandler));
598 task->start();
599#else
600 ASSERT_NOT_REACHED();
601 completionHandler(makeUnexpected(internalError(parameters.request.url())));
602#endif
603}

605void NetworkConnectionToWebProcess::preconnectTo(std::optional<uint64_t> preconnectionIdentifier, NetworkResourceLoadParameters&& loadParameters)
606{
607 CONNECTION_RELEASE_LOG(Loading, "preconnectTo: (parentPID=%d, pageProxyID=%" PRIu64 ", webPageID=%" PRIu64 ", frameID=%" PRIu64 ", resourceID=%" PRIu64 ")", loadParameters.parentPID, loadParameters.webPageProxyID.toUInt64(), loadParameters.webPageID.toUInt64(), loadParameters.webFrameID.toUInt64(), loadParameters.identifier);
608
609 ASSERT(!loadParameters.request.httpBody());
610
611 auto completionHandler = [this, protectedThis = makeRef(*this), preconnectionIdentifier = WTFMove(preconnectionIdentifier)](const ResourceError& error) {
612 if (preconnectionIdentifier)
613 didFinishPreconnection(*preconnectionIdentifier, error);
614 };
615
616#if ENABLE(LEGACY_CUSTOM_PROTOCOL_MANAGER)

987std::optional<NetworkActivityTracker> NetworkConnectionToWebProcess::startTrackingResourceLoad(PageIdentifier pageID, ResourceLoadIdentifier resourceID, bool isTopResource)
988{
989 if (m_sessionID.isEphemeral())
990 return std::nullopt;
991
992 // Either get the existing root activity tracker for this page or create a
993 // new one if this is the main resource.
994
995 size_t rootActivityIndex;
996 if (isTopResource) {
997 // If we're loading a page from the top, make sure any tracking of
998 // previous activity for this page is stopped.
999
1000 stopAllNetworkActivityTrackingForPage(pageID);
1001
1002 rootActivityIndex = m_networkActivityTrackers.size();
1003 m_networkActivityTrackers.constructAndAppend(pageID);
1004 m_networkActivityTrackers[rootActivityIndex].networkActivity.start();
1005
1006#if HAVE(NW_ACTIVITY)
1007 ASSERT(m_networkActivityTrackers[rootActivityIndex].networkActivity.getPlatformObject());
1008#endif
1009 } else {
1010 rootActivityIndex = findRootNetworkActivity(pageID);
1011
1012 // This could happen if the Networking process crashes, taking its
1013 // previous state with it.
1014 if (rootActivityIndex == notFound)
1015 return std::nullopt;
1016
1017#if HAVE(NW_ACTIVITY)
1018 ASSERT(m_networkActivityTrackers[rootActivityIndex].networkActivity.getPlatformObject());
1019#endif
1020 }
1021
1022 // Create a tracker for the loading of the new resource, setting the root
1023 // activity tracker as its parent.
1024
1025 size_t newActivityIndex = m_networkActivityTrackers.size();
1026 m_networkActivityTrackers.constructAndAppend(pageID, resourceID);
1027#if HAVE(NW_ACTIVITY)
1028 ASSERT(m_networkActivityTrackers[newActivityIndex].networkActivity.getPlatformObject());
1029#endif
1030
1031 auto& newActivityTracker = m_networkActivityTrackers[newActivityIndex];
1032 newActivityTracker.networkActivity.setParent(m_networkActivityTrackers[rootActivityIndex].networkActivity);
1033 newActivityTracker.networkActivity.start();
1034
1035 return newActivityTracker.networkActivity;
1036}

1043
