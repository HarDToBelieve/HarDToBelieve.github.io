58class NetworkLoadChecker : public CanMakeWeakPtr<NetworkLoadChecker> {
59 WTF_MAKE_FAST_ALLOCATED;
60public:
61 enum class LoadType : bool { MainFrame, Other };
62
63 NetworkLoadChecker(NetworkProcess&, NetworkResourceLoader*, NetworkSchemeRegistry*, WebCore::FetchOptions&&, PAL::SessionID, WebPageProxyIdentifier, WebCore::HTTPHeaderMap&&, URL&&, DocumentURL&&, RefPtr<WebCore::SecurityOrigin>&&, RefPtr<WebCore::SecurityOrigin>&& topOrigin, WebCore::PreflightPolicy, String&& referrer, bool shouldCaptureExtraNetworkLoadMetrics = false, LoadType requestLoadType = LoadType::Other);
64 ~NetworkLoadChecker();

66 struct RedirectionTriplet {
67 WebCore::ResourceRequest request;
68 WebCore::ResourceRequest redirectRequest;
69 WebCore::ResourceResponse redirectResponse;
70 };
71
72 using RequestOrRedirectionTripletOrError = Variant<WebCore::ResourceRequest, RedirectionTriplet, WebCore::ResourceError>;
73 using ValidationHandler = CompletionHandler<void(RequestOrRedirectionTripletOrError&&)>;
74 void check(WebCore::ResourceRequest&&, WebCore::ContentSecurityPolicyClient*, ValidationHandler&&);
75
76 using RedirectionRequestOrError = Expected<RedirectionTriplet, WebCore::ResourceError>;
77 using RedirectionValidationHandler = CompletionHandler<void(RedirectionRequestOrError&&)>;
78 void checkRedirection(WebCore::ResourceRequest&& request, WebCore::ResourceRequest&& redirectRequest, WebCore::ResourceResponse&& redirectResponse, WebCore::ContentSecurityPolicyClient*, RedirectionValidationHandler&&);
79
80 WebCore::ResourceError validateResponse(const WebCore::ResourceRequest&, WebCore::ResourceResponse&);
81
82 void setCSPResponseHeaders(WebCore::ContentSecurityPolicyResponseHeaders&& headers) { m_cspResponseHeaders = WTFMove(headers); }
83#if ENABLE(CONTENT_EXTENSIONS)
84 void setContentExtensionController(URL&& mainDocumentURL, std::optional<UserContentControllerIdentifier> identifier)
85 {
86 m_mainDocumentURL = WTFMove(mainDocumentURL);
87 m_userContentControllerIdentifier = identifier;
88 }
89#endif
90
91 NetworkProcess& networkProcess() { return m_networkProcess; }
92
93 const URL& url() const { return m_url; }
94 WebCore::StoredCredentialsPolicy storedCredentialsPolicy() const { return m_storedCredentialsPolicy; }
95
96 WebCore::NetworkLoadInformation takeNetworkLoadInformation() { return WTFMove(m_loadInformation); }
97 void storeRedirectionIfNeeded(const WebCore::ResourceRequest&, const WebCore::ResourceResponse&);
98
99 void enableContentExtensionsCheck() { m_checkContentExtensions = true; }
100
101private:
102 WebCore::ContentSecurityPolicy* contentSecurityPolicy();
103 bool isChecking() const { return !!m_corsPreflightChecker; }
104 bool isRedirected() const { return m_redirectCount; }

106 void checkRequest(WebCore::ResourceRequest&&, WebCore::ContentSecurityPolicyClient*, ValidationHandler&&);

113 bool doesNotNeedCORSCheck(const URL&) const;
114 void checkCORSRequest(WebCore::ResourceRequest&&, ValidationHandler&&);

116 void checkCORSRequestWithPreflight(WebCore::ResourceRequest&&, ValidationHandler&&);
117
118 RequestOrRedirectionTripletOrError accessControlErrorForValidationHandler(String&&);
