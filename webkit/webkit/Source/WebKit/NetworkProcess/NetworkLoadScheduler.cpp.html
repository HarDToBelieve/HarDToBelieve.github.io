32namespace WebKit {
33
34static constexpr size_t maximumActiveCountForLowPriority = 2;
35
36class NetworkLoadScheduler::HostContext {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 HostContext() = default;
40 ~HostContext();
41
42 void schedule(NetworkLoad&);
43 void unschedule(NetworkLoad&);
44 void prioritize(NetworkLoad&);
45
46private:
47 void start(NetworkLoad&);
48 bool shouldDelayLowPriority() const { return m_activeLoads.size() >= maximumActiveCountForLowPriority; }
49
50 HashSet<NetworkLoad*> m_activeLoads;
51 ListHashSet<NetworkLoad*> m_pendingLoads;
52};
53
54void NetworkLoadScheduler::HostContext::schedule(NetworkLoad& load)
55{
56 auto startImmediately = [&] {
57 auto& request = load.currentRequest();
58 if (request.priority() > WebCore::ResourceLoadPriority::Low)
59 return true;
60
61 if (request.isConditional())
62 return true;
63
64 if (!shouldDelayLowPriority())
65 return true;
66
67 return false;
68 }();
69
70 if (startImmediately) {
71 start(load);
72 return;
73 }
74
75 m_pendingLoads.add(&load);
76}
77
78void NetworkLoadScheduler::HostContext::unschedule(NetworkLoad& load)
79{
80 m_activeLoads.remove(&load);
81 m_pendingLoads.remove(&load);
82
83 if (m_pendingLoads.isEmpty())
84 return;
85 if (shouldDelayLowPriority())
86 return;
87
88 start(*m_pendingLoads.takeFirst());
89}
90
91void NetworkLoadScheduler::HostContext::prioritize(NetworkLoad& load)
92{
93 auto priority = load.parameters().request.priority();
94 load.reprioritizeRequest(++priority);
95
96 if (!m_pendingLoads.remove(&load))
97 return;
98
99 start(load);
100}
101
102void NetworkLoadScheduler::HostContext::start(NetworkLoad& load)
103{
104 m_activeLoads.add(&load);
105
106 load.start();
107}
108
109NetworkLoadScheduler::HostContext::~HostContext()
110{
111 for (auto* load : m_pendingLoads)
112 start(*load);
113}

115NetworkLoadScheduler::NetworkLoadScheduler() = default;
116NetworkLoadScheduler::~NetworkLoadScheduler() = default;

118void NetworkLoadScheduler::schedule(NetworkLoad& load)
119{
120 auto* context = contextForLoad(load);
121
122 if (!context) {
123 load.start();
124 return;
125 }
126
127 context->schedule(load);
128}
129
130void NetworkLoadScheduler::unschedule(NetworkLoad& load)
131{
132 if (auto* context = contextForLoad(load))
133 context->unschedule(load);
134}
135
136void NetworkLoadScheduler::setResourceLoadSchedulingMode(WebCore::PageIdentifier pageIdentifier, WebCore::LoadSchedulingMode mode)
137{
138 switch (mode) {
139 case WebCore::LoadSchedulingMode::Prioritized:
140 m_pageContexts.ensure(pageIdentifier, [&] {
141 return makeUnique<PageContext>();
142 });
143 break;
144 case WebCore::LoadSchedulingMode::Direct:
145 m_pageContexts.remove(pageIdentifier);
146 break;
147 }
148}
149
150void NetworkLoadScheduler::prioritizeLoads(const Vector<NetworkLoad*>& loads)
151{
152 for (auto* load : loads) {
153 if (auto* context = contextForLoad(*load))
154 context->prioritize(*load);
155 }
156}
157
158void NetworkLoadScheduler::clearPageData(WebCore::PageIdentifier pageIdentifier)
159{
160 m_pageContexts.remove(pageIdentifier);
161}
162
163auto NetworkLoadScheduler::contextForLoad(const NetworkLoad& load) -> HostContext*
164{
165 if (!load.url().protocolIsInHTTPFamily())
166 return nullptr;
167
168 auto* pageContext = m_pageContexts.get(load.parameters().webPageID);
169 if (!pageContext)
170 return nullptr;
171
172 auto host = load.url().host().toString();
173 return pageContext->ensure(host, [&] {
174 return makeUnique<HostContext>();
175 }).iterator->value.get();
176}
