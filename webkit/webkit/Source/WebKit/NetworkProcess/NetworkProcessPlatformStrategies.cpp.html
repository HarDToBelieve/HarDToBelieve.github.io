56BlobRegistry* NetworkProcessPlatformStrategies::createBlobRegistry()
57{
58 using namespace WebCore;
59 class EmptyBlobRegistry : public WebCore::BlobRegistry {
60 void registerFileBlobURL(const URL&, Ref<BlobDataFileReference>&&, const String& path, const String& contentType) final { ASSERT_NOT_REACHED(); }
61 void registerBlobURL(const URL&, Vector<BlobPart>&&, const String& contentType) final { ASSERT_NOT_REACHED(); }
62 void registerBlobURL(const URL&, const URL& srcURL) final { ASSERT_NOT_REACHED(); }
63 void registerBlobURLOptionallyFileBacked(const URL&, const URL& srcURL, RefPtr<BlobDataFileReference>&&, const String& contentType) final { ASSERT_NOT_REACHED(); }
64 void registerBlobURLForSlice(const URL&, const URL& srcURL, long long start, long long end, const String& contentType) final { ASSERT_NOT_REACHED(); }
65 void unregisterBlobURL(const URL&) final { ASSERT_NOT_REACHED(); }
66 unsigned long long blobSize(const URL&) final { ASSERT_NOT_REACHED(); return 0; }
67 void writeBlobsToTemporaryFiles(const Vector<String>& blobURLs, CompletionHandler<void(Vector<String>&& filePaths)>&&) final { ASSERT_NOT_REACHED(); }
68 };
69 static NeverDestroyed<EmptyBlobRegistry> blobRegistry;
70 return &blobRegistry.get();
71}
