28#include "NetworkResourceLoader.h"

30namespace WebKit {
31
32NetworkResourceLoadMap::NetworkResourceLoadMap(Function<void(bool hasUpload)>&& hasUploadChangeListener)
33 : m_hasUploadChangeListener(WTFMove(hasUploadChangeListener))
34{
35}
36
37NetworkResourceLoadMap::~NetworkResourceLoadMap()
38{
39 clear();
40}
41
42NetworkResourceLoadMap::MapType::AddResult NetworkResourceLoadMap::add(ResourceLoadIdentifier identifier, Ref<NetworkResourceLoader>&& loader)
43{
44 ASSERT(!m_loaders.contains(identifier));
45 bool hasUpload = loader->originalRequest().hasUpload();
46 auto result = m_loaders.add(identifier, WTFMove(loader));
47 if (hasUpload)
48 setHasUpload(true);
49 return result;
50}
51
52bool NetworkResourceLoadMap::remove(ResourceLoadIdentifier identifier)
53{
54 return !!take(identifier);
55}
56
57void NetworkResourceLoadMap::clear()
58{
59 m_loaders.clear();
60 setHasUpload(false);
61}
62
63RefPtr<NetworkResourceLoader> NetworkResourceLoadMap::take(ResourceLoadIdentifier identifier)
64{
65 auto loader = m_loaders.take(identifier);
66 if (!loader)
67 return nullptr;
68
69 if (loader->originalRequest().hasUpload())
70 setHasUpload(WTF::anyOf(m_loaders.values(), [](auto& loader) { return loader->originalRequest().hasUpload(); }));
71
72 return loader;
73}
74
75NetworkResourceLoader* NetworkResourceLoadMap::get(ResourceLoadIdentifier identifier) const
76{
77 return m_loaders.get(identifier);
78}
79
80void NetworkResourceLoadMap::setHasUpload(bool hasUpload)
81{
82 if (m_hasUpload == hasUpload)
83 return;
84
85 m_hasUpload = hasUpload;
86 if (m_hasUploadChangeListener)
87 m_hasUploadChangeListener(m_hasUpload);
88}
