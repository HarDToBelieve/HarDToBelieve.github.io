34void NetworkResourceLoadParameters::encode(IPC::Encoder& encoder) const
35{
36 encoder << identifier;
37 encoder << webPageProxyID;
38 encoder << webPageID;
39 encoder << webFrameID;
40 encoder << parentPID;
41 encoder << request;
42
43 encoder << static_cast<bool>(request.httpBody());
44 if (request.httpBody()) {
45 request.httpBody()->encode(encoder);
46
47 const Vector<FormDataElement>& elements = request.httpBody()->elements();
48 size_t fileCount = 0;
49 for (size_t i = 0, count = elements.size(); i < count; ++i) {
50 if (WTF::holds_alternative<FormDataElement::EncodedFileData>(elements[i].data))
51 ++fileCount;
52 }
53
54 SandboxExtension::HandleArray requestBodySandboxExtensions;
55 requestBodySandboxExtensions.allocate(fileCount);
56 size_t extensionIndex = 0;
57 for (size_t i = 0, count = elements.size(); i < count; ++i) {
58 const FormDataElement& element = elements[i];
59 if (auto* fileData = WTF::get_if<FormDataElement::EncodedFileData>(element.data)) {
60 const String& path = fileData->filename;
61 SandboxExtension::createHandle(path, SandboxExtension::Type::ReadOnly, requestBodySandboxExtensions[extensionIndex++]);
62 }
63 }
64 encoder << requestBodySandboxExtensions;
65 }
66
67 if (request.url().isLocalFile()) {
68 SandboxExtension::Handle requestSandboxExtension;
69#if HAVE(AUDIT_TOKEN)
70 if (networkProcessAuditToken)
71 SandboxExtension::createHandleForReadByAuditToken(request.url().fileSystemPath(), *networkProcessAuditToken, requestSandboxExtension);
72 else
73#endif
74 SandboxExtension::createHandle(request.url().fileSystemPath(), SandboxExtension::Type::ReadOnly, requestSandboxExtension);
75
76 encoder << requestSandboxExtension;
77 }
78
79 encoder << contentSniffingPolicy;
80 encoder << contentEncodingSniffingPolicy;
81 encoder << storedCredentialsPolicy;
82 encoder << clientCredentialPolicy;
83 encoder << shouldPreconnectOnly;
84 encoder << shouldClearReferrerOnHTTPSToHTTPRedirect;
85 encoder << needsCertificateInfo;
86 encoder << isMainFrameNavigation;
87 encoder << isMainResourceNavigationForAnyFrame;
88 encoder << shouldRelaxThirdPartyCookieBlocking;
89 encoder << maximumBufferingTime;
90
91 encoder << static_cast<bool>(sourceOrigin);
92 if (sourceOrigin)
93 encoder << *sourceOrigin;
94 encoder << static_cast<bool>(topOrigin);
95 if (sourceOrigin)
96 encoder << *topOrigin;
97 encoder << options;
98 encoder << cspResponseHeaders;
99 encoder << originalRequestHeaders;
100
101 encoder << shouldRestrictHTTPResponseAccess;
102
103 encoder << preflightPolicy;
104
105 encoder << shouldEnableCrossOriginResourcePolicy;
106
107 encoder << frameAncestorOrigins;
108 encoder << pageHasResourceLoadClient;
109 encoder << parentFrameID;
110 encoder << crossOriginAccessControlCheckEnabled;
111
112 encoder << documentURL;
113
114#if ENABLE(SERVICE_WORKER)
115 encoder << serviceWorkersMode;
116 encoder << serviceWorkerRegistrationIdentifier;
117 encoder << httpHeadersToKeep;
118#endif
119
120#if ENABLE(CONTENT_EXTENSIONS)
121 encoder << mainDocumentURL;
122 encoder << userContentControllerIdentifier;
123#endif
124
125 encoder << isNavigatingToAppBoundDomain;
126 encoder << pcmDataCarried;
127}

129std::optional<NetworkResourceLoadParameters> NetworkResourceLoadParameters::decode(IPC::Decoder& decoder)
130{
131 NetworkResourceLoadParameters result;
132
133 if (!decoder.decode(result.identifier))
134 return std::nullopt;
135
136 std::optional<WebPageProxyIdentifier> webPageProxyID;
137 decoder >> webPageProxyID;
138 if (!webPageProxyID)
139 return std::nullopt;
140 result.webPageProxyID = *webPageProxyID;
141
142 std::optional<PageIdentifier> webPageID;
143 decoder >> webPageID;
144 if (!webPageID)
145 return std::nullopt;
146 result.webPageID = *webPageID;
147
148 if (!decoder.decode(result.webFrameID))
149 return std::nullopt;
150
151 if (!decoder.decode(result.parentPID))
152 return std::nullopt;
153
154 if (!decoder.decode(result.request))
155 return std::nullopt;
156
157 bool hasHTTPBody;
158 if (!decoder.decode(hasHTTPBody))
159 return std::nullopt;
160
161 if (hasHTTPBody) {
162 RefPtr<FormData> formData = FormData::decode(decoder);
163 if (!formData)
164 return std::nullopt;
165 result.request.setHTTPBody(WTFMove(formData));
166
167 std::optional<SandboxExtension::HandleArray> requestBodySandboxExtensionHandles;
168 decoder >> requestBodySandboxExtensionHandles;
169 if (!requestBodySandboxExtensionHandles)
170 return std::nullopt;
171 for (size_t i = 0; i < requestBodySandboxExtensionHandles->size(); ++i) {
172 if (auto extension = SandboxExtension::create(WTFMove(requestBodySandboxExtensionHandles->at(i))))
173 result.requestBodySandboxExtensions.append(WTFMove(extension));
174 }
175 }
176
177 if (result.request.url().isLocalFile()) {
178 std::optional<SandboxExtension::Handle> resourceSandboxExtensionHandle;
179 decoder >> resourceSandboxExtensionHandle;
180 if (!resourceSandboxExtensionHandle)
181 return std::nullopt;
182 result.resourceSandboxExtension = SandboxExtension::create(WTFMove(*resourceSandboxExtensionHandle));
183 }
184
185 if (!decoder.decode(result.contentSniffingPolicy))
186 return std::nullopt;
187 if (!decoder.decode(result.contentEncodingSniffingPolicy))
188 return std::nullopt;
189 if (!decoder.decode(result.storedCredentialsPolicy))
190 return std::nullopt;
191 if (!decoder.decode(result.clientCredentialPolicy))
192 return std::nullopt;
193 if (!decoder.decode(result.shouldPreconnectOnly))
194 return std::nullopt;
195 if (!decoder.decode(result.shouldClearReferrerOnHTTPSToHTTPRedirect))
196 return std::nullopt;
197 if (!decoder.decode(result.needsCertificateInfo))
198 return std::nullopt;
199 if (!decoder.decode(result.isMainFrameNavigation))
200 return std::nullopt;
201 if (!decoder.decode(result.isMainResourceNavigationForAnyFrame))
202 return std::nullopt;
203 if (!decoder.decode(result.shouldRelaxThirdPartyCookieBlocking))
204 return std::nullopt;
205 if (!decoder.decode(result.maximumBufferingTime))
206 return std::nullopt;
207
208 bool hasSourceOrigin;
209 if (!decoder.decode(hasSourceOrigin))
210 return std::nullopt;
211 if (hasSourceOrigin) {
212 result.sourceOrigin = SecurityOrigin::decode(decoder);
213 if (!result.sourceOrigin)
214 return std::nullopt;
215 }
216
217 bool hasTopOrigin;
218 if (!decoder.decode(hasTopOrigin))
219 return std::nullopt;
220 if (hasTopOrigin) {
221 result.topOrigin = SecurityOrigin::decode(decoder);
222 if (!result.topOrigin)
223 return std::nullopt;
224 }
225
226 std::optional<FetchOptions> options;
227 decoder >> options;
228 if (!options)
229 return std::nullopt;
230 result.options = *options;
231
232 if (!decoder.decode(result.cspResponseHeaders))
233 return std::nullopt;
234 if (!decoder.decode(result.originalRequestHeaders))
235 return std::nullopt;
236
237 std::optional<bool> shouldRestrictHTTPResponseAccess;
238 decoder >> shouldRestrictHTTPResponseAccess;
239 if (!shouldRestrictHTTPResponseAccess)
240 return std::nullopt;
241 result.shouldRestrictHTTPResponseAccess = *shouldRestrictHTTPResponseAccess;
242
243 if (!decoder.decode(result.preflightPolicy))
244 return std::nullopt;
245
246 std::optional<bool> shouldEnableCrossOriginResourcePolicy;
247 decoder >> shouldEnableCrossOriginResourcePolicy;
248 if (!shouldEnableCrossOriginResourcePolicy)
249 return std::nullopt;
250 result.shouldEnableCrossOriginResourcePolicy = *shouldEnableCrossOriginResourcePolicy;
251
252 if (!decoder.decode(result.frameAncestorOrigins))
253 return std::nullopt;
254
255 std::optional<bool> pageHasResourceLoadClient;
256 decoder >> pageHasResourceLoadClient;
257 if (!pageHasResourceLoadClient)
258 return std::nullopt;
259 result.pageHasResourceLoadClient = *pageHasResourceLoadClient;
260
261 std::optional<std::optional<FrameIdentifier>> parentFrameID;
262 decoder >> parentFrameID;
263 if (!parentFrameID)
264 return std::nullopt;
265 result.parentFrameID = WTFMove(*parentFrameID);
266
267 std::optional<bool> crossOriginAccessControlCheckEnabled;
268 decoder >> crossOriginAccessControlCheckEnabled;
269 if (!crossOriginAccessControlCheckEnabled)
270 return std::nullopt;
271 result.crossOriginAccessControlCheckEnabled = *crossOriginAccessControlCheckEnabled;
272
273 std::optional<URL> documentURL;
274 decoder >> documentURL;
275 if (!documentURL)
276 return std::nullopt;
277 result.documentURL = *documentURL;
278
279#if ENABLE(SERVICE_WORKER)
280 std::optional<ServiceWorkersMode> serviceWorkersMode;
281 decoder >> serviceWorkersMode;
282 if (!serviceWorkersMode)
283 return std::nullopt;
284 result.serviceWorkersMode = *serviceWorkersMode;
285
286 std::optional<std::optional<ServiceWorkerRegistrationIdentifier>> serviceWorkerRegistrationIdentifier;
287 decoder >> serviceWorkerRegistrationIdentifier;
288 if (!serviceWorkerRegistrationIdentifier)
289 return std::nullopt;
290 result.serviceWorkerRegistrationIdentifier = *serviceWorkerRegistrationIdentifier;
291
292 std::optional<OptionSet<HTTPHeadersToKeepFromCleaning>> httpHeadersToKeep;
293 decoder >> httpHeadersToKeep;
294 if (!httpHeadersToKeep)
295 return std::nullopt;
296 result.httpHeadersToKeep = WTFMove(*httpHeadersToKeep);
297#endif
298
299#if ENABLE(CONTENT_EXTENSIONS)
300 if (!decoder.decode(result.mainDocumentURL))
301 return std::nullopt;
302
303 std::optional<std::optional<UserContentControllerIdentifier>> userContentControllerIdentifier;
304 decoder >> userContentControllerIdentifier;
305 if (!userContentControllerIdentifier)
306 return std::nullopt;
307 result.userContentControllerIdentifier = *userContentControllerIdentifier;
308#endif
309
310 std::optional<std::optional<NavigatingToAppBoundDomain>> isNavigatingToAppBoundDomain;
311 decoder >> isNavigatingToAppBoundDomain;
312 if (!isNavigatingToAppBoundDomain)
313 return std::nullopt;
314 result.isNavigatingToAppBoundDomain = *isNavigatingToAppBoundDomain;
315
316 std::optional<std::optional<WebCore::PrivateClickMeasurement::PcmDataCarried>> pcmDataCarried;
317 decoder >> pcmDataCarried;
318 if (!pcmDataCarried)
319 return std::nullopt;
320 result.pcmDataCarried = *pcmDataCarried;
321
322 return result;
323}
