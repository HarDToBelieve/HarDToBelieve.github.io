29namespace WebKit {
30
31void NetworkSchemeRegistry::registerURLSchemeAsCORSEnabled(String&& scheme)
32{
33 m_corsEnabledSchemes.add(WTFMove(scheme));
34}
35
36bool NetworkSchemeRegistry::shouldTreatURLSchemeAsCORSEnabled(StringView scheme)
37{
38 if (scheme.startsWith("http"))
39 return scheme.length() == 4 || (scheme.length() == 5 && scheme[4] == 's');
40 return m_corsEnabledSchemes.contains(scheme.toStringWithoutCopying());
41}
