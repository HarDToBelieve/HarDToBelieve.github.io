38Ref<NetworkSocketStream> NetworkSocketStream::create(NetworkProcess& networkProcess, URL&& url, PAL::SessionID sessionID, const String& credentialPartition, WebSocketIdentifier identifier, IPC::Connection& connection, SourceApplicationAuditToken&& auditData)
39{
40 return adoptRef(*new NetworkSocketStream(networkProcess, WTFMove(url), sessionID, credentialPartition, identifier, connection, WTFMove(auditData)));

43NetworkSocketStream::NetworkSocketStream(NetworkProcess& networkProcess, URL&& url, PAL::SessionID sessionID, const String& credentialPartition, WebSocketIdentifier identifier, IPC::Connection& connection, SourceApplicationAuditToken&& auditData)
44 : m_identifier(identifier)
45 , m_connection(connection)
46 , m_impl(SocketStreamHandleImpl::create(url, *this, sessionID, credentialPartition, WTFMove(auditData), NetworkStorageSessionProvider::create(networkProcess, sessionID).ptr()))
47 , m_delayFailTimer(*this, &NetworkSocketStream::sendDelayedFailMessage)
48{
49}

51void NetworkSocketStream::sendData(const IPC::DataReference& data, uint64_t identifier)
52{
53 m_impl->platformSend(data.data(), data.size(), [this, protectedThis = makeRef(*this), identifier] (bool success) {
54 send(Messages::WebSocketStream::DidSendData(identifier, success));
55 });
56}
57
58void NetworkSocketStream::sendHandshake(const IPC::DataReference& data, const std::optional<CookieRequestHeaderFieldProxy>& headerFieldProxy, uint64_t identifier)
59{
60 m_impl->platformSendHandshake(data.data(), data.size(), headerFieldProxy, [this, protectedThis = makeRef(*this), identifier] (bool success, bool didAccessSecureCookies) {
61 send(Messages::WebSocketStream::DidSendHandshake(identifier, success, didAccessSecureCookies));
62 });
63}

105static constexpr auto closedPortErrorCode = 61;
106
107void NetworkSocketStream::sendDelayedFailMessage()
108{
109 send(Messages::WebSocketStream::DidFailSocketStream(m_closedPortError));
110}
111
112void NetworkSocketStream::didFailSocketStream(SocketStreamHandle& handle, const SocketStreamError& error)
113{
114 ASSERT_UNUSED(handle, &handle == m_impl.ptr());
115
116 if (error.errorCode() == closedPortErrorCode) {
117 m_closedPortError = error;
118 m_delayFailTimer.startOneShot(NetworkProcess::randomClosedPortDelay());
119 } else
120 send(Messages::WebSocketStream::DidFailSocketStream(error));
121}
