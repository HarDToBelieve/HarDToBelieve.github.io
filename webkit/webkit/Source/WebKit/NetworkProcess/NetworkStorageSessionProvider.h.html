26#pragma once
27
28#include "NetworkProcess.h"
29#include <WebCore/NetworkStorageSession.h>
30#include <WebCore/StorageSessionProvider.h>
31#include <pal/SessionID.h>
32#include <wtf/WeakPtr.h>
33
34namespace WebKit {
35
36class NetworkStorageSessionProvider final : public WebCore::StorageSessionProvider {
37public:
38 static Ref<NetworkStorageSessionProvider> create(NetworkProcess& networkProcess, PAL::SessionID sessionID) { return adoptRef(*new NetworkStorageSessionProvider(networkProcess, sessionID)); }
39
40private:
41 NetworkStorageSessionProvider(NetworkProcess& networkProcess, PAL::SessionID sessionID)
42 : m_networkProcess(makeWeakPtr(networkProcess))
43 , m_sessionID(sessionID) { }
44
45 WebCore::NetworkStorageSession* storageSession() const final
46 {
47 if (m_networkProcess)
48 return m_networkProcess->storageSession(m_sessionID);
49 return nullptr;
50 }
51
52 WeakPtr<NetworkProcess> m_networkProcess;
53 PAL::SessionID m_sessionID;
54};
