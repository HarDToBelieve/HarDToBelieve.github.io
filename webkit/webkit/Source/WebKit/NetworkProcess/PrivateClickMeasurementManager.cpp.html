55PrivateClickMeasurementManager::PrivateClickMeasurementManager(NetworkSession& networkSession, NetworkProcess& networkProcess, PAL::SessionID sessionID, Function<void(NetworkLoadParameters&&, NetworkLoadCallback&&)>&& networkLoadFunction)
56 : m_firePendingAttributionRequestsTimer(*this, &PrivateClickMeasurementManager::firePendingAttributionRequests)
57 , m_networkSession(makeWeakPtr(networkSession))
58 , m_networkProcess(networkProcess)
59 , m_sessionID(sessionID)
60 , m_networkLoadFunction(WTFMove(networkLoadFunction))
61{
62 ASSERT(m_networkLoadFunction);
63
64 // We should send any pending attributions on session-start in case their

66 // delay in database startup.

70void PrivateClickMeasurementManager::storeUnattributed(PrivateClickMeasurement&& attribution)
71{
72 if (!featureEnabled())
73 return;
74
75 clearExpired();
76
77 if (attribution.ephemeralSourceNonce()) {
78 auto attributionCopy = attribution;
79 // This is guaranteed to be close in time to the navigational click which makes it likely to be personally identifiable.
80 getTokenPublicKey(WTFMove(attributionCopy), PrivateClickMeasurement::AttributionReportEndpoint::Source, PrivateClickMeasurement::PcmDataCarried::PersonallyIdentifiable, [weakThis = makeWeakPtr(*this), this] (PrivateClickMeasurement&& attribution, const String& publicKeyBase64URL) {
81 if (!weakThis)
82 return;
83
84 if (publicKeyBase64URL.isEmpty())
85 return;
86
87 if (m_fraudPreventionValuesForTesting)
88 attribution.setSourceUnlinkableTokenValue(m_fraudPreventionValuesForTesting->unlinkableToken);
89#if PLATFORM(COCOA)
90 else {
91 if (!attribution.calculateAndUpdateSourceUnlinkableToken(publicKeyBase64URL))
92 return;
93 }
94#endif
95
96 getSignedUnlinkableToken(WTFMove(attribution));
97 });
98 }
99
100 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, "[Private Click Measurement] Storing a click."_s);
101
102#if ENABLE(RESOURCE_LOAD_STATISTICS)
103 if (auto* resourceLoadStatistics = m_networkSession->resourceLoadStatistics())
104 resourceLoadStatistics->insertPrivateClickMeasurement(WTFMove(attribution), PrivateClickMeasurementAttributionType::Unattributed);
105#endif
106}

108static NetworkLoadParameters generateNetworkLoadParameters(URL&& url, const String& httpMethod, RefPtr<JSON::Object>&& jsonPayload, PrivateClickMeasurement::PcmDataCarried dataTypeCarried, bool isDebugModeEnabled)
109{
110 ResourceRequest request { WTFMove(url) };
111 request.setHTTPMethod(httpMethod);
112 request.setHTTPHeaderField(HTTPHeaderName::CacheControl, WebCore::HTTPHeaderValues::maxAge0());
113 if (jsonPayload) {
114 request.setHTTPContentType(WebCore::HTTPHeaderValues::applicationJSONContentType());
115 request.setHTTPBody(WebCore::FormData::create(jsonPayload->toJSONString().utf8().data()));
116 }
117
118 NetworkLoadParameters loadParameters;
119 loadParameters.request = request;
120 loadParameters.parentPID = presentingApplicationPID();
121 loadParameters.storedCredentialsPolicy = StoredCredentialsPolicy::EphemeralStateless;
122 loadParameters.shouldClearReferrerOnHTTPSToHTTPRedirect = true;
123 loadParameters.pcmDataCarried = UNLIKELY(isDebugModeEnabled) ? PrivateClickMeasurement::PcmDataCarried::PersonallyIdentifiable : dataTypeCarried;
124
125 return loadParameters;
126}

128static NetworkLoadParameters generateNetworkLoadParametersForHttpPost(URL&& url, Ref<JSON::Object>&& jsonPayload, PrivateClickMeasurement::PcmDataCarried dataTypeCarried, bool isDebugModeEnabled)
129{
130 return generateNetworkLoadParameters(WTFMove(url), "POST"_s, WTFMove(jsonPayload), dataTypeCarried, isDebugModeEnabled);
131}

133static NetworkLoadParameters generateNetworkLoadParametersForHttpGet(URL&& url, PrivateClickMeasurement::PcmDataCarried dataTypeCarried, bool isDebugModeEnabled)
134{
135 return generateNetworkLoadParameters(WTFMove(url), "GET"_s, nullptr, dataTypeCarried, isDebugModeEnabled);

138void PrivateClickMeasurementManager::getTokenPublicKey(PrivateClickMeasurement&& attribution, PrivateClickMeasurement::AttributionReportEndpoint attributionReportEndpoint, PrivateClickMeasurement::PcmDataCarried pcmDataCarried, Function<void(PrivateClickMeasurement&& attribution, const String& publicKeyBase64URL)>&& callback)
139{
140 if (!featureEnabled())
141 return;
142
143 auto tokenPublicKeyURL = attribution.tokenPublicKeyURL();
144 if (m_tokenPublicKeyURLForTesting) {
145 if (attributionReportEndpoint == PrivateClickMeasurement::AttributionReportEndpoint::Destination)
146 return;
147 tokenPublicKeyURL = *m_tokenPublicKeyURLForTesting;
148 // FIXME(225364)
149 pcmDataCarried = PrivateClickMeasurement::PcmDataCarried::NonPersonallyIdentifiable;
150 }
151
152 if (tokenPublicKeyURL.isEmpty() || !tokenPublicKeyURL.isValid())
153 return;
154
155 auto loadParameters = generateNetworkLoadParametersForHttpGet(WTFMove(tokenPublicKeyURL), pcmDataCarried, debugModeEnabled());
156
157 RELEASE_LOG_INFO(PrivateClickMeasurement, "About to fire a token public key request.");
158 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, "[Private Click Measurement] About to fire a token public key request."_s);
159
160 m_networkLoadFunction(WTFMove(loadParameters), [weakThis = makeWeakPtr(*this), this, attribution = WTFMove(attribution), callback = WTFMove(callback)] (auto& error, auto& response, auto& jsonObject) mutable {
161 if (!weakThis)
162 return;
163
164 if (!error.isNull()) {
165 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Error, makeString("[Private Click Measurement] Received error: '"_s, error.localizedDescription(), "' for token public key request."_s));
166 return;
167 }
168
169 if (!jsonObject) {
170 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Error, makeString("[Private Click Measurement] JSON response is empty for token public key request."_s));
171 return;
172 }
173
174 callback(WTFMove(attribution), jsonObject->getString("token_public_key"_s));
175 });
176

179void PrivateClickMeasurementManager::getSignedUnlinkableToken(PrivateClickMeasurement&& attribution)
180{
181 if (!featureEnabled())
182 return;
183
184 // This is guaranteed to be close in time to the navigational click which makes it likely to be personally identifiable.
185 auto pcmDataCarried = PrivateClickMeasurement::PcmDataCarried::PersonallyIdentifiable;
186 auto tokenSignatureURL = attribution.tokenSignatureURL();
187 if (m_tokenSignatureURLForTesting) {
188 tokenSignatureURL = *m_tokenSignatureURLForTesting;
189 // FIXME(225364)
190 pcmDataCarried = PrivateClickMeasurement::PcmDataCarried::NonPersonallyIdentifiable;
191 }
192
193 if (tokenSignatureURL.isEmpty() || !tokenSignatureURL.isValid())
194 return;
195
196 auto loadParameters = generateNetworkLoadParametersForHttpPost(WTFMove(tokenSignatureURL), attribution.tokenSignatureJSON(), pcmDataCarried, debugModeEnabled());
197
198 RELEASE_LOG_INFO(PrivateClickMeasurement, "About to fire a unlinkable token signing request.");
199 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, "[Private Click Measurement] About to fire a unlinkable token signing request."_s);
200
201 m_networkLoadFunction(WTFMove(loadParameters), [weakThis = makeWeakPtr(*this), this, attribution = WTFMove(attribution)] (auto& error, auto& response, auto& jsonObject) mutable {
202 if (!weakThis)
203 return;
204
205 if (!error.isNull()) {
206 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Error, makeString("[Private Click Measurement] Received error: '"_s, error.localizedDescription(), "' for secret token signing request."_s));
207 return;
208 }
209
210 if (!jsonObject) {
211 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Error, makeString("[Private Click Measurement] JSON response is empty for token signing request."_s));
212 return;
213 }
214
215 auto signatureBase64URL = jsonObject->getString("unlinkable_token"_s);
216 if (signatureBase64URL.isEmpty())
217 return;
218
219 // FIX NOW!
220 if (m_fraudPreventionValuesForTesting)
221 attribution.setSourceSecretToken({ m_fraudPreventionValuesForTesting->secretToken, m_fraudPreventionValuesForTesting->signature, m_fraudPreventionValuesForTesting->keyID });
222#if PLATFORM(COCOA)
223 else {
224 if (!attribution.calculateAndUpdateSourceSecretToken(signatureBase64URL))
225 return;
226 }
227#endif
228
229 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, "[Private Click Measurement] Storing a secret token."_s);
230
231#if ENABLE(RESOURCE_LOAD_STATISTICS)
232 if (auto* resourceLoadStatistics = m_networkSession->resourceLoadStatistics())
233 resourceLoadStatistics->insertPrivateClickMeasurement(WTFMove(attribution), PrivateClickMeasurementAttributionType::Unattributed);
234#endif
235 });

239void PrivateClickMeasurementManager::handleAttribution(AttributionTriggerData&& attributionTriggerData, const URL& requestURL, const WebCore::ResourceRequest& redirectRequest)
240{
241 if (!featureEnabled())
242 return;
243
244 RegistrableDomain redirectDomain { redirectRequest.url() };
245 auto& firstPartyURL = redirectRequest.firstPartyForCookies();
246
247 if (!redirectDomain.matches(requestURL)) {
248 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Warning, "[Private Click Measurement] Attribution was not accepted because the HTTP redirect was not same-site."_s);
249 return;
250 }
251
252 if (redirectDomain.matches(firstPartyURL)) {
253 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Warning, "[Private Click Measurement] Attribution was not accepted because it was requested in an HTTP redirect that is same-site as the first-party."_s);
254 return;
255 }
256
257 attribute(SourceSite { WTFMove(redirectDomain) }, AttributionDestinationSite { firstPartyURL }, WTFMove(attributionTriggerData));
258}

265void PrivateClickMeasurementManager::attribute(const SourceSite& sourceSite, const AttributionDestinationSite& destinationSite, AttributionTriggerData&& attributionTriggerData)
266{
267#if ENABLE(RESOURCE_LOAD_STATISTICS)
268 if (!featureEnabled())
269 return;
270
271 if (auto* resourceLoadStatistics = m_networkSession->resourceLoadStatistics()) {
272 resourceLoadStatistics->attributePrivateClickMeasurement(sourceSite, destinationSite, WTFMove(attributionTriggerData), [this, weakThis = makeWeakPtr(*this)] (auto attributionSecondsUntilSendData) {
273 if (!weakThis)
274 return;
275
276 if (!attributionSecondsUntilSendData)
277 return;
278
279 if (attributionSecondsUntilSendData.value().hasValidSecondsUntilSendValues()) {
280 auto minSecondsUntilSend = attributionSecondsUntilSendData.value().minSecondsUntilSend();
281
282 ASSERT(minSecondsUntilSend);
283 if (!minSecondsUntilSend)
284 return;
285
286 if (m_firePendingAttributionRequestsTimer.isActive() && m_firePendingAttributionRequestsTimer.nextFireInterval() < *minSecondsUntilSend)
287 return;
288
289 if (UNLIKELY(debugModeEnabled())) {
290 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, makeString("[Private Click Measurement] Setting timer for firing attribution request to the debug mode timeout of "_s, debugModeSecondsUntilSend.seconds(), " seconds where the regular timeout would have been "_s, minSecondsUntilSend.value().seconds(), " seconds."_s));
291 minSecondsUntilSend = debugModeSecondsUntilSend;
292 } else
293 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, makeString("[Private Click Measurement] Setting timer for firing attribution request to the timeout of "_s, minSecondsUntilSend.value().seconds(), " seconds."_s));
294
295 startTimer(*minSecondsUntilSend);
296 }
297 });
298 }
299#endif

302void PrivateClickMeasurementManager::fireConversionRequest(const PrivateClickMeasurement& attribution, PrivateClickMeasurement::AttributionReportEndpoint attributionReportEndpoint)
303{
304 if (!featureEnabled())
305 return;
306
307 if (!attribution.sourceUnlinkableToken()) {
308 fireConversionRequestImpl(attribution, attributionReportEndpoint);
309 return;
310 }
311
312 auto attributionCopy = attribution;
313 // This happens out of webpage context and with a long delay and is thus unlikely to be personally identifiable.
314 getTokenPublicKey(WTFMove(attributionCopy), attributionReportEndpoint, PrivateClickMeasurement::PcmDataCarried::NonPersonallyIdentifiable, [weakThis = makeWeakPtr(*this), this, attributionReportEndpoint] (PrivateClickMeasurement&& attribution, const String& publicKeyBase64URL) {
315 if (!weakThis)
316 return;
317
318 auto publicKeyData = base64URLDecode(publicKeyBase64URL);
319 if (!publicKeyData)
320 return;
321
322 auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
323 crypto->addBytes(publicKeyData->data(), publicKeyData->size());
324 auto publicKeyDataHash = crypto->computeHash();
325
326 auto keyID = base64URLEncodeToString(publicKeyDataHash.data(), publicKeyDataHash.size());
327 if (keyID != attribution.sourceUnlinkableToken()->keyIDBase64URL)
328 return;
329
330 fireConversionRequestImpl(attribution, attributionReportEndpoint);
331 });
332}

334void PrivateClickMeasurementManager::fireConversionRequestImpl(const PrivateClickMeasurement& attribution, PrivateClickMeasurement::AttributionReportEndpoint attributionReportEndpoint)
335{
336 URL attributionURL;
337 switch (attributionReportEndpoint) {
338 case PrivateClickMeasurement::AttributionReportEndpoint::Source:
339 attributionURL = m_attributionReportTestConfig ? m_attributionReportTestConfig->attributionReportSourceURL : attribution.attributionReportSourceURL();
340 break;
341 case PrivateClickMeasurement::AttributionReportEndpoint::Destination:
342 attributionURL = m_attributionReportTestConfig ? m_attributionReportTestConfig->attributionReportAttributeOnURL : attribution.attributionReportAttributeOnURL();
343 }
344
345 if (attributionURL.isEmpty() || !attributionURL.isValid())
346 return;
347
348 auto loadParameters = generateNetworkLoadParametersForHttpPost(WTFMove(attributionURL), attribution.attributionReportJSON(), PrivateClickMeasurement::PcmDataCarried::NonPersonallyIdentifiable, debugModeEnabled());
349
350 RELEASE_LOG_INFO(PrivateClickMeasurement, "About to fire an attribution request.");
351 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Log, "[Private Click Measurement] About to fire an attribution request."_s);
352
353 m_networkLoadFunction(WTFMove(loadParameters), [weakThis = makeWeakPtr(*this), this](auto& error, auto& response, auto&) {
354 if (!weakThis)
355 return;
356
357 if (!error.isNull()) {
358 m_networkProcess->broadcastConsoleMessage(m_sessionID, MessageSource::PrivateClickMeasurement, MessageLevel::Error, makeString("[Private Click Measurement] Received error: '"_s, error.localizedDescription(), "' for ad click attribution request."_s));
359 }
360 });
361}

374void PrivateClickMeasurementManager::firePendingAttributionRequests()
375{
376#if ENABLE(RESOURCE_LOAD_STATISTICS)
377 if (!featureEnabled())
378 return;
379
380 auto* resourceLoadStatistics = m_networkSession->resourceLoadStatistics();
381 if (!resourceLoadStatistics)
382 return;
383
384 resourceLoadStatistics->allAttributedPrivateClickMeasurement([this, weakThis = makeWeakPtr(*this)] (auto&& attributions) {
385 if (!weakThis)
386 return;
387 auto nextTimeToFire = Seconds::infinity();
388 bool hasSentAttribution = false;
389
390 for (auto& attribution : attributions) {
391 std::optional<WallTime> earliestTimeToSend = attribution.timesToSend().earliestTimeToSend();
392 std::optional<WebCore::PrivateClickMeasurement::AttributionReportEndpoint> attributionReportEndpoint = attribution.timesToSend().attributionReportEndpoint();
393
394 if (!earliestTimeToSend || !attributionReportEndpoint) {
395 ASSERT_NOT_REACHED();
396 continue;
397 }
398
399 auto now = WallTime::now();
400 if (*earliestTimeToSend <= now || m_isRunningTest || debugModeEnabled()) {
401 if (hasSentAttribution) {
402 // We've already sent an attribution this round. We should send additional overdue attributions at
403 // a random time between 15 and 30 minutes to avoid a burst of simultaneous attributions. If debug
404 // mode is enabled, this should be much shorter for easy testing.
405 auto interval = debugModeEnabled() ? debugModeSecondsUntilSend : 15_min + Seconds(cryptographicallyRandomNumber() % 900);
406 startTimer(interval);
407 return;
408 }
409
410 auto laterTimeToSend = attribution.timesToSend().latestTimeToSend();
411 fireConversionRequest(attribution, *attributionReportEndpoint);
412 clearSentAttribution(WTFMove(attribution), *attributionReportEndpoint);
413 hasSentAttribution = true;
414
415 // Update nextTimeToFire in case the later report time for this attribution is sooner than the scheduled next time to fire.
416 // Or, if debug mode is enabled, we should send the second report on a much shorter delay for easy testing.
417 if (laterTimeToSend)
418 nextTimeToFire = debugModeEnabled() ? debugModeSecondsUntilSend : std::min(nextTimeToFire, laterTimeToSend.value().secondsSinceEpoch());
419
420 continue;
421 }
422
423 auto seconds = *earliestTimeToSend - now;
424 nextTimeToFire = std::min(nextTimeToFire, seconds);
425
426 // Attributions are sorted by earliestTimeToSend, so the first time we hit this there can be no further attributions
427 // due for reporting, and nextTimeToFire is the minimum earliestTimeToSend value for any pending attribution.
428 break;
429 }
430
431 if (nextTimeToFire < Seconds::infinity())
432 startTimer(nextTimeToFire);
433 });
434#endif
435}
