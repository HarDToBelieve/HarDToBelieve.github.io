52ServiceWorkerFetchTask::ServiceWorkerFetchTask(WebSWServerConnection& swServerConnection, NetworkResourceLoader& loader, ResourceRequest&& request, SWServerConnectionIdentifier serverConnectionIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, ServiceWorkerRegistrationIdentifier serviceWorkerRegistrationIdentifier, bool shouldSoftUpdate)
53 : m_swServerConnection(makeWeakPtr(swServerConnection))
54 , m_loader(loader)
55 , m_fetchIdentifier(WebCore::FetchIdentifier::generate())
56 , m_serverConnectionIdentifier(serverConnectionIdentifier)
57 , m_serviceWorkerIdentifier(serviceWorkerIdentifier)
58 , m_currentRequest(WTFMove(request))
59 , m_timeoutTimer(*this, &ServiceWorkerFetchTask::timeoutTimerFired)
60 , m_serviceWorkerRegistrationIdentifier(serviceWorkerRegistrationIdentifier)
61 , m_shouldSoftUpdate(shouldSoftUpdate)
62{
63 SWFETCH_RELEASE_LOG("ServiceWorkerFetchTask: (serverConnectionIdentifier=%" PRIu64 ", serviceWorkerRegistrationIdentifier=%" PRIu64 ", serviceWorkerIdentifier=%" PRIu64 ")", m_serverConnectionIdentifier.toUInt64(), m_serviceWorkerRegistrationIdentifier.toUInt64(), m_serviceWorkerIdentifier.toUInt64());
64 m_timeoutTimer.startOneShot(loader.connectionToWebProcess().networkProcess().serviceWorkerFetchTimeout());
65}

67ServiceWorkerFetchTask::~ServiceWorkerFetchTask()
68{
69 SWFETCH_RELEASE_LOG("~ServiceWorkerFetchTask:");
70 if (m_serviceWorkerConnection)
71 m_serviceWorkerConnection->unregisterFetch(*this);
72}

74template<typename Message> bool ServiceWorkerFetchTask::sendToServiceWorker(Message&& message)
75{
76 return m_serviceWorkerConnection ? m_serviceWorkerConnection->ipcConnection().send(std::forward<Message>(message), 0) : false;
77}
78
79template<typename Message> bool ServiceWorkerFetchTask::sendToClient(Message&& message)
80{
81 return m_loader.connectionToWebProcess().connection().send(std::forward<Message>(message), m_loader.identifier());
82}

84void ServiceWorkerFetchTask::start(WebSWServerToContextConnection& serviceWorkerConnection)
85{
86 SWFETCH_RELEASE_LOG("start:");
87 m_serviceWorkerConnection = makeWeakPtr(serviceWorkerConnection);
88 serviceWorkerConnection.registerFetch(*this);
89
90 startFetch();
91}

93void ServiceWorkerFetchTask::contextClosed()
94{
95 SWFETCH_RELEASE_LOG("contextClosed: (m_isDone=%d, m_wasHandled=%d)", m_isDone, m_wasHandled);
96 m_serviceWorkerConnection = nullptr;
97 if (m_isDone)
98 return;
99
100 if (m_wasHandled) {
101 didFail(ResourceError { errorDomainWebKitInternal, 0, { }, "Service Worker context closed"_s });
102 return;
103 }
104 cannotHandle();
105}

107void ServiceWorkerFetchTask::startFetch()
108{
109 SWFETCH_RELEASE_LOG("startFetch");
110 m_loader.consumeSandboxExtensionsIfNeeded();
111 auto& options = m_loader.parameters().options;
112 auto referrer = m_currentRequest.httpReferrer();
113
114 // We are intercepting fetch calls after going through the HTTP layer, which may add some specific headers.
115 auto request = m_currentRequest;
116 cleanHTTPRequestHeadersForAccessControl(request, m_loader.parameters().httpHeadersToKeep);
117
118 bool isSent = sendToServiceWorker(Messages::WebSWContextManagerConnection::StartFetch { m_serverConnectionIdentifier, m_serviceWorkerIdentifier, m_fetchIdentifier, request, options, IPC::FormDataReference { m_currentRequest.httpBody() }, referrer });
119 ASSERT_UNUSED(isSent, isSent);
120}

122void ServiceWorkerFetchTask::didReceiveRedirectResponse(ResourceResponse&& response)
123{
124 if (m_isDone)
125 return;
126
127 SWFETCH_RELEASE_LOG("didReceiveRedirectResponse:");
128 m_wasHandled = true;
129 m_timeoutTimer.stop();
130 softUpdateIfNeeded();
131
132 response.setSource(ResourceResponse::Source::ServiceWorker);
133 auto newRequest = m_currentRequest.redirectedRequest(response, m_loader.parameters().shouldClearReferrerOnHTTPSToHTTPRedirect);
134
135 sendToClient(Messages::WebResourceLoader::WillSendRequest { newRequest, IPC::FormDataReference { newRequest.httpBody() }, response });
136}

138void ServiceWorkerFetchTask::didReceiveResponse(ResourceResponse&& response, bool needsContinueDidReceiveResponseMessage)
139{
140 if (m_isDone)
141 return;
142
143 SWFETCH_RELEASE_LOG("didReceiveResponse: (httpStatusCode=%d, MIMEType=%" PUBLIC_LOG_STRING ", expectedContentLength=%" PRId64 ", needsContinueDidReceiveResponseMessage=%d, source=%u)", response.httpStatusCode(), response.mimeType().utf8().data(), response.expectedContentLength(), needsContinueDidReceiveResponseMessage, static_cast<unsigned>(response.source()));
144 m_wasHandled = true;
145 m_timeoutTimer.stop();
146 softUpdateIfNeeded();
147
148 response.setSource(ResourceResponse::Source::ServiceWorker);
149 sendToClient(Messages::WebResourceLoader::DidReceiveResponse { response, needsContinueDidReceiveResponseMessage });
150}

170void ServiceWorkerFetchTask::didFinish()
171{
172 ASSERT(!m_timeoutTimer.isActive());
173 SWFETCH_RELEASE_LOG("didFinish:");
174
175 m_isDone = true;
176 m_timeoutTimer.stop();
177 sendToClient(Messages::WebResourceLoader::DidFinishResourceLoad { { } });
178}
179
180void ServiceWorkerFetchTask::didFail(const ResourceError& error)
181{
182 m_isDone = true;
183 if (m_timeoutTimer.isActive()) {
184 m_timeoutTimer.stop();
185 softUpdateIfNeeded();
186 }
187 SWFETCH_RELEASE_LOG_ERROR("didFail: (error.domain=%" PUBLIC_LOG_STRING ", error.code=%d)", error.domain().utf8().data(), error.errorCode());
188 m_loader.didFailLoading(error);
189}
190
191void ServiceWorkerFetchTask::didNotHandle()
192{
193 if (m_isDone)
194 return;
195
196 SWFETCH_RELEASE_LOG("didNotHandle:");
197 m_isDone = true;
198 m_timeoutTimer.stop();
199 softUpdateIfNeeded();
200
201 m_loader.serviceWorkerDidNotHandle(this);
202}

204void ServiceWorkerFetchTask::cannotHandle()
205{
206 SWFETCH_RELEASE_LOG("cannotHandle:");
207 // Make sure we call didNotHandle asynchronously because failing synchronously would get the NetworkResourceLoader in a bad state.
208 RunLoop::main().dispatch([weakThis = makeWeakPtr(this)] {
209 if (weakThis)
210 weakThis->didNotHandle();
211 });
212}

226void ServiceWorkerFetchTask::continueFetchTaskWith(ResourceRequest&& request)
227{
228 SWFETCH_RELEASE_LOG("continueFetchTaskWith: (hasServiceWorkerConnection=%d)", !!m_serviceWorkerConnection);
229 if (!m_serviceWorkerConnection) {
230 m_loader.serviceWorkerDidNotHandle(this);
231 return;
232 }
233 m_timeoutTimer.startOneShot(m_loader.connectionToWebProcess().networkProcess().serviceWorkerFetchTimeout());
234 m_currentRequest = WTFMove(request);
235 startFetch();
236}
237
238void ServiceWorkerFetchTask::timeoutTimerFired()
239{
240 ASSERT(!m_isDone);
241 ASSERT(!m_wasHandled);
242 SWFETCH_RELEASE_LOG_ERROR("timeoutTimerFired: (hasServiceWorkerConnection=%d)", !!m_serviceWorkerConnection);
243
244 softUpdateIfNeeded();
245
246 cannotHandle();
247
248 if (m_swServerConnection)
249 m_swServerConnection->fetchTaskTimedOut(serviceWorkerIdentifier());
250}
251
252void ServiceWorkerFetchTask::softUpdateIfNeeded()
253{
254 SWFETCH_RELEASE_LOG("softUpdateIfNeeded: (m_shouldSoftUpdate=%d)", m_shouldSoftUpdate);
255 if (!m_shouldSoftUpdate)
256 return;
257 if (auto* registration = m_loader.connectionToWebProcess().swConnection().server().getRegistration(m_serviceWorkerRegistrationIdentifier))
258 registration->scheduleSoftUpdate(m_loader.isAppBound() ? WebCore::IsAppBound::Yes : WebCore::IsAppBound::No);
259}
