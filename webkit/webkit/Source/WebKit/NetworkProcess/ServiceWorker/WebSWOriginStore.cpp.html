35namespace WebKit {
36
37using namespace WebCore;
38
39WebSWOriginStore::WebSWOriginStore()
40 : m_store(*this)
41{
42}
43
44void WebSWOriginStore::addToStore(const SecurityOriginData& origin)
45{
46 m_store.scheduleAddition(computeSharedStringHash(origin.toString()));
47 m_store.flushPendingChanges();
48}
49
50void WebSWOriginStore::removeFromStore(const SecurityOriginData& origin)
51{
52 m_store.scheduleRemoval(computeSharedStringHash(origin.toString()));
53 m_store.flushPendingChanges();
54}
55
56void WebSWOriginStore::clearStore()
57{
58 m_store.clear();
59}
60
61void WebSWOriginStore::importComplete()
62{
63 m_isImported = true;
64 for (auto& connection : m_webSWServerConnections)
65 connection.send(Messages::WebSWClientConnection::SetSWOriginTableIsImported());
66}
67
68void WebSWOriginStore::registerSWServerConnection(WebSWServerConnection& connection)
69{
70 m_webSWServerConnections.add(connection);
71
72 if (!m_store.isEmpty())
73 sendStoreHandle(connection);
74
75 if (m_isImported)
76 connection.send(Messages::WebSWClientConnection::SetSWOriginTableIsImported());
77}
78
79void WebSWOriginStore::unregisterSWServerConnection(WebSWServerConnection& connection)
80{
81 m_webSWServerConnections.remove(connection);
82}
83
84void WebSWOriginStore::sendStoreHandle(WebSWServerConnection& connection)
85{
86 SharedMemory::Handle handle;
87 if (!m_store.createSharedMemoryHandle(handle))
88 return;
89
90#if OS(DARWIN) || OS(WINDOWS)
91 uint64_t dataSize = handle.size();
92#else
93 uint64_t dataSize = 0;
94#endif
95 connection.send(Messages::WebSWClientConnection::SetSWOriginTableSharedMemory(SharedMemory::IPCHandle { WTFMove(handle), dataSize }));
96}
97
98void WebSWOriginStore::didInvalidateSharedMemory()
99{
100 for (auto& connection : m_webSWServerConnections)
101 sendStoreHandle(connection);
102}
