76WebSWServerConnection::~WebSWServerConnection()
77{
78 m_networkProcess->unregisterSWServerConnection(*this);
79 for (const auto& keyValue : m_clientOrigins)
80 server().unregisterServiceWorkerClient(keyValue.value, keyValue.key);
81 for (auto& completionHandler : m_unregisterJobs.values())
82 completionHandler(false);
83}

139void WebSWServerConnection::controlClient(ServiceWorkerClientIdentifier clientIdentifier, SWServerRegistration& registration, const ResourceRequest& request)
140{
141 // As per step 12 of https://w3c.github.io/ServiceWorker/#on-fetch-request-algorithm, the active service worker should be controlling the document.
142 // We register a temporary service worker client using the identifier provided by DocumentLoader and notify DocumentLoader about it.
143 // If notification is successful, DocumentLoader will unregister the temporary service worker client just after the document is created and registered as a client.
144 sendWithAsyncReply(Messages::WebSWClientConnection::SetDocumentIsControlled { clientIdentifier.contextIdentifier, registration.data() }, [weakThis = makeWeakPtr(this), this, clientIdentifier](bool isSuccess) {
145 if (!weakThis || isSuccess)
146 return;
147 unregisterServiceWorkerClient(clientIdentifier);
148 });
149
150 ServiceWorkerClientData data { clientIdentifier, ServiceWorkerClientType::Window, ServiceWorkerClientFrameType::None, request.url(), request.isAppBound() ? WebCore::LastNavigationWasAppBound::Yes : WebCore::LastNavigationWasAppBound::No };
151 registerServiceWorkerClient(SecurityOriginData { registration.key().topOrigin() }, WTFMove(data), registration.identifier(), request.httpUserAgent());
152}

154std::unique_ptr<ServiceWorkerFetchTask> WebSWServerConnection::createFetchTask(NetworkResourceLoader& loader, const ResourceRequest& request)
155{
156 if (loader.parameters().serviceWorkersMode == ServiceWorkersMode::None)
157 return nullptr;
158
159 if (!server().canHandleScheme(loader.originalRequest().url().protocol()))
160 return nullptr;
161
162 std::optional<ServiceWorkerRegistrationIdentifier> serviceWorkerRegistrationIdentifier;
163 if (loader.parameters().options.mode == FetchOptions::Mode::Navigate) {
164 auto topOrigin = loader.parameters().isMainFrameNavigation ? SecurityOriginData::fromURL(request.url()) : loader.parameters().topOrigin->data();
165 auto* registration = doRegistrationMatching(topOrigin, request.url());
166 if (!registration)
167 return nullptr;
168
169 serviceWorkerRegistrationIdentifier = registration->identifier();
170 controlClient(ServiceWorkerClientIdentifier { loader.connectionToWebProcess().webProcessIdentifier(), *loader.parameters().options.clientIdentifier }, *registration, request);
171 } else {
172 if (!loader.parameters().serviceWorkerRegistrationIdentifier)
173 return nullptr;
174
175 if (isPotentialNavigationOrSubresourceRequest(loader.parameters().options.destination))
176 return nullptr;
177
178 serviceWorkerRegistrationIdentifier = *loader.parameters().serviceWorkerRegistrationIdentifier;
179 }
180
181 auto* worker = server().activeWorkerFromRegistrationID(*serviceWorkerRegistrationIdentifier);
182 if (!worker) {
183 SWSERVERCONNECTION_RELEASE_LOG_ERROR("startFetch: DidNotHandle because no active worker for registration %" PRIu64, serviceWorkerRegistrationIdentifier->toUInt64());
184 return nullptr;
185 }
186
187 auto* registration = server().getRegistration(*serviceWorkerRegistrationIdentifier);
188 bool shouldSoftUpdate = registration && registration->shouldSoftUpdate(loader.parameters().options);
189 if (worker->shouldSkipFetchEvent()) {
190 if (shouldSoftUpdate)
191 registration->scheduleSoftUpdate(loader.isAppBound() ? WebCore::IsAppBound::Yes : WebCore::IsAppBound::No);
192
193 return nullptr;
194 }
195
196 if (worker->hasTimedOutAnyFetchTasks()) {
197 SWSERVERCONNECTION_RELEASE_LOG_ERROR("startFetch: DidNotHandle because worker %" PRIu64 " has some timeouts", worker->identifier().toUInt64());
198 return nullptr;
199 }
200
201 auto task = makeUnique<ServiceWorkerFetchTask>(*this, loader, ResourceRequest { request }, identifier(), worker->identifier(), *serviceWorkerRegistrationIdentifier, shouldSoftUpdate);
202 startFetch(*task, *worker);
203 return task;
204}

206void WebSWServerConnection::startFetch(ServiceWorkerFetchTask& task, SWServerWorker& worker)
207{
208 auto runServerWorkerAndStartFetch = [weakThis = makeWeakPtr(this), this, task = makeWeakPtr(task)](bool success) mutable {
209 if (!task)
210 return;
211
212 if (!weakThis) {
213 task->cannotHandle();
214 return;
215 }
216
217 if (!success) {
218 SWSERVERCONNECTION_RELEASE_LOG_ERROR("startFetch: fetchIdentifier=%" PRIu64 " DidNotHandle because worker did not become activated", task->fetchIdentifier().toUInt64());
219 task->cannotHandle();
220 return;
221 }
222
223 auto* worker = server().workerByID(task->serviceWorkerIdentifier());
224 if (!worker || worker->hasTimedOutAnyFetchTasks()) {
225 task->cannotHandle();
226 return;
227 }
228
229 if (!worker->contextConnection())
230 server().createContextConnection(worker->registrableDomain());
231
232 auto identifier = task->serviceWorkerIdentifier();
233 server().runServiceWorkerIfNecessary(identifier, [weakThis = WTFMove(weakThis), this, task = WTFMove(task)](auto* contextConnection) mutable {
234 if (!task)
235 return;
236
237 if (!weakThis) {
238 task->cannotHandle();
239 return;
240 }
241
242 if (!contextConnection) {
243 SWSERVERCONNECTION_RELEASE_LOG_ERROR("startFetch: fetchIdentifier=%s DidNotHandle because failed to run service worker", task->fetchIdentifier().loggingString().utf8().data());
244 task->cannotHandle();
245 return;
246 }
247 SWSERVERCONNECTION_RELEASE_LOG("startFetch: Starting fetch %" PRIu64 " via service worker %" PRIu64, task->fetchIdentifier().toUInt64(), task->serviceWorkerIdentifier().toUInt64());
248 static_cast<WebSWServerToContextConnection&>(*contextConnection).startFetch(*task);
249 });
250 };
251
252 worker.whenActivated(WTFMove(runServerWorkerAndStartFetch));
253}

255void WebSWServerConnection::postMessageToServiceWorker(ServiceWorkerIdentifier destinationIdentifier, MessageWithMessagePorts&& message, const ServiceWorkerOrClientIdentifier& sourceIdentifier)
256{
257 auto* destinationWorker = server().workerByID(destinationIdentifier);
258 if (!destinationWorker)
259 return;
260
261 std::optional<ServiceWorkerOrClientData> sourceData;
262 WTF::switchOn(sourceIdentifier, [&](ServiceWorkerIdentifier identifier) {
263 if (auto* sourceWorker = server().workerByID(identifier))
264 sourceData = ServiceWorkerOrClientData { sourceWorker->data() };
265 }, [&](ServiceWorkerClientIdentifier identifier) {
266 if (auto clientData = destinationWorker->findClientByIdentifier(identifier))
267 sourceData = ServiceWorkerOrClientData { *clientData };
268 });
269
270 if (!sourceData)
271 return;
272
273 // It's possible this specific worker cannot be re-run (e.g. its registration has been removed)
274 server().runServiceWorkerIfNecessary(destinationIdentifier, [destinationIdentifier, message = WTFMove(message), sourceData = WTFMove(*sourceData)](auto* contextConnection) mutable {
275 if (contextConnection)
276 sendToContextProcess(*contextConnection, Messages::WebSWContextManagerConnection::PostMessageToServiceWorker { destinationIdentifier, WTFMove(message), WTFMove(sourceData) });
277 });
278}

280void WebSWServerConnection::scheduleJobInServer(ServiceWorkerJobData&& jobData)
281{
282 ASSERT(!jobData.scopeURL.isNull());
283 if (jobData.scopeURL.isNull()) {
284 rejectJobInClient(jobData.identifier().jobIdentifier, ExceptionData { InvalidStateError, "Scope URL is empty"_s });
285 return;
286 }
287
288 SWSERVERCONNECTION_RELEASE_LOG("Scheduling ServiceWorker job %s in server", jobData.identifier().loggingString().utf8().data());
289 ASSERT(identifier() == jobData.connectionIdentifier());
290
291 server().scheduleJob(WTFMove(jobData));
292}

294URL WebSWServerConnection::clientURLFromIdentifier(DocumentOrWorkerIdentifier contextIdentifier)
295{
296 return WTF::switchOn(contextIdentifier, [&](DocumentIdentifier documentIdentifier) -> URL {
297 ServiceWorkerClientIdentifier clientIdentifier { identifier(), documentIdentifier };
298
299 auto iterator = m_clientOrigins.find(clientIdentifier);
300 if (iterator == m_clientOrigins.end())
301 return { };
302
303 auto clientData = server().serviceWorkerClientWithOriginByID(iterator->value, clientIdentifier);
304 if (!clientData)
305 return { };
306
307 return clientData->url;
308 }, [&](ServiceWorkerIdentifier serviceWorkerIdentifier) -> URL {
309 auto* worker = server().workerByID(serviceWorkerIdentifier);
310 if (!worker)
311 return { };
312 return worker->data().scriptURL;
313 });
314}

316void WebSWServerConnection::scheduleUnregisterJobInServer(ServiceWorkerJobIdentifier jobIdentifier, ServiceWorkerRegistrationIdentifier registrationIdentifier, DocumentOrWorkerIdentifier contextIdentifier, CompletionHandler<void(UnregisterJobResult&&)>&& completionHandler)
317{
318 SWSERVERCONNECTION_RELEASE_LOG("Scheduling unregister ServiceWorker job in server");
319
320 auto* registration = server().getRegistration(registrationIdentifier);
321 if (!registration)
322 return completionHandler(false);
323
324 auto clientURL = clientURLFromIdentifier(contextIdentifier);
325 if (!clientURL.isValid())
326 return completionHandler(makeUnexpected(ExceptionData { InvalidStateError, "Client is unknown"_s }));
327
328 ASSERT(!m_unregisterJobs.contains(jobIdentifier));
329 m_unregisterJobs.add(jobIdentifier, WTFMove(completionHandler));
330
331 server().scheduleUnregisterJob(ServiceWorkerJobDataIdentifier { identifier(), jobIdentifier }, *registration, contextIdentifier, WTFMove(clientURL));
332}

363void WebSWServerConnection::registerServiceWorkerClient(SecurityOriginData&& topOrigin, ServiceWorkerClientData&& data, const std::optional<ServiceWorkerRegistrationIdentifier>& controllingServiceWorkerRegistrationIdentifier, String&& userAgent)
364{
365 auto contextOrigin = SecurityOriginData::fromURL(data.url);
366 bool isNewOrigin = WTF::allOf(m_clientOrigins.values(), [&contextOrigin](auto& origin) {
367 return contextOrigin != origin.clientOrigin;
368 });
369 auto* contextConnection = isNewOrigin ? server().contextConnectionForRegistrableDomain(RegistrableDomain { contextOrigin }) : nullptr;
370
371 auto clientOrigin = ClientOrigin { WTFMove(topOrigin), WTFMove(contextOrigin) };
372 m_clientOrigins.add(data.identifier, clientOrigin);
373 server().registerServiceWorkerClient(WTFMove(clientOrigin), WTFMove(data), controllingServiceWorkerRegistrationIdentifier, WTFMove(userAgent));
374
375 if (!m_isThrottleable)
376 updateThrottleState();
377
378 if (contextConnection) {
379 auto& connection = static_cast<WebSWServerToContextConnection&>(*contextConnection);
380 m_networkProcess->parentProcessConnection()->send(Messages::NetworkProcessProxy::RegisterServiceWorkerClientProcess { identifier(), connection.webProcessIdentifier() }, 0);
381 }
382}

384void WebSWServerConnection::unregisterServiceWorkerClient(const ServiceWorkerClientIdentifier& clientIdentifier)
385{
386 auto iterator = m_clientOrigins.find(clientIdentifier);
387 if (iterator == m_clientOrigins.end())
388 return;
389
390 auto clientOrigin = iterator->value;
391
392 server().unregisterServiceWorkerClient(clientOrigin, clientIdentifier);
393 m_clientOrigins.remove(iterator);
394
395 if (!m_isThrottleable)
396 updateThrottleState();
397
398 bool isDeletedOrigin = WTF::allOf(m_clientOrigins.values(), [&clientOrigin](auto& origin) {
399 return clientOrigin.clientOrigin != origin.clientOrigin;
400 });
401
402 if (isDeletedOrigin) {
403 if (auto* contextConnection = server().contextConnectionForRegistrableDomain(RegistrableDomain { clientOrigin.clientOrigin })) {
404 auto& connection = static_cast<WebSWServerToContextConnection&>(*contextConnection);
405 m_networkProcess->parentProcessConnection()->send(Messages::NetworkProcessProxy::UnregisterServiceWorkerClientProcess { identifier(), connection.webProcessIdentifier() }, 0);
406 }
407 }
408}

410bool WebSWServerConnection::hasMatchingClient(const RegistrableDomain& domain) const
411{
412 return WTF::anyOf(m_clientOrigins.values(), [&domain](auto& origin) {
413 return domain.matches(origin.clientOrigin);
414 });

417bool WebSWServerConnection::computeThrottleState(const RegistrableDomain& domain) const
418{
419 return WTF::allOf(server().connections().values(), [&domain](auto& serverConnection) {
420 auto& connection = static_cast<WebSWServerConnection&>(*serverConnection);
421 return connection.isThrottleable() || !connection.hasMatchingClient(domain);
422 });
423}

425void WebSWServerConnection::setThrottleState(bool isThrottleable)
426{
427 m_isThrottleable = isThrottleable;
428 updateThrottleState();
429}
430
431void WebSWServerConnection::updateThrottleState()
432{
433 HashSet<SecurityOriginData> origins;
434 for (auto& origin : m_clientOrigins.values())
435 origins.add(origin.clientOrigin);
436
437 for (auto& origin : origins) {
438 if (auto* contextConnection = server().contextConnectionForRegistrableDomain(RegistrableDomain { origin })) {
439 auto& connection = static_cast<WebSWServerToContextConnection&>(*contextConnection);
440
441 if (connection.isThrottleable() == m_isThrottleable)
442 continue;
443 bool newThrottleState = computeThrottleState(connection.registrableDomain());
444 if (connection.isThrottleable() == newThrottleState)
445 continue;
446 connection.setThrottleState(newThrottleState);
447 }
448 }
449}

451void WebSWServerConnection::contextConnectionCreated(SWServerToContextConnection& contextConnection)
452{
453 auto& connection = static_cast<WebSWServerToContextConnection&>(contextConnection);
454 connection.setThrottleState(computeThrottleState(connection.registrableDomain()));
455
456 if (hasMatchingClient(connection.registrableDomain()))
457 m_networkProcess->parentProcessConnection()->send(Messages::NetworkProcessProxy::RegisterServiceWorkerClientProcess { identifier(), connection.webProcessIdentifier() }, 0);
458}
