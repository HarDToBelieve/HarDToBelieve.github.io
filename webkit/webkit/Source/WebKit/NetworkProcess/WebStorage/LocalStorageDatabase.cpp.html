64bool LocalStorageDatabase::openDatabase(ShouldCreateDatabase shouldCreateDatabase)
65{
66 ASSERT(!RunLoop::isMain());
67 if (!FileSystem::fileExists(m_databasePath)) {
68 if (shouldCreateDatabase == ShouldCreateDatabase::No)
69 return true;
70 m_items = HashMap<String, String> { };
71 }
72
73 if (m_databasePath.isEmpty()) {
74 LOG_ERROR("Filename for local storage database is empty - cannot open for persistent storage");
75 return false;
76 }
77
78 if (!m_database.open(m_databasePath)) {
79 LOG_ERROR("Failed to open database file %s for local storage", m_databasePath.utf8().data());
80 return false;
81 }
82
83 // Since a WorkQueue isn't bound to a specific thread, we have to disable threading checks
84 // even though we never access the database from different threads simultaneously.
85 m_database.disableThreadingChecks();
86
87 if (!migrateItemTableIfNeeded()) {
88 // We failed to migrate the item table. In order to avoid trying to migrate the table over and over,
89 // just delete it and start from scratch.
90 if (!m_database.executeCommand("DROP TABLE ItemTable"_s))
91 LOG_ERROR("Failed to delete table ItemTable for local storage");
92 }
93
94 if (!m_database.executeCommand("CREATE TABLE IF NOT EXISTS ItemTable (key TEXT UNIQUE ON CONFLICT REPLACE, value BLOB NOT NULL ON CONFLICT FAIL)"_s)) {
95 LOG_ERROR("Failed to create table ItemTable for local storage");
96 return false;
97 }
98
99 if (m_quotaInBytes != WebCore::StorageMap::noQuota)
100 m_database.setMaximumSize(m_quotaInBytes);
101
102 return true;
103}

105void LocalStorageDatabase::startTransactionIfNecessary()
106{
107 if (!m_transaction)
108 m_transaction = makeUnique<SQLiteTransaction>(m_database);
109
110 if (m_transaction->inProgress())
111 return;
112
113 m_transaction->begin();
114 m_workQueue->dispatchAfter(transactionDuration, [weakThis = makeWeakPtr(*this)] {
115 if (weakThis)
116 weakThis->m_transaction->commit();
117 });
118}

120bool LocalStorageDatabase::migrateItemTableIfNeeded()
121{
122 ASSERT(!RunLoop::isMain());
123 if (!m_database.tableExists("ItemTable"))
124 return true;
125
126 auto query = m_database.prepareStatement("SELECT value FROM ItemTable LIMIT 1"_s);
127
128 // This query isn't ever executed, it's just used to check the column type.
129 if (query && query->isColumnDeclaredAsBlob(0))
130 return true;
131
132 // Create a new table with the right type, copy all the data over to it and then replace the new table with the old table.
133 static const ASCIILiteral commands[] = {
134 "DROP TABLE IF EXISTS ItemTable2"_s,
135 "CREATE TABLE ItemTable2 (key TEXT UNIQUE ON CONFLICT REPLACE, value BLOB NOT NULL ON CONFLICT FAIL)"_s,
136 "INSERT INTO ItemTable2 SELECT * from ItemTable"_s,
137 "DROP TABLE ItemTable"_s,
138 "ALTER TABLE ItemTable2 RENAME TO ItemTable"_s,
139 ASCIILiteral::null(),
140 };
141
142 SQLiteTransaction transaction(m_database);
143 transaction.begin();
144
145 for (size_t i = 0; commands[i]; ++i) {
146 if (m_database.executeCommand(commands[i]))
147 continue;
148
149 LOG_ERROR("Failed to migrate table ItemTable for local storage when executing: %s", commands[i].characters());
150
151 return false;
152 }
153
154 transaction.commit();
155 return true;
156}

158HashMap<String, String> LocalStorageDatabase::items() const
159{
160 ASSERT(!RunLoop::isMain());
161 if (!m_database.isOpen())
162 return { };
163
164 HashMap<String, String> items;
165 if (m_items) {
166 items.reserveInitialCapacity(m_items->size());
167 for (auto& entry : *m_items) {
168 auto value = entry.value.isNull() ? itemBypassingCache(entry.key) : entry.value;
169 items.add(entry.key, WTFMove(value));
170 }
171 return items;
172 }
173
174 auto query = scopedStatement(m_getItemsStatement, getItemsQueryString);
175 if (!query) {
176 LOG_ERROR("Unable to select items from ItemTable for local storage");
177 return { };
178 }
179
180 m_items = HashMap<String, String> { };
181 int result = query->step();
182 while (result == SQLITE_ROW) {
183 String key = query->columnText(0);
184 String value = query->columnBlobAsString(1);
185 if (!key.isNull() && !value.isNull()) {
186 m_items->add(key, value.sizeInBytes() > maximumSizeForValuesKeptInMemory ? String() : value);
187 items.add(WTFMove(key), WTFMove(value));
188 }
189 result = query->step();
190 }
191
192 if (result != SQLITE_DONE)
193 LOG_ERROR("Error reading items from ItemTable for local storage");
194
195 return items;
196}

198void LocalStorageDatabase::removeItem(const String& key, String& oldValue)
199{
200 ASSERT(!RunLoop::isMain());
201 if (!m_database.isOpen())
202 return;
203
204 startTransactionIfNecessary();
205 oldValue = item(key);
206 if (oldValue.isNull())
207 return;
208
209 auto deleteStatement = scopedStatement(m_deleteItemStatement, "DELETE FROM ItemTable WHERE key=?"_s);
210 if (!deleteStatement) {
211 LOG_ERROR("Failed to prepare delete statement - cannot write to local storage database");
212 return;
213 }
214 deleteStatement->bindText(1, key);
215
216 int result = deleteStatement->step();
217 if (result != SQLITE_DONE) {
218 LOG_ERROR("Failed to delete item in the local storage database - %i", result);
219 return;
220 }
221
222 if (m_items)
223 m_items->remove(key);
224}

226String LocalStorageDatabase::item(const String& key) const
227{
228 ASSERT(!RunLoop::isMain());
229 if (!m_database.isOpen())
230 return { };
231
232 if (m_items) {
233 // Use find() instead of get() since a null String is a valid value here.
234 auto it = m_items->find(key);
235 if (it == m_items->end())
236 return { };
237 if (!it->value.isNull())
238 return it->value;
239 // The value is too large and needs to be fetched from the database.
240 }
241 return itemBypassingCache(key);
242}
243
244String LocalStorageDatabase::itemBypassingCache(const String& key) const
245{
246 auto query = scopedStatement(m_getItemStatement, "SELECT value FROM ItemTable WHERE key=?"_s);
247 if (!query) {
248 LOG_ERROR("Unable to get item from ItemTable for local storage");
249 return { };
250 }
251 query->bindText(1, key);
252
253 int result = query->step();
254 if (result == SQLITE_ROW)
255 return query->columnBlobAsString(0);
256 if (result != SQLITE_DONE)
257 LOG_ERROR("Error get item from ItemTable for local storage");
258 return { };
259}

261void LocalStorageDatabase::setItem(const String& key, const String& value, String& oldValue, bool& quotaException)
262{
263 ASSERT(!RunLoop::isMain());
264 if (!m_database.isOpen())
265 openDatabase(ShouldCreateDatabase::Yes);
266 if (!m_database.isOpen())
267 return;
268
269 startTransactionIfNecessary();
270 oldValue = item(key);
271
272 auto insertStatement = scopedStatement(m_insertStatement, "INSERT INTO ItemTable VALUES (?, ?)"_s);
273 if (!insertStatement) {
274 LOG_ERROR("Failed to prepare insert statement - cannot write to local storage database");
275 return;
276 }
277
278 insertStatement->bindText(1, key);
279 insertStatement->bindBlob(2, value);
280
281 int result = insertStatement->step();
282 if (result != SQLITE_DONE) {
283 LOG_ERROR("Failed to update item in the local storage database - %i", result);
284 if (result == SQLITE_FULL)
285 quotaException = true;
286 return;
287 }
288
289 if (m_items)
290 m_items->set(key, value.sizeInBytes() > maximumSizeForValuesKeptInMemory ? String() : value);
291}

293bool LocalStorageDatabase::clear()
294{
295 ASSERT(!RunLoop::isMain());
296 if (!m_database.isOpen())
297 return false;
298
299 if (m_items && m_items->isEmpty())
300 return false;
301
302 startTransactionIfNecessary();
303 auto clearStatement = scopedStatement(m_clearStatement, "DELETE FROM ItemTable"_s);
304 if (!clearStatement) {
305 LOG_ERROR("Failed to prepare clear statement - cannot write to local storage database");
306 return false;
307 }
308
309 int result = clearStatement->step();
310 if (result != SQLITE_DONE) {
311 LOG_ERROR("Failed to clear all items in the local storage database - %i", result);
312 return false;
313 }
314
315 if (m_items) {
316 m_items->clear();
317 return true;
318 }
319
320 return m_database.lastChanges() > 0;
321}

329void LocalStorageDatabase::close()
330{
331 ASSERT(!RunLoop::isMain());
332 if (m_isClosed)
333 return;
334 m_isClosed = true;
335
336 bool isEmpty = databaseIsEmpty();
337
338 m_clearStatement = nullptr;
339 m_insertStatement = nullptr;
340 m_getItemStatement = nullptr;
341 m_getItemsStatement = nullptr;
342 m_deleteItemStatement = nullptr;
343 m_items = std::nullopt;
344 if (auto transaction = std::exchange(m_transaction, nullptr))
345 transaction->commit();
346
347 if (m_database.isOpen())
348 m_database.close();
349
350 if (isEmpty)
351 SQLiteFileSystem::deleteDatabaseFile(m_databasePath);
352}

354bool LocalStorageDatabase::databaseIsEmpty() const
355{
356 ASSERT(!RunLoop::isMain());
357 if (!m_database.isOpen())
358 return false;
359
360 if (m_items)
361 return m_items->isEmpty();
362
363 auto query = m_database.prepareStatement("SELECT COUNT(*) FROM ItemTable"_s);
364 if (!query) {
365 LOG_ERROR("Unable to count number of rows in ItemTable for local storage");
366 return false;
367 }
368
369 if (query->step() != SQLITE_ROW) {
370 LOG_ERROR("No results when counting number of rows in ItemTable for local storage");
371 return false;
372 }
373
374 return !query->columnInt(0);
375}

377void LocalStorageDatabase::openIfExisting()
378{
379 ASSERT(!RunLoop::isMain());
380 if (m_database.isOpen())
381 return;
382
383 openDatabase(ShouldCreateDatabase::No);
384
385 // Prewarm the getItems statement for performance since the pages block synchronously on retrieving the items.
386 if (m_database.isOpen())
387 scopedStatement(m_getItemsStatement, getItemsQueryString);
388}

390SQLiteStatementAutoResetScope LocalStorageDatabase::scopedStatement(std::unique_ptr<SQLiteStatement>& statement, ASCIILiteral query) const
391{
392 ASSERT(!RunLoop::isMain());
393 if (!statement) {
394 auto statementOrError = m_database.prepareHeapStatement(query);
395 if (!statementOrError)
396 return SQLiteStatementAutoResetScope { };
397 statement = statementOrError.value().moveToUniquePtr();
398 ASSERT(m_database.isOpen());
399 }
400 return SQLiteStatementAutoResetScope { statement.get() };
401}
