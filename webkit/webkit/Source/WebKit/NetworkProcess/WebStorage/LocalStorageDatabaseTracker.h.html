35namespace WebKit {
36
37class LocalStorageDatabaseTracker : public RefCounted<LocalStorageDatabaseTracker> {
38public:
39 static Ref<LocalStorageDatabaseTracker> create(String&& localStorageDirectory);
40 ~LocalStorageDatabaseTracker();
41
42 String databasePath(const WebCore::SecurityOriginData&) const;
43
44 void didOpenDatabaseWithOrigin(const WebCore::SecurityOriginData&);
45 void deleteDatabaseWithOrigin(const WebCore::SecurityOriginData&);
46
47 // Returns a vector of the origins whose databases should be deleted.
48 Vector<WebCore::SecurityOriginData> databasesModifiedSince(WallTime);
49
50 Vector<WebCore::SecurityOriginData> origins() const;
51
52 struct OriginDetails {
53 String originIdentifier;
54 std::optional<WallTime> creationTime;
55 std::optional<WallTime> modificationTime;
56
57 template<class Encoder> void encode(Encoder&) const;
58 template<class Decoder> static std::optional<OriginDetails> decode(Decoder&);
59
60 OriginDetails isolatedCopy() const { return { originIdentifier.isolatedCopy(), creationTime, modificationTime }; }
61 };
62 Vector<OriginDetails> originDetailsCrossThreadCopy();
63
64private:
65 LocalStorageDatabaseTracker(String&& localStorageDirectory);
66
67 String databasePath(const String& filename) const;
68 String localStorageDirectory() const;
69
70 enum DatabaseOpeningStrategy {
71 CreateIfNonExistent,
72 SkipIfNonExistent
73 };
74
75 // It is not safe to use this member from a background thread, call localStorageDirectory() instead.
76 const String m_localStorageDirectory;
77
78#if PLATFORM(IOS_FAMILY)
79 void platformMaybeExcludeFromBackup() const;
80
81 mutable bool m_hasExcludedFromBackup { false };
82#endif
83};
84
85template<class Encoder>
86void LocalStorageDatabaseTracker::OriginDetails::encode(Encoder& encoder) const
87{
88 encoder << originIdentifier << creationTime << modificationTime;
89}
90
91template<class Decoder>
92std::optional<LocalStorageDatabaseTracker::OriginDetails> LocalStorageDatabaseTracker::OriginDetails::decode(Decoder& decoder)
93{
94 LocalStorageDatabaseTracker::OriginDetails result;
95 if (!decoder.decode(result.originIdentifier))
96 return std::nullopt;
97
98 if (!decoder.decode(result.creationTime))
99 return std::nullopt;
100
101 if (!decoder.decode(result.modificationTime))
102 return std::nullopt;
103
104 return result;
105}
