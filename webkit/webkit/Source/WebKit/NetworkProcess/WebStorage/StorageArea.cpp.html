38StorageArea::StorageArea(LocalStorageNamespace* localStorageNamespace, const SecurityOriginData& securityOrigin, unsigned quotaInBytes, Ref<WorkQueue>&& queue)
39 : m_localStorageNamespace(makeWeakPtr(localStorageNamespace))
40 , m_securityOrigin(securityOrigin)
41 , m_quotaInBytes(quotaInBytes)
42 , m_identifier(Identifier::generate())
43 , m_queue(WTFMove(queue))
44{
45 ASSERT(!RunLoop::isMain());
46 if (isEphemeral())
47 m_sessionStorageMap = makeUnique<StorageMap>(m_quotaInBytes);
48}
49
50StorageArea::~StorageArea()
51{
52 ASSERT(!RunLoop::isMain());
53
54 if (m_localStorageDatabase)
55 m_localStorageDatabase->close();
56}
57
58void StorageArea::addListener(IPC::Connection::UniqueID connectionID)
59{
60 ASSERT(!RunLoop::isMain());
61 ASSERT(!m_eventListeners.contains(connectionID));
62
63 if (m_eventListeners.isEmpty() && !isEphemeral())
64 ensureDatabase();
65
66 m_eventListeners.add(connectionID);
67}
68
69void StorageArea::removeListener(IPC::Connection::UniqueID connectionID)
70{
71 ASSERT(!RunLoop::isMain());
72 m_eventListeners.remove(connectionID);
73
74 if (!m_eventListeners.isEmpty())
75 return;
76
77 if (m_localStorageNamespace)
78 m_localStorageNamespace->removeStorageArea(m_securityOrigin);
79}
80
81bool StorageArea::hasListener(IPC::Connection::UniqueID connectionID) const
82{
83 ASSERT(!RunLoop::isMain());
84 return m_eventListeners.contains(connectionID);
85}

87std::unique_ptr<StorageArea> StorageArea::clone() const
88{
89 ASSERT(!RunLoop::isMain());
90 ASSERT(!m_localStorageNamespace);
91
92 auto storageArea = makeUnique<StorageArea>(nullptr, m_securityOrigin, m_quotaInBytes, m_queue.copyRef());
93 *storageArea->m_sessionStorageMap = *m_sessionStorageMap;
94
95 return storageArea;
96}

98void StorageArea::setItem(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier storageAreaImplID, const String& key, const String& value, const String& urlString, bool& quotaException)
99{
100 ASSERT(!RunLoop::isMain());
101
102 String oldValue;
103 if (isEphemeral())
104 m_sessionStorageMap->setItem(key, value, oldValue, quotaException);
105 else
106 ensureDatabase().setItem(key, value, oldValue, quotaException);
107
108 if (quotaException)
109 return;
110
111 dispatchEvents(sourceConnection, storageAreaImplID, key, oldValue, value, urlString);
112}
113
114void StorageArea::removeItem(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier storageAreaImplID, const String& key, const String& urlString)
115{
116 ASSERT(!RunLoop::isMain());
117
118 String oldValue;
119 if (isEphemeral())
120 m_sessionStorageMap->removeItem(key, oldValue);
121 else
122 ensureDatabase().removeItem(key, oldValue);
123 if (oldValue.isNull())
124 return;
125
126 dispatchEvents(sourceConnection, storageAreaImplID, key, oldValue, String(), urlString);
127}
128
129void StorageArea::clear(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier storageAreaImplID, const String& urlString)
130{
131 ASSERT(!RunLoop::isMain());
132
133 if (isEphemeral()) {
134 if (!m_sessionStorageMap->length())
135 return;
136 m_sessionStorageMap->clear();
137 } else {
138 if (!ensureDatabase().clear())
139 return;
140 }
141
142 dispatchEvents(sourceConnection, storageAreaImplID, String(), String(), String(), urlString);
143}

145HashMap<String, String> StorageArea::items() const
146{
147 ASSERT(!RunLoop::isMain());
148 if (isEphemeral())
149 return m_sessionStorageMap->items();
150
151 return ensureDatabase().items();
152}
153
154void StorageArea::clear()
155{
156 ASSERT(!RunLoop::isMain());
157 if (isEphemeral())
158 m_sessionStorageMap->clear();
159 else {
160 if (m_localStorageDatabase) {
161 m_localStorageDatabase->close();
162 m_localStorageDatabase = nullptr;
163 }
164 }
165
166 for (auto& listenerUniqueID : m_eventListeners)
167 IPC::Connection::send(listenerUniqueID, Messages::StorageAreaMap::ClearCache(), m_identifier.toUInt64());
168}
169
170LocalStorageDatabase& StorageArea::ensureDatabase() const
171{
172 ASSERT(!isEphemeral());
173
174 ASSERT(m_localStorageNamespace->storageManager()->localStorageDatabaseTracker());
175 // We open the database here even if we've already imported our items to ensure that the database is open if we need to write to it.
176 if (!m_localStorageDatabase) {
177 auto* localStorageDatabaseTracker = m_localStorageNamespace->storageManager()->localStorageDatabaseTracker();
178 m_localStorageDatabase = LocalStorageDatabase::create(m_queue.copyRef(), localStorageDatabaseTracker->databasePath(m_securityOrigin), m_quotaInBytes);
179 m_localStorageDatabase->openIfExisting();
180 }
181 return *m_localStorageDatabase;
182}
183
184void StorageArea::dispatchEvents(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier storageAreaImplID, const String& key, const String& oldValue, const String& newValue, const String& urlString) const
185{
186 ASSERT(!RunLoop::isMain());
187 ASSERT(storageAreaImplID);
188 for (auto& listenerUniqueID : m_eventListeners) {
189 auto optionalStorageAreaImplID = listenerUniqueID == sourceConnection ? std::make_optional(storageAreaImplID) : std::nullopt;
190 IPC::Connection::send(listenerUniqueID, Messages::StorageAreaMap::DispatchStorageEvent(optionalStorageAreaImplID, key, oldValue, newValue, urlString), m_identifier.toUInt64());
191 }
192}

194void StorageArea::close()
195{
196 if (!m_localStorageDatabase)
197 return;
198
199 m_localStorageDatabase->close();
200}
