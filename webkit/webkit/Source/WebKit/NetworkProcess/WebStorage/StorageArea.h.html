40namespace WebKit {
41
42class LocalStorageDatabase;
43class LocalStorageNamespace;
44
45class StorageArea : public CanMakeWeakPtr<StorageArea> {
46 WTF_MAKE_NONCOPYABLE(StorageArea);
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 using Identifier = StorageAreaIdentifier;
50
51 StorageArea(LocalStorageNamespace*, const WebCore::SecurityOriginData&, unsigned quotaInBytes, Ref<WorkQueue>&&);
52 ~StorageArea();
53
54 const WebCore::SecurityOriginData& securityOrigin() const { return m_securityOrigin; }
55 Identifier identifier() { return m_identifier; }
56
57 void addListener(IPC::Connection::UniqueID);
58 void removeListener(IPC::Connection::UniqueID);
59 bool hasListener(IPC::Connection::UniqueID connectionID) const;
60
61 std::unique_ptr<StorageArea> clone() const;
62
63 void setItem(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier, const String& key, const String& value, const String& urlString, bool& quotaException);
64 void removeItem(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier, const String& key, const String& urlString);
65 void clear(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier, const String& urlString);
66
67 HashMap<String, String> items() const;
68 void clear();
69
70 bool isEphemeral() const { return !m_localStorageNamespace; }
71
72 void close();
73
74 void syncToDatabase();
75 void handleLowMemoryWarning();
76
77private:
78 void dispatchEvents(IPC::Connection::UniqueID sourceConnection, StorageAreaImplIdentifier, const String& key, const String& oldValue, const String& newValue, const String& urlString) const;
79
80 LocalStorageDatabase& ensureDatabase() const;
81
82 // Will be null if the storage area belongs to a session storage namespace or the storage area is in an ephemeral session.
83 WeakPtr<LocalStorageNamespace> m_localStorageNamespace;
84 mutable RefPtr<LocalStorageDatabase> m_localStorageDatabase;
85
86 WebCore::SecurityOriginData m_securityOrigin;
87 unsigned m_quotaInBytes { 0 };
88
89 mutable std::unique_ptr<WebCore::StorageMap> m_sessionStorageMap;
90 HashSet<IPC::Connection::UniqueID> m_eventListeners;
91
92 Identifier m_identifier;
93 Ref<WorkQueue> m_queue;
94};
