52StorageManager::StorageManager(String&& localStorageDirectory)
53{
54 ASSERT(!RunLoop::isMain());
55
56 if (!localStorageDirectory.isNull())
57 m_localStorageDatabaseTracker = LocalStorageDatabaseTracker::create(WTFMove(localStorageDirectory));
58}

75{

77
78 ASSERT(m_sessionStorageNamespaces.contains(storageNamespaceID));
79 m_sessionStorageNamespaces.remove(storageNamespaceID);

82void StorageManager::cloneSessionStorageNamespace(StorageNamespaceIdentifier storageNamespaceID, StorageNamespaceIdentifier newStorageNamespaceID)
83{
84 ASSERT(!RunLoop::isMain());
85
86 SessionStorageNamespace* sessionStorageNamespace = m_sessionStorageNamespaces.get(storageNamespaceID);
87 if (!sessionStorageNamespace)
88 return;
89
90 SessionStorageNamespace* newSessionStorageNamespace = getOrCreateSessionStorageNamespace(newStorageNamespaceID);
91 ASSERT(newSessionStorageNamespace);
92
93 sessionStorageNamespace->cloneTo(*newSessionStorageNamespace);
94}

96HashSet<SecurityOriginData> StorageManager::getSessionStorageOriginsCrossThreadCopy() const
97{
98 ASSERT(!RunLoop::isMain());
99
100 HashSet<SecurityOriginData> origins;
101 for (const auto& sessionStorageNamespace : m_sessionStorageNamespaces.values()) {
102 for (auto& origin : sessionStorageNamespace->origins())
103 origins.add(crossThreadCopy(origin));
104 }
105
106 return origins;
107}

127HashSet<SecurityOriginData> StorageManager::getLocalStorageOriginsCrossThreadCopy() const
128{
129 ASSERT(!RunLoop::isMain());
130
131 HashSet<SecurityOriginData> origins;
132 if (m_localStorageDatabaseTracker) {
133 for (auto& origin : m_localStorageDatabaseTracker->origins())
134 origins.add(origin.isolatedCopy());
135 } else {
136 for (const auto& localStorageNameSpace : m_localStorageNamespaces.values()) {
137 for (auto& origin : localStorageNameSpace->ephemeralOrigins())
138 origins.add(origin.isolatedCopy());
139 }
140 }
141
142 for (auto& transientLocalStorageNamespace : m_transientLocalStorageNamespaces.values()) {
143 for (auto& origin : transientLocalStorageNamespace->origins())
144 origins.add(origin.isolatedCopy());
145 }
146
147 return origins;
148}

159void StorageManager::renameOrigin(const URL& oldURL, const URL& newURL)
160{
161 ASSERT(!RunLoop::isMain());
162 auto oldOrigin = WebCore::SecurityOriginData::fromURL(oldURL);
163 auto newOrigin = WebCore::SecurityOriginData::fromURL(newURL);
164 for (auto& localStorageNamespace : m_localStorageNamespaces.values())
165 localStorageNamespace->flushAndClose(oldOrigin);
166
167 if (auto* tracker = m_localStorageDatabaseTracker.get()) {
168 static const std::array<const char *, 3> suffixes { "", "-shm", "-wal" };
169 for (const auto* suffix : suffixes)
170 FileSystem::moveFile(makeString(tracker->databasePath(oldOrigin), suffix), makeString(tracker->databasePath(newOrigin), suffix));
171 }
172
173 for (auto& localStorageNamespace : m_localStorageNamespaces.values())
174 localStorageNamespace->clearStorageAreasMatchingOrigin(oldOrigin);
175}

177void StorageManager::deleteLocalStorageOriginsModifiedSince(WallTime time)
178{
179 ASSERT(!RunLoop::isMain());
180
181 if (m_localStorageDatabaseTracker) {
182 auto originsToDelete = m_localStorageDatabaseTracker->databasesModifiedSince(time);
183
184 for (auto& transientLocalStorageNamespace : m_transientLocalStorageNamespaces.values())
185 transientLocalStorageNamespace->clearAllStorageAreas();
186
187 for (const auto& origin : originsToDelete) {
188 for (auto& localStorageNamespace : m_localStorageNamespaces.values())
189 localStorageNamespace->clearStorageAreasMatchingOrigin(origin);
190 m_localStorageDatabaseTracker->deleteDatabaseWithOrigin(origin);
191 }
192 } else {
193 for (auto& localStorageNamespace : m_localStorageNamespaces.values())
194 localStorageNamespace->clearAllStorageAreas();
195 }
196}

198void StorageManager::deleteLocalStorageEntriesForOrigins(const Vector<SecurityOriginData>& origins)
199{
200 ASSERT(!RunLoop::isMain());
201
202 for (auto& origin : origins) {
203 for (auto& localStorageNamespace : m_localStorageNamespaces.values())
204 localStorageNamespace->clearStorageAreasMatchingOrigin(origin);
205
206 for (auto& transientLocalStorageNamespace : m_transientLocalStorageNamespaces.values())
207 transientLocalStorageNamespace->clearStorageAreasMatchingOrigin(origin);
208
209 if (m_localStorageDatabaseTracker)
210 m_localStorageDatabaseTracker->deleteDatabaseWithOrigin(origin);
211 }
212}

214StorageArea* StorageManager::createLocalStorageArea(StorageNamespaceIdentifier storageNamespaceID, SecurityOriginData&& origin, Ref<WorkQueue>&& workQueue)
215{
216 ASSERT(!RunLoop::isMain());
217
218 if (auto* localStorageNamespace = getOrCreateLocalStorageNamespace(storageNamespaceID))
219 return &localStorageNamespace->getOrCreateStorageArea(WTFMove(origin), m_localStorageDatabaseTracker ? LocalStorageNamespace::IsEphemeral::No : LocalStorageNamespace::IsEphemeral::Yes, WTFMove(workQueue));
220
221 return nullptr;
222}

224StorageArea* StorageManager::createTransientLocalStorageArea(StorageNamespaceIdentifier storageNamespaceID, SecurityOriginData&& topLevelOrigin, SecurityOriginData&& origin, Ref<WorkQueue>&& workQueue)
225{
226 ASSERT(!RunLoop::isMain());
227 ASSERT((HashMap<StorageNamespaceIdentifier, RefPtr<TransientLocalStorageNamespace>>::isValidKey(storageNamespaceID)));
228
229 if (auto* transientLocalStorageNamespace = getOrCreateTransientLocalStorageNamespace(storageNamespaceID, WTFMove(topLevelOrigin)))
230 return &transientLocalStorageNamespace->getOrCreateStorageArea(WTFMove(origin), WTFMove(workQueue));
231
232 return nullptr;
233}

246LocalStorageNamespace* StorageManager::getOrCreateLocalStorageNamespace(StorageNamespaceIdentifier storageNamespaceID)
247{
248 ASSERT(!RunLoop::isMain());
249
250 if (!m_localStorageNamespaces.isValidKey(storageNamespaceID))
251 return nullptr;
252
253 return m_localStorageNamespaces.ensure(storageNamespaceID, [this, storageNamespaceID]() {
254 return makeUnique<LocalStorageNamespace>(*this, storageNamespaceID);
255 }).iterator->value.get();
256}
257
258TransientLocalStorageNamespace* StorageManager::getOrCreateTransientLocalStorageNamespace(StorageNamespaceIdentifier storageNamespaceID, SecurityOriginData&& topLevelOrigin)
259{
260 ASSERT(!RunLoop::isMain());
261
262 if (!m_transientLocalStorageNamespaces.isValidKey({ storageNamespaceID, topLevelOrigin }))
263 return nullptr;
264
265 return m_transientLocalStorageNamespaces.ensure({ storageNamespaceID, WTFMove(topLevelOrigin) }, [] {
266 return makeUnique<TransientLocalStorageNamespace>();
267 }).iterator->value.get();
268}

270SessionStorageNamespace* StorageManager::getOrCreateSessionStorageNamespace(StorageNamespaceIdentifier storageNamespaceID)
271{
272 ASSERT(!RunLoop::isMain());
273
274 if (!m_sessionStorageNamespaces.isValidKey(storageNamespaceID))
275 return nullptr;
276
277 return m_sessionStorageNamespaces.ensure(storageNamespaceID, [] {
278 // We currently have no limit on session storage.
279 return makeUnique<SessionStorageNamespace>(std::numeric_limits<unsigned>::max());
280 }).iterator->value.get();
281}

289 m_sessionStorageNamespaces.clear();

292Vector<StorageAreaIdentifier> StorageManager::allStorageAreaIdentifiers() const
293{
294 ASSERT(!RunLoop::isMain());
295
296 Vector<StorageAreaIdentifier> identifiers;
297 for (const auto& localStorageNamespace : m_localStorageNamespaces.values()) {
298 for (auto key : localStorageNamespace->storageAreaIdentifiers())
299 identifiers.append(key);
300 }
301
302 for (const auto& trasientLocalStorageNamespace : m_transientLocalStorageNamespaces.values()) {
303 for (auto key : trasientLocalStorageNamespace->storageAreaIdentifiers())
304 identifiers.append(key);
305 }
306
307 for (const auto& sessionStorageNamespace : m_sessionStorageNamespaces.values()) {
308 for (auto key : sessionStorageNamespace->storageAreaIdentifiers())
309 identifiers.append(key);
310 }
311
312 return identifiers;
313}
