36#if HAVE(LSDATABASECONTEXT)
37static LSDatabaseContext *databaseContext()
38{
39 static dispatch_once_t once;
40 static LSDatabaseContext *context = nullptr;
41 dispatch_once(&once, ^{
42 context = [NSClassFromString(@"LSDatabaseContext") sharedDatabaseContext];
43 });
44 return context;
45}
46#endif
47
48LaunchServicesDatabaseObserver::LaunchServicesDatabaseObserver(NetworkProcess&)
49{
50#if HAVE(LSDATABASECONTEXT)
51 if (![databaseContext() respondsToSelector:@selector(addDatabaseChangeObserver4WebKit:)])
52 return;
53
54 m_observer = [databaseContext() addDatabaseChangeObserver4WebKit:^(xpc_object_t change) {
55 auto message = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
56 xpc_dictionary_set_string(message.get(), XPCEndpoint::xpcMessageNameKey, LaunchServicesDatabaseXPCConstants::xpcUpdateLaunchServicesDatabaseMessageName);
57 xpc_dictionary_set_value(message.get(), LaunchServicesDatabaseXPCConstants::xpcLaunchServicesDatabaseKey, change);
58
59 Locker locker { m_connectionsLock };
60 for (auto& connection : m_connections) {
61 RELEASE_ASSERT(xpc_get_type(connection.get()) == XPC_TYPE_CONNECTION);
62 xpc_connection_send_message(connection.get(), message.get());
63 }
64 }];
65#endif
66}

73void LaunchServicesDatabaseObserver::startObserving(OSObjectPtr<xpc_connection_t> connection)
74{
75 {
76 Locker locker { m_connectionsLock };
77 m_connections.append(connection);
78 }
79
80#if HAVE(LSDATABASECONTEXT)
81 if (![databaseContext() respondsToSelector:@selector(addDatabaseChangeObserver4WebKit:)]) {
82 auto message = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
83 xpc_dictionary_set_string(message.get(), XPCEndpoint::xpcMessageNameKey, LaunchServicesDatabaseXPCConstants::xpcUpdateLaunchServicesDatabaseMessageName);
84 xpc_connection_send_message(connection.get(), message.get());
85 return;
86 }
87
88 RetainPtr<id> observer = [databaseContext() addDatabaseChangeObserver4WebKit:^(xpc_object_t change) {
89 auto message = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
90 xpc_dictionary_set_string(message.get(), XPCEndpoint::xpcMessageNameKey, LaunchServicesDatabaseXPCConstants::xpcUpdateLaunchServicesDatabaseMessageName);
91 xpc_dictionary_set_value(message.get(), LaunchServicesDatabaseXPCConstants::xpcLaunchServicesDatabaseKey, change);
92
93 xpc_connection_send_message(connection.get(), message.get());
94 }];
95
96 [databaseContext() removeDatabaseChangeObserver4WebKit:observer.get()];
97#else
98 auto message = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
99 xpc_dictionary_set_string(message.get(), XPCEndpoint::xpcMessageNameKey, LaunchServicesDatabaseXPCConstants::xpcUpdateLaunchServicesDatabaseMessageName);
100 xpc_connection_send_message(connection.get(), message.get());
101#endif
102}
103
104LaunchServicesDatabaseObserver::~LaunchServicesDatabaseObserver()
105{
106#if HAVE(LSDATABASECONTEXT)
107 if (![databaseContext() respondsToSelector:@selector(removeDatabaseChangeObserver4WebKit:)])
108 return;
109
110 [databaseContext() removeDatabaseChangeObserver4WebKit:m_observer.get()];
111#endif
112}

129void LaunchServicesDatabaseObserver::handleEvent(xpc_connection_t connection, xpc_object_t event)
130{
131 if (xpc_get_type(event) == XPC_TYPE_ERROR) {
132 if (event != XPC_ERROR_CONNECTION_INVALID && event != XPC_ERROR_TERMINATION_IMMINENT)
133 return;
134
135 Locker locker { m_connectionsLock };
136 for (size_t i = 0; i < m_connections.size(); i++) {
137 if (m_connections[i].get() == connection) {
138 m_connections.remove(i);
139 break;
140 }
141 }
142 return;
143 }
144 if (xpc_get_type(event) == XPC_TYPE_DICTIONARY) {
145 String messageName = xpc_dictionary_get_string(event, xpcMessageNameKey);
146 if (messageName != LaunchServicesDatabaseXPCConstants::xpcRequestLaunchServicesDatabaseUpdateMessageName)
147 return;
148 startObserving(connection);
149 }
150}
