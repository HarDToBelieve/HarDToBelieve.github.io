55static void initializeNetworkSettings()
56{
57 static const unsigned preferredConnectionCount = 6;
58
59 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPLoadWidth, preferredConnectionCount);
60
61 Boolean keyExistsAndHasValidFormat = false;
62 Boolean prefValue = CFPreferencesGetAppBooleanValue(CFSTR("WebKitEnableHTTPPipelining"), kCFPreferencesCurrentApplication, &keyExistsAndHasValidFormat);
63 if (keyExistsAndHasValidFormat)
64 WebCore::ResourceRequest::setHTTPPipeliningEnabled(prefValue);
65
66 if (WebCore::ResourceRequest::resourcePrioritiesEnabled()) {
67 const unsigned fastLaneConnectionCount = 1;
68
69 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPPriorityNumLevels, WebCore::resourceLoadPriorityCount);
70 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPMinimumFastLanePriority, toPlatformRequestPriority(WebCore::ResourceLoadPriority::Medium));
71 _CFNetworkHTTPConnectionCacheSetLimit(kHTTPNumFastLanes, fastLaneConnectionCount);
72 }
73}

75void NetworkProcess::platformInitializeNetworkProcessCocoa(const NetworkProcessCreationParameters& parameters)
76{
77#if PLATFORM(IOS_FAMILY)
78 SandboxExtension::consumePermanently(parameters.cookieStorageDirectoryExtensionHandle);
79 SandboxExtension::consumePermanently(parameters.containerCachesDirectoryExtensionHandle);
80 SandboxExtension::consumePermanently(parameters.parentBundleDirectoryExtensionHandle);
81 SandboxExtension::consumePermanently(parameters.tempDirectoryExtensionHandle);
82#endif
83
84 _CFNetworkSetATSContext(parameters.networkATSContext.get());
85
86 m_uiProcessBundleIdentifier = parameters.uiProcessBundleIdentifier;
87
88 initializeNetworkSettings();
89
90#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
91 setSharedHTTPCookieStorage(parameters.uiProcessCookieStorageIdentifier);
92#endif
93
94 // Allow the network process to materialize files stored in the cloud so that loading/reading such files actually succeeds.
95 FileSystem::setAllowsMaterializingDatalessFiles(true, FileSystem::PolicyScope::Process);
96
97 // FIXME: Most of what this function does for cache size gets immediately overridden by setCacheModel().
98 // - memory cache size passed from UI process is always ignored;
99 // - disk cache size passed from UI process is effectively a minimum size.
100 // One non-obvious constraint is that we need to use -setSharedURLCache: even in testing mode, to prevent creating a default one on disk later, when some other code touches the cache.
101
102 m_cacheOptions = { NetworkCache::CacheOption::RegisterNotify };
103
104 // Disable NSURLCache.
105 auto urlCache(adoptNS([[NSURLCache alloc] initWithMemoryCapacity:0 diskCapacity:0 diskPath:nil]));
106 [NSURLCache setSharedURLCache:urlCache.get()];
107}

109RetainPtr<CFDataRef> NetworkProcess::sourceApplicationAuditData() const
110{
111#if USE(SOURCE_APPLICATION_AUDIT_DATA)
112 ASSERT(parentProcessConnection());
113 if (!parentProcessConnection())
114 return nullptr;
115 std::optional<audit_token_t> auditToken = parentProcessConnection()->getAuditToken();
116 if (!auditToken)
117 return nullptr;
118 return adoptCF(CFDataCreate(nullptr, (const UInt8*)&*auditToken, sizeof(*auditToken)));
119#else
120 return nullptr;
121#endif
122}

124#if !HAVE(HSTS_STORAGE)
125static void filterPreloadHSTSEntry(const void* key, const void* value, void* context)
126{
127 RELEASE_ASSERT(context);
128
129 ASSERT(key);
130 ASSERT(value);
131 if (!key || !value)
132 return;
133
134 ASSERT(key != kCFNull);
135 if (key == kCFNull)
136 return;
137
138 auto* hostnames = static_cast<HashSet<String>*>(context);
139 auto val = static_cast<CFDictionaryRef>(value);
140 if (CFDictionaryGetValue(val, _kCFNetworkHSTSPreloaded) != kCFBooleanTrue)
141 hostnames->add((CFStringRef)key);
142}
143#endif

145HashSet<String> NetworkProcess::hostNamesWithHSTSCache(PAL::SessionID sessionID) const
146{
147 HashSet<String> hostNames;
148#if HAVE(HSTS_STORAGE)
149 if (auto* networkSession = static_cast<NetworkSessionCocoa*>(this->networkSession(sessionID))) {
150 for (NSString *host in networkSession->hstsStorage().nonPreloadedHosts)
151 hostNames.add(host);
152 }
153#else
154 if (auto* session = storageSession(sessionID)) {
155 if (auto HSTSPolicies = adoptCF(_CFNetworkCopyHSTSPolicies(session->platformSession())))
156 CFDictionaryApplyFunction(HSTSPolicies.get(), filterPreloadHSTSEntry, &hostNames);
157 }
158#endif
159 return hostNames;
160}

162void NetworkProcess::deleteHSTSCacheForHostNames(PAL::SessionID sessionID, const Vector<String>& hostNames)
163{
164#if HAVE(HSTS_STORAGE)
165 if (auto* networkSession = static_cast<NetworkSessionCocoa*>(this->networkSession(sessionID))) {
166 for (auto& hostName : hostNames)
167 [networkSession->hstsStorage() resetHSTSForHost:hostName];
168 }
169#else
170 if (auto* session = storageSession(sessionID)) {
171 for (auto& hostName : hostNames) {
172 auto url = URL({ }, makeString("https://", hostName));
173 _CFNetworkResetHSTS(url.createCFURL().get(), session->platformSession());
174 }
175 }
176#endif
177}

179void NetworkProcess::clearHSTSCache(PAL::SessionID sessionID, WallTime modifiedSince)
180{
181 NSTimeInterval timeInterval = modifiedSince.secondsSinceEpoch().seconds();
182 NSDate *date = [NSDate dateWithTimeIntervalSince1970:timeInterval];
183#if HAVE(HSTS_STORAGE)
184 if (auto* networkSession = static_cast<NetworkSessionCocoa*>(this->networkSession(sessionID)))
185 [networkSession->hstsStorage() resetHSTSHostsSinceDate:date];
186#else
187 if (auto* session = storageSession(sessionID))
188 _CFNetworkResetHSTSHostsSinceDate(session->platformSession(), (__bridge CFDateRef)date);
189#endif
190}

192void NetworkProcess::clearDiskCache(WallTime modifiedSince, CompletionHandler<void()>&& completionHandler)
193{
194 if (!m_clearCacheDispatchGroup)
195 m_clearCacheDispatchGroup = adoptOSObject(dispatch_group_create());
196
197 auto group = m_clearCacheDispatchGroup.get();
198 dispatch_group_async(group, dispatch_get_main_queue(), makeBlockPtr([this, protectedThis = makeRef(*this), modifiedSince, completionHandler = WTFMove(completionHandler)] () mutable {
199 auto aggregator = CallbackAggregator::create(WTFMove(completionHandler));
200 forEachNetworkSession([modifiedSince, &aggregator](NetworkSession& session) {
201 if (auto* cache = session.cache())
202 cache->clear(modifiedSince, [aggregator] () { });
203 });
204 }).get());
205}

215void NetworkProcess::flushCookies(PAL::SessionID sessionID, CompletionHandler<void()>&& completionHandler)
216{
217 platformFlushCookies(sessionID, WTFMove(completionHandler));
218}
219
220void saveCookies(NSHTTPCookieStorage *cookieStorage, CompletionHandler<void()>&& completionHandler)
221{
222 ASSERT(RunLoop::isMain());
223 ASSERT(cookieStorage);
224 [cookieStorage _saveCookies:makeBlockPtr([completionHandler = WTFMove(completionHandler)]() mutable {
225 // CFNetwork may call the completion block on a background queue, so we need to redispatch to the main thread.
226 RunLoop::main().dispatch(WTFMove(completionHandler));
227 }).get()];
228}
229
230void NetworkProcess::platformFlushCookies(PAL::SessionID sessionID, CompletionHandler<void()>&& completionHandler)
231{
232 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
233 if (auto* networkStorageSession = storageSession(sessionID))
234 saveCookies(networkStorageSession->nsCookieStorage(), WTFMove(completionHandler));
235 else
236 completionHandler();
237}
