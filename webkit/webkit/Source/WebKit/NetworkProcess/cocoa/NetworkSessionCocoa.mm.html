103static NSURLSessionResponseDisposition toNSURLSessionResponseDisposition(WebCore::PolicyAction disposition)
104{
105 switch (disposition) {
106 case WebCore::PolicyAction::StopAllLoads:
107 ASSERT_NOT_REACHED();
108#if !ASSERT_ENABLED
109 FALLTHROUGH;
110#endif
111 case WebCore::PolicyAction::Ignore:
112 return NSURLSessionResponseCancel;
113 case WebCore::PolicyAction::Use:
114 return NSURLSessionResponseAllow;
115 case WebCore::PolicyAction::Download:
116 return NSURLSessionResponseBecomeDownload;
117 }
118}

127 case WebKit::AuthenticationChallengeDisposition::Cancel:
128 return NSURLSessionAuthChallengeCancelAuthenticationChallenge;
129 case WebKit::AuthenticationChallengeDisposition::RejectProtectionSpaceAndContinue:
130 return NSURLSessionAuthChallengeRejectProtectionSpace;
131 }
132}

134static WebCore::NetworkLoadPriority toNetworkLoadPriority(float priority)
135{
136 if (priority <= NSURLSessionTaskPriorityLow)
137 return WebCore::NetworkLoadPriority::Low;
138 if (priority >= NSURLSessionTaskPriorityHigh)
139 return WebCore::NetworkLoadPriority::High;
140 return WebCore::NetworkLoadPriority::Medium;
141}

559static void updateIgnoreStrictTransportSecuritySetting(RetainPtr<NSURLRequest>& request, bool shouldIgnoreHSTS)
560{
561 auto scheme = request.get().URL.scheme;
562 if ([scheme isEqualToString:@"https"]) {
563 if (shouldIgnoreHSTS && ignoreHSTS(request.get())) {
564 // The request was upgraded for some other reason than HSTS.
565 // Don't ignore HSTS to avoid the risk of another downgrade.
566 setIgnoreHSTS(request, false);
567 }

571 }

682- (void)URLSession:(NSURLSession *)session task:(NSURLSessionTask *)task didReceiveChallenge:(NSURLAuthenticationChallenge *)challenge completionHandler:(void (^)(NSURLSessionAuthChallengeDisposition disposition, NSURLCredential *credential))completionHandler
683{
684 auto* sessionCocoa = [self sessionFromTask: task];
685 if (!sessionCocoa || [task state] == NSURLSessionTaskStateCanceling) {
686 completionHandler(NSURLSessionAuthChallengeCancelAuthenticationChallenge, nil);
687 return;
688 }
689
690 auto taskIdentifier = task.taskIdentifier;
691 LOG(NetworkSession, "%llu didReceiveChallenge", taskIdentifier);
692
693 // Proxy authentication is handled by CFNetwork internally. We can get here if the user cancels
694 // CFNetwork authentication dialog, and we shouldn't ask the client to display another one in that case.
695 if (challenge.protectionSpace.isProxy && !sessionCocoa->preventsSystemHTTPProxyAuthentication()) {
696 completionHandler(NSURLSessionAuthChallengeUseCredential, nil);
697 return;
698 }
699
700 NegotiatedLegacyTLS negotiatedLegacyTLS = NegotiatedLegacyTLS::No;
701
702 if ([challenge.protectionSpace.authenticationMethod isEqualToString:NSURLAuthenticationMethodServerTrust]) {
703 if (NetworkSessionCocoa::allowsSpecificHTTPSCertificateForHost(challenge))
704 return completionHandler(NSURLSessionAuthChallengeUseCredential, [NSURLCredential credentialForTrust:challenge.protectionSpace.serverTrust]);
705
706#if HAVE(TLS_PROTOCOL_VERSION_T)
707 NSURLSessionTaskTransactionMetrics *metrics = task._incompleteTaskMetrics.transactionMetrics.lastObject;
708 auto tlsVersion = (tls_protocol_version_t)metrics.negotiatedTLSProtocolVersion.unsignedShortValue;
709 if (tlsVersion == tls_protocol_version_TLSv10 || tlsVersion == tls_protocol_version_TLSv11)
710 negotiatedLegacyTLS = NegotiatedLegacyTLS::Yes;
711#endif
712 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
713 if (negotiatedLegacyTLS == NegotiatedLegacyTLS::No && [task respondsToSelector:@selector(_TLSNegotiatedProtocolVersion)]) {
714 SSLProtocol tlsVersion = [task _TLSNegotiatedProtocolVersion];
715 if (tlsVersion == kTLSProtocol11 || tlsVersion == kTLSProtocol1)
716 negotiatedLegacyTLS = NegotiatedLegacyTLS::Yes;
717 }
718 ALLOW_DEPRECATED_DECLARATIONS_END
719
720 if (negotiatedLegacyTLS == NegotiatedLegacyTLS::Yes && task._preconnect)
721 return completionHandler(NSURLSessionAuthChallengeCancelAuthenticationChallenge, nil);
722
723 // Handle server trust evaluation at platform-level if requested, for performance reasons and to use ATS defaults.
724 if (sessionCocoa->fastServerTrustEvaluationEnabled() && negotiatedLegacyTLS == NegotiatedLegacyTLS::No) {
725 auto* networkDataTask = [self existingTask:task];
726 if (networkDataTask) {
727 NSURLProtectionSpace *protectionSpace = challenge.protectionSpace;
728 networkDataTask->didNegotiateModernTLS(URL(URL(), makeString(String(protectionSpace.protocol), "://", String(protectionSpace.host), ':', protectionSpace.port)));
729 }
730#if HAVE(CFNETWORK_NSURLSESSION_STRICTRUSTEVALUATE)
731 auto decisionHandler = makeBlockPtr([weakSelf = WeakObjCPtr<WKNetworkSessionDelegate>(self), sessionCocoa = makeWeakPtr(sessionCocoa), completionHandler = makeBlockPtr(completionHandler), taskIdentifier, networkDataTask = makeRefPtr(networkDataTask), negotiatedLegacyTLS](NSURLAuthenticationChallenge *challenge, OSStatus trustResult) mutable {
732 auto strongSelf = weakSelf.get();
733 if (!strongSelf)
734 return completionHandler(NSURLSessionAuthChallengeCancelAuthenticationChallenge, nil);
735 auto task = WTFMove(networkDataTask);
736 auto* session = sessionCocoa.get();
737 if (trustResult == noErr || !session) {
738 completionHandler(NSURLSessionAuthChallengePerformDefaultHandling, nil);
739 return;
740 }
741 processServerTrustEvaluation(*session, *strongSelf->_sessionWrapper, challenge, negotiatedLegacyTLS, taskIdentifier, task.get(), WTFMove(completionHandler));
742 });
743 [NSURLSession _strictTrustEvaluate:challenge queue:[NSOperationQueue mainQueue].underlyingQueue completionHandler:decisionHandler.get()];
744 return;
745#else
746 return completionHandler(NSURLSessionAuthChallengeRejectProtectionSpace, nil);
747#endif
748 }
749 }
750
751 sessionCocoa->continueDidReceiveChallenge(*_sessionWrapper, challenge, negotiatedLegacyTLS, taskIdentifier, [self existingTask:task], [completionHandler = makeBlockPtr(completionHandler)] (WebKit::AuthenticationChallengeDisposition disposition, const WebCore::Credential& credential) mutable {
752 completionHandler(toNSURLSessionAuthChallengeDisposition(disposition), credential.nsCredential());
753 });

756- (void)URLSession:(NSURLSession *)session task:(NSURLSessionTask *)task didCompleteWithError:(NSError *)error
757{
758 LOG(NetworkSession, "%llu didCompleteWithError %@", task.taskIdentifier, error);
759
760 if (error) {
761 NSDictionary *oldUserInfo = [error userInfo];
762 NSMutableDictionary *newUserInfo = oldUserInfo ? [NSMutableDictionary dictionaryWithDictionary:oldUserInfo] : [NSMutableDictionary dictionary];
763 newUserInfo[@"networkTaskDescription"] = [task description];
764 error = [NSError errorWithDomain:[error domain] code:[error code] userInfo:newUserInfo];
765 }
766
767 if (auto* networkDataTask = [self existingTask:task])
768 networkDataTask->didCompleteWithError(error, networkDataTask->networkLoadMetrics());
769 else if (error) {
770 if (!_sessionWrapper)
771 return;
772 auto downloadID = _sessionWrapper->downloadMap.take(task.taskIdentifier);
773 if (!downloadID)
774 return;
775 if (!_session)
776 return;
777 auto* download = _session->networkProcess().downloadManager().download(downloadID);
778 if (!download)
779 return;
780
781 NSData *resumeData = nil;
782 if (id userInfo = error.userInfo) {
783 if ([userInfo isKindOfClass:[NSDictionary class]]) {
784 resumeData = userInfo[@"NSURLSessionDownloadTaskResumeData"];
785 if (resumeData && ![resumeData isKindOfClass:[NSData class]]) {
786 RELEASE_LOG(NetworkSession, "Download task %llu finished with resume data of wrong class: %s", (unsigned long long)task.taskIdentifier, NSStringFromClass([resumeData class]).UTF8String);
787 ASSERT_NOT_REACHED();
788 resumeData = nil;
789 }
790 }
791 }
792
793#if USE(LEGACY_CFNETWORK_DOWNLOADS)
794 if (resumeData) {
795 // Mojave does not include the download location in the resume data from CFNetwork. Add it from the task.
796 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:resumeData error:nil]);
797 [unarchiver setDecodingFailurePolicy:NSDecodingFailurePolicyRaiseException];
798 auto dictionary = adoptNS(static_cast<NSMutableDictionary *>([[unarchiver decodeObjectOfClasses:[NSSet setWithObjects:[NSDictionary class], [NSArray class], [NSString class], [NSNumber class], [NSData class], [NSURL class], [NSURLRequest class], nil] forKey:@"NSKeyedArchiveRootObjectKey"] mutableCopy]));
799 [unarchiver finishDecoding];
800 [dictionary setObject:task._pathToDownloadTaskFile forKey:@"NSURLSessionResumeInfoLocalPath"];
801 auto encoder = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
802 [encoder encodeObject:dictionary.get() forKey:@"NSKeyedArchiveRootObjectKey"];
803 resumeData = [encoder encodedData];
804 }
805#endif
806
807 auto resumeDataReference = resumeData ? IPC::DataReference { static_cast<const uint8_t*>(resumeData.bytes), resumeData.length } : IPC::DataReference { };
808 download->didFail(error, resumeDataReference);
809 }

1487 case WebCore::StoredCredentialsPolicy::Use:
1488 LOG(NetworkSession, "Using isolated NSURLSession with credential storage.");

1493 case WebCore::StoredCredentialsPolicy::EphemeralStateless:

1495 }
1496 } (storedCredentialsPolicy);

1498 if (isolatedSessions.size() > maxNumberOfIsolatedSessions) {

1501 for (auto& key : isolatedSessions.keys()) {
1502 auto timestamp = isolatedSessions.get(key)->lastUsed;
1503 if (timestamp < oldestTimestamp) {
1504 oldestTimestamp = timestamp;
1505 keyToRemove = key;
1506 }
1507 }
1508 LOG(NetworkSession, "About to remove isolated NSURLSession.");

1510 }
1511
1512 RELEASE_ASSERT(isolatedSessions.size() <= maxNumberOfIsolatedSessions);
