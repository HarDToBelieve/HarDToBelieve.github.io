59static rtc::Thread& rtcNetworkThread()
60{
61 static NeverDestroyed<std::unique_ptr<rtc::Thread>> networkThread;
62 static std::once_flag onceKey;
63 std::call_once(onceKey, [] {
64 networkThread.get() = rtc::Thread::CreateWithSocketServer();
65 networkThread.get()->SetName("RTC Network Thread", nullptr);
66
67 auto result = networkThread.get()->Start();
68 ASSERT_UNUSED(result, result);
69 });
70 return *networkThread.get();
71}

83NetworkRTCProvider::NetworkRTCProvider(NetworkConnectionToWebProcess& connection)
84 : m_connection(&connection)
85 , m_ipcConnection(makeRef(connection.connection()))

87 , m_rtcNetworkThread(rtcNetworkThread())

107void NetworkRTCProvider::close()
108{
109 RTC_RELEASE_LOG("close");
110
111 // Cancel all pending DNS resolutions.
112 while (!m_resolvers.isEmpty())
113 stopResolver(*m_resolvers.keys().begin());
114
115 m_connection->connection().removeThreadMessageReceiver(Messages::NetworkRTCProvider::messageReceiverName());
116 m_connection = nullptr;
117 m_rtcMonitor.stopUpdating();
118
119 callOnRTCNetworkThread([this]() {
120 m_sockets.clear();
121 });
122}

124void NetworkRTCProvider::createSocket(LibWebRTCSocketIdentifier identifier, std::unique_ptr<rtc::AsyncPacketSocket>&& socket, Socket::Type type, Ref<IPC::Connection>&& connection)
125{
126 ASSERT(m_rtcNetworkThread.IsCurrent());
127 if (!socket) {
128 RTC_RELEASE_LOG_ERROR("createSocket with %u sockets is unable to create a new socket", m_sockets.size());
129 connection->send(Messages::LibWebRTCNetwork::SignalClose(identifier, 1), 0);
130 return;
131 }
132 addSocket(identifier, makeUnique<LibWebRTCSocketClient>(identifier, *this, WTFMove(socket), type, WTFMove(connection)));
133}

135void NetworkRTCProvider::createUDPSocket(LibWebRTCSocketIdentifier identifier, const RTCNetwork::SocketAddress& address, uint16_t minPort, uint16_t maxPort, bool isFirstParty, bool isRelayDisabled, WebCore::RegistrableDomain&& domain)
136{
137#if PLATFORM(COCOA)
138 if (m_platformUDPSocketsEnabled) {
139 if (auto socket = NetworkRTCUDPSocketCocoa::createUDPSocket(identifier, *this, address.value, minPort, maxPort, m_ipcConnection.copyRef(), isFirstParty, isRelayDisabled, WTFMove(domain))) {
140 addSocket(identifier, WTFMove(socket));
141 return;
142 }
143 }
144#endif
145
146 ASSERT(m_rtcNetworkThread.IsCurrent());
147 std::unique_ptr<rtc::AsyncPacketSocket> socket(m_packetSocketFactory->CreateUdpSocket(address.value, minPort, maxPort));
148 createSocket(identifier, WTFMove(socket), Socket::Type::UDP, m_ipcConnection.copyRef());
149}

151void NetworkRTCProvider::createServerTCPSocket(LibWebRTCSocketIdentifier identifier, const RTCNetwork::SocketAddress& address, uint16_t minPort, uint16_t maxPort, int options)
152{
153 ASSERT(m_rtcNetworkThread.IsCurrent());
154 callOnMainRunLoop([this, protectedThis = makeRef(*this), identifier, address, minPort, maxPort, options] {
155 if (!m_connection)
156 return;
157
158 if (!m_isListeningSocketAuthorized) {
159 m_connection->connection().send(Messages::LibWebRTCNetwork::SignalClose(identifier, 1), 0);
160 return;
161 }
162
163 callOnRTCNetworkThread([this, identifier, address = RTCNetwork::isolatedCopy(address.value), minPort, maxPort, options]() mutable {
164 std::unique_ptr<rtc::AsyncPacketSocket> socket(m_packetSocketFactory->CreateServerTcpSocket(address, minPort, maxPort, options));
165 createSocket(identifier, WTFMove(socket), Socket::Type::ServerTCP, m_ipcConnection.copyRef());
166 });
167 });
168}

177void NetworkRTCProvider::createClientTCPSocket(LibWebRTCSocketIdentifier identifier, const RTCNetwork::SocketAddress& localAddress, const RTCNetwork::SocketAddress& remoteAddress, String&& userAgent, int options)
178{
179 callOnMainRunLoop([this, protectedThis = makeRef(*this), identifier, localAddress, remoteAddress, userAgent = WTFMove(userAgent).isolatedCopy(), options]() mutable {
180 if (!m_connection)
181 return;
182
183 auto* session = m_connection->networkSession();
184 if (!session) {
185 m_connection->connection().send(Messages::LibWebRTCNetwork::SignalClose(identifier, 1), 0);
186 return;
187 }
188 callOnRTCNetworkThread([this, identifier, localAddress = RTCNetwork::isolatedCopy(localAddress.value), remoteAddress = RTCNetwork::isolatedCopy(remoteAddress.value), proxyInfo = proxyInfoFromSession(remoteAddress, *session), userAgent = WTFMove(userAgent).isolatedCopy(), options]() mutable {
189#if PLATFORM(COCOA)
190 if (m_platformTCPSocketsEnabled) {
191 if (auto socket = NetworkRTCTCPSocketCocoa::createClientTCPSocket(identifier, *this, remoteAddress, options, m_ipcConnection.copyRef())) {
192 addSocket(identifier, WTFMove(socket));
193 return;
194 }
195 }
196#endif
197 rtc::PacketSocketTcpOptions tcpOptions;
198 tcpOptions.opts = options;
199 std::unique_ptr<rtc::AsyncPacketSocket> socket(m_packetSocketFactory->CreateClientTcpSocket(localAddress, remoteAddress, proxyInfo, userAgent.utf8().data(), tcpOptions));
200 createSocket(identifier, WTFMove(socket), Socket::Type::ClientTCP, m_ipcConnection.copyRef());
201 });
202 });
203}

205void NetworkRTCProvider::wrapNewTCPConnection(LibWebRTCSocketIdentifier identifier, LibWebRTCSocketIdentifier newConnectionSocketIdentifier)
206{
207 ASSERT(m_rtcNetworkThread.IsCurrent());
208 auto socket = m_pendingIncomingSockets.take(newConnectionSocketIdentifier);
209 RELEASE_LOG_IF(!socket, WebRTC, "NetworkRTCProvider::wrapNewTCPConnection received an invalid socket identifier");
210 if (socket)
211 addSocket(identifier, makeUnique<LibWebRTCSocketClient>(identifier, *this, WTFMove(socket), Socket::Type::ServerConnectionTCP, m_ipcConnection.copyRef()));
212}

254void NetworkRTCProvider::newConnection(Socket& serverSocket, std::unique_ptr<rtc::AsyncPacketSocket>&& newSocket)
255{
256 ASSERT(m_rtcNetworkThread.IsCurrent());
257 auto incomingSocketIdentifier = LibWebRTCSocketIdentifier::generate();
258 m_ipcConnection->send(Messages::LibWebRTCNetwork::SignalNewConnection(serverSocket.identifier(), incomingSocketIdentifier, RTCNetwork::SocketAddress(newSocket->GetRemoteAddress())), 0);
259 m_pendingIncomingSockets.add(incomingSocketIdentifier, WTFMove(newSocket));
260}

267void NetworkRTCProvider::createResolver(LibWebRTCResolverIdentifier identifier, String&& address)
268{
269 if (!isMainRunLoop()) {
270 callOnMainRunLoop([this, protectedThis = makeRef(*this), identifier, address = WTFMove(address).isolatedCopy()]() mutable {
271 if (!m_connection)
272 return;
273 createResolver(identifier, WTFMove(address));
274 });
275 return;
276 }
277 WebCore::DNSCompletionHandler completionHandler = [this, identifier](auto&& result) {
278 if (!result.has_value()) {
279 if (result.error() != WebCore::DNSError::Cancelled)
280 m_connection->connection().send(Messages::WebRTCResolver::ResolvedAddressError(1), identifier);
281 return;
282 }
283
284 Vector<RTCNetwork::IPAddress> ipAddresses;
285 ipAddresses.reserveInitialCapacity(result.value().size());
286 for (auto& address : result.value()) {
287 if (address.isIPv4())
288 ipAddresses.uncheckedAppend(rtc::IPAddress { address.ipv4Address() });
289 else if (address.isIPv6())
290 ipAddresses.uncheckedAppend(rtc::IPAddress { address.ipv6Address() });
291 }
292
293 m_connection->connection().send(Messages::WebRTCResolver::SetResolvedAddress(ipAddresses), identifier);
294 };
295
296#if PLATFORM(COCOA)
297 auto resolver = NetworkRTCResolver::create(identifier, WTFMove(completionHandler));
298 resolver->start(address);
299 m_resolvers.add(identifier, WTFMove(resolver));
300#else
301 WebCore::resolveDNS(address, identifier.toUInt64(), WTFMove(completionHandler));
302#endif
303}

305void NetworkRTCProvider::stopResolver(LibWebRTCResolverIdentifier identifier)
306{
307 if (!isMainRunLoop()) {
308 callOnMainRunLoop([this, protectedThis = makeRef(*this), identifier] {
309 if (!m_connection)
310 return;
311 stopResolver(identifier);
312 });
313 return;
314 }
315#if PLATFORM(COCOA)
316 if (auto resolver = m_resolvers.take(identifier))
317 resolver->stop();
318#else
319 WebCore::stopResolveDNS(identifier.toUInt64());
320#endif
321}

323void NetworkRTCProvider::closeListeningSockets(Function<void()>&& completionHandler)
324{
325 ASSERT(isMainRunLoop());
326 if (!m_isListeningSocketAuthorized) {
327 completionHandler();
328 return;
329 }
330
331 m_isListeningSocketAuthorized = false;
332 callOnRTCNetworkThread([this, completionHandler = WTFMove(completionHandler)]() mutable {
333 Vector<LibWebRTCSocketIdentifier> listeningSocketIdentifiers;
334 for (auto& keyValue : m_sockets) {
335 if (keyValue.value->type() == Socket::Type::ServerTCP)
336 listeningSocketIdentifiers.append(keyValue.key);
337 }
338 for (auto id : listeningSocketIdentifiers)
339 m_sockets.get(id)->close();
340
341 callOnMainRunLoop([provider = makeRef(*this), listeningSocketIdentifiers = WTFMove(listeningSocketIdentifiers), completionHandler = WTFMove(completionHandler)] {
342 if (provider->m_connection) {
343 for (auto identifier : listeningSocketIdentifiers)
344 provider->m_connection->connection().send(Messages::LibWebRTCNetwork::SignalClose(identifier, ECONNABORTED), 0);
345 }
346 completionHandler();
347 });
348 });
349}

351struct NetworkMessageData : public rtc::MessageData {
352 NetworkMessageData(Ref<NetworkRTCProvider>&& rtcProvider, Function<void()>&& callback)
353 : rtcProvider(WTFMove(rtcProvider))
354 , callback(WTFMove(callback))

360void NetworkRTCProvider::OnMessage(rtc::Message* message)
361{
362 ASSERT(message->message_id == 1);
363 auto* data = static_cast<NetworkMessageData*>(message->pdata);
364 data->callback();
365 delete data;
366}

368void NetworkRTCProvider::callOnRTCNetworkThread(Function<void()>&& callback)
369{
370 m_rtcNetworkThread.Post(RTC_FROM_HERE, this, 1, new NetworkMessageData(*this, WTFMove(callback)));
371}
