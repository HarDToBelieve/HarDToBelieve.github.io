39static inline bool isEnabled(CFDictionaryRef proxy, CFStringRef key)
40{
41 auto enabled = static_cast<CFNumberRef>(CFDictionaryGetValue(proxy, key));
42
43 if (!enabled)
44 return false;
45
46 int result;
47 return CFNumberGetValue(enabled, kCFNumberIntType, &result);
48 return !!result;
49}
50
51static inline rtc::ProxyInfo createRTCProxy(CFDictionaryRef proxy, rtc::ProxyType type, CFStringRef nameKey, CFStringRef portKey)
52{
53 rtc::ProxyInfo info;
54 info.type = type;
55
56 auto proxyHostName = static_cast<CFStringRef>(CFDictionaryGetValue(proxy, nameKey));
57 if (!proxyHostName)
58 return { };
59 auto portNumber = static_cast<CFNumberRef>(CFDictionaryGetValue(proxy, portKey));
60 if (!portNumber)
61 return { };
62
63 int proxyPort;
64 if (!CFNumberGetValue(portNumber, kCFNumberSInt32Type, &proxyPort))
65 return { };
66
67 info.address = { std::string([(__bridge NSString *)proxyHostName UTF8String]), proxyPort };
68 info.autodetect = false;
69 return info;
70}

72rtc::ProxyInfo NetworkRTCProvider::proxyInfoFromSession(const RTCNetwork::SocketAddress&, NetworkSession& session)
73{
74 // FIXME: We should check for kCFNetworkProxiesExceptionsList to decide whether to use proxy or not.
75 // FIXME: We should also get username/password for authentication cases.
76 RetainPtr<CFDictionaryRef> proxyDictionary = static_cast<NetworkSessionCocoa&>(session).proxyConfiguration();
77 if (!proxyDictionary)
78 proxyDictionary = adoptCF(CFNetworkCopySystemProxySettings());
79#if PLATFORM(MAC)
80 if (isEnabled(proxyDictionary.get(), kCFNetworkProxiesHTTPSEnable))
81 return createRTCProxy(proxyDictionary.get(), rtc::PROXY_HTTPS, kCFNetworkProxiesHTTPSProxy, kCFNetworkProxiesHTTPSPort);
82
83 if (isEnabled(proxyDictionary.get(), kCFNetworkProxiesSOCKSEnable))
84 return createRTCProxy(proxyDictionary.get(), rtc::PROXY_SOCKS5, kCFNetworkProxiesSOCKSProxy, kCFNetworkProxiesSOCKSPort);
85#endif
86#if PLATFORM(IOS)
87 if (isEnabled(proxyDictionary.get(), kCFNetworkProxiesHTTPEnable))
88 return createRTCProxy(proxyDictionary.get(), rtc::PROXY_HTTPS, kCFNetworkProxiesHTTPProxy, kCFNetworkProxiesHTTPPort);
89#endif
90 return { };
91}
