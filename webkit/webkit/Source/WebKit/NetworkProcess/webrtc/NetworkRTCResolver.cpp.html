33namespace WebKit {

35// FIXME: Use the function after removing the NetworkRTCResolverCocoa.
36#if !PLATFORM(COCOA)
37std::unique_ptr<NetworkRTCResolver> NetworkRTCResolver::create(LibWebRTCResolverIdentifier identifier, WebCore::DNSCompletionHandler&& completionHandler)
38{
39 return std::unique_ptr<NetworkRTCResolver>(new NetworkRTCResolver(identifier, WTFMove(completionHandler)));
40}
41#endif
42
43NetworkRTCResolver::NetworkRTCResolver(LibWebRTCResolverIdentifier identifier, WebCore::DNSCompletionHandler&& completionHandler)
44 : m_identifier(identifier)
45 , m_completionHandler(WTFMove(completionHandler))
46{
47}
48
49NetworkRTCResolver::~NetworkRTCResolver()
50{
51 if (auto completionHandler = WTFMove(m_completionHandler))
52 completionHandler(makeUnexpected(WebCore::DNSError::Unknown));
53}
54
55void NetworkRTCResolver::start(const String& address)
56{
57 WebCore::resolveDNS(address, m_identifier.toUInt64(), WTFMove(m_completionHandler));
58}
59
60void NetworkRTCResolver::stop()
61{
62 WebCore::stopResolveDNS(m_identifier.toUInt64());
63}
