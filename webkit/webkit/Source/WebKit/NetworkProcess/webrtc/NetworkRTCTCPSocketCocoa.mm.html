42static dispatch_queue_t tcpSocketQueue()
43{
44 static dispatch_queue_t queue;
45 static dispatch_once_t onceToken;
46 dispatch_once(&onceToken, ^{
47 queue = dispatch_queue_create("WebRTC TCP socket queue", DISPATCH_QUEUE_CONCURRENT);
48 });
49 return queue;
50}

52std::unique_ptr<NetworkRTCProvider::Socket> NetworkRTCTCPSocketCocoa::createClientTCPSocket(LibWebRTCSocketIdentifier identifier, NetworkRTCProvider& rtcProvider, const rtc::SocketAddress& remoteAddress, int tcpOptions, Ref<IPC::Connection>&& connection)
53{
54 // FIXME: We should migrate ssltcp candidates, maybe support OPT_TLS_INSECURE as well.
55 if ((tcpOptions & rtc::PacketSocketFactory::OPT_TLS_FAKE) || (tcpOptions & rtc::PacketSocketFactory::OPT_TLS_INSECURE))
56 return nullptr;
57 return makeUnique<NetworkRTCTCPSocketCocoa>(identifier, rtcProvider, remoteAddress, tcpOptions, WTFMove(connection));
58}

60static inline void processIncomingData(RetainPtr<nw_connection_t>&& nwConnection, Function<Vector<uint8_t>(Vector<uint8_t>&&)>&& processData, Vector<uint8_t>&& buffer = { })
61{
62 auto nwConnectionReference = nwConnection.get();
63 nw_connection_receive(nwConnectionReference, 1, std::numeric_limits<uint32_t>::max(), makeBlockPtr([nwConnection = WTFMove(nwConnection), processData = WTFMove(processData), buffer = WTFMove(buffer)](dispatch_data_t content, nw_content_context_t context, bool isComplete, nw_error_t error) mutable {
64 if (content) {
65 dispatch_data_apply(content, makeBlockPtr([&](dispatch_data_t, size_t, const void* data, size_t size) {
66 // FIXME: Introduce uncheckedAppend version.
67 buffer.append(static_cast<const uint8_t*>(data), size);
68 return true;
69 }).get());
70 buffer = processData(WTFMove(buffer));
71 }
72 if (isComplete && context && nw_content_context_get_is_final(context))
73 return;
74 if (error) {
75 RELEASE_LOG_ERROR(WebRTC, "NetworkRTCTCPSocketCocoa processIncomingData failed with error %d", nw_error_get_error_code(error));
76 return;
77 }
78 processIncomingData(WTFMove(nwConnection), WTFMove(processData), WTFMove(buffer));
79 }).get());
80}

82NetworkRTCTCPSocketCocoa::NetworkRTCTCPSocketCocoa(LibWebRTCSocketIdentifier identifier, NetworkRTCProvider& rtcProvider, const rtc::SocketAddress& remoteAddress, int options, Ref<IPC::Connection>&& connection)
83 : m_identifier(identifier)
84 , m_rtcProvider(rtcProvider)
85 , m_connection(WTFMove(connection))
86 , m_isSTUN(options & rtc::PacketSocketFactory::OPT_STUN)
87{
88 auto hostName = remoteAddress.hostname();
89 if (hostName.empty())
90 hostName = remoteAddress.ipaddr().ToString();
91 auto host = adoptNS(nw_endpoint_create_host(hostName.c_str(), String::number(remoteAddress.port()).utf8().data()));
92 // FIXME: Handle TLS certificate validation like for other network code paths, using sec_protocol_options_set_verify_block
93 bool isTLS = options & rtc::PacketSocketFactory::OPT_TLS;
94 auto tcpTLS = adoptNS(nw_parameters_create_secure_tcp(isTLS ? NW_PARAMETERS_DEFAULT_CONFIGURATION : NW_PARAMETERS_DISABLE_PROTOCOL, ^(nw_protocol_options_t tcp_options) {
95 nw_tcp_options_set_no_delay(tcp_options, true);
96 }));
97
98 m_nwConnection = adoptNS(nw_connection_create(host.get(), tcpTLS.get()));
99
100 nw_connection_set_queue(m_nwConnection.get(), tcpSocketQueue());
101 nw_connection_set_state_changed_handler(m_nwConnection.get(), makeBlockPtr([identifier = m_identifier, rtcProvider = makeRef(rtcProvider), connection = m_connection.copyRef()](nw_connection_state_t state, _Nullable nw_error_t error) {
102 ASSERT_UNUSED(error, !error);
103 switch (state) {
104 case nw_connection_state_invalid:
105 case nw_connection_state_waiting:
106 case nw_connection_state_preparing:
107 return;
108 case nw_connection_state_ready:
109 connection->send(Messages::LibWebRTCNetwork::SignalConnect(identifier), 0);
110 return;
111 case nw_connection_state_failed:
112 rtcProvider->callOnRTCNetworkThread([rtcProvider, identifier] {
113 rtcProvider->closeSocket(identifier);
114 });
115 connection->send(Messages::LibWebRTCNetwork::SignalClose(identifier, -1), 0);
116 return;
117 case nw_connection_state_cancelled:
118 return;
119 }
120 }).get());
121
122 processIncomingData(m_nwConnection.get(), [identifier = m_identifier, connection = m_connection.copyRef(), ip = remoteAddress.ipaddr(), port = remoteAddress.port(), isSTUN = m_isSTUN](auto&& buffer) mutable {
123 return WebRTC::extractMessages(WTFMove(buffer), isSTUN ? WebRTC::MessageType::STUN : WebRTC::MessageType::Data, [&](auto* message, auto size) {
124 IPC::DataReference data(message, size);
125 connection->send(Messages::LibWebRTCNetwork::SignalReadPacket { identifier, data, RTCNetwork::IPAddress(ip), port, rtc::TimeMillis() * 1000 }, 0);
126 });
127 });
128
129 nw_connection_start(m_nwConnection.get());
130}

132void NetworkRTCTCPSocketCocoa::close()
133{
134 if (!m_nwConnection)
135 return;
136 nw_connection_cancel(m_nwConnection.get());
137 m_rtcProvider.takeSocket(m_identifier);
138}

145static RetainPtr<dispatch_data_t> dataFromVector(Vector<uint8_t>&& v)
146{
147 auto bufferSize = v.size();
148 auto rawPointer = v.releaseBuffer().leakPtr();
149 return adoptNS(dispatch_data_create(rawPointer, bufferSize, dispatch_get_main_queue(), ^{
150 fastFree(rawPointer);
151 }));
152}

154Vector<uint8_t> NetworkRTCTCPSocketCocoa::createMessageBuffer(const uint8_t* data, size_t size)
155{
156 if (size >= std::numeric_limits<uint16_t>::max())
157 return { };
158
159 if (m_isSTUN) {
160 auto messageLengths = WebRTC::getSTUNOrTURNMessageLengths(data, size);
161 if (!messageLengths)
162 return { };
163
164 ASSERT(messageLengths->messageLength == size);
165 ASSERT(messageLengths->messageLengthWithPadding >= size);
166 if (messageLengths->messageLengthWithPadding < size)
167 return { };
168
169 Vector<uint8_t> buffer;
170 buffer.reserveInitialCapacity(messageLengths->messageLengthWithPadding);
171 buffer.append(data, size);
172 for (size_t cptr = 0 ; cptr < messageLengths->messageLengthWithPadding - size; ++cptr)
173 buffer.uncheckedAppend(0);
174 return buffer;
175 }
176
177 // Prepend length.
178 Vector<uint8_t> buffer;
179 buffer.reserveInitialCapacity(size + 2);
180 buffer.uncheckedAppend((size >> 8) & 0xFF);
181 buffer.uncheckedAppend(size & 0xFF);
182 buffer.append(data, size);
183 return buffer;
184}

186void NetworkRTCTCPSocketCocoa::sendTo(const uint8_t* data, size_t size, const rtc::SocketAddress&, const rtc::PacketOptions& options)
187{
188 auto buffer = createMessageBuffer(data, size);
189 if (buffer.isEmpty())
190 return;
191
192 nw_connection_send(m_nwConnection.get(), dataFromVector(WTFMove(buffer)).get(), NW_CONNECTION_DEFAULT_MESSAGE_CONTEXT, true, makeBlockPtr([identifier = m_identifier, connection = m_connection.copyRef(), options](_Nullable nw_error_t) {
193 connection->send(Messages::LibWebRTCNetwork::SignalSentPacket { identifier, options.packet_id, rtc::TimeMillis() }, 0);
194 }).get());
195}
