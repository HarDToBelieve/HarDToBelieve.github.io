83static dispatch_queue_t udpSocketQueue()
84{
85 static dispatch_queue_t queue;
86 static dispatch_once_t onceToken;
87 dispatch_once(&onceToken, ^{
88 queue = dispatch_queue_create("WebRTC UDP socket queue", DISPATCH_QUEUE_CONCURRENT);
89 });
90 return queue;
91}

125static rtc::SocketAddress socketAddressFromIncomingConnection(nw_connection_t connection)
126{
127 auto endpoint = adoptNS(nw_connection_copy_endpoint(connection));
128 auto type = nw_endpoint_get_type(endpoint.get());
129 if (type == nw_endpoint_type_address) {
130 auto* ipAddress = nw_endpoint_copy_address_string(endpoint.get());
131 rtc::SocketAddress remoteAddress { ipAddress, nw_endpoint_get_port(endpoint.get()) };
132 free(ipAddress);
133 return remoteAddress;
134 }
135 return rtc::SocketAddress { nw_endpoint_get_hostname(endpoint.get()), nw_endpoint_get_port(endpoint.get()) };
136}

148NetworkRTCUDPSocketCocoaConnections::NetworkRTCUDPSocketCocoaConnections(WebCore::LibWebRTCSocketIdentifier identifier, NetworkRTCProvider& rtcProvider, const rtc::SocketAddress& address, Ref<IPC::Connection>&& connection, bool isFirstParty, bool isRelayDisabled, const WebCore::RegistrableDomain& domain)
149 : m_identifier(identifier)
150 , m_connection(WTFMove(connection))
151#if HAVE(NWPARAMETERS_TRACKER_API)
152 , m_isFirstParty(isFirstParty)
153 , m_isKnownTracker(isKnownTracker(domain))
154#endif
155 , m_shouldBypassRelay(isRelayDisabled)
156{
157 auto parameters = adoptNS(nw_parameters_create_secure_udp(NW_PARAMETERS_DISABLE_PROTOCOL, NW_PARAMETERS_DEFAULT_CONFIGURATION));
158 {
159 auto hostAddress = address.ipaddr().ToString();
160 if (address.ipaddr().IsNil())
161 hostAddress = address.hostname();
162 auto localEndpoint = adoptNS(nw_endpoint_create_host(hostAddress.c_str(), "0"));
163 m_address = { nw_endpoint_get_hostname(localEndpoint.get()), nw_endpoint_get_port(localEndpoint.get()) };
164 nw_parameters_set_local_endpoint(parameters.get(), localEndpoint.get());
165 }
166 configureParameters(parameters.get(), address.family() == AF_INET ? nw_ip_version_4 : nw_ip_version_6);
167
168 m_nwListener = adoptNS(nw_listener_create(parameters.get()));
169 nw_listener_set_queue(m_nwListener.get(), udpSocketQueue());
170
171 nw_listener_set_state_changed_handler(m_nwListener.get(), makeBlockPtr([nwListener = m_nwListener.get(), address = m_address, connection = m_connection.copyRef(), protectedRTCProvider = makeRef(rtcProvider), identifier = m_identifier](nw_listener_state_t state, nw_error_t error) mutable {
172 switch (state) {
173 case nw_listener_state_invalid:
174 case nw_listener_state_waiting:
175 break;
176 case nw_listener_state_ready:
177 address.SetPort(nw_listener_get_port(nwListener));
178 connection->send(Messages::LibWebRTCNetwork::SignalAddressReady(identifier, RTCNetwork::SocketAddress(address)), 0);
179 break;
180 case nw_listener_state_failed:
181 RELEASE_LOG_ERROR(WebRTC, "NetworkRTCUDPSocketCocoaConnections failed with error %d", error ? nw_error_get_error_code(error) : 0);
182 protectedRTCProvider->callOnRTCNetworkThread([protectedRTCProvider, identifier] {
183 protectedRTCProvider->closeSocket(identifier);
184 });
185 connection->send(Messages::LibWebRTCNetwork::SignalClose(identifier, -1), 0);
186 break;
187 case nw_listener_state_cancelled:
188 break;
189 }
190 }).get());
191
192 nw_listener_set_new_connection_handler(m_nwListener.get(), makeBlockPtr([protectedThis = makeRef(*this)](nw_connection_t connection) {
193 Locker locker { protectedThis->m_nwConnectionsLock };
194 if (protectedThis->m_isClosed)
195 return;
196
197 auto remoteAddress = socketAddressFromIncomingConnection(connection);
198 ASSERT(remoteAddress != HashTraits<rtc::SocketAddress>::emptyValue() && !HashTraits<rtc::SocketAddress>::isDeletedValue(remoteAddress));
199
200 protectedThis->m_nwConnections.set(remoteAddress, connection);
201 protectedThis->setupNWConnection(connection, remoteAddress);
202 }).get());
203
204 nw_listener_start(m_nwListener.get());
205}

208{
209 auto protocolStack = adoptNS(nw_parameters_copy_default_protocol_stack(parameters));
210 auto options = adoptNS(nw_protocol_stack_copy_internet_protocol(protocolStack.get()));
211 nw_ip_options_set_version(options.get(), version);
212
213 if (m_shouldBypassRelay)

215#if HAVE(NWPARAMETERS_TRACKER_API)
216 nw_parameters_set_is_third_party_web_content(parameters, m_isFirstParty);
217 nw_parameters_set_is_known_tracker(parameters, m_isKnownTracker);
218#endif
219}

221void NetworkRTCUDPSocketCocoaConnections::close()
222{
223 Locker locker { m_nwConnectionsLock };
224 m_isClosed = true;
225
226 nw_listener_cancel(m_nwListener.get());
227 m_nwListener = nullptr;
228
229 for (auto& nwConnection : m_nwConnections.values())
230 nw_connection_cancel(nwConnection.get());
231 m_nwConnections.clear();

239static inline void processUDPData(RetainPtr<nw_connection_t>&& nwConnection, Function<void(const uint8_t*, size_t)>&& processData)
240{
241 auto nwConnectionReference = nwConnection.get();
242 nw_connection_receive(nwConnectionReference, 1, std::numeric_limits<uint32_t>::max(), makeBlockPtr([nwConnection = WTFMove(nwConnection), processData = WTFMove(processData)](dispatch_data_t content, nw_content_context_t context, bool isComplete, nw_error_t error) mutable {
243 if (content) {
244 dispatch_data_apply(content, makeBlockPtr([&](dispatch_data_t, size_t, const void* data, size_t size) {
245 processData(static_cast<const uint8_t*>(data), size);
246 return true;
247 }).get());
248 }
249 if (isComplete && context && nw_content_context_get_is_final(context))
250 return;
251 if (error) {
252 RELEASE_LOG_ERROR(WebRTC, "NetworkRTCUDPSocketCocoaConnections failed processing UDP data with error %d", error ? nw_error_get_error_code(error) : 0);
253 return;
254 }
255 processUDPData(WTFMove(nwConnection), WTFMove(processData));
256 }).get());
257}

259RetainPtr<nw_connection_t> NetworkRTCUDPSocketCocoaConnections::createNWConnection(const rtc::SocketAddress& remoteAddress)
260{
261 auto parameters = adoptNS(nw_parameters_create_secure_udp(NW_PARAMETERS_DISABLE_PROTOCOL, NW_PARAMETERS_DEFAULT_CONFIGURATION));
262 {
263 auto hostAddress = m_address.ipaddr().ToString();
264 if (m_address.ipaddr().IsNil())
265 hostAddress = m_address.hostname();
266 auto localEndpoint = adoptNS(nw_endpoint_create_host(hostAddress.c_str(), String::number(m_address.port()).utf8().data()));
267 nw_parameters_set_local_endpoint(parameters.get(), localEndpoint.get());
268 }
269 configureParameters(parameters.get(), remoteAddress.family() == AF_INET ? nw_ip_version_4 : nw_ip_version_6);
270
271 auto remoteHostAddress = remoteAddress.ipaddr().ToString();
272 if (remoteAddress.ipaddr().IsNil())
273 remoteHostAddress = remoteAddress.hostname();
274 auto host = adoptNS(nw_endpoint_create_host(remoteHostAddress.c_str(), String::number(remoteAddress.port()).utf8().data()));
275 auto nwConnection = adoptNS(nw_connection_create(host.get(), parameters.get()));
276
277 setupNWConnection(nwConnection.get(), remoteAddress);
278 return nwConnection;
279}

281void NetworkRTCUDPSocketCocoaConnections::setupNWConnection(nw_connection_t nwConnection, const rtc::SocketAddress& remoteAddress)
282{
283 nw_connection_set_queue(nwConnection, udpSocketQueue());
284
285 processUDPData(nwConnection, [identifier = m_identifier, connection = m_connection.copyRef(), ip = remoteAddress.ipaddr(), port = remoteAddress.port()](auto* message, auto size) mutable {
286 IPC::DataReference data(message, size);
287 connection->send(Messages::LibWebRTCNetwork::SignalReadPacket { identifier, data, RTCNetwork::IPAddress(ip), port, rtc::TimeMillis() * 1000 }, 0);
288 });
289
290 nw_connection_start(nwConnection);
291}

293void NetworkRTCUDPSocketCocoaConnections::sendTo(const uint8_t* data, size_t size, const rtc::SocketAddress& remoteAddress, const rtc::PacketOptions& options)
294{
295 bool isInCorrectValue = (remoteAddress == HashTraits<rtc::SocketAddress>::emptyValue()) || HashTraits<rtc::SocketAddress>::isDeletedValue(remoteAddress);
296 ASSERT(!isInCorrectValue);
297 if (isInCorrectValue)
298 return;
299
300 nw_connection_t nwConnection;
301 {
302 Locker locker { m_nwConnectionsLock };
303 nwConnection = m_nwConnections.ensure(remoteAddress, [this, &remoteAddress] {
304 return createNWConnection(remoteAddress);
305 }).iterator->value.get();
306 }
307
308 auto value = adoptNS(dispatch_data_create(data, size, nullptr, DISPATCH_DATA_DESTRUCTOR_DEFAULT));
309 nw_connection_send(nwConnection, value.get(), NW_CONNECTION_DEFAULT_MESSAGE_CONTEXT, true, makeBlockPtr([identifier = m_identifier, connection = m_connection.copyRef(), options](_Nullable nw_error_t) {
310 connection->send(Messages::LibWebRTCNetwork::SignalSentPacket { identifier, options.packet_id, rtc::TimeMillis() }, 0);
311 }).get());
312}
