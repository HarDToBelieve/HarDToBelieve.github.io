32namespace IPC {
33namespace Detail {
34template<typename T, typename I> auto TestLegacyDecoder(int) -> decltype(I::decode(std::declval<Decoder&>(), std::declval<T&>()), std::true_type { });
35template<typename T, typename I> auto TestLegacyDecoder(long) -> std::false_type;
36template<typename T, typename I> auto TestModernDecoder(int) -> decltype(I::decode(std::declval<Decoder&>()), std::true_type { });
37template<typename T, typename I> auto TestModernDecoder(long) -> std::false_type;
38}
39template<typename T, typename I = T> struct HasLegacyDecoder : decltype(Detail::TestLegacyDecoder<T, I>(0)) { };
40template<typename T, typename I = T> inline constexpr bool HasLegacyDecoderV = HasLegacyDecoder<T, I>::value;
41template<typename T, typename I = T> struct HasModernDecoder : decltype(Detail::TestModernDecoder<T, I>(0)) { };
42template<typename T, typename I = T> inline constexpr bool HasModernDecoderV = HasModernDecoder<T, I>::value;
43
44template<typename T, typename = void> struct ArgumentCoder {
45 template<typename Encoder>
46 static void encode(Encoder& encoder, const T& t)
47 {
48 t.encode(encoder);
49 }
50
51 static std::optional<T> decode(Decoder& decoder)
52 {
53 if constexpr(HasModernDecoderV<T>)
54 return T::decode(decoder);
55 else {
56 T t;
57 if (T::decode(decoder, t))
58 return t;
59 return std::nullopt;
60 }
61 }
62
63 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, T& t)
64 {
65 if constexpr(HasLegacyDecoderV<T>)
66 return T::decode(decoder, t);
67 else {
68 std::optional<T> optional = T::decode(decoder);
69 if (!optional)
70 return false;
71 t = WTFMove(*optional);
72 return true;
73 }
74 }
75};
76
77template<typename T>
78struct ArgumentCoder<T, typename std::enable_if_t<std::is_arithmetic_v<T>>> {
79 template<typename Encoder>
80 static void encode(Encoder& encoder, T value)
81 {
82 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(&value), sizeof(T), alignof(T));
83 }
84
85 static std::optional<T> decode(Decoder& decoder)
86 {
87 T result;
88 if (decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(&result), sizeof(T), alignof(T)))
89 return result;
90 return std::nullopt;
91 }
92};
93
94template<typename T>
95struct ArgumentCoder<T, typename std::enable_if_t<std::is_enum_v<T>>> {
96 template<typename Encoder>
97 static void encode(Encoder& encoder, T value)
98 {
99 ASSERT(WTF::isValidEnum<T>(WTF::enumToUnderlyingType<T>(value)));
100 encoder << WTF::enumToUnderlyingType<T>(value);
101 }
102
103 static std::optional<T> decode(Decoder& decoder)
104 {
105 std::optional<std::underlying_type_t<T>> value;
106 decoder >> value;
107 if (value && WTF::isValidEnum<T>(*value))
108 return static_cast<T>(*value);
109 return std::nullopt;
110 }
111};
