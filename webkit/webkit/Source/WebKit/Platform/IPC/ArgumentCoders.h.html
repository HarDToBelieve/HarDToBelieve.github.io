43// An argument coder works on POD types
44template<typename T> struct SimpleArgumentCoder {
45 template<typename Encoder>
46 static void encode(Encoder& encoder, const T& t)
47 {
48 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(&t), sizeof(T), alignof(T));
49 }
50
51 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, T& t)
52 {
53 return decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(&t), sizeof(T), alignof(T));
54 }
55};

57template<typename T, size_t Extent> struct ArgumentCoder<ArrayReference<T, Extent>> {
58 using ArrayReferenceType = ArrayReference<T, Extent>;
59 template<typename Encoder>
60 static void encode(Encoder& encoder, const ArrayReferenceType& arrayReference)
61 {
62 if (!Extent)
63 return;
64 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(arrayReference.data()), arrayReference.size() * sizeof(T), alignof(T));
65 }
66 static std::optional<ArrayReferenceType> decode(Decoder& decoder)
67 {
68 if (!Extent)
69 return ArrayReferenceType();
70 const uint8_t* data = decoder.decodeFixedLengthReference(Extent * sizeof(T), alignof(T));
71 if (!data)
72 return std::nullopt;
73 return ArrayReferenceType(reinterpret_cast<const T*>(data), Extent);
74 }
75};

77template<typename T> struct ArgumentCoder<ArrayReference<T, arrayReferenceDynamicExtent>> {
78 using ArrayReferenceType = ArrayReference<T, arrayReferenceDynamicExtent>;
79 template<typename Encoder>
80 static void encode(Encoder& encoder, const ArrayReferenceType& arrayReference)
81 {
82 encoder << static_cast<uint64_t>(arrayReference.size());
83 if (!arrayReference.size())
84 return;
85 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(arrayReference.data()), arrayReference.size() * sizeof(T), alignof(T));
86 }
87 static std::optional<ArrayReferenceType> decode(Decoder& decoder)
88 {
89 uint64_t size;
90 if (!decoder.decode(size))
91 return std::nullopt;
92 if (!size)
93 return ArrayReferenceType();
94
95 auto dataSize = CheckedSize { size } * sizeof(T);
96 if (UNLIKELY(dataSize.hasOverflowed()))
97 return std::nullopt;
98
99 const uint8_t* data = decoder.decodeFixedLengthReference(dataSize, alignof(T));
100 if (!data)
101 return std::nullopt;
102 return ArrayReferenceType(reinterpret_cast<const T*>(data), static_cast<size_t>(size));
103 }
104};

106template<typename T> struct ArgumentCoder<OptionSet<T>> {
107 static void encode(Encoder& encoder, const OptionSet<T>& optionSet)
108 {
109 ASSERT(WTF::isValidOptionSet(optionSet));
110 encoder << optionSet.toRaw();
111 }
112
113 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, OptionSet<T>& optionSet)
114 {
115 typename OptionSet<T>::StorageType value;
116 if (!decoder.decode(value))
117 return false;
118 optionSet = OptionSet<T>::fromRaw(value);
119 if (!WTF::isValidOptionSet(optionSet))
120 return false;
121 return true;
122 }
123
124 static std::optional<OptionSet<T>> decode(Decoder& decoder)
125 {
126 std::optional<typename OptionSet<T>::StorageType> value;
127 decoder >> value;
128 if (!value)
129 return std::nullopt;
130 auto optionSet = OptionSet<T>::fromRaw(*value);
131 if (!WTF::isValidOptionSet(optionSet))
132 return std::nullopt;
133 return optionSet;
134 }
135};

137template<typename T> struct ArgumentCoder<std::optional<T>> {
138
139 template<typename Encoder> static void encode(Encoder& encoder, const std::optional<T>& optional)
140 {
141 if (!optional) {
142 encoder << false;
143 return;
144 }
145
146 encoder << true;
147 encoder << optional.value();
148 }
149
150 template<typename Decoder> static WARN_UNUSED_RETURN bool decode(Decoder& decoder, std::optional<T>& optional)
151 {
152 bool isEngaged;
153 if (!decoder.decode(isEngaged))
154 return false;
155
156 if (!isEngaged) {
157 optional = std::nullopt;
158 return true;
159 }
160
161 T value;
162 if (!decoder.decode(value))
163 return false;
164
165 optional = WTFMove(value);
166 return true;
167 }
168
169 template<typename Decoder> static std::optional<std::optional<T>> decode(Decoder& decoder)
170 {
171 std::optional<bool> isEngaged;
172 decoder >> isEngaged;
173 if (!isEngaged)
174 return std::nullopt;
175 if (*isEngaged) {
176 std::optional<T> value;
177 decoder >> value;
178 if (!value)
179 return std::nullopt;
180 return std::optional<std::optional<T>>(WTFMove(*value));
181 }
182 return std::optional<std::optional<T>>(std::optional<T>(std::nullopt));
183 }
184};

186template<typename T> struct ArgumentCoder<Box<T>> {
187 static void encode(Encoder& encoder, const Box<T>& box)
188 {
189 if (!box) {
190 encoder << false;
191 return;
192 }
193
194 encoder << true;
195 encoder << *box.get();
196 }
197
198 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, Box<T>& box)
199 {
200 bool isEngaged;
201 if (!decoder.decode(isEngaged))
202 return false;
203
204 if (!isEngaged) {
205 box = nullptr;
206 return true;
207 }
208
209 Box<T> value = Box<T>::create();
210 if (!decoder.decode(*value))
211 return false;
212
213 box = WTFMove(value);
214 return true;
215 }
216
217 static std::optional<Box<T>> decode(Decoder& decoder)
218 {
219 std::optional<bool> isEngaged;
220 decoder >> isEngaged;
221 if (!isEngaged)
222 return std::nullopt;
223 if (*isEngaged) {
224 std::optional<T> value;
225 decoder >> value;
226 if (!value)
227 return std::nullopt;
228 return std::optional<Box<T>>(Box<T>::create(WTFMove(*value)));
229 }
230 return std::optional<Box<T>>(Box<T>(nullptr));
231 }

234template<typename T, typename U> struct ArgumentCoder<std::pair<T, U>> {
235 static void encode(Encoder& encoder, const std::pair<T, U>& pair)
236 {
237 encoder << pair.first << pair.second;
238 }
239
240 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, std::pair<T, U>& pair)
241 {
242 T first;
243 if (!decoder.decode(first))
244 return false;
245
246 U second;
247 if (!decoder.decode(second))
248 return false;
249
250 pair.first = first;
251 pair.second = second;
252 return true;
253 }
254
255 static std::optional<std::pair<T, U>> decode(Decoder& decoder)
256 {
257 std::optional<T> first;
258 decoder >> first;
259 if (!first)
260 return std::nullopt;
261
262 std::optional<U> second;
263 decoder >> second;
264 if (!second)
265 return std::nullopt;
266
267 return {{ WTFMove(*first), WTFMove(*second) }};
268 }

289template <typename T, typename... Elements, size_t... Indices>
290auto tupleFromTupleAndObject(T&& object, std::tuple<Elements...>&& tuple, std::index_sequence<Indices...>)
291{
292 return std::make_tuple(WTFMove(object), WTFMove(std::get<Indices>(tuple))...);

296auto tupleFromTupleAndObject(T&& object, std::tuple<Elements...>&& tuple)
297{
298 return tupleFromTupleAndObject(WTFMove(object), WTFMove(tuple), std::index_sequence_for<Elements...>());

301template<typename Type, typename... Types>
302struct TupleDecoderImpl {
303 static std::optional<std::tuple<Type, Types...>> decode(Decoder& decoder)
304 {
305 std::optional<Type> optional;
306 decoder >> optional;
307 if (!optional)
308 return std::nullopt;
309
310 std::optional<std::tuple<Types...>> subTuple = TupleDecoderImpl<Types...>::decode(decoder);
311 if (!subTuple)
312 return std::nullopt;
313
314 return tupleFromTupleAndObject(WTFMove(*optional), WTFMove(*subTuple));
315 }
316};

383template<typename T, size_t inlineCapacity, typename OverflowHandler, size_t minCapacity> struct VectorArgumentCoder<false, T, inlineCapacity, OverflowHandler, minCapacity> {
384 template<typename Encoder>
385 static void encode(Encoder& encoder, const Vector<T, inlineCapacity, OverflowHandler, minCapacity>& vector)
386 {
387 encoder << static_cast<uint64_t>(vector.size());
388 for (size_t i = 0; i < vector.size(); ++i)
389 encoder << vector[i];
390 }
391
392 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, Vector<T, inlineCapacity, OverflowHandler, minCapacity>& vector)
393 {
394 std::optional<Vector<T, inlineCapacity, OverflowHandler, minCapacity>> optional;
395 decoder >> optional;
396 if (!optional)
397 return false;
398 vector = WTFMove(*optional);
399 return true;
400 }
401
402 static std::optional<Vector<T, inlineCapacity, OverflowHandler, minCapacity>> decode(Decoder& decoder)
403 {
404 uint64_t size;
405 if (!decoder.decode(size))
406 return std::nullopt;
407
408 Vector<T, inlineCapacity, OverflowHandler, minCapacity> vector;
409 for (size_t i = 0; i < size; ++i) {
410 std::optional<T> element;
411 decoder >> element;
412 if (!element)
413 return std::nullopt;
414 vector.append(WTFMove(*element));
415 }
416 vector.shrinkToFit();
417 return vector;
418 }
419};

421template<typename T, size_t inlineCapacity, typename OverflowHandler, size_t minCapacity> struct VectorArgumentCoder<true, T, inlineCapacity, OverflowHandler, minCapacity> {
422 template<typename Encoder>
423 static void encode(Encoder& encoder, const Vector<T, inlineCapacity, OverflowHandler, minCapacity>& vector)
424 {
425 encoder << static_cast<uint64_t>(vector.size());
426 encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(vector.data()), vector.size() * sizeof(T), alignof(T));
427 }
428
429 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, Vector<T, inlineCapacity, OverflowHandler, minCapacity>& vector)
430 {
431 uint64_t decodedSize;
432 if (!decoder.decode(decodedSize))
433 return false;
434
435 if (!isInBounds<size_t>(decodedSize))
436 return false;
437
438 auto size = static_cast<size_t>(decodedSize);
439
440 // Since we know the total size of the elements, we can allocate the vector in
441 // one fell swoop. Before allocating we must however make sure that the decoder buffer
442 // is big enough.
443 if (!decoder.bufferIsLargeEnoughToContain<T>(size))
444 return false;
445
446 Vector<T, inlineCapacity, OverflowHandler, minCapacity> temp;
447 temp.grow(size);
448
449 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(temp.data()), size * sizeof(T), alignof(T)))
450 return false;
451
452 vector.swap(temp);
453 return true;
454 }
455
456 static std::optional<Vector<T, inlineCapacity, OverflowHandler, minCapacity>> decode(Decoder& decoder)
457 {
458 uint64_t decodedSize;
459 if (!decoder.decode(decodedSize))
460 return std::nullopt;
461
462 if (!isInBounds<size_t>(decodedSize))
463 return std::nullopt;
464
465 auto size = static_cast<size_t>(decodedSize);
466
467 // Since we know the total size of the elements, we can allocate the vector in
468 // one fell swoop. Before allocating we must however make sure that the decoder buffer
469 // is big enough.
470 if (!decoder.bufferIsLargeEnoughToContain<T>(size))
471 return std::nullopt;
472
473 Vector<T, inlineCapacity, OverflowHandler, minCapacity> vector;
474 vector.grow(size);
475
476 if (!decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(vector.data()), size * sizeof(T), alignof(T)))
477 return std::nullopt;
478
479 return vector;
480 }
481};

485template<typename KeyArg, typename MappedArg, typename HashArg, typename KeyTraitsArg, typename MappedTraitsArg, typename HashTableTraits> struct ArgumentCoder<HashMap<KeyArg, MappedArg, HashArg, KeyTraitsArg, MappedTraitsArg, HashTableTraits>> {
486 typedef HashMap<KeyArg, MappedArg, HashArg, KeyTraitsArg, MappedTraitsArg, HashTableTraits> HashMapType;
487
488 static void encode(Encoder& encoder, const HashMapType& hashMap)
489 {
490 encoder << static_cast<uint32_t>(hashMap.size());
491 for (typename HashMapType::const_iterator it = hashMap.begin(), end = hashMap.end(); it != end; ++it)
492 encoder << *it;
493 }
494
495 static std::optional<HashMapType> decode(Decoder& decoder)
496 {
497 uint32_t hashMapSize;
498 if (!decoder.decode(hashMapSize))
499 return std::nullopt;
500
501 HashMapType hashMap;
502 for (uint32_t i = 0; i < hashMapSize; ++i) {
503 std::optional<KeyArg> key;
504 decoder >> key;
505 if (UNLIKELY(!key))
506 return std::nullopt;
507
508 std::optional<MappedArg> value;
509 decoder >> value;
510 if (UNLIKELY(!value))
511 return std::nullopt;
512
513 if (UNLIKELY(!HashMapType::isValidKey(*key)))
514 return std::nullopt;
515
516 if (UNLIKELY(!hashMap.add(WTFMove(*key), WTFMove(*value)).isNewEntry)) {
517 // The hash map already has the specified key, bail.
518 return std::nullopt;
519 }
520 }
521
522 return hashMap;
523 }
524
525 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, HashMapType& hashMap)
526 {
527 std::optional<HashMapType> tempHashMap;
528 decoder >> tempHashMap;
529 if (!tempHashMap)
530 return false;
531 hashMap.swap(*tempHashMap);
532 return true;
533 }
534};

536template<typename KeyArg, typename HashArg, typename KeyTraitsArg, typename HashTableTraits> struct ArgumentCoder<HashSet<KeyArg, HashArg, KeyTraitsArg, HashTableTraits>> {
537 typedef HashSet<KeyArg, HashArg, KeyTraitsArg, HashTableTraits> HashSetType;
538
539 static void encode(Encoder& encoder, const HashSetType& hashSet)
540 {
541 encoder << static_cast<uint64_t>(hashSet.size());
542 for (typename HashSetType::const_iterator it = hashSet.begin(), end = hashSet.end(); it != end; ++it)
543 encoder << *it;
544 }
545
546 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, HashSetType& hashSet)
547 {
548 std::optional<HashSetType> tempHashSet;
549 decoder >> tempHashSet;
550 if (!tempHashSet)
551 return false;
552
553 hashSet.swap(tempHashSet.value());
554 return true;
555 }
556
557 static std::optional<HashSetType> decode(Decoder& decoder)
558 {
559 uint64_t hashSetSize;
560 if (!decoder.decode(hashSetSize))
561 return std::nullopt;
562
563 HashSetType hashSet;
564 for (uint64_t i = 0; i < hashSetSize; ++i) {
565 std::optional<KeyArg> key;
566 decoder >> key;
567 if (!key)
568 return std::nullopt;
569
570 if (UNLIKELY(!HashSetType::isValidValue(*key)))
571 return std::nullopt;
572
573 if (UNLIKELY(!hashSet.add(WTFMove(*key)).isNewEntry)) {
574 // The hash set already has the specified key, bail.
575 return std::nullopt;
576 }
577 }
578
579 return hashSet;
580 }

583template<typename KeyArg, typename HashArg, typename KeyTraitsArg> struct ArgumentCoder<HashCountedSet<KeyArg, HashArg, KeyTraitsArg>> {
584 typedef HashCountedSet<KeyArg, HashArg, KeyTraitsArg> HashCountedSetType;
585
586 static void encode(Encoder& encoder, const HashCountedSetType& hashCountedSet)
587 {
588 encoder << static_cast<uint64_t>(hashCountedSet.size());
589
590 for (auto entry : hashCountedSet) {
591 encoder << entry.key;
592 encoder << entry.value;
593 }
594 }
595
596 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, HashCountedSetType& hashCountedSet)
597 {
598 uint64_t hashCountedSetSize;
599 if (!decoder.decode(hashCountedSetSize))
600 return false;
601
602 HashCountedSetType tempHashCountedSet;
603 for (uint64_t i = 0; i < hashCountedSetSize; ++i) {
604 KeyArg key;
605 if (!decoder.decode(key))
606 return false;
607
608 unsigned count;
609 if (!decoder.decode(count))
610 return false;
611
612 if (UNLIKELY(!HashCountedSetType::isValidValue(key)))
613 return false;
614
615 if (UNLIKELY(!tempHashCountedSet.add(key, count).isNewEntry)) {
616 // The hash counted set already has the specified key, bail.
617 return false;
618 }
619 }
620
621 hashCountedSet.swap(tempHashCountedSet);
622 return true;
623 }

626template<typename ValueType, typename ErrorType> struct ArgumentCoder<Expected<ValueType, ErrorType>> {
627 static void encode(Encoder& encoder, const Expected<ValueType, ErrorType>& expected)
628 {
629 if (!expected.has_value()) {
630 encoder << false;
631 encoder << expected.error();
632 return;
633 }
634 encoder << true;
635 encoder << expected.value();
636 }
637
638 static std::optional<Expected<ValueType, ErrorType>> decode(Decoder& decoder)
639 {
640 std::optional<bool> hasValue;
641 decoder >> hasValue;
642 if (!hasValue)
643 return std::nullopt;
644
645 if (*hasValue) {
646 std::optional<ValueType> value;
647 decoder >> value;
648 if (!value)
649 return std::nullopt;
650
651 Expected<ValueType, ErrorType> expected(WTFMove(*value));
652 return expected;
653 }
654 std::optional<ErrorType> error;
655 decoder >> error;
656 if (!error)
657 return std::nullopt;
658 return { makeUnexpected(WTFMove(*error)) };
659 }
660};

662template<size_t index, typename... Types>
663struct VariantCoder {
664 static void encode(Encoder& encoder, const WTF::Variant<Types...>& variant, unsigned i)
665 {
666 if (i == index) {
667 encoder << WTF::get<index>(variant);
668 return;
669 }
670 VariantCoder<index - 1, Types...>::encode(encoder, variant, i);
671 }
672
673 static std::optional<WTF::Variant<Types...>> decode(Decoder& decoder, unsigned i)
674 {
675 if (i == index) {
676 std::optional<typename WTF::variant_alternative<index, WTF::Variant<Types...>>::type> optional;
677 decoder >> optional;
678 if (!optional)
679 return std::nullopt;
680 return { WTFMove(*optional) };
681 }
682 return VariantCoder<index - 1, Types...>::decode(decoder, i);
683 }
684};
