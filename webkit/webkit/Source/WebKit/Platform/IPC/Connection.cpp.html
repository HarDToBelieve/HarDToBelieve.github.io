110 BinarySemaphore m_waitForSyncReplySemaphore;

112 // Protects m_didScheduleDispatchMessagesWorkSet and m_messagesToDispatchWhileWaitingForSyncReply.
113 Lock m_lock;

115 // The set of connections for which we've scheduled a call to dispatchMessageAndResetDidScheduleDispatchMessagesForConnection.
116 HashSet<RefPtr<Connection>> m_didScheduleDispatchMessagesWorkSet WTF_GUARDED_BY_LOCK(m_lock);

118 struct ConnectionAndIncomingMessage {
119 Ref<Connection> connection;
120 std::unique_ptr<Decoder> message;
121
122 void dispatch()
123 {
124 connection->dispatchMessage(WTFMove(message));
125 }
126 };
127 Deque<ConnectionAndIncomingMessage> m_messagesBeingDispatched; // Only used on the main thread.
128 Deque<ConnectionAndIncomingMessage> m_messagesToDispatchWhileWaitingForSyncReply WTF_GUARDED_BY_LOCK(m_lock);
129};

143void Connection::SyncMessageState::enqueueMatchingMessages(Connection& connection, MessageReceiveQueue& receiveQueue, ReceiverName receiverName, uint64_t destinationID)
144{
145 ASSERT(isMainRunLoop());
146 auto enqueueMatchingMessagesInContainer = [&](Deque<ConnectionAndIncomingMessage>& connectionAndMessages) {
147 Deque<ConnectionAndIncomingMessage> rest;
148 for (auto& connectionAndMessage : connectionAndMessages) {
149 if (connectionAndMessage.connection.ptr() == &connection && connectionAndMessage.message->messageReceiverName() == receiverName && (connectionAndMessage.message->destinationID() == destinationID || !destinationID))
150 receiveQueue.enqueueMessage(connection, WTFMove(connectionAndMessage.message));
151 else
152 rest.append(WTFMove(connectionAndMessage));
153 }
154 connectionAndMessages = WTFMove(rest);
155 };
156 Locker locker { m_lock };
157 enqueueMatchingMessagesInContainer(m_messagesBeingDispatched);
158 enqueueMatchingMessagesInContainer(m_messagesToDispatchWhileWaitingForSyncReply);
159}

161bool Connection::SyncMessageState::processIncomingMessage(Connection& connection, std::unique_ptr<Decoder>& message)
162{
163 switch (message->shouldDispatchMessageWhenWaitingForSyncReply()) {
164 case ShouldDispatchWhenWaitingForSyncReply::No:
165 return false;
166 case ShouldDispatchWhenWaitingForSyncReply::YesDuringUnboundedIPC:
167 if (!UnboundedSynchronousIPCScope::hasOngoingUnboundedSyncIPC())
168 return false;
169 break;
170 case ShouldDispatchWhenWaitingForSyncReply::Yes:
171 break;
172 }
173
174 bool shouldDispatch;
175 {
176 Locker locker { m_lock };
177 shouldDispatch = m_didScheduleDispatchMessagesWorkSet.add(&connection).isNewEntry;
178 ASSERT(connection.m_incomingMessagesLock.isHeld());
179 if (message->shouldMaintainOrderingWithAsyncMessages()) {
180 // This sync message should maintain ordering with async messages so we need to process the pending async messages first.
181 while (!connection.m_incomingMessages.isEmpty())
182 m_messagesToDispatchWhileWaitingForSyncReply.append(ConnectionAndIncomingMessage { connection, connection.m_incomingMessages.takeFirst() });
183 }
184 m_messagesToDispatchWhileWaitingForSyncReply.append(ConnectionAndIncomingMessage { connection, WTFMove(message) });
185 }
186
187 if (shouldDispatch) {
188 RunLoop::main().dispatch([this, protectedConnection = makeRef(connection)]() mutable {
189 dispatchMessagesAndResetDidScheduleDispatchMessagesForConnection(protectedConnection);
190 });
191 }
192
193 wakeUpClientRunLoop();
194
195 return true;
196}

198void Connection::SyncMessageState::dispatchMessages(Function<void(MessageName, uint64_t)>&& willDispatchMessage)
199{
200 ASSERT(RunLoop::isMain());
201
202 {
203 Locker locker { m_lock };
204 if (m_messagesBeingDispatched.isEmpty())
205 m_messagesBeingDispatched = std::exchange(m_messagesToDispatchWhileWaitingForSyncReply, { });
206 else {
207 while (!m_messagesToDispatchWhileWaitingForSyncReply.isEmpty())
208 m_messagesBeingDispatched.append(m_messagesToDispatchWhileWaitingForSyncReply.takeLast());
209 }
210 }
211
212 while (!m_messagesBeingDispatched.isEmpty()) {
213 auto messageToDispatch = m_messagesBeingDispatched.takeFirst();
214 if (willDispatchMessage)
215 willDispatchMessage(messageToDispatch.message->messageName(), messageToDispatch.message->destinationID());
216 messageToDispatch.dispatch();
217 }
218}

220void Connection::SyncMessageState::dispatchMessagesAndResetDidScheduleDispatchMessagesForConnection(Connection& connection)
221{
222 ASSERT(RunLoop::isMain());
223
224 {
225 Locker locker { m_lock };
226 ASSERT(m_didScheduleDispatchMessagesWorkSet.contains(&connection));
227 m_didScheduleDispatchMessagesWorkSet.remove(&connection);
228 ASSERT(m_messagesBeingDispatched.isEmpty());
229 Deque<ConnectionAndIncomingMessage> messagesToPutBack;
230 for (auto& connectionAndIncomingMessage : m_messagesToDispatchWhileWaitingForSyncReply) {
231 if (&connection == connectionAndIncomingMessage.connection.ptr())
232 m_messagesBeingDispatched.append(WTFMove(connectionAndIncomingMessage));
233 else
234 messagesToPutBack.append(WTFMove(connectionAndIncomingMessage));
235 }
236 m_messagesToDispatchWhileWaitingForSyncReply = WTFMove(messagesToPutBack);
237 }
238
239 while (!m_messagesBeingDispatched.isEmpty())
240 m_messagesBeingDispatched.takeFirst().dispatch();
241}

341// Enqueue any pending message to the MessageReceiveQueue that is meant to go on that queue. This is important to maintain the ordering of
342// IPC messages as some messages may get received on the IPC thread before the message receiver registered itself on the main thread.
343void Connection::enqueueMatchingMessagesToMessageReceiveQueue(MessageReceiveQueue& receiveQueue, ReceiverName receiverName, uint64_t destinationID)
344{
345 ASSERT(isMainRunLoop());
346
347 SyncMessageState::singleton().enqueueMatchingMessages(*this, receiveQueue, receiverName, destinationID);
348
349 Deque<std::unique_ptr<Decoder>> remainingIncomingMessages;
350 for (auto& message : m_incomingMessages) {
351 if (message->messageReceiverName() == receiverName && (message->destinationID() == destinationID || !destinationID))
352 receiveQueue.enqueueMessage(*this, WTFMove(message));
353 else
354 remainingIncomingMessages.append(WTFMove(message));
355 }
356 m_incomingMessages = WTFMove(remainingIncomingMessages);
357}

388void Connection::dispatchMessageReceiverMessage(MessageReceiver& messageReceiver, std::unique_ptr<Decoder>&& decoder)
389{
390 if (!decoder->isSyncMessage()) {
391 messageReceiver.didReceiveMessage(*this, *decoder);
392 return;
393 }
394
395 SyncRequestID syncRequestID;
396 if (UNLIKELY(!decoder->decode(syncRequestID))) {
397 // We received an invalid sync message.
398 // FIXME: Handle this.
399 return;
400 }
401
402 auto replyEncoder = makeUniqueRef<Encoder>(MessageName::SyncMessageReply, syncRequestID.toUInt64());
403
404 // Hand off both the decoder and encoder to the work queue message receiver.
405 bool wasHandled = messageReceiver.didReceiveSyncMessage(*this, *decoder, replyEncoder);
406
407 // FIXME: If the message was invalid, we should send back a SyncMessageError.
408 ASSERT(decoder->isValid());
409
410 if (!wasHandled)
411 sendSyncReply(WTFMove(replyEncoder));
412}

456bool Connection::sendMessage(UniqueRef<Encoder>&& encoder, OptionSet<SendOption> sendOptions)
457{
458 if (!isValid())
459 return false;
460
461#if ENABLE(IPC_TESTING_API)
462 if (isMainRunLoop()) {
463 bool hasDeadObservers = false;
464 for (auto& observerWeakPtr : m_messageObservers) {
465 if (auto* observer = observerWeakPtr.get())
466 observer->willSendMessage(encoder.get(), sendOptions);
467 else
468 hasDeadObservers = true;
469 }
470 if (hasDeadObservers)
471 m_messageObservers.removeAllMatching([](auto& observer) { return !observer; });
472 }
473#endif
474
475 if (isMainRunLoop() && m_inDispatchMessageMarkedToUseFullySynchronousModeForTesting && !encoder->isSyncMessage() && !(encoder->messageReceiverName() == ReceiverName::IPC) && !sendOptions.contains(SendOption::IgnoreFullySynchronousMode)) {
476 SyncRequestID syncRequestID;
477 auto wrappedMessage = createSyncMessageEncoder(MessageName::WrappedAsyncMessageForTesting, encoder->destinationID(), syncRequestID);
478 wrappedMessage->setFullySynchronousModeForTesting();
479 wrappedMessage->wrapForTesting(WTFMove(encoder));
480 return static_cast<bool>(sendSyncMessage(syncRequestID, WTFMove(wrappedMessage), Timeout::infinity(), { }));
481 }
482
483 if (sendOptions.contains(SendOption::DispatchMessageEvenWhenWaitingForSyncReply)
484 && (!m_onlySendMessagesAsDispatchWhenWaitingForSyncReplyWhenProcessingSuchAMessage
485 || m_inDispatchMessageMarkedDispatchWhenWaitingForSyncReplyCount))
486 encoder->setShouldDispatchMessageWhenWaitingForSyncReply(ShouldDispatchWhenWaitingForSyncReply::Yes);
487 else if (sendOptions.contains(SendOption::DispatchMessageEvenWhenWaitingForUnboundedSyncReply))
488 encoder->setShouldDispatchMessageWhenWaitingForSyncReply(ShouldDispatchWhenWaitingForSyncReply::YesDuringUnboundedIPC);
489
490 {
491 Locker locker { m_outgoingMessagesLock };
492 m_outgoingMessages.append(WTFMove(encoder));
493 }
494
495 // FIXME: We should add a boolean flag so we don't call this when work has already been scheduled.
496 m_connectionQueue->dispatch([protectedThis = makeRef(*this)]() mutable {
497 protectedThis->sendOutgoingMessages();
498 });
499 return true;
500}

512std::unique_ptr<Decoder> Connection::waitForMessage(MessageName messageName, uint64_t destinationID, Timeout timeout, OptionSet<WaitForOption> waitForOptions)
513{
514 ASSERT(RunLoop::isMain());
515 auto protectedThis = makeRef(*this);
516
517 timeout = timeoutRespectingIgnoreTimeoutsForTesting(timeout);
518
519 WaitForMessageState waitingForMessage(messageName, destinationID, waitForOptions);
520
521 {
522 Locker locker { m_waitForMessageLock };
523
524 // We don't support having multiple clients waiting for messages.
525 ASSERT(!m_waitingForMessage);
526 if (m_waitingForMessage)
527 return nullptr;
528
529 // If the connection is already invalidated, don't even start waiting.
530 // Once m_waitingForMessage is set, messageWaitingInterrupted will cover this instead.
531 if (!m_shouldWaitForMessages)
532 return nullptr;
533
534 bool hasIncomingSynchronousMessage = false;
535
536 // First, check if this message is already in the incoming messages queue.
537 {
538 Locker locker { m_incomingMessagesLock };
539 for (auto it = m_incomingMessages.begin(), end = m_incomingMessages.end(); it != end; ++it) {
540 std::unique_ptr<Decoder>& message = *it;
541
542 if (message->messageName() == messageName && message->destinationID() == destinationID) {
543 std::unique_ptr<Decoder> returnedMessage = WTFMove(message);
544
545 m_incomingMessages.remove(it);
546 return returnedMessage;
547 }
548
549 if (message->isSyncMessage())
550 hasIncomingSynchronousMessage = true;
551 }
552 }
553
554 // Don't even start waiting if we have InterruptWaitingIfSyncMessageArrives and there's a sync message already in the queue.
555 if (hasIncomingSynchronousMessage && waitForOptions.contains(WaitForOption::InterruptWaitingIfSyncMessageArrives)) {
556#if ASSERT_ENABLED
557 // We don't support having multiple clients waiting for messages.
558 ASSERT(!m_waitingForMessage);
559#endif
560 return nullptr;
561 }
562
563 m_waitingForMessage = &waitingForMessage;
564 }
565
566 // Now wait for it to be set.
567 while (true) {
568 // Handle any messages that are blocked on a response from us.
569 bool wasMessageToWaitForAlreadyDispatched = false;
570 SyncMessageState::singleton().dispatchMessages([&](auto nameOfMessageToDispatch, uint64_t destinationOfMessageToDispatch) {
571 wasMessageToWaitForAlreadyDispatched |= messageName == nameOfMessageToDispatch && destinationID == destinationOfMessageToDispatch;
572 });
573
574 Locker locker { m_waitForMessageLock };
575
576 if (wasMessageToWaitForAlreadyDispatched) {
577 m_waitingForMessage = nullptr;
578 break;
579 }
580
581 if (UNLIKELY(m_inDispatchSyncMessageCount && !timeout.isInfinity())) {
582 RELEASE_LOG_ERROR(IPC, "Connection::waitForMessage(%{public}s): Exiting immediately, since we're handling a sync message already", description(messageName));
583 m_waitingForMessage = nullptr;
584 break;
585 }
586
587 if (m_waitingForMessage->decoder) {
588 auto decoder = WTFMove(m_waitingForMessage->decoder);
589 m_waitingForMessage = nullptr;
590 return decoder;
591 }
592
593 // Now we wait.
594 bool didTimeout = !m_waitForMessageCondition.waitUntil(m_waitForMessageLock, timeout.deadline());
595 // We timed out, lost our connection, or a sync message came in with InterruptWaitingIfSyncMessageArrives, so stop waiting.
596 if (didTimeout || m_waitingForMessage->messageWaitingInterrupted) {
597 m_waitingForMessage = nullptr;
598 break;
599 }
600 }
601
602 return nullptr;
603}

605bool Connection::pushPendingSyncRequestID(SyncRequestID syncRequestID)
606{
607 {
608 Locker locker { m_syncReplyStateLock };
609 if (!m_shouldWaitForSyncReplies)
610 return false;

612 }

617void Connection::popPendingSyncRequestID(SyncRequestID syncRequestID)
618{
619 --m_inSendSyncCount;
620 Locker locker { m_syncReplyStateLock };
621 ASSERT_UNUSED(syncRequestID, m_pendingSyncReplies.last().syncRequestID == syncRequestID);
622 m_pendingSyncReplies.removeLast();
623}

625std::unique_ptr<Decoder> Connection::sendSyncMessage(SyncRequestID syncRequestID, UniqueRef<Encoder>&& encoder, Timeout timeout, OptionSet<SendSyncOption> sendSyncOptions)
626{
627 ASSERT(syncRequestID);
628 ASSERT(RunLoop::isMain());
629
630 if (!isValid()) {
631 didFailToSendSyncMessage();
632 return nullptr;
633 }
634 if (!pushPendingSyncRequestID(syncRequestID)) {
635 didFailToSendSyncMessage();
636 return nullptr;
637 }
638
639 // First send the message.
640 OptionSet<SendOption> sendOptions = IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply;
641 if (sendSyncOptions.contains(SendSyncOption::ForceDispatchWhenDestinationIsWaitingForUnboundedSyncReply))
642 sendOptions = sendOptions | IPC::SendOption::DispatchMessageEvenWhenWaitingForUnboundedSyncReply;
643
644 if (sendSyncOptions.contains(IPC::SendSyncOption::MaintainOrderingWithAsyncMessages))
645 encoder->setShouldMaintainOrderingWithAsyncMessages();
646
647 auto messageName = encoder->messageName();
648 sendMessage(WTFMove(encoder), sendOptions);
649
650 // Then wait for a reply. Waiting for a reply could involve dispatching incoming sync messages, so
651 // keep an extra reference to the connection here in case it's invalidated.
652 Ref<Connection> protect(*this);
653 std::unique_ptr<Decoder> reply = waitForSyncReply(syncRequestID, messageName, timeout, sendSyncOptions);
654
655 popPendingSyncRequestID(syncRequestID);
656
657 if (!reply)
658 didFailToSendSyncMessage();
659
660 return reply;
661}

663std::unique_ptr<Decoder> Connection::waitForSyncReply(SyncRequestID syncRequestID, MessageName messageName, Timeout timeout, OptionSet<SendSyncOption> sendSyncOptions)
664{
665 timeout = timeoutRespectingIgnoreTimeoutsForTesting(timeout);
666
667 willSendSyncMessage(sendSyncOptions);
668
669 bool timedOut = false;
670 while (!timedOut) {
671 // First, check if we have any messages that we need to process.
672 SyncMessageState::singleton().dispatchMessages();
673
674 {
675 Locker locker { m_syncReplyStateLock };
676
677 // Second, check if there is a sync reply at the top of the stack.
678 ASSERT(!m_pendingSyncReplies.isEmpty());
679
680 PendingSyncReply& pendingSyncReply = m_pendingSyncReplies.last();
681 ASSERT_UNUSED(syncRequestID, pendingSyncReply.syncRequestID == syncRequestID);
682
683 // We found the sync reply, or the connection was closed.
684 if (pendingSyncReply.didReceiveReply || !m_shouldWaitForSyncReplies) {
685 didReceiveSyncReply(sendSyncOptions);
686 return WTFMove(pendingSyncReply.replyDecoder);
687 }
688 }
689
690 // Processing a sync message could cause the connection to be invalidated.
691 // (If the handler ends up calling Connection::invalidate).
692 // If that happens, we need to stop waiting, or we'll hang since we won't get
693 // any more incoming messages.
694 if (!isValid()) {
695 RELEASE_LOG_ERROR(IPC, "Connection::waitForSyncReply: Connection no longer valid, id=%" PRIu64, syncRequestID.toUInt64());
696 didReceiveSyncReply(sendSyncOptions);
697 return nullptr;
698 }
699
700 // We didn't find a sync reply yet, keep waiting.
701 // This allows the WebProcess to still serve clients while waiting for the message to return.
702 // Notably, it can continue to process accessibility requests, which are on the main thread.
703 timedOut = !SyncMessageState::singleton().wait(timeout);
704 }
705
706#if OS(DARWIN)
707 RELEASE_LOG_ERROR(IPC, "Connection::waitForSyncReply: Timed-out while waiting for reply for %{public}s from process %d, id=%" PRIu64, description(messageName), remoteProcessID(), syncRequestID.toUInt64());
708#else
709 RELEASE_LOG_ERROR(IPC, "Connection::waitForSyncReply: Timed-out while waiting for reply for %s, id=%" PRIu64, description(messageName), syncRequestID.toUInt64());
710#endif
711
712 didReceiveSyncReply(sendSyncOptions);
713
714 return nullptr;
715}

717void Connection::processIncomingSyncReply(std::unique_ptr<Decoder> decoder)
718{
719 {
720 Locker locker { m_syncReplyStateLock };
721
722 // Go through the stack of sync requests that have pending replies and see which one
723 // this reply is for.
724 for (size_t i = m_pendingSyncReplies.size(); i > 0; --i) {
725 PendingSyncReply& pendingSyncReply = m_pendingSyncReplies[i - 1];
726
727 if (pendingSyncReply.syncRequestID.toUInt64() != decoder->destinationID())
728 continue;
729
730 ASSERT(!pendingSyncReply.replyDecoder);
731
732 pendingSyncReply.replyDecoder = WTFMove(decoder);
733 pendingSyncReply.didReceiveReply = true;
734
735 // We got a reply to the last send message, wake up the client run loop so it can be processed.
736 if (i == m_pendingSyncReplies.size())
737 SyncMessageState::singleton().wakeUpClientRunLoop();
738
739 return;
740 }
741 }
742
743 // If we get here, it means we got a reply for a message that wasn't in the sync request stack or map.
744 // This can happen if the send timed out, so it's fine to ignore.
745}

747void Connection::processIncomingMessage(std::unique_ptr<Decoder> message)
748{
749 ASSERT(message->messageReceiverName() != ReceiverName::Invalid);
750
751 if (message->messageName() == MessageName::SyncMessageReply) {
752 processIncomingSyncReply(WTFMove(message));
753 return;
754 }
755
756 if (!MessageReceiveQueueMap::isValidMessage(*message)) {
757 RunLoop::main().dispatch([protectedThis = makeRef(*this), messageName = message->messageName()]() mutable {
758 protectedThis->dispatchDidReceiveInvalidMessage(messageName);
759 });
760 return;
761 }
762
763 // FIXME: These are practically the same mutex, so maybe they could be merged.
764 Locker waitForMessagesLocker { m_waitForMessageLock };
765
766 Locker incomingMessagesLocker { m_incomingMessagesLock };
767 if (auto* receiveQueue = m_receiveQueues.get(*message)) {
768 receiveQueue->enqueueMessage(*this, WTFMove(message));
769 return;
770 }
771
772 if (message->isSyncMessage()) {
773 Locker locker { m_incomingSyncMessageCallbackLock };
774
775 for (auto& callback : m_incomingSyncMessageCallbacks.values())
776 m_incomingSyncMessageCallbackQueue->dispatch(WTFMove(callback));
777
778 m_incomingSyncMessageCallbacks.clear();
779 }
780
781 // Check if we're waiting for this message, or if we need to interrupt waiting due to an incoming sync message.
782 if (m_waitingForMessage && !m_waitingForMessage->decoder) {
783 if (m_waitingForMessage->messageName == message->messageName() && m_waitingForMessage->destinationID == message->destinationID()) {
784 m_waitingForMessage->decoder = WTFMove(message);
785 ASSERT(m_waitingForMessage->decoder);
786 m_waitForMessageCondition.notifyOne();
787 return;
788 }
789
790 if (m_waitingForMessage->waitForOptions.contains(WaitForOption::DispatchIncomingSyncMessagesWhileWaiting) && message->isSyncMessage() && SyncMessageState::singleton().processIncomingMessage(*this, message)) {
791 m_waitForMessageCondition.notifyOne();
792 return;
793 }
794
795 if (m_waitingForMessage->waitForOptions.contains(WaitForOption::InterruptWaitingIfSyncMessageArrives) && message->isSyncMessage()) {
796 m_waitingForMessage->messageWaitingInterrupted = true;
797 m_waitForMessageCondition.notifyOne();
798 enqueueIncomingMessage(WTFMove(message));
799 return;
800 }
801 }
802
803 // Check if this is a sync message or if it's a message that should be dispatched even when waiting for
804 // a sync reply. If it is, and we're waiting for a sync reply this message needs to be dispatched.
805 // If we don't we'll end up with a deadlock where both sync message senders are stuck waiting for a reply.
806 if (SyncMessageState::singleton().processIncomingMessage(*this, message))
807 return;
808
809 enqueueIncomingMessage(WTFMove(message));
810}

866void Connection::connectionDidClose()
867{
868 // The connection is now invalid.
869 platformInvalidate();
870
871 {
872 Locker locker { m_syncReplyStateLock };
873
874 ASSERT(m_shouldWaitForSyncReplies);
875 m_shouldWaitForSyncReplies = false;
876
877 if (!m_pendingSyncReplies.isEmpty())
878 SyncMessageState::singleton().wakeUpClientRunLoop();
879 }
880
881 {
882 Locker locker { m_waitForMessageLock };
883
884 ASSERT(m_shouldWaitForMessages);
885 m_shouldWaitForMessages = false;
886
887 if (m_waitingForMessage)
888 m_waitingForMessage->messageWaitingInterrupted = true;
889 }
890 m_waitForMessageCondition.notifyAll();
891
892 if (m_didCloseOnConnectionWorkQueueCallback)
893 m_didCloseOnConnectionWorkQueueCallback(this);
894
895 RunLoop::main().dispatch([protectedThis = makeRef(*this)]() mutable {
896 // If the connection has been explicitly invalidated before dispatchConnectionDidClose was called,
897 // then the connection will be invalid here.
898 if (!protectedThis->isValid())
899 return;
900
901 // Set m_isValid to false before calling didClose, otherwise, sendSync will try to send a message
902 // to the connection and will then wait indefinitely for a reply.
903 protectedThis->m_isValid = false;
904
905 protectedThis->m_client.didClose(protectedThis.get());
906
907 clearAsyncReplyHandlers(protectedThis.get());
908 });
909}

937void Connection::dispatchSyncMessage(Decoder& decoder)
938{
939 ASSERT(isMainRunLoop());
940 ASSERT(decoder.isSyncMessage());
941
942 SyncRequestID syncRequestID;
943 if (UNLIKELY(!decoder.decode(syncRequestID))) {
944 // We received an invalid sync message.
945 return;
946 }
947
948 ++m_inDispatchSyncMessageCount;
949 auto decrementSyncMessageCount = makeScopeExit([&] {
950 ASSERT(m_inDispatchSyncMessageCount);
951 --m_inDispatchSyncMessageCount;
952 });
953
954 auto replyEncoder = makeUniqueRef<Encoder>(MessageName::SyncMessageReply, syncRequestID.toUInt64());
955
956 bool wasHandled = false;
957 if (decoder.messageName() == MessageName::WrappedAsyncMessageForTesting) {
958 if (!m_fullySynchronousModeIsAllowedForTesting) {
959 decoder.markInvalid();
960 return;
961 }
962 std::unique_ptr<Decoder> unwrappedDecoder = Decoder::unwrapForTesting(decoder);
963 RELEASE_ASSERT(unwrappedDecoder);
964 processIncomingMessage(WTFMove(unwrappedDecoder));
965
966 SyncMessageState::singleton().dispatchMessages();
967 } else {
968 // Hand off both the decoder and encoder to the client.
969 wasHandled = m_client.didReceiveSyncMessage(*this, decoder, replyEncoder);
970 }
971
972 // FIXME: If the message was invalid, we should send back a SyncMessageError.
973#if ENABLE(IPC_TESTING_API)
974 ASSERT(decoder.isValid() || m_ignoreInvalidMessageForTesting);
975#else
976 ASSERT(decoder.isValid());
977#endif
978
979 if (!wasHandled)
980 sendSyncReply(WTFMove(replyEncoder));
981}

1031void Connection::dispatchMessage(Decoder& decoder)
1032{
1033 RELEASE_ASSERT(isValid());
1034 if (decoder.messageReceiverName() == ReceiverName::AsyncReply) {
1035 auto handler = takeAsyncReplyHandler(*this, decoder.destinationID());
1036 if (!handler) {
1037 ASSERT_NOT_REACHED();

1039 }
1040 handler(&decoder);
1041 return;
1042 }

1044#if ENABLE(IPC_TESTING_API)

1046 bool hasDeadObservers = false;

1050 else
1051 hasDeadObservers = true;

1055 }
1056#endif
1057
1058 m_client.didReceiveMessage(*this, decoder);
1059}

1061void Connection::dispatchMessage(std::unique_ptr<Decoder> message)
1062{
1063 ASSERT(RunLoop::isMain());
1064 if (!isValid())
1065 return;
1066
1067 {
1068 // FIXME: The matches here come from
1069 // m_messagesToDispatchWhileWaitingForSyncReply. This causes message
1070 // reordering, because some of the messages go to
1071 // SyncState::m_messagesToDispatchWhileWaitingForSyncReply while others
1072 // go to Connection::m_incomingMessages. Should be fixed by adding all
1073 // messages to one list.
1074 Locker incomingMessagesLocker { m_incomingMessagesLock };
1075 if (auto* receiveQueue = m_receiveQueues.get(*message)) {
1076 receiveQueue->enqueueMessage(*this, WTFMove(message));
1077 return;
1078 }
1079 }
1080
1081 if (message->shouldUseFullySynchronousModeForTesting()) {
1082 if (!m_fullySynchronousModeIsAllowedForTesting) {
1083 m_client.didReceiveInvalidMessage(*this, message->messageName());
1084 return;
1085 }
1086 m_inDispatchMessageMarkedToUseFullySynchronousModeForTesting++;
1087 }
1088
1089 m_inDispatchMessageCount++;
1090
1091 bool isDispatchingMessageWhileWaitingForSyncReply = (message->shouldDispatchMessageWhenWaitingForSyncReply() == ShouldDispatchWhenWaitingForSyncReply::Yes)
1092 || (message->shouldDispatchMessageWhenWaitingForSyncReply() == ShouldDispatchWhenWaitingForSyncReply::YesDuringUnboundedIPC && UnboundedSynchronousIPCScope::hasOngoingUnboundedSyncIPC());
1093
1094 if (isDispatchingMessageWhileWaitingForSyncReply)
1095 m_inDispatchMessageMarkedDispatchWhenWaitingForSyncReplyCount++;
1096
1097 bool oldDidReceiveInvalidMessage = m_didReceiveInvalidMessage;
1098 m_didReceiveInvalidMessage = false;
1099
1100 if (message->isSyncMessage())
1101 dispatchSyncMessage(*message);
1102 else
1103 dispatchMessage(*message);
1104
1105 m_didReceiveInvalidMessage |= !message->isValid();
1106 m_inDispatchMessageCount--;
1107
1108 // FIXME: For synchronous messages, we should not decrement the counter until we send a response.
1109 // Otherwise, we would deadlock if processing the message results in a sync message back after we exit this function.
1110 if (isDispatchingMessageWhileWaitingForSyncReply)
1111 m_inDispatchMessageMarkedDispatchWhenWaitingForSyncReplyCount--;
1112
1113 if (message->shouldUseFullySynchronousModeForTesting())
1114 m_inDispatchMessageMarkedToUseFullySynchronousModeForTesting--;
1115
1116 if (m_didReceiveInvalidMessage && isValid())
1117 m_client.didReceiveInvalidMessage(*this, message->messageName());
1118
1119 m_didReceiveInvalidMessage = oldDidReceiveInvalidMessage;
1120}

1143size_t Connection::MessagesThrottler::numberOfMessagesToProcess(size_t totalMessages)
1144{
1145 ASSERT(RunLoop::isMain());
1146
1147 // Never dispatch more than 600 messages without returning to the run loop, we can go as low as 60 with maximum throttling level.
1148 static const size_t maxIncomingMessagesDispatchingBatchSize { 600 };
1149 static const unsigned maxThrottlingLevel = 9;
1150
1151 size_t batchSize = maxIncomingMessagesDispatchingBatchSize / (m_throttlingLevel + 1);
1152
1153 if (totalMessages > maxIncomingMessagesDispatchingBatchSize)
1154 m_throttlingLevel = std::min(m_throttlingLevel + 1, maxThrottlingLevel);
1155 else if (m_throttlingLevel)
1156 --m_throttlingLevel;
1157
1158 return std::min(totalMessages, batchSize);
1159}

1174
1175void Connection::dispatchIncomingMessages()
1176{
1177 ASSERT(RunLoop::isMain());
1178
1179 std::unique_ptr<Decoder> message;
1180
1181 size_t messagesToProcess = 0;
1182 {
1183 Locker locker { m_incomingMessagesLock };
1184 if (m_incomingMessages.isEmpty())
1185 return;
1186
1187 message = m_incomingMessages.takeFirst();
1188
1189 // Incoming messages may get adding to the queue by the IPC thread while we're dispatching the messages below.
1190 // To make sure dispatchIncomingMessages() yields, we only ever process messages that were in the queue when
1191 // dispatchIncomingMessages() was called. Additionally, the MessageThrottler may further cap the number of
1192 // messages to process to make sure we give the main run loop a chance to process other events.
1193 messagesToProcess = m_incomingMessagesThrottler->numberOfMessagesToProcess(m_incomingMessages.size());
1194 if (messagesToProcess < m_incomingMessages.size()) {
1195 RELEASE_LOG_ERROR(IPC, "%p - Connection::dispatchIncomingMessages: IPC throttling was triggered (has %zu pending incoming messages, will only process %zu before yielding)", this, m_incomingMessages.size(), messagesToProcess);
1196#if PLATFORM(COCOA)
1197 RELEASE_LOG_ERROR(IPC, "%p - Connection::dispatchIncomingMessages: first IPC message in queue is %{public}s", this, description(message->messageName()));
1198#endif
1199 }
1200
1201 // Re-schedule ourselves *before* we dispatch the messages because we want to process follow-up messages if the client
1202 // spins a nested run loop while we're dispatching a message. Note that this means we can re-enter this method.
1203 if (!m_incomingMessages.isEmpty())
1204 m_incomingMessagesThrottler->scheduleMessagesDispatch();
1205 }
1206
1207 dispatchMessage(WTFMove(message));
1208
1209 for (size_t i = 1; i < messagesToProcess; ++i) {
1210 {
1211 Locker locker { m_incomingMessagesLock };
1212 if (m_incomingMessages.isEmpty())
1213 return;
1214
1215 message = m_incomingMessages.takeFirst();
1216 }
1217 dispatchMessage(WTFMove(message));
1218 }
1219}

1250CompletionHandler<void(Decoder*)> takeAsyncReplyHandler(Connection& connection, uint64_t identifier)
1251{
1252 Locker locker { asyncReplyHandlerMapLock };
1253 auto& map = asyncReplyHandlerMap();
1254 auto iterator = map.find(reinterpret_cast<uintptr_t>(&connection));
1255 if (iterator != map.end()) {
1256 if (!iterator->value.isValidKey(identifier)) {
1257 ASSERT_NOT_REACHED();
1258 connection.markCurrentlyDispatchedMessageAsInvalid();
1259 return nullptr;
1260 }
1261 ASSERT(iterator->value.contains(identifier));
1262 return iterator->value.take(identifier);
1263 }
1264 ASSERT_NOT_REACHED();
