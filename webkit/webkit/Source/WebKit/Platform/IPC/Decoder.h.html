88 template<typename T>
89 WARN_UNUSED_RETURN bool decode(T& t)
90 {
91 using Impl = ArgumentCoder<std::remove_const_t<std::remove_reference_t<T>>, void>;
92 if constexpr(HasLegacyDecoder<T, Impl>::value) {
93 if (UNLIKELY(!Impl::decode(*this, t))) {
94 markInvalid();
95 return false;
96 }
97 } else {
98 std::optional<T> optional;
99 *this >> optional;
100 if (UNLIKELY(!optional)) {
101 markInvalid();
102 return false;
103 }
104 t = WTFMove(*optional);
105 }
106 return true;
107 }
108
109 template<typename T>
110 Decoder& operator>>(std::optional<T>& t)
111 {
112 using Impl = ArgumentCoder<std::remove_const_t<std::remove_reference_t<T>>, void>;
113 if constexpr(HasModernDecoder<T, Impl>::value) {
114 t = Impl::decode(*this);
115 if (UNLIKELY(!t))
116 markInvalid();
117 } else {
118 T v;
119 if (LIKELY(Impl::decode(*this, v)))
120 t = WTFMove(v);
121 else
122 markInvalid();
123 }
124 return *this;
125 }

127 template<typename T>
128 bool bufferIsLargeEnoughToContain(size_t numElements) const
129 {
130 static_assert(std::is_arithmetic<T>::value, "Type T must have a fixed, known encoded size!");
131
132 if (numElements > std::numeric_limits<size_t>::max() / sizeof(T))
133 return false;
134
135 return bufferIsLargeEnoughToContain(alignof(T), numElements * sizeof(T));
136 }

142 template <typename T>
143 static std::optional<T> decodeSingleObject(const uint8_t* source, size_t numberOfBytes)
144 {
145 std::optional<T> result;
146 Decoder decoder(source, numberOfBytes, ConstructWithoutHeader);
147 if (!decoder.isValid())
148 return std::nullopt;
149
150 decoder >> result;
151 return result;
152 }
