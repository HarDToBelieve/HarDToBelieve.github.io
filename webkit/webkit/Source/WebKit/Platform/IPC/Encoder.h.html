27
28#include "Attachment.h"
29#include "MessageNames.h"

36namespace IPC {
37
38enum class MessageFlags : uint8_t;
39enum class ShouldDispatchWhenWaitingForSyncReply : uint8_t;
40
41template<typename, typename> struct ArgumentCoder;
42
43class Encoder final {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 Encoder(MessageName, uint64_t destinationID);
47 ~Encoder();
48
49 ReceiverName messageReceiverName() const { return receiverName(m_messageName); }
50 MessageName messageName() const { return m_messageName; }
51 uint64_t destinationID() const { return m_destinationID; }
52
53 bool isSyncMessage() const { return messageIsSync(messageName()); }
54
55 void setShouldDispatchMessageWhenWaitingForSyncReply(ShouldDispatchWhenWaitingForSyncReply);
56 ShouldDispatchWhenWaitingForSyncReply shouldDispatchMessageWhenWaitingForSyncReply() const;
57
58 void setFullySynchronousModeForTesting();
59 void setShouldMaintainOrderingWithAsyncMessages();
60
61 void wrapForTesting(UniqueRef<Encoder>&&);
62
63 void encodeFixedLengthData(const uint8_t* data, size_t, size_t alignment);
64
65 template<typename T>
66 Encoder& operator<<(T&& t)
67 {
68 ArgumentCoder<std::remove_const_t<std::remove_reference_t<T>>, void>::encode(*this, std::forward<T>(t));
69 return *this;
70 }
71
72 uint8_t* buffer() const { return m_buffer; }
73 size_t bufferSize() const { return m_bufferSize; }
74
75 void addAttachment(Attachment&&);
76 Vector<Attachment> releaseAttachments();
77 void reserve(size_t);
78
79 static const bool isIPCEncoder = true;
80
81 template<typename T>
82 static RefPtr<WebCore::SharedBuffer> encodeSingleObject(const T& object)
83 {
84 Encoder encoder(ConstructWithoutHeader);
85 encoder << object;
86
87 if (encoder.hasAttachments()) {
88 ASSERT_NOT_REACHED();
89 return nullptr;
90 }
91
92 return WebCore::SharedBuffer::create(encoder.buffer(), encoder.bufferSize());
93 }
94
95private:
96 enum ConstructWithoutHeaderTag { ConstructWithoutHeader };
97 Encoder(ConstructWithoutHeaderTag);
98
99 uint8_t* grow(size_t alignment, size_t);
100
101 bool hasAttachments() const;
102
103 void encodeHeader();
104 const OptionSet<MessageFlags>& messageFlags() const;
105 OptionSet<MessageFlags>& messageFlags();
106
107 MessageName m_messageName;
108 uint64_t m_destinationID;
109
110 uint8_t m_inlineBuffer[512];
111
112 uint8_t* m_buffer;
113 uint8_t* m_bufferPointer;
114
115 size_t m_bufferSize;
116 size_t m_bufferCapacity;
117
118 Vector<Attachment> m_attachments;
119};
120
121} // namespace IPC
