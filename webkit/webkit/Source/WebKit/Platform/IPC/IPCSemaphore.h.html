26#pragma once
27
28#include "Timeout.h"
29#include <wtf/Noncopyable.h>
30#include <wtf/Seconds.h>
31
32#if OS(DARWIN)
33#include <mach/semaphore.h>
34#include <wtf/MachSendRight.h>
35#elif OS(WINDOWS)
36#include <windows.h>
37#endif
38
39namespace IPC {
40
41class Decoder;
42class Encoder;
43
44class Semaphore {
45 WTF_MAKE_FAST_ALLOCATED;
46 WTF_MAKE_NONCOPYABLE(Semaphore);
47public:
48 Semaphore();
49 Semaphore(Semaphore&&);
50 ~Semaphore();
51 Semaphore& operator=(Semaphore&&);
52
53 void encode(Encoder&) const;
54 static std::optional<Semaphore> decode(Decoder&);
55
56 void signal();
57 bool wait();
58 bool waitFor(Timeout);
59
60#if OS(DARWIN)
61 explicit Semaphore(MachSendRight&&);
62
63 MachSendRight createSendRight() const;
64 explicit operator bool() const { return m_sendRight || m_semaphore != SEMAPHORE_NULL; }
65#elif OS(WINDOWS)
66 explicit Semaphore(HANDLE);
67#else
68 explicit operator bool() const { return true; }
69#endif
70
71private:
72 void destroy();
73#if OS(DARWIN)
74 MachSendRight m_sendRight;
75 semaphore_t m_semaphore { SEMAPHORE_NULL };
76#elif OS(WINDOWS)
77 HANDLE m_semaphoreHandle { nullptr };
78#endif
79};
80
81} // namespace IPC
