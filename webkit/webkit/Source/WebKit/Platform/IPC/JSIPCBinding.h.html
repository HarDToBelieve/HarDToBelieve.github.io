106template<typename U>
107JSC::JSValue jsValueForDecodedArgumentValue(JSC::JSGlobalObject* globalObject, OptionSet<U>&& value)
108{
109 auto& vm = globalObject->vm();
110 auto scope = DECLARE_THROW_SCOPE(vm);
111 auto result = jsValueForDecodedArgumentValue(globalObject, value.toRaw());
112 RETURN_IF_EXCEPTION(scope, JSC::JSValue());
113 result.getObject()->putDirect(vm, JSC::Identifier::fromString(vm, "isOptionSet"_s), JSC::jsBoolean(true));
114 RETURN_IF_EXCEPTION(scope, JSC::JSValue());
115 return result;
116}

129template<typename T, typename... Elements>
130std::optional<JSC::JSValue> putJSValueForDecodeArgumentInArray(JSC::JSGlobalObject* globalObject, IPC::Decoder& decoder, JSC::JSArray* array, size_t currentIndex, std::tuple<T, Elements...>*)
131{
132 auto startingBufferPosition = decoder.currentBufferPosition();
133 std::optional<T> value;
134 decoder >> value;
135 if (!value)
136 return std::nullopt;
137
138 auto jsValue = jsValueForDecodedArgumentValue(globalObject, WTFMove(*value));
139 if (jsValue.isEmpty())
140 return jsValue;
141
142 putJSValueForDecodedArgumentAtIndexOrArrayBufferIfUndefined(globalObject, array, currentIndex, jsValue,
143 decoder.buffer() + startingBufferPosition, decoder.currentBufferPosition() - startingBufferPosition);
144
145 std::tuple<Elements...>* dummyArguments = nullptr;
146 return putJSValueForDecodeArgumentInArray<Elements...>(globalObject, decoder, array, currentIndex + 1, dummyArguments);
147}
148
149template<typename T>
150static std::optional<JSC::JSValue> jsValueForDecodedArguments(JSC::JSGlobalObject* globalObject, IPC::Decoder& decoder)
151{
152 auto scope = DECLARE_THROW_SCOPE(globalObject->vm());
153 auto* array = JSC::constructEmptyArray(globalObject, nullptr);
154 RETURN_IF_EXCEPTION(scope, JSC::JSValue());
155 typename IPC::CodingType<T>::Type* dummyArguments = nullptr;
156 return putJSValueForDecodeArgumentInArray<>(globalObject, decoder, array, 0, dummyArguments);
157}
