29namespace IPC {

31void MessageReceiveQueueMap::addImpl(StoreType&& queue, ReceiverName receiverName, uint64_t destinationID)
32{
33 auto key = std::make_pair(static_cast<uint8_t>(receiverName), destinationID);
34 ASSERT(!m_queues.contains(key));
35 m_queues.add(key, WTFMove(queue));
36}
37
38void MessageReceiveQueueMap::remove(ReceiverName receiverName, uint64_t destinationID)
39{
40 auto key = std::make_pair(static_cast<uint8_t>(receiverName), destinationID);
41 ASSERT(m_queues.contains(key));
42 m_queues.remove(key);
43}
44
45MessageReceiveQueue* MessageReceiveQueueMap::get(const Decoder& message) const
46{
47 if (m_queues.isEmpty())
48 return nullptr;
49 auto key = std::make_pair(static_cast<uint8_t>(message.messageReceiverName()), 0);
50 auto it = m_queues.find(key);
51 if (it == m_queues.end()) {
52 key.second = message.destinationID();
53 if (key.second)
54 it = m_queues.find(key);
55 }
56 if (it == m_queues.end())
57 return nullptr;
58 return WTF::visit(
59 WTF::makeVisitor(
60 [](const std::unique_ptr<MessageReceiveQueue>& queue) { return queue.get(); },
61 [](MessageReceiveQueue* queue) { return queue; }
62 ), it->value);
63}
