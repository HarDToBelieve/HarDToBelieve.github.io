31namespace IPC {
32
33class FunctionDispatcherQueue final : public MessageReceiveQueue {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 FunctionDispatcherQueue(FunctionDispatcher& dispatcher, MessageReceiver& receiver)
37 : m_dispatcher(dispatcher)
38 , m_receiver(receiver)
39 {
40 }
41 ~FunctionDispatcherQueue() final = default;
42
43 void enqueueMessage(Connection& connection, std::unique_ptr<Decoder>&& message) final
44 {
45 m_dispatcher.dispatch([connection = makeRef(connection), message = WTFMove(message), &receiver = m_receiver]() mutable {
46 connection->dispatchMessageReceiverMessage(receiver, WTFMove(message));
47 });
48 }
49private:
50 FunctionDispatcher& m_dispatcher;
51 MessageReceiver& m_receiver;
52};
53
54class ThreadMessageReceiverQueue final : public MessageReceiveQueue {
55 WTF_MAKE_FAST_ALLOCATED;
56public:
57 ThreadMessageReceiverQueue(Connection::ThreadMessageReceiver& receiver)
58 : m_receiver(makeRef(receiver))
59 {
60 }
61 ~ThreadMessageReceiverQueue() final = default;
62
63 void enqueueMessage(Connection& connection, std::unique_ptr<Decoder>&& message) final
64 {
65 m_receiver->dispatchToThread([connection = makeRef(connection), message = WTFMove(message), receiver = m_receiver]() mutable {
66 connection->dispatchMessageReceiverMessage(receiver.get(), WTFMove(message));
67 });
68 }
69private:
70 Ref<Connection::ThreadMessageReceiver> m_receiver;
71};
