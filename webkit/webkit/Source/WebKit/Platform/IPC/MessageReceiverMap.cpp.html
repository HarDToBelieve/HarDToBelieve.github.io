42void MessageReceiverMap::addMessageReceiver(ReceiverName messageReceiverName, MessageReceiver& messageReceiver)
43{
44 ASSERT(!m_globalMessageReceivers.contains(messageReceiverName));
45
46 messageReceiver.willBeAddedToMessageReceiverMap();
47 m_globalMessageReceivers.set(messageReceiverName, makeWeakPtr(messageReceiver));
48}
49
50void MessageReceiverMap::addMessageReceiver(ReceiverName messageReceiverName, uint64_t destinationID, MessageReceiver& messageReceiver)
51{
52 ASSERT(destinationID);
53 ASSERT(!m_messageReceivers.contains(std::make_pair(messageReceiverName, destinationID)));
54 ASSERT(!m_globalMessageReceivers.contains(messageReceiverName));
55
56 messageReceiver.willBeAddedToMessageReceiverMap();
57 m_messageReceivers.set(std::make_pair(messageReceiverName, destinationID), makeWeakPtr(messageReceiver));
58}
59
60void MessageReceiverMap::removeMessageReceiver(ReceiverName messageReceiverName)
61{
62 auto it = m_globalMessageReceivers.find(messageReceiverName);
63 if (it == m_globalMessageReceivers.end()) {
64 ASSERT_NOT_REACHED();
65 return;
66 }
67
68 if (it->value)
69 it->value->willBeRemovedFromMessageReceiverMap();
70 m_globalMessageReceivers.remove(it);
71}
72
73void MessageReceiverMap::removeMessageReceiver(ReceiverName messageReceiverName, uint64_t destinationID)
74{
75 auto it = m_messageReceivers.find(std::make_pair(messageReceiverName, destinationID));
76 if (it == m_messageReceivers.end()) {
77 ASSERT_NOT_REACHED();
78 return;
79 }
80
81 if (it->value)
82 it->value->willBeRemovedFromMessageReceiverMap();
83 m_messageReceivers.remove(it);
84}
85
86void MessageReceiverMap::removeMessageReceiver(MessageReceiver& messageReceiver)
87{
88 Vector<ReceiverName> globalReceiversToRemove;
89 for (auto& [name, receiver] : m_globalMessageReceivers) {
90 if (receiver == &messageReceiver)
91 globalReceiversToRemove.append(name);
92 }
93
94 for (auto& globalReceiverToRemove : globalReceiversToRemove)
95 removeMessageReceiver(globalReceiverToRemove);
96
97 Vector<std::pair<ReceiverName, uint64_t>> receiversToRemove;
98 for (auto& [nameAndDestinationID, receiver] : m_messageReceivers) {
99 if (receiver == &messageReceiver)
100 receiversToRemove.append(std::make_pair(nameAndDestinationID.first, nameAndDestinationID.second));
101 }
102
103 for (auto& [name, destinationID] : receiversToRemove)
104 removeMessageReceiver(name, destinationID);
105}
106
107void MessageReceiverMap::invalidate()
108{
109 for (auto& messageReceiver : m_globalMessageReceivers.values())
110 messageReceiver->willBeRemovedFromMessageReceiverMap();
111 m_globalMessageReceivers.clear();
112
113
114 for (auto& messageReceiver : m_messageReceivers.values())
115 messageReceiver->willBeRemovedFromMessageReceiverMap();
116 m_messageReceivers.clear();
117}

119bool MessageReceiverMap::dispatchMessage(Connection& connection, Decoder& decoder)
120{
121 if (auto messageReceiver = m_globalMessageReceivers.get(decoder.messageReceiverName())) {
122 ASSERT(!decoder.destinationID());
123
124 messageReceiver->didReceiveMessage(connection, decoder);
125 return true;
126 }
127
128 if (auto messageReceiver = m_messageReceivers.get(std::make_pair(decoder.messageReceiverName(), decoder.destinationID()))) {
129 messageReceiver->didReceiveMessage(connection, decoder);
130 return true;
131 }
132
133 return false;
134}
135
136bool MessageReceiverMap::dispatchSyncMessage(Connection& connection, Decoder& decoder, UniqueRef<Encoder>& replyEncoder)
137{
138 if (auto messageReceiver = m_globalMessageReceivers.get(decoder.messageReceiverName())) {
139 ASSERT(!decoder.destinationID());
140 return messageReceiver->didReceiveSyncMessage(connection, decoder, replyEncoder);
141 }
142
143 if (auto messageReceiver = m_messageReceivers.get(std::make_pair(decoder.messageReceiverName(), decoder.destinationID())))
144 return messageReceiver->didReceiveSyncMessage(connection, decoder, replyEncoder);
145
146 return false;
147}
