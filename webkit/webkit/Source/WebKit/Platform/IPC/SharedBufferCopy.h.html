26// Encodes a SharedBuffer that is received as a copy of the decoded data in a new SharedBuffer.
27// To avoid copying from the Decoder, use SharedBufferDataReference to receive a DataReference instead.
28
29#pragma once
30
31#include <WebCore/SharedBuffer.h>
32
33namespace IPC {
34
35class Decoder;
36class Encoder;
37
38class SharedBufferCopy {
39public:
40 SharedBufferCopy() = default;
41
42 SharedBufferCopy(RefPtr<WebCore::SharedBuffer>&& buffer)
43 : m_buffer(WTFMove(buffer)) { }
44 SharedBufferCopy(Ref<WebCore::SharedBuffer>&& buffer)
45 : m_buffer(WTFMove(buffer)) { }
46 SharedBufferCopy(const WebCore::SharedBuffer& buffer)
47 : m_buffer(WebCore::SharedBuffer::create())
48 {
49 m_buffer->append(buffer);
50 }
51
52 RefPtr<WebCore::SharedBuffer>& buffer() { return m_buffer; }
53 const RefPtr<WebCore::SharedBuffer>& buffer() const { return m_buffer; }
54
55 const uint8_t* data() const { return m_buffer ? m_buffer->data() : nullptr; }
56 size_t size() const { return m_buffer ? m_buffer->size() : 0; }
57 bool isEmpty() const { return m_buffer ? m_buffer->isEmpty() : true; }
58
59 void encode(Encoder&) const;
60 static WARN_UNUSED_RETURN std::optional<SharedBufferCopy> decode(Decoder&);
61
62private:
63 RefPtr<WebCore::SharedBuffer> m_buffer;
64};
65
66} // namespace IPC
