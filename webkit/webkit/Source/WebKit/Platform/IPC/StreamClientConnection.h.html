39// A message stream is a half-duplex two-way stream of messages to a between the client and the
40// server.
41//
42// StreamClientConnection can send messages and receive synchronous replies
43// through this message stream or through IPC::Connection.
44//
45// The server will receive messages in order _for the destination messages_.
46// The whole IPC::Connection message order is not preserved.
47//
48// The StreamClientConnection trusts the StreamServerConnection.
49class StreamClientConnection final {
50 WTF_MAKE_NONCOPYABLE(StreamClientConnection);
51public:
52 StreamClientConnection(Connection&, size_t bufferSize);
53
54 StreamConnectionBuffer& streamBuffer() { return m_buffer; }
55 void setWakeUpSemaphore(IPC::Semaphore&&);
56
57 template<typename T, typename U> bool send(T&& message, ObjectIdentifier<U> destinationID, Timeout);
58
59 using SendSyncResult = Connection::SendSyncResult;
60 template<typename T, typename U>
61 SendSyncResult sendSync(T&& message, typename T::Reply&&, ObjectIdentifier<U> destinationID, Timeout);
62
63private:
64 struct Span {
65 uint8_t* data;
66 size_t size;
67 };
68 static constexpr size_t minimumMessageSize = StreamConnectionEncoder::minimumMessageSize;
69 static constexpr size_t messageAlignment = StreamConnectionEncoder::messageAlignment;
70 template<typename T>
71 bool trySendStream(T& message, Span&);
72 template<typename T>
73 std::optional<SendSyncResult> trySendSyncStream(T& message, typename T::Reply&, Timeout, Span&);
74 bool trySendDestinationIDIfNeeded(uint64_t destinationID, Timeout);
75 void sendProcessOutOfStreamMessage(Span&&);
76
77 std::optional<Span> tryAcquire(Timeout);
78 std::optional<Span> tryAcquireAll(Timeout);
79
80 enum class WakeUpServer : bool {
81 No,
82 Yes
83 };
84 WakeUpServer release(size_t writeSize);
85 void wakeUpServer();
86
87 Span alignedSpan(size_t offset, size_t limit);
88 size_t size(size_t offset, size_t limit);
89
90 size_t wrapOffset(size_t offset) const { return m_buffer.wrapOffset(offset); }
91 size_t alignOffset(size_t offset) const { return m_buffer.alignOffset<messageAlignment>(offset, minimumMessageSize); }
92 using ClientOffset = StreamConnectionBuffer::ClientOffset;
93 Atomic<ClientOffset>& sharedClientOffset() { return m_buffer.clientOffset(); }
94 using ClientLimit = StreamConnectionBuffer::ServerOffset;
95 Atomic<ClientLimit>& sharedClientLimit() { return m_buffer.serverOffset(); }
96 size_t toLimit(ClientLimit) const;
97 uint8_t* data() const { return m_buffer.data(); }
98 size_t dataSize() const { return m_buffer.dataSize(); }
99
100 Connection& m_connection;
101 uint64_t m_currentDestinationID { 0 };
102
103 size_t m_clientOffset { 0 };
104 StreamConnectionBuffer m_buffer;
105 std::optional<Semaphore> m_wakeUpSemaphore;
106};

108template<typename T, typename U>
109bool StreamClientConnection::send(T&& message, ObjectIdentifier<U> destinationID, Timeout timeout)
110{
111 static_assert(!T::isSync, "Message is sync!");
112 if (!trySendDestinationIDIfNeeded(destinationID.toUInt64(), timeout))
113 return false;
114 auto span = tryAcquire(timeout);
115 if (!span)
116 return false;
117 if constexpr(T::isStreamEncodable) {
118 if (trySendStream(message, *span))
119 return true;
120 }
121 sendProcessOutOfStreamMessage(WTFMove(*span));
122 if (!m_connection.send(WTFMove(message), destinationID, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply))
123 return false;
124 return true;
125}

127template<typename T>
128bool StreamClientConnection::trySendStream(T& message, Span& span)
129{
130 StreamConnectionEncoder messageEncoder { T::name(), span.data, span.size };
131 if (messageEncoder << message.arguments()) {
132 auto wakeupResult = release(messageEncoder.size());
133 if (wakeupResult == StreamClientConnection::WakeUpServer::Yes)
134 wakeUpServer();
135 return true;
136 }
137 return false;
138}

140template<typename T, typename U>
141StreamClientConnection::SendSyncResult StreamClientConnection::sendSync(T&& message, typename T::Reply&& reply, ObjectIdentifier<U> destinationID, Timeout timeout)
142{
143 static_assert(T::isSync, "Message is not sync!");
144 if (!trySendDestinationIDIfNeeded(destinationID.toUInt64(), timeout))
145 return { };
146 auto span = tryAcquire(timeout);
147 if (!span)
148 return { };
149 if constexpr(T::isStreamEncodable) {
150 auto maybeSendResult = trySendSyncStream(message, reply, timeout, *span);
151 if (maybeSendResult)
152 return WTFMove(*maybeSendResult);
153 }
154 sendProcessOutOfStreamMessage(WTFMove(*span));
155 return m_connection.sendSync(WTFMove(message), WTFMove(reply), destinationID.toUInt64(), timeout);
156}

158template<typename T>
159std::optional<StreamClientConnection::SendSyncResult> StreamClientConnection::trySendSyncStream(T& message, typename T::Reply& reply, Timeout timeout, Span& span)
160{
161 // In this function, SendSyncResult { } means error happened and caller should stop processing.
162 // std::nullopt means we couldn't send through the stream, so try sending out of stream.
163 auto syncRequestID = m_connection.makeSyncRequestID();
164 if (!m_connection.pushPendingSyncRequestID(syncRequestID))
165 return SendSyncResult { };
166
167 auto result = [&]() -> std::optional<SendSyncResult> {
168 StreamConnectionEncoder messageEncoder { T::name(), span.data, span.size };
169 if (!(messageEncoder << syncRequestID << message.arguments()))
170 return std::nullopt;
171 auto wakeupResult = release(messageEncoder.size());
172
173 if (wakeupResult == StreamClientConnection::WakeUpServer::Yes)
174 wakeUpServer();
175 if constexpr(T::isReplyStreamEncodable) {
176 auto replySpan = tryAcquireAll(timeout);
177 if (!replySpan)
178 return SendSyncResult { };
179 auto decoder = std::unique_ptr<Decoder> { new Decoder(replySpan->data, replySpan->size, m_currentDestinationID) };
180 if (decoder->messageName() != MessageName::ProcessOutOfStreamMessage) {
181 ASSERT(decoder->messageName() == MessageName::SyncMessageReply);
182 return decoder;
183 }
184 } else
185 m_clientOffset = 0;
186 return m_connection.waitForSyncReply(syncRequestID, T::name(), timeout, { });
187 }();
188 m_connection.popPendingSyncRequestID(syncRequestID);
189 if (result && *result) {
190 auto& decoder = **result;
191 std::optional<typename T::ReplyArguments> replyArguments;
192 decoder >> replyArguments;
193 if (!replyArguments)
194 return SendSyncResult { };
195 moveTuple(WTFMove(*replyArguments), reply);
196 }
197 return result;
198}

200inline bool StreamClientConnection::trySendDestinationIDIfNeeded(uint64_t destinationID, Timeout timeout)
201{
202 if (destinationID == m_currentDestinationID)
203 return true;
204 auto span = tryAcquire(timeout);
205 if (!span)
206 return false;
207 StreamConnectionEncoder encoder { MessageName::SetStreamDestinationID, span->data, span->size };
208 if (!(encoder << destinationID)) {
209 ASSERT_NOT_REACHED(); // Size of the minimum allocation is incorrect. Likely an alignment issue.
210 return false;
211 }
212 auto wakeupResult = release(encoder.size());
213 if (wakeupResult == StreamClientConnection::WakeUpServer::Yes)
214 wakeUpServer();
215 m_currentDestinationID = destinationID;
216 return true;
217}

219inline void StreamClientConnection::sendProcessOutOfStreamMessage(Span&& span)
220{
221 StreamConnectionEncoder encoder { MessageName::ProcessOutOfStreamMessage, span.data, span.size };
222 // Not notifying on wake up since the out-of-stream message will do that.
223 auto result = release(encoder.size());
224 UNUSED_VARIABLE(result);
225}
226inline std::optional<StreamClientConnection::Span> StreamClientConnection::tryAcquire(Timeout timeout)
227{
228 ClientLimit clientLimit = sharedClientLimit().load(std::memory_order_acquire);
229 // This would mean we try to send messages after a timeout. It is a programming error.
230 // Since the value is trusted, we only assert.
231 ASSERT(clientLimit != ClientLimit::clientIsWaitingTag);
232
233 for (;;) {
234 if (clientLimit != ClientLimit::clientIsWaitingTag) {
235 auto result = alignedSpan(m_clientOffset, toLimit(clientLimit));
236 if (result.size >= minimumMessageSize)
237 return result;
238 }
239 if (timeout.didTimeOut())
240 break;
241 ClientLimit oldClientLimit = sharedClientLimit().compareExchangeStrong(clientLimit, ClientLimit::clientIsWaitingTag, std::memory_order_acq_rel, std::memory_order_acq_rel);
242 if (clientLimit == oldClientLimit) {
243 m_buffer.clientWaitSemaphore().waitFor(timeout);
244 clientLimit = sharedClientLimit().load(std::memory_order_acquire);
245 } else
246 clientLimit = oldClientLimit;
247 // The alignedSpan uses the minimumMessageSize to calculate the next beginning position in the buffer,
248 // and not the size. The size might be more or less what is needed, depending on where the reader is.
249 // If there is no capacity for minimum message size, wait until more is available.
250 // In the case where clientOffset < clientLimit we can arrive to a situation where
251 // 0 < result.size < minimumMessageSize.
252 }
253 return std::nullopt;
254}

256inline std::optional<StreamClientConnection::Span> StreamClientConnection::tryAcquireAll(Timeout timeout)
257{
258 // This would mean we try to send messages after a timeout. It is a programming error.
259 // Since the value is trusted, we only assert.
260 ASSERT(sharedClientLimit().load(std::memory_order_acquire) != ClientLimit::clientIsWaitingTag);
261
262 // The server acknowledges that sync message has been processed by setting clientOffset == clientLimit == 0.
263 // Wait for this condition, or then the condition where server says that it started to sleep after setting that condition.
264 // The wait sequence involves two variables, so form a transaction by setting clientLimit == clientIsWaitingTag.
265 // The transaction is cancelled if the server has already set clientOffset == clientLimit == 0, otherwise it commits.
266 // If the transaction commits, server is guaranteed to signal.
267
268 for (;;) {
269 ClientLimit clientLimit = sharedClientLimit().exchange(ClientLimit::clientIsWaitingTag, std::memory_order_acq_rel);
270 ClientOffset clientOffset = sharedClientOffset().load(std::memory_order_acquire);
271 if (!clientLimit && (clientOffset == ClientOffset::serverIsSleepingTag || !clientOffset))
272 break;
273
274 m_buffer.clientWaitSemaphore().waitFor(timeout);
275 if (timeout.didTimeOut())
276 return std::nullopt;
277 }
278 // In case the transaction was cancelled, undo the transaction marker.
279 sharedClientLimit().store(static_cast<ClientLimit>(0), std::memory_order_release);
280 m_clientOffset = 0;
281 return alignedSpan(m_clientOffset, 0);
282}

284inline StreamClientConnection::WakeUpServer StreamClientConnection::release(size_t size)
285{
286 size = std::max(size, minimumMessageSize);
287 m_clientOffset = wrapOffset(alignOffset(m_clientOffset) + size);
288 ASSERT(m_clientOffset < dataSize());
289 // If the server wrote over the clientOffset with serverIsSleepingTag, we know it is sleeping.
290 ClientOffset oldClientOffset = sharedClientOffset().exchange(static_cast<ClientOffset>(m_clientOffset), std::memory_order_acq_rel);
291 if (oldClientOffset == ClientOffset::serverIsSleepingTag)
292 return WakeUpServer::Yes;
293 ASSERT(!(oldClientOffset & ClientOffset::serverIsSleepingTag));
294 return WakeUpServer::No;
295}

297inline StreamClientConnection::Span StreamClientConnection::alignedSpan(size_t offset, size_t limit)
298{
299 ASSERT(offset < dataSize());
300 ASSERT(limit < dataSize());
301 size_t aligned = alignOffset(offset);
302 size_t resultSize = 0;
303 if (offset < limit) {
304 if (aligned >= offset && aligned < limit)
305 resultSize = size(aligned, limit);
306 } else {
307 if (aligned >= offset || aligned < limit)
308 resultSize = size(aligned, limit);
309 }
310 return { data() + aligned, resultSize };
311}
312
313inline size_t StreamClientConnection::size(size_t offset, size_t limit)
314{
315 if (!limit)
316 return dataSize() - 1 - offset;
317 if (limit <= offset)
318 return dataSize() - offset;
319 return limit - offset - 1;
320}
321
322inline size_t StreamClientConnection::toLimit(ClientLimit clientLimit) const
323{
324 ASSERT(!(clientLimit & ClientLimit::clientIsWaitingTag));
325 ASSERT(static_cast<size_t>(clientLimit) <= dataSize() - 1);
326 return static_cast<size_t>(clientLimit);
327}
