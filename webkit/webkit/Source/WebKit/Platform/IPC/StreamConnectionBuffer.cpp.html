33static Ref<WebKit::SharedMemory> createMemory(size_t size)
34{
35 auto memory = WebKit::SharedMemory::allocate(size);
36 if (!memory)
37 CRASH();
38 return memory.releaseNonNull();
39}
40
41StreamConnectionBuffer::StreamConnectionBuffer(size_t memorySize)
42 : m_dataSize(memorySize - headerSize())
43 , m_sharedMemory(createMemory(memorySize))
44{
45 ASSERT(m_dataSize <= maximumSize());
46}
47
48StreamConnectionBuffer::StreamConnectionBuffer(Ref<WebKit::SharedMemory>&& memory, size_t memorySize, Semaphore&& clientWaitSemaphore)
49 : m_dataSize(memorySize - headerSize())
50 , m_sharedMemory(WTFMove(memory))
51 , m_clientWaitSemaphore(WTFMove(clientWaitSemaphore))
52{
53 ASSERT(m_dataSize <= maximumSize());
54}
55
56StreamConnectionBuffer::StreamConnectionBuffer(StreamConnectionBuffer&& other) = default;
57
58StreamConnectionBuffer::~StreamConnectionBuffer() = default;
59
60StreamConnectionBuffer& StreamConnectionBuffer::operator=(StreamConnectionBuffer&& other)
61{
62 if (this != &other) {
63 m_dataSize = other.m_dataSize;
64 m_sharedMemory = WTFMove(other.m_sharedMemory);
65 m_clientWaitSemaphore = WTFMove(other.m_clientWaitSemaphore);
66 }
67 return *this;
68}
69
70void StreamConnectionBuffer::encode(Encoder& encoder) const
71{
72 WebKit::SharedMemory::Handle handle;
73 if (!m_sharedMemory->createHandle(handle, WebKit::SharedMemory::Protection::ReadWrite))
74 CRASH();
75 WebKit::SharedMemory::IPCHandle ipcHandle { WTFMove(handle), m_sharedMemory->size() };
76 encoder << ipcHandle;
77 encoder << m_clientWaitSemaphore;
78}
79
80std::optional<StreamConnectionBuffer> StreamConnectionBuffer::decode(Decoder& decoder)
81{
82 std::optional<WebKit::SharedMemory::IPCHandle> ipcHandle;
83 decoder >> ipcHandle;
84 if (!ipcHandle)
85 return std::nullopt;
86 std::optional<Semaphore> semaphore;
87 decoder >> semaphore;
88 if (!semaphore)
89 return std::nullopt;
90 size_t dataSize = static_cast<size_t>(ipcHandle->dataSize);
91 if (dataSize < headerSize())
92 return std::nullopt;
93 if (dataSize > headerSize() + maximumSize())
94 return std::nullopt;
95 auto sharedMemory = WebKit::SharedMemory::map(ipcHandle->handle, WebKit::SharedMemory::Protection::ReadWrite);
96 if (sharedMemory->size() < dataSize)
97 return std::nullopt;
98 return StreamConnectionBuffer { sharedMemory.releaseNonNull(), dataSize, WTFMove(*semaphore) };
99}
