35// StreamConnectionBuffer is a shared "bi-partite" circular buffer supporting variable length messages, specific data
36// alignment with mandated minimum size. StreamClientConnection and StreamServerConnection use StreamConnectionBuffer to
37// communicate.
38//
39// The "bi-partite" is here to mean that writes and reads to the buffer must be continuous.
40//
41// The caller (client or server) "acquires" some amount of buffer data. The data pointer location depends on the amount
42// of data being acquired. The buffer returns all the available capacity -- this may be less than acquired.
43//
44// The caller "releases" the some amount of data, at least the amount acquired but potentially more.
45//
46// Both the client and the server must release the same amount of data at the same step of the acquire/release sequence.
47// Suppose the client releases first 8 and then 166 bytes. Upon the first acquire, the server must also release 8 bytes
48// and then upon the second acquire 166 bytes. Due to how alignment and minimum length affect the position at which the
49// memory can be referred to, the server cannot "read two times simultaneously" and release 174 bytes.
50//
51// The buffer also supports synchronous replies via "releaseAll/acquireAll" call pattern. The communication protocol can
52// establish cases where the client transfers ownership of the data to the server. In these cases the server will
53// acknowledge the read with "releaseAll" operation. The client will then read the reply via "acquireAll" operation. The
54// reply will be written to the beginning of the buffer. This mandates that in-place data references in the buffer
55// cannot be used simultaneously for message data and the reply data. In current IPC implementation, the message
56// processing uses the in-place data references, while the reply data is first constructed to allocated memory and then
57// copied to the message buffer.
58//
59// The circular buffer has following implementation:
60// * The client owns the data between [clientOffset, serverOffset[. When clientOffset == serverOffset, the client owns
61// all the data.
62// * The server owns the data between [serverOffset, clientOffset[.
63// * The exception to the above is when communication protocol can contain messages that denote that the server will own
64// all the data.
65// * The buffer can hold maximum of size - 1 values. The last value is reserved for indicating that the buffer is full.
66// FIXME: Maybe would be simpler implementation if it would use the "wrap" flag instead of the hole as the indicator.
67// This would move the alignedSpan implementation to the StreamConnectionBuffer.
68// * All atomic variable loads are untrusted, so they're clamped. Violations are not reported, though.
69// See SharedDisplayListHandle.
70class StreamConnectionBuffer {
71public:
72 explicit StreamConnectionBuffer(size_t memorySize);
73 StreamConnectionBuffer(StreamConnectionBuffer&&);
74 ~StreamConnectionBuffer();
75 StreamConnectionBuffer& operator=(StreamConnectionBuffer&&);
76
77 size_t wrapOffset(size_t offset) const
78 {
79 ASSERT(offset <= dataSize());
80 if (offset == dataSize())
81 return 0;
82 return offset;
83 }
84
85 template<size_t messageAlignment>
86 size_t alignOffset(size_t offset, size_t acquireSize) const
87 {
88 offset = roundUpToMultipleOf<messageAlignment>(offset);
89 if (offset + acquireSize >= dataSize())
90 offset = 0;
91 return offset;
92 }
93
94 // Value denoting the client index to the buffer, with special tag values.
95 // The client trusts this. The server converts the value to trusted size_t offset with a special function.
96 enum ClientOffset : size_t {
97 // Tag value stored in when the server is sleeping, e.g. not running.
98 serverIsSleepingTag = 1u << 31
99 };
100
101 // Value denoting the server index to the buffer, with special tag values.
102 // The client trusts this. The server converts the value to trusted size_t offset with a special function.
103 enum ServerOffset : size_t {
104 // Tag value stored when the client is waiting, e.g. waiting on a semaphore.
105 clientIsWaitingTag = 1u << 31
106 };
107
108 Atomic<ClientOffset>& clientOffset() { return header().clientOffset; }
109 Atomic<ServerOffset>& serverOffset() { return header().serverOffset; }
110 uint8_t* data() const { return static_cast<uint8_t*>(m_sharedMemory->data()) + headerSize(); }
111 size_t dataSize() const { return m_dataSize; }
112 Semaphore& clientWaitSemaphore() { return m_clientWaitSemaphore; }
113
114 static constexpr size_t maximumSize() { return std::min(static_cast<size_t>(ClientOffset::serverIsSleepingTag), static_cast<size_t>(ClientOffset::serverIsSleepingTag)) - 1; }
115 void encode(Encoder&) const;
116 static std::optional<StreamConnectionBuffer> decode(Decoder&);
117
118private:
119 StreamConnectionBuffer(Ref<WebKit::SharedMemory>&&, size_t memorySize, Semaphore&& clientWaitSemaphore);
120
121 struct Header {
122 Atomic<ServerOffset> serverOffset;
123 // Padding so that the variables mostly accessed by different processes do not share a cache line.
124 // This is an attempt to avoid cache-line induced reduction of parallel access.
125 alignas(sizeof(uint64_t[2])) Atomic<ClientOffset> clientOffset;
126 };
127 Header& header() const { return *reinterpret_cast<Header*>(m_sharedMemory->data()); }
128 static constexpr size_t headerSize() { return roundUpToMultipleOf<alignof(std::max_align_t)>(sizeof(Header)); }
129
130 size_t m_dataSize { 0 };
131 Ref<WebKit::SharedMemory> m_sharedMemory;
132 Semaphore m_clientWaitSemaphore;
133};
