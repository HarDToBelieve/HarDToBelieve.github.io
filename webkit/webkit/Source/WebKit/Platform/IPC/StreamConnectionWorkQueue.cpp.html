31StreamConnectionWorkQueue::StreamConnectionWorkQueue(const char* name)
32 : m_name(name)
33{
34}
35
36void StreamConnectionWorkQueue::dispatch(WTF::Function<void()>&& function)
37{
38 {
39 Locker locker { m_lock };
40 m_functions.append(WTFMove(function));
41 ASSERT(!m_shouldQuit); // Re-entering during shutdown not supported.
42 }
43 wakeUpProcessingThread();
44
45}
46void StreamConnectionWorkQueue::addStreamConnection(StreamServerConnectionBase& connection)
47{
48 {
49 Locker locker { m_lock };
50 m_connections.add(connection);
51 ASSERT(!m_shouldQuit); // Re-entering during shutdown not supported.
52 }
53 wakeUpProcessingThread();
54}
55
56void StreamConnectionWorkQueue::removeStreamConnection(StreamServerConnectionBase& connection)
57{
58 ASSERT(m_processingThread);
59 {
60 Locker locker { m_lock };
61 m_connections.remove(connection);
62 ASSERT(!m_shouldQuit); // Re-entering during shutdown not supported.
63 }
64 m_wakeUpSemaphore.signal();
65}
66
67void StreamConnectionWorkQueue::stop()
68{
69 m_shouldQuit = true;
70 if (!m_processingThread)
71 return;
72 m_wakeUpSemaphore.signal();
73 m_processingThread->waitForCompletion();
74 m_processingThread = nullptr;
75}
76
77void StreamConnectionWorkQueue::wakeUp()
78{
79 m_wakeUpSemaphore.signal();
80}
81
82IPC::Semaphore& StreamConnectionWorkQueue::wakeUpSemaphore()
83{
84 return m_wakeUpSemaphore;
85}
86
87void StreamConnectionWorkQueue::wakeUpProcessingThread()
88{
89 if (m_processingThread) {
90 m_wakeUpSemaphore.signal();
91 return;
92 }
93
94 auto task = [this]() mutable {
95 for (;;) {
96 processStreams();
97 if (m_shouldQuit) {
98 processStreams();
99 return;
100 }
101 m_wakeUpSemaphore.wait();
102 }
103 };
104 m_processingThread = Thread::create(m_name, WTFMove(task), ThreadType::Graphics, Thread::QOS::UserInteractive);
105}
106
107void StreamConnectionWorkQueue::processStreams()
108{
109 constexpr size_t defaultMessageLimit = 1000;
110 bool hasMoreToProcess = false;
111 do {
112 Deque<WTF::Function<void()>> functions;
113 HashSet<Ref<StreamServerConnectionBase>> connections;
114 {
115 Locker locker { m_lock };
116 functions.swap(m_functions);
117 connections = m_connections;
118 }
119 for (auto& function : functions)
120 WTFMove(function)();
121
122 hasMoreToProcess = false;
123 for (auto& connection : connections)
124 hasMoreToProcess |= connection->dispatchStreamMessages(defaultMessageLimit) == StreamServerConnectionBase::HasMoreMessages;
125 } while (hasMoreToProcess);
126}
