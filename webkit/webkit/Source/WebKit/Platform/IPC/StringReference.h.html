26#ifndef StringReference_h
27#define StringReference_h
28
29#include <string.h>
30#include <wtf/Forward.h>
31#include <wtf/HashTraits.h>
32
33namespace IPC {
34
35class Encoder;
36class Decoder;
37
38class StringReference {
39public:
40 StringReference()
41 : m_data(0)
42 , m_size(0)
43 {
44 }
45
46 StringReference(const char* data, size_t size)
47 : m_data(data)
48 , m_size(size)
49 {
50 }
51
52 template<size_t length>
53 StringReference(const char (&string)[length])
54 : m_data(string)
55 , m_size(length - 1)
56 {
57 }
58
59 bool isEmpty() const { return !m_size; }
60
61 size_t size() const { return m_size; }
62 const char* data() const { return m_data; }
63
64 CString toString() const;
65
66 void encode(Encoder&) const;
67 static WARN_UNUSED_RETURN bool decode(Decoder&, StringReference&);
68
69 struct Hash {
70 static unsigned hash(const StringReference& a);
71 static bool equal(const StringReference&, const StringReference&);
72 static const bool safeToCompareToEmptyOrDeleted = true;
73 };
74
75private:
76 friend struct HashTraits<IPC::StringReference>;
77
78 const char* m_data;
79 size_t m_size;
80};
81
82inline bool operator==(const StringReference& a, const StringReference& b)
83{
84 return a.size() == b.size() && !memcmp(a.data(), b.data(), a.size());
85}
86
87inline bool operator!=(const StringReference& a, const StringReference& b)
88{
89 return !(a == b);
90}
91
92inline bool StringReference::Hash::equal(const StringReference& a, const StringReference& b)
93{
94 return a == b;
95}
96
97} // namespace IPC
98
99namespace WTF {
100template<typename T> struct DefaultHash;
101
102template<> struct DefaultHash<IPC::StringReference> : IPC::StringReference::Hash { };
103
104template<> struct HashTraits<IPC::StringReference> : GenericHashTraits<IPC::StringReference> {
105 static const bool emptyValueIsZero = 0;
106 static void constructDeletedValue(IPC::StringReference& stringReference) { stringReference.m_size = std::numeric_limits<size_t>::max(); }
107 static bool isDeletedValue(const IPC::StringReference& stringReference) { return stringReference.size() == std::numeric_limits<size_t>::max(); }
108};
109
110} // namespace WTF

112#endif // StringReference_h
