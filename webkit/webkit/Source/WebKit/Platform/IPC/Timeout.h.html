26#pragma once
27
28#include <algorithm>
29#include <wtf/MonotonicTime.h>
30#include <wtf/TimeWithDynamicClockType.h>
31
32namespace IPC {
33
34class Timeout {
35public:
36 Timeout(Seconds timeDelta)
37 : m_deadline(MonotonicTime::now() + timeDelta)
38 {
39 }
40
41 static constexpr Timeout infinity() { return Timeout { }; }
42 bool isInfinity() const { return std::isinf(m_deadline); }
43 static Timeout now() { return 0_s; }
44 Seconds secondsUntilDeadline() const { return std::max(m_deadline - MonotonicTime::now(), 0_s ); }
45 constexpr MonotonicTime deadline() const { return m_deadline; }
46 bool didTimeOut() const { return MonotonicTime::now() >= m_deadline; }
47
48private:
49 explicit constexpr Timeout()
50 : m_deadline(MonotonicTime::infinity())
51 {
52 }
53
54 MonotonicTime m_deadline;
55};
56
57}
