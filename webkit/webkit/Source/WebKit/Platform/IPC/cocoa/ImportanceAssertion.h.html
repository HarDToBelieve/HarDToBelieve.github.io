26#pragma once
27
28#if PLATFORM(MAC)
29
30#include <mach/message.h>
31
32namespace IPC {
33
34class ImportanceAssertion {
35 WTF_MAKE_FAST_ALLOCATED;
36 WTF_MAKE_NONCOPYABLE(ImportanceAssertion);
37
38public:
39 explicit ImportanceAssertion(mach_msg_header_t* header)
40 : m_voucher(0)
41 {
42 if (MACH_MSGH_BITS_HAS_VOUCHER(header->msgh_bits)) {
43 m_voucher = header->msgh_voucher_port;
44 header->msgh_voucher_port = MACH_VOUCHER_NULL;
45 header->msgh_bits &= ~(MACH_MSGH_BITS_VOUCHER_MASK | MACH_MSGH_BITS_RAISEIMP);
46 }
47 }
48
49 ~ImportanceAssertion()
50 {
51 if (m_voucher) {
52 kern_return_t kr = mach_voucher_deallocate(m_voucher);
53 ASSERT_UNUSED(kr, !kr);
54 }
55 }
56
57private:
58 mach_voucher_t m_voucher;
59};
60
61}
62
63#endif // PLATFORM(MAC)
