35// Version of round_msg() using CheckedSize for extra safety.
36static inline CheckedSize safeRoundMsg(CheckedSize value)
37{
38 constexpr size_t alignment = sizeof(natural_t);
39 return ((value + (alignment - 1)) / alignment) * alignment;
40}
41
42std::unique_ptr<MachMessage> MachMessage::create(MessageName messageName, size_t size)
43{
44 auto bufferSize = CheckedSize(sizeof(MachMessage)) + size;
45 if (bufferSize.hasOverflowed())
46 return nullptr;
47 void* memory = WTF::fastZeroedMalloc(bufferSize);
48 return std::unique_ptr<MachMessage> { new (NotNull, memory) MachMessage { messageName, size } };
49}

51MachMessage::MachMessage(MessageName messageName, size_t size)
52 : m_messageName { messageName }
53 , m_size { size }
54{
55}
56
57MachMessage::~MachMessage()
58{
59 if (m_shouldFreeDescriptors)
60 ::mach_msg_destroy(header());
61}
62
63CheckedSize MachMessage::messageSize(size_t bodySize, size_t portDescriptorCount, size_t memoryDescriptorCount)
64{
65 CheckedSize messageSize = sizeof(mach_msg_header_t);
66 messageSize += bodySize;
67
68 if (portDescriptorCount || memoryDescriptorCount) {
69 messageSize += sizeof(mach_msg_body_t);
70 messageSize += (portDescriptorCount * sizeof(mach_msg_port_descriptor_t));
71 messageSize += (memoryDescriptorCount * sizeof(mach_msg_ool_descriptor_t));
72 }
73
74 return safeRoundMsg(messageSize);
75}

77void MachMessage::leakDescriptors()

79 m_shouldFreeDescriptors = false;
