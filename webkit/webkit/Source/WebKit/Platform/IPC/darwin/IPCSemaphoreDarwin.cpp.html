34Semaphore::Semaphore()
35{
36 auto ret = semaphore_create(mach_task_self(), &m_semaphore, SYNC_POLICY_FIFO, 0);
37 ASSERT_UNUSED(ret, ret == KERN_SUCCESS);
38}
39
40Semaphore::Semaphore(Semaphore&& other)
41 : m_sendRight(WTFMove(other.m_sendRight))
42 , m_semaphore(std::exchange(other.m_semaphore, SEMAPHORE_NULL))
43{
44}
45
46Semaphore::Semaphore(MachSendRight&& sendRight)
47 : m_sendRight(WTFMove(sendRight))
48 , m_semaphore(m_sendRight.sendRight())
49{
50 ASSERT(m_sendRight);
51}
52
53Semaphore::~Semaphore()
54{
55 destroy();
56}
57
58Semaphore& Semaphore::operator=(Semaphore&& other)
59{
60 if (this != &other) {
61 destroy();
62 m_sendRight = WTFMove(other.m_sendRight);
63 m_semaphore = std::exchange(other.m_semaphore, SEMAPHORE_NULL);
64 }
65
66 return *this;
67}
68
69void Semaphore::signal()
70{
71 auto ret = semaphore_signal(m_semaphore);
72 ASSERT_UNUSED(ret, ret == KERN_SUCCESS || ret == KERN_TERMINATED);
73}
74
75bool Semaphore::wait()
76{
77 auto ret = semaphore_wait(m_semaphore);
78 ASSERT(ret == KERN_SUCCESS || ret == KERN_TERMINATED);
79 return ret == KERN_SUCCESS;
80}
81
82bool Semaphore::waitFor(Timeout timeout)
83{
84 Seconds waitTime = timeout.secondsUntilDeadline();
85 auto seconds = waitTime.secondsAs<unsigned>();
86 auto ret = semaphore_timedwait(m_semaphore, { seconds, static_cast<clock_res_t>(waitTime.nanosecondsAs<uint64_t>() - seconds * NSEC_PER_SEC) });
87 ASSERT(ret == KERN_SUCCESS || ret == KERN_OPERATION_TIMED_OUT || ret == KERN_TERMINATED);
88 return ret == KERN_SUCCESS;
89}
90
91MachSendRight Semaphore::createSendRight() const
92{
93 return MachSendRight::create(m_semaphore);
94}
95
96void Semaphore::encode(Encoder& encoder) const
97{
98 encoder << createSendRight();
99}
100
101std::optional<Semaphore> Semaphore::decode(Decoder& decoder)
102{
103 MachSendRight sendRight;
104 if (!decoder.decode(sendRight))
105 return std::nullopt;
106 return std::optional<Semaphore> { std::in_place, WTFMove(sendRight) };
107}
108
109
110void Semaphore::destroy()
111{
112 if (m_sendRight) {
113 m_sendRight = MachSendRight();
114 m_semaphore = SEMAPHORE_NULL;
115 return;
116 }
117 if (m_semaphore == SEMAPHORE_NULL)
118 return;
119 auto ret = semaphore_destroy(mach_task_self(), m_semaphore);
120 ASSERT_UNUSED(ret, ret == KERN_SUCCESS);
121 m_semaphore = SEMAPHORE_NULL;
122}
