35RefPtr<SharedMemory> SharedMemory::copyBuffer(const SharedBuffer& buffer)
36{
37 if (buffer.isEmpty())
38 return nullptr;
39
40 auto sharedMemory = allocate(buffer.size());
41 if (!sharedMemory)
42 return nullptr;
43
44 auto sharedMemoryPtr = static_cast<char*>(sharedMemory->data());
45 for (auto& segmentEntry : buffer)
46 memcpy(sharedMemoryPtr + segmentEntry.beginPosition, segmentEntry.segment->data(), segmentEntry.segment->size());
47
48 return sharedMemory;
49}
