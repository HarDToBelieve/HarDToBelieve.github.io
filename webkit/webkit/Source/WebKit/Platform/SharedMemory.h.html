58enum class MemoryLedger { None, Default, Network, Media, Graphics, Neural };
59
60class SharedMemory : public ThreadSafeRefCounted<SharedMemory> {
61public:
62 enum class Protection {
63 ReadOnly,
64 ReadWrite
65 };
66
67 class Handle {
68 WTF_MAKE_NONCOPYABLE(Handle);
69 public:
70 Handle();
71 ~Handle();
72 Handle(Handle&&);
73 Handle& operator=(Handle&&);
74
75 bool isNull() const;
76
77#if OS(DARWIN) || OS(WINDOWS)
78 size_t size() const { return m_size; }
79#endif
80
81 // Take ownership of the memory for jetsam purposes.
82 void takeOwnershipOfMemory(MemoryLedger) const;
83
84 void clear();
85
86#if USE(UNIX_DOMAIN_SOCKETS)
87 IPC::Attachment releaseAttachment() const;
88 void adoptAttachment(IPC::Attachment&&);
89#endif
90#if OS(WINDOWS)
91 static void encodeHandle(IPC::Encoder&, HANDLE);
92 static std::optional<HANDLE> decodeHandle(IPC::Decoder&);
93#endif
94 private:
95 friend class SharedMemory;
96#if USE(UNIX_DOMAIN_SOCKETS)
97 mutable IPC::Attachment m_attachment;
98#elif OS(DARWIN)
99 mutable mach_port_t m_port { MACH_PORT_NULL };
100 size_t m_size;
101#elif OS(WINDOWS)
102 mutable HANDLE m_handle;
103 size_t m_size;
104#endif
105 };
106
107 struct IPCHandle {
108 IPCHandle() = default;
109 IPCHandle(Handle&& handle, uint64_t dataSize)
110 : handle(WTFMove(handle))
111 , dataSize(dataSize)
112 {
113 }
114 void encode(IPC::Encoder&) const;
115 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, IPCHandle&);
116
117 Handle handle;
118 uint64_t dataSize { 0 };
119 };
120
121 // FIXME: Change these factory functions to return Ref<SharedMemory> and crash on failure.
122 static RefPtr<SharedMemory> allocate(size_t);
123 static RefPtr<SharedMemory> copyBuffer(const WebCore::SharedBuffer&);
124 static RefPtr<SharedMemory> map(const Handle&, Protection);
125#if USE(UNIX_DOMAIN_SOCKETS)
126 static RefPtr<SharedMemory> wrapMap(void*, size_t, int fileDescriptor);
127#elif OS(DARWIN)
128 static RefPtr<SharedMemory> wrapMap(void*, size_t, Protection);
129#endif
130#if OS(WINDOWS)
131 static RefPtr<SharedMemory> adopt(HANDLE, size_t, Protection);
132#endif
133
134 ~SharedMemory();
135
136 bool createHandle(Handle&, Protection);
137
138 size_t size() const { return m_size; }
139 void* data() const
140 {
141 ASSERT(m_data);
142 return m_data;
143 }
144
145#if OS(WINDOWS)
146 HANDLE handle() const { return m_handle; }
147#endif
148
149#if PLATFORM(COCOA)
150 Protection protection() const { return m_protection; }
151#endif
152
153 // Return the system page size in bytes.
154 static unsigned systemPageSize();
155
156private:
157#if OS(DARWIN)
158 WTF::MachSendRight createSendRight(Protection) const;
159#endif
160
161 size_t m_size;
162 void* m_data;
163#if PLATFORM(COCOA)
164 Protection m_protection;
165#endif
166
167#if USE(UNIX_DOMAIN_SOCKETS)
168 std::optional<int> m_fileDescriptor;
169 bool m_isWrappingMap { false };
170#elif OS(DARWIN)
171 mach_port_t m_port { MACH_PORT_NULL };
172#elif OS(WINDOWS)
173 HANDLE m_handle;
174#endif
175};
176
177};
