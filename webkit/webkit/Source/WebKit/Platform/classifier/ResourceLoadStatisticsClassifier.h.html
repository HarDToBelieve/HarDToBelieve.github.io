28namespace WebCore {
29struct ResourceLoadStatistics;
30}
31
32namespace WebKit {
33
34enum ResourceLoadPrevalence {
35 Low = 1 << 0,
36 High = 1 << 1,
37 VeryHigh = 1 << 2,
38};
39
40class ResourceLoadStatisticsClassifier {
41public:
42 ResourceLoadStatisticsClassifier() = default;
43 virtual ~ResourceLoadStatisticsClassifier() = default;
44 ResourceLoadPrevalence calculateResourcePrevalence(const WebCore::ResourceLoadStatistics& resourceStatistic, ResourceLoadPrevalence currentPrevalence);
45 ResourceLoadPrevalence calculateResourcePrevalence(unsigned subresourceUnderTopFrameDomainsCount, unsigned subresourceUniqueRedirectsToCount, unsigned subframeUnderTopFrameOriginsCount, unsigned topFrameUniqueRedirectsToCount, ResourceLoadPrevalence currentPrevalence);
46protected:
47 virtual bool classify(unsigned subresourceUnderTopFrameDomainsCount, unsigned subresourceUniqueRedirectsToCount, unsigned subframeUnderTopFrameOriginsCount)
48 {
49 return classifyWithVectorThreshold(subresourceUnderTopFrameDomainsCount, subresourceUniqueRedirectsToCount, subframeUnderTopFrameOriginsCount);
50 }
51 bool classifyWithVectorThreshold(unsigned, unsigned, unsigned);
52};
