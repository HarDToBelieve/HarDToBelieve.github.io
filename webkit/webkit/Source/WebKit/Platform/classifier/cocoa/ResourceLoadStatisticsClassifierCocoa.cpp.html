40bool ResourceLoadStatisticsClassifierCocoa::classify(unsigned subresourceUnderTopFrameDomainsCount, unsigned subresourceUniqueRedirectsToCount, unsigned subframeUnderTopFrameOriginsCount)
41{
42 if (!canUseCorePrediction())
43 return classifyWithVectorThreshold(subresourceUnderTopFrameDomainsCount, subresourceUniqueRedirectsToCount, subframeUnderTopFrameOriginsCount);
44
45 Vector<svm_node> features;
46
47 if (subresourceUnderTopFrameDomainsCount)
48 features.append({1, static_cast<double>(subresourceUnderTopFrameDomainsCount)});
49 if (subresourceUniqueRedirectsToCount)
50 features.append({2, static_cast<double>(subresourceUniqueRedirectsToCount)});
51 if (subframeUnderTopFrameOriginsCount)
52 features.append({3, static_cast<double>(subframeUnderTopFrameOriginsCount)});
53
54 // Add termination node with index -1.
55 features.append({-1, -1});
56
57 double score;
58 int classification = svm_predict_values(singletonPredictionModel(), features.data(), &score);
59 LOG(ResourceLoadStatistics, "ResourceLoadStatisticsClassifierCocoa::classify(): Classified with CorePrediction.");
60 return classification < 0;
61}

63String ResourceLoadStatisticsClassifierCocoa::storagePath()
64{
65 CFBundleRef webKitBundle = CFBundleGetBundleWithIdentifier(CFSTR("com.apple.WebKit"));
66 RetainPtr<CFURLRef> resourceUrl = adoptCF(CFBundleCopyResourcesDirectoryURL(webKitBundle));
67 resourceUrl = adoptCF(CFURLCreateCopyAppendingPathComponent(nullptr, resourceUrl.get(), CFSTR("corePrediction_model"), false));
68 CFErrorRef error = nullptr;
69 resourceUrl = adoptCF(CFURLCreateFilePathURL(nullptr, resourceUrl.get(), &error));
70
71 if (error || !resourceUrl)
72 return String();
73
74 RetainPtr<CFStringRef> resourceUrlString = adoptCF(CFURLCopyFileSystemPath(resourceUrl.get(), kCFURLPOSIXPathStyle));
75 return String(resourceUrlString.get());
76}

78bool ResourceLoadStatisticsClassifierCocoa::canUseCorePrediction()
79{
80 if (m_haveLoadedModel)
81 return true;
82
83 if (!m_useCorePrediction)
84 return false;
85
86 if (svm_load_model) {
87 m_useCorePrediction = false;
88 return false;
89 }
90
91 String storagePathStr = storagePath();
92 if (storagePathStr.isNull() || storagePathStr.isEmpty()) {
93 m_useCorePrediction = false;
94 return false;
95 }
96
97 if (singletonPredictionModel()) {
98 m_haveLoadedModel = true;
99 return true;
100 }
101
102 m_useCorePrediction = false;
103 return false;
104}
105
106const struct svm_model* ResourceLoadStatisticsClassifierCocoa::singletonPredictionModel()
107{
108 static std::optional<struct svm_model*> corePredictionModel;
109 static dispatch_once_t onceToken;
110 dispatch_once(&onceToken, ^{
111 auto path = storagePath();
112 if (path.isEmpty())
113 return;
114
115 corePredictionModel = svm_load_model(path.utf8().data());
116 });
117
118 if (corePredictionModel && corePredictionModel.value())
119 return corePredictionModel.value();
120
121 WTFLogAlways("ResourceLoadStatisticsClassifierCocoa::singletonPredictionModel(): Couldn't load model file at path %s.", storagePath().utf8().data());
122 m_useCorePrediction = false;
123 return nullptr;
124}
