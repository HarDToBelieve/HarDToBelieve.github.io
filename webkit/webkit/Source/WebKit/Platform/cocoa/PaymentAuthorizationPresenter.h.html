39namespace WebCore {
40class Payment;
41class PaymentContact;
42class PaymentMerchantSession;
43class PaymentMethod;
44class PaymentSessionError;
45struct ApplePayCouponCodeUpdate;
46struct ApplePayPaymentMethodUpdate;
47struct ApplePayShippingContactUpdate;
48struct ApplePayShippingMethod;
49struct ApplePayShippingMethodUpdate;

52namespace WebKit {
53
54class PaymentAuthorizationPresenter : public CanMakeWeakPtr<PaymentAuthorizationPresenter> {
55 WTF_MAKE_FAST_ALLOCATED;
56 WTF_MAKE_NONCOPYABLE(PaymentAuthorizationPresenter);
57public:
58 struct Client {
59 virtual ~Client() = default;
60
61 virtual void presenterDidAuthorizePayment(PaymentAuthorizationPresenter&, const WebCore::Payment&) = 0;
62 virtual void presenterDidFinish(PaymentAuthorizationPresenter&, WebCore::PaymentSessionError&&) = 0;
63 virtual void presenterDidSelectPaymentMethod(PaymentAuthorizationPresenter&, const WebCore::PaymentMethod&) = 0;
64 virtual void presenterDidSelectShippingContact(PaymentAuthorizationPresenter&, const WebCore::PaymentContact&) = 0;
65 virtual void presenterDidSelectShippingMethod(PaymentAuthorizationPresenter&, const WebCore::ApplePayShippingMethod&) = 0;
66#if HAVE(PASSKIT_COUPON_CODE)
67 virtual void presenterDidChangeCouponCode(PaymentAuthorizationPresenter&, const String& couponCode) = 0;
68#endif
69 virtual void presenterWillValidateMerchant(PaymentAuthorizationPresenter&, const URL&) = 0;
70 };
71
72 virtual ~PaymentAuthorizationPresenter() = default;
73
74 Client& client() { return m_client; }
75
76 void completeMerchantValidation(const WebCore::PaymentMerchantSession&);
77 void completePaymentMethodSelection(std::optional<WebCore::ApplePayPaymentMethodUpdate>&&);
78 void completePaymentSession(const std::optional<WebCore::PaymentAuthorizationResult>&);
79 void completeShippingContactSelection(std::optional<WebCore::ApplePayShippingContactUpdate>&&);
80 void completeShippingMethodSelection(std::optional<WebCore::ApplePayShippingMethodUpdate>&&);
81#if HAVE(PASSKIT_COUPON_CODE)
82 void completeCouponCodeChange(std::optional<WebCore::ApplePayCouponCodeUpdate>&&);
83#endif
84
85 virtual void dismiss() = 0;
86#if PLATFORM(IOS_FAMILY)
87 virtual void present(UIViewController *, CompletionHandler<void(bool)>&&) = 0;
88#endif
89
90protected:
91 explicit PaymentAuthorizationPresenter(Client& client)
92 : m_client(client)
93 {
94 }
95
96 virtual WKPaymentAuthorizationDelegate *platformDelegate() = 0;
97
98private:
99 Client& m_client;
100};

102} // namespace WebKit
