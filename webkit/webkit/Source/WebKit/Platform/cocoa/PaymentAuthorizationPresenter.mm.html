69static PKPaymentAuthorizationStatus toPKPaymentAuthorizationStatus(WebCore::PaymentAuthorizationStatus status)
70{
71 switch (status) {
72 case WebCore::PaymentAuthorizationStatus::Success:
73 return PKPaymentAuthorizationStatusSuccess;
74 case WebCore::PaymentAuthorizationStatus::Failure:
75 return PKPaymentAuthorizationStatusFailure;
76 case WebCore::PaymentAuthorizationStatus::PINRequired:
77 return PKPaymentAuthorizationStatusPINRequired;
78 case WebCore::PaymentAuthorizationStatus::PINIncorrect:
79 return PKPaymentAuthorizationStatusPINIncorrect;
80 case WebCore::PaymentAuthorizationStatus::PINLockout:
81 return PKPaymentAuthorizationStatusPINLockout;
82 }
83}
84
85static PKPaymentErrorCode toPKPaymentErrorCode(WebCore::ApplePayErrorCode code)
86{
87 switch (code) {
88 case WebCore::ApplePayErrorCode::Unknown:
89 break;
90
91 case WebCore::ApplePayErrorCode::ShippingContactInvalid:
92 return PKPaymentShippingContactInvalidError;
93
94 case WebCore::ApplePayErrorCode::BillingContactInvalid:
95 return PKPaymentBillingContactInvalidError;
96
97 case WebCore::ApplePayErrorCode::AddressUnserviceable:
98 return PKPaymentShippingAddressUnserviceableError;
99
100 case WebCore::ApplePayErrorCode::CouponCodeInvalid:
101#if HAVE(PASSKIT_COUPON_CODE)
102 return PKPaymentCouponCodeInvalidError;
103#else
104 break;
105#endif
106
107 case WebCore::ApplePayErrorCode::CouponCodeExpired:
108#if HAVE(PASSKIT_COUPON_CODE)
109 return PKPaymentCouponCodeExpiredError;
110#else
111 break;
112#endif
113 }
114
115 return PKPaymentUnknownError;
116}

118static NSError *toNSError(const WebCore::ApplePayError& error)
119{
120 auto userInfo = adoptNS([[NSMutableDictionary alloc] init]);
121 [userInfo setObject:error.message() forKey:NSLocalizedDescriptionKey];
122
123 if (auto contactField = error.contactField()) {
124 NSString *pkContactField = nil;
125 NSString *postalAddressKey = nil;
126
127 switch (*contactField) {
128 case WebCore::ApplePayErrorContactField::PhoneNumber:
129 pkContactField = PAL::get_PassKit_PKContactFieldPhoneNumber();
130 break;
131
132 case WebCore::ApplePayErrorContactField::EmailAddress:
133 pkContactField = PAL::get_PassKit_PKContactFieldEmailAddress();
134 break;
135
136 case WebCore::ApplePayErrorContactField::Name:
137 pkContactField = PAL::get_PassKit_PKContactFieldName();
138 break;
139
140 case WebCore::ApplePayErrorContactField::PhoneticName:
141 pkContactField = PAL::get_PassKit_PKContactFieldPhoneticName();
142 break;
143
144 case WebCore::ApplePayErrorContactField::PostalAddress:
145 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
146 break;
147
148 case WebCore::ApplePayErrorContactField::AddressLines:
149 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
150 postalAddressKey = getCNPostalAddressStreetKey();
151 break;
152
153 case WebCore::ApplePayErrorContactField::SubLocality:
154 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
155 postalAddressKey = getCNPostalAddressSubLocalityKey();
156 break;
157
158 case WebCore::ApplePayErrorContactField::Locality:
159 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
160 postalAddressKey = getCNPostalAddressCityKey();
161 break;
162
163 case WebCore::ApplePayErrorContactField::PostalCode:
164 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
165 postalAddressKey = getCNPostalAddressPostalCodeKey();
166 break;
167
168 case WebCore::ApplePayErrorContactField::SubAdministrativeArea:
169 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
170 postalAddressKey = getCNPostalAddressSubAdministrativeAreaKey();
171 break;
172
173 case WebCore::ApplePayErrorContactField::AdministrativeArea:
174 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
175 postalAddressKey = getCNPostalAddressStateKey();
176 break;
177
178 case WebCore::ApplePayErrorContactField::Country:
179 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
180 postalAddressKey = getCNPostalAddressCountryKey();
181 break;
182
183 case WebCore::ApplePayErrorContactField::CountryCode:
184 pkContactField = PAL::get_PassKit_PKContactFieldPostalAddress();
185 postalAddressKey = getCNPostalAddressISOCountryCodeKey();
186 break;
187 }
188
189 [userInfo setObject:pkContactField forKey:PAL::get_PassKit_PKPaymentErrorContactFieldUserInfoKey()];
190 if (postalAddressKey)
191 [userInfo setObject:postalAddressKey forKey:PAL::get_PassKit_PKPaymentErrorPostalAddressUserInfoKey()];
192 }
193
194 return [NSError errorWithDomain:PAL::get_PassKit_PKPaymentErrorDomain() code:toPKPaymentErrorCode(error.code()) userInfo:userInfo.get()];
195}

197static RetainPtr<NSArray> toNSErrors(const Vector<RefPtr<WebCore::ApplePayError>>& errors)
198{
199 return createNSArray(errors, [] (auto& error) -> NSError * {
200 return error ? toNSError(*error) : nil;
201 });
202}
203
204static RetainPtr<NSArray> toPKShippingMethods(const Vector<WebCore::ApplePayShippingMethod>& shippingMethods)
205{
206 return createNSArray(shippingMethods, [] (auto& method) {
207 return toPKShippingMethod(method);
208 });
209}

217void PaymentAuthorizationPresenter::completePaymentMethodSelection(std::optional<WebCore::ApplePayPaymentMethodUpdate>&& update)
218{
219 ASSERT(platformDelegate());
220 if (!update) {
221 [platformDelegate() completePaymentMethodSelection:nil];
222 return;
223 }
224
225 auto paymentMethodUpdate = adoptNS([PAL::allocPKPaymentRequestPaymentMethodUpdateInstance() initWithPaymentSummaryItems:WebCore::platformSummaryItems(WTFMove(update->newTotal), WTFMove(update->newLineItems))]);
226#if HAVE(PASSKIT_INSTALLMENTS) && ENABLE(APPLE_PAY_INSTALLMENTS)
227 [paymentMethodUpdate setInstallmentGroupIdentifier:WTFMove(update->installmentGroupIdentifier)];
228#endif // HAVE(PASSKIT_INSTALLMENTS) && ENABLE(APPLE_PAY_INSTALLMENTS)
229#if defined(PaymentAuthorizationPresenterAdditions_completePaymentMethodSelection)
230 PaymentAuthorizationPresenterAdditions_completePaymentMethodSelection
231#endif
232 [platformDelegate() completePaymentMethodSelection:paymentMethodUpdate.get()];
233}

235void PaymentAuthorizationPresenter::completePaymentSession(const std::optional<WebCore::PaymentAuthorizationResult>& result)
236{
237 ASSERT(platformDelegate());
238 auto status = result ? toPKPaymentAuthorizationStatus(result->status) : PKPaymentAuthorizationStatusSuccess;
239 RetainPtr<NSArray> errors = result ? toNSErrors(result->errors) : @[ ];
240 [platformDelegate() completePaymentSession:status errors:errors.get()];
241}
