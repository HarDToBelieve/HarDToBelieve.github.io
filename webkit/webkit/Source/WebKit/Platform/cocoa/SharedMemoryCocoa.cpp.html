47static int toVMMemoryLedger(MemoryLedger memoryLedger)
48{
49 switch (memoryLedger) {
50 case MemoryLedger::None:
51 return VM_LEDGER_TAG_NONE;
52 case MemoryLedger::Default:
53 return VM_LEDGER_TAG_DEFAULT;
54 case MemoryLedger::Network:
55 return VM_LEDGER_TAG_NETWORK;
56 case MemoryLedger::Media:
57 return VM_LEDGER_TAG_MEDIA;
58 case MemoryLedger::Graphics:
59 return VM_LEDGER_TAG_GRAPHICS;
60 case MemoryLedger::Neural:
61 return VM_LEDGER_TAG_NEURAL;
62 }
63}

66static inline std::optional<size_t> safeRoundPage(size_t size)
67{
68 size_t roundedSize;
69 if (__builtin_add_overflow(size, static_cast<size_t>(PAGE_MASK), &roundedSize))
70 return std::nullopt;
71 roundedSize &= ~static_cast<size_t>(PAGE_MASK);
72 return roundedSize;
73}

99void SharedMemory::Handle::takeOwnershipOfMemory(MemoryLedger memoryLedger) const
100{
101#if HAVE(MACH_MEMORY_ENTRY)
102 if (!m_port)
103 return;
104
105 kern_return_t kr = mach_memory_entry_ownership(m_port, mach_task_self(), toVMMemoryLedger(memoryLedger), 0);
106 if (kr != KERN_SUCCESS)
107 RELEASE_LOG_ERROR(VirtualMemory, "SharedMemory::Handle::setOwnership: Failed ownership of shared memory. Error: %{public}s (%x)", mach_error_string(kr), kr);
108#else
109 UNUSED_PARAM(memoryLedger);
110#endif
111}

127void SharedMemory::IPCHandle::encode(IPC::Encoder& encoder) const
128{
129 encoder << static_cast<uint64_t>(handle.m_size);
130 encoder << dataSize;
131 encoder << IPC::MachPort(handle.m_port, MACH_MSG_TYPE_MOVE_SEND);
132 handle.m_port = MACH_PORT_NULL;

135bool SharedMemory::IPCHandle::decode(IPC::Decoder& decoder, IPCHandle& ipcHandle)
136{
137 ASSERT(!ipcHandle.handle.m_port);
138 ASSERT(!ipcHandle.handle.m_size);
139
140 SharedMemory::Handle handle;
141
142 uint64_t bufferSize;
143 if (!decoder.decode(bufferSize))
144 return false;
145
146 uint64_t dataLength;
147 if (!decoder.decode(dataLength))
148 return false;
149
150 // SharedMemory::Handle::size() is rounded up to the nearest page.
151 if (dataLength > bufferSize)
152 return false;
153
154 IPC::MachPort machPort;
155 if (!decoder.decode(machPort))
156 return false;
157
158 handle.m_size = bufferSize;
159 handle.m_port = machPort.port();
160 ipcHandle.handle = WTFMove(handle);
161 ipcHandle.dataSize = dataLength;
162 return true;
163}

175RefPtr<SharedMemory> SharedMemory::allocate(size_t size)
176{
177 ASSERT(size);
178
179 auto roundedSize = safeRoundPage(size);
180 if (!roundedSize) {
181 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::allocate: Failed to allocate shared memory (%zu bytes) due to overflow", nullptr, size);
182 return nullptr;
183 }
184
185 mach_vm_address_t address = 0;
186 // Using VM_FLAGS_PURGABLE so that we can later transfer ownership of the memory via mach_memory_entry_ownership().
187 kern_return_t kr = mach_vm_allocate(mach_task_self(), &address, *roundedSize, VM_FLAGS_ANYWHERE | VM_FLAGS_PURGABLE);
188 if (kr != KERN_SUCCESS) {
189#if RELEASE_LOG_DISABLED
190 LOG_ERROR("Failed to allocate mach_vm_allocate shared memory (%zu bytes). %s (%x)", size, mach_error_string(kr), kr);
191#else
192 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::allocate: Failed to allocate mach_vm_allocate shared memory (%zu bytes). %{public}s (%x)", nullptr, size, mach_error_string(kr), kr);
193#endif
194 return nullptr;
195 }
196
197 auto sharedMemory = adoptRef(*new SharedMemory);
198 sharedMemory->m_size = size;
199 sharedMemory->m_data = toPointer(address);
200 sharedMemory->m_port = MACH_PORT_NULL;
201 sharedMemory->m_protection = Protection::ReadWrite;
202
203 return WTFMove(sharedMemory);
204}

206static inline vm_prot_t machProtection(SharedMemory::Protection protection)
207{
208 switch (protection) {
209 case SharedMemory::Protection::ReadOnly:
210 return VM_PROT_READ;
211 case SharedMemory::Protection::ReadWrite:
212 return VM_PROT_READ | VM_PROT_WRITE;
213 }
214
215 ASSERT_NOT_REACHED();
216 return VM_PROT_NONE;
217}

219static WTF::MachSendRight makeMemoryEntry(size_t size, vm_offset_t offset, SharedMemory::Protection protection, mach_port_t parentEntry)
220{
221 auto roundedSize = safeRoundPage(size);
222 if (!roundedSize) {
223 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::makeMemoryEntry: Failed to create a mach port for shared memory (%zu bytes) due to overflow", nullptr, size);
224 return { };
225 }
226
227 memory_object_size_t memoryObjectSize = *roundedSize;
228 mach_port_t port = MACH_PORT_NULL;
229
230 // First try without MAP_MEM_VM_SHARE because it prevents memory ownership transfer. We only pass the MAP_MEM_VM_SHARE flag as a fallback.
231 kern_return_t kr = mach_make_memory_entry_64(mach_task_self(), &memoryObjectSize, offset, machProtection(protection) | VM_PROT_IS_MASK, &port, parentEntry);
232 if (kr != KERN_SUCCESS) {
233 RELEASE_LOG(VirtualMemory, "SharedMemory::makeMemoryEntry: Failed to create a mach port for shared memory, will try again with MAP_MEM_VM_SHARE flag. Error: %{public}s (%x)", mach_error_string(kr), kr);
234 kr = mach_make_memory_entry_64(mach_task_self(), &memoryObjectSize, offset, machProtection(protection) | VM_PROT_IS_MASK | MAP_MEM_VM_SHARE, &port, parentEntry);
235 if (kr != KERN_SUCCESS) {
236#if RELEASE_LOG_DISABLED
237 LOG_ERROR("Failed to create a mach port for shared memory. %s (%x)", mach_error_string(kr), kr);
238#else
239 RELEASE_LOG_ERROR(VirtualMemory, "SharedMemory::makeMemoryEntry: Failed to create a mach port for shared memory with MAP_MEM_VM_SHARE flag. Error: %{public}s (%x)", mach_error_string(kr), kr);
240#endif
241 return { };
242 }
243 }
244
245 RELEASE_ASSERT(memoryObjectSize >= size);
246
247 return WTF::MachSendRight::adopt(port);
248}

250RefPtr<SharedMemory> SharedMemory::wrapMap(void* data, size_t size, Protection protection)
251{
252 ASSERT(size);
253
254 auto sendRight = makeMemoryEntry(size, toVMAddress(data), protection, MACH_PORT_NULL);
255 if (!sendRight)
256 return nullptr;
257
258 auto sharedMemory(adoptRef(*new SharedMemory));
259 sharedMemory->m_size = size;
260 sharedMemory->m_data = nullptr;
261 sharedMemory->m_port = sendRight.leakSendRight();
262 sharedMemory->m_protection = protection;
263
264 return WTFMove(sharedMemory);
265}

267RefPtr<SharedMemory> SharedMemory::map(const Handle& handle, Protection protection)
268{
269 if (handle.isNull())
270 return nullptr;
271
272 vm_prot_t vmProtection = machProtection(protection);
273 mach_vm_address_t mappedAddress = 0;
274 kern_return_t kr = mach_vm_map(mach_task_self(), &mappedAddress, handle.m_size, 0, VM_FLAGS_ANYWHERE, handle.m_port, 0, false, vmProtection, vmProtection, VM_INHERIT_NONE);
275#if RELEASE_LOG_DISABLED
276 if (kr != KERN_SUCCESS)
277 return nullptr;
278#else
279 if (kr != KERN_SUCCESS) {
280 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::map: Failed to map shared memory. %{public}s (%x)", nullptr, mach_error_string(kr), kr);
281 return nullptr;
282 }
283#endif
284
285 auto sharedMemory(adoptRef(*new SharedMemory));
286 sharedMemory->m_size = handle.m_size;
287 sharedMemory->m_data = toPointer(mappedAddress);
288 sharedMemory->m_port = MACH_PORT_NULL;
289 sharedMemory->m_protection = protection;
290
291 return WTFMove(sharedMemory);
292}

294SharedMemory::~SharedMemory()
295{
296 if (m_data) {
297 auto roundedSize = safeRoundPage(m_size);
298 if (!roundedSize) {
299 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::~SharedMemory: Failed to deallocate memory (%zu bytes) due to overflow", this, m_size);
300 RELEASE_ASSERT_NOT_REACHED();
301 }
302
303 kern_return_t kr = mach_vm_deallocate(mach_task_self(), toVMAddress(m_data), *roundedSize);
304#if RELEASE_LOG_DISABLED
305 ASSERT_UNUSED(kr, kr == KERN_SUCCESS);
306#else
307 if (kr != KERN_SUCCESS) {
308 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::~SharedMemory: Failed to deallocate shared memory. %{public}s (%x)", this, mach_error_string(kr), kr);
309 ASSERT_NOT_REACHED();
310 }
311#endif
312 }
313
314 if (m_port) {
315 kern_return_t kr = mach_port_deallocate(mach_task_self(), m_port);
316#if RELEASE_LOG_DISABLED
317 ASSERT_UNUSED(kr, kr == KERN_SUCCESS);
318#else
319 if (kr != KERN_SUCCESS) {
320 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::~SharedMemory: Failed to deallocate port. %{public}s (%x)", this, mach_error_string(kr), kr);
321 ASSERT_NOT_REACHED();
322 }
323#endif
324 }

327bool SharedMemory::createHandle(Handle& handle, Protection protection)
328{
329 ASSERT(!handle.m_port);
330 ASSERT(!handle.m_size);
331
332 auto roundedSize = safeRoundPage(m_size);
333 if (!roundedSize) {
334 RELEASE_LOG_ERROR(VirtualMemory, "%p - SharedMemory::createHandle: Failed to create handle (%zu bytes) due to overflow", this, m_size);
335 return false;
336 }
337
338 auto sendRight = createSendRight(protection);
339 if (!sendRight)
340 return false;
341
342 handle.m_port = sendRight.leakSendRight();
343 handle.m_size = *roundedSize;
344
345 return true;
346}

353WTF::MachSendRight SharedMemory::createSendRight(Protection protection) const
354{
355 ASSERT(m_protection == protection || m_protection == Protection::ReadWrite && protection == Protection::ReadOnly);
356 ASSERT(!!m_data ^ !!m_port);
357
358 if (m_port && m_protection == protection)
359 return WTF::MachSendRight::create(m_port);
360
361 ASSERT(m_data);
362 return makeMemoryEntry(m_size, toVMAddress(m_data), protection, MACH_PORT_NULL);
363}
