31void setMachPortQueueLength(mach_port_t receivePort, mach_port_msgcount_t queueLength)
32{
33 ASSERT(MACH_PORT_VALID(receivePort));
34
35 mach_port_limits_t portLimits;
36 portLimits.mpl_qlimit = queueLength;
37
38 mach_port_set_attributes(mach_task_self(), receivePort, MACH_PORT_LIMITS_INFO, reinterpret_cast<mach_port_info_t>(&portLimits), MACH_PORT_LIMITS_INFO_COUNT);
39}
