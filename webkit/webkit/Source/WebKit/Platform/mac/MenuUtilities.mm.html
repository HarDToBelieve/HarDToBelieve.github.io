67#if HAVE(DATA_DETECTORS_MAC_ACTION)
68static DDMacAction *actionForMenuItem(NSMenuItem *item)
69#else
70static DDAction *actionForMenuItem(NSMenuItem *item)
71#endif
72{
73 NSDictionary *representedObject = item.representedObject;
74 if (![representedObject isKindOfClass:[NSDictionary class]])
75 return nil;
76
77 id action = [representedObject objectForKey:@"DDAction"];
78
79#if HAVE(DATA_DETECTORS_MAC_ACTION)
80 if (![action isKindOfClass:PAL::getDDMacActionClass()])
81 return nil;
82#else
83 if (![action isKindOfClass:PAL::getDDActionClass()])
84 return nil;
85#endif
86
87 return action;
88}

90NSMenuItem *menuItemForTelephoneNumber(const String& telephoneNumber)
91{
92 if (!PAL::isDataDetectorsFrameworkAvailable())
93 return nil;
94
95 auto actionContext = adoptNS([PAL::allocDDActionContextInstance() init]);
96 [actionContext setAllowedActionUTIs:@[ @"com.apple.dial" ]];
97
98 NSArray *proposedMenuItems = [[PAL::getDDActionsManagerClass() sharedManager] menuItemsForValue:(NSString *)telephoneNumber type:PAL::get_DataDetectorsCore_DDBinderPhoneNumberKey() service:nil context:actionContext.get()];
99 for (NSMenuItem *item in proposedMenuItems) {
100 auto action = actionForMenuItem(item);
101 if ([action.actionUTI hasPrefix:@"com.apple.dial"]) {
102 item.title = formattedPhoneNumberString(telephoneNumber);
103 return item;
104 }
105 }
106
107 return nil;
108}

110RetainPtr<NSMenu> menuForTelephoneNumber(const String& telephoneNumber)
111{
112 if (!PAL::isRevealFrameworkAvailable() || !PAL::isRevealCoreFrameworkAvailable())
113 return nil;
114
115 RetainPtr<NSMenu> menu = adoptNS([[NSMenu alloc] init]);
116 auto viewForPresenter = adoptNS([[NSView alloc] init]);
117 auto urlComponents = adoptNS([[NSURLComponents alloc] init]);
118 [urlComponents setScheme:@"tel"];
119 [urlComponents setPath:telephoneNumber];
120 auto item = adoptNS([PAL::allocRVItemInstance() initWithURL:[urlComponents URL] rangeInContext:NSMakeRange(0, telephoneNumber.length())]);
121 auto presenter = adoptNS([PAL::allocRVPresenterInstance() init]);
122 auto delegate = adoptNS([[WKEmptyPresenterHighlightDelegate alloc] init]);
123 auto context = adoptNS([PAL::allocRVPresentingContextInstance() initWithPointerLocationInView:NSZeroPoint inView:viewForPresenter.get() highlightDelegate:delegate.get()]);
124 NSArray *proposedMenuItems = [presenter menuItemsForItem:item.get() documentContext:nil presentingContext:context.get() options:nil];
125
126 [menu setItemArray:proposedMenuItems];
127
128 return menu;
129}
