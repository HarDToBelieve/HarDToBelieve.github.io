79void PluginProcess::initializeProcess(const AuxiliaryProcessInitializationParameters& parameters)
80{
81 WTF::setProcessPrivileges(allPrivileges());
82 WebCore::NetworkStorageSession::permitProcessToUseCookieAPI(true);
83 m_pluginPath = parameters.extraInitializationData.get("plugin-path");
84 platformInitializeProcess(parameters);
85}

87void PluginProcess::initializeConnection(IPC::Connection* connection)
88{
89 AuxiliaryProcess::initializeConnection(connection);
90
91 // We call _exit() directly from the background queue in case the main thread is unresponsive
92 // and AuxiliaryProcess::didClose() does not get called.
93 connection->setDidCloseOnConnectionWorkQueueCallback(callExit);
94}

96void PluginProcess::removeWebProcessConnection(WebProcessConnection* webProcessConnection)
97{
98 size_t vectorIndex = m_webProcessConnections.find(webProcessConnection);
99 ASSERT(vectorIndex != notFound);
100
101 m_webProcessConnections.remove(vectorIndex);
102
103 if (m_webProcessConnections.isEmpty() && m_pluginModule) {
104 // Decrement the load count. This is balanced by a call to incrementLoadCount in createWebProcessConnection.
105 m_pluginModule->decrementLoadCount();
106 }
107
108 enableTermination();
109}

111NetscapePluginModule* PluginProcess::netscapePluginModule()
112{
113 if (!m_pluginModule) {
114 ASSERT(!m_pluginPath.isNull());
115 m_pluginModule = NetscapePluginModule::getOrCreate(m_pluginPath);
116
117#if PLATFORM(MAC)
118 if (m_pluginModule) {
119 if (m_pluginModule->pluginQuirks().contains(PluginQuirks::PrognameShouldBeWebKitPluginHost))
120 *const_cast<const char**>(_NSGetProgname()) = "WebKitPluginHost";
121 }
122#endif
123 }
124
125 return m_pluginModule.get();
126}

145void PluginProcess::initializePluginProcess(PluginProcessCreationParameters&& parameters)
146{
147 ASSERT(!m_pluginModule);
148
149 auto& memoryPressureHandler = MemoryPressureHandler::singleton();
150 memoryPressureHandler.setLowMemoryHandler([this] (Critical, Synchronous) {
151 if (shouldTerminate())
152 terminate();
153 });
154 memoryPressureHandler.install();
155
156 m_supportsAsynchronousPluginInitialization = parameters.supportsAsynchronousPluginInitialization;
157 setMinimumLifetime(parameters.minimumLifetime);
158 setTerminationTimeout(parameters.terminationTimeout);
159
160 platformInitializePluginProcess(WTFMove(parameters));
161}

163void PluginProcess::createWebProcessConnection()
164{
165 // FIXME: Merge this with AuxiliaryProcess::createIPCConnectionPair().
166
167 bool didHaveAnyWebProcessConnections = !m_webProcessConnections.isEmpty();
168
169#if USE(UNIX_DOMAIN_SOCKETS)
170 IPC::Connection::SocketPair socketPair = IPC::Connection::createPlatformConnection();
171
172 auto connection = WebProcessConnection::create(socketPair.server);
173 m_webProcessConnections.append(WTFMove(connection));
174
175 IPC::Attachment clientSocket(socketPair.client);
176 parentProcessConnection()->send(Messages::PluginProcessProxy::DidCreateWebProcessConnection(clientSocket, m_supportsAsynchronousPluginInitialization), 0);
177#elif OS(DARWIN)
178 // Create the listening port.
179 mach_port_t listeningPort = MACH_PORT_NULL;
180 auto kr = mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &listeningPort);
181 if (kr != KERN_SUCCESS) {
182 LOG_ERROR("Could not allocate mach port, error %x", kr);
183 CRASH();
184 }
185
186 // Create a listening connection.
187 auto connection = WebProcessConnection::create(IPC::Connection::Identifier(listeningPort));
188
189 m_webProcessConnections.append(WTFMove(connection));
190
191 IPC::Attachment clientPort(listeningPort, MACH_MSG_TYPE_MAKE_SEND);
192 parentProcessConnection()->send(Messages::PluginProcessProxy::DidCreateWebProcessConnection(clientPort, m_supportsAsynchronousPluginInitialization), 0);
193#else
194 notImplemented();
195#endif
196
197 if (NetscapePluginModule* module = netscapePluginModule()) {
198 if (!didHaveAnyWebProcessConnections) {
199 // Increment the load count. This is matched by a call to decrementLoadCount in removeWebProcessConnection.
200 // We do this so that the plug-in module's NP_Shutdown won't be called until right before exiting.
201 module->incrementLoadCount();
202 }
203 }
204
205 disableTermination();
206}

208void PluginProcess::getSitesWithData(uint64_t callbackID)
209{
210 Vector<String> sites;
211 if (NetscapePluginModule* module = netscapePluginModule())
212 sites = module->sitesWithData();
213
214 parentProcessConnection()->send(Messages::PluginProcessProxy::DidGetSitesWithData(sites, callbackID), 0);
215}

217void PluginProcess::deleteWebsiteData(WallTime modifiedSince, uint64_t callbackID)
218{
219 if (auto* module = netscapePluginModule()) {
220 auto currentTime = WallTime::now();
221
222 if (currentTime > modifiedSince) {
223 uint64_t maximumAge = (currentTime - modifiedSince).secondsAs<uint64_t>();
224
225 module->clearSiteData(String(), NP_CLEAR_ALL, maximumAge);
226 }
227 }
228
229 parentProcessConnection()->send(Messages::PluginProcessProxy::DidDeleteWebsiteData(callbackID), 0);
230}
231
232void PluginProcess::deleteWebsiteDataForHostNames(const Vector<String>& hostNames, uint64_t callbackID)
233{
234 if (auto* module = netscapePluginModule()) {
235 for (auto& hostName : hostNames)
236 module->clearSiteData(hostName, NP_CLEAR_ALL, std::numeric_limits<uint64_t>::max());
237 }
238
239 parentProcessConnection()->send(Messages::PluginProcessProxy::DidDeleteWebsiteDataForHostNames(callbackID), 0);
240}

242void PluginProcess::setMinimumLifetime(Seconds lifetime)
243{
244 if (lifetime <= 0_s)
245 return;
246
247 disableTermination();
248
249 m_minimumLifetimeTimer.startOneShot(lifetime);
250}
251
252void PluginProcess::minimumLifetimeTimerFired()
253{
254 enableTermination();
255}
