72public:
73 FullscreenWindowTracker() { }

78private:
79 HashSet<CGWindowID> m_windows;

82static bool rectCoversAnyScreen(NSRect rect)
83{
84 for (NSScreen *screen in [NSScreen screens]) {
85 if (NSContainsRect(rect, [screen frame]))
86 return YES;
87 }
88 return NO;
89}
90
91static bool windowCoversAnyScreen(NSWindow *window)
92{
93 return rectCoversAnyScreen(window.frame);
94}
95
96static CGWindowID cgWindowID(NSWindow *window)
97{
98 return window.windowNumber;
99}
100
101template<typename T>
102void FullscreenWindowTracker::windowShown(T window)
103{
104 CGWindowID windowID = cgWindowID(window);
105 if (!windowID)
106 return;
107
108 // If this window is already visible then there is nothing to do.
109 if (m_windows.contains(windowID))
110 return;
111
112 // If the window is not full-screen then we're not interested in it.
113 if (!windowCoversAnyScreen(window))
114 return;
115
116 bool windowSetWasEmpty = m_windows.isEmpty();
117
118 m_windows.add(windowID);
119
120 // If this is the first full screen window to be shown, notify the UI process.
121 if (windowSetWasEmpty)
122 PluginProcess::singleton().setFullscreenWindowIsShowing(true);
123}
124
125template<typename T>
126void FullscreenWindowTracker::windowHidden(T window)
127{
128 CGWindowID windowID = cgWindowID(window);
129 if (!windowID)
130 return;
131
132 // If this is not a window that we're tracking then there is nothing to do.
133 if (!m_windows.remove(windowID))
134 return;
135
136 // If this was the last full screen window that was visible, notify the UI process.
137 if (m_windows.isEmpty())
138 PluginProcess::singleton().setFullscreenWindowIsShowing(false);
139}

152static unsigned modalCount;
153
154static void beginModal()
155{
156 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
157 // Make sure to make ourselves the front process
158 ProcessSerialNumber psn;
159 GetCurrentProcess(&psn);
160 SetFrontProcess(&psn);
161 ALLOW_DEPRECATED_DECLARATIONS_END
162
163 if (!modalCount++)
164 setModal(true);
165}
166
167static void endModal()
168{
169 if (!--modalCount)
170 setModal(false);
171}

186static bool isAdobeAcrobatAddress(const void* address)
187{
188 Dl_info imageInfo;
189 if (!dladdr(address, &imageInfo))
190 return false;
191
192 const char* pathSuffix = "/Contents/Frameworks/Acrobat.framework/Acrobat";
193
194 int pathSuffixLength = strlen(pathSuffix);
195 int imageFilePathLength = strlen(imageInfo.dli_fname);
196
197 if (imageFilePathLength < pathSuffixLength)
198 return false;
199
200 if (strcmp(imageInfo.dli_fname + (imageFilePathLength - pathSuffixLength), pathSuffix))
201 return false;
202
203 return true;
204}

206static bool stringCompare(CFStringRef a, CFStringRef b, CFStringCompareFlags options, void* returnAddress, CFComparisonResult& result)
207{
208 if (pthread_main_np() != 1)
209 return false;
210
211 if (!oldPluginProcessNameShouldEqualNewPluginProcessNameForAdobeReader)
212 return false;
213
214 if (options != kCFCompareCaseInsensitive)
215 return false;
216
217 const char* aCString = CFStringGetCStringPtr(a, kCFStringEncodingASCII);
218 if (!aCString)
219 return false;
220
221 const char* bCString = CFStringGetCStringPtr(b, kCFStringEncodingASCII);
222 if (!bCString)
223 return false;
224
225 if (strcmp(aCString, "com.apple.WebKit.PluginProcess"))
226 return false;
227
228 if (strcmp(bCString, "com.apple.WebKit.Plugin.64"))
229 return false;
230
231 // Check if the LHS string comes from the Acrobat framework.
232 if (!isAdobeAcrobatAddress(a))
233 return false;
234
235 // Check if the return adress is part of the Acrobat framework as well.
236 if (!isAdobeAcrobatAddress(returnAddress))
237 return false;
238
239 result = kCFCompareEqualTo;
240 return true;
241}

252}

279

293static void initializeCocoaOverrides()
294{
295 // Override -[NSConcreteTask launch:]
296 Method launchMethod = class_getInstanceMethod(objc_getClass("NSConcreteTask"), @selector(launch));
297 NSConcreteTask_launch = reinterpret_cast<void (*)(NSTask *, SEL)>(method_setImplementation(launchMethod, reinterpret_cast<IMP>(replacedNSConcreteTask_launch)));
298
299 // Override -[NSWorkspace launchApplicationAtURL:options:configuration:error:]
300 // FIXME: Are these deprecation allowances necessary?
301 Method launchApplicationAtURLOptionsConfigurationErrorMethod = class_getInstanceMethod(objc_getClass("NSWorkspace"), @selector(launchApplicationAtURL:options:configuration:error:));
302 NSWorkspace_launchApplicationAtURL_options_configuration_error = reinterpret_cast<NSRunningApplication *(*)(NSWorkspace *, SEL, NSURL *, NSWorkspaceLaunchOptions, NSDictionary *, NSError **)>(method_setImplementation(launchApplicationAtURLOptionsConfigurationErrorMethod, reinterpret_cast<IMP>(replacedNSWorkspace_launchApplicationAtURL_options_configuration_error)));
303
304 // Override -[NSWorkspace openFile:]
305 Method openFileMethod = class_getInstanceMethod(objc_getClass("NSWorkspace"), @selector(openFile:));
306 NSWorkspace_openFile = reinterpret_cast<BOOL (*)(NSWorkspace *, SEL, NSString *)>(method_setImplementation(openFileMethod, reinterpret_cast<IMP>(replacedNSWorkspace_openFile)));
307
308 // Override -[NSApplication runModalForWindow:]
309 Method runModalForWindowMethod = class_getInstanceMethod(objc_getClass("NSApplication"), @selector(runModalForWindow:));
310 NSApplication_RunModalForWindow = method_setImplementation(runModalForWindowMethod, reinterpret_cast<IMP>(replacedRunModalForWindow));
311
312 NSNotificationCenter *defaultCenter = [NSNotificationCenter defaultCenter];
313
314 // Track when any Cocoa window is about to be be shown.
315 [defaultCenter addObserverForName:NSWindowWillOrderOnScreenNotification
316 object:nil
317 queue:nil
318 usingBlock:^(NSNotification *notification) { fullscreenWindowTracker().windowShown([notification object]); }];
319 // Track when any Cocoa window is about to be hidden.
320 [defaultCenter addObserverForName:NSWindowWillOrderOffScreenNotification
321 object:nil
322 queue:nil
323 usingBlock:^(NSNotification *notification) { fullscreenWindowTracker().windowHidden([notification object]); }];
324

390void PluginProcess::platformInitializeProcess(const AuxiliaryProcessInitializationParameters& parameters)
391{
392 initializeShim();
393
394 initializeCocoaOverrides();
395
396 bool experimentalPlugInSandboxProfilesEnabled = parameters.extraInitializationData.get("experimental-sandbox-plugin") == "1";
397 WebCore::RuntimeEnabledFeatures::sharedFeatures().setExperimentalPlugInSandboxProfilesEnabled(experimentalPlugInSandboxProfilesEnabled);
398
399 // FIXME: It would be better to proxy SetCursor calls over to the UI process instead of
400 // allowing plug-ins to change the mouse cursor at any time.
401 // FIXME: SetsCursorInBackground connection property is deprecated in favor of kCGSSetsCursorInBackgroundTagBit window tag bit.
402 // <rdar://problem/7752422> asks for an API to set cursor from background processes.
403 CGSConnectionID cid = CGSMainConnectionID();
404 CGSSetConnectionProperty(cid, cid, CFSTR("SetsCursorInBackground"), (CFTypeRef)kCFBooleanTrue);
405
406 RetainPtr<CFURLRef> pluginURL = adoptCF(CFURLCreateWithFileSystemPath(0, m_pluginPath.createCFString().get(), kCFURLPOSIXPathStyle, false));
407 if (!pluginURL)
408 return;
409
410 RetainPtr<CFBundleRef> pluginBundle = adoptCF(CFBundleCreate(kCFAllocatorDefault, pluginURL.get()));
411 if (!pluginBundle)
412 return;
413
414 m_pluginBundleIdentifier = CFBundleGetIdentifier(pluginBundle.get());
415
416 if (m_pluginBundleIdentifier == "com.adobe.acrobat.pdfviewerNPAPI")
417 oldPluginProcessNameShouldEqualNewPluginProcessNameForAdobeReader = true;
418
419 // FIXME: Workaround for Java not liking its plugin process to be suppressed - <rdar://problem/14267843>
420 if (m_pluginBundleIdentifier == "com.oracle.java.JavaAppletPlugin")
421 (new UserActivity("com.oracle.java.JavaAppletPlugin"))->start();
422
423 if (!pluginHasSandboxProfile(m_pluginBundleIdentifier)) {
424 // Allow Apple Events from Citrix plugin. This can be removed when <rdar://problem/14012823> is fixed.
425 setenv("__APPLEEVENTSSERVICENAME", "com.apple.coreservices.appleevents", 1);
426 }
427}

431 NSString *applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ (%@ Internet plug-in)", "visible name of the plug-in host process. The first argument is the plug-in name and the second argument is the application name."), [[(NSString *)m_pluginPath lastPathComponent] stringByDeletingPathExtension], (NSString *)parameters.uiProcessName];

437void PluginProcess::initializeSandbox(const AuxiliaryProcessInitializationParameters& parameters, SandboxInitializationParameters& sandboxParameters)
438{
439 // PluginProcess may already be sandboxed if its parent process was sandboxed, and launched a child process instead of an XPC service.
440 // This is generally not expected, however we currently always spawn a child process to create a MIME type preferences file.
441 if (currentProcessIsSandboxed()) {
442 RELEASE_ASSERT(!parameters.connectionIdentifier.xpcConnection);
443 return;
444 }
445
446 char cacheDirectory[PATH_MAX];
447 if (!confstr(_CS_DARWIN_USER_CACHE_DIR, cacheDirectory, sizeof(cacheDirectory))) {
448 WTFLogAlways("PluginProcess: couldn't retrieve system cache directory path: %d\n", errno);
449 exit(EX_OSERR);
450 }
451
452 m_nsurlCacheDirectory = [[[NSFileManager defaultManager] stringWithFileSystemRepresentation:cacheDirectory length:strlen(cacheDirectory)] stringByAppendingPathComponent:[[NSBundle mainBundle] bundleIdentifier]];
453 if (![[NSFileManager defaultManager] createDirectoryAtURL:[NSURL fileURLWithPath:m_nsurlCacheDirectory isDirectory:YES] withIntermediateDirectories:YES attributes:nil error:nil]) {
454 WTFLogAlways("PluginProcess: couldn't create NSURL cache directory '%s'\n", cacheDirectory);
455 exit(EX_OSERR);
456 }
457
458 bool parentIsSandboxed = parameters.connectionIdentifier.xpcConnection && connectedProcessIsSandboxed(parameters.connectionIdentifier.xpcConnection.get());
459
460 if (parameters.extraInitializationData.get("disable-sandbox") == "1") {
461 if (parentIsSandboxed) {
462 WTFLogAlways("Sandboxed processes may not disable plug-in sandbox, terminating %s.", parameters.clientIdentifier.utf8().data());
463 exit(EX_OSERR);
464 }
465 return;
466 }
467
468 String sandboxProfile = pluginSandboxProfile(m_pluginBundleIdentifier);
469 if (sandboxProfile.isEmpty()) {
470 if (parentIsSandboxed) {
471 WTFLogAlways("Sandboxed processes may only use sandboxed plug-ins, terminating %s.", parameters.clientIdentifier.utf8().data());
472 exit(EX_OSERR);
473 }
474 return;
475 }
476
477 sandboxParameters.setSandboxProfile(sandboxProfile);
478
479 char temporaryDirectory[PATH_MAX];
480 if (!confstr(_CS_DARWIN_USER_TEMP_DIR, temporaryDirectory, sizeof(temporaryDirectory))) {
481 WTFLogAlways("PluginProcess: couldn't retrieve system temporary directory path: %d\n", errno);
482 exit(EX_OSERR);
483 }
484
485 if (strlcpy(temporaryDirectory, [[[[NSFileManager defaultManager] stringWithFileSystemRepresentation:temporaryDirectory length:strlen(temporaryDirectory)] stringByAppendingPathComponent:@"WebKitPlugin-XXXXXX"] fileSystemRepresentation], sizeof(temporaryDirectory)) >= sizeof(temporaryDirectory)
486 || !mkdtemp(temporaryDirectory)) {
487 WTFLogAlways("PluginProcess: couldn't create private temporary directory '%s'\n", temporaryDirectory);
488 exit(EX_OSERR);
489 }
490
491 sandboxParameters.setUserDirectorySuffix([[[[NSFileManager defaultManager] stringWithFileSystemRepresentation:temporaryDirectory length:strlen(temporaryDirectory)] lastPathComponent] fileSystemRepresentation]);
492
493 sandboxParameters.addPathParameter("PLUGIN_PATH", m_pluginPath);
494 sandboxParameters.addPathParameter("NSURL_CACHE_DIR", m_nsurlCacheDirectory);
495
496 [[NSUserDefaults standardUserDefaults] registerDefaults:@{ @"NSUseRemoteSavePanel" : @YES }];
497
498 AuxiliaryProcess::initializeSandbox(parameters, sandboxParameters);
499}
