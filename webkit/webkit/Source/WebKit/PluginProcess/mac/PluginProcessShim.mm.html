53// Simple Fake System V shared memory. This replacement API implements
54// usable system V shared memory for use within a single process. The memory
55// is not shared outside of the scope of the process.
56struct FakeSharedMemoryDescriptor {
57 FakeSharedMemoryDescriptor* next;
58 int referenceCount;
59 key_t key;
60 size_t requestedSize;
61 size_t mmapedSize;
62 int sharedMemoryFlags;
63 int sharedMemoryIdentifier;
64 void* mmapedAddress;
65};
66
67static FakeSharedMemoryDescriptor* shmDescriptorList = 0;
68static int fakeSharedMemoryIdentifier = 0;
69
70static FakeSharedMemoryDescriptor* findBySharedMemoryIdentifier(int sharedMemoryIdentifier)
71{
72 FakeSharedMemoryDescriptor* descriptorPtr = shmDescriptorList;
73
74 while (descriptorPtr) {
75 if (descriptorPtr->sharedMemoryIdentifier == sharedMemoryIdentifier)
76 break;
77 descriptorPtr = descriptorPtr->next;
78 }
79 return descriptorPtr;
80}
81
82static FakeSharedMemoryDescriptor* findBySharedMemoryAddress(const void* mmapedAddress)
83{
84 FakeSharedMemoryDescriptor* descriptorPtr = shmDescriptorList;
85
86 while (descriptorPtr) {
87 if (descriptorPtr->mmapedAddress == mmapedAddress)
88 break;
89
90 descriptorPtr = descriptorPtr->next;
91 }
92 return descriptorPtr;
93}

95static Boolean shim_disabled(void)
96{
97 static Boolean isFakeSHMDisabled;
98
99 static dispatch_once_t once;
100 dispatch_once(&once, ^() {
101 Boolean keyExistsAndHasValidFormat = false;
102 Boolean prefValue = CFPreferencesGetAppBooleanValue(CFSTR("WebKitDisableFakeSYSVSHM"), kCFPreferencesCurrentApplication, &keyExistsAndHasValidFormat);
103
104 if (keyExistsAndHasValidFormat && prefValue)
105 isFakeSHMDisabled = true;
106 else if (sandbox_check(getpid(), NULL, SANDBOX_FILTER_NONE) == 1)
107 isFakeSHMDisabled = false; // Sandboxed
108 else
109 isFakeSHMDisabled = true; // Not Sandboxed
110
111 });
112
113 return isFakeSHMDisabled;
114}

116static int shim_shmdt(const void* sharedAddress)
117{
118 if (shim_disabled())
119 return shmdt(sharedAddress);
120
121 FakeSharedMemoryDescriptor* descriptorPtr = findBySharedMemoryAddress(sharedAddress);
122 if (!descriptorPtr) {
123 errno = EINVAL;
124 return -1;
125 }
126
127 descriptorPtr->referenceCount--;
128 if (!descriptorPtr->referenceCount) {
129 munmap(descriptorPtr->mmapedAddress, descriptorPtr->mmapedSize);
130 descriptorPtr->mmapedAddress = 0;
131 }
132
133 return 0;
134}

136static void* shim_shmat(int sharedMemoryIdentifier, const void* requestedSharedAddress, int shmflg)
137{
138 if (shim_disabled())
139 return shmat(sharedMemoryIdentifier, requestedSharedAddress, shmflg);
140
141 FakeSharedMemoryDescriptor* descriptorPtr = findBySharedMemoryIdentifier(sharedMemoryIdentifier);
142 void* mappedAddress = (void*)-1;
143
144 if (!descriptorPtr) {
145 errno = EINVAL;
146 return mappedAddress;
147 }
148
149 if (descriptorPtr->mmapedAddress) {
150 if (!requestedSharedAddress || requestedSharedAddress == descriptorPtr->mmapedAddress) {
151 mappedAddress = descriptorPtr->mmapedAddress;
152 descriptorPtr->referenceCount++;
153 }
154 } else {
155 descriptorPtr->mmapedSize = (descriptorPtr->requestedSize + PAGE_SIZE) & ~(PAGE_SIZE - 1);
156 mappedAddress = descriptorPtr->mmapedAddress = mmap(const_cast<void*>(requestedSharedAddress),
157 descriptorPtr->mmapedSize, PROT_READ | PROT_WRITE, MAP_ANON | MAP_PRIVATE, -1, 0);
158 descriptorPtr->referenceCount++;
159 }
160
161 return mappedAddress;
162}

164static int shim_shmget(key_t key, size_t requestedSizeOfSharedMemory, int sharedMemoryFlags)
165{
166 if (shim_disabled())
167 return shmget(key, requestedSizeOfSharedMemory, sharedMemoryFlags);
168
169 FakeSharedMemoryDescriptor* descriptorPtr = shmDescriptorList;
170
171 while (descriptorPtr) {
172 // Are we looking for something we've already created?
173 if (descriptorPtr->key == key
174 && descriptorPtr->requestedSize == requestedSizeOfSharedMemory
175 && !((descriptorPtr->sharedMemoryFlags ^ sharedMemoryFlags) & 0777))
176 break;
177 descriptorPtr = descriptorPtr->next;
178 }
179
180 if (!descriptorPtr) {
181 descriptorPtr = (FakeSharedMemoryDescriptor*)malloc(sizeof(FakeSharedMemoryDescriptor));
182 if (!descriptorPtr) {
183 errno = ENOMEM;
184 return -1;
185 }
186 descriptorPtr->key = key;
187 descriptorPtr->requestedSize = requestedSizeOfSharedMemory;
188 descriptorPtr->sharedMemoryFlags = sharedMemoryFlags;
189 descriptorPtr->sharedMemoryIdentifier = ++fakeSharedMemoryIdentifier;
190 descriptorPtr->referenceCount = 0;
191 descriptorPtr->mmapedAddress = 0;
192 descriptorPtr->mmapedSize = 0;
193 descriptorPtr->next = shmDescriptorList;
194 shmDescriptorList = descriptorPtr;
195 }
196 return descriptorPtr->sharedMemoryIdentifier;
197}

199static int shim_shmctl(int sharedMemoryIdentifier, int cmd, struct shmid_ds* outputDescriptor)
200{
201 if (shim_disabled())
202 return shmctl(sharedMemoryIdentifier, cmd, outputDescriptor);
203
204 FakeSharedMemoryDescriptor* descriptorPtr = findBySharedMemoryIdentifier(sharedMemoryIdentifier);
205
206 if (!descriptorPtr) {
207 errno = EINVAL;
208 return -1;
209 }
210
211 switch (cmd) {
212 case IPC_SET:
213 case IPC_RMID:
214 errno = EPERM;
215 return -1;
216
217 case IPC_STAT:
218 outputDescriptor->shm_perm.cuid = outputDescriptor->shm_perm.uid = getuid();
219 outputDescriptor->shm_perm.cgid = outputDescriptor->shm_perm.gid = getgid();
220 outputDescriptor->shm_perm.mode = descriptorPtr->sharedMemoryFlags & 0777;
221
222 outputDescriptor->shm_segsz = descriptorPtr->requestedSize;
223
224 outputDescriptor->shm_cpid = outputDescriptor->shm_lpid = getpid();
225
226 outputDescriptor->shm_nattch = descriptorPtr->referenceCount;
227
228 outputDescriptor->shm_ctime = outputDescriptor->shm_atime = outputDescriptor->shm_dtime = time(0);
229
230 return 0;
231 }
232
233 errno = EINVAL;
234 return -1;
235}

242static CFComparisonResult shimCFStringCompare(CFStringRef a, CFStringRef b, CFStringCompareFlags options)
243{
244 if (pluginProcessShimCallbacks.stringCompare) {
245 CFComparisonResult result;
246 // FIXME (see rdar://72897291): Work around a Clang bug where __builtin_return_address()
247 // sometimes gives us a signed pointer, and sometimes does not.
248 void* returnAddress = __builtin_return_address(0);
249#if CPU(ARM64E)
250 returnAddress = ptrauth_strip(returnAddress, ptrauth_key_process_dependent_code);
251#endif
252 if (pluginProcessShimCallbacks.stringCompare(a, b, options, returnAddress, result))
253 return result;
254 }
255
256 return CFStringCompare(a, b, options);
257}
