43namespace API {
44
45class Data : public ObjectImpl<API::Object::Type::Data> {
46public:
47 using FreeDataFunction = void (*)(unsigned char*, const void* context);
48
49 static Ref<Data> createWithoutCopying(const unsigned char* bytes, size_t size, FreeDataFunction freeDataFunction, const void* context)
50 {
51 return adoptRef(*new Data(bytes, size, freeDataFunction, context));
52 }
53
54 static Ref<Data> create(const unsigned char* bytes, size_t size)
55 {
56 unsigned char *copiedBytes = nullptr;
57
58 if (size) {
59 copiedBytes = static_cast<unsigned char*>(fastMalloc(size));
60 memcpy(copiedBytes, bytes, size);
61 }
62
63 return createWithoutCopying(copiedBytes, size, [] (unsigned char* bytes, const void*) {
64 if (bytes)
65 fastFree(static_cast<void*>(bytes));
66 }, nullptr);
67 }
68
69 static Ref<Data> create(const Vector<unsigned char>& buffer)
70 {
71 return create(buffer.data(), buffer.size());
72 }
73
74 static Ref<Data> create(Vector<unsigned char>&& buffer)
75 {
76 auto bufferSize = buffer.size();
77 auto bufferPointer = buffer.releaseBuffer().leakPtr();
78 return createWithoutCopying(bufferPointer, bufferSize, [] (unsigned char* bytes, const void*) {
79 if (bytes)
80 WTF::VectorMalloc::free(bytes);
81 }, nullptr);
82 }
83
84 static RefPtr<Data> create(const IPC::DataReference&);
85
86#if PLATFORM(COCOA)
87 static Ref<Data> createWithoutCopying(RetainPtr<NSData>);
88#endif
89
90 ~Data()
91 {
92 m_freeDataFunction(const_cast<unsigned char*>(m_bytes), m_context);
93 }
94
95 const unsigned char* bytes() const { return m_bytes; }
96 size_t size() const { return m_size; }
97
98 IPC::DataReference dataReference() const { return IPC::DataReference(m_bytes, m_size); }
99
100 void encode(IPC::Encoder&) const;
101 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, RefPtr<API::Object>&);
102
103private:
104 Data(const unsigned char* bytes, size_t size, FreeDataFunction freeDataFunction, const void* context)
105 : m_bytes(bytes)
106 , m_size(size)
107 , m_freeDataFunction(freeDataFunction)
108 , m_context(context)
109 {
110 }
111
112 const unsigned char* m_bytes { nullptr };
113 size_t m_size { 0 };
114
115 FreeDataFunction m_freeDataFunction { nullptr };
116 const void* m_context { nullptr };
117};
