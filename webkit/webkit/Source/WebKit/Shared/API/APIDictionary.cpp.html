34Ref<Dictionary> Dictionary::create()
35{
36 return create({ });
37}
38
39Ref<Dictionary> Dictionary::create(MapType map)
40{
41 return adoptRef(*new Dictionary(WTFMove(map)));
42}
43
44Dictionary::Dictionary(MapType map)
45 : m_map(WTFMove(map))
46{
47}
48
49Dictionary::~Dictionary()
50{
51}
52
53Ref<Array> Dictionary::keys() const
54{
55 if (m_map.isEmpty())
56 return API::Array::create();
57
58 Vector<RefPtr<API::Object>> keys;
59 keys.reserveInitialCapacity(m_map.size());
60
61 for (const auto& key : m_map.keys())
62 keys.uncheckedAppend(API::String::create(key));
63
64 return API::Array::create(WTFMove(keys));
65}
66
67bool Dictionary::add(const WTF::String& key, RefPtr<API::Object>&& item)
68{
69 MapType::AddResult result = m_map.add(key, WTFMove(item));
70 return result.isNewEntry;
71}
72
73bool Dictionary::set(const WTF::String& key, RefPtr<API::Object>&& item)
74{
75 MapType::AddResult result = m_map.set(key, WTFMove(item));
76 return result.isNewEntry;
77}
78
79void Dictionary::remove(const WTF::String& key)
80{
81 m_map.remove(key);
82}
