26#ifndef APIDictionary_h
27#define APIDictionary_h
28
29#include "APIObject.h"
30#include <wtf/HashMap.h>
31#include <wtf/text/StringHash.h>
32#include <wtf/text/WTFString.h>
33
34namespace API {
35
36class Array;
37
38class Dictionary final : public ObjectImpl<Object::Type::Dictionary> {
39public:
40 typedef HashMap<WTF::String, RefPtr<Object>> MapType;
41
42 static Ref<Dictionary> create();
43 static Ref<Dictionary> create(MapType);
44
45 virtual ~Dictionary();
46
47 template<typename T>
48 T* get(const WTF::String& key) const
49 {
50 RefPtr<Object> item = m_map.get(key);
51 if (!item)
52 return 0;
53
54 if (item->type() != T::APIType)
55 return 0;
56
57 return static_cast<T*>(item.get());
58 }
59
60 Object* get(const WTF::String& key) const
61 {
62 return m_map.get(key);
63 }
64
65 Object* get(const WTF::String& key, bool& exists) const
66 {
67 auto it = m_map.find(key);
68 exists = it != m_map.end();
69 if (!exists)
70 return nullptr;
71
72 return it->value.get();
73 }
74
75 Ref<Array> keys() const;
76
77 bool add(const WTF::String& key, RefPtr<Object>&&);
78 bool set(const WTF::String& key, RefPtr<Object>&&);
79 void remove(const WTF::String& key);
80
81 size_t size() const { return m_map.size(); }
82
83 const MapType& map() const { return m_map; }
84
85private:
86 explicit Dictionary(MapType);
87
88 MapType m_map;
89};
90
91} // namespace API
92
93#endif // APIDictionary_h
