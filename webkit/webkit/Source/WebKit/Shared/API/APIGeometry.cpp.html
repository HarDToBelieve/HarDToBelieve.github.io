33namespace API {
34
35void Point::encode(IPC::Encoder& encoder) const
36{
37 encoder << m_point.x;
38 encoder << m_point.y;
39}
40
41bool Point::decode(IPC::Decoder& decoder, RefPtr<API::Object>& result)
42{
43 WKPoint point;
44 if (!decoder.decode(point.x))
45 return false;
46 if (!decoder.decode(point.y))
47 return false;
48
49 result = Point::create(point);
50 return true;
51}

73void Rect::encode(IPC::Encoder& encoder) const
74{
75 encoder << m_rect.origin.x;
76 encoder << m_rect.origin.y;
77 encoder << m_rect.size.width;
78 encoder << m_rect.size.height;
79}
80
81bool Rect::decode(IPC::Decoder& decoder, RefPtr<API::Object>& result)
82{
83 WKRect rect;
84 if (!decoder.decode(rect.origin.x))
85 return false;
86 if (!decoder.decode(rect.origin.y))
87 return false;
88 if (!decoder.decode(rect.size.width))
89 return false;
90 if (!decoder.decode(rect.size.height))
91 return false;
92
93 result = Rect::create(rect);
94 return true;
95}
