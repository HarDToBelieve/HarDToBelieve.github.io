26#ifndef APINumber_h
27#define APINumber_h
28
29#include "APIObject.h"
30#include "Decoder.h"
31#include "Encoder.h"
32
33namespace API {
34
35template<typename NumberType, API::Object::Type APIObjectType>
36class Number : public ObjectImpl<APIObjectType> {
37public:
38 static Ref<Number> create(NumberType value)
39 {
40 return adoptRef(*new Number(value));
41 }
42
43 NumberType value() const { return m_value; }
44
45 void encode(IPC::Encoder& encoder) const
46 {
47 encoder << m_value;
48 }
49
50 static WARN_UNUSED_RETURN bool decode(IPC::Decoder& decoder, RefPtr<Object>& result)
51 {
52 NumberType value;
53 if (!decoder.decode(value))
54 return false;
55
56 result = Number::create(value);
57 return true;
58 }
59
60private:
61 explicit Number(NumberType value)
62 : m_value(value)
63 {
64 }
65
66 const NumberType m_value;
67};
68
69typedef Number<bool, API::Object::Type::Boolean> Boolean;
70typedef Number<double, API::Object::Type::Double> Double;
71typedef Number<uint64_t, API::Object::Type::UInt64> UInt64;
72typedef Number<int64_t, API::Object::Type::Int64> Int64;
73
74} // namespace API
75
76#endif // APINumber_h
