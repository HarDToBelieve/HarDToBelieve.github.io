264template <Object::Type ArgumentType>
265class ObjectImpl : public Object {
266public:
267 static const Type APIType = ArgumentType;
268
269 virtual ~ObjectImpl()
270 {
271 }
272
273protected:
274 friend class Object;
275
276 ObjectImpl()
277 {
278 }
279
280 Type type() const override { return APIType; }
281
282#if DELEGATE_REF_COUNTING_TO_COCOA
283 void* operator new(size_t size) { return newObject(size, APIType); }
284 void* operator new(size_t, void* value) { return value; }
285#endif
286};
