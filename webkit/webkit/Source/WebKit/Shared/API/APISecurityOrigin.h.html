26#pragma once
27
28#include "APIObject.h"
29#include <WebCore/SecurityOrigin.h>
30#include <wtf/Ref.h>
31
32namespace API {
33
34class SecurityOrigin : public API::ObjectImpl<API::Object::Type::SecurityOrigin> {
35public:
36 static Ref<SecurityOrigin> createFromString(const WTF::String& string)
37 {
38 return adoptRef(*new SecurityOrigin(WebCore::SecurityOriginData::fromURL(WTF::URL(WTF::URL(), string))));
39 }
40
41 static Ref<SecurityOrigin> create(const WTF::String& protocol, const WTF::String& host, std::optional<uint16_t> port)
42 {
43 return adoptRef(*new SecurityOrigin({ protocol, host, port }));
44 }
45
46 static Ref<SecurityOrigin> create(const WebCore::SecurityOrigin& securityOrigin)
47 {
48 return adoptRef(*new SecurityOrigin(securityOrigin.data().isolatedCopy()));
49 }
50
51 const WebCore::SecurityOriginData& securityOrigin() const { return m_securityOrigin; }
52
53private:
54 SecurityOrigin(WebCore::SecurityOriginData&& securityOrigin)
55 : m_securityOrigin(WTFMove(securityOrigin))
56 {
57 }
58
59 WebCore::SecurityOriginData m_securityOrigin;
60};
61
62}
