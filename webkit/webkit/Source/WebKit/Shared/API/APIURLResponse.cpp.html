31namespace API {

33
34URLResponse::URLResponse(const WebCore::ResourceResponse& response)
35 : m_response(response)
36{
37}
38
39void URLResponse::encode(IPC::Encoder& encoder) const
40{
41 encoder << resourceResponse();
42}
43
44bool URLResponse::decode(IPC::Decoder& decoder, RefPtr<Object>& result)
45{
46 ResourceResponse response;
47 if (!decoder.decode(response))
48 return false;
49
50 result = create(response);
51 return true;
52}
