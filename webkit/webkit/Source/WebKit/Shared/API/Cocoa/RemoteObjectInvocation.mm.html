34RemoteObjectInvocation::RemoteObjectInvocation()
35{
36}
37
38RemoteObjectInvocation::RemoteObjectInvocation(const String& interfaceIdentifier, RefPtr<API::Dictionary>&& encodedInvocation, std::unique_ptr<ReplyInfo>&& replyInfo)
39 : m_interfaceIdentifier(interfaceIdentifier)
40 , m_encodedInvocation(WTFMove(encodedInvocation))
41 , m_replyInfo(WTFMove(replyInfo))
42{
43}
44
45void RemoteObjectInvocation::encode(IPC::Encoder& encoder) const
46{
47 encoder << m_interfaceIdentifier;
48 UserData::encode(encoder, m_encodedInvocation.get());
49 if (!m_replyInfo) {
50 encoder << false;
51 return;
52 }
53
54 encoder << true;
55 encoder << m_replyInfo->replyID;
56 encoder << m_replyInfo->blockSignature;
57}
58
59bool RemoteObjectInvocation::decode(IPC::Decoder& decoder, RemoteObjectInvocation& result)
60{
61 if (!decoder.decode(result.m_interfaceIdentifier))
62 return false;
63
64 RefPtr<API::Object> encodedInvocation;
65 if (!UserData::decode(decoder, encodedInvocation))
66 return false;
67
68 if (!encodedInvocation || encodedInvocation->type() != API::Object::Type::Dictionary)
69 return false;
70
71 result.m_encodedInvocation = static_cast<API::Dictionary*>(encodedInvocation.get());
72
73 bool hasReplyInfo;
74 if (!decoder.decode(hasReplyInfo))
75 return false;
76
77 if (hasReplyInfo) {
78 uint64_t replyID;
79 if (!decoder.decode(replyID))
80 return false;
81
82 String blockSignature;
83 if (!decoder.decode(blockSignature))
84 return false;
85
86 result.m_replyInfo = makeUnique<ReplyInfo>(replyID, WTFMove(blockSignature));
87 }
88
89 return true;
90}
