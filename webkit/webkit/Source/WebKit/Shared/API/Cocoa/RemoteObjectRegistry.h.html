35OBJC_CLASS _WKRemoteObjectRegistry;
36
37namespace IPC {
38class MessageSender;
39}
40
41namespace WebKit {
42
43class RemoteObjectInvocation;
44class UserData;
45
46class RemoteObjectRegistry : public IPC::MessageReceiver {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 virtual ~RemoteObjectRegistry();
50
51 virtual void sendInvocation(const RemoteObjectInvocation&);
52 void sendReplyBlock(uint64_t replyID, const UserData& blockInvocation);
53 void sendUnusedReply(uint64_t replyID);
54
55protected:
56 explicit RemoteObjectRegistry(_WKRemoteObjectRegistry *);
57
58private:
59 virtual std::unique_ptr<ProcessThrottler::BackgroundActivity> backgroundActivity(ASCIILiteral) { return nullptr; }
60 virtual IPC::MessageSender& messageSender() = 0;
61 virtual uint64_t messageDestinationID() = 0;
62
63 // IPC::MessageReceiver
64 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
65
66 // Message handlers
67 void invokeMethod(const RemoteObjectInvocation&);
68 void callReplyBlock(uint64_t replyID, const UserData& blockInvocation);
69 void releaseUnusedReplyBlock(uint64_t replyID);
70
71 WeakObjCPtr<_WKRemoteObjectRegistry> m_remoteObjectRegistry;
72 HashMap<uint64_t, std::unique_ptr<ProcessThrottler::BackgroundActivity>> m_pendingReplies;
73};
