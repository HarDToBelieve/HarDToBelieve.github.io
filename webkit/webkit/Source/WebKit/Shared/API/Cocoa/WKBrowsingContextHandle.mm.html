26#import "config.h"
27#import "WKBrowsingContextHandleInternal.h"
28
29#import "WebPage.h"
30#import "WebPageProxy.h"
31#import <wtf/HashFunctions.h>
32
33@implementation WKBrowsingContextHandle
34
35- (id)_initWithPageProxy:(NakedRef<WebKit::WebPageProxy>)page
36{
37 return [self _initWithPageProxyID:page->identifier() andWebPageID:page->webPageID()];
38}
39
40- (id)_initWithPage:(NakedRef<WebKit::WebPage>)page
41{
42 return [self _initWithPageProxyID:page->webPageProxyIdentifier() andWebPageID:page->identifier()];
43}
44
45- (id)_initWithPageProxyID:(WebKit::WebPageProxyIdentifier)pageProxyID andWebPageID:(WebCore::PageIdentifier)webPageID
46{
47 if (!(self = [super init]))
48 return nil;
49
50 _pageProxyID = pageProxyID;
51 _webPageID = webPageID;
52
53 return self;
54}
55
56- (NSUInteger)hash
57{
58 return computeHash(_pageProxyID, _webPageID);
59}
60
61- (BOOL)isEqual:(id)object
62{
63 if (![object isKindOfClass:[WKBrowsingContextHandle class]])
64 return NO;
65
66 return _pageProxyID == static_cast<WKBrowsingContextHandle *>(object)->_pageProxyID && _webPageID == static_cast<WKBrowsingContextHandle *>(object)->_webPageID;
67}
68
69- (void)encodeWithCoder:(NSCoder *)coder
70{
71 [coder encodeInt64:_pageProxyID.toUInt64() forKey:@"pageProxyID"];
72 [coder encodeInt64:_webPageID.toUInt64() forKey:@"webPageID"];
73}
74
75- (id)initWithCoder:(NSCoder *)coder
76{
77 if (!(self = [super init]))
78 return nil;
79
80 _pageProxyID = makeObjectIdentifier<WebKit::WebPageProxyIdentifierType>([coder decodeInt64ForKey:@"pageProxyID"]);
81 _webPageID = makeObjectIdentifier<WebCore::PageIdentifierType>([coder decodeInt64ForKey:@"webPageID"]);
82
83 return self;
84}
85
86+ (BOOL)supportsSecureCoding
87{
88 return YES;
89}
90
91- (id)copyWithZone:(NSZone *)zone
92{
93 return [[WKBrowsingContextHandle allocWithZone:zone] _initWithPageProxyID:_pageProxyID andWebPageID:_webPageID];
94}
95
96- (NSString *)description
97{
98 return [NSString stringWithFormat:@"<%@: %p; pageProxyID = %llu; webPageID = %llu>", NSStringFromClass(self.class), self, _pageProxyID.toUInt64(), _webPageID.toUInt64()];
99}
100
101@end
