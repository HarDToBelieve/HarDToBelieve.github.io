39- (instancetype)_initWithObjectRegistry:(_WKRemoteObjectRegistry *)objectRegistry interface:(_WKRemoteObjectInterface *)interface
40{
41 if (!(self = [super init]))
42 return nil;
43
44 _objectRegistry = objectRegistry;
45 _interface = interface;
46
47 return self;
48}
49
50- (BOOL)conformsToProtocol:(Protocol *)protocol
51{
52 return [super conformsToProtocol:protocol] || protocol_conformsToProtocol([_interface protocol], protocol);
53}
54
55static const char* methodArgumentTypeEncodingForSelector(Protocol *protocol, SEL selector)
56{
57 // First look at required methods.
58 struct objc_method_description method = protocol_getMethodDescription(protocol, selector, YES, YES);
59 if (method.name)
60 return method.types;
61
62 // Then look at optional methods.
63 method = protocol_getMethodDescription(protocol, selector, NO, YES);
64 if (method.name)
65 return method.types;
66
67 return nullptr;
68}
69
70- (NSMethodSignature *)methodSignatureForSelector:(SEL)selector
71{
72 if (!selector)
73 return nil;
74
75 Protocol *protocol = [_interface protocol];
76
77 const char* types = methodArgumentTypeEncodingForSelector(protocol, selector);
78 if (!types) {
79 // We didn't find anything the protocol, fall back to the superclass.
80 return [super methodSignatureForSelector:selector];
81 }
82
83 return [NSMethodSignature signatureWithObjCTypes:types];
84}

89}
