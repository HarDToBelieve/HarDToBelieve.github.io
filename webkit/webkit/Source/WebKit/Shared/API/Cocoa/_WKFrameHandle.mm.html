26#import "config.h"
27#import "_WKFrameHandleInternal.h"
28
29#import <WebCore/FrameIdentifier.h>
30#import <WebCore/WebCoreObjCExtras.h>
31
32@implementation _WKFrameHandle
33
34- (void)dealloc
35{
36 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(_WKFrameHandle.class, self))
37 return;
38
39 _frameHandle->~FrameHandle();
40
41 [super dealloc];
42}
43
44- (BOOL)isEqual:(id)object
45{
46 if (object == self)
47 return YES;
48
49 if (![object isKindOfClass:[_WKFrameHandle class]])
50 return NO;
51
52 return _frameHandle->frameID() == ((_WKFrameHandle *)object)->_frameHandle->frameID();
53}
54
55- (NSUInteger)hash
56{
57 return _frameHandle->frameID().toUInt64();
58}
59
60- (uint64_t)frameID
61{
62 return _frameHandle->frameID().toUInt64();
63}
64
65#pragma mark NSCopying protocol implementation
66
67- (id)copyWithZone:(NSZone *)zone
68{
69 return [self retain];
70}
71
72#pragma mark NSSecureCoding protocol implementation
73
74+ (BOOL)supportsSecureCoding
75{
76 return YES;
77}
78
79- (id)initWithCoder:(NSCoder *)decoder
80{
81 if (!(self = [super init]))
82 return nil;
83
84 NSNumber *frameID = [decoder decodeObjectOfClass:[NSNumber class] forKey:@"frameID"];
85 if (![frameID isKindOfClass:[NSNumber class]]) {
86 [self release];
87 return nil;
88 }
89
90 API::Object::constructInWrapper<API::FrameHandle>(self, makeObjectIdentifier<WebCore::FrameIdentifierType>(frameID.unsignedLongLongValue), false);
91
92 return self;
93}
94
95- (void)encodeWithCoder:(NSCoder *)coder
96{
97 [coder encodeObject:@(self.frameID) forKey:@"frameID"];
98}
99
100#pragma mark WKObject protocol implementation
101
102- (API::Object&)_apiObject
103{
104 return *_frameHandle;
105}
106
107@end
