87{

141- (void)_sendInvocation:(NSInvocation *)invocation interface:(_WKRemoteObjectInterface *)interface
142{
143 std::unique_ptr<WebKit::RemoteObjectInvocation::ReplyInfo> replyInfo;
144
145 NSMethodSignature *methodSignature = invocation.methodSignature;
146 for (NSUInteger i = 0, count = methodSignature.numberOfArguments; i < count; ++i) {
147 const char *type = [methodSignature getArgumentTypeAtIndex:i];
148
149 if (strcmp(type, "@?"))
150 continue;
151
152 if (replyInfo)
153 [NSException raise:NSInvalidArgumentException format:@"Only one reply block is allowed per message send. (%s)", sel_getName(invocation.selector)];
154
155 id replyBlock = nullptr;
156 [invocation getArgument:&replyBlock atIndex:i];
157 if (!replyBlock)
158 [NSException raise:NSInvalidArgumentException format:@"A NULL reply block was passed into a message. (%s)", sel_getName(invocation.selector)];
159
160 const char* replyBlockSignature = _Block_signature((__bridge void*)replyBlock);
161
162 if (strcmp([NSMethodSignature signatureWithObjCTypes:replyBlockSignature].methodReturnType, "v"))
163 [NSException raise:NSInvalidArgumentException format:@"Return value of block argument must be 'void'. (%s)", sel_getName(invocation.selector)];
164
165 replyInfo = makeUnique<WebKit::RemoteObjectInvocation::ReplyInfo>(generateReplyIdentifier(), replyBlockSignature);
166
167 // Replace the block object so we won't try to encode it.
168 id null = nullptr;
169 [invocation setArgument:&null atIndex:i];
170
171 ASSERT(!_pendingReplies.contains(replyInfo->replyID));
172 _pendingReplies.add(replyInfo->replyID, PendingReply(interface, invocation.selector, replyBlock));
173 }
174
175 RetainPtr<WKRemoteObjectEncoder> encoder = adoptNS([[WKRemoteObjectEncoder alloc] init]);
176
177 [encoder encodeObject:invocation forKey:invocationKey];
178
179 if (!_remoteObjectRegistry)
180 return;
181
182 _remoteObjectRegistry->sendInvocation(WebKit::RemoteObjectInvocation(interface.identifier, [encoder rootObjectDictionary], WTFMove(replyInfo)));
183}

190static bool blockSignaturesAreCompatible(const String& wire, const String& local)
191{
192 if (local == wire)
193 return true;
194
195 if (local.length() != wire.length())
196 return false;
197
198 unsigned length = local.length();
199 for (unsigned i = 0; i < length; i++) {
200 char localType = local[i];
201 char wireType = wire[i];
202
203 if (localType != wireType) {
204 // `bool` and `signed char` are interchangeable.
205 if (strchr("Bc", localType) && strchr("Bc", wireType))
206 continue;
207 return false;
208 }
209 }
210 return true;
211}

213static String replyBlockSignature(Protocol *protocol, SEL selector, NSUInteger blockIndex)
214{
215 // Required, non-inherited method:
216 const char* methodTypeEncoding = _protocol_getMethodTypeEncoding(protocol, selector, true, true);
217 // @optional, non-inherited method:
218 if (!methodTypeEncoding)
219 methodTypeEncoding = _protocol_getMethodTypeEncoding(protocol, selector, false, true);
220
221 ASSERT(methodTypeEncoding);
222 if (!methodTypeEncoding)
223 return { };
224
225 NSMethodSignature *targetMethodSignature = [NSMethodSignature signatureWithObjCTypes:methodTypeEncoding];
226 ASSERT(targetMethodSignature);
227 if (!targetMethodSignature)
228 return { };
229
230 return [targetMethodSignature _signatureForBlockAtArgumentIndex:blockIndex]._typeString.UTF8String;
231}

233- (void)_invokeMethod:(const WebKit::RemoteObjectInvocation&)remoteObjectInvocation
234{
235 auto& interfaceIdentifier = remoteObjectInvocation.interfaceIdentifier();
236 auto* encodedInvocation = remoteObjectInvocation.encodedInvocation();
237
238 auto interfaceAndObject = _exportedObjects.get(interfaceIdentifier);
239 if (!interfaceAndObject.second) {
240 NSLog(@"Did not find a registered object for the interface \"%@\"", (NSString *)interfaceIdentifier);
241 return;
242 }
243
244 RetainPtr<_WKRemoteObjectInterface> interface = interfaceAndObject.second;
245
246 auto decoder = adoptNS([[WKRemoteObjectDecoder alloc] initWithInterface:interface.get() rootObjectDictionary:encodedInvocation replyToSelector:nullptr]);
247
248 NSInvocation *invocation = nil;
249
250 @try {
251 invocation = [decoder decodeObjectOfClass:[NSInvocation class] forKey:invocationKey];
252 } @catch (NSException *exception) {
253 NSLog(@"Exception caught during decoding of message: %@", exception);
254 }
255
256 NSMethodSignature *methodSignature = invocation.methodSignature;
257 auto* replyInfo = remoteObjectInvocation.replyInfo();
258
259 // Look for the block argument (if any).
260 for (NSUInteger i = 0, count = methodSignature.numberOfArguments; i < count; ++i) {
261 const char *type = [methodSignature getArgumentTypeAtIndex:i];
262
263 if (strcmp(type, "@?"))
264 continue;
265
266 // We found the block.
267 // If the wire had no block signature but we expect one, we drop the message.
268 if (!replyInfo) {
269 NSLog(@"_invokeMethod: Expected reply block, but none provided");
270 return;
271 }
272
273 String wireBlockSignature = [NSMethodSignature signatureWithObjCTypes:replyInfo->blockSignature.utf8().data()]._typeString.UTF8String;
274 String expectedBlockSignature = replyBlockSignature([interface protocol], invocation.selector, i);
275
276 if (expectedBlockSignature.isNull()) {
277 NSLog(@"_invokeMethod: Failed to validate reply block signature: could not find local signature");
278 ASSERT_NOT_REACHED();
279 return;
280 }
281
282 if (!blockSignaturesAreCompatible(wireBlockSignature, expectedBlockSignature)) {
283 NSLog(@"_invokeMethod: Failed to validate reply block signature: %s != %s", wireBlockSignature.utf8().data(), expectedBlockSignature.utf8().data());
284 ASSERT_NOT_REACHED();
285 return;
286 }

291 class ReplyBlockCallChecker : public WTF::ThreadSafeRefCounted<ReplyBlockCallChecker> {
292 public:
293 static Ref<ReplyBlockCallChecker> create(_WKRemoteObjectRegistry *registry, uint64_t replyID) { return adoptRef(*new ReplyBlockCallChecker(registry, replyID)); }
294
295 ~ReplyBlockCallChecker()
296 {
297 if (m_didCallReplyBlock)
298 return;
299
300 // FIXME: Instead of not sending anything when the remote object registry is null, we should
301 // keep track of all reply block checkers and invalidate them (sending the unused reply message) in
302 // -[_WKRemoteObjectRegistry _invalidate].
303 if (!m_remoteObjectRegistry->_remoteObjectRegistry)
304 return;
305
306 m_remoteObjectRegistry->_remoteObjectRegistry->sendUnusedReply(m_replyID);
307 }
308
309 void didCallReplyBlock() { m_didCallReplyBlock = true; }
310
311 private:
312 ReplyBlockCallChecker(_WKRemoteObjectRegistry *registry, uint64_t replyID)
313 : m_remoteObjectRegistry(registry)
314 , m_replyID(replyID)
315 {
316 }
317
318 RetainPtr<_WKRemoteObjectRegistry> m_remoteObjectRegistry;
319 uint64_t m_replyID = 0;
320 bool m_didCallReplyBlock = false;
321 };
322
323 RefPtr<ReplyBlockCallChecker> checker = ReplyBlockCallChecker::create(self, replyID);
324 id replyBlock = __NSMakeSpecialForwardingCaptureBlock(wireBlockSignature.utf8().data(), [interface, remoteObjectRegistry, replyID, checker](NSInvocation *invocation) {
325 auto encoder = adoptNS([[WKRemoteObjectEncoder alloc] init]);
326 [encoder encodeObject:invocation forKey:invocationKey];
327
328 if (remoteObjectRegistry->_remoteObjectRegistry)
329 remoteObjectRegistry->_remoteObjectRegistry->sendReplyBlock(replyID, WebKit::UserData([encoder rootObjectDictionary]));
330 checker->didCallReplyBlock();
331 });
332
333 [invocation setArgument:&replyBlock atIndex:i];
334
335 // Make sure that the block won't be destroyed before the invocation.
336 objc_setAssociatedObject(invocation, replyBlockKey, replyBlock, OBJC_ASSOCIATION_RETAIN);
337 [replyBlock release];
338
339 break;
340 }
341
342 invocation.target = interfaceAndObject.first.get();
343
344 @try {
345 [invocation invoke];
346 } @catch (NSException *exception) {
347 NSLog(@"%@: Warning: Exception caught during invocation of received message, dropping incoming message .\nException: %@", self, exception);
348 }
349}

351- (void)_callReplyWithID:(uint64_t)replyID blockInvocation:(const WebKit::UserData&)blockInvocation
352{
353 auto encodedInvocation = blockInvocation.object();
354 if (!encodedInvocation || encodedInvocation->type() != API::Object::Type::Dictionary)
355 return;
356
357 auto it = _pendingReplies.find(replyID);
358 if (it == _pendingReplies.end())
359 return;
360
361 auto pendingReply = it->value;
362 _pendingReplies.remove(it);
363
364 auto decoder = adoptNS([[WKRemoteObjectDecoder alloc] initWithInterface:pendingReply.interface.get() rootObjectDictionary:static_cast<API::Dictionary*>(encodedInvocation) replyToSelector:pendingReply.selector]);
365
366 NSInvocation *replyInvocation = nil;
367
368 @try {
369 replyInvocation = [decoder decodeObjectOfClass:[NSInvocation class] forKey:invocationKey];
370 } @catch (NSException *exception) {
371 NSLog(@"Exception caught during decoding of reply: %@", exception);
372 return;
373 }
374
375 [replyInvocation setTarget:pendingReply.block.get()];
376 [replyInvocation invoke];
377}
