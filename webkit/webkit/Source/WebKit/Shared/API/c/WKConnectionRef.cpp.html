37void WKConnectionSetConnectionClient(WKConnectionRef connectionRef, const WKConnectionClientBase* wkClient)
38{
39 WebKit::toImpl(connectionRef)->initializeConnectionClient(wkClient);
40}

42void WKConnectionPostMessage(WKConnectionRef connectionRef, WKStringRef messageNameRef, WKTypeRef messageBodyRef)
43{
44 WebKit::toImpl(connectionRef)->postMessage(WebKit::toImpl(messageNameRef)->string(), WebKit::toImpl(messageBodyRef));
45}
