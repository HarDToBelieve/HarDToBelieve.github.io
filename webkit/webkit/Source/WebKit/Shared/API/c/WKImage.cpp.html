32WKTypeID WKImageGetTypeID(void)
33{
34 return WebKit::toAPI(WebKit::WebImage::APIType);
35}
36
37WKImageRef WKImageCreate(WKSize size, WKImageOptions options)
38{
39 auto webImage = WebKit::WebImage::create(WebKit::toIntSize(size), WebKit::toImageOptions(options));
40 return toAPI(webImage.leakRef());
41}
42
43WKSize WKImageGetSize(WKImageRef imageRef)
44{
45 return WebKit::toAPI(WebKit::toImpl(imageRef)->size());
46}
