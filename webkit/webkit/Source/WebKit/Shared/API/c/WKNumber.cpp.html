37WKBooleanRef WKBooleanCreate(bool value)
38{
39 auto booleanObject = API::Boolean::create(value);
40 return WebKit::toAPI(&booleanObject.leakRef());
41}

43bool WKBooleanGetValue(WKBooleanRef booleanRef)
