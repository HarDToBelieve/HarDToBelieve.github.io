124template<typename T, typename APIType = typename ImplTypeInfo<T>::APIType>
125auto toAPI(T* t) -> APIType
126{
127 return reinterpret_cast<APIType>(API::Object::wrap(t));
128}
129
130template<typename T, typename APIType = typename ImplTypeInfo<T>::APIType>
131auto toAPI(T& t) -> APIType
132{
133 return reinterpret_cast<APIType>(API::Object::wrap(&t));
134}
135
136template<typename T, typename ImplType = typename APITypeInfo<T>::ImplType>
137auto toImpl(T t) -> ImplType*
138{
139 return static_cast<ImplType*>(API::Object::unwrap(static_cast<void*>(const_cast<typename std::remove_const<typename std::remove_pointer<T>::type>::type*>(t))));
140}
141
142template<typename ImplType, typename APIType = typename ImplTypeInfo<ImplType>::APIType>
143class ProxyingRefPtr {
144public:
145 ProxyingRefPtr(RefPtr<ImplType>&& impl)
146 : m_impl(impl)
147 {
148 }
149
150 ProxyingRefPtr(Ref<ImplType>&& impl)
151 : m_impl(WTFMove(impl))
152 {
153 }
154
155 operator APIType() { return toAPI(m_impl.get()); }
156
157private:
158 RefPtr<ImplType> m_impl;
159};

295inline WKEventModifiers toAPI(OptionSet<WebEvent::Modifier> modifiers)
296{
297 WKEventModifiers wkModifiers = 0;
298 if (modifiers.contains(WebEvent::Modifier::ShiftKey))
299 wkModifiers |= kWKEventModifiersShiftKey;
300 if (modifiers.contains(WebEvent::Modifier::ControlKey))
301 wkModifiers |= kWKEventModifiersControlKey;
302 if (modifiers.contains(WebEvent::Modifier::AltKey))
303 wkModifiers |= kWKEventModifiersAltKey;
304 if (modifiers.contains(WebEvent::Modifier::MetaKey))
305 wkModifiers |= kWKEventModifiersMetaKey;
306 if (modifiers.contains(WebEvent::Modifier::CapsLockKey))
307 wkModifiers |= kWKEventModifiersCapsLockKey;
308 return wkModifiers;
309}
