32WKTypeID WKGetTypeID(WKTypeRef typeRef)
33{
34 return WebKit::toAPI(WebKit::toImpl(typeRef)->type());
35}

37WKTypeRef WKRetain(WKTypeRef typeRef)
38{
39 WebKit::toImpl(typeRef)->ref();
40
41 return typeRef;
42}
43
44void WKRelease(WKTypeRef typeRef)
45{
46 WebKit::toImpl(typeRef)->deref();
47}
