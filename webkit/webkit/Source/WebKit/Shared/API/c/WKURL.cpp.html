26#include "config.h"
27#include "WKURL.h"
28
29#include "WKAPICast.h"
30
31WKTypeID WKURLGetTypeID()
32{
33 return WebKit::toAPI(API::URL::APIType);
34}
35
36WKURLRef WKURLCreateWithUTF8CString(const char* string)
37{
38 return WebKit::toAPI(&API::URL::create(String::fromUTF8(string)).leakRef());
39}
40
41WKURLRef WKURLCreateWithBaseURL(WKURLRef baseURL, const char* relative)
42{
43 return WebKit::toAPI(&API::URL::create(WebKit::toImpl(baseURL), String::fromUTF8(relative)).leakRef());
44}
45
46WKStringRef WKURLCopyString(WKURLRef url)
47{
48 return WebKit::toCopiedAPI(WebKit::toImpl(url)->string());
49}
50
51bool WKURLIsEqual(WKURLRef a, WKURLRef b)
52{
53 return API::URL::equals(*WebKit::toImpl(a), *WebKit::toImpl(b));
54}
55
56WKStringRef WKURLCopyHostName(WKURLRef url)
57{
58 return WebKit::toCopiedAPI(WebKit::toImpl(url)->host());
59}
60
61WKStringRef WKURLCopyScheme(WKURLRef url)
62{
63 return WebKit::toCopiedAPI(WebKit::toImpl(url)->protocol());
64}
65
66WK_EXPORT WKStringRef WKURLCopyPath(WKURLRef url)
67{
68 return WebKit::toCopiedAPI(WebKit::toImpl(url)->path());
69}
70
71WKStringRef WKURLCopyLastPathComponent(WKURLRef url)
72{
73 return WebKit::toCopiedAPI(WebKit::toImpl(url)->lastPathComponent());
74}
