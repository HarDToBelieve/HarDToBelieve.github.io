34WKTypeID WKURLRequestGetTypeID()
35{
36 return WebKit::toAPI(API::URLRequest::APIType);
37}
38
39WKURLRequestRef WKURLRequestCreateWithWKURL(WKURLRef url)
40{
41 return WebKit::toAPI(&API::URLRequest::create(URL(URL(), WebKit::toImpl(url)->string())).leakRef());
42}
43
44WKURLRef WKURLRequestCopyURL(WKURLRequestRef requestRef)
45{
46 return WebKit::toCopiedURLAPI(WebKit::toImpl(requestRef)->resourceRequest().url());
47}
48
49WKURLRef WKURLRequestCopyFirstPartyForCookies(WKURLRequestRef requestRef)
50{
51 return WebKit::toCopiedURLAPI(WebKit::toImpl(requestRef)->resourceRequest().firstPartyForCookies());
52}
53
54WKStringRef WKURLRequestCopyHTTPMethod(WKURLRequestRef requestRef)
55{
56 return WebKit::toCopiedAPI(WebKit::toImpl(requestRef)->resourceRequest().httpMethod());
57}
58
59WKURLRequestRef WKURLRequestCopySettingHTTPBody(WKURLRequestRef requestRef, WKDataRef body)
60{
61 WebCore::ResourceRequest requestCopy(WebKit::toImpl(requestRef)->resourceRequest());
62 requestCopy.setHTTPBody(WebCore::FormData::create(WKDataGetBytes(body), WKDataGetSize(body)));
63 return WebKit::toAPI(&API::URLRequest::create(requestCopy).leakRef());
64}

66void WKURLRequestSetDefaultTimeoutInterval(double timeoutInterval)

68 API::URLRequest::setDefaultTimeoutInterval(timeoutInterval);
69}
