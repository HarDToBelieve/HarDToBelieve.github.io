26#include "config.h"
27#include "WKUserContentURLPattern.h"
28
29#include "APIUserContentURLPattern.h"
30#include "WKAPICast.h"
31#include "WKString.h"
32
33WKTypeID WKUserContentURLPatternGetTypeID()
34{
35 return WebKit::toAPI(API::UserContentURLPattern::APIType);
36}
37
38WKUserContentURLPatternRef WKUserContentURLPatternCreate(WKStringRef patternRef)
39{
40 return WebKit::toAPI(&API::UserContentURLPattern::create(WebKit::toImpl(patternRef)->string()).leakRef());
41}
42
43WKStringRef WKUserContentURLPatternCopyHost(WKUserContentURLPatternRef urlPatternRef)
44{
45 return WebKit::toCopiedAPI(WebKit::toImpl(urlPatternRef)->host());
46}
47
48WKStringRef WKUserContentURLPatternCopyScheme(WKUserContentURLPatternRef urlPatternRef)
49{
50 return WebKit::toCopiedAPI(WebKit::toImpl(urlPatternRef)->scheme());
51}
52
53bool WKUserContentURLPatternIsValid(WKUserContentURLPatternRef urlPatternRef)
54{
55 return WebKit::toImpl(urlPatternRef)->isValid();
56}
57
58bool WKUserContentURLPatternMatchesURL(WKUserContentURLPatternRef urlPatternRef, WKURLRef urlRef)
59{
60 return WebKit::toImpl(urlPatternRef)->matchesURL(WebKit::toWTFString(urlRef));
61}
62
63bool WKUserContentURLPatternMatchesSubdomains(WKUserContentURLPatternRef urlPatternRef)
64{
65 return WebKit::toImpl(urlPatternRef)->matchesSubdomains();
66}
