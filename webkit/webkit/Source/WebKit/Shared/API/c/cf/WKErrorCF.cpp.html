32WKErrorRef WKErrorCreateWithCFError(CFErrorRef cfError)
33{
34 return WebKit::toAPI(&API::Error::create(WebCore::ResourceError(cfError)).leakRef());
35}
36
37CFErrorRef WKErrorCopyCFError(CFAllocatorRef alloc, WKErrorRef error)
38{
39 WTF::RetainPtr<CFErrorRef> cfError = WebKit::toImpl(error)->platformError().cfError();
40 return cfError.leakRef();
41}
