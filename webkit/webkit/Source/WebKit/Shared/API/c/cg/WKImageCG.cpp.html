36CGImageRef WKImageCreateCGImage(WKImageRef imageRef)
37{
38 WebKit::WebImage* webImage = WebKit::toImpl(imageRef);
39 if (!webImage)
40 return nullptr;
41
42 return webImage->bitmap().makeCGImageCopy().leakRef();
43}
44
45WKImageRef WKImageCreateFromCGImage(CGImageRef imageRef, WKImageOptions options)
46{
47 if (!imageRef)
48 return nullptr;
49
50 auto nativeImage = WebCore::NativeImage::create(imageRef);
51 WebCore::IntSize imageSize = nativeImage->size();
52 auto webImage = WebKit::WebImage::create(imageSize, WebKit::toImageOptions(options));
53
54 auto graphicsContext = webImage->bitmap().createGraphicsContext();
55 if (!graphicsContext)
56 return nullptr;
57
58 WebCore::FloatRect rect(WebCore::FloatPoint(0, 0), imageSize);
59
60 graphicsContext->clearRect(rect);
61 graphicsContext->drawNativeImage(*nativeImage, imageSize, rect, rect);
62 return toAPI(webImage.leakRef());
63}
