32WKURLRequestRef WKURLRequestCreateWithNSURLRequest(NSURLRequest *urlRequest)
33{
34 if ([urlRequest HTTPBodyStream])
35 return nullptr;
36 RetainPtr<NSURLRequest> copiedURLRequest = adoptNS([urlRequest copy]);
37 return WebKit::toAPI(&API::URLRequest::create(copiedURLRequest.get()).leakRef());
38}
39
40NSURLRequest *WKURLRequestCopyNSURLRequest(WKURLRequestRef urlRequest)
41{
42 return [WebKit::toImpl(urlRequest)->resourceRequest().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody) copy];
43}
