39WKTypeID WKWebArchiveGetTypeID()
40{
41 return WebKit::toAPI(API::WebArchive::APIType);
42}
43
44WKWebArchiveRef WKWebArchiveCreate(WKWebArchiveResourceRef mainResourceRef, WKArrayRef subresourcesRef, WKArrayRef subframeArchivesRef)
45{
46 auto webArchive = API::WebArchive::create(WebKit::toImpl(mainResourceRef), WebKit::toImpl(subresourcesRef), WebKit::toImpl(subframeArchivesRef));
47 return WebKit::toAPI(&webArchive.leakRef());
48}
49
50WKWebArchiveRef WKWebArchiveCreateWithData(WKDataRef dataRef)
51{
52 auto webArchive = API::WebArchive::create(WebKit::toImpl(dataRef));
53 return WebKit::toAPI(&webArchive.leakRef());
54}
55
56WKWebArchiveRef WKWebArchiveCreateFromRange(WKBundleRangeHandleRef rangeHandleRef)
57{
58 auto webArchive = API::WebArchive::create(makeSimpleRange(WebKit::toImpl(rangeHandleRef)->coreRange()));
59 return WebKit::toAPI(&webArchive.leakRef());
60}
61
62WKWebArchiveResourceRef WKWebArchiveCopyMainResource(WKWebArchiveRef webArchiveRef)
63{
64 RefPtr<API::WebArchiveResource> mainResource = WebKit::toImpl(webArchiveRef)->mainResource();
65 return WebKit::toAPI(mainResource.leakRef());
66}
67
68WKArrayRef WKWebArchiveCopySubresources(WKWebArchiveRef webArchiveRef)
69{
70 RefPtr<API::Array> subresources = WebKit::toImpl(webArchiveRef)->subresources();
71 return WebKit::toAPI(subresources.leakRef());
72}
73
74WKArrayRef WKWebArchiveCopySubframeArchives(WKWebArchiveRef webArchiveRef)
75{
76 RefPtr<API::Array> subframeArchives = WebKit::toImpl(webArchiveRef)->subframeArchives();
77 return WebKit::toAPI(subframeArchives.leakRef());
78}
79
80WKDataRef WKWebArchiveCopyData(WKWebArchiveRef webArchiveRef)
81{
82 return WebKit::toAPI(&WebKit::toImpl(webArchiveRef)->data().leakRef());
83}
