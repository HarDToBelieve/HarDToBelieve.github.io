33WKTypeID WKWebArchiveResourceGetTypeID()
34{
35 return WebKit::toAPI(API::WebArchiveResource::APIType);
36}
37
38WKWebArchiveResourceRef WKWebArchiveResourceCreate(WKDataRef dataRef, WKURLRef URLRef, WKStringRef MIMETypeRef, WKStringRef textEncodingRef)
39{
40 auto webArchiveResource = API::WebArchiveResource::create(WebKit::toImpl(dataRef), WebKit::toWTFString(URLRef), WebKit::toWTFString(MIMETypeRef), WebKit::toWTFString(textEncodingRef));
41 return WebKit::toAPI(&webArchiveResource.leakRef());
42}
43
44WKDataRef WKWebArchiveResourceCopyData(WKWebArchiveResourceRef webArchiveResourceRef)
45{
46 return WebKit::toAPI(&WebKit::toImpl(webArchiveResourceRef)->data().leakRef());
47}
48
49WKURLRef WKWebArchiveResourceCopyURL(WKWebArchiveResourceRef webArchiveResourceRef)
50{
51 return WebKit::toCopiedURLAPI(WebKit::toImpl(webArchiveResourceRef)->URL());
52}
53
54WKStringRef WKWebArchiveResourceCopyMIMEType(WKWebArchiveResourceRef webArchiveResourceRef)
55{
56 return WebKit::toCopiedAPI(WebKit::toImpl(webArchiveResourceRef)->MIMEType());
57}
58
59WKStringRef WKWebArchiveResourceCopyTextEncoding(WKWebArchiveResourceRef webArchiveResourceRef)
60{
61 return WebKit::toCopiedAPI(WebKit::toImpl(webArchiveResourceRef)->textEncoding());
62}
