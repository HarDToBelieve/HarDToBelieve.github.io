60WebArchive::WebArchive(WebArchiveResource* mainResource, RefPtr<API::Array>&& subresources, RefPtr<API::Array>&& subframeArchives)
61 : m_cachedMainResource(mainResource)
62 , m_cachedSubresources(subresources)
63 , m_cachedSubframeArchives(subframeArchives)
64{
65 auto coreMainResource = m_cachedMainResource->coreArchiveResource();
66
67 Vector<Ref<ArchiveResource>> coreArchiveResources;
68 coreArchiveResources.reserveInitialCapacity(m_cachedSubresources->size());
69 for (size_t i = 0; i < m_cachedSubresources->size(); ++i) {
70 auto resource = m_cachedSubresources->at<WebArchiveResource>(i);
71 ASSERT(resource);
72 ASSERT(resource->coreArchiveResource());
73 coreArchiveResources.uncheckedAppend(*resource->coreArchiveResource());
74 }
75
76 Vector<Ref<LegacyWebArchive>> coreSubframeLegacyWebArchives;
77 coreSubframeLegacyWebArchives.reserveInitialCapacity(m_cachedSubframeArchives->size());
78 for (size_t i = 0; i < m_cachedSubframeArchives->size(); ++i) {
79 auto subframeWebArchive = m_cachedSubframeArchives->at<WebArchive>(i);
80 ASSERT(subframeWebArchive);
81 ASSERT(subframeWebArchive->coreLegacyWebArchive());
82 coreSubframeLegacyWebArchives.uncheckedAppend(*subframeWebArchive->coreLegacyWebArchive());
83 }
84
85 m_legacyWebArchive = LegacyWebArchive::create(*coreMainResource, WTFMove(coreArchiveResources), WTFMove(coreSubframeLegacyWebArchives));
86}

102WebArchiveResource* WebArchive::mainResource()
103{
104 if (!m_cachedMainResource)
105 m_cachedMainResource = WebArchiveResource::create(m_legacyWebArchive->mainResource());
106 return m_cachedMainResource.get();
107}

109API::Array* WebArchive::subresources()
110{
111 if (!m_cachedSubresources) {
112 Vector<RefPtr<API::Object>> subresources;
113 subresources.reserveInitialCapacity(m_legacyWebArchive->subresources().size());
114 for (auto& subresource : m_legacyWebArchive->subresources())
115 subresources.uncheckedAppend(WebArchiveResource::create(subresource.ptr()));
116 m_cachedSubresources = API::Array::create(WTFMove(subresources));
117 }
118
119 return m_cachedSubresources.get();
120}
121
122API::Array* WebArchive::subframeArchives()
123{
124 if (!m_cachedSubframeArchives) {
125 Vector<RefPtr<API::Object>> subframeWebArchives;
126 subframeWebArchives.reserveInitialCapacity(m_legacyWebArchive->subframeArchives().size());
127 for (auto& subframeArchive : m_legacyWebArchive->subframeArchives())
128 subframeWebArchives.uncheckedAppend(WebArchive::create(static_cast<LegacyWebArchive*>(subframeArchive.ptr())));
129 m_cachedSubframeArchives = API::Array::create(WTFMove(subframeWebArchives));
130 }
131
132 return m_cachedSubframeArchives.get();
133}

135static void releaseWebArchiveData(unsigned char*, const void* data)
136{
137 // Balanced by CFRetain in WebArchive::data().
138 CFRelease(data);
139}
140
141Ref<API::Data> WebArchive::data()
142{
143 RetainPtr<CFDataRef> rawDataRepresentation = m_legacyWebArchive->rawDataRepresentation();
144
145 // Balanced by CFRelease in releaseWebArchiveData.
146 CFRetain(rawDataRepresentation.get());
147
148 return API::Data::createWithoutCopying(CFDataGetBytePtr(rawDataRepresentation.get()), CFDataGetLength(rawDataRepresentation.get()), releaseWebArchiveData, rawDataRepresentation.get());
149}
