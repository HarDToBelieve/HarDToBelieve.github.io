175 enum class State {
176 // Idle - Nothing's happening.
177 Idle,
178
179 // Activating - Waiting to show the payment UI.
180 Activating,
181
182 // Active - Showing payment UI.
183 Active,
184
185 // Authorized - Dispatching the authorized event and waiting for the paymentSessionCompleted message.
186 Authorized,
187
188 // ShippingMethodSelected - Dispatching the shippingmethodselected event and waiting for a reply.
189 ShippingMethodSelected,
190
191 // ShippingContactSelected - Dispatching the shippingcontactselected event and waiting for a reply.
192 ShippingContactSelected,
193
194 // PaymentMethodSelected - Dispatching the paymentmethodselected event and waiting for a reply.
195 PaymentMethodSelected,
196
197#if ENABLE(APPLE_PAY_COUPON_CODE)
198 // CouponCodeChanged - Dispatching the couponcodechanged event and waiting for a reply.
199 CouponCodeChanged,
200#endif
201
202 // Completing - Completing the payment and waiting for presenterDidFinish to be called.
203 Completing,
204 } m_state { State::Idle };

206 enum class MerchantValidationState {
207 // Idle - Nothing's happening.
208 Idle,
209
210 // Validating - Dispatching the validatemerchant event and waiting for a reply.
211 Validating,
212
213 // ValidationComplete - A merchant session has been sent along to PassKit.
214 ValidationComplete
215 } m_merchantValidationState { MerchantValidationState::Idle };
