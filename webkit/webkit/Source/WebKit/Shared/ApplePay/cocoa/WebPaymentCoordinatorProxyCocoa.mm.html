111static RetainPtr<NSSet> toPKContactFields(const WebCore::ApplePaySessionPaymentRequest::ContactFields& contactFields)
112{
113 Vector<NSString *> result;
114
115 if (contactFields.postalAddress)
116 result.append(PAL::get_PassKit_PKContactFieldPostalAddress());
117 if (contactFields.phone)
118 result.append(PAL::get_PassKit_PKContactFieldPhoneNumber());
119 if (contactFields.email)
120 result.append(PAL::get_PassKit_PKContactFieldEmailAddress());
121 if (contactFields.name)
122 result.append(PAL::get_PassKit_PKContactFieldName());
123 if (contactFields.phoneticName)
124 result.append(PAL::get_PassKit_PKContactFieldPhoneticName());
125
126 return adoptNS([[NSSet alloc] initWithObjects:result.data() count:result.size()]);
127}

129NSDecimalNumber *toDecimalNumber(const String& amount)
130{
131 if (!amount)
132 return [NSDecimalNumber zero];
133 return [NSDecimalNumber decimalNumberWithString:amount locale:@{ NSLocaleDecimalSeparator : @"." }];
134}

136static PKMerchantCapability toPKMerchantCapabilities(const WebCore::ApplePaySessionPaymentRequest::MerchantCapabilities& merchantCapabilities)
137{
138 PKMerchantCapability result = 0;
139 if (merchantCapabilities.supports3DS)
140 result |= PKMerchantCapability3DS;
141 if (merchantCapabilities.supportsEMV)
142 result |= PKMerchantCapabilityEMV;
143 if (merchantCapabilities.supportsCredit)
144 result |= PKMerchantCapabilityCredit;
145 if (merchantCapabilities.supportsDebit)
146 result |= PKMerchantCapabilityDebit;
147
148 return result;
149}

171{
172 auto result = adoptNS([[NSDateComponents alloc] init]);
173 [result setCalendar:[NSCalendar calendarWithIdentifier:NSCalendarIdentifierGregorian]];

192PKShippingMethod *toPKShippingMethod(const WebCore::ApplePayShippingMethod& shippingMethod)
193{
194 PKShippingMethod *result = [PAL::getPKShippingMethodClass() summaryItemWithLabel:shippingMethod.label amount:toDecimalNumber(shippingMethod.amount)];
195 [result setIdentifier:shippingMethod.identifier];
196 [result setDetail:shippingMethod.detail];
197#if HAVE(PASSKIT_SHIPPING_METHOD_DATE_COMPONENTS_RANGE)
198 if (auto& dateComponentsRange = shippingMethod.dateComponentsRange)
199 [result setDateComponentsRange:toPKDateComponentsRange(*dateComponentsRange).get()];
200#endif
201 return result;
202}

222static RetainPtr<NSSet> toNSSet(const Vector<String>& strings)
223{
224 if (strings.isEmpty())
225 return nil;
226
227 auto mutableSet = adoptNS([[NSMutableSet alloc] initWithCapacity:strings.size()]);
228 for (auto& string : strings)
229 [mutableSet addObject:string];
230
231 return WTFMove(mutableSet);
232}

244RetainPtr<PKPaymentRequest> WebPaymentCoordinatorProxy::platformPaymentRequest(const URL& originatingURL, const Vector<URL>& linkIconURLs, const WebCore::ApplePaySessionPaymentRequest& paymentRequest)
245{
246 auto result = adoptNS([PAL::allocPKPaymentRequestInstance() init]);
247
248 [result setOriginatingURL:originatingURL];
249
250 [result setThumbnailURLs:createNSArray(linkIconURLs).get()];
251
252 [result setAPIType:toAPIType(paymentRequest.requester())];
253
254 [result setCountryCode:paymentRequest.countryCode()];
255 [result setCurrencyCode:paymentRequest.currencyCode()];
256 [result setBillingContact:paymentRequest.billingContact().pkContact()];
257 [result setShippingContact:paymentRequest.shippingContact().pkContact()];
258 [result setRequiredBillingContactFields:toPKContactFields(paymentRequest.requiredBillingContactFields()).get()];
259 [result setRequiredShippingContactFields:toPKContactFields(paymentRequest.requiredShippingContactFields()).get()];
260
261 [result setSupportedNetworks:createNSArray(paymentRequest.supportedNetworks()).get()];
262 [result setMerchantCapabilities:toPKMerchantCapabilities(paymentRequest.merchantCapabilities())];
263
264 [result setShippingType:toPKShippingType(paymentRequest.shippingType())];
265
266 [result setShippingMethods:createNSArray(paymentRequest.shippingMethods(), [] (auto& method) {
267 return toPKShippingMethod(method);
268 }).get()];
269
270 [result setPaymentSummaryItems:WebCore::platformSummaryItems(paymentRequest.total(), paymentRequest.lineItems())];
271
272 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
273 [result setExpectsMerchantSession:YES];
274 ALLOW_DEPRECATED_DECLARATIONS_END
275
276 if (!paymentRequest.applicationData().isNull()) {
277 auto applicationData = adoptNS([[NSData alloc] initWithBase64EncodedString:paymentRequest.applicationData() options:0]);
278 [result setApplicationData:applicationData.get()];
279 }
280
281 [result setSupportedCountries:toNSSet(paymentRequest.supportedCountries()).get()];
282
283#if HAVE(PASSKIT_BOUND_INTERFACE_IDENTIFIER)
284 auto& boundInterfaceIdentifier = m_client.paymentCoordinatorBoundInterfaceIdentifier(*this);
285 if (!boundInterfaceIdentifier.isEmpty())
286 [result setBoundInterfaceIdentifier:boundInterfaceIdentifier];
287#endif
288
289 // FIXME: Instead of using respondsToSelector, this should use a proper #if version check.
290 auto& bundleIdentifier = m_client.paymentCoordinatorSourceApplicationBundleIdentifier(*this);
291 if (!bundleIdentifier.isEmpty() && [result respondsToSelector:@selector(setSourceApplicationBundleIdentifier:)])
292 [result setSourceApplicationBundleIdentifier:bundleIdentifier];
293
294 auto& secondaryIdentifier = m_client.paymentCoordinatorSourceApplicationSecondaryIdentifier(*this);
295 if (!secondaryIdentifier.isEmpty() && [result respondsToSelector:@selector(setSourceApplicationSecondaryIdentifier:)])
296 [result setSourceApplicationSecondaryIdentifier:secondaryIdentifier];
297
298#if PLATFORM(IOS_FAMILY)
299 auto& serviceType = m_client.paymentCoordinatorCTDataConnectionServiceType(*this);
300 if (!serviceType.isEmpty() && [result respondsToSelector:@selector(setCTDataConnectionServiceType:)])
301 [result setCTDataConnectionServiceType:serviceType];
302#endif
303
304#if HAVE(PASSKIT_INSTALLMENTS)
305 if (PKPaymentInstallmentConfiguration *configuration = paymentRequest.installmentConfiguration().platformConfiguration()) {
306 [result setInstallmentConfiguration:configuration];
307 [result setRequestType:PKPaymentRequestTypeInstallment];
308 }
309#endif
310
311#if HAVE(PASSKIT_COUPON_CODE)
312 if (auto supportsCouponCode = paymentRequest.supportsCouponCode())
313 [result setSupportsCouponCode:*supportsCouponCode];
314
315 if (auto& couponCode = paymentRequest.couponCode(); !couponCode.isNull())
316 [result setCouponCode:couponCode];
317#endif
318
319#if HAVE(PASSKIT_SHIPPING_CONTACT_EDITING_MODE)
320 if (auto& shippingContactEditingMode = paymentRequest.shippingContactEditingMode())
321 [result setShippingContactEditingMode:toPKShippingContactEditingMode(*shippingContactEditingMode)];
322#endif
323
324#if defined(WebPaymentCoordinatorProxyCocoaAdditions_platformPaymentRequest)
325 WebPaymentCoordinatorProxyCocoaAdditions_platformPaymentRequest
326#endif
327
328 return result;
329}

427void WebPaymentCoordinatorProxy::platformBeginApplePaySetup(const PaymentSetupConfiguration& configuration, const PaymentSetupFeatures& features, Messages::WebPaymentCoordinatorProxy::BeginApplePaySetup::AsyncReply&& reply)
428{
429 UIViewController *presentingViewController = m_client.paymentCoordinatorPresentingViewController(*this);
430 if (!presentingViewController) {
431 reply(false);
432 return;
433 }
434
435 auto request = adoptNS([PAL::allocPKPaymentSetupRequestInstance() init]);
436 [request setConfiguration:configuration.platformConfiguration()];
437 [request setPaymentSetupFeatures:features.platformFeatures()];
438
439 auto paymentSetupViewController = adoptNS([PAL::allocPKPaymentSetupViewControllerInstance() initWithPaymentSetupRequest:request.get()]);
440 if (!paymentSetupViewController) {
441 reply(false);
442 return;
443 }
444
445 auto completion = makeBlockPtr([reply = WTFMove(reply)]() mutable {
446 RunLoop::main().dispatch([reply = WTFMove(reply)]() mutable {
447 reply(true);
448 });
449 });
450
451 endApplePaySetup();
452 [presentingViewController presentViewController:paymentSetupViewController.get() animated:YES completion:completion.get()];
453 m_paymentSetupViewController = paymentSetupViewController.get();
454}
