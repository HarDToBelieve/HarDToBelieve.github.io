38void WebPaymentCoordinatorProxy::platformCanMakePayments(CompletionHandler<void(bool)>&& completionHandler)
39{
40 if (!PAL::isPassKitFrameworkAvailable())
41 return completionHandler(false);
42
43 m_canMakePaymentsQueue->dispatch([theClass = retainPtr(PAL::getPKPaymentAuthorizationViewControllerClass()), completionHandler = WTFMove(completionHandler)]() mutable {
44 RunLoop::main().dispatch([canMakePayments = [theClass canMakePayments], completionHandler = WTFMove(completionHandler)]() mutable {
45 completionHandler(canMakePayments);
46 });
47 });
48}

50void WebPaymentCoordinatorProxy::platformShowPaymentUI(const URL& originatingURL, const Vector<URL>& linkIconURLStrings, const WebCore::ApplePaySessionPaymentRequest& request, CompletionHandler<void(bool)>&& completionHandler)
51{
52 if (!PAL::isPassKitFrameworkAvailable())
53 return completionHandler(false);
54
55 auto paymentRequest = platformPaymentRequest(originatingURL, linkIconURLStrings, request);
56
57 auto showPaymentUIRequestSeed = m_showPaymentUIRequestSeed;
58 auto weakThis = makeWeakPtr(*this);
59 [PAL::getPKPaymentAuthorizationViewControllerClass() requestViewControllerWithPaymentRequest:paymentRequest.get() completion:makeBlockPtr([paymentRequest, showPaymentUIRequestSeed, weakThis, completionHandler = WTFMove(completionHandler)](PKPaymentAuthorizationViewController *viewController, NSError *error) mutable {
60 auto paymentCoordinatorProxy = weakThis.get();
61 if (!paymentCoordinatorProxy)
62 return completionHandler(false);
63
64 if (error) {
65 LOG_ERROR("+[PKPaymentAuthorizationViewController requestViewControllerWithPaymentRequest:completion:] error %@", error);
66
67 completionHandler(false);
68 return;
69 }
70
71 if (showPaymentUIRequestSeed != paymentCoordinatorProxy->m_showPaymentUIRequestSeed) {
72 // We've already been asked to hide the payment UI. Don't attempt to show it.
73 return completionHandler(false);
74 }
75
76 NSWindow *presentingWindow = paymentCoordinatorProxy->m_client.paymentCoordinatorPresentingWindow(*paymentCoordinatorProxy);
77 if (!presentingWindow)
78 return completionHandler(false);
79
80 ASSERT(viewController);
81
82 paymentCoordinatorProxy->m_authorizationPresenter = makeUnique<PaymentAuthorizationViewController>(*paymentCoordinatorProxy, paymentRequest.get(), viewController);
83
84 ASSERT(!paymentCoordinatorProxy->m_sheetWindow);
85 paymentCoordinatorProxy->m_sheetWindow = [NSWindow windowWithContentViewController:viewController];
86
87 paymentCoordinatorProxy->m_sheetWindowWillCloseObserver = [[NSNotificationCenter defaultCenter] addObserverForName:NSWindowWillCloseNotification object:paymentCoordinatorProxy->m_sheetWindow.get() queue:nil usingBlock:[paymentCoordinatorProxy](NSNotification *) {
88 paymentCoordinatorProxy->didReachFinalState();
89 }];
90
91 [presentingWindow beginSheet:paymentCoordinatorProxy->m_sheetWindow.get() completionHandler:nullptr];
92
93 completionHandler(true);
94 }).get()];
95}

97void WebPaymentCoordinatorProxy::platformHidePaymentUI()
98{
99 if (m_state == State::Activating) {
100 ++m_showPaymentUIRequestSeed;
101
102 ASSERT(!m_authorizationPresenter);
103 ASSERT(!m_sheetWindow);
104 return;
105 }
106
107 ASSERT(m_authorizationPresenter);
108 ASSERT(m_sheetWindow);
109
110 [[NSNotificationCenter defaultCenter] removeObserver:m_sheetWindowWillCloseObserver.get()];
111 m_sheetWindowWillCloseObserver = nullptr;
112
113 [[m_sheetWindow sheetParent] endSheet:m_sheetWindow.get()];
114
115 if (m_authorizationPresenter)
116 m_authorizationPresenter->dismiss();
117
118 m_sheetWindow = nullptr;
119}
