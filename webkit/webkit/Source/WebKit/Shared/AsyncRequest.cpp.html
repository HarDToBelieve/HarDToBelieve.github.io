27#include "config.h"
28#include "AsyncRequest.h"
29
30#include <wtf/RunLoop.h>
31
32namespace WebKit {
33
34static uint64_t generateRequestID()
35{
36 ASSERT(RunLoop::isMain());
37 static uint64_t requestID = 0;
38 return ++requestID;
39}
40
41AsyncRequest::AsyncRequest(WTF::Function<void ()>&& abortHandler)
42 : m_abortHandler(WTFMove(abortHandler))
43 , m_requestID(generateRequestID())
44{
45}
46
47AsyncRequest::~AsyncRequest()
48{
49 ASSERT(!m_abortHandler);
50}
51
52void AsyncRequest::setAbortHandler(WTF::Function<void ()>&& handler)
53{
54 m_abortHandler = WTFMove(handler);
55}
56
57void AsyncRequest::requestAborted()
58{
59 if (m_abortHandler) {
60 m_abortHandler();
61 m_abortHandler = nullptr;
62 }
63
64 clearCompletionHandler();
65}
66
67} // namespace WebKit
