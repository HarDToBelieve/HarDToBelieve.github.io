62enum class AuthenticationChallengeDisposition : uint8_t;
63using ChallengeCompletionHandler = CompletionHandler<void(AuthenticationChallengeDisposition, const WebCore::Credential&)>;
64
65class AuthenticationManager : public NetworkProcessSupplement, public IPC::MessageReceiver {
66 WTF_MAKE_FAST_ALLOCATED;
67 WTF_MAKE_NONCOPYABLE(AuthenticationManager);
68public:
69 explicit AuthenticationManager(NetworkProcess&);
70
71 static const char* supplementName();
72
73 void didReceiveAuthenticationChallenge(PAL::SessionID, WebPageProxyIdentifier, const WebCore::SecurityOriginData* , const WebCore::AuthenticationChallenge&, NegotiatedLegacyTLS, ChallengeCompletionHandler&&);
74 void didReceiveAuthenticationChallenge(IPC::MessageSender& download, const WebCore::AuthenticationChallenge&, ChallengeCompletionHandler&&);
75
76 void completeAuthenticationChallenge(uint64_t challengeID, AuthenticationChallengeDisposition, WebCore::Credential&&);
77
78 void negotiatedLegacyTLS(WebPageProxyIdentifier) const;
79
80private:
81 struct Challenge {
82 WTF_MAKE_STRUCT_FAST_ALLOCATED;
83 Challenge(WebPageProxyIdentifier pageID, const WebCore::AuthenticationChallenge& challenge, ChallengeCompletionHandler&& completionHandler)
84 : pageID(pageID)
85 , challenge(challenge)
86 , completionHandler(WTFMove(completionHandler)) { }
87
88 WebPageProxyIdentifier pageID;
89 WebCore::AuthenticationChallenge challenge;
90 ChallengeCompletionHandler completionHandler;
91 };
92
93#if HAVE(SEC_KEY_PROXY)
94 // NetworkProcessSupplement
95 void initializeConnection(IPC::Connection*) final;
96#endif
97
98 // IPC::MessageReceiver
99 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
100
101 uint64_t addChallengeToChallengeMap(UniqueRef<Challenge>&&);
102 bool shouldCoalesceChallenge(WebPageProxyIdentifier, uint64_t challengeID, const WebCore::AuthenticationChallenge&) const;
103
104 Vector<uint64_t> coalesceChallengesMatching(uint64_t challengeID) const;
105
106 NetworkProcess& m_process;
107
108 HashMap<uint64_t, UniqueRef<Challenge>> m_challenges;
109};
