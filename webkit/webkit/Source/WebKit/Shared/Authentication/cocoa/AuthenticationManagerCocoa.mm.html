40void AuthenticationManager::initializeConnection(IPC::Connection* connection)
41{
42 RELEASE_ASSERT(isMainRunLoop());
43
44 if (!connection || !connection->xpcConnection()) {
45 ASSERT_NOT_REACHED();
46 return;
47 }
48
49 auto weakThis = makeWeakPtr(*this);
50 // The following xpc event handler overwrites the boostrap event handler and is only used
51 // to capture client certificate credential.
52 xpc_connection_set_event_handler(connection->xpcConnection(), ^(xpc_object_t event) {
53 callOnMainRunLoop([event = OSObjectPtr(event), weakThis = weakThis] {
54 RELEASE_ASSERT(isMainRunLoop());
55
56 xpc_type_t type = xpc_get_type(event.get());
57 if (type == XPC_TYPE_ERROR || !weakThis)
58 return;
59
60 if (type != XPC_TYPE_DICTIONARY || strcmp(xpc_dictionary_get_string(event.get(), ClientCertificateAuthentication::XPCMessageNameKey), ClientCertificateAuthentication::XPCMessageNameValue)) {
61 ASSERT_NOT_REACHED();
62 return;
63 }
64
65 auto challengeID = xpc_dictionary_get_uint64(event.get(), ClientCertificateAuthentication::XPCChallengeIDKey);
66 if (!challengeID)
67 return;
68
69 auto xpcEndPoint = xpc_dictionary_get_value(event.get(), ClientCertificateAuthentication::XPCSecKeyProxyEndpointKey);
70 if (!xpcEndPoint || xpc_get_type(xpcEndPoint) != XPC_TYPE_ENDPOINT)
71 return;
72 auto endPoint = adoptNS([[NSXPCListenerEndpoint alloc] init]);
73 [endPoint _setEndpoint:xpcEndPoint];
74 NSError *error = nil;
75 auto identity = adoptCF([SecKeyProxy createIdentityFromEndpoint:endPoint.get() error:&error]);
76 if (!identity || error) {
77 LOG_ERROR("Couldn't create identity from end point: %@", error);
78 return;
79 }
80
81 auto certificateDataArray = xpc_dictionary_get_array(event.get(), ClientCertificateAuthentication::XPCCertificatesKey);
82 if (!certificateDataArray)
83 return;
84 NSMutableArray *certificates = nil;
85 if (auto total = xpc_array_get_count(certificateDataArray)) {
86 certificates = [NSMutableArray arrayWithCapacity:total];
87 for (size_t i = 0; i < total; i++) {
88 auto certificateData = xpc_array_get_value(certificateDataArray, i);
89 auto cfData = adoptCF(CFDataCreate(nullptr, reinterpret_cast<const UInt8*>(xpc_data_get_bytes_ptr(certificateData)), xpc_data_get_length(certificateData)));
90 auto certificate = adoptCF(SecCertificateCreateWithData(nullptr, cfData.get()));
91 if (!certificate)
92 return;
93 [certificates addObject:(__bridge id)certificate.get()];
94 }
95 }
96
97 auto persistence = xpc_dictionary_get_uint64(event.get(), ClientCertificateAuthentication::XPCPersistenceKey);
98 if (persistence > static_cast<uint64_t>(NSURLCredentialPersistenceSynchronizable))
99 return;
100
101 weakThis->completeAuthenticationChallenge(challengeID, AuthenticationChallengeDisposition::UseCredential, WebCore::Credential(adoptNS([[NSURLCredential alloc] initWithIdentity:identity.get() certificates:certificates persistence:(NSURLCredentialPersistence)persistence]).get()));
102 });
103 });
104}
