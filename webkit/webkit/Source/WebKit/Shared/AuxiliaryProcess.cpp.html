64void AuxiliaryProcess::initialize(const AuxiliaryProcessInitializationParameters& parameters)
65{
66 WTF::RefCountedBase::enableThreadingChecksGlobally();
67
68 setAuxiliaryProcessType(parameters.processType);
69
70 RELEASE_ASSERT_WITH_MESSAGE(parameters.processIdentifier, "Unable to initialize child process without a WebCore process identifier");
71 Process::setIdentifier(*parameters.processIdentifier);
72
73 platformInitialize(parameters);
74
75#if PLATFORM(COCOA)
76 m_priorityBoostMessage = parameters.priorityBoostMessage;
77#endif
78
79 initializeProcess(parameters);
80
81 SandboxInitializationParameters sandboxParameters;
82 initializeSandbox(parameters, sandboxParameters);
83
84#if !LOG_DISABLED || !RELEASE_LOG_DISABLED
85 WebCore::initializeLogChannelsIfNecessary();
86 WebKit::initializeLogChannelsIfNecessary();
87#endif // !LOG_DISABLED || !RELEASE_LOG_DISABLED
88
89 initializeProcessName(parameters);
90
91 // In WebKit2, only the UI process should ever be generating certain identifiers.
92 PAL::SessionID::enableGenerationProtection();
93 ContentWorldIdentifier::enableGenerationProtection();
94 WebPageProxyIdentifier::enableGenerationProtection();
95
96 m_connection = IPC::Connection::createClientConnection(parameters.connectionIdentifier, *this);
97 initializeConnection(m_connection.get());
98 m_connection->open();
99}

146void AuxiliaryProcess::disableTermination()
147{
148 m_terminationCounter++;
149 m_terminationTimer.stop();
150}
151
152void AuxiliaryProcess::enableTermination()
153{
154 ASSERT(m_terminationCounter > 0);
155 m_terminationCounter--;
156
157 if (m_terminationCounter)
158 return;
159
160 if (!m_terminationTimeout) {
161 terminationTimerFired();
162 return;
163 }
164
165 m_terminationTimer.startOneShot(m_terminationTimeout);
166}

215std::optional<std::pair<IPC::Connection::Identifier, IPC::Attachment>> AuxiliaryProcess::createIPCConnectionPair()
216{
217#if USE(UNIX_DOMAIN_SOCKETS)
218 IPC::Connection::SocketPair socketPair = IPC::Connection::createPlatformConnection();
219 return std::make_pair(socketPair.server, IPC::Attachment { socketPair.client });
220#elif OS(DARWIN)
221 // Create the listening port.
222 mach_port_t listeningPort = MACH_PORT_NULL;
223 auto kr = mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &listeningPort);
224 if (kr != KERN_SUCCESS) {
225 RELEASE_LOG_ERROR(Process, "AuxiliaryProcess::createIPCConnectionPair: Could not allocate mach port, error %x", kr);
226 CRASH();
227 }
228 if (!MACH_PORT_VALID(listeningPort)) {
229 RELEASE_LOG_ERROR(Process, "AuxiliaryProcess::createIPCConnectionPair: Could not allocate mach port, returned port was invalid");
230 CRASH();
231 }
232 return std::make_pair(IPC::Connection::Identifier { listeningPort }, IPC::Attachment { listeningPort, MACH_MSG_TYPE_MAKE_SEND });
233#elif OS(WINDOWS)
234 IPC::Connection::Identifier serverIdentifier, clientIdentifier;
235 if (!IPC::Connection::createServerAndClientIdentifiers(serverIdentifier, clientIdentifier)) {
236 LOG_ERROR("Failed to create server and client identifiers");
237 CRASH();
238 }
239 return std::make_pair(serverIdentifier, IPC::Attachment { clientIdentifier });
240#else
241 notImplemented();
242 return { };
243#endif
244}
