26#ifndef BlobDataFileReferenceWithSandboxExtension_h
27#define BlobDataFileReferenceWithSandboxExtension_h
28
29#include <WebCore/BlobDataFileReference.h>
30
31namespace WebKit {
32
33class SandboxExtension;
34
35class BlobDataFileReferenceWithSandboxExtension final : public WebCore::BlobDataFileReference {
36public:
37 static Ref<BlobDataFileReference> create(const String& path, const String& replacementPath = { }, RefPtr<SandboxExtension>&& sandboxExtension = nullptr)
38 {
39 return adoptRef(*new BlobDataFileReferenceWithSandboxExtension(path, replacementPath, WTFMove(sandboxExtension)));
40 }
41
42private:
43 BlobDataFileReferenceWithSandboxExtension(const String& path, const String& replacementPath, RefPtr<SandboxExtension>&&);
44 virtual ~BlobDataFileReferenceWithSandboxExtension();
45
46 void prepareForFileAccess() override;
47 void revokeFileAccess() override;
48
49 RefPtr<SandboxExtension> m_sandboxExtension;
50};
51
52}
53
54#endif // BlobDataFileReferenceWithSandboxExtension_h
