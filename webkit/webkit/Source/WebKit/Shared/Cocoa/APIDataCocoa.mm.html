31Ref<Data> Data::createWithoutCopying(RetainPtr<NSData> data)
32{
33 const unsigned char* bytes = reinterpret_cast<const unsigned char*>([data bytes]);
34 size_t size = [data length];
35
36 return createWithoutCopying(bytes, size, [](unsigned char*, const void* data) {
37 if (!data)
38 return;
39 CFRelease(data);
40 }, static_cast<void*>(data.leakRef()));
41}
