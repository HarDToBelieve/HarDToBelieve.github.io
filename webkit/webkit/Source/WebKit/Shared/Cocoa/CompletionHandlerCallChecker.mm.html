36Ref<CompletionHandlerCallChecker> CompletionHandlerCallChecker::create(id delegate, SEL delegateMethodSelector)
37{
38 return adoptRef(*new CompletionHandlerCallChecker(object_getClass(delegate), delegateMethodSelector));
39}
40
41CompletionHandlerCallChecker::CompletionHandlerCallChecker(Class delegateClass, SEL delegateMethodSelector)
42 : m_delegateClass(delegateClass)
43 , m_delegateMethodSelector(delegateMethodSelector)
44 , m_didCallCompletionHandler(false)
45{
46}
47
48CompletionHandlerCallChecker::~CompletionHandlerCallChecker()
49{
50 if (m_didCallCompletionHandler)
51 return;
52
53 Class delegateClass = classImplementingDelegateMethod();
54 [NSException raise:NSInternalInconsistencyException format:@"Completion handler passed to %c[%@ %@] was not called", class_isMetaClass(delegateClass) ? '+' : '-', NSStringFromClass(delegateClass), NSStringFromSelector(m_delegateMethodSelector)];
55}
56
57void CompletionHandlerCallChecker::didCallCompletionHandler()
58{
59 ASSERT(!m_didCallCompletionHandler);
60 m_didCallCompletionHandler = true;
61}
62
63static bool shouldThrowExceptionForDuplicateCompletionHandlerCall()
64{
65 static bool shouldThrowException;
66 static std::once_flag once;
67 std::call_once(once, [] {
68 shouldThrowException = linkedOnOrAfter(WebCore::SDKVersion::FirstWithExceptionsForDuplicateCompletionHandlerCalls);
69 });
70 return shouldThrowException;
71}
72
73bool CompletionHandlerCallChecker::completionHandlerHasBeenCalled() const
74{
75 if (!m_didCallCompletionHandler)
76 return false;
77
78 if (shouldThrowExceptionForDuplicateCompletionHandlerCall()) {
79 Class delegateClass = classImplementingDelegateMethod();
80 [NSException raise:NSInternalInconsistencyException format:@"Completion handler passed to %c[%@ %@] was called more than once", class_isMetaClass(delegateClass) ? '+' : '-', NSStringFromClass(delegateClass), NSStringFromSelector(m_delegateMethodSelector)];
81 }
82
83 return true;
84}
85
86Class CompletionHandlerCallChecker::classImplementingDelegateMethod() const
87{
88 Class delegateClass = m_delegateClass;
89 Method delegateMethod = class_getInstanceMethod(delegateClass, m_delegateMethodSelector);
90
91 for (Class superclass = class_getSuperclass(delegateClass); superclass; superclass = class_getSuperclass(superclass)) {
92 if (class_getInstanceMethod(superclass, m_delegateMethodSelector) != delegateMethod)
93 break;
94
95 delegateClass = superclass;
96 }
97
98 return delegateClass;
99}
