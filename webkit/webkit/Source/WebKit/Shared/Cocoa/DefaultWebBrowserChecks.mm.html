48bool isRunningTest(const String& bundleID)
49{
50 return bundleID == "com.apple.WebKit.TestWebKitAPI"_s || bundleID == "com.apple.WebKit.WebKitTestRunner"_s || bundleID == "org.webkit.WebKitTestRunnerApp"_s;
51}

53std::optional<Vector<WebCore::RegistrableDomain>> getAppBoundDomainsTesting(const String& bundleID)
54{
55 if (bundleID.isNull())
56 return std::nullopt;
57
58 static auto appBoundDomainList = makeNeverDestroyed(MemoryCompactLookupOnlyRobinHoodHashMap<String, Vector<WebCore::RegistrableDomain>> {
59 {"inAppBrowserPrivacyTestIdentifier"_s, Vector<WebCore::RegistrableDomain> { WebCore::RegistrableDomain::uncheckedCreateFromRegistrableDomainString("127.0.0.1") }},
60 });
61
62 auto appBoundDomainIter = appBoundDomainList->find(bundleID);
63 if (appBoundDomainIter != appBoundDomainList->end())
64 return appBoundDomainIter->value;
65
66 return std::nullopt;
67}

87enum class ITPState : uint8_t {
88 Uninitialized,
89 Enabled,
90 Disabled

93static std::atomic<ITPState> currentITPState = ITPState::Uninitialized;

95bool hasRequestedCrossWebsiteTrackingPermission()
96{
97 ASSERT(!isInWebKitChildProcess());
98
99 static std::atomic<bool> hasRequestedCrossWebsiteTrackingPermission = [[NSBundle mainBundle] objectForInfoDictionaryKey:@"NSCrossWebsiteTrackingUsageDescription"];
100 return hasRequestedCrossWebsiteTrackingPermission;
101}

103static bool determineITPStateInternal(bool appWasLinkedOnOrAfter, const String& bundleIdentifier)
104{
105 ASSERT(!RunLoop::isMain());
106 ASSERT(!isInWebKitChildProcess());
107
108 if (!appWasLinkedOnOrAfter && !isFullWebBrowser(bundleIdentifier))
109 return false;
110
111 if (!isFullWebBrowser(bundleIdentifier) && !hasRequestedCrossWebsiteTrackingPermission())
112 return true;
113
114 TCCAccessPreflightResult result = kTCCAccessPreflightDenied;
115#if (PLATFORM(IOS) && __IPHONE_OS_VERSION_MIN_REQUIRED >= 140000) || (PLATFORM(MAC) && __MAC_OS_X_VERSION_MIN_REQUIRED >= 110000)
116 result = TCCAccessPreflight(get_TCC_kTCCServiceWebKitIntelligentTrackingPrevention(), nullptr);
117#endif
118 return result != kTCCAccessPreflightDenied;
119}

121static RefPtr<WorkQueue>& itpQueue()
122{
123 static NeverDestroyed<RefPtr<WorkQueue>> itpQueue;
124 return itpQueue;
125}
126
127void determineITPState()
128{
129 ASSERT(RunLoop::isMain());
130 if (currentITPState != ITPState::Uninitialized)
131 return;
132
133 bool appWasLinkedOnOrAfter = linkedOnOrAfter(WebCore::SDKVersion::FirstWithSessionCleanupByDefault);
134
135 itpQueue() = WorkQueue::create("com.apple.WebKit.itpCheckQueue");
136 itpQueue()->dispatch([appWasLinkedOnOrAfter, bundleIdentifier = WebCore::applicationBundleIdentifier().isolatedCopy()] {
137 currentITPState = determineITPStateInternal(appWasLinkedOnOrAfter, bundleIdentifier) ? ITPState::Enabled : ITPState::Disabled;
138 RunLoop::main().dispatch([] {
139 itpQueue() = nullptr;
140 });
141 });
142}

144bool doesAppHaveITPEnabled()
145{
146 ASSERT(!isInWebKitChildProcess());
147 ASSERT(RunLoop::isMain());
148 // If we're still computing the ITP state on the background thread, then synchronize with it.
149 if (itpQueue())
150 itpQueue()->dispatchSync([] { });
151 ASSERT(currentITPState != ITPState::Uninitialized);
152 return currentITPState == ITPState::Enabled;
153}

155bool doesParentProcessHaveITPEnabled(AuxiliaryProcess& auxiliaryProcess, bool hasRequestedCrossWebsiteTrackingPermission)
156{
157 ASSERT(isInWebKitChildProcess());
158 ASSERT(RunLoop::isMain());
159
160 if (!isParentProcessAFullWebBrowser(auxiliaryProcess) && !hasRequestedCrossWebsiteTrackingPermission)
161 return true;
162
163 static bool itpEnabled { true };
164 static dispatch_once_t once;
165 dispatch_once(&once, ^{
166
167 TCCAccessPreflightResult result = kTCCAccessPreflightDenied;
168#if (PLATFORM(IOS) && __IPHONE_OS_VERSION_MIN_REQUIRED >= 140000) || (PLATFORM(MAC) && __MAC_OS_X_VERSION_MIN_REQUIRED >= 110000)
169 RefPtr<IPC::Connection> connection = auxiliaryProcess.parentProcessConnection();
170 if (!connection) {
171 ASSERT_NOT_REACHED();
172 RELEASE_LOG_ERROR(IPC, "Unable to get parent process connection");
173 return;
174 }
175
176 auto auditToken = connection->getAuditToken();
177 if (!auditToken) {
178 ASSERT_NOT_REACHED();
179 RELEASE_LOG_ERROR(IPC, "Unable to get parent process audit token");
180 return;
181 }
182 result = TCCAccessPreflightWithAuditToken(get_TCC_kTCCServiceWebKitIntelligentTrackingPrevention(), auditToken.value(), nullptr);
183#endif
184 itpEnabled = result != kTCCAccessPreflightDenied;
185 });
186 return itpEnabled;
187}

189static std::atomic<bool> hasCheckedUsageStrings = false;
190bool hasProhibitedUsageStrings()
191{
192 ASSERT(!isInWebKitChildProcess());
193
194 static bool hasProhibitedUsageStrings = false;
195
196 if (hasCheckedUsageStrings)
197 return hasProhibitedUsageStrings;
198
199 NSDictionary *infoDictionary = [[NSBundle mainBundle] infoDictionary];
200 RELEASE_ASSERT(infoDictionary);
201
202 // See <rdar://problem/59979468> for details about how this list was selected.
203 auto prohibitedStrings = @[
204 @"NSHomeKitUsageDescription",
205 @"NSBluetoothAlwaysUsageDescription",
206 @"NSPhotoLibraryUsageDescription",
207 @"NSHealthShareUsageDescription",
208 @"NSHealthUpdateUsageDescription",
209 @"NSLocationAlwaysUsageDescription",
210 @"NSLocationAlwaysAndWhenInUseUsageDescription"
211 ];
212
213 for (NSString *prohibitedString : prohibitedStrings) {
214 if ([infoDictionary objectForKey:prohibitedString]) {
215 String message = [NSString stringWithFormat:@"[In-App Browser Privacy] %@ used prohibited usage string %@.", [[NSBundle mainBundle] bundleIdentifier], prohibitedString];
216 WTFLogAlways(message.utf8().data());
217 hasProhibitedUsageStrings = true;
218 break;
219 }
220 }
221 hasCheckedUsageStrings = true;
222 return hasProhibitedUsageStrings;
223}

225bool isParentProcessAFullWebBrowser(AuxiliaryProcess& auxiliaryProcess)
226{
227 ASSERT(isInWebKitChildProcess());
228
229 static bool fullWebBrowser { false };
230 static dispatch_once_t once;
231 dispatch_once(&once, ^{
232 RefPtr<IPC::Connection> connection = auxiliaryProcess.parentProcessConnection();
233 if (!connection) {
234 ASSERT_NOT_REACHED();
235 RELEASE_LOG_ERROR(IPC, "Unable to get parent process connection");
236 return;
237 }
238
239 auto auditToken = connection->getAuditToken();
240 if (!auditToken) {
241 ASSERT_NOT_REACHED();
242 RELEASE_LOG_ERROR(IPC, "Unable to get parent process audit token");
243 return;
244 }
245
246 fullWebBrowser = WTF::hasEntitlement(*auditToken, "com.apple.developer.web-browser");
247 });
248
249 return fullWebBrowser || isRunningTest(WebCore::applicationBundleIdentifier());
250}

252static bool isFullWebBrowser(const String& bundleIdentifier)
253{
254 ASSERT(!isInWebKitChildProcess());
255
256 static bool fullWebBrowser = WTF::processHasEntitlement("com.apple.developer.web-browser");
257
258 return fullWebBrowser || isRunningTest(bundleIdentifier);
259}

261bool isFullWebBrowser()
262{
263 ASSERT(!isInWebKitChildProcess());
264 ASSERT(RunLoop::isMain());
265
266 return isFullWebBrowser(WebCore::applicationBundleIdentifier());
