41class ReadOnlySharedRingBufferStorage : public WebCore::CARingBufferStorage {
42public:
43 ReadOnlySharedRingBufferStorage() = default;
44 explicit ReadOnlySharedRingBufferStorage(const SharedMemory::Handle&);
45
46 SharedMemory* storage() const { return m_storage.get(); }
47
48protected:
49 // WebCore::CARingBufferStorage
50 bool allocate(size_t byteCount, const WebCore::CAAudioStreamDescription& format, size_t frameCount) override;
51 void deallocate() override { }
52 void* data() final;
53 void getCurrentFrameBounds(uint64_t& startTime, uint64_t& endTime) final;
54 void setCurrentFrameBounds(uint64_t startFrame, uint64_t endFrame) override { }
55 uint64_t currentStartFrame() const final { return m_startFrame; }
56 uint64_t currentEndFrame() const final { return m_endFrame; }
57 void updateFrameBounds() final;
58 void flush() final;
59 size_t size() const final;
60
61 static constexpr unsigned boundsBufferSize { 32 };
62 struct FrameBounds {
63 std::pair<uint64_t, uint64_t> boundsBuffer[boundsBufferSize];
64 Atomic<unsigned> boundsBufferIndex { 0 };
65 };
66 const FrameBounds* sharedFrameBounds() const;
67 FrameBounds* sharedFrameBounds();
68
69 RefPtr<SharedMemory> m_storage;
70 uint64_t m_startFrame { 0 };
71 uint64_t m_endFrame { 0 };
72};
73
74class SharedRingBufferStorage final : public ReadOnlySharedRingBufferStorage {
75public:
76 SharedRingBufferStorage(Function<void(SharedMemory*, const WebCore::CAAudioStreamDescription& format, size_t frameCount)>&& storageChangedHandler)
77 : m_storageChangedHandler(WTFMove(storageChangedHandler))
78 {
79 }
80
81 void invalidate() { m_storageChangedHandler = nullptr; }
82
83private:
84 // WebCore::CARingBufferStorage
85 bool allocate(size_t byteCount, const WebCore::CAAudioStreamDescription& format, size_t frameCount) final;
86 void deallocate() final;
87 void setCurrentFrameBounds(uint64_t startFrame, uint64_t endFrame) final;
88
89 void setStorage(RefPtr<SharedMemory>&&, const WebCore::CAAudioStreamDescription& format, size_t frameCount);
90
91 Function<void(SharedMemory*, const WebCore::CAAudioStreamDescription& format, size_t frameCount)> m_storageChangedHandler;
92};
