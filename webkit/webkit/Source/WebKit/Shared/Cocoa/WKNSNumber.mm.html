26#import "config.h"
27#import "WKNSNumber.h"
28
29using namespace WebKit;
30
31@implementation WKNSNumber {
32 union {
33 API::ObjectStorage<API::Boolean> _boolean;
34 API::ObjectStorage<API::Double> _double;
35 API::ObjectStorage<API::UInt64> _uint64;
36 API::ObjectStorage<API::Int64> _int64;
37 } _number;
38}
39
40- (void)dealloc
41{
42 switch (_type) {
43 case API::Object::Type::Boolean:
44 _number._boolean->~Number<bool, API::Object::Type::Boolean>();
45 break;
46
47 case API::Object::Type::Double:
48 _number._double->~Number<double, API::Object::Type::Double>();
49 break;
50
51 case API::Object::Type::UInt64:
52 _number._uint64->~Number<uint64_t, API::Object::Type::UInt64>();
53 break;
54
55 case API::Object::Type::Int64:
56 _number._int64->~Number<int64_t, API::Object::Type::Int64>();
57 break;
58
59 default:
60 ASSERT_NOT_REACHED();
61 }
62
63 [super dealloc];
64}
65
66// MARK: NSValue primitive methods
67
68- (const char *)objCType
69{
70 switch (_type) {
71 case API::Object::Type::Boolean:
72 return @encode(bool);
73 break;
74
75 case API::Object::Type::Double:
76 return @encode(double);
77 break;
78
79 case API::Object::Type::UInt64:
80 return @encode(uint64_t);
81 break;
82
83 case API::Object::Type::Int64:
84 return @encode(int64_t);
85 break;
86
87 default:
88 ASSERT_NOT_REACHED();
89 }
90
91 return nullptr;
92}
93
94- (void)getValue:(void *)value
95{
96 switch (_type) {
97 case API::Object::Type::Boolean:
98 *reinterpret_cast<bool*>(value) = _number._boolean->value();
99 break;
100
101 case API::Object::Type::Double:
102 *reinterpret_cast<double*>(value) = _number._double->value();
103 break;
104
105 case API::Object::Type::UInt64:
106 *reinterpret_cast<uint64_t*>(value) = _number._uint64->value();
107 break;
108
109 case API::Object::Type::Int64:
110 *reinterpret_cast<int64_t*>(value) = _number._int64->value();
111 break;
112
113 default:
114 ASSERT_NOT_REACHED();
115 }
116}
117
118// MARK: NSNumber primitive methods
119
120- (char)charValue
121{
122 if (_type == API::Object::Type::Boolean)
123 return _number._boolean->value();
124
125 return super.charValue;
126}
127
128- (double)doubleValue
129{
130 if (_type == API::Object::Type::Double)
131 return _number._double->value();
132
133 return super.doubleValue;
134}
135
136- (unsigned long long)unsignedLongLongValue
137{
138 if (_type == API::Object::Type::UInt64)
139 return _number._uint64->value();
140
141 return super.unsignedLongLongValue;
142}
143
144- (long long)longLongValue
145{
146 if (_type == API::Object::Type::Int64)
147 return _number._int64->value();
148
149 return super.longLongValue;
150}

153
154- (id)copyWithZone:(NSZone *)zone

156 return [self retain];

159// MARK: WKObject protocol implementation
160
161- (API::Object&)_apiObject
162{
163 switch (_type) {
164 case API::Object::Type::Boolean:
165 return *_number._boolean;
166 break;
167
168 case API::Object::Type::Double:
169 return *_number._double;
170 break;
171
172 case API::Object::Type::UInt64:
173 return *_number._uint64;
174 break;
175
176 case API::Object::Type::Int64:
177 return *_number._int64;
178 break;
179
180 default:
181 ASSERT_NOT_REACHED();
182 }
183
184 return *_number._boolean;
185}
