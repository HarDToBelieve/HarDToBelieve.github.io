26#import "config.h"
27#import "WKNSURLRequest.h"
28
29#import "WKNSURLExtras.h"
30
31@implementation WKNSURLRequest
32
33- (NSObject *)_web_createTarget
34{
35 return [static_cast<API::URLRequest*>(&self._apiObject)->resourceRequest().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody) copy];
36}
37
38- (NSURL *)URL
39{
40 return [NSURL _web_URLWithWTFString:static_cast<API::URLRequest*>(&self._apiObject)->resourceRequest().url().string()];
41}
42
43#pragma mark NSCopying protocol implementation
44
45- (id)copyWithZone:(NSZone *)zone
46{
47 return [self retain];
48}
49
50@end
