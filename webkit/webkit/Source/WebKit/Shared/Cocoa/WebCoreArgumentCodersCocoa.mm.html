455void ArgumentCoder<Ref<WebCore::Font>>::encodePlatformData(Encoder& encoder, const Ref<WebCore::Font>& font)
456{
457 const auto& platformData = font->platformData();
458 encoder << platformData.orientation();
459 encoder << platformData.widthVariant();
460 encoder << platformData.textRenderingMode();
461 encoder << platformData.size();
462 encoder << platformData.syntheticBold();
463 encoder << platformData.syntheticOblique();
464
465 auto ctFont = platformData.font();
466 auto fontDescriptor = adoptCF(CTFontCopyFontDescriptor(ctFont));
467 auto attributes = adoptCF(CTFontDescriptorCopyAttributes(fontDescriptor.get()));
468 encoder << attributes;
469
470 const auto& creationData = platformData.creationData();
471 encoder << static_cast<bool>(creationData);
472 if (creationData) {
473 encoder << creationData->fontFaceData;
474 encoder << creationData->itemInCollection;
475 } else {
476 auto referenceURL = adoptCF(static_cast<CFURLRef>(CTFontCopyAttribute(ctFont, kCTFontReferenceURLAttribute)));
477 auto string = CFURLGetString(referenceURL.get());
478 encoder << String(string);
479 encoder << String(adoptCF(CTFontCopyPostScriptName(ctFont)).get());
480 }
481}

483static RetainPtr<CTFontDescriptorRef> findFontDescriptor(const String& referenceURL, const String& postScriptName)
484{
485 auto url = adoptCF(CFURLCreateWithString(kCFAllocatorDefault, referenceURL.createCFString().get(), nullptr));
486 if (!url)
487 return nullptr;
488 auto fontDescriptors = adoptCF(CTFontManagerCreateFontDescriptorsFromURL(url.get()));
489 if (!fontDescriptors || !CFArrayGetCount(fontDescriptors.get()))
490 return nullptr;
491 if (CFArrayGetCount(fontDescriptors.get()) == 1)
492 return static_cast<CTFontDescriptorRef>(CFArrayGetValueAtIndex(fontDescriptors.get(), 0));
493
494 // There's supposed to only be a single item in the array, but we can be defensive here.
495 for (CFIndex i = 0; i < CFArrayGetCount(fontDescriptors.get()); ++i) {
496 auto fontDescriptor = static_cast<CTFontDescriptorRef>(CFArrayGetValueAtIndex(fontDescriptors.get(), i));
497 auto currentPostScriptName = adoptCF(static_cast<CFStringRef>(CTFontDescriptorCopyAttribute(fontDescriptor, kCTFontNameAttribute)));
498 if (String(currentPostScriptName.get()) == postScriptName)
499 return fontDescriptor;
500 }
501 return nullptr;
502}

504std::optional<WebCore::FontPlatformData> ArgumentCoder<Ref<WebCore::Font>>::decodePlatformData(Decoder& decoder)
505{
506 std::optional<WebCore::FontOrientation> orientation;
507 decoder >> orientation;
508 if (!orientation)
509 return std::nullopt;
510
511 std::optional<WebCore::FontWidthVariant> widthVariant;
512 decoder >> widthVariant;
513 if (!widthVariant)
514 return std::nullopt;
515
516 std::optional<WebCore::TextRenderingMode> textRenderingMode;
517 decoder >> textRenderingMode;
518 if (!textRenderingMode)
519 return std::nullopt;
520
521 std::optional<float> size;
522 decoder >> size;
523 if (!size)
524 return std::nullopt;
525
526 std::optional<bool> syntheticBold;
527 decoder >> syntheticBold;
528 if (!syntheticBold)
529 return std::nullopt;
530
531 std::optional<bool> syntheticOblique;
532 decoder >> syntheticOblique;
533 if (!syntheticOblique)
534 return std::nullopt;
535
536 std::optional<RetainPtr<CFDictionaryRef>> attributes;
537 decoder >> attributes;
538 if (!attributes)
539 return std::nullopt;
540
541 std::optional<bool> includesCreationData;
542 decoder >> includesCreationData;
543 if (!includesCreationData)
544 return std::nullopt;
545
546 if (*includesCreationData) {
547 std::optional<Ref<WebCore::SharedBuffer>> fontFaceData;
548 decoder >> fontFaceData;
549 if (!fontFaceData)
550 return std::nullopt;
551
552 // Upon receipt, copy the data for security, so the sender can't scribble over it while we're using it.
553 auto localFontFaceData = WebCore::SharedBuffer::create(fontFaceData.value()->data(), fontFaceData.value()->size());
554
555 std::optional<String> itemInCollection;
556 decoder >> itemInCollection;
557 if (!itemInCollection)
558 return std::nullopt;
559
560 auto fontCustomPlatformData = createFontCustomPlatformData(localFontFaceData, *itemInCollection);
561 if (!fontCustomPlatformData)
562 return std::nullopt;
563 auto baseFontDescriptor = fontCustomPlatformData->fontDescriptor.get();
564 if (!baseFontDescriptor)
565 return std::nullopt;
566 auto fontDescriptor = adoptCF(CTFontDescriptorCreateCopyWithAttributes(baseFontDescriptor, attributes->get()));
567 auto ctFont = adoptCF(CTFontCreateWithFontDescriptor(fontDescriptor.get(), *size, nullptr));
568
569 auto creationData = WebCore::FontPlatformData::CreationData { localFontFaceData, *itemInCollection };
570 return WebCore::FontPlatformData(ctFont.get(), *size, *syntheticBold, *syntheticOblique, *orientation, *widthVariant, *textRenderingMode, &creationData);
571 }
572
573 std::optional<String> referenceURL;
574 decoder >> referenceURL;
575 if (!referenceURL)
576 return std::nullopt;
577
578 std::optional<String> postScriptName;
579 decoder >> postScriptName;
580 if (!postScriptName)
581 return std::nullopt;
582
583 auto fontDescriptor = findFontDescriptor(*referenceURL, *postScriptName);
584 if (!fontDescriptor)
585 return std::nullopt;
586 fontDescriptor = adoptCF(CTFontDescriptorCreateCopyWithAttributes(fontDescriptor.get(), attributes->get()));
587 auto ctFont = adoptCF(CTFontCreateWithFontDescriptor(fontDescriptor.get(), *size, nullptr));
588
589 return WebCore::FontPlatformData(ctFont.get(), *size, *syntheticBold, *syntheticOblique, *orientation, *widthVariant, *textRenderingMode);
590}

592void ArgumentCoder<WebCore::ResourceRequest>::encodePlatformData(Encoder& encoder, const WebCore::ResourceRequest& resourceRequest)
593{
594 auto requestToSerialize = retainPtr(resourceRequest.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody));
595
596 if (Class requestClass = [requestToSerialize class]; UNLIKELY(requestClass != [NSURLRequest class] && requestClass != [NSMutableURLRequest class])) {
597 WebCore::ResourceRequest request(requestToSerialize.get());
598 request.replacePlatformRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
599 requestToSerialize = retainPtr(request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody));
600 }
601 ASSERT([requestToSerialize class] == [NSURLRequest class] || [requestToSerialize class] == [NSMutableURLRequest class]);
602
603 bool requestIsPresent = requestToSerialize;
604 encoder << requestIsPresent;
605
606 if (!requestIsPresent)
607 return;
608
609 // We don't send HTTP body over IPC for better performance.
610 // Also, it's not always possible to do, as streams can only be created in process that does networking.
611 if ([requestToSerialize HTTPBody] || [requestToSerialize HTTPBodyStream]) {
612 auto mutableRequest = adoptNS([requestToSerialize mutableCopy]);
613 [mutableRequest setHTTPBody:nil];
614 [mutableRequest setHTTPBodyStream:nil];
615 requestToSerialize = WTFMove(mutableRequest);
616 }
617
618 IPC::encode(encoder, requestToSerialize.get());
619
620 encoder << resourceRequest.requester();
621 encoder << resourceRequest.isAppBound();
622}
623
624bool ArgumentCoder<WebCore::ResourceRequest>::decodePlatformData(Decoder& decoder, WebCore::ResourceRequest& resourceRequest)
625{
626 bool requestIsPresent;
627 if (!decoder.decode(requestIsPresent))
628 return false;
629
630 if (!requestIsPresent) {
631 resourceRequest = WebCore::ResourceRequest();
632 return true;
633 }
634
635 auto request = IPC::decode<NSURLRequest>(decoder, NSURLRequest.class);
636 if (!request)
637 return false;
638
639 WebCore::ResourceRequest::Requester requester;
640 if (!decoder.decode(requester))
641 return false;
642
643 bool isAppBound;
644 if (!decoder.decode(isAppBound))
645 return false;
646
647 resourceRequest = WebCore::ResourceRequest(request->get());
648 resourceRequest.setRequester(requester);
649 resourceRequest.setIsAppBound(isAppBound);
650
651 return true;
652}

691bool ArgumentCoder<WebCore::MediaPlaybackTargetContext>::decodePlatformData(Decoder& decoder, WebCore::MediaPlaybackTargetContext::Type contextType, WebCore::MediaPlaybackTargetContext& target)
692{
693 ASSERT(contextType != WebCore::MediaPlaybackTargetContext::Type::Mock);
694
695 if (contextType == WebCore::MediaPlaybackTargetContext::Type::AVOutputContext) {
696 if (![PAL::getAVOutputContextClass() conformsToProtocol:@protocol(NSSecureCoding)])
697 return false;
698

756std::optional<RetainPtr<CVPixelBufferRef>> ArgumentCoder<RetainPtr<CVPixelBufferRef>>::decode(Decoder& decoder)
757{
758 std::optional<MachSendRight> sendRight;
759 decoder >> sendRight;
760 if (!sendRight)
761 return std::nullopt;
762 RetainPtr<CVPixelBufferRef> pixelBuffer;
763 if (!*sendRight)
764 return pixelBuffer;
765 {
766 auto surface = adoptCF(IOSurfaceLookupFromMachPort(sendRight->sendRight()));

768 return std::nullopt;

770 auto status = WebCore::CVPixelBufferCreateWithIOSurface(kCFAllocatorDefault, surface.get(), nullptr, &rawBuffer);
771 if (status != noErr || !rawBuffer)
772 return std::nullopt;
773 pixelBuffer = adoptCF(rawBuffer);
774 }
775 return pixelBuffer;
