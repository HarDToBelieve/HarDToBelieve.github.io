33namespace WebKit {
34
35void XRDeviceInfo::encode(IPC::Encoder& encoder) const
36{
37 encoder << identifier << supportsOrientationTracking << supportsStereoRendering << features << recommendedResolution;
38}
39
40std::optional<XRDeviceInfo> XRDeviceInfo::decode(IPC::Decoder& decoder)
41{
42 XRDeviceInfo deviceFeatures;
43 if (!decoder.decode(deviceFeatures.identifier))
44 return std::nullopt;
45
46 if (!decoder.decode(deviceFeatures.supportsOrientationTracking))
47 return std::nullopt;
48
49 if (!decoder.decode(deviceFeatures.supportsStereoRendering))
50 return std::nullopt;
51
52 if (!decoder.decode(deviceFeatures.features))
53 return std::nullopt;
54
55 if (!decoder.decode(deviceFeatures.recommendedResolution))
56 return std::nullopt;
57
58 return deviceFeatures;
59}
