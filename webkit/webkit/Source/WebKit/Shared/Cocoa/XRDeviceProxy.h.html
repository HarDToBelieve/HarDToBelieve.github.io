36namespace WebKit {
37
38class PlatformXRSystemProxy;
39
40struct XRDeviceInfo;
41
42class XRDeviceProxy final : public PlatformXR::Device {
43public:
44 static Ref<XRDeviceProxy> create(XRDeviceInfo&&, PlatformXRSystemProxy&);
45 XRDeviceIdentifier identifier() const { return m_identifier; }
46
47 void sessionDidEnd();
48
49private:
50 XRDeviceProxy(XRDeviceInfo&&, PlatformXRSystemProxy&);
51
52 WebCore::IntSize recommendedResolution(PlatformXR::SessionMode) final { return m_recommendedResolution; }
53 void initializeTrackingAndRendering(PlatformXR::SessionMode) final;
54 void shutDownTrackingAndRendering() final;
55 bool supportsSessionShutdownNotification() const final { return true; }
56 void initializeReferenceSpace(PlatformXR::ReferenceSpaceType) final { }
57 Vector<PlatformXR::Device::ViewData> views(PlatformXR::SessionMode) const final;
58 void requestFrame(PlatformXR::Device::RequestFrameCallback&&) final;
59 std::optional<PlatformXR::LayerHandle> createLayerProjection(uint32_t, uint32_t, bool) final;
60 void deleteLayer(PlatformXR::LayerHandle) override { };
61 void submitFrame(Vector<PlatformXR::Device::Layer>&&) final;
62
63 XRDeviceIdentifier m_identifier;
64 WeakPtr<PlatformXRSystemProxy> m_xrSystem;
65 bool m_supportsStereoRendering { false };
66 WebCore::IntSize m_recommendedResolution { 0, 0 };
67};

69} // namespace WebKit
