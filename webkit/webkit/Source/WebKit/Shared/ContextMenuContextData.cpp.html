38ContextMenuContextData::ContextMenuContextData()
39 : m_type(Type::ContextMenu)
40#if ENABLE(SERVICE_CONTROLS)
41 , m_selectionIsEditable(false)
42#endif
43{
44}
45
46ContextMenuContextData::ContextMenuContextData(const WebCore::IntPoint& menuLocation, const Vector<WebKit::WebContextMenuItemData>& menuItems, const ContextMenuContext& context)
47#if ENABLE(SERVICE_CONTROLS)
48 : m_type(context.controlledImage() ? Type::ServicesMenu : context.type())
49#else
50 : m_type(context.type())
51#endif
52 , m_menuLocation(menuLocation)
53 , m_menuItems(menuItems)
54 , m_webHitTestResultData(context.hitTestResult(), true)
55 , m_selectedText(context.selectedText())
56#if ENABLE(SERVICE_CONTROLS)
57 , m_selectionIsEditable(false)
58#endif
59{
60#if ENABLE(SERVICE_CONTROLS)
61 Image* image = context.controlledImage();
62 if (!image)
63 return;
64
65 // FIXME: figure out the rounding strategy for ShareableBitmap.
66 m_controlledImage = ShareableBitmap::createShareable(IntSize(image->size()), { });
67 auto graphicsContext = m_controlledImage->createGraphicsContext();
68 if (!graphicsContext)
69 return;
70 graphicsContext->drawImage(*image, IntPoint());
71#endif
72}

74void ContextMenuContextData::encode(IPC::Encoder& encoder) const
75{
76 encoder << m_type;
77 encoder << m_menuLocation;
78 encoder << m_menuItems;
79 encoder << m_webHitTestResultData;
80 encoder << m_selectedText;
81
82#if ENABLE(SERVICE_CONTROLS)
83 ShareableBitmap::Handle handle;
84 if (m_controlledImage)
85 m_controlledImage->createHandle(handle, SharedMemory::Protection::ReadOnly);
86 encoder << handle;
87 encoder << m_controlledSelectionData;
88 encoder << m_selectedTelephoneNumbers;
89 encoder << m_selectionIsEditable;
90#endif
91}
92
93bool ContextMenuContextData::decode(IPC::Decoder& decoder, ContextMenuContextData& result)
94{
95 if (!decoder.decode(result.m_type))
96 return false;
97
98 if (!decoder.decode(result.m_menuLocation))
99 return false;
100
101 if (!decoder.decode(result.m_menuItems))
102 return false;
103
104 if (!decoder.decode(result.m_webHitTestResultData))
105 return false;
106
107 if (!decoder.decode(result.m_selectedText))
108 return false;
109
110#if ENABLE(SERVICE_CONTROLS)
111 ShareableBitmap::Handle handle;
112 if (!decoder.decode(handle))
113 return false;
114
115 if (!handle.isNull())
116 result.m_controlledImage = ShareableBitmap::create(handle, SharedMemory::Protection::ReadOnly);
117
118 if (!decoder.decode(result.m_controlledSelectionData))
119 return false;
120 if (!decoder.decode(result.m_selectedTelephoneNumbers))
121 return false;
122 if (!decoder.decode(result.m_selectionIsEditable))
123 return false;
124#endif
125
126 return true;
127}
128
129#if ENABLE(SERVICE_CONTROLS)
130bool ContextMenuContextData::controlledDataIsEditable() const
131{
132 if (!m_controlledSelectionData.isEmpty())
133 return m_selectionIsEditable;
134
135 if (m_controlledImage)
136 return m_webHitTestResultData.isContentEditable;
137
138 return false;
139}
140#endif
