26#pragma once
27
28#include "SandboxExtension.h"
29#include <WebCore/IDBResultData.h>
30#include <wtf/Noncopyable.h>
31
32namespace WebKit {
33
34class WebIDBResult {
35 WTF_MAKE_NONCOPYABLE(WebIDBResult);
36public:
37 WebIDBResult()
38 {
39 }
40
41 WebIDBResult(const WebCore::IDBResultData& resultData)
42 : m_resultData(resultData)
43 {
44 }
45
46 WebIDBResult(const WebCore::IDBResultData& resultData, SandboxExtension::HandleArray&& handles)
47 : m_resultData(resultData)
48 , m_handles(WTFMove(handles))
49 {
50 }
51
52 WebIDBResult(WebIDBResult&&) = default;
53 WebIDBResult& operator=(WebIDBResult&&) = default;
54
55 const WebCore::IDBResultData& resultData() const { return m_resultData; }
56 const SandboxExtension::HandleArray& handles() const { return m_handles; }
57
58 void encode(IPC::Encoder&) const;
59 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, WebIDBResult&);
60
61private:
62 WebCore::IDBResultData m_resultData;
63 SandboxExtension::HandleArray m_handles;
64};
65
66} // namespace WebKit
