34DebuggableInfoData DebuggableInfoData::empty()
35{
36 return {
37 Inspector::DebuggableType::Page,
38 "Unknown"_s,
39 "Unknown"_s,
40 "Unknown"_s,
41 false
42 };
43}
44
45void DebuggableInfoData::encode(IPC::Encoder& encoder) const
46{
47 encoder << debuggableType;
48 encoder << targetPlatformName;
49 encoder << targetBuildVersion;
50 encoder << targetProductVersion;
51 encoder << targetIsSimulator;
52}
53
54std::optional<DebuggableInfoData> DebuggableInfoData::decode(IPC::Decoder& decoder)
55{
56 std::optional<Inspector::DebuggableType> debuggableType;
57 decoder >> debuggableType;
58 if (!debuggableType)
59 return std::nullopt;
60
61 std::optional<String> targetPlatformName;
62 decoder >> targetPlatformName;
63 if (!targetPlatformName)
64 return std::nullopt;
65
66 std::optional<String> targetBuildVersion;
67 decoder >> targetBuildVersion;
68 if (!targetBuildVersion)
69 return std::nullopt;
70
71 std::optional<String> targetProductVersion;
72 decoder >> targetProductVersion;
73 if (!targetProductVersion)
74 return std::nullopt;
75
76 std::optional<bool> targetIsSimulator;
77 decoder >> targetIsSimulator;
78 if (!targetIsSimulator)
79 return std::nullopt;
80
81 return {{
82 *debuggableType,
83 *targetPlatformName,
84 *targetBuildVersion,
85 *targetProductVersion,
86 *targetIsSimulator
87 }};
88}
