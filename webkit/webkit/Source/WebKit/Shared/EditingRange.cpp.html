36std::optional<WebCore::SimpleRange> EditingRange::toRange(WebCore::Frame& frame, const EditingRange& editingRange, EditingRangeIsRelativeTo base)
37{
38 ASSERT(editingRange.location != notFound);
39 WebCore::CharacterRange range { editingRange.location, editingRange.length };
40
41 if (base == EditingRangeIsRelativeTo::EditableRoot) {
42 // Our critical assumption is that this code path is called by input methods that
43 // concentrate on a given area containing the selection.
44 // We have to do this because of text fields and textareas. The DOM for those is not
45 // directly in the document DOM, so serialization is problematic. Our solution is
46 // to use the root editable element of the selection start as the positional base.
47 // That fits with AppKit's idea of an input context.
48 auto* element = frame.selection().rootEditableElementOrDocumentElement();
49 if (!element)
50 return std::nullopt;
51 return resolveCharacterRange(makeRangeSelectingNodeContents(*element), range);
52 }
53
54 ASSERT(base == EditingRangeIsRelativeTo::Paragraph);
55
56 auto paragraphStart = makeBoundaryPoint(startOfParagraph(frame.selection().selection().visibleStart()));
57 if (!paragraphStart)
58 return std::nullopt;
59
60 auto scopeEnd = makeBoundaryPointAfterNodeContents(paragraphStart->container->treeScope().rootNode());
61 return WebCore::resolveCharacterRange({ WTFMove(*paragraphStart), WTFMove(scopeEnd) }, range);
62}
63
64EditingRange EditingRange::fromRange(WebCore::Frame& frame, const std::optional<WebCore::SimpleRange>& range, EditingRangeIsRelativeTo editingRangeIsRelativeTo)
65{
66 ASSERT(editingRangeIsRelativeTo == EditingRangeIsRelativeTo::EditableRoot);
67
68 if (!range)
69 return { };
70
71 auto* element = frame.selection().rootEditableElementOrDocumentElement();
72 if (!element)
73 return { };
74
75 auto relativeRange = characterRange(makeBoundaryPointBeforeNodeContents(*element), *range);
76 return EditingRange(relativeRange.location, relativeRange.length);
77}

81namespace IPC {
82
83void ArgumentCoder<WebKit::EditingRange>::encode(Encoder& encoder, const WebKit::EditingRange& editingRange)
84{
85 encoder << editingRange.location;
86 encoder << editingRange.length;
87}
88
89std::optional<WebKit::EditingRange> ArgumentCoder<WebKit::EditingRange>::decode(Decoder& decoder)
90{
91 WebKit::EditingRange editingRange;
92
93 if (!decoder.decode(editingRange.location))
94 return std::nullopt;
95 if (!decoder.decode(editingRange.length))
96 return std::nullopt;
97
98 return editingRange;
99}
