35void EditorState::encode(IPC::Encoder& encoder) const
36{
37 encoder << transactionID;
38 encoder << originIdentifierForPasteboard;
39 encoder << shouldIgnoreSelectionChanges;
40 encoder << selectionIsNone;
41 encoder << selectionIsRange;
42 encoder << selectionIsRangeInsideImageOverlay;
43 encoder << isContentEditable;
44 encoder << isContentRichlyEditable;
45 encoder << isInPasswordField;
46 encoder << isInPlugin;
47 encoder << hasComposition;
48 encoder << isMissingPostLayoutData;
49 if (!isMissingPostLayoutData)
50 m_postLayoutData.encode(encoder);
51}
52
53bool EditorState::decode(IPC::Decoder& decoder, EditorState& result)
54{
55 if (!decoder.decode(result.transactionID))
56 return false;
57
58 if (!decoder.decode(result.originIdentifierForPasteboard))
59 return false;
60
61 if (!decoder.decode(result.shouldIgnoreSelectionChanges))
62 return false;
63
64 if (!decoder.decode(result.selectionIsNone))
65 return false;
66
67 if (!decoder.decode(result.selectionIsRange))
68 return false;
69
70 if (!decoder.decode(result.selectionIsRangeInsideImageOverlay))
71 return false;
72
73 if (!decoder.decode(result.isContentEditable))
74 return false;
75
76 if (!decoder.decode(result.isContentRichlyEditable))
77 return false;
78
79 if (!decoder.decode(result.isInPasswordField))
80 return false;
81
82 if (!decoder.decode(result.isInPlugin))
83 return false;
84
85 if (!decoder.decode(result.hasComposition))
86 return false;
87
88 if (!decoder.decode(result.isMissingPostLayoutData))
89 return false;
90
91 if (!result.isMissingPostLayoutData) {
92 if (!PostLayoutData::decode(decoder, result.postLayoutData()))
93 return false;
94 }
95
96 return true;
97}

99void EditorState::PostLayoutData::encode(IPC::Encoder& encoder) const
100{
101 encoder << typingAttributes;
102#if PLATFORM(IOS_FAMILY) || PLATFORM(GTK) || PLATFORM(WPE)
103 encoder << caretRectAtStart;
104#endif
105#if PLATFORM(COCOA)
106 encoder << selectedTextLength;
107 encoder << textAlignment;
108 encoder << textColor;
109 encoder << enclosingListType;
110 encoder << baseWritingDirection;
111#endif
112#if PLATFORM(IOS_FAMILY)
113 encoder << selectionClipRect;
114 encoder << caretRectAtEnd;
115 encoder << selectionGeometries;
116 encoder << markedTextRects;
117 encoder << markedText;
118 encoder << markedTextCaretRectAtStart;
119 encoder << markedTextCaretRectAtEnd;
120 encoder << wordAtSelection;
121 encoder << characterAfterSelection;
122 encoder << characterBeforeSelection;
123 encoder << twoCharacterBeforeSelection;
124#if USE(DICTATION_ALTERNATIVES)
125 encoder << dictationContextsForSelection;
126#endif
127 encoder << isReplaceAllowed;
128 encoder << hasContent;
129 encoder << isStableStateUpdate;
130 encoder << insideFixedPosition;
131 encoder << hasPlainText;
132 encoder << editableRootIsTransparentOrFullyClipped;
133 encoder << caretColor;
134 encoder << atStartOfSentence;
135 encoder << selectionStartIsAtParagraphBoundary;
136 encoder << selectionEndIsAtParagraphBoundary;
137#endif
138#if PLATFORM(MAC)
139 encoder << selectionBoundingRect;
140 encoder << candidateRequestStartPosition;
141 encoder << paragraphContextForCandidateRequest;
142 encoder << stringForCandidateRequest;
143 encoder << canEnableAutomaticSpellingCorrection;
144#endif
145#if PLATFORM(GTK) || PLATFORM(WPE)
146 encoder << surroundingContext;
147 encoder << surroundingContextCursorPosition;
148 encoder << surroundingContextSelectionPosition;
149#endif
150 encoder << fontAttributes;
151 encoder << canCut;
152 encoder << canCopy;
153 encoder << canPaste;
154}

156bool EditorState::PostLayoutData::decode(IPC::Decoder& decoder, PostLayoutData& result)
157{
158 if (!decoder.decode(result.typingAttributes))
159 return false;
160#if PLATFORM(IOS_FAMILY) || PLATFORM(GTK) || PLATFORM(WPE)
161 if (!decoder.decode(result.caretRectAtStart))
162 return false;
163#endif
164#if PLATFORM(COCOA)
165 if (!decoder.decode(result.selectedTextLength))
166 return false;
167 if (!decoder.decode(result.textAlignment))
168 return false;
169 if (!decoder.decode(result.textColor))
170 return false;
171 if (!decoder.decode(result.enclosingListType))
172 return false;
173 if (!decoder.decode(result.baseWritingDirection))
174 return false;
175#endif
176#if PLATFORM(IOS_FAMILY)
177 if (!decoder.decode(result.selectionClipRect))
178 return false;
179 if (!decoder.decode(result.caretRectAtEnd))
180 return false;
181 if (!decoder.decode(result.selectionGeometries))
182 return false;
183 if (!decoder.decode(result.markedTextRects))
184 return false;
185 if (!decoder.decode(result.markedText))
186 return false;
187 if (!decoder.decode(result.markedTextCaretRectAtStart))
188 return false;
189 if (!decoder.decode(result.markedTextCaretRectAtEnd))
190 return false;
191 if (!decoder.decode(result.wordAtSelection))
192 return false;
193 if (!decoder.decode(result.characterAfterSelection))
194 return false;
195 if (!decoder.decode(result.characterBeforeSelection))
196 return false;
197 if (!decoder.decode(result.twoCharacterBeforeSelection))
198 return false;
199#if USE(DICTATION_ALTERNATIVES)
200 if (!decoder.decode(result.dictationContextsForSelection))
201 return false;
202#endif
203 if (!decoder.decode(result.isReplaceAllowed))
204 return false;
205 if (!decoder.decode(result.hasContent))
206 return false;
207 if (!decoder.decode(result.isStableStateUpdate))
208 return false;
209 if (!decoder.decode(result.insideFixedPosition))
210 return false;
211 if (!decoder.decode(result.hasPlainText))
212 return false;
213 if (!decoder.decode(result.editableRootIsTransparentOrFullyClipped))
214 return false;
215 if (!decoder.decode(result.caretColor))
216 return false;
217 if (!decoder.decode(result.atStartOfSentence))
218 return false;
219 if (!decoder.decode(result.selectionStartIsAtParagraphBoundary))
220 return false;
221 if (!decoder.decode(result.selectionEndIsAtParagraphBoundary))
222 return false;
223#endif
224#if PLATFORM(MAC)
225 if (!decoder.decode(result.selectionBoundingRect))
226 return false;
227
228 if (!decoder.decode(result.candidateRequestStartPosition))
229 return false;
230
231 if (!decoder.decode(result.paragraphContextForCandidateRequest))
232 return false;
233
234 if (!decoder.decode(result.stringForCandidateRequest))
235 return false;
236
237 if (!decoder.decode(result.canEnableAutomaticSpellingCorrection))
238 return false;
239#endif
240#if PLATFORM(GTK) || PLATFORM(WPE)
241 if (!decoder.decode(result.surroundingContext))
242 return false;
243 if (!decoder.decode(result.surroundingContextCursorPosition))
244 return false;
245 if (!decoder.decode(result.surroundingContextSelectionPosition))
246 return false;
247#endif
248
249 std::optional<std::optional<FontAttributes>> optionalFontAttributes;
250 decoder >> optionalFontAttributes;
251 if (!optionalFontAttributes)
252 return false;
253
254 result.fontAttributes = optionalFontAttributes.value();
255
256 if (!decoder.decode(result.canCut))
257 return false;
258 if (!decoder.decode(result.canCopy))
259 return false;
260 if (!decoder.decode(result.canPaste))
261 return false;
262
263 return true;
264}

266TextStream& operator<<(TextStream& ts, const EditorState& editorState)
267{
268 if (editorState.shouldIgnoreSelectionChanges)
269 ts.dumpProperty("shouldIgnoreSelectionChanges", editorState.shouldIgnoreSelectionChanges);
270 if (!editorState.selectionIsNone)
271 ts.dumpProperty("selectionIsNone", editorState.selectionIsNone);
272 if (editorState.selectionIsRange)
273 ts.dumpProperty("selectionIsRange", editorState.selectionIsRange);
274 if (editorState.isContentEditable)
275 ts.dumpProperty("isContentEditable", editorState.isContentEditable);
276 if (editorState.isContentRichlyEditable)
277 ts.dumpProperty("isContentRichlyEditable", editorState.isContentRichlyEditable);
278 if (editorState.isInPasswordField)
279 ts.dumpProperty("isInPasswordField", editorState.isInPasswordField);
280 if (editorState.isInPlugin)
281 ts.dumpProperty("isInPlugin", editorState.isInPlugin);
282 if (editorState.hasComposition)
283 ts.dumpProperty("hasComposition", editorState.hasComposition);
284 if (editorState.isMissingPostLayoutData)
285 ts.dumpProperty("isMissingPostLayoutData", editorState.isMissingPostLayoutData);
286
287 if (editorState.isMissingPostLayoutData)
288 return ts;
289
290 TextStream::GroupScope scope(ts);
291 ts << "postLayoutData";
292 if (editorState.postLayoutData().typingAttributes != AttributeNone)
293 ts.dumpProperty("typingAttributes", editorState.postLayoutData().typingAttributes);
294#if PLATFORM(IOS_FAMILY) || PLATFORM(GTK) || PLATFORM(WPE)
295 if (editorState.postLayoutData().caretRectAtStart != IntRect())
296 ts.dumpProperty("caretRectAtStart", editorState.postLayoutData().caretRectAtStart);
297#endif
298#if PLATFORM(COCOA)
299 if (editorState.postLayoutData().selectedTextLength)
300 ts.dumpProperty("selectedTextLength", editorState.postLayoutData().selectedTextLength);
301 if (editorState.postLayoutData().textAlignment != NoAlignment)
302 ts.dumpProperty("textAlignment", editorState.postLayoutData().textAlignment);
303 if (editorState.postLayoutData().textColor.isValid())
304 ts.dumpProperty("textColor", editorState.postLayoutData().textColor);
305 if (editorState.postLayoutData().enclosingListType != NoList)
306 ts.dumpProperty("enclosingListType", editorState.postLayoutData().enclosingListType);
307 if (editorState.postLayoutData().baseWritingDirection != WebCore::WritingDirection::Natural)
308 ts.dumpProperty("baseWritingDirection", static_cast<uint8_t>(editorState.postLayoutData().baseWritingDirection));
309#endif // PLATFORM(COCOA)
310#if PLATFORM(IOS_FAMILY)
311 if (editorState.postLayoutData().selectionClipRect != IntRect())
312 ts.dumpProperty("selectionClipRect", editorState.postLayoutData().selectionClipRect);
313 if (editorState.postLayoutData().caretRectAtEnd != IntRect())
314 ts.dumpProperty("caretRectAtEnd", editorState.postLayoutData().caretRectAtEnd);
315 if (!editorState.postLayoutData().selectionGeometries.isEmpty())
316 ts.dumpProperty("selectionGeometries", editorState.postLayoutData().selectionGeometries);
317 if (!editorState.postLayoutData().markedTextRects.isEmpty())
318 ts.dumpProperty("markedTextRects", editorState.postLayoutData().markedTextRects);
319 if (editorState.postLayoutData().markedText.length())
320 ts.dumpProperty("markedText", editorState.postLayoutData().markedText);
321 if (editorState.postLayoutData().markedTextCaretRectAtStart != IntRect())
322 ts.dumpProperty("markedTextCaretRectAtStart", editorState.postLayoutData().markedTextCaretRectAtStart);
323 if (editorState.postLayoutData().markedTextCaretRectAtEnd != IntRect())
324 ts.dumpProperty("markedTextCaretRectAtEnd", editorState.postLayoutData().markedTextCaretRectAtEnd);
325 if (editorState.postLayoutData().wordAtSelection.length())
326 ts.dumpProperty("wordAtSelection", editorState.postLayoutData().wordAtSelection);
327 if (editorState.postLayoutData().characterAfterSelection)
328 ts.dumpProperty("characterAfterSelection", editorState.postLayoutData().characterAfterSelection);
329 if (editorState.postLayoutData().characterBeforeSelection)
330 ts.dumpProperty("characterBeforeSelection", editorState.postLayoutData().characterBeforeSelection);
331 if (editorState.postLayoutData().twoCharacterBeforeSelection)
332 ts.dumpProperty("twoCharacterBeforeSelection", editorState.postLayoutData().twoCharacterBeforeSelection);
333
334 if (editorState.postLayoutData().isReplaceAllowed)
335 ts.dumpProperty("isReplaceAllowed", editorState.postLayoutData().isReplaceAllowed);
336 if (editorState.postLayoutData().hasContent)
337 ts.dumpProperty("hasContent", editorState.postLayoutData().hasContent);
338 ts.dumpProperty("isStableStateUpdate", editorState.postLayoutData().isStableStateUpdate);
339 if (editorState.postLayoutData().insideFixedPosition)
340 ts.dumpProperty("insideFixedPosition", editorState.postLayoutData().insideFixedPosition);
341 if (editorState.postLayoutData().caretColor.isValid())
342 ts.dumpProperty("caretColor", editorState.postLayoutData().caretColor);
343#endif
344#if PLATFORM(MAC)
345 if (editorState.postLayoutData().selectionBoundingRect != IntRect())
346 ts.dumpProperty("selectionBoundingRect", editorState.postLayoutData().selectionBoundingRect);
347 if (editorState.postLayoutData().candidateRequestStartPosition)
348 ts.dumpProperty("candidateRequestStartPosition", editorState.postLayoutData().candidateRequestStartPosition);
349 if (editorState.postLayoutData().paragraphContextForCandidateRequest.length())
350 ts.dumpProperty("paragraphContextForCandidateRequest", editorState.postLayoutData().paragraphContextForCandidateRequest);
351 if (editorState.postLayoutData().stringForCandidateRequest.length())
352 ts.dumpProperty("stringForCandidateRequest", editorState.postLayoutData().stringForCandidateRequest);
353 if (editorState.postLayoutData().canEnableAutomaticSpellingCorrection)
354 ts.dumpProperty("canEnableAutomaticSpellingCorrection", editorState.postLayoutData().canEnableAutomaticSpellingCorrection);
355#endif
356
357 if (editorState.postLayoutData().canCut)
358 ts.dumpProperty("canCut", editorState.postLayoutData().canCut);
359 if (editorState.postLayoutData().canCopy)
360 ts.dumpProperty("canCopy", editorState.postLayoutData().canCopy);
361 if (editorState.postLayoutData().canPaste)
362 ts.dumpProperty("canPaste", editorState.postLayoutData().canPaste);
363
364 return ts;
365}
