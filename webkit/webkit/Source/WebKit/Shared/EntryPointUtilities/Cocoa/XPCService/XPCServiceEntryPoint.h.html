51 XPCServiceInitializerDelegate(OSObjectPtr<xpc_connection_t> connection, xpc_object_t initializerMessage)
52 : m_connection(WTFMove(connection))

65 virtual bool getClientSDKVersion(uint32_t& clientSDKVersion);
66 virtual bool getClientProcessName(String& clientProcessName);

75};

87template<typename XPCServiceType, typename XPCServiceInitializerDelegateType>
88void XPCServiceInitializer(OSObjectPtr<xpc_connection_t> connection, xpc_object_t initializerMessage, xpc_object_t priorityBoostMessage)
89{
90 if (initializerMessage) {
91 if (xpc_dictionary_get_bool(initializerMessage, "configure-jsc-for-testing"))
92 JSC::Config::configureForTesting();
93 if (xpc_dictionary_get_bool(initializerMessage, "disable-jit"))
94 JSC::ExecutableAllocator::setJITEnabled(false);
95 }
96
97 XPCServiceInitializerDelegateType delegate(WTFMove(connection), initializerMessage);
98
99 // We don't want XPC to be in charge of whether the process should be terminated or not,
100 // so ensure that we have an outstanding transaction here. This is not needed on iOS because
101 // the UIProcess takes process assertions on behalf of its child processes.
102#if PLATFORM(MAC)
103 osTransaction() = adoptOSObject(os_transaction_create("WebKit XPC Service"));
104#endif
105
106 InitializeWebKit2();
107
108 if (!delegate.checkEntitlements())
109 exit(EXIT_FAILURE);
110
111 AuxiliaryProcessInitializationParameters parameters;
112 if (priorityBoostMessage)
113 parameters.priorityBoostMessage = priorityBoostMessage;
114
115 if (!delegate.getConnectionIdentifier(parameters.connectionIdentifier))
116 exit(EXIT_FAILURE);
117
118 if (!delegate.getClientIdentifier(parameters.clientIdentifier))
119 exit(EXIT_FAILURE);
120
121 // The host process may not have a bundle identifier (e.g. a command line app), so don't require one.
122 delegate.getClientBundleIdentifier(parameters.clientBundleIdentifier);
123
124 if (!delegate.getClientSDKVersion(parameters.clientSDKVersion))
125 exit(EXIT_FAILURE);
126
127 WebCore::ProcessIdentifier processIdentifier;
128 if (!delegate.getProcessIdentifier(processIdentifier))
129 exit(EXIT_FAILURE);
130 parameters.processIdentifier = processIdentifier;
131
132 if (!delegate.getClientProcessName(parameters.uiProcessName))
133 exit(EXIT_FAILURE);
134
135 if (!delegate.getExtraInitializationData(parameters.extraInitializationData))
136 exit(EXIT_FAILURE);
137
138 // Set the task default voucher to the current value (as propagated by XPC).
139 voucher_replace_default_voucher();
140
141#if HAVE(QOS_CLASSES)
142 if (parameters.extraInitializationData.contains("always-runs-at-background-priority"_s))
143 Thread::setGlobalMaxQOSClass(QOS_CLASS_UTILITY);
144#endif
145
146 parameters.processType = XPCServiceType::processType;
147
148 initializeAuxiliaryProcess<XPCServiceType>(WTFMove(parameters));
149}
