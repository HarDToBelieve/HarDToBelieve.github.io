42bool XPCServiceInitializerDelegate::checkEntitlements()
43{
44#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
45 if (isClientSandboxed()) {
46 // FIXME(<rdar://problem/54178641>): Remove this check once WebKit can work without network access.
47 if (hasEntitlement("com.apple.security.network.client"))
48 return true;
49
50 audit_token_t auditToken = { };
51 xpc_connection_get_audit_token(m_connection.get(), &auditToken);
52 if (auto rc = sandbox_check_by_audit_token(auditToken, "mach-lookup", static_cast<enum sandbox_filter_type>(SANDBOX_FILTER_GLOBAL_NAME | SANDBOX_CHECK_NO_REPORT), "com.apple.nsurlsessiond")) {
53 // FIXME (rdar://problem/54178641): This requirement is too strict, it should be possible to load file:// resources without network access.
54 NSLog(@"Application does not have permission to communicate with network resources. rc=%d : errno=%d", rc, errno);
55 return false;
56 }
57 }
58#endif
59 return true;
60}

62bool XPCServiceInitializerDelegate::getConnectionIdentifier(IPC::Connection::Identifier& identifier)
63{
64 mach_port_t port = xpc_dictionary_copy_mach_send(m_initializerMessage, "server-port");
65 if (!MACH_PORT_VALID(port))
66 return false;
67
68 identifier = IPC::Connection::Identifier(port, m_connection);
69 return true;
70}

91bool XPCServiceInitializerDelegate::getProcessIdentifier(WebCore::ProcessIdentifier& identifier)
92{
93 auto parsedIdentifier = parseInteger<uint64_t>(xpc_dictionary_get_string(m_initializerMessage, "process-identifier"));

107bool XPCServiceInitializerDelegate::getExtraInitializationData(HashMap<String, String>& extraInitializationData)
108{
109 xpc_object_t extraDataInitializationDataObject = xpc_dictionary_get_value(m_initializerMessage, "extra-initialization-data");
110
111 String inspectorProcess = xpc_dictionary_get_string(extraDataInitializationDataObject, "inspector-process");
112 if (!inspectorProcess.isEmpty())
113 extraInitializationData.add("inspector-process"_s, inspectorProcess);
114
115#if ENABLE(SERVICE_WORKER)
116 String serviceWorkerProcess = xpc_dictionary_get_string(extraDataInitializationDataObject, "service-worker-process");
117 if (!serviceWorkerProcess.isEmpty())
118 extraInitializationData.add("service-worker-process"_s, WTFMove(serviceWorkerProcess));
119 String registrableDomain = xpc_dictionary_get_string(extraDataInitializationDataObject, "registrable-domain");
120 if (!registrableDomain.isEmpty())
121 extraInitializationData.add("registrable-domain"_s, WTFMove(registrableDomain));
122#endif
123
124 String isPrewarmedProcess = xpc_dictionary_get_string(extraDataInitializationDataObject, "is-prewarmed");
125 if (!isPrewarmedProcess.isEmpty())
126 extraInitializationData.add("is-prewarmed"_s, isPrewarmedProcess);
127
128 if (!isClientSandboxed()) {
129 String userDirectorySuffix = xpc_dictionary_get_string(extraDataInitializationDataObject, "user-directory-suffix");
130 if (!userDirectorySuffix.isEmpty())
131 extraInitializationData.add("user-directory-suffix"_s, userDirectorySuffix);
132 }
133
134 String alwaysRunsAtBackgroundPriority = xpc_dictionary_get_string(extraDataInitializationDataObject, "always-runs-at-background-priority");
135 if (!alwaysRunsAtBackgroundPriority.isEmpty())
136 extraInitializationData.add("always-runs-at-background-priority"_s, alwaysRunsAtBackgroundPriority);
137
138 return true;
139}

159void XPCServiceExit(OSObjectPtr<xpc_object_t>&& priorityBoostMessage)
160{
161 // Make sure to destroy the priority boost message to avoid leaking a transaction.
162 priorityBoostMessage = nullptr;
163
164#if PLATFORM(MAC)
165 osTransaction() = nullptr;
166#endif
167
168 xpc_transaction_exit_clean();
169}
