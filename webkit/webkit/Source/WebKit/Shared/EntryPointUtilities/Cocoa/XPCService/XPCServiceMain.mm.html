42static void setAppleLanguagesPreference()
43{
44 auto bootstrap = adoptOSObject(xpc_copy_bootstrap());
45 if (!bootstrap)
46 return;
47
48 if (xpc_object_t languages = xpc_dictionary_get_value(bootstrap.get(), "OverrideLanguages")) {
49 @autoreleasepool {
50 NSDictionary *existingArguments = [[NSUserDefaults standardUserDefaults] volatileDomainForName:NSArgumentDomain];
51 auto newArguments = adoptNS([existingArguments mutableCopy]);
52 RetainPtr<NSMutableArray> newLanguages = adoptNS([[NSMutableArray alloc] init]);
53 xpc_array_apply(languages, ^(size_t index, xpc_object_t value) {
54 [newLanguages addObject:[NSString stringWithCString:xpc_string_get_string_ptr(value) encoding:NSUTF8StringEncoding]];
55 return true;
56 });
57 [newArguments setValue:newLanguages.get() forKey:@"AppleLanguages"];
58 [[NSUserDefaults standardUserDefaults] setVolatileDomain:newArguments.get() forName:NSArgumentDomain];
59 }
60 }

63static void XPCServiceEventHandler(xpc_connection_t peer)
64{
65 static NeverDestroyed<OSObjectPtr<xpc_object_t>> priorityBoostMessage;
66
67 OSObjectPtr<xpc_connection_t> retainedPeerConnection(peer);
68
69 xpc_connection_set_target_queue(peer, dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0));
70 xpc_connection_set_event_handler(peer, ^(xpc_object_t event) {
71 xpc_type_t type = xpc_get_type(event);
72 if (type != XPC_TYPE_DICTIONARY) {
73 RELEASE_LOG_ERROR(IPC, "XPCServiceEventHandler: Received unexpected XPC event type: %{public}s", xpc_type_get_name(type));
74 if (type == XPC_TYPE_ERROR) {
75 if (event == XPC_ERROR_CONNECTION_INVALID || event == XPC_ERROR_TERMINATION_IMMINENT) {
76 RELEASE_LOG_ERROR(IPC, "Exiting: Received XPC event type: %{public}s", event == XPC_ERROR_CONNECTION_INVALID ? "XPC_ERROR_CONNECTION_INVALID" : "XPC_ERROR_TERMINATION_IMMINENT");
77 // FIXME: Handle this case more gracefully.
78 [[NSRunLoop mainRunLoop] performBlock:^{
79 exit(EXIT_FAILURE);
80 }];
81 }
82 }
83 return;
84 }
85
86 auto* messageName = xpc_dictionary_get_string(event, "message-name");
87 if (!messageName) {
88 RELEASE_LOG_ERROR(IPC, "XPCServiceEventHandler: 'message-name' is not present in the XPC dictionary");
89 return;
90 }
91 if (!strcmp(messageName, "bootstrap")) {
92 const char* serviceName = xpc_dictionary_get_string(event, "service-name");
93 if (!serviceName) {
94 RELEASE_LOG_ERROR(IPC, "XPCServiceEventHandler: 'service-name' is not present in the XPC dictionary");
95 return;
96 }
97 CFStringRef entryPointFunctionName = nullptr;
98 if (!strncmp(serviceName, "com.apple.WebKit.WebContent", strlen("com.apple.WebKit.WebContent")))
99 entryPointFunctionName = CFSTR(STRINGIZE_VALUE_OF(WEBCONTENT_SERVICE_INITIALIZER));
100 else if (!strcmp(serviceName, "com.apple.WebKit.Networking"))
101 entryPointFunctionName = CFSTR(STRINGIZE_VALUE_OF(NETWORK_SERVICE_INITIALIZER));
102 else if (!strcmp(serviceName, "com.apple.WebKit.Plugin.64"))
103 entryPointFunctionName = CFSTR(STRINGIZE_VALUE_OF(PLUGIN_SERVICE_INITIALIZER));
104 else if (!strcmp(serviceName, "com.apple.WebKit.GPU"))
105 entryPointFunctionName = CFSTR(STRINGIZE_VALUE_OF(GPU_SERVICE_INITIALIZER));
106 else if (!strcmp(serviceName, "com.apple.WebKit.WebAuthn"))
107 entryPointFunctionName = CFSTR(STRINGIZE_VALUE_OF(WEBAUTHN_SERVICE_INITIALIZER));
108 else {
109 RELEASE_LOG_ERROR(IPC, "XPCServiceEventHandler: Unexpected 'service-name': %{public}s", serviceName);
110 return;
111 }
112
113 CFBundleRef webKitBundle = CFBundleGetBundleWithIdentifier(CFSTR("com.apple.WebKit"));
114 typedef void (*InitializerFunction)(xpc_connection_t, xpc_object_t, xpc_object_t);
115 InitializerFunction initializerFunctionPtr = reinterpret_cast<InitializerFunction>(CFBundleGetFunctionPointerForName(webKitBundle, entryPointFunctionName));
116 if (!initializerFunctionPtr) {
117 RELEASE_LOG_FAULT(IPC, "Exiting: Unable to find entry point in WebKit.framework with name: %s", [(__bridge NSString *)entryPointFunctionName UTF8String]);
118 [[NSRunLoop mainRunLoop] performBlock:^{
119 exit(EXIT_FAILURE);
120 }];
121 return;
122 }
123
124 auto reply = adoptOSObject(xpc_dictionary_create_reply(event));
125 xpc_dictionary_set_string(reply.get(), "message-name", "process-finished-launching");
126 xpc_connection_send_message(xpc_dictionary_get_remote_connection(event), reply.get());
127
128 int fd = xpc_dictionary_dup_fd(event, "stdout");
129 if (fd != -1)
130 dup2(fd, STDOUT_FILENO);
131
132 fd = xpc_dictionary_dup_fd(event, "stderr");
133 if (fd != -1)
134 dup2(fd, STDERR_FILENO);
135
136 WorkQueue::main().dispatchSync([initializerFunctionPtr, event = OSObjectPtr<xpc_object_t>(event), retainedPeerConnection] {
137 initializerFunctionPtr(retainedPeerConnection.get(), event.get(), priorityBoostMessage.get().get());
138
139 setAppleLanguagesPreference();
140 });
141
142 priorityBoostMessage.get() = nullptr;
143 return;
144 }
145
146 // Leak a boost onto the NetworkProcess.
147 if (!strcmp(messageName, "pre-bootstrap")) {
148 assert(!priorityBoostMessage.get());
149 priorityBoostMessage.get() = event;
150 return;
151 }
152
153 handleXPCEndpointMessages(event, messageName);
154 });
155
156 xpc_connection_resume(peer);
157}

168#if ENABLE(CFPREFS_DIRECT_MODE)
169static bool shouldEnableCFPrefsDirectMode(int argc, const char** argv)
170{
171 if (argc <= 0 || !argv[0])
172 return false;
173 if (strstr(argv[0], "com.apple.WebKit.WebContent"))
174 return true;
175 if (strstr(argv[0], "com.apple.WebKit.GPU"))
176 return true;
177 return false;
178}
179#endif
180
181int XPCServiceMain(int argc, const char** argv)
182{
183 ASSERT(argc >= 1);
184 ASSERT(argv[0]);
185#if ENABLE(CFPREFS_DIRECT_MODE)
186 if (shouldEnableCFPrefsDirectMode(argc, argv)) {
187 // Enable CFPrefs direct mode to avoid unsuccessfully attempting to connect to the daemon and getting blocked by the sandbox.
188 _CFPrefsSetDirectModeEnabled(YES);
189#if HAVE(CF_PREFS_SET_READ_ONLY)
190 _CFPrefsSetReadOnly(YES);
191#endif
192 }
193#else
194 UNUSED_PARAM(argc);
195 UNUSED_PARAM(argv);
196#endif
197
198 WTF::initializeMainThread();
199
200 auto bootstrap = adoptOSObject(xpc_copy_bootstrap());
201
202 if (bootstrap) {
203#if PLATFORM(IOS_FAMILY)
204 auto containerEnvironmentVariables = xpc_dictionary_get_value(bootstrap.get(), "ContainerEnvironmentVariables");
205 xpc_dictionary_apply(containerEnvironmentVariables, ^(const char *key, xpc_object_t value) {
206 setenv(key, xpc_string_get_string_ptr(value), 1);
207 return true;
208 });
209#endif
210#if PLATFORM(MAC)
211#if ASAN_ENABLED
212 // EXC_RESOURCE on ASAN builds freezes the process for several minutes: rdar://65027596
213 if (char *disableFreezingOnExcResource = getenv("DISABLE_FREEZING_ON_EXC_RESOURCE")) {
214 if (!strcasecmp(disableFreezingOnExcResource, "yes") || !strcasecmp(disableFreezingOnExcResource, "true") || !strcasecmp(disableFreezingOnExcResource, "1")) {
215 int val = 1;
216 int rc = sysctlbyname("debug.toggle_address_reuse", nullptr, 0, &val, sizeof(val));
217 if (rc < 0)
218 WTFLogAlways("failed to set debug.toggle_address_reuse: %d\n", rc);
219 else
220 WTFLogAlways("debug.toggle_address_reuse is now 1.\n");
221 }
222 }
223#endif
224 String webKitBundleVersion = xpc_dictionary_get_string(bootstrap.get(), "WebKitBundleVersion");
225 String expectedBundleVersion = [NSBundle bundleWithIdentifier:@"com.apple.WebKit"].infoDictionary[(__bridge NSString *)kCFBundleVersionKey];
226 if (!webKitBundleVersion.isNull() && !expectedBundleVersion.isNull() && webKitBundleVersion != expectedBundleVersion) {
227 auto errorMessage = makeString("WebKit framework version mismatch: ", webKitBundleVersion, " != ", expectedBundleVersion);
228 logAndSetCrashLogMessage(errorMessage.utf8().data());
229 crashDueWebKitFrameworkVersionMismatch();
230 }
231#endif
232 }
233
234#if PLATFORM(MAC)
235 // Don't allow Apple Events in WebKit processes. This can be removed when <rdar://problem/14012823> is fixed.
236 setenv("__APPLEEVENTSSERVICENAME", "", 1);
237#endif
238
239 xpc_main(XPCServiceEventHandler);
240 return 0;
241}
