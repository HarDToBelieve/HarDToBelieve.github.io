31namespace WebKit {
32
33void FrameInfoData::encode(IPC::Encoder& encoder) const
34{
35 encoder << isMainFrame;
36 encoder << request;
37 encoder << securityOrigin;
38 encoder << frameID;
39 encoder << parentFrameID;
40}
41
42std::optional<FrameInfoData> FrameInfoData::decode(IPC::Decoder& decoder)
43{
44 std::optional<bool> isMainFrame;
45 decoder >> isMainFrame;
46 if (!isMainFrame)
47 return std::nullopt;
48
49 std::optional<WebCore::ResourceRequest> request;
50 decoder >> request;
51 if (!request)
52 return std::nullopt;
53
54 std::optional<WebCore::SecurityOriginData> securityOrigin;
55 decoder >> securityOrigin;
56 if (!securityOrigin)
57 return std::nullopt;
58
59 std::optional<std::optional<WebCore::FrameIdentifier>> frameID;
60 decoder >> frameID;
61 if (!frameID)
62 return std::nullopt;
63
64 std::optional<std::optional<WebCore::FrameIdentifier>> parentFrameID;
65 decoder >> parentFrameID;
66 if (!parentFrameID)
67 return std::nullopt;
68
69 return {{
70 WTFMove(*isMainFrame),
71 WTFMove(*request),
72 WTFMove(*securityOrigin),
73 WTFMove(*frameID),
74 WTFMove(*parentFrameID)
75 }};
76}
