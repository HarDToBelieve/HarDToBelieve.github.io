38GamepadData::GamepadData(unsigned index, const String& id, const String& mapping, const Vector<SharedGamepadValue>& axisValues, const Vector<SharedGamepadValue>& buttonValues, MonotonicTime lastUpdateTime)
39 : m_index(index)
40 , m_id(id)
41 , m_mapping(mapping)
42 , m_axisValues(WTF::map(axisValues, [](const auto& value) { return value.value(); }))
43 , m_buttonValues(WTF::map(buttonValues, [](const auto& value) { return value.value(); }))
44 , m_lastUpdateTime(lastUpdateTime)
45{
46}
47
48void GamepadData::encode(IPC::Encoder& encoder) const
49{
50 encoder << m_isNull;
51 if (m_isNull)
52 return;
53
54 encoder << m_index << m_id << m_mapping << m_axisValues << m_buttonValues << m_lastUpdateTime;
55}
56
57std::optional<GamepadData> GamepadData::decode(IPC::Decoder& decoder)
58{
59 GamepadData data;
60 if (!decoder.decode(data.m_isNull))
61 return std::nullopt;
62
63 if (data.m_isNull)
64 return data;
65
66 if (!decoder.decode(data.m_index))
67 return std::nullopt;
68
69 if (!decoder.decode(data.m_id))
70 return std::nullopt;
71
72 if (!decoder.decode(data.m_mapping))
73 return std::nullopt;
74
75 if (!decoder.decode(data.m_axisValues))
76 return std::nullopt;
77
78 if (!decoder.decode(data.m_buttonValues))
79 return std::nullopt;
80
81 if (!decoder.decode(data.m_lastUpdateTime))
82 return std::nullopt;
83
84 return data;
85}

89String GamepadData::loggingString() const
90{
91 StringBuilder builder;
92
93 builder.append(m_axisValues.size(), " axes, ", m_buttonValues.size(), " buttons\n");
94
95 for (size_t i = 0; i < m_axisValues.size(); ++i)
96 builder.append(" Axis ", i, ": ", m_axisValues[i]);
97
98 builder.append('\n');
99
100 for (size_t i = 0; i < m_buttonValues.size(); ++i)
101 builder.append(" Button ", i, ": ", FormattedNumber::fixedPrecision(m_buttonValues[i]));
102
103 return builder.toString();
104}
