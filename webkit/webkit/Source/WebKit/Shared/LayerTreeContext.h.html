36namespace WebKit {

38enum class LayerHostingMode : uint8_t {
39 InProcess,
40#if HAVE(OUT_OF_PROCESS_LAYER_HOSTING)
41 OutOfProcess
42#endif
43};
44
45class LayerTreeContext {
46public:
47 LayerTreeContext();
48 ~LayerTreeContext();
49
50 void encode(IPC::Encoder&) const;
51 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, LayerTreeContext&);
52
53 bool isEmpty() const;
54
55 uint64_t contextID;
56};
57
58bool operator==(const LayerTreeContext&, const LayerTreeContext&);
59
60inline bool operator!=(const LayerTreeContext& a, const LayerTreeContext& b)
61{
62 return !(a == b);
63}
64
65}

72 WebKit::LayerHostingMode::OutOfProcess,
73#endif
74 WebKit::LayerHostingMode::InProcess
75 >;
76};
