38std::unique_ptr<KeyedDecoder> createForFile(const String& path)
39{
40 ASSERT(!RunLoop::isMain());
41
42 auto handle = FileSystem::openAndLockFile(path, FileSystem::FileOpenMode::Read);
43 if (handle == FileSystem::invalidPlatformFileHandle)
44 return nullptr;
45
46 auto fileSize = FileSystem::fileSize(handle).value_or(0);
47 if (!fileSize) {
48 FileSystem::unlockAndCloseFile(handle);
49 return nullptr;
50 }
51
52 size_t bytesToRead;
53 if (!WTF::convertSafely(fileSize, bytesToRead)) {
54 FileSystem::unlockAndCloseFile(handle);
55 return nullptr;
56 }
57
58 Vector<uint8_t> buffer(bytesToRead);
59 size_t totalBytesRead = FileSystem::readFromFile(handle, buffer.data(), buffer.size());
60
61 FileSystem::unlockAndCloseFile(handle);
62
63 if (totalBytesRead != bytesToRead)
64 return nullptr;
65
66 // FIXME: We should try to modify the constructor to pass &&.
67 return KeyedDecoder::decoder(buffer.data(), buffer.size());
68}

70void writeToDisk(std::unique_ptr<KeyedEncoder>&& encoder, String&& path)
71{
72 ASSERT(!RunLoop::isMain());
73
74 auto rawData = encoder->finishEncoding();
75 if (!rawData)
76 return;
77
78 FileSystem::PlatformFileHandle handle = FileSystem::openAndLockFile(path, FileSystem::FileOpenMode::Write);
79 if (handle == FileSystem::invalidPlatformFileHandle)
80 return;
81
82 auto writtenBytes = FileSystem::writeToFile(handle, rawData->data(), rawData->size());
83 FileSystem::unlockAndCloseFile(handle);
84
85 if (writtenBytes != static_cast<int64_t>(rawData->size()))
86 RELEASE_LOG_ERROR(DiskPersistency, "Disk persistency: We only wrote %d out of %zu bytes to disk", static_cast<unsigned>(writtenBytes), rawData->size());
87}
