36namespace WebKit {

38
39NPIdentifierData::NPIdentifierData()
40 : m_isString(false)
41 , m_number(0)
42{
43}
44
45
46NPIdentifierData NPIdentifierData::fromNPIdentifier(NPIdentifier npIdentifier)
47{
48 NPIdentifierData npIdentifierData;
49
50 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(npIdentifier);
51 npIdentifierData.m_isString = identifierRep->isString();
52
53 if (npIdentifierData.m_isString)
54 npIdentifierData.m_string = identifierRep->string();
55 else
56 npIdentifierData.m_number = identifierRep->number();
57
58 return npIdentifierData;
59}
60
61NPIdentifier NPIdentifierData::createNPIdentifier() const
62{
63 if (m_isString)
64 return static_cast<NPIdentifier>(IdentifierRep::get(m_string.data()));
65
66 return static_cast<NPIdentifier>(IdentifierRep::get(m_number));
67}
68
69void NPIdentifierData::encode(IPC::Encoder& encoder) const
70{
71 encoder << m_isString;
72 if (m_isString)
73 encoder << m_string;
74 else
75 encoder << m_number;
76}
77
78std::optional<NPIdentifierData> NPIdentifierData::decode(IPC::Decoder& decoder)
79{
80 NPIdentifierData result;
81 if (!decoder.decode(result.m_isString))
82 return std::nullopt;
83
84 if (result.m_isString) {
85 if (!decoder.decode(result.m_string))
86 return std::nullopt;
87 return result;
88 }
89
90 if (!decoder.decode(result.m_number))
91 return std::nullopt;
92 return result;
93}
