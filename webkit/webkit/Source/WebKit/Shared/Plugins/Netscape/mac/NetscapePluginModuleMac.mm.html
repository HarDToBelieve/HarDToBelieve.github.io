40static bool getPluginArchitecture(CFBundleRef bundle, PluginModuleInfo& plugin)
41{
42 RetainPtr<CFArrayRef> pluginArchitecturesArray = adoptCF(CFBundleCopyExecutableArchitectures(bundle));
43 if (!pluginArchitecturesArray)
44 return false;
45
46 // Turn the array into a set.
47 HashSet<unsigned> architectures;
48 for (CFIndex i = 0, numPluginArchitectures = CFArrayGetCount(pluginArchitecturesArray.get()); i < numPluginArchitectures; ++i) {
49 CFNumberRef number = static_cast<CFNumberRef>(CFArrayGetValueAtIndex(pluginArchitecturesArray.get(), i));
50
51 SInt32 architecture;
52 if (!CFNumberGetValue(number, kCFNumberSInt32Type, &architecture))
53 continue;
54 architectures.add(architecture);
55 }
56
57#ifdef __x86_64__
58 // We only support 64-bit Intel plug-ins on 64-bit Intel.
59 if (architectures.contains(kCFBundleExecutableArchitectureX86_64)) {
60 plugin.pluginArchitecture = CPU_TYPE_X86_64;
61 return true;
62 }
63#endif
64
65 return false;
66}

68static bool getPluginInfoFromPropertyLists(CFBundleRef bundle, PluginModuleInfo& plugin)
69{
70 RetainPtr<CFDictionaryRef> mimeTypes = static_cast<CFDictionaryRef>(CFBundleGetValueForInfoDictionaryKey(bundle, CFSTR("WebPluginMIMETypes")));
71 if (!mimeTypes || CFGetTypeID(mimeTypes.get()) != CFDictionaryGetTypeID())
72 return false;
73
74 // Get the plug-in name.
75 CFStringRef pluginName = static_cast<CFStringRef>(CFBundleGetValueForInfoDictionaryKey(bundle, CFSTR("WebPluginName")));
76 if (pluginName && CFGetTypeID(pluginName) == CFStringGetTypeID())
77 plugin.info.name = pluginName;
78
79 // Get the plug-in description.
80 CFStringRef pluginDescription = static_cast<CFStringRef>(CFBundleGetValueForInfoDictionaryKey(bundle, CFSTR("WebPluginDescription")));
81 if (pluginDescription && CFGetTypeID(pluginDescription) == CFStringGetTypeID())
82 plugin.info.desc = pluginDescription;
83
84 // Get the MIME type mapping dictionary.
85 CFIndex numMimeTypes = CFDictionaryGetCount(mimeTypes.get());
86 Vector<CFStringRef> mimeTypesVector(numMimeTypes);
87 Vector<CFDictionaryRef> mimeTypeInfoVector(numMimeTypes);
88 CFDictionaryGetKeysAndValues(mimeTypes.get(), reinterpret_cast<const void**>(mimeTypesVector.data()), reinterpret_cast<const void**>(mimeTypeInfoVector.data()));
89
90 for (CFIndex i = 0; i < numMimeTypes; ++i) {
91 WebCore::MimeClassInfo mimeClassInfo;
92
93 // If this MIME type is invalid, ignore it.
94 CFStringRef mimeType = mimeTypesVector[i];
95 if (!mimeType || CFGetTypeID(mimeType) != CFStringGetTypeID() || CFStringGetLength(mimeType) == 0)
96 continue;
97
98 // If this MIME type doesn't have a valid info dictionary, ignore it.
99 CFDictionaryRef mimeTypeInfo = mimeTypeInfoVector[i];
100 if (!mimeTypeInfo || CFGetTypeID(mimeTypeInfo) != CFDictionaryGetTypeID())
101 continue;
102
103 // FIXME: Consider storing disabled MIME types.
104 CFTypeRef isEnabled = CFDictionaryGetValue(mimeTypeInfo, CFSTR("WebPluginTypeEnabled"));
105 if (isEnabled) {
106 if (CFGetTypeID(isEnabled) == CFNumberGetTypeID()) {
107 int value;
108 if (!CFNumberGetValue(static_cast<CFNumberRef>(isEnabled), kCFNumberIntType, &value) || !value)
109 continue;
110 } else if (CFGetTypeID(isEnabled) == CFBooleanGetTypeID()) {
111 if (!CFBooleanGetValue(static_cast<CFBooleanRef>(isEnabled)))
112 continue;
113 } else
114 continue;
115 }
116
117 // Get the MIME type description.
118 CFStringRef mimeTypeDescription = static_cast<CFStringRef>(CFDictionaryGetValue(mimeTypeInfo, CFSTR("WebPluginTypeDescription")));
119 if (mimeTypeDescription && CFGetTypeID(mimeTypeDescription) != CFStringGetTypeID())
120 mimeTypeDescription = 0;
121
122 mimeClassInfo.type = String(mimeType).convertToASCIILowercase();
123 mimeClassInfo.desc = mimeTypeDescription;
124
125 // Now get the extensions for this MIME type.
126 CFIndex numExtensions = 0;
127 CFArrayRef extensionsArray = static_cast<CFArrayRef>(CFDictionaryGetValue(mimeTypeInfo, CFSTR("WebPluginExtensions")));
128 if (extensionsArray && CFGetTypeID(extensionsArray) == CFArrayGetTypeID())
129 numExtensions = CFArrayGetCount(extensionsArray);
130
131 for (CFIndex i = 0; i < numExtensions; ++i) {
132 CFStringRef extension = static_cast<CFStringRef>(CFArrayGetValueAtIndex(extensionsArray, i));
133 if (!extension || CFGetTypeID(extension) != CFStringGetTypeID())
134 continue;
135
136 // The DivX plug-in lists multiple extensions in a comma separated string instead of using
137 // multiple array elements in the property list. Work around this here by splitting the
138 // extension string into components.
139 Vector<String> extensionComponents = String(extension).convertToASCIILowercase().split(',');
140
141 for (auto& component : extensionComponents)
142 mimeClassInfo.extensions.append(component);
143 }
144
145 // Add this MIME type.
146 plugin.info.mimes.append(mimeClassInfo);
147 }
148
149 return true;
150}

152bool NetscapePluginModule::getPluginInfo(const String& pluginPath, PluginModuleInfo& plugin)
153{
154 RetainPtr<CFURLRef> bundleURL = adoptCF(CFURLCreateWithFileSystemPath(kCFAllocatorDefault, pluginPath.createCFString().get(), kCFURLPOSIXPathStyle, false));
155
156 // Try to initialize the bundle.
157 RetainPtr<CFBundleRef> bundle = adoptCF(_CFBundleCreateUnique(kCFAllocatorDefault, bundleURL.get()));
158 if (!bundle)
159 return false;
160
161 // Check if this bundle is an NPAPI plug-in.
162 UInt32 packageType = 0;
163 CFBundleGetPackageInfo(bundle.get(), &packageType, 0);
164 if (packageType != FOUR_CHAR_CODE('BRPL'))
165 return false;
166
167 // Check that the architecture is valid.
168 if (!getPluginArchitecture(bundle.get(), plugin))
169 return false;
170
171 plugin.path = pluginPath;
172 plugin.bundleIdentifier = CFBundleGetIdentifier(bundle.get());
173 if (CFTypeRef versionTypeRef = CFBundleGetValueForInfoDictionaryKey(bundle.get(), kCFBundleVersionKey)) {
174 if (CFGetTypeID(versionTypeRef) == CFStringGetTypeID())
175 plugin.versionString = static_cast<CFStringRef>(versionTypeRef);
176 }
177
178 if (CFTypeRef shortVersionTypeRef = CFBundleGetValueForInfoDictionaryKey(bundle.get(), CFSTR("CFBundleShortVersionString"))) {
179 if (CFGetTypeID(shortVersionTypeRef) == CFStringGetTypeID())
180 plugin.shortVersionString = static_cast<CFStringRef>(shortVersionTypeRef);
181 }
182
183 if (!getPluginInfoFromPropertyLists(bundle.get(), plugin))
184 return false;
185
186 plugin.hasSandboxProfile = pluginHasSandboxProfile(plugin.bundleIdentifier);
187
188 RetainPtr<CFStringRef> filename = adoptCF(CFURLCopyLastPathComponent(bundleURL.get()));
189 plugin.info.file = filename.get();
190
191 if (plugin.info.name.isNull())
192 plugin.info.name = plugin.info.file;
193 if (plugin.info.desc.isNull())
194 plugin.info.desc = plugin.info.file;
195
196 plugin.info.isApplicationPlugin = false;
197 plugin.info.clientLoadPolicy = WebCore::PluginLoadClientPolicy::Undefined;
198#if PLATFORM(MAC)
199 plugin.info.bundleIdentifier = plugin.bundleIdentifier;
200 plugin.info.versionString = plugin.versionString;
201#endif
202
203 return true;
204}

206// FIXME: This doesn't need to be platform-specific.
207class PluginVersion {
208public:
209 static PluginVersion parse(const String& versionString);
210
211 bool isLessThan(unsigned componentA) const;
212 bool isValid() const { return !m_versionComponents.isEmpty(); }
213
214private:
215 PluginVersion()
216 {
217 }
218
219 Vector<unsigned, 4> m_versionComponents;
220};
221
222PluginVersion PluginVersion::parse(const String& versionString)
223{
224 PluginVersion version;
225 for (auto component : StringView { versionString }.split('.')) {
226 auto versionComponent = parseIntegerAllowingTrailingJunk<unsigned>(component);
227 if (!versionComponent)
228 return { };
229 version.m_versionComponents.append(*versionComponent);
230 }
231 return version;
232}
233
234bool PluginVersion::isLessThan(unsigned componentA) const
235{
236 ASSERT(isValid());
237
238 return m_versionComponents[0] < componentA;
239}

241void NetscapePluginModule::determineQuirks()
242{
243 PluginModuleInfo plugin;
244 if (!getPluginInfo(m_pluginPath, plugin))
245 return;
246
247 if (plugin.bundleIdentifier == "com.macromedia.Flash Player.plugin") {
248 // Flash requires that the return value of getprogname() be "WebKitPluginHost".
249 m_pluginQuirks.add(PluginQuirks::PrognameShouldBeWebKitPluginHost);
250
251 // Flash supports snapshotting.
252 m_pluginQuirks.add(PluginQuirks::SupportsSnapshotting);
253
254 // Flash returns a retained Core Animation layer.
255 m_pluginQuirks.add(PluginQuirks::ReturnsRetainedCoreAnimationLayer);
256
257 // Flash has a bug where NSExceptions can be released too early.
258 m_pluginQuirks.add(PluginQuirks::LeakAllThrownNSExceptions);
259 }
260
261 if (plugin.bundleIdentifier == "com.microsoft.SilverlightPlugin") {
262 // Silverlight doesn't explicitly opt into transparency, so we'll do it whenever
263 // there's a 'background' attribute that's set to a transparent color.
264 m_pluginQuirks.add(PluginQuirks::MakeOpaqueUnlessTransparentSilverlightBackgroundAttributeExists);
265
266 // Silverlight has a workaround for a leak in Safari 2. This workaround is
267 // applied when the user agent does not contain "Version/3" so we append it
268 // at the end of the user agent.
269 m_pluginQuirks.add(PluginQuirks::AppendVersion3UserAgent);
270
271 PluginVersion pluginVersion = PluginVersion::parse(plugin.versionString);
272 if (pluginVersion.isValid()) {
273 if (pluginVersion.isLessThan(4)) {
274 // Versions of Silverlight prior to 4 don't retain the scriptable NPObject.
275 m_pluginQuirks.add(PluginQuirks::ReturnsNonRetainedScriptableNPObject);
276 }
277 }
278 }
279
280 if (plugin.bundleIdentifier == "com.apple.ist.ds.appleconnect.webplugin") {
281 // <rdar://problem/8440903>: AppleConnect has a bug where it does not
282 // understand the parameter names specified in the <object> element that
283 // embeds its plug-in.
284 m_pluginQuirks.add(PluginQuirks::WantsLowercaseParameterNames);
285 }
286
287 if (plugin.bundleIdentifier == "com.adobe.acrobat.pdfviewerNPAPI" || plugin.bundleIdentifier == "com.apple.testnetscapeplugin") {
288 // The Adobe Reader plug-in wants wheel events.
289 m_pluginQuirks.add(PluginQuirks::WantsWheelEvents);
290 }
291}
