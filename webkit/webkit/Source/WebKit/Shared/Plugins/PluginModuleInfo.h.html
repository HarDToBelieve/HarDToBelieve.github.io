34namespace WebKit {
35
36enum PluginModuleLoadPolicy {
37 // The plug-in module should be loaded normally.
38 PluginModuleLoadNormally,
39
40 // The plug-in module should be loaded normally. If the plug-in has a sandbox, it will be disabled.
41 PluginModuleLoadUnsandboxed,
42
43 // The plug-in should be blocked from being instantiated.
44 // Note that the plug-in will still be seen by e.g. navigator.plugins
45 PluginModuleBlockedForSecurity,
46 PluginModuleBlockedForCompatibility,
47};
48
49struct PluginModuleInfo {
50 String path;
51 WebCore::PluginInfo info;
52
53#if PLATFORM(COCOA)
54 cpu_type_t pluginArchitecture;
55 String bundleIdentifier;
56 String versionString;
57 String shortVersionString;
58 bool hasSandboxProfile;
59#endif
60};

62} // namespace WebKit
