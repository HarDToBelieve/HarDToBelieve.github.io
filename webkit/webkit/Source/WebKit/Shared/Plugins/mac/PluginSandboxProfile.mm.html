34namespace WebKit {
35
36static NSString *pluginSandboxProfileDirectory()
37{
38 NSBundle *bundle = [NSBundle bundleForClass:NSClassFromString(@"WKWebView")];
39 return [[bundle resourcePath] stringByAppendingPathComponent:@"PlugInSandboxProfiles"];
40}
41
42static NSString *pluginSandboxProfileName(const String& bundleIdentifier)
43{
44 // Fold all / characters to : to prevent the plugin bundle-id from trying to escape the profile directory
45 String sanitizedBundleIdentifier = bundleIdentifier;
46 sanitizedBundleIdentifier.replace('/', ':');
47
48 return [NSString stringWithFormat:@"%@.sb", (NSString *)sanitizedBundleIdentifier];
49}
50
51static String pluginSandboxCommonProfile()
52{
53 NSString *profilePath = [pluginSandboxProfileDirectory() stringByAppendingPathComponent:@"com.apple.WebKit.plugin-common.sb"];
54 return [NSString stringWithContentsOfFile:profilePath encoding:NSUTF8StringEncoding error:NULL];
55}
56
57String pluginSandboxProfile(const String& bundleIdentifier)
58{
59 if (bundleIdentifier.isEmpty())
60 return String();
61
62 NSString *profileName = pluginSandboxProfileName(bundleIdentifier);
63
64 NSString *profilePath = [pluginSandboxProfileDirectory() stringByAppendingPathComponent:profileName];
65 NSString *profileString = [NSString stringWithContentsOfFile:profilePath encoding:NSUTF8StringEncoding error:NULL];
66 if (!profileString)
67 return String();
68
69 return makeString(pluginSandboxCommonProfile(), String(profileString));
70}
71
72bool pluginHasSandboxProfile(const String& bundleIdentifier)
73{
74 if (bundleIdentifier.isEmpty())
75 return false;
76
77 NSString *profileName = pluginSandboxProfileName(bundleIdentifier);
78 NSString *profilePath = [pluginSandboxProfileDirectory() stringByAppendingPathComponent:profileName];
79
80 return [[NSFileManager defaultManager] fileExistsAtPath:profilePath];
81}
