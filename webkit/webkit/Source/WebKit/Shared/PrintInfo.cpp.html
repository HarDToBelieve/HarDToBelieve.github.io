36
37void PrintInfo::encode(IPC::Encoder& encoder) const
38{
39 encoder << pageSetupScaleFactor;
40 encoder << availablePaperWidth;
41 encoder << availablePaperHeight;
42 encoder << margin;
43
44#if PLATFORM(GTK)
45 IPC::encode(encoder, printSettings.get());
46 IPC::encode(encoder, pageSetup.get());
47 encoder << printMode;
48#endif
49
50#if PLATFORM(IOS_FAMILY)
51 encoder << snapshotFirstPage;
52#endif
53}
54
55bool PrintInfo::decode(IPC::Decoder& decoder, PrintInfo& info)
56{
57 if (!decoder.decode(info.pageSetupScaleFactor))
58 return false;
59 if (!decoder.decode(info.availablePaperWidth))
60 return false;
61 if (!decoder.decode(info.availablePaperHeight))
62 return false;
63 if (!decoder.decode(info.margin))
64 return false;
65
66#if PLATFORM(GTK)
67 if (!IPC::decode(decoder, info.printSettings))
68 return false;
69 if (!IPC::decode(decoder, info.pageSetup))
70 return false;
71 if (!decoder.decode(info.printMode))
72 return false;
73#endif
74
75#if PLATFORM(IOS_FAMILY)
76 if (!decoder.decode(info.snapshotFirstPage))
77 return false;
78#endif
79
80 return true;
81}
