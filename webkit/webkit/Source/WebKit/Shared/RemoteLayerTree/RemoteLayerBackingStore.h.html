44namespace WebKit {
45
46class PlatformCALayerRemote;
47
48class RemoteLayerBackingStore {
49 WTF_MAKE_NONCOPYABLE(RemoteLayerBackingStore);
50 WTF_MAKE_FAST_ALLOCATED;
51public:
52 RemoteLayerBackingStore(PlatformCALayerRemote*);
53 ~RemoteLayerBackingStore();
54
55 enum class Type : uint8_t {
56 IOSurface,
57 Bitmap,
58#if ENABLE(CG_DISPLAY_LIST_BACKED_IMAGE_BUFFER)
59 CGDisplayList,
60#endif
61 };
62
63 void ensureBackingStore(Type, WebCore::FloatSize, float scale, bool deepColor, bool isOpaque);
64
65 void setNeedsDisplay(const WebCore::IntRect);
66 void setNeedsDisplay();
67
68 bool display();
69
70 WebCore::FloatSize size() const { return m_size; }
71 float scale() const { return m_scale; }
72 Type type() const { return m_type; }
73 bool isOpaque() const { return m_isOpaque; }
74 unsigned bytesPerPixel() const;
75
76 PlatformCALayerRemote* layer() const { return m_layer; }
77
78 enum class LayerContentsType { IOSurface, CAMachPort };
79 void applyBackingStoreToLayer(CALayer *, LayerContentsType);
80
81 void encode(IPC::Encoder&) const;
82 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, RemoteLayerBackingStore&);
83
84 void enumerateRectsBeingDrawn(WebCore::GraphicsContext&, void (^)(WebCore::FloatRect));
85
86 bool hasFrontBuffer() const
87 {
88 return !!m_frontBuffer.imageBuffer;
89 }
90
91 std::unique_ptr<WebCore::ThreadSafeImageBufferFlusher> takePendingFlusher();
92
93 enum class BufferType {
94 Front,
95 Back,
96 SecondaryBack
97 };
98
99 // Returns true if it was able to fulfill the request. This can fail when trying to mark an in-use surface as volatile.
100 bool setBufferVolatility(BufferType, bool isVolatile);
101
102 MonotonicTime lastDisplayTime() const { return m_lastDisplayTime; }
103
104private:
105 void drawInContext(WebCore::GraphicsContext&);
106 void clearBackingStore();
107 void swapToValidFrontBuffer();
108
109 bool supportsPartialRepaint();
110
111 WebCore::PixelFormat pixelFormat() const;
112
113 PlatformCALayerRemote* m_layer;
114
115 WebCore::FloatSize m_size;
116 float m_scale;
117 bool m_isOpaque;
118
119 WebCore::Region m_dirtyRegion;
120
121 struct Buffer {
122 RefPtr<WebCore::ImageBuffer> imageBuffer;
123 bool isVolatile = false;
124
125 explicit operator bool() const
126 {
127 return !!imageBuffer;
128 }
129
130 void discard();
131 };
132
133 Buffer m_frontBuffer;
134 Buffer m_backBuffer;
135 Buffer m_secondaryBackBuffer;
136 std::optional<ImageBufferBackendHandle> m_bufferHandle;
137
138 std::unique_ptr<WebCore::ThreadSafeImageBufferFlusher> m_frontBufferFlusher;
139
140 Type m_type;
141 bool m_deepColor { false };
142
143 WebCore::RepaintRectList m_paintingRects;
144
145 MonotonicTime m_lastDisplayTime;
146};
