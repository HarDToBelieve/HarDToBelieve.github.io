54void RemoteLayerTreeTransaction::LayerCreationProperties::encode(IPC::Encoder& encoder) const
55{
56 encoder << layerID;
57 encoder << type;
58
59 // PlatformCALayerRemoteCustom
60 encoder << hostingContextID;
61 encoder << hostingDeviceScaleFactor;
62
63#if ENABLE(MODEL_ELEMENT)
64 // PlatformCALayerRemoteModelHosting
65 encoder << !!model;
66 if (model)
67 encoder << *model;
68#endif
69}

71auto RemoteLayerTreeTransaction::LayerCreationProperties::decode(IPC::Decoder& decoder) -> std::optional<LayerCreationProperties>
72{
73 LayerCreationProperties result;
74 if (!decoder.decode(result.layerID))
75 return std::nullopt;
76
77 if (!decoder.decode(result.type))
78 return std::nullopt;
79
80 // PlatformCALayerRemoteCustom
81 if (!decoder.decode(result.hostingContextID))
82 return std::nullopt;
83
84 if (!decoder.decode(result.hostingDeviceScaleFactor))
85 return std::nullopt;
86
87#if ENABLE(MODEL_ELEMENT)
88 // PlatformCALayerRemoteModelHosting
89 bool hasModel;
90 if (!decoder.decode(hasModel))
91 return std::nullopt;
92 if (hasModel) {
93 auto model = WebCore::Model::decode(decoder);
94 if (!model)
95 return std::nullopt;
96 result.model = model;
97 }
98#endif
99
100 return WTFMove(result);
101}

356 if (!decoder.decode(result.addedAnimations))
357 return false;
358
359 if (!decoder.decode(result.keyPathsOfAnimationsToRemove))
360 return false;
361 }
362
363 if (result.changedProperties & PositionChanged) {
364 if (!decoder.decode(result.position))
365 return false;
366 }
367
368 if (result.changedProperties & BoundsChanged) {
369 if (!decoder.decode(result.bounds))
370 return false;
371 }
372
373 if (result.changedProperties & BackgroundColorChanged) {
374 if (!decoder.decode(result.backgroundColor))
375 return false;
376 }

378 if (result.changedProperties & AnchorPointChanged) {
379 if (!decoder.decode(result.anchorPoint))

558 decoder >> eventRegion;
559 if (!eventRegion)
560 return false;

562 }
563
564#if HAVE(CORE_ANIMATION_SEPARATED_LAYERS)
565 if (result.changedProperties & SeparatedChanged) {
566 if (!decoder.decode(result.isSeparated))

568 }
569

571 if (result.changedProperties & SeparatedPortalChanged) {
572 if (!decoder.decode(result.isSeparatedPortal))
573 return false;
574 }
575
576 if (result.changedProperties & DescendentOfSeparatedPortalChanged) {
577 if (!decoder.decode(result.isDescendentOfSeparatedPortal))
578 return false;
579 }
580#endif
581#endif
582
583 return true;
584}

611
612 encoder << m_scrollPosition;
613
614 encoder << m_themeColor;

616 encoder << m_sampledPageTopColor;
617
618 encoder << m_pageScaleFactor;
619 encoder << m_minimumScaleFactor;
620 encoder << m_maximumScaleFactor;
621 encoder << m_initialScaleFactor;
622 encoder << m_viewportMetaTagWidth;
623
624 encoder << m_renderTreeSize;
625 encoder << m_transactionID;
626 encoder << m_activityStateChangeID;
627
628 encoder << m_newlyReachedPaintingMilestones;

649bool RemoteLayerTreeTransaction::decode(IPC::Decoder& decoder, RemoteLayerTreeTransaction& result)
650{
651 if (!decoder.decode(result.m_rootLayerID))
652 return false;
653 if (!result.m_rootLayerID)
654 return false;
655
656 if (!decoder.decode(result.m_createdLayers))
657 return false;
658
659 uint64_t numChangedLayerProperties;
660 if (!decoder.decode(numChangedLayerProperties))
661 return false;
662
663 for (uint64_t i = 0; i < numChangedLayerProperties; ++i) {
664 WebCore::GraphicsLayer::PlatformLayerID layerID;
665 if (!decoder.decode(layerID))
666 return false;
667
668 std::unique_ptr<LayerProperties> layerProperties = makeUnique<LayerProperties>();
669 if (!decoder.decode(*layerProperties))
670 return false;
671
672 result.changedLayerProperties().set(layerID, WTFMove(layerProperties));
673 }
674
675 if (!decoder.decode(result.m_destroyedLayerIDs))
676 return false;
677
678 for (auto& layerID : result.m_destroyedLayerIDs) {
679 if (!layerID)
680 return false;
681 }
682
683 if (!decoder.decode(result.m_videoLayerIDsPendingFullscreen))
684 return false;
685
686 if (!decoder.decode(result.m_layerIDsWithNewlyUnreachableBackingStore))
687 return false;
688
689 for (auto& layerID : result.m_layerIDsWithNewlyUnreachableBackingStore) {
690 if (!layerID)
691 return false;
692 }
693
694 if (!decoder.decode(result.m_contentsSize))
695 return false;
696
697 if (!decoder.decode(result.m_scrollOrigin))
698 return false;
699
700 if (!decoder.decode(result.m_baseLayoutViewportSize))
701 return false;
702
703 if (!decoder.decode(result.m_minStableLayoutViewportOrigin))
704 return false;
705
706 if (!decoder.decode(result.m_maxStableLayoutViewportOrigin))
707 return false;
708
709 if (!decoder.decode(result.m_scrollPosition))
710 return false;
711
712 if (!decoder.decode(result.m_themeColor))
713 return false;
714
715 if (!decoder.decode(result.m_pageExtendedBackgroundColor))
716 return false;
717
718 if (!decoder.decode(result.m_sampledPageTopColor))
719 return false;
720
721 if (!decoder.decode(result.m_pageScaleFactor))
722 return false;
723
724 if (!decoder.decode(result.m_minimumScaleFactor))
725 return false;
726
727 if (!decoder.decode(result.m_maximumScaleFactor))
728 return false;
729
730 if (!decoder.decode(result.m_initialScaleFactor))
731 return false;
732
733 if (!decoder.decode(result.m_viewportMetaTagWidth))
734 return false;
735
736 if (!decoder.decode(result.m_renderTreeSize))
737 return false;
738
739 if (!decoder.decode(result.m_transactionID))
740 return false;
741
742 if (!decoder.decode(result.m_activityStateChangeID))
743 return false;
744
745 if (!decoder.decode(result.m_newlyReachedPaintingMilestones))
746 return false;
747
748 if (!decoder.decode(result.m_scaleWasSetByUIProcess))
749 return false;
750
751 if (!decoder.decode(result.m_allowsUserScaling))
752 return false;
753
754 if (!decoder.decode(result.m_avoidsUnsafeArea))
755 return false;
756
757 if (!decoder.decode(result.m_viewportMetaTagWidthWasExplicit))
758 return false;
759
760 if (!decoder.decode(result.m_viewportMetaTagCameFromImageDocument))
761 return false;
762
763 if (!decoder.decode(result.m_isInStableState))
764 return false;
765
766 std::optional<Vector<TransactionCallbackID>> callbackIDs;
767 decoder >> callbackIDs;
768 if (!callbackIDs)
769 return false;
770 result.m_callbackIDs = WTFMove(*callbackIDs);
771
772 bool hasEditorState;
773 if (!decoder.decode(hasEditorState))
774 return false;
775
776 if (hasEditorState) {
777 EditorState editorState;
778 if (!decoder.decode(editorState))
779 return false;
780 result.setEditorState(editorState);
781 }
782
783 if (!decoder.decode(result.m_dynamicViewportSizeUpdateID))
784 return false;
785
786 return true;
787}

995String RemoteLayerTreeTransaction::description() const
996{
997 TextStream ts;
998
999 ts.startGroup();
1000 ts << "layer tree";
1001
1002 ts.dumpProperty("transactionID", m_transactionID);
1003 ts.dumpProperty("contentsSize", m_contentsSize);
1004 if (m_scrollOrigin != WebCore::IntPoint::zero())
1005 ts.dumpProperty("scrollOrigin", m_scrollOrigin);
1006
1007 ts.dumpProperty("baseLayoutViewportSize", WebCore::FloatSize(m_baseLayoutViewportSize));
1008
1009 if (m_minStableLayoutViewportOrigin != WebCore::LayoutPoint::zero())
1010 ts.dumpProperty("minStableLayoutViewportOrigin", WebCore::FloatPoint(m_minStableLayoutViewportOrigin));
1011 ts.dumpProperty("maxStableLayoutViewportOrigin", WebCore::FloatPoint(m_maxStableLayoutViewportOrigin));
1012
1013 if (m_pageScaleFactor != 1)
1014 ts.dumpProperty("pageScaleFactor", m_pageScaleFactor);
1015
1016 ts.dumpProperty("minimumScaleFactor", m_minimumScaleFactor);
1017 ts.dumpProperty("maximumScaleFactor", m_maximumScaleFactor);
1018 ts.dumpProperty("initialScaleFactor", m_initialScaleFactor);
1019 ts.dumpProperty("viewportMetaTagWidth", m_viewportMetaTagWidth);
1020 ts.dumpProperty("viewportMetaTagWidthWasExplicit", m_viewportMetaTagWidthWasExplicit);
1021 ts.dumpProperty("viewportMetaTagCameFromImageDocument", m_viewportMetaTagCameFromImageDocument);
1022 ts.dumpProperty("allowsUserScaling", m_allowsUserScaling);
1023 ts.dumpProperty("avoidsUnsafeArea", m_avoidsUnsafeArea);
1024 ts.dumpProperty("isInStableState", m_isInStableState);
1025 ts.dumpProperty("renderTreeSize", m_renderTreeSize);
1026 ts.dumpProperty("root-layer", m_rootLayerID);
1027
1028 if (!m_createdLayers.isEmpty()) {
1029 TextStream::GroupScope group(ts);
1030 ts << "created-layers";
1031 for (const auto& createdLayer : m_createdLayers) {
1032 TextStream::GroupScope group(ts);
1033 ts << createdLayer.type <<" " << createdLayer.layerID;
1034 switch (createdLayer.type) {
1035 case WebCore::PlatformCALayer::LayerTypeAVPlayerLayer:
1036 ts << " (context-id " << createdLayer.hostingContextID << ")";
1037 break;
1038 case WebCore::PlatformCALayer::LayerTypeContentsProvidedLayer:
1039 ts << " (context-id " << createdLayer.hostingContextID << ")";
1040 break;
1041 case WebCore::PlatformCALayer::LayerTypeCustom:
1042 ts << " (context-id " << createdLayer.hostingContextID << ")";
1043 break;
1044#if ENABLE(MODEL_ELEMENT)
1045 case WebCore::PlatformCALayer::LayerTypeModelLayer:
1046 ts << " (model " << *createdLayer.model << ")";
1047 break;
1048#endif
1049 default:
1050 break;
1051 }
1052 }
1053 }
1054
1055 dumpChangedLayers(ts, m_changedLayerProperties);
1056
1057 if (!m_destroyedLayerIDs.isEmpty())
1058 ts.dumpProperty<Vector<WebCore::GraphicsLayer::PlatformLayerID>>("destroyed-layers", m_destroyedLayerIDs);
1059
1060 if (m_editorState) {
1061 TextStream::GroupScope scope(ts);
1062 ts << "EditorState";
1063 ts << *m_editorState;
1064 }
1065
1066 ts.endGroup();
1067
1068 return ts.release();
1069}
