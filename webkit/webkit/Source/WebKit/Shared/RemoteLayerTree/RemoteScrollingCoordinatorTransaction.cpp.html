169void ArgumentCoder<ScrollingStateNode>::encode(Encoder& encoder, const ScrollingStateNode& node)
170{
171 encoder << node.nodeType();
172 encoder << node.scrollingNodeID();
173 encoder << node.parentNodeID();
174 encoder << node.changedProperties();
175
176 if (node.hasChangedProperty(ScrollingStateNode::Property::Layer))
177 encoder << static_cast<GraphicsLayer::PlatformLayerID>(node.layer());
178}
179
180bool ArgumentCoder<ScrollingStateNode>::decode(Decoder& decoder, ScrollingStateNode& node)
181{
182 // nodeType, scrollingNodeID and parentNodeID have already been decoded by the caller in order to create the node.
183 OptionSet<ScrollingStateNode::Property> changedProperties;
184 if (!decoder.decode(changedProperties))
185 return false;
186
187 node.setChangedProperties(changedProperties);
188 if (node.hasChangedProperty(ScrollingStateNode::Property::Layer)) {
189 GraphicsLayer::PlatformLayerID layerID;
190 if (!decoder.decode(layerID))
191 return false;
192 node.setLayer(layerID);
193 }
194
195 return true;
196}

568static void encodeNodeAndDescendants(IPC::Encoder& encoder, const ScrollingStateNode& stateNode, int& encodedNodeCount)
569{
570 ++encodedNodeCount;
571
572 switch (stateNode.nodeType()) {
573 case ScrollingNodeType::MainFrame:
574 case ScrollingNodeType::Subframe:
575 encoder << downcast<ScrollingStateFrameScrollingNode>(stateNode);
576 break;
577 case ScrollingNodeType::FrameHosting:
578 encoder << downcast<ScrollingStateFrameHostingNode>(stateNode);
579 break;
580 case ScrollingNodeType::Overflow:
581 encoder << downcast<ScrollingStateOverflowScrollingNode>(stateNode);
582 break;
583 case ScrollingNodeType::OverflowProxy:
584 encoder << downcast<ScrollingStateOverflowScrollProxyNode>(stateNode);
585 break;
586 case ScrollingNodeType::Fixed:
587 encoder << downcast<ScrollingStateFixedNode>(stateNode);
588 break;
589 case ScrollingNodeType::Sticky:
590 encoder << downcast<ScrollingStateStickyNode>(stateNode);
591 break;
592 case ScrollingNodeType::Positioned:
593 encoder << downcast<ScrollingStatePositionedNode>(stateNode);
594 break;
595 }
596
597 if (!stateNode.children())
598 return;
599
600 for (const auto& child : *stateNode.children())
601 encodeNodeAndDescendants(encoder, *child.get(), encodedNodeCount);
602}

604void RemoteScrollingCoordinatorTransaction::encode(IPC::Encoder& encoder) const
605{
606 int numNodes = m_scrollingStateTree ? m_scrollingStateTree->nodeCount() : 0;
607 encoder << numNodes;
608
609 bool hasNewRootNode = m_scrollingStateTree ? m_scrollingStateTree->hasNewRootStateNode() : false;
610 encoder << hasNewRootNode;
611
612 if (m_scrollingStateTree) {
613 encoder << m_scrollingStateTree->hasChangedProperties();
614
615 int numNodesEncoded = 0;
616 if (const ScrollingStateNode* rootNode = m_scrollingStateTree->rootStateNode())
617 encodeNodeAndDescendants(encoder, *rootNode, numNodesEncoded);
618
619 ASSERT_UNUSED(numNodesEncoded, numNodesEncoded == numNodes);
620 } else
621 encoder << Vector<ScrollingNodeID>();
622}

629bool RemoteScrollingCoordinatorTransaction::decode(IPC::Decoder& decoder)
630{
631 int numNodes;
632 if (!decoder.decode(numNodes))
633 return false;
634
635 bool hasNewRootNode;
636 if (!decoder.decode(hasNewRootNode))
637 return false;
638
639 m_scrollingStateTree = makeUnique<ScrollingStateTree>();
640
641 bool hasChangedProperties;
642 if (!decoder.decode(hasChangedProperties))
643 return false;
644
645 m_scrollingStateTree->setHasChangedProperties(hasChangedProperties);
646
647 for (int i = 0; i < numNodes; ++i) {
648 ScrollingNodeType nodeType;
649 if (!decoder.decode(nodeType))
650 return false;
651
652 ScrollingNodeID nodeID;
653 if (!decoder.decode(nodeID))
654 return false;
655
656 ScrollingNodeID parentNodeID;
657 if (!decoder.decode(parentNodeID))
658 return false;
659
660 m_scrollingStateTree->insertNode(nodeType, nodeID, parentNodeID, notFound);
661 ScrollingStateNode* newNode = m_scrollingStateTree->stateNodeForID(nodeID);
662 ASSERT(newNode);
663 ASSERT(!parentNodeID || newNode->parent());
664
665 switch (nodeType) {
666 case ScrollingNodeType::MainFrame:
667 case ScrollingNodeType::Subframe:
668 if (!decoder.decode(downcast<ScrollingStateFrameScrollingNode>(*newNode)))
669 return false;
670 break;
671 case ScrollingNodeType::FrameHosting:
672 if (!decoder.decode(downcast<ScrollingStateFrameHostingNode>(*newNode)))
673 return false;
674 break;
675 case ScrollingNodeType::Overflow:
676 if (!decoder.decode(downcast<ScrollingStateOverflowScrollingNode>(*newNode)))
677 return false;
678 break;
679 case ScrollingNodeType::OverflowProxy:
680 if (!decoder.decode(downcast<ScrollingStateOverflowScrollProxyNode>(*newNode)))
681 return false;
682 break;
683 case ScrollingNodeType::Fixed:
684 if (!decoder.decode(downcast<ScrollingStateFixedNode>(*newNode)))
685 return false;
686 break;
687 case ScrollingNodeType::Sticky:
688 if (!decoder.decode(downcast<ScrollingStateStickyNode>(*newNode)))
689 return false;
690 break;
691 case ScrollingNodeType::Positioned:
692 if (!decoder.decode(downcast<ScrollingStatePositionedNode>(*newNode)))
693 return false;
694 break;
695 }
696 }
697
698 m_scrollingStateTree->setHasNewRootStateNode(hasNewRootNode);
699
700 return true;
701}
