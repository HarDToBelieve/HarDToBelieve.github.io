34void RemoteScrollingUIState::encode(IPC::Encoder& encoder) const
35{
36 encoder << m_changes;
37
38 if (m_changes.contains(Changes::ScrollSnapNodes))
39 encoder << m_nodesWithActiveScrollSnap;
40
41 if (m_changes.contains(Changes::UserScrollNodes))
42 encoder << m_nodesWithActiveUserScrolls;
43}
44
45std::optional<RemoteScrollingUIState> RemoteScrollingUIState::decode(IPC::Decoder& decoder)
46{
47 RemoteScrollingUIState uiState;
48
49 if (!decoder.decode(uiState.m_changes))
50 return std::nullopt;
51
52 if (uiState.m_changes.contains(Changes::ScrollSnapNodes)) {
53 if (!decoder.decode(uiState.m_nodesWithActiveScrollSnap))
54 return std::nullopt;
55 }
56
57 if (uiState.m_changes.contains(Changes::UserScrollNodes)) {
58 if (!decoder.decode(uiState.m_nodesWithActiveUserScrolls))
59 return std::nullopt;
60 }
61
62 return uiState;
63}
64
65void RemoteScrollingUIState::reset()
66{
67 clearChanges();
68 m_nodesWithActiveScrollSnap.clear();
69 m_nodesWithActiveUserScrolls.clear();
70}
71
72void RemoteScrollingUIState::addNodeWithActiveScrollSnap(WebCore::ScrollingNodeID nodeID)
73{
74 auto addResult = m_nodesWithActiveScrollSnap.add(nodeID);
75 if (addResult.isNewEntry)
76 m_changes.add(Changes::ScrollSnapNodes);
77}
78
79void RemoteScrollingUIState::removeNodeWithActiveScrollSnap(WebCore::ScrollingNodeID nodeID)
80{
81 if (m_nodesWithActiveScrollSnap.remove(nodeID))
82 m_changes.add(Changes::ScrollSnapNodes);
83}
84
85void RemoteScrollingUIState::addNodeWithActiveUserScroll(WebCore::ScrollingNodeID nodeID)
86{
87 auto addResult = m_nodesWithActiveUserScrolls.add(nodeID);
88 if (addResult.isNewEntry)
89 m_changes.add(Changes::UserScrollNodes);
90}
91
92void RemoteScrollingUIState::removeNodeWithActiveUserScroll(WebCore::ScrollingNodeID nodeID)
93{
94 if (m_nodesWithActiveUserScrolls.remove(nodeID))
95 m_changes.add(Changes::UserScrollNodes);
96}
