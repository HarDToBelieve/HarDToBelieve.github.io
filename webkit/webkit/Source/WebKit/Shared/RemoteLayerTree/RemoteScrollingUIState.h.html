31namespace IPC {
32class Decoder;
33class Encoder;

36namespace WebCore {
37using ScrollingNodeID = uint64_t;
38}
39
40namespace WebKit {
41
42class RemoteScrollingUIState {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 enum class Changes : uint8_t {
46 ScrollSnapNodes = 1 << 0,
47 UserScrollNodes = 1 << 1,
48 };
49
50 void encode(IPC::Encoder&) const;
51 static std::optional<RemoteScrollingUIState> decode(IPC::Decoder&);
52
53 OptionSet<Changes> changes() const { return m_changes; }
54 void clearChanges() { m_changes = { }; }
55
56 void reset();
57
58 const HashSet<WebCore::ScrollingNodeID>& nodesWithActiveScrollSnap() const { return m_nodesWithActiveScrollSnap; }
59 void addNodeWithActiveScrollSnap(WebCore::ScrollingNodeID);
60 void removeNodeWithActiveScrollSnap(WebCore::ScrollingNodeID);
61
62 const HashSet<WebCore::ScrollingNodeID>& nodesWithActiveUserScrolls() const { return m_nodesWithActiveUserScrolls; }
63 void addNodeWithActiveUserScroll(WebCore::ScrollingNodeID);
64 void removeNodeWithActiveUserScroll(WebCore::ScrollingNodeID);
65
66private:
67 OptionSet<Changes> m_changes;
68 HashSet<WebCore::ScrollingNodeID> m_nodesWithActiveScrollSnap;
69 HashSet<WebCore::ScrollingNodeID> m_nodesWithActiveUserScrolls;
70};
71
72} // namespace WebKit

81 >;
82};
