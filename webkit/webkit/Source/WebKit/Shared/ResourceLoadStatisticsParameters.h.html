28#include "ArgumentCoders.h"

34namespace WebKit {
35
36struct ResourceLoadStatisticsParameters {
37
38 String directory;
39 SandboxExtension::Handle directoryExtensionHandle;
40 bool enabled { false };
41 bool isItpStateExplicitlySet { false };
42 bool enableLogTestingEvent { false };
43 bool shouldIncludeLocalhost { true };
44 bool enableDebugMode { false };
45#if ENABLE(RESOURCE_LOAD_STATISTICS)
46 WebCore::ThirdPartyCookieBlockingMode thirdPartyCookieBlockingMode { WebCore::ThirdPartyCookieBlockingMode::All };
47 WebCore::SameSiteStrictEnforcementEnabled sameSiteStrictEnforcementEnabled { WebCore::SameSiteStrictEnforcementEnabled::No };
48#endif
49 WebCore::FirstPartyWebsiteDataRemovalMode firstPartyWebsiteDataRemovalMode { WebCore::FirstPartyWebsiteDataRemovalMode::AllButCookies };
50 WebCore::RegistrableDomain standaloneApplicationDomain;
51 HashSet<WebCore::RegistrableDomain> appBoundDomains;
52 WebCore::RegistrableDomain manualPrevalentResource;
53
54 void encode(IPC::Encoder& encoder) const
55 {
56 encoder << directory;
57 encoder << directoryExtensionHandle;
58 encoder << enabled;
59 encoder << isItpStateExplicitlySet;
60 encoder << enableLogTestingEvent;
61 encoder << shouldIncludeLocalhost;
62 encoder << enableDebugMode;
63#if ENABLE(RESOURCE_LOAD_STATISTICS)
64 encoder << thirdPartyCookieBlockingMode;
65 encoder << sameSiteStrictEnforcementEnabled;
66#endif
67 encoder << firstPartyWebsiteDataRemovalMode;
68 encoder << standaloneApplicationDomain;
69 encoder << appBoundDomains;
70 encoder << manualPrevalentResource;
71 }
72
73 static std::optional<ResourceLoadStatisticsParameters> decode(IPC::Decoder& decoder)
74 {
75 std::optional<String> directory;
76 decoder >> directory;
77 if (!directory)
78 return std::nullopt;
79
80 std::optional<SandboxExtension::Handle> directoryExtensionHandle;
81 decoder >> directoryExtensionHandle;
82 if (!directoryExtensionHandle)
83 return std::nullopt;
84
85 std::optional<bool> enabled;
86 decoder >> enabled;
87 if (!enabled)
88 return std::nullopt;
89
90 std::optional<bool> isItpStateExplicitlySet;
91 decoder >> isItpStateExplicitlySet;
92 if (!isItpStateExplicitlySet)
93 return std::nullopt;
94
95 std::optional<bool> enableLogTestingEvent;
96 decoder >> enableLogTestingEvent;
97 if (!enableLogTestingEvent)
98 return std::nullopt;
99
100 std::optional<bool> shouldIncludeLocalhost;
101 decoder >> shouldIncludeLocalhost;
102 if (!shouldIncludeLocalhost)
103 return std::nullopt;
104
105 std::optional<bool> enableDebugMode;
106 decoder >> enableDebugMode;
107 if (!enableDebugMode)
108 return std::nullopt;
109
110#if ENABLE(RESOURCE_LOAD_STATISTICS)
111 std::optional<WebCore::ThirdPartyCookieBlockingMode> thirdPartyCookieBlockingMode;
112 decoder >> thirdPartyCookieBlockingMode;
113 if (!thirdPartyCookieBlockingMode)
114 return std::nullopt;
115
116 std::optional<WebCore::SameSiteStrictEnforcementEnabled> sameSiteStrictEnforcementEnabled;
117 decoder >> sameSiteStrictEnforcementEnabled;
118 if (!sameSiteStrictEnforcementEnabled)
119 return std::nullopt;
120#endif
121
122 std::optional<WebCore::FirstPartyWebsiteDataRemovalMode> firstPartyWebsiteDataRemovalMode;
123 decoder >> firstPartyWebsiteDataRemovalMode;
124 if (!firstPartyWebsiteDataRemovalMode)
125 return std::nullopt;
126
127 std::optional<WebCore::RegistrableDomain> standaloneApplicationDomain;
128 decoder >> standaloneApplicationDomain;
129 if (!standaloneApplicationDomain)
130 return std::nullopt;
131
132 std::optional<HashSet<WebCore::RegistrableDomain>> appBoundDomains;
133 decoder >> appBoundDomains;
134 if (!appBoundDomains)
135 return std::nullopt;
136
137 std::optional<WebCore::RegistrableDomain> manualPrevalentResource;
138 decoder >> manualPrevalentResource;
139 if (!manualPrevalentResource)
140 return std::nullopt;
141
142 return {{
143 WTFMove(*directory),
144 WTFMove(*directoryExtensionHandle),
145 WTFMove(*enabled),
146 WTFMove(*isItpStateExplicitlySet),
147 WTFMove(*enableLogTestingEvent),
148 WTFMove(*shouldIncludeLocalhost),
149 WTFMove(*enableDebugMode),
150#if ENABLE(RESOURCE_LOAD_STATISTICS)
151 WTFMove(*thirdPartyCookieBlockingMode),
152 WTFMove(*sameSiteStrictEnforcementEnabled),
153#endif
154 WTFMove(*firstPartyWebsiteDataRemovalMode),
155 WTFMove(*standaloneApplicationDomain),
156 WTFMove(*appBoundDomains),
157 WTFMove(*manualPrevalentResource),
158 }};
159 }
160};
