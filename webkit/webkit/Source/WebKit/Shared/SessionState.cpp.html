35void HTTPBody::Element::encode(IPC::Encoder& encoder) const
36{
37 encoder << type;
38 encoder << data;
39 encoder << filePath;
40 encoder << fileStart;
41 encoder << fileLength;
42 encoder << expectedFileModificationTime;
43 encoder << blobURLString;
44}
45
46static bool isValidEnum(HTTPBody::Element::Type type)
47{
48 switch (type) {
49 case HTTPBody::Element::Type::Data:
50 case HTTPBody::Element::Type::File:
51 case HTTPBody::Element::Type::Blob:
52 return true;
53 }
54
55 return false;
56}
57
58auto HTTPBody::Element::decode(IPC::Decoder& decoder) -> std::optional<Element>
59{
60 Element result;
61 if (!decoder.decode(result.type) || !isValidEnum(result.type))
62 return std::nullopt;
63 if (!decoder.decode(result.data))
64 return std::nullopt;
65 if (!decoder.decode(result.filePath))
66 return std::nullopt;
67 if (!decoder.decode(result.fileStart))
68 return std::nullopt;
69 if (!decoder.decode(result.fileLength))
70 return std::nullopt;
71 if (!decoder.decode(result.expectedFileModificationTime))
72 return std::nullopt;
73 if (!decoder.decode(result.blobURLString))
74 return std::nullopt;
75
76 return result;
77}

126std::optional<FrameState> FrameState::decode(IPC::Decoder& decoder)
127{
128 FrameState result;
129 if (!decoder.decode(result.urlString))
130 return std::nullopt;
131 if (!decoder.decode(result.originalURLString))
132 return std::nullopt;
133 if (!decoder.decode(result.referrer))
134 return std::nullopt;
135 if (!decoder.decode(result.target))
136 return std::nullopt;
137
138 if (!decoder.decode(result.m_documentState))
139 return std::nullopt;
140 result.validateDocumentState();
141
142 if (!decoder.decode(result.stateObjectData))
143 return std::nullopt;
144
145 if (!decoder.decode(result.documentSequenceNumber))
146 return std::nullopt;
147 if (!decoder.decode(result.itemSequenceNumber))
148 return std::nullopt;
149
150 if (!decoder.decode(result.scrollPosition))
151 return std::nullopt;
152 if (!decoder.decode(result.shouldRestoreScrollPosition))
153 return std::nullopt;
154 if (!decoder.decode(result.pageScaleFactor))
155 return std::nullopt;
156
157 if (!decoder.decode(result.httpBody))
158 return std::nullopt;
159
160#if PLATFORM(IOS_FAMILY)
161 if (!decoder.decode(result.exposedContentRect))
162 return std::nullopt;
163 if (!decoder.decode(result.unobscuredContentRect))
164 return std::nullopt;
165 if (!decoder.decode(result.minimumLayoutSizeInScrollViewCoordinates))
166 return std::nullopt;
167 if (!decoder.decode(result.contentSize))
168 return std::nullopt;
169 if (!decoder.decode(result.scaleIsInitial))
170 return std::nullopt;
171 if (!decoder.decode(result.obscuredInsets))
172 return std::nullopt;
173#endif
174
175 if (!decoder.decode(result.children))
176 return std::nullopt;
177
178 return result;
179}

181void PageState::encode(IPC::Encoder& encoder) const
182{
183 encoder << title << mainFrameState << !!sessionStateObject;
184
185 if (sessionStateObject)
186 encoder << sessionStateObject->toWireBytes();
187
188 encoder << shouldOpenExternalURLsPolicy;
189}
190
191bool PageState::decode(IPC::Decoder& decoder, PageState& result)
192{
193 if (!decoder.decode(result.title))
194 return false;
195 std::optional<FrameState> mainFrameState;
196 decoder >> mainFrameState;
197 if (!mainFrameState)
198 return false;
199 result.mainFrameState = WTFMove(*mainFrameState);
200
201 bool hasSessionState;
202 if (!decoder.decode(hasSessionState))
203 return false;
204
205 if (hasSessionState) {
206 Vector<uint8_t> wireBytes;
207 if (!decoder.decode(wireBytes))
208 return false;
209
210 result.sessionStateObject = SerializedScriptValue::createFromWireBytes(WTFMove(wireBytes));
211 }
212
213 std::optional<ShouldOpenExternalURLsPolicy> shouldOpenExternalURLsPolicy;
214 decoder >> shouldOpenExternalURLsPolicy;
215 if (!shouldOpenExternalURLsPolicy)
216 return false;
217
218 result.shouldOpenExternalURLsPolicy = *shouldOpenExternalURLsPolicy;
219 return true;
220}

229std::optional<BackForwardListItemState> BackForwardListItemState::decode(IPC::Decoder& decoder)
230{
231 BackForwardListItemState result;
232
233 auto identifier = BackForwardItemIdentifier::decode(decoder);
234 if (!identifier)
235 return std::nullopt;
236 result.identifier = *identifier;
237
238 if (!decoder.decode(result.pageState))
239 return std::nullopt;
240
241 if (!decoder.decode(result.hasCachedPage))
242 return std::nullopt;
243
244 return result;
245}

247void BackForwardListState::encode(IPC::Encoder& encoder) const
248{
249 encoder << items;

253std::optional<BackForwardListState> BackForwardListState::decode(IPC::Decoder& decoder)
254{
255 std::optional<Vector<BackForwardListItemState>> items;
256 decoder >> items;
257 if (!items)
258 return std::nullopt;
259
260 std::optional<uint32_t> currentIndex;
261 if (!decoder.decode(currentIndex))
262 return std::nullopt;
263
264 return {{ WTFMove(*items), WTFMove(currentIndex) }};
265}

267void FrameState::validateDocumentState() const
268{
269 for (auto& stateString : m_documentState) {
270 if (stateString.isNull())
271 continue;
272
273 if (!stateString.is8Bit())
274 continue;
275
276 // rdar://48634553 indicates 8-bit string can be invalid.
277 const LChar* characters8 = stateString.characters8();
278 for (unsigned i = 0; i < stateString.length(); ++i) {
279 auto character = characters8[i];
280 RELEASE_ASSERT(isLatin1(character));
281 }
282 }
283}

288
289 if (shouldValidate == ShouldValidate::Yes)
290 validateDocumentState();
291}
