47struct HTTPBody {
48 struct Element {
49 void encode(IPC::Encoder&) const;
50 static std::optional<Element> decode(IPC::Decoder&);
51
52 enum class Type {
53 Data,
54 File,
55 Blob,
56 };
57
58 // FIXME: This should be a Variant. It's also unclear why we don't just use FormDataElement here.
59 Type type = Type::Data;
60
61 // Data.
62 Vector<uint8_t> data;
63
64 // File.
65 String filePath;
66 int64_t fileStart;
67 std::optional<int64_t> fileLength;
68 std::optional<WallTime> expectedFileModificationTime;
69
70 // Blob.
71 String blobURLString;
72 };
73
74 void encode(IPC::Encoder&) const;
75 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, HTTPBody&);
76
77 String contentType;
78 Vector<Element> elements;
79};

81class FrameState {
82public:
83 void encode(IPC::Encoder&) const;
84 static std::optional<FrameState> decode(IPC::Decoder&);
85
86 // These are used to help debug <rdar://problem/48634553>.
87 FrameState() { RELEASE_ASSERT(RunLoop::isMain()); }
88 ~FrameState() { RELEASE_ASSERT(RunLoop::isMain()); }
89 const Vector<String>& documentState() const { return m_documentState; }
90 enum class ShouldValidate : bool { No, Yes };
91 void setDocumentState(const Vector<String>&, ShouldValidate = ShouldValidate::No);
92 void validateDocumentState() const;
93
94 String urlString;
95 String originalURLString;
96 String referrer;
97 String target;
98
99 std::optional<Vector<uint8_t>> stateObjectData;
100
101 int64_t documentSequenceNumber { 0 };
102 int64_t itemSequenceNumber { 0 };
103
104 WebCore::IntPoint scrollPosition;
105 bool shouldRestoreScrollPosition { true };
106 float pageScaleFactor { 0 };
107
108 std::optional<HTTPBody> httpBody;
109
110 // FIXME: These should not be per frame.
111#if PLATFORM(IOS_FAMILY)
112 WebCore::FloatRect exposedContentRect;
113 WebCore::IntRect unobscuredContentRect;
114 WebCore::FloatSize minimumLayoutSizeInScrollViewCoordinates;
115 WebCore::IntSize contentSize;
116 bool scaleIsInitial { false };
117 WebCore::FloatBoxExtent obscuredInsets;
118#endif
119
120 Vector<FrameState> children;
121
122private:
123 Vector<String> m_documentState;
124};
