35#if USE(DIRECT2D)
36interface ID2D1Bitmap;
37interface ID2D1RenderTarget;
38interface ID3D11Device1;
39interface IDXGIKeyedMutex;
40interface IDXGISurface1;
41
42#include <WebCore/COMPtr.h>
43#endif

45namespace WebCore {
46class Image;
47class GraphicsContext;
48}
49
50namespace WebKit {
51
52class ShareableBitmap : public ThreadSafeRefCounted<ShareableBitmap> {
53public:
54 struct Configuration {
55 std::optional<WebCore::DestinationColorSpace> colorSpace;
56 bool isOpaque { false };
57#if USE(DIRECT2D)
58 mutable HANDLE sharedResourceHandle { nullptr };
59#endif
60
61 void encode(IPC::Encoder&) const;
62 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, Configuration&);
63 };
64
65 class Handle {
66 WTF_MAKE_NONCOPYABLE(Handle);
67 public:
68 Handle();
69 Handle(Handle&&) = default;
70 Handle& operator=(Handle&&) = default;
71
72 bool isNull() const { return m_handle.isNull(); }
73
74 void clear();
75
76 void encode(IPC::Encoder&) const;
77 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, Handle&);
78
79 private:
80 friend class ShareableBitmap;
81
82 mutable SharedMemory::Handle m_handle;
83 WebCore::IntSize m_size;
84 Configuration m_configuration;
85 };
86
87 static CheckedUint32 numBytesForSize(WebCore::IntSize, const ShareableBitmap::Configuration&);
88 static CheckedUint32 calculateBytesPerRow(WebCore::IntSize, const Configuration&);
89 static CheckedUint32 calculateBytesPerPixel(const Configuration&);
90
91 // Create a shareable bitmap that uses malloced memory.
92 static RefPtr<ShareableBitmap> create(const WebCore::IntSize&, Configuration);
93
94 // Create a shareable bitmap whose backing memory can be shared with another process.
95 static RefPtr<ShareableBitmap> createShareable(const WebCore::IntSize&, Configuration);
96
97 // Create a shareable bitmap from an already existing shared memory block.
98 static RefPtr<ShareableBitmap> create(const WebCore::IntSize&, Configuration, RefPtr<SharedMemory>);
99
100 // Create a shareable bitmap from a handle.
101 static RefPtr<ShareableBitmap> create(const Handle&, SharedMemory::Protection = SharedMemory::Protection::ReadWrite);
102
103 // Create a handle.
104 bool createHandle(Handle&, SharedMemory::Protection = SharedMemory::Protection::ReadWrite) const;
105
106 ~ShareableBitmap();
107
108 const WebCore::IntSize& size() const { return m_size; }
109 WebCore::IntRect bounds() const { return WebCore::IntRect(WebCore::IntPoint(), size()); }
110
111 // Create a graphics context that can be used to paint into the backing store.
112 std::unique_ptr<WebCore::GraphicsContext> createGraphicsContext();
113
114 // Paint the backing store into the given context.
115 void paint(WebCore::GraphicsContext&, const WebCore::IntPoint& destination, const WebCore::IntRect& source);
116 void paint(WebCore::GraphicsContext&, float scaleFactor, const WebCore::IntPoint& destination, const WebCore::IntRect& source);
117
118 bool isBackedBySharedMemory() const { return m_sharedMemory; }
119
120 // This creates a bitmap image that directly references the shared bitmap data.
121 // This is only safe to use when we know that the contents of the shareable bitmap won't change.
122 RefPtr<WebCore::Image> createImage();
123
124#if USE(CG)
125 // This creates a copied CGImageRef (most likely a copy-on-write) of the shareable bitmap.
126 RetainPtr<CGImageRef> makeCGImageCopy();
127
128 // This creates a CGImageRef that directly references the shared bitmap data.
129 // This is only safe to use when we know that the contents of the shareable bitmap won't change.
130 RetainPtr<CGImageRef> makeCGImage();
131
132 WebCore::PlatformImagePtr createPlatformImage() { return makeCGImageCopy(); }
133#elif USE(CAIRO)
134 // This creates a BitmapImage that directly references the shared bitmap data.
135 // This is only safe to use when we know that the contents of the shareable bitmap won't change.
136 RefPtr<cairo_surface_t> createCairoSurface();
137
138 WebCore::PlatformImagePtr createPlatformImage() { return createCairoSurface(); }
139#elif USE(DIRECT2D)
140 COMPtr<ID2D1Bitmap> createDirect2DSurface(ID3D11Device1*, ID2D1RenderTarget*);
141 IDXGISurface1* dxSurface() { return m_surface.get(); }
142 void createSharedResource();
143 void disposeSharedResource();
144 void leakSharedResource();
145
146 WebCore::PlatformImagePtr createPlatformImage() { return nullptr; }
147#endif
148
149private:
150 ShareableBitmap(const WebCore::IntSize&, Configuration, void*);
151 ShareableBitmap(const WebCore::IntSize&, Configuration, RefPtr<SharedMemory>);
152
153#if USE(CG)
154 RetainPtr<CGImageRef> createCGImage(CGDataProviderRef) const;
155 static void releaseBitmapContextData(void* typelessBitmap, void* typelessData);
156 static void releaseDataProviderData(void* typelessBitmap, const void* typelessData, size_t);
157#endif
158
159#if USE(CAIRO)
160 static void releaseSurfaceData(void* typelessBitmap);
161#endif
162
163public:
164 void* data() const;
165 size_t bytesPerRow() const { return calculateBytesPerRow(m_size, m_configuration); }
166
167private:
168 size_t sizeInBytes() const { return numBytesForSize(m_size, m_configuration); }
169
170 WebCore::IntSize m_size;
171 Configuration m_configuration;
172
173#if USE(DIRECT2D)
174 COMPtr<IDXGISurface1> m_surface;
175 COMPtr<IDXGIKeyedMutex> m_surfaceMutex;
176 COMPtr<ID2D1Bitmap> m_bitmap;
177#endif
178
179 // If the shareable bitmap is backed by shared memory, this points to the shared memory object.
180 RefPtr<SharedMemory> m_sharedMemory;
181
182 // If the shareable bitmap is backed by fastMalloced memory, this points to the data.
183 void* m_data { nullptr };
184};
185
186} // namespace WebKit
