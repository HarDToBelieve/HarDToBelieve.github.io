37ShareableResource::Handle::Handle() = default;
38
39void ShareableResource::Handle::encode(IPC::Encoder& encoder) const
40{
41 encoder << SharedMemory::IPCHandle { WTFMove(m_handle), m_size };
42 encoder << m_offset;
43}
44
45bool ShareableResource::Handle::decode(IPC::Decoder& decoder, Handle& handle)
46{
47 SharedMemory::IPCHandle ipcHandle;
48 if (!decoder.decode(ipcHandle))
49 return false;
50 if (!decoder.decode(handle.m_offset))
51 return false;
52
53 handle.m_size = ipcHandle.dataSize;
54 handle.m_handle = WTFMove(ipcHandle.handle);
55 return true;
56}

58#if USE(CF)
59static void shareableResourceDeallocate(void *ptr, void *info)
60{
61 static_cast<ShareableResource*>(info)->deref(); // Balanced by ref() in createShareableResourceDeallocator()
62}
63
64static RetainPtr<CFAllocatorRef> createShareableResourceDeallocator(ShareableResource* resource)
65{
66 CFAllocatorContext context = { 0,
67 resource,
68 NULL, // retain
69 NULL, // release
70 NULL, // copyDescription
71 NULL, // allocate
72 NULL, // reallocate
73 shareableResourceDeallocate,
74 NULL, // preferredSize
75 };
76
77 return adoptCF(CFAllocatorCreate(kCFAllocatorDefault, &context));
78}
79#endif

81RefPtr<SharedBuffer> ShareableResource::wrapInSharedBuffer()
82{
83 ref(); // Balanced by deref when SharedBuffer is deallocated.
84
85#if USE(CF)
86 auto deallocator = createShareableResourceDeallocator(this);
87 auto cfData = adoptCF(CFDataCreateWithBytesNoCopy(kCFAllocatorDefault, data(), static_cast<CFIndex>(size()), deallocator.get()));
88 return SharedBuffer::create(cfData.get());
89#elif USE(GLIB)
90 GRefPtr<GBytes> bytes = adoptGRef(g_bytes_new_with_free_func(data(), size(), [](void* data) {
91 static_cast<ShareableResource*>(data)->deref();
92 }, this));
93 return SharedBuffer::create(bytes.get());
94#else
95 ASSERT_NOT_REACHED();
96 return nullptr;
97#endif
98}
99
100RefPtr<SharedBuffer> ShareableResource::Handle::tryWrapInSharedBuffer() const
101{
102 RefPtr<ShareableResource> resource = ShareableResource::map(*this);
103 if (!resource) {
104 LOG_ERROR("Failed to recreate ShareableResource from handle.");
105 return nullptr;
106 }
107
108 return resource->wrapInSharedBuffer();
109}

111RefPtr<ShareableResource> ShareableResource::create(Ref<SharedMemory>&& sharedMemory, unsigned offset, unsigned size)
112{
113 auto totalSize = CheckedSize(offset) + size;
114 if (totalSize.hasOverflowed()) {
115 LOG_ERROR("Failed to create ShareableResource from SharedMemory due to overflow.");
116 return nullptr;
117 }
118 if (totalSize > sharedMemory->size()) {
119 LOG_ERROR("Failed to create ShareableResource from SharedMemory due to mismatched buffer size.");
120 return nullptr;
121 }
122 return adoptRef(*new ShareableResource(WTFMove(sharedMemory), offset, size));
123}
124
125RefPtr<ShareableResource> ShareableResource::map(const Handle& handle)
126{
127 auto sharedMemory = SharedMemory::map(handle.m_handle, SharedMemory::Protection::ReadOnly);
128 if (!sharedMemory)
129 return nullptr;
130
131 return create(sharedMemory.releaseNonNull(), handle.m_offset, handle.m_size);
132}

134ShareableResource::ShareableResource(Ref<SharedMemory>&& sharedMemory, unsigned offset, unsigned size)
135 : m_sharedMemory(WTFMove(sharedMemory))
136 , m_offset(offset)
137 , m_size(size)
138{
139}

141ShareableResource::~ShareableResource()
142{
143}
144
145bool ShareableResource::createHandle(Handle& handle)
146{
147 if (!m_sharedMemory->createHandle(handle.m_handle, SharedMemory::Protection::ReadOnly))
148 return false;
149
150 handle.m_offset = m_offset;
151 handle.m_size = m_size;
152
153 return true;
154}
155
156const uint8_t* ShareableResource::data() const
157{
158 return static_cast<const uint8_t*>(m_sharedMemory->data()) + m_offset;
159}
160
161unsigned ShareableResource::size() const
162{
163 return m_size;
164}
