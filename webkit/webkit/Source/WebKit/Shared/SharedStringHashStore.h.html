27
28#include "SharedMemory.h"
29#include "SharedStringHashTable.h"
30#include <WebCore/SharedStringHash.h>
31#include <wtf/RunLoop.h>
32
33namespace WebKit {
34
35class SharedStringHashStore {
36public:
37 class Client {
38 public:
39 virtual ~Client() { }
40
41 virtual void didInvalidateSharedMemory() = 0;
42 virtual void didUpdateSharedStringHashes(const Vector<WebCore::SharedStringHash>& addedHashes, const Vector<WebCore::SharedStringHash>& removedHashes) { };
43 };
44
45 SharedStringHashStore(Client&);
46
47 bool createSharedMemoryHandle(SharedMemory::Handle&);
48
49 void scheduleAddition(WebCore::SharedStringHash);
50 void scheduleRemoval(WebCore::SharedStringHash);
51
52 bool contains(WebCore::SharedStringHash);
53 void clear();
54
55 bool isEmpty() const { return !m_keyCount; }
56
57 void flushPendingChanges();
58
59private:
60 void resizeTable(unsigned newTableSize);
61 void processPendingOperations();
62
63 struct Operation {
64 enum Type { Add, Remove };
65 Type type;
66 WebCore::SharedStringHash sharedStringHash;
67 };
68
69 Client& m_client;
70 unsigned m_keyCount { 0 };
71 unsigned m_tableSize { 0 };
72 SharedStringHashTable m_table;
73 Vector<Operation> m_pendingOperations;
74 RunLoop::Timer<SharedStringHashStore> m_pendingOperationsTimer;
75};
76
77} // namespace WebKit
