41TouchBarMenuData::TouchBarMenuData(const WebCore::HTMLMenuElement& element)
42{
43 if (!element.isTouchBarMenu())
44 return;
45
46 // FIXME: We can't rely on using the 'id' attribute of the element here to distinguish
47 // between different menu items. For instance, a menuitem may have the same 'id' as
48 // another, or have no 'id' at all.
49 m_id = element.attributeWithoutSynchronization(WebCore::HTMLNames::idAttr);
50 m_isPageCustomized = true;
51}
52
53void TouchBarMenuData::addMenuItem(const TouchBarMenuItemData& data)
54{
55 m_items.append(data);
56}
57
58void TouchBarMenuData::removeMenuItem(const TouchBarMenuItemData& data)
59{
60 m_items.removeFirst(data);
61}
62
63void TouchBarMenuData::encode(IPC::Encoder& encoder) const
64{
65 encoder << m_items;
66}
67
68bool TouchBarMenuData::decode(IPC::Decoder& decoder, TouchBarMenuData& data)
69{
70 return decoder.decode(data.m_items);
71}
