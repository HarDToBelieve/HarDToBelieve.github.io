43namespace WebKit {
44
45// Based on NSTouchBarItem types.
46enum ItemType {
47 Button
48};
49
50struct TouchBarMenuItemData {
51 explicit TouchBarMenuItemData() = default;
52 explicit TouchBarMenuItemData(const WebCore::HTMLMenuItemElement&);
53 explicit TouchBarMenuItemData(const TouchBarMenuItemData&) = default;
54
55 void encode(IPC::Encoder&) const;
56 static std::optional<TouchBarMenuItemData> decode(IPC::Decoder&);
57
58 ItemType type { ItemType::Button };
59 String identifier;
60 float priority { 0.0 };
61 bool validTouchBarDisplay { true };
62};
63
64// Touch Bar Menu Items will be ordered based on priority.
65inline bool operator<(const TouchBarMenuItemData& lhs, const TouchBarMenuItemData& rhs)
66{
67 return lhs.priority < rhs.priority;
68}
69
70inline bool operator>(const TouchBarMenuItemData& lhs, const TouchBarMenuItemData& rhs)
71{
72 return rhs < lhs;
73}
74
75inline bool operator<=(const TouchBarMenuItemData& lhs, const TouchBarMenuItemData& rhs)
76{
77 return !(lhs > rhs);
78}
79
80inline bool operator>=(const TouchBarMenuItemData& lhs, const TouchBarMenuItemData& rhs)
81{
82 return !(lhs < rhs);
83}
84
85inline bool operator==(const TouchBarMenuItemData& lhs, const TouchBarMenuItemData& rhs)
86{
87 return lhs.type == rhs.type
88 && lhs.identifier == rhs.identifier
89 && lhs.priority == rhs.priority;
90}
91
92inline bool operator!=(const TouchBarMenuItemData& lhs, const TouchBarMenuItemData& rhs)
93{
94 return !(lhs == rhs);
95}
