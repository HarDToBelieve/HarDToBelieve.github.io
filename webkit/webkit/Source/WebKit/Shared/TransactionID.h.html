26#pragma once
27
28#include <wtf/HashFunctions.h>
29#include <wtf/HashTraits.h>
30#include <wtf/text/TextStream.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebKit {
34
35template<typename T> class MonotonicObjectIdentifier {
36public:
37 MonotonicObjectIdentifier() = default;
38
39 MonotonicObjectIdentifier(WTF::HashTableDeletedValueType)
40 : m_identifier(hashTableDeletedValue())
41 { }
42
43 bool isHashTableDeletedValue() const { return m_identifier == hashTableDeletedValue(); }
44
45 template<typename Encoder> void encode(Encoder& encoder) const
46 {
47 ASSERT(isValidIdentifier(m_identifier));
48 encoder << m_identifier;
49 }
50 template<typename Decoder> static std::optional<MonotonicObjectIdentifier> decode(Decoder& decoder)
51 {
52 std::optional<uint64_t> identifier;
53 decoder >> identifier;
54 if (!identifier)
55 return std::nullopt;
56 ASSERT(isValidIdentifier(*identifier));
57 return MonotonicObjectIdentifier { *identifier };
58 }
59
60 bool operator==(const MonotonicObjectIdentifier& other) const
61 {
62 return m_identifier == other.m_identifier;
63 }
64
65 bool operator>(const MonotonicObjectIdentifier& other) const
66 {
67 return m_identifier > other.m_identifier;
68 }
69
70 bool operator>=(const MonotonicObjectIdentifier& other) const
71 {
72 return m_identifier >= other.m_identifier;
73 }
74
75 bool operator<(const MonotonicObjectIdentifier& other) const
76 {
77 return m_identifier < other.m_identifier;
78 }
79
80 bool operator<=(const MonotonicObjectIdentifier& other) const
81 {
82 return m_identifier <= other.m_identifier;
83 }
84
85 bool operator!=(const MonotonicObjectIdentifier& other) const
86 {
87 return m_identifier != other.m_identifier;
88 }
89
90 MonotonicObjectIdentifier& increment()
91 {
92 ++m_identifier;
93 return *this;
94 }
95
96 MonotonicObjectIdentifier next() const
97 {
98 return MonotonicObjectIdentifier(m_identifier + 1);
99 }
100
101 uint64_t toUInt64() const { return m_identifier; }
102 explicit operator bool() const { return m_identifier; }
103
104 String loggingString() const
105 {
106 return String::number(m_identifier);
107 }
108
109private:
110 template<typename U> friend MonotonicObjectIdentifier<U> makeMonotonicObjectIdentifier(uint64_t);
111 friend struct HashTraits<MonotonicObjectIdentifier>;
112 template<typename U> friend struct MonotonicObjectIdentifierHash;
113
114 static uint64_t hashTableDeletedValue() { return std::numeric_limits<uint64_t>::max(); }
115 static bool isValidIdentifier(uint64_t identifier) { return identifier != hashTableDeletedValue(); }
116
117 explicit MonotonicObjectIdentifier(uint64_t identifier)
118 : m_identifier(identifier)
119 {
120 }
121
122 uint64_t m_identifier { 0 };
123};
124
125template<typename T>
126TextStream& operator<<(TextStream& ts, const MonotonicObjectIdentifier<T>& identifier)
127{
128 ts << identifier.toUInt64();
129 return ts;
130}
131
132enum TransactionIDType { };
133using TransactionID = MonotonicObjectIdentifier<TransactionIDType>;
