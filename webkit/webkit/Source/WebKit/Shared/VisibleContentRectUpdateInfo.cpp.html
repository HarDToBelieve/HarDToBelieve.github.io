39void VisibleContentRectUpdateInfo::encode(IPC::Encoder& encoder) const
40{
41 encoder << m_exposedContentRect;
42 encoder << m_unobscuredContentRect;
43 encoder << m_contentInsets;
44 encoder << m_unobscuredContentRectRespectingInputViewBounds;
45 encoder << m_unobscuredRectInScrollViewCoordinates;
46 encoder << m_layoutViewportRect;
47 encoder << m_obscuredInsets;
48 encoder << m_unobscuredSafeAreaInsets;
49 encoder << m_scrollVelocity;
50 encoder << m_lastLayerTreeTransactionID;
51 encoder << m_scale;
52 encoder << m_viewStability;
53 encoder << m_isFirstUpdateForNewViewSize;
54 encoder << m_allowShrinkToFit;
55 encoder << m_enclosedInScrollableAncestorView;
56}
57
58bool VisibleContentRectUpdateInfo::decode(IPC::Decoder& decoder, VisibleContentRectUpdateInfo& result)
59{
60 if (!decoder.decode(result.m_exposedContentRect))
61 return false;
62 if (!decoder.decode(result.m_unobscuredContentRect))
63 return false;
64 if (!decoder.decode(result.m_contentInsets))
65 return false;
66 if (!decoder.decode(result.m_unobscuredContentRectRespectingInputViewBounds))
67 return false;
68 if (!decoder.decode(result.m_unobscuredRectInScrollViewCoordinates))
69 return false;
70 if (!decoder.decode(result.m_layoutViewportRect))
71 return false;
72 if (!decoder.decode(result.m_obscuredInsets))
73 return false;
74 if (!decoder.decode(result.m_unobscuredSafeAreaInsets))
75 return false;
76 if (!decoder.decode(result.m_scrollVelocity))
77 return false;
78 if (!decoder.decode(result.m_lastLayerTreeTransactionID))
79 return false;
80 if (!decoder.decode(result.m_scale))
81 return false;
82 if (!decoder.decode(result.m_viewStability))
83 return false;
84 if (!decoder.decode(result.m_isFirstUpdateForNewViewSize))
85 return false;
86 if (!decoder.decode(result.m_allowShrinkToFit))
87 return false;
88 if (!decoder.decode(result.m_enclosedInScrollableAncestorView))
89 return false;
90
91 return true;
92}

94String VisibleContentRectUpdateInfo::dump() const
95{
96 TextStream stream;
97 stream << *this;
98 return stream.release();
99}
100
101TextStream& operator<<(TextStream& ts, ViewStabilityFlag stabilityFlag)
102{
103 switch (stabilityFlag) {
104 case ViewStabilityFlag::ScrollViewInteracting: ts << "scroll view interacting"; break;
105 case ViewStabilityFlag::ScrollViewAnimatedScrollOrZoom: ts << "scroll view animated scroll or zoom"; break;
106 case ViewStabilityFlag::ScrollViewRubberBanding: ts << "scroll view rubberbanding"; break;
107 case ViewStabilityFlag::ChangingObscuredInsetsInteractively: ts << "changing obscured insets interactively"; break;
108 case ViewStabilityFlag::UnstableForTesting: ts << "unstable for testing"; break;
109 }
110 return ts;
111}
112
113TextStream& operator<<(TextStream& ts, const VisibleContentRectUpdateInfo& info)
114{
115 TextStream::GroupScope scope(ts);
116
117 ts << "VisibleContentRectUpdateInfo";
118
119 ts.dumpProperty("lastLayerTreeTransactionID", info.lastLayerTreeTransactionID());
120
121 ts.dumpProperty("exposedContentRect", info.exposedContentRect());
122 ts.dumpProperty("unobscuredContentRect", info.unobscuredContentRect());
123 ts.dumpProperty("contentInsets", info.contentInsets());
124 ts.dumpProperty("unobscuredContentRectRespectingInputViewBounds", info.unobscuredContentRectRespectingInputViewBounds());
125 ts.dumpProperty("unobscuredRectInScrollViewCoordinates", info.unobscuredRectInScrollViewCoordinates());
126 ts.dumpProperty("layoutViewportRect", info.layoutViewportRect());
127 ts.dumpProperty("obscuredInsets", info.obscuredInsets());
128 ts.dumpProperty("unobscuredSafeAreaInsets", info.unobscuredSafeAreaInsets());
129
130 ts.dumpProperty("scale", info.scale());
131 ts.dumpProperty("viewStability", info.viewStability());
132 ts.dumpProperty("isFirstUpdateForNewViewSize", info.isFirstUpdateForNewViewSize());
133 if (info.enclosedInScrollableAncestorView())
134 ts.dumpProperty("enclosedInScrollableAncestorView", info.enclosedInScrollableAncestorView());
135
136 ts.dumpProperty("allowShrinkToFit", info.allowShrinkToFit());
137 ts.dumpProperty("scrollVelocity", info.scrollVelocity());
138
139 return ts;
140}
