49enum class ViewStabilityFlag : uint8_t {
50 ScrollViewInteracting = 1 << 0, // Dragging, zooming, interrupting deceleration
51 ScrollViewAnimatedScrollOrZoom = 1 << 1, // Decelerating, scrolling to top, animated zoom
52 ScrollViewRubberBanding = 1 << 2,
53 ChangingObscuredInsetsInteractively = 1 << 3,
54 UnstableForTesting = 1 << 4
55};
56
57class VisibleContentRectUpdateInfo {
58public:
59 VisibleContentRectUpdateInfo() = default;
60
61 VisibleContentRectUpdateInfo(const WebCore::FloatRect& exposedContentRect, const WebCore::FloatRect& unobscuredContentRect, const WebCore::FloatBoxExtent& contentInsets, const WebCore::FloatRect& unobscuredRectInScrollViewCoordinates, const WebCore::FloatRect& unobscuredContentRectRespectingInputViewBounds, const WebCore::FloatRect& layoutViewportRect, const WebCore::FloatBoxExtent& obscuredInsets, const WebCore::FloatBoxExtent& unobscuredSafeAreaInsets, double scale, OptionSet<ViewStabilityFlag> viewStability, bool isFirstUpdateForNewViewSize, bool allowShrinkToFit, bool enclosedInScrollableAncestorView, const WebCore::VelocityData& scrollVelocity, TransactionID lastLayerTreeTransactionId)
62 : m_exposedContentRect(exposedContentRect)
63 , m_unobscuredContentRect(unobscuredContentRect)
64 , m_contentInsets(contentInsets)
65 , m_unobscuredContentRectRespectingInputViewBounds(unobscuredContentRectRespectingInputViewBounds)
66 , m_unobscuredRectInScrollViewCoordinates(unobscuredRectInScrollViewCoordinates)
67 , m_layoutViewportRect(layoutViewportRect)
68 , m_obscuredInsets(obscuredInsets)
69 , m_unobscuredSafeAreaInsets(unobscuredSafeAreaInsets)
70 , m_scrollVelocity(scrollVelocity)
71 , m_lastLayerTreeTransactionID(lastLayerTreeTransactionId)
72 , m_scale(scale)
73 , m_viewStability(viewStability)
74 , m_isFirstUpdateForNewViewSize(isFirstUpdateForNewViewSize)
75 , m_allowShrinkToFit(allowShrinkToFit)
76 , m_enclosedInScrollableAncestorView(enclosedInScrollableAncestorView)
77 {
78 }
79
80 const WebCore::FloatRect& exposedContentRect() const { return m_exposedContentRect; }
81 const WebCore::FloatRect& unobscuredContentRect() const { return m_unobscuredContentRect; }
82 const WebCore::FloatBoxExtent& contentInsets() const { return m_contentInsets; }
83 const WebCore::FloatRect& unobscuredRectInScrollViewCoordinates() const { return m_unobscuredRectInScrollViewCoordinates; }
84 const WebCore::FloatRect& unobscuredContentRectRespectingInputViewBounds() const { return m_unobscuredContentRectRespectingInputViewBounds; }
85 const WebCore::FloatRect& layoutViewportRect() const { return m_layoutViewportRect; }
86 const WebCore::VelocityData& scrollVelocity() const { return m_scrollVelocity; }
87 const WebCore::FloatBoxExtent& obscuredInsets() const { return m_obscuredInsets; }
88 const WebCore::FloatBoxExtent& unobscuredSafeAreaInsets() const { return m_unobscuredSafeAreaInsets; }
89
90 double scale() const { return m_scale; }
91 bool inStableState() const { return m_viewStability.isEmpty(); }
92 OptionSet<ViewStabilityFlag> viewStability() const { return m_viewStability; }
93 bool isFirstUpdateForNewViewSize() const { return m_isFirstUpdateForNewViewSize; }
94 bool allowShrinkToFit() const { return m_allowShrinkToFit; }
95 bool enclosedInScrollableAncestorView() const { return m_enclosedInScrollableAncestorView; }
96 TransactionID lastLayerTreeTransactionID() const { return m_lastLayerTreeTransactionID; }
97
98 MonotonicTime timestamp() const { return m_scrollVelocity.lastUpdateTime; }
99
100 void encode(IPC::Encoder&) const;
101 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, VisibleContentRectUpdateInfo&);
102
103 String dump() const;
104
105private:
106 WebCore::FloatRect m_exposedContentRect;
107 WebCore::FloatRect m_unobscuredContentRect;
108 WebCore::FloatBoxExtent m_contentInsets;
109 WebCore::FloatRect m_unobscuredContentRectRespectingInputViewBounds;
110 WebCore::FloatRect m_unobscuredRectInScrollViewCoordinates;
111 WebCore::FloatRect m_layoutViewportRect;
112 WebCore::FloatBoxExtent m_obscuredInsets;
113 WebCore::FloatBoxExtent m_unobscuredSafeAreaInsets;
114 WebCore::VelocityData m_scrollVelocity;
115 TransactionID m_lastLayerTreeTransactionID;
116 double m_scale { -1 };
117 OptionSet<ViewStabilityFlag> m_viewStability;
118 bool m_isFirstUpdateForNewViewSize { false };
119 bool m_allowShrinkToFit { false };
120 bool m_enclosedInScrollableAncestorView { false };
121};
122
123inline bool operator==(const VisibleContentRectUpdateInfo& a, const VisibleContentRectUpdateInfo& b)
124{
125 // Note: the comparison doesn't include timestamp and velocity since we care about equality based on the other data.
126 return a.scale() == b.scale()
127 && a.exposedContentRect() == b.exposedContentRect()
128 && a.unobscuredContentRect() == b.unobscuredContentRect()
129 && a.contentInsets() == b.contentInsets()
130 && a.unobscuredContentRectRespectingInputViewBounds() == b.unobscuredContentRectRespectingInputViewBounds()
131 && a.layoutViewportRect() == b.layoutViewportRect()
132 && a.obscuredInsets() == b.obscuredInsets()
133 && a.unobscuredSafeAreaInsets() == b.unobscuredSafeAreaInsets()
134 && a.scrollVelocity().equalIgnoringTimestamp(b.scrollVelocity())
135 && a.viewStability() == b.viewStability()
136 && a.isFirstUpdateForNewViewSize() == b.isFirstUpdateForNewViewSize()
137 && a.allowShrinkToFit() == b.allowShrinkToFit()
138 && a.enclosedInScrollableAncestorView() == b.enclosedInScrollableAncestorView();
139}
