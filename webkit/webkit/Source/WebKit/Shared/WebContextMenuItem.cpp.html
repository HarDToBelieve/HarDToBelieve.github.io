42Ref<WebContextMenuItem> WebContextMenuItem::create(const String& title, bool enabled, API::Array* submenuItems)
43{
44 size_t size = submenuItems->size();
45
46 Vector<WebContextMenuItemData> submenu;
47 submenu.reserveCapacity(size);
48
49 for (size_t i = 0; i < size; ++i) {
50 WebContextMenuItem* item = submenuItems->at<WebContextMenuItem>(i);
51 if (item)
52 submenu.append(item->data());
53 }
54
55 return adoptRef(*new WebContextMenuItem(WebContextMenuItemData(WebCore::ContextMenuItemTagNoAction, title, enabled, submenu))).leakRef();
56}

58WebContextMenuItem* WebContextMenuItem::separatorItem()
59{
60 static WebContextMenuItem* separatorItem = new WebContextMenuItem(WebContextMenuItemData(WebCore::SeparatorType, WebCore::ContextMenuItemTagNoAction, String(), true, false));
61 return separatorItem;
62}
63
64Ref<API::Array> WebContextMenuItem::submenuItemsAsAPIArray() const
65{
66 if (m_webContextMenuItemData.type() != WebCore::SubmenuType)
67 return API::Array::create();
68
69 Vector<RefPtr<API::Object>> submenuItems;
70 submenuItems.reserveInitialCapacity(m_webContextMenuItemData.submenu().size());
71
72 for (const auto& item : m_webContextMenuItemData.submenu())
73 submenuItems.uncheckedAppend(WebContextMenuItem::create(item));
74
75 return API::Array::create(WTFMove(submenuItems));
76}
