71WebContextMenuItemData::WebContextMenuItemData(const WebCore::ContextMenuItem& item)
72 : m_type(item.type())
73 , m_action(item.action())
74 , m_title(item.title())
75{
76 if (m_type == WebCore::SubmenuType) {
77 const Vector<WebCore::ContextMenuItem>& coreSubmenu = item.subMenuItems();
78 m_submenu = kitItems(coreSubmenu);
79 }
80
81 m_enabled = item.enabled();
82 m_checked = item.checked();
83 m_indentationLevel = item.indentationLevel();
84}
85
86ContextMenuItem WebContextMenuItemData::core() const
87{
88 if (m_type != SubmenuType)
89 return ContextMenuItem(m_type, m_action, m_title, m_enabled, m_checked, m_indentationLevel);
90
91 Vector<ContextMenuItem> subMenuItems = coreItems(m_submenu);
92 return ContextMenuItem(m_action, m_title, m_enabled, m_checked, subMenuItems, m_indentationLevel);
93}

116std::optional<WebContextMenuItemData> WebContextMenuItemData::decode(IPC::Decoder& decoder)
117{
118 WebCore::ContextMenuItemType type;
119 if (!decoder.decode(type))
120 return std::nullopt;
121
122 WebCore::ContextMenuAction action;
123 if (!decoder.decode(action))
124 return std::nullopt;
125
126 String title;
127 if (!decoder.decode(title))
128 return std::nullopt;
129
130 bool checked;
131 if (!decoder.decode(checked))
132 return std::nullopt;
133
134 bool enabled;
135 if (!decoder.decode(enabled))
136 return std::nullopt;
137
138 unsigned indentationLevel;
139 if (!decoder.decode(indentationLevel))
140 return std::nullopt;
141
142 std::optional<Vector<WebContextMenuItemData>> submenu;
143 decoder >> submenu;
144 if (!submenu)
145 return std::nullopt;
146
147 switch (type) {
148 case WebCore::ActionType:
149 case WebCore::SeparatorType:
150 case WebCore::CheckableActionType:
151 return {{ type, action, title, enabled, checked, indentationLevel }};
152 break;
153 case WebCore::SubmenuType:
154 return {{ action, title, enabled, WTFMove(*submenu), indentationLevel }};
155 break;
156 }
157 ASSERT_NOT_REACHED();
158 return std::nullopt;
159}

161Vector<WebContextMenuItemData> kitItems(const Vector<WebCore::ContextMenuItem>& coreItemVector)
162{
163 Vector<WebContextMenuItemData> result;
164 result.reserveCapacity(coreItemVector.size());
165 for (unsigned i = 0; i < coreItemVector.size(); ++i)
166 result.append(WebContextMenuItemData(coreItemVector[i]));
167
168 return result;
169}
170
171Vector<ContextMenuItem> coreItems(const Vector<WebContextMenuItemData>& kitItemVector)
172{
173 Vector<ContextMenuItem> result;
174 result.reserveCapacity(kitItemVector.size());
175 for (unsigned i = 0; i < kitItemVector.size(); ++i)
176 result.append(kitItemVector[i].core());
177
178 return result;
179}
