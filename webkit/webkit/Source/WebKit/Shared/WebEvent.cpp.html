33namespace WebKit {
34
35WebEvent::WebEvent()
36 : m_type(static_cast<uint32_t>(NoType))
37{
38}
39
40WebEvent::WebEvent(Type type, OptionSet<Modifier> modifiers, WallTime timestamp)
41 : m_type(type)
42 , m_modifiers(modifiers)
43 , m_timestamp(timestamp)
44{
45}
46
47void WebEvent::encode(IPC::Encoder& encoder) const
48{
49 encoder << m_type;
50 encoder << m_modifiers;
51 encoder << m_timestamp;
52}
53
54bool WebEvent::decode(IPC::Decoder& decoder, WebEvent& result)
55{
56 if (!decoder.decode(result.m_type))
57 return false;
58 if (!decoder.decode(result.m_modifiers))
59 return false;
60 if (!decoder.decode(result.m_timestamp))
61 return false;
62 return true;
63}
