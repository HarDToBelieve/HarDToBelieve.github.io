64WebHitTestResultData::WebHitTestResultData(const WebCore::HitTestResult& hitTestResult, bool includeImage)
65 : absoluteImageURL(hitTestResult.absoluteImageURL().string())
66 , absolutePDFURL(hitTestResult.absolutePDFURL().string())
67 , absoluteLinkURL(hitTestResult.absoluteLinkURL().string())
68 , absoluteMediaURL(hitTestResult.absoluteMediaURL().string())
69 , linkLabel(hitTestResult.textContent())
70 , linkTitle(hitTestResult.titleDisplayString())
71 , linkSuggestedFilename(hitTestResult.linkSuggestedFilename())
72 , isContentEditable(hitTestResult.isContentEditable())
73 , elementBoundingBox(elementBoundingBoxInWindowCoordinates(hitTestResult))
74 , isScrollbar(IsScrollbar::No)
75 , isSelected(hitTestResult.isSelected())
76 , isTextNode(hitTestResult.innerNode() && hitTestResult.innerNode()->isTextNode())
77 , isOverTextInsideFormControlElement(hitTestResult.isOverTextInsideFormControlElement())
78 , isDownloadableMedia(hitTestResult.isDownloadableMedia())
79 , imageSize(0)
80{
81 if (auto* scrollbar = hitTestResult.scrollbar())
82 isScrollbar = scrollbar->orientation() == HorizontalScrollbar ? IsScrollbar::Horizontal : IsScrollbar::Vertical;
83
84 if (!includeImage)
85 return;
86
87 if (Image* image = hitTestResult.image()) {
88 RefPtr<SharedBuffer> buffer = image->data();
89 if (buffer) {
90 imageSharedMemory = WebKit::SharedMemory::allocate(buffer->size());
91 memcpy(imageSharedMemory->data(), buffer->data(), buffer->size());
92 imageSize = buffer->size();
93 }
94 }
95
96 if (auto target = makeRefPtr(hitTestResult.innerNonSharedNode()); target && is<RenderImage>(target->renderer()))
97 imageBitmap = createShareableBitmap(*downcast<RenderImage>(target->renderer()));
98}

104void WebHitTestResultData::encode(IPC::Encoder& encoder) const
105{
106 encoder << absoluteImageURL;
107 encoder << absolutePDFURL;
108 encoder << absoluteLinkURL;
109 encoder << absoluteMediaURL;
110 encoder << linkLabel;
111 encoder << linkTitle;
112 encoder << linkSuggestedFilename;
113 encoder << isContentEditable;
114 encoder << elementBoundingBox;
115 encoder << isScrollbar;
116 encoder << isSelected;
117 encoder << isTextNode;
118 encoder << isOverTextInsideFormControlElement;
119 encoder << isDownloadableMedia;
120 encoder << lookupText;
121 encoder << toolTipText;
122 encoder << dictionaryPopupInfo;
123
124 WebKit::SharedMemory::Handle imageHandle;
125 if (imageSharedMemory && imageSharedMemory->data())
126 imageSharedMemory->createHandle(imageHandle, WebKit::SharedMemory::Protection::ReadOnly);
127
128 encoder << WebKit::SharedMemory::IPCHandle { WTFMove(imageHandle), imageSize };
129
130 ShareableBitmap::Handle imageBitmapHandle;
131 if (imageBitmap)
132 imageBitmap->createHandle(imageBitmapHandle, SharedMemory::Protection::ReadOnly);
133 encoder << imageBitmapHandle;
134
135 bool hasLinkTextIndicator = linkTextIndicator;
136 encoder << hasLinkTextIndicator;
137 if (hasLinkTextIndicator)
138 encoder << linkTextIndicator->data();
139
140 platformEncode(encoder);
141}

143bool WebHitTestResultData::decode(IPC::Decoder& decoder, WebHitTestResultData& hitTestResultData)
144{
145 if (!decoder.decode(hitTestResultData.absoluteImageURL)
146 || !decoder.decode(hitTestResultData.absolutePDFURL)
147 || !decoder.decode(hitTestResultData.absoluteLinkURL)
148 || !decoder.decode(hitTestResultData.absoluteMediaURL)
149 || !decoder.decode(hitTestResultData.linkLabel)
150 || !decoder.decode(hitTestResultData.linkTitle)
151 || !decoder.decode(hitTestResultData.linkSuggestedFilename)
152 || !decoder.decode(hitTestResultData.isContentEditable)
153 || !decoder.decode(hitTestResultData.elementBoundingBox)
154 || !decoder.decode(hitTestResultData.isScrollbar)
155 || !decoder.decode(hitTestResultData.isSelected)
156 || !decoder.decode(hitTestResultData.isTextNode)
157 || !decoder.decode(hitTestResultData.isOverTextInsideFormControlElement)
158 || !decoder.decode(hitTestResultData.isDownloadableMedia)
159 || !decoder.decode(hitTestResultData.lookupText)
160 || !decoder.decode(hitTestResultData.toolTipText)
161 || !decoder.decode(hitTestResultData.dictionaryPopupInfo))
162 return false;
163
164 WebKit::SharedMemory::IPCHandle imageHandle;
165 if (!decoder.decode(imageHandle))
166 return false;
167
168 hitTestResultData.imageSize = imageHandle.dataSize;
169 if (imageHandle.handle.isNull()) {
170 if (hitTestResultData.imageSize)
171 return false;
172 } else {
173 hitTestResultData.imageSharedMemory = WebKit::SharedMemory::map(imageHandle.handle, WebKit::SharedMemory::Protection::ReadOnly);
174 if (!hitTestResultData.imageSharedMemory)
175 return false;
176 if (!hitTestResultData.imageSize)
177 return false;
178 }
179
180 ShareableBitmap::Handle imageBitmapHandle;
181 if (!decoder.decode(imageBitmapHandle))
182 return false;
183
184 if (!imageBitmapHandle.isNull())
185 hitTestResultData.imageBitmap = ShareableBitmap::create(imageBitmapHandle, SharedMemory::Protection::ReadOnly);
186
187 bool hasLinkTextIndicator;
188 if (!decoder.decode(hasLinkTextIndicator))
189 return false;
190
191 if (hasLinkTextIndicator) {
192 std::optional<WebCore::TextIndicatorData> indicatorData;
193 decoder >> indicatorData;
194 if (!indicatorData)
195 return false;
196
197 hitTestResultData.linkTextIndicator = WebCore::TextIndicator::create(*indicatorData);
198 }
199
200 return platformDecode(decoder, hitTestResultData);
201}

214IntRect WebHitTestResultData::elementBoundingBoxInWindowCoordinates(const WebCore::HitTestResult& hitTestResult)
215{
216 Node* node = hitTestResult.innerNonSharedNode();
217 if (!node)
218 return IntRect();
219
220 Frame* frame = node->document().frame();
221 if (!frame)
222 return IntRect();
223
224 FrameView* view = frame->view();
225 if (!view)
226 return IntRect();
227
228 RenderObject* renderer = node->renderer();
229 if (!renderer)
230 return IntRect();
231
232 return view->contentsToWindow(renderer->absoluteBoundingBoxRect());
233}
