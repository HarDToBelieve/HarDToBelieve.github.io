151void WebKeyboardEvent::encode(IPC::Encoder& encoder) const
152{
153 WebEvent::encode(encoder);
154
155 encoder << m_text;
156 encoder << m_unmodifiedText;
157 encoder << m_key;
158 encoder << m_code;
159 encoder << m_keyIdentifier;
160 encoder << m_windowsVirtualKeyCode;
161 encoder << m_nativeVirtualKeyCode;
162 encoder << m_macCharCode;
163#if USE(APPKIT) || USE(UIKIT_KEYBOARD_ADDITIONS) || PLATFORM(GTK) || USE(LIBWPE)
164 encoder << m_handledByInputMethod;
165#endif
166#if PLATFORM(GTK) || USE(LIBWPE)
167 encoder << m_preeditUnderlines;
168 encoder << m_preeditSelectionRange;
169#endif
170#if USE(APPKIT) || PLATFORM(GTK)
171 encoder << m_commands;
172#endif
173 encoder << m_isAutoRepeat;
174 encoder << m_isKeypad;
175 encoder << m_isSystemKey;
176}

178bool WebKeyboardEvent::decode(IPC::Decoder& decoder, WebKeyboardEvent& result)
179{
180 if (!WebEvent::decode(decoder, result))
181 return false;
182
183 if (!decoder.decode(result.m_text))
184 return false;
185 if (!decoder.decode(result.m_unmodifiedText))
186 return false;
187 if (!decoder.decode(result.m_key))
188 return false;
189 if (!decoder.decode(result.m_code))
190 return false;
191 if (!decoder.decode(result.m_keyIdentifier))
192 return false;
193 if (!decoder.decode(result.m_windowsVirtualKeyCode))
194 return false;
195 if (!decoder.decode(result.m_nativeVirtualKeyCode))
196 return false;
197 if (!decoder.decode(result.m_macCharCode))
198 return false;
199#if USE(APPKIT) || USE(UIKIT_KEYBOARD_ADDITIONS) || PLATFORM(GTK) || USE(LIBWPE)
200 if (!decoder.decode(result.m_handledByInputMethod))
201 return false;
202#endif
203#if PLATFORM(GTK) || USE(LIBWPE)
204 if (!decoder.decode(result.m_preeditUnderlines))
205 return false;
206 if (!decoder.decode(result.m_preeditSelectionRange))
207 return false;
208#endif
209#if USE(APPKIT) || PLATFORM(GTK)
210 if (!decoder.decode(result.m_commands))
211 return false;
212#endif
213 if (!decoder.decode(result.m_isAutoRepeat))
214 return false;
215 if (!decoder.decode(result.m_isKeypad))
216 return false;
217 if (!decoder.decode(result.m_isSystemKey))
218 return false;
219
220 return true;
221}
222
223bool WebKeyboardEvent::isKeyboardEventType(Type type)
224{
225 return type == RawKeyDown || type == KeyDown || type == KeyUp || type == Char;
226}
