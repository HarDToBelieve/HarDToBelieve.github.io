65void WebMouseEvent::encode(IPC::Encoder& encoder) const
66{
67 WebEvent::encode(encoder);
68
69 encoder << m_button;
70 encoder << m_buttons;
71 encoder << m_position;
72 encoder << m_globalPosition;
73 encoder << m_deltaX;
74 encoder << m_deltaY;
75 encoder << m_deltaZ;
76 encoder << m_clickCount;
77#if PLATFORM(MAC)
78 encoder << m_eventNumber;
79 encoder << m_menuTypeForEvent;
80#endif
81 encoder << m_force;
82 encoder << m_syntheticClickType;
83 encoder << m_pointerId;
84 encoder << m_pointerType;
85 encoder << m_gestureWasCancelled;
86}

88bool WebMouseEvent::decode(IPC::Decoder& decoder, WebMouseEvent& result)
89{
90 if (!WebEvent::decode(decoder, result))
91 return false;
92
93 if (!decoder.decode(result.m_button))
94 return false;
95 if (!decoder.decode(result.m_buttons))
96 return false;
97 if (!decoder.decode(result.m_position))
98 return false;
99 if (!decoder.decode(result.m_globalPosition))
100 return false;
101 if (!decoder.decode(result.m_deltaX))
102 return false;
103 if (!decoder.decode(result.m_deltaY))
104 return false;
105 if (!decoder.decode(result.m_deltaZ))
106 return false;
107 if (!decoder.decode(result.m_clickCount))
108 return false;
109#if PLATFORM(MAC)
110 if (!decoder.decode(result.m_eventNumber))
111 return false;
112 if (!decoder.decode(result.m_menuTypeForEvent))
113 return false;
114#endif
115 if (!decoder.decode(result.m_force))
116 return false;
117
118 if (!decoder.decode(result.m_syntheticClickType))
119 return false;
120 if (!decoder.decode(result.m_pointerId))
121 return false;
122 if (!decoder.decode(result.m_pointerType))
123 return false;
124 if (!decoder.decode(result.m_gestureWasCancelled))
125 return false;
126
127 return true;
128}
129
130bool WebMouseEvent::isMouseEventType(Type type)
131{
132 return type == MouseDown || type == MouseUp || type == MouseMove || type == MouseForceUp || type == MouseForceDown || type == MouseForceChanged;
133}
