72#if PLATFORM(MAC) || PLATFORM(IOS_FAMILY)
73
74bool defaultDisallowSyncXHRDuringPageDismissalEnabled()
75{
76#if PLATFORM(MAC) || PLATFORM(MACCATALYST)
77 if (CFPreferencesGetAppBooleanValue(CFSTR("allowDeprecatedSynchronousXMLHttpRequestDuringUnload"), CFSTR("com.apple.WebKit"), nullptr)) {
78 WTFLogAlways("Allowing synchronous XHR during page unload due to managed preference");
79 return false;
80 }
81#elif PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST) && !PLATFORM(WATCHOS)
82 if (allowsDeprecatedSynchronousXMLHttpRequestDuringUnload()) {
83 WTFLogAlways("Allowing synchronous XHR during page unload due to managed preference");
84 return false;
85 }
86#endif
87 return true;
88}
89
90#endif

101static bool defaultAsyncFrameAndOverflowScrollingEnabled()
102{
103#if PLATFORM(IOS_FAMILY)
104 return true;
105#endif
106
107#if HAVE(SYSTEM_FEATURE_FLAGS)
108 return isFeatureFlagEnabled("async_frame_and_overflow_scrolling");
109#endif
110
111#if PLATFORM(MAC)
112 return true;
113#endif
114
115 return false;
116}

118bool defaultAsyncFrameScrollingEnabled()
119{
120#if USE(NICOSIA)
121 return true;
122#endif
123
124 return defaultAsyncFrameAndOverflowScrollingEnabled();
125}

132bool defaultOfflineWebApplicationCacheEnabled()
133{
134#if PLATFORM(COCOA)

136 return !newSDK;

140 return true;
