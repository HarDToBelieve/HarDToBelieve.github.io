26#pragma once
27
28#include "SandboxExtension.h"
29#include <WebCore/NetworkStorageSession.h>
30#include <pal/SessionID.h>
31#include <wtf/HashMap.h>
32
33namespace WebKit {
34
35struct WebProcessDataStoreParameters {
36 using TopFrameDomain = WebCore::RegistrableDomain;
37 using SubResourceDomain = WebCore::RegistrableDomain;
38
39 PAL::SessionID sessionID;
40 String applicationCacheDirectory;
41 SandboxExtension::Handle applicationCacheDirectoryExtensionHandle;
42 String applicationCacheFlatFileSubdirectoryName;
43 String mediaCacheDirectory;
44 SandboxExtension::Handle mediaCacheDirectoryExtensionHandle;
45 String mediaKeyStorageDirectory;
46 SandboxExtension::Handle mediaKeyStorageDirectoryExtensionHandle;
47 String javaScriptConfigurationDirectory;
48 SandboxExtension::Handle javaScriptConfigurationDirectoryExtensionHandle;
49#if ENABLE(RESOURCE_LOAD_STATISTICS)
50 WebCore::ThirdPartyCookieBlockingMode thirdPartyCookieBlockingMode { WebCore::ThirdPartyCookieBlockingMode::All };
51 HashSet<WebCore::RegistrableDomain> domainsWithUserInteraction;
52 HashMap<TopFrameDomain, SubResourceDomain> domainsWithStorageAccessQuirk;
53#endif
54#if HAVE(ARKIT_INLINE_PREVIEW)
55 String modelElementCacheDirectory;
56 SandboxExtension::Handle modelElementCacheDirectoryExtensionHandle;
57#endif
58 bool resourceLoadStatisticsEnabled { false };
59
60 template<class Encoder> void encode(Encoder&) const;
61 template<class Decoder> static std::optional<WebProcessDataStoreParameters> decode(Decoder&);
62};
63
64template<class Encoder>
65void WebProcessDataStoreParameters::encode(Encoder& encoder) const
66{
67 encoder << sessionID;
68 encoder << applicationCacheDirectory;
69 encoder << applicationCacheDirectoryExtensionHandle;
70 encoder << applicationCacheFlatFileSubdirectoryName;
71 encoder << mediaCacheDirectory;
72 encoder << mediaCacheDirectoryExtensionHandle;
73 encoder << mediaKeyStorageDirectory;
74 encoder << mediaKeyStorageDirectoryExtensionHandle;
75 encoder << javaScriptConfigurationDirectory;
76 encoder << javaScriptConfigurationDirectoryExtensionHandle;
77#if ENABLE(RESOURCE_LOAD_STATISTICS)
78 encoder << thirdPartyCookieBlockingMode;
79 encoder << domainsWithUserInteraction;
80 encoder << domainsWithStorageAccessQuirk;
81#endif
82#if HAVE(ARKIT_INLINE_PREVIEW)
83 encoder << modelElementCacheDirectory;
84 encoder << modelElementCacheDirectoryExtensionHandle;
85#endif
86 encoder << resourceLoadStatisticsEnabled;
87}
88
89template<class Decoder>
90std::optional<WebProcessDataStoreParameters> WebProcessDataStoreParameters::decode(Decoder& decoder)
91{
92 std::optional<PAL::SessionID> sessionID;
93 decoder >> sessionID;
94 if (!sessionID)
95 return std::nullopt;
96
97 String applicationCacheDirectory;
98 if (!decoder.decode(applicationCacheDirectory))
99 return std::nullopt;
100
101 std::optional<SandboxExtension::Handle> applicationCacheDirectoryExtensionHandle;
102 decoder >> applicationCacheDirectoryExtensionHandle;
103 if (!applicationCacheDirectoryExtensionHandle)
104 return std::nullopt;
105
106 String applicationCacheFlatFileSubdirectoryName;
107 if (!decoder.decode(applicationCacheFlatFileSubdirectoryName))
108 return std::nullopt;
109
110 String mediaCacheDirectory;
111 if (!decoder.decode(mediaCacheDirectory))
112 return std::nullopt;
113
114 std::optional<SandboxExtension::Handle> mediaCacheDirectoryExtensionHandle;
115 decoder >> mediaCacheDirectoryExtensionHandle;
116 if (!mediaCacheDirectoryExtensionHandle)
117 return std::nullopt;
118
119 String mediaKeyStorageDirectory;
120 if (!decoder.decode(mediaKeyStorageDirectory))
121 return std::nullopt;
122
123 std::optional<SandboxExtension::Handle> mediaKeyStorageDirectoryExtensionHandle;
124 decoder >> mediaKeyStorageDirectoryExtensionHandle;
125 if (!mediaKeyStorageDirectoryExtensionHandle)
126 return std::nullopt;
127
128 String javaScriptConfigurationDirectory;
129 if (!decoder.decode(javaScriptConfigurationDirectory))
130 return std::nullopt;
131
132 std::optional<SandboxExtension::Handle> javaScriptConfigurationDirectoryExtensionHandle;
133 decoder >> javaScriptConfigurationDirectoryExtensionHandle;
134 if (!javaScriptConfigurationDirectoryExtensionHandle)
135 return std::nullopt;
136
137#if ENABLE(RESOURCE_LOAD_STATISTICS)
138 std::optional<WebCore::ThirdPartyCookieBlockingMode> thirdPartyCookieBlockingMode;
139 decoder >> thirdPartyCookieBlockingMode;
140 if (!thirdPartyCookieBlockingMode)
141 return std::nullopt;
142
143 std::optional<HashSet<WebCore::RegistrableDomain>> domainsWithUserInteraction;
144 decoder >> domainsWithUserInteraction;
145 if (!domainsWithUserInteraction)
146 return std::nullopt;
147
148 std::optional<HashMap<TopFrameDomain, SubResourceDomain>> domainsWithStorageAccessQuirk;
149 decoder >> domainsWithStorageAccessQuirk;
150 if (!domainsWithStorageAccessQuirk)
151 return std::nullopt;
152#endif
153#if HAVE(ARKIT_INLINE_PREVIEW)
154 String modelElementCacheDirectory;
155 if (!decoder.decode(modelElementCacheDirectory))
156 return std::nullopt;
157
158 std::optional<SandboxExtension::Handle> modelElementCacheDirectoryExtensionHandle;
159 decoder >> modelElementCacheDirectoryExtensionHandle;
160 if (!modelElementCacheDirectoryExtensionHandle)
161 return std::nullopt;
162#endif
163
164 bool resourceLoadStatisticsEnabled = false;
165 if (!decoder.decode(resourceLoadStatisticsEnabled))
166 return std::nullopt;
167
168 return WebProcessDataStoreParameters {
169 WTFMove(*sessionID),
170 WTFMove(applicationCacheDirectory),
171 WTFMove(*applicationCacheDirectoryExtensionHandle),
172 WTFMove(applicationCacheFlatFileSubdirectoryName),
173 WTFMove(mediaCacheDirectory),
174 WTFMove(*mediaCacheDirectoryExtensionHandle),
175 WTFMove(mediaKeyStorageDirectory),
176 WTFMove(*mediaKeyStorageDirectoryExtensionHandle),
177 WTFMove(javaScriptConfigurationDirectory),
178 WTFMove(*javaScriptConfigurationDirectoryExtensionHandle),
179#if ENABLE(RESOURCE_LOAD_STATISTICS)
180 *thirdPartyCookieBlockingMode,
181 WTFMove(*domainsWithUserInteraction),
182 WTFMove(*domainsWithStorageAccessQuirk),
183#endif
184#if HAVE(ARKIT_INLINE_PREVIEW)
185 WTFMove(modelElementCacheDirectory),
186 WTFMove(*modelElementCacheDirectoryExtensionHandle),
187#endif
188 resourceLoadStatisticsEnabled
189 };
190}
