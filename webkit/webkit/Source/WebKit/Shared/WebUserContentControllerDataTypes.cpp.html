33void WebUserScriptData::encode(IPC::Encoder& encoder) const
34{
35 encoder << identifier;
36 encoder << worldIdentifier;
37 encoder << userScript;
38}
39
40std::optional<WebUserScriptData> WebUserScriptData::decode(IPC::Decoder& decoder)
41{
42 std::optional<uint64_t> identifier;
43 decoder >> identifier;
44 if (!identifier)
45 return std::nullopt;
46
47 std::optional<ContentWorldIdentifier> worldIdentifier;
48 decoder >> worldIdentifier;
49 if (!worldIdentifier)
50 return std::nullopt;
51
52 std::optional<WebCore::UserScript> userScript;
53 decoder >> userScript;
54 if (!userScript)
55 return std::nullopt;
56
57 return {{ WTFMove(*identifier), WTFMove(*worldIdentifier), WTFMove(*userScript) }};
58}

60void WebUserStyleSheetData::encode(IPC::Encoder& encoder) const
61{
62 encoder << identifier;
63 encoder << worldIdentifier;
64 encoder << userStyleSheet;
65}
66
67std::optional<WebUserStyleSheetData> WebUserStyleSheetData::decode(IPC::Decoder& decoder)
68{
69 std::optional<uint64_t> identifier;
70 decoder >> identifier;
71 if (!identifier)
72 return std::nullopt;
73
74 std::optional<ContentWorldIdentifier> worldIdentifier;
75 decoder >> worldIdentifier;
76 if (!worldIdentifier)
77 return std::nullopt;
78
79 WebCore::UserStyleSheet userStyleSheet;
80 if (!decoder.decode(userStyleSheet))
81 return std::nullopt;
82
83 return {{ WTFMove(*identifier), WTFMove(*worldIdentifier), WTFMove(userStyleSheet) }};
84}

111 return {{ WTFMove(*identifier), WTFMove(*worldIdentifier), WTFMove(*name) }};
