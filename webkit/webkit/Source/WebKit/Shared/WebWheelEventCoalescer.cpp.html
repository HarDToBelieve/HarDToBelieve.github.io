48bool WebWheelEventCoalescer::canCoalesce(const WebWheelEvent& a, const WebWheelEvent& b)
49{
50 if (a.position() != b.position())
51 return false;
52 if (a.globalPosition() != b.globalPosition())
53 return false;
54 if (a.modifiers() != b.modifiers())
55 return false;
56 if (a.granularity() != b.granularity())
57 return false;
58#if PLATFORM(COCOA)
59 if (a.phase() != b.phase())
60 return false;
61 if (a.momentumPhase() != b.momentumPhase())
62 return false;
63#endif
64#if PLATFORM(COCOA) || PLATFORM(GTK) || USE(LIBWPE)
65 if (a.hasPreciseScrollingDeltas() != b.hasPreciseScrollingDeltas())
66 return false;
67#endif
68
69 return true;
70}

72WebWheelEvent WebWheelEventCoalescer::coalesce(const WebWheelEvent& a, const WebWheelEvent& b)
73{
74 ASSERT(canCoalesce(a, b));
75
76 auto mergedDelta = a.delta() + b.delta();
77 auto mergedWheelTicks = a.wheelTicks() + b.wheelTicks();
78
79#if PLATFORM(COCOA)
80 auto mergedUnacceleratedScrollingDelta = a.unacceleratedScrollingDelta() + b.unacceleratedScrollingDelta();
81
82 return WebWheelEvent(WebEvent::Wheel, b.position(), b.globalPosition(), mergedDelta, mergedWheelTicks, b.granularity(), b.directionInvertedFromDevice(), b.phase(), b.momentumPhase(), b.hasPreciseScrollingDeltas(), b.scrollCount(), mergedUnacceleratedScrollingDelta, b.modifiers(), b.timestamp());
83#elif PLATFORM(GTK) || USE(LIBWPE)
84 return WebWheelEvent(WebEvent::Wheel, b.position(), b.globalPosition(), mergedDelta, mergedWheelTicks, b.phase(), b.momentumPhase(), b.granularity(), b.hasPreciseScrollingDeltas(), b.modifiers(), b.timestamp());
85#else
86 return WebWheelEvent(WebEvent::Wheel, b.position(), b.globalPosition(), mergedDelta, mergedWheelTicks, b.granularity(), b.modifiers(), b.timestamp());
87#endif
88}

90bool WebWheelEventCoalescer::isMomentumPhaseEvent(const WebWheelEvent& event)
91{
92 return event.phase() == WebWheelEvent::Phase::PhaseNone && event.momentumPhase() == WebWheelEvent::Phase::PhaseChanged;
93}

95bool WebWheelEventCoalescer::shouldDispatchEventNow(const WebWheelEvent& event) const
96{
97#if PLATFORM(GTK)
98 // Don't queue events representing a non-trivial scrolling phase to
99 // avoid having them trapped in the queue, potentially preventing a
100 // scrolling session to beginning or end correctly.
101 // This is only needed by platforms whose WebWheelEvent has this phase
102 // information (Cocoa and GTK+) but Cocoa was fine without it.
103 if (event.phase() == WebWheelEvent::Phase::PhaseNone
104 || event.phase() == WebWheelEvent::Phase::PhaseChanged
105 || event.momentumPhase() == WebWheelEvent::Phase::PhaseNone
106 || event.momentumPhase() == WebWheelEvent::Phase::PhaseChanged)
107 return true;
108#else
109 UNUSED_PARAM(event);
110#endif
111
112 return m_wheelEventQueue.size() >= wheelEventQueueSizeThreshold;
113}

115std::optional<WebWheelEvent> WebWheelEventCoalescer::nextEventToDispatch()
116{
117 if (m_wheelEventQueue.isEmpty())
118 return std::nullopt;
119
120 auto coalescedEvent = m_wheelEventQueue.takeFirst();
121
122 auto coalescedSequence = makeUnique<CoalescedEventSequence>();
123 coalescedSequence->append(coalescedEvent);
124
125 WebWheelEvent coalescedWebEvent = coalescedEvent;
126
127 while (!m_wheelEventQueue.isEmpty() && canCoalesce(coalescedWebEvent, m_wheelEventQueue.first())) {
128 auto firstEvent = m_wheelEventQueue.takeFirst();
129 coalescedSequence->append(firstEvent);
130 coalescedWebEvent = coalesce(coalescedWebEvent, firstEvent);
131 }
132
133#if !LOG_DISABLED
134 if (coalescedSequence->size() > 1)
135 LOG_WITH_STREAM(WheelEvents, stream << "WebWheelEventCoalescer::wheelEventWithCoalescing coalsesced " << *coalescedSequence << " into " << coalescedWebEvent);
136#endif
137
138 m_eventsBeingProcessed.append(WTFMove(coalescedSequence));
139 return coalescedWebEvent;
140}

142bool WebWheelEventCoalescer::shouldDispatchEvent(const NativeWebWheelEvent& event)
143{
144 LOG_WITH_STREAM(WheelEvents, stream << "WebWheelEventCoalescer::shouldDispatchEvent " << event << " (" << m_wheelEventQueue.size() << " events in the queue, " << m_eventsBeingProcessed.size() << " event sequences being processed, coalesce during decleration " << m_shouldCoalesceEventsDuringDeceleration << ")");
145
146 m_wheelEventQueue.append(event);
147
148 auto lastEventInterval = event.timestamp() - m_lastEventTime;
149 m_lastEventTime = event.timestamp();
150
151 if (isMomentumPhaseEvent(event) && shouldCoalesceEventsDuringDeceleration() && lastEventInterval) {
152 constexpr double momentumVelocityEventFrequencyReductionThreashold = 320.0; // Points per second.
153 auto instantaneousVelocity = std::max(std::abs(event.delta().width()), std::abs(event.delta().height())) / lastEventInterval.seconds();
154
155 constexpr auto maxCoalescingInterval = WebCore::FullSpeedAnimationInterval;
156 auto lastDispatchedEventInterval = event.timestamp() - m_lastDispatchedEventTime;
157
158 if (instantaneousVelocity < momentumVelocityEventFrequencyReductionThreashold && lastDispatchedEventInterval < maxCoalescingInterval) {
159 LOG_WITH_STREAM(WheelEvents, stream << " coalesced event that came within " << lastDispatchedEventInterval.milliseconds() << " of previous dispatch");
160 return false;
161 }
162 }
163
164 if (!m_eventsBeingProcessed.isEmpty()) {
165 if (!shouldDispatchEventNow(m_wheelEventQueue.last())) {
166 LOG_WITH_STREAM(WheelEvents, stream << "WebWheelEventCoalescer::shouldDispatchEvent - " << m_wheelEventQueue.size() << " events queued; not dispatching");
167 return false;
168 }
169 // The queue has too many wheel events, so push a new event.
170 // FIXME: This logic is confusing, and possibly not necessary.
171 }
172
173 m_lastDispatchedEventTime = event.timestamp();
174 return true;
175}

177NativeWebWheelEvent WebWheelEventCoalescer::takeOldestEventBeingProcessed()
178{
179 ASSERT(hasEventsBeingProcessed());
180 auto oldestSequence = m_eventsBeingProcessed.takeFirst();
181 return oldestSequence->last();
182}
