29#import <pal/spi/cocoa/NSURLConnectionSPI.h>
30#import <wtf/ProcessPrivilege.h>

32namespace WebKit {
33
34RetainPtr<CFHTTPCookieStorageRef> cookieStorageFromIdentifyingData(const Vector<uint8_t>& data)
35{
36 ASSERT(!data.isEmpty());
37 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
38
39 auto cookieStorageData = adoptCF(CFDataCreate(kCFAllocatorDefault, data.data(), data.size()));
40 auto cookieStorage = adoptCF(CFHTTPCookieStorageCreateFromIdentifyingData(kCFAllocatorDefault, cookieStorageData.get()));
41 ASSERT(cookieStorage);
42
43 CFHTTPCookieStorageScheduleWithRunLoop(cookieStorage.get(), [NSURLConnection resourceLoaderRunLoop], kCFRunLoopCommonModes);
44
45 return cookieStorage;
46}
47
48Vector<uint8_t> identifyingDataFromCookieStorage(CFHTTPCookieStorageRef cookieStorage)
49{
50 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
51
52 Vector<uint8_t> result;
53
54 auto cfData = adoptCF(CFHTTPCookieStorageCreateIdentifyingData(kCFAllocatorDefault, cookieStorage));
55 result.append(CFDataGetBytePtr(cfData.get()), CFDataGetLength(cfData.get()));
56
57 return result;
58}
