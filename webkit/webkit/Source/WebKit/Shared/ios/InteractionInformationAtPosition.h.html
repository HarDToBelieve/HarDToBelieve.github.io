44struct InteractionInformationAtPosition {

71#if ENABLE(DATALIST_ELEMENT)

74 bool shouldNotUseIBeamInEditableContent { false };

80 String title;

91 WebCore::FloatRect lineCaretExtent;
92
93 std::optional<WebCore::Cursor> cursor;

95 WebCore::TextIndicatorData linkIndicator;

97 String dataDetectorIdentifier;

105 // Copy compatible optional bits forward (for example, if we have a InteractionInformationAtPosition

108 void mergeCompatibleOptionalInformation(const InteractionInformationAtPosition& oldInformation);
109
110 void encode(IPC::Encoder&) const;
111 static WARN_UNUSED_RETURN bool decode(IPC::Decoder&, InteractionInformationAtPosition&);
