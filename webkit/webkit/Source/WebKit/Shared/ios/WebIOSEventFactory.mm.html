36UIKeyModifierFlags WebIOSEventFactory::toUIKeyModifierFlags(OptionSet<WebKit::WebEvent::Modifier> modifiers)
37{
38 UIKeyModifierFlags modifierFlags = 0;
39 if (modifiers.contains(WebKit::WebEvent::Modifier::ShiftKey))
40 modifierFlags |= UIKeyModifierShift;
41 if (modifiers.contains(WebKit::WebEvent::Modifier::ControlKey))
42 modifierFlags |= UIKeyModifierControl;
43 if (modifiers.contains(WebKit::WebEvent::Modifier::AltKey))
44 modifierFlags |= UIKeyModifierAlternate;
45 if (modifiers.contains(WebKit::WebEvent::Modifier::MetaKey))
46 modifierFlags |= UIKeyModifierCommand;
47 if (modifiers.contains(WebKit::WebEvent::Modifier::CapsLockKey))
48 modifierFlags |= UIKeyModifierAlphaShift;
49 return modifierFlags;
50}

52static OptionSet<WebKit::WebEvent::Modifier> modifiersForEvent(::WebEvent *event)
53{
54 OptionSet<WebKit::WebEvent::Modifier> modifiers;
55 WebEventFlags eventModifierFlags = event.modifierFlags;
56 if (eventModifierFlags & WebEventFlagMaskShiftKey)
57 modifiers.add(WebKit::WebEvent::Modifier::ShiftKey);
58 if (eventModifierFlags & WebEventFlagMaskControlKey)
59 modifiers.add(WebKit::WebEvent::Modifier::ControlKey);
60 if (eventModifierFlags & WebEventFlagMaskOptionKey)
61 modifiers.add(WebKit::WebEvent::Modifier::AltKey);
62 if (eventModifierFlags & WebEventFlagMaskCommandKey)
63 modifiers.add(WebKit::WebEvent::Modifier::MetaKey);
64 if (eventModifierFlags & WebEventFlagMaskLeftCapsLockKey)
65 modifiers.add(WebKit::WebEvent::Modifier::CapsLockKey);
66 return modifiers;
67}

69WebKit::WebKeyboardEvent WebIOSEventFactory::createWebKeyboardEvent(::WebEvent *event, bool handledByInputMethod)
70{
71 WebKit::WebEvent::Type type = (event.type == WebEventKeyUp) ? WebKit::WebEvent::KeyUp : WebKit::WebEvent::KeyDown;
72 String text;
73 String unmodifiedText;
74 bool autoRepeat;
75 if (event.keyboardFlags & WebEventKeyboardInputModifierFlagsChanged) {
76 text = emptyString();
77 unmodifiedText = emptyString();
78 autoRepeat = false;
79 } else {
80 text = event.characters;
81 unmodifiedText = event.charactersIgnoringModifiers;
82 autoRepeat = event.isKeyRepeating;
83 }
84 String key = WebCore::keyForKeyEvent(event);
85 String code = WebCore::codeForKeyEvent(event);
86 String keyIdentifier = WebCore::keyIdentifierForKeyEvent(event);
87 int windowsVirtualKeyCode = WebCore::windowsKeyCodeForKeyEvent(event);
88 // FIXME: This is not correct. WebEvent.keyCode represents the Windows native virtual key code.
89 int nativeVirtualKeyCode = event.keyCode;
90 int macCharCode = 0;
91 bool isKeypad = false;
92 bool isSystemKey = false;
93 auto modifiers = modifiersForEvent(event);
94 double timestamp = event.timestamp;
95
96 if (windowsVirtualKeyCode == '\r') {
97 text = "\r"_s;
98 unmodifiedText = text;
99 }
100
101 // The adjustments below are only needed in backward compatibility mode, but we cannot tell what mode we are in from here.
102
103 // Turn 0x7F into 8, because backspace needs to always be 8.
104 if (text == "\x7F")
105 text = "\x8"_s;
106 if (unmodifiedText == "\x7F")
107 unmodifiedText = "\x8"_s;
108 // Always use 9 for tab.
109 if (windowsVirtualKeyCode == 9) {
110 text = "\x9"_s;
111 unmodifiedText = text;
112 }
113
114 return WebKit::WebKeyboardEvent { type, text, unmodifiedText, key, code, keyIdentifier, windowsVirtualKeyCode, nativeVirtualKeyCode, macCharCode, handledByInputMethod, autoRepeat, isKeypad, isSystemKey, modifiers, WallTime::fromRawSeconds(timestamp) };
115}

117WebKit::WebMouseEvent WebIOSEventFactory::createWebMouseEvent(::WebEvent *event)
118{
119 // This currently only supports synthetic mouse moved events with no button pressed.
120 ASSERT_ARG(event, event.type == WebEventMouseMoved);
121
122 auto type = WebKit::WebEvent::MouseMove;
123 auto button = WebKit::WebMouseEvent::NoButton;
124 unsigned short buttons = 0;
125 auto position = WebCore::IntPoint(event.locationInWindow);
126 float deltaX = 0;
127 float deltaY = 0;
128 float deltaZ = 0;
129 int clickCount = 0;
130 double timestamp = event.timestamp;
131
132 return WebKit::WebMouseEvent(type, button, buttons, position, position, deltaX, deltaY, deltaZ, clickCount, OptionSet<WebKit::WebEvent::Modifier> { }, WallTime::fromRawSeconds(timestamp));
133}

136static WebKit::WebWheelEvent::Phase toWebPhase(UIScrollPhase phase)
137{
138 switch (phase) {
139 case UIScrollPhaseNone:
140 return WebKit::WebWheelEvent::PhaseNone;
141 case UIScrollPhaseMayBegin:
142 return WebKit::WebWheelEvent::PhaseMayBegin;
143 case UIScrollPhaseBegan:
144 return WebKit::WebWheelEvent::PhaseBegan;
145 case UIScrollPhaseChanged:
146 return WebKit::WebWheelEvent::PhaseChanged;
147 case UIScrollPhaseEnded:
148 return WebKit::WebWheelEvent::PhaseEnded;
149 case UIScrollPhaseCancelled:
150 return WebKit::WebWheelEvent::PhaseCancelled;
151 default:
152 ASSERT_NOT_REACHED();
153 return WebKit::WebWheelEvent::PhaseNone;
154 }
155}

157WebKit::WebWheelEvent WebIOSEventFactory::createWebWheelEvent(UIScrollEvent *event, UIView *contentView, std::optional<WebKit::WebWheelEvent::Phase> overridePhase)
158{
159 WebCore::IntPoint scrollLocation = WebCore::roundedIntPoint([event locationInView:contentView]);
160 CGVector deltaVector = [event _adjustedAcceleratedDeltaInView:contentView];
161 WebCore::FloatSize delta(deltaVector.dx, deltaVector.dy);
162 WebCore::FloatSize wheelTicks = delta;
163 wheelTicks.scale(1. / static_cast<float>(WebCore::Scrollbar::pixelsPerLineStep()));
164
165 return {
166 WebKit::WebEvent::Wheel,
167 scrollLocation,
168 scrollLocation,
169 delta,
170 wheelTicks,
171 WebKit::WebWheelEvent::Granularity::ScrollByPixelWheelEvent,
172 false,
173 overridePhase.value_or(toWebPhase(event.phase)),
174 WebKit::WebWheelEvent::PhaseNone,
175 true,
176 1,
177 delta,
178 { },
179 MonotonicTime::fromRawSeconds(event.timestamp).approximateWallTime()
180 };
181}
