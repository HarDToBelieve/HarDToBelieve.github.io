106constexpr unsigned guidSize = 36 + 1;
107constexpr unsigned versionSize = 31 + 1;
108
109struct CachedSandboxHeader {
110 uint32_t versionNumber;
111 uint32_t libsandboxVersion;
112 uint32_t headerSize;
113 uint32_t builtinSize; // If a builtin doesn't exist, this is UINT_MAX.
114 uint32_t dataSize;
115 char sandboxBuildID[guidSize];
116 char osVersion[versionSize];
117};
118// The file is layed out on disk like:
119// byte 0
120// CachedSandboxHeader <- sizeof(CachedSandboxHeader) bytes
121// SandboxHeader <- CachedSandboxHeader::headerSize bytes
122// [SandboxBuiltin] optional. Present if CachedSandboxHeader::builtinSize is not UINT_MAX. If present, builtinSize bytes (not including null termination).
123// SandboxData <- CachedSandboxHeader::dataSize bytes
124// byte N
125
126struct SandboxInfo {
127 SandboxInfo(const String& parentDirectoryPath, const String& directoryPath, const String& filePath, const SandboxParametersPtr& sandboxParameters, const CString& header, const WebCore::AuxiliaryProcessType& processType, const SandboxInitializationParameters& initializationParameters, const String& profileOrProfilePath, bool isProfilePath)
128 : parentDirectoryPath { parentDirectoryPath }
129 , directoryPath { directoryPath }
130 , filePath { filePath }
131 , sandboxParameters { sandboxParameters }
132 , header { header }
133 , processType { processType }
134 , initializationParameters { initializationParameters }
135 , profileOrProfilePath { profileOrProfilePath }
136 , isProfilePath { isProfilePath }
137 {
138 }
139
140 const String& parentDirectoryPath;
141 const String& directoryPath;
142 const String& filePath;
143 const SandboxParametersPtr& sandboxParameters;
144 const CString& header;
145 const WebCore::AuxiliaryProcessType& processType;
146 const SandboxInitializationParameters& initializationParameters;
147 const String& profileOrProfilePath;
148 const bool isProfilePath;
149};

165static OSStatus enableSandboxStyleFileQuarantine()
166{
167#if !PLATFORM(MACCATALYST)
168 qtn_proc_t quarantineProperties = qtn_proc_alloc();
169 auto quarantinePropertiesDeleter = makeScopeExit([quarantineProperties]() {
170 qtn_proc_free(quarantineProperties);
171 });
172
173
174 if (qtn_proc_init_with_self(quarantineProperties)) {
175 // See <rdar://problem/13463752>.
176 qtn_proc_init(quarantineProperties);
177 }
178
179 if (auto error = qtn_proc_set_flags(quarantineProperties, QTN_FLAG_SANDBOX))
180 return error;
181
182 // QTN_FLAG_SANDBOX is silently ignored if security.mac.qtn.sandbox_enforce sysctl is 0.
183 // In that case, quarantine falls back to advisory QTN_FLAG_DOWNLOAD.
184 return qtn_proc_apply_to_self(quarantineProperties);
185#else
186 return false;
187#endif
188}

199static std::optional<Vector<char>> fileContents(const String& path, bool shouldLock = false, OptionSet<FileSystem::FileLockMode> lockMode = FileSystem::FileLockMode::Exclusive)
200{
201 FileHandle file = shouldLock ? FileHandle(path, FileSystem::FileOpenMode::Read, lockMode) : FileHandle(path, FileSystem::FileOpenMode::Read);
202 file.open();
203 if (!file)
204 return std::nullopt;
205
206 char chunk[4096];
207 constexpr size_t chunkSize = WTF_ARRAY_LENGTH(chunk);
208 size_t contentSize = 0;
209 Vector<char> contents;
210 contents.reserveInitialCapacity(chunkSize);
211 while (size_t bytesRead = file.read(chunk, chunkSize)) {
212 contents.append(chunk, bytesRead);
213 contentSize += bytesRead;
214 }
215 contents.resize(contentSize);
216
217 return contents;
218}

244static std::optional<CString> setAndSerializeSandboxParameters(const SandboxInitializationParameters& initializationParameters, const SandboxParametersPtr& sandboxParameters, const String& profileOrProfilePath, bool isProfilePath)
245{
246 StringBuilder builder;
247 for (size_t i = 0; i < initializationParameters.count(); ++i) {
248 const char* name = initializationParameters.name(i);
249 const char* value = initializationParameters.value(i);
250 if (sandbox_set_param(sandboxParameters.get(), name, value)) {
251 WTFLogAlways("%s: Could not set sandbox parameter: %s\n", getprogname(), strerror(errno));
252 CRASH();
253 }
254 builder.append(name, ':', value, ':');
255 }
256 if (isProfilePath) {
257 auto contents = fileContents(profileOrProfilePath);
258 if (!contents)
259 return std::nullopt;
260 builder.appendCharacters(contents->data(), contents->size());
261 } else
262 builder.append(profileOrProfilePath);
263 return builder.toString().ascii();
264}

266static String sandboxDataVaultParentDirectory()
267{
268 char temp[PATH_MAX];
269 size_t length = confstr(_CS_DARWIN_USER_CACHE_DIR, temp, sizeof(temp));
270 if (!length) {
271 WTFLogAlways("%s: Could not retrieve user temporary directory path: %s\n", getprogname(), strerror(errno));
272 exit(EX_NOPERM);
273 }
274 RELEASE_ASSERT(length <= sizeof(temp));
275 char resolvedPath[PATH_MAX];
276 if (!realpath(temp, resolvedPath)) {
277 WTFLogAlways("%s: Could not canonicalize user temporary directory path: %s\n", getprogname(), strerror(errno));
278 exit(EX_NOPERM);
279 }
280 return resolvedPath;
281}

283static String sandboxDirectory(WebCore::AuxiliaryProcessType processType, const String& parentDirectory)
284{
285 StringBuilder directory;
286 directory.append(parentDirectory);
287 switch (processType) {
288 case WebCore::AuxiliaryProcessType::WebContent:
289 directory.append("/com.apple.WebKit.WebContent.Sandbox");
290 break;
291 case WebCore::AuxiliaryProcessType::Network:
292 directory.append("/com.apple.WebKit.Networking.Sandbox");
293 break;
294 case WebCore::AuxiliaryProcessType::Plugin:
295 directory.append("/com.apple.WebKit.Plugin.Sandbox");
296 break;
297#if ENABLE(GPU_PROCESS)
298 case WebCore::AuxiliaryProcessType::GPU:
299 directory.append("/com.apple.WebKit.GPU.Sandbox");
300 break;
301#endif
302#if ENABLE(WEB_AUTHN)
303 case WebCore::AuxiliaryProcessType::WebAuthn:
304 directory.append("/com.apple.WebKit.WebAuthn.Sandbox");
305 break;
306#endif
307 }
308
309#if !USE(APPLE_INTERNAL_SDK)
310 // Add .OpenSource suffix so that open source builds don't try to access a data vault used by system Safari.
311 directory.append(".OpenSource");
312#endif
313
314 return directory.toString();
315}

317static String sandboxFilePath(const String& directoryPath, const CString& header)
318{
319 // Make the filename semi-unique based on the contents of the header.
320
321 auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
322 crypto->addBytes(header.data(), header.length());
323 auto hash = crypto->computeHash();
324
325 return makeString(directoryPath, "/CompiledSandbox+", base64URLEncoded(hash.data(), hash.size()));
326}

328static bool ensureSandboxCacheDirectory(const SandboxInfo& info)
329{
330 if (FileSystem::fileTypeFollowingSymlinks(info.parentDirectoryPath) != FileSystem::FileType::Directory) {
331 FileSystem::makeAllDirectories(info.parentDirectoryPath);
332 if (FileSystem::fileTypeFollowingSymlinks(info.parentDirectoryPath) != FileSystem::FileType::Directory) {
333 WTFLogAlways("%s: Could not create sandbox directory\n", getprogname());
334 return false;
335 }
336 }
337
338#if USE(APPLE_INTERNAL_SDK)
339 const char* storageClass = processStorageClass(info.processType);
340 CString directoryPath = FileSystem::fileSystemRepresentation(info.directoryPath);
341 if (directoryPath.isNull())
342 return false;
343
344 auto makeDataVault = [&] {
345 do {
346 if (!rootless_mkdir_datavault(directoryPath.data(), 0700, storageClass))
347 return true;
348 } while (errno == EAGAIN);
349 return false;
350 };
351
352 if (makeDataVault())
353 return true;
354
355 if (errno == EEXIST) {
356 // The directory already exists. First we'll check if it is a data vault. If it is then
357 // we are the ones who created it and we can continue. If it is not a datavault then we'll just
358 // delete it and try to make a new one.
359 if (!rootless_check_datavault_flag(directoryPath.data(), storageClass))
360 return true;
361
362 if (FileSystem::fileType(info.directoryPath) == FileSystem::FileType::Directory) {
363 if (!FileSystem::deleteNonEmptyDirectory(info.directoryPath))
364 return false;
365 } else {
366 if (!FileSystem::deleteFile(info.directoryPath))
367 return false;
368 }
369
370 if (!makeDataVault())
371 return false;
372 } else {
373 WTFLogAlways("%s: Sandbox directory couldn't be created: ", getprogname(), strerror(errno));
374 return false;
375 }
376#else
377 bool hasSandboxDirectory = FileSystem::fileTypeFollowingSymlinks(info.directoryPath) == FileSystem::FileType::Directory;
378 if (!hasSandboxDirectory) {
379 if (FileSystem::makeAllDirectories(info.directoryPath)) {
380 ASSERT(FileSystem::fileTypeFollowingSymlinks(info.directoryPath) == FileSystem::FileType::Directory);
381 hasSandboxDirectory = true;
382 } else {
383 // We may have raced with someone else making it. That's ok.
384 hasSandboxDirectory = FileSystem::fileTypeFollowingSymlinks(info.directoryPath) == FileSystem::FileType::Directory;
385 }
386 }
387
388 if (!hasSandboxDirectory) {
389 // Bailing because we don't have a sandbox directory.
390 return false;
391 }
392#endif // USE(APPLE_INTERNAL_SDK)
393
394 return true;
395}

397static bool writeSandboxDataToCacheFile(const SandboxInfo& info, const Vector<char>& cacheFile)
398{
399 FileHandle file { info.filePath, FileSystem::FileOpenMode::Write, FileSystem::FileLockMode::Exclusive };
400 return file.write(cacheFile.data(), cacheFile.size()) == safeCast<int>(cacheFile.size());
401}

403static SandboxProfilePtr compileAndCacheSandboxProfile(const SandboxInfo& info)
404{
405 if (!ensureSandboxCacheDirectory(info))
406 return nullptr;
407
408 char* error = nullptr;
409 CString profileOrProfilePath = info.isProfilePath ? FileSystem::fileSystemRepresentation(info.profileOrProfilePath) : info.profileOrProfilePath.utf8();
410 if (profileOrProfilePath.isNull())
411 return nullptr;
412 SandboxProfilePtr sandboxProfile { info.isProfilePath ? sandbox_compile_file(profileOrProfilePath.data(), info.sandboxParameters.get(), &error) : sandbox_compile_string(profileOrProfilePath.data(), info.sandboxParameters.get(), &error) };
413 if (!sandboxProfile) {
414 WTFLogAlways("%s: Could not compile WebContent sandbox: %s\n", getprogname(), error);
415 return nullptr;
416 }
417
418 const bool haveBuiltin = sandboxProfile->builtin;
419 int32_t libsandboxVersion = NSVersionOfRunTimeLibrary("sandbox");
420 RELEASE_ASSERT(libsandboxVersion > 0);
421 String osVersion = systemMarketingVersion();
422
423 CachedSandboxHeader cachedHeader {
424 CachedSandboxVersionNumber,
425 static_cast<uint32_t>(libsandboxVersion),
426 safeCast<uint32_t>(info.header.length()),
427 haveBuiltin ? safeCast<uint32_t>(strlen(sandboxProfile->builtin)) : std::numeric_limits<uint32_t>::max(),
428 safeCast<uint32_t>(sandboxProfile->size),
429 { 0 },
430 { 0 }
431 };
432
433 size_t copied = strlcpy(cachedHeader.sandboxBuildID, SANDBOX_BUILD_ID, sizeof(cachedHeader.sandboxBuildID));
434 ASSERT(copied == guidSize - 1);
435 copied = strlcpy(cachedHeader.osVersion, osVersion.utf8().data(), sizeof(cachedHeader.osVersion));
436 ASSERT(copied < versionSize - 1);
437
438 const size_t expectedFileSize = sizeof(cachedHeader) + cachedHeader.headerSize + (haveBuiltin ? cachedHeader.builtinSize : 0) + cachedHeader.dataSize;
439
440 Vector<char> cacheFile;
441 cacheFile.reserveInitialCapacity(expectedFileSize);
442 cacheFile.append(bitwise_cast<uint8_t*>(&cachedHeader), sizeof(CachedSandboxHeader));
443 cacheFile.append(info.header.data(), info.header.length());
444 if (haveBuiltin)
445 cacheFile.append(sandboxProfile->builtin, cachedHeader.builtinSize);
446 cacheFile.append(sandboxProfile->data, cachedHeader.dataSize);
447
448 if (!writeSandboxDataToCacheFile(info, cacheFile))
449 WTFLogAlways("%s: Unable to cache compiled sandbox\n", getprogname());
450
451 return sandboxProfile;
452}

454static bool tryApplyCachedSandbox(const SandboxInfo& info)
455{
456#if USE(APPLE_INTERNAL_SDK)
457 CString directoryPath = FileSystem::fileSystemRepresentation(info.directoryPath);
458 if (directoryPath.isNull())
459 return false;
460 if (rootless_check_datavault_flag(directoryPath.data(), processStorageClass(info.processType)))
461 return false;
462#endif
463
464 auto contents = fileContents(info.filePath, true, FileSystem::FileLockMode::Shared);
465 if (!contents || contents->isEmpty())
466 return false;
467 Vector<char> cachedSandboxContents = WTFMove(*contents);
468 if (sizeof(CachedSandboxHeader) > cachedSandboxContents.size())
469 return false;
470
471 // This data may be corrupted if the sandbox file was cached on a different platform with different endianness
472 CachedSandboxHeader cachedSandboxHeader;
473 memcpy(&cachedSandboxHeader, cachedSandboxContents.data(), sizeof(CachedSandboxHeader));
474 int32_t libsandboxVersion = NSVersionOfRunTimeLibrary("sandbox");
475 RELEASE_ASSERT(libsandboxVersion > 0);
476 String osVersion = systemMarketingVersion();
477
478 if (cachedSandboxHeader.versionNumber != CachedSandboxVersionNumber)
479 return false;
480 if (static_cast<uint32_t>(libsandboxVersion) != cachedSandboxHeader.libsandboxVersion)
481 return false;
482 if (std::strcmp(cachedSandboxHeader.sandboxBuildID, SANDBOX_BUILD_ID))
483 return false;
484 if (cachedSandboxHeader.osVersion != osVersion)
485 return false;
486
487 const bool haveBuiltin = cachedSandboxHeader.builtinSize != std::numeric_limits<uint32_t>::max();
488
489 // These values are computed based on the disk layout specified below the definition of the CachedSandboxHeader struct
490 // and must be changed if the layout changes.
491 const char* sandboxHeaderPtr = bitwise_cast<char *>(cachedSandboxContents.data()) + sizeof(CachedSandboxHeader);
492 const char* sandboxBuiltinPtr = sandboxHeaderPtr + cachedSandboxHeader.headerSize;
493 unsigned char* sandboxDataPtr = bitwise_cast<unsigned char*>(haveBuiltin ? sandboxBuiltinPtr + cachedSandboxHeader.builtinSize : sandboxBuiltinPtr);
494
495 size_t expectedFileSize = sizeof(CachedSandboxHeader) + cachedSandboxHeader.headerSize + cachedSandboxHeader.dataSize;
496 if (haveBuiltin)
497 expectedFileSize += cachedSandboxHeader.builtinSize;
498 if (cachedSandboxContents.size() != expectedFileSize)
499 return false;
500 if (cachedSandboxHeader.headerSize != info.header.length())
501 return false;
502 if (memcmp(sandboxHeaderPtr, info.header.data(), info.header.length()))
503 return false;
504
505 SandboxProfile profile { };
506 CString builtin;
507 profile.builtin = nullptr;
508 profile.size = cachedSandboxHeader.dataSize;
509 if (haveBuiltin) {
510 builtin = CString::newUninitialized(cachedSandboxHeader.builtinSize, profile.builtin);
511 if (builtin.isNull())
512 return false;
513 memcpy(profile.builtin, sandboxBuiltinPtr, cachedSandboxHeader.builtinSize);
514 }
515 ASSERT(static_cast<void *>(sandboxDataPtr + profile.size) <= static_cast<void *>(cachedSandboxContents.data() + cachedSandboxContents.size()));
516 profile.data = sandboxDataPtr;
517
518 setNotifyOptions();
519
520 if (sandbox_apply(&profile)) {
521 WTFLogAlways("%s: Could not apply cached sandbox: %s\n", getprogname(), strerror(errno));
522 return false;
523 }
524
525 return true;
526}

535static void getSandboxProfileOrProfilePath(const SandboxInitializationParameters& parameters, String& profileOrProfilePath, bool& isProfilePath)
536{
537 switch (parameters.mode()) {
538 case SandboxInitializationParameters::ProfileSelectionMode::UseDefaultSandboxProfilePath:
539 profileOrProfilePath = [webKit2Bundle() pathForResource:[[NSBundle mainBundle] bundleIdentifier] ofType:@"sb"];
540 isProfilePath = true;
541 return;
542 case SandboxInitializationParameters::ProfileSelectionMode::UseOverrideSandboxProfilePath:
543 profileOrProfilePath = parameters.overrideSandboxProfilePath();
544 isProfilePath = true;
545 return;
546 case SandboxInitializationParameters::ProfileSelectionMode::UseSandboxProfile:
547 profileOrProfilePath = parameters.sandboxProfile();
548 isProfilePath = false;
549 return;
550 }
551}

553static bool compileAndApplySandboxSlowCase(const String& profileOrProfilePath, bool isProfilePath, const SandboxInitializationParameters& parameters)
554{
555 char* errorBuf;
556 CString temp = isProfilePath ? FileSystem::fileSystemRepresentation(profileOrProfilePath) : profileOrProfilePath.utf8();
557 uint64_t flags = isProfilePath ? SANDBOX_NAMED_EXTERNAL : 0;
558
559 setNotifyOptions();
560
561 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
562 if (sandbox_init_with_parameters(temp.data(), flags, parameters.namedParameterArray(), &errorBuf)) {
563 ALLOW_DEPRECATED_DECLARATIONS_END
564 WTFLogAlways("%s: Could not initialize sandbox profile [%s], error '%s'\n", getprogname(), temp.data(), errorBuf);
565 for (size_t i = 0, count = parameters.count(); i != count; ++i)
566 WTFLogAlways("%s=%s\n", parameters.name(i), parameters.value(i));
567 return false;
568 }
569 return true;
570}

572static bool applySandbox(const AuxiliaryProcessInitializationParameters& parameters, const SandboxInitializationParameters& sandboxInitializationParameters, const String& dataVaultParentDirectory)
573{
574 String profileOrProfilePath;
575 bool isProfilePath;
576 getSandboxProfileOrProfilePath(sandboxInitializationParameters, profileOrProfilePath, isProfilePath);
577 if (profileOrProfilePath.isEmpty()) {
578 WTFLogAlways("%s: Profile path is invalid\n", getprogname());
579 CRASH();
580 }
581
582#if USE(CACHE_COMPILED_SANDBOX)
583 // The plugin process's DARWIN_USER_TEMP_DIR and DARWIN_USER_CACHE_DIR sandbox parameters are randomized so
584 // so the compiled sandbox should not be cached because it won't be reused.
585 if (parameters.processType == WebCore::AuxiliaryProcessType::Plugin)
586 return compileAndApplySandboxSlowCase(profileOrProfilePath, isProfilePath, sandboxInitializationParameters);
587
588 SandboxParametersPtr sandboxParameters { sandbox_create_params() };
589 if (!sandboxParameters) {
590 WTFLogAlways("%s: Could not create sandbox parameters\n", getprogname());
591 CRASH();
592 }
593 auto header = setAndSerializeSandboxParameters(sandboxInitializationParameters, sandboxParameters, profileOrProfilePath, isProfilePath);
594 if (!header) {
595 WTFLogAlways("%s: Sandbox parameters are invalid\n", getprogname());
596 CRASH();
597 }
598
599 String directoryPath { sandboxDirectory(parameters.processType, dataVaultParentDirectory) };
600 String filePath = sandboxFilePath(directoryPath, *header);
601 SandboxInfo info {
602 dataVaultParentDirectory,
603 directoryPath,
604 filePath,
605 sandboxParameters,
606 *header,
607 parameters.processType,
608 sandboxInitializationParameters,
609 profileOrProfilePath,
610 isProfilePath
611 };
612
613 if (tryApplyCachedSandbox(info))
614 return true;
615
616 SandboxProfilePtr sandboxProfile = compileAndCacheSandboxProfile(info);
617 if (!sandboxProfile)
618 return compileAndApplySandboxSlowCase(profileOrProfilePath, isProfilePath, sandboxInitializationParameters);
619
620 setNotifyOptions();
621
622 if (sandbox_apply(sandboxProfile.get())) {
623 WTFLogAlways("%s: Could not apply compiled sandbox: %s\n", getprogname(), strerror(errno));
624 CRASH();
625 }
626
627 return true;
628#else
629 UNUSED_PARAM(parameters);
630 UNUSED_PARAM(dataVaultParentDirectory);
631 return compileAndApplySandboxSlowCase(profileOrProfilePath, isProfilePath, sandboxInitializationParameters);
632#endif // USE(CACHE_COMPILED_SANDBOX)
633}

635static void initializeSandboxParameters(const AuxiliaryProcessInitializationParameters& parameters, SandboxInitializationParameters& sandboxParameters)
636{
637 // Verify user directory suffix.
638 if (sandboxParameters.userDirectorySuffix().isNull()) {
639 auto userDirectorySuffix = parameters.extraInitializationData.find("user-directory-suffix");
640 if (userDirectorySuffix != parameters.extraInitializationData.end()) {
641 String suffix = userDirectorySuffix->value;
642 auto firstPathSeparator = suffix.find("/");
643 if (firstPathSeparator != notFound)
644 suffix.truncate(firstPathSeparator);
645 sandboxParameters.setUserDirectorySuffix(suffix);
646 } else {
647 String clientIdentifier = codeSigningIdentifier(parameters.connectionIdentifier.xpcConnection.get());
648 if (clientIdentifier.isNull())
649 clientIdentifier = parameters.clientIdentifier;
650 sandboxParameters.setUserDirectorySuffix(makeString([[NSBundle mainBundle] bundleIdentifier], '+', clientIdentifier));
651 }
652 }
653
654 String osSystemMarketingVersion = systemMarketingVersion();
655 Vector<String> osVersionParts = osSystemMarketingVersion.split('.');
656 if (osVersionParts.size() < 2) {
657 WTFLogAlways("%s: Couldn't find OS Version\n", getprogname());
658 exit(EX_NOPERM);
659 }
660 String osVersion = osVersionParts[0] + '.' + osVersionParts[1];
661 sandboxParameters.addParameter("_OS_VERSION", osVersion.utf8().data());
662
663 // Use private temporary and cache directories.
664 setenv("DIRHELPER_USER_DIR_SUFFIX", FileSystem::fileSystemRepresentation(sandboxParameters.userDirectorySuffix()).data(), 1);
665 char temporaryDirectory[PATH_MAX];
666 if (!confstr(_CS_DARWIN_USER_TEMP_DIR, temporaryDirectory, sizeof(temporaryDirectory))) {
667 WTFLogAlways("%s: couldn't retrieve private temporary directory path: %d\n", getprogname(), errno);
668 exit(EX_NOPERM);
669 }
670 setenv("TMPDIR", temporaryDirectory, 1);
671
672 sandboxParameters.addPathParameter("WEBKIT2_FRAMEWORK_DIR", [[webKit2Bundle() bundlePath] stringByDeletingLastPathComponent]);
673 sandboxParameters.addConfDirectoryParameter("DARWIN_USER_TEMP_DIR", _CS_DARWIN_USER_TEMP_DIR);
674 sandboxParameters.addConfDirectoryParameter("DARWIN_USER_CACHE_DIR", _CS_DARWIN_USER_CACHE_DIR);
675
676 char buffer[4096];
677 int bufferSize = sizeof(buffer);
678 struct passwd pwd;
679 struct passwd* result = 0;
680 if (getpwuid_r(getuid(), &pwd, buffer, bufferSize, &result) || !result) {
681 WTFLogAlways("%s: Couldn't find home directory\n", getprogname());
682 exit(EX_NOPERM);
683 }
684
685 sandboxParameters.addPathParameter("HOME_DIR", pwd.pw_dir);
686 String path = String::fromUTF8(pwd.pw_dir);
687 path.append("/Library");
688 sandboxParameters.addPathParameter("HOME_LIBRARY_DIR", FileSystem::fileSystemRepresentation(path).data());
689 path.append("/Preferences");
690 sandboxParameters.addPathParameter("HOME_LIBRARY_PREFERENCES_DIR", FileSystem::fileSystemRepresentation(path).data());
691
692#if CPU(X86_64)
693 sandboxParameters.addParameter("CPU", "x86_64");
694#elif CPU(ARM64)
695 sandboxParameters.addParameter("CPU", "arm64");
696#else
697#error "Unknown architecture."
698#endif
699 if (mbr_close_connectionsPtr())
700 mbr_close_connectionsPtr()();
701 if (lookup_close_connectionsPtr())
702 lookup_close_connectionsPtr()();
703}

705void AuxiliaryProcess::initializeSandbox(const AuxiliaryProcessInitializationParameters& parameters, SandboxInitializationParameters& sandboxParameters)
706{
707 TraceScope traceScope(InitializeSandboxStart, InitializeSandboxEnd);
708
709#if USE(CACHE_COMPILED_SANDBOX)
710 // This must be called before initializeSandboxParameters so that the path does not include the user directory suffix.
711 // We don't want the user directory suffix because we want all processes of the same type to use the same cache directory.
712 String dataVaultParentDirectory { sandboxDataVaultParentDirectory() };
713#else
714 String dataVaultParentDirectory;
715#endif
716
717 initializeSandboxParameters(parameters, sandboxParameters);
718
719 if (!applySandbox(parameters, sandboxParameters, dataVaultParentDirectory)) {
720 WTFLogAlways("%s: Unable to apply sandbox\n", getprogname());
721 CRASH();
722 }
723
724 if (shouldOverrideQuarantine()) {
725 // This will override LSFileQuarantineEnabled from Info.plist unless sandbox quarantine is globally disabled.
726 OSStatus error = enableSandboxStyleFileQuarantine();
727 if (error) {
728 WTFLogAlways("%s: Couldn't enable sandbox style file quarantine: %ld\n", getprogname(), static_cast<long>(error));
729 exit(EX_NOPERM);
730 }
731 }

755void AuxiliaryProcess::setQOS(int latencyQOS, int throughputQOS)
756{
757 if (!latencyQOS && !throughputQOS)
758 return;
759
760 struct task_qos_policy qosinfo = {
761 latencyQOS ? LATENCY_QOS_TIER_0 + latencyQOS - 1 : LATENCY_QOS_TIER_UNSPECIFIED,

763 };
764
765 task_policy_set(mach_task_self(), TASK_OVERRIDE_QOS_POLICY, (task_policy_t)&qosinfo, TASK_QOS_POLICY_COUNT);
766}
