61class AbortAction {
62public:
63 AbortAction(Condition& condition)
64 : m_condition(condition)
65 {
66 if (noErr != PAL::FigThreadRegisterAbortAction(action, this, &m_token))
67 m_token = nullptr;
68 }
69
70 ~AbortAction()
71 {
72 if (m_token)
73 PAL::FigThreadUnregisterAbortAction(m_token);
74 }
75
76 bool aborted() const { return m_aborted; }
77
78private:
79 static void action(void* refcon)
80 {
81 auto thisPtr = static_cast<AbortAction*>(refcon);
82 thisPtr->m_aborted = true;
83 thisPtr->m_condition.notifyAll();
84 }
85
86 Condition& m_condition;
87 FigThreadAbortActionToken m_token { nullptr };
88 bool m_aborted { false };
89};

125void MediaFormatReader::parseByteSource(RetainPtr<MTPluginByteSourceRef>&& byteSource)
126{
127 ASSERT(isMainRunLoop());
128
129 static NeverDestroyed<ContentType> contentType("video/webm"_s);
130 auto parser = SourceBufferParserWebM::create(contentType);
131 if (!parser) {
132 Locker locker { m_parseTracksLock };
133 m_parseTracksStatus = kMTPluginFormatReaderError_AllocationFailure;
134 return;
135 }
136
137 if (!m_logger) {
138 m_logger = makeRefPtr(Document::sharedLogger());
139 m_logIdentifier = nextLogIdentifier();
140 }
141
142 ALWAYS_LOG_IF_POSSIBLE(LOGIDENTIFIER);
143 parser->setLogger(*m_logger, m_logIdentifier);
144
145 // Set a minimum audio sample duration of 0 so the parser creates indivisible samples with byte source ranges.
146 parser->setMinimumAudioSampleDuration(0);
147
148 parser->setCallOnClientThreadCallback([](Function<void()>&& function) {
149 MediaTrackReader::storageQueue().dispatch(WTFMove(function));
150 });
151
152 parser->setDidParseInitializationDataCallback([this, protectedThis = makeRef(*this)](SourceBufferParser::InitializationSegment&& initializationSegment) {
153 didParseTracks(WTFMove(initializationSegment), noErr);
154 });
155
156 parser->setDidEncounterErrorDuringParsingCallback([this, protectedThis = makeRef(*this)](uint64_t errorCode) {
157 didParseTracks({ }, errorCode);
158 });
159
160 parser->setDidProvideMediaDataCallback([this, protectedThis = makeRef(*this)](Ref<MediaSample>&& mediaSample, uint64_t trackID, const String& mediaType) {
161 didProvideMediaData(WTFMove(mediaSample), trackID, mediaType);
162 });
163
164 Locker locker { m_parseTracksLock };
165 m_byteSource = WTFMove(byteSource);
166 m_parseTracksStatus = std::nullopt;
167 m_duration = MediaTime::invalidTime();
168 m_trackReaders.clear();
169
170 readerQueue().dispatch([this, protectedThis = makeRef(*this), byteSource = m_byteSource, parser = parser.releaseNonNull()]() mutable {
171 parser->appendData(WTFMove(byteSource));
172 MediaTrackReader::storageQueue().dispatch([this, protectedThis = makeRef(*this), parser = WTFMove(parser)]() mutable {
173 finishParsing(WTFMove(parser));
174 });
175 });
176}

178void MediaFormatReader::didParseTracks(SourceBufferPrivateClient::InitializationSegment&& segment, uint64_t errorCode)
179{
180 ASSERT(!isMainRunLoop());
181
182 Locker locker { m_parseTracksLock };
183 ASSERT(!m_parseTracksStatus);
184 ASSERT(m_duration.isInvalid());
185 ASSERT(m_trackReaders.isEmpty());
186
187 ALWAYS_LOG_IF_POSSIBLE(LOGIDENTIFIER);
188 if (errorCode)
189 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, errorCode);
190
191 m_parseTracksStatus = errorCode ? kMTPluginFormatReaderError_ParsingFailure : noErr;
192 m_duration = WTFMove(segment.duration);
193
194 for (auto& videoTrack : segment.videoTracks) {
195 auto track = videoTrack.track.get();
196 auto trackReader = MediaTrackReader::create(allocator(), *this, kCMMediaType_Video, *track->trackUID(), track->defaultEnabled());
197 if (!trackReader)
198 continue;
199
200 m_trackReaders.append(trackReader.releaseNonNull());
201 }
202
203 for (auto& audioTrack : segment.audioTracks) {
204 auto track = audioTrack.track.get();
205 auto trackReader = MediaTrackReader::create(allocator(), *this, kCMMediaType_Audio, *track->trackUID(), track->defaultEnabled());
206 if (!trackReader)
207 continue;
208
209 m_trackReaders.append(trackReader.releaseNonNull());
210 }
211
212 for (auto& textTrack : segment.textTracks) {
213 auto track = textTrack.track.get();
214 if (auto trackReader = MediaTrackReader::create(allocator(), *this, kCMMediaType_Text, *track->trackUID(), track->defaultEnabled()))
215 m_trackReaders.append(trackReader.releaseNonNull());
216 }
217
218 m_parseTracksCondition.notifyAll();
219}

221void MediaFormatReader::didProvideMediaData(Ref<MediaSample>&& mediaSample, uint64_t trackID, const String&)
222{
223 ASSERT(!isMainRunLoop());
224
225 Locker locker { m_parseTracksLock };
226 auto trackIndex = m_trackReaders.findMatching([&](auto& track) {
227 return track->trackID() == trackID;
228 });
229
230 if (trackIndex != notFound)
231 m_trackReaders[trackIndex]->addSample(WTFMove(mediaSample), m_byteSource.get());
232}

234void MediaFormatReader::finishParsing(Ref<SourceBufferParser>&& parser)
235{
236 ASSERT(!isMainRunLoop());
237 ALWAYS_LOG_IF_POSSIBLE(LOGIDENTIFIER);
238
239 Locker locker { m_parseTracksLock };
240 ASSERT(m_parseTracksStatus.has_value());
241
242 for (auto& trackReader : m_trackReaders)
243 trackReader->finishParsing();
244
245 if (m_duration.isIndefinite()) {
246 MediaTime greatestPresentationTime { MediaTime::invalidTime() };
247 for (auto& trackReader : m_trackReaders) {
248 if (greatestPresentationTime.isInvalid() || trackReader->greatestPresentationTime() > greatestPresentationTime)
249 greatestPresentationTime = trackReader->greatestPresentationTime();
250 }
251 if (greatestPresentationTime.isValid())
252 m_duration = greatestPresentationTime;
253 }
254
255 parser->setDidParseInitializationDataCallback(nullptr);
256 parser->setDidEncounterErrorDuringParsingCallback(nullptr);
257 parser->setDidProvideMediaDataCallback(nullptr);
258 parser->resetParserState();
259}

261OSStatus MediaFormatReader::copyProperty(CFStringRef key, CFAllocatorRef allocator, void* valueCopy)
262{
263 if (!CFEqual(key, PAL::get_MediaToolbox_kMTPluginFormatReaderProperty_Duration())) {
264 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "asked for unsupported property ", String(key));
265 return kCMBaseObjectError_ValueNotAvailable;
266 }
267
268 AbortAction action { m_parseTracksCondition };
269
270 Locker locker { m_parseTracksLock };
271 m_parseTracksCondition.wait(m_parseTracksLock, [&] {
272 assertIsHeld(m_parseTracksLock);
273 return m_parseTracksStatus.has_value() || action.aborted();
274 });
275
276 if (m_duration.isIndefinite())
277 return kCMBaseObjectError_ValueNotAvailable;
278
279 if (auto leakedDuration = adoptCF(PAL::CMTimeCopyAsDictionary(PAL::toCMTime(m_duration), allocator)).leakRef()) {
280 *reinterpret_cast<CFDictionaryRef*>(valueCopy) = leakedDuration;
281 return noErr;
282 }
283
284 ERROR_LOG_IF_POSSIBLE(LOGIDENTIFIER, "Could not retrieve value for property ", String(key));
285 return kCMBaseObjectError_ValueNotAvailable;
286}

288OSStatus MediaFormatReader::copyTrackArray(CFArrayRef* trackArrayCopy)
289{
290 AbortAction action { m_parseTracksCondition };
291
292 Locker locker { m_parseTracksLock };
293 m_parseTracksCondition.wait(m_parseTracksLock, [&] {
294 assertIsHeld(m_parseTracksLock);
295 return m_parseTracksStatus.has_value() || action.aborted();
296 });
297
298 if (*m_parseTracksStatus != noErr)
299 return *m_parseTracksStatus;
300
301 auto mutableArray = adoptCF(CFArrayCreateMutable(allocator(), Checked<CFIndex>(m_trackReaders.size()), &kCFTypeArrayCallBacks));
302 for (auto& trackReader : m_trackReaders)
303 CFArrayAppendValue(mutableArray.get(), trackReader->wrapper());
304
305 *trackArrayCopy = adoptCF(CFArrayCreateCopy(allocator(), mutableArray.get())).leakRef();
306 return noErr;
307}
