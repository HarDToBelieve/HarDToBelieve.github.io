28#if ENABLE(WEBM_FORMAT_READER)

30#include <WebCore/MediaSampleAVFObjC.h>
31#include <pal/spi/cocoa/MediaToolboxSPI.h>
32#include <wtf/MediaTime.h>
33#include <wtf/RetainPtr.h>
34
35namespace WebKit {
36
37class MediaSampleByteRange final : public WebCore::MediaSampleAVFObjC {
38public:
39 static Ref<MediaSampleByteRange> create(WebCore::MediaSample& sample, MTPluginByteSourceRef byteSource, uint64_t trackID)
40 {
41 return adoptRef(*new MediaSampleByteRange(sample, byteSource, trackID));
42 }
43
44 MediaTime presentationTime() const final { return m_presentationTime; }
45 MediaTime decodeTime() const final { return m_decodeTime; }
46 MediaTime duration() const final { return m_duration; }
47 size_t sizeInBytes() const final { return m_sizeInBytes; }
48 WebCore::FloatSize presentationSize() const final { return m_presentationSize; }
49 SampleFlags flags() const final { return m_flags; }
50 std::optional<ByteRange> byteRange() const final { return m_byteRange; }
51
52 AtomString trackID() const final;
53 WebCore::PlatformSample platformSample() final;
54 void offsetTimestampsBy(const MediaTime&) final;
55 void setTimestamps(const MediaTime&, const MediaTime&) final;
56
57private:
58 MediaSampleByteRange(MediaSample&, MTPluginByteSourceRef, uint64_t trackID);
59
60 MediaTime m_presentationTime;
61 MediaTime m_decodeTime;
62 MediaTime m_duration;
63 ByteRange m_byteRange;
64 uint64_t m_trackID;
65 size_t m_sizeInBytes;
66 WebCore::FloatSize m_presentationSize;
67 RetainPtr<MTPluginByteSourceRef> m_byteSource;
68 RetainPtr<CMFormatDescriptionRef> m_formatDescription;
69 SampleFlags m_flags;
70};
71
72} // namespace WebKit

74#endif // ENABLE(WEBM_FORMAT_READER)
