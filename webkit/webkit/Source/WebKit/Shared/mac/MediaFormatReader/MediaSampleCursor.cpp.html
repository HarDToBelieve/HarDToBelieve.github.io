48static MediaTime assumedDecodeTime(const MediaTime& presentationTime, DecodeOrderSampleMap& map)
49{
50 MediaSample& firstSample = *map.begin()->second;
51 return firstSample.decodeTime() == firstSample.presentationTime() ? presentationTime : MediaTime::invalidTime();
52}

68static DecodeOrderSampleMap::iterator upperBound(DecodeOrderSampleMap& map, const MediaTime& time)
69{
70 auto decodeTime = assumedDecodeTime(time, map);
71 return map.findSampleAfterDecodeKey(std::make_pair(decodeTime, time));
72}

74static PresentationOrderSampleMap::iterator upperBound(PresentationOrderSampleMap& map, const MediaTime& time)
75{
76 return map.findSampleStartingAfterPresentationTime(time);
77}

79template<typename OrderedMap>
80static int64_t stepIterator(int64_t stepsRemaining, typename OrderedMap::iterator& iterator, OrderedMap& samples)
81{
82 ASSERT(iterator != samples.end());
83 if (stepsRemaining < 0)
84 for (; stepsRemaining && iterator != samples.begin(); ++stepsRemaining, --iterator) { }
85 else
86 for (auto next = std::next(iterator); stepsRemaining && next != samples.end(); --stepsRemaining, iterator = next++) { }
87 return stepsRemaining;
88}
89
90template<typename OrderedMap>
91static bool stepTime(const MediaTime& delta, MediaTime& time, OrderedMap& samples, bool hasAllSamples, const MediaTime& trackDuration)
92{
93 time += delta;
94 MediaSample& lastSample = *samples.rbegin()->second;
95 auto firstTime = samples.begin()->second->presentationTime();
96 auto lastTime = hasAllSamples ? lastSample.presentationTime() + lastSample.duration() : trackDuration;
97 return time < firstTime || time >= lastTime;
98}

139template<typename OrderedMap>
140std::optional<typename OrderedMap::iterator> MediaSampleCursor::locateIterator(OrderedMap& samples, bool hasAllSamples) const
141{
142 ASSERT(m_locatorLock.isLocked());
143 using Iterator = typename OrderedMap::iterator;
144 return WTF::switchOn(m_locator,
145 [&](const MediaTime& presentationTime) -> std::optional<Iterator> {
146 assertIsHeld(m_locatorLock);
147 auto iterator = upperBound(samples, presentationTime);
148 if (iterator == samples.begin())
149 m_locator = WTFMove(iterator);
150 else if (hasAllSamples || iterator != samples.end())
151 m_locator = std::prev(iterator);
152 else
153 return std::nullopt;
154 return locateIterator(samples, hasAllSamples);
155 },
156 [&](const auto& otherIterator) {
157 assertIsHeld(m_locatorLock);
158 m_locator = otherIterator->second->presentationTime();
159 return locateIterator(samples, hasAllSamples);
160 },
161 [&](const Iterator& iterator) {
162 return iterator;
163 }
164 );
165}

167MediaSample* MediaSampleCursor::locateMediaSample(SampleMap& samples, bool hasAllSamples) const
168{
169 ASSERT(m_locatorLock.isLocked());
170 return WTF::switchOn(m_locator,
171 [&](const MediaTime&) -> MediaSample* {
172 assertIsHeld(m_locatorLock);
173 auto iterator = locateIterator(samples.presentationOrder(), hasAllSamples);
174 if (!iterator)
175 return nullptr;
176 return locateMediaSample(samples, hasAllSamples);
177 },
178 [&](const auto& iterator) {
179 return iterator->second.get();
180 }
181 );
182}

184MediaSampleCursor::Timing MediaSampleCursor::locateTiming(SampleMap& samples, bool hasAllSamples) const
185{
186 ASSERT(m_locatorLock.isLocked());
187 return WTF::switchOn(m_locator,
188 [&](const MediaTime& presentationTime) {
189 assertIsHeld(m_locatorLock);
190 if (locateMediaSample(samples, hasAllSamples))
191 return locateTiming(samples, hasAllSamples);
192 auto& clampedPresentationTime = std::min(presentationTime, m_trackReader->duration());
193 return Timing {
194 assumedDecodeTime(clampedPresentationTime, samples.decodeOrder()),
195 clampedPresentationTime,
196 MediaTime::zeroTime(),
197 };
198 },
199 [&](const auto& iterator) {
200 return Timing {
201 iterator->second->decodeTime(),
202 iterator->second->presentationTime(),
203 iterator->second->duration(),
204 };
205 }
206 );
207}

209template<typename OrderedMap>
210OSStatus MediaSampleCursor::stepInOrderedMap(int64_t stepsToTake, int64_t& stepsTaken)
211{
212 return getSampleMap([&](SampleMap& samples, bool hasAllSamples) -> OSStatus {
213 assertIsHeld(m_locatorLock);
214 auto& orderedMap = orderedSamples<OrderedMap>(samples);
215 if (auto iterator = locateIterator(orderedMap, hasAllSamples)) {
216 auto stepsRemaining = stepIterator(stepsToTake, *iterator, orderedMap);
217 m_locator = WTFMove(*iterator);
218 stepsTaken = stepsToTake - stepsRemaining;
219 return noErr;
220 }
221 return kMTPluginSampleCursorError_LocationNotAvailable;
222 });
223}
224
225OSStatus MediaSampleCursor::stepInPresentationTime(const MediaTime& delta, Boolean& wasPinned)
226{
227 return getSampleMap([&](SampleMap& samples, bool hasAllSamples) -> OSStatus {
228 assertIsHeld(m_locatorLock);
229 auto timing = locateTiming(samples, hasAllSamples);
230 wasPinned = stepTime(delta, timing.presentationTime, samples.presentationOrder(), hasAllSamples, m_trackReader->duration());
231 m_locator = timing.presentationTime;
232 return noErr;
233 });
234}

236template<typename Function>
237OSStatus MediaSampleCursor::getSampleMap(Function&& function) const
238{
239 OSStatus status = noErr;
240 Locker locker { m_locatorLock };
241 m_trackReader->waitForSample([&](SampleMap& samples, bool hasAllSamples) {
242 if (!samples.size())
243 ERROR_LOG(LOGIDENTIFIER, "track ", m_trackReader->trackID(), " finished parsing with no samples.");
244 status = samples.size() ? function(samples, hasAllSamples) : kMTPluginSampleCursorError_NoSamples;
245 return true;
246 });
247 return status;
248}

250template<typename Function>
251OSStatus MediaSampleCursor::getMediaSample(Function&& function) const
252{
253 return getSampleMap([&](SampleMap& samples, bool hasAllSamples) -> OSStatus {
254 assertIsHeld(m_locatorLock);
255 auto sample = locateMediaSample(samples, hasAllSamples);
256 if (!sample)
257 return kMTPluginSampleCursorError_LocationNotAvailable;
258 DEBUG_LOG(LOGIDENTIFIER, "sample: ", *sample);
259 function(*sample);
260 return noErr;
261 });
262}

264template<typename Function>
265OSStatus MediaSampleCursor::getTiming(Function&& function) const
266{
267 return getSampleMap([&](SampleMap& samples, bool hasAllSamples) {

269 auto timing = locateTiming(samples, hasAllSamples);
270 DEBUG_LOG(LOGIDENTIFIER, "decodeTime: ", timing.decodeTime, ", presentationTime: ", timing.presentationTime, ", duration: ", timing.duration);
271 function(timing);
272 return noErr;
273 });

308CFComparisonResult MediaSampleCursor::compareInDecodeOrder(MTPluginSampleCursorRef otherCursorRef) const
309{
310 MediaSampleCursor* otherCursor = unwrap(otherCursorRef);
311 if (!otherCursor || m_trackReader.ptr() != otherCursor->m_trackReader.ptr())
312 return kCFCompareLessThan;
313
314 if (this == otherCursor) {
315 RELEASE_ASSERT(wrapper() == otherCursorRef);
316 return kCFCompareEqualTo;
317 }
318
319 auto presentationTime = MediaTime::invalidTime();
320 getTiming([&](const Timing& timing) {
321 presentationTime = timing.presentationTime;
322 });
323
324 auto otherPresentationTime = MediaTime::invalidTime();
325 otherCursor->getTiming([&](const Timing& timing) {
326 otherPresentationTime = timing.presentationTime;
327 });
328
329 switch (presentationTime.compare(otherPresentationTime)) {
330 case MediaTime::LessThan:
331 return kCFCompareLessThan;
332 case MediaTime::EqualTo:
333 return kCFCompareEqualTo;
334 case MediaTime::GreaterThan:
335 return kCFCompareGreaterThan;
336 }
337}

350OSStatus MediaSampleCursor::getSyncInfo(MTPluginSampleCursorSyncInfo* syncInfo) const
351{
352 OSStatus syncInfoStatus = noErr;
353 auto getSampleStatus = getMediaSample([&](MediaSample& sample) {
354 if (sample.hasSyncInfo()) {
355 *syncInfo = {
356 .fullSync = sample.isSync()
357 };
358 return;
359 }
360 syncInfoStatus = kCMBaseObjectError_ValueNotAvailable;
361 });
362 if (syncInfoStatus != noErr)
363 return syncInfoStatus;
364 return getSampleStatus;
365}

388OSStatus MediaSampleCursor::getPlayableHorizon(CMTime* playableHorizon) const
389{
390 return getSampleMap([&](SampleMap& samples, bool hasAllSamples) {
391 assertIsHeld(m_locatorLock);
392 MediaSample& lastSample = *samples.decodeOrder().rbegin()->second;
393 auto timing = locateTiming(samples, hasAllSamples);
394 *playableHorizon = PAL::toCMTime(lastSample.presentationTime() + lastSample.duration() - timing.presentationTime);
395 return noErr;
396 });
