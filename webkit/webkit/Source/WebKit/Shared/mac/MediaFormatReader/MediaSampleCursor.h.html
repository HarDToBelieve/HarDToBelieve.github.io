116constexpr MediaSampleCursor::WrapperClass MediaSampleCursor::wrapperClass()
117{
118 return {
119#if HAVE(MT_PLUGIN_SAMPLE_CURSOR_PLAYABLE_HORIZON)
120 .version = kMTPluginSampleCursor_ClassVersion_4,
121#else
122 .version = kMTPluginSampleCursor_ClassVersion_3,
123#endif
124 .copy = [](MTPluginSampleCursorRef sampleCursor, MTPluginSampleCursorRef* sampleCursorCopy) {
125 return unwrap(sampleCursor)->copy(sampleCursorCopy);
126 },
127 .stepInDecodeOrderAndReportStepsTaken = [](MTPluginSampleCursorRef sampleCursor, int64_t stepCount, int64_t* stepsTaken) {
128 return unwrap(sampleCursor)->stepInDecodeOrderAndReportStepsTaken(stepCount, stepsTaken);
129 },
130 .stepInPresentationOrderAndReportStepsTaken = [](MTPluginSampleCursorRef sampleCursor, int64_t stepCount, int64_t* stepsTaken) {
131 return unwrap(sampleCursor)->stepInPresentationOrderAndReportStepsTaken(stepCount, stepsTaken);
132 },
133 .stepByDecodeTime = [](MTPluginSampleCursorRef sampleCursor, CMTime deltaDecodeTime, Boolean* positionWasPinned) {
134 return unwrap(sampleCursor)->stepByDecodeTime(deltaDecodeTime, positionWasPinned);
135 },
136 .stepByPresentationTime = [](MTPluginSampleCursorRef sampleCursor, CMTime deltaPresentationTime, Boolean* positionWasPinned) {
137 return unwrap(sampleCursor)->stepByPresentationTime(deltaPresentationTime, positionWasPinned);
138 },
139 .compareInDecodeOrder = [](MTPluginSampleCursorRef sampleCursor, MTPluginSampleCursorRef otherSampleCursor) {
140 return unwrap(sampleCursor)->compareInDecodeOrder(otherSampleCursor);
141 },
142 .getSampleTiming = [](MTPluginSampleCursorRef sampleCursor, CMSampleTimingInfo* sampleTimingInfo) {
143 return unwrap(sampleCursor)->getSampleTiming(sampleTimingInfo);
144 },
145 .getSyncInfo = [](MTPluginSampleCursorRef sampleCursor, MTPluginSampleCursorSyncInfo* syncInfo) -> OSStatus {
146 return unwrap(sampleCursor)->getSyncInfo(syncInfo);
147 },
148 .copyFormatDescription = [](MTPluginSampleCursorRef sampleCursor, CMFormatDescriptionRef* formatDescription) {
149 return unwrap(sampleCursor)->copyFormatDescription(formatDescription);
150 },
151 .copySampleLocation = [](MTPluginSampleCursorRef sampleCursor, MTPluginSampleCursorStorageRange* storageRange, MTPluginByteSourceRef* byteSource) {
152 return unwrap(sampleCursor)->copySampleLocation(storageRange, byteSource);
153 },
154#if HAVE(MT_PLUGIN_SAMPLE_CURSOR_PLAYABLE_HORIZON)
155 .getPlayableHorizon = [](MTPluginSampleCursorRef sampleCursor, CMTime* playableHorizon) {
156 return unwrap(sampleCursor)->getPlayableHorizon(playableHorizon);
157 },
158#endif
159 };
