40static bool shouldTransformGraph(id object, const ObjCObjectGraph::Transformer& transformer)
41{
42 if (NSArray *array = dynamic_objc_cast<NSArray>(object)) {
43 for (id element in array) {
44 if (shouldTransformGraph(element, transformer))
45 return true;
46 }
47 }
48
49 if (NSDictionary *dictionary = dynamic_objc_cast<NSDictionary>(object)) {
50 bool result = false;
51 [dictionary enumerateKeysAndObjectsUsingBlock:[&transformer, &result](id key, id object, BOOL* stop) {
52 if (shouldTransformGraph(object, transformer)) {
53 result = true;
54 *stop = YES;
55 }
56 }];
57
58 return result;
59 }
60
61 return transformer.shouldTransformObject(object);
62}
63
64static RetainPtr<id> transformGraph(id object, const ObjCObjectGraph::Transformer& transformer)
65{
66 if (NSArray *array = dynamic_objc_cast<NSArray>(object)) {
67 auto result = adoptNS([[NSMutableArray alloc] initWithCapacity:array.count]);
68 for (id element in array)
69 [result addObject:transformGraph(element, transformer).get()];
70
71 return result;
72 }
73
74 if (NSDictionary *dictionary = dynamic_objc_cast<NSDictionary>(object)) {
75 auto result = adoptNS([[NSMutableDictionary alloc] initWithCapacity:dictionary.count]);
76 [dictionary enumerateKeysAndObjectsUsingBlock:[&result, &transformer](id key, id object, BOOL*) {
77 [result setObject:transformGraph(object, transformer).get() forKey:key];
78 }];
79
80 return result;
81 }
82
83 return transformer.transformObject(object);
84}
85
86RetainPtr<id> ObjCObjectGraph::transform(id object, const Transformer& transformer)
87{
88 if (!object)
89 return nullptr;
90
91 if (!shouldTransformGraph(object, transformer))
92 return object;
93
94 return transformGraph(object, transformer);
95}

97enum class ObjCType : uint8_t {
98 Null,
99
100 NSArray,
101 NSData,
102 NSDate,
103 NSDictionary,
104 NSNumber,
105 NSString,
106
107 WKBrowsingContextHandle,
108 WKTypeRefWrapper,
109};

111static std::optional<ObjCType> typeFromObject(id object)
112{
113 ASSERT(object);
114
115 if (dynamic_objc_cast<NSArray>(object))
116 return ObjCType::NSArray;
117 if (dynamic_objc_cast<NSData>(object))
118 return ObjCType::NSData;
119 if (dynamic_objc_cast<NSDate>(object))
120 return ObjCType::NSDate;
121 if (dynamic_objc_cast<NSDictionary>(object))
122 return ObjCType::NSDictionary;
123 if (dynamic_objc_cast<NSNumber>(object))
124 return ObjCType::NSNumber;
125 if (dynamic_objc_cast<NSString>(object))
126 return ObjCType::NSString;
127
128 if (dynamic_objc_cast<WKBrowsingContextHandle>(object))
129 return ObjCType::WKBrowsingContextHandle;
130 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
131 if (dynamic_objc_cast<WKTypeRefWrapper>(object))
132 return ObjCType::WKTypeRefWrapper;
133 ALLOW_DEPRECATED_DECLARATIONS_END
134
135 return std::nullopt;
136}

138void ObjCObjectGraph::encode(IPC::Encoder& encoder, id object)
139{
140 if (!object) {
141 encoder << static_cast<uint32_t>(ObjCType::Null);
142 return;
143 }
144
145 auto type = typeFromObject(object);
146 if (!type)
147 [NSException raise:NSInvalidArgumentException format:@"Can not encode objects of class type '%@'", static_cast<NSString *>(NSStringFromClass([object class]))];
148
149 encoder << *type;
150
151 switch (type.value()) {
152 case ObjCType::Null:
153 return;
154
155 case ObjCType::NSArray: {
156 NSArray *array = object;
157
158 encoder << static_cast<uint64_t>(array.count);
159 for (id element in array)
160 encode(encoder, element);
161 return;
162 }
163
164 case ObjCType::NSData: {
165 IPC::encode(encoder, static_cast<NSData *>(object));
166 return;
167 }
168
169 case ObjCType::NSDate: {
170 IPC::encode(encoder, static_cast<NSDate *>(object));
171 return;
172 }
173
174 case ObjCType::NSDictionary: {
175 NSDictionary *dictionary = object;
176
177 encoder << static_cast<uint64_t>(dictionary.count);
178 [dictionary enumerateKeysAndObjectsUsingBlock:[&encoder](id key, id object, BOOL *stop) {
179 encode(encoder, key);
180 encode(encoder, object);
181 }];
182 return;
183 }
184
185 case ObjCType::NSNumber: {
186 IPC::encode(encoder, static_cast<NSNumber *>(object));
187 return;
188 }
189
190 case ObjCType::NSString: {
191 IPC::encode(encoder, static_cast<NSString *>(object));
192 return;
193 }
194
195 case ObjCType::WKBrowsingContextHandle: {
196 encoder << static_cast<WKBrowsingContextHandle *>(object).pageProxyID;
197 encoder << static_cast<WKBrowsingContextHandle *>(object).webPageID;
198 return;
199 }
200
201 case ObjCType::WKTypeRefWrapper: {
202 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
203 UserData::encode(encoder, toImpl(static_cast<WKTypeRefWrapper *>(object).object));
204 ALLOW_DEPRECATED_DECLARATIONS_END
205 return;
206 }
207 }
208
209 ASSERT_NOT_REACHED();
210}

217bool ObjCObjectGraph::decode(IPC::Decoder& decoder, RetainPtr<id>& result)
218{
219 ObjCType type;
220 if (!decoder.decode(type))
221 return false;
222
223 switch (type) {
224 case ObjCType::Null: {
225 result = nil;
226 return true;
227 }
228
229 case ObjCType::NSArray: {
230 uint64_t size;
231 if (!decoder.decode(size))
232 return false;
233
234 auto array = adoptNS([[NSMutableArray alloc] init]);
235 for (uint64_t i = 0; i < size; ++i) {
236 RetainPtr<id> element;
237 if (!decode(decoder, element))
238 return false;
239 [array addObject:element.get()];
240 }
241
242 result = WTFMove(array);
243 return true;
244 }
245
246 case ObjCType::NSData: {
247 RetainPtr<NSData> data;
248 if (!IPC::decode(decoder, data))
249 return false;
250
251 result = WTFMove(data);
252 return true;
253 }
254
255 case ObjCType::NSDate: {
256 RetainPtr<NSDate> date;
257 if (!IPC::decode(decoder, date))
258 return false;
259
260 result = WTFMove(date);
261 return true;
262 }
263
264 case ObjCType::NSDictionary: {
265 uint64_t size;
266 if (!decoder.decode(size))
267 return false;
268
269 auto dictionary = adoptNS([[NSMutableDictionary alloc] init]);
270 for (uint64_t i = 0; i < size; ++i) {
271 RetainPtr<id> key;
272 if (!decode(decoder, key))
273 return false;
274
275 RetainPtr<id> object;
276 if (!decode(decoder, object))
277 return false;
278
279 @try {
280 [dictionary setObject:object.get() forKey:key.get()];
281 } @catch (id) {
282 return false;
283 }
284 }
285
286 result = WTFMove(dictionary);
287 return true;
288 }
289
290 case ObjCType::NSNumber: {
291 RetainPtr<NSNumber> number;
292 if (!IPC::decode(decoder, number))
293 return false;
294
295 result = WTFMove(number);
296 return true;
297 }
298
299 case ObjCType::NSString: {
300 RetainPtr<NSString> string;
301 if (!IPC::decode(decoder, string))
302 return false;
303
304 result = WTFMove(string);
305 return true;
306 }
307
308 case ObjCType::WKBrowsingContextHandle: {
309 std::optional<WebPageProxyIdentifier> pageProxyID;
310 decoder >> pageProxyID;
311 if (!pageProxyID)
312 return false;
313 std::optional<WebCore::PageIdentifier> webPageID;
314 decoder >> webPageID;
315 if (!webPageID)
316 return false;
317
318 result = adoptNS([[WKBrowsingContextHandle alloc] _initWithPageProxyID:*pageProxyID andWebPageID:*webPageID]);
319 return true;
320 }
321
322 case ObjCType::WKTypeRefWrapper: {
323 RefPtr<API::Object> object;
324 if (!UserData::decode(decoder, object))
325 return false;
326
327 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
328 result = adoptNS([[WKTypeRefWrapper alloc] initWithObject:toAPI(object.get())]);
329 ALLOW_DEPRECATED_DECLARATIONS_END
330 return true;
331 }
332 }
333
334 ASSERT_NOT_REACHED();
335 return false;
336}

338bool ObjCObjectGraph::decode(IPC::Decoder& decoder, RefPtr<API::Object>& result)
339{
340 RetainPtr<id> rootObject;
341 if (!decode(decoder, rootObject))
342 return false;
343
344 result = ObjCObjectGraph::create(rootObject.get());
345 return true;
346}
