34SecItemRequestData::SecItemRequestData(Type type, CFDictionaryRef query)
35 : m_type(type)
36 , m_queryDictionary(query)
37{
38}
39
40SecItemRequestData::SecItemRequestData(Type type, CFDictionaryRef query, CFDictionaryRef attributesToMatch)
41 : m_type(type)
42 , m_queryDictionary(query)
43 , m_attributesToMatch(attributesToMatch)
44{
45}
46
47void SecItemRequestData::encode(IPC::Encoder& encoder) const
48{
49 encoder << m_type;
50
51 encoder << static_cast<bool>(m_queryDictionary);
52 if (m_queryDictionary)
53 encoder << m_queryDictionary;
54
55 encoder << static_cast<bool>(m_attributesToMatch);
56 if (m_attributesToMatch)
57 encoder << m_attributesToMatch;
58}
59
60bool SecItemRequestData::decode(IPC::Decoder& decoder, SecItemRequestData& secItemRequestData)
61{
62 if (!decoder.decode(secItemRequestData.m_type))
63 return false;
64
65 bool expectQuery;
66 if (!decoder.decode(expectQuery))
67 return false;
68
69 if (expectQuery && !decoder.decode(secItemRequestData.m_queryDictionary))
70 return false;
71
72 bool expectAttributes;
73 if (!decoder.decode(expectAttributes))
74 return false;
75
76 if (expectAttributes && !decoder.decode(secItemRequestData.m_attributesToMatch))
77 return false;
78
79 return true;
80}
