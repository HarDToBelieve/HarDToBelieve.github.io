47void ArgumentCoder<WebCore::CertificateInfo>::encode(Encoder& encoder, const WebCore::CertificateInfo& certificateInfo)
48{
49 encoder << certificateInfo.type();
50
51 switch (certificateInfo.type()) {
52#if HAVE(SEC_TRUST_SERIALIZATION)
53 case WebCore::CertificateInfo::Type::Trust:
54 encoder << certificateInfo.trust();
55 break;
56#endif
57 case WebCore::CertificateInfo::Type::CertificateChain:
58 encoder << certificateInfo.certificateChain();
59 break;
60 case WebCore::CertificateInfo::Type::None:
61 // Do nothing.
62 break;
63 }
64}
65
66bool ArgumentCoder<WebCore::CertificateInfo>::decode(Decoder& decoder, WebCore::CertificateInfo& certificateInfo)
67{
68 WebCore::CertificateInfo::Type certificateInfoType;
69 if (!decoder.decode(certificateInfoType))
70 return false;
71
72 switch (certificateInfoType) {
73#if HAVE(SEC_TRUST_SERIALIZATION)
74 case WebCore::CertificateInfo::Type::Trust: {
75 RetainPtr<SecTrustRef> trust;
76 if (!decoder.decode(trust) || !trust)
77 return false;
78
79 certificateInfo = WebCore::CertificateInfo(WTFMove(trust));
80 return true;
81 }
82#endif
83 case WebCore::CertificateInfo::Type::CertificateChain: {
84 RetainPtr<CFArrayRef> certificateChain;
85 if (!decoder.decode(certificateChain) || !certificateChain)
86 return false;
87
88 certificateInfo = WebCore::CertificateInfo(WTFMove(certificateChain));
89 return true;
90 }
91 case WebCore::CertificateInfo::Type::None:
92 // Do nothing.
93 break;
94 }
95
96 return true;
97}

99static void encodeNSError(Encoder& encoder, NSError *nsError)
100{
101 String domain = [nsError domain];
102 encoder << domain;
103
104 int64_t code = [nsError code];
105 encoder << code;
106
107 NSDictionary *userInfo = [nsError userInfo];
108
109 RetainPtr<CFMutableDictionaryRef> filteredUserInfo = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, userInfo.count, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
110
111 [userInfo enumerateKeysAndObjectsUsingBlock:^(id key, id value, BOOL*) {
112 if ([value isKindOfClass:[NSString class]] || [value isKindOfClass:[NSURL class]] || [value isKindOfClass:[NSNumber class]])
113 CFDictionarySetValue(filteredUserInfo.get(), (__bridge CFTypeRef)key, (__bridge CFTypeRef)value);
114 }];
115
116 if (NSArray *clientIdentityAndCertificates = [userInfo objectForKey:@"NSErrorClientCertificateChainKey"]) {
117 ASSERT([clientIdentityAndCertificates isKindOfClass:[NSArray class]]);
118 ASSERT(^{
119 for (id object in clientIdentityAndCertificates) {
120 if (CFGetTypeID((__bridge CFTypeRef)object) != SecIdentityGetTypeID() && CFGetTypeID((__bridge CFTypeRef)object) != SecCertificateGetTypeID())
121 return false;
122 }
123 return true;
124 }());
125
126 // Turn SecIdentity members into SecCertificate to strip out private key information.
127 id clientCertificates = [NSMutableArray arrayWithCapacity:clientIdentityAndCertificates.count];
128 for (id object in clientIdentityAndCertificates) {
129 if (CFGetTypeID((__bridge CFTypeRef)object) != SecIdentityGetTypeID()) {
130 [clientCertificates addObject:object];
131 continue;
132 }
133 SecCertificateRef certificate = nil;
134 OSStatus status = SecIdentityCopyCertificate((SecIdentityRef)object, &certificate);
135 RetainPtr<SecCertificateRef> retainCertificate = adoptCF(certificate);
136 // The SecIdentity member is the key information of this attribute. Without it, we should nil
137 // the attribute.
138 if (status != errSecSuccess) {
139 LOG_ERROR("Failed to encode nsError.userInfo[NSErrorClientCertificateChainKey]: %d", status);
140 clientCertificates = nil;
141 break;
142 }
143 [clientCertificates addObject:(__bridge id)certificate];
144 }
145 CFDictionarySetValue(filteredUserInfo.get(), CFSTR("NSErrorClientCertificateChainKey"), (__bridge CFTypeRef)clientCertificates);
146 }
147
148 id peerCertificateChain = [userInfo objectForKey:@"NSErrorPeerCertificateChainKey"];
149 if (!peerCertificateChain) {
150 if (SecTrustRef peerTrust = (__bridge SecTrustRef)[userInfo objectForKey:NSURLErrorFailingURLPeerTrustErrorKey]) {
151#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
152 peerCertificateChain = (__bridge NSArray *)adoptCF(SecTrustCopyCertificateChain(peerTrust)).autorelease();
153#else
154 CFIndex count = SecTrustGetCertificateCount(peerTrust);
155 peerCertificateChain = [NSMutableArray arrayWithCapacity:count];
156 for (CFIndex i = 0; i < count; ++i)
157 [peerCertificateChain addObject:(__bridge id)SecTrustGetCertificateAtIndex(peerTrust, i)];
158#endif
159 }
160 }
161 ASSERT(!peerCertificateChain || [peerCertificateChain isKindOfClass:[NSArray class]]);
162 if (peerCertificateChain)
163 CFDictionarySetValue(filteredUserInfo.get(), CFSTR("NSErrorPeerCertificateChainKey"), (__bridge CFTypeRef)peerCertificateChain);
164
165#if HAVE(SEC_TRUST_SERIALIZATION)
166 if (SecTrustRef peerTrust = (__bridge SecTrustRef)[userInfo objectForKey:NSURLErrorFailingURLPeerTrustErrorKey])
167 CFDictionarySetValue(filteredUserInfo.get(), (__bridge CFStringRef)NSURLErrorFailingURLPeerTrustErrorKey, peerTrust);
168#endif
169
170 encoder << static_cast<CFDictionaryRef>(filteredUserInfo.get());
171
172 if (id underlyingError = [userInfo objectForKey:NSUnderlyingErrorKey]) {
173 ASSERT([underlyingError isKindOfClass:[NSError class]]);
174 encoder << true;
175 encodeNSError(encoder, underlyingError);
176 } else
177 encoder << false;
178}

185static RetainPtr<NSError> decodeNSError(Decoder& decoder)
186{
187 String domain;
188 if (!decoder.decode(domain))
189 return nil;
190
191 int64_t code;
192 if (!decoder.decode(code))
193 return nil;
194
195 RetainPtr<CFDictionaryRef> userInfo;
196 if (!decoder.decode(userInfo) || !userInfo)
197 return nil;
198
199 bool hasUnderlyingError = false;
200 if (!decoder.decode(hasUnderlyingError))
201 return nil;
202
203 if (hasUnderlyingError) {
204 auto underlyingNSError = decodeNSError(decoder);
205 if (!underlyingNSError)
206 return nil;
207
208 auto mutableUserInfo = adoptCF(CFDictionaryCreateMutableCopy(kCFAllocatorDefault, CFDictionaryGetCount(userInfo.get()) + 1, userInfo.get()));
209 CFDictionarySetValue(mutableUserInfo.get(), (__bridge CFStringRef)NSUnderlyingErrorKey, (__bridge CFTypeRef)underlyingNSError.get());
210 userInfo = WTFMove(mutableUserInfo);
211 }
212
213 return adoptNS([[NSError alloc] initWithDomain:domain code:code userInfo:(__bridge NSDictionary *)userInfo.get()]);
214}

267{

284std::optional<WebCore::KeypressCommand> ArgumentCoder<WebCore::KeypressCommand>::decode(Decoder& decoder)
285{
286 std::optional<String> commandName;
287 decoder >> commandName;
288 if (!commandName)
289 return std::nullopt;
290
291 std::optional<String> text;
292 decoder >> text;
293 if (!text)
294 return std::nullopt;
295
296 WebCore::KeypressCommand command;
297 command.commandName = WTFMove(*commandName);
298 command.text = WTFMove(*text);
299 return WTFMove(command);
300}

377void ArgumentCoder<WebCore::ContentFilterUnblockHandler>::encode(Encoder& encoder, const WebCore::ContentFilterUnblockHandler& contentFilterUnblockHandler)
378{
379 auto archiver = adoptNS([[NSKeyedArchiver alloc] initRequiringSecureCoding:YES]);
380 contentFilterUnblockHandler.encode(archiver.get());
381 encoder << (__bridge CFDataRef)archiver.get().encodedData;
382}
383
384bool ArgumentCoder<WebCore::ContentFilterUnblockHandler>::decode(Decoder& decoder, WebCore::ContentFilterUnblockHandler& contentFilterUnblockHandler)
385{
386 RetainPtr<CFDataRef> data;
387 if (!decoder.decode(data) || !data)
388 return false;
389
390 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:(__bridge NSData *)data.get() error:nullptr]);
391 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
392 if (!WebCore::ContentFilterUnblockHandler::decode(unarchiver.get(), contentFilterUnblockHandler))
393 return false;
394
395 [unarchiver finishDecoding];
396 return true;
397}

402void ArgumentCoder<WebCore::SerializedPlatformDataCueValue>::encodePlatformData(Encoder& encoder, const WebCore::SerializedPlatformDataCueValue& value)
403{
404 ASSERT(value.platformType() == WebCore::SerializedPlatformDataCueValue::PlatformType::ObjC);

409std::optional<WebCore::SerializedPlatformDataCueValue> ArgumentCoder<WebCore::SerializedPlatformDataCueValue>::decodePlatformData(Decoder& decoder, WebCore::SerializedPlatformDataCueValue::PlatformType platformType)
410{
411 ASSERT(platformType == WebCore::SerializedPlatformDataCueValue::PlatformType::ObjC);
412
413 if (platformType != WebCore::SerializedPlatformDataCueValue::PlatformType::ObjC)
414 return std::nullopt;
415
416 auto object = decodeObject(decoder, WebCore::SerializedPlatformDataCueMac::allowedClassesForNativeValues());
417 if (!object)
418 return std::nullopt;
419
420 return WebCore::SerializedPlatformDataCueValue { platformType, object.value().get() };
421}
