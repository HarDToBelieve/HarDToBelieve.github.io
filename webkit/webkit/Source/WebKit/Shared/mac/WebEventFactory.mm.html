48static WebMouseEvent::Button currentMouseButton()
49{
50 NSUInteger pressedMouseButtons = [NSEvent pressedMouseButtons];
51 if (!pressedMouseButtons)
52 return WebMouseEvent::NoButton;
53 if (pressedMouseButtons == 1 << 0)
54 return WebMouseEvent::LeftButton;
55 if (pressedMouseButtons == 1 << 1)
56 return WebMouseEvent::RightButton;
57 return WebMouseEvent::MiddleButton;
58}

60static WebMouseEvent::Button mouseButtonForEvent(NSEvent *event)
61{
62 switch ([event type]) {
63 case NSEventTypeLeftMouseDown:
64 case NSEventTypeLeftMouseUp:
65 case NSEventTypeLeftMouseDragged:
66 return WebMouseEvent::LeftButton;
67 case NSEventTypeRightMouseDown:
68 case NSEventTypeRightMouseUp:
69 case NSEventTypeRightMouseDragged:
70 return WebMouseEvent::RightButton;
71 case NSEventTypeOtherMouseDown:
72 case NSEventTypeOtherMouseUp:
73 case NSEventTypeOtherMouseDragged:
74 return WebMouseEvent::MiddleButton;
75 case NSEventTypePressure:
76 case NSEventTypeMouseEntered:
77 case NSEventTypeMouseExited:
78 return currentMouseButton();
79 default:
80 return WebMouseEvent::NoButton;
81 }
82}

89static WebEvent::Type mouseEventTypeForEvent(NSEvent* event)
90{
91 switch ([event type]) {
92 case NSEventTypeLeftMouseDragged:
93 case NSEventTypeMouseEntered:
94 case NSEventTypeMouseExited:
95 case NSEventTypeMouseMoved:
96 case NSEventTypeOtherMouseDragged:
97 case NSEventTypeRightMouseDragged:
98 return WebEvent::MouseMove;
99 case NSEventTypeLeftMouseDown:
100 case NSEventTypeRightMouseDown:
101 case NSEventTypeOtherMouseDown:
102 return WebEvent::MouseDown;
103 case NSEventTypeLeftMouseUp:
104 case NSEventTypeRightMouseUp:
105 case NSEventTypeOtherMouseUp:
106 return WebEvent::MouseUp;
107 default:
108 return WebEvent::MouseMove;
109 }
110}

130static NSPoint globalPointForEvent(NSEvent *event)
131{
132 switch ([event type]) {
133 case NSEventTypePressure:
134 case NSEventTypeLeftMouseDown:
135 case NSEventTypeLeftMouseUp:
136 case NSEventTypeLeftMouseDragged:
137 case NSEventTypeMouseEntered:
138 case NSEventTypeMouseExited:
139 case NSEventTypeMouseMoved:
140 case NSEventTypeOtherMouseDown:
141 case NSEventTypeOtherMouseUp:
142 case NSEventTypeOtherMouseDragged:
143 case NSEventTypeRightMouseDown:
144 case NSEventTypeRightMouseUp:

153static NSPoint pointForEvent(NSEvent *event, NSView *windowView)
154{
155 switch ([event type]) {
156 case NSEventTypePressure:
157 case NSEventTypeLeftMouseDown:
158 case NSEventTypeLeftMouseUp:
159 case NSEventTypeLeftMouseDragged:
160 case NSEventTypeMouseEntered:
161 case NSEventTypeMouseExited:
162 case NSEventTypeMouseMoved:
163 case NSEventTypeOtherMouseDown:
164 case NSEventTypeOtherMouseUp:
165 case NSEventTypeOtherMouseDragged:
166 case NSEventTypeRightMouseDown:
167 case NSEventTypeRightMouseUp:
168 case NSEventTypeRightMouseDragged:
169 case NSEventTypeScrollWheel: {
170 // Note: This will have its origin at the bottom left of the window unless windowView is flipped.
171 // In those cases, the Y coordinate gets flipped by Widget::convertFromContainingWindow.
172 NSPoint location = [event locationInWindow];
173 if (windowView)
174 location = [windowView convertPoint:location fromView:nil];
175 return location;
176 }
177 default:
178 return NSZeroPoint;
179 }
180}

182static WebWheelEvent::Phase phaseForEvent(NSEvent *event)
183{
184 uint32_t phase = WebWheelEvent::PhaseNone;
185 if ([event phase] & NSEventPhaseBegan)
186 phase |= WebWheelEvent::PhaseBegan;
187 if ([event phase] & NSEventPhaseStationary)
188 phase |= WebWheelEvent::PhaseStationary;
189 if ([event phase] & NSEventPhaseChanged)
190 phase |= WebWheelEvent::PhaseChanged;
191 if ([event phase] & NSEventPhaseEnded)
192 phase |= WebWheelEvent::PhaseEnded;
193 if ([event phase] & NSEventPhaseCancelled)
194 phase |= WebWheelEvent::PhaseCancelled;
195 if ([event phase] & NSEventPhaseMayBegin)
196 phase |= WebWheelEvent::PhaseMayBegin;
197
198 return static_cast<WebWheelEvent::Phase>(phase);
199}

201static WebWheelEvent::Phase momentumPhaseForEvent(NSEvent *event)
202{
203 uint32_t phase = WebWheelEvent::PhaseNone;
204
205 if ([event momentumPhase] & NSEventPhaseBegan)
206 phase |= WebWheelEvent::PhaseBegan;
207 if ([event momentumPhase] & NSEventPhaseStationary)
208 phase |= WebWheelEvent::PhaseStationary;
209 if ([event momentumPhase] & NSEventPhaseChanged)
210 phase |= WebWheelEvent::PhaseChanged;
211 if ([event momentumPhase] & NSEventPhaseEnded)
212 phase |= WebWheelEvent::PhaseEnded;
213 if ([event momentumPhase] & NSEventPhaseCancelled)
214 phase |= WebWheelEvent::PhaseCancelled;
215
216 return static_cast<WebWheelEvent::Phase>(phase);
217}

220{
221 if (replacesSoftSpace)
222 return emptyString();

224 return emptyString();

228static inline String unmodifiedTextFromEvent(NSEvent* event, bool replacesSoftSpace)
229{
230 if (replacesSoftSpace)
231 return emptyString();
232 if ([event type] == NSEventTypeFlagsChanged)
233 return emptyString();
234 return String([event charactersIgnoringModifiers]);

237static bool isKeypadEvent(NSEvent* event)
238{
239 // Check that this is the type of event that has a keyCode.
240 switch ([event type]) {
241 case NSEventTypeKeyDown:
242 case NSEventTypeKeyUp:
243 case NSEventTypeFlagsChanged:
244 break;
245 default:
246 return false;
247 }
248
249 switch ([event keyCode]) {
250 case 71: // Clear
251 case 81: // =
252 case 75: // /
253 case 67: // *
254 case 78: // -
255 case 69: // +
256 case 76: // Enter
257 case 65: // .
258 case 82: // 0
259 case 83: // 1
260 case 84: // 2
261 case 85: // 3
262 case 86: // 4
263 case 87: // 5
264 case 88: // 6
265 case 89: // 7
266 case 91: // 8
267 case 92: // 9
268 return true;
269 }
270
271 return false;
272}

274static inline bool isKeyUpEvent(NSEvent *event)
275{
276 if ([event type] != NSEventTypeFlagsChanged)
277 return [event type] == NSEventTypeKeyUp;
278 // FIXME: This logic fails if the user presses both Shift keys at once, for example:
279 // we treat releasing one of them as keyDown.
280 switch ([event keyCode]) {
281 case 54: // Right Command
282 case 55: // Left Command
283 return !([event modifierFlags] & NSEventModifierFlagCommand);
284
285 case 57: // Capslock
286 return !([event modifierFlags] & NSEventModifierFlagCapsLock);
287
288 case 56: // Left Shift
289 case 60: // Right Shift
290 return !([event modifierFlags] & NSEventModifierFlagShift);
291
292 case 58: // Left Alt
293 case 61: // Right Alt
294 return !([event modifierFlags] & NSEventModifierFlagOption);
295
296 case 59: // Left Ctrl
297 case 62: // Right Ctrl
298 return !([event modifierFlags] & NSEventModifierFlagControl);
299
300 case 63: // Function
301 return !([event modifierFlags] & NSEventModifierFlagFunction);
302 }
303 return false;
304}

306static inline OptionSet<WebEvent::Modifier> modifiersForEvent(NSEvent *event)
307{
308 OptionSet<WebEvent::Modifier> modifiers;
309 if ([event modifierFlags] & NSEventModifierFlagCapsLock)
310 modifiers.add(WebEvent::Modifier::CapsLockKey);
311 if ([event modifierFlags] & NSEventModifierFlagShift)
312 modifiers.add(WebEvent::Modifier::ShiftKey);
313 if ([event modifierFlags] & NSEventModifierFlagControl)
314 modifiers.add(WebEvent::Modifier::ControlKey);
315 if ([event modifierFlags] & NSEventModifierFlagOption)
316 modifiers.add(WebEvent::Modifier::AltKey);
317 if ([event modifierFlags] & NSEventModifierFlagCommand)
318 modifiers.add(WebEvent::Modifier::MetaKey);
319 return modifiers;
320}

332WebMouseEvent WebEventFactory::createWebMouseEvent(NSEvent *event, NSEvent *lastPressureEvent, NSView *windowView)
333{
334 NSPoint position = pointForEvent(event, windowView);
335 NSPoint globalPosition = globalPointForEvent(event);
336
337 WebEvent::Type type = mouseEventTypeForEvent(event);
338 if ([event type] == NSEventTypePressure) {
339 // Since AppKit doesn't send mouse events for force down or force up, we have to use the current pressure
340 // event and lastPressureEvent to detect if this is MouseForceDown, MouseForceUp, or just MouseForceChanged.
341 if (lastPressureEvent.stage == 1 && event.stage == 2)
342 type = WebEvent::MouseForceDown;
343 else if (lastPressureEvent.stage == 2 && event.stage == 1)
344 type = WebEvent::MouseForceUp;
345 else
346 type = WebEvent::MouseForceChanged;
347 }
348
349 WebMouseEvent::Button button = mouseButtonForEvent(event);
350 unsigned short buttons = currentlyPressedMouseButtons();
351 float deltaX = [event deltaX];
352 float deltaY = [event deltaY];
353 float deltaZ = [event deltaZ];
354 int clickCount = clickCountForEvent(event);
355 auto modifiers = modifiersForEvent(event);
356 auto timestamp = WebCore::eventTimeStampSince1970(event);
357 int eventNumber = [event eventNumber];
358 int menuTypeForEvent = typeForEvent(event);
359
360 int stage = [event type] == NSEventTypePressure ? event.stage : lastPressureEvent.stage;
361 double pressure = [event type] == NSEventTypePressure ? event.pressure : lastPressureEvent.pressure;
362 double force = pressure + stage;
363
364 return WebMouseEvent(type, button, buttons, WebCore::IntPoint(position), WebCore::IntPoint(globalPosition), deltaX, deltaY, deltaZ, clickCount, modifiers, timestamp, force, WebMouseEvent::SyntheticClickType::NoTap, eventNumber, menuTypeForEvent);
365}

367WebWheelEvent WebEventFactory::createWebWheelEvent(NSEvent *event, NSView *windowView)
368{
369 NSPoint position = pointForEvent(event, windowView);
370 NSPoint globalPosition = globalPointForEvent(event);
371
372 BOOL continuous;
373 float deltaX = 0;
374 float deltaY = 0;
375 float wheelTicksX = 0;
376 float wheelTicksY = 0;
377
378 WebCore::getWheelEventDeltas(event, deltaX, deltaY, continuous);
379
380 if (continuous) {
381 // smooth scroll events
382 wheelTicksX = deltaX / static_cast<float>(WebCore::Scrollbar::pixelsPerLineStep());
383 wheelTicksY = deltaY / static_cast<float>(WebCore::Scrollbar::pixelsPerLineStep());
384 } else {
385 // plain old wheel events
386 wheelTicksX = deltaX;
387 wheelTicksY = deltaY;
388 deltaX *= static_cast<float>(WebCore::Scrollbar::pixelsPerLineStep());
389 deltaY *= static_cast<float>(WebCore::Scrollbar::pixelsPerLineStep());
390 }
391
392 WebWheelEvent::Granularity granularity = WebWheelEvent::ScrollByPixelWheelEvent;
393 bool directionInvertedFromDevice = [event isDirectionInvertedFromDevice];
394 WebWheelEvent::Phase phase = phaseForEvent(event);
395 WebWheelEvent::Phase momentumPhase = momentumPhaseForEvent(event);
396 bool hasPreciseScrollingDeltas = continuous;
397
398 uint32_t scrollCount;
399 WebCore::FloatSize unacceleratedScrollingDelta;
400
401 static bool nsEventSupportsScrollCount = [NSEvent instancesRespondToSelector:@selector(_scrollCount)];
402 if (nsEventSupportsScrollCount) {
403 scrollCount = [event _scrollCount];
404 unacceleratedScrollingDelta = WebCore::FloatSize([event _unacceleratedScrollingDeltaX], [event _unacceleratedScrollingDeltaY]);
405 } else {
406 scrollCount = 0;
407 unacceleratedScrollingDelta = WebCore::FloatSize(deltaX, deltaY);
408 }
409
410 auto modifiers = modifiersForEvent(event);
411 auto timestamp = WebCore::eventTimeStampSince1970(event);
412
413 return WebWheelEvent(WebEvent::Wheel, WebCore::IntPoint(position), WebCore::IntPoint(globalPosition), WebCore::FloatSize(deltaX, deltaY), WebCore::FloatSize(wheelTicksX, wheelTicksY), granularity, directionInvertedFromDevice, phase, momentumPhase, hasPreciseScrollingDeltas, scrollCount, unacceleratedScrollingDelta, modifiers, timestamp);
414}

416WebKeyboardEvent WebEventFactory::createWebKeyboardEvent(NSEvent *event, bool handledByInputMethod, bool replacesSoftSpace, const Vector<WebCore::KeypressCommand>& commands)
417{
418 WebEvent::Type type = isKeyUpEvent(event) ? WebEvent::KeyUp : WebEvent::KeyDown;
419 String text = textFromEvent(event, replacesSoftSpace);
420 String unmodifiedText = unmodifiedTextFromEvent(event, replacesSoftSpace);
421 String key = WebCore::keyForKeyEvent(event);
422 String code = WebCore::codeForKeyEvent(event);
423 String keyIdentifier = WebCore::keyIdentifierForKeyEvent(event);
424 int windowsVirtualKeyCode = WebCore::windowsKeyCodeForKeyEvent(event);
425 int nativeVirtualKeyCode = [event keyCode];
426 int macCharCode = WebCore::keyCharForEvent(event);
427 bool autoRepeat = [event type] != NSEventTypeFlagsChanged && [event isARepeat];
428 bool isKeypad = isKeypadEvent(event);
429 bool isSystemKey = false; // SystemKey is always false on the Mac.
430 auto modifiers = modifiersForEvent(event);
431 auto timestamp = WebCore::eventTimeStampSince1970(event);
432
433 // Always use 13 for Enter/Return -- we don't want to use AppKit's different character for Enter.
434 if (windowsVirtualKeyCode == VK_RETURN) {
435 text = "\r";
436 unmodifiedText = "\r";
437 }
438
439 // AppKit sets text to "\x7F" for backspace, but the correct KeyboardEvent character code is 8.
440 if (windowsVirtualKeyCode == VK_BACK) {
441 text = "\x8";
442 unmodifiedText = "\x8";
443 }
444
445 // Always use 9 for Tab -- we don't want to use AppKit's different character for shift-tab.
446 if (windowsVirtualKeyCode == VK_TAB) {
447 text = "\x9";
448 unmodifiedText = "\x9";
449 }
450
451 return WebKeyboardEvent(type, text, unmodifiedText, key, code, keyIdentifier, windowsVirtualKeyCode, nativeVirtualKeyCode, macCharCode, handledByInputMethod, commands, autoRepeat, isKeypad, isSystemKey, modifiers, timestamp);
452}

454NSEventModifierFlags WebEventFactory::toNSEventModifierFlags(OptionSet<WebKit::WebEvent::Modifier> modifiers)
455{
456 NSEventModifierFlags modifierFlags = 0;
457 if (modifiers.contains(WebKit::WebEvent::Modifier::CapsLockKey))
458 modifierFlags |= NSEventModifierFlagCapsLock;
459 if (modifiers.contains(WebKit::WebEvent::Modifier::ShiftKey))
460 modifierFlags |= NSEventModifierFlagShift;
461 if (modifiers.contains(WebKit::WebEvent::Modifier::ControlKey))
462 modifierFlags |= NSEventModifierFlagControl;
463 if (modifiers.contains(WebKit::WebEvent::Modifier::AltKey))
464 modifierFlags |= NSEventModifierFlagOption;
465 if (modifiers.contains(WebKit::WebEvent::Modifier::MetaKey))
466 modifierFlags |= NSEventModifierFlagCommand;
467 return modifierFlags;
468}

470NSInteger WebEventFactory::toNSButtonNumber(WebKit::WebMouseEvent::Button mouseButton)
471{
472 switch (mouseButton) {
473 case WebKit::WebMouseEvent::NoButton:
474 return 0;
475 case WebKit::WebMouseEvent::LeftButton:
476 return 1 << 0;
477 case WebKit::WebMouseEvent::RightButton:
478 return 1 << 1;
479 case WebKit::WebMouseEvent::MiddleButton:
480 return 1 << 2;
481 }
482 ASSERT_NOT_REACHED();
483 return 0;
484}
