41void WebHitTestResultData::platformEncode(IPC::Encoder& encoder) const
42{
43 bool hasActionContext = detectedDataActionContext;
44 encoder << hasActionContext;
45 if (!hasActionContext)
46 return;
47
48 encoder << detectedDataActionContext;
49
50 encoder << detectedDataBoundingBox;
51 encoder << detectedDataOriginatingPageOverlay;
52
53 bool hasDetectedDataTextIndicator = detectedDataTextIndicator;
54 encoder << hasDetectedDataTextIndicator;
55 if (hasDetectedDataTextIndicator)
56 encoder << detectedDataTextIndicator->data();
57}
58
59bool WebHitTestResultData::platformDecode(IPC::Decoder& decoder, WebHitTestResultData& hitTestResultData)
60{
61 bool hasActionContext;
62 if (!decoder.decode(hasActionContext))
63 return false;
64
65 if (!hasActionContext)
66 return true;
67 ASSERT(PAL::isDataDetectorsFrameworkAvailable());
68
69 auto detectedDataActionContext = IPC::decode<DDActionContext>(decoder, PAL::getDDActionContextClass());
70 if (!detectedDataActionContext)
71 return false;
72
73 hitTestResultData.detectedDataActionContext = WTFMove(*detectedDataActionContext);
74
75 if (!decoder.decode(hitTestResultData.detectedDataBoundingBox))
76 return false;
77
78 if (!decoder.decode(hitTestResultData.detectedDataOriginatingPageOverlay))
79 return false;
80
81 bool hasDetectedDataTextIndicator;
82 if (!decoder.decode(hasDetectedDataTextIndicator))
83 return false;
84
85 if (hasDetectedDataTextIndicator) {
86 std::optional<WebCore::TextIndicatorData> indicatorData;
87 decoder >> indicatorData;
88 if (!indicatorData)
89 return false;
90
91 hitTestResultData.detectedDataTextIndicator = WebCore::TextIndicator::create(*indicatorData);
92 }
93
94 return true;
95}
